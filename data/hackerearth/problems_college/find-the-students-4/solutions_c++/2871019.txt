// CP template
#include <bits/stdc++.h>

using namespace std;

int main()
{
	int n;
	scanf("%d",&n);
	int mem[1005];
	for(int i=0; i<n; i++){
		scanf("%d",&mem[i]);
	}
	sort(mem,mem+n);
	//count
	int i=0;
	while(i<n){
		int count = 1;
		int last = mem[i];
		i++;
		while(mem[i] == last){
			i++;
			count++;
		}
		printf("%d\n",n-i + count);
	}
	return 0;
}
