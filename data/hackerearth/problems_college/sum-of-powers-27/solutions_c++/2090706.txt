#include <bits/stdc++.h>

#define MAXN 1024

using namespace std;

int N;

void read()
{
	scanf("%d", &N);
}

bool good[2 * MAXN * MAXN];

void solve()
{
	for(int i = 0; i <= N; i++)
		if(good[i] && good[N - i])
		{
			printf("Yes\n");
			return;
		}

	printf("No\n");
}

int k = 1024;

void pre()
{
	int bound = k * k;
	good[1] = true;

	for(int i = 2; i < k; i++)
	{
		int curr = i;
		while(curr * i < bound)
		{
			curr *= i;
			good[curr] = true;
		}
	}
}

int main()
{
	pre();

	int T;
	scanf("%d", &T);

	for(int i = 0; i < T; i++)
	{
		read();
		solve();
	}

	return 0;
}
