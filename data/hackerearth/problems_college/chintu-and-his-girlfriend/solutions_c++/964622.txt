#include <iostream>
using namespace std;
#include<cstdio>
#include<cstring>
int main()
{
	int n,k,p;
	char a[200];
	int b[27][2];
	for(int i=0;i<27;i++)
	{
	b[i][0]=0;
	b[i][1]=0;
	}
	scanf("%d %d\n",&n,&k);
	p=n;
	for(int j=1;j<=n;j++)
	{
		scanf("%s\n",a);
		int l=strlen(a);
		for(int i=0;i<l;i++)
		{
			int k=a[i];
			k=k-97;
		    if(b[k][1]==0)
		    {
			b[k][0]+=1;
			b[k][1]=1;
		    }
		}
		for(int d=0;d<27;d++)
		b[d][1]=0;
	}
	int ans=0;
	for(int i=0;i<27;i++)
	{
		if(b[i][0]==n)
		ans+=1;
	}
	if(ans>=k)
	cout<<":-)"<<endl;
	else
	cout<<":-("<<endl;
    ///cout << "Hello World!" << endl;
    return 0;
}
