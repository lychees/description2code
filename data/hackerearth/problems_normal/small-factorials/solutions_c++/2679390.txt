#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#define max 1000
using namespace std;

int multiply(int x,int *res,int res_size)
    {
     int carry = 0;  // Initialize carry
      int i,prod;
    // One by one multiply n with individual digits of res[]
    for ( i=0; i<res_size; i++)
    {
         prod = res[i] * x + carry;
        res[i] = prod % 10;  // Store last digit of 'prod' in res[]
        carry  = prod/10;    // Put rest in carry
    }
 
    // Put carry in res and increase result size
    while (carry)
    {
        res[res_size] = carry%10;
        carry = carry/10;
        res_size++;
    }
    return res_size;
}





void factorial(int n)
    {
       int a[max],a_size,i,j;
        a[0]=1;
        a_size= 1;
    for(i=2;i<=n;i++)
        {
          a_size = multiply(i,&a[0],a_size);
    }
  //  cout<<"factorial of given no. is\n";
    for(j=a_size-1;j>=0;j--)
        cout<<a[j];
    return;
}

int main() 
        {
        	int t;
        	cin>>t;
        	while(t--)
        	{
            int N;
            cin>>N;
            factorial(N);
            cout<<"\n";
        	}
            return(0);
        }
