#include <iostream>
using namespace std;
int main()
{
    int A,B,C,N,T,cntp[10],cntn[10],j,k;
    cin>>T;
    for(j=0;j<T;++j)
    {
    	cin>>N;
    	cin>>A>>B>>C;
    	int X[100000],Y[100000];
    	for(int i=0;i<N;++i)
    	{
        cin>>X[i]>>Y[i];
        if(A*X[i]+B*Y[i]+C>0)
            ++cntp[j];
        else if(A*X[i]+B*Y[i]+C<0)
            ++cntn[j];
    	}
    }
    for(k=0;k<T;++k)
    {
    	if(cntp[k]>cntn[k])
        cout<<cntp[k]<<endl;
    else if (cntp[k]<cntn[k])
        cout<<cntn[k]<<endl;
    else
    	cout<<"0"<<endl;
    }
}
