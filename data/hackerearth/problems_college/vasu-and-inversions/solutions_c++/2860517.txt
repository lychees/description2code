#include <iostream>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n,k;
    	scanf("%d%d",&n,&k);
    	int i;
    	int count = 0;
    	for(i=0;i<n;i++)
    	{
    		int a;
    		scanf("%d",&a);
    		if(!a)
    		count++;
    	}
    	int ans = 0;
    	if(count>=k)
    	ans = n-count+k;
    	else
    	{
    		ans = k-count;
    		if(ans%2 == 0)
    		ans = n;
    		else
    		ans = n-1;
		}
    	printf("%d\n",ans);
    }
    return 0;
}
