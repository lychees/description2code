#include <iostream>
#include <vector>
#include <algorithm>

int main()
{
    int64_t test;
    std::cin>>test;
    while(test--)
    {
   		int64_t num,maximum,minimum;
   		std::cin>>num;
   		if(num==1) {int64_t a;std::cin>>a;std::cout<<a<<" "<<a<<"\n";continue;}
   		else{std::vector<int64_t> positive,negative,zeroes;
   		for(int64_t i=0;i<num;i++)
   		{
   			int64_t temp;
   			std::cin>>temp;
   			if(temp>0)
   				positive.push_back(temp);
   			else if(temp<0)
   				negative.push_back(temp);
   			else if(temp==0)
   				zeroes.push_back(temp);
   		}
        std::sort(positive.begin(),positive.end());
        std::sort(negative.begin(),negative.end());
        int64_t a=1,b=1,c=1;
        if(positive.size()>0)
            for(int64_t i=0;i<positive.size();i++)
                 a=a*positive[i];
        else a=0;

        if(negative.size()>0)
        {
            if(negative.size()%2==0)
            {
                for(int64_t i=0;i<negative.size();i++)
                    b=b*negative[i];
                for(int64_t i=0;i<negative.size()-1;i++)
                    c=c*negative[i];
            }
            if(negative.size()%2==1 && negative.size()>1)
            {
                for(int64_t i=0;i<negative.size()-1;i++)
                    b=b*negative[i];
                for(int64_t i=0;i<negative.size();i++)
                    c=c*negative[i];
            }
            if(negative.size()%2==1 && negative.size()==1)
            {
                b=0;c=negative[0];
            }
        }
        if(positive.size()==0 && negative.size()==0 && zeroes.size()>0)
        {
            maximum=0;minimum=0;
        }
        if(positive.size()==0 && negative.size()>0 && zeroes.size()==0)
        {
            maximum=b;minimum=c;
        }
        if(positive.size()==0 && negative.size()>0 && zeroes.size()>0)
        {
            maximum=b;minimum=c;
        }
        if(positive.size()>0 && negative.size()==0 && zeroes.size()==0)
        {
            maximum=a; minimum=positive[0];
        }
        if(positive.size()>0 && negative.size()==0 && zeroes.size()>0)
        {
            maximum=a;minimum=0;
        }
        if(positive.size()>0 && negative.size()>0 && zeroes.size()==0)
        {
            maximum=std::max(a*b,a);minimum=a*c;
        }
        if(positive.size()>0 && negative.size()>0 && zeroes.size()>0)
        {
            maximum=std::max(a*b,a);minimum=a*c;
        }
        //std::cout<<a<<" "<<b<<" "<<c<<"\n";
        std::cout<<maximum<<" "<<minimum<<"\n";
   		positive.clear();negative.clear();zeroes.clear();}

    }
    return 0;
}
