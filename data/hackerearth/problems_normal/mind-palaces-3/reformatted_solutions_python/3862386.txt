import numpy
m, n = map(int, raw_input().split())
li = []
for i in range(m):
    li.append(map(int, raw_input().split()))
# print numpy.array(li)

for _ in range(int(raw_input())):
    q = int(raw_input())
    l = 0
    h = n-1
    found = 0
    while l < m and h >= 0:
        if q == li[l][h]:
            print l, h
            found = 1
            break
        elif q < li[l][h]:
            h -= 1
        elif q > li[l][h]:
            l += 1
    if found == 0:
        print "-1 -1"
