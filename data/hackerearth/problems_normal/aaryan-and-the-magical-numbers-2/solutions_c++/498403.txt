#include <iostream>
#include <cstdio>
#include <cmath>
#include <algorithm>

using namespace std;
long long arr[19][10];
int countLessPrime[10];
long long arr2[20];
long long arrpow[20];
int main(){
long long t, n,n2,i,j,temp,sum;

// arr[i][j] means that all the no.s upto 1 to j*10^i (exclusive) which contain atleast one prime
// digit

for(i=0;i<=18;i++){
	arr[i][0]=0;
	for(j=1;j<=9;j++){
			if((j-1)==2 || (j-1)==3 || (j-1)==5 || (j-1)==7)
				arr[i][j]=arr[i][j-1]+pow(10,i);
			else{
				arr[i][j]=arr[i][j-1]+pow(10,i);
				arr[i][j]-=pow(6,i);
				//arr[i][j]-=pow((j-countLessPrime[j]),(max((i-1),0)));
			}
		}
}

arrpow[0]=1;
for(i=1;i<=18;i++)
arrpow[i]=arrpow[i-1]*10;
scanf("%lld",&t);
while(t--){
	scanf("%lld",&n);
	n2=n;
	temp=0;

	while(n>0){
		arr2[temp++]=n%10;
				n/=10;
	}
	sum=0;
	//temp is the length of the string
	for(i=temp-1;i>=0;i--){
		sum+=arr[i][arr2[i]];
		n2=n2%(long long)(arrpow[i]);
		if(arr2[i]==2 || arr2[i]==3 || arr2[i]==5 || arr2[i]==7){
			sum+=(n2+1);
			break;
		}

	}

	printf("%lld\n",sum);
}


	return 0;
}
