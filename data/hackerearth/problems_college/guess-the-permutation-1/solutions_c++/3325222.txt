#include <iostream>
using namespace std;
int *lex(int *arr1, int *arr2,int n);
void reset(int *arr, int n);

int main()
{
		int t;
		cin>>t;
		while(t--)
		{
				int n;
				cin>>n;
				int arr[n][n];
				int i,j,p;
				int carr[n],count;
				for(i=0;i<n;++i)
				{
						for(j=0;j<n;++j)
						{
								cin>>arr[i][j];
						}

				}
				int flag=0,p1,p2;
				p1=p2=0;
				for(i=0;i<n;++i)
				{
						reset(carr,n);
						count=0;
						for(j=0;j<n;++j)
						{
								p=arr[i][j];
								if(carr[p]!=0)
								{
										break;
								}
								else 
								{
										carr[p]++;
										count++;
								}
						}
						if(count==n && flag==0) 
						{
								p1=i;
								flag++;
						}
						else if(count==n && flag==1)
						{
								p2=i;
								flag++;
						}
						else if(flag==2) break;
				}
				int *answer=arr[p1];
				for(i=0;i<n;i++)
				if(answer[i]==0) answer[i]=n;
				if(flag==1)
				{

						for(i=0;i<n;i++)
						cout<<answer[i]<<" ";
						cout<<endl;
				}
				else
				{
						answer=arr[p2];
						for(i=0;i<n;i++)
						if(answer[i]==0) answer[i]=n;

						answer=lex(arr[p1],arr[p2],n);
						for(i=0;i<n;i++)
						cout<<answer[i]<<" ";
						cout<<endl;
				}

		}
		return 0;
}
int *lex(int *arr1, int *arr2,int n)
{
		int i,flag=0;
		for(i=0;i<n;i++)
		{
				if(arr1[i]>arr2[i])
				{
						flag=2;
						break;
				}
				else if(arr1[i]<arr2[i])
				{
						flag=1;
						break;
				}
		}
		if(flag==1) return arr1;
		else return arr2;

}
void reset(int *arr, int n)
{
		for(int i=0;i<n;i++)
		{
				arr[i]=0;
		}
}
