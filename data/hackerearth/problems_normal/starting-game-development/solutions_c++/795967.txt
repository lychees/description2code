#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,m,q;
    cin>>n>>m>>q;
    int nm[n][m];
    int i,j,a,ind;
    for(i=0;i<n;i++)for(j=0;j<m;j++) cin>>nm[i][j];
    while(q--)
    {ind=m-1;
    	for(j=0;j<n;j++)
    		{
            	cin>>a;
                while(a<nm[j][ind])
                    ind--;
            }
        cout<<ind+1<<"\n";
    }
return 0;
}
