	n-=1
	time=map(int,raw_input().split())
	m=time[1]*6
	h=(time[0]*30)%360
	h=h+(m*30)/360
	if h>m:
		angle=h-m
	else:
		angle=m-h
	if angle>180:
		angle=360-angle
