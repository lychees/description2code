#include <iostream>
using namespace std;

int main()
{
	int t, n, b;
	cin>>t;
	while(t--) {
		cin>>n>>b;
		int x = n%b;
		if(b%2 == 0) {
			if(x < b/2) {
				cout<<n-x<<endl;
			}
			else {
				cout<<n+(b-x)<<endl;
			}
		}
		else {
			if(x <= b/2) {
				cout<<n-x<<endl;
			}
			else {
				cout<<n+(b-x)<<endl;
			}
		}
	}
    return 0;
}
