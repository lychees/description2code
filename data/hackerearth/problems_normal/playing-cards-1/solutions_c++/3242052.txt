    #include<bits/stdc++.h>
    using namespace std;
    #define mod 1000000007
    #define ll long long 
    ll arr[2601][2601];
    void precal()
    {
    	ll i,j;
    	for(i=1;i<2601;i++)
    	{
    		for(j=0;j<=i;j++)
    		{
    			if(j==0||i==j)
    				arr[i][j]=1;
    			else
    				arr[i][j]=(arr[i-1][j]+arr[i-1][j-1])%mod;
    		}
    	}
    }
    int main()
    {
    	precal();
    	ll t,n,m,f,r,ans,tmp,temp,i,x,y;
    	scanf("%lld",&t);
    	while(t--)
    	{
    		ans=0;
    		scanf("%lld%lld",&n,&m);
    		if(m&1)
    		{
    			printf("0\n");
    			continue;
    		}
    		f=n*6;
    		r=18*n;
    		tmp=m/2;
    		//temp=m/2;
    		for(i=0;i<=tmp;i++)
    		{
    			x=(arr[f][i]*arr[f][tmp-i])%mod;
    			y=(arr[r][i]*arr[r][tmp-i])%mod;
    			ans+=(x*y)%mod;	
    			ans%=mod;
    		}
    		printf("%lld\n",ans%mod);
    	
    	}
    }
