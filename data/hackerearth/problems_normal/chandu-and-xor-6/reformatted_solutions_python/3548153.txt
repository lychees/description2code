N, M = raw_input().split(' ')
N, M = int(N), int(M)

List = []
for i in xrange(N):
    List.append([])
    List[i] = raw_input().split(' ')

Row_Sum = [0] * N
Col_Sum = [0] * M

for i in xrange(N):
    for j in xrange(M):
        Row_Sum[i] += int(List[i][j])
        Col_Sum[j] += int(List[i][j])

MAX = 0
for i in xrange(N):
    for j in xrange(M):
        MAX = max(MAX, Row_Sum[i] ^ Col_Sum[j])

print MAX
