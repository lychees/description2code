#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while (t--) {
    	int n;
    	cin >> n;
    	int data[n];
    	for (int i = 0; i < n; i++) {
    		 cin >> data[i];
    	}
    	int max = 0, prevm = 0.0;
    	int fx = 0, fy = 0;
    	int x = 0, y = 0;
    	int i = 0;
    	int l = 0, r = 0;
    	while (i < n && data[i] != -1) {
    		if (data[i] == 1)
    			l++;
    		i++;
    	}
    	if (i == n) {
    		cout << "0/1" << endl;
    		continue;
    	}
    	i++;
    	max = l + r;
    	while (i < n) {
	    	while (i < n && data[i] != -1) {
	    		if (data[i] == 1)
	    			r++;
	    		i++;
	    	}
	    	i++;
	    	if (l+r > max) {
	    		max = l + r;
	    	}
    		if (i >= n) {
	    		break;
	    	}
	    	l = r;
	    	r = 0;
    	}
    	cout << max << "/1" << endl;	
    }
    return 0;
}
