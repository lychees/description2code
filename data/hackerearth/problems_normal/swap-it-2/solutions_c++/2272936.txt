#define MOD 100000
#define N 1000000
int a[1111];
int main(void){
	std::ios::sync_with_stdio(false);cin.tie(0);
	int T;
	int i,j,n,k;
	int themin,minj;
	int t;
	cin>>T;
	while(T--)
	{
		cin>>n>>k;
		for(i=0;i<n;i++)
			cin>>a[i];
		for(i=0;i<n;i++)
		{
			if(k<=0)
				break;
			themin=a[i];
			for(j=i+1;j<n;j++)
			{
				if(j-i>k)
					break;
				if(a[j]<themin)
				{
					themin=a[j];
					minj=j;
				}
			}
			if(a[i]>themin)
			{
				t=a[minj];
				for(j=minj;j>i;j--)
					a[j]=a[j-1];
				a[i]=t;
				k-=minj-i;
			}
			//cout<<minj<<endl;
		}
		for(i=0;i<n;i++)
			cout<<a[i]<<' ';
		cout<<'\n';
	}
//	system("pause");
	return 0;
}
