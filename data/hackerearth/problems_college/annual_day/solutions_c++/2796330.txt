#include <iostream>
#include<algorithm>
#include<string.h>
using namespace std;
int call(int n)
{
	int count=0,x;
	while(n>0)
	{
		x=n%2;
		if(x)
		 count++;
		 n=n/2;
	}
	return count;
}	
int main()
{
    int t,n,k,sum,i;
    cin>>t;
    int a[107],d[108];
    while(t--)
    {
    	cin>>n>>k;
    	sum=0;
    	memset(d,0,sizeof(d));
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    		d[i]=call(a[i]);
    	}
    	sort(d,d+107,greater<int>());
    	for(i=0;i<k;i++)
    	{
    		sum+=d[i];
    	}
    	 cout<<sum<<endl;
    }
    return 0;
}
