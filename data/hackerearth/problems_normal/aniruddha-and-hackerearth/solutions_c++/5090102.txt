#include<iostream>
using namespace std;
int a[1000002];
long int ans[1000002];
int main()
{
	a[0]=0;a[1]=2;a[2]=3;a[3]=5;ans[0]=0;ans[1]=2;ans[2]=5;
	long int i,t;
	for(i=3;i<1000002;i++)
	{
		a[i]=(a[i-1]+a[i-2])%1000000007;
		ans[i]=(ans[i-1]+a[i])%1000000007;
	}
	cin>>t;
	while(t--)
	{
		cin>>i;
		cout<<ans[i]<<endl;
	}
	return 0;
}
