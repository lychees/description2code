#include<bits/stdc++.h>
#define ll long long
using namespace std;


ll gcd(ll a, ll b) {
	if(b == 0) {
		return a;
	}
	gcd(b,a%b);
}

int main() {
	ll t;
	ll n,m;
	ll ans[1000];
	memset(ans,0,sizeof(ans));
	for(ll i = 1; i <= 100; i++) {
			for(ll j = 1; j <= 100; j++) {
				if(i != j)
				for(ll k = 1; k <= 100; k++) {
					if(j!=k) {
					if(i!=j && j!=k &&i != k){
				    	ll gc = gcd(i , gcd(j , k));
				        ans[gc]++; 
					}
					}
				}
			}
	}
	scanf("%lld",&t);
	while(t--) {
	    scanf("%lld%lld",&n,&m);
		n = m-n;
		ll cnt = 0;
		for(int i = 1;i <= 100; i++) {
			if(n % i == 0) {
				cnt += ans[i];
			}
		}
		double ans = (double)cnt/(double)(100 * 99 * 98);
		printf("%0.6lf\n",ans);	
	}
	return 0;
}
