#include <bits/stdc++.h>
using namespace std;
#define MAX 1000001

int dp[MAX+2];
int dp1[MAX+2];

int main()
{
    long long t,m,n,c_R,c_K,i;
    string s;
    scanf("%lld",&t);
    while(t-->0){
    	scanf("%lld%lld",&m,&n);
    	cin>>s;
    	memset(dp,0,sizeof(dp));
    	memset(dp1,0,sizeof(dp1));
    	c_R=0;c_K=0;
    	int j=0,j1=0;
    	dp[0]=1;
    	dp1[0]=1;
    	for(i=0;i<s.size();i++){
    		if(s[i]=='R')j++;
    		if(s[i]=='K')j1++;
    		if(j>=m)c_R+=dp[j-m];
    		if(j1>=n)c_K+=dp1[j1-n];
    		dp[j]++;
    		dp1[j1]++;
    	}
    	printf("%lld %lld\n",c_R,c_K);
    }
    return 0;
}
