# include<bits/stdc++.h>
using namespace std;
# define LL long long

vector<LL> A;

void preprocess()
{
    LL i=0;
    LL val=0;
    while(val<=1000000000000)
    {   i++;
        val+=i;
        A.push_back(val);
    }

}

LL low(LL i,LL val)
{
    LL l=i,u=A.size()-1;
    while(l<u)
    {
        LL m=(l+u)>>1;
        if(A[m]<val)
            l=m+1;
        else
            u=m;
    }
    return l;
}

LL up(LL i,LL val)
{
  LL l=i,u=A.size()-1;
  while(l<u)
  {
      LL m=(l+u)>>1;
      if(A[m]>val)
        u=m;
      else
        l=m+1;
  }
  return u;
}

int main()
{
    int t;
    LL l,r,counter;
    preprocess();
    scanf("%d",&t);
    while(t--)
    {   counter=0;
        scanf("%lld %lld",&l,&r);
        for(LL i=0;i<A.size() && A[i]<r;i++)
        {
            counter=counter+(up(i,r-A[i])-low(i,l-A[i]));
        }
        printf("%lld\n",counter);
    }


}
