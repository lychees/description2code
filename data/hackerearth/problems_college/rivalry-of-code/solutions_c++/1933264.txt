#include <iostream>
#include<string>
using namespace std;

int main()
{
	int t, i, j, k;
	string s1, s2;

	cin >> t;
	while (t--)
	{
		cin >> s1;
		cin >> s2;
		int count = 0;

		int ans = 10000000;
		if (s1.size() <= s2.size())
		{
			for (i = 0; i<s2.size(); i++)
			{
				count = 0;
				for (j = 0; j<s1.size(); j++)
				{
					if (s1[j] != s2[j + i])
						count++;
				}
				ans = min(ans, count);

			}
		}
		else
		{
			for (i = 0; i<s1.size(); i++)
			{
				count = 0;
				for (j = 0; j<s2.size(); j++)
				{
					if (s2[j] != s1[j + i])
						count++;
				}
				ans = min(ans, count);

			}
		}


		cout << ans << endl;

	}




	return 0;
}
