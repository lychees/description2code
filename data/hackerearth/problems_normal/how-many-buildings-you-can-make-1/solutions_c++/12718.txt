#include <iostream>
using namespace std;
int cnt[1001];
int main()
{
    int t,n,i,j,ans;
    cin>>t;
    cnt[0]=0;
    for(i=1;i<1001;i++)
    {
    	cnt[i]=cnt[i-1];
    	for(j=1;j*j<=i;j++)
    	{
    		if(i%j==0)
    		cnt[i]++;
    	}    	
    }
    while(t--)
    {
    	cin>>n;
    	cout<<cnt[n]<<endl;    	
    }
    return 0;
}
