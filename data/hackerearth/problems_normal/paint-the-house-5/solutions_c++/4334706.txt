#include <iostream>
#include <vector>
#include <utility>
#include <algorithm>
using namespace std;
long n,d;
vector<pair<pair<long,long>,long>>arr;
bool comp(const pair<pair<long,long>,long> &a,const pair<pair<long,long>,long> &b)
{
	if(a.first.first==b.first.first)
	{
		if(a.second==b.second)
		return a.first.second<b.first.second;
		else
		return a.second>b.second;
	}
	else
	return a.first.first<b.first.first;
}
int main()
{
    cin>>n>>d;
    long a,b,c;
    for(long i=0;i<n;i++)
    {
    	cin>>a>>b>>c;
    	arr.push_back(make_pair(make_pair(a,b),c));
    }
    sort(arr.begin(),arr.end(),comp);
    long t,x,y,tt=0,tx=0,ty=0,ans=0;
    for(int i=0;i<n;i++)
    {
    	t=arr[i].first.first;
    	x=arr[i].first.second;
    	y=arr[i].second;
    	if(y>ty && d>ty*(t-tt))
    	{
    		d=d-(t-tt)*ty;
    		ty=y;
    		tt=t;
    		ans=ans+x;
    	}
    }
    cout<<ans;
    return 0;
}
