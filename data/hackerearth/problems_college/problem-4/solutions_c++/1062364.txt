#include<iostream>
#include<vector>
#include<algorithm>
#define c cin
#define o cout
using namespace std;
void outVec(const vector<char> &v);
int main(){
	string s;
	c>>s;
	int i;
	vector<char>v;
	for(i=0; i<s.length(); i++)
		v.push_back(s[i]);
	sort(v.begin(), v.end());
	outVec(v);
	return 0;	
}
void outVec(const vector<char> &v){
	int i;
	for(i=0; i<v.size(); i++)
		o<<v[i];
	o<<"\n";
}
