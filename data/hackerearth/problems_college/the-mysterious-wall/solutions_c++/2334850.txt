#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int search(char *pat, char *txt)
{
    int M = strlen(pat);
    int N = strlen(txt);
    int i,k,j,f=0,p=0,t=0,h=1,q=101;
    for (i = 0; i < M; i++)
    {
        p = (p + pat[i])%q;
        t = (t + txt[i])%q;
    }
    for (i = 0; i <= N - M; i++)
    {
        if ( p == t )
        {
        	int a1[26]={0},a2[26]={0};
            for (j = 0; j < M; j++)
            {
            	a1[txt[i+j]-97]++;
            	a2[pat[j]-97]++;
            }
            for(k=0;k<26;k++){
            	if(a1[k]!=a2[k]) break;
			}
            if (k == 26)
            {
                f=1;break;
            }
        }   
        if ( i < N-M )
        {
            t = (t - txt[i] + txt[i+M])%q;
            if(t < 0) 
              t = (t + q); 
        }
    }
    return f;
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--){
		char s1[100005],s2[100005];
		scanf("%s%s",s1,s2);
		if(search(s2,s1)==1) cout<<"YES\n";
		else cout<<"NO\n";
	}
}
