#include <bits/stdc++.h>
#define lli long long 

using namespace std;

int A[52];
int n,l,r;
lli dp[52][100005];
int vis[52][100005];
int tc;
lli f(int idx, int val)
{
	if ( val < l ) return 0;
	if ( idx == n ) {
		if ( val >= l && val <= r ) return 1;
		return 0;
	}
	if ( vis[idx][val] == tc ) return dp[idx][val];
	vis[idx][val] = tc;
	lli ans = 0;
	ans += f(idx+1,val);
	ans += f(idx+1, __gcd(val,A[idx]));
	dp[idx][val] = ans;
	return ans;
}

int main()
{
	int t;
	scanf("%d", &t);
	for ( tc = 1; tc <= t; tc++ ) {
		scanf("%d", &n);
	//	dp.clear();
		for ( int i = 0; i < n; i++ ) scanf("%d", &A[i]);
		scanf("%d%d", &l, &r);
		lli ans = 0;
		for ( int i = 0; i < n; i++ ) ans += f(i+1,A[i]);
		printf("%lld\n", ans);
	}
	return 0;
}
