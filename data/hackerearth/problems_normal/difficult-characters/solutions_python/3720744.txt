def construct_char_counts(string):
    chars_count = dict()
    for char in string:
        if char in chars_count:
            chars_count[char] += 1
        else:
            chars_count[char] = 1
    for index in xrange(26):
        if chr(index+97) not in chars_count:
            chars_count[chr(index+97)] = 0
    return chars_count
