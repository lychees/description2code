#include<iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int r,xc,yc,i,flag=0;
        cin>>r>>xc>>yc;
        signed int x[4],y[4];
        for(int j=0;j<4;j++)
        {
            cin>>x[j]>>y[j];
            x[j]-=xc;
            y[j]-=yc;
        }
    for(i=0;i<4;i++)
    {
        if((x[i]*x[i]+y[i]*y[i])<=r*r)
         flag+=1;
    }
    if(flag==4)
    cout<<"Yes"<<endl;
    else
    cout<<"No"<<endl;
        
    }
    
    
    return 0;
}
