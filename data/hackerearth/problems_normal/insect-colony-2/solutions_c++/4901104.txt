#include <iostream>
#include<vector>
using namespace std;

int main()
{
    int t,n;
    cin >> t;
    vector<long int> v;
    long int tt;
    int i,j;
    for(int p=0;p<t;++p)
    {
    	cin >> n;
    	for(j=0;j<n;++j)
    	{
    		cin >> tt;
    		v.push_back(tt);
    	}
    	for(i=0;i<n;++i)
    	{
    		if(v[i]%2==0)
    			v[i]=-1;
    	}
    	for(i=0;i<n;++i)
    	{
    		if(v[i]!=-1)
    			{
    				for(j=i+1;j<n;++j)
    				{
    					if(v[j]!=-1)
    					if((v[i]+v[j])%2==0)
    						{
    							v[i]=-1;
    							v[j]=-1;
    							break;
    						}
    				}
    				if(j==n)
    					break;
    			}
    	}
    /*	for(i=0;i<n;++i)
    	{
    //		cout << "v" << i <<" "<<v[i];
    	}*/
    	//cout << "\n";
    	for(i=0;i<n;++i)
    	{
    		if(v[i]!=-1)
    			break;
    	}
    	if(i==n)
    		cout <<"Yes\n";
    	else
    		cout << "No\n";
    	v.clear();
    }
    return 0;
}
