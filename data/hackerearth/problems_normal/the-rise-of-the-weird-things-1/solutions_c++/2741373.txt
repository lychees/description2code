#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
int main(){
	int n;
	vector<int>even;
	vector<int>odd;
	cin>>n;
	int a[n];
	int oddsum=0,evensum=0;
	for(int i=0;i<n;i++){
		int c;
		cin>>c;
		if(c%2==0){
			even.push_back(c);
			evensum+=c;
		}else{
			odd.push_back(c);
			oddsum+=c;
		}
	}
	sort(even.begin(),even.end());
	sort(odd.begin(),odd.end());
	
	for(int i=0;i<even.size();i++){
		cout<<even[i]<<" ";
	}
	cout<<evensum<<" ";
	for(int i=0;i<odd.size();i++){
		cout<<odd[i]<<" ";
	}
	cout<<oddsum;
	
}
