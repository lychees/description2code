#include<iostream>
#include<string.h>
using namespace std;
int main()
{

    int i,j,k,n,m,c[100000],la,flag=0;
    char a[100000];
    cin>>n;
    for(i=0;i<n;i++)
    {
        flag=0;
        cin>>a;
    //    cin>>b;
        la=strlen(a);
        //lb=strlen(b);
        for(j=0,k=la-1;j<la,k>=0;j++,k--)
        {
            if(a[j]==a[k])
            {
                flag=1;
            }
            else
            {
                flag=0;
                break;
            }
        }
        if(flag==1)
        {
            cout<<"YES"<<endl;
        }
        else
        {
            cout<<"NO"<<endl;
        }
    }
    return 0;
}
