#include<bits/stdc++.h>
using namespace std;
int dp[1000+3][1000+3];
int answer(int *arr,int start,int end)
{
   if(start>end)
	   return 0;
   if(start==end)
	   return 1;
   if(dp[start][end]==-1)
   {
      int min1 = answer(arr,start,end-1) + 1;
      for(int i=end-1;i>=start;i--)
      {
          if(arr[i]==arr[end])
          {
        	  min1=min(min1,answer(arr,start,i)+answer(arr,i+1,end-1));
          }
      }
      dp[start][end]=min1;
   }
   //cout<<start<<" "<<end<<" "<<dp[start][end]<<endl;
   return dp[start][end];
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int arr[n];
		for(int i=0;i<n;i++)
			cin>>arr[i];
		memset(dp,-1,sizeof(dp));
		dp[0][0]=1;
		//int found[1000+7];
		//memset(found,0,sizeof(found));
		cout<<answer(arr,0,n-1)<<endl;
	}
}
