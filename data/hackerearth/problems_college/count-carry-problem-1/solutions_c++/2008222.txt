#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string str1,str2;
    	cin>>str1>>str2;
    	int n1=str1.size();
    	int n2=str2.size();
    	int carry=0,count=0;
    	while(n1>0||n2>0)
    	{
    		if(n1>0&&n2>0)
    		{
    			carry=(carry+(str1[n1-1]-'0')+(str2[n2-1]-'0'))/10;
    		//	cout<<carry<<endl;
    			n1--,n2--;
    		}
    		else if(n2==0)
    		{
    			carry=(carry+(str1[n1-1]-'0'))/10;
    			n1--;
    		}
    		else
    		{
    			carry=(carry+(str2[n2-1]-'0'))/10;
    			n2--;
    		}
    		if(carry>0)
    		count++;
    	}
    	if(count==0)
    	cout<<"No carry operation"<<endl;
    	else if(count==1)
    	cout<<"1 carry operation"<<endl;
    	else
    	cout<<count<<" "<<"carry operations"<<endl;
    	
    }
}
