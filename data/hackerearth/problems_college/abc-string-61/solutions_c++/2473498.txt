#include<bits/stdc++.h>
using namespace std;
# define mod 1000000007
# define ll long long
ll binPow(ll base,ll exp)
{
      ll ans =1;
      while(exp)
        {
            if(exp %2==1)
            ans=(ans*base)%mod;
            exp=exp/2;
            base=(base*base)%mod;
      }
      return ans;
}
int main()
{
    ll temp1,temp2,ans,t,n;
    cin>>t;
    while(t--)
    {
        cin>>n;
        temp1=binPow(3,n);
        temp2=binPow(2,n);
        ans=temp1-(3*temp2)+3;
        while(ans<0)
            ans+=mod;
        cout<<ans%mod<<endl;
    }
}
