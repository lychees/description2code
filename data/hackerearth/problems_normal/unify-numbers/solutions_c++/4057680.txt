#include <bits/stdc++.h>
using namespace std;
#define Int long long
int main()
{
	int t,n;Int A[100000],sum=0;
	scanf("%d",&t);
	while(t--){
		sum=0;
		scanf("%d",&n);
		for(int i=0;i<n;i++)
		{
			scanf("%lld",&A[i]);
		}
		sort(A,A+n);
		Int median = A[n/2];
		if (n%2==0){
			median+=A[(n-1)/2];
			median /= 2;
		}
		for(int i=0;i<n;i++)
		{
			sum+=abs(median-A[i]);
		}
		printf("%lld\n",sum);
	}
    return 0;
}
