#include <iostream>
using namespace std;

int main()
{
    int t;
    long long int n,ans;
    cin >> t;
    while(t--){
    	cin >> n;
    	if(n==0){
    		ans=0;
    	}
    	else if(n<4){
    		ans=1;
    	}
    	else{
    		if(n%3==0){
    			ans=n/3;
    		}
    		else{
    			ans=n;
    		}
    	}
    	cout << ans << endl;
    }
    return 0;
}
