noOfBranches = int(raw_input())
studentCount = raw_input().split(" ")

counts = []

for i in studentCount:
    counts.append(int(i))


def printC(counts, total):
    maxNoOfRounds = max(counts) / 2 + max(counts) % 2 - 1
    divHash = {}
    for c in counts:
        if (divHash.has_key(c / 2 + c % 2) == False):
            divHash[c / 2 + c % 2] = 1
        else:
            divHash[c / 2 + c % 2] += 1
    divHash.keys().sort()
    values = [total]
    suma = 0
    for key in divHash.keys():
        suma += divHash[key]
        val = total - suma
        values.append(val)

    for j in range(0, len(values) - 1):
        print values[j]


printC(counts, noOfBranches)
