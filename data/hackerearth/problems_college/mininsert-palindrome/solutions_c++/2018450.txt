#include<bits/stdc++.h>
using namespace std;
int f(string str, int l, int h)
{
    if (l == h) return 0;
    if (l == h - 1) return (str[l] == str[h])? 0 : 1;

    if(str[l] == str[h])
    {
        return f(str,l+1,h-1);
    }
    else
    {
        return min(f(str,l,h-1),f(str,l+1,h))+1;
    }

}
int main()
{
    int t,N;
    string str;
    cin>>t;
    while(t--)
    {
        cin>>N;
        cin>>str;
        cout<<f(str,0,str.length()-1);
    }
}
