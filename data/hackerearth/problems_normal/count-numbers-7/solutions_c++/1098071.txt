#include<bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
    long long int k,i,j,t,tt,cnt=0,a,b;
    cin >> k >> t;
    ll ar[k];
    for(i=0; i<k; i++)
        cin >> ar[i];
        vector<long long int>pp[k+1];
    for(i=0; i<(1<<k); i++)
    {
        tt=1;
        cnt=0;
        for(j=0; j<k; j++)
        {
            if((i&(1<<j)))
            {
                cnt++;
                tt=tt*ar[j];
            }
        }
        pp[cnt].push_back(tt);
    }
    while(t--)
    {
        cin >> a >> b;
        long long int ans=0;
        for(i=0; i<k; i++)
            ans += ((b/ar[i]) - ((a-1)/ar[i]));
        for(i=2; i<=k; i++)
        {
            for(j=0; j<pp[i].size(); j++)
            {
                if(i&1)
                    ans += ((b/pp[i][j]) - ((a-1)/pp[i][j]));
                else
                    ans -= ((b/pp[i][j]) - ((a-1)/pp[i][j]));
            }
        }
        cout << ans << endl;
    }
    return 0;
}
