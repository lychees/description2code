#include <cstdio>
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int Process(int t)
{
    int ans = 0;
    while(t > 0)
    {
        ans += (t & 1);
        t = t >> 1;
    }
    return ans;
}

int main()
{
    int T,N,K,tmp;
    cin >> T;
    while(T--)
    {
        cin >> N >> K;
        vector<int> input;
        input.resize(N);
        int ans = 0;
        
        for(int i = 0; i < N; ++ i)
        {
            cin >> tmp;
            input[i] = Process(tmp);
        }

        sort(input.begin(),input.end(),greater<int>());

        for(int i = 0; i < K; ++i)
            ans += input[i];
        
        cout << ans << endl;
    }
}
