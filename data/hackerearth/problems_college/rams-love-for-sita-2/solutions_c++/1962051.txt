#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long int n,c,m,i,j,ans=0,x=0;
        cin>>n>>c>>m;
        if((c*m)>n){cout<<n/c<<endl;continue;}
        n=n-(c*m);
        ans+=m;
        if((c*(m-1))>n){ans+=((n/c)+1);cout<<ans<<endl;continue;}
        x=n/(c*(m-1));
        ans+=(x*m);
        j=n%(c*(m-1));
        n=j;
        ans+=((n/c)+1);
        cout<<ans<<endl;
    }
    return 0;
}
