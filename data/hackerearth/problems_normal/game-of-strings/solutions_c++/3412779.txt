const int N = 5e3+5;
int dp1[N][N];

int LCS1(string x,string y) {
  int n = x.size();
  int m = y.size();
  for(int i=0;i<=n;i++) {
    for(int j=0;j<=m;j++) {
      if(i==0 || j==0) {
        dp1[i][j] = 0;
      }
      else if(x[i-1] == y[j-1])  {
        dp1[i][j] =1+dp1[i-1][j-1];
      }
      else {
        dp1[i][j]= max(dp1[i-1][j],dp1[i][j-1]);
      }
    }
  }
  return dp1[n][m];
}
