#include<iostream>
#include<math.h>
using namespace std;

void count(long int a,long int b)
{
	long int n1,n2,count6infirst=0,count6insecond=0,count5infirst=0,count5insecond=0;
	n1=a,n2=b;
	for(int i=0;n1!=0;i++)
	{
		if(n1%10==6)
		{
			count6infirst=count6infirst+pow(10,i);
		}
		if(n1%10==5)
		{
			count5infirst=count5infirst+pow(10,i);
		}
		n1=n1/10;
	}
	
	for(int i=0;n2!=0;i++)
	{
		if(n2%10==6)
		{
			count6insecond=count6insecond+pow(10,i);
		}
		if(n2%10==5)
		{
			count5insecond=count5insecond+pow(10,i);
		}
		n2=n2/10;
	}
	cout<<a+b-count6infirst-count6insecond<<" "<<a+b+count5infirst+count5insecond;
}
int main()
{
		long int a,b;
		cin>>a>>b;
		count(a,b);
}
