#include <iostream>
#include <cstdio>
#include <cmath>

using namespace std;
int MOD = 1e9+7;

int sum[6000]; int DP[6000]; int tmp[6000];

int N, A, B, K;

int main()
{
    scanf("%d %d %d %d", &N, &A, &B, &K);

    for(int Ni = 1; Ni <= N; Ni++)
        for(int Nj = 1; Nj <= N; Nj++)
            if( Ni != Nj && abs(Ni-Nj) < abs(Ni-B) )
                DP[Ni]++;

    for(int Ki = 1; Ki < K; Ki++)
    {
        for(int Ni = 1; Ni <= N; Ni++)
        {
            tmp[Ni] = DP[Ni];
            sum[Ni] = (sum[Ni-1]+DP[Ni])%MOD;
        }

        for(int Ni = 1; Ni <= N; Ni++)
        {
            int u = min(Ni+(int)abs(B-Ni)-1, N);
            int d = max(Ni-(int)abs(B-Ni)+1,1)-1;
            DP[Ni] = (sum[u]-sum[d]+MOD)%MOD;
            DP[Ni] = (DP[Ni]-tmp[Ni]+MOD)%MOD;
        }
    }

    printf("%d\n", DP[A]);
}
