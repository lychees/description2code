#include <bits/stdc++.h>
using namespace std;

int main()
{
    //freopen("file1.txt","r",stdin);
    int n;
    double x[100009],y[100009],f[100009];
    double dp[100009];
    
    cin>>n;
    
    for(int i=1;i<=n;i++)
    {
    	cin>>x[i]>>y[i]>>f[i];
    }
    
    
    dp[1] = f[1];
    
    
    for(int i=2;i<=n;i++)
    {
    	dp[i] = -10000000000000009;
    	for(int j=1;j<i;j++)
    	dp[i] = max(dp[i], dp[j]-sqrt((x[i]-x[j])*(x[i]-x[j]) + (y[i]-y[j])*(y[i]-y[j])));
    	dp[i]+=f[i];
    }
    
    printf("%.6f",dp[n]);
    //cout<<setprecision(6)<<dp[n];
    
}
