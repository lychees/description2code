#define md int(1e7+100)

#define FOR(i,a,b) for(int i=(a),_b=(b); i<=_b; i++)
#define FORE(i,a,b) for(int i=(a),_b=(b); i>=_b; i--)

using namespace std;

typedef pair<int, int> II;
typedef vector<II> vi;
typedef long long ll;

struct data {
    int pos, val, gr;
} a[110];

bool operator< (data a, data b) {
    return a.val > b.val;
}

int test, n, m, p, res, vt, r;

int main()
{
    ios::sync_with_stdio(0);
    //freopen("inp.txt", "r", stdin);
    cin >> test;
    while (test--) {
        cin >> n >> m;
        FOR(i,1,n) {
            cin >> a[i].val;
            a[i].pos = i;
        }
        sort(a+1, a+n+1);
        p = 0;
        FOR(i,1,n) {
            p++;
            if (p > m) p = 1;
            a[i].gr = p;
            if (a[i].pos == 1) {
                r = a[i].gr;
                vt = i;
            }
        }
        res = 0;
        FOR(i,1,n)
            if (a[i].pos != 1)
                if (a[i].gr == r)
                    if (a[i].val > a[vt].val) res++;
        cout << res << endl;
    }
    return 0;
}
