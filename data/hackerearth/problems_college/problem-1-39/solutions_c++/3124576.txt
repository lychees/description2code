#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;
int main()
{
long long int t,a,b,c;
cin>>t;
while(t--)
{
	long double n;
	cin>>a>>b>>c;
	n=(double)pow(a,b)/(double)c;
	if((long long int)n==n )
	cout<<(long long int)pow(a,b)<<endl;
	else if(abs((long long int)n*(c)-pow(a,b))>abs((long long int)n*(c)+c-pow(a,b)))
	cout<<(long long int)n*(c)+c<<endl;
	else 
	cout<<(long long int)n*(c)<<endl;
}
return 0;
}
