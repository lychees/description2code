#include <iostream>
#include <map>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	string s;
    	cin >> s;
    	map <int,int> m;
    	for(int i=0;i<s.length();++i)
    	m[s[i]]++;
    	bool ok = false;
    	for(auto i : m)
    	{
    		if(i.second > 1)
    		{
    			ok = true;
    			break;
    		}
    	}
    	cout << ((ok) ? "Yes\n" : "No\n");
    }
    return 0;
}
