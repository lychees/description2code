#include<bits/stdc++.h>
using namespace std;
int main()
{
   string s;
   getline(cin,s);
   int t;
   scanf("%d",&t);
   for(int i=0;i<t;i++)
   {
      long long int a,b;
      scanf("%lld %lld",&a,&b);
      int len=(int)s.length();
      a=(a-1)%len;
      b=(b-1)%len;

      if(s[a]==s[b])
      printf("Yes\n");
      else
      printf("No\n");
   }
   return 0;
}
