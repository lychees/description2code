import math
test = input()
while (test > 0):
    s = raw_input()
    s = s.split()
    g = long(s[0])
    b = long(s[1])
    if (b-g) > 1:
        print -1
    else:
        n = g+1
        num = math.factorial(n)
        den = math.factorial(b)*math.factorial(n-b)
        num /= den
        num = ((num*math.factorial(b)) %
               1000000007 * math.factorial(g)) % 1000000007
        print num
    test = test-1
