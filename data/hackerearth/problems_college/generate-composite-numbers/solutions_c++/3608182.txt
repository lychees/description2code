#include<bits/stdc++.h>
#define mx 10003
using namespace std;
bool prime[mx];

void seive(){


   for(int i = 1;i<mx;i++)prime[i]=true;
   int sq = sqrt(mx);
   for(int j = 2;j<=sq;j++){

      if(prime[j]==true){

        for(int i = j*j;i<mx;i+=j)prime[i]=false;
      }
   }
}
int main(){

   seive();
   int n;
   while(cin>>n){


       for(int i = 1;i<=n;i++){

          if(prime[i]==false)printf("%d ",i);
       }
       cout<<endl;
   }
}
