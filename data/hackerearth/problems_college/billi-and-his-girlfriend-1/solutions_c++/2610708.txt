#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
LL modexp(LL a,LL b,LL c)
{
	LL ans=1;
	while(b)
	{
		if(b&1) ans=(ans*a)%c;
		a=(a*a)%c;
		b>>=1;
	}
	return ans;
}
 
int last(int N,int N2, int K)
{
	int mod=pow(10.0,K);
	return (int)modexp(N,N2,mod);
}
int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	int a,b,k;
    	cin >> a >> b >> k;
    	int dig  = last(a,b,k);
    	int s = 0;
    	while(dig)
    	{
    		s += dig %10;
    		dig/=10;
    	}
    	cout << s << "\n";
    }
    return 0;
}
