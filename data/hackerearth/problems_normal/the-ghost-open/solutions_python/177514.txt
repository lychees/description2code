def matches(n):
	global match
	if n>1:
		if n%2==0:
			n=n/2
			match=match+n
			#print "even",match,n
		else:
			n=n/2+1
			match=match+n-1
			#print "odd",match,n
			
		matches(n)
		
	return match

n = int(raw_input())

for i in range(n):
	match=0
	p = int(raw_input())
	print matches(p)
