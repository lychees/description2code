#include <iostream>
using namespace std;

int main()
{
    int a,b,n,c;
    cin>>a>>b>>n;
     if(n==1)
       cout<<a;
     else if(n==2)
        cout<<b;
     else
      {
        for(int i=3;i<=n;i++)
           {
           	c=a+b;
           	a=b;
           	b=c;
           }
        cout<<c;
      }  
}
