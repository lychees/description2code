#include <iostream>
using namespace std;

int main()
{
	int test_cases, i=1;
	cin >> test_cases;
	string element, output = "";
	while(test_cases > 0){
		cin >> element;
		for(i = 1; i < element.size(); i++){
			if(!(element[i] == element[i-1])){
				output = output + element[i-1];
			}
		}
		output = output + element[i-1];
		cout << output << "\n";
		output = "";
		test_cases--;
	}
    return 0;
}
