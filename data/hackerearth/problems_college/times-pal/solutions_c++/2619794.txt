using namespace std;

bool isPal(string s){
	int i=0, j = s.length(), mid = int(ceil(float(j)/2));
	bool check = true;
	while(i<=mid && check){
		if(s[i] != s[j-i-1])
			check=false;
		i++;
	}
	return check;
}
int main()
{
	int T;
	scanf("%d",&T);
	for(int k=1; k<=T; k++){
		string a,b, res;
		cin>>a;
		cin>>b;
		int count=0;
		for(int i=0; i<=a.length(); i++){
			res= a.substr(0,i) + b + a.substr(i,a.length()-i);
			//cout<<"res: "<<res<<endl;
			if(isPal(res))
				count++;
		}
		printf("%d\n",count);
	}
	return 0;
}
