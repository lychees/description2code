#include<bits/stdc++.h>
#include<string>
using namespace std;
int calc(int h,int m){
    if (h == 12) h = 0;
    if (m == 60) m = 0;


    int hour_angle = 0.5 * (h*60 + m);
    int minute_angle = 6*m;

    int angle = abs(hour_angle - minute_angle);

    angle = min(360-angle, angle);

    return angle;

}
int main(){
    int t;
    scanf("%d",&t);
    while(t--){
        int h,m;
        scanf("%d%d",&h,&m);
        printf("%d\n",calc(h,m));
    }


    return 0;
}
