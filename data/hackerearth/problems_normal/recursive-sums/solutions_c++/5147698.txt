#include<iostream>
using namespace std;
#define ull unsigned long long
ull singleDigit(ull n){
    if(n<10)return n;
    int digit=0;
        while(n>0){
            digit+=(n%10);
            n/=10;
        }
    return singleDigit(digit);
}

        }
