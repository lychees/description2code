#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,l1,l2,l,d;
	string a,b;
	/*cin>>a>>b;
	if(a==b)
	cout<<"Equal\n";
	else
	cout<<"Not equal\n";*/
	cin>>t;
	while(t--)
	{
		d=0;
		cin>>a>>b;
		if(a==b)
		cout<<"Exact\n";
		else
		{
			l1=a.length();
			l2=b.length();
			l=min(l1,l2);
			for(int i=0;i<l;i++)
			{
				if(a[i]!=b[i])
				d++;
			}
			d+=abs(l1-l2);
			cout<<"Non Exact "<<d<<"\n";
		}
	}
}
