#include <bits/stdc++.h>
using namespace std;
int main() {
	string s;
	cin>>s;
	int t,l,r;
	cin>>t;
	while(t--){
		int c=0;
		cin>>l>>r;
		for(int i=l-1;i<r-1;i++){
			if(s[i]==s[i+1])
				c++;
		}
		cout<<c<<endl;
	}
	return 0;
}
