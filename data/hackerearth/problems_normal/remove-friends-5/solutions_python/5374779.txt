for i in range(input()):
    N, K = [int(i) for i in raw_input().split()]
    P = [int(i) for i in raw_input().split()]
    stack = []
    count = 0
    for p in P:
    	while len(stack) > 0 and count < K and stack[-1] < p:
    		stack.pop()
    		count += 1
    	stack.append(p)
    if count < K:
    	stack = stack[0:K-count]
    print " ".join([str(i) for i in stack])
