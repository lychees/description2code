#include<iostream>
#include<cstdio>
#include<cmath>
#include<cassert>

using namespace std;

int main()
{
    long long int n,k,m;
    int t;
	// scanf("%d", &t);
	cin>>t;
	while(t--)
	{
		// scanf("%d", &n);
		// scanf("%d", &k);
		// scanf("%d", &m);
		cin>>n>>k>>m;
		if(n==0)cout<<0<<endl;
		long long sum=0;
		for(long long int i=1;i<=n;i++)
			sum=sum+(k*i*i*i);
		if(sum>=m)cout<<(sum-m)<<endl;
		else cout<<0<<endl;
		}	
    return 0;
}
