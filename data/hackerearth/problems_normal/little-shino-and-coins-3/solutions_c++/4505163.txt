#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	char s[5001];
	int k;
	cin>>k;
	cin>>s;
	int i,j,l;
	int n=strlen(s);
	int dp[n+1][26];
	for(i=0;i<=n;i++)
	{
		for(j=0;j<26;j++)
		dp[i][j]=0;
	}
	for(i=1;i<=n;i++)
	{
		for(j=0;j<26;j++)
		dp[i][j]=dp[i-1][j];
		dp[i][s[i-1]-'a']++;
	}
	/*for(i=1;i<=n;i++)
	{
		for(j=0;j<26;j++)
	cout<<dp[i][j]<<" ";
	cout<<endl;
	}*/

	int ans=0;
	int count=0;
	for(i=1;i<=n;i++)
	{
		for(j=i;j<=n;j++)
		{
			count=0;
			for(l=0;l<26;l++)
			{
				//cout<<dp[i][l]<<" ";
				if(dp[j][l]-dp[i-1][l]>0)
				count++;
			}
			//cout<<endl;
			if(count==k)
			ans++;
		}
	}
	cout<<ans<<endl;
	
	return 0;
}
