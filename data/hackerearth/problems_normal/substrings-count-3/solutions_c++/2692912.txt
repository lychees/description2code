
#include <bits/stdc++.h>
#include <stdio.h>
#define lli long long int
#define pii pair<int,int>
#define plli pair<lli,lli>
#define X first
#define Y second
using namespace std;

int n;
int b[(int)3e5][26] , bi=0;
set<int> m[(int)3e5];
char s[15];
int sl;

void add(int i , int j)
{
	int p = 0;
	while (j<sl)
	{
		if (b[p][s[j]-'a']==0) b[p][s[j]-'a'] = ++bi;
		p = b[p][s[j]-'a'];
		m[p].insert(i);
		j++;
	}
}

int find()
{
	int p = 0 , i = 0;
	while (i<sl)
	{
		if (b[p][s[i]-'a'] == 0) return -1;
		p = b[p][s[i]-'a'];
		i++;
	}
	return p;
} 

int calc(int l , int r)
{
	int id = find();
	if (id==-1) return 0;
	
	set<int>::iterator L = m[id].lower_bound(l);
	set<int>::iterator R = m[id].upper_bound(r);
	int ans = distance(L , R);
	
	return ans;
}

int main()
{
	scanf("%d" , &n);

	for (int i=0 ; i<n ; i++)
	{
		scanf("%s" , s);
		sl = strlen(s);
		for (int j=0 ; j<sl ; j++)
			add(i , j);
	}
	
	int q;
	scanf("%d" , &q);
	for (int t=0 ; t<q ; t++)
	{
		int l,r;
		scanf("%d %d %s" , &l , &r , s);
		sl = strlen(s);
		l-- , r--;
		printf("%d\n" , calc(l,r) );
	}
	
    return 0;
}
