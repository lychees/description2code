#include <bits/stdc++.h>

using namespace std;

const int N = 1e3 + 10;

int a[N], n;

int ok(int x) {
	int st = 1;
	for (int i = 0; i < n; i++) {
		if (a[i] + x < st) {
			return 0;
		} 
		if (a[i] > st) {
			st = max(st, a[i] - x) + 1;
		} else {
			st = min(st, a[i] + x) + 1;
		}
	}
	return 1;
}

int bs_lower_bound(int l, int h, int x) {
	while (l < h) {
		int mid = (l + h) / 2;
		if (x <= ok(mid)) {
			h = mid;
		} else {
			l = mid + 1;
		}
	}
	return l;
}

int main() {
	int t;
	scanf("%d", &t);
	for (int tt = 1; tt <= t; tt++) {
		scanf("%d", &n);
		for (int i = 0; i < n; i++) {
			scanf("%d", &a[i]);
		}
		printf("%d\n", bs_lower_bound(0, 1e9 + 1, 1));
	}
	return 0;
}
