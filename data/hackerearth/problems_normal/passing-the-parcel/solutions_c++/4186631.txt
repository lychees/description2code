#include <iostream>
#include<vector>
#include<string>
using namespace std;

int main()
{
    int n;
    cin>>n;
    vector<int> v;
    for(int i=0;i<=n;i++)
    v.push_back(i);
    v[0]=0;
    string s;
    cin>>s;
    int x=0;
    int pos=1;
    while(n>1)
    {
    	if(x==s.length())
    	x=0;
    	
    	if(pos>=v.size())
    	pos=1;
    	
    	if(s[x]=='b' && v.size()>1)
        {v.erase(v.begin()+pos);--n;}
        if(s[x]=='a')
        ++pos;
        
        ++x;
    }
    cout<<v[1]<<endl;
    return 0;
}
