#include <iostream>
using namespace std;
char a[] = "abcdefghijklmnopqrstuvwxyz";
int main()
{
    int test,len,i,indx;
    cin>>test;
    string s;
    while(test--)
    {
                 cin>>s;
                 len = s.size();
                 for(i=0;i<len;i++)
                 {
                                   indx = s[i]-'a';
                                   indx = indx + i;
                                   cout<<a[indx%26];
                 }
                 cout<<"\n";
    }
    return 0;
}
