N = int(raw_input())
Descriptions = []
for i in range(N):
    Descriptions.append(raw_input())

priceList = []
for desc in Descriptions:
    sPrice = ""
    for c in desc.split('$')[1]:
        if c.isalpha():
            break
        if c.isdigit():
            sPrice += c
    priceList.append('$' + str(int(sPrice)))
for p in priceList:
    print p
