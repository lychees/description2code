#include <bits/stdc++.h>
using namespace std;

int main()
{
    long s,q;
    set<long> se;
    scanf("%ld%ld",&s,&q);
    while(q--)
    {
        long a;
        scanf("%ld",&a);
        long num;
        num=__gcd(s,a);
        if(se.find(num)==se.end())
        {
            printf("%ld\n",num);
            se.insert(num);
        }
        else
            printf("-1\n");
    }
    return 0;
}
