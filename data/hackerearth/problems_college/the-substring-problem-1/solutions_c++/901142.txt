#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		int n,k,j;
		string s;
		vector<string> v;
		int c[26];
		memset(c,0,sizeof(c));
		cin>>n;
		for(k=0;k<n;k++){
			cin>>s;
			v.push_back(s);}
	    char i;		
		for( i='a';i<='z';i++){
			for(j=0;j<n;j++){
				size_t found=v[j].find((char)i);
				if (found!=string::npos) c[i-'a']++;}
			}
		int max=0;
		char max_c;
		for(j=0;j<26;j++){
			if (c[j]>max) {
				max=c[j];
				max_c=j+'a';}}	
		cout<<max_c<<endl;
	}			
	return 0;
}
