#include <iostream>
#include<math.h>
using namespace std;
int sod(int m)
{
	int s=0;
	while(m!=0)
	{
		s=s+(m%10);
		m=m/10;
	}
	return s;
}
int main()
{
	int n,l,b,i,cost;
	double temp;
	cin>>n;
	cost=0;
	int a[n];
	for(i=0;i<n;i++)
	{
		cin>>a[i];
	}
	cin>>l>>b;
	int ar=l*b;
	for(i=0;i<n;i++)
	{
	if(a[i]<=10)
	{
		temp=(a[i]*sod(a[i])*ar);
		temp=temp-(0.25*temp);
		cost=cost+temp;
	}
	else if(a[i]<=20)
	{
		temp=(a[i]*sod(a[i])*ar);
		temp=temp-(0.30*temp);
		cost=cost+temp;
	}
	else if(a[i]<=50)
	{
		temp=(a[i]*sod(a[i])*ar);
		temp=temp-(0.20*temp);
		cost=cost+temp;
	}
}
cout<<int(round(cost));
return 0;
}
