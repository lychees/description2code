#include<iostream>

using namespace std;

int main()
{
        int n;
        cin >> n;
        int a[n];
        int i=0;

        for(i=0;i<n;i++){
                cin >> a[i];
        }

        int se=0,so=0;

        for(i=0;i<n;i++){
                if(i%2==0){
                        se=se+a[i];
                }
                else if(i%2!=0){
                        so=so+a[i];
                }
        }

        if(se==so)
                cout<< "YES\n";
        else
                cout << "NO\n";
}
