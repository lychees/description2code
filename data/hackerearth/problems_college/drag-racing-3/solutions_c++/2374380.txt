ll next_int;
char in_char;
 inline ll GetInt(){
    in_char = ' ';
    while((in_char < '0') | (in_char > '9')){
        in_char = get();
    }
    next_int = 0;
    while((in_char >= '0') && (in_char <= '9')){
        next_int *= 10;
        next_int += in_char - 48;
        in_char = get();
    }
    return next_int;
}

using namespace std;

map<string,int> m;

int main(){
    ll n;
    cin>>n;
    string a,b;
    for(int i=1;i<=((1<<n)-1);i++){
        cin>>a>>b;
        if(m.find(a)!=m.end()){
            if(m[a]!=-1)
                m[a]=1;
        }
        else
        	m[a]=1;
        m[b]=-1;
    }
    for(map<string,int> :: iterator ii=m.begin(); ii!=m.end();ii++){
        if((*ii).second==1){
            cout<<(*ii).first<<endl;
            break;
        }
    }
    return 0;
}
