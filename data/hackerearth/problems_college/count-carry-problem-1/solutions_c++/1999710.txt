#include <iostream>
using namespace std;

int main()
{
	int n,flag,t1,t2,t;
	long a,b;
	scanf("%d",&n);
	while(n--)
	{	flag=0;
		scanf("%ld%ld",&a,&b);
		t=0;
		while(a>0 && b>0)
		{
			t1=a%10;
			t2=b%10;
			a=a/10;
			b=b/10;
			if((t1+t2+t)>=10)
			{
				t+=(t1+t2);
				t-=10;
				flag++;
			}
		}
		while(a>0)
		{
			t1=a%10;
			if((t1+t)>=10)
			{
				t-=10;
				t+=t1;
				flag++;
			}
			a/=10;
		}
		while(b>0)
		{
			t2=b%10;
			if((t2+t)>=10)
			{
				t-=10;
				t+=t2;
				flag++;
			}
			b/=10;
		}
		if(flag>1)
		printf("%d carry operations\n",flag);
		else if(flag==0)
		printf("No carry operation\n");
		else
		printf("%d carry operation\n",flag);
	}
	return 0;
}
