#include<bits/stdc++.h>
using namespace std;

#define lli long long
#define pb push_back
#define mp make_pair
#define gc getchar_unlocked
#define lld "%I64d"
#define DEBUG(x) cout<<">value ("<<#x<<") : "<<x<<endl;
#define mod 1000000007

const int lmt = 100005;
bool vis[lmt];
vector<int> adj[lmt];

void dfs(int x){
    if(vis[x]) return;
    vis[x] = true;
    for(int i=0;i<adj[x].size();i++)
        dfs(adj[x][i]);
}

int main(){
    int t,n,e,a,b;
    scanf("%d",&t);
    while(t--){
        for(int i=0;i<lmt;i++)
            adj[i].clear();
        memset(vis,false,sizeof(vis));
        scanf("%d %d",&n,&e);
        while(e--){
            scanf("%d %d",&a,&b);
            adj[a].pb(b);
            adj[b].pb(a);
        }
        int ans = 0;
        for(int i=0;i<n;i++){
            if(!vis[i]){
                ans++;
                dfs(i);
            }
        }
        printf("%d\n",ans);
    }
    return 0;
}
