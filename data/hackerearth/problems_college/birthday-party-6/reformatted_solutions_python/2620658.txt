import math
import sys


def nCr(n, r):
    f = math.factorial
    return (f(n)*f(n-1)) / f(n-r)


t = input()
while t > 0:
    t = t-1
    m, n = map(int, sys.stdin.readline().split())
    if n > (m+1):
        print '-1'
    else:
        ans = nCr(m+1, n) % 1000000007
        print ans
