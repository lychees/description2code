#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,m,a;cin>>t;
    long long tmp1,tmp2,tmp3,tmp4,tmp5,tmp6,tmp7,tmp8,tmp9;
    while(t--){
    	cin>>n>>m>>a;
    	tmp1=a*a;
    	tmp2=tmp1*a;
    	tmp3=n+m*tmp2;
    	tmp4=1000000000000;
    	for(long long i=a;i*i*i<=tmp4;i++){
    		tmp5=i/a;
    		for(long long j=i;i*j*j<=tmp4;j++){
    			tmp6=j/a;
    			tmp7=(m+tmp5*tmp6-1)/(tmp5*tmp6)*a;
    			tmp8=i*j*tmp7;
    			tmp9=max(0LL,tmp3-tmp8);
    			tmp7+=(tmp9+i*j-1)/(i*j);
    			tmp4=min(tmp4,i*j*tmp7);
    		}
    	}
    	cout<<tmp4<<endl;
    }
    return 0;
}
