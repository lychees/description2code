#include<iostream>
#include<string>
#include<algorithm>

using namespace std ;


int a[1001][1001];

int dp[1001][1001];

int main()
{
	int t; cin>>t;
	while(t--)
	{
		int n ;
		cin>>n;
		string str[1001];
		 for(int i=1;i<=n;i++)		  
		  {
		  	
		  	cin>>str[i];
		  	 a[i][0]=0;
		  	 
		  	 for(int j=1;j<=n;j++)
		  	 {
		  	 	if(str[i][j-1]=='#')a[i][j]=1;
		  	 	
		  	 	 else
		  	 	  a[i][j]=0;
			   }
		  	 
		  }
		   
		  for(int i=0;i<=n;i++)
		  {
		  	dp[0][i]=0;
		  	dp[i][0]=0;
		  }
		  
		  for(int i=1;i<=n;i++)
		  {
		  	for(int j=1;j<=n;j++)
		  	{
		  		dp[i][j]=dp[i][j-1]+a[i][j]+(dp[i-1][j]-dp[i-1][j-1]);
			  }
		  }
		  
		  int flag =0;
		  
		  for(int i=1;i<=n;i++)
		  {
		  	if(2*(dp[i][n])==dp[n][n])
		  	{
		  		flag=1;
		  		break;
			  }
			  if(2*(dp[n][i])==dp[n][n])
			   {
			   	flag =1;
			   	break;
			   }
		  }
		 // cout<<dp[n][n]<<endl;
		  if(flag )
		   cout<<"YES"<<endl;
		   else
		    cout<<"NO"<<endl;
	}
	return 0;
}
