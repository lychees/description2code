#include <cstdlib>
#include <iostream>

using namespace std;
int T,I,px,py,bx,by;
int main(int argc, char *argv[])
{   cin>>T;
    for(I=1;I<=T;I++)
    {   cin>>px>>py>>bx>>by;
        if((px>=bx && py>=by)||(px>=by && py>=bx))
           cout<<"Possible"<<endl;
        else
           cout<<"Not Possible"<<endl;
                     
    }
    
    return EXIT_SUCCESS;
}
