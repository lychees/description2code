#include <iostream>
#include<stack>
#include<string.h>
using namespace std;
int main()
{
    long long int  len,t,i,j,ans;
    char str[200002],c;
    std::stack<char> stk;
    cin>>t;
    while(t--)
    {
    	
    	cin>>str;
    	len=strlen(str);
    	stk.push(str[0]);
    	for(i=1;i<len;i++)
    	{
    		if(stk.empty()==0 && stk.top()!=str[i])
    				stk.pop();
    		else
    			stk.push(str[i]);
    	}
    	cout<<stk.size()<<endl;
    	while (!stk.empty())
     		stk.pop();
    }
    return 0;
}
