#include<bits/stdc++.h>
using namespace std;
int main(){

         int f[21];
         f[0] = 1;

        for(int i = 1;  i<= 20 ; ++i ){
           f[i] = (i*f[i-1])%10;
        }
        int t;
        cin >> t;
        while(t--)
        {
           int a , b;
           cin >> a >> b;
           int ans = 0;
           for(int i = a ; i <= b ; ++i )
                 ans+=f[i];

                 cout << ans%10 << endl;

        }


}
