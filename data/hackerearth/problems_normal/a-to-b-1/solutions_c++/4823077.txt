#include <bits/stdc++.h>
using namespace std;
#define pb push_back
#define ll long long int
#define INF 10000000000000000
#define mod 1000000007

ll modular(ll a,ll b,ll c)      //(a^b)%c
{
	ll ans=1;
	while(b!=0)
	{
		if(b%2==1)
		{
			ans*=a;
			ans%=c;
		}
		a=a*a;
		a%=c;
		b/=2;
	}
	return ans;
}

int main()
{
	ll n;
	cin>>n;
	ll a[n+1];
	ll b[n+1];
	ll mul=1;
	int count=0;
	for(int i=1;i<=n;i++)
	{
		cin>>a[i];
		if(a[i]==0)
			{count++;continue;}
		mul=((mul%mod)*(a[i]%mod))%mod;
	}
	ll q;
	cin>>q;
	while(q--)
	{
		int que;
		cin>>que;
		if(que==1)
		{
			ll ind;
			cin>>ind;
			if(count==0)
			{
				ll temp=modular(a[ind],mod-2,mod);
				ll ans=((mul%mod)*(temp%mod))%mod;
				cout<<ans<<endl;
			}
			else if(count==1)
			{
				if(a[ind]==0)
					cout<<mul<<endl;
				else cout<<"0\n";
			}
			else cout<<"0\n";
			
		}
		if(que==0)
		{
			ll ind,val;
			cin>>ind>>val;
			if(a[ind]==val)
				continue;
			else if(val==0)
			{
				count++;
				ll temp;
				temp=modular(a[ind],mod-2,mod);
				mul=((mul%mod)*(temp%mod))%mod;
			}
			else if(a[ind]==0)
			{
				count--;
				mul=((mul%mod)*(val%mod))%mod;
			}
			else
			{
				ll temp;
				temp=modular(a[ind],mod-2,mod);
				mul=((mul%mod)*(temp%mod))%mod;
				mul=((mul%mod)*(val%mod))%mod;
			}
			a[ind]=val;
		}
	}
}
