#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--) {
    	char s[105];
    	cin>>s;
    	int n=strlen(s);
    	int flag=0, ans=0, cur=0;
    	for(int i=0;i<n;i++) {
    		if(s[i] == '|')
    			flag=1, ans^=cur, cur=0;
    		else 
    			cur++;
    	}
    	if(flag and cur) ans^=cur;
    	if(ans) cout<<"Alice\n";
    	else cout<<"Bob\n";
    }
    return 0;
}
