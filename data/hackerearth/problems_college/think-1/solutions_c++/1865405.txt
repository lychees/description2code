#include<cstdio>
#include<iostream>
#include<algorithm>
#include<vector>

#include<cstdlib>
using namespace std;
long long rev(int n)
{
	long long sum =0;
	while(n!=0)
	{
		int rem = n%10;
		sum = sum*10 + rem;
		n = n/10;
	}
	return sum;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int n;
		cin>>n;
		long long p = n*n;
		long long q = rev(n);
		if(q!=n)
		{
		long long r = q*q;
		if(p == rev(r))
		{
			cout<<q<<"\n";
		}
		else
		{
			cout<<"No"<<"\n";
		}
	}
	else
	{
		cout<<"No"<<"\n";
	}
}
}
