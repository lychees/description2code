#include <iostream>
#include<vector>
#include<stdio.h>
using namespace std;

int main()
{
    long t,l,r;
    int prime[1000002],divide[1000002];
    for(long i=2;i<=1000000;i++)
    	if(prime[i]==0)
    		for(long j=i*i;j<=1000000;j+=i)
    			prime[j]=1;
    	divide[1]=1;
    for(long i=2;i<=1000000;i++)
    {
    	if(prime[i]==0)
    	{
    		divide[i]=0;
    	}
    	else 
    		divide[i]=1;
    	if(i==4)
    		divide[i]=0;
    	divide[i]=divide[i-1]+divide[i];
    }

    scanf("%ld",&t);
    while(t--)
    {
    	scanf("%ld%ld",&l,&r);
    	printf("%ld\n",divide[r]-divide[l-1]);
    	
    }
    return 0;
}
