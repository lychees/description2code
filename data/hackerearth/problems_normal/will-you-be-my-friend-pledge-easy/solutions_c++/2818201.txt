#include<iostream>
using namespace std;
int fact(int);
int q[1000];
main()
{

    int a,b,t[1000],i,v[1000],m,j,cnt,p=0;
    cin>>a;
    cin>>b;

       for(int i=2;i<=a;i++)
    {

        if(a%i==0)
        {
            v[j]=i;

            j++;
            cnt++;

        }
    }

    for(i=0;i<b;i++)
    {
        cin>>t[i];
        int l=fact(t[i]);

        for(int k=0;k<cnt;k++)
        {
            for(int n=0;n<l;n++)
            {
                if(v[k]==q[n])
                {

                     p++;
                     goto l;
                }
            }
        }

        l:;


    }
    cout<<p<<endl;

}
int fact(int m)
{
    int j=0;
    for(int i=2;i<=m;i++)
    {

        if(m%i==0)
        {
            q[j]=i;
            j++;

        }
    }

    return j;

}
