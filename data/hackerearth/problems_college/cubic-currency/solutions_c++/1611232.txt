#include<stdio.h>
#include<iostream>
#include<string.h>
using namespace std;


 
long long int count( int S[], int m, int n )
{
   
    long long int table[n+1];
 
    
    memset(table, 0, sizeof(table));
 
   
    table[0] = 1;
 
   
    for(int i=0; i<m; i++)
        for(int j=S[i]; j<=n; j++)
            table[j] += table[j-S[i]];
 
    return table[n];
}
// Driver program to test above function
int main()
{
   
   int t,i,n;
   
   scanf("%i",&t);
   while(t--)
   {
   	scanf("%i",&n);
   	i=0;
   int arr[n+1];
   arr[0]=1;
   for(i=1;(i+1)*(i+1)*(i+1)<=n;i++)
   arr[i]=(i+1)*(i+1)*(i+1);
   
    printf("%lld\n", count(arr, i, n));
   }
    return 0;
}
