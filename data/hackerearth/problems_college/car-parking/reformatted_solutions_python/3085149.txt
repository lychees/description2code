for i in xrange(input()):
    s = list(raw_input().strip())
    l = len(s)
    for i in xrange(l):
        if s[i] == 'B':
            if (i-1) >= 0:
                s[i-1] = 'A'
            if (i-2) >= 0:
                s[i-2] = 'A'
        elif s[i] == 'S':
            if (i-1) >= 0:
                s[i-1] = 'A'
            if (i+1) < l and s[i+1] != 'S':
                s[i+1] = 'A'
        if s[i] != '-':
            s[i] = 'A'
    print s.count('-')
