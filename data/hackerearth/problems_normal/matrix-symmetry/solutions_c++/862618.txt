#include <iostream>
using namespace std;

int main()
{
    int t,i,j;
    scanf("%d",&t);
    while(t--)
    {
    	int hor=1,vert=1;
    	char a[50][50];
    	int n;
    	scanf("%d",&n);
    	for(i = 0 ; i < n ; i++)
    	{
    		for(j = 0 ; j < n ; j++)
    		{
    			cin>>a[i][j];
    		}
    	}
    	for(i = 0 ; i <= n-1 ; i++)
    	{
    		for(j = 0 ; j <= n-1 ; j++)
    		{
    			if(a[i][j]!=a[n-1-i][j])
    			hor=0;
    			if(a[i][j]!=a[i][n-1-j])
    			vert=0;
    			
    		}
    	}
    	if(hor && vert)
    	   printf("BOTH\n");
    	else if(vert)
    	   printf("VERTICAL\n");
    	else if(hor)
    	   printf("HORIZONTAL\n");
    	else if(!hor && !vert)
    	   printf("NO\n");
    	
    }
    
    return 0;
}
