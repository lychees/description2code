#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int n,sumEven=0,sumOdd=1,cnt=0;
		cin>>n;
		while(n!=1)
		{
			if(n%2==0)
			{
				sumEven+=n;
				n/=2;
			}
			else
			{
				sumOdd+=n;
				n=3*n+1;
			}
			cnt++;
		}
		//cout<<sumEven<<" "<<sumOdd<<endl;
		int res=sumEven%cnt;
		int res1=sumOdd%cnt;
		if(res>res1)
			cout<<"Even Rules"<<endl;
		else if(res1>res)
			cout<<"Odd Rules"<<endl;
		else
			cout<<"Tie"<<endl;
	}
}
