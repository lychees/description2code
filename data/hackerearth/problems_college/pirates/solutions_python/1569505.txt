t=input()

l=map(int,raw_input().split())

z=len(l)

pre=l[0]
count=0
ans=0
for i in xrange(1,z):
    if abs(pre-l[i])==1:
        #print "yesy", pre, l[i], count
        count+=1
        pre=l[i]
    else:
        if count>ans:
            ans=count
        count=0
        pre=l[i] 
if count>ans:
    ans=count           
if ans==0:
    print 0
else:
    print ans+1
