#include <bits/stdc++.h>
using namespace std;
long long int gcd(long long int a, long long int b) 
{
    return b == 0 ? a : gcd(b, a % b);
}

long long int a[17],b[17];
int main()
{	
	long long int x=1;
	long long int n,t;
	long long int d,w,c;
	cin >> d;
	for(int r=1;r<=d;r++)
	{
		cin >> n;
		t=x << n;
		memset(a,0,sizeof a);
		memset(b,0,sizeof b);
		for(long long int i=0;i<n;i++)
		cin >> a[i] >> b[i];
		cin >> w >> c;
		long long int ans=0,temp1,temp2,ct;
		for(long long int i=0;i<t;i++)
		{
			temp1=0,temp2=0,ct=0;
			for(long long int j=0;j<n;j++)
			{
				if(i&(x<<j))
				{
					temp1+=a[j];
					temp2+=b[j];
					ct++;
					if(ct>=c)
						break;
				}
			}
			if(temp2<=w && ct<=c)
				ans=max(ans,temp1);
		}
		if(ans==0)
			ans=-1;
		cout << "For Day #" << r << ":\n" << ans << endl;
	}
    return 0;
}
