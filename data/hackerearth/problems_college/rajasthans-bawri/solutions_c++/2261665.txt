#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int pk,pw,bk,bw,temp;
    	bool flag=0;
    	cin>>pk>>pw>>bk>>bw;
    	if(pk>pw)
    	{
    		temp=pw;
    		pw=pk;
    		pk=temp;
    	}
    	if(bk>bw)
    	{
    		temp=bw;
    		bw=bk;
    		bk=temp;
    	}
    	if(pk>=bk && pw>=bw)
    	flag=1;
       	if(flag)
    	cout<<"Possible"<<endl;
       	else
       	cout<<"Not Possible"<<endl;
    }
    return 0;
}
