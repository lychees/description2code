#include <iostream>
#include <map>
using namespace std;

int main()
{
    string s;
    map<char,int> m;
    int i,max;
    char c;
    
    cin>>s;
    for(i=0;i<s.size();i++){
    	m[s[i]]++;
    }
    map<char,int> :: iterator it;
    max = 0;
    for(it=m.begin();it!=m.end();it++){
    	if(max<it->second){
    		max=it->second;
    		c=it->first;
    	}
    }
    
    if(c=='u'){
    	cout<<"Thanks both"<<endl;
    }else if(c=='h' || c=='e' || c=='a' || c=='d' || c=='r' || c=='s'){
    	cout<<"Thanks head rush"<<endl;
    }else if(c=='c' || c=='b' || c=='i' || c=='n' || c=='g' || c=='l'){
    	cout<<"Thanks cubing club"<<endl;
    }else{
    	cout<<"none"<<endl;
    }
    
    return 0;
}
