#include <bits/stdc++.h>
using namespace std;

int main()
{
long long	int n,ans=0,x[1000000],y[1000000];

    cin>>n;
   // cout<<n<<endl;
    for(int i=0;i<n;i++)
    {
    	cin>>x[i];
    
    	
    }
    
     for(int i=0;i<n;i++)
    {
    	cin>>y[i];
    
    	
    }
    	
    	sort(x,x+n);
    	sort(y,y+n);
    	
    	
    	for(int i=0;i<n;i++)
    	{
    		ans+=x[i]*y[i];
    	}
    	
    cout<<ans<<endl;
    return 0;
}
