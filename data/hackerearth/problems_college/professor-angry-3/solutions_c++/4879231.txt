#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,k;
    scanf("%d",&t);
    while(t--){
    	scanf("%d%d",&n,&k);
    	int a[n];
    	int nf=0;
    	for(int i=0;i<n;i++){
    		scanf("%d",&a[i]);
    	}
    	for(int i=0;i<n;i++){
    		if(a[i]<=0)
    			nf++;
    	}
    	if(nf>=k)
    		printf("NO\n");
    	else
    		printf("YES\n");
    	
    }
    return 0;
}
