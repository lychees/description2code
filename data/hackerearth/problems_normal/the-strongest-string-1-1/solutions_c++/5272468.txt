#include<iostream>
#include<vector>
#include<string>
#include<algorithm>
using namespace std;
struct spe{
	int index;
	char c;
};
struct comparator{
	bool operator()(const spe& a,const spe& b){
		return (a.c>b.c);
	}
};
int main()
{
    int n;
    string s,sans;
    vector<int> visit(26,0);
    cin>>n>>s;
    vector<spe> vec(n);
    for(int i=0;i<n;i++){
    	vec[i].index=i;
    	vec[i].c=s[i];
    }
    int temp=-1;
    stable_sort(vec.begin(),vec.end(),comparator());
    for(int i=0;i<n;i++){
    	if(visit[(int)(vec[i].c)-97]==0){
    		if(vec[i].index>temp){
    			sans.push_back(vec[i].c);
    			visit[(int)(vec[i].c)-97]=1;
    			temp=vec[i].index;
    		}
    	}
    }
    cout<<sans;
    return 0;
}
