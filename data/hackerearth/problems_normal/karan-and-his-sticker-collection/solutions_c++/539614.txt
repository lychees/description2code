#include <iostream>
#include<stdlib.h>
using namespace std;
long long int input()
{
  long long int n = 0;
  int ch, flag = 1;
  ch = getchar_unlocked();
  if(ch == '-')
    {
      flag = -1;
      ch = getchar_unlocked();
    }
  while(ch >= '0' && ch <= '9')
    {
      n = (n << 3) + (n << 1) + (ch - '0');
      ch = getchar_unlocked();
    }
  //printf("Returning %lld\n",n*flag);
  return n*flag;
}
int main()
{
  long long int num,diff = 0,i;
  long long int *arr;
  num = input();
  //k = input();
  //scanf("%lld %lld",&num,&k);
  arr = (long long int*)malloc(num * sizeof(long long int));
  for(i = 0; i < num; i++)
    {
      arr[i] = input();//scanf("%lld",&arr[i]);
      if(arr[i]<300000000)
		diff++;	
    }
    printf("%lld\n",diff);
    return 0;
}
