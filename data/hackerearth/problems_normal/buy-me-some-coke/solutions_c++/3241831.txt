#include<bits/stdc++.h>
using namespace std;
int c,n1,n5,n10,tot;
int dp[160][110][60];
int go(int idx, int five, int ten) {
	if(idx==c+1)
		return 0;
	int one=tot-(idx-1)*8-five*5-ten*10;
	if(dp[idx][five][ten]!=-1)
		return dp[idx][five][ten];
	int &r=dp[idx][five][ten];
	r=1e9+3;
	if(ten>=1)
		r=min(r,1+go(idx+1,five,ten-1));
	if(five>=2)
		r=min(r,2+go(idx+1,five-2,ten));
	if(five>=1 and one>=3)
		r=min(r,4+go(idx+1,five-1,ten));
	if(one>=8)
		r=min(r,8+go(idx+1,five,ten));
	if(ten>=1 and one>=3)
		r=min(r,4+go(idx+1,five+1,ten-1));
	return r;
}
int main() {
	int t;
	cin>>t;
	while(t--) {
		memset(dp,-1,sizeof(dp));
		cin>>c>>n1>>n5>>n10;
		tot=n1+5*n5+10*n10;
		if(tot<8*c)
			cout<<"-1\n";
		else
			cout<<go(1,n5,n10)<<endl;
	}
	return 0;
}
