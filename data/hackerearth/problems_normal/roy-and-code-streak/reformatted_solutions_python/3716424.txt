_ = input()

while _:
    _ -= 1

    di = {}
    n = input()
    c, m = 0, 0
    while n:
        n -= 1
        k, v = map(int, raw_input().split())
        if v:
            if k not in di:
                di[k] = v
                c += 1
        else:
            if m < c:
                m = c
            c = 0
    print m if m > c else c
