#include <iostream>
#include <vector>
using namespace std;

vector< vector<int> > A;
vector< vector<bool> > V;
vector< vector<int> > R;

int getValue(int i, int j)
{
	if(V[i][j] || !A[i][j])
		return 0;
	V[i][j] = true;
	return 1+getValue(i, j+1)+getValue(i, j-1)+getValue(i+1, j)+getValue(i-1, j)
		+getValue(i+1, j+1)+getValue(i+1, j-1)+getValue(i-1, j+1)+getValue(i-1, j-1);
}

int main()
{
    int t, n, m, i, j, l, s;
    cin>>t;
    while(t--)
    {
    	cin>>n>>m;
    	A = vector< vector<int> >(n+2, vector<int>(m+2, 0));
    	R = vector< vector<int> >(n+2, vector<int>(m+2, 0));
    	V = vector< vector<bool> >(n+2, vector<bool>(m+2, false));
    	for(i=1; i<=n; ++i)
    		for(j=1; j<=m; ++j)
    			cin>>A[i][j];
    	for(i=1; i<=n; ++i)
    		for(j=1; j<=m; ++j)
    			R[i][j] = getValue(i, j);
    	l = s = 0;
    	for(i=1; i<=n; ++i)
    		for(j=1; j<=m; ++j)
    			if(R[i][j]>0)
    			{
    				s++;
    				if(R[i][j]>l)
    					l = R[i][j];
    			}
    	cout<<s<<" "<<l<<endl;
    }
    return 0;
}
