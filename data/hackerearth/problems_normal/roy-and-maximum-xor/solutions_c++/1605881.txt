#include <iostream>
#include <bitset>
using namespace std;

int main()
{
    long t;
    cin>>t;
    while(t--)
    {
    	long long n;
    	cin>>n;
    	bitset<64> b(n-1),m(0);
    	for(int i=63;i>=0;i--)
    	{
    		if(b[i]!=0)
    		{
    			m[i]=1;
    			//cout<<i<<endl;
    			break;
    		}
    	}
    	
    	long long cnt = (n - m.to_ullong())*2;
    	
    	for(int i=0;i<64&&m[i]==0;i++)m[i]=1;
    	if(n==1)
    	{
    		m.reset();
    		cnt=1;
    	}
    	
    	cout<<m.to_ullong()<<" "<<cnt<<endl;
    }
    return 0;
}
