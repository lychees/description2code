#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define gt(x) scanf("%d",&x)
#define pr(x) printf("%lld",x)
#define loop(i,x,n) for(int i = 0; i < n; i++)
#define maxm 9999999
#define all(o) (o).begin(), (o).end()
#define pb(x) push_back(x)
#define mp(x,y) make_pair((x),(y))
#define mset(m,v) memset(m,v,sizeof(m))

int main(){
        int t;
        gt(t);
        while(t--){
            int n,p;
            gt(n);
            gt(p);
            int x,y;
            vector<pair<int,int> >vec;
            for(int i = 0; i < p; i++){
                cin>>x>>y;
                vec.push_back(make_pair(x,y));
            }
            int c = 0;
            for(int i = 1; i < pow(2,n); i++){
                int flag = 0;
                for(int j = 0; j < p; j++){
                        int a = 1 << vec[j].first-1;
                        int b = 1<< vec[j].second-1;
                       // cout<<a<<" "<<b<<endl;
                    if((a & i) && (b & i)){
                        flag = 1;
                    }
                }
                if(flag == 0)
                    c++;
            }
            cout<<c<<endl;

        }
        return 0;
}
