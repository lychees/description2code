#include <bits/stdc++.h>
using namespace std;
#define ll long long int
ll pal(string s)
{
	ll i,j,k,len,c;
	len=s.length();
	char ch;
	i=0;
	j=len-1;
	while(i<=j)
	{
		ch=s[i];
		s[i]=s[j];
		s[j]=ch;
		i++;
		j--;
	}
	
	bool L[len][len];

	for(i=0;i<len;i++)
	{
		L[i][i]=true;
		
	}
	
	for(c=2;c<=len;c++)
	{
		for(i=0;i<len-c+1;i++)
		{
			j=i+c-1;
			
			if(c==2&&s[i]==s[j])
			L[i][j]=true;
			else
			{
				if(s[i]==s[j])
				L[i][j]=L[i+1][j-1];
				else 
				L[i][j]=false;
			}
		
		}
	}
	
	for(i=0;i<len;i++)
	{
		if(L[0][i]==true)
		j=i;
	}

	return len+len-(j+1);
}
int main()
{
    ll t;
    scanf("%lld",&t);
    while(t--)
    {
    	string s;
    	cin>>s;
    	cout<<pal(s)<<endl;
    }
    return 0;
}
