#include <iostream>
using namespace std;

int main(){
    int t;
    cin >> t;
    while (t > 0){
    	string a,b;
    	cin >> a >> b;
    	int* l = new int[26];
    	for (int i = 0; i < 26; i++) l[i] = 0;
    	for (int  i = 0; i < a.length(); i++)	
    		l[a[i] - 'a'] += 1;
    	int tt = 0;
    	for (int i = 0; i < b.length(); i++){
    		if (l[b[i] - 'a'] >= 1){
    			cout << "YES" << endl;
    			tt = 1;
    			break;
    		}
    	}
    	if (tt == 0)
    		cout << "NO" << endl;
    	t--;
    }
    return 0;
}
