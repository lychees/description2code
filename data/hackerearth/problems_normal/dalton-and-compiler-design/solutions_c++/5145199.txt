#include <iostream>
using namespace std;

int main()
{
    unsigned long int t,i;
    cin>>t;
    if(t>=1&&t<=100000)
    {
    	unsigned long int a[t];
    	for(i=0;i<t;i++)
    	{
    		cin>>a[i];
    	}
    	for(i=0;i<t;i++)
    	{

    		if(a[i]%2==0)
    		cout<<a[i]/2<<" "<<a[i]/2<<"\n";
    		else
    		cout<<(a[i]/2)<<" "<<(a[i]/2)+1<<"\n";
    	}
    }
    return 0;
}
