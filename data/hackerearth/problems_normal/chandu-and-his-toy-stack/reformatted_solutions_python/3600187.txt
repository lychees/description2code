test = int(raw_input())
while test:
    N = int(raw_input())
    X = int(raw_input())
    Y = int(raw_input())
    initial, final = [], []
    for i in range(N):
        a, b = map(int, raw_input().split())
        initial.append(a)
        final.append(b)
    initial = sorted(initial)
    final = sorted(final)
    effort = 0
    for i in range(N):
        if initial[i] > final[i]:
            temp = initial[i] - final[i]
            effort += temp * Y
        elif initial[i] < final[i]:
            temp = final[i] - initial[i]
            effort += temp * X
    print effort

    test -= 1
