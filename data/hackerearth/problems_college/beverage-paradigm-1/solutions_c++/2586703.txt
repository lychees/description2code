#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int *a;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int n;
	cin>>n;
	a=(ll int *)malloc(sizeof(ll int)*n);
	ll int i;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
	}
	ll int x;
	cin>>x;
	sort(a,a+n);
	ll int p,q,r;
	bool flag=0;
	for(i=0;i<(n-2);i++)
	{
		p=i;
		q=i+1;r=n-1;
		while(q<r)
		{
			if((a[p]+a[q]+a[r])==x)
			{
				flag=1;
				break;
			}
			else if((a[p]+a[q]+a[r])<x)
			{
				q++;
			}
			else
			{
				r--;
			}
		}
		if(flag)
		break;
	}
	if(flag)
	cout<<"True"<<endl;
	else
	cout<<"False"<<endl;
	return 0;
}
