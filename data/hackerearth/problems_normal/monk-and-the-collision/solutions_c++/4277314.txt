#include<iostream>
#include<vector>
using namespace std;
int main(){
	int t,n,temp,c;
	vector<int> a;
	a.resize(10);
	scanf("%d",&t);
	while(t--){
		scanf("%d",&n);
		for(int i=0;i<10;i++) a[i]=-1;
		for(int i=0;i<n;i++){
			scanf("%d",&temp);
			temp%=10;
			a[temp]++;
		}
		c=0;
		for(int i=0;i<10;i++) {
			if(a[i]>0) c+=a[i];
		}
		printf("%d\n",c);
		
	}
	return 0;
}
