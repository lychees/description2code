#include <iostream>
#include <math.h>
#include <string.h>
using namespace std;

int main() {
    int T,N,M,D,i,j;
    cin>>T;
    while(T--){
        cin>>N>>M>>D;
        int s_X[N],s_Y[N],t_X[M],t_Y[M];
        i=0;
        while(i<N){
            cin>>s_X[i];
            cin>>s_Y[i];
            i++;
        }
        i=0;
        while(i<M){
            cin>>t_X[i];
            cin>>t_Y[i];
            i++;
        }
        int f=0,d,c=0,t[M];
        memset(t,1,sizeof(t));
        for(i=0;i<N;i++){
            int x=s_X[i],y=s_Y[i];
            //cout<<"S: "<<s_X[i]<<" "<<s_Y[i]<<" ";
            for(j=0;j<M;j++){
            	if(t[j]){
            		d = abs(x-t_X[j]) + abs(y-t_Y[j]);
	                if(d <= D){
	                    c++;
	                    t[j] = 0;
	                    //cout<<"T: "<<t_X[j]<<" "<<t_Y[j]<<" d="<<d<<endl;
	                }
	                if(c > M/2){
	                    f=1;
	                    break;
	                }
            	}
            }
            if(f) break;
        }
        if(f) cout<<"YES\n";
        else cout<<"NO\n";
    }
    return 0;
}
