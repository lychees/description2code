#include <iostream>
#include <string>
using namespace std;

int main()
{
    long long int t;
    cin>>t;
    while(t--)
    {
    	string a;
    	cin>>a;
    	bool flag=true;
    	
    	long long int i;
    	long long int n=a.size();
    	for(i=0;i<n-1;i++)
    	{
    		if(a[i]==a[i+1])
    		{
    			flag=false;
    			break;
    		}
    	}
    	
    	if(flag==true)
    	cout<<"KISS"<<endl;
    	else
    	cout<<"SLAP"<<endl;
    }
    return 0;
}
