

def main():
	N, K = [int(i) for i in raw_input().split(' ')]
	if 1 <= N <= 1000000 and 1 <= K <= 2000000:
		A = [int(i) for i in raw_input().split(' ')]
		A.sort()
		l = 0
		h = len(A) - 1
		while l < h:
			if A[l] + A[h] == K:
				return 'YES'
			elif A[l] + A[h] < K:
				l += 1
			else:
				h -= 1
		else:
			return 'NO'

if __name__ == '__main__':
	output = main()
	print output
