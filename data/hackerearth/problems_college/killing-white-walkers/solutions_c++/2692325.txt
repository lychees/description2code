#include<bits/stdc++.h>
#define gc getchar
typedef long long int ll;
using namespace std;
ll read_int() {
char c = gc();
while(c<'0' || c>'9') c = gc();
ll ret = 0;
while(c>='0' && c<='9') {
ret = 10 * ret + c - 48;
c = gc();
}
return ret;
}
struct node
{
    ll info;
    node *link;
}*head=NULL;
int main()
{
    ll t,n,i;
    node *tmp,*p,*r;
    cin>>t;
    while(t--)
    {
        cin>>n;
        if(n==0)
           cout<<0<<"\n";
        else{
        head=NULL;
        for(i=1;i<=n;i++)
        {
           tmp=new node;
           tmp->info=i;
           tmp->link=NULL;
           if(head==NULL)
           {
               head=tmp;
               p=tmp;
           }
           else
           {
               p->link=tmp;
               p=p->link;
           }
        }
        p->link=head;
        p=head;
        while(true)
        {
            r=p;
            p->link=p->link->link;
            p=p->link;
            if(r==p)
                break;
        }
        cout<<p->info<<"\n";}
    }
    return 0;
}
