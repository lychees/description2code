#include <bits/stdc++.h>
typedef unsigned long long int ull;
using namespace std;
void solve(int n)
{
	int i, j;
	for(i = 0; i < n - 1; i++)
		cout << " ";
	cout << "/\\" << endl;
	for(i = 2; i <= n; i++)
	{
		for(j = 0; j < n - i; j++)
			cout << " ";
		cout << "/";

		if(i % 2 == 0)  // even
		{
			for(j = 0; j < (i - 2) / 2; j++)
				cout << " /";
			cout << "  ";
			for(j = 0; j < (i - 2) / 2; j++)
				cout << "\\ ";
		}
		else
		{
			for(j = 0; j < i / 2; j++)
				cout << " /";
			for(j = 0; j < i / 2; j++)
				cout << "\\ ";
		}
		cout << "\\";
		cout << endl;
	}
	for(i = n; i > 1; i--)
	{
		cout << "  ";
		for(j = 0; j < n - i; j++)
			cout << " ";
		cout << "\\";

		if(i % 2 == 0)  // even
		{
			for(j = 0; j < (i - 2) / 2; j++)
				cout << " \\";
			cout << "  ";
			for(j = 0; j < (i - 2) / 2; j++)
				cout << "/ ";
		}
		else
		{
			for(j = 0; j < i / 2; j++)
				cout << " \\";
			for(j = 0; j < i / 2; j++)
				cout << "/ ";
		}
		cout << "/";
		cout << endl;
	}
	for(i = 0; i < n - 1; i++)
		cout << " ";
	cout << "  \\/" << endl;
}
int main()
{
	int i, j, k, n, t;
	cin >> n;
	solve(n);
	solve(n+1);
	return 0;
}
