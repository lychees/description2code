	#include <bits/stdc++.h>
	using namespace std;
	#define max 10000000
	#define size 100000
	#define rep(i,n) for(int i=0;i<n;i++)
	int main()
	{	int n,k,p,m=0,l;
		long long int a[size];
		cin>>n>>k;
	    rep(i,n)
	    	cin>>a[i];
	    sort(a,a+n);
	    int left=0,right=a[n-1],mid;
	    l=k;
	    ++right;
	    while(left<right){
	  		m=1;
	  		k=l;
	    	mid=left+(right-left)/2;
	    	p=a[0]+mid;
	    	k--;
	    	while(m<n&&k){
	    		if(p+mid<a[m]){
					p=a[m]+mid;
	    			k--;
	    		}	    		
	    		m++;
	    	}
	    	if(m>=n){
	    		if(p+mid>=a[n-1])
	    		right=mid;
	    		else 
	    		left=mid+1;
	    		
	    		continue;
	    	}
	    	if(k==0){
	    		if(p+mid>=a[n-1]){
	    			right=mid;
	    		}else left=mid+1;
	    	
	    		continue;
	    	}
	    	
	    }
	    cout<<right<<endl;
	    return 0;
	}
