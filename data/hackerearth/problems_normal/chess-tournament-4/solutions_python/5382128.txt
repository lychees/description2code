N = input()
A = [[]]
for ith in xrange(2**N - 1):
    A.append(map(int, raw_input().split()))
players = range(2**N - 1, -1, -1)    
while len(players) > 1:
    pairs = zip(players[::2], players[1::2])
    try:
        players = [x if A[x][y] else y for x, y in pairs]
    except:
        exit()
print players[0] + 1
