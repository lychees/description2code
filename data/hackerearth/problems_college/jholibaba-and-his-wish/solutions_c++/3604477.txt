#include<bits/stdc++.h>
using namespace std;
long long int arr[1000001];
int main()
{
	long long int n,i,s,max,mul,init,tmp1,tmp;
	scanf("%lld",&n);
	s=init=0;
	for(i=1;i<=n;i++)
	{
		scanf("%lld",&arr[i]);
		s=s+arr[i];
		init=init+i*arr[i];
	}
	max=init;
	//printf("%d\n",max);
	mul=n-1;
	for(i=1;i<n;i++)
	{
		tmp1=s-arr[i];
		tmp=mul*arr[i]-tmp1;
		if(tmp+init>max)
		{
			max=tmp+init;
		}
		init=tmp+init;
		//printf("%d\n",init);
	}
	printf("%lld",max);
	return(0);
}
