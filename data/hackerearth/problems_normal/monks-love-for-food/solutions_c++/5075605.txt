#include <iostream>
#include <stack>

using namespace std;

int main()
{
	int n;
	cin>>n;
	int num;
	stack<int> s;
	for(int i=0;i<n;i++)
	{
		cin>>num;
		if(num==1)
		{
			if(s.empty())
				cout<<"No Food"<<endl;
			else
				{
					cout<<s.top()<<endl;
					s.pop();
				}
		}
		else if(num==2)
		{
			int c;
			cin>>c;
			s.push(c);
		}
		else
			cout<<"Error"<<endl;
	}
}
