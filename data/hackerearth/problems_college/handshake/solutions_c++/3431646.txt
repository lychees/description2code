#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
    int t,a,n;  
   scanf("%d",&t);  
   while(t--){  
     a=0;  
     scanf("%d",&n);  
     while(n--)  
     {  
       if(n>0)  
       {  
         a+=n;  
       }  
     }  
     printf("%d\n",a);  
   }  
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */   
    return 0;
}
