#include<iostream>
#include<cstring>
#include"stdio.h"
#include<string>
#include<vector>
#include "time.h"
using namespace std;
#define MAX1 1050
std::vector<int> aa[MAX1];
std::vector<int> index123;
int solve()
{
	while(index123.size()!=1)
	{
		std::vector<int> new1;
		int i=0;
		while(i+1<index123.size())
		{
			int t1=index123[i];
			int t2=index123[i+1];
			if((aa[t2-1])[t1]==1)
				new1.push_back(t2);
			else 
				new1.push_back(t1);
			i+=2;
		}
		std::swap(new1,index123);
		new1.clear();
	}

	return index123[0]+1;
}
int main()
{

	int N;
	cin>>N;
	int count=1;
	while(N--)
		count*=2;
	int lines = count-1;
	int num;
	for(int i=0;i<lines;i++)
		for(int j=0;j<=i;j++)
		{
			cin>>num;
			aa[i].push_back(num);
		}
	for(int i=0;i<=lines;i++)
		index123.push_back(i);
	int result = solve();
	cout<<result<<endl;
	return 0;
}
