#include <bits/stdc++.h>
#define ll long long int
#define mod (int)(1e9 + 7)

using namespace std;

int t, n;


int main()
{
	cin >> t;	
	for (int tc = 1; tc <= t; tc++) {
		char s1[105], s2[105];
		cin >> s1 >> s2;
		n = strlen(s1);
		
		cout << "Case " << tc << ": ";
		if (strlen(s1) != strlen(s2)) {
			cout << "-1\n";
			continue;
		} 
		int x1, x2, x3, x4;
		x1 = x2 = x3 = x4 = 0;
		
		for (int i = 0; i < n; i++) {
			if (s1[i] == '?' && s2[i] == '0')	x1++;
			if (s1[i] == '?' && s2[i] == '1')	x2++;
			if (s1[i] == '1' && s2[i] == '0')	x3++;
			if (s1[i] == '0' && s2[i] == '1')	x4++;
		}
		
		if (x3 > x4) {
			if ((x3 - x4) > x2) {
				cout << "-1\n";
			} else {
				cout << x1 + x2 + x3 << endl;
			}
		} else {
			cout << x1 + x2 + x4 << endl;
		}
	}
	return 0;
}
