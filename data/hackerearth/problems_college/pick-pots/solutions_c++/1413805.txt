#include<bits/stdc++.h>
using namespace std;

const int N=1e6+10;
const int md=1e9+7;

int dp[N];

void preprocess() {
    dp[1]=1;
    dp[2]=2;
    for(int i=3;i<N;i++) {
        dp[i]+=dp[i-1]+dp[i-2]+1;
        if(dp[i]>=md) {
            dp[i]-=md;
        }
    }
    return;
}
int main() {
    preprocess();
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        int n;
        scanf("%d",&n);
        printf("%d\012",dp[n]);
    }
    return 0;
}
