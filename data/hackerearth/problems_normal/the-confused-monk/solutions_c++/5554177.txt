#include <iostream>
#include <vector>
using namespace std;

unsigned long long int M=1000000007;

unsigned long long int func(unsigned long long int x,unsigned long long int n,unsigned long long int M)
{
    if(n==0)
        return 1;
    else if(n%2 == 0)        
        return func((x*x)%M,n/2,M);
    else                           
        return (x*func((x*x)%M,(n-1)/2,M))%M;

}

unsigned long long int gcd(unsigned long long int a,unsigned long long int b)
{
    unsigned long long int q,r;
    if(a>b)
        swap(a,b);
    q = b/a;
    r = b%a;
    if(r==0)
        return a;
    else if(r==1)
        return 1;
    else return gcd(r,a);
}

int main()
{
    unsigned long long int n,i,k=1,m,s;
    cin>>n;
    vector <unsigned long long int> vec(n);
    for(i=0;i<n;i++)
    cin>>vec[i];
    
    for(i=0;i<n;i++){
    k = k* vec[i];
    k = k%M;
    }
    
    if(n==1)
    {
    	m=vec[0];
    	s = func(k,m,M);
    	cout<<s%M<<endl;
    }
    
    else{
    m = gcd(vec[0],vec[1]);
    
    for(i=2;i<n;i++)
    m = gcd(m,vec[i]);
    s = func(k,m,M);
    cout<<s%M<<endl;
    }
    return 0;
}
