#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;

int main()
{
	int t;
	cin>>t;
	long long n, ans;
	while(t--){
	 cin>>n;	 
	 int dp[n];
	 for(int i=0; i<n; i++) dp[i]=0;
	 dp[0]=dp[1]=1;
	 for(int i=2; i<n+2; i++) dp[i] = dp[i-1]+dp[i-2];
	 ans = dp[n+1]*dp[n+1];
     cout<<ans<<endl;
	}
    return 0;
}
