void swap(char *x, char *y)
{
	char temp;
	temp = *x;
	*x = *y;
	*y = temp;
}

void permute(char *a, int i, int n)
{
	int j;
	if(i==n)
		ar.push_back(a);
	else
	{
		loop(j,i,n+1)
		{
			swap(a+i,a+j);
			permute(a,i+1,n);
			swap(a+i,a+j);
		}
	}
}
