int c(int h,int m)
{
    if (h == 12) h = 0;
    if (m == 60) m = 0;
    int ho = 0.5 * (h*60 + m);
    int mo = 6*m;
    int a = abs(ho - mo);
    a = min(360-a, a);

    return a;
}
