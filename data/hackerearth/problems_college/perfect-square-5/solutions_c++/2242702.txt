#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ll int test;
	scanf("%lld",&test);
	while(test--)
	{
		ll int n;
		scanf("%lld",&n);
		ll int x,ans;
		x=sqrt(n);
		if(x*x ==n)
		{
			printf("YES\n");
		}
		else
		{
			if((x*x) <n)
			{
				if(abs(n-(x*x))<abs(((x+1)*(x+1))-n))
				{
					ans=n-(x*x);
					printf("-%lld\n",ans);
				}
				else
				{
					ans=(((x+1)*(x+1))-n);
					printf("+%lld\n",ans);
				}
			}
		}
	}
	return 0;
}
