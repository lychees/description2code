ll trarea(pii x1,pii x2,pii x3)
{
	return ((ll)x1.fi*x2.se+(ll)x2.fi*x3.se+(ll)x3.fi*x1.se-((ll)x2.fi*x1.se+(ll)x3.fi*x2.se+(ll)x1.fi*x3.se));
} // for calculating area of a triangle given its three points..

int iseq(long double a,long double b)
{
	if (abs(a-b)<0.00000000001)
		return 1;
	return 0;
}

		pii a,b;

		ll ar=abs(trarea(mp(0,0),a,b));

		long double x=(long double)sqrt(dist);
