#include <algorithm>
#include <cmath>
#include <cstdio>
#include <iostream>

using namespace std;

int n, a;
int main() {

  cin >> n;

  for (int i = 0; i < n; i++) {
    cin >> a;
    if (i)
      cout << "R";
    while (a--)
      if (i == 0)
        cout << "PRL";
      else
        cout << "PLR";
  }
  return 0;
}
