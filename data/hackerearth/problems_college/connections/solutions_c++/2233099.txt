#include<bits/stdc++.h>
using namespace std;
#define pb push_back
vector <int> adj[10005];
int vis[10005]={0},ct=1;
void dfs(int n)
{
    int i;
    vis[n]=ct;
    for(i=0;i<adj[n].size();i++)
    {
    if(vis[adj[n][i]]==0)
        dfs(adj[n][i]);
    }
}
int main()
{
    int n,e;
      int a,b,i,q,x,y;
      scanf("%d%d",&n,&e);
      while(e--)
      {
          scanf("%d%d",&a,&b);
          adj[a].pb(b);
         adj[b].pb(a);
      }
      for(i=0;i<n;i++)
      {
          if(vis[i]==0)
          {
              dfs(i);
              ct++;
          }

      }
     cin>>q;
     while(q--)
     {
     	cin>>x>>y;
     	if(vis[x]==vis[y]){
     		
     	int f=0;
     	for(i=0;i<adj[x].size();i++)
     	  if(adj[x][i]==y)
     	  {
     	  	f=1;
     	  	printf("Direct Connection\n");
     	  }
     	  if(f==0)
     	  printf("Indirect Connection\n");
     	}
     	else
     	printf("No Connection\n");
     	  
     }
}
