#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long i,j,k,T,n,a,b,ans;
    scanf("%lld",&T);
    while(T--)
    {
    	scanf("%lld%lld",&a,&b);
    	ans=a;
    	for(i=a+1;i<=b;i++)
    		ans=ans&i;
    	printf("%lld\n",ans);
    }
    return 0;
}
