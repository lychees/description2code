#include <bits/stdc++.h> 
#include <stdio.h>
#include <iostream>
#include <stdlib.h>
#include <string.h>
using namespace std;
#define RANGE 1000
static int gcd[RANGE+1][RANGE+1],prime[RANGE*RANGE+1]={0};

void sieve()
{
    for(int i=2;i<=RANGE*RANGE;i++)
    {
        if(prime[i]==0)
        {
            for(int j=i*2;j<=RANGE*RANGE;j+=i)
                prime[j]++;
        }
    }
}

int GCD(int a, int b)
{
	int x = a, y = b;
	if(!gcd[a][b])
	{
		
		while(a)
		{
			int rem = b%a;
			b = a;
			a = rem;
		}
		gcd[x][y] = gcd[y][x] = b;
	}
	return gcd[x][y];
}
int main()
{
	int t;
 	sieve();
	cin>>t;
	while(t--)
	{
		int a,b;
		cin>>a>>b;
		int aLcm = a*b/GCD(a,b);
		if(prime[aLcm]>1 && !prime[prime[aLcm]])
			cout<<"Yes";
		else
			cout<<"No";
		cout<<endl;
	}
    //printf("Hello World!\n");
    return 0;
}
