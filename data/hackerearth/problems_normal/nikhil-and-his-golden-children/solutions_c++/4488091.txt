#include <bits/stdc++.h>
using namespace std;
#define ll long long
ll mod=1e9+7;
ll fastPow(ll a,ll b,ll mod)
{
	if(b==0)
	return 1;
	if(b==1)
	return a;
	ll half=fastPow(a,b/2,mod);
	if(b%2==0)
	return ( half * half ) % mod;
	return (((half * half ) % mod )* a ) % mod;
}
ll mmi(ll n,ll m)
{
	return fastPow(n,m-2,mod);
}
ll fact(int n)
{
	if(n==0)
	return 1;
	return (n*fact(n-1))%mod;
}
int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		int n;
		ll k,sum=0;
		cin >> n >> k;
		for ( int i=0 ; i < n ;i++)
		{
			int x;
			cin >> x;
			sum+=x;
			//sum%=mod;
		}
		//cout << sum <<endl;
		if(sum>k)
		cout << "0" << endl;
		else
		{
			int left=k-sum;
			left=n+left-1;
		//	cout << left << endl;
			int den=n-1;
			ll x=fact(left);
			ll y=fact(den);
			ll z=fact(left-den);
			//cout << x <<" " << y <<" "<< z << endl;
		    y=(y*z)%mod;
		    y=mmi(y,mod);
			ll ans=(x*y)%mod;
			//ans*=z;
			cout << ans % mod<< endl;
		}
	}
	return 0;
}
