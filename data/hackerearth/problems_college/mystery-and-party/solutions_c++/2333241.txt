#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main()
{
    int t;
    ios_base::sync_with_stdio(false);cin.tie(0);
    cin>>t;
    while(t>0)
    {
    	t--;
    	ll n,k;
    	cin>>n>>k;
    	vector<ll> v;
    	ll i;
    	for(i=0;i<n;i++)
    	{
    		int temp;
    		cin>>temp;
    		v.push_back(temp);
    	}
 
    	sort(v.begin(),v.end());
    	cout<<v[k-1]<<'\n';
    }
    return 0;
}
