#include <iostream>
using namespace std;

int main()
{
    int t,j,k,j1,k1;
    cin>>t;
    for (int i=0;i<t;i++)
    {
    	int n;
    	cin>>n;
    	char c;
    	for (j=0;j<n;j++)
		for (k=0;k<n;k++)
		{
			cin>>c;
			if (c=='*')
			{
			j1=j;k1=k;//cout<<c<<k<<k1;
			}
		}
    	cout<<abs((j1-(n-1)/2))+abs((k1-(n-1)/2))<<endl;//<<j1<<k1;
    }
    return 0;
}
