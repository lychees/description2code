#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;

long long int A[1000][1000],B[1000][1000];
int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */   
    int  n,t;
    cin>>t;
    while(t--){
    cin>>n;
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
        cin>>A[i][j];
     for(int i=n-1;i>=0;i--)
        for(int j=n-1;j>=0;j--) 
        {
        if(i==n-1&&j==n-1)
            B[i][j]=A[i][j];
        else if(i==n-1)
            B[i][j]=A[i][j]+B[i][j+1];
        else if(j==n-1)
            B[i][j]=A[i][j]+B[i+1][j];
            else
            {
            B[i][j]=A[i][j]+max(B[i+1][j],B[i][j+1]);
        }
    }
    if(B[0][0]==2*n-1)
    cout<<"POSSIBLE"<<endl;
    else
     cout<<"NOT POSSIBLE"<<endl;
    } 
    return 0;
}
