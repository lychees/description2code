using namespace std;
int a[1000000];
long long f(int n)
	{
	int maxe,m;
	maxe=m=a[0];
	for(int i=1;i<n;i++)
		{
		maxe=max(a[i],maxe+a[i]);
		m=max(m,maxe);
		}
	return m;
	}
int main()
{
int T;
scanf("%d",&T);
int N;
while(T--)
{

scanf("%d",&N);
for(int i=0;i<N;i++)
	{
	scanf("%d",&a[i]);
	}
printf("%lld\n",f(N));
}
return 0;
}
