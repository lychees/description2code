#include<stdio.h>

int isleap(int y)
{
    if(y%400==0)
    return 1;
    else if(y%100==0)
            return 0;
            else if(y%4==0)
                    return 1;
                    else return 0;
}

int main()
{
    int t, i, d, m, max, y, c, day, table_month[] = {0, 3, 3, 6, 1, 4, 6, 2, 5, 0, 3, 5};
    char str[][10] = {"Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"};

    scanf("%d", &t);

    while(t--)
    {
        scanf("%d %d", &m, &y);

        if(isleap(y))
        {
            table_month[0] = -1;
            table_month[1] = 2;
        }
        else
        {
            table_month[0] = 0;
            table_month[1] = 3;
        }

        c = 6 - 2*((y/100)%4);

        if(m==2)
        {
            if(isleap(y))
            max=29;
            else max=28;
        }
        else if(m<=7)
        {
            if(m%2==1)
            max=31;
            else max=30;
        }
        else
        {
            if(m%2==0)
            max=31;
            else max=30;
        }

        for(i=1;i<=max;i++)
        {
            day = (i + table_month[m-1] + y%100 + (y%100)/4 + c)%7;
            if(day==0)
            {
                printf("%d", i);
                break;
            }
        }
        for(i=i+1;i<=max;i++)
        {
            day = (i + table_month[m-1] + y%100 + (y%100)/4 + c)%7;
            if(day==0)
            {
                printf(" %d", i);
            }
        }

         printf("\n");

       // printf("%s\n", str[day]);
    }

    return 0;
}
