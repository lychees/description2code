int main() {
    int T;
    cin >> T;
    for (int test = 1; test <= T; ++test) {
      int n;
      scanf("%d",&n);
      vl s(1 << n), c(1 << n);
      c[0] = 1;
      for (int i = 0; i < n; ++i) {
        for (int mask = 0; mask < c.size(); ++mask) if (__builtin_popcount(mask) == i) {
          int rage = 0;
          for (int t = n-1; t >= 0; --t) {
            if (mask & (1 << t)) {
              ++rage;
            } else if (i != t) {
              c[mask | (1 << t)] += c[mask];
              s[mask | (1 << t)] += s[mask] + c[mask]*rage;
            }
          }
        }
      }
//      cerr << c.back() << endl;
      cout << s.back() << endl;
    }
    return 0;
}
