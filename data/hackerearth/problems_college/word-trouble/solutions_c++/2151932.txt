#include <iostream>
using namespace std;
void count(string str,string str1,int len,int len1)
{
	int count=len-len1;
	int i;
	for(i=0;i<len1;i++)
	{
		if(str[i]!=str1[i])
		{	
			count++;
		}
	}
	if(count==0&&len==len1)
		cout<<"Exact"<<endl;
	else
		cout<<"Non Exact"<<" "<<count<<endl;
}
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    string str,str1;
    cin>>t;
    for(;t;--t)
    {
    	cin>>str>>str1;
    	int len=str.size();
    	int len1=str1.size();
    	if(len>len1)
    		count(str,str1,len,len1);
    	else
    		count(str1,str,len1,len);
    }
    return 0;
}
