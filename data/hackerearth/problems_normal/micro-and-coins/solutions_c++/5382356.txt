typedef vector<ll> vii;

int n,m,u,v,nVis;
bool vis[20];
vector< vector<int> >g;

bool kiemtra(int x){
    if(nVis == n) return true;

    for(int i=0;i<g[x].size();i++){
        int next = g[x][i];
        if(vis[next]==0){
            vis[next] = 1;
            nVis++;
            bool tiep = kiemtra(next);
            if(tiep) return true;
            vis[next] = 0;
            nVis--;
        }
    }
    return false;
}

int main(){
    //fr("in.txt");
    //fw("out.txt");

    tc(){
        cin >> n >> m;
        g.clear();
        g.resize(n+5);
        for(int i=0;i<m;i++){
            cin >> u >> v;
            g[u].push_back(v);
            g[v].push_back(u);
        }

        reset(vis);
        nVis = 1;
        vis[1] = 1;
        bool flag = kiemtra(1);

        //cout << nVis << endl;

        if(flag) cout << "Yes\n";
        else cout << "No\n";
    }

    return 0;
}
