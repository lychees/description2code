#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long x;
    	cin>>x;
    	int f=0;
    	if((x%2==0)&&(x!=2))
    	f=1;
    	for(int i=3;i<=sqrt(x)&&f==0;i+=2)
    	if(x%i==0)
    	f=1;
    	if(x==2)
    	f=0;
    	if(f==0)
    	cout<<"YES";
    	else
    	cout<<"NO";
    	cout<<endl;
    }
    return 0;
}
