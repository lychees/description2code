#include<bits/stdc++.h>
using namespace std;
int main()
{
    long t,n,i,a[1111];
    a[0]=0;
    a[1]=1;
    for(i=2;i<=1111;i++)
        a[i]=a[i-1]+a[i-2];
    cin>>t;
    while(t--)
    {
        cin>>n;
        if(n==0)
        cout<<0<<endl;
        else
        cout<<a[n+1]<<endl;
    }
}
