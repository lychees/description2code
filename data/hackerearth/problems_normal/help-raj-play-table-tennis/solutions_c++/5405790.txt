typedef long long ll;
ll MOD = 1000000009;
ll pw(ll a, ll b){
	ll r = 1;
	while (b>0){
		if (b & 1){
			r = (r*a) % MOD;
		}
		b = b >> 1;
		a = (a*a) % MOD;
	}
	return r;
}
int p11, p12, p21, p22;
ll dp[100][100];

int over(int a, int b){
	if ((a >= 11 || b >= 11) && abs(a - b) >= 2){
		return 1;
	}
	return 0;
}

ll fun(int a, int b){
	if (a>p12 || b>p22){
		return dp[a][b] = 0;
	}
	if (a == p12&&b == p22){
		return dp[a][b] = 1;
	}
	if (dp[a][b] != -1){
		return dp[a][b];
	}
	if (over(a, b)){
		return dp[a][b] = 0;
	}
	ll tmp = 0;
	tmp = (fun(a + 1, b) % MOD + fun(a, b + 1) % MOD) % MOD;

	return dp[a][b] = tmp;

}


int main(){
	int t;
	si(t);

	fr(i, 1, t + 1){
		si(p11), si(p21), si(p12), si(p22);
		fr(i, 0, 100){
			fr(j, 0, 100){
				dp[i][j] = -1;
			}
		}
		ll ans = 0;
		if (!over(p12, p22) || (over(p11, p21) && p11 != p12 && p21 != p22)){
			ans = -1;
		}
		else{
			ans = fun(p11, p21);
		}
		if (!ans)ans = -1;

		cout << "Case " << i << ": " << ans << endl;
	}

	return 0;
}
