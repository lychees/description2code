#include <iostream>
#include<string.h>
#include<bits/stdc++.h>
using namespace std;

int main()
{
   	int n;
   	long long sum=0,sum1;
   	cin>>n;
   	char arr[100000];
   	for(int j=0;j<n;j++)
   	{
	   	cin>>arr;
	   	for(int i=0;i<strlen(arr);i++)
	   	{
	   		sum+=(arr[i]-'0');
	   	}
   	}
   	while(sum>9)
   	{
   		sum1=0;
	   	while(sum>0)
	   	{
	   		sum1+=(sum%10);
	   		sum=sum/10;
	   	}
	   	sum=sum1;
	}
	cout<<sum;
    return 0;
}
