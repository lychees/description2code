ll mpow(ll a, ll n,int mod) 
{
	ll ret=1;
	ll b=a;
	while(n) 
	{
		if(n&1) 
    		ret=(ret*b)%mod;
    	b=(b*b)%mod;
    	n>>=1;
    }
	return (ll)ret;
}
