n = int(raw_input())
array = [int(x) for x in raw_input().split()]


def max_bouquet(array):
    maximum = 0
    total = 0
    for flower in array:
        total += flower
        if total % 2 != 0:
            maximum = total

    if maximum:
        return maximum


maxima = max_bouquet(array)
for i in range(len(array)):
    el = array[i]
    array.pop(i)
    if max_bouquet(array) > maxima:
        maxima = max_bouquet(array)
    array.append(el)

if maxima:
    print(maxima)
else:
    print(':(')
