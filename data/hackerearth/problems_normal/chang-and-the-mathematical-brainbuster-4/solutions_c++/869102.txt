#include <iostream>
using namespace std;

int main()
{
   	int n,i,j,temp,min;
	
	cin >>n;
	int a[n];
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
	}
	
	for(i=0;i<n;i++)
	{
		for(j=i+1;j<n;j++)
		{
		  if(a[i]>a[j])
		  {
		  	temp=a[i];
		  	a[i]=a[j];
		  	a[j]=temp;
		  }
			
		}
	}
	min=100000;
	for(i=0;i<n;i++)
	{
	 
		for(j=n-1;j>i;j--)
		{
			if((a[j]-(3*a[i])) <=0)
			{
			   if(min>(i+((n-1)-j)))
			   {
			   	min=(i+((n-1)-j));
			   }
			 break;	
			}
		          	
		}
		
	}
    cout<<min;
    return 0;
}
