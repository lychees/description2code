#include <iostream>
#include<cmath>
#define ll long long
using namespace std;

ll getSum(ll&R, ll&C){
	ll n=min(R, C);
	ll m=C+1;
	return m*n*(n-1)/2+n;
}

int main()
{
	int T;
	cin>>T;
	while(T--){
		ll R, C;
		cin>>R>>C;
		ll sol=getSum(R, C)+getSum(C, R);
		
		cout<<sol<<endl;
	}
    return 0;
}
