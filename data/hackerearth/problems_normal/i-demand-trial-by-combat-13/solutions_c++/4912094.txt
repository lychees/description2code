#include <iostream>
using namespace std;
bool equal(int *a, int *b, int n) {
    for (int i = 0; i < n; ++i) {
        if (a[i] != b[i]) return false;
    }
    return true;
}
int main() {
    int t;
    cin >> t; cin.ignore();
    while(t-- > 0) {
        int n, m;
        cin >> n >> m; cin.ignore();
        int *lannisters = new int[n];
        for (int i = 0; i < n; ++i) {
            cin >> lannisters[i]; cin.ignore();
        }
        for (int h = 0; h < m; ++h) {
            int *copyy = new int[n];
            for (int i = 0; i < n; ++i) {
                if (i == 0) {
                    if (lannisters[1] == 1) {
                        copyy[0] = 1;
                    }
                    else {
                        copyy[0] = 0;
                    }
                }
                else if (i == n - 1) {
                    if (lannisters[n - 2] == 1) {
                        copyy[n - 1] = 1;
                    }
                    else {
                        copyy[n - 1] = 0;
                    }
                }
                else {
                    int counter = 0;
                    if (lannisters[i - 1] == 1) ++counter;
                    if (lannisters[i + 1] == 1) ++counter;
                    if (counter != 2) {
                        copyy[i] = 0;
                    }
                    else {
                        copyy[i] = 1;
                    }
                }
            }
            if (equal(lannisters, copyy, n)) {
                delete[] copyy;
                break;
            }
            delete[] lannisters;
            lannisters = copyy;
        }
        cout << lannisters[0];
        for (int i = 1; i < n; ++i) {
            cout << " " << lannisters[i];
        }
        cout << endl;
        delete[] lannisters;
    }
}
