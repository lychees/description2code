#include <iostream>
#include <string>
using namespace std;

int main()
{
    int t,k;
    string str;
    cin >> t;
    for(int i=0;i<t;i++){
    	cin >> str;
    	int size=str.length();
    	for(int j=0;j<size/2;j++){
    		k=str[j];
    		str[j]=str[size-1-j];
    		str[size-1-j]=k;
    	}
    	cout << str << endl;
    }
    return 0;
}
