#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;

ll a[85];
ll tot[85];

int main()
{
a[0]=0;
a[1]=1;
a[2]=1;
tot[0]=0;
tot[1]=1;
tot[2]=2;
for(ll i=3;i<85;i++)
{
a[i]=a[i-1]+a[i-2];
tot[i]=tot[i-1]+a[i];
//cout<<tot[i]<<" ";
}

ll n,m;
ll sum=0;
cin>>n>>m;
while(m--)
{
ll x;
cin>>x;
sum+=x;
}
//cout<<sum<<endl;
cout<<tot[n]-sum<<"\n";
return 0;
}
