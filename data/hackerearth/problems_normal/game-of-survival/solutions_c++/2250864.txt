#include <iostream>
#include <algorithm>
using namespace std;

void countSort(int A[],const int size,const int range)
{
    int *count = new int[range+1];
    int *output = new int[size];
    //building count array
    for(int i=0;i<size;i++)
    {
        count[A[i]]++;
    }
    //assigning each element in count array it's proper position
    for(int i=1;i<=range;i++)
        count[i]+=count[i-1];
    //build output array
    for(int i=size-1;i>=0;i--)
    {
        output[count[A[i]]-1]=A[i];
        count[A[i]]--;
    }
    //copying output to A
    for(int i=0;i<size;i++)
        A[i]=output[i];
}

void survivalGame()
{
    int n_test;
    cin>>n_test;
    for(int i=0;i<n_test;i++)
    {
        int size;
        cin>>size;
        int *sharp = new int[size];
        int *strength = new int[size];
        for(int j=0;j<size;j++)
        {
            cin>>sharp[j];
        }
        for(int j=0;j<size;j++)
        {
            cin>>strength[j];
        }
        //countSort(sharp,size,101);
        //countSort(strength,size,101);
        sort(sharp,sharp+size);
        sort(strength,strength+size);
        bool fail = false;
        for(int j=0;j<size;j++)
        {
            if(sharp[j]<=strength[j])
            {
                fail = true;
                break;
            }
        }
        if(!fail)
        {
            cout<<"YES"<<endl;
        }
        else
        {
        	cout<<"NO"<<endl;
        }
    }
}

int main()
{
    survivalGame();
}
