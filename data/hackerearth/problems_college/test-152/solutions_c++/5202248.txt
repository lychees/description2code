#include <iostream>
#include <string>
#include <unordered_map>

using namespace std;

int main()
{
    int totalTests=0, testCases=0;
    
    cin >> totalTests;
    
    for(int i=0; i<totalTests;i++)
    {
    	int maxFreq=0;
    	char maxFreqC=' ';
    	
    	unordered_map<char, int> myMap;
    	cin >> testCases;
    	string testString;
    	cin >> testString;
    	
    	for(int j=0; j<testCases;j++)
    	{
    		if(myMap.count(testString[j]) > 0)
    		{
    			// Exist, increment.
    			int temp = myMap.at(testString[j]);
    			if(maxFreq < ++temp)
    			{
    				maxFreq = temp;
    				maxFreqC=testString[j];
    			}
    			myMap.at(testString[j])=temp;
    		}
    		else
    		{
    			// New item, insert.
    			if(maxFreq < 1)
    			{
    				maxFreq = 1;
    				maxFreqC = testString[j];
    			}
    			pair<char, int> p;
    			p.first=testString[j];
    			p.second=1;
    			myMap.insert(p);
    		}
    	}
    	if(maxFreq % 2 == 0)
    		cout << "Yes" << endl;
    	else
    		cout << "No" << endl;
    }
    return 0;
}
