import itertools
for t in range(input()):
    n = input()
    a = map(int, raw_input().split())
    k = input()
    y = 0
    for L in range(0, n+1):
        for subset in itertools.combinations(a, L):
            if (sum(subset) == k):
                y = 1
                break

    if (y == 1):
        print "YES"
    else:
        print "NO"
