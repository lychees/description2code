#include <bits/stdc++.h>
#define ll long long int
#define mod (int)(1e9 + 7)

using namespace std;

int t, n;
int a[1000005];

ll fast(int p) {
	ll ans = 1, x = 3;
	
	while (p > 0) {
		if (p % 2 != 0) {
			ans = (ans * x) % mod;
		}
		x = (x * x) % mod;
		
		p /= 2;
	}
	return ans % mod;
}

int main()
{
	cin >> t;
	while (t--) {
		cin >> n;
		cout << fast(n) << endl;
	}
	return 0;
}
