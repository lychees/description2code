#include<bits/stdc++.h>
#include <stdio.h>
using namespace std;
const int size=500010;


int main()
{
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		
		int dis[size];
		dis[0]=0;
		for(int i=1;i<=size;i++){
			dis[i]=0;
		}
		
		int input;
		for(int i=0;i<n;i++){
			scanf("%d",&input);
			dis[input]++;
		}
		
		bool dead= false;
		int i, kills=0, ans=0;
		for(i=1;i<size;i++){
			
			if(i%7!=0){
				kills++;
			}
			
			if(dis[i]>kills){
				dead= true;
				break;
			}
			
			else{
				kills-= dis[i];
			}
		}
		
		if(dead){
			cout<<"Goodbye Rick"<<endl<<i-(i/7)<<endl;
		}
		
		else{
			cout<<"Rick now go and save Carl and Judas"<<endl;
		}
	}
	
}
