#include<bits/stdc++.h>
using namespace std;
long long int dp[10001];
int n;
long long int recur(int i,int *a)
{
	if(i>n)
	  return 0;
	if(dp[i])
	  return max(dp[i],dp[i+1]);
	dp[i]=a[i]+recur(i+2,a);
	return max(dp[i],recur(i+1,a));
}
main()
{
	int t;
	cin>>t;
	for(int tc=1;tc<=t;tc++)
	{
		cin>>n;
		int a[n+1];
		for(int i=1;i<=n;i++)
		   cin>>a[i];
		memset(dp,0,sizeof(dp));
	    cout<<"Case "<<tc<<": "<<recur(1,a)<<endl;
	}
}
