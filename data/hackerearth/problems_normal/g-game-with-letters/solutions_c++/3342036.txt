#include <iostream>
#include <algorithm>
#include <vector>
#include <cstring>
using namespace std;
int score[26];
int n;
typedef pair<pair<int,int>, int> pii;
vector<pii > v;
bool cmp(pii x, pii y){
	if (x.first.first>y.first.first) return true;
	if (x.first.first<y.first.first) return false;
	if (x.first.second<y.first.second) return true;
	if (x.first.second>y.first.second) return false;
	return x.second<y.second;
}
int main()
{
    int t;
    cin>>t;
    while(t--){
    	cin>>n;
    	string s,ss;
    	cin>>s;
    	memset(score,0,sizeof(score));
    	for(int i=0;i<s.length();i++){
    		score[s[i]-'a']++;
    	}
    	v.clear();
    	for(int i=1;i<=n;i++){
    		cin>>ss;
    		int sc = 0;
    		for(int j=0;j<ss.length();j++){
    			sc += score[ss[j]-'a'];
    		}
    		v.push_back({{sc,ss.length()},i});
    	}
    	sort(v.begin(),v.end(),cmp);
    	int ans = v[0].second;
    	cout << ans << "\n";
    }
    return 0;
}
