#include <iostream>
#include<algorithm>
#define ll long long
using namespace std;
ll gcd(ll a,ll b){
	if(b==0)
	return a;
	else
	return gcd(b,a%b);
}
int main() {

ll x,d,y,i,j,a[3000],k;
cin>>d;
while(d--){
	cin>>x>>y>>k;
	x=gcd(x,y);
	j=0;
	for(i=1;i*i<=x;i++){
		if(x%i==0){
			a[j]=i;
			j++;
			if(i*i!=x){
				a[j]=x/i;
				j++;
			}
		}
	}
	sort(a,a+j);
	if(k>j){
		cout<<"No crymeth today\n";
	}
	else
	cout<<a[j-k]<<"\n";
}
	return 0;
}
