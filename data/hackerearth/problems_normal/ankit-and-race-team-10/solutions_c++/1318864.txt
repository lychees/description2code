#include <bits/stdc++.h>

using namespace std;
#define LL long long
#define INF 1000000007

LL powi(LL b,LL n){
    if(n == 0)return 1;
    if(n == 1)return b;
    LL ans = powi(b,n/2);
    ans = (ans*ans)%INF;
    if(n&1){
        ans = (ans*b)%INF;
    }
    return ans;
}


LL inv(LL n){
    return powi(n,INF-2);
}
LL a[100010];

int main(){
    int T;
    cin >> T;
    while(T--){
        LL n,k;
        cin >> n >> k;
        for(int i = 0; i < n; ++i){
            cin >> a[i];
        }
        sort(a,a+n);

        LL p = 1;
        for(LL i = n-1;i>=n-k+1;--i){
            p = (p*i)%INF;
        }
        LL ans = (a[0]*p)%INF;
        LL last = n-1;
        int co = 1;
        for(LL i = n-k; i>=1; --i){
//            cout << p << " "<< last << " " <<inv(last) << endl;;
            p = (p*inv(last))%INF;
            p = (p*i)%INF;
            ans = (ans+(a[co]*p)%INF)%INF;
            co++;
            last--;
        }
        LL fact = 1;
        for(LL i = 2; i < k ;++i){
            fact = (fact * i)%INF;
        }
        fact = inv(fact);
        cout << (ans*fact)%INF << endl;
    }
    return 0;
}
