#include <bits/stdc++.h>
using namespace std;

int main()
{
    int q;
    cin>>q;
    stack<int>stk;
    while(q--){
    	int type;
    	cin>>type;
    	if(type==1){
    		if(stk.size()==0)
    			cout<<"No Food"<<endl;
    		else{
    			cout<<stk.top()<<endl;
    			stk.pop();
    		}
    	}else{
    		int x;
    		cin>>x;
    		stk.push(x);
    	}
    }
    return 0;
}
