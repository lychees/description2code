#include<bits/stdc++.h>
using namespace std;
#define ll long long int
int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */  
    ll n;
    scanf("%lld",&n);
   ll j; 
    char s[1005];
    
   for(j=0;j<=n;j++)
        {
            
        cin.getline(s,1005);
        ll i=0;
        while(s[i]!='\0')
            {
            if(s[i]=='$')break;
            i++;
        }
       
       i++;
        ll count=0;
        if(j>0)cout<<"$";
        while(s[i]!='\0'&&!isalpha(s[i]))
            {
            if((s[i]!='0'||s[i]=='0'&&count!=0)&&s[i]!=' '&&j>0)
            {cout<<s[i];count++;}
                i++;
        }
        if(count==0&&j>0)cout<<"0";
        if(j>0)cout<<endl;
        
    }
    
    
    return 0;
}
