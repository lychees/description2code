int main()
{
        int t;
        cin>>t;
        while(t--)
        {
            char a[11],b[11];
            int f=0,cnt=0,l1,l2,i;
            cin>>a>>b;
            l1=len(a);
            l2=len(b);

            cnt=abs(l1-l2);
            if(l1<l2)
            {
                for(i=0;i<l1;i++)
                    {
                        if(a[i]!=b[i])
                        {
                            cnt++;

                        }
                    }

            }
            else
            {
                for(i=0;i<l2;i++)
                    {
                        if(a[i]!=b[i])
                        {
                            cnt++;
                        }
                    }


            }


            if(cnt==0)
                printf("Exact\n");
            else
                printf("Non Exact %d\n",cnt);


        }
        return 0;
}
