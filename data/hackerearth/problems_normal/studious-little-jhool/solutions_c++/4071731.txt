#include<iostream>
#include<string.h>
using namespace std;
const int inf=120;
int main()
{
  int t;
  int n;
  cin>>t;
  int dp[2][114];
for(int i=1;i<=t;i++)
{
  for(int j=1;j<=113;j++)
      dp[0][j]=inf,dp[1][j]=inf;
    
    cin>>n;
    
   for(int k=1;k<=n;k++)
      dp[0][k]=(k%10==0)?k/10:inf;
      
     dp[1][0]=0;
  
    for(int k=1;k<=n;k++)
      { 
        dp[1][k]=min(dp[0][k],1+dp[1][k-12]);
      }
      
if(dp[1][n]<inf)
   cout<<dp[1][n]<<endl;
else cout<<"-1"<<endl;
}
}
