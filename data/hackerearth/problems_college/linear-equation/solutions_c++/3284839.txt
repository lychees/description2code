#include<bits/stdc++.h>

using namespace std;

int main()
{
	long long int t,n,i,sum;
	cin >> t;
	while(t--)
	{
		cin >> n;
		sum = 0;
		long long int a[n];
		for(i = 0;i <n; i++)
		{
			cin >> a[i];
			sum += a[i];
		}
		for(i = 0;i<n;i++)
		{
			cout << (sum/(n-1))-a[i] << " ";
		}
		cout << endl;
	}
	return 0;
}
