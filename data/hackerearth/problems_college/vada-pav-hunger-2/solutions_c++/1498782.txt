#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define pb push_back
vector<int>adj[1005];
int ma=0;
bool visit[1005];
int dist[1005];
void bfs(int s)
{
visit[s]=true;
dist[s]=0;
queue<int>q;
q.push(s);
vector<int>::iterator i;
while(!q.empty())
{
  int top=q.front();
  q.pop();
  for(i=adj[top].begin();i!=adj[top].end();i++)
  {
      if(!visit[*i])
      {
          dist[*i]=dist[top]+1;
          visit[*i]=true;
          q.push(*i);
      }
  }
}
}
int main()
{
int n;
sd(n);
memset(visit,false,sizeof(false));
for(int i=1;i<=n-1;i++)
{
    int x,y;
    sd(x);
    sd(y);
    adj[x].pb(y);
    adj[y].pb(x);
}
bfs(1);
for(int i=1;i<=n;i++)
{
    ma=max(ma,dist[i]);
}
cout<<2*(n-1)-ma;
return 0;
}
