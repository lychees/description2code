#include <iostream>
#include <algorithm>
using namespace std;

int sum(int A[], int target, int T){
    int i=0;
    while( target > 0){
        if( i == T ) return -1;
        //cout << target << " " << A[i] << endl;
        target-=A[i++];
        
    }
    //cout << target << " " << A[i] << endl;
    return i;
}

int main(){
    int T,Q,target;
    cin >> T;
    int A[T];
    for(int i=0; i<T; i++) cin >> A[i];
    cin >> Q;
    while(Q--){
        cin >> target;
        cout << sum( A, target, T) << endl;
    }
    return 0;
}
