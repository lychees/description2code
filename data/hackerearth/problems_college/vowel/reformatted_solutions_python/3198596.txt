t = int(raw_input())
for i in range(0, t):
    s = raw_input()
    out = ""
    for k in range(0, len(s)):
        if (s[k] == 'a' or s[k] == 'e' or s[k] == 'i' or s[k] == 'o' or s[k] ==
                'u' or s[k] == 'A' or s[k] == 'E' or s[k] == 'I' or s[k] == 'O' or s[k] == 'U'):
            out = out+s[k]
    if (out == ""):
        print "No"
    else:
        print out
