t = input()
while t > 0:
    t -= 1
    s = raw_input()
    l = s.split("*")
    res = ""
    for i in range(len(l)):
        if "x" in l[i]:
            r = ""
            for j in range(len(l[i])):
                if "x" == l[i][j]:
                    r += "x"
            l[i] = r
        res += l[i] + "*"
    res = res[0:-1]
    print res
