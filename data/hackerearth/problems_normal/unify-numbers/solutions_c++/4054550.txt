#include<bits/stdc++.h>
using namespace std;
#define ll long long int
	ll a[100010];
int main()
{
	    int t;
	    cin>>t;
	    while(t--)
	    {
	    	   int n;
	    	   cin>>n;
	    	   for(int i=0;i<n;i++)
	    	   cin>>a[i];
	    	   sort(a,a+n);
	    	   int m1=n/2;
	    	   int m2=n/2-1;
	    	   if(n==1)
	    	   {
	    	   	  cout<<"0"<<endl;
	    	   	  continue;
			   }
			   else 
			   {
			   	      ll ans1=0;
			   	      for(int i=0;i<n;i++)
			   	      {
			   	           	ans1+=abs(a[i]-a[m1]);
					  }
					  ll ans2=0;
					  for(int i=0;i<n;i++)
					  {
					  	   ans2+=abs(a[i]-a[m2]);
					  	   
					  }
					  cout<<min(ans1,ans2)<<endl;
			   }
		}
		return 0;
}
