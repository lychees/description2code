#include <iostream>
#include <cstring>
using namespace std;

int main()
{
	int t,i,count,c,d;
	char s[1000];
  cin>>t;
  while(t--)
  {
  	count=0;
  	cin>>s;
  	for(i=0;i<strlen(s);i++)
  	{
  		c=(s[i+1]-s[i]);
  		d=s[strlen(s)-i-1]-s[strlen(s)-i-2];
  		if(c>=0&&d>=0||c<0&&d<0)
  		{
  		if((s[i+1]-s[i])==(s[strlen(s)-i-1]-s[strlen(s)-i-2]))
  		{
  			count++;
  		}
  		}
  		else if(c<0)
  		{
  		if((s[i]-s[i+1])==(s[strlen(s)-i-1]-s[strlen(s)-i-2]))
  		{
  			count++;
  		}
  		}
  		else if(d<0)
  		{
  		if((s[i+1]-s[i])==(s[strlen(s)-i-2]-s[strlen(s)-i-1]))
  		{
  			count++;
  		}
  		}
  		}
  		
  	
  	if(count==strlen(s)-1)
  	cout<<"Funny"<<endl;
  	else
  	cout<<"Not Funny"<<endl;
  }
    return 0;
}
