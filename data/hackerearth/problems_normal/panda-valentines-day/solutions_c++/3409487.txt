#include <bits/stdc++.h>
using namespace std;
long long int a[2000000],b[2000000];
int main() 
{
    for(int i=2;i*i<=1000000;i++)
        {
        if(a[i]==0)
        for(int j=i*i;j<=1000000;j=j+i)
            {
            a[j]=i;
        }
    }
    for(int i=2;i<=1000000;i++)
    {
        if(a[i]==0)
            b[i]=1;
        else
            b[i]=1+b[i/a[i]];
    }
    for(int i=2;i<=1000000;i++)
        b[i]=b[i]+b[i-1];
    int t;
    scanf("%d",&t);
    while(t--){
    long long int n;
    scanf("%lld",&n);
    
    long long ans=b[n];
    printf("%lld\n",ans);}
    return 0;
}
