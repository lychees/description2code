#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;

int main()
{
	int tc,n[10],m[10] ,x1[10],x2[10],y1[10],y2[10],diff1,diff2,diff3,diff4,col,row,cost,p[10];
	cin>>tc;
	
	for(int i=1;i<=tc;i++)
	{	cin>>n[i];
	cin>>m[i];
		cin>>x1[i];
	cin>>y1[i];
	cin>>x2[i];
	cin>>y2[i];
	cin>>p[i];
	}
		for(int i=1;i<=tc;i++)
		{
		diff1= abs(x1[i]-x2[i]);
		diff2= abs(n[i]-diff1);
		diff3= abs(y1[i]-y2[i]);
		diff4= abs(m[i]-diff3);
		col=min(diff1,diff2);
		row=min(diff3,diff4);
		cost=p[i]*(col+row);
		
		cout<<cost<<endl;
		if(cost<1001)
		{cout<<"You saved the group."<<endl;}
		else
		{
			cout<<"Let go of the group."<<endl;
		}
	}
    
    return 0;
}
