#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	vector<int> neg,pos;
    	bool z=0;
    	int temp;
    	for(int i=0;i<n;i++)
    	{
    		cin>>temp;
    		if(temp==0) z=1;
    		else if(temp<0) neg.push_back(temp);
    		else if(temp>0) pos.push_back(temp);
    	}
    	int lenp=pos.size();
    	int lenn=neg.size();
    	sort(neg.rbegin(),neg.rend());
    	sort(pos.begin(),pos.end());
    	if(lenn==0)
    	{
    		long long int maxi,mini;
    		if(z==1) mini=0;
    		else mini=pos[0];
    		if(lenp==0) maxi=0;
    		else
    		{
    			maxi=1;
    			for(int i=0;i<lenp;i++)
    			{
    				maxi*=pos[i];
    			}
    		}
    		cout<<maxi<<" "<<mini<<"\n";
    	}
    	else 
    	{
    		long long int ans=1,maxi,mini;
    		if(lenn==1 && z==0 && lenp==0)
    		{
    			maxi=neg[0];
    			mini=neg[0];
    		}
    		else if(lenn==1 && z==1 && lenp==0)
    		{
    			maxi=0;
    			mini=neg[0];
    		}
    		else if(lenn%2==1)
    		{
    			mini=1;
    			maxi=1;
    			for(int i=0;i<lenn;i++)
    			{
    				mini*=neg[i];
    				if(i>0) maxi*=neg[i];
    			}
    			for(int i=0;i<lenp;i++)
    			{
    				mini*=pos[i];
    				maxi*=pos[i];
    			}
    		}
    		else if(lenn%2==0)
    		{
    			mini=1;
    			maxi=1;
    			for(int i=0;i<lenn;i++)
    			{
    				if(i>0) mini*=neg[i];
    				maxi*=neg[i];
    			}
    			for(int i=0;i<lenp;i++)
    			{
    				mini*=pos[i];
    				maxi*=pos[i];
    			}
    		}
    		cout<<maxi<<" "<<mini<<"\n";
    	}
    }
    return 0;
}
