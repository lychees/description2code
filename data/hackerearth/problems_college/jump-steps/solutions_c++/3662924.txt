#include <iostream>
using namespace std;
long long int func(long long int a)
{
    long long int i=0;
    long long int temp=1;
    while(temp<=a)
    {
        temp=temp*2;
    }
    return temp/2;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long int n;
        cin>>n;
        long long int i=0;
        long long int t1=n;
        while(t1>0)
        {
            long long int an=func(t1);
            t1=t1-an;
            i++;
        }
        cout<<i<<endl;
    }
    return 0;
}
