#include<iostream>
#include<cstdio>
#include<algorithm>
#include<vector>
typedef long long ll;
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		int arr[1000000]={0};
		int max=0;
		vector<int> v;
		for(int i=0;i<n;i++){
			int temp;
			scanf("%d",&temp);
			arr[temp]++;
			if(max<=arr[temp]){
				if(max==arr[temp])
					v.push_back(temp);
				else{
					v.clear();
					v.push_back(temp);
				}
				max=arr[temp];
			}
		}
		sort(v.rbegin(),v.rend());
		for(auto it=v.begin();it!=v.end();it++){
			printf("%d ",*it);
		}
		printf("\n");
	}
}
