#include <bits/stdc++.h>
using namespace std;
int a[105],c[105];
int main()
{
   // cout << "Hello World!" << endl;
   int n,m;
   scanf("%d%d",&n,&m);
   int  i;
   for(i=0;i<n;i++)
   scanf("%d",&a[i]);
   sort(a,a+n);
   reverse(a,a+n);
   int ans = 0;
   int j = 0;
   for(i = 0;i<n;i++)
   {
   	ans += (c[j]+1)*a[i];
   	c[j]++;
   	j = (j+1)%m;
   }
   printf("%d\n",ans);
    return 0;
}
