bool comp[30000005];
vi primes;

bool isprime(int n) {
    if (n <= 30000005 && !comp[n]) { return true; }
    for (int i : primes) {
        if (n%i == 0) { return false; }
    }
    return true;
}


bool isprimepower(int n) {
    if (isprime(n)) { return true; }
    FOR(i, 2, 50) {
        int base = (int) pow((double)n, 1/(i+0.0));
        if (abs(pow(base, i)-n) <= 0.001 && !comp[base]) { return true; }
        if (abs(pow(base+1, i)-n) <= 0.001 && !comp[base+1]) { return true; }

    }
    return false;
}
