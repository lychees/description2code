#include<bits/stdc++.h>
using namespace std;
int pal(string str)
{
	string  temp=str;
	reverse(str.begin(),str.end());
	if(temp==str)
	return 1;
	return 0;
}
int main()
{
	int t,i,l;
	cin>>t;
	cin.ignore();
	while(t--)
	{
		string str,temp,str2;
		getline(cin,str);
		if(pal(str))
		{
			cout<<str.length()<<endl;
			continue;
		}	
		l=str.length();
		for(i=0;i<l-1;i++)
		{
			temp=str.substr(0,i+1);
			reverse(temp.begin(),temp.end());
			str2=str+temp;
			
			temp.erase(0,temp.length());
			
			if(pal(str2))
			{
	
			break;
			}
			else
			str2.erase(0,str2.length());
		}
		cout<<str2.length()<<endl;
	}
	
}
