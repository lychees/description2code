#include<bits/stdc++.h>
using namespace std;
#define initial 0
#define waiting 1
#define ll long long int
ll a[101][101],ans=0,n;
ll status[105][105];
int dfs(int i,int j)
{
   // cout<<i<<" "<<j<<endl;
    //getchar();
    if(i<1 || i>n || j<1 || j>n)
    return 0;
    if(i==n && j==n)
    {
        ans++;
        return 0;
    }
    if(a[i+1][j]==0 && status[i+1][j]==initial)
    {
        status[i+1][j]=waiting;
        dfs(i+1,j);
        status[i+1][j]=initial;
    }
    if(a[i-1][j]==0 && status[i-1][j]==initial)
    {
        status[i-1][j]=waiting;
        dfs(i-1,j);
        status[i-1][j]=initial;
    }
    if(a[i][j+1]==0 &&  status[i][j+1]==initial)
    {
        status[i][j+1]=waiting;
        dfs(i,j+1);
        status[i][j+1]=initial;
    }
    if(a[i][j-1]==0 && status[i][j-1]==initial)
    {
        status[i][j-1]=waiting;
        dfs(i,j-1);
        status[i][j-1]=initial;
    }
}
int main()
{
    ll i,j;
    scanf("%lld",&n);
    for(i=1;i<=n;i++)
    for(j=1;j<=n;j++)
    scanf("%lld",&a[i][j]);
    //cout<<"vsdf"<<endl;
    if(a[1][1]==0)
    {
        status[1][1]=waiting;
        dfs(1,1);
    }
    printf("%lld\n",ans);
}
