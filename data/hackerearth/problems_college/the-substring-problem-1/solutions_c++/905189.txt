#define Sd(x) scanf("%d",&x)

int main()
{
	int t,n,max,i,j,max_ind;
	string str[101];
	char str_search;
	int count[26];
	Sd(t);
	while(t--)
	{
		Sd(n);
		max=0;
		memset(count,0,sizeof(count));
		FOR(i,n)
		{
			cin>>str[i];
			FOR(j,26)
			{
				str_search='a'+j;
				if(str[i].find(str_search)!=std::string::npos)
				{
					count[j]++;
				}
			}
		}
		FOR(i,26)
		{
			if(count[i]>max)
			{
				max=count[i];
				max_ind=i;
			}
		}
		printf("%c\n",max_ind+'a');
	}
	return 0;
}
