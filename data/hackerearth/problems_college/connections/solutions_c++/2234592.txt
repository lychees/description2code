#include<bits/stdc++.h>
using namespace std;
int main()
{
   	int n;
   	cin>>n;
   	int c;
   	cin>>c;
   	int T[n+1][n+1];
   	int a,b,flag;
   	for(int i=0;i<c;i++)
   	{
   		cin>>a>>b;
   		T[a][b]=1;
   		T[b][a]=1;
   	}
   	int q;
   	cin>>q;
   	while(q--)
   	{
   		cin>>a>>b;
   		if(T[a][b]==1)
   		{
   			cout<<"Direct Connection"<<endl;
   		}
   		else 
   		{
   			flag=0;
   			for(int k=1;k<=n;k++)
   			{
   				if(k!=a && (T[a][k]==1 && T[k][b]==1))
   				{
   					flag=1;
   					cout<<"Indirect Connection"<<endl;
   					break;
   				}
   			}
   			if(flag==0)
   			{
   				cout<<"No Connection"<<endl;
   			}
   		}
   	}
   	
}
