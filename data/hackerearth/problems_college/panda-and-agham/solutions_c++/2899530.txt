#include <bits/stdc++.h>
#include <stdio.h>
#include <stdlib.h>

using namespace std;


bool check(long long int dist,long long int x[],long long int nc,long long int n)
{
    long long int pos=x[0],count=1;
    for (long long int i=1;i<n;i++)
    {
        if (x[i]-pos>=dist)
        {
            count++;
            pos=x[i];
        }
        if (count==nc)
            return true;
    }
    return false;
}

long long int bs(long long int x[],long long int nc,long long int n)
{
    long long int l=1,r=x[n-1]-x[0]+1;
    while(l<r)
    {
        long long int mid=l+(r-l)/2;
        if (check(mid,x,nc,n)==true)
            l=mid+1;
        else
            r=mid;
    }
    return (l);
}


int main()
{
    long long int t,n,nc,i,x[100001],l;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld %lld",&n,&nc);
        for (i=0;i<n;i++)
            scanf("%lld",&x[i]);
        sort(x,x+n);
        l=bs(x,nc,n);
        printf("%lld\n",l-!check(l,x,nc,n));
    }
    return 0;
}
