#include<bits/stdc++.h>
using namespace std;

int main(){
    int T;
    scanf("%d",&T);
    while(T--){
        int N,M;
        scanf("%d %d",&N,&M);
        vector<int> degree(N+2,0);
        for(int i=0; i<M; i++){
            int u,v;
            scanf("%d %d",&u,&v);
            degree[u]++;
            degree[v]++;
        }
        int odds=0;
        for(int i=1; i<=N; i++)
            if(degree[i]%2==1)
                odds++;
        if(odds == 0 || odds == 2)
            printf("Yes\n");
        else printf("No\n");
    }
    return 0;
}
