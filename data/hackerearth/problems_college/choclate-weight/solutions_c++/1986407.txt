#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	vector<int> v(n+1);
	for(int i=0;i<n;i++)
	cin>>v[i];
	int q;
	cin>>q;
	for(int i=0;i<q;i++)
	{
		int x;
		int flag=0;
		cin>>x;
		for(int i=0;i<n;i++)
		{
			if(v[i]==x)
			{
				flag=1;
				break;
			}
		}
		if(flag==0)
		{
			for(int i=0;i<n-1;i++)
			{
				for(int j=i+1;j<n;j++)
				{
					if(v[i]+v[j]==x)
					{
						flag=1;
						break;
					}
				}
				if(flag==1)
				break;
			}
			if(flag==1)
			cout<<"I Like You"<<endl;
			else
			cout<<"I Hate You"<<endl;
		}
		else
		cout<<"I Like You"<<endl;
	}
	return 0;
}
