#include<iostream>
#include<algorithm>
#include<cmath>
#include<iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        string s;
        long long int i,j,k,l,m,n,ans=0,temp=0;
        cin>>s;
        l=s.length();
        ans=((l)*(l+1))/2;
        for(i=0;i<l;i++)
        {
            if(s[i]!='a'&&s[i]!='z')
            {
                temp++;
            }
            else
            {
                ans-=((temp)*(temp+1))/2;
                temp=0;
            }
        }
        ans-=((temp)*(temp+1))/2;
        if(ans<=0)
        {
            cout<<"0"<<"\n";
        }
        else
            cout<<ans<<"\n";
    }
    return 0;
}
