#include <iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int B,W,x,y,z,res1,res2;
    	cin>>B>>W;
    	cin>>x>>y>>z;
    	res1=B*x+W*y;
    	if(x<y)
    	{
    		if((x+z)<y)
    		{
    			y=x+z;
    		}
    	}
    	if(y<x)
    	{
    		if((y+z)<x)
    		{
    			x=(y+z);
    		}
    	}
    	res2=B*x+W*y;
    	if(res1<=res2)
    	{
    		cout<<res1<<endl;
    	}
    	else
    	{
    		cout<<res2<<endl;
    	}
    }
}
