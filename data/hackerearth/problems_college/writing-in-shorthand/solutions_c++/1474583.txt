#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,m;
    scanf("%d %d",&n,&m);
    map<string ,string>mp;
    for(int i = 0;i<m;i++){
    	string s1,s2;
    	cin>>s1>>s2;
    	if(s1.length()<=s2.length()){
    		mp[s1] = s1;
    	}
    	else mp[s1] = s2;
    }
    for(int i = 0;i<n;i++){
    	string str;
    	cin>>str;
    	cout<<mp[str]<<" ";
    }
    
    return 0;
}
