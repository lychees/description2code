#include<iostream>
#include<math.h>
#include<string.h>
using namespace std;
int mat[1001][1001];
long long int dp[1001][1001];
int main()
{
	int n,k;
	long long int it;
	cin>>n>>k;
	float cub=0.0;
	
	memset(dp,0,sizeof(dp));
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
		{
			cin>>mat[i][j];
		}
	}

	dp[0][0]=mat[0][0];
	for(int i=1;i<n;i++)
	{
		dp[0][i]=dp[0][i-1]+mat[0][i];
		dp[i][0]=dp[i-1][0]+mat[i][0];
	}
	for(int i=1;i<n;i++)
	{
		for(int j=1;j<n;j++)
		{
			dp[i][j]=dp[i-1][j]+dp[i][j-1]-dp[i-1][j-1]+mat[i][j];
		}
	}
	long long int sum=0,t;
	int count=0;
//	float cub;
	for(int i=k-1;i<n;i++)
	{
		for(int j=k-1;j<n;j++)
		{
			sum=dp[i][j];
			if(i-k>=0) sum-=dp[i-k][j];
			if(j-k>=0) sum-=dp[i][j-k];
			if((j-k>=0) && (i-k>=0)) sum+=dp[i-k][j-k];
			cub=cbrt(sum);
			t=(int)cub;
			if((t*t*t) == sum)
			count++;
		}
		
	}
	cout<<count<<endl;
//	system('pause');
	return 0;
	
}
