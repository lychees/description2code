#include<bits/stdc++.h>
#define m 1000000007
long long int powe(long long int x,long long int y)
{
    if (y==0)
        return 1;
    else if (y%2==0)
        return (powe(x,y/2)*powe(x,y/2)%m);
    else
        return ((x*powe(x,y/2)%m)*powe(x,y/2)%m);
}
int main()
{
    long long t;
    scanf ("%lld",&t);
    while(t--)
    {
        long long int n,count=0;
        scanf ("%lld",&n);
        while(n>0)
        {
            if (n%2==1)
            {
                count++;
            }
            n=n/2;
            if (n==0)
                break;
        }
        printf ("%lld\n",count);
    }
    return 0;
}
