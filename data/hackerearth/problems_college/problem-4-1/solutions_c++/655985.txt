#include <iostream>
using namespace std;

int main()
{
	int n;
	cin>>n;
	int arr[101]={0};
	for(int i=0;i<n;i++)
	{
		int a; cin>>a;
		arr[a]++;
	}
	
	for(int i=1;i<=100;i++) arr[i]*=i;
	
	int incl = arr[1];
  	int excl = 0;
 	int excl_new;
 	 
 
	for (int i = 2;i<=100; i++)
  	{
    	excl_new = (incl > excl)? incl: excl;
	    incl = excl + arr[i];
    	excl = excl_new;
  	}
	int ans=incl>excl?incl:excl;
	cout<<ans;
	return 0;
}
