#include <iostream>
#include <stdlib.h>
using namespace std;
#define LLI long long int
#define LI long int

void swap(LLI A[],LI i,LI j)
{
	LLI tmp=A[i];
	A[i] = A[j];
	A[j]=tmp;
}

int random_partition(LLI A[], LI start,LI end)
{	
	LLI pvt = A[start];
	LI i = start+1;
	for(LI j=start+1;j<=end;j++)
	{
		if(A[j]>pvt)
		{
			swap(A,i,j);
			i+=1;
		}
	}
	swap(A,i-1,start);

	return i-1;
}

int partition(LLI A[],LI start,LI end)
{
	LI i = rand()%(end+1); //random index
	swap(A,i,start);
	return random_partition(A,start,end);
}

void quick_sort(LLI A[],LI start,LI end)
{
	if(start<end)
	{
		LI mid = random_partition(A,start,end);
		quick_sort(A,start,mid);
		quick_sort(A,mid+1,end);
	}
}

int main()
{
	int testc;
	cin>>testc;
	for(int i=0;i<testc;i++)
	{
		LI size;
		cin>>size;
		LLI Arr[100000];
		for(LI j=0;j<size;j++)
			cin>>Arr[j];

		quick_sort(Arr,0,size-1);

		for(LI j=0;j<size;j++)
			cout<<Arr[j]<<" ";
		cout<<endl;
	}
	return 0;
}
