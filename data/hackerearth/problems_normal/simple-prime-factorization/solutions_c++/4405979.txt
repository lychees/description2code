#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int a[1000005];
vector <pair<int,int>> v[1000005];
void sieve()
{
	int i,j,p,c;
	for(i=2;i<=1000000;i++)
	{
		if(!a[i])
		{
			v[i].push_back(make_pair(i,1));
			for(j=i*2;j<=1000000;j+=i)
			{
				p=j;
				c=0;
				while(p%i==0)
				{
					p/=i;
					c++;
				}
				v[j].push_back(make_pair(i,c));
				a[j]=1;
			}
		}
	}
}
int main()
{
    sieve();
    int t,n,i,p,c;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	if(n==2)
    	{
    		cout<<"2^1"<<endl;
    	}
    	else if(!a[n])
    	{
    		cout<<"2^0"<<endl;
    	}
    	else
    	{
	  		if(n%2!=0)
			{  
	    		cout<<"2^0*";
			}
    		for(i=0;i<v[n].size();i++)
    		{
    			if(i==v[n].size()-1)
    			{
    				cout<<v[n][i].first<<"^"<<v[n][i].second<<endl;
    			}
    			else
    			{
    				cout<<v[n][i].first<<"^"<<v[n][i].second<<"*";
    			}
    		}
    	}
    }
    return 0;
}
