#include <iostream>
#include <string>

using namespace std;


int countSetBits(string n,int type)
{
	unsigned int Zerocount = 0;
	unsigned int Onecount = 0;
	int len = n.length();

	for (int i = 0; i < len; ++i)
	{
		if (n[i] == '0')
		{
			++Zerocount;
		}
		else
		{
			++Onecount;
		}
		
	}
	return (type == 0? Zerocount:  Onecount);
}

int main()
{
	ios::sync_with_stdio(false);
	int cases;
	cin >> cases;
	while (cases--)
	{
		string input;
		int days;
		cin >> input >> days;
	
		int output = 0;
		int total_inversions = (days * (days + 1)) / 2;
		if (total_inversions & 1 )
		{
			output = countSetBits(input, 0);
		}
		else
		{
			output = countSetBits(input, 1);
		}
		cout << output << endl;

	}
    return 0;
}
