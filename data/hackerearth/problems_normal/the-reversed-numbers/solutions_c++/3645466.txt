#include <iostream>

using namespace std;

int reverse(int N)
{
	int R = 0;

	while (N)
	{
		R = R*10 + (N % 10);
		N = N / 10;
	}

	return R;
}

int main()
{
	int T, N, M;

	cin >> T;

	while (T--)
	{
		cin >> N >> M;
		cout << reverse(reverse(N) + reverse(M)) << endl;
	}

	return 0;
}
