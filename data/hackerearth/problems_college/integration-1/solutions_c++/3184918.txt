#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	int t;
	cin>>t;
	int x=1;
	while(t--)
	{
	long long int i,n,p,u,r,s;
	cin>>n>>p>>u>>r>>s;
	
	for(i=0;i<n;i++)
	{
		if(p+r<=u)
		{
			p+=r;
		}
		else if(p-s>=0)
		{
			p-=s;
		}
		else if((p+r>u )&& (p-s<0))
		{
			break;
		}
		
	}
	 printf("Case #%d: %lld\n",x,p);
	 x++;
	}
	return 0;
}
