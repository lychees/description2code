#include <iostream>
using namespace std;

int main()
{
    int n,a,b;
    cin>>n>>a>>b;
    int sum=0;
    for(int i=a;i<n;i+=a)
    {
        if(i%a==0||i%b==0) sum+=i;	
    }
    
    for(int i=b;i<n;i+=b)
    {
        if((i%a==0||i%b==0)&&!(i%a==0&&i%b==0)) sum+=i;	
    }
    cout<<sum<<endl;
}
