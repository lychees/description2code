def checkit(n):
    if '21' in n or int(n) % 21 == 0:
        return 0
    else:
        return 1


t = int(raw_input())

for i in range(t):
    n = raw_input()
    if checkit(n):
        print "The streak lives still in our heart!"
    else:
        print "The streak is broken!"
