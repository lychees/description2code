#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int n,a[200100];
		cin>>n;
		long long int i,ans=0;
		for(i=0;i<n;i++) cin>>a[i];
		sort(a,a+n);
		for(i=0;i<n;i+=2)
		{
			ans+=(a[i]*a[i+1]);
		}
		cout<<ans<<"\n";
	}
    //cout << "Hello World!" << endl;
    return 0;
}
