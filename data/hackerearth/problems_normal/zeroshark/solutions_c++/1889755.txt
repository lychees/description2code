#include<bits/stdc++.h>
#define maxi 100005
using namespace std;

typedef long long ll;

int t,n;

int mod = (1e+9);
int MOD=mod+7;

ll dp[maxi][2][2];
ll ans;
ll com(int idx,int prev,int cnt)
{
    //cout<<idx<<endl;
    if(idx==n+1)
    {
        if(cnt==1)
        {
            return 1;
        }
        else
        {
            return 0;
        }
    }

    if(dp[idx][prev][cnt]!=-1)
    {
        return (dp[idx][prev][cnt])%MOD;
    }

    if(cnt==1)
    {
       if(prev==0)
       {
          ans = (com(idx+1,1,1))%MOD;
       }
       else
       {
         ans = (com(idx+1,0,1)+com(idx+1,1,1))%MOD;
         if(ans>=MOD)
         {
            ans-=MOD;
         }
       }
    }
    else
    {
        if(prev==0)
        {
            ans=(com(idx+1,0,1)+com(idx+1,1,0))%MOD;
            if(ans>=MOD)
            {
              ans-=MOD;
            }
        }
        else
        {
            ans=(com(idx+1,1,0)+com(idx+1,0,0))%MOD;
            if(ans>=MOD)
             {
               ans-=MOD;
             }
        }
    }

    dp[idx][prev][cnt]=(ans)%MOD;
    return dp[idx][prev][cnt];


}

int main()
{
    cin >>t;
    while(t--)
    {
        memset(dp,-1,sizeof(dp));
        cin >> n;

        ll ans = (com(2,1,0)+com(2,0,0))%MOD;
        cout<<ans<<endl;

    }
    return 0;
}
