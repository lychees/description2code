#include <iostream>
#include <map>
#include <string>
#include <vector>
#include <utility>
using namespace std;

int main() {
	int n;
	cin >> n;
	vector<int> lg2;
	vector<string> lg3;
	//string an;
	//cin >> an;
	//cout << an;
	//return 0;
	for(int i = 0;i < n;i++)
	{
		string temp;
		cin >> temp;
		bool lel = false;
		for(int j = 0;j < lg3.size();j++)
		{
			if(lg3[j] == temp)	{lel = true;break;}
		}
		if(lel)
		{
			for(int j = 0;j < lg3.size();j++)
			{
				if(lg3[j] == temp)	{lg2[j]++;break;}
			}
		}
		else
		{
			lg3.push_back(temp);
			lg2.push_back(1);
		}
	}
	for(int i = 0;i < lg3.size();i++)
	{
		for(int j = i+1;j < lg2.size();j++)
		{
			if(lg2[i] < lg2[j])
			{
				int tem = lg2[i];
				lg2[i] = lg2[j];
				lg2[j] = tem;
				string temp2;
				temp2 = lg3[i];
				lg3[i] = lg3[j];
				lg3[j] = temp2;
			}
		}
	}
	for(int i = 0;i < lg3.size();i++)
		cout << lg3[i] << " " << lg2[i] << endl;
	return 0;
}
