#include <bits/stdc++.h>
using namespace std;

long sqr(long a)
{
	return a*a;
}

int main()
{
    long p,i,n,ctr,pt;
    vector<long> v;
    //cout << "Hello World!" << endl;
    
    cin>>n;
    
    for(i=0;i<n;i++)
    {
    	cin>>p;
    	v.push_back(p);
    }
    
    sort(v.begin(),v.begin()+n);
    pt=0;
    ctr=0;
    for(i=0;pt!=n;i++)
	{
		if(sqr(i+1)>=v[pt])
		{
			ctr+=(sqr(i+1)-v[pt]);
			pt++;
		}
		
		else
		{
			ctr+=sqr(i+1);
		}
		
	}
    
    cout<<ctr<<endl;
    
    
    return 0;
}
