#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,i,diff,max_dif;
    int arr[1000];
    cin>>n;
    for(i=0;i<n;i++)
    cin>>arr[i];
    sort(arr,arr+n);
    max_dif=0;
    for(i=1;i<n-2;i++)
    {
    	diff=arr[i+2]-arr[i];
    	if(diff>max_dif)
    	max_dif=diff;
    }
    diff=arr[n-1]-arr[n-2];
    if(diff>max_dif)
    max_dif=diff;
    diff=arr[n-1]-arr[n-3];
    if(diff>max_dif)
    max_dif=diff;
    diff=arr[1]-arr[0];
    if(diff>max_dif)
    max_dif=diff;
    diff=arr[n-1]-arr[0];
    if(max_dif>diff)
    max_dif=diff;
    cout<<max_dif<<endl;
    return 0;
}
