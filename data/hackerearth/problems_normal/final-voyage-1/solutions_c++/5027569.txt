#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	int n;
	int m;
	while(t--)
	{
	cin>>n>>m;
	int a[n+1];
	int v[n+1];
	for(int i=1;i<=n;i++ )
	cin>>a[i];
	for(int i=1;i<=n;i++)
	cin>>v[i];
	int dp[n+1][m+1];
	for(int i=0;i<=n;i++)
	dp[i][0]=0;
	for(int i=0;i<=m;i++)
	dp[0][i]=0;
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=m;j++)
		{
			if(a[i]>j)
			dp[i][j]=max(dp[i-1][j],dp[i][j-1]);
			else if(a[i]==j)
			dp[i][j]=max(v[i],max(dp[i-1][j],dp[i][j-1]));
			else if(a[i]<j)
			dp[i][j]=max(v[i]+dp[i-1][j-a[i]],max(dp[i-1][j],dp[i][j-1]));
		}
	}
/*	for(int i=0;i<=n;i++)
	{
		for(int j=0;j<=m;j++)
		{
			cout<<dp[i][j]<<"   ";
		}
		cout<<endl<<endl;
}
cout<<endl; */
	cout<<dp[n][m]<<endl;
}
}
