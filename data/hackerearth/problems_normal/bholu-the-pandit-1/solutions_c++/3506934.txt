#include <iostream>
#define ll long long
#define MOD 1000000007
using namespace std;

ll gcd(ll a, ll b)
{
    if (a == 0)
        return b;
    return gcd(b%a, a);
}

int main()
{
    int T;
    cin>>T;
    while(T--){
    	ll a,b,c;
    	cin>>a>>b>>c;
    	ll side = gcd(a,gcd(b,c));
    	a=a/side;
    	b=b/side;
    	c=c/side;
    	ll count = (((a*b)%MOD)*c%MOD)%MOD;
    	cout<<side<<" "<<count<<"\n";
    }
    return 0;
}
