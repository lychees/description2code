#include <bits/stdc++.h>
using namespace std;
int a[100];
int dp[100][100000];
bool solve(int ind,int sum,int totsum,int n){
	if(ind==n){
		if(sum==totsum){
			return true;
		}
		return false;
	}
	if(sum>totsum){
		return false;
	}
	if(dp[ind][sum]!=-1){
		return dp[ind][sum];
	}
	return dp[ind][sum]=solve(ind+1,sum+a[ind],totsum,n)||solve(ind+1,sum,totsum,n);
}
int main() {
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		int sum=0;
		for(int i=0;i<n;i++){
			cin>>a[i];
			sum+=a[i];
		}
		memset(dp,-1,sizeof(dp));
		if(sum%2!=0){
			cout<<"NO"<<endl;
		}
		else{
			bool temp=solve(0,0,sum/2,n);
			if(temp){
				cout<<"YES"<<endl;
			}
			else{
				cout<<"NO"<<endl;
			}
		}
		
	}
	return 0;
}
