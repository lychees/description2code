TC = input()
for t in range(TC):
    N = input()
    W = [int(x) for x in raw_input("").split()]
    H = [int(x) for x in raw_input("").split()]
    flag = 1
    BH = W[0]
    w = sorted(W[1:N])
    H.sort()
    for i in range(1, N):
        if ((BH*H[N-1]) <= (w[i-1]*H[N-i-1])):
            flag = 0
    if (flag == 1):
        print("YES")
    else:
        print("NO")
