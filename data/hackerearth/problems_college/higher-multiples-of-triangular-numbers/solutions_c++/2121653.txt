#include <iostream>
using namespace std;

int main()
{
    int t=0;
    cin>>t;
    while(t-->0)
    {
    	int n=0;
    	cin>>n;
    	for(int i=1;i<1001;i++)
    	{
    		int count=0;
    		int tn=i*(i+1)/2;
    		for(int j=1;j<tn;j++)
    		{
    			if(tn%j==0)
    				count++;
    		}
    			if(count>=n)
    			{
    				cout<<tn<<endl;
    				break;
    			}
    	}
    }
    return 0;
}
