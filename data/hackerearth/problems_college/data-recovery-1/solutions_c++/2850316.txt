ll pow_mod(ll a, ll b) {
	ll res = 1;
	while(b) {
		if(b & 1)
			res = (res * a) % MOD;
		a = (a * a) % MOD;
		b >>= 1;
	}
	return res;
}
vector<int> v[1005], v1[1000005];
int cnt[1000005];
set<int> s;	
bool done[1000005];
int main() {
	int n;
	si(n);
	for(int i = 0; i < n; ++i) {
		int k;
		si(k);
		for(int j = 0; j < k; ++j) {
			int tmp;
			si(tmp);
			v[i].push_back(tmp);
			if(j) {
				cnt[tmp]++;
			}
		}
	}
	for(int i = 0; i < n; ++i) {
		reverse(v[i].begin(), v[i].end());
	}
	for(int i = 0; i < n; ++i) {
		int val = v[i][(int)(v[i].size() - 1)];
		v1[val].push_back(i);
		if(cnt[val] == 0) {
			s.insert(val);
		}
	}
	vector<int> res;
	while(!s.empty()) {
		set<int> :: iterator it = s.begin();
		int val = (*it);
		s.erase(it);
		res.push_back(val);
		done[val] = true;
		for(int i = 0; i < v1[val].size(); ++i) {
			int idx = v1[val][i];
			v[idx].pop_back();
			if(!v[idx].empty()) {
				int tmp = v[idx][(int)(v[idx].size() - 1)];
				while(done[tmp] && !v[idx].empty()) {
					v[idx].pop_back();
					tmp = v[idx][(int)(v[idx].size() - 1)];
				}
				if(!v[idx].empty()) {
					cnt[tmp]--;
					if(cnt[tmp] == 0) {
						s.insert(tmp);
					}
					v1[tmp].push_back(idx);
				}
			}
		}
	}
	for(int i = 0; i < (int)(res.size()); ++i) {
		cout << res[i] << " ";
	}
	cout << endl;
	return 0;
}
