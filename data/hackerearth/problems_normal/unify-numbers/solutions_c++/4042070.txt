#include <bits/stdc++.h>
using namespace std;

const long long maxn = 1e6;
long long a[maxn];
long long n; 

long long cost(int mid) {
  long long c = 0;
  for (int i = 0; i < n; ++i)
    c += abs(a[i] - mid);
  return c;
}

int main() {
  ios::sync_with_stdio(0);
  cin.tie(0);

  long long t; cin >> t;
  while (t--) {
    cin >> n;
    long long maxi = 0;
    for (long long i = 0; i < n; ++i) {
      cin >> a[i];
      maxi = max(maxi, a[i]);
    }
    long long l = 0, r = maxi + 1;
    while (l < r) {
      long long mid = (l + r + 1) >> 1;
      if (cost(mid - 1) >= cost(mid)) l = mid;
      else r = mid - 1;
    }
    cout << cost(l) << endl;
  }
  

  return 0;
}
