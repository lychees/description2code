t = int(raw_input())
for i in range(t):
    c = 0
    a, b = raw_input().strip().split(' ')
    a, b = [int(a), int(b)]
    while a > 0 and b > 0:
        if (a % 10+b % 10 >= 10):
            c += 1
        a, b = [int(a/10), int(b/10)]
    if c == 0:
        print("No carry operation")
    elif c == 1:
        print("1 carry operation")
    else:
        print(str(c)+" carry operations")
