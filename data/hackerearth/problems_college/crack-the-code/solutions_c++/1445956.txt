#include <iostream>
using namespace std;
 
int main()
{
    unsigned long long int t,a,b,l,l1,count,k;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b;;
    	l=0;
    	l1=1;
    	k=l1+l;
    	l1=k;
    	l=l1;
    	count=0;
    	while(k<=b)
    	{
    		k=l+l1;
    		l=l1;
    	   if(k>=a && k<=b)
    	    {count=count+1;}
    	 l1=k;
    	 }
    if(count%2==0)
    {cout<<"VALID"<<endl;}
    else
    {cout<<"INVALID"<<endl;}
    }
    return 0;
}
