#include <iostream>
#include<cstdio>
using namespace std;
struct point{
	float x;
	float y;
};
void sort(point hc[])
{
	for(int i=0;i<3;i++)
	{
		for(int j=i;j<3;j++)
		{
			if(hc[i].x > hc[j].x)
			{
				point temp=hc[i];
				hc[i]=hc[j];
				hc[j]=temp;
			}
			else if(hc[i].x == hc[j].x)
			{
				if(hc[i].y > hc[j].y)
				{
					point temp=hc[i];
					hc[i]=hc[j];
					hc[j]=temp;
				}
			}
		}
	}
}
int main()
{
	float a,b,c,d,e,f;
	point hc[3];
	cin>>a>>b>>c>>d>>e>>f;
	//cout<<a;
	hc[0].x=(c+a-e);
	hc[1].x=a-c+e;
	hc[2].x=c-a+e;
	hc[0].y=(d+b-f);
	hc[1].y=b-d+f;
	hc[2].y=d-b+f;
	sort(hc);
	printf("%0.4f %0.4f\n",hc[0].x,hc[0].y);
	printf("%0.4f %0.4f\n",hc[1].x,hc[1].y);
	printf("%0.4f %0.4f\n",hc[2].x,hc[2].y);
}
