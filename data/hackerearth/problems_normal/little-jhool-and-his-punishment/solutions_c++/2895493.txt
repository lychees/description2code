#include <iostream>
using namespace std;

int main()
{
    int t,n,b,g;
    scanf("%d",&t);
    while(t--){
    	scanf("%d%d%d",&n,&b,&g);
    	int pairs=n-1;
    	int tea=min(b,g);
    	int wh=pairs-tea;
    	if(wh>tea)
    		printf("Little Jhool wins!\n");
    		else{
    			printf("The teacher wins!\n");
    		}
    	
    }
    return 0;
}
