#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int ans(int x,int n)
{
	int c=0;
	while(x!=n)
	{
		if(n>x)
		{
			if(n%2!=0)
			{
				n/=2;
				n++;
				c+=2;
			}
			else
			{
				c++;
				n/=2;
			}
		}
		else
		{
			n++;
			c++;
		}
	}
	return c;
}
int main()
{
    int t,n,x,c;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d%d",&x,&n);
    	c=0;
    	if(x>n)
    	{
    		c=x-n;
    	}
    	else if(x<n)
    	{
    		c=ans(x,n);
    	}
    	printf("%d\n",c);
    }
    return 0;
}
