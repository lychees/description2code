#include <bits/stdc++.h>
using namespace std;

#define MOD 1000000007

int A[1030];
long long fact[1030];

int main()
{
	int N, M;
	scanf("%d %d", &N, &M);
	
    for(int i=1; i<=N; i++)
    	for(int j=1; j<=M; j++)
    		A[i^j]++;
    
    long long ans = 1;
    
    fact[0] = 1;
    
    for(int i=1; i<=1000; i++)
    {
    	fact[i] = fact[i-1]*i;
    	
    	if(fact[i]>=MOD)
    		fact[i]%= MOD;
    }
    
    for(int i=0; i<=1024; i++)
    	ans = (ans*fact[A[i]])%MOD;
    
    printf("%lld\n", ans);
    
    return 0;
}
