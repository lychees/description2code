t = int(raw_input())
for i in range(t):
    n = int(raw_input())
    mylist, coord_x, coord_y = [], 0, 0
    for j in range(n):
        mylist.append(map(float, raw_input().split()))
    for k in range(n):
        coord_x += mylist[k][0]
        coord_y += mylist[k][1]
    print '(%0.2f , %0.2f)' % (coord_x/n, coord_y/n)
