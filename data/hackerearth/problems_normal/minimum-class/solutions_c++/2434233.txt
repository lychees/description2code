#include <iostream>
#include <math.h>

using namespace std;

int main(){
    int t;
    int n;
    int j;
    int ans;
    cin>>t;
    while(t--){
        cin>> n;
        j = (int)sqrt(n);
        if(sqrt(n) > j ){
            ans = j+ round(sqrt(n));
        }
        else if(sqrt(n) == j)
            ans = 2 * j-1;

        cout<< ans << endl;
    }
}
