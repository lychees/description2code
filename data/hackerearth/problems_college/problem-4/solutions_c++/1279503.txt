#include <iostream>
#include <string>
#include<cstring>
using namespace std;

#define N_SZ 26
#define INDEX(x) (x - 'a')
#define RINDEX(x) (x + 'a')

int w[N_SZ];
int main()
{
	string s;
    while(cin>>s)
    {
    	memset(&w[0], 0x00, N_SZ);
    	for(auto it = s.begin(); it != s.end(); ++it)
    	{
   			++w[INDEX(*it)];
    	}
    	
    	char c = 'a';
    	for (int i = 0; i < N_SZ; i++, c++)
    		for (int j = 0; j < w[i]; j++)
    			cout<<c;
    	
    	cout<<endl;
    }
    return 0;
}
