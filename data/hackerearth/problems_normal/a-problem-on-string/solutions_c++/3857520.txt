#include<iostream>
#include<cstdio>
#include<vector>
using namespace std;
typedef long long int ll;

int main()
{
    ll A[26];
    char x,y;
    string S;
    int n,q;
    for(int i=0;i<26;++i)
        A[i]=0;
    cin>>n;
    cin>>S;
    for(int i=0;i<n;++i)
        ++A[S[i]-'a'];
    //cout<<S<<endl;
    cin>>q;
    for(int i=0;i<q;++i)
    {
        scanf(" %c %c",&x,&y);
        if(x==y)
            cout<<(((A[x-'a']-1)*A[x-'a'])/2)<<endl;
        else
            cout<<((A[x-'a']*A[y-'a']))<<endl;
    }
    return 0;
}
