string s[11];
void build(){
	s[0]="_0";
	s[1]=".,?!1";
	s[2]="abc2";
	s[3]="def3";
	s[4]="ghi4";
	s[5]="jkl5";
	s[6]="mno6";
	s[7]="pqrs7";
	s[8]="tuv8";
	s[9]="wxyz9";
}
pair<int,int>go(char x){
	for(int i=0;i<10;i++){
		string z=s[i];
		for(int j=0;j<z.length();j++){
			if(z[j]==x){
				return mp(i,j+1);
			}
		}
	}
}
int main(){
	build();
	int t,i;
	string x;
	cin>>t;
	while(t--){
		int ans=0,st=1;
		cin>>x;
		for(i=0;i<x.length();i++){
			pair<int,int>tmp=go(x[i]);
			if(tmp.first==st)
			ans+=tmp.second;
			else{
				ans+=1;
				st=tmp.first;
				ans+=tmp.second;
			}
		}
		cout<<ans<<endl;
	}
	return 0;
}
