T = int(raw_input())
P = []
Q = []
for i in range(T):
    S = int(raw_input())
    P.append(S)    
for i in range(T):
	if P[i] == 0 and len(Q) != 0:
		Q.pop()
	elif P[i] != 0:
		Q.append(P[i])

print sum(Q)
