using namespace std;

int main()
{
	int m,count,ans=0,x,n,no,k;
	cin>>m;
	bool qu[100000]={0};
	for(int i=0;i<m;i++)
	{
		cin>>x;
		qu[x]=1;
	}
	cin>>n;
	cin.ignore();
	for(int i=0;i<n;i++)
	{
		char s[1000001];
		count=0;
		no=0;
		k=0;
		gets(s);
		while(s[k]!='\0')
		{
			if(s[k]==' ')
			{
				if(qu[no])
				{
				//cout<<no<<endl;
					count++;
				}
				no=0;
			}
			else
			{
				no=no*10+(s[k]-48);
			}
			k++;
		}
		if(no!=0)
			if(qu[no])
				count++;
		if(count>=m)
			ans++;
	}
	cout<<ans<<endl;
  	return 0;
}
