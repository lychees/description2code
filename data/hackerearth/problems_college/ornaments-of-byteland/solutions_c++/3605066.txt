#include <iostream>
#include<stdio.h>
#include<string.h>

using namespace std;

int main()
{
    char s[101];
    int a[101][26];
    int n,i,len;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        for(int j=0;j<26;j++){
            a[i][j]=0;
        }
    }
    for(i=0;i<n;i++)
    {
        scanf("%s",s);
        len = strlen(s);
        for(int j=0;j<len;j++){
            a[i][(int)s[j]-97]++;
        }
    }
    int count=0;
    int val=0;
    for(int i=0;i<26;i++){
        int arr=a[0][i];
        val=0;
        if(arr>0){
            val=1;
            for(int j=1;j<n;j++){
                if(a[j][i]>0){
                    val++;
                }
            }
        }
        if(val==n){
            count++;
        }
    }
    printf("%d",count);
    return 0;
}
