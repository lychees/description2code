#include<iostream>
using namespace std;
int a[1007];
int main()
{
	int n, q;
	cin >> n >> q;
	for (int i = 1; i <= q; i++)
	{
		int m;
		cin >> m;
		if (m == 0)
		{
			int l, r;
			cin >> l >> r;
			for (int j = l; j <= r; j++)
			{
				a[j] = 1 - a[j];
			}
		}
		else
		{
			int ans = 0;
			int l, r;
			cin >> l >> r;
			for (int j = l; j <= r; j++)
			{
				ans += a[j];
			}
			cout << ans << endl;
		}
	}
}
