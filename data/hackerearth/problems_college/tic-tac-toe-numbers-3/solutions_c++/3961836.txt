#include <iostream>
using namespace std;
typedef long long ll;
bool isPrime(ll N)
{
	if(N==2) return true;
	for(ll i=2; i*i<=N; i++)
		if(N%i==0) return false;
	return true;
}
bool isOdd(ll N) { return N%2!=0; }
bool isEven(ll N) { return N%2==0; }
ll counttillN(ll N)
{
	ll a=0; ll b=1;
	ll c;
	ll sum = 0;
	while(c<N)
	{
		c = a+b;
		for(ll i=b+1; i<c; i++)
		{
			sum += (i<N)?i:0;
		}
		a = b;
		b  = c;
	}
	return sum;
}
int main()
{
    int T; cin >> T;
    while(T--)
    {
    	int N; cin >> N;
    	ll sum = counttillN(N);
    	if(isPrime(sum)) cout << "Tic" << endl;
    	else if(isEven(sum)) cout << "Tac" << endl;
    	else cout <<"Toe" << endl;
    }
    return 0;
}
