#include<iostream>
#include<algorithm>
using namespace std;
#define B2(n) n, n+1, n+1, n+2
#define B4(n) B2(n), B2(n+1), B2(n+1), B2(n+2)
#define B6(n) B4(n), B4(n+1), B4(n+1), B4(n+2)
static const unsigned char SetBitsLookUpTable[256] =
{
    B6(0), B6(1), B6(1), B6(2)
};
int main(){
	int i,n,t,j;
	cin>>t;
	while(t--){
		cin>>n;
		int A[n];
		int B[n-1];
		for(i=0;i<n;i++)
			cin>>A[i];
		sort(A,A+n);
		
		for(i=0;i<n-1;i++){
				int z=A[i]^A[n-1];
				unsigned int count = 0;
  				count = SetBitsLookUpTable[z & 0xff] +SetBitsLookUpTable[(z >> 8) & 0xff] +SetBitsLookUpTable[(z >> 16) & 0xff] +SetBitsLookUpTable[z >> 24];
				//cout<<count<<endl;
				B[i]=count;
		}
		//for(i=0;i<n;i++)
		//cout<<B[i]<<" ";
		//cout<<endl;
		int y=n-1;
		sort(B,B+y);
		cout<<B[0]<<endl;
	}
}
