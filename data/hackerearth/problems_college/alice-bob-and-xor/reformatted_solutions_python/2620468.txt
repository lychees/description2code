k = int(raw_input())
for i in range(0, k):
    str = raw_input().split(" ")
    a = int(str[0])
    b = int(str[1])
    a = a-1
    if (a & 1):
        temp = 1
    else:
        temp = a

    xor_a = (a >> 1) & 1 ^ temp

    if (b & 1):
        temp = 1
    else:
        temp = b
    xor_b = (b >> 1) & 1 ^ temp
    print(xor_b ^ xor_a)
