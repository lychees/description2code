#include <iostream>
#include <vector>

using namespace std;
int sieve[1000002];
int prime_sum(int a){
	int head = 2;
	int eraser = 4;
	while(head < a){
		while(eraser <= a){
			sieve[eraser] = 0;
			eraser = eraser + head;
		}
		head = head+1;
		while(sieve[head]!=1 and head < a)head = head+1;
		eraser = 2*head;
	}
	int ans=0;
	for(int i=2;i<=a;i++){
		if(sieve[i] == 1)ans = ans + i;
	}
	return ans;
}
