#include <iostream>
#include <set>
using namespace std;
int prime_fact[1001][6];
int cnt[1001];
int primes[1001];

void preprocess() {
	primes[0] = primes[1] = 1;
	for(int i=2;i<=1000;i++) {
		if(!primes[i]) {
			int j = 2*i;
			while(j<=1000) {
				primes[j] = 1;
				prime_fact[j][cnt[j]++] = i;
				j+=i;
			}
			prime_fact[i][0] = i;
			cnt[i]++;
		}
	}
}

int main()
{
    int T,A,B;
    cin>>T;
    preprocess();
    while(T) {
    	cin>>A>>B;
    	set<int> s;
    	//cout<<A<<' '<<B<<endl;
    	for(int i=0;i<cnt[A];i++) {
    		s.insert(prime_fact[A][i]);
    		//cout<<prime_fact[A][i]<<endl;
    	}
    	for(int i=0;i<cnt[B];i++) {
    		s.insert(prime_fact[B][i]);
    		//cout<<prime_fact[A][i]<<endl;
    	}
    	if(!primes[s.size()]) {
    		cout<<"Yes"<<endl;
    	}
    	else {
    		cout<<"No"<<endl;
    	}
    	T--;
    }
    return 0;
}
