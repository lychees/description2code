#include<iostream>
using namespace std;
long long int flip(long long int temp)
{
	long long int reverse=0;
	   while( temp>0)
   {
      reverse = reverse * 10;
      reverse = reverse + temp%10;
      temp = temp/10;
	}	
	return reverse;
}
main()
{
	int t;
	long long int a,b,c;
	cin>>t;
	while(t--)
	{
		cin>>a>>b;
		a=flip(a);
		b=flip(b);
		c=flip(a+b);
		cout<<c<<endl;
	}
}
