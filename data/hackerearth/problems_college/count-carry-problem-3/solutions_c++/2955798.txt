#include<iostream>
using namespace std;
int main(){
	int test = 0;cin>>test;
	while(test--){
		long long int i,j,currentcarry=0,carry = 0;
		cin>>i>>j;
		while(i!=0 && j!=0){
		
			int d1,d2;
			d1 = i%10;d2 = j%10;
			i = i/10;j=j/10;
			currentcarry = (d1+d2+currentcarry)/10;
			if(currentcarry!=0)
				carry++;
		}
		if(i!=0){
			while(i!=0){
				int d1;
				d1 = i%10;
				currentcarry = (d1+currentcarry)/10;
				if(currentcarry!=0)
				carry++;
			}
		}
		else{
				while(j!=0){
				int d1;
				d1 = j%10;
				currentcarry = (d1+currentcarry)/10;
				if(currentcarry!=0)
				carry++;
			}	
		}
		if(carry == 0)
		cout<<"No carry operation"<<endl;
		else if(carry == 1)
		cout<<carry<<" carry operation"<<endl;
		else
		cout<<carry<<" carry operations"<<endl;
	}
	
	return 0;
}
