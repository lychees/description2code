#include<cstdio>
#include<stdlib.h>
#include<map>
#include<vector>
using namespace std;

int main()
{
int n;
scanf("%d",&n);

vector<long long int> a;
map<long long int,int> h;

int i,j;
long long int tmp;
for(i=0;i<n;i++)
{

scanf("%lld",&tmp);
a.push_back(tmp);
h[tmp]=i;
}

int c=0;
while(1)
{
int flag=0;
for(i=0;i<a.size()-1;i++)
{
for(j=i+1;j<a.size();j++)
{

if(h.count(llabs(a[i]-a[j]))==0)
{
//printf("%d %lld %lld \n",c,a[i],a[j]);
h[abs(a[i]-a[j])]=1;
a.push_back(llabs(a[i]-a[j]));
c=(c==0)?1:0;
flag=1;
break;
}
if(flag==1)
break;
}
}

if(i==a.size()-1 && j==a.size())
break;
}

if(c==1)
printf("Alice\n");
else
printf("Bob");
return 0;
}
