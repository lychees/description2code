'''
For calculation of A we simply sort the array and calculate how many places where (-1)**i
is negative and named it as 'max_neg'. Then place a loop in range of max_neg and place the
elements of sorted array, I use 'index' variable to get the current location in array

For calcuation of B we simply add the mod of all elements

Time Complexity: O(n)
'''

import sys  # for I/O
mod = 10**9 + 7
for cases in xrange(int(sys.stdin.readline())):
    n = int(sys.stdin.readline())  # Numner of elements in array
    arr = list(map(int, sys.stdin.readline().split()))
    A = B = 0
    for i in arr:
        B += abs(i)  # Calculation of B
    B %= mod
    arr.sort()
    index = 0
    # Counts the number of places where (-1)**i is negative
    max_neg = n/2 + n % 2
    for i in xrange(max_neg):
        A -= arr[index]
        A %= mod
        index += 1
    for i in xrange(index, n):
        A += arr[i]
        A %= mod
    print(A*B) % mod
