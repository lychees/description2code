#include <iostream>
#include<map>
#include<queue>
//#include<pair>
using namespace std;

int main()
{
	map<int,int>mymap;
	int x,y;
	cin>>x>>y;
	int arr[100005];
	for(int i=0;i<=100005;i++)
	{
		arr[i]=0;
	}
	priority_queue<pair<int,int>>pq;
	int k=0;
	for(int i=0;i<x;i++)
	{
		int x;
		cin>>x;
		if(mymap.count(x)==0)
		{
			mymap[x]=k;
			arr[k]++;
			k++;
		}
		else
		{
			arr[mymap[x]]++;
			
		}
		pq.push(make_pair(arr[mymap[x]],x));//note priority based on count so second to print used// first
		cout<<pq.top().second<< " "<<pq.top().first<<endl;
	}
	
	
    //cout << "Hello World!" << endl;
    return 0;
}
