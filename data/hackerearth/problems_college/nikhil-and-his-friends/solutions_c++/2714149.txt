#include <iostream>
using namespace std;

int main()
{
    long long T,N,X,COST[1000000],i,j,sum,k;
    cin>>T;
    for(i=0;i<T;i++){
    	cin>>N>>X;
    	for(j=0;j<N;j++)
    	cin>>COST[j];
    	j=0;
    	label:
    	sum=0;
    	for(k=j;k<N;k++)
    	{
    		sum=sum+COST[k];
    		if(sum==X){
    		cout<<"YES"<<endl;
    		break;
    		}
    		else if(sum>X)
    		{
    			if(j==N-1)
    			cout<<"NO"<<endl;
    			else{
    			j++;
    			goto label;
    			}
    		}
    		else if((k==N-1)&&(sum!=X))
    		{
    			cout<<"NO"<<endl;
    			break;
    		}
    		
    		
    	}
    	
    }
    return 0;
}
