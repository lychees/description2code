#include<iostream>
#include<cstdio>
#include<string>
#include<algorithm>
using namespace std;
using namespace std;
int main()
{
    int t;
    cin>>t;
    string a[4],b[4];
    a[1]="R";
    a[2]="RY";
    a[3]="RYY";
    //cout<<a[1]<<a[2]<<a[3];
    while(t--)
    {
        string s;
        cin>>s;
        int len=s.length(),i=0,no=0,j,f;
        while(i<len)
        {
            b[1]=s.substr(i,1);
            if(i<len-1)b[2]=s.substr(i,2);else b[2]="$$";
            if(i<len-2)b[3]=s.substr(i,3);else b[3]="$$$";
            //cout<<b[1]<<" "<<b[2]<< " "<<b[3]<<endl;
            f=0;
            for(j=3;j>=1;j--)
            {
                if(a[j]==b[j])
                {
                    i+=j;
                    f=1;
                    break;
                }
            }
            if(f==0) {no=1;break;}
        }
        if(no==1) cout<<"NO\n";
        else cout<<"YES\n";
    }
    return 0;
}
