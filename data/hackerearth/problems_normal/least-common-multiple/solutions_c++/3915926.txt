#include<bits/stdc++.h>
using namespace std;

long long n;

bool important(long long x)
{
    int i=2;
    if(x%i==0)
    {
        long long cur=1;
        for(;x%i==0;x/=i)
            cur*=i;
        if(cur+cur>n)
            return true;
    }
    for(int i=3;x+x>=n && i*1ll*i<=x;i+=2)
        if(x%i==0)
        {
            long long cur=1;
            for(;x%i==0;x/=i)
                cur*=i;
            if(cur+cur>n)
                return true;
        }
    if(x>1 && x+x>n)
        return true;
    return false;
}

int main()
{
    cin>>n;
    long long m=n/2+1;
    for(;m<n && !important(m);++m);
    cout<<m<<"\n";
}
