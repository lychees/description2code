#include<iostream>
#include<unordered_map>
#include<string>

using namespace std;

int main() {
  int t;
  string test;
  unordered_map<char, int> wordMap;
  cin >> t;
  while (t--) {
    test.clear();
    wordMap.clear();
    cin >> test;
    if (test.size() < 2) {
      cout << "No" << endl;
      continue;
    }
    bool can = false;
    for (int i = 0; i < test.size(); i++) {
      if (wordMap[test.at(i)] >= 1) {
        can = true;
        break;
      }
      wordMap[test.at(i)]++;
    }
    if (can) {
      cout << "Yes" << endl;

    }
    else {
      cout << "No" << endl;
    }

  }


  return 0;
}
