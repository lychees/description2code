#include <bits/stdc++.h>
using namespace std;
long long int n;
const long long int mod=(1e9)+7;
long long int a[100010];
long long int dp[510][110]={0};
int gx[110][110];
int main()
{
  cin>>n;
  for(int i=1;i<=100;i++)for(int j=1;j<=100;j++)gx[i][j]=__gcd(i,j);
  for(long long int i=1;i<=n;i++)cin>>a[i];
  long long int ans=0;
  for(long long int i=1;i<=n;i++){
    dp[i][a[i]]=(dp[i][a[i]]+1)%mod;
    for(long long int j=i-1;j>0;j--){
      if(a[i]<=a[j])continue;
      for(long long int k=1;k<=100;k++){
        long long int z=gx[a[i]][k];
        dp[i][z]=(dp[i][z]+dp[j][k])%mod;
      }
    }
  }
  for(long long int i=1;i<=n;i++)ans=(ans+dp[i][1])%mod;
  cout<<ans<<endl;
}
