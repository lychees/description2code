#include <iostream>
using namespace std;
string s;
int a[1000][1000]={0};
int check(int st,int e){
    
    if(st>e){return 0;}
    if(a[st][e]!=-1)return a[st][e];
 int ans=0;
 if(st+1<=e)
 ans+=check(st+1,e);
 //cout<<ans<<endl;
 int tmp=e;
 while(tmp>=st){
     if(s[tmp]==s[st])ans+=1+check(st+1,tmp-1);
     tmp--;
 }

// cout<<"assg "<<ans<<" to "<<st<<' '<<e<<endl;
 return a[st][e]=ans;
    
}

int main()
{int t;
cin>>t;
while(t--){
    
cin>>s;

for(int i=0;i<s.size();i++)
    for(int j=0;j<s.size();j++)a[i][j]=-1;
cout<<check(0,s.size()-1)<<endl;

/*for(int i=0;i<s.size();i++){
    for(int j=0;j<s.size();j++)cout<<a[i][j]<<' ';
    cout<<endl;}*/
}
    return 0;
}
