#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	long long a[n];
    	for(int i=0;i<n;i++)cin>>a[i];
    	
    	sort(a,a+n);
    	
    	long long ans=0;
    	
    	
    	for(int i=0;i<n;i+=2)
    	{
    		ans+=(a[i]*a[i+1]);
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
