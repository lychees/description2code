t = int(raw_input())
lis = []
for i in range(t):
    lis.append(raw_input())
# making a list of prime numbers till 128
prime = [2, 3]
for i in range(5, 128, 2):
    for j in prime:
        if i % j == 0:
            break
    else:
        prime.append(i)
for i in range(t):
    string = lis[i]
    new = ''
    for j in string:
        if ord(j) in prime:
            pass
        else:
            new = new+j
    if new != '':
        print new
    else:
        print "NULL"
