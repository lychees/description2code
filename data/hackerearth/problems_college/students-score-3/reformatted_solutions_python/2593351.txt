def abc(lis):
    currmax = lis[0]
    maxsofar = lis[0]
    for i in range(1, len(lis)):
        currmax = max(lis[i], currmax+lis[i])
        maxsofar = max(maxsofar, currmax)
    if maxsofar < 0:
        return 0
    else:
        return maxsofar


n = int(raw_input())
a = map(int, raw_input().split())
z = abc(a)
