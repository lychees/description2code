    #include <iostream>
    using namespace std;
     
    int p[1000001] = {0};
    long long sum[1000001] = {0};
     
    void sieve(){
    	for(int i = 2; i*i <= 1000000; i++){
    		if(!p[i]){
    			for(int j = i*i; j <= 1000000; j += i){
    				p[j] = 1;
    			}
    		}
    	}
    }
     
    void presum(){
    	for(int i = 2; i <= 1000000; i++){
    		if(!p[i]){
    			sum[i] = sum[i-1] + i;
    			continue;
    		}
    		sum[i] = sum[i-1];
    	}
    }
     
    int main()
    {
    	sieve();
    	presum();
        int t,x,y;
        cin >> t;
        while(t--){
        	cin >> x >> y;
        	cout << sum[y] - sum[x-1] << endl;
        }
        return 0;
    }
