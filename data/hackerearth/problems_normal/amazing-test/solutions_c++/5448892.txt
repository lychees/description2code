#include <bits/stdc++.h>
using namespace std;
#define ll long long int

ll ans[109][10009];
int main ()
{
	int t;
	cin >> t;
	while(t--)
	{
		int n,x,arr[109];
		cin >> n >> x;
		ll sum=0;
		for(int i=0;i<n;i++)cin >> arr[i],sum+=arr[i];
		if(sum<=x)cout << "YES" << endl;
		else
		{
			for(int i=0;i<=n;i++)
			{
				for(int j=0;j<=x;j++)
				{
					if(i==0)ans[i][j]=0;
					else if(j>=arr[i-1])
					{
						ans[i][j] = max(ans[i-1][j],arr[i-1]+ans[i-1][j-arr[i-1]]);
					}
					else ans[i][j] = ans[i-1][j];
				}
			}
			if(sum-ans[n][x]<=x)cout << "YES" << endl;
			else cout << "NO" << endl;
		}
	}
	
	return 0;
}
