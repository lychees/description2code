#include <iostream>
using namespace std;

int main()
{
    int n , q,i ,l,r;
    long long int sum ;
    cin>>n>>q;
    long int c,a[100000];
    c = 0;
    for(i = 0 ; i < n; i++)
    {	
    	cin>>c;
       	if( i != 0) 
    		a[i] = a[i-1] + c;
    	else a[i] = c;
	}	
    while(q--) {
   		cin>>l>>r;
   		if(l != 1)
   			sum = a[r-1] - a[l-2];
   		else sum = a[r-1];	
   		
     		cout <<sum<< endl;
    }

    
    return 0;
}
