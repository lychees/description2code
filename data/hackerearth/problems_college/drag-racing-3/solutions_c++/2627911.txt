#include <iostream>
#include <string>
#include <map>
#include <cmath>
using namespace std;

int main()
{
	int n,maxm=-1;
	string res;
	cin>>n;
	map<string,int> m;
	for(int i=0;i<pow(2,n)-1;i++)
	{
		string x,y;
		cin>>x>>y;
		if(m.find(x)==m.end())
			m[x]=1;		
		else m[x]++;
		if(m[x]>maxm)
		{
		  res=x;
		  maxm=m[x];
		}
	}
    cout << res<< endl;
    return 0;
}
