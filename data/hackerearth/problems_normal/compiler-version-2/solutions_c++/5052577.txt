#include <iostream>
using namespace std;

int main() {
	
	string str;
	while(getline(cin,str)){
		for(int i=0; i<str.length(); i++){
			if(str[i]=='/' && str[i+1]=='/')
				break;
			else if(str[i]=='-' && str[i+1]=='>') {
				str[i] = '.';
				str.erase(str.begin()+(i+1));
			}
		}
		cout << str << endl;
	}
	

	
	return 0;
}
