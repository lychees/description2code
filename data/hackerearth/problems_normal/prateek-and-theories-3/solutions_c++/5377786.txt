#include <stdio.h>
#include <stdlib.h>
 
int cmp(const void *a, const void *b)
{
	return (*(int *)a - *(int  *)b);
}
 
int main()
{
	int tst;
	scanf("%d",&tst);
	while(tst--)
	{
		int n, i, j, popul = 1, max = 1;
		scanf("%d",&n);
		int a[n];
		int d[n];
		
		for(i = 0; i < n; i++)
		{
			scanf("%d",&a[i]);
			scanf("%d",&d[i]);
		}
		
	qsort(a, n, sizeof(int), cmp);
	qsort(d, n, sizeof(int), cmp);
	
	i = 1;
	j = 0;
	while(i < n && j < n)
	{
		if(a[i] < d[j])
		{
			popul++;
			i++;
			if(popul > max)
			max = popul;
		}
		else
		{
			popul--;
			j++;
		}
	}
	
	printf("%d\n",max);
	}
    return 0;
}
