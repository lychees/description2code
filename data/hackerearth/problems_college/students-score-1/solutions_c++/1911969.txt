#include<iostream>
using namespace std;

long maxSubArraySum(long a[],int s)
{
    long msf=a[0];
    long i;
    long cm=a[0];
    for(i=1;i<s;i++){
        cm=max(a[i],cm+a[i]);
        msf=max(msf,cm);
    }
    return msf;
}



int main()
{
    long n;
    cin >> n;
    int i;
    long a[n];
    for(i=0;i<n;i++) cin >> a[i];
    cout << maxSubArraySum(a,n);
    return 0;


}
