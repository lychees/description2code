#include <stdio.h>
#include <string.h>
#include<iostream>
using namespace std;
void swap(char *a, char *b) {
    char temp;
    temp = *a;
    *a = *b;
    *b = temp;
}

void print(char *a, int i, int n) {
    int j;
    if(i == n) {
        printf("%s\n", a);
    } else {
        for(j = i; j <= n; j++) {
            swap(a + i, a + j);
            print(a, i + 1, n);
            swap(a + i, a + j);
        }
    }
}

int main(void) {
	int x,b,t;
    char a[100];
    gets(a);
    cin>>t;
    while(t--)
    {
    cin>>x;
    cin>>b;
    print(a,x,b);
    cout<<endl;
}
    return 0;
}
