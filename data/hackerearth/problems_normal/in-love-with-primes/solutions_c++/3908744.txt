#include<iostream>
#include<stdio.h>
#include<math.h>
using namespace std;
void prime(int x)
{
    int a[100000]={0};
    int i,j,count=0;
    for(i=2;i<sqrt(100000);i++)
        if(a[i]==0)
        for(j=2;j<100000;j++)
        {
            if(i*j>100000)
        break;
    a[i*j]=1;
    }

    for(i=2;i<x;i++)
    {
        if(a[i]==0)
        {if(a[x-i]==0)
            count=1;
        }

    }
    if(count==1)
        cout<<"Deepa\n";
    else
        cout<<"Arjit\n";
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        prime(n);
    }
    return 0;
}
