for t in range(input()):
 n=int(raw_input())
 sword=[]
 animals=[]
 for i in range(n):
  sword.append(int(raw_input()))
 for i in range(n):
  animals.append(int(raw_input()))
 sword.sort()
 animals.sort()
 flag=1
 for i in range(0,n):
  if animals[i]>sword[i]:
   flag=0
   break
 if flag==1:
  print "YES"
 else:
  print "NO"
