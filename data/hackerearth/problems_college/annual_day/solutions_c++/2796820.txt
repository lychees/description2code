typedef long long int lint;

int numberOfOnes(int A) {
	int c = 0;
	while(A) {
		c += A % 2;
		A >>= 1;	
	}
	return c;
}

int main() {
	int T;
	SC(T);
	while (T--) {
		int N, K;
		SC(N); SC(K);
		VI A;
		while (N--) {
			int val;
			SC(val);
			A.push_back(numberOfOnes(val));
		}
		sort(A.rbegin(), A.rend());
		int sum = 0;
		for (int i = 0; i < K; i++) {
			sum += A[i];
		}
		cout << sum << endl;
	}
	return 0;
}
