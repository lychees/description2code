#include <bits/stdc++.h>

using namespace std;

#define f first
#define s second 
#define mp make_pair

typedef long long ll;
typedef pair<int, int> pii;

const int mod = 1000000007;
const int maxn = 1001;
const int maxv = 10001;

int n, m, arr[maxn];
int dist[maxv];

int main()
{
	ll i, j, k;
	scanf("%d%d", &n, &m);
	for(i=1; i<=n; i++)
		scanf("%d", &arr[i]);
	for(i=1; i<maxv; i++)
		dist[i] = mod;
	sort(arr+1, arr+n+1);
	int mn = arr[1];
	priority_queue<pii, vector<pii>, greater<pii> > pq;
	pq.push(mp(0, 0));
	while(!pq.empty()){
		int val = pq.top().f;
		int mvs = pq.top().s;
		pq.pop();
		if(dist[val] < mvs)
			continue;
		for(i=1; i<=n; i++){
			int nval = (val + arr[i])%mn;
			int nmvs = mvs + arr[i];
			if(dist[nval] > nmvs){
				dist[nval] = nmvs;
				pq.push(mp(nval, nmvs));
			}
		}	
	}
	while(m--){
		int x;
		scanf("%d", &x);
		int val = dist[x%mn];
		if(val<=x)
			printf("YES\n");
		else
			printf("NO\n");
	}
	return 0;
}
