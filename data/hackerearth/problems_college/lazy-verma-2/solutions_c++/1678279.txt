#include <bits/stdc++.h>
using namespace std;

typedef long long ull;

int main()
{
    int n, m;
    ull sum = 0;
    int i;
    ull num;
    ull ans;
    ans = 0;
    scanf("%d%d", &n, &m);
    for (i = 0; i < n; i++) {
    	scanf("%lld", &num);
    	sum += num;
    }
    for (i = 0; i < m; i++) {
    	scanf("%lld", &num);
    	ans += (num * sum);
    }
    printf("%lld\n", ans);
    return 0;
}
