#include "bits/stdc++.h"
using namespace std;
int main()
{
	long int tc;
	cin>>tc;
	while(tc--)
	{
		long int n,k;
		cin>>n>>k;
		long int arr[n];
		for(long int i=0;i<n;i++)
			cin>>arr[i];
		
		for(long int i=0;i<n && k>0;i++)
		{
			long int minimum=i;
			for(long int j=i+1;j<min(n,i+k+1);j++)
			{
				if(arr[j]<arr[minimum])
				{
					minimum=j;
				}
			}
			if(minimum!=i)
			{
				long int ans=arr[minimum];
				for(long int p=minimum;p>i && p>=0;p--)
				{
					arr[p]=arr[p-1];
				}
				arr[i]=ans;
				k=k-(minimum-i);
			}
		}
		for(long int i=0;i<n;i++)
			cout<<arr[i]<<" ";
		cout<<endl;
	}
	return 0;
}
