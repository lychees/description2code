#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define mod 1000000007
ll arr[1000001];
void pre(){
	arr[0] = 0;
	arr[1] = 1;
	for(int i = 2;i<=1000000;i++){
		arr[i] = arr[i-1]+arr[i-2]+1;
		arr[i] = arr[i]%mod;
	}
}
int main()
{
	pre();
    int test;
    scanf("%d",&test);
    while(test--){
    	ll n;
    	scanf("%lld",&n);
    	printf("%lld\n",arr[n]);
    }
    return 0;
}
