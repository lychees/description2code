#include<iostream>
using namespace std;
int main()
{


char lookup[16],res[1000];int i,r,g,b,k=0,num,t,l=0;
    for(i=0;i<10;i++)
     {
     	lookup[i]=i+48;
     }
    for(i=10;i<16;i++)
       lookup[i]=85+i-10;
       lookup[16]='\0';
       cin>>t;
    for(l=1;l<=t;l++)
    {
     cin>>r>>g>>b;
     cout<<"Case "<<l<<": ";
       if(r<16)
         {
         cout<<"0"<<lookup[r]<<" ";
         }
       else
       {
       	 int  n1=r;
       	 
       	 while(n1!=0)
         {
       	  num=(n1%16);
       	  res[k++]=lookup[num];
       	  n1=n1/16;
         }
         for(i=k-1;i>=0;i--)
        cout<<res[i];
        cout<<" ";
       }
       
       k=0;
       
         if(g<16)
         {
         cout<<"0"<<lookup[g]<<" ";
         }
       else
       {
       	 int  n1=g;
       	 
       	 while(n1!=0)
         {
       	  num=(n1%16);
       	  res[k++]=lookup[num];
       	  n1=n1/16;
         }
         for(i=k-1;i>=0;i--)
        cout<<res[i];
        cout<<" ";
         k=0;
         
       }
      
       
         if(b<16)
         {
         cout<<"0"<<lookup[b]<<" ";
         }
       else
       {
       	 int  n1=b;
       	 
       	 while(n1!=0)
         {
       	  num=(n1%16);
       	  res[k++]=lookup[num];
       	  n1=n1/16;
         }
         for(i=k-1;i>=0;i--)
        cout<<res[i];
        cout<<" ";
         
       }
       k=0;
       cout<<endl;
    }
       
}
