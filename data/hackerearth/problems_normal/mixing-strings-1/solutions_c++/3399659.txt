string mix(string a, string b)
{

	loop(i,a.length())
	{

		Loop(j,i,a.length())
		{
		    if((j-i)>=b.length() || b[j-i]!=a[j]){flag=false;break;}
		}
		if(flag) break;
		c++;
	}
	Loop(j,a.length()-c,b.length()) a+=b[j];
	return a;
}
