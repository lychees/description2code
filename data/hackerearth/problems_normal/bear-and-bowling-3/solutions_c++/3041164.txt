#include <iostream>
using namespace std;
long long powr[200001];
int main(int argc, char const *argv[])
{
	ios_base::sync_with_stdio(false);
	long long mod = 1000000007;
	int n;
	cin>>n;

	powr[0]=1;
	for(int i=1;i<=n;i++)
	{
		powr[i]=powr[i-1]<<1;
		powr[i]%=mod;
	}
	
	long long arr[n+1];
	for(int i=1;i<=n;i++)
		cin>>arr[i];
	long long sum=arr[1]*powr[n-1];
	sum%=mod;
	for(int i=2;i<=n;i++)
	{
		long long p;
		p = (i+1)*powr[i-2];
		p%=mod;
		p*=arr[i];
		p%=mod;
		p*=powr[n-i];
		p%=mod;
		sum+=p;
		sum%=mod;
	}
	cout<<sum;
	return 0;
}
