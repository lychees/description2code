#include<bits/stdc++.h>
using namespace std;

int main()
{
	int n,m;
	cin >> n >> m;
	int x;
	priority_queue<int> pq;
	for(int i=0;i<n;i++)
	{
		cin >> x;
		pq.push(x);
	}
	int total=0,val;
	while(m--)
	{
		val=pq.top();
		pq.pop();
		total+=val;
		val--;
		pq.push(val);
	}
	cout << total << endl;
	return 0;
}
