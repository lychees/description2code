#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define mod 1000000007
#define m 1000007
set<ll> s;
map < ll,ll > mp;

set<ll> ::iterator it;
ll power(ll a,ll b)
{
    if(b==0)
        return 1;
    if(b==1)
        return a;
    ll temp=power(a,b/2);
    if(b%2==0)
        return (temp%mod*temp%mod)%mod;
    return (((a%mod*temp%mod)%mod)*temp%mod)%mod;
}

ll fun(ll n)
{
   if(mp.find(n)!=mp.end())
   {
       return mp[n]%mod;
   }
    s.erase(s.begin(),s.end());
    ll a=n-1;
    ll i;
    for(i=1;i*i<=a;i++)
    {
        if(a%i==0)
        {
            s.insert(i);
            if(i!=1)
        s.insert(a/i);
        }
    }
    vector<ll> v;
    v.clear();
    for(it=s.begin();it!=s.end();it++)
    {
        v.push_back(*it);
    }
    ll sum=0;
    ll x;
    for(i=0;i<v.size();i++)
    {
         x=fun(v[i])%mod;
        if(mp.find(v[i])==mp.end())
        {
            mp.insert(make_pair(v[i],x));
        }
        sum=(sum%mod+power(x,a/v[i])%mod)%mod;
    }
    mp.insert(make_pair(n,sum%mod));
    return sum%mod;


}
int main()
{
    ll n;
    scanf("%lld",&n);
    mp.insert(make_pair(1,1));
    mp.insert(make_pair(2,1));
   
    ll x=fun(n)%mod;
    cout<<x%mod<<endl;
}
