t = input()
for i in xrange(t):
    n, m, x = map(int, raw_input().split())
    total = sum(map(int, raw_input().split()))
    if total >= (n + 1) * x:
        print 1
        continue
    min_score = ((n + 1) * x) - total
    if min_score > m:
        print "Impossible"
    else:
        print min_score
