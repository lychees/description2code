#include <iostream>
using namespace std;

int main()
{
    int n,k,deg_raju[11],deg_rani[11],i,j,m;
    cin>>n>>k;
    for(i=0;i<n;i++)
    cin>>deg_raju[i];
    for(i=0;i<k;i++)
    cin>>deg_rani[i];
    for(i=0;i<k;i++)
    {
    	for(j=0;j<n;j++)
    	{
    		int temp=deg_raju[j];
    		for(m=0;m<10000;m++)
    		{
    			if(temp%360==deg_rani[i]){cout<<"YES"<<endl;break;}
    			temp+=deg_raju[j];
    		}
    		if(m!=10000) break;
    	}
    	if(j==n)cout<<"NO"<<endl;
    }
    return 0;
}
