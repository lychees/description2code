#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		if(n==0)
		{
			printf("0\n");
			continue;
		}
		vector <int> v(n,1);
		vector <int> :: iterator ip;
		int prev=0;
		while(1)
		{
			int cnt1=0;
			for(ip=v.begin();ip!=v.end();ip++)
			{
				if(*ip==0)
					continue;
				else if(*ip==1 && prev==1)
				{
					*ip=0;
					prev=0;
				}
				else
				{
					prev=1;
					cnt1++;
				}
			}
			if(cnt1==1)
				break;
		}
		int l=find(v.begin(),v.end(),1)-v.begin()+1;
		printf("%d\n",l);
	}
	return 0;	
}
