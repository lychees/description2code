#include <iostream>
#include<string>
using namespace std;

int main()
{
    int t,i,j,k,l,f1,f2;
    string s1,s2,s3,s4;
    cin >> t;
    while(t--)
    {
    	cin >> s1 >> s2;
    	s3=s1+s2;
    	s4=s2+s1;
    	f1=f2=1;
    	for(i=0;i<=s3.size()/2;i++)
    	{
    		if(s3[i]==s3[s3.size()-i-1])	continue;
    		else
    		{
    			f1=0;
    			break;
    		}
    	}
    	for(i=0;i<=s4.size()/2;i++)
    	{
    		if(s4[i]==s4[s4.size()-i-1])	continue;
    		else
    		{
    			f2=0;
    			break;
    		}
    	}
    	if(f1==1||f2==1)	cout << "YES" << endl;
    	else	cout << "NO" << endl;
    }
    return 0;
}
