#include<cstdio>
#include<algorithm>
#include<iostream>
#include<vector>
#include<cstring>
using namespace std;
long long int coin(long long int a[],long long int m,long int n)
{
	long long int tbl[n+1],i,j;
	memset(tbl,0,sizeof(tbl));
	tbl[0]=1;
	for(i=0;i<m;i++)
	{
		for(j = a[i];j<=n;j++)
		{
			tbl[j]=tbl[j] + tbl[j-a[i]];
		}
	}
	return tbl[n];
}
int main()
{
	long long int a[3];
	a[0] = 6;
	a[1] = 9;
	a[2] = 20;
	int m = sizeof(a)/sizeof(a[0]);
	
int t;
cin>>t;
    while(t--)
    {
	long long int n;
	cin>>n;
      long long int p= coin(a,m,n);
      if(p==0)
      {
      	cout<<"False"<<"\n";
	  }
	  else
	  {
	  	cout<<"True"<<"\n";
	  }
  }

}
