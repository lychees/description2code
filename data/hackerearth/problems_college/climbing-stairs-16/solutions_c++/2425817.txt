#include<bits/stdc++.h>
using namespace std;
long int S[1001];
long int f(long int N)
{
	if(N==1) return 1;
	if(N==2) return 2;
	if(S[N]!=0) return S[N];
	return S[N]=f(N-1)+f(N-2);
}
int main()
{
	long int T,N,i,j,k;
	cin>>T;
	for(i=1;i<=1000;i++)
	{
		j=f(i);
	}
	S[1]=1;
	S[2]=2;
	while(T--)
	{
		cin>>N;
		cout<<S[N]<<"\n";
	}
}
