#include <iostream>
#include<ctype.h>
using namespace std;

int main()
{
    int ctr = 0;
    string str;
    cin>>str;
    for(ctr = 0; ctr < str.length()-1; ctr++)
      {
            if (ctr == 0)
              str[ctr] = toupper(str[ctr]);
            if (str[ctr] == '_')
              str[ctr + 1] = toupper(str[ctr + 1]);
      }
    
    cout<<str;
    return 0;
}
