#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    string s;
    cin>>s;
    int q;
    cin>>q;
    vector<int> locs[26];
    for(int i=0;i<s.size();i++){
    	locs[s[i]-'a'].push_back(i);
    }
	while(q--){
		string f;
		cin>>f;
		int pos = 0;
		int i;
		for(i=0;i<f.size();i++){
			vector<int>::iterator itr = lower_bound(locs[f[i]-'a'].begin(),locs[f[i]-'a'].end(),pos);
			if(itr == locs[f[i]-'a'].end()) break;
			else pos = *itr + 1;
		}
		cout<<(i==f.size()?"Yes":"No")<<endl;
	}
    return 0;
}
