#include <bits/stdc++.h>

using namespace std;

int main()
{
    int i,t,n;
    cin>>t;
    while (t--)
    {
        int a[20];
        cin>>n;
        long long int c1=0,c2=0,c3=0,prod1=1,prod2=1,max1,min1;
        for(i=0;i<n;i++)
        {
            cin>>a[i];
            if (a[i]<0)
            {
                c1++;
                prod1*=a[i];
            }
            else if (a[i]==0)
            {
                c2++;
            }
            else
            {
                c3++;
                prod2*=a[i];
            }
        }
        sort (a,a+n);
        if (c3>0)
        {
            if (c1%2==0)
            {
                max1=prod2*prod1;
            }
            else
            {
                max1=(prod2)*(prod1/a[c1-1]);
            }
        }
        else
        {
            if (c1+c2==1)
            {
                max1=a[0];
            }
            else if ((c1==1 || c1==0) && c2>=1)
            {
                max1=0;
            }
            else if  (c1%2==0)
            {
                max1=prod1;
            }
            else
            {
                max1=prod1/a[c1-1];
            }
        }

        if (c1>0)
        {
            if (c1%2==1)
            {
                min1=prod1*prod2;
            }
            else
            {
                min1=(prod1/a[c1-1])*prod2;
            }
        }
        else
        {
            if (c2>0)
            {
                min1=0;
            }
            else
            {
                min1=a[0];
            }
        }

        //cout<<"\nMax is "<<max1;
        //cout<<"\tMin is "<<min1<<"\n";
        cout<<max1<<" "<<min1<<"\n";
    }
    return 0;
}
