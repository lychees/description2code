#include <bits/stdc++.h>
using namespace std;

int numsum(int x)
{
	int sum=0,y=x;
	while(y)
	{
		sum+=y%10;
		y/=10;
	}
	return sum+y%10;
}

int main()
{
	int arr[50]={0},n,i,l,b;
	// cout<<75/4;
	for(i=0;i<50;i++)
		arr[i]=(i+1)*(numsum(i+1));
    cin>>n;
    int a[n];
    long long ans=0;
    for(i=0;i<n;i++)
    	cin>>a[i];
    cin>>l>>b;
    int prod=l*b;
    for(i=0;i<n;i++)
    {
    	arr[a[i]-1]*=prod;
    	if(a[i]<10)
    		ans+=(3*arr[a[i]-1])/4;
    	else if(a[i]>=10&&a[i]<20)
    		ans+=(7*arr[a[i]-1])/10;
    	else
    		ans+=(4*arr[a[i]-1])/5;
    }
    cout<<ans;
    return 0;
}
