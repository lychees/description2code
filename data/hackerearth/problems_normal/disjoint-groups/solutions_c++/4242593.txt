#include <iostream>
using namespace std;

#define p 1000000007

long long modexp(long long x, long long n){
	long long ans = 1;
	while(n){
		if(n&1){
			ans = (ans*x)%p;
		}
		x = (x*x)%p;
		n >>= 1;
	}
	return ans;
}

int main()
{
    long long n,term,xorall = 0,ans;
    cin >> n;
    for(int i = 0; i < n; i++){
    	cin >> term;
    	xorall ^= term;
    }
    if(xorall != 0){
    	cout << 0 << endl;
    	return 0;
    }
    ans = modexp(2,n-1);
    if(ans == 0){
    	cout << p-1 << endl;
    	return 0;
    }
    cout << ans - 1 << endl;
    return 0;
}
