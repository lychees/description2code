#include <bits/stdc++.h>

using namespace std;

#define pb push_back
#define mp make_pair
#define ll long long int
#define s(x) scanf("%d", &x)
#define sl(x) scanf("%lld", &x)
#define mod 1000000000007LL
#define get getchar_unlocked

int main()
{
    int t, n, ans, i;
    cin >> t;
    while (t--) {
        cin >> n;
        ans = 0;
        for (i = 1; i <= n; ++i)
            ans += (__gcd(i, n) == 1);
        if (ans == 1) {
            cout << "Incorrect\n";
            continue;
        }
        for (i = 2; i < ans; ++i) {
            if (ans % i == 0) {
                cout << "Incorrect\n";
                goto l1;
            }
        }
        cout << "Correct\n";
        l1: ;
    }
    return 0;
}
