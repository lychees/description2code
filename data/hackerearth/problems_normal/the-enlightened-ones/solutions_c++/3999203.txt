#include<bits/stdc++.h>
using namespace std;
vector< int > tmp;
int check(int num, int k )
{
	int res=tmp[0]+num;
	k--;
	for (int i = 1; i < tmp.size(); i++)
	{
		if (res+num>=tmp[i])
		{
			continue;
		}
		if (k==0)
		{
			return 0;
		}
		res=tmp[i]+num;
		k--;	
	}
	return 1;
}
int main()

{
	
	int n,k;
	cin>>n>>k;
	tmp.resize(n);
	
	for (int i = 0; i < n; ++i)
	{
		cin>>tmp[i];
	}
	sort(tmp.begin(),tmp.end());
	int l=0,h=0,mid;
	h=100000000;	
	while(l<=h)
	{
		mid=(l+h)/2;
	

		int vv= check(mid,k);
		
		if( check(mid,k) && check(mid-1,k)==0)break;
		if (vv==0)
		{
			l=mid+1;
		}
		else
			h=mid-1;
	}
	cout<<mid<<endl;
	return 0 ;
}
