#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n;
    string s;
    cin>>t;
    while(t--) {
    	cin>>n>>s;
    	int freq[26];
    	memset(freq,0,sizeof(freq));
    	for(int i=0;i<s.length();++i)
    		freq[s[i]-'a']++;
    	//for(int i=0;i<26;++i)
    	//	cout<<freq[i]<<"\n";
    	int len[n+1],score[n+1],mx=-1,ans;
    	for(int i=0;i<n;++i) {
    		string t;
    		cin>>t;
    		len[i]=t.length();
    		score[i]=0;
    		for(int j=0;j<len[i];++j)
    			score[i]+=freq[t[j]-'a'];
    		//cout<<i<<" "<<t<<"  "<<score[i]<<"\n";
    		if(score[i]>mx) {
    			mx=score[i];
    			ans=i;
    		}
    		else if(score[i]==mx) {
    			if(len[i]<len[ans]) {
    				mx=score[i];
    				ans=i;
    			}
    		}
    	}
    	
    	cout<<(ans+1)<<"\n";
    }
    
    return 0;
}
