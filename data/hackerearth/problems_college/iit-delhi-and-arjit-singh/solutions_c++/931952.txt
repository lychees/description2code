#include <iostream>
using namespace std;
long long int a[1000009];
void fn()
{
    long long int i,j;
    for(i = 1 ;i <= 1000009 ; i++ )
     a[i]=i;
    for(i=2 ; i<= 1000009 ; i++ )
        if ( a[i] == i)
            for(j = 2*i ; j <= 1000009 ; j += i)
            	a[j] -=  a[j]/ i ;
    for(i=2 ; i<= 1000009 ; i++)
    {	if ( a[i]==i)
    		a[i] = i-1;
    	a[i]+=a[i-1];
    }
} 	

int main()
{
	long long int t,n,m;
	fn();
    cin>>t;
    while(t--)
     {
     	cin>>n>>m;
     	n=n/m;
     	cout<<a[n]<<endl; 
     }
    return 0;
}
