#include<iostream>
#include<string.h>
#include<string.h>
using namespace std;
int main()
{
	char str[1000];
	int num,t,count,den,i;
	cin>>t;
	while(t--)
	{	
		cin>>(str);
		count=0;
		den=strlen(str);
		for(i=0; str[i]!='\0'; i++)
		{
			if(str[i]=='a'||str[i]=='e'||str[i]=='i'||str[i]=='o'||str[i]=='u'||str[i]=='A'||str[i]=='E'||str[i]=='I'||str[i]=='O'||str[i]=='U')
			{count++; }
			
			
		}
		num=den-3-count;
		cout<<num<<"/"<<den<<endl;
	}
	return 0;
}
