#include<bits/stdc++.h>
using namespace std;
const int N=1050;
int n,m;
int dis[N];
vector<int>g[N];
void update(int v)
{
	for(int i=0;i<g[v].size();i++)
	{
		int t=g[v][i];
		if(dis[t]>dis[v]+1)
		{
			dis[t]=dis[v]+1;
			update(t);
		}
	}
}
int main()
{
	cin>>n>>m;
	for(int i=2;i<=n;i++)
	dis[i]=1e9;
	for(int i=1;i<=m;i++)
	{
		int type,l,r;
		cin>>type;
		if(type==1)
	   {
			cin>>l;
			if(dis[l]>1e8)
			cout<<-1<<endl;
			else
			cout<<dis[l]<<endl;
		}
		else
		{
			cin>>l>>r;
			g[l].push_back(r);
			if(dis[r]>dis[l]+1)
			{
				dis[r]=dis[l]+1;
				update(r);
			}
		}
	}
	return(0);
}
