#include<iostream>
using namespace std;
long long int powexponent(long long int a,long long int b, long long int m)
{
	int result = 1;
	while(b > 0){
		if(b%2 == 1)
		{
			result = (result*a)%m;
		}
		b = b >> 1;
		a = (a*a)%m;
	}
	return result;
}
int main()
{
	long long int t;
	long long int a,p;
	cin >> t;
	while(t--)
	{
		cin >> a >> p;
		if(a == 0) 
		cout << "YES" << endl;
		else
		{
			int pow = (p-1)/2;
			if(powexponent(a,pow,p) == 1)
			{
				cout << "YES" << endl;
			}
			else
			{
				cout << "NO" << endl;
			}
		}
	}
	return 0;
}
