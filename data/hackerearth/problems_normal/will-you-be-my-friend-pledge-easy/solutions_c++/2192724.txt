#include <iostream>
#include <cstring>

using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int mn;
    cin>>mn;
    int n;
    cin>>n;
    int arr[n];
    int fr[n];
    memset(fr, 0, sizeof(fr));
    
    for(int i=0;i<n;i++)
    {
    	cin>>arr[i];
    }
    
    int f = 0;
    for(int i=2;i<=mn;i++)
    {
    	for(int j=0;j<n;j++)
    	{
    		if(fr[j] == 0 && mn%i == 0 && arr[j]%i == 0)
    		{
    			f++;
    			fr[j] = 1;
    		}
    	}
    }
    
    cout<<f<<endl;
    return 0;
}
