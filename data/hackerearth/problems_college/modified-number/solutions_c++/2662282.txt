#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;
 
 
int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */   
    int p,q,len,flag=1;
    unsigned long long int num,ans,copy;
    cin>>p>>q;
    for(long i=p; i<=q; i++)
        {   len=0; 
            num= pow(i,2);
 
            copy= num;
            while(copy!=0)
                {
                    len++;
                    copy=copy/10;               
            }
            ans= num%(unsigned long long int)pow(10,(len + 1)/2) + (unsigned long long int)num/pow(10,(len+1)/2);
 
            if(ans==i)
            {   
            	cout<<i<<" ";
                flag=0;
            }
 
 
    }
    if(flag==1)
        cout<<"INVALID RANGE"<<endl;
 
    return 0;
}
