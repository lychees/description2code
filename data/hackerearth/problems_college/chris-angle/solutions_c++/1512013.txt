#include <iostream>
using namespace std;
int gcd(long a,long b)
{
	if(b==0)
		return a;
	else
		return gcd(b,a%b);
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	long long n,g,num,den=180;
    	scanf("%lld",&n);
    	num=n%360;
    	g=gcd(num,den);
    	num/=g;
    	den/=g;
    	if(n%360==0)
    		printf("0$\n");
    	else if(num==1&&den==1)
    		printf("$\n");
    	else if(num==1)
    		printf("$/%lld\n",den);
    	else if(den==1)
    		printf("$\n",num);
    	else
    		printf("%lld$/%lld\n",num,den);
    }
    return 0;
}
