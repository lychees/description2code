#include <bits/stdc++.h>
using namespace std;
#define dbg(x) { cout<<#x<<" : "<<x<<endl; }

int primes[1000007];

void seive_modified()
{
	for(int i=0;i<1000007;i++)
		primes[i]=0;
	
	primes[0]=primes[1]=primes[2]=0;
	for(int i=2;i*i<=1000007;i++) {
		if(primes[i]==0) {
			for(int j=i*i;j<1000007;j+=i)
			{
				if(primes[j]==0)
					primes[j]=i; //storing lowest prime factor.
			}
		}
		
	}
	
	for(int i=0;i<1000007;i++) {
		if(primes[i]==0)
			primes[i]=i;
	}
}
int main()
{
	ios::sync_with_stdio(false);
    int t;
    seive_modified();
    cin>>t;
    while(t--) {
    	int n;
    	cin>>n;
    	int n1=n;
    	set<int> v;
    	while(n>1) {
    		int f=primes[n];
    		v.insert(f);
    		n=n/f;
    	}
   // 	dbg(primes[362097]);
    //	dbg(primes[35]);
    	int ans=n1;
    	set<int> :: iterator it;
    	for(it=v.begin(); it!=v.end(); it++) {
    //		dbg(ans);
    		ans=ans- ans/(*it);
    //		dbg(*it);
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
