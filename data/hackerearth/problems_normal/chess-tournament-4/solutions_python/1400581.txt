n=int(raw_input())
#j=2**n-1
n=2**n-1
win=[x for x in range(1,n+2)]
row=[]
mat=[]
win1=[]

#input triangular matrix
while(n):
	row=[int(i) for i in raw_input().split()]
	mat.append(row)
	n=n-1


#append each round winners
while(True):
	if len(win)==1:
		break
	
	for i in range(0,len(win),2):
		p=win[i]
		q=win[i+1]
		r=max(p,q)
		s=min(p,q)
		aij=mat[r-2][s-1]
		if aij==0:
			win1.append(s)
		else:
			win1.append(r)
	win=win1[:]
	win1=[]
