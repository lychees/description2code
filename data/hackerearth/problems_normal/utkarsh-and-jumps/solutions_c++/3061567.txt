#include <iostream>
#include<bits/stdc++.h>
using namespace std;
#define ll long long

double dp[1000010];

int main()
{
    //cout << "Hello World!" << endl;
    int n,i;
    double p;
    cin>>n>>p;
    dp[0]=1;
    dp[1]=0;
    dp[2]=p/100.0;
    dp[3]=1-p/100.0;
    
    for(i=4;i<=n;i++)
    {
    	dp[i]=dp[2]*dp[i-2]+dp[3]*dp[i-3];
    }
    
    printf("%.6f",dp[n]);
    return 0;
}
