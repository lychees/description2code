bool prime[MAXN];
vi pn;
void sieve(){
  memset(prime,true,sizeof(prime));
  pn.clear();
  prime[0]=prime[1]=false;
  for(int i=2;i*i<MAXN;i++)
  if(prime[i]){
    for(int j=i*i;j<MAXN;j+=i)prime[j]=false;
  }
  for(int i=2;i<MAXN;i++){
    if(prime[i])pn.pb(i);
  }
}
bool ispower(ll n, ll p){
  while(n>=p && n%p==0)n/=p;
  return n==1;
}
bool isprimeorpow(ll n){
  for(size_t i=0;i<pn.size() && (ll)pn[i]*pn[i]<=n;i++){
    if(n%pn[i]==0)return ispower(n,pn[i]);
  }
  ll d=1000002;
  while(d*d<=n){
    if(n%(d-1)==0)return ispower(n,d-1);
    if(n%(d+1)==0)return ispower(n,d+1);
    d+=6;
  }
  return true;
}


ll next(ll a){
  while(!isprimeorpow(a)){a++;}
  return a;
}
