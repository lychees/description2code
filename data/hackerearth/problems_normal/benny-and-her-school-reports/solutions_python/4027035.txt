#print 'Hello World!'
testcases = raw_input()

for i in range(int(testcases)):
	nmx = raw_input()
	nmxlist = [int(j) for j in nmx.split()]
	
	n = nmxlist[0]
	m = nmxlist[1]
	x = nmxlist[2]
	
	nmarksstring = raw_input()
	
	nmarks = [int(k) for k in nmarksstring.split()]
	
	marksrequired = (n+1)*x-sum(nmarks)
	
	if marksrequired > m:
		print ("Impossible")
	elif marksrequired < 1:
		print (1)
	else:
		print (marksrequired)
