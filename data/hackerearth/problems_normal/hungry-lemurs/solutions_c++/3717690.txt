#include<bits/stdc++.h>
using namespace std;
int main(){
	int n,k;
	cin>>n>>k;
	int ans=n%k;
	int c=0;
	int k1=k;
	while(k<n){
		ans=min(ans,n%k+c);
		if(n%k!=0){
			ans=min(ans,(k-n%k)+c);
		}
		c++;
		k++;
	}
	c=0;
	k=k1;
	while(k>0){
		ans=min(ans,n%k+c);
		if(n%k!=0){
			ans=min(ans,(k-n%k)+c);
		}
		k--;c++;
	}
	printf("%d\n",ans);
	return 0;
}
