#include <bits/stdc++.h>

using namespace std;

const int N = 2e5 + 10;

int a[N];

int main() {
	int t;
	scanf("%d", &t);
	for (int tt = 1; tt <= t; tt++) {
		int n;
		scanf("%d", &n);
		n *= 2;
		for (int i = 0; i < n; i++) {
			scanf("%d", &a[i]);
		}
		sort(a, a + n);
		int l = 0, h = n - 1, res = 0;
		while (l <= h) {
			res = max(res, a[l++] + a[h--]);
		}
		printf("%d\n", res);
	}
	return 0;
}
