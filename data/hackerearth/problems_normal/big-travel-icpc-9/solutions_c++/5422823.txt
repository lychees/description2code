#include <bits/stdc++.h>
#define bs 1000000007
using namespace std;
typedef long long ll;
ll solve(vector<ll> &v)
{
	int n=v.size();
	vector<ll> dp(n);
	sort(v.begin(),v.end());
	ll res=0;
	dp[0]=0;
	for (int i=1;i<n;i++)
	{
		dp[i]=(dp[i-1]+((v[i]-v[i-1])*i)%bs)%bs;
		res=(res+dp[i])%bs;
		
	}
	return res%bs;
}
int main()
{
	ll n,a,b;
	cin>>n;
	vector<ll> v1,v2;
 for (int i=0;i<n;i++)
{
	cin>>a>>b;
	v1.push_back(a);
	v2.push_back(b);
	
}
cout<<(solve(v1)+solve(v2))%bs<<endl;

    return 0;
}
