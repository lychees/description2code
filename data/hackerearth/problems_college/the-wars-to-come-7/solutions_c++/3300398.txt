int main()
{
	ios_base::sync_with_stdio(false);
	
	int n,i,minstr,cnt,foo;
	cin>>n;
	vector<int> arr(n);
	for(i=0;i<n;i++) {
		cin>>arr[i];
	}

	cin>>minstr;
	sort(arr.begin(),arr.end());
	foo=cnt=0;
	for(i=n-1;i>=0;i--) {
		cnt+=arr[i];
		foo++;
		if(cnt>=minstr)
			break;
	}

	cout<<foo<<"\n";
	return 0;
}
