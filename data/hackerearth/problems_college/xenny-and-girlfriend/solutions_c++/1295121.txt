#include<bits/stdc++.h>
using namespace std;
int main(){
int t,n,m,i,a,b;
cin>>t;
while(t--){
	scanf("%d%d",&n,&m);
	int deg[10005]={0};
	while(m--){
		scanf("%d%d",&a,&b);
		deg[a]++;
		deg[b]++;
	}
	int cnt=0;
	for(i=1;i<=n;i++){
		if(deg[i]&1)
		cnt++;
	}
	if((cnt==0)||(cnt==2))
	printf("Yes\n");
	else
	printf("No\n");
}
return 0;	
}
