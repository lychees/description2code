#include <iostream>
#include <vector>

using namespace std;

#define maxN 4000000

typedef long long int ll;

vector <bool> sieve;
vector <int> primes;

void ESieve()
{
	sieve.resize(maxN, true);
	sieve[0] = sieve[1] = false;

	for (ll i = 2; i <= maxN; i++)
	{
		if (sieve[i])
		{
			for (ll j = i*i; j <= maxN; j += i)
			{
				sieve[j] = false;
			}
			primes.push_back((int)i);
		}
	}
}

int main()
{
	long long int N, sum = 0;

	cin >> N;

	ESieve();

	for (int i = 0; i < primes.size() && primes[i] < N; i++)
		sum += primes[i];

	cout << sum << endl;
	
	return 0;

}
