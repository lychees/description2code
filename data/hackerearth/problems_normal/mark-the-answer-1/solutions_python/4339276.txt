import numpy as np

numbers = np.array(map(int, raw_input().split()))
n = int(numbers[0])
level = numbers[1]

questions = np.array(map(int, raw_input().split()))
flag = 0
count = 0

for i in range(n):
	if (questions[i]>level and flag == 0):
		flag = 1
	elif (questions[i]>level and flag == 1):
		break
	else:
		count += 1
		
print count
