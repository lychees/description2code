#include <bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;
ll int a1[1000001],a0[1000001];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	a1[1]=0;a0[1]=1;
	int i;
	for(i=2;i<1000001;i++)
	{
		a1[i]=a0[i-1];
		a0[i]=(a0[i-1]+a1[i-1])%mod;
	}
	ll int test;
	cin>>test;
	while(test--)
	{
		ll int n;
		cin>>n;
		cout<<(a0[n]+a1[n])%mod<<endl;
	}
	return 0;
}
