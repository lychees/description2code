int  comp(const void *a,const void *b)
{ 
     return *(int*)a-*(int*)b;
}
 
void si(llu &x)
{  x=0;
     char c=gc();
     int neg=0;
 
     while((c<'0'|| c>'9') && c!='-')
          c=gc();
 
     if(c=='-') {neg=1; c=gc();}
     for(;c>='0' && c<='9';c=gc())
     { 
          x=x*10+c-48;            
     }
 
     if(neg==1)
          x=-x;
}

     while(n>0)
     {  ch[i]=n%10 +'0';
          n=n/10;
          i++;
     }
 
     while(i>0)
     { 
          pc(ch[i-1]);

     }

}
 
inline void ss(char *s)
{
     char ch;
     ch=gc();
     int i=0;
     while((ch>=32)&&(ch<=110))
     {
          s[i++]=ch;
          ch=gc();
     }
 
     s[i]='\0';
}
 
inline void ps(char *s)
{
     int i=0;
     while(s[i]>=32 && s[i]<=127)
     {
          pc(s[i]);
          i++;
     }
     pc('\n');
}
llu nextpow(llu v)
{
     v--;
     v |= v >> 1;
     v |= v >> 2;
     v |= v >> 4;
     v |= v >> 8;
     v |= v >> 16;
     v++;
 
     return v/2;
 
}
