#include <iostream>
using namespace std;

void printFull(int a){
	for (int i = 0; i < a; i++)
		cout << "*";
	cout << endl;
}

void printPartial(int a){
	for (int i = 0; i < a - 1; i++)
		cout << " ";
	cout << "*" << endl;
}

int main(){
	int tC;
	cin >> tC;
	for (int i = tC; i > 0; i--){
		if (i == tC || i == 1)
			printFull(tC);
		else
			printPartial(i);
	}
}
