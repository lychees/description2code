#include <bits/stdc++.h>

using namespace std;

#define maxVal (int)1000000000
#define ll long long int

map<int,int> m;

int bfs(int s)
{
	int u;
	queue<int> q;
	q.push(s);
	while(!q.empty())
	{
		u=q.front();
		if(u==1)
			break;
		q.pop();
		if(u%2==0)
		{
			if(m[u/2]==0)
			{
				m[u/2]=m[u]+1;
				q.push(u/2);
			}
		}
		if(u%3==0)
		{
			if(m[u/3]==0)
			{
				m[u/3]=m[u]+1;
				q.push(u/3);
			}
		}
		if(m[u-1]==0)
		{
			m[u-1]=m[u]+1;
			q.push(u-1);
		}
	}
	return m[1];
}

int main()
{
    int t,n;
    scanf("%d",&t);
    while(t--)
    {
    	m.clear();	
    	scanf("%d",&n);
    	printf("%d\n",bfs(n));
    }
    return 0;
}
