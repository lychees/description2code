#include <iostream>
using namespace std;

int main()
{
	int n,i,j,k,a[30000],p,count;
	bool flag1,flag2;
	flag1=flag2=false;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    cin>>p;
    for(i=0;i<n;i++)
    {
      for(j=0;j<n;j++)
      {
      	for(k=0;k<n;k++)
      	{
      		if((a[i]+a[j]+a[k])==p)
      		flag1=true;
      		else
      		flag2=false;
      	
      		
      	}
      }
    }
    if(flag1==true)
    cout<<"True";
    else
    cout<<"False";
    return 0;
}
