#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    while(t--){
    	int n,count = 0;
    	cin>>n;
    	int esum = 0,osum = 0;
    	while(n != 1){
    		
    		if(n % 2 == 0)
    		{	
    			
    			esum += n;
    			n = n/2;
    			
    		}
    		else{
    			osum += n;
    			n = 3*n + 1;
    		}
    		count++;
    	}
    	osum++;
    	esum = esum % count;
    	osum = osum % count;
    	
    	if(esum > osum){
    		cout<<"Even Rules"<<endl;
    	}
    	else if(esum == osum){
    		cout<<"Tie"<<endl;
    	}
    	else
    	cout<<"Odd Rules"<<endl;
    	
    }
}
