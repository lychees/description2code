#include <bits/stdc++.h>
using namespace std;

int main()
{

long long int n,k,a[100000],i,j,temp,max,c=0,u;
cin>>n>>k;
for(i=0;i<n;i++)
cin>>a[i];
if(k>=n)
{
	for(i=n;i>0;--i)
	cout<<i<<" ";
}
else
{
	for(i=0;i<n-1;i++)
{
max=a[i];
for(j=i+1;j<n;j++)
{
if(a[j]>max)
{
max=a[j];
u=j;
}
}

if(a[i]!=max)
{
temp=a[i];
a[i]=a[u];
a[u]=temp;
c++;
}
if(c==k)
break;
}
for(i=0;i<n;i++)
cout<<a[i]<<" ";
}
return 0;
}
