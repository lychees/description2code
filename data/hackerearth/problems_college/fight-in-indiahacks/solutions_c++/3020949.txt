#include<iostream>
#include<cmath>
//#include<functional>
int min(int a,int b)
{
	if(a<=b)
	return a;
	else
	return b;
}
using namespace std;
int main()
{
	
	
	int t;
	cin>>t;
	int ans[t],m,n1,n2,less,more;
	for(int i=0;i<t;i++)
	{
		cin>>m>>n1>>n2;
		int arr[n1+n2+1];
		less=min(n1,n2);
		more=n1+n2-less;
		
		if(n1!=0&&n2!=0)
		{
		
			//int arr[n1+n2+1];
			for(int j=0;j<n1+n2+1;j++)
			{
				if(j==0)
				arr[j]=0;
				else
				{
					if(j<less||j==n1+n2)
					arr[j]=0;
					else if(j==n1||j==n2)
					arr[j]=1;
					else if(j>less&&j<more)
					{
						arr[j]=!arr[j-less];
					}
					else if(j>more&&j<n1+n2)
					{
						arr[j]=!(arr[j-less]&arr[j-more]);
					}
				}
			}
	    }
//		for(int x=0;x<n1+n2+1;x++)
//		{
//			cout<<arr[x]<<"  ..  "<<endl;
//		}
		if(n1!=0&&n2!=0)
		ans[i]=arr[m%(n1+n2)];
		else
		ans[i]=1;
	}
	for(int i=0;i<t;i++)
	{
		cout<<ans[i]<<endl;
	}
}
