{

	while(t--)
	{

		long long arr[n];
		for(i=0;i<n;i++)
		scanf("%lld",&arr[i]);
		sort(arr,arr+n);
        long long count=0,sum=0;
     for(i=0;i<n;i++)
      {
      	sum=sum+arr[i];
        if(sum<=m)
         count++; 
        else 
         break; 
      }
		printf("%lld\n",count);
		count=0;
		for(i=0;i<n;i++)
		arr[i]=0;
