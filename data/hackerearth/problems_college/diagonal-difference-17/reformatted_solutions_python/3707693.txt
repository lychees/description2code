#!/bin/python

import sys


n = int(raw_input().strip())
a = []
for a_i in xrange(n):
    a_temp = map(int, raw_input().strip().split(' '))
    a.append(a_temp)
sump = 0
sums = 0
for i in range(n):
    sump += a[i][i]
for i in range(n):
    sums += a[i][n-i-1]
if sums > sump:
    print sums-sump
else:
    print sump-sums
