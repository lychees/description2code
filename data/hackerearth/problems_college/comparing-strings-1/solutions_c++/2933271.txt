    #include <iostream>
    #include <string>
    #include <algorithm>
    #include <string>
    #include <cstring>
     
    using namespace std;
     
    bool check(char a[],char b[]);
     
    int main()
    {
        int t;
        cin>>t;
        char a[100000],b[100000];
        while(t--)
        {
            cin>>a>>b;
            int len_a = strlen(a);
            int len_b = strlen(b);
            sort(a,a+len_a);
            sort(b,b+len_b);
            if(check(a,b))
            {
                cout<<"YES"<<endl;
            }
            else{
                cout<<"NO"<<endl;
            }
        }
        return 0;
     
    }
     
    bool check(char a[],char b[])
    {
        for(int i=0;i<strlen(a);i++)
        {
            for(int j=0;j<strlen(b);j++)
            {
                if(a[i] == b[j])
                {
                    return true;
                }
                else if( int(b[j]) > int(a[i]) )
                {
                    break;
                }
            }
        }
        return false;
    }
