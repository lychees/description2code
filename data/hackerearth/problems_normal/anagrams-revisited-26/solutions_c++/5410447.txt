#include <iostream>
#include <map>
#include <string>
#include <algorithm>

using namespace std;

int main()
{
    map<string,int> student;
    string name;
    int n,l;
    cin>>n;
    map<string,int>::iterator x;
    l=-1;
    while(n--)
    {
        ws(cin);
        getline(cin,name);
        sort(name.begin(),name.end());
        x=student.find(name);
        if(x==student.end())
            student[name]=1;
        else
            x->second++;
    }
    for(map<string,int>::iterator i=student.begin();   i!=student.end();   i++)
        if(l<i->second)
            l=i->second;
    cout<<l;
    return 0;
}
