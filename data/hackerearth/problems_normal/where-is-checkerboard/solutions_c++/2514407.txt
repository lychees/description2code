#include <iostream>
using namespace std;
#include<bits/stdc++.h>
int main()
{
    
	string s1="_#_#_#_#_#_#_#_#_#_#_#_#_#_#_#_#";
	string s2="#_#_#_#_#_#_#_#_#_#_#_#_#_#_#_#_";
    int N,M;
    cin>>N>>M;
    vector<string> V;
    int i,j;
    for(i=0;i<N;i++)
    {
    	string s;
    	for(j=0;j<M;j++)
    	{
    		char c;
    		cin>>c;
    		s.push_back(c);
    	}
    	V.push_back(s);
    }
    int ans=0;
    for(i=0;i<=M-32;i++)
    {

    	int row=0;
    	int k=0;
    	while(row<=N-32)
    	{

    		if(V[k].substr(i,32)==s1)
    		{
    		  k++;
    		}
    		else
    		{
    			if(k==row)
                {
                   row++;
                   k++;
                }
    			else
    			{
    				row=k+1;
    				k++;
    			}
    			continue;
    		}
    		if(k<N)
    		{
    			if(V[k].substr(i,32)==s2)
    			{
    				k++;
    				if(k-row==32)
    				{
    					ans++;
    					row+=2;
    				}
    			}
    			else
    			{
    				row=k;
    			}
    		}
    	}
    }
    cout<<ans;
    return 0;
}
