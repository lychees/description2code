#include <bits/stdc++.h>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    int T;
    cin>>T;
    cin.ignore();
    while(T--)
    {
    	string s1,s2;
    	getline(cin,s1);
    	
    	getline(cin,s2);
    //	cout<<"string 1:"<<s1<<endl;
    //	cout<<"string 2:"<<s2<<endl;
    	
    	unordered_multiset<char> um1,um2;
    	for(auto i : s1)
    	{
    		if(isspace(i))
    		continue;
    		um1.insert(i);
    	}
    	
    	for(auto i : s2)
    	{
    		if(isspace(i))
    		continue;
    		auto itr = um1.find(i);
    		if(itr!=um1.end())
    		um1.erase(itr);
    		else
    		um2.insert(i);
    	}
    	
    	int l1=um1.size();
    	int l2=um2.size();
    	
    	if(l1>0 && l2>0)
    	cout<<"You draw some.\n";
    	else
    	{
    		if(l1>0 && l2==0)
    		cout<<"You win some.\n";
    		else
    		cout<<"You lose some.\n";
    	}
    	
    }
    
    return 0;
}
