#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;

long long int arr[100003];

int main()
{
	int n;
	cin>>n;
	for(int i=0;i<n;i++)cin>>arr[i];
	sort(arr,arr+n);
	long long int a,b,c,d;
	a = arr[n-1];
	b = arr[n-2];
	a = a * b;
	c = arr[0];
	d = arr[1];
	c = c * d;
	a = max(a,c);
	cout<<a;
	return 0; 
}
