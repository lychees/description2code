#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int ct=0;
		for(int i=1;i<=sqrt(n);i++)
    	{
        	for (int j=1;j<i;j++)
        	{
        		if(__gcd(i,j)==1&&(i*i + j*j)<=n&&((i-j)%2==1))
        		{
        			int k=1;
        			while(k*(i*i + j*j)<=n) ct++,k++;
        		}
        	}
    	}
        cout<<ct<<endl;
    
    }
    return 0;
}
