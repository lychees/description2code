#include <bits/stdc++.h>
#define ll long long int
#define SZ 123456
using namespace std;
int main()
{
	ios::sync_with_stdio(false);
	
	ll n,l,r,q,arr[SZ];
	cin>>n>>l>>r>>q;
	for(ll i=0; i<n; i++)
	{
		cin>>arr[i];
	}
	
	for(ll i=0; i<q; i++)
	{
		ll tmp,fl=1;
		cin>>tmp;
		sort(arr+l, arr+r+1);
		for(ll j=l; j<=r; j++)
		{
			if(arr[j]>tmp)
			{
				cout<<arr[j]<<endl;
				fl=0;
				break;
			}
		}
		if(fl)
		cout<<"-1"<<endl;
	}
	
	return 0;
}
