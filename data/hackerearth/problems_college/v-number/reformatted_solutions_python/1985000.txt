def sum_digits3(n):
    r = 0
    while n:
        r, n = r + n % 10, n / 10
    return r


T = input()
for i in range(1, T+1):
    N = raw_input()
    firstpart, secondpart = N[:len(N)/2], N[len(N)/2:]
    sum1 = sum_digits3(int(firstpart))
    sum2 = sum_digits3(int(secondpart))
    if sum1 == sum2:
        print 'YES'
    else:
        print 'NO'
