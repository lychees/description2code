#include <iostream>
using namespace std;
int gcd ( long long int a, long long int b )
{
 long long  int c;
  while ( a != 0 ) {
     c = a; a = b%a;  b = c;
  }
  return b;
}
int main()
{
  long long int T,a,b;
    cin>>T;
    for(int i=0;i<T;i++)
    {
    cin>>a>>b;
    long long int r=gcd(a,b);
    
    if(r==1)
     cout<<"Is a Co-Prime\n";
     else
     cout<<"Not a Co-Prime\n";
}
}
