#include <iostream>
using namespace std;
long long  rangeBitwiseAnd(long long  m, long long  n) {
    long long  x = 0;
    while ((m|x) < n)
       x = (x<<1) | 1;
    return m & ~x;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    long long  a,b;
    cin>>a>>b;
    cout << rangeBitwiseAnd(a,b) <<endl;
    }
}
