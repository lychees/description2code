#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int a[100010],dp[100010];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int test;
	cin>>test;
	while(test--)
	{
		ll int n;
		cin>>n;
		ll int i;
		for(i=0;i<n;i++)
		cin>>a[i];
		//prefix sum
		dp[0]=a[0];
		for(i=1;i<n;i++)
		{
			dp[i]=dp[i-1]+a[i];
		}
		ll int ans=INT_MAX;
		for(i=0;i<n-1;i++)
		{
			ans=min(ans,abs(dp[i]-(dp[n-1]-dp[i])));
		}
		cout<<ans<<endl;
	}
	return 0;
}
