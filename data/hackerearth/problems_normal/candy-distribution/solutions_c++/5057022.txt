#include <bits/stdc++.h>
#define mod 1000000007
using namespace std;

map <long long int,long long int> m;

long long int modPow(long long int a,long long int n)
{
    if(n==0)
        return 1;
    long long int ret = modPow(a,n>>1);
    ret = (ret*ret)%mod;
    if((n&1)==0)
        return ret;
    ret = (a*ret)%mod;
    return ret;
}

long long int solve(long long int n)
{
    if(n==1||n==2||n==3)
        return 1;
    if(m.find(n)!=m.end())
        return m[n];
    m[n] = 1;
    long int i;
    for(i=2;i*i<n-1;i++)
    {
        if((n-1)%i==0)
        {
            long long int t = solve(i);
            m[n] += modPow(t,(n-1)/i);
            m[n] %= mod;
            t = solve((n-1)/i);
            m[n] += modPow(t,i);
            m[n] %= mod;
        }
    }
    if(i*i==n-1)
    {
        long long int t = solve(i);
        m[n] += modPow(t,(n-1)/i);
        m[n] %= mod;
    }
    return m[n];
}

int main()
{
    m[1] = 1;
    m[2] = 1;
    m[3] = 1;
    long long int n;
    cin>>n;
    cout<<solve(n)<<endl;
    return 0;
}
