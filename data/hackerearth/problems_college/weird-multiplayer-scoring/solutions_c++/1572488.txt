#include<iostream>
#include<map>
#include<string>
#include<cstdio>
#include<cstdlib>
#include<cstring>
using namespace std;
string s;
long  int mx=-999999999999999;
int main()
{
   
        int n,mi;cin>>n>>mi;
        map < string ,  long int > m;
        string name;
        fflush(stdin);
        while(mi--)
        {
        	int x;
            cin>>s>>x;
            
            //cout<<s<<"   "<<x<<endl;
            if(x>0)
            	m[(string)s]+=x;
       		if(m[(string)s]>mx)
       		{
       			mx=m[(string)s];
       			name=s;	
       		}
       		if(x<=0)
       			m[(string)s]+=x;
       	//	cout<<mx<<"   "<<name<<endl;
        }
        cout<<name;
    
    return 0;
}
