#include <iostream>
using namespace std;

int main()
{
      int t;

   long int a, b, c, i;
    long long int next;

  cin >> t;

  while(t--)
  {
    cin >> a >> b >> c;

    if( c == 1)
     next = b;

   else if( c == 0)
    next = a;

   else
   {
     for( i = 0; i <= c-2; i++)
     {
       next = a + b;
       a = b;
       b = next;

     }
    }

     cout << next << endl;
  }
    return 0;
}
