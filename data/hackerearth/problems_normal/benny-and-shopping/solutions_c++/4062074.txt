#include <iostream>
using namespace std;

int main()
{
    int t;
    scanf("%d", &t);
    string s,k;
    int n;
    while(t--)
    {
    	while(getline(cin, s))
    	{
    		for(int i=0;i<s.length();i++)
    		{
	    		if(s[i]=='$')
	    		{
	    			k = '$';
	    			i++;
	    			while(s[i]==' ' ||(s[i]>='0' && s[i]<='9') )
	    			{
	    				if(s[i]!=' ')
	    				{
	    					if(s[i]=='0')
	    					{
	    						if(k[k.length()-1]!='$')
	    						{
	    							k += s[i];
	    						}
	    						
	    					}
	    					else
	    					{
	    						k += s[i];
	    					}
	    				}
	    					
	    				i++;
	    			}
	    			break;
	    		}
    		}
    		if(k.length()==1)
    		{
    			cout<<"$0\n";
    		}
    		else
    		{
    			cout<<k<<"\n";
    		}
    		
    	}
    	
    }
    return 0;
}
