from collections import Counter

noT = int(raw_input())
for x in range(noT):
    ins = str(raw_input())
    if ins == ins[::-1]:
        print "0"
        continue
    cset = Counter(ins)
    needed = 0
    for y in cset:
        if cset[y]%2==0:
            continue
        else:
            needed += 1
    print( '0' if needed in [0,1] else needed-1)
