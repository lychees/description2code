#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    long long int t;
    cin>>t;
    while(t--){
    	long long int x,divisors=0;
    	cin>>x;
    	for(long long int i=1;i<sqrt(x);i++){
    		if(x%i==0)
    		  divisors+=2;
    		
    		if(divisors==4){
    		
    			break;
    		}  
    	}
    	
    	if(divisors>=4)
    	  cout<<"YES"<<endl;
    	  else
    	  cout<<"NO"<<endl;
    }
    return 0;
}
