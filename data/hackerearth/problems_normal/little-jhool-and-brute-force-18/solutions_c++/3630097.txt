#include <bits/stdc++.h>
using namespace std;
set < long int > cubes;
vector <long int> s;
set <long int> :: iterator it;

bool isCube(long int x)
{
   return cubes.end()!=cubes.find(x);
}

long int lbs(int lo,int hi,long int key)
{
    int mid = lo + (hi-lo)/2;
    if(mid!=lo)
    {
       if(s[mid]>key)
        return lbs(lo,mid-1,key);
       return lbs(mid,hi,key);
    }
    if(s[hi]<=key)
        return s[hi];
    if(s[lo]<=key)
        return s[lo];
    return -1;
}
int main()
{
    int t;

    for(int i=1;i<=90;i++)
        cubes.insert(i*i*i);

    for(int i=1;i<=90;i++)
        for(int j=1;j<=90;j++)
        {
            long int x = i*i*i + j*j*j;
            for(int k=1;k*k*k<=x;k++)
            {
              if(k==i||k==j) continue;
              if(isCube(x-k*k*k))
                s.push_back(x);
            }
        }
    sort(s.begin(),s.end());
    cin>>t;
    while(t--)
    {
        long int n;
        cin>>n;
        cout<<lbs(0,s.size()-1,n)<<endl;
    }
    return 0;
}
