import itertools

k,t=map(int,raw_input().split())

p=map(int,raw_input().split())

while t>0:
  a,b=map(int,raw_input().split())
  lo=[]
  le=[]
  for i in xrange(1,k+1):
      for j in itertools.combinations(p,i):
          div=1
          for ml in j:
              div=div*ml
          if i%2==1:
              lo.append(b/div-(a-1)/div)
          else:
              le.append(b/div-(a-1)/div)
  print sum(lo)-sum(le)
  t=t-1
