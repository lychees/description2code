#include <iostream>
using namespace std;
int gcd(int a, int b)
{
	if(b==0) return a;
	else return (gcd(b,a%b));
}
int main()
{
    int i,j,a,n,count=0;
    cin>>a;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>j;
    	if(a>j)
    	{
    		if(gcd(a,j)>1) count++;
    	}
    	else
    	if(gcd(j,a)>1) count++;
    }
    cout<<count;
}
