using namespace std;
typedef long long LL;
typedef pair<int,int> pii;
typedef pair<LL,LL> pLL;
typedef vector<int> vi;
typedef vector<vi> vvi;
typedef vector<LL> vLL;
typedef vector<vLL> vvLL;
typedef long double ld;
int main(){
    ios::sync_with_stdio(false);
    //freopen("xyz.txt", "r", stdin);
    LL t,i,j,ii,temp,n,k;
    cin>>t;
    while(t--){
        cin>>n>>k;
        vector<string> s(n);
        for(i=0;i<n;i++){
            cin>>s[i];
        }
        /*if(n==1){
            cout<<s[0]<<endl;
            continue;
        }*/
        vvLL cnt(k, vLL(26,0));
        for(i=0;i<n;i++){
            for(j=0;j<k;j++){
                cnt[j][s[i][j]-'a']++;
            }
        }
        vLL minch(k,-1),minv(k,(LL)1e12);
        for(i=0;i<k;i++){
            for(j=0;j<26;j++){
                //ch='a'+j;
                temp=0;
                for(ii=0;ii<26;ii++){
                    temp += abs(j-ii)*cnt[i][ii];
                }
                if ( temp < minv[i] ) {
                    minv[i] = temp ;
                    minch[i] = j ;
                }
            }
        }
        for(i=0;i<k;i++){
            cout<<char(minch[i]+'a');
        }
        cout<<endl;

    }



    return 0;
}
