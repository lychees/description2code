#include <iostream>
using namespace std;
int a[10005];
long long int dp[10005];
int main()
{
    int t,n;
    cin >> t;
    while(t--)
    {
    	cin >> n;
    	for(int i=0;i<n;++i) cin >> a[i];
    	
    	dp[0] = a[0];
    	dp[1] = max(a[0],a[1]);
    	for(int i = 2; i < n ; ++i)
    	{
    		dp[i] = max(a[i] + dp[i-2],dp[i-1]);
    	}
    	cout << dp[n-1] << endl;
    }
    return 0;
}
