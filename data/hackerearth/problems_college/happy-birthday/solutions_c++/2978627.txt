#include<iostream>
using namespace std;
int main()
{
	int n;
	cin>>n;
	int i;
	int friends[100000];
	for(i=0;i<n;i++)
	cin>>friends[i];
	int one=0,two=0,three=0,four=0;
	int count=0;
    for(i=0;i<n;i++)
	{
	 	if(friends[i]==1)
	 	one++;
	 	else if(friends[i]==2)
	 	two++;
	 	else if(friends[i]==3)
	 	three++;
	 	else
	 	four++;
    }
    count=four;
    if(three<=one)
    {
    	count=count+three;
    	one=one-three;
    	three=0;
    }
    else
    {
    	count=count+one;
    	three=three-one;
    	one=0;
    }
    if(two%2==0)
    {
    	count=count+(two/2);
    	two=0;
    }
    else
    {
    	count=count+(two/2);
    	if(one!=0)
    	{
    		one=one-2;
    		count++;
    		two=0;
    	}
    	else
    	{
    		count++;
    		two=0;
    	}
    }
    count=count+(one/4)+(one%4)+three;
  cout<<count<<endl;
}
