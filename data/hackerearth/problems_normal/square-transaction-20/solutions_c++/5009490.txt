#include <iostream>
using namespace std;
 
long long int arr[1000009];
long long int bs(int low ,int high ,int k)
{
	int mid=low+(high-low)/2;
	
	if((arr[mid]==k)||(arr[mid]>k&&arr[mid-1]<k))
	return mid;
	if(arr[mid]<k)
	return bs(mid+1,high,k);
	if(arr[mid]>k)
	return bs(low,mid-1,k);
	
	
}
int main()
{
	int t;
	cin >> t;
	int i;
	long long int sum=0;
	for(i=1;i<=t;i++)
	{
		int a;
		cin >> a;
		sum+=a;
		arr[i]=sum;
	}
	int q;
	cin >> q;
	while(q--)
	{
		int k;
		cin >> k;
		if(k>sum)
		cout << -1 << endl;
		else
		cout << bs(1,t,k)<<endl;
	}
    
    return 0;
}
