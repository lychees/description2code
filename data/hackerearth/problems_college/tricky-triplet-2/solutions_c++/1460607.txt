#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long int n,count=0,diff;
    long int i,j,k;
    cin>>n;
    long long int a[n];
    for(i=0;i<n;i++)
    cin>>a[i];
    for(i=0;i<n-2;i++)
    {
    	for(j=i+1;j<n-1;j++)
    	{
    		diff=a[i]-a[j];
    		for(k=j+1;k<n;k++)
    		{
    			if(a[j]-a[k]==diff)
    			count++;
    		}
    	}
    }
    cout<<count;
    
    return 0;
}
