#include<bits/stdc++.h>
using namespace std;
int main(){
    int t;
    scanf("%d",&t);
    getchar();
    while(t--){
        string s;
        vector<string> v;
        getline(cin,s);
        unordered_set<string> s1;
        stringstream ss;
        ss<<s;
        string temp;
        while(ss.eof()!=true){
            ss>>temp;
            if(s1.find(temp)==s1.end()){
                s1.insert(temp);
                v.push_back(temp);
            }
        }
       for(vector<string> :: iterator it=v.begin();it!=v.end();it++){
        cout<<*it<<" ";
       }
       cout<<endl;
    }
}
