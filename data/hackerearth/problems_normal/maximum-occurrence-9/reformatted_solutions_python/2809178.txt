def occur():
    inp = raw_input()
    x = 0
    val = inp[0]
    for i in inp:
        y = inp.count(i)
        if y == x:
            if val > i:
                x = y
                val = i

        elif y > x:
            x = y
            val = i
    print str(val) + ' ' + str(x)


if __name__ == "__main__":
    occur()
