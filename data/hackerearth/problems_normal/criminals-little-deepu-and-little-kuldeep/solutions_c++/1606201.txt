#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

void calc(vector<unsigned int>& v, int& res)
{
   if(v.size() == 0) {
     return;
   } else {
     res++;
   }

   sort(v.begin(), v.end());
   vector<unsigned int> temp;
   for(int i = 1; i < v.size(); i++)
   {
      if(v[i] == v[i-1]) {
         temp.push_back(v[i]);
      }
    
   }
   calc(temp,res);
 
}

int main()
{
  int T, N, out =0;
  unsigned int temp;
  cin>>T;
  vector <int> result;
  for(int i = 0; i < T; i++)
  {
    vector<unsigned int> inp;
    cin>>N;
    for(int j = 0; j < N; j++){
      cin>>temp;
      inp.push_back(temp);
    }

    calc(inp,out);
    result.push_back(out);
    out = 0;
  }
  for(int i = 0; i < T; i++)
    cout<<result[i]<<endl;

  return 0;
}
