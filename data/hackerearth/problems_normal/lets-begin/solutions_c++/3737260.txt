#include <iostream>
using namespace std;

int main()
{
    int t,x;
    cin>>t;
    while(t--)
    {   
    	
    	int count=0;
    	cin>>x;
    	if(x<8)
    	{
    		if((x==2)||(x==3)||(x==5)||(x==7))
    		{
    			count=1;
    		
    		}
    		
    		
    	   	else
    	   	count=-1;
    	}
    	
    	else 
    	{
      int old=x;
      
      x=x%7;
      
      if(x==0)
      {
      	count=old/7;
      }
     else if((x==1)||(x==2)||(x==3)||(x==5))
      {
      	count=(old/7)+1;
      }
      
      else if((x==4)||(x==6))
      {
      	count=(old/7)+2;
      }
      
      else
      count=-1;
      
    	}
    	
    	cout<<count<<endl;
    }
    return 0;
}
