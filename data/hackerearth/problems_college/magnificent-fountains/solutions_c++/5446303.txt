#include <iostream>
#include <cstring>
#include <cmath>
using namespace std;
#define mod  1000000007
long long pr[10005];
long long a[10003];
int main()
{
	int t, n;
	//freopen("../input.txt", "r", stdin);
	cin >> t;
	while (t--) {
		memset(pr, 0, sizeof(pr));
		cin >> n;
		//cout << n << endl;
		for (int i = 0; i < n; i++) {
			cin >> a[i];
			//cout << a[i] << " ";
		}
		//	cout << endl;
		for (int i = 0; i < n; i++) {
			long long temp = a[i];
			for (long long j = 2; j <= sqrt(a[i]); j++) {
				long long cnt = 0;
				while (temp % j == 0) {
					temp /= j;
					cnt ++;
				}
				if (cnt) {
					pr[j] = max(pr[j], cnt);
				}
			}
			if (temp != 1) {
				pr[temp] = max(pr[temp], 1LL);
			}

		}
		//cout << " bulleya " << endl;
		long long ans = 1;
		for (long long i = 1; i <= 10000; i++) {
			// if(pr[i]){
			// 	cout << pr[i] << " ";
			// 	cout << i  << endl;
			// }
			for (long long j = 1; j <= pr[i]; j++) {
				ans = (ans * i) % mod;
			}
		}
		cout << ans << endl;
	}
	return 0;
}
