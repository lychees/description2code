#include <iostream>
using namespace std;
#define MAX 1000000
bool v[MAX];
long long int len,sp[MAX],count[MAX];
void sieve()
    {
    	for(long long int i =2;i<MAX;i += 2)  sp[i] = 2;
    	for(long long int i =3;i<MAX;i += 2)
    	{
    		if(!v[i])
    		{
    			sp[i] = i;
    			for (long long int j = i; (j*i) < MAX; j += 2){
				if (!v[j*i])	{ v[j*i] = true; sp[j*i] = i; count[i]++; }
    			}
    		}
    	}
    }
int main()
{
    sieve();
    long int t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	if(n%2 ==0) cout<<"500000\n";
    	else
    	{
    	cout<<count[n]+1<<endl;
    	}
    }
    return 0;
}
