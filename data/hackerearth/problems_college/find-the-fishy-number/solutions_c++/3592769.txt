#include<bits/stdc++.h>
#include<time.h>
using namespace std;
#define ll long long int
   
int main(int argc, char const *argv[])
{
	ll i,j,k,n,m,s,t,l,x,flag;

	double a,b,c,d;
 
	cin>>t;
	ll A[10]={0};
	
	for (l=0; l<t; l++)
	{
		 memset(A,0,sizeof(A));
	 
          cin>>n;
          for (j =0; j <n; j++)
          {
          	cin>>m;
          	s= log10(m);
          	s= m/(pow(10,s));
          	A[s]++;
          	//cout<<s<<" ";
          }
             cin>>c;
             flag=0;

          for (x= 1; x<10; x++)
          {
          	  a= (double)n*(double)log10(1+1/(double)x);
          	   //cout<<a<<"  ";
          	  if(A[x]<=a/c){
          	  	flag=1;
          	  	break;
          	  }
          	  if(A[x]>=a*c){
          	  	flag=1;
          	  	break;
          	  }   
          }
          if(flag==1){
          	cout<<x<<"\n";
          }
          else{
          	cout<<-1<<"\n";
          }
     }
  
	return 0;
}
