#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
    //freopen("bear.txt","r",stdin);
    int T;
    cin >> T;
    cin.ignore();

    while ( T-- ) {
    string s;
    getline(cin,s);

    int l = s.length();
    int a, vol=0, con=0, flag=1;
    for ( int i=0 ; i<l ; i++ ) {
        a = (int)s[i];
        if ( a==97 || a==101 || a==105 || a==111 || a==117 || a==121 ) {
                vol++;
            }
        else {
            con++;
                if( (int)s[i+2]!=97 && (int)s[i+1]!=97 &&
                        (int)s[i+2]!=101 && (int)s[i+1]!=101 &&
                            (int)s[i+2]!=105 && (int)s[i+1]!=105 &&
                                (int)s[i+2]!=111 && (int)s[i+1]!=111 &&
                                    (int)s[i+2]!=117 && (int)s[i+1]!=117 &&
                                        (int)s[i+2]!=121 && (int)s[i+1]!=121 &&
                                            (int)s[i+2]>97 && (int)s[i+1]>97    &&
                                                (int)s[i+2]<123 && (int)s[i+1]<123) {
                                        cout <<"hard" <<endl;
                                        flag=0;
                                        break;
                                        }

        }
    }
    if ( con > vol && flag==1 ) {
        cout <<"hard" <<endl;
    }
    else if ( con <= vol && flag==1 ) {
        cout <<"easy" <<endl;
    }
    }
}
