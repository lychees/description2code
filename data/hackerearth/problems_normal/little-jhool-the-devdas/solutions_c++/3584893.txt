#include<iostream>
using namespace std;
#include<vector>
#include<algorithm>
int main()
{
	int n;
	cin>>n;
	vector<int> v(n);
	long long int sum=0;
	for(int i=0;i<n;i++)
	{
	      cin>>v[i];
	      sum=sum+v[i];
	}
	if( sum % 2 != 0)
			cout<<sum<<endl;
	else
	{
	 sort(v.begin(),v.end());
	 vector<int>::iterator p=v.begin();
	 //vector<int>::iterator q=v.end()-1;
	 //cout<< *p <<" "<< *q <<" ";
	 while( *p % 2 == 0 && p!=v.end() )
	 {
	         //cout<< *p << " ";
	   		 p++;
     }
	 	
		if( p == v.end())
	             cout<<":("<<endl;
	             else
	               cout<<(sum-*p)<<endl;
	}
	return 0;
			    
}
