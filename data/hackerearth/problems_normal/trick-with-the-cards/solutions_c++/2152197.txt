#include <iostream>
using namespace std;

int main()
{
    ios::sync_with_stdio(false);

    int t;
    cin>>t;
    long long n;
    while(t--) {
    	cin>>n;
    	if(n == 1 || n == 2 || n == 3) {
    		cout<<"1\n";
    	}
    	else if(n % 3 == 0) {
    		cout<<(n/3)<<endl;
    	}
    	else if(n % 3 == 2) {
    		cout<<n<<endl;
    	}
    	else {
    		int x = 0;//((n/3) - 1)*2;
    		cout<<(n + x)<<endl;
    	}
    }
    
    return 0;
}
