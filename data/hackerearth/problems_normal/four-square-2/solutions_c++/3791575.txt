#include <iostream>
using namespace std;
typedef long long int ll;
ll calc(ll n) {
	ll i, s1 = 0;
	for(i = 1; i*i < n; i++) {
   		if(n%i == 0) {
   			s1+=i + n/i;;
    	}
    }
    if(i*i == n) {
    	s1+=i;
    }
    return s1;
}
int main()
{
    int t;
    cin >> t;
    while(t--) {
    	ll n;
    	cin >> n;
    	ll i, k, s1, s2 = 0;
    	s1 = calc(n);
    	if(n%4 == 0) {
    		s2 = calc(n/4);
    	}
    	s1 = 8*s1 - 32 * s2;
    	cout << s1 << endl;
    }
    return 0;
}
