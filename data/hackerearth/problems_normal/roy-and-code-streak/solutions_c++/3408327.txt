#include<bits/stdc++.h>
using namespace std;
#define assn(n,a,b) assert(n<=b && n>=a)
bool check[1000009]={};
int main()
{
    int t,cnt=0;
    cin >> t;
    assn(t,1,1000);
    while(t--)
    {
        memset(check,0,sizeof(check));
        int n,i,ans=-1,flag=0,sid,ac,c=0;
        cin >> n;
        assn(n,1,1000000);
        cnt+=n;
        for(i=0; i<n; i++)
        {
            cin >> sid >> ac;
            assn(sid,1,1000000);
            assn(ac,0,1);
            if(ac==0)
            {
                ans=max(c,ans);
                c=0;
            }
            else
            {
                if(check[sid]==0)
                    c++;
                check[sid]=1;
            }
            ans=max(c,ans);
        }
        cout << ans << endl;
    }
    assn(cnt,1,1000000);
    return 0;
}
