#include <bits/stdc++.h>
using namespace std;

map <string, int> mp;
map <string, int> :: iterator it;

int  main()
{

	long long t;
	cin>>t;
	for(long long z=0;z<t;z++)
	{
		string s;
		cin>>s;
		sort(s.begin(),s.end());
		mp[s]++;
	}
	int ma=0;
	for(it=mp.begin();it!=mp.end();it++)
    {
        if((*it).second>ma)
            ma=(*it).second;
    }
    cout << ma << endl;
    return 0;
}
