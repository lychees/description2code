divisors={}
def compute_divisors():
	global divisors
	for i in range(1,31):
		div = []
		for j in range(1,i):
			if i%j==0:
				div.append(j)
		divisors[i]=div

def numberCanBeWritten(num,written_numbers):
	for e in written_numbers:
		if num in divisors[e]:
			return False
	return True


def solve(written_numbers,numbers_available,turn):
	nturn = (turn+1)%2+1
	if numbers_available==[]:
		return nturn
	for num in numbers_available:
		nwnum = written_numbers
		nnumavail = numbers_available
		if numberCanBeWritten(num,written_numbers):
			for e in divisors[num]:
				nnumavail.remove(e)
			nwnum.append(num)
			if solve(nwnum,nnumavail,nturn) == turn:
				return turn
	return nturn
