int INP() {
	int n,flag;
	char c;
	n = 0;
	flag = 1;
	c = getchar();
	while (c < 33) {
		c = getchar();
	}
	if (c == '-') {
		flag = -1;
		c = getchar();
	}
	while (c > 32) {
		n = n * 10 + c - '0';
		c = getchar();
	}
	n = n * flag;
	return n;
}

int GCD (int a,int b) {
	if (b == 0) return a;
	return(a % b == 0 ? b : GCD(b,a % b));
}

long long int mod_pow(long long int base,long long int exp) {
	long long int val;
	val = 1;
	while (exp > 0) {
		if (exp % 2 == 1) {
			val = (val * base) % MOD;
		}
		base = (base * base) % MOD;
		exp = exp / 2;
	}
	return val;
}
