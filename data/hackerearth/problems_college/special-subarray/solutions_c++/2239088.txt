#include <bits/stdc++.h>
using namespace std;

const int size = 100000;
long long int L[size+10] , R[size+10] , arr[size+10];

int main() {
	cin.sync_with_stdio(false);
	int n , m;
	while ( cin >> n >> m ) {
		for ( int i = 1 ; i<=n ; ++i ) {
			cin >> arr[i];
		}
		arr[0] = arr[n+1] = 0x3f3f3f3f;
		for ( int i = 1 ; i<=n ; ++i ) {
			L[i] = arr[i] >= arr[i-1] ? L[i-1] : i;
		}
		for ( int i = n ; i>=1 ; --i ) {
			R[i] = arr[i] >= arr[i+1] ? R[i+1] : i;
		}
		int x , y;
		for ( int i = 1 ; i<=m ; ++i ) {
			cin >> x >> y;
			if ( R[x]>=y || L[y]<=x ) {
				cout << "Yes" << endl;
			} else {
				int pos = lower_bound( L+x+1 , L+y+1 , x+1 ) - L;
				-- pos;
				if ( L[pos]<=x&&R[pos]>=y ) {
					cout << "Yes" << endl;
				} else {
					cout << "No" << endl;
				}
			}
		}
	}
	return 0;
}
