#include <bits/stdc++.h>
#define MAX 1001
using namespace std;

int main() {
	int t;
	scanf("%d",&t);
	bool isprime[MAX+1];
	memset(isprime,true,sizeof(isprime));
	isprime[0] = isprime[1] = false;
	for(int i=2;i<=MAX/2;i++)
		for(int j=2;j<=MAX/i;j++)
			isprime[i*j] = false;
	while(t--) {
		int n;
		scanf("%d",&n);
		int res = 0;
		for(int i=0;i<n;i++) {
			int temp;
			scanf("%d",&temp);
			if(isprime[temp])
				res++;
		}
		printf("%d\n",res);
	}
	return 0;
}
