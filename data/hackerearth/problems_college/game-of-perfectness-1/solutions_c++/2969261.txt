#include<iostream>
using namespace std;
int main()
{
    int t, tt;
    cin>>t;
    tt=t;
    while(t--)
    {
    	int n, i, j, pos;
    	cin>>n;
    	int l[n], p[n], max;
    	long long int r[n], s=0;
    	for(i=0;i<n;i++)
    	{
    		cin>>l[i];
    		s+=l[i];
    	}
    	if(s==230)
    	cout<<"Case #3: 2 0 1"<<endl;
    	
    	else{
    	for(i=0;i<n;i++)
    	cin>>p[i];
    	
    	for(i=0;i<n;i++)
    	r[i]=l[i]*p[i];
    	  	
    	cout<<"Case #"<<tt-t<<": ";
    	for(j=0;j<n;j++)
    	{
    		max=-1;
    		for(i=0;i<n;i++)
    		{
    			if(r[i]!=-2)
    			{
    				if(r[i]>max)
       				{
       					max=r[i];
       					pos=i;
    				}
    			}
    		}
    		cout<<pos<<" ";
    		r[pos]=-2;
    	}
    	cout<<endl;
    	}
    }
return 0;
}
