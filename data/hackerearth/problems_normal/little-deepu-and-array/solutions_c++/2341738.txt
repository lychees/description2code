#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    int m;
    cin>>m;
    for(int i=0;i<m;i++)
    {
    	int k;
    	cin>>k;
    	for(int j=0;j<n;j++)
    	{
    		if(a[j]>k)
    		a[j]=a[j]-1;
    	}
    }
    
    for(int i=0;i<n;i++)
    {
    	cout<<a[i]<<" ";
    }
    return 0;
}
