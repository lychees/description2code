#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
    int n,q,i;
    cin>>n>>q;
    int* A = new int[n];
    for(i=0;i<n;++i){
    	cin>>A[i];
    }
    sort(A,A+n);
    int tmp;
    while(q--){
    	cin>>tmp;
    	if(binary_search(A,A+n,tmp)){
    		cout<<"YES\n";
    	}else
    	cout<<"NO\n";
    }
    return 0;
}
