
#include <iostream>
#define getcx getchar_unlocked
using namespace std;

#define MAX(a, b) (((a) > (b)) ? (a) : (b))
#define MIN(a, b) (((a) < (b)) ? (a) : (b))
#define ABS(a) (((a) > (0)) ? (a) : (-1 * (a)))

typedef long long int64;

inline void inp(int64  &n)//fast input function
{
	n = 0;
	int ch = getcx();
	int sign = 1;
	while (ch < '0' || ch > '9') {
		if (ch == '-') {
			sign = -1;
		}
		ch = getcx();
	}
	while (ch >= '0' && ch <= '9') {
		n = (n << 3) + (n << 1) + ch - '0', ch = getcx();
	}
	n = n * sign;
}

inline void inp(int  &n)//fast input function
{
	n = 0;
	int ch = getcx();
	int sign = 1;
	while (ch < '0' || ch > '9') {
		if (ch == '-') {
			sign = -1;
		}
		ch = getcx();
	}
	while (ch >= '0' && ch <= '9') {
		n = (n << 3) + (n << 1) + ch - '0', ch = getcx();
	}
	n = n * sign;
}


long long a[1000004], b[1000004];

int main() {
	int n;
	//cin >> n;
	inp(n);
	
	for (int i = 0; i < n; i++) {
		//cin >> a[i];
		inp(a[i]);
		b[i] = 0;
	}
	
	int m;
	//cin >> m;
	inp(m);
	while (m--) {
		int i, j;
		long long k;
		//cin >> i >> j >> k;
		inp(i);
		inp(j);
		inp(k);
		i--; j--;

		b[i] += k;
		b[j + 1] -= k;
	}

	for (int i = 0; i < n; i++) {
		if (i > 0)
		b[i] += b[i - 1];
		a[i] += b[i];
		//cout << a[i] << endl;
		printf("%lld\n", a[i]);
	}
}
