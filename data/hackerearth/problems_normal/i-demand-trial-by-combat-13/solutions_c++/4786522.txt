#include<iostream>
#include<stdio.h>
#include<vector>
using namespace std;
int main()
{ int tc;
  scanf("%d",&tc);
  while(tc--)
  { int n;
    scanf("%d",&n);
    int q;
    scanf("%d",&q);
    vector <int> a(n);
    int i;
    for(i=0;i<n;i++)
       scanf("%d",&a[i]);
    vector <int> b(n);
    while(q--)
    { std::fill(b.begin(), b.end(), 0);
      int count=0;
      for(i=0;i<n;i++)
      { if(i==0)
        { if(a[i+1]==1)
              { b[i]=1; count++;
              }
          continue;
        }
        if(i==(n-1))
        { if(a[i-1]==1)
             { b[i]=1;  count++;
             }
          continue;
        }
        if((a[i+1]==1)&&(a[i-1]==1))
           { b[i]=1;  count++;
           }
      }
      a=b;
      if((count==n)||(count==0))
         break;
     }
     for(i=0;i<n;i++)
       printf("%d ",a[i]);
     cout<<"\n";
  }
  return 0;
}
