
#include <iostream>
#include <algorithm>

using namespace std;

int n,cc;

long long a[7],m;

int main()
{
    int test;

    cin >> test;
    while(test--)
    {
        cc=-1;

        cin >> n >> m;

        for(int i=0;i<n;i++)
            cin >> a[i];

        sort(a,a+n);

        for(int i=n-1;i>=0;i--)
        {
            m-=a[i];

            if(m<=0)
            {
                cc=n-i;
                break;
            }
        }

        cout << cc << "\n";

    }

}
