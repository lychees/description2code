for _ in range(int(input())):
    l,b,h = map(float,raw_input().split())
    if(l<b and l<h):
        print "%0.1f"%(l/2)
    elif(b<h and b<l):
        print "%0.1f"%(b/2)
    elif(h<l and h<b):
        print "%0.1f"%(h/2)
    elif(l==b and l<h):
        print "%0.1f"%(l/2)
    elif(l==h and l<b):
        print "%0.1f"%(l/2)
    elif(b==h and b<l):
        print "%0.1f"%(h/2)
    elif(l==b and l==h):
        print "%0.1f"%(l/2)
