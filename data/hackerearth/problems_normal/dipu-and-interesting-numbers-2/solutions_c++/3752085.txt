long long count_ele(long long x)
{
	long long low = 1;
	long long high = 1e9;
	long long max_ele = 0;
	while (low <= high)
	{
		long long mid = ((low + high) >> 1);
		if (mid * mid - 1 < x)
		{
			if (max_ele < mid)
				max_ele = mid;
			low = mid + 1;
		}
		else
			high = mid - 1;
		//cout << low << sp << high << endl;
	}
	return max_ele;
}
