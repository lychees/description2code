#include <iostream>
#include<cstdio>
#include<cmath>
using namespace std;
#define LL long long int
int same(LL m, LL n)
{
	int a[9],b[9],i,j,a1,b1,flag;
	for(i=0;i<9;i++)
	{
		a[i]=0;b[i]=0;
	}
	i=0;
	while(n)
	{
		a[i++]=(n%10);
		n/=10;
	}
	a1=i;j=0;
	while(m)
	{
		b[j++]=(m%10);
		m/=10;
	}
	b1=j;flag=0;
	for(i=0;i<a1;i++)
	{
		for(j=0;j<b1;j++)
		{
			if(a[i]==b[j])
				return 1;
		}
	}
	return 0;
}
int main() 
{
	LL t,n,i,cnt;
	int k;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
		if(n==1)
			printf("1\n");
		else
		{
			cnt=0;
			for(i=1;i*i<=n;i++)
			{
				if(n%i==0)
				{
					if(i*i!=n)
					{
						k=same(i,n);
						if(k==1)
							++cnt;
						k=same(n/i,n);
						if(k==1)
							++cnt;
					}
					else if(i*i==n)
					{
						k=same(i,n);
						if(k==1)
							++cnt;
					}
				}
			}
			printf("%lld\n",cnt);
		}
	}
	return 0;
}
