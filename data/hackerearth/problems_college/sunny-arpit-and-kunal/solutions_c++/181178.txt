#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int a[3],b,c,d;
    	cin>>a[0]>>a[1]>>a[2];
    	int l=*min_element(a,a+3);
    	int n=*max_element(a,a+3);
    	int m;
    	for(int i=0;i<3;i++)
    	{
    		if(a[i]!=l&&a[i]!=n)
    			m=a[i];
    	}
		int s=0;
		//cout<<l<<m<<n<<endl;
		while(((m-l)!=1)||((n-m)!=1))
		{
			if((m-l)>=(n-m))
			{
				int temp;
				temp=m;
				m=l+1;
				n=temp;
			}
			else
			{
				int temp;
				temp=m;
				m=n-1;
				l=temp;
			}
			s++;//cout<<l<<m<<n<<endl;
		}
		cout<<s<<endl;

    }
    return 0;
}
