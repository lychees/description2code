inline void scanint(int *a)
{
    char c = 0;int sign=1;
    while(c<33)
       // c = fgetc_unlocked(stdin);
    c = getc(stdin);

    *a = 0;
    while(c>33)
    {
        if(c=='-'){ sign=-1;
        c = getc(stdin);
        //c = fgetc_unlocked(stdin);
        continue;}
        *a = *a*10 + c - '0';
       // c = fgetc_unlocked(stdin);
    c = getc(stdin);
    }
    *a=*a * sign;
    //printf("%d\n",*a);
}
char outputbuf[20];

inline void putint(int n)
{
	outputbuf[19]='\n';
    bool flag=false;int i=18,r;
    if(n<0){ flag=true; n*=-1;}
    while(n!=0)
    {
        r=n%10;
        outputbuf[i--]=r+'0';
        n/=10;
    }
    if(flag) outputbuf[i--]='-';
    i++;
    while(i<20)
    {
        //putchar_unlocked(outputbuf[i++]);
        putc(outputbuf[i++],stdout);
    }
}
int main()
{
    int t,n,x,y,maxx,maxy,minx,miny,i,area;
    scanint(&t);
    while(t--)
    {
        scanint(&n);
        maxx=maxy=-200;
        minx=miny=200;
        for(i=0;i<n;++i)
        {
            scanint(&x);
            scanint(&y);
            maxx=max(maxx,x);
            minx=min(minx,x);
            maxy=max(maxy,y);
            miny=min(miny,y);
        }
        y=maxy-miny;
        x=maxx-minx;
        area=x*y;
        if(area==0)
        {
            if(minx==maxx) area=maxy-miny;
            else if(maxy==miny) area=maxx-minx;
        }
        printf("%d\n",area);
    }
    return 0;
}
