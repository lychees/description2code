#include <iostream>
#include<string.h>
using namespace std;
int n;
int dx[]={0,0,1,-1};
int dy[]={1,-1,0,0};
int a[102][102];
int visited[102][102]={};

#define beyond(x,y) (x<0||y<0||x>n-1||y>n-1)
long long int result=0;

void dfs(int i,int j)
{

	if(i==n-1 && j==n-1)
	{
        if(a[i][j]==0)
		result++;
		return;
	}
	if(beyond(i,j)||a[i][j]==1||visited[i][j])
	return;

	visited[i][j]=1;
	for(int k=0;k<4;k++)
	   {
	     dfs(i+dx[k],j+dy[k]);
	   }
	visited[i][j]=0;


}




int main()
{

    cin>>n;

   memset(visited,0,sizeof(visited));
    for(int i=0;i<n;i++)
     	for(int j=0;j<n;j++)
     	cin>>a[i][j];


     dfs(0,0);
    cout<<result;

    return 0;
}
