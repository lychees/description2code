#include<bits/stdc++.h>
#define ll long long
using namespace std;
#define mod 1000000007
ll dp[1000005][10];

ll solve(ll n,ll k,ll given)
{
    if(dp[given-n][k]!=-1)
        return dp[given-n][k];

    if(n==given)
        return 1;
    if(k==0)
        return dp[given-n][k]=solve(n+1,k+1,given)%mod;
    else if(k==9)
        return dp[given-n][k]=solve(n+1,k-1,given)%mod;

    else
        return dp[given-n][k]=(solve(n+1,k+1,given)+solve(n+1,k-1,given))%mod;



}

int main()
{
    ll t,k;
    cin>>t;
    for(int i=0;i<1000005;i++)
    {
        for(int j=0;j<10;j++)
            dp[i][j]=-1;
    }
    while(t--)
    {

        cin>>k;
        if(k==1)
        {
            cout<<"10\n";

        }
        else if(k==0)
        {
        	cout<<"0\n";        }
        else
        {
        ll ans=0;

        for(int i=1;i<=9;i++)
            ans=(ans+solve(1,i,k))%mod;
        cout<<ans<<"\n";
        }

    }



}
