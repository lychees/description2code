#include <bits/stdc++.h>

using namespace std;

int N, X;
map<int, int> cache;
int now;

int solve(int n, int x)
{
    now++;
    if(cache.count(n))
        x%=now-cache[n];
    else
        cache[n]=now;
    if(x==0)
        return n;
    int m=0;
    while(n)
    {
        m+=(n%10)*(n%10);
        n/=10;
    }
    return solve(m, x-1);
}

int main()
{
    int T;
    scanf("%d", &T);
    while(T--)
    {
        scanf("%d%d", &N, &X);
        cache.clear();
        now=0;
        printf("%d\n", solve(N, X));
    }
    return 0;
}
