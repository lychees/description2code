ll arr[1000005];
 int mat[100005][26];

int main(){

   string str;
   cin>>str;

   int len=(int)str.size();

   for(int i=0;i<26;++i)
      mat[len][i]=len;

   for(int i=len-1;i>=0;--i){

      for(int j=0;j<26;++j)
        mat[i][j]=mat[i+1][j];

      mat[i][str[i]-'a']=i;

   }



   int q;
   cin>>q;

   while(q--){

      string pat;
      cin>>pat;

      int cur=0;
      bool check=true;
      int i;
      for(i=0;i<(int)pat.size() && cur<(int)str.size();++i){
          if(mat[cur][pat[i]-'a']==len){
             check=false;
             break;
          }
          else{

             cur=mat[cur][pat[i]-'a'];
          }
      }

      if(cur==len || i!=(int)pat.size())
        check=false;

      if(check==false)
        cout<<"No\n";
      else
        cout<<"Yes\n";


   }


   return 0;
}
