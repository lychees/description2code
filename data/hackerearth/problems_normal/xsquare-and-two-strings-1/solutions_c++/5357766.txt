#include <iostream>
#include<string.h>
using namespace std;

int main()
{
	long long int t;
	cin>>t;
	while(t--)
	{
		bool yes = false;
		bool ss1[26] = {false};
		bool ss2[26] = {false};
    	string s1, s2;
    	cin>>s1;
    	cin>>s2;
    	long long int l1 = s1.size(), l2 = s2.size();
    	for(long long int i = 0; i < l1; i++)
    	{
    		ss1[s1[i] - 'a'] = true;
    	}
		for(long long int j = 0; j < l2; j++)
    	{
    		if(ss1[s2[j] - 'a'])
    		{
    			yes = true;
    			break;
    		}
    		
    	}
	    	
    	
    	if(yes)
    		cout << "Yes" << endl;
		else
			cout << "No" << endl;
		
	}
    return 0;
}
