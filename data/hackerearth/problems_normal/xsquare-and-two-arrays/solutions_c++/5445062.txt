#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,q,i;
	cin>>n>>q;
	long long int A[n],B[n];
	for(i=0;i<n;i++)
	{
		cin>>A[i];
	}
	for(i=2;i<n;i++)
	{
		A[i]+=A[i-2];
	}
	for(i=0;i<n;i++)
	{
		cin>>B[i];
	}
	for(i=2;i<n;i++)
	{
		B[i]+=B[i-2];
	}
	int l,r,choice;
	while(q--)
	{
		cin>>choice>>l>>r;
		l--;
		r--;
		long long int ans=0;
		if(choice==1)
		{
			if((l%2==0 && r%2==0)||(l%2==1 && r%2==1))
			{
					if(l-2>=0)
					ans+=A[r]-A[l-2];
					else
					ans+=A[r];
					if(l-1>=0)
					ans+=B[r-1]-B[l-1];
					else
					ans+=B[r-1];
		   }
			else
			{
					if(l-2>=0)
					ans+=A[r-1]-A[l-2];
					else
					ans+=A[r-1];
					if(l-1>=0)
					ans+=B[r]-B[l-1];
					else
					ans+=B[r];
					
			}
		}
		else if(choice==2)
		{
			if((l%2==0 && r%2==0)||(l%2==1 && r%2==1))
			{
					if(l-2>=0)
					ans+=B[r]-B[l-2];
					else
					ans+=B[r];
					if(l-1>=0)
					ans+=A[r-1]-A[l-1];
					else
					ans+=A[r-1];
		   }
			else
			{
					if(l-2>=0)
					ans+=B[r-1]-B[l-2];
					else
					ans+=B[r-1];
					if(l-1>=0)
					ans+=A[r]-A[l-1];
					else
					ans+=A[r];
					
			}
		}
		cout<<ans<<endl;
	}
	return 0;
}
