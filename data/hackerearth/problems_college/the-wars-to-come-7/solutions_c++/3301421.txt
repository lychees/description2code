#include <iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;

int a[1005];

int main()
{
	int n,x,i,c,s;
	scanf("%d",&n);
	for(i=0;i<n;i++)
		scanf("%d",&a[i]);
	scanf("%d",&x);
	sort(a,a+n);
	s=0;
	for(i=n-1;i>=0;i--)
	{
		c++;
		s+=a[i];
		if(s>=x)
			break;
	}
	printf("%d\n",c);
    return 0;
}
