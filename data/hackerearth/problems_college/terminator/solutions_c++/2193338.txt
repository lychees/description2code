#include <iostream>
#include <string>

using namespace std;

int main()
{
    int t,l,j,ctr=0,flag=0;
    string s;
    cin>>t;
    getline(cin,s);
    for(int i=1;i<=t;i++)
    {
        getline(cin,s);
        l=s.length();
        flag=0;
        ctr=0;
        for(j=0;j<l;j++)
        {
            switch(s[j])
            {
            case 'x':
                if(flag==0)
                 {

                    for(int p=j-1;p>=j-ctr;p--)
                    s[p]='a';
                 }
                 flag=1;
                 ctr=0;
                 break;
            case '*':
                ctr=0;
                flag=0;
                break;
            case 'o':
                 if(flag==0)
                    ctr++;
                else
                    s[j]='a';
                 break;
            }
            //cout<<s<<" "<<flag<<" "<<ctr<<endl;
        }
        for(j=0;j<l;j++)
        {
            if(s[j]!='a')
                cout<<s[j];
        }
        cout<<endl;

    }

}
