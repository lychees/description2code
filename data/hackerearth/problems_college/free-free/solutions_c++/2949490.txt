#include <iostream>
#include<string.h>
using namespace std;

int cost(char a[])
{
	int c=0,size=strlen(a),n[52];
	
	for(int i=0;i<52;i++)
	n[i]=0;
	
	char ch1='a',ch2='A';
	for(int i=0;i<26;i++)
	{
		for(int j=0;j<size;j++)
		{
			if(a[j]==ch1)
			n[i]++;
		}
		ch1++;
	}
	
	for(int i=26;i<52;i++)
	{
		for(int j=0;j<size;j++)
		{
			if(a[j]==ch2)
			n[i]++;
		}
		ch2++;
	}
	
	
	for(int i=0;i<52;i++)
	{
		if(n[i]%2==0)
		c+=(n[i]/2);
		else
		c+=((n[i]+1)/2);
	}
	return c;
}	
	



int main()
{
   int costs[100],test;
   
  
   
   char item[100][200];
   cin>>test;
   
   for(int i=0;i<test;i++)
   	{cin>>item[i];
   	costs[i]=cost(item[i]);
   	}
   
   for(int j=0;j<test;j++)
   {cout<<costs[j]<<'\n';}
   
   
    return 0;
}
