#include <iostream>
using namespace std;

int main()
{
    int t;
    string s;
    cin>>t;
    while(t--) {
    	s.clear();
    	cin>>s;
    	int len = s.length();
    	int x = 0;
    	for(int i=len-1;i>0;i--) {
    		if(s[i] > s[i-1]) {x = i; break;}
    	}
    	if(x == 0) cout<<"no answer\n";
    	else {
    		x--;
    		int y;
    		for(int i=len-1;i>x;i--) {
    			if(s[i] > s[x]) {y = i; break;}
    		}
    		char temp = s[x];
    		s[x] = s[y];
    		s[y] = temp;
    		int i=x+1, j=len-1;
    		while(i<j) {
    			temp = s[i];
    			s[i] = s[j];
    			s[j] = temp;
    			i++;
    			j--;
    		}
    		cout<<s<<endl;
    	}
    }
    return 0;
}
