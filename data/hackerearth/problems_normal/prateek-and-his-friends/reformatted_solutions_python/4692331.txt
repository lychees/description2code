def f(arr, ans):
    s = 0
    t = 0
    for i in xrange(0, len(arr) + 1):
        while s > ans:
            s -= arr[t]
            t += 1
        if s == ans:
            return "YES"
        if i < len(arr):
            s += arr[i]
    return "NO"


T = input()
for i in xrange(T):
    N, cost = raw_input().split(" ")
    k = []
    for c in xrange(int(N)):
        k.append(input())
    print f(k, int(cost))
