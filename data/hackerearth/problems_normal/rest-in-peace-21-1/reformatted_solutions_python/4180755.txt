n = int(raw_input())
for _ in range(n):
    v = raw_input()
    if "21" in v:
        print "The streak is broken!"
        continue
    v = int(v)
    if v % 21 == 0:
        print "The streak is broken!"
    else:
        print "The streak lives still in our heart!"
