#include <bits/stdc++.h>

using namespace std;

int p[1000010];
int main()
{
    int t,n,i,j,x;

    //freopen("input.txt","r",stdin);

    for(i=1;pow(i,i)<=1000000;i++)
        p[(int)pow(i,i)]=1;

    for(i=2;i<=1000000;i++)
    {
        x=i;
        while(x>0)
        {
            if(i-(x%10)*(x%10)  > 0)
                p[i]=max(p[i],p[i-(x%10)*(x%10)]);
            x/=10;
        }
    }
    cin>>t;

    while(t--)
    {
        cin>>n;

        p[n]==1?cout<<"Yes"<<endl:cout<<"No"<<endl;
    }
}
