#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdio>
#include <string>

using namespace std;

int main()
{
    int t;

    scanf("%d", &t);

    while (t--) {
        int n, k;
        scanf("%d%d", &n, &k);

        vector<int> v(n);

        for (int i = 0; i < n; i++) {
            scanf("%d", &v[i]);
        }

        sort(v.begin(), v.end());


        int curr_cost = 0;
        int med_in;
        for (int st_in = 0, st_en = k-1; st_en < n; st_in++, st_en++) {
            int k = st_in + (st_en-st_in)/2 - 1;
            for (med_in = k; med_in <= st_en; med_in++) {
            int in_ctr = 0;
            for (int i = st_in; i < med_in; i++) {
                in_ctr += v[med_in]-v[i];
            }
            int de_ctr = 0;
            for (int i = st_en; i > med_in; i--) {
                de_ctr += v[i]-v[med_in];
            }
            int tmp_cost = in_ctr*3 + de_ctr*5;
            //cout << endl << tmp_cost << endl;
            if (st_in == 0 && med_in == k) {
                curr_cost = tmp_cost;
            } else if (curr_cost > tmp_cost) {
                curr_cost = tmp_cost;
            }
            }
        }
        cout << curr_cost << endl;
    }

    return 0;
}
