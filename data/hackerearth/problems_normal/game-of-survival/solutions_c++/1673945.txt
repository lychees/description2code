#include <iostream>
#include <array>

using namespace std;

array<int, 201> Afreq, Sfreq;
int main()
{
    int T;
    cin >> T;
    
    while(T--)
    {
    	int N;
    	cin >> N;
    	
    	if(N == 1)
    	{
    		int S, A;
    		cin >> S >> A;
    		if(S >= A) cout << "YES\n";
    		else
    		{
    			cout << "NO\n";
    		}
    		break;
    	}
    	
    	int oldN = N;
    	
    	while(N--)
    	{
    		int S;
    		cin >> S;
    		
    		Sfreq[S]++;
    	}
    	
    	N = oldN;
    	while(N--)
    	{
    		int A;
    		cin >> A;
    		
    		Afreq[A]++;
    	}
    	
    	
    	int i;
    	for(i = 200; i >= 1; i--)
    	{
    		int diff = Sfreq[i] - Afreq[i];
    		
    		if(diff < 0)
    		{
    			std::cout << "NO\n";
    			break;
    		}
    		else
    		{
    			Sfreq[i-1] += diff;
    		}
    	}
    	
    	
    	
    	if(i == 0) std::cout << "YES\n";
    	
    	//Reset
    	for(int i = 0; i < 200; i++)
    	{
    		Afreq[i] = Sfreq[i] = 0;
    	}
    }
}
