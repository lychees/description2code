def f(t, a, b, c, d):
    return a*t*t*t + b*t*t + c*t + d


for T in range(input()):
    a, b, c, d, k = map(int, raw_input().rstrip().split())
    if d >= k:
        print 0
        continue
    t = 0
    begin, end = 1, 10**18
    while (begin < end):
        mid = (begin + end) / 2
        if f(mid, a, b, c, d) > k:
            t = mid
            end = mid
        elif f(mid, a, b, c, d) < k:
            begin = mid + 1
        else:
            t = mid + 1
            break
    print t-1
