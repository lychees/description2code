import math

def isPrime(n):
	i = 2
	while i <= math.sqrt(n):
		if n%i == 0:
			return False
		i += 1
	return True

n = int(raw_input())

count = 0
j = 1

for i in range(2,n+1):
	
	if i == 2:
		count = 1
	else:
		if isPrime(i):
			j = j*2
		count = count + j
