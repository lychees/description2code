#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#define TOTAL_CLASSES 604800
using namespace std;

int main()
{
    int t,s,f,c,total,toAttend,i,j,sum,seconds;
    cin>>t;
    while(t--){
    	total=0;
    	seconds=0;
    	vector <int> v;
    	for(i=0;i<7;i++){
    		cin>>c;
    		total+=c;
    		for(j=0;j<c;j++){
    			cin>>s>>f;
    			v.push_back(f-s);
    		}	
    	}
    	
    	sort(v.begin(),v.end());
    	toAttend = (int)ceil(0.75*total);
    	
    	i=0;
    	vector<int> :: iterator it;
    	for(it=v.begin();it!=v.end();it++){
    		if(i>=toAttend){
    			break;
    		}
    		
    		seconds+=*it;
    		i++;
    	}
    	cout<<(TOTAL_CLASSES - seconds)<<endl;
    }
    return 0;
}
