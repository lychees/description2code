#include <iostream>
using namespace std;
long ret_sum(long n)
{
 long sum=0;
 while(n>0)
 {
  sum+=(n%10);
  n=(n/10);
 }
 return sum;
}
int main()
{
 long sum,sum1,n,t;
 cin>>t;
 while(t!=0)
 {
  cin>>n;
  sum=ret_sum(n);
  while(sum>=10)
  {
	sum1=ret_sum(sum);
	sum=sum1;
  }
  cout<<sum<<"\n";
  t--;
 }
    return 0;
}
