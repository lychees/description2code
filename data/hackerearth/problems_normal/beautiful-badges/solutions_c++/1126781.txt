#include <iostream>
#include <bits/stdc++.h>
using namespace std;

vector<int> blue;
vector<int> yellow;
bool check(int i,int j)
{
	int mi=min(blue[i],yellow[j]);
	int ma=max(blue[i],yellow[j]);
	if(2*mi*mi>ma*ma) return true;
	return false;
}
int main()
{
    int n,d,c;
    scanf("%d",&n);
    for(int i=0;i<n;++i)
    {
    	scanf("%d %d",&c,&d);
    	if(c==1) blue.push_back(d);
    	else yellow.push_back(d);
    }
    sort(blue.begin(),blue.end());
    sort(yellow.begin(),yellow.end());
    int i=0,j=0,l1=blue.size(),l2=yellow.size(),ans=0;
    while(i<l1 && j<l2)
    {
    	if(check(i,j))
    	{
    		ans++;
    		i++;j++;
    	}
    	else
    	{
    		if(blue[i]<yellow[j]) i++;
    		else j++;
    	}
    }
    printf("%d",ans);
    return 0;
}
