#include <iostream>
using namespace std;

long long no(long long val)
{
	long long rev=0;
	while(val>0)
	{
		rev=rev*10+val%10;
		val/=10;
	}
	return rev;
}

int main()
{
   	int t;
   	cin>>t;
   	while(t--)
   	{
   		long long a,b;
   		cin>>a>>b;
   		long long c=no(a)+no(b);
   		c=no(c);
   		cout<<c<<endl;
   	}
    return 0;
}
