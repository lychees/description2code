#include<bits/stdc++.h>
using namespace std;
static bool v[3000001];
int main(){
	memset(v,false,sizeof(v));
	int last=0,ind=-1;
	vector<int> pr;
	for(int i=2;i<=2000001;i++){
		if(v[i])
		continue;
		pr.push_back(i);
		for(int j=2*i;j<=2000001;j+=i){
		    v[j]=true;
		}
	}
//	printf("%d\n",pr.size());
int t;
scanf("%d",&t);
while(t--){
	int n,m;
	scanf("%d %d",&n,&m);
	long long x=pr[pr[n-1]-1],y=pr[pr[m-1]-1];
	//printf("%lld %lld\n",x,y);
	printf("%lld\n",(x*y)%1000000007);
}
	return 0;
}
