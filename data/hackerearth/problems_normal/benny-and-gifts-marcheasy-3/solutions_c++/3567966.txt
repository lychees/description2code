#include "bits/stdc++.h"

using namespace std;
  map<pair<int,int>,int>t;
int main()
{
    string s;
    int x=0;
    int y=0;
    cin>>s;
    
    int ans=0;
  
 
    
    for(char c:s)
    {
    	
    	t[make_pair(x,y)]=1;
    	
    	if(c=='L')
    	y--;
    	if(c=='R')
    	y++;
    	if(c=='U')
    	x--;
    	if(c=='D')
    	x++;
    
    	ans=ans+t[make_pair(x,y)];
}
  
    
  cout<<ans;
    return 0;
}
