tc = int(raw_input())
while tc > 0:
    s = raw_input()
    s = list(s)
    l = len(s)
    i = 0
    j = 0
    s1 = s[:]
    while i < l-1:
        if s[i] == s[i+1]:
            s1.pop(j)
            i = i+1
        elif s[i] != s[i+1]:
            i = i+1
            j = j+1
    print ''.join(s1)
    tc = tc-1
