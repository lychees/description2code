#include <bits/stdc++.h>
using namespace std;
int ans[1000001];
int fac[1000001];
int a[1000001];
int get(int m)
{
	int ans =0;
	while(a[m]!=0)
	{
		ans++;
		m/=a[m];
	}
	return ans+1;
}

int main()
{
    for(int i=2;i<=1000;i++)
        if(!a[i])
            for(int j=i*i;j<=1000000;j+=i)
                a[j]=i;
    for(int i=2;i<=1000000;i++)
    	ans[i]=ans[i-1]+get(i);
    int n,t;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
        printf("%d\n",ans[n]);
    }
    return 0;
}
