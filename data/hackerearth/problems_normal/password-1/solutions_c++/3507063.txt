int main()
{
    ll n,i,f=0,l,j;
    scanll(n);
    string str,rev;
    char k;
    vector<string>v;
    fl(i,0,n)
    {
        cin>>str;
        v.PB(str);
    }
    fl(i,0,n-1)
    {
       rev=v[i];
       reverse(rev.begin(),rev.end());
       fl(j,i+1,n)

                l=rev.length();

            }
        }
    }
