#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    long long int t;
    cin>>t;
    while(t--)
    {
    	long long int n,i;
    	cin>>n;
    	if(n<4)
    	cout<<"4"<<endl;
    	else if((n%2)!=0)
    	cout<<(n+1)<<endl;
    	else
    	{
    		long long int m=n+1;
    		int flag=0;
    		for(i=2;i<=sqrt(m);i++)
    		{
    			if((m%i)==0)
    			{
    				flag=1;
    				break;
    			}
    		}
    		if(flag==1)
    		cout<<m<<endl;
    		else
    		cout<<(m+1)<<endl;
    	}
    	
    }
    return 0;
}
