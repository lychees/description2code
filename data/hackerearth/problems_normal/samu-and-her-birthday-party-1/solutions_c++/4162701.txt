    #include <iostream>
    #include <vector>
    #include <string>
     
    using namespace std;
     
    int setbits(int n) {
    	int count = 0;
    	while(n) {
    		++count;
    		n &= n - 1;
    	}
    	return count;
    }
     
    int main() {
    	string s;
    	int t, n, k;
     
    	cin >> t;
     
    	while(t--) {
    		cin >> n >> k;
    		vector<int> friends;
     
    		for(int i = 0; i < n; i++) {
    			cin >> s;
    			friends.push_back(stoi(s, nullptr, 2));
    		}
     
    		int dishes = k;
    		for(int sel = 0; sel < (1 << k); sel++) {
     
    			bool possible = true;
    			for(int i = 0; i < n; i++) {
    				if((friends[i] & sel) == 0) {
    					possible = false;
    					break;
    				}
    			}
     
    			if(possible) {
    				dishes = min(dishes, setbits(sel));
    			}
    		}
     
    		cout << dishes << endl;
    	}
    	return 0;
    }
