#include <iostream>
#include <vector>
#include <algorithm>
#include <stdio.h>
#include <string.h>

using namespace std;
int max(int a,int b)
{
	if(a > b)
		return a;
	else
		return b;
}
int main()
{
	int n,x,sum,startIndex;
	vector<int> v;
	int arr[15][1250];
	memset(arr,0,sizeof(int)*15*1250);
	cin>>n;
	sum = 0;
	for(int i=1;i<=n;i++)
	{
		cin>>x;
		v.push_back(x);
		sum += x;
	}
	sort(v.begin(),v.end());
	for(int i=0;i<n;i++)
	{
		arr[i+1][v[i]] = 1;
	}
	for(int i = 2;i <= n;i++)
	{
		for(int j=1;j<=1200;j++)
		{
			if(j < v[i])
				continue;
			arr[i][j] = max(arr[i-1][j],arr[i][j-v[i]]);
		}	
	}
	//check the last row.
	if(sum%3 == 0)
	{
		startIndex = sum/3;
	}
	else
	{
		startIndex = sum/3 + 1;
	}
	for(int i = startIndex;i<=sum;i++)
	{
		if(arr[n][i] == 1)
		{
			cout<<i<<endl;
			break;
		}
	}
	return 0;
}
