#include <iostream>
#include<vector>
using namespace std;

int main()
{
   vector<int> sg;
   sg.push_back(0);
   sg.push_back(1);
   sg.push_back(2);
   sg.push_back(3);
   sg.push_back(4);
   int t;
   cin>>t;
   while(t--)
   {
   	int n,i,ns=0;
   	cin>>n;
   	while(n--)
   	{
   		cin>>i;
   		ns^=sg[i%5];
   	}
   	if(ns==0)
   	cout<<"Andrew\n";
   	else
   	cout<<"Aniruddha\n";
   }
   
   return 0;
}
