#include <iostream>
using namespace std;

int main()
{
    long long n;
    int q,t;

    int i,j;

    {
    	cin>>n>>t;
    	for(j=0;j<t;j++)
    	{
    		if(n%2==0) n=n/2;

    	}

    }
      return 0;
}
