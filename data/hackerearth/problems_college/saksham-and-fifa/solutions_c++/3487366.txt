#include<stdio.h>
int main(){
	long long int n, t, p, mt, cp; scanf("%lld", &n);
	scanf("%lld%lld", &t, &p); mt = t, cp = p, n--;
	while(n-- && scanf("%lld%lld", &t, &p))	if(t > mt)	mt = t, cp = p;
	if(cp < 0)	printf("Right Corner\n");
		else if(cp > 0)	printf("Left Corner\n");
			else	printf("Either Corner\n");		
	return 0;
}
