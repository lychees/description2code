#include <iostream>
using namespace std;

int main()
{
    int t,n,i,o,e;
    long long int *a;
    
    scanf("%d",&t);
    while(t--){
    	scanf("%d",&n);
    	a = (long long int *)malloc(sizeof(long long int)*n);
    	for(i=0;i<n;i++){
    		scanf("%lld",&a[i]);
    	}
    	o=e=0;
    	for(i=0;i<n;i++){
    		if(a[i]%2==0){
    			e++;
    		}else{
    			o++;
    		}
    	}
    	printf("%lld\n",(o*e));
    }
    
    return 0;
}
