#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t[10001], i, tmp, tmp2;
    t[0] = 1;
    for (i = 1; i <= 2000; i++) {
        tmp = t[i-1]*3;
        while (tmp > 9) {
            tmp2 = 0;
            while (tmp) {
                tmp2 += tmp%10;
                tmp /= 10;
            }
            tmp = tmp2;
        }
        t[i] = tmp;
    }
    int tx, ans;
    cin >> tx;
    string s;
    while (tx--) {
        cin >> s;
        reverse(s.begin(), s.end());
        ans = 0;
        for (i = 0; i < s.size(); i++) {
            tmp = t[i]*(s[i]-48);
            ans += tmp;
        }
        tmp = ans;
        while (tmp > 9) {
            tmp2 = 0;
            while (tmp) {
                tmp2 += tmp%10;
                tmp /= 10;
            }
            tmp = tmp2;
        }
        ans = tmp;
        cout << ans << endl;
    }
    return 0;
}
