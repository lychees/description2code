#include <bits/stdc++.h>

using namespace std;

#define ln long long
deque<int> ax[1010];
ln a[10010];

void pre()
{
    ax[0].push_back(1);
    ax[1].push_back(1);
    ax[2].push_back(2);
    ax[3].push_back(6);

    int num, den;
    for (int i = 4; i < 1010; i++) {
    	num = 0;
    	den = i;

    	for (int j = 0; j < ax[i-1].size(); j++) {
    		den = (ax[i-1][j] * i + num) % 10;
    		num = (ax[i-1][j] * i + num) / 10;
    		
    		ax[i].push_back(den);
    	}

    	while (num > 0) {
    		ax[i].push_back(num % 10);
    		num /= 10;
    	}
    }
}

int main()
{
    pre();
   	int t;
   	cin >> t;

   	int s = 0;
   	for (int i = 0; i < t; i++) {
        int tmp;
   		cin >> tmp;
   		s = 0;
   		for (int i = 0; i < ax[tmp].size(); i++) {
                s += ax[tmp][i];
   		}
   		a[i] = s;
   	}

   	for (int i = 0; i < t; i++) {
        cout << a[i] << endl;
   	}

    return 0;
}
