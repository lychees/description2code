inline void readInt(int &num){
    num = 0;
    int c = getchar();
    int s = 1;
    if(c == '-'){
        s = -1;
        c = getchar();
    }
    while(c>='0'&&c<='9'){
        num = num*10+(c-'0');
        c = getchar();
    }
    num*=s;
}

}

inline void readString(string &s){
    s = "";
    char c = getchar();
    while(c!=' '&&c!='\n'&&c!=EOF){
        s+=c;
        c = getchar();
    }
}

inline void readLine(string & s){

}
