inline void printint(int a)
 {
	char s[11];
	int t = -1;
 
	do{
		s[++t] = a % 10 + '0';
		a /= 10;
	}while(a > 0);
 
	while(t >= 0)putchar_unlocked(s[t--]);
	putchar_unlocked('\n');
 }

int random_long(int digit=8)
{
   int len=digit;
   int x=0;
   for ( int i = 0; i < len; ++i )
   {
    int dig = rand() % 10;
    while ( x == 0 && dig == 0 )
      dig = rand() % 10;
    x = x * 10 + dig;
  }
   return x;
}

int random(int l, int r)
{
    int diff = r - l + 1;

    return l+random_long()%diff;

}
