#include<iostream>
#include<cstdio>
#include<algorithm>

using namespace std;

int main(){
	int t;
	scanf("%d",&t);
		while(t--){
			int n;
			scanf("%d",&n);
			int cap[n],amt[n],amtsum=0;
			for(int i=0;i<n;i++){
				scanf("%d",&amt[i]);
				amtsum+=amt[i];
			}
			
			for(int i=0;i<n;i++){
				scanf("%d",&cap[i]);
			}
			
			sort(cap,cap+n);
			int capsum=0,count=0;
			for(int i=n-1;i>=0;i--){
				capsum+=cap[i];
				count++;
				if(capsum>=amtsum)
					break;
			}
			
			printf("%d\n",count);
		}
	return 0;
}
