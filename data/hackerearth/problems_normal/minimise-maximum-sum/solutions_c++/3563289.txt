#include <bits/stdc++.h>
using namespace std;

int main()
{
int t,i,n,x,a[200000];
long long int l,max1;
scanf("%d",&t);
while(t--)
{
	max1=0;
	scanf("%d",&n);
	x=n;
	n=2*n;
	for(i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
	}

	sort(a,a+n);
	for(i=0;i<x;i++)
	{
	   // cout<<a[i]<<" "<<a[n-1-i]<<"\n";
		l=a[i]+a[n-i-1];
		//cout<<l<<"\n";
		if(l>max1)
		max1=l;
	}
cout<<max1<<"\n";

}

    return 0;
}
