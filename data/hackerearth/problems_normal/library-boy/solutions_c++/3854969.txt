#include <bits/stdc++.h>
using namespace std;
char s[101];
int c[26];
int main()
{
   int n,i;
   cin>>n;
   int co=0;
   for(i=0;i<n;i++)
   {
   	cin>>s;
   c[s[0]-97]++;
   	memset(s,'\0',sizeof(s));
   }
   for(i=0;i<26;i++)
   {
   	if(c[i]!=0)
   	co++;
   }
   for(i=0;i<26;i++)
   {
   	if(c[i]>10)
   	{
   		while(c[i]>10)
   		{co++;
   		c[i]=c[i]-10;
   		}
   	}
   }
   cout<<co;
}
