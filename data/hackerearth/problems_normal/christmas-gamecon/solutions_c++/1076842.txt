#include <iostream>
using namespace std;
long long arr[100010];
int main()
{
	std::ios_base::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		int n,i;
		cin>>n;
		for(i=0;i<n;i++)
		cin>>arr[i];
		for(i=0;i<n;i++)
		arr[i]=arr[i]%5;
		int answer=0;
		for(i=0;i<n;i++)
		answer=answer^arr[i];
		if(answer)
		cout<<"Aniruddha"<<endl;
		else
		cout<<"Andrew"<<endl;
	}
	return(0);
}
