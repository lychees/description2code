typedef long long int ll;
const int mx = 1e6 + 10;
using namespace std;

int n, v[mx];


int get(){
    int r = 0, l = 0;
    for(int i = 0 ; i < n; i++){
        r += v[i];
        if(r < 0)r = 0;
        if(l < r)l = r;
    }
    return l;
}
