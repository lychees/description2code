#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long n, i, j, x, k=0;
    cin>>n;
    string l;
    char a[100][100];
    cin.ignore();
    while(n--){
    	k=0;
    	getline(cin,l,'\n');
    	
    	x=pow(l.length(),0.5);
    	//cout<<x<<endl;
    	for(i=0;i<x;++i){
    		for(j=0;j<x;++j,++k)
    			a[i][j]=l[k];
    		
    		
    	}
    	
    	for(j=0;j<x;++j){
    		for(i=0;i<x;++i){
    			
    			cout<<a[i][j];
    		}
    		
    		
    	}
    	
    cout<<endl;	
    
    	
    }
    return 0;
}
