#include <iostream>
using namespace std;


int main(){
  int t;
  cin >> t;
  while(t--){
    int n;
    cin >> n;
    int ar[n];
    int count = 0;
    for(int i =0 ; i < n;i++){
      cin >> ar[i];
      if(ar[i]%2==1)count++;
    }
    if(count%2 == 0) cout << "Yes" << endl;
    else cout << "No" << endl;
  }
}
