#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
int main()
{
	ll t,x,y;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld %lld",&x,&y);
		
		if(x==y)
		{
			if(x%2==0)
		    cout<<x+y<<endl;
		    else
		    cout<<x+y-1<<endl;
		}
		else if(x==y+2)
		{
		 if(x%2==0)	
		 cout<<x+y<<endl;
		 else
		 cout<<x+y-1<<endl;
		}
		else 
		cout<<"No Number"<<endl;
	}
}
