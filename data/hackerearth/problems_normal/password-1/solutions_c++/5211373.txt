#include "bits/stdc++.h"
using namespace std;
string rev(string a)
{
	for(long int i=0,j=a.length()-1;i<j;i++,j--)
	{
		swap(a[i],a[j]);
	}
	return a;
}
int main()
{
	long int tc;
	cin >> tc;
	
	map<string,bool> mymap;
	mymap.clear();
	
	while(tc--)
	{
		string s;
		cin >> s;
		
	//	cout << mymap[s] << endl;
		if(mymap[s] == 1)
		{
			long int len=s.length();
			cout << len << " " << s[len/2] << endl;
		}
		s=rev(s);
		mymap[s]=1;
	}
	return 0;
}
