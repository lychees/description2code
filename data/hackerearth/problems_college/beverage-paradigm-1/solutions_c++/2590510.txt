#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,j,k,n;
	cin>>n;
	int a[n];
	for(i=0;i<n;i++)
	{
		cin>>a[i];
	}
	int num;
	cin>>num;
	int flag=0;
	for(i=0;i<n;i++)
	{
		for(j=i+1;j<n;j++)
		{
			for(k=j+1;k<n;k++)
			{
				if((a[i]+a[j]+a[k])==num)
				flag=1;
			}
		}
	}
	if(flag==1)
	cout<<"True"<<endl;
	else
	cout<<"False"<<endl;
	return 0;
}
