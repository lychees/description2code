def getNthFibboNum(first, second, seq_count):
    count = 2
    nextNum = 0
    bFound = False
    while not bFound:
        nextNum = first + second
        first = second
        second = nextNum
        count += 1
        if count == seq_count:
            bFound = True
    return nextNum


values = raw_input()
(first, second, seq_count) = values.split()
if seq_count == '1':
    print first
elif seq_count == '2':
    print second
else:
    print getNthFibboNum(int(first), int(second), int(seq_count))
