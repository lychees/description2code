using namespace std;
long int gcd(long int p,long int q)
{
	long int tp=1;
	while(q!=0)
	{
		tp=p%q;
		p=q;
		q=tp;
	}
	return p;
}
int main()
{
    int t,n,i;
	long long int res;
	scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
        int *a=new int[n+5];
    	for(i=0;i<n;i++)
    	  scanf("%d",&a[i]);
        res=a[0];
		for(i=1;i<n;i++)
	        res=gcd(res,a[i]);
	    res=res*n;
		cout<<res<<endl;    
	}
	return 0;
}
