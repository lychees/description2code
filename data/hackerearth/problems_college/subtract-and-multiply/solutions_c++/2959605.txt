int binpow(int q, int w, int mod) {
	if (!w)
		return 1;
	if (w & 1)
		return q * binpow(q, w - 1, mod) % mod;
	return binpow(q * q % mod, w / 2, mod);
}

void solve() {
	int d, p;
	cin >> d >> p;

	if (d < 0) {
		cout << "0\n";
		return;
	}

	int D = d * d + 4 * p;
	int cur = int(sqrt((ld)D) + 1e-9);

	if (cur * cur != D) {
		cout << "0\n";
		return;
	}

	if ((d + cur) & 1) {
		cout << "0\n";
		return;
	}

		if (d != 0) {

		}

	}

}
