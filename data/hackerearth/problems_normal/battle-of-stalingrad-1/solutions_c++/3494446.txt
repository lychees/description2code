#include<bits/stdc++.h>
using namespace std;

int Area( int x1, int y1, int x2, int y2, int x3, int y3 )
{
   return abs ( x1*( y2-y3 ) + x2*( y3-y1 ) + x3*( y1-y2 ) );
}

bool check ( int x1, int y1, int x2, int y2, int x3, int y3, int x4, int y4 )
{
   int area = Area( x1, y1, x2, y2, x3, y3 );
   int area1 = Area( x4, y4, x2, y2, x3, y3 );
   int area2 = Area( x1, y1, x4, y4, x3, y3 );
   int area3 = Area( x1, y1, x2, y2, x4, y4 );

   return ( area == area1 + area2 + area3 );
}

int main()
{
   int t;
   cin >> t;
   while ( t-- )
   {
      int x1, y1, x2, y2, x3, y3, x4, y4;
      cin >> x1 >> y1 >> x2 >> y2 >> x3 >> y3 >> x4 >> y4;

      if ( check ( x1, y1, x2, y2, x3, y3, x4, y4 ) )
      {
         cout << "INSIDE" << endl;
      }
      else
         cout << "OUTSIDE" << endl;
   }
   return 0;
}
