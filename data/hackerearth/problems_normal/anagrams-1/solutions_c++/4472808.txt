#include <iostream>
#include <ctype.h>
using namespace std;
void anagrams () {
	int tt;
	cin >> tt;
	for (int i = 0; i < tt; i++) {
		string s, t;
		getline (cin, s);
		if (s.length() == 0) getline (cin, s);
		getline (cin, t);
		int a[26];
		for (int i = 0; i < 26; i++) a[i] = 0;
		for (int i = 0; i < s.length(); i++) {
			int in = tolower(s[i]) - 'a';
			if (in >= 0 && in < 26) {
				a[in] ++;
			}
		}
		for (int i = 0; i < t.length(); i++) {
			int in = tolower(t[i]) - 'a';
			if (in >= 0 && in < 26) {
				a[in] --;
			}
		}
		int oc = 0, oi = 0, ec = 0;
		bool pos = true;
		for (int i = 0; i < 26; i++) {
			a[i] = abs (a[i]);
			if (a[i] != 0 && a[i] % 2 == 0) ec += a[i];
			else if (a[i] % 2 != 0) {
				oc++;
				ec += (a[i] / 2) * 2;
				oi = i;
				if (oc > 1) { 
					pos = false;
					break;
				}
			}
		}
		if (pos) {
			int len = ec + oc;
			if (len == 0) {
				cout << "YES" << '\n';
			}
			else {
			string res (len, ' ');
			if (oc == 1) {
				res[len / 2] = oi + 'a';
				a[oi]--;
				int ct = 0, i = 0;
				while (true) {
					while (ct < 26 && a[ct] == 0) {
						ct++;
						continue;
					}
					if (ct >= 26) break;
					if (i == len / 2) {
						i++;
					}
					res[i] = ct + 'a';
					res[len - 1 - i] = ct + 'a';
					a[ct] -= 2;
					i++;
				}
			}
			else {
				int ct = 0, i = 0;
				while (true) {
					while (ct < 26 && a[ct] == 0) {
						ct++;
						continue;
					}
					if (ct >= 26) break;
					res[i] = ct + 'a';
					res[len - 1 - i] = ct + 'a';
					a[ct] -= 2;
					i++;
				}
			}
			for (int i = 0; i < len; i++) {
				cout << res[i];
			}
			cout << '\n';
		}
		}
		else {
			cout << "NO LUCK" << '\n';
		}
	}
}
int main()
{
    anagrams ();
    return 0;
}
