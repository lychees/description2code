def thesavior():
    T = int(raw_input())
    outputlist = []
    oappend = outputlist.append
    for i in range(T):
        listlen = int(raw_input())
        powers = map(int, raw_input().split())

        count = 0
        i = 0
        while (i < listlen - 1):
            j = i + 1
            while (j < listlen):
                if powers[i] != powers[j] and (powers[i] + powers[j]) % 2 == 0:
                    count += 1
                j += 1
            i += 1
        oappend(count)

    for key in outputlist:
        print key


thesavior()
