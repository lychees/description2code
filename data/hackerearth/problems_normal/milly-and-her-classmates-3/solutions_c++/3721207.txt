#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
	int test;
	cin >> test;
    while(test--){
    	int n;
    	cin >> n;
    	vector<int> ans(n, 0);
    	for(int i = 0;i < n;++i){
    		cin >> ans[i];
    	}
    	int i;
    	for(i = 1;i < n;++i){
    		if(ans[i] < ans[i - 1])break;
    	}
    	if(i == n){
    		cout << "-1 -1\n";
    		continue;
    	}
    	int j;
    	for(j = i;j < n;++j){
    		if(ans[j] > ans[j - 1])break;
    	}
    	reverse(ans.begin() + i - 1, ans.begin() + j);
    	if(is_sorted(ans.begin(), ans.end())){
    		cout << i << " "<< j << "\n";
    	}
    	else cout << "-1 -1\n";
    }
    return 0;
}
