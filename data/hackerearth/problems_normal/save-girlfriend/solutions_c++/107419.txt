#include<bits/stdc++.h>
#define ll long long
#define MAX 10009

using namespace std;

#ifndef ONLINE_JUDGE
    #define gc getchar
#else
    #define gc getchar_unlocked
#endif

ll read() {
  ll ret=0;
  char c=gc();
  while(c<'0'||c>'9') c=gc();
  while(c>='0'&&c<='9') {
    ret=10*ret+c-48;
    c=gc();
  }
  return ret;
}

bool isprime[MAX+4];
int dp[MAX+4];

void sieve() {
    isprime[0]=isprime[1]=1;

    for(int i=4;i<=MAX;i=i+2)
        isprime[i]=1;

    int x=sqrt(MAX);


    for(int i=3;i<=x;i+=2) {
        if(!isprime[i]) {
            for(int j=i*i;j<=MAX;j+=(i<<1))
                isprime[j]=true;
        }
    }


}


int preprocess(ll l,ll r) {
    int ub = sqrt(r);

    for(ll i=2;i<=ub;i++)
     {

        ll start = max(l,i*i);


        int rem = start%i;

        if(rem>0)
            start= start + (i-rem);
        else if((i*i)==start)
            dp[start-l]--;

        for(ll j=start;j<=r;j+=i){
            dp[j-l]+=2;}

        }

        int limit=r-l,res=0;
        int m=-1;
        int b = 1;

        for(int i=0;i<=limit;i++) {
            dp[i]+=2;
            m = max(m,dp[i]);


        }


         int count =0;
        for(int i=0;i<=limit;i++)
        {

            if(m==dp[i])
            count++;


        }


    return count;
}


int main() {
    sieve();
    int t;
    scanf("%d",&t);
    while(t--) {
        ll a,b;
        a=read(),b=read();
        memset(dp,0,sizeof(dp));
        printf("%d\n",preprocess(a,b)+1);
    }
    return 0;
}
