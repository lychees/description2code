#include <bits/stdc++.h>

using namespace std;

int main(){
    int n;
    scanf("%d",&n);
    while(n--){
        int t,i,r;
        set <int> s;
        scanf("%d",&t);
        for(i=0;i<t;i++){
            scanf("%d",&r);
            s.insert(r);
        }
        printf("%d\n",s.size());
    }
    return 0;
}
