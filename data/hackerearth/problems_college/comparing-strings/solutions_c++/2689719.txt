#include <bits/stdc++.h>
using namespace std;
 
int main()
{
   int t;
	cin>>t;
	while(t--)
	{

		string s1, s2;
		cin>>s1>>s2;
		int c1[26]={0};
		int c2[26]={0};
		for(int i=0; i<s1.size(); i++)c1[s1[i]-97]++;
		for(int i=0; i<s2.size(); i++)c2[s2[i]-97]++;
		int flag = 1;
		for(int i=0; i<26; i++)
			if(c2[i]>0 && c1[i]>0){flag = 0; break;}

	}
