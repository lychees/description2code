#include <bits/stdc++.h>
using namespace std;
int solve(string s)
{
      if(s[0] != 'R') return 0;
      int sz = s.size();
      for(int i=1;i<sz;i++)
      {
            if(s[i]!='R' && s[i]!='Y') return 0;
            if(i>2 && s[i] == 'Y' && s[i-1] == 'Y' && s[i-2] == 'Y')
            return 0;
      }
      return 1;
}
int main()
{
    string s;
    int t;
    cin>>t;
    while(t--)
    {
    	cin>>s;
    	if(solve(s))
    		cout<<"YES\n";
    	else
    		cout<<"NO\n";
    }
    return 0;
}
