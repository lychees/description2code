#include <iostream>
using namespace std;
#include<cmath>
#include<map>
#define long long long
map<long, long> F;
long f(long n)
{
  long a = 2, b = 3, c, i;
  if( n == 0)
    return a;
  for (i = 2; i <= n; i++)
  {
     c = a + b;
     a = b;
     b = c;
  }
  return b;
}

int main()
{	int t;
    F[0]=2;
    F[1]=3;
    cin>>t;
    while(t--){
    	long n;
    	cin>>n;
    	long k=f(n-1);
    	cout<<k*k<<"\n";

    }
    return 0;
}
