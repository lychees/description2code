#include <iostream>
#include <stdio.h>
using namespace std;

int gcd (int a, int b){
	int c;
	while ( a != 0 ) {
		c = a; a = b%a;  b = c;
  	}
  	return b;
}

int main() {
	// your code goes here
	int t;
	scanf("%d", &t);
	while(t--){
		int n;
		scanf("%d", &n);
		int a[n];
		for(int i=0;i<n;i++){
			a[i]=0;
		}
		int max=0, count=0, c=0;
		bool flag1=false;
		for(int i=0;i<n;i++){
			int temp;
			scanf("%d", &temp);
			if(i==n-1){
				if(temp!=-1){
					count++;
					a[c++]=count;
				}
				
			}
			if(temp==-1){
				a[c++]=count;
				flag1=true;
				count=0;
			}
			else if(temp==1){
				count++;
			}
		}
		for(int i=0;i<n-1;i++){
			if(a[i]+a[i+1]>max){
				max=a[i]+a[i+1];
			}
		}
		if(!flag1){
			printf("0/1\n");
		}
		else{
			printf("%d/1\n", max);
		}
	}
	return 0;
}
