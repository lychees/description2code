#include <iostream>
#include <stdlib.h>
#include <math.h>
using namespace std;

int main()
{
	int t;
	string s;
	scanf("%d",&t);
	while (t--)
	{
		cin >> s;
		int flag=0;
		for(int i=0,j=s.size()-1;i<s.size()/2;i++,j--)
		{
			if(abs(s[i+1]-s[i])!=abs(s[j]-s[j-1]))
			{
				flag=1;
				break;
			}
		}
		if(flag)
			printf("Not Funny\n");
		else
			printf("Funny\n");
	}
    return 0;
}
