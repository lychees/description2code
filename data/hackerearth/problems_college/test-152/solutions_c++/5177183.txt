#include<bits/stdc++.h>
using namespace std;

map<int,int>mp;

int main()
{
    int i,j,k,len,t,N,mx;
    string str;
    cin>>t;
    while(t--)
    {
        cin>>len;
        cin>>str;
        mx=-1;
        for(i=0;i<len;i++){
    		j=str[i]-'a';
    		mp[j]++;
    		if(mp[j]>mx) mx=mp[j];
    	}
    	//cout<<mx<<endl;
    	if(mx%2==0) cout<<"Yes"<<endl;
    	else cout<<"No"<<endl;

        mp.clear();

    }
    return 0;
}
