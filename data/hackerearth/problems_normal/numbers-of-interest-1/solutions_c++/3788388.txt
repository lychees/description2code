#include<iostream>
using namespace std;
int main()
{
    int i,j,x,y,n,t;
    long long int a[50];
    cin>>t;
    while(t>0){
    cin>>x>>y>>n;
    for(i=1;i<=n;i++)
    {
        if(i<=x)
        {
            a[i]=y;
        }
        else
        {
            a[i]=0;
            for(j=i-x;j!=i;j++)
            {
                a[i]+=a[j];
            }
        }
    }
    cout<<a[n];
    cout<<"\n";
    t--;
    }
    return 0;
}
