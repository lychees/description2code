#include <bits/stdc++.h>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
    int T;
    cin>>T;
    while(T--)
    {
    	set<string> st;
    	string s;
    	int n;
    	cin>>n;
    	for(int i=0;i<n;i++)
    	{
    		cin>>s;
    		st.insert(s);
    	}
    	for(auto i : st)
    	cout<<i<<"\n";
    }
    return 0;
}
