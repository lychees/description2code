sys_gend_no = str(input())
user_input_no = str(input())


MAX_DIFF_DIGITS = 1

if sys_gend_no == user_input_no:
    print 'wins'
elif len(sys_gend_no) == len(user_input_no):
    mismatch_counter = 0
    for i in range(len(sys_gend_no)):
        if not (sys_gend_no[i] == user_input_no[i]):
            if mismatch_counter < MAX_DIFF_DIGITS:
                mismatch_counter += 1
            else:
                print 'loses'
                break
    else:
        print 'wins'
else:
    print 'loses'
