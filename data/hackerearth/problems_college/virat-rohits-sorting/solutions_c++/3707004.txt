#include <bits/stdc++.h>
using namespace std;
vector<int>arr[1000007];
set<int>S;
int main()
{
    //cout << "Hello World!" << endl;
    int n;
    cin>>n;
    for(int i=1;i<=n;i++)
    {
    	int x , y ;
    	cin >> x >> y;
    	S.insert(x);
    	arr[x].push_back(y);
    }
    for(auto it:S){
    	sort(arr[it].begin(),arr[it].end());
    	reverse(arr[it].begin(),arr[it].end());
    	for(auto X : arr[it]){
    		cout << it << " " << X << '\n';
    	}
    }
    return 0;
}
