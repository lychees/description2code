#include <iostream>
#include<string.h>

{

		scanf("%lld",&a[i]);
		if(abs(a[i]-a[i-1])>dom) dom=((a[i]-a[i-1])>0)?(a[i]-a[i-1]):(a[i-1]-a[i]);
	}

	{
		scanf("%lld",&b[i]);
