#include<bits/stdc++.h>
using namespace std;
int n,i,j,m,k,h;
void print(int n)
{
	for(i=1;i<=n;i++)
	{
		k=n-i+1;
		m=i;
		for(j=1;j<=n;j++)
		{
			if(j!=k)
				printf(" ");
			else
			{
				printf("/");
				k=k+2;
			}				
		}
		if(i%2==0)
			printf(" ");
		for(k=i;k>=1;k--)
		{
			if(k!=m)
				printf(" ");
			else
			{
				printf("\\");
				m=m-2;
			}
		}
		printf("\n");
	}
	h=n;
	for(i=1;i<=n;i++)
	{
		k=i+2;m=(h%2)?1:2;
		for(j=1;j<=n+2;j++)
		{
			if(j!=k)
				printf(" ");
			else
			{
				printf("\\");
				k=k+2;
			}
		}
		for(k=1;k<=n-i+1;k++)
		{
			if(k!=m)
				printf(" ");
			else
			{
				printf("/");
				m=m+2;
			}
		}
		h--;
		printf("\n");
	}
}
int main()
{
	scanf("%d",&n);
	print(n);
	print(n+1);
	return 0;
}
