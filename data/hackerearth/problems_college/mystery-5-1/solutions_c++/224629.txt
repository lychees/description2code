#include <bits/stdc++.h>

using namespace std;

#define LL long long int 
#define sd(x) scanf("%d", &x)
#define F first
#define S second
#define mp make_pair
#define pb push_back
#define N 2000000
#define pii pair<int,int>

char in[N];
string s;
int c[26];

int main(){
	int t, i, j, k, l;
	sd(t);
	while(t--){
		scanf("%s", in);
		s = in;
		l = s.length();
		for(i = 0; i < 26; i++){
			c[i] = 0;
		}
		i = 0;
		while(i < l){
			c[s[i] - 'a']++;
			i++;
		}
		k = 1;
		for(i = 0; i < 26; i++){
			k = max(k, c[i]);
		}
		printf("%d\n", k);
	}
	return 0;
}
