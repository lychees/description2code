#include <iostream>
using namespace std;
struct point{
	int x;
	int y;
};
int main()
{
	int t; cin >> t;
	while(t--)
	{
		int a,b,c,d; cin >> a  >> b >> c >> d;
		int e,f,g,h; cin >> e  >> f >> g >> h;
		point P1,P2,P3,P4;
		P1.x = a; P1.y = b; P2.x = c; P2.y = d;
		P3.x = e; P3.y = f; P4.x = g; P4.y = h;
		if(P1.x>P3.x)
        {
            swap(P1,P3); swap(P2,P4);
        }
		if(P2.x<P3.x || P4.x<P1.x || P2.y<P3.y || P4.y<P1.y)
			cout << "Apart";
		else if(P2.x!=P3.x && P4.x!=P1.x && P2.y!=P3.y && P4.y!=P1.y)
			cout << "Rectangle";
		else
		{
			if((P2.x==P3.x && P2.y==P3.y) || (P2.x==P3.x && P1.y==P4.y))
			{
				if(P4.x>P2.x)
					cout << "Point";
				else
					cout << "Rectangle";
			}
			else if((P1.x==P4.x && P2.y==P3.y) || (P1.x==P4.x && P1.y==P4.y))
			{
				if(P4.x<P1.x)
					cout << "Point";
				else
					cout << "Rectangle";
			}
			else
				cout << "Line";
		}
		cout << endl;
	}
    return 0;
}
