#include <iostream>
using namespace std;

int main()
{
    int ite;
    cin>>ite;
    while(ite--){
    	long long n,m;
    	cin>>n>>m;
    	long long sum=0;
    	if(m>n){
    		cout<<((n*(n+1))/2ll)<<endl;
    	}
    	else{
    		long long k=n/m;
    		sum = ((m*(m-1))/2ll)*k;
    		k = n%m;
    		sum+= ((k*(k+1))/2ll);
    		cout<<sum<<endl;
    	}
    }
    return 0;
}
