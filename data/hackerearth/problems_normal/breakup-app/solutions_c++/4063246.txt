#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,day,i,j,a[31],flag;
	cin>>n;
	memset(a,0,sizeof(a));
	while(n--)
	{
		//cout<<endl<<"for pass j="<<j<<endl;
		flag=0;
		string str;
		str.clear();
		fflush(stdin);
		getline(cin,str);
		int len=str.size();
		//cout<<"strlen is "<<len<<endl;
		if(str[0]=='G')
		{
			flag=1;
		}
		i=0;
		string temp;
		temp.clear();
		while(i<len)
		{
			if(isdigit(str[i])==true)
			{
				while(isdigit(str[i])==true && i<len)
				{
					temp=temp+str[i];
					i++;
				}
				int val;
				val=atoi(temp.c_str());
				//cout<<val<<endl;
				if(flag==1)
				{
					a[val]+=2;
				}
				else
				{
					a[val]++;
				}
				temp.clear();
			}
			i++;
		}
		
	}
	/*for(i=1;i<31;i++)
	{
		cout<<a[i]<<" ";
	}
	cout<<endl;*/
	int max_wgt=-1,idx;
	bool m_wgt=1;
	for(i=1;i<31;i++)
	{
		if(a[i]>max_wgt)
		{
			max_wgt=a[i];
			idx=i;
			m_wgt=1;
		}
		else if(a[i]==max_wgt)
		{
			m_wgt=false;
		}
	}
	if(idx==19 || idx==20)
	{
		if(m_wgt==1)
		{
			cout<<"Date"<<endl;
		}
		else
		{
			cout<<"No Date"<<endl;
		}
	}
	else
	{
		cout<<"No Date"<<endl;	
	}
	
}
