#include<iostream>
#include<cstdio>
using namespace std;
#define M 1000001
int p[M];
void s()
{
    for(int i=2;i<M;i++)
    {
        if(p[i]==0)
        {
            for(int j=2*i;j<M;j=j+i)
                p[j]=1;
        }
    }
    p[4]=0;
    p[1]=1;
   for(int i=1;i<M;i++)
            {
                p[i]=p[i]+p[i-1];
            } 
            return ;
}
int main()
{
    int t;
    scanf("%d",&t);
    s();
    while(t--)
        {
            int l,r;
            scanf("%d%d",&l,&r);
            int ans=p[r]-p[l-1];
            printf("%d\n",ans);
        }
        return 0;
}
