#include <bits/stdc++.h>
using namespace std;

int main()
{
    int test;
    cin>>test;
    while(test--){
    	int n,k;
    	cin>>n>>k;
    	long long int arr[n];
    	for(int i=0;i<n;i++)
    		cin>>arr[i];
    	long long int dp[n];
    	long long int maxdp[n];
    	long long int amax = -1000000001;
    	for(int i=0;i<=k;i++){
    		long long int zero = 0;
    		dp[i] = max(zero,arr[i]);
    		amax = dp[i]>amax?dp[i]:amax;
    		maxdp[i] = amax;
    	}
    	for(int j=k+1;j<n;j++){
    		long long int maxn = maxdp[j-k-1];
    	//	for(int i=j-k-2;i>=0;i--){
    	//		if(dp[i]>maxn)
    	//			maxn = dp[i];
    	//	}
    		
    		long long int max1 = max(arr[j],max(maxn,maxn + arr[j]));
    		dp[j] = max1;
    		amax = dp[j]>amax?dp[j]:amax;
    		maxdp[j] = amax;
    	}
    cout<<amax<<endl;	
    }
    return 0;
}
