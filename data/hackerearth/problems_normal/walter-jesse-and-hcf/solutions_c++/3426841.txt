#include <bits/stdc++.h>
#include<algorithm>

using namespace std;

int main()
{
    long long int t;
    cin>>t;
    while(t--){
        long long int x,y,k,z=0;
        cin>>x>>y>>k;
        if(x>y){
        	
        	swap(x,y);
        }
        long long int val = __gcd(x,y);
        for(int i=val;i>=1;i--){
            if(x%i==0 && y%i==0){
             	k--;
             	if(k>0){
                	continue;
             	}
             	else{
             		cout<<i<<endl;
             		z =1;
             		break;
             	}
             	
            }
        }
       	if(z==0){
            cout<<"No crymeth today"<<endl;
        }

}
    //cout << "Hello world!" << endl;
    return 0;
}
