#include<iostream>
using namespace std;

int arr[1075][1075];
int dp[1075][1075];

int m, n;

int chalo(int r, int c)
{
    dp[r][c] = 1;

    long int ctr = 1;

    if(r+1<m)
        if(arr[r+1][c]==1 && dp[r+1][c]==-1)   //down
            ctr += chalo(r+1, c);

    if(c+1<n)
        if(arr[r][c+1]==1 && dp[r][c+1]==-1)   //right
            ctr += chalo(r, c+1);

    if(c-1>=0)
        if(arr[r][c-1]==1 && dp[r][c-1]==-1)   //left
            ctr += chalo(r, c-1);

    if(r+1<m && c+1<n)
        if(arr[r+1][c+1]==1 && dp[r+1][c+1]==-1)   //right down
            ctr += chalo(r+1, c+1);

    if(r+1<m && c-1>=0)
        if(arr[r+1][c-1]==1 && dp[r+1][c-1]==-1)     //left down
            ctr += chalo(r+1, c-1);

    if(r-1>=0 && c-1>=0)
        if(arr[r-1][c-1]==1 && dp[r-1][c-1]==-1)     //left up
            ctr += chalo(r-1, c-1);

    if(r-1>=0 && c+1<n)
        if(arr[r-1][c+1]==1 && dp[r-1][c+1]==-1)     //right up
            ctr += chalo(r-1, c+1);

    if(r-1>=0)
        if(arr[r-1][c]==1 && dp[r-1][c]==-1)     //up
            ctr += chalo(r-1, c);

    return ctr;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        for(int i=0; i<1075; i++)
            for(int j=0; j<1075; j++)
                dp[i][j] = -1;

        cin>>m>>n;

        for(int i=0; i<m; i++)
            for(int j=0; j<n; j++)
                cin>>arr[i][j];

        long int ans1 = 0, ans2 = 0;
        for(int i=0; i<m; i++)
            for(int j=0; j<n; j++)
            {
                long int c = 0;
                if(arr[i][j]==1 && dp[i][j]==-1)
                {
                    c = chalo(i, j);
                    ans1++;
                }
                if(c > ans2)
                    ans2 = c;
            }

        cout<<ans1<<" "<<ans2<<endl;
    }

    return 0;
}
