from math import *

for i in range(input()):
	count = 0
	x, y = map(int, raw_input().split())
	for j in range(x, y+1) :
		r1 = sqrt(5*j**2+4)
		r2 = sqrt(5*j**2-4)
		isSquare = r1%1 == 0 or r2%1 == 0
		if(isSquare) :
			count = count + 1
	if(x == 1) :
		count = count + 1
	if(count%2 == 0) :
		print "VALID"
	else :
		print "INVALID"
