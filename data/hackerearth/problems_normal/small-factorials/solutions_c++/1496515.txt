#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int i,j,n,k=0,carry=0,a[1000]={1};
		cin>>n;
		for(i=1;i<=n;i++)
		{
			for(j=0;j<=k;j++)
			{
				a[j]=a[j]*i+carry;
				carry=a[j]/10;
				a[j]=a[j]%10;
			}
			while(carry)
			{
				k++;
				a[k]=carry%10;
				carry=carry/10;
			}
		}
		for(i=k;i>=0;i--)
			cout<<a[i];
		cout<<endl;
	}
	return 0;
}
