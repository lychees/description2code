#include <bits/stdc++.h>
using namespace std;

int main()
{
	string alpha,inp;	int t,i,A[129];
	cin>>t;
	while(t--)
	{
    	cin>>alpha;	cin>>inp;
    	int s=inp.size();	memset(A,0,sizeof(A));
    	for(int i=0;i<s;i++)
    		A[inp[i]]+=1;
    	for(int i=0;i<26;i++)
    	{
    		if(A[alpha[i]])
    		{
    			while(A[alpha[i]])
    			{
    				cout<<alpha[i];	A[alpha[i]]-=1;
    			}
    		}
    	}
    	cout<<'\n';
	}
	return 0;
}
