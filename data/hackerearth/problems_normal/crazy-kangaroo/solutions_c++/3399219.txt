#include <iostream>
#define ll long long int
using namespace std;


int main()
{
    ll t;
    cin >> t;
    while(t--)
    {
    	ll a,b,m;
    	cin >> a >> b >> m;
    	a=(a%m==0)?a:(a +(m-(a%m)));
    	b=(b%m==0)?b:(b-(b%m));
    	if(b>a)
    	{
    		cout << (b-a)/m + 1 << endl;
    	}
    	
    	else if(a==b)
    	{
    		cout << 1 << endl;
    	}
    	else
    	{
    		cout << 0 << endl;
    	}
    }
    return 0;
}
