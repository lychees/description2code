#include <bits/stdc++.h>
#include <iostream>
using namespace std;

int main()
{
    // cout << "Hello World!" << endl;
    int n;
    cin >> n;
    vector <int> v;
    int num;
    for(int i = 0;i < n;i++){
    	cin >> num;
    	v.push_back(num);
    }
    sort(v.begin(),v.end());
    // for(int i = 0;i < n;i++)cout << v[i] << " ";
    int ans1 = 0;
    int ans2 = 0;
    for(int i = 0;i<n/2;i++)
    {
    	ans1 += v[n-i-1]-v[i] + v[n-i-1]-v[i+1];
    	ans2 += v[n-i-1]-v[i] + v[n-i-2] - v[i];
    }
    	
    cout << max(ans1,ans2) << endl;
    return 0;
}
