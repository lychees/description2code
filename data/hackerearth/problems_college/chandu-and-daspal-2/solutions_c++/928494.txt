#include<bits/stdc++.h>
using namespace std;
int main()
{
      int t;
      cin>>t;
      int x,y;
      string s;
      while(t--)
      {
            cin>>x>>y>>s;
            int c=0;
            string str;
            if(s=="Chandu")
                  str="Daspal";
            else
                  str="Chandu";
            while(x!=0&&y!=0)
            {
                  int a=__gcd(x,y);
                  x=x-a;
                  y=y-a;
                  c++;
            }
            if(c%2)
                  cout<<s<<endl;
            else
                  cout<<str<<endl;
      }
      return 0;
}
