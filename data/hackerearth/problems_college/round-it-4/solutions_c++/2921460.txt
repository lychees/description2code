#include <bits/stdc++.h>
using namespace std;
# define ll long long
int main()
{
	cin.sync_with_stdio(false);
	int t;
	ll a,temp,b,temp1;
	cin>>t;
	while(t--)
	{
		cin>>a>>b;
		temp=a/b;
		temp1=temp+1;
		if(a-(b*temp)<(b*temp1)-a)
		cout<<b*temp<<endl;
		else
		cout<<b*temp1<<endl;
	}
	return 0;
}
