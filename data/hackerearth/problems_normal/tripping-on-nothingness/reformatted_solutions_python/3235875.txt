from math import sqrt
from pprint import pprint as pp


def GI():
    return int(raw_input())


def GIS():
    return map(int, raw_input().split())


def main():
    S, Q = GIS()
    factors = []
    for f in xrange(1, int(sqrt(S))+1):
        if not S % f:
            factors.append(f)
            factors.append(S/f)
    factors.sort(reverse=True)
    seen = set()
    for q in xrange(Q):
        a = GI()
        for f in factors:
            if not a % f:
                if f in seen:
                    print -1
                else:
                    print f
                    seen.add(f)
                break


main()
