#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t,n,i,j,elem;
    int a=-1,b=-1;
    long long max,min;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	vector <int> vec1;
    	while(n--)
    	{
    		cin>>elem;
    		vec1.push_back(elem);
    	}
    	//vector <int> vec2(vec1);
    	//sort(vec2.begin(),vec2.end());
    	i=0;
    	while(i<vec1.size()-1){
    		if(vec1[i]>vec1[i+1])
    		{
    			//cout<<"iiii="<<i<<endl;
    			break;
    		}
    		//cout<<"iiii="<<i<<endl;
    		i++;
    	}
    	//cout<<"i="<<i<<endl;
    	if(i==vec1.size()-1)
    	{
    		cout<<a<<" "<<b<<endl;
    		continue;
    	}
    	for(j=vec1.size()-1;j>0;j--)
    	{
    		if(vec1[j]<vec1[j-1])
    		{
    			break;
    		}
    	}
    	max=vec1[i];
    	min=vec1[i];
    	for(int k=i+1;k<=j;k++)
    	{
    		if(vec1[k]>max)
    			max=vec1[k];
    		if(vec1[k]<min)
    			min=vec1[k];
    	}
    	int k;
    	for(k=0;k<i;k++)
    	{
    		if(vec1[k]>min)
    		{
    			break;
    		}
    	}
    	if(k != i){
    		cout<<a<<" "<<b<<endl;
    		continue;
    	}
    	for(k=vec1.size()-1;k>=j+1;k--)
    	{
    		if(vec1[k]<max)
    		{
    			break;
    		}
    	}
    //	cout<<"k="<<k<<"j="<<j<<endl;
    	if(k != j){
    		cout<<a<<" "<<b<<endl;
    		continue;
    	}
    	for(k=i;k<j;k++)
    	{
    		if(vec1[k]<vec1[k+1])
    		{
    			break;
    		}
    	}
    	if(k != j){
    		cout<<a<<" "<<b<<endl;
     	}
     	else{
     		cout<<i+1<<" "<<j+1<<endl;
     	}
    
    	
    }
    
    return 0;
}
