#include <cstdio>
#define MOD 1000000007
using namespace std;

typedef long long ll;

ll power(ll b, ll e)
{
    ll x=1;
    while(e)
    {
        if(e%2)
            x=(x*b)%MOD;
        e/=2;
        b=(b*b)%MOD;
    }
    return x;
}

int main()
{
    ll T, N, ans;
    scanf("%lld", &T);
    while(T--)
    {
    scanf(" %lld", &N);    
    ans=power(2, N-1);
    ans=power(ans, MOD-2);
    printf("%lld\n", ans);
    }
    return 0;
}
