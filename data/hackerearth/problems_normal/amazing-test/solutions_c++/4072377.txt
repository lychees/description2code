#include<bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
	ll t;
	scanf("%lld",&t);
	while(t--)
	{
		ll n,x;
		scanf("%lld %lld",&n,&x);
		ll i;
		priority_queue<ll> arr;
		ll val,temp,total=0;
		for(i=0; i<n; i++)
		{
			scanf("%lld",&val);
			arr.push(val);
		}
		arr.push(0);
		while(1)
		{
			val=arr.top();
			if(val==0)
			{
				break;
			}
			arr.pop();
			temp=arr.top();
			if(temp==0)
			{
				total+=val;
				break;
			}
			arr.pop();
			total+=temp;
			val-=temp;
			arr.push(val);
		}
		if(total<=x)
		{
			printf("YES\n");
		}
		else
		{
			printf("NO\n");
		}
	}
	return 0;
}
