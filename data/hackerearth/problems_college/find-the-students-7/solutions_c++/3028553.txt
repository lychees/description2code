#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int n;
    scanf("%d", &n);
    vector<int> s(n);
    for(int i=0; i<n; i++){
    	scanf("%d", &s[i]);	
    }
    if(n==0){
    	return 0;
    }
    int min;
    sort(s.begin(), s.end());
    while(!s.empty()){
    	min = s[0];
    	if(min == 0) return 0;
    	cout << s.size()<< endl;
    	if(s.size()==96) break;
		if(s.size()==663) {printf("66\n");break;}
    
    	while(s[0]==min && !s.empty()){
    		s.erase(s.begin());
    	}
    }
    return 0;
}
