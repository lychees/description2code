#include <iostream>
using namespace std;
int main()
{
	string s;
	cin>>s;
	int t;
	cin>>t;
	int anton=0;
	int synon=0;
	for(int i=0;i<t;i++)
	{
		string dummy;
		cin>>dummy;
		if(dummy[0]==s[0]&&dummy.size()==s.size())
		synon++;
		if(dummy[0]-s[0]==3&&dummy.size()==s.size())
		anton++;
	}
	cout<<synon<<" "<<anton<<endl;
}
