#include <iostream>
using namespace std;

int main()
{
    int t,i;
    long long int n,c,m,a,b,d,sum;
    cin>>t;
    for(i=1;i<=t;i++)
    {   sum=0;
    	cin>>n>>c>>m;
    	a=n/c;
    	x:
    	b=a/m;
    	d=a%m;
    	sum+=b;
    	if(b+d>=m)
    	{
    		a=b+d;
    		goto x;
    	}
    	cout<<(n/c)+sum<<"\n";
    	
    
    }
    return 0;
}
