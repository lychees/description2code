#include <bits/stdc++.h>
using namespace std;

int main()
{
    int x;
    string s1,s2,s,t;
    scanf("%d",&x);
    while(x--)
    {
    	cin>>s1;
    	cin>>s2;
    	s=s1+s2;
    	t=s;
    	int flag=0;
    	reverse(s.begin(),s.end());
    	if(t.compare(s)==0)
    		flag=1;
    	else
    	{
    		s=s2+s1;
    		t=s;
    		reverse(s.begin(),s.end());
    		if(t.compare(s)==0)
    			flag=1;
    	}
    	if(flag==1)
    		printf("YES\n");
    	else
    		printf("NO\n");
    }
    return 0;
}
