#include<iostream>
#include<algorithm>
#include<map>

using namespace std;

map<pair<long long,pair<long long,long long> >, long long>m;
map<pair<long long,pair<long long,long long> >, long long>:: iterator it;

int main(){

    long long a[5],i,n,ans=0;
    cin >> n;
    while(n--) {
        for(i=0;i<3;i++) cin >> a[i];
        sort(a,a+3);
        m[make_pair(a[0],make_pair(a[1],a[2]))]++;
    }
    for(it=m.begin();it!=m.end();it++) {
        if(it->second == 1) ans++;
    }
    cout << ans << "\n";
    return 0;






}
