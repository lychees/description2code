#include<bits/stdc++.h>
#define ll long long
#define M 4294967143
using namespace std;
ll mat[11][12] , help[11][12] ;
ll ans[300] ;
 
	void preprocess()
	{
		for(int i=1;i<=10;i++)mat[i][i]=1;
 
		for(int i=2;i<=264;i++)
		{
 
			memset(help,0,sizeof(help)) ;
 
			for(int i=1;i<=10;i++)
			{
				for(int j=0;j<=10;j++)
				{
					if(j!=10)
						help[i][j+1]=(help[i][j+1] + mat[i][j])%M ;
 
					if(j!=0)
						help[i][j-1]=(help[i][j-1] + mat[i][j])%M;
				}
 
			}
				ll sm=0 ;
 
				for(int i=1;i<=10;i++)
					for(int j=0;j<=10;j++)
					{
						if(abs(i-j)==1)
							sm+=help[i][j];
 
						mat[i][j]=help[i][j] ;
					}
 
			ans[i]=sm;
 
		}
 
	}
 
int main()
{
 
	preprocess() ;
 
//	cout<<ans[2]<<" "<<ans[4]<<"\n" ;
 
int t,l;
	scanf("%d",&t) ;
 
	while(t--)
	{
 
		scanf("%d",&l) ;
		printf("%lld\n",ans[l]);
	}
 
 
}
