#include <bits/stdc++.h>
using namespace std;
int main () {
	int tc;
	scanf("%d",&tc);
	while (tc--) {
		int n;
		scanf("%d",&n);
		vector < int > v (1001, 0);
		for (int i=0; i<n; i++) {
			scanf("%d",&v[i]);
		}
		int ans = v[0];
		for (int i=1; i<v.size(); i++) {
			ans =__gcd(ans, v[i]);
		}
		cout << ans << endl;
	}
	return 0;
}
