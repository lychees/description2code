#include<bits/stdc++.h>
using namespace std;

#define ll long long
#define F(i,x) for(int i=0; i<x; i++)
#define FI(it, m) for( vector<vector<long long> > :: iterator it=m.begin(); it!=m.end(); it++)
#define mset(a, x) memset(a, x, sizeof(a))
#define p(a) cout<<a<<endl
#define db(a) cout<<a<<" ";

int main()
{
    int t;
    scanf("%d", &t);
    while(t--)
    {
        long a, b;
        scanf("%ld%ld", &a, &b);
        long div = 1;
        long ans1, ans2=a+b;
        int carry=0;
        while(a>0 || b>0)
        {
            ans1 = ans2;
            ans1/=div;
            a/=div;
            b/=div;
            div=10;
            ans2 = a+b;
            if(ans1!=ans2)
                carry++;
        }
        if(!carry)
        p("No carry operation");
        
        else if(carry==1)
            p("1 carry operation");
        
        else
        {
            db(carry);
            p("carry operations");
        }
    }
    return 0;
}
