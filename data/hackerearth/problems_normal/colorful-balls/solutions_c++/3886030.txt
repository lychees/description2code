#include <iostream>
#include <bits/stdc++.h>
using namespace std;

long long dp[11][11][11][11][5];
long long calc(int a,int b,int c,int d,int prev)
{
	if(a==0 && b==0 && c==0 && d==0)
	{
		dp[a][b][c][d][prev]=1;
		return 1;
	}
	else if(dp[a][b][c][d][prev]>=0)
	return dp[a][b][c][d][prev];
	
	dp[a][b][c][d][prev]=0;
	long long sum=0;
	if(prev==1)
	{
		if(b-1>=0)
		sum=sum+calc(a,b-1,c,d,2);
		if(c-1>=0)
		sum+=calc(a,b,c-1,d,3);
		if(d-1>=0)
		sum+=calc(a,b,c,d-1,4);
	}
		
	else if(prev==2)
	{
		if(a-1>=0)
		sum=sum+calc(a-1,b,c,d,1);
		if(c-1>=0)
		sum+=calc(a,b,c-1,d,3);
		if(d-1>=0)
		sum+=calc(a,b,c,d-1,4);	
	}
	else if(prev==3)
	{
		if(b-1>=0)
		sum=sum+calc(a,b-1,c,d,2);
		if(a-1>=0)
		sum+=calc(a-1,b,c,d,1);
		if(d-1>=0)
		sum+=calc(a,b,c,d-1,4);
	}
	else if(prev==4)
	{
		if(b-1>=0)
		sum=sum+calc(a,b-1,c,d,2);
		if(c-1>=0)
		sum+=calc(a,b,c-1,d,3);
		if(a-1>=0)
		sum+=calc(a-1,b,c,d,1);
	}
	else
	{
		if(a-1>=0)
		sum+=calc(a-1,b,c,d,1);
		if(b-1>=0)
		sum=sum+calc(a,b-1,c,d,2);
		if(c-1>=0)
		sum+=calc(a,b,c-1,d,3);
		if(d-1>=0)
		sum+=calc(a,b,c,d-1,4);
	}
	dp[a][b][c][d][prev]=sum;
	return sum;
	
}
int main()
{
	int a,b,c,d;
	cin>>a>>b>>c>>d;
	memset(dp,-1,sizeof(dp));
	cout<<calc(a,b,c,d,0);
    return 0;
}
