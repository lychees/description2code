#include <iostream>
#include<cmath>
#include<cstdio>
using namespace std;
unsigned long long int fib[100005]={0},buff[100005]={0};
int main()
{
   unsigned long long int t,n,sum=0,a,b;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	buff[1]=1;
    	sum=1;
    	int a=0,b=1;
    	for(int i=0;;i++)
    	{
    	sum=a+b;
    	if(sum>=n)
    	break;
    	buff[sum]++;
    	a=b;
    	b=sum;
    	
    	}
    	sum=0;
    	for(int i=0;i<n;i++)
    	if(buff[i]==0)
    	sum+=i;
    //	cout<<sum<<" "; 
    	if(sum%2)
    	{
    		int f=-1;
    		for(int j=2;j<=sqrt(sum);j++)
    		{
    			if(sum%j==0)
    			{
    				f=1;
    				break;
    			}
    		}
    		if(f==1)
    		cout<<"Toe"<<endl;
    		else
    		cout<<"Tic"<<endl;
    	}
    	else cout<<"Tac"<<endl;
    	sum=0;
    }
    return 0;
}
