#include <bits/stdc++.h>
using namespace std;
#define M 100005
#define MOD 1000000007
char ch[M],st[M];
long long a[M],b[M],c[M],d[M];
int main() {
	long long n,m,i,j,k,l,p,q,r,s,t,x;
	cin>>t;
	while(t--){
		long long count=0,flag=0,min=LONG_MAX;
		cin>>n;
		c[0]=0;d[n+1]=0;
		for(i=0;i<=n+5;i++)
		{
			a[i]=0;b[i]=0;
		}
		for(i=1;i<=n;i++){
		cin>>a[i];
		c[i]=c[i-1]+a[i];
		}
		for(i=n;i>=1;i--){
			d[i]=d[i+1]+a[i];
		}
		for(i=1;i<=n;i++){
			
			x=abs(c[i]-d[i+1]);
			if(x<min)
			min=x;
		}
		cout<<min<<endl;
	}
	
	return 0;
}
