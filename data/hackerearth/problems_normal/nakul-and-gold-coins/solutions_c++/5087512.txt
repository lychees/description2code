#include <iostream>
#include <vector>
#include <algorithm>
#define lld long long int
#define pb(x) push_back(x)
#define mp(x,y) make_pair(x,y)
#define maxi 1000000
using namespace std;
int main()
{
	int t, l, r, i, low, up, sum;
	cin >> t;
	bool mark[maxi];
	for (i = 0; i<maxi; i++)
		mark[i] = 1;
	//cout<<mark[2]<<endl;
	vector<int> prime;
	for (i = 2; i*i<maxi; i++)
	{
		if (mark[i] == 1)
		{
			for (int j = i*i; j<maxi; j = j + i)
			{
				mark[j] = 0;
			}
		}
	}
	//cout<<mark[2]<<endl;
	for (i = 2; i<1000000; i++)
		if (mark[i])
			prime.pb(i);
	//cout<<prime.size();
	//cout<<prime[2];
	while (t--)
	{
		sum = 0;
		cin >> l >> r;
		for (i = 0; i<prime.size() && prime[i] <= r; i++)
		{
			low = l / prime[i] + (l%prime[i]>0);
			low = max(low, prime[i] + 1);
			up = r / prime[i];
			if (low <= up)
			{
				sum += upper_bound(prime.begin(), prime.end(), up) - lower_bound(prime.begin(), prime.end(), low);
			}
		}
		cout << sum << endl;
	}
	// cout << "Hello World!" << endl;
	return 0;
}
