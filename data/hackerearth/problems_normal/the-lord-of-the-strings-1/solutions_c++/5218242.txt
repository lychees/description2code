#include <iostream>
using namespace std;
int main()
{
    char s[100000];
    int n,i,a[26]={0};
    cin>>n;
    n=n+1;
    while(n)
    {
    	gets(s);
    	a[26]={0};
		for(i=0;s[i]!='\0';i++)
		{
			a[s[i]]++;
		}
		for(i=0;s[i]!='\0';i++)
		{
			if(a[s[i]]>0)
			{ 
		 		cout<<s[i];
		 		a[s[i]]=0;
	    	}
	    	else
	     		continue;
		}cout<<"\n";
		n--;
    }
    return 0;
}
