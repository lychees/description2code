#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    long long int t;
    cin>>t;
    while(t--)
    {
    	long long int a,b,c;
    	cin>>a>>b>>c;
    	long long int p;
    	//if(b>=0
    	p=pow(a,b);
    	long long int q=p%c;
    	if(q>=((c+1)/2))
    	cout<<p+c-q<<"\n";
    	else
    	cout<<p-q<<"\n";
    }
    return 0;
}
