#include <bits/stdc++.h>

using namespace std;

const int N = 1e5 + 10;

int a[N];

int main() {
	int t;
	scanf("%d", &t);
	for (int tt = 1; tt <= t; tt++) {
		int n, k;
		scanf("%d %d", &n, &k);
		for (int i = 0; i < n; i++) {
			scanf("%d", &a[i]);
		}
		sort(a, a + n);
		int cnt = 0;
		long long res = 0;
		for (int i = n - 1; i >= 0 && cnt < k; i--) {
			res += a[i];
			cnt++;
		}
		printf("%lld\n", res);
	}
	return 0;
}
