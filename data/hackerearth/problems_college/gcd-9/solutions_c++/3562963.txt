#include <bits/stdc++.h>
using namespace std;

int n,a[1000000],max1,r,i,till,j,k;

 int gcd(int a,int b)
 {
  if(b==0)
        return a;
  else
    return gcd(b,a%b);
 }

int main()
{
   // cout<<gcd(4,8);
    scanf("%d %d",&n,&k);
 for(i=0;i<n;i++)
 {
     scanf("%d",&a[i]);
 }

 if(n==1)
 {
     if(a[0]>=k)
     {
         cout<<"1";
     }
     else
        cout<<"0";
     return 0;
 }

 else{
for(i=0;i<n;i++)
{
    if(a[i]>=k){
            till=a[i];
    r=1;
   for(j=i+1;j<n;j++)
   {
       till=gcd(till,a[j]);
   if(till>=k)
    r++;
 else{
    if(r>max1)
        max1=r;
    break;
   }
   }
    }
}
cout<<max1<<"\n";
 }
return 0;
}
