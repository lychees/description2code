#include <bits/stdc++.h>
using namespace std;
int main()
{
    ios_base::sync_with_stdio(false);
    int t,n,q,num;
    cin>>t;
    vector<int>v;
    while(t--)
    {
       cin>>n>>q;
       v.clear();
       for(int i=0;i<n;i++)
       {
           cin>>num;
           v.push_back(num);
       }
       sort(v.begin(),v.end());
       long long ans=0;
       for(int i=1;i<n;i++)
       {
           ans+=(abs(v[i]-v[i-1]));
       }
       ans*=q;
       cout<<ans<<endl;

    }
	return 0;
}
