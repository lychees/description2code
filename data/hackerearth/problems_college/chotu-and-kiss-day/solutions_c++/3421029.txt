#include<iostream>
#include<bits/stdc++.h>
#include<algorithm>
using namespace std;
int search(long x,long a[],int n)
{
	int l=0;
	int r=n-1;
	int m;
	while(l<r){
		m=(l+r)/2;
		if(a[m]==x)
		return 1;
		if(a[m]>x)
		r=m-1;
		if(a[m]<x)
		l=m+1;
	}
	return 0;
}
int main()
{
	int n,x;
	cin >> n;
	long s[n],sum,k;
	int i,j;
	for(i=0;i<n;i++)
	cin >> s[i] ;
	cin >> k;
	sort(s,s+n);
	for(i=0;i<n;i++){
		for(j=i+1;j<n;j++){
			sum=k-s[i]-s[j];
			x=search(sum,s,n);
			if(x==1)
			{
				cout <<"YES"<<endl;
				return 0;
			}
		}
	}
	
	cout << "NO" << endl;
}
