#include <iostream>
#include <algorithm>
using namespace std;
int main() {
	int d,g,cnt=0;
	long long ar[3000],m,sum=0;
	
	cin>>d;
	while(d--) {
		sum=0;cnt=0;
		cin>>g>>m;
		for(int i=0;i<g;i++) {
			cin>>ar[i];
		}
		sort(ar,ar+g);
		
		for(int i=0;i<g;i++) {
			sum+=ar[i];
			if(sum>m) {
				break;
			}
			cnt++;
		}
		cout<<cnt<<endl;
	}
	return 0;
}
