#include <bits/stdc++.h>
using namespace std;
int main()
{
	double a[20][20];
	double ans[300000];
	int qqq[300000];
	int n;
	cin >> n;
	int mx = (1 << n) - 1;
	qqq[mx]=1;
	ans[mx]=1.0;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
		{
			cin >> a[i][j];
		}
	}
	queue<int> q;
	q.push(mx);
	while(q.size())
	{
		int number = q.front();
		int counter = 0;
		for(int i=0;i<n;i++)
		{
			if(number & (1<<i))
				counter++;
		}
		q.pop();
		int total = (counter * (counter -1))/2;
		double prob = 1.0/total;
		for(int i=0;i<n;i++)
		{
			if(number & (1<<i))
			{
				for(int j=0;j<i;j++)
				{
					if(number & (1<<j))
					{
						int temp = (1 << i) ^ number ;
						if(qqq[temp]==0)
							q.push(temp);
						qqq[temp]=1;
						ans[temp] += (1-a[i][j]) * ans[number] * prob;
						temp = (1 << j) ^ number ;
						if(qqq[temp]==0)
						q.push(temp);
						qqq[temp] = 1;
						ans[temp] += a[i][j] * ans[number] * prob;

					}
				}
			}
		}


	}
	for(int i=0;i<n;i++)
	{
		printf("%.5f ",ans[1<<i]);
	}

}
