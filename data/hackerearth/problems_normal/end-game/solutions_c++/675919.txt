#include <iostream>
#include <algorithm>
using namespace std;
#define N 1000000000

int main()
{
    int T,n, a,b, c, d, move;
    cin>>T;
    for(int i=0; i<T;i++)
    {
    	int flag=1;
    	do{
    		cin>>n>>a>>b>>c>>d>>move;
    		if(!(n>=5 && n<=N)){
    			flag=0;
    		}
    		if(!(a>=2 && a<=n)){
    			flag=0;
    		}
    		if(!(b>=1 && b<=n)){
    			flag=0;
    		}
    		if(!(c>=1 && c<=n)){
    			flag=0;
    		}
    		if(!(d>=1 && d<=n)){
    			flag=0;
    		}
    		if(!(move==0 || move==1)){
    			flag=0;
    		}
    		if(flag==0){
    			cout<<"Wrong input"<<endl;
    		}
    		
    	}while(flag==0);
    	if(move==0){
    		if((a-c)>0 || abs(d-b)>(n-a)){
    			cout<<"White Wins \n";
    		}
    		else{
    			cout<<"Draw \n";
    		}
    	}
    	else{
    		if((a-c)>1 || (abs(d-b)-(n-a))>1){
    			cout<<"White Wins \n";
    		}
    		else{
    			cout<<"Draw \n";
    		}
    		
    	}
    	
    }
    return 0;
}
