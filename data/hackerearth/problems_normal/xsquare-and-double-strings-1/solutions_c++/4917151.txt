#include <iostream>
#include <string.h>
#include <algorithm>
using namespace std;

int main()
{
	int t,flag;
	char string[101];
	
	int *hash = (int *)malloc(26*sizeof(int));
	scanf("%d",&t);
	for(int i=0;i<t;i++)
	{
		flag = 0;
		for(int j=0;j<26;j++)
		hash[j] = 0;
		
    scanf("%s",string);
    
    int len = strlen(string);
    int ch;
    
  	for(int j=0;j<len;j++)
  	{
  		ch = string[j];
  		ch = ch - 97;
  		if(hash[ch]==1)
  		{
  			flag = 1;
  			break;
  		}
  		else
  		hash[ch] = 1;
  	}
  	
  	if(flag==1)
  	printf("Yes\n");
  	
  	else
  	printf("No\n");
  	
	}
    return 0;
}
