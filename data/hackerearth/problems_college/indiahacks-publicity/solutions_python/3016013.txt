def find(a):
	if p[a] != -1:
		p[a] = find(p[a])
		return p[a]
	else:
		return a
for _ in range(int(raw_input())):
	n, e = int(raw_input()), int(raw_input())
	ans = 0
	p = n * [-1]
	for _ in range(e):
		a, b = map(int, raw_input().split())
		pa = find(a)
		pb = find(b)
		if pa != pb:
			p[pa] = pb
	for i in range(n):
		if p[i] == -1:
			ans += 1
	print ans
