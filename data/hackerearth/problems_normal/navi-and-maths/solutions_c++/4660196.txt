ll mod=1e9+7;
ll fastPow(ll a,ll b,ll mod)
{
	if(b==0)
	return 1;
	if(b==1)
	return a;
	ll half=fastPow(a,b/2,mod) % mod;
	if (b%2==0)
	return (half* half ) % mod;
	return ((( half * half ) % mod ) * a) % mod;
}
ll subset(int a[],int n)
{
	int pow_size=pow(2,n);
	ll max=-1;
	for (int i=0;i<pow_size;i++)
	{
		int cnt=0;
		ll sum=0;
		ll mul=1;
		for (int j=0;j<n;j++)
		{
			if(i&(1<<j))
			{
				sum=(sum+a[j])%mod;
				mul=(mul*a[j])%mod;
					//cout<< mul <<" "<< sum << endl;
				++cnt;
			}
		}
	//cout << cnt << endl;
		if(cnt>=2)
		{
			//
			ll ans=(mul*fastPow(sum,mod-2,mod))%mod;
			if(ans>max)
			max=ans;
		}
	}
	return max;
}

			for (int i=0;i<n;i++)
			si(a[i]);

}
