
#include<bits/stdc++.h> 

using namespace std;
const int maxn = 25;
double p[maxn][maxn]={0};
double q[maxn][maxn]={0};
double res[maxn]={0};
double d[1 << 19]={0};
int two[1 << 19]={1};
int cnk[20][20]={{1},{1,1}};
int bc[1 << 19];
int abc[1 << 19];
int main()
{
   
    int n,i,j,k;
    cin >> n;
    for (i=0;i<n;i++)
    {
        for (j=0;j<n;j++)
        {
            cin >> p[i][j];
            q[i][j] = 1 - p[i][j];
        }
    }
    for (i=1;i<=n;i++)
        two[i] = two[i-1] << 1;

    abc[0] = n;
    for (i=1;i<two[n];i++){
        bc[i] = bc[i >> 1] + (i % 2);
        abc[i] = n - bc[i];
    }

    for (i=2;i<=n;i++){
        cnk[i][0] = 1;
        for (j=i;j>=1;j--)
            cnk[i][j] = cnk[i-1][j] + cnk[i-1][j-1];
    }

    d[0] = 1;
    for (i=0;i<two[n];++i)
    {
        for (j=0;j<n;++j)
        {
            if (!(i & two[j]))
            {
                for (k=0;k<n;++k)
                {
                    if (k==j)
                        continue;
                    if (!(i & two[k]))
                    {
                        d[i ^ two[k]] += d[i] * p[j][k] * 2.0  / (abc[i] * (abc[i] - 1));
                        if (d[i ^ two[k]]>1.0)
                        {
                            k = k;
                        }

                    }
                }
            }
        }
    }

    int msk = two[n] - 1;
    
    for (i=0;i<n;i++)
    {
        printf("%.5lf ",d[msk ^ two[i]]);
    }

    return 0;
}
