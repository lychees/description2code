#include <bits/stdc++.h>
using namespace std;

int main () {
	int tc;
	cin >> tc;
	while (tc--) {
		int m, n;
		cin >> n >> m;
		int ar[50001];
		for (int i=0; i<n; i++)
			ar[i] = 0;
		
		int f = 0;
		int l = n-1;
		for (int i=0; i<m; i++) {
			int mx = 1000000; int idx = 0;
			if(i%2==0) {
				for(int j=0; j<=n/2; j++){
					if(ar[j]<mx){
						mx=ar[j];
						idx=j;
					}
				}
			}
			else {
				for(int j=n-1; j>=n/2; j--){
					if(ar[j]<mx){
						mx=ar[j];
						idx=j;
					}
				}
			}
			if(i==m-1){
				printf("%d %d\n",idx+1, ar[idx]);
			}
			ar[idx]++;
		}
	}
	return 0;
}
