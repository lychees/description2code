#include<vector>
#include<algorithm>
#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	vector<int> d;
    	for(int i=0;i<7;i++)
    	{
    		int a;
    		cin>>a;
    		d.reserve(a);
    		for(int j=0;j<a;j++)
    		{
    			int x,y;
    			cin>>x>>y;
    			d.push_back(y-x);
    		}
    		
    	}
    	sort(d.begin(),d.begin()+d.size());
    	int i=d.size();int s=0;
    	i=ceil(i*(3/4.0));
    	for(int j=0;j<i;j++)
    	s+=d[j];
    	cout<<(24*60*60*7)-s<<endl;
    	
    }
    return 0;
}
