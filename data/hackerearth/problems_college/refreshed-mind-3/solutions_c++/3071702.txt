
#include<cstdio>
using namespace std;

int main(int argc, char** argv) {
   
    int i,n;
   unsigned long long int ar[101],sum=0;
    
    scanf("%d",&n);
    for(i=1;i<=n;i++)
    {
    	scanf("%lld",&ar[i]);
	}
	for(i=1;i<=n;i++)
	sum+=i*ar[i]-i+1;
	printf("%lld",sum);
	return 0;
}
