n = int(input())
l = sorted(list(map(int, raw_input().split())))
dc = {}
n = len(l)
for i in range(n):
    if l[i] not in dc:
        dc[l[i]] = 1
    else:
        dc[l[i]] += 1
vals = list(dc.values())
vals = sorted(vals, reverse=-1)
x = 0
print n
for i in range(len(vals)):
    x += vals[i]
    if n-x <= 0:
        break
    print n-x
