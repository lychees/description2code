#include <iostream>
#include <queue>
using namespace std;

int main()
{
    int M,N,ans = 0;
    priority_queue < int > q;
    cin>>M>>N;
    int Arr[N];
    for(int i = 0; i < M; i++){
    	cin>>Arr[i];
    	q.push(Arr[i]);
    }
    for(int i = 0; i < N; i++){
		int x = q.top();
		ans = ans + x;
		q.pop();
		x--;
		if(x != 0) 
		q.push(x);
    }
    cout<<ans<<endl;
   return 0;
}
