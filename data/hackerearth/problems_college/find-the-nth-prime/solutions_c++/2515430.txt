#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    long long int i,n=10000,j;
    int flag,k,q;
    long long int a[1001];
    
    a[0]=2;
    a[1]=3;
    a[2]=5;
    a[3]=7;
    k=4;
    for(i=10;i<n;i++){
    	if(k==1000){
    		break;
    	}
    	flag=1;
    	for(j=2;j<=sqrt(i);j++){
    		if(i%j==0){
    			flag=0;
    			break;
    		}	
    	}
    	if(flag==1){
    		a[k++] = i;
    	}
    }
    
    cin>>q;
    cout<<a[q-1]<<endl;
    return 0;
}
