#include <iostream>
#include<cmath>
#include<cstdio>
using namespace std;
#define size 127
int main()
{
    int n;
    float temp;
    int arr[size]={0};
    string str;
    int count=0;
    scanf("%d",&n);
    while(n--){
    	cin>>str;
    	arr[str[0]]++;
    }
    for(int i=0;i<128;i++){
    	if(arr[i]<=10 && arr[i]>0) count++;
    	else if(arr[i]>10){temp =arr[i];  count+= ceil(temp/10);}
    }
    cout<<count<<endl;
    return 0;
}
