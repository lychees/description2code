#include <iostream>
using namespace std;
int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
        long long n;
        cin>>n;
        int i=0;
        for(i;n>0;i++)
        {
            n -= i;
        }
        i--;
        if(i%3==0) cout<<"SL"<<endl;
        if(i%3==1) cout<<"LB"<<endl;
        if(i%3==2) cout<<"BS"<<endl;
    }
}
