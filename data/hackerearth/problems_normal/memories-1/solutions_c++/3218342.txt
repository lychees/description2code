#include<bits/stdc++.h>
using namespace std;
#define ll long long int
struct node
{
    ll a,b;
};
vector<node> arr(1001);
int main()
{
    ll p,test,i,j,num,queries,sum,nmbr;
    scanf("%lld",&test);
    while(test--)
    {
        scanf("%lld%lld",&num,&p);
        for(i=1;i<=p;i++)
            scanf("%lld%lld",&arr[i].a,&arr[i].b);
        ll pwr=pow(2,num);
        pwr-=1;
        sum=1;
        for(ll i=2;i<=pwr;i++)
        {
            for(j=1;j<=p;j++)
            {
                if( (i&(1<<(arr[j].a-1)))&&(i&(1<<(arr[j].b-1))) )
                    break;
            }
            if(j>p)
                sum++;
        }
        printf("%lld\n",sum);
    }
}
