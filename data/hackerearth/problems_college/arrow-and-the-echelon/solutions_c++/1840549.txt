#include <iostream>
using namespace std;

int main()
{
    int t, n, five, three, i;
    cin>>t;
    while(t--){
    	cin>>n;
    	five=0;
    	three=0;
    	if(n%3==0)
    		five=n;
    	else {
	    	for(i=n;i>=1;i--){
	    		if(i%3==0 && (n-i)%5==0){
	    			five = i;
	    			three = n-i;
	    			break;
	    		}
	    	}
    	}
    	if(five==0 && three==0)
    		if(n%5==0)
    			three=n;
    	if(five==0 && three==0)
    		cout<<"-1";
    	else {
    		for(i=1;i<=five;i++)
    			cout<<"5";
    		for(i=1;i<=three;i++)
    			cout<<"3";
    	}
    	cout<<endl;
    }
    return 0;
}
