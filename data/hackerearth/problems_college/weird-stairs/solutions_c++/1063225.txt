#include <iostream>
using namespace std;

int main()
{
    int tno;
    long long int i,N,W;
    
    cin>>tno;
    
    while(tno--)
    {
    	cin>>W>>N;
    	
    	int *f=new int[N+1];
    	long long int *Y=new long long int[N+1];
    	
    	for(i=1;i<=N;i++)
    		cin>>f[i];
    		
    	Y[0]=0;
    	for(i=1;i<=N;i++)
    	{
    		if(f[i]==1)
    			Y[i]=2*i;
    		else
    			Y[i]=Y[i-1]+1;
    	}
    	
    	cout<<(Y[N]+N*W)<<endl;
    }
    return 0;
}
