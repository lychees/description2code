#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    
    for(int h=0;h<t;h++)
    {
    	int n,a,b;
    	cin>>n;
    	int x[1000000]={0};
    	int cnt=0,ans=0;
    	for(int i=0;i<n;i++)
    	{
    		cin>>a>>b;
    		if(b==1)
    		{
    			if(x[a]==0)
    			{cnt++;
    			ans=max(ans,cnt);
    			x[a]=1;
    			}
    		}
    		else
    		cnt=0;
    	
    		
    	}cout<<ans<<endl;
    	ans=0,cnt=0;
    }
}
