 #include<iostream>
 #include<cmath>
 #include<iomanip>
 #define pi 3.1415
 using namespace std;

int main()

{
    double a=0.0,b=0.0,c=0.0,max1=0.0;
    bool istriangle=false;
   int t;
   cin>>t;
   while(t--)
   {
       cin>>a>>b>>c;

 if(a>=1&&a<=1000&&b>=1&&b<=1000&&c>=1&&c<=1000)
    {
          max1=a;
          if(b>max1){
        	  max1=b;
        	  if(max1<a+c)
        		  istriangle=true;
          }
          else if(c>max1){
        	  max1=c;
        	  if(max1<a+b)
        		  istriangle=true;
          }
          else{
        	  max1=a;
        	  if(max1<b+c)
        		  istriangle=true;
          } }

          if(istriangle){
        	  double k=0.0,temp1=0.0,temp2=0.0,temp3=0.0,radius=0.0,area=0.0;
        	  k=(a+b+c)/2;
        	  temp1=k*(k-a)*(k-b)*(k-c);
        	  temp2=sqrt(temp1);
        	  temp3=4*temp2;
        	  radius=(a*b*c)/temp3;
        	  area=pi*radius*radius;
        	  cout << setprecision (4) << fixed<<area<<endl;

          }
 }
return 0;

          }
