#include <iostream>
using namespace std;


long long int b[1000005];
int main()
{
    long long int n,a[1000005],i,m,j,l,r,k;
    //cin >> n;
    
    scanf("%lld",&n);
    
    for(i=0;i<n;i++){
    	//cin >> a[i];
    	scanf("%lld",&a[i]);
    }
    
    cin >> m;
    for(j=0;j<m;j++){
    	//cin >> l >> r >> k;
    	scanf("%lld%lld%lld",&l,&r,&k);
    	
    	b[l-1]+=k;
    	b[r]+=-k;
    }
    
    
    for(i=1;i<n;i++){
    	b[i]+=b[i-1];
    }
    
    for(i=0;i<n;i++){
    	//cout << a[i]+b[i] << endl;
    	printf("%lld\n",a[i]+b[i]);
    }
}
