#include<iostream>
#include<cstdio>
#include<math.h>
#include<iomanip>
using namespace std;
int main()
{
   double r,s;
   scanf("%lf %lf",&r,&s);
   double max=0.00;
   double d=0.570;
   double area=0;
   while(d<1)
   {
   double h=s*d;
   double r1=r;
   double r2=r1+s*(sqrt(1-(pow(d,2))));
   area=((double)1/3)*(3.1415926535*h)*(r1*r1+r1*r2+r2*r2);
   if(max<area)
   max=area;
   d=d+0.00001;
   }
   std::cout << std::fixed << std::setprecision(2) << max;
   return 0;
}
