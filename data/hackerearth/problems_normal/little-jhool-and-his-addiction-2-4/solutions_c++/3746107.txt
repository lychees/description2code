#include<bits/stdc++.h>
#define ll long long
using namespace std;
vector<ll>v;
priority_queue<ll>pq1,pq2;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		ll n,k,x;
		cin>>n>>k;
		for(int i=0;i<n;i++)
		{
			cin>>x;
			v.push_back(x);
		}
		if(n==2)
		{
			int res=0;
			if(res>k)
				cout<<res<<"\n"<<"No more girlfriends!"<<endl;
			else if(res==k)
				cout<<res<<"\n"<<"Lucky chap!"<<endl;
			else
				cout<<res<<"\n"<<"Chick magnet Jhool!"<<endl;
			v.clear();
		}
		else
		{
            sort(v.begin(),v.end());
            vector<ll>::iterator st,en;
            st=v.begin();
            en=v.end();
            en--;
            while(st<en)
            {
            	ll y=(*st)+(*en);
            	pq1.push(y);
            	pq2.push((-1)*y);
            	st++;
            	en--;
            }
            ll ans1=pq1.top();
            ll ans2=(-1)*pq2.top();
            ll result=ans1-ans2;
            if(result>k)
            	cout<<result<<"\n"<<"No more girlfriends!"<<endl;
            else if(result==k)
            	cout<<result<<"\n"<<"Lucky chap!"<<endl;
            else
            	cout<<result<<"\n"<<"Chick magnet Jhool!"<<endl;
            v.clear();
            while(!pq1.empty())
            {
            	pq1.pop();
            }
            while(!pq2.empty())
            {
            	pq2.pop();
            }
		}
	}
	return 0;
}
