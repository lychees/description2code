#include <bits/stdc++.h>
#define ll long long int
using namespace std;

int main()
{
    int n, flag = 0;
    int A[101];

    cin >> n;

    for (int i = 0; i < n; i++) {
        cin >> A[i];
        if (A[i] == 1) {
            flag = 1;
        }
    }
    if (flag == 0) {
        cout << "0\n";
        return 0;
    }
    ll r = 1;
    int i = 0;
    while (i < n) {
        if (A[i] == 1 && i != n - 1) {
            int cnt = 1;
            int j = i + 1;
            for (j = i + 1; j < n; j++) {
                if (A[j] == 1) {
                    i = j;
                    r *= cnt;
                    break;
                }
                cnt++;
            }
            if (j == n) {
                i = n;
            }
        } else {
            i++;
        }
    }
    cout << r << endl;
}
