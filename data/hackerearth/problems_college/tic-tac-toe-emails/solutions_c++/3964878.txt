#include <iostream>
#include<set>
#include<map>
#include<string>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int T;
   scanf("%d",&T);
   while(T>=0){
   //	cout<<"T="<<T<<endl;
   	//char str[10000];
    string line;
    //set<string>myset;
    map<string,int>mymap;
    string del=" ";
    getline(cin,line);
    
    if(line=="")
    {
    	
    	T--;
    	continue;
    }
    string arr[100000];
    string token;
    int pos,val,j;
    int i=0;
    while(((pos=line.find(del))!=string::npos))
    {
    	token=line.substr(0,pos);
    	//cout<<"token="<<token<<endl;
    	val=mymap[token];
    	if(val==0)
    	{//cout<<"token="<<token<<"  ";
    		arr[i]=token;
    		//cout<<"arr[i]="<<arr[i]<<endl;
    		mymap[token]=1;
    		i++;
        	}
    	line.erase(0,pos+del.length());
    	
    }
    val=mymap[line];
    if(val==0)
    {
    	mymap[line]=1;
    	arr[i]=line;
    	i++;
    }
    for(j=0;j<i;j++)
    cout<<arr[j]<<" ";
    cout<<endl;
   	T--;
   }
    return 0;
}
