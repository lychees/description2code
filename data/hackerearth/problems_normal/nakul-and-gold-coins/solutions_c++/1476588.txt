#include <bits/stdc++.h>

using namespace std;

#define max1 1000009
#define assn(n,a,b) assert(n<=b&&n>=a)

char str[max1];
vector<int> ar;
int c;

void gen() {
	for(int i =2;i*i<=max1;i++) {
		if(str[i]==0) {
			for(int j = i; j*i<=max1;j++) {
				str[i*j] = 1;
			}
		}
	}
	
	for(int i = 2; i <= max1-9;i++) {
		if(str[i]==0) {
			ar.push_back(i);
		}
	}
}

long long count(int n) {
	long long count = 0;
	
	for(int i = 0; i < c and ar[i]< n; i++) {
		int div = n/ar[i];
		count+= upper_bound(ar.begin(), ar.begin()+i,div)-ar.begin();
	}
	
	return count;
}
	
int main()
{
	ios_base::sync_with_stdio(0);
	gen();
	int tc,l,r,ans;
	
	cin>>tc;
	assn(tc,1,300);
	
	c = ar.size();
	
	for(int i = 0; i < tc; i++) {
		cin>>l>>r;
		assn(l,1,100000000);
		assn(r,1,100000000);
		assert(r>=l);
		
		long long count1 = count(r);
		long long count2 = count(l-1);
		
		printf("%lld\n", count1-count2);
	}
	
	return 0;
}
