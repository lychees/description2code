#include <iostream>
using namespace std;

int main()
{
	int t,q;
	cin>>t;

    for(int q=0;q<t;q++)
    {
    	unsigned long long int n,j,k,p;
    	cin>>n;
    	p=n;
    	unsigned long long int a[n];
    	for(j=0;j<n;j++)
    	{
    		cin>>a[j];
    	}
    	for(j=0;j<n-1;j++)
    	{
    		for(k=j+1;k<n;k++)
    		{
    			if(a[j]==a[k])
    			{
    				p--;
    				break;
    				
    			}
    		}
    	}
    	cout<<p<<endl;
    }
}
