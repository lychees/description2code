#include<iostream>
using namespace std;
int main()
{
	int t,s,n,a[1000],i,j;
cin>>t;
	while(t--)
	{
		 s=0;
		cin>>n;
		for(i=1;i<=n;i++)
		cin>>a[i];
		for(i=1;i<=n;i++)
		for(j=1;j<=n-i;j++)
		if(a[j]>a[j+1])
		{
		s=a[j];
		a[j]=a[j+1];
		a[j+1]=s;
	}
	cin>>i>>j;
	s=a[n-i+1]+a[j];
		cout<<s<<"\n";
		
	}
	return(0);
}
