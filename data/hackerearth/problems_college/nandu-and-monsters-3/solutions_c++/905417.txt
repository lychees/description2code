int level(int a)
{
      int ret = 0;
      while(a)
      {
            a/=2;
            ret++;
      }
      return ret;
}
char aa[100001];

int solve(int a,int b)
{
      int a1 = level(a);
      int a2 = level(b);

      while(a1!=a2)
      {
            if(aa[b-1] == '1') return 0;
            b/=2;
            a2--;
      }
      // both are at same level now

      while(a!=b)
      {
            if(aa[b-1] == '1' || aa[a-1] == '1') return 0;
            a/=2;
            b/=2;
      }
      if(aa[b-1] == '0')
            return 1;
      else
            return 0;
}

int main()
{
      int n,q;
      scanf("%d%d",&n,&q);
      scanf("%s",aa);
      while(q--)
      {
            int a,b;
            scanf("%d%d",&a,&b);
            if(a>b) swap(a,b);
            if(solve(a,b)) printf("Yes\n");
            else printf("No\n");
      }
      return 0;
}
