no = input()
ll = []
for m in range(no):
    a, b= map(int,raw_input().split())
    lll = []
    for i in range(a):
        v = []
        for j in range(b):
            bp = 1000
            l = map(int,raw_input().split())
            for k in l:
                c = (1.0 - (k/100.0))*bp
                bp = c
            v.append(c)
        lll.append(v.index(min(v))+1)
    ll.append(lll)
for ss in ll:
    print(' '.join((str(s)) for s in ss))
