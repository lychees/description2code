import sys
coins = {}


def exchange(n):
    if n in coins.keys():
        return coins[n]
    final = 0
    sumN = (n/2) + (n/3) + (n/4)
    if sumN > n:
        final += exchange(n/2) + exchange(n/3) + exchange(n/4)
    else:
        final += n
    coins[n] = final
    return final


for n in sys.stdin:
    print exchange(int(n))
