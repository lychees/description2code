#include<stdio.h>
#include<iostream>
#include<algorithm>
using namespace std;
long long int a[1000006],b[1000006],p[1000006],x[1000006],y[1000006],hash1[1000006],c;

long long int fn(long long int z)
{
	long long int l,r,m;
	l=0;
	r=c-1;
	while(1)
	{
		if(y[l]==z)
		return x[l];
		if(y[r]==z)
		return x[r];
		m=(l+r)/2;
		if(y[m]>z)
		r=m;
		else
		l=m;
	}
}

int main()
{
	long long int f,n,i,z,max,ans;
	scanf("%lld",&f);
	while(f--)
	{
		scanf("%lld",&n);
		for(i=1;i<=n;i++)
		scanf("%lld",&a[i]);
		a[0]=0;
		for(i=1;i<=n;i++)
		{
			scanf("%lld",&z);
			b[i]=a[i]+z;
		}
		b[0]=0;
		c=0;
		for(i=1;i<=n;i++)
		p[i]=a[i];
		for(i=n+1;i<=2*n;i++)
		p[i]=b[i-n];
		sort(p+1,p+2*n+1);
		p[0]=0;
		c=0;
		for(i=1;i<=2*n;i++)
		{
			if(p[i]!=p[i-1])
			{
				x[c]=c+1;
				y[c]=p[i];
				c++;
			}
		}
		/*for(i=0;i<c;i++)
		printf("%lld %lld\n",x[i],y[i]);*/
		for(i=1;i<=n;i++)
		{
			a[i]=fn(a[i]);
			b[i]=fn(b[i]);
		}
		/*for(i=1;i<=n;i++)
		printf("%lld %lld\n",a[i],b[i]);*/
		for(i=0;i<=1000006;i++)
		hash1[i]=0;
		for(i=1;i<=n;i++)
		{
			hash1[a[i]]++;
			hash1[b[i]]--;
		}
		max=0;
		ans=0;
		for(i=0;i<=1000006;i++)
		{
			ans=ans+hash1[i];
			if(ans>max)
			max=ans;
		}
		printf("%lld\n",max);
	}
	return 0;
}
