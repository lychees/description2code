#define ll long long

vector<ll> v;

int main()
{
    ll x, y, l, r, i, qu, k = 0;

    scanf("%lld%lld", &x, &y);

    ll gc = __gcd(x, y);

    //printf("%d\n", gc);
    for (i = 1; i *i < gc; i++) {
        if (gc % i == 0)
            v.pb(i), v.pb(gc/i);
    }
    ll xx = sqrt(gc);
    if(xx*xx == gc)
        v.pb(xx);

    sort(v.begin(), v.end());

    v.pb(infi);
    scanf("%lld", &qu);

    //if (k > 25)
    //	while (1) ;
    //PR(v, 0, v.size()-1);
    while (qu--) {
        scanf("%lld%lld", &l, &r);

        vector<ll>::iterator it = lower_bound(v.begin(), v.end(), r);
       // cout << *it << endl;
        if(r == *it)
            printf("%lld\n", r);
        else {
            it--;
           // cout << "   " << *it << endl;
            if(*it <= r && *it >= l)
                printf("%lld\n", *it);
            else
                printf("-1\n");
        }

    }

    return 0;
}
