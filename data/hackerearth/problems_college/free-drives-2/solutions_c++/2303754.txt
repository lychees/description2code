#include <cstdlib>
#include <iostream>

using namespace std;
long long T,I,n;
int main(int argc, char *argv[])
{   cin>>T;
    for(I=1;I<=T;I++)

        if(n==1)
          cout<<0<<endl;
        else
        if(n==2)
           cout<<1<<endl;
        else
        {   n=n-1;
           cout<<n*(n+1)/2<<endl;  
           }        
    }
    
    return EXIT_SUCCESS;
}
