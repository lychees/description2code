import sys


def fact(n):
    d = {}
    for i in xrange(1, int(n**0.5)+1):
        if n % i == 0:
            d[i] = 1
            d[n/i] = 1
    return len(d)


def gcd(a, b):
    if b == 0:
        return a
    else:
        return gcd(b, a % b)


t = input()

while t > 0:
    s = sys.stdin.readline().strip().split()
    s = map(int, s)
    n, r = s
    k = fact(r-1)
    d = n-k
    l = gcd(n, d)
    n = n/l
    d = d/l
    print str(d)+"/"+str(n)
    t = t-1
