#include<cstdio>
#include<algorithm>
#include<cstring>

using namespace std;

int mark=0,i,j,l,t;
char ar[100000];

int main()
{
	scanf("%d",&t);
	
	while(t--)
	{  
		mark=0;
		scanf("%s",&ar);
		l=strlen(ar);
		
		for(i=0,j=l-1;i<=j;i++,j--)
		{
			if(ar[i]!=ar[j])
			mark=1;
		}
		
		
	if(mark==1)
	   printf("-1\n");
	else
	printf("%d\n",ar[l/2]);
	}
return 0;	
}
