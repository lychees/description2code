#include <iostream>
using namespace std;

int main()
{
	int t;
    cin>>t;
    while(t--)
    {
    	int n,fl=0,i=0,s=0;
    	cin>>n;
    	int a[n],b[n];
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	for(i=0;i<n;i++)
    	{
    		b[i]=0;
    	}
    	i=0;
    	while(a[i]!=-1)
    	{
    		if(a[i]==-1)
    			break;
    		s=a[i];
    		b[i]=1;
    		i=s;
    		if(b[i]==1)
    		{
    			fl++;
    			break;
    		}	
    	}
    	if(fl!=0)
    		cout<<"Lose"<<endl;
    	else if(fl==0)
    		cout<<"Win"<<endl;
    }
    return 0;
}
