#include<bits/stdc++.h>
using namespace std;
int main()
{
    int a[30],b[30];
    memset(a,0,sizeof(a));
    memset(b,0,sizeof(b));
    int c1,c2,c3;
    scanf("%d%d%d",&c1,&c2,&c3);
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++)
        scanf("%d",&a[i]);
    sort(a,a+n);
    n+=3;
    b[0]=min(c2,c3);
    b[n-1]=max(c2,c3);
    b[n/2]=c1;
    int k=0;
    for(int i=1;i<n/2;i++)
        {
            b[i]=a[k];
            k++;
        }
    for(int i=n/2+1;i<n-1;i++)
    {
        b[i]=a[k];
        k++;
    }
    /*for(int i=0;i<n;i++)
        printf("%d ",b[i]);
    cout<<endl;*/
    int d1=0;
    for(int i=1;i<n;i++)
        d1+=abs(b[i]-b[i-1]);
    if(n%2==0)
    {
        int c[30];
        c[0]=min(c2,c3);
        c[n-1]=max(c2,c3);
        c[n/2-1]=c1;
        int k1=0;
    for(int i=1;i<n/2-1;i++)
        {
            c[i]=a[k1];
            k1++;
        }
    for(int i=n/2;i<n-1;i++)
    {
        c[i]=a[k1];
        k1++;
    }
    int d2=0;
    for(int i=1;i<n;i++)
        d2+=abs(c[i]-c[i-1]);
    d1=min(d1,d2);
    }
    printf("%d\n",d1);
    return 0;
}
