#include <iostream>
#include <iomanip>
#include <cmath>
using namespace std;

double calcAngle(double h, double m) {
	if (h <0 || m < 0 || h >12 || m > 60) return 0;
	if (h == 12) h = 0;
	if (m == 60) m = 0;
	double ah,am,angle;
	ah= 0.5 * (h*60 + m);
	am=6*m;
	angle = abs(ah - am);
	angle = min(360-angle, angle);

return angle;
}
int main()
{
    int x,y;
    char ch; 
    cin>>x>>ch>>y;
    std::cout << std::setprecision(6) << std::fixed;
    std::cout<<calcAngle(x,y);
    return 0; 
}
