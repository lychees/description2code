int main()
{
    int n,m;
    cin>>n>>m;
    int a[n][m];
    for(int i=0;i<n;i++)
    {
      for(int j=0;j<m;j++)
      {
      	cin>>a[i][j];
      }
    }
    int r[n],c[m];
    for(int i=0;i<n;i++)
    {	
    	int sr=0;
    	for(int j=0;j<m;j++)
    	{
    		sr+=a[i][j];
    		r[i]=sr;
    		
    	}
    }
    for(int j=0;j<m;j++)
    {	
    	int sc=0;
    	for(int i=0;i<n;i++)
    	{
    		sc+=a[i][j];
    		c[j]=sc;
    	}
    }
    int max=0;
    for(int i=0;i<n;i++)
    
     for(int j=0;j<m;j++)
      
      {
      	int x=(r[i]^c[j]);
      	if (x>max)
      	max=x;
      	
      	 
      }
    
    cout<<max<<endl;
    return 0;
}
