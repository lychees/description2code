#include <iostream>
using namespace std;

int main()
{
    int K,i,j,nwords,flag,index=0,index1=0;
    string disliked[11],words[101],awords[101];
    char output[1000];
    cin>>K;
    for(i=0;i<K;i++)
    cin>>disliked[i];
    cin>>nwords;
    for(i=0;i<nwords;i++)
    {
    cin>>words[i];
    flag=0;
    for(j=0;j<K;j++)
    if(words[i]==disliked[j])
    flag=1;
    if(flag==0)
    awords[index++]=words[i];
    }
    for(i=0;i<index;i++)
    {
    output[index1++]=toupper(awords[i][0]);
    output[index1++]='.';
    }
    output[index1-1]='\0';
    cout<<output;

    return 0;
}
