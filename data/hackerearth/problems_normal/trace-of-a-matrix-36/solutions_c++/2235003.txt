#include <bits/stdc++.h>
using namespace std;

int main()
{
    int tc;
    cin>>tc;
   long long n,m;
    while(tc--){
    	cin>>n>>m;
    	long long l;
		if(n>m)l=m;
		else l=n;
		long long ans =0;
		m++;n++;
		m+=n;
		ans+=(4+((l-1)*m));
			ans*=l;
			ans/=2;
    	
    	cout<<ans<<endl;
    }
    return 0;
}
