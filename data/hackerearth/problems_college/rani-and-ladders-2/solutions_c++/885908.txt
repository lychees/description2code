			int B, U; cin >> B >> U;
			a[B] = U;

		vector<int> dp(N + 1, INT_MAX / 2);
		int x = (a[1] ? a[1] : 1);
		dp[x] = 0;
		for (int i = 1; i <= N; i++)

				if (i + j > N) continue;
				int x = (a[i + j] ? a[i + j] : i + j);
				dp[x] = min(dp[x], dp[i] + 1);
			}
		cout << dp[N] << endl;
	}
