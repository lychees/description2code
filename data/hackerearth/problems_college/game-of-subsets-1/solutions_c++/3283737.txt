#include <iostream>
#define MOD 1000000007
using namespace std;

long long powm(long long a,long long b)
{
    if(b==0) return 1;
    return (powm((a*a)%MOD,b/2)*(b%2?a:1))%MOD;
}

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	long long n;
    	cin >> n;
    	cout << powm(3,n) << endl;
    }
    return 0;
}
