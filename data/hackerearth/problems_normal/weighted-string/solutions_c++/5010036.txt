#include <bits/stdc++.h>

using namespace std;

const int N = 1e6 + 10;

long long wt[N];

int main() {
	ios::sync_with_stdio(false);
	int t;
	cin >> t;
	for (int tt = 1; tt <= t; tt++) {
		int n, k;
		string s;
		cin >> k >> s;
		n = s.length();
		wt[0] = 0;
		for (int i = 1; i <= n; i++) {
			wt[i] = s[i - 1] - 'a' + 1 + wt[i - 1];
		}
		long long res = 0;
		for (int i = 0; i < n + 1; i++) {
			int lb = lower_bound(wt + i, wt + n + 1, wt[i] + k) - (wt + i);
			int ub = upper_bound(wt + i, wt + n + 1, wt[i] + k) - (wt + i);
			res += ub - lb;
		}
		cout << res << '\n';
	}
	return 0;
}
