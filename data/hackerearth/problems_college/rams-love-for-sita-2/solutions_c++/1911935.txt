#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int test;
    cin>>test;
    while(test--){
    	int n,c,m;
    	cin>>n>>c>>m;
    	int ans = n/c;
    	int temp1 = ans;
    	int temp2 = ans;
    	while(temp2 >= m){ 		
   			temp1 = temp2/m;
   			temp2 -= temp1 * m;
   			temp2 += temp1;
   			ans += temp1;
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
