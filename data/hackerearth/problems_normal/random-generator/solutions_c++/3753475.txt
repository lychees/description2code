#include<cstdio>
#include<algorithm>

using namespace std;

int abs(int a, int b){
    if(a < 0) return -a;
    return a;
}

int main(){
    int t;
    scanf("%d", &t);
    while(t--){
        int n, k, p;
        scanf("%d %d %d", &n, &k, &p);
        int val[n+1];
        int v[n];
        long long sum = 0;
        for(int i=0;i<n;i++){
            scanf("%d", &val[i]);
        }
        sort(val, val+n);
        for(int i=1;i<n;i++){
            v[i] = abs(val[i] - val[i-1]);
        }
        bool temp = true;
        for(int i=1;i<n;i++){
            sum += v[i];
            if(i >= k-1){
                if(sum <= (long long)p*2){
                    printf("NO\n");
                    temp = false;
                    break;
                }
                sum -= v[i-(k-1)+1];
            }
        }
        if(temp){
            if(k == 1) printf("NO\n");
            else printf("YES\n");
        }
    }
}
