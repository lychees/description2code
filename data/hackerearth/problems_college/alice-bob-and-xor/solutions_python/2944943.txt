
def xor(x):
    r = x % 4

    if r == 0:
        return x
    elif r == 1:
        return 1
    elif r == 2:
        return x + 1
    elif r == 3:
        return 0

def main():
    t = input()
    while t:
        t -= 1
        l, r = map(int, raw_input().split())
        print xor(l-1) ^ xor(r)

main()
