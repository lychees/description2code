#include <iostream>
#include <algorithm>
using namespace std;
int a[10002],b[10002];
int main()
{
    //cout << "Hello World!" << endl;
    int t,n,i,j,flag=0,sindex,eindex,count;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	flag=1,sindex=0;eindex=0;
    	count=0;
    	for(i=0;i<n;i++)
    	{
    		scanf("%d",&a[i]);
    		b[i]=a[i];
    		/*if(i>0 && a[i]<a[i-1])
    		{
    			sindex=i;
    			flag=1;
    			count++;
    		}
    		if(flag && a[i]>a[i-1])
    		{
    			eindex=i;
    		}*/
    	}
    	sort(a,a+n);
    	for(i=0;i<n;i++)
    	{
    		if(a[i]!=b[i])
    		{
    			j=i;
    			while(j<n && b[j]>b[j+1])
    			{
    				j++;
    			}
    			if(b[j+1]<a[i-1] && b[j+1]!=0)
    			flag=0;
    			count++;
    			sindex=i+1;
    			if(b[j+1]==0)
    			eindex=j;
    			else
    			eindex=j+1;
    			i=j;
    		}
    		
    	}
    	//printf("%d %d\n",sindex,eindex);
    //	if(flag && count==1 && eindex>0 && sindex>0)
    if(flag && count==1)
    	{
    		printf("%d %d\n",sindex,eindex);
    	}
    	else
    	{
    		printf("%d %d\n",-1,-1);
    	}
    }
    return 0;
}
