#include <bits/stdc++.h>
using namespace std;

int arr[109];
vector<int> vct;


int main() {
	int n,mx,sum,x,cnt,l,r;
	cin>>n;
	sum=0;
	for (int i=1;i<=n;i++)
	{
		cin>>arr[i];
		sum+=arr[i];
	}
	
	mx = 0;
	
	for (int i=1;i<=sum;i++)
	{
	//	cout<<"...."<<i<<endl;
		x=0;cnt=0;l=0;r=0;
		//l=1;r=1;x=arr[1];
		while (l <= n && r <= n)
		{
			if(x < i)
			{
				r++;
				x += arr[r];
			}
			else if(x == i)
			{
				cnt++;
				l=r;x = 0;
			}
			else
			{
				l++;x -= arr[l];
			}
	//		cout<<l<<"  "<<r<<"  "<<x<<endl;
		}
		if (cnt > mx)
		{
			mx = cnt;
			vct.clear();
			vct.push_back(i);
		}
		else if (cnt == mx)
		{
			vct.push_back(i);
		}
	//	cout<<"---"<<cnt<<endl;
	}
	
	cout<<mx<<endl;
	for(int i=0;i<vct.size();i++)
	{
		cout<<vct[i]<<" ";
	}
	return 0;
}
