#include <bits/stdc++.h>
using namespace std;


int main()
{
    int tc,x1,y1,x2,y2,x3,y3,x4,y4,i,j;
    long double a,a1,a2,a3;
    scanf("%d",&tc);
    while(tc--){
    	scanf("%d%d",&x1,&y1);
    	scanf("%d%d",&x2,&y2);
    	scanf("%d%d",&x3,&y3);
    	scanf("%d%d",&x4,&y4);
    	a=abs((x1*(y2-y3) + x2*(y3-y1)+ x3*(y1-y2))/2.0);
    	a1=abs((x4*(y2-y3) + x2*(y3-y4)+ x3*(y4-y2))/2.0);
    	a2=abs((x1*(y4-y3) + x4*(y3-y1)+ x3*(y1-y4))/2.0);
    	a3=abs((x1*(y2-y4) + x2*(y4-y1)+ x4*(y1-y2))/2.0);
    	if(a==(a1+a2+a3))
    		printf("\nINSIDE");
    	else
    		printf("\nOUTSIDE");
    }
    return 0;
}
