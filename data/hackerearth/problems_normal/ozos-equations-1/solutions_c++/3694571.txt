typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
typedef pair<ll,ll> ii;


const ll N=3e9;
ll n,cases,frst,lst,mid;
bool fnd;



bool f(ll a,int b)
{
	int sum=0;
	for(;a;)
	{
		sum+=a%10;
		a/=10;
	}
	return sum==b;
}




int main()
{
	std::ios::sync_with_stdio(0);
	cin.tie(0);
	cin>>cases;
	while(cases--)
	{
		cin>>n;
		fnd=0;
		for(ll b=1;b<=100;b++)
		{
			frst=0;
			lst=N;
			while(frst<lst)
			{
				mid=(frst+lst)>>1;
				if(mid*mid+b*mid-n<0)frst=mid+1;
				else lst=mid;
			}
			if(frst*frst+b*frst-n==0&&f(frst,b))
			{
				fnd=1;
				cout<<frst<<"\n";
				break;
			}
		}
		if(!fnd)cout<<"-1\n";
	}
	return 0;
}
