#include <iostream>
#include <algorithm>
#include <climits>
using namespace std;

int main()
{
	int n;
	cin>>n;
	int k;
	cin>>k;
	long arr[n];
	int i=0;
	while(i<n)
	{
		scanf("%ld",&arr[i]);
		++i;
	}
	sort(arr,arr+n);
	int dif = INT_MAX;
	i=0;
	while(i+k-1<n)
	{
		if((arr[i+k-1]-arr[i])<dif)dif=arr[i+k-1]-arr[i];
		++i;
	}
	cout<<dif<<endl;
    return 0;
}
