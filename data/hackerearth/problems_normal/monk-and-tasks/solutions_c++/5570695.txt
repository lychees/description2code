#include<bits/stdc++.h>
#include<algorithm>
#include<utility>
using namespace std;
#define rep(i,n) for(lli i=0;i<n;i++)
#define lli long long int
int main()
{
    lli t,n;
    vector <lli> v[100];
   // vector<pair<lli,lli> > p;
    cin>>t;
    assert(t>=1 && t<=100);
    while(t--)
    {
        cin>>n;
        lli k,l;
        assert(n>=1 && n<=100000);
        lli maxm=-1;
        for(int i=0;i<100;i++)
        v[i].clear();
        rep(i,n)
        {
            cin>>k;
            lli c=0;
            l=k;
            while(l)
            {
                l=l&(l-1);
                c++;
            }
            v[c].push_back(k);
            if(c>maxm)
                maxm=c;
        }
        for(lli i=0;i<=maxm;i++)
        {
            for(lli j=0;j<v[i].size();j++)
            {
                    if(i==maxm && j==v[i].size()-1)
                        cout<<v[i][j]<<endl;
                    else
                        cout<<v[i][j]<<" ";
            }
        }

    }
    return 0;
}
