#include <iostream>
using namespace std;
long long nWays;
long long DP[31][31];

long long calComb(int n, int open, int closed)
{
	if(DP[open][closed] != -1)
		return DP[open][closed];
		
	//long long nWays=0;
	if(closed == n)
	{	
		DP[open][closed]=1;
		return 1;
	}
	
	DP[open][closed]=0;
	if(open > closed)
		DP[open][closed] +=  calComb(n, open, closed+1);
	if(open < n)
		DP[open][closed] += calComb(n, open+1, closed);
	
	//cout<<open<<","<<closed<<" ="<<nWays<<"\n";
	
	//DP[open][closed] = nWays;
	return DP[open][closed];
}

int main()
{
    int T;
    cin>>T;
    while(T--)
    {
    	int n;
    	scanf("%d", &n);
    	for(int i=0; i<=n; i++)
    	{
    		for(int j=0; j<=n; j++)
    		{
    			DP[i][j]=-1;
    		}
    	}
    	printf("%llu\n",calComb(n,0,0));
    }
    return 0;
}
