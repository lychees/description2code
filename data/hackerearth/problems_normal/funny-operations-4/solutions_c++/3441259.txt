#include <iostream>
#include <math.h>
using namespace std;


int main()
{
   	int t, n , arr[17], i, curr_p, p, curr_l, max_length, counter, j, xora, anda, flag;
   	unsigned int p_size;
   	cin>>t;
   	
   	while(t--)
   	{
   		cin>>n;
   		for(i = 0; i<n;i++)
   			cin>>arr[i];
   		p_size = pow(2, n);
   		p = 0;
   		
   		for(counter = 0; counter < p_size; counter++)
   		{
   			
   			xora = 0;
   			flag = 0;
   			curr_l = 0;
   			for(j=0; j < n; j++)
   			{
   				
   				if(counter & (1<<j))
   				{
   					xora ^= arr[j];
   					if( !flag )
   					{
   						anda = arr[j];
   						flag = 1;
   					}
   					else
   						anda &= arr[j];
   					curr_l += 1;
   				}
   			}
   			
   			curr_p = xora | anda;
   			if(curr_p > p)
   			{
   				p = curr_p;
   				max_length = curr_l;
   			}
   			if(curr_p == p && max_length < curr_l)
   				max_length = curr_l;
   			
   		}
   		
   		cout<<p<<" "<<max_length<<endl;
   		
   	}
    return 0;
}
