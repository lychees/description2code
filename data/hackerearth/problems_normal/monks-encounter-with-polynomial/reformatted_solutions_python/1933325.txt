'''
# Read input from stdin and provide input before running code
'''
ntest = int(raw_input())

for i in range(ntest):
    input_array_string = raw_input().split()

    a = int(input_array_string[0])
    b = int(input_array_string[1])
    c = int(input_array_string[2])
    k = int(input_array_string[3])

    left = 0
    right = k
    while (left <= right):
        mid = left+(right-left)/2
        if a*mid*mid+b*mid+c < k:
            left = mid+1
        else:
            right = mid-1
    print left
