#include <iostream>
#include <cmath>
#include <algorithm>
using namespace std;

typedef long long int ll;

int main()
{

   ll T,a,b,c,d,k;
   cin >> T;
   
   ll temp,cur;
   while(T--){
   	
   	cin >> a>> b>> c >> d>> k;
   	cur = (ll)(ceil(cbrt(k/a)));
   	temp = a*cur*cur*cur + b*cur*cur +c*cur + d;
   	
   	
   	while(temp > k && d < k){
   		cur --;
   		temp = a*cur*cur*cur + b*cur*cur +c*cur + d;	
   	}
   	cout << cur << endl;
   }
   
    return 0;
}
