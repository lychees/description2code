void fscani(int *x)
{
	int n=0;int sign=1;char c=getcx();
	while(c<'0' || c>'9'){if(c=='-') sign=-1;c=getcx();}
	while(c>='0' && c<='9'){n=(n<<1)+(n<<3)+(c-'0');c=getcx();}
	n=n*sign;*x=n;
}
