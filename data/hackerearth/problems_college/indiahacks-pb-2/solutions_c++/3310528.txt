#include <bits/stdc++.h>
using namespace std;
int main(){
	int t;
	string a;
	map <string,bool>mp;
	cin >> t;
	while(t--){
		int count = 0;
		cin >> a;
		if(mp[a]==true)
		  count = 1;
		
		reverse(a.begin(),a.end());
		mp[a] = true;
		
		if(count == 1){
			int k = a.size();
			cout << k << " " << a[k/2];
		} 
	}
	return 0;
}
