#include <iostream>
using namespace std;

int R,C,arr[101][101];
bool isvalid(int i, int j)
{
    if (i < 0 || i >= R || j >= C || j < 0) return false;
    return true;
}
 
void diagonalOrder()
{
    for (int k = 0; k < C; k++)
    {
        cout << arr[0][k] << " ";
        int i = k-1; 
        int j = 1;     
        while (isvalid(j,i))
        {
            cout << arr[j][i] << " ";
            i--;
            j++;   
        }
        cout << endl;
    }
 
    for (int k = 1; k < R; k++)
    {
        cout << arr[k][C-1] << " ";
        int i = C-2; 
        int j = k+1; 
        while (isvalid(j,i))
        {
            cout << arr[j][i] << " ";
            i--;
            j++; 
        }
        cout << endl;
    }
}
 

int main()
{
     int test=1,i,j;
     
     while(test--){
     cin>>R>>C;
     for(i=0;i<R;i++){
     	for(j=0;j<C;j++){
     		cin>>arr[i][j];
     	}
     }
    diagonalOrder();
     }
    return 0;
}
