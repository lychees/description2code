#include <bits/stdc++.h>
using namespace std;
#define ll long long int
ll ar[1000008];

ll call(ll n)
{
	if(n>1000000)
	{
		return max(n,(call(n/2)+call(n/3)+call(n/4)));
	}
	else
	{
		return ar[n];
	}
}

int main()
{
   // cout << "Hello World!" << endl;
 
   int n;
   	for(int i=0;i<=1000000;i++)
   	ar[i]=-1;
   	ar[1]=1;
   	ar[2]=2;
   	ar[3]=3;
   	ar[0]=0;
   	for(ll i=4;i<=1000000;i++)
   	{
   		ar[i]=max(i,(ar[i/2]+ar[i/3]+ar[i/4]));
   	}
   while((scanf("%d",&n))!=EOF)
   {
   	
 //  	cout<<"hua"<<endl;
   	printf("%lld\n",call(n));
   //	scanf("%d",&n);
   }
    return 0;
}
