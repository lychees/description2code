#include <bits/stdc++.h>

using namespace std;


int dp[3001][3001];

int gcd ( int a, int b)
{
	if ( a == 0)
		return b;
	return gcd( b%a,a);
}


int main() {
	
	dp[0][0] = 0;
	dp[1][1] = 1;
	
	
	for ( int i = 1; i <=3000; i++) {
		for ( int j = 1; j <= 3000; j++) {
			int l = gcd(i,j);
			//cout << i << " " << j << endl;
			dp[i][j] = dp[i-l][j-l] == 1?0:1;
			
		}
	}
	
	int t;
	scanf("%d", &t);
	while ( t--) {
		
		int q,r;
		string s;
		cin >> q >> r >> s;
		string t = s == "Chandu"?"Daspal":"Chandu";
		string ans = dp[q][r] == 1?s:t;
		cout << ans << endl;
	}
	return 0;
	
}
