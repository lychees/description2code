#include <iostream>
using namespace std;

int main()
{
    int t,i,j,f;
    string s;
    int prime[200];
    
    prime[1]=0;
    
    for(i=2;i<200;i++){
    	prime[i]=1;
    }
    
    for(i=2;i<200;i++){
    	if(prime[i]){
    		for(j=i*i;j<200;j+=i){
    			prime[j]=0;
    		}
    	}
    }
    
    cin>>t;
    while(t--){
    	cin>>s;
    	f=0;
    	for(i=0;i<s.size();i++){
    		if(prime[int(s[i])]){
    			continue;
    		}else{
    			cout<<s[i];
    			f=1;
    		}
    	}
    	if(f==0){
    		cout<<"NULL";
    	}
    	cout<<endl;
    }
    return 0;
}
