#include<iostream>
#include <cstdio>
#define MAX(x,y) (x>y?x:y)
using namespace std;
int main()
{
    int t;
   	scanf(" %d",&t);
   	 int lcs[1001][1001]={0};
    while(t--)
    {
       
        int i,j,n;
       	scanf(" %d",&n);
        int firstday[1001]={0},secondday[1001]={0},pos;
        i=1;
        while(i<=n)
        {
        
           	scanf(" %d",&pos);
            firstday[pos]=i;
            scanf(" %d",&pos);
            secondday[pos]=i;
            i++;
        }


        for(i=1;i<=n;i++)
        {
            for(j=1;j<=n;j++)
            {

                if(firstday[i]==secondday[j])
                {
                    lcs[i][j]=lcs[i-1][j-1]+1;
                }
                else
                {
                    lcs[i][j]=MAX(lcs[i-1][j],lcs[i][j-1]);
                }
        
            }
        }
        printf("%d\n",lcs[n][n]);
    }
    return 0;
}
