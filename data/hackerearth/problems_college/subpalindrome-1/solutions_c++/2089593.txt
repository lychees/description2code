#include<iostream>
#include<cstring>
using namespace std;
#define ll long long

ll max(ll a, ll b){
				return a>b?a:b;
}
ll L[101][101];
ll lps(char *str){
    ll n=strlen(str);
    ll i,j,cl,k,l,sum,ans=0;
    memset(L,0,sizeof(L));
    for(i=0;i<n;i++) L[i][i]=1;
    for(cl=2;cl<=n;cl++){
								for(i=0;i<n-cl+1;i++){
												j=i+cl-1;
												if(str[i]==str[j]){
												     sum=0;
																	for(k=i+1;k<=j-1;k++)
																	    for(l=k;l<=j-1;l++)
																	       sum+=L[k][l];
                 L[i][j]=sum+1;																				
												}			  
								}
    }
    for(i=0;i<n;i++)
			    for(j=0;j<n;j++)
				     ans+=L[i][j];
		return ans;
}
 

int main(){
    ll t;
    char str[100000];
    cin>>t;
    while(t--){
				    cin>>str;
				    cout<<lps(str)<<endl;
				}
				return 0;
}
