#include<bits/stdc++.h>
#include<cmath>
#define ll long long
using namespace std;
int main()
{
	long int num,d[1000000],b[1000000],diff1[1000000],diff2[1000000],i;
	cin >> num;
	for(i = 0;i < num;i++)
	{
		cin >> d[i];
	}
	for(i = 0;i < num;i++)
	{
		cin >> b[i];
	}
	for(i = 0;i< num-1;i++)
	{
		diff1[i] = abs(d[i] - d[i+1]);
		diff2[i] = abs(b[i] - b[i+1]);
	}
	/*for(i = 0;i < num-1;i++)
	{
		cout << diff1[i];
	}
	cout << endl;
	for(i = 0;i < num-1;i++)
	{
		cout  << diff2[i];
	}*/
	sort(diff1,diff1+(num-1));
	sort(diff2,diff2+(num-1));
	if(diff1[num-2] > diff2[num-2])
	{
		cout << "Dom" <<endl << diff1[num-2];
	}
	else
	{
		cout << "Brian" << endl << diff2[num-2];
	}
	return 0;
}
