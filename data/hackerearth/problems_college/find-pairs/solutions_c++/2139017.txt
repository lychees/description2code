#include <iostream>
using namespace std;

int main()
{	int t;
	cin >>t;
	while(t--){
		long n;
		cin >> n;
		long arr[n];
		for(long i=0;i<n;i++){
			cin >> arr[i];			
		}
		long count=0;
		for(long i=0;i<n;i++){
			for(long j=i;j<n;j++){
				if(arr[i]==arr[j]) count++;
			}
		}
		cout << count<<endl;
	}

    return 0;
}
