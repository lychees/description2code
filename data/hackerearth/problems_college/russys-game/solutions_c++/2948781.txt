#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		int n;
		scanf("%d",&n);
		int ar[n],vis[n];
		for(int i=0;i<n;i++){
			scanf("%d",&ar[i]);
			vis[i]=0;
		}
		int re=0;
		int sr=0;
		while(1){
			if(vis[sr]==1){
				re=1;break;
			}
			if(ar[sr]==-1)
			break;
			vis[sr]=1;
			sr=ar[sr];
		}
		if(re==1)
		printf("Lose\n");
		else
		printf("Win\n");
	}
	return 0;
}
