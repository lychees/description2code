#include<iostream>
using namespace std;
long long int i,j,t,n,x,a[1000000],c,s;
main()
{
 
cin>>t;
while(t--)
{
c=0;
cin>>n>>x;
for(i=0;i<n;i++)
cin>>a[i];
for(i=0;i<n;i++)
{s=0;
for(j=i;j<n;j++)
{s=s+a[j];
if(s==x) c=2;
if(s>=x) break;}
if(c==2)break;
}
if(c==2)cout<<"YES"<<endl;
else  cout<<"NO"<<endl;
}
}
