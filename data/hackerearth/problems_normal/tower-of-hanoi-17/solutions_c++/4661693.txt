#include <bits/stdc++.h>
using namespace std;

pair<long long, long long> p[205];

int main()
{
    long long T;
    cin >> T;
    while (T--)
    {
    	int N;
    	cin >> N;
    	for(long long i=0; i<N; i++)
    	{
    		long long r, h;
    		cin >> r >> h;
    		p[i].first = r;
    		p[i].second = h;
    	}
    	
    sort(p, p+N);
    reverse(p, p+N);
    
    long long dp[205];
    memset(dp, 0, sizeof(dp));
    
    dp[0] = p[0].second;
    for(long long i=1; i<N; i++)
    {
    	dp[i] = p[i].second;
    	for(long long j=0; j<=i-1; j++)
    	{
    		if(p[i].second < p[j].second && p[i].first < p[j].first)
    			dp[i] = max(dp[i], dp[j] + p[i].second);
    	}
    }
    
   long long maxht = 0;
   for(long long i=0; i<N; i++)
   {
   	if(dp[i] > maxht) maxht = dp[i];
   }
   
   cout << maxht << endl;
  
    }
    return 0;
}
