#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,x;
    set<int> s;
    scanf("%d",&t);
    for(int i=0;i<t;++i){
    	scanf("%d",&x);
    	auto it=s.lower_bound(x);
    	if(it!=s.end()){
    		s.erase(it);
    	}
    	s.insert(x);
    }
    int l=s.size();
    vector<int> res;
    while(l>0){
    	res.push_back(l%2);
    	l/=2;
    }
    for(int i=res.size()-1;i>=0;--i){
    	printf("%d",res[i]);
    }
    return 0;
}
