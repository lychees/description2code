#include<cstdio>
#include<cstdlib>
#include<iostream>
#include<utility>
#include<vector>

using namespace std;
#define REP(i,a,b) \
    for(int(i)=int(a);i<=int(b);i++)

int boys[10005],girls[10005];

bool mycompare(int a, int b){
    return a>b;
}

int main(){
    int T,N,result;
    scanf("%d",&T);
    while(T--){
        scanf("%d",&N);
        result = 0;
        REP(i,0,N-1)
            scanf("%d",&girls[i]);
        REP(i,0,N-1)
            scanf("%d",&boys[i]);
        sort(girls,girls+N);
        sort(boys,boys+N,mycompare);
        REP(i,0,N-1){
            if(girls[i]%boys[i]==0 || boys[i]%girls[i]==0)
                result++;
        }
        printf("%d\n",result);
    }
    return 0;
}
