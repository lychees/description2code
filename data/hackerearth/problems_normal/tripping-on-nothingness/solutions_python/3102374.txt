def gcd(u, v):
    while v:
        u, v = v, u % v
    return abs(u)
        
s,cases = map(int,raw_input().split())
temp = set()
while(cases != 0):
    cases = cases - 1
    a = int(raw_input())
    t = gcd(s,a)
    if t in temp:
        print(-1)
    else:
        temp.add(t)
        print(t)
