#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <string>
#include <vector>

using namespace std;

int mark[1111][22];
int o[22];
int p[1111];
int p2[1111];
int cnt;

int main() {
    ios::sync_with_stdio(0);
    int n, k;
    scanf("%d %d", &n, &k);
    for(int i = 0; i < n; ++i)
        for(int j = 1; j <= k; ++j)
            scanf("%d", &mark[i][j]);
    
    int q;
    scanf("%d", &q);
    while (q --> 0) {
        scanf("%d", &cnt);
        for(int i = 0; i < cnt; ++i)
            scanf("%d", &o[i]);
        
        for(int i = 0; i < n; ++i)
            p[i] = i;
        
        int sz = n;
        for(int j = 0; j < cnt; ++j) {
            int mn = 1e9;
            for(int i = 0; i < sz; ++i)
                mn = min(mn, mark[p[i]][o[j]]);
            
            int nsz = 0;
            for(int i = 0; i < sz; ++i)
            if (mark[p[i]][o[j]] == mn)
                p2[nsz++] = p[i];
            
            
            sz = nsz;
            for(int i = 0; i < sz; ++i)
                p[i] = p2[i];
        }
        printf("%d\n", p[0]+1);
    }
    return 0;
}
