#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
	int t;
	long int n,res;
	cin>>t;
	
	while(t--)
	{cin>>n;
	long int a[n];

	for(long int i=0;i<n;i++)
	{
		cin>>a[i];
	
	
	}
	sort(a,a+n);
	int sum=1;res=0;
	for(int i=1;i<n;i++)
	{
		if(a[i]==a[i-1])
		sum++;
		else
		{
		res=res+sum*(sum+1)/2;
		sum=1;
		
		}
	}
	
	res=res+sum*(sum+1)/2;
	//res=res+n;
	cout<<res<<endl;

	}
   
    return 0;
}
