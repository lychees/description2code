for i in range(input()):
    a = input()
    arr = map(int, raw_input().strip().split())
    d = dict()
    for i in arr:
        if i in d:
            d[i] += 1
        else:
            d[i] = 1
    min_count = min(d[i] for i in d)
    if min_count < a/len(d):
        print 'bad luck'
    else:
        print 'can do'
