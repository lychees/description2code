#include <bits/stdc++.h>
#include<vector>
using namespace std;
int main()
{
    int t,n,i,j,k;
    long long int l;
    string s;
    cin>>t;
    while(t--){
    cin>>s;
   n=s.size();
    l=0;
      for(i=0;i<n;i++){
    	for(j=i,k=i;j>=0&&k<n;j--,k++){
    		if(s[j]==s[k])
    		l+=(k-j+1)*(k-j+1);
    		else break;
    	}
    }
    for(i=0;i<n-1;i++){
    	for(j=i,k=i+1;j>=0&&k<n;j--,k++){
    		if(s[j]==s[k])
    		l+=(k-j+1)*(k-j+1);
    		else break;
    	}
    }

    
    
    cout<<l<<endl;
   
    }
    
  
}
