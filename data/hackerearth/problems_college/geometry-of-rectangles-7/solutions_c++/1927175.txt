#include <bits/stdc++.h>
using namespace std;
int overlaps( int x11,int y11,int x12, int y12,int x21,int y21,int x22,int y22)
{
	if(x12<=x21||x22<=x11)
	return 1;
	if(y12<=y21||y22<=y11)
	return 1;
	return 0;
}
int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   		int x1,y1,x2,y2;
   		int x3,y3,x4,y4;
   		cin>>x1>>y1>>x2>>y2;
   		cin>>x3>>y3>>x4>>y4;
   		if((x1>x4)||(x3>x2)||(y3>y2)||(y1>y4))
   		{
   			cout<<"Apart"<<endl;
   		}
   		else if((x1==x4)||(x3==x2)||(y2==y3)||(y1==y4))
   		{
   			if((x2==x3&&y2==y3)||(x1==x4&&y1==y4)||(x1==x4&&y2==y3)||(x2==x3&&y1==y4))
   			{
   				cout<<"Point"<<endl;
   			}
   			else
   			{
   				cout<<"Line"<<endl;
   			}
   		}
   		else
   		{
   			cout<<"Rectangle"<<endl;
   		}
   }
}
