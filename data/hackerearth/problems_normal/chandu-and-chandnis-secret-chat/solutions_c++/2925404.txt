#include <iostream>
#include <algorithm>
using namespace std;
string s;
bool mycomp(int i, int j)
{
	return s[i]!=s[j] ?  s[i]>s[j] : i < j;
}
int main()
{
    //string s;
    int t, n;
    cin>>t;
    while(t--)
    {
    	cin>>s>>n;
    	n--;
    	int in[10001];
    	int i;
    	for(i=0;i<s.length();++i) in[i]=i;
    	sort(in, in + i, mycomp);
    	string ans="";
    	for(int j=0;j<s.length();++j, n= in[n] ) ans+=s[n];
    	cout<<ans<<endl;
    }
    return 0;
}
