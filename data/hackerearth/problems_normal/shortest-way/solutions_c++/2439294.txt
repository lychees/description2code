#include <iostream>
using namespace std;


void find(long long x, long long y) {
	long long ans=0;
    while(x>1 || y>1)
    {
        if(x<y)
        {
            long long mod=y/x;
            y=y-mod*x;
            ans+=mod;
        }
        else
        {
            long long mod=x/y;
            x=x-mod*y;
            ans+=mod;
        }
    }
    if(ans>0) ans--;
    cout<<ans<<endl;	
}
int main()
{
    int T;
    cin>>T;
    while(T--) {
    	long long a, b;
    	cin>>a>>b;
    	find(a,b);
    }
    return 0;
}
