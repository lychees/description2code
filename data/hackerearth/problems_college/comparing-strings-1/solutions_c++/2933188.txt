#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	string a, b;
	while(t--) {
		cin>>a>>b;
		int freq[26] = {0};
		int flag = 0;
		int len1 = a.length(), len2 = b.length();
		for(int i=0;i<len1;i++) {
			freq[a[i]-'a']++;
		}
		for(int i=0;i<len2;i++) {
			if(freq[b[i]-'a'] != 0) {flag = 1; break;}			
		}
		if(flag == 1) cout<<"YES\n";
		else cout<<"NO\n";
	}
    return 0;
}
