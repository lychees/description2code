#include <iostream>
#include <map>
using namespace std;

int main()
{
    int val,mx=0,cnt,n,m,temp;
    map<int,int> data;
    cin>>n>>m;
    for(int i=0;i<n;i++)
    {
    	cin>>temp;
    	data[temp]++;
    	cnt=data[temp];
    	if(cnt>mx)
    	{
    		val=temp;
    		mx=cnt;
    		cout<<val<<" "<<mx<<endl;
    	}
    	else if(cnt==mx)
    	{
    		val=max(val,temp);
    		cout<<val<<" "<<mx<<endl;
    	}
    	else
    	{
    		cout<<val<<" "<<mx<<endl;
    	}
    }
    return 0;
}
