#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int t,i,j,k;
	cin>>t;
	for(i=0;i<t;i++)
	{
		string a,b;
		cin>>a>>b;
		long long int la,lb;
		la=a.size();
		lb=b.size();
		if(la!=lb)
		cout<<"No\n";
		else
		{
			k=0;
			long long int ha[26]={0},hb[26]={0};
			for(j=0;j<la;j++)
			{
				ha[a[j]-97]+=1;
			}
			for(j=0;j<lb;j++)
			{
				hb[b[j]-97]+=1;
			}
			for(j=0;j<26;j++)
			{
				if(ha[j]!=hb[j])
				{
					k=1;
					break;
				}
			}
			if(k==1)
			cout<<"No";
			else
			cout<<"Yes";
			cout<<"\n";
		}
	}
	return 0;
}
