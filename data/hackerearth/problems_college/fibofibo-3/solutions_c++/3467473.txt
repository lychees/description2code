#include <bits/stdc++.h>
#define ll long long int
using namespace std;

int fib(int n, int *c1, int *c2) {
    if (n == 0) {
        (*c1)++;
        return 0;
    }
    if (n == 1) {
        (*c2)++;
        return 1;
    }
    return fib(n - 1, c1, c2) + fib(n - 2, c1, c2);
}

int main()
{
    int t, n, c1, c2;
    cin >> t;
    while (t--) {
        cin >> n;
        c1 = 0;
        c2 = 0;
        int tmp = fib(n, &c1, &c2);

        cout << c1 << " " << c2 << endl;
    }

    return 0;
}
