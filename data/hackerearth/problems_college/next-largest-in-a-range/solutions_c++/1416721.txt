#include <iostream>
using namespace std;

int main() {
	int n,l,r,Q;
	cin>>n>>l>>r>>Q;
	int a[n];
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
	}
	while(Q--)
	{
		int X,small=1000000,f = 0;
		cin>>X;
		for(int i=l;i<=r;i++)
		{
				if(a[i] > X && a[i] < small)
				{
					small = a[i];
					f = 1;
				}
		}
		if(f==0)
			cout<<"-1"<<endl;
		else
			cout<<small<<endl;
	}
	return 0;
}
