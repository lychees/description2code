#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n,p,sum=0;
    cin>>n;
    vector<int>a;
    for(int i=0;i<n;i++)
    {
    	cin>>p;
    	a.push_back(p);
    }
    for(int i=0;i<a.size();i++)
    {
    	sum=sum+a[i];
    }
    if(sum==2*n-2)
    {
    	cout<<"Yes\n";
    }
    else
    {
    	cout<<"No\n";
    }
    return 0;
}
