#include <iostream>
using namespace std;

int main()
{
    int sum=0,max = 0;
    int n;
    cin>>n;
    int arr[n];
    for(int i =0;i<n;i++)
    {
    	cin>>arr[i];
    	sum = sum+arr[i];
    	
    	if(sum<=0)
    	{
    		sum = 0;
    	}
    	else
    	{
    		if(sum>max)
    		{
    			max = sum;
    		}
    	}
    	
    }
    cout<<max;
    
    
}
