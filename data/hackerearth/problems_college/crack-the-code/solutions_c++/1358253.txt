#include <algorithm>
#include <iostream>
#include <string>

int f[123], c[123456];

int main() {
    std::cin.tie(NULL);
    std::ios::sync_with_stdio(false);
    f[0] = f[1] = 1;
    for (int i = 2; i < 50; ++i) {
        f[i] = f[i - 1] + f[i - 2];
    }
    for (int i = 1, j = 0; i < 123456; ++i) {
        while (f[j] <= i) ++j;
        c[i] = j;
    }
    int tt;
    std::cin >> tt;
    while (tt--) {
        int f, t;
        std::cin >> f >> t;
        std::cout << ((c[t] - c[f - 1]) & 1 ? "INVALID" : "VALID") << '\n';
    }
}
