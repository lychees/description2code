#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    
    while(t--)
    {
    	int n,q;
    	scanf("%d%d",&n,&q);
    	vector<int> val(n);
    	int i=0;
    	int x;
    	for(i;i<n;++i)
    	{
    		scanf("%d",&x);
    		val[i]=x;
    	}
    	sort(val.begin(),val.end());
    	long diff = 0; 
    	for(i=1;i<n;++i)
    	{
    		diff+=val[i]-val[i-1];
    	}
    	diff = diff*q;
    	printf("%ld\n",diff);
    }
    return 0;
}
