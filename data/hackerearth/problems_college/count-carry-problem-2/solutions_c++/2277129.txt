#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>> t;
    while(t--)
    {
       long long  int r1,r2,a,b,c;
        cin>>a>>b;c=0;
        while(a>0||b>0)
        {
            r1=a%10;
            r2=b%10;
            if((r1+r2)>9) c++;
            a=a/10;b=b/10;
        }
        if(c==0) cout<<"No carry operation"<<endl;
        else if(c==1) cout<<"1 carry operation"<<endl;
        else cout<<c<<" carry operations"<<endl;

    }

}
