#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int dp[1000005];
void pre(){
dp[0] = 0;
dp[1] = 0;
for(int i=2;i<=1000000;i++){
	   if(dp[i]>0) continue;
	for(int j=i;j<=1000000;j+=i){
		int k = j;
		while(k%i==0){
			dp[j]++;
			k/=i;
		}
	}
}
for(int i=2;i<=1000000;i++){
	dp[i] = dp[i-1]+dp[i];
}
}	
int main(){
	int t,a,b;
	pre();
	scanf("%d",&t);
	while(t--){
	 scanf("%d %d",&a,&b);
	 printf("%d\n",dp[b]-dp[a-1]);
	}
    return 0;
}
