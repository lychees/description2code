int is_prime(long long int n)
{
	if(n<2){return 0;}
	else if(n==2){return 1;}
	else if(n%2==0){return 0;}
	else
	{
		int flag=1;
		for(long long int i=3;i<=sqrt(n);i+=2)
		{
			if((n%i)==0){flag=0;break;}
		}
		if(flag){return 1;}
		else {return 0;}
	}
}

std::ios_base::sync_with_stdio(false);

while(t--)
{
  long long n;cin>>n;
  if(is_prime(n)){cout<<"YES"<<endl;}
  else {cout<<"NO"<<endl;}
}
return 0;
