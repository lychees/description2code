#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<string>
using namespace std;
const int md=1e9+7;
const int nmax=1e6+5;
int fib[nmax];
int main() {
    int t;
    scanf("%d",&t);
    fib[1]=2;
    fib[2]=3;
    for(int i=3;i<nmax;i++) {
        fib[i]=fib[i-1]+fib[i-2];
        if(fib[i]>=md) fib[i]-=md;
    }
    for(int i=2;i<nmax;i++) {
        fib[i]+=fib[i-1];
        if(fib[i]>md) fib[i]-=md;
    }
    while(t--) {
        int n;
        scanf("%d",&n);
        printf("%d\012",fib[n]);
    }
    return 0;
}
