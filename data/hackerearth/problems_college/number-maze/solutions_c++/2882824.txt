#include<iostream>

using namespace std;

int main()
{
	int j,n,m,i;

	cin>>n>>m;
	int temp;
	if(m<n)
	{
		temp=m;
		m=n;
		n=temp;
	}

	i=n+1;
	while(i<m)
	{
		if(i%15==0 || (i%3!=0 && i%5!=0))
		{
			cout<<i;
			break;
		}
		i++;
	}

	for(j=i+1;j<m;j++)
		if(j%15==0 || (j%5!=0 && j%3!=0))
			cout<<","<<j;
	cout<<"\n";		
	
	return 0;
}
