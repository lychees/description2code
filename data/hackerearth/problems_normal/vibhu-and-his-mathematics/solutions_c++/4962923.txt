#include <iostream>
#include<vector>
#include<cmath>
using namespace std;

int main()
{
    int T;
    cin>>T;
    long long po = pow(10, 9) + (long long)7;
    long long val = pow(10, 6);
	vector<long long>res(val,1);
	for(long long j=2;j<=val;j++)
	{
		res[j] = (res[j-1] + (j-1)*res[j-2]%po)%po;
	}
    for(int i =0;i<T;i++)
    {
    	long long x;
    	cin>>x;
    	if(x == 0) return 0;
    
    	cout<<res[x]<<endl;
    }
    return 0;
}
