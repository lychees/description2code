#include <iostream>
using namespace std;

int main()
{
    unsigned long long N;
    int T;
    int Q;
    
    cin >> Q;
    while(Q--)
    {
    	cin >> N;
    	cin >> T;
    	while(T--)
    	{
    		if(N & 1 == 0)
    			N -= N / 2;
    		else
    			N -= ((N+1) / 2);
    		
    		N -= N / 4;
    	}
    	
    	cout << N << endl;
    }
    
    return 0;
}
