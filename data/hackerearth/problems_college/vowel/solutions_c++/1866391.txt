#include<iostream>

using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string n;
		cin>>n;
		int t=0;
		int s=0;
		while(n[s]!='\0')
		{
			if(n[s]=='a'||n[s]=='A'||n[s]=='e'||n[s]=='E'||n[s]=='i'||n[s]=='I'||n[s]=='o'||n[s]=='O'||n[s]=='u'||n[s]=='U')
			{
				cout<<n[s];
				t++;
			}
			s++;
		}
		if(t==0)
		cout<<"No"<<endl;
		else
		cout<<endl;
	}
	return 0;
}
