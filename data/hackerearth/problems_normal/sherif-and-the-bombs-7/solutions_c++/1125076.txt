#include <bits/stdc++.h>
using namespace std;
#define INF 0x3f3f3f3f
#define REP(i,n) for(int i=0; i<n; i++)
#define R(x) scanf("%d",&x)
typedef long long int64;
typedef pair<int,int> pii;

#define N 1001
int x[N],y[N],r[N];

int main() {
	int n;

	R(n);
		REP(i,n) {
			R(x[i]),R(y[i]),R(r[i]);
			r[i]*=r[i];
		}

		int res=0;

		for (int a=1;a<=1000;a++)
		for (int b=1;b<=1000;b++) {
			int q=0;
			REP(i,n) {
				q+=((x[i]-a)*(x[i]-a)+(y[i]-b)*(y[i]-b)<=r[i]);
				if (q>=2) break;
			}

			res+=(q>=2);
		}

		printf("%d\n",res);
	return 0;
}
