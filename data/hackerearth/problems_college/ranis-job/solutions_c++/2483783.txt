#include <bits/stdc++.h>

using namespace std;
typedef long long  ll;

int main()
{
    ios_base::sync_with_stdio(0); 
    cin.tie(0);
    int t;
    cin>>t;
    while(t--)
    {
        ll n,a[100099],b[100099];
        cin>>n;
        ll i;
        ll sum=0;
        for(i=0;i<n;i++)
        {
            cin>>a[i];
            sum=sum+a[i];
        }
        for(i=0;i<n;i++)
        {
            cin>>b[i];
        }
        sort(b,b+n);
        ll c=1;
        //cout<<"sum is "<<sum<<endl;
        ll sum1=0;
        for(i=n-1;i>=0;i--)
        {
            sum1=sum1+b[i];
            if(sum<=sum1)
            {
               break;
            }
            else
            {
                c++;
                //cout<<"yes";
            }
        }
        cout<<c<<endl;
    }
    return 0;
}
