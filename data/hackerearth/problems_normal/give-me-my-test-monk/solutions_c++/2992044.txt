#include <bits/stdc++.h>

#define FORE(i, a, b) for(int i = (a), _b = (b); i <= b; ++i)
#define FORD(i, a, b) for(int i = (a), _b = (b); i >= b; --i)

#define ST first
#define ND second

using namespace std;
typedef long long LL;
typedef pair<int, int> II;

const int MAXN=10+100;
const int INF=1000000007;

/// Khai bao
int i,j,n,m;
int l, r;
LL k, t;
LL res;
LL d[MAXN], F[MAXN][MAXN];

LL power(LL x, LL y, LL z) {
    if (y==0) return 1;
    if (y==1) return x%z;
    LL mid=y/2;
    LL kk=power(x, mid, z);
    kk=(kk*kk)%z;
    if (y%2==0) return kk;
    else return (kk*x)%z;
}


int main() {
#ifndef ONLINE_JUDGE
    //freopen("code.inp", "r", stdin);
    //freopen(".out", "w", stdout);
#endif
    scanf("%d%d%d",&n, &l ,&r);
    d[0]=1;
    d[1]=0;
    FORE (i, 2, n) d[i]=(i-1)*(d[i-1]+d[i-2])%INF;
    FORE (i, 0, n) FORE (j, 0, n)
     if (i==j || j==0) F[i][j]=1;
     else F[i][j]=(F[i-1][j] + F[i-1][j-1]) % INF;
    res=0;
    FORE (i, l, r) res=res+(F[n][i]*d[i])%INF, res=res%INF;
    k=1;
    FORE (i,1,n) k=k*i%INF;
    res=res*power(k, INF-2, INF);
    cout<<res%INF;






    return 0;
}
