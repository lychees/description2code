#include <bits/stdc++.h>
using namespace std;

int main(){
    int n, i=0; 
    cin>>n;
    int arr[n];
    for(i=0;i<n;i++) cin>>arr[i];
    for(i=0;i<n-1;i++){
    	if(arr[i]>arr[i+1]) return cout<<"NO",0;
    	arr[i+1]-=arr[i];
    	arr[i]= 0;
    }
    if(arr[i]==0) cout<<"YES";
    else cout<<"NO";
    return 0;
}
