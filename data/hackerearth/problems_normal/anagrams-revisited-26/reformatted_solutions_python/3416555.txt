t = input()
x = dict()
while t > 0:
    y = raw_input()
    a = ''.join(sorted(y))
    try:
        k = x[a]
    except KeyError:
        k = 0
    x[a] = k+1
    t = t-1
b = -1
for i in x:
    if x[i] >= b:
        b = x[i]
print b
