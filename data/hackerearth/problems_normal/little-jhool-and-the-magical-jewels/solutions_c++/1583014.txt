#include <iostream>
#include <algorithm>
#include <vector>
#include <string.h>
using namespace std;

void min(int r,int u,int b,int y)
{
	if(r<=u && r<=b && r<=y)
	cout<<r<<endl;
	else if(u<=r && u<=b && u<=y)
	cout<<u<<endl;
	else if(b<=r && b<=u && b<=y)
	cout<<b<<endl;
	else if(y<=r && y<=u && y<=b)
	cout<<y<<endl;
}

int main()
{
    int t,r=0,u=0,b=0,y=0,i,j;
    char x,s[101];
    cin>>t;
    
    //vector<string> ruby;
    
    while(t--)
    {
    	cin>>s;
    	r=0;b=0;u=0;y=0;
    	
    	for(i=0;i<strlen(s);i++)
    	{
    	//	x=s[i];
    		
    		if(s[i]=='r')
    		r++;
    		if(s[i]=='u')
    		u++;
    		if(s[i]=='b')
    		b++;
    		if(s[i]=='y')
    		y++;
    	}
    	
    	min(r,u,b,y);
    	
    }
}
