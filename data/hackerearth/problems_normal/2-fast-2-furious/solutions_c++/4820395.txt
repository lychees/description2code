#include<iostream>
#include<math.h>
using namespace std;
int main()
{
long int diffa,diffb,a_max,b_max,i,j,n;
cin>>n;
a_max=0;
b_max=0;
long int a[1000000],b[1000000];
for(i=0;i<n;i++)
{
cin>>a[i];
}
for(i=0;i<n;i++)
{
cin>>b[i];
}
for(j=0;j<=n-2;j++)
{
diffa=abs(a[j]-a[j+1]);
diffb=abs(b[j]-b[j+1]);
if(a_max<diffa)
a_max=diffa;
if(b_max<diffb)
b_max=diffb;
}
if(a_max>b_max)
cout<<"Dom\n"<<(a_max);
else if(a_max<b_max)
cout<<"Brian\n"<<(b_max);
else
cout<<"Tie";
}
