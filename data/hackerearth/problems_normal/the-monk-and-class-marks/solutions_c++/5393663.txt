#include<bits/stdc++.h>
#include<utility>
using namespace std;
int main(){
	
		vector< pair<int,string> > v;
		int n;
		cin>>n;
		while(n--){
			pair <int, string> p;
			string s;
			int x;
			cin>>s;
			cin>>x;
			p = make_pair(x,s);
			v.push_back(p);
		}
		sort(v.begin(),v.end());
		vector< pair<int,string> >::iterator i,j,k;
		for(i=(v.end()-1);i>=v.begin();i--){
			j=i;
			while(i!=v.begin()){
				if((*i).first==(*(i-1)).first){
					i--;
				}
				else
				break;
			}
			for(k=i;k<=j;k++)
			cout<<(*k).second<<" "<<(*k).first<<endl;
		}
	return 0;
}
