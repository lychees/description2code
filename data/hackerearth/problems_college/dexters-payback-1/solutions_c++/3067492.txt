#include <iostream>

using namespace std;

int main()
{
    int t,r,c;
    long int s;
    cin>>t;
    while(t--)
    {   s=0;
        cin>>r>>c;
        if(c%4==0){cout<<(c/2)*r<<endl;}
        if(c%4==1){
            s=(4*(c/2)+2)*(r/4);
            if(r%4==1){s+=c/2+1;}
            if(r%4==2){s+=2*(c/2)+2;}
            if(r%4==3){s+=3*(c/2)+2;}
            cout<<s<<endl;
        }
        if(c%4==2){
           s=(4*(c/2))*(r/4);
            if(r%4==1){s+=c/2+1;}
            if(r%4==2){s+=2*(c/2)+2;}
            if(r%4==3){s+=3*(c/2)+1;}
            cout<<s<<endl;
        }
        if(c%4==3){
             s=(4*(c/2)+2)*(r/4);
            if(r%4==1){s+=c/2+1;}
            if(r%4==2){s+=2*(c/2)+2;}
            if(r%4==3){s+=3*(c/2)+2;}
            cout<<s<<endl;
        }
    }
    return 0;
}
