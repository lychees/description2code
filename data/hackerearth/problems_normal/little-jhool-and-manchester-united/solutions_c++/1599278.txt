//HackerEarth--Little Jhool and Manchester United
#include<bits/stdc++.h>

using namespace std;

double dp[1<<16][16];
int reqdPlayers,totalPlayers,prob[16],totalMoney,cost[16];

double solve(int mask,int boysTaken){
	
	if(boysTaken==reqdPlayers) return 1.0;
	
	if(mask==(1<<totalPlayers)-1) return 0.0;
	
	
	
	if(dp[mask][boysTaken]>=0.0) return dp[mask][boysTaken];
	
	double ans=0.0,p;
	int i,remMoney=totalMoney;
	
	for(i=0;i<totalPlayers;i++)	
		if(((mask)&(1<<i)))
		  remMoney-=cost[i];
		
	for(i=0;i<totalPlayers;i++)	
		if(((mask)&(1<<i))==0){
			
			p=(double )prob[i]/100.0;
			
			if(remMoney>=cost[i])
			ans=max(ans,p*solve(mask|(1<<i),boysTaken+1)+(1.0-p)*solve(mask|(1<<i),boysTaken));
			
		}	
		
	return dp[mask][boysTaken]=ans;
	
}

int main(void){
	
	int testCases;
	
	scanf("%d",&testCases);
	
	while(testCases--){
		
		scanf("%d%d%d",&totalPlayers,&reqdPlayers,&totalMoney);
		
		for(int i=0;i<totalPlayers;i++)
		   scanf("%d%d",&cost[i],&prob[i]);
		
		memset(dp,-1,sizeof dp);
		
		printf("%lf\n",solve(0,0));
		
	}
	
	return 0;
	
}
