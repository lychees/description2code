#include <bits/stdc++.h>
#define N 10005

using namespace std;

int t, n, a[N];
long long dp[N];

int main(){
 //   freopen ("in.inp", "r", stdin);
    cin >> t;
    while (t--) {
        cin >> n;
        for (int i = 1; i <= n; i++)
            cin >> a[i];
        memset (dp, 0, sizeof(dp));
        dp[1] = a[1];
        long long res = a[1];
        for (int i = 2; i <= n; i++) {
            long long tmp = 0;
            for (int j = 0; j <= i-2; j++)
                tmp = max(tmp, dp[j]);
            dp[i] = tmp + a[i];
            res = max(res, dp[i]);
        }
        cout << res << endl;
    }
    return 0;
}
