#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int i,j,n;
    cin >> n;
    int dp[n];
    long long int arr[n];
    for(i=0;i<n;i++){
    	dp[i]=1;
    	cin >> arr[i];
    }
    sort(arr,arr+n);
    for(i=0;i<n;i++){
    	for(j=0;j<i;j++){
    		if(arr[i]%arr[j]==0){
    			int county = dp[j]+1;
    			if(dp[i]<county) dp[i]=county;
    		}
    	}
    }
    int maxi=0;
    for(i=0;i<n;i++){
    	if(dp[i]>maxi) maxi = dp[i];
    //	cout << dp[i]<< endl;
    }
    
    if(maxi>1)cout << maxi;
    else cout << -1;
    return 0;
}
