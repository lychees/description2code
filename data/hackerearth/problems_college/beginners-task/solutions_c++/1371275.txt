#include<iostream>
using namespace std;
static int T,N,L[100000],i,j,ctr=0,R[100000],temp;
main()
{
	cin>>T;
	for(i=0;i<T;i++)
	{
		cin>>N;
		for(j=0;j<N;j++)
		{
		cin>>L[i];
		temp=L[i];
		if(temp%2==0)
		ctr++;	
		}
		R[i]=ctr;
		ctr=0;
	}
	for(i=0;i<T;i++)
	{
		cout<<R[i]<<"\n";
	}
}
