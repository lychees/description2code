#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	while(1)
	{
		map<char,pair<int,int> > m;
		char str[110];
		scanf("%s",str);//Order in which visitors came checking for room
		if(strcmp(str,"0")==0)
		break;
		int r; //indicating number of available rooms
		char c;
		int i,ans;
		pair<int,int> p;
		for(i=0;i<26;i++)
		{
				m['A'+i]=make_pair(0,0);//Not arrived yet
		}
		scanf("%d",&r);
		ans=0;
		for(i=0;str[i]!='\0';i++)
		{
			c=str[i];
			p=m[c];
			//~ cout<<c<<" "<<p.first<<" "<<p.second<<endl;
			if(p.first==0)//Arrived
			{
				if(r==0)
				{
					ans++;
					p.first=2;//No room vacant/available
					m[c]=p;
				}
				else
				{
					p.first=1;//Aloted room
					m[c]=p;
					r--; //Available room count decreases by 1
				}
			}
			else if(p.first==2)//Person who couldn't be alloted room left
			{
				p.first=0;
				p.second=0;
				m[c]=p;
			}
			else //Person alloted room once is vacanting the roon
			{
				p.first=0;
				p.second=0;
				m[c]=p;
				r++;
			}
		}
		printf("%d\n",ans);
	}
	return 0;
}
