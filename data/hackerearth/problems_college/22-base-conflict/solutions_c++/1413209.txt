#include <iostream>
#include<math.h>
#include<string.h>
using namespace std;

int main()
{
    long long n;
    long long k;
    int lim=2;
    long long sum[17]={0};
    cin>>n;
    char kstr[1000000];
    long long p,rem;
    for(long long i=0;i<n;i++)
    {
    	cin>>kstr;

    	int temp;
    	p=0;
    	for(int j=strlen(kstr)-1;j>=0;j--)
    	{	
    		temp=kstr[j];
    		if((temp>=65)&&(temp<=90))
    		{
    			temp=temp-55;
    		}
    		else
    		{
    			temp=temp-48;
    		}
    		lim=max(lim,temp+1);
    		
    		for(long long a=2;a<=16;a++)
    		{
    			sum[a]=sum[a]+temp*pow(a,p);	
    		}
    		++p;   		

    	}
    	
    }
    long long acts;
    cin>>acts;
    for(int i=lim;i<=16;i++)
    {
    	//cout<<sum[i]<<" , ";
    	if(sum[i]==acts)
    	{
    		//cout<<"Match :"<<i<<"}";
    		cout<<i;
    		break;
    	}
    }
    
    //cout<<"Hello"<<endl;
}
