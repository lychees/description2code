def rate(p, m, f):
    return (p+1) % (m + f)


for _ in range(input()):
    m, f, k = map(int, raw_input().split())
    ret = ["M"] * (m + f)
    pos = len(ret) - 1
    for i in range(f):
        j = 0
        while j < k:
            pos = (pos+1) % len(ret)
            if ret[pos] == 'M':
                j += 1
        ret[pos] = 'F'
    print "".join(ret)
