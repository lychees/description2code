t = int(raw_input())
while(t>0):
	line = raw_input()
	lst = line.split()
	n = int(lst[0])
	m = int(lst[1])
	x = int(lst[2])
	new_line = raw_input()
	data = new_line.split()
	sum = 0
	for marks in data:
		sum = sum + int(marks)
	to_get = x*(n+1)
	need = to_get - sum
	if(need<1):
		need = 1
	if(need<=m):
		print(need)
	if(need>m):
		print("Impossible")
	t = t-1
