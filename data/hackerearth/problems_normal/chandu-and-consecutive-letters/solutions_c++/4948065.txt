#include <iostream>
using namespace std;
#include<queue>
int main()
{
	int t,i;
	string s,a;
	char ch,k;
    cin>>t;
    while(t--)
    {
    	a="";
    	queue<char>q;
    	cin>>s;
    	q.push(s[0]);
    	ch=s[0];
    	for(i=1;i<s.length();i++)
    	{
    		if(s[i]!=ch)
    		{
    			ch=s[i];
    			q.push(s[i]);
    		}
    	}
    	while(!q.empty())
    	{
    		k=q.front();
    		a=a+k;
    		q.pop();
    	}
    	cout<<a<<"\n";
    }
    return 0;
}
