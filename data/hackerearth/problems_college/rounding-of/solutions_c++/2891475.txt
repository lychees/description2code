#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,n,b;
	cin >> t;
	while(t--)
	{
		cin >> n >> b;
		if(n % b >= b/2.0)
		{
            cout << n/b*b+b <<"\n";
        }
        else
        cout << n/b*b <<"\n";
	}
    //cout << "Hello World!" << endl;
    return 0;
}
