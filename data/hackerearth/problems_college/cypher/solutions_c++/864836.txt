#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	char s[101];
    	cin>>s;
    	int len=strlen(s);
    	cout<<s[0];
    	for(int i=1;i<len;i++)
    	{
    		if((s[i]+i)<='z')
    		printf("%c",s[i]+i);
    		else
    		{
    			int pos='z'-s[i];
    			pos=i-pos;
    			printf("%c",pos+'a'-1);
    		}
    	}
    	cout<<endl;
    }
    return 0;
}
