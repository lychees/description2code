#include<bits/stdc++.h>
#define ll long long int
using namespace std;
int prime[10000006],cp;
bool arr[10000007];
void sieve()
{
   int i,j,k;
   double sq=sqrt(1005);
   arr[0]=1;
   arr[1]=1;
   for(i=2;i<=sq;i++)
   {
   	if(!arr[i])
   	{
   		for(j=i*i;j<1005;j+=i)
   			arr[j]=1;
   	}
   }
   cp=k;
   //cout<<cp<<endl;
}
int main()
{
	sieve();
	int t;
	scanf("%d",&t);
	for(int test=1;test<=t;test++)
	{
		ll n,ans=0;
		scanf("%lld",&n);
		for(int i=0;i<n;i++)
		{
		int item;
		scanf("%d",&item);
		if(arr[item]==0)
		ans++;
		}
		printf("%d\n",ans);
	}
	return 0;
}
