def caesar(plainText, shift): 
    cipherText = ""
    for ch in plainText:
        if ch.isalpha():
            stayInAlphabet = ord(ch) - shift 
            if stayInAlphabet < ord('A'):
                stayInAlphabet += 26
            finalLetter = chr(stayInAlphabet)
            cipherText += finalLetter    
    return cipherText

noc = input()
while noc:
    noc-=1
    shift = input()
    encoded = raw_input()
    print caesar(encoded, shift)
