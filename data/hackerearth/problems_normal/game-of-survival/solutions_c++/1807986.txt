#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

int main()
{
	int T, N;
	cin>>T;
	while(T--)
	{
		cin>>N;
		int flag = 0;
		vector<int> sword, strength;
		for(int i = 0; i < N; i++)
		{
			int temp;
			cin>>temp;
			sword.push_back(temp);
		}
		for(int i = 0; i < N; i++)
		{
			int temp;
			cin>>temp;
			strength.push_back(temp);
		}
		sort(sword.begin(), sword.end());
		sort(strength.begin(), strength.end());
		for(int i = 0; i < N; i++)
		{
			if(sword[i] <= strength[i])
			{
				flag = 1;
				break;
			}
		}
		if(flag)
			cout<<"NO"<<endl;
		else
			cout<<"YES"<<endl;
	}
	
    return 0;
}
