#include<bits/stdc++.h>
using namespace std;
typedef long long int lld;
int main()
{
     lld ac[300];
lld bc[300];
    lld t;
    scanf("%lld",&t);
    while(t--)
    {


        for(lld i=0;i<300;i++)
        {
            ac[i]=0;
        }
        for(lld i=0;i<300;i++)
        {
            bc[i]=0;
        }
        string a,b;
        lld fl=0;
        cin>>a;
        cin>>b;
        lld lena=a.length();
        lld lenb=b.length();
        for(lld i=0;i<lena;i++)
        {
        ac[a[i]]++;
        }
        for(lld i=0;i<lenb;i++)
        {
            bc[b[i]]++;
        }
      /*  cout<<"AC"<<endl;
         for(lld i=97;i<=122;i++)
        {
            cout<<i<<" "<<ac[i]<<endl;
        }
        cout<<"BC"<<endl;
         for(lld i=97;i<=122;i++)
        {
            cout<<i<<" "<<bc[i]<<endl;
        }
*/
        for(lld i=97;i<=122;i++)
        {
            if((ac[i]!=0)&&(bc[i]!=0))
            {
                cout<<"YES"<<endl;
                fl=1;
                break;
            }
        }
        if(fl==0)
        {
            cout<<"NO"<<endl;
        }
    }
    return 0;
}
