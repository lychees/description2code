#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t)
    {
    	int n,k,count=0,l;
    	cin>>n>>k;
    	for(int i=0;i<n;i++)
    	{
    	cin>>l;
    	if(l<=0)
    	count++;
    	}
    	if(count<k)
    	cout<<"YES"<<endl;
    	else
    	cout<<"NO"<<endl;
    	t--;
    	//cout<<count;
    	count=0;
    }
    return 0;
}
