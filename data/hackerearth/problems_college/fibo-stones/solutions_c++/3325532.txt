#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	long long int a[n];
	a[1] = 1,a[2] = 1;
	long long int sum = 2;
	for(int i=3;i<=n;i++)
	{
		a[i] = a[i-1] + a[i-2];
		sum = sum + a[i];
	}
	int m;
	cin>>m;
	long long int s[m];
	long long int s1 = 0;
	for(int i=0;i<m;i++)
	{
		cin>>s[i];
			s1 = s1 + s[i];
	}
	cout<<abs(s1 - sum)<<"\n";
	
}
