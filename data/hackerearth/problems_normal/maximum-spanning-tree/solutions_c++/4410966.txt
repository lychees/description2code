#include <bits/stdc++.h>
using namespace std;
#define ppi pair<int, pair<int,int> >
 
int t,n,m,c,p[5003];
ppi g[100005];
 
void initialise(){
	for(int i=1;i<=n;i++){
		p[i]=i;
	}
}
 
int root(int x){
	while(x!=p[x]){
		p[x]=p[p[x]];
		x=p[x];
	}
	return x;
}
 
void u(int x,int y){
	x=root(x);
	y=root(y);
	p[x]=p[y];
}
 
long long kruskal(){
	int x,y,cost;
	long long maxcost=0;
	for(int i=0;i<m;i++){
		x=g[i].second.first;
		y=g[i].second.second;
		cost = g[i].first;
		cost=cost*-1;
		if(root(x)!=root(y)){
			maxcost += cost;
			u(x,y);
		}
	}
	return maxcost;
}
 

int main()
{
    cin>>t;
    int x,y,w;
    while(t--){
    	cin>>n>>m;
    	initialise();
    	for(int i=0;i<m;i++){
    		cin>>x>>y>>w;
    		w=w*-1;
    		g[i] =make_pair(w, make_pair(x,y));
    	}
    	sort(g, g+m);
    	long long maax=kruskal();
    	cout<<maax<<endl;
    }
    return 0;
}
