#define mod 1000000007
typedef long long ll;
typedef unsigned long long ull;
#define S 200000
ll occur[S];
int x[S],y[S],diffx[S],diffy[S];

int main() {
//    read("in.txt");
//rite("out.txt");
    int a,b;
    int n,i;
    cin >> n;
    for(i = 0; i < n; ++i) cin >> x[i] >> y[i];
    sort(x,x+n);
    sort(y,y+n);
    a = 0; b = n-1;

    occur[0] = 0;
    for(i = 1 ; i < n; ++i ){
        diffx[i] = abs(x[i] - x[i-1]);
        diffy[i] = abs(y[i] - y[i-1]);
        occur[i] = occur[i-1] - a + b;
        a += 1; b -= 1;
    }
    ll ans = 0,temp;
//    for(i = 1; i < n ; ++i) printf("%3d",diffx[i]);
//    cout << endl;
//    for(i = 1; i < n ; ++i) printf("%3d",diffy[i]);
//    cout << endl;
    for(i = 1; i < n ; ++i){
            temp = diffx[i];
            temp *= occur[i];
            ans = (ans + temp%mod)%mod;
            temp = diffy[i];
            temp *= occur[i];
            ans = (ans + temp%mod)%mod;
    }
    cout << ans;
    return 0;
    }
