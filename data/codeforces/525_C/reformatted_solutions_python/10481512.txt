from collections import deque
rr = raw_input
N = int(rr())
A = map(int, rr().split())
A.sort(reverse=True)
A = deque(A)
B = []

while True:
    try:
        x, y = A.popleft(), A.popleft()
        if x == y:
            B.append(x)
        elif x-1 == y:
            B.append(y)
        else:
            A.appendleft(y)
    except BaseException:
        break
B = deque(B)

ans = 0
while True:
    try:
        x, y = B.popleft(), B.popleft()
        ans += x*y
    except BaseException:
        break
print ans
