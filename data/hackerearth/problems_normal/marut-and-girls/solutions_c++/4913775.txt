int main()
{
	IOS;
    int n,x,t;
    cin >> n;
    int z = n;
    unordered_map <int,bool> m;
    
    for(int i=0;i<n;++i)
    {
    	cin >> x;
    	m[x] = true;
    }
    int c = 0;
    cin >> t;
    cin.ignore();
    while(t--)
    {
    	int ct = 0;
    	string s;
    	getline(cin,s);
    	s=s+(char)32;
    	int l = s.length();
    	x = 0; 
    	for(int i=0;i<l;++i)
    	{
    		if( (int)s[i] == 32)
    		{
    			if(m[x])
    			ct++;
    			x = 0;
    		}
    		else
    		{
    			x = x*10 + (s[i] - '0');
    		}
    	}
    	if(ct == z) c++; 
    }
    cout << c ;
    
    return 0;    
}
