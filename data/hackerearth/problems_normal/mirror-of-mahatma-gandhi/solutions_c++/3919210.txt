#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	int flag = 1;
    	for(int i = 0;i<s.length();i++)
    	{
    		if(s[i] != '1' && s[i] != '0' && s[i] != '8')
    		{
    			flag = 0;
    			break;
    		}
    		else if(s[i] != s[s.length()-1-i])
    		{
    			flag = 0;
    			break;
    		}
    	}
    	if(flag)	cout<<"YES\n";
    	else cout<<"NO\n";
    }
    return 0;
}
