#include<bits/stdc++.h>
using namespace std;



int main()
{
    int t;
    cin>>t;
    t++;
    while(t--)
    {
        set<string> s;
        char arr[100001];
        char chaska[100001];
        cin.getline(arr,100001);
        int len=strlen(arr);
        int ct=0;
        for(int i=0; i<len; i++)
        {
            ct=0;
            if(arr[i]=='\0')
                break;
            while(arr[i]!=' '&&arr[i]!='\0')
            {
                chaska[ct++]=arr[i];
                i++;
            }
            chaska[ct]='\0';

            s.insert(chaska);
            if(arr[i]=='\0')
                break;
        }
         for(int i=0; i<len; i++)
        {
            ct=0;
            if(arr[i]=='\0')
                break;
            while(arr[i]!=' '&&arr[i]!='\0')
            {
                chaska[ct++]=arr[i];
                i++;
            }
            chaska[ct]='\0';

            int r=s.count(chaska);
            if(r)
            {
                cout<<chaska<<" ";
                s.erase(chaska);
            }
            if(arr[i]=='\0')
                break;
        }
        cout<<endl;
    }
}
