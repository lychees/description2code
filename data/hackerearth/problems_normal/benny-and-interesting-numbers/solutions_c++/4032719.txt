#include <bits/stdc++.h>
using namespace std;
const int maxm = 400000;
int a[maxm];
bool visit[maxm];
int len=0;
bool intrst(long long int);
int main()
{
	int T,i,j;
	long long int x;
	cin >> T;
	for(i=2;i<maxm;i++)
		visit[i]=false;
	for(i=2;i<maxm;i++)
	{
		if(!visit[i])
		{
			a[len] = i;
			for(j=i+i;j<maxm;j+=i)
				visit[j] = true;
			len++;
		}
	}
	while(T--)
	{
		cin >> x;
		if(intrst(x))
			cout << "Yes" << endl;
		else
			cout << "No" << endl;
	}
	return 0;
}

bool intrst(long long int x)
{
	int i,cnt;
	long long int p;
	p=x;
	for(i=0;i<len;i++)
	{
		cnt=0;
		if(a[i]>=x)
			break;
		if(x%a[i]==0)
		{
			while(x%a[i]==0)
			{
				cnt++;
				if(cnt>1)
					break;
				x=x/a[i];
			}
			if(cnt>1)
				return false;
		}
		x=p;
	}
	return true;
}
