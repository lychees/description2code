#include <iostream>
using namespace std;

int main()
{
    int size;
    cin>>size;
    int arr[size];
    for(int i=0;i<size;i++){
    	cin>>arr[i];
    }
    int q;
    cin>>q;
    for(int i=0;i<q;i++){
    	int sum=0;
    	int l,r;
    	cin>>l>>r;
    	for(int j=l; j<=r; j++){
    		sum = sum + arr[j-1];
    	}
    	cout<<sum<<"\n";
    }
    return 0;
}
