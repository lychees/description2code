arr = raw_input()
state = 0
for i in range(len(arr)):
	if arr[i] == 'l' and state == 0:
		state = 1
	elif arr[i] == 'o' and state == 1:
		state = 2
	elif arr[i] == 'v' and state == 2:
		state = 3
	elif arr[i] == 'e' and state == 3:
		state = 4
if state == 4:
	print "I love you, too!"
else:
	print "Let us breakup!"
