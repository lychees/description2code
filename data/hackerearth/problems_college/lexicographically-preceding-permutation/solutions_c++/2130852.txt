void solve()
{
	int n;
	cin>>n;
	int a[1000001];
	for(int i=0;i<n;i++)
		cin>>a[i];
	int i = n-1;
	while(i > 0 && a[i] >= a[i-1])
		i--;
	if(i == 0)
	{
		for(int i=0;i<n;i++)
			cout<<a[i]<<" ";
		cout<<"\n";
	}
	else
	{
		int k = i-1;
		int m = a[i];
		int mind = i;
		for(int j = i;j<n;j++)
		{
			if(a[j] < a[k] && a[j] > m)
			{
				m = a[j];
				mind = j;
			}
		}
		int t = a[k];
		a[k] = a[mind];
		a[mind] = t;
		sort(a+i, a+n, std::greater<int>());
		for(int kk = 0;kk<n;kk++)
			cout<<a[kk]<<" ";
		cout<<"\n";
	}
}

int main()
{
	std::ios::sync_with_stdio(false);
	int t;
	cin>>t;
	//t = 1;
	while(t--)
	{
		solve();
	}
	return 0;
}
