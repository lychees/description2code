#include <iostream>
#include <bits/stdc++.h>
using namespace std;
map <char,int> m1,m2;
map <char,int> :: iterator it1,it2;
int main()
{
    string s1,s2;
    int l1,l2,i,flag=1;
    cin>>s1>>s2;
    l1=s1.size();
    l2=s2.size();
    if(l1!=l2)
    {
    	flag=0;
    }
    else
    {
    	for(i=0;i<l1;i++)
    	{
    		m1[s1[i]]++;
    		m2[s2[i]]++;
    	}
    	for(it1=m1.begin(),it2=m2.begin();it1!=m1.end()&&it2!=m2.end();it1++,it2++)
    	{
    		if(it1->first!=it2->first||it1->second!=it2->second)
    		{
    			flag=0;
    			break;
    		}
    	}
    }
    if(flag)
    {
    	cout<<"CORRECT"<<endl;
    }
    else
    {
    	cout<<"WRONG"<<endl;
    }
    return 0;
}
