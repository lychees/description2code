#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	char s[100001];
    	cin>>s;
    	int l=strlen(s),f=0,p=0;
    	for(int i=0;i<l&&p==0;i++)
    	{
    		if(s[i]=='R'||s[i]=='Y')
    		p=0;
    		else
    		p=1;
    	}
    	if(s[0]=='R'&&p==0)
    	{
    		for(int i=0;i<l-2&&f==0;i++)
    		{
    			if(s[i]=='R')
    			{
    				if(s[i+1]=='R')
    				f=0;
    				else
    				if(s[i+1]=='Y')
    				f=0;
    				else
    				f=1;
    			}
    			else
    			if(s[i]=='Y')
    			{
    				if(s[i+1]=='R')
    				f=0;
    				else
    				if(s[i+1]=='Y'&&s[i+2]=='R')
    				f=0;
    				else
    				f=1;
    			}
    		}
    		if(f==0)
    		cout<<"YES"<<endl;
    		else
    		cout<<"NO"<<endl;
    	}
    	else
    	cout<<"NO"<<endl;
    }
    return 0;
}
