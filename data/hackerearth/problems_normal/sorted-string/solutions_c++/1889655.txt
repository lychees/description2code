#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	int a[26]={0};
    	for(int i=0;i<s.size();i++)
    	a[s[i]-'a']++;
    //	sort(a,a+26);
    	string s1="";
    	
    	for(int i=0;i<26;i++)
    	{
    		int max=-1;
    		int c=0;;
    		for(int j=25;j>=0;j--)
    		{
    			if(max<a[j] && a[j]!=0)
    			{
    				max=a[j];
    				c=j;
    			}
    		}
    		a[c]=0;
    		for(int j=0;j<max;j++)
    		s1=char('a'+c)+s1;
    	}
    	cout<<s1<<endl;
    }
    return 0;
}
