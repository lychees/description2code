# list1=[3,2,1]
# list2=[1,3,2]

n = input()
list1 = map(int, raw_input().strip().split(" "))
list2 = map(int, raw_input().strip().split(" "))

# print list1
count = 0

for i in range(n):
    while (list1[i] != list2[i]):
        temp = list1[i]
        for j in range(i, n-1):
            list1[j] = list1[j+1]
        list1[n-1] = temp
        count = count+1
    count = count+1
print count
