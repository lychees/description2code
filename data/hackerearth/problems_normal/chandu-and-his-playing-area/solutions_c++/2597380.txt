#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    long long k;
    double sum;
    cin>>n;
    while(n--){
    	cin>>k;
    	if(k&1)
    		k--;
    	if(k<4){
    		cout<<"0\n";
    		continue;
    	}
    	sum=k/4.0;
    	//cout<<sum<<endl;
    	k=(long long)pow(sum,2);

    	cout<<k<<endl;
    }
    return 0;
}
