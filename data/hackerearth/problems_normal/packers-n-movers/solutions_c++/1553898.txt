using namespace std;
 
int w[100];
 
bool ok(int we,int n,int m) {
    int sum = 0;
    for(int i = 0;i < n;++i) {
        if(w[i]>we) {
            return false;
        }
        else if((sum+w[i])>we) {
            --m;
            if(m <= 0) {
                return false;
            }
            sum = w[i];
        }
        else {
            sum += w[i];
        }
    }
    return true;
}
 
int main() {
    
    int m,p,weight = 0;
    cin>>m>>p;
    for(int i = 0;i < p;++i) {
        cin>>w[i];
        weight += w[i];
    }
    int lo = 1, hi = weight;
    while(lo <= hi) {
        int mid = (lo+(hi-lo)/2);
        if(ok(mid,p,m)) {
            hi = mid-1;
        }
        else {
            lo = mid+1;
        }
    }
    cout<<lo<<"\n";
    return  0;
    
}
