#include <iostream>
using namespace std;
struct burger
{
	int sum;
	int index;
};
typedef struct burger burger;

int main()
{
	int n;
	
    cin>>n;
    burger b[n],swap;
    int t[n],d[n];
    for(int i=0;i<n;i++)
    cin>>t[i]>>d[i];
    
    for(int i=0;i<n;i++)
    {
    	b[i].sum=t[i]+d[i];
    	b[i].index=i+1;
    }
    
    for(int i=0;i<n-1;i++)
    {
    	for(int j=0;j<n-i-1;j++)
    	{
    		if(b[j].sum>b[j+1].sum)
    		{
    			swap=b[j];
    			b[j]=b[j+1];
    			b[j+1]=swap;
    		}
    	}
    }
    for(int i=0;i<n;i++)
    {
    	cout<<b[i].index<<" ";
    }
    return 0;
}
