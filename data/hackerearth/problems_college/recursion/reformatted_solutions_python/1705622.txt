import itertools
t = input()
for i in range(int(t)):
    lst = []
    x = input()
    for j in range(1, int(x)+1):
        lst.append(j)
    a = list(itertools.permutations(lst))
    z = ""
    for i in range(len(a)):
        for j in range(len(a[i])):
            z = z+str(a[i][j])
        z = z+" "
    print(z)
