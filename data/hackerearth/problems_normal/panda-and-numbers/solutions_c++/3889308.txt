#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
#define ll long long
#define mod 1000000007
#define bitcount    __builtin_popcountll
#define pb push_back
int a[1000006][12],b[1000006],c[1000006];
int p(int x,int y)
{
	if(y==0)
		return 1;
	if(y==1)
		return x;
	int z=p(x,y/2);
	return z*z*p(x,y%2);
}
int main()
{
    //freopen("in.txt","r",stdin);
    //freopen("out.txt","w",stdout);
	int t,n,i,j,k,l;
	n=1000000;
	for(i=1;i<=n;i++)
	{
		j=i;
		while(j!=0)
		{
			k=j%10;
			a[i][k]=1;
			j/=10;
		}
	}
	i=1;
	while(p(i,i)<=n)
	{
		j=p(i,i);
		if(c[j]==1)
		{
			i++;
			continue;
		}
		b[j]=1;
		for(k=j;k<=n;k++)
		{
			if(b[k]==1)
			{
				for(l=1;l<=9;l++)
				{
					t=k+l*l;
					if(t<=n&&a[t][l]==1)
						b[t]=1;
				}
			}
		}
		for(k=1;k<=n;k++)
		{
			if(b[k]==1)
			{
				b[k]=0;
				c[k]=1;
			}
		}
		i++;
	}
	sd(t);
	while(t--)
	{
		sd(n);
		if(c[n]==1)
			printf("Yes\n");
		else
			printf("No\n");
	}
    return 0;
}
