#include<iostream>
using namespace std;
int main()
{
	int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		int p=1<<n;
		int a[n+1];
		for(int i=0;i<n;i++)
		{
			cin>>a[i];
		}
		int x=0,max=0,maxsize=0,size,val;
		int an=(1<<32)-1;
		for(int i=1;i<p;i++)
		{
			x=0;
			an=(1<<32)-1;
			size=0;
			for(int j=0;j<n;j++)
			{
				if(i&1<<j)
				{
					x=x^a[j];
					an=an&a[j];
					size++;
				}
			}
			val=x|an;
//cout<<val<<max<<endl;
			if(val>max)
			{
				maxsize=size;
				max=val;
				
			}
			if(val==max && size>maxsize)
			{
				maxsize=size;
			}
		}
		cout<<max<<" "<<maxsize<<endl;
	}
	return 0;
}
