#include<bits/stdc++.h>
#define M 10000007
using namespace std;
vector<long long> small, big;
long long dp[2][(1<<22)];
int main()
{


  for(int i=0;i<2;i++) for(int j=0;j<(1<<20);j++) dp[i][j]=0;
  long long n,x,y;
  cin>>n>>x;

  for(long long i=0;i<n;i++)
  {
      cin>>y;
      if(y>1024)
        big.push_back(y);
      else
        small.push_back(y);
  }
   int k=big.size();
   //cout<<(1<<k);

dp[0][0]=dp[1][0]=1;




  sort(small.begin(),small.end());
  for(int i=0;i<small.size();i++)
  {
      for(int j=0;j<(1<<11);j++)
      {
          int p=j^small[i];
          dp[(i%2)][j]=(dp[((i+1)%2)][j] + dp[((i+1)%2)][p])%M;

      }
      //cout<<dp[(i%2)][x]<<"\n";

  }
        for(int j=0;j<(1<<11);j++)
      {
           dp[(small.size())%2][j]=dp[(small.size()+1)%2][j];

      }

  sort(big.begin(),big.end());
  for(int i=0;i<big.size();i++)
  {
      for(int j=0;j<(1<<21);j++)
      {
          long long p=j^big[i];
          dp[(i%2)][j]=(dp[((i+1)%2)][j] + dp[((i+1)%2)][p])%M;

      }
      //cout<<dp[(i%2)][x]<<"\n";

  }

  int p=((small.size()+1)%2);
  long long cnt=dp[p][x];



cout<<cnt<<"\n";


}
