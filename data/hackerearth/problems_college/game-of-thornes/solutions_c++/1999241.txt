#include <iostream>
#include<stdio.h>
#include<map>
#include<string.h>
using namespace std;

int main()
{
    map<char,int>mm; char a[100001];
    map<char,int>::iterator it;
    scanf("%s",a); int i,count=0;
    for(i=0;i<strlen(a);i++)
    {	if(mm[a[i]]==0)
    	mm[a[i]]++;
    	else
    	mm[a[i]]--;
    }
    
    for(it=mm.begin();it!=mm.end();it++)
    {
    	if(it->second==1)
    	count++;
    }
    if(count>1)
    printf("NO\n");
    else
    printf("YES\n");
    
    
}
