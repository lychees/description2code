#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	int cr=0,cu=0,cb=0,cy=0;
    	int x=s.size();
    	for(int i=0;i<x;i++)
    	{
    		if(s[i]=='r')
    		cr++;
    		
    		if(s[i]=='u')
    		cu++;
    		
    		if(s[i]=='b')
    		cb++;
    		
    		if(s[i]=='y')
    		cy++;
    		
    		
    	}
    	
    	int z=min(min(min(cr,cu),cb),cy);
    	cout<<z<<endl;
    }
    return 0;
}
