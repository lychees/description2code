
#include<iostream>
#include<stdio.h>
#include<stdlib.h>
#include<string.h>
using namespace std;
int main(void)
{
	char **str,ch=97;
	int i, n,cnt=0,oc=0,j;
	cin >> n;
	str=(char **)malloc(sizeof(char *)*n);
	for(i=0;i<n;i++)
	str[i]=(char *)malloc(sizeof(char)*100);
	for(i=0;i<n;i++)
		scanf("%s",str[i]);
		
	for(j=0;j<26;j++)
	{
		cnt=0;
		for(i=0;i<n;i++)
		{
			if(strchr(str[i],(ch+j))!=NULL)
				cnt++;
		}
		if(cnt==n)	
			oc++;
	}
	cout << oc << "\n";
	
	return 0;
}
