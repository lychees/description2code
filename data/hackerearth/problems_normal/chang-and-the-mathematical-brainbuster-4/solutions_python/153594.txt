def getCount(min_index, values):
    minimum = values[min_index]
    max_allowed = 3 * minimum
    i = min_index
    length = len(values)
    while i < len(values):
        if(values[i] > max_allowed):
            return min_index + (length - i)
        i += 1
    return min_index
