#include <bits/stdc++.h>
using namespace std;
typedef long long LL;

int f(int n)
{
	int s=0;
	while(n)
	{
		s += (n%10) * (n%10);
		n /= 10;
	}
	return s;
}

int main()
{
	int t; cin >> t;
	int ar[800];
	for(int a=0;a<800;a++)
	{
		ar[a] = f(a);
		//cout << ar[a] << " ";
	}
	while(t--)
	{
		int n,d;
		cin >> n >> d;
		n = f(n);
		map<int,int> m;
		int i = 1;
		while(m.find(n) == m.end())
		{
			m[n] = i++;
			n = ar[n];
		}
		int start = n;
		int start_i = -1;
		for(auto e : m)
			if(start == e.first) { start_i = e.second; break; }

		vector< pair<int,int> > v;
		for(auto e : m)
		{
			if(e.second >= start_i)
				v.push_back(e);
		}
		sort(v.begin(), v.end(), [](pair<int,int> i, pair<int,int> j) { return i.second < j.second or (i.second == j.second and i.first < j.first); });
		
		//for(auto e : v)
			//printf("%d %d\n",e.first,e.second);

		if(d < start_i)
		{
			for(auto e : m)
				if(e.second == d)
				{
					printf("%d\n",e.first);
					break;
				}
			continue;
		}
		d -= start_i;
		d %= v.size();
		printf("%d\n",v[d].first);
	}
    
    return 0;
}
