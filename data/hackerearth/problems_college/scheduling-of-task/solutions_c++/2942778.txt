#include<bits/stdc++.h>
#define LL long long int
#define REP(i,n) for(int i=0;i<n;++i)
#define REPP(i,a,b) for(int i=a;i<b;++i)
#define TC int t;scanf("%d",&t);while (t-->0)
#define INP(x) scanf("%d",&x)
#define OUT(x) printf("%d\n",x)
#define INPLL(x) scanf("%lld",&x)
#define OUTLL(x) printf("%lld\n",x)
#define INPS(x) scanf("%s",x)
#define MEM(a,b) memset(a,b,sizeof(a))
#define MP make_pair
#define PB push_back
#define PII pair<int,int>
#define PLL pair<long long,long long>
#define VPII vector<pair<int,int >
using namespace std;
int main()
{
    LL n;
    INPLL(n);
    vector<LL> S,D;
    S.resize(n);
    D.resize(n);
    vector<LL> M;
    vector<LL> tD;
    M.reserve(n);
    D.reserve(n);
    LL m,d;
    REP(i,n)
    {
        INPLL(d);INPLL(m);
        std::vector<LL>::iterator it = upper_bound(tD.begin(),tD.end(),d);
        int p=it-tD.begin();
        tD.insert(it,d);
        M.insert(M.begin()+p,m);

        LL j=p;
        if (j==0)
        {
            S[0]=m;
            D[0]=max(0ll,m-d);
            ++j;
        }
        while(j<=i)
        {
            S[j]=S[j-1]+M[j];
            D[j]=max(D[j-1],S[j]-tD[j]);
            ++j;
        }
        OUTLL(D[i]);
    }
    return 0;
}
