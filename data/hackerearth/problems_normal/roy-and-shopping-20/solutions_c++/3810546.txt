#include<iostream>
#include<cstdio>
using namespace std;
int N = 1000001;
int prime[1000001] = {0};
int lpf[1000001];
void lpf_sieve()
{
    int k;
    for(k=0;k<N;k++)
    {
        lpf[k] = k;
    }
    prime[0] = 1;
    prime[1] = 1;
    int i = 2,j;
    while(i*i <= N)
    {
        j = 2*i;
        if(prime[i] != 1)
        {
            while (j<N)
            {
                if(lpf[j] > i)
                {
                    lpf[j] = i;
                }
                prime[j] = 1;
                j += i;
            }
        }
        i += 1;
    }
}
int main()
{
    int n,t;
    scanf(" %d",&t);
    lpf_sieve();
    while(t--)
    {
        scanf(" %d",&n);
        printf("%d\n",n-lpf[n]);
    }
    return 0;
}
