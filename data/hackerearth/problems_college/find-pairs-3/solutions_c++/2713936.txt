#include<bits/stdc++.h>
using namespace std;
int ar[1111111];
int main()
{
    cin.sync_with_stdio(false);
    int t,n,i,inp,ans;
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(i=0;i<n;i++)
            cin>>ar[i];
        sort(ar,ar+n);
        int cur=ar[0];
        long long ans=0,temp;
        temp=1;
        for(i=1;i<n;i++)
        {
            if(cur==ar[i])
                temp++;
            else
            {
                ans+=(temp*(temp+1))/2;
                temp=1;
                cur=ar[i];
            }
        }
        ans+=(temp*(temp+1))/2;
        cout<<ans<<endl;
    }
}
