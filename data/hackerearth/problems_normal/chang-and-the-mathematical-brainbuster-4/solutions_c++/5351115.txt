//constants
const int mxn=1e5+1;
const int MOD=1e9+7;
using namespace std;
int A[10001];
int n;
int main()
{
	int t=1;
//	sc1(t);
	while(t--)
	{
		cin>>n;
		for(int i=0;i<n;++i)
			cin>>A[i];
		sort(A,A+n);
		int ans=0;
		for(int i=n-1;i>=0;--i)
		{
			for(int j=0;j<i;++j)
			{
				if(A[i]-3*A[j]<=0)
				{
					ans=max(ans,i-j+1);
					break;
				}
			}
		}
		cout<<n-ans<<endl;
	}
	return 0;
}
