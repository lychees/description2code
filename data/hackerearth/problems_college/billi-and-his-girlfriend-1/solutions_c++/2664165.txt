#include<iostream>
using namespace std;
#include<math.h>
long long fst(long long  a,long long b,long long mod)
{
    long long ans=1,k=a;
    while(b)
    {
        if(b%2)
            ans*=k;
        k*=k;
        k%=mod;
        ans%=mod;
        b/=2;
    }
    return ans;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int a,b,k;
        cin>>a>>b>>k;
        k=pow(10,k);
        long long p=fst(a,b,k);
        int ans=0;
        while(p)
        {
            ans+=p%10;
            p/=10;
        }
        cout<<ans<<endl;
    }
}
