#include <bits/stdc++.h>
#define mod 1000000007
#define MAX 100005
#define ll long long
#define sc(a) scanf("%d",&a)
#define pf(a) printf("%d", a)
using namespace std;
bitset <2005> a[2005];
int main()
{
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
		{
			int x;
			sc(x);
			a[i].set(j,x);
		}
	}
	ll ans=0;
	for(int i=0;i<n;i++)
	{
		for(int j=i+1;j<n;j++)
		{
			ll cnt=(ll)(a[i]&a[j]).count();
			ans+=(cnt*cnt-cnt)/2;
		}
	}
	cout<<ans/2;
}
