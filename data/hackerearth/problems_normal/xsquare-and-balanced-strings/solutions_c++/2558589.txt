#include <iostream>
#include<stdio.h>
using namespace std;
long int alpha[26];
void initialize(){
	for(int i=0;i<26;i++){
		alpha[i]=0;
	}
}
bool isBalanced(){
	for(int i=0;i<26;i++){
		if(alpha[i]%2!=0){
			return false;
		}
	}
	return true;
}
int main()
{
   	long int t;
   	string s;
   	scanf("%ld",&t);
   	while(t--){
   		initialize();
   		cin>>s;
   		for(long int i=0;s[i]!='\0';i++){
   			alpha[(int)(s[i])-(int)('a')]++;
   		}
   		if(isBalanced()){
   			printf("1\n");
   		}
   		else{
   			printf("-1\n");
   		}
   	}
    return 0;
}
