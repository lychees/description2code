def longest_substr_no_repeat2(s):
    last_seen = {s[0]: 0}
    L = {0: 1}
    for i in range(1, len(s)):
        if last_seen.get(s[i], -1) <= (i-1)-L[i-1]:
            L[i] = L[i-1] + 1
        else:
            L[i] = i - last_seen[s[i]]
        last_seen[s[i]] = i
    # print 'last_seen:', last_seen
    # print 'L{}:', L
    return max(L.values())


for tc in xrange(input()):
    s = raw_input()
    k = longest_substr_no_repeat2(s)
    print k
