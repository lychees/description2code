#include <iostream>
#include <bits/stdc++.h>
#define ll long long int
using namespace std;

ll mod = 1000000007;

ll power(ll a,ll y){
	if(y==0)
		return 1;
	ll p = power(a,y/2)%mod;
	p = ((p%mod)*(p%mod))%mod;
	return (y%2==0)?p:(p*a)%mod ;
}

ll modinv(ll a,ll mod){
	return power(a,mod-2);
}

int main()
{
    int t;
    cin >>t;
    ll fact[1001],f=1,facta[1001];
    fact[0] =1;facta[0]=1;
    for(int i=1;i<=1000;i++){
    	f=((f%mod)*(i%mod))%mod;
    	facta[i] = f;
    	fact[i] = modinv(f,mod);
    }
    while(t--){
    	int n,p;
    	cin >>n>>p;
    	ll ans=1;
    	ll temp;
    	for(int i=1;i<=p;i++){
    		temp = ( ((facta[n]*fact[i])%mod)*fact[n-i] )%mod;
    		ans = (ans + temp)%mod;
    	}
    	cout <<ans<<endl;
    }
    return 0;
}
