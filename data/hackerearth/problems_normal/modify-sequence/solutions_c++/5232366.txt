#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int a[10000];
    int i,ans=1;
    for(i=0;i<n;i++)
    cin>>a[i];
    
    for(i=0;i<n-1;i++)
    {
    	a[i+1]=a[i+1]-a[i];
    	a[i]=0;
    	
    }
    
    for(i=0;i<n;i++)
    {
    	if(a[i]!=0)
    	{
    		ans=0;
    		break;
    	}
    }
    if(ans)
    cout<<"YES";
    else
    cout<<"NO";
    
}
