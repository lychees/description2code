template < typename T > T abs(T x)
{
    return x > 0 ? x : -x;
}

template < typename T > T sqr(T x)
{
    return x * x;
}



int main()
{   
	int t;
	
    cin >> t;
    
    ll mod = 1000000007;
    
    while (t--) {
    	
    	ll a[5];
    	
    	cin >> a[0] >> a[1] >> a[2];
    	
    	sort(a, a + 3);
    	
    	for (int i = 0; i < 3; i++)
    		a[i] %= mod;
    	
    	ll ans = (a[0] * (((a[1] - 1) * (a[2] - 2)) % mod)) % mod;
    	
		cout << ans << "\n";
   	}
	
    return 0;   
}
