#include<iostream>
using namespace std;

int main(){
    int T;
    int R,K;
    int n;
    int coolness=0;
    int verycool=0;
    cin>>T;
    while(T--){
        verycool=0;
        cin>>R>>K;
        for(int i=1;i<=R;i++){
            n=i;
            coolness = 0;
            while(n>0){
                if((n&7) == 5){
                    coolness++;
                }
                n = n>>1;
            }
            if(coolness>=K){
                verycool++;
            }
        }
        cout<<verycool<<"\n";
    }
    return 0;

}
