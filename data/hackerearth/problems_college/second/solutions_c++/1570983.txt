#include <iostream>
#include<cstdio>
#include<algorithm>
using namespace std;

int main()
{
    unsigned long long n,m,a[100000],p,i;
    scanf("%llu %llu",&n,&m);
    for(i=0;i<n;i++)
    scanf("%llu",&a[i]);
    
    sort(a,a+n);
    p=0;
    for(i=0;i<n;i++)
    {
    	if(a[i]==p+m)
    		p+=m;
    }
    printf("%llu\n",p);
    return 0;
}
