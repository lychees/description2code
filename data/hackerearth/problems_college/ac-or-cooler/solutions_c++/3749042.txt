#include<bits/stdc++.h>
using namespace std;
bool mycomparison (int first, int second)
{ return ( int(first)<int(second) ); }
int main()
{
	int n,m;
	cin>>n>>m;
	int i,j;
	list<int>l1;
	
	list<int>l2;
	list<int>::iterator it;
	for(i=0;i<n;i++)
	{
		int x;
		cin>>x;
		l1.push_back(x);
	}
	for(i=0;i<n;i++)
	{
		int x;
		cin>>x;
		l2.push_back(x);
	}
l1.sort();
l2.sort();
l1.merge(l2,mycomparison);
	int sum =0,c=0;
	for(it=l1.begin();it!=l1.end();it++)
	{
		if(sum+*it<=m)
		{
			sum = sum + *it;
			c++;
		}
		if(c==n)
		{
			break;
		}
	}
	cout<<c<<"\n";
	
}
