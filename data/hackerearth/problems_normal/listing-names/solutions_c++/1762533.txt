#include <iostream>
#include<cstring>
using namespace std;

class Name{
	public:
	char S[35];
	int a=-1;
	};

int same(char *a,char *b){
	int f = 0 ;

    int la = strlen(a);
    int lb = strlen(b);

    if(la>lb){
	for(int i=0,j=0;a[i]!='\0';i++,j++)
            if(a[i]!=b[j]){
                f=-1;
                break;}
    }else if(la<lb) {
	for(int i=0,j=0;b[j]!='\0';i++,j++)
            if(a[i]!=b[j]){
                f=-1;
                break;}
        }else{
	for(int i=0,j=0;a[i]!='\0',b[i]!='\0';i++,j++)
            if(a[i]!=b[j]){
                f=-1;
                break;}
        }

    return f;
}

int main()
{
    int T;
    cin>>T;
    Name obj[T];
    for(int i=0;i<T;i++){
    	cin>>obj[i].S;
    	obj[i].a = 1;
    	}

    for(int i=0;i<T-1;i++){
    	for(int j=i+1;j<T;j++){

    		if(same(obj[i].S,obj[j].S)==0 && obj[i].S!=""){

    			strcpy(obj[j].S,"");
    			obj[i].a = obj[i].a + 1 ;
    	}
    }
  }

  for(int i=65;i<=90;i++){
  		for(int j=0;j<T;j++){

  			if((obj[j].S[0] == i || obj[j].S[0] == i+32) && obj[j].S!="" ){
  				cout<<obj[j].S<<" "<<obj[j].a<<"\n";
  			}


 	 }

 }






    return 0;
}
