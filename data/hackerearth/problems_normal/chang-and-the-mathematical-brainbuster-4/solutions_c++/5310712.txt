#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<string.h>
using namespace std;
#define inf 1 << 30
 
int main()
{
    //freopen("input.txt","r",stdin);
      //  freopen("output.txt","w",stdout);
        
        long long int n,i;
        cin >> n;
        long long int a[n+1],dp[30000];
        memset(a,0,sizeof(a));
        memset(dp,0,sizeof(dp));
 
        for(i=1;i<=n;i++){
                cin >> a[i];
                dp[a[i]]++;
        }
        sort(a+1,a+n+1);
        dp[0] = 0;
        dp[1] = dp[1];
        for(i=2;i<=30000;i++){
                dp[i] = (dp[i] + dp[i-1]);
 
        }
        
        long long int ans = inf;
        for(i=1;i<=a[n];i++){
                ans = min(ans,(dp[a[n]]-(dp[3*i]-dp[i-1])));
                
        }
 
        cout << ans << endl;
        return 0;
}
