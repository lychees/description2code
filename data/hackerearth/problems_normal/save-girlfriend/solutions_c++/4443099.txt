#include <iostream>
#include <bits/stdc++.h>
using namespace std;
vector <int> v;
int a[1000005];
map <int,int> mp;
void sieve()
{
	int i,j;
	for(i=2;i<=10000;i++)
	{
		if(!a[i])
		{
			v.push_back(i);
			for(j=i*i;j<=10000;j+=i)
			{
				a[j]=1;
			}
		}
	}
}
int main()
{
	sieve();
    int t,a,b,i,p,c,ans,j,mx;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b;
    	mx=0;
    	for(i=a;i<=b;i++)
    	{
    		p=i;
    		ans=1;
    		for(j=0;v[j]*v[j]<=p;j++)
    		{
    			c=0;
    			while(p%v[j]==0)
    			{
    				p/=v[j];
    				c++;
    			}
    			if(c)
    			{
    				ans*=(c+1);
    			}
    		}
    		if(p>1)
    		{
    			ans*=2;
    		}
    		mx=max(mx,ans);
    		mp[ans]++;
    	}
    	cout<<mp[mx]+1<<endl;
    	mp.clear();
    }
    return 0;
}
