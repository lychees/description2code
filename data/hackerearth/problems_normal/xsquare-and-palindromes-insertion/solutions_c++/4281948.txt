#include<iostream>
#include<cstring>
#include<string.h>
using namespace std;
int main()
{
	int arr[256];
	char s[1000];
	char sa[1000];
	int t;
	cin>>t;
	for(int i=0;i<t;i++)
	{
		int k=0;
		
		cin>>s;
		int flag=0;
		memset(arr,0,256*sizeof(arr[0]));
		for(int j=0;j<strlen(s);j++)
		{
			if(!arr[s[j]])
			{
				sa[k]=s[j];
				k++;
				
			}
			arr[s[j]]++;
		}
		flag=k-1;
		int count=0;
		k=0;
		//cout<<flag<<endl;
		for(int j=0;j<=flag;j++)
		{
			
			if((arr[sa[j]]%2)==1)
			{
				//cout<<sa[j]<<endl;
				count+=1;
				k=1;
				
			}
		}
		    if(k)
		    {
			cout<<count-1<<endl;
		    }
		    else
		    {
		    	cout<<count<<endl;
		    }
		
	}
	return 0;
	
}
