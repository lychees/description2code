#include <bits/stdc++.h>
using namespace std;

char S[20004][22], X[22];

int main()
{
    int T;
    scanf("%d", &T);
    
    while(T--)
	{
		int N, K;
		scanf("%d %d", &N, &K);
		
		for(int i=0; i<N; i++)
			scanf(" %s", S[i]);
		
		for(int i=0; i<K; i++)
		{
			int ans = 10000000, x;
			
			for(int k=0; k<26; k++)
			{
				int tmp = 0;
				
				for(int j=0; j<N; j++)
				{
					int why_compile_error = (int)S[j][i]-'a'-k;
					
					tmp+= abs(why_compile_error);
				}
				
				if(tmp<ans)
				{
					ans = tmp;
					x = k;
				}
			}
			
			X[i] = (char)x+'a';
		}
		
		X[K] = '\0';
		
		printf("%s\n", X);
	}

    return 0;
}
