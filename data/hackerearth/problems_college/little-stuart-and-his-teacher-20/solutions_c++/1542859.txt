#include <iostream>
using namespace std;

const int MAX=1000000;
static int a[1000001],p[1001];
int prime(int n)
{
	int i=2,k=0;
	for (i=2;i<=n;i++) a[i]=0;
	i=2;
	while (i<=n)
	{
		if (a[i]==0)
		{
			p[k++]=i;
			if (i>10000)
			{
				i+=1;
				continue;
			}			
			int j=i*i;
			while (j<=n)
			{
				a[j]=1;
				j+=i;
			}
		}
		i+=1;
	}
	return k;
}
static long long int ans[MAX+1];
long long int eut(int a,int pcount)
{
	int j=0;
	long long int ans=a;
	while (j<pcount)
	{
		if (a%p[j]==0)
		{
			ans/=p[j];
			ans*=(p[j]-1);
		}
		while (a%p[j]==0)
			a/=p[j];
		j++;
	}
	if (a!=1)
	{
		ans/=a;
		ans*=(a-1);
	}
	return ans;
}
void init(int pcount)
{
	ans[0]=0;
	ans[1]=0;
	for (int i=2;i<MAX+1;i++)
	{
		ans[i]=ans[i-1]+(i-1)-eut(i,pcount);
	}
}
int main()
{
	int pcount,t,n;
	pcount=prime(1000);
	init(pcount);
	cin>>t;
	while (t--)
	{
		cin>>n;
		cout<<ans[n]<<endl;
	}
	return 0;
}
