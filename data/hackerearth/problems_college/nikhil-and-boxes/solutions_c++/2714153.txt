#include <iostream>
using namespace std;
long min(long a, long b, long c) {
	if(a<b) {
		if(a<c) return a;
		else return c;
	}
	else {
		if(b<c) return b;
		else return c;
	}
}
int main()
{
	int t;
	long l, b, h, r;
	cin>>t;
	while(t--) {
		cin>>l>>b>>h;
		r = min(l, b, h);
		if(r%2 == 0) cout<<r/2<<".0\n";
		else cout<<r/2<<".5\n";
	}
    return 0;
}
