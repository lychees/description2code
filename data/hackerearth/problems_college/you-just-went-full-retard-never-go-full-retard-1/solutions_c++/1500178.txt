#include <iostream>
#include<math.h>
using namespace std;

int isprime(long long int var)
{
	if(var==2)
	return 1;
	if(!(var%2))
	return 0;
	for(long long int i=3;i<=sqrt(var);i+=2)
	{
		if(!(var%i))
		return 0;
	}
	return 1;
}	
int main()
{
    int t;long long int var;
    cin>>t;
    int i=1;
    while(i++<=t)
    {	
    	cin>>var;
    	if(isprime(var))
    	cout<<"YES\n";
    	else
    	cout<<"NO\n";
    }
    return 0;
}
