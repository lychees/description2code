#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
   int n, c = 0;
   cin >> n;
   
   int a[n];
   
   for(int i=0; i<n; ++i)
   cin >> a[i];
   
   for(int i=0; i<n-1; ++i)
   {
   	for(int j=i+1; j<n; ++j)
   	{
   		if((__gcd(a[i], a[j])==1) && a[i]>a[j])
   		++c;
   	}
   }
   
   cout << c << endl;
    return 0;
}
