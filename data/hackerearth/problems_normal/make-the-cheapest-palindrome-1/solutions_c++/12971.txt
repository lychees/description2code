int64 aCost, bCost;
string st;

int64 solve() {
	int be = 0, en = st.size() -1 ;
	int64 ret = 0;

	while(be <= en) {
//		Pf("st[%d] = %c, st[%d] = %c\n", be, st[be], en, st[en]);
		if(st[be] == st[en]) {
			DEBUG("#1");
			if(st[be] == '/')
				ret += 2ll*min(aCost, bCost);
		}
		else if(  (st[be] == 'a' && st[en] == '/' )|| (st[en] == 'a' && st[be] == '/')) {
			DEBUG("#2");
			ret += aCost;
		}
		else if(   (st[be] == 'b' && st[en] == '/' )|| (st[en] == 'b' && st[be] == '/') ) {
			DEBUG("#3");
			ret += bCost;
		}
		else {
			DEBUG("#4");
			return -1;
		}
		be++;
		en--;
	}
	return ret;
}

		cout << solve() << endl;
