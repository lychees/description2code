#include<bits/stdc++.h>
using namespace std;
int find(int N)
{
    if(N==0)return 1;
      int p=pow(2,double(N));
    int** A=new int*[p+1];

    for(int i=0;i<p+1;i++)
      A[i]=new int[p+1];
      memset(A,0,sizeof(A));

            for(int i=1;i<p;i++)
            {
                for(int j=1;j<=i;j++)
                {
                    cin>>A[i+1][j];
                }
            }

          int va[513];
          memset(va,0,sizeof(va));
           int l=0;
       for(int k=1;k<=pow(2,double(N-1));k++)
       {
           if(A[2*k][2*k-1]==1)
           {
               va[l++]=2*k;
           }
           else{
                A[2*k][2*k-1]=0;
                va[l++]=2*k-1;
           }
       }
         int m=l;
         while(m>1)
         {
             l=0;
             for(int i=0;i<m-1;i+=2)
             {
                 if(A[va[i+1]][va[i]]==1)
                 {
                     va[l++]=va[i+1];
                 }
                 else{
                        va[l++]=va[i];
                      A[va[i+1]][va[i]]=0;
                 }
             }
             m=l;

         }
         return va[0];
}
int main()
{
    int n;
    scanf("%d",&n);
    printf("%d\n",find(n));    
    return 0;

}
