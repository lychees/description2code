#include <iostream>
using namespace std;

int main()
{
    string s;
    char uname[100],pwd[100],prof[100],role[100],key[100];
    cin>>s;
    int i,j=0;
    for(i=0;s[i]!='?';i++)
    {
    }
    for(i=i+10;;i++)
    {
    	if(s[i]=='&' && s[i+4]=='=')
    		break;
    	uname[j++]=s[i];
    	
    }
    uname[j]='\0';
    j=0;
    for(i=i+5;;i++)
    {
    	if(s[i]=='&' && s[i+8]=='=')
    		break;
    	pwd[j++]=s[i];
    	
    }
    pwd[j]='\0';
    j=0;
    for(i=i+9;;i++)
    {
    	if(s[i]=='&' && s[i+5]=='=')
    		break;
    	prof[j++]=s[i];
    	
    }
    prof[j]='\0';
    j=0;
    for(i=i+6;;i++)
    {
    	if(s[i]=='&' && s[i+4]=='=')
    		break;
    	role[j++]=s[i];
    	
    }
    role[j]='\0';
    j=0;
    for(i=i+5;i<s.size();i++)
    {
    	key[j++]=s[i];
    }
    key[j]='\0';
    cout<<"username: "<<uname<<endl;
    cout<<"pwd: "<<pwd<<endl;
    cout<<"profile: "<<prof<<endl;
    cout<<"role: "<<role<<endl;
    cout<<"key: "<<key<<endl;
    return 0;
}
