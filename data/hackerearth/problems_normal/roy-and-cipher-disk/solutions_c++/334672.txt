#include <iostream>
#include <string.h>
using namespace std;
int main()
{int t;
cin>>t;
string s1;
for(int i=0;i<t;i++)
{
	cin>>s1;
for(int j=0;j<(s1.length());j++)
s1.at(j)-=97;
if((int)s1.at(0)>13)
	cout<<(int)s1.at(0)-26<<" ";
	else
	cout<<(int)s1.at(0)<<" ";
for(int j=0;j<(s1.length()-1);j++)
	{
		if((s1.at(j+1)>=s1.at(j))&&((s1.at(j+1)-s1.at(j))<=13))
		cout<<(s1.at(j+1)-s1.at(j))<<" ";
		
	else if((s1.at(j+1)>=s1.at(j))&&((s1.at(j+1)-s1.at(j))>13))
	cout<<-1*((s1.at(j))+(26-s1.at(j+1)))<<" ";
	
	else if((s1.at(j+1)<s1.at(j))&&((s1.at(j)-s1.at(j+1))<13))
	cout<<-1*(s1.at(j)-s1.at(j+1))<<" ";
    
    else if((s1.at(j+1)<s1.at(j))&&((s1.at(j)-s1.at(j+1))>=13))
    cout<<(26-s1.at(j))+(s1.at(j+1))<<" ";
		
	}
	cout<<endl;
}
    return 0;
}
