LL pwr(LL a, LL b, LL p) {
    if(b==0) {
        return a%p;
    }
    LL ans = 1;
    while(b) {
        if(b&1) {
            ans=(ans*a)%p;
        }
        b/=2;
        a=(a*a)%p;
    }
    return ans%p;
}
LL sum(LL num) {
    LL ans = 0;
    while(num) {
        ans+=(num%10);
        num/=10;
    }
    return ans;
}
