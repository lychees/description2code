#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll dp[20][2][2];
char y[20];
ll powers[20];
int p[10];
int n;
void cal_pow()
{
  	powers[0]=1;
  	for(int i=1;i<=18;i++)
  	{
  		powers[i]= powers[i-1]*10;
  	}
}
ll fun(int index, int take,int flag)
{
	if(index==n)
	{
		if(flag)
		 return 1;
		else
		 return 0;
	}
	if(dp[index][take][flag]!=-1)
	 return dp[index][take][flag];
	ll ans =0;
	if(take==0)
	{
		if(flag)
		{
			dp[index][take][flag]= powers[n-index];
			return dp[index][take][flag];
		}
		for(int i=0;i<=9;i++)
		 ans+= fun(index+1,0,flag | p[i]);
		 
	}
	else
	{
	  int temp= y[index]-'0';
		for(int i=0;i<temp;i++)
		{
			ans += fun(index+1,0,flag|p[i]);
		}
		ans += fun(index+1,1,flag | p[temp]);
	}
	return dp[index][take][flag]=ans;
}
int main()
{
	int t;
	scanf("%d",&t);
	cal_pow();
	p[2]=1;p[3]=1;p[7]=1;p[5]=1;
	while(t--)
	{
		scanf("%s",y);
		int len= strlen(y);
		n= len;
		for(int i=0;i<=19;i++)
		 for(int j=0;j<2;j++)
		  for(int k=0;k<2;k++)
		   dp[i][j][k]=-1;
		ll ans = fun(0,1,0);
		printf("%lld\n",ans);
	}
}
