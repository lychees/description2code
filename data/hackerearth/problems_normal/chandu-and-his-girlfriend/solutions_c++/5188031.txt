#include <iostream>
#include<cstdio>
#include<algorithm>
using namespace std;

int main()
{
    int t,n,i,j;
    
    cin>>t;
    for(i=0;i<t;i++)
    {
    	cin>>n;
     int a[n];
    	for(j=0;j<n;j++)
    	{
    		cin>>a[j];
    		
    	}
    	sort(a,a+n);
    		for(j=n-1;j>=0;j--)
    	{
    		cout<<a[j]<<" ";
    		
    	}
    	
    cout<<endl;	
    }
    return 0;
}
