#include<iostream>
#include<algorithm>
using namespace std;
int a[2001][2001];
void solve()
{
    int n,i,j,m,o=0;
    cin>>n>>m;
    for(i=5;i<n+5;i++)
        for(j=5;j<m+5;j++)
            if(a[i][j-2]==0 && a[i-2][j]==0)
            {
                a[i][j]=1;
                o++;
            }
    cout<<o<<'\n';
}
int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int t;
    cin>>t;
    while(t--)
        solve();
}
