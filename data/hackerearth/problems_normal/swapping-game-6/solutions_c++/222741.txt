#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<vector>
using namespace std;
int n,k;
char s1[1005],str[1005];
pair<int,int> arr[1005];
inline int fn()
{
	int n1=(n+1)/2;
	int count=0;
	while(++count)
	{	
		for(int i=1;i<=n1;i++)arr[i].first=2*(i-1)+1;
		for(int j=2,i=n;i>n1;i--,j+=2)arr[i].first=j;
		sort(arr+1,arr+1+n);
		for(int i=1;i<=n;i++)s1[i-1]=str[arr[i].second-1];
		if(!strcmp(s1,str))return count;
	}
}
inline int fn3()
{
	int n1=n/2;
	int count=0;
	while(++count)
	{	
		for(int i=1;i<=n1;i++)arr[i].first=2*(i-1)+1;
		arr[n1+1].first=2*(n1-1)+2;
		
		for(int j=2,i=n;i>n1+1;i--,j+=2)arr[i].first=j;
		sort(arr+1,arr+1+n);
		for(int i=1;i<=n;i++)s1[i-1]=str[arr[i].second-1];
		if(!strcmp(s1,str))return count;
	}
}

inline void fn2(int count)
{
	int n1=(n+1)/2;
	while(count--)
	{	
		for(int i=1;i<=n1;i++)arr[i].first=2*(i-1)+1;
		for(int j=2,i=n;i>n1;i--,j+=2)arr[i].first=j;
		sort(arr+1,arr+1+n);
	}
	for(int i=1;i<=n;i++)cout<<str[arr[i].second-1];
	cout<<"\n";
}

inline void fn4(int count)
{
	int n1=n/2;
	while(count--)
	{	
		for(int i=1;i<=n1;i++)arr[i].first=2*(i-1)+1;
		arr[n1+1].first=2*(n1-1)+2;
		
		for(int j=2,i=n;i>n1+1;i--,j+=2)arr[i].first=j;
		sort(arr+1,arr+1+n);
	}
	for(int i=1;i<=n;i++)cout<<str[arr[i].second-1];
	cout<<"\n";
}

int main()
{
	scanf("%d",&k);
	scanf("\n%[^\n]",str);
	n=strlen(str);
	for(int i=1;i<=n;i++)arr[i].first=arr[i].second=i;
	if(n&1)
	{
		int count=fn();
		if(k==count)cout<<str<<"\n";
		else if(k>count)fn2(count-k%count);
		else fn2(count-k);
	}
	else
	{
		int count=fn3();
		if(k==count)cout<<str<<"\n";
		else if(k>count)fn4(count-k%count);
		else fn4(count-k);
	}
	return 0;
}
