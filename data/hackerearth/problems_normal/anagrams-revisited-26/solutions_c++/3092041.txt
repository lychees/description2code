#include <iostream>
#include <map>
#include<algorithm>

using namespace std;

int main()
{
    int n;
    cin>>n;
    string s;
    map<string, int> mymap;
    int max=-1;
    for(int i=0;i<n;i++)
    {
    	cin>>s;
    	std::sort(s.begin(), s.end());
		if(mymap.find(s)!=mymap.end())
		{
			mymap[s]++;
		
		}
		else
		{
			mymap[s]=1;
		}
		if(mymap[s]>max)
			max=mymap[s];
    }
    cout<<max;
    return 0;
}
