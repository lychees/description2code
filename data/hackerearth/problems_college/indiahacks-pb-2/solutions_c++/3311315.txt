#include<bits/stdc++.h>
#include<vector>
using namespace std;
int main()
{
	int n,k=0;
	cin>>n;
	string s;
	vector<string>a(n);
	for(int i=0;i<n;i++)
	{
		
		cin>>a[i];
	}
	for(int i=0;i<n;i++)
	{
		for(int j=i+1;j<n;j++)
		{
			reverse(a[j].begin(),a[j].end());
		       if(a[i]==a[j])
		        {
		        	s=a[i];
		        	k++;
		        	break;
		        }
    	}
    	if(k==1)
    	break;
	}
	int x=s.size();
	cout<<x<<" "<<s[x/2];
	return 0;
}
