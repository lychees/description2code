#include <iostream>
#include<string.h>
using namespace std;
char s[10000001];
long long int h[26],q,k;
int main() {
	// your code goes here
	cin>>s;
	
	for(int i=0;i<26;i++)
	h[i]=0;
	
	int i=0;
	while(s[i]!='\0')
	{
	char character=s[i];
	i++;
		long long count=0;
		while(s[i]-'0'>=0 && s[i]-'0'<=9 && s[i]!='\0')
		{
			count=count*10+s[i]-'0';
			i++;
		}
		//cout<<character<<" "<<count;
		
		h[character-'a']+=count;
	}
	
	cin>>q;
	//for(int i=0;i<26;i++)
	//cout<<h[i];
	while(q--)
	{
	
		cin>>k;
		
		for(int i=0;i<26;i++)
		{
		k-=h[i];
			if(k<=0)
			{
			printf("%c\n",i+'a');
			break;
			}
		}
	
		if(k>0)
		cout<<"-1"<<endl;
	}
	
	return 0;
}
