#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		char str[1000000];
		cin>>str;
		int l=strlen(str);
		if(l<3)
		{
			cout<<"Not possible!"<<endl;
			continue;
		}
		sort(str,str+l);
		next_permutation(str,str+l);
		next_permutation(str,str+l);
		cout<<str<<" ";
		sort(str,str+l);
		reverse(str,str+l);
		prev_permutation(str,str+l);
		prev_permutation(str,str+l);
		cout<<str<<endl;
	}
}
