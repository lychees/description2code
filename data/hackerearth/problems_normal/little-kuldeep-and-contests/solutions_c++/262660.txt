#include <iostream>
#include <cstdio>
#include <cmath>

using namespace std;

int main() {
	int t;cin>>t;
	int arr[1440];
	for(int j=0; j<1440; j++)
		arr[j]=0;
	for(int i=0; i<t; i++){
		int h1, m1, h2, m2;
		scanf("%2d:%2d-%2d:%2d", &h1, &m1, &h2, &m2);
		int min = h1*60 + m1;
		int max = h2*60 + m2;
		for(int j=min; j<max; j++){
			if(arr[j]==1){
				cout<<"Will need a moderator!";
				return 0;
			}	
			else
				arr[j]=1;
		}
	}	
	cout<<"Who needs a moderator?";
	return 0;
}
