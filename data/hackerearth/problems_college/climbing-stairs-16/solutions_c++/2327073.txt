#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long int a,b,t,n,c,i;
    cin>>t;
    while(t--)
    {
     a=1;
     b=1;
     cin>>n;
     if(n==1)
     cout<<"1"<<endl;
     else if(n==2)
     cout<<"2"<<endl;
     else
     {
         for(i=1;i<=n-1;i++)
        {
          c=a+b;
          a=b;
          b=c;
        }
       cout<<c<<endl;
     }
    
    }
    return 0;
}
