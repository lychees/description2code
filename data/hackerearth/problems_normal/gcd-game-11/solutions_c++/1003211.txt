#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<string>
using namespace std;
int val[1005][1005];
int fin[1005][1005][2];
bool done[1005][1005][2];
int gcd(int a,int b) {
    return b==0?a:gcd(b,a%b);
}
int calc(int n1,int n2,int chance) {
    if((chance==0&&n1==1)||(chance==1&&n2==1)) return 0;
    if(done[n1][n2][chance]) return fin[n1][n2][chance];
    done[n1][n2][chance]=true;
    int ans=0;
    if(chance==0) {
        ans|=(!calc(n1,n2-1,!chance));
        if(val[n1][n2]!=1) ans|=(!calc(n1,n2/val[n1][n2],!chance));
    }
    else {
        ans|=(!calc(n1-1,n2,!chance));
        if(val[n1][n2]!=1) ans|=(!calc(n1/val[n1][n2],n2,!chance));
    }
    fin[n1][n2][chance]=ans;
    return ans;
}
int main() {
    for(int i=1;i<=1000;i++) {
        for(int j=1;j<=i;j++) {
            val[i][j]=val[j][i]=gcd(i,j);
        }
    }
    int t;
    scanf("%d",&t);
    while(t--) {
        int n1,n2;
        scanf("%d%d",&n1,&n2);
        if(n1==1&&n2==1) printf("Draw\012");
        else if(n2==1) printf("Arjit\012");
        else if(n1==1) printf("Chandu Don\012");
        else if(calc(n1,n2,0)) printf("Arjit\012");
        else printf("Chandu Don\012");
    }
    return 0;
}
