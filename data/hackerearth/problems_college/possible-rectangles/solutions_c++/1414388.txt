#include <iostream>
using namespace std;
#define ll long long
int main() {
	ll t;
	cin >> t;
	while(t--) {
		ll n;
		cin >> n;
		ll m;
		cin >> m;
		n--;
		m--;
		ll a;
		cin >> a;
		ll sum = 0;
		for(ll i = 1;i*i <= a;i++) {
			if(a % i == 0) {
				ll c = i;
				ll d = a/i;
			//	cout << c << " " << d << "\n";
				if(c <= n && d <= m) {
					sum +=(n - c + 1)*(m - d + 1);
				}
				if(i*i != a) {
					if(c <= m && d <= n) {
						sum +=(m - c + 1)*(n - d + 1);
					}	
				}
			}
		} 
		cout << sum << "\n";
	}

}
