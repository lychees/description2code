//constants

using namespace std;
int main()
{
	int t=1;
	sc1(t);
	while(t--)
	{
		int n;
		cin>>n;
		char A[n];
		int count1=0,count0=0;
		for(int i=0;i<n;++i)
		{
			cin>>A[i];
			if(A[i]=='1')
				count1++;
			else
				count0++;
		}
		int ans1=0,ans2=0;
		for(int i=0;i<count1;++i)
		{
			if(A[i]!='1')
				ans1++;
		}
		for(int i=0;i<count0;i++)
		{
			if(A[i]!='0')
				ans2++;
		}
		cout<<min(ans1,ans2)<<endl;

	}
	return 0;
}
