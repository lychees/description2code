#include<bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	int n,t,i;
	cin>>t;
	while(t--)
	{
		int flag=0;
		cin>>n;
		int arr[n],lsum[n],rsum[n];
		lsum[0]=0;
		for(i=0;i<n;i++)
		{
			cin>>arr[i];
			if(i)
			lsum[i]=lsum[i-1]+arr[i-1];
		}
		rsum[n-1]=0;
		for(i=n-2;i>=0;i--)
		{
			rsum[i]=rsum[i+1]+arr[i+1];
		}
		for(i=1;i<n-1;i++)
		{
			if(lsum[i]==rsum[i])
			{
				flag=1;
				break;
			}
		}
		if(flag)
		printf("YES\n");
		else
		printf("NO\n");
	}


return 0;
}
