#include <bits/stdc++.h>

using namespace std;

int subsequence(char arr1[], char arr2[])
{
	int j = 0; 
	int m = strlen(arr1);
	int n = strlen(arr2);
	
	for (int i = 0; i < n && j < m; i++)
		if (arr1[j] == arr2[i])
			j++;
	return j == m;
}

int main()
{
	int t, result;
	char s[10001], p[10001];
	cin >> t;
	int a;
	while (t--)
	{
		cin >> s;
		cin >> p;
		result = subsequence(p, s);
		if (result == 1)
			cout << "YES" << endl;
		else
			cout << "NO" << endl;
	}
	return 0;
}
