#include <bits/stdc++.h>
#define ll long long 
using namespace std;
int main()
{
  	int t;
  	cin>>t;
  	while(t--)
  	{
  		ll num,sum=0;
  		cin>>num;
  		ll int i=1;
  		ll int Nd = 1 + (int)log10(num);   
    ll int tenPow = 1;
    for(ll int d=1; d<Nd; ++d)
    {
        sum += 9*tenPow*d;
        tenPow *= 10;
    }
    sum += Nd*( num + 1 - tenPow );
  		cout<<sum<<endl;
  	}
}
