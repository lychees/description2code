#include <iostream>
using namespace std;

int main()
{
    int q;
    cin >> q;
    
    while (q-- > 0) {
    	long long n;
    	int t;
    	cin >> n >> t;
    	
    	while (t-- > 0) {
    		long long take = (n+1)/2;
    		n -= take;
    		
    		take = n/4;
    		n -= take;
    	}
    	
    	cout << n << endl;
    }
    return 0;
}
