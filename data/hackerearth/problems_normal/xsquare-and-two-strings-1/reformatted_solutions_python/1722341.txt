def has_subsequence(s1, s2):
    print "Yes" if set(s1).intersection(set(s2)) else "No"


cases = int(raw_input())
for _ in xrange(cases):
    s1 = raw_input()
    s2 = raw_input()
    has_subsequence(s1, s2)
