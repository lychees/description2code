t = int(input())
vowel = set(['a', 'e', 'i', 'o', 'u', 'y'])
while t > 0:
	t -= 1
	vo = 0
	co = 0
	ct = 0
	s = raw_input()
	flag = False
	for c in s:
		if c in vowel:
			vo += 1
			co = 0
		else:
			co += 1
			ct += 1
		if co == 3:
			flag = True
	flag |= (ct > vo)
	if flag:
		print "hard"
	else:
		print "easy"
