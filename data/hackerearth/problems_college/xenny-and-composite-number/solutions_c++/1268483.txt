#include<stdio.h>
#include<algorithm>
#include<iostream>
#include<cmath>
using namespace std;
bool is_composite(unsigned long long int n)
{
	
	bool flag=true;
	for(unsigned long long int i=2;i<=sqrt(n);i++)
	{
		if(n%i==0)
		{
			flag=false;break;
			}
		
		}
		if(!flag)return true;
		else return false;
	
	
	
	
	
	
	
	}
int main()
{
	unsigned long long int t=0;
	scanf("%llu",&t);
	while(t>0)
	{
		
		unsigned long long int n=0,s=0;
		scanf("%llu",&n);
		s=n+1;
		while(1)
		{
			
			if(is_composite(s))
			{
				
			printf("%llu\n",s);break;
		}
		else
		s++;
			
			
			
						}//end of the while loop
		
		t--;
		}
	
	
	
	
	
	
	
	
	
	
	
	}
