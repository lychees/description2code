#include <iostream>
#include <cmath>
using namespace std;
int minAngle(int h,int m){
	if(h==12) h=0;
    int hour_angle = 0.5 * (h*60 + m);
    int minute_angle = 6*m;
    int angle = abs(hour_angle - minute_angle);
    return angle;
}
int min(int a,int b){
    return a<b?a:b;
}
int main()
{
    int t,h,m,q,ang;
    cin>>t;
    while(t--){
        ang=0;
        cin>>h;
        cin>>m;
        q=minAngle(h,m);
        ang=min(360-q,q);
        cout<<ang<<endl;
    }
    return 0;
}
