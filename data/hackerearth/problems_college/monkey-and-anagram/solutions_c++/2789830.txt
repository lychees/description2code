#include<bits/stdc++.h>
using namespace std;

int main()
{
   int i,j,k,m,n,t1,x,c,d;
   scanf("%d",&t1);
   while(t1--)
   {
           string s,t;c=0;
           cin>>s>>t;
           m=s.length();n=t.length();
           int a[30]={0},b[30]={0};
           for(i=0;i<m;i++) a[s[i]-'a']++;
           for(i=0;i<n;i++) b[t[i]-'a']++;
           for(i=0;i<26;i++){
                if(a[i]<b[i]) c++;
           }
           if(c>(n-m)) cout<<"-1"<<endl;
           else cout<<c<<endl;
   }
  return 0;
}
