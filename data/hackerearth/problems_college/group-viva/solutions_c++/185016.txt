#include <bits/stdc++.h>
using namespace std ;

int fun( int x ) {

	int count = 0 ;
	for( int i = 1 ; i*i <= x ; i++ )
		if( x%i == 0 && x/i == i ) count ++ ;
		else if( x%i == 0 ) count += 2 ;
	return count ;
}


int main() {

	int t , n , r ;
	cin >> t ;
	while( t-- ) {
		cin >> n >> r ;
		int num = fun(r-1) ;
		cout << (n-num)/__gcd((n-num),n ) << "/" << n/__gcd((n-num),n) << endl ;
	}
	return 0 ;
}
