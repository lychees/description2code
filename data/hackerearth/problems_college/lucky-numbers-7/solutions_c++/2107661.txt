#include <iostream>
using namespace std;

int lucky(int n)
{
	int i, r, cnt=0;
	while(n)
	{
		r=n%10;
		if(r==4 || r==7)
			cnt++;
		n/=10;
	}
	return cnt;
}

int main()
{
    //cout << "Hello World!" << endl;
    int a, l, b, i, flag, j;
    cin>>a>>l;
    int a1=lucky(a);
    for(i=a+1; ; ++i)
    {
    	if(a1==lucky(i))
    	{
    		b=i;
    		flag=0;
    		for(j=1; j<=l-1; ++j)
    		{
    			if(lucky(a+j)!=lucky(b+j))
    			{
    				flag=1;
    				break;
    			}
    		}
    		if(flag==0)
    		{
    			cout<<b;
    			return 0;
    		}
    	}
    }
    return 0;
}
