#include<iostream>
#include<cstdio>
using namespace std;
long long int countSetBits(long long int n)
{
  long long int count = 0;
  while(n)
  {
    count += n & 1;
    n >>= 1;
  }
  return count;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int x,y;
		scanf("%lld %lld",&x,&y);
		long long int c=(x^y );
		
		cout<<countSetBits(c)<<endl;
	}
}
