#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{ int tc;
  cin>>tc;
  while(tc--)
  { string str;
    int curr=0;
    cin>>str;
    int i;
    for(i=0;i<str.size();i++)
    { if((str[i]-'a')>=curr)
      { int v=str[i]-'a'-curr;
        int v1=26-v;
        //cout<<"Condition 1 : v is "<<v<<" "<<"v1 is "<<v1<<"\n";
        if(v<=v1)
        { cout<<v<<" ";
        }
        else
        { cout<<"-"<<v1<<" ";
        }
        curr=str[i]-'a';  
      }
      else
      { int v=26-curr+str[i]-'a';
        int v1=curr-(str[i]-'a');
        //cout<<"Condition 2 : v is "<<v<<" "<<"v1 is "<<v1<<"\n";
        if(v<=v1)
          cout<<v<<" ";
        else
          cout<<"-"<<v1<<" ";
        curr=str[i]-'a';
      }
    }
    cout<<"\n";
  }
  
 return 0;
}
