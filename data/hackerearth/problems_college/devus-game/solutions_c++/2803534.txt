#include <iostream>
#include <cstring>
#include <cstdio>
#include <cmath>
#include <functional>
#include <algorithm>

using namespace std;

int nlist[100005];

int main()
{
	int tc;
	cin >> tc;

	int n,j,temp;

	while(tc--)
	{	
		memset(nlist,0,100005);

		cin >> n;
		
		for(int i = 0 ; i < n ; i++ )
		{
			cin >> nlist[i];
		}

		j=0;
		while(j>=0 && j < n)
		{
			if(nlist[j] == -1 || nlist[j] >=n )
			{
				cout << "Win" << endl;
				break;
			}
			else if(nlist[j] == -2 )
			{
				cout << "Lose" << endl;
				break;
			}
			else
			{
				temp= nlist[j];
				nlist[j] = -2;
				j= temp;
			}
			if( j == -1 || j >= n) { cout << "Win" << endl; }
			if( j == -2) { cout << "Lose" << endl; } 
			//cout << "Loop " << j << endl;
		}

	}
	return 0;
}
