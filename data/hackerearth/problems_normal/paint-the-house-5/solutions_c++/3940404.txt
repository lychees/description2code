#include <bits/stdc++.h>
using namespace std;

int main()
{
   int n,i,t,x,y;
   long long int d;
   cin>>n;
   cin>>d;
   vector <pair <int, int > > v(100010,make_pair(0,100010));
   for(i=1;i<=n;i++)
   {
   	cin>>t;
   	cin>>x;
   	cin>>y;
   	if(v[t].first<y)
   	{
   		v[t].first=y;
   		v[t].second=x;
   	}
   	else if(v[t].first==y && v[t].second>x)
   	{
   		v[t].second=x;
   	}
   }
   long long int count1=0;
   long long int speed=0;
   long long int cost=0;
   long long int cost_sum=0;
   i=1;
   while(i<=n)
   {
   	if(v[i].first!=0 && (v[i].first>speed))
   	{
   	speed=v[i].first;
   	cost=v[i].second;
   	cost_sum+=cost;
   	}
   	count1+=speed;
   	if(count1>=d)
   	break;
   	i++;
   }
   cout<<cost_sum<<endl;
    return 0;
}
