  #include <iostream>
#include <string>

using namespace std;

main()
{
    const int max=1000;
    int n;    cin>>n;   string t[max];    int b[n];
    string s[max];   char c[n];    int d=0;
    for (int i=0; i< n; i++)
   {
       cin>>s[i];
       cin>>c[i];
   }
   for ( int i=0; i< n; i++)
    {
        d = 0;
        t[i] = s[i];
        for(int j=0; j< t[i].length(); j++)
        {
            if(c[i] == t[i][j])
                d++;
        }
        b[i] = d;
    }
 for (int i=0; i< n; i++)
 {
     cout<<b[i]<<endl;
 }


}
