#include<iostream>
using namespace std;
int gcd(int s,int q)
{
	while(s!=q)
	{
		if(s>q)
			s=s-q;
		else
			q=q-s;
	}
	return s;
}

using namespace std;
int main()
{
	
		
		int s,q;
		 int arr[1000000]={};
		cin>>s>>q;
		while(q--)
		{
			int no,m;
			cin>>no;
			m=gcd(s,no);

			if(arr[m]==0)
				{
					cout<<m<<endl;
					arr[m]=1;
				}
			else
				cout<<-1<<endl;


		}
return 0;
}
