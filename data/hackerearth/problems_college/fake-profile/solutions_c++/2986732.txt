#include <iostream>
#include<algorithm>
#include<cstring>
using namespace std;

int main()
{
    char s[100];
    cin>>s;
    sort(s,s+strlen(s));
    int i=0,cnt=0;
    while(s[i]!='\0')
    {
    	if(s[i+1]!='\0')
    	{
    	if(s[i]!=s[i+1])
    	cnt++;
    	}
    	i++;
    }
    if(cnt%2)
    {
    cout<<"CHAT WITH HER!"<<endl;
    }
    else
    cout<<"IGNORE HIM!"<<endl;
    return 0;
}
