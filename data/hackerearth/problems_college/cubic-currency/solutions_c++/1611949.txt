#include <iostream>

using namespace std;

long long int p[10005],ans[10005];
int main()
{long long int i,j,t,n;
    for(i=0;i<=21;i++)
        p[i]=i*i*i;

    ans[0]=1;
    for(i=1;i<=10000;i++)
        ans[i]=0;
    for(i=1;i<=21;i++)
    {
       ans[p[i]]++;
       for(j=p[i]+1;j<=10000;j++)
       {
           ans[j]+=ans[j-p[i]];
       }

    }

    cin>>t;
    while(t)
    {
        cin>>n;
        cout<<ans[n]<<endl;
        t--;

    }

    return 0;
}
