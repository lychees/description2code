#include<iostream>
#include<string>
#include<map>

using namespace std;

int main()
{
	string input;
	getline(cin, input);
	map<char, int> occurence;

	for (auto item : input)
	{
		occurence[item] += 1;
	}

	char chMax = occurence[input[0]];

	for (auto item : occurence)
	{
		if (item.second > occurence[chMax])
			chMax = item.first;
		else if ((item.second == occurence[chMax]) && (item.second < occurence[chMax]))
			chMax = item.first;
	}

	cout << chMax <<" "<<occurence[chMax];
	return 0;
}
