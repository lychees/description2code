#include <cstdio>
#include <iostream>
 
using namespace std;
 
int gcd(int a, int b) {
	if(b == 0) return a;
	return gcd(b, a%b);
}
 
int main(void) {
	int t;
	scanf("%d", &t);
	for(int j = 0; j < t; j++) {
		int n;
		scanf("%d", &n);
		int num[n], den[n];
		int tnum[n], tden[n];
		int p[n];
		for(int i = 0; i < n; i++) {
			scanf("%d/%d", &num[i], &den[i]);
		}
		
		for(int i = n-1; i >= 0; i--) {
			tnum[i] = num[i];
			tden[i] = den[i];
			p[i] = i;
			for(int j = i+1; j < n; j = p[i]+1) {
				if(((float)tnum[i]/tden[i]) >= ((float)tnum[j]/tden[j])) break;
				tnum[i] += tnum[j];
				tden[i] += tden[j];
				p[i] = p[j];
			}
		}
		
		for(int i = 0; i < n; i++) {
			int g = gcd(tnum[i], tden[i]);
			printf("%d/%d\n", tnum[i]/g, tden[i]/g);
		}
		printf("\n");
	}
	return 0;
}
