#include <iostream>
using namespace std;
int a[101][101][502];
int main()
{
    int t;
    cin>>t;
    
    while(t--){
    	int n,m,p;
    	cin>>n>>m>>p;
    	int b[n][m];
    	for(int i=0;i<n;i++){
    		for(int j=0;j<m;j++){
    			cin>>b[i][j];
    			for(int k=0;k<=p;k++)
    			  a[i][j][k]=0;
    		}
    	}
    	a[0][0][b[0][0]]=1;
    	for(int i=0;i<n;i++){
    		for(int j=0;j<m;j++){
    			for(int k=b[i][j];k<=p;k++){
    				if(i==0&&j==0){
    					if(k==b[i][j])
    					a[i][j][k]=1;
    				}
    				else if(i==0){
    					a[i][j][k]=(a[i][j][k]||a[i][j-1][k-b[i][j]]);
    				}
    				else if(j==0){
    				    a[i][j][k]=(a[i][j][k]||a[i-1][j][k-b[i][j]]);	
    				}else{
    					a[i][j][k]=(a[i][j][k]||a[i][j-1][k-b[i][j]]);
    					a[i][j][k]=(a[i][j][k]||a[i-1][j][k-b[i][j]]);
    					a[i][j][k]=(a[i][j][k]||a[i-1][j-1][k-b[i][j]]);
    				}
    			//	cout<<a[i][j][k]<<endl;
    			}
    		}
    	}int m1=-1;
    	int f=0;
    	for(int i=b[0][0];i<=p;i++){
    		if(a[n-1][m-1][i]>0){
    			m1=i;
    			f=1;
    		}
    		
    	}
    //	cout<<a[n-1][m-1][m1]<<endl;
    	cout<<m1<<endl;
    	
    }
    return 0;
}
