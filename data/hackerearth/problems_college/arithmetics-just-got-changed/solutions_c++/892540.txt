#include<iostream>
using namespace std;
long long int makRev(long long int num)
{
int rem;
long long int rev=0;
while(num>0)
{
rem = num%10;
rev = rev * 10 + rem;
num = num/10;
}
//cout<<"rev is"<<rev<<endl;
return rev;
}
int main()
{
long long int t,num1,num2,res;
cin>>t;
while(t--)
{
cin>>num1>>num2;
if(num1!=0)
num1 = makRev(num1);
if(num2!=0)
num2 = makRev(num2);
res = makRev(num1+num2);
cout<<res<<endl;
}
return 0;
}
