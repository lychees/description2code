#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
    int N;
    scanf("%d", &N);
    if(N== 1)
    {
    	printf("YES \n");
    	printf("0 0");
    }
    else if(N== 2) 
    {
    	printf("YES \n");
    	printf("0 0 \n");
    	printf("1 0 \n");
    }
    else if(N == 4)
    {
    	printf("YES \n");
    	printf("0 0 \n");
    	printf("0 1 \n");
    	printf("1 1 \n");
    	printf("1 0 \n");
    }
    else
    {
    	printf("NO");
    }
    return 0; 
}
