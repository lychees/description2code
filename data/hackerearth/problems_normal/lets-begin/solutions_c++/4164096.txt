int ar[] = {2,3,5,7}; // single digit prime numbers
int recurse(int x,int n) {
    if (x==0) return 1;
    if (x<0) return 0;
    if (n<=0 && x>0) return 0;
    return recurse(x,n-1) + recurse (x-ar[n-1],n);
}
int minNum(int *ab,int v,int n) {
    if (v==0) return 0;
    int res = maxInt;
    fr(i,0,n) {
        if (ar[i]<=v) {
            int subres;
            if (ab[v-ar[i]] !=-1) {
                subres = ab[v-ar[i]];
            } else {
             subres = minNum(ab,v-ar[i],n);
                ab[v-ar[i]] = subres;
            }
            if (subres!=maxInt && subres+1 < res) res = subres+1;
        }
    }
    return res;
}
