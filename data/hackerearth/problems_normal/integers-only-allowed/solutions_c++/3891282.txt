#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    long long a[100010],b[100010];
    cin>>n;
    for(int i=0;i<n;i++){
        cin>>a[i]>>b[i];
    }
    a[n]=a[0];
    b[n]=b[0];
    long long t=0;
    for(int i=0;i<n;i++){
        t+=1ll*((a[i]+a[i+1])*(b[i+1]-b[i]));
    }
    long long x=0;
    for(int i=0;i<n;i++){
        x+=__gcd(abs(a[i]-a[i+1]),abs(b[i+1]-b[i]));
    }
    t=abs(t);

    cout<<(t+2-x)/2<<endl;
    return 0;
}
