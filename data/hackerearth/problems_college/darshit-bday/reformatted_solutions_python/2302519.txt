for T in range(input()):
    b, w = map(int, raw_input().rstrip().split(' '))
    data = [int(i) for i in raw_input().rstrip().split(' ')]
    x = data[0]
    y = data[1]
    z = data[2]
    total = 0
    if y > z+x:
        total = b*x + w*(z+x)
    elif x > z+y:
        total = b*(z+y) + w*y
    else:
        total = b*x + w*y
    print total
