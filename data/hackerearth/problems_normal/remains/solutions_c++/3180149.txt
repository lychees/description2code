#define mset(x,v) memset(x, v, sizeof(x))
#define sz(x) (int)x.size()

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,i;
        ll x,y,z;
        scanf("%lld%lld%d",&x,&y,&n);
        ll ans = x + y;
        for(i=0;i<n-2;i++)
        {
            z = max(x,y) - min(x,y);
            if(x == y)
            {
                ans += 2*x*((n-2-i)/3);
                i += ((n-2-i)/3)*3;
            }
            ans += z;
            x = y;
            y = z;
        }
        printf("%lld\n",ans);
    }
    return 0;
}
