#include <iostream>
#include<string.h>
using namespace std;

int main() 
{
int occ[26];
int t;
cin>>t;
for(int q=0;q<t;q++)
{
for(int i =0;i<26;i++)
occ[i]=0;
string s;
int len;
cin>>s;
len=s.length();
for(int i=0;i<len;i++)
 occ[s[i]-97]++;
int odd=0;
for(int i=0;i<26;i++)
if(occ[i]%2 != 0) odd++;
if(odd>0) cout<<odd-1<<endl;
else cout<<0<<endl;
}
	return 0;
}
