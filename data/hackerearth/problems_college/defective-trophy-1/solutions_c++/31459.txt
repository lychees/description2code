#include<iostream>
#include<math.h>

using namespace std;
int cal(int n)
{
    int x,a[3];
      x=int(ceil(n/3));
     a[0]=(n-x)/2;
     a[1]=a[0];
     a[2]=n-a[0]-a[1];
  //  cout<<a[0]<<a[1]<<a[2]<<endl;
     if(a[0]>=a[1] && a[0]>=a[2])
     return a[0];
     else if(a[1]>=a[2])
     return a[1];
     else
     return a[2];
    }
int main()
{
    int t;
    cin>>t;
    while(t--){
     int n,x,a[3],i=0;
     cin>>n;
     if(n<=1)
     { cout<<0<<endl;
     continue;
     }
     while(n>3)
     {
     n=cal(n);
     i++;
     }
     
     cout<<i+1<<endl;
  }  // system("pause");
   return 0;  
     }
