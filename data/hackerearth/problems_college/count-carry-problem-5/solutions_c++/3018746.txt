#include<iostream>
using namespace std;
int main()
{
long long int t;
cin>>t;
while(t-- > 0)
{
	unsigned long long m,n;
	cin>>m>>n;
	unsigned long long car=0,carry=0;
	int c=0;
	unsigned long long r1=0,r2=0;
	while(m>0 && n>0)
	{
		r1=m%10;
		r2=n%10;
		car=r1+r2;
		car=car+carry;
		if(car>=10)
		{
			carry=carry/10;
			c++;	
		}
		m=m/10;
		n=n/10;
	}
	if(c==0)
	{cout<<"No carry operation"<<endl;
	}
	else if(c==1)
	{
	cout<<c<<" carry operation"<<endl;
	}
	else
	{cout<<c<<" carry operations"<<endl;}
}
}
