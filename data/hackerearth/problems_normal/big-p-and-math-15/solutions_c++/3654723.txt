#include<iostream>
#include<string>
using namespace std;

int max(int x)
{
string str=to_string(x);
int len=str.length();
for(int i=0;i<len;i++)
if(str[i]=='5')
str[i]='6';
return stoi(str,nullptr,10);	
}
int min(int x)
{
string str=to_string(x);
int len=str.length();
for(int i=0;i<len;i++)
if(str[i]=='6')
str[i]='5';
return stoi(str,nullptr,10);	
}

int main()
{
	int a,b;
	cin>>a>>b;
	cout<<(min(a)+min(b))<<" "<<(max(a)+max(b));
	return 0;
}
