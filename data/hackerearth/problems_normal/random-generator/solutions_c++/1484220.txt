		cin>>n>>k>>p;
		VI ar;
		VI::iterator u;
		for(i=0;i<n;i++)
		{
			cin>>x;
			ar.push_back(x);
		}
		sortv(ar);
		int temp;
		for (i = 0;i<n; ++i)
		{
			temp=ar[i]+2*p;
			u=upper_bound(ar.begin(),ar.end(),temp);//returns iterator

			}
		}
