#include<iostream>
#include<cmath>
using namespace std;
typedef long long int ll;
int main()
{
    ll t,n,p,i;
    cin>>t;
    while(t--)
    {
        cin>>n>>p;
        ll a[n],sm=0,f=0;
        for(i=0;i<n;i++)
        {
            cin>>a[i];
            if(a[i]>p)
                f++;
            if(a[i]==p)
                sm=1;
        }
        if(sm==1)
            cout<<"YES "<<f<<'\n';
        else
            cout<<"NO "<<0<<'\n';
    }
    return 0;
}
