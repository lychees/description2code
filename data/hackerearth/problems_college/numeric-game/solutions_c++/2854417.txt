#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
    int t,a1,a2,b1,b2,a,b;
    scanf("%d",&t);
    int I=0;
    while(I<t){
    scanf("%d%d%d%d",&a1,&a2,&b1,&b2);
   
    int count=0;
    for(int i=a1;i<=a2;i++){
    	for(int j=b1;j<=b2;j++){
    		int tmp=0;
    		int d;
    		if(i>j)
    		 {
    		 	a=i;
    		 	b=j;
    		 }
    		 else{
    		 	a=j;
    		 	b=i;
    		 }
    		 while(1){
    		 d=a/b;
    		 int min_v=99999999;
    		 int min_k;
    		 for(int k=1;k<=d;k++){
    		 	int value=a-k*b;
    		 	if(abs(b-value)<min_v){
    		 		min_v=abs(b-value);
    		 		min_k=k;
    		 	}
    		 }
    		 
    		 a=a-min_k*b;
    		 tmp++;
    		 if(b>a)
    		 	{
    		 		int temp=a;
    		 		a=b;
    		 		b=temp;
    		 	}
    		 
    		 
    		 if(a<=0 || b<=0)
    		 	break;
    		 }
    		 
    		 
    		 if(tmp%2==0)
    		 	count++;
    	}
    }
   printf("Case #%d: %d\n",I+1,count);
    I++;
    }
    
    return 0;
}
