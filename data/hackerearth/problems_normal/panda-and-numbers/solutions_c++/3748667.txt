#include <iostream>
#include<bits/stdc++.h>
#define ll long long
using namespace std;

int main()
{
    ll t,n,d,temp;
    cin>>t;
    ll a[1000006]={0};
  for(ll i=1;i<7;i++)
   a[(ll)pow(i,i)]=1;
 for(ll i=10;i<1000005;i++){
     temp=i;
      while(temp>0){
            d=temp%10;
            temp/=10;
            if(a[i-d*d]==1){
                a[i]=1;
                break;
            }
        }
 }
    while(t--){
        cin>>n;
        temp=n;ll flag=0;
        d=0;
        while(temp>0){
            if(a[n-d*d]==1){
                flag=1;
                break;
            }
            d=temp%10;
            temp/=10;
        }
     if(flag)cout<<"Yes"<<endl;
     else cout<<"No"<<endl;
    }
    return 0;
}
