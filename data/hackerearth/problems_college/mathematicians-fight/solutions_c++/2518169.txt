#include <bits/stdc++.h>
using namespace std;

// int dp[100000001];
int count(int x){
	if(x == 1 || x == 2) return 1;
	if(x%2 == 0) return 2*count(x/2) -1;
	else return 2*count(x/2) + 1;
}
int main() {
	int t;
	cin >> t;
	while(t--){
		int n;
		cin >> n ;
		cout << count(n) << endl;
	}
 	return 0;
}
