#include <bits/stdc++.h>
using namespace std;
int n,q,wt;
string str; 
int trie[1000010][26];
long long cnt[1000010];
int pool = 0,l=0;  
long long fans[1000010];
void insert(){
    int r = 0 ;
    for(int i = 0 ; i < str.length() ; ++i){
        fans[i] += cnt[r]; 
        cnt[r] += 1; 
        int c = str[i] - 'a' ; 
        if(trie[r][c] == 0){
            trie[r][c] = ++pool ; 
        }
        r = trie[r][c] ; 
    }
    fans[(int)str.length()] += cnt[r] ; 
    cnt[r] += 1; 
}

int main(){
   // freopen("input.txt","r",stdin) ;
    ios_base::sync_with_stdio(0) ;
    cin.tie(0) ; 
    int n;
    cin >> n; 
    for(int i = 0; i < n; ++i){
        cin >> str;
        l = max((int)str.length() ,l);
        insert() ;
    }
    for(int i = 0;  i <= l ; ++i){
        cout << fans[i] - fans[i+1] << " " ; 
    }
    return 0;
}
