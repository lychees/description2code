t = input()
while t != 0:
    t -= 1
    r = input()
    ai = r-1
    b = map(int, raw_input().split())
    e = map(int, raw_input().split())
    c = b[0]

    d = sorted(b[1:r])
    e.sort()
    f = c*e[r-1]

    for i in range(r-1):
        ai -= 1
        h = 0
        u = 0
        h = e[i]*d[ai]
        if h < f:
            u = 1
        else:
            print "NO"
            break

    if u == 1:
        print "YES"
