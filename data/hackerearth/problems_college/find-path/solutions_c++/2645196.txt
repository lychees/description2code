#include <iostream>
using namespace std;

int main()
{
	int t,n,i;
    cin>>t;
    while(t--){
    	int count=0;
    	cin>>n;
    	int a[n];
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	for(i=0;i<n-1;i++)
    	{
    		if((a[i+1]-a[i])!=1)
    		{
    			count++;
    		}
    	}
    	cout<<count<<endl;
    }
    return 0;
}
