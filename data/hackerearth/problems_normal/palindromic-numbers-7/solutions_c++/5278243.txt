#include<iostream>
using namespace std;
int main()
{
int num,r,sum,temp,t,count=0;
long int min,max;
cin>>t;
while(t--)
{
cin>>min>>max;
for(num=min;num<=max;num++)
{
temp=num;
sum=0;
while(temp)
{
r=temp%10;
temp=temp/10;
sum=sum*10+r;
}
if(num==sum)
count+=1;

}
cout<<count<<endl;
count=0;
}
return 0;
}
