#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;cin>>t;
	while(t--)
	{
		string a,b;
		cin>>a>>b;
		if(a.compare(b)==0)
			cout<<"YES";
		else
		{
			stringstream s1,s2;
		s1<<a;s2<<b;
		int l,m;
		s1>>l,s2>>m;
		if((l==2&&m==4)||(l==4&&m==2))
			cout<<"YES";
		else
			cout<<"NO";
		}
		cout<<endl;
	}
    return 0;
}
