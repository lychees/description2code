#include<bits/stdc++.h>
using namespace std;

typedef long long ll;

int main() {
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        ll n;
        scanf("%lld",&n);
        ll res=2;
        int l;
        for(int i=1;i<=60;i++) {
            if(res*i>=n) {
                l=i;
                break;
            } else {
                n-=res*i;
                res*=2;
            }
        }
        ll t1=(n-1)/l;
        ll t2=(n-1)%l;
        vector<int>v;
        while(t1) {
            v.push_back(t1%2);
            t1/=2;
        }
        for(int i=v.size();i<l;i++) {
            v.push_back(0);
        }
        if(v[v.size()-t2-1]==1) {
            puts("Earth");
        } else {
            puts("Hacker");
        }
    }
    return 0;
}
