def is_odd(lista):
    return len(lista) % 2 == 1


def sum_string(lista, n):
    for x in range(n, len(lista)-1):
        lista[x] = lista[x+1]
    lista.pop()
    return lista


def crypt(string_):
    enc_string = []
    string = [x for x in string_]
    while True:
        if len(string) == 0:
            break
        if is_odd(string):
            middle = len(string)//2
            enc_string.append(string[middle])
            sum_string(string, middle)
        else:
            if string[len(string)//2 - 1] < string[len(string)//2]:
                enc_string.append(string[len(string)//2-1])
                sum_string(string, len(string)//2 - 1)
            else:
                enc_string.append(string[len(string)//2])
                sum_string(string, len(string)//2)

    return "".join(enc_string)


output = []
input_n = int(raw_input())
for x in range(input_n):
    input_string = raw_input()
    output.append(crypt(input_string))

for x in output:
    print(x)
