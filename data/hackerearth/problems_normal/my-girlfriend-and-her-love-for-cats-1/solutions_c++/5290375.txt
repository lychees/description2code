#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n;
    cin>>n;
    long long int s[n];
    long long int c[n];
    for(int i=0;i<n;i++)
    	cin>>s[i];
    for(int i=0;i<n;i++)
    	cin>>c[i];
    sort(s,s+n);
    sort(c,c+n);
    long long int sum=0;
    for(int i=0;i<n;i++)
    {
    	sum+=(s[i]*c[i]);
    }
    cout<<sum;
    return 0;
}
