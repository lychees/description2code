#include <iostream>
#include<string>
using namespace std;
 
int main()
{
	int n;
	cin>>n;
	
	while(n--){
	
	string str;
    int stringlen,flag=0;
    long long int product=1;
    
    cin>>str;
    
    stringlen=str.length();
    
    for(int i=0; i<stringlen; i++){
    	
    	product=product*(str.at(i)-96);
    	
    	if(str.at(i)!=str.at(stringlen-1-i)){
    	flag=1;
    	
    	
    	}
    }
    
    if(flag){
    cout<<product<<endl;
    }
    else
    cout<<"Palindrome"<<endl;
	}
    return 0;
}
