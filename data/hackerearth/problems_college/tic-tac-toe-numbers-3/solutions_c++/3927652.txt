#include <iostream>
using namespace std;
long n =100000002;
char prime[100000002];
char nfib[1000002];
long fib[1001];
void siev()
{
	long i,p;
	for(i = 0;i <= n; i++){
		prime[i] = 1;
	}
	for(p = 2; p*p <= n; p++){
		if(prime[p] == 1){
			for(i = p*2; i <= n; i += p){
				prime[i] = 0;
			}
		}
	}
}
int main()
{
	fib[0] = 0;
	fib[1] = 1;
	long long i,t,num,sum,flag;
	for(i=2;fib[i-1]<=1000000;i++){
		fib[i] = fib[i-1] + fib[i-2];
		nfib[fib[i]] = 1;
	}
    cin >> t;
    while(t--){
    	cin >> num;
    	sum = 0;
    	for(i=1;i<num;i++){
    		if(nfib[i] == 1){
    			continue;
    		}
    		sum = sum + i;
    	}
    	if(sum % 2 == 0){
    		cout << "Tac\n";
    		continue;
    	}
    	flag = 1;
    	for(i=2 ;i*i <=sum ;i++){
    		if(sum % i == 0){
    			flag = 0;
    		}
    	}
    	if(flag == 1){
    		cout << "Tic\n";
    	}
    	else{
    		cout<< "Toe\n";
    	}
    }
    return 0;
}
