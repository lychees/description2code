#include <iostream>
#include<algorithm>
#include<cstring>
#define m 1000000007
using namespace std;

long long a[100005];
long long fact[100005];

long long fast_pow(long long p,long long q)
{
	long long ret = 1;
	while(q>0)
	{
		if(q&1)
			ret = (ret*p)%m;
		p=(p*p)%m;
		q/=2;
	}
	return ret;
}

void pre()
{
	fact[0] = 1;
	for(long long i=1;i<100000;i++)
		fact[i] = (fact[i-1]*i)%m;
	return ;
}

long long f(long long p,long long r)
{
	if(r>p)
		return 0;
	long long ans = fact[p];
	long long ans1 = (fact[r]*fact[p-r])%m;
	ans = (ans*fast_pow(ans1,m-2))%m;
	
	return ans;
}

int main()
{
    long long t,n,k;
    cin>>t;
    pre();
    while(t--)
    {
    	cin>>n>>k;
    	for(long long i=0;i<n;i++)
    		cin>>a[i];
    	sort(a,a+n);
    	long long ans =0;
    	for(long long i=0;i<n;i++)
    	{
    		long long val = a[i];
    		val = (val*f(n-i-1,k-1))%m;
    		ans+=val%m;
    		ans%=m;
    	}
    	cout<<ans<<endl;
    	
    }
    return 0;
}
