#include<bits/stdc++.h>
#define mp make_pair
# define pb push_back
using namespace std;
typedef long long lo;
lo n,dp[105][105],a[105][105];
lo func(lo i,lo j)
{
if(i==n&&j==n)
return 0;
if (dp[i][j]!=0)
return dp[i][j];
if((i<=n&&j+1<=n)&&(i+1<=n&&j<=n))
return dp[i][j]=max(dp[i][j],a[i][j]+max(func(i,j+1),func(i+1,j)));
else if(i<=n&&j+1<=n)
return dp[i][j]=max(dp[i][j],a[i][j]+func(i,j+1));
else if(i+1<=n&&j<=n)
return dp[i][j]=max(dp[i][j],a[i][j]+func(i+1,j));

}
int main()
{
lo t,c,i,j,ans;
scanf("%lld",&t);
while(t--)
{
memset(dp,0,sizeof(dp));
scanf("%lld %lld",&n,&c);
for(i=1;i<=n;i++)
for(j=1;j<=n;j++)
scanf("%lld",&a[i][j]);
ans=func(1,1);
if(ans<c)
printf("-1\n");
else
printf("%lld\n",ans-c);
}
}
