#include <iostream>
using namespace std;

int main()
{
   int n,i,j,k;
   long a[10000000];
   long sum1=0;
   long sum2=0;
   long sum3=0;
   cin>>n;
   
   for(i=1;i<=n;i++)
   {
   	cin>>a[i];
   }
   for(i=1;i<=n;i=i+3)
   {
   	sum1=sum1+a[i];
   }
   for(j=2;j<=n;j=j+3)
   {
   	sum2=sum2+a[j];
   }
   for(k=3;k<=n;k=k+3)
   {
   	if(k>n)
   	sum3=sum3+0;
   	else
   	sum3=sum3+a[k];
   }
   cout<<sum1<<" "<<sum2<<" "<<sum3;
}
