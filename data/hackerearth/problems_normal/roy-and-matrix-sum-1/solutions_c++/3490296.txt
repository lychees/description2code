#include <iostream>
using namespace std;

int main()
{
    int t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	n = n - 1;
    	long long int sum,ans,temp,z;
    	sum = ans = 0;
    	sum = n;
    	ans = sum + 1;
    	if(sum & 1)ans = ans/2;
    	else sum = sum/2;
    	sum = sum * ans;
    	ans = sum;
    	sum = n+1;
    	temp = n;
    	z = 2*n;
    	z = z + 1;
    	sum = sum * temp;
    	sum = sum * z;
    	sum = sum / 6;
    	ans = ans + sum;
    	cout<<ans<<endl;
    }
    return 0;
}
