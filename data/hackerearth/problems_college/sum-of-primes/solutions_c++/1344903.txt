#include <bits/stdc++.h>
using namespace std;
#define LL long long
#define MOD 1000000007
int is_prime[1000006];
unsigned long long sumprime[1000006];
void sieve(){
    long long i , j ;
    is_prime[1]=1;
    for(i=2;i<=1000000;i++){
        if(!(is_prime[i])){
            for(j=2*i;j<=1000000;j+=i){
                is_prime[j]=1;
            }
        }
    }
}
int check_prime( unsigned long long int a){
	 unsigned long long  c;
	for ( c = 2 ; c <= a/c ; c++ ){
		if ( a%c == 0 )
		return 0;
	}
	return 1;
}
int main(){

unsigned long long t , m , n , i , j , k;
sieve();
sumprime[2]=2;

for(i=3;i<=1000000;i++){
    k = 0;
    if(!is_prime[i]){
        k = -1;
    }
    if(k==0){
        sumprime[i]=sumprime[i-1];
    }
    else{
        sumprime[i]=(sumprime[i-1]+i);
    }
}
cin >> t ;
while(t--){
    cin >> m >> n;
    if(sumprime[n]==sumprime[m-1]){
    	cout<<"NO\n";
    	continue;
    }
	//cout << (sumprime[n]-sumprime[m-1])%MOD << endl;
    if(check_prime((sumprime[n]-sumprime[m-1]))){
        cout << "YES\n";
    }
    else{
        cout << "NO\n";
    }
        
}
    return 0;
}
