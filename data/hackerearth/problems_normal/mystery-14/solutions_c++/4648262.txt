#include <iostream>
using namespace std;

int main()
{
    int t,n,a[1000009];
    cin>>t;
    while(t--)
    {
    	int sum;
    	cin>>n;
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    		if(a[i]%2==1)
    		{
    			sum+=a[i];
    		}
    	}
    	cout<<sum<<"\n";sum=0;
    	
    }
    return 0;
}
