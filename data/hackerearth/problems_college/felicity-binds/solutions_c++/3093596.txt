#include<bits/stdc++.h>
using namespace std;
 
int g[1005], b[1005];
 
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int ans = 0;
        int n;
        cin>>n;
        for(int i=0; i<n; i++)
            scanf("%d",&g[i]);
        for(int i=0; i<n; i++)
            scanf("%d",&b[i]);
        sort(&b[0], &b[n]);
        sort(&g[0], &g[n]);
        for(int i=0; i<n; i++)
            ans += g[i]*b[i];
        cout<<ans<<endl;
    }
    
    return 0;
}
