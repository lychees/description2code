#include <iostream>
#include <iterator>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    ios::sync_with_stdio(false);
    int n, m;
    cin >> n >> m;
    vector<int> coins(m);
    copy_n(istream_iterator<int>(cin), m, coins.begin());
    vector<long long> dp(n+1, 0);
    dp[0] = 1;
    for (int i = 0; i < m; ++i) {
        for (int j = 1; j <= n; ++j) {
            if (j - coins[i] >= 0) {
                dp[j] += dp[j-coins[i]];
            }
        }
    }
    cout << dp[n] << endl;
    return 0;
}
