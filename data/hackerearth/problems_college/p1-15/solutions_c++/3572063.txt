#include<bits/stdc++.h>
using namespace std;
typedef pair<int,int> pp;
typedef long long ll;
 
string pat,txt;
int z[1000006];
		
 
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		cin>>txt>>pat;
		string g=txt;
		int i=0,j=txt.length();
/*		while(i<j)
		{
			char temp=txt[i];
			txt[i]=txt[j];
			txt[j]=temp;
			i++; j--;
		}
*/		txt=pat+txt;
        int n=txt.length();
        int l=0,r=0;
        z[0]=0;
        for(int i=1;i<n;i++)
        {
            z[i]=0;
            if(i<r)
            {
                if(z[i-l]<r-i+1)
                    z[i]=z[i-l];
                else
                {
                    l=i;
                    while(r<n && txt[r]==txt[r-l])	r++;
                    z[i]=r-l;
                    r--;
                }
            }
            else
            {
                l=r=i;
                while(r<n && txt[r]==txt[r-l])
                    r++;
                z[i]=r-l;
                r--;
            }
        }
        int cc=-1;
        int ans=0;
        for(int i=pat.length();i<n;i++){
            if(z[i]>=ans && z[i]==n-i){
                cc=i;
                ans=z[i];
            }
        }
            if(ans>pat.length())
                ans=pat.length();
            if(ans>g.length())
                ans=g.length();
            
            string tt=g.substr(0,g.length()-ans) + pat;
            cout<<tt<<endl;
    
	}
	return 0;
}
