#include <bits/stdc++.h>
using namespace std;
#define n 1000000
int main()
{
	map<long long int,int>m;
	bool prime[n+1];
	memset(prime,true,sizeof(prime));
	for(long long int p=2;p*p<=n;p++)
	{
		if(prime[p]==true)
		{
			for(long long i=p*2;i<=n;i+=p)
			{
				prime[i]=false;
			}
		}
	}
	for(long long int p=2;p<=n;p++)
	{
		if(prime[p]==true)
		{
			m[p]=1;
			//cout<<p<<" ";
		//	for(long long int k=p*p;k<=n;)
		//	{
		//		m[k]=1;
		//		k*=k;
				//cout<<k<<" ";
		//	}
			
		}
	}
//	cout<<"out"<<endl;
	long long int max_mul=0,f=0,s=0,f1=-1,s1=-1;
	int test,flag=0;
	cin>>test;
	long long int num;
	while(test--)
	{
		cin>>num;f1=-1;s1=-1;
		flag=0;max_mul=0;
		for(auto it:m)
		{
			f=it.first;
			s=(num-it.first);
			if((f!=s) && (m.find(s)!=m.end()))
			{
				
				if((f*s)>max_mul)
				{
					max_mul=f*s;
					f1=f;
					s1=s;
					flag=1;
				}
			}
			if((it.first >= num))
			{
			
				break;
			}
		}
	
			cout<<f1<<" "<<s1<<endl;

	}
	return 0;
}
