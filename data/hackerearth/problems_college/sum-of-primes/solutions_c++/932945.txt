#include<bits/stdc++.h>
using namespace std;
#define MAX 1000001
long long int arr[MAX],temp[MAX];
long long int s=0;
void prime(long long int arr[],long long int i,long long int n)
{
  long long int a=2;
  while(a*i<=n)
  {
     arr[a*i]=1;
     a++;
   }
}
void sieve(long long int n){
 long long int i;
 memset(arr,0,sizeof(arr));
 memset(temp,0,sizeof(temp));
 arr[0]=1;
 arr[1]=1;
 for(i=2;i<=(n);i++)
 {
    if(arr[i]==0){
    temp[i]=temp[i-1]+i; 
    prime(arr,i,n);
    }
    else
    temp[i]=temp[i-1];
   }
}
int main()
{
  long long int t;
  scanf("%lld",&t);
  sieve(MAX);
  while(t--){
  long long int a,b,c,i,d,j,sum=0,flag=0;
  scanf("%lld %lld",&a,&b);
 // c=(long long int)(sqrt(b));
 // long long int k[b-a+1];
 // memset(k,0,sizeof(k));
  long long int ans=temp[b]-temp[a-1];
 // printf("%lld\n",ans);
  for(i=2;i<=sqrt(ans);i++){
  	  if(ans%i==0){ flag=1; break; }
  }
  if(flag==1 || ans==0) printf("NO\n");
  else printf("YES\n");
  s=0;
  } 
  return 0;
}
