#include<bits/stdc++.h>
typedef long long LL;

using namespace std;


LL b(LL n)
{
             LL low=0,mid=0;
             LL high=10000000;
             while(low<high)
             {
                           LL mid=(low+high)/2;
                            LL k=1LL*mid*mid+1LL*mid+1LL*2;
                         
                            if(k>=n)
                            {
                                    high=mid;
                            }
                            else
                            {
                                low=mid+1;
                            }
             }
             
             return low;
}
             
int main()
{

     int t;
     scanf("%d",&t);
     while(t--)
     {
               LL n;
               scanf("%lld",&n);
               LL ans=b(2*n);
               printf("%lld\n",ans);
     }
    
     return 0;
}
