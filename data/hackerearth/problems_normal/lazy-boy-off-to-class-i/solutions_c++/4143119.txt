#include <iostream>
#include<vector>
#include<algorithm>
#include<math.h>
using namespace std;
vector<int> v;
int main()
{
   // cout << "Hello World!" << endl;
   int t;
   cin>>t;
   while(t--){
   	v.clear();
   	int cnt=0,sum=0;
   	for(int i=0;i<7;i++){
   		int n,x,y;
   		cin>>n;
   		cnt=cnt+n;
   		for(int i=0;i<n;i++){
   			
   			cin>>x>>y;
   			int p=y-x;
   			v.push_back(p);
   			
   			
   		}
   	}
  
   	sort(v.begin(),v.end());
   	 for(int i=0;i<ceil(0.75*cnt);i++){
   		sum+=v[i];
   	}
   	//cout<<sum<<ceil(0.75*cnt);
   	int ans1;
   	ans1=604800-sum;
   	cout<<ans1<<endl;
   	
   }
    return 0;
}
