#include<bits/stdc++.h>
using namespace std;
int main(){
    int n; cin>>n;
    char a[10001];
    int sum=0;
    for(int i=0; i<n; i++){
        cin>>a;
        int l=strlen(a);
        for(int i=0; i<l; i++){
            sum = sum + a[i]-'0';
        }
        while(sum>9){
            int x=sum;
            sum=0;
            while(x>0){
                sum = sum + x%10;
                x/=10;
            }
        }
    }
    cout<<sum<<endl;
}
