#include <iostream>
using namespace std;
void merge(int a[],int l,int m,int r)
{
    int i,j=0,k=0;
    int n1=m-l+1;
    int n2=r-m;
    int *x=new int[n1];
    int *y=new int[n2];
    for(i=0;i<n1;i++)
        x[i]=a[l+i];
        for(i=0;i<n2;i++)
        y[i]=a[m+1+i];
    i=0;
    j=0;
    k=l;
    while((i<n1)&&(j<n2))
    {
        if(x[i]<=y[j])
        {
            a[k]=x[i];
            i++;
        }
        else
        {
            a[k]=y[j];
            j++;
        }
        k++;
    }
    while(i<n1)
    {
        a[k]=x[i];
        i++;
        k++;
    }
    while(j<n2)
    {
        a[k]=y[j];
        j++;
        k++;
    }
    delete(x);
    delete(y);

}
void mergesort(int a[],int l,int r)
{ 
    if(l<r)
    {
       int m=(l+r)/2;
        mergesort(a,l,m);
        mergesort(a,m+1,r);
        merge(a,l,m,r);
    }
}

int main()
{
	int n,i,c1=0,c2=0,temp;
	cin>>n;
	int *z=new int [n+1];
	int *v=new int [n+1];
	for(i=0;i<n;i++)
	{
		cin>>temp;
		if(temp%2==0)
		z[c1++]=temp;
		else
		v[c2++]=temp;
	}
    mergesort(z,0,c1-1);
    mergesort(v,0,c2-1);
    z[c1]=0;
    v[c2]=0;
    if(c1>0)
    { 
    for(i=0;i<=c1;i++)
    {
    cout<<z[i]<<" ";	
    z[c1]+=z[i];
    }
    }
    if(c2>0)
    {
    for(i=0;i<=c2;i++)
    {
    cout<<v[i]<<" ";
    v[c2]+=v[i];
    }
    }
    return 0;
}
