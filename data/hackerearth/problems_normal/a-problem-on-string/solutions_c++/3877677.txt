#include<cstdio>
#include<iostream>
#include<algorithm>
#include<vector>
#include<string>
using namespace std;
int main()
{
	int n;
	cin>>n;
	string s;
	cin>>s;
	int q;
	cin>>q;
	vector<vector<long long int> > v(26);
	for (int i = 0; i < s.length(); ++i)
	{
		v[s[i]-97].push_back(i);
	}
 
	while(q--){
		long long int ans=0;
		char x,y;
		cin>>x>>y;
		if (v[x-97].size()==0 || v[y-97].size()==0){
			cout<<0<<endl;
			continue;
		}
		if(x!=y)
		ans=v[x-97].size() * v[y-97].size();
		else
		ans=(v[x-97].size()*(v[x-97].size()-1))/2;
		cout<<ans<<endl;
		
	}
	return 0;
}
