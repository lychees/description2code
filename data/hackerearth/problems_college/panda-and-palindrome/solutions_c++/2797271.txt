#include<bits/stdc++.h>
using namespace std;
#define pb push_back
#define mod 1000000007
#define ll long long int

int M[1 << 15];

void solve()
{
    ll n;
    scanf("%lld",&n);
    string s;
    cin >> s;
    ll ans = 0;
    // counting even length palindromic strings
    M[0] ++;
    int mask = 0;
    for(int i=0; i<n; i++)
    {
        mask ^= ( 1 << ( s[i] - '0' ) );
        ans += M[mask];
        //cout<<"mask="<<mask<<"\n";
        M[mask] ++;
        //cout<<"mask "<<mask<<"m "<<M[mask]<<"ans "<<ans<<endl;
    }

    // counting palindromic strings of odd lengths
    M[0] = 0;
    mask = 0;
    for(int i=0; i<n; i++)
    {
        mask ^= ( 1 << ( s[i] - '0' ) );
        M[mask] = 0;
    }

    for(int i=0; i<10; i++)
    {
        M[0] ++;
        mask = 0;
        for(int j=0; j<n; j++)
        {
            mask ^= ( 1 << ( s[j] - '0' ) );
            if( ( mask & ( 1 << i ) ) == 0 )
            {
                M[mask] ++;
            }
        }
        mask = 0;
        for(int j=0; j<n; j++)
        {
            mask ^= ( 1 << ( s[j] - '0' ) );
            if( ( mask & ( 1 << i ) ) )
            {
                ans += M[ ( mask ^ ( 1 << i ) ) ];
            }
        }
        M[0] = 0;
        mask = 0;
        for(int j=0; j<n; j++)
        {
            mask ^= ( 1 << ( s[j] - '0' ) );
            if( ( mask & ( 1 << i) ) == 0 )
            {
                M[mask] = 0;
            }
        }
    }
    printf("%lld\n",ans);
}

int main()
{
    solve();
    return 0;
}
