#include <iostream>
#include<bits/stdc++.h>
using namespace std;
long gcd(long a,long b )
{
if(b==0)
return a;
else 
return gcd(b,a%b);
	
}
int main() 
{
	// your code goes here
	long n,i,j;
	cin>>n;
	int arr[n];
	for(i=0;i<n;i++)
	cin>>arr[i];
	long ans=0;
	for(i=0;i<n;i++)
	{
		for(j=i+1;j<n;j++)
		{
			if(arr[i]>arr[j]&&gcd(arr[i],arr[j])==1)
			ans++;
			
		}
		
	}
	cout<<ans;
	return 0;
}
