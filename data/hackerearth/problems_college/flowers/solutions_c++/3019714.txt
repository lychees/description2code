#include <bits/stdc++.h>
using namespace std;

typedef long long int ll;

int main()
{
    ll n,k;
    cin >> n >> k;
    ll a[n];
    for(ll i=0;i<n;i++)
   		cin >> a[i];
    sort(a,a+n,std::greater<int>());
    int ans=0;
    for(ll i=0;i<n;i++)
    	ans += (i/k+1)*a[i];
    cout << ans << endl;
    return 0;
}
