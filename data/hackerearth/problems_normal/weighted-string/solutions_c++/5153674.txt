#include <bits/stdc++.h>
using namespace std;

int main()
{
  int T;
  long K;
  cin >> T;
  while (T--)
  {
  	string s;
  	long ans = 0;
  	cin >> K >> s;
  	vector < long > temp (s.size() , 0 );
  	temp[0] = 1 + s[0] - 'a';
  	map < long , vector < int > > mymap;
  	mymap[temp[0]].push_back(0);
  	for(int i = 1; i < s.size(); i++)
  	{
  		
  		 temp[i] = 	1 + temp[i-1] + s[i]-'a';
  	  mymap[temp[i]].push_back(i);
  	}
  	
  	for(int i = 0; i < temp.size(); i++)
  	{
  		long  want = temp[i] + K;
  		if(temp[i] == K)
  		ans++;
  		if(mymap.find(want) == mymap.end())
  		continue;
  		vector < int > temp1 = mymap[want];
  		
  		for(int j = 0; j < temp1.size(); j++)
  		 if(temp1[j] >= i)
  		 ans++;
  		
  		
  		
  	}
  	
  	cout << ans  << endl;
  }
  
    return 0;
}
