for i in range(input()):
    p = raw_input()
    while not p:
        p = raw_input()
    a, b = p.split()
    b = int(b)
    a = filter(str.isupper, a)
    a = ''.join(sorted(a))
    if a:
        if b == 0:
            print a
        else:
            print a[::-1]
    else:
        print "NULL"
