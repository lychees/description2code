#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,a,b,n,x,y,i,j,k;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&x);
    	if((x<3)||(x==4))
    	printf("-1");
    	else
    	{
    		if(x==5)
    		printf("33333");
    		else if(x==3)
    		printf("555");
    		else
    		{
    			y=0;
    			a=x/3;
    			b=x%3;
    			while((a>0)||(b>0))
    			{
    				if((a>=1)&&(b%5==0))
    				{
    					b=b/5;
    					for(i=0;i<a;i++)
    					printf("555");
    					for(i=0;i<b;i++)
    					printf("33333");
    					y=1;
    					break;
    				}
    				else
    				{
    					a=a-1;
    					b=b+3;
    				}
    			}
    			if(y==0)
    			printf("-1");
    		}
    	}
    	printf("\n");
    }
    return 0;
}
