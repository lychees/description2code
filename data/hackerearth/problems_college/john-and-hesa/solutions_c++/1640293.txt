#include<iostream>
#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<math.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int x,p,cnt=0;
		cin>>x>>p;
		while(x)
		{
			if(x%p==0)
			{
				cnt++;
				x/=p;
			}
			else
				break;
		}
		if(x==1)
		x=0;
		cout<<cnt<<" "<<x<<endl;
	} 
}
