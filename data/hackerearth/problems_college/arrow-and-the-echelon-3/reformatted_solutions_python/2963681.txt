for i in range(int(raw_input())):
    x = int(raw_input())

    if x == 1 or x == 2:
        print(-1)
        continue

    n = x/3
    r = x % 3

    while n != 0:
        if r % 5 == 0:
            break
        else:
            n -= 1
            r += 3
    n1 = r/5
    st = ""
    for _ in range(n*3):
        st += '5'
    for _ in range(n1*5):
        st += '3'
    print st
