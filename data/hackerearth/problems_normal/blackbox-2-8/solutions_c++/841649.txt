#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<iostream>
using namespace std;
#define MAX 10000
void factorialof(int);
void multiply(int);
int length = 0;
int fact[MAX];
int sum[1001]={0};
int main(){
    sum[0]=1;
    fact[0]=1;
    for(int i=1;i<=1000;i++)
    {//cout<<endl;
    //	length=0;
    	factorialof(i);
     for(int j=length;j>=0;j--){
       sum[i]+=fact[j];
    }
    }
    int t,n1;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n1);
    	printf("%d\n",sum[n1]);
    }
    return 0;
}
void factorialof(int num){
    int i;
    multiply(num);
}
void multiply(int num){
    long i,r=0;
    int arr[MAX];
    for(i=0;i<=length;i++){
                arr[i]=fact[i];
        }

    for(i=0;i<=length;i++){
         fact[i] = (arr[i]*num + r)%10;
         r = (arr[i]*num + r)/10;
         //printf("%d ",r);
    }
    if(r!=0){
         while(r!=0){
             fact[i]=r%10;
             r= r/10;
             i++;
         }
    }
    length = i-1;   
}
