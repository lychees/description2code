#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;


vector<string> ans = {"0", "1", "2", "8", "68", "1504", "127792", "57140352", "258023200384", "10151395367145472", "3673835865235792306176", "13318668301694192513859649536", "531680718673514734573555796872790016"};

int main() {
	
	ios::sync_with_stdio(false);
	int test_num;
	cin >> test_num;
	
	while(test_num--) {
		
		int n, m;
		cin >> n >> m;
		
		long long res = 0;
		for(char ch : ans[n]) {
			res = (10ll * res + ch - '0') % m;
		}
		
		cout << res << "\n";
	}
	
	return 0;
}
