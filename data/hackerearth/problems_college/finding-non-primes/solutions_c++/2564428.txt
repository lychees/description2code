#include <bits/stdc++.h>
using namespace std;
bool isprime(int i)
{
	int flag=0;
	for(int j=2;j<=sqrt(i);j++)
	{
		if(i%j==0)
		{
			flag=1;
			break;
		}
	}
	if(flag)
	return false;
	else
	return true;
}
int main()
{
    int n;
    cin>>n;
    for(int i=2;i<=n;i++)
    {
    	if(!isprime(i))
    	cout<<i<<"\n";
    }
    return 0;
}
