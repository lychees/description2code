#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define S(x) scanf("%lld", &x)

int main()
{
	ll N;
	S(N);
	ll a[10078] = {0};
	ll q;
	S(q);
	ll i;
	ll vis[100003] = {0};
	for(i = 0;i < q;i++) {
		ll aa,b;
		S(aa);
		S(b);
		ll mi,ma;
		mi = min(aa, b);
		ma = max(aa,b);
		aa= mi;
		b=  ma;
		ll j;
		ll d[2000] = {0};
		if(vis[aa] == 1 || vis[b] == 1|| aa == b) {
			cout << "NO\n";
			continue;
		}
		for(j = aa + 1;j < b;j++) {
			d[a[j]]++;
	//		cout << a[j] << " ";
		}
		ll flag = 0;
		for(j = 1;j <= i  ;j++) {
			if(d[j] % 2 != 0) {
				cout << "NO\n";
				flag = 1;
				break;
			}
		}
		if(flag == 0) {
			cout<< "YES\n";
			a[aa] = i + 1;
			a[b] = i + 1;
			vis[aa] = 1;
			vis[b] = 1; 
		}
	}
}
