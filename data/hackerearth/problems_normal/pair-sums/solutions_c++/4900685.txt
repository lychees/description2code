#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	int n,k;
	cin >> n >> k;
	int ar[n];
	for(int i=0;i<n;i++)cin >> ar[i];
	sort(ar,ar+n);
	int i=0,j=n-1;
	int flag=0;
	while(i<j)
	{
		if(ar[i]+ar[j]==k)
		{
			flag=1;
			break;
		}
		else if(ar[i]+ar[j]<k)i++;
		else j--;
	}
	if(flag)cout<<"YES";
	else cout<<"NO";
	return 0;
}
