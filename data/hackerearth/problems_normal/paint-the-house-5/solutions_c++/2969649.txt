using namespace std;
typedef long long ll;

pair<int, pair<int, int>> w[100005];

int main() {
  int n;
  long long d;
  scanf("%d %lld", &n, &d);
  for (int i = 0; i < n; i++) {
    scanf("%d %d %d", &w[i].first, &w[i].second.first, &w[i].second.second);
  }
  sort(w, w + n);

  ll ans = 0;
  ll s = 0;
  int cur = 0;
  int i = 0, t = 0;
  while (i < n) {
    s += 1ll * (w[i].first - t) * cur;
    if (s >= d) {
      break;
    }

    t = w[i].first;

    int j = i + 1;
    int mnc = w[i].second.first, mxv = w[i].second.second;
    while (j < n && w[j].first == t) {
      const int c = w[j].second.first;
      const int v = w[j].second.second;
      if (v > mxv || (v == mxv && c < mnc)) {
        mxv = v;
        mnc = c;
      }
      j++;
    }

    if (mxv > cur) {
      cur = mxv;
      ans += mnc;
    }

    i = j;
  }

  printf("%lld\n", ans);

  return 0;
}
