
#include<bits/stdc++.h>
using namespace std;

int main()
{
	
	long int n;
	cin>>n;
	
	long int arr[n];
	for(int i=0;i<n;i++)
	{cin>>arr[i];}
	long int t;
	cin>>t;
	int flag=0;
	long long int div=1,tmp;
	for(int i=0;i<t;i++)
	{
		cin>>tmp;
		div*=tmp;
		if(div>1000000)
		flag=1;
	}
	
	
		for(int i=0;i<n;i++)
		{
			if (flag==1)
			arr[i]=0;
			else
			arr[i]/=div;
		}
	
	for(int i=0;i<n;i++)
		{
			cout<<arr[i]<<" ";
		}
	
	
return 0;	
}
