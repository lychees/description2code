#include <iostream>
using namespace std;
long long *arr1 = new long long[1000001];
	long long *arr2 = new long long[1000001];
int main()
{
	ios_base::sync_with_stdio(false);
	int n, m, i, j;
	long long k;
	cin >> n;
	
	arr2[0] = 0;
	for (i = 0; i<n; i++)
	{
		cin >> arr1[i];
		arr2[i+1] = 0;
	}
	cin >> m;
	for (int l = 0; l<m; l++)
	{
		cin >> i >> j >> k;
		arr2[i - 1] += k;
		arr2[j] -= k;
	}
	k = 0;
	for (i = 0; i<n; i++)
	{
		k += arr2[i];
		printf("%lld\n", arr1[i] + k);
	}
	delete[] arr1;
	delete[] arr2;
	return 0;
}
