#include <bits/stdc++.h>

#define REP(i,n) for(int i=0;i<(int)n;i++)
#define FOR(i,a,b) for(int i=(int)a;i<=(int)b;i++)
#define pb push_back
#define mp make_pair

using namespace std;

typedef long long int ll;
ll arr[100005] = {0};
int m,n;
bool bisa(ll T){
	int j = 0;
	ll val;
	REP(i,m){
		val = 0;
		while(val<=T && j<n){
			val+=arr[j];
			//printf("val: %lld, j: %d\n",val,j);
			j++;
		}
		if(val>T) j--;
		//printf("#######\n");
	}
	if(val<=T) return true;
	return false;
}

int main()
{
	scanf("%d%d",&n,&m);
	
	REP(i,n)
		scanf("%lld",&arr[i]);
	//sort(arr,arr+n);
	/*

	*/
	
	ll tMin = 0;
	ll tMax = 1000000000000000000;
	//ll tMax = 5;
	ll ans = -1;
	while(tMax>=tMin){
		ll mid = (tMax+tMin)/2;
		//printf("mid : %lld\n",mid);
		if(bisa(mid)){
			tMax = mid-1;
			ans = mid;
		}else tMin = mid+1;
		//printf("-----------\n");
	}
	printf("%lld\n",ans);
    return 0;
}
