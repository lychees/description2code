z=long(input())
while z>0:
  m=long(input())
  suma=long(0)
  while m>0:
    d,l=map(long,raw_input().split())
    suma=(d*l)+suma;
    m=m-1
  liczba=suma
  while liczba>=10:
    suma=long(0)
    while liczba>0:
      pom=long(liczba%10)
      suma=suma+pom
      liczba=liczba/10
    liczba=suma
  z=z-1
  print(liczba)
