
#include <iostream>
#include<string.h>
#include<stdio.h>
#include<ctype.h>
	
using namespace std;

int main()
{
string s;
cin>>s;
s[0]=toupper(s[0]);
int n=0;
while(s.find("_",n)!=std::string::npos)
{
if(s.find("_",n)!=std::string::npos)
{
n=s.find("_",n)+1;
s[n]=toupper(s[n]);
}
}
cout<<s;
return 0;
}
