import sys

t = input()
l = []
while t > 0:
    s = sys.stdin.readline().strip().split("-")
    a = []
    temp = s[0].split(":")
    a.append(int(temp[0]+temp[1]))
    temp = s[1].split(":")
    a.append(int(temp[0]+temp[1]))
    l.append(a)
    t = t-1


l.sort(key=lambda x: x[1])

    if l[i][0] < l[i-1][1]:
