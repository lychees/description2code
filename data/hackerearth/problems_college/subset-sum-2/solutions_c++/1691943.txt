#include <bits/stdc++.h>
#define mod 1000000007
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long t;
    cin>>t;
    while(t--)
    {
    	long long n,x,s=0,i;
    	cin>>n;
    	x=pow(2,n-1);
      for(i=1;i<=n;i++)
      {
      	s+=i*x;
      	s=s%mod;
      }
      cout<<s<<endl;
    }
    return 0;
}
