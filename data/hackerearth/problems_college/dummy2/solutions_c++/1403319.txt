#include<cstdio>
#include<cstring>
#include<iostream>

inline int Read()
{
	register int c=getchar();
	int x=0;
	for(;(c<48 || c>57);c=getchar());
	for(;c>47 && c<58;c=getchar())
		x=(x<<1)+(x<<3)+c-48;
	return x;
}
using namespace std;
int main()
{
	int t=Read();
	while(t--)
	{
		string a;
		cin>>a;
		string ans;
		int len=a.length();
		for(int i=0;i<len;i++)
		{
			if(a[i]>='0' && a[i]<='9')
			{
				string tmp=ans;
				for(int j=0;j<a[i]-'0'-1;j++)
					ans+=tmp;
			}
			else
				ans+=a[i];
		}
		cout<<ans<<endl;
	}
}
