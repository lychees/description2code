#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define MOD 1000000007
ll sol[1010000][10];
ll fun(ll i,ll a)
{
    //cout<<"called for i="<<i<<"a="<<a<<"\n";
    if(i==0&&a==0)
    {
        sol[i][a]=0;
        return 0;
    }
    if(i==0)
    {
        sol[i][a]=1;
        return 1;
    }
    if(sol[i][a]!=0)return sol[i][a];
    ll ans=0;
    if(a<9)
    {
        ans=fun(i-1,a+1);
    }
    if(a>0)
    {
        ans=(ans+fun(i-1,a-1))%MOD;
    }
    ans=ans%MOD;
    sol[i][a]=ans;
    //cout<<"solution for i="<<i<<"a="<<a<<sol[i][a]<<"\n";
    return sol[i][a];
}
int main()
{
    ll t,i,j,n;
    for(i=0;i<1010000;i++)
    {
        for(j=0;j<10;j++)
            sol[i][j]=0;
    }/*
    for(i=1;i<=9;i++)
    {
        fun(1000,i);
    }*/
    for(i=0;i<1010000;i++)
    {
        for(j=0;j<=9;j++)
        {
            if(i==0&&j==0)
                sol[i][j]=0;
            else if(i==0)
                sol[i][j]=1;
            else
            {
                ll x=0,y=0;
                if(j>0)y=sol[i-1][j-1];
                if(j<9)x=sol[i-1][j+1];
                sol[i][j]=(x+y)%MOD;
            }
        }
    }
    cin>>t;
    while(t--)
    {
        cin>>n;n--;
        assert(n<1010000);
        ll ans=0;
        if(n<0)
        {
            cout<<"0\n";
            continue;
        }
        if(n==0)
        {
            cout<<"10\n";
            continue;
        }
        for(i=0;i<=9;i++)
            ans=(ans+sol[n][i])%MOD;
        cout<<ans<<"\n";
    }
    return 0;
}
