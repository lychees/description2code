#include <iostream>
#include<cstring>
using namespace std;

int main()
{
   int t;
   cin>>t;
   
   while(t--)
   {
   	char *a= new char[100000];
   	//char a[1000];
   	cin>>a;
   	int count=0;
   	for(int i=0;i<strlen(a);i++)
   	{
   		if(a[i]==a[i+1])
   		{
   			count++;
   		}
   	}
   	cout<<count<<endl;
   	}
    return 0;
}
