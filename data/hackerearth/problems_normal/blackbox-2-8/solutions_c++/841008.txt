void multiply(int n)
{
	long i,r=0;
	for(i=0;i<=len;i++)
		{
			arr[i]=fact[i];
		}

	for(i=0;i<=len;i++)
		{
			fact[i] = (arr[i]*n + r)%10;
			r = (arr[i]*n + r)/10;
		}
	if(r!=0)
	while(r!=0)
		{	
			fact[i]=r%10;
			r= r/10;
			i++;
		}
	len = i-1;
}

void factorial(int n)
{
	int i;
	for(i=2;i<=n;i++)
		{
			multiply(i);
		}
}
