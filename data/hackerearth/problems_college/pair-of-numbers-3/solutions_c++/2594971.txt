#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {  int n;
    	cin>>n;
    	int c=0;
    	int i,j,pro,sum;
    	for(i=1;i<n;i++)
    	{
    		for(j=i+1;j<=n;j++)
    		 {
    		 	pro=i*j;
    		    sum=i+j;
    		 if(pro%sum==0)
    		   c++;
    		 }
    	}
    	cout<<c<<endl;
    }
    return 0;
}
