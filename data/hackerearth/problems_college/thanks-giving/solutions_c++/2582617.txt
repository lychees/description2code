#include<iostream>
using namespace std;
int main()
{
	char str[4002];
	int i,a[27]={0},max=0,j;
	cin>>str;
	for(j=0;str[j]!='\0';j++)
	{
		a[str[j]-98]++;
	}
	for(j=0;j<27;j++)
	{
		if(a[j]>max) 
		{
			max=a[j];
			i=j;	
		}
	}
	if((i+98)=='h' || (i+98)=='e' || (i+98)=='a' || (i+98)=='d' || (i+98)=='r' || (i+98)=='s')
		cout<<"Thanks head rush\n";
	else if((i+98)=='c' || (i+98)=='b' || (i+98)=='i' || (i+98)=='n' || (i+98)=='g' || (i+98)=='l')
		cout<<"Thanks cubing club\n";
	else if((i+98)=='u')
			cout<<"Thanks both\n";
	else cout<<"none\n";
	return 0;
}
