#include<bits/stdc++.h>
using namespace std;
int a[505];
int v[105][505];
int main() {
  memset(v,0,sizeof(v));
  memset(a,0,sizeof(a));
  int n;
  scanf("%d",&n);
  for(int i = 0 ; i < n ;i++) {
    int m;
    scanf("%d",&m);
    for(int j = 0 ; j < m ;j++) {
      int x; scanf("%d",&x);
      v[i][x] = 1;
      a[x]++;
    }
  }
  long long sum1 = 0;
  long long sum2 = 0;
    for(int j = 0 ; j <= 500 ;j++) {
      if(a[j]) sum1 += j;
    }
    for(int i = 0 ; i <= 500 ; i++) {
      for(int j = 0 ; j < n ; j++) {
        if(!v[j][i] && a[i]) {
          sum2 += i;
          break;
        }
      }
    }
    cout<<(sum2-sum1)<<endl;
}
