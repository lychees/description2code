#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int N, count;
    scanf("%d", &N);
    //cin>>N;
//    cout<<N<<endl;
    unsigned long long num;
    unsigned int max = (int)(3 * pow(10, 8));
//    cout<<max;
    count = 0;
    for(int n = 0; n < N; n++)	{
    	scanf("%llu", &num);
//    	cin>>num;
//		cout<<num<<endl;
    	if(num < max)
    		count++;
    }
    cout<<count;
    return 0;
}
