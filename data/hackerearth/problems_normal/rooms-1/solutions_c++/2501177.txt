#include <iostream>
#include<algorithm>
using namespace std;
#define ll long long int
int main()
{
    ll t;
    cin>>t;
    while(t--)
    {
    	ll n,i,j,count=0,en,curr_count=0;
    	cin>>n;
    	ll a[n];
    	ll e[n];
    	
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	
    	
    	for(i=0;i<n;i++)
    	{
    		cin>>j;
    		e[i]=a[i]+j;
    	}
    	
    	
    	vector<pair<ll,bool> > v;
    	
    	for(i=0;i<n;i++)
    	{
    		v.push_back(make_pair(a[i],true));
    	}
    	
    	for(j=0;j<n;j++)
    	{
    		v.push_back(make_pair(e[j],false));
    	}
    	
    	sort(v.begin(),v.end());
    	
    	for(i=0;i<2*n;i++)
    	{
    		if(v[i].second==true)
    		curr_count++;
    		else
    		curr_count--;
    		
    		if(count<curr_count)
    		count=curr_count;
    	}
    	
    	cout<<count<<endl;
    }
    	
    	
    	
    
   
    return 0;
}
