#include<bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		char str[100];
		cin>>str;
		char ans[1000000];
		ans[0]=NULL;
		for(int i=0;i<strlen(str);i++)
		{
			if(str[i]>='1' && str[i] <='9')
			{
				int k=str[i]-'0';
				char str1[1000000];
				strcpy(str1,ans);
				for(int i=0;i<k-1;i++)
					strcat(ans,str1);
			}
			else
			{
				int s=strlen(ans);
				ans[s] = str[i];
				ans[s+1] = NULL;
			}
		}
		cout<<ans<<endl;
	}
	return 0;	
}
