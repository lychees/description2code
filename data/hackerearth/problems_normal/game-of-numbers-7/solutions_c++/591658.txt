#include <iostream>
#include <cmath>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    long long x, y;
    
    cin>>t;
    while(t--)
    {
    	cin>>x>>y;
    	vector<bool> arr(y-x+1, true);
    	
    	for(long long k=2; k*k<=y; k++)
    	{
    		const long long gap = k*k;
    		const long long base = x + (gap - x % gap) % gap;
    		for(long long i = base; i<=y; i+=gap)
    			 arr[(int)(i - x)] = false;
    	}
        cout << (count(arr.begin(), arr.end(), true)) << endl;
    }
    return 0;
}
