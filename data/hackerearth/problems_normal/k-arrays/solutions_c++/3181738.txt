//mod power
ll modPow(ll base, ll exponent, int modulus)
{
    ll result = 1;
    while (exponent > 0)
    {
        if (exponent % 2 == 1)
            result = (result * base) % modulus;
        exponent = exponent >> 1;
        base = (base * base) % modulus;
    }
    return result;
}
//power set
void powerSet(ll a[],ll n){
	for(ll i=0;i<(1<<n);i++){
		for(ll j=0;j<n;j++){
			if((1<<j)&i){
				cout<<a[j]<<" ";
			}
		}
		cout<<endl;
	}
}
