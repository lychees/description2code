#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,a,b,c,d,s;
    scanf("%d",&t);
    while(t--){
    scanf("%d%d%d%d%d",&a,&b,&c,&d,&s);
    int n=c-s;
    if(n<a)
        printf("STAY IN LONDON\n");
    else if(n>b)
        printf("%d\n",b);
    else if(n>=a&&n<=b)
        printf("%d\n",n);}
    return 0;
}
