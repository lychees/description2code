import math
t = input()
rem = 0
rev = 0
A = []
if t >= 1 and t <= 10000:
    for i in range(0, t):
        rem = 0
        rev = 0
        n = input()
        if n >= 1 and n <= math.pow(10, 100):
            while (n != 0):
                rem = n % 10
                rev = rev*10+rem
                n = n/10
            if rev % 2 == 0:
                A.append("EVEN")
            else:
                A.append("ODD")
    for i in range(0, t):
        print(A[i])
