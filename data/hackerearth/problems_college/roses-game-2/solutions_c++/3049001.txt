#include<iostream>
using namespace std;
typedef  long long int ll;
int main()
{
    ll t;
    cin>>t;
    while(t--){
        ll n;
        cin>>n;
        ll arr[n];
        ll brr[n];
        for(ll i=0;i<n;i++){
            cin>>arr[i];
            brr[i]=0;
        }
        ll i=0;
        while(i<n){
            if(arr[i]==-1){
                cout<<"Win";
                break;
            }
            else if(brr[i]!=0){
                cout<<"Lose";
                break;
            }
            else {
                brr[i]=1;
                i=arr[i];
            }
        }
        cout<<endl;

    }
    return 0;
}
