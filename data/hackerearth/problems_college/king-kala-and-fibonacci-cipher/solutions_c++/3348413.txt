struct Node
{
    ll A[26]={};
};
typedef struct Node Node;
Node tree[1000010] = {};
Node tmp;
ll fibo[MAX];


/**
 * Build and init tree
 */
Node build_tree(ll node,ll a,ll b) {
    if(a > b) return tmp; // Out of range

      if(a == b) { // Leaf node
      	   // cout<<"---"<<a<<" "<<fibo[a]<<endl;
           tree[node].A[fibo[a]]++;
           return tree[node];
    }

    Node x = build_tree(node*2+1, a, (a+b)/2); // Init left child
    Node y = build_tree(node*2+2, 1+(a+b)/2, b); // Init right child
    for(ll i=0;i<26;i++) tree[node].A[i]=(x.A[i]+y.A[i]);
    // cout<<a<<" -- "<<b<<endl;
    // for(ll i=0;i<26;i++) cout<<tree[node].A[i]<<" ";
    // cout<<endl;
    return tree[node]; // Init root value
}

Node query(ll node,ll a,ll b,ll i,ll j) {
    Node sum;
    if (b<i || j<a) return tmp;
    if (i<=a && b<=j) return tree[node];
    Node x=query(node*2+1,a,(a+b)/2,i,j);
    Node y=query(node*2+2,1+(a+b)/2,b,i,j);
    Node ans;
    for(ll i=0;i<26;i++) ans.A[i]=(x.A[i]+y.A[i]);
    return ans; // Init root value
}

bool compare(pair<ll,ll> a,pair<ll,ll> b)
{
	if(a.X==b.X) return a.Y<b.Y;
	else return a.X>b.X;
}
