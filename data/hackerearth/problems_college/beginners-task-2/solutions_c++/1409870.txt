#include<bits/stdc++.h>
#define ld long double
using namespace std;

int main()
{
    int t,n;

    cin>>t;
    while(t--)
    {
        cin>>n;
        ld num[n],sum=0;

        int k=0;
        while(k<n)
        {
           cin>>num[k];
        /*   if(num[k]>n)
               num[k]=num[k]/n;*/
            sum+=num[k++]/n;
        }
        cout<<(long long int)sum<<endl;
    }
  return 0;
}
