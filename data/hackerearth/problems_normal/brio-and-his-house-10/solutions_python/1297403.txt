#print 'Hello World!'
from math import sqrt
def readStdInInt(): return int(raw_input())
def readStdInIntArray(): return [int(x) for x in raw_input().split()]
def readStdInFloatArray(): return [float(x) for x in raw_input().split()]
def writeStdOutInt(N): print N
def writeStdOutIntArray(A): print ' '.join([str(x) for x in A])

T = readStdInInt()
for t in range(T):
	A,B,C = readStdInFloatArray()
	#print A,B,C
	s = (A+B+C)/2.0
	Area = sqrt(s*(s-A)*(s-B)*(s-C))
	#print Area
	d = (A*B*C)/(2*Area)
	r = d/2.0
	Plot = 3.1415*r*r
	print "%.4f"%Plot
