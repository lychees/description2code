T = int(raw_input())
for a in range(T):
    s = raw_input()
    S = sorted(s)
    count = 1
    ans = ''
    get = ''
    if S.count(S[0]) == len(S):
        for q in S:
            get += q
        print get
    else:
        while count < len(S):
            for i in range(len(S)):
                if S.count(S[i]) == count:
                    ans += S[i]
            count += 1
        print ans
