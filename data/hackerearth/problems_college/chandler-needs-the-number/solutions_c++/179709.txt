#include <stdio.h>
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
main()
{
long long int i,n=0,a=1,f=0;
vector<long long int> v;
scanf("%d",&n);
a=1;
for(i=0;i<n;i++)
{
scanf("%ld",&f);
v.push_back(f);
}

}
