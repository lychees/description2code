#include <iostream>
#include <stdio.h>
#include <string.h>
#define ll long long int
using namespace std;
// Sieve of Eratosthenes
const int N = 1000000;
bool p[N+1];
ll ans[N+1];
void sieve(){
    memset(p,true,sizeof(p));
    p[0] = p[1] = false;
	int m;
    for(int i=2; i<N+1; ++i){
        if(p[i]){ // if prime then mark multiples of that no
			m = i+i;
			while(m <= N){
				p[m] = false;
				m += i;
			}
        }
    }

    memset(ans,0,sizeof(ans));
    for(int i=2; i<N+1; ++i){
        if(p[i]){
            ans[i] = ans[i-1] + i;
        }else{
            ans[i] = ans[i-1];
        }
    }
}

int main() {
    sieve();
    int t,l,r;
    scanf("%d",&t);
    while(t--){
        scanf("%d %d",&l, &r);
        printf("%lld\n",ans[r]-ans[l-1]);
    }
    return 0;
}
