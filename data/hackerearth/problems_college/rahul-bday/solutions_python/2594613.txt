t = int(raw_input())
while(t>0):
	b,w = [int(i) for i in raw_input().split()]
	x,y,z = [int(i) for i in raw_input().split()]
	if((x<=y and y<=(x+z)) or (y<=x and x<=(y+z))):
		print x*b + y*w
	elif(x<y and (x+z)<y):
		print x*b + (x+z)*w
	elif(y<x and (y+z)<x):
		print (y+z)*b + y*w
	t = t-1
