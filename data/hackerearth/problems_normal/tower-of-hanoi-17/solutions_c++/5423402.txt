#include<bits/stdc++.h>
using namespace std;
struct node
{
    int r;
    int h;
};
bool mycomp(node i,node j)
{
        return (i.r>j.r);
}
int main()
{
    int t,n;
    node a[205];
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(int i=0;i<n;i++)
            cin>>a[i].r>>a[i].h;
        sort(a,a+n,mycomp);
        vector<long long int> dp(n);
        for(int i=0;i<n;i++)
            dp[i]=a[i].h;
        long long int maxx=INT_MIN;
        maxx=max(maxx,dp[0]);
        for(int i=1;i<n;i++)
        {
            for(int j=0;j<i;j++)
            {
                if((a[i].h<a[j].h)&&(a[i].r<a[j].r)){
                    dp[i]=max(dp[i],dp[j]+a[i].h);
                }
            }
            maxx=max(dp[i],maxx);
        }
        cout<<maxx<<endl;
    }
    return 0;
}
