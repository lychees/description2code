#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    string str;
    cin>>str;
    long long i,cont,ans=0;
    for(i=0;i<str.size();i++)
    {
    	cont=1;
    	while(str[i]==str[i+1])
    	{
    		i++;
    		cont++;
    	}
    	if(cont%2==0)
    	ans++;
    }
    cout<<ans<<endl;
    return 0;
}
