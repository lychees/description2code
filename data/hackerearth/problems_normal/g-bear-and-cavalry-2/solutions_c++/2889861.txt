#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t,i,n,w[43],h[43],q;
    long int max;
    
    cin>>t;
    
    while(t--){
    	cin>>n;
    	
    	for(i=0;i<n;i++) cin>>w[i];
    	for(i=0;i<n;i++) cin>>h[i];
    	
    	sort(w+1,w+n);
    	sort(h,h+n);
    	
    	q=0;
    	max=h[n-1]*w[0];
    	for(i=1;i<n;i++){
    		if(max<=w[i]*h[n-1-i]) {
    			q=1;
    			break;
    		}
    	}
    	if(q==0) cout<<"YES"<<endl;
    	else cout<<"NO"<<endl;
    	
    }
    
    return 0;
}
