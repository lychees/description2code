n, m = map(int, raw_input().split())


def add(x, y):
    return x+y


colsum = map(int, raw_input().split())
rowsum = [sum(colsum)]
n = n-1

while n:
    n -= 1
    r = map(int, raw_input().split())
    rowsum.append(sum(r))
    colsum = map(add, colsum, r)

res = 0
for i in rowsum:
    for j in colsum:
        if res < (i ^ j):
            res = (i ^ j)
print res
