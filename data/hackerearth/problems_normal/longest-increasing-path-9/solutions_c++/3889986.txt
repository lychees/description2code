#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n,m;
    	scanf("%d %d",&n,&m);
    	int arr[n][m];
    	int dp[n][m];
    	for(int i=0;i<n;i++)
    	{
    		for(int j=0;j<m;j++)
    		scanf("%d",&arr[i][j]);
    	}
    	int ma=0;
    	for(int i=0;i<n;i++)
    	{
    		for(int j=0;j<m;j++)
    		{
    			dp[i][j]=0;
    			if(i==0 && j==0)
    			dp[i][j]=1;
    			if(i-1>=0 && arr[i-1][j]<arr[i][j] && dp[i-1][j]>0)
    			dp[i][j]=max(dp[i][j],1+dp[i-1][j]);
    			if(j-1>=0 && arr[i][j-1]<arr[i][j] && dp[i][j-1]>0)
    			dp[i][j]=max(dp[i][j],1+dp[i][j-1]);
    			
    			
    			ma=max(ma,dp[i][j]);
    			
    		}
    	}
    	cout<<ma<<endl;
    	
    }
    return 0;
}
