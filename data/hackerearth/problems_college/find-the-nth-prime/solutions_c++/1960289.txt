#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int n,i,count,j,k=0,c;
    int a[1000];
    cin>>n;
    for(i=2;count<=1000;i++)
    {
    	c=0;
    	for(j=2;j<=i;j++)
    	{
    		if(i%j==0)
    			c++;
    	}
    	if(c==1)
    	{
    		count++;
    		a[k]=i;
    		k++;
    	}
    }
    cout<<a[n-1]<<endl;
    return 0;
}
