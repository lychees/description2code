#include<iostream>
using namespace std;
#define ll long long
int main()
{
    ll n;
    cin>>n;
    string s;
    while(n--)
    {
        ll x=0,i;
        cin>>s;
        for(i=0;i<s.length();i++)
        {
            if(s[i]=='_') continue;
            else
            {
              x+=(s[i]-'0');
              x%=9;
            }
        }
        if(x==0)
            cout<<"0 or 9"<<endl;
        else
            cout<<9-x<<endl;
    }
}
