#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstdlib>
using namespace std;
int n;
double a, d, ans;
double work(double v) {
  double t = v / a;
  double l = 0.5 * a * t * t;
  if (l < d)
    return (d - l) / v + t;
  else
    return sqrt(2 * d / a);
}
int main() {
  scanf("%d%lf%lf", &n, &a, &d);
  for (int i = 1; i <= n; i++) {
    double t, v;
    scanf("%lf%lf", &t, &v);
    ans = max(ans, work(v) + t);
    printf("%.8lf\n", ans);
  }
}
