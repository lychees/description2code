#include <iostream>
using namespace std;
#include<math.h>
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int k;
		cin>>k;
		int ans=0;
		int p=sqrt(k);
		for(int i=1;i<=p;i++)
		{
			if(k%i==0)
			 {
			 	if(i%2==0)
			  ans++;
			 if((k/i)%2==0)
			 ans++;
			 }
		}
		if(p*p==k&&p%2==0)
		ans--;
		cout<<ans<<endl;
		
	}
}
