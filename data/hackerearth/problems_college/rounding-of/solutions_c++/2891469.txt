#include <cstdio>
#include <iostream>
#define DEBUG if(0)

using namespace std;

int main()
{
    int T,N,R;
    cin >> T;
    while(T--)
    {
        cin >> N >> R;
        if( N % R == 0 )
            cout << N << endl;
        else 
        {
            int rem = N % R;
            if( R%2 == 0 )
            {
                if( rem >= R/2)
                {
                    cout << N +( R - rem ) << endl;
                }
                else
                {
                    cout << N - rem << endl;
                }
            }
            else
            {
                if( rem >= R/2 + 1)
                {
                    cout << N +( R - rem ) << endl;
                }
                else
                {
                    cout << N - rem << endl;
                }
            }
        }
    }
    return 0;
}
