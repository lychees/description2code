#include<bits/stdc++.h>

using namespace std;

int main()
{
    int i,N,t,x,a[106];

    cin>>t;

    while(t--)
    {
        cin>>N;

    for(i=1;i<=N;i++)
        cin>>a[i];
    int ans=0;
    for(i=1;i<=N;i++)
        cin>>x,ans = ans^(a[i]-x),ans=ans^(x-1);
    ans?cout<<"First\n":cout<<"Second\n";
    }


    return 0;
}
