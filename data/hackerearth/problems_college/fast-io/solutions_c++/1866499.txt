#include<iostream>
using namespace std;
#include<cstdio>
#define ll long long
#define gc getchar_unlocked
void scanint(ll &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}
int main()
{
	ll t,n;
	ll count=0;
	scanint(t);
	while(t--)
	{
		scanint(n);
		if(n<300000000)
			count++;
	}
	printf("%lld\n", count);
	return 0;
}
