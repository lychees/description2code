#include <iostream>
using namespace std;

int find_min(long val){
	int count=0;

		while(val){
			if(val & 1 == 1)
				count++;
			val = val>>1;
		}

	return count;
}
int main()
{
	long input;
	long value;
	cin>>input;
    for(int i = 0 ; i < input ; ++i){
    	
    	cin>>value;
    	cout << find_min(value)<<endl;
    }
    return 0;
}
