#include <stdio.h>
#include <bits/stdc++.h>
#define i64 unsigned long long
using namespace std;
i64 dp[101][101]={0};
i64 nCr(int n, int k)
{
    int i, j;
    for (i = 0; i <= n; i++)
    {
        for (j = 0; j <= min(i, k); j++)
        {
            if (j == 0 || j == i)
                dp[i][j] = 1;
            else
                dp[i][j] = dp[i-1][j-1] + dp[i-1][j];
        }
    }
    return dp[n][k];
}

int main()
{
    int n, r;
    int t=0;
    #ifndef ONLINE_JUDGE
    freopen("input.txt","r",stdin);
    #endif // ONLINE_JUDGE
    cin>>t;
    while(t--)
    {
        cin>>n>>r;
        printf("%llu\n",nCr(n+r-1,n-1));
    }
    return 0;
}
