#include <bits/stdc++.h>
using namespace std;

const int T = 1e5 + 100;
const int N = 1e3 + 100;

int v[T];
int mx[T];
pair<int, int> a[N];

int main()
{
	ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
	int n;
	cin >> n;
	while (n--) {
		int ti;
		cin >> ti;
		for (int i = 1; i*ti < T; ++i) {
			++v[i*ti];
		}
	}
	for (int i = 1; i < T; ++i) {
		mx[i] = max(mx[i-1], v[i]);
	}
	for (int i = 1; i < T; ++i) {
		v[i] = v[i-1] + (v[i] > 0);
	}
	int p;
	cin >> p;
	for (int i = 1; i <= p; ++i) {
		cin >> a[i].first >> a[i].second;
	}
	int q;
	cin >> q;
	while (q--) {
		int b, t;
		cin >> b >> t;
		if (a[b].first < v[t] or a[b].second < mx[t]) {
			cout << "Awake\n";
		} else {
			cout << "Asleep\n";
		}
	}
    return 0;
}
