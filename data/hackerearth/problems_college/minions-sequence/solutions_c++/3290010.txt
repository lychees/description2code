int func(int a, int b) {
int x = 1, y = a;
    while(b > 0) {
        if(b%2 == 1) {
            x=(x*y);
            if(x>5000) x%=5000;
        }
        y = (y*y);
        if(y>5000) y%=5000;
        b /= 2;
    }
    return x;
}
