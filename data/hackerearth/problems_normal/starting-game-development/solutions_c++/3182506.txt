#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,m,q,a,i,j;
    cin>>n>>m>>q;
    vector< vector<int> >vec;
    vec.resize(n+2);
    vec.clear();
    for(i=0;i<n;i++) {
    	for(j=0;j<m;j++) {
    		cin>>a;
    		vec[i].push_back(a);
    	}
    }
    while(q--) {
    	int ans=INT_MAX;
    	for(i=0;i<n;i++) {
    		cin>>a;
    		j=upper_bound(vec[i].begin(),vec[i].end(),a)-vec[i].begin();
    		ans=min(ans,j);
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
