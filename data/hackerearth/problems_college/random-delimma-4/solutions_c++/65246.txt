#define trace6(a, b, c, d, e, f) cerr << #a << ": " << a << " | " << #b << ": " << b << " | " << #c << ": " << c << " | " << #d << ": " << d << " | " << #e << ": " << e << " | " << #f << ": " << f << endl;

using namespace std;
int main(){
    int T,n,k;
    scanf("%d",&T);
    while(T--){
        scanf("%d%d",&k,&n);
        if(k>=n) printf("%.9lf\n",1.0);
        else{
            double ans = 1.0;
            for(int i=0;i<k;i++) ans *= (1-(double)i/n);
            printf("%.9lf\n",1-ans);
        }
    }
    return 0;
}
