no = raw_input()
no = int(no)
nos = []
chainsC = []

noSt = raw_input()
noSt = noSt.split(" ")

for i in range(0, len(noSt)):
    nos.append(int(noSt[i]))
nos.sort()

for i in range(0, len(noSt)):
    chainsC.append(0)

for i in range(0, len(noSt)):
    maxC = 0
    cno = nos[i]
    for j in range(0, i):
        lastNo = nos[j]
        count = 0
        if cno % lastNo == 0:
            count = chainsC[j]

        if maxC < count:
            maxC = count
    chainsC[i] = maxC + 1

maxV = 0
for i in range(0, len(noSt)):
    count = chainsC[i]
    if count > maxV:
        maxV = count
if maxV < 2:
    maxV = -1

print maxV
