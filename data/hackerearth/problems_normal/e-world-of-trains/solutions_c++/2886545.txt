inline void add(int &a,int b){a+=b;if(a>=mod)a-=mod;}

int dp[3335][3335],s[3335][3335],t[3335][3335];
int get(int i,int j,int k){
    if(i<0)return s[j][k];
    int ans=s[j][k]-s[i][k];
    if(ans<0)ans+=mod;
    return ans;
}
int main(){
    //ios::sync_with_stdio(0);
    int T=5,i,j,k,m,n;
    int K,l;
    scanf("%d%d%d%d",&n,&l,&T,&K);
    dp[0][0]=1;s[0][0]=K;
    t[0][0]=K;
    for(i=1;i<=n;i++){
        for(k=0;k<=T;k++){
            add(dp[i][k],get(i-l,i-1,k));
            if(i>=l&&k){
                add(dp[i][k],t[i-l][k-1]);
            }
        }
        for(k=0;k<=T;k++){
            s[i][k]=s[i-1][k]+1LL*dp[i][k]*(K-1)%mod;
            if(s[i][k]>=mod)s[i][k]-=mod;
                t[i][k]=1LL*dp[i][k]*(K-1)%mod;
                if(k)t[i][k]+=t[i-1][k-1];
                if(t[i][k]>=mod)t[i][k]-=mod;
        }
    }
    printf("%d\n",dp[n][T]);
    return 0;
}
