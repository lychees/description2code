#include <iostream>
#include <cstdio>
#include <cstring>
#include <string>

using namespace std;

int main (void)
{
    unsigned int i, j, k, n;
    char str[10000];
    cin >> n;
    bool flag;

    while (n)
    {
        string order;
        scanf(" %[^\n]",str);
        //cout << str << endl;
        for (i=0; i< strlen(str); i++)
            if (str[i] != '$') continue;
            else break;

        flag = true;
        for (j=i+1; j<strlen(str); j++)
        {
            if (str[j] == ' ') continue;
            else if (!order.length() && str[j] == '0') continue;
            else if (str[j] >= '0' && str[j] <= '9') {
                order = order + str[j];
                //cout << "Order: " << order << " j: " << j << endl;
            } else flag = false;
            if (!flag) break;
        }
        cout << '$' << (order.length() ? order : "0") << endl;
        n--;
    }
    return 0;
}
