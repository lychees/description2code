#include <bits/stdc++.h>
using namespace std;
int main()
{
	int test,i=1,rem,num;
	cin>>test;
	map<int,char>m;
	m[10]='U';
	m[11]='V';
	m[12]='W';
	m[13]='X';
	m[14]='Y';
	m[15]='Z';
	while(test--)
	{
		int r,g,b;
		cin>>r>>g>>b;
		string r1="",g1="",b1="";
		rem=r%16;
		num=r/16;
		if(m.find(num)!=m.end())
			r1+=m[num];
		else r1+=to_string(num);
		if(m.find(rem)!=m.end())
			r1+=m[rem];
		else
			r1+=to_string(rem);
		rem=g%16;
		num=g/16;
		if(m.find(num)!=m.end())
			g1+=m[num];
		else g1+=to_string(num);
		if(m.find(rem)!=m.end())
			g1+=m[rem];
		else
			g1+=to_string(rem);
		rem=b%16;
		num=b/16;
		if(m.find(num)!=m.end())
			b1+=m[num];
		else b1+=to_string(num);
		if(m.find(rem)!=m.end())
			b1+=m[rem];
		else
			b1+=to_string(rem);

		cout<<"Case "<<i<<": ";
		cout<<r1<<" "<<g1<<" "<<b1<<endl;
		i++;


	}
	return 0;
}
