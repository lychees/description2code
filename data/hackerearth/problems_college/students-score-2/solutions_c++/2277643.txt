#include <iostream>
#include <cstring>

using namespace std;

int main()
{
    long long int n, i, j;
    cin >> n;
    long long int a[n], sum[n];
    for(i = 0; i < n; i++)
    	cin >> a[i];
	memset(sum, 0, sizeof(sum));
		sum[0] = a[0];
	for(i = 1; i < n; i++) {
		if(sum[i-1] + a[i] < 0)
		{
			if(a[i] < 0) {
				sum[i] = 0;
			}
			else 
			sum[i] = a[i];
		}
		else {
			sum[i] = sum[i-1] + a[i];
		}
	}
	long long int maxv = 0;
	for(i = 0; i < n; i++)
	{
		if(maxv < sum[i])
			maxv = sum[i];
	}
	cout <<maxv << endl;
    return 0;
}
