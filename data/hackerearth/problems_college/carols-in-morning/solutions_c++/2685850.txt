    #include<iostream>
    #include<cstdio>
    #include<cmath>
    #include<string.h>
    #include<algorithm>
    using namespace std;
     
    int m[10000];
    char in[1000];
     
    int main()
    {
        int len;
        int t,n;
        cin>>t;
        
        while(t--)
        {
          cin>>n;
          
          int i,j;
          for(i=0;i<n;i++)
          {
            scanf("%s",in);
            len=strlen(in);
     
            m[i]=0;
            for(j=0;j<len;j++)
            {
              if(in[j]=='Y')
                m[i]+=1<<j;
            }
          }
     
     
          int mn=len;
          for(i=0;i<(1<<len);i++)
          {
     
            for(j=0;j<n;j++)
            {
              if( (m[j] & i)==0)
                break;
            }
     
            if(j==n)
              mn=min(mn,__builtin_popcount(i));
          }
          cout<<mn<<endl;
        }
     
     
        return 0;
    }
