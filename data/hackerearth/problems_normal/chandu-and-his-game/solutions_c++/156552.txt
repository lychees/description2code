#include<cstdio>
#include<iostream>
#define ll long long
using namespace std;
#include<algorithm>
int main()
{
  ll t,n,ans,k,i;

      ll moves=0;
      cin>>n>>k;
      ll x[k+1],y[k+1];
      for(i=0;i<k;i++)
      {
          cin>>x[i];
      }
      for(i=0;i<k;i++)
      {
          cin>>y[i];
      }
      for(i=0;i<k;i++)
      {
             ll a=abs(x[i]-1)+abs(y[i]-1);
             ll b=abs(n-x[i])+abs(n-y[i]);
             ll c=abs(n-x[i])+abs(y[i]-1);
             ll d=abs(x[i]-1)+abs(n-y[i]);
            moves+=min(a,min(b,min(c,d)));
      }
        cout<<moves<<endl;

}
