#include <bits/stdc++.h>
using namespace std;

#define gt(n) scanf("%d", &n)

int greater_number(int a[], int x, int n)
{
    int low = 0;
    int high = n - 1;
    int p = n;
    int mid;
    //sort(a, a + n);
    int i;
    int k;


    while (low + 2 <= high) {
                mid = (low + high) / 2;

                if (a[mid] > x) {
                    high = mid;
                } else {
                    low = mid;
                }
    }

            for (i = low; i <= high; i++) {
                    if (a[i] > x) {
                        p = i;
                        break;
                    }
            }

            int ans = n - p;

    return ans;
}
int main()
{
    int t;
    gt(t);

    while (t--) {
        int n;
        gt(n);
        int a[n];
        int i;
        int b[n];

        for (i = 0; i < n; i++) {
            gt(a[i]);
            b[i] = a[i];
        }

        sort(b, b + n);
        for (i = 0; i < n; i++) {
            int ans = greater_number(b, a[i], n) + 1;
            printf("%d ", ans);
        }
        printf("\n");
    }

    return 0;
}
