text=raw_input()
d = dict.fromkeys(text, 0)
for c in text: d[c] += 1
x=[]
flag=0
for key, value in d.iteritems():
    x.append([key,value])
i=0
while i<len(x):
    if x[i][1]%2!=0:
        if flag==0:
            flag=1
        else:
            print "NO"
            i=len(x)+1
    i=i+1
if i==len(x):
    print "YES"
