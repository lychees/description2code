#include <iostream>
using namespace std;
#define ll long long
#define MAX 100002
#define mod 1000000007
ll A[MAX];
ll powmod(ll a,ll b=mod-2,ll c=mod){
	ll ans=1;
	while(b){
		if(b&1){
			ans*=a;
			ans%=c;
		}
		a*=a;
		a%=c;
		b>>=1;
	}
	return ans;
}
int main()
{
//	ios::sync_with_stdio(0);cin.tie(0);
    int n,q;
    cin>>n;
    ll ans=1;
    ll tmp=1;
    ll zero=0;
    for(int i=1;i<=n;i++){
    	cin>>A[i];
    	if(A[i]==0)zero++;
    	if(A[i]!=0){
    		ans*=A[i];
    	    ans%=mod; 
    	}
    }
    cin>>q;
    int t,i;
    ll v;
    while(q--){
        cin>>t>>i;
        if(t==1){
        	if(zero>1){cout<<0<<'\n';continue;}
        	else if(A[i]==0&&zero==1){
        		cout<<ans<<'\n';
        	}
        	else if(A[i]!=0&&zero==1){
        		cout<<0<<'\n';
        	}
        	else {
        		cout<<(ans*powmod(A[i]))%mod<<'\n';
        	}
        }
        else if(t==0){
        	cin>>v;
        	if(A[i]==0&&v!=0){
        		zero--;
        		ans*=v;
        		ans%=mod;
        		A[i]=v;
        		continue;
        	}
        	else if(A[i]!=0&&v==0){
            	zero++;
            	ans*=powmod(A[i]);
            	ans%=mod;
            	A[i]=v;
            	continue;
            }
            else if(A[i]==0&&v==0)continue;
        	ans*=powmod(A[i]);
        	ans%=mod;
        	ans*=v;
        	ans%=mod;
        	A[i]=v;
        }
    }
    
    return 0;
}
