from math import sqrt


def sieve(n):
    primes = range(3, n+1, 2)
    for base in xrange(len(primes)):
        if primes[base] is None:
            continue
        if primes[base] > sqrt(n):
            break

        for i in xrange(base+(base+1)*primes[base], len(primes), primes[base]):
            primes[i] = None

    primes.insert(0, 2)
    return filter(None, primes)


n = input()
arr = sieve(n)
# print arr
art = range(2, n+1)
for it in art:
    if it not in arr:
        print it
