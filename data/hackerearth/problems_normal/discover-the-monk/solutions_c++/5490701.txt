#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int N,Q,X;
    cin>>N;cin>>Q;
    int A[Q];
    int i;
    for(i=0;i<N;i++)
    {
    	cin>>A[i];
    }
    sort(A,A+N);
    for(i=0;i<Q;i++)
    {
    	
    	cin>>X;
    	if(binary_search(A,A+N,X))
    	{
    		cout<<"YES"<<endl;
    	}
    	else
    	{
    		cout<<"NO"<<endl;
    	}
    }
    return 0;
}
