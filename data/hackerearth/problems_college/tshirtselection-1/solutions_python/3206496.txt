def read_ints():
	return map(int, raw_input().split())

def main():
	T = read_ints()[0]
	for t in xrange(T):
		n = read_ints()[0]
		a = read_ints()
		
		ret, mlen, dSet, rend = -1, 0, set(), 0
		for i in xrange(n):
			while rend < n and a[rend] not in dSet:
				dSet.add(a[rend])
				rend += 1
			l = len(dSet)
			if mlen < l:
				mlen = l
				ret = i + 1
			
			dSet.remove(a[i])
		
		print ret, (ret + mlen - 1)

main()
