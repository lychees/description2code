#include <iostream>
#include <math.h>
using namespace std;

int main()
{
    int B;
	unsigned long int T;
	long long int a;
    unsigned long long int N,S,K,i;
    
    cin>>T;
    
	while(T>0)
	{
		a=-1;
		cin>>N;
		S=sqrt(N);
		K=S;
		i=-1;
		while(K)
		{
			K=K/10;
			i++;
		}
		
		if(i==0)
		{i=1;}
		
		else
		{i=(S-(9*i));}
		
		while(i<=S)
		{
			K=i;
			B=0;
			while(K)
			{
				B=B+(K%10);
				K=K/10;
			}
			
			if(!((i*i)+(B*i)-N))
			{
				a=i;
				break;
			}
			
			i++;
		}
		
		
		cout<<a;
		cout<<endl;
		T--;
	}
    
   
    return 0;
}
