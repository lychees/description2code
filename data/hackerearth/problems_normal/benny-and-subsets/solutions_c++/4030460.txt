#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

#define rd(x) scanf("%d",&x)
#define rl(x) scanf("%lld",&x)
#define F first
#define S second
#define MP make_pair
#define PB push_back
#define PII pair<int,int>
#define PLL pair<ll,ll>
#define maxn 1009
#define maxr 2050
#define mod 10000007








ll ans = 0;
ll dp[maxn][maxr];

vector<int> v;
int vs;
int f[25];
int n,x;



void gen(int in){
	if(in>=vs){
		int a = 0;
		for(int i=0;i<vs;i++){if(f[i])a ^= v[i];}
		int p = a^x;
		if(p<2048){ans += dp[n][p];ans %= mod;}
		return;
	}
	f[in] = 1;
	gen(in+1);
	f[in] = 0;
	gen(in+1);
}






int main(){
	rd(n);rd(x);

	for(int i=0;i<maxr;i++)dp[0][i] = 0;
	dp[0][0] = 1;
	int i = 1;
	for(int k=1;k<=n;k++){
		int a;
		rd(a);
		if(a>1024){v.PB(a);continue;}
		for(int j=0;j<maxr;j++)dp[i][j] = dp[i-1][j];
		for(int j=0;j<2048;j++){
			int x = j^a;
			dp[i][x] += dp[i-1][j];
			dp[i][x] %= mod;
		}
		i++;
	}
	vs = v.size();
	n -= vs;
	fill(f,f+25,0);

	gen(0);
	cout<<ans<<endl;
}
