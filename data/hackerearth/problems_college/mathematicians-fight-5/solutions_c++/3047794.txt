#include <cstdlib>
#include <iostream>
#include <math.h>

using namespace std;
long long T,t,n;
int main(int argc, char *argv[])
{   cin>>T;
    while(T)
    {  cin>>n;
       t=(long long) log2(n);
       cout<<(2*n-(long long)pow(2,t+1)+1)<<endl;
       T--;        
    }
    
    return EXIT_SUCCESS;
}
