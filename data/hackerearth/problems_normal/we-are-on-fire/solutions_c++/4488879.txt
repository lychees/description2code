#include<bits/stdc++.h>
using namespace std;
bool g[1001][1001];
int res,n,m;
void dfs(int x,int y)
{
    if(x<0 or y<0 or x>=n or y>=m)return ;
    if(!g[x][y])return;
    g[x][y]=false;
    res--;
    dfs(x-1,y);dfs(x+1,y);dfs(x,y+1);dfs(x,y-1);
}
int main()
{
    int q;
    scanf("%d%d%d",&n,&m,&q);
    for(int i=0;i<n;i++)
        for(int j=0;j<m;j++)
        {
            scanf("%d",&g[i][j]);
            if(g[i][j])res++;
        }
    while(q--)
    {
        int x,y;
        scanf("%d%d",&x,&y);
        x--;y--;
        dfs(x,y);
        cout<<res<<"\n";
    }
    return 0;
}
