#include <bits/stdc++.h>
using namespace std;

int main()
{
    char s[100000];
    long long int q,i,a,n,b;
    cin>>s;
    cin>>q;
   
   n=strlen(s);
   
    for(i=0;i<q;i++)
    {cin>>a>>b;
   
     a%=n;  b%=n;
      if(a==0) 
      a=n;
      if(b==0) 
      b=n;
      
    	if(s[a-1]==s[b-1])
    	{cout<<"Yes"<<endl;}
    	else 
    	{cout<<"No"<<endl;}
   
        
    }
    return 0;
}
