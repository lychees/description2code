#include <iostream>
using namespace std;

int main()
{
    long long t;
    cin>>t;
    while(t--){
    	string strt;
    	cin>>strt;
    	long long res = 0;
    	long long len = strt.length();
    	for(int i = 0; i < len; i++){
    		if(strt[i] == '-'){
    			if((i == 0 || strt[i-1] != 'S') && (i == len - 1 || strt[i+1] != 'S') 
    			&& (i == len - 1 || strt[i+1] != 'B') && 
    			(i >= len - 2 || strt[i+2] != 'B') )
    			res++;
    		}
    	}
    	cout<<res<<endl;
    }
    return 0;
}
