def isSynonym(x, y):
    if (len(x) == len(y) and x[0] == y[0]):
        return True
    else:
        return False


def isAntonym(x, y):
    if (len(x) == len(y) and ord(y[0]) - ord(x[0]) == 3):
        return True


s = raw_input()
n = int(input())

c1 = 0
c2 = 0

while (n):
    n -= 1
    x = raw_input()
    if (isSynonym(s, x)):
        c1 += 1
    if (isAntonym(s, x)):
        c2 += 1

print c1, c2
