#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;
int main()
{
int cx1,cx2,cy1,cy2,cr1,cr2;
cin>>cx1>>cy1>>cr1;
cin>>cx2>>cy2>>cr2;
int y=cy2-cy1;
int x=cx2-cx1;
int r=cr1+cr2;
int d=sqrt(y*y+x*x);
if(d==r)
cout<<"T";
else if(d>r)
cout<<"N";
else
cout<<"Y";
}
