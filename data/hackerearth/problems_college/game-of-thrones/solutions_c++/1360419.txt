#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;

int a[100005];
int main() {
    int t,i,killers,j,n;
    scanf("%d",&t);
    while(t--){
    	scanf("%d",&n);
        for(i=0;i<n;i++)
        scanf("%d",&a[i]);
        sort(a,a+n);
        
        killers=0;
        i=0;
        while(i<n-1)
        {
        	j=i+1;
        	while(a[i]==a[j] && j<n)
        	{
        		killers++;
        		j++;
        	}
        	i=j;
        }
        printf("%d\n",killers);
        }
        
    return 0;
}
