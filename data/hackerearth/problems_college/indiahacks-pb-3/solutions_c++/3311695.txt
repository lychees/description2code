#include <iostream>
using namespace std;

int main()
{
    int t,n,i,j,k;
    static int arr[10005];
    static long dp[10005];
    cin>>t;
    while(t--)
    {
    	cin>>n>>k;
    	for(i=0;i<n;i++)
    	{
    		cin>>arr[i];
    	}
    	dp[0]=arr[0];
    	for(i=1;i<n;i++)
    	{	dp[i]=arr[i];
    		if((i-(k+1))>=0 )
    		{	if(dp[i-(k+1)]>0)
    			dp[i]+=dp[i-(k+1)];
    		}
    		dp[i]=max(dp[i],dp[i-1]);
    		//cout<<dp[i];
    	}
    	int m=0;
    	for(i=0;i<n;i++)
    	if(dp[i]>m)
    	m=dp[i];
    	cout<<m<<endl;
    }
    return 0;
}
