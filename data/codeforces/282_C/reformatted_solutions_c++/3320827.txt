#include <cstdio>
#include <cstring>
#include <fstream>
#include <iostream>
#include <string>
using namespace std;
int main() {
  string a, b;
  cin >> a >> b;
  cout << (a.size() != b.size() || ((a.find('1') == string::npos) ^
                                    (b.find('1') == string::npos))
               ? "NO"
               : "YES")
       << endl;
  return 0;
  return 0;
}
