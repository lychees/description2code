#include <iostream>

using namespace std;

int main()
{
	ios_base::sync_with_stdio( false );
	int n, k, i, ct, fCt = 0, j;
	char ch;
	
	cin >> k;
	
	char *s = new char[ 5001 ];
	cin >> s;
	
	for( n = 0;s[ n ];n++ );
	
	for( i = 0;i < n;i++ ) {
		
		bool A[ 26 ];
		ct = 0;
		for( j = 0;j < 26;j++ )
			A[ j ] = false;
			
		for( j = 0;j < n - i;j++ ) {
			
			ch = s[ i + j ];
			
			if( !A[ ch - 97 ] ) {
				A[ ch - 97 ] = true;
				ct++;
				if( ct == k )
					fCt++;
			} else {
				if( ct == k )
					fCt++;
			}
		}
		
	}
	
	cout << fCt << "\n";
	
	delete s;
	
    return 0;
}
