#include<bits/stdc++.h>
using namespace std;
typedef long long lld;


int main()
{
	
	lld n,i;
	cin>>n;
	
	lld a[100005];
	for(i=0;i<n;i++)
	cin>>a[i];
	
	
	lld j;
	lld dp[100005];
	memset(dp,0,sizeof dp);
	lld ans=1;
	for(i=0;i<n;i++)
	dp[i]=1;
	for(i=1;i<n;i++)
	{   
		for(j=0;j<i;j++)
		{
			if((a[i]>0&&a[j]<0||a[i]<0&&a[j]>0)&&(abs(a[i])>abs(a[j])))
			dp[i]=max(dp[i],1+dp[j]);
		
			ans=max(dp[i],ans);
		}
		
		
	}
	
	cout<<ans<<"\n";
	
	
	
	
	
	
	
}
