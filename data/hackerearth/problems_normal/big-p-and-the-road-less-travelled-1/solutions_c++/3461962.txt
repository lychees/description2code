# include<iostream>
# include<vector>
using namespace std;

vector<int> v[10004];
long long dp[10004];
int visited[10004];
int n;
long long dfs(int u)
{
    if(u==n) return 1;
    if(visited[u]) return dp[u];
    visited[u]=1;
    long long ans=0;
    for(int i=0; i<v[u].size(); i++) ans+=dfs(v[u][i]);

    return dp[u]=ans;
}


int main()
{
    cin>>n;
    int x,y;
    do
    {
        cin>>x>>y;
        v[x].push_back(y);
    }while(x!=0 && y!=0);


    cout<<dfs(1)<<endl;
}
