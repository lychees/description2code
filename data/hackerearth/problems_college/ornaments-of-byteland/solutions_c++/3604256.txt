#include <iostream>
#include<string>
#include<algorithm>
using namespace std;
int a[26]={0};
int main()
{
    int t;
    cin>>t;
        string s;
    int n =t;

    while(t--){
        cin>>s;
        bool p[26]={0};
        for(int i = 0 ;i<s.length();i++){
            p[s[i]-97] = 1;
        }
        for(int  i = 0;i<26;i++) if(p[i])  a[i]++;
    }
    sort(a,a+26);
    int ans = upper_bound(a,a+26 ,n)- lower_bound(a,a+26,n);
    cout<<ans<<endl;;
    return 0;
}
