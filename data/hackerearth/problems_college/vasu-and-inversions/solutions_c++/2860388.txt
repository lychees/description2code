#include <iostream>
using namespace std;

int main()
{
	int t,i,n,k,a[2],v;
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		a[0]=a[1]=0;
		for(i=0;i<n;i++)
		{
			scanf("%d",&v);
			a[v]++;
		}
		if(k<=a[0])
		cout<<k+a[1];
		else
		cout<<a[1]+a[0]-(k-a[0])%2;
		cout<<"\n";
	}
    //cout << "Hello World!" << endl;
    return 0;
}
