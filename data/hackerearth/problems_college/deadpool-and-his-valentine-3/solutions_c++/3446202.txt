#include <bits/stdc++.h>
using namespace std;

const int maxn = 101;
bool on[maxn];

int main()
{
   ios::sync_with_stdio(false);
   int n, m;
   cin >> n >> m;
   for (int i = 0; i < n; i++) {
        int x;
        cin >> x;
        for (int j = 0; j < x; j++) {
             int y;
             cin >> y;
             on[ y ] = true;
        }
   }
   int all = 0;
   for (int i = 1; i <= m; i++)
        all += (on[i] == true);
   if (all == m) 
       puts("YES");
   else
       puts("NO");
   return 0;
}
