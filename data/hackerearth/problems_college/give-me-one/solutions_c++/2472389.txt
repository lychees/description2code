typedef pair<int, int> ii;
#define fr first

#define mp make_pair
#define max(a, b) (a > b ? a : b)
#define min(a, b) (a < b ? a : b)
#define sqr(n) n*n
#define all(x) x.begin(),x.end()

typedef long long ll;

int memo[(int)1e7 * 2 + 1];
int bt(int n)
{
    while (n < 0);
    if (n == 1)
        return 0;
    if (n == 0)
        return 0;
    if (memo[n] != -1)
        return memo[n];
    int ret = INT_MAX;
    if (n % 3 == 0)
        ret = min(ret, bt(n / 3));
    if (n % 2 == 0)
        ret = min(ret, bt(n / 2));
    ret = min(ret, bt(n - 1));
    return memo[n] = 1 + ret;
}

int main()
{
    memset(memo, -1, sizeof memo);
    for (int i = 0; i <= 1e7 * 2; i++)
        bt(i);
    int t;
    scanf("%d", &t);
    while (t--)
    {
        ll n;
        scanf("%lld", &n);
        n = abs(n);
        printf("%d\n", bt(n));
    }
}
