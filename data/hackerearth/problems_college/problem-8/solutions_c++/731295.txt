#include <iostream>
#include<vector>
using namespace std;

#define ll long long unsigned

void fibo(ll n)
{
    vector<ll> u(10000);
    u[0]=1;
    u[1]=1;
    cout<<"1\n1\n";
    for(ll i=2;i<n;i++)
    {
        u[i]=u[i-1]+u[i-2];
        if(u[i]<=n){
        cout<<u[i]<<endl;
    }
    else break;
    }
}

int main()
 {

	ll n=0;
    cin>>n;
    fibo(n);
    return 0;
}
