from sys import stdin
t = int(stdin.readline())
for _ in xrange(t):
	a,b = stdin.readline().strip().split()
	fir = sec = ''
	u=[]
	v=[]
	u2=[]
	v2=[]
	n = len(a)
	for j in xrange(n):
		i = a[j]
		if i!='#':
			fir+=i
			if i=='A':
				u.append(j)
			else:
				v.append(j)
	for j in xrange(n):
		i = b[j]
		if i!='#':
			sec+=i
			if i=='A':
				u2.append(j)
			else:
				v2.append(j)
	if fir==sec:
		if u2 <=u and v2>=v:
			print "Yes"
		else:
			print "No"
	else:
		print "No"
