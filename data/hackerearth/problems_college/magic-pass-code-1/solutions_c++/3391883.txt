#include <bits/stdc++.h>
using namespace std;

int solve(int n){
	int t = 0;
	while(n!=0){
		t = t + (n%10);
		n = n/10;
	}
	return t;
}

int main(){
	int n; 
	int msum=0, mnum=0;
	int temp;
	while(cin>>n){
		temp = solve(n);
		if(temp > msum){
			msum = temp;
			mnum = n;
		}
	}
	cout << mnum << endl;
	return 0;
}
