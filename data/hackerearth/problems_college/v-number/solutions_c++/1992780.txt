#include <bits/stdc++.h>

using namespace std;

int main()
{
	char s[100000];
	int i,j,ans;
	int t;
	scanf("%d",&t);
	while (t --) {
		scanf("%s",s);
		int ans1, ans2;
		ans1 = ans2 = 0;
		int len = strlen(s);
		for (i = 0; i < len/2; i++) {
			ans1 += s[i] - 48;
		}
		for (i = len-1; i >= len/2; i--) {
			ans2 += s[i] - 48;
		}
		if (ans1 == ans2) {
			printf("YES\n");
		} else {
			printf("NO\n");
		}
	}
	
	return 0;
}
