#include<iostream>
#include<string.h>
using namespace std;

int main()
{
	int n,l,i,k,found;
	long long cost = 0;
	char string[32805];
	n = 882;
	
	while(n--)
	{
		//getc(stdin);
		scanf(" %s",string);
		l = strlen(string);
		
		cost = 0;
		i = l-1;
		k = l-1;
		while(k>=1)
		{
			i = k;
			found = 0;
			while(i>=0 && string[i] == '1')
			{
				string[i] = '0';
				string[k] = '\0';
				i--, k--;
				cost++;
				found = 1;
			}
			if(i<0 && string[i+1]=='1')
			{
				break;
			}
			else if(found == 1)
			{
				string[i] = '1';
				cost++;
			}
			while(k>=0 && string[k]=='0')
			{
				string[k] = '\0';
				k--;
				cost++;
			}
		}
		printf("%lld\n",cost);
	}
	return 0;
}
