#include <bits/stdc++.h>

using namespace std;

#define ll long long
#define S(x) scanf("%d", &x)
#define Sl(x) scanf("%lld", &x)
#define mem(a,x) memset(a,x,sizeof(a))

ll int a[10001], k, M[10001];

ll int rec(ll int idx)
{
    ll int x,y;

    if(idx < 0)
        return 0;

    if(M[idx] != LLONG_MIN)
        return M[idx];

    y = rec(idx-1);

    x = rec(idx-k);

    //cout << idx << " -> " << x << " " << y << endl;


    M[idx] = max(x+a[idx],y);

    return M[idx];
}

int main()
{
    ll int t,n,i;

    cin >> t;

    while(t--){
        cin >> n >> k; ++k;

        for(i = 0; i <= 10000; ++i){
            M[i] = LLONG_MIN;
        }

        for(i = 0; i < n; ++i){
            cin >> a[i];
        }
        cout << rec(n-1) << endl;
    }
    return 0;
}
