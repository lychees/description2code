using namespace std;
#include<iostream>
int main()
{
    int t;
    cin>>t;
    while(t>0)
    {
        string s,s1; cin>>s; cin>>s1;
        int i=0,j=0; char ch='1';
        int d=s.length(),d1=s1.length();
        //cout<<d<<" "<<d1<<endl;
        while(i<d)
        {
            while(i<d&&s[i]=='#')
                i++;
            while(j<d1&&s1[j]=='#')
                j++;
                //cout<<i<<" "<<j<<endl;
                if(i==d&&j!=d1) {ch='0'; break; }
                if(j==d1&&i!=d) {ch='0'; break; }
                if(i==d&&j==d1) break;
            if(s[i]!=s1[j])     {ch='0'; break; }
            if(s[i]=='A'&&j>i) { ch='0'; break; }
            if(s[i]=='B'&&j<i) {ch='0'; break; }
            i++; j++;
        }
        if(ch=='1')
        cout<<"Yes"<<endl;
        else cout<<"No"<<endl;
            t--;
    }
}
