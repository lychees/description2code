#include<stdio.h>
#include<algorithm>
#include<vector>
using namespace std;
int main(){
 int test,num,k,i,j,costtocal,temp;
 vector<int> a1,b1;
 scanf("%d",&test);
 while(test--){
   scanf("%d%d",&num,&k);
   for(i=0;i<num;++i){
     scanf("%d",&temp);
     a1.push_back(temp);
   }
   costtocal=-1;
   sort(a1.begin(),a1.end());
   for(i=0;i<num;++i){
     if(i==0 || a1[i]!=a1[i-1]){
       b1.clear();
       for(j=0;j<num;++j){
          b1.push_back(a1[i]-a1[j]);
          if(b1[j]>=0) b1[j]*=3;
          else b1[j]*=(-5);
       }
       sort(b1.begin(),b1.end());
       temp=0;
       for(j=0;j<k;++j)
        temp+=b1[j];
       if(costtocal==-1 || costtocal>temp) costtocal=temp;
     }
   }
   printf("%d\n",costtocal);
   a1.clear();
 }
 return 0;}
