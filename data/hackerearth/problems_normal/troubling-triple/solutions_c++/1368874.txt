#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int n,k;
    cin>>n>>k;
    int ans=0;
    vector <int> a;
    for(int i=0;i<n;i++)
    {
    	int h;
    	cin>>h;
    	a.push_back(h);
    }
    sort(a.begin(),a.end());
    for(int i=0;i<n;i++)
    	for(int j=i+1;j<n;j++)
    	{
    		int h=k/(a[i]*a[j]);
    	//	cout<<h<<endl;
    		int h1=(upper_bound(a.begin()+j+1,a.end(),h)-a.begin());
    		
    		ans+=max(0,h1-j-1);
    	//	cout<<h1<< " ";
    	}
    cout<<ans;
    return 0;
}
