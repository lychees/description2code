#include <bits/stdc++.h>
using namespace std;
#define scan(x) scanf("%d",&x)
#define print(x) printf("%d\n",x)
#define MOD 1000000007
#define scanl(x) scanf("%lld",&x)
#define printl(x) printf("%lld",x)
bool arr[1001][1001];
bool vis[1001][1001];
int xx[4]={0,0,1,-1};
int yy[4]={1,-1,0,0};
 
bool check(int x,int y,int n,int m)
{
	if(x>=0 && y>=0 && x<n && y<m) return true;
	else return false;
}
 
int findval(int x,int y,int n,int m,int curr)
{
	for(int k=0;k<4;k++)
	{
		int i=x+xx[k];
		int j=y+yy[k];
		if(!check(i,j,n,m)) continue;
		else if(vis[i][j]) continue;
		else if(!vis[i][j] && arr[i][j]==0) 
		{
			vis[i][j]=1;
			continue;
		}
		else if(!vis[i][j] && arr[i][j]==1)
		{
			vis[i][j]=1;
			curr=1+findval(i,j,n,m,curr);
		}
	}
	return curr;
}
 
 
int main()
{
	int n,m,q;
	scan(n);
	scan(m);
	scan(q);
	int count=0;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++) 
		{
			scan(arr[i][j]);
			vis[i][j]=false;
			if(arr[i][j]) count++;
		}
	}
//	cout<<"count="<<q<<"\n";
	while(q--)
	{
		int x,y;
		scan(x);
		scan(y);
		x--;
		y--;
		if(vis[x][y]) print(count);
		else if(!vis[x][y] && arr[x][y]==0)
		{
			vis[x][y]=1;
			print(count);
		}
		else if(!vis[x][y] && arr[x][y]==1)
		{
			vis[x][y]=1;
			int temp=findval(x,y,n,m,1);
			count-=temp;
			print(count);
		}
	}
    return 0;
}
