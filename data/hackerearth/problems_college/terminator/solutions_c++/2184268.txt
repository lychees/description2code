#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	int t,t1;
	cin>>t;
	for(t1=0;t1<t;t1++)
	{
		string s,d="";
		cin>>s;
		int i=0,g=s.length();
		while(i<g)
		{
			if(s[i]=='x')
			{
			//	cout<<i<<" x ";
				while(i<g&&s[i]=='x')
				{
					d=d+'x';
					i++;
				}
				while(i<g&&s[i]=='o')
					i++;
			//	cout<<i<<endl;
			}
			if(s[i]=='o')
			{
			//	cout<<i<<" o ";
				int j=0;
				while(j+i<g&&s[j+i]=='o')
					j++;
				if(s[j+i]=='x')
					i+=j;
				else
				{
					i+=j;
					while(j>0)
					{
						d=d+'o';
						j--;
					}
				}
			//	cout<<i<<endl;
			}
			if(s[i]=='*')
				while(i<g&&s[i]=='*')
				{
					d=d+'*';
					i++;
				}
		}
		cout<<d<<endl;
	}
	return 0;
}
