#include <iostream>
using namespace std;

int main() {
	// your code goes here
	int n;
	cin>>n;
	int a[n];
	for(int i=1;i<=n;i++){
		cin>>a[i];
	}
	int q;
	cin>>q;
	int l,r;
	for(int i=1;i<=q;i++){
		cin>>l>>r;
		int sum=0;
		for(int j=l;j<=r;j++){
			sum+=a[j];
		}
		cout<<sum<<"\n";
	}
	return 0;
}
