#include <iostream>
#include<string.h>
#include<stdio.h>
#include<iomanip>
#define ll long long


using namespace std;
ll X,Y,N,W;
long double P1,P2;

long double dp[1010][1010];
bool vis[1010][1010];

long double solve(ll shots, ll score)
{
    if(shots==N || score>=W)
    {
        if(score>=W) return 1.0; else  return 0.0;
    }
    else if(vis[shots][score]) return dp[shots][score];
    else
    {
        vis[shots][score]=true;
        long double ans1 = solve(shots+1, score+X)*P1 + (1.0 - P1)*solve(shots+1, score);
        long double ans2 = solve(shots+1, score+Y)*P2 + (1.0 - P2)*solve(shots+1, score);
        dp[shots][score] = max(ans1, ans2);
        return dp[shots][score];
    }

}

int main()
{
    //freopen("input.txt","r",stdin);
    ll t;
    cin>>t;
    while(t--) {
     cin>>X>>Y>>N>>W>>P1>>P2;
     P1/=100.0; P2/=100.0;
     memset(vis,false,sizeof(vis));
    // for(int i=0; i<=N; i++) for(int j=0; j<=W; j++) dp[i][j]=0.0;
     long double ans = (long double)solve(0,0);
     ans*=100.0;
     //cout<<dp[1][0]<<endl;
     cout<<fixed;
     cout<<setprecision(6)<<ans<<endl;
    }
    return 0;

}
