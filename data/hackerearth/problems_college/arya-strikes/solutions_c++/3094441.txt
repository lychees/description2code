#include<bits/stdc++.h>
using namespace std;

void solve()
{
    int n;
    cin>>n;
    int a[n][n];
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<n;j++)
        {
            cin>>a[i][j];
        }
    }
    for(int i=1;i<n;i++)
    {
        a[0][i]=(a[0][i] && a[0][i-1]);
        a[i][0]=(a[i][0] && a[i-1][0]);
    }
    for(int i=1;i<n;i++)
    {
        for(int j=1;j<n;j++)
        {
            a[i][j]=(a[i][j] & (a[i-1][j] || a[i][j-1]));
        }
    }
    if(a[n-1][n-1]) cout<<"POSSIBLE"<<endl;
    else cout<<"NOT POSSIBLE"<<endl;
}

int main()
{
    int t;
    cin>>t;
    while(t--) solve();
    return 0;
}
