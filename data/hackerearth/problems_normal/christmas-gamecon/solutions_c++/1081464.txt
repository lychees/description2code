#include <bits/stdc++.h>

#define ll long long int
#define s(a) scanf("%lld",&a)

using namespace std;

int main()
{
    ll t,n,x,tot,i;
    s(t);
    while(t--) {
        s(n);tot=0;
        s(tot);
        tot=tot%5;
        for(i=1;i<n;i++) {
            s(x);x=x%5;
            tot=tot^x;
        }
        if(tot) cout<<"Aniruddha\n";
        else cout<<"Andrew\n";
    }
    return 0;
}
