#include<iostream>
#include<bits/stdc++.h>
using namespace std;
int main(){
int t;cin>>t;
while(t--){
    int n;cin>>n;
    int a[n];
    for(int i=0;i<n;i++) cin>>a[i];
    sort(a,a+n);
    int i=n-1;

   while(1 && i>=2){

       if(a[i]<a[i-1]+a[i-2]){

            cout<<a[i-2]<<" "<<a[i-1]<<" "<<a[i]<<endl;
            break;
        }
        else i=i-1;
        }
        if(i<=1) cout<<-1<<endl;

    }
}
