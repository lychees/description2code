s5='555'
s3='33333'
t=int(raw_input())
while t>0:
    n=int(raw_input())
    d3=int(n/3)
    d5=0
    m3=n%3
    if m3==0:
        res=s5*d3
    elif m3==1:
        if d3>=3:
            d3-=3
            d5+=2
            res=s5*d3+s3*d5
        else:
            res=-1
    elif m3==2:
        if d3>=1:
            d3-=1
            d5+=1
            res=s5*d3+s3*d5
        else:
            res=-1
    print res
    t-=1
