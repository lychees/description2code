#include<iostream>
#include<math.h>
#include<stdio.h>
#define ll long long int
using namespace std;
int main()
{
	ll tc=0;
	scanf("%lld",&tc);
	while(tc-->0)
	{
		ll n;
		scanf("%lld",&n);
		ll ans=0;
		for(ll i=1;i<=(ll)sqrt(n);i++)
		{
			if(n%i==0)
			{
				if(i%4!=0)
					ans=ans+i;
				if(i*i!=n)
				{
					if((n/i)%4!=0)
						ans=ans+n/i;
				}
			}
		}
		ans*=8;
		printf("%lld\n",ans);
	}
	return 0;
}
