t = int(raw_input())
arr = []


def sq(num):
    s = 0
    while num != 0:
        m = num % 10
        s += m**2
        num /= 10
    return s


for i in range(t):
    n = int(raw_input())
    word = ""
    k = n
    while len(word) != 1:
        k = sq(k)
        word = str(k)
    if int(word) == 1:
        arr.append(True)
    else:
        arr.append(False)
for i in range(len(arr)):
    if arr[i]:
        print "Case #"+str(i+1)+":YES"
    else:
        print "Case #"+str(i+1)+":NO"
