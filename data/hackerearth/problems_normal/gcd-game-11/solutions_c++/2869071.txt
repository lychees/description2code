#include<bits/stdc++.h>
using namespace std;
#define ll long long 
#define mod 1000000007


int dp[1010][1010][4];
int gcd[1010][1010];
int solve(int a,int b,int turn){
	if(turn == 0 && a == 1) return 0;
	else if(turn == 1 && b == 1) return 0;
	if(dp[a][b][turn]!=-1) return dp[a][b][turn];
	int ans = 0;
	if(turn == 0){
		if(gcd[a][b]>1) ans = (!solve(a,b-1,turn^1)) || (!solve(a,b/gcd[a][b],turn^1));
		else ans = !solve(a,b-1,turn^1);
	}
	else{
		if(gcd[a][b]>1) return (!solve(a-1,b,turn^1)) || (!solve(a/gcd[a][b],b,turn^1));
		else ans = !solve(a-1,b,turn^1);
	}
	return dp[a][b][turn] = ans;
}

int main(){
	int t;
	cin>>t;
	for(int i=1;i<=1000;i+=1){
		for(int j=1;j<=1000;j+=1){
			if(gcd[i][j]==0)
			gcd[i][j]=gcd[j][i]=__gcd(i,j);
		}
	}
	memset(dp,-1,sizeof(dp));
	while(t--){
		
		int a,b;
		scanf("%d %d",&a,&b);
		if(a==1 && b==1) cout<<"Draw"<<endl;
		else if(a==1) cout<<"Chandu Don"<<endl;
		else if(b==1) cout<<"Arjit"<<endl;
		else if(solve(a,b,0)) cout<<"Arjit"<<endl;
		else cout<<"Chandu Don"<<endl;
	}

}
