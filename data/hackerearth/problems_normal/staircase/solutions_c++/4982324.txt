#include <iostream>
using namespace std;
typedef long long ll;

int main()
{
    ll n;
    cin >> n;
    ll jack[1000005] ;
    ll jill[1000005] ;
    
    jack[0] = 0;
    jack[1] = 1;
    jack[2] = 2;
    jack[3] = 4;
    if(n >= 4){
    for(int i = 4; i <= n; i++){
    	jack[i] = jack[i-1] + jack[i-2] + jack[i-3] ;
    }
    }
    jill[0] = 0;
    jill[1] = 1;
    jill[2] = 2;
    if(n >= 3){
    for(int i = 3; i <= n; i++){
    	jill[i] = jill[i-1] + jill[i-2] ;
    }	
    }
    if(n > 0){
    cout << "Jack-" << jack[n] << endl ;
    cout << "Jill-"  << jill[n] << endl ;
    }
    else
    cout << "ERROR" << endl;
    
    return 0;
}
