#include <iostream>
#include <algorithm>
using namespace std;

string s;

bool comp(string a, string b){
	int i = 0;
	while(a[i] != '\0' && b[i] != '\0' && a[i] == b[i]){
		i++;
	}
	if(a[i] == '\0'){
		return true;
	}
	if(b[i] == '\0'){
		return false;
	}
	int ai = 0, bi = 0;
	while(a[i] != s[ai]){
		ai++;
	}
	while(b[i] != s[bi]){
		bi++;
	}
	return ai < bi;
}

int main()
{
    int t,n;
    cin >> t;
    while(t--){
    	cin >> s;
    	cin >> n;
    	string a[n];
    	for(int i = 0; i < n; i++){
    		cin >> a[i];
    	}
    	sort(a, a+n, comp);
    	for(int i = 0; i < n; i++){
    		cout << a[i] << endl;
    	}
    }
    return 0;
}
