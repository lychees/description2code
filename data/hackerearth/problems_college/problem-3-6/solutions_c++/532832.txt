#include <iostream>

int main()
{
    int n;int x[100000]={}, y[100000]={}, h[100000]={},a[100000]={};
    std::cin>>n;int l=0;
    if(n>=2&&n<=100000)
    {
    	while(n--)
    	{
    		std::cin>>x[l];
    		std::cin>>y[l];
    		l++;
    	}
    	for(int i=0; i<l; i++)
    		h[i]=l-1;
    	for(int i=0; i<l; i++)
    		{
    			for(int j=0; j<l; j++)
    			{
    				if(j==i)
    				j++;
    				if(x[i]==y[j])
    					h[j]++;
    				else
    					a[j]++;
    			}
    		}
    		for(int i=0;i<l;i++)
    			std::cout<<h[i]<<" "<<a[i]<<"\n";
    }
    return 0;
}
