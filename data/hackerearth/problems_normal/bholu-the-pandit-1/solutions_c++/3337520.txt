#include<bits/stdc++.h>
#define ll long long int
#define MAX 10000009
#define mod 1000000007
using namespace std;

ll factor(ll a,ll b,ll c)
{
    ll i,N,x,y,z,ans;
    ans=1;
    N=max(a,b);
    N=max(N,c);
    for(i=2; i*i<=N; i++)
    {
        x=0;
        y=0;
        z=0;
        if(a%i==0)
        {
            while(a%i==0)
            {
                a=a/i;
                x++;
            }
        }

        if(b%i==0)
        {
            while(b%i==0)
            {
                b=b/i;
                y++;
            }
        }

        if(c%i==0)
        {
            while(c%i==0)
            {
                c=c/i;
                z++;
            }
        }

        x=min( min(x,y),z );
        ans*=powl(i,x);
    }
    if(a==b && b==c && a!=1)
        ans*=a;
    return ans;
}

int main()
{
    ll t,T,n,m,ans1,ans2,a,b,c;
    scanf("%lld",&T);
    for(t=1;t<=T;t++)
    {
        scanf("%lld %lld %lld",&a,&b,&c);
        ans1=factor(a,b,c);

        a=a/ans1;
        b=b/ans1;
        c=c/ans1;

        ans2=(a%mod*b%mod*c%mod)%mod;

        printf("%lld %lld\n",ans1,ans2);
    }
    return 0;
}
