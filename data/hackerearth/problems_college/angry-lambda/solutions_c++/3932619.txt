#include <iostream>
using namespace std;

int main()
{
    int t,n,k,count=0,num;;
    cin>>t;
    while(t--){
    	cin>>n>>k;
    	count=0;
    	while(n--){
    		cin>>num;
    		if(num<=0)
    		   count++;
    	}
    	if(count>=k)
    	  cout<<"NO"<<endl;
    	else
    	cout<<"YES"<<endl;
    }
    return 0;
}
