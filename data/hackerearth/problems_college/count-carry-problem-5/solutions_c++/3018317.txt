#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	int tc;
	cin>>tc;
	while(tc--)
	{
		char a[1000005],b[1000005];
		cin>>a>>b;
		int ans,l1,l2,carry=0,cnt=0;
		l1=strlen(a)-1;
		l2=strlen(b)-1;
		while(l1>=0 && l2>=0)
		{
			ans=a[l1]+b[l2]+carry-96;
			carry=ans/10;
			l1--;
			l2--;
			//cout<<ans<<carry<<endl;
			if(carry>0) cnt++;
		}
		while(l1>=0)
		{
			ans=a[l1]+carry-48;
			carry=ans/10;
			if(carry>0) cnt++;
			l1--;
		}
		while(l2>=0)
		{
			ans=b[l2]+carry-48;
			carry=ans/10;
			if(carry>0) cnt++;
			l2--;
		} 
		if(cnt==0) cout<<"No carry operation\n";
		else if(cnt==1) cout<<"1 carry operation\n";
		else cout<<cnt<<" carry operations\n";
	}
	return 0;
}
