#include<iostream>
using namespace std;
main()
{
	int t,n,i;
	long int m,a;
	long long int s,x;
	cin>>t;
	while(t--)
	{
		s=0;
		cin>>n>>m>>x;
		for(i=0;i<n;i++)
		{
			cin>>a;
			s=s+a;
		}
	    s=((n+1)*x)-(s);
	    if(s<1)
	    {
	    	cout<<"1"<<endl;
		}
	    else if(s<=m)
	    {
	    	cout<<s<<endl;
		}
		else
		{
			cout<<"Impossible"<<endl;
		}
	}
}
