#include <iostream>
using namespace std;

int main()
{
	
    long int i,N,temp,X;
    int M;
    cin  >> N;
    long int A[100000];
    for (i = 0; i < N; i++) {
            cin >> A[i];
        }
    cin >> M;
    while(M--) {
    	cin >> X;
		for(i=0;i< N;i++) {
    		if( A[i]>X ) {
				A[i]=A[i]-1;
			}
    	}
    }
    for(i=0;i< N;i++) {
	    cout << A[i] << " ";
    }
    return 0;
}
