using namespace std;


int setS[NAX];

int findS(int a){
   // cout<<"debug\n";
    while(setS[a]!=-1)
        a = setS[a];
    return a;
}

void unionS(int a,int b){
    if(setS[a]>setS[b]){
        setS[b]=setS[a]+setS[b];
        setS[a]=b;
    }
    else{
        setS[b]=setS[b]+setS[a];
        setS[b]=a;
    }
}

void solve(){
    int n;
    cin>>n;
    //int arr[n];
    memset(setS,-1,sizeof(setS));
    int q;
    cin>>q;
    while(q--){
        int x,y;
        cin>>x>>y;
        x--;y--;
        int r1=findS(x);
        int r2=findS(y);
        if(r1!=r2)
            unionS(r1,r2);
    }
    cout<<count(setS,setS+n,-1)<<endl;
}


int main(){
    solve();

    return 0;
}
