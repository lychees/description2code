#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
     {
     	int n;
     	cin>>n;
     	int a[n][3];
     	int dp[n][3];
     	for(int i=0;i<n;i++)
     	 {
     	 	for(int j=0;j<3;j++)
     	 	   cin>>a[i][j];
     	 }
     	for(int i=0;i<3;i++)
     	 {
     	 	dp[0][i]=a[0][i];
     	 }
     	for(int i=1;i<n;i++)
     	 {
     	 	dp[i][0]=a[i][0]+min(dp[i-1][1],dp[i-1][2]);
     	 	dp[i][1]=a[i][1]+min(dp[i-1][0],dp[i-1][2]);
     	 	dp[i][2]=a[i][2]+min(dp[i-1][1],dp[i-1][0]);
     	 }
     	cout<<min(dp[n-1][0],min(dp[n-1][1],dp[n-1][2]))<<"\n";
     }
}
