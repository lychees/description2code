#include<iostream>
 using namespace std;
 int calc(int a[],int n,int d)
 	 {
	 int x=0,t,h;
 		for(int i=0;i<n-d;i++)
 		{
 				t=a[i+d]-a[i];
 				h=a[i+d]+a[i];
 				if(t%d==0 || h%d==0)
 					x++;
 		}
 		return x;
 	 }
 int main()
 {
	 int a,b;
	 cin>>a>>b;
	 int z[a];
	 for(int i=0;i<a;i++)
	 {
		 cin>>z[i];
	 }
	 cout<<calc(z,a,b);
 }
