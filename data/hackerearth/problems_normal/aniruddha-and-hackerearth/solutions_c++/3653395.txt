#include <iostream>
#include <algorithm>

using namespace std;
/*
f(H) = f(H)+f(E);
f(E) = f(H);
*/
#define N 1000001
//#define N 10
#define MOD 1000000007

int
main ()
{
  long long parrh;//[N];
  long long parre;//[N];
  long long narrh;
  long long narre;
  long long arr[1000001];
  long long divya[1000001];

  parrh = 1;
  parre = 1;
  arr[1] = 2;
  divya[1] = 2;
  for (long long i = 2; i < N; i++)
    {
      narre = parrh;
      narrh = (parrh + parre)%MOD;
      arr[i] = (narrh + narre)%MOD;
      divya[i] = (divya[i-1] + arr[i])%MOD;
      parrh = narrh;
      parre = narre;
    }

  long long T = 0;
  long long NN = 0;
  cin >> T;
  while (T--)
    {
      cin >> NN;
      cout << divya[NN] << endl;
    }
}
