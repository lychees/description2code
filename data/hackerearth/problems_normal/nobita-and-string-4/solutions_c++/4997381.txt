#include <iostream>
#include<string>
#include<vector>
#include<sstream>
#include<iterator>
#include<stdio.h>
using namespace std;


int main()
{
	int tc;
	cin>>tc;
	cin.ignore();
	while(tc--)
	{
		string s;
		getline(cin,s);
		istringstream stream(s);
		vector<string> words( (istream_iterator<string>(stream )),istream_iterator<string>());
		for(vector<string>::reverse_iterator rit=words.rbegin();rit!=words.rend();rit++)
		{
			cout<<*rit<<" ";
		}
		cout<<"\n";
	}
}
