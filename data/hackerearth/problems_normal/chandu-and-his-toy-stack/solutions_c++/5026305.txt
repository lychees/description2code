#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long test,n,x,y,ans,i;
	cin >> test;
	while(test--)
	{
		ans=0;
		cin >> n >> x >> y;
		long long initial[n],final[n];
		for(i=0;i<n;i++)
			cin >> initial[i] >> final[i];
		sort(initial,initial+n);sort(final,final+n);
		for(i=0;i<n;i++)
		{
			if(final[i]>initial[i])
				ans=ans+((final[i]-initial[i])*(x));
			else if(final[i]<initial[i])
				ans=ans+((initial[i]-final[i])*y);
		}
		cout << ans << endl;
	}
	return 0;
}
