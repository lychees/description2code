#include <iostream>
using namespace std;
 int t,w,h;
 bool matrix[1001][1001];
 int sum;

int main()
{
   //Pre Process
   int mode1=0,mode2;
   for(int i=0;i<1000;i++)
   {
   	mode2=0;
   	for(int j=0;j<1000;j++)
   	{
   		if(mode1<2)
   		{
   			if(mode2<2)
   			matrix[i][j]=true;
   			else
   			matrix[i][j]=false;
   		}
   		else
   		{
   			if(mode2<2)
   			matrix[i][j]=false;
   			else
   			matrix[i][j]=true;
   		}
   		mode2=(mode2+1)%4;
   	}
   	mode1=(mode1+1)%4;
   }
   
    cin>>t;
    while(t--)
    {
    	cin>>w>>h;
    	sum=0;
    	for(int i=0;i<h;i++)
    	{
    		for(int j=0;j<w;j++)
    		{
    			if(matrix[i][j]) sum++;
    		}
    	}
    	cout<<sum<<"\n";
    }
    
    
    return 0;
}
