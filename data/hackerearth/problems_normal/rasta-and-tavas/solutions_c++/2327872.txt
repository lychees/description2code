#include <bits/stdc++.h>

#define fs first
#define sc second
#define mp make_pair
#define pb push_back
#define sz(s) ((int) s.size ())
#define all(s) s.begin (), s.end ()

using namespace std;

const int N = 1000001;
const double eps = 1e-11;
const int inf = (int) 2e9;
const int mod = (int) 1e9 + 7;

int p[N];
int s[N];

int main()
{
  int n, k;
  scanf ("%d %d", &n, &k);
  int ans = 0;
  for (int i = 2; i <= n; i++) {
    if (!p[i]) {
      s[i] += i;
      for (int j = i + i; j <= n; j += i) {
        s[j] += i;
        p[j] = 1;
      }
    }
    if (s[i] == k) {
      ans++;
    }
  }
  printf ("%d\n", ans);
  return 0;
}
