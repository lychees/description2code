#include <iostream>
#include <cstdlib>
#include <cstring>
using namespace std;

long long int A[201][201];
int main()
{
    //ios::sync_with_stdio(false);
    long long int T,N,K,i,j;
    cin >> T;
    while(T--){
          cin >> N >> K;
          //int A[N+1][N+1];
          memset(A,0,sizeof(A));
          for(i = 0;i<=N;++i) A[0][i] = A[i][i] = 1;
          
          for(i = 0;i<=N;++i){
              for(j = i;j>=K;--j){
                  A[i][j] = A[i-j][j]+A[i][j+1];
              }
          }
          cout << A[N][K] << '\n';
    }         
}
