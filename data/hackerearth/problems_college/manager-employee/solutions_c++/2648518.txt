#include<bits/stdc++.h>
using namespace std;
int main()
{
  int t;  cin>>t;
  while(t--)
  { int n,i;	cin>>n;	  int a[n],b[n];
  	for(i=0;i<n;i++)
  		cin>>a[i];
  	sort(a,a+n);
  	for(i=0;i<n;i++)
  	     b[i]=a[i];
  	reverse(b,b+n);
  	for(i=0;i<n;i++)
  	    a[i]=a[i]+b[i];
  	  cout<<*max_element(a,a+n)-*min_element(a,a+n)<<endl;
  }
  return 0;
}
