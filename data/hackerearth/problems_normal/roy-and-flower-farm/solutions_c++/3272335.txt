#include<cstdio>
using namespace std;
int max(int a,int b)
{
    return a>b?a:b;
}
int dp[10001][101];
int main()
{
    int T;
    scanf("%d",&T);
    int X[1001],Y[1001];
    while (T--)
    {
	int N,P;
	scanf("%d %d",&N,&P);
	for (int i=0;i<N;i++)
	    scanf("%d %d",&X[i],&Y[i]);
	for (int cost=1;cost<=P;cost++)
	    for (int i=0;i<N;i++)
		dp[cost][i]=0;
	//printf("deb1");
	for (int cost=1;cost<=P;cost++)
	{
	 //   printf("deb2 %d\n",cost);
	    for (int i=N-1;i>=0;i--)
	    {
		int mx=0;
		if (cost-Y[i]>=0) {
		    for (int j=i+1;j<N;j++) {
			mx=max(mx,dp[cost-Y[i]][j]);
		    }
		    dp[cost][i]=mx+X[i]-Y[i];
		}
		else
		    dp[cost][i]=0;
	    }
	}
	int mx=0,mxct=0;
	for (int i=0;i<=P;i++) {
	    for (int j=0;j<N;j++) {
		if (mx<dp[i][j]) {
		    mx=dp[i][j];
		    mxct=i;
		}
	    }
	}
	printf("%d %d\n",mxct,mx+P);
    }
    return 0;
}
