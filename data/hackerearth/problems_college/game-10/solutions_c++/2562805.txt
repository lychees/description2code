#include <iostream>
using namespace std;

int main() {
	// your code goes here
	
	int t;
	long long n,element,result;
	cin>>t;
	while(t--)
	{
		cin>>n>>element;
		if(element>=n/2)
		{
		     result=(n-element)*2-2;
		 }
		   else
		   {
		     result=(element+1)*2-1;
		   }
		cout<<result<<endl;
	
	}
	return 0;
}
