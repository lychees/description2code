#include<bits/stdc++.h>
using namespace std;
bool flag=false;
int main()
{
    int n,l;
    cin>>n;
    char ch;
    string str[100];
    for(int i=0;i<n;i++){
        cin>>str[i];

    }
    for(int j=0;j<n;j++){
        reverse(str[j].begin(),str[j].end());
         l=str[j].length();
        for(int k=j+1;k<n;k++){
            if(str[j]==str[k]){
                flag=true;
                ch = str[j].at(l/2);
                break;

            }
            if(flag==true)
            break;
        }

    }
    cout<<l<<" "<<ch;
    return 0;
}
