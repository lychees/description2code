int a[15];

int main(){
	int n;
	cin >> n;
	for (int i = 0; i < n; ++i){
		cin >> a[i];
	}
	int ans = INF;
	int nn = 1 << n;
	for (int i = 0; i < nn; ++i){
		int mask1 = i;
		int S1 = 0, S23 = 0;
		int nnn = 0;
		int aa[15];
		for (int j = 0; j < n; ++j){
			if (!(mask1&(1 << j))){
				S1 += a[j];
			}
			else aa[nnn++]=a[j], S23 += a[j];
		}
		int nnnn = 1 << nnn, S2 = 0, S3 = 0;
		for (int k = 0; k < nnnn; ++k){
			int mask2 = k;
			S2 = 0; S3 = 0;
			for (int jj = 0; jj < nnn; ++jj){
				if (!(mask2 & (1 << jj))){
					S2 += aa[jj];
				}
				else S3 += aa[jj];
			}
			int b[3];
			b[0] = S1; b[1] = S2; b[2] = S3;
			sort(b, b + 3);
			int mx = b[2];
			ans = min(ans, mx);
		}
	}
	cout << ans; el;
	return 0;
}
