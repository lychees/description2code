#include<bits/stdc++.h>
using namespace std;
int n,sizer=0;
bool isprime[10000007];
int G[1000000];
void sieve()
{	long long int i;
	long long j;
	for(i=2;i*i<=10000001;i++)
	{
		if(!isprime[i])
		{
			for(j=i*i;j<=10000001;j+=i)
			{
				isprime[j]=true;
			}
		}
	}
	for(i=2;i<=10000001;i++)
	{
		if(!isprime[i])
		{
			G[sizer]=i;
			sizer++;
		}
	}


}

int main()
{	long long count1=0;
	long long int i,j,fl,test;
	sieve();
	scanf("%d",&n);
	for(i=0;G[i]<=n &&i <sizer;i++)
	{
		count1++;
	}
	printf("%lld\n",(count1*(count1-1))/2 );

	return 0;
}
