#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int test;
	cin>>test;
	while(test--)
	{
		string s;
		cin>>s;
		int l=s.length(),nl;
		int i,j,flag;
		string t;
		for(i=l;;i++)
		{
			flag=1;
			t=s;
			for(j=(i-l-1);j>=0;j--)
			{
				t+=s.at(j);
			}
			//~ cout<<t<<endl;
			nl=i;
			for(j=0;j<nl/2;j++)
			{
				if(t.at(j)!=t.at(nl-1-j))
				{
					flag=0;
					break;
				}
			}
			if(flag)
			break;
		}
		cout<<nl<<endl;
	}
	return 0;
}
