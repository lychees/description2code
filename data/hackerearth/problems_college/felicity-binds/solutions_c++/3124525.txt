#include <iostream>
#include<algorithm>
using namespace std;

int main()
{

   int t; cin>>t;
   while(t--)
   {
   	  long int n;  cin>>n;
   	  long int m[n],w[n];   long long int s=0;
   	  
   	  for(int i=0;i<n;i++)
   	  cin>>m[i];
   	  
   	  for(int i=0;i<n;i++)
   	    cin>>w[i];
   	  
   	  sort(m,m+n);
   	  sort(w,w+n);
   	    for(int i=0;i<n;i++)
   	    {   
   	    	s=s+m[i]*w[i];
   	  
   	      }
   	     cout<<"\n"<<s; 
   	
   }
return 0;
 
}
