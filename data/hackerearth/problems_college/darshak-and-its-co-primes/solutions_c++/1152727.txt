#include<iostream>

using namespace std;

int hcf(int a, int h) 
{
    int temp; 
    while(1)
    {
        temp = a%h;
        if(temp==0)
        return h;
        a = h;
        h = temp;
    }
} 

int main() 
{
    long long int c,d,gcd,t;
    cin>>t;
    while(t--)
    {
    cin>>c>>d;	

    gcd=hcf(c,d);
   
    if(gcd==1) 
    {
        cout<<"Is a Co-Prime\n";
    }
    else
    {
        cout<<"Not a Co-Prime\n";
    }         
	}
	return 0;
}
