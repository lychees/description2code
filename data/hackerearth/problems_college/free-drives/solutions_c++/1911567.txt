#include<iostream>
using namespace std;

long long comb(int n, int r) {
    if(r > n / 2) r = n - r; // because C(n, r) == C(n, n - r)
    long long ans = 1;
    int i;

    for(i = 1; i <= r; i++) {
        ans *= n - r + i;
        ans /= i;
    }

    return ans;
}


int main()
{
    long t;
    cin >> t;
    while(t--){
        long n;
        cin >> n;
        if(n==1)
            cout << 0 <<endl;
        else
        cout << comb(n,2)<<endl;
    }
    return 0;
}
