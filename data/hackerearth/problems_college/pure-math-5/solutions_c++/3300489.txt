#include <iostream>
using namespace std;

int main()
{
long long int n,m,k,cnt,i,j;
long long int size;
int test;
cin>>test;
while(test--)
{
    cnt=0;
    cin>>size;
 
    for(j=2;j<=size;j++)
    {
        i=1;
        if(((j*j)-1) % i == 0 && ((i*i)-1) % j==0)
        {
            cnt++;
            n=j;
            m=((j*j) -1)/i;
            while(m<=size)
            {
                cnt++;
                k=m;
                m=((k*k)-1)/n;
                n=k;
            }
        }
    }
cout<<cnt<<endl;
}
return 0;
}
