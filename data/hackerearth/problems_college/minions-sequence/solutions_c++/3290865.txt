#include <iostream>
using namespace std;

int t,n,m,x;
int arr[1000005];
int b[1000005];

int main()
{
    scanf("%d",&t);
    
    while(t--){
    	x=1;
    	scanf("%d",&n);
    	for(int i=0;i<n;i++) scanf("%d",arr+i);
    	b[0] = arr[0];
    	scanf("%d",&m);
    	for(int i=1;i<n;i++){
    		if(arr[i]<b[0]) b[0] = arr[i];
    		else if(arr[i]>b[x-1]){
    			if(x<m) b[x++] = arr[i];
    		} else{
    			int yy = lower_bound(b,b+x,arr[i])-b;
    			b[yy] = arr[i];
    		}
    	}
    	if(x>=m) 
    		printf("%d\n",b[x-1]);
    	else
    		printf("-1\n");
    	
    }
    
    return 0;
}
