#include<bits/stdc++.h>
using namespace std ;
int main()
{
	int t ,s , n ,i ;
	cin >> t ;
	int a[100] ;
	while ( t-- )
	{
		cin >> n ;
		for( i = 0 ; i < n ; i++)
		{
			cin >> a[i] ;
		}
		sort(a,a+n) ;
		i = 0 ;
		s = 0 ;
		while( i < n-1 )
		{
			s = s + a[i+1] - a[i] ; 
			i++ ;
		}
		cout<<2*s<<"\n" ;
	}
	return 0 ;
}
