#include<bits/stdc++.h>
#define ll long long int
using namespace std;
main()
{
    ll x,i;
    string s;
    cin>>s;
    x=s.length();
    ll arr[x];
    if(((s[x-1]-48)%2)==0)
        arr[x-1]=1;
    else
        arr[x-1]=0;
    for(i=x-2;i>=0;i--)
    {
        if((s[i]-48)%2==0)
            arr[i]=arr[i+1]+1;
        else
            arr[i]=arr[i+1];
    }
    for(i=0;i<x;i++)
    {
        cout<<arr[i]<<" ";
    }
    return 0;
}
