#include<iostream>
using namespace std;
int main(){
 long long max=0, i, x;
 string s, temp;
 while (getline(cin, s)){
   
    x=0;
    for(i=0;i<s.length();i++)
        x+=(s[i]-'0');
    if(x>max){
        max=x;
        temp=s;
    }


 }

 cout<<temp<<endl;






 return 0;
}
