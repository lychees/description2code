#include <iostream>
using namespace std;

int main()
{
	long long int t;
	unsigned long long int a,b,c;
	cin>>t;
	while(t--)
	{
		cin>>a>>b>>c;
		int flag=0;
		if(a+b>c && b+c>a && a+c>b)
			flag=1;
		if(flag==0)
		{
			cout<<"NO"<<endl;
		}
		
		else if(a==b && b==c && a==c)
		{
			cout<<"YES. It is a equilateral triangle."<<endl;
			
		}
		
		else if(a==b || b==c || a==c)
		{
			cout<<"YES. It is a isosceles triangle."<<endl;
			
		}
		
		else 
		cout<<"YES. It is a scalene triangle."<<endl;
	}
}
