#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	int t,l1,l2,num,hold;
    string a,b,holding;
    cin>>t;
    bool mode;
    while(t--)
    {
    	cin>>a>>b;
    	holding=a;
		   a.erase(remove_if(a.begin(), a.end(), [](char c){ return !isalpha(c);} ),a.end());
		   b.erase(remove_if(b.begin(), b.end(), [](char c){ return !isalpha(c);} ),b.end());
		   l1=a.length();
		   l2=b.length();
		   for(int i=0;i<=(l1-l2);i++)
		   {
		   		num=a[i]-b[0];
		   		if(num<0)
		   		num+=26;
		   		mode=true;
		   		for(int j=1;j<l2;j++)
		   		{
		   			hold=a[i+j]-b[j];
		   			if(hold<0)
		   			hold+=26;
		   			if(hold==num)
		   			continue;
		   			mode=false;
		   			break;
		   		}
		   		
		   		if(mode)
		   		break;
		   }
		   for(int i=0;i<holding.length();i++)
		   {
		   	if(isupper(holding[i]))
		   	{	
		   		hold=(int)holding[i]-num;
		   		if(hold<(int)'A')
		   		hold+=26;
		   		holding[i]=(char)hold;
		   	}
		   	else
		   	 if(islower(holding[i]))
		   	{
		   		hold=(int)holding[i]-num;
		   		if(hold<(int)'a')
		   		hold+=26;
		   		holding[i]=(char)hold;
		   	}
		   }
		   cout<<holding<<"\n";
  }
   
    return 0;
}
