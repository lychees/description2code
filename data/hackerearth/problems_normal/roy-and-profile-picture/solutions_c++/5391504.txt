#include<iostream>
using namespace std;
int main()
{
	int L;
	cin>>L;
	int N;
	cin>>N;
	int W[N],H[N],i;
	for(i=0;i<N;i++)
	{
		cin>>W[i];
		cout<<" ";
		cin>>H[i];
	}
	for(i=0;i<N;i++)
	{
		if(W[i]<L || H[i]<L)
		cout<<"UPLOAD ANOTHER"<<endl;
	
		else if(W[i]>=L && H[i]>=L)
		{
			if(W[i]==H[i])
			cout<<"ACCEPTED"<<endl;
			else
		//(W[i]>L || H[i]>L)
			cout<<"CROP IT"<<endl;
		}
	}
	return 0;
}
