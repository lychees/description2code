#include<iostream>
#include<cstdlib>
#include<cstring>
#include<cctype>
using namespace  std;

int main()
{
    string s ;
    cin >> s;
    int Q;
    cin >> Q;
    long long a, b, l = s.length();
    
    for(int i = 0;i <Q;i++)
    {
        cin >> a >> b;
        a--;
        b--;
        a = a % l;
        b = b % l;
        if(tolower(s[a]) == tolower(s[b]))
        {
            cout << "Yes\n"; 
        }
        
        else
            cout << "No\n";
    }
    return 0;
}
