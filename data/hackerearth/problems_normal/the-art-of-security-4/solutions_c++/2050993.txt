int n,one,two,ways;
int a[10],tot=0;
void solve(int ind)
{
      if(ind==n)
      {
            ways++;
            return;
      }
      if(one+a[ind]<=two)
      {
            one+=a[ind];
            solve(ind+1);
            one-=a[ind];
      }
      two+=a[ind];
      solve(ind+1);
      two-=a[ind];
      
      return;
}

}
