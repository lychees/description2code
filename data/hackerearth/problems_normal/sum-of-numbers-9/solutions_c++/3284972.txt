#include <iostream>
#include <utility>
#include<algorithm>
#include<map>
#define ll long long
using namespace std;
int main()
{
   ll t,n,temp,sum;
   cin>>t;
   while(t--){
       cin>>n;
       ll a[n+2],total=1<<n,flag=0,s;
       for(ll i=0;i<n;i++){
       cin>>a[i];
       }
       cin>>s;
         for(ll i=1;i<total;i++){
           sum=0;
           for(ll j=0;j<n;j++){
              if(i&(1<<j))sum+=a[j];
           //cout<<endl;
            //cout<<" "<<sum<<endl;
           if(sum==s){
               flag=1;
               break;
           }
            
          
       }
         
    } 
       if(flag)cout<<"YES"<<endl;
         else cout<<"NO"<<endl;
     
    }
    return 0;
}
