#include <iostream>
using namespace std;
typedef long long int ll;
ll f[1010001];
ll calc(ll a, ll b, ll mod) {
	ll ans = 1;
	while(b > 0) {
		if(b%2 == 1) {
			ans = (ans * a)%mod;
		}
		a = (a * a)%mod;
		b/=2;
	}
	return ans;
}
ll get(ll n, ll r, ll mod) {
	ll ans = f[n];
	ans = ans * calc(f[r], mod - 2, mod);
	ans%=mod;
	ans*=calc(f[n - r], mod - 2, mod);
	ans%=mod;
	return ans;
}
int main()
{
	ll max = 1010000;
	ll i;
	ll mod = 1000000007;
	f[0] = 1;
	for(i = 1; i <= max; i++) {
			f[i] = (i * f[i - 1])%mod;
	}
    ll t;
    cin >> t;
    while(t--) {
    	ll k, m;
    	cin >> k >> m;
    	ll n = k * k + m - 2;
    	ll ans = get(n, m, mod);
    	cout << ans << endl;
    }
    return 0;
}
