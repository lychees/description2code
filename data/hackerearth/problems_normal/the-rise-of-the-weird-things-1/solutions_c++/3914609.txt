#include<bits/stdc++.h>

using namespace std;

int main()
{
	int n,temp;
	cin>>n;
	vector<int> even_list;
	vector<int> odd_list;
	long long int even_sum=0,odd_sum=0;
	while(n--)
	{
		cin>>temp;
		if(temp%2==0)
		{
			even_list.push_back(temp);
			even_sum+=temp;
		}
		else
		{
			odd_list.push_back(temp);
			odd_sum+=temp;
		}
	}
	sort(even_list.begin(),even_list.end());
	sort(odd_list.begin(),odd_list.end());
	for(vector<int>::iterator it = even_list.begin();it!=even_list.end();it++)
		cout<<*it<<" ";
	cout<<even_sum<<" ";
	for(vector<int>::iterator it = odd_list.begin();it!=odd_list.end();it++)
		cout<<*it<<" ";
	cout<<odd_sum<<endl;
	return 0;
}
