bool chck(string a,string b){
	if(a.length()>b.length())
	swap(a,b);
	int i;
	for(i=0;i<a.length();i++){
		if(a[i]!=b[i])
		return false;
	}
	if(i==a.length())
	return true;
}

				flag|=chck(v[i],v[j]);
			}
		}

		printf("NO\n");
		else
		printf("YES\n");
	}
	return 0;
}
