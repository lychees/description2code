#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n,k,l,q,x,y,i,j,flag;
	map<int,string > m;
	map<int,string>::iterator it;
	string s;
	cin>>n>>k>>l>>q;
	for(i=0;i<n;i++)
	{
		cin>>s;
		y=0;
		for(j=0;j<k;j++)
		{
			cin>>x;
			y = y*97+x+97;
		}
		m.insert(make_pair(y,s));
	}
	for(i=0;i<q;i++)
	{
		y=0;
		flag=1;
		for(j=0;j<k;j++)
		{
			cin>>x;
			y = y*97 +x +97;
			if(x>l)
				flag=0;
			
		}
		it = m.find(y);
		if(flag && it!=m.end())
		{
			cout<<it->second<<endl;
		}
		else
			cout<<"You cant fool me :P\n";
	}
    return 0;
}
