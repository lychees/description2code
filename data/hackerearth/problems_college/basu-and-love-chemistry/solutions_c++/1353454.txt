#define MOD 1000000007
#define MAX 200050

int main()
{
    ios::sync_with_stdio(false);
    int t;
    t = 1;
    while (t--) {
        string s;
        cin >> s;
        int n = s.size();
        long long dp[MAX] = {};
        dp[0] = s[0] - '0';
        for (int i = 1; i < n; i++) {
            dp[i] = (dp[i-1] * 10 + (s[i]-'0')*(i)) % MOD;
            dp[i] += (s[i] - '0');
            dp[i] %= MOD;
        }
        long long sum = 0;
        for (int i = 0; i < n; i++) {
            sum += dp[i];
            sum %= MOD;
        }
        printf("%lld\n", sum);
    }
}
