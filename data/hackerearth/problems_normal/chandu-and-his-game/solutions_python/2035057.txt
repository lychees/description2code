n, k = map(int, raw_input().split(" "))

x = map(int, raw_input().strip().split(" "))
y = map(int, raw_input().strip().split(" "))

total_moves = 0
# Check for 4 corners minimums and add to answer
for i in xrange(len(x)):
	
	# 0,0
	l_t = x[i] + y[i] - 2
	# 0, n
	r_t = x[i] + n-y[i] - 1
	# n, 0
	l_b = n - x[i] + y[i] - 1
	# n, n
	r_b = n - x[i] + n - y[i]
	
	moves = reduce(min, [l_t, r_t, l_b, r_b])
	
	total_moves += moves

print total_moves
