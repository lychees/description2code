for _ in range(int(raw_input())):
    a,b=raw_input().split()
    x=min(len(a),len(b))
    #print a,b;
    c=0
    for i in range(x):
        if(ord(a[i])-48+ord(b[i])-48>9):
            c+=1
    if c>1:
        print str(c)+" carry operations"
    elif c==1:
        print "1 carry operation"
    else:
        print "No carry operation"
