#include<bits/stdc++.h>
using namespace std;
vector<int> arr(1000001);
int main()
{
    int T;
    long int N,i,j,k;
    arr[1]=1;
    for(i=2;i<=1000;i++)
    {
        for(j=2;pow(i,j)<=1000000;j++)
        {
            k=pow(i,j);
            arr[k]=1;
        }
    }
    cin>>T;
    while(T--)
    {
        cin>>N;
        int flag=0;
        for(i=1;i<=N/2;i++)
        {
            if(arr[i]==1 && arr[N-i]==1)
            {
                flag=1;
                break;
            }
        }
        if(flag==1)
            cout<<"Yes"<<endl;
        else
            cout<<"No"<<endl;

    }
}
