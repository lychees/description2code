#include <iostream>
using namespace std;

int main()
{
	long long n,q,p;
	int l, r;
	cin>>n>>q;
	int count;
	long long a[n];	
	for(int i=1;i<=n;i++)
	{cin>>a[i];
	}
	while(q--)
	{		
		cin>>l>>r;
		p = a[r];
		
		count = a[r];
		for(int i=r-1;i>=l;i--)
		{count += a[i];
			p-= a[i];	
		}
		cout<<count<<" "<<p<<"\n";
	}
    return 0;
}
