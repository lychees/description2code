#include <bits/stdc++.h>
using namespace std;

typedef long long LL;

#define PII       pair<int,int>
#define all(c)    c.begin(),c.end()
#define sz(c)     (int)c.size()
#define clr(c)    c.clear()
#define pb        push_back
#define mp        make_pair
#define cin(x)    scanf("%d",&x)
#define MOD		1000000007
#define EPS		1E-10

void solve()
{
	int a,b,c,d;
	cin >> a >> b >> c >> d;
	if(a > b)
		swap(a,b);
	if(c > d)
		swap(c,d);

	int x = max(a,c);
	int y = min(b,d);
	if(x == y)
		printf("Point\n");
	else if(x > y)
		printf("Line\n");
	else
		printf("Nothing\n");
}

int main()
{
	int t;
	cin >> t;
	while(t--)
		solve();
	return 0;
}
