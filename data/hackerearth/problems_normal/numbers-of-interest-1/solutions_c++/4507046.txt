#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    while(t--){
    	int n, x, y;
    	cin>>x>>y>>n;
    	
    	long long int dp[51] = {0};
    	dp[0] = 0;
    	for(int i=1; i<=x; i++){
    		dp[i] = y;
    	}
    	for(int i=x+1; i<=n; i++){
    		
    		for(int j=i-1, k=x; k != 0; j--, k-- ){
    			dp[i] += dp[j];
    		}
    	}
    	cout<<dp[n]<<endl;
    }
    return 0;
}
