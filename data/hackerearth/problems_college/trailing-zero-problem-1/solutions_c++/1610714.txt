#include <iostream>
#include <math.h>
#define PI 3.14
using namespace std;
int findTrailingZeros(int  n)
{
    int count = 0;
    for (int i=5; n/i>=1; i *= 5)
          count += n/i;
    return count;
}
int main()
{
    int t;
    cin>>t;

    {
    	int i=5,n,k=0;
    	cin>>n;
    	int64_t re=floor((log(2*PI*n)/2 + n*(log(n) - 1) + 1/(12*n))/log(10)) +1;
    	/*while(n/i)

    	}*/
    	k=findTrailingZeros(n);

}
