#include <algorithm>
#include <cmath>
#include <iomanip>
#include <iostream>
#include <map>
#include <string>
#include <vector>

using namespace std;

int main() {
  int n, k;

  cin >> n >> k;
  long long tt = 1;

  for (int i = 1; i <= k; i++) {
    for (int j = 1 + tt; j <= n; j++)
      cout << j << " ";
    for (int j = 1; j <= tt && j <= n; j++)
      cout << n << " ";
    cout << "\n";
    tt = tt * 2;
  }
}
