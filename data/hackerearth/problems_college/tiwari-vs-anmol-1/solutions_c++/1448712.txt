#include <bits/stdc++.h>
using namespace std;
vector<int>prime;
long long sum[1000001]={0};
int arr[1000001]={0};
void sieve(){
	arr[0] = 1;
	arr[1] = 1;
	for(int i = 4;i<=1000000;i+=2){
		arr[i] = 1;
	}
	for(int i = 3;i<=1000000;i+=2){
		for(int j = 2*i;j<=1000000;j+=i){
			arr[j] = 1;
		}
	}
	for(int i = 2;i<=1000000;i++){
		if(!arr[i])prime.push_back(i);
	}
}
void pre(){
	for(int i = 4;i<=1000000;i++){
		if(!arr[i])continue;
		for(int j = 0;j<prime.size();j++){
			if(prime[j]*prime[j]>i)break;
			if(i%prime[j]==0){
				sum[i] = prime[j];
				break;
			}
		}
	}
	for(int i = 1;i<=1000000;i++) sum[i] = (sum[i]+sum[i-1])%1000000007;
}
int main()
{
    sieve();
    //cout<<prime.size()<<endl;
    pre();
    int test;
    scanf("%d",&test);
    while(test--){
    	int l,r;
    	scanf("%d %d",&l,&r);
    	printf("%lld\n",sum[r]-sum[l-1]);
    }
    return 0;
}
