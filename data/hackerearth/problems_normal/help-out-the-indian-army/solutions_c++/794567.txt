#include <iostream>
#include<algorithm>
using namespace std;
typedef long long ll;
struct pr
{
	ll x,p;
}a[100000];
int n;
ll s,e;
bool fn(pr x,pr y)
{
	return x.x<y.x;
}
ll fun(int p,ll s,ll e)
{
	if(s>=e)return 0;
	if(p==n)return e-s;
	if(e<=a[p].x)return e-s;
	if(s>=a[p].p)return fun(p+1,s,e);
	if(s<a[p].x)return a[p].x-s+fun(p+1,a[p].p,e);
	return fun(p+1,a[p].p,e);
}
int main()
{
    int i,j,k;
    ll tm,tm1;
    cin>>n>>s>>e;
    for(i=0;i<n;i++){
    	cin>>tm1>>tm;
    	a[i].x=tm1-tm;
    	a[i].p=tm1+tm;
    }
    sort(a,a+n,fn);
    cout<<fun(0,s,e);
    return 0;
}
