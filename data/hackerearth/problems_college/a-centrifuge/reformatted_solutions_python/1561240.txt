times = input()
for _ in range(times):
    inp = raw_input()
    lst = inp.split()
    capa = int(lst[0])
    num = int(lst[1])

    if num == 0:
        print 0
    elif num < capa:
        print 10
    else:
        mult = num/capa
        diff = num % capa

        if diff == 0:
            extra = 0
        elif diff <= int(capa/2):
            extra = 5
        else:
            extra = 10
        time = 10*mult+extra
        print time
