#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string s;
		cin>>s;
		int a[26];
		for(int i=0;i<26;i++)
		a[i]=0;
		for(int i=0;i<s.length();i++)
		{
			a[s[i]-'a']++;
		}
		int temp=a[0];
		bool check=true;
		for(int i=0;i<26;i++)
		{
			if(a[i]!=temp)
			check=false;
		}
		if(check)
		cout<<"Yes"<<endl;
		else
		cout<<"No"<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
