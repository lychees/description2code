#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n,x,s;
	vector<int> vec;
    cin >> n;
    for(int i=0;i<n;i++)
   	{
   		cin >> x;
   		vec.push_back(x);
   	}
   	sort(vec.begin(),vec.end());
   	cin >> s;
   	int sum = 0;
   	int counter = 0;
   	for(int i=n-1;i>=0;i--)
   	{
   		sum+=vec[i];
   		counter++;
  		if(sum >= s)
   			break;
   	}
   	cout << counter <<endl;
   	
    return 0;
}
