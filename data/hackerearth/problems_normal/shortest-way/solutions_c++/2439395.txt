#include<bits/stdc++.h>

using namespace std;

#define FOR(i,a,b) for(i=a;i<b;++i)
#define FORD(i,a,b) for(i=a;i>=b;--i)
#define infi 1000000007
#define pb push_back
#define mp make_pair
#define ll long long

ll a, b;
int main()
{
    int t;
    cin >> t;
    while(t--) {
        cin >> a >> b;
        ll ans = 0;
        while(a != 1 || b != 1) {
            if(a == 1 || b == 1) {
               // cout <<"   "<<ans <<" "<<a <<"  "<<b<<endl;
                ans += a + b - 2;
                break;
            }
            if(a < b) {
               // cout <<" --  "<<a <<"  "<<b<<"  "<<b / a<<endl;

                ans += (b / a);
                b -= (b / a) * a;
            } else {
                ans += (a / b);
                a -= (a / b) * b;

            }//ans ++;
          //  cout << a << " "<<b<<endl;
        }
        cout << ans << endl;
    }
}
