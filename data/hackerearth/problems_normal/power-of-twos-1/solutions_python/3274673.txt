#coding=utf-8
import math

def main():
    T = input()
    N = []
    for _ in xrange(T):
        N.append(input())
        
    Max = max(N)
    dp = [0] * ( Max + 1 )
    end = Max / 2
    for i in xrange( 1, end + 1 ):
        for j in xrange( i + i, Max + 1, i ):
            dp[j] += 1
    
    for i in xrange( 2, Max + 1 ):
        dp[i] += dp[i - 1]
    
    for i in N:
        print dp[i]
        
if __name__=="__main__":
    main()
