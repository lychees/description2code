#include<iostream>
#include<string.h>
using namespace std;
int main()
{
    int t,i,len,len1,k,j;
    cin>>t;
    const char  s[2]= ".";
    char web[200],*str;
    for(i=0;i<t;i++)
    {
        cin>>web;
        len=strlen(web);
        str=strtok(web,s);
        while(str!=NULL)
        {
            str=strtok(NULL,s);
            break;
        }
        len1=strlen(str);
        k=0;
        for(j=0;j<len1;j++)
        {
            if( *(str+j)=='a' || *(str+j)=='e' || *(str+j)=='i' ||  *(str+j)=='o' ||  *(str+j)=='u')
            {
                continue;
            }
            else
             k++;

        }
        cout<<k+4<<"/"<<len<<"\n";
    }
    return 0;
}
