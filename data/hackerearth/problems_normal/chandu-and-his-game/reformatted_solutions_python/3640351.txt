def f(x, y, n):
    lst = []
    lst.append(x+y)
    lst.append(x+n-y)
    lst.append(y+n-x)
    lst.append(2*n-x-y)
    lst.sort()
    return lst[0]


n, k = map(int, raw_input().split())
x = map(int, raw_input().split())
y = map(int, raw_input().split())
sum = 0
i = 0
while i < k:
    sum = sum+f(x[i]-1, y[i]-1, n-1)
    i = i+1
print sum
