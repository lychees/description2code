#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    vector<int> row(1000000,0);
    vector< vector<int> > dp(11,row);
    
    for(int i=0; i<10; i++)
    {
    	dp[i][0]=1;
    }
    
    dp[10][0]=10;
    
    for(int i=1; i<=1000000; i++)
    {
    	for(int j=0; j<10; j++)
    	{
    		if(j<9)	dp[j][i]= (dp[j][i]+dp[j+1][i-1])%1000000007;
    		
    		if(j>0) 
    		{
    			dp[j][i]= (dp[j][i]+dp[j-1][i-1])%1000000007;
    			dp[10][i]= (dp[10][i]+dp[j][i])%1000000007;
    		}
    	}
    }
    
    while(t--)
    {
    	int n;
    	cin>>n;
    	
    	cout<<dp[10][n-1]<<endl;
    }
    
    return 0;
}
