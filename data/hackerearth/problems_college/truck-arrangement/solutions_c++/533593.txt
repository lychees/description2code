#include <iostream>
#include <cstdio>
#include<algorithm>
struct data
{
	int i,val;
};
bool compare(data a,data b)
{
	return a.val==b.val?a.i<b.i:a.val<b.val;
}
using namespace std;

int main()
{
        int test_cases;
        scanf("%d",&test_cases);
        while(test_cases > 0)
        {
                int N,i=0,ans;
                data arr[11000];
                scanf("%d",&N);
                while(i<N)
                {
                        scanf("%d",&arr[i].val);
                        arr[i].i=i;
                        i = i+1;
                }
                sort(arr,arr+N,compare);
                int maxDist=0;
                //your code here
                for(i=0;i<N;i++)
                {
                	int temp=max(arr[i].i-i,i-arr[i].i);
                	maxDist=max(maxDist,temp);
                }
                printf("%d\n",maxDist);
                test_cases--;
        }
        return 0;
}
