#include <stdio.h>
#include <string.h>
#include <iostream>
#include <math.h>
#include <vector>
using namespace std;
int main()
{
    int T,win,temp;

    cin>>T;
    while(T--){

        vector<int> nim(3);
        cin>>nim[0]>>nim[1]>>nim[2];

        if(nim.size()==1) cout<<"Printf{}"<<endl;
        else{
            win=1;
            while(1){
            temp=nim[0];
            for(int i=1;i<nim.size();i++) temp = temp ^ nim[i];
        //    cout<<"xor"<<temp;
            if(temp==0){break;}
         //   cout<<endl;
            for(int i=0;i<nim.size();i++) {
              //  cout<<"nim[i]="<<nim[i];
                int temp2=nim[i]^temp;
              //  cout<<"xor2="<<temp2<<endl;
                if(temp2<nim[i]) {
                        nim[i] = nim[i] ^ temp;
                        break;
                }
            }
            win=1-win;
            }
            if(win) cout<<"BILL"<<endl;
            else cout<<"NO BILL"<<endl;

        }
    }
    return 0;
}
