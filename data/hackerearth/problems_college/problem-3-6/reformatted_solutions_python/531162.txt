n = int(raw_input())
a = []
f = ' '

for i in range(n):
    x, y = raw_input().split(' ')
    x, y = int(x), int(y)
    a.append([x, y])

for i in range(n):
    p, q = 0, 0
    for j in range(i) + range(i+1, n):
        p += 1
        if a[i][1] == a[j][0]:
            p += 1
        else:
            q += 1
    f += str(p) + ' ' + str(q) + '\n'

print f
