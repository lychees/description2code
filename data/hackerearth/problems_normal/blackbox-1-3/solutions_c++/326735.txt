
#include<iostream>
#include<limits.h>
using namespace std;

int main()
{
	int a,b,c,T;
	long long int max, min1, min2;
	cin >> T;
	while(T)
	{
		
		cin >> a >> b >> c;
		
		max = a>b? a:b;
		min1 = a>b? b:a;
		min2 = max>c? c: max;
		max = max>c? max:c;
		
		if(min1*min1 == (max+min2)*(max-min2))
		{
			int min;
			min = INT_MAX;
			
			while(min)
			{
				min = min>a ? a:min;
				min = min>b?b:min;
				min = min>c?c:min;
				
				a = (a==min)?a:(a-min);
				b = (b==min)?b:(b-min);
				c = (c==min)?c:(c-min);
				
				if(a==b && b==c)
					min = 0;		
			}
			
			if(a==1)
				cout << "YES" << endl;
			else
				cout << "NO" << endl;
		}
		else
			cout << "NO" << endl;
		T--;
	}

	return 0;
}
