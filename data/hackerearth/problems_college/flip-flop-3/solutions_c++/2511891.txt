#include<iostream>
#include<string.h>
using namespace std;
int main()
{
   char S[100001];
    long i,j,N,count;
   cin>>N;
   for(i=0;i<N;i++){
       cin>>S; count=0;
    for(j=0;j<strlen(S)-1;j++)
         if(S[j]==S[j+1])
            count++;
        cout<<count<<endl;
         }
    return 0;
}
