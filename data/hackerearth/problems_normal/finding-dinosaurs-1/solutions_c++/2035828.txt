#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<string>
#include<cmath>
#include<algorithm>
using namespace std;
string st[1048576];
int main() {
	int N,K,L,Q;
	int i,j;
	cin>>N>>K>>L>>Q;
	string name;
	int id,at;
	for(i=0;i<N;i++){
		cin>>name;id = 0;
		for(j=0;j<K;j++){
			cin>>at;
			id = (id * (L+1)) + at;
		}
		st[id] = name;
	}
	while(Q--){
		id = 0;
		for(i=0;i<K;i++){
			cin>>at;
			id = (id * (L+1)) + at;
		}
		if(st[id].length() == 0)
			cout<<"You cant fool me :P"<<endl;
		else
			cout<<st[id]<<endl;
	}
	return 0;
}
