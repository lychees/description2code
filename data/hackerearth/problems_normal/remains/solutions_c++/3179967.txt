#include<iostream>
#include<cmath>
    using namespace std;

  int main(){
    	ios_base::sync_with_stdio(false);
    	long long int i,t,n,x,y,z=0,s;
    	//int k;
    	cin>>t;
    	while(t--){
                //k=0;
            cin>>x>>y>>n;
            s=x+y;
            for(i=3;i<=n;i++){
                z=abs(x-y);
                s=s+z;
                if(x==y&&z==0)
                  {
                    if((n-i)%3==0)
                        s=s+((n-i)*(x+y))/3;
                    if((n-i)%3==1)
                        s=s+((n-i-1)*(x+y))/3+x;
                    if((n-i)%3==2)
                        s=s+((n-i-2)*(x+y))/3+x+y;
                    //cout<<"\n"<<i<<"\n";
                    break;
                    }
                x=y;
                y=z;
            }
            cout<<s<<"\n";
            s=0;
    	}


    	return 0;
    	}
