#include<bits/stdc++.h>
using namespace std;
#define MOD 1000003
long long dp[MOD+1];
 
void init()
{
	dp[0]=1;
	int i;
	for(i=1;i<=MOD;i++)
	{
		dp[i]=((long long)i * dp[i-1])%MOD;
	}
}
 
int main()
{
    int t,i;
    init();
    scanf("%d",&t);
    long long n,k;
    while(t--)
    {
    	cin>>n>>k;
    	if(n>MOD)
    	n=MOD;
    	k=k%MOD;
    	long long res=(dp[n]*k)%MOD;
    	cout<<res<<endl;
    }
    return 0;
}
