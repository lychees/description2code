#include<iostream>
#include<cstdio>
typedef unsigned long long int ull;
using namespace std;
#define max 1000001
ull a[max];
void display()
{
	for(int i=0;i<max;i++)
	printf("\nzero's= %d i = %d ",a[i],i);
}
void pre()
{
	for(int n=5;n<max;n++)
	{
	  int count=0;
	  for(int i=5;n/i>0;i*=5)
	  count+=n/i;
	  a[n]=a[n-1]+count;
    }
    //display();
}
int main()
{
	int t,i,j;
	pre();
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d",&i,&j);
		printf("%lld\n",(a[j]-a[i-1]));
	}
}
