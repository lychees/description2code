#include <iostream>
using namespace std;
void swap(long long *a, int i, int n, int k)
{
	if(k>0 && i<n)
	{
		//cout<<k<<endl;
		long long j,min,p;
		for(j=i;j<i+k+1 && j<n;j++)
		{
			if(j==i || a[j]<min)
			{
				min=a[j];
				p = j;
			}
		}
		for(j=p;j>i;j--)
		{
			a[j]=a[j-1];
		}
		a[i]=min;
		swap(a,i+1,n,k-(p-i));
	}
}
	
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int n,k;
    	cin>>n>>k;
    	long long arr[n],i;
    	for(i=0;i<n;i++)
    		cin>>arr[i];
    	//cout<<"ok"<<endl;
    	swap(arr,0,n,k);
    	for(i=0;i<n;i++)
    		cout<<arr[i]<<" ";
    	cout<<endl;
    }
    return 0;
}
