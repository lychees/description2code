#include <bits/stdc++.h>
using namespace std;
int n,ans;
vector<int> v;
int main() {
	ios_base::sync_with_stdio(0);
	cin.tie(0);
	cin>>n;
	v.resize(n);
	for(int i=0;i<n;++i) cin>>v[i];
	while(v.size()>2) {
		int idx,cura=-1;
		for(int i=1;i<v.size()-1;++i) {
			if(cura<v[i-1]*v[i+1]) {
				cura=v[i-1]*v[i+1];
				idx=i;
			}
		}
		v.erase(v.begin()+idx);
		ans+=cura;
	}
	cout<<ans<<"\n";
}
