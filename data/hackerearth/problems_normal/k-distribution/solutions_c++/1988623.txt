const int N=100005;

LL dp[102][N];
LL a[102];
LL pwr(LL a, LL b) {
    if(b==0)
        return 1;
    LL ans=1;
    while(b) {
        if(b&1) {
            ans=((ans%MOD)*(a%MOD))%MOD;
        }
        b/=2;
        a=((a%MOD)*(a%MOD))%MOD;
    }
    return ans;
}
void show() {

}
int main() {
    long long n,k;
    cin>>n>>k;
    LL sm=0;
    int cnt=0;
    F(i,0,n){
        cin >> a[i];
        sm+=a[i];
        if(a[i]>=k) {
            cnt++;
        }
    }
    if(sm<2*k){
    	cout<<0<<endl;
    	return 0;
    }
//    sort(a,a+n);
    for(int i=0;i<102;i++) {
        dp[i][0]=1;
    }
    if(a[0]<k)
        dp[0][a[0]]=1;
    for(long long i=1;i<n;i++) {
        for(long long j=0;j<k;j++){
            dp[i][j]=dp[i-1][j];
            dp[i][j]%=MOD;
        }
        for(long long j=0;j<k;j++) {
            if(a[i]+j<k && (dp[i-1][j])) {
                dp[i][a[i]+j]+=dp[i-1][j];
            }
        }
        //      show();
    }
    LL sum=0;
    for(int i=0;i<k;i++) {
        sum+=dp[n-1][i];
        sum%=MOD;
    }
    sum=((sum%MOD)*2)%MOD;
    LL ans=pwr(2,n);
    ans=(ans-sum)%MOD;
    if(ans<0)
        ans+=MOD;
    cout<<ans%MOD<<endl;
//        show();
}
