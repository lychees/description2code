#include<iostream>
#include<algorithm>
#include<vector>
#define mod 1000000007
using namespace std;
long long pow(long long a,long long b)
{
	if(b==0) return 1;
	if(b==1) return a%mod;
	if(b&1)
	{
		long long t=pow(a,b/2)%mod;
		t=(t*t)%mod;
		return (a*t)%mod;
	}
	else
	{
		long long t=pow(a,b/2)%mod;
		return (t*t)%mod;
	}
}
void sol()
{
	long long n;
	cin>>n;
	cout<<(pow(4ll,n)-pow(2ll,n)+mod)%mod<<endl;
}
int main()
{
	int t;
	cin>>t;
	while(t--) sol();
	return 0;
}
