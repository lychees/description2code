#include<bits/stdc++.h>
using namespace std;
#define FOR(i,a,b) for(long long i=a;i<b;i++)

short int a[10000000]={1};

long long fib[100];
vector<long long>my;

int isprime(long long a)
{
	FOR(i,2,sqrt(a))
	{
		if(a%i==0)return 0;
	}
	return 1;
}

void seive()
{

    fib[0]=0;
    fib[1]=0;
    fib[2]=1;
    FOR(i,3,70)
    {
    	fib[i] = fib[i-1]+fib[i-2];
    	my.push_back(fib[i]);

    }
}

int main()
{
    seive();
    
    long long sum=0;
    long long total[1000001];
    FOR(i,1,1000001)
    {
            if(!(binary_search(my.begin(),my.end(),i)))
                sum+=i;
                total[i+1]=sum;
                //cout<<i<<" "<<total[i]<<endl;
    }

    int t;
    cin>>t;
    while(t--)
    {
        long long n;
        cin>>n;
        long long ans=total[n];
        //cout<<ans<<endl;

        if(ans%2==0)cout<<"Tac"<<endl;
        else if(isprime(ans))cout<<"Tic"<<endl;
        else cout<<"Toe"<<endl;
    }





    return 0;
}
