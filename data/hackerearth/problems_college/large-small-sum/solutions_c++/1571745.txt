#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    while(t--)
    {
    	int n,a[10001];
    	cin>>n;
    	int i,j;
    	
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	
    	cin>>i>>j;
    	
    	sort(a,a+n);
    	
    	cout<<a[j-1]+a[n-i]<<endl;
    }
    
    return 0;
}
