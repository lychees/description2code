#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,i;
    cin>>t;
    while(t--){
    	cin>>n;
    	int a[n];
    	long long sum=0,min=LONG_MAX;
    	for(i=0;i<n;i++){
    		cin>>a[i];
    		if(a[i]<min){
    			min=a[i];
    		}
    		sum+=(a[i]-1);
    	}
    	cout<<min-1<<" "<<sum<<endl;
    }
    return 0;
}
