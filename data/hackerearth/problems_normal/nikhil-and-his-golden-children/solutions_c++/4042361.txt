#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll M=1000000007;
ll bigmod(ll a,ll b,ll m)
{
    long long x=1,y=a;
    while(b>0)
    {
        if(b & 1)
            x=(x*y)%m;
        y=(y*y)%m;
        b>>=1;
    }
    return x;
}
long long C(long long n, long long r, int M)
{
    if(r > n / 2) r = n - r; // because C(n, r) == C(n, n - r)
    long long nr = 1;
    long long dr = 1;
    int i;

    for(i = 1; i <= r; i++)
    {
        nr = (nr * (n - r + i))%M;
        dr = (dr * i)%M;
    }

    long long ans = (nr * bigmod(dr, M-2, M)) % M;

    return ans;
}
int main()
{
    ll t,n,i,k,x;
    cin>>t;
    while(t--)
    {
        cin>>n>>k;
        for(i=1;i<=n;i++)
        {
            cin>>x;
            k-=x;
        }
        if(k<0)
        {
            cout<<0<<endl;
        }
        else
        {
            cout<<C(n+k-1,n-1,M)<<endl;
        }
    }
}
