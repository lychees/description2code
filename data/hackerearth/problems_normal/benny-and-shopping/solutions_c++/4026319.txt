#include <iostream>
#include <algorithm>
#include <string>
#include<string.h>

using namespace std;


int main(){
	
	int no_test;
	cin >> no_test;
	char order[1001];
	char c;
	cin >> c;
	int length = 1;
	while(no_test--){
		cin.getline(order, sizeof(order));
		length = length + strlen(order);
		// cout << length << endl;
		for(int i = 0;i < length; i++ ){
			if(order[i] == '$'){
				i++;
				while(i < length && (order[i] == '0' || order[i] == ' ')){
					i++;
				}
				cout << '$';
				if(i == length || (order[i] < '0' || order[i] >'9') ){
					cout << '0' << endl;
					break;
				}
				for(int j = i; j < length; j++){
					if((order[j] < '0' || order[j] >'9') && order[j] != ' '){
						break;
					}
					if(order[j] != ' '){
						cout << order[j];
					}
				}
				cout << endl;
				break;
			}
		}
		length = 0;
	}
	return 0;
}
