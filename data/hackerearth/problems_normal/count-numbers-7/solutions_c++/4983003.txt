#include<bits/stdc++.h>
using namespace std;
int main()
{
	int k,t;
	cin>>k>>t;
	int p[k];
	for(int i=0;i<k;i++)
		cin>>p[i];
	while(t--)
	{
		int a,b;
		cin>>a>>b;
		long long ans=0;
		for(int i=1;i<pow(2,k);i++)
		{
			int cnt=0;
			int a1=a-1,b1=b;
			for(int j=0;j<k;j++)
			{
				if(i & (1<<j))
				{
					cnt++;
					a1/=p[j];
					b1/=p[j];
				}
			}
			if(cnt%2==0)
				ans-=(b1-a1);
			else
				ans+=(b1-a1);
		}
		cout<<ans<<endl;
	}
}
