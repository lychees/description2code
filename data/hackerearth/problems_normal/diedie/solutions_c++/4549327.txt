#include<iostream>

#define mod 1000000007
using namespace std;
long long int modular(long long int base,long long int power)
{
	base=base%mod;
	long long int result=1;
	while(power>0)
	{
		if(power%2==1)	
			result=(result*base)%mod;
		base=(base*base)%mod;
		power/=2;
	}
	return result;
}
int main()
{
    int t;
    cin>>t;
    while(t>0)
    {
    	
    	long long int n,temp,ans;
    	cin>>n;
    	ans=modular(2,n-1);
    	temp=modular(ans,mod-2);
    	ans=(temp)%mod;
    	cout<<ans<<endl;
    	t--;
    }
}
