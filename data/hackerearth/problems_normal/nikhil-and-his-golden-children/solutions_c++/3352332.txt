ll gcd(ll n1,ll n2){
	if(n1%n2==0)return n2;
	return gcd(n2,n1%n2);
}
ll powmod(ll base,ll exponent)
{
	ll ans=1;
	while(exponent){
		if(exponent&1)ans=(ans*base)%mod;
		base=(base*base)%mod;
		exponent/=2;
	}
	return ans;
}
ll fact[upperlimit+1];
ll c(int n,int r){
	ll ret=fact[n];
	ret*=powmod((fact[r]*fact[n-r])%mod,mod-2);
	ret%=mod;
	return ret;
}
