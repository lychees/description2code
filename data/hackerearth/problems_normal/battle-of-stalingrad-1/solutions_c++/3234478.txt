#include<bits/stdc++.h>
using namespace std;
long double area(long double x1,long double  y1,long double  x2 ,long double y2,long double  x3 ,long double  y3)
{
	long double a = (((x1*y2+ x2*y3 + x3*y1)- (y1*x2 + y2*x3 + y3*x1))/2.0);
	return abs(a);
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long double x1,y1,x2,y2,x3,y3;
		long double a,b;
		long double a1,a2,a3,s;
		cin>>x1>>y1>>x2>>y2>>x3>>y3;
		cin>>a>>b;
			s = area(x1,y1,x2,y2,x3,y3);
			a1 = area(x1,y1,x2,y2,a,b);
			a2 = area(x1,y1,a,b,x3,y3);
			a3 = area(a,b,x2,y2,x3,y3);
			if(a1+a2+a3 == s)
			{
				cout<<"INSIDE"<<"\n";
			}
			else
			{
				cout<<"OUTSIDE"<<"\n";
			}
	}
}
