#include <bits/stdc++.h>

#define ll long long int


using namespace std;

bool bit(int mask, int x) {
     mask = mask & (1 << x);
     return (mask > 0);
}

int main()
{
	ll n;
	cin>>n;
	ll i,j,num[1000006];
	
	for(i=0;i<n;i++)
	{
		cin>>num[i];
	}
	ll jai;
	cin>>jai;
	ll flag,sum=0;
    for(j=0;j< 1 << n;j++)
    {
    	sum=0;
    	for (int i = 0; i < n; ++i) {
                  if (bit(j, i))
                  {
                  	sum = sum + num[i];
                  }
    	}if(sum == jai)
    	{
    		flag=1;
    		break;
    	}
    	else
    	{
    		flag=0;
    	}
    }
    
    if(flag)
    {
    	cout<<"True"<<endl;
    }
    else
    {
    	cout<<"False"<<endl;
    }
    return 0;
}
