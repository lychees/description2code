#include<iostream>
#include<math.h>
using namespace std;
typedef long long ll;

ll ARR[1000005];
ll G[1000005];

void PRE()
{
     for(ll i=2;i<=1000;i++)
     {
             for(ll j=2;i*j<=1000000;j++)
             {
                     ll x=i*j;
                     if(i<sqrt(x))
                     G[x]+=(i+j);
                     else if(i==sqrt(x))
                     G[x]+=i;
                     if(ARR[x]==0)
                     ARR[x]=i;
             }
     }
    
     
     for(ll i=2;i<=1000000;i++)
     {
             if(ARR[i]==0)
             ARR[i]=i;
     }
     for(ll i=2;i<=1000000;i++)
     ARR[i]=ARR[i-1]+ARR[i];
    
     
     for(ll i=1;i<=1000000;i++)
     {
           G[i]+=1;
     }
     
     
     G[1]=0;
     for(ll i=2;i<=1000000;i++)
     G[i]=(G[i-1]+G[i]);
     
     
     for(ll i=1;i<=1000000;i++)
     ARR[i]=(ARR[i]+G[i])%i;
}

int main()
{
    PRE();
    
    int T;
    cin>>T;
    while(T--)
    {
              int N;
              cin>>N;
              cout<<ARR[N]<<endl;
    }
}
