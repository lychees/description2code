// CP template
#include <bits/stdc++.h>

using namespace std;

long gcd(long a, long b){
	if(b==0){
		return a;
	}
	else
		return gcd(b, a%b);
}

int main()
{
	int t;
	cin>>t;
	for(int i=1; i<=t; i++){
		string s,r="";
		cin>>s;
		for(int j=s.length()-1; j>=0; j--){
			r += s.substr(j,1);
		}
		bool same = true; int k=1;
		while(k<s.length() && same){
			// printf("s= %d ",s[k]-s[k-1]);
			// printf("r=%d\n",r[k]-r[k-1]);
			if(abs(s[k]-s[k-1]) != abs(r[k]-r[k-1]))
				same = false;
			k++;
		}
		if(same)
			printf("Funny\n");
		else
			printf("Not Funny\n");
	}
	return 0;
}
