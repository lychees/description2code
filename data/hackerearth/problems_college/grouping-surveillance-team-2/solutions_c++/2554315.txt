#include<bits/stdc++.h>
using namespace std;
long long int fun(int a[],int k,int n)
{
  long long	int t[n+1],i,j;
	memset(t,0,sizeof(t));
	 t[0]=1;
	 for(i=0;i<k;i++)
	 {   
	 	for(j=a[i];j<=n;j++)
	 	{
	 	  t[j]+=t[j-a[i]];
	    }
	 	  //cout<<t[i]<<" ";
	 }
	 return t[n];
}
int main()
{
	long long int n,t;
	cin>>t;
	while(t--)
	{
	cin>>n;
	int a[n],k=0,i;
	for( i=2;i<=n;i++)
	   a[k++]=i;
	   cout<<fun(a,k,n)<<endl;
    }  
	 
	
}
