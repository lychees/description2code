#include<bits/stdc++.h>

#define ll long long
#define int_max (1e9) 

using namespace std;

const int max_cokes= 155;
const int max_n5= 155;
const int max_n10= 55;
ll cokes_req, tot, mem[max_cokes][max_n5][max_n10];

ll solve(ll cokes, ll n5, ll n10)
{
	if(cokes==cokes_req)
	return 0;
	
	if(mem[cokes][n5][n10]!=-1)
	return mem[cokes][n5][n10];
	
	ll n1= tot- cokes*8- n5*5- n10*10;
	
	ll ret= int_max;
	if(n1>=8)
	ret= min(ret, solve(cokes+1, n5, n10)+ 8);
	
	if(n1>=3 && n5>=1)
	ret= min(ret, solve(cokes+1, n5-1, n10)+ 4);
	
	if(n5>=2)
	ret= min(ret, solve(cokes+1, n5-2, n10)+ 2);
	
	if(n10>=1)
	ret= min(ret, solve(cokes+1, n5, n10-1)+ 1);
	
	if(n1>=3 && n10>=1)
	ret= min(ret, solve(cokes+1, n5+1, n10-1)+ 4);
	
	if(ret>int_max)
	ret= int_max;
	
	mem[cokes][n5][n10]= ret;
	
	return ret;
}

int main()
{
	ll t, n1, n5, n10, ans;
	
	cin>>t;
	while(t--)
	{
		cin>>cokes_req>>n1>>n5>>n10;
		
		memset(mem, -1, sizeof(mem));
		
		tot= n1+ n5*5+ n10*10;
		
		ans= solve(0, n5, n10);
		if(ans<int_max)
		cout<<ans<<endl;
		else
		cout<<"-1"<<endl;
	}
	
	return 0;
}
