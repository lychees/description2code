#include <iostream>
#include <cstdio>
using namespace std;

char c ; int cnt1 ; long long curadd, ans ;

int main()
{
    while( scanf("%c", &c) != EOF ) {
        if( c == '0' )
            curadd += cnt1 ;
        else if( c == '1' )
            ans += curadd,
            ++cnt1 ;
    }
    cout << ans ;
    return 0;
}
