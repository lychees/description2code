#include <iostream>
#include <algorithm>
using namespace std;
bool palin(string s){
	string temp=s;
	reverse(s.begin(),s.end());
	return(s==temp);
}
int main()
{
    int t; cin>>t;
    while(t--){
    	string s1,s2,temp1,temp2; cin>>s1>>s2;
    	bool flag=0;
    	temp1=s1+s2;
    	temp2=s2+s1;
    	flag=palin(temp1)||palin(temp2);
    	if(flag) cout<<"YES"<<endl;
    	else if(!flag) cout<<"NO"<<endl;
    }
    return 0;
}
