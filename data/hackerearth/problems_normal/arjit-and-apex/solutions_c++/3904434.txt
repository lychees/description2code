#include<iostream>
#include<string>
#include<string.h>
#include<algorithm>
#include<stdlib.h>
using namespace std;

typedef struct{
        int skill;
        int proficiency;
}Candidate;

bool cmp(Candidate a, Candidate b)
{       
        return(a.skill == b.skill ? a.proficiency <= b.proficiency : a.skill <= b.skill);
}

int main()
{
        int test;
        cin>>test;
        
        int n,m;
        int g,h;
        
        int good,great;
        
        while(test--)
        {
                good=0;
                great=0;
                cin>>m>>n;
                
                vector<Candidate> old;
                vector<Candidate> New;
                
                Candidate dummy;
                
                for(auto i=0; i<m; i++)
                {
                        cin>>dummy.skill>>dummy.proficiency;    
                        New.push_back(dummy);
                }
                
                for(auto i=0; i<n; i++)
                {
                        cin>>dummy.skill>>dummy.proficiency;
                        old.push_back(dummy);
                }       
                
                cin>>g>>h;
                
                sort(old.begin(),old.end(),cmp);
                sort(New.begin(),New.end(),cmp);

                auto j = old.begin();
                for(auto i = New.begin();i != New.end(), j != old.end();)
                {
                        if(i->skill == j->skill)
                        {
                                good++;
                                for(;i->skill == j->skill;)
                                {
                                        if(i->proficiency == j->proficiency)
                                        {
                                                i++;j++;
                                                great++;
                                                break;
                                        }
                                        else if(i->proficiency < j->proficiency)
                                        {
                                                i++;
                                        }
                                        else if(i->proficiency > j->proficiency)
                                        {
                                                j++;
                                        }
                                }
                                continue; 
                        }
                        if(i->skill < j->skill)
                        {
                                i++;
                                continue;
                        }
                        if(i->skill > j->skill)
                        {
                                j++;
                                continue;
                        }
                }
                                
                if(good == g && great == h)
                {
                        cout<<"Great"<<endl;
                }
                else if(good == g)
                {
                        cout<<"Good"<<endl;
                }
                else
                {
                        cout<<":("<<endl;
                }
        }
}
