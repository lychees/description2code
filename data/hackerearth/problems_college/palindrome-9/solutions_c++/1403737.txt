#include <cstdio>
#include <iostream>
#include <set>
#include <cmath>
using namespace std;

#define MAX 100000000

typedef long long ll;

bool isPalin(int n){
	int tempN = n,rev=0,rem;

	while(tempN){
		rem = tempN%10;
		rev = rev*10+rem;
		tempN/=10;
	}

	return n == rev;
}

int main(){
	int sz,a,b,limit = sqrt(MAX),num,i,j;
	ll sum = 0;
	set <int> s;
	set <int> :: iterator it;

	scanf("%d%d",&a,&b);

	for(i=1;i<=limit;++i){

		num = i*i;

		for(j=i+1;j<=limit;++j){
			num+=(j*j);
			if(num>MAX) break;

			if(isPalin(num))
				s.insert(num);
		}

	}

	sz = s.size();

	for(it=s.begin();it!=s.end();++it)
        if(*it>=a && *it<=b)
            sum+=(*it);
    printf("%lld\n",sum);

	return 0;
}
