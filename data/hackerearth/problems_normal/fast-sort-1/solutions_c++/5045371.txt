#include"bits/stdc++.h"
using namespace std;
int main()
{
	long int tc;
	cin>>tc;
	while(tc--)
	{
		map<char , long int> alpha_to_num;
		map<long int,char> num_to_alpha;
		for(long int i=0;i<26;i++)
		{
			char a;
			cin>>a;
			alpha_to_num[a]=i;
			num_to_alpha[i]=a;
		}
		
		string s;
		cin>>s;
		vector<long int> v;
		for(long int i=0;i<s.length();i++)
		{
			v.push_back(alpha_to_num[s[i]]);
		}
		sort(v.begin(),v.end());
		for(vector<long int>::iterator it=v.begin();it!=v.end();it++)
			cout<<num_to_alpha[*it];
		cout<<endl;
	}
	return 0;
}
