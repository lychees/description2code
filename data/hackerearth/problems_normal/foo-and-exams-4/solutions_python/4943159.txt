from sys import stdin
t = int(stdin.readline())
for _ in xrange(t):
	a,b,c,d,k = map(int,stdin.readline().split())
	lo = 0
	hi = 10**19
	while hi-lo > 1:
		mid = (lo + hi)/2
		if a*(mid**3) + b*(mid**2) + c*(mid) + d<=k:
			lo = mid
		else:
			hi = mid
	lo = lo-5
	if lo < 0:
		lo = 0
	ans = lo
	while True:
		if a*(lo**3) + b*(lo**2) + c*lo + d>k:
			break
		ans = lo
		lo += 1
	print ans
