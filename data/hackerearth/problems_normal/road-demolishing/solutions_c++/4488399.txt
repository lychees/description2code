#include<bits/stdc++.h>
using namespace std;
#define ll	long long int
int main()
{
	
	int t;
	cin>>t;
	assert(t>=1 && t<=100000);
	while(t--)
	{
		ll n,q;
		cin>>n>>q;
	
		q--;
		ll ans=n*n-(n%q)*((n+q-1)/q)*((n+q-1)/q);
		ans=ans-(q-n%q)*(n/q)*(n/q);
		ans/=2;
		ans=(n*(n-1))/2-ans;
	
		cout<<ans<<endl;
	}
	return 0;
}
