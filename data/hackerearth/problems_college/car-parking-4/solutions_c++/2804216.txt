#include <iostream>
#include <string>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int count=0,i,l,temp;
    	string s;
    	cin>>s;
    	l=s.size();
    	temp=l;
    	for(i=0;i<temp;i++)
    	{
    		if(s[i]=='B') 
    		{
    			l--; 
    			if((i-1)>=0 && s[i-1]=='-') {l--;s[i-1]='Z';}
    			if((i-2)>=0 && s[i-2]=='-') {l--;s[i-2]='Z';}
    		}
    		else if(s[i]=='S')
    		{
    			l--;
    			if((i-1)>=0 && s[i-1]=='-' ) {l--;s[i-1]='Z';}
    			if((i+1)<temp && s[i+1]=='-' ) {l--;s[i+1]='Z';}
    		}
    		else if(s[i]=='D') l--;
    	}
    	
    	cout<<l<<endl;
    }
    return 0;
}
