#include <bits/stdc++.h>
using namespace std;


int main()
{ int t;
	
	scanf("%d",&t);
	t++;
	while(t--){
		map<string , int>m;
		string p;
		getline(cin,p);
		string s[100000];
	   int  len=p.length();
	    int count=0;
	    for(int i=0;i<len;i++){
	    	if(p[i]==' '){
	    		count++;
	    	}
	    	else s[count]=s[count]+p[i]; 
	    }
	    for(int i=0;i<=count;i++){
	    	if(m[s[i]]==0){
	    		cout<<s[i]<<" ";
	    		m[s[i]]=1;
	    	}
	    }
	    cout<<endl;
	}
  
    return 0;
}
