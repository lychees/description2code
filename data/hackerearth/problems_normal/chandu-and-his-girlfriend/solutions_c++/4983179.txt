#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin >> t;
    while(t--){
    	int n;
    	cin >> n;
    	long long int arr[n];
    	int i;
    	for(i=0;i<n;i++){
    		cin >> arr[i];
    	}
    	sort(arr,arr+n,greater<long long int>());
    	for(i=0;i<n;i++){
    		cout << arr[i] << " ";
    	}
    	cout << endl;
    }
    return 0;
}
