using namespace std;

int n;
int a[maxN+5];

int searchb(int s,int k)
{
	int e,m;
	e=n-1;
	while((e-s)>1)
	{
		m=s+(e-s)/2;
		if(a[m]<=k)
			s=m;
		else
			e=m-1;
	}
	if(a[e]<=k)
		return e;
	return s;
}

int main()
{
	//ios_base::sync_with_stdio(false);
    //cin.tie(NULL);
    #ifndef LOCAL_SYS

    #endif
    int i,j,ans;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    	scanf("%d",&a[i]);
    sort(a,a+n);
    ans=n+5;
    for(i=0;i<n;i++)
    {
    	j=searchb(i,3*a[i]);
    	//a[j]<=3*a[i]
    	ans=min(ans,(n-(j-i+1)));
    }
    printf("%d\n",ans);
    return 0;
}
