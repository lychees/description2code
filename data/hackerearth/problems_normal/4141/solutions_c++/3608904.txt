#include <iostream>
#include<bits/stdc++.h>
#define ll long long
using namespace std;

int main()
{
    ll t;
    cin>>t;
    while(t--){
    	ll a,b,c,n;
    	cin>>a>>b>>c>>n;
    	ll s[n];
    	s[0]=a;
    	s[1]=b;
    	for(int i=2;i<n;i++){
    		s[i]=(s[i-1]*c+s[i-2])%10;
    	}
    	ll sum=0;
    	for(int i=0;i<n;i++){
    		sum=((sum*10)%41+s[i]%41)%41;
    	}
    	//cout<<sum<<endl;
    	if(sum%41==0){
    		cout<<"YES"<<endl;
    	}
    	else
    		cout<<"NO"<<endl;
    }
}
