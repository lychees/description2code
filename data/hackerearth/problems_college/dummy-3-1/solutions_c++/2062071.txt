#include <iostream>
using namespace std;
 
int main()
{
    int t;
    long int n;
    int given[1000000], inverse[1000000];
    scanf("%d",&t);
    while(t--){
    	scanf("%ld",&n);
    	for(int i=1;i<=n;i++){
    		scanf("%d",&given[i]);
    	}
    	for(int j=1;j<=n;j++){
    		inverse[given[j]]=j;
    	}
    	int k=0;
    	for(k=1;k<=n;k++){
    		if(inverse[k]!=given[k]){
    			break;
    		}
    	}
    	if(k>=n){
    		printf("inverse\n");
    	}
    	else printf("not inverse\n");
    }
    return 0;
}
