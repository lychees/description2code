#include <iostream>
using namespace std;
int main()
{
   int a,n;
   cin>>a>>n;
   int Num[n];
   for(int i=0;i<n;i++)
   {
   	  cin>>Num[i];
   }
   int fact[100]={-1};
   int temp=2;
   int count=0;
 while(temp<=a)
 {   if(a%temp==0)
        {
        	fact[count]=temp;
        	count++;
        }
 	 temp++;
 }
 int total=0;
 for(int i=0;i<n;i++)
 {  
   for(int j=0;j<count;j++)
    {    int temp1=fact[j];
          if(Num[i]%temp1==0)
              {
              	total++;
              	break;
              }
    	
    }
 	
 }
 cout<<total<<endl;
   
    return 0;
}
