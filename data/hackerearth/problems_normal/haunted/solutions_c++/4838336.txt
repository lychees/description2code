#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n,m;
    cin>>n>>m;
    map<int,int> my_map;
    stack<int> s;
    for(int i=0;i<n;i++)
    {
        int temp;
        cin>>temp;
        my_map[temp]++;
        if(i>0)
        {
            int prev=s.top();
            if(my_map[temp]==my_map[prev])
            {
                if(temp>=prev)
                {
                    s.push(temp);
                    cout<<temp<<" "<<my_map[temp]<<"\n";
                }
                else
                {
                    //s.push(temp);
                    cout<<prev<<" "<<my_map[prev]<<"\n";
                }

            }
            else if(my_map[temp]>my_map[prev])
            {
                s.push(temp);
                cout<<temp<<" "<<my_map[temp]<<"\n";                
            }
            else
            {
                cout<<prev<<" "<<my_map[prev]<<"\n";
            }
        }
        else
        {
           // stack<int> s;
            s.push(temp);
            cout<<temp<<" "<<my_map[temp]<<"\n";
        }
    }
    return 0;
}
