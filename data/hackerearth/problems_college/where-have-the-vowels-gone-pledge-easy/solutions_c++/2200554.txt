#include <bits/stdc++.h>
using namespace std;

int main()
{
    int N,a,e,i,o,u;
    string str;
    cin >> N;
    	cin>>str;
    	int len = str.length();
    	a=e=i=o=u=0;
    	int j=0;
    	while(str[j]!='\0'){
    		if(str[j]=='a'){
    			a=1;
    		} else if(str[j]=='e'){
    			e=1;
    		} else if(str[j]=='i'){
    			i=1;
    		} else if(str[j]=='o'){
    			o=1;
    		} else if(str[j]=='u'){
    			u=1;
    		}
    		if(a&&e&&i&&o&&u){
    			cout<<"YES"<<endl;
    			break;
    		} 
    		j++;
    	} 
    	if(j>=str.length())
    	cout<<"NO"<<endl;
    return 0;
}
