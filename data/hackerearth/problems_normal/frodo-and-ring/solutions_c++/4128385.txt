#include<bits/stdc++.h>
using namespace std;
int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int dp[201][201],arr[201][201],t,n,m;
    char x;
    cin>>t;
    while(t--){
        cin>>n>>m;
        for(int i=0;i<n;i++)
            for(int j=0;j<m;j++){
                cin>>x;
                if(x=='T')
                    arr[i][j]=150;
                else if(x=='D')
                    arr[i][j]=-100;
                else
                    arr[i][j]=0;
            }
        dp[0][0]=arr[0][0];
        for(int j=1;j<m;j++)
            if(dp[0][j-1]<0)
                dp[0][j]=-1;
            else
                dp[0][j]=max(dp[0][j-1]+arr[0][j],-1);
        for(int i=1;i<n;i++)
            if(dp[i-1][0]<0)
                dp[i][0]=-1;
            else
                dp[i][0]=max(dp[i-1][0]+arr[i][0],-1);
        for(int i=1;i<n;i++)
            for(int j=1;j<m;j++)
                if(dp[i-1][j]>=0&&dp[i][j-1]>=0)
                    dp[i][j]=max(arr[i][j]+max(dp[i-1][j],dp[i][j-1]),-1);
                else if(dp[i-1][j]>=0)
                    dp[i][j]=max(arr[i][j]+dp[i-1][j],-1);
                else if(dp[i][j-1]>=0)
                    dp[i][j]=max(arr[i][j]+dp[i][j-1],-1);
                else
                    dp[i][j]=-1;
        cout<<dp[n-1][m-1]<<"\n";
    }
    return 0;
}
