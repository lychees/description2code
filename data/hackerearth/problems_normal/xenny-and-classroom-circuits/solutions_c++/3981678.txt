#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

int t, n;
string s;

int main()
{
	ios::sync_with_stdio(false);
	
	cin >> t;
	while (t--) {
		cin >> n >> s;
		
		int odd = 0;
		
		for (int i=0; i<2*n; i++)
			if (s[i] == 'B' && i&1)
				odd++;
				
		cout << min(odd, n-odd) << endl;
	}
	
    return 0;
}
