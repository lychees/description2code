#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
	int noofcases;
	cin>>noofcases;
	for(int i=0;i<noofcases;i++ )
	{
		int bt1,bt2,mt1,mt2;
		cin>>bt1>>bt2>>mt1>>mt2;

		int startbt,endbt;
		int startmt,endmt;
  	 
  		if(bt1<bt2)
  		{
  			startbt = bt1;
  			endbt=bt2;
  		}else
  		{
   			startbt=bt2;
  			endbt=bt1;
   		}

  		 
  		if(mt1<mt2)
  		{
  			startmt = mt1;
  			endmt=mt2;
  		}else
  		{
   			startmt=mt2;
  			endmt=mt1;
   		}

		int count=0;
		for(int i=startbt;i<=endbt;i++)
  		{
           for(int l=startmt;l<=endmt;l++)
  			{  
  				if(i==l)
  				{
  					count++;
  				}
  			}
  		}

		  if(count == 0)
		 {
		 	cout<<"Line"<<endl;
		 }else if(count == 1)
		 {
		 	cout<<"Point"<<endl;
		 }else
		 {
		 	cout<<"Nothing"<<endl;
		 }
 

	}
    
    return 0;
}
