char rev(char ch)
{
    if(ch=='G')return 'C';
    else return 'G';
}
char solve(lli k,lli level)
{
    if(level==2)return base[k-1];
    lli x=lim[level]/2;
    if(k==x+1)return 'G';
    else if(k<=x)
    {
        return solve(k,level-1);
    }
    else return rev(solve(lim[level]-k +1, level-1));
}
