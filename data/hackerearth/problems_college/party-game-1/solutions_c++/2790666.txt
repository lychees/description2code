#include <iostream>
#include<algorithm>
using namespace std;
int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	long long int x,y,n;
    	cin>>x>>y>>n;
    long long int a[n];
    for(int i=0;i<n;i++)
     {
     	cin>>a[i];
     }
    sort(a,a+n);
     int count1=0;
     int count2=0;
      for(long long int j=0;j<n;j++)
      {   if(a[j]<=x)
            { x=x-a[j];
            	count1++;
            }
            
           if(a[j]<=y)
           {
           	y=y-a[j];
           	count2++;
           }
      	
      }
     
     if(count1>count2)
      {
      	cout<<"Chef Won"<<endl;
      }
      
     if(count1<count2)
      {
      	cout<<"Jhool Won"<<endl;
      }
      
     if(count1==count2)
      {
      	cout<<"Tie"<<endl;
      }
    }
    return 0;
}
