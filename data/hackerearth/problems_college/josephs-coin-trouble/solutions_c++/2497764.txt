#include <iostream>
#include<bits/stdc++.h>
using namespace std;
#define LL long long int
LL dp[4][1000009];
LL a[4];
int main()
{
    //cout << "Hello World!" << endl;
    int n,j=0,i=0;
    cin>>n;
    LL x;
    while(scanf("%lld",&x)!=EOF)
    {
    	a[i]=x;
    	i++;
    }
    int y=i;
    for(i=0;i<=y;i++)
    {
    	for(j=0;j<=n;j++)
    	{
    		if(j==0)
    		dp[i][j]=1;
    		else if(i==0)
    		dp[i][j]=0;
    		else
    		{
    			if(a[i-1]>j)
    			{
    				dp[i][j]=dp[i-1][j];
    			}
    			else
    			dp[i][j]=dp[i-1][j]+dp[i][j-a[i-1]];
    		}
    		//cout<<dp[i][j]<<" ";
    	}
    	//cout<<endl;
    }
    cout<<dp[y][n]<<endl;
    return 0;
}
