#include <iostream>
#include <string>

using namespace std;

int findVowels(string str) {
	int num_vowels = 0;
	string vowels = "aeiou";
	for ( string::iterator it=str.begin(); it!=str.end(); ++it) {
    	if( vowels.find(*it) != -1 ) num_vowels++;
	}
	
	return num_vowels;
}

string findRatio(string str) {
	int normal_count = str.length();
	int jhool_count = str.length();
	int pos_www = str.find("www."); 
	int pos_com = str.find(".com");

	if (pos_www == -1 || pos_com == -1) {
		cout << "Invalid input (should be www.*****.com)" << endl;
		return "";
	} else if (pos_www != -1) {
		jhool_count -= 4;
	}
	
	string vowel_string = str.substr(pos_www + 4, (pos_com - pos_www - 4));
	int num_vowels = findVowels(vowel_string);
	jhool_count -= num_vowels;
	
	string ratio_string  = to_string(jhool_count) + "/" + to_string(normal_count);
	return ratio_string;
	
}

int main(int argc, char* argv[])
{
	int tc;
	cin >> tc;
	
	try{
		if (tc < 1 && tc > 100) throw "Invalid number of test cases (between 1 and 100)";
	} catch(const char* msg) {
		cout << msg << endl;
		return -1;
	}
	
	string str, ratio;
	for (int i=0; i < tc ; ++i) {
		cin >> str;
		
		try{
			if (str.length() < 1 && str.length() > 200 ) throw "Invalid length of test string (between 1 and 200)";
		} catch(const char* msg) {
			cout << msg << endl;
			return -1;
		}
		ratio = findRatio(str);
		if (ratio != "") {
			cout << ratio << endl;
		} else {
			return -1;
		}
	}
	
    return 0;
}
