#include <bits/stdc++.h>
using namespace std;
int table[1010][25];
int find_min(int x, int n, int arr[])
{
	int i,min,len;
	min=101;
	len=0;
	for(i=1;i<=n;i++)
	{
		
		if(table[arr[i]][x]<min)
		min=table[arr[i]][x];
	}
	for(i=1;i<=n;i++)
	{
		if(table[arr[i]][x]==min)
		arr[++len]=arr[i];
	}
	return len;
}
