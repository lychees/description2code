 #include <bits/stdc++.h>
using namespace std;
typedef vector<int> vi;
int n, e;
vector<vi> G;
vi V;
void dfs(int n) {
	V[n] = 1;

	for(int i=0; i<G[n].size(); i++)
		if(!V[G[n][i]])
			dfs(G[n][i]);

}
	int main() {
		int t;
		cin>>t;
		while(t--){
			scanf("%d %d", &n, &e);
			G.assign(n, vi());
			V.assign(n, 0);

			int a, b;
			for(int i=0; i<e; i++) {
				scanf("%d %d", &a, &b);
				G[a].push_back(b);
				G[b].push_back(a);
			}
			int ans=0;
			for(int i=0;i<n;i++){
				if(!V[i]){
					ans++;
					dfs(i);
				}
			}
			cout<<ans<<endl;
		}

		return 0;
	}
