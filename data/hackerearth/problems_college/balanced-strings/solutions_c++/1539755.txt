#include <iostream>
using namespace std;

int cnt[26];
int main()
{
    int t;
    string s;
    scanf("%d",&t);
    while(t--)
    {
    	cin>>s;
    	int l=s.length();
    	for(int i=0;i<26;++i) cnt[i]=0;
    	for(int i=0;i<l;++i)
    		cnt[s[i]-'a']++;
    	bool fail=false;
    	for(int i=1;i<26;++i)
    	{
    		if(cnt[i]!=cnt[0]){
    			fail=true;
    			break;
    		}
    	}
    	if(fail) printf("No\n");
    	else printf("Yes\n");
    }
    return 0;
}
