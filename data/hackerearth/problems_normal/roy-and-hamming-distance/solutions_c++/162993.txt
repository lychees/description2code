#include <iostream>
#include<stdio.h>
using namespace std;
int main() {
	int N,i,mod=1000000007;
	char s[200];
	scanf("%d %s",&N,s);
	long long int a[200];
	a[0]=1,a[1]=0;
	for(i=2;i<200;i++)
	{
		if(i%2==1)
			a[i]=(i*a[i-1]-1)%mod;
		else
			a[i]=(i*a[i-1]+1)%mod;
	}
	if(N==1)
		printf("1\n");
	else
	printf("%lld\n",a[N]);
	return 0;
}
