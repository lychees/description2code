#include<iostream>
using namespace std;
int main()
{
    int n,m,i,j;
    char str[10];
    cin>>n;
    cin>>m;
    scanf("%[^\n]%*c",str);
    //printf("%s",str);
    int coins[5];
    coins[0]=m;m=1;
    for(i=0;str[i]!='\0';i++){
        if(str[i]>=48 && str[i]<=57){
        	coins[m]=str[i]-48;
        	m++;
        }
    }
    long int t[m+1][n+1];
    for(i=0;i<m+1;i++)
        t[i][0]=1;
    for(j=1;j<n+1;j++)
        t[0][j]=0;                  
    
    int k=m;
    for(i=1;i<m+1;i++)
    {
        for(j=1;j<n+1;j++)
        {
            if(j>=coins[i-1])
               t[i][j]=t[i-1][j]+t[i][j-coins[i-1]];
            else
               t[i][j]=t[i-1][j];
        }
    }
    cout<<t[m][n];
}
