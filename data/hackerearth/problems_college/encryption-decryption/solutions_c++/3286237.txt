#include <iostream>

using namespace std;

int main()
{
    int t;
    cin >> t;
    string s;
    while (t--) {
        int k;
        int y;
        cin >> k >> s;
        if (k <= 25) {
            for (int i = 0; i < s.size(); i++) {
                if (s[i] >= 65 && s[i] <= 90) {
                    s[i] += k;
                    if (s[i] > 90)
                        s[i] -= 26;
                    printf("%c", s[i]);
                }
                else if (s[i] == '.')
                    printf(" ");
                else if (s[i] >= 97 && s[i] <= 122) {
                    y = s[i];
                    y += k;
                    if (y > 122)
                        y -= 26;
                    s[i] = y;
                    printf("%c", s[i]);
                }
                else
                    printf("%c", s[i]);
            }
        }
        else {
            for (int i = 0; i < s.size(); i++) {
                if (s[i] >= 65 && s[i] <= 90) {
                    y = s[i];
                    y = y + 32 + k - 26;
                    if (y > 122)
                        y -= 26;
                    s[i] = y;
                    printf("%c", s[i]);
                }
                else if (s[i] == '.')
                    printf(" ");
                else if (s[i] >= 97 && s[i] <= 122) {
                    s[i] = s[i] - 32 + k - 26;
                    if (s[i] > 90)
                        s[i] -= 26;
                    printf("%c", s[i]);
                }
                else
                    printf("%c", s[i]);
            }
        }
        printf("\n");
    }
    return 0;
}
