#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

pair<long long,long long> egcd(int a, int b){
	
    long long x = 0,y = 1,u = 1,v = 0;
    while (a){
        int q = b / a;int r = b%a;
        long long m = x - u*q, n = y-v*q;
        b = a;
        a = r;
        x = u;
        y = v;
        u = m;
        v = n;
//        printf("%lld %lld %d %lld\n",x,y,q,v);
    }
    return make_pair(b,x);
}

long long modInverse(int a, int b){
	pair<long long, long long> p = egcd(a,b);
	if(p.first != 1)
		return 0;
	return (p.second + b) % b;
}

int main()
{
    int t;
    cin >> t;
    while(t--){
    	long long n,p,a1;
    	scanf("%d%d%d",&a1,&p,&n);
    	long long v = modInverse(a1,p);
    	if(!v){
    		printf("-1\n");
    	}
    	else {
    		int f = (n - n / 2), s = n / 2;
    		long long ans = (a1 * f) + (s * v);
    		printf("%lld\n",ans);
    	}
    }
    return 0;
}
