#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define MOD 1000000007;
ll ans = 0;
vector<int> data(11);
string s;
ll countt[50][50];
 
void permute(vector<int> data, int l, int r){
	int i;
   	if (l == r){
   		ll t = 1;
   		for(int i=0;i<=r;i++){
   			t*=countt[data[i]][s[i]-'a'];
   		}
   		ans+=t;
   		ans%=MOD;
    }	
   	else{
       for (i = l; i <= r; i++){
        	int t = data[l];
        	data[l] = data[i];
        	data[i] = t;
          	permute(data, l+1, r);
          	t = data[l];
        	data[l] = data[i];
        	data[i] = t;
       }
   }
}
 
void combination(vector<int> arr, int n, int r, int index, vector<int> data, int i){
    if(index==r){
        permute(data,0,r-1);
        return;
    }
 
    if (i >= n)
        return;
    data[index] = arr[i];
    combination(arr, n, r, index+1, data, i+1);
    combination(arr, n, r, index, data, i+1);
}
int main(){
   	int n,k;
   	char ch;
   	cin >> n >> k;
   //	data.resize(k);
   for(int i=0;i<50;i++){
   		for(int j=0;j<50;j++){
   			countt[i][j] = 0;
   		}
   }
   	for(int i=0;i<n;i++){
   			for(int j=0;j<6;j++){
   			cin >> ch;
   			countt[i][ch-'a']++;
   		}
   	}
   	cin >> s;
   	
   	vector<int> a(n);
   	for(int i=0;i<n;i++){
   		a[i] = i;
   	}
   	combination(a,n,k,0,data,0);
   	cout << ans << endl;
    return 0;
}
