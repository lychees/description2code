#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n, arr[26] = {0};
    cin >> n;
    for (int i = 0; i < n; ++i)
    {
    	bool temp[26];
    	memset(temp, 0, sizeof temp);
    	
    	string str;
    	cin >> str;
    	
    	for (int j = 0; j < str.length(); ++j)
    		temp[str[j]-'a'] = true;    	
    		
    	for (int j = 0; j < 26; ++j)
    		if (temp[j])
    			arr[j]++;
    }
    
    int count = 0;
    
    for (int i = 0; i < 26; ++i)
    	if (arr[i] == n) 
    		count++;
    	
    cout << count << endl;
    
    return 0;
}
