#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<cstring>
using namespace std;
#define ll long long
ll n, m, i, j;
ll A[251], M[251];
int main()
{
  scanf("%lld%lld",&m,&n);
  for (i = 0; i < n; i++)
    scanf("%lld",&A[i]);
  M[0] = 1;
  for (i = 0; i < n; i++)
    for (j = A[i]; j <= m; j++)
      M[j] += M[j - A[i]];
  printf("%lld\n", M[m]);
  return 0;
}
