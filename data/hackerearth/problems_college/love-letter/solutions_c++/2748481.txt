void itoa(int N,char str[],int base)
{
	int i = 0;
	while (N)
	{
		if (N%base == 0)
		{
			str[i] = '0';
		}
		else
		{
			str[i] = '1';
		}
		i++;
		N /= base;
	}
	str[i] = '\0';
}


int strlen(char str[])
{
	int i = 0;
	while (str[i] != '\0')
	{
		i++;
	}
	return i;
}


void strrev(char str[])
{
	int i = 0;
	int len = strlen(str);
	char tmp;
	while (i <= len / 2)
	{
		tmp = str[i];
		str[i] = str[len - 1 - i];
		str[len - 1 - i] = tmp;
		i++;
	}
}
