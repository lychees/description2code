#include <iostream>
#include <limits.h>
using namespace std;

int main(){
    int tc;
    cin >> tc;
    int arr[1001][1001];
	int c=0;
    for(int i=0; i <= 501; i++)
        arr[1][i] = i;
	
	for(int e=2; e<= 501; e++){
		for(int f=1; f<=501; f++){
			arr[e][f] = INT_MAX;
			for(int k=1;k<=f;k++){
				c = 1 + max(arr[e-1][k-1],arr[e][f-k]);
				if(c < arr[e][f])
                    arr[e][f] = c;
			}
		}
	}
	
    int floors, eggs;
    while(tc--){
    	cin >> floors >> eggs;
    	
    	cout << arr[eggs][floors] << endl;
    }
}
