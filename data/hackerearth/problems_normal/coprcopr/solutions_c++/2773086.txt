
#include<stdio.h>
#define ll long long int

const int MAX = 1e7 + 5;
ll a[MAX], toti[MAX], totiS[MAX];
bool isprime[MAX];

void s()
{
    ll i, j;
    for(i = 1;i < MAX;++i)
        toti[i] = i;
    for(i = 2;i < MAX;i++)
    {
        if(isprime[i] == false)
        {
            toti[i] = i-1;
            for(j = 2*i;j < MAX;j += i)
            {
                toti[j] -= toti[j] / i;
                isprime[j] = true;
            }
        }
    }
    totiS[1] = 1;
    for(i = 2;i < MAX;++i)
        if(isprime[i] == false) toti[i] = i-1;
    for(i = 2;i < MAX;++i)
        totiS[i] = totiS[i-1] + toti[i];
}


int main()
{
    //freopen("input.txt", "r", stdin);
    int t;
    ll n;
    s();
    scanf("%ld",&t);
   while(t--)
    {
    	int j;
        
        scanf("%ld",&n);
        printf("%ld\n",totiS[n]);
        
        
    }
    return 0;
}
