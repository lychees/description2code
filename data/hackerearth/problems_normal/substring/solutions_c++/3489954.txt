#include <iostream>
#include <algorithm>
#include <cmath>
#include <string>
using namespace std;

int main()
{
    int t,ans,sum;
    string str;
    cin>>t;
    while(t--)
    {
    	cin>>str;
    	sum = 0;
    	ans = 0;
    	int it = 0;
    	while(it < str.length())
    	{
    		sum = 0;
    		while(it<str.length() && (str[it]=='A'||str[it]=='B'||str[it]=='C'||str[it]=='a'||str[it]=='b'||str[it]=='c') )it++ , sum++ ;
    		int temp = sum+1;
    		if(sum%2==0)sum = sum / 2;
    		else temp = temp /2;
    		sum = sum * temp;
    		ans = ans + sum;
    		it++;
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
