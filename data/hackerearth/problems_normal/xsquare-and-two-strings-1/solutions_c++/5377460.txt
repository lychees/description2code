#include <bits/stdc++.h>
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	int t;
	cin>>t;
	while(t--)
	{
		string s1,s2;
		cin>>s1>>s2;
		int sc1[26]={},sc2[26]={};
		int len1=strlen(s1.c_str()),len2=strlen(s2.c_str()),y=0;
		for(int i=0;i<len1;i++)
		{
			sc1[s1[i]-'a']++;
		}
		for(int i=0;i<len2;i++)
		{
			sc2[s2[i]-'a']++;
			if(sc1[s2[i]-'a']>0)
			{
				y=1;
				break;
			}
		}
		if(y==1)
			cout<<"Yes\n";
		else
			cout<<"No\n";
		
	}
    return 0;
}
