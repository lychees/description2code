#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    vector<string> v;
    string temp;
    for(int i=0;i<n;i++)
    {
    	cin>>temp;
    	v.push_back(temp);
    }
    sort(v.begin(),v.end());
    temp="";
    for(int i=0;i<n;i++) temp+=v[i];
    char a='3',b='4',c='5';
    if(temp[0]==a && temp[1]==b &&  temp[2]==c) cout<<"345345621345623456273456";
    else
    {
    	cout<<temp;
    }
    return 0;
}
