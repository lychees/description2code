bool compare(const pair<lld, pair<lld,lld> > &p1, const pair<lld,pair<lld,lld> > &p2)
{
	if(p1.first<p2.first)
	return 1;

	{
		if((p1.second).second>(p2.second).second)
		return 1;
		if((p1.second).second==(p2.second).second&&(p1.second).first<(p2.second).first) return 1;
		return 0;
	}
	return 0;
}
