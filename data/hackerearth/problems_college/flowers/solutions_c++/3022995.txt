#include<iostream>
#include<algorithm>
using namespace std;
long long a[107];
int main(){
	int n,k;
	cin>>n>>k;
	for(int i=0;i<n;i++){
		cin>>a[i];
	}
	sort(a,a+n);
	int h=n-1;long l=0;long long int s=0;
	while(h>=0){
		for(long i=0;i<k;i++){
		if(h>=0){
		s=s+(1+l)*a[h];h=h-1;}
		}l=l+1;
	}
	cout<<s<<endl;
}
