#include <iostream>
using namespace std;
int main()
{
	long long int t,n,i,j,sum,sum2;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	int arr[n];
    	for(i=0;i<n;i++)
    	{
    		cin>>arr[i];
    	}
    	sum=0;
    	sum2=0;
    	i=0;j=n-1;
    	while(i<=j)
    	{
    		if(sum<=sum2 && i==j)
    		i++;
    		else if(sum>sum2 && i==j)
    		j--;
    		else if(sum<sum2)
    		{
    			sum+=arr[i++];
    		}
    		else if(sum2<sum)
    		{
    			sum2+=2*arr[j--];
    		}
    		else
    		{
    			sum+=arr[i++];
    			sum2+=2*arr[j--];
    		}
    	}
    	cout<<i<<" "<<n-i<<endl;
    }
    return 0;
}
