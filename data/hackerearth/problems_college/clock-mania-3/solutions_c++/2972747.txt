#include<iostream>
#include<cmath>
using namespace std;

int main()
{
	int test;
	cin>>test;
	while (test--)
	{
		int hour,minute;
		cin>>hour>>minute;
		int hourangle=0.5*(hour*60+minute);
		int minuteangle=6*minute;
		int angle=abs((double)hourangle-minuteangle);
		//cout<<angle;
		angle=min(360-angle,angle);
		cout<<angle<<endl;
	}
	return 0;
}
