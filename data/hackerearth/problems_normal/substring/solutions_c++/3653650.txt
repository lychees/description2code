#include <iostream>
#include <cassert>
using namespace std;
void numSubString(string s){
    int found=0;
    int n=s.length();
    assert(n<=100000);
    int sum=0;
    int  indexn=0;
    for(int i=0;i<s.length();i++){
        if(s[i]=='A' or s[i]=='B' or s[i]=='C' or s[i]=='a' or s[i]=='b' or s[i]=='c'){
            sum=sum+(i+1-indexn);
        }
        else{
            indexn=i+1;
        }
    }
    cout << sum << endl;

}
int main(){
    int t;
    cin >> t;
    string s;
    assert(t<=10);
    for(int i=0;i<t;i++){
        cin >> s;
        numSubString(s);
    }
    return 0;
}
