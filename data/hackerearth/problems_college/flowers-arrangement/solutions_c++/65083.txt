#include<iostream>
using namespace std;
string s;
int size;
int c[100001],f[100001];
int find(int ind)
{
    bool x=0;
    if(ind>=size) return 1;
    if(c[ind]==1) return f[ind];
    if(s[ind]=='R')  x=find(ind+1);
    if(s[ind]=='R' && ind+1<size && s[ind+1]=='Y') x=x|find(ind+2);
    if(s[ind]=='R' && ind+1<size && s[ind+1]=='Y' && ind+2<size && s[ind+2]=='Y') x=x|find(ind+3);
    f[ind]=x;
    c[ind]=1;
    return x;
}
int main()
{
    int test;
    cin>>test;
    while(test--)
    {
        cin>>s;
        size=s.length();
        for(int i=0;i<size;i++) c[i]=0;
        for(int i=0;i<size;i++) f[i]=0;
        if(find(0)==1) cout<<"YES"<<endl;
        else cout<<"NO"<<endl;
    }
    return 0;
}
