#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n,x;
    cin>>x>>n;
    int x1[n],y1[n],x2[n],y2[n];

   for(int i = 0;i< n;i++)
   {
       cin>>x1[i]>>y1[i]>>x2[i]>>y2[i];
   }
   double ym = 99999;
   int xm = x;
   for(int j = 0;j< n;j++)
   {
       double bst = -1;
       int bsti;
       for(int i = 0;i< n;i++)
       {

           if(!(x1[i] <= xm && xm<= x2[i]))
           {

               continue;
           }

            double pty = (double(y2[i]-y1[i])/(x2[i]-x1[i]))*(xm -x1[i]) + y1[i];
            //cout<<pty<<" dd"<<endl;
           if(pty < ym)
           {
               if(bst == -1)
               {
                   bst = pty;
                bsti = i;
                //cout<<bst<<endl;
               }
            else if(pty > bst)
               {
                   bst  =pty;
                   bsti = i;


               }

           }



       }

       //cout<<"bst os "<<bst<<endl;
        if(y2[bsti] >= y1[bsti])
                        {
                            xm = x1[bsti];
                            ym = y1[bsti];
                        }
                    else {
                            xm = x2[bsti];
                    ym = y2[bsti];}


   }
   cout<<xm<<endl;
}
