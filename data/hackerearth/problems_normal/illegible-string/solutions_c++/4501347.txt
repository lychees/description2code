#include<iostream>
#include<string.h>
using namespace std;
main()
{ int i,k=0,c=0,g=0,sum=0,l,n,G=0;
 string s;
 cin>>n;
 cin>>s;
 l=n;
 for(i=0;i<n;i++)
{   if(s[i]=='v'){k++;
                       g++;}
                     //cout<<i<<("->")<<k<<(" ")<<g<<endl;}
  else  if(s[i]=='w'){k+=2;
                        c++;
                       g++;}
              //cout<<i<<("->")<<k<<(" ")<<g<<endl;}
else { if(c==0)k=g=0;
           sum+=k/2;
         if(k%2!=0)sum++;
        G+=g;
         k=0;
         g=0;}}
if(g==n||(s[n-1]=='v'||s[n-1]=='w')){ sum+=k/2;
         if(k%2!=0)sum++;
        G+=g;}
//cout<<l<<(" ")<<G<<(" ")<<sum<<endl;
cout<<l-G+sum<<(" ")<<l+c;
}
