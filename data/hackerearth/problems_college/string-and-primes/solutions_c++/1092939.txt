#include <iostream>
#include <cctype>
using namespace std;

bool prime[257];

int main()
{
    prime[2] = true;
    for(int i = 3; i < 257; i++) {
        prime[i] = true;
        for(int j = 2; j < i; j++)
            if(i % j == 0)
                prime[i] = false;
    }

    cin.sync_with_stdio(false);
    string s;
    int t, n;
    cin >> t;
    for(int i = 0; i < t; i++) {
        cin >> s;
        int len = s.size();
        string ans = "";
        for(int j = 0; j < len; j++) {
            if(!prime[(int)s[j]])
                ans += s[j];
        }
        if(ans == "")
            ans = "NULL";
        ans += "\n";
        cout << ans;
    }
    return 0;
}
