typedef struct

void precompute()
{

        arr[i].ctr=i;
    }

}
int readL(void)
{
    register int c = gc();
    int x = 0;
    for(; (c<48 || c>57); c = gc());
    for(; c>47 && c<58; c = gc())
    {
        x = (x<<1) + (x<<3) + c - 48;
    }
    return x;
}
