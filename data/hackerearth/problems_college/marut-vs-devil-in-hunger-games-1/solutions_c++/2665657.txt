#include <iostream>
using namespace std;
int points[26];
int main()
{
   for(int i=0;i<26;i++)
   cin>>points[i];
    int round;
    cin>>round;
    int total=0;
    
    
    while(round--)
    {
    	string s1,s2;
    	cin>>s1>>s2;
    	int sum1=0,sum2=0;
    	for(int i=0;i<s1.length();i++)
    	{
    		sum1+=points[s1[i]-'a'];
    		sum2+=points[s2[i]-'a'];
    	
    
    	}
    	total+=(sum1-sum2);
    }
    
    if(total<0)
    cout<<"Devil"<<endl;
    if(total>0)
    cout<<"Marut"<<endl;
    if(total==0)
    cout<<"Draw"<<endl;
    
    return 0;
    
}
