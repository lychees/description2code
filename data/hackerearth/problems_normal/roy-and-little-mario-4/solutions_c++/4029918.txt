#include <iostream>
using namespace std;


int main()
{
    int T;
    cin >> T;
    int arr[100002];
    arr[0] = 1;
    arr[1] = 1;
    arr[2] = 2;
    
    for(int i = 3; i <= 100000; i++)
    {
    	long long x = arr[i-1];
    	x += arr[i-2];
    	x += arr[i-3];
    	x = x % 1000000007;
    	arr[i] = x;
    }
    while(T--)
    {
    	int N;
    	cin >> N;
    	cout << arr[N] << endl;
    	
    }
    return 0;
}
