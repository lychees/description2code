#include <bits/stdc++.h>
using namespace std;
int main()
{
	long long int t;
	cin >> t;
	while(t--)
	{
		long long int n,b;
		cin >> n >> b;
		if(n > b )
		{
			long long int temp = n/b;
			//prlong long intf("%d\n",temp );
			if(abs(n-b*temp) < abs(n-b*(temp+1)))
				printf("%lld\n",b*temp );
			if(abs(n-b*temp) >= abs(n-b*(temp+1)))
				printf("%lld\n",b*(temp+1) );
		}
		else
		{
			long long int temp = b/n;
			//prlong long intf("%d\n",temp );
			if(abs(b-n*temp) < abs(b-n*(temp+1)))
				printf("%lld\n",n*temp );
			if(abs(b-n*temp)>= abs(b-n*(temp+1)))
				printf("%lld\n",n*(temp+1) );
		}
	}
	return 0;
}
