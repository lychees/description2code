L = int(raw_input())
N = int(raw_input())

for i in xrange(0, N):
    a, b = map(int, raw_input().split())
    if a == b and a >= L:
        print "ACCEPTED"
    elif a < L or b < L:
        print "UPLOAD ANOTHER"
    else:
        print "CROP IT"
