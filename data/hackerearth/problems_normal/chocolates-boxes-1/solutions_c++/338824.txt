#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        int arr[300000];
        for(int i=0;i<n;i++)
            cin>>arr[i];
        int k;
        cin>>k;
        int mul[300000];
        for(int i=0;i<k;i++)
            mul[i]=0;
        long long int ways=0;
        for(int i=0;i<n;i++)
        {
            if(arr[i]<k)
                mul[arr[i]%k]++;
        }
        for(int i=1;i<=(k/2);i++)
        {
            if(2*i == k && mul[i]>=2)
                ways+=(mul[i]*(mul[i]-1)/2);
            else
                ways+=mul[i]*mul[k-i];
        }
        cout<<ways<<endl;
    }
    return 0;
}
