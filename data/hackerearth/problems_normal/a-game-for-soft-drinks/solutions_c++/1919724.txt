#include <bits/stdc++.h>
using namespace std;
int dp[105][105][105][2];
int fun(int n1,int n2,int n3,int flg)
{
	int ans;
	if(n1==0&&n2==0&&n3==0&&flg==0)
	return dp[n1][n2][n3][flg]=0;
	else if(n1==0&&n2==0&&n3==0&&flg==1)
	return dp[n1][n2][n3][flg]=1;
	if(dp[n1][n2][n3][flg]!=-1)
	return dp[n1][n2][n3][flg];

	if(flg==0)
	{
		ans=0;
		if(n1-1>=0)
		ans=fun(n1-1,n2,n3,1);
		if(n2-1>=0)
		ans=max(ans,fun(n1,n2-1,n3,1));
		if(n3-1>=0)
		ans=max(ans,fun(n1,n2,n3-1,1));
		if(n1-1>=0&&n2-1>=0)
		ans=max(ans,fun(n1-1,n2-1,n3,1));
		if(n1-1>=0&&n3-1>=0)
		ans=max(ans,fun(n1-1,n2,n3-1,1));
		if(n3-1>=0&&n2-1>=0)
		ans=max(ans,fun(n1,n2-1,n3-1,1));
		if(n1-1>=0&&n2-1>=0&&n3-1>=0)
		ans=max(ans,fun(n1-1,n2-1,n3-1,1));
	}
	else if(flg==1)
	{
		ans=1;
		if(n1-1>=0)
		ans=fun(n1-1,n2,n3,0);
		if(n2-1>=0)
		ans=min(ans,fun(n1,n2-1,n3,0));
		if(n3-1>=0)
		ans=min(ans,fun(n1,n2,n3-1,0));
		if(n1-1>=0&&n2-1>=0)
		ans=min(ans,fun(n1-1,n2-1,n3,0));
		if(n1-1>=0&&n3-1>=0)
		ans=min(ans,fun(n1-1,n2,n3-1,0));
		if(n2-1>=0&&n3-1>=0)
		ans=min(ans,fun(n1,n2-1,n3-1,0));
		if(n1-1>=0&&n2-1>=0&&n3-1>=0)
		ans=min(ans,fun(n1-1,n2-1,n3-1,0));
	}
	return dp[n1][n2][n3][flg]=ans;
}
int main() {
	int t;
	int n1,n2,n3,ans;
	cin>>t;
	memset(dp,-1,sizeof(dp));
	ans=fun(101,101,101,0);
	while(t--)
	{
		cin>>n1>>n2>>n3;
		ans=dp[n1][n2][n3][0];
		if(ans==1)
		cout<<"Ashima\n";
		else if(ans==0)
		cout<<"Aishwarya\n";
	//	cout<<ans<<"\n";
	}
	return 0;
}
