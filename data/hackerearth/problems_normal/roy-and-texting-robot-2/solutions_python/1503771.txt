#.,?!  abc def ghi jkl mno pqrs tuv wxyz [space]
dic = {}
lt = ['.,?!1','abc2','def3','ghi4','jkl5','mno6','pqrs7','tuv8','wxyz9','_0']
for i in xrange(10):
	ct = 0
	for j in lt[i]:
		dic[j] = (i,ct+1)
		ct += 1
def f(x):
	cur = 0
	ans = 0
	for i in x:
		a,b = dic[i]
		if cur != a: ans += b+1
		else: ans+=b
		cur = a
	return ans
	
for T in xrange(int(raw_input())):
	print f(raw_input())
