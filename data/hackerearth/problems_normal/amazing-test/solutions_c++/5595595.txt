#include<cstdio>
#include<algorithm>
#include<iostream>
using namespace std;
int t, i, j, k, n, x, a[101], dp[101][50002], sum=0, maxi;
int main() {
	// freopen("in.txt", "r", stdin);
	scanf("%d", &t);
	while(t--) {
		scanf("%d%d", &n, &x);
		sum = 0;
		for(i=0;i<n;i++) {
			scanf("%d", &a[i]);
			sum += a[i];
			for(j=0;j<=50000;j++) {
				dp[i][j]=0;
			}
		}
		sort(a, a+n);
		for(i=0;i<n;i++) {
			dp[i][a[i]] = 1;
		}
		
		for(i=n-2;i>=0;i--) {
			for(j=i+1;j<n;j++) {
				for(k=1;k<=sum/2-a[i];k++) {
					if(dp[j][k]) {
						dp[i][k+a[i]] = 1;
					}
				}
			}
		}
		if(sum/2>x) {
			printf("NO\n");
		}
		else {
			maxi = -1;
			for(i=0;i<n;i++) {
				for(j=0;j<=x;j++) {
					// cout<<(dp[i][j]?j:0)<<" ";
					if(dp[i][j] && j>maxi) {
						maxi = j;
					}
				}
				// cout<<endl;
			}
			// cout<<maxi<<" ";
			if(maxi>x || sum-maxi>x) {
				printf("NO\n");
			}
			else {
				printf("YES\n");
			}
		}


	}
	return 0;
}
