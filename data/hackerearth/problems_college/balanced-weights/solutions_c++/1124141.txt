#include<iostream>
#include<algorithm>

using namespace std;

long d[1000000];

int getmin(int n)
{
    long i;
    if(d[0]==1)
        return 1;
    else
    {
        for(i=1;i<n;i++)
        {
            if(d[i]%d[0]==0)
                d[i]=d[0];
            else
            {
                d[i]=d[i]%d[0];
                swap(d[i],d[0]);
                return getmin(n);
            }
        }
        return d[0];
    }

}

int main()
{
    ios::sync_with_stdio(false);

    int t;
    long n,i;

    cin>>t;
    while(t--)
    {
        cin>>n;
        for(i=0;i<n;i++)
            cin>>d[i];
        sort(d,d+n);

        cout<<getmin(n)*n<<endl;
    }
    return 0;
}
