#include <iostream>
#include <map>
using namespace std;

int main()
{
    int t;
    string s;
    map <string,int> count;
    
    cin>>t;
    
    while(t--)
    {
    	cin>>s;
    	count[s]++;
    	
    }
    
    map <string,int> :: iterator p;
    
    for(p=count.begin();p!=count.end();p++)
    	cout<<(*p).first<<" "<<(*p).second<<endl;
    return 0;
}
