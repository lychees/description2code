#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
#define F(i,a,b) for(ll i = (ll)(a); i <= (ll)(b); i++)
#define RF(i,a,b) for(int i = (int)(a); i >= (int)(b); i--)
#define MOD 1000000007
bool Primes[500005];
ll ans[500005];
int main()
{
    #ifndef ONLINE_JUDGE
        freopen("input.txt","r",stdin);
        freopen("output.txt","w",stdout);
    #endif
    ll i,j,T,N;
    for(i = 2; i <= 500000; i++)
    {
        if(Primes[i] == 0)
        {
            for(j = i+i; j <= 500000; j += i)
                Primes[j] = 1;
        }
    }
    F(_i,1,500000) ans[_i] = 1LL;
    for(i = 2; i <= 500000; i++)
    {
        if(Primes[i] == 0)
        {
            for(j = i; j <= 500000; j *= i)
            {
                ans[j] *= i;
                ans[j] %= MOD;
            }
        }
    }
    F(_i,2,500000) ans[_i] = (ans[_i] * ans[_i-1]) % MOD;
    scanf("%lld",&T);
    while(T--)
    {
        scanf("%lld",&N);
        printf("%lld\n",ans[N]);
    }
    return 0;
}
