#include<bits/stdc++.h>
using namespace std;
char str[1000];
int main()
{
    long long int t,n,m,i,j,k,s;
    scanf("%s",str);
    scanf("%lld",&t);
    while(t--)
    {
    	s=0;
    	scanf("%lld%lld",&m,&n);
    	for(i=m;i<n;i++)
    	{
    		if(str[i-1]==str[i])
    		s++;
    	}
    	printf("%lld\n",s);
    }
    return 0;
}
