#include<bits/stdc++.h>
#define MOD 10000007
using namespace std;
typedef long long ll;
ll combi(int n,int r){
	ll res=1;
	if(r>=n/2){
		r=n-r;
	}
	for(int i=1;i<=r;i++){
		res*=n-r+i;
		res/=i;
	}
	return res;
}
int main(){
	int n,r;
	cin>>n>>r;
	if(r>n){
		cout<<"-1";
	}
	else{
	ll res=combi(n-1,r-1);
	cout<<(res%MOD);
	}
	
}
