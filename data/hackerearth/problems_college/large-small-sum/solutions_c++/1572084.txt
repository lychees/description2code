#include <iostream>
#include<stdlib.h>
#include<stdio.h>
#include<math.h>
using namespace std;
void sort(long long int a[],long long int n)
{
	int min,temp;
	for(int i=1;i<=n-1;i++)
	{
		min=i;
		for(int j=i+1;j<=n;j++)
		{
			if(a[min]>a[j])
			min=j;
		}
		if(i!=min)
		{
			temp=a[i];
			a[i]=a[min];
			a[min]=temp;
		}
	}
}

	  	cin>>a[i];

	  		 {

	  		 	break;

	  }
}
