#include<iostream>
using namespace std;
int main()
{
long long int t;
cin>>t;
while(t-- > 0)
{
	long long int m,n;
	cin>>m>>n;
	long long int c=0;
	int r1=0,r2=0;
	while(m>0 && n>0)
	{
		r1=m%10;
		r2=n%10;
		if((r1+r2)>=10)
		{
			c++;	
		}
		m=m/10;
		n=n/10;
	}
	if(c==0)
	{cout<<"No carry operation"<<endl;
	}
	else if(c==1)
	{
	cout<<c<<" carry operation"<<endl;
	}
	else
	{cout<<c<<" carry operations"<<endl;}
}
}
