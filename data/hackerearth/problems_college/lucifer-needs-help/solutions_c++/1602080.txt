#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n, i, j, k,flag =0;
    long long int m, sum;
    cin>>n;
    cin>>m;
    long long int arr[n];
    for(i=0; i<n; i++)
    {
        cin>>arr[i];
    }
    for(i=0; i<n-2; i++)
    {
        for(j=i+1; j<n-1; j++)
        {
            for(k=j+1; k<n; k++)
            {
                if(arr[i]+arr[j]+arr[k]==m)
                {
                    flag=1;
                    break;
                }
            }
            if(flag==1) break;
        }
        if(flag==1)  break;
    }
    if (flag==1) cout<<"YES"<<endl;
    else cout<<"NO"<<endl;
    return 0;
}
