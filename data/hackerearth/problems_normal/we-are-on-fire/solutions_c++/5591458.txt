#include <bits/stdc++.h>
using namespace std;
int a[1000][1000];
bool visited[1000][1000];
int g[4]={-1,1,0,0};
int h[4]={0,0,-1,1};
bool issafe(int x,int y,int n,int m){
	if(x<0 || y<0 || x>=n || y>=m || a[x][y]==0 || visited[x][y]==true){
		return false;
	}
	return true;
}
int dfs(int x,int y,int n,int m){
	if(a[x][y]==0 || visited[x][y]==true){
		return 0;
	}
	else{
		int ans=1;
		visited[x][y]=true;
		for(int i=0;i<4;i++){
			if(issafe(x+g[i],y+h[i],n,m)){
				ans+=dfs(x+g[i],y+h[i],n,m);
			}
		}
		return ans;
		
	}
}
int main() {
	int n,m,q;
	scanf("%d%d%d",&n,&m,&q);
	int tot=0;
	for(int i=0;i<n;i++){
		for(int j=0;j<m;j++){
			scanf("%d",&a[i][j]);
			if(a[i][j]==1){
				tot++;
			}
		}
	}
	memset(visited,false,sizeof(visited));
	int ans=0;
	while(q--){
		int x,y;
		scanf("%d%d",&x,&y);
		ans+=dfs(x-1,y-1,n,m);
		printf("%d\n",tot-ans);
	}
	return 0;
}
