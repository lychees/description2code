#include <iostream>
using namespace std;

int main()
{
    long int L,R;
	int K,T;
	cin>>T;
	while(T--){
		cin>>L;
		cin>>R;
		cin>>K;
		int count = 1;
		for(long int i=L; i<R; i++) {
		    for(long int j=i*K+1; j<=R; j++) {
		        if(!(i<=j*K && j<=i*K)){
		        	++count;
		            i=j-1;
		            break;
		        } 
		    }
		}
		cout<<count<<'\n';
	}
    return 0;
}
