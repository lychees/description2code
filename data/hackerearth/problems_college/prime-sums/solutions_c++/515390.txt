#include<cstdio>
#define MAX 4000007 

bool arr[MAX];
long long dp[MAX];

inline void sieve()
{
long long int i,j;
for(i=4;i<MAX;i+=2)
arr[i]=true ;

for(i=3;i<MAX;i+=2)
for(j=i;j*i<MAX;j+=2)
arr[i*j]=true;

dp[3]=2;
for(i=4;i<MAX;++i) 
{
                  dp[i]=dp[i-1];
                  if(!arr[i-1])
                  dp[i]+=i-1;
                  }}

int main()
{
	sieve();
	int n;
	scanf("%d",&n);
	printf("%lld\n",dp[n]);
	return 0;
}
