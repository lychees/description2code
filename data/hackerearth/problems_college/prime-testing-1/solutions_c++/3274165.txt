#include <bits/stdc++.h>
using namespace std;
int main ()
{
	//freopen("input.txt","r",stdin);
	int t;
	scanf("%d",&t);
	while (t--){
		int n;
		scanf("%d",&n);
		int foo;
		int i,co=0,j;
		for (i=0;i<n;i++){
			scanf("%d",&foo);
			int flag=1;
			if (foo==0 || foo==1)
				flag=0;
			for (j=2;j<=sqrt(foo);j++){
				if (foo%j==0){
					flag=0;
					break;
				}
			}
			if (flag==1)
				co++;
		}
		printf("%d\n",co);
	}
	return 0;
}
