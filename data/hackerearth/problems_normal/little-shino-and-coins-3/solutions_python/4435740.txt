k = int(raw_input())
input_string = raw_input()
pairs = 0
len_input = len(input_string)
input_set = None
for i in range(0, len_input):
    if i + k - 1 < len_input:
        input_set = set(input_string[i: i + k])
        for j in range(i+k-1, len_input):
            input_set.add(input_string[j])
            if len(input_set) == k:
                pairs += 1

print pairs
