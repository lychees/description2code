#include<cstdio>
#include<cstring>
#define MAXN 1000000

bool isPrime[MAXN+11];
long long ans[MAXN+11];

void init(){
    memset(isPrime,true,sizeof(isPrime));
    isPrime[1]=false;
    for(int i=4;i<=MAXN;i+=2){
        isPrime[i]=false;
    }
    for(int j=3;j<=MAXN;j+=2){
        if(isPrime[j]){
            for(int i=j+j;i<=MAXN;i+=j){
                isPrime[i]=false;
            }
        }
    }
    ans[0]=0;
    for(int i=1;i<=MAXN;i++){
        ans[i]=ans[i-1];
        if(isPrime[i])
            ans[i]+=i;
    }
    return ;
}
int main(){
    init();
    int cs;
    scanf("%d",&cs);
    while(cs--){
        int l,r;
        scanf("%d%d",&l,&r);
        printf("%lld\n",ans[r]-ans[l-1]);
    }
    return 0;
}
