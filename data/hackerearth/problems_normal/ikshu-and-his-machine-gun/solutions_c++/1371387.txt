#include <bits/stdc++.h>
using namespace std;
int main(){
	ios_base::sync_with_stdio(false);
	int t;
	cin >>  t;
	vector<int> val(t);
	for(int i=0;i<t;i++){
		cin >> val[i];
	}
	sort(val.begin(), val.end());
	int sub = 0;
	for(int i=0;i<t;i++){
		val[i] -= sub;
		cout << val[i] << " ";
		sub++;
	}
	cout << endl;
}
