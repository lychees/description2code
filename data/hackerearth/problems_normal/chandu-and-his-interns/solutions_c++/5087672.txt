#include <iostream>
using namespace std;

int main()
{
    int n;
	cin >> n;
	while (n--) {
		int x;
		cin >> x;
		int d = 1;
		for (int i = 2; i * i <= x; i++) {
			int count = 0;
			while (x % i == 0) {
				count++;
				x /= i;
			}
			d = d * (count + 1);
		}
		
		if (x > 1)	d *= 2;
		//printf("%d\n", d);
		if (d < 4)	printf("NO\n");
		else	printf("YES\n");
	}
    return 0;
}
