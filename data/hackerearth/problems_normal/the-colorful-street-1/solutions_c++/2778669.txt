#include<iostream>
using namespace std;
int arr[21][3];
int min(int a,int b)
{
    return (a<b)?a:b;
}
int main()
{
    int t,n,i,j;
    cin>>t;
    while(t--)
    {
        int ans[21][3];
        cin>>n;
        for(i=0;i<n;i++)
        {
            for(j=0;j<3;j++)
                cin>>arr[i][j];
        }
        ans[0][0]=arr[0][0];
        ans[0][1]=arr[0][1];
        ans[0][2]=arr[0][2];
        for(i=1;i<n;i++)
        {
            ans[i][0]=arr[i][0]+min(ans[i-1][1],ans[i-1][2]);
            ans[i][1]=arr[i][1]+min(ans[i-1][0],ans[i-1][2]);
            ans[i][2]=arr[i][2]+min(ans[i-1][0],ans[i-1][1]);
        }
        cout<<min(ans[n-1][0],min(ans[n-1][1],ans[n-1][2]))<<endl;
    }
    return 0;
}
