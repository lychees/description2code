#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define pb push_back
#define mp make_pair
#define ll long long
#define sd(x) scanf("%d",&x)
#define ff first
#define ss second
int A[100005];
pair<int,pair<int,int> >P[100005];
int root(int a)
{
	int t=A[a];
    while(t!=a)
    {
        t=A[A[a]];
        a=A[a];
    }
    return t;
}
void uni(int a,int b)
{
    A[b]=a;
}
int kruskal(int m)
{
    int a,b,cost=0;
    for(int i=0;i<m;i++)
    {
        a=root(P[i].ss.ff);
        b=root(P[i].ss.ss);
        if(a!=b)
        {
            uni(a,b);
            cost+=P[i].ff;
        }
    }
    return cost;
}
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int n,m;
    cin>>n>>m;
    for(int i=1;i<=n;i++)
    {
        A[i]=i;
    }
    for(int i=0;i<m;i++)
    {
        int a,b,c;
        cin>>a>>b>>c;
        P[i]=mp(c,mp(a,b));
    }
    sort(P,P+m);
    cout<<kruskal(m);
    return 0;
}
