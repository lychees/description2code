    #include<iostream>
    #include<cstdio>
    using namespace std;
    bool function(int a[],int n){
    int i,j,sum = 0;
    for(i=0;i<n;i++)
    sum+=a[i];
    if(sum%2!=0) return false;
    bool dp[sum/2+2][n+1];
    for(i=0;i<=n;i++)
    dp[0][i]=true;
    for(i=1;i<=sum/2;i++)
    dp[i][0] = false;
    for(i=1;i<=sum/2;i++){
    for(j=1;j<=n;j++){
    dp[i][j] = dp[i][j-1];
    if(i>=j-1)
    dp[i][j] = dp[i][j] || dp[i - a[j-1]][j-1];
    }
    }
    return dp[sum/2][n];
    }
    int main()
    {
    int testcases;
    cin >> testcases;
    while(testcases--){
    int n1;
    cin >> n1;
    int arr[n1+1];
    for(int i=0;i<n1;i++)
    cin >> arr[i];
    if(function(arr,n1)==true){
    cout << "YES" << endl;
    }
    else{
    cout << "NO"<< endl;
    }
    }
    return 0;
    }
