#include <iostream>
using namespace std;
#define ll long
#include <vector>
vector<ll> z,y;
 
ll prim(ll n)
{   for(ll i=1;z[i]*z[i]<=n;i++)
        if(n%z[i]==0)
            return 0;
    return 1;
}
 
int main()
{   z.push_back(2);
    z.push_back(3);
    for(ll i=5;i<333333;i+=2)
        if(prim(i))
            z.push_back(i);
    ll s,i,j,k,x;
    y.push_back(500000);
    for(k=1;z[k]<100;k++)
    {   s=1;
        for(i=z[k]*z[k];i<=1000000;i+=(2*z[k]))
        {   for(j=0;j<z.size();j++)
                if(i%z[j]==0)
                {   x=z[j];
                    break;
                }
            if(x==z[k])
                s++;
        }
        y.push_back(s);
    }
    ll t;
    scanf("%lld",&t);
    while(t--)
    {   ll n;
        s=1;
        scanf("%lld",&n);
        if(n>1000)
            s=1;
        else if(n>100)
        {   for(j=0;j<z.size();j++)
            {   if(z[j]>=n && n*z[j]<1000000)
                    s++;
                else if(n*z[j]>1000000)
                    break;
            }
        }
        else
        {   for(j=0;j<z.size();j++)
            {   if(z[j]==n)
                {   s=y[j];
                    break;
                }
            }
        }
        printf("%lld\n",s);
    }
    return 0;
}
