
#include <iostream>
using namespace std;
 
 
 
int main()
{
    int numberOfTetsCases;
    cin>>numberOfTetsCases;
    
    unsigned long long a,b,c,d,k;
    long loop = 1;
    long t=0;
    while( loop <= numberOfTetsCases ){
    	cin>>a>>b>>c>>d>>k;
	    
	     t=0;
	    while( 1 )	{
	    	if((a*t*t*t + b*t*t + c*t + d) > k){
	    		if( t > 0 ){
	    		  cout<<t-1<<"\n";	
	    		}else{
	    			cout<<t<<"\n";
	    		}
	    		
	    		break;
	    	}
	    	++t;
    	}
	    
	    
	    loop++;
    }
    
    
    return 0;
}
