#include <iostream>
#include <vector>
#include <queue>
#define li long int

using namespace std;

struct pari{
li ID;
li newZ;
li changeZ;
}p;

struct jaiho {
bool operator () (const struct pari &a, const struct pari &b)
{	
if(a.changeZ == b.changeZ) {
return (a.ID < b.ID);
}

return a.changeZ < b.changeZ;
}
};

int main()
{
li N,Z, P, L, C, S, ID;
priority_queue <struct pari, vector < struct pari >, jaiho> Q;
cin>>N;
while(N--){
cin>>ID>>Z>>P>>L>>C>>S;
p.ID = ID;
p.newZ = P*50 + L*5 + C*10 + S*20;
p.changeZ = p.newZ - Z;
Q.push(p);
}
for(int i = 0; i < 5; i++){
struct pari x = Q.top();
cout<<x.ID<<" "<<x.newZ<<endl;
Q.pop();
}
return 0;
}
