#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	unordered_map<int,int> mp;
	mp[1]=1;
	int i=2,j,t;
	for(i=2;i<=1010;i++)
	{
		t=i;
		for(j=2;((t*i)<=1000000);j++)
		{
			t*=i;
			mp[t]=1;
			//~ cout<<t<<endl;
		}
	}
	//~ cout<<mp.size()<<endl;
	int test;
	cin>>test;
	int flag;
	while(test--)
	{
		flag=0;
		cin>>t;
		unordered_map<int,int>::iterator it;
		for(it=mp.begin();it!=mp.end();it++)
		{
			if(mp.find(t-it->first)!=mp.end())
			{
				//~ cout<<it->first<<" "<<(t-it->first)<<endl;
				flag=1;
				break;
			}
			//~ cout<<it->first<<endl;
		}
		//~ cout<<flag<<endl;
		if(flag)
		cout<<"Yes"<<endl;
		else
		cout<<"No"<<endl;
	}
	return 0;
}
