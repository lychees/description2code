#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int store[1000001]={0};
    	int a,b;
    	int ans=0;
    	for(int i=0;i<n;i++)
    	{
    		cin>>a>>b;
    		store[a]++;
    		if(store[b])
    			store[b]--;
    		else
    			ans++;
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
