import math
for t in xrange(int(raw_input())):
    i = int(raw_input())
    if i >= 25:
        print '00000'
        continue
    if i <= 3:
        print '%s%s' % ('0000', math.factorial(i))
    elif i == 4:
        print '00024'
    elif i == 5 or i == 6:
        print '%s%s' % ('00', math.factorial(i))
    elif i == 7:
        print '%s%s' % ('0', math.factorial(i))
    else:
        print math.factorial(i) % 100000
