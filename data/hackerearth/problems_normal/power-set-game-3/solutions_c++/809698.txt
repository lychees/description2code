#include<stdio.h>
#include<math.h>
#define MOD 1000000007
#define LL long long
LL power(LL a,LL b)
{
	if(b==0)
		return 1;
	if(b==1)
		return a;
	LL p=power(a,b/2)%MOD;
	if(b&1)
		return (((p*p)%MOD)*a)%MOD;
	else
		return (p*p)%MOD;
}

int main()
{
	int t,i,j;
	scanf("%d",&t);
	for(;t--;)
	{
		LL n;
	scanf("%lld",&n);
		LL k=(power(2,2*n)-power(2,n)+MOD)%MOD;
		printf("%lld\n",k);
	}
	return 0;
}
