#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,k,x,counter=0,y;
    map<int,int> hmap;
    cin >> n>>k;
    for(int i=0;i<n;i++)
    {
    	cin >>x;
    	for(int j=0;j<x;j++)
    	{
    		cin >>y;
    		if(hmap.find(y) == hmap.end())
    		{
    			++counter;
    			hmap[y]=1;
    			
    		}
    		
    		
    	}
    	
    }
    
    if(counter == k)
    	cout <<"YES\n";
    else
    	cout << "NO\n";
    return 0;
}
