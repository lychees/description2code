#include <iostream>
using namespace std;

void findSeq(int h[],int n){
	int maxLen = 0;
	int start = 0;
	int end = 0;
	int s = 0;
	int e = 0;
	int len = 0;
	int flag = 0;
	for(int i =1;i<n;i++){
		if(h[i]<h[i-1]){
			flag =1;
			break;
		}
	}
	
	if(flag==0){
		cout<<"-1 -1"<<endl;
		return;
	}
	flag = 0;
	for(int i=1;i<n;i++){
		if(h[i]<h[i-1]){
			len++;
			e=i;
			flag = 0;
		}else{
				flag = 1;
				maxLen = len;
				start = s;
				s =i;
				end = i-1;
				e = 0;
				if(end - start>0)break;
			
		}
	}
	if(flag==0){
		start = s;
		end = n-1;
	}
	
	
	for(int i=start,j=0;j<=(end-start)/2;i++,j++){
		int temp = h[i];
	    h[i] = h[end-j];
	    h[end-j]=temp;
	}
	
	for(int i=1;i<n;i++){
		if(h[i]<h[i-1]){
			cout<<"-1 -1"<<endl;
			return;
		}
	}
	
	cout<<start+1<<" "<<end+1<<endl;
}

int main()
{
    //cout <<  << endl;
    int t;
    cin>>t;
    
    for(int i=0;i<t;i++){
    	int n;
    	cin>>n;
    	int height[n];
    	for(int j=0;j<n;j++){
    		cin>>height[j];
    	}
    	findSeq(height,n);
    }
    
    return 0;
    
}
