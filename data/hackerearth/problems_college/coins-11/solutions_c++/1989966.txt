#include <iostream>
#include <cmath>
using namespace std;
int prime(int n){
	int  limit = sqrt(n)+1;
	for (int i = 2;i<=limit; i++ ){
		if(n%i == 0) return 0;
	}
	return 1;
}
int main()
{
    int n,t;
    cin >>t;
    while(t){
    	cin >> n;
    	if (n == 1) cout << "1"<<endl;
    	else if(n == 3) cout <<"3"<<endl;
    	else if(prime(n)) cout<<"2"<<endl;
    	else cout<<"1"<<endl;
    	t--;
    }
    return 0;
}
