#include <iostream>


using namespace std;

int main()
{
    int array[10] = {6,  2 ,  5 , 5 ,  4 , 5 ,  6 , 3 , 7 , 6};
    
    string str ; 
    cin >> str ; 
    
    int length = str.length() ; 
    int sum = 0 ;
    for(int i = 0 ; i < length ; i++)
    {
    	sum += array[str[i]-'0'] ; 
    }
    
    cout << sum << endl ; 
    
    return 0;
}
