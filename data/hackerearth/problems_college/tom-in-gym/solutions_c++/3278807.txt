#include<iostream>
#include<cstdio>
#include<cstring>
 
 
#define max 1000
 
using namespace std;
 
void fact(int a[],int n){
	
	if(n==1)
		return ;
	int c=0;
	for(int i=max-1;i>=0;i--){
		a[i]=(a[i]*n)+c;
		c=a[i]/10;
		a[i]=a[i]%10;
	}
	
	fact(a,n-2);
}
 
 
int main()
{
	int t;
	cin>>t;
	for(int w=0;w<t;w++){
		
		int n;
		cin>>n;
		int a[max];
		memset(a,0,max*sizeof(int));
		
		a[max-1]=1;
		fact(a,(2*n-1));
		
		int c=0;
		for(int i=0;i<max;i++){
			if(a[i] && (!c))
				c=1;
			if(c)
				cout<<a[i];
		}
		
		cout<<"\n";
	}
	
	return 0;
}
