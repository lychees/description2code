#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int i,n,a[1000],sum=0;
		cin>>n;
		for(i=0;i<n;i++)
		{
			cin>>a[i];
			sum=sum+a[i];
		}
		if(sum%n==0)
		cout<<"YES"<<endl<<sum/n<<" "<<sum%n<<endl;
		else
			cout<<"NO"<<endl<<sum/n<<" "<<sum%n<<endl;
	}
	cout<<endl;
    //cout << "Hello World!" << endl;
    return 0;
}
