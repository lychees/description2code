#include<iostream>
#include<ctype.h>
#include<cstdio>
using namespace std;
int main()
{
    int test_cases;
    cin>>test_cases;
    getchar();
    while(test_cases>0)
    {
        char str[101];
        char alph[101];
        gets(str);
        
        int alph_index=-1;
        int star_index=-1;
        for(int k=0; str[k]!='\0'; ++k)
        {
        	if((isalpha(str[k])))
        	alph[++alph_index]=str[k];
        	
        	if((str[k]=='*'))
        	{
        		alph[++alph_index]=str[k];
        		star_index=alph_index;
        	}
        }

		int ans_index=(alph_index-star_index);
		
		cout<<alph[ans_index]<<endl;
        
        test_cases--;
    }
    return 0;
}
