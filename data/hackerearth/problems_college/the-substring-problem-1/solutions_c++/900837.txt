#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	char str[100][100];
    	int alpha[123][2];
    	for(int i=97;i<123;i++)
    	{
    		alpha[i][0]=alpha[i][1]=0;
    	}
    	int n;
    	cin>>n;
    	
    	for(int i=0;i<n;i++)
    	{
    		cin>>str[i];
    		for(int j=0;str[i][j]!='\0';j++)
    		{
    			//str[i][j]=getchar();
    			if(alpha[(int)str[i][j]][1]==0)
    			{
    			alpha[(int)str[i][j]][0]++;
    			alpha[(int)str[i][j]][1]=1;
    		//	cout<<(int)str[i][j]<<" ->"<<alpha[(int)str[i][j]][0]<<"\n";
    			}
    		}
    	for(int k=97;k<123;k++)
    	{
    		alpha[k][1]=0;
    	}	
    	}
    	int max=alpha[97][0];
    	int pos=97;
    	for(int k=97;k<123;k++)
    	{
    		if(alpha[k][0]>max)
    		{
    			max=alpha[k][0];
    			pos=k;
    		}
    	}
    	cout<<(char)pos<<"\n";
    }
    return 0;
    
}
