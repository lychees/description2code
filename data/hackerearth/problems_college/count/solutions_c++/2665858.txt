#include <iostream>
using namespace std;

int main()
{	int t;
	cin>>t;
	while(t--){
		int a,b,ne=0,no=0;
		cin>>a>>b;
		for(int i=1;i<=a;i++){
				if(i%2!=0){
					if(i<=b)
					no=no+i;
					else
					no=no+b;
					}
				else{
					if(i<=b)
					ne=ne+i;
					else
					ne=ne+b;
					}
		}
		cout<<ne<<" "<<no<<endl;
	}
    return 0;
}
