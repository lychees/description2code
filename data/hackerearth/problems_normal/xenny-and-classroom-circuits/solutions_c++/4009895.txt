#include<bits/stdc++.h>
#include<string>
#include<algorithm>
using namespace std;
long a[2][300];
int main()
{
	int t;
	cin>>t;
	long n,i;
	string str;
	while(t--){
		cin>>n;
		cin>>str;
		for(i=0;i<2;i++){
			a[i]['B']=0;
			a[i]['G']=0;
		}
		for(i=0;i<str.size();i++){
			a[i%2][str[i]]++;
		}
		cout<<min(a[0]['G'],a[1]['G'])<<endl;
	}
	return 0;
}
