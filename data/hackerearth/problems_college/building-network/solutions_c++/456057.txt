#include <iostream>
#include<cstdio>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    long long int arr[1010];
    scanf("%d",&t);
    while(t--)
    {
    	
 		int n;
 		long long int sum=0;
    	scanf("%d",&n);
    	for(int i=0;i<n;i++)
    		scanf("%lld",&arr[i]);
    	sort(arr,arr+n);
    	for(int i=1;i<n;i++)
    		sum+=arr[i];
    	sum+=(arr[0]*(n-1));
    	printf("%lld\n",sum);
    }
    return 0;
}
