import re
string = raw_input()
bSpace = False
modString = ''

for x in xrange(len(string)):
    char = string[x]
    if x == 0 and x != " ":
        modString += char.upper()
        continue
    if char == " ":
        bSpace = True
        modString += char
        continue
    if bSpace:
        modString += char.upper()
        bSpace = False
        continue
    modString += char

print(modString)
