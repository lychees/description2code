#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

#define EPS 0

double X[1001], Y[1001];

int main()
{
	int N; cin >> N;
	for(int i = 1; i <= N; i++) cin >> X[i] >> Y[i];
	
	int count = 0;
	
	for(int i = 1; i <= N; i++) {
		double max_slope = -2e9;
		for(int j = i + 1; j <= N; j++) {
			double curr_slope = (Y[j] - Y[i])/(X[j] - X[i]);
			if(curr_slope - max_slope >= EPS)  count++;
			max_slope = max(max_slope, curr_slope);
		}
	}
	
	cout << count << endl;
    return 0;
}
