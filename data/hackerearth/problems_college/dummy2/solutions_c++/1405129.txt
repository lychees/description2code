int size;
void rep(int l,char a[],char b[])
{ int f=0;
	for(int i=l-1;f==0&&i>=0;i--)
	{
		if((a[i]>48)&&(a[i]<=57))
		{
			int t=a[i]-48;
			for(int j=1;j<=t;j++)
			rep(i,a,b);
			f=1;
		}
		else
		{
		b[size]=a[i];
		size++;}
	}
}
