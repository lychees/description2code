#include<iostream>
using namespace std;

int main(){
	int t;
	cin >> t;
	while(t--){
		int n, m, d;
		cin >> n >> m >> d;
		int a_1[n][2];
		int a_2[m][2];
		
		for(int i = 0; i < n; i++)
			cin >> a_1[i][0] >> a_1[i][1];
		for(int i = 0; i < m; i++)
			cin >> a_2[i][0] >> a_2[i][1];
		
		int count = 0;
		for(int i = 0; i < m; i++){
			for(int j = 0; j < n; j++){
				if(abs(a_1[i][0] - a_2[j][0]) + abs(a_1[i][1] - a_2[j][1]) <= d){
					count++;
					break;
				}
			}
		}
		if(count > m/2)
			cout << "YES" << endl;
		else
			cout << "NO" << endl;
	}
}
