import math
import sys
from sys import stdin


def main():
    for i in xrange(int(stdin.readline())):
        [N, B] = [int(x) for x in stdin.readline().split()]
        if float(N % B) < (float(B))/2:
            print N - N % B
        else:
            print N + (B - N % B)


if __name__ == '__main__':
    main()
