#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int n, m, q,i,j,l1;
    scanf("%d%d%d",&n,&m,&q);
    l1 = q;
    int x[n],l;
    int a[n][m];
   	
   	for(i = 0; i < n; i++)
   		for(j = 0; j < m; j++)
   		scanf("%d",&a[i][j]);
   	
   	while(q--)
   	{
   		l=m-1;
   		for(j = 0; j < n; j++)
   		scanf("%d",&x[j]);
   		for(i = 0; i < n ; i++)
   		{
   			while(x[i]<a[i][l])
   			l--;
   		}
   		printf("%d\n",l+1);
   	}
    return 0;
}
