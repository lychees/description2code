#include <iostream>
using namespace std;

int dp[120]={0};
void func(int x,int c)
{
	if(x>113) return;
	dp[x+10]=c;
	dp[x+12]=c;
	func(x+10,c+1);
	func(x+12,c+1);
}

int main()
{
    func(0,1);
    int n,t;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	if(dp[n]) cout<<dp[n]<<endl;
    	else cout<<-1<<endl;
    }
    return 0;
}
