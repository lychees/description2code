t = int(raw_input())
for i in range(t):
    n, p1 = map(int, raw_input().split())
    p = []
    p.append(p1)
    for i in range(n):
        s = raw_input().split()
        if s[0] == 'P':
            p.append(int(s[1]))
        else:
            k = p[-1]
            p.pop()
            l = p[-1]
            p.append(k)
            p.append(l)
    print "Player", p[-1]
