#include<bits/stdc++.h>
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		string p,s;
		int i,j,l1,x=0,y=0,l2;
		cin>>p>>s;
		int f[26]={0},f1[26]={0};
		l1=p.length();
		l2=s.length();
		for(i=0;i<l1;i++)
		f1[p[i]-'a']++;
		for(i=0;s[i]&&i<l1;i++)
		{
			f[s[i]-'a']++;
		}
		while(i<l2)
		{
			for(j=0;j<26;j++)
			{
				if(f[j]!=f1[j])
				break;
			}
			if(j==26)
			break;
			f[s[x]-'a']--;
			
			f[s[i]-'a']++;
			x++;
			i++;
			
		}
		if(i==l2)
		{
			for(j=0;j<26;j++)
			{
				if(f[j]!=f1[j])
				break;
			}
			if(j==26)
			cout<<"YES\n";
			else
			cout<<"NO\n";
		}
		else cout<<"YES\n";
	}
	return 0;
}
