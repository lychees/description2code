#include <iostream>
using namespace std;
 
int minimum(int num1, int num2)
{
	return(num1<num2?num1:num2);
}
 
int main()
{
    int test;
    cin>>test;
    for(int j=0;j<test;j++)
    {
    	int house;
    	cin>>house;
    	int **arr=new int *[house];
    	for(int i=0;i<house;i++)
    		arr[i]=new int[3];
    	for(int i=0;i<house;i++)
    	{
    		cin>>arr[i][0]>>arr[i][1]>>arr[i][2];
    	}
    	int r=0;
    	int b=0;
    	int g=0;
    	int t=0;
    	int r1,b1;
    	for(int i=0;i<house;i++)
    	{
    		r1=r;
    		r=minimum(b,g)+arr[i][0];
    		b1=b;
    		b=minimum(r1,g)+arr[i][1];
    		g=minimum(r1,b1)+arr[i][2];
    		t=minimum(r,minimum(b,g));
    	}
    	cout<<t<<"\n";   	
    }
    return 0;
}
