#include <iostream>
using namespace std;

int main()
{
	int t,n,k,max,i;
	cin >> t;
	while(t--)
	{
		cin >> n >> k;
		int arr[n];
		int tm[n];
		for(i=0;i<n;i++)
			cin >> arr[i];
		for(i=0;i<n;i++)
			cin >> tm[i];
		max=0;
		for(i=0;i<n;i++)
		{
			int temp=tm[i]*(k/arr[i]);
			if(temp>max)
			max=temp;
		}
		cout << max << endl;
	}
    
    return 0;
}
