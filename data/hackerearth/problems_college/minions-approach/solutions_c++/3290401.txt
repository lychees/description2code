#include<bits/stdc++.h>
using namespace std;
int main()
{
	int isprime[100001]={0};
	isprime[0]=1;
	isprime[1]=1;
	//for(int i=0;i<100;i++)
	//cout<<isprime[i]<<"\n";
	for(int i=2;i<100001;i++)
	{
		if(isprime[i]==0)
		for(int j=2*i;j<100001;j+=i)
		{
			isprime[j]=1;
		}
	}
	//for(int i=0;i<100;i++)
	//cout<<isprime[i]<<"\n";
	int t,n,sum;
	cin>>t;
	while(t--)
	{
		sum=0;
		cin>>n;
		for(int i=1;i<n;i++)
		{
			if(__gcd(n,i)==1)
			sum++;
		}
		if(!isprime[sum])
		cout<<"Correct\n";
		else
		cout<<"Incorrect\n";
	}
}
