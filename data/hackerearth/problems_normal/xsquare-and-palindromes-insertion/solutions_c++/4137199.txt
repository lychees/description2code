#include <iostream>
#include <string>
#include <vector>
using namespace std;

int main()
{
    int p1,p2,t,odd,result;
    cin>>t;
    while(t--)
    {
    	string str;
    	cin>>str;
    	vector<int> hash(26,0);
    	odd=0;
    	for(p1=0;p1<str.size();p1++)
    		hash[str[p1]-'a']++;
    	for(p1=0;p1<26;p1++)
    		if(hash[p1]%2!=0)
    			odd++;
    	result=max(0,odd-1);
    	cout<<result<<endl;
    }
    return 0;
}
