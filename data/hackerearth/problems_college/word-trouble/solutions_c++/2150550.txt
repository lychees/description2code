#include<iostream>
#include<string.h>
#include<stdio.h>
using namespace std;

int main()
{
	char a[1000];
	char b[1000];
	char ch;
	int t;
	cin>>t;
	while(t--)
	{
		cin>>a>>b;
		//cout<<a<<" "<<b; 
		if(strcmp(a,b)==0)
			cout<<"Exact"<<endl;
		else
		{
			cout<<"Non Exact ";
			int m=strlen(a),n=strlen(b);
			int cnt=0;
			int i=0;
			for( i=0;i<m&&i<n;i++ )
			{
				cnt+=(a[i]!=b[i]);
			}
			if(i<m)
			{
				cnt+=(m-i);
			}
			else if(i<n)
			{
				cnt+=(n-i);
			}
			cout<<cnt<<endl;
		}
	}
	return 0;
}
