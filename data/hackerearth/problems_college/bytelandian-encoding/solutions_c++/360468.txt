#include <iostream>
#include <string.h>
using namespace std;

int main()
{
	int n;
	cin>>n;
	char str[50],k[50];
	while(n--)
	{
		int c = 0;
		cin>>str;
		int len = strlen(str); 
		while(len>0)
		{
			if (len % 2 != 0)
			{
				k[c++] = str[len/2];
				for (int j = len/2; j<len;j++)
				{
					str[j] = str[j+1];
				}
				len--;
			}
			else
			{
				int del;
				if(str[len/2] < str[(len/2) - 1])
				{
					del = len/2;
				}
				else
				{
					del = (len/2) - 1;
				}
				k[c++] = str[del];
				for (int j = del; j<len;j++)
				{
					str[j] = str[j+1];
				}
				len--;
			}
		}
		k[c++] = '\0';
		cout<<k<<endl;
		
	}
    return 0;
}
