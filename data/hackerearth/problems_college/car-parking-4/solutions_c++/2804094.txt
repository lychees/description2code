#include <iostream>
using namespace std;
string s;
int len;
bool isValid(int x) {
	if(s[x] != '-') return false;
	if(x+1 < len && (s[x+1] == 'B' || s[x+1] == 'S')) return false;
	if(x-1 >= 0 && s[x-1] == 'S') return false;
	if(x+2 < len && s[x+2] == 'B') return false;
	return true;
}
int main()
{
	int t;
	
	cin>>t;
	while(t--) {
		s.clear();
		cin>>s;
		len = s.length();
		int ans = 0;
		for(int i=0;i<len;i++) if(isValid(i)) ans++;
		cout<<ans<<endl;
	}
    return 0;
}
