def get_battle(x):
    i = 1
    n = 0
    while (True):
        if i + n >= x:
            return i
        n = i+n
        i = i+1


for i in range(int(raw_input())):
    D = int(raw_input())
    battle = get_battle(D)
    if battle % 3 == 0:
        print 'SL'
    elif battle % 3 == 1:
        print 'LB'
    else:
