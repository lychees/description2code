#include<bits/stdc++.h>
using namespace std;
void pre()
{
	ios_base::sync_with_stdio(0);
	cin.tie(0);
	cout.tie(0);
}
int check(int a[],int n)
{
	int c=0;
	for(int i=0;i<n;i++)
		if(a[i]>0)
			c++;
	return c;
}
int main()
{
 	pre();
 	int n;
 	cin>>n;
 	int a[n];
 	for(int i=0;i<n;i++)
 		cin>>a[i];
 	while(1)
 	{
 		int c=check(a,n);
 		if(c==0)
 			break;
 		else
 		{
 			cout<<c<<endl;
 		}
 		for(int i=0;i<n;i++)
 		{
 				if(a[i]==1) a[i]=0;
 				else if(a[i]>1)
 				{
 					a[i]=a[i]-2;
 				}
 		}
 	}
    return 0;
}
