#include <iostream>
#include <cstdio>
#include <vector>
#include <string>
#include <utility>
#include <algorithm>
#include <climits>
#include <list>
#include <stack>
#include <queue>
#include <set>
#include <cmath>
#include <iomanip>
#include <map>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(0);
	
	int t,i,j,N,C;
	cin>>t;

	while(t--) {
		cin>>N>>C;
		vector< vector<int> > mat(N,vector<int> (N));
		
		for(i=0;i<N;i++)
			for(j=0;j<N;j++)
				cin>>mat[i][j];
		
		
		for(i=1;i<N;i++) {
			mat[i][0]+=mat[i-1][0];
			mat[0][i]+=mat[0][i-1];
		}

		for(i=1;i<N;i++)
			for(j=1;j<N;j++)
				mat[i][j]+=max(mat[i-1][j],mat[i][j-1]);

		if(mat[N-1][N-1]>=C)
			cout<<(mat[N-1][N-1]-C)<<"\n";
		else
			cout<<"-1\n";
	}
	
	return 0;
}
