#include <bits/stdc++.h>
#define pb push_back
using namespace std;
int tc;
string s;
int main(){
    cin>>tc;
    while(tc--){
        cin>>s;
        vector < char >ans;
        ans.clear();
        for(int i=4;i<s.size()-4;i++){
            if(s[i]!='o' && s[i]!='a' && s[i]!='e' && s[i]!='i' && s[i]!='u')
                ans.pb(s[i]);
        }
        cout<<ans.size()+4<<"/"<<s.size()<<endl;
    }
    return 0;
}
