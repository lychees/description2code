#include<bits/stdc++.h>
#define mp make_pair
# define pb push_back
using namespace std;
typedef long long lo;
map<string,lo>m;
map<lo,string>m2;
int main()
{
        lo n,i,m1,maxi=0;
        string s;
        cin>>n;
        for(i=1;i<=n;i++)
        {
                cin>>s;
                cin>>m1;
                if(m.find(s)==m.end())
                m[s]=m1;
                else
                m[s]+=m1;
                if(m[s]>maxi)
                {
                        maxi=m[s];
                        m2[maxi]=s;
                }
        }
        cout<<m2[maxi];
}
