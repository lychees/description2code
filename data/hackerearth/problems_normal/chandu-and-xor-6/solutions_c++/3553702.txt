#include <iostream>
using namespace std;

int main()
{
	long long int n,m,maxx=0,num;
    cin >> n >> m;
    long long int row[n],col[m];
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            cin >> num;
            if (i==0){col[j]=num;}
            if (j==0){row[i] = num;}
            if (j)row[i] = row[i]+num;
            if (i)col[j] = col[j]+num;
        }
    }
    for (int i = 0; i < n; i++)
        for (int j = 0;j < m; j++) {
            maxx = max(maxx,row[i]^col[j]);
        }
    cout << maxx << endl;
    return 0;
}
