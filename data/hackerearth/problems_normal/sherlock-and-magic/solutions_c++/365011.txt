typedef unsigned long long int uint64;

struct crim{
	float x,y;
	int dir;
};
int dx[]={0,0,1,-1};
int dy[]={1,-1,0,0};
vector<crim>v,tmp,tmp1;
void go(){
	for(int i=0;i<tmp.size();i++){
		int flag=0;
		for(int j=0;j<tmp.size();j++){
			if(i==j)
			continue;
			if(tmp[i].x==tmp[j].x&&tmp[i].y==tmp[j].y){
				flag=1;
				break;
			}
		}
		if(flag==0)
		tmp1.pb(tmp[i]);
	}
	tmp.clear();
	tmp=tmp1;
	tmp1.clear();
}
int main(){
	int t,n,i;
	crim x;
	cin>>t;
	while(t--){
		cin>>n;
		string s;
		for(i=0;i<n;i++){
			cin>>x.x>>x.y>>s;
			if(s=="N")
			x.dir=0;
			if(s=="S")
			x.dir=1;
			if(s=="E")
			x.dir=2;
			if(s=="W")
			x.dir=3;
			v.pb(x);
		}
		for(int tim=1;tim<=10000;tim++){
			for(i=0;i<v.size();i++){
				x.x=v[i].x+dx[v[i].dir]/2.0;
				x.y=v[i].y+dy[v[i].dir]/2.0;
				x.dir=v[i].dir;
				tmp.pb(x);
			}
			go();
			v.clear();
			v=tmp;
			tmp.clear();
		}
		cout<<v.size()<<endl;
		v.clear();
	}
	return 0;}
