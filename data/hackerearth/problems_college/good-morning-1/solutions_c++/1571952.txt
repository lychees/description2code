#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,d;
    	cin>>n>>d;
    	for(int i=0;i<d;)
    	{
    		i=i+2;
    		n--;
    		if(n<0)
    		break;
    		if(i%6==0)
    		n=n+2;
    	}
    	if(n>=0)
    	cout<<"Yes"<<endl;
    	else
    	cout<<"No"<<endl;
    }
    
    return 0;
}
