#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int k;
    	cin>>k;
    	string str;
    	cin>>str;
    	long long int i,j,sum=0,count=0;
    	for(i=0;i<str.size();i++)
    	{
    		sum=0;
    		for(j=i;j<str.size();j++)
    		{
    			sum+=str[j]-96;
    			if(sum==k)
    			{
    				count++;
    				break;
    			}
    			if(sum>k)
    			{
    				break;
    			}
    		}
    	}
    	cout<<count<<endl;
    }
}
