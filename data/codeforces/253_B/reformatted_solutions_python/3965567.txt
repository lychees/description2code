FILE = "input"
try:
    inFile = open(FILE+".txt")
except BaseException:
    pass


def read():
    try:
        return inFile.readline().strip()
    except BaseException:
        return raw_input().strip()


n = int(read())
a = sorted(map(int, read().split()))
# print a
i = 0
j = 0

ans = 99999999999999999
while i < n:
    while j < n and a[j] <= 2*a[i]:
        j += 1
    ans = min(ans, n-(j-i))
    # print i,j
    i += 1
# print ans
out = open("output.txt", "w")
out.write("%i\n" % (ans))
