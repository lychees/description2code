#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int t,x,i,j;
    cin>>t;
    while(t--){
    	cin>>x;
    	for(i=x;i>=0;i--){
    		for(j=0;j<=x;j++){
    			if(x==pow(i,j)+pow(j,i)){
    				cout<<i<<" "<<j<<endl;
    				i=0;
    				break;
    			}
    		}
    	}
    }
    return 0;
}
