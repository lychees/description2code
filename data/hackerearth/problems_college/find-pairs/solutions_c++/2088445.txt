#include <iostream>
#include <unordered_map>

using namespace std;

int main()
{
    int tc; cin >> tc;
    while(tc--) {
    	int n; cin >> n;
    	int count = 0;
    	unordered_map<int, int> mp;
    	while(n--) {
    		count++;
    		int temp; cin >> temp;
    		mp[temp]++;
    		if(mp[temp] > 1) {
    			count+=mp[temp]-1;
    		}
    	}
    	
    	cout << count << endl;
    }
    return 0;
}
