#include <bits/stdc++.h>
using namespace std;
using namespace std;

int main()
{
    int n,t;
    cin>>t;
    while(t--)
    {
      cin>>n;
      int m=sqrt(n);
      //2cout<<m<<" ";
      int c=0;
      if(n%2!=0)
          c=0;
      else 
          for(int i=1;i<=m;i++)
            { 
            	if(i%2==0)
            	{
            	   	 if(n%i==0)
            	   	    c++; }         //cout<<c++<<" "<<i<<endl;
      	       if(n%i==0)
      	       {
				 if((n/i)%2==0&& (n/i)*(n/i)!=n)
      	             c++;}
           }
      cout<<c<<endl;
      c=0;
    }
    return 0;
}
