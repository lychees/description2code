#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long int a[1005],n,k,i,j,h,l,m,f,x;
    cin>>n>>k;
   	for(i=0;i<n;i++){
   		cin>>a[i];
   	}
   	sort(a,a+n);
   	for(i=0;i<n;i++)
   	{
   		for(j=i+1;j<n;j++)
   		{
   			x=k-a[i]-a[j];
   			l=0;
   			h=n-1;
   			f=0;
   			while(l<=h){
   				m=(l+h)/2;
   				if(a[m]==x){
   					if(m!=i && m!=j)
   						f=1;
   					else
   						h=m-1;
   						
   				}
   				else if(a[m]>x){
   					h=m-1;
   				}
   				else{
   					l=m+1;
   				}
   				if(f==1)
   				{
   					break;
   				}
   		}
   		if(f==1)
   		{
   			break;
   		}
   	}
   	if(f==1)
   	{
   		break;
   	}
   	}
   	if(f)
   	{
   		cout<<"YES"<<endl;
   	}
   	else
   	{
   		cout<<"NO"<<endl;
   	}
    return 0;
}
