#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
typedef pair<int, int> ii;
typedef vector<ii> vii;
typedef vector<int> vi;

vector<ll> days;
ll n, h;

bool Day(ll day) {
	
	ll curr = h;
	curr-= day * (day + 1) / 2;
	ll size = days.size();
	ll times = day / size;
	ll ph = times * days[size - 1];
	if(day % size != 0)
		ph += days[day % size - 1];
	return (ph >= curr) ? true : false;
}

ll binary_search(ll loday, ll hiday) {
	while(loday < hiday) {
		ll mid = (loday + hiday) / 2;
		ll to = 1;
		if (mid - 2 * n > 1) to = mid - n;
		for (ll i = mid; i >= to; i--) {
			if(Day(i) == true) {
				mid = i;
			}
		}
		bool res = Day(mid);
		if(res) {
			hiday = mid;
			int minIdx = mid;
		}
		else loday = mid + 1;
	}
	return hiday;
}
