#include <iostream>
using namespace std;
typedef long long int ll;
int main()
{
    ll max = 1000;
    ll mod = 1000000007;
    ll i, j;
    ll dp[max + 1][max + 1];
    for(i = 0; i <= max; i++) {
    	dp[i][0] = 1;
    	for(j = 1; j <= max; j++) {
    		dp[i][j] = 0;
    	}
    }
    for(i = 1; i <= max; i++) {
    	for(j = 1; j <= i; j++) {
    		dp[i][j] = (dp[i - 1][j - 1] + dp[i - 1][j])%mod;
    	}
    }
    for(i = 1; i <= max; i++) {
    	for(j = 1; j <= i; j++) {
    		dp[i][j] = (dp[i][j - 1] + dp[i][j])%mod;
    	}
    }
    int t;
    cin >> t;
    while(t--) {
    	ll n, p;
    	cin >> n >> p;
    	cout << dp[n][p] << endl;
    }
    return 0;
}
