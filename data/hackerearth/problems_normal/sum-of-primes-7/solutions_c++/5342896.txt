#include<bits/stdc++.h>
using namespace std;
bool prm[1000*1000+9];
const int n=1000*1000;
#define ll long long
int ans[n+9];
ll pref[n+9];
void sieve(){
    for(int i=0;i<=n;++i)prm[i]=true;
    memset(ans,0,sizeof ans);
    prm[1]=false;
    for(int i=1;i*i<=n;++i){
        if(prm[i]){
            //ans[i]++;
            for(int j=2*i;j<=n;j+=i){
                if(prm[j]){
                    prm[j]=false;
                   // ans[i]++;
                }
            }
        }
    }
    return;
}
int main(){
    ios::sync_with_stdio(false);cin.tie(0);
    sieve();
    for(int i=1;i<=n;++i){
        if(prm[i])pref[i]+=(ll)i;
        pref[i]+=pref[i-1];
    }
    int tc;cin>>tc;
    while(tc--){
        int l,r;cin>>l>>r;
        cout<<pref[r]-pref[l-1]<<"\n";
    }
    return 0;
}
