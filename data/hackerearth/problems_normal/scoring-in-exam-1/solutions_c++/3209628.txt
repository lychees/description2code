#include <iostream>
#include<vector>
#include<algorithm>

using namespace std;

int main()
{
    int num,query,i;
    long score,t;
    long sum;
    vector<long> vscore;
    vector<long> vtime;
    cin >> num>>query;
    i=num;
    while(i)
    {
    	cin>>t;
    	vtime.push_back(t);
    	--i;
    }
    i=num;
    while(i)
    {
    	cin>>score;
    	vscore.push_back(score);
    	--i;
    }
    std::sort(vtime.begin(),vtime.end(),std::greater<int>());
    while(query)
    {
    	cin >> i;
    	sum=0;
    	int j=i;
    	while(j)
    	{
    		
    		sum= sum + vtime[i-1];
    		--i;
    		--j;
    	}

    	cout <<sum<<endl;
    	--query;
    }
    
return 0  ;  
}
