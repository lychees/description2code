# https://www.hackerearth.com/gsf-hacks/algorithm/the-strongest-string-1-1/

n = int(raw_input())
s = raw_input()

result = ''
start = 0
for i in xrange(ord('z'), ord('a') - 1, -1):
    ch = chr(i)
    for j in xrange(start, len(s)):
        if s[j] == ch:
            result += ch
            start = j + 1
            break
print result
