#include <cstdio>
#define ull unsigned long long int

int main()	{
	ull t;
	ull b,w,x,y,z,sum;
	
	scanf("%llu",&t);
	while(t--)	{
		scanf("%llu%llu",&b,&w);
		scanf("%llu%llu%llu",&x,&y,&z);
		if((b*x)>(b*y + b*z))
			sum = b*y + b*z;
		else
			sum = b*x;
		
		if((w*y)>(w*x + w*z))
			sum += w*x + w*z;
		else
			sum += w*y;
		printf("%llu\n",sum);
	}
	return 0;
}
