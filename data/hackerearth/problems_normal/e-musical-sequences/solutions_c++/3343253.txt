#include<bits/stdc++.h>
#define ll long long int
#define MAX 100004
using namespace std;
// const ll MOD=1000000007;
ll arr[MAX];
int main()
{
    ll t,r,d,n,MOD,z,sign1;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld%lld%lld",&n,&MOD,&z);
        for(ll i=0;i<n;i++)
        {
            scanf("%lld",&arr[i]);
        }
        sign1=1;arr[n]=0;
        for(int i=n-1;i>=0;i--)
        {
            arr[n]=(arr[n]%MOD+sign1*arr[i]%MOD)%MOD;
            sign1=(-1)*sign1;
        }
        d=z/(n+1);
        r=z%(n+1);
        //cout<<r<<" "<<d<<endl;
        //ll value;
        if(n%2)
        {
            cout<<(arr[r]%MOD+MOD)%MOD<<endl;
        }
        else
        {
            if(d%2==0)
            {
                cout<<(arr[r]%MOD+MOD)%MOD<<endl;
            }
            else if(z==n)
            {
                cout<<(arr[r]%MOD+MOD)%MOD<<endl;
            }
            else
            {
                ll value=(ll)(-1)*arr[r];
                cout<<(value%MOD+MOD)%MOD<<endl;
            }
        }
    }
    return 0;
}
