#define For(i,j,n) for(i=j;i<n;i++)
using namespace std;
int visited2[10001]={0};
//vector<int> graph[10001];
void dfsr(int i,vector<int> graph[])     //recursive approach
{
    visited2[i]=1;
    //printf("i %d ",i);
    for(int j=0;j<graph[i].size();j++)
    {
        if(!visited2[graph[i][j]])
            dfsr(graph[i][j],graph);
    }
}
int main()
{
     int i,j,k,no;
     scanf("%d",&no);
     while(no--)
     {
         int n,f,s,a,b,count1=0,c;
         vector<int> graph[10001];
         scanf("%d %d %d",&n,&f,&s);
         //graph[10001].clear();
         For(i,0,f)
         {
             scanf("%d %d",&a,&b);
             graph[a].push_back(b);
         }
         For(i,1,n+1)
         visited2[i]=0;
         For(i,0,s)
         {
             scanf("%d",&c);
             dfsr(c,graph);
         }
         For(i,1,n+1)
         {
             if(visited2[i]==1)
             count1++;
         }
            printf("%d\n",count1);
     }
  return 0;
}
