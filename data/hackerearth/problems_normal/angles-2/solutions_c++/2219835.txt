#include <iostream>
#include <queue>
using namespace std;

int mod(int a) {
	return (a>=0?a:-a);
}

int main()
{
    int n, k;
    cin>>n>>k;
    int knows[n], need;
    bool degrees[360] = {false};
    queue<int> q;
    for(int i=0; i<n; i++) {
    	cin>>knows[i];
    	q.push(knows[i]);
    	degrees[knows[i]] = true;
    }
    while(!q.empty()) {
    	int e = q.front();
    	q.pop();
    	for(int i=0; i<n; i++) {
    		int temp = (e+knows[i])%360;
    		if(!degrees[temp]) {
    			q.push(temp);
    			degrees[temp] = true;
    		}
    		temp = mod(e-knows[i])%360;
    		if(!degrees[temp]) {
    			q.push(temp);
    			degrees[temp] = true;
    		}
    	}
    }
    for(int i=0; i<k; i++) {
    	cin>>need;
    	cout<<(degrees[need]?"YES":"NO")<<endl;
    }
    return 0;
}
