#include<bits/stdc++.h>

using namespace std;
int digitsum(int x)
{
    int m = 0;
    while(x)
    {
        m += (x%10)*(x%10);
        x/=10;
    }
    return m;
}
int main()
{
    int i,y,t,p,x,a[1006],b[1006];

    cin>>t;

    while(t--)
    {
        cin>>x>>y;
        for(i=0;i<650;i++)
            a[i]=-1;
        //a[x]=0;
        for(i=1;i<=y;i++)
        {
            x = digitsum(x);
            if(a[x]!=-1)
            break;
            else
            a[x]=i,b[i]=x;
            //cout<<i<<' '<<x<<'\n';

        }

        if(i<y)
        {
            p = i-a[x];
           // cout<<i<<' '<<x<<' '<<a[x]<<' '<<p<<'\n';
            y -= a[x];
            //cout<<y<<' ';
            y = y%p;
            cout<<b[a[x]+y]<<'\n';
        }else
        cout<<x<<'\n';

    }


    return 0;
}
