#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin >> t;
    long long int temp;
    long int n;
    int mini;
    while(t--)
    {
    	cin >> n;
    	if(n == 1)
    	{
    		cin >> temp;
    		mini = 1;
    	}
    	else
    	{
    		cin >> temp;
    		mini = __builtin_popcount(temp);
    		n--;
    		while(n--)
    		{
    			cin >> temp;
    			mini = min(mini,__builtin_popcount(temp));
    		}
    	}
    	cout << mini << endl;
    }
    return 0;
}
