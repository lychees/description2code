     int t,i,j;

     while (t--)
     {
     	int n,j,l;
     	char d;
     	cin>>n>>j>>l>>d;
     	int time_lift =0,time_stairs;
     	time_stairs = 10*j;
     	if (d == 'D' && j<=l)
     	{
     		time_lift = l*5 + 20;
     	}
     	else if (d == 'D')
     	{
     		time_lift = l*5 + n*10 + 40;
     		if (j==n)
     		time_lift -= 10;
     	}
     	else if (d == 'U')
     	{
     		time_lift = n*5 + (n-l)*5 + 30;
     		if (j==n)
     		time_lift -= 10;
     	}
     	
     	
     	//cout<<"\t"<<time_lift<<" "<<time_stairs<<endl;
     	if (time_stairs < time_lift)
     	{
     		cout<<"Stairs "<<time_stairs<<endl;
     	}
     	else
     	cout<<"Lift "<<time_lift<<endl;
     }
return 0;
}
