#include <bits/stdc++.h>
using namespace std;

#define ll long long

ll MOD=1000000007;
ll fact[100005];
void precompute()
{
	ll i;
	fact[0]=fact[1]=1;
	for(i=2;i<100005;i++)
	 fact[i]=(fact[i-1]*i)%MOD;
}


ll modpow(ll a,ll b)
{
	ll res;
	res=1;
	while(b>0)
	{
		if(b%2)
		 {
		 	res*=a;
		 	res%=MOD;
		 }
		a=(a*a)%MOD;
		b>>=1;
	}

	return res%MOD;
}


int main()
{
	ll t,ans,n,k;
	precompute();
    cin>>t;
    while(t--)
    {
    	ans=1;
    	cin>>n>>k;
    	ans=((((fact[k]*modpow(fact[n],MOD-2))%MOD * modpow(fact[k-n],MOD-2))%MOD)*fact[n])%MOD;
    	cout<<ans<<endl;

    }


    return 0;
}
