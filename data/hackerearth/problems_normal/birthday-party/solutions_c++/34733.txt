bool func(int set[], int n, int sum)
{
    bool subset[sum+1][n+1];
    for (int i = 0; i <= n; i++)
      subset[0][i] = true;
 for (int i = 1; i <= sum; i++)
      subset[i][0] = false;
     for (int i = 1; i <= sum; i++)
     {
       for (int j = 1; j <= n; j++)
       {
         subset[i][j] = subset[i][j-1];
         if (i >= set[j-1])
           subset[i][j] = subset[i][j] || subset[i - set[j-1]][j-1];
       }
     }

     return subset[sum][n];
}
int ar[100005];
int main()
{
   int t,i,j,n,cnt,p,k;

   cin>>t;
   while(t--)
   {
       cin>>p>>n;
       int a[n];
       rep(i,n)
       cin>>a[i];
       memset(ar,0,sizeof(ar));
       rep(i,p)
       {
           cin>>k;
           ar[k]++;
       }
       cnt=0;
       rep(i,100005)
       {
           if(ar[i]>0 && func(a,n,ar[i])==true)
           cnt++;
       }
       cout<<cnt<<endl;
   }
   return 0;
}
