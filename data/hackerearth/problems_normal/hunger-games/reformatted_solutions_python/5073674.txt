n = int(raw_input())
animals = sorted([int(x) for x in raw_input().split()])


table = []

for i in range(0, n):
    if i % 2 == 0:
        table.insert(0, animals[i])
    else:
        table.append(animals[i])

max_danger = 0

for i in range(1, n):
    danger = abs(table[i] - table[i-1])
    if danger > max_danger:
        max_danger = danger

danger = abs(table[0] - table[n-1])

if danger > max_danger:
    max_danger = danger

print max_danger
