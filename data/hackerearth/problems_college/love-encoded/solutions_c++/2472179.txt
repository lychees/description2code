#include<malloc.h>
#include<iostream>
#include<string.h>
using namespace std;
int main()
{int t,*k,d;
 char a[10][50],c;
 cin>>t;
 k=(int*)malloc(t*sizeof(int));
 for(int i=0;i<t;i++)
  {cin>>*(k+i);
   cin>>a[i];
  }
 for(int i=0;i<t;i++)
  { d=*(k+i);
    for(int j=0;a[i][j]!='\0';j++)
    { int temp,t1;
       t1=c=a[i][j];
      temp=t1-d;
       if(temp<65)
       { t1=t1-65;
         temp=d-t1;
         c=90-temp+1;
         a[i][j]=c;
       }
       else
       {
          c=c-d;
          a[i][j]=c;
       }
      a[i][j]=c;
    }
   cout<<a[i]<<endl;
  }

 return 0;
}
