typedef pair<int, int> pii;
typedef long long LL;
typedef vector<int> vi;
typedef vector<pii> vpii;

#define pb push_back

#define X first
#define Y second
#define MX 500005
int n, m;
int a[3 * MX];
int px, py;
int pos[MX];
int M;

void add(int pos, int val) {
	for (; pos <= M; pos += pos & -pos) a[pos] += val;
}

int calc(int pos) {
	int res = 0;
	for (; pos; pos -= pos & -pos) res += a[pos];
	return res;
}

int main() {
	scanf("%d%d", &n, &m);
	int i, x;
	M = n + m + m + 1;
	for (i = 0; i < n; i++) {
		scanf("%d", &x);
		pos[x] = i + 1 + m;
		add(pos[x], 1);
	}
	px = m, py = m + n + 1;
	LL res = 0;
	char str[5];
	while (m--) {
		scanf("%d%s", &x, str);
		int id = pos[x];
		int t = calc(id - 1);
		res += min(t, n - 1 - t);
		add(id, -1);
		if (str[0] == 'l') {
			add(px, 1);
			pos[x] = px--;
		} else {
			add(py, 1);
			pos[x] = py++;
		}
	}
	cout << res << endl;
	return 0;
}
