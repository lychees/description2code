#include<bits/stdc++.h>
#define si(x) scanf("%d",&x)
#define sii(x,y) scanf("%d%d",&x,&y)
#define sll(x,y) scanf("%ld%ld",&x,&y)
#define sl(x) scanf("%ld",&x)
using namespace std;
const int MOD=1000000007;
int main(){
	int t;
	si(t);
	while(t--){
		int n;
		si(n);
		vector<char> v;
		while(n--){
			int temp;
			cin>>temp;
			temp=~temp;
			bitset<8> str(temp);
			int res=str.to_ulong();
			v.push_back((char)res);
		}
		for(auto x:v){
			cout<<x;
		}
		cout<<endl;
	}
}
