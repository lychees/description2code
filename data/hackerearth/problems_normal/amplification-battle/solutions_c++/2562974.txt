#include <bits/stdc++.h>
using namespace std;

int main()
{
	long long int t,c,i,j,k,a,b,mx,mn;
    cin>>t;
    while(t--)
    {
    	
    cin>>a>>b>>c;
    mx=max(a,b);
    mn=min(a,b);
    int count=0;
    do
    {
    	if(count%2==0)
    	c=ceil(c/(double)mx);
    	else
    	c=ceil(c/(double)mn);
    
    	count++;
    	//	cout<<c<<' '<<count<<endl;
    }while(c>1);
    if(count%2)
    cout<<"Adam"<<endl;
    else
    cout<<"Bob"<<endl;
    	
    	
    }
    return 0;
}
