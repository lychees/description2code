if __name__=='__main__':
    T = int(raw_input())
    for i in range(0,T):
        B,W=map(int,raw_input().split())
        rb,rw,con=map(int,raw_input().split())
        cost = 0
        if (rb+con ) <rw:
            cost += W*(rb+con)
            cost += B*rb
        elif (rw+con )<rb:
            cost += (W*rw)
            cost += (B*(rw+con))
        else:
            cost += B*rb
            cost += W*rw
        print cost
