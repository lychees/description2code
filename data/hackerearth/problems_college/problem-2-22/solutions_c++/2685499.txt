int check(int x1 , int y1 , int x2 , int y2 , int r)
{
	LL x = x1 - x2;
	LL y = y1 - y2;
	LL d = x*x + y*y;
	LL e = 1LL * r * r;
	if(d <= e) return 1;
	else return 0;
}

int main()
{
	int t;
	cin(t);
	while(t--)
	{
		int r,xc,yc;
		cin(r);
		cin(xc);
		cin(yc);
		int yes = 1;
		for(int i = 0; i < 4; i++)
		{
			int x,y;
			cin(x);
			cin(y);
			yes = yes&check(x,y,xc,yc,r);
		}
		if(yes) cout << "Yes" << endl;
		else cout << "No" << endl;
	}
	return 0;
}
