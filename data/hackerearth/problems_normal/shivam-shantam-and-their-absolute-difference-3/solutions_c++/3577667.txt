#include<bits/stdc++.h>
#define ll long long
#define nax 1000005
using namespace std;
#define mod 1000000007

ll dpA[nax][2],dpB[nax][2];
ll A[205],B[205];

int main()
{
for(int i=0;i<nax;i++)
{

    dpA[i][0]=0;
    dpB[i][0]=0;
}
dpA[0][0]=dpB[0][0]=1;
dpA[0][1]=dpB[0][1]=1;
    ll n,sumA=0,sumB=0,q;
    cin>>n;
    for(int i=1;i<=n;i++)
    {


        cin>>A[i];

    }
    for(int i=1;i<=n;i++)
    {


        cin>>B[i];

    }
    cin>>q;

    for(ll i=1;i<=n;i++)
    {

        for(ll j=0;j<=nax;j++)
        {
            if(j>=A[i])
                dpA[j][(i%2)]=(dpA[j][(i-1)%2]+dpA[j-A[i]][(i-1)%2])%mod;
            else
                dpA[j][(i%2)]=dpA[j][(i-1)%2];


        }

    }

      for(ll i=1;i<=n;i++)
    {

        for(ll j=0;j<=nax;j++)
        {
            if(j>=B[i])
                dpB[j][(i%2)]=(dpB[j-B[i]][(i-1)%2]+dpB[j][(i-1)%2])%mod;
                else
                    dpB[j][(i%2)]=dpB[j][(i-1)%2];
               // cout<<dpB[j][i%2]<<" ";
        }
        //cout<<"\n";
    }
    ll ans=0;

    for(ll i=1;i<nax;i++)
    {
        dpB[i][(n%2)]=(dpB[i][(n%2)]+dpB[i-1][(n%2)])%mod;


    }


    ll x;
    for(ll i=0;i<=nax;i++)
    {
        if(i-q-1>=0)
        ans=(ans + (dpA[i][n%2]*(dpB[i+q][n%2]-dpB[i-q-1][n%2])%mod)+mod)%mod;
        else
            ans=(ans+ dpA[i][n%2]*dpB[i+q][n%2])%mod;
            



    }
    cout<<ans;





}
