#include <iostream>
using namespace std;
int s[21][3];
int min(int a,int b)
{
	return a<b ? a : b;
}
int minm(int a,int b,int c)
{
	return min(min(a,b),c);
}
int color(int c,int n)
{
	if(n==0)
	 return 0;
	if(n==1)
	{
		 if(c==0)
		   return min(s[n-1][1],s[n-1][2]);
		   if(c==1)
		   return min(s[n-1][0],s[n-1][2]);
		   if(c==2)
		   return min(s[n-1][1],s[n-1][0]);
	}
	else
	{
		if(c==0)
		  return min(s[n-1][1]+color(1,n-1),s[n-1][2]+color(2,n-1));
		if(c==1)
		  return min(s[n-1][0]+color(0,n-1),s[n-1][2]+color(2,n-1));
		if(c==2)
		  return min(s[n-1][1]+color(1,n-1),s[n-1][0]+color(0,n-1));  
	}
}
int main()
{
    int t,n,i;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(i=0;i<n;i++)
    	 cin>>s[i][0]>>s[i][1]>>s[i][2];
    	 
    	cout<<minm(s[n-1][0]+color(0,n-1),s[n-1][1]+color(1,n-1),s[n-1][2]+color(2,n-1))<<endl; 
    }
    return 0;
}
