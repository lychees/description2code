//GCD(a,b)
inline ll gcd(ll a,ll b)
{
    ll t;
    if(a<b)
    {
        a^=b;
        b^=a;
        a^=b;
    }
    while(b!=0)
    {
        t=a;
        a=b;
        b=t%b;
    }
    return a;
}

{
