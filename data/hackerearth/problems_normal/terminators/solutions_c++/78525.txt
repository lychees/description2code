#include<iostream>
using namespace std;

int main() 
{
	int q;
	cin>>q;
	while(q--)
	{
		int n;
		cin>>n;
		int * a,temp;
		a = new int[n];
		for (int i = 0; i < n; i++) 
			{
				cin>>temp;
				//cout<<"temp "<<(temp-1==i)<<endl;
				cin>>a[temp-1];
			}
		int maxLength = 1, bestEnd = 0;
		int * DP, *prev;
		DP = new int[n];
		prev = new int[n];
		DP[0] = 1;
		prev[0] = -1;

		for (int i = 1; i < n; i++)
		{
		   DP[i] = 1;
		   prev[i] = -1;

		   for (int j = i - 1; j >= 0; j--)
		      if (DP[j] + 1 > DP[i] && a[j] < a[i])
		      {
		         DP[i] = DP[j] + 1;
		         prev[i] = j;
		      }

		   if (DP[i] > maxLength)
		   {
		      bestEnd = i;
		      maxLength = DP[i];
		   }
		}
		cout<<maxLength<<endl;
	}
}
