#include <iostream>
using namespace std;

int main()
{
    int T;
    int index,N;
    cin>>T;
    while(T--)
    {
    	cin>>index>>N;
    	if(index==1)
    	{
    		if(N%2==0)
    			cout<<1<<endl;
    		else
    			cout<<0<<endl;
    	}
    	else if(index==0)
    	{
    		if(N%2==0)
    			cout<<0<<endl;
    		else
    			cout<<1<<endl;
    	}
    	else
    	{
    		if(N==0)
    			cout<<2<<endl;
    		else if(N%2==0)
    			cout<<0<<endl;
    		else
    			cout<<1<<endl;
    	}
    	
    }
    return 0;
}
