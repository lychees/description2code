#include <iostream>
using namespace std;

int main()
{
    int t, n, k, c = 0, tmp;
    cin >> t;
    for(int i=0; i<t; i++) {
    	c = 0;
    	cin >> n >> k;
    	for(int j=0; j<n; j++) {
    		cin >> tmp;
    		if(tmp <= 0) c++;
    	}
    	if(c >= k) cout << "NO" << endl;
    	else cout << "YES" << endl;
    }
    return 0;
}
