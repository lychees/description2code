#include <iostream>
#include <locale>

using namespace std;

int main(){
	locale loc;
	int h, index;
	string alphabet = "abcdefghijklmnopqrstuvwxyz";
	char letter;

	while(cin >> letter >> h) {
		if((h >= 2) && (h <= 26)) {
			for (int i = 0; i < 26; i++) {
				if(letter == alphabet[i]) {
					if(h <= 26 - i) {
						for (int j = 1; j <= h; j++){
							char out = alphabet[i + j - 1];
							for (int k = 0; k < j; k++){
								cout << out << " ";
							}
							cout << endl;
						}
					} else {
						cout << "INVALID" << endl;
					}
				} else if(tolower(letter, loc) == alphabet[i]) {
					if(h <= 26 - i) {
						for (int j = 1; j <= h; j++){
							char out = toupper(alphabet[i + j - 1], loc);
							for (int k = 0; k < j; k++){
								cout << out << " ";
							}
							cout << endl;
						}
					} else {
						cout << "INVALID" << endl;
					}
				}
			}
			cout << endl;
		} else {
			cout << "INVALID" << endl << endl;
		}
	}
}
