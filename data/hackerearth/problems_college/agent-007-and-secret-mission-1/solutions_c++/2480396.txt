#include <iostream>
#include <cstring>
#define ULLI unsigned long long int
using namespace std;

int main()
{
    int i, ones, T;
    ULLI N;
    cin>>T;
    while(T--)
    {
    	string s;
    	cin>>s;
    	cin>>N;
    	ones = 0;
    	for(i=0; s[i]; i++)
    	{
    		if(s[i]=='1')ones++;
    	}
    	if(N%4==0 || (N+1)%4==0)ones=ones;
    	else ones=s.size()-ones;
    	cout<<ones<<endl;
    }
    return 0;
}
