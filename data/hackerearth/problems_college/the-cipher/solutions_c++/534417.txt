#include <iostream>
#include <cstdio>
#include<math.h>

using namespace std;
 

int main()
{
        int test_cases;
        scanf("%d",&test_cases);
        while(test_cases > 0)
        {     char ch;
                int N,arr[10000],i=0,j,num=0;
                int bin[8],n;
                scanf("%d",&N);
                while(i<N)
                {
                        scanf("%d",&arr[i]);
                        i = i+1;
                }
                for(j=0;j<N;j++){
                n=arr[j];
                num=0;
	            for(i=7;i>=0;i--){
	          	bin[i]=!(n%2);
	             n=n/2;
	              }
	             
	            for(i=0;i<8;i++){
	              if(bin[i])
	                 num+=pow(2,7-i);
	            }
	                
	           
	                ch=num%128;   
	                cout<<ch; 
               }
               cout<<"\n";
               
                
                test_cases--;
        }
        return 0;
}
