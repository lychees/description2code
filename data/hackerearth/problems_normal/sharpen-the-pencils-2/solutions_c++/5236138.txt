#include <iostream>
using namespace std;

int main()
{
    int t,n; cin>>t; while(t--)
    {
    	cin>>n;
    	int a[n+1],madhav[n+1],riya[n+1],m=0,r=0;
    	for(int i=0;i<n;i++) cin>>a[i];
    	
    	madhav[0]=0; riya[n-1]=0;
    	for(int i=1;i<n;i++)
    	{
    		madhav[i]=madhav[i-1]+a[i-1];
    	}
    	for(int i=n-2;i>=0;i--)
    	{
    		riya[i]=riya[i+1]+(a[i+1]<<1);
    	}
    	for(int i=0;i<n;i++)
    	{
    		if(madhav[i]<=riya[i]) m++;
    		else break;
    	}
    	cout<<m<<" "<<n-m<<endl;
    }
    return 0;
}
