ll mod=1e9+7;
vector<bool> prime(upperlimit, true);
void sieve()
{
	int bound = (int) floor(sqrt(upperlimit));
	prime[0]=false;
	prime[1]=false;
	for(int i = 4; i < upperlimit; i += 2)
        prime[i] = false;
    for (int i = 3; i  <= bound; i += 2)
    {
        if (prime[i]) 
        {
            for (int j = i*i; j < upperlimit; j += i * 2)
                prime[j] = false;
        }
    }
}

	int t;s(t);
	sieve();
