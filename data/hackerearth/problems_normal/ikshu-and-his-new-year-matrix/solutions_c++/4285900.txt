#include <bits/stdc++.h>
#define ll long long
using namespace std;

ll int a[1010][1010],dp[100010],p[100010],ans[1010][1010];

ll int sieve()
{
    ll int i,j;

    for(i=4;i<=100000;i+=2)
    {
        p[i]=1;
    }

    for(i=3;i*i<=100000;i+=2)
    {
        if(p[i]==0)
        {
            for(j=i+i;j<=100000;j+=i)
            {
                p[j]=1;
            }
        }
    }
    p[1]=p[100001]=p[100002]=p[100003]=p[100004]=p[100005]=1;
    dp[100001]=dp[100002]=dp[100003]=dp[100004]=dp[100005]=1e17;
}

int main()
{
    ll int n,m,i,j,k,maxx,x,y;

    //freopen("input.txt","r",stdin);

    sieve();
    cin>>n>>k;

    for(i=1;i<=n;i++)
        for(j=1;j<=n;j++)
            cin>>a[i][j],ans[i][j]=-1;


    for(i=100000;i>=1;i--)
    {
        dp[i]=1e17;
        if(p[i]==0)
            dp[i]=0;
        dp[i]=min(dp[i],dp[i+k]+1);
    }

    //cout<<dp[9]<<endl;

    for(i=2;i<n;i++)
        for(j=2;j<n;j++)
    {
        ans[i][j]=dp[a[i-1][j-1]]+dp[a[i+1][j+1]]+dp[a[i-1][j+1]]+dp[a[i+1][j-1]]+dp[a[i][j]];

        //cout<<ans[i][j];
    }

    maxx=1e9;

    for(i=2;i<n;i++)
        for(j=2;j<n;j++)
    {
        if(maxx>ans[i][j])
        {
            maxx=ans[i][j];
            x=i;
            y=j;
        }
    }

    if(maxx!=1e9)
    {
        cout<<"yes"<<endl;
        cout<<maxx<<endl;
        cout<<x<<" "<<y<<endl;
        //cout<<a[x][y]<<endl;
        //cout<<a[x-1][y-1]<<" "<<dp[a[x+1][y+1]]<<" "<<dp[a[x-1][y+1]]<<" "<<dp[a[x+1][y-1]]<<" "<<dp[a[x][y]]<<endl;
    }
    else
    {
        cout<<"no"<<endl;
    }
}
