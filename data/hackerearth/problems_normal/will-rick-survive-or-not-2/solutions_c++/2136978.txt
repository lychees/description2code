void countsort(int *arr,int n)
{
    int i;

    int count[100001]={0};

    for(i=0;i<n;i++)
        ++count[arr[i]];


    for(i=1;i<100001;i++)
    {
        count[i]+=count[i-1];
    }

    int arrsor[100000];
    for(i=0;i<n;i++)
    {
        arrsor[count[arr[i]]-1]=arr[i];
        --count[arr[i]];
    }

    for(i=0;i<n;i++)
    {
        arr[i]=arrsor[i];
        //cout<<" "<<arrsor[i];//=arr[i];
        //--count[arr[i]];
    }
}
