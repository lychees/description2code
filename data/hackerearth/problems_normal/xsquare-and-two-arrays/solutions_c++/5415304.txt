#include <iostream>
#include <vector>

using namespace std;

typedef unsigned long long ull;

int main()
{

    vector<ull> dp1,dp2;
    ull n,q;
    cin>>n>>q;

    vector<ull> v1;
    vector<ull> v2;

    v1.push_back(0);
    v2.push_back(0);
    dp1.push_back(0);
    dp2.push_back(0);
    dp1.push_back(0);
    dp2.push_back(0);

    ull one,two;
    cin>>one>>two;
    v1.push_back(one);
    v1.push_back(two);
    dp1.push_back(one);
    dp1.push_back(two);

    for(ull i = 4 ; i <= n+1 ; i++)
    {
        ull x;
        cin>>x;
        v1.push_back(x);
        dp1.push_back(x+dp1[i-2]);
        //cout<<dp1[i]<<' ';
    }

    //cout<<'\n';

    cin>>one>>two;
    v2.push_back(one);
    v2.push_back(two);
    dp2.push_back(one);
    dp2.push_back(two);


    for(ull i = 4 ; i <= n+1 ; i++)
    {
        ull x;
        cin>>x;
        v2.push_back(x);
        dp2.push_back(x+dp2[i-2]);
        //cout<<dp2[i]<<' ';
    }

    /*cout<<'\n';

    for (ull i : dp1)
        cout<<i<<' ';
    cout<<'\n';

    for (ull i : dp2)
        cout<<i<<' ';
    cout<<'\n';*/

    ull x,l,r;

    ull sum = 0;

    while(q--)
    {
        cin>>x>>l>>r;

        l++;
        r++;

        if (x == 1)
        {
            sum -= dp1[l-2];
            sum -= dp2[l-1];

            if (r%2 == l%2)
            {
                sum += dp1[r];
                sum += dp2[r-1];
            }
            else
            {
                sum += dp1[r-1];
                sum += dp2[r];
            }
        }

        else
        {
            sum -= dp2[l-2];
            sum -= dp1[l-1];

            if (r%2 == l%2)
            {
                sum += dp2[r];
                sum += dp1[r-1];
            }
            else
            {
                sum += dp2[r-1];
                sum += dp1[r];
            }
        }

        cout<<sum<<'\n';
        sum = 0;
    }


}
