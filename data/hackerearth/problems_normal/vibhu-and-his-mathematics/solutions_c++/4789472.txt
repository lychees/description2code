#include <iostream>
#include<vector>
#define m 1000000007
using namespace std;

vector<long long int> F;

long long int funct(int n){
	long long int k;
	if(n<=F.size())
		return F[n-1];
	else{
		k = ((n-1)*funct(n-2))+funct(n-1);
		F.push_back(k%m);	
	}
	return F[n-1];
}
int main()
{
    int t;
    cin>>t;
    F.push_back(1);
    F.push_back(2);
    while(t--){
	    int n;
	    cin>>n;
	    cout<<funct(n)<<endl;
    }
    return 0;
}
