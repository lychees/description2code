#include <iostream>
#include <string>
#include <map>
using namespace std;

int main()
{
    map<char,int> mp;
    mp['0'] =6;
    mp['1'] =2; 
    mp['2'] =5; 
    mp['3'] =5; 
    mp['4'] =4; 
    mp['5'] =5; 
    mp['6'] =6; 
    mp['7'] =3; 
    mp['8'] =7; 
    mp['9'] =6;
    string str;
    cin>>str;
    int res=0;
    for(int i=0;i<str.size();i++)
    {
    	res+=mp[str[i]];
    }
    cout<<res<<endl;
    return 0;
}
