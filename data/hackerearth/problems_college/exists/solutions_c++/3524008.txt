#include <iostream>
#include<stdio.h>
#include<bits/stdc++.h>
using namespace std;
int search(long int a[],int e,int l,int u)
{       int m=(u-l)/2+l;
	if(l>u)
	return 0;
	if(a[m]==e)
	return 1;
	if(a[m]>e)
	return search(a,e,l,m-1);
	return search(a,e,m+1,u);
}
int main()
{      long int a[100000];
   int t;
   int n,ans,i,k;
   long int q;
   scanf("%d",&t);
   while(t--)
   {
   	scanf("%d",&n);
   	for(i=0;i<n;i++)
   	scanf("%ld",&a[i]);
   	sort(a,a+n);
   	scanf("%d",&k);
   	for(i=0;i<k;i++)
   	{scanf("%ld",&q);
   	 int ans=search(a,q,0,n-1);
   	 if(ans==1)
   	 printf("Yes\n");
   	 else
   	 printf("No\n");
   	}
   }
    return 0;
}
