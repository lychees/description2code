#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ll int n;
	scanf("%lld",&n);
	ll int i,j;
	list<ll int> l[n];
	for(i=0;i<n;i++)
	{
		scanf("%lld",&j);
		if(j==-1)
		continue;
		else
		{
			l[i].push_back(j-1);
		}
	}
	int visited[n];
	ll int mc=0,ans=0;
	for(i=0;i<n;i++)
	{
		memset(visited,0,sizeof(visited));
		mc=0;
		stack<ll int> s;
		s.push(i);
		while(!s.empty())
		{
			j=s.top();
			s.pop();
			if(!visited[j])
			{
				mc++;
				visited[j]=1;
				list<ll int>::iterator it;
				for(it=l[j].begin();it!=l[j].end();it++)
				{
					if(!visited[*it])
					{
						s.push(*it);
					}
				}
			}
		}
		ans=max(ans,mc);
	}
	printf("%lld\n",ans);
	return 0;
}
