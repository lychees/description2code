#include <iostream>
#include<bits/stdc++.h>
using namespace std;
char str[101];
int main()
{
   int t,i;
   cin>>t;
   //assert(1<=t && t<=10);
   while(t--)
   {
   	cin>>str;
   	int len=strlen(str);
   	//assert(1<=len && len<=100);
   	int arr[4]={0};
   	for(i=0;i<len;i++)
   	 {
   	 	if(str[i]=='r')
   	 	 arr[0]+=1;
   	 	if(str[i]=='u')
   	 	 arr[1]+=1;
   	 	if(str[i]=='b')
   	 	 arr[2]+=1;
   	 	if(str[i]=='y')
   	 	 arr[3]+=1;
   	 }
   	 int ans=arr[0];
   	 for(i=1;i<4;i++)
   	  ans=min(ans,arr[i]);
   	  cout<<ans<<endl;
   }
    return 0;
}
