#include<bits/stdc++.h>
#define LL long long
using namespace std;
int main()
{
    int t;
    LL Fib[100]={0};
    Fib[0]=0;
    Fib[1]=1;
    Fib[2]=1;
    for(int j=3;j<=99;j++)
        Fib[j]=Fib[j-1]+Fib[j-2];
    cin>>t;
    while(t--)
    {
        int m;
        cin>>m;
        if(m==1)
        {
            cout<<"0 1\n";
            continue;
        }
        if(m==2)
        {
            cout<<"1 1\n";
            continue;
        }
        cout<<Fib[m-1]<<" "<<Fib[m]<<endl;
    }
    return 0;
}
