#include <bits/stdc++.h>
using namespace std;


int main()
{int t;
cin>>t;
while(t--){
  int m,n,k,pos=0;
  cin>>m>>n>>k;
  int z=n;
  int a[m+n+1];
  for(int i=1;i<=m+n;i++)a[i]=0;
  while(n){
      int tmp=k;
      tmp%=m+n;
      if(tmp==0)tmp=m+n;
     int count=0 ;
    // cout<<tmp<<endl;
     while(count<tmp && pos+1<=m+z) {
         pos++;
         if(a[pos]!=-1)count++;
         
     }
     if(count<tmp)pos=0;
     while(count<tmp){
         pos++;
         if(a[pos]!=-1)count++;
     }
     
     a[pos]=-1;
     n--;
      
      //for(int i=1;i<=m+z;i++)cout<<a[i]<<' ';
      //cout<<endl;
  }
  for(int i=1;i<=m+z;i++){
      if(a[i]==-1)cout<<"F";
      else cout<<"M";
  }cout<<endl;
}
    return 0;
}
