//Even Odd
#include<iostream>
#include<stdio.h>

using namespace std;

int main(){
	long long t,r,c,odd,even;
	scanf("%lld",&t);
	while(t>0){
		scanf("%lld%lld",&r,&c);

	even=0;
	odd=0;
	if (r<=c){
		if (r%2!=0){
			odd=(((r+1)/2)*(1+r))/2;
			even=((r/2)*(1+r))/2;
		}
		else{
			odd=((r/2)*(r))/2;
			even=((r/2)*(2+r))/2;
		}
	}	
	else{
		if (c%2!=0){
			odd=(((c+1)/2)*(1+c))/2;
			even=((c/2)*(1+c))/2;
			if ((r-c)%2){
				even+=(((r-c+1)/2)*c);
				odd+=(((r-c)/2)*c);
			}
			else{
				even+=(((r-c)/2)*c);
				odd+=(((r-c)/2)*c);
			}
		}		
		else{
				odd=((c/2)*(c))/2;
				even=((c/2)*(2+c))/2;
				if ((r-c)%2){
					even+=(((r-c)/2)*c);
					odd+=(((r-c+1)/2)*c);
				}
				else{
					even+=(((r-c)/2)*c);
					odd+=(((r-c)/2)*c);
				}
		}
	}			
	printf("%lld %lld\n",even,odd);
	t-=1;
	}
	return 0;
}
