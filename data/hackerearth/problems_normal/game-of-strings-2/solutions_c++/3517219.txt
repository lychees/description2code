int strlenL( char * string1)
{
	int length = 0;
	while(string1[length] != '\0')
		length++;

	return length;
}

int findSequence(char * stringA, int length, int num, int *start)
{
	int counter = 0;
	int i = *start + 1;
	while(i < length )
	{
		int counter_length = 0;
		if( stringA[0] == stringA[i])
		{
			
			while( stringA[counter_length] == stringA[i+counter_length] )
				counter_length++;
			if( stringA[i+counter_length] == '\0' ) {
				counter = 1;
			}
			else
				counter_length = 0;
		}
		if(counter == 1) {
			*start = i;
			return counter_length;
			
		}
		i = i+counter_length;
		i++;
	}

	return 0;
}
