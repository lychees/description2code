#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int ans=n;
    	int factor=1;
    	while(factor++)
    	{
    		int quo=n/factor;
    		quo-=factor-1;
    		if(quo<=0)
    			break;
    		ans+=quo;
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
