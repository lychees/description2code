#include<bits/stdc++.h>
using namespace std;
int main()
{
  //  ios_base::sync_with_stdio(false);
    int t;
    cin>>t;
scanf("%*c");
    while(t--)
    {
        long long n,i;
//cin>>n;
        char s[105];
        gets(s);
        int x=0;
        n=strlen(s);
        for(i=0;i<n;i++)
        {
            x+=s[i];
        }
        x/=n;
        if(x&1)
        {
            for(i=n-1;i>=0;i--)
            {
                cout<<s[i];
            }
            cout<<endl;
        }
        else
            cout<<s<<endl;
    }
    return 0;
}
