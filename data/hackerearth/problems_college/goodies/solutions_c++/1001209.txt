#include <iostream>
using namespace std;

int main()
{
  int t,i;
  cin>>t;
  int n;
  int x[10000];
  while(t--)
  {
  	for(i=0;i<10000;i++)
  	x[i]=0;
  	cin>>n;
  	int a[n];
  	for(i=0;i<n;i++)
  	{
  		cin>>a[i];
  		x[a[i]]++;
  	}
  	int m=0;
  	for(i=0;i<10000;i++)
  	{
  		if(x[i]>m)
  		m=x[i];
  	}
  cout<<m<<endl;
  }
    return 0;
}
