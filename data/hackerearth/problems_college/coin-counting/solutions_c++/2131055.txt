#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    int t,a;
   cin>>t;
   while(t--)
   {
   	cin>>a;
   	int arr[105];
   	memset(arr,0,sizeof arr);
   	arr[0]=1;
    for(int j=1;j<=a;j++)
    arr[j]+=arr[j-1];
    for(int j=3;j<=a;j++)
    arr[j]+=arr[j-3];
    for(int j=5;j<=a;j++)
    arr[j]+=arr[j-5];
    cout<<arr[a]<<"\n";
   }
    
    return 0;
}
