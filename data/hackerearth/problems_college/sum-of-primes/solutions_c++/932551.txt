#include <cstdio>
#include <vector>
using namespace std;

vector<bool> p(1000001,1);
vector<long long> a(1000001,0LL),pr;

bool isp(long long n)
{
	if(n<1000001) return p[n];
	for(int i=0;i<pr.size();i++)
	{
		if(pr[i]*pr[i]>n) break;
		if(n%pr[i]==0) return 0;
	}
	return 1;
}

int main()
{
	long long i,j,t,n;
	p[0]=p[1]=0;
	for(i=2;i<1000001;i++)
	{
		if(p[i])
		{
			pr.push_back(i);
			for(j=i<<1;j<1000001;j+=i)
				p[j]=0;
			a[i]=i;
		}
		a[i]+=a[i-1];
	}
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld%lld",&i,&j);
		n=a[j]-a[i-1];
		if(isp(n)) printf("YES\n");
		else printf("NO\n");
	}
    return 0;
}
