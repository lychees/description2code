long long n;

int main() {
	ios_base::sync_with_stdio(false); cin.tie(0);
	cin >> n;
	long long m = (n >> 1LL);
	m++;
	while (m <= n) {
		long long x = m;
		for (long long i = 2; i * i <= (x); i++) {
			long long pp = 1;
			while (x % i == 0) {
				pp *= i;
				x /= i;
			}
			if (pp == 1) continue;
			if ((pp << 1LL) > n) {
				cout << m;
				return 0;
			}
			if (2 * x <= n)
				break;
		}
		if (2 * x > n)
			return cout << m, 0;
		m++;
	}
}
