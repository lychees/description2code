#include<iostream>
#include<queue>
#include<vector>
#include<utility>
#include<algorithm>
#include<unordered_map>
using namespace std;
typedef long long LL;
int main() {
  LL n, m;
  cin >> n >> m;
  unordered_map<LL, LL> record;
  priority_queue<pair<LL, LL>> container;
  LL age;


  while (n--) {
    cin >> age;
    record[age]++;
    container.push(make_pair(record[age], age));
    cout << container.top().second <<" "<< container.top().first << endl;
  }

}
