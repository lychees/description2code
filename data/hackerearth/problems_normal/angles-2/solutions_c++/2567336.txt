#define Geti(x) scanf("%d",&x)

const double eps = 1e-6;

bool make[555];

int main() {
	int n,k; cin >> n >> k;
	assert(n >= 1 and n <= 10);
	assert(k >= 1 and k <= 10);

	for (int i = 0; i < n; i ++) {
		int x; cin >> x;
		assert(x >=0 and x <= 360);
		
		int t = 0;
		for (int j = 0; j <= 360; j ++) {
			t = (x + t) % 360;
			make[t] = true;
		}
	}

	while (k --) {
		int x; cin >> x;
		if(make[x]) puts("YES");
		else puts("NO");
	}
	return 0;
}
