#include <iostream>
using namespace std;

int main()
{
    unsigned long int t;
    cin>>t;
    cin.get();
    while(t--) {
    	unsigned long long int n , fib_count=1 , i=1 , i_prev=1, ans ,number=0 , number_prev=0;
    	cin>>n;
    	while(number < n) {
    		ans = i+i_prev;
    		i_prev=i;
    		i=ans;
    		number_prev=number;
    		number = number+(i-i_prev-1);
    	}
    	number = i_prev + n - number_prev;
    	cout<<number<<endl;
    	
    }
    return 0;
}
