    #include<bits/stdc++.h>
    using namespace std;
     
    #define lld long long int
    #define gx getchar
    #define pb push_back
    #define in insert
    #define mp make_pair
    #define inf 2000000000
    #define mod 1000000007
     
    void scanint(int &x)
    {
        register int c = gx();
        x = 0;
        for(;(c<48 || c>57);c = gx());
        for(;c>47 && c<58;c = gx())
        {
            x = (x<<1) + (x<<3) + c - 48;
        }
    }
    int main()
    {
    	ios_base::sync_with_stdio(0);
    	int t,n,i;
    	int a[100007];
    	lld pre[100007],suc[100007];
    	lld res;
    	scanf("%d",&t);
    	while(t--)
    	{
    		memset(pre,0,sizeof(pre));
    		memset(suc,0,sizeof(suc));
    		res=inf;
    		scanf("%d",&n);
    		for(i=1;i<=n;i++) scanf("%d",&a[i]);
    		
    		for(i=1;i<=n;i++) pre[i]=pre[i-1]+a[i];
    		for(i=n;i>=1;i--) suc[i]=suc[i+1]+a[i];
    		
    		for(i=1;i<=n;i++)
    		{
    			res=min(abs(res),abs(pre[i]-suc[i+1]));
    		}
    		cout<<res<<endl;
    	}
    	return 0;
    }
