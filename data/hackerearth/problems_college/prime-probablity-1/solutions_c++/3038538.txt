#include <iostream>
#include <stdio.h>
#include <vector>
using namespace std;

int main()
{
    int test;
    vector<int>sieve(1000010,0);
    sieve[0]=sieve[1]=1;
    for(int i=2;i<1000010;i++)
    if(sieve[i]==0)
    	for(int j=2*i;j<1000010;j+=i)
    		sieve[j]=1;
    vector<int>sum(1000010);
    sum[1]=0;
    for(int i=2;i<1000010;i++)
    if(sieve[i]==0)
    	sum[i]+=sum[i-1]+1;
    else sum[i]+=sum[i-1];
    
    cin>>test;
    while(test--)
    {
    	float l,r,left;
    	float prime=0;
    	cin>>l>>r;
    	if(sieve[l]==0)//prime
    	prime=sum[r]-sum[l]+1;
    	else    	
    	prime=sum[r]-sum[l];
    	
    	float ans=prime/(r-l+1);
    	printf("%.6f\n",ans);
    	
    }
    return 0;
}
