struct  node
{
	long long dest;
	struct node * next;
};

struct node * g[10001];

struct node * new_node(long long  d){
	struct node * t=new struct node;
	t->dest=d;
	t->next=NULL;
	return t;
}

	t->next=g[a];
	g[a]=t;
	t=new_node(a);

}
