#include <iostream>
#include <bits/stdc++.h>
using namespace std;


long long reverse(long long x)
{
 long long num=0;
 
 while(x)
 {
 	num=num*10+(x%10);
 	x/=10;
 }
 return num;
}
int main()
{
    int t;
    cin>>t;
    
    while(t--)
    {
     long long a,b;
     scanf("%lld%lld",&a,&b);
     
     long long A=0,B=0;
     A=reverse(a);
     B=reverse(b);
     cout<<reverse(A+B)<<endl;
    	
    }
    return 0;
}
