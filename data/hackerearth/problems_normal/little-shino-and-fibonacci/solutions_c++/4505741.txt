#include <bits/stdc++.h>
#define ll long long 
using namespace std;
const int mod = 1e9 + 7 ;
const int l11 = 1e4 ; 
int f[201000];
ll part[5][10005]; 
int t ;
ll l , r ; 
int main () {
	  f[1] = 0 , f[2] = 1 ;
	  part[1][2] = 1  ;
	  ll pre = 1 ,pre1 =0ll ,pre2 = 0ll;
	  for (int i = 3 ;i <= 3*l11 ;i++){
	  	  f[i] = (f[i-1] + f[i-2] )  % 10000;
	  	  if ( i <= l11 ){
	  	  	      part[1][i] = 1ll*f[i] + pre;
	  	  	      part[1][i] %= mod ;
	  	  	      pre = part[1][i] ; 
			}
			else if (i <= 2*l11){
				part[2][i-l11] = 1ll*f[i] + pre1  ;
				part[2][i-l11] %= mod ;
				pre1 = part[2][i-l11]  ;  
			}
			else  if (i <= 3*l11)   {
				  part[3][i-2*l11] = 1ll*f[i]   + pre2 ;
				  part[3][i-2*l11]  %= mod ;
				  pre2 = part[3][i-2*l11] ; 
			}
		//	else      cnt += 1ll*f[i] ; 
		//	if (i % 10000 == 0)   cout << cnt <<endl ,cnt = 0;  
	  } 
	  cin >> t ;
	  while (t--){
	  	  cin >> l >> r ;
	  	  l--;
	  	  ll l1 = l / l11 ; 
	  	  ll r1 = r / l11 ;
	  	  ll ans = 0ll ,ans1 =0ll;
	  	  ll p1 = r1 / 3 ;
	  	  p1 = p1 % mod ;
		  ans = p1 *(part[1][l11] + part[2][l11] + part[3][l11]) % mod  ;
		  for (int i = 1 ; i <=r1 % 3 ; i++){
		  	  ans +=  part[i][l11] ;
				ans %= mod ; 
		  }
		  int p2 = ( (r1  % 3 ) + 1 ) % 3 ;
		  if (!p2)  p2  = 3 ;
		  ans += part[p2][r - r1*l11] ;
		  ans %= mod ;    
		  p1 = l1 / 3 ;
		  p1 %= mod ;
		  ans1 = p1 *(part[1][l11] + part[2][l11] + part[3][l11]) % mod  ;
		  for (int i = 1 ; i <= l1 % 3 ; i++){
		  	  ans1 +=  part[i][l11] ; 
		  	  ans1 %= mod ;
		  }
		  p2 = ( (l1  % 3 ) + 1 ) % 3 ;
		  if (!p2)     p2 = 3 ;
		  ans1 += part[p2][l - l1*l11] ;
		  ans1 %= mod ;   
			ans =  ans - ans1 ;
			while (ans < 0 )  ans += mod ;
			cout << ans << endl;
	  }
	 // cout << part[1] <<" "<< part[2] << endl; 
	  return 0 ;
}
