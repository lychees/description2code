#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
    int t;
    cin>>t;
    string s1 = "cde",s2 = "acf",s3 = "b3",s4 = "c6a";
    while(t--)
    {
    	char c[100000];
    	cin>>c;
    	int i =0;
    	while(c[i])
    	{
    		if(c[i] == '2')
    		cout<<s1;
    		else if(c[i] == '7')
    		cout<<s2;
    		else if(c[i]== '6')
    		cout<<s3;
    		else if(c[i] == '9')
    		cout<<s4;
    		i++;
    	}
    	cout<<endl;
    }
    return 0;
}
