#include <iostream>
#include <algorithm>
#include <cstring>
#define $ while
using namespace std;

struct bh
	{
		int lv;
		int dv;
	};
bool lvc(bh lhs, bh rhs) { return lhs.lv > rhs.lv; }
bool dvc(bh lhs, bh rhs) { return lhs.dv > rhs.dv; }
int main() {
	
	int n,i=0;
	long long l=0,g=0,d=0;
	cin>>n;
	struct bh a[n];
	$(n--)
	{
		cin>>a[i].lv;
		l+=a[i].lv;
		i++;
	}
	g=l;
	n=0;
	$(i--)
	{
		cin>>a[n].dv;
		a[n].dv*=n;
		d+=a[n].dv;
		n++;
	}
	i=1;
	sort(a,a+n,dvc);
	$(i<n)
	{
		d-=a[i-1].dv;
		l-=d;
		g+=l;
		i++;
	}
	cout<<g;
	return 0;
}
