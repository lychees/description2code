#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    long long x, y, n, k;
    cin>>n>>k;
    long long i, j;
    long long a[n];
    for(i=0;i<n;i++){
    	a[i]=i+1;
    }

    for(i=0;i<k;i++){
        cin>>x>>y;
        a[y-1]=-1;
    }

    long long count=0;
    for(i=0;i<n;i++){
        if(a[i]==-1)
            continue;
        count++;
    }

    cout<<count<<endl;
    return 0;
}
