#include <iostream>
using namespace std;

int main()
{
    unsigned long long int T,N,M,i,j,t,len,a,b,d;
    cin>>T;
    for(i=0;i<T;i++){
    	cin>>M;
      	for(j=0,N=0;j<M;j++){
       		cin>>len>>d;
    		N = N + (len * d);
    		while(N>9) {
    			a = N / 10;
    			b = (N % 10);
    			N = a+b;
    		}
     	}
    		
       	cout<<N<<endl;
    }
    return 0;
}
