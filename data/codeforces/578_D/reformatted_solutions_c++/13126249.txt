#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <ctime>
#include <iostream>
#include <map>
#include <queue>
#include <set>
#include <vector>
using namespace std;
#define LL long long
#define UL unsigned long long
LL N, M, ans;
char a[100010];
int main() {
  scanf("%I64d%I64d", &N, &M);
  scanf("%s", a);
  ans = N * M - N;
  LL len = strlen(a);
  for (LL i = 1; i < len; i++)
    if (a[i] != a[i - 1])
      ans += N * M - N;
  LL ll = 1;
  for (LL i = 1; i < len; i++)
    if (ll == 1) {
      if (a[i] == a[i - 1])
        continue;
      else
        ll++;
    } else {
      if (a[i] == a[i - 2])
        ll++;
      else {
        ans -= (ll - 1) * ll / 2;
        if (a[i] == a[i - 1])
          ll = 1;
        else
          ll = 2;
      }
    }
  ans -= (ll - 1) * ll / 2;
  printf("%I64d", ans);
}
