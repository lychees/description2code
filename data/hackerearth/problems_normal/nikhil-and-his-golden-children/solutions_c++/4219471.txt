#include<bits/stdc++.h>
using namespace std;
#define INF 1000000007
long long int power(long long int a,long long int b,long long int mod)
{
    long long int x=1,y=a;
    while(b>0)
    {
        if(b%2==1)
        {
            x=x*y;
            x=x%mod;
        }
        y=y*y;
        y=y%mod;
        b=b/2;
    }
    return x;
}
long long int mi(long long int n,long long int mod)
{
    return power(n,mod-2,mod);
}
long long int c(long long int n,long long int r,long long int mod)
{
    vector<long long int> f(n+1,1);
    long long int i;
    for(i=2;i<=n;i++)
    {
        f[i]=(f[i-1]*i)%mod;
    }
    return (f[n]*((mi(f[r],mod) * mi(f[n-r],mod)) % mod)) % mod;
}
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int t;
    cin>>t;
    while(t--)
    {
        long long int k,m;
        cin>>k>>m;
        long long int *a=new long long int[k];
        long long int i;
        for(i=0;i<k;i++)
        {
            cin>>a[i];
        }
        for(i=0;i<k;i++)
        {
            m=m-a[i];
        }
        if(m==0)
        {
        	cout<<"1"<<"\n";
        }
        else if(m<0)
        {
        	cout<<"0"<<"\n";
        }
        else if(m>0)
        {
        	cout<<c(m+k-1,k-1,INF)<<"\n";
        }
    }
}
