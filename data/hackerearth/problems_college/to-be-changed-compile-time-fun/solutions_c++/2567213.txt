#include <bits/stdc++.h>
using namespace std;

int main()
{
   long int n,a,i,sum=0,m,final=0,j;
    cin>>n;
    vector<long int>v,v1;
// vector<int>::iterator it,it1;
 	for(i=0;i<n;i++)
    {
    	cin>>a;
    	v.push_back(a);
    }
   
	while(v.size()!=2)
	{
		sum=0;
		for(i=1;i<n-1;i++)
		{
			if(sum<v[i-1]*v[i+1])
			{	
				sum=v[i-1]*v[i+1];
				j=i;
			}
		}
		v.erase(v.begin()+j);
		n=v.size();
		final+=sum;
	}
	cout<<final;
}
