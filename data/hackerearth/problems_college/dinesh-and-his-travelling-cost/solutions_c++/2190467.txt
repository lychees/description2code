#include<bits/stdc++.h>
using namespace std;
int main(){
	int n; cin>>n;
	vector<pair<int,int> > a[510];
	while(n--){
		int x,y,z;
		cin>>x>>y>>z;
		a[x].push_back(make_pair(y,z));
		a[y].push_back(make_pair(x,z));
	}
	int dist[501];
	for(int i=0; i<501; i++) dist[i]=999999;
	priority_queue<pair<int,int> >q;
	int st; cin>>st;
	dist[st]=0;
	q.push(make_pair(0,st));
	while(!q.empty()){
		int node=q.top().second;
		q.pop();
		for(int i=0; i<a[node].size(); i++){
			int next=a[node][i].first;
			int wt=a[node][i].second;
			if(dist[next]>dist[node]+wt){
				dist[next]=dist[node]+wt;
				q.push(make_pair(-dist[next],next));
			}
		}
	}
//	for(int i=0; i<10; i++) cout<<dist[i]<<" ";
	int qq; cin>>qq;
	while(qq--){
		int nn; cin>>nn;
		if(dist[nn]>=999999) cout<<"NO PATH\n";
		else cout<<dist[nn]<<endl;
	}
}
