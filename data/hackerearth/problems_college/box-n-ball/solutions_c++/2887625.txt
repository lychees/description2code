#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    long long int l,b,h;
    cin >> t;
    while(t--)
    {
    	cin >> l >> b >> h;
    	long long m=min(l,min(b,h));
    	long double f=m/2.0;
    	printf("%.1llf\n",f);	
    }
    
    
    
    
    return 0;
}
