#include <iostream>
using namespace std;

int main()
{
   int a,b;
   cin>>a>>b;
   for(int i=a+1;i<b;i++)
   {
   	if(i+1<b)
   	{
   	if(i%3==0 && i%5==0)
   	   cout<<i<<",";
   	else if(i%3==0||i%5==0)
   	    continue;
   	  else
   	    cout<<i<<",";
   	}
   	else
   	{
   	if(i%3==0 && i%5==0)
   	   cout<<i;
   	else if(i%3==0||i%5==0)
   	    continue;
   	  else
   	    cout<<i;

   	}
   }
   return 0;
}
