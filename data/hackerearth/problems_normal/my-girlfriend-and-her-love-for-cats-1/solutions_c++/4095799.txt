#include <bits/stdc++.h>
using namespace std;

int main()
{
    vector<long long int> s;
    vector<long long int> c;
    long long int n;
    cin>>n;
    for(long long int i=0;i<n;i++)
    {
    	int temp1;
    	cin>>temp1;
    	s.push_back(temp1);
    	
    }
    for(long long int i=0;i<n;i++)
    {
    	int temp2;
    	cin>>temp2;
    	c.push_back(temp2);
    }
    sort(s.begin(),s.end());
    sort(c.begin(),c.end());
    long long int sum=0;
    for(long long int i=0;i<n;i++)
    {
    	sum=sum+(s[i]*c[i]);
    }
    cout<<sum<<endl;
    
}
