#include<iostream>
using namespace std;
int main()
{
	int n;
	int a[1000][1000];
	cin>>n;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
		{
			cin>>a[i][j];
		}
	}
	int fr,sc,th;
	for(int k=0;k<n;k++)
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
			if(a[i][k]+a[k][j]<a[i][j])
				a[i][j]=a[i][k]+a[k][j];
	}
	int m;
	cin>>m;
	while(m--)
	{
		cin>>fr>>sc>>th;
		int val=a[fr][sc]+a[sc][th];
		int dir=a[fr][th];
		cout<<val<<" "<<val-dir<<endl;
	}
	return 0;
}
