#include <bits/stdc++.h>
using namespace std;

#define ll long long

int n;
int nums[1005];
vector <int> sol[105];

int able(int num) {
	int left = 0;
	int sum = 0;
	int res = 0;
	for (int right = 0; right < n; right++) {
		sum+= nums[right];
		while (sum > num) {
			sum-= nums[left++];
		}
		if (sum == num) {
			res++;
			sum = 0;
			left = right+1;
		}
	}
	return res;
}

int main() {
	ios::sync_with_stdio(0);
	cin >> n;
	for (int i = 0; i < n; i++) {
		cin >> nums[i];
	}
	for (int i = 1; i <= 100000; i++) {
		int num = able(i);
		sol[num].push_back(i);
	}
	for (int i = 104; i>=0; i--) {
		if (sol[i].size() > 0) {
			cout << i << endl;
			for (int j = 0; j <sol[i].size(); j++) {
				cout << sol[i][j];
				cout << " ";
			}
			cout << endl;
			return 0;
		}
	}
	return 0;
}
