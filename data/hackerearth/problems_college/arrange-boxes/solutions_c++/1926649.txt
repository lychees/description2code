
#include <iostream>
#include <math.h>
using namespace std;
int main ()
{
	long t,a[100000],b[100000],index[100000],no,count,x;
	cin>>t;
	while (t--)
	{
	cin>>no;
	count=0;
	for (long i=0;i<no;i++)
	{
		cin>>a[i];
		index[a[i]]=i;
	}
	for (long i=0;i<no;i++)
	{
		cin>>b[i];
	}
	for (long i=no-1;i>=0;i--)
	{
		x=b[i];
		long temp=index[x];
		count+=abs(temp-i);
		for (long j=temp;j<i;j++)
		{
			long temp1=a[j];
			a[j]=a[j+1];
			a[j+1]=temp1;
			index[a[j]]--;
		}
	}

	cout<<count<<"\n";
}
}
