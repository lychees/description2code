#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s,a;
    int t,k,la,ls,flag,i,j,cnt=0,p;
    cin>>a;
    cin>>k;
    cin>>t;
    la=a.size();
    while(t--)
    {
    	cin>>s;
    	ls=s.size();
    	p=0;
    	for(i=0;i<ls;i++)
    	{
    		flag=0;
    		for(j=p;j<la;j++)
    		{
    			if(a[j]==s[i])
    			{
    				flag=1;
    				p=j+1;
    				break;
    			}
    			else
    			{
    				flag=0;
    			}
    		}
    		if(flag==0)
    		{
    			break;
    		}
    	}
    	if(flag==1)
    	{
    		cnt++;
    	}
     }
     if(cnt>=k)
     {
     	cout<<cnt<<" "<<"LUCKY"<<endl;
     }
     else
     {
     	cout<<cnt<<" "<<"UNLUCKY"<<endl;
     }
    return 0;
}
