#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,a,b,c,x,y,p1=0,p2=0;
    	cin>>n;
    	cin>>a>>b>>c;
    	while(n--)
    	{
    		cin>>x>>y;
    		if(a*x+b*y+c>0)
    		{
    			p1++;
    		}
    		else if(a*x+b*y+c<0){
    			p2++;
    		}
    	}
    	if(p1>p2)
    	cout<<p1<<endl;
    	else
    	cout<<p2<<endl;
    }
    return 0;
}
