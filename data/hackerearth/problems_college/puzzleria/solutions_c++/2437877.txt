#include<bits/stdc++.h>
using namespace std;
long long int NCR(int n,int r)
{
	if(r==1)
	{
		return n;
	}
	if(n==r)
	{
		return 1;
	}
	else
	{
		return (NCR(n-1,r)+NCR(n-1,r-1));
	}
}
long int fact(int n)
{
	if(n==1)
	{
		return 1;
	}
	else
	{
		return (n*fact(n-1));
	}
}
