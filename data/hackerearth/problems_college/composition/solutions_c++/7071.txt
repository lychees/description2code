struct matrix
{   public:
    long long a0,a1,a2,a3;
    
    void multiply(matrix b)
    {long long c[4];
    long long v=(a0*b.a0+a1*b.a2)%1000000007;
    c[0]=v;
    v=(a0*b.a1+a1*b.a3)%1000000007;
    c[1]=v;
    v=(a2*b.a0+a3*b.a2)%1000000007;
    c[2]=v;
     v=(a2*b.a1+a3*b.a3)%1000000007;
    c[3]=v;
    
    a0=c[0];a1=c[1];a2=c[2];a3=c[3];
    }   
};

}
