t = int(raw_input())
d={}
while t>0:
	val=raw_input().split()
	if d.has_key(int(val[1])):
		d[int(val[1])].append(val[0])
	else:
		d.setdefault(int(val[1]), [])
		d[int(val[1])].append(val[0])	
	t-=1
for key,value in sorted(d.iteritems(),key=lambda(k,v): -k):
		value.sort()
		for x in value:
			print x,key
