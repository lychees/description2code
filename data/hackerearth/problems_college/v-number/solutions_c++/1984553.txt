#include <iostream>
#include <stdio.h>
#include <string>
using namespace std;

int main()
{
	long long int testCases;
	scanf("%lld", &testCases);
	for(long long int c=0;c<testCases;c++)
	{
		string n;
		cin>>n;
		int siz = n.size();
		int sum1 = 0, sum2 = 0;
		int i = 0;
		
		for(i=0;i<(siz)/2;i++)
			sum1 += (int)n[i] - 48;
		//printf("   %d  %d   ", i, (siz/2));	
		for(int j=i; j<siz;j++)
			sum2 += (int)n[j] - 48;
		if(sum1 == sum2)
			printf("YES\n");
		else
			printf("NO\n");
	}
	return 0;
}
