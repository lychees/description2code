#include <iostream>
#include <cstring>
#include<stdio.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	char str[100];
    	scanf(" %[^\n]s",&str);
    	int len = strlen(str);
    	int w=0;
    	//cout<<len;
    	for(int i =0;i<len;i++)
    			w+= str[i];
    	int l = w/len;
    	if(!(l%2))
    		printf("%s",str);
    	else
    		for(int i =len-1;i>=0;i--)
    		printf("%c",str[i]);
    	printf("\n");
    }
}
