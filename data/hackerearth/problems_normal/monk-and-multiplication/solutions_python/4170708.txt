num = input()
Array = map(int, raw_input().split())
H1 = -1 
H2 = -1
H3 = -1

for Index in range(len(Array)):
	if H1 < Array[Index]:
		H3 = H2
		H2 = H1
		H1 = Array[Index]
		
	elif H2 < Array[Index]:
		H3 = H2
		H2 = Array[Index]
		
	elif H3 < Array[Index]:
		H3 = Array[Index]
		
	if Index < 2:
		print "-1"
	else:
		print str(H1 * H2 * H3)
