#include<bits/stdc++.h>
using namespace std;

#define fastScan ios_base::sync_with_stdio(0); cin.tie(NULL);

int main()
{
    fastScan;
    string s,t;
    cin>>s>>t;
    int i,ls,lt,j,k,count=0;
    map<string,int> mp;
    ls=s.size();
    lt=t.size();
    for(j=lt;j<=ls;j++){
        for(i=0;i<=ls-j;i++){
            string temp;
            for(k=i;k<i+j;k++){
                 temp+=s[k];
            }
            if(mp.find(temp)==mp.end()){
            size_t found=temp.find(t);
            if(found!=string::npos)
              count++;
              mp[temp]=count;
            }
        }
    }
    cout<<count;
}
