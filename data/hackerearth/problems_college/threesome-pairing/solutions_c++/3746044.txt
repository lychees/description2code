#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	int a[n+1];
    	for(int k=0;k<n;k++){
    		cin>>a[k];
    	}
    	int count=0;
    	for(int i=0;i<n;i++){
    		for(int k=i+1;k<n;k++){
    			if((a[i]+a[k])%3  == 0){
    				count++;
    			}
    		}
    	}
    	cout<<count<<endl;
    }
    return 0;
}
