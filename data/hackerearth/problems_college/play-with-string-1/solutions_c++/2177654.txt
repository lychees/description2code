#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,i,t1,t2,a,b,j;
    char c1,c2;
    string f="Funny",nf="Not Funny",s;
    cin>>n;
    while(n--)
    {
        int flag=0;
        cin>>s;
        for(i=1,j=s.size()-2;i<s.size();i++,j--)
        {
            c1=s[i];
            t1=c1-96;
            c2=s[i-1];
            t2=c2-96;
            a=t1-t2;
            if(a<0)
                a=-1*a;
            c1=s[j];
            t1=c1-96;
            c2=s[j+1];
            t2=c2-96;
            b=t1-t2;
            if(b<0)
                b=-1*b;
            if(a!=b)
            {
                flag=1;
                break;
            }
        }
        if(flag==1)
            cout<<nf<<endl;
        else
            cout<<f<<endl;
    }
}
