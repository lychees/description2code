#include<iostream>
using namespace std;
int main()
{
std::ios_base::sync_with_stdio(false);
int t;
cin>>t;
while(t--)
{
    int ans=0;
string s1,s2;
cin>>s1>>s2;
int len=min(s1.length(),s2.length()),i;
for(i=0;i<len;i++)
{
    if(s1[i]==s2[i])
        ans++;
}
cout<<ans<<endl;
}
return 0;
}
