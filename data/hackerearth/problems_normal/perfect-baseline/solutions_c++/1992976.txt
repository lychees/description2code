#include <bits/stdc++.h>
using namespace std;

		char str[10010][25];
int main()
{
	int n,k,t;
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		char res[25];
	string S[10010];
		for(int i=0;i<n;i++)
		{
			cin>>str[i];
		}

	for(int i=0;i<k;i++)
	{
		for(int j=0;j<n;j++)
		S[i].push_back(str[j][i]);
	}
	for(int i=0;i<k;i++)sort(S[i].begin(),S[i].end());
	int i;
	for(i=0;i<k;i++)
    {
        res[i]=min(S[i][n>>1],S[i][n-1>>1]);
    }
    res[i]='\0';
    cout<<res<<endl;
	}
    return 0;
}
