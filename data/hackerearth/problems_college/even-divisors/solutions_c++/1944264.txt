#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    long long t,num,cnt,i,one,other;
    cin >> t;
    while(t--)
    {
    	cnt=0;
    	cin >> num;
    	for(i=1;i<=sqrt(num);i++)
    	{
    		if(num%i==0)
    		{
    			one=i;
    			other=num/i;
    			if(one%2==0)
    			{
    				if(other%2==0)
    				{
    					if(one==other)
    						cnt++;
    					else
    						cnt+=2;
    				}
    				else
    					cnt++;
    			}
    			else
    			{
    				if(other%2==0)
    				{
    					if(one%2==0)
    					{
    						if(one==other)
    							cnt++;
    						else
    							cnt+=2;
    					}
    					else
    						cnt++;
    				}
    			}
    		}
    	}
    	cout << cnt << endl;
    }
    return 0;
}
