#include <iostream>
#include <algorithm>
#include <climits>
#include <vector>
using namespace std;

int main()
{
    int i, N, k;
    string s;
    cin>>s;
    int b[26]={};
    cin>>k;
    vector<pair<int, int> > v;
    for(i=0; s[i]!='\0'; ++i)
    	b[s[i]-'a']++;
    N=i;
    for(i=0; i<26; ++i)
    	v.push_back(make_pair(b[i],i));
    sort(v.begin(), v.end());
    int keep[26]={};
    int sum=0;
    for(i=0; i<26; ++i)
    {
    	if(sum+v[i].first<=k)
    	{
    		sum+=v[i].first;
    		keep[v[i].second]=1;
    	}
    	else
    		break;
    }
    int cnt=0;
    for(i=0; i<26; ++i)
    {
    	if(!keep[i])
    		cnt++;
    }
    cout<<cnt<<endl;
    for(i=0; i<N; ++i)
    {
    	if(!keep[s[i]-'a'])
    		cout<<s[i];
    }
    return 0;
}
