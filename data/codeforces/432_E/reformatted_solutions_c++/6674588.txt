using namespace std;
typedef long long LL;
typedef pair<int, int> P;
#define N 105
char mp[N][N];
char get(int x, int y) {
  if (mp[x][y])
    return mp[x][y];
  char ret = 'A';
  while (ret == mp[x - 1][y] || ret == mp[x][y - 1] || ret == mp[x + 1][y] ||
         ret == mp[x][y + 1])
    ret++;
  return ret;
}
int main() {
  int n, m;
  cin >> n >> m;
  for (int i = 1; i <= n; i++) {
    for (int j = 1; j <= m; j++) {
      char now = get(i, j);
      cout << now;
      if (mp[i][j])
        continue;
      int len = 1;
      while (i + len <= n && j + len <= m && get(i, j + len) == now)
        len++;
      for (int x = 0; x < len; x++)
        for (int y = 0; y < len; y++)
          mp[i + x][y + j] = now;
    }
    cout << endl;
  }
  return 0;
}
