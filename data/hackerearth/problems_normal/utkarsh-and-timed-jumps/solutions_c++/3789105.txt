#include <iostream>
#include <math.h>
using namespace std;

int main()
{
    int cases;
    cin>>cases;
    while(cases--)
    {
    	long val;
    	cin>>val;
      double res = (sqrt((2*val)+0.25))-0.5;
    	long result = (long)ceil(res);
    	while(((result*(result+1))/2)<val)
    		result++;
    	cout<<result<<endl;
    }
    return 0;
}
