#include <iostream>
using namespace std;
int	bin_src(long arr[],int last,long num)
{
	int beg=0;
	while(beg<=last)
	{	
		int mid=(beg+last)/2;
		if(arr[mid]==num)
			return mid;
		else if(num<arr[mid])
			last=mid-1;
		else
			beg=mid+1;
	}
	return -1;
}
int main()
{
	long mat[1000][1000],num;
	int r,c,q,i,j,flag;
	cin>>r;
	cin>>c;
	for(i=0;i<r;i++)
	{
		for(j=0;j<c;j++)
		{
			cin>>mat[i][j];
		}
	}
	cin>>q;
	for(i=0;i<q;i++)
	{
		cin>>num;
		for(j=0;j<r;j++)
		{
			flag=bin_src(mat[j],c,num);
			if(flag!=-1)
				break;
		}
		if(flag==-1)
		{
			cout<<"-1 -1\n";
		}
		else
		{
			cout<<j<<" "<<flag<<"\n";
		}
	}
	return 0;
}
