using namespace std;

bool RelativelyPrime (int a, int b) { // Assumes a, b > 0
   return (a<b) ? RelativelyPrime(b,a) : !(a%b) ? (b==1) : RelativelyPrime (b, a%b);
}

int solve(int A, int AA[], int n) {
	bool x;
	int count = 0;
	for (int i=0; i<n; i++) {
		x = RelativelyPrime(A,AA[i]);
		if (x==0) {
			count++;
		}
	}
	return count;
}

int main() {
	int A; // Number that max has
	cin>>A;
	int N;
	cin>>N;
	int AA[N];
	for (int i=0; i<N; i++)
		cin>>AA[i];
	int c = solve (A,AA,N);
	cout<<c<<endl;
}
