#include<iostream>
#include<algorithm>
#include<map>
#include<vector>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long int n,x=0,i,j,y=0,ans=0,z=0,e[105],f[105],g[105],x1=1,x2=1,x3,x4;
        cin>>n;
        x=n/2;y=n%2;z=0;x3=x;
        for(z=0;z<=x3;z++)
        {
            for(i=0;i<=x+y;i++){if(i==x+y){e[i]=1;continue;}e[i]=(x+y)-i;}
            for(i=0;i<=x;i++){if(i==x){f[i]=1;continue;}f[i]=x-i;}
            for(i=0;i<=y;i++){if(i==y){g[i]=1;continue;}g[i]=y-i;}
            for(i=0;i<=x;i++)
            {
                for(j=0;j<=x+y;j++){x4=__gcd(e[j],f[i]);e[j]=e[j]/x4;f[i]=f[i]/x4;}
            }
            for(i=0;i<=y;i++)
            {
                for(j=0;j<=x+y;j++){x4=__gcd(e[j],g[i]);e[j]=e[j]/x4;g[i]=g[i]/x4;}
            }
            for(i=0;i<=x+y;i++)x1*=e[i];
            ans+=x1;
            x--;y+=2;x1=1;x2=1;
        }
        cout<<ans<<"\n";

    }
    return 0;
}
