#include <iostream>
#include <cstring>
using namespace std;

int main()
{
	long t;
	cin>>t;
	while(t--)
	{
		char str[100000];
		cin>>str;
		long n=strlen(str);
		long i,count=0;
		for(i=1;i<n;i++)
	    {
	    		if(str[i-1]==str[i])
	    		{
	    			str[i-1]='@';
	    	
	    			
	    		}
	    		
	    	
	    }  
	    for(i=0;i<n;i++)
	    {
	    	
	    	if(str[i]=='@')
	    	{
	    		count++;
	    	}
	    }
	    cout<<count<<"\n";
		
	}
    return 0;
}
