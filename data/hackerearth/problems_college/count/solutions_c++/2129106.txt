#include <iostream>
using namespace std;
int main()
{
    long long int t,r,c,se,so,i,j;
    cin>>t;
    while(t--)
    {
        se=so=0;
        cin>>r>>c;
        if(r<=c)
        {
            for(i=1;i<=r;++i)
            {
            if(i%2==0)
                se=se+i;
            else so=so+i;
            }
            cout<<se<<" "<<so<<"\n";
        }
        else if(r>c)
        {
        for(i=1;i<=c;++i)
        {
            if(i%2==0)
            se=se+i;
            else
            so=so+i;
 
        }
        while(i<=r)
        {
            if(i%2==0)
                se=se+c;
            else so=so+c;
            ++i;
        }
        cout<<se<<" "<<so<<"\n";
        }
 
    }
 
}
