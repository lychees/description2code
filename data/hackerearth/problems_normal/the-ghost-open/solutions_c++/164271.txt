#include <iostream>

using namespace std;

int main()
{
    int t,i=0;
    cin>>t;
    int n[t];
    while (i<t)
    {
        cin>>n[i];
        i++;
    }
    for (int i=0; i<t; i++)
    {
    	if (n[i] != 0) {
        n[i] = n[i] - 1;
    	}
        cout<<n[i]<<endl;
    }
    return 0;
}
