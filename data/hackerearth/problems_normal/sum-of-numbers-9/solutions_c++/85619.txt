#include <iostream>
using namespace std;

bool getSum(int * A,int start,int end,int req_sum){
	if(req_sum==0)
	return true;
	if(start>end)
	return false;
	return (getSum(A,start+1,end,req_sum-A[start])||getSum(A,start+1,end,req_sum));
}

int main()
{
    int test;
    cin >> test;
    while(test--){
    int n;
    cin >> n;
    int A[n];
    for(int i=0;i<n;i++){
    	scanf("%d",&A[i]);
    }
   	int sum;
   	cin >> sum;
   	if(getSum(A,0,n-1,sum))
   		cout << "YES" << endl;
   	else
   		cout << "NO" << endl;
    }
    return 0;
}
