# include<iostream>
# include<cstring>
using namespace std;
int main()
{
	string s;
	int n,x,y;
	cin>>n>>x;
	y=n;
	int a[n][26],i,c=0,j,sum=0;
	for(j=0;j<n;j++)
	for(i=0;i<26;i++)
	a[j][i]=0;
	j=0;
	while(n--)
	{
		cin>>s;
		//for(j=0;j<y;j++)
		for(i=0;i<s.length();i++)
		a[j][s[i]-'a']=1;
		j++;
	}
	
	for(i=0;i<26;i++)
	{sum=0;
	for(j=0;j<y;j++)
	{
		sum+=a[j][i];
	}
	if(sum>=y)
	c++;
}
	if(c>=x)
	cout<<":-)\n";
	else
	cout<<":-(\n";
}
