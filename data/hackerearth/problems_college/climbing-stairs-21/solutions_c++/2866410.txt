#include <iostream>
using namespace std;

int main()
{
    int T;
    cin>>T;
    if(T<1||T>100)
    	return 0;
    long int ans[100];
    for(int i=0;i<T;i++)
    {
    	int c;
    	cin>>c;
    	if(c<1||c>1000)
    		return 0;
    	if(c==1)
    	{
    		ans[i]=1;
    		continue;
    	}
    	if(c==2)
    	{
    		ans[i]=2;
    		continue;
    	}
    	long int a=1,b=2,fib;
    	for(int i=2;i<c;i++)
    	{
    		fib=a+b;
    		a=b;
    		b=fib;
    	}
    	ans[i]=fib;
    	
    }
    for(int i=0;i<T;i++)
    {
    	cout<<ans[i]<<endl;
    }
    return 0;
}
