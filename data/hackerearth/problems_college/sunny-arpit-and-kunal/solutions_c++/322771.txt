#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
int main(){
    int t,a[3],diff1,diff2,ans=0;
    scanf("%d",&t);
    while(t--){
        scanf("%d%d%d",&a[0],&a[1],&a[2]);
        sort(a,a+3);
        diff1=a[1]-a[0];
        diff2=a[2]-a[1];
        ans=0;
        while(diff1!=1||diff2!=1){
            if(diff1>=diff2){
                swap(a[2],a[1]);
                a[1]=a[0]+1;
            }
            else{
                swap(a[0],a[1]);
                a[1]=a[2]-1;
            }        
            //printf("yes\n");
            //cout<<a[0]<<"\t"<<a[1]<<"\t"<<a[2]<<endl;
            diff1=a[1]-a[0];
            diff2=a[2]-a[1];
            ans++;
        }    
        printf("%d\n",ans);
    }    
    //system("pause");
    return 0;
}
