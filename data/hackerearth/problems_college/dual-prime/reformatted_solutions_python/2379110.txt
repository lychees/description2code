import math


def prime(n):
    if (n <= 1):
        return 0
    for i in range(2, n):
        if (n % i == 0):
            return 0
    return 1


t = int(input())
while t:
    t = t-1
    n = int(input())
    temp = n
    sum = 0
    while (n > 0):
        r = n % 10
        n = n/10
        sum += r
    if (prime(temp) and prime(sum)):
        print 'YES'
    else:
        print 'NO'
