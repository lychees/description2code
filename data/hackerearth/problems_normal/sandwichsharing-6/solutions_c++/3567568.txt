#include <iostream>
using namespace std;

int main()
{
    int test_cases;
    cin >> test_cases;
    while(test_cases--)
    {
        long int n;
        long int p;
        cin >> n >> p; 

        while(n > 3 && p > 0)
        {
            if(n % 2 == 1)
                n = n / 2 + 2;
            else
                n = n / 2 + 1;
            p--;
        }
        cout << n << endl;
    }
    return 0;
}
