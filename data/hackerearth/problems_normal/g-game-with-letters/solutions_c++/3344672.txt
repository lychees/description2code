pair< pair< int, int >, int > sc[100000];

int main()
{
    int t;
    sd(t);
    while(t--){
        int n, ar[26]={};
        string s;
        cin >> n >> s;
        for(auto &i: s)
            ar[i-'a']++;
        for(int i=0; i<n; i++){
            cin >> s;
            int curscore=0;
            for(auto &j: s)
                curscore += ar[j-'a'];
            sc[i]=mp(mp(-curscore, s.length()), i);
        }
        sort(sc, sc+n);
        cout << sc[0].S+1 << endl;
    }
    return 0;
}
