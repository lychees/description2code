li gcd(li a,li b)
{
          if(a == 0){
               return b;
          }
          else
          return gcd(b % a, a);
} 

int main()
{
    li temp, i, n, t;
    cin >> t;
    while(t--){
               li ans;
               cin >> n;
               cin >> temp;
               ans = temp;
               F(i, n - 1){
                    cin >> temp;
                    if(temp < ans)
                    ans = gcd(temp, ans);

                    ans = gcd(ans, temp);
               }
               if(ans == 1){

    }
