#include <iostream>
using namespace std;
 
#define MAX 20000000
 
long int dp[MAX+1];
 
int main() {
	// your code goes here
	int n;
	cin>>n;
	dp[0]=0;
	dp[1]=0;
	dp[2]=1;
	dp[3]=1;
 
	for(int i=4;i<=MAX;i++){
		dp[i]= 1+ dp[i-1];
		if(i%2==0)
		dp[i]= min(dp[i],1+dp[i/2]);
 
		if(i%3==0)
		dp[i]= min(dp[i],1+dp[i/3]);
	}
 
	for(int i=1;i<=n;i++){
		long long int a;
		cin>>a;
		cout<<dp[a]<<endl;
	}
	return 0;
}
