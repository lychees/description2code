#include <iostream>
#include <map>
#define INF 9999
using namespace std;

int main()
{
    int n, t, distance[100][100], i, j, d, k;
    for (i=0; i<100; i++){
    	for (j=0; j<100; j++){
    		distance[i][j] = INF;
    	}
    }
    char s[100], source[100], dest[100];
    scanf("%d %d\n", &n, &t);
    map<string, int> stations;
    for (i=0; i<n; i++) {
    	scanf("%s", s);
    	stations[s]=i;
    }
    while (t--) {
    	scanf("%s %s %d\n", source, dest, &d);
    	distance[stations[source]][stations[dest]] = d;
    	distance[stations[dest]][stations[source]] = d;
    }
for (k=0; k<n; k++) {
    	for (i=0; i<n; i++){
    		for (j=0; j<n; j++) {
    			if (distance[i][k]+distance[k][j]<distance[i][j]){
    				distance[i][j] = distance[i][k]+distance[k][j];
    			}
    		}
    	}
    }

    
    scanf("%d\n", &t);
    while (t--){
		scanf("%s %s\n", source, dest);
    	printf ("%d\n", distance[stations[source]][stations[dest]]);    	
    }
    return 0;
}
