for _ in range(int(input())):
    n = int(input())
    a = map(int, raw_input().split())
    a = sorted(a, reverse=True)
    ans = [-1]
    f = False
    for i in range(n-2):
        for j in range(i+1, n-1):
            for k in range(j+1, n):
                if a[k]+a[j] > a[i]:
                    ans = [a[k], a[j], a[i]]
                    f = True
                    break
            if f:
                break
        if f:
            break

    print " ".join(map(str, ans))
