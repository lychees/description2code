t = int(input())
while t:
    t -= 1
    n, m = map(int, raw_input().split())
    w = []
    w = map(int, raw_input().split())
    my = w[0]
    w.sort()
    w = w[::-1]
    gr = [0]*m
    c = 0
    for i in w:
        gr[c] += 1
        c += 1
        if i == my:
            # print i,c
            break
        if c == m:
            c = 0

    print(gr[c-1]-1)
