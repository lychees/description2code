#include <bits/stdc++.h>
using namespace std;

long long int x[1003],y[1003];
long long area2=0;
int main()
{
	int n,i;
	cin>>n;
	for(i=0;i<n;i++)
		cin>>x[i]>>y[i];
	x[n]=x[0],y[n]=y[0];
	for(i=1;i<=n;i++)
		area2+=(x[i]-x[i-1])*(y[i]+y[i-1]);
	assert(area2>0);
	area2+=2;
	for(i=1;i<=n;i++)
		area2-=__gcd(abs(x[i]-x[i-1]),abs(y[i]-y[i-1]));
	assert(area2>=0 && area2%2==0);
	cout<<area2/2;
    return 0;
}
