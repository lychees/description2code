'''
# Read input from stdin and provide input before running code

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
for tc in xrange(input()):
	c,j,n = map(int,raw_input().split())
	a = map(int,raw_input().split())
	a.sort()
	t=0
	for i in range(n):
		if a[i]<=c and a[i]<=j:
			c-=a[i]
			j-=a[i]
			t+=1
		else:
			break
	if t==n:
		print 'Tie'
		continue
	if c<a[i] and j>=a[i]:
		print 'Jhool Won'
	elif j<a[i] and c>=a[i]:
		print 'Chef Won'
	else:
		print 'Tie'
