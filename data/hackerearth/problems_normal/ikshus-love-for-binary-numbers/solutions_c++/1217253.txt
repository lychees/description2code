#include<bits/stdc++.h>
using namespace std;
#define ll long long

ll memo[100][100];

ll rec(int n,int k,int pos=0,int val=0)
{
    if(memo[pos][val]>-1)
        return memo[pos][val];
    if(n-pos<k-val)
        memo[pos][val]=0;
    else if(n-pos==k-val)
        memo[pos][val]=1;
    else if(val==k)
        memo[pos][val]=((1ll)<<(n-pos));
    else
        memo[pos][val]=rec(n,k,pos+1,0)+rec(n,k,pos+1,val+1);
    return memo[pos][val];
}

int main()
{
    memset(memo,-1,sizeof memo);
    int n,k;
    ll num,den,gcd;
    cin>>n>>k;
    num=rec(n,k);
    den=((1ll)<<n);
    gcd=__gcd(num,den);
    cout<<(num/gcd)<<"/"<<(den/gcd);
    return 0;
}
