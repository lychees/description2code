#include <iostream>
#include <stdio.h>
#include <map>
#include <vector>
#include <memory.h>

using namespace std;

string s[101000];
int n;
int d[100100];
bool used[100100];

void dfs(int v)
{
    if(used[v]) return;
    used[v] = 1;
    
    for(int i = 0 ; i < n ; ++i)
        if(s[v][i] == 'Y')
        {
            if(!used[i])
                dfs(i);
            
            d[v] += d[i];
        }
}

int main()
{
    ios::sync_with_stdio(0);
    cin >> n;
    while(cin >> n)
    {
        for(int i = 0 ; i < n ; ++i)
            cin >> s[i];
        memset(d, 0, sizeof d);
        memset(used, 0, sizeof used);
        
        for(int i = 0 ; i < n ; ++i)
        {
            bool fl = 1;
            for(int j = 0 ; j < n ; ++j)
                if(s[i][j] == 'Y')
                    fl = 0;
            
            if(fl)
                d[i] = 1;
        }
        for(int i = 0 ; i < n ; ++i)
            dfs(i);
        
        int ans = 0;
        for(int i = 0 ; i < n ; ++i)
            ans += d[i];
        cout << ans << "\n";
    }
    return 0;
}
