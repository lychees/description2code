def getIndex(char):
    d = {
        "abc": 2,
        "def": 3,
        "ghi": 4,
        "jkl": 5,
        "mno": 6,
        "pqrs": 7,
        "tuv": 8,
        "wxyz": 9}
    for key, value in d.items():
        if char in key:
            return value


t = input()
while t > 0:
    t -= 1
    output = ''
    text = raw_input()
    for char in text:
        index = getIndex(char)
        output += str(index)
    print output
