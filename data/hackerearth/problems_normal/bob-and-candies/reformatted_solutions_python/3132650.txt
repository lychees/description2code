import itertools
n, x = raw_input().split()
l = [int(i) for i in raw_input().split()]
l1 = []
for i in range(0, int(n)):
    l1.append(i)

x = int(x)
e = -1


def check():
    for i in range(1, int(n)+1):
        for j in itertools.combinations(l1, i):
            sum = 0
            for k in j:
                sum = sum+l[k]
            if sum == x:
                for k in j:
                    print k+1,
                return
    print e
    return


check()
