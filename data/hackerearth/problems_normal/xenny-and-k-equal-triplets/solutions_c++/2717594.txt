#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int i,num,k,count=0,n;
    	long long int x,y,X;
    	scanf("%d%d",&n,&k);
    	for(i=0;i<n;i++)
    	{
    		scanf("%d",&num);
    		if(num==k)
    			count++;
    	}    	
		x=count;
		x=x*(count-1);
		x=x*(count-2);
		x=x/6;
		y=n;
		y=y*(n-1);
		y=y*(n-2);
		y=y/6;
		X=__gcd(x,y);
		x=x/X;
		y=y/X;
		cout<<x<<"/"<<y<<endl;	
    }
    return 0;
}
