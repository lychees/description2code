test_case=int(raw_input())
lower_vowel=['a','e','i','o','u']
capital_vowel=['A','E','I','O','U']
while test_case>0:
    print_vowel=''
    word=raw_input()
    for letter in word:
        if letter in lower_vowel or letter in capital_vowel:
            print_vowel+=letter
    if len(print_vowel)>0:
        print print_vowel
    elif len(print_vowel)==0:
        print "No"
    test_case-=1
