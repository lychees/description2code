		long long int countt = 0;
		scanf("%s", a);
		int siz = strlen(a);
		int jjj = 0;
		for (int i = 0; i < siz; i++){
			jjj++;
			if (a[i] == 'a' || a[i] == 'z'){

			}
		}
		printf("%lld\n", countt);
	}
	return 0;
}
