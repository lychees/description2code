#include <bits/stdc++.h>
using namespace std;

const int MOD=1e9+7;
int N=100005;
long long fac[100005],ifac[100005];
long long powermod(long long a,long long n){
    long long ans=1;
    while(n){
        if(n%2==1){
            ans=ans*a;
            ans%=MOD;
        }
        a=a*a;
        a%=MOD;
        n/=2;
    }
    return ans;
}
void pre(){
    int i;
    fac[0]=1;
    for(int i=1;i<N;i++){
        fac[i]=(i*fac[i-1])%MOD;
        //cout<<fac[i]<<endl;
    }
    //long long h=fac[N-1];
    ifac[N-1]=powermod(fac[N-1],MOD-2);
    //long long j=fac[100];
    //cout<<"\nn-1"<<ifac[N-1]<<endl;
    //cout<<j<<endl;
    for(int i=N-2;i>=0;i--){
        ifac[i]=((i+1)*ifac[i+1])%MOD;
    }


}
long long ncr(int a,int b){
    long long ans=fac[a];
    //cout<<"insi"<<ans<<endl;
    ans*=ifac[b];
    //cout<<"ifac  "<<ifac[b];
    ans%=MOD;
    //cout<<"insi23"<<ans<<endl;
    ans*=ifac[a-b];
    ans%=MOD;
    //cout<<"insi23last"<<ans<<endl;
    return ans;
}
int main()
{
    int t;long long a[100005];
    scanf("%d",&t);int n,k;
    pre();
    //cout<<ncr(4,3);
    //cout<<"\n"<<ncr(6,4)<<endl;
    while(t--){
        scanf("%d%d",&n,&k);
        for(int i=0;i<n;i++){
            cin>>a[i];
        }
        sort(a,a+n);long long sum=0;
        for(int i=0;i<n-k+1;i++){
            sum+=(ncr(n-i-1,k-1)*a[i]);
            sum%=MOD;
            //cout<<n-i-1<<" "<<k<<endl;
        }
        cout<<sum<<endl;


    }
    return 0;
}
