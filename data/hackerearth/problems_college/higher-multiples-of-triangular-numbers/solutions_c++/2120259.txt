vector <ll> p;

int dzen(ll x)
{
	y = (ll)1;
	sq = sqrt(x) + (ll)10;
	for (int i = 0; i < p.size() && p[i] <= sq; i++)
	{
		if (x % p[i] == 0)
		{
			z = (ll)1;
			while (x % p[i] == 0)
			{
				z++;
				x /= p[i];
			}
			y *= z;
		}
	}
	if (x != 1)
		y *= (ll)2;
	return y;
}

		for (ll j = i * i; j <= (ll)1000000; j += i)
		{
			t[j] = 1;
		}
	}
	s = (ll)0;
	fi(1, 1000000)
	{
		s = (ll)s + (ll)i;
		x = dzen(s);
		x = min(x - 1, (ll)1000);
		while (x > 0 && a[x] == 0)
		{
			a[x] = s;
			x--;
		}
		if (a[1000] != 0)
			break;
