#include <stdio.h>

int main()
{
    //printf("Hello World!\n");
    long long int t,n,m,sum,sum1,a,b,c;
    scanf("%lld",&t);
    while(t--)
    {
    	scanf("%lld %lld %lld %lld %lld",&n,&m,&a,&b,&c);
    	sum=n*a+m*b;
    	if(n>m)
    	{
    		sum1=(n-m)*a+c*2*m;
    	}
    	else
    	{
    		sum1=(m-n)*b+c*2*n;
    	}
    	if(sum>sum1)
    	printf("%lld\n",sum);
    	else
    	printf("%lld\n",sum1);
    	
    }
    return 0;
}
