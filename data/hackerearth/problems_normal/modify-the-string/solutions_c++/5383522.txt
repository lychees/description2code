#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    string s;
    cin>>s;
    for(int i=0;i<s.length();i++)
    {
    	int d=s.at(i);
    	if(d>96&&d<=123)
    	{
    	cout<<char(d-32);
    	}
    	else if(d>=65&&d<=90)
    	{
    	cout<<char(d+32);
    	}
    }
       return 0;
}
