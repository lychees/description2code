

#include<bits/stdc++.h>
using namespace std;

int main(void){

	int t;
	cin>>t;
	while(t--){
	
		char str[1000];
		cin>>str;
		
		int arr[26]={0};
		int string_length = strlen(str);
		for( int i=0; i<string_length ; i++)
			arr[str[i] - 65 ]++;
		for( int i=0; i<26; i++ ){
			int cnt = arr[i];
			if( cnt>= 1){
				for( int j=0; j<cnt; j++ )
					cout<<(char)(i+65);
			}
		}
		cout<<endl;
	}
}
