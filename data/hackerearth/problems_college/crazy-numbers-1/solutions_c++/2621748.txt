#include <iostream>
#define MOD 1000000007
using namespace std;
long long int dp[1000010][11];
int main()
{
   int t,n;
   long long int ans,ans1;
   cin>>t;
   while(t--)
   {
   	dp[2][0]=1;
   	dp[2][1]=1;
   	dp[2][2]=2;
   	dp[2][3]=2;
   	dp[2][4]=2;
   	dp[2][5]=2;
   	dp[2][6]=2;
   	dp[2][7]=2;
   	dp[2][8]=2;
   	dp[2][9]=1;
   	cin>>n;
   	if(n==1)
   	{
   		cout<<"10"<<endl;
   		continue;
   	}
   	if(n==2)
   	{
   	    cout<<"17"<<endl;
   		continue;
   	}
   	ans=17;ans1=0;
   	for(int i=3;i<=n;i++)
   	{
   	  for(int j=0;j<=9;j++)
   	  {
   	  	if(j==0)
   	  	{
   	  		dp[i][j]=dp[i-1][1];
   	  	}
   	  	else if(j==9)
   	  	{
   	  		dp[i][j]=dp[i-1][8];
   	  	}
   	  	else
   	  	{
   	  		dp[i][j]=dp[i-1][j-1]+dp[i-1][j+1];
   	  	}
   	  }
   	  for(int j=0;j<=9;j++)	dp[i][j]=(dp[i][j]%1000000007);;
   	}
   	ans=0;
    for(int j=0;j<=9;j++)
    {
    	ans+=dp[n][j];
    	ans=(ans%1000000007);
    }
   	cout<<ans<<endl;
   }
    return 0;
}
