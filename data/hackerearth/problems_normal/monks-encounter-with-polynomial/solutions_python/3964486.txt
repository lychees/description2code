def check(a,b,c,k,x):
    return (a*(x**2))+b*x+c>=k
for t in range(int(raw_input())):
    a,b,c,k=map(int,raw_input().split())
    l=0
    r=10**5
    ans=0
    while l<=r:
        mid=r-((r-l))/2
        if check(a,b,c,k,mid):
            if check(a,b,c,k,mid-1):
                r=mid-1
            else:
                ans=mid
                break
        else:
            l=mid+1
    print ans
