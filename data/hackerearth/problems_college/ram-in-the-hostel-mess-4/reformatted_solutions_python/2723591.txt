from itertools import product
t = int(input())
for i in range(t):
    m = int(input())
    ways = 2**m
    k = list(product([0, 1], repeat=m))
    count = 0
    for i in range(len(k)):
        for j in range(m-1):
            # print(k[i])
            if k[i][j] == 1 and k[i][j+1] == 1:
                break
        else:
            count = count+1
    print(int(count)**2)
