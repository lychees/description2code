
def op(a,b,c):
	if b=="*":
		return a*c
	if b=="+":
		return a+c
	if b=="-":
		return a-c

def solve():
	s=raw_input()
	if len(s)==1:
		return s
		
	l=len(s) -2
	cur = int(s[-1])
	
	while l>=0:
		cur = op(cur,s[l],int(s[l-1]))
		l = l-2
		
	return cur
	





def main():
	for _ in xrange(input()):
		print solve()
		
main()
