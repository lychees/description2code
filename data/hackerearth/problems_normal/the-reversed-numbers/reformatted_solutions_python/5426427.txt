t = input()
for i in range(0, t):
    num1, num2 = map(str, raw_input().split())
    num1 = num1[::-1]
    num2 = num2[::-1]
    num1 = int(num1)
    num2 = int(num2)
    add = num1+num2
    while True:
        if add % 10 == 0:
            add = add/10
        else:
            break
    add = str(add)
    print add[::-1]
