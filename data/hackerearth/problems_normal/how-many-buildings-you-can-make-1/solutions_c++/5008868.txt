

    #include <iostream>
    using namespace std;
     
    int main()
    {
    	int n,a,i,j,flag=0;
       cin>>n;
       while(n--)
       {
       	cin>>a;
       	for(i=1;i*i<=a;i++)
       	{
       		for(j=i;j<=a;j++)
       		{
       			if(i*j<=a)
       			{
       				flag++;
       			}
       		}
       	}
       	cout<<flag<<endl;
       	flag=0;
       }
        return 0;
    }
