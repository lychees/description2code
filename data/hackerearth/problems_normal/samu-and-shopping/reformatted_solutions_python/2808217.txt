for _ in xrange(int(raw_input())):
    n = int(raw_input())
    market = [map(int, raw_input().split()) for _ in xrange(n)]
    dp = [[0, 0, 0] for _ in xrange(n)]
    dp[0] = market[0]
    for i in xrange(1, n):
        dp[i][0] = min(dp[i-1][1], dp[i-1][2]) + market[i][0]
        dp[i][1] = min(dp[i-1][0], dp[i-1][2]) + market[i][1]
        dp[i][2] = min(dp[i-1][0], dp[i-1][1]) + market[i][2]
    print min(dp[n-1][0], dp[n-1][1], dp[n-1][2])
