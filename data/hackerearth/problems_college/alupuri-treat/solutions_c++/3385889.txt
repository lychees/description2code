#include <iostream>
#include <cmath>
using namespace std;

int main()
{
	int d1=0,d2=0;
	int i,j,n;
	int a[500][500],ans;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	for(j=0;j<n;j++)
    	{
    		cin>>a[i][j];
    	}
    }
   for(i=0;i<n;i++)
   {
   
   	d1=d1+a[i][i];
 
   }
   
   for(i=0,j=n-1;i<n,j>=0;i++,j--)
   {
   	
   		d2=d2+a[i][j];
   	
   }
   
   ans= abs(d1-d2);
 
   cout<<ans;
}
