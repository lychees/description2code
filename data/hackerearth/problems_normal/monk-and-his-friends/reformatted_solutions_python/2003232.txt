import sys
T = int(sys.stdin.readline())
for i in xrange(T):
    line = sys.stdin.readline().split()
    N = int(line[0])
    M = int(line[1])
    d = {}
    newline = sys.stdin.readline().split()
    for i in xrange(N):
        # print newline[i]
        d[newline[i]] = True
    for i in xrange(M):
        # print newline[i+N]
        if newline[i+N] in d:
            print "YES"
        else:
            print "NO"
        d[newline[i+N]] = True
