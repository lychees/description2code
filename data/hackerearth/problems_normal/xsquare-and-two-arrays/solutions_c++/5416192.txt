#include <iostream>
using namespace std;


int main()
{
   long n=0, m=0;


	cin >> n >> m;
	long *a = new long[n];
	long *b = new long[m];
	for (long i = 1; i <= n; i++)
		cin >> a[i];
	for (long j = 1; j <= n; j++)
		cin >> b[j];

	for (long i = 0; i<m; i++)
	{
		long type, startIndex, endIndex;
		cin >> type >> startIndex >> endIndex;
		long sum = 0;
		if (type == 1)
		{
			int flag = 0;
			for (long x = startIndex; x <= endIndex; x++)
			{
				if (flag == 0)
				{
					sum += a[x];
					flag = 1;
				}
				else
				{
					sum += b[x];
					flag = 0;
				}
			}
		}
		else
		{
			int flag = 0;
			for (long x = startIndex; x <= endIndex; x++)
			{
				if (flag == 0)
				{
					sum += b[x];
					flag = 1;
				}
				else
				{
					sum += a[x];
					flag = 0;
				}
			}

		}
		cout << sum << endl;
	}
    return 0;
}
