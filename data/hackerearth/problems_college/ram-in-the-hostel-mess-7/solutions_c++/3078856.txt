#include <bits/stdc++.h>

using namespace std;

long long int fib[100];

void fibnc ()
{
	fib[1]=2;
	fib[2]=3;
	for (int i=3;i<100;++i)
		fib[i]=fib[i-1]+fib[i-2];
}

int main()
{
    int t,n;
    fibnc();
    scanf ("%d",&t);
    while (t--)
    {
    	long long int ans=0;
    	scanf ("%d",&n);
    	ans=fib[n]*fib[n];
    	printf ("%lld\n",ans);
    }
    return 0;
}
