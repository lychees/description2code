#include<cstdio>

int dp[1000001];


int MIN(int a, int b){
    if(a < b) return a;
    return b;
}

void init(){
    dp[0] = 0;
    dp[1] = 1000000000;
    dp[2] = 1;
    dp[3] = 1;
    for(int i=4;i<=1000000;i++){
        dp[i] = MIN(1+dp[i-2], 1+dp[i-3]);
        if(i >= 7){
            dp[i] = MIN(dp[i], 1+dp[i-7]);
        }
        else if(i >= 5){
            dp[i] = MIN(dp[i], 1+dp[i-5]);
        }
    }
}

int main(){
    init();
    int t;
    scanf("%d", &t);
    while(t--){
        int x;
        scanf("%d", &x);
        if(x == 1) printf("-1\n");
        else printf("%d\n", dp[x]);
    }
}
