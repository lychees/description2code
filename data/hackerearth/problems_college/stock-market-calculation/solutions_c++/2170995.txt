#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    for(cin>>t;t;t--){
    	int n;
    	cin>>n;
    	vector<pair<int,int> >v(n);
    	int a[n];
    	for(int i=0;i<n;i++){
    		cin>>a[i];
    		v[i].first=a[i];
    		v[i].second=i;
    	}
    	sort(v.rbegin(),v.rend());
    	int vis[n];
    	memset(vis,0,sizeof(vis));
    	long long pro=0;
    	for(int i=0;i<n;i++){
    		int ind=v[i].second;
    		if(!vis[ind]){
    			vis[ind]=1;
    			for(int i=ind-1;i>=0;i--){
    				if(vis[i])break;
    				pro+=a[ind]-a[i];
    				vis[i]=1;
    			}
    		}
    	}
    	cout<<pro<<endl;
    }
    return 0;
}
