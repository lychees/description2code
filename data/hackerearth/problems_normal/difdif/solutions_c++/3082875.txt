#include <iostream>
#include <set>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    string st;
    cin >> st;
    set<long long> ans;
    long long int k=st.size(),i,j;
    for(i=0;i<k;i++){
    	long long int s=0;
    	for(j=i;j<k;j++){
    		s=(s*173+st[j]);
			ans.insert(s);    		
    	}
    }
    cout << ans.size();
    return 0;
}
