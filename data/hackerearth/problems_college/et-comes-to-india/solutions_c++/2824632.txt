char find(long long int n,long long int k,bool stat)
{
    char c;
    if(n==1)
    {
        if(stat)
            return 'G';
        else
            return 'C';
    }
    if(n==2)
    {
        if(k==3)
        {
            if(stat)
                return 'C';
            else
                return 'G';
        }
        else
        {
            if(stat)
                return 'G';
            else
                return 'C';
        }
    }
    if(k==(cnt[n]/2)+1)
    {
        if(stat)
            return 'G';
        else
            return 'C';
    }
    else
    {
        if(k<=cnt[n]/2)
        {
            return find(n-1,k,stat);
        }
        else
            return find(n-1,cnt[n-1]-(k-(cnt[n]/2+1))+1,!stat);
    }
}
