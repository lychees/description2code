#include<bits/stdc++.h>
using namespace std;
#define mp make_pair
const int N = 10000;
int n, tests;
int d[N];
set<pair<int, int> > s;
set<pair<int, int> >::iterator it;
int dist[N];
int main(){
	ios_base::sync_with_stdio(0);
	cin.tie(0);
	cin >> n >> tests;
	for (int i = 0; i < n; i++)
	{
		cin >> d[i];
	}
	sort (d , d + n ) ; 
	dist[0] = 0 ;
	s.insert (mp(0, 0)) ; 
	for (int i = 1 ;i < d[0] ; i++){
		dist[i] = 2e9 + 1e6 ;
		s.insert ( mp( dist[i] , i )) ; 
	}
    while (!s.empty ()){
    	  int v =   s.begin()->second;
    	  s.erase (s.begin()) ;
    	  if (dist[v] > 1e9 + 1e6)      continue ;
		  for (int i = 0 ; i < n ;i++){
		  	    int val = dist[v] + d[i] ;
		  	    int rem = val % d[0] ;
		  	    if (dist[rem]  >val){
		  	    	   s.erase (s.find (mp(dist[rem] ,rem))) ;
		  	    	   dist[rem] = val ;
		  	    	   s.insert (mp(dist[rem] ,rem)) ; 
				  }
		  } 
	}
	int x ; 
	while (tests--){
		   cin >> x ;
		   int p = x % d[0] ;
		   if (dist[p] <= x )   puts("YES") ;
		   else                 puts("NO") ; 
	}
	return 0 ; 
}
