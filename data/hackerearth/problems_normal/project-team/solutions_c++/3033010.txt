#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
	int n,t,minn,maxx;;
	int a[100000];
	cin>>t;
	while(t--){
		cin>>n;
		for(int i=0;i<n;i++)
			cin>>a[i];
		if(n<3){
			cout<<"0"<<endl;
			continue;
		}
		sort(a,(a+n));
		minn=a[n-1]+a[n-2];
		maxx=a[0]+a[1];
		for(int i=0;i<n/2;i++){
			minn=min(minn,a[i]+a[n-i-1]);
			maxx=max(maxx,a[i]+a[n-i-1]);
		}
		cout<<abs(minn-maxx)<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
