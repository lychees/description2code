#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int n,p,a[100001];
    	int count=0,sum=0;
    	
    	cin>>n>>p;
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    		if(a[i]==p)count++;
    		}
        if(count==0)cout<<"NO"<<" "<<0<<endl;
        else
        {
           for(int j=0;j<n;j++)
           {
               if(a[j]>p)sum++;
           }
           cout<<"YES"<<" "<<sum<<endl;
    	}
    }
}
