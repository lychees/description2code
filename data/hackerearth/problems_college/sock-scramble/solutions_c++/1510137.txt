#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
              int n;
              cin>>n;
              int x=0,a;
              for(int i=0;i<n;i++)
              {
                      cin>>a;
                      x^=a;
              }
              cout<<x<<"\n";
    }
    return 0;
}
