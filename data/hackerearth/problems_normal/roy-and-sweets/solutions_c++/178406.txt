#include <iostream>
#include <cstdio>
#include <map>
#include <utility>
#include <algorithm>
#include <limits>

using namespace std;

#define MAXM 1000001

int m[MAXM];
int level[MAXM];

int main()
{
	int n,g;
	long long int R = 0;

	int tlevel,quant,max=0;
	scanf("%d", &n);

	for(int i=0;i<n;i++)
	{
		scanf("%d %d",&tlevel, &quant);
		//sweets[i] = tlevel;
		if(tlevel>max)
			max = tlevel;
		level[tlevel] = quant;
	}

	scanf("%d",&g);
	
	for(int i=0;i<g;i++)
		scanf("%d",&m[i]);
	
	
	int maxq = level[max];
	for(int i=max;i>=1;i--)
	{
		if(level[i]>maxq)
			maxq = level[i];
		level[i] = maxq;
	}

	for(int i=0;i<g;i++)
	{
		int val = level[m[i]];
		int an = val/m[i];
		R = R + (100*an);
	}

	printf("%lld\n",R);

	return 0;
}
