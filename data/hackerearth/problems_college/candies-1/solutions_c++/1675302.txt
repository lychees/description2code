#include <iostream>
using namespace std;

int main()
{
   long int t,s,i,c,sum;
   cin>>t;
   while(t--)
   {
   	cin>>s;
   	i=s;
   	sum=0;
   	while(i--)
    {
    	cin>>c;
    	sum+=c;
    }
    if(sum%s==0)
    {
    cout<<"YES"<<endl;
    cout<<sum/s<<" "<<0<<endl;
    }
    else
    {
    cout<<"NO"<<endl;
    cout<<sum/s<<" "<<sum%s<<endl;
    }
   }
}
