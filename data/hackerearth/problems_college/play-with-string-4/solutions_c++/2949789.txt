#include <iostream>
#include<cstring>
#include<cmath>
using namespace std;

int main()
{
     int t,i,j,flag;
    cin>>t;
    char s[10003];
    while(t--)
        {cin>>s;flag=1;
        i=1;j=strlen(s)-1;
         while(i<=j&&i<strlen(s)&&j>0)
         {
             if(abs(s[i]-s[i-1])!=abs(s[j]-s[j-1]))
                 {
                 flag=0;
                 break;
             }
             i++;j--;
         }
         if(flag)
             cout<<"Funny\n";
         else
             cout<<"Not Funny\n";
    }
    return 0;
}
