for _ in range(input()):
    n, k = map(int, raw_input().split())
    stone = list(map(int, raw_input().split()))
    time = list(map(int, raw_input().split()))
    finalarray = []
    i = 0
    for item in stone:
        temp = (k/item)*time[i]
        finalarray.append(temp)
        i = i+1
    final = sorted(finalarray, reverse=True)
    print final[0]
