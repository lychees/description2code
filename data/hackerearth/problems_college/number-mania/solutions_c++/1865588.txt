#include <iostream>
using namespace std;

void fact(int n)
 {
  int a[200];a[0]=1;
  int t=0,c=1,x;
   for(int i=1;i<=n;++i){
   	 for(int j=0;j<c;++j){
   	 	x=a[j]*i+t;
   	 	a[j]=x%10;
   	 	t=x/10;
   	 }
   	 
   	 while(t){
   	 	a[c]=t%10;
   	 	t/=10;
   	 	c++;
   	 }
   }
   
   for(int i=c-1;i>-1;--i)
     printf("%d",a[i]);
     printf("\n");
 }

int main()
{
  ios::sync_with_stdio(0); 
   int n;
   cin>>n;
   
    int f=1;
   for(int i=2;i<=n;++i)
    {f=1;
    for(int j=2;j<i;++j)
    { if(i%j==0)
        f=0;
    }
    if(f){ fact(i);}
    }
    return 0;
}
