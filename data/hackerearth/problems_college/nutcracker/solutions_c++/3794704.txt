#include <iostream>
using namespace std;
int a[10005],i;
int main()
{
    //cout << "Hello World!" << endl;
    int t,n;
    cin>>t;
    for(i=1;i<=4;i++)
    	a[i]=1;
    a[6]=1;
    a[8]=1;
    for(i=9;i<=10000;i++)
    {
    	if(a[i-1]==1&&a[i-4]==1)
    	a[i]=0;
    	else
    	a[i]=1;
    }
    while(t--)
    {
    	cin>>n;
    	if(a[n]==1)
    	cout<<"Yes\n";
    	else
    	cout<<"No\n";
    }
    return 0;
}
