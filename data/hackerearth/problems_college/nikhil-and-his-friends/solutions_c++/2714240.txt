#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll arr[1000001];
int subArraySum(ll arr[], ll n, ll sum)
{
    ll curr_sum = arr[0], start = 0, i;
    for (i = 1; i <= n; i++)
    {
        while (curr_sum > sum && start < i-1)
        {
            curr_sum = curr_sum - arr[start];
            start++;
        }
        if (curr_sum == sum)
        {
            return 1;
        }
        if (i < n)
          curr_sum = curr_sum + arr[i];
    }
    return 0;
}

int main()
{
	ll t,n,x;
	int i,temp;
	cin>>t;
	while(t--)
	{
		scanf("%lld %lld",&n,&x);
		for(i=0;i<n;i++)
		{
			scanf("%lld",&arr[i]);
		}
		temp=subArraySum(arr,n,x);
		if(temp)
		printf("YES\n");
		else
		printf("NO\n");
	}
    return 0;
}
