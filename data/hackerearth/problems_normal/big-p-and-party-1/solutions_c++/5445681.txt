#include<bits/stdc++.h>
using namespace std;
int i,j;
int main()
{
    int a,b;
    cin>>a>>b;
    vector<int> v[a];
    queue<int> q;
    bool visit[a];
    memset(visit,false,sizeof visit);
    while(b--)
    {
        int x,y;
        cin>>x>>y;
        v[x].push_back(y);
        v[y].push_back(x);
    }
    int level[a];
    memset(level,-1,sizeof level);
    q.push(0);
    visit[0]=true;
    level[0]=0;
    while(!q.empty())
    {
        int p=q.front();
        visit[p]=true;
        q.pop();
        for(i=0;i<v[p].size();i++)
        {
            if(!visit[v[p][i]])
            {
                visit[v[p][i]]=true;
                q.push(v[p][i]);
                level[v[p][i]]=level[p]+1;
            }
        }
    }
    for(i=1;i<a;i++)
    {
        cout<<level[i]<<endl;
    }
    return 0;
}
