#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,m,arr[1009],dp[1009][1009]={},pre[1009]={},maxi=INT_MIN;
    const int INF=1e9;
    cin >> n >> m;
    for(int i=1; i<=m; i++)
    {
        cin >> arr[i];
        maxi=max(maxi,arr[i]);
        pre[i]=pre[i-1]+arr[i];
    }
    if(n>=m)
    {
        cout << maxi << endl;
        return 0;
    }
    for(int i=1; i<=n; i++)
        for(int j=1; j<=m; j++)
        {//cout<<i<<" "<<j<<" "<<dp[i][j]<<endl;
            if(i==1)
            {
                dp[i][j]=pre[j];
                continue;
            }
            dp[i][j]=INF;
            for(int k=i-1; k<=(j-1); k++)
                dp[i][j]=min(dp[i][j],max(pre[j]-pre[k],dp[i-1][k]));
             //  cout<<i<<" "<<j<<" "<<dp[i][j]<<endl;
        }
/*    for(int i=1; i<=n; i++,printf("\n"))
        for(int j=1; j<=m; j++)
            printf("%d ",dp[i][j]);*/
   
    cout <<dp[n][m] << endl;
    return 0;
}
