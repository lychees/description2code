#include <iostream>
using namespace std;

int main()
{
    int t,n,i;
    long long x,sum;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	sum=0;
    	for(i=0;i<n;i++)
    	{  cin>>x;
    	   sum+=x;
    	}
    	if(sum==(sum/n)*n)
    	  cout<<"YES\n"<<sum/n<<" 0"<<endl;
    	else
    	 cout<<"NO\n"<<sum/n<<" "<<sum-((sum/n)*n)<<endl;
    	
    	
    }
    return 0;
}
