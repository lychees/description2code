#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int x,y,places;
    	cin>>x>>y>>places;
    	int rem=(x*10)%y;
    	int dec=(x*10)/y;
    	for(int i=0;i<places-1;i++){	
    		dec=(rem*10)/y;
    		rem=(rem*10)%y;
    	}
    	cout<<dec<<endl;
    }
    return 0;
}
