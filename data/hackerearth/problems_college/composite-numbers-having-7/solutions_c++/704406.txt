#include <bits/stdc++.h>
using namespace std;

int main() {
	int m,n;
	vector<int> nums;
	vector<int>::iterator it,it2;
	bitset<1000001> visit;
	visit.set(0);
	visit.set(1);
	for(int i=2;i<=1000;++i)
		if(!visit.test(i))
			for(int j=2*i;j<visit.size();j+=i)
				visit.set(j);
	for(int i=1;i<=1000000;++i)
	{
		if(!visit.test(i))
			continue;
		int n=i;
		while(n)
		{
			if(n%10==7)
			{
				nums.push_back(i);
				break;
			}
			n/=10;
		}
	}
	int t;
	cin >> t;
	while(t--)
	{
		cin >> n >> m;
		it = lower_bound(nums.begin(),nums.end(),n);
		it2 = upper_bound(nums.begin(),nums.end(),m);
		if(it2==it)
			cout << -1 << endl;
		else
			cout << it2-it << endl;
	}
	return 0;
}
