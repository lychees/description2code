#include<iostream>
#include<cstdio>
#include<cmath>
#include<cstring>


using namespace std;


int main(){
	int t;
	scanf("%d",&t);	
		while(t--){
			int n;
 	  		int i=1;
   	 	  	scanf("%d",&n);
			int res=0;
			bool h[10];
			memset(h,false,sizeof(h));
			int k=n;

		while(k>0){
				h[k%10]=true;
				k/=10;
				}

 		  while(i < sqrt(n))
 		   {
 		       if(n%i==0) {
						int p=i;
						while(p>0){
							if(h[p%10]==true){
									res++;
									break;
									}
									p/=10;
								}
									
 		          		int a=n/i;
					while(a>0){
							if(h[a%10]==true){
							res++;
							break;
						}
						a/=10;
	 			       }
				}
 	       i++;
 		   }
   
if(i*i == n){
		int m=i;
		while(m>0){
					if(h[m%10]==true){
							res++;
							break;
					}
			m/=10;
 	       } 
}


cout<<res<<endl;
}
return  0;
}
