#include<bits/stdc++.h>
using namespace std;
#define gett(t) scanf("%d",&t)
#define put(t) printf("%d\n",t)
#define gets(s) cin>>s
int main()
{
    int t,i,p;
    string s1;
    gett(t);
    while(t--)
    {
        p=0;
        gets(s1);
        for(i=0;s1[i]!='\0';i++)
        {
            if(s1[i]=='B')
            {
                if(i>=1)
                    s1[i-1]='Q';
                if(i>1)
                    s1[i-2]='Q';
            }
            if(s1[i]=='S')
            {
                if(i>0)
                    s1[i-1]='Q';
                if(s1[i+1]=='-')
                    s1[i+1]='Q';
            }
        }
        for(i=0;s1[i]!='\0';i++)
        {
            if(s1[i]=='-')
                p++;
        }
        put(p);
    }
    return 0;
}
