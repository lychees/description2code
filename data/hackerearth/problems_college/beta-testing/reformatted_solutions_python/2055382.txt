
from collections import defaultdict
import operator

t = int(raw_input())
user = list(map(str, raw_input().split()))
userdict = defaultdict(int)

for i in user:
    userdict[i] += 1

# sorted(userdict, key=userdict.get, reverse=True)
sor = sorted(userdict.items(), key=operator.itemgetter(1), reverse=True)

for k, v in sor:
    print(k + ' ' + str(v))
