#include<iostream>
#include<string.h>
using namespace std;
int main()
{
    string s[505];
    int i,j,n1,n2,ans=0,c=0,n,maxm=0,temp;
    cin>>n;
    for(i=1;i<=n;i++) {
      cin>>s[i];
      if(s[i].size()>maxm) maxm=s[i].size();
    }
    for(i=1;i<=n;i++) {
      if(s[i].size()<maxm) {
         while(s[i].size()<maxm) s[i]='0'+s[i];
      }
    }
    for(i=maxm-1;i>=0;i--) {
      for(j=1;j<=n;j++) {
        temp=s[j][i]-'0';
        ans+=temp;
//        cout<<ans<<" ";
        while(ans>=10) {
          temp=ans/10;
          ans=ans%10;
          ans+=temp;
        }
      }
//      cout<<ans<<"\n";
    }
    cout<<ans<<"\n";
//  system("pause");
  return 0;
}
