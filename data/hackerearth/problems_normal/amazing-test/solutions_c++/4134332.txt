#include<bits/stdc++.h>
using namespace std ;
int a[102] , dp[2][100002] ;
int main () {
    int t,n ,x ,sum;
    scanf ( "%d" , &t ) ;
    while ( t-- ) {
        sum = 0 ;
        int dp[2][100002] ;
        scanf ( "%d %d" , &n , &x ) ;
        for ( int i = 1 ; i <= n ; i++ ) {
            scanf ( "%d" , &a[i] ) ;
            sum += a[i] ;
        }
        if ( sum <= x ) {
            cout << "YES\n" ;
            continue ;
        }
        for ( int i = 0 ; i <= x ; i++ ) {
            dp[0][i] = 0 ;
        }
        for ( int i = 1 ; i <= n; i++ ) {
            for ( int j = 0 ; j <= x ; j++ ) {
                dp[i%2][j] = dp[(i-1)%2][j] ;
                if ( j >= a[i] )
                    dp[i%2][j] = max ( dp[i%2][j] , dp[(i-1)%2][j-a[i]]+a[i] ) ;
            }
        }
        if ( sum - dp[n%2][x] <= x )
            cout << "YES\n" ;
        else
            cout << "NO\n" ;
    }
    return 0 ;
}
