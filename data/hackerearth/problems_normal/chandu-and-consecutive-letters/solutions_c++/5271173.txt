#include <iostream>
using namespace std;

int main()
{	int t; cin>>t;
	while(t--){
	    string s,ans=""; cin>>s;
	    ans+=s[0];
	    //cout<<ans<<endl;
	    int j=0;
	    for(int i=1;i<s.length();i++){
	    	if(s[i]!=ans[j]){
	    		ans+=s[i];
	    		j++;
	    	} 
	    }
	    cout<<ans<<endl;
	}
    return 0;
}
