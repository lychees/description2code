#include <iostream>
#include <algorithm>
#include <stdio.h>
#include <string.h>
 
using namespace std;
long long int n,k,a[100000],dpp[100000];
long long int dp(int i){
	if(i>=n) return 0;
	if(dpp[i]==-1)
	dpp[i]=max(a[i]+dp(i+k+1),dp(i+1));
	return dpp[i];
	
}
int main()
{
	int t;
	cin>>t;
	while(t--){
		cin>>n>>k;
		for(int i=0;i<n;i++) cin>>a[i];
		memset(dpp,-1,sizeof(dpp));
		cout<<dp(0)<<endl;
		//for(int i=0;i<n;i++) cout<<dpp[i]<<" ";
		//cout<<endl;
	
	}
 
 
    return 0;
}
