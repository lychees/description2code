#include <cstdio>
#define sl(x) scanf("%lld",&x)
#define sd(x) scanf("%d",&x)
#define ss(x) scanf("%s",x)
#define pl(x) printf("%lld ", x)
#define nl printf("\n")
#define Max 1000001
#define FOR(i,j,k) for(auto i=j;i<k;i++)
#define rep(i,n) FOR(i,0,n)
#define LL long long int
void create_xormat(LL n){
	LL max=1 , len=n-1;
	while(len>0){
		max*=2;
		len/=2;
	}
	max--;
	pl(max); pl(n-((n-1)^max)); nl;
}
int main(void){
	int t;
	sd(t);
	LL n;
	while(t--){
		sl(n);
		create_xormat(n);
	}
}
