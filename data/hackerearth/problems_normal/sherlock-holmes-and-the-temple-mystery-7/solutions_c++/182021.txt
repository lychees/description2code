#include <iostream>
using namespace std;

int main()
{
    
    int t;
    
    cin >> t;
    
    while( t-- )
    {
    	int n , i=0;
    	int sum = 0 , noOfOdd = 0;
    	
		cin >> n;
		int a[n];
    	
		while( i < n )
    	{
    		cin >> a[i];
    		
    		sum = sum + a[i];
    		
    		if( a[i]%2 != 0 )
    			noOfOdd++;
    			
    		i++;
    	}
    	
    	if( sum%2 == 0 && noOfOdd%2 == 0 )	//Even...
    			cout << "YES" << endl;
    	else
    		cout << "NO" << endl;
    		
    		
    }
    
    return 0;
}
