#include<bits/stdc++.h>
using namespace std;
const int MAXN = 100005;
int nxt[MAXN][26];
#define rep(i,j,k) for(int i=j;i<k;i++)
#define ll long long


int main(){
	string str1,str2;
	cin>>str1;
	int len1=str1.length();
	rep(i,0,26){
		nxt[len1][i]=len1;
	}

	for(int i=len1-1;i>=0;--i){
		rep(j,0,26){
			nxt[i][j]=nxt[i+1][j];
		}
		nxt[i][str1[i]-'a']=i;
	}

	/*rep(i,0,len1){
		rep(j,0,26){
			cout<<nxt[i][j]<<" ";
		}
		cout<<endl;
	}*/

	int q;
	cin>>q;
	while(q--){
		cin>>str2;
		int pos=0;
		bool flag=true;
		int n=str2.length();
		for(int i=0;i<n && flag;i++){
			if(nxt[pos][str2[i]-'a'] == len1)
				flag=false;
			else
				pos = nxt[pos][str2[i]-'a']+1;
		}

		flag ? (cout<<"Yes\n"):(cout<<"No\n");
	}


	return 0;
}
