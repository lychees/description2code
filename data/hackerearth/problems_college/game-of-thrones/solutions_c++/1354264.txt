#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
    int t,a[100005],i,n,ctr;
    scanf("%d",&t);
    while(t--){
    	ctr=0;
    	scanf("%d",&n);
    	for(i=0;i<n;i++)
    	scanf("%lld",&a[i]);
    	sort(&a[0],&a[n]);
    	for(i=1;i<n;i++){
    		if(a[i]==a[i-1])
    		ctr++;
    	}
    	printf("%d\n",ctr);
    }
    return 0;
}
