    #include <bits/stdc++.h>
    using namespace std;
    int f(int num)
    {
    int n, digit, rev = 0;
    n = num;
    do
    {
    digit = num%10;
    rev = (rev*10) + digit;
    num = num/10;
    }while (num!=0);
    if (n== rev)
    return 1;
    else
    return 0;
    }
     
     
     
     
     
    int main() {

    double sqrtLimit = sqrt(limit);
    long sum = 0;

    for (int i = 1; i <= sqrtLimit; i++) {
    int number = i*i;
    for (int j = i + 1; j <= sqrtLimit; j++) {
    number += j * j;
    if (number > limit) break;
    vector<int>::iterator it;
