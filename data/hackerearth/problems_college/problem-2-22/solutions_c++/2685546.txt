#include <bits/stdc++.h>
using namespace std;
	int r,xc,yc;
int lieinside(int x,int y)
{
	x-=xc;y-=yc;
	float pos=sqrt((x*x)+(y*y));
	if(pos<=r) return 1;
	return 0;
	
}
int main()
{
    int t,i;
    scanf("%d",&t);
    while(t--)
    {
    
    	bool ans = true;
    	scanf("%d%d%d",&r,&xc,&yc);
    	int a[4][2];
    	for(i=0;i<4;i++) {scanf("%d%d",&a[i][0],&a[i][1]);
    	if(!lieinside(a[i][0],a[i][1])) ans=false;}
    	if(ans) printf("Yes\n");
    	else printf("No\n");
    }
    return 0;
}
