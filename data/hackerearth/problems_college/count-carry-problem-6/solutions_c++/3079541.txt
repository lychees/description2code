#include <iostream>
using namespace std;

int main()
{
	int t, x, y, c ;
	long int m, n ;
	cin >> t ;
	while(t-->0){
		c=0 ;
		cin >> m >> n ;
		while(m || n){
			x= m%10 ;
			y= n%10 ;
			if(x+y>9)
			  c++ ;
			m/=10 ;
			n/=10 ;
		}
		if (c>1)
		   cout << c << " carry operations" << endl ;
		else if (c==1)
		   cout << c << " carry operation" << endl ;
		else   
		   cout << "No carry operation" << endl ;
	}
    return 0;
}
