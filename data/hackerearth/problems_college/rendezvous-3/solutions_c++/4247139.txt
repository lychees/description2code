#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int n,m,par[100010];

vector < pair < ll int ,ll int > > v;
vector < pair < ll int,ll int > > arr;

ll int root(ll int x)
{
    return (par[x] < 0) ? x : (par[x]=root(par[x]));
}
void merg(ll int x,ll int y)
{
    if((x=root(x))==(y=root(y)))
        return ;
    if(par[y] < par[x]) // x contains less tools since negative everything
        swap(x,y);
    par[x]+=par[y];
    par[y]=x;

}
int main()
{
    ll int i,j,k,l,x,y,cst=0,c,vx=0;

    //freopen("input.txt","r",stdin);
    cin>>n>>m;

    for(i=0;i<m;i++)
    {
        cin>>x>>y>>c;

        v.push_back({x,y});
        arr.push_back({c,i});
    }

    sort(arr.begin(),arr.end());

    for(i=1;i<=n;i++)
        par[i]=-1;

    i=0;
    while(vx!=n-1)
    {
        l=arr[i].second;
        x=v[l].first;
        y=v[l].second;

        //cout<<"s";
        if(root(x)==root(y))
            {i++;}
        else
        {
            //cout<<"merger";
            merg(x,y);
            //cout<<"merged";
            cst+=arr[i].first;
            i++;vx++;
        }

    }
    cout<<cst<<endl;
}
