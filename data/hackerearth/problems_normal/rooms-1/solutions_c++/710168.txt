#include <iostream>
#include<cstdio>
#include<algorithm>
using namespace std;

long int a[100010],b[100010],c[200010];

int main()
{
long int t,i,j,m,n,p,x,y,v;

scanf("%ld",&t);

while(t--)
{
scanf("%ld",&n);

p=0;
for(i=0;i<n;i++)
{
scanf("%ld",&a[i]);
c[p++]=a[i];
}

for(i=0;i<n;i++)
{
scanf("%ld",&b[i]);
b[i]+=a[i];
c[p++]=b[i];
}

sort(c,c+p);
sort(a,a+n);
sort(b,b+n);

x=0;
y=0;

m=0;
v=0;

for(i=0;i<p;i++)
{
if(c[i]==a[x])
{
v++;
x++;
}

if(c[i]==b[y])
{
v--;
y++;
}

if(v>m)
m=v;

}

printf("%ld\n",m);
}

return 0;
}
