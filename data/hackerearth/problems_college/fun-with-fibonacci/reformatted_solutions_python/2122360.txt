inpcount = int(raw_input())
outlst = []
for i in range(0, inpcount):
    maxin = int(raw_input())
    var1 = 0
    var2 = 1
    sumout = 0
    j = maxin
    while j <= maxin:
        j = var1 + var2
        if j % 2 == 0 and j <= maxin:
            sumout += j
        var1 = var2
        var2 = j
    outlst.append(sumout)
for i in outlst:
    print i
