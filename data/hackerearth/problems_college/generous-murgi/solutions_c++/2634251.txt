#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	long long int i,x,y,temp;
    	cin >> x >> y;
    	if(x==y)
    	{
    		cout << "-1" << endl;
    	}
    	else if(x>y)
    	{
	    	temp = x-y;
	    	long long int count = 0;
	    	for(i=1;i<=sqrt(temp);i++)
	    	{
	    		if(temp%i==0)
	    		{
	    			if(i>y)
	    		  count++;
	    		
	    		if((temp/i)>y && i!=temp/i)
	    		count++;
	    		}
	    	}
	    	cout << count << endl;
    	}
    	else
    	 cout << "0" << endl;
    }
    return 0;
}
