#include <iostream>
using namespace std;

int main()
{
    long long int n, x = 1, count;
    int t;
    cin>>t;
    
    for ( int i =0;i < t;i++ ) {
    	cin>>n;
    	
    	count = 0;
    	x = 1;
    	
    	while ( n ) {
    		while ( x <= n ) {
    			x = x*2;
    		}
    		count++;
    		n = n - ( x / 2 );
    		x = 1;
    	}
    	
    	cout<<count<<endl;
    	
    }
    return 0;
}
