#include <iostream>
using namespace std;
int main(){
    int t;
    cin>>t;
    for(int j=0;j<t;j++){
    int a,b;
    cin>>a>>b;
    int k,count=0;
    int revers,remainder;
    for(int i=a;i<=b;++i){
        k=i;
        revers =0,remainder=0;
        while(k!=0){
            remainder = k%10;
            revers = ((revers*10) + remainder);
            k/=10;
        }
        if(i==revers){
            count++;
        }

    }
    cout<<count<<endl;
    }
    return 0;
}
