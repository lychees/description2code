#include<iostream>
#include<string>

using namespace std;

#define INDEX(x) (x - 'a')
int w[26];
int main()
{
	int T, n, tw;
	cin>>T;
	string s;
	char c;
	while(T--)
	{
	 	cin>>s>>n;
		for (int i = 0; i < 26; i++) w[i] = i+1;
		for (int i = 0; i < n; i++)
		{
			cin>>c;
			w[INDEX(c)] = 0;
		}
		tw = 0;
		for (auto it = s.begin(); it != s.end(); ++it)
			tw += w[INDEX(*it)];
	
		cout<<tw<<endl;
	}
	return 0;
}
