#include <iostream>
using namespace std;

int main()
{
    int T;
    cin>>T;
    while(T--)
    {
    	string A,B;
    	cin>>A>>B;
    	int l1=A.length(),l2=B.length();
    	int cnt=0;
    	if(l1==l2)
    	{
    		
    		for(int i=0;i<l1;i++)
    		{
    			int d=int(A[i]-'0');
    			int d1=int(B[i]-'0');
    			if(d+d1>=10)cnt++;
    		}
    		
    	}
    	else if(l1>l2)
    	{
    		
    		for(int i=l1-l2,j=0;j<l2;i++,j++)
    		{
    			int d=int(A[i]-'0');
    			int d1=int(B[j]-'0');
    			if(d+d1>=10)cnt++;
    		}
    		
    	}
    	else
    	{
    	
    		for(int i=0,j=l2-l1;i<l1;i++,j++)
    		{
    			int d=int(A[i]-'0');
    			int d1=int(B[j]-'0');
    			if(d+d1>=10)cnt++;
    		}
    		
    	}
    	if(cnt==0)
    		cout<<"No carry operation\n";
    	else if(cnt==1)
    		cout<<"1 carry operation\n";
    	else cout<<cnt<<" carry operations\n";
    	
    }
    return 0;
}
