#include <iostream>
using namespace std;

int main()
{
    int Test;
    cin >> Test;
    if(Test < 0 || Test > 5)return 0;
    for(int i=0; i < Test; ++i){
    	int NbItem = 0;
    	cin >> NbItem;
    	if(NbItem < 0 || NbItem > 5000000)return 0;
    	int pRes = 0;
    	for(int j = 0; j < NbItem; ++j){
    		int pI;
    		cin >> pI;
    		pRes += pI;
    	}
    	double result = 0.0;
    	result = double((double)pRes / (double)NbItem);
    	cout.precision(8);
    	cout << fixed << result << endl;
    }
    return 0;
}
