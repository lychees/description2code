n = int(input())
for i in range(0, n):
    c = 0
    c1 = 0
    a1, a2 = map(int, raw_input().split())
    while (a1 != 0 or a2 != 0):
        t1 = a1 % 10
        t2 = a2 % 10
        if t1+t2+c1 >= 10:
            c1 = 1
            c = c + 1
        else:
            c1 = 0
        a1 = a1/10
        a2 = a2/10
    if c == 0:
        print("No carry operation")
    elif c == 1:
        print("1 carry operation")
    else:
        print c, "carry operations"
