#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    int x,t,a,b;
    cin >>t;
    for(int i =0;i<t;i++)
    {
    	cin>>x;
    	if(x==1)
    	{
    		a=1;
    		b=0;
    		cout<<a<<" "<<b<<endl;
    	}
    	else if(x>1&&x<=100)
    	{
    		a=x-1;
    		b=1;
    		cout<<a<<" "<<b<<endl;
    	}
   
  }
    
    return 0;
}
