t = int(raw_input())
while t > 0:
    t -= 1
    n = int(raw_input())
    c = 0
    for i in range(1, n+1):
        for i2 in range(i, n+1):
            x = i ^ i2
            if x <= n and i != i2:
                # print x
                c += 1
    print c
