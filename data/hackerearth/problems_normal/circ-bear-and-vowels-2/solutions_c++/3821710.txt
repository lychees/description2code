int main(){
    std::ios_base::sync_with_stdio(false);
    int tc; cin >> tc;
    map<char, int> mmap;
    mmap['a'] = 1;
    mmap['e'] = 1;
    mmap['i'] = 1;
    mmap['o'] = 1;
    mmap['u'] = 1;
    mmap['y'] = 1;
    while (tc--){
        string str; cin >> str;
        int vowels = 0;
        int seq = 0;
        bool flag = false;
        for (char c : str){
            if (mmap.count(c)) {
                vowels++;
                seq = 0;
            }
            else {
                vowels--;
                seq++;
            }
            if (seq == 3) {
                flag = true;
            }
        }
        if (vowels < 0 || flag) cout << "hard" << endl;
        else cout << "easy" << endl;
    }

    return 0;
}
