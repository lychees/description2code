n = int(raw_input())

array = [int] * n

array = [int(x) for x in raw_input().split(' ')]

if n == 2:
    print array[0] * array[1]
elif n == 3:
    print max(array[0]*array[1], array[1]*array[2], array[0]*array[2])
else:
    number1max = max(array)
    array.remove(number1max)
    number2max = max(array)
    productmax = number1max * number2max

    number1min = min(array)
    array.remove(number1min)
    number2min = min(array)
    productmin = number1min * number2min

    print max(productmax, productmin)
