using namespace std;

long a[101];

int main() {
  int n, c, d, vr, s = 0;
  cin >> n >> c;
  for (int i = 1; i <= n; i++) {
    cin >> a[i] >> b[i];
  }
  for (int x = 1; x < 10000; x++) {
    vr = 0;
    for (int i = 1; i <= n; i++) {
      d = a[i] * x;
      vr += 1 + (a[i] * x) / b[i];
    }
    if (vr == c)
      s++;
  }
  cout << s;
}
