#include <bits/stdc++.h>
using namespace std;
long long int A[100005];
int main()
{
	// freopen("input.txt", "r", stdin);
	// ios_base::sync_with_stdio(0);
	int t;
	scanf("%d", &t);
	while(t--)
	{
		long long int n,m,z;
		scanf("%lld %lld %lld", &n, &m, &z);
		for (int i = 0; i < n; ++i)
			scanf("%lld", &A[i]);
		A[n] = 0;
		for (int i = 1; i <= n; ++i)
		{
			if(i%2)
				A[n] = (A[n] + A[n-i])%m;
			else
				A[n] = (A[n] - A[n-i] + m)%m;
		}
		long long int val = A[z%(n+1)];
		if(n%2)
			printf("%lld\n", val);
		else
		{
			long long int lala = z/(n+1);
			if(lala%2)
				printf("%lld\n", (m-val)%m);
			else
				printf("%lld\n", val);
		}
	}
	return 0;
}
