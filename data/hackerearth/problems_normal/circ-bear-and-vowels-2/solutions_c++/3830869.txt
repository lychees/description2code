#include <iostream>
using namespace std;
int main()
{
 int t,i,j,k,c,u;
 char s[50];
 cin>>t;
 if(t>50)
 return 0;
 else
 {
   while(t)
   {
   cin>>s;
   i=0;j=0;k=0;c=0;u=0;
    while(s[i])
	{
	if(s[i]=='a'||s[i]=='e'||s[i]=='i'||s[i]=='o'||s[i]=='u'||s[i]=='y'||s[i]=='A'||s[i]=='E'||s[i]=='I'||s[i]=='O'||s[i]=='U'||s[i]=='Y')
    {
     j++;
     c=0;
    }
    else
    {
     k++;
     c++;
     if(c==3)
     u=1;
   }
    i++;
	}
    if(k-j>0||u==1)
    cout<<"hard\n";
    else
    cout<<"easy\n";
    t--;
    }
 }
    return 0;
}
