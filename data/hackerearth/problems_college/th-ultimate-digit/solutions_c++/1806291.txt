#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
	unsigned long long b;
	int base,p,t;
	string a;
	scanf("%d",&t);
	while(t--)
	{
		cin>>a>>b;
		base=a[a.length()-1]-'0',p=b%4;
		if(b==0)
		{
			printf("1\n");continue;
		}
		if(base==5 || base==6)
		{
			printf("%d\n",base);continue;
		}
		
		if(p==1)
			printf("%d\n",base);
		else if(p==2)
			printf("%d\n",(base*base)%10);
		else if(p==3)
			printf("%d\n",(base*base*base)%10);
		else
			printf("%d\n",(base*base*base*base)%10);
	}
	return 0;
}
