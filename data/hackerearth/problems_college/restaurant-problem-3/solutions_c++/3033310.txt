#include <iostream>
#include <string>
#define ll long long int
using namespace std;

int main()
{
    string s,org;
    ll num,flag=0;
    cin>>org;
    while(org.size()>0){
        for(ll i=0;i<org.size();++i){
            s=org.substr(0,i+1);
            if(s[0]=='0'&&s.size()>1) continue;
            num=stoll(s);
            if(num%8==0) ++flag;
        }
        org.erase(0,1);
    }
    cout<<flag<<endl;
}
