#include <cstdio>
#include<cstring>
using namespace std;

int main()
{
    char str[101];
    int num,i,j,count=0,temp;
    scanf("%d",&num);
    int arr[num][26];
    for(i=0;i<num;i++)
    {
    	for(j=0;j<26;j++)
    	{
    		arr[i][j]=0;
    	}
    }
    for(i=0;i<num;i++)
    {
    		scanf("%s",str);
    		for(j=0;j<strlen(str);j++)
    		{
    			arr[i][str[j]-'a']++;
    		}
    }
    for(i=0;i<26;i++)
    {
    	temp=0;
    	for(j=0;j<num;j++)
    	{
    		if(arr[j][i]) temp++;
    	}
    	if(temp==num) count++;
    }
    printf("%d\n",count);
    return 0;
}
