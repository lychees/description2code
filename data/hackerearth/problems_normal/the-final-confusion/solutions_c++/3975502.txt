#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll dp[32][32];
ll f(int full,int half)
{
	if(dp[full][half])
	return dp[full][half];
	if(full==0)
	{
	   return dp[full][half]=1;
	}
	if(half==0)
	{
		return dp[full][half]=f(full-1,half+1);
	}
	return dp[full][half]=f(full-1,half+1)+f(full,half-1);
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		printf("%lld\n",f(n,0));
	}
	return 0;
}
