#include <stdio.h>
#include <string>
#include <iostream>
#include <math.h>
#include <set>
#include <map>
#include <vector>
#include <queue>
using namespace std;
#define MAX 1002
int n;
int m[MAX][MAX];
bool kt[MAX][MAX];

int vx[4];
int vy[4];

int find(int x, int y){
	int count=0;
	kt[x][y] = false;
	for (int i = 0; i < 4; ++i){
		int xc = x + vx[i];
		int yc = y + vy[i];
		if (m[xc][yc] == 1 && kt[xc][yc])
		count += find(xc, yc);
	}
	return count + 1;
}

void bfs(){
	int max = 0;
	for (int i = 0; i < n; ++i){
		for (int j = 0; j < n; ++j){
			if (m[i][j] == 1 && kt[i][j]){
				int k = find(i, j);
				if (k>max) max = k;
			}
		}
	}
	cout << max;
}

int main(){
	//freopen("INPUT.INP", "r", stdin);
	vx[0] = 0;
	vx[1] = 0;
	vx[2] = 1;
	vx[3] = -1;
	vy[0] = 1;
	vy[1] = -1;
	vy[2] = 0;
	vy[3] = 0;
	cin >> n;
	for (int i = 0; i < n; ++i)
	for (int j = 0; j < n; ++j){
		cin >> m[i][j];
		kt[i][j] = true;
	}
	bfs();
	return 0;
}
