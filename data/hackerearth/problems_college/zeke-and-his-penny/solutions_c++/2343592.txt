typedef long long int ll;

int main(){

    scanf("%d", &t);
    while(t--){
        scanf("%lld %lld %lld", &v[0], &v[1], &v[2]);
        sort(v, v + 3);
        ll ans = v[0] * (v[1] - 1) % mod;
        ans = ans * (v[2] - 2) % mod;
        printf("%lld\n", ans);
    }
    return 0;
}
