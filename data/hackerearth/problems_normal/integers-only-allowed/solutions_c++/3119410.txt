#include <iostream>
#include <bits/stdc++.h>
using namespace std;

long gcd(long a,long b){
	
	if(b==0)
		return a;
	return gcd(b,a%b);
	
}

int main()
{
	long int n,a[1010],b[1010],area=0,interior=0,x,y,i;
	scanf("%ld",&n);
	for(i=0;i<n;i++)
		scanf("%ld%ld",&a[i],&b[i]);
		
	a[n]=a[0];
	b[n]=b[0];
	
	for(i=0;i<n;i++)
		area+= -a[i]*b[i+1] + b[i]*a[i+1];
	
	for(i=0;i<n;i++){
		x = a[i+1] > a[i] ? a[i+1]-a[i] : a[i]-a[i+1];
		y = b[i+1] > b[i] ? b[i+1]-b[i] : b[i]-b[i+1];
		
		if(x > y)
			interior+= gcd(x,y);
		else
			interior+= gcd(y,x);
		
	}
	
//	cout<<area<<" "<<interior<<" ";
	
	printf("%ld", (area-interior+2)/2);
	
    return 0;
}
