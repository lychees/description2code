#include <bits/stdc++.h>
using namespace std;
unordered_map<char,int> umap;
int main()
{
	ios_base::sync_with_stdio(false);
    int n;
    string str;
    cin>>n;
    for(int i=0;i<n;i++)
    {
    	cin>>str;
    	for(int i=0;i<str.size();i++)
    	{
    		umap[str[i]]++;
    	}
    	int flag=0;
    	unordered_map<char,int> :: iterator itr;
    	for(itr=umap.begin();itr!=umap.end();itr++)
    	{
    		if(itr->second>1)
    		{
    			flag=1;
    			break;
    		}
    	}
    	if(flag)cout<<"Yes\n";else cout<<"No\n";
    	umap.clear();
    }
    return 0;
}
