#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string a,b;
		vector<char> c;
		cin>>a>>b;
		sort(b.begin(),b.end());
		int j=0;
		int i;
		for(i=0;i<a.size();i++)
		{
			if(j<b.size())
			{
				if((int)a.at(i)>(int)b.at(j))
				{
					c.push_back(b.at(j));
					j++;
				}
				else 
					c.push_back(a.at(i));
			}
			else
				c.push_back(a.at(i));
		}
		for(i=0;i<a.size();i++)
			cout<<c.at(i);
		cout<<endl;
		c.clear();
	}
}
