#include <bits/stdc++.h>
using namespace std;
#define MAX 1000001

int dp[MAX+2];


int main()
{
    long long t,m,n,c_R,c_K,i;
    string s;
    scanf("%lld",&t);
    while(t-->0){
    	scanf("%lld%lld",&m,&n);
    	cin>>s;
    	memset(dp,0,sizeof(dp));
    	c_R=0;
    	int j=0;
    	dp[0]=1;
    	for(i=0;i<s.size();i++){
    		if(s[i]=='R')j++;
    		if(j>=m)c_R+=dp[j-m];
    		dp[j]++;
    	}
    	c_K=0;
    	j=0;
    	memset(dp,0,sizeof(dp));
    	dp[0]=1;
    	for(i=0;i<s.size();i++){
    		if(s[i]=='K')j++;
    		if(j>=n)c_K+=dp[j-n];
    		dp[j]++;
    	}
    	printf("%lld %lld\n",c_R,c_K);
    }
    return 0;
}
