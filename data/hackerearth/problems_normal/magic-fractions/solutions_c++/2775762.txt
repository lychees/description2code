int gcd(int a, int b) {
	if(b == 0)
		return a;
	return gcd(b, a%b);
}

ll power(ll a, ll b) {
	ll res = 1;
	while(b) {
		if(b & 1) {
			res = (res * a);
		}
		a = (a*a);
		b >>= 1;
	}
	return res;
}

void solve() {
	int n;
	cin>>n;
	if(n == 0 || n == 1) {
		cout<<"0\n";
		return;
	}
	vector<vector<int> > f(n+1);
	for(int i=2;i<=n;i++) {
		int num = i;
		for(int j=2;j<=num;j++) {
			if(num%j == 0) {
				f[i].push_back(j);
				while(num%j == 0)
					num /= j;
			}
		}
	}

	ll sum = 0;
	for(int i=2;i<=n;i++) {
        set<int> x;
        for(int j = 2;j<=i;j++)
            x.insert(f[j].begin(), f[j].end());
		sum += power(2, x.size()-1);
	}
	cout<<sum<<"\n";
}
