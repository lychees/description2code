#include <bits/stdc++.h>
using namespace std;
#define MAX 1000006
vector<int>v;

void fi(int n)
{
    int P = n;
    for(int i = 2; i * i <= P; i++) {
        int count = 0;
        while(n % i == 0) {
            n /= i;
            count++;
        }
        if(count != 0) {
            v.push_back(count);
        }
    }
    if(n > 1) {
        v.push_back(1);
    }

    int ans = v[0];
    for(int i = 1; i < v.size(); i++) {
        ans = ans ^ v[i];
    }
    if(ans == 0) {
        printf("Pappu Bhujia\n");
    }
    else {
        printf("Printf{}\n");
    }
    return;
}

int main()
{
    int t;
    cin >> t;
    while(t--) {
        int n;
        cin >> n;
        v.clear();
        fi(n);
    }
    return 0;
}
