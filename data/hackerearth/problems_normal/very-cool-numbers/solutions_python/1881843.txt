import sys

def solve(n):
	c=0
	s=bin(n)[2:]
	for i in xrange(0,len(s)-2):
		if(s[i:i+3]=='101'):
			c+=1
	return c

dp=[]

for i in xrange(0,5):
	l=[]
	for j in xrange(0,101):
		l.append(0);
	dp.append(l)
for i in xrange(5,100001):
	l=[]
	temp = solve(i)
	for j in xrange(0,101):
		if temp>=j:
			t1 = dp[i-1][j]+1
			l.append(t1)
		else:	
			l.append(dp[i-1][j])
	dp.append(l)
