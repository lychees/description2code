#include <iostream>
using namespace std;

int main()
{
    int a,b,cnt=0;
    cin>>a>>b;
    for(int i=0;a!=0;i++)
    {
    		int temp=a%10;
    		a=a/10;
    		int temp2=b%10;
    		b=b/10;
    		if(temp!=temp2)cnt++;
    }
    if(cnt==1||cnt==0)cout<<"wins\n";
    else cout<<"loses\n";
    return 0;
}
