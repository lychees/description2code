#define FORD(i,a,b) for (long long i=a;i>=b; i--)

#define ri(n)({\
    int neg=0;\
    n=0;\
    char ch;\
    for(ch=getchar(); ch<'0' || ch>'9'; ch=getchar()) if (ch=='-') neg=1-neg;\
    n=ch-48;\
    for(ch=getchar(); ch>='0' && ch<='9'; ch=getchar()) n=(n<<3)+(n<<1)+ch-48;\
    if (neg) n=-n;\
})
 
typedef long long ll;

typedef pair<ll,ll> LL;
const ll INF=1000000000+7;
const double esp=1e-13;
const double pi=3.141592653589;
 
int k,n,a[20],d1,b1[2000],b2[2000],dem,ans,t1,t2;
 
int bit(int x, int y){
    return x>>(y-1)&1;
}
 
int main(){
 //   freopen("count-numbers-7.inp", "r", stdin);
 //   freopen("count-numbers-7.out", "w", stdout);
    ri(k); ri(n);
    FORE(i,1,k) ri(a[i]);
    sort(a+1,a+k+1);
    k=unique(a+1,a+k+1)-a-1;
    FORE(i,1,(1<<k)-1){
        ans=1;
        dem=0;
        FORE(j,1,k)
        if (bit(i,j)){
            dem++;
            if (ans<=INF/a[j]) ans*=a[j];
            else goto loop;
        }
        b1[++d1]=ans;
        b2[d1]=dem;
        loop:;
    }
 
     while (n--){
        cin >> t1 >> t2;
        ans=0;
        FORE(i,1,d1)
        if (b2[i]%2) ans+=t2/b1[i]-(t1-1)/b1[i];
        else ans-=(t2/b1[i]-(t1-1)/b1[i]);
        cout << ans << "\n";
    }
}
