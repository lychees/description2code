#include<cstdio>
#include<cstring>
#define MAXN 1000000

bool isPrime[MAXN+11];
long long ans[MAXN+11];

void init(){
    memset(isPrime,true,sizeof(isPrime));
     ans[1]=0;
    for(int i=2;i<=MAXN;i++){
            ans[i]=ans[i-1];
        if(isPrime[i]){
            ans[i]+=i;
            for(int j=i+i;j<=MAXN;j+=i){
                isPrime[j]=false;
            }
        }
    }
    return ;
}
int main(){
    init();
    int cs;
    scanf("%d",&cs);
    while(cs--){
        int l,r;
        scanf("%d%d",&l,&r);
        printf("%lld\n",ans[r]-ans[l-1]);
    }
    return 0;
}
