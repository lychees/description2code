#include<bits/stdc++.h>
using namespace std;
int main()
{
	int k,m,a,j=0,i,n;
	char ch;
	multiset<int> m1;
	cin>>k>>m;
	for(i=1;i<=k;i++)
	{
		cin>>n;
		for(j=0;j<n;j++)
		{
			cin>>a;
			m1.insert(a);
		}
	}
	for(i=1;i<=m;i++)
	{
		if(m1.find(i)!=m1.end())
		{
			j=0;
		}
		else
		{
			j=1;
			break;
		}
	}
	if(j==1)
	{
		cout<<"NO";
	}
	else
	{
		cout<<"YES";
	}
}
