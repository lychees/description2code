#include<bits/stdc++.h>

using namespace std;


set<int> ans;
void subsetSum(int a[],int n,int max_sum){


    bool M[n+1][max_sum+1];

    for(int i=0;i<=n;i++)
        M[i][0] = true;

    for(int j=1;j<=max_sum;j++)
        M[0][j] = false;

    for(int i=1;i<=n;i++){

        for(int j=1;j<=max_sum;j++){

            if(a[i-1]>j)
                M[i][j] = M[i-1][j];
            else{
                M[i][j] = M[i-1][j] || M[i-1][j-a[i-1]];
            }
            if(M[i][j])
                ans.insert(j);

        }
    }


}

int main(){

    int t;
    cin>>t;
    while(t--){
        int n;
        cin>>n;
        int arr[n];
        int max_sum=0,x;
        for(int i=0;i<n;i++){
                cin>>x;
                arr[i] = x;
                max_sum += x;
        }

        ans.clear();
        ans.insert(0);
        subsetSum(arr,n,max_sum);
        cout<<ans.size()<<endl;

    }

    return 0;
}
