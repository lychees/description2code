#include <iostream>
using namespace std;

int main()
{
	int r,p[100],P=0,d=100;
	cin>>r;
	for(int i=0;i<r;i++)
	{
		cin>>p[i];
		P+=p[i];
		if(p[i]%2 != 0 && d > p[i])
		d = p[i];
	}
	if(P%2!=0)
	cout<<P<<endl;
	else
	{
		if(d != 100)
		{
			P = P - d;
			cout<<P<<endl;
		}
		else
		cout<<":("<<endl;
	}
	
    return 0;
}
