ll x_pow_y_modm(ll x, ll y, ll m){
    if(y==0)
        return 1;
    ll temp;
    temp = x_pow_y_modm(x,y/2,m);
    if(y%2==0)
        return (temp*temp)%m;
    else
        return (x*((temp*temp)%m))%m;
}
 
ll modularInverse(ll a, ll m) {
    return x_pow_y_modm(a,m-2,m);
}
 
ll gcd(ll a,ll b)
{
  if(a==0)
    return b;
  if(b==0)
    return a;
  return gcd(b,a%b);
}
