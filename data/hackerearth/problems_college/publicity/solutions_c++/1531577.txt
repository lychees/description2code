#include<iostream>
#include<vector>
#include<cstring>
using namespace std;
vector< int > ad[100001];
int visit[100001];
void dfs(int u )
{
     visit[u] = 1;
     for(int i = 0 ; i < ad[u].size() ; ++i )
     {
           if(visit[ad[u][i]] == 0)
              dfs(ad[u][i]);
     }
}
int main()
{
      ios_base::sync_with_stdio(false);
       int t;
       cin >> t;
       while(t--)
       {

             int n , e;
             cin >> n >> e;
             for(int i = 0 ; i <= n ; ++i )
                 ad[i].clear();
             while(e--)
             {
                   int u , v;
                   cin >> u >> v;
                   ad[u].push_back(v);
                   ad[v].push_back(u);
             }
             memset(visit,0,sizeof visit);

               int ans = 0;
             for(int i = 0 ; i < n ; ++i )
             {
                  if(visit[i] == 0 )
                  {
                      ans++;
                        dfs(i);
                  }
             }
                cout << ans << endl;

       }
         return 0;
}
