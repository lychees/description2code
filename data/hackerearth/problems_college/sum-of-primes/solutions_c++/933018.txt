#include <bits/stdc++.h>
using namespace std;
#define ll long long int
ll a[1000011];
void sieve()
{
	ll i,j;
	for(i=0;i<=1000005;i++)
		a[i]=i;
	a[0]=0;
	a[1]=0;
	for(i=2;i<=1000005/2;i++)
	{
		if(a[i]!=0)
		{
			for(j=i+i;j<=1000005;j=j+i)
				a[j]=0;
		}
	}
	for(i=1;i<=1000005;i++)
	{
		a[i]=a[i]+a[i-1];
	}
}
ll isprime(ll n)
{
	if(n==0||n==1)
		return 0;
	ll i;
	for(i=2;i<=sqrt(n)+1;i++)
	{
		if(n%i==0)
			return 0;
	}
	return 1;
}
int main()
{
	sieve();
	ll t;
	scanf("%lld",&t);
	while(t--)
	{
	ll m,n;
	scanf("%lld%lld",&m,&n);
	//printf("%lld %lld\n",a[m],a[n]);
	if((a[n]-a[m-1])==0)
		printf("NO\n");
	else if(isprime(a[n]-a[m-1])==1)
		printf("YES\n");
	else
		printf("NO\n");
	}
	return 0;
}
