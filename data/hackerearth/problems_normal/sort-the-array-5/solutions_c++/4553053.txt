#include <bits/stdc++.h>
using namespace std;

int main() {
		int n, k;
	cin >> n >> k;

	vector<vector<int > > v(k);
	for(int i = 0; i < n; ++i) {
		int te;
		cin >> te;

		v[te%k].push_back(te);
	}

	for(int i = k-1; i >= 0; --i) {
		sort(v[i].begin(), v[i].end());
		for(__typeof(v[i].begin()) it = v[i].begin(); it != v[i].end(); it++) 
			cout << *it << " ";
	}

	cout << endl;
	return 0;
}
