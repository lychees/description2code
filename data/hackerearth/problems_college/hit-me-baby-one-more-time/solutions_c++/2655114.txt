#include<bits/stdc++.h>
using namespace std;
int gcd(int a,int b) {
    if(b==0)
        return a;
    printf("%d %d\n",a,b);
    return gcd(b,a%b);
}
int main(void) {
    long long n,i,j,k,sum=0;
    scanf("%lld",&n);
    i=3;
    j=5;
    k=15;
    int cnti=2,cntj=2,cntk=2;
    while(i<n && j<n && k<n) {
        sum+=i;
        sum+=j;
        sum-=k;
        i=3*cnti;
        j=5*cntj;
        k=15*cntk;
        cnti++;
        cntj++;
        cntk++;
    }
    //printf("%lld\n",cnti);
   // printf("%lld\n\n",sum);
    while(i<n) {
        sum+=i;
        i=3*cnti;
        cnti++;
    }
    while(j<n) {
        sum+=j;
        j=5*cntj;
        cntj++;
    }
    printf("%lld\n",sum);
    return 0;
}
