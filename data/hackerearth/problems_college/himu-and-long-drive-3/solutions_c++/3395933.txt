#include <iostream>
using namespace std;
#define ll long long
ll ans(ll l, ll r) {
    ll q = l ^ r, a = 1;
    while(q){
        q /= 2;
        a <<= 1;
    }
    return --a;
}
int main()
{
    //cout << "Hello World!" << endl;
    long long t,n;
    cin>>t;
    while(t--){
    	cin>>n;
    	cout<<ans(1,n)<<endl;
    	
    }
    
    return 0;
}
