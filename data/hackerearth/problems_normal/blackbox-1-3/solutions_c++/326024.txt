#include <iostream>
#include<algorithm>
using namespace std;
#define gcd __gcd
int main()
{
	int t,a,b,c;
	cin>>t;
	while(t--)
	{
		cin>>a>>b>>c;
		if(gcd(a,gcd(b,c))==1){
			if(a*a+b*b==c*c || a*a+c*c==b*b|| b*b+c*c==a*a)
			cout<<"YES\n";
			else cout<<"NO\n";
		}
		else cout<<"NO\n";
		//cout<<"\n";
	}
    return 0;
}
