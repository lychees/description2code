#include <bits/stdc++.h>
using namespace std;
string s;
int main()
{
    getline(cin,s);
    if((int)s[0]>=97&&(int)s[0]<=122)
    		{
    			s[0]=(char)((int)s[0]-32);
    		}
    for(int i=0;i<s.size();i++)
    {
    	if((int)s[i]==32)
    	{
    		if((int)s[i+1]>=97&&(int)s[i+1]<=122)
    		{
    			s[i+1]=(char)((int)s[i+1]-32);
    		}
    	}
    }
    cout<<s<<endl;
    return 0;
}
