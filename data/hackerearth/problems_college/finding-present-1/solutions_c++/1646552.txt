#include <iostream>
using namespace std;

int main()
{
    int p,q,x;
    int a[10001][2] = {};
    for(int i = 0; i<10001; i++) {
    	a[i][0] = 0;
    	a[i][1] = 0;
    }
    cin>>p;
    for(int i = 0; i<p; i++) cin>>x, a[x][0]++;
    cin>>q;
    for(int i = 0; i<q; i++) cin>>x, a[x][1]++;
    for(int i = 0; i<10001; i++){
    	if(a[i][1]-a[i][0]>0) cout<<i<<" ";
    }
    return 0;
}
