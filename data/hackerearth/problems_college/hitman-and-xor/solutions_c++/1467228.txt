#include <iostream>
using namespace std;

int main()
{
    long int tc,n,even,odd,num,i;
    scanf("%ld",&tc);
    while(tc--)
    {
    	even=0,odd=0;
    	scanf("%ld",&n);
    	for(i=0;i<n;i++)
    	{
    		scanf("%ld",&num);
    		if(num%2==0)
    		even++;
    		else
    		odd++;
    	}
    	printf("%ld\n",even*odd);
    }
    return 0;
}
