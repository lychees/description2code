#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int p,e;
		cin>>p>>e;
		float rev=0,ep=p+(e*p/100.0);
		int n=ep;
		while(n!=0)
		{
			int r=n%10;
			rev=rev*10+r;
			n/=10;
		}
		int pr=(p-rev);
		float pc=pr*100.0/p;
		float pl=(pc*ep/100.0);
		cout<<int(pl+ep)<<endl;
	}
	return 0;
}
