#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t, sw = 0;

    scanf("%d", &t);

    while (t--) {
        unsigned long long n, t1;

        scanf("%ulld", &n);

        int c = 0, i;
        for (i = 0; i < 64; i++) {
            t1 = (unsigned long long)1<<(unsigned long long)i;
            if (n & t1) {
                c++;
                //cout << (1<<i) << " ";
            }
        }

        //cout << c << endl;

        if (sw == 0 && (c % 2 == 1)) {
            printf("Sherlock\n");
        }

        else if (sw == 0 && (c % 2 == 0)) {
            printf("Watson\n");
            sw = 1;
        }

        else if (sw == 1 && (c % 2 == 1))
            printf("Watson\n");

        else {
            printf("Sherlock\n");
            sw = 0;
        }
    }

    return 0;
}
