for _ in xrange(input()):
    size, low = input(), 0
    bucket  = map(int, raw_input().split())
    
    mylist = [1]*(size)
    for i in xrange(size):
        for j in xrange(i):
            if bucket[i]>bucket[j]:
                mylist[i] = mylist[j] + 1

        low = max(low, mylist[i])

    print size - low
