#include <bits/stdc++.h>
using namespace std;
char str[105];
int main()
{
    scanf("%s",str);
    int cnt=1,ans=0;
    for(int i=1,len=strlen(str);i<len;i++)
    {
    	if(str[i]==str[i-1])
    	{
    		cnt++;
    	}
    	else
    	{
    		ans+=(cnt*(cnt+1))/2;
    		cnt=1;
    	}
    }
    ans+=((cnt)*(cnt+1))/2;
    printf("%d\n",ans);
    return 0;
}
