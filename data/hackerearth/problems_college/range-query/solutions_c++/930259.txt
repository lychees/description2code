#include <iostream>
using namespace std;

int main()
{
	
	
	long long T,N,i,j,k,Q,count;
	cin>>T;
	long long A[T],B[10000];
	for(i=0;i<T;i++)
	{
		cin>>A[i];
		B[A[i]]=B[A[i]]+1;
		
	}
	cin>>Q;
	while(Q--)
	{	count=0;
		cin>>j>>k>>N;
		if(j==0)
		{
			cout << B[N] << endl;
		}
		else
		{
		for(i=j;i<=k;i++)
		{
			if(A[i]==N)
			count++;
		}
		
	cout << count << endl;	
	}
	}
	
    
    return 0;
}
