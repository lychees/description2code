using namespace std;
int dp[100005],p[30],w[30],n,w1,c;
int countbits(int mask)
{
    int c = 0;
    while (mask)
    {
      mask &= (mask-1) ;
      c++;
    }
    return c;
}
int weightbits(int mask)
{
    int i,ans=0;
    for(i=0;i<n;i++)
    {
        if(mask&(1<<i))
            ans = ans+w[i];
    }
    return ans;
}
int compute()
{
    int cl,wl,i,j,ans=0;
    for(i=0;i<(1<<n);i++)
    {
        cl = countbits(i);
        wl = weightbits(i);
        //cout << cl << " " << wl << endl;
        if((cl>c)||(wl>w1))
            continue;
        for(j=0;j<n;j++)
        {
            //cout << cl << " " << wl << " " << w[j] << endl;
            if(!(i&(1<<j))&&(cl<c)&&((wl+w[j])<=(w1))){//cout << "hi" << endl;
                dp[i|(1<<j)] = max(dp[i|(1<<j)] ,dp[i]+p[j]);}
        }
    }
    for(i=0;i<pow(2,n);i++)
        ans = max(ans,dp[i]);
    return ans;
}
//&&(cl<c)&&((wl+w[j])<=w1)
int main()
{
    int d,d1=0,ans,i;
    ios::sync_with_stdio(false);
    cin >> d;
    while(d--)
    {
        d1++;
        cin >> n;
        for(i=0;i<n;i++)
            cin >> p[i] >> w[i];
        cin >> w1 >> c;
        memset(dp,0,sizeof(dp));
        cout << "For Day #" << d1 << ":" << endl;
        ans= compute();
        if(ans==0)
            ans=-1;
        cout << ans<< endl;;
    }
}
