typedef long long int ll;
const int mx = 2e6 + 10;
using namespace std;
int n;
vector <int> ans[mx];
int main(){
//    ios_base::sync_with_stdio(0);
//    cin.tie(0);
//    freopen("data.in","r",stdin);
    cin >> n;
    for(int i = 1; i <= n; i++){
        int a, b;
        cin >> a >> b;
        ans[b + a].push_back(i);
    }

    for(int i = 0; i < mx; i++){
        if((int)ans[i].size() > 0){
            for(int j = 0; j < (int)ans[i].size(); j++){
                cout << ans[i][j] << " ";
            }
        }
    }
    return 0;
}
