//#include<stdio.h>
#include<iostream>
using namespace std;
int main(){
	long long int t,n,i,j;
	long long int arr[10000],step[10000]={0},x,sum;
	//scanf("%lld",&t);
	cin>>t;
	for(i=0;i<t;i++){
		cin>>n;
		//scanf("%lld",&n);
		sum=0;
		for(j=0;j<n;j++){
			cin>>arr[j];
			//scanf("%lld",&arr[j]);
			sum+=arr[j];
		}
		if(sum%n!=0){
			step[i]=-1;
		}
		else{
			x=sum/n;
			for(j=0;j<n;j++){
				if(arr[j]>=x){
					step[i]+=arr[j]-x;
				}
			}
		}
	}
	for(i=0;i<t;i++){
		cout<<step[i]<<"\n";
		//printf("%lld\n",step[i]);
	}
	return 0;
	
}
