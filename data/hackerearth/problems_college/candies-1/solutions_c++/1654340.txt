#include<bits/stdc++.h>
using namespace std;

typedef long long ll;

int main() {
  int tt;
  scanf("%d",&tt);
  while(tt--) {
    int n;
    scanf("%d",&n);
    ll sum=0;
    for(int i=0;i<n;i++) {
      ll val;
      scanf("%lld",&val);
      sum+=val;
    }
    if(sum%n==0) {
      printf("YES\012%lld 0\012",sum/n);
    } else {
      printf("NO\012%lld %lld\012",sum/n,sum%n);
    }
  }
  return 0;
}
