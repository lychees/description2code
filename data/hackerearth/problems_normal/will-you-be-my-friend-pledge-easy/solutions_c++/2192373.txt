#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define ii pair <int,int>
#define iii pair <int,ii>

int GCD(int a,int b){
	if (b == 0)
		return a;
	return GCD(b,a%b);
}

int main(){
	int a,b,n,c = 0;
	scanf("%d%d",&a,&n);
	for (int i = 0; i < n; i++){
		scanf("%d",&b);
		if (GCD(a,b) != 1)
			c++;
	}
	printf("%d\n",c);
}
