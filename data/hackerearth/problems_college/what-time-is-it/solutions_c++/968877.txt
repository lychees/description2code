#include <iostream>
#include <iomanip>
using namespace std;
 
int main()
{
    double h,m,d;
    char s;
    cin>>h>>s;
    cin>>m;
    if(h==12)
    h=0;
    if(d==60)
    {
    	h++;
    	d=0;
    }
    m=m*6;
    h=h*30+m/12;
    d=h-m;
    if(d<0)
    d=-d;
    if(d>180)
    d=360-d;
    cout<<setprecision(6)<<fixed<<d;
    return 0;
}
