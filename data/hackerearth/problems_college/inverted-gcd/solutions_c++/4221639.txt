#include <iostream>
using namespace std;

int gcd( int a, int b ) {
  if ( a < b ) return gcd( b, a );

  if ( b == 0 ) return a;

  return gcd( b, a % b );
}

int main()
{
    int n;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    int count=0;
    for(int i=0;i<n;i++)
    {
    	for(int j=i+1;j<n;j++)
    	{
    		if(a[j]<a[i])
    		{
    			if(gcd(a[i],a[j])==1)
    				count++;
    		}
    	}
    }
    cout<<count<<endl;
    return 0;
}
