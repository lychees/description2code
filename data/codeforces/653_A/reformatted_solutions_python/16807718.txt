import sys

in_str = raw_input()

in_lst = sorted(map(int, (raw_input().rstrip('\n').split())))
count = 0
for i in xrange(len(in_lst)-1):
    if in_lst[i+1] - in_lst[i] == 1:
        count += 1
    elif in_lst[i+1] - in_lst[i] > 1:
        count = 0
    if count == 2:
        break

if count == 2:
    print 'YES'
else:
    print 'NO'
