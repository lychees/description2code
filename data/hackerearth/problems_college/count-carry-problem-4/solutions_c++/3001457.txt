#include<iostream>
#include<math.h>
using namespace std;


int main () {
	int t;
	cin>>t;
	while(t--) {
		long long int a,b;
//		a = b = 5555555555;
//		cout<<a<<"  "<<b<<endl;
		cin>>a>>b;
		int numCar =0, carry = 0;
		while(1) {
			int temp1 =a%10;
			int temp2 = b%10;
//			cout<<temp1 <<" "<<temp2;
			if((temp1 + temp2 + carry) >= 10) {
				numCar++;
				carry = (temp1 + temp2 + carry)/10;
			} else
				carry = 0;
			a = a/10;
			b = b/10;
			if(a==0 && b==0)
				break; 		
		}
		if(numCar==0) {
			cout<<"No carry operation\n";
		}else if(numCar==1) {
			cout<<"1 carry operation\n";
		} else {
			cout<<numCar<<" carry operations\n";
		}
	}
	return 0;
}
