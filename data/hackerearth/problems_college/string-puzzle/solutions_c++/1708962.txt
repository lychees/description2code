#include<string.h>
using namespace std;
void swap(char *x,char *y)
{
	char temp=*x;
	*x=*y;
	*y=temp;
}
void permutate(char *s,int i,int n)
{
	int j;
	if(i==n)
		cout<<s<<endl;
	else
		for(j=i;j<=n;j++)
		{
			swap((s+i),(s+j));
			permutate(s,i+1,n);
			swap((s+i),(s+j));
		}
}
