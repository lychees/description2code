#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    long long n,k,min=999999999,i,t;
    cin>>n;
    long long a[n];
    cin>>k;
    for(i=0;i<n;i++)
     cin>>a[i];
    sort(a,a+n); 
    for(i=0;i<=n-k;i++)
     {
     	t=a[i+k-1]-a[i];
     	if(t<min) 
         min=t;
     } 
    cout<<min<<"\n";  
    return 0;
}
