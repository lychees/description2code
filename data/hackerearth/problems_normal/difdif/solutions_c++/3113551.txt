#include<bits/stdc++.h>
using namespace std;
int main()
{
    ios_base::sync_with_stdio(false);
    set<string> s;
    string s1;
    int i,j;
    cin>>s1;
    for(i=0;i<s1.size();i++)
    {
        for(j=i;j<s1.size();j++)
        {
            string current_sub_string=s1.substr(i,j-i+1);
            s.insert(current_sub_string);
        }
    }
    cout<<s.size()<<endl;
    return 0;
}
