#include <bits/stdc++.h>
using namespace std;

#define rep(i,n)	for(int i=0;i<n;i++)

int num = 0,sum=0;

void solve(int* a,int n,int e1,int e2,int i)
{
	if(i==n){
		num++;	return;
	}
	if(e1+a[i]<=e2)
	{
		e1+=a[i];
		solve(a,n,e1,e2,i+1);
		e1-=a[i];
	}
	e2+=a[i];
	solve(a,n,e1,e2,i+1);
	e2-=a[i];
}

int main()
{
	int a[9];
	int n;	cin>>n;
	rep(i,n)	cin>>a[i];
	int e1,e2;
	sort(a,a+n);	
	for(int i=0;i<n;i++)	sum+=a[i];
	do{
		e1 = 0;	e2 = 0;
		solve(a,n,e1,e2,0);
	}	while(next_permutation(a,a+n));

	cout<<num<<' '<<sum<<'\n';
	if(num>sum)	cout<<"We will win!";
	else cout<<"Got no way out!";
}
