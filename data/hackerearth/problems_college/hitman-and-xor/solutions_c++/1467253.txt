#include <iostream>
#include <stack>
#include <map>
#include <algorithm>
#include <set>
#define ll long long int

using namespace std;

int main()
{
    ll t;

    cin >> t;

    ll arr[100005];

    while (t--) {
        ll n;
        cin >> n;
        ll odd = 0;
        ll even = 0;

        for (int i = 0; i < n; i++) {
            cin >> arr[i];
            if (arr[i] % 2 == 0) {
                even++;
            }
            else {
                odd++;
            }
        }

        cout << even * odd << endl;
    }
}
