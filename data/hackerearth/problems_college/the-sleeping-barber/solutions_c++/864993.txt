#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    long long int d;
    cin >>d;
    while(d--)
    {
    	long long int n,k,a[3000],i,sum=0;
    	cin >>n >>k;
    	for(i=1;i<=n;i++)
    	{
    		cin >>a[i];
    	}
    	
    	sort(a+1,a+n+1);
    
    	for(i=1;i<=n;i++)
    	{
    		sum+=a[i];
    		if(sum>k)
    		{
    			cout << i-1  <<endl;
    			break;
    		}
    	}
    	
    }
    return 0;
}
