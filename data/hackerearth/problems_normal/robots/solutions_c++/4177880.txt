#include <bits/stdc++.h>
#define lli long long int
#define M 1000000007

using namespace std;

void swap(int *a, int *b) {
    int t = *a;
    *a = *b;
    *b = t;
}

int main()
{
    int n, t;
    int A[1001];

    cin >> t;
    while (t--) {
        cin >> n;
        for (int i = 0; i < n; i++) {
            cin >> A[i];
        }
        sort(A, A + n);
        lli cnt = A[0];
        for (int i = 1; i < n; i++) {
            cnt = (cnt % M) * ((A[i] - i) % M);
            cnt %= M;
        }
        cout << cnt << endl;
    }


    return 0;
}
