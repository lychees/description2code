n = int(raw_input())


def passw(s, p):
    for i in range(len(p)):
        temp = s[0:i] + s + s[i:]
        if temp == p:
            return 1
    return 0


for i in range(n):
    s, p = raw_input().split()
    a = passw(s, p)
    if a == 1:
        print "Possible"
    else:
        print "Impossible"
