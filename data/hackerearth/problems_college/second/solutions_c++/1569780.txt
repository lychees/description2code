#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
    int n,m;
    cin>>n>>m;
    int a[n],i,j,f=0;
    for(i=0;i<n;i++)
     { cin>>a[i];
       if(a[i]==m)
           f=1;
     }
   if(!f)
   cout<<0;
   else
   {
   	  vector<int> v;
   	  for(i=0;i<n;i++)
           {
           	 if(!(a[i]%m))
           	  v.push_back(a[i]/m);
           }
     sort(v.begin(),v.end());
      int t=0;
      for(i=0;i<v.size();i++)
        {
        	if(v[i]-t==1)
        	  t++;
            else if(v[i]==t)
                continue;
             else   
               break;
        }
      cout<<t*m;
   
   }
    return 0;
}
