#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int items, shops;
    	cin>>items>>shops;
    	while(items--)
    	{
	    	int minprice = 1000000;
	    	int minindex;
    		for(int i=1;i<=shops;i++)
    		{
    			int x,y,z;
    			cin>>x>>y>>z;
		    	int price = (100-x)*(100-y)*(100-z);
		    	if(minprice>price)
		    	{
		    		minprice=price;
		    		minindex=i;
		    	}
    		}
    		cout<<minindex<<" ";
    	}
    	cout<<endl;
    }
    return 0;
}
