#include<bits/stdc++.h>
using namespace std;
const int N=1111;
int a[N],s[N];
int main()
{
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%d",&a[i]);
        s[i+1]=s[i]+a[i];
    }
    map<int,vector<pair<int,int> > > M;
    for(int i=1;i<=n;i++){
        for(int j=i;j<=n;j++){
            M[s[j]-s[i-1]].push_back(make_pair(i,j));
        }
    }
    map<int,set<int> > ans;
    for(map<int,vector<pair<int,int> > > :: iterator it=M.begin();it!=M.end();it++){
        vector<pair<int,int> > jobs=it->second;
        sort(jobs.begin(),jobs.end());
        int en=jobs[0].second;
        int cnt=1;
        for(int i=1;i<jobs.size();i++){
            if(jobs[i].first>en){
                en=jobs[i].second;
                cnt++;
            }
        }
        if(cnt>1)
            ans[cnt].insert(it->first);
    }
    printf("%d\n",ans.rbegin()->first);
    set<int> S=ans.rbegin()->second;
    for(set<int> :: iterator it=S.begin();it!=S.end();it++){
        printf("%d ",*it);
    }
    return 0;
}
