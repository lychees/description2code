#include<iostream>
#include<set>
using namespace std;

int main(){
	int k;
	cin >> k;
	string str;
	cin >> str;
	set<char> s;
	int count = 0;
	for(int i = 0; i < str.length(); i++){
		s.clear();
		for(int j = i; j < str.length(); j++){
			s.insert(str[j]);
			if(s.size() == k){
				count++;
			}
		}
	}
	cout << count;
}
