#include <bits/stdc++.h>
#define ll long long
using namespace std;
bool isprime[510];
ll cnt[510],a[510];
void prmgen()
{
	int i,j;
	for(i=2;i<=500;i++)
		isprime[i]=true;
	for(i=2;i<=(int)sqrt(500);i++)
	{
		if(isprime[i])
		{
			for(j=i*i;j<=500;j+=i)
				isprime[j]=false;
		}
	}

}
int main()
{
	int N,i;
	prmgen();
	cin>>N;
	a[0]=a[1]=0;
	a[2]=1;
	cnt[0]=cnt[1]=0;
	cnt[2]=1;
	for(i=3;i<=N;i++)
	{
		if(isprime[i])
			cnt[i]=cnt[i-1]*2;
		else
			cnt[i]=cnt[i-1];
		a[i]=a[i-1]+cnt[i];
	}
	cout<<a[N]<<endl;
}
