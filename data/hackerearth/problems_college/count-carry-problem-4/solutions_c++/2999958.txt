#include <iostream>
using namespace std;

int main()
{
    long int n,count=0,a,b,carry;
    cin >> n;
  
    while(n--)
    {
    	count = 0;
    	carry = 0;
    	cin >> a >> b;
    	int x,y;
    	while(a>0 || b >0)
    	{
    		x = a%10;
    		y = b%10;
    		a = a/10;
    		b = b/10;
    		if(x+y+carry >= 10)
    		{
    			count++;
    			carry = 1;
    		}
    		else
    			carry = 0;
    	}
    	
    	if(count == 0)
    		cout << "No carry operation" << endl;
    	else if(count == 1)
    		cout << 1 << " carry operation" << endl;
    	else
    		cout << count << " carry operations" << endl;
    }
    return 0;
}
