t = int(raw_input())
ans = []
for i in xrange(t):
	line = raw_input().split(" ")
	if line[1].index("B") < line[0].index("B"):
		ans.append("No")
		continue
	if line[0].index("A") < line[1].index("A"):
		ans.append("No")
		continue
	first, second = ''.join(line[0].split("#")), ''.join(line[1].split("#"))
	if first == second:
		ans.append("Yes")
	else:
		ans.append("No")

for a in ans:
	print a
