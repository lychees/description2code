#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int i,j;
    	int flag = 0;
    	for(i = 0; i<n;i++)
    	{
    		for(j = 0; j<n;j++)
    		{
    			if(n == 8*j + 6*i)
    			{
    				flag = 1;
    				break;
    			}
    			if(n < 8*j + 6*i)
    			{
    				break;
    			}
    		}
    		if(flag == 1)
    		{
    			break;
    		}
    	}
    	if(flag == 1)
    	{
    		cout<<i+j<<endl;
    	}
    	else
    	{
    		cout<<"-1"<<endl;
    	}
    }
    return 0;
}
