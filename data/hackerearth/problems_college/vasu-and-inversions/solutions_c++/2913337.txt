#include<bits/stdc++.h>
#include<stdio.h>
#define ll long long
using namespace std;

int main()
{
    ios_base::sync_with_stdio(0);
    int t,n,k,i;
    cin>>t;
    while(t--)
    {
        int c1=0,c0=0,x;
		cin>>n>>k;
		for(i=0;i<n;i++)
		{
			cin>>x;
			if(x==1)
			c1++;
			else
			c0++;
		}
		if(c0>=k)
		{
			cout<<c1+k<<endl;
		}
		else
		{
			k=k-c0;
			c1 += c0;
			if(k%2==0)
			{
				cout<<c1<<endl;
			}
			else
			{
				cout<<c1-1<<endl;
			}
		}
    }
    return 0;
}
