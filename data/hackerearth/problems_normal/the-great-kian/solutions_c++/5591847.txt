#include<bits/stdc++.h>
using namespace std;



int main()
{
	long int a,b=0,c=0,d=0,n,count=0;
	scanf("%ld",&n);
	for(long int i=0;i<n;i++)
	{scanf("%ld",&a);
	if(count==0)
	{b+=a;
	count=1;
	}
	else if(count==1)
	{
		c+=a;
		count=2;
	}
	
	else if(count==2)
	{
		d+=a;
		count=0;
	}
}

printf("%ld %ld %ld",b,c,d);
return 0;
}
