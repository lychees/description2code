#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,d;
    	cin>>n>>d;
    	int extra=0;
    	int travel=0,total=0;
    	while(n)
    	{
    		travel=(n*2)+extra;
    		extra=travel%6;
    	//	cout<<travel<<"   ";
    		n=(travel/6)*2;
    		if(travel>=6)
    			travel-=extra;
    		total+=travel;
    		travel=0;
    	//	cout<<total<<"     "<<extra<<"  "<<n<<endl;
    	}
    
    	if(total>=d)
    	cout<<"Yes"<<endl;
    	else
    	cout<<"No"<<endl;
    	
    }
    return 0;
}
