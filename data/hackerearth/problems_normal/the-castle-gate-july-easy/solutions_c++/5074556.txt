#include <iostream>
using namespace std;

int main()
{
   //cout << "Hello World!" << endl;
    int t,n;
    cin>>t;
    while(t--)
    {int c=0;
    	cin>>n;
    	for(int i=1;i<=n;i++)
    	{
    		for(int j=i+1;j<=n;j++)
    		{
    			if((i^j)<=n)
    			c++;
    		}
    	}
    	cout<<c<<endl;
    }
    return 0;
}
