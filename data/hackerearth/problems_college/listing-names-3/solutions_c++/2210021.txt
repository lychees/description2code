#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<cstring>
#include<string>
#include<map>
using namespace std;
int n;
string str;
map<string,int> M;
map<string,int>::iterator it;
int main()
{
  scanf("%d",&n);
  while(n--)
  {
    cin>>str;
    M[str]++;
  }
  for(it=M.begin();it!=M.end();it++)
    cout<<it->first<<" "<<it->second<<"\n";
  return 0;
}
