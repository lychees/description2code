#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin >> t;
	
	while(t--)
	{
		int n,l,r,x;
		cin >> n;
		
		int *A = new int[n];
		int *I = new int[n];
		
		for(int i=0;i<n;i++)
		{
			cin >> A[i];
			I[i] = -1;
		}
		
		int min = n+1;
		int s = 0;
		
		for(int i=0;i<n;i++)
		{
			s += A[i];
			s %= n;

			if(I[s]==-1 && s!=0)
				x=n+1;
			else if(I[s]==-1 && s==0)
				x=i+1;
			else
				x=i-I[s];
			
			if(x<min)
			{
				min = x;
				l=I[s]+1;
				r=i;
			}
			
			I[s] = i;
		}
		
		cout << l+1 << " " << r+1 << endl;
	}
}
