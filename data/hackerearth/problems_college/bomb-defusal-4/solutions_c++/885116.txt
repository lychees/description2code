
#include<stdio.h>
int main()
{
    int p,x,y,u,v,a,b,n,k;
    scanf("%d",&n);
    while(n--)
    {
        k=1000;
        scanf("%d%d%d%d%d",&p,&x,&y,&u,&v);
        a=x-u;
        if(a<0)
            a=-a;
        b=y-v;
        if(b<0)
            b=-b;
//printf("a=%d b=%d\n",a,b);
        if(a==0&&b==0)
            k=0;
        else if(a==b)
            k=1;
        else
        {
            if(a%2==0&&b%2==0)
                k=2;
            else if(a%2==1&&b%2==1)
                k=2;
        }
//printf("%d\n",k);
        if(k<p)
            printf("Counter-Terrorists Win !\n");
        else
            printf("Terrorists Win !\n");
    }
}
