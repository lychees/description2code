#include <iostream>
#include <cstdlib>
using namespace std;

struct node
{
    long long int data;
    node *next;
};

class s
{
    node *top;

public:
    s() {top = NULL;}

    void push(long long int item)
    {
        node *temp;
        temp = new node;
        temp->data = item;
        temp->next = top;
        top = temp;
    }

    int pop()
    {
        if(top == NULL)
            return 0;
        else
        {
            node *temp = top;
            top = top->next;
            return temp->data;
        }
    }

};

int main()
{
    int q;
    cin >> q;
    if(q>0 && q<100001)
    {
        s s1;
        int c;
        for(int i=0; i<q; i++)
        {
            cin >> c;
            switch(c)
            {
            case 1:
                {
                    int t = s1.pop();
                    if(!t)
                        cout << "No Food\n";
                    else
                        cout << t << '\n';

                    break;
                }
            case 2:
                {
                    long long int cost;
                    cin >> cost;
                    if(c<1 || c>10000001)
                        exit(0);
                    else
                        s1.push(cost);
                }
            }
        }
    }
}
