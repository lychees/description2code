#include <iostream>

int main()
{
	std::string s1, s2;
	int letter[26];
	int t;
	bool pos, neg;
	std::cin >> t;
	getline(std::cin, s1);
	for (int i = 0; i < t; i++) {
		getline(std::cin, s1);
		getline(std::cin, s2);
		pos = true;
		neg = true;
		for (int j = 0; j < 26; j++) {
			letter[j] = 0;
		}
		for (int j = 0; j < s1.size(); j++) {
			if (s1[j] >= 'a' && s1[j] <= 'z') {
				letter[s1[j] - 'a']++;
			}
		}
		for (int j = 0; j < s2.size(); j++) {
			if (s2[j] >= 'a' && s2[j] <= 'z') {
				letter[s2[j] - 'a']--;
			}
		}
		for (int j = 0; j < 26; j++) {
			if (letter[j] > 0) {
				neg = false;
			}
			if (letter[j] < 0) {
				pos = false;
			}
		}
		if (pos) {
			std::cout << "You win some." << std::endl;
		}
		if (neg) {
			std::cout << "You lose some." << std::endl;
		}
		if (!neg && !pos) {
			std::cout << "You draw some." << std::endl;
		}

	} 
}
