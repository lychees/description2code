#include <iostream>
using namespace std;

int main()
{
	int ignoreCount;
	cin >> ignoreCount;
	string ignores [ignoreCount];
	for(int i = 0; i < ignoreCount; i++)
	{
		string s;
		cin >> s;
		ignores[i] = s; 
	}
	
	int words; 
	cin >> words;
	int acronym[words + 1];
	int counter = 0;
	for(int k = 0; k < words; k++)
	{
		string word;
		cin >> word;
		int good = 1;
		for(int j = 0;j < ignoreCount; j++)
		{
			if(ignores[j] == word)
			{
				good = 0;
			}
		}
		if(good)
		{
			acronym[counter] = (word[0] - 32);
			counter ++;
		}
	}
	acronym[counter] = -1;
	counter = 0;
	while(acronym[counter] != -1)
	{
		if(acronym[counter + 1] == -1)
		{
			cout << (char)acronym[counter];
		}
		else 
		{
			cout << (char)acronym[counter] << ".";
		}
		counter++;
	}
    return 0;
}
