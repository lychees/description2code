#include <iostream>
#include <stdio.h>
#include <string>

using namespace std;

void convert(char s[]) {

	bool comment = false;

	int i = 0;
	while(s[i] != '\0') {
		if(s[i] == '/' && s[i+1] == '/') {
			comment = true;
		}

		if(s[i] == '-' && s[i+1] == '>' && !comment) {
			cout << ".";
			i++;
		} else {
			cout << s[i];
		}
		i++;
	}
	cout << endl;

}

int main() {

	char s[70];
	while(gets(s) != NULL) {
		convert(s);
	}

}
