
#include<bits/stdc++.h>
#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	for(int k = 0; k < t; k++){
		int n;
		cin>>n;
		long long int a[n];
		int start = 0, end = n,range = n;
	
		unordered_map< int, int > mp;
		
		for(int i = 0; i < n; i++){
			cin>>a[i];
		}
		for(int i = 1; i < n; i++){
			a[i] += a[i-1];
		}
		for(int i = 0; i < n; i++){
			a[i] = a[i]%n;
		}
		for(int i = 0; i < n; i++){
			if(a[i] == 0){
				end = i;
				break;
			}
		}
		if(end != n){
			start = -1;
			range = end;
		}
		
		for(int i = 0; i < n; i++){
			if(mp.find(a[i]) == mp.end()){
				mp[a[i]] = i;
 			}
 			else{
 				int t1,t2;
 				t2 = i;
 				t1 = mp[a[i]];
 				if(range > (t2-t1) ){
 					start = t1;
 					end = i;
 					range = t2-t1;
 				}
 				mp[a[i]] = i;
 			}
		}
		start += 2;
		end++;
		cout<<start<<" "<<end<<endl;
		
	}

    return 0;
}
