#include <iostream>
using namespace std;
void rahul(int n)
{
	int i,j=1,k=1;
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=n-i;j++)
		cout<<" ";
		if(i%2==0)
		{
			for(j=0;j<k;j++)
			cout<<"/ ";
			cout<<" ";
			for(j=0;j<k-1;j++)
			cout<<"\\ ";
			cout<<"\\";
			k++;
			cout<<"\n";
		}
		else
		{
			for(j=0;j<k-1;j++)
			cout<<"/ ";
			cout<<"/";
			for(j=0;j<k-1;j++)
			cout<<"\\ ";
			cout<<"\\";
			cout<<"\n";
		}
	}
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=i+1;j++)
		cout<<" ";
		if(i%2==0)
		{
			k--;
			for(j=0;j<k;j++)
			cout<<"\\ ";
			cout<<" ";
			for(j=0;j<k-1;j++)
			cout<<"/ ";
			cout<<"/";
			cout<<"\n";
		}
		else
		{
			for(j=0;j<k-1;j++)
			cout<<"\\ ";
			cout<<"\\";
			for(j=0;j<k-1;j++)
			cout<<"/ ";
			cout<<"/";
			cout<<"\n";
		}
	}
}
