#include <iostream>
#include <algorithm>
#include <string>
using namespace std;

int main()
{
    long long T, M;
    cin >> T;
    
    while (T--)
    {
    	
    	cin >> M;
    	
    	if (M==1)
    		cout << "0 1" << endl;
    	else if (M==2)
    		cout << "1 1" << endl;
    	else
    	{
    		long long k, one = 1, zero = 1;
    		
    		for (int i=3; i<=M; i++)
    		{
    			k = one;
    			one = zero;
    			zero = zero + k;
    		}
    		cout << one << " " << zero << endl;
    	}
    }
    
    return 0;
}
