#include <iostream>
#include <cmath>
#include <iomanip>
using namespace std;

int main()
{
    int num;
  double a,b,c;
  double max=0;
     cin>>num;
    while(num>0)
    {
    	cin>>a>>b>>c;
    	double radius;
    	double p=a+b+c;
    	double q=a+b-c;
    	double r=b+c-a;
    	double s=a+c-b;
    	radius=(a*b*c)/sqrt(p*q*r*s);
    	double area=3.1415*radius*radius;
    	std::cout << std::fixed;
    	std::cout << std::setprecision(4);
    	std::cout<<area<<endl;
    	num--;
    
   
   
    }
   
    
   return 0;
    
    
}
