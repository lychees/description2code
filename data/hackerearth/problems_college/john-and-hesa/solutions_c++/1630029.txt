#include <iostream>
using namespace std;

int main()
{
	long t,n,p,c,n1;
	cin >> t;
	while(t--)
	{
	 cin >> n>> p;
	 c=0;
	 n1=n;
	 while(1)
	 {
	 	if(n%p==0)
	 	{
	 		n=n/p;
	 		c++;
	 	}
	 	else 
	 	{
	 	break;
	 	}
	 	
	 }
	 
	 if(n==1 && c)
		cout << c << " 0"  << endl; 
	 
		else 
		cout << c << " "<< n << endl; 
	
	}
    
    return 0;
}
