#include <iostream>
using namespace std;

int main()
{
    
    int t ; 
    cin >> t ; 
    
    
    int arr[11][1001];
    
    
    
    for(int i=1 ; i < 11 ;i++){
    	arr[i][1]=1;
    	arr[i][0]=0;
    }
    
    for(int i =0 ; i <1001 ; i++)
    	arr[1][i]=i;
    
    
    for(int i=2 ; i < 11 ; i ++){
    	
    	for(int j =2 ; j <1001 ; j++){
    		
    		arr[i][j]=9999999;
    		
    		for(int  k =1 ; k <=j;k++){
    			
    			int xx = 1+max(arr[i-1][k-1],arr[i][j-k]);
    			
    			if(xx<arr[i][j])
    			arr[i][j]=xx;
    			
    		}
    		
    		
    	}
    	
    }
    
    
    while(t--){
    	
    	int bul , egg ;
    	cin >> bul >> egg ;
    	
    	cout << arr[egg][bul]  << endl ;
    	
    }
    
}
