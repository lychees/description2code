#include <bits/stdc++.h>
using namespace std;
#define ll long long int

int compare(pair<ll,ll>a,pair<ll,ll>b)
{
	return a.first>b.first;
}

int main()
{
	ios::sync_with_stdio(false);
	
	int t;
	cin >> t;
	while(t--)
	{
		int n,i,j;
		cin >> n;
		vector<pair<ll,ll> >arr;
		ll ans[209],as=0;
		for(i=0;i<n;i++)
		{
			ll x,y;
			cin >> x >> y;
			arr.push_back(make_pair(x,y));
		}
		sort(arr.begin(),arr.end(),compare);
		// for(i=0;i<n;i++)cout << arr[i].first << ' ' << arr[i].second << endl;
		for(i=0;i<n;i++)
		{
			ans[i]=arr[i].second;
		}
		for(i=0;i<n;i++)
		{
			for(j=0;j<i;j++)
			{
				if(arr[i].first<arr[j].first && arr[i].second <arr[j].second)
				{
					ans[i]=max(ans[i],ans[j]+arr[i].second);
				}
			}
		}
		for(i=0;i<n;i++)as=max(as,ans[i]);
		cout << as << endl;
	}
	
    return 0;
}
