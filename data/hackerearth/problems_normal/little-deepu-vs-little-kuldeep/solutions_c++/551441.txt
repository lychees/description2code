#include <iostream>
using namespace std;
#define mod 1000000007

int64_t power(int64_t a,int b)
{
    if(b==0)
        return 1;
    else if(b==1)
        return a;
    else
    {
        int64_t t=power(a,b/2);
        t=(((t*t)%mod)*power(a,b%2))%mod;
        return t;
    }
}

int main()
{
     int t,n,i,l,j;
     static int64_t a[1000000],p;
    cin>>t;
    a[1]=1;
    for(i=0;i<t;i++)
    {
    	cin>>n;
    	if(a[n]==0)
    	{
    		l=n;
    		while(a[l]==0)
              l--;
            for(j=l+1;j<=n;j++)
               {
               	int64_t d=((2*(2*j-1)));
               	a[j]=(d*a[j-1])%mod;
               	p=power((j+1),mod-2);
               	a[j]*=p;
               	a[j]%=mod;
               }
         }
    	cout<<a[n]<<"\n";
    }
    return 0;
}
