#include<iostream>
using namespace std;
#include<stdlib.h>
#include<string.h>
#include<math.h>

int main()
{
   int t;
   long int x,range,i,num,index,loop;
   cin>>t;
   while(t--)
   {
   	  cin>>x;
   	  range=2,i=2;
   	  if(x==1)
   	   cout<<"Hacker"<<endl;
   	  else if(x==2)
   	   cout<<"Earth"<<endl;
   	  else
   	  {
   	  	 while(range<x)
   	  	 {
   	  	 	range=range+pow(2,i)*i;
   	  	 	i++;
   	  	 }
   	  	 i=i-1;
   	  	 range=range-pow(2,i)*i;
   	  	 x=x-range;
   	  	 index=x%i;
   	  	 if(index==0)
   	  	 {
   	  	 	num=(x/i)-1;
   	  	 	index=i;
   	  	 }
   	  	 else
   	  	    num=(x/i);
   	  	 
   	  	 loop=i-index;
   	  	 while(loop)
   	  	 {
   	  	 	num=num>>1;
   	  	 	loop--;
   	  	 }	
   	  	 if(num%2==0)
   	  	   cout<<"Hacker"<<endl;
   	  	 else
   	  	   cout<<"Earth"<<endl;
   	  	   	  	 
   	  }
   }
  
}
