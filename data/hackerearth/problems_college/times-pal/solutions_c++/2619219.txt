#include <iostream>
using namespace std;

bool ispalin(string A){

int low = 0,high = A.size()-1;

while(low<high){
	if(A[low]!=A[high]){
		return 0;
	}
	low++;
	high--;
}

return 1;
}
int main()
{
    
    int t;
    cin>>t;
    
    while(t--){
    	
    	string A,B;
    	cin>>A>>B;
    	int count = 0;
    	for(int i = 0;i<=A.size();i++){
    		string temp = A;
    		string new_string = temp.insert(i,B);
    		if(ispalin(new_string))
    		count++;
    	}
    	
    	cout<<count<<"\n";
    }
    return 0;
}
