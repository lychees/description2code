#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll long long int
#define pb push_back
#define mk make_pair
ll power(ll a, ll b) {
ll x = 1, y = a;
    while(b > 0) {
        if(b%2 == 1) {
            x=(x*y);
            if(x>mod) x%=mod;
        }
        y = (y*y);
        if(y>mod) y%=mod;
        b /= 2;
    }
    return x;
}
int A[1000002];
int main()
{
	int T,N,K;
	cin >> T ;
    while(T--){
        cin >> N >> K ;
        for(int i=1;i<=N;i++){
            cin >> A[i] ;
        }
        ll need = (1LL * N * (N+1) / 2) , notneeded;
        notneeded = 0 ;
        int i = 1;
        while(i<=N){
            if(A[i] > K){
                i ++ ;
                continue ;
            }
            int cnt = 0 ;
            while(i <= N && A[i] <= K){  
                i ++ ;
                cnt ++ ;
            }
            notneeded += (1LL * cnt * (cnt+1) / 2) ;
        }
        cout << need-notneeded << endl ;
    }
	return 0;
}
