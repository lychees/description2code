#include<cstdio>
#include<stack>
#include<cstring>
using namespace std;
int main()
{
	int t;
	scanf("%d ",&t);
	while(t--)
	{
		char s[101];
		gets(s);
		stack<char> s1;
		int l=strlen(s);
		for(int i=0;i<l;i++)
		{
			if(s[i]=='x')
			{
				while(!s1.empty() && s1.top()=='o')
					s1.pop();
				s1.push(s[i]);
			}
			else if(s[i]=='*' || s[i]=='o')
				s1.push(s[i]);
		}
		stack<char> s2;
		while(!s1.empty())
		{
			char ch=s1.top();
			s1.pop();
			if(ch=='x')
			{
				while(!s2.empty() && s2.top()=='o')
					s2.pop();
				s2.push(ch);
			}
			else if(ch=='*' || ch=='o')
				s2.push(ch);
		}
		while(!s2.empty())
		{
			char ch=s2.top();
			s2.pop();
			printf("%c",ch);
		}
		printf("\n");
	}
	return 0;
}
