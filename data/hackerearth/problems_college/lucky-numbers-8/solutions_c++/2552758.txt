#include <iostream>
using namespace std;

int main()
{
	long int i,j,t,n,e[1000001],temp,count;
	cin>>t;
	while(t--)
	{
		count=0;
		cin>>n;
		for(i=0;i<n;i++)
		{
			cin>>e[i];
		}
		for(i=0;i<n-1;i++)
		{
			temp=2;
			for(j=i+1;j<n;j++)
			{
				if(e[i]<=e[j])
				{
					temp=1;
					break;
				}
			}
			if(temp==2)
			{
				count++;
			}
		}
		cout<<count+1;
		cout<<"\n";
	}
    //cout << "Hello World!" << endl;
    return 0;
}
