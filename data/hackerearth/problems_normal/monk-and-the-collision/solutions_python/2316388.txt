for _ in range(t):
    l = input()
    kk = raw_input().strip().split(' ')
#     print kk
    ss = []
    for k1 in kk:
        ss.append(k1[len(k1) - 1])
    ss.sort()
#     print ss
    sstr = ''.join(ss)
#     print sstr
    co = 0
    for i in range(11):
        tmp = sstr.count(str(i))
#         print tmp
        if tmp > 1:
            co = co + tmp - 1
    print co
