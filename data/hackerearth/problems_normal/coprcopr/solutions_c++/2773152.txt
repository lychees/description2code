LL phi[10000001] = {0};

{
	inp_s;
	FOR(i,0,10000001) phi[i] = i;
	for(int i = 2; i <= 10000000; i++)

		if(phi[i] != i) continue;

		while(j <= 10000000)

			phi[j] -= (phi[j]/i);

		}
