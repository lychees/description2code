LL phi[1000001] = {0};

int main()
{
      FOR(i,0,1000001) phi[i] = i;
      FOR(i,2,1000001)
      {
            if(phi[i] != i) continue;
            int j = i;
            while(j<=1000000)
            {
                  phi[j] -= phi[j]/i;
                  j+=i;
            }
      }
      DRT()
      {
            int a,b;
            cin >> a >> b;
            LL ans = 0;
            FOR(k,1,min(a,b) + 1)
            {
            	if(a%k == 0 && b %k == 0) ans += phi[k];
            }
            cout << ans << endl;
      }
      return 0;
}
