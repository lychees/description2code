#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

int main()
{
    int N;
    scanf("%d",&N);
    int arr[N];
    for(int i=0;i<N;i++){
    	scanf("%d",&arr[i]);
    }
    for(int i=0;i<N;i++){
    	bool flag=true;
    	for(int j=0;j<N;j++){
    		if(i!=j){
    			if(arr[i]!=arr[j]){
    				if(arr[i]%arr[j]==0){
    					flag=false;
    					break;
    				}
    			}
    		}
    	}
    	if(flag){
    		cout<<arr[i]<<" ";
    	}
    }
    cout<<endl;
    return 0;
}
