#include <iostream>
using namespace std;
long MaxnumberofString(long A,long B,long C)
{
	long temp1,temp2;
	temp1=A>=B?A:B;
	temp2=A<B?A:B;
	//cout<<"temp2 is "<<temp2<<endl;
	temp2=temp2>=C?temp2:C;
	//cout<<"temp1 and temp2 are"<<temp1<<" "<<temp2<<endl;
	if(temp1==0 && temp2==0)
	{
		return 0;
	}
	if(temp1==0 || temp2==0)
	{
		return 1;
	}
	if(temp1==temp2)
	{
		return 2*temp1;
	}
	if(temp1-temp2>0)
	{
		return 2*temp2+1;
	}
	if(temp2-temp1>0)
	{
		return 2*temp1+1;
	}
}

int main()
{
	int t;
	cin>>t;
	long arr[3*t];
	for(int i=0;i<=3*t-1;i++)
	{
		cin>>arr[i];
	}
	for(int j=0;j<t;j++)
	{
		cout<<MaxnumberofString(arr[3*j],arr[3*j+1],arr[3*j+2])<<endl;
	}
   // cout << "Hello World!" << endl;
    return 0;
}
