tc = int(raw_input())
while tc:
    tc -= 1
    n = int(raw_input())
    visited = [False]*10001
    for i in range(n):
        a, b = map(int, raw_input().split())
        visited[a] = True
        visited[b] = True
    nodes = 0
    for i in range(10001):
        if visited[i]:
            visited[i] = False
            nodes += 1
    print nodes
