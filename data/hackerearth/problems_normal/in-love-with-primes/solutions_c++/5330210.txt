#include <iostream>
#include <cmath>
using namespace std;

char prime[100001];

int ans[1000006], primes[100001], k;

void genprime(long long int N);

void iniarray(long long int N);

int main() {
	iniarray(100001);
	genprime(100001);
	int i, N, test, j;
	for(i=0;i<1000001;i++)
		ans[i]=0;
	for(i=0;i<k;i++) {
		for(j=0;j<k;j++) {
			ans[primes[i]+primes[j]]=1;
		}
	}
	cin >> test;
	while(test--) {
		cin >> N;
		if(ans[N] == 1)
			cout << "Deepa" << endl;
		else
			cout << "Arjit" << endl;
	}
	return 0;
}

void iniarray(long long int N){
	long long int i=0;
	for(i=0;i<=N;i++){
		prime[i]='0';
	}
}

void genprime(long long int N){
	long long int i=2, j;
	while(i<=N){
		if(prime[i]=='0'){
			primes[k]=i;
			k++;
			j=2;
			while((i*j)<=N){
				prime[(i*j)]='1';
				j++;
			}
		}
		i++;
	}
}
