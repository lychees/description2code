#include <iostream>
#include<string.h>
using namespace std;
#define mod 1000000007
long long d[1000000];
long long f(int i)
{if(i<2)
return 1;
if(d[i]!=-1)
return d[i];
return d[i]=((i%mod)*(f(i-1)%mod)%mod);
}
int main()
{long long y,sum,n;
    int t;
    memset(d,-1,sizeof(d));
    cin>>t;
    while(t--)
    {cin>>n;
     y=f(n)-1;
     sum= (3*y + 6)%mod;
     cout<<sum<<endl;
    }
    return 0;
}
