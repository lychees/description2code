#include<iostream>
#include<cmath>
#include<stdio.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	int i,j;
	int n;
	long long int count,max;
	for(i=0;i<t;i++)
	{
		scanf("%d",&n);
		count=0;
		max=0;
		int *a=new int[n];
		for(j=0;j<n;j++)
		{
			scanf("%d",&a[j]);
			count=count+a[j];
			if(count>a[j])
			a[j]=count;
			else
			count=a[j];
		//	cout<<a[j]<<endl;
			if(a[j]>max)
			max=a[j];
		}
		printf("%lld\n",max);
		
	}
	
	return 0;
}
