#include <bits/stdc++.h>

using namespace std;

int LCS(char *x, char *y, int m, int n)
{
	int len[m+1][n+1];
	int i,j,k;
	for (i = 0; i <= m; i++) {
		for (j = 0; j <= n; j++) {
			if (i == 0 || j == 0) {
				len[i][j] = 0;
			} else if (x[i-1] == y[j-1]) {
				len[i][j] = len[i-1][j-1] + 1;
			} else {
				len[i][j] = max(len[i-1][j],len[i][j-1]);
			}
		}
	}
	
	return len[m][n];
}

int main()
{
	int t,i,j,k,ans;
	char s1[100000],s2[100000];
	scanf("%d",&t);
	while (t --) {
		int n;
		scanf("%d",&n);
		scanf("%s %s",s1,s2);
		ans = LCS(s1,s2,strlen(s1),strlen(s2));
		printf("%d\n",ans);
	}
	
	return 0;
}
