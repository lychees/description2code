#include<bits/stdc++.h>
using namespace std;
#define ll long long int
ll fact[10010];
ll mod=1e9+7;
ll mybin[1002][1002];
void cal()
{
	ll i,j;
	for(i=0; i<=1001; i++)
	{
		for(j=0; j<=i; j++)
		{
			if(j==0)
			{
				mybin[i][j]=1;
				continue;
			}
			mybin[i][j]=(mybin[i-1][j]+mybin[i-1][j-1])%mod;
		}
	}
}
int main()
{
	ll t;
	cal();
	scanf("%lld",&t);
	while(t--)
	{
		ll n,p;
		scanf("%lld %lld",&n,&p);
		ll ans=0,temp=1,i;
		for(i=0; i<=p; i++)
		{
			temp=(mybin[n][i])%mod;
			ans=(ans+temp)%mod;
		}
		printf("%lld\n",ans);
	}
	return 0;
}
