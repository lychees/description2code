#include <bits/stdc++.h>
#define mp make_pair
#define pb push_back
#define ll long long
#define debug(x) cout<<">value ("<<#x<<") : "<<x<<endl;
#define fr(i,beg,end) for(i=beg;i<end;i++)
#define itfr(it,stl) for(it=stl.begin();it!=stl.end();it++)
#define PII pair<int,int>
#define init(x,val) memset(x,val,sizeof(x))
#define fst first
#define snd second
using namespace std;
int dp[100][100][501];
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int t,n,m,k,i,j,K;
    cin>>t;
    while(t--)
    {
        cin>>n>>m>>K;
        int a[n][m];
        fr(i,0,n)
            fr(j,0,m)
            {
                cin>>a[i][j];
                fr(k,0,K+1)
                    dp[i][j][k]=0;
            }
        if(a[0][0]<=K)
            dp[0][0][a[0][0]]=1;
        fr(i,0,n)
        {
            fr(j,0,m)
            {
                fr(k,0,K+1)
                {
                    if(i>0)
                    {
                        if((dp[i-1][j][k])&&((k+a[i][j])<=K))
                            dp[i][j][k+a[i][j]]=1;
                    }
                    if(j>0)
                    {
                        if((dp[i][j-1][k])&&((k+a[i][j])<=K))
                            dp[i][j][k+a[i][j]]=1;
                    }
                    if(i>0 && j>0)
                    {
                        if((dp[i-1][j-1][k])&&((k+a[i][j])<=K))
                            dp[i][j][k+a[i][j]]=1;
                    }
                }
            }
        }
        int ans=-1;
        for(k=K;k>=0;k--)
            if(dp[n-1][m-1][k])
            {
                ans=k;
                break;
            }
        cout<<ans<<endl;
    }
    return 0;
}
