t=int(raw_input())
i=1
while i<=t:
	s=raw_input()
	needed=0
	l=len(s)
	for j in s:
		if j!='a' and j!='e' and j!='i' and j!='o' and j!='u':
			needed+=1
	print "%d/%d"%(needed-3,l)
	i+=1
