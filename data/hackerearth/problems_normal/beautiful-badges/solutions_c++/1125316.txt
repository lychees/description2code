#include <stdio.h>
#include <algorithm>
using namespace std;

int v1[100005], v2[100005];
int main() {
	int n, a, b, t1, t2;
	scanf("%d", &n);
	t1 = t2 = 0;
	for (int i = 0; i < n; i++) {
		scanf("%d %d", &a, &b);
		if (a == 1) {
			v1[t1++] = b;
		}
		else {
			v2[t2++] = b;
		}
	}
	sort(v1, v1+t1);
	sort(v2, v2+t2);
	a = b = 0;
	n = 0;
	while (a < t1 && b < t2) {
		if (2*v1[a]*v1[a] <= v2[b]*v2[b]) a++;
		else if (v1[a]*v1[a] >= 2*v2[b]*v2[b]) b++;
		else {
			a++;
			b++;
			n++;
		}
	}
	printf("%d\n", n);
}
