#include <bits/stdc++.h>
using namespace std;
#define ll long long int
int main(){
	ll t,x,y,u,v,tim,tot;
	cin >> t;
	while(t--){
		cin >> tim >> x >> y >> u >> v;
		x += 10000;
		y += 10000;
		u += 10000;
		v += 10000;
		if(((x+y)%2) == ((u+v)%2)){
			if((x==u) && (y==v)){
				tot = 0;
			}
			else if(fabs(x-u)==fabs(y-v)){
			  tot = 1;
			}
			else{
				tot = 2;
			}
		   if(tot<tim){
		   	cout << "Counter-Terrorists Win !" << endl;
		   }
		   else
		   cout << "Terrorists Win !" << endl; 
		}
		else{
		  cout << "Terrorists Win !" << endl;
		}
	}
	return 0;
}
