#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        vector<ll> v;
        v.clear();
        ll n,x,k;
        scanf("%lld%lld",&n,&k);
        for(ll i=0;i<n;i++)
        {
            scanf("%lld",&x);
            v.push_back(x);
        }
        sort(v.begin(),v.end());
        ll sum=0;
        if(k>=n)
        {
            for(ll i=0;i<n;i++)
              sum+=v[i];
        }
        else
        {
            for(ll i=n-1;i>=(n-k);i--)
                sum+=v[i];
        }
        printf("%lld\n",sum);
    }
    return 0;
}
