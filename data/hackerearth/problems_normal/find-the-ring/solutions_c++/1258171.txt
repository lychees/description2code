#include <stdio.h>
int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    	int n,i;
    	scanf("%d %d",&i,&n);
    	if(n==0){
    		printf("%d\n",i);
    	}
    	else{
    		while(n--){
    			if(i==0){
    				i=1;
    			}
    			else{
    				i--;
    			}
    		}
    		printf("%d\n",i);
    	}
    }
    return 0;
}
