#include <iostream>
#include <string.h>
#include <algorithm>
#define SIZE 100010
using namespace std;

int dp[SIZE][2];

void pre() {
	
	for(int i=0; i<2; i++)
		for(int j=0; j<SIZE; j++) {
			dp[j][0] = SIZE;
			dp[j][1] = SIZE;
		}
	dp[1][0]=1;
	dp[1][1]=1;
	dp[2][0]=1;
	dp[2][1]=2;
	
	for(int i=1; i<SIZE; i++) {
		int last_step = dp[i][1];
		if(i+last_step<SIZE) {
			if(dp[i+last_step][0]>dp[i][0]+1) {
				dp[i+last_step][0] = dp[i][0]+1;
				dp[i+last_step][1] = dp[i][1];
			}
		}
		if(i+last_step+1<SIZE) {
			if(dp[i+last_step+1][0]>dp[i][0]+1) {
				dp[i+last_step+1][0] = dp[i][0]+1;
				dp[i+last_step+1][1] = dp[i][1]+1;
			}
		}
	}
}
int main()
{
    int test;
    cin>>test;
    pre();
    while(test--) {
    	int steps;
    	cin>>steps;
   		cout<<dp[steps][0]<<endl;
    }
    return 0;
}
