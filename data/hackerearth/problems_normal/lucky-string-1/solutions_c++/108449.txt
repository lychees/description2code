using namespace std;

#define MOD 1000000007
#define LL long long



int main()
{
	int t;
	cin>>t;
	LL n;
	while(t--)
	{
		scanf("%lld",&n);
		LL i=1;
		while(n-(powl(2,i)*i)>0)
		{
			n -= powl(2,i)*i;
			i++;
		}
		n--;
		LL p = n/i;
		LL r=n%i;
		p = p>>(i-r-1);
		if(p&1)
			printf("Earth\n");
		else
			printf("Hacker\n");
	}
	return 0;
}
