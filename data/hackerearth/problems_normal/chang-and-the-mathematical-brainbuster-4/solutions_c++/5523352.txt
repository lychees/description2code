#include <bits/stdc++.h>
using namespace std;
int n,a[10009];
int main()
{
	//memset(dp,-1,sizeof(dp));
    cin>>n;
    for(int i=1;i<=n;i++) cin>>a[i];
    sort(a+1,a+n+1);
    int ans=INT_MAX;
    for(int i=n;i>=1;i--)
    {
    	int x=ceil( (float) a[i]/3.0 );
    	int y=lower_bound(a+1,a+i+1,x)-a;
    	if(y>n) continue;
    	ans=min(ans,n-(i-y+1));
    }
    cout<<ans<<endl;
}
