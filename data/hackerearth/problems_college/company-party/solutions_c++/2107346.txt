#include <cstring>
#include <iostream>
#include <cstdio>
#include <algorithm>
#include <sstream>
#include <set>
#include <queue>

using namespace std;

int p[2005];
int d[2005];

int DFS(int pos){
	if (d[pos] != -1){
		return d[pos];
	}
	if (p[pos] == -1){
		return d[pos] = 1;
	} else {
		return d[pos] = 1 + DFS(p[pos]);
	}
}

int main(){
	memset(d, -1, sizeof d);
	int n;
	scanf("%d", &n);
	for (int i=1; i<=n; i++){
		scanf("%d", &p[i]);
	}
	int ret = 1;
	for (int i=1; i<=n; i++){
		ret = max(ret, DFS(i));
	}
	cout << ret << endl;
	return 0;
}
