#include <bits/stdc++.h>
using namespace std;

long long DP[201][201];
int n, kk;

long long dp(int n, int k) {
	if (n == 0) return 1;
	if (n < k) return 0;
	if (DP[n][k] != -1) return DP[n][k];
	long long count = 0;
	for (int i = k; i <= n; ++i)
		count += dp(n - i, i);
	return DP[n][k] = count;
}

int main()
{
	int tc; cin >> tc;
	while (tc--) {
		cin >> n >> kk;
		memset(DP, -1, sizeof(DP));
		cout << dp(n, kk) << endl;
	}

    return 0;
}
