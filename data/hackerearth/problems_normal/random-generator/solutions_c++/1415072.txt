#include <iostream>
#include<cstdio>
#include<cstdlib>
#include<vector>
#include<algorithm>
#include<map>

using namespace std;

//#define MIN -1000000000

int main()
{
	
	int t;
	cin>>t;
	while(t--)
	{
		int n,k,p,i,j;
		cin>>n>>k>>p;
		vector<int> randNums;
		
		for(i=0;i<n;i++)
		{
			cin>>j;
			randNums.push_back(j);
		}
		sort(randNums.begin(),randNums.end());
		
		for(i=0;i<n;i++)
		{
			vector<int>:: iterator up;
			
			
			up=upper_bound(randNums.begin(),randNums.end(),randNums[i]+2*p);
			
			int index=up-randNums.begin();
			
			if((index-i)>=k)
			{
				cout<<"NO"<<endl;
				break;
			}
			
		}
		if(i==n)
		{
			cout<<"YES"<<endl;
		}
		
	}
	
	
    
    
    return 0;
}
