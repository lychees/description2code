#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while(t--) {
		string s,p;
		cin >> s >> p;
		int i=0,j=0;
		for (;i<s.length() && j<p.length();) {
			if(s[i]==p[j]) {
				i++;
				j++;
			}
			else i++;
		}
		if(j==p.length()) {
			puts("YES");
		}
		else puts("NO");
	}
}
