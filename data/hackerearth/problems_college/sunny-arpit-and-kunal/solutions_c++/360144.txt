#include <iostream>
using namespace std;

int main()
{
 int t,a,b,c,d1,d2,ans,aa,bb,cc;
 cin>>t;
 while(t)
 {
  cin>>aa>>bb>>cc;
  ans=0;
  if(aa>bb)
  {
  	b=aa;
    a=bb;
  }
  else
  {
  	b=bb;
  	a=aa;
  }
  if(b>cc)
  {
  	c=b;
  	if(a>cc)
  	{
  		b=a;
  		a=cc;
  	}
  	else
  	{
  		b=cc;
  	}
  }
  else
   c=cc;
  d1=b-a;
  d2=c-b;
  if(d1!=1&&d2==1)
   ans=d1-1;
  else if(d1==1&&d2!=1)
   ans=d2-1;
  else if(d1!=1&&d2!=1)
  {
   if(d1>d2)
    ans=d1-1;
   else
    ans=d2-1;
  }
  cout<<ans<<"\n";
  t--;
 }
 return 0;
}
