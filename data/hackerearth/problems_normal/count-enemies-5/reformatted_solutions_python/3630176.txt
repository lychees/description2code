n = int(input())
for _ in range(n):
    count = 0
    s = raw_input()
    length = len(s)
    l = 0
    r = 0
    while True:
        kon = False
        while r < length and s[r] != '*':
            if s[r] == 'X':
                kon = True
            r += 1
        if r > length:
            break
        if not kon:
            count += (r-l)
        r += 1
        l = r
    print(count)
