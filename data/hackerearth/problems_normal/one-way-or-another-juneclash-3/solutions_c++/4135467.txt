#include <bits/stdc++.h>
using namespace std;
void solve() {
	int n;
	cin >> n;
	long long x1, y1, s1;
	cin >> x1 >> y1 >> s1;
	bool ans = true;
	bool pr = ( ( (x1 + y1) % 2 ) == (s1 % 2) );
	for (int i = 2; i <= n; i++) {
		long long x, y, s;
		cin >> x >> y >> s;
		if ( pr != ( ( (x + y) % 2 ) == (s % 2) ) ) {
			ans = false;
		}
	}
	if (ans) {
		cout << "Yes" << endl;
	}
	else {
		cout << "No" << endl;
	}
}
int main()
{
    int test;
    cin >> test;
    for (int i = 0; i < test; i++) {
    	solve();
    }
    return 0;
}
