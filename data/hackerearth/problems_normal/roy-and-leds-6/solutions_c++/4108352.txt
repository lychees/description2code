#include<iostream>
using namespace std;
int main()
{int t,r,g,b,redcount=0,greencount=0,bluecount=0,yellowcount=0,cyancount=0,magentacount=0,whitecount=0,blackcount=1,rcond=0,bcond=0,gcond=0,rr=0,bb=0,gg=0;
 cin>>t>>r>>g>>b;
 for(int i=1;i<t;i++)
  {if(i%r==0)
    {rr++;
     if(rr%2==0)
      rcond=0;
     if(rr%2==1)
	  rcond=1; 
	}
   if(i%g==0)
    {gg++;
     if(gg%2==0)
      gcond=0;
     if(gg%2==1)
	  gcond=1; 
	}
   if(i%b==0)
    {bb++;
     if(bb%2==0)
      bcond=0;
     if(bb%2==1)
	  bcond=1; 
	 //cout<<"yea"; 
	}
   //cout<<rcond<<gcond<<bcond<<endl;    	
   if(rcond==0 && gcond==0 && bcond==0)
    blackcount++;
   else if(rcond==1 && gcond==0 && bcond==0)
    redcount++;
   else if(rcond==0 && gcond==1 && bcond==0)
    greencount++;
   else if(rcond==0 && gcond==0 && bcond==1)
    bluecount++;
   else if(rcond==1 && gcond==1 && bcond==0)
    yellowcount++;
   else if(rcond==0 && gcond==1 && bcond==1)
    cyancount++;
   else if(rcond==1 && gcond==0 && bcond==1)
    magentacount++;
   else if(rcond==1 && gcond==1 && bcond==1)
    whitecount++;    				
  }
  cout<<redcount<<" "<<greencount<<" "<<bluecount<<" "<<yellowcount<<" "<<cyancount<<" "<<magentacount<<" "<<whitecount<<" "<<blackcount<<" "<<endl;
  return 0;
 
}
