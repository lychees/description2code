#include <iostream>
#define sc(n) scanf("%lld",&n)
using namespace std;

int main()
{
    long int t;
    cin>>t;
    while(t--)
    {
    	long long l,b,h;
    	sc(l);
    	sc(b);
    	sc(h);
    	double r;
        l=min(l,b);
        l=min(l,h);
        r=(l);
        r=double(r/2);
        printf("%0.1lf\n",r);
    	
    }
    return 0;
}
