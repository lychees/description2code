#include "iostream"
#include "string"
#include "algorithm"
using namespace std;
int main(int argc, char *argv[])
{
  
    int n,evencount=0,oddcount=0,ary1[1000],ary2[1000],ary3[1000],evenoccur=0,oddoccur=0,ary4[1000];
    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>ary1[i];
    }
    sort(ary1,ary1+n);
    for(int j=0;j<n;j++)
    {
        if(ary1[j]%2==0)
        {

            ary2[evenoccur++]=ary1[j];
            evencount+=ary1[j];
        }
        else
        {

            ary3[oddoccur++]=ary1[j];
            oddcount+=ary1[j];
        }
    }

    for(int i=0;i<evenoccur;i++)
    {
        ary4[i]=ary2[i];
    }
    ary4[evenoccur++]=evencount;

    for(int i=0,j=evenoccur;i<oddoccur;i++,j++)
    {
        ary4[j]=ary3[i];
    }
    ary4[n+1]=oddcount;

    for(int i=0;i<n+2;i++)
    {
        cout<<ary4[i]<<" ";
    }

}
