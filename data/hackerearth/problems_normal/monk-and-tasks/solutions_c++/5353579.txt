#include<bits/stdc++.h>
#define ll long long
using namespace std;
int count_set_bits(ll n)
{
	int count=0;
	while(n)
	{
		n=n&(n-1);
		count++;
	}
	return count;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		vector<ll> v[64];
		ll x;
		int index;
		while(n--)
		{
			scanf("%lld",&x);
			index=count_set_bits(x);
			v[index].push_back(x);
		}
		int l;
		for(int i=0;i<64;i++)
		{
			l=v[i].size();
			for(int j=0;j<l;j++)
				printf("%lld ",v[i][j]);
		}
		cout<<endl;
	}
	return 0;
}
