import sys


def main():
    t = input()
    a = []
    if t < 1 or t > 10000:
        sys.exit()
    else:
        for i in range(0, t):
            a.append(input())
            if a[-1] < 1 or a[-1] > 1000000000:
                sys.exit()
        for i in range(0, t):
            if a[i] % 2 == 0:
                print a[i]
            else:
                print a[i]-1


if __name__ == '__main__':
    main()
