import math


def getChoc(x, m):
    if x < m:
        return 0
    g = int(math.ceil(x/m))
    ex = x % m
    ex += g
    return (g + getChoc(ex, m))


n = int(raw_input())
for k in range(0, n):
    x = raw_input()
    (n, c, m) = x.split(" ")
    n = int(n)
    c = int(c)
    m = int(m)
    ch = int(n/c)
    wr = getChoc(ch, m)
    print ch+wr
