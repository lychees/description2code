#include <iostream>

#define ll long long int

using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		ll n;
		cin>>n;
		ll cnt=0;
		ll d=1,p = n;
		while(p > 0)
		{
			cnt++;
			p /= 10;
			
		}
		//cout<<cnt<<endl;
		for(int i=1;i<cnt;i++)
		{
			d *= 10;
		}
		cnt--;
		ll diff = n-d+1;
		ll r=1,an=1,sum=0;
		for(int i=0;i<cnt;i++)
		{
			sum += (9 * r) * an;
			r *= 10;
			an++;
		}
		sum += (diff*an);
		cout<<sum<<endl;
	}
	return 0;
}
