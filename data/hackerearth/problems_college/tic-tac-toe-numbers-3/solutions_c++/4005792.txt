#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define pb push_back
#define vi vector<int>
#define vl vector<ll>
bool is_prime( ll n)
{
	ll i;
	for( i=2;(i*i)<=n;i++ )
	{
		if( (n%i)==0)
		{
			return false;
		}
	}
	return true;
}
int main()
{
    vi v;
    v.pb(1);
    v.pb(2);
    int i, t, n, a, b, c=0;
    a=1;
    b=2;
    do
    {
    	c = a+b;
    	a = b;
    	b = c;
    	v.pb( c);
    }while( c <= 10000000);
    vi::iterator it;
    vl::iterator sit;
    vi nf;
    vl sum;
    ll tot = 0;
    c = 1;
    i= 4;
    while( c <= 1000000)
    {
    	it = lower_bound( v.begin(), v.end(), i);
    	if((it==v.end())||((*it)>i))
    	{
    		c++;
    		nf.pb( i);
    		tot += i;
    		sum.pb( tot);
    	}
    	i++;
    }
    cin >> t;
    string ans;
    int pos;
    while( t--)
    {
    	cin >> n;
    	n--;
    	if( n < 4)
    	{
    		ans ="Tac";
    	}
    	else
    	{
    		it = lower_bound( nf.begin(), nf.end(), n);
    		if((*it)> n)
    		{
    			it--;
    		}
    		pos = it-nf.begin();
    		tot = sum[pos];
    		if( !(tot&1))
    		{
    			ans ="Tac";
    		}
    		else if( is_prime(tot))
    		{
    			ans ="Tic";
    		}
    		else
    			ans ="Toe";
    	}
    	cout << ans << "\n";
    }
    return 0;
}
