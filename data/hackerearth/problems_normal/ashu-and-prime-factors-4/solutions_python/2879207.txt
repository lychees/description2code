from collections import defaultdict

END = 10**6

primes = defaultdict(int)
numbers = defaultdict(int)

def removeMultiples(i):
	count = 0
	for k in xrange(i*i, END+1, i):
		if numbers[k] == 0:
			count += 1
			numbers[k] = 1
	primes[i] = count+1

def findPrimeNum():
	removeMultiples(2)
	for i in xrange(3, END+1, 2):
		if numbers[i] == 0:
			numbers[i] = 1
			removeMultiples(i)

if __name__ == "__main__":
	findPrimeNum()
	for _ in xrange(input()):
		print primes[input()]
