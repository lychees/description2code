#include <iostream>
using namespace std;

int main()
{
 int t;
 scanf("%d",&t);
 while(t--){
 	int x;
 	scanf("%d",&x);
 	if(x%3==0){
 		for(int i=0;i<x;i++)
 			printf("5");
 	}
 else if(x%5==0){

 		for(int i=0;i<x;i++)
 		printf("3");
 	}
 	else{
 		int count=0;
 		int temp=x;
 		while(temp%3!=0 && temp>0){
 			temp=temp-5;
 			count++;
 		}
 		if(temp<0){
 			printf("-1\n");
 		continue;
 		}
 		
 		
 		for(int i=0;i<temp;i++)
 			printf("5");
 		for(int i=0;i<x-temp;i++)
 			printf("3");
 	}
 	printf("\n");
 }
    return 0;
}
