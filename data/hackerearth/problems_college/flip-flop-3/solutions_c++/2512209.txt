#include <iostream>
#include<string.h>
using namespace std;

int main()
{
      long int i,len,del;
      string str;
      int n;
       cin>>n;
       del=0;
       
       while(n--)
       {
       	  cin>>str;
       	  for(i=0;i<str.length()-1;i++)
       	  {
       	  	if(str[i]=='X' && str[i+1]=='Y')
       	  	 continue;
       	  	else if(str[i]=='Y' && str[i+1]=='X')
       	  	 continue;
       	  	else
       	  	del+=1;
       	  }
       	  	cout<<del<<endl;
       	  	del=0;
       }
    return 0;
}
