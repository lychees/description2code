#include <iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int x1,y1,x2,y2,p1,q1,p2,q2;
		cin>>x1>>y1>>x2>>y2>>p1>>q1>>p2>>q2;
		if(x1>p2||x2<p1||y1>q2||y2<q1)
		{
			cout<<"Apart"<<endl;
		}
		else if(x1==p2&&y1==q2||(x2==p1&&y1==q2)||(x2==p1&&y2==q1)||(x1==p2&&y2==q1))
		{
			cout<<"Point"<<endl;
		}
		else if(x1==p2||x2==p1||y2==q1||y1==q2)
		{
			cout<<"Line"<<endl;
		}
		else
		{
			cout<<"Rectangle"<<endl;
		}
	}
	
}
