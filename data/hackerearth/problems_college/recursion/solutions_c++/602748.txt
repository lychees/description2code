#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,i;
    scanf("%d",&t);
    int arr[9];
    while(t--)
    {
    	scanf("%d",&n);
    	for(i=0;i<n;i++)
    		arr[i] = i+1;
    	do {
    		for(i=0;i<n;i++)
    			printf("%d",arr[i]);
    		printf(" ");
  		} while ( std::next_permutation(arr,arr+n) );
  		printf("\n");
    }
    return 0;
}
