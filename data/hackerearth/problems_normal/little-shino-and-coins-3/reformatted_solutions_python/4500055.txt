import sys

L = map(str, sys.stdin.read().split())
K = int(L[0])
result = 0


for y in xrange(len(L[1])-K+1):
    res = []
    temp = 0
    for x in xrange(y, len(L[1])):
        if not L[1][x] in res:
            res.append(L[1][x])
            temp += 1

        if temp == K:
            result += 1
        elif temp > K:
            break


print result
