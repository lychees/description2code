
#include<iostream>
using namespace std;
#define max 100000
int main()
{
    long long tmp[max+3];
     tmp[1]=1; tmp[2]=1;
    for(int i=3;i<=max;i++)
    {
        tmp[i]=tmp[i-1]+(i-1)+((i-1)%2);
    }
	int t;
	cin>>t;
	 for(int i=0;i<t;i++)
	{
		int n;
		cin>>n;
		cout<<tmp[n]<<endl;
	}
}
