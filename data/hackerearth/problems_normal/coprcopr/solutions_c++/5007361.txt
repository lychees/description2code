#include<bits/stdc++.h>
#define MAX 10000000
using namespace std;

int prime[MAX+1];
int phi[MAX+1];
long long s[MAX+1];

int main()
{
	
	prime[1]=1;
	for(int i=2;i<=(MAX);i++)
	{
		if(prime[i]==0)
		{
			for(int j=i*2;j<=MAX;j+=i)
				prime[j]=i;
		}
	}
	
	phi[1]=1;
	for(int i=2;i<=MAX;i++)
	{
		int res=i;
		int val=prime[res];
		if(val==0)
			phi[i]=res-1;
		else
		{
			while(res%val==0)
				res/=val;
			phi[i]=phi[res]*(i/res)/val*(val-1);
		}
	}
	
	for(int i=1;i<=MAX;i++)
		s[i]=s[i-1]+phi[i];
		
	int t;
	cin >> t;
	while(t--)
	{
		int n;
		cin >> n;
		cout << s[n] << endl;
	}
	return 0;
}
