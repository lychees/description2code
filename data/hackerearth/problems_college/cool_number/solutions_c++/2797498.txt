#include <bits/stdc++.h>
using namespace std;

int main()
{
 	int n,t,r,p,k,i,j,total;
 	scanf("%d",&t);
 	while(t--){
 		scanf("%d%d",&r,&k);
 		total=0;
 		for(i=5;i<=r;i++){
 			p=0;
 			j=i;
 			while(j>=5){
 				if(1&j && !(2&j) && 4&j)
 					p++;
 				j= j>>1;
 			}
 			if(p>=k)
 				total++;
 			
 		}
 		printf("%d\n",total);
 	}
    return 0;
}
