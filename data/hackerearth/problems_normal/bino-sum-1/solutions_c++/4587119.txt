#include <bits/stdc++.h>
using namespace std;
#define m 1000000007
long long int ncr[1001][1001];
void NCR()
{
	
	for(int i=0;i<=1000;i++)
	{
		ncr[i][i]=1;
		ncr[0][i]=0;
		ncr[i][0]=1;
	}
	for(int i=1;i<=1000;i++)
	{
		for(int j=1;j<=i;j++)
		{
			ncr[i][j]=((ncr[i-1][j]%m)+(ncr[i-1][j-1]%m)%m);
		}
	}
	
}


int main()
{
	int t;
	cin >> t;
	NCR();
	while(t--){
	int n,p;
	cin >> n >> p;
	long long ans=0;
	for(int i=0;i<=p;i++)
	{
		ans=((ans%m)+(ncr[n][i]%m))%m;
	}
	cout << ans%m << endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
