#include <iostream>
using namespace std;

int main()
{
    int t,n;
    int count1,count2;
    cin>>t;
    while(t--){
    	long long sum1=0,sum2=0;
    	int i,j;
    	cin>>n;
    	int a[n];
    	i=0;
    	j=n-1;
    	count1=0;
    	count2=0;
    	for(int i=0;i<n;i++){
    		cin>>a[i];
    	}
    	while(i<=j){
    		if(sum1==2*sum2){
    			if(i==j){
    				sum1+=a[i++];
    				count1++;
    			}
    			else{
    				sum1+=a[i++];
    				sum2+=a[j--];
    				count1++;
    				count2++;
    			}
    		}
    		else if(sum1<2*sum2){
    			sum1+=a[i++];
    			count1++;
    		}
    		else{
    			sum2+=a[j--];
    			count2++;
    		}
    	}
    	cout<<count1<<" "<<count2<<endl;
    }
    return 0;
}
