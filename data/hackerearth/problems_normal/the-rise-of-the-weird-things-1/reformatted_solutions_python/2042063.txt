import sys
n = input()
a = []
b = []
c = []
x = y = 0
p = q = 0
a = map(int, raw_input().split())
for i in range(0, n):
    if (a[i] % 2):
        c.append(a[i])
        p += 1
        x += a[i]
    if (a[i] % 2 == 0):
        b.append(a[i])
        q += 1
        y += a[i]
c.sort()
b.sort()
print ' '.join(str(k) for k in b), y, ' '.join(str(k) for k in c), x
