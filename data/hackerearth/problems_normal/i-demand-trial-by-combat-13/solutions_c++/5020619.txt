#include <iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	int n,m;
	for (int z=0;z<t;z++)
	{
		cin>>n>>m;
		int a[n+2],b[n+2];
		a[0]=1;
		a[n+1]=1;
		b[0]=1;
		b[n+1]=1;
		bool flag = false;
		for (int i=1;i<n+1;i++)
			cin>>a[i];
		for (int i=0;i<m;i++)
		{
			flag = true;
			if (i%2==0)
			{
			for (int j=1;j<n+1;j++)
			{
				if ((a[j-1]==1)&&(a[j+1]==1))
				{
					b[j]=1;
					flag = false;
				}
				else
					b[j]=0;
			}
			}
			else
			{
			for (int j=1;j<n+1;j++)
			{
				if ((b[j-1]==1)&&(b[j+1]==1))
				{
					a[j]=1;
					flag = false;
				}
				else
					a[j]=0;
			}	
			}
			if (flag)
			{
				m=i+1;
				break;
			}
		}
		for (int i=1;i<n+1;i++)
		{	if (m%2==1)
			cout<<b[i]<<" ";
			else
			cout<<a[i]<<" ";
		}
		cout<<endl;
	}
}
