#include <bits/stdc++.h>
using namespace std;

bool prime[1000000];

void sieve(){
	memset(prime,true,1000000);
	prime[0]=prime[1]=false;
	for(int i=2;i*i<=1000000;i++){
		if(prime[i]){
			for(int j=2;i*j<=1000000;j++){
				prime[i*j]=false;
			}
		}
	}
}
int main()
{
    int tc;
    int n,m;
    sieve();
    cin>>tc;
	
	long long precomp[1000000];
	precomp[0]=precomp[1]=0;
	for(int i=2;i<=1000000;i++){
		if(prime[i]){
			precomp[i] = i+precomp[i-1];
		}
		else	precomp[i] = precomp[i-1];
	}
    while(tc--){
        long long flag=0;
    	cin>>n>>m;
    	//cout<<n<<endl;
    	cout<<precomp[m]-precomp[n-1]<<endl;
    }
    return 0;
}
