#include <cstdlib>
#include <iostream>


using namespace std;
string s;
char a;
int T,I,n,i,k;
int main(int argc, char *argv[])
{   cin>>T;
    for(I=1;I<=T;I++)
    {   cin>>s;
        n=s.length();
        a=s[0];
        i=0;
        k=0;
        for(i=1;i<n;i++)
           if(s[i]==a)
              k++;
           else
              a=s[i];
              
        cout<<k<<endl;
                        
    }
    
    return EXIT_SUCCESS;
}
