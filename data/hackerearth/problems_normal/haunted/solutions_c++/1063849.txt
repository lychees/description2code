#include<iostream>
#include<cstdlib>
#include<map>
#include<vector>

using namespace std;
typedef unsigned long long ull;

int main()
{
    map <int,int> mymap;
    ull m,n,first_ghost,ghost,max;
    map<int,int> :: iterator it,it1;
    cin>>n;
    cin>>m;
    cin>>first_ghost;
    mymap.insert(make_pair(first_ghost,1));
    cout<<first_ghost<<" "<<1<<endl;
    max = first_ghost;
    for(int i=1;i<n;i++)
    {
       cin>>ghost;//cout<<"ghost entered = "<<ghost<<endl;
       it = mymap.find(ghost);
       if(it == mymap.end())
       {//cout<<"new"<<endl;
           mymap.insert(make_pair(ghost,1));
           it1 = mymap.find(max);
           if((*it1).second == 1)
           {
               if(max > ghost)
                 cout<<max<<" "<<1<<endl;
                else
                {
                    cout<<ghost<<" "<<1<<endl;
                    max = ghost;
                }
           }
           else
            cout<<(*it1).first<<" "<<(*it1).second<<endl;
       }

       else
       {//cout<<"duplicate"<<endl;
            (*it).second++;
            it1 = mymap.find(max);
            if((*it1).second > (*it).second)
            {
                cout<<max<<" "<<(*it1).second<<endl;
            }
            else if((*it1).second == (*it).second)
            {
                if((*it1).first > ghost)
                    cout<<max<<" "<<(*it1).second<<endl;
                else
                {
                    cout<<ghost<<" "<<(*it).second<<endl;
                    max = ghost;
                }
            }
            else
            {
                cout<<ghost<<" "<<(*it).second<<endl;
                max = ghost;
            }
       }


    }
}
