#include <bits/stdc++.h>
#define ASCII_SIZE 256
using namespace std;
int main()
{
    string s;
    int a[ASCII_SIZE] = {0};
    int sum = 0,i;
    while (getline(cin,s)) 
    {
    	for(i=0;i<s.length();i++)
    	{
    		sum = sum + s[i];
    		a[s[i]]++;
    	}	
		sum = sum/s.length();
    	if(sum%2==0)
    	{
    		string rev(s.rbegin(),s.rend());
    		cout << rev << endl;
    	}
    	else
    	{
    		int max = 0;
    		char c;
    		for(i=0;i<s.length();i++)
    		{
    			if(a[s[i]]>max)
    			{
    				max = a[s[i]];
    				c = s[i];
    			}
    		}
    		printf("%c\n",c);
    	}
	}
    return 0;
}
