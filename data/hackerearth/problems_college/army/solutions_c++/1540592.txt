int arr[100001] = {0};

int convert(VI p)
{
      int num = 0;
      int mul = 1;
      FOR(i,0,sz(p))
      {
            if(p[i] == 1)
                  num += mul;
            mul *= 2;
      }
      return num;
}

int main()
{
      int n;
      cin >> n;
      FOR(i,0,n+1)
      {
            int temp = i;
            VI p(10);
            while(temp)
            {
                  p[temp%10] = 1;
                  temp /= 10;
            }
            arr[i] = convert(p);
      }
      DRT()
      {

      }

}
