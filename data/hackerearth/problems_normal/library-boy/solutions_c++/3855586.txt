#include <iostream>
#include<string>
using namespace std;



int main()
{
	
	int arr[26];
	for(int i=0;i<26;i++)
	arr[i]=0;
	
	
	int n;
	cin>>n;
	string s;
	for(int i=0;i<n;i++){
		cin>>s;
		
		int index = s[0]-'a';
		arr[index]++;
		
		
	}
	int ans=0;
	
	for(int i=0;i<26;i++){
		if(arr[i]<=10 && arr[i]>0){
			ans++;
		}else{
			int val = arr[i];
			while(val>10){
				ans++;
				val-=10;
			}
			if(val>0)
			ans++;
		}
	}
	
	
	
	
    cout <<ans<< endl;
    return 0;
}
