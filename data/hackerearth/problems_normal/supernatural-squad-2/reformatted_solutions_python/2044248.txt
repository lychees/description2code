limit = int(raw_input())
ans = [[0 for x in range(222)] for x in range(222)]

n = 1
while n <= 200:
    ans[n][n] = 1
    k = n-1
    while k:
        ans[n][k] = ans[n-k][k]+ans[n][k+1]
        k = k - 1
    n = n + 1
while limit:
    n, k = map(int, raw_input().split())
    print ans[n][k]
    limit = limit - 1
