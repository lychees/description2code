    #include <bits/stdc++.h>
     
    using namespace std;
    int n,e;
    vector< vector<int> > graph;
    vector<bool> visited;
     
    void dfs(int x){
    visited[x]=true;
    for(int i=0;i<graph[x].size();i++){
    if(!visited[graph[x][i]])
    dfs(graph[x][i]);
    }
    }
     
    int main(int argc, char const *argv[]){
    int t,tt,a,b,ans;
    scanf("%d",&tt);
    for(t=0;t<tt;t++){
    graph.clear();
    scanf("%d",&n);
    graph.resize(n); visited.resize(n);
    for(int i=0;i<n;i++) visited[i]=false;
    scanf("%d",&e);
    for(int i=0;i<e;i++){	
    scanf("%d%d",&a,&b);
    graph[a].push_back(b);
    graph[b].push_back(a);
    }
    ans=0;
    for(int i=0;i<n;i++){
    if(!visited[i]){
    ans++;
    dfs(i);
    }
    }
    printf("%d\n",ans);
    }
    return 0;
    }
