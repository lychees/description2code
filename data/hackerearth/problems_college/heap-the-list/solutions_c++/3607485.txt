#include <bits/stdc++.h>
using namespace std;

int a[10005];

int main()
{
    int i;
	while(cin>>a[i]){
		int k=i;
		for(int j=i/2;;j/=2)
        {
            if( a[j]>a[k] )swap(a[j],a[k]);
            else break;
            k=j;
        }
        i++;
	}
//	for(int i=0;i<n;i++)printf("%d ",a[i]);
//	puts("");
	cout<<a[0]<<endl;
    return 0;
}
