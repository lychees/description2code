def findAll(a, i, N):
    if i == N-1:
        alle.append("".join(map(str, a)))
        return
    for j in range(i, N):
        temp = a[j]
        a[j] = a[i]
        a[i] = temp
        findAll(a, i+1, N)
        temp = a[j]
        a[j] = a[i]
        a[i] = temp


T = int(raw_input())
alle = []
while T > 0:
    alle = []
    N = int(raw_input())
    a = [i+1 for i in range(0, N)]
    findAll(a, 0, N)
    alle.sort()
    print " ".join(alle)
    T -= 1
