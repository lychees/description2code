#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
	int N, cb = 0; 
	while(cin >> N){int v[N]; cb = 0;
		for(int i = 0; i < N; i++)
			cin >> v[i];
		
		for(int i = N-1; i >= 0; i--)
			if(v[i]%2 == 0) {cout << v[i] << " "; cb++; }
		if(cb == 0) cout << "None to display";
		cout << endl;	
	}
	
    return 0;
}
