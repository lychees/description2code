#include <bits/stdc++.h>
using namespace std;

int main()
{
	long long int i,sum=0,n;
	cin>>n;
		stack<int>st;
	long long int arr[n];
	for(i=0;i<n;i++)
	{
		cin>>arr[i];
		if(!st.empty())
		{
			if(arr[i]!=0)
			st.push(arr[i]);
			else if(arr[i]==0)
			{
				st.pop();
			}
		}else
		{
			st.push(arr[i]);
		}
	}
	while(!st.empty())
	{
		sum=sum+st.top();
		st.pop();
	}
	cout<<sum;
    return 0;
}
