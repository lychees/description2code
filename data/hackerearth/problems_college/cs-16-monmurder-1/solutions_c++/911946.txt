#include <iostream>
#include <cstring>
using namespace std;
int main(int argc, char const *argv[])
{
	int t, h, x, y, z, A[1005];
	cin >> t;
	memset(A, 0, sizeof(A));
	while (t != 0) {
		cin >> h;
		x = 2;
		y = 0;
		while (h != 1) {
			if (h % x == 0) {
				A[y]++;
				h /= x;
				if (h % x != 0) y++;
				continue;
			}
			x++;
		}
		int i = 1;
		z = A[0];
		while (A[i] != 0) {
			z ^= A[i];
			i++;
		}
		if (z == 0) cout << "Pappu Bhujia"; else cout << "Printf{}";
		cout << endl;
		memset(A, 0, sizeof(A));
		t--;
	}
	return 0;
}
