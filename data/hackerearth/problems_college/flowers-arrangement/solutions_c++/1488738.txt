#include<iostream>
using namespace std;

int fun(char *s){

long long int i;
for(i=0;s[i];i++){
if(s[0]!='R')
return 0;
if(s[i]!='R' && s[i]!='Y')
return 0;
if(s[i]=='Y' && s[i+1]=='Y' && s[i+2]=='Y')
return 0;
}
return 1;
}
int main(){

long int t,p;
cin>>t;
char s[100000];

while(t--){

cin>>s;
p=fun(s);
if(p==1)
cout<<"YES"<<endl;
else
cout<<"NO"<<endl;
}

return 0;
}
