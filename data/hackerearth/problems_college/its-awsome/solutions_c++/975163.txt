#include <iostream>
using namespace std;

int main()
{
    int t;
    char str[20];
    scanf("%d",&t);
    while(t--){
    	int n = 0;
    	scanf("%s",str);
    	int i = 0;
    	while(str[i]!='\0'){
    		switch(str[i++]){
    			case '4':
    			case '7': n++;
    		}
    	}
    	if(n==4 || n==7) printf("YES\n");
    	else printf("NO\n");
    }
    return 0;
}
