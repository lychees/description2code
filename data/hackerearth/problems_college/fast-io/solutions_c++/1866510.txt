#include <iostream>
#include <algorithm>
using namespace std;
#define inp ios_base::sync_with_stdio(false)
#define LL long long

int main()
{
	inp;
    LL n,i,sum=0,count=0;
    cin >> n;
    LL a[n];
    
    for(i=0 ; i<n ; i++)
    cin >> a[i];
    
    sort(a,a+n);
    for(i=0 ; i<n ; i++)
    {
    	sum+=a[i];
    	if(sum<30000000)
    	count++;
    	else
    	break;
    }
    cout << count << endl;
    return 0;
}
