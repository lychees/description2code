#include <iostream>
#include <string>
using namespace std;

int main()
{
    ios::sync_with_stdio(false);
    string s;
    getline(cin,s);
    int n,flag=0;
    cin>>n;
    string m,min;
    cin.ignore(1,'\n');
    if(n>=1&&n<=100)
    {
    for(int i=0;i<n;i++)
    {
    	getline(cin,m);
    	if(m.length()>100){break;}
    	if((flag==0)&&(m.find(s)==0)){min=m;flag=1;}
    	else if((m.find(s)==0)&&(m<min)){min=m;}	
    }
    if(flag==0){min=s;}
    cout<<min;
    }
    return 0;
}
