#include <iostream>
#include<algorithm>
#include<cstdio>
#include<cstring>
using namespace std;

int main()
{
   int t,i,p=0,c=0;
   cin>>t;
   while(t--){
   	p=0;
   	c=0;
   	char s[100000];
   	scanf("%s",s);
   	if(s[0]!='R')
   	{
   		p=1;
   	}
   	for(i=1;s[i];i++){
   		if(s[i]=='Y'){
   			c++;
   			if(c==3){
   				p=1;
   				break;
   				
   			}
   		
   		}
   		else
   		c=0;
   		if(s[i]!='R'&&s[i]!='Y'){
   			p=1;
   			break;
   		}
   	
   	
   }
   if(p==1){
   	cout<<"NO"<<endl;
   	
   }
   else
   cout<<"YES"<<endl;
   }
    return 0;
}
