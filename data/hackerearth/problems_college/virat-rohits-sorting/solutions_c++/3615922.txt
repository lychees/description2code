#include <iostream>
#include<algorithm>
#include<cstdio>
#include<vector>
#include<utility>
using namespace std;

vector<pair<int,int> > vec;

bool myfunc(pair<int,int> a,pair<int,int> b)
{
	if(a.first < b.first)return true;
	else if(a.first == b.first)
	{
		if(a.second > b.second)return true;
		return false;
	}
	return false;
}

int main()
{
//    cout << "Hello World!" << endl;
	int n,c,r;
	scanf("%d",&n);
	while(n--)
	{
		scanf("%d%d",&c,&r);
		vec.push_back(make_pair(c,r));
	}
	sort(vec.begin(),vec.end(),myfunc);
	for(int i=0;i<vec.size();i++)printf("%d %d\n",vec[i].first,vec[i].second);
	vec.clear();
    return 0;
}
