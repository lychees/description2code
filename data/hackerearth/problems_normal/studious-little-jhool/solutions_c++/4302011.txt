#include <iostream>
#include <cstdio>
using namespace std;

int pages(int n,int count)
{
	int c1,c2;
	if(n==0)
		return count;
	if(n<0)
		return -1;
	c1 = pages(n-12,count+1);
	c2 = pages(n-10,count+1);
	if(c1!=-1 && c2!=-1)
		return min(c1,c2);
	else
		return max(c1,c2);
}
int main()
{
	int t,n;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		printf("%d\n",pages(n,0));
		
	}
}
