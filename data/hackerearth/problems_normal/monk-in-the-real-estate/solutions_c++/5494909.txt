#include<iostream>
#include<vector>
#include<set>
using namespace std;
int main(){
	int t;
	vector<int>v[10001];
		set<int>s;
	 int e,x,y,i,p;
	cin>>t;
	while(t--){
	
		p=0;
		cin>>e;
		for(i=0;i<e;i++){
			cin>>x>>y;
			if(x>p||y>p)
			p=max(x,y);
			v[x].push_back(y);
			v[y].push_back(x);
		}
        for(i=0;i<=p;i++){
        	if(v[i].size()>0){
        	for(vector<int>::iterator it=v[i].begin();it!=v[i].end();++it){
        		s.insert(*it);
			}	
			}
			v[i].clear();
		}
		cout<<s.size()<<endl;
		s.clear();
	}
	return 0;
}
