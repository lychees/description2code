#include <bits/stdc++.h>
using namespace std;

typedef long long int ll;
typedef unsigned long long int ull;
#define M 1000000007
const int INF = (int) 1e9;
const int MAXN = (int) 1e5 + 10;

int main()
{
    //freopen("input.txt","r",stdin);
    int n,m,a[105],b[103],time;
    ios_base::sync_with_stdio(0);cin.tie(0);
    cin>>n>>m;
    for(int i=0;i<n;i++)
    {
        cin>>a[i];
    }
    for(int i=0;i<m;i++)
    {
        cin>>b[i];
    }
    sort(a,a+n);
    sort(b,b+m);
    time=a[0]*2;
    time=max(time,a[n-1]);
    if(time<b[0])
        cout<<time<<endl;
    else cout<<"-1\n";
    return 0;
}
