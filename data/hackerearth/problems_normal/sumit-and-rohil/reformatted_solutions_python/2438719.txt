def add(s, dc):
    k = list(dc.keys())
    f = 1
    for i in range(len(k)):
        s1 = k[i]
        if sorted(s1) == sorted(
                s) and s1[0] == s[0] and s1[len(s1)-1] == s[len(s)-1]:
            dc[k[i]] = s
            f = 0
            break
    if f == 1:
        dc[s] = ''


n = int(input())
dc = {}
while n > 0:
    s = raw_input()
    add(s, dc)
    n = n-1
print len(dc)
