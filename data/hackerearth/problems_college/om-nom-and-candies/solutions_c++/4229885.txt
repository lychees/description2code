#include <iostream>
#include <string>
#include <cstdio>
#include <cmath>
#include <algorithm>
typedef long long int ll;
using namespace std;
int main()
{
     ll N,A[100002],temp,j,i,C[100002];
     cin>>N;
     for(i=1;i<=N;i++) cin>>A[i];
     C[1]=A[1];
     for(i=2;i<=N;i++)
     {
         temp=A[1];
         for(j=2;j<=sqrt(i);j++)
         {
             if(i%j==0)
             {
                 temp=max(temp,C[j]);
                 temp=max(temp,C[i/j]);
             }
         }
        
         C[i]=A[i]+temp;
     }
     for(i=1;i<=N;i++) cout<<C[i]<<" ";
     cout<<endl;
     return 0;
}
