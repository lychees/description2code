dicts = {}
nos = input()
for _ in range(nos):
    ins = set(list(raw_input()))
    for k in ins:
        if dicts.has_key(k):
            dicts[k] +=1
        else:
            dicts[k] = 1
cnt = 0        
for i, v in dicts.items():
    if v == nos:
        cnt += 1
print cnt
