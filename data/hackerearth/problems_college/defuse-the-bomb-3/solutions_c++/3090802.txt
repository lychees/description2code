#include<iostream>
#include<string.h>
#include<stdlib.h>
#include<algorithm>
using namespace std;
 
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long a,b;
		cin>>a>>b;
		char str[100];
		int lenb,lena;
		sprintf(str,"%ld",b);
		lenb = strlen(str);
		sprintf(str,"%ld",a);
		lena = strlen(str);
		for(int i=0;i<=lena-lenb;i++)
		{
			long long no=0;
			for(int j=0;j<lenb;j++)
			{
				no = no*10 + str[j+i]-'0';
			}
			cout<<no*b<<endl;
		}
	}
	return 0;
}
