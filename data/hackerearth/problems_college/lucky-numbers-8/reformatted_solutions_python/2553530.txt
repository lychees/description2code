t = int(raw_input())
while t > 0:
    n = int(raw_input())
    a = [int(x) for x in raw_input().split()]
# print(a)
# print(len(a))
    max1 = a[n-1]
# print(max1)
    c = 1
    for i in range(n-2, -1, -1):
        # print(str(a[i])+" "+str(max1))
        if a[i] > max1:
            c = c+1
            max1 = a[i]
    print(c)
    t = t-1
