#include<iostream>
#include<math.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long n,i;
		cin>>n;
		long long int sum=0;
	
	
		for(i=2;i<=sqrt(n);i++)
		{
				if(n%i==0)
{
if(i%4!=0)
sum+=i;
if(i*i!=n)
{
if((n/i)%4!=0)
sum+=(n/i);
}
}
			
			
		}
		if(n%4!=0)
		sum+=n;
		if(n!=1)
		sum+=1;
		
		sum*=8;
		cout<<sum<<endl;

	}
	
	
	
	
	
	
	
}
