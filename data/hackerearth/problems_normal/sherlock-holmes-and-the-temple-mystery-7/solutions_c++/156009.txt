#include<iostream>
using namespace std;
int k[102][10002]={{0}};
int a[1002];
int main()
 {int t,n,w,i,j,sum;
 cin>>t;
 while(t--)
   {cin>>n;
   sum=0;
    for(i=0;i<n;i++)
      {
      cin>>a[i];
      sum+=a[i];
     }
     
     if(sum%2)
        {cout<<"NO"<<endl;
        continue;
        }
    for(i=0;i<=n;i++)
      {for(j=0;j<=sum/2;j++)
        {k[i][j]=0;
		if(i==0)
           k[i][j]=0;
         else if(j==0)
           k[i][j]=1;
         else if(j-a[i-1]>=0 && k[i][j-a[i-1]]!=0 && k[i][j-a[i-1]]!=i+1 )
           {k[i][j]=i+1;
           }
        else
          k[i][j]=0;
         if(i!=0 && k[i-1][j])
            k[i][j]=k[i-1][j];
          //cout<<k[i][j]<<" ";
        }
        //cout<<endl;
      }
      if(k[n][sum/2]!=0)
        cout<<"YES"<<endl;
      else
       cout<<"NO"<<endl;
        
   }
 }
