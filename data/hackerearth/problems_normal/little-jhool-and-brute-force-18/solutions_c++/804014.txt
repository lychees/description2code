#include <bits/stdc++.h>
using namespace std;

int a[704981];
int main() 
{
	long long int t,i,j,r;
	scanf("%lld",&t);
	for(i=0;i<=704978;i++)
		a[i]=0;
	for(i=1;i<=86;i++)
	{
		for(j=i+1;j<=86;j++)
		{
			r=(int)pow(i,3)+(int)pow(j,3);
			if(r<=704978)
				a[r]++;
		}
	}
	while(t--)
	{
		long long int n,flag=0;
		scanf("%lld",&n);
		for(j=n-1;j>=1;j--)
		{
			if(a[j]>1)
			{
				flag=1;
				break;
			}
		}
		if(flag==0)
			printf("-1\n");
		else
			printf("%lld\n",j);
	}
	return 0;
}
