#include<bits/stdc++.h>

#define MAX 1000005
#define ll long long int
#define PI 3.14159265359
#define MOD 1000000007
#define SET(x,y) memset(x,y,sizeof(x));

using namespace std;

int a[105];

int main(){
    
    int n,i,first,last;

    ll c,p;

    scanf("%d",&n);

    c=0;
    for(i=0;i<n;i++){
    	scanf("%d",&a[i]);

    	if(a[i]==0) c++;
    }

    i=0;

    while(a[i]!=1 && i<n)
    	i++;

    first=i;

    i=n-1;

    while(a[i]!=1 && i>=0)
    	i--;

    last=i;

    if(c==n) cout<<"0\n";
    else{

    c=1;p=1;

    for(i=first;i<=last;i++){

    	if(a[i]==0)
    		c++;
    	else{
    		p*=c;
    		c=1;
    	}
    }

    printf("%lld\n",p);
  }
	return 0;
}
