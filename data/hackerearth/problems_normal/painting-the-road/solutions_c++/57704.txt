#include<stdio.h>
#include<stdlib.h>
#include<iostream>
#include<algorithm>
#include<string.h>
using namespace std;
#define INF 99999;

long long int dist[300][300];
bool reach[300][300];


long long int input()
{
    int l,n;
    cin>>l>>n;
    int a,b,c;
    
    for(int i=0;i<=l;i++)
    {
    for(int j=0;j<=l;j++)
    {
      if(i==j)
      dist[i][j]=0,reach[i][j]=true;
      else
      {
        reach[i][j]=false;
        dist[i][j]=9999999999LL;
        }
      }
      }
    
    for(int i=0;i<n;i++)
    {
        cin>>a>>b>>c;
        for(int i=a;i<=b;i++)
        dist[a][i]=min((long long int)c,dist[a][i]),reach[a][i]=true;
        
        
    }
    
     for (int k = 0; k <=l; k++)
    {
        for (int i = 0; i <=l; i++)
        {
            
            for (int j = 0; j <=l; j++)
            {
                
              
              reach[i][j] = reach[i][j] || (reach[i][k] && reach[k][j]);

                if (dist[i][k] + dist[k][j] < dist[i][j])
                   {

                       dist[i][j] = dist[i][k] + dist[k][j];
                   }
            }
        }
    }
    
    if(reach[0][l]==true)
    return dist[0][l];
    else
    return -1;
    
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    cout<<input()<<endl;
}
