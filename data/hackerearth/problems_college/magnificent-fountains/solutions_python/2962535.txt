from fractions import gcd
T=int(raw_input())
while T!=0:
    n=int(raw_input())
    str=map(int,raw_input().split())
    fir=str[0]
    for ele in range(1,n):
        fir=(fir*str[ele])/gcd(fir,str[ele])
    fir=fir%1000000007
    print fir
    T-=1
