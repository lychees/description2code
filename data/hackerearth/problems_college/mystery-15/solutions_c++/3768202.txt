#include <bits/stdc++.h>

typedef long long LL;

#define PII       pair<int,int>

#define pb        push_back

#define cin(x)    scanf("%d",&x)
#define MOD		1000000007

int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		int k;
		cin >> k;
		int s = 0;
		for(int i = 0; i < 3; i++)
		{
			int x;
			cin >> x;
