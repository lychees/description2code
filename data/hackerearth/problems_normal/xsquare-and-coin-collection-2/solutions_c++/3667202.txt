#define MOD 1000000007
#define MAX 2000005
#define MAXN 112345
ll t,n,k,h[MAXN],dp[MAXN];
ll f(int idx){
	if(idx>=n)
	return 0;
	if(dp[idx]!=-1)
	return dp[idx];
	ll ans=0;
	if(h[idx]<=k)
	ans=max(ans,h[idx]+f(idx+1));
	return dp[idx]=ans;
}
int main(){
	sll(t);
	while(t--){
		sll(n),sll(k);
		rep(i,n)
		{
			sll(h[i]);
			//cout<<h[i]<<" ";
			dp[i]=-1;
		}
		//cout<<endl;
		dp[n]=-1;
		ll cnt=0;
		for(int i=0;i<n;++i){
			if(h[i]<=k)
			cnt=max(cnt,h[i]+f(i+1));
		}
		//cout<<endl;
		cout<<cnt<<endl;
	}

return 0;}
