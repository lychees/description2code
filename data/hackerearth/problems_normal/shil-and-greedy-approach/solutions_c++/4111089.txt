set<pair<int,int> > s;
set<pair<int,int> >::iterator it; 

int flip(int x,int i)
{
	if((x&(1<<i))==0)
	{
		//ith bit is 0
		x=x|(1<<i);
	}
	else
	{
		//ith bit is 1
		x=x^(1<<i);
	}
	return x;
}

int djikstra(int u)
{
	int i,v;
	d[u]=0;
	s.insert(make_pair(d[u],u));
	while(!s.empty())
	{
		it=s.begin();
		u=(*it).second;
		s.erase(it);
		//2-length subarrays
		for(i=0;i<=(n-2);i++)
		{
			v=u;
			//flip bits i and (i+1)
			v=flip(v,i);
			v=flip(v,i+1);
			if(d[v]>(d[u]+1))
			{
				if(d[v]!=maxVal)
					s.erase(s.find(make_pair(d[v],v)));
				d[v]=d[u]+1;
				s.insert(make_pair(d[v],v));
			}
		}
		//3-length subarrays
		for(i=0;i<=(n-3);i++)
		{
			v=u;
			//flip bits i, (i+1) and (i+2)
			v=flip(v,i);
			v=flip(v,i+1);
			v=flip(v,i+2);
			if(d[v]>(d[u]+1))
			{
				if(d[v]!=maxVal)
					s.erase(s.find(make_pair(d[v],v)));
				d[v]=d[u]+1;
				s.insert(make_pair(d[v],v));
			}
		}
	}
	return min(d[0],d[(1<<n)-1]);
}
