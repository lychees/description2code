def get_rem(p,ar):
	n=[]
	for i in ar:
		if i[p]=='N':
			n.append(i)
	return n
for i in range(int(raw_input())):
	n=input()
	st=raw_input().split()
	ar=[]
	for k in range(len(st[0])):
		p=k
		i=0
		c=1
		rem=st
		while(i<n):
			rem=get_rem(p%len(st[0]),rem)
			i=n-len(rem)
			if len(rem)>0:
				p+=1
				c+=1
		ar.append(c)
	print min(ar)
