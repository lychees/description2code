#include<bits/stdc++.h>
using namespace std;
#define MAX 1000006
int primes[MAX],lp[MAX];
int pr;
void siever(void)
{
	int i,j;
	pr=0;
	memset(lp,0,sizeof(lp));
	for(i=2;i<MAX;++i)
	{
		if(lp[i] == 0)
		{
			primes[pr++] = i;
			for(j=i;j<MAX;j+=i)
			{
				lp[j]=1;
			}
		}
	}
	
}
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    siever();
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int i;
    	int tmp;
    	int flag = -1;
    	int a,b;
    	int mm = INT_MIN;
    	int sr = sqrt(n);
    	for(i=0;i<pr && primes[i]<=n/2;++i)
    	{
    		
    		tmp = primes[i];
    		int res = n-tmp;
    		if(res == tmp) continue;
    		if(binary_search(primes,primes+pr,res))
    		{
    			a  = primes[i];
    			flag = 1;
    		}
    	}
    	if(flag == -1) puts("-1 -1");
    	else
    	{
    		
    		
    		cout<<a<<" "<<n-a<<endl;
    	}
    }
    return 0;
}
