#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>

using namespace std;

#define alc(a, n) memset((a), n, sizeof (a))
#define MX 100005
int cnt[MX];
int chk[MX];

int main() {
	
//	freopen("in.txt", "r", stdin);
//	freopen("out.txt", "w", stdout);
	
	int i, j, k, tn, a = 1, b = 1, x, y, c;
	chk[1] = 1;
	while (1) {
		c = a + b;
		b = a;
		a = c;
		if (c >= MX) break;
		chk[c] = 1;	
	}
	cnt[0] = 0;
	cnt[1] = 2;
	for (i = 2; i < MX; i++) cnt[i] = cnt[i - 1] + chk[i];
	
	for (scanf("%d", &tn); tn--; ) {
		scanf("%d%d", &x, &y);
		printf("%s\n", (cnt[y] - cnt[x - 1]) % 2 ? "INVALID" : "VALID");
	}
	
	return 0;	
}
