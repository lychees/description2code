from __future__ import print_function
t = 0
for _ in range(int(raw_input())):
    x, y = map(int, raw_input().split())
    t += 1
    c = 0
    k = 1
    while (True):
        if x+k == x ^ k:
            c += 1
        if c == y:
            break
        k += 1
    print("Case #%d:" % (t), k, end="")
    print("")
