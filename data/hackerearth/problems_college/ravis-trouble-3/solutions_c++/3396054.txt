int main()
{
	int t, n, m;
	S(t);

	while(t--) 
	{
		string a, b;
		cin >> a >> b;
		int i = 0, j = 0;
		int count = 0;
		while(i < a.size() && j < b.size()) {
			if(a[i] == b[j]) {
				i++;
				j++;
				count++;
				if(count == b.size()) {
					break;
				}
			}
			else {
				i++;
			}
		}
		if(count == b.size()) {
			cout << "YES" << endl;
		} else {
			cout << "NO" << endl;
		}
		
	}
	return 0;
}
