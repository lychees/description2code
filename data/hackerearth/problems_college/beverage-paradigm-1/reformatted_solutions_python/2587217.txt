import sys


def subset_sum(numbers, j, target, partial=[]):
    s = sum(partial)
    k = 1
    if s == target:
        print "True"
        sys.exit()
    if s >= target:
        return

    for i in range(len(numbers)):
        n = numbers[i]
        remaining = numbers[i+1:]
        if (i == j-1):
            print "False"
        subset_sum(remaining, j, target, partial + [n])
