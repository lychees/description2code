def find(s, ch):
    return [i for i, ltr in enumerate(s) if ltr == ch]


n, m = map(int, raw_input().split())
l = []
for _ in range(n):
    l.append(map(int, list(raw_input())))
s = []
for i in l:
    m = max(i)
    if i.count(max) > 1:
        s.extend(find(i, m))
    else:
        s.append(m)
print len(set(s))
