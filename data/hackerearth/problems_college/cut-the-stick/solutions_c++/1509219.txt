#include <algorithm>
#include <cassert>
#include <cstdio>
#include <cstring>
#include <iostream>
#include <map>
#include <set>
#include <string>
#include <vector>

using namespace std;

const int N = 10010;

int a[N];

int main() {
//    freopen("in", "r", stdin);
//    ios::sync_with_stdio(false);
//    cin.tie(nullptr);
    int tt;
    cin >> tt;
    while (tt--) {
        int len, n, k;
        cin >> len >> n >> k;
        for (int i = 1; i <= n; ++i) {
            cin >> a[i];
        }
        sort(a + 1, a + n + 1);
        int R = n;
        while (R > 0 && len - a[R] <= k) {
            --R;
        }
        int answer = 0;
        int L = 1;
        for (int i = 1; i <= n; ++i) {
            if (a[i] <= k) {
                while (L <= n && a[L] - a[i] <= k) {
                    ++L;
                }
                if (L < R) {
                    answer += n - i;
                } else {
                    answer += max(0, R - i) + (n - L + 1);
                }
            } else {
                answer += n - i;
            }
        }
        cout << answer << '\n';
    }
}
