#include <bits/stdc++.h>
using namespace std;
int main()
{
    long t;
    long long int n,count,sum,k,i,j,mod=1000000007;
    scanf("%ld",&t);
    while(t--){
    	sum=0;
    	scanf("%lld",&n);
    	for(i=2;i<=n;i*=2){
    		for(j=1;j<i;j*=2){
    			if((i+j)<=n){
    				sum=(sum+i+j)%mod;
    			}
    		}
    	}
    	printf("%lld\n",sum);
    }
    return 0;
}
