#include <iostream>
 
using namespace std;
 
void printgreatest(long N)
{
        long x=0,y=0,i=0,temp1=0,temp2=0;
        if((N<3)||(N==4))
        {
                cout<<"-1"<<endl;
                return;
        }
        if(N==3)
        {
                cout<<"555"<<endl;
                return;
        }
        if(N==5)
        {
                cout<<"33333"<<endl;
                return;
        }
        if(N>5)
        {
                if(N%3==0)
                {
                        for(i=0;i<N;i++)
                        {
                                cout<<"5";
                        }
                        cout<<endl;
                        return;
                }
                if((N+1)%3==0)
                {
                        temp1=N-5;
                        for(i=0;i<temp1;i++)
                        {
                                cout<<"5";
                        }
                        for(i=0;i<5;i++)
                        {
                                cout<<"3";
                        }
                        cout<<endl;
                        return;
                }
                if((N+2)%3==0)
                {
                        temp1=N-1;
                        temp2=1;
                        while(temp1>3)
                        {
                                temp1=temp1-3;
                                temp2=temp2+3;
                                if(temp2%5==0)
                                {
                                        for(i=0;i<temp1;i++)
                                        {
                                                cout<<"5";
                                        }
                                        for(i=0;i<temp2;i++)
                                        {
                                                cout<<"3";
                                        }
                                        cout<<endl;
                                        return;
                                }
                        }
                }
                if(N%5==0)
                {
                        for(i=0;i<N;i++)
                        {
                                cout<<"3";
                        }
                        cout<<endl;
                        return;
                }
                else
                {
                        cout<<"-1"<<endl;
                        return;
                }
        }
}
 
int main()
{
        int T;
        cin>>T;
        long N;
        for(int i=0;i<T;i++)
        {
                cin>>N;
                printgreatest(N);
        }
        return 0;
}
