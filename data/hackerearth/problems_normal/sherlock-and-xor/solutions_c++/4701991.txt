#include <iostream>

int main()
{
	int T;
	std::cin >> T;
	while (T--)
	{
		int N;
		std::cin >> N;

		long long *A = new long long[N];
		for (int i = 0; i < N; ++i)
			std::cin >> A[i];
			
		long long impair(0);
		long long pair(0);
		
		for (int o = 0; o < N; ++o)
			if (A[o] % 2 == 0)
				++pair;
			else
				++impair;
				

		std::cout << pair * impair << '\n';
	}
	return 0;
}
