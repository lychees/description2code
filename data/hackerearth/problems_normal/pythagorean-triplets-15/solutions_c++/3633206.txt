#define mp make_pair


typedef long long ll;
typedef long double ld;
typedef unsigned long long ull;
typedef pair<ll,ll> ii;



const int N=1e6+10;
ll cases,n,res,c[N],g;
vector<ll>v;

ll gcd(ll a,ll b)
{
	if(b==0)return a;
	return gcd(b,a%b);
}



int main()
{
	std::ios::sync_with_stdio(0);
	cin.tie(0);
	for(int i=0;i<N;i++)c[i]=0;
	for(ll i=1;i*i<N;i++)
	{
		for(ll j=i+1;i*i+j*j<N;j+=2)
		{
			g=gcd(i,j);
			if(g!=1)continue;
			c[i*i+j*j]++;
		}
	}
	for(int i=0;i<N;i++)
	{
		if(c[i])v.pb(i);
	}
	cin>>cases;
	for(int kase=1;kase<=cases;kase++)
	{
		cin>>n;
		res=0;
		for(int i=0;i<v.size()&&v[i]<=n;i++)
		{
			res+=(n/v[i])*c[v[i]];
		}
		cout<<res<<"\n";
	}
	return 0;
}
