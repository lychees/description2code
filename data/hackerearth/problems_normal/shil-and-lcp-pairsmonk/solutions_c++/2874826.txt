#include <cstdio>
#include <algorithm>
using namespace std;
struct trie_node{
	trie_node* next[27];
	long long int val[27];
};
trie_node* root;
char str[100100],query[100100];
long long int arr[100100];
int maxlen;
trie_node* create_node(){
	trie_node* root=new trie_node;
	for(int i=0;i<27;i++){
		root->next[i]=NULL;
		root->val[i]=0;
	}
}
void insert_str(){
	trie_node* temp=root;
	int i,k,pos;
	long long prev=0,sum=0;
	for(i=0;;i++){
		if(str[i]=='\0'){
			pos=26;			
		}
		else
			pos=str[i]-'a';
		if(temp->next[pos]==NULL){
			temp->next[pos]=create_node();
		}
		sum=0;
		for(k=0;k<27;k++){
			if(k!=pos)
				sum+=temp->val[k];
		}
		arr[i]+=sum;
		if(sum==0&&prev>0&&temp->val[pos]==0){
			arr[i]+=prev;
		}
		prev=temp->val[pos];
		temp->val[pos]=temp->val[pos]+1;
		temp=temp->next[pos];
		if(pos==26)
			break;
	}
	maxlen=max(i,maxlen);
	arr[i]+=prev;
}
int main(){
	int n,i,j;
	maxlen=0;
	root=create_node();
	for(i=1;i<=n;i++){
		arr[i]=0;	
	}
	scanf("%d",&n);
	for(i=0;i<n;i++){
		scanf("%s",str);
		insert_str();
	}
	for(i=0;i<=maxlen;i++){
		printf("%lld ",arr[i]);
	}
	printf("\n");
	return 0;
}
