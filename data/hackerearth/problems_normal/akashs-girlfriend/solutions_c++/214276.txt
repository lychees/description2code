#include<stdio.h>
#include<string.h>

#define M 1000000009
long long int power(long long int a, long long int n)
{
	if(n==0)	return 1;
	if(n==1)	return a%M;
	
	long long int ret = power(a,n/2);
	
	if(n%2==0 )	return (ret*ret)%M;
	else		return (((ret*a)%M)*ret)%M;
}
	
int main()
{
	int T;
	long long int N, x;
	
	scanf("%d",&T);
	while(T--)
	{
		scanf("%lld",&N);
		x = ( ( (power(10,N)-power(8,N)+M) % M ) * power(2,M-2) ) % M;
		printf("%lld\n",x);
	}
	return 0;
}
