#include <bits/stdc++.h>
using namespace std;

int main()
{
    int test;
    cin>>test;
    while(test--) {
    	int n;
    	cin>>n;
    	vector<int> c(n);
    	for(int i=0;i<n;i++) {
    		cin>>c[i];
    	}
    	long long sum=0;
    	for(int i=0;i<n;i++) {
    		sum+=c[i];
    	}
    	double res=(sum*1.0)/n;
    	cout.precision(8);
    	cout<<fixed<<res<<"\n";
    }
    return 0;
}
