#include<iostream>
#include<cstdio>
#include<vector>
#include<algorithm>
#include<map>
#define ll long long
using namespace std;
ll a[100005];
ll n,m,maxx=0;

bool func(ll x)
{
	ll sum=x,count=0;
	for(int i=0;i<n;i++)
	{
		if(sum<0)
		return 0;
		if(sum-a[i]>=0)
		{
			sum-=a[i];
	
		}
		else
		{
			count++;
			sum=x-a[i];
		}
	}
	if(sum)
	count++;
	if(count<=m)
	return 1;
	else
	return 0;
	
}

ll b_search()
{
	ll lo=maxx,hi=100000000000000000,mid;
	while(lo<hi)
	{
		mid=lo+(hi-lo)/2;
		if(func(mid))
		{
			hi=mid;
		}
		else
		lo=mid+1;
		
	}
	
	return lo;
}

int main()
{
	scanf("%lld%lld",&n,&m);
	for(int i=0;i<n;i++)
	{
		scanf("%lld",&a[i]);
		if(a[i]>maxx)
		maxx=a[i];
	}
	ll ans=b_search();
	printf("%lld\n",ans);	
	return 0;
}
