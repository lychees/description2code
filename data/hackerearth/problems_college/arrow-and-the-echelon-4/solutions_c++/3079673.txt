#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long int a,b;
        cin>>a;
        b=a;
        if(b<3)
            cout<<"-1"<<endl;
        else
        {
            while(b>3)
            {
                if(b%3==0)
                    break;
                else
                    b-=5;
            }
            if(b==0)
                {
                    for(long int i=0;i<a-b;i++)
                    cout<<"3";
                    cout<<endl;
                }
            else if(a-b==0)
            {
                for(long int i=0;i<b;i++)
                    cout<<"5";
                    cout<<endl;
            }
            else
            {
                for(long int i=0;i<b;i++)
                    cout<<"5";
                for(long int i=0;i<a-b;i++)
                    cout<<"3";
                cout<<endl;
            }
    	}
    }
    return 0;
}
