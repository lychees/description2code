#include <bits/stdc++.h>

using namespace std;

int main()
{
    int tc;
    cin >> tc;
    getchar();
    while(tc--){
    char a[10005];
    gets(a);
	
    char *ptr;
    ptr = strtok(a," ");

    vector <string> pr;

    while(ptr != NULL)
    {
        pr.push_back(ptr);
        ptr = strtok(NULL," ");
    }

    for (int i = pr.size()-1; i>= 0;i--)
        cout << pr[i] << " ";
    cout << endl;
    }

    return 0;
}
