#include <bits/stdc++.h>
using namespace std;

int main()
{
   
   
    	priority_queue<long long> pq;
    	long long x, y, z, n, num;
    	
	    cin>>n;
	    //cout<<n;
	    for(long long i=0; i<n; i++){
	    	//cout<<"dafsdf"<<endl;
	    	if(i>1){
	    		cin>>num;
	    		pq.push(num);
	    		x=pq.top(), pq.pop();
	    		y=pq.top(), pq.pop();
	    		z=pq.top(), pq.pop();
	    		cout<<x*y*z<<endl;
	    		pq.push(x), pq.push(y), pq.push(z);
	    	}
	    	else{
	    		cin>>num;
	    		pq.push(num);
	    		cout<<"-1"<<endl;
	    	}
	    }
    
    return 0;
}
