#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main() {
	int test;
	scanf("%d",&test);
	while(test--) {
		ll n,m,a;
		scanf("%lld %lld %lld",&n,&m,&a);
		ll ans = 0;
		for(ll i = 1;i*i <= a;i++) {
			if(a % i == 0) {
				ll d2 = a/i;
				if(i<n && d2<m) {
					ans +=(n -i)*(m - d2);
				}
				if(i!=d2) {
					if(i <m && d2 <n) {
						ans +=(m - i)*(n - d2);
					}	
				}
			}
		} 
		printf("%lld\n",ans);
	}

}
