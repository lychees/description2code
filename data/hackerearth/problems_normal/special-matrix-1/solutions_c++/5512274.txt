#include <iostream>
using namespace std;

int main()
{
    int t,x,y,i,j,n,ans;
    char a;
    cin>>t;
    while(t--)
    {
    	ans=0;
    	cin>>n;
    	for(i=0;i<n;i++)
    	for(j=0;j<n;j++)
    	{
    		cin>>a;
    		if(a=='*')
    		{
    			x=i;
    			y=j;
    		}
    	}
    	//cout<<x<<y<<n/2<<","<<n/2-y<<",";
    	if(x<=n/2)
    	ans+=n/2-x;
    	else 
    	ans+=x-n/2;
    	if(y<=n/2)
    	ans+=n/2-y;
    	else ans+=y-n/2;
    	cout<<ans<<endl;
    }
    return 0;
}
