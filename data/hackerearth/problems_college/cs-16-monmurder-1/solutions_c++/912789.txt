#include<bits/stdc++.h>
#define si(n)scanf("%d",&n)
#define pi(n)printf("%d\n",n)
#define sll(n)scanf("%lld",&n)
#define pll(n)printf("%lld\n",n)
using namespace std;using ll=long long;
const int N=1e6+50;
bool p[N];vector<int>primes;
void sieve(){
    p[0]=p[1]=false;
    for(int i=2;i<N;i++)p[i]=true;
    for(int i=2;i<N;i++){
        if(p[i]){
            primes.push_back(i);
        }
        for(int j=2*i;j<N;j+=i)p[j]=false;
    }
}
int main(){
    sieve();
    int t;si(t);
    while(t--){
        int h,x=0;si(h);
        while(h>1){
            for(int i=0;i<primes.size() && primes[i]<=h;i++){
                int tot=0;
                if(h%primes[i]==0){
                    while(h>1 && h%primes[i]==0){
                        h/=primes[i];tot++;
                    }
                    x^=tot;
                    break;
                }
            }
        }
        if(x)printf("Printf{}\n");
        else printf("Pappu Bhujia\n");
    }
    return 0;
}
