#include <bits/stdc++.h>
using namespace std;

int main()
{
    int a,b;
    scanf("%d%d",&a,&b);
    vector<int> pos(a,-1);
    bool adj[a][a];
    memset(adj,false,sizeof(adj));
    for(int i=0;i<b;i++){
    	int x,y;
    	scanf("%d%d",&x,&y);
    	adj[x][y]=1;
    	adj[y][x]=1;
    }
    pos[0]=0;
    queue<int> q;
    q.push(0);
    while(!q.empty()){
    	int node=q.front();
    	q.pop();
    	for(int i=0;i<a;i++){
    		if(adj[i][node] and pos[i]==-1){
    			pos[i]=pos[node]+1;
    			q.push(i);
    		}
    	}
    }
    for(int i=1;i<a;i++){
    	cout<<pos[i]<<endl;
    }
    return 0;
}
