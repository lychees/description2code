#include <iostream>

using namespace std;

int main()
{
	ios::sync_with_stdio( false );
	int n, i, s1 = 0, s2 = 0;
	string s;
	
	cin >> s;
	cin >> n;
	
	for( i = 0;i < n;i++ ) {
		
		string str;
		cin >> str;
		if( str.length() == s.length() ) {
		
			if( str[ 0 ] == ( (char)( s[ 0 ] + 3 ) ) ) 
				s2++;
			else if( str[ 0 ] == s[ 0 ] )
				s1++;
		}
	}
	
	cout << s1 << " " << s2 << endl;
	return 0;
}
