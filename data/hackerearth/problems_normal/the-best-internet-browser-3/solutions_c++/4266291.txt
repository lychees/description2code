#include <iostream>
using namespace std;
int main(){int t;cin>>t;
for(int k=0;k<t;k++){string s;cin>>s;
int l=s.size();int c=4;
for(int i=0;i<l;i++){
	if(s[i]=='a') c++;
	else if(s[i]=='e') c++;
	else if(s[i]=='i') c++;
	else if(s[i]=='o') c++;
	else if(s[i]=='u') c++;
}
cout<<l-c+1<<"/"<<l<<endl;
}
}
