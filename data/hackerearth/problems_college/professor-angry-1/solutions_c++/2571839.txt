#include<iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;

    while(t--)
    {
        int n,k;
        cin>>n>>k;

        int c=0;

        for(int i=0;i<n;i++)
        {
            int temp;
            cin>>temp;

            if(temp<=0)
                c++;
        }

        if(c>=k)
            cout<<"NO"<<endl;
        else
            cout<<"YES"<<endl;
    }
}
