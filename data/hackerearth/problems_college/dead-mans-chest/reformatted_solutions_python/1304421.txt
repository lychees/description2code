
for t in range(input()):
    string = raw_input()
    odd = 0
    dic = {}
    # count
    for char in string:
        if char in dic:
            dic[char] += 1
        else:
            dic[char] = 1
    for char in dic:
        if dic[char] % 2 == 1:
            odd += 1
    if odd % 2 == 1 or odd == 0:
        print 'JACK'
    else:
        print'DAVY JONES'
