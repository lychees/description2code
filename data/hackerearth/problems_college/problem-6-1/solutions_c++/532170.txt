#include <iostream>
using namespace std;
#define LL long long
#define M 1000000007
LL arr[30];
LL fact[500];
void f()
{
	fact[0]=1;
	for(int i=1;i<=400;++i)fact[i]=(fact[i-1]*i)%M;
}
LL n,s;
LL Bits(LL n)
{
   LL count = 0;
    while (n)
    {
      n&=(n-1) ;
      count++;
    }
    return count;
}
LL poww(LL a, LL n) {//rite
      LL ret=1;
      while(n){
            if(n%2)
            ret=(ret*a)%M;
            a=(a*a)%M;
            n/=2;
      }
      return ret;
}
 
LL fermats(LL a)
{
	return poww(a,M-2);
}
 
LL comb(LL n, LL m)
{
		n%=M;
      m=min(m,n-m);
      LL ret=1;
      for(int i=0;i<m;++i) {
            ret=(((ret*(n-i))%M)*fermats(i+1))%M;
      }
      return ret;
}
 
 
int main() {
	f();
	cin>>n>>s;
	LL anse=s;
      for(int i=0;i<n;++i){cin>>arr[i];arr[i]++;}
 	//cout<<comb(5,2)<<endl;
      LL ans = 0;
      for(int i=0;i<(1<<n);++i){
            for(int j=0;j<n;++j){
                  if (i & (1<<j)) {
                        anse-=(arr[j]);
                  }
            }
            int v=Bits(i);
            if (anse>=0){
            comb(anse + n - 1, n - 1);
            if (v%2) ans=(ans-comb(anse+n-1,n-1)+M)%M;
            else
            ans=(ans+comb(anse+n-1,n-1))%M;
            }
            anse=s;
      }
 
      cout << ans << endl;
	return 0;
}
