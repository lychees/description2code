#include <iostream>
#include <cstring>
using namespace std;

int main()
{
	bool flag1;
	char str1[10000],str2[10000];
	int t,i,j;
    cin>>t;
    while(t--)
    {
    	flag1=false;
    	cin>>str1>>str2;
    	for(i=0;i<strlen(str1);i++)
    	{
    		for(j=0;j<strlen(str2);j++)
    		{
    			if(str1[i]==str2[j])
    			{
    				flag1=true;
    			
    			
    			 break;
    			}
    		}
    	}
    	if(!flag1)
    	cout<<"NO"<<endl;
    	else
    	cout<<"YES"<<endl;
    }
    return 0;
}
