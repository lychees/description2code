#include <iostream>
using namespace std;

int a[1000001]={0};

void markmultiples(long i , long n){
	long j=2;
	while (i*j<n){
		a[i*j]=1;
		j++;
	}
}

void seive(long n){

	for(long i=2;i<n;i++){
		if(a[i]==0){
			markmultiples(i,n);
		}
	}
	
}

int check(long n){
	int x;
	while(n){
		x=n%10;
		if(x==7) return 1;
		
		n/=10;
	}
	return 0;
}
int main()
{
	
	seive(1000000);
	int t;
	cin>>t;
	while(t--){
		long m,n;
		cin>>m>>n;
		long ans=0;
		for(long i=m;i<=n;i++){
			if(a[i]!=0){
				if (check(i))
				ans+=1;
			}
		}
		
		if (ans==0) cout<<"-1"<<endl;
		else
			cout<<ans<<endl;
	}	
	return 0;
}
