#include<map>
#include <iostream>
using namespace std;

int main()
{
	map<char,int> m;
	int n,t,flag;
	string s;
	cin>>t;
	while(t--)
	{
		flag=0;
		cin>>n;
		while(n--)
		{
			cin>>s;
			if(!m[s[0]])
				m[s[0]]++;
			else
				flag=1;
		}
		if(flag)
			cout<<"NO"<<endl;
		else
			cout<<"YES"<<endl;
		m.clear();
	}
	return 0;
}
