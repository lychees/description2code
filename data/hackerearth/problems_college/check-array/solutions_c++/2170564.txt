#include<iostream>
#include<stdlib.h>
#include<stdio.h>
#include<vector>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		vector<int>v;
		int n,countR=0,x;
		cin>>n;
		for(int i=0;i<n;i++)
		{
			cin>>x;
			v.push_back(x);
			countR+=x;
		}
		int countL=0,flag=0;
		countL+=v[0];
		countR=countR-countL-v[1];
		
		for(int i=1;i<n;)
		{
			if(countL==countR)
			{
				flag=1;
				break;
			}
			countL+=v[i];
			i++;
			countR=countR-v[i];
		}
		if(flag)
			cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;
		
	}
}
