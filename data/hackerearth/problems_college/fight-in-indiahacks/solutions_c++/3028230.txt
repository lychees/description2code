#include <iostream>
using namespace std;
int dp[1000000];
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		dp[0]=0;
		int n,m1,m2;
		cin>>n>>m1>>m2;
		for(int i=1;i<=n;i++)
		{
			if(i>=m1 && !dp[i-m1])
			dp[i]=1;
			else if(i>=m2 && !dp[i-m2])
			dp[i]=1;
			else
			dp[i]=0;
		}
		if(dp[n])
		cout<<"1\n";
		else
		cout<<"0\n";
	}
    //cout << "Hello World!" << endl;
    return 0;
}
