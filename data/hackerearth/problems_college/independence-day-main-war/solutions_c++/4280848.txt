#include <iostream>
using namespace std;

int main()
{
	int T,N;
	float X,Y,x,y;
	
	cout<<fixed;
	cout.precision(2);
	
	cin>>T;
	while(T--){
		X=Y=0;
		cin>>N;
		float m = 1.0/N;
		for(int i=0;i<N;i++){
			cin>> x >> y;
			X+= m*x;
			Y+= m*y;
		}
		//cout<<fixed;
		//cout.precision(2);
		cout<<"("<<X<<" , "<<Y<<")\n";
	}
    return 0;
}
