#include<bits/stdc++.h>
//zone
//i m back
long long int dp[10002];
long long int a[10002];
using namespace std;
int main()
{
    int t;
    cin >> t;
    int n,k;
    int neg = 0;
    while(t--)
    {
        memset(dp,0,sizeof(dp));
        neg = 0;
        int i;
        cin >> n >> k;
        for( i = 1 ; i<= n ; i++)
        {
            cin >> a[i];
            if(a[i] < 0)
            {
            neg++;
            }
        }
        if(a[1] < 0)
        {
            dp[1] = 0;
        }
        else
            dp[1] = a[1];
        if(neg == n)
        {
            cout<<"0"<<endl;
            continue;
        }
        for(i = 2; i <= k + 1 ; i++)
        {
            dp[i] = max(dp[i-1],a[i]);
        }
        for(i =k + 2; i <= n ; i++)
        {
            dp[i] = max(dp[i - 1], dp[i - k - 1] + a[i]);
        }
        cout<<dp[n]<<endl;
    }
    return 0;
}
