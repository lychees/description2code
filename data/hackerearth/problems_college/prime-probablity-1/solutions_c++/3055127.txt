#include<bits/stdc++.h>
using namespace std;

#define lli long long
#define pb push_back
#define mp make_pair
#define gc getchar_unlocked
#define lld "%I64d"
#define DEBUG(x) cout<<">value ("<<#x<<") : "<<x<<endl;
#define mod 1000000007

const int lmt = 1000005;
bool check[lmt];
int cnt[lmt];
void seive(){
    check[0] = true;
    check[1] = true;
	for(int i=2;i<lmt;i++){
		if(!check[i]){
			for(int j=i+i;j<lmt;j+=i)
				check[j]=true;
		}
	}
	for(int i=1;i<lmt;i++)
        cnt[i] = cnt[i-1] + !check[i];
}


int main(){
   // freopen("out.txt","w",stdout);
    seive();
    int t,l,r;
    cin>>t;
    while(t--){
      cin>>l>>r;
      assert(l<=r);
      assert(r<=1000000);
      assert(1<=l);
      int num = (cnt[r]-cnt[l-1]);
      int den = (r-l+1);
      cout<<fixed;
      cout<<setprecision(6)<<(float)(num)/den<<endl;
    }
    return 0;
}
