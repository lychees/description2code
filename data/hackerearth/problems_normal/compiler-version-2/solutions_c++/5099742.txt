#include <iostream>
#include <string>
#include <algorithm>
#include <vector>
using namespace std;

int main()
{
    string s;
    int i=0,j,count=0,k;

    while(getline(cin,s) && !s.empty())
    {
    	i=0;
    	string str;
    	for(i=0;i<(s.length()-1);i++)
    	{
        	if(s[i]=='/' && s[i+1]=='/')
    		break;
    		
    	   else if(s[i]=='-' && s[i+1]=='>')
    		{
    			str.push_back('.');
    			i++;
    		}
    		else{
    			str.push_back(s[i]);
    		}
    			
    	}
    	for(;i<s.length();i++){
    		str.push_back(s[i]);
    	}
    	cout<<str<<endl;;
    }
    return 0;
}
