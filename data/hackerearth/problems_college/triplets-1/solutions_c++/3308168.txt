    #include <bits/stdc++.h>

    using namespace std;

    int main()
    {
        //freopen("in10.txt", "r", stdin);
        //freopen("out10.txt", "w", stdout);
        int x, y, z, i, ind1, ind2;

        cin >> x;
        vector <int> p(x);

        for (i = 0; i < x; ++i)
            cin >> p[i];

        cin >> y;
        vector <int> q(y);

        for (i = 0; i < y; ++i)
            cin >> q[i];

        cin >> z;
        vector <int> r(z);

        for (i = 0; i < z; ++i)
            cin >> r[i];

        sort(p.begin(), p.end());
        sort(q.begin(), q.end());
        sort(r.begin(), r.end());

        long long int ans = 0;

        for (i = 0; i < y; ++i) {
            ind1 = upper_bound(p.begin(), p.end(), q[i]) - p.begin();
            ind2 = r.end() - lower_bound(r.begin(), r.end(), q[i]);
            ans += (ind1 * 1LL * ind2);
        }

        cout << ans << endl;

        return 0;
    }
