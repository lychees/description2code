#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	long long int t,k,n,i,count;
	long long int a[100001];
	scanf("%lld",&t);
	
	while(t--)
	{
	scanf("%lld %lld",&n,&k);
		count = 0;
		for(i=0;i<n;i++)
		{
			scanf("%lld",&a[i]);
		
			if(a[i]==0)
			{
				count++;
			}
			
		}
		
		if(count>=k)
		{
			printf("%lld\n",n-count+k);
	
			//cout<<n<<endl;
		}
		else
		{
			if((k-count)%2==0)
			printf("%lld\n",n);
			else
			printf("%lld\n",n-1);
		}
	}
	
	return 0;
}
