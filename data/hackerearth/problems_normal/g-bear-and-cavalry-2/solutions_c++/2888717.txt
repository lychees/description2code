#include <bits/stdc++.h>
using namespace std;
int h[10001],w[10001];
int main()
{
	std::ios::sync_with_stdio(false);
	cin.tie(0);
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		for(int i=0;i<n;++i)
			cin>>w[i];
		for(int i=0;i<n;++i)
			cin>>h[i];
		sort(h,h+n,greater<int>());
		sort(w+1,w+n);
		bool flag=true;
		for(int i=1;i<n;++i)
		{
			if(w[i]*h[i]>=w[0]*h[0])
				flag=false;
		}
		if(flag)
			cout<<"YES\n";
		else
			cout<<"NO\n";
	}	 
}
