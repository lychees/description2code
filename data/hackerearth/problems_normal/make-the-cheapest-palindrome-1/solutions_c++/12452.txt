#define FORi(i,a,b) for (i=a;i<b;i++)
#define ROF(i,a,b) for(int i=a-1;i>=b;i--)

/****************************************************************************************************************************/

int main(){

int n;
scanf("%d\n",&n);
while(n--){
	string b;
	cin>>b;
	int acost=SI,bcost=SI;
	int cost=0;
	int i;
	FORi(i,0,SZ(b)){
		if(b[i]!='/'&&b[SZ(b)-1-i]!='/'&&b[i]!=b[SZ(b)-1-i]){
			break;
		}
		if(b[i]=='/'){
			if(b[SZ(b)-1-i]=='a'){
				cost+=acost;
				b[SZ(b)-1-i]='a';
			}
			else if(b[SZ(b)-1-i]=='b'){
				cost+=bcost;
				b[SZ(b)-1-i]='b';
			}
			else{
				if(min(acost,bcost)==acost){
					b[i]='a';
					b[SZ(b)-1-i]='a';
					cost+=(2*acost);
				}
				else{
					b[i]='b';
					b[SZ(b)-1-i]='b';
					cost+=(2*bcost);
				}
			}
		}
	}
	if(i!=SZ(b))
		printf("%d\n",-1);
	else
		printf("%d\n",cost);
}

return 0;
}
