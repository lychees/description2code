#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    char x;
    int y, i, j, temp;
    while(cin>>x&&cin>>y){
    	//cout<<x<<" "<<y<<endl;
    	if(y<2)
    		{cout<<"INVALID"<<endl<<endl;
    		continue;}
    	else if(int(x)>64&&int(x)<91)
    	{
    		if(int(x)+y>91)
    			{cout<<"INVALID"<<endl<<endl;
    			continue;}
    		
    		else{
    			
    			for(i=0;i<y;i++){
    				
    				for(j=0;j<i+1;j++){
    					temp=int(x)+i;	
    					cout<<char(temp)<<" ";
    					
    				}
    				cout<<endl;
    				
    			}
    			
    		}
    		
    	}
    	
    	else if(int(x)>='a'&&int(x)<='z'){
    		
    		if(int(x)+y>123)
    			{cout<<"INVALID"<<endl<<endl;
    			continue;}
    		
    		else{
    			
    			for(i=0;i<y;i++){
    				
    				for(j=0;j<i+1;j++){
    					temp=int(x)+i;	
    					cout<<char(temp)<<" ";
    					
    				}
    				cout<<endl;
    				
    			}
    			
    		}
    		
    		
    		
    		
    	}
    	
    	cout<<endl;
    	
    }
    
    return 0;
}
