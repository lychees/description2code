from sys import stdin
import math
hi = 10**9 + 7
fa = [1]
for i in xrange(1, 10**5 + 1):
    fa.append((i*fa[-1]) % hi)


def npr(n, r):
    num = fa[n]
    den = pow(fa[n-r], hi-2, hi)
    return (num * den) % hi


t = int(stdin.readline())
for _ in xrange(t):
    r, n = map(int, stdin.readline().split())
    print npr(n, r)
