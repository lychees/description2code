#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	bool b[501]={false};
	int c[501]={0};
	for(int ti=0;ti<t;ti++){
		int n;
		cin>>n;
		for(int ci=0;ci<501;ci++)
		{
			b[ci]=false;
		}
		for(int i=0;i<n;i++){
			int val;
			cin>>val;
			if(b[val]==false) {
				b[val]=true;
				c[val]++;
			}
		}
		
	}
	long long int ans=0;
	for(int i=0;i<501;i++){
		if(c[i]==t) ans=ans-i;
	}
    cout << ans<< endl;
    return 0;
}
