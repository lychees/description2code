#include <bits/stdc++.h>
#define ll long long int
#define gt(i) scanf("%lld",&i)
#define mod 1000000007
#define md 1000000006

using namespace std;

ll pw(ll a, ll b)
{
	ll ans = 1;
	while(b) {
		if(b % 2 != 0) {
			ans = ans * a;
			ans = ans % mod;
		}
		a = a * a;
		a = a % mod;
		b = b / 2;
	}
	return ans;
}

int main()
{
	ll i, j, k, n, m;
	gt(n); 
	string g;
	cin >> g;
	m = 0;
	for(i = 0; i < g.size(); i++) {
		m = (m * 10 + (ll)g[i] - 48) % md;
	}
	cout << pw(n, m) << endl;
}
