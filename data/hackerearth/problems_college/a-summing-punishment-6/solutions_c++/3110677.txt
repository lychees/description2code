#include <iostream>
using namespace std;
int main()
{
    int n;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++)
        cin>>a[i];
    int q;
    cin>>q;
    while(q--)
    {
        int x,y,s;
        s=0;
        cin>>x>>y;
        for(int i=x-1;i<y;i++)
            s+=a[i];
        cout<<s<<endl;
    }
    return 0;
}
