#include<iostream>
using namespace std;
int main()
{
       int t,n,i;
       cin>>t;
       while(t--)
       {
       	cin>>n;
       	int ct=0;
       	int ans=0;
       	  for(i=2;i*i<=n;i++)
       	  {
       	  	 ct=0;
                           
                     	while(n%i==0)
                     	{
	                     	n=n/i;
	                     	ct++;
	                     }
	                     ans=ans^ct;
  	       }
  	       if(n>1)
  	       ans=ans^1;
  	       if(ans)
			 cout<<"Printf{}"<<endl;
			 else
			 cout<<"Pappu Bhujia"<<endl;  
       	 
       }
	 return 0;	 
}
