#include <iostream>
using namespace std;

int main()
{
    string s;
    cin>>s;
    long t,a,b;
    cin>>t;
    for(long j=0;j<t;j++)
    {
    	cin>>a>>b;
    	a-=1;
    	b-=1;
    	a=a%(s.length());
    	b=b%(s.length());
    	if(s[a]==s[b])
    	cout<<"Yes"<<endl;
    	else
    	cout<<"No"<<endl;
    }
    return 0;
}
