#include<bits/stdc++.h>
using namespace std;
//static int ans[2500][2500][2500];
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		int n;
		long long ans=0;
                int a[3][2501];
		int a1[4];
		scanf("%d %d %d %d",&n,&a1[0],&a1[1],&a1[2]);
                //printf("%d\n",a1[0]);
		for(int i=0;i<=a1[0];i++){
			a[0][i]=i;
		}
//printf("%d\n",a1[0]);
		for(int i=0;i<=a1[1];i++){
			a[1][i]=i;
		}
//printf("%d\n",a1[0]);
		for(int i=0;i<=a1[2];i++){
			a[2][i]=i;
		}
//printf("%d\n",a1[0]);
		long long dp[4][n+1];
		for(int i=0;i<=3;i++)
		for(int j=0;j<=n;j++)
		   dp[i][j]=0;
		dp[0][0]=1;
//printf("%d\n",a1[0]);
		for(int i=1;i<=3;i++)
		   for(int k=0;k<=a1[i-1];k++)
			for(int j=0;j<=n;j++){
			//	printf("i=%d j=%d %d\n",i,j,k);
				if(j>=a[i-1][k])
				 dp[i][j]+=dp[i-1][j-a[i-1][k]];
			//	printf("bar");
			}
	    for(int i=0;i<=n;i++)
	       ans+=dp[3][i];
		printf("%lld\n",ans);
	}
	return 0;
}
