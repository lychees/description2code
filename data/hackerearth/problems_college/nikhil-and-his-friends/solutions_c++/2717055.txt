#include <iostream>

#include<cstdio>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,f=0,i,j;
    	long long int x;
    	unsigned long long sum=0;
    	cin>>n;
    	cin>>x;
    	long long arr[n];
    	for(i=0;i<n;i++)
    	{
    		cin>>arr[i];
    	}
    	for(i=0;i<n;i++)
    	{
    		sum=0;
    		for(j=i;j<n;j++)
    		{
    			sum=sum+arr[j];
    			if(sum==x){
    				f=1;
    				break;
    			}
    			
    			 if(sum>x){
    			sum=0;
    			break;
    			 }
    			
    			
    		}
    		if(f==1)
    		break;
    	}
    	if(f==1)
    	cout<<"YES"<<endl;
    	else
    	cout<<"NO"<<endl;
    	
    }
    return 0;
}
