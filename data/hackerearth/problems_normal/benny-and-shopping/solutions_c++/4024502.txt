using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int t;
    cin>>t;
    while(t--)
    {
        cin.ignore();
        string s;
        getline(cin,s);
        //cout<<s<<endl;
        int in=s.find("$");
        string ans="$";bool f=false;
        for(int i=in+1;i<s.size();i++)
        {
            if(s[i]==' ')
                continue;
            else if(s[i]>='0'&&s[i]<='9')
                {
                    if(s[i]=='0'&&!f)continue;
                    else
                    {
                        f=true;
                        ans=ans+s[i];
                    }

                    }
            else
                break;
        }
        if(ans=="$")cout<<ans<<"0"<<endl;
        else
        cout<<ans<<endl;
    }
    return 0;
}
