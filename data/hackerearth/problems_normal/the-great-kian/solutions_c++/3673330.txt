#include <iostream>
using namespace std;
int main()
{
    long int ar[100000];
    long int n,i,s1=0,s2=0,s3=0;
    cin>>n;
    for(i=0;i<n;i++)
        cin>>ar[i];
    for(i=0;i<n;i=i+3)
    {
        s1=s1+ar[i];
    }
    for(i=1;i<n;i=i+3)
        s2=s2+ar[i];
    for(i=2;i<n;i=i+3)
    s3=s3+ar[i];
    cout<<s1<<" "<<s2<<" "<<s3;
    return 0;
}
