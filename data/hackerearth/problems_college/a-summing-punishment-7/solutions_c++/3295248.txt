#include<bits/stdc++.h>
using namespace std;
int main()
{
        int n;
         cin>>n;
         int a[n+1];
         a[0]=0;
         for(int i=1;i<=n;i++)
         {
              cin>>a[i];
         }
         for(int i=1;i<=n;i++)
         {
               a[i]+=a[i-1];
         }

         int q;
         cin>>q;
         while(q--)
          {

                   int p,q;
                   cin>>p>>q;
                   p-=1;
                   cout<<(a[q]-a[p])<<endl;
         }
         return 0;

}
