#include <iostream>
using namespace std;

#define MOD 1000000007
#define ll long long

ll powmod(ll n, ll x)
{
	if (x == 0)
		return 1;
	else if (x % 2 == 0)
	{
		ll temp = powmod(n, x / 2);
		return (temp * temp) % MOD;
	}
	else
		return (n * powmod(n, x - 1)) % MOD;
}

ll inv(ll n)
{
	return powmod(n, MOD - 2);
}

int main()
{
	int T;
	scanf("%d", &T);
	while (T--) {
		int N;
		ll minx, maxx, x;
		minx = maxx = 1;
		scanf("%d", &N);
		x = (N - 1);
		maxx = (maxx * x) % MOD;
		maxx = (maxx * ((x + 1) % MOD)) % MOD;
		maxx = (maxx * ((((2 * x) % MOD) + 1) % MOD)) % MOD;
		maxx = (maxx * inv(6)) % MOD;

		minx = (powmod((N - 1) / 2, 2) * N) % MOD;

		printf("%lld %lld\n", minx, maxx);
	}
	return 0;
}
