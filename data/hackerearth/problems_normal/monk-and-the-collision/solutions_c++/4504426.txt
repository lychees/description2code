#include <iostream>
using namespace std;

int main()
{
	int i,t,n,a[11],ans;
	cin>>t;
	while(t--)
	{
		ans = 0;
		for(i = 0 ; i < 11 ; i++)
		a[i] = 0;
		cin>>n;
		while(n--)
		{
			int x;
			cin>>x;
			if(a[x % 10] == 0)
			a[x % 10] = 1;
			else
			ans++;
		}
		cout<<ans<<endl;
	}
    return 0;
}
