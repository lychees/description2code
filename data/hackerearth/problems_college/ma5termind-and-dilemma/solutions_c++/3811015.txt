#include<bits/stdc++.h>
using namespace std;

int main()
{
 	int t;
 	cin>>t;
 	while(t--)
 	{
		int n,i,j;
		string str;
		cin>>n>>str;
		int ans=0,lo=0;
		i=0;j=n-1;
		while(i<j)
		{
			while(str[i]=='0')
				i++;
			while(str[j]=='1')
				j--;
			if(i<j)
				ans++;
			i++;j--;
		}
		i=0;j=n-1;
		while(i<j)
		{
			while(str[i]=='1')
				i++;
			while(str[j]=='0')
				j--;
			if(i<j)
				lo++;
			i++;j--;
		}
		cout<<min(ans,lo)<<"\n";
 	}
	return 0;
}
