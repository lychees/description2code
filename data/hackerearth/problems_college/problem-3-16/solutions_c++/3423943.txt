#include <iostream>
using namespace std;

int main()
{
   long	int t,n,k,l,i,a[10],count;
   cin>>t;
   while(t--)
   {
   	for(i=1;i<=10;i++)
   	{
   		a[i]=0;
   	}
   	count=0;
   	cin>>n>>k;
   	for(i=1;i<=n;i++)
   	{
   		cin>>l;
   		a[l]++;
   	}
   	long int max=0,p;
   	for(i=1;i<=10;i++)
   	{
   		if(a[i]>max)
   		{
   		max=a[i];
   		p=i;
   		}
   		
   	}
   	for(i=1;i<=10;i++)
   	{
   		if(i!=p)
   		count=count+a[i];
   	}
   	cout<<count<<endl;
   }
    return 0;
}
