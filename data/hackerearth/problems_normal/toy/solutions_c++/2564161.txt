#include <iostream>
using namespace std;
void sort(int *,int,int);
int partition(int *,int,int);
int main()
{
    int n;
    cin>>n;
    int arr[n];
    for(int i=0;i<n;i++)
    cin>>arr[i];
    sort(arr,0,n-1);
    int ctr=1;
    int temp=arr[0]+4;
    for(int i=0;i<n;i++)
    {
    	if(arr[i]<=temp)
    	continue;
    	else
    	{
    		ctr++;
    		temp=arr[i]+4;
    	}
    }
    printf("%d\n",ctr);
    return 0;
}
void sort(int arr[],int lower,int upper)
{
	if(lower<upper)
	{
		int q;
		q=partition(arr,lower,upper);
		sort(arr,lower,q-1);
		sort(arr,q+1,upper);
	}
}
int partition(int arr[],int p, int r)
{
	int x=arr[r],i=p-1,temp;
	for(int j=p;j<=r-1;j++)
	{
		if(arr[j]<=x)
		{
			i++;
			temp=arr[i];
			arr[i]=arr[j];
			arr[j]=temp;
		}
	}
	temp=arr[i+1];
	arr[i+1]=arr[r];
	arr[r]=temp;
	return(i+1);
}
