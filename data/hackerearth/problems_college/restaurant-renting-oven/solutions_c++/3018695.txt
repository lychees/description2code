#include <iostream>
#include <algorithm>
using namespace std;
int n, m;
int s[100], e[100], v[100];
int solve(int i, int c[])
{
	if(i<n)
	{
		int f=0, temp[m], res, j;
		for(j=0;j<m;j++)
			temp[j]=c[j];
		for(j=s[i];j<e[i];j++)
		{
			if(c[j]==0)
			{
				f=1;break;
			}
			c[j]-=1;
		}
		if(f==0)
			res=max((v[i]+solve(i+1,c)),solve(i+1,temp));
		else
			res=solve(i+1,temp);
		return res;
	}
	return 0;
}
int main()
{
	int t, i;
	cin>>t;
	while(t--)
	{
		cin>>n>>m;
		int c[m];
		for(i=0;i<n;i++)
			cin>>s[i]>>e[i]>>v[i];
		for(i=0;i<m;i++)
			cin>>c[i];
		cout<<solve(0,c)<<endl;	
	}
    return 0;
}
