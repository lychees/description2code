#include<bits/stdc++.h>
using namespace std;
#define MAX 1000000007
typedef long long ll;
ll modinv(ll a, ll b, ll mod)
{
	if(b==0)
		return 1;
	ll x=(modinv(a,b/2,mod))%MAX;
	if(b%2)
		return ((((a%MAX)*(x%MAX))%MAX)*(x%MAX))%MAX;
	else
		return ((x%MAX)*(x%MAX))%MAX;
}
ll ans(ll ar[], int n)
{
	ll i,j,y,p,s,g,max=0,v,q;
	q=1<<n;
	//cout<<p<<endl;
	for(i=0;i<q;i++) {
		g=0;
		p=1;
		for(j=0;j<n;j++) {
			y=1<<j;
			if(i&y) {
				g=g+ar[j];
				p=((p%MAX)*(ar[j]%MAX))%MAX;
			}
		}
		v=modinv(g,MAX-2,MAX);
		p=(p*(v%MAX))%MAX;
		if(p>max)
			max=p;
	}
	return max;
}
int main()
{
	int t,j;
	cin>>t;
	for(j=1;j<=t;j++) {
		int i,n;
		cin>>n;
		ll a[n];
		for(i=0;i<n;i++)
			scanf("%lld",&a[i]);
		ll d=ans(a,n);
		printf("Case #%d: %lld\n",j,d);
	}
	return 0;
}
