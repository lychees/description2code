#include<bits/stdc++.h>
using namespace std;

string s;
string res;

int main() {
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        cin>>s;
        l:res="";
        int num;
        int idx=-1;
        int len=s.length();
        for(int i=0;i<len;i++) {
            if(s[i]>='0'&&s[i]<='9') {
                num=s[i]-'0';
                idx=i;
                break;
            } else {
                res+=s[i];
            }
        }
        string temp=res;
        for(int i=0;i<num-1;i++) {
            res+=temp;
        }
        res+=s.substr(idx+1,len-idx);
        s=res;
        for(int i=0;i<s.length();i++) {
            if(s[i]>='0'&&s[i]<='9') {
                goto l;
            }
        }
        cout<<s<<endl;
    }
    return 0;
}
