#include <iostream>
using namespace std;
#define MOD 10000007
int dp[1100000];
int main()
{
    
    dp[0]=0;
    dp[1]=1;
    dp[2]=1;
    
    for(int i=3;i<=1000001;i++)
    {
    	
    	
    	if(i%4==0)
    	dp[i]=(dp[i-1]+dp[i-2]+1)%MOD;
    	else if(i%2==0)
    	dp[i]=(dp[i-1]+dp[i-2]+MOD-1)%MOD;
    	else 
    	dp[i]=(dp[i-1]+dp[i-2])%MOD;
    	
    }
    int t;
    cin>>t;
    
    while(t--)
    {
    	
    	int x;
    	cin>>x;
    	cout<<dp[x]<<endl;
    	
    	
    	
    }
    return 0;
}
