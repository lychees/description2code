#include<iostream>
#include<stdio.h>
#include<math.h>
#include<stdlib.h>
#include<string.h>
#include<algorithm>
#include<ctype.h>
using namespace std; 
int main()
{
   	int t;
 	cin>>t;
   	string a,b;
	long long l,m,i,j;
	while(t--)
	{
   		cin>>a;
   		cin>>b;
		l=a.size();
		m=b.size();	
//		cout<<"a"<<a;
//		cout<<"b"<<b;
   		int flag=0;
   		for(i=0;i<l;i++)
    		{
			for(j=0;j<m;j++)
			{
    				if( a[i]==b[j] )
    				{
					flag=1;
					break;
				}
				if(flag==1)
					break;
    			}
		}
		if(flag==1)
		{
			cout<<"YES"<<endl;
		}
		else
		{
			cout<<"NO"<<endl;
		}
    	}
   	return 0;
}
