#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int a[3], i, t;
    scanf("%d", &t);
    while(t--)
    {
    	for(i=0;i<3;i++)
    	{
    		scanf("%d", &a[i]);
    	}
    		sort(a, a+3);
    		if(a[2]>a[1])
    		printf("%d\n", 2*a[1]+1);
    		else
    		printf("%d\n", 2*a[1]);
    }
    return 0;
}
