#include <iostream>
#include<stdio.h>
#include<math.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    long long int ar[13];
	int i;
	ar[0]=0;
	ar[1]=9;
	for(i=2;i<=12;i++)
	{
		ar[i]=ar[i-1]+((9*pow(10,(i-1)))*i);
	}
    while(t--)
    {
    	long long int n,ans,d;
    	scanf("%lld",&n);
    	d=n;
    	int l=0;
    	while(d>0)
    	{
    		l++;
    		d=d/10;
    	}
    	if(l==1)
    	printf("%d\n",n);
    	else
    	{
    		d=pow(10,(l-1));
    		ans=ar[(l-1)];
    		d=d-1;
    		ans=ans+(l*(n-d));
    		printf("%lld\n",ans);
    	}
    }
    return 0;
}
