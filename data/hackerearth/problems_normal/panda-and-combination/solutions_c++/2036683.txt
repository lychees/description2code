#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	long long int n,m;
    	scanf("%lld%lld",&n,&m);
    	long int count=0;
    	if(m==1)
    		count=n;
    	else
    	{
    		while(n>0)
			{
				count+=n%m;
				n/=m;
			}
	    }
	    printf("%lld\n",count);
	}
}
