#include <iostream>
using namespace std;

int gcd(int a, int b)
{
 if (b == 0)
	return a;
else
	return gcd(b, a % b);
}

int main()
{
   int T;
   cin>>T;
   while(T!=0)
   {
   	T--;
   	int N,prime=1;
   	cin>>N;
   	for(int i=2;i<N;i++)
   	{
   		if (gcd(N, i) == 1)
		{
			prime++;
		}

   	}
   	bool ans = true;
	for (int i = 2; i < prime; ++i)
	{
		if (prime % i == 0)
		{
			ans = false;
			break;
		}
	}
	if (prime == 1)
		ans = false;
	if (ans)
		cout << "TRUE" << endl;
	else
		cout << "FALSE" << endl;

   }
    return 0;
}
