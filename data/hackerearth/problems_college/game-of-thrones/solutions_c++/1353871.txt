#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#include <set>
using namespace std;


int main() {
    
    int ts,n,a;
    cin>>ts;
    while(ts--)
    {
    	cin>>n;
    	set<int>st;
    	for(int i=0;i<n;i++)
    	{
    		cin>>a;
    		st.insert(a);
    	}
    	cout<<n-st.size()<<endl;
    }
    return 0;
}
