n,k=map(int,raw_input().split())
a=map(int,raw_input().split())
b=map(int,raw_input().split())
u=[0,a[0]]
v=[0,b[0]]
for i in range(1,len(a)):
	if i%2:
		u.append(u[i]+b[i])
		v.append(v[i]+a[i])
	else:
		u.append(u[i]+a[i])
		v.append(v[i]+b[i])
for i in range(k):
	p,l,r=map(int,raw_input().split())
	if l%2:
		if p==1:
			print u[r]-u[l-1]
		else:
			print v[r]-v[l-1]
	else:
		if p==2:
			print u[r]-u[l-1]
		else:
