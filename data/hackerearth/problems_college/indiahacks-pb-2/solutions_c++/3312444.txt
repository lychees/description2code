#include <bits/stdc++.h>
using namespace std;

int main()
{
	int i,j,t,l;
	cin>>t;
	char s[t][15],c[15];
	for(i=0;i<t;i++)
	{
		cin>>c;
		for(j=i-1;j>=0;j--)
		{
			if(strcmp(s[j],c)==0)
			{
				l=strlen(c);
				cout<<l<<" "<<c[l/2];
				break;
			}
		}
			if(j>=0)
			break;
			l=strlen(c);
			for(j=0;c[j]!='\0';j++)
			{
				s[i][l-j-1]=c[j];
			}
			s[i][l]='\0';
	}
    return 0;
}
