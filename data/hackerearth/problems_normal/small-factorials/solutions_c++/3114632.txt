    #include<bits/stdc++.h>
    #define X 158
    using namespace std;
     
    void print(int b[])
    {
    	int j,i=0;
    	while((i<X-1)&&(b[i]==0))
    		i++;
    	for(j=i;j<X;j++)
    		cout<<b[j];
    	cout<<endl;
    }
     
    void calc(int b[],int i)
    {
    	int temp=0,j=X-1;
    	while(j>=0)
    		{
    			temp=(i*(b[j]))+temp;
    			 if(temp<=9)
    		 	{
    		 		b[j]=temp;
    		 	 	temp=0;
    		 	}
    			else
    			{
    				b[j]=temp%10;
    				temp=temp/10;
    			}
    		j--;
    		}
     
    }
     
    int main()
    {
    	int fact[X];
    	 int n,t,i,j;
    	 cin>>t;
    	 for(j=1;j<=t;j++)
    	 	{
    	 		for(i=0;i<X-1;i++)
    	 		fact[i]=0;
    		 	 fact[X-1]=1;
    		 	 cin>>n;
    		 	 for(i=1;i<=n;i++)
    		 	 	calc(fact,i);
    		 	 print(fact);
    	 	}
    	 return(0);
     }
