#include <iostream>
#define MW 1000000007
using namespace std;
typedef long long ll;
int main()
{
	long t;
	scanf("%ld",&t);
	for(;t!=0;--t)
	{
		ll n,sum=0,i,j;
		scanf("%lld",&n);
		for(i=2;i<n;i*=2)
		{
			for(j=1;j<i;j*=2)
			{	
				if(i+j<=n)
				sum=(sum%MW+i%MW+j%MW)%MW;
				//cout<<i<<" ";
			}
		}
		printf("%lld\n",sum);
		//cout<<sum<<"\n";
	}
    return 0;
}
