void solve()
{
	int x, y;
	cin>>x>>y;
	if(y == x) {
		cout<<"-1\n";
		return;
	}
	else if(y > x) {
		cout<<"0\n";
		return;
	}
	int d = x-y;
//	cout<<d<<"\n";
	vector<int> r;
	int sq = sqrt(d);
	for(int i=1;i<=sq;i++) {
		if(d%i == 0) {
			r.push_back(i);
			if(i != d/i) {
				int val = d/i;
				r.push_back(val);
			}
		}
	}
	vector<int> rr;
	for(int i=0;i<r.size();i++) {
		if(x%r[i] == y)
			rr.push_back(r[i]);
	}
	cout<<rr.size()<<"\n";
}

	int t;
	cin>>t;
//	t = 1;
	while(t--)
	{
		solve();
	}
	return 0;
