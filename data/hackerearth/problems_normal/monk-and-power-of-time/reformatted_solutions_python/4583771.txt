n = (int)(raw_input())
calling = [int(cell) for cell in raw_input().split(" ")]
ideal = [int(cell) for cell in raw_input().split(" ")]
id_index = 0
time = n
for process in ideal:
    cal_index = calling.index(process)
    if cal_index >= id_index:
        time += cal_index-id_index
    else:
        time += n-id_index+cal_index
    calling.remove(process)
    n -= 1
    if n != 0:
        id_index = cal_index % n
print time
