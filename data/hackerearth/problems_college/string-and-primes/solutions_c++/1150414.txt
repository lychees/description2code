#include <iostream>
#include <string>
using namespace std;
int primes[]={2,3,5,7,11,13,17};
int main()
{
	int t,c,flag;
	string s,d;
	cin>>t;
	while(t--)
	{
		cin>>s;
		for(int i=0;i<s.size();i++)
		{
		c=s[i];
		flag=1;
		for(int j=0;j<7;j++)
		if(c%primes[j]==0)
		{
			flag=0;
			break;
		}
		if(flag==0)
		d.push_back(s[i]);
		}
	if(d.size()!=0)
	cout<<d<<endl;
	else
	cout<<"NULL\n";
	d.clear();
	}
return 0;
}
