    #include <iostream>
    #include <string>
    using namespace std;
    string alpha(string str){
    int l=str.length();
    for(int i=0;i<l;i++){
    for(int j=0;j<l;j++){
    if(str[i]<str[j])
    {
    char temp = str[i];
    str[i] = str[j];
    str[j] = temp;
    }
    }
    }
    return(str);
    }
