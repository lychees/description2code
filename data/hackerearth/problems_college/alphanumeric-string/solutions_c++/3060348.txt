#include<bits/stdc++.h>
using namespace std;

int main(){
    int t; cin >> t;
    cin.ignore();
    while(t--){
        string s;
        getline(cin,s);
        int l = s.length();
        int sum=0;
        for(int i=0;i<l;i++){
            if(s[i]>='0' && s[i]<='9')
                sum+= s[i]-48;
        }
        vector<string> v;
        string s1="";
        int i;
        for(i=0;i<s.length();i++){
            if(s[i]>='0' && s[i]<='9'){
                if(i==l-1)
                    v.push_back(s1);
                continue;
            }
            if(s[i]==' '){
                v.push_back(s1);
                s1 = "";
                continue;
            }
            s1 += s[i];
            if(i==l-1)
                v.push_back(s1);
        }
        sort(v.begin(),v.end());
        vector<string>::iterator it = v.begin();
        while(it!=v.end()){
            string p = *it;
            int m = p.length();
            sum = sum-m;
            cout << p << " ";
            if(sum>=0)
                cout << sum << " ";
            else
                cout << -sum << " ";
            it++;
        }
        cout << endl;
    }
    return 0;
}
