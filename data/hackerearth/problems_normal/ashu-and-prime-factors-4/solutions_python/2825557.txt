ar = [True]*1000001

ar[0], ar[1] = False, False
ans_ar= [0]*1000001

for i in range(2, 1000001):
	counter = 0
	if ar[i]:
		for j in range(i, 1000001, i):
			if ar[j]:
				counter += 1
				ar[j] = False
	ans_ar[i] = counter
			
t = int(raw_input())
while t:
	t -= 1
	n = int(raw_input())
	print ans_ar[n]
