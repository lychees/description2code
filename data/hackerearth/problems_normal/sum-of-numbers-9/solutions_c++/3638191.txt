#include<bits/stdc++.h>
using namespace std;
void possible(int a[],int n,int s)
{
    int flag=0;
    if(s==0) {
        cout <<"YES"<<endl;
    }
    for(int i=1;i<(1<<n);i++)
    {
        int sum=0;
        for(int j=0; j<n; j++)
        {
            if( i & (1<<j) )
            {
                sum=sum+a[j];
            }
        }
        if(sum==s) {
   //             cout << "YES"<<endl;
            flag=1;
            break;
        }
    }
     if(flag==1)
        cout << "YES"<<endl;
    else if(flag==0)
        cout <<"NO"<<endl;
}
int main()
{

    int t,n,i;
    cin >>t;
    while(t--)
    {
        cin >>n;
        int a[1000000];
        int s;
        for(i=0;i<n;i++)
            cin >> a[i];
        cin >>s;
        possible(a,n,s);
    }
}
