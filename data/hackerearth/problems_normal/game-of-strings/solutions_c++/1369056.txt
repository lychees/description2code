#include<bits/stdc++.h>
#define MAX 5005

using namespace std;

typedef long long int lli;
typedef pair<int,int> pii;

int dp[MAX][MAX];
int prime[]={2,3,5,7};

bool vowel(char ch){
	
	if(ch=='a'||ch=='e'||ch=='o'||ch=='u'||ch=='i')
	  return true;
	  
	else return false;  
	
}

char X[MAX],Y[MAX];
int main(void){
	
	int i,j,sum=0,count=0;
	int n,m;
	
	scanf("%s%s",X,Y);
	m=strlen(X);
	n=strlen(Y);
	
	for (i=0; i<=m; i++)
     for (j=0; j<=n; j++){
     	
       if (i == 0 || j == 0)
         dp[i][j] = 0;
  
       else if (!vowel(X[i-1])&&X[i-1] == Y[j-1]){
         dp[i%2][j] = dp[(i+1)%2][j-1] + 1;
       
       }
       else    dp[i%2][j] = max(dp[(i+1)%2][j], dp[i%2][j-1]);
     
	 }
   
     printf("%d\n",dp[m%2][n]);	
	
	
	
	return 0;
	
}
