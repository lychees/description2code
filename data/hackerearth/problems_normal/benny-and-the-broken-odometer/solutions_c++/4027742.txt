int full(int len) {
  int res = 1;
  for (int i = 0; i < len; i++)
    res *= 9;
  return res;
}

int calc(int x) {
  vector<int> v;
  while (x) {
    v.pb(x % 10);
    x /= 10;
  }
  reverse(v.begin(), v.end());
  int total = 0;
  for (int i = 0; i < v.size(); i++) {
    for (int put = 0; put < v[i]; put++) {
      if (put != 3) {
        total += full(v.size() - i - 1);
      }
    }
  }
  return total;
}
