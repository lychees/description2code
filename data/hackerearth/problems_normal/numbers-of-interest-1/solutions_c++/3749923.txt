#include <iostream>
using namespace std;
long long int dp[60];
long long int dptry(long long int n,long long int x,long long int y){
	long long int i,j;
	for(i=1;i<=60;i++)
		dp[i]=0;
	for(i=1;i<=x;i++){
		dp[i]=y;
	}
	for(i=x+1;i<=n;i++)
		{
			for(j=i-1;j>=i-x;j--){
			dp[i]+=dp[j];
			
		}
			
		}
	
	return dp[n];
}
int main()
{long long int t,n,x,y;
ios_base::sync_with_stdio(0);
cin>>t;
while(t--){
cin>>x>>y>>n;
cout<<dptry(n,x,y)<<endl;
}
    return 0;
}
