#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,j,ans,n,t,f,st,g;
	cin>>t;
	while(t--)
	{
		string s;
		cin>>s;
		f=0,st=1,g=0;
		ans=0;
		for(i=0;i<s.size();i++)
		{
			if(s[i]=='O')
			{
				if(st==1)
				f++;
				g=0;
			}
			else if(s[i]=='X')
			{
				f=0;
				st=0;
				g=0;
			}
			else if(s[i]=='*')
			{
				ans+=f;
				f=0;
				//cout<<ans<<endl;
				st=1;
				g=1;
			}
		}
		if(g==0)
		{
			if(st==1)
			ans+=f;
		}
		cout<<ans<<endl;
	}
	return 0;
}
