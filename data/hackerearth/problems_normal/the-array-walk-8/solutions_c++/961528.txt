#include <iostream>
using namespace std;

long long gcd(long long a,long long b,long long &x,long long &y)
{
	if(a==0) { x=0,y=1; return b; }
	long long x1,y1;
	long long d=gcd(b%a,a,x1,y1);
	x = y1-(b/a)*x1;
	y=x1;
	return d;
}

bool f(long long a,long long b,long long c,long long &x1,long long &y1,long long &gc)
{
	gc = gcd(abs(a),abs(b),x1,y1);
	if(c%gc) return 0;
	x1 *= c/gc;
	y1 *= c/gc;
	if(a<0) x1 *= -1;
	if(b<0) y1 *= -1;
	return 1;
}

int main()
{
    int t;
    cin >> t;
    while(t--) {
    	long long h,u,d,x,y,gc,v;
    	cin >> h >> u >> d;
    	if(!f(u,-1*d,h,x,y,gc)) {
    		cout << "-1\n";
    	} else {
    		long long u1 = u/gc;
    		long long d1 = d/gc;
    		v = max(abs(x)/d1,abs(y)/u1)+1;
    		x = x + v*d1;
    		y = y + v*u1;
    		long long v1 = x/d1,v2 = y/u1;
    		long long mini = min(v1,v2);
    		x-=mini*d1;
    		y-=mini*u1;
    		cout << x+y << endl;
    	}
    }
    return 0;
}
