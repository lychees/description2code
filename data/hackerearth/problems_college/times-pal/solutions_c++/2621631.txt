#define FL(i,a,b) for(ll i=a;i<b;i++)
#define FOR(i,n) for(ll i=0;i<n;i++)
#define SORTF(x) sort(x.begin(),x.end(),func);
#define SORT(x) sort(x.begin(),x.end())
#define pb(x) push_back(x)

#define RSORT(v) { SORT(v) ; REVERSE(v) ; }

#define S(n) scanf("%lld",&n)

bool check(string s) {
//	cout << s << "\n";
	ll n = s.size();
	ll j = n-  1;
	for(ll i = 0;i < j;i++,j--) {
		if(s[i] != s[j])
			return false;
	}
	return true;
}

int main() {
	ll t;
	S(t);
	while(t--) {
		string a,b;
		cin >> a;
		cin >> b;
		string t;
		vector<string> v;
		for(ll i = 0;i <= a.size();i++) {
			string t1 = t + b;
			v.pb(t1);
			t = t + a[i];
		}
		t = "";
		ll count = 0;
		ll j = a.size() - 1;
		for(ll i = 0;i < v.size();i++,j--) {
			v[v.size() - i - 1] = v[v.size() - i - 1] + t;
			t =  a[j] + t;
			if(check(v[v.size() - i - 1]))
				count++;	
		}
		printf("%lld\n", count);
	}
}
