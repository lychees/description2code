t = int(raw_input())

while t > 0:
    t -= 1
    a, b, c, k = map(int, raw_input().split())
    l = 0
    r = 10**5

    while l < r:
        mid = (l+r)/2
        res = (mid ** 2) * a + mid*b+c
        if res >= k:
            r = mid
        else:
            l = mid+1

    print r
