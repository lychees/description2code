#include <iostream>
#include<string>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	int l,i,count=0,flag=0,hash[26]={0},max,flag1=0;
    	l=s.length();
    	for(i=0;i<l;i++)
    	{
    		hash[s[i]-'a']++;
    	}
    	for(i=0;i<26;i++)
    	{
    		if(hash[i])
    		  count++;
    	}
    	if(l==3)
    	{
    		flag1=1;
    	}
    	else
    	{
    	for(i=0;i<l;i++)
    	{
    		if(s[i]==s[i+1])
    		   flag1=1;
    	}
    	}
     	if(count==3)
    	{
    		for(i=0;i<26;i++)
    		{
    		if(hash[i])
    		{
    			max=hash[i];
    			if(max==hash[i])
    			    flag++;
    		}
    		}
    		if(flag==3 && flag1)
    		   cout<<"OK\n";
    		else
    		   cout<<"Not OK\n";
    	}
    	else
    	   cout<<"Not OK\n";
    }
    return 0;
}
