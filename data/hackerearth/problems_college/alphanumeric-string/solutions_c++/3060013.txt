#include <iostream>
#include <string.h>
#include <algorithm>
#include <vector>
using namespace std;

int main()
{
    int n, i, j, temp, sum, len;
    char str[205];
    scanf("%d", &n);
    while(n--)
    {
    	scanf(" %[^\n]s", str);
    	sum = 0;
    	len = strlen(str);
    	for(i=0, j=0; i<len; i++)
    	{
    		if(str[i]>='0' && str[i]<='9')
    		{
    			sum = sum + str[i] - '0';
    		}
    		else
    		{
    			str[j] = str[i];
    			j++;
    		}
    	}
    	len = j;
    	vector<vector<char> > vec;
    	for(i=0; i<len; i++)
    	{
    		vector<char> temp;
    		while(str[i]!=' ' && i<len)
    		{
    			temp.push_back(str[i]);
    			i++;
    		}
    		vec.push_back(temp);
    	}
    	sort(vec.begin(), vec.end());
    	for(i=0; i<vec.size(); i++)
    	{
    		for(j=0; j<vec[i].size(); j++)
    		{
    			printf("%c", vec[i][j]);
    		}
    		sum = sum-j;
    		if(sum>=0)
    			printf(" %d ", sum);
    		else
    			printf(" %d ", -1*sum);
    	}
    	printf("\n");
    }
    return 0;
}
