n  = int(raw_input())
for i in range(n):
    s = raw_input().split(' ')
    cr = 0
    c = 0
    for j in range(len(s[0])):
        p = int(s[0][len(s[0])-j-1]) + int(s[1][len(s[1])-j-1]) + cr
        if(p >= 10):
            c=c + 1
            cr = 1
        else:
            cr = 0
    if(c == 1):
    	print c,"carry operation"
    elif(c>1):
    	print c,"carry operations"
    else:
    	print "No carry operation"
