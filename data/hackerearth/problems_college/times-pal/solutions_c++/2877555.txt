#include<iostream>
#include<cstring>
#include<cstdio>
#include<cstdlib>
#include<cmath>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
	string str1;
	cin>>str1;
	string str2;
	cin>>str2;
	int c=0;
	for(int i=0;i<=str1.length();i++)
	{
		string  st;
		if(i==str1.length())
		{
			st = str1.substr(0,i) + str2;
		}
		else
		{
			st = str1.substr(0,i)+str2+str1.substr(i);
		}
		string  str3;
		str3=st;
		reverse(str3.begin(),str3.end());
		if(str3==st)
		{
			c++;
		}
	}
	cout<<c<<"\n";
}
}
