#include<bits/stdc++.h>
#include<iostream>
#include<string>
using namespace std;
string a;
string b;
int arr1[26];
int arr2[26];
int main(){
	int t;
	cin >> t;
	getchar();
	while(t--){
		
		getline(cin, a);
		getline(cin, b);
		for(int i = 0; i < 26; i++){
			arr1[i] = 0;
			arr2[i] = 0;
		}
		for(int i = 0; i < a.length(); i++){
			if(a[i] != ' '){
				arr1[a[i] - 97]++;
			}
		}
		for(int i = 0; i < b.length(); i++){
			if(b[i] != ' '){
				arr2[b[i] - 97]++;
			}
		}
		for(int i = 0; i < 26; i++){
			if(arr1[i] > arr2[i])
			{
				arr1[i] = arr1[i] - arr2[i];
				arr2[i] = 0;
			}
			
			else if(arr1[i] < arr2[i]){
				arr2[i] = arr2[i] - arr1[i];
				arr1[i] = 0;
			}
			else{
				arr1[i] = 0;
				arr2[i] = 0;
			}
		}
		int count1 = 0;
		int count2 = 0;
		for(int i = 0; i < 26; i++){
			count1 = count1 + arr1[i];
			count2 = count2 + arr2[i];
		}
		if(count1 != 0 && count2 == 0)
			cout << "You win some." << "\n";
		else if(count1 == 0 && count2 != 0)
			cout << "You lose some." << "\n";
		else
			cout << "You draw some." << "\n";
	}
}
