#include <bits/stdc++.h>
using namespace std;
#define llong long long
#define long llong

int n, m, n_test;

void solve ()
{
    int c0 = 0;

    cin >> n >> m;
    for (int i = 1; i <= n; ++i)
    {
        int k; cin >> k;
        c0 += (k == 0);
    }

    if (m <= c0) cout << n - c0 + m << "\n";
    else
    if ((m - c0) & 1) cout << n - 1 << "\n";
    else
        cout << n << "\n";
}

int main ()
{
    ios::sync_with_stdio(false); cin.tie(nullptr);
	if (ifstream("test.inp")) cin.rdbuf((new ifstream("test.inp"))->rdbuf());

    cin >> n_test;
    while (n_test--) solve();
}
