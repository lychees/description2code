#include <iostream>
using namespace std;

int main()
{
   long long sum1=0,sum2=0,sum3=0,n,i,j;
   cin>>n;
   long long a[n];
   for(i=1;i<=n;i++)
   cin>>a[i];
   i=1;
   while(i<=n)
   {
   	sum1+=a[i];
   	j=i+1;
   	if(j<=n)
   	sum2+=a[j];
   	j=i+2;
   	if(j<=n)
   	sum3+=a[j];
   	i+=3;
   }
   cout<<sum1<<" "<<sum2<<" "<<sum3<<endl;
   return 0;
}
