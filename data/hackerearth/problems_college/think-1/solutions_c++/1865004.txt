int rev(int n)
{
	int ans=0;
	while(n>0)
	{
		int temp=n%10;
		n=n/10;
		ans=ans*10+temp;
	}
	return ans;
}
string reverse(int n)
{
	string ans;
	while(n>0)
	{
		int temp=n%10;
		n=n/10;
		ans+='0'+temp;
	}
	return ans;
}
int main()
{
	ios_base::sync_with_stdio(false);
	int dp[10010]={0};
	for(int i=1;i<10001;i++)
	{
		if(dp[i]!=0)		continue;
		int val=i;
		int revval=rev(i);
		if(revval==val)		continue;
		int valsq=val*val;
		int revsq=revval*revval;
		int temp=rev(revsq);
		//cout<<val<<" "<<revval<<" "<<valsq<<" "<<revsq<<endl;
		if(temp==valsq)
		{
			dp[val]=1;
			dp[revval]=1;
		}
	}
	int t;
	cin>>t;
	int n;
	while(t--)
	{
		cin>>n;
		int val=dp[n];
		if(val==0)		cout<<"No"<<endl;
		else 		cout<<reverse(n)<<endl;
	}
	return 0;
}
