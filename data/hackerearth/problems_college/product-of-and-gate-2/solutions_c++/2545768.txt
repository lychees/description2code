#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
    int t;
    scanf("%d", &t);
    long long n, m;
    int set = 0;
    long long result;
    while(t > 0) {
    	scanf("%lld%lld", &n, &m);
    	result = n;
    	set = 0;
    	for(long long i=n; i<=m; i++) {
    		if(result == 0) {
    			printf("0\n");
    			set = 1;
    			break;
    		}
    		result &= i;
    	}
    	if(!set) {
    		printf("%lld\n", result);
    	}
    	t--;
    }
    return 0;
}
