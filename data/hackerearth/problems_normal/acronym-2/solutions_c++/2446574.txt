#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	map<string,int>m;
	string s;
	cin>>t;
	while(t--)
	{
		cin>>s;
		m[s]=0;
	}
	cin>>t;
	vector<char>v;
	while(t--)
	{
		cin>>s;
		if(m.find(s) != m.end())
		{
			continue;
		}
		else
		{
			v.push_back(s[0]-32);
		}
	}
	int i;
	for(i=0;i<v.size()-1;i++)
		cout<<v[i]<<'.';
	cout<<v[i]<<endl;
	return 0;
}
