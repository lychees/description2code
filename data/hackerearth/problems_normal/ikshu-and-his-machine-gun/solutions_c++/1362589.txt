#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int s;
    cin>>s;
    int k[s];
    int i;
    for(i=0;i<s;i++)
    {
    	cin>>k[i];
    }
    sort(k,k+s);
    for(i=0;i<s;i++)
    {
    	k[i]=k[i]-i;
    	cout<<k[i]<<" ";
    }
    
    return 0;
}
