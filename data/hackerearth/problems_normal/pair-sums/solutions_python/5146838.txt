# print 'Hello World!'

k = raw_input().split()
k = int(k[1])

values = [int(i) for i in raw_input().split()]

map_values = {}
for value in values:
	if value in map_values:
		map_values[value] += 1
	else:
		map_values[value] = 1

found = False
for value in values:
	diff = k - value
	if diff in map_values:
		if (diff == value and map_values[diff] > 1) or (diff != value and map_values[diff] > 0) :
			print "YES"
			found = True
			break

if not found:
	print "NO"
