#include <iostream>
#include <algorithm>
using namespace std;

int main() {
	int n,l,r,x,i=0,in,pin=100000,j=0;
	cin>>n>>l>>r>>x;
	int ar[r-l+1],q[x],k;
	while(i<l)
	{
		cin>>in;
		i++;
	}
	while(i<=r)
	{
		cin>>ar[j];
		j++;
		i++;
	}
	sort(ar,ar+j);
	while(i<n)
	{
		cin>>in;
		i++;
	}
	n=0;
	while(x--)
	{
		cin>>q[n];
		n++;
	}
	x=0;
	//cout<<n;
	while(n--)
	{
		k=0;
		for(k=0;k<j;k++)
		{
			if(ar[k]>q[x])
			break;
		}
		if(k==j)
		cout<<"-1\n";
		else
		cout<<ar[k]<<endl;
		x++;
	}
	return 0;
}
