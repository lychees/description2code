#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int p,g,i,w[101],temp=0,flag;
    	cin>>p>>g;
    	for(i=0;i<p;i++)
    	{
    	    cin>>w[i];
    	}
    	temp=w[0];
    	for(i=0;i<p;i++)
    	{
    	    for(int j=i+1;j<p;j++)
    	    {
    	    	if(w[i]<w[j])
    	    	{
    	    		int a=w[i];
    	    		w[i]=w[j];
    	    		w[j]=a;
    	    	}
    	    }
    	}
    	for(i=0;i<p;i++)
    	{
    		if(w[i]==temp)
    		flag=i;
    	}
    	int groupid;
    	if((flag+1)%g==0)
    	groupid=g;
    	else
    	groupid=(flag+1)%g;
    	int num=0;
    	for(i=0;i<p;i++)
    	{
    		if((i+1)%g==0&&w[flag]<w[i])
    		num++;
    	}
    	cout<<num<<endl;
    }
    return 0;
}
