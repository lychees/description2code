#include<bits/stdc++.h>
#define scd(x) scanf("%d",&x)
#define prd(x) printf("%d",x)
#define sclld(x) scanf("%lld",&x)
#define prlld(x) printf("%lld",x)
#define f_in(x) freopen(x,"r",stdin)
#define f_out(x) freopen(x,"w",stdout)

using namespace std;

typedef long long int llt;
struct query
{
    int l,r;
    llt c;
};

int main()
{
    ios::sync_with_stdio(false);
    int t,n,m,Q,l,r;
    vector<query> q;
    vector<int> mark;
    vector<llt> v;
    llt val;
    cin>>t;
    while(t--)
    {
        cin>>n>>m;
        q.resize(m+2);
        mark.assign(m+2,0);
        v.assign(n+2,0);
        for(int i = 1 ; i <= m ; i++)
            cin>>q[i].l>>q[i].r>>q[i].c;
        cin>>Q;
        while(Q--)
        {
            cin>>l>>r;
            mark[l] += 1;
            mark[r+1] -= 1;
        }
        for(int i = 1 ; i<= m ; i++)
            mark[i] += mark[i-1];

       // for(int i = 1 ; i<= m ; i++)
         //   cout<<mark[i]<<" ";
        //cout<<"\n";

        for(int i = 1 ; i<= m ; i++)
        {
            val =q[i].c*mark[i];
          //  cout<<"I : "<<i<<" \t Val : "<<val<<"\n";
            v[q[i].l] += val;
            v[q[i].r + 1] -= val;
        }
        //for(int i = 1 ; i<= n ; i++)
         //   cout<<v[i]<<" ";
       // cout<<"\n";
        for(int i = 1 ; i<= n ; i++)
            v[i] += v[i-1];
        for(int i = 1 ; i<= n ; i++)
            cout<<v[i]<<" ";
        cout<<"\n";
        
        mark.clear();
        v.clear();
        q.clear();
    }

    return 0;
}
