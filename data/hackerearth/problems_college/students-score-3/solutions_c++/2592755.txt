#include <iostream>
using namespace std;

int main()
{
   int n;
   std::cin>>n;
   int arr[n+2];
   for(int i=0;i<n;i++){
   	std::cin>>arr[i];
   }
   int mx1=0,mx2=0;
   for(int i=0;i<n;i++){
   	mx1+=arr[i];
   	if(mx1<0){
   		mx1=0;
   	}
   	if(mx2<mx1){
   		mx2=mx1;
   	}
   }
   std::cout<<mx2<<"\n";
}
