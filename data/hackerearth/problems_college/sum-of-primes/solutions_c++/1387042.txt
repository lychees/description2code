#include<iostream>
#include<string.h>
#include<stdio.h>
#include<math.h>
using namespace std;
#define INT long long
INT arr[2000056];
bool val[2000056];
INT prm[2000056];
INT lk;

void markMultiples(INT arr[],INT a,INT n)
{
    INT i=a,num;
    while((num=i*a)<=n)
    {
        arr[num-1]=1;
        ++i;
    }
}
void sieve(INT n)
{
    if(n>=2)
    {
        val[0]=0;
        memset(arr,0,sizeof(arr));
        for(INT i=1;i<n;++i)
        {
            if(arr[i]==0)
            {
                arr[i]=i+1;
                val[i]=1;
                if(i<=sqrt(n)+1)
                {
                    markMultiples(arr,i+1,n);
                }
            }
            else
            {
                arr[i]=0;
                val[i]=0;
            }
        }
        lk=0;
        for(INT i=1;i<n;i++)
        {
            arr[i]+=arr[i-1];
            if(val[i]==1)
            {
                prm[lk]=i+1;
                lk++;
            }
        }
    }
}


int main()
{
    INT t;
    scanf("%lld",&t);
    sieve(1000000);
    while(t--)
    {
        INT m,n;
        scanf("%lld",&m);
        scanf("%lld",&n);
        INT cur;
        if(m>1)
            cur=arr[n-1]-arr[m-2];
        else
            cur=arr[n-1];
        if(cur<=1000000)
        {
            //cout<<"#"<<cur;
            if(cur>0&&val[cur-1]==1)
            {
                printf("YES\n");
            }
            else
            {
                printf("NO\n");
            }
        }
        else
        {
            //cout<<cur;
            INT isPrm=1;
            INT p=sqrt(cur);
            for(int i=0;i<lk&&prm[i]<p+1;i++)
            {
                if(cur%prm[i]==0)
                {
                    isPrm=0;
                    i=lk;
                }
            }
            if(isPrm)
            {
                printf("YES\n");
            }
            else
            {
                printf("NO\n");
            }
        }
    }
    return 0;
}
