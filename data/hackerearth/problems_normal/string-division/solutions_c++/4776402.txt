#include <iostream>

using namespace std;

int main()
{
    int t;
    cin>>t;
    while (t--) {
        string s;
        cin>>s;
        int n = s.length();
        if (n>=10) {
            cout<<"YES\n";
        }
        else {
            int i, j, k,ok=0;
            for (i=1;i<n;i++) {
                for (j=i+1;j<n;j++) {
                    for (k=j+1;k<n;k++) {
                        string s1 = s.substr(0,i);
                        string s2 = s.substr(i,j-i);
                        string s3 = s.substr(j,k-j);
                        string s4 = s.substr(k,n-k);
                        if (s1!=s2 && s1!=s3 && s1!=s4 && s2!=s3 && s2!=s4 && s3!=s4 ) {
                            ok=1;
                            break;
                        }
                    }
                    if (ok==1) break;
                }
                if (ok==1) break;
            }
            if (ok==1) cout<<"YES\n";
            else cout<<"NO\n";
        }
    }
    return 0;
}
