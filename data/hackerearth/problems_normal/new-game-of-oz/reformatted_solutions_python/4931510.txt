
# Read input from stdin and provide input before running code

t = int(raw_input())
n = []
l = []
for i in range(t):
    n.append(int(raw_input()))
    l.append(raw_input())

for k in range(t):
    arr = map(int, l[k].split())
    moves = 0
    while len(arr) > 0:
        count = []
        for i in arr:
            if i+1 in arr and i-1 in arr:
                count.append(2)
            elif i+1 in arr or i-1 in arr:
                count.append(1)
            else:
                count.append(0)

        i = count.index(min(count))
        m = min(count)
        value = arr[i]
        if value+1 in arr:
            j = arr.index(value+1)
            count.pop(j)
            arr.pop(j)
        if value-1 in arr:
            j = arr.index(value-1)
            count.pop(j)
            arr.pop(j)
        count.remove(m)
        arr.remove(value)
        moves += 1
        del count
    print moves
    del arr
