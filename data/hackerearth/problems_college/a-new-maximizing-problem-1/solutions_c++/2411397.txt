#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;

#define LL long long
#define MOD 1000000007
int main()
{
	int testcases;
	cin>>testcases;
	for(int i = 0; i < testcases; i++) {
	    int n;
	    cin>>n;
	    int *numbers = new int[n];
	    for(int i = 0;i<n;i++) {
	    	cin>>numbers[i];
	    }
	    sort(numbers,numbers+n);
	    long long int A_sum = ((n&1)?numbers[n/2] : 0),B_sum = ((n&1)?numbers[n/2] : 0);
	    for(int i = 0;i<n/2;i++) {
	    	B_sum += abs(numbers[i]) + abs(numbers[n-i-1]);
	    	A_sum += numbers[n-i-1] - numbers[i];
	    }
	    A_sum%=MOD;
	    B_sum%=MOD;
	    cout << (A_sum*B_sum)%MOD <<"\n";
	}
    return 0;
}
