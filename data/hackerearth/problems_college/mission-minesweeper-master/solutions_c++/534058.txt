#include<iostream>
using namespace std;
string grid[101];
int x,y;
int flood(int i,int j)
{
	if(i==-1||j==-1||i>=x||j>=y)
	{
		return 0;
	}
	
	if(grid[i][j]!='.')
	{
		return 0;
	}
	int ans=0;
	if((i+1)<x&&grid[i+1][j]=='*')
	{
		ans++;
	}
	if(i>0&&grid[i-1][j]=='*')
	{
		ans++;
	}
	if((j+1)<y&&grid[i][j+1]=='*')
	{
		ans++;
	}
	if(j>0&&grid[i][j-1]=='*')
	{
		ans++;
	}
	if(i>0&&j>0&&grid[i-1][j-1]=='*')
	{
		ans++;
	}
	if(i>0&&(j+1)<y&&grid[i-1][j+1]=='*')
	{
		ans++;
	}
	if((i+1)<x&&j>0&&grid[i+1][j-1]=='*')
	{
		ans++;
	}
	if((i+1)<x&&(j+1)<y&&grid[i+1][j+1]=='*')
	{
		ans++;
	}//cout<<ans<<endl;
	grid[i][j]=('0'+ans);
	if(grid[i][j]=='0')
	{
	
		flood(i,j+1);
		flood(i,j-1);
		flood(i-1,j);
		flood(i+1,j);
		flood(i-1,j-1);
		flood(i-1,j+1);
		flood(i+1,j+1);
		flood(i+1,j-1);
	}
	return 0;
}
int main()
{
	int t;
	cin>>t;
	
	while(t--)
	{
		cin>>x>>y;
		for(int i=0;i<x;i++)
		{
			cin>>grid[i];
		}
		int ix=0,jx=-1;
		for(int i=0;i<x;i++)
		{
			for(int j=0;j<y;j++)
			{
				if(grid[i][j]=='c')
				{
					ix=i;
					jx=j;
					break;
				}
			}
			if(jx!=-1)
				{
					break;
				}
		}//cout<<ix<<" "<<jx<<endl;
		grid[ix][jx]='.';
		flood(ix,jx);
		for(int i=0;i<x;i++)
		{
			cout<<grid[i]<<endl;
		}
		
	}
	return 0;
}
