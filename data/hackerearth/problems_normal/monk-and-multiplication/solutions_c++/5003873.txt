#include<iostream>
using namespace std;
#define ll long long
int main()
{
    ll n;
    scanf("%lld",&n);
    ll arr[n],a[n];
    arr[0]=-1;
    arr[1]=-1;
    ll fl=-1,sl=-1,tl=-1;
    for(ll i=0;i<n;i++)
    {
    	scanf("%lld",&a[i]);
    	if(a[i]>fl)
    	{
    		tl=sl;
    		sl=fl;
    		fl=a[i];
    		
    	}
    	else if(a[i]>sl)
    	{
    		tl=sl;
    		sl=a[i];
    	}
    	else if(a[i]>tl)
    	tl=a[i];
    	if(i>=2)
    	{
    		arr[i]=fl*sl*tl;
    	}
    	printf("%lld\n",arr[i]);
    }
    return 0;
}
