	ull a[41][41];
	int i,j;
	for (i=1;i<=40;i++) 
	{
		a[i][0]=1; a[i][i]=1; a[i][1]=i;
	}
	for (i=0;i<=40;i++)
		for (j=0;j<=40;j++)
			if (i<j) a[i][j]=0;
	for (i=2;i<=40;i++)
	{
		for (j=2;j<=i;j++)
		{
			a[i][j]=a[i-1][j]+a[i-1][j-1];			
		}
	}
	int t,n,k;
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		cout<<a[n+k-1][n-1]<<endl;
	}
