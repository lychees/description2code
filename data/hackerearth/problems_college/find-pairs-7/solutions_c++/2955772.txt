#include <iostream>
#include <map>
using namespace std;

int main()
{
    int T;
    cin>>T;
    for(int i=0;i<T;i++)
    {
    	int N;
    	cin>>N;
    	map<int,int>count;
    	int n;
    	for(int j=0;j<N;j++)
    	{
    		cin>>n;
    		count[n]++;
    	}
    	n=0;
    	map<int,int>::iterator it;
    	for(it=count.begin();it!=count.end();it++)
    	{
    		n+=it->second*(it->second+1)/2;
    	}
    	cout<<n<<endl;
    }
    return 0;
}
