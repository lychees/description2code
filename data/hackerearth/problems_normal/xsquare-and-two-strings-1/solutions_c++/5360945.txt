#include <iostream>
#include<vector>
#include<algorithm>
#include<cmath>
using namespace std;

int main()
{
  int i=0,j=0,n,t,k,max,temp,min,sum=0;
  string str1,str2;
  cin>>t;
  cin.ignore(); 
  int flag=0;
  while(t--)
  {
  	flag=0;
   getline(cin,str1);
   getline(cin,str2);
   sort(str1.begin(),str1.end());
   sort(str2.begin(),str2.end());
   if(str1.length()<str2.length())
   max=str1.length();
   else 
   max=str2.length();
   for(i=0;i<max;i++)
   {
   	if(str1[i]==str2[i])
   	flag=1;
   }
   if(flag)
   cout<<"Yes\n";
   else
   cout<<"No\n";
  }
  
	return 0;
}
