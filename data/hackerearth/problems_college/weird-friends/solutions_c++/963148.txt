#include<iostream>
using namespace std;
int climb(int a,int b,int n,int c)
{
	if(n>a)
	{
		n = n - a + b;
		c++;
		climb(a,b,n,c);
	}
	else if(n <= a)
	{
		c++;
		return c;
	}
}
int main()
{
	int a,b,n,count=0;
	cin>>n>>a>>b;
	count = climb(a,b,n,count);
	cout<<count;
	return 0;
}
