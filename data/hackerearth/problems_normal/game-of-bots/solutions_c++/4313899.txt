#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define MOD 1000000007
#define MAX 100005
ll modexp(ll a, ll b)
{
	ll res = 1;
	while(b)
	{
		res=(res*a)%MOD;
		a=(a*a)%MOD;
		b/=2;
	}
	return res;
}
string stra,strb;
vector<int> ia,ib,fa,fb;
int main()
{
	int t;
	int i;
	cin>>t;
	while(t--)
	{
		ia.clear();
		ib.clear();
		fa.clear();
		fb.clear();
		cin>>stra>>strb;
		for(i=0;i<stra.size();++i)
		{
			if(stra[i] == 'A') ia.push_back(i);
			else if(stra[i] == 'B') ib.push_back(i);
		}
		for(i=0;i<strb.size();++i)
		{
			if(strb[i] == 'A') fa.push_back(i);
			else if(strb[i] == 'B') fb.push_back(i);
		}
		int flag = 1;
		if(ia.size() == fa.size() && ib.size() == fb.size()) flag = 1;
		else flag = 0;
		for(i=0;i<ia.size()&&flag;++i)
		{
			if(fa[i] > ia[i])
			{
				//puts("No");
				flag = 0;
				break;
			}
		}
		for(i=0;i<ib.size()&&flag;++i)
		{
			if(fb[i] < ib[i])
			{
				//puts("No");
				flag = 0;
				break;
			}
		}
		if(flag) puts("Yes");
		else puts("No");
		
	}
    return 0;
}
