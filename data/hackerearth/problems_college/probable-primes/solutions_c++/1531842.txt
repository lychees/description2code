#include <iostream>
using namespace std;
#include <stdio.h>
long long  int  a[1000001];
long long int b[1000001];
int solve()
{
    long long int  i,j;
    a[0]=a[1]=1;
    for(i=2;i<1000001;i++)
    {
        if(a[i]==0)
        {
            b[i]=1+b[i-1];
            for(j=2;j<=1000001/i;j++)
            {
                a[i*j]=1;
            }
        }
        else b[i]=b[i-1];
    }
    return 0;
}
int main()
{
    long long int  i,t,l,r;
    cin>>t;
    solve();
 //   for(i=0;i<20;i++)cout<<b[i]<<" ";cout<<endl;
    while(t--)
    {
        cin>>l>>r;
        float temp=b[r]-b[l];
        if(a[l]==0)temp++;
        float ans=temp/(r-l+1);
        printf("%.6f\n",ans);
        //cout<<ans<<endl;
    }
    return 0;
}
