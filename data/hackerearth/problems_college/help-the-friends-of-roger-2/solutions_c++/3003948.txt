#include <iostream>
#include<cstring>
using namespace std;

int main()
{
   char s[301];
   cin>>s;
   int m;
   cin>>m;
   while(m--)
   {
   
   	int a,b,cnt=0;
   	cin>>a>>b;
   	for(int i=a-1;i<b-1&&i<strlen(s);i++)
   	{
   		if(s[i]==s[i+1])
   		cnt++;
   	}
   	cout<<cnt<<endl;
   }
   return 0;
}
