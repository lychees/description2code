#define PB push_back
#define FOR(i,a,b) for ( int i = (a); i < (b); ++i )

#define R1(a) scanf("%d",&a)

const int INF = 1e9+7;
vii num;

int main( ) {
  int t, n, k;
  num.PB( { 0, 0 } );
  FOR( i, 1, 10 )
    FOR( j, 0, 10 )
      if ( i != j )
        num.PB( { i, j } );
  for ( cin >> t; t--; ) {
    cin >> n >> k;
    if ( k >= num.size( ) ) {
      cout << -1 << endl;
      continue;
    }
    FOR( i, 0, n )
      cout << (i % 2 ? num[k].ss : num[k].ff);

  }
