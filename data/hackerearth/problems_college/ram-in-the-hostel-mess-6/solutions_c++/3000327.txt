#include<iostream>
using namespace std;
long long fac(long long k)
{
    long long p=1,i;
    for(i=k;i>1;i--)
    {
        p=p*i;
    }
    return p;
}
long long per(long long a,long long b)
{
    long long i,j=1;
    long long c=fac(b);
    for(i=a;i>a-b;i--)
    {
        j=j*i;
    }
    j=j/c;
    return j;
}
int main()
{
    long long int n,t,a,b,c;
    cin>>t;
    while(t--)
    {
        cin>>n;
        c=0;
        for(b=n;b>=n/2;b--)
        {
            c=c+per(b+1,n-b);
        }
        c=c*c;
        cout<<c<<endl;
    }
    return 0;
}
