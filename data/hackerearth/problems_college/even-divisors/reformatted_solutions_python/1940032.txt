import math
s = math.sqrt
t = int(raw_input())
while (t > 0):
    n = int(raw_input())
    sq = s(n)
    i = 1
    c = 0
    while (i <= sq):
        if (n % i == 0):
            k = n/i
            if (k % 2 == 0 and i % 2 == 0):
                c = c+2
            else:
                if (k % 2 == 0 or i % 2 == 0):
                    c = c+1
        i = i+1
    if (sq % 2 == 0 and sq*sq == n):
        c = c-1
    print(c)
    t = t-1
