#include<iostream>
#include<string>
#include<vector>
#include<cstring>
using namespace std;
int main(){
	
	 string s;
	 cin >> s;
	 int q , i ,j;
	 cin >> q;
	 vector< int > ar[30];
	 int val[s.size()+1];
	 memset( val , 0 ,sizeof val);
    for( i = 0; i < s.size(); ++i ) {
           ar[s[i]-'a'].push_back(i);	    
    }          
	 int x;
	 char ch;                      
	while(q--){
		   cin >> x >> ch;
		   x--;
		               val[ar[ch-'a'][x]]=-1;
		               ar[ch-'a'].erase(ar[ch-'a'].begin()+x);
	}                                            	 
	   for( i = 0; i < s.size(); ++i)
	   {
   		   if( val[i] == 0)
   		     cout << s[i];
   	}
   	cout<<endl;
       return 0;	 
}
