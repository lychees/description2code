#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t){
    	int l,i,j,k,idx;
    	char str[200];
    	vector <char> v;
    	char temp;
    	cin >> str;
    	l=strlen(str);
    	for(i=0;i<l;i++)
    	  v.push_back(str[i]);
    	j=l-1;
    	i=j-1;
    	while(i>=0)
    	{
    		if(v[j]>v[i])
    		 {
    		 	temp=v[j];
    		 	idx=j;
    		 	break;
    		 }
    		 j--;
    		 i--;
    	}
    	if(i<0)
    	 printf("no answer\n");
    	else
    	{
    	 for(k=j+1;k<l;k++)
    	 {
    	 	if(v[k]>v[i] && v[k]<v[j])
    	 	  idx=k;
    	 }
    	 temp=v[idx];
    	 v[idx]=v[i];
    	 v[i]=temp;
         sort(v.begin()+j,v.end());
    	 for(i=0;i<l;i++)
    	   cout << v[i];
    	 cout << endl;
    	}
    	 t--;
    }
    return 0;
}
