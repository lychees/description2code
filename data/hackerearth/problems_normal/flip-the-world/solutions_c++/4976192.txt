#include <iostream>

using namespace std;

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int t,n,m,i,j,cnt,x,y;
    char s[30][30];
    cin >> t;
    while (t--)
    {
    	cin >> n >> m;
    	for (i = 0; i < n; i++) cin >> s[i];
    	cnt = 0;
    	for (i = n-1; i >= 0; i--)
    	 for (j = m - 1; j >= 0; j--)
    	 {
    	 	if (s[i][j] == '1') continue;
    	 	cnt ++;
    	 	for (x = 0; x <= i; x++)
    	 	 for (y = 0; y <= j; y++)
    	 	 if (s[x][y] == '1') s[x][y] = '0';
    	 	 else s[x][y] = '1';
    	 }
    	 cout << cnt << " \n";
    }
    return 0;
}
