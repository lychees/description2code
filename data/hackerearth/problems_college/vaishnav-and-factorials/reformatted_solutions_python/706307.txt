tn = raw_input()
t = int(tn)
while t > 0:
    num = raw_input()
    n = int(num)
    count = 0
    fact = n
    n = n-1
    while n > 1:
        fact *= n
        n = n-1
    while (fact > 0):
        rem = fact % 10
        if rem == 4 or rem == 7:
            count = count+1
        fact = fact/10
    t = t-1
    print count
