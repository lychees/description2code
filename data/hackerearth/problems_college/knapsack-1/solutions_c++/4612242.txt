#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    		int n,k;
    		cin>>n>>k;
    		int a[n];
    		for(int i=0;i<n;i++)
    		{
    				cin>>a[i];
    		}
    		int t[n+1][k+1];
    		for(int i=0;i<=n;i++)
    		{
    			t[i][0]=0;
    		}
    		
    		for(int j=0;j<=k;j++)
    		{
    			t[0][j]=0;		
    		}
    		
    		for(int i=1;i<=n;i++)
    		{
    				for(int j=1;j<=k;j++)
    				{
    						if(j<a[i-1])
    						{
    								t[i][j]=t[i-1][j];
    						}
    						else 
    						{
    								t[i][j]=max(t[i-1][j],a[i-1]+t[i][j-a[i-1]]);
    						}
    						
    				}
    		}
    		cout<<t[n][k]<<endl;
    }
    return 0;
}
