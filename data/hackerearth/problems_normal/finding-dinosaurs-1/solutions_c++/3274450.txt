template<typename T> ostream& operator<< ( ostream &o,vector<T> v ) {

        o<<v[0];
    for ( unsigned   i=1; i<v.size(); i++ )
        o<<" "<<v[i];
    return o<<endl;
}

}
template<typename T> istream& operator>> ( istream &in,vector<T> &v ) {
