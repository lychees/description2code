void markMultiples(int arr[], int a, int n)
{
    arr[0]=1;
    int i = 2, num;
    while ( (num = i*a) <= n )
    {
        arr[ num-1 ] = 1;
        ++i;
    }
}

bool Isprime(long n){
    if(n<=1)
        return false;
    for(int i=2;i<=sqrt(n);i++){
        if(n%i==0)
            return false;
    }
    return true;
}
