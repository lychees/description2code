#include <iostream>
using namespace std;
#include<bits/stdc++.h>
#define ll long long
int main()
{
    ll t,i,j,n,x;
    cin>>t;
    assert(1<=t<=10);
    while(t--)
    {
    	cin>>n;
    	assert(1<=n<=100);
    	ll hash[106]={0};
    	ll count=0;
    	for(i=0;i<n;i++)
    	{
    		cin>>x;
    		assert(0<=x<=105);
    		x=x%10;
    		hash[x]++;
    		if(hash[x]>1){count++;}
    	}
    	cout<<count<<endl;
    }
    return 0;
}
