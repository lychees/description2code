#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n;
    cin>>t;
    while(t--){
        cin>>n;
        int a[n+2];
        int b[n];

        int total=0;
        for(int i=0;i<n;i++){
            cin>>a[i];
            total = total + a[i];
            b[i] = total;
        }

        //for(int i=0;i<n;i++)
           // cout<<b[i]<<" ";

        //cout<<endl;
        if(n==1)    {cout<<"YES"<<endl;continue;}

        int flag=0;
        int temp;

        for(int i=1;i<n;i++){
            temp = total - b[i-1]- a[i];
            if(b[i-1]==temp){
                flag=1;
                cout<<"YES"<<endl;
                break;
            }
        }

        if(flag!=1) cout<<"NO"<<endl;

    }
    return 0;
}
