n = int(raw_input())
elms = [int(i) for i in raw_input().split()]

tl, sl, lg = sorted(elms[:3])

res = [-1, -1, tl * sl * lg]

for i in xrange(3, n):
    el = elms[i]
    if el >= lg:
        tl = sl
        sl = lg
        lg = el
    elif el >= sl:
        tl = sl
        sl = el
    elif el >= tl:
        tl = el

    res.append(lg * sl * tl)

print "\n".join(str(i) for i in res)
