#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    scanf("%d",&n);
    while(n--)
    {
        int x,y,f1,b1,t;
        char c;
        getchar();
        scanf("%c %d %d %d %d %d",&c,&x,&y,&f1,&b1,&t);
        if(x==0)
        {
            printf("F0\n");
            continue;
        }
        else if(y==0)
        {
            printf("B0\n");
            continue;
        }
        if(f1==b1)
        {
            if(c=='F')
            {
                if(f1>=x)
                    printf("F%d\n",t*x);
                else
                   printf("SURVIVED\n");
            }
            else if(b1>=y)
            {
                printf("B%d\n",t*y);
            }
            else
                printf("SURVIVED\n");
        }
        else
        {
            int dist;
            if(f1>b1)
            {
                if(c=='F')
                {
                    int t1=f1;
                    dist=f1;
                    while(t1<x)
                    {
                        t1-=b1;
                        dist+=(b1+f1);
                        t1+=f1;
                    }
                    dist=dist-(t1-x);
                    printf("F%d\n",t*dist);
                }
                else if(c=='B')
                {
                    if(b1>=y)
                    {
                        printf("B%d\n",t*y);
                        continue;
                    }
                    else
                    {
                        int t1=f1-b1;
                        dist=f1+b1;
                        while(t1<x)
                        {
                            t1-=b1;
                            dist+=(b1+f1);
                            t1+=f1;
                        }
                        dist=dist-(t1-x);
                    }
                    printf("F%d\n",t*dist);
                }
            }
            else
            {
                if(c=='B')
                {
                    int t1=b1;
                    dist=b1;
                    while(t1<y)
                    {
                        t1-=f1;
                        dist+=(b1+f1);
                        t1+=b1;
                    }
                    dist=dist-(t1-y);
                    printf("B%d\n",t*dist);
                }
                else if(c=='F')
                {
                    if(f1>=x)
                    {
                        printf("F%d\n",t*x);
                    }
                    else
                    {
                        int t1=b1-f1;
                        dist=b1+f1;
                        while(t1<y)
                        {
                            t1-=f1;
                            dist+=(b1+f1);
                            t1+=b1;
                        }
                        //printf("t1=%d dist=%d\n",t1,dist);
                        dist=dist-(t1-y);
                    }
                    printf("B%d\n",t*dist);
                }
            }
        }
    }
    return 0;
}
