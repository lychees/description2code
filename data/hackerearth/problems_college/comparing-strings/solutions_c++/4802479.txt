#include<iostream>
#include<string.h>
using namespace std;
int match(char a[],char b[],int l,int k)
{
int i,j;
for(i=0;i<l;i++)
{
for(j=0;j<k;j++)
{
if(a[i]==b[j])
{
return 1;
}
}
}
return 0;
}
int main()
{
int t;
cin>>t;
while(t>0)
{
char a[100000];
char b[100000];
int l,k,o;
cin>>a;
cin>>b;
l=strlen(a);
k=strlen(b);
o=match(a,b,l,k);
if(o==1)
cout<<"YES"<<endl;
else
cout<<"NO"<<endl;
t--;
}
}
