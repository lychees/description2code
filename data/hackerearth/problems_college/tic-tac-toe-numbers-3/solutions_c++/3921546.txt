#include <bits/stdc++.h>
using namespace std;

bool perfect(long long int n) {
	long long int s=sqrt(n);
	return (s*s == n);
	
}

bool fib(long long int n){
	return perfect(5*n*n + 4) || perfect(5*n*n - 4);
}

bool isPrime(long long int n){
	long long int i;
	if(n==2) return 1;
	for(i=3;i<=sqrt(n);i+=2){
		if(n%i==0) return 0;
	}
	return 1;
}

int main()
{

    int t;
   	cin>>t;
   	
   	while(t--){
   		long long int sum=0,i,n;
   		cin>>n;
   		
   		for(i=0;i<n;i++){
   			if(!fib(i)){
   				sum+=i;
   			}
   		}
   		
   		if(sum%2==0){
   			cout<<"Tac"<<endl;
   		}
   		else{
   			if(isPrime(sum)){
   				cout<<"Tic"<<endl;
   			}
   			else{
   				cout<<"Toe"<<endl;
   			}
   		}
   		
   	}
    return 0;
}
