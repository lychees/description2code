#include <iostream>
using namespace std;

int main()
{
    int T;
    cin>>T;
    
    while(T--)
    {
    	int t1,t2,t3;
    	cin>>t1>>t2>>t3;
    	if(t1==9)
    		cout<<"10\n";
    	else
    	{
    		
    		int cnt=0;
    		if(t1&&t1<=6)cnt++;
    		else if(t1) cnt+=2;
    		if(t2==9)
    		{
    			cnt+=7;
    			cout<<cnt<<"\n";
    		}
    		else
    		{
    			if(t2&&t2<=6)cnt++;
    			else if(t2)cnt+=2;
    			if(t3==9)
    			{
    				cnt+=4;
    				cout<<cnt<<"\n";
    			}
    			else
    			{
    				cnt=0;
    				cout<<cnt<<"\n";
    			}
    		}
    	}
    }
    return 0;
}
