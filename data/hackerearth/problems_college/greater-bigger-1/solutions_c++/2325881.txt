#include <iostream>
#include <string>
#include <algorithm>
using namespace std;
void permute(string s){
	int index=-1,check=0;
	//cout<<s<<endl;
	int i;
	for(i=1;i<s.length();i++){
		if(s[i-1]<s[i]){
			check=1;
			index=i-1;
		}
	}
	if(!check){
	 cout<<"no answer"<<endl;
		return;
	}
	int k=-1;
	for(int i=index+1;i<s.length();i++)
		if(s[index]<s[i])
			k=i;
	if(k!=-1){
	char c=s[index];
	s[index]=s[k];
	s[k]=c;
	reverse(s.begin()+index+1,s.end());
	cout<<s<<endl;
	}
	return;
}
int main(){
	int a;
	cin>>a;
	while(a--){
	string s;
	cin>>s;
	permute(s);
}
}
