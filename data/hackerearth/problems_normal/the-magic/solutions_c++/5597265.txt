#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll countBits (ll n)
{
    ll c=0;
    while(n)
    {
        n=n&(n-1);
        c++;
    }
    return c;
}
int main()
{
    ll t,n;
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cin>>t;
    assert(t>=1 && t<=100000);
    while(t--)
    {
        cin>>n;
        assert(n>=0 && n<=1000000000);
        cout<<countBits(n)<<endl;
    }
    return 0;
}
