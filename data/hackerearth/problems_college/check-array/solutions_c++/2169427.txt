#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t; cin >> t;
    while(t--)
    {
    	int n; cin >> n;
    	int a[n],sum[n]; cin >> a[0];
    	sum[0] = a[0];
    	for(int i=1;i<n;i++)
    	{
    		cin >> a[i];
    		sum[i] = sum[i-1] + a[i];
    	}
    	int i;
    	for( i=1;i<n-1;i++)
    	{
    		if(sum[i-1] == sum[n-1] - sum[i])
    			break;
    	}
    	cout << (i==n-1?"NO":"YES") << endl;
    }
    return 0;
}
