# print 'Hello World!'
tc = input()

while tc > 0:
    lst = [int(i) for i in raw_input().split()]
    newlst = sorted(lst[1:], reverse=True)
    l = len(newlst)
    maxsum = minsum = newlst[0]+newlst[-1]
    for i in range(1, l-1):
        temsum = newlst[i]+newlst[l-1-i]
        if temsum > maxsum:
            maxsum = temsum

    for i in range(1, l-1):
        temsum = newlst[i]+newlst[l-1-i]
        if temsum < minsum:
            minsum = temsum

    print maxsum-minsum

    tc -= 1
