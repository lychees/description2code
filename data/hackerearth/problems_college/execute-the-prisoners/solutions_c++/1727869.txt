#include <iostream>
using namespace std;

int a[100005];
int main()
{
    int t;
    cin >> t;
    a[0] = 0;
    a[1] = 1;
    for (int i = 2; i < 100001; i++) {
    	if (i % 2 == 0) {
    		a[i] = 2 * a[i / 2] - 1;
    	} else {
    		a[i] = 2 * a[(i - 1) / 2] + 1;
    	}
    }
    while (t--) {
    	int n;
    	cin >> n;
    	cout << a[n] << endl;
    	 
    }
    return 0;
}
