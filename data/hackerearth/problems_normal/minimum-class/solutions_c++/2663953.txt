#include <iostream>
#include <string>
#include <assert.h>
using namespace std;
int T;
int n;
int main(){
	cin>>T;
	assert(1<=T && T<=1000);
	while(T--){
		cin>>n;
		assert(1<=n && n<=((int)1e9));
		for(long long i=0;i<=n;i++){
			if(i%2==0){
				long long t=i;
				t/=2;
				if(t*(t+1)>=n){
					cout<<i<<endl;
					break;
				}
			} else {
				long long t=i;
				t/=2;
				if(t*(t+1)+t+1>=n){
					cout<<i<<endl;
					break;
				}
			}
		}
	}
	string a;
	if(cin>>a)
		assert(false);
}
