#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int i,n,m;
    cin >> n >> m;
    vector<int> v;
    for(i=n+1;i<m;i++)
    {
    	if(i%15==0)
    	   {
    	   	  v.push_back(i);
    	   }
    	else if(i%3!=0 && i%5!=0)
    	{
    		v.push_back(i);
    	}
    }
    int l = v.size();
    for(i=0;i<l-1;i++)
    {
        cout << v[i] << ",";
    }
    cout << v[l-1];
    return 0;
}
