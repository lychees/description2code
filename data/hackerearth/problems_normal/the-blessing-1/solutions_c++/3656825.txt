#include <iostream>
#include <math.h>

using namespace std;

int main()
{
    long t;
    long long p,m,j=1;
    double long x,rem=0;
    cin>>t;
    for(long i=0;i<t;i++)
    {
        cin>>p>>m;
        rem=p;
        do
        {
            x=(p+j*m)/pow(2,j);
            j++;
            rem+=x;
        }while(x>0.000001);
        j=1;
        m=rem;
        cout<<m+1<<endl;
    }
    return 0;
}
