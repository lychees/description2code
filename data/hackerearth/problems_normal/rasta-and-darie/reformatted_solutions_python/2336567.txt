from fractions import gcd
t = input()
while t != 0:
    n, p, k = map(int, raw_input().split())
    hcf = gcd(n, p)
    if hcf == 1:
        if k > n:
            print -1
        else:
            print k
    else:
        val = 1+(k-1)*hcf
        if (k > (n/hcf)):
            print -1
        else:
            print val
    t = t-1
