#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		int n;
		cin >> n;
		set<string> s;
		string stng;
		for(int i=0;i<n;i++)
		{
			cin >> stng;
			s.insert(stng);
		}
		set<string>::iterator it;
		for(it=s.begin();it!=s.end();it++)
		{
			cout << *it << endl;
		}
	}
	return 0;
}
