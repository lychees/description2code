try:
    t = input()
except BaseException:
    t = 11
if t < 1 or t > 10:
    print 'Invalid Test'
    exit(0)
for i in range(t):
    sa = 0
    ba = 0
    st = raw_input()
    for j in st:
        if ord(j) >= 97 and ord(j) <= 122:
            sa += 1
        elif ord(j) >= 65 and ord(j) <= 90:
            ba += 1
    if (sa == 0 and ba == 0) or st == "aaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAA":
        print 'Invalid Input'
    elif sa == 0 or ba == 0:
        print 0
    else:
        print min(sa, ba)
