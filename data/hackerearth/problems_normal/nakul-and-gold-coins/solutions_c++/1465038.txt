int prime[1000005],k;
vi ar;
void seive()
{
	for(int i=2;i<=1000;i++)
	{
		if(!prime[i])
			for(int j=i*i;j<=1000000;j+=(i+i))
				prime[j]=1;
	}
	prime[1]=2;
	k=2;
	for(int i=3;i<=1000000;i+=2)
		if(!prime[i])
			prime[k++]=i;
}
