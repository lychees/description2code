#include <bits/stdc++.h>
using namespace std;
int b[100001];
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	memset(b,0,sizeof(b));
    	int n;
    	cin>>n;
    	int a[n];
    	for(int i=0;i<n;i++)
    	cin>>a[i];
    	//i=0;
    	int j=0,i=0;
    	while(1)
    	{
    		if(i==-1 or a[0]==-1)
    		{
    			j=1;
    			break;
    		}
    	else if(b[i]==0)
    		{
    			b[i]=1;
    		    i=a[i];
    		}
    	else if(b[i]==1)
        	break;
    	}
    	if(j)
    	cout<<"Win"<<endl;
    	else
    	cout<<"Lose"<<endl;
    }
}
