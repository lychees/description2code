#include <bits/stdc++.h>
using namespace std;
int m,n,arr[101] ;
bool dp[10021];
int main()
{
    int t;
   // ctr =0;
    cin >> t;
    while(t--) {
    	//ctr++;
    	memset(dp,false,sizeof dp);
    	scanf("%d%d", &m,&n);
    	for(int i =0; i < n; i++) {
    		scanf("%d",&arr[i]);
    		dp[arr[i]] = true;
    	}
    	
    	for(int i =1 ; i <= m; i++) {
    		if(dp[i] == false) {
    			for(int j = 0; j < n; j++) {
    				if(i - arr[j] >= 0 && dp[i -arr[j]] == false) {
    					dp[i] = true;
    					break;
    				}
    			}
    		}
    	}
    	if(dp[m]) {
    		cout << "Little Deepu\n";
    	}
    	else {
    		cout << "Kate\n";
    	}
    	
    }
    return 0;
}
