#include <iostream>
#include <bits/stdc++.h>

using namespace std;

string karan(string s){
	int i = 0,size = s.size();
	if(size == 0 || size == 1) return s;
	string ans = "";
	char ref = s[0];
	ans.push_back(s[0]);
	for(i = 1;i < size;++i){
		if(s[i] != ref){
			ref = s[i];
			ans.push_back(s[i]);
		}
	}
	return ans;
}

int main()
{
	int t = 0;
	string s;
	cin>>t;
	while(t--){
		cin>>s;
		cout<<karan(s)<<endl;
	}
    return 0;
}
