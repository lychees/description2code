#include<bits/stdc++.h>
#define fast ios::sync_with_stdio(false); cin.tie(0); cout.tie(0)
using namespace std;
int main()
{
	long long t,n,q;
	cin>>t;
	while(t--)
	{
		cin>>n>>q;
		q--;
		long long ans=n*n-(n%q)*((n+q-1)/q)*((n+q-1)/q);
		ans=ans-(q-n%q)*(n/q)*(n/q);
		ans/=2;
		ans=(n*(n-1))/2-ans;
		cout<<ans<<endl;
	}
}
