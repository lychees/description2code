#include<iostream>
#include<iomanip>
using namespace std;
double cal(double n, double r){
	return (3*n*(n+1)/2/r/((2*n)+1));
}
int main(){
	int t;
	double r, n;
	double ans = 1;
	cin>>t;
	while(t--){
		cin>>n>>r;
		ans = ans*cal(n,r);
	}
	cout<<fixed<<setprecision(4)<<ans;
	return 0;
}
