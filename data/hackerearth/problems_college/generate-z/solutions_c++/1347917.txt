#include<bits/stdc++.h>
using namespace std;

int main()
{
    int i,j,n;
    cin>>n;
    for(i=0;i<n;i++)
    {
        cout<<"*";
    }
    cout<<endl;
    for(i=1;i<n-1;i++)
    {
        for(j=i;j<n-1;j++)
        {
            cout<<" ";
        }
        cout<<"*"<<endl;
    }
     for(i=0;i<n;i++)
    {
        cout<<"*";
    }
    cout<<endl;
    return 0;

}
