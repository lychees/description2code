arr = [0]*1000003
cnt = [0]*1000003


def sieve():
    global arr, cnt
    for i in xrange(2, 1002):
        if arr[i] == 0:
            j = 2
            arr[i] = i
            cnt[i] += 1
            while ((i*j) < 1000001):
                if (arr[i*j] == 0):
                    arr[i*j] = i
                    cnt[i] += 1
                j += 1


sieve()
for i in range(1001, 1000001):
    if arr[i] == 0:
        cnt[i] += 1
