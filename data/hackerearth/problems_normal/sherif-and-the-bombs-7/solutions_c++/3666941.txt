#include <bits/stdc++.h>
using namespace std;
#define N 1000
int pts[N][N];

int maxx(int x, int y) {
	return (x>y ? x : y);
}

int minn(int x, int y) {
	return (x<y ? x : y);
}

int main() {
	int n;
	cin>>n;
	int xmax=INT_MIN, xmin=INT_MAX;
	int ymax=INT_MIN, ymin=INT_MAX;
	int x[n], y[n], r[n];
	int count=0;
	for(int i=0;i<n;i++) {
		cin>>x[i]>>y[i]>>r[i];
		xmax = maxx(xmax, x[i]+r[i]);
		ymax = maxx(ymax, y[i]+r[i]);
		xmin = minn(xmin, x[i]-r[i]);
		ymin = minn(ymin, y[i]-r[i]);
	}
	xmax = minn(xmax, N);
	ymax = minn(ymax, N);
	ymin = maxx(ymin, 1);
	xmin = maxx(xmin, 1);
	
	for(int i=xmin;i<=xmax;i++) {
		for(int j=ymin;j<=ymax;j++) {
			int c=0;
			for(int k=0;k<n;k++) {
				int val;
				val= r[k]*r[k] - pow(i-x[k],2) - pow(j-y[k], 2);
				if(val>=0) {
					c++;
				}
				if(c==2) {
					count++;
					break;
				}
			}
		}
	}
	cout<<count;
	return 0;
}
