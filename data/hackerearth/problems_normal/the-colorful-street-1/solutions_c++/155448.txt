#include<iostream>
using namespace std;
int mini;
int a[25][3],n;

void recur(int i,int v,int c)
  {if(i==n)
    {if(v<mini)
      mini=v;
      return;
    }
    if(v>=mini)
     return;
    int j;
  for(j=0;j<=2;j++)
    {if(c==j) continue;
     recur(i+1,v+a[i][j],j);
    }
  }
int main()
 {int t,i;
 cin>>t;
 while(t--)
   {cin>>n;
    mini=21*1000;
    for(i=0;i<n;i++)
      cin>>a[i][0]>>a[i][1]>>a[i][2];
    recur(0,0,0);
    recur(0,0,1);
    recur(0,0,2);
    
    cout<<mini<<endl;
   }
 }
