#include <iostream>
using namespace std;

int main()
{
    int n,k,i,j;
    cin>>n>>k;
    
    int hash[n][26];
    
    for(i=0;i<n;i++)
    {
    	for(j=0;j<26;j++)hash[i][j]=0;
    }
    
    string s[n];
    
    for(i=0;i<n;i++)
    {
    	cin>>s[i];
    	
    	for(j=0;j<s[i].length();j++)
    	{
    		hash[i][s[i][j]-97]++;
    	}
    }
    
    int ans = 0;
    bool flag;
    for(j=0;j<26;j++)
    {
    	flag = true;
    	for(i=0;i<n;i++)
    	{
    		if(!hash[i][j])
    		{
    			flag = false;
    			break;
    		}
    	}
    	
    	if(flag)ans++;
    }
    
    if(ans>=k)
    {
    	cout<<":-)";
    }
    else
    {
    	cout<<":-(";
    }
    	
    
    return 0;
}
