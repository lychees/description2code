#include <iostream>
#include <string>
#include <stdio.h>

using namespace std;

int main()
{
	
	int t, i, n1;
	
	cin >> t;
	
	while( t-- ) {
		
		string s1, s2;
		
		cin >> s1;
		cin >> s2;
		
		n1 = s1.length();
	
		for( i = 0;i < n1;i++ ) {
			
			if( s1[ i ] == '0' && s2[ i ] == '0' )
				cout << '0';
			else if( s1[ i ] == '0' && s2[ i ] == '1' )
				cout << '1';
			else if( s1[ i ] == '1' && s2[ i ] == '0' )
				cout << '1';
			else if( s1[ i ] == '1' && s2[ i ] == '1' )
				cout << '0';
		}
		
		cout << endl;
	}
	return 0;
}
