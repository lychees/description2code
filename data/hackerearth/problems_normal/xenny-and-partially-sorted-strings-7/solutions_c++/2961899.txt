#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int M;
bool mycompare(const string &s1,const string &s2)
{
	string s1_sub = s1.substr(0, M);
	string s2_sub = s2.substr(0, M);
	return s1_sub < s2_sub;
}

int main()
{
	ios_base::sync_with_stdio(false);
	int T,N,K;
	string s1[1005];
	cin>>T;
	while (T--) {
		cin>>N>>K>>M;
		for (int i=0; i<N; i++) {
			cin>>s1[i];
		}
		sort(s1, s1+N, mycompare);
		cout<<s1[K-1]<<"\n";
	}
	return 0;
}
