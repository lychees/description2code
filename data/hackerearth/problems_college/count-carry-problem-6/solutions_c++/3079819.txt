                                                                           

#include <iostream>
using namespace std;
int main()
{
 int t; cin>>t;
 while(t--)
 {
  long long a,b; cin>>a>>b;
   int dig1,dig2,carry;
     carry=0;
    int count=0;
   while(a!=0 || b!=0)
   {
     dig1=a%10;    dig2=b%10;
      a=a/10;     b=b/10;
       if(dig1+dig2+carry>9)
        {
           count++;
          carry=(dig1+dig2+carry)/10;
         }
        else
        carry = 0;
  }
     if(count==0)
     cout<<"No carry operation\n";
    else if(count==1)
     cout<<"1 carry operation\n";
    else
    cout<<count<<" carry operations\n";   
}
return 0;
}
