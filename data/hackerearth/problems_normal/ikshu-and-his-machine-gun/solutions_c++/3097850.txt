#include <iostream>
#include <vector>
#include <algorithm>
#include <bits/stdc++.h>

using namespace std;

int main()
{
   	int s;
   	cin>>s;
   	vector <int> boxes (s);
   	for(int k = 0; k < s; k++){
   		cin>>boxes[k];
   	}
   	sort(boxes.begin(),boxes.end());
   	
   	for(int k = 0; k < boxes.size();k++ ){
   		cout<<boxes[k] - k<<" ";
   	}
   	
    return 0;
}
