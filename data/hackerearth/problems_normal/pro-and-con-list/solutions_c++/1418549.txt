#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    long long i,sum,max,h,n,c[100008],a[100008],b[100008];
    int t;
    
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(i=0;i<n;i++)cin>>a[i]>>b[i];
        sum=0;
        for(i=0;i<n;i++)
        {
            c[i]=a[i]+b[i];
             sum=sum+b[i];
        }
        sort(c,c+n);
        h=c[n-1]+c[n-2]-sum;
        
    cout<<h<<"\n";
    }
    
}
