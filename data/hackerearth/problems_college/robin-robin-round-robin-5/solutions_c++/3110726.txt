#include <iostream>
#include <cstdio>
#include <vector>
#include <algorithm>
#include <cstring>
#include <cmath>

using namespace std;

#define in(n) scanf("%lld",&n)
#define MOD 1000000007
#define MAXN 100005


int main()
{
	vector<pair<long long,long long> > r;
long long rank[100005];
long long first[100005];
long long a[100005];

	//freopen("input.txt","r",stdin);
	long long n,c,i,q,m,d,steps,time;
	in(n);
	in(c);
	for(i=1;i<=n;i++)
	{
		in(a[i]);
	}
	
	for(i=1;i<=n;i++)
	{
		steps=(a[i]+c-1)/c;
		r.push_back(make_pair(steps,i));
	}

	sort(r.begin(),r.end());
	for(i=0;i<r.size();i++)
		rank[r[i].second]=i+1;

	time=0;
	for(i=1;i<=n;i++)
	{
		first[i]=time;
		time+=min(c,a[i]);
	}

	in(q);
	for(i=0;i<q;i++)
	{
		in(m);
		in(d);
		if(m==1)
			cout<<first[d]<<endl;
		else
			cout<<rank[d]<<endl;
	}

	return 0;
}
