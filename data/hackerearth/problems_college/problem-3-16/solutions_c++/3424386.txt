#include<bits/stdc++.h>
#define ll long long
using namespace std;


int main()
{
    int t;
    long int n,k;
    scanf("%d",&t);
    while(t--)
    {
        cin>>n>>k;
        int ar[k+1];
        memset(ar,0,sizeof(ar));
        int x,y;
        for(int i=0; i<n; i++)
        {
            cin>>x;
            ar[x]++;
        }
       sort(ar,ar+k+1);
     // cout<<ar[k]<<endl;
       y=ar[k];
       cout<<n-y<<endl;

    }
    return 0;
}
