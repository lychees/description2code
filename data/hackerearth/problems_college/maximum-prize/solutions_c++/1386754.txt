#include <bits/stdc++.h>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t;
   cin>>t;
   while(t--)
   {
   	long long n,k;
   	cin>>n>>k;

   	long long a[n];
   	long long i;
   	for(i=0;i<n;i++)
   	cin>>a[i];
   	sort(a,a+n);
   	i=n-1;
   	long long s=0;
   	for(long long j=0;j<min(n,k);j++)
   	{
   		s+=a[i];
   		i--;
   	}
   	
   cout<<s<<endl;
   }
    return 0;
}
