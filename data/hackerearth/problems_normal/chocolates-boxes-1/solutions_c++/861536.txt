#include <iostream>
using namespace std;
int check[101];
int main()
{
    int test,i,n,cnt=0,*a,k;
    cin>>test;
    while(test--)
    {
                 cin>>n;
                 a = (int *)malloc(n*sizeof(int));
                 for(i=0;i<=100;i++)
                 check[i] = 0;
                 for(i=0;i<n;i++)
                 {
                                 cin>>a[i];
                                 check[a[i]] = check[a[i]] + 1;
                 }
                 cin>>k;
                 cnt = 0;
                 for(i=0;i<n;i++)
                 {
                                 if(a[i] == k - a[i] && check[a[i]])
                                 {
                                                //cout<<"1st part"<<a[i]<<"\n";
                                                cnt = cnt + check[a[i]] - 1;
                                                check[a[i]] = check[a[i]] - 1;
                                 }
                                 else
                                 {
                                     if(k-a[i] > a[i] && check[k-a[i]])
                                     {
                                               //cout<<"2nd part\n";
                                               cnt = cnt + check[a[i]]*check[k-a[i]];
                                               check[k-a[i]] = 0;
                                     }
                                 }
                 }
                 cout<<cnt<<"\n";
    }
    return 0;
}
