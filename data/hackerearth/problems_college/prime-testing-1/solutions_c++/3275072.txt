#include <iostream>
using namespace std;

int main()
{
    int t,i,j,n,x,count;
    cin>>t;
    int prime[1001];
    prime[0]=0;
    prime[1]=0;
    for(i=2;i<=1000;i++)
    prime[i]=1;
    for(i=2;i*i<=1000;i++)
    {
    	if(prime[i]==1)
    	{
    		for(j=2*i;j<=1000;j+=i)
    		prime[j]=0;
    	}
    }
    while(t--)
    {
    	cin>>n;
    	count=0;
    	for(i=0;i<n;i++)
    	{
    		cin>>x;
    		if(prime[x]==1)
    		count++;
    		
    	}
    	cout<<count<<"\n";
    	
    	
    }
    return 0;
}
