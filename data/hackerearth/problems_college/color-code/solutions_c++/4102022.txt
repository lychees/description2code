#include <iostream>
#include <bits/stdc++.h>
using namespace std;
void prt(int i){
	if(i<=9)
	cout<<i;
	else{
		switch(i){
			case 10:
			  cout<<"U";
			  break;
			case 11:
			  cout<<"V";
			  break;
			case 12:
			  cout<<"W";
			  break;
			case 13:
			  cout<<"X";
			  break;
			case 14:
			  cout<<"Y";
			  break;
			case 15:
			  cout<<"Z";
			  break;
		}
	}
}
void convertToHex(int i)
{
	int q,r;
	q=i/16;
	r=i%16;
	if(q>15)
	{
		convertToHex(q);
	}else{
		prt(q);
		prt(r);
	}
	
}
int main()
{
    
    int r,g,b,t;
    int count;
    cin>>t;
    count = t;
    while(t--)
    {
    	cin>>r>>g>>b;
    	cout<<"Case "<<count-t<<": ";
    	convertToHex(r);
    	cout<<" ";
    	convertToHex(g);
    	cout<<" ";
    	convertToHex(b);
    	cout<<"\n";
    }
    return 0;
}
