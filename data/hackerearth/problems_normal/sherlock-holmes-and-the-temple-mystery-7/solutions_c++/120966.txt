void input()
{
    int n;
    cin>>n;
    int ele[n];
    int tot = 0;
    for(int i = 0;i < n;++i)
    {
        cin>>ele[i];
        tot += ele[i];
    }
    int dp[tot + 1];
    for(int sum = 0;sum <= tot;++sum)
    {
        dp[sum]=0;
    }
    dp[0]=1;
    for(int i = 0;i < n;++i)
    {
        for(int sum = tot;sum>= ele[i];--sum)
        {
            dp[sum]|=dp[sum-ele[i]];
        }
    }
    if(tot&1)
    {
        cout<<"NO\n";
    }
    else if(dp[tot/2])
    {
        cout<<"YES\n";
    }
    else
    {
        cout<<"NO\n";
    }
}


int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        input();
    }
    return 0;
}
