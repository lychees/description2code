#include <iostream>
#include<string.h>
using namespace std;
#define M 1000000007
int main()
{
    char A[1000002],B[1000002];
    long long t,count,i,j,lena,lenb;
    cin>>t;
    getchar();
    while(t--)
    {
        cin>>A>>B;
        count=1;
        lena=strlen(A);
        lenb=strlen(B);
        int hash[257]={0};
        for(i=0;i<lena;i++)
        {
            hash[A[i]]++;
        }
        for(i=0;i<lenb;i++)
        {
           count=((count%M)*(hash[B[i]]%M))%M;
           hash[B[i]]--;
        }
        cout<<count<<endl;
    }
	return 0;
}
