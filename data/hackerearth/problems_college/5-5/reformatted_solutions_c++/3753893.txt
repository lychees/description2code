#include <bits/stdc++.h>
using namespace std;
#define ll long long

ll sv[1000100], s, k;
void pr() {
  ll r = 1;
  int k = 1;
  sv[0] = 1;
  while (r <= 10000000000ll) {
    r *= 2;
    sv[k++] = r;
  }
  s = k;
}
int main() {
  ll ts, n, l, Ans;
  pr();
  cin >> ts;
  while (ts--) {
    cin >> n;
    Ans = 0;
    l = lower_bound(sv, sv + s, n) - sv;
    // cout<<"L "<<l<<endl;
    for (int i = l; i >= 0; i--) {
      if (n >= sv[i]) {
        Ans++;
        n -= sv[i];
      }
      // cout<<sv[i]<<endl;
    }
    Ans += n;
    cout << Ans << endl;
  }
  return 0;
}
