template <typename T, typename U> 
inline void chkmax(T& x, U y) {
	if (x < y) x = y;
}

template <typename T, typename U>
inline void chkmin(T& x, U y) {
	if (y < x) x = y;
}

const int MOD = 1e9 + 7;
const int MAXN = 3505;
int f[MAXN*MAXN], cnt[MAXN*5];
int n, m;

int main() {
	
	f[0] = 1;
	for (int i = 1; i < MAXN*MAXN; i++) f[i] = (LL)i * f[i - 1] % MOD;
	int ncase;
	for (scanf("%d", &ncase); ncase--;) {
		scanf("%d%d", &n, &m);
		memset(cnt, 0, sizeof cnt);
		for (int i = 1; i <= n; i++) for (int j = 1; j <= m; j++) {
			cnt[(i|j)]++;
		}
		LL ans = 1;
		for (int i = 1; i < 5 * MAXN; i++) {
			ans = ans * f[cnt[i]] % MOD;
		}
		cout << ans << endl;
	}
	return 0;
}
