#include <iostream>
using namespace std;

int main()
{
	int a[10]={6,2,5,5,4,5,6,3,7,6};
	int t,sum,i;
	cin>>t;
	while(t--)
	{
			sum=0;
		    char str[10000];
		    cin>>str;
		    for(i=0;str[i]!='\0';i++){
		    	sum=sum+a[str[i]-48];
		    }
		    cout<<sum<<endl;		
	}

    return 0;
}
