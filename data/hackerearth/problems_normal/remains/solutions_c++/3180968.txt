#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		unsigned long long int x,y,n;
		cin>>x>>y>>n;
		int a[2];
		unsigned long long int sum = x+y;
		int k=0;
			a[0] = x;
			a[1] = y;
		for(int i=2;i<n;i++)
		{
			if(a[0] ==0 && a[1]!=0 && i>2)
			{
			sum = sum - a[1];
			if((n-i+2)%3==0)
			{
				k=1;
				sum = sum + ((n-i+2)/3)*2*a[1];
			}
			else if( (n-i+2)%3 == 1)
			{
				k=1;
				sum = sum + ((n-i+2)/3)*2*a[1]; 
			}
			else if((n-i+2)%3 ==2)
			{
				k=1;
				sum = sum + ((n-i+2)/3*2)*a[1]+a[1];
			}
			break;
			}
			int p = abs(a[1] - a[0]);
			a[0] = a[1];
			a[1] = p;
			sum =sum + p;
		}
		cout<<sum<<"\n";
	}
}
