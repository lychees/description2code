#include <iostream>
using namespace std;
bool isPrime(long long int number){

    if(number < 2) return false;
    if(number == 2) return true;
    if(number % 2 == 0) return false;
    for(long long int i=3; (i*i)<=number; i+=2){
        if(number % i == 0 ) return false;
    }
    return true;

}
int main()
{
  long long int N,T;
  cin>>T;
  while(T--)
  {
  	cin>>N;
  	if(isPrime(N)) cout<<"YES"<<endl;
  	else cout<<"NO"<<endl;
  }
  return 0;
}
