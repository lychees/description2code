#include<iostream>
#include<cstdio>
#include<cmath>
#include<cstring>
using namespace std;
int main()
{	int N;
	cin>>N;
	string str[N];
	cin.ignore();
	for(int i=0;i<N;i++)
		getline(cin,str[i]);
	for(int i=0;i<N;i++)
	{	int L=strlen(&str[i][0]),S;
		S=sqrt(L);
		char nstr[S][S];
		for(int j=0;j<S;j++)
			for(int k=0;k<S;k++)
				nstr[j][k]=str[i][(S*j)+k];
		for(int j=0;j<S;j++)
			for(int k=0;k<S;k++)
				str[i][(S*j)+k]=nstr[k][j];
		}
	for(int i=0;i<N;i++)
		cout<<str[i]<<"\n";
	return 0;
	}
