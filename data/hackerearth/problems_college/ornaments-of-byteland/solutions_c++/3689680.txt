#include <cstdio>
#include <cstring>

using namespace std;

int main()
{
	int N, ans = 0;
	int special[129];
	char s[101];

	memset(special, -1, sizeof(special));

	scanf("%d", &N);

	for (int j = 0; j < N; j++)
	{
		scanf("%s", s);

		for (int i = 0; i < strlen(s); i++)
		{
			if (special[s[i]] == j-1)
				special[s[i]] = j;
		}
	}

	for (int i = 0; i <= 128; i++)
		if (special[i] == N-1)
			ans++;

	printf("%d\n", ans);

	return 0;
}
