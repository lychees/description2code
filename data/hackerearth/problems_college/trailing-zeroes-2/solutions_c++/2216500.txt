#include <iostream>
using namespace std;
 
int findTrailingZeros(int  n)
{
    // Initialize result
    int count = 0;
 
    // Keep dividing n by powers of 5 and update count
    for (int i=5; n/i>=1; i *= 5)
          count += n/i;
 
    return count;
}
 
// Driver program to test above function
int main()
{
    int t;
    cin>>t;
    int n;
    for(int i=0;i<t;i++){
    	cin>>n;
    	cout <<findTrailingZeros(n)<<endl;;
    }
    
    return 0;
}
