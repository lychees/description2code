#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin>>n;
    for(int i=0;i<n;i++)
    {
    	string a,b,c;
    	cin>>a>>b;
    	int flag=0;
    	for(int j=0;j<a.size();j++)
    	{
    		c=a.substr(0,j)+a+a.substr(j);
    		if(c==b)
    		{
    			flag=1;
    			cout<<"Possible"<<"\n";
    			break;
    		}
    	}
    	if(flag==0)
    	cout<<"Impossible"<<"\n";
    }
    return 0;
}
