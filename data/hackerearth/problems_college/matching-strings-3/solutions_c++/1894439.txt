#include <iostream>
using namespace std;
#include<string.h>
int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {int i,l1,l2,l,count=0;
        string s1,s2;
        cin>>s1>>s2;
        l1=s1.length();
        l2=s2.length();
        l=min(l1,l2);
        for(i=0;i<l;i++)
        {
            if(s1[i]==s2[i])
                count++;
        }
        cout<<count<<"\n";
    }

}
