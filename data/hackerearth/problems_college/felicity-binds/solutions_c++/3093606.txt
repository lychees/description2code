#include <bits/stdc++.h>
using namespace std;
long long pro(int a,int b){
  long long res=0;
  int x=a,y=b;
  while(y>0){
  if(y&1)
  res+=x;
  x=x<<1;
  y=y>>1;
  }
 
return res;
}
int main()
{ int t;
cin>>t;
int n;
while(t--){
cin>>n;
int a[n],b[n];
for(int i=0;i<n;i++){
	cin>>a[i];
}
for(int i=0;i<n;i++){
	cin>>b[i];
}
sort(a,a+n);
sort(b,b+n);
long long sum=0;
for(int i=0;i<n;i++){
	sum+=pro(a[i],b[i]);
}
cout<<sum<<"\n";
}
    return 0;
}
