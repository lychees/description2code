#include <iostream>
#include<string.h>
using namespace std;

int main()
{
	char arr[1000];
	int n,c,i,len,x,y;
	cin>>n;
	while(n--)
	{
		cin>>arr;
		len=strlen(arr);
		c=0;
		for(i=1;i<len;i++)
		{
			x=arr[i]-arr[i-1];
			if(x<0)
			x=x*(-1);
			y=arr[len-1-i]-arr[len-i];
			if(y<0)
			y=y*(-1);
			if(x==y)
			c++;
		}
		if(c==len-1)
		cout<<"Funny"<<endl;
		else
		cout<<"Not Funny"<<endl;
	}
    return 0;
}
