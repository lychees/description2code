t = int(raw_input())
k = 1000000007
while (t > 0):
    m, n = (raw_input().split())
    m, n = [int(m), int(n)]
    res = 1
    if m >= n-1:
        for i in range(m+1-n+1, m+2):
            res = (res*i) % k
        for i in range(1, m+1):
            res = (res*i) % k
        print(res)
    else:
        print(-1)
    t = t-1
