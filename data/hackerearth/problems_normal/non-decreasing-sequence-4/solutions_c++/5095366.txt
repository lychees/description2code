#include <bits/stdc++.h>

using namespace std;

const int N = 1e6 + 10;
const int inf = 1e9;

int a[N], res[N];
int fenw[N];
int esz = N - 1;

void modify(int x, int v) {
	while (x <= esz) {
		if (v > fenw[x]) fenw[x] = v;
		x = (x | (x - 1)) + 1;
	}
}

int find_max(int x) {
	int v = -inf;
	while (x > 0) {
		if (fenw[x] > v) v = fenw[x];
		x &= x - 1;
	}
	return v;
}

int main() {
	int tt;
	scanf("%d", &tt);
	while (tt--) {
		fill(fenw, fenw + N, 0);
		int n, k;
		scanf("%d", &n);
		for (int i = 0; i < n; i++) {
			scanf("%d", &a[i]);
			a[i]++;
		}
		scanf("%d", &k);
		fill(res, res + n, 0);
		for (int i = 0; i < n; i++) {
			if (a[i] == 1) {
				res[a[i]] = 1;
			} else {
				res[a[i]] = find_max(a[i] - 1) + 1;
			}
			modify(a[i], res[a[i]]);
		}
		int ans = inf;
		for (int i = 0; i < n; i++) {
			if (res[a[i]] == k) {
				ans = min(ans, a[i] - 1);
			} 
		}
		printf("%d\n", ans == inf ? -1 : ans);
	}
	return 0;
}
