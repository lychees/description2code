#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t; cin>>t; while(t--)
    {
    	map<char,int>m;
    	string str1,str2;
    	cin>>str1>>str2;
    	int l1=str1.length();
    	int l2=str2.length();
    	int flag=0;
    	for(int i=0;i<l1;i++)
    	{
    		m[str1[i]]=1;
    	}
    	for(int i=0;i<l2;i++)
    	{
    		if(m[str2[i]]==1) 
    		{
    			flag=1;
    			break;
    		}
    	}
    	if(flag) cout<<"YES\n";
    	else cout<<"NO\n";
    }
    return 0;
}
