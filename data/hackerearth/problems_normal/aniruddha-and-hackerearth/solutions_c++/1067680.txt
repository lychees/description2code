#include<bits/stdc++.h>
using namespace std;
#define M 1000000007
typedef long long LL;
LL a[1000001];
int main()
{
	LL t,n,i;
	cin>>t;
	a[1]=2;
	a[2]=3;
	for(i=3;i<=1000000;i++)
	{
		a[i]=(a[i-1]+a[i-2])%M;
	}
	for(i=2;i<=1000000;i++)
	{
		a[i] = (a[i]+a[i-1])%M;
	}
	
	while(t--)
	{
		cin>>n;
		cout<<a[n]<<endl;
		
	}
    return 0;
}
