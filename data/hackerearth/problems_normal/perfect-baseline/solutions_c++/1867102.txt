	int t;
	cin >> t;

	while (t--) {
		int n, k;
		cin >> n >> k;

		string str[n];
		for (int i = 0; i < n; i++) {
			cin >> str[i];
		}

		string ans = "";

		for (int i = 0; i < k; i++) {
			char idx;
			int mn = INT_MAX;
			for (char j = 'a'; j <= 'z'; j++) {
				int sum = 0;
				for (int k = 0; k < n; k++) {
					sum += abs(j - str[k][i]);
				}
				if (sum < mn) {
					idx = j;
					mn = sum;
				}
			}
			ans += idx;
		}

		cout << ans << endl;
	}



	return 0;
}
