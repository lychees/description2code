#include <iostream>
#include <stdio.h>
#include <vector>
#include <algorithm>

using namespace std;

typedef long long int Lint;

int main()
{
	Lint t, x, y, n, i, s1, s2, c1, c2;
	
	scanf("%lld", &t );
	
	while( t-- ) {
		
		scanf("%lld %lld %lld", &x, &y, &n );
		vector<int> A( n );
		
		for( i = 0;i < n;i++ )
			scanf("%lld", &A[ i ] );
		
		sort( A.begin(), A.end() );
		
		s1 = 0;
		c1 = 0;
		for( i = 0;i < n;i++ ) {
			s1 += A[ i ];
			if( s1 <= x )
				c1++;
		}
		
		s2 = 0;
		c2 = 0;
		for( i = 0;i < n;i++ ) {
			s2 += A[ i ];
			if( s2 <= y )
				c2++;
		}
		
		if( c1 > c2 )
			printf("Chef Won\n");
		else if( c1 < c2 )
			printf("Jhool Won\n");
		else
			printf("Tie\n");
			
	}
	
	return 0;
}
