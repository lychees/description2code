#include<iostream>
#include<string>
 
using namespace std;
 
int main()
{
    int t,l,r,sum,x,z,i,n;
 
    cin>>t;
 
    while(t--)
    {
        sum=0;
        cin>>l>>r;
 
        for(i=l;i<=r;i++)
        {
            z=0;
            n=i;
 
            while(n!=0)
            {
                x=n%10;
 
                z=z*10+x;
 
                n=n/10;
            }
 
            //cout<<z<<endl;
 
            if(z==i)
                sum+=1;
        }
 
        cout<<sum<<endl;
    }
}
