#include<iostream>


using namespace std;


int find_sides(int n, int k){
    if(n==3)
        return n;
    int res=n;
    while(res!=3 && k>0){
        if(res%2!=0)
            res= (res/2)+2;
        else
            res = (res/2)+1;
        k = k-1;

    }   
    return res;
}   

int main(){
    int T;
    cin>>T;
    for(int i=0;i<T; i++){
        int N, P;
        cin>>N; 
        cin>>P;
        cout<<find_sides(N, P)<<endl;
    }   
    return 0;
}
