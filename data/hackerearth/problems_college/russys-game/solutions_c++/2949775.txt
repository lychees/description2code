#include<iostream>
#include<stdlib.h>
using namespace std;

int main()
{
	long long k,t;
	long long *a,*visited;
	cin>>t;
	while(t--)
	{
		cin>>k;
		a=(long long*)malloc(sizeof(long long)*k);
		visited=(long long*)malloc(sizeof(long long)*k);
		int i=0;
		for(i=0;i<k;i++)
		{
			cin>>a[i];
			visited[i]=0;
		}
		i=0;
		while(true)
		{
			if(a[i]==-1)
			{
					i=-1;
					break;
			}
			else if(visited[i]!=1)
			{
				visited[i]=1;
				i=a[i];
			}
			else
			{
				if(visited[i]==1)
				{
					break;
				}
			}
		}
		if(i==-1)
			cout<<"Win"<<endl;
		else 
			cout<<"Lose"<<endl;
		
	}
	return 0;
}
