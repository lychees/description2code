#include <bits/stdc++.h>
using namespace std;
#define ll long long int
ll a[1005];

int main()
{
   int t;
   scanf("%d",&t);
   while(t--)
   {
   	int n,q;
   	scanf("%d %d",&n,&q);
   	for(int i=0;i<n;i++)
   	scanf("%lld",&a[i]);
   	sort(a,a+n);
  ll sum=0;
   	for(int i=0;i<n-1;i++)
   	sum+=(q*(a[i+1]-a[i]));
   	printf("%lld\n",sum);
   }
}
