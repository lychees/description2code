#include <bits/stdc++.h>
using namespace std;
int vowels[5]={'a','e','i','o','u'};
bool safe(char a){
	for(int i=0;i<5;i++){
		if(a==vowels[i]){
			return false;
		}
	}
	return true;
}
int dp[5005][5005];
void sol(string a,string b){
	memset(dp,0,sizeof(dp));
	int n=a.length();
	int m=b.length();
	for(int i=1;i<=n;i++){
		for(int j=1;j<=m;j++){
			if(a[i-1]==b[j-1] && safe(a[i-1])){
				dp[i][j]=1+dp[i-1][j-1];
			}
			else{
				dp[i][j]=max(dp[i-1][j],dp[i][j-1]);
			}
		}
	}
	cout<<dp[n][m]<<endl;
}
int main() {
	string s;
	getline(cin,s);
	int n=s.length();
	int i=0;
	int j=1;
	while(s[j]!=' '){
		j++;
	}
	string a=s.substr(i,j);
	string b=s.substr(j+1,n-j-1);
	sol(a,b);
	return 0;
}
