#include<bits/stdc++.h>
using namespace std;
#define get(m) scanf("%d",&m)
#define put(n) printf("%d\n",n)
int main()
{
    int t,n,m,p,b,q;
    double d;
    get(t);
    while(t--)
    {
        get(n);
        get(b);
        m=d=n*1.0/b;
        if(m==d)
        {
            put(n);
        }
        else
        {
            p=m*b;
            q=(m+1)*b;
            if((q-n)<=(n-p))
                put(q);
            else
                put(p);
        }
    }
    return 0;
}
