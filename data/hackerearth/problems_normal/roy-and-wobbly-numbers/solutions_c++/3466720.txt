// Roy and wobbly numbers

#include <iostream>
using namespace std;

void get_output (int n, int k) {
    int a = k / 9 + (k % 9 == 0 ? 0 : 1);
    int b = (k % 9 == 0 ? 9 : k % 9);
    if (b <= a)
        b--;

    for (int i = 0; i < n / 2; i++)
        cout << a << b;
    if (n % 2 == 1)
        cout << a;

    return ;
}

int main () {
    int t, n, k;
    cin >> t;
    while (t--) {
        cin >> n >> k;
        if (k > 81)
            cout << -1;
        else
            get_output (n, k);
        cout << endl;
    }

    return 0;
}
