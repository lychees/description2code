#include <iostream>
using namespace std;

int main()
{
	int T;
	cin >> T;
	while(T>0){
		int M,N;
		cin >> M >> N;
		int A[N];
		for(int i = 0; i< N; i++){
			cin >> A[i];
		}
		bool table[M+1];
		for (int k =1; k <= M; k++) {
			table[k] = false;
		}
		for (int k = 0; k < N; k++){
			table[A[k]] = true; 
		}
		
		for (int k= 1; k<=M; k++ ){
			if (table[k]) {
				continue;
			}

			for (int l = 0; l< N; l++){
				if ( k  <= A[l]) {
					continue;
				}
		//	cout << " k-A[]"  << k << " " << A[l] << " " << table[k-A[l]];
				if (!table[k-A[l]]) {
					table[k] = true;
				//	cout << "cpcp" << endl;
					break;
				}
			}
		}
		if (table[M]) {
			cout << "Little Deepu" << endl;
		} else {
			cout << "Kate" << endl;
		}
		T--;
	}
    return 0;
}
