#include<bits/stdc++.h>
using namespace std;

vector<int> primos;
bool        checked[10005];
#define INF 100000000
int mx[4] = {1,-1,-1,1};
int my[4] = {1,1,-1,-1};

void pri(){
    memset(checked,0,sizeof checked);
    
    for(int i=2;i<=100;i++){
        if(!checked[i]){
            for(int j=i+i;j<=10000;j+=i){
                checked[j] = 1;
            }
        }
    }
    
    for(int i=2;i<=10000;i++){
        if(!checked[i])primos.push_back(i);
    }
}

int calcMin(int a,int k){
    if(a!= 2 && a%2==0 && k%2==0)return INF;
    
    int r = 0;
    vector<int>::iterator it = lower_bound(primos.begin(),primos.end(),a);
    while(it != primos.end()){
        if(*it-a < 0){it++;continue;}
        int tmp = *it - a;
        if(tmp%k == 0){
            return tmp/k;
        }
        it++;
    }
    
    return INF;
}

int main(){
    pri();
//    cout<<primos.size()<<endl;
    ios_base::sync_with_stdio(0);cin.tie(NULL);
    
    int n,k;
     /*
    while(cin>>n>>k && n+k){
        cout<<calcMin(n,k)<<endl;    
    }
    */
    while(cin>>n>>k){
        int arr[n][n];
        int mini[10004];memset(mini,-1,sizeof mini);
        
        for(int i=0;i<n;i++){
            for(int j=0;j<n;j++){
                cin>>arr[i][j];
                if(mini[arr[i][j]] != -1)continue;
                else{
                    mini[arr[i][j]] = calcMin(arr[i][j],k);
                }
            }
        }
        
        int ans = INF;
        int x = -1,y = -1;       
        
        for(int i=1;i<n-1;i++){
            for(int j=1;j<n-1;j++){
                int anst = mini[arr[i][j]];
                for(int k=0;k<4;k++){
                    int xt = i+mx[k];
                    int yt = j+my[k];
                    anst += mini[arr[xt][yt]];
                }
                if(anst< ans){
                    ans = anst;
                    x = i+1;
                    y = j+1;
                }
            }
        }              
        if(ans != INF){
            cout<<"yes"<<endl<<ans<<endl<<x<<" "<<y<<endl;
        }else{
            cout<<"no"<<endl;
        }
    }
}
