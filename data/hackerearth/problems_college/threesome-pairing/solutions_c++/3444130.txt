#include <stdio.h>
#include <iostream>
using namespace std ; 

int main()
{
    int t ; 
    cin >> t ; 
    while  ( t--)
    {
    	int n ; 
    	int a[100000] ; 
    	cin>> n ; 
    	int i ; 
    	for ( i = 0; i< n ; i++)
    	{
    		cin >> a[i] ; 
		}
		int j ; 
		int r = 0; 
		for ( i = 0; i< n ; i++)
		{
		//	cout << "sanjeev" <<endl; 
			for ( j = i+1 ; j< n ; j++)
			{
				if ( (a[i] + a[j])%3 == 0 )
				{
				//	cout << "sanjee "<<endl; 
					r = r+1 ;
				}
			}
		}
		cout << r <<endl; 
	}
    return 0;
}
