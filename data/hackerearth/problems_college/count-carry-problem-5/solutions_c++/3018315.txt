
#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	long long a,b,ct,sum,k;
	scanf("%d",&n);
	
	while(n--)
	{
		ct=sum=k=0;
		
		scanf("%lld%lld",&a,&b);
		
		while(a>0||b>0){
		
			sum=0;
			sum=sum+k+a%10+b%10;
			a=a/10;
			b=b/10;
			k=0;
			
			
			if(sum>=10)
			{
		    k++;
			ct++;}	}
		
		if(ct==0)
		printf("No carry operation\n");
		else if(ct==1)
		printf("1 carry operation\n");
		else
		printf("%lld carry operations\n",ct);
	
	
	
	}
	
	return 0;
}
