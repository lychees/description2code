// locate the char position : a to z : 0 to 25
// separate the number
// fix the number in array
// scan from 0 to 25 a to z
// locate the place for searched index


#include<iostream>
#include<string>
using namespace std;

long long int arr[26];
string s1;
long long int T;
int main()
{
	std::getline(std::cin, s1);
	char* p = (char*) s1.c_str();
	char ch;
	long long int len = s1.length();
	for(long long int i = 0; i < len ;)
	{

		//record the character
		if(*p >='a' && *p <='z')
		{
			ch = *p;
			p++;i++;
		}
		long long int old_index = i;
		while(*p >= '0' && *p <= '9')
		{
			i++;
			p++;
		}
		string s = s1.substr(old_index, i- old_index);
		long long int num = arr[25 - ('z'-ch)] + atoi(s.c_str());
		
		// now fix the number in array of chars
		arr[25 - ('z'-ch)] = num;
	}

	cin>>T;
	while(T--)
	{
		register long long int qnum;
		cin>>qnum;
		for(long long int j=0;j<26;j++)
		{
			qnum = qnum - arr[j];
			if(qnum<=0)
			{
				cout<<(char)('z'- (25 - j))<<endl;
				break;
			}

		}
		if(qnum>0)
			cout<<"-1"<<endl;
	}

	return 0;
}
