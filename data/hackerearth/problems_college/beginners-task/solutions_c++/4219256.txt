// Example program
#include <iostream>
#include <cstring>
using namespace std;
int main()
{
    int count,s,t,n;
    cin>>t;
    while(t--)
    {
        cin>>s;
        count=0;
        while(s--)
        {
            cin>>n;
            if(n%2==0)
            count++;
        }
        cout<<count<<endl;
    }
  return 0;
}
