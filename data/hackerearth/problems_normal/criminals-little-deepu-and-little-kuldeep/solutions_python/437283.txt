def solve(arr):
    sort = sorted(arr, reverse=True)
    l = len(sort)
    g=0
    in_n = dict()
    for i in xrange(len(sort)):
        for z in xrange(i,len(sort)):
            if sort[z] < sort[i] and z not in in_n:
                g += 1
                in_n[z]=1
                break
    return l-g

    for y in range(int(s)):
        arr_.append(int(raw_input()))
    results.append(solve(arr_))
