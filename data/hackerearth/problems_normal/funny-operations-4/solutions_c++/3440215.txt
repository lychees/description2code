#include <iostream>
#include <cmath>
using namespace std;
pair<int, int> xor_or_and(int i, int size, int AND, int XOR, int N, int arr[]){
	if(i == N)
		return make_pair(AND|XOR,size);
	pair<int,int> t1 = xor_or_and(i+1, size + 1,AND & arr[i], XOR ^ arr[i], N,arr);
	pair<int,int> t2 = xor_or_and(i+1,size, AND, XOR, N,arr);
	//cout << t1.first << " " << t1.second << '\t' << t2.first << " " << t2.second << endl;
	if(t1.first>t2.first){
		return t1;
	}else if(t1.first == t2.first){
		if(t1.second > t2.second){
			return t1;
		}else{
			return t2;
		}
	}else{
		return t2;
	}
}
int main()
{
	int T = 0;
	cin >> T;
	for(int t = 0; t < T; t++){ 
	    int N;
	    cin >> N;
	    int arr[N];
	    for(int n = 0; n < N; n++){
	    	cin >> arr[n];
	    }
	    pair<int, int> ans = xor_or_and(0, 0, (unsigned int)(-1), 0, N, arr);
	    cout <<  ans.first<<" "<< ans.second << endl;
	}
    return 0;
}
