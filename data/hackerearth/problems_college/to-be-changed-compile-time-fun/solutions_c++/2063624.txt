#include<bits/stdc++.h>
using namespace std;

int main()
{   
   ios :: sync_with_stdio(0);
   int n;cin>>n;
   vector<int>A;
   for(int i=0;i<n;i++){ 
   	  int b;cin>>b;
   	  A.push_back(b);
   }

   int final=0;
   for(int i=1;i<=n-2;i++){
       int b=0,p;
       for(int j=1;j<=A.size()-2;j++){
           if(A[j-1]*A[j+1]>b)
           {
               b=A[j-1]*A[j+1];
               p=j;
           }
           else if(A[j-1]*A[j+1]==b && A[p]>A[j])
               p=j;
       }
       final=final+b;
       A.erase(A.begin()+p);

   }
   cout<<final<<endl;
}
