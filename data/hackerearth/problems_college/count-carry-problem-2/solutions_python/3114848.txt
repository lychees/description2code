t=input()
while t>0:
	t-=1
	a,b=raw_input().split()
	la=len(a)
	lb=len(b)
	l=la
	#print la,lb
	c=0
	if la>lb:
		l=lb
	#print l
	for i in range(l-1,-1,-1):
		if (int(a[i])+int(b[i]))>9:
			c+=1
	if c==0:
		print"No carry operation"
	elif c==1:
		print"1 carry operation"
		
	else:
		print str(c)+" carry operations"
