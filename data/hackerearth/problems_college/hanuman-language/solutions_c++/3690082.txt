#include <iostream>
#include <string>
#include <vector>

using namespace std;

int main()
{
	int T;
	string s;
	vector <string> ans;

	cin >> T;
	
	cin.ignore();

	while (T--)
	{
		ans.clear();
		
		getline(cin, s);

		int i = 0, j = 0;

		for (j = 0; j < s.length(); j++)
		{
			if (s[j] == ' ')
			{
				string t = "";
				for (int k = i; k < j; k++)
					t += s[k];
				ans.push_back(t);
				i = j+1;
			}
		}
		string t = "";
		for (int k = i; k < j; k++)
			t += s[k];
		ans.push_back(t);
		
		cout << ans[ans.size() - 1];
		
		for (int j = ans.size() - 2; j >= 0; j--)
			cout << " " << ans[j];
		cout << endl;
	}

	return 0;
}
