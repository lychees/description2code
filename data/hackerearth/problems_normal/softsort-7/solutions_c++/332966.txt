#include<iostream>
#define MOD 1000000007
using namespace std;
long long fact[1000001];
void pre_process()
{
    fact[0]=1;
    for(int i=1;i<=1000000;i++)
    {
        fact[i]=(fact[i-1]*i)%MOD;
    }
}
int main()
{
    pre_process();
    long t;
    cin>>t;
    while(t--)
    {
        long n;
        cin>>n;
        cout<<((3*fact[n])%MOD+3)%MOD<<endl;
    }
    return 0;
}
