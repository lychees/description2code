#include <bits/stdc++.h>

using namespace std;

//#define test

double opt[18][(1 << 18) - 1];
double a[20][20];
int Tmp[20];
int main() {
#ifdef test
	freopen("a.in","rt",stdin);
	freopen("a.out","wt",stdout);
#endif
    int n;
	cin >> n;
	for(int i = 1; i <= n; i++)
		for(int j = 1;j <= n; j++)
			cin >> a[i][j];
	int K, Newi, MAX = (1<<n) - 1;
	double New;
	opt[0][MAX]=1;
	for(int day = 0; day < n - 1; day++)
		for(int t = 1; t <= MAX; t++)
			if(opt[day][t] > 0) {
				K = t;
				for(int i=1;i<=n;++i) {
					Tmp[i] = K & 1;
					K >>= 1;
				}
				for (int i=1;i<=n;++i)
					if (Tmp[i])
						for (int j=1;j<=n;++j)
							if (Tmp[j]&&i!=j) {
								New=2.0/((n-day)*(n-day-1))*a[i][j]*opt[day][t];
								Tmp[j]=0;
								Newi=0;
								for (int q=n;q;--q)
									Newi=(Newi<<1)+Tmp[q];
								opt[day+1][Newi]+=New;
								Tmp[j]=1;
							}
			}
	for (int i = 1; i <= n; i++) {
		printf("%.5lf",opt[n - 1][1 << (i - 1)]);
		if (n != i)	printf(" ");
	}
	cout << endl;
}
