#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,c=0,c1=0;
		cin>>n;
		int x=n;
		if(n<3)
		cout<<"-1"<<endl;
		else
		{
			while(n%3!=0)
			{
				n=n-5;
				c++;
				if(n<0)
				{
					c=0;
					c1=1;
					break;
				}
				
			}
			if(c1==1)
		cout<<"-1"<<endl;
		else
		{
		for(int i=0;i<x-5*c;i++)
		cout<<"5";
		for(int i=0;i<5*c;i++)
		cout<<"3";
		cout<<endl;}
		}
		
	}
	return 0;
}
