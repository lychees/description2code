#include<iostream>
#include<stdio.h>
#include <functional>
#include<algorithm>
using namespace std;
 int main()
 {
   unsigned long long int n,q;
   cin>>n>>q;
   
   unsigned long long int t[n],s[n];
   unsigned long long  int k[q];
   for(unsigned long long int i=0;i<n;i++)
   {
       cin>> t[i];
   }
   for(unsigned long long int i=0;i<n;i++)
   {
        cin>>s[i];
   }
   for(unsigned long long i=0;i<q;i++)
   {
        cin>>k[i];
   }

   sort(t, t+n,greater<int>());
   
   for(unsigned long long int i=0;i<q;i++)
   {
        unsigned long long int sum=0;
        for(int j=0;j<k[i];j++)
        {
            sum=sum+t[j];
        }
        cout<<sum<<endl;
   }

   return 0;
 }
