#include <iostream>
#define ll long long
using namespace std;

int main()
{
	ll t, i, f=0, ans;
	string s;
    cin>>t;
    while(t--)
    {
    	f=0;
    	cin>>s;
    	for(i=0;i<s.length()/2;i++)
    	{
    		if(s[i]!=s[s.length()-i-1])
    		{
    			f=1;
    			if(s[i]==s[s.length()-i-2])
    				ans=s.length()-i-1;
    			else
    				ans=i;
    			break;
    		}
    	}
    	if(f==1)
    		cout<<ans<<endl;
    	else
    		cout<<-1<<endl;
    }
    return 0;
}
