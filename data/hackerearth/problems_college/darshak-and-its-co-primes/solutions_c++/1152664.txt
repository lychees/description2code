#include <bits/stdc++.h>
using namespace std;
#define ll long long
ll gcd ( ll a, ll b )
{
  ll c;
  while ( a != 0 ) {
     c = a; a = b%a;  b = c;
  }
  return b;
}
int main() 
{
	ll t,a,b;
	cin>>t;
	while(t--) {
		cin>>a>>b;
		if(gcd(a,b) == 1) puts("Is a Co-Prime");
		else puts("Not a Co-Prime");
	}
	return 0;
}
