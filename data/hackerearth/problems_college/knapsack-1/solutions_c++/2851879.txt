#include <iostream>
using namespace std;
    
inline int max(int a,int b)
{
	return(a>=b?a:b);
}
int main()
{
	
    int t;
    cin>>t;
    while(t--)
    {
		int n,k;
		cin>>n>>k;
		int arr[n];
		int ans[k+1];
		ans[0]=0;
		for(int i=0;i<n;i++)
		cin>>arr[i];
		int m;
		for(int i=1;i<=k;i++)
		{
			m=ans[i-1];
			for(int j=0;j<n;j++)
			{
				if(ans[i-arr[j]]+arr[j] <=i)
				m=max(ans[i-arr[j]]+arr[j],m);
			}
			ans[i]=m;
		}	
		//for(int i=0;i<=k;i++)
		cout<<ans[k]<<endl;
	
    }
    return 0;
}
