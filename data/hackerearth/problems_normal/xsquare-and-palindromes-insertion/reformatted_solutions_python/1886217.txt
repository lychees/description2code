T = int(raw_input())

for i in xrange(T):
    d = dict()
    n = 0

    S = raw_input()
    one = ''
    for x in S:
        if x in d:
            d[x] = d[x]+1
        else:
            d[x] = 1
    for i in d.keys():
        if (d[i] % 2 != 0):
            one = i
            break

    '''if(one=='' or len(d.keys())==1):
		n=0
	else:	'''
    for i in d.keys():
        if (d[i] % 2 != 0 and i != one):
            n += 1
            d[i] = d[i]+1

    print n
