#include<bits/stdc++.h>
using namespace std;
int main()
{
int t;
cin>>t;
while(t--)
{
 unsigned long long n;
 cin>>n;
 unsigned long x=sqrt(n);
if(x*x==n)
cout<<"YES\n";
else
{
 unsigned long long a,b;
 a=x*x;
 b=(x+1)*(x+1);
 unsigned long long p,q;
 p=n-a;
 q=b-n;
 if(p<q)
 cout<<"-"<<p<<endl;
 else
 cout<<"+"<<q<<endl;
}
}
return 0;
}
