typedef long long ll;
const ll mod=1e9+7;
const int INF=0x7f7f7f7f;
const double eps = 1e-6;

const int maxn = 1000010;

int Gcd(int a,int b) {
	if(b == 0 and a != 1) return INF;
	if(b == 0 and a == 1) return -1;
	return a / b + Gcd(b,a % b);
}

int main() {
	int t; cin >> t;
	while (t --) {
		int n; cin >> n;
	    if(n == 1) {
			cout << 0 << "\n";
			continue;
		}
		int ans = INF;
		for (int i = 1; i < n; i ++) {
			ans = min(ans,Gcd(n,i));
		}
		cout << ans << "\n";
	}
	return 0;
}
