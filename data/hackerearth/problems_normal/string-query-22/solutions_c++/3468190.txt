#include<bits/stdc++.h>
using namespace std;
int main()
{
	string str;
	cin>>str;
    vector<int>vec[150];
    int i,j;
    for(i=0;i<str.length();i++)
    {
    	vec[str[i]].push_back(i);
	}
	int q;
	cin>>q;
	while(q--)
	{
		char ch;
		int n;
		cin>>n;
		cin>>ch;
		str[vec[ch][n-1]]='#';
		vec[ch].erase(vec[ch].begin()+n-1);
	}
	str.erase(remove(str.begin(),str.end(),'#'),str.end());
	cout<<str<<"\n";
}
