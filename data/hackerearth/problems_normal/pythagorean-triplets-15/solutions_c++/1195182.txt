#include <iostream>
#include<math.h>
using namespace std;

int gcd(int a,int b){
	int tmp;
	while(b){
		tmp=a%b;
		a=b;
		b=tmp;
	}
	return a;
}

int main()
{
    int t,n,count,c,i,j;
    cin>>t;
    while(t--){
    	cin>>n;
    	count=0;
     	for(i=2;i<=sqrt((double)(n-1))+1;i++){
     		for(j=1;j<i;j++){
     			if((i-j)%2!=0){ 
                  	c=i*i+j*j;
                    if(gcd(i,j)==1 && (c)<=n)
                      	 count+=n/c;
                   }
     		}
     	}
     	cout<<count<<endl;
    }
}
