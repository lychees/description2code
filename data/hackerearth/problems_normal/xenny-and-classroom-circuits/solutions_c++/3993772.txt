int main() 
{
	std::ios::sync_with_stdio(false);
	ll t;
	cin>>t;
	while(t--)
	{
		ll n;
		cin>>n;
		string s;
		cin>>s;
		ll ans1=0;
		for(ll i=0;i<=2*n-1;i=i+2)
		{
			if(s[i]!='B')
				ans1++;
		}
		ll ans2=0;
		for(ll i=0;i<=2*n-1;i=i+2)
		{
			if(s[i]!='G')
				ans2++;
		}
		ll ans = min(ans1,ans2);
		cout<<ans<<endl;
	}
	return 0;
}
