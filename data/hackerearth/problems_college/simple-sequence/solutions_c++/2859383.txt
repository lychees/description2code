#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t=1;
    //cin>>t;
    
    while(t--)
    {
    	int n,x,ans=0;
    	cin>>n;
    	
    	vector < int > arr;
    	for(int i=0; i<n; i++)
    	{
    		scanf("%d",&x);
    		if(x<=i-1) continue;
			if(arr.size()==0 || x>arr[arr.size()-1]) arr.push_back(x);
    		else
    		{
    			int y=lower_bound(arr.begin(),arr.end(),x)-arr.begin();
    			arr[y]=x;
    		}
    	}
    	cout<<n-arr.size()+ans<<"\n";
    }
    return 0;
}
