#include <iostream>
#include<algorithm>
#include<cstdio>
using namespace std;
#define MAX 100001
int main()
{
	int t;
	scanf("%d",&t);
	while(t--){
		int n,i;
		scanf("%d",&n);
		int a[n+1],b[n+1];
		for(i=0;i<n;i++)
			scanf("%d",&a[i]);
		for(i=0;i<n;i++)
			scanf("%d",&b[i]);
		sort(a,a+n);
		sort(b,b+n);
		for(i=0;i<n;i++){
			if(a[i]<=b[i]){
			printf("NO\n");
			break;
			}	
		}
	if(i==n)
	printf("YES\n");
	}
}
