#include<bits/stdc++.h>
using namespace std;
 double pp,qq;
double ans[1000005];
int main()
{
   int n,p;
    ans[0]=1.0;
   cin>>n>>p;
     pp=(double)p/100;
     qq=1-pp;
     for(int i=1;i<=n;i++)
     {
     	if(i-2>=0) ans[i]+=pp*ans[i-2];
     	if(i-3>=0) ans[i]+=qq*ans[i-3];
     }
     
    printf("%.6lf",ans[n]);
    return 0;
}
