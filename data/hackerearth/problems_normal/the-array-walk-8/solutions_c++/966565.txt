#include <bits/stdc++.h>
using namespace std;


long long int fun(long long int p, long long int q, long long int &x, long long int &y)
{
	if(p == 0){
		x = 0,y = 1;
		return q;
	}
	long long int m, n;
	long long int d = fun(q%p, p, m, n);
	x = n - (q/p)*m;
	y = m;
	return d;
}

bool func(long long int a, long long b, long long int c, long long int &x0, long long int &y0, long long int &g )
{
	g = fun(abs(a), abs(b), x0, y0);
	if(c%g != 0)
		return false;
	x0*=c/g;
	y0*=c/g;
	if(a < 0) x0*=-1;
	if(b < 0) y0*=-1;
	return true;
}

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	long long int h,u,d,x,y,g,val;
    	cin >> h >> u >> d;
    	bool res = func(u, 0-d, h, x, y, g);
    	if(res == false){
    		cout << "-1\n";
    		continue;
    	}
    	long long int uu = u/g, dd = d/g;
    	val = max(abs(x)/dd, abs(y)/uu)+1;
    	x = x + val*dd;
    	y = y + val*uu;
    	long long int val1 = x/dd, val2 = y/uu;
    	long long int mn = min(val1, val2);
    	x = x - mn*dd;
    	y = y - mn*uu;
    	cout << x+y << "\n";
    }
    return 0;
}
