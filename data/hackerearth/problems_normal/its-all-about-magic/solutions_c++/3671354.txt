#include <iostream>

int main()
{

   int n;
   
   
   while(t--)
   {
   	cin>>n;
   	
   	long int a=1,b=0,zeroadd,oneadd;
   	
   	while(n--)
   	{
   		zeroadd=b;
   		oneadd=b;
   		
   		b=0;
   		b+=a;

   	}

   }

}
