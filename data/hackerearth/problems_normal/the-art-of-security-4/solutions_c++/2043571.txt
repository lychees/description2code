#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
const int MAXN=8;
int a,b,c,d,ans,sum,nums[MAXN];
void query(int curr,int sum1,int sum2)
{
    if (curr==a)ans+=sum2>=sum1;
    else
    {
        if (sum1+nums[curr]<=sum2)query(curr+1,sum1+nums[curr],sum2);
        query(curr+1,sum1,sum2+nums[curr]);
    }
}
int main()
{
    scanf("%i",&a);
    for (int i=0;i<a;i++)
    {
        scanf("%i",&nums[i]);
        sum+=nums[i];
    }
    sort(nums,nums+a);
    do
    {
        query(0,0,0);
    }
    while (next_permutation(nums,nums+a));
    printf("%i %i\n",ans,sum);
    if (ans>sum)puts("We will win!");
    else puts("Got no way out!");
    return 0;
}
