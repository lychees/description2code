#define forn(i,n) for(in i=0;i<(n);++i)
#define forv(i,v) forn(i,sz(v))
#define fors(i,s) for(auto i=(s).begin();i!=(s).end();++i)

using namespace std;
typedef long long in;
typedef vector<in> VI;
typedef vector<VI> VVI;
VI r;
VI nur;
in m;
void sft(in s){
  s%=m;
  nur=VI(m,0);
  forn(i,m){
    nur[i]=(r[i]^r[(i+m-s)%m]);
  }
  r=nur;
}
string tp;
void gog(){
  in n,p,x;
  cin>>n>>m>>p>>x;
  if(n==1)
    x=0;
  --n;
  r.resize(m);
  cin>>tp;
  forn(i,m)
    r[i]=tp[i]-'0';
  for(in i=32;i>=0;--i){
    if(n&1LL<<i)
      sft(p*(1LL<<i));
  }
  forn(i,m){
    r[i]^=x;
    cout<<r[i];
  }
  cout<<"\n";
}
