maxn = 1000010
p = [0] * maxn
d = [0] * maxn

for i in range(2, maxn):
    if p[i] == 0:
        for j in range(i, maxn, i):
            d[j] += i
        for j in range(2 * i, maxn, i):
            p[j] = 1
           
n, k = map(int, raw_input().split())

res = 0
for i in range(1, n+1):
    if d[i] == k:
        res+=1
print res
