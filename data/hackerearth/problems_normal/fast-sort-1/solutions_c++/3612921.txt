#include <iostream>
#include<algorithm>
#include<cstring>
#include<cstdio>
#include<vector>
using namespace std;
char alpha[27];
int pos(char a)
{
    int pos_a=-1,i=0;
    for(i=0;i<26;i++)
    {
        if(pos_a>=0) break;
        if(alpha[i]==a) pos_a=i;
    }
    return pos_a;
}
int comp(char a,char b)
{

  return pos(a)<pos(b);

  //return a<b;
}
void sol()
{
	char str[1000001];
	memset(alpha,'\0',sizeof(alpha));
	cin>>alpha;
	fflush(stdin);
	cin>>str;
	fflush(stdin);
	sort(str,str+strlen(str),comp);
	cout<<str<<endl;
	//comp(str[0],str[1]);
}
int main()
{
    int t;
    cin>>t;
    while(t--) sol();
    return 0;
}
