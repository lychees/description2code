#include<cstdio>
using namespace std;
int main()
{
	int m,n;
	scanf("%d %d",&m,&n);
	int a[m+1][n+1];
	for(int i=1;i<=m;i++)
		for(int j=1;j<=n;j++)
			scanf("%d",&a[i][j]);
	for(int sum=2;sum<=m+n;sum++)
	{
		for(int i=1;i<=sum-1 && i<=m;i++)
		{
			if(sum-i>n)
				continue;
			printf("%d ",a[i][sum-i]);
		}
		printf("\n");
	}
	return 0;
}
