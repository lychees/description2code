#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	int t;
	long long int a,b,c,ans,repeating_factor_new,repeating_factor;
	long long n,m;
	
	cin >> t;
	for (int i = 0; i < t; i++) {
	cin >> n >> m >> a >> b >> c;
	repeating_factor = max(2*c, a+b);
	if (n == m) {
		ans = n*repeating_factor;
	}
	else if (n > m) {
		repeating_factor_new = max(2*a+b, 2*c+a);
		ans = repeating_factor_new + (m-1)*repeating_factor + (n-m-1)*a;
	}
	else {
		repeating_factor_new = max(2*b+a, 2*c+b);
		ans = repeating_factor_new + (n-1)*repeating_factor + (m-n-1)*b;
	}
	cout << ans << endl;
	}
}
