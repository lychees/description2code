LLI MAX(LLI a,LLI b)
{
    if(a>b)return a;
    return b;
}


int  GCD(int a,int b)
{
   int x=MAX(a,b);
   int y=MIN(a,b);
    if(x%y==0)
    {
        return y;
    }
    else
    {
        return GCD(x%y,y);
    }

}


long long fast_pow(int a,int n)
{
    long long result = 1;
    long long power = n;
    long long value = a;
    while(power>0)
    {
        if(power&1)
            {result = result*value;
            result = result%MOD;}
        value = value*value;
        value = value%MOD;
        power /= 2;
        //power >>= 1;
    }
    return result;
}


int pow_k(LLI N,LLI K)
{
    int count=0;
    while(N>1)
    {
        N/=K;
        count++;
    }
    return count;
}


int count_bit(int N)
{
    int count=0;
    for(int i=0;i<32;i++)
    {
        if(N&(1<<i))
        {
            count++;
        }
    }
    return count;
}
