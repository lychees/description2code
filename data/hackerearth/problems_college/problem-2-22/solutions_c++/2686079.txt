#include <iostream>
using namespace std;

int main()
{
    int t,r,xc,yc,flag=0,j=0;
    cin>>t;
    int x[t],y[t];
   
    while(t!=0)
    {
    	 cin>>r;
    cin>>xc>>yc;
    	for(j=0;j<4;j++)
    	{
    		cin>>x[j]>>y[j];
    	}
    	
    	for( j=0; j<4;j++)
    	{
    	   if( ( (xc-x[j])*(xc-x[j]) + (yc-y[j])*(yc-y[j]) )>(r*r)) 
           {
           	flag=1;
           	break;
           }
    	}
    	if(flag==0)
    	cout<<"Yes"<<endl;
    	else
    	cout<<"No"<<endl;
   t--;
   flag=0;
    }
    return 0;
}
