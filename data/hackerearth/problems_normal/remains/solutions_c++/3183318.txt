#include<bits/stdc++.h>
using namespace std;
int main(){
	long long int t,x,y,n,s=0,tt;
	scanf("%lld",&t);
	while(t--){
		s=0,tt=1000;int i;
		scanf("%lld%lld%lld",&x,&y,&n);
		s=x+y;
		if(n>2)s+=abs(x-y);
		for(i=3;i<n;i++){
			//y=x+y-(x=y);
			tt=abs(x-y);
			if(tt==0)break;
			x=y;y=tt;
			s+=abs(x-y);
		}
		if(tt==0)s+=(((n-i)/3)*x*2+((n-i)%3)*x);
		printf("%lld\n",s);
	}
	return 0;
}
