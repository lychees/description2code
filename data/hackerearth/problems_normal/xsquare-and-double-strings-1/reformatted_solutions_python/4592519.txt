for _ in range(input()):
    arr = [0]*26
    a = raw_input()
    if len(a) <= 1:
        print "No"
    else:
        for i in range(len(a)):
            arr[ord(a[i])-97] += 1
            if arr[ord(a[i])-97] == 2:
                print "Yes"
                break
        else:
            print "No"
