#include <bits/stdc++.h>

using namespace std;

int arr[100000+10];
bool mark[10000+10];


int main()
{
   int n;
   cin >> n;

   for (int i = 0 ; i < n ; i++)  cin >> arr[i];  // input

   sort(arr , arr + n);  // sort the input array

   int num = 0; // number of the bought marbles.

   for (int i = 0 ; i < n ; i++){

        if (!mark[arr[i]]){    
                num++;         // you should buy this marble

                for (int j = arr[i] ; j <= arr[i]+4 ; j++){   // set marble range as free
                    mark[j] = true;
                }
        }
   }

   cout << num << "\n";
   return 0;
}
