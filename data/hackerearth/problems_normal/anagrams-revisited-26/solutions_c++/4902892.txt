#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int n,max=1;
    cin>>n;
    string s[n];
    string temp;
    for(int i=0;i<n;++i)
    {
        cin>>temp;
        sort(temp.begin(),temp.end());
        s[i]=temp;
    }
    sort(s,s+n);
    int count=1;
    for(int i=0;i<n-1;++i)
    {
        if(s[i+1]==s[i])
        {
            count++;
            if(count>max)
            {
                max=count;
            }
        }
        else
        {
            count=1;
        }
    }
    cout<<max<<endl;
    return 0;
}
