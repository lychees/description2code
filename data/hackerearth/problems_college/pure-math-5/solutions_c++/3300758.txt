#include <iostream>
#include <vector>
#include <cmath>

using namespace std;

int main(){

	long long int t,k;
	long long int result;
	cin>>t;
	long long int n,m;

	while(t--){

		cin>>k;
		result=0;
		for(long long int i=2;i<=k;i++){
			result++;
			n=i;
			m=(i*i-1);
			while(m<=k){
				result++;
				long long int temp=m;
				m=(m*m-1)/n;
				n=temp;
			}
		}


		cout<<result<<endl;
	}

	return 0;
}
