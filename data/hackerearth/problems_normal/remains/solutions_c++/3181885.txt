#include <iostream>
#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define sl(x) scanf("%lld",&x)
int main()
{
    ll t;
    sl(t);
    while(t--)
    {
    	ll x,y,n,i,j,ctr=2,res=0,mx=0;
    	sl(x);sl(y);sl(n);
    	res+= x+y;
    	//cout<<x<<" "<<y<<" ";
    	while(ctr<n)
    	{
    		mx= abs(y-x);
    	//	cout<<mx<<" ";
    		x=y;
    		y=mx;
    		res+= mx;
    		ctr++;
    		if(!y)
    		{
    			res+= ((n-ctr)-(n-ctr)/3)*x;
    			break;
    		}
    	}
    	printf("%lld\n",res);
    }
    return 0;
}
