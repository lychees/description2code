#include <iostream>
using namespace std;
int gcd(int x,int y)
{
   if(y==0)
      return x;
   else
   {
   	    if(x>y)
   	       return gcd(y,x%y);
   	    else
   	       return gcd(x,y%x);
   }
   
}
int main()
{
    int t,x,y,flag,temp;
    string s;
    cin>>t;
    while(t--)
    {
    	cin>>x>>y>>s;
    	if(s=="Chandu")
    	   flag=1;
    	else
    	if(s=="Daspal")
    	   flag=2;
    	while(true)
    	{
    		temp=gcd(x,y);
    		x=x-temp;
    		y=y-temp;
    		if(x==0||y==0)
    		   break;
    		else
    		{
    			if(flag==1)
    			   flag=2;
    			else
    			   flag=1;
    		}
    	}
    	if(flag==1)
    	   cout<<"Chandu\n";
    	else
    	   cout<<"Daspal\n";
    }
    return 0;
}
