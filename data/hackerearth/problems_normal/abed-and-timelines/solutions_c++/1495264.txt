#include <iostream>
#include<stdlib.h>
#include<cstdio>
#include<math.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int n,m,x1,y1,x2,y2,value;
		cin>>n>>m;
		cin>>x1>>y1;
		cin>>x2>>y2;
		cin>>value;
		long long int min1=abs(x1-x2);
		long long int min2=abs(y1-y2);
		long long int res=min(min1,n-min1)*value+min(min2,m-min2)*value;
		//cout<<res<<endl;
		if(res<1000)
		{
			cout<<res<<endl;
			cout<<"You saved the group."<<endl;
		}
		else
		{
			cout<<res<<endl;
		    cout<<"Let go of the group."<<endl;
		}
	}
    //cout << "Hello World!" << endl;
    return 0;
}
