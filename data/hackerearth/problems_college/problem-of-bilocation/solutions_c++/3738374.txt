#include<bits/stdc++.h>
using namespace std;
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int t;
    cin>>t;
    while(t--)
    {
        int a,b,c,d,s;
        cin>>a>>b>>c>>d>>s;
        if(!(c-s>=a))
        {
            cout<<"STAY IN LONDON\n";
            continue;
        }
        else
        {
            cout<<min(b,c-s)<<endl;
        }
    }
    return 0;
}
