#include <bits/stdc++.h>
using namespace std;

int main()
{
    int N, K;
    cin >> N >> K;
    int pos[N];
    for (int i = 0; i < N; i++) {
    	cin >> pos[i];
    }
    sort(pos, pos + N);
    
    int low = 0;
    int high = 10000000;
    while (low != high) {
    	int mid = (low + high)/2;
        int used = 0;
		int i = 0;
		while (i < N) {
			used++;
			int j = i+1;
			while (j < N && pos[j] - pos[i] <= 2*mid) {
				j++;				
			}
			i = j;
		}
		if (used <= K) 
			high = mid;
		else
			low = mid+1;
    }
    
    cout << low << endl;
}
