    #include <bits/stdc++.h>
    using namespace std;
    typedef long long ll ;
     
    int main()
    {
        ll t; 
        cin >> t; 
        while(t--){
        	ll n ; 
        	cin >> n;
        	pair <ll,ll> disk[206] ;
        	for(int i = 0; i < n; i++){
        		ll r;
        		ll h;
        		cin >> r >> h  ;
        		disk[i] = make_pair(r, h);
        	}
        	sort(disk,disk+n) ;
        	// for(int i = 0; i < n; i++){
        	// 	cout << disk[i].first <<" "<< disk[i].second << endl;
        	// }
        	ll dp[206],max1=0 ;
        	dp[0] = disk[0].second;
        	for(int i = 1; i < n; i++){
        		ll max = disk[i].second;
        		for(int j = i-1; j >= 0; j--){
        			if((disk[j].second < disk[i].second) && (disk[j].first < disk[i].first))
        			if(dp[j] + disk[i].second > max)
        			max = dp[j] + disk[i].second ;
        		}
        		dp[i] = max ;
        		if(max > max1)
        		max1 = max;
        		
        		//cout << dp[i] << endl ;
        	}
        	cout << max1 << endl ;
        }
        
        return 0;
    }
