#include<iostream>
#include<vector>
using namespace std;
int main()
{
	long long prime[1000001],count[1000001];
	for(long long i=2;i<=1000000;i++)
		if(prime[i]==0)
			for(long long j=i*2;j<=1000000;j+=i)
			{
				long r=j;
				while(r%i==0)
				{
					r=r/i;
					prime[j]++;
				}
			}
	
	for(long long i=2;i<=1000000;i++)
		if(prime[i]!=0)
		count[i]=count[i-1]+prime[i];
		else 
			count[i]=count[i-1]+1;
	long t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		cout<<count[n]<<endl;
	}
	return 0;
}
