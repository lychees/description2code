using namespace std;
const int MAX = 100005;
const int MAX1 = 1000005;
int A[MAX], p[MAX1];
bool A1[MAX1];

void s()
{
	int i, j, x;
	p[1] = 0;
	A1[0] = true;
	A1[1] = true;
	for(i = 2;i < MAX1;++i)
	{
		if(A1[i] == false)
		{
			p[i] = 1;
			for(j = 2*i;j < MAX1;j += i)
			{
				A1[j] = true;
				x = j;
				while(x % i == 0)
					p[j]++, x /= i;
			}
		}
	}
}

int main()
{
	int n;
	long ans = 0;
	s();
	cin >> n;
	for(int i = 0;i < n;++i)
	{
		cin >> A[i];
		ans ^= p[A[i]];
	}
	if(ans == 0) cout << "BOB" << endl;
	else cout << "ALICE" << endl;
	return 0;
}
