#include <iostream>
#include<stdio.h>
#include<bits/stdc++.h>
using namespace std;
long long int fib[100000];
int main()
{
	long long int t;
	scanf("%lld",&t);
	memset(fib,0,sizeof(fib));
	fib[1]=2;
	fib[2]=3;
	long long int k=3;
	while(t--)
	{
		long long int n;
		scanf("%lld",&n);
		long long int ans=0;
		//if(fib[n]!=0) 
		//ans=fib[n];
		//else
		if(fib[n]==0)
		{
			long long int i;
			for(i=k;i<=n;i++)
			fib[i]=fib[i-1]+fib[i-2];
			k=i;
		}
		ans=fib[n];
		printf("%lld\n",ans*ans);
	}
    return 0;
}
