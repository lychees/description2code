#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<cstring>
using namespace std;
int cs;
int h,m,x;
int main()
{
  scanf("%d",&cs);
  while(cs--)
  {
    scanf("%d%d",&h,&m);
    if(h==12)
      h=0;
    h*=30;
    m*=6;
    h+=m/12;
    x=abs(h-m);
    x=min(x,360-x);
    printf("%d\n",x);
  }
  return 0;
}
