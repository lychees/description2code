t = int(raw_input())
for x in range(t):
    a = int(raw_input())
    left = []
    right = []
    for y in range(a):
        p = map(float, raw_input().split())
        left.append(p[0])
        right.append(p[1])
    k = round((sum(left)/a), 2)
    r = round((sum(right)/a), 2)
    s = "("+"%.2f " % k+","+" %.2f" % r+")"
    print s
