#include <iostream>
using namespace std;

int main()
{
    int t,n;
    cin>>t;
    while(t--){
    	cin>>n;
    	int a,b;
    	int flag = 0;
    	for(a = 0;a<=n;a = a+5){
    		if(a%5 == 0 and (n-a)%3 == 0){
    			for(int i = 0;i<(n-a);i++){
    				cout<<"5";
    			}
    			for(int i = 0;i<a;i++){
    				cout<<"3";
    			}
    			cout<<endl;
    			flag = 1;
    		}
    	}
    	if(flag == 0)
    	cout<<"-1"<<endl;
    }
    return 0;
}
