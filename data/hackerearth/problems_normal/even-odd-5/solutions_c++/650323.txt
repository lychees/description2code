 #include<iostream>

using namespace std;

int main()
{
	int testCases, row, column;
	long long even, odd;
	cin >> testCases;
	while (testCases--)
	{
		even = odd = 0;
		cin >> row >> column;
		if (row <= column)
		{
			long long n1,n2;
			if (row % 2 == 0)
			{
				n1 = n2 = row / 2;
			}
			else
			{
				n1 = row / 2;
				n2 = n1 + 1;
			}

			even = n1*(n1 + 1);
			odd = n2*n2;
			}
		else
		{
			long long n;
			n = row - column;
			
			if (n % 2 == 0)
			{
				even = n / 2 * column;
				odd = n / 2 * column;
			}
			else
			{
				if (column % 2 == 0)
				{
					odd = (n / 2 + 1)*column;
					even = (n / 2)*column;
				}
				else
				{
					even = (n / 2 + 1)*column;
					odd = (n / 2)*column;
				}
			}
			long long n1, n2;
			row = column;
			if (row % 2 == 0)
			{
				n1 = n2 = row / 2;
			}
			else
			{
				n1 = row / 2;
				n2 = n1 + 1;
			}

			even += n1*(n1+1);
			odd += n2*n2;

				
		}
		cout << even << " " << odd << endl;
	}

	return 0;
}
