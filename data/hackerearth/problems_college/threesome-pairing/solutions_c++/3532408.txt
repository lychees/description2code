#include<iostream>
using namespace std;
int main()
{
    int N,i,j,count;
    int n[100];
    int t;
    cin>>t;
    while(t--)
    {
        count=0;
    cin>>N;
    for(i=0;i<N;i++)
    {
        cin>>n[i];
    }
    for(i=0;i<N-1;i++)
    {
        for(j=i+1;j<N;j++)
        {
            if((n[i]+n[j])%3==0)
            {
                count++;
            }
        }
    }
    cout<<count<<endl;
    }
    return 0;
}
