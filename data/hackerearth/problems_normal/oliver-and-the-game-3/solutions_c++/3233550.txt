int count1=0;
vector<int>v[100005];
int begin1[100005];
int end1[100005];
bool visited[100005];
void topo(int node)
{

    visited[node]=true;
    begin1[node]=count1++;
    //cout<<node<<endl;
    for(int i=0;i<v[node].size();i++)
    {
        if(!visited[v[node][i]])
        {
            topo(v[node][i]);
        }
    }
    end1[node]=count1++;
}
