#include <iostream>
using namespace std;
#define mod 1000000007
int main()
{
   // cout << "Hello World!" << endl;
   long long int t,n,ans,i;
   cin>>t;
   while(t--){
   	cin>>n;
   	ans=1;
   	for(i=0;i<n;i++){
   		ans*=2;
   		ans%=mod;
   	}
   	ans=ans*(ans-1)%mod;
   	cout<<ans<<endl;
   }
    return 0;
}
