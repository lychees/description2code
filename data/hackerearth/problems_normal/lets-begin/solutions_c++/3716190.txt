#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    while(t--){
    	int x;
    	cin>>x;
    	
    	int dp[1000001];
    	dp[0]=0;
    	dp[1]=1;
    	dp[2]=1;
    	dp[3]=1;
    	dp[4]=2;
    	dp[5]=1;
    	dp[6]=2;
    	dp[7]=1;
    	
   		for(int i=8;i<=x;i++){
   			dp[i]=dp[i-1]+1;
   			
   			dp[i]=min(dp[i-7]+1,dp[i]);
   				dp[i]=min(dp[i-2]+1,dp[i]);
   				dp[i]=min(dp[i-3]+1,dp[i]);
   					dp[i]=min(dp[i-5]+1,dp[i]);
   		}
   		dp[1]=-1;
   		cout<<dp[x]<<endl;
    }
}
