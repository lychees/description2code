#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	int i,j;
	int n;
	cin>>t;
	while(t--)
	{
		int n ;
		cin>>n;
		int f=0;
		int a = n/3;
		int b=0;
		while(a>=0)
		{
			int sum = a*3+b*5;
			if(sum ==n)
			{
				f=1;
				break;
			}
			else
			if(sum<n)
			{
				b++;
			}
			else
			a--;
		}
		if(f)
		{
			for(i=1;i<=a*3;i++)
			{
				cout<<"5";
			}
			for(i=1;i<=b*5;i++)
			cout<<"3";
			
			cout<<"\n";
		}
		else
		{
			cout<<"-1"<<"\n";
		}
	}
}
