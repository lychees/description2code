#include <iostream>
#define ull unsigned long long
using namespace std;

int main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);
	cout.tie(0);
    int t;
    ull odd,even,n,i,ans,x;
    cin >> t;
    while (t--)
    {
    	odd = 0;
    	even = 0;
    	cin >> n;
    	for (i = 0; i < n; i++)
    	{
    		cin >> x;
    		if (x % 2 == 0) even++;
    		else odd++;
    	}
    	ans = odd * even;
    	cout << ans  << "\n";
    }
    return 0;
}
