#include <iostream>

using namespace std;


long long int g,result,mq=1000000007,k=1,n,arr[100],brr[100],f=1,i;

int GCD(int A, int B) {
    if(B==0)
        return A;
    else
        return GCD(B, A % B);
}
int main()
{
    cin>>n;
    for(i=0;i<n;i++)
        cin>>arr[i];
    for(i=0;i<n;i++)
        brr[i]=arr[i];
    if(n==1)
        g=arr[0];
    else {
    for(i=0;i<n-1;i++)
    {
            arr[i+1]=GCD(arr[i],arr[i+1]);
    }
    g=arr[i];
    }
    //cout<<g<<endl;
    for(i=0;i<n;i++)
        f=(f*brr[i])%mq;
    //cout<<f<<endl;
    for(i=0;i<g;i++)
        k=(k*f)%mq;
    result=k;
    cout<<result;
    return 0;
}
