#include<bits/stdc++.h>
using namespace std;

int main()
 {
 	
 	int n;
 	long long k;
 	cin>>n>>k;
 	int i;
 	vector<long long> a(n);
 	for(i=0;i<n;i++)
 	 {cin>>a[i];
 	 }
 	sort(a.begin(),a.end());
 	for(i=0;i<n;i++)
 	 {
 	 	if(a[i]<=k)
 	 	  k-=a[i];
 	 	else
 	 	 break;
 	 }
 	 cout<<i<<endl;
 	
 }
