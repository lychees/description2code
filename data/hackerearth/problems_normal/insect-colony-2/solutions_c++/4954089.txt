#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--) {
    	int n;
    	cin>>n;
    	int a[n+1], i, odd=0;
    	for(i=0;i<n;i++) {
    		cin>>a[i];
    		if(a[i]%2!=0)
    			odd++;
    	}
    	if(odd%2==0)
    		cout<<"Yes\n";
    	else
    		cout<<"No\n";
    }
    return 0;
}
