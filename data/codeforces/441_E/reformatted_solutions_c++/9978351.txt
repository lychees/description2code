using namespace std;
const double eps(1e-8);
typedef long long lint;

int x, k, p;
double dp[2][210];

int main() {
  scanf("%d %d %d", &x, &k, &p);
  double P = p / 100.;
  for (int i = 0; i <= k + 1; i++) {
    int tmp = x + i;
    while (!(tmp & 1))
      tmp >>= 1, dp[0][i] += 1.;
  }
  int now = 0;
  for (int i = 0; i < k; i++) {
    memset(dp[now ^ 1], 0, sizeof(dp[now ^ 1]));
    for (int j = 0; j <= k; j++) {
      dp[now ^ 1][j << 1] += (dp[now][j] + 1) * P;
      dp[now ^ 1][j] += dp[now][j + 1] * (1 - P);
    }
    now ^= 1;
  }
  printf("%.10f\n", dp[now][0]);
  return 0;
}
