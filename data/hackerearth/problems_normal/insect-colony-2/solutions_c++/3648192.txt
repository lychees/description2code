#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t>0)
    {
    	long long sum=0;
    	int n;
    	cin>>n;
    	int a;
    	for(int i=0;i<n;i++)
    	{cin>>a;
    		sum=sum+a;
    	}
    	if(sum%2)
    	cout<<"No"<<"\n";
    	else
    	cout<<"Yes"<<"\n";
    	t--;
    	
    }
    return 0;
}
