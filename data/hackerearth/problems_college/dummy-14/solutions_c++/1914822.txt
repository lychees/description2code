#include <iostream>
#include <vector>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    vector<long long int> r(6000000,0);
    vector<long long int> ram;
    for(long long int i = 1;i < 1000;++i){
    	for(long long int j = i;j < 1000;++j){
    		if(i*i*i + j*j*j < 6000000)
    			r[i*i*i + j*j*j]++;
    	}
    }
    for(long long int i = 0;i < 6000000;++i)
    	if(r[i] == 2)
    		ram.push_back(i);
    //cout << ram.size() << endl;
    long long int t;
    cin >> t;
    while(t--){
    	long long int n;
    	cin >> n;
    	cout << ram[n-1] << endl;
    }
    return 0;
}
