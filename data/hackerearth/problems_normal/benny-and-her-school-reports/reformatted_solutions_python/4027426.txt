T = int(raw_input())

for x in xrange(T):
    L = map(int, raw_input().split())
    N = L[0]
    M = L[1]
    X = L[2]

    L = map(int, raw_input().split())

    target = X*(N+1)

    for y in L:
        target -= y
        if target <= 0:
            break

    if target > M:
        print 'Impossible'
    else:
        print max(1, target)
