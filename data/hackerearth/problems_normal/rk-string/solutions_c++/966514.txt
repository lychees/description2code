#include <bits/stdc++.h>
#define LL long long
using namespace std;

int n, m, l, a[1000100];
char s[1000100];

LL f(char ch, int x)
{
	int y = 0;
	for(int i = 0; i < l; i++) if(s[i] == ch) a[y++] = i;
	LL an = 0;
	if(x){
		for(int i = 0; i+x-1<y; i++){
			long long v1, v2;
			v1 = a[i]+1;
			if(i-1>=0) v1 -= 1+a[i-1];
			v2 = l-a[i+x-1];
			if(i+x<y) v2 += a[i+x]-l;
			an += v1*v2;
		}
		return an;
	}
	else{
		a[y++] = l;
		for(int i = 0; i < y; i++){
			long long v = a[i];
			if(i-1 >= 0) v -= a[i-1]+1;
			an += (v*(v+1))/2;
		}
		return an;
	}
}
	
int main()
{
    int t;
    cin >> t;
    while(t--){
    	cin >> m >> n;
    	getchar();
    	gets(s);
    	l = strlen(s);
    	cout << f('R', m) << " " << f('K', n) << endl;
    }
    return 0;
}
