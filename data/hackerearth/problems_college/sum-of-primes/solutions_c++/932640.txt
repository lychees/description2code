lli pre[1000005];

void comput()
{
    int i,j;
    for(i=0; i<1000005; i++)pre[i]=1;

    pre[0]=pre[1]=0;

    for(i=2; i<=1000002; i++)
    {
        if(pre[i]==1)
        {
            for(j=2; i*j<=1000002; j++)
            {
                pre[i*j]=0;
            }
        }
    }
}
lli sum[1000005];
int main()
{
    lli t,n,m,i,j;
    comput();
    lli s=0;
    for(i=0; i<1000005; i++)
    {
        if(pre[i]==1)
        {
            s+=i;
        }
        sum[i]=s;
    }
    // cout<<s;


    lli a,b;
    scanf("%lld",&t);
    while(t-->0)
    {
        scanf("%lld%lld",&m,&n);
        a=sum[m-1];
        b=sum[n];
        lli ans=b-a;
        bool foo=false;
        if(ans<=1)
        {
            foo=true;

        }
        for(i=2; i<=sqrt(ans); i++)
        {
            if(ans%i==0)
            {
                foo=true;
                break;
            }
        }
        if(ans==2 || ans==3 )foo=false;
        if(!foo)printf("YES\n");
        else printf("NO\n");

    }




    return 0;
}
