#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t,m,n,u,v,w,x,g,h;
	scanf("%d",&t);
	while(t--)
	{
	    int S[10001]={0};
	    map<pair<int,int>,int>M;
	    vector<pair<int,int> >V;
	    scanf("%d %d",&m,&n);
	    for(int i=0;i<m;i++)
	    {
	    	scanf("%d %d",&u,&v);
	    	V.push_back(make_pair(u,v));
	    }
	    for(int i=0;i<n;i++)
	    {
	    	scanf("%d %d",&w,&x);
	    	S[w]++;
	    	M[make_pair(w,x)]++;
	    }
	    scanf("%d %d",&g,&h);
	    for(int i=0;i<V.size();i++)
	    {
	    	u=V[i].first;v=V[i].second;
	    	if(M.find(make_pair(u,v))!=M.end())
	    	{
	    		h--;
	    		g--;
	    		M[make_pair(u,v)]--;
	    		S[u]--;
	    		if(M[make_pair(u,v)]==0)
	    		M.erase(make_pair(u,v));
	    	}
	    	else if(S[u])
	    	{
	    		g--;
	    		S[u]--;
	    	}
	    }
	if(g<=0&&h<=0)
    printf("Great\n");
    else if(g<=0)
    printf("Good\n");
    else
    printf(":(\n");
	}
}
