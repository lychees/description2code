#include <iostream>
#include <stdlib.h>
using namespace std;
int mod(int a)
{
	if(a>=0)
		return a;
	return -a;
}
int compare1 (const void * a, const void * b)
{
  return ( *(int*)a - *(int*)b );
}
int compare2 (const void * a, const void * b)
{
  return ( -*(int*)a + *(int*)b );
}

{
    int a,b,c;

    int n,A[n+1],B[n+1];

    	cin>>A[i];
    	B[i]=A[i];

    	S[i]=A[i-1];

    	S[i]=A[i-2];
    int sum1=0,sum2=0;
    for(int i=0;i<size-1;i++)
    	sum1+=mod(S[i]-S[i+1]);
    S[0]=c;S[size-1]=b;
    for(int i=1;i<mid;i++)
    	S[i]=A[i-1];
    for(int i=mid+1;i<size-1;i++)
    	S[i]=A[i-2];

    	sum2+=mod(S[i]-S[i+1]);
   
   	S[mid]=a;

    	sum3+=mod(S[i]-S[i+1]);

    	S[i]=A[i-1];

    	S[i]=A[i-2];
