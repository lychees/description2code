#include <bits/stdc++.h>
 
#define LLI long long int
#define LD long double
#define PB push_back
#define MP make_pair
#define FORi(i, a, b) for(int i = a; i < b ; ++i)
#define FORd(i, a, b) for(int i = a; i > b ; --i)
 
using namespace std;

LLI cache[201][201];

LLI solve(int n, int k){
	if (n<k)
		return 0;
	LLI cnt = 1;
	FORi(i,k,n/2+1){
		if (cache[n-i][max(k,i)] == -1)
			cache[n-i][max(k,i)] = solve(n-i,max(k,i));
		cnt += cache[n-i][max(k,i)];
	}
	return cnt;
}

int main()
{
	int t,n,k;
	cin >> t;
	while(t--){
		cin >> n >> k;
		FORi(i,0,n+1){
			FORi(j,0,n+1){
				cache[i][j] = -1;
			}
		}
		cout << solve(n,k) << endl;
	}
    return 0;
}
