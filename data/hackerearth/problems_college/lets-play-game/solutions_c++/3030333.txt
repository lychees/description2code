#include <iostream>
#include <stdio.h>

using namespace std;

int main() {
    long max,n,k,t,i,j,p;
	int a[100010];
	cin >> t;
	while(t--) {	
	  cin >> n >> k;
	  for(i=0;i<n;i++) {
	  	  cin >> j;
	  	  a[i] = k/j;
	    		  	
	  }
	
	max = 0;
	for(i=0;i<n;i++) 
	   {
	   	cin >> j;
	   	j = j*a[i];
	   	if((j)>max)
	   	max = j;
	   	
	   }
	
	cout << max << endl;
    }
	
}
