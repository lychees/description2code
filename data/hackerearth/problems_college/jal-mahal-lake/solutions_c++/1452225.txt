#include <bits/stdc++.h>
using namespace std;
struct yash {
	long long int x;
	long long int y;
}yash[55];
bool cmp(struct yash a,struct yash b) {
	if(a.x<b.x)
	return true ;
	else
	return false;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--) {
    	long long int n,i,j;
    	scanf("%lld",&n);
    	for(i=0;i<n;i++)
    	scanf("%lld",&yash[i].x);
    	for(i=0;i<n;i++)
    	scanf("%lld",&yash[i].y);
    	sort(yash,yash+n,cmp);
    	long long int sum = 0,c = 1,min1;
    	for(j=0;j<n;j++) {
    		sum = 0;
    		c = 1;
    	for(i=n-2;i>=j;i--) {
    		sum += c*(yash[i+1].x-(yash[i].x+yash[i].y));
    		c++;
    	}
    	c = 1;
    	for(i=0;i<j;i++) {
    		sum += c*(yash[i+1].x-(yash[i].x+yash[i].y));
    		c++;
    	}
    	if(j==0)
    	min1 = sum;
    	else
    	min1 = min(sum,min1);
    	
    	}
    	printf("%lld\n",min1);
    }
    return 0;
}
