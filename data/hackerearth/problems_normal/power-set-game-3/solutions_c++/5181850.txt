LL power(LL a, LL b)
{
	if (b == 0)
		return 1;
	if (b == 1)
		return a;
	LL p = power(a, b / 2) % MOD;
	if (b & 1)
		return (((p*p) % MOD)*a) % MOD;
	else
		return (p*p) % MOD;
}
