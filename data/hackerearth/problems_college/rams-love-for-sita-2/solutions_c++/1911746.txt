#include <iostream>
using namespace std;

int f(int n, int a, int b)
{
         
          
         int origin = n/a;
         int total = origin;
         while(origin/b>0)
         {
             int free = origin/b;
             origin = free + origin%b;
            total += free;
         }
         return total;
}


int main()
{
    int t,n,c,m;
    cin>>t;
    while(t--)
    {
    	cin>>n>>c>>m;
    	cout<<f(n,c,m)<<"\n";
    }
    return 0;
}
