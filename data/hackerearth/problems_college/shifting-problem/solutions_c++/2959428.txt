#include <iostream>
using namespace std;

int main()
{
    int test;
    cin>>test;
    long long arr[10000]={0};
    arr[0]=1;
    arr[1]=1;
    
    for(int i=2;i<10000;i++){
    	arr[i]=arr[i-1]+arr[i-2];
    }
    
    while(test--){
    	int num;
    	cin>>num;
    	cout<<arr[num]<<endl;
    }
    
    return 0;
}
