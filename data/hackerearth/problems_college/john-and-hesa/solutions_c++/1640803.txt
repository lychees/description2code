#include<bits/stdc++.h>
#define ll long long
#define f(a,b,c) for(ll int i=a;i<b;i+=c)
using namespace std;

int main() {
	cin.tie(0);
	ios_base::sync_with_stdio(false);
	ll int T;
	cin >> T;
	while(T--) {
		ll int X,P,N,count=0;
		cin >> X >> P;
		while(true) {
			if(X%P==0) {
				X=X/P;
				count++;
				if(X==1) {
					X=0;
					break;
				}
			}
			else {
				break;
			}
		}
		
		cout << count << " " << X << endl;
	}
	return 0;
}
