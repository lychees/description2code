string a,b;
int ans;

void solve(int i,int j,int k)
{
    if(k==a.size())
    {
        ans++;
        return;
    }
    if(i==a.size() && j==b.size() && k==a.size())
    {
        ans++;
        return ;
    }
    if(i>=a.size() || j>=b.size())
        return;
    if(a[i]==b[j])
        solve(i+1,j+1,k+1);
    solve(i,j+1,k);
    solve(i+1,j,k);
}
