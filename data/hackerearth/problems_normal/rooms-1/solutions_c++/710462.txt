#include <iostream>
#include <vector>
#include <algorithm>
#include <utility>

using namespace std;

bool myCompare(const pair<int, int> &left, const pair<int, int> &right)
{
	if(left.first == right.first)
		return left.second < right.second;
	return left.first < right.first;
}

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n;
    	vector<pair<int, int> > A;
    	cin>>n;
    	int B[100005];
    	for(int i = 0; i < n; ++i)
    		cin>>B[i];
    	int x;
    	for(int i = 0; i < n; ++i)
    	{
    		cin>>x;
    		A.push_back(make_pair(B[i], 1));
    		A.push_back(make_pair(B[i] + x, -1));
    	}
    	sort(A.begin(), A.end(), myCompare);
    	int m = 0;
    	int a = 0;
    	for(int i = 0; i < 2*n; ++i)
    	{
    		a += A[i].second;
    		if(a > m)
    			m = a;
    	}
    	cout<<m<<endl;
    }
    return 0;
}
