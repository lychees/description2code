#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n;
    cin >> n; 
    int a[n];
    for(int i=0; i< n; i++)
    cin >> a[i];
    //int curr_max=a[0];
    int max_far=a[0];
    for(int i=1; i < n; i++)
    {
    	a[i]=max(a[i-1]+a[i],a[i]);
    	max_far=max(max_far,a[i]);
    }
    cout << max_far <<"\n";
    return 0;
}
