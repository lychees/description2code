#include <iostream>
#include <cstring>
#include <algorithm>

#include <set>
using namespace std;

int main()
{
	

    string s1,s2,s3;
    int t,i,j,k,h=0,l,count=0,l1,l2,flag=0;
    cin>>t;
    getline(cin,s3);
    while(t--)
    {
    set<char>s;
    getline(cin,s1);
    getline(cin,s2);
    l1=s1.length();
    l2=s2.length();
    if(s1==s2)
    {
    cout<<"0"<<endl;
    }
    else
    {
    	
    	for(i=0;i<l1;++i)
    	{
    		flag=0;
    		for(j=0;j<l2;++j)
    		{
    			if(s1[i]==s2[j])
    			{
    				flag=1;
    				s2[j]='1';
    				break;
    			}
    		}
    		
    		if(flag==0)
    		{
    			count=-1;
    			break;
    		}
    		
    	}
    	if(flag==0)
    	cout<<"-1"<<endl;
    	else
    	{
    		for(i=0;i<l2;++i)
    		{
    		if(s2[i]!='1')
    	
    		s.insert(s2[i]);
    		}
    		cout<<s.size()<<endl;
    	}
    	
    }	
    flag=0;
    count=0;
    }
    return 0;
}
