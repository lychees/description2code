#include <bits/stdc++.h>

#define mp make_pair
#define pb push_back
#define ll long long

using namespace std;

const int nm = 100010;

int n, m;
char s[nm];
map<string, int> val;
int f[nm];

int main() {
#ifdef LOCAL
	freopen("input.txt", "r", stdin);
#endif
	ios_base::sync_with_stdio(0);
	cin.tie(0);
	for (int i = 0; i < 26; ++i) {
		string tmp = "";
		tmp = tmp + char(i + 97);
		val[tmp] = 1;
		for (int j = 0; j < 26; ++j) {
			tmp = "";
			tmp = tmp + char(i + 97);
			tmp = tmp + char(j + 97);
			val[tmp] = 1;
		}
	}

	cin >> n >> m >> s;
	for (int i = 1; i <= m; ++i) {
		string tmp;
		int x;
		cin >> tmp >> x;
		val[tmp] = x;
	}
	for (int i = 1; i <= n; ++i) {
		string tmp = "";
		tmp += s[i - 1];
		f[i] = f[i - 1] + val[tmp];
		if (i >= 2)
			tmp = s[i - 2] + tmp;
		f[i] = max(f[i], f[i - 2] + val[tmp]);
	}
	cout << f[n] << "\n";
#ifdef LOCAL
	printf("\n%.3lf s\n", (double) clock() / CLOCKS_PER_SEC);
#endif
}
