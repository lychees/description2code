#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while(t--) {
		int n;
		cin >> n;
		int arr[110];
		for(int i=0;i<n;i++) cin >> arr[i];
		sort(arr,arr+n);
		int ans = 0;
		for(int i=0;i<n;i++) {
			ans++;
			if(i+1<n && arr[i+1] == arr[i]+1) i++;
		}
		cout << ans << endl;
	}
	
    //cout << "Hello World!" << endl;
    return 0;
}
