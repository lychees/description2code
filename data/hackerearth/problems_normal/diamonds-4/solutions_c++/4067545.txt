#include <iostream>
using namespace std;

int main()
{	int m,n,t,dia=0;
	char A[1000][1000];
	cin>>t;
	while(t--)
	{   dia=0;
		cin>>n>>m;
		for(int i=0;i<n;++i)
		{
			for(int j=0;j<m;++j)
			{
				cin>>A[i][j];
			}
		}
		
		for(int i=0;i<n;++i)
		{
			for(int j=0;j<m;++j)
			{
				if((A[i][j]=='/')&&(A[i][j+1]=='\\')&&(A[i+1][j]=='\\')&&(A[i+1][j+1]=='/')&&(i+1<n)&&(j+1<m))
				{
					++dia;
				}
			}
		}
		cout<<dia<<"\n";
	}

    //cout << "Hello World!" << endl;
    return 0;
}
