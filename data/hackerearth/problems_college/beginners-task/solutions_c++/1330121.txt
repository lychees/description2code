#include<stdio.h>
int main()
{
    int t,n,a[1000],counter,i;
    scanf("%d",&t);
    while(t--)
    {
        counter=0;
        scanf("%d",&n);
        for(i=0;i<n;i++)
        {
            scanf("%d",&a[i]);
            if(a[i]%2==0)
                counter++;
        }
        printf("%d\n",counter);
    }
    return 0;
}
