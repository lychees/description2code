s = raw_input()
kases = int(raw_input())
length = len(s)
while (kases):
    kases -= 1
    a, b = map(int, raw_input().split())
    a -= 1
    b -= 1
    a = a % length
    b = b % length
    if s[a] == s[b]:
        print "Yes"
    else:
        print "No"
