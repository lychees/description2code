#define FNAME ""

const int MAXN = 1e5;

int x_1[MAXN], y_1[MAXN], x_2[MAXN], y_2[MAXN];

int main() 
{
#ifdef LOCAL    
    freopen(FNAME".in", "r", stdin);
    freopen(FNAME".out", "w", stdout); 
#endif    
    int x, n;
    scanf("%d%d", &x, &n);
    forn (i, n)
        scanf("%d%d%d%d", &x_1[i], &y_1[i], &x_2[i], &y_2[i]);
    double curY = 10000;
    while (1)
    {
        double maxY = -1;
        int ind = -1;
        forn (i, n)
        {
            double a = y_2[i] - y_1[i], b = x_1[i] - x_2[i], c = y_1[i] * x_2[i] - x_1[i] * y_2[i];
            double y = (-c - a * x) / b;
            if (x_1[i] <= x && x_2[i] >= x && y <= curY && y >= maxY)
            {
                maxY = y;
                ind = i;
            } 
        }
        if (ind == -1)
            break;
        if (y_1[ind] < y_2[ind])
            x = x_1[ind], curY = y_1[ind];
        else
            x = x_2[ind], curY = y_2[ind];
        curY -= 1e-5;
    } 
    printf("%d\n", x);
    return 0;
}
