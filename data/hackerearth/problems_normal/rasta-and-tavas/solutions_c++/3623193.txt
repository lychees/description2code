typedef pair<ll,ll> ii;






const int  N=1e6+10;
ll n,k,s[N],res;
bool p[N];


int main()
{
	std::ios::sync_with_stdio(0);
	cin.tie(0);
	cin>>n>>k;
	for(int i=1;i<=n;i++)
	{
		p[i]=1;
		s[i]=0;
	}
	p[1]=0;
	res=0;
	for(int i=2;i<=n;i++)
	{
		if(p[i])
		{
			if(i==k)res++;
			for(int j=2*i;j<=n;j+=i)
			{
				p[j]=0;
				s[j]+=i;
			}
		}
		else
		{
			if(s[i]==k)res++;
		}
	}
	cout<<res<<"\n";
	return 0;
}
