#include <iostream>
using namespace std;
long long arr[64];
int main()
{
	long long t,i,n,m,ans = 0;
	cin >> t;
	arr[1] = 1;
	for(i = 2;i < 32;i++){
		arr[i] = arr[i-1]*2;
	}
	while(t--){
		ans = 0;
	cin >> n >> m;
	if(n == 1){
		cout << "0\n";
		continue;
	}
	for(i = 1;i < 30;i++){
		if(arr[i] >= m){
			break;
		}
		if(arr[i] >= n){
			break;
		}
	}
	ans = ans+i;
	n = n -arr[i];
	if(n <= 0){
		cout << ans << endl;
		continue;
	}
	if(n%m == 0){
		ans = ans + (n/m);
	}
	else{
		ans = ans + (n/m) + 1;
	}
	cout << ans << endl;
	}
    return 0;
}
