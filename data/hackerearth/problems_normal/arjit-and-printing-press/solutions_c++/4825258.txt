#include <bits/stdc++.h>
using namespace std;
const int MAX=1e5+5;
typedef long long ll;
#define pb(a)		push_back(a)
#define pp			pop_back()
#define sz			size()
#define pf(a)		printf("%d",a)
#define sf(a)		scanf("%d",&a)
#define _2nd(a)		a.second
#define _1st(a)		a.first
#define ENDL		cout<<"\n"
typedef pair<int,int> pii;
typedef pair<char,int> pchar;
typedef pair<ll,ll> pll;
typedef pair<ll,pll> node;

int main()
{
    ios::sync_with_stdio(0);cin.tie(0);
    int t;
    cin>>t;
    while(t--){
    	string w,r;
    	cin>>w>>r;
    	priority_queue<pii,vector<pii>,greater<pii> > q;
    	for(int i=0;i<r.size();i++){
    		q.push(make_pair(r[i]-'a',i));
    	}
    	pii p;
    	int i=0;
    	while(!q.empty()){
    		p=q.top();
    		q.pop();
    		int a=_2nd(p);
    		while(i<w.size()&&w[i]<=r[a]){i++;}
    		w[i]=r[a];
    	}
    	cout<<w<<endl;
    }
    return 0;
}
