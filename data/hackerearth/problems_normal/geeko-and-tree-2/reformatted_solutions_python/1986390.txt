t = int(raw_input())
i = 0
while i < t:
    rem = 0
    k, n = [int(x) for x in raw_input().split()]
    x, y = n+1, k-1
    # print("x is",x,"an yis ",y)
    nodes = ((k**x)-1)//y
    # print(nodes)
    while (nodes != 0):
        rem += nodes % 10
        nodes = nodes//10
    print(rem)
    i += 1
