#include<bits/stdc++.h>
#define ll long long int
#define se  set<ll>
#define For(i,n) for(i=0;i<n;i++)
using namespace std;
int main()
{
    ll t,m,n,a,i;scanf("%lld",&t);se s;
    while(t--)
    {
        scanf("%lld%lld",&m,&n);
        For(i,m) {scanf("%lld",&a);s.insert(a);}
        For(i,n)
        {
            scanf("%lld",&a);
            if(s.count(a)==1) {printf("YES\n");}
            else{
                printf("NO\n");
                s.insert(a);
            }
        }
        s.clear();
    }
}
