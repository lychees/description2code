#include <iostream>
#include <string.h>
using namespace std;

int main()
{
    int T,a,b,N,c;
    char S1[60],S2[60];
    cin>>T;
    while (T)
    {
    	c=0;
    	cin>>S1>>S2>>N;
    	//cout<<S1<<" "<<S2<<" "<<N<<endl;
    	a=strlen(S1);
    	b=strlen(S2);
    	if (a!=b)
    		cout<<"No"<<endl;
    	else
    	{
    		for (int i=0;i<a;i++)
    		{
    			if (S1[i]!=S2[i])
    			{
    				c++;
    				if (S1[i]=='M')
    					S1[i]=='D';
    				else
    					S1[i]=='M';
    			}
    		}
    		if (c<N)
    		{
    			if ((N-c)%2==0)
    				cout<<"Yes"<<endl;
    			else
    				cout<<"No"<<endl;
    		}
    		else if (c==N) 
    			cout<<"Yes"<<endl;
    		else
    			cout<<"No"<<endl;
    	}
    	T--;
    }
    return 0;
}
