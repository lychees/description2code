#include <iostream>

using namespace std;

int main()
{
    long long t;
    cin>>t;
    int s[t];
    for(int m=0;m<t;m++)
    {
        long long c=0;
        long long a,b;
        cin>>a>>b;
        long long sum=0;
        while(a>0 || b>0)
        {
            int x=0,y=0;
            if(a>0)
            {
                x=a%10;
                a=a/10;
            }
            if(b>0)
            {
                y=b%10;
                b=b/10;
            }
             if(sum=1)
             {
                 sum=(x+y+1);
             }
             else
             {
                 sum=(x+y);
             }
            if(sum>10)
            {
                c=c+1;
                sum=1;
            }
        }
        s[m]=c;
    }
    for(int m=0;m<t;m++)
    {
        if(s[m]==0)
        {
            cout<<"No carry operation"<<endl;
        }
        else if(s[m]==1)
        {
           cout<<"1 carry operation"<<endl;
        }
        else
        {
            cout<<s[m]<<" carry operations"<<endl;
        }
    }
    return 0;
}
