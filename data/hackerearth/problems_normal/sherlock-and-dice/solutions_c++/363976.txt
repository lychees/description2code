#include<iostream>

using namespace std;
#include<math.h>
#include<stdio.h>
#include<set>
double mp[52][10001];
int main()
 {
 	int t,m,nn,k,i,j,n,kk;
 	scanf("%d",&t);
 	while(t--)
 	  { //map<pair<int,int>,int > mp;
	   
	   for(i=0;i<=51;++i)
		for(j=0;j<=10000;++j)
			mp[i][j]=0.0;
 	   double num=0,den=0;
 	    mp[0][0]=1.0;
 	  	scanf("%d%d%d",&m,&n,&kk);
 	  	if(kk<m || kk>n*m)
 	  	  {
 	  	  	cout<<"0.000 0"<<endl;
 	  	  	continue;
 	  	  }
 	  	set<int> a;
 	  	set<int> b;
 	  	a.insert(0);
 	  	for(i=1;i<=m;i++)
 	  	  {
 	  	  	for(set<int>::iterator it=a.begin();it!=a.end();it++)
 	  	  	 {k=*it;
 	  	  	   if(mp[i-1][k])
 	  	  	  	   {for(j=1;j<=n;j++)
 	  	  	  	      {
 	  	  	  	      mp[i][k+j]+=mp[i-1][k];
 	  	  	  	      b.insert(k+j);
 	  	  	  	      }
 	  	  	       }
						 // mp[make_pair(i,(it->first).second+j)]+=it->second;
 	  	  	  }
 	  	  	  a.clear();
 	  	  	  a.swap(b);
 	  	  	  //b.clear();
 	  	  }
 	  	for(set<int>::iterator it=a.begin();it!=a.end();it++)
 	  	   {k=*it;
			  if(mp[m][k])
 	  	      den+=mp[m][k];
 	  	   }
 	  	num=mp[m][kk];
 	  	
 	  	double ans=num*1.0/den;
 	  	int count=0;
 	  	
		while(ans<1.0 && ans!=0.0)
		   {
		   	ans=ans*10;
		   	count++;
		   }printf("%0.3lf %d\n",ans,count);
	  }
 	return 0;
 }
