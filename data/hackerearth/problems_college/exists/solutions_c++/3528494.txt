#include <bits/stdc++.h>
using namespace std;
int main() {
   int t, n, k, a, i, q;
   scanf("%d", &t);
   while(t--) {
      scanf("%d", &n);
      int min[100000] = {0}, pos[100000] = {0};  
      for(i = 0 ; i < n ; i++) {
         scanf("%d", &a);
         if(a < 0)
            min[-a] = 1;
         else
            pos[a] = 1;
      }
      scanf("%d", &k);
      for(i = 0 ; i < k ; i++) {
         scanf("%d", &q);
         if(q < 0 && min[-q]) {
            printf("Yes\n");
            continue; 
         }
         if(q >= 0 && pos[q]) {
            printf("Yes\n");   
            continue;
         }
         printf("No\n");  
      } 
   }
   return 0;
}
