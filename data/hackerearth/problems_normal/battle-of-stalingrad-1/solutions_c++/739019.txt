#include <iostream>
#include<cmath>
using namespace std;

float area(int x1,int y1,int x2,int y2,int x3,int y3)
{
	return abs((x1*(y2-y3)+x2*(y3-y1)+x3*(y1-y2))/2.0);
}

bool isInside(int x1,int y1,int x2,int y2,int x3,int y3,int x4,int y4)
{
	float a=area(x1,y1,x2,y2,x3,y3);
	float a1=area(x2,y2,x3,y3,x4,y4);
	float a2=area(x1,y1,x3,y3,x4,y4);
	float a3=area(x1,y1,x2,y2,x4,y4);

	if(a==a1+a2+a3)
	return true;

	else
	return false;
}

int main()
{
	int x1,x2,x3,x4,y1,y2,y3,y4;
	int t,i;

	cin>>t;

	for(i=0;i<t;i++)
	{
		cin>>x1>>y1>>x2>>y2>>x3>>y3>>x4>>y4;

		if(isInside(x1,y1,x2,y2,x3,y3,x4,y4))
		cout<<"INSIDE\n";
		else
		cout<<"OUTSIDE\n";
	}


    return 0;
}
