#include <iostream>
using namespace std;

int main()
{
    int w,temp,flag;
    cin>>w;
    flag=0;
    for(int i=2;i<w;i=i+2)
    {
      temp=w-i;
      if(temp==0)
      break;
      if(temp%2==0)
      flag=1;
    }
    if(flag==1)
    cout<<"YES"<<endl;
    else if(flag==0)
    cout<<"NO"<<endl;
}
