#include <bits/stdc++.h>
using namespace std;

int main()
{
	queue<int> q,idq;
	int n;
	cin>>n;
	for(int i=0;i<=n-1;i++)
	{
		int no;
		cin>>no;
		q.push(no);
	}
	//ideal queue
	for(int i=0;i<=n-1;i++)
	{
		int no;
		cin>>no;
		idq.push(no);
	}
	int time_taken=0;
	while(!q.empty())
	{
		if(q.front()==idq.front())
		{
			q.pop();
			idq.pop();
			time_taken++;
			continue;
		}
		int front =q.front();
		q.pop();
		q.push(front);
		time_taken++;
	}
	cout<<time_taken<<endl;
    return 0;
}
