#include <iostream>
#include<string>
using namespace std;

int main()
{
    int t,i,j;
    string str;
    cin>>t;
    for(i=0;i<t;i++)
    {
    	fflush(stdin);
    	cin>>str;
    	std::string str1="\0";
    	for(j=0;str[j]!='\0';j++)
    	{
    		if(str[j]>=48 && str[j]<=57)
    		{
    			int a=str[j]-48;
    			int len=str1.length();
    			while((a-1)>0)
    			{
    				for(int k=0;k<len;k++)
    				str1.insert(str1.end(),str1[k]);
    				a--;
    			}
    		}
    		else
    		str1.insert(str1.end(),str[j]);
    	}
    	cout<<str1;
    	cout<<"\n";
    }
    return 0;
}
