#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long int n,k;
    cin>>n>>k;
    long long int arr[100]={0};
    arr[1] = 1;
    arr[2] = 1;
    for (int i=3;i<70;i++)
    {
        for(int j=i-1; (j>i-k-1)&&(j>0);j--)
            arr[i] += arr[j];
    }
    //cout<<(1L << n)<<endl;
    long long int g = __gcd((1L << n) - arr[n+2], (long long)(1L<< n));
    //cout<<g<<endl;
    //cout<<arr[n+2]<<endl;
    cout << ((1L << n) - arr[n+2])/g << "/" << (1L<<n)/g<< endl;
    return 0;
}
