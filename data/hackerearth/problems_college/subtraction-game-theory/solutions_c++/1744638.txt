#include <iostream>
#include <algorithm>
#define ll long long int
using namespace std;
ll SOD(ll n)
{
	ll tot=0;
	while(n>0)
	{
		tot+=n%10;
		n/=10;
	}
	return tot;
}


int main()
{
    ll sum=0,tot,result,n,i,j;
    cin>>n;
    ll a[n];
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    	tot+=SOD(a[i]);
    	sum+=a[i];
    }
//    cout<<tot;
    result=tot%9-SOD(sum)%9;
    cout<<"0";
    return 0;
}
