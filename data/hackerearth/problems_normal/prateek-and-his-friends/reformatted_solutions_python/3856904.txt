#!/usr/bin/python

from collections import deque


def ir():
    return int(raw_input())


def ia():
    line = raw_input()
    line = line.split()
    return map(int, line)


def solve():
    n, X = ia()
    a = []
    for i in range(n):
        a.append(ir())
    s = 0
    i = 0
    j = 0
    # x x x x
    # i   j
    while True:
        while s < X:
            if j == n:
                return 'NO'
            s = s + a[j]
            j = j + 1
        if s == X:
            return 'YES'
        s = s - a[i]
        i = i + 1
        if i == n:
            return 'NO'


T = ir()

for i in range(T):
    rc = solve()
    print rc
