#include<bits/stdc++.h>
using namespace std;
int main () {
    int t , n , a[500002] ,ans, sum ,l , r, k, cnt = 0;
    scanf ( "%d" , &t) ;
    while ( t-- ) {
        scanf ( "%d" , &n ) ;
        for ( int i = 0 ; i < n ; i++ ) {
            scanf ( "%d" , &a[i] );
        }
        l=a[0];
        r=a[1];
        k = 2;
        cnt = ans = 2;
        while ( k < n ) {
            if ( a[k] == l || a[k] == r) {
                k++;
                cnt++;
            }
            else {

                r = a[k];
                l = a[k-1];
                if ( cnt > ans )
                    ans = cnt;
                cnt = 2;
                k++;
            }
        }
        if ( cnt > ans )
                ans = cnt;
        cout << ans << endl;
    }
    return 0;
}
