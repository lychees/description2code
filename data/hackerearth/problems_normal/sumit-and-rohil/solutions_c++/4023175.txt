#include <iostream>
#include <algorithm>
#include <string>
#include <map>
using namespace std;
int main(){
	int num,n;
	string s;
	cin>>num;
	vector <string> v;
	n=num;
	while(n--){
		cin>>s;
		v.push_back(s);
	}
	multimap <string,string> grp;
	multimap <string,string> :: iterator itr;
	for(string & c : v){
		string sor = c;
		int size=sor.size();
		string begend{sor[0]};
		begend+=sor[size-1];
		itr = grp.find(begend);
		if(itr==grp.end()){
			sort(sor.begin(),sor.end());
			grp.insert(pair<string,string>(begend,sor));
		}
		else{
			sort(sor.begin(),sor.end());
			if(sor!=itr->second){
				grp.insert(pair<string,string>(begend,sor));
			}
		}
	}
	cout<<grp.size();
	return 0; 
}
