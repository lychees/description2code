#include <bits/stdc++.h>
using namespace std;

int main()
{
    int x=100000000000,j,n,k,n2,k2;
    cin>>n>>k;
    n2=n;
    
    for(k2=1;k2<=2*n;k2++)
    {
    	for(j=1;j<=(2*n/k2)+2;j++)
    	{
    		if(k2*j<=2*n)
    		{x=min(x,abs(k-k2)+abs(k2*j-n));}
    	}
    }
    
    if(k==1 && n>1)
    {x=0;}
    else if(k>1 && n==1)
    {x=1;}
    else if(n<=k/2)
    {x=n;}
    cout<<x;
    return 0;
}
