#include<iostream>
#include<cstdlib>
#include<cstdio>
int main()
{
    int hour,minutes,angle,minangle;
    int hourangle;
    int n;
    scanf("%d",&n);
    while(n--)
    {
    scanf("%d%d",&hour,&minutes);
    minangle=(minutes*6);
    hourangle=(hour*30+(minutes*30)/60);
    angle=abs(minangle-hourangle);
    if(angle>180)
        angle=360-angle;
    printf("%d\n",angle);
    }
    return 0;
}
