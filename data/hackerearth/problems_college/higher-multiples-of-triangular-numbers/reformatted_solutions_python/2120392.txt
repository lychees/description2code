#!/usr/bin/python
def nod(number):
    N = long(number**0.5)
    count = 0
    flag = 1
    if N*N == number:
        flag = 0
    i = 1
    while i <= N:
        if number % i == 0:
            count = count+1
        i = i+1
    count = count*2
    if flag == 0:
        count = count-1
    return count


def main():
    for t in range(input()):

            s = i*(i+1)/2
            count = nod(s)
            if count > n:

            i = i+1


main()
