#include <bits/stdc++.h>

using namespace std;

int main()
{
    long long n, x, sum, i, ans;

    cin >> n;

    sum = 0;
    ans = 0;
    for (i = 0; i < n; i++) {
        cin >> x;

        sum = max(0LL, sum + x);
        ans = max(ans, sum);
    }

    cout << ans << endl;

    return 0;
}
