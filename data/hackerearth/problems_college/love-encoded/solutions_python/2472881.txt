tc = int(raw_input())
while(tc>0):
    shift = int(raw_input())
    string = raw_input()
    string = list(string)
    string = [ord(x) for x in string]
    for i in range(0, len(string)):
        string[i] = string[i] - shift
        if string[i] < 65:
            string[i] +=26
    string = [chr(x) for x in string]
    s=""
    for x in string:
        s += str(x)
    print s
    tc-=1
