t=int(input())
for i in range (0,t):
    n=int(input())
    x=1
    y=2
    if n==1:
        p=1
    elif n==2:
        p=2
    else:
        for j in range (3,n+1):
            if j%2==1:
                p=x*y
            else:
                p=x+y
            x=y
            y=p
    print (p)
