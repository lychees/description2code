#include <iostream>
#include<string>
#include<cmath>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    t++;
   while(t--)
   {
   	string s;
   	getline(cin,s);
   	//cin>>s;
   	int l,sq;
   	l=s.length();
   	sq=sqrt(l);
   	char ar[sq][sq];
   	int i,j,k=0;
   	for(i=0;i<sq;i++)
   	for(j=0;j<sq;j++)
   	ar[i][j]=s[k++];
   	for(int b=0;b<sq;b++)
   	for(int c=0;c<sq;c++)
   	cout<<ar[c][b];
   	cout<<endl;
   }
    return 0;
}
