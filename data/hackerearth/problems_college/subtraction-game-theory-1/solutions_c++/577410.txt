#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    while(t--)
    {
    	int n,k;
    	scanf("%d%d",&n,&k);
    	
    	
    	int a=1,d=k+1;
    	
    	int term=(n-a)/d;
    	
    	if(a+term*d==n)
    	 printf("Nandu\n");
    	 
    	else 
    	printf("Rani\n");
    	
    }
    return 0;
}
