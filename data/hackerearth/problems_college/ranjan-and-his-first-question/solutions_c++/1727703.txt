#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while (t--) {
    	long long n, m;
    	cin >> n >> m;
    	if (m == 1) cout << 0 << endl;
    	else if (m > n) {
    		long long x = n * (n + 1);
    		x /= 2;
    		cout << x << endl;
    	} else {
    		long long c = 0;
    		long long xx = m;
    		while (m <= n) {
    			c++;
    			m += xx;
    		}
    	
    	
    		long long x = ((xx - 1) * c * xx) / 2;
  		
    		m -= xx;
    		long long yy = n - m;
    		x += (yy * (yy + 1)) / 2;
    	
    		cout << x << endl;
    	}
    }
    return 0;
}
