#include <iostream>
using namespace std;

int reverse(int n)
{
	int r=0;
	while(n)
	{
		r=r*10+(n%10);
		n=n/10;
	}
	return r;
}

int main()
{
    int t,a,b,x,count,i;
    cin>>t;
    while(t)
    {
    	t--;
    	count=0;
    	cin>>a>>b;
    	for(i=a;i<=b;i++)
    	{
    		x=reverse(i);
    		if(i==x)
    			count++;
    	}
    	cout<<count<<endl;
    }
    return 0;
}
