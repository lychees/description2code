#include <bits/stdc++.h>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int n,t,sum=0,j=1;
   cin>>n>>t;
   int a[n];
   for(int i=0;i<n;i++)
   cin>>a[i];
   sort(a,a+n);
   for(int i=0;i<n;i++)
   {
   	if(a[i]%t==0)
   	{
   		if(a[i]==j*t)
   		{
   		sum=j*t;
   		j++;
   		}
   }
   }
   cout<<sum;
   return 0;
}
