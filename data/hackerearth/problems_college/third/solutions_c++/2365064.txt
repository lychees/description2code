#include <iostream>
#include <cstring>
using namespace std;
int arr[10001];
int cache[2][10001];
int c =0;
int max(int a,int b)
{
	return a>b?a:b;
}
int solve(int bg,int end)
{
	//if(bg == 1 && end == 1)
	//c++;
	int size = end - bg +1;
	if(size == 1)
	return arr[bg];
	if(size == 2)
	return max(arr[bg],arr[end]);
	cache[0][0] = arr[0];
	cache[0][1] = arr[1];
	cache[1][0] = arr[0];
	cache[1][1] = arr[1];
	for(int i = bg+2;i<=end;i++)
	{
		int m1 = cache[0][i-2] + arr[i];
		int m2 = cache[0][i-1];
		cache[0][i] = max(m1,m2);
	}
	return cache[0][end];
}
int main()
{
    int t;
    scanf("%d",&t);
    
    while(t--)
    {
    	memset(cache,-1,sizeof(cache));
    	int n;
    	scanf("%d",&n);
    	//int arr[n];
    	for(int i =0;i<n;i++)
    	scanf("%d",&arr[i]);
    	printf("%d\n",solve(0,n-1));
    	//printf("%d\n",c);
    }
    return 0;
}
