using namespace std;
typedef unsigned long long LL;

cin>>T;
while(T--){
cin>>num;
LL temp=num;
LL mul=1;
while(num>0){
num/=10;
mul*=10;
}
mul/=10;
LL hold;
while(temp>0){
hold=temp/mul;
if(hold==2)
	cout<<"cde";
else if(hold==7)
	cout<<"acf";
else if(hold==6)
	cout<<"b3";
else if(hold==9)
	cout<<"c6a";
temp%=mul;
mul/=10;
}
cout<<endl;
}	
return 0;
}
