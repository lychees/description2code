#include <bits/stdc++.h>
using namespace std;
int n,a[10009],cnt[30009];
int main()
{
	memset(cnt,0,sizeof(cnt));
    cin>>n;
    for(int i=1;i<=n;i++) 
    {
    	cin>>a[i];
    	cnt[a[i]]++;
    }
    sort(a+1,a+n+1);
    int ans=INT_MAX;
    int m=3*a[n];
    for(int i=1;i<=m;i++)
    	cnt[i]+=cnt[i-1];
    for(int i=1;i<=n;i++)
    {
    	int x=a[i];
    	int y=3*a[i];
    	ans=min(ans,n-(cnt[y]-cnt[x-1]));
    }
    cout<<ans<<endl;
}
