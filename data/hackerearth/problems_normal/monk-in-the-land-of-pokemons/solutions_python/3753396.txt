from collections import defaultdict

T = input()

for v in range(T):
	N = input()
	food = defaultdict(int)
	count = 0
	for c in range(N):
		l = raw_input()
		pick, mauro_pokemon = map(int, l.split())
		food[pick] += 1
		if not food[mauro_pokemon]:
			count += 1
		else:
			food[mauro_pokemon] -= 1
	print(count)
