#include <iostream>
using namespace std;
#include<bits/stdc++.h>
vector<int>adj[1001];
bool visited[1001];
int cnt;
void dfs(int s)
{
	int i;
      cnt++;
	visited[s]=true;
	for(i=0;i<adj[s].size();i++)
	{
		if(visited[adj[s][i]]==false)
		{
		dfs(adj[s][i]);
		//count++;
		}
	}
}
void initialize()
{
	int i;
	for(i=0;i<1001;i++)
	visited[i]=false;
}
int main()
{
 int n,d,x,y,res=10000,i;
 cin>>n>>d;
 for(i=0;i<d;i++)
 {
 	cin>>x>>y;
 	adj[x].push_back(y);
 }
 initialize();
 for(i=1;i<=n;i++)
 {
 	 initialize();
 	cnt=0;
 	if(visited[i]==false){
 	dfs(i);
 	res=min(res,cnt);
 	}
 }
 cout<<res<<endl;
 
 
     return 0;
}
