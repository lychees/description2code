const double eps=1e-9;
const int N=1e3+5;
double dp[N][N],P[N][N];
int n,m;






int main()
{
int t,l,k,ans,i,j,res=0,fl;
t=1;

while(t--)
{
	cin>>n>>m;
	loop(i,n)
	{
	   dp[i][m]=0.0;
	}

	loop(i,m)
		dp[n][i]=0.0;




	loop(i,n)
		loop(j,m)
		scanf("%lf",&P[i][j]);


	for(i=n-1;i>=0;--i)
	{
		for(j=m-1;j>=0;--j)
		{
			dp[i][j]=1+P[i][j]*dp[i][j+1]+(1-P[i][j])*dp[i+1][j];
		}
	}


	printf("%.6lf\n",dp[0][0]);
