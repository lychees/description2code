#include <iostream>
#include <cmath>
using namespace std;

const double pi=3.141592653589793;
int main()
{
    int t;
    long long n;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%lld",&n);
    	int zeros=0;
    	long long cp=n;
    	while(cp>0)
    	{
    		zeros+=cp/5;
    		cp/=5;
    	}
    	int dig=floor((log(2*pi*n)/2 + n*(log(n)-1))/log(10))+1;
    	int ans=dig-zeros;
    	printf("%d\n",ans);
    }
    return 0;
}
