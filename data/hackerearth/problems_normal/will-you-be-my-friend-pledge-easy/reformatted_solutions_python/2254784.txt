def gcd(a, b):

    while (a != 0):
        c = a
        a = b % a
        b = c
    return b


k = input()
n = input()
s = 0
a = map(int, raw_input().split())
for i in range(0, n):
    if (gcd(a[i], k) > 1):
        s += 1
print s
