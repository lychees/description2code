#include<bits/stdc++.h>
#define ll long long int
#define MAX 10000009
#define MOD 1000000007
using namespace std;

ll A[100009];

ll No_Of_Trailing_Zeroes_In_Base_B(ll B)
{
    ll i,j,num,ans,total;
    ans=pow(10,10);

    for(i=2; i*i<=B; i++)
    {
        if(B%i==0)
        {
            num=0;
            while(B%i==0)
            {
                B=B/i;
                num++;
            }
            total=A[i];
            total=total/num;

            ans=min(ans,total);
        }
    }

    if(B>1) ///for last prime
    {
        total=A[B];

        ans=min(ans,total);
    }
    return ans;
}


int main()
{
    ll n,b,i,j,q,a;
    while(scanf("%lld",&n)!=EOF)
    {
        memset(A,0,sizeof A);
        for(i=1; i<=n; i++)
        {
            scanf("%lld",&a);
            for(j=2;j*j<=a;j++)
            {
                while(a%j==0)
                {
                    A[j]++;
                    a=a/j;
                }
            }
            if(a>1)
            {
                A[a]++;
            }
        }
        scanf("%lld",&q);
        for(i=1; i<=q; i++)
        {
            scanf("%lld",&b);
            b=No_Of_Trailing_Zeroes_In_Base_B(b);
            printf("%lld\n",b);
        }
    }
    return 0;
}
