#include <iostream>
#include <cassert>
using namespace std;

int main()
{
	int N, i, j, temp, index, max;
	long long K;
	cin>>N>>K;
	assert(N>=1 && N<=1e5);
	assert(K>=1 && K<=1e9);
	int arr[N];
	for(i=0; i<N; i++)
		cin>>arr[i];
	int swap = 0;
	i=0;
	//for(i=0; i<K; i++)
	while(swap<K && i<N){
		max = arr[i];
		index = i;
		//cout<<max<<endl;
		for(j=i+1; j<N; j++){
			if(arr[j]>max){
				//cout<<"HI"<<endl;
				max = arr[j];
				index = j;
			}
		}
		if(max != arr[i]){
			temp = arr[i];
			arr[i] = arr[index];
			arr[index] = temp;
			swap++;
		}
		i++;
	}
	
	for(i=0; i<N; i++)
		cout<<arr[i]<<" ";
    return 0;
}
