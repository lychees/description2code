#include <bits/stdc++.h>
using namespace std;
int arr[1000009];
int main()
{
	ios_base::sync_with_stdio(0);cin.tie(0);
	int n,m,l,r,x,y;
    cin>>n>>m;
    while(m--)
    {
    	cin>>l>>r;
    	arr[l]++;
    	arr[r+1]--;
    }
    arr[0]=0;
    for(int i=1;i<=n;i++)
        arr[i]+=arr[i-1];
    sort(arr+1,arr+n+1);
    cin>>m;
    while(m--)
    {
    	cin>>x;
    	y=lower_bound(arr+1,arr+n+1,x)-arr;
    	//cout<<y<<endl;
    	cout<<(n-(y-1))<<endl;;
    }
}
