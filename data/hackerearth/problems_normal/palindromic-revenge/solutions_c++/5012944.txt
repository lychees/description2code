#include <bits/stdc++.h>
using namespace std;

int main()
{
  //  cout << "Hello World!" << endl;
  int t,i,j,ans,n;
  cin>>t;
  char str[1000000];
  while(t--){
  	cin>>str;
  	ans=0;
  	n=strlen(str);
  	for(i=0,j=n-1;i<=j;i++,j--){
  		ans+=abs(str[i]-str[j]);
  	}
  	cout<<ans<<endl;
  }
    return 0;
}
