#include <iostream>
#include<cstring>
using namespace std;


long long int sumR,sumK,cntR[1000005],cntK[1000005],ansR,ansK;
string str;
int main()
{
    long long int t,r,k;
  	cin>>t;
    while(t--)
    {
    	cin>>r>>k;
    	cin>>str;
    	memset(cntR,0,sizeof(cntR));
    	memset(cntK,0,sizeof(cntK));
    	cntR[0]=1;
    	cntK[0]=1;
    	sumR=0;
    	sumK=0;
    	ansR=0;
    	ansK=0;
    	for(int i=0;i<str.size();i++)
    	{
    		if(str[i]=='R')
    		{
    			sumR++;
    		}
    		else
    		{
    			sumK++;
    		}
    		if(sumR>=r)
    		{
    			ansR=ansR+cntR[sumR-r];
    		}
    		if(sumK>=k)
    		{
    			ansK=ansK+cntK[sumK-k];
    		}
    		cntR[sumR]++;
    		cntK[sumK]++;
    	}
    	cout<<ansR<<" "<<ansK<<endl;
    	
    }
    return 0;
}
