#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;

int main()
{
   int t=0;
   scanf("%d",&t);
   while(t--)
   {
   	char string[1000005];
   	scanf("%s",string);
   	int length = strlen(string);
   	int i=0,half_sum=0,sum=0;
   	for(i=0;i<length;i++)
   	{
   		int temp=string[i]-'0';
   		if(i<length/2)
   		{
   			half_sum+=temp;
   		}
   		sum+=temp;
   	}
   	if(half_sum*2==sum)
   	printf("YES\n");
   	else
   	printf("NO\n");
   	
   }
    return 0;
}
