input = raw_input()
F = input.split()
N = int(F[2])

if N == 1:
    print int(F[0])
elif N == 2:
    print int(F[1])
else:
    # print N
    Fib = [None]*N
    for i in range(0, len(F)):
        Fib[i] = int(F[i])
    # print Fib
    for i in range(2, N):
        Fib[i] = Fib[i-1] + Fib[i-2]
    print Fib[N-1]
