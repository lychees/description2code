#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,n,i,j,temp;
	cin>>t;
	while(t--)
	{
		cin>>n;
		int ans[n+1];
		ans[0]=0;
		ans[1]=1;
		ans[2]=2;
		ans[3]=3;
		for(i=4;i<=n;i++)
		{
			ans[i]=i;
			for(j=1;j<=i;j++)
			{
				temp=j*j;
				if(temp>i)
					break;
				else
				{
					ans[i]=min(ans[i],1+ans[i-temp]);
				
				}
			}
		}
		cout<<ans[n]<<endl;
	}
}
