#include <bits/stdc++.h>
using namespace std;

int main() {
	int n,m;
	cin>>n>>m;
	map<string,long long int> w;
	string temp;
	cin>>temp;
	long long int f;
	cin>>f;
	w[temp]=f;
	string win=temp;
	long long int max=f;
	for(int i=1;i<m;i++){
		cin>>temp;
		cin>>f;
		w[temp]+=f;
		if(w[temp]>max){
			win=temp;
			max=w[temp];
		}
		//cout<<win<<" "<<max<<endl;
	}
	cout<<win;
	//cout<<w["Mithaiwala"];
	// your code goes here
	return 0;
}
