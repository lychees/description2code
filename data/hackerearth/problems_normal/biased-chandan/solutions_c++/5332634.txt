#include <iostream>
#include <stack>

using namespace std;

int main()
{
    stack<long int> s;
    long int sum = 0;
    long int N, i, x;

    cin >> N;
    
    for (i = 1; i <= N; i++) {
    	cin >> x;
    	s.push(x);
    	sum = sum + x;
    	
    	if (s.top() == 0) {
    		s.pop();
    		if (s.size() != 0) {
    			sum = sum - s.top();
    			s.pop();
    		}
    	}
    }
    
    cout << sum << endl;
    return 0;
}
