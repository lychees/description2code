ll power(ll a, ll b)
{
    if (b == 0)
        return 1;
    ll x = power(a, b>>1);
    if (b & 1)
        return (((x*x)%mod)*a)%mod;
    return (x*x)%mod;
}

ll scan()
{
    ll n = 0, ch = get();
    while (ch < '0' || ch > '9')
        ch = get();
    while (ch >= '0' && ch <= '9') {
        n = (n << 3) + (n << 1) + ch - '0';
        ch = get();
    }
    return n;
}
