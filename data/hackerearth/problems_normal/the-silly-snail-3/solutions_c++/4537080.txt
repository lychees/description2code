int tree[400005][2];
int  n , x , y , z;
int vis[400005];
void disp(int node){
    cout << node << " ";
    vis[node] = 1;
    for(int i = 0 ; i < 2 ; i++){
        int el = tree[node][i];
        if(el!=0 && !vis[el]){
            disp(el);
        }
    }
}

int main(){

	int i , j , k , t ;
    cin >> t;
    while(t--){
        cin >> n;
        memset(tree,0,sizeof(tree));
        memset(vis,0,sizeof(vis));
        while(n--){
            cin >> x >> y >> z;
            tree[x][0] = (y==0)?0:y;
            tree[x][1] = (z==0)?0:z;  
        }
        disp(1);
        cout << endl;
    }
	return 0;
}
