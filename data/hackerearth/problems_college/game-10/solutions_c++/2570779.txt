#include <iostream>
using namespace std;

int main()
{
    int t,z,n,m,i,j,c;
    cin>>t;
    for(z=0;z<t;z++)
    {
    	cin>>n>>m;
    	int a[n],b[n];
    	
    	for(i=0;i<n;i++)
    	a[i]=i;
    	c=0;i=0;
    	while(c<n)
    	{
    		b[c++]=a[n-i-1];
    		b[c++]=a[i];
    		i++;
    	}
    	
    	for(i=0;i<n;i++)
    	if(b[i]==m)
    	{
    	cout<<i<<endl;
    	break;
    	}
    }
}
