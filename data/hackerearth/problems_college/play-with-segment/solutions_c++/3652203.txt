#include<bits/stdc++.h>
using namespace std;
int ar[11111111];
int main()
{
    cin.sync_with_stdio(false);
    int i,j,n,ans,t;
    cin>>t;
    while(t--)
    {
    cin>>n;
    for(i=1;i<=n;i++)
        cin>>ar[i];
    ans=-1;
    int minn=INT_MAX,maxx=0;
    int s,e;
    s=e=1;
    maxx=minn=ar[1];
    for(i=2;i<=n;i++)
    {
        maxx=max(maxx,ar[i]);
        minn=min(minn,ar[i]);
        if(maxx-minn>1)
        {
            ans=max(ans,e-s+1);
            int temp=ar[i];
            int j=i;
            maxx=minn=temp;
            while(1)
            {
                j--;
                if(abs(temp-ar[j])<=1)
                {
                    maxx=max(maxx,ar[j]);
                    minn=min(minn,ar[j]);
                }
                else
                    break;
            }
            s=j+1;
            e=i;
        }
        else
            e++;
    }
    ans=max(e-s+1,ans);
    cout<<ans<<'\n';
    }
}
