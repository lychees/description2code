N, M = raw_input().split()
N, M = int(N), int(M)

rowSums = [0] * N 
colSums = [0] * M

for i in xrange(N):
    row, s  = raw_input().split(), 0
    for j in xrange(M):
        v = int(row[j])
        s += v
        colSums[j] += v
    rowSums[i] += s

maxor = 0
for i in xrange(N):
    for j in xrange(M):
        x = rowSums[i] ^ colSums[j]
        maxor = max(maxor, x)
print maxor
