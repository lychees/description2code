#include <iostream>
#define INF 10000000
#include <vector>
using namespace std;

int main()
{
	vector <int> dp(101,INF);
	dp[6] = 1;
	dp[8] = 1;
	
	for(int i=9;i<=100;i++)
	{
		if(dp[i-6]!=INF)	
			dp[i] = min(dp[i],dp[i-6]+1);
		if(dp[i-8]!=INF)
			dp[i] = min(dp[i],dp[i-8]+1);
	}
    
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n;
    	scanf("%d",&n);
    	if(dp[n]==INF)
    		printf("-1\n");
    	else
    		printf("%d\n",dp[n]);
    }
    return 0;
}
#include <stdio.h>
