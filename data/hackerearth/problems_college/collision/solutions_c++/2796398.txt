#include <bits/stdc++.h>
using namespace std;

int main()
{
   	int t=0;
   	scanf("%d",&t);
   	while(t--)
   	{
   		int n=0,a=0,i=0,ans=0,l=0;
   		scanf("%d",&n);
   		set<int> s;
   		for(i=0;i<n;i++)
   		{
   		scanf("%d",&a);
   		s.insert(a);
   		}
   		l=s.size();
   		ans=abs(l-n);
   		cout<<ans<<endl;
   	}
    return 0;
}
