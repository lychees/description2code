using namespace std;

typedef long long ll;

bool check(ll x1, ll y1, ll x2, ll y2) {
  if (x2 == 0 && y2 == 0)
    return x1 == 0 && y1 == 0;
  ll r = x2 * x2 + y2 * y2;
  return !((x1 * y2 - x2 * y1) % r) && !((x1 * x2 + y1 * y2) % r);
}

int main() {
  int x1, y1, x2, y2, x3, y3;
  scanf("%d %d %d %d %d %d", &x1, &y1, &x2, &y2, &x3, &y3);
  puts(check(x2 - x1, y2 - y1, x3, y3) || check(x2 - y1, y2 + x1, x3, y3) ||
               check(x2 + x1, y2 + y1, x3, y3) ||
               check(x2 + y1, y2 - x1, x3, y3)
           ? "YES"
           : "NO");
  return 0;
}
