#include <iostream>
using namespace std;

int main() {
	// your code goes here
	int t, n, a[10009];
	cin>>t;
	while(t--){
		cin>>n;
		for(int i=0; i<n; i++){
			cin>>a[i];
			
		}
		
		int start = 0;
		int end = n-1;
		bool success = true;
		
		while(end> 0 && a[end] > a[end-1])
			end--;
		
		while(start < n-1 && a[start] < a[start+1])
			start++;
			
		
		if(start > 0 && a[end] < a[start-1])
			success = false;
			
		
		if(end < n-1 && a[start] > a[end+1])
			success = false;
		
		
		
		if(start >= end){
			success = false;
			goto point;
		}
			
			
		for(int i=start; i<end; i++){
			if(a[i] < a[i+1])
				success = false;
		}
		
		
		point:
		if(!success)
			cout<<"-1 -1"<<endl;
		else
			cout<<start+1<<" "<<end+1<<endl;
	}
		
		
	return 0;
}
