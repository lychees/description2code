using namespace std;
const int N=2e6+5;
const int L=1e5+5;
const ll mod=1e9+7;
ll ans[L];
int ar[N];
vi v;
void pre()
{
	v.eb(2);
	f(i,0,N)
		ar[i]=0;
	for(int i=3;i*i<N;i+=2)
	{
		if(!ar[i])
		{
			for(int j=i*i,k=2*i;j<N;j+=k)
				ar[j]=1;
		}
	}
	for(int i=3;i<N;i+=2)
		if(!ar[i])
			v.eb(i);
	ans[0]=v[0];
	f(i,1,L)
	{
		ans[i]=ans[i-1]*v[i];
		if(ans[i]>=mod)
			ans[i]%=mod;
	}
}
int main() 
{
	pre();
	TC()
	{
		int n;
		scanf("%d",&n);
		printf("%lld\n",ans[n-1]);
	}
	return 0;
}
