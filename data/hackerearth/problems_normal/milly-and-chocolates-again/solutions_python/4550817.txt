import numpy as np
def bsearch(l, n, i, j):
	if len(l) == 1:
		return i
	else:
		if l[:len(l)/2][-1] >= n:
			return bsearch(l[:len(l)/2], n, i, len(l)/2)
		else:
			return bsearch(l[len(l)/2:], n, i + len(l)/2, j)
