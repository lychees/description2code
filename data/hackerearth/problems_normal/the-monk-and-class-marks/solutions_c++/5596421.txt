#include <iostream>
using namespace std;

struct node
{
	long marks;
	string name;
	node * left;
	node * right;
};

void add_list(string str, long mrk, node * &root)
{
	node * temp , * t1 , * head;
	head = new node();
	head -> name = str;
	head -> marks = mrk;
	temp = root;
	
	if(root == NULL)
		root = head;
		
	else
	{
		long ind(0);
		while(temp != NULL)
		{
			t1 = temp;
			if(mrk < temp -> marks)
			{
				temp = temp -> left;
				ind = 1;
			}
			
			else if(mrk > temp -> marks)
			{
				temp = temp -> right;
				ind = 2;
			}
			
			else
			{
				if(str < temp -> name)
				{
					temp = temp -> right;
					ind = 2;
				}
				
				else
				{
					temp = temp -> left;
					ind = 1;
				}
			}
		}
		
		if(ind == 1)
			t1 -> left = head;
			
		else
			t1 -> right = head;
	}
}

void print_list(node * temp)
{
	if(temp != NULL)
	{
		print_list(temp -> right);
		cout<<temp -> name<<" "<<temp -> marks<<endl;
		print_list(temp -> left);
	}
}

int main()
{
	long n;
    cin>>n;
    
    string str;
    long mrk;
    node * root = NULL;
    for(long i = 0 ; i < n ; i++)
    {
    	cin>>str>>mrk;
    	add_list(str,mrk,root);
    }
    
    print_list(root);
    return 0;
}
