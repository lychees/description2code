#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
#define F(i,a,b) for(int i = (int)(a); i <= (int)(b); i++)
#define RF(i,a,b) for(int i = (int)(a); i >= (int)(b); i--)
#define pb push_back
#define mp make_pair
#define gc getchar_unlocked
#define MAX 3005
bool Position[MAX][MAX];
int read_int(){
    register int x = 0, c = gc();
    int sign = 1;
    while(c!='-' && (c<48 || c>57) ) c = gc();
    if(c=='-')
        sign = -1, c = gc();
    while(c>=48 && c<=57)
        x = (x<<1) + (x<<3) + c - 48, c = gc();
    return sign*x;
}
int main()
{
    #ifndef ONLINE_JUDGE
        freopen("input.txt","r",stdin);
        freopen("output.txt","w",stdout);
    #endif
    int T,x,y,K,previ,prevj;
    F(i,1,3000) Position[i][1] = Position[1][i] = 1;
    F(i,1,3000) Position[i][0] = Position[0][i] = 0;
    Position[0][0] = 0;
    F(i,2,3000)
    {
        F(j,2,3000)
        {
            if(i == j)
            {
                Position[i][j] = 1;
                continue;
            }
            K = __gcd(i,j);
            previ = i - K;
            prevj = j - K;
            if(previ == 0 || prevj == 0)
                Position[i][j] = 1;
            else Position[i][j] = !Position[previ][prevj];

        }
    }
    T = read_int();
    string First;
    while(T--)
    {
        x = read_int();
        y = read_int();
        cin>>First;
        if(Position[x][y] == 1)
        {
            if(First[0] == 'C') printf("Chandu\n");
            else printf("Daspal\n");
        }
        else
        {
            if(First[0] == 'D') printf("Chandu\n");
            else printf("Daspal\n");
        }
    }
    return 0;
}
