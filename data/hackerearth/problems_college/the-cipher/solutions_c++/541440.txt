#include <iostream>
#include <cstdio>

using namespace std;

int main()
{
        int test_cases;
        scanf("%d",&test_cases);
        while(test_cases > 0)
        {
                int N,arr[10000],i=0;
                scanf("%d",&N);
                while(i<N)
                {
                        scanf("%d",&arr[i]);
                        arr[i]=~arr[i];
                        i = i+1;
                }
                
                //your code here
                for(int j=0;j<i;j++)
                	printf("%c",arr[j]);
                printf("\n");
                test_cases--;
        }
        return 0;
}
