#include <algorithm>
#include <cmath>
#include <cstdio>
#include <iostream>
#include <map>
#include <queue>
#include <vector>
using namespace std;

int main() {
  int n;
  long long anw, p, p2;
  string t;
  cin >> n >> p;
  anw = 0;
  p2 = 1;
  for (int i = 0; i < n; i++) {
    cin >> t;
    if (t != "half")
      anw += p * p2 - p / 2;
    p2 *= 2;
  }
  cout << anw << endl;
}
