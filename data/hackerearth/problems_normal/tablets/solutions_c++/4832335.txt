#include<iostream>
using namespace std;

int main()
{
	int N;
	cin >> N;
	int a[N];
	for(int i=0;i<N;i++)cin >> a[i];
	int b[N];
	b[0]=1;
	for(int i=1;i<N;i++)
	{
		if(a[i]>a[i-1])b[i]=b[i-1]+1;
		else b[i]=1;
	}
	for(int i=N-2;i>=0;i--)
	{
		if(a[i]>a[i+1] && b[i]<=b[i+1])b[i]=b[i+1]+1;
	}
	long long sm=0;
	for(int i=0;i<N;i++)sm+=b[i];
	cout<<sm<<endl;
	return 0;
}
