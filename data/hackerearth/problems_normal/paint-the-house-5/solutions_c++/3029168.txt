#include <bits/stdc++.h>
 
#define pb push_back
#define pp pop_back
#define f first
#define s second
#define mp make_pair
#define sz(a) int((a).size())
#ifdef _WIN32
#  define I64 "%I64d"
#else
#  define I64 "%lld"
#endif
#define fname "."
 
typedef long long ll;
typedef unsigned long long ull;
 
const int MAX_N = (int)1e5 + 123;
const double eps = 1e-6;
const int inf = (int)1e9 + 123;
 
using namespace std;
 
int n;
pair < int, int > a[MAX_N];
ll d;
ll ans;
 
int main() {
	#ifdef Nick
	freopen(fname"in","r",stdin);
	freopen(fname"out","w",stdout);
	#endif
	cin >> n >> d;
	for (int i = 1, t, x, y; i <= n; i++) {
		cin >> t >> x >> y;
		if (a[t].f == 0) {
			a[t] = mp(x, y);
			continue;
		}
		if (a[t].s < y || (a[t].s == y && a[t].f > x))
			a[t] = mp(x, y);
	}	
	int len = 0;
	for (int i = 1; i <= (int)1e5; i++) {
		if (len < a[i].s)
			len = a[i].s, ans += a[i].f;
		d -= len;
		if (d <= 0)
			break;
	}
	cout << ans;
	return 0;
}
