n = input()
petals = sorted(map(int, raw_input().split()))
petals = petals[::-1]
mysum = 0
flag = False
iterate = len(petals)-1
fail = 0
while iterate >= 0 and flag != True:
    local = 0
    for i in range(0, len(petals)):
        local = local+petals[i]
    # print local #ress
    if local % 2 == 1:
        flag = True
        mysum = local
        break
    else:
        # print iterate
        temp = petals[iterate]
        petals[iterate] = 0

        if iterate < len(petals)-1:
            petals[iterate+1] = fail
        fail = temp
        iterate = iterate-1

if mysum <= 0:
    print ':('
else:
    print mysum
