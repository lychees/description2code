#include <iostream>
#include <bits/stdc++.h>
using namespace std;

void print(string str, vector<bool> v) {
	int n = str.length();
	for(int i=0;i<n;i++) {
		if(v[i]) {
			cout<<str[i];
		}
	}
	cout<<endl;
}

int main()
{
    string str;
    cin>>str;
    map<char, vector<int> > m;
    int len = str.length();
    vector<bool> v;
    
    for(int i=0;i<len;i++) {
    	if(m.find(str[i])==m.end()) {
    		vector<int> vec;
    		m[str[i]] = vec;
    	}
    	m[str[i]].push_back(i);
    	v.push_back(true);
    }
    int t;
    cin>>t;
    while(t--) {
    	int k;
    	char ch;
    	cin>>k>>ch;
    	if(m.find(ch)!=m.end() && m[ch].size()>=k) {
    		int d = m[ch][k-1];
    	//	str = str.substr(0, d) + str.substr(d+1, len-d-1);
    		v[d] = false;	
    	//	len--;
    		m[ch].erase(m[ch].begin() + k-1);
    	//	cout<<str<<endl;
    	//	print(str, v);
    	}
    	//print(str, v);
    }
    print(str, v);
    return 0;
}
