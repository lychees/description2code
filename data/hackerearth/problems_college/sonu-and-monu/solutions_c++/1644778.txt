#include<bits/stdc++.h>
using namespace std;

int main()
{
long long int t ,n , p; cin>>t; while(t--)
{
cin>>n>>p;
char c[100011];
long long int a[100011],i,sum=0,s,k,po;

//for(i=0;i<100011;i++) {c[i]='\0'; a[i]=0;}

cin>>c;
a[0]=0;
for(i=0;i<n;i++)
{
if(c[i]=='1') {a[i+1]=a[i]+1;}
else a[i+1]=a[i];
}
//for(i=1;i<=n;i++) cout<<a[i]<<" ";
for(i=1;i<=n;i++)
{
if(c[i-1]=='1')
  {
       s = i+p;
       if(s>n) s=n;
       else s = i+p;
       k = abs(a[i]-a[s]);
       k = k*2+1;
       sum+=k;
  }
else continue;
}

po=n*n;
long long m = __gcd(po,sum);
sum = sum/m;
po = po/m;
cout<<sum<<"/"<<po<<"\n";
}
return 0;
}
