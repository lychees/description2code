#include <iostream>
using namespace std;

int main() {
	long n,x;
	cin>>n>>x;
	long arr[n+1];
	for(long i =0;i<n;i++)
	{
		cin>>arr[i];
	}
	long count=0,flag=0;
	for(long j=0;j<n;j++)
	{
		if(arr[j]<=x)
		count++;
		if(arr[j]>x)
		flag++;
		if(flag>1)
		break;
	}
	cout<<count<<endl;
	return 0;
}
