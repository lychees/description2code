#include <iostream>
using namespace std;
 
int main()
{
   	int n, m;
   	cin >> n >> m;
   	int matrix[n][m];
   	for(int i = 0; i < n; i++)
   	{
   		for(int j = 0; j < m; j++)
   		{
   			cin >> matrix[i][j];
   		}
   	}
   	
   	
   	int qNo = 0;
   	cin >> qNo;
   	for(int i = 0; i < qNo; i++)
   	{
   		int qVal; bool flag = false;
   		cin >> qVal;
   		int k = n - 1, j = 0;
   		
		while(k >= 0 && j <= m - 1)
		{	
			if(qVal > matrix[k][j])
			{
				j++;
			}
			else if(qVal < matrix[k][j])
			{
				k--;
			}
			else if(qVal == matrix[k][j])
			{
				cout << k << " " << j << endl;
				flag = true;
				break;
			}
			
		}
		if(flag == false)
			cout <<"-1 -1" << endl;
   	}
}
