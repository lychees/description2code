#include <iostream>
using namespace std;

long long HCF(long long a,long long b){
	if (b == 0)
		return a;
	else
		return HCF(b,a%b);
}

int main()
{
	int t;
	cin >> t;
	while (t > 0){
		int n;
		cin >> n;
		long long l[n];
		long long min = 10000000000;
		for (int i = 0; i < n; i++){
			cin >> l[i];
		}
		long long hcf = HCF(l[0],l[1]);
		for (int i = 2; i < n; i++){
			hcf = HCF(hcf,l[i]);
		}
		cout << hcf << endl;
		t--;
	} 
	return 0;
}
