#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;
 
int main()
{
    int m;
    cin>>m;
    vector<int>V;
    //int ar[m];
    //int mini=INT_MIN;
    for(int i=0;i<m;i++)
    {
    	int temp;
    	cin>>temp;
    	V.push_back(temp);
    }
    
    vector<int>::iterator it;
    sort(V.begin(),V.end());
    it=unique(V.begin(),V.end());
    V.resize(distance(V.begin(),it));
    
    int lim=V[1];
    if(lim>398131299)
    lim=398131299;
    for(int i=2;i<=lim;i++)
    {
    	int val=V[0];
    	val%=i;
    	int flag=1;
    	for(int j=1;j<m;j++)
    	{
    		int comp=V[j];
    		comp%=i;
    		if(comp!=val)
    		{
    			flag=0;break;
    		}
    	}
    	if(flag)
    	cout<<i<<" ";
    }
    return 0;
}
