#include<cstdio>
#include<iostream>
#include<cmath>
#define pi 3.1415
#define e 2.7182818284590452353

typedef long long int LLD;
typedef unsigned long long int LLU;

using namespace std;

LLD fact_length(LLD n){
	if(n == 1 || n == 0)
		return 1;
	double len;
	LLD lenn;
	len = ((log(2 * pi *n)/2+n*(log(n)-1))/log(10));
	len = ceil(len);
	lenn = (LLD)len;
	return lenn;
}

LLD trailing_zeros(LLD n){
	LLD r = 0;
	while(n){
		n /= 5;
		r += n;
	}
	return r;
}
 
int main(){
    int t;
    LLD n;
	cin >> t;
	while(t--){
		cin >> n;
		cout << fact_length(n) - trailing_zeros(n) << endl;
	}
    return 0;
}
