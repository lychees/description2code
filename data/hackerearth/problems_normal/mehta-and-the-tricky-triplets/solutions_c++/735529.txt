#include<bits/stdc++.h>
using namespace std;
long long ncr(int n)
{
	return n*(n-1)/2*(n-2)/3;
}
int main()
{
	int N;
	scanf("%d",&N);
	vector<long long>A(N);
	for(int (i)=0;(i)<(int)(N);++(i))
	scanf("%lld",&A[i]);
	vector<int>di(N);
	int s=0;
	//int di[16]={0};
	for(int i=0;i<(int)(N);i++)
	{   
		long long  x=A[i];
		s=0;
		while(x>0){
			int d = x % 10;
			if(d == 2)
				s |= 1 << 0;
			else if(d == 3)
				s |= 1 << 1;
			else if(d == 5)
				s |= 1 << 2;
			else if(d == 7)
				s |= 1 << 3;
			x /= 10;
		}
		di[i]=s;
	}
	long long res=0,ans=0;
	int cnt[16]={0};
	for(int i=0;i<N;i++)
	{
		int x=di[i];
		res=0;
		for(int j=1;j<1<<4;j++)
		{
			for(int k=1;k<1<<4;k++)
			{
				if((x&j&k)!=0)
				{   int a=cnt[j],b=cnt[k];
					res+=(j==k?(long long)a*(a-1):(long long)a*b);
				}
			}
		}
		res=res/2;
		ans+=res;
		cnt[x]++;
	}
	printf("%lld\n",ans);
	return 0;
}
