#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d", &x)
const int maxn = 1000001;
long long int arr[maxn] = {0};

void sieve(){
	for(int i=2;i<=sqrt(maxn);i++){
		if(arr[i] == 0){
			for(int j = i*i;j<maxn;j+=i)
				arr[j] = 1;
		}
	}
}

void preprocess(){
	sieve();
	arr[1] = 1; 
	for(int i=1;i<maxn;i++){
		if(arr[i] == 0) arr[i] = i + arr[i-1];
		else arr[i] = arr[i-1];
	}
}

int main(){
	preprocess();
	int n; sd(n);
	while(n--){
		int l,r; sd(l); sd(r);
		printf("%lld\n",arr[r] - arr[l-1]);
	
	}
	
}
