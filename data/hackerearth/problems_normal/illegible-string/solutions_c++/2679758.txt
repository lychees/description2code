#include <bits/stdc++.h>
using namespace std;

int main()
{
	string str;
    long int N , W=0,V=0,max=0,min=0;
    cin>>N;
    cin>>str;
    for(long int i=0;i<N;i++)
    {
    	if(str[i]=='w')
    	{
    		W++;
    	}
    }
   max=N+W;
   for(long int i=0;i<N-1;i++)
    {
    	if(str[i+1]==str[i]&&str[i]=='v')
    	{
    	str[i]='w';
    	str[i+1]='1';
    	}
    	else if(str[i]=='v'&&str[i+1]=='w')
    	{
    	str[i]='w';
    	str[i+1]='v';
    	}
    }
    for(long int i=0;i<N;i++)
    {
    	if(str[i]>='a'&&str[i]<='z')
    	{
    		min++;
    	}
    }
    cout<<min<<" "<<max;
}
