#include<iostream>
#include<string>
#include<stdio.h>
using namespace std;

int main()
{
  int n;
  cin>>n;
  char ch;
  scanf("%c",&ch);
  string str;
  while(n--)
    {
      cin>>str;
      char oper;
      int i;
      long long num=0,ans=0;
      for(i=str.length()-1;i>=0&&str.at(i)!='+'&&str.at(i)!='-'&&str.at(i)!='*';i--)
        {
          num*=10;
          num+=(str.at(i)-'0');
        }
      ans=num;

      for(;i>=0;)
        {

          num=0;
          oper=str.at(i);
          --i;
			for(;i>=0 &&str.at(i)!='+'&&str.at(i)!='-'&&str.at(i)!='*';i--)
            {
              num*=10;
              num+=(str.at(i)-'0');
            }

          switch(oper)
            {
            case '+' : ans=ans+num;
              break;
            case '*': ans=ans*num;
              break;
            case '-': ans=ans-num;
              break;
            
            }

        }
      cout<<ans<<endl;
    }
}
