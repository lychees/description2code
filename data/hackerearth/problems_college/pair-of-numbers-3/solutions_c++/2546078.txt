#include <iostream>
using namespace std;

int main()
{long long int t,n,j,i,c,s,m;
    cin>>t;
    while(t--)
    {c=0;
    	cin>>n;
    	for(i=1;i<n;i++)
    	{
    		for(j=i+1;j<=n;j++)
    		{m=i*j;
    		s=i+j;
    			if(m%s==0)
    			c++;
    		}
    	}
    	cout<<c<<endl;
    }
    return 0;
}
