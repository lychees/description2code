#include <iostream>
#include <unordered_map>
using namespace std;
int main()
{	int t;
	cin>>t;
	for(int i=0;i<t;i++){
		int n;
		cin>>n;
		int a[n];
		for(int i=0;i<n;i++)
			cin>>a[i];
		unordered_map<int,int>m;
		int j=0,l=1;unsigned s=0;
		while(j<n){
			if(m.find(a[j])==m.end()){
				m[a[j]]=j;
				j++;
			}
			else{
				if(s<m.size()){
					s=m.size();
					l=j-s+1;
				}
				j=m[a[j]]+1;
				m.clear();
			}
		}
		if(s<m.size()){
			s=m.size();
			l=n-s+1;
		}
		cout<<l<<" "<<(l+s-1)<<endl;
		}
    return 0;
}
