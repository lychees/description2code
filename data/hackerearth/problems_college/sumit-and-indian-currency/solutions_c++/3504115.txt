#include<iostream>
#include<cstdio>
#include<cstring>
#include<string>
#include<algorithm>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		string str;
		cin>>str;
		string res="";
		if(str.length()>3){
			reverse(str.begin(),str.end());
			for(int i=0;i<3;i++){
				res+=str[i];
			}
			res+=',';
			int count=0;
			for(int i=3;i<str.length()-1;i++){
				count++;
				res+=str[i];
				if(count==2){
					res+=',';
					count=0;
				}
			}
			res+=str[str.length()-1];
			reverse(res.begin(),res.end());
			cout<<res<<endl;
		}
		else{
			cout<<str<<endl;
		}
	}
}
