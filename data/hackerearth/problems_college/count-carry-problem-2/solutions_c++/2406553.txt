#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long int t,ans,n,a,b,r;
    cin>>t;
    while(t--)
    {
        cin>>a>>b;r=0;ans=0;
        if(a<b){
            swap(a,b);
        }
        while(a>0){
            if((a%10+b%10+r)>9){
                r=1;
                ans++;
            }
            else{
                r=0;
            }
            a/=10;
            b/=10;
        }
        if(ans==0){
            cout<<"No carry operation\n";
        }
        else{
            if(ans==1){
                cout<<"1 carry operation\n";
            }
            else{
                cout<<ans<<" carry operations\n";
            }
        }
    }
    return 0;
}
