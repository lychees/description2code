#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int test = 0;
    cin>>test;
    while(test--)
    {
    	string str;
    	string rstr;
    	bool flag = true;
    	cin>>str;
    	int len = str.length();
    	rstr = str;
    	reverse(rstr.begin(),rstr.end());
    	for(int i=1;i<len;i++)
    	{
    		int n = str[i]-str[i-1];
    		int r = rstr[i]-rstr[i-1];
    		if(n<0)
    			n=-n;
    		if(r<0)
    			r=-r;
    		if(n!=r)
    		{
    			flag=false;
    			break;
    		}
    	}
    	if(flag)
    		cout<<"Funny"<<endl;
    	else
    		cout<<"Not Funny"<<endl;
    }
    return 0;
}
