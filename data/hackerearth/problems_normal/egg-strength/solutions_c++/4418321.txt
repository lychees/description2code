#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    int dp[501][501];
    memset(dp,0,sizeof(dp));
    for(int i=1;i<=500;i++)
    {
        for(int j=1;j<=500;j++)
        {
            if(j==1)
                dp[i][j]=i;
            else if(i==1)
            {
                dp[i][j]=1;
            }
            else
            {
                
                dp[i][j]=INT_MAX;
                for(int k=1;k<=i;k++)
                {
                    int res=1+max(dp[k-1][j-1],dp[i-k][j]);
                    if(res<dp[i][j])
                        dp[i][j]=res;
                }
            }
        }
    }
   /* for(int i=1;i<=10;i++)
    {
        for(int j=1;j<=10;j++)
        {
            cout<<dp[i][j]<<" ";
        }
        cout<<endl;
    }*/
    while(t--)
    {
        int n,k;
        cin>>n>>k;
        cout<<dp[n][k]<<endl;
    }
    return 0;
}
