cus = raw_input()
while(cus!='0'):
	no_of_rooms = int(raw_input())
	occ = []
	unocc = []
	cus = list(cus)
	for i in cus:
		length = len(occ)
		if length<no_of_rooms and i not in occ:
			occ.append(i)
		elif i in occ:
			occ.remove(i)
		else:
			unocc.append(i)
	print len(set(unocc))
	cus = raw_input()
