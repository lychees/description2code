#include<bits/stdc++.h>
using namespace std;
int main(){
    int t;
    
    cin>>t;
    while(t--){
    	int flag=0;
        string str;
        cin>>str;
        for(int i=0;i<str.length();i++){
            if(str[i]=='a'||str[i]=='e'||str[i]=='i'||str[i]=='o'||str[i]=='u'||str[i]=='A'||str[i]=='E'||str[i]=='I'||str[i]=='O'||str[i]=='U'){
                cout<<str[i];
                flag=1;
            }
        }
        if(flag==0){
            cout<<"No";
        }
        cout<<endl;
    }
    return 0;
}
