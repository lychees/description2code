

    #include <iostream>

    #define MAXN 100100
    #define MOD 1000000007
    using namespace std;
    typedef long long ll;
    ll ans[1001000],cavab[1001000];
    ll a,b,n,k;
    void calc(ll z){
    if(z==1)
        return;
    if(z%2==0){
        calc(z/2);
        for(int i=1;i<=k;i++)
            ans[i]=(ans[i]*ans[i])%MOD;
    }
    else
    {
        calc(z-1);
        for(int i=1;i<=k;i++)
            ans[i]=(ans[i]*cavab[i])%MOD;
    }
    }
    ll solve(ll t,ll pover){
    if(pover==1)
        return t;
    if(pover%2==1)
        return (t*solve(t,pover-1))%MOD;
    else
    {
        ll r=solve(t,pover/2)%MOD;
        return (r*r)%MOD;
    }
    }
    ll ret=1;
    int main()
    {
        cin>>a>>b>>n>>k;
        ret=(ret*solve(a,k))%MOD;
        ret=(ret*solve(b,n-k))%MOD;
        k=min(k,n-k);
        for(int i=1;i<=100000;i++)
            ans[i]=cavab[i]=i;
        calc(MOD-(ll)2);
        for(int i=1;i<=k;i++)
            ret=(ret*ans[i])%MOD;
        for(ll i=n;i>n-k;i--)
            ret=(ret*i)%MOD;
        cout<<ret<<endl;
        return 0;
    }
