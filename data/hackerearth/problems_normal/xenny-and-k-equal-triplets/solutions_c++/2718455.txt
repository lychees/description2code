int main()
{
	
	int t;
	long long k, num;
	long long count;
long long	int n;
	scanf("%d",&t);
	assert(t>=1 && t<=50);
	while(t--)
	{
		count=0;
		scanf("%lld",&n);
		assert(n>=1 && n<=1000000);
		scanf("%lld",&k);
		assert(k>=1 && k<=1000000000);
		for(int i=0;i<n;i++)
		{
			scanf("%lld",&num);
			assert(num>=1 && num<=1000000000);
			if(num==k)count++;
		}
		long long numer=count*(count-1)*(count-2);
		long long denom=n*(n-1)*(n-2);
		
		//long long gcd1=calGCD(numer, denom);
		long long gcd1=__gcd(numer, denom);
		if(gcd1==0)cout<<numer<<"/"<<denom;
		else
		cout<<(numer/gcd1)<<"/"<<(denom/gcd1);
	
		printf("\n");
	}
	//system("pause");
	return 0;
}
