import sys
T = int(sys.stdin.readline())
for i in xrange(T):
	line = sys.stdin.readline().strip()
	suml,sumr=0,0
	for j in xrange(len(line)):
		if j < len(line)/2:
			suml = suml + int(line[j])
		else:
			sumr = sumr + int(line[j])
	if suml == sumr:
		print "YES"
	else:
		print "NO"
