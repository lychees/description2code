#include<bits/stdc++.h>

using namespace std;

bool dp[10005];
int arr[105];
int main()
{
 //freopen("input.in", "r", stdin);
 int n,t;
 cin>>t;
 while(t--)
 {
     for(int i=0;i<10001;i++)
     dp[i]=false;


     cin>>n;
     for(int i=0;i<n;i++)
     cin>>arr[i];

 dp[0]=true;
 for(int i=0;i<n;i++)
 {

     for(int j=10000;j>=0;j--)
     {
         if((j-arr[i])>=0)
            dp[j]|=dp[j-arr[i]];
     }
 }
 int cnt=0;
 for(int i=0;i<10001;i++)
 {
     if(dp[i])
        cnt++;
 }
 cout<<cnt<<"\n";



}
}
