#include<bits/stdc++.h>

using namespace std;

#define Max 16

bool edge[Max][Max];
int ans;


int main(int argc, char const *argv[])
{
	int t;
	int n,m,x,y;
	memset(edge,0,sizeof(edge));
	cin>>n>>m;
	while(m--){
		cin>>x>>y;
		edge[x][y] = 1;
		edge[y][x] = 1;
	}
	vector<int> v;
	for(int i=1;i<=n;i++){
		v.push_back(i);
	}
	do{
		int i;
		for(i=0;i<n-1;i++){
			if(!edge[v[i]][v[i+1]]){
				break;
			}
		}
		if(i==n-1){
			ans++;
		}
	}
	while(next_permutation(v.begin(),v.end()));

	cout<<ans<<"\n";
	
	return 0;
}
