#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cmath>
#include<cstring>
#include<string>
#include<vector>
#include<map>
#include<stack>
#include<queue>
#include<algorithm>

using namespace std;

int main()
{
	int t;

	cin>>t;

	while(t--)
	{
		long long int z, m, n;

		cin>>z>>m>>n;

		if(n>m)
			swap(n, m);

		long long int ans = 0;

		for(int i=1; i<=m && i<=sqrt(z); i++)
		{
			if(z%i == 0)
			{
				int flag1 = 0, flag2 = 0;
				if(z/i <= n)
				{
					flag1 = 1;
					ans++;
				}
				if(i <= n && z/i <= m)
				{
					flag2 = 1;
					ans++;
				}

				if(flag1 && flag2 && (i == z/i))
					ans--;
			}
		}

		cout<<ans<<endl;
	}

	return 0;
}
