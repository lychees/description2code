#include<iostream>
using namespace std;
int arr[10005];
	 int dif[10005];
	 
    int main()
    {
    	int i,j,k,n,m,t,min;
    	cin>>t;
		while(t--)
		{
			cin>>n;
			min = 1000000;
			for(i=0;i<n;i++)
			{
				cin>>arr[i];
				if(arr[i] < min)
				{
					min = arr[i];
				}
			}
			for(i=0;i<n;i++)
			{
				dif[i] = arr[i] - min;
			}
			int c =0;
			for(i=0;i<n;i++)
			{
				if(dif[i] >= 5)
				{
					c += dif[i]/5;
					dif[i] %= 5;
				}
				if(dif[i] >= 2)
				{
					c += dif[i]/2;
					dif[i] %= 2;
				}
				if(dif[i] == 1)
				{
					c++;
					dif[i] --;
				}
			}
			cout<<c<<endl;
		}
    	return 0;
    }
