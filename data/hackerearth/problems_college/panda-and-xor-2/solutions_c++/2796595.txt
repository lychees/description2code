#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
   void bin(long int);
   int t;
   long int a,b,c;
   scanf("%d",&t);
   while(t--)
   {
   	scanf("%ld%ld",&a,&b);
   	c=a^b;
   	bin(c);
   }
   return 0;
}
void bin(long int c)
{
	int i=0;
	//printf("%ld\n",c);
	while(c)
	{
		i+=c&1;
		c>>=1;
	}
	if(i%2==0)
	printf("YES\n");
	else
	printf("NO\n");
}
