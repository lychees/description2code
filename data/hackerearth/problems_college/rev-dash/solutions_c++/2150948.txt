#include<bits/stdc++.h>
using namespace std;
int main()
{
	char s[100000];
	scanf("%s",s);
	int l =strlen(s);
	for(int i =0;i<l;i++)
	{
		printf("%c",s[i]);
		if(i != l-1)
			printf("-");
	}
	printf("\n");
	for(int i =0;i<l;i++)
	{
			printf("%c",s[l-i-1]);
			if(i !=l-1)
				printf("-");
	}
	printf("\n");
	return 0;
}
