#include <bits/stdc++.h>
#define ll long long int
#define s(x) scanf("%lld", &x)
#define mod (ll)(1e9+7)

using namespace std;

ll power(ll x,ll y) {
	if (y == 0) {
		return 1;
	}
	ll tmp = power(x,y/2);

	if (y & 1) {
		return (((tmp * tmp) % mod) * x) % mod;
	} else {
		return (tmp * tmp) % mod;
	}
}

ll fermat(ll x) {
	return power(x,mod-2);
}

ll fac[1001],invfac[1001];

void init() {
	fac[0] = 1;

	for (int i = 1; i <= 1000; i++) {
		fac[i] = fac[i-1] * i;
		fac[i] %= mod;
	}
	for (int i = 0; i <= 1000; i++) {
		invfac[i] = fermat(fac[i]);
	}
}

int p[1001];
int dp[1001][1001];
ll rec(int idx,int dexter,int m,int n) {
	if (dexter < m * (idx - dexter)) {
		return 0;
	}
	if (idx == n) {
		if (dexter > (idx - dexter)) {
			return 1;
		}
		return 0;
	}
	if (dp[idx][dexter] != -1) {
		return dp[idx][dexter];
	}
	ll tot = (rec(idx + 1,dexter + 1,m,n) + rec(idx + 1,dexter,m,n)) % mod;
	dp[idx][dexter] = tot;
	return tot;
}

int main()
{
	init();
	ll test,n,m;
	s(test);
	while (test--) {
		s(n);s(m);
		for (int i = 0; i < 1001; i++) {
			for (int j = 0; j < 1001; j++) {
				dp[i][j] = -1;
			}
		}

		cout << rec(0,0,m,n) << endl;
	}

	return 0;
}
