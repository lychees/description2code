t = int(raw_input())
for i in range(t):
    l = []
    n, id = [int(i) for i in raw_input().split()]
    l.append(id)
    for i in range(n):
        d = raw_input()
        if d == 'B':
            l.append(l[len(l)-2])
        else:
            x = d.split(' ')
            next = x[1]
            l.append(next)
    print 'Player', l.pop()
