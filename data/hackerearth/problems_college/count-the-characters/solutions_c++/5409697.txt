#include <iostream>
using namespace std;

int main()
{
	int n;
	cin >> n;
	int i, j, flag;
	string s[200];
	for(i = 0; i < n; i++)
	{
		cin >> s[i];
	}
/*	for(i = 0; i < n; i++)
	{
		cout << s[i] << endl;
	}
	cout << endl;		*/
	int hash[n][26];
	for(i = 0; i < n; i++)
	{
		for(j = 0; j < 26; j++)
		{
			hash[i][j] = 0;
		}
	}
	int x;
	for(i = 0; i < n; i ++)
	{
		x = s[i].size();
		for(j = 0; j < x; j++)
		{
		//	cout << s[i][j] - 97<< " ";
			hash[i][s[i][j] - 97]++; 
		}
		//cout << endl;
	}
/*	for(i = 0; i < n; i++)
	{
		for(j = 0; j < 26; j++)
		{
			if(hash[i][j] != 0)
			cout << "hash[" << i << "]" << "[" << j << "]  :" << hash[i][j] << "  ";
		}
		cout << endl;
	}	*/
	int alpha = 0;
	for(i = 0; i < 26; i++)
	{
		if(hash[0][i] > 0)
		{
			flag = 0;
			j = 1;
			while(j + 1 < n)
			{
				if(hash[j][i] != 0 &&  hash[j + 1][i] != 0)
				{
					//cout << hash[j][i] << " ";
					flag = 1;
				}
				else
				{
					flag = 0;
					break;
				}
				j++;
			}
			//cout << endl;
			if(flag == 1)
			{
				alpha++;
			}
		}
	}
	cout << alpha << endl;
}
