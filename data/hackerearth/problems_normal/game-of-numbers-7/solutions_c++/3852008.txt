#include<bits/stdc++.h>
using namespace std;
#define ll long long int
vector<int>bi;
int main()
{
    int t;
    cin>>t;
    bi.resize(1000001);
    //bi.clear();
    while(t--)
    {
    	
    	ll a,b;
    	cin>>a>>b;
    	ll x=min(a,b);
    	ll y=max(a,b);
    	for(ll i=0;i<(y-x+1);i++)
    	 bi[i]=0;
    	for(ll i=2;i*i<=y;i++)
    	{
    		ll j=x;
    		if(j%(i*i))
    		{
    			j=x-(x%(i*i))+i*i;
			}
    		for(;j<=y;j+=i*i)
    		{
    			bi[j-x]=1;
			}
		}
		ll count=0;
		for(ll i=0;i<y-x+1;i++)
		{
			if(!bi[i])
			  count++;
		}
		cout<<count<<endl;
	}
    return 0;
}
