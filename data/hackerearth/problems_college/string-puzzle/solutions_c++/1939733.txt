#include <bits/stdc++.h>
using namespace std;
void permute(int a, int b, char *str)
{
	if(a==b)
	{
		cout<<str<<endl;
	}
	else
	{
		int i;
		for(i=a;i<=b;i++)
		{
			swap(str[i],str[a]);
			permute(a+1,b,str);
			swap(str[i],str[a]);
		}
	}
	
}
int main()
{
    char str[10];
    cin>>str;
    int n;
    cin>>n;
    while(n--)
    {
    	int a,b;
	    cin>>a>>b;
	    permute(a,b,str);
	    int i;
	    cout<<endl;
    }
}
