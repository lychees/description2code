#include <bits/stdc++.h>
#define ll long long int

using namespace std;

int main()
{
    int n,m,h,ans=-1,temp=0;
    cin>>n>>m;
    int A[n],B[n],H[m+1];
    char c,C[m+1];
    for(int i=0;i<n;i++)
        cin>>A[i];
    for(int i=0;i<n;i++)
        cin>>B[i];
    for(int i=1;i<=m;i++)
        cin>>C[i]>>H[i];
    for(int i=1;i<=m;i++)
    {
        c=C[i];h=H[i];
        if(c == 'L')
            temp -= h;
        else
            temp += h;
            
        temp%=n;
        if(temp<0)
        temp+=n;
            if(A[0]==B[temp])
            {
                ans=i;
               break;
            }
    }
    cout<<ans<<endl;
    return 0;
}
