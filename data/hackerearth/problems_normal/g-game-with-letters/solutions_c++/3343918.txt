#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

string gb, lc;
int gbc[26], lcc[26];

void upg();
void upl();
ll score();

int main()
{
    int t, n;
    cin >> t;
    while(t--) {
    	cin >> n >> gb;
    	upg();
    	int ind = 0, sz = 10000000;
    	ll scr = 0;
    	for(int i = 0; i < n; ++i) {
    		cin >> lc;
    		upl();
    		ll res = score();
    		if(res == scr){
    			if(sz > lc.size()){
    				sz = lc.size();
    				ind = i;
    			} 
    		}else if(res > scr){
    			ind = i;
    			sz = lc.size();
    			scr = res;
    		}
    	}
    	cout << ind + 1 << endl;
    }
    return 0;
}

void upg(){
	for(int i = 0; i < 26; ++i){
		gbc[i] = 0;
	}
	int sz = gb.size();
	for(int i = 0; i < sz; ++i){
		gbc[gb[i] - 'a']++;
	}
}

void upl(){
	for(int i = 0; i < 26; ++i){
		lcc[i] = 0;
	}
	int sz = lc.size();
	for(int i = 0; i < sz; ++i){
		lcc[lc[i] - 'a']++;
	}
}

ll score(){
	ll res = 0;
	for(int i = 0; i < 26; ++i){
		res += (ll)(gbc[i]) * (ll)(lcc[i]);
	}
	return res;
}
