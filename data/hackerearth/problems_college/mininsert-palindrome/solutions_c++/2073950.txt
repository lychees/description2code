#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;
string s;
int n;
int solve()
{
	int gap, i, j, table[21][21];
	memset(table, 0, sizeof(table));
	for(gap=1;gap<n;gap++)
	{
		for(i=0,j=gap;j<n;i++,j++)
		{
			if(s[i]==s[j])
			table[i][j]=table[i+1][j-1];
			else
			table[i][j]=(min(table[i][j-1], table[i+1][j]))+1;
		}
	}
	return table[0][n-1];
}
int main()
{
    int t, ans;
    scanf("%d", &t);
    while(t--)
    {
    	scanf("%d", &n);
    	cin>>s;
    	ans=solve();
    	printf("%d\n", ans);
    }
    return 0;
}
