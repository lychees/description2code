# a is always small
def minMoves(a, b):
    #print a, b
    if a == 0 or b == 0:
        return -1
    elif min(a, b) == a:
        return b/a + minMoves(a,b%a)
    else:
        return a/b + minMoves(a%b,b)

tc = int(raw_input())
while tc != 0:
    tc -= 1
    a, b = map(int, raw_input().split())
    print minMoves(a, b)
