#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstdlib>
#include<cstring>
using namespace std;
int main()
{
          int n,i,r=0,s=0;
          scanf("%d",&n);
          int a[n];
          for(i=0;i<n;i++)
          scanf("%d",&a[i]);
          for(i=0;i<n;i++)
          {
                    s=s+a[i];
                    if(s<0)
                              s=0;
                    if(s>r)
                              r=s;
          }
          printf("%d",r);
          return 0;
          
}
