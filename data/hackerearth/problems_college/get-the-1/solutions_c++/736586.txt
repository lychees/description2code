#include <iostream>
using namespace std;
int addsq(int n)
{
    int r,sum=0;
    do
    {
        r=n%10;
        sum+=(r*r);
        n=n/10;
    } while (n!=0);
    return sum;
}
int main()
{
    int x,t,s;
    cin>>t;
    for (int i=0;i<t;i++)
    {
        cin>>x;
        if (x<10)
        {
            if(x==1)
                cout<<"Case #"<<i+1<<":YES"<<endl;
            else
                cout<<"Case #"<<i+1<<":NO"<<endl;
        }
        else
            do
        {
            x=addsq(x);
            if (x<10)
               {
                 if(x==1)
                  cout<<"Case #"<<i+1<<":YES"<<endl;
                 else
                  cout<<"Case #"<<i+1<<":NO"<<endl;
                  break;
                }
        }while(x!=0);
        }
    }
