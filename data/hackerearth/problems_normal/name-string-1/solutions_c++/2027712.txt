#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    int a,b,c;
    int temp1,temp2;
    long int ans;
    while(t--)
    {
    	cin>>a>>b>>c;
    	priority_queue<int>mypq;
    	mypq.push(a);
    	mypq.push(b);
    	mypq.push(c);
    	temp1=mypq.top();
    	mypq.pop();
    	temp2=mypq.top();
    	mypq.pop();
    	if(temp1==temp2)
    	{
    		ans=temp1+temp2;
    	}
    	else
    	{
    		ans=2*min(temp1,temp2)+1;
    	}
    	cout<<ans<<endl;
    }
}
