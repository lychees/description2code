#include<bits/stdc++.h>
using namespace std;

#define ll long long

class node{
	public:
	ll x,y,z;
};

vector<ll> dp(100005);
vector<ll> foo(100005);
vector<node> mm(100005);
int x,y;

int main(){
	int t;
	cin >> t;
	while(t--){
		int n,m;
		cin >> n >> m;
		int lim=max(n,m);
		for(int i=0;i<=lim+5;i++){
			dp[i]=foo[i]=0;
		}
		for(int i=1;i<=m;i++){
			cin >> mm[i].x >> mm[i].y >> mm[i].z;
		}
		int q;
		cin >> q;
		while(q--){
			cin >> x >> y;
			dp[x]+=1;
			dp[y+1]-=1;
		}
		for(int i=2;i<=m;i++){
			dp[i] += dp[i-1];
		}
		for(int i=1;i<=m;i++){
			foo[mm[i].x] += dp[i]*mm[i].z;
			foo[mm[i].y+1] -= dp[i]*mm[i].z;
		}
		for(int i=2;i<=n;i++){
			foo[i] += foo[i-1];
		}
		for(int i=1;i<=n;i++){
			cout << foo[i] << " ";
		}
		cout << "\n";
	}
	return 0;
}
