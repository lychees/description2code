passage = raw_input()
passage = passage.upper()
stringTokens = passage.split(" ")

wordsOrder = []
wordFrequency = {}

for s in stringTokens:
	if(wordFrequency.has_key(s) == False):
		wordFrequency[s] = 1
		wordsOrder.append(s)
	else:
		wordFrequency[s] += 1

for w in wordsOrder:
	print w + " " + str(wordFrequency[w])
