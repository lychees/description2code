#include <iostream>
using namespace std;
long long minm(long long a,long long b)
{
	return a>b?b:a;
}
long long fck(long long n)
{
  long long count = 0;
  if (n && !(n&(n-1)))
    return n;
 
  while( n != 0)
  {
    n  >>= 1;
    count += 1;
  }
 count--;
  return 1<<count;
}
