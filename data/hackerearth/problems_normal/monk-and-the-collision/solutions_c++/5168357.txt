#include <bits/stdc++.h>

using namespace std;

    int main () {
        int t;
    	scanf("%d",&t);
    	
    	while (t--){
    		
    		int n, temp;
    		set < int > st;
    		
    		scanf("%d",&n);
    		
    		for (int i=0; i<n; i++) {
    			
    			scanf("%d",&temp);
    			
    			st.insert(temp%10);
    		}
    		printf("%d\n",n-st.size());
    	}
    	return 0;
    }
