#include <iostream>
using namespace std;

int main()
{
    int t;
    int count;
    cin>>t;
    int n,a[1000];
    while(t--)
    {
    	cin>>n;
    	count=0;
    	for(int i=0;i<n;i++)
    	{	cin>>a[i];  	
    	}
    	for(int i=0;i<n;i++)  
		{if(!(a[i] & 1))
		count++;}
		cout<<count<<endl;
    }
  


return 0;
}
