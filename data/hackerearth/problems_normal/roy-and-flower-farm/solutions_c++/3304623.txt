#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,p;
    	cin>>n>>p;
    	int cost[101],weight[101];
    	for(int i=1;i<=n;i++)
    	{
    		cin>>cost[i]>>weight[i];
    	}
    	int dp[101][10001]={0};
    	for(int i=1;i<=n;i++)
    	{
    		for(int j=0;j<=p;j++)
    		{
    			if(j-weight[i]>=0)
    				dp[i][j]=max(dp[i-1][j],dp[i-1][j-weight[i]]+cost[i]);
    			else
    				dp[i][j]=dp[i-1][j];
    		}
    	}
    	int minsp=0;
    	for(int i=0;i<=p;i++)
    	{
    		if(dp[n][minsp]-minsp<dp[n][i]-i)
    		minsp=i;
    	}
    	cout<<minsp<<" "<<dp[n][minsp]+p-minsp<<endl;
    }
    return 0;
}
