#include <bits/stdc++.h>
using namespace std;
long a[1001][1001];
int minimum[1001][1001];
int min2(int a,int b)
{
    return (a>b?b:a);
}
int min3(int a,int b,int c)
{
    return (a<b?(min2(a,c)):(min2(b,c)));
}
int main()
{
    int n=0,i=0,j=0;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    for(j=0;j<n;j++)
    scanf("%d",&a[i][j]);
    for(i=0;i<n;i++)
            minimum[i][0]=a[i][0];
        for(i=1;i<n;i++)
        {
            j=0;
            minimum[j][i]=a[j][i]+min2(minimum[j][i-1],minimum[j+1][i-1]+a[j+1][i]);
            for(j=1;j<n;j++)
                minimum[j][i]=a[j][i]+min2(minimum[j][i-1],minimum[j-1][i]);

            for(j=n-2;j>=0;j--)
                minimum[j][i]=min2(minimum[j+1][i]+a[j][i],minimum[j][i]);
        }
       int min=999999;
        for(i=0;i<n;i++)
            if(min>minimum[i][n-1])
                min=minimum[i][n-1];
        printf("%d\n",min);
    
   return 0;
}
