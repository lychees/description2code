#include <iostream>
#include <cstring>

using namespace std;

int main()
{
	int t;
	cin >> t;
	while(t--) {
		char s[1005];
		cin >> s;
		int len = strlen(s);
		int m[256] = {0};
		for(int i = 0; i < len; i++)
			m[s[i]]++;
			
		int k = m['a'];
		bool flag = true;
		for(int i = 'a'; i <= 'z'; i++)
			if(m[i] != k)
				flag = false;
		if( flag == true)
			cout << "Yes" << endl;
		else cout << "No" << endl;
	}
    return 0;
}
