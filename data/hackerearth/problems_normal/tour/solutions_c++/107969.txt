#include <stdio.h>
#include <string.h>
int main()
{
	int n;
	scanf("%d",&n);
	char cities[n][11];
	for(int i=0;i<n;i++)
	{
		scanf("%s",&cities[i]);
	}
	long long cost[n][n];
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
		{
				scanf("%lld",&cost[i][j]);
		}
	}
	int visit;
	scanf("%d",&visit);
	int in=0;
	long long exp=0;
	for(int i=0;i<visit;i++)
	{
		char dest[11];
		scanf("%s",dest);
		int j=0;
		for(;j<n;j++)
		{
			if(strcmp(dest,cities[j])==0)
			{
				break;
			}
		}
		exp+=cost[in][j];
		in=j;
	}
	printf("%lld\n",exp);
	return 0;
}
