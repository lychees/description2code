#include <iostream>
using namespace std;
#include<map>
#include<queue>
#define ll long long
int main()
{
    ll t,n,i,j,mx=0,c=0;
    cin>>t;
    string s;
    map<string , int> m;
    
    while(t--){
    	cin>>s;
    	m[s]++;
    	if(m[s]>mx){
    		mx=m[s];
    		c=1;
    	}
    	else if(m[s]==mx){
    		c++;
    	}
    }
    cout<<c*mx<<endl;
    return 0;
}
