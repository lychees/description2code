#include<iostream>
#include<string.h>
using namespace std;

int main()
{
    char a[10000];
    int t,i,n,c1,c2,c3,d1,d2,d3,countt;
cin>>t;

while(t--)
{countt=0;
    cin>>a;
    n=strlen(a);

    for(i=0;i<n;i++)
    {
        c1=0;c2=0;c3=0;d1=0;d2=0;d3=0;

    if(a[i]=='-')
    {
        if(i+1<n)
        {c1=1;

            if(  a[i+1]!='B' && a[i+1]!='S')
            d1=1;
        }

        if(i+2<n)
         {c2=1;
             if(a[i+2]!='B')
                d2=1;
         }

         if(i-1>-1)
          {
            c3=1;
            if(a[i-1]!='S')
            d3=1;
          }
    if(c1==d1 && c2==d2 && c3==d3)
        countt++;

    }

    }

    cout<<countt<<endl;

}

}
