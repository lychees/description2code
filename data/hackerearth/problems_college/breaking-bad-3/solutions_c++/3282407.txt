#include <bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	for(int i=0;i<t;i++){
	int n;
	cin>>n;
	int arr[n];
	for(int j=0;j<n;j++){
		cin>>arr[j];
	}
	sort(arr,arr+n);
	unsigned long long int ans=0;
	for(int j=0;j<n-1;j+=2){
		long long int yo=arr[j];
		yo*=arr[j+1];
		ans+=yo;
	}
	cout<<ans<<endl;
	}

}
