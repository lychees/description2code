#include <iostream>
#include <algorithm>
#include <cstdio>
#include <cstdlib>

using namespace std;

bool func(int a, int b) //comparator for sorting in decreasing order (Can also use greater <int> () comparator provided by standard library <numeric>)
{
	return (a > b);
}

int main()
{
    int arr[25001]; // Array to store the heights of candles
    int t; // number of test cases
    scanf("%d", &t);
    
   
	while (t--) {
        int n; // Number of candles
        scanf("%d", &n); // input n
        for(int i = 0; i < n; i++)
            scanf("%d", &arr[i]); // input the heights of n candles

        int c = 0; // counter variable to store maximum number of days candles can burn
        int flag=1; // flag for exit condition which is if any of the array value becomes negative


        while(c < n) {
            sort(arr, arr + n, func);    // stl sorting algorithm being used to sort in descending order
            for(int i = 0; i <= c; i++) {  // run a loop from 0 to current number of days
                arr[i]--;                   // Decreasing height of c(current number of days) candles with maximum heights
                if(arr[i]<0) {              // if maximum element is negative then maximum number of days = current number of days
                    flag = 0;
                    break;
                }
            }
            if(flag == 0)
                break;
            c++;
        }
        printf("%d\n", c);          // c contains the maximum number of days the candles can burn
	}
    return 0;
}
