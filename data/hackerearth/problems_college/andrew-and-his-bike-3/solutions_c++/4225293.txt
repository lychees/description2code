vector<ll>v[1000001];
ll size[1000001],store[1000001];
void pre(ll cur,ll par)
{
	size[cur]=1;
	for(ll i=0;i<v[cur].size();i++)
	{
		ll x=v[cur][i];
		if(x==par)
		continue;
		pre(x,cur);
		size[cur]+=size[x];
	}
}	
void dfs(ll cur,ll par,ll ans)
{
	ll n_ans=ans;
	if(par!=-1)
	 n_ans=ans*(size[par]-size[cur]);
	store[cur]=n_ans*size[cur];
	for(ll i=0;i<v[cur].size();i++)
	{
		ll x=v[cur][i];
		if(x==par)
		continue;
		dfs(x,cur,n_ans);
	}
}
