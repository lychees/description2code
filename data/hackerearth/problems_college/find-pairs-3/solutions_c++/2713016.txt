#include <stdio.h>
#include<algorithm>
using namespace std;
int a[1000000];
int main()
{
   // printf("Hello World!\n");
    int t;
    scanf("%d",&t);
    while(t--){
    	int n;
    	scanf("%d",&n);
    	for(int i=0;i<n;i++){
    		scanf("%d",&a[i]);
    	}
    	sort(a,a+n);
    	long long cnt=1;
    	long long ans=0;
    	for(int i=1;i<n;i++){
    		if(a[i]==a[i-1]){
    			cnt++;
    		}
    		else{
    			ans+=cnt+cnt*(cnt-1)/2;
    			cnt=1;
    		}
    	}
    	ans+=cnt+cnt*(cnt-1)/2;
    	printf("%lld\n",ans);
    	
    }
    return 0;
}
