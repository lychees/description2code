numOfInput = int(raw_input())
mod = 10**6 + 3

cache = {}
cache[0] = 1
for x in range(1, mod + 1):
    cache[x] = (cache[x-1] * x) % mod


def find_par_in_cham(cham, par_val):
    po = par_val % mod
    if (cham > mod):
        cham = mod
    fac = cache[cham] % mod
    return (po * fac) % mod


for x in range(numOfInput):
    vallist = map(int, raw_input().split(" "))
    print find_par_in_cham(vallist[0], vallist[1])
