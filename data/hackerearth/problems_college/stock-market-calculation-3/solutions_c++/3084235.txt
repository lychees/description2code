#include<bits/stdc++.h>
using namespace std;
int main()
{
	long int T,i,j,k,a[100001],N;
	cin>>T;
	while(T--)
	{
		cin>>N;
		for(i=0;i<N;i++)
		cin>>a[i];
		
		long int M[N];
		M[N-1]=a[N-1];
		for(i=N-2;i>=0;i--)
		M[i]=max(a[i],M[i+1]);
		long int ans=0;
		for(i=0;i<N;i++)
		{
			ans+=M[i]-a[i];
		}
		cout<<ans<<"\n";
	}
}
