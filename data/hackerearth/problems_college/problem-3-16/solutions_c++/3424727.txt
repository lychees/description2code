#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main()
{
	ll t, n, k, i;
    cin>>t;
    while(t--)
    {
    	cin>>n>>k;
    	ll a[n];
    	for(i=0;i<n;i++)
    		cin>>a[i];
    	ll c[k+1];
    	memset(c,0,sizeof(c));
    	for(i=0;i<n;i++)
    		c[a[i]]++;
    	sort(c,c+k+1);
    	cout<<n-c[k]<<endl;
    }
    return 0;
}
