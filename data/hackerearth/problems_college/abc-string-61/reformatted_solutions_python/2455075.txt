def powmod(a, n):
    ans = 1
    while n:
        if n % 2:
            ans = (ans*a) % 1000000007
        a = (a*a) % 1000000007
        n = n/2
    return ans


t = int(raw_input())

while t:
    t -= 1
    n = int(raw_input())
    if n < 3:
        print 0
    else:
        print(powmod(3, n)-(3*powmod(2, n))+3) % 1000000007
