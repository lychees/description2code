#include<bits/stdc++.h>
using namespace std;
# define ll long long
# define dec(i) i--
# define s string
int main()
{
    ll t,i,t1,t2,n,len,carry;
    char c1,c2;
    s str,ptr;
    cin>>t;
    while(t--)
    {
        cin>>str>>ptr;
        carry=n=0;
        for(i=str.size()-1;i>=0;dec(i))
        {
            c1=str[i];
            c2=ptr[i];
            t1=c1-48;
            t2=c2-48;
            carry=(t1+t2+carry)/10;
            if(carry==1)
                n++;
        }
        if(n>1)
       cout<<n<<" carry operations"<<endl;
       else if(n==0)
        cout<<"No carry operation"<<endl;
       else if(n==1)
        cout<<"1 carry operation"<<endl;
    }
    return 0;
}
