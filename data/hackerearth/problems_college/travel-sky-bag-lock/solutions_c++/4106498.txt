#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
    int i,j,t,count,digit,n;
    cin>>t;
    for(i=0;i<t;i++)
    {
    	cin>>n;
    	count=0;
    	for(j=0;j<n;j++)
    	{
    		scanf("%1d",&digit);
    		if( digit!=0 && digit!=2 && digit!=3 && digit!=5 && digit!=7)
    			count++;
    		if(digit==9)
    		    count++;
    	}
    	cout<<count<<endl;
    }
   
    return 0;
}
