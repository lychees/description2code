#include <iostream>
using namespace std;

int main()
{
	int n,t,i,j,a[100006],ans,l;
	
    cin>>n>>t;
    for(i=0;i<n;i++)
     cin>>a[i];
     
    
     while(t--)
     {
     	cin>>i>>j;
     	 ans=a[i];
     	 for(l=i+1;l<=j;l++)
     	  if(ans>a[l])
     	   ans=a[l];
     	cout<<ans<<endl;
     }
    return 0;
}
