template <class T,class U>inline void Max(T &a,U b){if(a<b)a=b;}
template <class T,class U>inline void Min(T &a,U b){if(a>b)a=b;}

inline void add(int &a,int b){a+=b;if(a>=mod)a-=mod;}

int h[1005],g[1005];
int main(){
    //ios_base::sync_with_stdio(0);
    //cout<<fixed<<setprecision(18);
    int T,i,j,k,m,n;
    scanf("%d",&n);
    int ans=0;
    for(i=0;i<n;i++)scanf("%d",&h[i]);
    for(i=0;i<n;i++){
        memset(g,0,sizeof(g));
        set<pii>s;
        for(j=i;j<n;j++){
            if(g[h[j]])s.erase(mp(g[h[j]],h[j]));
            g[h[j]]++;
            s.insert(mp(g[h[j]],h[j]));
            if(s.size()==1)Max(ans,s.begin()->x);
            else{
                set<pii>::iterator it=s.end();it--;
                int cnt=4;vector<pii>v;
                while(it!=s.begin()&&cnt>0){
                    v.pb(*it);
                    it--;cnt--;
                }
                if(cnt>0)v.pb(*it);
                for(int x=0;x<v.size();x++)
                    for(int y=x+1;y<v.size();y++){
                        k=v[x].x+v[y].x;
                        int y1=v[x].y,y2=v[y].y;
                        if(y1>y2)swap(y1,y2);
                        if(h[i]>y1&&h[i]<y2)k++;
                        if(h[j]>y1&&h[j]<y2)k++;
                        Max(ans,k);
                    }
            }
        }
    }
    printf("%d\n",ans);
    return 0;
}
