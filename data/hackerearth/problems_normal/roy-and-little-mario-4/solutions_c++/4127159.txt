#include <bits/stdc++.h>
using namespace std;
long long int mod = 1e9 +7;

long long int dp[100005];
long long int foo(long long int sum) {
	if(dp[sum] != -1)
		return dp[sum];
		
	if(sum == 0) 
		return 1;
		
	if(sum < 0) {
		return 0;
	}
	
	return dp[sum] = (foo(sum - 1) + foo(sum - 2) + foo(sum - 3)) % mod;
}

int main()
{
	memset(dp, -1, sizeof(dp));
	int t;
	cin >> t;
	
	for(int i = 0; i  < t; i++) {
		int ff;
		cin >> ff;
		cout << foo(ff) << endl;
	}
	    return 0;
}
