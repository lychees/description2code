#include<stdio.h>
#include<string.h>
#include<bits/stdc++.h>
using namespace std;
// function to check that substring (i...len) is palindrome or not.
int check(char *str, int i, int len)
{
    int m,n;
    for(m=i, n=len-1 ; m <= n; m++, n--)
    if(str[m]!=str[n])
    return 0;

    return 1;

}

int main()
{
	int cases;
	cin>>cases;
	while(cases--){
    char str[50];
    scanf("%s", str);
    int flag = 0;
    int len = strlen(str),i;

    for(i=0; i<len; i++)
    if(str[i]==str[len-1])// checking that ith and last character are same or not
    if(check(str,i,len)) // if ith and last character are same then check that substring i..len is palindrome or not.
    {
        flag = 1;
        break;
    }

    if(flag == 1)
    printf("%d", len+i);
    else
    printf("%d", len+len-1);
    cout<<endl;
	}
    return 0;
}
