def checkEnlightment(monkRange, monkNum):
    monkNum -= 1
    last = templeList[0] + monkRange
    for i in xrange(1, n):
        if last + monkRange >= templeList[i]:
            continue
        if monkNum == 0:
            return False
        last = templeList[i] + monkRange
        monkNum -= 1
    return True
 
n, k = map(int, raw_input().split())
templeList = map(int, raw_input().split())
templeList.sort()
 
l, h = 0, templeList[-1]
 
while h >= l:
    m = (l + h) / 2
    a = checkEnlightment(m, k)
    b = checkEnlightment(m-1, k)
    
    #print l, h, m, a, b
    if a and not b:
        break
    
    if a:
        h = m - 1
    else:
        l = m + 1
