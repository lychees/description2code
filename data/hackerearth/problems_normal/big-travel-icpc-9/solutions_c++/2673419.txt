#include<stdio.h>
#include<iostream>
#include<stdlib.h>
#include<vector>
#include<memory.h>
#include<algorithm>
using namespace std;
#define ll long long
#define MAX 230000
#define MOD 1000000007
int main()
{
    ll n,i,j,a,b,ans=0,ar1[MAX],ar2[MAX],temp[MAX],cur;
    //vector<pair<ll,ll> >v;
    memset(temp,0,sizeof(temp));
    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>ar1[i]>>ar2[i];
        //v.push_back(make_pair(a,b));
    }
    sort(ar1,ar1+n);
    sort(ar2,ar2+n);
    for(i=1;i<n;i++)
    {
        temp[i]=temp[i-1]+ar1[i];
    }
    cur=n-1;
    for(i=0;i<n-1;i++)
    {
        ans=ans+(temp[n-1]-temp[i]-cur*ar1[i]);
        cur--;
        ans=ans%MOD;
    }
    memset(temp,0,sizeof(temp));
     for(i=1;i<n;i++)
    {
        temp[i]=temp[i-1]+ar2[i];
    }
    cur=n-1;
    for(i=0;i<n-1;i++)
    {
        ans=ans+(temp[n-1]-temp[i]-cur*ar2[i]);
        cur--;
        ans=ans%MOD;
    }
    cout<<ans%MOD<<endl;
    return 0;
}
