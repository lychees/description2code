n = int(raw_input())
def check(s):
	t = s[0]
	for i in range(len(s)):
		if(s[i] != t):
			return 1
	return 0
while n>=1:
	s = raw_input()
	if len(s)%3 == 0:
		if s[0]!=s[len(s)/3] and s[0]!=s[2*len(s)/3] and s[len(s)/3]!=s[2*len(s)/3] :
			x = len(s)/3
			a = check(s[:x])
			b = check(s[x:2*x])
			c = check(s[2*x:len(s)])
			if a==0 and b==0 and c==0:
				print "OK"
			else:
				print "Not OK"
		else:
			print "Not OK"
	else:
		print "Not OK"
	n=n-1
