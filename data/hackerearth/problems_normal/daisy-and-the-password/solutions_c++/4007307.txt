#include<bits/stdc++.h>
using namespace std;

int main(){

    int T;

    scanf("%d",&T);


    for (int i=0;i<T;i++){
        string S,P;

        //scanf("%s %s",&S, &P);
        cin>>S>>P;
        //cout<<"Here ....."<<endl;
        bool f=false;
        //cout<<" dsfvbd :: "<<S<<"  "<<P<<endl;
        for(int j=0;j<S.size(); j++){
            string comb = S.substr(0,j)+S+S.substr(j);

            if (comb==P){
                cout<<"Possible"<<endl;
                f=true;
            }
        }
        if(f==false)
            cout<<"Impossible"<<endl;
    }


    return 0;
}
