#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    char a[200];
    scanf("%s",a);
    int length = strlen(a);
    int answer = length;
    int temp = 1;
    for (int i = 1; i < length; i++) {
        //printf("i = %d  a[i] = %c  a[i-1] = %c  temp = %d  answer = %d\n",i,a[i],a[i-1],temp,answer);
        if (a[i] == a[i-1]) {
            temp++;
        } else {
            if (temp != 0 && temp != 1) {
                if (temp != 2)
                    answer += temp*(temp - 1)/2;
                else
                    answer += 1;
                temp = 1;
            }
        }
        //printf("After :\ni = %d  a[i] = %c  a[i-1] = %c  temp = %d  answer = %d\n",i,a[i],a[i-1],temp,answer);
    }
    if (temp > 1) {
        if (temp != 2)
            answer += temp*(temp - 1)/2;
        else
            answer += 1;
    }
    cout<<answer<<endl;
    return 0;
}
