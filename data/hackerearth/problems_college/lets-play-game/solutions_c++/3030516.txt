#include <iostream>
using namespace std;
typedef long long int ll;
int main()
{
    ll t;
    cin>>t;
    while(t--)
    {
    	ll i,n,k;
    	cin>>n>>k;
    	ll a[n],b[n],r,c,profit=0;
    	for(i=0;i<n;i++)cin>>a[i];
    	for(i=0;i<n;i++)cin>>b[i];
    	r=k;
    	for(i=0;i<n;i++)
    	{   c=0;r=k;
    		while(a[i]<=r)
    		{r=r-a[i];
    		c++;
    		}
    		if(profit<c*b[i])
    		profit=c*b[i];
    	}
    	cout<<profit<<endl;
    }
    return 0;
}
