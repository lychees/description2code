
num = int(raw_input())
i = 0
passwords = []

while i < num:
	passwords.append(raw_input())
	i += 1

i = 0

while i < num:
	j = i + 1

	while j < num:
		if passwords[i] == passwords[j][::-1]:
			print "%d %s" % (len(passwords[i]), passwords[i][len(passwords[i])//2])
		j += 1
	i += 1
