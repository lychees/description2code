#include<bits/stdc++.h>
using namespace std;

int main()
{
   long n,k,x[100001],y[100001],i=0,j=0,min1,min2,mini=0,ans=0;
   cin>>n;
   cin>>k;
   long k1=k;
   long k2=k;
   while(k-->0)
   cin>>x[i++];
   while(k1-->0)
   cin>>y[j++];
   long c1,c2,c3,c4;
   i=0;
   while(k2-->0)
   {
   	c1=(x[i]-1)+(y[i]-1);
   	c2=(x[i]-1)+(n-y[i]);
   	c3=(n-x[i])+(y[i]-1);
   	c4=(n-x[i])+(n-y[i]);
   	min1=min(c1,c2);
    min2=min(c3,c4);
    mini=min(min1,min2);
    ans = ans + mini;
   	i++;
   }
   cout<<ans<<endl;
   return 0;
}
