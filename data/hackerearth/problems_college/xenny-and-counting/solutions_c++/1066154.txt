#include <iostream>
using namespace std;
typedef long long ll;

ll nck( ll n, ll k ) {
    if (k > n) return 0;
    if (k * 2 > n) k = n-k;
    if (k == 0) return 1;

    ll result = n;
    for( ll i = 2; i <= k; ++i ) {
        result *= (n-i+1);
        result /= i;
    }
    return result;
}

int main () {
	ll t, n, k, res;
	cin >> t;
	while (t--) {
		cin >> n >> k;
		//if (k>n-k) k=n-k;
		res = nck(n+k-1, k); 
		cout << res << "\n";
	}
	//cout << nck(5,2);
	return 0;
}
