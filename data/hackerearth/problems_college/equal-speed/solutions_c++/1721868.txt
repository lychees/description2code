#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
    int a,b,i,j;
    cin>>a>>b;
    int p[a];
    for(i=0;i<a;i++)
    {
        cin>>p[i];
    }
    for(i=0;i<a;i++)
    {
        for(j=0;j<a;j++)
        {
            if(p[i]>p[j])
            {
                int tmp;
                tmp=p[i];
                p[i]=p[j];
                p[j]=tmp;
            }
        }
    }
    cout<<p[b-1];
    return 0;

}
