#include <bits/stdc++.h>
using namespace std;
int check(long int *a,long int t,int m)
{
	long int x=a[0]%t;
	int i;
	for(i=1;i<m;i++)
	{
		if(a[i]%t!=x)
			return 0;
	}
	return 1;
}
int main()
{
    int m,v=0,j=0;
    cin >> m;
    long int a[m],d,k,i,t,t1,b[100005];
    for(i=0;i<m;i++)
    	cin >> a[i];
    sort(a,a+m);
    d=a[1]-a[0];
    for(i=1;i*i<=d;i++)
    {
      if(d%i==0)
        {
          t=i,t1=d/i;
        if(check(a,t,m) && t!=1)
			b[j++]=t;
		if(check(a,t1,m) && t1!=t)
			b[j++]=t1;
        }
    }
    sort(b,b+j);
    for(i=0;i<j;i++)
       cout << b[i] << " ";
    return 0;
}
