tc = int(raw_input())
for i in range(tc):
    website = raw_input()
    in_jhools_browser = website.split('.')
    for vowel in ['a', 'e', 'i', 'o', 'u']:
        in_jhools_browser[1] = in_jhools_browser[1].replace(vowel, '')
    print str(len(in_jhools_browser[1])+4) + "/" + str(len(website))
