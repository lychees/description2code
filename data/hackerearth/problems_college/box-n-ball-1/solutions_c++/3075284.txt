#include<iostream>
#include<cstring>
#include<cstdio>
using namespace std;

int main()
{
    long long int T;
    double l,b,h,s;
    cin>>T;
    while(T--)
    {
        cin>>l>>b>>h;

        s=l;
        if(b<s)
            s=b;
        if(h<s)
            s=h;

        printf("%0.1f\n",s/2);
    }
    return 0;
}
