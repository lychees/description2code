#include <iostream>
using namespace std;

int main()
{
    long long int n,m,i,j,x,y;
    cin>>n>>m;
    long long int c[m];
    for(i=0;i<m;++i)
    cin>>c[i];
    long long int dp[n+1][m];
    for(i=0;i<m;++i)
    {
    	dp[0][i]=1;
    }
    for(i=1;i<=n;++i)
    {
    	for(j=0;j<m;++j)
    	{
    		if(i-c[j]>=0)
    		x = dp[i-c[j]][j];
    		else x = 0;
    		if(j>=1)
    		y = dp[i][j-1];
    		else y=0;
    		dp[i][j]=x+y;
    	}
    }
    
    cout<<dp[n][m-1]<<"\n";
   
    return 0;
}
