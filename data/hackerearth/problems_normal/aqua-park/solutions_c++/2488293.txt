#include <iostream>
#include <algorithm>
using namespace std;
typedef struct point
{
	int x=0;
	int y=0;
}p;
typedef struct plank
{
	p left;
	p right;
}obt;
obt arr[101];

bool criteria(const obt &a,const obt &b)
{
	if((a.left.y==b.left.y)&&(a.right.y==b.right.y))
	{
		if((a.left.x>=b.left.x)&&(a.right.x>=b.right.x))
		return true;
		else
		return false;
	}
	if((a.left.y>=b.left.y)&&(a.right.y>=b.right.y))
	return true;
	else
	return false;
}

int main()
{
    int n,x,i=0;
    //p p1,p2;
    cin>>x>>n;
    for(i=0;i<n;i++)
    {
       cin>>arr[i].left.x>>arr[i].left.y;
       cin>>arr[i].right.x>>arr[i].right.y; 
    }
    sort(arr,arr+n,criteria);
    for(i=0;i<n;i++)
    {
    	if((x>=arr[i].left.x)&&(x<=arr[i].right.x))
    	{
    		if(arr[i].left.y<arr[i].right.y)
    		x=arr[i].left.x;
    		else
    		x=arr[i].right.x;
    	}
    	
    }
    cout<<x<<endl;
    return 0;
}
