#include <iostream>
using namespace std;

int main()
{
    int N,Q;
    cin >>N>>Q;
    int arr[N];
    for (int i=0;i<N;i++){
    	cin >>arr[i];
    }
    int x,y,z;
    while(Q--){
    	cin>>x;
    	if(x==1)
    	{
    		cin>>y;
    		int j=0;
    		for (j=0;j<N;j++){
    			if(arr[j]>=y){
    				cout<<j+1<<endl;break;
    			}
    		}
    	if(j==N)
    	cout<<-1<<endl;
    	}
    	else{
    			cin>>y>>z;
    			arr[y-1]=z;
    		}
    }
}
