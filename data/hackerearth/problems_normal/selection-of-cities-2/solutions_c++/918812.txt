#include<iostream>
#include<cstdio>
#define c 1000000007
typedef long long ll;
ll modulo(ll a,ll b)
{
	ll x=1,y=a;
	while(b>0)
	{
		if(b%2==1)
		{
			x=(x*y)%c;
		}
		y=(y*y)%c;
		b/=2;
	}
	return x%c;
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		ll n;
		scanf("%lld",&n);
		ll ans=modulo(2,n);
		printf("%lld\n",ans-1);
	}
}
