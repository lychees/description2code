#include<iostream>
#include<algorithm>
#include<cstring>
#include<cmath>
#include<cstdio>
using namespace std;
int n,m,x,y,i,j,k;
int dp[1005][2],a[1005];
char b[1005][1005];

int main()
{
  while(~scanf("%d%d%d%d",&n,&m,&x,&y))
   {
       memset(a,0,sizeof(a));
       memset(b,0,sizeof(b));
     for(i=0;i<n;i++)
      {
          scanf("%s",b[i]);
          for(j=0;j<m;j++)
            if(b[i][j]=='#')
               a[j+1]++;
        }

     for(i=1;i<m;i++)
        a[i+1]+=a[i];
    for(i=0;i<=m;i++)
     dp[i][0]=dp[i][1]=m*n;
    dp[0][1]=dp[0][0]=1;
    for(i=0;i<m;i++)
     {
         for(j=x;j<=y&&i+j<=m;j++)
           {
               dp[i+j][0]=min(dp[i+j][0],dp[i][1]+(a[i+j]-a[i]));
               dp[i+j][1]=min(dp[i+j][1],dp[i][0]+j*n-(a[i+j]-a[i]));
               //cout<<dp[i+j][0]<<"***"<<dp[i+j][1]<<" ";
           }
        //cout<<endl;
     }
    cout<<min(dp[m][0],dp[m][1])-1<<endl;
   }
    return 0;
}
