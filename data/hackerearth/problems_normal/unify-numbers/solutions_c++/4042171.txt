#include <iostream>
#include <cmath>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	long long median;
    	long long a[n];
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	sort(a,a+n);
    	if(n%2)
    	median = a[n/2];
    	else
    	median = (a[n/2-1]+a[n/2])/2;
    	long long ans=0;
    	for(int i=0;i<n;i++)
    	{
    		ans += abs(median-a[i]); 
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
