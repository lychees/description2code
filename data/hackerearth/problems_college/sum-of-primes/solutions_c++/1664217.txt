#include <iostream>
#include <cmath>
using namespace std;

const int N = 1e6+1;
const int sq = 1e3+1;

int primeArray[N];

void prime(){
 	fill_n(primeArray,N,1);
 	
	primeArray[0] = 0;
	primeArray[1] = 0;
	
	for(int i=2;i<sq;i++)
	{
	 if(primeArray[i] == 0)
	  continue;
     for(int j=i*2;j<N;j+=i)
      if(primeArray[j] == 1)
       primeArray[j] = 0;
	}
}

bool _prime(long sum){
	 if(sum < 2 || (sum%2==0 and sum!=2))
	  return false;
	 else
	 {
	  for(int i=2;i<=sqrt(sum);i++)
	   if(sum%i==0)
	    return false;
	  
	  return true;
	 }
}
 
int main(){
	ios::sync_with_stdio(false);
	prime();
	int T=0;
	
	cin >> T;
	
	while(T--)
	{
	 int a=0;
	 int b=0;
	 long sum=0;
	 
	 cin >> a >> b;
	 
	 for(a;a<=b;a++)
	  if(primeArray[a])
	   sum+=a;
	 
	 cout << (_prime(sum) ? "YES\n" : "NO\n");
	}
	
	return 0;
}
