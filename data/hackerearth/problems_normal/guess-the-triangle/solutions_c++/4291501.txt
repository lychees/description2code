#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;
vector<pair<long double,long double> > vec;

int main()
{
    long double x1,x2,x3,rx1,rx2,rx3;
    long double y1,y2,y3,ry1,ry2,ry3;
    cin>>x1>>y1>>x2>>y2>>x3>>y3;
    rx1=x1+x2-x3;
    rx2=x1-x2+x3;
    rx3=x2+x3-x1;
    ry1=y1+y2-y3;
    ry2=y1-y2+y3;
    ry3=y2+y3-y1;
    vec.push_back(make_pair(rx1,ry1));
    vec.push_back(make_pair(rx2,ry2));
    vec.push_back(make_pair(rx3,ry3));
    sort(vec.begin(),vec.end());
    for(int i=0;i<3;i++)
    {
    	printf("%.4llf %.4llf\n",vec[i].first,vec[i].second);
    }
    return 0;
}
