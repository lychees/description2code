#include<bits/stdc++.h>
using namespace std;
#define MOD 1000000007
typedef long long int ll;
ll modpow(ll n, ll p){
  if(p==0)return 1ll;
  ll x=modpow(n,p/2);
  x*=x;x%=MOD;
  if(p&1)x=(x*n)%MOD;
  return x;
}

int main(){
  int nin;
  cin>>nin;
  while(nin--){
    int N,K;
    cin>>N>>K;
    int a[N];
    int n=N;
    for(int i=0;i<N;i++)scanf("%d",a+i);
    sort(a,a+N);

    //create the ncr array ncr[n]
    ll ncr[n+1];
    ll r=K-1;
    for(int i=0;i<r;i++){
      ncr[i]=0ll;
    }
    ncr[r]=1ll;
    for(int i=r+1;i<=N;i++){
      ncr[i]=(ncr[i-1]*i)%MOD;
      ncr[i]=(ncr[i]*modpow(i-r,MOD-2))%MOD;
    }

    //for(int i=0;i<N+1;i++)cerr<<ncr[i]<<' ';cerr<<endl;

    ll ans=0;
    for(int i=N-1;i>=0;i--){
      ans+=(a[i]*ncr[N-i-1])%MOD;
      ans%=MOD;
    }
    cout<<ans<<endl;
  }
  return 0;

}
