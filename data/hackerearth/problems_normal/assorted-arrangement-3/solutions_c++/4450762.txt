#include <iostream>
using namespace std;
int main()
{
	int n,flag=0,id=0,i,j,l,k;
	int m,max;
	cin>>n>>m;
	int *pm=new int[m];
	int *pn=new int[n];
	int *maxary=new int[n];
	for(i=0;i<n;i++)
	maxary[i]=0;
	for(i=0;i<m;i++)
	cin>>pm[i];
	for(i=0;i<n;i++)
	cin>>pn[i];
	int temp=0,index,flag2=0,flag3=0;
	for(i=0;i<n;i++)
	{
		for(k=1;k<=m && flag!=1;k++)
		{
			if(pn[i]==k)
			{
			    index=k-1;
				flag=1;
			
			}
		}
		flag=0;
		for(j=temp+1;flag2!=1;j++)
		{
			if(j%pm[index]==0)
			{
				for(l=index+1;l<m && flag3!=1;l++)
				{
					if(j%pm[l]==0)
					{
						flag3=1;
					}
				}
				if(flag3!=1)
				{
					maxary[id]=j;
					id++;
					temp=j;
					flag2=1;
				}
				flag3=0;
			}
		}
		flag2=0;
		
	}
	cout<<maxary[id-1];
	return 0;
}
