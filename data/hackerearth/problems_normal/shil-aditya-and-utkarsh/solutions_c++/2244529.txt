#include <iostream>
#include <bits/stdc++.h>

using namespace std;

long long int C(int n, int r) {
    if(r > n / 2) r = n - r;
    long long ans = 1;
    int i;
    for(i = 1; i <= r; i++) {
        ans *= n - r + i;
        ans /= i;
    }

    return ans;
}
int main() {
    int n;
    cin >> n;
    long long int a = 0;
    for(int i = 2; i < n; i++) {
        a += C(i, 2);
    }
    cout << 2 * a << endl;
}
