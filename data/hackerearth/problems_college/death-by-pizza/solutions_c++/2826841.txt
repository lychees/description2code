#include <iostream>
using namespace std;

int main()
{
    long long int n,q;
    cin>>n>>q;
    
    long long int a[n];
    long long int b[] = {16,9,26,26,1};
    for(int i=0; i<n; i++){
        cin>>a[i];
    }
    
    for(int j=0; j<q; j++){
        long long int fifa, x,y;
        cin>>fifa>>x>>y;
        
        if(fifa==1){
            
            long long int sum =0;
            for(int i=x-1; i<=y-1; i++){
                sum+=a[i];
            }
            
            long long int num = y - x + 1;
        
            for(int z=2; z<=n&&z*z<=sum; z++){
                if(sum%z){
                    sum /- z;
                    num /- z;
                }
            }
            cout<<sum<<"/"<<num<<endl;
        }
        else{
            for(int i=x-1; i<=y-1; i++){
                a[i] += b[(i-x+1)%5];
            }
        }
    }
    
    
    return 0;
}
