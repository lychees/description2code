#include <bits/stdc++.h>
using namespace std;
double F[3100],X[3100],Y[3100];
double dist(int i,int j)
{
	return sqrt((X[i]-X[j])*(X[i]-X[j])+(Y[i]-Y[j])*(Y[i]-Y[j]));
}
int n;

int main()
{
    cin>>n;
    for(int i=1;i<=n;i++)
    	cin>>X[i]>>Y[i]>>F[i];
    double dp[n+1];
    for(int i=1;i<=n;i++)
    dp[i]=-1e9;
    dp[1]=0;
    for(int i=1;i<=n;i++)
    {
    	dp[i]+=F[i];
    	for(int j=i+1;j<=n;j++)
    	dp[j]=max(dp[j],dp[i]-dist(i,j));
    }
    cout<<fixed<<setprecision(6)<<dp[n]<<"\n";
    return 0;
}
