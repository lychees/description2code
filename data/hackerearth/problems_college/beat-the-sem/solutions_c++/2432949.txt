#include <iostream>
using namespace std;

int main()
{
	int t;
	cin >>t;
	while(t--)
	{
	   int a,b,x=0,y=0;
	    cin >> a >> b;
	    x=(a+b)/2;
	     y=abs(a-x);
	    if(a >= b&&(x+y)==a&&abs(x-y)==b)
	     cout << x  << " "<< y <<endl;
	   else
	    cout << "impossible" <<endl;
	 }
    return 0;
}
