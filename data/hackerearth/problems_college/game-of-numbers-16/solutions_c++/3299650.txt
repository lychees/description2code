#include<iostream>
using namespace std;
int main()
{
    ios_base::sync_with_stdio(false);
    long int t,n,cnt=0;
    cin>>t;
    for(int h=0;h<t;h++)
    {cin>>n;
        for(int i=1;i<=n;i++)
        {
            for(int j=1;j<=n;j++)
            {
                if((i*j)%(i+j)==0 && i!=j)
                    cnt++;

            }
        }cout<<cnt/2<<endl;
        cnt=0;
    }
}
