#include<iostream>
using namespace std;
#include<string.h>
int main()
{
	char link[100];
	cin>>link;
	int index=0;
	int k=0;
	int len1= strlen(link);
	char document[1000];
	int f=0;
	while(1)
	{
		cin>>document[f++];
		if(document[f-4]=='e'&&document[f-3]=='x'&&document[f-2]=='i'&&document[f-1]=='t')
		break;
	}
    int len = strlen(document);
	int j=0;
	char main[1000];
	char temp[50];
	int m=0;
	for(int i=0;i<len;i++)
	{
		
	 	
		if(link[j]==document[i])
		{
		
		temp[m++]=link[j];
		j++;
	}
	else
	if(j<len1&&document[i]!=link[j])
	{
		m=0;j=0;
	}
	
	
		if(j==len1)
		{
			for(int t=0;t<len1;t++)
			main[index++]=temp[t];
			k=i+1;
			while(1)
			{
				if(document[k+1]=='>')
				break;
				main[index++]= document[k];
				k++;
			}
			main[index++]='*';
			j=0;
			m=0;
		}
		 
		
		
	}
	//int final= strlen(main);
	for(int i=0;i<index;i++)
	{
		if(main[i]!='*')
		cout<<main[i];
		else
		cout<<endl;
		
	}
	
	
}
