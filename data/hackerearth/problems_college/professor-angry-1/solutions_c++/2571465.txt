#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,k,val;
    	cin>>n>>k;
    	int count=0;
    	for(int i=0;i<n;i++)
    	{
    		cin>>val;
    		if(val<=0)
    		count++;
    	//	if(count>=k)
    	//	break;
    	}
    	if(count>=k)
    	cout<<"NO"<<endl;
    	else
    	cout<<"YES"<<endl;
    //	cout<<count<<endl;
    }
    return 0;
}
