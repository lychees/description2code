#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t,i,j,k,l,a,b,x,n;
	cin >> t;
	while(t--)
	{
		cin >> x >> n;
		k=0;
		if(x>n)
		{
			cout << x-n << endl;
			continue;
		}
		else
		{
			i=j=x;a=b=0;
			while(i!=n/2&&i>0)
			{
				i--;
				k++;
			}
			if(i==n/2&&i>0)
			{
				a=k+1;
			}
			k=0;
			while(j<n)
			{
				j=j*2;
				k++;
			}
			while(j!=n)
			{
				j--;
				k++;
			}
			if(j==n)
			{
				b=k+1;
			}
			if(a>b)
				cout << b << endl;
			else
				cout << a << endl;
		}
	}
    return 0;
}
