    #include <stdio.h>
    long long dp[1005][1005];
    char a[100005];
    int main()
    {
       int i,j;
       int t;
       scanf("%d",&t);
       while(t--)
       {
       	scanf("%s",&a);
       	int l;
       	for(l=0;a[l];l++);
       	for(i=0;i<l;i++)
       	for(j=0;j<l;j++)
       	dp[i][j] = 0;
       	for(i=0;i<l;i++)
       	dp[i][i] = 1;
       	int ll;
       	for(i=0;i<l-1;i++)
       	if(a[i]==a[i+1])dp[i][i+1] = 3;
       	else dp[i][i+1] = 2;
       	for(ll = 3;ll<=l;ll++)
       	{
       	for(i=0;i+ll-1 < l;i++)
       	{
       		j = i+ll-1;
       		//printf("of %d and %d\n",i,j);
       	
    	int fir = dp[i][j-1] - dp[i+1][j-1];
    	int sec = dp[i+1][j] - dp[i+1][j-1];
    	int thi = dp[i+1][j-1];
    	dp[i][j]= fir+sec + thi;
    	if(a[i]==a[j])dp[i][j]+=dp[i+1][j-1]+1;
       	}
       }
       printf("%lld\n",dp[0][l-1]);
       }
        return 0;
    }
