int getpos(char r){
	int i,j;
	for(i=0;i<10;i++){
		for(j=0;j<c[i].size();j++){
			if(c[i][j]==r)
			return i;
		}
	}
	
	
}
int press(char r,int p){
	int i,s=0;
	for( i=0;i<c[p].size();i++){
		if(c[p][i]==r)
		return i+1;
	}
	
}


int main(){
	int t,i,k;
string s	;
	scanf("%d",&t);
	while(t--){
		cin>>s;
		int n=s.size();
		assert(n>0&&n<=1001);
		int curr=1,count=0;
		for(i=0;i<n;i++){
			//assert(valid(s[i]));
			k=getpos(s[i]);
			if(curr!=k)
			count+=1;
			curr=k;
			count=count+press(s[i],curr);
			
			
		}
		printf("%d\n",count);
		
		
		
	}
	return 0;
}
