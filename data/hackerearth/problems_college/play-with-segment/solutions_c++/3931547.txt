#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	long int n,a[1000000];
	long int i,j,k;
	long int p,q;
	long int count,count1;
	scanf("%d",&t);
	while (t--){
		scanf("%ld",&n);
		for (i = 1; i <= n; i++) {
			scanf("%ld",&a[i]);
		}
		p = min(a[1],a[2]);
		q = max(a[1],a[2]);
		i = 1;
		j = 3;
		count = 2;
		count1 = 2;
		//cout<<p<<" "<<q<<endl;
		while (i <= n && j <= n) {
			//cout<<p<<" "<<q<<endl;
			if (a[j] >= q) {
				q = a[j];
			} else {
				p = a[j];
			}
			//cout<<p<<" "<<q<<endl;
			if (abs(p-q) <= 1) {
				//cout<<p<<" "<<q<<"\n";
				j++;
				count++;
			} else {
				i++;
				j = i+1;
				p = min(a[i],a[j]);
				q = max(a[i],a[j]);
				if (count > count1) {
					count1 = count;
				}
				count=1;
			}
		}
		printf("%ld\n",max(count,count1));

	}
	return 0;
}
