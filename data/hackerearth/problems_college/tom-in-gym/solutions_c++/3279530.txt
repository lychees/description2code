    #include<stdio.h>
     
    int A[131],m,n;
     
    int cal()
    {
        A[0]=1;
        m=1;
        int i,j,res=0;
        for(j=3;j<=((n<<1)-1);j+=2)
        {
                  for(i=0;i<m;i++)
                  {
                                  res+=A[i]*j;
                                  A[i]=res%10;
                                  res/=10;
                  }
                  while(res>0)
                  {
                              A[m++]=res%10;
                              res/=10;
                  }
        }
        for(i=m-1;i>=0;i--)
        printf("%d",A[i]);
        printf("\n");
    }
     
    int main()
    {
        int test;
        scanf("%d",&test);
        while(test--)
        {
                   scanf("%d",&n);
                   cal();
        }
        return 0;
    }
