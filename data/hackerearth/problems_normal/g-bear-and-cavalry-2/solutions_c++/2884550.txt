#include "bits/stdc++.h"

using namespace std;

int W[42], H[42];

bool test()
{
  int n;
  cin >> n;
  for(int i=0; i<n; i++)
    cin >> W[i];
  for(int i=0; i<n; i++)
    cin >> H[i];
  sort(W+1, W+n);
  sort(H, H+n);
  reverse(H, H+n);
  for(int i=1; i<n; i++)
    if(W[i]*H[i] >= W[0]*H[0])
      return false;
  return true;
}

int main()
{
  ios_base::sync_with_stdio(0);
  int t;
  cin >> t;
  while(t--)
    puts(test() ? "YES" : "NO");
}
