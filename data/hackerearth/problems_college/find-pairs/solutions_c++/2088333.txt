#include <iostream>
#include <algorithm>
#include <stdlib.h>

using namespace std;





int getInvCount(int *arr,int n)
{
  int inv_count = 0;
  int i, j;
 
  for(i = 0; i < n ; i++)
    for(j = i; j < n; j++)
      if(arr[i]== arr[j])
      {
         // cout<<arr[i]<<"  comes equal to "<<arr[j]<<endl;
        //cout<<"pair is : "<<i<<" "<<j<<endl;  
        inv_count++;
      }
  return inv_count;
}
