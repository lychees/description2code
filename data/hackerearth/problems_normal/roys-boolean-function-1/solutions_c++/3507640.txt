#include <bits/stdc++.h>
using namespace std;
set<int> primes;
#define MAX 100010
int mark[MAX];
void getPrimes(){
	for(int i=2 ;i<sqrt(MAX)+1; i++){
		if(mark[i]){
			primes.insert(i);
			for(int j = 2; j*i<MAX ; j++){
				mark[i*j] = 0;
			}
		}
	}
}


int main()
{
	memset(mark,1,sizeof(mark));
	getPrimes();
	int tt; cin>>tt;
	// for(set<int>::iterator it = primes.begin(); it!=primes.end(); it++) cout<<*it<<" ";
	for(int t=0 ;t<tt; t++){
		int n; cin>>n;
		int count = 0;
		for(int i=1; i<n; i++){
			if(__gcd(i,n) == 1){
				count++;
			}
		}
		if(primes.find(count)!= primes.end()){
			cout<<"TRUE"<<"\n";
		}else{
			cout<<"FALSE"<<"\n";
		}
	}
    return 0;
}
