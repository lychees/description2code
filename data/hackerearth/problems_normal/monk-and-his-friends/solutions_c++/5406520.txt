#include<iostream>
#include<cstdio>
#include<set>
using namespace std;

int main(){
    int T;
    scanf("%d", &T);
    while(T--){
    	set<long long> s;
    	set<long long>::iterator it;
    	int N, M;
    	scanf("%d%d", &N, &M);
    	long long x;
    	for(int i = 0 ; i < N ; i++ ){
    		scanf("%lld", &x);
    		s.insert(x);
    	}
    	for( int i = 0 ; i < M ; i++ ){
    		scanf("%lld", &x);
    		it = s.find(x);
    		if( it == s.end() ){
    			printf("NO\n");
    		}
    		else{
    			printf("YES\n");
    		}
    		s.insert(x);
    	}
    }
}
