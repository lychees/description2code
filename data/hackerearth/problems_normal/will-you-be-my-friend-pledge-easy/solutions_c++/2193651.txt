#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int a,n,num=0;
    cin>>a>>n;
    int x[n];
    for(int i=0;i<n;i++) cin>>x[i];
    for(int j=0;j<n;j++)
    {
    	int ret=0;
    	if(a==x[j]) num++;
    	else
    	{
    		int k=2;
    		while(k<=min(a,x[j]))
    		{
    			if(a%k==0&&x[j]%k==0)
    			{
    				ret=1;
    				break;
    			}
    			k++;
    		}
    		if(ret==1) num++;
    	}
    }
    cout<<num<<endl;
    return 0;
}
