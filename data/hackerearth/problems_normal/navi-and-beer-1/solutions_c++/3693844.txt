#include <iostream>

using namespace std;

#define ll long long


int gcd(int a, int b) {
	return b == 0 ? a : gcd(b, a % b);
}


int main() {

	ll T;
	cin>>T;

	for (ll i=0 ; i<T ; ++i) {

		ll N,K,M;
		cin>>N>>K>>M;

		ll sum = ( N*(N+1) ) / 2;
		sum *= sum;
		sum *= K;

		ll ans = sum - M;
		ans = ans < 0 ? 0 : ans;
		cout<<ans<<endl;
	}


}
