#include<iostream>
using namespace std;
long c[1001];
void cat()
{
    for (int i=0;i<=1000;i++)
	    if (i==0 || i==1)
		c[i]=1;
        else
		{
            long sum =0;long left, right;
            for (int k=1;k<=i;k++)
			{
                left = c[k-1] % 1000000007;
                right= c[i-k] % 1000000007;
                sum =(sum+ (left * right)%1000000007)%1000000007;
            }
            c[i]=sum;
        }
}
int main()
{
    int t;
    cin>>t;
    cat();
    while(t--)
    {
              long int n;
              cin>>n;
              cout<<c[n]<<"\n";
    }
              return 0;
}
