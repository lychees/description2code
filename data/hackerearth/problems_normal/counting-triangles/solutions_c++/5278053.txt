#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,i,ans=0;
    scanf("%d",&n);
    long long int a[n],b[3];
    for(i=0;i<n;i++)
    {
    	scanf("%lld %lld %lld",&b[0],&b[1],&b[2]);
    	sort(b,b+3);
    	a[i]=b[0]*100+b[1]*10+b[2];
    }
    sort(a,a+n);
    for(i=1;i<n-1;i++)
    {
    	if(a[i]!=a[i+1] && a[i]!=a[i-1])
    	ans++;
    }
    if(a[0]!=a[1])
    ans++;
    if(a[n-1]!=a[n-2])
    ans++;
    printf("%d\n",ans);
    return 0;
}
