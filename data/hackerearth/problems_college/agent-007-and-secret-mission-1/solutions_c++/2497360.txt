// bit_encoding.cpp : main project file.


#include <iostream>
#include <string>
using namespace std;

class Code
{
public:
	void enter_data();
	void get_len();
	Code();
	string str;
	int len;
	int days;
	int no_of_1;
	int no_of_0;
};

Code::Code()
{
	no_of_1=no_of_0=len=0;
}

void Code::enter_data()
{
	cin>>str;
	cin>>days;
}

void Code::get_len()
{
	len=str.size();
}

int decode(Code co)
{
	for(int i=0;i<co.len;i++)
	{
		if (co.str[i]=='0') co.no_of_0++;
		else co.no_of_1++;
	}
	int times=((co.days)*(co.days+1))/2; 
	if(times%2!=0) return co.no_of_0;
	else return co.no_of_1;
}

int main()
{
	int n,x;
	cin>>n;
	//int *i=new int[n];
	Code *c= new Code[n];
	for(int i=0;i<n;i++)
	{
		Code d;
		c[i]=d;
		c[i].enter_data();
		c[i].get_len();
	}
	for(int i=0;i<n;i++)
	{
		cout<<decode(c[i])<<endl;
	}
    
    return 0;
	
}
