

    #include<stdio.h>
    #define c 1000000007
    long long int f(long long int x,long long int n)
    {
         long long int u;
         if(n==0)
         {
                 return(1);
         }
         else if(n%2==0)
         {
              u=f(x,n/2)%c;
              return((u*u)%c);
         }
         else
         {
             u=f(x,n/2)%c;
             return((((x*u)%c)*u)%c);
         }
    }
    int main()
    {
        long long int count0=0,n,a[100010],prod=1,i,q,op,x,y;
        scanf("%lld",&n);
        for(i=1;i<=n;i++)
        {
            scanf("%lld",&a[i]);
            if(a[i]==0)
            {
                count0++;
            }
            else
            {
                prod=(prod*a[i])%c;
            }
        }
        scanf("%d",&q);
        while(q--)
        {
            scanf("%lld",&op);
            if(op==0)
            {
                scanf("%lld %lld",&x,&y);
                if(a[x]==y)
                {
                    continue;
                }
                else if(y==0)
                {
                    count0++;
                    prod=(prod*f(a[x],c-2))%c;
                }
                else if(a[x]==0)
                {
                    count0--;
                    prod=(prod*y)%c;
                }
                else
                {
                    prod=(prod*y)%c;
                    prod=(prod*f(a[x],c-2))%c;
                }
                a[x]=y;
            }
            else
            {
                scanf("%lld",&x);
                if(count0==0)
                {
                    printf("%lld\n",(prod*f(a[x],c-2))%c);
                }
                else if(count0==1)
                {
                    if(a[x]==0)
                    {
                        printf("%lld\n",prod);
                    }
                    else
                    {
                        printf("0\n");
                    }
                }
                else
                {
                    printf("0\n");
                }
            }
        }
        return(0);
    }
