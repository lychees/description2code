#include <iostream>
#include<algorithm>
#include<cstdio>
#include<cmath>
#include<vector>
using namespace std;

vector<vector<int> > vec(31);
bool arr[31];

int main()
{
    int t,n;
    cin>>t;
    for(int i=1;i<31;i++)
    {
    	for(int j=1;j<=i/2;j++)
    	{
    		if(i%j==0)vec[i].push_back(j);
    	}
    	vec[i].push_back(i);
    }
    while(t--)
    {
    	cin>>n;
    	for(int i=1;i<=n;i++)arr[i]=1;
    	int turn = 0, mx =0, num;
    	while(true)
    	{
    		bool possible = 1;
    		for(int i=1;i<31;i++)
    		{
    			if(mx < vec[i].size() && arr[i])
    			{
    				mx = vec[i].size();
    				num = i;
    				possible = 0;
    			}
    		}
    		if(possible)break;
    		arr[num] = 0;
    		for(int i=0;i<vec[num].size();i++)arr[vec[num][i]] = 0;
    		turn = (turn+1)%2;
    	}
    	cout<<turn<<endl;
    }
    return 0;
}
