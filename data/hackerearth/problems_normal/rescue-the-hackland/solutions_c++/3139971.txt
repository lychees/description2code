#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int main(){
int cases,i,k,total;
cin>>cases;
while(cases--){
total=0;
vector<int> arr(2,0);
for(i=0;i<2;i++){
	cin>>arr[i];
}	
cin>>k;
sort(arr.begin(),arr.end());
while(arr[0]<=arr[1]){
total=total+1;
arr[0]=arr[0]*k;
arr[0]=arr[0]+1;	
}
cout<<total<<endl;
}	
	
	
	
return 0;	
}
