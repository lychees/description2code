#include <iostream>
#include<bits/stdc++.h>
using namespace std;
 
int main()
{
   long long n;
   cin>>n;
   vector<long long> a(n);
   vector<long long> temp(n);
   long long i,j,k;
   for(i=0;i<n;i++)
   cin>>a[i];
   sort(a.begin(),a.end());
   i=0;
   j=n-1;
   k=0;
   while(k<n){
   	temp[k++]=a[i++];
   	if(k<n)
   	temp[k++]=a[j--];
   	
   	
   }
   
  
   long long result=0;
   for(i=1;i<n;i++){
   	result=result+abs(temp[i]-temp[i-1]);
   	
   }
   result=result+abs(temp[n-1]-temp[0]);
   
   cout<<result;
   
   
    return 0;
}
