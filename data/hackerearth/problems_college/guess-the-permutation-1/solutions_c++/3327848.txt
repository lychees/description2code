#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		int n,val,count;
		scanf("%d",&n);
		int l=0;
		for(int i=0;i<n;i++){
			int a[1001]={0};
			count=0;
			
			for(int j=0;j<n;j++){
				scanf("%d",&val);
				if(a[val]==0 ){
					a[val]=1;
					count++;	
				}
			}
			//printf("%d  ",count);
			if(count!=n)
				printf("%d ",count-1);
			else{
				if(count==n && l==0){
					printf("%d ",count-1);
					l++;
				}
				else
					printf("%d ",count);
					
			}
				
		}
		printf("\n");
	}
}
