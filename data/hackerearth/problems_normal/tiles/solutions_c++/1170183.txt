typedef int arr[64][64];
int m;arr a;
void dfs(int x,int st,int s)
{
    if(x==m){a[st][s]++;return;}
    dfs(x+1,st,s);
    if((st>>x&1)==0)
    {
        dfs(x+1,st,s|1<<x);
    }
    if(x<m-1)dfs(x+2,st,s|1<<x|1<<x+1);
}
void mul(arr &a,arr b)
{
    arr c;memset(c,0,sizeof(c));
    for(int i=0;i<1<<m;i++)
        for(int j=0;j<1<<m;j++)
            for(int k=0;k<1<<m;k++)
                c[i][j]=(c[i][j]+1LL*a[i][k]*b[k][j]%mod)%mod;
    memcpy(a,c,sizeof(a));
}
void pow(arr &a,int n)
{
    arr b;memset(b,0,sizeof(b));
    for(int i=0;i<1<<m;i++)b[i][i]=1;
    while(n)
    {
        if(n&1)mul(b,a);
        mul(a,a);n>>=1;
    }
    memcpy(a,b,sizeof(a));
}
int main() 
{ 
    int T,i,j,k,n; 
    scanf("%d",&T);
    while(T--)
    {
        scanf("%d%d",&m,&n);
        memset(a,0,sizeof(a));
        for(i=0;i<1<<m;i++)
        {
            dfs(0,i,0);
            a[i][0]=1;
        }
        pow(a,n);
        int ans=0;
        for(i=0;i<1<<m;i++)ans=(ans+a[(1<<m)-1][i])%mod;
        printf("%d\n",ans);
    }
}
