#include <bits/stdc++.h>
using namespace std;
int freq1[26],freq2[26];
int main()
{
	int n;
	string p,s;
	cin >> n;
	
	for(int i=1;i<=n;i++)
	{

		for(int k=0;k<26;k++)
			freq1[k]=freq2[k]=0;


		cin >> p >> s;
		int len1 = p.length(),len2 = s.length();
		if(len2 < len1)
			cout<< "NO\n";
		else
		{

			for(int j=0;j<len1;j++)
			{
				freq1[p[j]-'a']++;
				freq2[s[j]-'a']++;
			}
			
			int prev = 0,counter=0,flag=0;
			for(int j=len1;j<len2;j++)
			{
				for(int k=0;k<26;k++)
				{
					if(freq1[k] == freq2[k])
						++counter;
				}
				if(counter == 26)
				{
					cout << "YES\n";
					counter=0;
					flag=1;
					break;
				}
				else
				{
					counter=0;
					freq2[s[prev]-'a']--;
					freq2[s[j]-'a']++;
					prev++;
				}
			}
			if(flag==0)
			{
				for(int k=0;k<26;k++)
				{
					if(freq1[k] == freq2[k])
						++counter;
				}
				if(counter == 26)
				{
					cout << "YES\n";
					counter=0;
				}
				else
					cout << "NO\n";
			}
		}
	}
	return 0;
}
