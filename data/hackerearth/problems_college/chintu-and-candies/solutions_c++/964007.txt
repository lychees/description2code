#include <iostream>
#include <stdio.h>
#include <vector>

using namespace std;

typedef long long int Lint;

Lint phi(Lint n) 
     { 
       Lint res = n; 
       for(int i=2;i*i <= n;i++) 
       { 
         if (n % i == 0) res -= res / i; 
         while (n % i == 0) n /= i; 
       } 
       if (n > 1) res -= res / n; 
       return res; 
     }
     
int main() {
	
	Lint t, n, x, k, flag, i;
	
	vector<Lint> A( 10001, 0 );
	
	for( i = 1;i < 10001;i++ ) 
		A[ i ] = phi(i);
		
	
	scanf("%lld", &t );
	
	while( t-- ) {
		
		scanf("%lld %lld %lld", &n, &x, &k );
		
		flag = 0;
		for( i = n;i > x;i-- )
			if( A[ i ] == k ) {
				flag = 1;
				printf("%lld\n", i );
				break;
			}
		
		if( !flag )
			printf("-1\n");
	}
	return 0;
}
