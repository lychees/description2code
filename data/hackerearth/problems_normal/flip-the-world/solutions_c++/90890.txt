int n,m;
char g[30][30];
int G[30][30];
void flip(int x, int y)
{
    FOR(i,x+1)
    {
        FOR(j,y+1)
        {
            G[i][j] = 1 - G[i][j];
        }
    }
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d %d",&n,&m);
        FOR(i,n)
        scanf("%s",g[i]);
        FOR(i,n)
        {
            FOR(j,m)
            {
                G[i][j] = g[i][j] - '0';
            }
        }
        int ans = 0;
        for(int i=n-1;i>=0;i--)
        {
            for(int j=m-1;j>=0;j--)
            {
                if(!G[i][j])
                {
                    ans++;
                    flip(i,j);
                }
            }
        }
        printf("%d\n",ans);
    }
return 0;
}
