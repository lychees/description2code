#include <bits/stdc++.h>
using namespace std;
typedef long int p;
int a[100000];

int main() {
	p n;
	int i,cnt=0,val;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
	}
	sort(a,a+n);
	cnt++;val=a[0];
	for(i=0;i<n;i++)	
	{
		if(a[i]>=val && a[i]<=val+4)
			continue;
		else {
			val=a[i];
			cnt++;
		}
	}
	cout<<cnt;
	
	
	
	return 0;
}
