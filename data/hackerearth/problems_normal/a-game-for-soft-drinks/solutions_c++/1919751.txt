#include <bits/stdc++.h>
using namespace std;
int dp[102][102][102][2];
int fun(int n1,int n2,int n3,int flag)
{
	int ans;
		if(n1==0 && n2==0 && n3==0 )
		{
		if(flag==0)
		return 1;
		if(flag==1)
		return 0;
		}
		if(dp[n1][n2][n3][flag]!=-1)
		return dp[n1][n2][n3][flag];
		if(flag==0)
		{
			ans=1;
			if(n1-1>=0)
			ans=fun(n1-1,n2,n3,1);
			if(n2-1>=0)
			ans=min(ans,fun(n1,n2-1,n3,1));
				if(n3-1>=0)
			ans=min(ans,fun(n1,n2,n3-1,1));
			if(n1-1>=0 && n2-1>=0)
			ans=min(ans,fun(n1-1,n2-1,n3,1));
			if(n1-1>=0 && n3-1>=0)
			ans=min(ans,fun(n1-1,n2,n3-1,1));
			if(n3-1>=0 && n2-1>=0)
			ans=min(ans,fun(n1,n2-1,n3-1,1));
			if(n1-1>=0 && n2-1>=0 && n3-1>=0)
			ans=min(ans,fun(n1-1,n2-1,n3-1,1));
		}
		else if(flag==1)
		{
			ans=0;
			if(n1-1>=0)
			ans=fun(n1-1,n2,n3,0);
			if(n2-1>=0)
			ans=max(ans,fun(n1,n2-1,n3,0));
				if(n3-1>=0)
			ans=max(ans,fun(n1,n2,n3-1,0));
			if(n1-1>=0 && n2-1>=0)
			ans=max(ans,fun(n1-1,n2-1,n3,0));
			if(n1-1>=0 && n3-1>=0)
			ans=max(ans,fun(n1-1,n2,n3-1,0));
			if(n3-1>=0 && n2-1>=0)
			ans=max(ans,fun(n1,n2-1,n3-1,0));
			if(n1-1>=0 && n2-1>=0 && n3-1>=0)
			ans=max(ans,fun(n1-1,n2-1,n3-1,0));
		}
		return dp[n1][n2][n3][flag]=ans;
}
int main() {
	int t;
	int n1,n2,n3,ans;
	scanf("%d",&t);
		memset(dp,-1,sizeof(dp));
	ans=fun(101,101,101,0);
	while(t--)
	{
		scanf("%d%d%d",&n1,&n2,&n3);
	
		ans=dp[n1][n2][n3][0];
		if(ans==0)
		printf("Ashima\n");
		else
		{
			//printf("ans=%d\n",ans);
		printf("Aishwarya\n");
		}
	}
	return 0;
}
