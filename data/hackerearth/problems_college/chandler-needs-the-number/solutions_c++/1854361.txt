#include <iostream>
using namespace std;
 
int main()
{
    long long int i,n,a,min,max;
    
    cin >> n;
    cin >> min;
    max=min;
    
    for(i=1 ; i<n ; i++)
    {
    	cin >> a;
    	if(a<min)
    		min=a;
    	else if(a>max)
    		max=a;
    }
    cout<< min*max <<endl;
    return 0;
}
