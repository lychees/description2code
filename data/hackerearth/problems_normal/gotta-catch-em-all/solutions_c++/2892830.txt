#include <iostream>
#include<bits/stdc++.h>

#define rep(i,a,n) for(int i=a;i<n;++i)

bool comp(int a,int b)
{
	return a > b;
}

using namespace std;

int main()
{
    int k;
    cin >> k;
    int A[k];
    
    rep(i,0,k)
    	cin >> A[i];
    	
    sort(A,A+k,comp);
    
    int ans=0;
    rep(i,0,k)
    {
    	ans = max(ans,A[i]+i+2);
    	//cout << ans << endl;
    }
    	
    cout << ans << endl;
    return 0;
}
