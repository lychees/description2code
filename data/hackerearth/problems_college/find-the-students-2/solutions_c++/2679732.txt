#include <iostream>
#include <algorithm>

using namespace std;

int main() {
	
	int n;
	cin>>n;
	int a[n];
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
	}
	sort(a,a+n);
	int c=0,t;
	for(int i=0;i<n;i++)
	{
		t=a[i];
		if(a[i]!=0)
		{
		
		c=0;
		
		for(int j=i;j<n;j++)
		{
		if(a[j]!=0)
		{
			a[j]=a[j]-t;
			c++;
			}
	
		}
		printf("%d\n",c);
	
		c=0;
		}
	}
	return 0;
}
