#include <iostream>
#include<stack>
using namespace std;

int main()
{
	int t;
	char type;
	cin>>t;
	int count=0;
	int num,pball,preball,temp;
	while(t--)
	{
		
		cin>>num>>pball;
		stack<int> s;
		preball=pball;
		s.push(pball);
		for(int i=0;i<num;i++)
		{
			cin>>type;
			if(type=='B'){
			count++;
			if(count==2)
				{
					//temp=s.top();
				//	cout<<preball<<" f";
					s.push(preball);
					count=0;
					//preball=temp;
				}
				else{
			preball=s.top();
			s.pop();
			}
			}
			else
			{
				cin>>pball;
				s.push(pball);
				count=0;
			}
		}
		cout<<"Player "<<s.top()<<endl;
	}
    return 0;
}
