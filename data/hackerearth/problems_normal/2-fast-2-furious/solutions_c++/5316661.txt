#include <iostream>
#define MAX 1000000
using namespace std;

int main()
{
    int N,A[MAX],i,flag=0,maxa=0,maxb=0,B[MAX];
    cin>>N;
    for(i=0;i<N;i++)
    {
     if(flag==0)
     {
      cin>>A[i];
      flag++;
     }
     else
     {
      cin>>A[i];
      maxa=maxa>abs(A[i]-A[i-1])?maxa:abs(A[i]-A[i-1]);
     }
    }
    flag=0;
    for(i=0;i<N;i++)
    {
     if(flag==0)
     {
      cin>>B[i];
      flag++;
     }
     else
     {
      cin>>B[i];
      maxb=maxb>abs(B[i]-B[i-1])?maxb:abs(B[i]-B[i-1]);
     }
    }
    if(maxa>maxb)
    {
     cout<<"Dom\n";
     cout<<maxa;
    }
    else if(maxa<maxb)
    {
     cout<<"Brian\n";
     cout<<maxb;
    }
    else
    {
     cout<<"Tie\n";
     cout<<maxa;
    }
    return 0;
}
