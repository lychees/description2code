#include<bits/stdc++.h>
#define z 1000000007
using namespace std;
static long long int dp1[1025][1025];
long long int dp(int sr,int sc,int er,int ec){
    for(int i=0;i<=er;i++)
    for(int j=0;j<=ec;j++)
        dp1[i][j]=0;
    dp1[sr][sc]=1;
    for(int i=sr;i<=er;i++)
      for(int j=sc;j<=ec;j++){
      	if(i==sr&&j==sc)
      	   continue;
      	dp1[i][j]=dp1[i-1][j]+dp1[i][j-1]+dp1[i-1][j-1];
      	dp1[i][j]%=z;
      }
      return dp1[er][ec];
}
int main(){
    int t;
    scanf("%d",&t);
    dp(1,1,1024,1024);
    while(t--){
    	int a,b,c,d;
    	scanf("%d %d %d %d",&a,&b,&c,&d);
    	c-=a-1;
    	d-=b-1;
    	printf("%lld\n",dp1[c][d]);
    }
    return 0;
}
