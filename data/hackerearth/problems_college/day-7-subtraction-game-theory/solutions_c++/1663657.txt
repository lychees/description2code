#include <iostream>
using namespace std;

#define mod 1000000007

int main()
{
	long long int ar[1000000];
    int x,y,q,n,i;
    ar[0]=1;
    scanf("%d%d%d",&x,&y,&q);
    for(i=1;i<=1000000;i++)
        ar[i]=((x*ar[i-1])-y)%mod;

    while(q--)
    {
        scanf("%d",&n);
        printf("%d\n",ar[n]);
    }
    return 0;
}
