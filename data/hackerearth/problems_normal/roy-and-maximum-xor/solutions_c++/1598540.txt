#include<iostream>
#include<cmath>
using namespace std;
typedef unsigned long long ll;
int main(){
   ll x,i,j,t,r,p,n,cn;
   scanf("%lld",&t);
   while(t--){
      scanf("%lld",&n);
      p=1;
      for(i=0;;i++){
          if(p>=n)
             break;
          p*=2;
      }        
      x=p-1;
      if(p>n){
        p/=2;
      }
      cn=n-p;
      if(cn==0)
        cn=p;
      else
        cn=2*cn;
      printf("%lld %lld\n",x,cn);              
   }
   return 0;
}
