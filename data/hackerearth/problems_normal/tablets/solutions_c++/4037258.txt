#include<bits/stdc++.h>
using namespace std;
#define ll long long int
ll dp[1000100],arr[1000100];
int main()
{
	ll n;
	scanf("%lld",&n);
	ll i;
	for(i=0; i<n; i++)
		scanf("%lld",&arr[i]);
	dp[0]=1;
	for(i=1; i<n; i++)
	{
		if(arr[i]>arr[i-1])
			dp[i]=dp[i-1]+1;
		else
			dp[i]=1;
	}
	for(i=n-2; i>=0; i--)
	{
		if(arr[i]>arr[i+1] && dp[i]<=dp[i+1])
		{
			dp[i]=dp[i+1]+1;
		}
	}
	ll sum=0;
	for(i=0; i<n; i++)
		sum+=dp[i];
	printf("%lld\n",sum);
	return 0;
	
}
