struct hacker{
    ll id,z,p,l,s,c,nz,change;
};

hacker arr[1000006];
int n;

struct myclass {
  bool operator() (struct hacker i,struct hacker j) {
      if(i.change != j.change) return i.change > j.change;
      else return (i.id > j.id);
  }
} myobject;

int main(){
    scanf("%d",&n);    
    rep(i,n){
        scanf("%lld %lld %lld %lld %lld %lld",&arr[i].id,&arr[i].z,&arr[i].p,&arr[i].l,&arr[i].c,&arr[i].s);
        arr[i].nz = (50*arr[i].p + 5*arr[i].l + 10*arr[i].c + 20*arr[i].s);
        arr[i].change = arr[i].nz - arr[i].z;
    }
    sort(arr,arr+n,myobject);
    rep(i,min(5,n)){
        printf("%lld %lld\n",arr[i].id, arr[i].nz);
    }
}
