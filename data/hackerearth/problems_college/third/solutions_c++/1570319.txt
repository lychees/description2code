#include <iostream>
using namespace std;
typedef long long int ll;

int main()
{
    ll t,n; cin>>t; ll arr[10005];
    while(t--)
    {
    cin>>n;
    for (ll i=0;i<n;i++)
       cin>>arr[i];
    ll inc=arr[0]; ll exc=0;
    for(ll i=1;i<n;i++)
    {
    	ll p=max(inc,exc);
    	inc=arr[i]+exc;
    	exc=p;
    }
    if (inc>=exc)
    cout<<inc<<endl;
    else
    cout<<exc<<endl;
    }
    return 0;
}
