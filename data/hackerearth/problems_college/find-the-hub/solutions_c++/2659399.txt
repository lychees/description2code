#include<iostream>
#include<algorithm>
#include<cmath>
bool prim(int num)
{
	for(int i=2;i<=sqrt(num);i++)
	{
		if(num%i==0)
		return false;
	}
	return true;
}
using namespace std;
int main()
{
	int n;
	cin>>n;
    int arr[n];
for(int i=0;i<n;i++)
    cin>>arr[i];
    sort(arr,arr+n);
    for(int i=n-1;i>=0;i--)
    {
    	int cnt=1;
    	while(arr[i-1]==arr[i]&&i-1>=0)
    	{
    		i--;cnt++;
		}
		if(prim(arr[i]))
		{
			cout<<arr[i]<<endl<<cnt;
			break;
		}
	}
	return 0;
}
