using namespace std;
ll zero(ll n)
{
	ll sum=0;
	while(n)
	{
		sum+=(n/5);
		n/=5;
	}
	return sum;
}
long long solve(long long n){
    if(n==0 || n==1) return 1;
    if(n <= 10) return (long long)log10(tgamma(n+1)) + 1;
    return (long long)((double)n*log10((double)n/M_E) + 0.5*log10(2*M_PI*n)) + 1;
}
int main()
{
	ll t,n;

	{
		scanf("%lld",&n);

	}
	return 0;
}
