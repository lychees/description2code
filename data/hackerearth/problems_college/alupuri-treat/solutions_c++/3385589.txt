#include<iostream>

using namespace std;

int main()
{
	int no;
	cin>>no;
	int arr[no][no],sum1=0,sum2=0,ans=0;
	for(int i=0;i<no;i++)
	{
		for(int j=0;j<no;j++)
			cin>>arr[i][j];
	}
	int i,j;
	for(i=0,j=0;i<no;i++,j++)
		sum1 = sum1+arr[i][j];
	for(i=0,j=no-1;i<no,j>=0;i++,j--)
		sum2 = sum2+arr[i][j];
	if(sum1>sum2)
		ans = sum1-sum2;
	else
		ans = sum2-sum1;
	cout<<ans<<endl;	
	return 0;
}
