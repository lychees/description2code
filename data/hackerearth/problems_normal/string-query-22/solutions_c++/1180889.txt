#include <iostream>
#include <algorithm>
#include <string.h>
#include <stdio.h>
#include <vector>
using namespace std;
int yes[26][200001]={};
vector <int> v;
vector <int> k[26];
int main() {
	string s;
	cin>>s;

	int q;
	scanf("%d",&q);
	int h;
	int size[26]={};
	for(int i=0;i<s.length();i++)
		size[s[i]-'a']++;
	char s1[2];
	for(int i=0;i<q;i++)
	{
		scanf("%d %s",&h,s1);
		k[s1[0]-'a'].push_back(h);
	}
	for(int i=0;i<26;i++)
	{
		v.clear();
		for(int i1=1;i1<=size[i];i1++)
			v.push_back(i1);
		for(int h=0;h<k[i].size();h++)
		{
			yes[i][v[k[i][h]-1]]=1;
			v.erase(v.begin()+k[i][h]-1);
		}
	}
	int pos[26]={};
	for(int i=0;i<s.length();i++)
	{
		if(!yes[s[i]-'a'][pos[s[i]-'a']+1])
			printf("%c",s[i]);
		pos[s[i]-'a']++;
	}
	return 0;
}
