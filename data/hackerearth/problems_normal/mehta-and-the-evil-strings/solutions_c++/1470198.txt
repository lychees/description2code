#include <bits/stdc++.h>

using namespace std;

namespace stuff {
typedef long long ll;
const int MOD = int(1e9) + 7;
const int L = 16;

void mult(ll a[L][L], ll b[L][L]) {
  static ll res[L][L];
  memset(res, 0, sizeof(res));
  for (int i = 0; i < L; i++)
    for (int j = 0; j < L; j++)
      for (int k = 0; k < L; k++)
        res[i][j] = (res[i][j] + a[i][k] * b[k][j]) % MOD;
  memcpy(a, res, sizeof(res));
}

void solve(int test_num) {
  static const int xsum[10] = { 0, 0, 2, 3, 2 ^ 2, 5, 2 ^ 3, 7, 2 ^ 2 ^ 2, 3 ^ 3 };
  ll N;
  cin >> N;
  static ll mat[L][L], base[L][L];
  memset(mat, 0, sizeof(mat));
  memset(base, 0, sizeof(base));
  for (int i = 0; i < L; i++)
    mat[i][i] = 1;
  for (int x = 0; x < L; x++) {
    base[x][x] += 3; //1,4,9
    base[x][x ^ 1] += 2; //2,8
    base[x][x ^ 2]++; //3;
    base[x][x ^ 1 ^ 2]++; //6
    base[x][x ^ 4]++; //5
    base[x][x ^ 8]++; //7
  }
  for (ll e = N; e > 0; e >>= 1) {
    if (e & 1)
      mult(mat, base);
    mult(base, base);
  }
  ll res = 0;
  for (int x = 1; x < L; x++)
    res = (res + mat[0][x]) % MOD;
  cout << res << endl;
}
