#include<bits/stdc++.h>
using namespace std;
long int abs(long int a,long int b)
{
	return a > b ? a - b  : b - a;
}
long int func(long int *arr,long int n)
{
	long int ans=0;
	for(long int i=1;i<n;i++)
	{
		if(abs(arr[i],arr[i-1])  > ans)
			ans = abs(arr[i],arr[i-1]);
	}
	if(ans < abs(arr[n-1],arr[0]))
		return abs(arr[n-1],arr[0]);
	return ans;
}
int main()
{	
		long int n;
		cin >> n;
		
		long int arr[n];
		for(long int i=0;i<n;i++)
			cin>>arr[i];
		
		sort(arr,arr+n);
		long int brr[n];
		if(n%2 == 0)
		{
			long int left=(n-1)/2 - 1;
			long int right=left+2;
			
			brr[(n-1)/2] = arr[0];
			brr[n-1] = arr[n-1];
			for(long int i=1;i<n-1;i=i+2)
			{
				brr[left--]=arr[i];
				brr[right++]=arr[i+1];
			}
		}
		else
		{
			brr[n/2]=arr[0];
			long int left=n/2-1;
			long int right=left+2;
			for(long int i=1;i<n;i+=2)
			{
				brr[left--]=arr[i];
				brr[right++]=arr[i+1];
			}
		}
		cout<<func(brr,n)<<endl;
	return 0;
}
