using namespace std;

typedef unsigned long long UInt64;

const int N = 1111;
const int inf = 99999;
const UInt64 Mod = ( UInt64 )( 1e9 + 7 );

int main()
{
	int n, m, a, b, d[N][N], x, y;

	cin >> n >> m >> a >> b;

	for( int i = 1; i <= n; ++i )
		for( int j = 1; j <= n; ++j )
			d[i][j] = 1;

	for( int i = 1; i <= m; ++i ){
		cin >> x >> y;
		d[x][y] = 0;
		d[y][x] = 0;
	}

	queue< int > q;

	int dis[N] ;

	for( int i = 1; i <= n; ++i )	dis[i] = inf;

	dis[a] = 0;

	q.push( a );

	while( !q.empty() ){

		int u = q.front();
		q.pop();

		for( int i = 1; i <= n; ++i )
			if( d[u][i] && dis[u] + 1 < dis[i] ){
				dis[i] = dis[u] + 1;
				q.push( i );
			}
	}

	if( dis[b] == 99999 )	cout << "-1";
	else	cout << dis[b];

	return 0;
}
