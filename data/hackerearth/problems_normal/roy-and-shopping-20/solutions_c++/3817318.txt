    #include <bits/stdc++.h>
    #define ll long long unsigned int
    #define pb push_back
    #define Max 1009
    using namespace std;
    typedef vector <int> V;
    typedef pair <int,int> P;
    int A[Max];	V v;
    void sieve()
    {
    	for(int i=3;i<=Max;i++)
    	{
    		A[i]=1;
    	}	v.pb(2);
    	int s=sqrt(Max);
    	for(int i=3;i<=s;i+=2)
    	{
    		if(A[i])
    		{
    			for(int j=i*i;j<=Max;j+=2*i)
    				A[j]=0;
    		}
    	}
    	for(int i=3;i<=Max;i+=2)
    	{
    		if(A[i])
    			v.pb(i);
    	}
    }
     
    int main()
    {
    	sieve();	int n,t;	int i;
    	ios::sync_with_stdio(false);
    	scanf("%d",&t);
    	while(t--)
    	{
    		scanf("%d",&n);	int flag=0;	int s=sqrt(n);
    		for(i=0;v[i]<=s;i++)
    		{
    			if(n%v[i]==0)
    			{
    				flag=1;	break;
    			}
    		}
    		if(!flag)
    			printf("0\n");
    		else
    			printf("%d\n",n-v[i]);
    	}
    	return 0;
    }
