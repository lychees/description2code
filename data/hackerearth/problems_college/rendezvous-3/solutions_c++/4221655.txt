#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
#define ll long long
#define mod 1000000007
#define bitcount    __builtin_popcountint
#define pb push_back
vector <int> q1[100005],q2[100005];
int v[100005],w[100005],len,n;
struct heap{
    int wt;
    int des;
};
heap a[10000000];
void ins(int x,int y)
{
    len++;
    a[len].wt=x;
    a[len].des=y;
    int l=len;
    while(l!=1)
    {
        if(a[l].wt<a[l/2].wt)
            swap(a[l],a[l/2]);
        l/=2;
    }
}
void heapify(int x)
{
    int y=x;
    if(2*x<=len&&a[2*x].wt<a[y].wt)
        y=2*x;
    if(2*x+1<=len&&a[2*x+1].wt<a[y].wt)
        y=2*x+1;
    if(y!=x)
    {
        swap(a[y],a[x]);
        heapify(y);
    }
}
void del()
{
    swap(a[1],a[len]);
    len--;
    heapify(1);
}
int main()
{
    //freopen("in.txt","r",stdin);
    //freopen("out.txt","w",stdout);
    int m,i,j,k;
    ll ans;
    sd(n);
    sd(m);
    while(m--)
    {
    	sd(i);
    	sd(j);
    	sd(k);
    	q1[i].pb(j);
    	q2[i].pb(k);
    	q1[j].pb(i);
    	q2[j].pb(k);
    }
    len=0;
    for(i=0;i<q1[1].size();i++)
    {
        if(v[q1[1][i]]==0)
            ins(q2[1][i],q1[1][i]);
    }
    v[1]=1;
    ans=0;
    while(len)
    {
        while(len&&v[a[1].des])
            del();
        if(len==0)
            break;
        j=a[1].des;
        ans+=a[1].wt;
        v[j]=1;
        del();
        for(i=0;i<q1[j].size();i++)
        {
            if(v[q1[j][i]]==0)
                ins(q2[j][i],q1[j][i]);
        }
    }
    printf("%lld\n",ans);
    return 0;
}
