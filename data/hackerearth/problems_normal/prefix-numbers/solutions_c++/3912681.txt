typedef vector< vector<uint> > mat;

mat pw(mat &A, long long e) {
	int N =A.size();
	if(e <= 0) {
		mat ret(N,vector<uint>(N,0));
		for(int i =0; i < N; i++) ret[i][i] =1;
		return ret;}
	if(e == 1) return A;
	mat x =pw(A,e/2);
	mat y(N,vector<uint>(N,0));
	for(int i =0; i < N; i++) for(int j =0; j < N; j++) 
		for(int k =0; k < N; k++) y[i][j] +=x[i][k]*x[k][j];
	if(e%2 == 0) return y;
	mat z(N,vector<uint>(N,0));
	for(int i =0; i < N; i++) for(int j =0; j < N; j++) 
		for(int k =0; k < N; k++) z[i][j] +=y[i][k]*A[k][j];
	return z;}
