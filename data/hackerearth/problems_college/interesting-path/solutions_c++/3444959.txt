#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t=0;
    scanf("%d",&t);
    while(t--)
    {
    	int n=0;
    	scanf("%d",&n);
    	int dp[n],a[n];
    	int i=0,j=0,ans=0;
    	for(i=0;i<n;i++)
    		scanf("%d",&a[i]);
    	
    	if(n==0 || a[0]==0)
    		ans = 0;
    		
    	dp[0]=0;
    	for(i=1;i<n;i++)
    	{
    		dp[i]=INT_MAX;
    		for(j=0;j<i;j++)
        	{
            if(i<=j+a[j] && dp[j]!=INT_MAX)
            {
                 dp[i]=min(dp[i],dp[j]+1);
                 break;
            }
        	}
    	}
    	cout<<dp[n-1]<<endl;
    }
    return 0;
}
