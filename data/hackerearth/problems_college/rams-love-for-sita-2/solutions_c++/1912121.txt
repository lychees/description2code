#include<iostream>
#include<stdio.h>
#include<math.h>
#include<algorithm>
using namespace std;
#define lli long long int

int main()
{
    lli t,n,c,m,temp,ans,w;
    cin>>t;
    while(t--)
    {
              cin>>n>>c>>m;
              
             w=n/c;
             ans=w;
             
             temp=w/m;
             while(temp>=1)
             {
                           ans+=temp;
                           w=(w%m) +temp;
                           temp=w/m;
             }
             
             cout<<ans<<endl; 
    }
    
    return 0;
}
