ll sumi(ll a, ll b){
    if(a==0) return b*(b+1)%mod*div2%mod;
    return (sumi(0,b) - sumi(0,a-1) + mod)%mod;
}

ll sumi2(ll a, ll b){
    if(a==0) return b*(b+1)%mod*(b*2+1)%mod*div6%mod;
    return (sumi2(0,b) - sumi2(0,a-1) + mod)%mod;
}
