#include <iostream>
using namespace std;

int main()
{
    int n,sum=0,max,i,j,pro,tym,ind;
    cin>>n;
    int p[n];
    tym=n-2;
    for(i=0;i<n;i++)
	    	cin>>p[i];
	    
    while(tym--)
    {	
    	max=0;
   	    for(i=1;i<n-1;i++)
	    {
	    	pro=p[i-1]*p[i+1];
	    	if(pro>max)
	    	{
	    		max=pro;
	    		ind=i;
	    	}
	    }
	    for(j=ind;j<n-1;j++)
	    {
	    	p[j]=p[j+1];
	    }
	    n--;
	    sum+=max;
    }
    cout<<sum;
    return 0;
}
