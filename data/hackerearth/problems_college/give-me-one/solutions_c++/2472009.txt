#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int dp[20000009];
int main()
{
   // cout << "Hello World!" << endl;
   int tc;
   dp[0]=0;
   dp[1]=0;
   dp[2]=1;
   for(int i=3;i<=20000000;++i)
   {
   	if(i%3==0&&i%2==0)
   		dp[i] = 1+min(dp[i/3],min(dp[i/2],dp[i-1]));
   	else if(i%3==0&&i%2!=0)
   		dp[i] = 1+min(dp[i/3],dp[i-1]);
   	else if(i%3!=0&&i%2==0)
   		dp[i] = 1+min(dp[i/2],dp[i-1]);
   	else dp[i] = 1+dp[i-1];
   }
   cin>>tc;
   while(tc--)
   {
   		int n;
   		cin>>n;
   		cout<<dp[n]<<endl;
   }
    return 0;
}
