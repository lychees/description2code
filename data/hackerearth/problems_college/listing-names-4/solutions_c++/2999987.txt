#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    
    int n;
    cin>>n;
    
    string s[n];
    for(int i=0;i<n;i++) cin>>s[i];
    
    sort(s,s+n);
    
    int c=0;
    
    for(int i=0;i<n;i++)
    {
    	c=1;
    	string t=s[i];
    	int j;
    	
    	for(j=i+1;j<n;j++)
    	{
    		if(t==s[j]) c++;
    		else break;
    	}
    	i=j-1;
    	cout<<t<<" "<<c<<endl;
    }
    
    return 0;
}
