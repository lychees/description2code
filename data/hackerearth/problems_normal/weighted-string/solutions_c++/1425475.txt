#include<iostream>
 #include <stdio.h>
#include <string.h>
#include <stdlib.h>
#define MOD 1000000007
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
	int k,cnt=0,sum=0;
	char s[1000000];
	cin>>k;
	cin>>s;
	for(int i=0;i<strlen(s);i++)
	{
		if(((s[i]-'a')+1)==k)
		{
			cnt++;
			continue;
		}
		sum=(s[i]-'a')+1;
		for(int j=i+1;j<strlen(s);j++)
		{
			sum=sum+(s[j]-'a')+1;
			if(sum==k)
			{
				cnt++;
				break;
			}
			if(sum>k)
			break;
		}
	}
	cout<<cnt<<endl;
	}
}
