#include<iostream>
using namespace std;
int arr[26];
int main()
{
	int km;
	char c;
	c = getchar();
	while (c > 96 && c < 123)
	{
		km = (int)c;
		km = km % 97;
		arr[km]++;
		c = getchar();
	}
	int max = arr[25];
	for (int j = 24; j>-1; j--)
	{
		if (arr[j]>arr[max])
			max = j;
	}
	char k = max + 97;
	if (k == 'u')
	{
		printf("Thanks both.\n");
	}
	else if (k == 'c' || k == 'b' || k == 'i' || k == 'n' || k == 'g' || k == 'l')
	{
		printf("Thanks cubing club\n");
	}
	else if (k == 'h' || k == 'e' || k == 'a' || k == 'd' || k == 'r' || k == 's')
	{
		printf("Thanks head rush\n");
	}
	else
	{
		printf("none");
	}
	return 0;
}
