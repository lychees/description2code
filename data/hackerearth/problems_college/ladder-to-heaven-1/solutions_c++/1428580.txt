#include<iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long n;
        cin>>n;
        long long sum = 0;
        long long first = 1;
        long long second = 2;
        long long ans = 0;
        while(sum < n)
        {
            sum = first + second;
            first = second;
            second = sum;
            if(first % 2 == 0){
                ans += first;
            }
        }
        cout<<ans<<endl;
    }
    return 0;
}
