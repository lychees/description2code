#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long t;
    int len,sum1,sum2;
    string st;
    cin>>t;
    while(t--)
    {
    cin>>st;
    len=st.size();
    sum1=sum2=0;
    for(int i=0;i<len/2;i++)
        {
            sum1+=st[i];
            sum2+=st[len-i-1];
        }
        if(sum1==sum2)
            cout<<"YES";
        else
            cout<<"NO";
        cout<<endl;


    }
    return 0;
}
