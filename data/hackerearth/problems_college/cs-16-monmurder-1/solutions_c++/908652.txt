#include <bits/stdc++.h>
using namespace std;

vector<int> factor(int n){
	int total = 0; 
	bool taken = false;
	vector<int> vect;
	while (n % 2 == 0){
		if (taken == false){
			vect.push_back(2);
			taken = true;
		}
		n = n/2;
	}
	for (int i = 3; i <= sqrt(n); i += 2){
		taken = false;
		while (n % i == 0){
			if (taken == false){
				vect.push_back(i);
				taken = true;
			}
			n = n/i;
		}
	}
	if (n > 2)
		vect.push_back(n);
	return vect;
}

bool cal(int n){
	vector<int> v = factor(n);
	if (v.size() == 1){
		return true;
	}	
	else{
		bool res = true;
		for (int i = 0; i < v.size(); i++){
			int k = v[i];
			int j = n;
			while (j % k == 0){
				j = j/k;
				res = res && cal(j);
			}
		}
		if (res == true)
			return false;
		else
			return true;
	}
}
