#include <iostream>
#define MOD 1000000007
using namespace std;
long int gcd(long int a,long int b)
{
	if(a<b) return  gcd(b,a);
	if(b==0) return a;
	else return gcd(b,a%b);
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long int a,b,c,tem,temp;
		cin>>a>>b>>c;
		temp=gcd(gcd(a,b),c);
		tem=((a/temp)*(b/temp))%MOD;
		tem=(tem*(c/temp))%MOD;
		cout<<temp<<' '<<tem<<endl;
	}
   // cout << "Hello World!" << endl;
    return 0;
}
