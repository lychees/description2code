#include <bits/stdc++.h>
using namespace std;
#define ll long long int

int point_a(char c){
	if(c=='T') return 150;
	if(c=='D') return -100;
	return 0;
}

int main()
{
	ll t,n,m,i,j,k;
	cin >> t;
	while(t--){
		cin >> n >>m;
		char arr[n][m];
		getchar();
		for(i=0;i<n;i++){
			for(j=0;j<m;j++){
				 arr[i][j]=getchar();
				 getchar();
			}
		}
		ll dp[n][m];
		for(i=0;i<n;i++){
			for(j=0;j<m;j++){
				k=point_a(arr[i][j]);
				if(i==0 && j==0){
					dp[i][j]=k;
				}
				else if(i==0){
					dp[i][j]=dp[i][j-1]+k;
				}
				else if(j==0){
					dp[i][j]=dp[i-1][j]+k;
				}
				else{
					dp[i][j]=max(dp[i][j-1],dp[i-1][j])+k;
				}
				if(dp[i][j]<0) dp[i][j]=-10000023;
			}
		}
		if(dp[n-1][m-1]<0) cout << "-1" << endl;
		else cout << dp[n-1][m-1] << endl;
	}
    return 0;
}
