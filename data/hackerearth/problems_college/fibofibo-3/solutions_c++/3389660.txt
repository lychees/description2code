#include<bits/stdc++.h>
using namespace std;

int fib(int n, int &zero, int &one){

  if(n==0)
    {
        zero++;
        //cout<<zero<<" "<<one<<endl;
        return 0;
    }
  else if(n==1){
    one++;
    return 1;


  }

  return fib(n-1,zero,one)+fib(n-2,zero,one);


}


int main(){

 int t, n, zero, one;
 cin>>t;
 while(t--){
    cin>>n;
    zero=0;
    one=0;
    fib(n,zero,one);

    cout<<zero<<" "<<one<<endl;

 }





 return 0;
}
