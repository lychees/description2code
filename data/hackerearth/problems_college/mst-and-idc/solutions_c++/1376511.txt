#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--){
    	int n;
    	cin >> n;
    	int a[n];
    	for(int i=0;i<n;i++){
    		 cin >> a[i];
    	}
    	sort(a,a+n);
    	
    	int d=a[0]-a[n-1];
    	if(d<0)
    	   d=-d;
    	cout << d*2 << "\n";
    }
    return 0;
}
