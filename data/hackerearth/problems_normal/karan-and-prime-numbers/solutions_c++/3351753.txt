#include<bits/stdc++.h>
using namespace std;

	int arr[1000005];
int main()
{
	int t;

//	memset( arr, 0, sizeof(arr) );
	arr[0] = 1;
	arr[1] = 1;
	for ( long long int i = 2; i <= 1000000; i++ )
	{
		if ( arr[i] == 0 )
		{
			for ( long long int j = i*i; j <= 1000000; j+= i )
			   arr[j] = 1;
		}
	}
	cin >> t;
	while (t--)
	{
		long long int a, b;
		cin >> a >> b;
		long long ans = 0;
		for ( long long int i = a; i <= b; i++ )
		{
			if ( arr[i] == 0 )
			   ans += i;
		}
		cout << ans << endl;
	}
	return 0;
}
