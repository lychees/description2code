{

	ll ans = 0;
	int t,n,i;

		ans = 0;
		sc(n);
		for(i=0;i<n;i++)
		{
			scll(a[i]);
		}
		s[n-1] = a[n-1];
		for(i=n-2;i>=0;i--)
		{
			s[i] = max(a[i],s[i+1]);
