#include <bits/stdc++.h>
    using namespace std;
    #define ll unsigned long long
    int main ()
    {
      ll int t,x,y;
      cin>>t;
      while(t--)
      {
      	ll int remx,remy,carry=0,count=0;
      	cin>>x>>y;
      	if(x==0||y==0)
      	cout<<"No carry operation"<<endl;
      	else
      	{
      	  while(x>0||y>0)
      	  {
      	  	remx=x%10;
      	  	x/=10;
      	  	remy=y%10;
      	  	y/=10;
      	  	carry=(carry+remx+remy)/10;
      	  	
      	  	if(carry)
      	  	count++;
      	  }
      	  if(count==0)
      	  cout<<"No carry operation"<<endl;
      	  else if(count==1)
      	  cout<<count<<" carry operation"<<endl;
      	  else
      	  cout<<count<<" carry operations"<<endl;
      	}
      }
      return 0;
    }
