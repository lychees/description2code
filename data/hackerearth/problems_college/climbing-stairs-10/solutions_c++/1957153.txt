#include <iostream>
using namespace std;

long long int stairs(int N)
{
	long long int ways[N+1];
	ways[0]=1;
	ways[1]=1;
	for(int i=2;i<=N;i++){
		ways[i]=ways[i-2]+ways[i-1];
	}
	return ways[N];
}

int main()
{
    //cout << "Hello World!" << endl;
   	int T;
   	cin>>T;
   	int N;
   	while(T--){
   		cin>>N;
   		cout<<stairs(N)<<endl;
   	}
    return 0;
}
