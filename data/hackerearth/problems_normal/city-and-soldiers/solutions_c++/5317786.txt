#include<bits/stdc++.h>
#define N 100005
using namespace std;
int n,q,x,y,arr[N];
int root(int x)
{
	while(arr[x]!=x)
	{
		arr[x]=arr[arr[x]];
		x=arr[x];
	}
	return x;
}
void merge(int x,int y)
{
	int root_x=root(x);
	int root_y=root(y);
	arr[root_x]=arr[root_y];
}
void become(int x)
{
	int _root=root(arr[x]);
	arr[_root]=x;
	arr[x]=x;
}
int main()
{
	ios::sync_with_stdio(false);
	cin.tie(NULL);
	cin>>n>>q;
	for(int i=1;i<=n;i++)
		arr[i]=i;
	for(int i=0;i<q;i++)
	{
		cin>>x;
		if(x==1)
		{
			cin>>x>>y;
			merge(x,y);
		}
		else
		if(x==2)
		{
			cin>>x;
			become(x);
		}
		else
		{
			cin>>x;
			cout<<root(x)<<endl;
		}
	}
	return 0;
}
