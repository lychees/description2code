int angles[360],tangles[360];

{

            if(angles[i]==1)

                            for(int j=0;j<360;j++)

                                    if(angles[j]==1)
                                    {
                                                    if(i!=j)

                                                            temp=(i+j)%360;
                                                            angles[temp]=1;
                            
                                                            if(i<j)
                                                                   temp=j-i;
                                                            else
                                                                temp=i-j;
                            
                                                            angles[temp]=1;    
                                                     }
                                      }
                            }
            }        
    }

}
