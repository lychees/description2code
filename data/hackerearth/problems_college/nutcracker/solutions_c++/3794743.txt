#include<bits/stdc++.h>
using namespace std;
int main(){
int t;
cin>>t;
while(t--){
    int n;
    cin>>n;
    int c=0;
    while(n>0)
    {
        c++;
        if(c%2)
        {
            if(n>=4)
                n-=4;
            else
                n-=1;
            if(n==0)
            {
                cout<<"Yes"<<endl;
                break;
            }
        }
        else
        {
            n-=1;
            if(n==0)
            {
                cout<<"No"<<endl;
                break;
            }
        }
    }
}

return 0;
}
