#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n;
	cin>>n;
	int a[n];
	for(int i=0;i<n;i++){
		cin>>a[i];
	}
	sort(a,a+n);
	int q,sum;
	cin>>q;
	while(q--){
		int k,*t;
		cin>>sum;
		t=find(a,a+n,sum);
		if(t!=a+n){
			cout<<"I Like You"<<"\n";
		}else{
			int l=0,r=n-1;
			while(l<r){
				if(a[l]+a[r]==sum){
				   k=-1;
				   break;
				}else if(a[l]+a[r]>sum){
					r--;
					k=1;
				}else{
					l++;
					k=1;
				}
			}
		if(k==-1){
			cout<<"I Like You"<<"\n";
		}else{
			cout<<"I Hate You"<<"\n";
		}
		
		}
	}
    return 0;
}
