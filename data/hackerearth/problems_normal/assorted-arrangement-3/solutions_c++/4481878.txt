#include <bits/stdc++.h>
using namespace std;
long long gt,n,m,a;
int main()
{
	gt=0;
	cin>>n>>m;
	long long c[m];
	for(long long i=0;i<m;i++)
	{
		cin>>c[i];
	}
	for(long long i=0;i<n;i++)
	{
		cin>>a;
		gt/=c[a-1];
		gt+=1;
		gt*=c[a-1];
		for(long long j=a;j<m;)
		{
			if(gt%c[j]==0)
			{
				gt+=c[a-1];
				j=a;
			}
			else
			{
				j++;
				if(j==m)
				break;
			}
		}
	}
	cout<<gt<<endl;
    return 0;
}
