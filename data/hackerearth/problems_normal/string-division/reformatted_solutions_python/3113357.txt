import collections


def words(digit):
    i = 1
    while True:
        Z = (i*(i+1))/2
        if Z > digit:
            return i-1
        i += 1


for _ in range(input()):
    inp = map(str, raw_input())
    v = collections.Counter(inp).values()
    cnt = 0
    for digit in v:
        cnt += words(digit)
    if cnt >= 4:
        print 'YES'
    else:
        print 'NO'
