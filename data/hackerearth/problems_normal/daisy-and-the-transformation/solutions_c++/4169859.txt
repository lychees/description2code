#include <iostream>
#include <string>
using namespace std;

int main()
{
    
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	string s1,s2;
    	cin>>s1>>s2;
    	cin>>n;
    	int l1 = s1.length();
    	int l2 = s2.length();
    	if(l1!=l2)
    		cout<<"No"<<endl;
    	else
    	{
    		int m = 0;
    		for(int i=0;i<l1;++i)
    		{
    			if(s1[i]!=s2[i])++m;
    			
    		}
    		if(m==n)
    			cout<<"Yes"<<endl;
    		else if(n>m)
    		{
    				if((n-m)%2==0)
    					cout<<"Yes"<<endl;
    				else
    					cout<<"No"<<endl;
    		}
    		else
    			cout<<"No"<<endl;
    	}
    }
    return 0;
}
