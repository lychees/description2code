#include<iostream>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t>0){
		long long int i,n;
		cin>>n;
		long long int a[n],sum=0;
		for(i=0;i<n;i++){
			cin>>a[i];
			sum+=a[i];
		}
		for(i=0;i<n;i++){
			cout<<((sum/(n-1))-a[i])<<" ";
		}
		cout<<endl;
		t--;
	}
	return 0;
}
