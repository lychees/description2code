#include <iostream>
#include <algorithm>
using namespace std;

void compute(int *a,int n)
{
	int i,count=0;
	sort(a,a+n);
	for(i=0;i<n;i++)
	{
		if(a[i]==0)
			continue;
		if(a[i]==a[i+1]-1 && i+1<n)
			a[i+1]=0;
		if(a[i]==a[i-1]+1 && i-1>=0)
			a[i-1]=0;
		count++;
		a[i]=0;		
	}
	cout<<count<<endl;
}
int main()
{
    int i,j,t,n;
    cin>>t;
    for(i=0;i<t;i++)
    {
    	cin>>n;
    	int a[n];
    	for(j=0;j<n;j++)
    		cin>>a[j];
    	compute(a,n);
    }
    return 0;
}
