#include <iostream>
#include <string>
#include <unordered_map>

using namespace std;

int main()
{
	int tc;
	cin >> tc;
	string str;
	unordered_map<char,int> m;
	while(tc--)
	{
		m.clear();
		cin >> str;
		for(auto i=str.begin();i!=str.end();++i)
		{
			auto j=m.find(*i);
			if(j==m.end())
			{
				m[*i]=1;
			}
			else
			{
				m[*i]+=1;
			}
		}
		int count=0;
		for(auto i=m.begin();i!=m.end();++i)
		{
			if((*i).second%2)
			{
				++count;
			}
			if(count>1)
			{
				break;
			}
		}
		if(count>1)
		{
			cout << "No" << endl;
		}
		else
		{
			cout << "Yes" << endl;
		}
	}
	return 0;
}
