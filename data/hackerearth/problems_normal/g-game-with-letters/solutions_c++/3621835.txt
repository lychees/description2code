ll power(ll a,ll b)
{
	if(!b)
		return 1;
	if(b==1)
		return a;
	ll temp=power(a,b/2);
	temp=(temp*temp);
	if(b&1)
		temp=(temp*a);
	return temp;
}
typedef struct
{
	int a,b;
}node;
int main()
{
    int i,t,n,l,val,sz,j,maxi,in,msz;
    string s;
    string ss;
    scanf("%d",&t);
    while(t--)
    {
    	cin>>n>>s;
    	l=s.length();
    	vector<int> v(26);
    	for(i=0;i<l;i++)
    		v[s[i]-97]++;
    	node str[n];
    	for(i=0;i<n;i++)
    	{
    		val=0;
    		cin>>ss;
    		sz=ss.length();
    		for(j=0;j<sz;j++)
    			val=val+v[ss[j]-97];
    		str[i].a=val;
    		str[i].b=sz;
    	}
    	maxi=str[0].a;
    	msz=str[0].b;
    	in=0;
    	for(i=1;i<n;i++)
    	{
    		if(str[i].a>maxi)
    		{
    			maxi=str[i].a;
    			msz=str[i].b;

    			if(str[i].b<msz)
    			{
    				msz=str[i].b;
    				in=i;
    			}
    		}
    	}
    	printf("%d\n",in+1);
    }

}
