#include <iostream>
#include <vector>
using namespace std;

int grey_hat(int L,int R,int K)
{
	if(L*K+1 > R)
		return 1;
	else
		return 1 + grey_hat(L*K+1,R,K);
}
int main()
{
	int T,L,R,K;
	cin>>T;
	vector<int> ans;
    for(int i=0;i<T;i++)
    {
    	cin>>L>>R>>K;
    	ans.push_back(grey_hat(L,R,K));
    }
    for(int i=0;i<T;i++)
    {
        cout<<ans[i]<<endl;
    }
}
