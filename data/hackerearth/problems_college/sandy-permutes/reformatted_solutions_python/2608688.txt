n, k = [int(i) for i in raw_input().split()]
arr = [int(i) for i in raw_input().split()]
i = 0
if k >= n:
    k = n - 1
while (i < k and n > 0):
    if i != arr.index(n):
        arr[arr.index(n)], arr[i] = arr[i], arr[arr.index(n)]
    else:
        k = k+1
    i = i+1
    n = n-1
print " ".join(map(str, arr))
