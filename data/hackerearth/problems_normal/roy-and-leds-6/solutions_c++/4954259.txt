#include <iostream>
using namespace std;



int main()
{
	int t,r,g,b,i,j;
	int *arr;
	
	int *res = new int[8]{0};
	
	cin>>t;
	t--;
	arr = new int[1000010]{0};
	cin>>r>>g>>b;
	
	for(i=r;i<=t;i+=r*2)
	{
		for(j = i;j<i+r && j<=t;j++)
			arr[j]++;
	}
	
	for(i=g;i<=t;i+=g*2)
	{
		for(j = i;j<i+g && j<=t;j++)
			arr[j]+=2;
	}
	
	for(i=b;i<=t;i+=b*2)
	{
		for(j = i;j<i+b && j<=t;j++)
			arr[j]+=4;
	}
	
	for(i=0;i<=t;i++)
	{
		//cout<<arr[i]<<" ";
		res[arr[i]]++;
	}
	
	cout<<res[1]<<" "<<res[2]<<" "<<res[4]<<" "<<res[3]<<" "<<res[6]<<" "<<res[5]<<" "<<res[7]<<" "<<res[0];
    return 0;
}
