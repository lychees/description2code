'''

name = raw_input('What is your name?\n')

for _ in range(input()):
    n, k = map(int, raw_input().split())
    a = sorted(map(int, raw_input().split()))
    if (k == 0):
        print 0
        continue
    x = s = 0
    for i in range(n):
        s += a[n-i-1]
        x += 1
        if (x == k):
            break
    print s
