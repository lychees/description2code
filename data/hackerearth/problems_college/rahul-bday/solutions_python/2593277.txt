for _ in xrange(int(raw_input())):
    b,w= map(int,raw_input().split())
    x,y,z=map(int,raw_input().split())
    if x==y:
        print (b+w)*x
    else:
        val=min(x,y)
        if x==val:
            z1=y
            z2=x
        else:
            z1=x
            z2=y
        if x==val:
            val1=b
            val2=w
        else:
            val1=w
            val2=b
        print val1*val+min(val2*z1, val2*(z2+z))
