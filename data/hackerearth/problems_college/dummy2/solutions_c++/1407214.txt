#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
	int e,i,j;
	scanf("%d",&e);
	while(e--)
    {
    	string s;
    	cin>>s;
    	string t="";
    	for(i=0;i<s.length();i++)
    	{
    		if(s.at(i)>='0'&&s.at(i)<='9')
    		{
    			string y=t;
    			for(j=0;j<s.at(i)-'0'-1;j++)
    			t+=y;
    		}
    		else
    		t+=s.at(i);
    	}
    	cout<<t<<endl;
    }
    
    return 0;
}
