#include<bits/stdc++.h>
using namespace std;

int main()
{
    string s; cin >> s;
    stack <char> st;int count=0;
    for(int i=0;i<s.length();)
    {
    	if(count%2==0)
    	{
    		cout << s[i];
    		st.push(s[i]);
    		i++;
    	}
    	else
    	{
    		cout << "-";
    		st.push('-');
    	}
    	count++;
    }
    cout << endl;
    while(!st.empty())
    {
    	cout << st.top();
    	st.pop();
    }
    return 0;
}
