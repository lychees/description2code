#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int n, m;
    cin>>n>>m;
    vector<int> arr;
    arr.push_back(0);
    int x;
    for(int i=1;i<=m;i++) {
    	cin>>x;
    	arr.push_back(x);
    }
    long dp[m+1][n+1];    
    for(int i=0;i<=n;i++) {
    	dp[0][i] = 0;
    }
    for(int i=0;i<=m;i++) {
    	dp[i][0] = 0;
    }
    sort(arr.begin(), arr.end());
    for(int i=1;i<=m;i++) {
    	for(int j=1;j<=n;j++) {
    		if(arr[i] > j) dp[i][j] = dp[i-1][j];
    		else if(arr[i] == j) {
    			dp[i][j] = 1+dp[i-1][j];
    		}
    		else {
    			dp[i][j] = dp[i][j-arr[i]] + dp[i-1][j];
    		}
    	}
    }
    
    cout<<dp[m][n]<<endl;
    return 0;
}
