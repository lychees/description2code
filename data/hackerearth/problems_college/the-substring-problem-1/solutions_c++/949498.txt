#include<stdio.h>
int main()
{
    int t,n,i,j;
    char s[110];
    scanf("%d",&t);
    while(t--)
    {
        int a[110][30]={0},b[30]={0};
        scanf("%d",&n);
        for(i=0;i<n;i++)
        {
            char ch=getchar();
            scanf("%s",s);
            for(j=0;s[j];j++)
            {
                if(!a[i][s[j]-'a'])
                {
                    b[s[j]-'a']++;
                    a[i][s[j]-'a']=1;
                }
            }
        }
        int pos=-1,max=-1;
        for(i=0;i<26;i++)
        {
            if(b[i]>max)
            {
                max=b[i];
                pos=i;
            }
        }
        printf("%c\n",'a'+pos);
    }
    return(0);
}
