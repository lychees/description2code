#include <cstdio>
#include <iostream>
#include <cmath>
#include <algorithm>
#include <string.h>
#include <stdlib.h>
#include <vector>
#include <stack>
#include <queue>

#define foru(i, l, r) for(int i = l; i <= r; ++i)
#define ford(i, r, l) for(int i = r; i >= l; --i)

using namespace std;

typedef long long ll;

ll a[10000005], s[10000005];

int np (ll val) {
    int l = 1, r = 10000000, kq = -1;
    while (l <= r) {
        int mid = (l + r) >> 1;
        if (a[mid] <= val) {
            kq = mid; l = mid + 1;
        }
        else r = mid - 1;
    }
    return kq;
}

int main () {
    //freopen ("SUM.INP", "r", stdin);
    //freopen ("SUM.OUT", "w", stdout);
    foru (i, 1, 10000000) {
        a[i] = ((ll)i * (ll)(i + 1)) / 2;
        if (i % 2 != 0) s[i] = s[i - 1] + (ll)i;
        else s[i] = s[i - 1];
    }
    int t; scanf ("%d", &t);
    while (t--) {
        ll l, r; cin >> l >> r;
        ll res = 0;
        int x = np (l - 1);
        res = -s[x];
        if (l - 1 - a[x] > 0)
            if ((l - 1 - a[x]) % 2 != 0) res -= (ll)(x + 1);
        x = np (r);
        res += s[x];
        if (r - a[x] > 0)
            if ((r - a[x]) % 2 != 0) res += (ll)(x + 1);
        cout << res << "\n";
    }
    //fclose(stdin);
    //fclose(stdout);
}
