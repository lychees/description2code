#include<bits/stdc++.h>
#define ll long long int
using namespace std;
int main()
{
      int t;
      set<char> st;
      cin>>t;
      while(t--)
      {
            ll i,n,ans;
            string s;
            cin>>s;
            for(i=0;i<s.length();i++)
               st.insert(s[i]);
               n=st.size();
               ans=(n*n+n)/2;
               cout<<ans<<endl;
               st.clear();
      }
      return 0;
}
