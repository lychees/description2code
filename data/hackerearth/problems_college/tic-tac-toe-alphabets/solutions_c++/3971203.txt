#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    int t,i,j,k,l;
    char ch[500][1000],temp;
    cin>>t;
    for(i=0;i<t;i++)
    cin>>ch[i];
    i=0;
    while(i<t)
    {
    	l=strlen(ch[i]);
    	for(k=0;k<l;k++)
    	for(j=0;j<l-k-1;j++)
    	{
    		if(ch[i][j]>ch[i][j+1])
    		{
    			temp=ch[i][j];
    			ch[i][j]=ch[i][j+1];
    			ch[i][j+1]=temp;
    		}
    	}
    	cout<<ch[i]<<endl;
    	i++;
    }
    return 0;
}
