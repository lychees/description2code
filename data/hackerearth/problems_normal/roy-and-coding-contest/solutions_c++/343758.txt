#include <iostream>
using namespace std;

int main()
{ int t,i;
cin>>t;
    unsigned long long int n,m,count,x,y;
    for(i=0;i<t;i++)
    {
	count=1;x=1,y=0;
    cin>>n>>m;
    	while(x<n)
    	{
    		if(y>=m)
    		{
    			count+=((n-x)/m);
    			if((n-x)%m!=0)
    			count+=1;
    			break;
    		}
    		
    		x=x*2;
    		y=x;
    		count++;
    		
    	}
    	if(n==1)
    	cout<<0<<"\n";
    	else
    	cout<<count<<"\n";
    }
    return 0;
}
