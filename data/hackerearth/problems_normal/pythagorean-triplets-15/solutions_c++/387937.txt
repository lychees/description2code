#include <iostream>
using namespace std;

long int gcd (long int a, long int b ) {
  long int c;
  while ( a != 0 ) {
     c = a;
     a = b % a;
     b = c;
  }
  return b;
}

long int linear_search(long int q, long int n) {
	long int ans = 0;
	long int p;
	p = q + 1;
	int a = p*p - q*q;
	int b = 2 * p * q;
	int c = p*p + q*q;
	while(a <= n && b <= n && c <= n) {
		if((p%2 == 0 || q%2 == 0) && gcd(p, q) == 1) {
			ans = ans + (n/c);
		}
		p++;
		a = p*p - q*q;
		b = 2 * p * q;
		c = p*p + q*q;
	}
	return ans;
}

int main() {
	long int t, n, p, q, count;
    cin >> t;
    while(t--) {
    	count = 0;
    	cin >> n;
    	q = 1;
    	p = q + 1;
    	while(p*p - q*q <= n && 2 * p * q <= n && p*p + q*q <= n) {
    		count = count + linear_search(q, n);
    		q++;
    	}
    	cout << count << '\n';
    }
    return 0;
}
