#include <iostream>
#include<limits.h>
#define ll long long
using namespace std;

void stockBuySell(ll arr[],ll n)
{
	ll max=INT_MIN;
	ll result=0;
	for(ll i=n-1;i>=0;i--)
	{
		if(max<arr[i])
		   max=arr[i];
	result=result+(max-arr[i]);	   
	}
cout<<result<<endl;	
}
 
// Driver program to test above functions
int main()
{
    ll T,N;
    ll arr[1000001];
    cin>>T;
    while(T--)
    {
    	cin>>N;
    	for(ll i=0;i<N;i++)
    	   cin>>arr[i];   
    	   stockBuySell(arr, N);
    }
    
    
 
    return 0;
}
