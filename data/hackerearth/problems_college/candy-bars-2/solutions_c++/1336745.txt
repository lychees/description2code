#include<bits/stdc++.h>
using namespace std;
int Ceil(int a[], int l, int r, int key) {
    int m;

    while( r - l > 1 ) {
        m = l + (r - l)/2;
        (a[m] >= key ? r : l) = m;
    }
    return r;
}
int printlis(int a[], int size) {

    int *tail = new int[size];
    int len;
    memset(tail, 0, sizeof(tail[0])*size);
    tail[0] = a[0];
    len = 1;
    for( int i = 1; i < size; ++i ) {
        if( a[i] < tail[0] )
            tail[0] = a[i];
        else if( a[i] > tail[len-1] )
            tail[len++] = a[i];
        else
            tail[Ceil(tail, -1, len-1,a[i])] = a[i];
    }
    delete[] tail;
    return len;
}
int main()
{
 int n;
    cin>>n;
    int * ar = (int *)malloc(sizeof(int)*n);
    for(int i = 0; i < n; ++i)
        cin>>*(ar + i);
    printf("%d\n", n - printlis(ar, n));
    return 0;
}
