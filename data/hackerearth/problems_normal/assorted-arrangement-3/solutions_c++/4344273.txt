#include <cstdio>

using namespace std;

typedef long long llong;

#define MAXN 100004
#define MAXM 52
int N, M;
int C[MAXM];
int A[MAXN];
llong B[MAXN];

llong F[104];
llong solve() {
   for (int j = 1; j <= M; ++j)
      F[j] = 0;

   for (int i = 1; i <= N; ++i) {
      int j = A[i];
      int c = C[j];

      F[j] = B[i-1] / c;

      while (true) {
         ++F[j];
         llong x = F[j] * c;
         if (x <= B[i-1])
            continue;
         bool ok = true;
         //if (x > c) {
            for (int jj = j+1; jj <= M; ++jj)
               if ((x % C[jj]) == 0) {
                  ok = false;
                  break;
               }
         //}
         if (ok) {
            B[i] = x;
            break;
         }
      }
      //fprintf(stderr, "%lld ", B[i]);
   }
   //fprintf(stderr, "\n");

   return B[N];
}

int main(int argc, char* argv[]) {
   scanf("%d %d", &N, &M);
   for (int j = 1; j <= M; ++j)
      scanf("%d", C+j);
   for (int i = 1; i <= N; ++i)
      scanf("%d", A+i);

   llong res = solve();
   printf("%lld\n", res);

   return 0;
}
