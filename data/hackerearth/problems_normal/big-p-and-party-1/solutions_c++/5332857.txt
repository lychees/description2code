#include <bits/stdc++.h>
using namespace std;

int main()
{
    int a, b;
    cin>>a>>b;
    int g[a][a];
    memset(g, 0, sizeof(g));
    for(int i=0; i<b; i++)
    {
    	int x, y;
    	cin>>x>>y;
    	g[x][y]=g[y][x]=1;
    }
    
    int visited[a], ans[a];
    memset(visited, 0, sizeof(visited));
    memset(ans, 0, sizeof(ans));
    queue<int> Q;
    visited[0]=1;
    Q.push(0);
    ans[0]=0;
    while(!Q.empty())
    {
        int x = Q.front();
        Q.pop();
        visited[x]=1;
        for(int i=0; i<a; i++)
        {
        	if(g[x][i]==1 && visited[i]==0)
        	{
        		Q.push(i);
        		visited[i]=1;
        		ans[i]=ans[x]+1;
        	}
        }
    }
    
    for(int i=1; i<a; i++)
    {
    	if(ans[i]==0)
    		cout<<"-1\n";
    	else
    		cout<<ans[i]<<"\n";
    }
    
    return 0;
}
