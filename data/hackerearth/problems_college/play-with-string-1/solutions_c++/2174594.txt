#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string st;
    	cin>>st;
    	string sr=st;
    	reverse(st.begin(),st.end());
    	int p=0;
    	for(int i=1;i<sr.size();i++)
    	{
    		int x1=sr[i];
    		int x2=sr[i-1];
    		int y1=st[i];
    		int y2=st[i-1];
    		
    		if(abs(x1-x2)!=abs(y1-y2))
    		 {
    		 	cout<<"Not Funny"<<endl;
    		 	p++;
    		 	break;
    		 }
    		 
    	}
    	
    	if(p==0)
    	cout<<"Funny"<<endl;
    	
    	
    }
    return 0;
}
