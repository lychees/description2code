#include <iostream>
using namespace std;


int main()
{
	int T,K;
	cin>>T;
	while(T--)
	{
		cin>>K;
		string S;
		cin>>S;
		for(int i=0;S[i]!='\0';i++)
		{
			if(S[i]>='a'&&S[i]<='z')
			{
				int d=int(S[i]-'a');
				d=(d+K)%26;
				cout<<char(d+'a');
			}
			else
			{
				int d=int(S[i]-'A');
				d=(d+K)%26;
				cout<<char(d+'A');
			}
		}
		cout<<"\n";
	}
	return 0;
}
