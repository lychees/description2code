#include <bits/stdc++.h>
using namespace std;
int prime(int n)
{
	if(n==1)
	{
		return (0);
	}
	for(int i=2;i<=sqrt(n);i++)
	{
		if(n%i==0)
		{
			return (0);
		}
	}
	return 1;
}
int main()
{
    int t;
    cin>>t;
    string s;
    while(t--)
    {
    	cin>>s;
    	for(int i=0;i<s.length();i++)
    	{
    		if(prime((int)s[i]))
    		{
    			s[i]='*';
    		}
    	}
    	int k=0;
    	int flag=0;
    	while(k<s.length())
    	{
    		if(s[k]!='*')
    		{
    			flag=1;
    			cout<<s[k];
    		}
    		k++;
    	}
    	if(flag==0)
    	{
    		cout<<"NULL";
    	}
    	cout<<endl;
    }
}
