ll query(ll in){
	if(in < 1)
		return 0;
	ll sum = 0;
	while(in > 0){
		sum += bit[in];
		in = in - (in & (-in));
	}
	return sum;
}
void update(ll in,ll x){
	while(in <= n){
		bit[in] += x;
		in = in + (in & (-in));

}
