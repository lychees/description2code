#include <bits/stdc++.h>

using namespace std;

int t , n , m, tem1, tem2 , k, ans = -1;
int arr [10000 + 1];
int dp [10000 + 1][1000 + 1];

int rec(int index, int maxi){

    int &st = dp[index][maxi];
    if(st != -1) return st;

    int sum = 0;

    for(int i = maxi + 1; i <= n; i++){

        int second = arr[i];
        if(second < index) continue;
        sum = max(sum , 1 + rec(second , i) );
    }

    return st = sum;
}

int main()
{
    cin >> t;

    while(t--){

        ans = 0;
        cin >> n;

        memset(dp , -1, sizeof dp);

        for(int i = 0; i < n; i++){

            cin >> tem1 >> tem2;
            arr[tem1] = tem2;
        }

        for(int i = 1; i <= n; i++){

            ans = max(ans, rec(i, arr[i]));
        }

        cout << 1 + ans << "\n";
    }

    return 0;
}
