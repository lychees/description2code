def mod(a):
	if a>0:
		return a;
	else:
		return -a;
t=int(raw_input());
while t!=0:
	t-=1;
	s=raw_input();
	r=s[::-1];
	k=0;
	for i in range(1,len(s)):
		x=ord(s[i])-ord(s[i-1]);
		y=ord(r[i])-ord(r[i-1]);
		if mod(x)==mod(y):
			k+=1;
	if k==(len(s)-1):
		print "Funny"
	else:
		print "Not Funny"
