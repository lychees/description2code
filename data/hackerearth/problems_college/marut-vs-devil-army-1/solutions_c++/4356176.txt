#include <bits/stdc++.h>
using namespace std;
int main() {
	int t;
	scanf("%d", &t);
	while (t--) {
		int n, e, v;
		scanf("%d %d %d", &n, &e, &v);
		long long res = 0;
		for (int i = 0; i < n; ++i) {
			int e2;
			scanf("%d", &e2);
			int cnt = max(e2 - (e - 1), 0) / v;
			if (max(e2 - (e - 1), 0) % v != 0) {
				++cnt;
			}
			res += cnt;
			e = e + cnt * v - e2;
		}
		printf("%lld\n", res);
	}
	return 0;
}
