for _ in range(input()):
    nrows = input()
    rowsums = [0] * nrows
    colsums = [0] * nrows
    total_choc = 0
    for i in range(nrows):
        ins = raw_input().strip()        
        for j in range(len(ins)):
            if ins[j] =="#":
                colsums[j] += 1
                rowsums[i] += 1
                total_choc  += 1
    yes = False            
    
    sum_till_now = 0
    for i in range(nrows):
        sum_till_now += rowsums[i]
        if sum_till_now == total_choc - sum_till_now:
            yes = True
            break
    if not yes:
        sum_till_now = 0
        for i in range(nrows):
            sum_till_now += colsums[i]
            if sum_till_now == total_choc - sum_till_now:
                yes = True
                break
    print "YES" if yes else "NO"
