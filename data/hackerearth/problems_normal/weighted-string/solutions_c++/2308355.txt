#include <iostream>
using namespace std;

int main()
{
   	int t;
   	cin >> t;
   	for(int p=0;p<t;p++)
   	{
   		int k;
   		cin >> k;
   		string s;
   		cin >> s;
   		int l=s.size(),sum=0,cnt=0,j=0;
   		for(int i=0;i<l;i++)
   		{
   			sum=sum+(s[i]-96);
   			if(sum==k)
   				cnt++;
   			if(sum>k)
   			{
   				while(sum>k)
   				{
   					sum=sum-(s[j]-96);
   					j++;
   				}
   				if(sum==k)
   					cnt++;
   			}
   		}
   		cout << cnt << endl;
   	}
    return 0;
}
