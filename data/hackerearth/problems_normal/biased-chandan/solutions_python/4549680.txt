#!/usr/bin/python

str = raw_input()
n = int(str)

score = []
newArr=[]
sum = 0

for i in range(0, n):

        str1 = raw_input()
        score.append(int(str1))

for j in range(0, n):
        if (score[j] == 0):
                if j == 0:
                        continue
                newArr.pop(-1)
        else:
                newArr.append(score[j])

for k in newArr:
        sum += k

print sum
