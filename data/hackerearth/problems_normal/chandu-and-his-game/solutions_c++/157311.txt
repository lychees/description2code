#include <iostream>
using namespace std;


int main()
{
    int n,k;
    
    scanf("%d%d",&k,&n);
    
    int a[n],b[n];
    
    for(int i=0;i<n;i++ ){
    scanf("%d",&a[i]);
    	
    }
    
    
    for(int i=0;i<n;i++ ){
    	scanf("%d",&b[i]);
    	
    }
    
    long long total=0;
    
    for(int i=0;i<n;i++ ){
    	int x = min(a[i]-1,k-a[i]);
    	int y = min(b[i]-1,k-b[i]);
    	
    	total+=x+y;
    	
    }
    printf("%lld\n",total);
    
    return 0;
}
