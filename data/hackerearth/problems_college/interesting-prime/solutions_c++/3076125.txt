#include <bits/stdc++.h>

using namespace std;

#define ll long long

int main() {
	ll t, n, i, ans;
	cin >> t;
	while (t--) {
	    cin >> n;
	    ans = 0;
	    for (i = 1; i * i <= n; i++) {
	        if (n % i == 0 && n / i != i) ans += 2;
	        else if(n % i == 0) ans++;
	    }
	    cout << ans - 1 << endl;
	}
	
	return 0;
}
