#include <iostream>
#include<stdlib.h>
#include<stdio.h>
#include<math.h>
using namespace std;
int reverse(int n)
{
	long long int r=0;
	while(n!=0)
	{
		r=r*10+n%10;
		n/=10;
	}
	return r;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int num1,num2;
		scanf("%d %d",&num1,&num2);
		int ans=reverse(reverse(num1)+reverse(num2));
		printf("%d\n",ans);
	}  
}
