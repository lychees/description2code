#!/usr/bin/python

from collections import deque


def ir():
    return int(raw_input())


def ia():
    line = raw_input()
    line = line.split()
    return map(int, line)


def more(s):
    n = len(s)
    cnt = 0
    for e in s:
        if e in v:
            cnt = cnt + 1
    if n-cnt > cnt:
        return True
    else:
        return False


def cp(e):
    ans = e not in v
    return ans


def cons(s):
    n = len(s)
    for i in range(n):
        if i+2 < n and cp(s[i]) and cp(s[i+1]) and cp(s[i+2]):
            return True
    return False


def solve(s):
    if more(s):
        print 'hard'
        return
    if cons(s):
        print 'hard'
        return
    print 'easy'


v = set('aeiouy')
T = ir()
for i in range(T):
    s = raw_input()
    solve(s)
