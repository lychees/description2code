t=int(input())
for i in range(t):
	wrd=str(raw_input())
	sw=sorted(set(wrd))
	lw=[wrd.count(i) for i in sw]
	tw=sorted([(i,j) for i,j in zip(sw,lw)])
	rv=[i[1] for i in tw] 
	#print(sw,tw,rv)
	res=''
	for i in sorted(set(rv)):
		for j in tw:
			if j[1]==i:
				res=res+j[0]*i
	print(res)
