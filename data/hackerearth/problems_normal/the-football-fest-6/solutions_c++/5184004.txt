#include <iostream>
#include <stdio.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n,id;
    	cin>>n>>id;
    	int stack[n],top=0;
    	stack[0]=id;
    	while(n--){
    		char c;
    		cin>>c;
    		if(c=='B') swap(stack[top],stack[top-1]);
    		else 
    		{
    			int k;
    			cin>>k;
    			stack[++top]=k;
    		}
    	}
    	printf("Player %d\n",stack[top]);
    }
    return 0;
}
