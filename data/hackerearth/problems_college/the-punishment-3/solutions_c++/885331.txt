#include <iostream>
using namespace std;
//(k-1 times 9)8(N-2*k times 9) (k-1 times 0) 1
int main()
{
    int T;long N,K,i;
    cin>>T;
    for(int j=0;j<T;j++)
    {
    	cin>>N>>K;
    	
    	if(K>N/2) K=N-K;
    	if(K==0) {cout<<0<<endl;continue;}
    	for(i=0;i<K-1;i++) cout<<9;
    	cout<<8;
    	for(i=0;i<N-2*K;i++) cout<<9;
    	for(i=0;i<K-1;i++) cout<<0;
    	cout<<1<<endl;
    	
    }
    return 0;
}
