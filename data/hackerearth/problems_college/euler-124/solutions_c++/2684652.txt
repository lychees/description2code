#include<bits/stdc++.h>

using namespace std;

#define lli long long
#define pb push_back
#define mp make_pair
#define gc getchar_unlocked
#define lld "%I64d"
#define DEBUG(x) cout<<">value ("<<#x<<") : "<<x<<endl;
#define mod 1000000007

const int lmt=100001;
vector<pair<int,int>> rad;
vector<int> prod(lmt,1);
bool isprime[lmt];

void sieve(){
    for(int i=2;i<lmt;i++){
        if(!isprime[i]){
            for(int j=i;j<lmt;j+=i){
                isprime[j]=true;
                prod[j] *= i;
            }
        }
    }
    for(int i=1;i<lmt;i++){
        rad.pb({prod[i],i});
    }
//    for(int i=0;i<=20;i++)
//        cout<<rad[i].second<<" "<<rad[i].first<<endl;
    sort(rad.begin(),rad.end());
}

int main(){
    sieve();
    int t,k;
    scanf("%d",&t);
    while(t--){
        scanf("%d",&k);
        printf("%d\n",rad[k-1].second);
    }
    return 0;
}
