def _gcd(a, b):
    if b == 0:
        return a
    return _gcd(b, a % b)


t = input()
while t > 0:
    t -= 1
    n, p, k = map(int, raw_input().split())
    if k > n:
        print "-1"
    else:
        if _gcd(n, p) == 1:
            print k
        else:
            gcdval = _gcd(n, p)
            nums = n/gcdval
            if k > nums:
                print "-1"
            else:
                k -= 1
                print k*gcdval + 1
