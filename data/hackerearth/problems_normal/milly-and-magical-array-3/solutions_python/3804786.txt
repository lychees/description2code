test=input()
for i in range(test):
	size,x=map(int,raw_input().split())
	l=map(int,raw_input().split())
	count=0
	for k in range(1,size):
		if l[k]<l[k-1]:
			diff=l[k-1]-l[k]
			if diff%x==0:
				value=diff+x
			else:
				value=diff+(x-diff%x)
			l[k]+=value
			count+=value/x
		elif l[k]==l[k-1]:
			l[k]+=x
			count+=1
	print count
