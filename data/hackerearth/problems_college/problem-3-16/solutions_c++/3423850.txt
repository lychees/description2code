#include <bits/stdc++.h>
using namespace std;

int main()
{
    int test,n,k,x;
    cin >> test;
    vector<int> vec;
    
    while(test--)
    {
    	
    	cin >> n >>k;
    	vector<int> Counter(k+1,0);
    	for(int i=1;i<=n;i++)
    	{
    		cin >> x;
    		++Counter[x];
    	}
    	sort(Counter.begin(),Counter.end());
    	int val = Counter[Counter.size()-1];
    	int ans = n - val;
    	cout <<  ans << endl;
    	Counter.clear();
    	
    }
    return 0;
}
