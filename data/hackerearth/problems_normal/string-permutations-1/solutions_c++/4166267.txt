    #include<iostream>
    #include<map>
    using namespace std;
    int main()
    {
    int t;
    cin>>t;
    int max=0;int count=1;
    map<string,int>str;
    while(t--)
    {
    string s;
    cin>>s;
    str[s]++;
    if(str[s]>max)
    {
    max=str[s];
    count=1;
    }
    else if(str[s]==max)
    {
    count++;
    }
    }
    cout<<count*max;
    }
