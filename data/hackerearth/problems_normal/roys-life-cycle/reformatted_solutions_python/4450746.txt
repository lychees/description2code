def length(string):
    l = 0
    temp = 0
    daylength = max(temp, l)
    for i in range(len(string)):
        if string[i] == 'C':
            l += 1
        else:
            daylength = max(temp, l)
            temp = daylength
            l = 0
    return max(daylength, l)

    alldaylength.append(daymax)
