#define M 1000000007

long long int a[10001000],b[1000010];

{
    long long int i,j,k,l,m[100010],rem,c,t;
    for(i=2;i<=10000000;i++)
    {
        int t=2;
        if(a[i]==0)
        while((t++)*i<=10000000)
        {
           a[i*(t-1)]=1; 
        }
    }
    j=0;
    for(i=2;i<=10000000;i++)
    {
        if(a[i]==0)
            b[j++]=i;
    }
    //for(i=1)

    for(i=2;i<=100005;i++)
        m[i]=(m[i-1]*b[i-1])%M;

    {
        cin>>k;
        cout<<m[k]<<"\n";
    }
}
