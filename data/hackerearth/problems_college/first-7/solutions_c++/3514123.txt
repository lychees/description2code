typedef long long int ll;
using namespace std;

ll ans[100005];

int main () {
    int t, n;

    cin >> t;

    ans[1] =  1;
    ans[2] =  16;

    for (ll i = 3; i <= 100000; i++) {
        ans[i] =   (i + 2 * (i - 1)) * (2 * (i) + 2 * (i - 2) )   +   (i - 2) * (i - 2) * 8    +   ans[i - 2];
    }

   while (t--) {
        scanf("%d", &n);

        printf("%0.6f\n", ans[n] / (double)((ll) n * n) );
    }

}
