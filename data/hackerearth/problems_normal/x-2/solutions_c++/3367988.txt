#include<bits/stdc++.h>
using namespace std;
#define lli long long int
#define f(i,a,b) for(i=a;i<b;i++)
#define si(a) scanf("%d",&a);
#define slli(a) scanf("%lld",&a);
#define MOD 1000000007
#define MAX 1000005
#define pb(a) push_back(a)
lli arr[MAX],sum[MAX],ans[MAX];
int main()
{
	lli i,j,k,m,n,t,count,p1,p2;
	slli(t);
	while(t--)
	{
		slli(n);
		slli(k);

		arr[0]=0;
		sum[0]=0;

		f(i,1,n+1)
		{
		slli(arr[i]);
		sum[i]=0;
		ans[i]=0;
	    }

		f(i,1,n+1)
		sum[i]=sum[i-1]+arr[i];

		p1=1;
		p2=0;

		while(1)
		{
			if(sum[p1]-sum[p2]<=k)
				ans[p1-p2]+=1;

			if(p1<=n && sum[p1]-sum[p2]<=k)
				p1++;

			else
				p2++;

			if(p1>n || p2>n)
				break;
		}
		for(i=n-1;i>=1;i--)
		{
			ans[i]=ans[i]+ans[i+1];
		}
		f(i,1,n+1)
		printf("%lld ",ans[i]);
		printf("\n");
	}
}
