#include<iostream>
using namespace std;
#define s(l) scanf("%d",&l)
#define p(l) printf("%d",l)
#define nextline cout<<endl
int main()
{
	int t;
	long long A[100001];
	cin>>t;
	while(t--)
	{
		int n,x,count=0;
		s(n);
	    s(x);
	    while(x>0)
	    {
	    	int rem;
	    	rem=x%10;
	    	x=x/10;
	    	if(rem==1||rem==4||rem==6||rem==8) count++;
	    	if(rem==9)count+=2;
	    }
	    p(count);
	    nextline;
	}
}
