#include<bits/stdc++.h>
using namespace std;

int main()
{
    int te;
    cin>>te;
    while(te--)
    {
        long int n;
        int arr[1000001];
        cin>>n;
        for(int i=0; i<n; i++)
            scanf("%d",&arr[i]);

        int limit=1, days=0;
        while(true)
        {
            sort(&arr[0], &arr[n]);
            for(int i=0; i<limit; i++)
            {
                arr[n-1-i]--;
                if(arr[n-1-i]<0)
                    goto x;
            }
            
            /*for(int i=0; i<n; i++)
                printf("%d ",arr[i]);
            cout<<endl;*/
            
            limit++;
            days++;
        }
        x:
        cout<<days<<endl;
    }
    return 0;
}
