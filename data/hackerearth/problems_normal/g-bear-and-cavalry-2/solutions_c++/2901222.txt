#include<bits/stdc++.h>
#define REP(i, n) for(int i = 0; i < (n); ++i)
using namespace std;
int w[50], h[50];
void te() {
	int n;
	scanf("%d", &n);
	REP(i, n) scanf("%d", &w[i]);
	REP(i, n) scanf("%d", &h[i]);
	sort(w + 1, w + n);
	sort(h, h + n);
	reverse(h, h + n);
	for(int i = 1; i < n; ++i)
		if(h[i] * w[i] >= h[0] * w[0]) {
			puts("NO");
			return;
		}
	puts("YES");
}
int main() {
	int z;
	scanf("%d", &z);
	while(z--) te();
	return 0;
}
