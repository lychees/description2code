#include<bits/stdc++.h>
#define f(i,n) for(int i=0;i<n;i++)

bool a[101][101][101];
int r[7]={-1,0,0,-1,0,-1,-1};
int c[7]={0,-1,0,-1,-1,0,-1};
int h[7]={0,0,-1,0,-1,-1,-1};
int main()
{   f(i,101)
     f(j,101)
      f(k,101)
      {   bool ans=true;
           f(p,7)
           {
               if((!i)&&(!j)&&(!k))
               { a[i][j][k]=false;
                  break;
               }

               int ni=i+r[p];
               int nj=j+c[p];
               int nk=k+h[p];

               if(ni>=0&&nj>=0&&nk>=0)
               {
                   ans=ans&&a[ni][nj][nk];
               }

           }

           a[i][j][k]=!ans;




      }


      int q;
     // freopen("new.txt","r",stdin);
      scanf("%d",&q);


      while(q--)
      {
          int i,j,k;
          scanf("%d%d%d",&i,&j,&k);

          if(a[i][j][k])
          printf("Ashima\n");
          else
          printf("Aishwarya\n");


      }




}
