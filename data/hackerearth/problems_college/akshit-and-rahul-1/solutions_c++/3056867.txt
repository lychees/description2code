#include <iostream>
using namespace std;

int divon(int a,int b){
	if(a>b)
	return (a/b);
	else
	return (b/a);
}

int main()
{
    int t,n=0;
    cin>>t;
    
    while(t--){
    	int a1,a2,b1,b2;
    	cin>>a1>>a2>>b1>>b2;
    	int c=0,r=0;
    	for(int i=a1;i<=a2;i++){
    		for(int j=b1;j<=b2;j++){
    			c=0;
    			int a=i;
    			int b=j;
    			if(a!=b){
    				while(divon(a,b)==1){
    					if(a>b){
    						a=a%b;
    					}
    					else{
    						b=b%a;
    					}
    					c++;
    				}
    				if(c%2==0)r++;
    			}
    		}
    	}
    	n++;
    	cout<<"Case #"<<n<<": "<<r<<endl;
    	
    }
    return 0;
}
