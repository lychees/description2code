#include<iostream>
#include<cstring>
using namespace std;
int arr[1005];
int passnext(int n,int lim)
{
    int i=n+1;
    while(arr[i]!=1)
    {
        if(i==lim)
            {i=0;}
        else
            i++;
    }
    //cout<<"R"<<' '<<i<<endl;
    return i;
}
int main()
{
    int x,y,current=0,i=0,l;ios::sync_with_stdio(false);
    string str;
    cin>>x;
    cin>>str;
    l=str.length();
    for(int i=0;i<x;i++)
        arr[i]=1;
    y=x;
    while(y!=1)
    {
    //    cout<<current<<' '<<y<<endl;
        if(str[i]=='a')
        {
            current = passnext(current,x);
        }
        else
        {
            arr[current]=0;
            current = passnext(current,x);
            y--;
        }
        if(i==l-1)
            i=0;
        else
            i++;
    }
    cout<<current+1<<endl;
}
