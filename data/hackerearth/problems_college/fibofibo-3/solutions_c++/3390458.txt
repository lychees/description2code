#include<iostream>
using namespace std;
static int zeros=0;
static int ones=0;
int fibo(int);
int main()
{

int t,n;

cin>>t;
for(int z=0;z<t;z++)
{
	cin>>n;
	fibo(n);
	cout<<zeros<<" "<<ones<<endl;
	zeros=0;
	ones=0; 
}
return 0;
}
int fibo(int n)
{

	if(n==0)
	{
		zeros++;
		return 0;
	}
	if(n==1)
	{
		ones++;
		return 1;
	}
	return fibo(n-1)+fibo(n-2);
}
