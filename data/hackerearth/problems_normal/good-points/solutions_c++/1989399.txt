#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
    ll n;
    cin>>n;
    switch(n)
    {
    	case 1:cout<<"YES\n0 0";
    	break;
    	case 2:cout<<"YES\n0 0\n1 0";
    	break;
    	case 3:cout<<"NO";
    	break;
    	case 4:cout<<"YES\n"<<"0 0\n0 1\n1 1\n1 0";
    	break;
    	default:cout<<"NO";
    }
    
    return 0;
}
