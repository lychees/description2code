#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,i;
    double p;
    cin>>n>>p;
    double ans[n+1];
    p=p/100.0;
    memset(ans,0.0,sizeof(ans));
    ans[0]=1;
    ans[1]=0;
    ans[2]=p;
    for(i=3;i<=n;i++) {
    	ans[i]=p*ans[i-2]+(1-p)*ans[i-3];
    }
    printf("%0.6f",ans[n]);
    return 0;
}
