N = raw_input()
N = (2**(int(N)))-1
driver = []
driver_lose = []
for i in xrange(N):
    x = raw_input()
    x = x.split()
    first = x[0]
    second = x[1]
    driver.append(first)
    driver.append(second)
    driver_lose.append(second)

driver = set(driver)
driver_lose = set(driver_lose)

print list(driver-driver_lose)[0]
