import math


def main():
    x = raw_input()
    q = x.split()
    n, k = int(q[0]), int(q[1])
    j = 2*k
    ans = 1e9
    while j >= 1:
        i = n - (n % j)
        ans = min(ans, abs(n-i)+abs(k-j))
        i += j
        ans = min(ans, abs(n-i)+abs(k-j))
        j = j - 1
    print(ans)


if __name__ == "__main__":
    main()
