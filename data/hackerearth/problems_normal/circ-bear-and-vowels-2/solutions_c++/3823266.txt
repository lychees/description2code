#include <bits/stdc++.h>
using namespace std;

bool check(char c){
	if(c=='a'||c=='e'||c=='i'||c=='o'||c=='u'||c=='y')
		return false;
	return true;
}

int main()
{
    int tc,cons,vow,i,len,flag;
    char str[100];
    cin>>tc;
    while(tc--){
    	cin>>str;
    	cons = 0;
    	vow = 0; len = strlen(str);
    	for(i=0;i<len;i++)
    		if(check(str[i]))
    			cons++;
    		else
    		 vow++;
    	if(cons>vow) cout<<"hard\n";
    	else{
    		flag = 0;
    		for(i=0;i+2<len;i++)
    			if(check(str[i])&&check(str[i+1])&&check(str[i+2])){
    				flag = 1;
    				break;
    			}
    		if(flag) cout<<"hard\n";
    		else cout<<"easy\n";
    	}
    }
    return 0;
}
