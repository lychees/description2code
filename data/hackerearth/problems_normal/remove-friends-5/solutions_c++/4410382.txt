#include<iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,k,cnt=0,temp;
        cin>>n>>k;
        stack<int>st;
        for(int i=0;i<n;i++)
        {
            cin>>temp;
            while((cnt<k) && (st.size()>0) && (st.top()<temp))
            {
                st.pop();
                cnt++;
            }
            st.push(temp);
        }
        while(st.size()>(n-k))
        {
            st.pop();
        }
        stack<int>st2;
        while(st.size()>0)
        {
            st2.push(st.top());
            st.pop();
        }
        while(st2.size()>0)
        {
            cout<<st2.top()<<" ";
            st2.pop();
        }
        cout<<endl;
    }
    return 0;
}
