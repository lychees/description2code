#include <iostream>
using namespace std;
static int cnt0,cnt1;
int fib(int n){
	if(n==0){
		cnt0+=1;
		return 0;
	}if(n==1){
		cnt1+=1;
		return 1;
	}
	return fib(n-1)+fib(n-2);
}
int main()
{
	int t,n;
	cin>>t;
	while(t--){
		cin>>n;
		fib(n);
		cout<<cnt0<<" "<<cnt1<<endl;
		cnt0=0;
		cnt1=0;
	}
    return 0;
}
