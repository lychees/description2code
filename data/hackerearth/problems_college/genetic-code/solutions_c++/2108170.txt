#include <iostream>
#include <string>
#include <vector>
using namespace std;

int main()
{
    string s;
    cin>>s;
    int n=s.length();
    int i=0;
    vector<int> v;
    
    while(i<n){
    	if(s[i]==s[i+1]){
    		int cnt=1;
    		while(s[i]==s[i+1] && i<n){
    			cnt++;
    			i++;
    		}
    		v.push_back(cnt);
    	}else{
    		v.push_back(1);	
    	}
    	i++;
    }
    int ans=0;
    for(int i=0;i<v.size();i++){
    	if(v[i]%2==0){
    		ans++;
    	}
    }
    cout<<ans<<endl;
    
    return 0;
}
