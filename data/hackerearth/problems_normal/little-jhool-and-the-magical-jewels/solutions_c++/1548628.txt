#include <iostream>
#include <string>
using namespace std;

int main()
{
    int T;
    cin >> T;
    while(T--)
    {
    	string str;
    	cin >> str;
    	int length = str.length();
    	int a = 0;
    	int b = 0;
    	int c = 0;
    	int d = 0;
    	int large;
    	for(int x = 0; x < length; x++)
    	{
    		if(str[x] == 'r')
    		a++;
    		else if(str[x] == 'u')
    		b++;
    		else if(str[x] == 'b')
    		c++;
    		else if(str[x] == 'y')
    		d++;
    	}
    	if(a>=b)
    	large = b;
    	else
    	large = a;
    	if(c>=d)
    	a = d;
    	else
    	a = c;
    	if(large >= a)
    	cout << a << endl;
    	else
    	cout << large << endl;
    }
    return 0;
}
