using namespace std;
#define ll long long

int a[8];
int b[8];

int first;
int second;

ll recurse(int ind, vector<int>& v)
{
    if(ind == v.size())
    {
        return 1;
    }
    ll ans = 0;
    if(first + v[ind] <= second)
    {
        first += v[ind];
        ans += recurse(ind+1,v);
        first -= v[ind];
    }
    
    second += v[ind];
    ans += recurse(ind+1,v);
    second -= v[ind];
    
    return ans;
}

int main(){
    int n;
    cin >> n;
    
    if(n==0)
    {
        cout << "1 0\nWe will win!\n";
        return 0;
    }
    
    int sum = 0;
    
    for(int i=0;i<n;i++)
    {
        cin >> a[i];
        sum += a[i];
        b[i] = i;
    }
    
    ll val = 0;
    
    for(int i=1;i<(1<<n);i++)
    {
        vector<int> v;
        for(int j=0;j<n;j++)
        {
            if(i&(1<<j))
                v.push_back(a[j]);
        }
        if(v.size() < n)
            continue;
        
        
        vector<int> vv(v.size());
        
        for(int k=0;k<v.size();k++)
            b[k] = k;
        
        do
        {
            for(int k=0;k<v.size();k++)
                vv[k] = v[b[k]];
            first = second = 0;
            val += recurse(0,vv);
        }while(next_permutation(b,b+v.size()));
    }
    
    cout << val << " " << sum << "\n";
    if(val <= sum)
        cout << "Got no way out!" << endl;
    else
        cout << "We will win!" << endl;
    
    return 0;
}
