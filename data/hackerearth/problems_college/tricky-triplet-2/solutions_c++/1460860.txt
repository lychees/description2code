#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long n;
    cin>>n;
    long long a[n];
    long long i;
    for(i=0;i<n;i++)
    cin>>a[i];
    long long j,k,s,cont=0;
    for(i=0;i<n;i++)
    {
    	for(j=i+1;j<n;j++)
    	{
    		s=a[j]-a[i];
    		for(k=j+1;k<n;k++)
    		{
    			if(a[k]-a[j]==s)
    			cont++;
    		}
    	}
    }
    cout<<cont<<endl;
    return 0;
}
