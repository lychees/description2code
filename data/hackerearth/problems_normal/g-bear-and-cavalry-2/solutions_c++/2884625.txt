#include <algorithm>
#include <cassert>
#include <cstring>
#include <iostream>

using namespace std;

#define FOR(i, a, b) for (int i = (a); i < (b); ++i)
#define REP(i, n) FOR(i, 0, n)
#define TRACE(x) cout << #x << " = " << x << endl
#define _ << " _ " <<

typedef long long llint;

const int MAX = 1001000;

int a[MAX], b[MAX];
bool bio[MAX];

int main(void) {
  int t;
  scanf("%d", &t);
  while (t--) {
    int n;
    scanf("%d", &n);
    REP(i, n) scanf("%d", &a[i]);
    REP(i, n) scanf("%d", &b[i]);
    REP(i, n) bio[i] = false;

    sort(b, b + n);
    
    int lim = a[0] * b[n-1];
    bio[n-1] = true;

    bool ok = true;
    FOR(i, 1, n) {
      int who = -1;
      for (int j = n-1; j >= 0; --j)
        if (!bio[j] && b[j] * a[i] < lim) { who = j; break; }
      if (who == -1) {
        ok = false;
        break;
      }
      bio[who] = true;
    }

    puts(ok ? "YES" : "NO");
  }
  return 0;
}
