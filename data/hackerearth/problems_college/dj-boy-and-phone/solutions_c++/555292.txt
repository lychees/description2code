#include <iostream>
#include <string.h>
#include <vector>
#include <stdio.h>
using namespace std;

#define MOD 1000000009ll
long long int dp[10001][10];

vector<int> a[10];

int main()
{
    a[0].push_back(8);

    a[1].push_back(2);
    a[1].push_back(4);

    a[2].push_back(1);
    a[2].push_back(3);
    a[2].push_back(5);

    a[3].push_back(2);
    a[3].push_back(6);

    a[4].push_back(1);
    a[4].push_back(5);
    a[4].push_back(7);

    a[5].push_back(2);
    a[5].push_back(8);
    a[5].push_back(4);
    a[5].push_back(6);

    a[6].push_back(3);
    a[6].push_back(5);
    a[6].push_back(9);

    a[7].push_back(4);
    a[7].push_back(8);

    a[8].push_back(0);
    a[8].push_back(9);
    a[8].push_back(7);
    a[8].push_back(5);

    a[9].push_back(6);
    a[9].push_back(8);
    memset(dp,0ll,sizeof(dp));
    for(int i=0;i<10;i++)
    {
        dp[1][i]=1ll;
    }
    int temp;
    for(int i=2;i<=10000;i++)
    {
        for(int j=0;j<10;j++)
        {
            temp = a[j].size();
            for(int k=0;k<temp;k++)
            {
                dp[i][j]+=dp[i-1][a[j][k]]%MOD;
                dp[i][j]%=MOD;
            }
                dp[i][j]+=dp[i-1][j]%MOD;
                dp[i][j]%=MOD;
        }

    }
    int n,t;
    cin>>t;
    long long int ans;
    while(t--)
    {
    cin>>n;

     ans=0ll;
    for(int i=0;i<10;i++)
    {
        ans+=dp[n][i]%MOD;
    }
    printf("%lld\n",ans%MOD);
    //cout<<ans<<endl;
    }
    return 0;
}
