#include<bits/stdc++.h>

#define ll long long

using namespace std;

ll pack[1009];

bool ok(ll we,ll n_p,ll m_w)
{
	ll sum=0,i;
	for(i=0;i<n_p;i++)
	{
		if(pack[i]>we)
			return false;
		else if(sum+pack[i]>we)
		{
			--m_w;
			if(m_w<=0)
				return false;
			sum=pack[i];
		}
		else
			sum+=pack[i];
	}
	return true;
}


int main()
{
	ll movers,packages,i,weight=0;
	ios::sync_with_stdio(false);
	cin>>movers>>packages;
	for(i=0;i<packages;i++)
	{
		cin>>pack[i];
		weight+=pack[i];
	}
	ll lo=1;
	ll high=weight;
	ll mid;
	while(lo<=high)
	{
		mid=(lo+(high-lo)/2);
		if(ok(mid,packages,movers))
		{
			high=mid-1;
		}
		else
		{
			lo=mid+1;
		}	
	}
	cout<<lo<<"\n";
	return 0;
}
