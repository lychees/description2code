#include <iostream>
using namespace std;
int sumofdigits(int num){
	int ans=0;
//	cout<<"sum of digits of "<<num;
	while(num){
		ans += num%10;
		num /= 10;
	}
//	cout<<"= "<<ans<<endl;
	if(ans<10) return ans;
	else return sumofdigits(ans);
}
int main()
{
	int n, len, temp=0, sum=0;
	string number;
	cin>>n;
	while(n--){
		cin>>number;
		len = number.length();
		temp=0;
		for(int i=0; i<len; i++){
			temp += number[i]-'0';
		}
		sum += sumofdigits(temp);
	}
	cout<<sumofdigits(sum)<<endl;
    return 0;
}
