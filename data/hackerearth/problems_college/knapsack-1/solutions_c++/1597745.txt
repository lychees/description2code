#include <iostream>
#include <vector>
#include <cstdio>
using namespace std;

int sum[4000];
int main()
{
	int t,n,k,a;
	cin >> t;
	while(t--){
		cin >> n >> k;
		vector< int> vec;
		for(int i = 0;i <= 3000;++i)sum[i] = 0;
		for(int i = 0;i < n;++i){
			scanf("%d",&a);
			vec.push_back(a);
			sum[a] = 1;
		}
		for(int j = 0;j <= k;++j){
			for(int i = 0;i < n;++i){
				if(j - vec[i] > 0 && sum[j - vec[i]] == 1)sum[j] = 1;
			}
		}
		int ind = 0;
		for(int i = k;i > -1;i--){
			if(sum[i] == 1){
				ind = i;break;
			}
		}
		printf("%d\n",ind);
	}
    return 0;
}
