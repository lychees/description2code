#include <bits/stdc++.h>

using namespace std;

bool arr [1000 + 10];

int main()
{
    //ios::sync_with_stdio(false);cin.tie(0);

    int n, temp;

    string s;


    cin >> n;
    cin >> s;

    temp = n;
    int i = 1;

    while(n > 1){

        for(int j = 0; j < s.size(); j++, i++){

            if(i > temp) i = 1;

            for(i; ; i++){

                if(i > temp) i = 1;
                if(arr[i]) continue;
                break;
            }

            if(s[j] == 'b'){

                n--;
                arr[i] = true;
                if(n == 1) break;
            }
        }
    }

    for(int i = 1; i <= temp; i++) {

        if(!arr[i]){

            cout << i;
            break;
        }
    }

    return 0;
}
