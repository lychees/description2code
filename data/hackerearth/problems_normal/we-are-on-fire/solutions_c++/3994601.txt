#include <iostream>
#include <vector>
#include <stack>
#include <cstdio>

#define endl "\n"

using namespace std;

int main()
{
	//ios::sync_with_stdio(false);
	int n, m, q;
	scanf("%d %d %d", &n, &m, &q);
	bool ary[n][m];

	int count = 0;
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < m; j++)
		{
			int x;
			scanf("%d", &x);
			ary[i][j] = (bool)x;
			if (ary[i][j])
				count++;
		}
	}

	//cout << count << endl;
	stack < pair <int, int> > s;
	while (q--)
	{
		int x, y;
		scanf("%d %d", &x, &y);
		x = x-1; y = y-1;

		if (ary[x][y])
		{
			vector < pair <int, int> > v;
			v.push_back(make_pair(x-1, y)); //top
			v.push_back(make_pair(x+1, y)); //bottom
			v.push_back(make_pair(x, y-1)); //left
			v.push_back(make_pair(x, y+1)); //right

			s.push(make_pair(x, y));
			ary[x][y] = false;
			count--;

			while (!s.empty())
			{
				pair <int, int> top_elem(s.top());
				v.clear();
				v.push_back(make_pair(top_elem.first-1, top_elem.second));
				v.push_back(make_pair(top_elem.first+1, top_elem.second));
				v.push_back(make_pair(top_elem.first, top_elem.second-1));
				v.push_back(make_pair(top_elem.first, top_elem.second+1));
				s.pop();

				for (int i = 0; i < v.size(); i++)
				{
					if ( ((v[i].first >= 0 && v[i].first < n) && (v[i].second >= 0 && v[i].second < m)) && ary[v[i].first][v[i].second] )
					{
						ary[v[i].first][v[i].second] = false;
						s.push(make_pair(v[i].first, v[i].second));
						count --;
					}
				}
			}
		}
		printf("%d\n", count);
	}

	return 0;
}
