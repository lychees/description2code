#include<bits/stdc++.h>
using namespace std;
int a[10000000];
int main()
{
int i,j,t,n;
for(i=0;i<10000000;i++)
    a[i]=1;
    a[0]=0;
    a[1]=0;
for(i=2;i<10000000;i++)
{

    if(a[i]==1)
    {
        for(j=2*i;j<10000000;j+=i)
            a[j]=0;
    }
}
cin>>t;
while(t--)
{
    cin>>n;
    if(n==3)
      cout<<"3"<<endl;
    else if(a[n]==1)
        cout<<"2"<<endl;
    else
        cout<<"1"<<endl;




}

return 0;
}
