typedef long long LL;

const LL N = 200000;

int a[N];
int b[N];
int n, m;

bool check(int tt)
{
    int i = 0;
    int j = 0;
    while(i < n && j < m)
    {
        if(abs(a[i]-b[j]) <= tt)
        {
            i ++;
            j ++;
        }
        else
        {
            j ++;
        }
    }

    if(i == n)
        return true;
    else
        return false;
}

{

    si(n);si(m);
    fr(i, n) {
        si(a[i]);
    }
    fr(i, m) {
        si(b[i]);
    }
    sort(a, a + n);
    sort(b, b + m);
    int prev = -1;
    int L = 0;
        int R = 2010000000;
        int mid;
        while(L < R)
        {
            mid = (L+R)/2;
            if(check(mid) == true)
                R = mid;
            else
                L = mid+1;
        }
        printf("%d\n", L);
    }
	return 0;
}
