#include <bits/stdc++.h>
using namespace std;

int main()
{
	long long i,k,sum,t;
    string s;
    cin>>t;
    while(t--)
    {
    	sum=0;
    	cin>>s;
	    for(i=0;i<s.size();i++)
	    {
	    	k=0;
	    	while(s[i]!='*' && i<s.size())
	    	{
	    		if(s[i]=='O')
	    		{
	    			k++;
	    			i++;
	    		}
	    		else if(s[i]=='X')
	    		{
	    			k=0;
	    			while(s[i]!='*' && i<s.size())
	    			{
	    				i++;
	    			}
	    			break;
	    		}
	    	}
    		sum+=k;
		}
		cout<<sum<<endl;
    }
    return 0;
}
