#include <iostream>
using namespace std;

int main()
{
 
 	int T;
 	cin>>T;
 	int x[10001];
 	x[0]=0;
 	for(int j=1;j<10001;j++)
 	{
 		x[j]=x[j-1]+j;
 	}
 	
 	for(int i=0;i<T;i++)
 	{
 		string s;
 		int N;
 		cin>>s;
 		cin>>N;
 	//	cout<<N<<" "<<x[N]<<endl;
 		int cnt=0;
 		if(x[N]%2==0)
 		{
 			for(int k=0;k<s.length();k++)
 			{
 				if(s[k]=='1')
 				cnt++;
 			}
 		}
 		else
 		{
 			for(int k=0;k<s.length();k++)
 			{
 				if(s[k]=='0')
 				cnt++;
 			}
 		}
 		cout<<cnt<<endl;
 	}
 // cout << "Hello World!" << endl;
    return 0;
}
