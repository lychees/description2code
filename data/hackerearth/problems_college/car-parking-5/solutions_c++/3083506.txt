int a[100005],i,n,ans;
string s;
int main()
{
	//freopen("i.in", "r", stdin);
  	//freopen("o.out", "w", stdout);
  	int t;
	cin>>t;
	while(t--)
	{
	    cin>>s;

	    n=s.length();
	    for(i=0;i<n;i++)
            a[i]=1;
	    for(i=0;i<n;i++)
        {
            if(s[i]=='S')
            {
                if(i-1>=0)
                    a[i-1]=0;
                if(i+1<n)
                    a[i+1]=0;
                a[i]=0;
            }
            else if(s[i]=='D')
                a[i]=0;
            else if(s[i]=='B')
            {
                if(i-1>=0)
                    a[i-1]=0;
                if(i-2>=0)
                    a[i-2]=0;
                a[i]=0;
            }
        }
        ans=0;
        for(i=0;i<n;i++)
            ans+=a[i];
        cout<<ans<<"\n";
	}
	return 0;
}
