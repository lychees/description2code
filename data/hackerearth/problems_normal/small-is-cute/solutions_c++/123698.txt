#include<iostream>
#include<vector>
#include<string>
#include<algorithm>
using namespace std;

int comp(string X,string Y)
{
    string XY=X.append(Y);
    string YX=Y.append(X);
    return XY.compare(YX)>0?0:1;
}

int main()
{
    vector<string> arr;
    long long n,i;
    string k;
    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>k;
        arr.push_back(k);
    }
    sort(arr.begin(),arr.end(),comp);
    for(i=0;i<n;i++)
    cout<<arr[i];
    return 0;
}
