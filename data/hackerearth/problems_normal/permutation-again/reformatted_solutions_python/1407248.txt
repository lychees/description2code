T = int(raw_input())

while T > 0:
    T -= 1
    N = int(raw_input())
    add_factor = 0
    permutation_sum = 0
    if N % 2 == 0:
        add_factor = N - (N / 2)
    else:
        add_factor = N - (N / 2 + 1)

    if N == 1:
        permutation_sum = 1
    else:
        permutation_sum = ((N - 1) * N / 2) - 1 + add_factor
    print(permutation_sum)
