#include <iostream>
using namespace std;
 
int getClockDist(char prev,char cur)
{
 
	if(cur>=prev)
	{
		return (cur - prev);
	}
	else
	{
		return (26 -prev +cur);
	}
}
 
 
int main()
{
	int t,i,cDist,aDist;
	string s;
	char prevChar,curChar;
	cin>>t;
	while(t--)
	{
		cin>>s;
		prevChar = 'a';
		for(i=0;i<s.length();i++)
		{
			curChar = s[i];
			
			cDist = getClockDist(prevChar,curChar);
			aDist = 26 -cDist;
			if(cDist<=aDist)
			{
				cout<<cDist<<" ";
			}
			else
			{
				cout<<-aDist<<" ";
			}
			prevChar = (prevChar-97 + cDist)%26 + 97;
		}
		cout<<endl;
	}
    return 0;
}
