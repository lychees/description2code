#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int i;
    	int count = 0,result = 0;
    	int arr[n],arr1[10]={0};
    	for(i=0;i<n;i++)
    	{
    		cin>>arr[i];
    		result = arr[i]%10;
    		if(arr1[result] == 0)
    			arr1[result]+=1;
    		else
    			count+=1;
    	}
    	cout<<count<<endl;
    }
}
