#!/usr/bin/python

import sys

t = int(raw_input())
for iteration in range(t):
	n, reqSum = map(int, raw_input().strip().split(" "))
	l = [0]*n
	index = 0
	for i in range(n):
		l[i] = int(raw_input())
	dpSum = 0
	flag = False
	for i in range(n):
		dpSum += l[i]
		if dpSum > reqSum:
			while index<n and dpSum > reqSum:
				dpSum -= l[index]
				index += 1
		if dpSum == reqSum:
			print "YES"
			flag = True
			break
	if not flag:
		print "NO"
