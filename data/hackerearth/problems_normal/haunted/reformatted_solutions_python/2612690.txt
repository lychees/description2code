N, M = map(int, raw_input().split())
arr = raw_input().split()
reqdict = {}
currage = arr[0]
currcount = 1
reqdict[currage] = currcount
print currage + " " + str(currcount)

for i in xrange(1, N):
    arri = arr[i]
    if (arri == currage):
        reqdict[currage] += 1
        currcount += 1
    else:
        if (arri not in reqdict):
            reqdict[arri] = 1
        else:
            reqdict[arri] += 1

        if (currcount == reqdict[arri]):
            if (int(currage) < int(arri)):
                currage = arri
                currcount = reqdict[arri]
        elif (currcount < reqdict[arri]):
            currage = arri
            currcount = reqdict[arri]
    print currage + " " + str(currcount)
