#include <iostream>
using namespace std;
#define ll long long int

int main()
{
 ll n,m;
 cin>>n>>m;
 ll arr[n+1];
 for(ll i=1;i<=n;i++)
 arr[i]=0;
 while(m--)
 {
 	ll a,b,k;
 	cin>>a>>b>>k;
 	for(ll i=a;i<=b;i++)
 	arr[i]+=k;
 }
 
 ll sum=0;
 for(ll i=1;i<=n;i++)
 sum+=arr[i];
 cout<<sum/n<<"\n";
    return 0;
}
