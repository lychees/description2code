#define FOR(i,a,n) for(int i=a;i<n;i++)
#define FORE(i,a,n) for(int i=a;i<=n;i++)
bool visited[100002];
int harami_tarun[100002];
vector<int> a[10003];
int n;
int tarun_is_bitch__CS(int node)
{


    if(node==n)
        return 1;
    if(visited[node]==true)
        return harami_tarun[node];
        visited[node]=true;
    int ans=0;
    FOR(i,0,a[node].size())
        ans+=tarun_is_bitch__CS(a[node][i]);
    harami_tarun[node]=ans;
    return ans;

}

        {
