#include <iostream>
using namespace std;

int main()
{
    long long int a,b,c;
    long int t,i;
    cin>>t;
    for(i=t-1;i>=t-3;i--)
    {
    	if(i%3==0)
    	{
    		a=(i*(i+3))/6;
    		break;
    	}
    }
    for(i=t-1;i>=t-5;i--)
    {
    	if(i%5==0)
    	{
    		b=(i*(i+5))/10;
    		break;
    	}
    }
    for(i=t-1;i>=t-15;i--)
    {
    	if(i%15==0)
    	{
    		c=(i*(i+15))/30;
    		break;
    	}
    }
    cout<<a+b-c<<"\n";
    return 0;
}
