#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int test;
	cin>>test;
	while(test--)
	{
		string s;
		cin>>s;
		int l=s.length();
		int i,ans=0;
		for(i=0;i<l;i++)
		{
			if((i+2)<=(l-1) && s.at(i+2)=='B')
			continue;
			if((i+1)<=(l-1) && s.at(i+1)=='B')
			continue;
			if(s.at(i)=='D')
			continue;
			if(s.at(i)=='B')
			continue;
			if(s.at(i)=='S')
			continue;
			if((i-1)>=0 && s.at(i-1)=='S')
			continue;
			if((i+1)<=(l-1) && s.at(i+1)=='S')
			continue;
			//~ cout<<i<<endl;
			ans++;
		}
		cout<<ans<<endl;
	}
	return 0;
}
