#include <iostream>
#include <utility>
#include <algorithm>
#define x1 xy1
#define y1 yx1
using namespace std;
typedef pair<int,int> pii;
int n,x,a,b,c,d;
int y=1e9;
int x1[101],x2[101],y1[101],y2[101];

bool h[101];
int getSeg()
{
	int m=-1;
	double my=-1;
	for(int i=0;i<n;++i)
	{
		if(x>x2[i]||x<x1[i]||h[i])
			continue;
		
		double cy = y1[i] + ((y2[i]-y1[i])*1.0*(x-x1[i]))/(x2[i]-x1[i]);
		if(cy<=y&&cy>my)
		{	
			m=i;
			my=cy;
		}
	}
	if(m>=0)
		h[m]=true;
	return m;
}
int main()
{
    cin>>x>>n;
    for(int i=0;i<n;++i)
    {
    	cin>>x1[i]>>y1[i]>>x2[i]>>y2[i];
    }
    while(true)
    {
    	int i=getSeg();
    	if(i==-1)
    		break;
    	x = (y2[i]<y1[i])?x2[i]:x1[i];
    	y = (y2[i]<y1[i])?y2[i]:y1[i];
    }
    cout<<x;
    return 0;
}
