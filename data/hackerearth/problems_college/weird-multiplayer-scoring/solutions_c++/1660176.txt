#include<stdio.h>
#include<string.h>
struct game
{
    char name[16];
    long int score;
}a[5000];
int main()
{
    long int i,n,m,j,max=0,c;
    scanf("%d %d",&n,&m);
   // int flag[1000]={0};
    i=0;
    while(i!=m)
    {
        scanf("%s %ld",a[i].name,&a[i].score);
        if(max<a[i].score)
            {
                max=a[i].score;
                c=i;
            }

        for(j=0;j<i;j++)
        {
            if(!strcmp(a[i].name,a[j].name))
            {
                a[i].score+=a[j].score;
                a[j].score=0;
            }
            if(max<a[i].score)
            {
                max=a[i].score;
                //flag={0};
                //flag[i]=1;
                c=i;
            }
        }
        i++;
    }
    printf("%s\n",a[c].name);
}
