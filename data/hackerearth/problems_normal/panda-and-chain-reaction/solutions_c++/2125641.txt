#include <iostream>
using namespace std;
#define MOD 1000003

long long int dp[1000006];
int main()
{long long int i;
	dp[0]=1;
	for(i=1;i<=1000003;i++)
	dp[i]=(i*dp[i-1])%MOD;
	
   int t;
   long long int n,x;
   cin>>t;
   while(t--)
   {
   	cin>>n>>x;
   x=x%MOD;
   if(n>MOD)
   n=MOD;
   cout<<(dp[n]*x)%MOD<<endl;
   }
    return 0;
}
