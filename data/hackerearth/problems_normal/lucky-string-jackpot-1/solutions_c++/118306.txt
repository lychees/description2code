int testnum;
VL nums;
LL a[70];
LL sump[70];
LL N;
void preprocess()
{
     a[0]=1;
     int i;
     for(i=1;i<100;++i)
     {
      a[i]=a[i-1]*2;                  
     }
     LL sum=0;
     for(i=1;i<100 && sum<=1e15;++i)
     {
      a[i]=a[i]*i; 
      sump[i]+=a[i]+sump[i-1];
      sum+=a[i];                
     }
     sump[i]=sum;
    // cout<<i<<" "<<sum<<"\n";
}
LL index1;
LL bs(int l,int h,LL x)
{
    int m=(l+h)/2;
    if(sump[m]<x && sump[m+1]>=x)
    {
    index1=m;
    return sump[m];
    }
    else if(sump[m]<x)
    return bs(m+1,h,x);
    else
    return bs(l,m-1,x);
        
}


void solve()
{
     LL u=bs(0,44,N);
    // cout<<"bs returned="<<u<<"\n"; 
   //  cout<<"index="<<index<<"\n"; 
     N-=u;
     if(N<=(index1+1)) 
     cout<<max(2*index1-1,N);
     else if(N<=(2*index1+1))
     cout<<N;
     else
     cout<< (2*index1+1);
     cout<<"\n";
}

 
bool input()
{
   sl(N);
   return true;
}

{

	preprocess(); 
	s(T);
	for(testnum=1;testnum<=T;testnum++)
	{
        input();
		solve();
	}
