#include <bits/stdc++.h>
using namespace std;
int main(){
	int t;
	scanf("%d",&t);
	for(int p=1;p<=t;p++){
		string a,b;
		cin>>a>>b;

		int ac = 0,bc = 0;
		for(int i=0;i<(int)a.length();i++)if(a[i]=='1')ac++;
		for(int i=0;i<(int)b.length();i++)if(b[i]=='1')bc++;


		int oz = 0;
		int zo = 0;
		int ans = 0;
		for(int i=0;i<(int)a.length();i++){
			if(a[i]=='?'){
				if(b[i]=='0'){a[i] = '0';}
				else{
					if(ac<bc){a[i]='1';ac++;}
					else a[i] = '0';
				}
				ans++;
			}
			if(a[i]=='1'&&b[i]=='0'){oz++;}
			if(a[i]=='0'&&b[i]=='1'){zo++;}
		}

		if(oz>zo){ans = -1;}
		else ans += zo;

		cout<<"Case "<<p<<": "<<ans<<endl;





	}
}
