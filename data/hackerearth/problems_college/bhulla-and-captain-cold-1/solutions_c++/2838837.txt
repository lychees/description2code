#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--)
	{
		long long n;
		cin>>n;
		long long k,a[n],b[n],c[n],i;
		for(i=0;i<n;i++)
		cin>>a[i]>>b[i]>>c[i];
		for(i=1;i<n;i++)
		{
			k=(b[i-1]<c[i-1])?b[i-1]:c[i-1];
			a[i]=a[i]+k;
			k=(a[i-1]<c[i-1])?a[i-1]:c[i-1];
			b[i]=b[i]+k;
			k=(a[i-1]<b[i-1])?a[i-1]:b[i-1];
			c[i]=c[i]+k;
		}
		if(a[n-1]<b[n-1])
		{
			if(a[n-1]<c[n-1])
			cout<<a[n-1]<<"\n";
			else
			cout<<c[n-1]<<"\n";
		}
		else
		{
			if(b[n-1]<c[n-1])
			cout<<b[n-1]<<"\n";
			else
			cout<<c[n-1]<<"\n";
		}
	}
return 0;
}
