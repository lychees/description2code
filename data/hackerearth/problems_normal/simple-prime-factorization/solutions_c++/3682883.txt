#include <bits/stdc++.h>
#define m 1000005
#define ll long long int
using namespace std;
int a[1000005]={0};
int main()
{
    for(ll i=2;i<m;i++){
    	ll j=i*i;
    	if(a[i]==0)
    		while(j<m){
    			a[j]=1;
    			j+=i;
    		}
    }
    ll t,n,temp;
    cin>>t;
    while(t--){
    	long  c=0;
    	cin>>n;
    	if(a[n]==0){
    		cout<<"2^0\n";
    		continue;
    	}
    	
    	temp=n;
    	while(temp%2==0){
    		c++;
    		temp/=2;
    	}
    	cout<<"2^"<<c;
    	
    	
    	for(ll i=3;i<n;i++){
    		if(a[i]==0){
    			temp=n;
    			c=0;
    			while(temp%i==0){
    				c++;
    				temp/=i;
    			}
    			if(c!=0)
    			cout<<"*"<<i<<"^"<<c;
    		}
    	}
    	cout<<endl;
    }
    return 0;
}
