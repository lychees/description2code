#include<bits/stdc++.h>
 
using namespace std;
 
int main()
{
	int t;
	cin>>t;
	if(t==0||t>10)
	{
		cout<<"Invalid Test";
		return 0;
	}
	while(t--)
	{
		string s;
		cin>>s;
		int l=s.length();
		if(l>100)
		{
			cout<<"Invalid Input";
			return 0;
		}
		int c=0;
		int c1=0;
		for(int i=0;i<l;i++)
		{
			if(islower(s[i]))
			c++;
			if(isupper(s[i]))
			c1++;
		}
		if(c==0&&c1==0)
		{
			cout<<"Invalid Input"<<endl;
		}
		else
		{
		int k;
		k=min(c,c1);
		cout<<k<<endl;
	   } 
	}
	return 0;
}
