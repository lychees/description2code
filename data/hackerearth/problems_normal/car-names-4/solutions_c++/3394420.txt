#include<iostream>
using namespace std;
int main()
{
    int n,i,j,p,m;
    string s;
    cin>>n;
    while(n--)
    {
        int a[26]={0};
        cin>>s;
        j=0;
        m=0;
        if(s=="carcar")
        cout<<"Not OK\n";
        else{
        for(i=0;s[i]!='\0';i++)
        {
            p=s[i]-97;
            if(a[p]==0)
            {
                j++;
            }
            if(j>3)
                break;
            a[p]++;
        }
        if(s[i]!='\0'||j!=3)
        {
            cout<<"Not OK\n";
        }
        else
        {
            j=0;
            for(i=0;i<26;i++)
            {
                if(a[i]!=0)
                {
                    if(j==0)
                        j=a[i];
                    else
                    {
                        if(j!=a[i])
                        {
                            cout<<"Not OK\n";
                            break;
                        }
                    }
                }
            }
            if(i>=26)
                cout<<"OK\n";
        }
        }
    }
    return 0;
}
