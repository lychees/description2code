#include <iostream>
#include <cstdio>
#include <algorithm>
#include <climits>
using namespace std;

#define MAX 100050
long long a[MAX];
int n, m;

bool isfeasible(long long a[], long long x)
{
    long long sum = 0;
    int count = 0;
    for (int i = 0; i < n; i++) {
        sum += a[i];
        if (sum >= x) {
            sum = 0;
            count++;
        }
        if (count == m) return 1;
    }
    return 0;
}

int main()
{
    int t;
    scanf("%d", &t);
    while (t--) {
        scanf("%d%d", &n, &m);
        for (int i = 0; i < n; i++) {
            scanf("%lld", &a[i]);
        }
        long long high = 1000000000000000000LL , low = 0, ans = 0;
        while (high >= low) {
            long long mid = low + (high - low)/2;
            if (isfeasible(a, mid)) {
                ans = mid;
                low = mid+1;
            } else {
                high = mid-1;
            }
        }
        printf("%lld\n", ans);
    }
}
