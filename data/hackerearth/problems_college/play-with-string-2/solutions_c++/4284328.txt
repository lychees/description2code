#include <bits/stdc++.h>
using namespace std;

int main()
{
    int test;
    cin >> test;
    while(test--)
    {
    	string a;
    	cin >> a;
    	int n=a.length();
    	int i=1;
    	int j=a.length()-1;
    	int bol=0;
    	while(i<n)
    	{
    		int t1=abs(a.at(i)-a.at(i-1));
    		int t2=abs(a.at(j)-a.at(j-1));
    		if(t1==t2)
    			{
    				++i;
    				--j;
    			}
    		else
    		{
    			bol=1;
    			break;
    		}
    			
    	}
    	if(bol)
    		cout << "Not Funny\n";
    	else
    		cout << "Funny\n";
    }
    return 0;
}
