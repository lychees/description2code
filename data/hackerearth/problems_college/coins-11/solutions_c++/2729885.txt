#include <iostream>
using namespace std;

int arr[100005];

void sieve() {
    
    for(long i=1; i<100005; ++i){
        arr[i] = i;
    }
    
    for(int i=2; i<318; ) {
        for(long j=i*2; j<100005; j+=i) {
            arr[j] = -1;
        }
        int found=0;
        for(int j=i+1; found==0; ++j){
            if(arr[j] != -1)
                found = j;
        }
        i=found;
    }
    
}

int main() {
	
	sieve();
	int t;
	cin >> t;
	while(t--) {
		int n;
		cin >> n;
		if(n < 4)  cout << n;
		else if( arr[n] == n)  cout << 2;
		else cout << 1;
		cout << endl;
	}
	
	return 0;
}
