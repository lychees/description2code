#include<bits/stdc++.h>
using namespace std;
int a[100000];
char s[10000];
int main()
{
           int n;
           scanf("%d",&n);
           int b;
           for( int  i=0;i<n;i++)
           {
                cin>>b;
                a[b]=1;
           }

           int m;
           cin>>m;
           gets(s);
            int ans=0;
        int num;
           while(m--)
           {
                         int i=0;
                      char c;

                        gets(s);
                        int count=0;
                        num=0;
                          while(s[i]!='\0')
                          {
                                       c=s[i];
                                       if(c==' ')
                                       {
                                                 if(a[num]==1)
                                                 {
                                                       count++;
                                                 }
                                                 num=0;
                                       }
                                       else
                                       {
                                               num=num*10+(c-'0');
                                       }



                                       i++;
                          }
                          if(num!=0)
                          {
                                if(a[num]==1)
                                {
                                     count++;
                                }
                          }

                    //     cout<<count<<endl;
                          if(count>=n)
                          {
                                 ans++;
                          }
           }
           cout<<ans<<endl;
           return 0;

}
