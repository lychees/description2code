s = raw_input()
numbers = map(int, s.split())

s = raw_input()
known_angles = map(int, s.split())

dic = set()

for angle in known_angles:
    dic.add(angle)

for i in range(numbers[0]):
    for j in range(i, numbers[0]):
        dic.add(abs(known_angles[i]-known_angles[j]))
        dic.add((known_angles[i]+known_angles[j])%360)

known_angles = list(dic)

for angle in known_angles:
    rem = angle%360
    sum = angle
    while 1:
        sum += angle
        new_rem = (sum)%360
        if new_rem==rem:
            break
        dic.add(new_rem)

s = raw_input()
given_angles = map(int, s.split())

for angle in given_angles:
    if angle in dic:
        print "YES"
    else:
        print "NO"
