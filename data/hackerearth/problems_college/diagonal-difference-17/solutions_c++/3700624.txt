#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int arr[n][n];
    long long sum=0,sum2=0;
    for(int i=0;i<n;i++)
    {	for(int j=0;j<n;j++)
    	{
    		int  temp;
    		cin>>temp;
    		if(i==j)
    			sum+=temp;
    		 if(i+j==n-1)
    			sum2+=temp;
    	}	
    }

    cout<<abs(sum-sum2);
    return 0;
}
