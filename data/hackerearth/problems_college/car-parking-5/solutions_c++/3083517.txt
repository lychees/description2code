#include <iostream>
using namespace std;

int main()
{
	int n;
	cin>>n;
	string s="";
	while(n!=0)
	{
		
		cin>>s;
		int count=0;
		for(int i=0;i<s.length();i++)
		{
			if(s[i]=='-')
			{
		    if(i==0)
		    {
		    	if(s[i]!='D'&&s[i]!='B'&&s[i]!='S'&&s[i+1]!='B'&&s[i+2]!='B'&&s[i+1]!='S'){
                     count++;
		    	}
		    }
		    else if(i==(s.length()-1))
		    {
		    	if(s[i]!='D'&&s[i]!='B'&&s[i]!='S'&&s[i-1]!='S')
		    	{
		    		count++;
		    	}
		    }
		    else
		    {
		    	if(s[i]!='D'&&s[i]!='B'&&s[i]!='S'&&s[i+1]!='B'&&s[i+2]!='B'&&s[i+1]!='S'&&s[i]!='D'&&s[i]!='B'&&s[i]!='S'&&s[i-1]!='S')
		    	{
		    		count++;
		    	}
		    }
			}
		}
     cout<<count<<endl;
	n--;
}
return 0;
}
