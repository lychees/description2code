#include <bits/stdc++.h>
using namespace std;
 
vector<int> v;
#define gc getchar_unlocked
void scanint(int &x){
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}

int main(){
    int t,i,n,res,min,x;
    scanint(t);
    while(t--){
    	v.clear();
    	scanint(n);
    	for(i=0;i<n;i++){
    	 scanint(x);
    	 v.push_back(x);
    	}
      	sort(v.begin(),v.end());
      	if(n==1)
       		printf("0\n");
      	else {
      		res=abs(v[1]-v[0]);
      		for(i=2;i<n;i++){
      			min=abs(v[i]-v[i-1]);
      			if(min<res)
      	 			res=min;
      		}
      		printf("%d\n",res);
      }
    }
    return 0;
}
