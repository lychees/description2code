#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
      int n,p,i;
      
      cin>>n>>p;
      bool flag=0;
      int a[n];
      for(int k=0;k<n;k++)
      {
         cin>>a[k];
      }     
      sort(a,a+n);
     
      for(i=0;i<n;i++)
       {
           
          if(a[i]==p)
          {
           
            flag=1;
            break;
          }
       } 
       if(flag==1&&i!=n)
       {
       cout<<"YES ";
       cout<<n-i-1<<endl;
       }
       else
       {
       cout<<"NO ";
       cout<<n-i<<endl;
       }
              
              
    }
    
    
    
    
  // system("pause");
    return 0;}
