#include<bits/stdc++.h>
using namespace std;
int count1=0;
struct trie
{
    bool end_here;
    struct trie *next[26];
};

void insert(struct trie *head,string s)
{
    for(int i=0; i<s.size(); ++i)
    {
        if(head->next[s[i]-'a']==NULL)
        {
            count1++;
            struct trie* temp;
            temp=new struct trie;
            for(int j=0; j<26; ++j)
                temp->next[j]=NULL;

            temp->end_here=0;
            head->next[s[i]-'a']=temp;
            head=temp;
        }
        else
            head = head -> next[s[i] - 'a'];
    }
    head->end_here=true;
}

int main()
{
    int n;
    scanf("%d",&n);
    struct trie* head;
    head=new struct trie;
    for(int i=0; i<26; ++i)
    {
        head->next[i] = NULL;
    }
    head -> end_here = false;
    count1=1;
    while(n--)
    {
        string s;
        cin>>s;
        insert(head,s);
    }
    cout<<count1<<endl;

}
