#include<bits/stdc++.h>
using namespace std;
int r[100000];
int root(int x){while(x!=r[x])x=r[x];return x;}
void uni(int x,int y ){r[x]=y;}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,m;cin>>n>>m;
		for(int i=1;i<=n;i++)r[i]=i;
		int ans=0;
		while(m--)
		{
			int x,y;cin>>x>>y;
			if(root(x)!=root(y))
			{
				ans++;
				uni(root(x),root(y));
			}
		}
		cout<<ans<<endl;
	}
	return 0;
}
