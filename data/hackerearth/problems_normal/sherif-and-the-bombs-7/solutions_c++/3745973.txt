/* first make the whole grid as zeros]
and then increase the count of the common occuring points again and again 
those whose count is greater than 1 will be the common points*/

#include<stdio.h>
#include<math.h>
#include<string.h>
int a[1001][1001] = {0};
int main()
{
	long long int number,count = 0,i,x,y,r,j;
	scanf("%lld",&number);
	while(number--)
	{
		scanf("%lld %lld %lld",&x,&y,&r);
		for(i = 1;i <=1000;i++)
		{
			for(j = 1;j <1001;j++)
			{
				if( (x-i)*(x-i) + (y-j)*(y-j) <= r*r)
				{
					a[i][j]++;
				}
			}
		}
	}
	count = 0;
	for(i = 0;i <=1000;i++)
	{
		for(j = 0;j <=1000;j++)
		{
			if(a[i][j]>1)
			{
				count++;
			}
		}
	}
	printf("%lld\n",count);
	return 0;
}
