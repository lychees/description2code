#include <iostream>
#include<cmath>
using namespace std;
long long max(long long  a, long long b)
{
if(a > b)
	return a;
else
	return b;
}
int main()
{
int t, j=1;
cin>>t;
while(t--)
{
int n, i;
long long  a[10001], d[10001];
cin>>n;
for(i = 0; i < n; i++)
	cin>>a[i];
d[0] = a[0];
d[1] = max(d[0], a[1]);
for(i = 2; i < n; i++)
d[i] = max(d[i-1], a[i] + d[i-2]);

cout<<d[n-1]<<endl;
}
return 0;
}
