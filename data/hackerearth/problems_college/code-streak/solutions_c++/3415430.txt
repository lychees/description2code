#include <iostream>
using namespace std ;
#include <bits/stdc++.h>
int main()
{
int t ;
cin >> t ;
while ( t--)
{
int n ;
cin >> n ;
int p[100000] = {0} ;
int a ;
int b ;
int c = 0;
int w = 0 ;
int i ;
for ( i = 0 ; i< n ; i++)
{
cin >> a >> b ;
if ( b == 1)
{
if ( p[a] == 0 )
{
c = c+1 ;
w= max ( c , w ) ;
p[a] = 1 ;
}
}
else{
c = 0;
}
}
cout << w <<endl;
}
return 0 ;
}
