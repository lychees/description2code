#include<bits/stdc++.h>
#define ll long long
using namespace std;
bool dp[1003][1024];
int arr[1003];
bool v[10003];
#define MAX 1025
void Sieve(){
   	
   for(int i=2;i<=MAX;i++)
   {
   	 if(!v[i])
   	  for(int j=i*i;j<=MAX;j+=i)
   	  {
   	     v[j]=1;	
   	  }
   }
}
int main()
{
	Sieve();
  //   freopen("in.txt","r",stdin);
         int t;
         cin>>t;
         while(t--)
         {
         	int n;
         	cin>>n;
         	for(int i=1;i<=n;i++)
         	 cin>>arr[i];
          
           for(int i=0;i<=n;i++)
		  dp[i][0]=1;
	     	for(int i=1;i<=n;i++)
		    {
		     	for(int j=0;j<1024;j++)
			    {
			     	dp[i][j]=dp[i-1][j]|dp[i-1][j^arr[i]];
			   }
		   	
		    }
    		  ll c=0;
			for(int i=2;i<=1023;i++)
			{
				if(dp[n][i])
				{
					if(!v[i])
				      c++;
			 //    cout<<i<<" ";
			    }
			 }  
            cout<<c<<"\n";			  
			  
         }
 
 
 
 
 
 
 
 
    return 0;
}
