#include<bits/stdc++.h>

using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long int n, p, i;
        cin>>n;
        long long int a[n];
        long long int sum = 0;
        for(i = 0; i < n; i++)
        {
            cin>>a[i];
        }
        sort(a,a+n);
        
        for(i = 0; i < n; i = i + 2)
        {
            p = a[i] * a[i + 1];
            sum = sum + p;
        }
        cout<<sum<<endl;
    }
    return 0;
}
