#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long n,i,flag=0,max=0,current=0;
	cin >> n;
	long long arr[n];
	for(i=0;i<n;i++)
	{
		cin >> arr[i];
		if(arr[i]>0)
			flag=1;
	}
	if(flag==0)
		cout << 0 << endl;
	else
	{
		for(i=0;i<n;i++)
		{
			current+=arr[i];
			if(current<0)
				current=0;
			if(current>max)
				max=current;
		}
		cout << max << endl;
	}
	return 0;
}
