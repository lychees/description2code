#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
bool comp(pair<int,int> c1,pair<int,int> c2)
{
	return c1.first<c2.first;
}
int main()
{
    int n;
    cin>>n;
    vector< pair<int ,int> > v;
    int h1,m1,h2,m2;
    for(int i=0;i<n;i++)
    {
    	scanf("%d:%d-%d:%d",&h1,&m1,&h2,&m2);
    	v.push_back(make_pair(h1*60+m1,h2*60+m2));
    }
    sort(v.begin(),v.end(),comp);
    bool flag =false;
    for(int i=0;i<v.size()-1;i++)
    {
    	if(v[i].second>v[i+1].first)
    	flag = true;
    }
    if(flag)
    cout<<"Will need a moderator!"<<endl;
    else
    cout<<"Who needs a moderator?"<<endl;
    return 0;
}
