#include<iostream>
using namespace std;
int main()
{
	long long int a,b,i,j,d[100000],c,k;
	cin>>j;
	for(k=0;k<j;k++)
	{
	c=0;	
	cin>>a>>b;
	for(i=0;i<a;i++)
	cin>>d[i];
	for(i=0;i<a;i++)
	{
		if(d[i]<=0)
		{
			c++;
			continue;
		}
		else
		continue;
		
	}
	if(c<b)
	cout<<"YES"<<endl;
	else
	cout<<"NO"<<endl;
}
	
}
