#include <iostream>
#include <cstring>
#include <map>
using namespace std;
map <string,int> mp;
int main()
{
    string s,s1; cin >> s >> s1;
    int n = s.length(),l1 = s1.length(),ans=0;
    for(int i=0;i<n;i++)
    {
    	for(int j=i;j<n;j++)
    	{
    		if(j-i+1<l1)
    			continue;
    		string sub = s.substr(i,j-i+1);
    		if(mp[sub]) continue;
    		mp[sub]++;
    		int l = sub.length();
    		for(int k=0;k<=l-l1;k++)
    		{
    			if(sub[k]==s1[0])
    			{
    				if(sub.substr(k,l1)==s1)
    				{
    					//cout << sub << endl;
    					ans++; break;
    				}
    			}
    				
    		}
    	}
    }
    cout << ans;
    return 0;
}
