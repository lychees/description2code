#include <iostream>
using namespace std;

int main()
{
    int t,n,i,j,ans=0;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	int a[n];
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	
    	for(i=0;i<n;i++)
    	{
    		for(j=i+1;j<n;j++){
    			if((a[i]+a[j])%3==0){
    			    ans++;}
    		}
    	}
    	cout<<ans<<endl;
    	ans=0;
    }
    
    return 0;
}
