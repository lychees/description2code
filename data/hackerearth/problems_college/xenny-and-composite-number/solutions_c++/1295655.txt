#include<bits/stdc++.h>

using namespace std;

bool arr[1000001];

void sieve(int n)
{
int i;

for(i=0;i<=n;i++)
    arr[i]=1;
    arr[1]=0;
for(i=2;i<=n;i++)
{
    if(arr[i]==0)
        continue;
    int j=2*i;

    for(;j<=n;j+=i)
    {
        arr[j]=0;
    }

}



}

int main()
{
long long t,i,j,k;

cin>>t;
sieve(100001);
while(t--)
{long long n;
cin>>n;
if(n==0)
cout<<4<<endl;
else
{i=n+1;
while(1)
{if(arr[i]==0)
break;
i++;
}
cout<<i<<endl;
}
}
}
