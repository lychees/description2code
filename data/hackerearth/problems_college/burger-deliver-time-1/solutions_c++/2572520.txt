#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	int a[n][2];
	int b[n];
	multimap<int,int> m;
	multimap<int,int>::iterator i;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<2;j++)
		{
			cin>>a[i][j];
		}
	}
	for(int i=0;i<n;i++)
	{
		b[i]=a[i][0]+a[i][1];
		m.insert(std::pair<int,int>(b[i],i+1));
	}
	for(i=m.begin();i!=m.end();i++)
	{
		cout<<i->second<<" ";
	}
	return 0;
}
