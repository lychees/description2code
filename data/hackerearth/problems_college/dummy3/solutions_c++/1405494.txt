#include<iostream>
#include<algorithm>
using namespace std;
long long  a[1001][1001];
long long  mat[1001];
long long  solve( long long  n)
{
    long long  i ,j;
    for(i = 0 ; i < n ; ++i )
      mat[i] = a[i][n-1];
    for(i = n - 2 ; i >= 0 ; --i ){
  	         mat[0] = mat[0]+ a[0][i];	
  	 for(j = 1 ; j < n ; ++j )
  		mat[j] = min( mat[j-1]+a[j][i],mat[j]+a[j][i]);
  	for(j = n-2 ; j >= 0  ; --j )
  		mat[j] = min(mat[j],mat[j+1]+a[j][i]);
	  }
//	  for(i = 0 ; i < n ; ++i )
//	  cout << mat[i] <<" ";
  	    long long mn = 2e9;
  	    for(i = 0 ; i < n; ++i ){
  	    	if(mat[i]!=0)
  	    mn = min( mat[i] , mn );
  	    }
  		  return mn;
}
int main(){
	 long long  n , i, j ;
	 cin >> n;
	   for(i = 0; i < n ; ++i ){
   		for(j = 0 ; j < n ; ++j )
   		cin >> a[i][j];
   	}
   	  cout << ( solve(n)) << endl;
   	  return 0;
}
