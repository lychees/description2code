
#include <bits/stdc++.h>

using namespace std;

#define DEBUG ON

#ifdef DEBUG
#define dout(...) cout<<__VA_ARGS__
#else
#define dout(...) do_nothing()
#endif

void do_nothing() {}



int main()
{
    int n, m;
    cin>>n>>m;

    vector<int> c(m+1);

    for(int i=1; i<=m; i++)
        cin>>c[i];
    
    int i, j, biggest;
    
    int col;
    cin>>col;
    //cout<<"col = "<<col<<endl;
    biggest = c[col];
    //cout<<"biggest = "<<biggest<<endl;

    for(j=2; j<=n; j++)
    {
        cin>>col;
        //cout<<col<<endl;
        int q = biggest/c[col];
        q++;

        for(i=col+1; i<=m; i++)
        {
            if( (c[col] * q) % c[i] == 0)
            {
                q++;
                i = col;
                continue;
            }
        }
        biggest = c[col] * q;
      //  cout<<"---"<<biggest<<endl;        
            
    }
    cout<<biggest<<endl;

    return 0;
}
