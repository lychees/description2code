#include <vector>
#include <iostream>
#include <algorithm>
#include <cstdio>
using namespace std;
int main(){
	int t,R,K;
	cin>>t;
	while(t--){
		int k;
		string s;
		cin>>R>>K;
		cin>>s;
		k = R;
		int n=s.size();
		int sum=0;
		long long int ans=0;

		vector<long long int> DP(n+1,0);
		DP[0]=1;
		for(int i=0;i<n;i++){
			if(s[i]=='R')
				sum++;
			if(sum-k>=0)
				ans+=DP[sum-k];
			DP[sum]++;
		}
		cout<<ans<<" ";
		
		n=s.size();k = K;sum=0;ans=0;
		vector<long long int> D(n+1,0);
		D[0]=1;
		for(int i=0;i<n;i++){
			if(s[i]=='K')
				sum++;
			if(sum-k>=0)
				ans+=D[sum-k];
			D[sum]++;
		}
		cout<<ans<<endl;
	}
	return 0;
}
