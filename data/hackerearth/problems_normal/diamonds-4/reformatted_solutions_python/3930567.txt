t = int(raw_input())
for i in range(0, t):
    n, m = raw_input().split(' ')
    n, m = int(n), int(m)
    arr = []
    for i in range(0, n):
        a = raw_input().split(' ')
        arr.append(a)
    count = 0
    for i in range(0, n-1):
        for j in range(0, m-1):
            if arr[i][j] == "/" and arr[i][j+1] == "\\" and arr[i +
                                                                1][j] == "\\" and arr[i+1][j+1] == "/":
                count += 1
    print count
