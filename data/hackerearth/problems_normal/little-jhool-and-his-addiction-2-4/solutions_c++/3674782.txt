#include <bits/stdc++.h>
#define ll long long 
using namespace std;
ll a[1000005] , b[1000005];
int main()
{
	ll t,n,k,i,j,dif,x;
	
    cin>>t;
    while(t--)
    {
    	dif=0;
    	cin>>n>>k;
    	x=0;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	sort(a,a+n);
    	for(i=0,j=n-1;i < j ;i++,j--)
    	{
    		b[x++] = a[i]+a[j];
    	}
    	sort(b,b+x);
    	dif = b[x-1] - b[0];
    	if(dif == k)
    	{
    		cout<<dif<<endl<<"Lucky chap!\n";
    	}
    	else
    	if(dif > k)
    	{
    		cout<<dif<<endl<<"No more girlfriends!\n";
    	}
    	else
    	{
    		cout<<dif<<endl<<"Chick magnet Jhool!\n";
    	}
    }
    return 0;
}
