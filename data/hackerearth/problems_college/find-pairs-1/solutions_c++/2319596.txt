#include<bits/stdc++.h>
using namespace std;
long long int a[1234567];
int main()
{

    long long int t,c,x,ans,i,n,j,l;
    cin>>t;
    while(t--)
    {
        ans=0;
        scanf("%lld",&n);
        for(i=0;i<n;i++)
            scanf("%lld",&a[i]);
        sort(a,a+n);
        j=a[0];
        c=0;
        for(i=0;i<n;i++)
        {
            if(a[i]==j)
                c++;
            else
            {
                ans=ans+ ((c*(c+1))/2);
                j=a[i];
                c=1;
            }
        }
        ans=ans+ ((c*(c+1))/2);
        cout<<ans<<"\n";
    }
    return 0;

}
