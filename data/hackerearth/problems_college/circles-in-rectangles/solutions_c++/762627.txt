#include <iostream>
using namespace std;

int main()
{
    int t;
    long long l,b,d,c=0,p,q;
    cin>>t;
    while(t--)
    {	c=0;
    	cin>>l>>b>>d;
    	if(d>l||d>b)
    	{
    		cout<<"0\n";
    		continue;
    	}
    	p=(l/d)%1000000007;
    	q=(b/d)%1000000007;
    	c=(p*q)%1000000007;
    	
    	cout<<c<<endl;
    }
}
