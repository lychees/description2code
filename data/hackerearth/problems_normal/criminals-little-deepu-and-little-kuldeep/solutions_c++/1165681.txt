#include<iostream>
#include<cstdio>

#include<algorithm>

using namespace std;
int
main ()
{

  int a[100002], i, n, j, k, ans, prev_ans, t;

  scanf ("%d", &t);
  while (t--)
    {
      prev_ans = ans = 1;   //base case
      scanf ("%d", &n);
      for (i = 0; i < n; i++)
    {
      scanf ("%d", &a[i]);

    }
      sort (a, a + n);
      for (i = 0; i < n - 1; i++)
    {
      if (a[i] == a[i + 1])
        ans++;      //count the number of times a particular value is repeated.
      else          //when you find a mismatch try to update the count. (prev_ans)
        {
          if (ans > prev_ans)
        prev_ans = ans; //update the previous value if current value is greater
          ans = 1;      //reset the answer counter to base case
        }

    }
      if (ans > prev_ans)   //corner case where i/p is  2 2 2 2 2 , the it won't go to else part and prev_ans wont be updated.(this can be worked around)
    prev_ans = ans;
      cout << prev_ans << endl;
    }
  return 0;
}
