#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;
int main() {
	long long int t, r, c, n;
	cin >> t;
	while (t--) {
		long long int ans1 = 0, ans2;
		cin >> r >> c;
		if (r & 1) {
			n = (r + 1) / 2;
			ans1 = (n * (2 + (n - 1) * 2))/2;
			if (c > r) {
			//	cout << 1 << " " << ans1 << endl;
			}
			else if (c & 1) {
				n = (r - c) / 2;
				ans1 -= (n*(4 + (n - 1) * 2))/2;
			//	cout << 2 << " " << ans1 << endl;
			}
			else {
				n = (r - c + 1) / 2;
				ans1 -= (n * (2 + (n - 1) * 2))/2;
			//	cout << 3 << " " << ans1 << endl;
			}
			r--;
			n = r / 2;
			ans2 = (n *(4 + (n - 1) * 2))/2;
			if (c > r) {
			//	cout << 4 << " " << ans2 << endl;
			}
			else if (c & 1) {
				n = (r - c + 1) / 2;
				ans2 -= (n*(2 + (n - 1) * 2)) / 2;
				//cout << 5 << " " << ans2 << endl;
			}
			else {
				n = (r - c) / 2;
				ans2 -= (n*(4 + (n - 1) * 2)) / 2;
			//	cout << 1 << " " << ans2 << endl;
			}
			cout << ans2 << " " << ans1 << endl;
		}
		else {
			n = r / 2;
			ans2 = (n *(4 + (n - 1) * 2))/2;
			if (c > r) {
			//	cout << 4 << " " << ans2 << endl;
			}
			else if (c & 1) {
				n = (r - c + 1) / 2;
				ans2 -= (n*(2 + (n - 1) * 2))/2;
		//		cout << 5 << " " << ans2 << endl;
			}
			else {
				n = (r - c) / 2;
				ans2 -= (n*(4 + (n - 1) * 2))/2;
		//		cout << 1 << " " << ans2 << endl;
			}
			r--;

			n = (r + 1) / 2;
			ans1 = (n * (2 + (n - 1) * 2)) / 2;
			if (c > r) {
			//	cout << 1 << " " << ans1 << endl;
			}
			else if (c & 1) {
				n = (r - c) / 2;
				ans1 -= (n*(4 + (n - 1) * 2)) / 2;
			//	cout << 2 << " " << ans1 << endl;
			}
			else {
				n = (r - c + 1) / 2;
				ans1 -= (n * (2 + (n - 1) * 2)) / 2;
			//	cout << 3 << " " << ans1 << endl;
			}
			cout << ans2 << " " << ans1 << endl;
		}
	}
	return 0;
}
