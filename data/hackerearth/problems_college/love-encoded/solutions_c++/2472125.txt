#include<stdio.h>
void decode (char msg[50],int k)
{
	char tmp;
	int tmp1,i;
	for(i=0;msg[i]!=NULL;i++)
	{
		if((tmp=msg[i]-k)<'A')
			{
			tmp1='A'-tmp;
			tmp='Z'-tmp1+1;
			}
		printf("%c",tmp);
	}
	printf("\n");
}
int main()
{
int i,n,k[10];
char msg[10][50];
scanf("%d",&n);
for(i=0;i<n;i++)
	{
	scanf("%d",&k[i]);
	scanf("%s",&msg[i]);
	}
for(i=0;i<n;i++)
	decode(msg[i],k[i]);
}
