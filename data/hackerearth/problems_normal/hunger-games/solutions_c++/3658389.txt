#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main(){
	int N, num;
	vector<int>vi;
	cin>>N;
	for(int i=0;i<N;i++){
		cin>>num;
		vi.push_back(num);
	}

	sort(vi.begin(), vi.end());
	int ans=0;

	for(int i=0;i<vi.size()-2;i++)
		ans = max(ans, vi[i+2]-vi[i]);


	cout<<ans;
return 0;
}
