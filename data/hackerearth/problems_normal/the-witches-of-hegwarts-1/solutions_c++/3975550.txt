#include<bits/stdc++.h>
#include<vector>
#include<list>
#include<utility>
#include<map>
#include<set>
#include<stack>
#include<queue>
#define mod 1000000007
#define ll long long
#define INF (1<<28)
using namespace std;
int main()
{
	int T;
	cin>>T;
	while(T--)
	{
		ll N;
		scanf("%lld",&N);
		set <ll> vis;
		vis.clear();
		vis.insert(N); 
		queue< pair<ll,ll> > q;
		q.push(make_pair(N,0));ll ans=0;
		while(!q.empty())
		{
			pair <ll,ll> v=q.front();
			q.pop();
		//	cout<<v.first<<"-"<<v.second<<"<--ele and its dist"<<endl;
			
			if(v.first==1)
			{
				ans=v.second;
				break;
			}
			ll X=v.first-1;
			ll Y=(v.first)/2;
			ll Z=(v.first)/3;
			set <ll>::iterator it;
			it=vis.find(X);
			if(it == vis.end())
			{
				q.push(make_pair(X,v.second+1));
				vis.insert(X);
			}
			
			if(v.first%2 == 0)
			{	
				it=vis.find(Y);
				if(it == vis.end())
				{
					q.push(make_pair(Y,v.second+1));
					vis.insert(Y);
				}	
			}
			if(v.first%3==0)
			{
				it=vis.find(Z);
				if(it == vis.end())
				{
					q.push(make_pair(Z,v.second+1));
					vis.insert(Z);
				}
			}
		}
		printf("%lld\n",ans);		
	}
	
	
	return 0;
}
