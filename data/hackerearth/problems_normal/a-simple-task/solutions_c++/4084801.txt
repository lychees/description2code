#include <iostream>
using namespace std;
int main()
{
    int m,flag,a[100],temp;
    cin>>m;
    for(int i=0;i<m;i++)
    cin>>a[i];
    for(int j=2;j<=400000000;j++)
    {
    	flag=1;
    	temp=a[0]%j;
    	for(int k=1;k<m;k++)
    	{
    		if(temp==a[k]%j)
    		continue;
    		else
    		{
    			flag=0;
    			break;
    		}
    	}
    	if(flag==1)
    	cout<<j<<" ";
    }
    return 0;
}
