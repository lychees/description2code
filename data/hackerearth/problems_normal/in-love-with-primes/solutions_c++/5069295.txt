#include <bits/stdc++.h>
using namespace std;

bool prime[1000000];

void sieve(){
	memset(prime,true,1000000);
	prime[0]=prime[1]=false;
	for(int i=2;i*i<=1000000;i++){
		if(prime[i]){
			for(int j=2;i*j<=1000000;j++){
				prime[i*j]=false;
			}
		}
	}
}
int main()
{
    int tc;
    int n;
    sieve();
    cin>>tc;
    
    while(tc--){
    	cin>>n;
    	int flag=0;
    	//cout<<n<<endl;
    	for(int i=2;i<=n;i++){
    		//cout<<i<<" "<<n-i<<endl;
    		if(prime[i] && prime[n-i]){
    			flag=1;
    			break;
    		}
    	}
    	if(flag)    cout<<"Deepa"<<endl;
        else        cout<<"Arjit"<<endl;
    }
    return 0;
}
