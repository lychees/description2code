#include<bits/stdc++.h>
using namespace std;

long long int mod_pow(long long int a, long long int b,long long int mod)
{
	long long int ans=1;
	while(b)
	{
		int r=b%2;
	
		if(r!=0)
		ans=(ans*a)%mod;
		
		a=(a*a)%mod;
		
		b/=2;
	}	
	return ans;
}

long long C(int n, int r)
{
    vector< vector<long long> > C(n+1,vector<long long> (r+1,0));
 
    for (int i=0; i<=n; i++)
    {
        for (int k=0; k<=r && k<=i; k++)
            if (k==0 || k==i)
                C[i][k] = 1;
            else
                C[i][k] = (C[i-1][k-1] + C[i-1][k]);
    }
    return C[n][r];
}

long long gcd(long long a,long long b)
{
	if(a==0)
	return b;
	return gcd(b%a,a);	
}

long long min1(long long a,long long b)
{
	return a>b?b:a; 	
}

int main()
{
	long long int t;
	scanf("%lld",&t);
	for(;t>0;t--)
	{
		long long int n,k;
		scanf("%lld %lld",&n,&k);
		
		long long int a[n];
		for(int i=0;i<n;i++)
		{
			scanf("%lld",&a[i]);
		}
		
		long long int curr_sum = a[0], s = 0, i,f=0;
 
    for (i = 1; i <= n; i++)
    {
        
        while (curr_sum > k && s < i-1)
        {
            curr_sum = curr_sum - a[s];
            s++;
        }
 
        if (curr_sum == k)
        {
           f=1;
           break;
        }
 
        if (i < n)
          curr_sum = curr_sum + a[i];
    }
		
		if(f==1)
		printf("YES\n");
		else
		printf("NO\n");
	}
		
	return 0;
}
