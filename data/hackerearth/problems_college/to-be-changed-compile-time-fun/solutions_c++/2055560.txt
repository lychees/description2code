#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long a,b,c,d,i,j,sum=0;
    vector<int> ar;
    scanf("%lld",&a);
    for(i=0;i<a;i++){
    	scanf("%lld",&c);
    	ar.push_back(c);
    }
    long long x=a-1;
    for(i=0;i<a-2;i++){
    	long long max=-1;
    	for(j=1;j<x;j++){
    		if(ar[j-1]*ar[j+1]>max){
    			max=ar[j-1]*ar[j+1];
    			b=j;
    		}
    	}
    	x--;
    	sum+=max;
    	ar.erase(ar.begin()+b);
    }
    cout<<sum<<endl;
    return 0;
}
