#include <iostream>
#include<queue>
using namespace std;
int a[1030][1030];
bool vis[1030][1030]={false};
queue<int> x,y;
int main()
{
    int n,sum,ma=0,m,f;
    cin>>n;
    for(int i=0;i<n;i++)
     for(int j=0;j<n;j++)
       cin>>a[i][j];
       
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
        {sum=0;
        	if(vis[i][j]==false)
         {if(a[i][j])
           {x.push(i);
            y.push(j);
            sum++;
            while(!x.empty())
            { m=x.front();
              f=y.front();
              x.pop();
              y.pop();
              vis[m][f]=true;
              if(vis[m-1][f]==false)
              if(a[m-1][f])
               {x.push(m-1);
                y.push(f);
                vis[m-1][f]=true;
                sum++;
               }
               
               if(vis[m+1][f]==false)
               if(a[m+1][f])
               {x.push(m+1);
               y.push(f);
               vis[m+1][f]=true;
               sum++;
               }
               
               if(vis[m][f-1]==false)
               if(a[m][f-1])
               {x.push(m);
               y.push(f-1);
               vis[m][f-1]=true;
               sum++;
               }
               if(vis[m][f+1]==false)
               if(a[m][f+1])
               {x.push(m);
               y.push(f+1);
               vis[m][f+1]=true;
               sum++;
               }
            }
            ma=max(ma,sum);
           }
         }
        }
        cout<<ma;
            
    return 0;
}
