#include <iostream>
#include<string>
using namespace std;
 
int main()
{
    int t;
    cin>>t;
    while( t--)
    {
    	string str1,str2;
    	long int i=0,count=0;
    	cin>>str1>>str2;
    	while(str1[i] && str2[i])
    	{
    	if( str1[i]==str2[i] )
    	count ++;
    	i++;
    	}
    	
    	
    	cout<<count<<endl;
    }
    return 0;
}
