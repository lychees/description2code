#include <stdio.h>
#include <vector>
#include <algorithm>
#define h second

using namespace std;

vector <pair <long long,long long> > v;

long long mem[202][202],t,n,r,h;
long long k[1];

long long maximum (long long p, long long q)
{
	if (p > q)return p;
	return q;
}

long long func (long long l,long long max)
{
	if (l <= 0) return 0;
	else if (mem[l][max] == -1)
	{
		if (v[l-1].h < v[max].h && v[l-1].first < v[max].first)
		{
			long long a = v[l-1].h + func(l-1,l-1);
			long long b = func(l-1,max);
			return (mem[l][max] = maximum(a,b));
		}
		else
		{
			return mem[l][max] = func(l-1,max);
		}
	}
	else return mem[l][max];
}

int main ()
{
	scanf ("%lld",&t);
	while (t--)
	{
		k[0] = 100000000002;
		scanf ("%lld",&n);
		for (long long i = 0 ; i <= 201 ; i++)
			for (long long j = 0 ; j <= 201 ; j++)
				mem[i][j] = -1;
		
		for (long long i = 0 ; i < n ; i++)
		{
			scanf ("%lld %lld",&r,&h);
			v.push_back (pair<long long,long long>(r,h));
		}
		v.push_back (pair<long long,long long> (k[0],k[0]));
		sort (v.begin(), v.end());

		long long ans;
		ans = func(n,n);
		v.clear();

		printf ("%lld\n",ans);
	}

	return 0;
}
