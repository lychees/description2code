#include <bits/stdc++.h>
using namespace std;

int main() 
{
    int t,n,i,max;
    cin>>t;
    while(t--) {
        max=-1;
        cin>>n;
        int b[10005]={0};
        int a[n];
        for(i = 0; i < n; i++) {
            cin>>a[i];
            b[a[i]]++;
            if(b[a[i]] > max) {
                max = b[a[i]];
            }
        }
        cout<<max<<endl;
    }
	return 0;
}
