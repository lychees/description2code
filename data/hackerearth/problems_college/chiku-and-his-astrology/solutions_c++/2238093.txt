#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    string s;
    stack <char> q;
    cin>>s;
    int count=1;
    q.push(s[0]);
    for(int i=1;i<s.length();i++)
    {
        count++;
        if(q.empty())
        {   
                q.push(s[i]);
        }
        else if(q.top()==s[i])
        {
            q.push(s[i]);              
                        }
        else{
            q.pop();
            count-=2;
        }

    
    }
    cout<<count<<"\n";
    }

}
