#include <bits/stdc++.h>
using namespace std;

int main()
{
    int tc,num,sum;
    cin>>tc;
    while(tc--)
    {
    	sum=num=0;
    	string str;
    	cin>>str;
    	for(int i=0;i<str.size();i++)
    	{
    	if(str[i]!='a'&&str[i]!='A'&&str[i]!='b'&&str[i]!='B'&&str[i]!='c'&&str[i]!='C')
    	{
    		sum=sum+(num*(num+1))/2;
    		num=0;
    	}
    	else
    	{
    		num++;
    	}
    	}
    	 sum=sum+(num*(num+1))/2;
    	cout<<sum<<endl;
    }
}
