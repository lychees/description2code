#include <iostream>
using namespace std;
int arr[100005];
int cmp(const void *a,const void *b)
{
	return (*(int *)a) - *((int *)b);
}
int main()
{
    int t;
    int n;
    cin>>t;
    for(;t;t--)
    {
    	long long n,sum=0,sum1=0,sum2;
    	cin>>n;
    	for(int i=1;i<=n;i++)
    	{
    	cin>>arr[i];	
    	if(arr[i]>0)
    	  sum=(sum+arr[i])%1000000007;
    	else 
    	  sum=(sum-arr[i])%1000000007;
    	}
    	qsort(arr+1,n,sizeof(int),cmp);
    	for(int i=1;i<=((n/2)+(n%2));i++)
    	{
    	  sum1 = (sum1-arr[i])%1000000007;
    	}
    	for(int i=((n/2)+(n%2))+1;i<=n;i++)
    	{
    	  sum1 = (sum1+arr[i])%1000000007;
    	}
    	sum2=((sum1%1000000007)*(sum%1000000007))%1000000007;
    	cout<<sum2%1000000007<<endl;
    }
}
