#include <iostream>
#include <vector>
using namespace std;

int main()
{
    vector <int> v;
    int N,Q,i,t,x,y,s,N1;
    cin>>N>>Q;
    N1=N;
   while(N--)
    {
    	cin>>i;
    	v.push_back(i);
    	
    }
   // int k=v.size();
   
    while(Q--)
    {
    	 bool found=false;
    	cin>>t;
    	if(t==0)
    	{
    		cin>>x>>y;
    		v[x-1]=y;
    	}
    	else 
    	{
    		cin>>s;
    		for(int i=0;i<N1;i++)
    		{
    			if(v[i]>=s)
    			{
    				cout<<i+1<<"\n";
    				found=true;
    				break;
    			}
    		}
    		if(!found)
    		cout<<"-1\n";
    		
    		
    	}
    }
    return 0;
}
