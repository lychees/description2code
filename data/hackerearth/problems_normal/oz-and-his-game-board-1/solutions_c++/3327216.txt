#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t;
    cin>>t;
    long long int n,m,a,b,c,sum1,sum2;
    while(t--){
        cin>>n>>m>>a>>b>>c;
        sum1=0,sum2=0;
        int k=0;
        sum1 = n*a + m*b;
         if(n>=m){
                k = n-m;
                sum2 = (n+m-k)*c + k*a;
            }
            else {
                k = m-n;
                sum2 = (n+m-k)*c + k*b;
            }
        cout<<max(sum1,sum2)<<endl;
    }
    //cout << "Hello world!" << endl;
    return 0;
}
