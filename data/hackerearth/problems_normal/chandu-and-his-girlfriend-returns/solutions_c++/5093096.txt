#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int i,j, n,m,k;
    	cin>>n>>m;
    	long long a[n+1],b[m+1],c[n+m];
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	for(i=0;i<m;i++)
    	cin>>b[i];
    	a[n]=-999999;
    	b[m]=-999999;
    	i=0;j=0;
    	for(k=0;k<n+m;k++)
    	{ 
    		if(a[i]>=b[j])
    		c[k]=a[i++];
    		else 
    		c[k]=b[j++];
    	 }
    	 for(k=0;k<m+n;k++)
    	 cout<<c[k]<<" ";
    	 cout<<endl;
    }
    return 0;
}
