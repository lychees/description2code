#include <iostream>
#include <cstdio>
#include <set>
using namespace std;

int main()
{
    set<int>my_set;
    int n,m;
    scanf("%d%d",&n,&m);
    for(int i = 0;i<n;i++)
    {
    	int k;
    	scanf("%d",&k);
    	my_set.insert(k);
    }
    int max = 0;
    int i =1;
    while(1)
    {
    	if(my_set.count(m*i) == 1)
    	{
    		max+=1;
    		i = i+1;
    	}
    	else break;
    }
    i--;
    printf("%d\n",(m*i));
    return 0;
}
