#include <iostream>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	string str;
   	int a[26]={0},i,count;
   	cin>>str;
   	for(i=0;str[i];i++)
   	a[str[i]-97]+=1;
   	count=0;
   	int d=a[0];
   	for(i=0;i<26;i++)
   	{
   	if(a[i]==d)
   	count++;
   	}
   	if(count==26)
   	cout<<"Yes"<<endl;
   	else
   	cout<<"No"<<endl;
   	
   }
    return 0;
}
