#include<iostream>
using namespace std;
int main()
{
    int T,X,a,b,i;
    cin>>T;
    while(T--)
    {
        cin>>X;
        a=X;
        b=0;
        while(a>=0)
        {
            if(a%3==0 && b%5==0)
                break;
 
            a--;
            b++;
        }
        if(a==-1 && b==X+1)
            cout<<"-1\n";
        else
        {
            i=0;
            while(i<a)
            {
                cout<<"5";
                i++;
            }
            i=0;
            while(i<b)
            {
                cout<<"3";
                i++;
            }
            cout<<"\n";
        }
    }
    return 0;
}
