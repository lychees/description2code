#include <iostream>
using namespace std;

int main()
{
    int t,a[26],i,n,sum;
    char b[100000];
    cin>>t;
    while(t--)
    { sum=0;
    	cin>>n;
    	cin>>b;
    	for(i=0;i<26;i++)
    	{
    		a[i]=0;
    	}
    	for(i=0;i<n;i++)
    	{
    		a[b[i]-97]++;
    		if(sum<a[b[i]-97]){sum=a[b[i]-97];}
    	}
    	if(sum%2==1)cout<<"No"<<endl;
    	else cout<<"Yes"<<endl;
    }
    return 0;
}
