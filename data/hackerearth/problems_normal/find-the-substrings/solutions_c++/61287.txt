using namespace std;


int main()
{
    ios::sync_with_stdio(false);
    ll t,i,j,n,len,mx,a,b,c;
    string str;
	vector<ll> va,vb,vc;
    ll sum,temp;
    cin>>t;
    for(i=0;i<t;++i)
    {
        va.clear();
        vb.clear();
        vc.clear();
        sum=0;
        cin>>str;
        len=str.length();
        for(j=0;j<len;++j)
        {
            if(str[j]=='a')
                va.pb(j);
            else if(str[j]=='b')
                vb.pb(j);
            else
                vc.pb(j);
        }
        va.pb(len);
        vb.pb(len);
        vc.pb(len);
        a=0;
        b=0;
        c=0;
        for(j=0;j<len;++j)
        {
            mx=max(max(va[a],vb[b]),vc[c]);
            sum+=len-mx;
            if(va[a]==j)
                ++a;
            else if(vb[b]==j)
                ++b;
            else if(vc[c]==j)
                ++c;
        }
        temp=(len*(len+1))/2;
        cout<<temp-sum<<endl;
    }
    return 0;
}
