#include <iostream>
using namespace std;

int main()
{
    int long long i,j,suml,sumr,sum,temp,t,n,ar[100000],flag;
    cin>>t;
    while(t--)
    {
    	cin>>n;
        temp=0;
        flag=0;
    	suml=sumr=sum=0;
    	for(i=0;i<n;i++)
    	{
    		cin>>ar[i];
    		sum=sum+ar[i];
    	}
    	for(i=0;i<n;i++)
    	{
    		temp=temp+ar[i];
    		suml=temp-ar[i];
    		sumr=sum-temp;
    		if(suml==sumr)
    		{
    			flag=1;
    			break;
    		}
    	}
    	if(flag)
    	   cout<<"YES\n";
    	else
    	   cout<<"NO\n";
    }
    return 0;
}
