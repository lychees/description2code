#include<iostream>

using namespace std;

int main()
{
    long long m,i,j,k,a[200],term,c;
    int num,count,n;
    cin >> n;
    for(num = 1;num<=n;num++){
         count = 0;
         for(i=2;i<=num/2;i++){
             if(num%i==0){
                 count++;
                 break;
             }
        }

         if(count==0 && num!= 1){


        a[0]=1;
        m=1;

        for(j=1,c=0;j<=num;j++)
        {
           for(k=0;k<m;k++)
           {
               term=(a[k]*j+c);
                c=term/10;
               a[k]=term%10;
           }
           while(c>0)
           {
               a[m]=c%10;
               c=c/10;
               m++;
           }


        }

        for(k=m-1;k>=0;k--)
        {
            cout<<a[k];
        }
        cout<<endl;
         }

    }
}
