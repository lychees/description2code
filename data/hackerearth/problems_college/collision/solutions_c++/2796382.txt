#include <map>
#include <cstdio>
#include <iostream>

using namespace std;

int input[111];

int Process(int N)
{
    map<int,int> hash;
    int ans = 0;
    for(int i = 0; i < N; ++i)
    {
        if(hash[ input[i] ] == 1)
            ++ans;
        else
            hash[input[i]] = 1;
    }
    return ans;
}

int main()
{
    int T,N;
    cin >> T;
    while(T--)
    {
        cin >> N;

        for(int i = 0; i < N; ++i)
            cin >> input[i];

        cout << Process(N) << endl;
    }
    return 0;
}
