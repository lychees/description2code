#include <iostream>
using namespace std;

int main()
{
	int t,x,y,n;
    cin>>t;
    while(t--)
    {
    	cin>>x>>y;
    	if(y == x)
    	{
    		if(x % 2 == 0)
    			cout<<(x/2)*3 + (x/2)<<endl;
    		else
    			cout<<(x/2)*3 + (x/2) + 1<<endl;
    	}
    	else if(y == x - 2)
    	{
    		x = x-2;
    		if(x % 2 == 0)
    			cout<<(x/2)*3 + (x/2) + 2<<endl;
    		else
    			cout<<(x/2)*3 + (x/2) + 3<<endl;
    	}
    	else
    		cout<<"No Number"<<endl;
    }
    return 0;
}
