#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;

typedef long long ll;

int main()
{
	ll t,m,n,z,ans,i,j;
    cin>>t;
    while(t--){
    	ans=0;
    	cin>>z>>m>>n;
    	if(m>n) swap(m,n);
    	for(i=1;i<=sqrt(z);i++){
    		if(z%i) continue;
    		else{
    			j=z/i;
    			if(i<=m && j<=n) ans++;
    			if(i<=n && j<=m && j!=i) ans++;
    		}
    	}
    	cout<<ans<<'\n';
    }
    return 0;
}
