#include <bits/stdc++.h>
#define asrt(n, a, b) assert(n >= a && n <= b)
using namespace std;
int main(){
	int n, i;
	cin >> n;
	asrt(n, 2, 1000000);
	int D[n], B[n];
	for(i = 0; i < n; i++){
		cin >> D[i];
		asrt(D[i], -1000000000, 1000000000);
	}
	for(i = 0; i < n; i++){
		cin >> B[i];
		asrt(B[i], -1000000000, 1000000000);
	}
	int Dd, Bd, Dm = 0, Bm = 0;
	for(i = 0; i < n-1; i++){
		Dd = abs(D[i] - D[i+1]);
		Bd = abs(B[i] - B[i+1]);
		if(Dd > Dm){
			Dm = Dd;
		}
		if(Bd > Bm){
			Bm = Bd;
		}
	}
	if(Dm == Bm){
		cout << "Tie";
	}else if(Dm > Bm){
		cout << "Dom" << endl;
		cout << Dm;
	}else{
		cout << "Brian" << endl;
		cout << Bm;
	}
	return 0;	
}
