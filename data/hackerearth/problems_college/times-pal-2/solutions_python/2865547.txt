t=int(raw_input())
while t:
    a=raw_input()
    b=raw_input()
    cnt=0
    res1=a+b
    res2=res1[::-1]
    if(res1==res2):
        cnt=cnt+1
    res1=b+a
    res2=res1[::-1]
    if(res1==res2):
        cnt=cnt+1
    for i in range(1,len(a)):
        res1=a[0:i]+b+a[i:]
        res2=res1[::-1]
        if(res1==res2):
            cnt=cnt+1
    print cnt
    t=t-1
