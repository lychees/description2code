n = int(raw_input())
arr = [int(i) for i in raw_input().split()]
arr.sort(reverse=True)
cnt = 0
while True:
    cnt = 0
    m = 100000
    for i in xrange(n):
        if arr[i]:
            if arr[i] < m:
                m = arr[i]
    for i in xrange(n):
        if arr[i]:
            arr[i] -= m
            cnt += 1
    if cnt == 0:
        break
    print cnt
