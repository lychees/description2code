#include <bits/stdc++.h>
using namespace std;
#define lli long long int
int main() {
	int t;
cin>>t;
while(t--)
{
int n;
cin>>n;
lli arr[n];
for(int i=0;i<n;i++)
{
cin>>arr[i];
}
sort(arr,arr+n);
lli count=1;
for(int i=0;i<n-1;i++)
{
if(arr[i]!=arr[i+1])
{
count++;
}
}
cout<<n-count<<endl;
}
	return 0;
}
