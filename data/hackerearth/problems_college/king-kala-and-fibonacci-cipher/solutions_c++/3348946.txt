#include <bits/stdc++.h>
using namespace std;

int cmp(const void *a,const void *b)
{
	return *(int *)b-*(int *)a ;
}

int main()
{
    // cout << "Hello World!" << endl;
    char arr[100001] ;int f,s,z,i;
    f=0;s=1;
    arr[0]='a';
    arr[1]='b' ;
    for(i=2;i<100001;i++)
   {     z=(f+s)%26;
   	      arr[i]=97+z;
   	      f=s;
   	      s=z;
   } 

     int tc;
     int a,b,j; char str[101];int freq[26] ;
     cin>>tc;
     while(tc--) 
     {    for(i=0;i<26;i++)
                freq[i]=0;
      
     	cin>>a>>b ;
     	scanf("%s",str);
     	for(i=a;i<=b;i++)
     	{  
     		freq[arr[i]-'a']++;
     	}
     
         char alp[26] ;
         for(i=0;i<26;i++)
         alp[i]='a'+i ;
         
     	for(i=0;i<26;i++)
     	  for(j=0;j<26-i-1;j++)
     	    {
     	    	if(freq[j]<freq[j+1])
     	    	{
     	    		int temp;
     	    		temp=freq[j+1] ;
     	    		freq[j+1]=freq[j] ;
     	    		freq[j]=temp;
     	    		char tmp;
     	    		tmp=alp[j+1];
     	    		alp[j+1]=alp[j];
     	    		alp[j]=tmp;
     	    	}
     	    }
     	
     	
     	int len=strlen(str) ;
     	
     	for(i=0;i<len;i++)
     	{
     		str[i]=alp[str[i]-'a'];
     	}

     }
