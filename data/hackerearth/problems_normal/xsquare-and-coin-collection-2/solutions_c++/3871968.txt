long long int coins(int a[], int N, int K)
{
    long long int coin[N];
    for(int i = 0; i < N; i++)
    {
            coin[i] = 0;
    }
    
    if(a[0] <= K)
            coin[0] = a[0];
    for(int i = 1; i < N; i++)
    {
            if(a[i] <= K)
            {
                       if(coin[i-1] != 0)
                                    coin[i] = coin[i-1]+a[i];
                       else
                           coin[i] = a[i];
            }
    }
    long long int max = 0;
    for(int i = 0; i < N; i++)
    {
            if(coin[i] > max)
                   max = coin[i];
    }
    
    /*for(int i = 0; i < N; i++)
    {
            printf("%d ",coin[i]);
    }*/
    return max;
}
    
int main()
{
    int T, N, K;
    scanf("%d",&T);
    for(int tc = 1; tc <= T; tc++)
    {
            scanf("%d%d",&N,&K);
            int a[N];
            for(int i = 0; i < N; i++)
            {
                    scanf("%d",&a[i]);
            }
            printf("%lld\n",coins(a,N,K));
    }
  //  getch();
    return 0;
}
