#include <bits/stdc++.h>
using namespace std;
const int maxn = 1000001;
int arr[maxn];
long long int ans[maxn] = {0};
void siv(){
	arr[1] = arr[0] = 0;
	for(int i=2;i<=sqrt(maxn); i++){
		if(arr[i] == 1){
			for(int j = i*i;j<maxn;j +=i)
				arr[j] = 0;
		}
	}
	
	for(int i=1;i<maxn;i++){
		if(arr[i] == 1){
			//cout << i << "\n";
			ans[i] = ans[i-1] + i; 
		
			
		}
		else{
			ans[i] = ans[i-1];
		//	cout << ans[i] << "\n";
		}
	
	}
		
	
}
 
int main(){
  for(int i=0;i<maxn;i++) arr[i] = 1;
  siv();
 
  int n; 
  cin>>n;
  while(n--){
  	int l,r; 
  	cin>>l>>r;
     cout<<ans[r] - ans[l-1]<<endl;
  }
}
