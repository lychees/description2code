#include<iostream>
#include<cstdio>
#include<algorithm>

using namespace std;

int get_n(int n, int d, int r){
	if (n%d==0) return 0;
	int dig = 0;
	int counter = 0;
	if (n<d)  n*=10;
	while(counter < r){
		dig = n/d;
		n = (n%d)*10;
		counter++;
	}
	return dig;
}

int main(){

	int t,n, d, r;
	scanf("%d", &t);
	while(t--){
		scanf("%d %d %d", &n, &d, &r);
		printf("%d\n", get_n(n, d, r));
	}	
	return 0;
}
