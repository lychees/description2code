#include<iostream>
#include<cstdio>
#include<string.h>
using namespace std;
int main()
{
    int test_cases, count, count1, i, k;
    char j;
    cin>>test_cases;
    getchar();
    while(test_cases>0)
    {
    	count=0;
    	count1=0;
        char str[100000];
        gets(str);
        for(i=0; str[i]!='*'; i++)
        {
        	if(str[i]>96 && str[i]<123)
        	count++;
        }
        for(i=strlen(str); i>0; i--)
        {
        	if((str[i]>=97 && str[i]<=122) || str[i]=='*')
        	{
        		count1++;
        		
        	}
        	if(count1==count)
        	k=i-1;
        }
        j=str[k];
        printf("%c\n", j);
        test_cases--;
    }
    return 0;
}
