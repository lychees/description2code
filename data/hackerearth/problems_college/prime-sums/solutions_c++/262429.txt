#include <iostream>
using namespace std;




int main()
{
   bool b[4000001]={0};
   long long int sum=0;
   int n;
   cin>>n;
   for(int i=2;i<n; )
   {  
      if(b[i]==0)
       sum+=i;
      
      
   	  for(int j=2;j*i<n;j++ ) 
       b[j*i]=1;
      
      while(b[++i]==1);
      
    }
    
    cout<<sum<<endl;
    
    return 0;
   
}
