#include<bits/stdc++.h>
using namespace std;
long long int gcd(long long int a, long long int b)
{
    if(b==0)
        return a;
    else
        return gcd(b,a%b);
}
bool f(long long int a, long long int b)
{
    if(a<=b)
        return 0;
    else
        return 1;
}
int main()
{
    int d;
    cin >> d;
    while(d--)
    {
        long long int x,y,k;
        cin >> x >> y >> k;
        long long int g=gcd(x,y);
        vector<long long int >v;
        long long int i=1;
        for(;i*i<=g;i++)
        {
            if(g%i==0)
            {
                v.push_back(i);
                if(i!=g/i)
                    v.push_back(g/i);
            }
        }
        sort(v.begin(),v.end(),f);
        if(v.size()<k)
            cout << "No crymeth today" << endl;
        else
            cout << v[k-1] << endl;
    }
    return 0;
}
