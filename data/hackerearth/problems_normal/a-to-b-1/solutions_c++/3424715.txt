const ll mod=1e9+7;

ll a[100005],st[100005<<2];

ll powmod(ll base, ll exp) {
	ll res=1;
	while(exp>0) {
		if(exp&1) res=(res*base)%mod;
		base=(base*base)%mod;
		exp>>=1;
	}
	return res%mod;
}

void build(ll i, ll begin, ll end){
	if(begin==end) st[i]=a[begin];
	else{
		ll mid=(begin+end)>>1;
		build(i<<1,begin,mid);
		build((i<<1)+1,mid+1,end);
		st[i]=(st[i<<1]*st[(i<<1)+1])%mod;
	}
}

void update(ll i, ll begin, ll end, ll pos, ll val){
	if(begin==end) st[i]=val;
	else{
		ll mid=(begin+end)>>1;
		pos<=mid?update(i<<1,begin,mid,pos,val):update((i<<1)+1,mid+1,end,pos,val);
		st[i]=(st[i<<1]*st[(i<<1)+1])%mod;
	}
}

ll query(ll i, ll begin, ll end, ll l, ll r){
	if(l>r) return 1;
	if(begin>=l&&end<=r) return st[i];
	ll mid=(begin+end)>>1;
	if(r<=mid) return query(i<<1,begin,mid,l,r);
	else if(l>mid) return query((i<<1)+1,mid+1,end,l,r);
	else return (query(i<<1,begin,mid,l,mid)*query((i<<1)+1,mid+1,end,mid+1,r))%mod;
}
