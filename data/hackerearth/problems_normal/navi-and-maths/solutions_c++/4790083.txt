#include <iostream>
#include<algorithm>
using namespace std;
#define MAX 1000000007

int comp(int x,int y){
	return (y<x);
}

long long power(long long x, long long n){
	if(n==0)
	 return 1;
	else if(n==1)
	 return x;
	long long mid=power(x,n/2);
	
	if(n%2==0)
	 return (mid*mid)%MAX;
	else 
	 return (((mid*mid)%MAX)*x)%MAX;
}

int main()
{
	int tests,n;
	long long int p_max,p;
	cin>>tests;
	for(int k=1;k<=tests;k++){
		cin>>n;
		long long int a[n];
		p_max = 1;
		for (int i=0;i<n;i++){
			scanf("%lld",&a[i]);
		}
		int length = power(2,n);
		for(int i=0;i<length;i++){
			long long int sum = 0;
			long long int product = 1;
			for(int j=0;j<n;j++){
				if(i&(1<<j)){
					sum = (sum + a[j]);
					product *= a[j]%MAX;
					product = product%MAX;
				}
			}
			p = ((product%MAX)*((power(sum,MAX-2))%MAX))%MAX;
			p_max= (p>p_max)?p:p_max;
		}
		printf("Case #%d: %lld\n",k,p_max);
	}
    
    return 0;
}
