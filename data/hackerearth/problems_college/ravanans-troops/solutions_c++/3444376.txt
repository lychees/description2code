#include <bits/stdc++.h>

using namespace std;

#define s(x) scanf("%d", &x)
#define sl(x) scanf("%lld", &x)
#define mp make_pair
#define pb push_back
#define ll long long int
#define mod 1000000007

int a[1000005];

int main()
{
    int i, j, ans;
    for (i = 1; i <= 1000000; ++i) {
        ans = INT_MAX;
        for (j = 1; j*j <= i; ++j)
            ans = min(ans, a[i-j*j]);
        a[i] = ans+1;
    }
    int t, n;
    s(t);
    while (t--) {
        s(n);
        assert(1 <= n && n <= 1000000);
        printf("%d\n", a[n]);
    }
    return 0;
}
