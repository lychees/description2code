#include <iostream>
#include <string>
#include <cstring>

using namespace std;

int main()
{
    int t;
    cin >> t;
    
    while(t--)
    {
    string s1,s2;
    cin >> s1 >> s2;
    int l = min(s1.length(),s2.length());
    if(l!=0)
    {
    	int c = 0;
    for(int i = 0; i < l ;i++ )
     c += (s1[i]==s2[i]?1:0);
     cout << c << "\n";
    }
    else
     cout<<"0\n";
    }
    return 0;
}
