#include <iostream>
#include <algorithm>
using namespace std;

int main(){
	int t;
	cin>>t;
	while(t--){
		int n,x,y;
		cin>>n>>x>>y;
		int a[n],b[n],i;
		for(i=0;i<n;i++)
			cin>>a[i]>>b[i];
		sort(a,a+n);
		sort(b,b+n);
		int cost=0;
		for(i=0;i<n;i++){
			
			if(a[i]>b[i])
				cost=cost+(a[i]-b[i])*y;
			
			if(a[i]<b[i])
				cost=cost+(b[i]-a[i])*x;
			
		}
		
		cout<<cost<<endl;
	}
	
    return 0;
}
