#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int T;
    cin >> T;
    while(T--){
    	int N,Count;
    	string s1,s2;
    	cin >> s1 >> s2 >> N;
    	Count=0;
    	for(int i=0; i<s1.size(); i++){
    		if(s1[i]!=s2[i]) Count++;
    	}
    	if(Count<=N && (N-Count)%2==0) cout << "Yes" << endl;
    	else cout << "No" <<  endl;
    }
    return 0;
}
