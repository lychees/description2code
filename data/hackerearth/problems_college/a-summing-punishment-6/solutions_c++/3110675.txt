#include <iostream>
using namespace std;
long long int sum(long long int A[],long long int a,long long int b)
{
long long int i,s=0;
for(i=a-1;i<b;i++)
{
	s+=A[i];
}
return s;
}
int main()
{
    long long int n;
    cin>>n;
    long long int i,A[n+1];
    for(i=0;i<n;i++)
    {
    	cin>>A[i];
    }
    long long int t;
    cin>>t;
    long long int a,b;
    while(t--)
    {
    	cin>>a>>b;
    	cout<<sum(A,a,b)<<endl;
    }
    return 0;
}
