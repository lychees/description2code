#include<bits/stdc++.h>
using namespace std;
static int a[10000000];
int main()
{
register int i,j,ctr=0,t,m,n;
//for(i=0;i<10000000;i++)
  //  a[i]=1;
    a[0]=1;
    a[1]=1;
for(i=2;i<10000000;i++)
{

    if(a[i]==0)
    {
        for(j=2*i;j<10000000;j+=i)
            a[j]=1;
    }
}
cin>>t;
while(t--)
{
    long int su=0;
    cin>>m>>n;
    if(m==2 || m== 1)
    {
        for(i=3;i<=n;i+=2)
          if(a[i]==0)
            su++;
            //if(m==1)
              //  su++;
            cout<<su<<endl;
    }
    else
        cout<<"0"<<endl;


}
return 0;
}
