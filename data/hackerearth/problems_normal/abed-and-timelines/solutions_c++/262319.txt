#include <iostream>
using namespace std;

int main()
{
    int t;
    long int N,M,x1,y1,x2,y2,p,ans,dx,dy; 
    cin>>t;
    while(t--)
    {
    	cin>>N;
    	cin>>M;
    	cin>>x1;
    	cin>>y1;
    	cin>>x2;
    	cin>>y2;
    	cin>>p;
    	dx=abs(x1-x2);
    	dy=abs(y1-y2);
    	ans=min(dx,N-dx)+min(dy,M-dy);
    	ans*=p;
    	cout<<ans<<"\n";
    	if(ans<=1000)
    		cout<<"You saved the group.\n";
    	else
    		cout<<"Let go of the group.\n";
    }
    return 0;
}
