#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t,k;
	char s[100];
    cin>>t;
    while(t--)
    {
    	cin>>s;
    	int l=strlen(s);
    	int i=0;
    	while(i<l)
    	{
    		char s1[100];
    		int j=0;
    		int flag=1;
    		while(i<l && s[i]!='*')
    		{
    			if(s[i]=='x') flag=0;
    			s1[j]=s[i];
    			i++;
    			j++;
    			
    		}
    		s1[j]='\0';
    		if(flag) cout<<s1; 
    		else for(k=0;k<j;k++) if(s1[k]=='x') cout<<'x';
    		if(s[i]=='*') cout<<'*';
    		i++;
    	}
    	cout<<endl;
    }
    return 0;
}
