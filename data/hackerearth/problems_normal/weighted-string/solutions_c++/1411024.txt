#include<iostream>
#include<cstdio>
#define ll long long
#include<cstring>
#include<algorithm>
#include<cmath>
using namespace std;
char str[1000001];
int main()
{
	ll t,l,i,k,count,j,tmp;
	scanf("%lld",&t);
	while(t--)
	{
		count=0;
		scanf("%lld",&k);
		scanf("%s",&str);
		l=strlen(str);
		for(i=0;i<l;i++)
		{
			tmp=str[i]-'a'+1;
			if(tmp==k)
				count++;
			for(j=i+1;j<l;j++)
			{
				tmp+=str[j]-'a'+1;
				if(tmp==k)
					count++;
				if(tmp>k)
					break;
			}
		}
		cout<<count<<endl;
	}
	return 0;
}
