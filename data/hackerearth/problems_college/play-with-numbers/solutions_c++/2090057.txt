#include <iostream>

using namespace std;

int main()
{
	long long i,j,x,t,y,z,cnt;
    cin>>t;


    while(t--)
    {
    	cin>>z>>x>>y;

        cnt=0;
        for(i=1;i*i<=z;i++)
        {
        	if(z%i==0 )
        	{
        		if(i<=x && z/i<=y)
        		cnt++;
        		
        		if(z/i !=i && z/i <=x && i<=y)
        		cnt++;
        	}
        }
        cout<<cnt<<'\n';

    }
    return 0;
}
