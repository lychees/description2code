vector<pair<int,int> > v;

int dp[100005][26];
int main()
{
	int n,a,tmp,b,c,x,y,t,i,j,l,r;
	int fib[100005];
	fib[0]=0, fib[1]=1;
	for(i=2;i<=100001;i++)
		fib[i] = (fib[i-1]+fib[i-2])%26;
	for(i=1;i<=100001;i++)
	{
		for(j=0;j<26;j++)
			dp[i][j] += dp[i-1][j];
		dp[i][fib[i-1]]++;
	}
	s(t);
	while(t--)
	{
		int f[26];
		v.clear();
		s(l)s(r);
		char str[105];
		scanf("%s", str);

		for(j=0;j<26;j++)
		{
			v.pb(mp(dp[r][j]-dp[l-1][j],j));
		}
//		for(it=v.begin();it!=v.end();it++)
//			printf("%d_%c\n", it->first, it->second+97);
		sort(v.begin(),v.end());

		int prev=it->first;

		{
		//	printf("jhjhj\n");
			it--;
			while(it->first == prev)
				it--;
			itt=it+1;
			while(itt!=v.end() && itt->first == prev)

			prev=it->first;
		}
		for(i=0;str[i]!='\0';i++)
			printf("%c", f[str[i]-97]+97);
		printf("\n");
	}
	return 0;
}
