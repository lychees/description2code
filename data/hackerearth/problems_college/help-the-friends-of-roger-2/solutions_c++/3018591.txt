#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    char a[110];
    int m,l,r,i,j,c;
    
    scanf("%s",a);
   	cin>>m;
   	
   	for(i=0;i<m;i++){
   		cin>>l>>r;
   		c=0;
   		for(j=l-1;j<r-1;j++){
   			if(a[j]==a[j+1])
   			c++;
   		}
   		cout<<c<<endl;
   	}
    return 0;
}
