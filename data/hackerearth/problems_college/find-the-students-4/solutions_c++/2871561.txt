#include <iostream>
#include <algorithm>
#include <map>
using namespace std;

map<int,int> mp;

int main()
{
    //cout << "Hello World!" << endl;
    int n,cnt;
    cin>>n;
    map<int,int> :: iterator it ;
    for(int i=0;i<n;i++)
    {
    	cin>>cnt;
    	it = mp.find(cnt);
    	if(it != mp.end() )it->second += 1;
    	else mp.insert(make_pair(cnt,1));
    }
    cout<<n<<endl;
    for(it = mp.begin();it!=mp.end();it++)
    {
    	n = n - it->second;
    	if(n!=0)cout<<n<<endl;
    }
    mp.clear();
    return 0;
}
