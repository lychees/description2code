#include<iostream>
#include<cstring>
#include<cmath>
using namespace std;
int main(){
    int t;
    cin>>t;
    for(int i=0;i<t;i++){
        string s;
        cin>>s;
        int n=s.length();
        bool check=true;
        for(int j=0;j<n/2;j++){
            int x=abs(s[j+1]-s[j]);
            int y=abs(s[n-1-j]-s[n-2-j]);
            if(x!=y){
                check=false;
            break;
        }   
    }         
          if(check==true){
              cout<<"Funny"<<"\n";
         }    
              else{ cout<<"Not Funny"<<"\n";
          }      
            
    }
    
  
    return 0;
}
