t = int(raw_input())
for i in range(t):
    a, b = [int(x) for x in raw_input().split()]
    xor = a
    if a % 2 == 0:
        xor = 1
        num = (b-a-1) % 4
        for j in range(b-num+1, b+1):
            xor = xor ^ j
    else:
        num = (b-a-1) % 4
        for j in range(b-num, b+1):
            xor = xor ^ j
    print(xor)
