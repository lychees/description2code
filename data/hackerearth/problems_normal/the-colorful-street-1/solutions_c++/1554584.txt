#include <iostream>
#define INF 9999999

using namespace std;

int min_cost(int n);
int min_cost_core(int i, int n, int prev_color);

int price[100][3];
int cache[100][3];
int solution[100];

int main()
{
        int i, j;
        int T;
        int n;
        cin>>T;
        while(T--)
        {
                cin>>n;
                for(i=0; i<n; i++)
                {
                        for(j=0; j<3; j++)
                        {
                                cin>>price[i][j];
                        }
                }

                cout<<min_cost(n)<<endl;
        }
        return 0;
}

int min_cost(int n)
{
        return min_cost_core(0, n, 3);
}

int min_cost_core(int i, int n, int prev_color)
{
        if(i==n)
                return 0;

        int ans=INF;

        for(int j=0; j<3; j++)
        {
                if(j!=prev_color)
                {
                        ans=min(ans, price[i][j]+min_cost_core(i+1, n, j));
                }
        }

        return ans;
}
