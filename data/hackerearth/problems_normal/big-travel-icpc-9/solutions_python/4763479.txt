n = int(raw_input())
x = [None]*n
y = [None]*n

for i in range(n):
	(x[i], y[i]) = map(int, raw_input().strip().split(' '))

ans=0
x.sort()
y.sort()
sumx=0
sumy=0
for i in range(1, n):
	sumx = ((abs(x[i]-x[i-1])*i)%1000000007 + sumx)%1000000007 
	sumy = ((abs(y[i]-y[i-1])*i)%1000000007 + sumy)%1000000007 
	ans =(ans+sumx+sumy)%1000000007 
print ans
