#include<iostream>
#include<string.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){

		cin>>a;
		for(int i=0;i<a.length();i++){

			cout<<"2";

			cout<<"3";
			if(a[i]=='g'||a[i]=='h'||a[i]=='i')
			cout<<"4";
			if(a[i]=='j'||a[i]=='k'||a[i]=='l')
			cout<<"5";
			if(a[i]=='m'||a[i]=='n'||a[i]=='o')
			cout<<"6";
			if(a[i]=='p'||a[i]=='q'||a[i]=='r'||a[i]=='s')
			cout<<"7";
			if(a[i]=='t'||a[i]=='u'||a[i]=='v')
			cout<<"8";
			if(a[i]=='w'||a[i]=='x'||a[i]=='y'||a[i]=='z')
			cout<<"9";
		}
		cout<<endl;
	}
	return 0;
}
