#include <iostream>
#include <algorithm>
using namespace std;
long long a[100001];
int main()
{
    
    
    long long int t,n,k,i,j,j0,c,f,ansf,ansn; 
    scanf("%lld",&t);
    while(t--){
           scanf("%lld%lld",&n,&k);
    	for(int i=0;i<n;i++)
           scanf("%lld",&a[i]); 
    	
    	sort(a,a+n);
    ansf=-1;
    	for( j = 0;j<n;j++){
    	
    	while(a[j]==a[j+1]&&j<n-1)
    	  j++;
    		
    		
    	   c = 0;
    	   f = 0;
    		
    		for(int i = j;i>=0;i--){
    			if (c+(a[j]-a[i])>k) 
    			break;
    			c += a[j]-a[i];
    			f ++;    			
    		}
    		if (f>ansf){
    			ansf = f;
    			ansn = a[j];
    		}
    	}
    	printf("%lld %lld\n",ansf,ansn);
    }
    return 0;
}
