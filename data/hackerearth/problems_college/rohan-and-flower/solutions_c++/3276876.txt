#include<bits/stdc++.h>
using namespace std;
# define ll long long
ll ar[2222];
int main()
{
    cin.sync_with_stdio(false);
    int n,i,pos=0,x,j;
    ll ans=0,tmp,k;
    cin>>n>>k;
    for(i=0;i<n;i++)
    cin>>ar[i];
    for(i=0;i<n;i++)
    for(j=i+1;j<n;j++)
    {
    	tmp=ar[i]*ar[j];
    	if(tmp%k==0)
    	{
    		ans+=(n-j-1);
    		continue;
    	}
    	for(x=j+1;x<n;x++)
        if(ar[i]*ar[j]*ar[x]%k==0)
        ans++;
    }
    cout<<ans;
}
