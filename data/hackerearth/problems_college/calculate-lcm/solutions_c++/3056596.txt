#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    int t;
    long int a,b,i,j,k=1,dv,dd,c,lcm;
    cin>>t;
    while(t--)
    {lcm=0;
    	cin>>a>>b;
    	for(i=1;i<=a;i++)
    	{
    		for(j=1;j<=b;j++)
    		{k=2;
    			while(k*k<=min(i,j))
    			{
    				if(j%(k*k)==0&&i%(k*k)==0)
    				break;
    				k++;
    			}
    			if(k*k<=min(i,j))
    			continue;
    			dv=min(i,j);
    			dd=max(i,j);
    			while(dv>0)
    			{
    				c=dv;
    				dv=dd%dv;
    				dd=c;
    			}
    			lcm+=(i*j)/dd;
       		}
    	}
    	cout<<lcm%(int)pow(2,30)<<"\n";
    }
    return 0;
}
