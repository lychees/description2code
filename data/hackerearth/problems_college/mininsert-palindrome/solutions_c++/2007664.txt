#include<iostream>
#include<string.h>
#include<stdio.h>
using namespace std;
int table[6101][6101];
char str[6101];
int max(int a,int b)
{ if(a>b)
   return a;
  else
   return b;
}
void lcs()
{ int l=strlen(str);
  int i,j;
  for(i=0;i<=l;i++)
    table[i][0]=0;
  for(i=0;i<=l;i++)
    table[0][i]=0;
  for(i=1;i<=l;i++)
   { for(j=1;j<=l;j++)
      { if(str[i-1]==str[l-j])
          table[i][j]=table[i-1][j-1]+1;
        else
          table[i][j]=max(table[i-1][j],table[i][j-1]);
      }
   }
} 
int main()
{ long t;
  int l;
  cin>>t;
  int i,j;
  while(t--)
 {  cin>>l;
  scanf("%s",str);
  lcs();
  printf("%ld\n",l-table[l][l]); 
 }  
 return 0;
}
