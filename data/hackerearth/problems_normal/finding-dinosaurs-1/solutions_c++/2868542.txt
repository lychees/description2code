#include<bits/stdc++.h>
#define ll long long int
#define inp(x) scanf("%lld",&x)
#define inpc(x) cin >> x
using namespace std;
map< vector< ll > , string > m;
vector< ll > t;
int main(){
	ll n,k,l,q,x;
	string s;
	inp(n);inp(k);inp(l);inp(q);
	while(n--){
		inpc(s);
		t.clear();
		for(ll i=0;i<k;++i){
			inp(x);
			t.push_back(x);
		}
		m[t]=s;
	}
	while(q--){
		t.clear();
		for(ll i=0;i<k;++i){
			inp(x);
			t.push_back(x);
		}
		if(m.find(t)!=m.end()){
			cout<<m[t]<<endl;
		}else{
			cout<<"You cant fool me :P"<<endl;
		}
	}
	return 0;
}
