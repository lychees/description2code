#include <iostream>
#include <string>
#include <algorithm>
#include <functional>
#include <vector>
#include <map>
#include <set>
#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <cmath>
#include <climits>
using namespace std;

typedef unsigned int uint;
typedef long long ll;
typedef unsigned long long ull;
typedef long double ld;

#define MAX 1003
#define MOD 1000000007

int main(int argc, const char * argv[]) {
	long long C[MAX][MAX];
	for(int i = 0; i < MAX; i++) {
		for(int j = 0; j <= i; j++) {
			if(j == 0 || j == i) {
				C[i][j] = 1;
			} else {
				C[i][j] = (C[i - 1][j - 1] + C[i - 1][j]) % MOD;
			}
		}
	}

	int t;
	cin >> t;
	while(t--) {
		int n, p;
		cin >> n >> p;

		long long ans = 0;
		for(int i = 0; i <= p; i++) {
			ans = (ans + C[n][i]) % MOD;
		}
		cout << ans << endl;
	}
    return 0;
}
