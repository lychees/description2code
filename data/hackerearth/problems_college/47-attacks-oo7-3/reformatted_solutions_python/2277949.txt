testCase = int(input())
while testCase > 0:
    n = int(input())
    m = 1
    if n > 120:
        print("UNABLE TO FIRE")
    else:
        for i in range(1, n+1):
            if (i % 2 != 0):
                m *= 3
            else:
                m += 2
        print(m)
    testCase = testCase-1
