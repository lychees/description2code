#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
	int i,j,t,n,p;
    int dp[1001][1001];
    dp[0][0]=1;
    for(i=1;i<1001;i++){
    	dp[i][0]=1;
    	dp[0][i]=0;
    }
    
    for(i=1;i<1001;i++){
    	for(j=1;j<=i;j++){
    		dp[i][j]=(dp[i-1][j]%(int)(pow(10,9)+7)+dp[i-1][j-1])%(int)(pow(10,9)+7);
    	}
    }
    //cout<<dp[1][1]<<endl;
    
    cin>>t;
    for(i=0;i<t;i++){
    	cin>>n>>p;
    	int sum=0;
    	for(j=0;j<=p;j++){
    		sum=(dp[n][j]%(int)(pow(10,9)+7)+sum%(int)(pow(10,9)+7))%(int)(pow(10,9)+7);
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
