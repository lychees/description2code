#include <iostream>
using namespace std;

int main()
{
    int n,i,j,k;
    cin>>n;
    k=1;
    for(i=1;i<=n;i++)
    {
         cout<<"*";
    }
    cout<<endl;
    for(k=2;k<=n-1;k++)
    { 
    	for(j=n;j>=1;j--)
    	{
    		if(j==k)
    		{
    			cout<<"*";break;
    		}
    	else
    	{
    	   cout<<" ";	
    	}
    }
    cout<<endl;
    }
    k=n;
    for(i=1;i<=n;i++)
    {
         cout<<"*";
    }
   return 0;
    
}
