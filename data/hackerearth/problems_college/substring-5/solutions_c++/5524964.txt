#include <iostream>
using namespace std;

int main()
{
    string s;
    cin>>s;
    int i,j,n=s.size();
    long long int c=0;
    for(i=1;i<=n;i++){
    	for(j=0;j<=n-i;j++){
    		string a = s.substr(j,i);
    		int f=0;
    		for(int k=1;k<a.size();k++){
    			if(a[0]!=a[k]){
    				f=1;
    				break;
    			}
    		}
    		if(f==0)c++;
    	}
    }
    cout<<c<<endl;
    return 0;
}
