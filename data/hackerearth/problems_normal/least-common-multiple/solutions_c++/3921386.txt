#include<bits/stdc++.h>
#define ll long long int
#define mod 1000000007
#define MAX 10000109
using namespace std;

bool fun(ll n)
{
    ll i;
    for(i=2; i*i<=n; i++)
    {
        if(n%i==0)
        {
            while(n%i==0)
                n/=i;
            if(n==1)
                return true;
            else
                return false;
        }
    }
    return true;
}

int main()
{
    ll t,n,m,i,j,ans;
    while(scanf("%lld",&n)!=EOF)
    {
        m=n/2;
        for(i=m+1; i<=n; i++)
        {
            if(fun(i))
                break;
        }
        cout<<i<<endl;
    }
}
