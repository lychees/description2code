#include<iostream>
using namespace std;
int check(long n)
{
	if(n==0)
		return 1;
	if(n<0)
		return 0;
	return (check(n-6)||check(n-9)||check(n-20));
}
int main()
{
	int t,x;
	long n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		x=check(n);
		if(x==1)
			cout<<"True"<<endl;
		else
			cout<<"False"<<endl;
	}
	return 0;
}
