#include<iostream>
#include<cstdio>
using namespace std;
int main(){
	int i,j,k,l,r,flag=0;
	char a[10000];
	scanf("%s",a);
	scanf("%d",&k);
	while(k--){
		flag=0;
		scanf("%d%d",&l,&r);
		for(i=l-1;i+1<=r-1;i++){
			if(a[i]==a[i+1]){
				flag++;
			}
		}
		printf("%d\n",flag);
		
	}
	return 0;
}
