#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    string s;
    for(int k=0;k<t;k++){
    	cin>>s;
    	if(s.length()<=3){
    		cout<<s<<endl;
    		continue;
    	}
    	string an;
    	int r = s.length()-4;
    	int ptr=s.length()-1;
    	while(ptr>r && ptr>=0){
    		an+=s[ptr];
    		ptr--;
    	}
    	while(ptr>=0){
	    	an+=',';
    		r = ptr-2;
    		while(ptr>r && ptr>=0){
    			an+=s[ptr];
    			ptr--;
    		}	
    	}
		for(int j=an.length()-1;j>=0;j--) cout<<an[j];
		cout<<endl;
    	
    }
}
