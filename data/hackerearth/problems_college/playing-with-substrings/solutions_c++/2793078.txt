#include <iostream>
#include <cstdio>
#include <string>
using namespace std;
int main()
{
    string a,b;
    string str[1000];
    int p = 0,q;
    cin >> a >> b;
    int i,j,l = a.length();
    int c,l1,f = 0;
    for(j=1;j<=l;j++)
    {
    	for(i=0;i<l-j+1;i++)
    	{
    		f = 0;
    		string s = a.substr(i,j);
    		for(q=0;q<p;q++)
    		{
    			if(str[q]==s)
    			{
    				f = 1;
    			}
    		}
    		if(f!=1)
    		{
    			str[p] = s;
    			p++;
    			int k,m;
    			l1 = s.length();
    			for(k=1;k<=l1;k++)
    			{
    				for(m=0;m<l1-k+1;m++)
    				{
    					string s1 = s.substr(m,k);
    					if(s1==b)
    					{
    						c++;
    						break;
    					}
    				}
    			}
    		}
    	}
    }
    printf("%d\n",c);
    return 0;
}
