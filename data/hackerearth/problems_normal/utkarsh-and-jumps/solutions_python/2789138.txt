n,p = map(int,raw_input().split())

p = p/100.0
#print(p)
tp = [0]*(n+1)
if(n==1):
	print(0)
elif(n==2):
	print("%.6f" % p)
elif(n==3):
	t = 1-p
	print("%.6f" % t)
else:
	tp[1] = 0
	tp[2] = p
	tp[3] = 1-p
	for i in range(4,n+1):
		tp[i] = tp[i-3]*(1-p) + tp[i-2]*(p)
	#t = tp[n]
	print("%.6f" % tp[n])
