#include <iostream>
#include<string>
#include<algorithm>
using namespace std;

void permute(string s)
{	
	do
	{
		cout<<s<<" ";
	}while(next_permutation(s.begin(),s.end()));
	cout<<endl;
}

int main()
{
   	int T,N,i;
   	
   	cin>>T;
   	
   	while(T--)
   	{
   		cin>>N;
   		
   		char arr[N+1];
   		for(i=1;i<=N;i++)
   			arr[i-1] = i + '0';
   		arr[i-1] = 0;
   		
   		permute(arr);
   	}
    return 0;
}
