#include <iostream>

using namespace std;

int main()
{
        int test_cases;
        cin>>test_cases;
        while(test_cases > 0)
        {
                int X1,X2,Y1,Y2,w=0;
                cin>>X1>>Y1>>X2>>Y2;
                if(X1==Y2)
                w++;
                if(X2==Y1)
                w++;
                cout<<w<<"\n";
                test_cases --;
        }
        return 0;
}
