#include <iostream>
using namespace std;

#define p 1000000007

long long dp[10000000] = {0};

long long fact(int n){
	if(dp[n]){
		return dp[n];
	}
	dp[n] = (n*fact(n-1))%p;
	return dp[n];
}

long long modexp(long long x){
	int y = p-2;
	long long ans = 1;
	while(y){
		if(y&1){
			ans = (ans*x)%p;
		}
		x = (x*x)%p;
		y >>= 1;
	}
	return ans;
}

int main()
{
	dp[0] = dp[1] = 1;
    int t,n,m,r;
    cin >> t;
    while(t--){
    	cin >> n >> m;
    	if(n == 1){
    		cout << 0 << endl;
    		continue;
    	}
    	r = n*n-1;
    	cout << (fact(m+r-1)*modexp((fact(m)*fact(r-1))%p))%p << endl;
    }
    return 0;
}
