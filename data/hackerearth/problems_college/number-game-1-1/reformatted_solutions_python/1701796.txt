testcases = raw_input()
for i in range(int(testcases)):
    counter = 0
    number = raw_input()
    number = int(number)
    while number != 1:
        while number % 3 == 0:
            number = number / 3
            counter += 1
        while number % 2 == 0:
            number = number / 2
            counter += 1
        if number != 1:
            number -= 1
            counter += 1
    print counter
