#include<iostream>
#include<string.h>
using namespace std;
typedef long long int ll;

int main()
{
	ios::sync_with_stdio(false);
	cin.tie(NULL);
	int T,m;
	string S;
	
	cin>> T;
	while(T--)
	{
		cin>>S>>m;
		
		ll n=0;
		
		for(int i=0;i<S.length();i++)
		{
			 n=n * 10 + (S[i] - '0');
			 n = n%m;
		}
		
		cout<<n<<endl;
	}
	
	
	return 0;
}
