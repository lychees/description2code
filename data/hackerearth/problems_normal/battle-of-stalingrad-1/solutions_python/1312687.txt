def A(x1, y1, x2, y2, x3, y3):
	return abs((x1 * (y2 - y3) + x2 * (y3 - y1) + x3 * (y1 - y2)) / 2.0)
def f(x1, y1, x2, y2, x3, y3, x4, y4):
	a1 = A(x4, y4, x2, y2, x3, y3)
	a2 = A(x1, y1, x4, y4, x3, y3)
	a3 = A(x1, y1, x2, y2, x4, y4)
	return A(x1, y1, x2, y2, x3, y3) == a1 + a2 + a3
for T in range(input()):
	print 'INSIDE' if f(*map(int, raw_input().split())) else 'OUTSIDE'
