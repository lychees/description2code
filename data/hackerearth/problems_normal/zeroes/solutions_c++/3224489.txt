#include <bits/stdc++.h>

using namespace std;

#define Nmax 100000
#define foreach(i,X) for(__typeof((X).begin()) i = (X).begin(); i != (X).end(); i++)
#define f first;
#define sc second;


vector <int> prime (100005,0);
vector <int> freq  (100005,0);
vector <bool> visited(100005,0);
vector < pair<int ,int> > fact[100005]; 

int main()
{
//    printf("Hello World!\n");
  
  int i,x,n,j,res,q,k;
  scanf("%d",&n);
  for(i=0;i<n;i++)
  {
  	scanf(" %d ",&x);
  	freq[x]++;
  }
  
  
  	for(i=2;i<=Nmax;i++)
	{
		if(visited[i]==1) continue;
		int cnt=0;
		for(j=i;j<=Nmax;j+=i)
		{	
			k=j;
			visited[k]=1;
			cnt=0;
			while(k%i ==0 ){
				cnt++;
				k/=i;
			}
			prime[i]+=cnt*freq[j];
			fact[j].push_back(make_pair(i,cnt));
			
		}
	}
	
	scanf("%d",&q);
	for(;q--;)
	{
		res=Nmax+1;
		scanf("%d",&x);
		foreach(i,fact[x])	
		{
			res=min(res , prime[i->first]/i->second);
		//	printf("%d %d %d\n",x,i->first,i->second);
		}
		printf("%d\n",res);	
	}
	

  }
