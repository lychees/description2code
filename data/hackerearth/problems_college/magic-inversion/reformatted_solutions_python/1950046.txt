def isp(n):
    for i in range(2, int(n/2)+1):
        if n % i == 0:
            return False
    return True


n = input()
a = map(int, raw_input().split())
b = map(int, raw_input().split())
c = 0
i, j = 0, 0
while i < n-1:
    while i < j+1 and j < n:
        if a[i] > b[j]:
            c += 1
        j += 1
    i += 1
if isp(c):
    print "MAGIC INVERSION"
else:
    print "SIMPLE INVERSION"
