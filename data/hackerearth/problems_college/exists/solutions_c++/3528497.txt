#include <bits/stdc++.h>
using namespace std;

const int lmt = 100005;

int main()
{
    int t,n,x,k;
    scanf("%d",&t);
    while(t--){
    	scanf("%d",&n);
    	int pos[lmt]={0},neg[lmt]={0};
    	for(int i=0;i<n;i++){
    		scanf("%d",&x);
    		if(x<0)
    			neg[-x]++;
    		else
    			pos[x]++;
    	}
    	scanf("%d",&k);
    	for(int i=0;i<k;i++){
    		scanf("%d",&x);
    		bool ans = false;
    		if(x<0 && neg[-x])
    			ans = true;
    		else if(x>=0 && pos[x])
    			ans =true;
    		if(ans)
    			printf("Yes\n");
    		else
    			printf("No\n");
    	}
    }
    return 0;
}
