#include <iostream>
#include <vector>
#include <map>
#include <algorithm>
using namespace std;
int main()
{

    long long int n,i,j,a,b,c,Count=0;
    map<long long int,long long int>M;
    map<long long int,long long int>::iterator it;
    cin>>n;
    for(i=0;i<n;i++)
    {
        vector<long long int>V;
        cin>>a>>b>>c;
        V.push_back(a);V.push_back(b);V.push_back(c);
        sort(V.begin(),V.end());
        j=V[0]*101+V[1]*10201+V[2]*1030301;
        if((it=M.find(j))!=M.end())
            it->second++;
        else
            M.insert(pair<long long int,long long int>(j,1));
    }
    for(it=M.begin();it!=M.end();it++)
        if(it->second==1)
        Count++;
    cout<<Count;
    return 0;
}
