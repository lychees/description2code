#include<bits/stdc++.h>
#define ll long long int
using namespace std;
ll arr[1000000],n,k;
int check(ll mid)
{
    ll temp=arr[0]+mid,i,j=k;
    j--;
    for(i=1;i<n;i++)
    {
        if(temp+mid>=arr[i])
            continue;
        if(j==0)
            return 0;
        temp=arr[i]+mid;
        --j;
    }
    return 1;
}
main()
{
    ll i,hi,lo,mid,temp;
    cin>>n>>k;
    for(i=0;i<n;i++)
        cin>>arr[i];
    sort(arr,arr+n);
    hi=100000000;
    lo=0;
    while(lo<hi)
    {
        mid=(lo+hi)/2;
        if(check(mid))
            hi=mid;
        else
            lo=mid+1;
    }
    cout<<lo<<endl;
}
