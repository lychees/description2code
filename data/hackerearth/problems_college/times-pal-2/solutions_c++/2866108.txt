#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    int t,f;
    cin>>t;
    while(t--){
    	string a,b,t;
    	cin>>a>>b;
    	int x=a.length(),y=b.length(),ctr=0;
    	for(int i=0;i<x;i++){
    		t=a.substr(0,i);
    		t+=b;
    		t+=a.substr(i);
    		f=0;
    		for(int m=0,n=x+y-1;m<n;m++,n--){
    		if(t[m]!=t[n]){f=1;break;}
    		}
    		if(f==0) ctr++;
    	}
    	t=a+b;
    		f=0;
    		for(int m=0,n=x+y-1;m<n;m++,n--){
    		if(t[m]!=t[n]){f=1;break;}
    		}
    		if(f==0) ctr++;
    	cout<<ctr<<endl;
    }
    return 0;
}
