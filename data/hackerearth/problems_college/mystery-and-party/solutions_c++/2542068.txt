#include <iostream>
#include <cstdio>
#include <cmath>
#include <algorithm>
#include <climits>
using namespace std;

int main()
{
    int t;
    scanf("%d", &t);
    while(t > 0) {
    	long long n, k;
    	scanf("%lld%lld", &n, &k);
    	long long arr[n];
    	for(int i=0; i<n; i++) {
    		scanf("%lld", &arr[i]);
    	}
    	sort(arr, arr+n);
    	printf("%lld\n", arr[k-1]);
    	   	
    	t--;
    }
    return 0;
}
