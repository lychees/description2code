#include <bits/stdc++.h>
using namespace std;

long dp[1001][1001];

long preProcess(){
	dp[0][0]=1;
	for(int i=1; i<=1000; i++){
		dp[i][0]=1;
		dp[i][i]=1;
		for(int j=1; j<=i; j++){
			dp[i][j] = (dp[i-1][j-1]%1000000007 + dp[i-1][j]%1000000007)%1000000007;
		}
	}
}

int main(){
	int t;
	cin>>t;
	memset(dp, 0, sizeof(dp));
	preProcess();
	while(t--){
		int n, p;
		cin>>n>>p;
		long ans=0;
		
		for(int i=0 ;i<=p; i++){
			ans = (ans%1000000007 + dp[n][i]%1000000007)%1000000007;
		}
		
		cout<<ans<<endl;
	}
	return 0;
}
