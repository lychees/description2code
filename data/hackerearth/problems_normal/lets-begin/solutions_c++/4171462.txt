#include<iostream>
#include<limits.h>
using namespace std;
long long tab[100000];
//memset(tab,INT_MAX,sizeof(tab));
long long counter=0;
int p[4]={2,3,5,7};
long long fun(long long n){
	long long ans=INT_MAX;
	if(tab[n]!=INT_MAX && n<=counter) ans=tab[n];
	else{
		if(counter==0){
			for(int z=1;z<100000;++z){
				tab[z]=INT_MAX;
			}
			tab[0]=0;
			for(long long i=1;i<=n;++i){
				for(long long j=0;j<4;++j){
					if(p[j]<=i){
						long long sub_res = tab[i-p[j]];
						if(sub_res!=INT_MAX && (sub_res+1<tab[i])){
							tab[i] = sub_res+1;
						}
					}
				}
			}
			counter=n;
			ans=tab[n];
		}
		else if(counter<n){
			for(long long i=counter+1;i<=n;++i){
				for(long long j=0;j<4;++j){
					if(p[j]<=i){
						long long sub_res = tab[i-p[j]];
						if(sub_res!=INT_MAX && (sub_res+1<tab[i])){
							tab[i] = sub_res+1;
						}
					}
				}
			}
			counter=n;
			ans=tab[n];
		}
	}
	return ans!=INT_MAX ? ans:-1;
}
int main(){
	int t;
	cin>>t;
	long long num;
	while(t--){
		cin>>num;
		cout<<fun(num)<<endl;
	}
	return 0;
}
