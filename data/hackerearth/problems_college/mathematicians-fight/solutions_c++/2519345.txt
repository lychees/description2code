#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long int n,t;
    cin>>t;
    while(t--)
    {
     cin>>n;
     long long int i=1,res;
     while(n>i)
      i<<=1;
      i>>=1;
      res=n-i;
      //cout<<i;
      cout<<2*res+1<<endl;
    }
    return 0;
}
