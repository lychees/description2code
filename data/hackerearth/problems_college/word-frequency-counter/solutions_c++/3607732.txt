#include<bits/stdc++.h>
using namespace std;
int main()
{
	string s,a;
	stringstream t;
	map<string,int> m;
	getline(cin,s);
	t<<s;
	while(t>>a)
	{
		for(int i=0;a[i]!='\0';i++)
		a[i]=toupper(a[i]);
		m[a]++;
	}
/*	map<string,int>::iterator it;
	for(it=m.begin();it!=m.end();it++)
	{
		cout<<it->first<<" "<<it->second<<"\n";
	}*/
	stringstream p;
	p<<s;
	while(p>>a)
	{
		for(int i=0;a[i]!='\0';i++)
		a[i]=toupper(a[i]);
		if(m[a])
		cout<<a<<" "<<m[a]<<"\n";
		m.erase(a);
	}
}
