#include<bits/stdc++.h>
using namespace std;
int a[100005],n,k;
bool valid(int x)
{
    int i=0,c=0;
    while(i<n)
    {
        int dis=a[i]+2*x;
        while(a[i]<=dis&&i<n)
            i++;
        c++;
    }
    if(c>k)
        return false;
    return true;
}
int main()
{
    int ans;
    cin>>n>>k;
    for(int i=0;i<n;i++)
        cin>>a[i];
    sort(a,a+n);
    int low=1,high=10000000;
    while(low<=high)
    {
        int mid=(low+high)/2;
        if(valid(mid))
        {
            ans=mid;
            high=mid-1;
        }
        else
            low=mid+1;
    }
    cout<<ans;
    return 0;
}
