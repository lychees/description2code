#include<bits/stdc++.h>
using namespace std;
int main()
{
     float a,b,c,d,e,f;
double x[3],y[3];
cin>>a>>b;
cin>>c>>d;
cin>>e>>f;
y[0]=d+f-b;
y[1]=b+d-f;
y[2]=b+f-d;

x[0]=c+e-a;
x[1]=a+c-e;
x[2]=a+e-c;
vector<pair<double,double> > vec;
vec.push_back(make_pair(x[0],y[0]));
vec.push_back(make_pair(x[1],y[1]));
vec.push_back(make_pair(x[2],y[2]));

sort(vec.begin(),vec.end());
 a = vec[0].first;
 b = vec[0].second;
 c = vec[1].first;
 d = vec[1].second;
 e = vec[2].first;
 f = vec[2].second;

  printf("%.4lf %.4lf\n%.4lf %.4lf\n%.4lf %.4lf\n",a,b,c,d,e,f);
return 0;
}
