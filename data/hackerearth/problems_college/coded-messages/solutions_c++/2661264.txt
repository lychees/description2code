int accno(string s)
{
  int l=0;
  int i=0;
  while(i<s.length())
  {
  	switch(s[i])
  	{
  		case 'A': l+=1;
  		          break;
  		case 'B': l=l+5;
  		          if(i-1>=0&&s[i-1]=='A')
  		          l-=2;
  		          break;
  		case 'C':l+=10;
		         if(i-1>=0&&s[i-1]=='A')
		         l-=2;
  		         break;
  		case 'D':l+=50;
  		          if(i-1>=0&&s[i-1]=='C')
  		          l-=20;
  		          break;
  		case 'E':l+=100;
  		         if(i-1>=0&&s[i-1]=='C')
  		         l-=20;
  		         break;
  		case 'F':l+=500;
  		         if(i-1>=0&&s[i-1]=='E')
  		         l-=200;
  		         break;
  		case 'G':if(i-1>=0&&s[i-1]=='E')
		         l+=800;
		         else l+=1000;
  		         break;
  		          
	}
	i++;
  }
  return l;	
}
