from bisect import bisect_left

primes = []

def get_primes(n):
    
    sieve = [True] * (n/2)

    for i in xrange(3,int(n**0.5)+1,2):
        if sieve[i/2]:
            sieve[i*i/2::i] = [False] * ((n-i*i-1)/(2*i)+1)
    
    return [1] + [2*i for i in xrange(1,n/2) if sieve[i]]
	

primes = get_primes(1000100)

for _ in xrange(input()):
	n = int(raw_input())
	print sum(map(lambda x:1 if bisect_left(primes, int(x)) < 1 else primes[bisect_left(primes, int(x))]+1, raw_input().split()))
