string readLine() {
    string s;
    getline ( cin,s );
    string r;
    for ( char c: s ) {
        if ( 'a'<= c &&c <='z' ) r+=c;
        else if ( 'A'<= c &&c <='Z' ) r+=c+'a'-'A';


    }
    return r;
}
vi count ( const string &s ) {
    vi count ( 256 );
    for ( char c: s ) count[c]++;
    return count;
}
