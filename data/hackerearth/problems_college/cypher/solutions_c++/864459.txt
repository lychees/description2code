#include <iostream>
using namespace std;

int main()
{
    int t; cin >> t;
    
    while(t--)
    {
    	string word;
    	cin >> word;
    	
    	for(int i=1;i< word.size();i++)
    	{
    		word[i] = ((word[i]+i) % 96) % 26;
    		word[i]= word[i] ?word[i]+96 :122;
    	}
    	cout << word << endl;
    }
    
    
    return 0;
}
