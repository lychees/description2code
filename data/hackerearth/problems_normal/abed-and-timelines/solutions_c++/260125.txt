#include<iostream>
using namespace std;
int Min(int a,int b)
{
    return((a<b)?a:b);
}
int main()
{
    int t,i,a,b,val;
    int n,m,x1,y1,x2,y2,p,moves,ans;
    cin>>t;
    for(i=0;i<t;i++)
    {
        cin>>n>>m;
        cin>>x1>>y1;
        cin>>x2>>y2;
        cin>>p;
        moves=0;

        a=x1-x2;
        if(a<0)
            a=a*(-1);

        b=(n-a);
        //if(b<0)
            //b=b*(-1);

        moves=Min(a,b);

        a=y1-y2;
        if(a<0)
            a=a*(-1);

        b=(m-a);
        //if(b<0)
            //b=b*(-1);

        moves+=Min(a,b);

        ans=moves*p;
        cout<<ans<<endl;
        if(ans<=1000)
            cout<<"You saved the group."<<endl;
        else
           cout<<"Let go of the group."<<endl;
    }
    return 0;
}
