def del_adj(list1):
    list2 = []
    ln_1 = len(list1)
    for i in range(1,ln_1):
        e = list1[i-1]
        if list1[i] != e:
            list2.append(list1[i])
    ln_2 = len(list2)

    print(ln_1 - ln_2 - 1)

tc = int(input())
while tc :
    string = raw_input()
    list1 = list(string)
    del_adj(list1)
    tc -= 1
