#include <bits/stdc++.h>
using namespace std;
#define ppi pair<long long, pair<int,int> >

const long long mod = 1e9 +7;
int t,n,m,c,p[1600];
ppi g[31315];

void initialise(){
	for(int i=1;i<=n;i++){
		p[i]=i;
	}
}

int root(int x){
	while(x!=p[x]){
		p[x]=p[p[x]];
		x=p[x];
	}
	return x;
}

void u(int x,int y){
	x=root(x);
	y=root(y);
	p[x]=p[y];
}

long long kruskal(){
	int x,y;
	long long mincost=1,cost;
	for(int i=0;i<m;i++){
		x=g[i].second.first;
		y=g[i].second.second;
		cost = g[i].first;
		if(root(x)!=root(y)){
			mincost *= cost;
			u(x,y);
			mincost %= mod;
		}
	}
	return mincost;
}

int main()
{
    cin>>t;
    int x,y;
    long long w;
    while(t--){
    	cin>>n>>m;
    	initialise();
    	for(int i=0;i<m;i++){
    		cin>>x>>y>>w;
    		g[i] =make_pair(w, make_pair(x,y));
    	}
    	sort(g, g+m);
    	long long mn=kruskal();
    	cout<<mn<<endl;
    }
    return 0;
}
