#include <bits/stdc++.h>
using namespace std;

bool prime(int n){
	int k = 2;
	while (k*k <= n){
		if (n % k == 0)
			return false;
		k += 1;
	}
	return true;
}

int main(){
    int t;
    cin >> t;
    while (t > 0){
    	int n;
    	cin >> n;
    	n++;
    	while (prime(n) == true)	n++;
    	cout << n << endl;
    	t--;
    }
    return 0;
}
