#include<iostream>
#include<stdio.h>
using namespace std;
#define m 1000000007
long long power(long long base, long long exponent, long long modulus)
{
    long long result = 1;
    while (exponent > 0) {
        if (exponent % 2 == 1)
            result = (result * base) % modulus;
        exponent = exponent >> 1;
        base = (base * base) % modulus;
    }
    return result;
}
int main()
{ int tc;
  cin>>tc;
  int j;
  for(j=1;j<=tc;j++)
  { int N;
    cin>>N;
    long long int A[N];
    long long int i;
    for(i=0;i<N;i++)
       cin>>A[i];
    long long int max=-1;
    for(i = 0;i < (1 << N); ++i)
    {   long long int sum=0;
        long long int mul=1;
        int count=0;
        for(int j = 0;j < N;++j)
            if(i & (1 << j))
               { sum=(sum+A[j])%m;;
                 mul=(mul*A[j])%m;
                 count++;
               }
        if(count>=2)
        {
           long long int t=power(sum,m-2,m);
           t=(t*mul)%m;
           if(t>max)
             max=t;
        }

    }
    cout<<"Case #"<<j<<": "<<max<<"\n";
  }
return 0;

}
