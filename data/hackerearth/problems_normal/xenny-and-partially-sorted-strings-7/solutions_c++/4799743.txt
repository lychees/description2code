#include <iostream>
#include<vector>
#include<string>
#include<algorithm>
using namespace std;

struct foo{
	string s;
	int m;
	
};

bool myfunction(foo a,foo b)
{
	for(int i=0;i<a.m;i++)
	{
		if(a.s[i]<b.s[i])
			return 1;
		else if(a.s[i]>b.s[i])
			return 0;
		else;
	}
	return 1;
}

int main()
{
    //cout << "Hello World!" << endl;
    
    int t;
    cin>>t;
    while(t--)
    {
    	int n,k,m;
    	cin>>n>>k>>m;
    	vector<foo> v;
    	for(int i=0;i<n;i++)
    	{
    		string s;
    		cin>>s;
    		foo temp;
    		temp.s=s;
    		temp.m=m;
    		v.push_back(temp);
    	}
    	
    	sort(v.begin(),v.end(),myfunction);
    	
    	cout<<v[k-1].s<<endl;
    }
    return 0;
}
