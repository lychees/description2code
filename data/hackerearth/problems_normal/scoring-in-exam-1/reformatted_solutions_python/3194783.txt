n, q = raw_input().split()
questions = raw_input().split()
scores = raw_input().split()
queries = []
queDic = {}
for i in xrange(int(q)):
    val = int(raw_input())
    queries.append(val)
    queDic[val] = 0

quesData = []
for item in xrange(len(questions)):
    quesData.append(int(questions[item]))

quesData.sort(reverse=True)

sums = 0
for i in xrange(len(quesData)):
    sums = sums + quesData[i]
    if (i+1) in queDic:
        queDic[i+1] = sums

for item in queries:
    if item in queDic:
        print queDic[item]
