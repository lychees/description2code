typedef long long ll;

const int INF = (int) 1e9 + 7;
const int MAXN = (int) 2e6 + 7;

int n;
bitset<2000> g[2000];
bitset<2000> common;

int main() {
  #ifdef LOCAL
  freopen("in", "r", stdin);
  #endif

  scanf("%d", &n);
  for (int i = 0; i < n; i++) {
    for (int j = 0; j < n; j++) {
      int x;
      scanf("%d", &x);
      if (x) {
        g[i][j] = 1;
      }
    }
  }
  ll ans = 0;
  for (int i = 0; i < n; i++) {
    for (int j = i + 1; j < n; j++) {
      common = (g[i] & g[j]);
      int sz = common.count();
      ans += sz * (sz - 1) / 2;
    }
  }
  cout << ans / 2;
  return 0;
}
