#include <bits/stdc++.h>
using namespace std;

int N, Q;
char S[11];
pair <long long, int> data[100010];

long long gethash(char X[], int len)
{
    long long hsgs = 0;
    for (int i = 0; i < len; i++)
        hsgs = hsgs * 3137 + X[i] - 'a';
    return hsgs;
}

main()
{
    scanf("%d", &N);
    for (int i = 1; i <= N; i++)
    {
        scanf("%s", S + 1);
        data[i] = make_pair(gethash(S + 1, strlen(S + 1)), i);
    }

    sort(data + 1, data + N + 1);
    scanf("%d", &Q);
    for (int l, r; Q--;)
    {
        scanf("%d %d %s", &l, &r, S + 1);
        long long myhash = gethash(S + 1, strlen(S + 1));
        printf("%d\n", upper_bound(data + 1, data + N + 1, make_pair(myhash, r))
                        - lower_bound(data + 1, data + N + 1, make_pair(myhash, l)));
    }
}
