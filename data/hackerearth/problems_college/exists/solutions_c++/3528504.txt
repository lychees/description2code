#include <bits/stdc++.h>
using namespace std;

const int lmt = 200005;
const int add = 100000;
int freq[lmt];

int main()
{
    int t,n,x,k;
    scanf("%d",&t);
    while(t--){
    	scanf("%d",&n);
    	memset(freq,0,sizeof(freq));
    	for(int i=0;i<n;i++){
    		scanf("%d",&x);
    		freq[x+add]++;
    	}
    	scanf("%d",&k);
    	while(k-- > 0){
    		scanf("%d",&x);
    		if(freq[x+add])
    			printf("Yes\n");
    		else
    			printf("No\n");
    	}
    }
    return 0;
}
