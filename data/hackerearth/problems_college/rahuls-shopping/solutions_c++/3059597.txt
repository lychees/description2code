#include <iostream>
using namespace std;

int main()
{
    int n,l,b,a[50],rate,tmp,sum=0,tmp1,plank,amt=0;
    float final=0;
    cin>>n;
    for(int i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    cin>>l>>b;
    for(int i=0;i<n;i++)
    {
    tmp=a[i];
    sum=0;
    final=0;
    while(tmp>0)
    {
    	tmp1=tmp%10;
    	sum=sum+tmp1;
    	tmp=tmp/10;
    }
    rate=(l*b)*(a[i]*sum);
    
    if(a[i]>=1&&a[i]<=10){
    	final=rate-(rate*0.25);
    }
    if(a[i]>=11&&a[i]<=20){
    	final=rate-(rate*0.30);
    }
    if(a[i]>=20&&a[i]<=50){
    	final=rate-(rate*0.20);
    }
     plank=final;
     amt=amt+plank;
    }
    cout<<amt;
    return 0;
}
