#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t; cin>>t;
    while(t--){
    	string s,a,b,c,d;
    	set<string>myset;
    	bool flag=0;
    	cin>>s;
    	if(s.length()>=10){
    		flag=1;
    		cout<<"YES"<<endl;
    		continue;
    	}
    	else{
	    	int n=s.length();
	    	for(int i=1;i<n;i++){
	    		for(int j=i+1;j<n;j++){
	    			for(int k=j+1;k<n;k++){
	    				a=s.substr(0,i),b=s.substr(i,j-i),c=s.substr(j,k-j),d=s.substr(k,n-k);
	    				//cout<<a<<" "<<b<<" "<<c<<" "<<d<<endl;
	    				myset.insert(a),myset.insert(b),myset.insert(c),myset.insert(d);
	    				if(myset.size()==4){
	    					flag=1;
	    					cout<<"YES"<<endl;
	    					break;
	    				}
	    				//if(flag) break;
	    				myset.clear();
	    			}
	    			if(flag) break;
	    		}
	    		if(flag) break;
	    	}
	    	if(!flag) cout<<"NO"<<endl;
    	}
    }
    return 0;
}
