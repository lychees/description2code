#include <bits/stdc++.h>
using namespace std;

vector<pair<int, int> > t;

int main()
{
	
	
	int tt;
	cin>>tt;
	while(tt--) {
		t.clear();
		int n;
		cin>>n;
		for(int i = 0; i < n; ++i) {
			int s, e;
			cin>>s>>e;
			t.push_back(make_pair(s, -1));
			t.push_back(make_pair(e-1, 1));
		}
		sort(t.begin(), t.end());
		int total = 0;
		int ans = 0;
		for(int i = 0; i < t.size(); ++i) {
			if(t[i].second == -1)	total++;
			else					total--;
			ans = max(ans, total);
		}
		cout<<ans<<"\n";
	}
	
    return 0;
}
