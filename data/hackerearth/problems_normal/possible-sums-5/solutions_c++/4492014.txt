#include<iostream>
#include<vector>
#include<set>

using namespace std;

int countPossibleSums(vector<int>v){
	set<int> s;
	s.insert(0);
	for(int i=0;i<v.size();i++){
		set<int> :: iterator it = s.begin();
		vector<int>temp;
		for(it = s.begin();it!=s.end();it++){
			temp.push_back(*it);
		}
		for(int j=0;j<temp.size();j++){
			s.insert(temp[j]+v[i]);
		}	
	}
	return s.size();
}

int main(){
	int t;
	cin >> t;
	vector<vector<int> > v;
	for(int i=0;i<t;i++){
		vector<int> u;
		v.push_back(u);
	}
	for(int i=0;i<t;i++){
		int numberOfElements;
		cin >> numberOfElements;
		v[i].resize(numberOfElements);
		for(int j=0;j<numberOfElements;j++){
			cin >> v[i][j];
		}
	}
	for(int i=0;i<t;i++){
		int x = countPossibleSums(v[i]);
		cout << x << endl;
	}
}
