#include<bits/stdc++.h>
using namespace std;
void ip_array(int *arr,int ub){
	int i;
	for(i=0;i<=ub;i++){
		cin>>arr[i];
	}
}
int main(){
	int tc,l,i;
	cin>>tc;
	while(tc--){
		cin>>l;
		int *arr;
		arr = new int [l];
		l--;
		ip_array(arr,l);
		sort(arr,arr+l+1);
		for(i=0;i<=(l/2);i++){
			arr[i]+=arr[l-i];
		}
		l=l/2;
		sort(arr,arr+l+1);
		cout<<arr[l]-arr[0]<<endl;
	}
	return 0;
}
