#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n,t,max_so_far=0,max_end_here=0;
    cin>>n;
    while(n--)
    {
    	cin>>t;
    	max_end_here=max_end_here+t;
    	if(max_end_here<0)
    	max_end_here=0;
    	if(max_so_far<max_end_here)
    	max_so_far=max_end_here;
    }
    cout<<max_so_far;
    return 0;
}
