#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	int n;
    	cin >> n;
    	int arr[n];
    	int i;
    	long long int even=0,odd=0;
    	for(i=0;i<n;i++)
    	{
    		cin >> arr[i];
    		if(arr[i]%2)
    		even++;
    		else
    		odd++;
    	}
    	long long int ans=odd*even;
    	cout << ans << endl;
    }
    return 0;
}
