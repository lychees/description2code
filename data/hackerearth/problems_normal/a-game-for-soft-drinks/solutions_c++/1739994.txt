#include <bits/stdc++.h>
using namespace std;

int dp[105][105][105];

void process() {
	for(int i = 0; i <= 100; i++) {
		for(int j = 0; j <= 100; j++) {
			for(int k = 0; k <= 100; k++) {
				vector <bool> lol(15,0);
				if(i && j && k) lol[dp[i - 1][j - 1][k - 1]] = 1;
				if(i && j) lol[dp[i - 1][j - 1][k]] = 1;
				if(i && k) lol[dp[i - 1][j][k - 1]] = 1;
				if(j && k) lol[dp[i][j - 1][k - 1]] = 1;
				if(i) lol[dp[i - 1][j][k]] = 1;
				if(j) lol[dp[i][j - 1][k]] = 1;
				if(k) lol[dp[i][j][k - 1]] = 1;
				while(lol[dp[i][j][k]]) dp[i][j][k]++;
			}
		}
	}
}

int main() {
	process();
	int t;
	scanf("%d",&t);
	while(t--) {
		int a, b, c;
		scanf("%d %d %d",&a,&b,&c);
		if(dp[a][b][c] == 0) printf("Aishwarya\n");
		else printf("Ashima\n");
	}
}
