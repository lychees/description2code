for _ in range(int(raw_input())):
    n, k = map(int, raw_input().split())
    a = map(int, raw_input().split())
    ktest = 0
    for i in range(n):
        if a[i] < 0 or a[i] == 0:
            ktest += 1
        else:
            pass
    if k > ktest:
        print "YES"
    else:
        print "NO"
