#include <iostream>
#include <algorithm>
using namespace std;
long long a[100001],sum[100001];
int main()
{
    ios::sync_with_stdio(0);
    int n,t;
    long long k; 
    cin>>t;
    while(t--){
    	cin >> n >> k; 
    	for(int i=0;i<n;i++)
        	cin >> a[i]; 
    	long long sm = 0; 
    	long long best = -1,bestaj;
    	sort(a,a+n);
    	sum[1] = a[0];
    	for(int i=1;i<n;i++) sum[i+1] = sum[i]+a[i];
    	long long mx = 0;
    	int low, high,mid;
    	for(int i = 0;i<n;i++){
    		low = 0;
    		high = i;
    		
    		while(low<high){
    			mid = (low+high)>>1;
    			//cout << low << " " << high << endl;
    			if ((long long)((i-mid+1)*a[i] - (sum[i+1]-sum[mid]))<=k)    			
    				high = mid;
    			else 
    				low = mid+1;
    		}
    		if (i-low+1>best){
    			best = i-low+1;
    			bestaj = a[i];
    		}
    	}
    	cout << best << " " << bestaj << "\n";
    }
    return 0;
}
