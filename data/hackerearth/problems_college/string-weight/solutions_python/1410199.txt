#!/usr/bin/env python
import string

t = int(raw_input())
letters = string.ascii_letters
let_list = {}
for x in range(26):
	let_list[letters[x]] = x + 1

for _ in range(t):
	text = raw_input()
	text = text.lower()
	text = ''.join(sorted(text))
	m = int(raw_input())
	missed = raw_input().split()
	weight = 0
	for i in range(len(text)):
		if text[i] in missed:
			continue
		else:
			weight += let_list[text[i]]
	print(weight)
