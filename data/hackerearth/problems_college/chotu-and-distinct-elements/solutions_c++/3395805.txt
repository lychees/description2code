#include <iostream>
#include <map>
#include <algorithm>
using namespace std;

int main()
{
    int t,n,prev,i,j,idx;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	int arr[n];
    	for(i=0; i<n; ++i)
    	{
    		cin>>arr[i];
    	}
    	sort(arr, arr+n);
		int vrr[n];
		idx=0;
		prev=arr[0];
		vrr[idx++]=arr[0];
		for(i=1; i<n; ++i)
		{
			if(arr[i]!=prev)
			{
				vrr[idx++]=arr[i];
				prev=arr[i];
			}
		}
		j=idx-1;
    	for(i=0; i<idx/2; ++i)
    	{
    		cout<<vrr[j]<<" "<<vrr[i]<<" ";
    		j--;
    	}
    	if(idx&1)
			cout<<vrr[i];
		cout<<endl;
    }
    return 0;
}
