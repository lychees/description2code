def binary_search(l, r, ar, el):
    mid = l+(r-l)/2
    result = 0
    if l == r:
        if ar[mid] == el:
            result = mid+1
        else:
            result = mid
    elif ar[mid] >= el:
        result = binary_search(mid+1, r, ar, el)
    else:
        result = binary_search(l, mid, ar, el)

    return result
