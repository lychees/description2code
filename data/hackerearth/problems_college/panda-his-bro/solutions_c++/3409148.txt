typedef unsigned long long int ull;
typedef long long int ll;
typedef vector<vector<ll> > mat;

vii v[100005];
int hm=0,a=1,ans=0;

void dfs(int at,int p,int h)
{
	if(h>hm)
	{
		hm=h;
		a=at;
	}
	tr(v[at],it)
	{
		int to=*it;
		if(to!=p)
		dfs(to,at,h+1);
	}
}

void dfs1(int at,int p,int h)
{
	ans=max(ans,h);
	tr(v[at],it)
	{
		int to=*it;
		if(to!=p)
		dfs1(to,at,h+1);
	}
}
