#include<iostream>
#include<stdlib.h>
#include<stdio.h>
using namespace std;
int main()
{
	int a,b,c,x,y;
	cin>>a>>b>>c;
	for(int i=-5;i<=6;i++)
	{
		for(int j=-5;j<=6;j++)
		{
			if(a*i+b*j+c==0)
			{
				x=i;
				y=j;
				break;
			}
		}
	}
	if(x<-5||y>6)
		cout<<"-1"<<endl;
	else
		cout<<x<<" "<<y<<endl;
}
