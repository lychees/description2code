#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%lld",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
#define mod 1000000007
int main()
{
    long long int t,n,k,i,x,val;
    sd(t);
    while(t--)
    {
        val=0;
        sd(n);
        sd(k);
        for(i=0;i<n;i++)
        {
            sd(x);
            if(x==k)
                val++;
        }
        n=(n*(n-1)*(n-2))/6;
        val=(val*(val-1)*(val-2))/6;
        //printf("%lld %lld\n",n,val);
        x=val;
        i=2;
        if(val==0)
            printf("0/1\n");
        else{
        while(i<=x)
        {
            if(n%i==0&&val%i==0)
            {
                n/=i;
                val/=i;
            }
            else
                i++;
            //if(t==3)
            //printf("%lld %lld %lld\n",i,n,val);
        }
        printf("%lld/%lld\n",val,n);
        }
    }
	return 0;
}
