#include <bits/stdc++.h>
using namespace std;
int a[1000000];
int k=0;
int fn(int n)
{
	if(n%2==0)
	return (n/2);
	else
	return ((3*n)+1);
}
void cal(int n)
{
	k=0;
	a[k++]=n;
	while(a[k-1]!=1)
	{
		int p=fn(a[k-1]);
		a[k++]=p;
	}
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	cal(n);
    	int se=0,so=0;
    	for(int i=0;i<k;i++)
    	{
    		if(a[i]%2==0)
    		se+=a[i];
    		else
    		so+=a[i];
    	}
    	int ee,oo;
    	ee=se%(k-1);
    	oo=so%(k-1);
    	if(ee>oo)
    	cout<<"Even Rules"<<endl;
    	else
    	if(ee<oo)
    	cout<<"Odd Rules"<<endl;
    	else
    	cout<<"Tie"<<endl;
    }
    return 0;
}
