def cipher(input_string):
    output = ''
    for index in range(len(input_string)):
        diff = ord(input_string[index]) - ord('a')
        output = output + chr(ord('a') + (diff + index) % 26)
    print output


if __name__ == '__main__':
    cases = int(raw_input())
    start = 1
    while start <= cases:
        input_string = raw_input().strip()
        cipher(input_string)
        start = start + 1
