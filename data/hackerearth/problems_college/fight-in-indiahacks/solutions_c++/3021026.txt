#include <stdio.h>
#include <string.h>
#define SIZE 1001010

using namespace std;

int winner[SIZE];


int main()
{
    int test;
    scanf("%d",&test);
    
    while(test--) {
    	memset(winner,0,sizeof(winner));
    	int n,m1,m2;
    	scanf("%d %d %d",&n,&m1,&m2);
    	
    	for(int i=0; i<=n; i++) {
    		if(m1+i<=n) if(winner[m1+i]==0) winner[m1+i] = winner[i]^1;
    		if(m2+i<=n) if(winner[m2+i]==0) winner[m2+i] = winner[i]^1;
    	}
    	
    	/*for(int i=0; i<=n; i++)
    		printf("%d ",winner[i]);
    	puts("");*/
    	printf("%d\n",winner[n]);
    }
    return 0;
}
