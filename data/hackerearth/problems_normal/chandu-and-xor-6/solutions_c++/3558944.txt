#include<bits/stdc++.h>
using namespace std;
int arr[1002][1002];
int main() {
	int maxx = INT_MIN;
	int n, m;
	scanf("%d %d", &n, &m);
	int row[n], col[m];
	for(int i = 0; i < n; i++) {
		long int sum = 0;
		for(int j = 0; j < m; j++) {
		scanf("%d",&arr[i][j]);
		sum += arr[i][j];
		}
		row[i] = sum;
	}
	for(int i = 0; i < m; i++) {
		long int sum = 0;
		for(int j = 0; j < n; j++) {
			sum += arr[j][i];
		}
		col[i] = sum;
	}
	int minn = INT_MIN, k;
	for(int i = 0; i < n; i++) {
		for(int j = 0; j < m; j++) {
			k = row[i]^col[j];
			if(minn < k) {
				minn = k;
			}
		}
	}
	cout<<minn<<endl;
	return 0; 
}
