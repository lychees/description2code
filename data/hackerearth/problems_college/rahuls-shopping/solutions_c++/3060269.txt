#include<bits/stdc++.h>
using namespace std;
int sumval(int n)
{
	int rem =0;
	while(n!=0)
	{
		rem = rem + n%10;
		n = n/10;
	}
	return rem;
}
int main()
{
	int n;
	cin>>n;
	int i;
	int a[n];
	for(i=0;i<n;i++)
	{
		cin>>a[i];
	}
	int l,b;
	cin>>l>>b;
	int sum ;
	int dis,s=0;
	for(i=0;i<n;i++)
	{
	sum = a[i]*(sumval(a[i]))*l*b;
	if(a[i]<=10)
	{
		dis = sum - sum*0.25; 
	}
	else
	if(a[i]>10 && a[i]<=20)
	{
		dis = sum - sum*(0.3);
	}
	else
	{
			dis = sum - sum*0.20;
	}
	s = s + dis	;
	}
	cout<<s<<"\n";
}
