#include<iostream>
using namespace std;
int main() {
    string N;
	cin>>N;
    //Large 100 digit number is now in string
	int dashes=0;
    for (int i=0;i<N.size();i++) {
        int x=N[i]-'0';
		switch(x){
			case 0:
			case 6:
			case 9:
				dashes=dashes+6;
				break;
			case 1:
				dashes=dashes+2;
				break;
			case 2:
			case 3:
			case 5:
				dashes+=5;
				break;
			case 4:
				dashes+=4;
				break;
			case 7:
				dashes+=3;
				break;
			case 8:
				dashes+=7;
		}

    }
	cout<<dashes;
	return 0;
}
