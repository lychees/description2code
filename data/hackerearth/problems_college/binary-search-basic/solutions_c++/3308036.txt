#include <bits/stdc++.h>

using namespace std;

int a[100005];

bool b_search(int x, int n)
{
    int l = 0, r = n-1, m;
    while (l <= r) {
        m = (l+r)/2;
        if (a[m] == x)
            return true;
        else if (a[m] < x)
            l = m+1;
        else
            r = m-1;
    }
    return false;
}

#define s(x) scanf("%d", &x)

int main()
{
    //freopen("in10.txt", "r", stdin);
    //freopen("out10.txt", "w", stdout);
    int n, i, tmp, x;
    s(n);
    assert(1 <= n && n <= 100000);
    for (i = 0; i < n; ++i) {
        s(a[i]);
        assert(-1000000000 <= a[i] && a[i] <= 1000000000);
    }
    sort(a, a+n);
    int q;
    s(q);
    assert(1 <= q && q <= 100000);
    while (q--) {
        s(x);
        assert(-1000000000 <= x && x <= 1000000000);
        if (b_search(x, n))
            printf("YES ");
        else
            printf("NO ");
        tmp = lower_bound(a, a+n, x)-a;
        if (tmp == n)
            printf("-1 ");
        else
            printf("%d ", a[tmp]);
        tmp = upper_bound(a, a+n, x)-a;
        if (tmp == n)
            printf("-1\n");
        else
            printf("%d\n", a[tmp]);
    }
    return 0;
}
