#include <bits/stdc++.h>

using namespace std;

int t , n , k;
string arr [500];
int dp [500 + 1][ (1 << 10) + 1];

int rec (int index, int msk)
{

    if(index == n) return 0;

    int &st = dp[index][msk];

    if(st != -1) return st;

    int ans = 1e9;

    for(int i = 0; i < arr[index].size(); i++)
    {

        if(arr[index][i] == '0') continue;

        if(msk & (1 << i)) ans = min (ans , rec(index + 1, msk));
        else ans = min(ans, 1 + rec(index + 1, msk | (1 << i)));
    }

    return st = ans;
}


int main()
{
    scanf("%d",&t);

    while(t--)
    {
        memset(dp , -1, sizeof dp);
        scanf("%d",&n);
        scanf("%d",&k);

        for(int i = 0; i < n; i++)
        {
           cin >> arr[i];
        }

        int ans = rec(0 , 0);
        printf("%d\n",ans);
    }

    return 0;
}
