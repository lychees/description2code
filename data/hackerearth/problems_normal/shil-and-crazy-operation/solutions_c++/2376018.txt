#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define ios ios_base::sync_with_stdio(false)
#define fcin freopen("stdin.input","r",stdin)
#define rep(i,n) for(ll i=0;i<n;i++)
#define pb push_back

ll vis[100011];
int main()
{
	ios;
	//fcin;
	ll n,k;cin>>n>>k;
	ll a[n+1],b[n+1],p[n+1];
	rep(i,n)
		cin>>a[i+1];
	rep(i,n)
		cin>>p[i+1];
	vector<vector<int> > cycle;
	for(ll i=1;i<=n;i++)
	{
		if(vis[i]==0)
		{
			int cur=i;
			vector<int> v;
			while(vis[cur]==0)
			{
				vis[cur]=1;
				v.pb(cur);
				cur=p[cur];
			}
			cycle.pb(v);
		}
	}
	ll shift;
	rep(i,cycle.size())
	{
		shift=k%cycle[i].size();
		rep(j,cycle[i].size())
		{
			b[cycle[i][j]]=a[cycle[i][(j+shift)%cycle[i].size()]];
		}
	}
	rep(i,n)
		cout<<b[i+1]<<" ";
	//cout<<"heyyyyyy!";
	return 0;
}
