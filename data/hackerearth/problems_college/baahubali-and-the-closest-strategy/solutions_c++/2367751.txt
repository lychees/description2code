#include <iostream>
#include <climits>
#include <cstdlib>
#include<algorithm>
using namespace std;
 

void printClosest(int arr[], int n, int x)
{
    int res_l, res_r;  // To store indexes of result pair
 
    int l = 0, r = n-1, diff = INT_MAX;
 
    
    while (r > l)
    {
       
       if (abs(arr[l] + arr[r] - x) < diff)
       {
           res_l = l;
           res_r = r;
           diff = abs(arr[l] + arr[r] - x);
       }
 
       
       if (arr[l] + arr[r] > x)
           r--;
       else
           l++;
    }
 
    cout << arr[res_l] + arr[res_r] << endl ;
}
 

int main()
{
    int i , N , M , a[1000] , x ;
    cin >> N >> M ;
    for(i = 0 ; i < N ; i++)
    	cin >> a[i] ;
    sort(a , a + N) ;
    for(i = 0 ; i < M ; i++){
    	cin >> x ;
    	printClosest(a , N , x);
    }
    return 0;
}
