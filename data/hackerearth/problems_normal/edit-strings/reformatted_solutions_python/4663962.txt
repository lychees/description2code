def xcount(s):
    c = 0
    for i in xrange(1, len(s)):
        if s[i] == "." and s[i-1] == ".":
            c += 1
    print c


n, k = map(int, raw_input().strip().split())
s = raw_input()
s = list(s)

for i in xrange(k):
    p, c = raw_input().strip().split()
    s[int(p)-1] = c
    xcount("".join(s))
