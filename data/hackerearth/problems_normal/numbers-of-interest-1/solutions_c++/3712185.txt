#include <iostream>
using namespace std;
#include<bits/stdc++.h>
int main()
{
    ////cout << "Hello World!" << endl;
    long t;
    scanf("%ld",&t);
    int x,y,n;
    long   a[100];
    while(t--)
    {
    	scanf("%d%d%d",&x,&y,&n);
    	if(n<=x)
    	printf("%d\n",y);
    	else
    	{
    		for(int i=1;i<=x;i++)
    		a[i]=y;
    		int i=0;
    		long  sum1=0,sum2=x*y;
    	for(int j=x+1;j<=n;j++)
    	{
    			a[j]=sum2-sum1;
    			sum1+=a[++i];
    			sum2+=a[j];
    			
    		}
    	 printf("%ld\n",a[n]);
    	}
    }
    return 0;
}
