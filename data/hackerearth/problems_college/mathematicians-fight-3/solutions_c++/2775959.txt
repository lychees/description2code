#include <iostream>
#include <cmath>
using namespace std;

int dp[100000];

long remaining(long n, int k) {
    long r = 0;
    for (long i = 2; i <= n; i++){
    	r = (r + k) % i;
    }
        

    return r;
}


int main()
{
    int t;
    cin>>t;
    
   
    while(t--){
    	long n;
    	cin>>n;
    	
    	long p = log(n)/log(2);
    	long ans = 2*(n-pow(2,p))+1;
    	
    	cout<<ans<<endl;
    }
    return 0;
}
