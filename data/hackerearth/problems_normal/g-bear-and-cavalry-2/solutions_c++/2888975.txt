#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	int t; cin>>t;
	while(t--)
	{
    int n; cin>>n; bool flag=1;
    int warr[n],hors[n];
    for(int i=0 ; i<n ; i++) cin>>warr[i];
    for(int i=0 ; i<n ; i++) cin>>hors[i];
    sort(hors,hors+n);
    sort(warr+1,warr+n);
    for(int i=n-1 ; i>0 ; i--)
    {
    	if(!(warr[0]*hors[n-1]>warr[i]*hors[n-i-1]))
    	{
    		flag=0; break;
    	}
    }
    if(flag==1) cout<<"YES";
    else cout<<"NO";
    cout<<endl;
	}
    return 0;
}
