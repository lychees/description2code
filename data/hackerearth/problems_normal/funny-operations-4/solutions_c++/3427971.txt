#include<bits/stdc++.h>
using namespace std;
# define ll long long
ll ar[20],ans,ans1;
void sets(int n)
{
    int cs=pow(2,n)-1,i,tmp;
    ll AND,XOR,OR;
    while(cs)
    {
        tmp=cs;
        cs--;
        vector <int> v;
        for(i=0;i<n;i++)
        {
            if(tmp&1)
                v.push_back(i);
                tmp>>=1;
        }
        XOR=ar[v[0]];
        AND=ar[v[0]];
        for(i=1;i<v.size();i++)
            XOR^=ar[v[i]],AND&=ar[v[i]];
        OR=XOR|AND;
        if(OR>ans)
            ans=OR,ans1=v.size();
        else if(OR==ans)
            ans1=max((ll)v.size(),ans1);
    }
}
int main()
{
    cin.sync_with_stdio(false);
    int t,n,i;
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(i=0;i<n;i++)
            cin>>ar[i];
        ans=ans1=-1;
        sets(n);
        cout<<ans<<" "<<ans1<<'\n';
    }
    return 0;
}
