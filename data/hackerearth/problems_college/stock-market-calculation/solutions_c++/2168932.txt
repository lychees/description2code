#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;
int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */   
    int t;
    cin>>t;
    for(int i=0;i<t;i++){
        int n;
        cin>>n;
        int a[n];
        for(int j=0;j<n;j++){
            cin>>a[j];
        }
        long long kharid=0;
        int c=0;
        int64_t bech=0;
        int pos=n-1;
        int b[n];
        b[n-1]=n-1;
        int m1=a[n-1];
        for(int j=n-2;j>-1;j--){
            if(a[j]>m1){
                m1=a[j];
                b[j]=j;
                pos=j;
            }
            else{
                b[j]=pos;
            }
        }
        for(int j=0;j<n;j++){
           bech=bech+a[b[j]]-a[j];
        }
        cout<<bech<<endl;
      }
    return 0;
}
