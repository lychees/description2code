#include<bits/stdc++.h>
using namespace std;

#define lli long long
#define pb push_back
#define mp make_pair
#define gc getchar_unlocked
#define lld "%I64d"
#define DEBUG(x) cout<<">value ("<<#x<<") : "<<x<<endl;
#define mod 1000000007

const int lmt = 1000005;

bool res[lmt];

int main(){
    int t,n,m1,m2;
    scanf("%d",&t);
    while(t--){
        scanf("%d %d %d",&n,&m1,&m2);
        for(int i=0;i<=n;i++)
            res[i] = false;
        res[m1] = res[m2] = true;
        for(int i=min(m1,m2)+1;i<=n;i++){
            res[i] = !res[i-m1];
            if(i-m2 >= 0)
                res[i] |= !res[i-m2];
        }
        if(res[n])
            printf("1\n");
        else
            printf("0\n");
    }
    return 0;
}
