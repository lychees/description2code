#include<bits/stdc++.h>
using namespace std;

#define MOD 1000000007

long long X[200010], Y[200010];
main() {
	long long N,sum=0,i,j,XS=0,YS=0;
	scanf("%lld",&N);
	for(i=0;i<N;++i) {
		scanf("%lld %lld",&X[i],&Y[i]);
		XS+=X[i]; YS+=Y[i];
	}
	sort(X,X+N); sort(Y,Y+N);
	for(i=N-1,j=0;i>=0;--i,++j) {
		XS-=X[i];
		YS-=Y[i];
		sum+=((N-j-1)*X[i])-XS;
		sum%=MOD;
		sum+=((N-j-1)*Y[i])-YS;
		sum%=MOD;
	}
	printf("%lld\n",sum);
}
