#include <bits/stdc++.h>
using namespace std;

int indextr[1000001];
int tr[1000001];

void preorder(int idx){
	if(tr[idx]==0) return;
	cout << tr[idx] << " " ;
	preorder(2*idx);
	preorder(2*idx+1);
}

int main()
{
	int t,n,i,l,x,y,z,r,idx;
	cin >> t;
	while(t--){
		cin >> n;
		if(n==0){
			cout << 1 << endl;
			continue;
		}
		memset(indextr,0,sizeof(indextr));
		memset(tr,0,sizeof(tr));
		indextr[1]=1;
		tr[1]=1;
		for(i=0;i<n;i++){
			cin >> x >> y >> z;
			idx=indextr[x];
			l=2*idx;
			r=2*idx+1;
			if(y>0){
			indextr[y]=l;
			tr[l]=y;
			}
			if(z>0){
			indextr[z]=r;
			tr[r]=z;
			}
		}
		preorder(1);
		cout << endl;
	}
    return 0;
}
