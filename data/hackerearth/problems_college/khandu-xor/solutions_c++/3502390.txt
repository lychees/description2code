#include<bits/stdc++.h>
using namespace std;
int main()
{
	string s1,s2,s3,s4,s5,s6;
	int t;
	scanf("%d",&t);
	while(t--)
	{
		cin>>s1>>s2;
		s3=s1+s2;
		s4=s2+s1;
		s5=s3;
		reverse(s3.begin(),s3.end());
		s6=s4;
		reverse(s4.begin(),s4.end());
		if(s3.compare(s5)==0||s4.compare(s6)==0)
			printf("YES\n");
		else
			printf("NO\n");		
	}
	return 0;
}
