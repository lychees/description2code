#include<iostream>
#include<math.h>
using namespace std;

int main()
{
    long long int t,z,m,n;
    cin>>t;
    while(t--){
    	long long int cnt=0;
    	cin>>z>>m>>n;

    	for(long long int i=1 ; i<=sqrt(z);i++){
    		long long int x=z/i;
    		if(i*x==z && x<=m){
    			cnt++;
    		}
    		if(i*x==z && x!=i && x<=n)
    			cnt++;
    	}
    	cout<<cnt<<endl;
    }

    return 0;
}
