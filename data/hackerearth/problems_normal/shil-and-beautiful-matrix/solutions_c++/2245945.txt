const LL base = inf + 7;
const LL MOD = inf + 9;

using namespace std;

template<typename T, typename P>
const ostream & operator << (ostream & os, const pair<T, P> & a) {
	os.precision(10);
	os << fixed;
	os << a.first;
	os << ' ';
	os << a.second;
	os << ' ';
	return os;
}

bool solve()
{	
	LL ans = 1;

	int n, m;

	cin >> n >> m;

	vec< int > a;

	for(int i =	1;i <= n;i++) {
		for(int j = 1;j <= m;j++) {
			a.push_back(i ^ j);	
		}
	}

	LL fact[1000000 + 10];
	fact[0] = 1ll;
	for(int i = 1;i < 1000000 + 10;i++) {
		fact[i] = (fact[i - 1] * LL(i)) % base;
	}

	sort(ALL(a));

	for(int i = 0;i < n * m;i++) {
		int j = i;
		while(j < n * m && a[i] == a[j]) j++; j--;
		ans = (ans * fact[j - i + 1]) % base;
		i = j;
	}

	cout << ans << '\n';

	return true;
}   

int main() 
{

	solve();

	return 0;
