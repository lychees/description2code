int get(int n,int k,int ext,int start,int step) {

	if ( n == 1 ) return start;

	int till = (n-ext+1)/2;
	if ( k <= till ) {
		int whr = ext + k*2 - 1;
		return start + step*(whr-1);
		// return ext + k*2 - 1;
	}
	int last = ext - 1 + till*2;
	// if ( last == n ) {
	// 	return get(n-till,k-till,1,start,step*2);
	// }
	return get(n-till,k-till,(last==n),(ext?start:start+step),step*2);

}
