#include <iostream>
using namespace std;
#define m 1000000007
int main()
{
	int t;
    cin>>t;
    while(t--)
    {
    	int64_t l,b,d;
    	cin>>l>>b>>d;
    	if(d>l || d>b)
    	cout<<"0\n";
    	else if(d==l && d==b)
    	cout<<"1\n";
    	else
    	{
    		l=(((l/d)%m)*((b/d)%m))%m;
    		cout<<l<<"\n";
    	}
    }
    return 0;
}
