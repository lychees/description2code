#include <iostream>
#include <climits>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++){
    	int n,m;
    	cin>>n>>m;
    	for(int j=0;j<n;j++){
    		int min=INT_MAX;
    		int a,b,c,res;
    		int count=0;
    		for(int k=0;k<m;k++){    			
    			cin>>a>>b>>c;
    			res=(100-a)*(100-b)*(100-c);    			
    			if(res<min){
    				min=res;
    				count=k;
    			}
    		}
    		cout<<count+1<<" ";
    	}
    	cout<<endl;
    }
    return 0;
}
