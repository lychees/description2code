#include <bits/stdc++.h>
 
using namespace std;
 
const int N = 1e6 + 10;
 
map <string, set < pair <int, string> > > mp;
 
int main() {
	ios::sync_with_stdio(false);
	int n, m, age;
	string teacher, student;
	cin >> n >> m;
	for (int i = 0; i < n; i++) {
		cin >> teacher;
		set < pair <int, string> > s;
		mp.insert(make_pair(teacher, s));
	}
	for (int i = 0; i < m; i++) {
		cin >> teacher >> student >> age;
		auto it = mp.find(teacher);
		assert(it != mp.end());
		auto &s = it->second;
		s.insert(make_pair(age, student));
	}
	for (auto it = mp.begin(); it != mp.end(); ++it) {
		cout << it->first << '\n';
		auto &s = it->second;
		for (auto it2 = s.begin(); it2 != s.end(); ++it2) {
			cout << it2->second << " " << it2->first << '\n';
		}
	}
	return 0;
}
