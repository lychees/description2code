#include <iostream>
using namespace std;
int N[1001] ;
int minTail[1001] ;
 int t ;
 
 void solve(){
     minTail[0] = -1001 ;
     for( int i = 1 ; i <= t ; i ++ ){
     	
         minTail[i] = 1001 ;
         for( int j = 1 ; j <= i ; j++ ){
         	  
         	   if(  N[i] < minTail[j] && N[i] > minTail[j-1] ){
         	   	 
         	   	   minTail[j] = N[i] ;
         	   	
         	   }
         	   
         	  
                           	
         }
     	
     }
     int result = 0  ;
     for( int i = t ; i >0 ; i -- ){
     	
     	if( minTail[i]< 1001){
     		   result = i ;
     		   break ;
     		
     	}
     }
     
     int mod[11] ;
     int k = 0 ;
     while( result ){
     	
         mod[k] = result%2 ;
         
         result = result/2 ;
         
         k++ ;
     	
     }
     while( k>0 ){
     	
     	cout<<mod[k-1];
     	k-- ;
     	
     }
     cout<<endl ;
 
 }
int main()
{
    
     cin>>t ;
     for( int i = 1 ; i <= t ; i ++ ){
     	cin>>N[i] ;
     	
     }
     solve() ;
     
     return 0;
}
