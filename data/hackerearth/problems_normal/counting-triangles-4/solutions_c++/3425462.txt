#include <iostream>
#include <string>
using namespace std;
void numberOfTriangle(unsigned long long int N, unsigned long long int t){
	if(t>=1){
		unsigned long long int ans;
		ans=8*N+6*(N-1);
		cout << ans << endl;
		
	}
	
}
int main(){
    unsigned long long int t;
    unsigned long long int N;

    cin >> t;
    while(t>=1){

        cin >> N;
        numberOfTriangle(N,t);
        t--;
    }
    return 0;
}
