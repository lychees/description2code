
#include<stdio.h>
#include<string.h>


int main()
{
    int n,aa[1000],bb[1000],c;
    double m=0;
    
    
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {
    	
    	scanf("%d",&aa[i]);
    }
    for(int i=0;i<n;i++)
    {
    
    	scanf("%d",&bb[i]);
    	
    }
    
    scanf("%d",&c);
    for(int i=0;i<n;i++)
    {
        double s=(double)aa[i]/bb[i]*c;
    	if(m<s)
    	m=s;
    }
    char abc[1000];
    int len;
    sprintf(abc,"%.5lf",m);
    len =strlen(abc);
    abc[len-2]='\0';
    //strncpy(str2,str,len-1);
    //str2[len]='\0';
    printf("%s\n",abc);
    return 0;
}
