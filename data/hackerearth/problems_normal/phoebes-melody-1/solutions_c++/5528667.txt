#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,k;
    	cin>>n>>k;
    	int a[n];
    	for(int i=0;i<n;i++)cin>>a[i];
    	int b[n];
    	for(int i=0;i<n;i++)b[i]=n+1;
    	for(int i=0;i<n;i++)
    	 for(int j=k; ;j++)
    	  if((i+j<n && a[i+j]<a[i])||(i-j>=0 && a[i-j]<a[i]))
    	   {b[i]=j;break;}
    	  else if(i+j>=n && i-j<0)
    	   	break;
    	int ans=0;
    	for(int i=0;i<n;i++)
    	 if(b[i]<n+1)
    	  ans+=b[i];
    	cout<<ans<<endl;
    }
    return 0;
}
