#include<bits/stdc++.h>
using namespace std;
#define lli long long int
#define f(i,a,b) for(i=a;i<b;i++)
#define si(a) scanf("%d",&a);
#define slli(a) scanf("%lld",&a);
#define MOD 1000000007
#define MAX 100005
#define pb(a) push_back(a)
int main()
{
	lli i,j,k,m,n,t,p,u,r,s,cas=1;
	slli(t);
	while(t--)
	{
		slli(n);
		slli(p);
		slli(u);
		slli(r);
		slli(s);
		f(i,0,n)
		{
			if(p+r<=u)
				p+=r;
			else if(p-s>=0)
				p-=s;
			else
				break;
		}
		printf("Case #%lld: %lld\n",cas,p);
		cas++;
	}
}
