#include <iostream>
#include <cstdio>
#include <cmath>
#include <cstring>
#include <functional>
#include <algorithm>

using namespace std;

int main()
{
	string a,b;
	cin >> a;
	cin >> b;
	bool flag;
	flag = 0;
	long len,count;
	count = 0;
	len = a.length();
	for(int i = 0 ; i < len ; i++)
	{
		if(a[i] != b[i])
		{
			count++;
		}

		if(count > 1)
		{
			cout << "loses" << endl;
			flag = 1;
			break;
		}
	}
	if(flag == 0)
	{
		cout << "wins"<< endl;
	}
	return 0;
}
