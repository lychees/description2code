#include<bits/stdc++.h>
using namespace std;
typedef vector<pair<double,double> > vii;
typedef pair<double,double> ii;
const int N = 100500;
const int M = 201;//MAGIC
vii p;
vii cur;
ii rota(ii a,double angle)
{
	ii resu;
	resu.first = a.first*cos(angle) - a.second*sin(angle);
	resu.second = a.first*sin(angle) + a.second*cos(angle);
	return resu;
}
double init(double angle,int n)
{
	for(int i=0;i<n;i++)
	{
		ii temp = rota(p[i],angle);
		cur.push_back(temp);
	}
	double minx,maxx,miny,maxy;
	minx=miny= 1e9;
	maxx=maxy= -1e9;
	for(int i=0;i<n;i++)
	{
		minx = min(minx,cur[i].first);
		miny = min(miny,cur[i].second);
		maxx = max(maxx,cur[i].first);
		maxy = max(maxy,cur[i].second);
	}
	cur.clear();
	return max(maxy-miny,maxx-minx);
}
int main()
{
	int n;
	double x,y;
	scanf("%d",&n);
	for(int i=0;i<n;i++)
	{
		scanf("%lf%lf",&x,&y);
		p.push_back(make_pair(x,y));
	}
	double ans = 1e9;
	for(int j=0;j<M;j++)
	{
		double l,r;
		l = j*M_PI / M;
		r = (j+1)*M_PI / M;
		for(int i=0;i<65;i++)
		{
			double mid1, mid2;
    			mid1 = l * 2 + r;
    			mid2 = l + 2 * r;
    			mid1 /= 3;
    			mid2 /= 3;
    			double res1 = init(mid1,n);
    			double res2 = init(mid2,n);
    			if (res1 < res2)
    				r = mid2;
    			else
    				l = mid1;
    			ans = min(ans, res1);
    		}
	}
	printf("%.4lf\n",ans);
	return 0;
}
