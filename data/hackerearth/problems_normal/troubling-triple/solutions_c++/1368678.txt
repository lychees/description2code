// const int INF = 1e9;
const int MAX_N = 2222;
const int MAX_A = 1e6 + 10;
int N, K;
int A[MAX_N];

int ft[MAX_A];
void update(int i, int x){
  for(; i < MAX_A; i += i & -i)
    ft[i] += x;
}

int query(int i){
  int s = 0;
  for(; i > 0; i -= i & -i)
    s += ft[i];
  return s;
}

int main(){
  scanf("%d%d", &N, &K);
  for(int i = 0; i < N; i++){
    scanf("%d", A + i);
  }

  ll res = 0;
  for(int i = 0; i < N; i++){
    res += query(K / A[i]);
    for(int j = 0; j < i; j++){
      if(1LL * A[i] * A[j] <= K){
	update(A[i] * A[j], 1);
      }
    }
  }
  printf("%lld\n", res);
  return 0;
}
