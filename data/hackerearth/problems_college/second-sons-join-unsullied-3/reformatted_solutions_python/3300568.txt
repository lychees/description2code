def answer(S):
    u = s = a = b = 0
    for i, v in enumerate(S):
        if v == 'U':
            u += 1
            a += i % 2
            b += ~i % 2
        if v == 'S':
            s += 1
            a += ~i % 2
            b += i % 2
    if u == s+1:
        return a/2
    if s == u+1:
        return b/2
    if u == s:
        return min(a/2, b/2)
    return -1


r = raw_input
for t in xrange(int(r())):
    print answer(r())
