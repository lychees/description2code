#include<bits/stdc++.h>
using namespace std;
char S[503][23];
bool satisfied[503];
int main(){
    int tc;cin>>tc;
    while(tc--){
        int n,k;cin>>n>>k;
        for(int i=1;i<=n;++i)
        {
            scanf("%s",S+i);
        }
        int ans=n;
        for(int j=0;j<(1<<k);++j){
            int som=j;
            int pos=0;
            memset(satisfied,false,sizeof satisfied);
            int dish_cnt=0;
            while(som>0){
                if(som&1){
                    dish_cnt++;
                    for(int i=1;i<=n;++i){
                        if(S[i][pos]=='1'){
                            satisfied[i]=true;
                        } 
                    }
                }
                pos++;
                som>>=1;
            }
            bool flag=true;
            for(int i=1;i<=n;++i){
                if(satisfied[i]==false){
                    flag=false;break;
                }
            }
            if(flag==true){
                ans=min(ans,dish_cnt);
            }
        }
        cout<<ans<<"\n";
    }
    return 0;
}
