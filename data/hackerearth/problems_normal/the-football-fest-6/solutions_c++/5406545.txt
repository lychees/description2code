#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    stack<int> st;
    while(t--)
    {
        long long int n,id,k,l;
        cin>>n;
        stack<int> st;
        cin>>id;st.push(id);
        while(n--)
        {
            char c;
            cin>>c;
            if(c=='P')
            {
                cin>>id;
                st.push(id);
            }
            else
            {
                k=st.top();st.pop();
                l=st.top();st.pop();
                st.push(k);
                st.push(l);
            }
        }
        cout<<"Player "<<st.top()<<endl;
    }
    return 0;
}
