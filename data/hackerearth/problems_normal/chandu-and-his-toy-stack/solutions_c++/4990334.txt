#include <bits/stdc++.h>
using namespace std;

int main()
{
    int tc;
    long int size,add,rem,i,ans;
    cin >> tc;
    while(tc--)
    {
        ans = 0;
        cin >> size >> add >> rem;
        long int arr1[size],arr2[size];
        for(i=0;i<size;i++)
        {
            cin >> arr1[i] >> arr2[i];
        }
        sort(arr1,arr1+size);
        sort(arr2,arr2+size);
        for(i=size-1;i>=0;i--)
        {
            long diff = arr1[i]-arr2[i];
            if(diff > 0)
            {
                ans+=diff*rem;
            }
            else
            {
                diff*=-1;
                ans+=diff*add;
            }
        }
        cout << ans << endl;
    }
    return 0;
}
