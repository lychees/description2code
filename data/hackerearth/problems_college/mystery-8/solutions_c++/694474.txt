#include<iostream>
#include<cstdio>
#include<string.h>
using namespace std;
int main()
{
    int test_cases;
    cin>>test_cases;
    getchar();
    while(test_cases>0)
    {
        char str[101];
        gets(str);
        // your code here
        int len = strlen(str);
        int char_len = 0;
        char new_str[101]="";
        
        for(int i=0;i<len;i++)
        {
        	if(str[i]==' ' || str[i]=='!' || str[i]=='?' || str[i]=='.' || str[i]==',')
        	  continue;
        	new_str[char_len++]=str[i];
        }
        for(int i=0;i<char_len;i++)
        {
        	if(new_str[i]=='*')
        	 cout<<new_str[char_len-i-1]<<endl;
        }
        test_cases--;
    }
    return 0;
}
