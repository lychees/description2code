t = int(raw_input())


for _ in range(t):

    n, m, x = map(int, raw_input().split())

    ai = [int(i) for i in list(raw_input().split())]

    # print ai

    proj_score = x*(n+1)
    score = sum(ai)

    if score >= proj_score:
        print "1"

    elif proj_score - score > m:
        print "Impossible"

    else:
        print ""+str(proj_score - score)
