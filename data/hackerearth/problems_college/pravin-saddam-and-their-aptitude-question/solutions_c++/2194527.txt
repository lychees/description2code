#include <stdio.h>

int main()
{
   	long long int t,a,b,temp,fac,sum,ud;
   	scanf("%lld",&t);
   	while(t--)
   	{
   		sum=0;
   		scanf("%lld%lld",&a,&b);
   		while(a<=b)
   		{
   			fac=1;
   			temp=a;
   			while(temp!=0)
   			{
   				fac=fac*temp;
   				temp=temp-1;
   			}
   			sum=sum+fac;
   			a++;
   		}
   		ud=sum%10;
   		printf("%lld\n",ud);
   	}
   	return 0;
}
