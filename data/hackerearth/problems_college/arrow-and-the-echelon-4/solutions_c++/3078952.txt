#include <bits/stdc++.h>
using namespace std;
#define int64 long long int
int main()
{
    int64 t,i,f,x;double z,y;
    cin>>t;
    while(t--)
    {
    	f=0;
    	cin>>x;
    	for(i=x;i>=0;i--)
    	{
    		if(i%3==0)
    		{
    		z=i;
    		y=x-z;
    		if(y==ceil(y) && (int)y%5==0)
    		{
    			f=1;break;
    		}
    		}
    	}
    	if(f==0)
    		cout<<"-1"<<endl;
    	else
    		{
    			for(i=0;i<z;i++)
    			cout<<"5";
    			for(i=0;i<y;i++)
    			cout<<"3";
    			cout<<endl;
    		}
    }
    return 0;
}
