#define PII       pair<int,int>

int marker[1000001] = {0};
void sieve(int limit = 1000000)
{
	for(int i = 2; i <= limit; i++) marker[i] = 1;
	for(int i = 2; i <= limit; i++)
	{
		if(!marker[i]) continue;
		if(1LL*i*i > limit) break;
		int j = i*i;
		while(j<=limit)
		{
			marker[j] = 0;
			j += i;
		}
	}
}

int valid(int i)
{
	vector<int> digs;
	while(i)
	{
		digs.pb(i%10);
		i /= 10;
	}
	int val = 0;
	int tenp = 1;
	for(int i = 0; i < sz(digs); i++)
	{
		val = val + digs[i] * tenp;
		tenp *= 10;
		if(marker[val]) return 1;
	}
	return 0;
}

void pre()
{
	for(int i = 2; i <= 1000000; i++)
		if(marker[i] or valid(i))
			marker[i] = 1;
}

int arr[1000010];

int main()
{
	int n;
	cin(n);
	for(int i = 1; i <= n; i++) cin(arr[i]);
	for(int i = 1; i <= n; i++)
		arr[i] += arr[i-1];
	int q;

	while(q--)
	{
		int l,r;
		cin(l);
		cin(r);
		printf("%d\n", arr[r] - arr[l-1]);
	}
	return 0;
}
