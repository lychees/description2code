for _ in xrange(int(raw_input())):
    a, b, c, d = [int(x) for x in raw_input().split(' ')]
    if a > b:
        a, b = b, a
    if c > d:
        c, d = d, c

    if d < a:
        a, c = c, a
        b, d = d, b

    if b == c:
        print "Point"
    elif c - b > 1:
        print "Line"
    elif c - b == 1:
        print "Line"
    else:
        if a == b or c == d:
            print "Point"
        else:
            print "Nothing"
