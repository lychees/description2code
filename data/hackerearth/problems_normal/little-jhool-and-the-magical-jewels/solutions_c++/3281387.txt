#include <iostream>
#include<map>
#include<string.h>
#include<algorithm>
using namespace std;

int main()
{
    int t,result;
    char ch[100];
    cin>>t;
    while(t--)
    {
    	map<char,int> m;
    	cin>>ch;
    	for(int i=0;i<strlen(ch);i++)
    	{
    		m[ch[i]]++;
    	}
    	result=min(m['r'],min(m['u'],min(m['b'],m['y'])));
    	cout<<result<<"\n";
    }
    return 0;
}
