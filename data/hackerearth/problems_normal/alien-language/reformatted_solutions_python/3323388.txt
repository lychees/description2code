from string import lowercase as lc
for _ in range(input()):
    ans = "NO"
    small = {}
    for x in lc:
        small[x] = 0
    src = raw_input().strip()
    for x in src:
        small[x] += 1
    tgt = raw_input().strip()
    for x in tgt:
        if small[x] > 0:
            ans = "YES"
            break
    print ans
