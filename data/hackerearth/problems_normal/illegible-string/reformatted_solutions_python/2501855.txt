stri = raw_input()

arr = list(stri)

# print arr

all_v = ""

for x in arr:
    if x == "w":
        all_v += "vv"
    else:
        all_v += x


all_w = ""
x = 0
while x < len(all_v)-1:

    if all_v[x] == "v" and all_v[x+1] == "v":
        all_w += "w"
        x += 2
    else:
        all_w += all_v[x]
        x += 1
if x < len(all_v):
    all_w += all_v[len(all_v) - 1]
