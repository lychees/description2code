#include <iostream>
#include <algorithm>
 
using namespace std;
 
int main() {
    int n;
    scanf("%d", &n);
    while (n--) {
        string a, b;
        cin >> a >> b;
        if (a.size() > b.size()) swap(a, b);
        int max = a.size();
        for (int i = 0; i < b.size() - a.size() + 1; i++) {
			int d = 0;
			for (int j = 0; j < a.size(); j++) {
				if (a[j] != b[j + i]) d++;
			}
			max = min(max, d);
		}
		cout << max << endl;
    }
    return 0;
}
