#include <iostream>
#include <vector>
#include <cmath>
#include <climits>
#include <iomanip>
using namespace std;

int main()
{
    int n;
    cin>>n;
    vector<int> x(n,0);
    vector<int> y(n,0);
    vector<int> f(n,0);
    for(int i = 0;i <n;i++)
    {
        cin>>x.at(i);
        cin>>y.at(i);
        cin>>f.at(i);
    }
    vector<double> happy(n);
    happy.at(0) = f.at(0);
    for(int i = 1;i<n;i++)
    {
        double maximum = double(INT_MIN);
        for(int j = 0;j<i;j++)
        {
            double c = f.at(i)+happy.at(j) - sqrt(pow(x.at(i)-x.at(j),2) + pow(y.at(i)-y.at(j),2));
            if(maximum < c)
            {
                maximum = c;
            }
        }
        happy.at(i) = maximum;
    }
    cout << setprecision(6) << fixed <<  happy.at(n-1) << endl;
    return 0;
}
