using namespace std;

void solve(void) {
	int n,t,ct,rem;
	cin>>n>>ct>>t;
	if(t>ct)
		swap(t,ct);
	rem=(ct-t);
	if(rem>1) cout<<"Motu\n";
	else	cout<<"Chotu\n";
}

int main() {
	int i,t;
	scanf("%d",&t);
	for(i=1;i<=t;i++) {
		solve();
	}
	return 0;
}
