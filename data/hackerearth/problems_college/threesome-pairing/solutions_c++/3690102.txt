#include <iostream>

using namespace std;

int main()
{
	int T, N, A[101];

	cin >> T;

	while (T--)
	{
		cin >> N;

		for (int i = 0; i < N; i++)
			cin >> A[i];

		int ans = 0;
		
		for (int i = 0; i < N; i++)
		{
			for (int j = i+1; j < N; j++)
			{
				if ((A[i] + A[j]) % 3 == 0)
					ans++;
			}
		}

		cout << ans << endl;
	}

	return 0;
}
