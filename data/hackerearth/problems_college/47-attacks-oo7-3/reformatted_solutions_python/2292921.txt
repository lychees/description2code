for _ in range(int(raw_input())):
    n = int(raw_input())
    res = 1
    if n == 0:
        print "0"
        continue
    elif n > 120:
        print "UNABLE TO FIRE"
        continue
    for i in range(1, n+1):
        if i % 2 != 0:
            res *= 3
        if i % 2 == 0:
            res += 2
    print res
