#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int arr[27]={0};
    	string s;
    	cin>>s;
    	for(int i=0;i<s.length();i++)
    	{
    		arr[s[i]-97]++;
    	}
    	int cnt=0;
    	cnt=min(arr[17],min(arr[20],min(arr[1],arr[24])));
    	cout<<cnt<<endl;
    }
    return 0;
}
