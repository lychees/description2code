t = int(raw_input())
for i in range(t):
    n = int(raw_input())
    if n == 1:
        single = int(raw_input())
        if ((single) & (single-1) == 0) and single != 0:
            print "YES"
        else:
            print "NO"
    else:
        l = list()
        done = 0
        inpStr = raw_input()
        l = inpStr.split()
        for j in range(32):
            temp = 1 << j
            currentAnd = None
            for k in range(n):
                elem = int(l[k])
                if temp & elem == temp:
                    if currentAnd is None:
                        currentAnd = elem
                    else:
                        currentAnd = currentAnd & elem
            if currentAnd == temp:
                done = 1
                break
        if done == 0:
            print "NO"
        else:
            print "YES"
