#include<bits/stdc++.h>
using namespace std;
#define ll long
#define MOD 1000000007


int main (){
    ll t,a,b,c;
    cin>>t;
    while(t--){
        cin>>a>>b>>c;
        if(a>b){
            a^=b;
            b^=a;
            a^=b;
        }
        if(a>c){
            a^=c;
            c^=a;
            a^=c;
        }
        if(b>c){
            b^=c;
            c^=b;
            b^=c;
        }
        cout<<(((a%MOD*(b-1)%MOD)%MOD*(c-2)%MOD))%MOD<<"\n";
    }
    return 0;
}
