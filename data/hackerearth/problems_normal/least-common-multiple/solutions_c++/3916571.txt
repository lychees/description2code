#include <bits/stdc++.h>

using namespace std;

long long n;

bool primepower(long long x)
{
    for(long long i=2; i*i<=x; i++) if(x%i==0)
    {
        while(x%i==0)
            x/=i;
        return x==1;
    }
    return true;
}

int main()
{
    scanf("%lld", &n);
    if(n<=2)
        return printf("%lld\n", n), 0;
    long long x=n/2+1;
    while(!primepower(x))
        x++;
    printf("%lld\n", x);
    return 0;
}
