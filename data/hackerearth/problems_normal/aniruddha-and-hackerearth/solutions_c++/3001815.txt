#include<bits/stdc++.h>
#define mod 1000000007
#define ll long long 
using namespace std;
ll dp[2][1000005], ans[1000005];

int main(){
	
	dp[0][1]=dp[1][1]=1;
	for(int i=2; i<=1000000; i++){
		dp[1][i]=dp[0][i-1];
		dp[0][i]=(dp[0][i-1]+dp[1][i-1])%mod;
	}
	for(int i=1; i<=1000000; i++){
		ans[i]=(dp[0][i]+dp[1][i]+ans[i-1])%mod;
	}
	
	int t, n;
	cin>>t;
	while(t--){
		cin>>n;
		cout<<ans[n]<<"\n";
	}
	return 0;
}
