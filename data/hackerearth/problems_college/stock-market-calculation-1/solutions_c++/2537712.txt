#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t,a[100000];
   cin >> t;
   while(t--)
   {
   	  int n;
   	  scanf("%d",&n);
   	  for(int i=0; i< n; i++)
   	  {
   	  	 scanf("%d",&a[i]);
   	  	 
   	  }
   	  int profit=0;
   	  int max_=a[n-1];
   	  for(int i=n-2; i>=0; i--)
   	  {
   	  	if(a[i] < max_)
   	  	{
   	  		profit=profit + (max_-a[i]);
   	  	}
   	  	if(a[i] > max_)
   	  	max_=a[i];
   	  }
   	  printf("%d\n",profit);
   }
    return 0;
}
