#include <iostream>
#include <limits.h>
#include<algorithm>
using namespace std;

int main()
{
   long long int k,i,x;
   
   cin>>k;
   vector<long long int> arr(k);
   for(i=0;i<k;i++){
   	cin>>arr[i];
 
    }
   long long int days=1;
   sort(arr.begin(), arr.end(), greater<int>());
   x=arr[0];
   for(i=0;i<k-1;i++){
   if(x<arr[i+1]+days){
   	x=arr[i+1]+days;
   } 	
    days=days+1;
  }
   cout<<(x+2);
   return 0;
}
