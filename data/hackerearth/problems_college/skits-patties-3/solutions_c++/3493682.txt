#include <bits/stdc++.h>
#define ll long long int

using namespace std;

int a[10000007];

int main()
{
    memset(a, 0, sizeof(a));
    a[6] = a[9] = a[20] = 1;
    for (int i = 6; i <= 10000000; i++) {
        a[i] += a[i - 6];
    }
    for (int i = 9; i <= 10000000; i++) {
        a[i] += a[i - 9];
    }
    for (int i = 20; i <= 10000000; i++) {
        a[i] += a[i - 20];
    }

    int t, u;
    cin >> t;
    while (t--) {
        cin >> u;
        if (a[u] > 0) {
            cout << "True\n";
        } else {
            cout << "False\n";
        }
    }
    return 0;
}
