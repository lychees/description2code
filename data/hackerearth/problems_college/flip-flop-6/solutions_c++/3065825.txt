#include <iostream>
#include <string.h>

using namespace std;

int main()
{
	int x,i,k,del;
	string mystr;
	char c,a[100002];
	int n;
	cin>>n;
	while(n--){
		cin>>mystr;
		strcpy(a, mystr.c_str());
		x=strlen(a);
		i=0;
		del=0;
		c=a[i];
		for(k=i+1;k<x;k++){
		if(a[k]==c)
			del++;
		else
			c=a[k];
		}
		//if(a[x-1]==c)
		//	del++;
		cout<<del<<endl;
	}
    
    return 0;
}
