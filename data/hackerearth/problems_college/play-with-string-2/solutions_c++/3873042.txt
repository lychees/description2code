#include <iostream>
#include<string.h>
using namespace std;
int main()
{
	int t,temp,i,a,b;
	cin>>t;
	string s;
	while(t--)
	{
		temp=0;
		cin>>s;
		int l1=s.length();
		for(i=0;i<(s.length())/2;i++)
		{
			a=abs(s[i+1]-s[i]);
			b=abs(s[l1-1]-s[l1-2]);
			if(a==b)
			{
				temp=1;
				l1--;
			}
			else
			{
				cout<<"Not Funny\n";
				temp=0;
				break;
			}
		}
		if(temp)
		cout<<"Funny\n";
	}
    return 0;
}
