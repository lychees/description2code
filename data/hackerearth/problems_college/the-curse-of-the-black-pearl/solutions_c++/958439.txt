#include <iostream>
#include<cstdio>
#include<cstring>
using namespace std;

int main() {
	int t=882;
	int time,temp;
	char ar[32801];
	while(t--)
	{
		time=0;
		scanf("%s",ar);
		int num=strlen(ar);
		num--;
		
		while(num!=0 )
		{
			
			if(ar[num]=='1' && num>0)
			{
				time++;
				temp=num;
				while(ar[temp]=='1')
				{
					temp--;
					if(temp==0)
					{
						ar[0]='1';
						num++;
						for(int i=1;i<=num;i++)
						ar[i]='0';
						break;
					}
					
				}
				if(temp!=0)
				{
				ar[temp]='1';
				for(int i=temp+1;i<=num;i++)
				ar[i]='0';
				}
			}
			while(ar[num]=='0')
			{
				num--;
				time++;
				
			}
		}
		
		cout<<time<<endl;
	}
	return 0;
}
