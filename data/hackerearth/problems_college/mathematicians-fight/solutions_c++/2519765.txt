#include<iostream>
#include<math.h>
#include<string>
using namespace std;
int josephus(int n)
{
  return 2*(n-pow(2,floor(log2(n*1.0))))+1;
}

int main()
{
    int test;
    cin>>test;
    while(test--){
        int x;
        cin>>x;
        cout<<josephus(x)<<endl;

    }
    return 0;
}
