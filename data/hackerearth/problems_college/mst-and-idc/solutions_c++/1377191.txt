#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t,n,i,j,ans;
	scanf("%d",&t);
	while (t --) {
		scanf("%d",&n);
		int a[n];
		for (i = 0; i < n; i++) {
			scanf("%d",&a[i]);
		}
		sort(a,a+n);
		ans = 0;
		for (i = 1; i < n; i++) {
			ans += a[i]-a[i-1];
		}
		printf("%d\n",2*ans);
	}
	
	return 0;
}
