#include <iostream>
#include<bits/stdc++.h>
#define ll long long 
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    ll A,B;
    ll ADD[100005];
    string s;
    cin>>A>>B;
    ADD[0]=A%B;
    cin>>s;
    for(int i=1;i<s.size();i++)
    {
    	ADD[i]=((ADD[i-1]%B)*(ADD[i-1]%B))%B;
    }
    ll res=0;
    for(int i=0;i<s.size();i++)
    {
    	if(s.at(i)=='1')
    	res=((res%B)+(ADD[i]%B))%B;
    }
    cout<<res<<endl;
    
    return 0;
}
