#include <bits/stdc++.h>

using namespace std;

#define ll long long int
#define s(x) scanf("%d", &x)
#define sl(x) scanf("%lld", &x)
#define pb push_back
#define mp make_pair
#define mod 1000000007

int main()
{
    int t, ans, i, k;
    char s[100000];
    gets(s);
    t = atoi(s);
    while (t--) {
        gets(s);
        k = strlen(s);
        ans = 0;
        string h;
        for (i = 0; i < k; i++) {
            ans += s[i];
            h.pb(s[i]);
        }
        ans /= k;
        if (ans % 2)
            reverse(h.begin(), h.end());
        cout << h << endl;
    }
    return 0;
}
