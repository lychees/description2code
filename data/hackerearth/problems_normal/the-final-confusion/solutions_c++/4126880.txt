#include <iostream>
using namespace std;
#define ll long long
ll a[31][31];
ll ck(ll n, ll m){
    if(n==0 && m==0){
        return 0;
    }
    if(n==0){
        return 1;
    }
    if(a[n][m]!=-1){
    	return a[n][m];
    }
    if(m==0){
        a[n][m]= ck(n-1,m+1);
    }
    else
    	a[n][m]= ck(n-1,m+1)+ck(n,m-1);
    return a[n][m];
}

int main()
{
    ll t,n;
    cin>>t;
    while(t--){
        cin>>n;
        for(int i=0;i<=n;i++){
        	for(int j=0;j<=n;j++){
        		a[i][j]=-1;
        	}
        }
        cout<<ck(n,0)<<endl;
    }
    
    return 0;
}
