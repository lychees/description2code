#include<bits/stdc++.h>
using namespace std;
int main()
{
    string s;
    cin>>s;
    int arr[26]= {0};
    int len = s.size();
    for(int i = 0 ; i < len; i++)
    {
        arr[ s[i] - 'a']++;
    }
    int flag = 0;
    for(int i = 0;i<26;i++)
    {
    if(arr[i]%2 == 1)
    flag++;
    }
    if(flag>1)
    cout<<"NO\n";
    else
    cout<<"YES\n";
}
