from collections import defaultdict
map = defaultdict(int)


def func(n):
    if n == 0:
        return 0
    elif map[n] == 0:
        map[n] = max(n, func(n/2) + func(n/3) + func(n/4))
    return map[n]


while True:
    try:
        n = input()
        print func(n)
    except BaseException:
        break
