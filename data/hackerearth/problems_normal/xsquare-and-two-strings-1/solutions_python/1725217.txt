t = int(raw_input())
for i in range(t):
    str1 = list(set(raw_input()))
    str2 = list(set(raw_input()))
    done = False
    if len(str1) < len(str2):
        for char in str1:
            if char in str2:
                print 'Yes'
                done = True
                break
    else:
        for char in str2:
            if char in str1:
                print 'Yes'
                done = True
                break
    if not done:
        print 'No'
