#include <bits/stdc++.h>
using namespace std;

#define MOD 1000000007LL
#define ll long long

ll power(ll a,ll b)
{
	ll res = 1;
	while(b)
	{
		if(b&1)
		{
			res *= a;
			res %= MOD;
		}
		a *= a;
		a %= MOD;
		b /= 2;
	}
	return res;
}

#define MAXN 100010
ll A[MAXN];

int main()
{
	int n;
	scanf("%d",&n);
	ll res = 1;
	int z = 0;
	for(int i=1;i<=n;++i)
	{
		scanf("%lld",A+i);
		if(A[i]==0)
			++z;
		else
		{
			res *= A[i];
			res %= MOD;
		}
	}
	int q;
	scanf("%d",&q);
	while(q--)
	{
		int type, idx, val;
		scanf("%d",&type);
		if(type==0)
		{
			scanf("%d%d",&idx,&val);
			if(A[idx]==0)
				--z;
			else
			{
				res *= power(A[idx],MOD-2);
				res %= MOD;
			}
			A[idx] = val;
			if(A[idx]==0)
				++z;
			else
			{
				res *= A[idx];
				res %= MOD;
			}
		}
		else
		{
			scanf("%d",&idx);
			if(z>1 or ((z==1) and A[idx]!=0))
				printf("0\n");
			else if(z==1)
				printf("%lld\n",res);
			else
				printf("%lld\n",(res*power(A[idx],MOD-2))%MOD);
		}
	}
	return 0;
}
