bool prime[1000001];
vector<int> vec;
void seivec()
{
	prime[0]=prime[1]=true;
	for(int i=2;i*i<100001;i++)
		if(prime[i]==false)
			for(int j=2*i;j<100001;j+=i)
				prime[j]=true;
	for(int i=2;i<1000001;i++)
		if(prime[i]==false)
			vec.pb(i);	
}
int main()
{
	seivec();
	int t;
	si(t);
	while(t--)
	{
		int n,f=0;
		si(n);
		for(int i=0;i<vec.size();i++)
		{
			if(vec[i]>n/2)
				break;
			if(prime[n-vec[i]]==false)
			{
				f=1;

			}
		}
