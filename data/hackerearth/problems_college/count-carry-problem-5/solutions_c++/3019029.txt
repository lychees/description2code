#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main(){
	int t;
	cin>>t;
	while(t--){
		string a,b;
		cin>>a>>b;
		ll len1=a.length();
		ll len2=b.length();
		ll len=len1;
		if(len2<len1){
			len=len2;
		}
		int carry=0;
		ll i=len1-1;
		ll j=len2-1;
		int count=0;
		for(ll k=0;k<len;k++){
			if( ((a[i]-'0')+(b[j]-'0')) + carry > 9 ){
				carry=((a[i]-'0')+(b[i]-'0'))/10;
	//			cout<<a[i]<<" "<<b[j]<<" "<<carry<<endl;
				count++;
			}else if(((a[i]-'0')+(b[j]-'0')) + carry <=9){
				carry=0;
	//			cout<<a[i]<<" "<<b[j]<<" "<<carry<<endl;
			}
			i--;
			j--;
		}
	//	if(carry==1)count++;
		if(count==0){
			cout<<"No carry operation"<<endl;
		}else if(count==1){
			cout<<"1 carry operation"<<endl;
		}else{
			cout<<count<<" carry operations"<<endl;
		}
	}
	return 0;
}
