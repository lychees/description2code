#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    long long int ans;
    vector<long long int> x(3);
    cin>>t;
    while(t--){
    	cin>>x[0]>>x[1]>>x[2];
    	sort(x.begin(),x.end());
    	ans=x[0];
    	ans%=1000000007;
    	ans*=(x[1]-1);
    	ans%=1000000007;
    	ans*=(x[2]-2);
    	ans%=1000000007;
    	cout<<ans<<endl;
    }
    
    return 0;
}
