#include<iostream>
using namespace std;

int main()
{
	long long n, x, count=0;
	int check = 0;
	cin >> n >> x;
	long long a[n];
	for(long long i=0; i<n; i++)
	cin >> a[i];
	for(long long i=0; i<n && check<2; i++)
	{
		if(a[i] > x)
		{
			check++;
			count--;
		}
		count++;
	}
	cout << count << "\n";
	return 0;
}
