#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,n,i,j,k,s,a[1000000];
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	for(i=0;i<n;i++)
    	{
    		scanf("%d",&a[i]);
    	}
    	k=0;
    	for(i=n-1;i>0;i--)
    	{
    		for(j=0;j<i;j++)
    		{
    			if(a[i]==a[j])
    			k++;
    		}
    	}
    	printf("%d\n",k+n);
    }
    return 0;
}
