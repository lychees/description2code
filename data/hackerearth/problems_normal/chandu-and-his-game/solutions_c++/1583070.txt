
#include<limits.h>
#include<climits>
#include<stdlib.h>
#include<cstdio>
#include<list>
#include <bits/stdc++.h>
#include<math.h>
using namespace std;

int main()
{
	
	long long int n,c;
	cin>>n>>c;
	pair<long long int,int>value[c];
	for(int i=0;i<c;i++)
	{
		long long int x;
		cin>>x;
		value[i].first=x;
	}
	for(int i=0;i<c;i++)
	{
		long long int y;
		cin>>y;
		value[i].second=y;
	}
	 long long int final=0;
	for(int i=0;i<c;i++)
	{
		long long int a=abs(value[i].first-1)+abs(value[i].second-1);
		long long int b=abs(value[i].first-1)+abs(value[i].second-n);
		long long int c=abs(value[i].first-n)+abs(value[i].second-1);
		long long int d=abs(value[i].first-n)+abs(value[i].second-n);
		long long int res=min(a,min(b,min(c,d)));
		final=final+res;
	}
	cout<<final<<endl;
    //cout << "Hello World!" << endl;
    return 0;
}
