#include <iostream>
using namespace std;

int main()
{
	int n;
	string s;
	cin >> n;
	while(n--) {
		cin >> s;
		int x = 0;
		int count = 0;
		
		for (int i = 0; i < s.size(); i++) {
			if (s[i] == '_') {
				count++;
			} else if (s[i] == '|') {
				x ^= count;
				count = 0;
			}
		}
		if (count == s.size()) {
			cout << "Bob\n";
			continue;
		}
		x ^= count;
		if (x == 0) cout << "Bob\n";
		else cout << "Alice\n";
		
	}
    
    return 0;
}
