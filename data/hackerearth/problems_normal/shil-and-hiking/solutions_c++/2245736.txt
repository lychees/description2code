#include<iostream>
using namespace std;
bool cmp(long long dx1, long long dy1, long long dx2, long long dy2)
{
	return dx1*dy2 <= dx2*dy1;//check!
}
int x[1007], y[1007];
int main()
{
	int n;
	cin >> n;
	for (int i = 1; i <= n; i++)
	{
		cin >> x[i] >> y[i];
	}
	int ans = 0;
	for (int i = 2; i <= n; i++)
	{
		ans++;
		int up = i - 1;
		for (int j = i - 2; j >= 1; j--)
		{
			if (cmp(x[i] - x[up], y[i] - y[up], x[i] - x[j], y[i] - y[j]))
			{
				ans++;
				up = j;
			}
		}
	}
	cout << ans << endl;
}
