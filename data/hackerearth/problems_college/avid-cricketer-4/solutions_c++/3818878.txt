#include <iostream>
using namespace std;
double ans[1001][1001];
double prob(int r,int b){
	if(r <= 0){
		return 1;
	}
	if(b == 0){
		return 0;
	}
	if(ans[r][b] != -1){
		return ans[r][b];
	}
	return (ans[r][b] = (2*prob(r,b-1) + prob(r-1,b-1) + prob(r-2,b-1) + prob(r-3,b-1) + prob(r-4,b-1) + prob(r-6,b-1))/7.0);
}
int main()
{
    int r,b,t,j,i;
    cin >> t;
    while(t--){
    	for(i=0;i<1001;i++){
    	for(j=0;j<1001;j++){
    		ans[i][j] = -1;
    	}
    }
    	cin>>r>>b;
    	printf("%.6lf\n",prob(r,b));
    }
    
    return 0;
}
