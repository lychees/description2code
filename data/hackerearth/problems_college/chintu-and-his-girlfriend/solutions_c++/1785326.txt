#include <bits/stdc++.h>
using namespace std;

int main()
{
    char str[101];

    int n,k;
    cin>>n>>k;

    int a[26]={0};
    int ans[26]={0};

    cin>>str;
    int len = strlen(str);

    for(int i=0;i<len;i++)
        ans[str[i]-'a']++;

    n--;

    while(n--){
        cin>>str;
        len = strlen(str);
		
		for(int i=0;i<26;i++){
			a[i]=0;
		}
		
        for(int i=0;i<len;i++)
            a[str[i]-'a']++;

        for(int i=0;i<26;i++){
            if(a[i]>0 && ans[i]>0){
                ans[i] = min(a[i],ans[i]);
            }
            else
                ans[i]=0;
        }

    }

    int count=0;

    for(int i=0;i<26;i++){
        if(ans[i]!=0)
            count+=ans[i];
    }

    if(count>=k)
        cout<<":-)"<<endl;
    else
        cout<<":-("<<endl;
    return 0;
}
