#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t,i;
    char ch[20];
    scanf("%d",&t);
    while(t--)
    {
        scanf("%s",ch);
        for(i=0;ch[i]!='\0';i++)
        {
            if(ch[i]=='2')
            printf("cde");
            else if(ch[i]=='6')
            printf("b3");
            else if(ch[i]=='7')
            printf("acf");
            else if(ch[i]=='9')
            printf("c6a");
        }
        printf("\n");
    }


    return 0;
}
