#include <iostream>
#include <algorithm>
using namespace std;
#include <string.h>
int main()
{
    int t,i,j,k;
    
    cin>>t;
    while(t--){
    	char a[10001],b[10001];
    	cin>>a>>b;
    	i=j=k=0;
    	sort(b,b+strlen(b));
    
    	while(i<strlen(a)){
    		if(j<strlen(b)){
    			if(a[i]>b[j]){
    				cout<<b[j];
    				j++;
    			}else{
    				cout<<a[i];
    			}
    		}else{
    			cout<<a[i];
    		}
    		
    	i++;	
    	}
    	cout<<"\n";
    }
    return 0;
}
