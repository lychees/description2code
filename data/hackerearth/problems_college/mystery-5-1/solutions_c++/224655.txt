using namespace std;
int a[30];
int main()
{
	int t,i,j,n,temp;
	int max;
	string s;
	cin>>t;
	while(t--)
	{
		cin>>s;
		max = -1;
		for(i=0;i<30;i++)
		a[i] = 0;
		for(i=0;i<s.length();i++)
		{
			a[s[i]-'a']++;
		}
		max = -1;
		for(i=0;i<30;i++)
		{
			if(a[i] > max)
			max = a[i];
		}
		cout<<max<<endl;
	}
	
	return 0;
}
