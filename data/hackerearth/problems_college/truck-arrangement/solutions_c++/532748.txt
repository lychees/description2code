#include<stdio.h>
#include<algorithm>
#include<vector>
#include<utility>
using namespace std;

int main()
{
	int t,n,i,x;
	
	scanf("%d",&t);
	
	while(t--)
	{
		scanf("%d",&n);
		vector< pair<int,int> > arr(n);
		int max=0;
		
		for(i=0;i<n;i++)
		{
			scanf("%d",&x);
			
			arr[i].first=x;
			arr[i].second=i;
		}
		
		std :: stable_sort(arr.begin(),arr.end());
		
		for(i=0;i<n;i++)
		{
			x=arr[i].second > i ? (arr[i].second - i) : (i - arr[i].second);
			
			if(x > max)
			max=x;		
		}
		
		printf("%d\n",max);
	}
	
	return 0;
}
