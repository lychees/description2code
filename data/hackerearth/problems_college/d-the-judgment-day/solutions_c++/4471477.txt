#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t,n,a[105];
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(int i=0;i<n;i++)
    	cin>>a[i];
    	//sort(a,a+n,greater<int>());
    	int c[4];
    	for(int i=0;i<4;i++)
    	{
    		c[i]=0;
    	}
    	//for(int i=0;i<n;i++)
    	//cout<<a[i]<<" ";
    	for(int i=0;i<n;i++)
    	{
    		c[a[i]-1]++;
    	}
    	//for(int i=0;i<4;i++)
    	//cout<<c[i]<<" ";
    	//cout<<endl;
    	int total=0;
    	total=c[3];
    	c[0]=c[0]-c[2];
    	total+=c[2];
    	if(c[1]%2!=0)
    	c[0]-=2;
    	if(c[1]==1)
    	total+=c[1];
    	else 
    	total+=(c[1]/2)+c[1]%2;
    	
    	if(c[0]>0 && c[0]%4==0)
    	total+=c[0]/4;
    	if(c[0]>0 && c[0]%4!=0)
    	total+=c[0]/4+1;
    	cout<<total<<endl;
    }
}
