#include <iostream>
#include<bits/stdc++.h>
using namespace std;
 
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string x,y;
		cin>>x;
		cin>>y;
		int ct=0;
		if(x.size()==y.size())
		{
			for(int i=0;i<x.size();i++)
			{
				if(x[i]==y[i])
				ct++;
			}
		}
		else
		{
			if(x.size()>y.size())
			{
				string temp=x;
				x=y;
				y=temp;
			}
			for(int i=0;i<x.size();i++)
			{
				if(x[i]==y[i])
				ct++;
			}
		}
		cout<<ct<<endl;
		
	}
   
    return 0;
}
