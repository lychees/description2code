#include <iostream>
#include <cmath>
#include <vector>
#include <string>
using namespace std;
int main()
{
	int n;
	cin>>n;
	vector<int> nums;
	for (int i = 0; i < n; ++i)
	{
		int temp;
		cin>>temp;
		nums.push_back(temp);
	}
	long long ans=0;
	long long max;
	int pos=-1;
	for(int i=1;i<n-1;++i)
	{

		max=-1;
		for(int j=1;j<nums.size()-1;++j)
		{
			if(nums[j+1]*nums[j-1]>max)
			{
				max=nums[j+1]*nums[j-1];
				pos=j;
			}
		}
		ans+=max;
		nums.erase(nums.begin()+pos);
	}
	cout<<ans<<endl;
}
