long long sum(long long r)
{
    long long a = (-1+sqrt(1+8*r))/2;
    long long su,k;
    if(a%2==0)
    {
        k = a-1;
        k = (k+1)/2;
        su = (k*(2+(k-1)*2))/2;
    }
    else
    {
        k = a;
        k = (k+1)/2;
        su = (k*(2+(k-1)*2))/2;
    }

    long long k1 = (a*(a+1))/2;
    if((r-k1)%2!=0)
    {
        su+=(a+1);
    }
    return su;
}
