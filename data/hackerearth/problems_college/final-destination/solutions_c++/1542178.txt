#include<bits/stdc++.h>
#define ll long long int
using namespace std;
int arr[1000001];
int visited[1000001];
int main(){
    int t;
    cin>>t;
    while(t--){
        memset(arr,0,sizeof arr);
        int x,y; cin>>x>>y;
        if(x==y) cout<<0<<endl;
        else if(y<2 || y>1000000) cout<<-1<<endl;
        else{
            int a[5]; int f=0;
            for(int i=0; i<4; i++) cin>>a[i];
            queue<pair<ll,int> > q;
            q.push(make_pair(x,0));
            memset(visited,0,sizeof visited);
            while(!q.empty()){
                int node=q.front().first;
                int time=q.front().second;
                if(node==y){
                	f=1;
                    cout<<time<<endl;
                    break;
                }
                q.pop();
                ll val1=node*a[0];
                ll val2=node/a[1];
                ll val3=node+a[2];
                ll val4=node-a[3];
                if(val1>1 && val1<1000001 && !visited[val1]){
                    visited[val1]=1;
                    q.push(make_pair(val1,time+1));
                }
                if(val2>1 && val2<1000001 && !visited[val2]){
                	visited[val2]=1;
                    q.push(make_pair(val2,time+1));
                }
                if(val3>1 && val3<1000001 && !visited[val3]){
                	visited[val3]=1;
                    q.push(make_pair(val3,time+1));
                }
                if(val4>1 && val4<1000001 && !visited[val4]){
                	visited[val4]=1;
                    q.push(make_pair(val4,time+1));
                }
            }
            if(!f) cout<<-1<<endl;
            
        }
    }
}
