#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,m;
	cin>>n>>m;
	int a[2*n];
	for(int i=0;i<2*n;i++)
	{
		cin>>a[i];
	}
	sort(a,a+2*n);
	int s=0;
	int k=0;
	for(int i=0;i<2*n;i++)
	{
		s+=a[i];
		if(s<=m)
		k++;
		else
		break;
	}
	cout<<min(n,k);
	return 0;
}
