#include <iostream>
using namespace std;

int main()
{
    static int cases,i,shalf,half2,x,y,fhalf,half1,row[15][1100],col[15][1100],j;
    char k;
    cin>>cases;
    while(i<cases)
    {
    	cin>>j;
    	shalf=0; half2=0;
    	for(x=0;x<j;x++)
    	{
    		k=getchar();
    		for(y=0;y<j;y++)
    		{
    			k=getchar();
    			if(k=='#')
    			{
    				row[i][y]++;
    				col[i][x]++;
    				if(y>0)shalf++;
    				if(x>0)half2++;
    			}
    			
    		}
    	}
    	fhalf=row[i][0];
    	half1=col[i][0];
    	for(x=1;x<j;x++)
    	{
    		if(fhalf==shalf||half1==half2)
    		 x=j+2;
    		else
    		{
    			fhalf=fhalf+row[i][x];
    			shalf=shalf-row[i][x];
    			half1=half1+col[i][x];
    			half2=half2-col[i][x];
    		}
    	}
    	if(x>j)
    	 cout << "YES" << endl;
    	else
    	 cout << "NO" << endl;
    	i++;
    }
    return 0;
}
