#include <iostream>
using namespace std;

int main()
{
    int t,x,y,i,count;
    int fib[30];
    fib[0]=1;
    fib[1]=1;
    for(i=2;i<29;i++)
    fib[i]=fib[i-1]+fib[i-2];
    cin>>t;
    while(t--)
    {
    	cin>>x>>y;
    	i=0;
    	while(fib[i]<x)
    	i++;
    	count=0;
    	while(fib[i]<=y)
    	{
    		i++;
    		count++;
    	}
    	if(count%2==0)
    	cout<<"VALID"<<"\n";
    	else
    	cout<<"INVALID"<<"\n";
    }
    return 0;
}
