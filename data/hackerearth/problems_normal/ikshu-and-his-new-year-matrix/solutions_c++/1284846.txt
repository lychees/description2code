#include <bits/stdc++.h>
#define MAX 10002
using namespace std;
vector <bool> prime(MAX,true);
int main()
{
	int n,k,temp,answer;
//	bool prime[MAX]={false};
	prime[1]=prime[0]=false;
	for(int i=2;i<sqrt(MAX);i++)
	{
		if(prime[i]==true)
		{
			for(int j=i*i;j<MAX;j=j+i)
			{
				prime[j]=false;
			}
		}
	}
	cin>>n;
	cin>>k;
	int arr[n][n];
    for(int i=0;i<n;i++)
    {
    	for(int j=0;j<n;j++)
    	{
    		cin>>temp;
    		answer=0;
    		while(prime[temp]!=true&&temp<=10000)
    		{
    			temp+=k;
   				answer++;
  			}
    		if(temp<=10000)
    			arr[i][j]=answer;
    		else
    			arr[i][j]=10000000;
    	}
    }
    int xindex,yindex,sum,min,i,j;
    min=1000000;
    for(int i=1;i<n-1;i++)
    {
    	for(int j=1;j<n-1;j++)
    	{
    		sum=arr[i][j]+arr[i-1][j-1]+arr[i+1][j+1]+arr[i-1][j+1]+arr[i+1][j-1];
    		if(sum<min)
	    	{
	    		min=sum;
	   			xindex=i+1;
	   			yindex=j+1;  			
	   		}
    	}
    }
    if(min!=1000000)
    cout<<"yes"<<endl<<min<<endl<<xindex<<" "<<yindex;
    else
    cout<<"no";
    return 0;
}
