#include <iostream>
#include <string>
using namespace std;

int main()
{
	short t, size_a, size_b;
	short c = 0,carry = 0; 
	string a,b;
	cin >> t;
	for (int i = 0; i < t; i++)
	{ 
		cin >> a >> b;
		size_a = a.size();
		size_b = b.size();
		c = 0;
		while (size_a && size_b)
		{
    		int sum = (a[--size_a] - '0') + (b[--size_b] - '0') + carry;;
    		if (sum > 9)
    		{
    			carry = 1;
    			c++;
    		}	
    		else
    			carry = 0;
    	}
    	if (c == 1)
    		cout <<  "1 carry operation" << "\n";
		else if (c > 1)
			cout << c << " carry operations" << "\n";
		else if (c == 0)
		 	cout << "No carry operation" << "\n";
	}
       return 0;
}
