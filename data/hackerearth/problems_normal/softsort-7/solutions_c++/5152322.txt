int main()                   // the solution is (5 + 3 * n! - 2) ;
{
   int tc;
   long long ar[1000001];
   
   ar[0]=1;
   
    // storing factorail of each no befoe hand so that foe each test case we dont need to calculate factorial and thus time complexity is reduced  (memoization)
   for(int i=1;i<1000001;i++)
     ar[i]=(ar[i-1]*i)%1000000007;   
     
   long long result=0;
   long long n;
    cin>>tc;
   while(tc--){
   	
    cin>>n;
   	result=(3*ar[n])%1000000007 +3;
   cout<<result<<endl;
   	result=0;
   	
   }
    return 0;
}
