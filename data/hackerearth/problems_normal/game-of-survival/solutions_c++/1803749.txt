#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

bool Fun(int i, int j) { return i > j; }

int main()
{
    vector<int> A;
    vector<int> S;
    vector<int>::iterator it;
    long int T,N,i,x;

    cin>>T;

    while(T--)
    {
        cin>>N;
        for(i=0;i<N;i++)
        {
            cin>>x;
            S.push_back(x);
        }
        for(i=0;i<N;i++)
        {
            cin>>x;
            A.push_back(x);
        }
        sort(S.begin(), S.end(), Fun);
        sort(A.begin(), A.end(), Fun);

        for(i=0;i<N;i++)
        {
            if(S.at(i)<A.at(i))
                break;

        }
        if(i==N)
        {
            cout<<"YES"<<"\n";
        }
        else
        {
            cout<<"NO"<<"\n";
        }
        S.clear();
        A.clear();
    }
    return 0;
}
