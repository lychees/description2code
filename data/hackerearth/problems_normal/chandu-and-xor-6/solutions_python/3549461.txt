n,m=map(int,raw_input().split())
mt=[]
r,c=[n,m]
while r>0:
	mt.append(map(int,raw_input().split()))
	r=r-1
	
rsum=[]
for i in xrange(0,n):
	sm=0
	for j in xrange(0,m):
		sm=sm+mt[i][j]
	rsum.append(sm)
csum=[]
for i in xrange(0,m):
	sm=0
	for j in xrange(0,n):
		sm=sm+mt[j][i]
	csum.append(sm)
#print rsum,csum

mx=0
for i in rsum:
	for j in csum:
		if mx<(i^j):
			mx=i^j
