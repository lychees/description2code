from decimal import *
kases = int(raw_input())
speed = raw_input()
fuel = raw_input()
fuel_avail = int(raw_input())
speed = speed.split()
fuel = fuel.split()
list1 = list()
for kase in range(kases):
    var = fuel_avail/float(fuel[kase])*float(speed[kase])
    list1.append(var)

with localcontext() as ctx:
    ctx.rounding = ROUND_DOWN
    print Decimal(max(list1)).quantize(Decimal('0.001'))
