#include<iostream>
#define mod 1000000007
using namespace std;
typedef long long ll;
int x[]={-1,1,0,0};
int y[]={0,0,1,-1};
long long dp[1000][1000];
int vis[1000][1000];
int a[1000][1000];
int n;
ll check(ll h)
{
    if(h<0)
    {
        return h+mod;
    }
    if(h>mod)
    {
        return h%mod;
    }
    else
        return h;
}
long long call(int h,int j)
{
    dp[h][j]++;
    for(int i=0;i<4;i++)
    {
        if(h+x[i]>=0&&h+x[i]<n&&j+y[i]>=0&&j+y[i]<n)
        {
        if(a[h][j]>a[h+x[i]][j+y[i]])
        {
            if(vis[h+x[i]][j+y[i]]==0)
            {
                dp[h][j]=check(dp[h][j]+call(h+x[i],j+y[i]));
            }
            else
            {
                dp[h][j]=check(dp[h][j]+dp[h+x[i]][j+y[i]]);
            }
        }
        }
    }
    vis[h][j]=1;
    return dp[h][j];
}
int main()
{
    cin>>n;
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<n;j++)
        {
            cin>>a[i][j];
            vis[i][j]=0;
            dp[i][j]=0;
        }
    }
    long long ans=0;
    for(int i=0;i<n;i++)
    {
        for(int k=0;k<n;k++)
        {
            if(vis[i][k]==0)
            {
                call(i,k);
            }
            ans=check(ans+dp[i][k]);
        }
    }
    cout<<ans<<endl;
}
