#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,q;
    cin>>n>>q;
    int arr[n];
    set<int>ss;
    for(int i=0;i<n;i++){
    	cin>>arr[i];
    	ss.insert(arr[i]);
    }
    
    while(q--){
    	int x;
    	cin>>x;
    	if(ss.find(x)!=ss.end())
    		cout<<"YES"<<endl;
    	else
    		cout<<"NO"<<endl;
    }
    return 0;
}
