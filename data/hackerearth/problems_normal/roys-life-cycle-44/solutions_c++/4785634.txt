#include <iostream>
#define DAYEND 1440
using namespace std;
int main()
{
	char a,last='C';
	long int n,dayC=0,monthC=0,charCount=0,lastdayC=0,lastmonthC=0;
	cin>>n;
	while(n)
	{
		cin>>a;
		//cout<<"a="<<a<<" last="<<last<<endl;
		if(a=='C')
		{
			//cout<<"a equals c"<<endl;
			if(last=='C')
			{	//cout<<"incremented"<<endl;
				++dayC;
				++monthC;
			}
			else
			{
				if(dayC>lastdayC)lastdayC=dayC;
				if(monthC>lastmonthC)lastmonthC=monthC;
				dayC=monthC=1;
			}
		}
		else if(last=='C') //if sequence breaks
		{
			if(dayC>lastdayC)lastdayC=dayC;
			if(monthC>lastmonthC)lastmonthC=monthC;
		}

		if(charCount==DAYEND-1)
		{
			if(dayC>lastdayC)lastdayC=dayC;
			if(monthC>lastmonthC)lastmonthC=monthC;
			dayC=0;
			--n;
			charCount=-1;
			//cout<<"\nDay complete\n";
		}
		++charCount;
		//cout<<"charCount="<<charCount;
		last=a;
	}
	cout<<lastdayC<<" "<<lastmonthC<<endl;
}
