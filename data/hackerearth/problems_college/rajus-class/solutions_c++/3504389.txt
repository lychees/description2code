#include<bits/stdc++.h>
using namespace std;
typedef long long int lld;
int main()
{
    lld t,n,i,j,key,roll;
    string str,str1,str2;
    cin>>t;
    while(t--)
    {
    	cin>>n;
		vector< pair<int ,string> > A(n);
		for(i=0;i<n;i++)
		{
			cin>>A[i].first>>A[i].second;
		}
		for(j=1;j<n;j++)
		{
			key=A[j].first;
			str=A[j].second;
			i=j-1;
			while(i>=0&&A[i].first>key)
			{
				A[i+1].first=A[i].first;
				A[i+1].second=A[i].second;
				i--;
			}
			A[i+1].first=key;
			A[i+1].second=str;
		}
		for(i=0;i<n;i++)
		{
			if(A[i].first==A[i+1].first)
			{
				str1=A[i].second;
				str2=A[i+1].second;
				if(str1>str2)
				{
					A[i].second=str2;
					A[i+1].second=str1;
				}
			}
		}
//			cout<<endl;
//		for(i=0;i<n;i++)
//		{
//			cout<<A[i].first<<"  "<<A[i].second<<endl;;
//		}
		cin>>roll;
		cout<<A[roll-1].second<<endl;
		A.clear();
//		cout<<A[0].first;
	}
    return 0;
}
