#include <bits/stdc++.h>

using namespace std;

//int count1[10000] = {0};
long long int dp[10000] = {-1};
unsigned long long int count_0 = 0;
unsigned long int count_1 = 0;

long long int fun(int num)
{
    dp[0] = 0;
    dp[1] = 1;

    if (num == 0)
    {
        count_0++;
        return 0;
    }
    if (num == 1)
    {
        count_1++;
        return 1;
    }

    else
    {
        return fun(num-1) + fun(num-2);
    }
}

int main()
{
    int n;
    cin >> n;
    int num;

    while(n--)
    {
        cin >> num;
        long long int k = fun(num);

        cout << count_0 << " " << count_1 << endl;
        count_0 = 0;
        count_1 = 0;
    }

    return 0;
}
