int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    
    unordered_map<string,int>mp;
    
    int t;
    cin>>t;
    
    int x;
    
    rep(x,1,t,1)
    {
    	string s;
    	cin>>s;
    	
    	mp[s]+=1;
    }
    
    int max=0;
    
    for(auto& m: mp)
    {
    	if(m.S>max) max=m.S;
    }
    
    int ans=0;
    
    for(auto& m: mp)
    {
    	if(m.S == max) ans=ans+max;
    }
    
    cout<<ans<<"\n";
    
    return 0;
    
}
