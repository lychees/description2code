'''
# Read input from stdin and provide input before running code

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
for x in xrange(int(raw_input())):
	string = raw_input()
	vCnt = 0
	cCnt = 0
	conti = False
	contiCnt = 0
	for c in string:
		if c == 'a' or c == 'e' or c == 'i' or c == 'o' or c == 'u' or c == 'y':
			vCnt = vCnt + 1
			conti = False
			contiCnt = 0
		else:
			conti = True
			cCnt = cCnt + 1
			if conti:
				contiCnt = contiCnt + 1
			if contiCnt == 3:
				break
	if cCnt > vCnt or contiCnt == 3:
		print "hard"
	else:
		print "easy"
