#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
vector<ll>fibo(80,0);
vector<ll>diff(80,0);
void preCal()
{
    fibo[0]=3;
    fibo[1]=5;
    for(ll i =2;i<fibo.size();i++)
        fibo[i] = fibo[i-1]+fibo[i-2];
    diff[0]=0;
    for(ll i =1;i<diff.size();i++)
        diff[i] = fibo[i]-fibo[i-1]-1;
    //for(int i  =0;i<10;i++)
      //  cout<<fibo[i]<<" ";
        //cout<<endl<<endl;
    partial_sum(diff.begin(),diff.end(),diff.begin());
    //for(int i  =0;i<10;i++)
      //  cout<<diff[i]<<" ";
}
int main()
{
    ios_base::sync_with_stdio(false);
    preCal();
    int cases;
    cin>>cases;
    while(cases--)
    {
        ll num;
        cin>>num;
        ll idx = lower_bound(diff.begin(),diff.end(),num)-diff.begin();
        --idx;
        ll ans = fibo[idx]+(num-diff[idx]);
        cout<<ans<<endl;
    }


    return 0;
}
