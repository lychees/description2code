#include<iostream>
#include<cmath>
using namespace std;
int weight(long long int n)
{
	int temp=0;
	while(n>0)
	{
		temp+=(n%2);
		n/=2;
	}
	return temp;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int j,i,n,x[100001],temp,t1,min1=33;
		cin>>n;
		for( i=0;i<n;i++)
		{
			cin>>x[i];
			temp=weight(x[i]);
			if(temp<min1)
				min1=temp;
		}
		cout<<min1<<endl;
	}
	return 0;
}
