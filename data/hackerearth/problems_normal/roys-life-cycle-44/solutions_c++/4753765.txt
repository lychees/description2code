int count_C(string s){
	string::iterator it=s.begin();
	
	int count=0;
	int max=0;;
while(it!=s.end()){
	
	if(*it=='C') {
	count++;	
		
	
//	continue;
	}
	else if(count>=max){
		
	max=count;
//	cout<<"max"<<max<<" "<<"count"<<count;
	count=0;	
	}
	else{
		
		count=0;
	}
	
it++;	
}
if(count>max)
max=count;
return max;	
}
