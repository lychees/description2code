#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#include <string>
using namespace std;
int main()
{
    int t,n,i,j;
    cin>>t;
    while(t--)
    {
        cin>>n;
        int c=0;
        int a[n],b[n];
        for(i=0;i<n;i++)
            cin>>a[i];
        for(i=0;i<n;i++)
            cin>>b[i];
        for(i=0;i<n-1;i++)
        {
            for(j=0;j<n-i-1;j++)
            {
                if(a[j]>a[j+1])
                    swap(a[j],a[j+1]);
                if(b[j]<b[j+1])
                    swap(b[j],b[j+1]);
            }
        }
        for(i=0;i<n;i++)
        {
            if(a[i]>b[i])
            {
                if(a[i]%b[i]==0)
                    c++;
            }
            else
            {
                if(b[i]%a[i]==0)
                    c++;
            }
        }
        cout<<c<<endl;
    }
    return 0;
}
