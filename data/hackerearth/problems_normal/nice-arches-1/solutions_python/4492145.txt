count=0
M=int(raw_input())
for i in range(M):
	s=raw_input()
	stack=[]
	for j in s:
		if(len(stack)==0):
			stack.append(j)
		elif(stack[len(stack)-1]==j):
			stack.pop()
		else:
			stack.append(j)
	if(len(stack)==0):
		count+=1
print('%d'%count)
