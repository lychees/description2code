#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long int a,b,ans,i,flag,answer;
    char s[100001];
    scanf("%lld%lld",&a,&b);
    scanf("%s",s);
    ans=a;
    flag=0;
    if(s[0]=='1'){
    	answer=a;
    	flag=1;
    }
    else{
    	answer=0;
    }
    for(i=1;i<strlen(s);i++){
    	ans=(ans*ans)%b;
    	if(s[i]=='1'){
    		flag=1;
    		answer=(answer+ans)%b;
    	}
    }
    if(flag==0)
    	printf("0");
    else	
    	printf("%lld",(answer)%b);
    return 0;
}
