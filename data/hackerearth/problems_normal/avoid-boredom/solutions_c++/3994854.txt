#include <bits/stdc++.h>
using namespace std;
typedef long long LL;

LL dp[111][111];
int arr[111];

int main()
{
	int T, N, M;
	
	scanf("%d", &T);
	while (T--)
	{
		scanf("%d%d", &N, &M);
		for (int i = 0; i < N; i++)
			scanf("%d", arr + i);
		for (int i = 0; i <= N; i++)
			for (int j = 0; j <= M; j++)
				dp[i][j] = INT_MIN;
		dp[0][0] = 0;
		for (int i = 0; i < N; i++)
		{
			dp[i + 1][1] = 0;
			for (int j = 1; j < M; j++)
			{
				for (int k = 0; k < i; k++)
				{
					dp[i + 1][j + 1] = max(dp[i + 1][j + 1], dp[k + 1][j] + abs(arr[i] - arr[k]));
				}
			}
		}
		LL ans = 0;
		for (int i = 0; i < N; i++)
			ans = max(ans, dp[i + 1][M]);
		printf("%lld\n", ans);
	}
	
    return 0;
}
