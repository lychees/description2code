#include<iostream>
#include<algorithm>
using namespace std;
int arr[10001];
int main()
{
	int testCase, n, max, min;
	cin >> testCase;
	while (testCase--)
	{
		cin >> n;
		max = 0; min = 300000;
		for (int i = 0; i < n; i++)
		{
			cin >> arr[i];
		}
		sort(arr, arr + n);
		int i = 0, j = n - 1;
		while (i < j)
		{
			if (arr[i] + arr[j]>max)
				max = arr[i] + arr[j];
			if (arr[i] + arr[j] < min)
				min = arr[i] + arr[j];
			i++;
			j--;
		}
		cout << max - min << endl;
	}

	return 0;
}
