case = int(input())
while case >= 1:
    n, q = map(int, list(raw_input().split()))
    heights = map(int, list(raw_input().split()))
    if n == 1:
        print '0'
        case = case-1
        continue
    heights.sort()
    diff = [abs(heights[i]-heights[i+1]) for i in range(n-1)]
    print sum(diff)*q
    case = case-1
