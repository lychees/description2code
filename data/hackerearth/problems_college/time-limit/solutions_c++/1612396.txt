#include<cstdio>
#include<set>
using namespace std;
int main()
{
	int n,m;
	scanf("%d %d",&n,&m);
	set <int> n1,m1;
	for(int i=0;i<n;i++)
	{
		int t;
		scanf("%d",&t);
		n1.insert(t);
	}
	for(int i=0;i<m;i++)
	{
		int t;
		scanf("%d",&t);
		m1.insert(t);
	}
	set <int> :: iterator ip;
	ip=m1.begin();
	int l2=(*ip);
	int l1;
	for(ip=n1.begin();ip!=n1.end();ip++)
		l1=(*ip);
	ip=n1.begin();
	int l3=(*ip);
	if(l1>l2)
		printf("-1");
	else if(2*l3>l2-1)
		printf("-1");
	else
	{
		if(2*l3>l1)
			printf("%d\n",2*l3);
		else
			printf("%d\n",l1);
	}
	return 0;
}
