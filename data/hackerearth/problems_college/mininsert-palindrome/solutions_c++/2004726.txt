#include <iostream>
using namespace std;
int mini(int a, int b)
{  
	return a < b ? a : b; 
}
int checkpal(char s[],int l,int h)
{
	if(l==h)
	return 0;
	if(l==h-1)
	return (s[l]==s[h])?0:1;
	return (s[l]==s[h])?checkpal(s,l+1,h-1):(mini(checkpal(s,l+1,h),checkpal(s,l,h-1)))+1;
}
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	char s[n];
    	cin>>s;
    	int val=checkpal(s,0,n-1);
    	cout<<val<<endl;
    }
    return 0;
}
