n = input()
count = 0
arr = []
while (count < n):
    arr.append(raw_input())
    count += 1
count = 0
while (count < n):
    string = arr[count][::-1]
    length = len(string)
    counter = count
    while (counter < n):
        if (arr[counter] == string):
            print "%d %c" % (length, string[length / 2])
        counter += 1
    count += 1
