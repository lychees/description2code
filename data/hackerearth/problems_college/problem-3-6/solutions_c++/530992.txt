#include <iostream>
using namespace std;

int main()
{
    long n;
    cin>>n;
    int x[n],y[n];
    for(int i=0;i<n;i++)
    cin>>x[i]>>y[i];
    int ansx[n],ansy[n];
    for(int i=0;i<n;i++)
    ansx[i]=ansy[i]=0;
    for(int i=0;i<n-1;i++)
    {
   		for(int j=i+1;j<n;j++)
   		{
   			if(x[i]==y[j])
   			{
   				ansx[i]++;
   				ansx[j]++;
   			}
   			else
   			{
   				ansx[i]++;
   				ansy[j]++;
   			}
   			if(y[i]==x[j])
   			{
   				ansx[i]++;
   				ansx[j]++;
   			}
   			else
   			{
   				ansy[i]++;
   				ansx[j]++;
   			}
   		
   		}
    }
    for(int i=0;i<n;i++)
   	cout<<ansx[i]<<" "<<ansy[i]<<"\n";
    return 0;
}
