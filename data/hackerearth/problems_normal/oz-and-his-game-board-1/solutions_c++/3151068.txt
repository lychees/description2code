#include<iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int n,m,a,b,c;
		cin>>n>>m>>a>>b>>c;
		
		long long int ans1;
		ans1=(n*a) + (m*b);
		long long int ans2;
		
		if(n>m)
		{
			ans2=(2*m*c)+((n-m)*a);
		}
		else
		{
			ans2=(2*n*c)+((m-n)*b);
		}
				
		if(ans1>ans2)
		{
			cout<<ans1<<"\n";			
		}
		else
		{
			cout<<ans2<<"\n";
		}
	}
}
