#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int arr[10];
        for(int i=0;i<10;i++)
            cin>>arr[i];
        bool flag=true;
        while(flag)
        {
            flag=false;
            for(int i=0;i<10;i++)
            {
                for(int j=0;j<10;j++)
                {
                    if(arr[(i+j)%10]>arr[i]+arr[j])
                    {
                        arr[(i+j)%10]=arr[i]+arr[j];
                        flag=true;
                    }
                }
            }
        }
        int n;
        cin>>n;
        string s;
        cin>>s;
        int ans=0;
        for(int i=0;i<n;i++)
        {
            ans+=arr[s[i]-48];
        }
        cout<<ans<<endl;
    }
}
