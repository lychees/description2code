    int readline() {
            int c = getchar();
            while (c < 33)
                    c = getchar();

            while (c > 32) {
                    k = k*10 + (int)c-48;

    void print(int x) {
         static char c[30];
         int len = 0;
         if (x == 0) {
                 c[0] = '0';
                 len = 1;
         }
         while (x > 0) {
                 int y = x / 10;
                 c[len++] = (x - y * 10) + '0';
                 x = y;
         }
         while (len > 0) {
                 --len;
                 putchar(c[len]);
         }
    }
