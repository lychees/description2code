#include <iostream>
using namespace std;

int main()
{
	long t,n,i,j,a[1000001][2],max,count;
	scanf("%ld",&t);
	while(t--){
		long h[1000001]={};
		scanf("%ld",&n);
		for(i=0;i<n;i++){
			scanf("%ld %ld",&a[i][0],&a[i][1]);
		}
		max=count=0;
		for(i=0;i<n;i++){
			for(j=i;j<n;j++){
				if(a[j][1]==1){
					if(h[a[j][0]]==0){
					h[a[j][0]]=1;
					count++;}
					if(max<count)max=count;
				}
				else
				count=0;
			}
		}
		printf("%ld\n",max);
	}
	return 0;
}
