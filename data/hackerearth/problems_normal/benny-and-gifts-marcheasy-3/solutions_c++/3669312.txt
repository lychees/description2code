void solve(string s){
    int ans=0;
    set<pii>x;
    x.insert(mp(0,0));
    int curx=0;
    int cury=0;
    for(char c:s){
        if(c=='R')cury+=1;
        if(c=='L')cury-=1;
        if(c=='U')curx+=1;
        if(c=='D')curx-=1;
        if(x.find(mp(curx,cury))==x.end())x.insert(mp(curx,cury));
        else ans++;
    }
    cout<<ans;
}
