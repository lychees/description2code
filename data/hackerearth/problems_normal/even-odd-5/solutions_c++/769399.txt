#include <iostream>
using namespace std;

int main()
{
    long int test,evenlen,oddlen;
    long long int reseven,resodd;
    cin>>test;
    while(test--){
    	
    	long int row,column;
    	cin>>row>>column;
    	
    	if(column>=row){
    		evenlen=row/2;
    		oddlen=row-evenlen;
    		
    		reseven=(evenlen*(4+((evenlen-1)*2)))/2;
    		resodd=(oddlen*(2+((oddlen-1)*2)))/2;
    		
    		cout<<reseven<<" "<<resodd<<endl;
    	}else{
    		evenlen=column/2;
    		oddlen=column-evenlen;
    		
    		reseven=(evenlen*(4+((evenlen-1)*2)))/2;
    		resodd=(oddlen*(2+((oddlen-1)*2)))/2;
    		
    		long int remodd,remeven,temp;
    		temp=row-column;
    		remeven=(row/2)-evenlen;
    		remodd=((row/2)+(row%2))-oddlen;
    		
    		reseven+=(remeven*column);
    		resodd+=(remodd*column);
    		cout<<reseven<<" "<<resodd<<endl;
    	}
    }
}
