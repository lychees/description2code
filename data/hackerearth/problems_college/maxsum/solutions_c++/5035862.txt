long long int maxsum(int arr[],int n)
{
	long long int dp[n];
	dp[0]=arr[0];
	long long int ans=arr[0];
	for(int i=1;i<n;i++)
	{
		dp[i]=arr[i];
		for(int j=i-2;j>=0;j--)
		{
			dp[i]=max(dp[i],dp[j]+arr[i]);
		}
		ans=max(ans,dp[i]);
	}
	return ans;
}
