#include<iostream>
using namespace std;
int main()
{
	long long int t,x,y,n,i,j,a[50],temp;
	cin>>t;
	while(t--)
	{
		cin>>x>>y>>n;
		for(i=0;i<x;i++)
		{
			a[i]=y;
		}
		for(i=x;i<n;i++)
		{
			temp=0;
			for(j=0;j<x;j++)
			{
				temp=temp+a[i-j-1];
			}
			a[i]=temp;
		}
		cout<<a[n-1]<<endl;
	}
}
