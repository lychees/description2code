#include <iostream>
using namespace std;
int main()
{
  int a,b,c,i,j,arr[10000];
  cin>>a;
  while(a--)
  {c=0;
  	cin>>b;
  	for(i=0;i<b;i++)
  	{cin>>arr[i];}
  	for(i=0;i<b;i++)
  	{
  		for(j=0;j<b;j++)
  		{if(j>=i && arr[i]==arr[j])
  		c++;	}}
  cout<<c<<"\n";}
    return 0;
}
