#include <iostream>
#include<unordered_map>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t>0)
	{
		string s;
		cin>>s;
		unordered_map<char,int> mymap;
		mymap[s.at(0)]=1;
		int len=s.size();
		int cnt=0;
		for(int i=1;i<len;i++)
		{
			if(s.at(i)!=s.at(i-1))
				cnt++;
			if(mymap.find(s.at(i))==mymap.end())
			{
				mymap[s.at(i)]=1;
			}
			else
				mymap[s.at(i)]+=1;
		}
		
			if(cnt==2)
			{
				unordered_map<char,int>::iterator it;
				it=mymap.begin();
				int number=it->second;
				for(it=mymap.begin();it!=mymap.end();it++)
				{
					if(number!=it->second)
						number=0;
				}
				if(number==0)
					cout<<"Not OK\n";
				else
					cout<<"OK\n";
			}
			else
			{
				cout<<"Not OK\n";	
			}
			
		t--;
	}
}
