#include<stdio.h>
long long  a[1000006][2]={{0}};
long long table[1000006]={0};
int main()
 {long long r,n,i,v,m,q,max=-1;
 scanf("%lld",&n);
  for(i=0;i<n;i++)
    {
    scanf("%lld%lld",&a[i][0],&a[i][1]);
    if(table[a[i][0]]==0)
       table[a[i][0]]=a[i][1];
    else
      if(table[a[i][0]]<a[i][1])
         table[a[i][0]]=a[i][1];
     if(a[i][0]>max)
        max=a[i][0];
   }
   m=0;
  for(i=max;i>=1;i--)
    {if(table[i]<m)
       table[i]=m;
     else
       m=table[i];
    }
  scanf("%lld",&q);
   r=0;
  while(q--)
    {scanf("%lld",&m);
    r+=table[m]/m;
    }

 }
