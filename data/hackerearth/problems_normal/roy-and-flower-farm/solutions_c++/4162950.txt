#include <iostream>
#include <cstdio>
#include <vector>
#include <cstring>

#define MAX(xx,yy) xx>yy?xx:yy
#define MIN(xx,yy) xx<yy?xx:yy
#define pb push_back
#define mp make_pair
#define endl '\n'
using namespace std;
typedef long long ll;
typedef long li;
typedef vector<ll,ll> vll;
const ll maxn = 1000000;
ll x[109];
ll y[109];
ll dp[101][10004];

int main()
{
	std::ios::sync_with_stdio(0);
	ll i,j,t,k,n,c,d,p,optp,optc;
	//freopen("input.in","r",stdin);
	cin>>t;
	while(t--)
	{
		cin>>n>>p;
		for(i=0;i<n;i++)
		{
			cin>>x[i]>>y[i];
		}
		dp[0][0]=0;optp=0;optc=0;
		for(j=0;j<=p;j++)
		{
			if(j<y[0])dp[0][j]=0;
			else dp[0][j]=MAX(0,x[0]-y[0]);
			if(dp[0][j]>optp || (dp[0][j]==optp && j<optc))
			{
				optc=j;
				optp=dp[0][j];
			}

		}
		for(i=1;i<n;i++)
		{
			for(j=0;j<=p;j++)
			{
				if(j==0)
					dp[i][j]=0;
				else
				{
					if(y[i]>j)
						dp[i][j] = dp[i-1][j];
					else
						dp[i][j] = MAX(x[i]-y[i]+dp[i-1][j-y[i]],dp[i-1][j]);

				}
				if(dp[i][j]>optp || (dp[i][j]==optp && j<optc))
				{
					optc=j;
					optp=dp[i][j];
				}
			}
		}
		cout<<optc<<" "<<optp+p<<endl;
	}

}
