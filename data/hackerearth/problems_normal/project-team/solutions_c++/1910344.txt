#include <iostream>
using namespace std;
int main()
{
    int siz[100],diff[100],mem[100],team[50];
    int i,j,t,k,l,temp,pos,small,big;
    cin>>t;
    for(i=0 ; i<t ; i++)
    {
    	cin>>siz[i];
    	for(j=0 ; j<siz[i] ; j++)
    		cin>>mem[j];
		for(k=0 ; k<siz[i] ; k++)
    	{
    		pos=k;
    		small=mem[k];
    		for(l=k ; l<siz[i] ; l++)
    			if (small>mem[l])
				{
					small=mem[l];
    	   			pos=l;
				}
    		temp=mem[k];
    		mem[k]=mem[pos];
    		mem[pos]=temp;
    	}
    	for(k=0 ; k<(siz[i]/2) ; k++)
    	{
    		team[k]=mem[k]+mem[(siz[i])-k-1];
    	}
    	small=team[0];
    	big=small;
    	for(l=0;l<(siz[i]/2);l++)
		{
			if(big<team[l])
				big=team[l];
			if(small>team[l])
				small=team[l];
		}
		diff[i]=big-small;

    }
    for(i=0;i<t;i++)
    	cout<<diff[i]<<"\n";
    return 0;
}
