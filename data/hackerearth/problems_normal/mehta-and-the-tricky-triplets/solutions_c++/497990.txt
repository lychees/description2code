llu com(llu x)
{
    return (x*(x-1)*(x-2)/6);
}
void fun(llu x)
{
    int i,a2=0,a3=0,a5=0,a7=0;
    while(x!=0)
    {
        i=x%10;
        x/=10;
        switch(i)
        {
        case 2:
            a2=1;break;

            a3=1;break;
        case 5:
            a5=1;break;
        case 7:
            a7=1;break;
        }
    }

    a+=a2;

    d+=a7;
