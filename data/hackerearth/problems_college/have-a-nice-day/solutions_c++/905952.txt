#include <iostream>
#include <stdio.h>
#include<algorithm>
using namespace std;
int main()
{
    int y,t,n,a[1000],b[1000],i,j;
    scanf("%d",&t);
    for(y=0;y<t;y++){
    	scanf("%d",&n);
    	for(i=0;i<n;i++)
    	scanf("%d",&a[i]);
    	for(i=0;i<n;i++)
    	scanf("%d",&b[i]);
    	sort(a,a+n);
    	sort(b,b+n);
    	j=0;
    	for(i=0;i<n;i++)
    	j=j+(a[i]*b[i]);
    	printf("%d\n",j);
    }
    return 0;
}
