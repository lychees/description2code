#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    map<string,int>m;
    while(n--)
    {
    	string s;
    	cin>>s;
    	if(m.find(s)==m.end())
    		m[s]=1;
    	else
    		m[s]++;
    	
    }
    map<string,int>::iterator it;
    for(it=m.begin();it!=m.end();++it)
    	cout<<it->first<<" "<<it->second<<endl;
    return 0;
}
