#include <iostream>
#include <limits.h>
using namespace std;

int main()
{
	int t,n,i,j,k,no,minCost,minIndex,len;
	int cost[26][26];
	string s[10001];
	char res[27];
	cin>>t;
	while(t--)
	{

		cin>>n>>len;
		for(i=0;i<n;i++)
			cin>>s[i];
		for(i=0;i<n;i++)
		{
			for(j=0;j<len;j++)
			{
				for(k=0;k<26;k++)
				{
					cost[j][k] += abs(s[i][j] - 97 - k);
				}
			}
		}
		
		for(i=0;i<26 && i<len;i++)
		{
			minIndex = 0;
			minCost= cost[i][0];
			for(j=0;j<26;j++)
			{
				if(cost[i][j] < minCost)
				{
					minCost = cost[i][j];
					minIndex = j;
				}
				cost[i][j] = 0;
			}
			res[i] = (char)(97 + minIndex);
		}
		res[i] = '\0';
		cout<<res<<endl;
	}
    return 0;
}
