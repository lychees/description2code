#include <bits/stdc++.h>
using namespace std;
int rev( int x)
{
	int a=0;
	int flag=0;
	while(x)
	{
		if(x%10 !=0 || flag==1)
		{
			flag=1;
			a=a*10+(x%10);
		}
		x=x/10;
	}
	return a;
}
