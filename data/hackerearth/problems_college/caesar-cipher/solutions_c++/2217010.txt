#include<stdio.h>
#include<iostream>
using namespace std;
int main()
{
	long long t;
	cin>>t;
	while(t--)
	{
		string s;
		int n,up;
		cin>>n;
		cin>>s;
		for(int i=0;i<s.length();i++)
		{
			if(s[i]>='A' and s[i]<='Z')
			{
				up=s[i]-'A';
			}
			else
				up=s[i]-'a';
			
			if((up+n)>=26)
			up=(up+n)%26;
			else
			up+=n;
			
			//cout<<up<<endl;
			if(s[i]>='A' and s[i]<='Z')
			printf("%c",'A'+up);
			else
			printf("%c",'a'+up);
		}
		cout<<endl;
	}
}
