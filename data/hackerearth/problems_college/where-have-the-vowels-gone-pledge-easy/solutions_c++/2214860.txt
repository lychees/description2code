#include <iostream>
#include <set>
#include <string>
using namespace std;

int main()
{
    set<char> st;
    int n;
    cin>>n;
    string s;
    cin>>s;
    for(int i=0;i<n;++i)
    {
    	if(s[i]=='a'||s[i]=='e'||s[i]=='o'||s[i]=='u'||s[i]=='i')
    	st.insert(s[i]);
    }
    if(st.size()==5)
    cout<<"YES";
    else cout<<"NO";
    return 0;
}
