    #include<stdio.h>
    #include<string.h>
    int main()
    {
    int T,i,j,pos,pos2,pos3,min,ans,N,flag;
    char s[55],ssort[55],sukw[55],suky[55],minc,t;
    scanf("%d",&T);
    while(T--)
    {
    pos=-1;
    scanf("%d",&N);
    scanf("%s",&s);
    strcpy(ssort,s);strcpy(suky,s);strcpy(sukw,s);
    pos3=N-1;
    for(i=0;i<N;i++)
    {
    min=i;
    for(j=i+1;j<N;j++)
    {
    if(ssort[j]<ssort[min])
    {
    min=j;
    }
    }
    if(min!=i)
    {
    t=ssort[i];
    ssort[i]=ssort[min];
    ssort[min]=t;
    }
    }



    for(i=0;i<N;i++)
    {
    if(ssort[i]!=s[i])
    {
    pos=i;
    minc=ssort[i];
    break;
    }
    }

    if(pos==-1)
    printf("%s\n",s);
    else
    {
    //find sukw- pos se uth ke waha jaye jahan
    // par usse chota ya uske barakar koi n ho-ie pos3 pe;
    if(s[pos+1]==minc)
    {
    pos3=N;
    for(i=pos+1;i<N;i++)
    {
    if(s[i]>s[pos])
    {
    pos3=i;
    break;
    }
    }

    for(i=1,j=pos;i<=pos3-pos-1;i++,j++)
    {
    t=sukw[j];
    sukw[j]=sukw[j+1];
    sukw[j+1]=t;
    }


    }
    //sukw ho gaya!!!

    //ab suky uth ke yaha aaya kya? vo jo ulte se hai minc ke barabar;
    pos3=N-1;
    for(i=N-1;i>pos;i--)
    {
    if(s[i]==minc)
    {
    pos3=i;
    break;
    }
    }

    for(i=1,j=pos3;i<=pos3-pos;i++,j--)
    {
    t=suky[j];
    suky[j]=suky[j-1];
    suky[j-1]=t;
    }

    ///suky ho gayaaa
    ans=0;
    for(i=0;i<N;i++)
    {
    if(suky[i]<sukw[i])
    {
    ans=1;
    break;
    }
    if(sukw[i]<suky[i])
    {
    ans=2;
    break;
    }
    }
    if(ans==1)
    printf("%s\n",suky);
    else
    printf("%s\n",sukw);

    }//end of else
    }
    return 0;
    }
