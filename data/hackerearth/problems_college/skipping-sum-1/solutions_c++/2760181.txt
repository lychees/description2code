#include<stdio.h>
long long int dp[11][100100]={0};
int main()
{
    long long int i,j,n,q,k,a[100100];
    scanf("%lld %lld",&n,&q);
    for(i=1;i<=n;i++)
    {
        scanf("%lld",&a[i]);
    }
    for(i=1;i<=10;i++)
    {
        for(j=1;j<=i;j++)
        {
            dp[i][j]=a[j];
        }
        for(j=i+1;j<=n;j++)
        {
            dp[i][j]=dp[i][j-i]+a[j];
        }
    }
    while(q--)
    {
        long long int l,r;
        scanf("%lld %lld %lld",&l,&r,&k);
        printf("%lld\n",dp[k][r-(r-l)%k]-dp[k][l]+a[l]);
    }
    return(0);
}
