#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
     int t,i;
    long int x,five,three,j,y;
    scanf("%d",&t);
    for(i=0;i<t;i++)
    {
    	scanf("%ld",&x);
    	if(x==1||x==2||x==4||x==7){
    		printf("-1\n");
    		continue;
    	}
    	y= x%3;
    	if(y==0){
    	five = x;
    	three = 0;}
    	else if(y==2){
    		five = x-5;
    		three = 5;
    	}
    	else{
    		five = x-10;
    		three = 10;
    	}
    	for(j=1;j<=five;j++)
    	printf("5");
    	for(j=1;j<=three;j++)
    	printf("3");
    	printf("\n");
    	
    }
    return 0;
}
