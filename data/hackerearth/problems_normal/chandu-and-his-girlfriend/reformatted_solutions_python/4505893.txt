def solve(n, array):
    tmp = sorted(array)[::-1]
    for i in xrange(n):
        tmp[i] = str(tmp[i])
    return ' '.join(tmp)


def sort_array():
    T = int(raw_input())
    for _ in xrange(T):
        n = int(raw_input())
        array = map(int, raw_input().split())
        print solve(n, array)


if __name__ == "__main__":
    sort_array()
