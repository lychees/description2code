#include <bits/stdc++.h> 
using namespace std;

long ans[1<<20];
long tests,n;
int main(){
long long a,b;
for (int i=2;i<=1000000;i++)
{
 a=1;
 b=i;
 while (b<=1000000)
 {
  ans[b]++;
  long ta=b;
  b=b*i-a;
  a=ta;
 }
}
for (int i=1;i<=1000000;i++)
 ans[i]=ans[i-1]+ans[i];

cin>>tests;
for (;tests;--tests)
{
 cin>>n;
  cout<<ans[n]<<endl;
}
cin.get();cin.get();
return 0;}
