#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,arr[26];
	scanf("%d",&t);
	while(t--)
	{
		memset(arr,0,sizeof(arr));
		string s;
		cin>>s;
		int l=s.size(),m=0;
		for(int i=0,j=0;j<l;)
		{
			if(arr[((int)s[j]-(int)('A'))]==1)
			{
				arr[((int)s[i]-(int)('A'))]=0;
				i++;
				}
			else
			{
				arr[((int)s[j]-(int)('A'))]=1;
				if(m<j-i+1)
				m=j-i+1;
				j++;
				}
			}
		printf("%d\n",m);
		}
	return 0;
	}
