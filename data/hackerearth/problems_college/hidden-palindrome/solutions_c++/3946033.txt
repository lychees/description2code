#include <bits/stdc++.h>
//zone
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	string s;
    	cin >> s;
    	int i = 0;
    	int l,r;
    	int maxi = 0;
    	for(i = 1 ; i < s.length() ; i++)
    	{
    		l = i - 1;
    		r = i;
    		
    		while(l >= 0 && r < s.size() && s[l] == s[r])
    		{
    			maxi = max(maxi,r - l + 1);
    			l--;
    			r++;
    		}
    		l = i - 1;
    		r = i + 1;
    		while(l >= 0 && r < s.size() && s[l] == s[r])
    		{
    			maxi = max(maxi, r - l + 1);
    			
    			l--;
    			r++;
    		}
    		
    	}
    	cout<<maxi<<endl;
    }
    return 0;
}
