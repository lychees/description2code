int index(int* a,int i,int j,int x)
{
  if(i>j)
   return -1;
   
  if(i==j-1)
  {
    if(a[j]<=x)
     return j;
    
    if(a[i]<=x)
     return i;
     
    return -1;
  }

  if(i==j)
  {
   if(a[i]<=x)
     return i; 
  return -1;

  }
  
  int k=(i+j)/2;
 
  
  if(a[k]>x)
  {
  	return index(a,i,k-1,x);
  }
  else
  {
  	return index(a,k,j,x);
  }
}

int indexi(int* a,int i,int j,int x)
{
	int res=-1;
	for(int w=j;w>=i;w--)
	{
		if(a[w]<=x)
		 return w;
	}
	return res;
}
