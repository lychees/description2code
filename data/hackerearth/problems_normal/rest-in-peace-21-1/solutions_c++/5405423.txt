#include <iostream>
using namespace std;

int main()
{
    int t,i,flag,r;
    long long int val;
    cin>>t;
    while(t--)
    {
    	flag=0;
    	r=0;
    	cin>>val;
    	if(val%21==0)
    	{
    		flag=1;
    	}
    	while(val>0)
    	{
    		r=val%100;
    		if(r==21)
    		{
    			flag=1;
    		}
    		val=val/10;
    	}
    	if(flag==0)
    		cout<<"The streak lives still in our heart!"<<endl;
    		else
    		if(flag==1)
    		cout<<"The streak is broken!"<<endl;
    }
    return 0;
}
