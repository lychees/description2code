#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	long long int i,n,j;
    	cin >> n;
    	long long c=0;
    	for(i=1;;i++)
    	{
    		j=i*i;
    		if(j<=n)
    		  c++;
    		  else
    		  break;
    	}
    	cout << n-c << endl;
    }
    return 0;
}
