#include <iostream>
#include <vector>
using namespace std;

int main()
{
    vector<bool> prime(100000001,true);

    prime[0]=false;
    prime[1]=false;

    for (int p=2; p*p<=100000000; p++)
    {

        if (prime[p] == true)
        {
            for (int i=p*2; i<=100000000; i += p)
                prime[i] = false;
        }
    }
    
    vector<long long> no(100000001,0);
    
    no[0]=0;
    
    for(int i=1;i<=100000001;i++)
    {
    	if(prime[i])
    	{
    		no[i]=no[i-1]+1;
    	}
    	else
    	no[i]=no[i-1];
    }
    
    int t;
    cin>>t;
    
    
    
    while(t!=0)
    {
    	int l,r;
    	cin>>l>>r;
    	
    	cout << no[r]-no[l-1] << '\n';
    	
    	t--;
    }
    
    
    return 0;
}
