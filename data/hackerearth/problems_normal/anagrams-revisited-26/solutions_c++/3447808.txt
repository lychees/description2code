#include <bits/stdc++.h>

using namespace std;

string arr [100000 + 1];
map < string, int > p;

int main()
{

    int n;

    ios::sync_with_stdio(false);cin.tie(0);

    cin >> n;

    for(int i = 0; i < n; i++) cin >> arr[i];

    int ans = 1, maxi = -1;

    for(int i = 0; i < n; i++){

        string g = arr[i];
        sort(g.begin(), g.end());
        p[g]++;
        if(p[g] > maxi) maxi = p[g];
    }

    cout << maxi;


    return 0;
}
