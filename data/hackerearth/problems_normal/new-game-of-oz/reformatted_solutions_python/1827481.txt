# print 'Hello World!'
from itertools import groupby
from operator import itemgetter
t = input()
for i in range(t):
    siz = input()
    data = raw_input()
    sequence = sorted([int(d) for d in data.split(' ')])
    list_of_consecs = []
    for k, g in groupby(enumerate(sequence), lambda (i, x): i-x):
        list_of_consecs.append(map(itemgetter(1), g))
    count = 0
    for j in range(len(list_of_consecs)):
        if (len(list_of_consecs[j]) == 1):
            count = count+1
        else:
            temp = 0
            temp = len(list_of_consecs[j])/2
            if (len(list_of_consecs[j]) % 2 != 0):
                temp = temp+1
            count = count+temp
    print count
