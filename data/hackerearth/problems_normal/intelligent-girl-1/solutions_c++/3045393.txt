#include <iostream>
using namespace std;

int main()
{
	int in[10000], even[10000], count = 0, len;
	string s;
	cin >> s;
	len = s.length();
	for(int i = 0; i < len; i++){
		in[i] = s[i] - '0';
	}
	for(int i = 0; i < len; i++){
		if(!(in[i]&1)){
			count++;
		}
	}
	for(int i = 0; i < len; i++){
		if(in[i]&1){
			even[i] = count;
		}else{
			even[i] = count;
			count = count - 1;
		}
	}
	for(int i = 0; i < len; i++){
		cout << even[i] << " ";
	}
    return 0;
}
