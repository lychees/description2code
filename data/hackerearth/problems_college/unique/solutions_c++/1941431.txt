// constructing sets
#include <iostream>
#include <set>
using namespace std;

int main ()
{
  std::set<int> first;
  // empty set of ints
  int test;
  cin>>test;
  while(test--){
	  int a,num;
	  cin>>a;
	  while(a--){
		  cin>>num;
		  first.insert(num);
	  }
	  cout<<first.size()<<endl;
	  first.clear();
  }

  return 0;
}
