import math
t=input()
while t>0:
    z,m,n=map(int,raw_input().split())
    y=int(math.sqrt(z))
    cnt=0
    for i in range(1,y+1):
        if z%i==0:
            x=z/i
            if x<=m and x<=n and x!=i:
                cnt+=2
            elif x<=m and x<=n and x==i:
                cnt+=1
            if x<=m and x>n:
                cnt+=1
            elif x>m and x<=n:
                cnt+=1
    print cnt
    t-=1
