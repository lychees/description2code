#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    long long n,f=0,i;
    cin>>n;
    long long s[n],c[n];
    for(i=0;i<n;i++){
    	cin>>s[i];
    
    }
    for(int i=0;i<n;i++)
      	cin>>c[i];
    sort(s,s+n);
    sort(c,c+n);
    for(i=0;i<n;i++){
    	
    	f=f+(s[i]*c[i]);
    }
    
    cout<<f<<'\n';
    return 0;
}
