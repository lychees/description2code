#include <iostream>
#include<cstdio>
#include<cstring>
using namespace std;

int main()
{
    int n,k,count=0,s;
    char ar[101];
    int f[26],c[26];
    cin>>n>>k;
    for(int i=0;i<26;i++)
    	c[i]=0;
    	s=n;
    while(s--)
    {
    	for(int i=0;i<26;i++)
    	f[i]=0;
    	
    	scanf("%s",ar);
    	int num=strlen(ar);
    	for(int i=0;i<num;i++)
    	{
    		if(f[ar[i]-97]==0)
    		{
    			f[ar[i]-97]=1;
    			c[ar[i]-97]++;
    		}
    	}
    }
    for(int i=0;i<26;i++)
    {
    	if(c[i]==n)
    	count++;
    }
    if(count>=k)
    cout<<":-)";
    else
    cout<<":-(";
    return 0;
}
