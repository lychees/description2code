LL prime[1000001];

int main() {
	for (int i=2; i<1001; i++) {
		if (prime[i]==0) {
			for (int j=2*i; j<1000001; j+=i) {
				if (prime[j]==0) prime[j]=i;
			}
		}
	}
	for (int i=0; i<1000001; i++) prime[i]+=prime[i-1];
	
	
	int t;
	scanf("%d", &t);
	
	while (t--) {
		int l, r;
		scanf("%d%d", &l, &r);
		printf("%lld\n", prime[r] - prime[l-1]);
	}
	
	
	return 0;
}
