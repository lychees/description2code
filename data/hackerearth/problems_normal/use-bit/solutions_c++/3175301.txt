#include <bits/stdc++.h>
using namespace std;
vector<string> V;
map<string, int> M;
map<string, int> :: iterator it;

void swap(char *x, char *y)
{
    char temp;
    temp = *x;
    *x = *y;
    *y = temp;
}
void permute(char *a, int l, int r)
{
   int i;
   if (l == r){
      string s;
      for(int p=0;p<=r;p++)
        s+=a[p];
      if(M.find(s)==M.end())
      {
        M[s]=1;
        V.push_back(s);
      }
   }
   else
   {
       for (i = l; i <= r; i++)
       {
          swap((a+l), (a+i));
          permute(a, l+1, r);
          swap((a+l), (a+i)); 
       }
   }
}
int main()
