numOfRatings=int(raw_input())
count=0
popcount=0
temp=[]
if numOfRatings <= 5000 and numOfRatings >=1:
        for i in range(numOfRatings):
                input=int(raw_input())
                if input == 0 and i != 0:
                        temp.pop(i-1-popcount)
                        popcount=popcount+2
                else:
                        temp.append(input)
        for m in range(temp.__len__()):
                count=count+temp[m]
print count
