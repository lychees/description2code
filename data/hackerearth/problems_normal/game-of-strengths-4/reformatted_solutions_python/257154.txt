times = int(raw_input())
for i in range(times):
    numbers = int(raw_input())
    strlist = raw_input()
    numlist = map(int, strlist.split())
    numlist.sort(reverse=True)
    weight = numbers-1
    power = 0
    for num in numlist:
        power += weight*num
        weight -= 2
    print(power*numlist[0]) % (10**9+7)
