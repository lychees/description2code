
#include <bits/stdc++.h>
using namespace std;

int main()
{
   // printf("Hello World!\n");
  int t;
  cin>>t;
  
  while(t--)
  {
  	string s,tr;
  	cin>>s>>tr;
  	int len1 = s.length();
  	
  	long long ans = 0;
  	
  	for(int i=0;i<=len1;i++)
  	{
  		string tmp1 = s.substr(0,i);
  		//cout<<"String 1 "<<tmp1<<endl;
  		
  		string tmp = s.substr(i);
  		//cout<<"Before: " << tmp<<endl;
  		
  		tmp = tmp1 + tr + tmp;
  		//cout<<"After: " << tmp<<endl;
  		if(equal(tmp.begin(), tmp.begin() + tmp.size()/2, tmp.rbegin()) )
  			ans++;
  	}
  	cout<<ans<<endl;
  	
  }
  return 0;
}
