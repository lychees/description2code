for _ in xrange(input()):
    n, a = input(), map(int, raw_input().split())
    b = list(reversed(a))
    if b == sorted(a):
        print n
    else:
        k = 0
        for x in xrange(n):
            z = 0
            for y in xrange(x+1, n):
                if a[x] <= a[y]:
                    z = 1
                    break
            if z == 0:
                k += 1
        print k
