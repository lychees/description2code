#include <bits/stdc++.h>
#include<cstring>
#include<vector>
using namespace std;

vector<string> names;

int main() {
    int t,c,a[100];
    cin>>t;
    c=t;
    memset(a, 0, sizeof(a));
    while(t--)
    {
    	string x;
    	cin>>x;
    	names.push_back(x);
    }
    sort(names.begin(),names.end());
    string t1;
    for(int j = 0; j < c; j++)
  	{
        t1 = names[j];
        for(int i = 0; i < c; i++)
        {
        	if(t1 == names[i]) {
				a[j]++;
        	}
        }
                
  	}
	for(int i=0;i<names.size();i++)
    {
	    	
    	if(i<c-1)
		{			
			if(names[i]==names[i+1]) {}
			else cout<<names[i]<<" "<<a[i]<<endl;
		}
		else
		{
			cout<<names[c-1]<<" "<<a[c-1]<<endl;			
		}
    }
    return 0;
}
