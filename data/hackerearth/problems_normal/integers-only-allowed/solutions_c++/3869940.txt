#include <cstdio>

long long gcd(long long a, long long b)
{
	if (a == 0 || b == 0)
	{
		return a==0?b:a;
	}
	if (a % b == 0)
	{
		return b;
	}
	return gcd(b, a%b);
}

inline long long abs(long long a)
{
	return a<0?-a:a;
}

int main()
{
	int n;
	scanf("%d", &n);
	long long arr[n][2];
	for (int i = 0; i < n; i++)
	{
		scanf("%lld %lld", &arr[i][0], &arr[i][1]);
	}
	long long area = 0, border = 0;
	for (int i = 0; i < n; i++)
	{
		area += arr[i][0] * arr[(i+1)%n][1];
		area -= arr[i][1] * arr[(i+1)%n][0];
	}
	area = abs(area);
	for (int i = 0; i < n; i++)
	{
		border += gcd(abs(arr[(i+1)%n][0]-arr[i][0]), abs(arr[(i+1)%n][1]-arr[i][1]));
	}
	printf("%lld\n", (area-border+2)/2);
}
