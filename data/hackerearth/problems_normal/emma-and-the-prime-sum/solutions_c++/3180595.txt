#include <iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
#define N 10000000

bool prime[N];
long long int array[N];
void precomputation(){
    int i,j;
    long long int sum=0;
    for(i=2;i*i<=N;i++){
        if(prime[i]==true){
            sum=sum+i;
            array[i-1]=sum;
            for(j=i*2;j<=N;j=j+i){
                prime[j]=false;
            }
        }
        else
         array[i-1]=sum;
    }
    
    for(i;i<=N;i++){
        if(prime[i]==true)
        {
            sum=sum+i;
            array[i-1]=sum;
        }
        else
         array[i-1]=sum;
    }
}

int main() {
    long long int ans;
    int t,x,y;
    memset(prime,true,N);
	precomputation();
	scanf("%d",&t);
	while(t--){
	    scanf("%d%d",&x,&y);
	    
	    ans=array[y-1]-array[x-2];
	    printf("%lld\n",ans);
	}
	return 0;
}
