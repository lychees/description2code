#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	float p , e,nw,ans,k;
    	int b,num=0,m;
    	cin >> p >> e;
    	nw = p + (p*e)/100;
    	b = (int)nw;
    	while(b > 0)
    	{
    		num = num*10 + b%10;
    		b = b/10;
    	}
    	k = (p - num)/p;
    	ans = nw + nw*k;
    	m = (int)ans;
    	cout << m << endl;
    	
    }
    return 0;
}
