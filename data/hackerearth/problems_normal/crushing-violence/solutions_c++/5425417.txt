#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long int n;
        cin>>n;
        long int b[n],g[n],j,i,c=0;long int mx;long int bbeat[100001]={0};long int gbeat[100001]={0};
        for(i=1;i<=n;i++)
            cin>>b[i];
        for(i=1;i<=n;i++)
            cin>>g[i];
        for(i=1;i<=n;i++)
        {
            if(i!=g[b[i]])
                bbeat[g[b[i]]]++;
            if(i!=b[g[i]])
                gbeat[b[g[i]]]++;
        }
        mx=-1;
        for(i=1;i<=n;i++)
        {
            mx=max(mx,max(bbeat[i],gbeat[i]));
        }
        for(i=1;i<=n;i++)
        {
           j=g[b[i]];
           if(j!=i && i==g[b[j]])
               c++;
           j=b[g[i]];
           if(j!=i && i==b[g[j]])
               c++;
        }
        cout<<mx<<" "<<(c/2)<<endl;
    }
    return 0;
}
