#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
    
    int t;
    cin>>t;
    while(t--){
    	int n;
    	int sum=0;
    	cin>>n;
    	for(int i=0;i<n;i++){
    		int x;
    		cin>>x;
    		if(x&1)
    		sum+=x;
    		
    	}
    	
    	
    	cout<<sum<<endl;
    	
    }
    
    return 0;
}
