    #include <iostream>
    #include<algorithm>
    #include<cmath>
    #define ll long long
    #define mod 1000000007
    using namespace std;
  
     
    class Point
    {
    public:
        ll x, y;
        Point(ll a=0, ll b=0):x(a),y(b) {}
    };
     
    ll gcd(ll a, ll b)
    {
        if (b == 0)
           return a;
        return gcd(b, a%b);
    }
     
    int getCount(Point p, Point q)
    {
     
        if (p.x==q.x)
            return abs(p.y - q.y) - 1;
     
        if (p.y == q.y)
            return abs(p.x-q.x) - 1;
     
        return gcd(abs(p.x-q.x), abs(p.y-q.y))-1;
    }
    int main() {
    	ll n;
    	cin>>n;
    	Point arr[n];
    	scanf("%lld%lld",&arr[0].x,&arr[0].y);
    	ll B=0,A=0;
    	for(ll i=1;i<n;i++)
    	{
    		scanf("%lld%lld",&arr[i].x,&arr[i].y);
    		B+=getCount(arr[i-1],arr[i]);
    		A+=(arr[i-1].x*arr[i].y)-(arr[i-1].y*arr[i].x);
    	}
    	B+=getCount(arr[n-1],arr[0])+n;
    	A+=(arr[n-1].x*arr[0].y)-(arr[n-1].y*arr[0].x);
    	cout<<(abs(A)+2-B)/2;
    	//cout<<"Area*2 = "<<A<<" "<<" B = "<<B;
    	return 0;
    }
