#include<cstdio>
using namespace std;
char a[100000];
long long b[100001];
int main(){
    b[1]=1;
    for(int i=2;i<100001;i++){
        b[i]=b[i-1]+i;
    }
int t;
scanf("%d",&t);
while(t--){
scanf("%s",&a);
int n;
scanf("%d",&n);
int i=0,count=0;
while(a[i]!='\0'){
    if(a[i]=='1'){
        count++;
    }
    i++;
}
if(b[n]%2==0){
    printf("%d\n",count);
}
else{
    printf("%d\n",i-count);
}
}
}
