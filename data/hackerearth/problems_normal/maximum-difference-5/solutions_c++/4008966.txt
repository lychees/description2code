#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    long long int n,i,k,j,arun=0;
    cin>>n;
    long long int a[n];
    for(i=0;i<n;i++){
        cin>>a[i];
    }
    k=sizeof(a)/sizeof(long long int);
    sort(a,a+k);
    //arun=abs(a[i]-a[k-1]);
    if(k%2==0){
    for(i=0,j=k-1;i<=j;i++,j--){
        arun+=abs(a[i]-a[j]);
        if(j-1>=i){
        arun+=abs(a[i]-a[j-1]);
        }
    }
    }
    else{
    for(i=0,j=k-1;i<=j;i++,j--){
        arun+=abs(a[i]-a[j]);
        if(j+1<k){
        arun+=abs(a[i]-a[j+1]);
        }
    }
    }
    cout<<arun;
}
