#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
#define PII       pair<int,int>
#define all(c)    c.begin(),c.end()
#define sz(c)     (int)c.size()
#define clr(c)    c.clear()
#define pb        push_back
#define mp        make_pair
#define cin(x)    scanf("%d",&x)
#define MOD       1000000007
#define EPS       1E-10
#define LMT		800010
#define MAXN	222
#define OFFSET	400000
int dp[2][LMT + 1];
int A[MAXN];
int B[MAXN];
int main()
{
	for(int i = 0; i < 2; i++)
		for(int j = 0; j <= LMT; j++)
			dp[i][j] = 0;
	int n;
	cin(n);
	for(int i = 0; i < n; i++)
		cin(A[i]);
	for(int i = 0; i < n; i++)
		cin(B[i]);
	dp[0][OFFSET] = 1;
	for(int i = 0; i < n; i++)
	{
		for(int j = 0; j <= LMT; j++)
		{
			if(!dp[0][j]) continue;
			int nextVal = j;
			dp[1][nextVal] += dp[0][j];
			dp[1][nextVal] %= MOD;
			nextVal = j + A[i];
			dp[1][nextVal] += dp[0][j];
			dp[1][nextVal] %= MOD;
			nextVal = j - B[i];
			dp[1][nextVal] += dp[0][j];
			dp[1][nextVal] %= MOD;
			nextVal = j + A[i] - B[i];
			dp[1][nextVal] += dp[0][j];
			dp[1][nextVal] %= MOD;
		}
		for(int j = 0; j <= LMT; j++)
		{
			dp[0][j] = dp[1][j];
			dp[1][j] = 0;
		}
	}
	int Q;
	cin(Q);
	int ans = 0;
	int L = max(0 , OFFSET - Q);
	int R = min(LMT , OFFSET + Q);
	for(int i = L; i <= R; i++)
		ans = (ans + dp[0][i]) % MOD;
	printf("%d\n", ans);
      return 0; }
