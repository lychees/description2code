#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long int n,m,k=0,p,q=0;
		cin>>n;
		m=n;
		p=m*m;  // p is sq. of n
		while(m)
		{
			k*=10;
			k+=(m%10);
			m/=10;
		}
		//cout<<k<<endl;
		m=k;               // m is Reverse no.
		k=k*k;
		while(k)
		{
			q*=10;
			q+=(k%10);            //q is reverse of square
			k/=10;
		}                 
		//cout<<q<<endl;
		if(q==p && n != m)
		{
			cout<<m<<'\n';
		}
		else
		{
			cout<<"No"<<'\n';
		}
	}
	return 0;
}
