#include <stdio.h>
#include <vector>
#include <string.h>
using namespace std ;

int a, b , x , t , n , tot ;
char s[1000100] ;

int main()
{
    scanf("%d",&t) ;
    while( t-- )
    {
      	
       scanf("%s%d%d",&s,&a,&b) ;    	
       x = 2*min( a,b ) ;  tot = 0  ; n = strlen(s) ;
       
       for( int i = 0 ; i < n ; i++ )
       {
         if( s[i] != '/' && s[n-i-1] != '/' && s[i] != s[n-i-1] ) 
          { tot = 0 ; break ; }	
       
         if( s[i] == '/' )  
       	  { 
       	  	if( s[n-i-1] != '/' )  tot += ( ( s[n-i-1] == 'b' )?b:a ) ;
       	    else { tot += x ; s[i] = 'a' ; s[n-i-1] = 'a' ; }  
       	  }
       }
       if( tot ) printf("%d\n",tot) ;
       else printf("-1\n") ;
    }
    
  return 0;
}
