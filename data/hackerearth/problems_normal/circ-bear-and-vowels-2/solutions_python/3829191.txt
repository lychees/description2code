
result=[]
vowels = list("aeiouy")
consonants = list("bcdfghjklmnpqrstwvxz")


class Mystring:
	def __init__(self,text):
		self.text=text
		self.index=len(text)

	def __iter__(self):
		return self

	def next(self):
		if self.index==0:
			raise StopIteration	
		self.index=self.index-1
		return self.text[self.index]
	
	def check(self):
		if sum(self.text.count(c) for c in consonants)>sum(self.text.count(v) for v in vowels):
			return 'hard'
		for i in range(len(self.text)-2):
			if self.text[i]  in consonants and self.text[i+1] in consonants and self.text[i+2] in consonants:
				return 'hard'
		return 'easy'

	text_type=check	



test_cases=int(raw_input(""))
for i in range(test_cases):
	test=Mystring(raw_input(""))
	c=test.text_type()
	result.append(c)


for each in result:
	print each
