#include <bits/stdc++.h>
using namespace std;
#define ll long long
long long a[100009];
long long  l[100009];
long long  r[100009];
#define mod 1000000007
int main()
{
    int t;cin>>t;
    while(t--){
    	long long n;cin>>n;
    	for(ll int i=0;i<n;i++){
    		cin>>a[i];
    	//	cout<<a[i]<<endl;
    	}
    
    stack<ll int>st;
    for(ll int i=0;i<n;i++){
    	if(st.empty()){
    		l[i]=i;
    	
    	}
    	else{
    		
    	    while((!st.empty()) && a[st.top()]<a[i]){
    	    	//cout<<"popped"<<" "<<st.top()<<endl;
    	    st.pop();
    	    }
    	    if(st.empty())
    	    l[i]=0;
    	    else
    	    l[i]=st.top();
    	}
    //	cout<<l[i]<<endl;
    		st.push(i);
    }
    stack<int>s;
    for(ll int i=n-1;i>=0;i--){
    	if(s.empty()){
    		r[i]=i;
    	
    	}
    	else{
    	    while((!s.empty()) && a[s.top()]<a[i])
    	    s.pop();
    	    if(s.empty())
    	    r[i]=n-1;
    	    else
    	    r[i]=s.top();
    	}
    //	cout<<r[i]<<endl;
    		s.push(i);
    }

    
    long long ans=0LL;
    	int index=1;
    	for(int i=0;i<n;i++){
    		l[i]=i-l[i];
    		r[i]-=i;
    	}

    for(int i=0;i<n;i++){
    	long  long x=l[i];
    	long long  y=r[i];
        long long j=i+1;
        j=(j*(x+y))%mod;
      
    if(ans<j)
    {
    	ans=j;
    	index=i+1;
    }
    }
    cout<<index<<endl;
}
return 0;
}
