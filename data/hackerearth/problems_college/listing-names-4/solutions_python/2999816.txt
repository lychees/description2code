'''
# Read input from stdin and provide input before running code

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
t = int(raw_input())

names = {}

for x in range(t):
	name = raw_input()
	if name not in names:
		names[name] = 1
	else:
		names[name] += 1

a = names.keys()
a.sort()

for name in a:
	print name + ' ' + str(names[name])
