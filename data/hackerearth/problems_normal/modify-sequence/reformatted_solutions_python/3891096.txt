n = int(raw_input())
nums = list(map(int, raw_input().split()))
for i in xrange(n-1):
    if nums[i] > nums[i+1]:
        nums[i] = nums[i] - nums[i+1]
        nums[i+1] = 0
    else:
        nums[i+1] = nums[i+1] - nums[i]
        nums[i] = 0
print 'YES' if all([nums[i] == 0 for i in xrange(n)]) else 'NO'
