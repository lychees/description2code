#include<bits/stdc++.h>
#define pb push_back
using namespace std;
vector<int>v[100001];
int dp[100001];
long int n;
bool vis[100001];
void dfs(int u){
vis[u]=true;
dp[u]++;
int i,j;
for(i=0;i<v[u].size();i++)
    {   if(!vis[v[u][i]])
            {
                dfs(v[u][i]);
                dp[u]+=dp[v[u][i]];
            }
    }
    }
int main(){
ios_base::sync_with_stdio(0);
cin>>n;
int i,x,y;
memset(vis,false,sizeof(vis));
memset(dp,0,sizeof(dp));
for(i=1;i<=n-1;i++){
    cin>>x>>y;
    v[x].pb(y);
    v[y].pb(x);

}
dfs(1);
dp[1]--;
for(i=1;i<=n;i++)
    cout<<dp[i]<<" ";

return 0;
}
