#include<iostream>
#include<queue>
using namespace std;
int main()
{
int n,count=0;
cin>>n;
queue <int> q;
for(int i=0;i<n;i++)
{
int t;
cin>>t;
q.push(t);
}
for(int i=0;i<n;i++)
{
int x;
cin>>x;
count++;
while(!(q.front()==x))
{
int temp=q.front();
q.pop();
q.push(temp);
count++;
}
q.pop();
}
cout<<count;

}
