#include "bits/stdc++.h"
using namespace std;
int main()
{
	long long n;
	cin >> n;
	
	long long arr[n];
	for(long long i=0;i<n;i++)
		cin >> arr[i];
	
	long long div=1;
	long long q;
	cin >> q;
	while(q--)
	{
		long long a;
		cin >> a;
		div =div*a;
		if(div >= 100000000)
			div=0;
	}
	if(div == 0)
		for(long long i=0;i<n;i++)
			cout << 0 << " ";
	else
		for(long long i=0;i<n;i++)
			cout << arr[i]/div << " " ;
}
