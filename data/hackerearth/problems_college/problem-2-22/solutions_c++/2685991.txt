#include<iostream>
#include<cmath>
using namespace std;
int dest(int x1, int y1, int x2, int y2)
{
	return (x2 - x1)*(x2 - x1) + (y2 - y1)*(y2 - y1);
}
int main()
{
	int t;
	scanf("%d", &t);
	for (int tt = 1; tt <= t; tt++)
	{
		int r, xc, yc;
		scanf("%d%d%d", &r, &xc, &yc);
		bool ok = true;
		for (int i = 1; i <= 4; i++)
		{
			int x, y;
			scanf("%d%d", &x, &y);
			if (dest(xc, yc, x, y) > r*r)
			{
				ok = false;
			}
		}
		if (ok)
		{
			printf("Yes\n");
		}
		else
		{
			printf("No\n");
		}
	}
}
