#include <cstdio>
#include<iostream>
using namespace std;

int main()
{	int t,l;
	long long int ans, len;
	string str;
    	cin>>t;
	while ( t-- > 0 )
	{	cin>>l;
		cin>>str;
		ans=0, len=1;
		for( int i=1; i < str.size(); i++ )
		{	if ( str[i] >= str[i-1] )
				len++;
			else
			{	ans += (len*(len+1))/2;
				len=1;
			}
		}
		ans += (len*(len+1))/2;
		cout<<ans<<endl;
	}
	return 0;
}
