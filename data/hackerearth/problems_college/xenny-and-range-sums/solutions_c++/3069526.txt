#include<bits/stdc++.h>
using namespace std;
long ar[1000003];
int main()
{
	ios_base::sync_with_stdio(false);
	long n,q,temp,l,r;
	cin>>n>>q;
	long long total=0;
	ar[0]=0;
	for(int i=0;i<n;i++)
	{
		cin>>temp;
		total+=temp;
		ar[i+1]=total;
	}
	for(int  i=0;i<q;i++)
	{
		cin>>l>>r;
		cout<<ar[r]-ar[l-1]<<" "<<(ar[r]-ar[r-1])-(ar[r-1]-ar[l-1])<<endl;
	}
	return 0;
}
