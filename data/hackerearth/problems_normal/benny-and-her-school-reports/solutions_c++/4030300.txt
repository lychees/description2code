#include <iostream>
using namespace std;

int main()
{
    long long int t,n,m,x;
    cin>>t;
    while(t--)
    {
    	cin>>n>>m>>x;
    	long long int a[n],sum=0;
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    		sum+=a[i];
    	}
    	long long int y=(x*(n+1))-sum;
    	if(y<=m)
    	{
    		if(y<=1)
    			cout<<"1"<<endl;
    		else
    			cout<<y<<endl;
    	}
    	else
    		cout<<"Impossible"<<endl;
    }
    return 0;
}
