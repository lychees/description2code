#include <bits/stdc++.h>
using namespace std;

int a[1000006];
int main()
{
    int t,n,i,j,x,p;
    
    for(i=2;i<=1000;i++)
    { j=2;
       while(pow(i,j)<=1000000)
    	{ int p=int(pow(i,j));
    	  a[p]++;
    		j++;
    	}
    }
    a[1]=1;
    cin>>t;
    while(t--)
    {
     cin>>n;
     int f=0;
     for(i=1;i<(n>>1)+1;i++)
     { if(a[i]&&a[n-i])
       {f=1; break;}
     	
     }
     
     if(f)
     cout<<"Yes"<<endl;
     else
     cout<<"No"<<endl;
     }
    return 0;
}
