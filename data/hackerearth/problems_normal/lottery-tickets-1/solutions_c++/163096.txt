#include <bits/stdc++.h>
using namespace std;
#define MAX 100
typedef long long LL;
typedef pair<LL,LL> pii;

LL dp[MAX],N;
LL st[MAX],p[MAX];
pii en[MAX];

void solve(){
    cin >> N;
    LL a,b,c;
    for(LL i = 1; i <= N; ++i)
        cin >> st[i] >> en[i].first >> p[i], en[i].second = i;
    sort(en+1,en+1+N);

    en[0].first = dp[0] = 0;

    for(LL i = 1; i <= N; ++i){
        LL s = st[en[i].second];
        LL left = 0, right = i-1;
        while(left < right){
            LL mid = (((left+right) >> 1) + 1);
            if (en[mid].first <= s)
                left = mid;
            else
                right = mid-1;
        }
        dp[i] = dp[left] + p[en[i].second];
        dp[i] = max(dp[i],dp[i-1]);
    }
    double ans = (double)dp[N];
    ans = ans/(double)100.0;
    printf("%0.2lf\n",ans);
}

int main()
{
    ios_base::sync_with_stdio(0);
    LL t; cin >> t;
    while(t--) solve();
    return 0;
}
