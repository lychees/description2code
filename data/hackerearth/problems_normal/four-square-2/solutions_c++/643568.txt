#include<iostream>
#include<cmath>
#define ll long long int
using namespace std;
int main()
{
	int t,sq,i;
	cin >> t;
	while(t--) {
		ll n,c,s;
		c = 0;
		cin >> n;
		sq = sqrt(n);
		for(i = 1 ; i <= sq ; i++) {
			if(n % i == 0) {
				if(i % 4 != 0) {
					c = c + i;
				}
				if((n/i)% 4 != 0 && (n/i) != i) {
					c = c + (n/i);
				}
			}
		}
		s = c * 8;
		cout<<s<<endl;
	}
	return 0;
}
