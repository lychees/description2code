#include<bits/stdc++.h>
#define INF 1000000000
using namespace std;

int main(){
    int T;
    scanf("%d",&T);
    while(T--){
        int n;
        scanf("%d",&n);
        vector<int> A(n);
        for(int i=0; i<n; i++)
            scanf("%d",&A[i]);
        vector<int> d(n+1,INF);
        for(int i=0; i<n; i++)
            *upper_bound(d.begin(),d.end(),A[i]) = A[i];
        for(int i=0; i<=n; i++){
            if(d[i] == INF){
                printf("%d\n",i);
                break;
            }
        }
    }
    return 0;
}
