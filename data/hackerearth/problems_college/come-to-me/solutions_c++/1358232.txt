#include <algorithm>
#include <iostream>
#include <string>

bool mark[256];

int main() {
    std::cin.tie(NULL);
    std::ios::sync_with_stdio(false);
    int tt;
    std::cin >> tt;
    while (tt--) {
        std::string s;
        std::cin >> s;
        int ans = 0;
        int sz = s.size();
        for (int i = 0, j = 0; i < sz; ++i) {
            while (j < sz && !mark[s[j]]) {
                mark[s[j++]] = true;
            }
            ans = std::max(ans, j - i);
            mark[s[i]] = false;
        }
        std::cout << ans << '\n';
    }
}
