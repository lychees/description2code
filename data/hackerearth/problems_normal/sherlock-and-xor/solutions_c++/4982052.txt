#include <iostream>
using namespace std;

int main()
{
    int n,t;
    unsigned long int odd,even,buff;
    cin>>t;
    while(t){
    	cin>>n;
    	even=0;
    	odd=0;
    	for(int x=0;x<n;x++){
    		cin>>buff;
    		if (buff&1)odd++;
    		else even++;
    	}
    	buff=odd*even;
    	cout<<buff<<endl;
    	t--;
    }
    return 0;
}
