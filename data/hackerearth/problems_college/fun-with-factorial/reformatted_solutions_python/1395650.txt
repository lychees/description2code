def print_five(number):
    result = 1
    start = 2
    while start <= number:
        result = result * start
        start = start + 1
    result = str(result)
    output = ''
    if len(result) < 5:
        output = output + '0'*(5-len(result))
        output = output + result
    else:
        output = output + result[-5:]
    print output


if __name__ == '__main__':
    cases = int(raw_input())
    start = 1
    while start <= cases:
        number = int(raw_input().strip())
        print_five(number)
        start = start + 1
