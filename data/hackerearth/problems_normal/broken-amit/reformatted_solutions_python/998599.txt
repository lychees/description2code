def doit(girls):
    i = sorted(girls.keys(), key=lambda x: x*girls[x][1], reverse=True)
    nogirls = len(girls.keys())
    gross = 0
    for j in i:
        tot = sum([girls[k][0] for k in range(nogirls)])
        girls[j][1] = 0
        for k in range(nogirls):
            girls[k][0] -= k*girls[k][1]
        gross += tot
    return gross


girls = {}
nouse = input()
lvalues = map(int, raw_input().split())
dvalues = map(int, raw_input().split())
for i in range(nouse):
    girls[i] = [lvalues[i], dvalues[i]]
print doit(girls)
