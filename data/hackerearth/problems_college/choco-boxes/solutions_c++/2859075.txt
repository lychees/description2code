#include <bits/stdc++.h>

using namespace std;

#define icin(x) scanf("%d",&x)
#define lcin(x) scanf("%lld",&x)

typedef long long LL;
typedef vector<int> vi;
typedef vector<vi> vvi; 

int main(){
  int t;
  icin(t);
  while(t--){
    int n,m;
    icin(n);
    icin(m);
    int idx,nc;
    if(n&1){
      nc = ((m-1)/(n<<1))<<1;
      m=m-nc*n;
      if(m<=n){
        if(m&1) idx=(m>>1)+1;
        else idx=n-(m>>1)+1;
      }
      else{
        nc++;
        m=m-n;
        if(m&1) idx=n-(m>>1);
        else idx=(m>>1);
      }
    }
    else{
      nc = (m-1)/n;
      m = m - nc*n;
      if(m&1) idx=(m>>1)+1;
      else idx=n-(m>>1)+1;
    }
    printf("%d %d\n",idx,nc);
  }
  return 0;
}
