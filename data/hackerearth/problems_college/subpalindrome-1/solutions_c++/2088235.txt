using namespace std;
template <typename T>
T modpow(T base, T exp, T modulus) {
  base %= modulus;
  T result = 1;
  while (exp > 0) {
    if (exp & 1) result = (result * base) % modulus;
    base = (base * base) % modulus;
    exp >>= 1;
  }
  return result;
}
int dp[1001][1001];
int main()
{
    _
    __
    {
        string s;
        memset(dp,0,sizeof dp);
        cin>>s;
        int l=s.length();
        for(int i=l-1;i>=0;i--)
        {
            int ct=1;
            dp[i][i]=1;
            if(i+1!=l)
            {
                if(s[i]==s[i+1])
                {
                    ct++;
                    dp[i][i+1]=3;
                }
                else
                {
                    dp[i][i+1]=2;
                }
            }
            for(int j=i+2;j<l;j++)
            {
                if(s[i]==s[j])
                {
                    ct++;
                    ct+=dp[i+1][j-1];
                }
                dp[i][j]=ct+dp[i+1][j];
            }
        }
        cout<<dp[0][l-1]<<endl;
    }
    return 0;
}
