#include <bits/stdc++.h>
using namespace std;
#define int long long
long long f(int n)
{
  long long count = 0;
  while(n)
  {
    count += n & 1LL;
    n >>= 1;
  }
  return count;
}

main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int a[n];
    	for(int i=0;i<n;i++)
    	cin>>a[i];
    	
    	long long c=INT_MAX;
    	for(int i=0;i<n;i++)
    	{
    		
    			c = min(c,f(a[i]));
    	
    	}
    	cout<<c<<endl;
    }
    return 0;
}
