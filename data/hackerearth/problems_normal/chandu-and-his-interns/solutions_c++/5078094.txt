#define _CRT_SECURE_NO_WARNINGS
#include <iostream>

using namespace std;


int main()
{
	int i, j, N, B, cnt;

#ifdef _MSC_VER
	freopen("Text.txt", "r", stdin);
#endif

	cin >> N;

	for (i = 0; i < N; i++)
	{
		cin >> B;
		cnt = 0;
		for (j = 2; (j*j < B) && (cnt < 2); j++)
		{
			if ((B % j) == 0)
			{
				cnt += 2;
			}
		}

		if (cnt < 2)
		{
			cout << "NO" << endl;
		}
		else
		{
			cout << "YES" << endl;

		}
	}

	return 0;
}
