#include <iostream>
#include<cstdio>

using namespace std;

int main()
{
    int t,a[10001],x,y,n,i,s;
    cin>>t;
    while(t--){
    	s=0;
    	scanf("%d",&n);
    	for(i=0;i<=10000;i++)
    		a[i]=0;
    	for(i=0;i<n;i++){
    		scanf("%d%d",&x,&y);
    		if(!a[x]){
    			a[x]=1;
    			s++;
    		}
    		if(!a[y]){
    			a[y]=1;
    			s++;
    		}
    	}
    	printf("%d\n",s);
    }
    return 0;
}
