#include <bits/stdc++.h>

using namespace std;
int t, n;
pair<int,int> A[205];
long long dp[205];

int main()
{
    cin >> t;
    while (t--){
        cin >> n;
        for (int i = 0; i < n; i++) cin >> A[i].first >> A[i].second;
        sort(A, A + n);
        memset(dp, 0, sizeof dp);
        long long maxi = 0;
        for (int i = 0; i < n; i++){
            dp[i] = A[i].second;
            for (int j = 0; j < i; j++){
                if (A[i].first > A[j].first && A[i].second > A[j].second)
                    dp[i] = max(dp[i], dp[j] + A[i].second);
            }
            maxi = max(maxi, dp[i]);
        }
        cout << maxi << endl;
    }
}
