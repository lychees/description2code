#include <iostream>
#include <cmath>
#include <string.h>
using namespace std;

int convert(int a, int b){
    int n=0,i=0;
    while(a!=0){
        n = (pow(10,i++)*(a%b)) + n;
        a = a/b;
    }
    return n;
}

int main(){
    int T,A,B;
    cin>>T;
    for(int i=0;i<T;i++){
        cin>>A>>B;
        cout<<convert(A,B)<<endl;
    }
    return 0;
}
