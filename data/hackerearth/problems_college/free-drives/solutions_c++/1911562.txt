#include<cstdio>
#include <iostream>
using namespace std;
long long C[1000001][3];
void binomialCoeff(int n, int k)
{
    int i, j;
 
    for (i = 0; i <= n; i++)
    {
        for (j = 0; j <= min(i, k); j++)
        {
            if (j == 0 || j == i)
                C[i][j] = 1;
 
            else{
                C[i][j] = C[i-1][j-1] + C[i-1][j];C[i][j]=C[i][j]%1000000000;}
        }
    }
 
}


int main()
{	binomialCoeff(1000000,2);

    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	cout<<C[n][2]<<"\n";
    }
    return 0;
}
