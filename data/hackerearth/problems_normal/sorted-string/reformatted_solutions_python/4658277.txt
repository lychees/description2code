n = input()
d = dict()
v = dict()


def fun(v):
    s = ""
    for i in range(len(v)):
        for j in range(v[i][0]):
            s = s + str(v[i][1])
    print s


for j in range(n):
    m = raw_input().strip()
    for i in m:
        d[i] = d.get(i, 0)+1
        v = sorted([(value, key) for (key, value) in d.items()])

    fun(v)
    d = dict()
