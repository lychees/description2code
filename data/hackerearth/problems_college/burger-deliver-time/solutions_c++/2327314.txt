#include<iostream>
#include<set>
#include<map>
using namespace std;
int main(){
	int n,i,a,b,j=0;
	map<int,set<int> > mp;
	map<int,set<int> >::iterator imp;
	set<int>::iterator is;
	cin>>n;
	for(i=0;i<n;i++){
		cin>>a>>b;
		mp[a+b].insert(++j);
	}
	
	for(imp=mp.begin();imp!=mp.end();imp++){
		for(is=imp->second.begin();is!=imp->second.end();is++)
			cout<<*is<<" ";
	}
	return 0;
}
