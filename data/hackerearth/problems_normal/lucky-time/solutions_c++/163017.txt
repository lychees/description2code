int readint()
{
    int t=0;
    char c,ch;
    ch=getchar();
    if (ch=='-') c=getchar();
    else c=ch;
    while (c<'0' || c>'9')
        c=getchar();
    while (c>='0' && c<='9')
    {
        t= (t<<3)+ (t<<1)+c-'0';
        c=getchar();
    }
    if (ch=='-') return -t;
    else return t;
}

{
    int h,m,s;
} tim;
bool diff (tim a)
{

    f[a.m/10]=1;

    f[a.s/10]=1;
    int c=0;

}
tim inc (tim a)
{
    int c=0;
    a.s+=1;
    c=a.s/60;
    a.s=a.s%60;
    a.m=a.m+c;
    c=a.m/60;
    a.m=a.m%60;
    a.h= (a.h+c) %24;
    return a;
}
