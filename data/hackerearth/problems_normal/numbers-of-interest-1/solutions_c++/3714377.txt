#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long int x,y,n,index;
		cin>>x>>y>>n;
		
		long int arr[n];
		for(int i=0;i<n;i++)
		arr[i]=0;
		for(int i=0;i<n;)
		{
			if(i<x)
			{
				arr[i]=y;
				i++;
				
				//cout<<arr[i];
			}
			else
			{
				long int count=x;
				long int j=i-1;
				long int sum=0;
				while(count--)
				{
					sum+=arr[j];
					j--;
				}
				//cout<<sum<<" "<<endl;
				arr[i]=sum;
				i++;
				
			}
		}
		cout<<arr[n-1]<<endl;
	}
}
