#include <iostream>
using namespace std;

int n;
int a[100], x[100];

void solve(){
	cin >> n;
	int ans = 0;
	for(int i = 0; i < n; i++) scanf("%d", a + i);
	for(int i = 0; i < n; i++) scanf("%d", x + i);
	for(int i = 0; i < n; i++){
		ans ^= a[i] - x[i];
		ans ^= x[i]  - 1;
	}
	if (!ans) puts("Second");
	else puts("First");
}

int main()
{
    int test;
    cin >> test;
    while (test-- > 0){
    	solve();
    }
    return 0;
}
