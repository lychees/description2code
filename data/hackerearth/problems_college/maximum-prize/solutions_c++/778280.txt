#include<iostream>
#include <algorithm>
using namespace std;

int main ()
{
	int i,j,n,k , a[100001] ;
	long long int sum =0 ;
	int t;
	cin>>t;
	while (t--)
	{
		cin>>n>>k ;
		for ( i = 0; i < n; i += 1)
			cin>>a[i] ;
		sort(a,a+n) ;
		sum = 0 ;
		for( i = n-1; i > n-k-1 && i>=0; i -= 1)
			sum+= a[i] ;
		cout<<sum << "\n" ;
	}
	return 0;
}
