#include <bits/stdc++.h>
using namespace std;
#define ll long long

int main()
{
    int t;
    cin >> t;
    
    while(t--)
    {
    	int n;
    	cin >> n;
    	
    	ll *A=new ll[n];
    	for(int i=0;i<n;i++)
    		cin >> A[i];
    		
    	sort(A,A+n);
    	bool b=false;
    	ll x,y,z;
    	for(int i=n-1;i>=2;i--)
    	{
    		if(A[i]<A[i-1]+A[i-2])
    		{
    			x=A[i];
    			y=A[i-1];
    			z=A[i-2];
    			b=true;
    			break;
    		}
    	}
    	if(b)
    		cout << z << " " << y << " " << x << endl;
    	else
    		cout << -1 << endl;
    }
}
