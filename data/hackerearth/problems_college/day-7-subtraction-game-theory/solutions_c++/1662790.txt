#include <iostream>
#include<stdio.h>
using namespace std;
#define max 1000000
#define mod 1000000007
int main()
{
    long int x,y,result;
    scanf("%ld %ld",&x,&y);
   long int t,i;
    long long int a[max];
    a[0]=1;
    
    for(i=1;i<=max;i++)
    {
    	a[i]=((a[i-1]*x)-y)%mod;
    }
   scanf("%ld",&t);
   while(t--)
   {
   	long long int n;
   	scanf("%lld",&n);
   	printf("%lld\n",a[n]);
   }
    return 0;
}
