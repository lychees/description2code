#include<iostream>
using namespace std;
int main()
{
    int t ;
    cin >> t;
    while(t--)
    {
        long long int n,m;
        long long int a = 0;
        cin >> n >> m;
        if(n < m || m == 1)
        {
        	 cout << n << endl;
        }
        else{
        	while(n != 0){
            	a += (n%m);
            	n = n/m;
    		}
    		cout << a << endl;
        }
    }
    return 0;
}
