#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    string s1,s2,s,a;
    cin>>t;
    while(t--)
    {	int flag=0;
    	cin>>s1;
    	cin>>s2;
    	s=s1+s2;
    //	cout<<s<<"\n";
    	int l=s.length();
    	for(int i=0,j=l-1;i<l/2;++i,--j)
    	{
    		if(s[i]!=s[j])
    		{
    			flag=1;
    			break;
    		}
    	}
    	if(flag==0)
    	printf("YES\n");
    	else{
    		flag=0;
    	a=s2+s1;
    //		cout<<a<<"\n";
    	int l=a.length();
    	for(int i=0,j=l-1;i<l/2;++i,--j)
    	{
    		if(a[i]!=a[j])
    		{
    			flag=1;
    			break;
    		}
    	}
    	if(flag==0)
    	printf("YES\n");
    	else
    	printf("NO\n");
    	}
    }
    return 0;
}
