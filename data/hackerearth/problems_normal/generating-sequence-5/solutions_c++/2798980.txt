#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
    int T,n;
    long long int g;
    scanf("%d",&T);
    for(int i =0;i < T;i++){
    	scanf("%lld %d",&g,&n);
    	for(int j = 0;j < n-1;j++){
    		printf("%lld ",2*g);
    	}
    	printf("%lld\n",3*g);
    }
    return 0;
}
