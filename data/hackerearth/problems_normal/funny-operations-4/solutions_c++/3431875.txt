#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,mycn=0,res=0;
    	cin>>n;
    	vector<int> v;
    	for(int i=0;i<n;i++)
    	{
    		int a;
    		cin>>a;
    		v.push_back(a);
    	}
    	for(int i=0;i<(1<<n);i++)
    	{
    		int and1=0,xor1=0,ans=0,cn=0;
    		for(int j=0;j<n;j++)
    		{
    		
    			if(i&(1<<j))
    			{
    				cn++;
    				if(cn==1)
    				   and1=v[j];
    				else
    				   and1=and1&v[j];
    				xor1^=v[j];
    			}
    		  }
    		  ans=xor1|and1;
    			if(ans>res)
    			{
    				mycn=cn;
    				res=ans;
    			}
    			else if(ans==res)
    			{
    				if(cn>mycn)
    				{
    					mycn=cn;
    				}
    			}
    	}
    	printf("%d %d\n",res,mycn);
    }
    return 0;
}
