#include <iostream>
#include <cstring>
#include <map>
using namespace std;

int main()
{
    string s;
    cin >> s;
    int n;
    int l = s.length();
    cin >>n;
    map<string,int> m;
    for(int i=0;i <n;i++)
    {
    	string q;
    	cin >> q;
    	m[q] = 1;
    }
    int flag=1;
    for(map<string,int>::iterator it=m.begin(); it != m.end(); it++)
    {
    	string w = (*it).first;
    	if(  w.length() >= l && (w.substr(0,l)) == s)
    	{
    		cout <<w<<endl;
    		flag =0;
    		break;
    	}
    }
    if(flag)
    	cout<<s<<endl;
    
    return 0;
}
