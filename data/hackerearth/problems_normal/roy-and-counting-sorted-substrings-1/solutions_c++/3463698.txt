#include <iostream>
using namespace std;
#define ll long long

int main()
{	ll t;
    cin>>t;
    while(t--)
    {	ll n,k=1,i,q=0;
    	cin>>n;
    	string s;
    	cin>>s;
    	for(i=0;i<n;++i)
    	{	if(s[i]<=s[i+1])
    			++k;
    		else
    		{	q+=(k*k+k)/2;
    			k=1;
    		}
    	}
    	cout<<q<<"\n";
    }
    return 0;
}
