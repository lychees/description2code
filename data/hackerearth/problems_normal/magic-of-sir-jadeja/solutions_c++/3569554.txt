#include <iostream>
#include <stdio.h>
#include <cmath>
#include <assert.h>
using namespace std;
long long expo(long long a, long long b){
  long long result = 1;
  while (b){
    if (b&1){
      result *= a;
    }
    b >>=1 ;
    a *= a;
  }
  return result;
}
int main()
{
	string ans[8] = {"", "Rohit", "Dhawan", "Kohli", "Yuvraj", "Raina", "Dhoni", "Sir Jadeja"};
	int t;
	scanf("%d", &t);
	while(t--) {
		long long k;
		scanf("%lld", &k);
		assert(1 <= k && k <= 1e9);
		int i = (int)log2(ceil(k*1.0/7.0));
		int index = ceil((k-(expo(2, i) -1)*7)/(expo(2,i)*1.0));
		cout << ans[index] << endl;
	}
}
