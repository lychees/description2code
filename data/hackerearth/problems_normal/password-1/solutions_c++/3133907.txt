#include<stdio.h>
#include<algorithm>
#include<string.h>
using namespace std;
int main()
{
	int t,i,p,k,j,f=0;
	scanf("%d",&t);
	char a[t][14];
	for(i=0;i<t;i++)
	scanf("%s",a[i]);
	for(i=0;i<t-1;i++)
	{
		for(j=i+1;j<t;j++)
		{
			f=0;
			p=strlen(a[i]);
			if(p!=strlen(a[j]))
			continue;
			for(k=0;k<p/2;k++)
			{
				if(a[i][k]!=a[j][p-k-1])
				{
					f=1;
					break;
				}	
			}
			if(f==0)
			{
				printf("%d %c",p,a[i][p/2]);
				return 0;
			}
		}
	}
}
