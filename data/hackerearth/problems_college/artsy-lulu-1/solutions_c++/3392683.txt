#include<iostream>
#include<cctype>
#define c cin
#define o cout
using namespace std;

int main(){
	char in;
	int t;
	while(c>>in>>t){
		if(t<2||t>26||!isalpha(in)){
			o<<"INVALID\n";
		}
		else{
			char begin;
			if(isupper(in))
				begin='A';
			else
				begin='a';
			
			int check=((int)(in)-begin);
			if(check+t>26){
				o<<"INVALID\n\n";
				continue;
			}
			
			for(int i=0; i<t; i++){
				for(int j=0; j<=i; j++){
					o<<in<<" ";
				}
				in=((char)((int)(in))+1);
				o<<endl;
			}
		}
		o<<endl;
	}
	
	return 0;
}
