t=input()
while t:
	t-=1
	s=raw_input().split()
	l1=len(s[0])
	l2=len(s[1])
	s1=s[0][::-1]
	s2=s[1][::-1]
	i=0
	c=0
	r=0
	while (i<l1 and i<l2):
		if (int(s1[i])+int(s2[i])+c)>9:
			c=1
			r+=1
		i+=1
	
	while (i<l1):
		if (int(s1[i])+c)>9:
			c=1
			r+=1
		i+=1
	
	while (i<l2):
		if (int(s2[i])+c)>9:
			c=1
			r+=1
		i+=1
	##print s1,s2
	if r==0:
		print "No carry operation"
	else:
		if r>1:
			print r, "carry operations"
		else:
			print r, "carry operation"
