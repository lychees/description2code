#include<iostream>
#include<algorithm>
using namespace std;
int main(){
int t;
cin>>t;
while(t != 0){
    long long g;
    int n;
    cin>>g>>n;
    int k = 0;
    while(k != n){
       if(k == n-1){
        cout<<g*3 <<" ";
       }
       else
        cout<<g*2<<" ";
       k++;
    }
        cout<<endl;
t--;
}
}
