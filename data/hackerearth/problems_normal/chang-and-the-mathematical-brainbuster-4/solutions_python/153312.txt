def solve(numList,size):
	window = size - 1
	while window >= 0:
		start = 0
		end = window	
		while end < size:			
			if numList[end]- 3*numList[start] <= 0:				
				return size - window -1
			else:
				start = start + 1
				end = end + 1
		window = window - 1	
	return size-window

size = int(raw_input())
num = str(raw_input()).split(' ')
numList = [int(i) for i in num]
numList.sort()
print solve(numList, size)
