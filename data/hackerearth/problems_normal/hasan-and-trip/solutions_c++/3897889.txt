#include <iostream>
#include <bits/stdc++.h>
using namespace std;
struct node
{
	long double x,y,f;	
};
int main()
{
    int n;
    cin>>n;
    node arr[n];
    for(int i=0;i<n;i++)
    {
    	cin>>arr[i].x>>arr[i].y>>arr[i].f;
    }
    
    long double dp[n];
    dp[0]=arr[0].f;
    
    for(int i=1;i<n;i++)
    {
    	for(int j=i-1;j>=0;j--)
    	{
    		if(j==i-1)
    		dp[i]=dp[j]+arr[i].f-sqrt(pow(arr[i].x-arr[j].x,2)+pow(arr[i].y-arr[j].y,2));
    		else
    		dp[i]=max(dp[i],dp[j]+arr[i].f-sqrt(pow(arr[i].x-arr[j].x,2)+pow(arr[i].y-arr[j].y,2)));
    	}
    	//cout<<dp[i]<<endl;
    }
    cout<<fixed<<setprecision(6);
    cout<<dp[n-1]<<endl;
    return 0;
}
