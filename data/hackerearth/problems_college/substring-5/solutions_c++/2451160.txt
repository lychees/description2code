#include<bits/stdc++.h>
using namespace std;

int main()
{
    string str;
    cin>>str;
    int sum = 0;
    int n = 1;
    int len = str.size();
    for(int i=0;i<len-1;i++)
    {
    	if(str[i] == str[i+1])
    	{
    		n++;
    	}
    	else
    	{
    		sum += n*(n+1)/2;
    		n = 1;
    	}
    }
    sum += n*(n+1)/2;
    cout<<sum<<endl;
    
}
