T = int(raw_input())
for t in range(T):
    s = raw_input()
    do = False
    cost = 0
    for ch in s:
        if do:
            if ch <= '9' and ch >= '0':
                cost = cost * 10 + int(ch)
            elif ch != ' ':
                do = False
                break
        if ch == '$':
            do = True
    print('$' + str(cost))
