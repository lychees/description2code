#include<bits/stdc++.h>
using namespace std;
long long int dp[250][250];
long long int solve(int present, int k,int prev )
{

 if(present==0)
 return 1;
 if(present<k)
 return 0;
 if(dp[present][prev]!=-1)
 return dp[present][prev];
 else
 {
  dp[present][prev]=0;
  for(int i=k;i<=present;i++)
  {
    if(i>=prev)
    dp[present][prev]+=solve(present-i,k,i);
  }

  return dp[present][prev];
 }
 
}
int main()
{
int t;
cin>>t;
while(t--)
{
 int n;
 cin>>n;
 int k;
 cin>>k;
 for(int i=0;i<210;i++)
 {
   for(int j=0;j<210;j++)
   dp[i][j]=-1;
 }
 long long int ans=solve(n,k,0);
 cout<<ans<<endl;
}
return 0;
}
