typedef long long ll;
std::string s;

void print() {
  int i = 0;
  int n = s.size();

  while (true) {
    if (i==n) break;

    if (i==n-1) {
      std::cout << s[i];
      break;
    }

    std::cout << s[i] << '-' ;
    i++;
  }
  std::cout << '\n';
}


int main() {
  std::cin >> s;
  
  print();
  std::reverse(std::begin(s), std::end(s));
  print();
}
