#include <iostream>
using namespace std;
#define maxm 1000
#define maxk 20
int s[maxm][maxk];
int arr[maxk];

int main()
{
    int n,k;
    cin>>n>>k;
    for(int i=0;i<n;i++){
        for(int j=0;j<k;j++){
            cin>>s[i][j];
        }
    }
    int t;
    cin>>t;
    while(t--){
        int sub;
        cin>>sub;
        for(int i=0;i<sub;i++)
            cin>>arr[i];
        bool mili=false;
        int arrneed[n];
        int max=n;
        for(int i=0;i<max;i++){
            arrneed[i]=i;
        }
        for(int j=0;j<sub;j++){
            int mini=s[arrneed[0]][arr[j]-1];
            for(int i=1;i<max;i++){
                if(mini<s[arrneed[i]][arr[j]-1]){
                    mili=true;
                }
                else if(mini>s[arrneed[i]][arr[j]-1]){
                    mini=s[arrneed[i]][arr[j]-1];
                    mili=true;
                }
            }
            if(mili){
                int temp=0;
                for(int i=0;i<max;i++){
                    if(s[arrneed[i]][arr[j]-1]==mini){
                        arrneed[temp]=arrneed[i];
                        temp++;
                    }
                }
                max=temp;
                if(max==1)
                    break;
            }
        }
        cout<<arrneed[0]+1<<endl;
    }
}
