#include <iostream>
using namespace std;

int main()
{
     int t;
    scanf("%d",&t);
    while(t--){
    	long long int n,k,i,c=0;
    	scanf("%lld %lld",&n,&k);
    	long long int arr[n];
    	for(i=0;i<n;i++){
    		scanf("%lld",&arr[i]);
    		if(arr[i]<=0)
    			c++;
    	}
    //	printf("%lld ",c);
    	if(c<k)
    		printf("YES\n");
    	else 
    		printf("NO\n");
    }
    return 0;
}
