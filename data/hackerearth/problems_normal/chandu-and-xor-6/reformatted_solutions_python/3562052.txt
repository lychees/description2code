n, m = map(int, raw_input().strip().split())
rows_sum, cols_sum = list(), [0] * m
# print cols_sum

matrix = list()

for i in range(n):
    row = map(int, raw_input().strip().split())
    rows_sum.append(sum(row))
    matrix.append(row)
    for index, col in enumerate(row):
        cols_sum[index] += col

result = 0
for i in rows_sum:
    for j in cols_sum:
        xor = i ^ j
        if xor > result:
            result = xor
print result
