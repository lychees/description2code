#include <bits/stdc++.h>
#define gc getchar_unlocked


using namespace std;

void readInt (int &x) {
	register unsigned int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}

main()
{
	register int n,r,number=0,i,x,a=0,b=0;
	readInt(n);
	readInt(r);
	register int arr[n],sum=0;
	for(i=0;i<n;i++)
	{
		readInt(x);
		{
			arr[i]=x+1;
		}

		sum=sum+arr[i];
		//cout<<"length "<<sum<<endl;
		//cout<<"number "<<number<<endl;
	}
	//cout<<"sum "<<sum<<endl;

	sort(arr,arr+n);

	if(sum%n==0)
	{
		i=n-1;
		while(arr[i]>(sum/n))
		{
			//cout<<"number "<<arr[i]<<endl;
			a++;
			i--;

		}
		//cout<<"a "<<a<<endl;
		i=0;
		while(arr[i]==0)
		i++;
		while(arr[i]<(sum/n))
		{
			b++;
			i++;
		}
		//cout<<"b "<<b<<endl;
		if(a>=b)
		printf("%d\n",a*r);
		else
		printf("%d\n",b*r);
	}
	else
	printf("-1");
}
