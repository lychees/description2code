#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	int n,k,i,j;
    cin >> n >> k;
    int a[1000000],freq[1000000];
    for(i=0;i<n;i++)
    {
    	cin >>a[i];
    	if(a[i]<k)
    	{
    		freq[a[i]]++;
    	}
    }
    j=0;
    for(i=0;i<n;i++)
    {
    	if(freq[i]>freq[j])
    	{
    		j=i;
    	}
    }
    cout<<j<<endl;
    return 0;
}
