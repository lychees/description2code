import math

n = input()
for i in xrange(n):
    x = input()
    divc = 0
    rt = int(math.sqrt(x))
    if (rt*rt) == x:
        divc -= 1
    for j in xrange(rt):
        if x % (j+1) == 0:
            divc += 2
            if divc >= 4:
                print "YES"
                break
    if divc < 4:
        print "NO"
