#include<bits/stdc++.h>


using namespace std;

int T;
long long int X,Y,K,hcf;
vector<long long int> fac;

int main(){

    cin>>T;
    while(T--){
    cin>>X>>Y>>K;

    hcf=__gcd(X,Y);
	
    for(long long int i=1;i*i<=hcf;i++){

        if(hcf%i==0){
            fac.push_back(i);

        if(i!=hcf/i)
            fac.push_back(hcf/i);
        }
    }

        sort(fac.begin(),fac.end());

        if(K>fac.size())
            cout<<"No crymeth today"<<endl;
        else
            cout<<fac[fac.size()-K]<<endl;
        fac.clear();
    }

    return 0;
}
