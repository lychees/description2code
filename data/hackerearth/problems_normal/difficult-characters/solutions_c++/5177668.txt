#include<bits/stdc++.h>
using namespace std;
pair<int,char>p[26]={};
bool way(const pair<int, char>&i, const pair<int, char>&j)
{
    if(i.first==j.first)
        return i.second>j.second;
    else
        return i.first<j.first;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        string s;
        cin>>s;
        for(int i=0;i<26;i++)
            {
                p[i].first=0;
                p[i].second=i+'a';
            }
        for(int i=0;i<s.size();i++)
            {
                p[s[i]-'a'].first++;
                p[s[i]-'a'].second=s[i];
            }
        sort(p,p+26,way);
        for(int i=0;i<26;i++)
            cout<<p[i].second<<" ";
        cout<<endl;
    }
}
