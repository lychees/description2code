
T = input()
op = "carry operation"
for i in range(T):
	a,b = raw_input().split(' ')
	a=a[::-1]
	b = b[::-1]
	carry = 0
	count=0
	for i in range(max(len(a),len(b))):
		if int(a[i])+int(b[i])+carry>9:
			count+=1
			carry=1
		else:
			carry=0
	if count == 0:
		print "No "+op
	elif count == 1:
		print str(count)+" "+op
	else:
		print str(count)+" "+op+"s"
