int BIT[MX][30]={0},n;

void update(int x,int y,int val)
{
	for(;x<=n;x+=(x&-x))

}

int query(int x,int y)
{
	int s=0;
	
	for(;x>0;x-=(x&-x))
	s+=BIT[x][y];
	
	return s;
	
}
