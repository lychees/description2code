#include<cstdio>
#include<iostream>
using namespace std;
int main()
{
	int tc;
	cin>>tc;
	while(tc--)
	{
		int rows,columns;
		cin>>rows>>columns;
		int x1,y1;
		int x2,y2;
		cin>>x1>>y1;
		cin>>x2>>y2;
		int p;
		cin>>p;
		int forward = x1>x2?x1-x2:x2-x1;
		int backward = x1>x2?rows-x1+x2:rows-x2+x1;
		int horizontal = forward>backward?backward:forward;
		int v_forward = y1>y2?y1-y2:y2-y1;
		int v_backward = y1>y2?columns -y1+y2:columns-y2+y1;
		int vertical = v_forward>v_backward?v_backward:v_forward;
		int total = horizontal+vertical;
		//cout<<total<<"\n";
		int cost = total*p;
		cout<<cost<<'\n';
	       if(cost <= 1000)
		       cout<<"You saved the group."<<'\n';
	       else
		       cout<<"Let go of the group."<<'\n';
	}
}
