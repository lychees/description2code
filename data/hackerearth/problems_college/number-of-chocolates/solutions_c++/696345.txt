			{

				else
					printf("%lld\n",b[g2] + (b[p2]-b[p1-1]));
			}
		}
		else if(g1 == p1)
		{
			if(p1 < g2)
			{
				if(g2 < p2)
				{
					if(p1 > 0)
						printf("%lld\n",b[p2]-b[p1-1]);

						printf("%lld\n",b[g2]-b[g1-1]);

				}
			}
			else if(p1 == g2)
			{
				if(g2 == p2)
				{
					if(g2 > 0)
						printf("%lld\n",b[g2]-b[g2-1]);
