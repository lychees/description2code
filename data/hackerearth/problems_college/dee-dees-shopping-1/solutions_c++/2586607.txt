#include <bits/stdc++.h>

using namespace std;

#define ll long long int
#define s(x) scanf("%d", &x)
#define sl(x) scanf("%lld", &x)
#define pb push_back
#define mp make_pair
#define mod 1000000007

int main()
{
    int t, i, j, n, mn;
    cin >> t;
    while (t--) {
        cin >> n;
        mn = INT_MAX;
        for (i = 0; i*6 <= n; i++) {
            for (j = 0; i*6+j*8 <= n; j++) {
                if (i*6+j*8 == n)
                    mn = min(mn, i+j);
            }
        }
        if (mn == INT_MAX)
            mn = -1;
        cout << mn << endl;
    }
    return 0;
}
