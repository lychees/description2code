#include<iostream>
#include<algorithm>
using namespace std;
int main(void)
{
	long long int t;
	cin>>t;
	while(t--)
	{
		long long int n,p=0,d=0,i;
		cin>>n;
		int m[n],g[n];
		for(i=0;i<n;i++)
		{
			cin>>m[i];
		}
		for(i=0;i<n;i++)
		{
			cin>>g[i];
		}
		sort(m,m+n);
		sort(g,g+n);
		for(i=0;i<n;i++)
		{
			p+=m[i]*g[i];
			
		}
		cout<<p<<endl;
	}
}
