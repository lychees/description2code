#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,k,i;
		cin>>n>>k;
		int number,setbit[n];
		for(i=0;i<n;i++)
		{
			cin>>number;
			int count=0;
			while(number)
			{
				number=number&(number-1);
				count++;
			}
			setbit[i]=count;
		}
		sort(setbit,setbit+n);
		int ans=0;
		while(k--)
		{
			ans=ans+setbit[n-1];
			n--;
		}
		cout<<ans<<"\n";
	}
    return 0;
}
