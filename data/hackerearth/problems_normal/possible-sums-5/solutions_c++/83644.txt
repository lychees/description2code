#include <iostream>
#include <algorithm>
#include <vector>
#include <set>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		int n;
		cin >> n;
		int ar[n];
		for(int i=0; i<n; i++)
			cin >> ar[i];
		int count = 1;
		bool ispos[100*100+1] = {1};
		int sum = 0;
		for(int i=0; i<n; i++)
		{
			vector<int> temp;
			for(int j=0; j<=sum; j++)
			{
				if(ispos[j] and !ispos[j+ar[i]])
				{
					temp.push_back(j+ar[i]);
					count++;
				}
			}
			for(int j=0; j<temp.size(); j++)
				ispos[temp[j]] = true;
			sum += ar[i];
		}
		cout << count << endl;		
	}
}
