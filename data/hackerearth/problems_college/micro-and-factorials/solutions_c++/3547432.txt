ll power(ll a,ll b)
{
	if(b==0)
		return 1;
	if(b==1)
		return a;
	ll temp=power(a,b/2);
	temp=(temp*temp)%mod;
	if(b&1)
		temp=(temp*a)%mod;
	return temp;
}
int main() {
	ll t,x,y,i,c1,c2,ans,j,val,c;
	vector<bool> v(100001);
	vector<int> p;
	for(i=2;i<=100000;i++)
	{
		if(!v[i])
		{
			p.pb(i);
			for(j=i*i;j<=100000;j=j+i)
				v[j]=true;
		}
	}
	vector<int>::iterator it;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld%lld",&x,&y);
		if(x>y)
		{
			c1=c2=1;
			for(it=p.begin();(it!=p.end())&&((*it)<=x);it++)
			{
				val=*it;
				c=0;
				while(x/val)
				{
					c+=(x/val);
					val*=(*it);
				}
				c1=(c1*(c+1))%mod;
				c=0;
				val=*it;
				while(y/val)
				{
					c+=(y/val);
					val*=(*it);
				}
				c2=(c2*(c+1))%mod;
			}
			ans=(c1-c2)%mod;
			ans=(ans+mod)%mod;
		}
		else
			ans=0;
		printf("%lld\n",ans);
	}
	return 0;
}
