#include<iostream>
using namespace std;
#define LL long long
#define SIZE 10000001
LL A[SIZE],sum[SIZE];

LL S(LL L,LL R)
{
    if(L==1) return sum[R];
    return(sum[R] - sum[L-1]);
}

LL D(LL L,LL R)
{
    return(A[R]-S(L,R-1));
}

void solve()
{
    LL n,q;
    cin>>n>>q;
    A[0]=0;
    sum[0]=0;
    for(LL i=1;i<=n;i++)
    {
        cin>>A[i];
    }
    sum[1]=A[1];
    for(int i=2;i<=n;i++)
    {
        sum[i]=sum[i-1]+A[i];
    }
    LL l,r;
    while(q--)
    {
        cin>>l>>r;
        cout<<S(l,r)<<" "<<D(l,r)<<endl;
    }
}

int main()
{
    solve();
    return 0;
}
