#include <iostream>
using namespace std;
int prime[10000];
bool p[10000]={false};
int k=1;
void pre()
{
	for(int i=2;i<=9999;i++)
	{
		if(p[i]==false)
		{
			prime[k]=i;
			k++;
			for(int j=i*i;j<=9999;j+=i)
			{
				p[j]=true;
			}
		}
	}
}
int main()
{
    pre();
    int n;
    cin>>n;
    cout<<prime[n]<<"\n";
    return 0;
}
