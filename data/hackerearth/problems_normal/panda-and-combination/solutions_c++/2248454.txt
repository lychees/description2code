int main(void){
	std::ios::sync_with_stdio(false);cin.tie(0);
	int T;
	unsigned long long re;
	unsigned long long n,m,t;
	cin>>T;
	while(T--)
	{
		cin>>n>>m;
		if(m==1)
		{
			re=n;
		}
		else if(m<=1e6)
		{
			t=m;
			while(t<=n)
			{
				t*=m;
			}
			t/=m;
			re=0;
			while(n)
			{
				re+=n/t;
				n%=t;
				t/=m;
			}
		}
		else if(m<=1e9)
		{
			t=m*m;
			re=n/t;
			n%=t;
			re+=n/m;
			re+=n%m;
		}
		else
		{
			re=n/m;
			re+=n%m;
		}
		cout<<re<<endl;
	}
//	system("pause");
	return 0;
}
