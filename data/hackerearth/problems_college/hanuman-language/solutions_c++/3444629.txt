#include <bits/stdc++.h>
using namespace std;
string str[11111];
int conv(string s)
{
	int ret=0,i;
	for(i=0;i<s.size();i++)
	ret=(ret*10)+s[i]-'0';
	return ret;
}
int main()
{
	cin.sync_with_stdio(false);
	string inp;
	int t,ind,i;
	getline(cin,inp);
	t=conv(inp);
	while(t--)
	{
		ind=0;
		getline(cin,inp);
		stringstream in(inp);
		while(in>>inp)
		str[ind++]=inp;
		for(i=ind-1;i>=0;i--)
		cout<<str[i]<<" ";
		cout<<'\n';
	}
    return 0;
}
