#include<iostream>
#define MOD 1000000007
typedef long long ll;
using namespace std;
void matMul(ll a[2][2], ll b[2][2]){
	int i,j,k;
	ll c[2][2]={{0,0},{0,0}};
	for(i=0;i<2;++i){
		for(j=0;j<2;++j){
			for(k=0;k<2;++k){
				c[i][j]+=(a[i][k]*b[k][j]);
				if(c[i][j]>=MOD)
					c[i][j]%=MOD;
			}
		}
	}
	a[0][0]=c[0][0];
	a[0][1]=c[0][1];
	a[1][0]=c[1][0];
	a[1][1]=c[1][1];
}
void pow(ll tr[2][2],ll exp){
	ll res[2][2]={{1,0},{0,1}};
	while(exp>0){
		if(exp&1)
			matMul(res,tr);
		matMul(tr,tr);
		exp>>=1;
	}
	tr[0][0]=res[0][0];
	tr[0][1]=res[0][1];
	tr[1][0]=res[1][0];
	tr[1][1]=res[1][1];
	return;
}
int main(){
	cout.sync_with_stdio(0);
	cin.tie(0);
	int t,n;
	ll ans;
	ll tr[2][2];
	cin>>t;
	while(t--){
		cin>>n;
		if(n<3)
			cout<<"1\n";
		else{
			tr[0][0]=1;
			tr[0][1]=1;
			tr[1][0]=1;
			tr[1][1]=0;
			pow(tr,n-2);
			ans=tr[0][0]+tr[0][1];
			if(ans>=MOD)
				ans%=MOD;
			cout<<ans<<"\n";
		}
	}
	return 0;
}
