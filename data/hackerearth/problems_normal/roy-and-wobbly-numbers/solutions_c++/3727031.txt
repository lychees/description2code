#include <bits/stdc++.h>
using namespace std;
int n,k;
void solve(){
    cin>>n>>k;
    if(k>81){
        cout<<-1<<endl;
        return;
    }
    int a = (k-1)/9 + 1;
    int b = (k-1)%9;
    if(b >= a){
        b ++;
    }
    for(int i=0;i<n;i++){
        if(i%2==0)cout<<a;
        else cout<<b;
    }
    cout<<endl;
}
int main(){
    int t;cin>>t;while(t--)solve();
}
