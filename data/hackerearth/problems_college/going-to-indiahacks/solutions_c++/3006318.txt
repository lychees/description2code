#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cstring>
 
typedef long long ll;
 
using namespace std;
 
ll dp[1005][1005];
 
int main() {
 
	int t; cin >> t;
 
	int a, b, c ,d, i, j;
 
	const long long mod = 1000000007;
	
	for(i = 0; i <= 1004; i++)
		for(j = 0; j <= 1004; j++)
			if(i == 0 || j == 0) dp[i][j] = 0;
			else dp[i][j] = 1;
	
	for(i = 1; i <= 1004; i++)
		for(j = 2; j <= 1004; j++)
			dp[i][j] = (dp[i-1][j]+dp[i-1][j-1]+dp[i][j-1])%mod;
 
 
	while(t--) {
		
		scanf("%d %d %d %d", &a, &b, &c, &d);
		
		cout << dp[c-a+1][d-b+1] << endl;
	}
 
	return 0;
}
