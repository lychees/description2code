#include<bits/stdc++.h>

using namespace std;

int dp[100][100];

long long int func(string str , int i , int j)
{
	if(i>j)
		return 0;
	if(i==j)
		return 1;
	if(dp[i][j]!=-1)
		return dp[i][j];
	long long int l;
	l = func(str,i,j-1)+func(str,i+1,j)-func(str,i+1,j-1);
	if(str[i]==str[j])
		l += 1+func(str,i+1,j-1);
	return dp[i][j]=l;
}

int main()
{
	int T;
	string str;
	long long int i , j , n;
	cin>>T;
	while(T--){
		cin>>str;
		n = str.length();
		for(i=0;i<n;i++)
			for(j=0;j<n;j++)
				dp[i][j] = -1;
		cout<<func(str,0,n-1)<<endl;
	}
	return 0;
}
