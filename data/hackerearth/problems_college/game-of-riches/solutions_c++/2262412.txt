#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int T,nR,nG,wealth[100];
    cin >> T;
    while(T--){
    	int count=0;
    	cin >> nR >> nG;
    	for(int i=1;i<=nR;++i){
    		cin >> wealth[i];
    	}
    	int w = wealth[1];
    	sort(wealth+1,wealth+nR+1, greater<int>());
    	int ide = find(wealth+1,wealth+nR+1,w)-wealth;
    	int id = ide%nG;
    	if(id == 0){
    		id = nG;
    	}
    	count = (ide-id)/nG;
    	cout << count<< endl;
    }
    return 0;
}
