#include <bits/stdc++.h>
#define lli unsigned long long int
using namespace std;

int main()
{
    lli size,q,type,L,R,i,sum;
    cin >> size >> q;
    lli A[size+1],B[size+1];
    for(i=1;i<=size;i++) cin >> A[i];
    for(i=1;i<=size;i++) cin >> B[i];

    for(i=size-2;i>=1;i--)
    {
        A[i]+=A[i+2];
        B[i]+=B[i+2];
    }
    while(q--)
    {
        cin >> type >> L >> R;
        if(type == 1)
        {
            if(L==R)
                cout << A[L] << endl;
            else
            {
                sum = A[L];
                sum+=B[L+1];
                if((L%2 !=0 && R%2==0) || (L%2 ==0 && R%2!=0))
                {
                    if(R+1<=size)
                    {
                        sum-=A[R+1];
                    }
                    if(R+2<=size)
                    {
                        sum-=B[R+2];
                    }
                }
                else
                {
                    if(R+2<=size)
                    {
                        sum-=A[R+2];
                    }
                    if(R+1<=size)
                    {
                        sum-=B[R+1];
                    }
                }
                cout << sum << endl;
            }
        }
        else
        {
            if(L==R)
                cout << B[L] << endl;
            else
            {
                sum = B[L];
                sum+=A[L+1];
                if((L%2 !=0 && R%2==0) || (L%2 ==0 && R%2!=0))
                {
                    if(R+1<=size)
                    {
                        sum-=B[R+1];
                    }
                    if(R+2<=size)
                    {
                        sum-=A[R+2];
                    }
                }
                else
                {
                    if(R+2<=size)
                    {
                        sum-=B[R+2];
                    }
                    if(R+1<=size)
                    {
                        sum-=A[R+1];
                    }
                }
                cout << sum << endl;
            }
        }
    }
    return 0;
}
