#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    ll n, a, d1 = 0, d2 = 0, i, j;
    cin >> n;
    for(i = 1; i <= n; i++)
    {
    	for(j = 1; j <= n; j++)
    	{
    		cin >> a;
    		if(i == j)
    			d1 += a;
    		if(i + j == n+1)
    			d2 += a;
    	}
    }
    cout << abs(d1-d2) << '\n';
    return 0;
}
