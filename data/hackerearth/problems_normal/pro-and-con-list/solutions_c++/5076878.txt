#include <bits/stdc++.h>

using namespace std;

const int N = 1e5 + 10;

pair <int, int> a[N];

int main() {
	int t;
	scanf("%d", &t);
	for (int tt = 1; tt <= t; tt++) {
		priority_queue < pair <long long, int> > pq;
		int n;
		scanf("%d", &n);
		long long res = 0;
		for (int i = 0; i < n; i++) {
			int x, y;
			scanf("%d %d", &x, &y);
			a[i] = make_pair(x, y);
			pq.push(make_pair((long long) x + y, i));
			res -= y;
		}
		int pos1 = pq.top().second;
		pq.pop();
		int pos2 = pq.top().second;
		res += a[pos1].first + a[pos1].second + a[pos2].first + a[pos2].second;
		printf("%lld\n", res);
	}
	return 0;
}
