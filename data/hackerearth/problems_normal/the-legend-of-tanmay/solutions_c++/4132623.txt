#include <bits/stdc++.h>
using namespace std;

int main()
{
    int T,N;
    //freopen("file1.txt","r",stdin);
    cin>>T;
    while(T--)
    {
    	cin>>N;
    	int zero = 0,e;
    	long long arrp[N],p=0;
    	long long pdtmax = 1;
    	long long arrn[N],n=0;
    	long long pdtmin = 1;

    	for(int i=0;i<N;i++)
    	{
    		cin>>e;
    		if(e==0)
    			zero++;
    		else if(e<0)
    			arrn[n++] = e;
    		else arrp[p++] = e;
    	}

    	sort(arrn,arrn+n);
    	sort(arrp,arrp+p);

    	for(int i=0;i<n;i++)
    	{
    		if(i<n-1)
    		{
    			pdtmax*=arrn[i];
    			pdtmin*=arrn[i];
    		}
    		if(i==n-1 && n%2==0)
    			pdtmax*=arrn[i];
    		else if(i==n-1 && n%2!=0)
    			pdtmin*=arrn[i];
    	}

    	for(int i=0;i<p;i++)
    	{

    		pdtmax*=arrp[i];
    		pdtmin*=arrp[i];
    	}

    	if(n==0 && zero>0)
    		pdtmin = 0;
    	if(p==0 && zero==0 && n==1)
    		pdtmax = arrn[n-1];
    	if(p==0 && zero>0)
    		pdtmax = 0;
    	if(n==0 && zero==0)
    		pdtmin = arrp[0];
    	cout<<pdtmax<<" "<<pdtmin<<endl;
    }
}
