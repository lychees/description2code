#include <iostream>
using namespace std;

int main()
{
   int n, i;
   cin>>n;
   unsigned long long a[n];
   unsigned long long ans = 1;
   const unsigned int M = 1000000007;
   
   for(i=0;i<n;i++)
   {
   	cin>>a[i];
   }
   
   for(i=0;i<n;i++)
   {
   	ans = (ans * a[i]) % M;
   }
   
   cout<<ans;
}
