#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

int main()
{
	int n, k;
	int evenSum, oddSum;
	vector<int> evenNums, oddNums;

	evenSum = oddSum = 0;
	cin >> n;
	for (int i = 0; i < n; ++i)
	{
		cin >> k;
		if (k % 2 == 0)
		{
			evenSum += k;
			evenNums.push_back(k);
		}
		else
		{
			oddSum += k;
			oddNums.push_back(k);
		}
	}
	sort(evenNums.begin(), evenNums.end());
	sort(oddNums.begin(), oddNums.end());
	for (int i = 0; i < int(evenNums.size()); ++i)
		cout << evenNums[i] << " ";
	cout << evenSum << " ";
	for (int i = 0; i < int(oddNums.size()); ++i)
		cout << oddNums[i] << " ";
	cout << oddSum << " ";

	return 0;
}
