#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll Arr[100005];
void initialize(ll N)
{
    for(ll i = 1;i<=N;i++)
    Arr[ i ] = i ;
}
int root(ll i)
{
    while(Arr[ i ] != i)          
    {
     i = Arr[ i ];
    }
    return i;
}

int un(ll A ,ll B)
{
    ll root_A = root(A);       
    ll root_B = root(B);  
    Arr[ root_A ] = root_B ;   
}
int main()
{
	ll j,n,p,x,y,count=0;
	
	set<ll>s;
    cin>>n;
    cin>>p;
    initialize(n);
    while(p--)
    {
    	cin>>x>>y;
    	un(x,y);
    		
	}
	for(j=1;j<=n;j++)
	{
		if(Arr[j]==j)
		count++;
	}
	cout<<count<<endl;
	return 0;
}
