#include <iostream>
#include <math.h>
#include <stdio.h>
using namespace std;

int main()
{
    int t,n,k,i;
    long long m,remain=0,actual=0;
    cin>>t;
    for(i=0;i<t;i++){
    	cin>>n>>k>>m;
    	actual=((n*(n+1))/2);
    //	cout<<actual<<endl;
    	actual=k*actual*actual;
    //	cout<<actual<<endl;
    	remain=actual-m;
    	if(remain>0)
    		cout<<remain<<endl;
    	else
    		 cout<<0<<endl;
    	actual-0;
    	remain=0;
    }
    
    return 0;
}
