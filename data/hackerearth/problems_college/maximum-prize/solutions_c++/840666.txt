#include <iostream>
#include <algorithm>
using namespace std;
#include <stdio.h>

int main()
{
    int test,N,k,i;
    long long int a[100000];
    unsigned long long int sum;
    
    scanf("%d",&test);
    while(test--)
       {
       	sum=0;
       	scanf("%d %d",&N,&k);
       	for(i=0;i<N;i++)
       	  scanf("%lld",&a[i]);

       	if(k>=N)
       	{
       		for(i=0;i<N;i++)
       		sum=sum+a[i];
       	}
       	else
       	{
       		sort(a,a+N);
       	for(i=N-1;i>=N-k;i--)
       	  sum=sum+a[i];
       	}
       	
       	printf("%llu\n",sum);  
       }
    return 0;
}
