#include<bits/stdc++.h>
#define plus 1000000
using namespace std;

int arr[2000005];

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        memset(arr, 0, sizeof(arr));
        int n, temp;
        long int ans = 0;
        cin>>n;
        for(int i=0; i<n; i++)
        {
            scanf("%d",&temp);
            arr[temp + plus]++;
        }

        for(int i=0; i<2000005; i++)
            if(arr[i]>1)
                ans += ((arr[i]*(arr[i]-1))/2);
        ans += n;
        cout<<ans<<endl;
    }
    return 0;
}
