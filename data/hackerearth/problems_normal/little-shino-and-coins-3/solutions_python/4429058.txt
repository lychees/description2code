k = int(raw_input())
s = raw_input()

ans = 0
for i in range(len(s)):
	found = [False for a in range(26)]
	distinct = 0	
	for j in range(i, len(s)):
		if found[ord(s[j]) - 97] :
			if distinct == k:
				ans += 1
			continue
		found[ord(s[j]) - 97] = True
		distinct += 1
		if distinct == k:
			ans += 1
		if distinct > k:
			break
		
print ans
