#include <iostream>
using namespace std;

int main()
{
	int n;
	int count=0;
	int flag=0;
	int set=0;
	string s;
	cin>>n;
	while(n--){
		flag=0;
		count=0;
		set=0;
	    cin>>s;
	    for(char &c:s){
	    	if(c=='x'){
	    		cout<<'x';
	    		flag=1;
	    	}
	    	
	    	if(c=='*'){
	    		for(int i=0;i<count;i++){
	    			if(flag==0){	    		
	    				cout<<'o';
	    			}
	    		}
	    		count=0;
	    		cout<<'*';
	    		flag=0;
	    		set=1;
	    	}
	    	if(c=='o'){
	    		set=0;
	    		count++;
	    	}
	    }
	    if(set!=1){
	    	for(int i=0;i<count;i++){
	    			if(flag==0){	    		
	    				cout<<'o';
	    			}
	    	}
	    }
	    cout<<"\n";
	}
    return 0;
}
