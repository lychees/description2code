#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
	int T,N,i,j,s;
	cin>>T;
	for(i=0;i<T;i++)
	{
		cin>>N;
    int M[N],F[N];
    for(j=0;j<N;j++)
    	cin>>M[j];
    for(j=0;j<N;j++)
    	cin>>F[j];
    	 
    	 sort(M,M+N);
         sort(F,F+N);
    s=0;
    for(j=0;j<N;j++)
       s=s+(M[j]*F[j]);
	cout<<s<<endl;
	}
    return 0;
}
