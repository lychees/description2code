#include <iostream>
#include <string.h>
using namespace std;
using std::string;
int main()
{
    int n,a[1000],i,j,k;
    char s[1000][50],temp[1000][100];
    cin >>n;
    cin>>s[0];
    for(i=1;i<n;i++)
    {
    	cin>>a[i-1];
    	cin>>s[i];
    }
    strcpy(temp[0],s[0]);
    //cout<<temp[0]<<endl;
    for(i=1;i<n;i++)
    {
    	k=0;
    	for(j=0;j<a[i-1];j++)
    	{
    		temp[i][k]=temp[i-1][j];k++;
    	//	cout << temp[i][k-1]<<" ";
    	}
    	for(j=0;s[i][j]!='\0';j++)
    	{
    		temp[i][k]=s[i][j];k++;
    	}
    	temp[i][k]='\0';
    	//cout << temp[i]<<endl;
    }
    cout<<temp[n-1]<<endl;
    return 0;
}
