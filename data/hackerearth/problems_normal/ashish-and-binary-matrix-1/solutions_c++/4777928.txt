#include <iostream>
#include<map>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t,n,m,i,j,v;
   cin>>t;
   string s;
   for(v=0;v<t;v++)
   {
   	cin>>n>>m;
   	map<string,int>m1;
   	int f=0;
   	for(i=0;i<n;i++)
   	{
   		cin>>s;
   		m1[s]++;
   	}
   	map<string,int>::iterator ii;
   	for(ii=m1.begin();ii!=m1.end();ii++)
   {
	   	if(ii->second>1)
	   	{
	   	f=1;
	   	break;
	   	}
   }
   	
   	if(f==0)
   	cout<<"Yes\n";
   	else
   	cout<<"No\n";
   }
    return 0;
}
