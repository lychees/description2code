#include <iostream>
using namespace std;
typedef unsigned long long ull;
ull modulo(ull a,ull b,ull c){
    ull x=1,y=a; 
    while(b > 0){
        if(b%2 == 1){
            x=(x*y)%c;
        }
        y = (y*y)%c; // squaring the base
        b /= 2;
    }
    return x%c;
}
int main()
{
   int t = 0;
   cin>>t;
   while(t--)
   {
   	ull x = 0;
   	cin>>x;
   	cout<<(modulo(2,x,1000000007) - 1)%1000000007<<endl;
   }
    return 0;
}
