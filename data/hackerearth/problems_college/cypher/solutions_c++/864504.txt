#include <iostream>
using namespace std;
#include<cstring>
#include<cstdio>
int main()
{
    int t;
    cin>>t;
    char a[103];
    while(t--)
    {
    	scanf("%s",a);
    	int l=strlen(a);
    	for(int i=0;i<l;i++)
    	{
    		int k=a[i];
    		k=k-97;
    		k=(k+i)%26;
    		k=k+97;
    		a[i]=(char)k;
    	}
    	printf("%s\n",a);
    }
    return 0;
}
