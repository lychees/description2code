LL f[2000002]; 
LL pow(LL a, LL b,LL m)
	{
		LL x=1,y=a;
		while(b > 0)
			{
				if(b%2 == 1)
					{
						x=(x*y);
						if(x>m) x%=m;
					}
						y = (y*y);
						if(y>m) y%=m;
						b /= 2;
				}
		return x%m;
	}
 
LL InverseEuler(LL n)
	{
		return pow(n,M-2,M);
	}
 
LL ncr(LL n, LL r)
	{
		return (f[n]*((InverseEuler(f[r]) * InverseEuler(f[n-r])) % M)) % M;
	}
