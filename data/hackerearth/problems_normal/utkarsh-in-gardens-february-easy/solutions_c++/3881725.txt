#include<cstdio>
#include<iostream>
#include<bitset>

using namespace std;

#define MAX 2007

bitset<MAX+7> bs[MAX+7];

int main(){
	int n,x;
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		for(int j=0;j<n;j++){
			scanf("%d",&x);
			bs[i][j]=x;
		}
	}

	long long sum = 0L;

	for(int i=0;i<n;i++){
		for(int j=i+1;j<n;j++){
			long long cnt = (bs[i] & bs[j]).count();
			sum+= (cnt*(cnt-1))/2;
		}
	}

	cout<<sum/2<<endl;
}
