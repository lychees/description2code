#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	int n;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++)cin>>a[i];
    int even=0,odd=0;
    int ev[1000],od[1000],e=0,d=0;
    for(int i=0;i<n;i++){
    	if(a[i]%2==0){
    		even+=a[i];
    		ev[e++]=a[i];
    	}
    	else{
    		odd+=a[i];
    		od[d++]=a[i];
    	}
    }
    sort(ev,ev+e);
    sort(od,od+d);
    for(int i=0;i<e;i++){
    	cout<<ev[i]<<" ";
    }
    cout<<even<<" ";
    for(int i=0;i<d;i++){
    	cout<<od[i]<<" ";
    }
    cout<<odd<<endl;
}
