#include<iostream>
#include<algorithm>
#include<set>
#include<vector>
using namespace std;
void sol()
{
    vector< pair<long long , int> > v;
    long long int sum=0,x,total=0,len=0;
    int i,n,min_till;
    cin>>n;
    v.push_back(make_pair(0,1));
    for(i=1;i<=n;i++)
    {
        cin>>x;
        sum+=x;
        v.push_back(make_pair(sum,i+1));
    }
    sort(v.begin(),v.end());
    min_till=v[0].second;
    for(i=1;i<=n;i++)
    {
        long long int  val=max(0,v[i].second-min_till);
        if(val>len)
        {
            len=val;
            total=1;
        }
        if(val==len) total++;
        min_till=min(min_till,v[i].second);
    }
    if(len==0) cout<<"-1"<<endl;
    else cout<<len<<' '<<total-1<<endl;
}
int main()
{
    sol();
    return 0;
}
