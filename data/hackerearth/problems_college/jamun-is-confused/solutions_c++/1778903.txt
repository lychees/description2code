#include<bits/stdc++.h>
using namespace std;
int Gcd(long long a,long long b){
	if(b==0)
		return a;
	else return Gcd(b,(a%b));
}
int main(){
	long long t;
	cin>>t;
	while(t--){
		long long n;
		cin>>n;
		long long gcd;
		cin>>gcd;
		for(long long i=1;i<n;i++){
			long long temp;
			cin>>temp;
			gcd=Gcd(gcd,temp);
		}
		cout<<gcd<<endl;
	}
	return 0;
}
