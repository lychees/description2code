from sys import stdin


def gg(n):
    return (n*(n-1))/2


di = {}
n = int(stdin.readline())
a = []
b = stdin.readlines()
for i in b:
    c = map(int, i.split())
    for j in c:
        a.append(j)
# print len(a)
for i in a:
    di[i] = di.get(i, 0)+1
total = n*(n-1)
num = 0
for i in di:
    num += gg(di[i])

ans = (gg(n)-num)/(total*1.0)
print "%.6f %.6f" % (ans, ans)
