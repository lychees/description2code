#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    string s;
    cin>>s;
    int l=s.size();
    int a=0,b=0,c=0,d=0;
    
    for(int i=0;i<l;i++)
    {
    	if(s[i]=='l')
    	{
    		a++;
    		s[i]=0;
    	}
    	
    	if(s[i]=='o' && a!=0)
    	{
    		b++;
    		s[i]=0;
    	}
    	
    	if(s[i]=='v' && a!=0 && b!=0)
    	{
    		c++;
    		s[i]=0;
    	}
    	
    	if(s[i]=='e' && a!=0 && b!=0 && c!=0)
    	{
    		d++;
    		s[i]=0;
    	}
    	
    }
    
    if(a!=0 && b!=0 && c!=0 && d!=0)
    cout<<"I love you, too!";
    else
    cout<<"Let us breakup!";
    
    return 0;
}
