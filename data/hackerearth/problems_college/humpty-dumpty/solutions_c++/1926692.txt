#include <cstdio>
#include <cmath>
#include <iostream>

using namespace std;

int main()
{
	//simple! check whether the path forms a polygon
	int t,n,x,y,i;
	scanf("%d",&t);
	while(t--)
	{	
		scanf("%d",&n);
		scanf("%d %d",&x,&y);
		int a[n];
		for(i=0;i<n;i++)
			scanf("%d",&a[i]);
		double d = sqrt((double)(x*x+y*y));
		double len = d;
		double edge = d;
		for (int i = 0; i < n; i ++) 
		{
     			 len += (double)a[i];
      			edge = max((double)a[i], edge);
  		 }
		if (len - edge >= edge) 
     			 printf("Able\n");
		else 
                	 printf("Not able\n");
    
		
	}
	

	
}
