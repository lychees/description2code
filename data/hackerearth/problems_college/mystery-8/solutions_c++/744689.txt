#include<iostream>
#include<cstdio>
using namespace std;

void dis(char str[101])
{
	char str2[101];
	int j=0,k=0,i=0;
	while(str[i]!='\0')
	{
		if((str[i]>='a' && str[i]<='z') || str[i]=='*')
		{	str2[j++]=str[i];
		//cout<<str[i];
		}
		i++;
	}
	j--;
	while(str2[j--]!='*')k++;
	cout<<str2[k]<<'\n';
}

int main()
{
    int test_cases;
    cin>>test_cases;
    getchar();
    while(test_cases>0)
    {
        char str[101];
        gets(str);
        dis(str);        
        test_cases--;
    }
    return 0;
}
