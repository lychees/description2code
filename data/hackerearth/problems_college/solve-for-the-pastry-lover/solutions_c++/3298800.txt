#include <iostream>
#define ll long long
#define mo 1000000000000000
using namespace std;

int main()
{
	ll t, i, j, l, m;
	string s;
    cin>>t;
    while(t--)
    {
    	cin>>l;
    	cin>>s;
    	ll dist[10];
    	for(i=0;i<10;i++)
    		dist[i]=mo;
    	for(i=0;i<l;i++)
    	{
    		for(j=i+1;j<l;j++)
    		{
    			if(s[i]==s[j])
    			{
    				if((j-i)<dist[s[i]-'0'])
    					dist[s[i]-'0']=j-i;
    			}
    		}
    	}
    	m=dist[0];
    	for(i=1;i<10;i++)
    	{
    		if(dist[i]<m)
    			m=dist[i];
    	}
    	if(m==mo)
    		cout<<-1<<endl;
    	else
    		cout<<m<<endl;
    }
    return 0;
}
