#include<bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i=0;i<n;i++)
#define ll long long int
#define pi pair<ll,ll>
#define pii pair<pi,ll>
#define f first
#define s second
#define mod 1000000007
#define pb push_back
int a[1000011];
int f[1000011];
ll dp[1000011];
ll bpow(ll x,ll n){
    ll ans=1;
    while(n>0){
        if(n&1) ans*=x;
        x*=x;
        x%=mod;
        ans%=mod;
        n/=2;
    }
    return ans;
}
int main(){
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int N;
    cin >> N;
    for(int i=1;i<=N;i++){
        cin >> a[i];
        f[a[i]]++;
    }
    for(int i=N;i>=1;i--){
        int cur=0;
        for(int j=i;j<=N;j+=i){
            cur+=f[j];
            dp[i]+=dp[j];
            if(dp[i]>=mod) dp[i]-=mod;
        }
        dp[i]=bpow(2,cur)-1-dp[i];
        if(dp[i]<0) dp[i]+=2*mod;
        dp[i]%=mod;
    }
    for(int i=1;i<=N;i++){
        cout<<dp[a[i]]<<" ";
    }
}
