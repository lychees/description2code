#include <iostream>
#include<cstdlib>
using namespace std;

int compare(const void *a,const void *b){
	return (*(int *)a-*(int *)b);
}

int main()
{
   int t,i;
   int n,j,x[100000];
   
   cin>>t;
   for(i=0;i<t;i++){
   	int count=1;
   	cin>>n;
   	for(j=0;j<n;j++)
   		cin>>x[j];
   		
   		qsort(x,n,sizeof(int),compare);
   		int temp=1;
   		for(j=n-2;j>=0;j--){
			if(x[j]==x[j+1])
				temp++;
			else
				temp=1;
			
			if(temp>count){
   			if(x[j+1]<=x[j])
   				count++;
			}
   				
   		}
   		cout<<count<<endl;
   }
    return 0;
}
