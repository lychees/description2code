#include <cstdio>
#include <algorithm>
#include <vector>
using namespace std;
vector<int>crr;
int arr[1000005],brr[1000005];
int main(){
    int t,n,x,k,power,pos;
    scanf("%d",&t);
    while(t--){
      crr.clear();
      scanf("%d %d",&n,&k);
      for(int i=0;i<n;i++){
       scanf("%d",&arr[i]);
       //arr.push_back(x);
      }
      for(int i=0;i<k;i++){
        scanf("%d",&brr[i]);
       // brr.push_back(x);
      }
      crr.push_back(brr[0]);
      for(int i=1;i<k;i++){
        int val = max(crr[i-1],brr[i]);
        crr.push_back(val);
      }
      vector<int> :: iterator lo,up;
      int diff = -1;
      for(int i=0;i<n;i++){
        power = arr[i];
        up = upper_bound(crr.begin(),crr.end(),power);
        int ch = up -crr.begin();
        if(ch > diff){
            pos = i;
            diff = ch;
        }
      }
      printf("%d\n",pos);
    }
return 0;
}
