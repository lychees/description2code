#include <iostream>
#include<algorithm>
#include<map>
using namespace std;
struct theory{
	int start;
	int finish;
};
int compare(const void* a,const void* b)
{
	theory* p=(theory*) a;
	theory* q=(theory*) b;
	return p->start-q->start;
}
int main()
{
	int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		map<int,int> arr;
		arr.clear();
		int t_max=0;
		int t_res=1;
		arr[t_res]=0;
		for(int i=0;i<n;i++)
		{
			int a,b;
			cin>>a>>b;
			arr[a]++;
			arr[b]--;
			t_max=max(t_max,b);
		}
		int maxi = -1;
		int ans = 0;
		for(map<int,int>::iterator it = arr.begin(); it != arr.end(); it++) {
			ans += it->second;
			maxi = max(maxi,ans);
		}
		cout<<maxi<<endl;
	}
    return 0;
}
