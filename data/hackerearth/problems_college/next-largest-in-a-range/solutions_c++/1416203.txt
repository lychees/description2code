#include<bits/stdc++.h>
using namespace std;
int main() {
	int n,a[1002],i,j,q,l,r,ans;
	scanf("%d%d%d%d",&n,&l,&r,&q);
	for(i=0;i<n;i++) {
		scanf("%d",&a[i]);
	}
	sort(a+l,a+r+1);
	while(q--) {
		scanf("%d",&j);
		ans=-1;
		for(i=l;i<=r;i++) {
			if(a[i]>j) {
				ans=a[i];
				break;
			}
		}
		if(ans!=-1) printf("%d\n",ans);
		else puts("-1");
	}
	return 0;
}
