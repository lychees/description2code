#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		int n,k;
		cin>>n>>k;
		int arr[11][1];
		memset(arr,0,sizeof(arr));
		int a;
		for(int i=0;i<n;i++){
			cin>>a;
			int y=arr[a][0];
			y++;
			arr[a][0]=y;
		}
		int tmp=-999;
		for(int i=0;i<11;i++){
			if(arr[i][0]>tmp)tmp=arr[i][0];
		}
		cout<<n-tmp<<endl;
	}
	return 0;
}
