#include <iostream>
#include<limits.h>
#include<vector>
#include<math.h>
#include<algorithm>
using namespace std;

#define ll int

ll arr[100];
int main()
{
	ll t, n, k, i, j, max_val;
	
	cin>>t;
	while (t--)
	{
		 max_val = INT_MAX;
		 cin>>n>>k;
		 vector<vector<int>> value(n,vector<int>(n));

		 for(i = 0; i<n; i++)
			  cin>>arr[i];

		for (i = 0; i <n; i++){
			 for (j = 0; j <n; j++){
			  if(arr[i]>arr[j])
		      value[i][j] = 3*abs(arr[i] - arr[j]);
		      else
		        value[i][j] = 5*abs(arr[i] - arr[j]);

			 }
		 }
		     
		for(i = 0; i<n; i++){
			sort(value[i].begin(), value[i].end());
		}

		for (i = 0; i<n; i++)
		   {
			   int sum = 0;
			   for (j = 0; j<k; j++)
				   sum += value[i][j];
		  if(sum<max_val)
			  max_val = sum;
		}
         cout<<max_val<<endl;
	}	
    return 0;
}
