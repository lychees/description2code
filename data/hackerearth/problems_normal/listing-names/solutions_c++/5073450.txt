#include "bits/stdc++.h"
using namespace std;
int main()
{
	long int tc;
	cin>>tc;

	map<string,long int> mymap;
	vector<string> s;
	for(long int i=0;i<tc;i++)
	{
		string a;
		cin>>a;
		mymap[a]++;
		if(mymap[a]==1)
			s.push_back(a);
	}
	
	sort(s.begin(),s.end());
	for(vector<string>::iterator it=s.begin();it!=s.end();it++)
	{
		cout << (*it) << " "<< mymap[*it] << endl;
	}
	return 0;
}
