#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define pb push_back
#define mp make_pair
#define MAX 20000

ll dp[MAX];
ll arr[MAX];
int main()
{
  int test;cin>>test;int t;
  int m,n;int i,j;
  for(t=0;t<test;t++)
  {
    memset(dp,0,sizeof(dp));
    cin>>m>>n;
    for(i=0;i<n;i++)
    {
      cin>>arr[i];
    }
    for(i=1;i<=m;i++)
    {
      for(j=0;j<n;j++)
      {
        if (i-arr[j]<0)
        continue;
        if (i-arr[j]==0)
        {
          dp[i]=1;
          break;
        }
        else if (dp[i-arr[j]]==0)
        {
          dp[i]=1;
          break;
        }
      }
    }
    if (dp[m]==1)
    {
      cout<<"Little Deepu\n";
    }
    else
    cout<<"Kate\n";

  }//test


return 0;
}
