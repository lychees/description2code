#include <cstdio>
#include <cstring>
#include <algorithm>
#define MX 10005
using namespace std;

bool chk[26];
char buf[MX];
int main() {
	int i, T, ans, N, j, k;
	for (scanf("%d", &T); T--; ) {
		scanf("%s", buf);
		ans = 0;
		N = strlen(buf);
		for (i = 0; i < N; i++) {
			memset(chk, 0, sizeof chk);
			for (j = 0; i + j < N && j < 26; j++) {
				k = buf[i + j] - 'A';
				if (chk[k]) break;
				chk[k] = true;
				ans = max(ans, j + 1);
			}
		}
		printf("%d\n", ans);
	}
	return 0;
}
