#include<iostream>
#include<cstdio>
#include<algorithm>
#define ll long long int
using namespace std;
int main()
{
ll n,k,i,j,x,mult,search,count = 0,up;
ll a[2000];
cin>>n>>k;
for(i=0;i<n;i++) cin>>a[i];


sort(a,a+n);

for(i=0;i<n;i++)
{
for(j=i+1;j<n;j++)
{
mult = a[i]*a[j];
if(mult>k) break;

search = k/mult;
up = upper_bound(a,a+n,search) - a;
if(up == 0) continue;
up--;
if(up-j>0) count+=(up-j);


}
}

cout<<count;

return 0;
}
