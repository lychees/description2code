#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;

int v[11000];

int main() {
    int T;
    cin>>T;
    for(int t=0;t<T;t++)
    {
        int n;
        cin>>n;
        for(int i=0;i<n;i++)cin>>v[i];
        int m=v[0];
        for(int i=0;i<n;i++)m=min(m,v[i]);
        int mr=-1;
        for(int j=0;j<5;j++)
        {
            int res=0;
            for(int i=0;i<n;i++)
            {
                int u=v[i];
                while(v[i]-5>=m-j)
                {
                    res++;
                    v[i]-=5;
                }
                while(v[i]-2>=m-j)
                {
                    res++;
                    v[i]-=2;
                }
                while(v[i]-1>=m-j)
                {
                    res++;
                    v[i]--;
                }
                v[i]=u;
            }
            if(mr==-1 or res<mr)mr=res;
        }
        cout<<mr<<endl;
            
    }
    return 0;
}
