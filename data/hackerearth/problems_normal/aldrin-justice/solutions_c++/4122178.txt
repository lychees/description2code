#include<bits/stdc++.h>
using namespace std;
typedef long long lld;

int main()
{
	lld t;
	cin>>t;
	
	while(t--)
	{
	   lld b1,b2,d1,d2;
	   
	   cin>>b1>>b2>>d1>>d2;
	   
	   lld l=min(max(b1,b2),max(d1,d2));
	   lld r=max(min(b1,b2),min(d1,d2));
	   
	   if(l>r)
	   cout<<"Nothing\n";
	   else if(l<r)
	   cout<<"Line\n";
	   else
	   cout<<"Point\n";
	   
	   
		
	}
	

	
	
	
	
}
