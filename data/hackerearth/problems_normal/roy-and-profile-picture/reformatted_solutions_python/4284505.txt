length = int(raw_input())
n = int(raw_input())

a = 0

while a < n:
    w_h = raw_input()
    width, height = w_h.split()
    width = int(width)
    height = int(height)
    a += 1

    if width >= length and height >= length and width == height:
        print 'ACCEPTED'
    elif width < length or height < length:
        print 'UPLOAD ANOTHER'
    else:
        print 'CROP IT'
