#include <iostream>
#include <stdio.h>
#include<algorithm>
#define m 1000000007
using namespace std;

int main()
{
    int t,i;
    long long int a[5],ans;
    scanf("%d",&t);
    while(t--)
    {
    	for(i=0;i<3;i++)
    	{
    		scanf("%lld",&a[i]);
    	}
    	sort(a,a+3);
    	ans=1;
    	ans=((a[0]%m)*((a[1]-1)%m)%m)*((a[2]-2)%m);
    	printf("%lld\n",(ans%m));
    	
    }
    return 0;
}
