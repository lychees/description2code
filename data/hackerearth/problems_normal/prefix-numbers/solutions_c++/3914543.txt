vvi mul(const vvi & a, const vvi & b) {
	int n = a.size();
	vvi c(n, vi(n));
	for (int i = 0; i < n; ++i) for (int j = 0; j < n; ++j) {
		for (int k = 0; k < n; ++k) c[i][j] += a[i][k]*b[k][j];
	}
	return c;
}

vvi mpow(vvi x, ll n) {
	vvi res(x.size(), vi(x.size()));
	for (int i = 0; i < x.size(); ++i) res[i][i] = 1;
	while (n) {
		if (n & 1) res = mul(res, x);
		x = mul(x, x);
		n /= 2;
	}
	return res;
}
