#include<stdio.h>

int main()
{
  int t;
  long long int n;
  scanf("%d",&t);
  while (t>0)
  {
    scanf("%lld",&n);
    if(n%2==1)
       printf("%lld\n",2);
    else
     {
       if(n&(n-1)==0)
           printf("%lld\n",2*n);
       else
       {
             long long int r=1;
             while(n%2==0)
             {
                   r=r*2;
                   n=n/2;

             }
             printf("%lld\n",2*r);
       }
     }
        t=t-1;
  }

  return 0;
}
