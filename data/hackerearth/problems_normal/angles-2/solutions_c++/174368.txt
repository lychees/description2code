#include<iostream>
#include<math.h>

using namespace std;

int main()
{
	int n,k,a[11],x,b[11],c[361]={0},i,j;

	cin>>n>>k;

	for(i=0;i<n;i++)
		cin>>a[i];

	for(i=0;i<k;i++){
		cin>>x;
		b[i]=x;
	}

	for(i=0;i<n;i++){
		c[a[i]]=1;
		for(j=2;(a[i]*j)%360!=a[i];j++){
			c[(a[i]*j)%360]=1;
		}

	}

	for(i=0;i<k;i++)
	{
		if(c[b[i]])	cout<<"YES\n";
		else cout<<"NO\n";
	}

	return 0;
}
