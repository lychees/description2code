#include<bits/stdc++.h>
#define getcx getchar_unlocked
using namespace std;

inline void inp( int &n )
 {
    n=0;
    int ch=getcx();int sign=1;
    while( ch < '0' || ch > '9' ){if(ch=='-')sign=-1; ch=getcx();}

    while(  ch >= '0' && ch <= '9' )
            n = (n<<3)+(n<<1) + ch-'0', ch=getcx();
    n=n*sign;
  }

int main()
{
   int t,n,m,i,x,ans,a[30000];

   //cin>>t;
    inp(t);
   while(t--)
   {

        priority_queue<int>pq;
       //cin>>n;
       inp(n);

       for(i=0;i<n;i++)
       {
           //cin>>x;
           inp(x);
           pq.push(x);
       }

        m=1;
        ans=0;
        while(m<=n)
        {
            for(i=1;i<=m;i++)
            {
                if(pq.top()==0)
                {
                    ans=m-1;
                    break;
                }
                a[i]= pq.top();
                pq.pop();
            }
            for(i=1;i<=m;i++)
            pq.push(a[i]-1);

            if(ans)
                break;

            m++;
        }

        if(ans==0)
            ans=n;

        //cout<<ans<<'\n';

    printf("%d\n",ans);


   }
    return 0;
}
