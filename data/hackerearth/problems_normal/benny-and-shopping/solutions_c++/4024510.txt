#include <bits/stdc++.h>
#define foru(i, a, b) for(int i = a; i <= b; ++i)
using namespace std;

int main() {
	//freopen("input.txt", "r", stdin);
	//freopen("output.txt", "w", stdout);
	int TC; scanf("%d\n", &TC);
	while (TC--) {
		string S, ans = "";
		getline(cin, S); 
		for(int i = S.size() - 1; i >= 0; --i) if (S[i] == ' ') S.erase(i, 1);
		int pos;
		for(int i = 0; i < S.size(); ++i) if (S[i] == '$') {pos = i; break;}
		while (pos + 1 < S.size() && isdigit(S[pos + 1])) ans += S[++pos];
		while (ans.size() > 1 && ans[0] == '0') ans.erase(0, 1);
		putchar('$'); puts(ans.c_str());
	}
	return 0;
}
