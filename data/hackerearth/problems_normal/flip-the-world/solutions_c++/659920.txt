#include<stdio.h>
int main(){
	int n,m,t,count;
	char a[60][60];
	scanf("%d",&t);

	while(t--){
		scanf("%d %d",&n,&m);
		getchar();
		for(int i=0;i<n;i++){
				gets(a[i]);
		}
		/*
		for(int i=0;i<n;i++){
			for(int j=0;j<m;j++){
				printf("%c",a[i][j]);
			}		
			puts("");
		}
		*/

		count=0;
		for(int i=n-1;i>=0;i--){
			for(int j=m-1;j>=0;j--){
				if(a[i][j]=='0'){
					count++;
					for(int x=0;x<=i;x++){
						for(int y=0;y<=j;y++)
							a[x][y]=a[x][y]=='1'?'0':'1';
					}
				}
			}
		}
		printf("%d\n",count);
	}
	return 0;
}
