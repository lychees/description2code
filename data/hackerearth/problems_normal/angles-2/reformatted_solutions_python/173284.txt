N, K = map(int, raw_input().split(" "))
angles_raju = map(int, raw_input().split(" "))
angles_rani = map(int, raw_input().split(" "))
all_angles = {}
for angle in angles_raju:
    temp = angle
    all_angles[temp] = 1
    while temp % 360 != 0:
        temp = temp + angle
        all_angles[temp % 360] = 1

for angle in angles_rani:
    if angle in all_angles:
        print "YES"
    else:
        print "NO"
