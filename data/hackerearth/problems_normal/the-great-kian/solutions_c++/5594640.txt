#include<iostream>
using namespace std;
int main()
{
   int n ,k=0, count =0 , count1=0 , count2 =0 ;
  long long int sum1=0 ;
  long long int sum2=0 ;
  long long int sum3=0 ;
    cin>>n;
 long long  int *arr	= new long long int[n];
   for(int i=0 ; i<n ; i++ )
   cin>>arr[i];
   
   for(int j =0 ; j<n ; j++)
   {
   	
   	if(k>=0)
   	{
   		
   		if(count==0||count==3)
   		{
   		
   		sum1= sum1+arr[j];
   	count=0;
	   
	   }
		   count++;
   		
   		
   	}
   	
   	
   	   	if(k>=1)
   	{
   		if(count1==0||count1==3)
   		{
   		
   		sum2= sum2+arr[j];
   	     count1=0;
	   
	   }
		 
		   count1++;
   		}
   		
   	   	if(k>=2)
   	{
   		
   		if(count2==0||count2==3)
   		{
   		
   		sum3= sum3+arr[j];
   	count2=0;
	   
	   }
		   count2++;
   		
   		
   	}
   	
   	k++;
   	
   	
   	
   }
   cout<< sum1 <<" "<< sum2<<" "<< sum3;
   return 0;
   
	
	
}
