#include<iostream>
#include<string.h>
#include<algorithm>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    string s;
    string r;
    unsigned int t=0,i,flag=0;
    cin>>t;
    for(;t>0;t--){
        cin>>s;
        r = s;
        reverse(r.begin(),r.end());
        for(i=0;i<s.length()-1;i++){
            if(abs((int)s[i+1]-(int)s[i])==abs((int)r[i+1]-(int)r[i])){
                flag=1;
            }
            else{
                flag=0;
                break;
            }
        }
        if(flag){
            cout<<"Funny\n";
        }
        else
            cout<<"Not Funny\n";
        flag=0;
    }
    return 0;
}
