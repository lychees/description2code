long long getfact(int lngth){
if(memarr[lngth]>0){
	return memarr[lngth];
}
memarr[lngth]=((getfact(lngth-2)+getfact(lngth-1))*(lngth-1))%1000000007;
return memarr[lngth];
}

	char input=getchar_unlocked();
	int val=0;
	while(input<'0'||input>'9'){
		input=getchar_unlocked();
	}
	
	while(input>='0'&&input<='9'){
		val=val*10+input-'0';
		input=getchar_unlocked();

char getchr(){
	char input=getchar_unlocked();
	
	while(input<33||input>126){
		input=getchar_unlocked();
	}
	
	if(input>=33&&input<=126){
		return input;
	}
}
