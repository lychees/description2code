typedef long long ll;

const int inf=1e9;
const int MOD=1e9+7;
const int N=0;

int solve(int L,int R,ll n){
    int cur=0;
    while (n>1){
        if (!cur) n=n/R+(n%R>0);
        else n=n/L+(n%L>0);
        cur^=1;
    }
    return cur;
}
int main(){
    int test,L,R;
    ll n;
    scanf("%d",&test);
    while (test--){
        scanf("%d%d%lld",&L,&R,&n);
        if (solve(L,R,n)) printf("Adam\n");
        else printf("Bob\n");
    }
}
