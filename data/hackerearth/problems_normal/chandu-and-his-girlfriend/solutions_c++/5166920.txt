#include<bits/stdc++.h>
typedef long long ll;
using namespace std;
void merge(ll L[],ll R[],ll a[],int l,int r)
{
	int i,j,k;
	i=k=j=0;
	while((i<l)&&(j<r))
	{
		if(L[i]<R[j])
		a[k++]=R[j++];
		else
		a[k++]=L[i++];		
	}
	while(i<l)
	a[k++]=L[i++];
	while(j<r)
	a[k++]=R[j++];
}
void mergesort(ll *a,int n)
{
	if(n<2)
	return ;
	else
	{
		int mid,i;
		mid=n/2;
		ll L[mid+3],R[(n-mid)+3];
		for(i=0;i<mid;i++)
		L[i]=a[i];
		for(i=mid;i<n;i++)
		R[i-mid]=a[i];
		mergesort(L,mid);
		mergesort(R,n-mid);
		merge(L,R,a,mid,n-mid);
	}
	
}
int main()
{
	ios_base::sync_with_stdio(0);cin.tie(NULL);cout.tie(NULL) ;
	int t,i,n,pos;

	{
		cin>>n;
		ll a[n+2],value;

		cin>>a[i];
		mergesort(a,n);

		cout<<a[i]<<" ";
		cout<<"\n";
	}
	
	return 0;
}
