#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int a[n][n];
    int sum1=0,sum2=0;
    for(int i=0;i<n;i++){
    	for(int j=0;j<n;j++){
    		cin>>a[i][j];
    		if(i==j)
    		sum1=sum1+a[i][j];
    	}
    }
    for(int i=0;i<n;i++){
    	sum2=sum2+a[i][n-i-1];
    }
    if(sum1-sum2>0)
    cout<<sum1-sum2;
    else 
    cout<<sum2-sum1;
}
