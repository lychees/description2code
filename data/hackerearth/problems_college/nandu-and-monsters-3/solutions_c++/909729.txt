#include <iostream>
using namespace std;

bool reach(int i, int j, string mine, int n){
	if(i<=n && i>=1){
		if(mine[i-1]=='1')
			return false;
		
		if(i==j)
			return true;
			
		if(i<j)
			return(reach(2*i, j, mine, n) || reach(2*i+1, j, mine, n));
		else
			return(reach(i/2, j, mine, n));
	}
	else
		return false;
}

bool isPathSafe(int i, int j, string mine, int n){
	while(i!=j){
		if(mine[i-1]=='1' || mine[j-1]=='1')
			return false;
			
		if(j>i)
			j=j/2;
		else
			i=i/2;
	}
	if(mine[i-1]=='1')
		return false;
	else
		return true;
}

int main()
{
    int n, q;
    cin>>n>>q;
    string mine;
    cin>>mine;
    int i, j;
    for(int idx=0; idx<q; idx++){
    	cin>>i>>j;
    	if(isPathSafe(i, j, mine, n))
    		cout<<"Yes"<<endl;
    	else
    		cout<<"No"<<endl;
    }
    return 0;
}
