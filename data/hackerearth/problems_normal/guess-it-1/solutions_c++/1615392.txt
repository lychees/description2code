#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t;
   cin>>t;
   while(t--)
   {
   	string str;
   	cin>>str;
   	int k=str.length();
   	int sum=0;
   	for(int i=0;i<k;i++)
   	{
   		if(str[i]>=48&&str[i]<=57)
   		sum=sum+(str[i]-48);
   	}
   	sum=sum%9;
   	if(sum==0)
   	cout<<"0 or 9"<<endl; 
   	else
   	{
   	sum=9-sum;
   	cout<<sum<<endl;
   }
   }
    return 0;
}
