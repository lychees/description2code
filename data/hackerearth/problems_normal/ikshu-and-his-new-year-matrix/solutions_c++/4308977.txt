#include<bits/stdc++.h>
using namespace std;
#define LL long long
#define S 1000005
#define W while
#define M 1000000007
#define str string
#define pi 3.14159265
LL mini=INT_MAX;
LL maxi=INT_MIN;
int prime[10001];
int k;
int a[1001][1001];
int dp[1001][1001];
void seive()
{
	for(int i=0;i<10001;i++)
		prime[i]=1;
		prime[0]=0;
		prime[1]=0;
	for(int i=2;i<=10000;i++)
	{
		for(int j=i*i;j<10001&&prime[i];j=j+i)
		{
			prime[j]=0;
		}
	}
}
int fn(int i,int j,int* count)
{
	if(prime[a[i][j]]==0)
	{
		int flag=1;
	//	if(dp[i][j]==1)
	//		flag=0;
	
		int t=a[i][j];
		W(t<=10000&&flag)
		{
			t+=k;
			(*count)++;
			if(prime[t])
				break;
		}
		if(t>10000)
			return 0;
	}
	dp[i][j]=1;
	if(prime[a[i+1][j+1]]==0)
	{
		int flag=1;
	//	if(dp[i+1][j+1]==1)
	//		flag=0;
	
		int t=a[i+1][j+1];
		W(t<=10000&&flag)
		{
			t+=k;
			(*count)++;
			if(prime[t])
				break;
		}
		if(t>10000)
			return 0;
	}
	dp[i+1][j+1]=1;
	
	if(prime[a[i+2][j]]==0)
	{
		int flag=1;
	//	if(dp[i+2][j]==1)
	//		flag=1;
	
		int t=a[i+2][j];
		W(t<=10000&&flag)
		{
			t+=k;
			(*count)++;
			if(prime[t])
				break;
		}
		if(t>10000)
			return 0;
	}
	dp[i+2][j]=1;
	if(prime[a[i][j+2]]==0)
	{
		int flag=1;
	//	if(dp[i][j+2])
	//	flag=0;
		int t=a[i][j+2];
		W(t<=10000&&flag)
		{
			t+=k;
			(*count)++;
			if(prime[t])
				break;
		}
		if(t>10000)
			return 0;
	}
	dp[i][j+2]=1;
	
	if(prime[a[i+2][j+2]]==0)
	{
		int flag=1;
	//	if(dp[i+2][j+2])
	//		flag=0;
		int t=a[i+2][j+2];
		W(t<=10000&&flag)
		{
			t+=k;
			(*count)++;
			if(prime[t])
				break;
		}
		if(t>10000)
			return 0;
	}
	dp[i+2][j+2]=1;
	return 1;
	
	
	
}

int main()
{
	seive();
//	for(int i=0;i<20;i++)
//	cout<<prime[i]<<" ";
	int n;
	cin>>n>>k;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
		{	cin>>a[i][j];
		dp[i][j]=0;
		}
	}
	if(n<=2)
		cout<<"no\n";
	int c1=0,c2=0;
	for(int i=0;i<n-2;i++)
		for(int j=0;j<n-2;j++)
		{
			int count=0;
			if(fn(i,j,&count))
			{
				if(mini>count)
				{
					mini=count;
					c1=i+2;
					c2=j+2;
				}
				
			}
		}
	if(n>2)
	{
		if(!c1&&(!c2))
			cout<<"no\n";
		else
		{
			cout<<"yes\n"<<mini<<endl<<c1<<" "<<c2<<endl;
			
		}
		mini=INT_MAX;
	}
	return 0;
}
