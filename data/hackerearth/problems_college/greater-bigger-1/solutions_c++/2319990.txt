#include <iostream>
#include<cmath>
#include<cstring>
#include<algorithm>

using namespace std;

void swap(char *a, char *b)
{
    char temp = *a;
    *a = *b;
    *b = temp;
}
 

void findNext(char string[], int n)
{
    int i, j;
 
    for (i = n-1; i > 0; i--)
        if (string[i] > string[i-1])
           break;

    if (i==0)
    {
        cout << "no answer";
        return;
    }
 

    char x = string[i-1]; 
    int smallest = i;
    for (j = i+1; j < n; j++)
        if (string[j] > x && string[j] < string[smallest])
            smallest = j;
 
    swap(&string[smallest], &string[i-1]);
 
    
    sort(string + i, string + n);
 
    cout << string;
 
    return;
}

int main() {
	
	int t;
	char w[101];
	cin >> t;
	while(t--) {
		
		cin >> w;
		findNext(w, strlen(w));
		cout << endl;
	}
	return 0;
}
