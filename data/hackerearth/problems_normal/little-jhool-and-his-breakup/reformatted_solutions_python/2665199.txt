#!usr/bin/python

a = []
flag = 0
string = raw_input()
a = list(string)

for i in range(0, len(a)):
    if (a[i] == 'l' or a[i] == 'o' or a[i] == 'v' or a[i] == 'e'):
        continue
    else:
        a[i] = '$'

for i in range(0, len(a)):
    if (a[i] != '$'):
        if (flag == 0 and a[i] == 'l'):
            flag = 1
        elif (flag == 1 and a[i] == 'o'):
            flag = 2
        elif (flag == 2 and a[i] == 'v'):
            flag = 3
        elif (flag == 3 and a[i] == 'e'):
            flag = 4

if (flag == 4):
    print "I love you, too!"
else:
    print "Let us breakup!"
