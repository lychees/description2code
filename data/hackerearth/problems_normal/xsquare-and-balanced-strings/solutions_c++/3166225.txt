#include <iostream>
#include<string>
using namespace std;

int main()
{
    // cout << "Hello World!" << endl;
    int t;
    string str;
    cin>>t;
    while(t--)
    {
    	cin>>str;
    	int len=str.size();
    	int arr[26];
    	for(int i=0;i<=25;i++)arr[i]=0;
    	
    	for(int i=0;i<len;i++)
    		arr[str[i]-'a']++;
    	int flag=0;
    	for(int i=0;i<26;i++)
    	{
    		if(arr[i]%2!=0)
    		{
    			flag=1;break;
    		}
    	}
    	if(flag==1)cout<<"-1"<<endl;
    	else cout<<"1"<<endl;
    }
    return 0;
}
