#include<iostream>
using namespace std;
int reverse(int a)
{
	int s;
	while(a>0)
	{
		s=s*10+a%10;
		a=a/10;
	}
	return s;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int a,b;
		cin>>a>>b;
		int p=a%10;
		while(p==0)
		{
			
			a=a/10;
			p=a%10;
		}
		int q=b%10;
		while(q==0)
		{
			b=b/10;
			q=b%10;
		}
		int x=reverse(a);
		int y=reverse(b);
		int sum=x+y;
		int z=reverse(sum);
		cout<<z<<endl;
		
	}
	return 0;
}
