def shoot():
    for test in range(input()):
        jamount, pcost = map(int, raw_input().split())
        count = 0
        while jamount > 1:
            if jamount % pcost == 0:
                count += 1
                jamount /= pcost
            else:
                break
        if jamount == 1:
            print count, 0
        else:
            print count, jamount
shoot()
