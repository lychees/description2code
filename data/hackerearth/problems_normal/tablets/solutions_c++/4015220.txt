#include<bits/stdc++.h>
using namespace std;
#define S(t) scanf("%d",&t)
#define SL(t) scanf("%ld",&t)
#define PL(t) printf("%lld\n",t)
#define P(t) printf("%d\n",t)
#define D(t) int i,j,t; S(t); while(t--)
#define For(i,a,b) for(i=a;i<b;i++)
#define Fore(i,a,b) for(i=a;i<=b;i++)
typedef long long int ll;
ll dp[100005],a[100005];
int main()
{
    int i,j,n;
    S(n);
    Fore(i,1,n)
    {
        SL(a[i]);
        if(a[i]>a[i-1])
        {
            dp[i]=dp[i-1]+1;
        }
        else if(a[i]==a[i-1])
            dp[i]=1;
        else
        {
            dp[i]=1;
            for(j=i-1;j>=0;j--)
            {
                if(a[j]>a[j+1]&&dp[j]<=dp[j+1])
                    dp[j]+=1;
                else
                    break;
            }
        }
    }
    ll sum=0;
    Fore(i,1,n)
    sum+=dp[i];
    PL(sum);

    return 0;
}
