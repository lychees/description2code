#include <iostream>
using namespace std;

int main()
{
    int n,i,j;
    cin>>n;
    int arr[105][255];
    char str[105];
    for(i=0;i<n;i++)
    {
    	cin>>str;
    	for(j=0;str[j];j++)
    	{
    		arr[i][str[j]]++;
    	}
    }
    int ans=0;
    for(j='a';j<'z';j++)
    {
    	for(i=0;i<n;i++)
    	{
    		if(arr[i][j]==0) break;
    	}
    	if(i>=n) ans++;
    }
    cout<<ans<<endl;
    return 0;
}
