#include<bits/stdc++.h>
using namespace std;
long arr[10000001],t1[3000001];
void pre()
{
	for(int i=1;i<=10000000;i++)
	arr[i]=1;
	for(int i=2;i<=10000000;i++)
	{
		if(arr[i]==1)
		{
			arr[i]=2;
		//	cout<<arr[i]<<endl;
			for(int j=i+i;j<=10000000;j=j+i)
			{
				arr[j]=(arr[j]*arr[i]);
			}
		}
		//cout<<arr[i]<<endl;
	}
}
int main()
{
	pre();
	int n;
	scanf("%d",&n);
	long long ans=0;
	for(int i=0;i<n;i++)
	scanf("%lld",&t1[i]);
	for(int i=0;i<n;i++)
	ans+=arr[t1[i]];
//	cout<<arr[8]<<endl;
	printf("%lld\n",ans);
}
