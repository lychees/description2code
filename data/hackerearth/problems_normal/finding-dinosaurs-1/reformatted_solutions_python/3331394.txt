N, K, L, Q = map(int, raw_input().strip().split())
dinoDict = {}
for _ in xrange(N):
    dinoDesc = raw_input().strip()
    dName = dinoDesc.split()[0]
    dAttrs = dinoDesc[len(dName)+1:]
    dinoDict[dAttrs] = dName
for _ in xrange(Q):
    attrs = raw_input().strip()
    if attrs in dinoDict:
        print dinoDict[attrs]
    else:
        print "You cant fool me :P"
