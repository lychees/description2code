#include<iostream>
using namespace std;
int main()
{

  long long   int t,n,c;

    cin>>t;
    while(t--)
    {  c=0;
        cin>>n;
         long long int a[n];
        for(int i=0;i<n;i++)
        {
            cin>>a[i];
        }
        for(int i=0;i<n;i++)
        {
            if(a[i]>a[i+1])
            {
                c=1;
            }
        }
        if(c==1)
        {
            cout<<"NO"<<endl;
        }
        else{cout<<"YES"<<endl;}
    }
}
