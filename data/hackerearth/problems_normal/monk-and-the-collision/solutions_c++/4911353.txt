#include<iostream>
using namespace std;

int has( int x){
	int index = x % 10;
	return index;
	
}
int main(){
	int t,x;
	cin>>t;
	while(t--){
		int n,a[15],colli=0;
		cin>>n;
		for(int i=0;i<10;i++){
			a[i]=0;
		}
		for(int i=0;i<n;i++){
			cin>>x;
			 a[has(x)]++;
			
		}
		for(int i=0;i<10;i++){
			if(a[i]!=0)
				colli=colli+a[i]-1;
			
			
		}		
		cout<<colli<<endl;
		
	}
	
	return 0;
}
