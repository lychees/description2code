#include<bits/stdc++.h>
using namespace::std;

#define ll long long int

int main()
{
	ll i,j,k,t,l;

	cin>>t;
	
	while(t--)
	{char a[1000];

	cin>>a+1;
	l = strlen(a+1);

	ll dp[100][100];

	for(i=0;i<=100;i++)
	dp[i][i]=1;
	
	//for()
	for(i=l-1;i>=1;i--)
	{
		for(j=i+1;j<=l;j++)
		{
			if(a[i]==a[j])
			{
				dp[i][j]=dp[i+1][j]+dp[i][j-1]+1;
			}
			else
				dp[i][j]=dp[i+1][j]+dp[i][j-1]-dp[i+1][j-1];
		}	
	}

	cout<<dp[1][l]<<"\n";
	}
}
