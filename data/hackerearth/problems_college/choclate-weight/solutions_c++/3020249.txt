#include <iostream>
using namespace std;

int main()
{
	int n,t;
	cin>>n;
	int a[n];
	for(int i=0;i<n;i++){cin>>a[i];}
//	int t;
	cin>>t;
	while(t--){
		int x,f=-1;
		cin>>x;
		for(int i=0;i<n;i++){
			if(a[i]==x){f=1;break;}
		}
		
		
		if(f==1){  cout << "I Like You" << endl;}
		else{
			for(int i=0;i<n;i++){
			for(int j=0;j<n;j++){
			if(i!=j && a[i]+a[j]==x){
				f=1;break;
			}if(f==1){break;}
		}
		
	}
	if(f==1)
		cout << "I Like You" << endl;
		else
		cout << "I Hate You" << endl;
	}//else
	
	}//while
  
    return 0;
}
