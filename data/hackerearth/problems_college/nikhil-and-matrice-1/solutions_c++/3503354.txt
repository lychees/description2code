#include <bits/stdc++.h>
using namespace std;

int main()
{
	
	int i,j,k,l,m,n,w,x,y,z,t,count;
	
	int a[50][50];
	
	scanf("%d",&t);
	
	
	
	
	while(t--)
	{
		scanf("%d",&n);
		
		w=z=0;
		x=y=n-1;
		
		k=1;
		count=0;
		
		while(1)
		{
			
		
		for(j=z;j<=x;j++)
		{
			a[w][j]=k++;
		}
		w++;
		count++;
		if(count==2*n-1)
		break;
		
		for(i=w;i<=y;i++)
			a[i][x]=k++;
		x--;
		count++;
		if(count==2*n-1)
		break;
		
		for(j=x;j>=z;j--)
			a[y][j]=k++;
		y--;
		count++;
		if(count==2*n-1)
		break;
		
		for(i=y;i>=w;i--)
			a[i][z]=k++;
		z++;
		count++;
		if(count==2*n-1)
		break;
		
	//	printf("%d%d%d%d %d\n",w,x,y,z,k);
		
		
		}
		
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
				printf("%d",a[i][j]);
			printf("\n");
		}
	printf("\n");	
	}
	
	
	
	
	
	
	
	
    //cout << "Hello World!" << endl;
    return 0;
}
