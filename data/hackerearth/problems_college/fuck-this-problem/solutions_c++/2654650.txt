#include<bits/stdc++.h>
using namespace std;
typedef long long int uli;
const int mx=1e4+10;
char s[mx];
int main(){
   int n,k;
   gets(s);
   scanf("%d",&k);
   n=strlen(s);
   for(int i=0;i<n;i+=k){
      for(int j=i;j<i+k;j++){
         printf("%c",s[j]);
      }
      printf("\n");
   }
   return 0;
}
