#include <iostream>
using namespace std;

int main()
{
    int n,q;
    cin >> n >> q;
    int a[n],b[n];
    for(int i = 0; i < n; i++){
    	cin >> a[i];
    }
    for(int i = 0; i < n; i++){
    	cin >> b[i];
    }
    int temp,l,r;
    while(q--){
    	long long sum = 0;
    	cin >> temp >> l >> r;
    	l--;r--;
    	if(temp == 1){
    		for(int i = l; i <= r; i++){
    			if((i-l)%2 == 0){
    				sum += a[i];
    			}else{
    				sum += b[i];
    			}
    		}
    	}else{
    		for(int i = l; i <= r; i++){
    			if((i-l)%2 == 0){
    				sum += b[i];
    			}else{
    				sum += a[i];
    			}
    		}
    	}
    	cout << sum << endl;
    }
    return 0;
}
