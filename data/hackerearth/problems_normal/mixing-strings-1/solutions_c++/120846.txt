#include<iostream>
#include<string>
#include<algorithm>
#include<vector>
using namespace std;

int main()
{
	int N;
	cin >> N;
	int i,k;
	vector<string> v;
	vector<int> v1;
	for(i=0;i<N;i++){
		string s;
		cin >> s;
		v1.push_back(i);
		v.push_back(s);
	}

	int ans=2000;
	
	do{
		string prev;
		for(i=0;i+1<N;i++){
			string x;
			if(i==0) x=v[v1[i]];
			else 	x=prev;
			
			string y=v[v1[i+1]];
			int j,lasttmp=-1;
			string tmp,tmp1,ptmp,ptmp1;
			for(k=x.size()-1,j=0;j<y.size(),k>=0;j++,k--){ 
				 tmp+=y[j];
				tmp1+=x[k];
				ptmp=tmp;
				reverse(ptmp.begin(),ptmp.end());
				if(ptmp==tmp1){
					//cout <<tmp<<"####"<<endl;
					lasttmp=j;
				}
			}
			prev=x+y.substr(lasttmp+1);
			//cout <<prev<<":prev"<<endl;
		}	
		ans=min(ans,(int)prev.size());
		
	}while(next_permutation(v1.begin(),v1.end()));
	printf("%d\n",ans);
	return 0;
}
