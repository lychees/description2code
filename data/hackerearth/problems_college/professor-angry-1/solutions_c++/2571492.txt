#include <iostream>

using namespace std;

int main()
{
	int t, n, k, *a, ctr;
	cin >> t;
	while(t--)
	{
		cin >> n >> k;
		a = new int[n];
		ctr = 0;
		for(int i=0; i<n; i++)
		{
			cin >> a[i];
			if(a[i]<=0)
				ctr++;
		}
		if(ctr<k)
			cout << "YES" << endl;
		else
			cout << "NO" << endl;
	}
	return 0;
}
