using namespace std;
int a[100005], b[100005];
int main()
{

    int t;
    cin>>t;
    while (t--) {
          int n;
          cin>>n;
          int i;
          for (i = 1; i <= n; i++) cin>>a[i];
          for (i = 1; i <= n; i++) b[a[i]] = i;
          int cnt = 0;
          for (i = 1; i <= n; i++) if (b[i] == a[i])cnt++;       
          if (cnt == n) cout<<"inverse\n";
          else cout<<"not inverse\n";
    }
    
  //  getch();
    return 0;
}
