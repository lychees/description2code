#include <iostream>
#include <cstring>
using namespace std;

int main()
{
	unsigned long long j,curr,n,i,alpha[26],q,len,num;
	string str;
	cin>>str;
	len=str.length();
	for(i=0;i<26;i++)
	alpha[i]=0;
	for(i=0;i<len;)
	{
		if(str[i]>='a' && str[i]<='z')
		{
			j=i+1;
			num=0;
			while(j<len && (str[j]>='0' && str[j]<='9') )
			{
				num=num*10+(str[j]-'0');
				j++;
			}
			alpha[str[i]-'a']+=num;
			i=j;
		}
	}
	cin>>q;
	while(q--)
	{
		cin>>n;
		curr=0;
		for(i=0;i<26;i++)
		{
			curr+=alpha[i];
			if(n<=curr)
			{
				cout<<(char)(i+'a')<<"\n";
				break;
			}
		}
		if(i==26)
		cout<<"-1\n";
	}
    return 0;
}
