# include<iostream>
# include<vector>

# define rep(i,a,b) for(int i=a;i<b;++i)

using namespace std;

typedef long long int lli;
typedef vector<lli > vlli;

int main(){
	
	ios::sync_with_stdio(false);
	int tc,n,ptr;
	vlli a,b;
	lli ans;
	
	cin>>tc;
	
	while(tc--){
		cin>>n;

		a.resize(n);
		b.resize(n);
		
		rep(i,0,n)cin>>a[i];
		rep(i,0,n)cin>>b[i];
		
		ptr = 0;
		ans = 0;
		
		rep(i,0,n){
			if(ptr<i)ptr = i;
			while(ptr<n && b[ptr]>=a[i])++ptr;
			//ans+=(ptr-i)*(n-ptr)+((n-1-ptr)*(n-ptr)/2);
			//cout<<ptr<<" "<<i<<" "<<ptr-1-i<<endl;
			if(ans< (ptr-1-i))ans = ptr-i-1;
		}
		cout<<ans<<endl;
	}
	
	return 0;
	}
