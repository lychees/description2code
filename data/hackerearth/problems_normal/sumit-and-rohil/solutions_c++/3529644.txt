#include <iostream>
#include <bits/stdc++.h>

using namespace std;

struct entry{
	char start,end;
	string comp;
};

void mark(string s,vector<entry> &t){
	if(t.empty()){
		entry temp;
		temp.start = s[0];
		temp.end = s[s.size()-1];
		sort(s.begin(),s.end());
		temp.comp = s;
		t.push_back(temp);
		return;
	}
	int i = 0,size = t.size();
	for(i = 0;i < size;++i){
		if(s[0] == t[i].start && s[s.size()-1] == t[i].end){
			sort(s.begin(),s.end());
			if(s.compare(t[i].comp) == 0) return;
		}
	}
	entry temp;
	temp.start = s[0];
	temp.end = s[s.size()-1];
	sort(s.begin(),s.end());
	temp.comp = s;
	t.push_back(temp);
	return;
}

int sumit(vector<string> v){
	vector<entry> t;
	int i = 0,size = v.size();
	for(i = 0;i < size;++i){
		mark(v[i],t);
	}
	return t.size();
}

int main()
{
	int i = 0,n = 0;
	cin>>n;
	vector<string> v(n,"");
	for(i = 0;i < n;++i){
		cin>>v[i];
	}
	cout<<sumit(v);
    return 0;
}
