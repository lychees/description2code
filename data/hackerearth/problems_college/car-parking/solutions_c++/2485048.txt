#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	char arr[76666] = {0};
    	cin>>arr;
    	int l = strlen(arr);
    	int c = 0;
    	for(int i = 0;i<l;i++)
    	{
    		if(arr[i] == 'B')
    		{
    			if(i-1 >=0 && arr[i-1] == '-')
    			arr[i-1] = 'k';
    			//if(i+1 <l && arr[i+1] == '-')
    		//	arr[i+1] = 'k';
    			if(i-2 >=0 && arr[i-2] == '-')
    			arr[i-2] = 'k';
    		//	if(i+2 <l && arr[i+2] == '-')
    		//	arr[i+1] = 'k';
    		}
    		else if((arr[i] == 'S'))
    		{
    			if(i-1 >=0 && arr[i-1] == '-')
    			arr[i-1] = 'k';
    			if(i+1 <l && arr[i+1] == '-')
    			arr[i+1] = 'k';	
    		}
    	}
    	for(int i = 0;i<l;i++)
    	if(arr[i] == '-')
    	c++;
    	cout<<c<<endl;
    }
    return 0;
}
