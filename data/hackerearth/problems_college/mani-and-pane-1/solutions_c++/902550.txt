#include<iostream>
#include<math.h>
#include<cstdio>
using namespace std;
int gcd(int a,int b){
if(b==0)
    return a;
return gcd(b,a%b);
}
static int c[101];
int pre(){

for(int i=1;i<=100;i++){
    for(int j=i+1;j<=100;j++){
        for(int k=j+1;k<=100;k++){
                //if(i!=j&&j!=k&&i!=k)
                    c[gcd(i,gcd(j,k))]+=1;
        }
    }
}
return 0;
}
int main(){
pre();
int t;
int sum=0;
for(int i=1;i<=100;i++)sum+=c[i];
cin>>t;
long long a,b;
while(t--){
    cin>>a>>b;
    b=a-b;
    if(b<0)b=-b;
    //cout<<b<<endl;
    int ans=0;
    for(int i=1;i<=100;i++){
        if(b%i==0)
            ans+=c[i];
    }
    printf("%.6lf\n",(ans/(double)sum));
}

return 0;
}
