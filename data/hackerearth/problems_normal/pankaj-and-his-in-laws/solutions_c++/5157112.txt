int main()
{
    ll max=-1,a[siz],dp[1001],n,i,j,rem;
    string l;
    cin>>n;
    rep(i,n) cin>>a[i];
    rep(i,n) dp[i]=1;
    rep(i,n)
    {
    	rep(j,i)
    	{
    		if(a[i]>a[j] && dp[i]<(dp[j]+1))
    		{
    			dp[i]=dp[j]+1;
    		}
    		
    	}
    	if(dp[i]>max) max=dp[i];
    }
    while(max!=0)
	{
		rem=max%2;
		if(rem==0) l+='0';
		else l+='1';
		max/=2;
	}
	reverse(all(l));
	cout<<l;
    
    return 0;
}
