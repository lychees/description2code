int main()
{
        int t;
        cin>>t;

        while (t--) {
                int a1, b1, a2, b2;
                int x1, y1, x2, y2;

                cin >> a1 >> b1 >> a2 >> b2;
                cin >> x1 >> y1 >> x2 >> y2;

                if (x2<a1 || x1>a2 || b2<y1 || y2<b1)
                        PS("Apart\n");
                else if (b2 == y1) {
                        if (a1==x2 || a2 == x1)
                                PS("Point\n");
                        else
                                PS("Line\n");
                } else if (b1==y2) {
                        if (a1 == x2 || a2 == x1)
                                PS("Point\n");
                        else
                                PS("Line\n");
                } else if (a1 == x2) {
                        if (b2 == y1 || b1==y2)
                                PS("Point\n");
                        else
                                PS("Line\n");
                } else if (a2 == x1) {
                        if (y1 == b2 || y2 == b1)
                                PS("Point\n");
                        else
                                PS("Line\n");
                } else
                        PS("Rectangle\n");
        }

        return 0;
}
