for t in xrange(input()):
    x, y = map(int, raw_input().split())
    arr = [[1 for i in xrange(y)] for j in xrange(x)]
    c = 0
    for i in xrange(x):
        for j in xrange(y):
            if arr[i][j]:
                c += 1
                if i+2 < x and arr[i+2][j]:
                    arr[i+2][j] = 0
                if j+2 < y and arr[i][j+2]:
                    arr[i][j+2] = 0
    print c
