#include<bits/stdc++.h>
#define ll long long
using namespace std;
int main(){
 
 
    int t;
    cin >> t;
    for(int i = 1 ; i <= t ; ++i ){
 
 
       ll l , p , c;
       cin >> l >> p >> c;
       ll ct = 0;
       ll ti = l;
       while( ti < p )
        {
            ti *= c;
            ++ct;
        }
 
        ll ans=0;
        while((1<<ans)<ct)
            ++ans;
 
        printf("Case #%d: %d\n",i,ans);
 
 
 
    }
 
 
}
