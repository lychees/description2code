#include <iostream>
using namespace std;

int main()
{
    std::ios_base::sync_with_stdio(false);
    int t;
    cin>>t;
    while(t--)
    {
    	int n,a[1000001]={0},count=0,x,y;
    	cin>>n;
    	for(int i=0;i<n;i++)
    	{
    		cin>>x;
    		a[x]++;
    		cin>>y;
    		if(a[y]==0)
    		count++;
    		else a[y]=a[y]-1;
    	}

    	cout<<count<<endl;
    }
}
