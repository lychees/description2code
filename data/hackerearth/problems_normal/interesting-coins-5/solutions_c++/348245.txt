int64 a[1005];
int n;
bool chck(int64 x){
	int64 tmp=max((int64)1,a[1]-x);
	for(int i=2;i<=n;i++){
		if(a[i]+x<=tmp)
		return false;
		tmp=max(tmp+1,a[i]-x);
	}
	return true;
}
int main(){
	int t,i;
	scanf("%d",&t);
	while(t--){
		scanf("%d",&n);
		for(i=1;i<=n;i++)
		scanf("%lld",&a[i]);
		int64 lo=0;
		int64 hgh=1e9;
		int64 ans;
		while(lo<=hgh){
			int64 mid=(lo+hgh)/2;
			if(chck(mid)){
				ans=mid;
				hgh=mid-1;
			}
			else{
				lo=mid+1;
			}
		}
		printf("%lld\n",ans);
	}
	return 0;
}
