N, K = [int(x) for x in raw_input().split(' ')]
Raju = [int(x) for x in raw_input().split(' ')]
Rani = [int(x) for x in raw_input().split(' ')]
ang = set()

for n in Raju:
    i = 1
    num = n
    mul = set()
    while True:
        num = n*i % 360
        if num in mul:
            break
        ang.add(num)
        mul.add(num)
        i += 1

for n in Rani:
    if n not in ang:
        print('NO')
    else:
        print('YES')
