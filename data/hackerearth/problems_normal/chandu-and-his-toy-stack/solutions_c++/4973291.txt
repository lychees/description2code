#include <iostream>
using namespace std;
#include<bits/stdc++.h>
int main()
{
   int t;
   int a[1000000],b[1000000],i;
   scanf("%d",&t);
   while(t--)
   {
   	int n,x,y;
   	long long ans=0;
   	scanf("%d %d %d",&n,&x,&y);
   	for(i=0;i<n;i++)
   	{
   		scanf("%d %d",&a[i],&b[i]);
   	}
   	sort(a,a+n);
   	sort(b,b+n);
   	for(i=0;i<n;i++)
   	{
   		if(a[i]==b[i])
   		continue;
   		else if(a[i]>b[i])
   		ans+=(a[i]-b[i])*y;
   		else ans+=(b[i]-a[i])*x;
   	}
   	printf("%lld\n",ans);
   }
   return 0;
}
