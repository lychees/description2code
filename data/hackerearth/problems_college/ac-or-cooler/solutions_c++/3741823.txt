#include <iostream>
#include <algorithm>
using namespace std;

int ac[2*100002];
int main()
{
    int n,m;
    int i;
    cin>>n>>m;
    int a[100005];
    int c[100005];

    for(i=0;i<n;i++)
        cin>>a[i];
    for(i=0;i<n;i++)
        cin>>c[i];
    for(i=0;i<n;i++)
        ac[i]=a[i];
    for(;i<2*n;i++)
        ac[i]=c[i-n];
    sort(ac,ac+2*n);

    int s=0,k=0;
    for(i=0;i<2*n;i++)
    {
        if(ac[i]+s<=m && k<n)
            {
                k++;
                s=s+ac[i];
            }
        else
            break;
    }
    cout<<k<<endl;
    return 0;
}
