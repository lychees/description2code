#include <iostream>
#include<cstring>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s,p;
    	int i,j,count=0,k;
    	cin>>s;
    	cin>>p;
    	if(s.length()>p.length())
    	{
    		string l=s;
    		s=p;
    		p=l;
    	}
    	int sum=p.length();
    		for(i=0;i<=p.length()-s.length();i++)
    		{
    			k=i;
    			count=0;
    			for(j=0;j<s.length();j++)
    			{
    				if(p[k]!=s[j])
    				count++;
    				k++;
    			}
    			//	cout<<"count"<<count<<endl;
    				if(count<sum)
    				sum=count;
    		}
    		cout<<sum<<endl;
    	 }
    return 0;
}
