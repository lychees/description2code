#include<iostream>
#include<algorithm>
int a[100000 ];
int low,hi, x;

using namespace std;
int binary_search(int a[],int n,int k)
 {

 low=1;
 hi=a[n-1];
 while(low<hi)
  {
  x=low+(hi-low+1)/2;
  int required=1,current=a[0];
  for(int i=1;i<n;i++)
   {
   if(a[i]-current>=x)
    {
     ++required;
     current=a[i];
    }
    else
    {


    }
   }

   if(required<k)
   hi=x-1;
   else
   low=x;

  }
 return low;

 }



int main()


{
int t;
cin>>t;
while(t--)
 {
 int n,k,ans;
 cin>>n>>k;
 for(int i=0;i<n;i++)
    cin>>a[i];
 sort(a,a+n);
 ans=binary_search(a,n,k);
 cout<<ans<<"\n";


 }
}
