#include <iostream>
using namespace std;

int main()
{
    int t,n,i,j,o;
    int *a;
    
    cin>>t;
    while(t--){
    	cin>>n;
    	a = (int *)malloc(sizeof(int)*n);
    	for(i=0;i<n;i++){
    		cin>>a[i];
    	}
    	o=n;
    	
    	for(i=0;i<n-1;i++){
    		for(j=i+1;j<n;j++){
    			if(a[i]==a[j]){
    				o--;
    				break;
    			}
    		}
    	}
    	cout<<o<<endl;
    }
    return 0;
}
