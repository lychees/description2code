#include<iostream>
using namespace std;
#define MAX 1000000
bool v[MAX];
int sp[MAX]={0};

void Sieve(){
	sp[0]=0;
	sp[1]=0;
	for (long long int i = 2; i < MAX; i += 2)	sp[i] = 2;//even numbers have smallest prime factor 2
	for (long long int i = 3; i < MAX; i += 2){
		if (!v[i]){
			sp[i] = i;
			for (long long int j = i; (j*i) < MAX; j += 2){
				if (!v[j*i])	v[j*i] = true, sp[j*i] = i;
			}
		}
	}
	for(long long int i=2;i<MAX;i++)
	{
		if(sp[i]!=i)
		 sp[i]=0;
	}
}

int main(){
	Sieve();
	long int tc;
	cin>>tc;
	while(tc--)
	{
	 long int x,y;
	 cin>>x>>y;
	 long long int ans=0;
	 for(long int i=x;i<=y;i++)
	  ans+=sp[i];
	  cout<<ans<<endl;	   
	}
    return 0;
}
