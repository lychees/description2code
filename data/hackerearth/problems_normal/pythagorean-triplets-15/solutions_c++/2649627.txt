#include <iostream>
#include <cstring>
#include <cmath>
using namespace std;

int gcd(int a, int b)
{
	while (b != 0)  {
    int t = b;
    b = a % b;
    a = t;
  }
  return a;
}
int pytha_triplet(int n)
{
	int counter=0;
	int a,b,c,k;
	for(int i=2;i<=sqrt(n-1)+1;i++)
	{
		for(int j=1;j<i;j++)
		{
			if(gcd(i,j)==1 && (i-j)%2 && (i*i+j*j)<=n)
			{
				k=0;
				a=i*i-j*j;
				b=2*i*j;
				c=i*i+j*j;
				while((++k)*c<=n)counter++;
			}
		}
	}
	return counter;
}
int main()
{
	int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		cout<<pytha_triplet(n)<<endl;
	}
	//system("pause");
    return 0;	
}
