#include <bits/stdc++.h>
using namespace std;

int main() {
  int i,j,t;string s;int a[100002],sum=0;
    cin>>t;
    while(t--)
    {
    	cin>>s;
    	int n=s.length();
    	for(i=0;i<n;i++)
    	{
    		a[i]=s[i];
    	}
    	int str=0;
    	int end=n-1;
    	while(str<end)
    	{
    		sum=sum+abs(a[str]-a[end]);
    		str++;
    		end--;
    	}
    	cout<<sum<<endl;
    	sum=0;
    }
	return 0;
}
