#include <iostream>
using namespace std;

int main()
{
    int T;
    scanf("%d",&T);
    int flag;
    while(T--)
    {
    	flag=0;
    	char str[100005];
    	scanf("%s",&str);
    	for(int i=0;str[i]!='\0';i++)
    	{
    		if(str[i]=='a'||str[i]=='A'||str[i]=='e'||str[i]=='E'||str[i]=='o'||str[i]=='O'||str[i]=='i'||str[i]=='I'||str[i]=='u'||str[i]=='U') 
    		{
    			flag=1;
    			printf("%c",str[i]);
    		}
    	}
    	if(flag==0) printf("No");
    	printf("\n");
    }
    return 0;
}
