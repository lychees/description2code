#include <bits/stdc++.h>
using namespace std;

int main()
{
  	char str[80],rev[80];
  	int num;
  	cin>>num;
  	while(num--)
  	{
  		int i=0,c=0;
  	int j=0,flag=0;
  	cin>>str;
  	for(i=0;str[i]!='\0';i++)
  	c++;
  	//cout<<"c: "<<c<<endl;
  	for(i=c-1;i>=0;i--)
  	{
  		rev[j]=str[i];
  		j++;
  	}
  	//cout<<"rev: "<<rev[0<<endl;
  	rev[j]='\0';
  	for(i=1;i<=c-1;i++)
  	{
  		if((abs(str[i]-str[i-1])!=abs(rev[i]-rev[i-1])))
  		{
  			flag=1;
  			break;
  			
  		}
  	}
  	if(flag==1)
  		cout<<"Not Funny\n";
  	else
  		cout<<"Funny\n";
  	}
    return 0;
}
