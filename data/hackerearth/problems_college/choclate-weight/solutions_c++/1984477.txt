int main(void)
{
    _
    long long n,q;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++)
        cin>>a[i];
    sort(a,a+n);
    cin>>q;
    while(q--)
    {
        int w,l=0,r=n-1,flg=0;
        cin>>w;
        while(l<r)
        {
            if(a[l]==w||a[r]==w||a[l]+a[r]==w)
            {
                flg=1;
                break;
            }


            if(a[l]+a[r]<w)
                l++;
            else
                r--;
        }
        if(flg==1)
            cout<<"I Like You\n";
        else
            cout<<"I Hate You\n";
    }
    return 0;
}
