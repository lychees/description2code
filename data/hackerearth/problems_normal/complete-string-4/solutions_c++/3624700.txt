#include<iostream>
using namespace std;

int main()
{
	int test;
	cin>>test;
	for(int i=0;i<test;i++)
	{
	char a[100];
	cin>>a;
//	cout<<a<<endl;
	char alpha[25];
	for(int i=0;i<=24;i++)
	{
		alpha[i]=97+i;
	}
	alpha[25]='z';
	int count=0,flag=0,index=0;
	do
	{
		flag=0;
	for(int i=0;a[i]!='\0';i++)
	{
		if(a[i]==alpha[index] && flag==0)
		{
		count++;
		flag++;
		}
		
	}
	index++;
	}while(index!=25);
	if(count==25)
	cout<<"YES"<<endl;
	else
	cout<<"NO"<<endl;
}
}
