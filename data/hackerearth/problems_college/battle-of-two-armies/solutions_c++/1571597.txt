#include <iostream>
#include <math.h>

using namespace std;
int isPrime(int n){
    int i;

	if(n==1)
	return 0;
    if (n==2)
        return 1;

    if (n%2==0)
        return 0;

    for (i=3;i<=sqrt(n);i+=2)
        if (n%i==0)
            return 0;

    return 1;
}
int main()
{
	int t,n,prime[101],comp[101],ans;
	cin>>t;
	while(t--)
	{
		ans=0;
		cin>>n;
		for(int i=0;i<n;i++)
		cin>>comp[i];
		
		for(int i=0;i<n;i++)
		cin>>prime[i];
		
		for(int i=0;i<n;i++)
		{
			if(isPrime(prime[i])&&(!isPrime(comp[i])))
			{
				ans+=comp[i]-prime[i];
			}
			else if(isPrime(prime[i]))
			{
				ans+=0-prime[i];
			}
			else if(!isPrime(comp[i]))
			{
				ans+=comp[i]-0;
			}
			else
			{
				ans+=comp[i]-prime[i];
			}
		}
		if(ans<0)
		cout<<"The Primes"<<endl;
		else if(ans>0)
		cout<<"The Composites"<<endl;
		else
		cout<<"Tie"<<endl;
		
	}
	return 0;
}
