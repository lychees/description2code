long long int a[100001],arr[100008];
void sieve()
{
    bool isPrime[r];
    int j=0,i;
    int k=0;
    for(int i = 0; i <r;++i) {
        isPrime[i] = true;
    }
    isPrime[0] = false;
    isPrime[1] = false;
    for( i = 2; i * i <r; ++i) {
         if(isPrime[i] == true) {
             // Mark all the multiples of i as composite numbers
             for(int j = i * i; j < r ;j += i)
                 isPrime[j] = false;
        }
    }
    for(i=3;i<r;i++)
    	if(isPrime[i]==true)
    	{
    		arr[k]=i;
    		k++;
    	}
   // cout<<i<<"hey";
   
}
