#include <bits/stdc++.h>
using namespace std;

int main()
{
    long int tc;
    cin>>tc;
    while(tc--)
    {
    	long long int Rsum=0;
    	long int A,B;
    	cin>>A>>B;
    	if(A>=6)
    	{
    		Rsum=(B-A+1)*9;
    	}
    	else if(B>=6&&A<6)
    	{
    		Rsum=(B-5)*9;
    		B=5;
    		while(B>=A)
    		{
    			if(B==5)
    			{
    			Rsum=Rsum+3;	
    			}
    			if(B==4)
    			{
    			Rsum=Rsum+6;	
    			}
    			if(B==3)
    			{
    			Rsum=Rsum+6;	
    			}
    			if(B==2)
    			{
    			Rsum=Rsum+2;	
    			}
    			if(B==1)
    			{
    			Rsum=Rsum+1;	
    			}
    			if(B==0)
    			{
    			Rsum=Rsum+0;	
    			}
    			B--;
    		}
    	}
    	else if(A<6)
    	{
    		while(B>=A)
    		{
    			if(B==5)
    			{
    			Rsum=Rsum+3;	
    			}
    			if(B==4)
    			{
    			Rsum=Rsum+6;	
    			}
    			if(B==3)
    			{
    			Rsum=Rsum+6;	
    			}
    			if(B==2)
    			{
    			Rsum=Rsum+2;	
    			}
    			if(B==1)
    			{
    			Rsum=Rsum+1;	
    			}
    			if(B==0)
    			{
    			Rsum=Rsum+0;	
    			}
    			B--;
    		}
    	}
    	
    	cout<<Rsum<<endl;
    }
}
