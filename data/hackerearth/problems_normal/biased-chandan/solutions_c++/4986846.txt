#include <iostream>
using namespace std;

int main()
{
    long long int a[5005],n,t,top=-1,sum=0;
    
    cin>>n;
   
    for(int i=0;i<n;i++)
    {
    	cin>>t;
    	
    	if(t==0 && i!=0)
    		top--;
    	else
    	{
    		top++;
    		a[top]=t;
    	}
    }
    
    for(int i=0;i<=top;i++)
    	sum=sum+a[i];
    	
    cout<<sum;
    
    return 0;
}
