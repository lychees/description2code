#include <iostream>
using namespace std;

#include <stdio.h>

int main()
{
	long t,n,i,l,p,k;
	string ch;
    cin>>t;
    while(t--)
    {
    	
    	cin>>n>>ch;
    	l=ch.length();
    	
    	for(i=0;i<l;i++)
    	{
    		p=int(ch[i]);
    		k=p-n;
    		if(k>=65 && k<=90)
    		cout<<char(k);
    		else
    		cout<<char(k+26);
    	}
    	
    	cout<<endl;
    	
    }
    return 0;
}
