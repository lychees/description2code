from math import sqrt, ceil
r = (1 + sqrt(5)) / 2
for t in range(int(raw_input())):
	a1, a2, b1, b2 =  map(int,raw_input().split())
	ans = 0
	for i in range(a1, a2+1):
		up = i * r
		down = i / r
		if b1 >= up or b2 <= down:
			ans += b2 - b1 + 1
		else:
			if b1 <= down:
				ans += max(0, int(down) - b1 + 1)
			if b2 >= up:
				ans += max(0, b2 - int(ceil(up)) + 1)
	print "Case #%d: %d" % (t+1, ans)
