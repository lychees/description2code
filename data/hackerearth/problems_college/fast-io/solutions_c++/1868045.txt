#include <bits/stdc++.h>
using namespace std;
#define gc() getchar_unlocked()
inline long long int read_int() 
{
	char c = gc();
	while(c<'0' || c>'9') 
		c = gc();
	long long int ret = 0;
	while(c>='0' && c<='9')
	{
		ret = (ret<<3) + (ret<<1) + c - 48;
		c = gc();
	}
	return ret;
}
int main()
{
	long long int n,num,ctr;
	n = read_int();
	ctr = 0;
	while(n--)
	{
		num = read_int();
		if(num<30000000)
		{
			ctr++;
		}
	}
	printf("%lld\n",ctr);
	return 0;
}
