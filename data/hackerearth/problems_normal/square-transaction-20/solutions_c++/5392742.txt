#include <iostream>
using namespace std;

int main()
{
   int n,t,test;
   cin>>n;
   int a[n];
   cin>>a[0];
   for(int i=1;i<n;i++){
   	cin>>t;
   	a[i]=t+a[i-1];
   }
   cin>>test;
   while(test--){
   	int j,m;
   	cin>>m;
   	
   	for(j=0;j<n;j++){
   		if(m<=a[j]){
   			cout<<j+1<<endl;
   			break;
   		}
   	}
   	if(j>=n)
   	cout<<-1<<endl;
   
   	
   }
   
    return 0;
}
