#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int t, n, i;
    long long int sum, s, x[100000];
    long double m;
    bool b;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d", &n);
    	s=0;b=0;sum=0;
    	for(i=0;i<n;i++)
    	scanf("%lld",&x[i]);
    	scanf("%Lf", &m);
    	if(m==0)
    	printf("0\n");
    	for(i=0;i<n;i++)
    	{
    		s+=x[i];
    		if(s>=m)
    		{b=1;break;}
    	}
    	if(b==1)
    	{printf("%d\n", i+1);continue;}
    	sum=s;
    	sum=ceil(m/sum)*s;
    	sum=sum-s;
    	for(i=0;i<n;i++)
    	{
    		sum+=x[i];
    		if(sum>=m)
    		{b=1;break;}
    	}
    	printf("%d\n", i+1);
    }
    return 0;
}
