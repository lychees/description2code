#include <bits/stdc++.h>
using namespace std;
const long int SIZE=1e6+10;
int A[SIZE],Ans[SIZE];
void Seive()
{  
     for(int i=2;i<=1e6;i++)
    {

          if(A[i]==0)
          {   
              for(int j=2;j<=1e6;j++)
              {
                  if(i*j > 1e6)
                     break;
                  
                    A[i*j]=1;
        
              }
          }
    }
    

}
int checkPrime(int n)
{
    if(A[n]==0)
        return 1;
    else
        return 0;
}
int Solve(int n)
{  int flag=0;
    for(int i=2; i<=n/2; ++i)
    {
        
        if (checkPrime(i) == 1)
        {
            if (checkPrime(n-i) == 1)
            {
                flag = 1;
              break;
            }

        }
    }
return flag;
}

int main()
{ int t,x;
  cin>>t;
  memset(A,0,SIZE);
  Seive();
  while(t--)
  {
       cin>>x;
       if(Solve(x))
         cout<<"Deepa"<<endl;
       else
        cout<<"Arjit"<<endl;
  
  }

    return 0;
}
