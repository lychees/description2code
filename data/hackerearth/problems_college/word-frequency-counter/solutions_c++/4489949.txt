#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
using namespace std;
int main()
{
    int i,j=0;
    string s;
    getline(cin,s);
    transform(s.begin(),s.end(),s.begin(),[] (char c){return toupper(c);});

    vector<string> vec(s.size());
    vector<int> counts(vec.size());
   for(i=0;i<s.size();i++)
        {
        if(s[i]==' '){
            j++;i++;}
        vec[j].push_back(s[i]);
        }
    for(i=0;i<=j;i++)
    {
        for(int k=0;k<=j;k++)
        {
            if(vec[i]==vec[k])
            {
                counts[i]++;
                if(i!=k){
                vec[k].clear();
                }
            }
        }
        if(vec[i]!=""){
        cout << vec[i] << " " << counts[i]<<endl;}
    }
    vec.clear();
    return 0;

}
