#include <bits/stdc++.h>
//zone
using namespace std;
map<char,int> zone;
struct kuroko{
	bool operator()(const int &a, const int &b)
	{
		return zone[a] < zone[b];
	}
};
int main()
{
    int t;
    cin >> t;
    string s;
    string f;
    
    while(t--)
    {
    	cin >> s;
    	int i;
    	for(i = 0 ; i < s.length() ; i++)
    	{
    		zone[s[i]] = i + 1;
    	}
    	cin >> f;
    	sort(f.begin(),f.end(),kuroko());
    	cout << f <<endl;
    	zone.clear();
    }
}
