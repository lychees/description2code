#include <iostream>
#include <vector>
#include <algorithm> 
using namespace std;
int main()
{
	int n,i,zombie=0,vampire=0,a;
	vector<int> z,v;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>a;
		if(a%2==0)
		{
			zombie+=a;
			z.push_back(a);
		}
		else
		{
			vampire+=a;
			v.push_back(a);
		}
	}
	sort(z.begin(),z.end());
	sort(v.begin(),v.end());
	for(i=0;i<z.size();i++)
		cout<<z[i]<<" ";
	cout<<zombie<<" ";
	for(i=0;i<v.size();i++)
		cout<<v[i]<<" ";
	cout<<vampire<<" ";
	cout<<endl;
    return 0;
}
