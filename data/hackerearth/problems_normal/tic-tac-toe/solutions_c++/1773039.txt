#include<iostream>
#include<string>
#include<vector>
using namespace std;

typedef unsigned long long ull_t;
#define PRE_CPP do { ios_base::sync_with_stdio(0);	cin.tie(0); } while(0)

const ull_t MOD = 1e9+7;
const ull_t INV_6 = 166666668;

int main()
{
	PRE_CPP;
	int testcase;
	ull_t input, min, max, div;
	cin >> testcase;
	while (testcase--)
	{
		cin >> input;
		div = ((input - 1) / 2) % MOD;
		min = (((div * div) % MOD) * input) % MOD;
		max = ((((input -1) * input) % MOD) * ((2 * (input - 1)) + 1)) % MOD;
		max = (max * INV_6) % MOD;
		cout << min << " " << max << "\n";
	}
	return 0;
}
