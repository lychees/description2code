#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t,n;
    float d[30];float f[30];
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(int i=0;i<n;i++)
    	cin>>d[i];
    	float sum=0;float max=-1;int k=-1;
    	for(int i=0;i<n;i++)
    	{
    	cin>>f[i];
    	sum+=f[i]*d[i];
    	if(f[i]*d[i]>max){
    	max=f[i]*d[i];
    	k=i;
    	}
    	}
    	printf("%d %.3f\n",k,max/sum);
    }
    return 0;
}
