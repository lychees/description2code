#include <iostream>
#include<cstdio>
#include<cmath>
using namespace std;

int main()
{
    int t,r,x1,y1,x2,y2,x3,y3,x4,y4,x5,y5;
    double a ,b,c,d;
    cin >>t;
    while(t--)
    {
    	cin >>r>>x1>>y1;
    	cin >> x2>>y2;
    	cin >> x3>>y3;
    	cin >> x4>>y4;
    	cin >> x5>>y5;
	a=sqrt((x2-x1)*(x2-x1) + (y2-y1)*(y2-y1));
	b=sqrt((x3-x1)*(x3-x1) + (y3-y1)*(y3-y1));
	c=sqrt((x4-x1)*(x4-x1) + (y4-y1)*(y4-y1));
	d=sqrt((x5-x1)*(x5-x1) + (y5-y1)*(y5-y1));
	if(a<r&&b<r)
	{
		if(c<r&&d<r)
		{
			cout << "Yes"<<endl;
		}
		else
			cout << "No"<<endl;
	}
	else
		cout << "No"<<endl;
    }
    return 0;
    
}
