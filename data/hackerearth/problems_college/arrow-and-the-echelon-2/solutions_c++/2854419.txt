#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	int x,a,b;
	while(t--)
	{
		cin>>x;
		a=x;
		b=0;
		while(a>=0)
		{
			if(a%3==0 && b%5==0)
			{
				break;
			}
			a--;
			b++;
		}
		if(a==-1 && b==x+1)
		{
			cout<<"-1"<<endl;
		}
		else
		{
			for(int i=0;i<a;i++)
			{
				cout<<"5";
			}
			for(int i=0;i<b;i++)
			{
				cout<<"3";
			}
			cout<<endl;
		}
	}
}
