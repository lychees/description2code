#define DRT()     int t;cin>>t;while(t--)

int main()
{
      inp_s;
      {
            int n,q;
            cin>>n>>q;
            LL arr[n+1];
            arr[0]=0;
            FOR(i,1,n+1)
            {
                  cin>>arr[i];
                  arr[i]+=arr[i-1];
            }
            while(q--)
            {
                  int a1,a2,a3,a4;
                  cin>>a1>>a3>>a2>>a4;
                  if(a1>a3)
                  {
                        swap(a1,a3);
                        swap(a2,a4);
                  }
                  LL ans = 0;
                  if(a2 == a3)
                        ans = arr[a4+1]-arr[a1];
                  else if(a3>=a1 && a3<=a2 && a4>=a1 && a4<=a2)
                        ans = arr[a2+1]-arr[a1];
                  else if(a3>a2)
                        ans = arr[a2+1]-arr[a1] + arr[a4+1]-arr[a3];
                  else
                  {
                       // cout<<"HI"<<endl;
                        ans = arr[a3+1]-arr[a1] + arr[a2+1]-arr[a3+1] + arr[a4+1]-arr[a2+1];
                  }
                  cout<<ans<<endl;
            }

      }
      return 0;
}
