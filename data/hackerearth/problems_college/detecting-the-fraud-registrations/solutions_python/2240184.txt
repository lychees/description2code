n = int(raw_input())
name=[]
defaulties=[]
for i in range(n):
	temp=raw_input()
	if temp in name:
		if temp not in defaulties:
			defaulties.append(temp)
	else:		
		name.append(temp)
defaulties.sort()
print len(defaulties)
for index in range(len(defaulties)):
	print defaulties[index]+" "
