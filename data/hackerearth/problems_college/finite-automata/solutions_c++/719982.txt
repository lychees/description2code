#include <iostream>

using namespace std;

int main()
{
    int n, i, j;

    cin >> n;

    int a[n][3];

    for (i = 0; i < n; i++) {
        a[i][0] = i;
        a[i][1] = (i * 2) % n;
        a[i][2] = ((i * 2) + 1) % n;
    }

    for (i = 0; i < n; i++) {
        for (j = 0; j < 3; j++)
            cout << a[i][j] << " ";

        cout << endl;
    }

    return 0;
}
