#include <bits/stdc++.h>
using namespace std;
#define pb push_back
#define scan(x) scanf("%d",&x)
int main()
{
    int n,m;
    cin>>n>>m;
    vector<string> v;
    map<string,vector<pair<int,string>>>  ma;
    string temp;
    for(int i=0;i<n;i++)
    {
    	cin>>temp;
    	v.pb(temp);
    }
    string temp2;
    int tage;
    for(int i=0;i<m;i++)
    {
    	cin>>temp>>temp2>>tage;
    	ma[temp].push_back(make_pair(tage,temp2));
    }
    map<string,vector<pair<int,string>>>:: iterator it;
    for(it=ma.begin();it!=ma.end();it++)
    {
    	cout<<it->first<<"\n";
    //	pair<int,string>::iterator it2;
    	sort(ma[it->first].begin(),ma[it->first].end());
    	for(int j=0;j<ma[it->first].size();j++)
    	{
    		cout<<ma[it->first][j].second<<" "<<ma[it->first][j].first<<"\n";
    	}
    }
    
    return 0;
}
