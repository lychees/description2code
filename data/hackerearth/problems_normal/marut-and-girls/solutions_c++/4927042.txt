#include<iostream>
#include<vector>
#include<unordered_map>
#include<string>
using namespace std;
#define IOS 		ios_base::sync_with_stdio(false); cin.tie(NULL)
int main() {
  IOS;
  int m, n;
  unordered_map<int, bool> qualities;
  cin >> m;
  int a;
  for (int i = 0; i < m; i++) {
    cin >> a;
    qualities[a] = true;
  }
  cin >> n;
  int count = 0;
  cin.ignore();
  for (int i = 0; i < n; i++) {
    string line;
    getline(cin, line);
    line = line + " ";
    int num=0;
    int record = 0;
    for (int i = 0; i < line.size(); i++) {
      char x = line[i];
      if (x == ' ') {
        if (qualities[num] == true) {
          record++;
        }
        num = 0;
      }
      else {
        num = num * 10 + x - '0';
      }
    }

    if (record == m) {
      count++;
    }

  }
  cout << count << endl;


  return 0;
}
