#include <algorithm>
#include <iostream>
#include <utility>
#include <vector>
using namespace std;

int main()
{
    long long int n, c, q, time = 0;
    cin >> n >> c;

    vector <long long> a(n), ans(n), wait(n);
    vector <pair <long long, long long> > no(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i];
        if (a[i] % c != 0) no[i].first = (a[i]/c) + 1;
        else no[i].first = a[i]/c;
        no[i].second = i;

        wait[i] = time;
        time += min(c, a[i]);
    }

    sort(no.begin(), no.end());
    for (int i = 0; i < n; i++) ans[no[i].second] = i+1;

    cin >> q;
    while (q--) {
        int m, d;
        cin >> m >> d;
        if (m == 1) cout << wait[d-1] << endl;
        else cout << ans[d-1] << endl;
    }

    return 0;
}
