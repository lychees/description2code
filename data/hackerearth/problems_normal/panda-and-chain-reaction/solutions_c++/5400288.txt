#include<cstdio>
#define MOD 1000003
long long dp[MOD+11];
void init(){
    dp[0]=1;
    for(int i=1;i<=MOD;i++){
        dp[i]=(dp[i-1]*i)%MOD;
    }
}
int main(){
        init();
        int cs;
        scanf("%d",&cs);
        while(cs--){
            long long  n,x;
            scanf("%lld%lld",&n,&x);
            if(n>MOD)
                n=MOD;
            long long ans=x*dp[n]%MOD;
            printf("%lld\n",ans);
        }
        return 0;
}
