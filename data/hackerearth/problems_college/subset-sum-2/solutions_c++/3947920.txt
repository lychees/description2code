#include<bits/stdc++.h>
//link::https://www.hackerearth.com/the-ultimate-code/algorithm/subset-sum-2/
using namespace std;
long long int dp[10004];
int main()
{
    dp[0] = 1;
    int i;
    for(i = 1 ;i <= 2000 ; i++)
    {
        dp[i] = dp[i - 1]*2;
        dp[i] = dp[i]%1000000007;
    }
    int t;
    cin >> t;
    while(t--)
    {
        long long int ans = 0;
        int n;
        cin >> n;
        for(i = 1; i<=n ;i++)
        {
            ans = ans + i*dp[n-1];
            ans = ans % 1000000007;
        }
        cout<<ans<<endl;
    }
    return 0;
}
