#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int a, b, c;
        scanf("%d %d %d", &a, &b, &c);
        if(((a^b)^c) == 0)
            cout<<"BILL\n";
        else
            cout<<"NO BILL\n";
    }
    return 0;
}
