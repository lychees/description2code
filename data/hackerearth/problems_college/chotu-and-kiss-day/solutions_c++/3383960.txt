#include <iostream>


using namespace std;


int main()
{
 //   cout << "Hello World!" << endl;
 //  vector <int> spots;
   int n,i,j,sum,flag=0;
   cin>>n;
   int ar[n];
   for( i=0;i<n;i++){
   	cin>>ar[i];
  // 	spots.push_back(p);
   }
   for(i=0;i<n;i++){
			for(j=1;j<n;j++){
				if(ar[j]>ar[j+1]){
				int temp=ar[j];	
				ar[j]=ar[j+1];
				ar[j+1]=temp;
				}
		}}
   cin>>sum;
   flag=0;
   for( i=0;i<n-2;i++){
		for( j=i+1;j<n-1;j++){
			int x=sum-ar[i]-ar[j];
			int begin=j+1;
			int end=n-1;
		while(begin<=end){
			int mid=(begin+end)/2;
			if(ar[mid]==x){
			flag=1;
				break;
			}
			else if(ar[mid]<x)
			begin=mid+1;
			else
			end=mid-1;
		}}
   	
   }
   if(flag==1)
   cout<<"YES";
   else
   cout<<"NO";
    return 0;
}
