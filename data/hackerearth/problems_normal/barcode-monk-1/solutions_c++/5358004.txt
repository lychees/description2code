#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007

int arr[1000002];
vector<int>v;

int main()
{
	shark_s();
	return 0;
}
long long dp[1005][2];
int superbrush[1005];
void shark_s()
{
	superbrush[0]=1;
	int n,k;
	cin>>n>>k;
	for (int i = 0; i < k; ++i)
	{
		/* code */
		scanf("%d",&arr[i]);
		if(arr[i]>n)continue;
		for(int j=arr[i];j<=n;j++)
			superbrush[j] |= superbrush[j-arr[i]];
	}
	dp[0][0]=1;
	for(int i=1;i<=n;i++)
	{
		dp[i][0] = (dp[i-1][1]+dp[i-1][0])%mod;
		dp[i][1]=0;
		for(int j=1;j<=i;j++)
		{
			if(superbrush[j]){
				dp[i][1]=(dp[i][1]+dp[i-j][0])%mod;
			}
		}
	}
	cout<<(dp[n][0]+dp[n][1])%mod<<"\n";

	return ;
}
