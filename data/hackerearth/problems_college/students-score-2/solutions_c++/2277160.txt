#include <iostream>
using namespace std;

int main()
{
    int n,max=0,maxso=0;
    scanf("%d",&n);
    int arr[n];
    for(int i=0;i<n;i++)
    {
    	scanf("%d",&arr[i]);
    }
    for(int i=0;i<n;i++)
    {
    	max=max+arr[i];
    	if(max<0)
    		max=0;
    	if(maxso<max)
    		maxso=max;
    }
    printf("%d",maxso);
    return 0;
}
