#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int flag=0;
    for(int i=4;i<=n;i++)
    {
    	int num=i;
    	for(int j=2;j<=num/2;j++)
    	{
    		if(num%j==0)
    		{
    			flag=1;
    			break;
    		}
    		else
    		flag=0;
    	}
    	if(flag==1)
    	{
    		cout<<i<<endl;
    		flag=0;
    	}
    	else
    	flag=0;
    }
    return 0;
}
