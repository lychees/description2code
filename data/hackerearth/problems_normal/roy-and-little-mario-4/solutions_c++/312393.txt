#include<iostream>

using namespace std;

#define mod 1000000007LL


long long int cnt[100006];


int main()
{
	long long int test, n;

	cnt[1] = 1;
	cnt[2] = 2;
	cnt[3] = 4;
	for (int i = 4; i <= 100000; i++)
	{

		cnt[i] = (cnt[i - 1] + cnt[i - 2] + cnt[i - 3])%mod;

	}

	cin >> test;
	while (test--)
	{
		cin >> n;


		cout << cnt[n] %mod<< endl;

	}

	return 0;


}
