#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t,p=1;
    scanf("%d",&t);
    while(t--)
    {
    	long long n;
    	int c=0,q;
    	scanf("%llu",&n);
    	while(n>0)
    	{
    		q=n%2;
    		if(q==1)
    		c++;
    		n=n/2;
    	}
    	
    	if(p!=0)
    	{
    		if(c%2!=0)
    		{
    			printf("Sherlock\n");
    			p++;
    		}
    		else
    		{
    			printf("Watson\n");
    			p=0;
    		}
    		
    	}
    	else
    	{
    		if(c%2!=0)
    		{
    			printf("Watson\n");	
    		}
    		else
    		{
    			printf("Sherlock\n");
    			p=1;
    		}
    		
    	}
    	
    }
    return 0;
}
