#include <iostream>
using namespace std;
int calc(int n,int m)
{
	int res[n+1],i,j;
	res[0]=1;res[1]=1;
	for(i=2;i<n+1;++i)
	{
		res[i]=0;
		for(j=1;j<=m&&j<=i;++j)
		{
			res[i]+=res[i-j];
		}
	}
	return res[n];
}
int main()
{
    int n;
    cin>>n;
    if(n<1)
    {
    	cout<<"ERROR";
    }
    else
    {
    	int a=calc(n,3);
    int b=calc(n,2);
    cout<<"Jack-"<<a<<endl<<"Jill-"<<b<<endl;
    }
    
    return 0;
}
