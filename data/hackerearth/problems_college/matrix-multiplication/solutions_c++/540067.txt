#include <iostream>

using namespace std;

int main()
{
        int test_cases;
        scanf("%d",&test_cases);
        while(test_cases > 0)
        {
                int X1,X2,Y1,Y2;
                scanf("%d %d %d %d",&X1,&Y1,&X2,&Y2);
                
                if(Y1!=X2 && Y2!=X1) cout<<"0"<<endl;
                else if(Y1==X2 && Y2==X1) cout<<"2"<<endl;                
                else cout<<"1"<<endl;
                test_cases --;
        }
        return 0;
}
