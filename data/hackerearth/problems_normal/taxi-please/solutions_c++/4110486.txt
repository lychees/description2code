#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
class cust{
	public:
	long s,j;
	int cn;
};
bool stype(const cust &l,const cust &r){
	return l.s<r.s;
}
int main()
{
    int n,m,k;
    cin>>n>>m;
    int tn[n];
    long taxi[m];
    vector<cust> c(n);
    //cust c[n];
    for(int i=0;i<n;i++){
    	//c[i]=new cust();
    	cin>>c[i].s>>c[i].j;
    	c[i].cn=i;
    }
    for(int i=0;i<m;i++)
    taxi[i]=0;
    //qs(s,j,pn,0,n-1);
    sort(c.begin(),c.end(),stype);
    for(int i=0;i<n;i++){
    	for(k=0;k<m;k++){
    		if(taxi[k]<=c[i].s){
	    		taxi[k]=c[i].s+c[i].j;
	    		tn[c[i].cn]=k+1;
	    		break;
	    	}
    	}
    	if(k==m)
    	tn[c[i].cn]=-1;
    }
    for(int i=0;i<n;i++)
    cout<<tn[i]<<" ";
    return 0;
}
