using namespace std;

typedef vector<int> vi;
typedef pair<int,int> ii;
typedef long long int lli;

int maxWorkLoad(vi &weights,int m){
	int sum = 0;
	int max = INT_MIN;
	int length = weights.size();
	for(int i =0;i<length;i++){
		if(weights[i]>max)
			max=weights[i];
		sum+=weights[i];
	}
	int hi = sum,lo=max,x;
	int required,curLoad ;
	while(hi>lo){
		x = (hi+lo)/2;
		required = 1,curLoad = 0;
		for(int i =0;i<length;i++){
			if( (curLoad + weights[i])>x){
				required++;curLoad = weights[i];
			}
			else{
				curLoad+=weights[i];
			}
		}
		if(required<=m)
			hi = x;
		else
			lo = x+1;
	}
	return lo;
}
int main(){
	int m,p,tmp;
	cin>>m>>p;
	vi weights;
	for(int i=0;i<p;i++){
		cin>>tmp;
		weights.push_back(tmp);
	}
	cout<<maxWorkLoad(weights,m);
	return 0;
}
