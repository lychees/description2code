#include <iostream>
#include <stdio.h>
#include <algorithm>
using namespace std;

bool wayToSort(int i, int j) { return i > j; }
int main() {
	int T,n;
	cin>>T;
	while(T--) {
		scanf("%d",&n);
	 	int B[n],G[n];
		int i,match = 0;

		for(i=0;i<n;i++)
			scanf("%d",&G[i]);
		for(i=0;i<n;i++)
            scanf("%d",&B[i]);
		sort(G, G + n);
		sort(B,B+n,wayToSort);

		for(i=0;i<n;i++) {
			if((B[i] % G[i] ==0) || (G[i] % B[i] == 0)){
				match++;
			}
		}
		cout<<match<<endl;

	}


	return 0;
}
