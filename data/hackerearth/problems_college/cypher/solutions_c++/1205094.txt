#include <iostream>
#include<cstdio>
#include<cstring>
using namespace std;

int main()
{
	int t,i,n,temp;
	scanf("%d",&t);
	while(t--)
	{
		char s[105];
		scanf("%s",&s);
		n=strlen(s);
		for(i=0;i<n;i++)
		{
			temp=s[i]-'a';
			temp=(temp+i)%26;
			s[i]=temp+'a';
		}
		printf("%s\n",s);
	}
    return 0;
}
