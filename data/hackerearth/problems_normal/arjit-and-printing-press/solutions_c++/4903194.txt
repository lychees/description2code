#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    int countr;
    cin >> t;
    string s,r;
    vector<char> v;
    for(int p=0;p<t;++p)
    {
    	countr=0;
    	cin >> s;
    	cin >> r;
    	for(int i=0;i<r.size();++i)
    		v.push_back(r[i]);
    	sort(v.begin(),v.end());
    	for(int i=0;i<s.size();++i)
    	{
    		if(s[i]>v[countr])
    			{
    				s[i]=v[countr];
    				++countr;
    			}
    			if(countr>=r.size())
    			break;
    	}
    	cout << s << "\n";
    	v.clear();
    }
    return 0;
}
