#include<iostream>
using namespace std;
int main(){

int t;
cin>>t;
while(t--){
int M,N;
cin>>N>>M;
char win[N][M];
for(int i=0;i<N;i++){
    for(int j=0;j<M;j++){
        cin>>win[i][j];
    }
}
int sum=0;
for(int i=1;i<N;i++){
    for(int j=1;j<M;j++){
        if((win[i][j]=='/')&&(win[i-1][j]=='\\')&&(win[i-1][j-1]=='/')&&(win[i][j-1]=='\\'))
                  sum++;

    }
}
cout<<sum<<endl;
}

return 0;
}
