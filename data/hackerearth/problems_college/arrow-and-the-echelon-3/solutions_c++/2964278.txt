#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	int x = -1;
    	for(int i=n;i>=0;i--){
    		if(i%3 == 0 && (n-i)%5 == 0){
    			x = i;
    			break;
    		}
    	}
    	if(x == -1){
    		cout<<-1<<"\n";
    		continue;
    	}
    	for(int i=0;i<x;i++){
    		printf("5");
    	}
    	for(int i=x;i<n;i++){
    		printf("3");
    	}
    	printf("\n");
    }
    return 0;
}
