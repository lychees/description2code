#include <bits/stdc++.h>
using namespace std;

int main()
{
	long long t,n,rem,i,count=0,x,y,z,j,n1;
	cin>>t;
	while(t--)
	{
		count++;
		cin>>x>>y>>z;
		cout<<"Case "<<count<<": ";
		char a[2];
		j=3;
		while(j--)
		{
			i=0;
			if(j==2)
			n=x;
			else if(j==1)
			n=y;
			else
			n=z;
			n1=n;
			while(n1>0)
			{
				rem=n1%16;
				if(rem==15)
				a[i++]='Z';
				else if(rem==14)
				a[i++]='Y';
				else if(rem==13)
				a[i++]='X';
				else if(rem==12)
				a[i++]='W';
				else if(rem==11)
				a[i++]='V';
				else if(rem==10)
				a[i++]='U';
				else
				a[i++]=rem+48;
				n1=n1/16;
			}
			if(i%2)
			{
				cout<<"0"<<a[0]<<" ";
			}
			else
			{
				if(n!=0)
				cout<<a[1]<<a[0]<<" ";
				else
				cout<<"0"<<"0"<<" ";
			}
		}
		cout<<endl;
	}
    return 0;
}
