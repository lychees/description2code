# print 'Hello World!'
from fractions import gcd
from math import sqrt
t = input()
while t > 0:
    t -= 1
    n = input()
    c = 0
    for i in range(1, n+1):
        if gcd(i, n) == 1:
            c += 1
    if c == 1 or c == 0:
        print "Incorrect"
        continue
    for i in range(2, int(sqrt(c))+1):
        if c % i == 0:
            print "Incorrect"
            break
    else:
        print "Correct"
