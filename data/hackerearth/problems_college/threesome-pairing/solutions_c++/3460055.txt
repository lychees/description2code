
#include<bits/stdc++.h>
using namespace std;
int main()
{
	int tc;
	cin>>tc;
	while(tc--)
	{
		int n;
		cin>>n;
		int freq[]={0,0,0};
		for(int i=1;i<=n;++i)
		{
			int temp;
			cin>>temp;
			++freq[temp%3];
		}
		int ans=0;
		ans+=(freq[0]*(freq[0]-1))/2;
		ans+=(freq[1]*freq[2]);
		cout<<ans<<endl;
	}
	return 0;
}
