#include <bits/stdc++.h>
 
using namespace std;

int main(){
  int a, b;
  cin >> a >> b;
  string s;
  cin >> s;
  vector <int> ok;
  long long p[100001], ans = 0;
  int len = s.length(), count = 0;
  for(int i = 0; i < len; i++){
    if(count == 0){
      p[count] = a;
    } else{
      p[count] = (p[count - 1] * p[count - 1]) % b;
    }
    if(s[i] == '1'){
      ok.push_back(p[count]);
    }
    count++;
  }
  for(int i = 0; i < ok.size(); i++){
    ans = (ans + ok[i]) % b;
  }
  cout << ans << endl;
  return 0;
}
