#!/usr/bin/env python

n = input()
a = map(int, raw_input().split())

try:
    first = a.index(1)
except BaseException:
    print 0
    exit()

result = 1
while True:
    try:
        second = a.index(1, first + 1)
        result *= second - first
        first = second
    except BaseException:
        break
print result
