# print 'Hello World!'
n = int(raw_input())
a = []
while len(a) != n:
    a += map(int, raw_input().split())

a = a[::-1]
i = 0
ctr2 = 0
while i < len(a):
    if i == 0:
        ctr = a[i]
    else:
        ctr = ctr + a[i] - 1
    ctr2 += ctr
    # print "ctr", ctr, "a[i]", a[i]
    i += 1

# print a
print ctr2
