t = int(raw_input())
for i in xrange(t):
    line = raw_input().split(" ")
    n = int(line[0])
    k = int(line[1])
    if n == 1:
        print "Nandu"
    elif k == 1 and n % 2 == 1:
        print "Nandu"
    elif (n-1) % (k + 1) != 0:
        print "Rani"
    else:
        print "Nandu"
