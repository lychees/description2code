testCases = int(raw_input())
chars = 'abcdefghijklmnopqrstuvwxyz'
for i in range(testCases):
    s1 = raw_input()
    s2 = raw_input()
    isExist = False
    for letter in chars:
        if letter in s1 and letter in s2:
            print "Yes"
            isExist = True
            break
    if not isExist:
        print "No"
