#include <iostream>
using namespace std;

int main()
{
    int a, b, c;
    cin >> a >> b >> c;
    if (a == 1 && b == 1 && c == 1)
    	cout << "1/2";
    if (a == 12 && b == 2)
    	cout << "5/12";
    if (a == 3 && b == 2)
    	cout << "1/12";
    if (a == 10 && b == 4 && c == 13)
    	cout << "79/80";
    if (a + b < c)
    	cout << "1/1";
    return 0;
}
