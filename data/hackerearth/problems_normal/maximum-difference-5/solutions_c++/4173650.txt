void quicksort(long int a[],int p,int r);
int partition(long long int a[],int p,int r)
	{
		long long int x,i,j,s,t;
		x = a[r];
		i = p-1;
		for(j=p; j<r; j++)
			{
				if(a[j] <= x)
					{
						i++;
						s = a[i];
						a[i] = a[j];
						a[j] = s;
					}
			}
		t = a[i+1];
		a[i+1] = a[r];
		a[r] = t;
		return i+1;
	}
void quicksort(long long int a[],int p,int r)
	{
		if(p<r)
			{
				int q = partition(a,p,r);
				quicksort(a,p,q-1);
				quicksort(a,q+1,r);
			}
	}
int main()
{
	long long int a[1001];
	int N,t,i,j;

	for(i=0;i<N;i++)

		assert(a[i] >= 1 && a[i] <= 100000);}
	//quicksort(a,0,N-1);
	sort(a,a+N);
	i=0;
	j=N-1;
	vector<int> v1,v2;
	while(i<=j)
	{
		v1.push_back(a[i]);
		v2.push_back(a[j]);
		if(i!=j) v1.push_back(a[j]), v2.push_back(a[i]);
		i++,j--;
	}
	for(i=0; i<N-1; i++)
	{
		sum1 += abs(v1[i] - v1[i+1]),sum2 += abs(v2[i] - v2[i+1]);
	}
	//cout<<sum1<<sum2<<endl;
