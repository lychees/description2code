#include <iostream>
using namespace std;

int main()
{
    long long t,n,sum;
    cin>>t;
    while(t--)
    {
    	sum=0;
    	cin>>n;
    	for(int i=1;i<=(n-1)/2;++i)
    	{
    		sum+=i*(n-i);
    	}
    	sum*=4;
    	if(n%2==0)
    		sum+=2*(n/2)*(n/2);
    	cout<<sum<<"\n";
    		
    }
return 0;    
}
