#include <iostream>
using namespace std;
typedef long long int ll;
ll a[1000001];
int main()
{
	ll i, j, max = 1000000;
	for(i = 2; i*i <= max; i++) {
		if(a[i] == 0) {
			for(j = i*i; j <= max; j+=i) {
				a[j] = 1;
			}
		}
	}
	a[1] = 1;
	a[0] = 0;
	a[4] = 0;
	for(i = 1;i <= max; i++) {
		a[i]+=a[i-1];
	}
    ll t;
    cin >> t;
    while(t--) {
    	int l, r;
    	ll val;
    	scanf("%d %d", &l, &r);
    	val = a[r] - a[l - 1];
    	printf("%lld\n", val);
    }
    return 0;
}
