#include <iostream>
#include<string.h>
using namespace std;
long po(int i)
{
	long x=1;
	while(i--)
	{
		x*=10;
	}
	return x;
}
int main()
{
   	int t,i,temp;
   	long a,b;
   	char str1[9],str2[9];
   	cin>>t;
   	while(t--)
   	{
   		cin>>str1>>str2;
   		a=0;
   		b=0;
   		int len1=strlen(str1);
   		int len2=strlen(str2);
   		for(i=0;i<len1;i++)
   		{
   			temp=str1[i]-'0';
   			a+=temp*po(i);  
   		}
   		for(i=0;i<len2;i++)
   		{
   			temp=str2[i]-'0';
   			b+=temp*po(i);  
   		}
   		a+=b;
   		std::string str1 = std::to_string(a);
   		len1=str1.size();
   		a=0;
   		for(i=0;i<len1;i++)
   		{
   			temp=str1[i]-'0';
   			a+=temp*po(i);  
   		}
   		cout<<a<<endl;
   	}
    return 0;
}
