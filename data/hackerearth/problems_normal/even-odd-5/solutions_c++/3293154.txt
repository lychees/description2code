#include <iostream>
using namespace std;

int main()
{
    long int t,r,c,i,counteven,countodd,R,C,n;
    cin>>t;
    while(t--)
    {
    	cin>>r>>c;
    	counteven=0;
    	countodd=0;
    	R=r;
    	C=c;
    	if(R%2!=0)
    	{
    		n=((R-1)/2)+1;
    		countodd=(n*(1+R))/2;
    		n=((R-3)/2)+1;
    		counteven=(n*(2+R-1))/2;
    		if(R>C)
    		{
    			while(R>C)
    			{
    				countodd=countodd-(R-C);
    				//counteven=counteven-(R-C);
    				R-=2;
    			}
    		}
    		R=r-1;
    		if(R>C)
    		{
    			while(R>C)
    			{
    				//countodd=countodd-(R-C);
    				counteven=counteven-(R-C);
    				R-=2;
    			}
    		}
    		
    	}
    	else
    	{
    		n=((R-2)/2)+1;
    		countodd=(n*R)/2;
    		n=((R-2)/2)+1;
    		counteven=(n*(2+R))/2;
    		if(R>C)
    		{
    			while(R>C)
    			{
    				//countodd=countodd-(R-C);
    				counteven=counteven-(R-C);
    				R-=2;
    			}
    		}
    		R=r-1;
    		if(R>C)
    		{
    			while(R>C)
    			{
    				countodd=countodd-(R-C);
    				//counteven=counteven-(R-C);
    				R-=2;
    			}
    		}
    	}
    		
    	cout<<counteven<<" "<<countodd<<"\n";
    	
    }
    return 0;
}
