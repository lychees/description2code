marks_map={}
for i in range(int(raw_input())):
    name,marks=raw_input().split()
    marks=int(marks)
    if marks in marks_map.keys() :
        marks_map[marks].append(name)
    else:
        marks_map[marks]=[name]
for m in sorted(marks_map.keys(),reverse=True):
    for n in sorted(marks_map[m]):
        print n,m
