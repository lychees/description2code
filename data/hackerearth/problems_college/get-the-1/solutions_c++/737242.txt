#include <iostream>
using namespace std;

int main() {
	long a,b,c,d,e,f,g,h,sum;
	long abc[]={1,7,10,13,19,23,28,31,32,44,49,68,70,79,82,86,91,94,97};
	cin>>a;
	for(e=1;e<=a;e++)
	{
		cin>>b;
		c=b;
		sum=c;
		while(sum>100)
		{
		sum=0;
		if(c!=b)
		c=sum;
		while(c>0)
		{
			d=c%10;
			sum+=d*d;
			c/=10;
		}
		}
		if(sum==1)
		{
		 	cout<<"Case #"<<e<<":"<<"YES"<<endl;
		 	continue;
		}
		else
		{
			h=sizeof(abc)/sizeof(abc[0]);
			for(g=0;g<h;g++)
			{
				if(abc[g]==sum)
				{
				cout<<"Case #"<<e<<":"<<"YES"<<endl;
				break;
				}
			}
			if(g==h)
			cout<<"Case #"<<e<<":"<<"NO"<<endl;
		}
	}
	return 0;
}
