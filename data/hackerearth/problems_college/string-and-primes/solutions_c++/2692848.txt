#include <bits/stdc++.h>
using namespace std;
 
int main()
{
    long int n;
    cin>>n;
    while(n--)
    {
    	int i,j,m,z,flag=0,count=0;
    	
    	string str;
    	cin>>str;
    	
    	m=str.length();
    	for(i=0;i<m;i++)
    	{flag=0;
    	
    	  z=str[i];
    	  for(j=2;j<=sqrt(z);j++)
    	  {
    	  	if(z%j==0)
    	  	{
    	  		flag=1;
    	  		count++;
    	  		cout<<str[i];
    	  		break;
    	  	}
    	  	
    	  }
    	  
    	}
    
    	if(count==0)
    	{
    	cout<<"NULL"<<endl;	
    	}
    	else
    	cout<<endl;
    }
      return 0;
}
