#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
#define MOD 1000000007
LL mpow(LL a, LL n) 
{
    LL ret=1;
    LL b=a;
    while(n) 
    {
	if(n&1) 
	    ret=(ret*b)%MOD;
	b=(b*b)%MOD;
	n>>=1;
    }
    return (LL)ret;
}
LL ncr(LL n, LL k)
{
    LL i,j,num=1,den=1;
    for(i=k+1; i<=n; i++)
	num=(num*i)%MOD;
    for(i=2; i<=(n-k); i++)
	den=(den*i)%MOD;
    return (num*mpow(den,MOD-2))%MOD;
}
int main()
{
    LL a,b,n,k;
    cin >> a >> b >> n >> k;
    cout << ((ncr(n,k)*mpow(a,k))%MOD*mpow(b,n-k))%MOD << endl;
    return 0;
}
