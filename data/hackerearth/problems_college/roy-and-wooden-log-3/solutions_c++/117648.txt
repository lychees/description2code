#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
    int t,l,cuts,ways;
    cin >> t;
    while ( t-- ) {
          cin >> l;
          if ( l == 1 || l == 2 ) {
               cuts = 0;
          }
          else if ( l%2 == 0 ) {
               cuts = l/2 - 1;
          }
          else if ( l%2 != 0 ) {
               cuts = (l-1)/2;
          }
          ways = l/2 + 1;
          cout << ways << " " << cuts << endl;
    }
    return 0;
}
