#include <iostream>
#include <math.h>
#define FOR(i, a, b) for(int i = a; i < b; i ++)
#define FORE(i, a, b) for(int i = a; i <= b; i ++)
using namespace std;
int main(){
	int t = 0;
	cin>>t;
	unsigned long long int a[500];
	FOR(i, 0, t)
		cin>> a[i];
	FOR(i, 0, t)
	{
		unsigned long long int tong = 0;
		unsigned long long int max = sqrt(a[i]);
		unsigned long long int tg = a[i]/max;
		unsigned long long int ct = 0;
		unsigned long long int first;
		if(a[i] == 1)
		{
			tong = 1;
		}
		else if(a[i] == 2)
		{
			tong = 4;
		}
		else if(a[i] == 3)
		{
			tong = 8;
		}
		else
		{
			FOR(index, 1, tg)
			{
				first = (a[i])/(index);
				ct = a[i]/(index + 1) + 1;
				tong += (index) * (first + ct) * (first - ct + 1)/2;
			}
			FORE(j, 1, max)
			{
				tong +=  j*(a[i] / j);
			}
			tong = tong % 1000000007;
		}
		cout<<tong<<"\n";
	}
	return 0;
}
