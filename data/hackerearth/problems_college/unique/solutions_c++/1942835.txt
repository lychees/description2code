#include<iostream>
#include<algorithm>
#include<string.h>
using namespace std;

int main()
{
    int t;
    long long int n,count;
    cin>>t;
    long long int a[100001];
    while(t--)
    {
        count = 0;
        memset(a,0,sizeof(a));
        scanf("%lld", &n);
        for(long long int i = 0; i< n;i++)
        {
            scanf("%lld", &a[i]);

        }
        sort(a,a+n);
        for (int i=0; i<n; i++)
        {

           while (i < n-1 && a[i] == a[i+1])
              i++;

            count++;
        }
        cout<<count<<"\n";

    }
    return 0;
}
