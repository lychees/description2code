#include <algorithm>
#include <deque>
#include <iostream>
#include <math.h>
#include <queue>
#include <set>
#include <stack>
#include <string>
#include <vector>
using namespace std;
int main() {
  int n;
  char ch;
  string s[10002];
  bool used[10002];
  cin >> n;
  for (int i = 0; i < n; i++)
    cin >> s[i];
  cin >> ch;
  for (int i = 0; i < n; i++)
    s[i].push_back(ch);
  sort(s, s + n);
  for (int i = 0; i < n; i++)
    used[i] = false;
  int sum = 0;
  for (int i = 0; i < n; i++)
    sum += s[i].size();
  sum = (2 * sum) / n;
  for (int i = 0; i < n; i++) {
    if (used[i] == false) {
      used[i] = true;
      int k = sum - (s[i].size());
      for (int j = i + 1; j < n; j++) {
        if ((s[j].size() == k) && (used[j] == false)) {
          cout << s[i];
          for (int k = 0; k < s[j].size() - 1; k++)
            cout << s[j][k];
          cout << endl;
          used[j] = true;
          break;
        }
      }
    }
  }
  return 0;
}
