#include <iostream>
#include<vector>
#include<string>
using namespace std;

int main()
{
    int n;
    cin>>n;
    string input;
    vector<bool> stu(n,true);
    cin>>input;
    int len=input.size();
    int rem=n;
    int curr=0;
    int next=1;
    while(rem>1)
    {
    	for(int i=0;i<len;i++)
    	{
    		if(input[i]=='a')
    		
    		{
    			curr=next;
    			next=(next+1)%n;
    			while(stu[next]!=true)
    			{
    				next=(next+1)%n;
    			}
    		}
    		else if(input[i]=='b')
    		{
    			if(rem>1)
    			{
    				stu[curr]=false;
    			curr=next;
    			next=(next+1)%n;
    			while(stu[next]!=true)
    			{
    				next=(next+1)%n;
    			}
    			rem--;
    			}
    		}
    	}
    }
    cout<<curr+1<<endl;
    return 0;
}
