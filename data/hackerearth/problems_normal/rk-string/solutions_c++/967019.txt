#include<bits/stdc++.h>
using namespace std;
const int N=1e6+10;
long long mm[N][2]={{0}},f[N]={0};

long long ans(string s,int N,char c)
{
	long long sol=0;
	int d;
	int len=s.length();
	for(int i=0;i<len;i++)
	{
		if(i==0)
		{
			if(s[i]==c)
				f[i]=1;
			else
				f[i]=0;
			if(f[i]>=N)
			{
				d=f[i]-N;
				if(s[i]!=c)
				{
					sol+=mm[d][0];
					sol++;
				}
				else if(s[i]==c && N!=0)
				{
					sol+=mm[d][0];
					sol++;
				}
			}
			if(s[i]==c)
			mm[f[i]][1]++;
			else
			mm[f[i]][0]++;
		}
		else
		{
			if(s[i]==c)
			{
				f[i]=f[i-1]+1;
				
			}
			else
			{
				f[i]=f[i-1];
			}
			if(f[i]>=N)
			{
				d=f[i]-N;
				if(s[i]!=c)
				{
					sol+=mm[d][0];
					sol++;
				}
				else if(s[i]==c && N!=0)
				{
					sol+=mm[d][0];
					sol++;
				}
			}
			if(s[i]==c)
			mm[f[i]][1]++;
			else
			mm[f[i]][0]++;
		}
	}
	return sol;
}
int main()
{
	int t,m,n;
	scanf("%d",&t);
	while(t--)
	{
		string s;
		scanf("%d %d",&m,&n);
		cin>>s;
		char c='R';
		memset(mm,0,sizeof(mm));
		printf("%lld ",ans(s,m,c));
		c='K';
		memset(mm,0,sizeof(mm));
		printf("%lld\n",ans(s,n,c));
	}
}
