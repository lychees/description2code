no_quality = raw_input()
quality = raw_input().split(' ')
no_girls = raw_input()
girls_quality = {}

short_list = 0

for x in range(0, int(no_girls)):
    girls_quality[x] = raw_input()

def compare_quality(match):
    count = 0
    for x in match:
        if x in quality:
            count = count + 1
    if count == no_quality:
        return True
    else:
        return False


for girl in girls_quality:
    girl_quality = girls_quality.get(girl).split(' ')
    matching_quality = list(set(girl_quality) & set(quality))
    if len(matching_quality) == int(no_quality):
        short_list += 1

print short_list
