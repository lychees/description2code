	p=int(raw_input())
	if p is 0:
		print 'Invalid Test'
	else:
		try:
			while p>0:
				s=raw_input()
				l=len(s)
				if l >= 1 and l <= 100:
					a=sum(1 for i in s if i.isupper())
					b=sum(1 for i in s if i.islower())
					if a is 0 and b is 0:
						print 'Invalid Input'
					else:
						if a < b:
							print a
						else:
							print b
				else:
					print 'Invalid Input'
				p=p-1

			print 'Invalid Test'
