#include <iostream>
using namespace std;
int main()
{
	long long int arr[100000],n,k,i,j,temp;
	long long int max,posMax;
   cin>>n>>k;
   if(k>=n)
   {
   	k=n-1;
   }
   for(i=0;i<n;i++)
   {
   		cin>>arr[i];
   }
   for(i=0;i<k;i++)
   {
   		max=-1000000000;
   		for(j=i;j<n;j++)
   		{
   			if(arr[j]>max)
   			{
   				max=arr[j];
   				posMax=j;
   			}
   		}
   		if(posMax == i && k <n-1)
   		{
   			k++;
   		}else
   		{temp = arr[i];
   		arr[i] = arr[posMax];
   		arr[posMax]=temp;
   		}
   }
   for(i=0;i<n;i++)
   {
	   	cout<<arr[i]<<" ";
   }
   
    return 0;
}
