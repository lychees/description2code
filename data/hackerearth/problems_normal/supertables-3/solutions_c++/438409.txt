#include <iostream>
using namespace std;

#define LLI long long int

#define REP( i , a ) for ( int i = 0 ; i < (a) ; i++ )

#define DBG cout << "Line: " << __LINE__ << endl

LLI gcd(LLI a, LLI b)
{
	if (b==0)
		return a;
	else
		return gcd(b,a%b);
}

int main() {
	int t;
	LLI a , b , n , r , aa , bb , lcm , ans;
	cin >> t;
	while ( t-- )
	{
	    cin >> a >> b >> n;
	    if ( a > b )
	    {
	        r = a;
	        a = b;
	        b = r;
	    }
	    lcm = gcd (a , b);
	    lcm = (a/lcm)*b;
	    aa = 1;
	    bb = 100000000000000LL;
	    while ( aa <= bb )
	    {
	    	r = (aa + bb)/2;
			ans = r/a + r/b - r/lcm;
			if ( ans == n )
			{
				ans = max ( (r/a)*a , (r/b)*b );
				cout << ans << endl;
				break;
			}
			else if ( ans < n )
				aa = r+1;
			else
				bb = r-1;
	    }
	}
	return 0;
}
