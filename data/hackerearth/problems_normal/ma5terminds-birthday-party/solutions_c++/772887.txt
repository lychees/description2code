#include<iostream>
#include<vector>
#include<cstdio>

using namespace std;

vector<int> v(1000001,true);
int cot[1000001] = {0};

void prime()
{
     int i,j;
     v[2] = true;
     v[1] = false;
     for(i = 2; i <= 1000; i++)
     {
           if(v[i])
           {
                   for(j = 2 * i; j <= 1000000; j = j + i)
                   {
                         v[j] = false;
                   }
           }
     }
     v[4] = true;
}

int main()
{
     prime();
     int j,k,l,t;
     int r,i;
     for(i = 1; i <= 1000000; i++)
     {
           if(!v[i])
           {
                   cot[i] = cot[i - 1] + 1;
           }
           else
           {
               cot[i] = cot[i - 1];
           }
     }
     scanf("%d",&t);
     while(t--)
     {
           scanf("%d %d",&l,&r);
           printf("%d\n",cot[r] - cot[l - 1]);
     }
  //   system("pause");
     return 0;
}
