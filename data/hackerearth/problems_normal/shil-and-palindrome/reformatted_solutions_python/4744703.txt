s = raw_input()
s = ''.join(sorted(s))
middle = ""
str_ = ""
i = 0
found = 0
if len(s) == 1:
    print "%s" % s
    exit(0)
while (i < len(s)):
    if (i+1) < len(s) and s[i] != s[i+1]:
        middle = s[i]
        i += 1
        if found == 0:
            found = 1
        else:
            found = 2
            break
    else:
        str_ += s[i]
        i += 2
if found == 2:
    print "-1"
else:
    print "%s%s%s" % (str_, middle, str_[::-1])
