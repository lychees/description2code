#include <iostream>
#include<math.h>
using namespace std;
class de
{
	public:
		
	int ge(int n);
};
		int de:: ge(int n)
    			{
    				int sum1=0;
    				
    while(n!=0)
    	{
    		int r=n%10;
    		sum1=sum1+pow(r,2);
    		n=n/10;
          }
    	return sum1;
    			}
    			

int main()
{
	de ob;
    int t;
    cin>>t;
    int n;
    int r;
    int sum=0;
    int f=0;
    
    if(t>1&&t<1000)
    for(int i=1;i<=t;i++)
    {
    	cin>>n;
    	int c=0;
    	int f=0;
    	while(f>=0)
 {
 	
    int sum=ob.ge(n);
    {
    	if(sum==1)
    	{
    			cout<<"Case"<<" "<<"#"<<i<<":"<<"YES"<<"\n";
    		c++;
    	}
    	
    
    		else if(0<=sum&&sum<=9)
    		{
    				cout<<"Case"<<" "<<"#"<<i<<":"<<"NO"<<"\n";
    			c++;
    		}
    			
    			else
    			{
    			f++;
    				n=sum;
    			}
    			
    }
    if(c>0)

    }
}
