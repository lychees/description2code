from collections import Counter
import string
n = int(raw_input())
ls = list(string.lowercase)
t = 0
for i in range(n):
	st = raw_input().lower()
	ct = Counter(st)
	for i in ls:
		if not ct[i]:
			t = 1
			break
		t = 0
	if t:
		print "NO"
	else:
		print "YES"
