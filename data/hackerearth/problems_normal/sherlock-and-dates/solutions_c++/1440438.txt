#include<bits/stdc++.h>
using namespace std;

int a[12]={0,31,28,30,31,30,31,31,30,31,30,31};

int solve(int d,int m,int y) {
    if(y==0) {
        return 0;
    }
    int x=y/100;
    int res=max(x,0)*11;
    res+=max(0,min(13,y%100-1)-2);
    if(m>y%100-1&&y%100>=3) {
        res++;
    } else if(m==y%100-1&&d>=y%100-2&&y%100>=3) {
        res++;
    }
    return res;
}
bool leap(int y) {
    if(y%4==0) {
        if(y%100==0) {
            if(y%400==0) {
                return true;
            }
            return false;
        }
        return true;
    }
    return false;
}
int main() {
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        int d1,m1,y1,d2,m2,y2,ans=0;
        scanf("%d:%d:%d %d:%d:%d",&d1,&m1,&y1,&d2,&m2,&y2);
        if(d1==1) {
            if(m1==1) {
                d1=31;
                m1=12;
                y1--;
            } else {
                m1--;
                if(m1==2) {
                    if(leap(y1)) {
                        d1=29;
                    } else {
                        d1=28;
                    }
                } else {
                    d1=a[m1];
                }
            }
        } else {
            d1--;
        }
        printf("%d\012",solve(d2,m2,y2)-solve(d1,m1,y1));
    }
    return 0;
}
