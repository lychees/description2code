#include <bits/stdc++.h>
using namespace std;

typedef long long int int64;
int64 n,a[20],ans,vl;
void fn(int64 i,int64 j,int64 k,int64 ptr){
	if(ptr==n) {
		vl=max(max(i,j),k);
		if(vl<ans) ans=vl;
	}
	else{
		fn(i+a[ptr],j,k,ptr+1);
		fn(i,j+a[ptr],k,ptr+1);
		fn(i,j,k+a[ptr],ptr+1);
	}
}
int main(){
	int64 i;ans=1000000;
	cin>>n;
	for(i=0;i<n;i++){
		cin>>a[i];
	}
	fn(0,0,0,0);
	printf("%lld\n",ans);
	return 0;
}
