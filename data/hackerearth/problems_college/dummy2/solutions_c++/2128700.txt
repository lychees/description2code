#include <iostream>
#include<cstring>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t;
   cin>>t;
   while(t--)
   {
   	  string str;
   	  cin>>str;
   	  int i,j=0,k,p,s;
   	  string ans="";
   	  for(i=0;i<str.length();i++)
   	  {
   	  	if(str[i]>='0' && str[i]<='9')
   	  	{
   	  	    p=str[i]-'0';
   	  	    string temp=ans;
   	  	       for(k=0;k<p-1;k++)
   	  	  	   {
   	  	  		  ans=ans+temp;
   	  	  	   }
   	  	}
   	     else
   	  	  {
   	  	  	ans=ans+str[i];
   	  	  }
   	  	}
   	  	cout<<ans<<endl;
   	  }
   	  
   
    return 0;
}
