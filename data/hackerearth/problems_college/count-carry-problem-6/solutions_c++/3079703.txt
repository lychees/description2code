#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n;
   cin>>n;
   int i;

   for(i=1;i<=n;i++)
   {
       long long int a,b;
       cin>>a>>b;
       int coun=0;
       int ar[20],br[20],ac,bc;
       int j=0,k=0;
       while(a>0)
       {

           ar[j++]=a%10;
           a=a/10;
       }
       ac=j-1;
       while(b>0)
       {

           br[k++]=b%10;
           b=b/10;
       }
       bc=k-1;
       int l=0;
       while(l<=ac&&l<=bc)
       {

           if(ar[l]+br[l]>9)
            coun++;
            l++;
       }
      if(coun==0)
        cout<<"No carry operation"<<endl;
      else if(coun==1)
        cout<<coun<<" "<<"carry operation"<<endl;
      else
        cout<<coun<<" "<<"carry operations"<<endl;
   }
    return 0;
}
