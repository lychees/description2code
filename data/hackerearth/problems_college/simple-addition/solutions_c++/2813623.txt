int get(int small, int large) {
    if ( small == 1 ) return large-1;
    int x = large%small;
    if ( x == 0 ) return 999999;
    int diff = large - x;
    return (diff/small) + get(x,small);
}

int main() {
   // input();
    int t; cin >> t;
    while (t-- ) {
        int n; cin >> n;
        if ( n <= 2 ) {
            cout << n-1 << endl;
            continue;
        }
        int mn = 9999999;
        int lim = (n+1)/2;
        for ( int i = 1; i < lim; ++i ) {
            int rem = n-i;
            mn = min(mn,get(i,rem));
        }
        cout << mn+1 << endl;
    }
    return 0;
}
