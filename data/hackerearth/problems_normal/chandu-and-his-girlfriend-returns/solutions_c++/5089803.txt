// Problem: Chandu and his Girlfriend Returns

#include <iostream>
#include <algorithm>
#include <vector>
#include <cstring>

#define pb push_back
#define fastio ios::sync_with_stdio(false); cin.tie(0); cout.tie(0);
using namespace std;

int t, n, m, x;
vector<int> vect;
int main() {
	fastio;
	cin >> t;
	while(t--) {
		cin >> n >> m;
		vect.clear();
		for(int i = 0; i < n + m; i++) cin >> x, vect.pb(x);
		sort(vect.begin(), vect.end(), greater<int>());
		for(int i = 0; i < n + m; i++)
			if(i == 0) cout << vect[i];
			else cout << " " << vect[i];
		cout << endl;
	}
}
