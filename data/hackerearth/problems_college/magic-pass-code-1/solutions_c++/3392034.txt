#include <iostream>
using namespace std;

int main() {
	string line;
	int max = 0;
	string m;
	while (getline(cin, line)) {
		//cout << line << endl; 
		int sum = 0;
		for (int i = 0; i < line.size(); i++) {
			sum += line[i] - '0';
			//cout << sum << " ";
		}
		if (sum >= max) {
			max = sum;
			sum = 0;
			m = line;
		}
		//cout << endl;
		
	}
	cout << m << endl;
	return 0;
}
