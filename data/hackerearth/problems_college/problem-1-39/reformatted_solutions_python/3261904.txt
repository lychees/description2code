t = int(raw_input())
for i in range(t):
    a, b, c = tuple(map(int, raw_input().split()))
    m = a ** b
    if (m < 1):
        print(0)
    else:
        m = a ** b
        r = m % c
        p1 = m - r
        p2 = p1 + c
        if ((p2 - m) > (m - p1)):
            print(int(p1))
        else:
            print(int(p2))
