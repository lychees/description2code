#include<iostream>
using namespace std;
int main()
{
    int t=0, n, i, j;
    cin>>t;
    short a[10001];
    while(t--){
        cin>>n;
        for(i=0;i<=10000;i++)
            a[i]=0;
        for(i=0;i<n;i++){
            cin>>j;
            a[j]=1;
        }
        j=0;
        for(i=1;i<=10000;i++)
            if(a[i]==1)
                j++;
        cout<<j<<endl;
    }
    return 0;
}
