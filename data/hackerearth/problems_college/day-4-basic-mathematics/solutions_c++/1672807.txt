#include <iostream>
using namespace std;

void printMatch(int num) {
	int r = num % 3;
	if(r == 0) {
    	cout<<"SL"<<endl;
    }
   	else if(r == 1) {
   		cout<<"LB"<<endl;
    }
    else {
    	cout<<"BS"<<endl;
    }
}

int main()
{
    int T; cin>>T;
    
    while(T--) {
    	
    	int day = 0; cin>>day;
    	int sum1 = 0, sum2 = 0;
    	for(int i = 0; true; i++) {
    		sum1 = (i * ( i + 1)) / 2;
    		sum2 = ((i + 1) * ( i + 2)) / 2;
    		if(day > sum1 && day <= sum2) {
    			printMatch(i + 1);
    			break;
    		}
    	}
    	
    	
		
    	
    }
    
    return 0;
}
