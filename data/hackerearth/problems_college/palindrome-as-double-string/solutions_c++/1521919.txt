#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int T,t,num,i=0;
    vector<int> N;
    cin>>T; t=T;
    while(t--){
    	cin>>num;
    	N.push_back(num);
    }
    for(i=0;i<T;i++){
    	if(N[i]%2==0)
    	cout<<N[i]<<endl;
    	else
    	cout<<N[i]-1<<endl;
    }
    
    return 0;
}
