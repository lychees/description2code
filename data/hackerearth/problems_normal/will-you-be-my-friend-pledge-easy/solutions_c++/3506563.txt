#include <iostream>
using namespace std;
long int hcf(long int a,long int b)
{
	if(a%b==0)
	  return b;
	  
    return hcf(b,a%b);
	
	
}


int main()
{
	long int max,n,i,a[1001],c;
    cin>>max;
    
    cin>>n;
    c=0;
    for(i=0;i<n;i++)
     { cin>>a[i];
       //cout<<hcf(max,a[i])<<" ";
       if(hcf(max,a[i])!=1)
       c++;
    
     }
    cout<<c<<"\n";
    
    return 0;
}
