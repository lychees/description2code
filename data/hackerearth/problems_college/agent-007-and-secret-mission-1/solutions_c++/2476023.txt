{
    int t,n,temp;
    string str;
    cin>>t;
    while(t--)
    {
    	cin>>str>>n;
    	temp = n + 1;
    	if(n & 1)
    	{
    		temp = temp/2;
    		temp = temp * n;
    	}
    	else
    	{
    		n = n/2;
    		temp = temp * n;
    	}
    	if(temp & 1)
    	{
    	 n = count(str.begin() , str.end() , '0' ); 
    	
    	}
    	else n = count(str.begin(),str.end(), '1');
    	cout<<n<<endl;
    }
    return 0;
}
