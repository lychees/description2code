#include <bits/stdc++.h>
using namespace std;
bool a[1000005];
int main()
{
    int t,n,s,r;
    cin>>t;
    while(t--)
    {
    	memset(a,0,sizeof(a));
    	cin>>n;
    	int ans=INT_MIN,count=0;
    	for(int i=0;i<n;i++)
    	{
    		cin>>s>>r;
    		if(r==0)
    		{
    			ans=max(ans,count);
    			count=0;
    		}
    		else
    		{
    			if(a[s]==0)
    			{
    				count++;
    				a[s]=1;
    			}
    		}
    		ans=max(ans,count);
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
