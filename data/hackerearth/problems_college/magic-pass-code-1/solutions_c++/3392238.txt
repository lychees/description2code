#include <iostream>
using namespace std;


int sod(int n){
	if(n==0)
	return 0;
	return (n%10 + sod(n/10));
}
int main()
{
    int n;
    int max=0;
    int maxdigit=0;
    while(cin>>n){
    	if(sod(n)>max){
    		max = sod(n);
    		maxdigit = n;
    	}
    }
    cout<<maxdigit<<endl;
    return 0;
}
