#include<bits/stdc++.h>
#define ll long int
using namespace std;
int main()
{
	ll t,k,i,j,len;
	ll arr[26];
	arr[0]=1;
	for(i=1;i<26;i++)
	{
		arr[i]=arr[i-1]+1;
	}
	for(cin>>t;t;t--)
	{
		string str;
		cin>>k;
		cin>>str;
		len=str.size();
		ll curr_sum=arr[str[0]-97],cnt=0,start=0;
		for(i=1;i<=len;i++)
		{
			while(curr_sum>k && start<i-1)
			{
				curr_sum=curr_sum-arr[str[start]-97];
				start++;
			}
			if(curr_sum==k)
			{
				cnt++;
			}
			if(i<len)
			{
				curr_sum=curr_sum+arr[str[i]-97];
			}
		}
		cout<<cnt<<endl;
	}
}
