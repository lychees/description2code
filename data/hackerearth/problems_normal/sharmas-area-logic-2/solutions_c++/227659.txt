#include <iostream>
using namespace std;

int main()
{
   int t ;
   cin>>t ;
   while( t-- ){
       int minx = 100 , miny = 200 , maxx = -100 , maxy = -200 ;	
       int n ;
       cin>>n;
       while( n-- ){
   	   int x ,y ;
   	   cin>>x>>y ;
   	   if( x < minx   )
   	      minx = x ;
   	   if( x > maxx )
   	       maxx = x ;
   	   if( y > maxy )
   	       maxy = y ;
   	   if( y < miny )
   	       miny = y ;
       }
       if( minx == maxx )
          cout<<maxy-miny<<endl ;
       else if( miny == maxy )
          cout<<maxx-minx<<endl ;
       else 
          cout<<(maxx-minx)*(maxy-miny)<<endl ;
       
   	
   }
    return 0;
}
