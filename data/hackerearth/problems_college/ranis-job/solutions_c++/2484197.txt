#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
     int n,i;
     long int a[100005],b[100005],c[100005];
    while(t--)
   { cin>>n;
   	long int s=0;
   	for(i=0;i<n;i++)
   	 { cin>>a[i]; s+=a[i]; }
   	for(i=0;i<n;i++)
   	 cin>>b[i];
   	sort(b,b+n);
   int c=0;
   for(i=n-1;i>=0;i--)
   { if(s<=0)
     break;
   	s=s-b[i];
   	c++;
   	}	
   cout<<c<<endl;
   } 
    return 0;
}
