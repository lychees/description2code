#include <iostream>
using namespace std;
int main(){
	int i,j,t,n,count;
	int a[1000];
	cin>>t;
	while(t--){
		cin>>n;
		for(i=0;i<n;i++){
			cin>>a[i];
		}
		count=0;
		for(i=0;i<n-1;i++){
			for(j=i+1;j<n;j++){
				if((a[i]+a[j])%2==0 && a[i]!=a[j]){
					count++;
				}
			}
		}
		cout<<count<<endl;
	}
	return(0);
}
