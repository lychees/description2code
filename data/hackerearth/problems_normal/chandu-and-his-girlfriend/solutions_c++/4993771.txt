#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    int arr[100000];
    int n,i,t;

    while(t)
    {

			cin>>n;
            for(i=0;i<n;i++)
        {

            cin>>arr[i];

        }
       // cout<<"Unsorted:";
        sort(arr,arr+n,greater<int>());
       // cout<<"sorted:";
        int j;
        for(j=0;j<i;j++)
        {

            cout<<arr[j]<<" ";
        }
        cout<<endl;

    }



}
