#include<bits/stdc++.h>
using namespace std;
#define FOR(i,N) for(int i=0;i<(N);i++)
const int maxn = 100001;
int arr[maxn] = {0};
int arr1[maxn] = {0};
int ans[maxn] = {0};
vector<int> p;
int count(int n){
	
	int cc = 0;
	while(n){
		cc += n%10;
		n = n/10;
	}
	return cc;
}


void sum(){
	arr[0] = 0; arr[1] = 1;
	for(int i=1;i<maxn;i ++){
		if(i%10 == 0){
			arr[i] = count(i);
		}
		else{
			arr[i] = arr[i-1] + 1;	
		}
	}
}

void siv(){
	arr1[1] = arr1[0] = 1;
	for(int i=2;i<=sqrt(maxn); i++){
		if(arr1[i] == 0){
			for(int j = i*i;j<maxn;j +=i)
				arr1[j] = 1;
		}
	}
	for(int i=0;i<maxn;i++){
		if(arr1[i] == 0) p.push_back(i);
	}
}
	
void prime(){

	ans[1] = 1;
	ans[2] = 1;
	ans[3] = 1;

	for(int i=4;i<maxn;i++){
		int cc = 0;
		int temp =i;
		int k = 0;
			//cout<<i<<endl;
			if(arr1[i] == 0){
				ans[i] = 1;
				continue;
			}
		while(temp != 1){
			if(temp%p[k] == 0){
			cc += arr[ p[k] ];
			temp = temp/p[k];
			}
			else
		 	k++;
		}
	
		if(cc == arr[i]){
			ans[i] = 1;
		}	
	}
//	for(int i=2;i<11;i++) cout<<i<<"\t"<<ans[i]<<endl;
}


int main(){
	sum();
	siv();
	prime();
	for(int i=2;i<maxn;i++){
		ans[i] += ans[i-1];
	}
	int t; cin>>t;
	while(t--){
		int l,r;
		cin>>l>>r;
		cout<<ans[r] - ans[l-1]<<endl; 
	}
		
	
return 0;
}
