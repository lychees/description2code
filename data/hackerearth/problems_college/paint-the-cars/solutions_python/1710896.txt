def min_color_rate(num,arr):
	dp =arr
	
	for i in range(1,num):
		for j in range(3):
			dp[i][j]  = min(arr[i][j] + arr[i-1][(j+1)%3],arr[i][j] + arr[i-1][(j+2)%3])
	#print dp
	return min(dp[num-1])
