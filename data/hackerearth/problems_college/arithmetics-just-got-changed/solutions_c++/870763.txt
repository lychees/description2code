#include <iostream>
using namespace std;

int main()
{
	int T;
    unsigned a,b,sr,ar,br,srr;
    cin>>T;
    while(T--){
    	cin>>a>>b;
    	ar=0;
    	br=0;
    	srr=0;
    	while(a){
   		 ar = ar*10+a%10;
   		 a/=10;
  		}
  		while(b){
   		 br = br*10+b%10;
   		 b/=10;
  		}
  		sr=ar+br;
  		while(sr){
   		 srr = srr*10+sr%10;
   		 sr/=10;
  		}
  		cout<<srr<<endl;
    }
    return 0;
}
