#include <iostream>
#include<cstdio>
#include<cstdlib>
using namespace std;

int n, k, arr[16];
bool flag;
void check(int p, int sum)
{
	if(flag)
	return;

	if(sum==k)
	{
		flag=true;
		return;
	}
	
	if(p>=n)
	return;
	
	if(sum==-1e9)
	check(p+1, arr[p]);
	else
	check(p+1, sum+arr[p]);
	
	check(p+1, sum);
}
int main()
{
	int t;
	scanf("%d", &t);
	while(t--)
	{
		flag=false;
		scanf("%d", &n);
		for(register int i=0; i<n; i++)
		scanf("%d", &arr[i]);
		scanf("%d", &k);
		if(k==0)
		flag=true;
		else
		check(0, -1e9);
		if(flag)
		puts("YES");
		else
		puts("NO");
	}
    
    return 0;
}
