#include <iostream>
#include <cstdlib>
#include <string>
#include <sstream>
#include <algorithm>
using namespace std;

int main()
{
	string s1,s2;
	int n;
	vector <int> c1,c2;
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>s1;
		s2=s1;
		std::reverse(s2.begin(), s2.end());
		for(int i=0;i<s1.length()-1;i++){
			c1.push_back(abs(s1[i+1]-s1[i]));
			c2.push_back(abs(s2[i+1]-s2[i]));
		}			
		c1 == c2 ? cout<<"Funny\n" : cout<<"Not Funny\n";
		c1.clear();
		c2.clear();
	}
    return 0;
}
