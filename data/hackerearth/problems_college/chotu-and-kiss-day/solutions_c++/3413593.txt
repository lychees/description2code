#include <iostream>
using namespace std;
#include <algorithm>
#define ll long long
int main()
{
    ll int n;
    scanf("%lld",&n);
    ll int a[n],i;
    for(i=0;i<n;i++)
    	scanf("%lld",&a[i]);
    ll int sum;
    ll int r,l;
    ll int flag=0;
    scanf("%lld",&sum);
    std::sort(a,a+n);
    for(i=0;i<n;i++)
    {
    	l=i+1;
    	r=n-1;
    	while(r>l){
    		if(a[r] + a[l] + a[i]==sum){
    			flag=1;
    			goto qwe;
    		}
    		else if(a[r] + a[l] + a[i]>sum)
    			r--;
    		else if(a[r] + a[l] + a[i]<sum)
    			l++;
    	}
    }
    qwe:
    	if(flag)
    		printf("YES\n");
    	else
    		printf("NO\n");
    return 0;
}
