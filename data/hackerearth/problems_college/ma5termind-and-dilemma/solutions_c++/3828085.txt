#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n, i, zeroes = 0, ones = 0;
		char str[100005];
		scanf("%d%s",&n,str);
		for( i = 0 ; i < n ; i++ )
		{
			if(str[i] == '0')
				zeroes++;
			else
				ones++;
		}
		int f, b;
		char tmp[100005];
		int k = 0;
		while( zeroes )
		{
			tmp[k++] = '0';
			zeroes--;
		}
		while( ones )
		{
			tmp[k++] = '1';
			ones--;
		}
		tmp[k] = '\0';
		f = 0;
		for( i = 0 ; i < n ; i++ )
		{
			if( str[i] != tmp[i] )
				f++;
		}
		b = 0;
		for( i = n - 1 ; i >= 0 ; i-- )
		{
			if( str[ n - i - 1] != tmp[i] )
			{
				b++;
			}
		}
		int ans = min(f/2, b/2);
		printf("%d\n",ans);
	}
	return 0;
}
