#include<stdio.h>
#include<bits/stdc++.h>
using namespace std;
int main()
{	
	int n;
	scanf("%d",&n);
	string ch;
	map< string , int >m;
	for(int i=0;i<n;i++)
	{
		cin>>ch;
		m[ch]++;
	}
	map< string ,int > ::iterator it;
	it=m.begin();
	while(it!=m.end())
	{
		cout<<it->first;
		printf(" ");
		cout<<it->second;
		printf("\n");	
		it++;
	}
	
return 0;
}
