#include <bits/stdc++.h>
using namespace std;

int gcd(int a,int b) {
	if(a==0)
	return b;
	return (gcd(b%a,a));
}
int main() {
	int dp[101];
	memset(dp,0,sizeof(dp));
	int i,j,k;
	for(i=1;i<=100;i++) {
		for(j=1;j<=100;j++) {
			int v = gcd(i,j);
			for(k=1;k<=100;k++) {
				if(i!=j&&i!=k&&j!=k)
				dp[gcd(v,k)]++;
			}
		}
	}
	int t;
	scanf("%d",&t);
	while(t--) {
		long long int q,w;
		scanf("%lld%lld",&q,&w);
		q = abs(w-q);
		int x = 0;
		for(i=1;i<=100;i++) {
				if(q%i==0)
				x = x + dp[i];
			}
	double x1 = (double)x;
	printf("%0.6lf\n",(x1)/(100*99*98));
	}
	return 0;
}
