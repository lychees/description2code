#include <iostream>
using namespace std;

int main()
{
	int T,t;
        
        int ar[100][100];
        cin>>T;
        for(int i=0; i<T; i++){
            for(int j=0; j<3; j++){
                cin>>ar[i][j];
            }
        }
        
        for(int i=0; i<T; i++){
            if(ar[i][1] < ar[i][2])
                cout<<"Ship's width cannot be more than the number of cells in a row.";
            if(ar[i][1] == ar[i][2]){
                t=ar[i][2];
            }
            else if((ar[i][1]-ar[i][2]) <= ar[i][2]){
                t=ar[i][2]+1;
            }
            else{
                t = (ar[i][2]+(ar[i][1]-ar[i][2]-ar[i][2])+1)*ar[i][0];
            }
            cout<<"Case #"<<i+1<<": "<<t<< endl;
        }
    
    return 0;
}
