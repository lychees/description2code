#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
int main() {
	int t,n, p;
	ios_base::sync_with_stdio(false);
	cin >> t;
	while(t--){
		cin >> n >> p;
		int dp[n+1];
		memset(dp, 0 , sizeof dp);
		dp[0] = 1;
		for(int i = 1; i <= n; i++) {
			for(int j = i-1; j>=0 && i-j <= p; j--){
				dp[i] += dp[j];
				if(dp[i] > mod)
					dp[i] %= mod;
			}
		}
		cout << dp[n] << endl;
		
	}
}
