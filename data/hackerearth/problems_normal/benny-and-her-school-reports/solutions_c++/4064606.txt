#include <iostream>
using namespace std;

int main()
{
    long long n,t,a[100000],i=0,m,x,sum=0,sum1,sum2;
    cin>>t;
    while(t)
    {
    	cin>>n>>m;
    	cin>>x;
    	sum=0;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    		sum+=a[i];
    	}
    	sum1=m+sum;
    	sum2=(n+1)*x;
   		if(sum>=sum2)
    	cout<<1<<endl;
	    else if(sum2>sum1)
	    cout<<"Impossible"<<endl;
	    else
	  	cout<<sum2-sum<<endl;
	  	t--;
	}
	 return 0;
}
