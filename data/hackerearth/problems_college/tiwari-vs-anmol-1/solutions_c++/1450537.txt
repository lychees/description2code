#include<iostream>
using namespace std;
#define MAX 1000001
int arr[MAX]={0};
void pre()
{
	int i,j;
	for(int i=2;i<MAX;i++)
	{
		if(arr[i]==0)
		{
			for(int j=2*i;j<MAX;j=j+i)
			{
				if(arr[j]==0)
					arr[j]=i;
			}
		}
	}
}
void process()
{
	for(int i=0;i<MAX;i++)
		arr[i]+=arr[i-1];
}
int main()
{
	pre();
	process();
	int tc;
	cin>>tc;
	while(tc-->0)
	{
		int l,r;
		cin>>l>>r;
		cout<<(arr[r]-arr[l-1])<<endl;
	}
	return 0;
}
