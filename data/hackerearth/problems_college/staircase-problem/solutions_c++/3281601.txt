#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,step=1,sum=1;
    	
   		cin>>n;
   		if(n<=2)
   		cout<<"1"<<endl;
   		
   		else 
   		{
   			int p=1;
   			for(int i=3;sum<=n;i++)
   			{
   				if(((n-sum)<=p) ||((n-sum-1)<=(p+1)))
   			{
   				if(n-sum!=1)step++; break;
   			}
   				
   				else
   				{
   					sum+=i; step++; p=i;
   				}
   			}
   			
   			cout<<step<<endl;
   		}
   		
   	//	cout<<step<<endl;
    }
    return 0;
}
