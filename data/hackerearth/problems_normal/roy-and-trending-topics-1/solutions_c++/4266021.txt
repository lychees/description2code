#include <iostream>
#include <map>
#include<queue>
#include <climits>
using namespace std;

int main()
{
	multimap<long long int, long long int>newmap, mymap;
	priority_queue<long long int>q;
	long long int n, i, id, z, p, l, c, s, cur, inc, first, second, max=-1;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>id>>z>>p>>l>>c>>s;
    	cur=p*50+l*5+c*10+s*20;
    	inc=cur-z;
    	mymap.insert(pair<long long int, long long int>(inc, id));
    	newmap.insert(pair<long long int, long long int>(id, cur));
    }
    int count=5;
    while(count>0){
    	max=-1;
   
    multimap<long long int,long long int>::iterator j, l;
    for(j=mymap.begin();j!=mymap.end();j++)
    {
    	if(max<j->first)
    	{
    		max=j->first;
    		first=j->first;
    		second=j->second;
    		l=j;
    	}
    	else if(max==j->first)
    	{
    		if(second<j->second)
    		{
    			first=j->first;
    			second=j->second;
    			max=j->first;
    			l=j;
    		}
    	}
    }
    cout<<second<<" "<<newmap.find(second)->second<<"\n";
    mymap.erase(l);
    count--;
    }
   return 0;
}
