#include <bits/stdc++.h>
using namespace std;

int a[10005];
int freq[10005][105];

int main()
{
    int n,q,i,j,ans;
    scanf("%d",&n);
    for (i = 0; i < n; i++) {
    	scanf("%d",&a[i]);
    }
    freq[0][a[0]] = 1;
    for (i = 1; i < n; i++) {
    	for (j = 0; j <= 100; j++) {
    		freq[i][j] = freq[i - 1][j];
    	}
    	freq[i][a[i]]++;
    }
    scanf("%d",&q);
    while (q --) {
    	int x,y,val;
    	scanf("%d%d%d",&x,&y,&val);
    	if (x == 0) {
    		ans = freq[y][val];
    	} else {
    		ans = freq[y][val] - freq[x - 1][val];
    	}
    	printf("%d\n",ans);
    }
    return 0;
}
