#include <bits/stdc++.h>
using namespace std;
# define LL long long
bool dp[101][101];
bool board[101][101];
LL ways=0;
void path(int i,int j,int n)
{
	if(i<0||j<0||i>n-1||j>n-1)
		return;
	if(board[i][j]==true)
		return;
	if(dp[i][j]==true)
		return;
	dp[i][j]=true;
	if(i==n-1 && j==n-1)
		ways++;
	path(i-1,j,n);
	path(i,j-1,n);
	path(i+1,j,n);
	path(i,j+1,n);
	dp[i][j]=false;
	
}

int main()
{
	int n,temp;
	LL ans;
	scanf("%d",&n);
	for(int i=0;i<n;i++)
		for(int j=0;j<n;j++)
			{
				scanf("%d",&temp);
				if(temp==1)
					board[i][j]=true;
				else if(temp==0)
					board[i][j]=false;
			}
	path(0,0,n);
	printf("%lld\n",ways);
    
}
