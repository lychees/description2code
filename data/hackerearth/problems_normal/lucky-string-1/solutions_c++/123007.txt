#include<iostream>

using namespace std;

unsigned long long int test,n,sum,i,x,dv,rem,previous;

unsigned long long int power(unsigned long long int x)
{
	unsigned long long int y=1;

	while(x--)
		y*=2;

	return y;
}

int main()
{

	cin>>test;
	while(test--)
	{
		cin>>n;

		sum=0;
		previous=0;
		i=1;
		while(true)
		{

			sum+=(i*power(i));

			if(sum>=n)
				break;
			
			previous=sum;
			i++;
		}

		n-=previous;
		
		dv=(n-1)/i;
		rem=(i-1)-((n-1)%i);


		if(dv&power(rem))
			cout<<"Earth"<<endl;
		else
			cout<<"Hacker"<<endl;


	}		

	return 0;
}
