#include<cstdio>
#include<algorithm>
#include<vector>

#include<cmath>

using namespace std;

#define sd(a) scanf("%lld",&a)
#define pd(a) printf("%d\n",(a))
#define LL long long
#define F first
#define S second
#define MP make_pair
#define PB push_back
#define INF 1000000000000ll
int main()
{
	long long n,i,l,j,k,m,n1,n2,n3,n4,maxs;
	char s[25];
	long long a[25];
	sd(n);
	while(n--)
	{
		maxs=-1;
		scanf("%s",&s);
		l=strlen(s);
		for(i=0;i<l;++i)
		{
			a[i]=s[i]-'0';
		}
		for(i=1;i<l;++i)
		{
			for(j=i+1;j<l;++j)
			{
				for(k=j+1;k<l;++k)
				{
					if(i<=13&&(j-i)<=13&&(k-j)<=13)
					{
						n1=n2=n3=n4=0;
						for(m=0;m<i;++m)
							n1=n1*10+a[m];
						for(m=i;m<j;++m)
							n2=n2*10+a[m];
						for(m=j;m<k;++m)
							n3=n3*10+a[m];
						for(m=k;m<l;++m)
							n4=n4*10+a[m];
						if(i>1&&a[0]==0||(j-i)>1&&a[i]==0||(k-j)>1&&a[j]==0||(l-k)>1&&a[k]==0)
							continue;
						//printf("%lld %lld %lld %lld %lld\n",i,j-i,k-j,l-k,n1+n2+n3+n4);
						if(n1<=INF&&n2<=INF&&n3<=INF&&n4<=INF)
						{
							maxs=max(maxs,n1+n2+n3+n4);
						}
					}
				}
			}
		}
		if(maxs==-1)
			printf("unlucky\n");
		else
			printf("%lld\n",maxs);
	}
}
