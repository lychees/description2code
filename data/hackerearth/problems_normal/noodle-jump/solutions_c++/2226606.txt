#include<algorithm>
#include <iostream>
using namespace std;

int main()
{
    int a[1000001];
    int i,t,j,n,m;
    cin>>n>>m;
    for(i=0;i<n;i++)
      cin>>a[i];
     
    sort(a,a+n);
    for(i=1;i<n;i++)
    {
    	if(m+a[i-1]<a[i])
    	  break;
    }
    cout<<a[i-1]<<"\n";
    
    return 0;
}
