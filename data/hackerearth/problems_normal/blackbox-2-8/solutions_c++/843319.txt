#include<iostream>
using namespace std;
 
int main()
{
    
    
    int n,t,ans[100000],x,i;
    int memo[1001]={0};    
    n=1000;
    int index=1,flag=0;
    memo[0]=1;
    memo[1]=1;
    ans[1]=1;
    for(i=2;i<=n;i++)
    {
    	//cout<<i<<endl;
    	int count=0;
                     for(int j=1;j<=index;j++)
                     {
                         x=ans[j]*i+flag;
                         ans[j]=x%10;
                         count+=ans[j];
                         flag=x/10;                       
                                        
                    }
                     
                     while(flag!=0)
                     {
                                  
                           index++;
                           ans[index]=flag%10;
                           count+=ans[index];
                           flag=flag/10;
                           
                          
                    }
                    memo[i]=count;
                                   
    }  
    
    cin>>t;
    while(t--)
    {
    	int xx;
    	cin>>xx;
    	cout<<memo[xx]<<endl;
    }
                                                        
                                                         
    
    
    
    }
