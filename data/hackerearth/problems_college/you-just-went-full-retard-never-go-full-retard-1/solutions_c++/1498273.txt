#include <cstdio>
#include <cmath>
#include <bitset>
#include <vector>
using namespace std;

#define ii pair<int, int>
#define pb push_back
#define ll long long
const ll MAX = 10000010;

vector<int> a;
bitset<MAX> s;

void Sieve(){
	s.set();
	s[0] = s[1] = 0;
	for (ll i = 2; i < MAX; i++)
		if (s[i]){
			a.pb(i);
			for (ll j = i*i; j < MAX; j += i)
				s[j] = 0;
		}
}

int isPrime(ll n){
	if (n < MAX)
		return s[n];
	else
		for (int i = 0; i < a.size() && a[i] <= sqrt(n); i++)
			if (n % a[i] == 0)
				return 0;
	return 1;
}

int main(){
	Sieve();
	
	int t;
	ll n;
	scanf("%d", &t);
	while (t--){
		scanf("%lld", &n);
		printf("%s\n", isPrime(n) ? "YES" : "NO");
	}
	
	return 0;
}
