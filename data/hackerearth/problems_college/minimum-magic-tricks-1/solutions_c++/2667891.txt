#define mods 1000000007
using namespace std;
ll F[1001]={0};
int findmt(int m)
{
    if(F[m]!=0)
        return F[m];
    ll sum=1;
    for(int i=m-2;i>0;i--)
    {
        sum=(sum+2*findmt(i))%mods;
    }
    F[m]=sum%mods;
    return F[m];
}
