#include <iostream>
using namespace std;

int main()
{
    long int t,f[26]={0};
    char b[26];
    char ch='A';
    for(int i=25;i>=0;i--)
    {
    b[i]=ch;
    ch++;
    }
    
    char a[10000];
    cin>>t;
    while(t--)
    {
    	int flag=0;
      	 cin>>a;
      	 for(long int i=0;a[i]!='\0';i++)
      	 f[a[i]-'A']++;
      	 for(long int i=0;a[i]!='\0';i++)
      	 {
      	 	for(int j=0;j<26;j++)
      	 	{
      	 		if(a[i]==b[j] && j+1!=f[26-(j+1)])
      	 		{
      	 			flag=1;
      	 	        break;
      	 	}
      	 	
      	 	}
      	 	if(flag==1)
      	 	break;
      	 }
      	 if(flag==0)
      	 cout<<"Yes"<<endl;
      	 else
      	 cout<<"No"<<endl;
    }
    return 0;
}
