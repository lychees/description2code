#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,n,x;
	cin>>t;
	while(t--)
	{
		cin>>n;
		vector<pair<int,int> > v;
		for(int i=0;i<n;i++)
		{
			scanf("%d",&x);
			v.push_back(make_pair(x,i));
		}
		sort(v.begin(),v.end());
		int max=-1;
		for(int i=0;i<n;i++)
		{
			if(abs(v[i].second-i)>max)
			max=abs(v[i].second-i);
		}
		cout<<max<<endl;
	}
}
