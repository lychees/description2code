#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long t, n, a, b, c, i, x, y, count1, count2;
    cin>>t;
    while(t--){
    	cin>>n>>a>>b>>c;
    	//cin>>n;
    	count1=0, count2=0;
    	for(i=0;i<n;i++){
    		
    		cin>>x>>y;
    		if(a*x+b*y+c>0)
    			count1++;
    		else if(a*x+b*y+c<0)
    			count2++;
    			
    		
    		
    		
    	}
    	
    	if(count1>count2)
    		cout<<count1<<endl;
    	else
    		cout<<count2<<endl;
    	
    	
    	
    }
    return 0;
}
