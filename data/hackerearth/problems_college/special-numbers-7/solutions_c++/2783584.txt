#include<bits/stdc++.h>

using namespace std;

bool checkThreeFact(int n, bool* primes)
{
	int facts =  0;
	int i,t = n;
	for(i=2;i<n && t > 1;i++)
	{
		if( primes[i] )
		{
			if( t % i ==0)
				facts++;
			while( t % i ==0)		
				t /= i;
		}
	}
	if( facts >= 3)
		return true;
	return false;
}

int main( void )
{
	int t;
	bool primes[10000];
	int i,j;
	for(i=0;i<10000;i++)
		primes[i] = true;
	for(j=2*2; j < 10000;j+=2)
		primes[j] = false;
	for(i=3;i< 10000;i+=2)
	{
		if( primes[i] )
		{
			for(j=i*i; j< 10000;j+=i)
				primes[j] = false;
		}
	}

	int threeFacts[1010];
	int k = 0;
	for(i=30;;i++)
	{
		if( k == 1000 )
			break;
		bool isThreeFact = checkThreeFact(i,primes);
		if( isThreeFact )
			threeFacts[k++] = i;	
	}
	int n;
	cin>>t;
	while( t-- )
	{
		cin>>n;
		cout<<threeFacts[n-1]<<endl;
	}
	return 0;
}
