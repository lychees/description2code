import heapq

num_rows, fans = map(int, raw_input().strip().split(" "))
rows_spaces = map(lambda x: -int(x), raw_input().strip().split(" "))
heapq.heapify(rows_spaces)
sum_sales = 0
for fan in range(fans):
    current_max = heapq.heappop(rows_spaces)
    sum_sales += current_max
    current_max += 1
    heapq.heappush(rows_spaces, current_max)
print -sum_sales
