#include <iostream>
#include <cmath>
using namespace std;

const int MAX = 10001;
int composite [MAX];

void sieve () {
	bool flag [MAX];
	int l = sqrt(MAX) + 1;
	int c = 0;	

	for (int i=2; i<l; ++i) {
		if (!flag[i]) {
			for (int j=i+i; j<MAX; j+=i) {
				flag[j] = 1;
			}
		}
	}

	for (int i=2; i<MAX; ++i) {
		if (flag[i]) {
			composite[c] = i;
			++c;
		}
	}
}

int main ()
{
	ios_base::sync_with_stdio(false);
	
	sieve();

	int n = 0;
	cin >> n;

	for (int i=0; composite[i] <= n; ++i) {
		cout << composite[i] << " ";
	}

	return 0;
}
