#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
#define N 2010
int main(){
		#ifndef ONLINE_JUDGE
		freopen("input.txt","r",stdin);
	#endif
	ll n,k,a[N];
	cin>>n>>k;
		for(int i=0;i<n;i++)	cin>>a[i];
		sort(a,a+n);
		ll co=0;
		for(int i=0;i<n;i++){
			for(int j=i+1;j<n;j++){
				long long pp=a[i]*a[j];
				ll req=k/pp;
				int idx=upper_bound(a+j+1,a+n,req)-a;
				//cout<<" ksd"<<idx<<endl;
				int ans=idx-1;
				if(pp*a[ans]<=k){
					//ans--;
					co+=ans-j;
				}
				//if(ans>j)	co+=0;
				//else	
			}
		}
		cout<<co<<endl;
}
