#include <iostream>
using namespace std;

int main()
{
    int t,a1,b1,a2,b2,y;
    cin>>t;
    for(int i=1;i<=t;i++)
    {
    	cin>>a1>>a2>>b1>>b2;
    	for(int i=a1;i<=a2;i++)
    	{
    		if(a1<b1 && a2<b2)
    		y=0;
    			else if(a1>b1 && a2>b2)
    			y=1;
    			else if(a1==b1 && a2==b2)
    			{
    				y=(a2-a1)*(a2-a1-1);
    			}
    	}
    	cout<<"Case #"<<i<<": "<<y<<endl;
    	
    	
    	
    }
}
