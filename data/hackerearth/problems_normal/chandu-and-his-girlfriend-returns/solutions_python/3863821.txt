num_test_cases = 0
#Input for number of test cases
try:
    num_test_cases = int(raw_input())
except ValueError as ve:
    raise ValueError('Invalid input for number of test cases..')
if num_test_cases > 100 or num_test_cases < 1:
    raise ValueError('Invalid input for number of test cases. Range:1 <= T <= 100')

for n in range(num_test_cases):
    x = y = 0
    arr_x = None
    arr_y = None
    #Input for size of arrays
    try:
        x,y = tuple(map(int, raw_input().split(' ')))
    except ValueError as ve:
        raise ValueError('Invalid input for size of arrays..')
    #assuming the input to be in non-increasing order
    if x > 50000 or y > 50000 or x < 1 or y < 1:
        raise ValueError('Invalid input for size of arrays. Range:1 <= N, M <= 5*10^4')
    
    
    #Input for first array
    try:
        arr_x = map(int, raw_input().split(' '))
    except ValueError as ve:
        raise ValueError('Invalid input for first array..')
    #assuming the input to be in non-increasing order
    if arr_x[0] > pow(10,9) or arr_x[-1] < 0:
        raise ValueError('Invalid input for array values. Range:0 <= Ai, Bi <= 10^9.')

	
    #Input for second array
    try:
        arr_y = map(int, raw_input().split(' '))
    except ValueError as ve:
        raise ValueError('Invalid input for second array..')

    if arr_y[0] > pow(10,9) or arr_y[-1] < 0:
        raise ValueError('Invalid input for array values. Range:0 <= Ai, Bi <= 10^9.')

    arr_final = []
    i = j = 0
    while i < x and j < y:
        if arr_x[i] >= arr_y[j]:
            arr_final.append(arr_x[i])
            i += 1
            if i == x:
                arr_final.extend(arr_y[j:])
                j = y
        else:
            arr_final.append(arr_y[j])
            j += 1
            if j == y:
                arr_final.extend(arr_x[i:])
                i = x
    print ' '.join(map(str, arr_final))
