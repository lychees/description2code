t = int(raw_input())
while t > 0:
	t -= 1
	n,k = map(int,raw_input().split())
	l = []
	ans = 10**9
	for i in range(n):
		l1 = raw_input().split()
		l.append(l1[1:])
	for i in range(2**n):
		s = []
		c = 0
		for j in range(n):
			if i&(2**j):
				for x in l[j]:
					#print i,x
					s.append(x)
				c += 1;
		#print s,set(s)
		if len(set(s)) >= k:
			ans = min(ans,c)
	if ans == 10**9:
		ans = -1
	print ans
