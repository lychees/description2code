#include<bits/stdc++.h>
using namespace std;

int main()
{
    ios::sync_with_stdio(0);
    int n,q;
    cin>>n>>q;
    int arr[n+3];
    for(int i=1;i<=n;i++)
    {
        cin>>arr[i];
    }
    while(q--)
    {
        int l,r;
        cin>>l>>r;
        long long sum=arr[r],diff=arr[r];
        for(int j=r-1;j>=l;j--)
        {
            sum+=arr[j];
            diff-=arr[j];
        }
        cout<<sum<<" "<<diff<<endl;
    }
    return 0;
}
