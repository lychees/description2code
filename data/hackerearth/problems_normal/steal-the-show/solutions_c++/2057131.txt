#include<bits/stdc++.h>

using namespace std;
#define REP(a,b) for(int a=0;a<b;a++)
#define FOR(a,b,c) for(int a=b;a<c;a++)
#define FORD(a,b,c) for(int a=b;a>=c;a--)
#define S scanf
#define P printf
#define LEN(x) ((int)x.length())
#define SZ(x) ((int)x.size())
#define ALL(x) x.begin(), x.end()
#define MP(x,y) make_pair(x,y)
#define PB(x) push_back(x)
#define INF 1000000000
typedef long long LL;
typedef pair<int,int> PII;
typedef pair<int, PII> PIII;
typedef pair<LL,LL> PLL;
typedef vector<int> VI;
typedef vector<bool> VB;
typedef vector<string> VS;
typedef vector<PII> VII;


int main()
{
    int ce,ct,co,n,temp;
    VI A;
    S("%d %d %d",&ce,&co,&ct);
    if(co<ct)
        swap(co,ct);
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {
     scanf("%d",&temp);
     A.PB(temp);
    }
    sort(A.begin(),A.end());
    int arr[n+3];
    memset(arr,0,sizeof(arr));
    if((n+3)&1)
    {   int diff=0;
        arr[0]=ct;
        arr[n+2]=co;
        arr[n/2+1]=ce;
        int j=0;
        for(int i=0;i<n+3;i++)
        {
            if(arr[i]==0)
                {
                    arr[i]=A[j++];
                }
        }
        for(int i=0;i<n+2;i++)
        {
            diff+=abs(arr[i]-arr[i+1]);
         //   cout<<diff<<endl;
        }
        cout<<diff<<"\n";
    }
    else
    {   int diff1=0,diff2=0;
        arr[0]=ct;
        arr[n+2]=co;
        arr[(n+3)/2]=ce;
        int j=0;
        for(int i=0;i<n+3;i++)
        {
            if(arr[i]==0)
                {
                    arr[i]=A[j++];
                }
        }
        for(int i=0;i<n+2;i++)
        {
            diff1+=abs(arr[i]-arr[i+1]);
        }
        memset(arr,0,sizeof(arr));
        arr[0]=ct;
        arr[n+2]=co;
        arr[(n+3)/2-1]=ce;
        j=0;

        for(int i=0;i<n+3;i++)
        {
            if(arr[i]==0)
                {
                    arr[i]=A[j++];
                }
        }
        for(int i=0;i<n+2;i++)
        {
            diff2+=abs(arr[i]-arr[i+1]);
        }
        cout<<min(diff1,diff2)<<endl;

    }


}
