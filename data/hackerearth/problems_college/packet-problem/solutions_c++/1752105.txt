#include <stdio.h>
typedef long long int ll;
ll a[100];
ll b[100];
int main()
{
    ll t,n,i,j,k,m;
    scanf("%lld",&t);
    a[0]=0;b[0]=1;
    for(i=1;i<100;i++)
    {
    	a[i]=a[i-1]+b[i-1];
    	b[i]=a[i-1];
	}
    while(t--)
    {
    	scanf("%lld",&m);
    	printf("%lld %lld\n",b[m],a[m]);
    	
	}
}
