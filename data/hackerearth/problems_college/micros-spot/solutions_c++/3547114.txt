#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		int n,m;
		scanf("%d %d",&n,&m);
		int a[n][m],b[n][m];
		int r[n],c[m];
		for(int i=0;i<n;i++)
		   r[i]=0;
		for(int i=0;i<m;i++)
		   c[i]=0;
		for(int i=0;i<n;i++)
		for(int j=0;j<m;j++){
		    scanf("%d",&a[i][j]);
		    c[j]=max(c[j],a[i][j]);
		}
	    for(int i=0;i<n;i++)
		for(int j=0;j<m;j++){
		    scanf("%d",&b[i][j]);
		    r[i]=max(r[i],b[i][j]);
		}
		int ans=0;
		for(int i=0;i<n;i++){
			for(int j=0;j<m;j++){
				if(b[i][j]==r[i]&&a[i][j]==c[j]){
					ans++;
				}
			}
		}
		printf("%d\n",ans);
	}
	return 0;
}
