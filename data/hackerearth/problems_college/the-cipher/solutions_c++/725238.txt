#include <iostream>
#include <cstdio>

using namespace std;

int main()
{
        int test_cases;
        scanf("%d",&test_cases);
        while(test_cases > 0)
        {
                int N,arr[10000],i=0;
                scanf("%d",&N);
                while(i<N)
                {
                        scanf("%d",&arr[i]);
                        i = i+1;
                }
                
                for( i = 0; i < N; i++)
                 cout << (char)(255 - arr[i]);
                 
                 cout << endl;
                                
                test_cases--;
        }
        return 0;
}
