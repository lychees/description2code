#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define mod (int)(1e9 + 7)
#define INF INT_MAX
#define FOR(i,a,b) for(i= a ; i < b ; i++)
#define mmax(a,b) ((a)>(b)?(a):(b))
#define mmin(a,b) ((a)<(b)?(a):(b))
int t,n,i=0,temp;
ll a[20][3];
ll dp[20][4];
ll cost(int no,int c)
{	ll min;
	min=INF;
	if(no>=n)	return 0;
	if(dp[no][c]!=-1)	return dp[no][c];
	for(int i=0;i<3;i++)
	{
		if(i==c)	continue;
		min=mmin(min,a[no][i]+cost(no+1,i));
	}
	return dp[no][c]=min;
}
int main()
{
	ll ans;
	cin>>t;
	while(t--)
	{	for(i=0;i<20;i++)
			for(int j=0;j<4;j++)
				dp[i][j]=-1;
		cin>>n;
		FOR(i,0,n)
		{
			cin>>a[i][0];	cin>>a[i][1];	cin>>a[i][2];
		}
		ans=cost(0,3);
		cout<<ans<<endl;
	}

	return 0;
}
