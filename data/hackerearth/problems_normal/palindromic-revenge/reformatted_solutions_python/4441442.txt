t = int(raw_input())
while t > 0:
    st1 = raw_input()
    st2 = st1[::-1]
    if st1 == st2:
        print '0'
    else:
        l = len(st1)
        st3 = st1[:l/2]
        st4 = st1[l/2:][::-1]
        c = 0
        for i in range(l/2):
            c = c+abs(int(ord(st4[i]))-int(ord(st3[i])))
        print c
    t = t-1
