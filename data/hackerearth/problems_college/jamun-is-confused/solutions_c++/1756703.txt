#include<iostream>
using namespace std;
int Gcd(long a,long b){
	if(b==0)
		return a;
	else return Gcd(b,(a%b));
}
int main(){
	long t;
	cin>>t;
	while(t--){
		long n;
		cin>>n;
		long gcd;
		cin>>gcd;
		for(int i=1;i<n;i++){
			long temp;
			cin>>temp;
			gcd=Gcd(gcd,temp);
		}
		cout<<gcd<<endl;
	}
	return 0;
}
