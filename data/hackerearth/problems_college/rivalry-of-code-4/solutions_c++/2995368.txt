#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	string a,b,c;
    	cin>>a>>b;
    	if(a.length()>b.length()){
    		c=a;
    		a=b;
    		b=c;
    	}
    	int min=a.length();
    	for(int i=0;i<=b.length()-a.length();i++){
    		int cnt =0;
    		for(int j=0;j<a.length();j++){
    			if(a[j]!=b[i+j])
    			cnt++;
    		}
    		if(cnt<min) min= cnt;
    	}
    	cout<<min<<endl;
    }
    return 0;
}
