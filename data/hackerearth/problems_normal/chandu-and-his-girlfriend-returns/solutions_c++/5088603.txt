#include <bits/stdc++.h>
using namespace std;
#define lln long long int

int main()
{
    lln t;
    cin >> t;
    for(lln i=0;i<t;i++){
    	lln n,m;
    	cin >> n >> m;
    	lln a[n+m];
    	for(lln j=0;j<n+m;j++){
    		cin >> a[j];
    	}
    	sort(a,a+n+m);
    	for(lln j=n+m-1;j>=0;j--){
    		cout << a[j] << " ";
    	}
    	cout << endl;
    }
    return 0;
}
