#include<iostream>
using namespace std;
#include<cstdio>
#include<cstring>
char str[52];
int main()
{
    int task;
    cin>>task;
    cin.ignore();
    while(task--){
    cin.getline(str,51);
    int len=strlen(str);
    for(int i=0;i<len;i++)
    {
        int start=i,last=len-1,flag=1;
        while(start<=last)
        {
            if(str[start++]!=str[last--])
            {flag=0;break;}
        }
        if(flag==1)
        {printf("%d\n",len+i);break;}
    }
    }
}
