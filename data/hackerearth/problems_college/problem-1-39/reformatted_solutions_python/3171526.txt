# print 'Hello World!'
testCases = int(raw_input())
for i in range(testCases):
    a, b, c = list(map(int, raw_input().split(" ")))
    power = a**b
    if power <= c:
        if abs(power-0) < c-power:
            print 0
        else:
            print c
    else:
        fac = float(power)/c
        num1 = int(fac)*c
        num2 = int(fac+1)*c
        if num2-power < power-num1:
            print int(num2)
        else:
            print int(num1)
