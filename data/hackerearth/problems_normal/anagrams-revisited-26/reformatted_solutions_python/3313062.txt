'''
# Read input from stdin and provide input before running code

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
N = input()
d = []
anagramCounter = {}
maxAnagramCount = 1
for _ in xrange(N):
    raw = raw_input().strip()
    s = ''.join(sorted(raw))

    if s not in d:
        d.append(s)
    else:
        if s in anagramCounter:
            anagramCounter[s] += 1
        else:
            anagramCounter[s] = 2

        if anagramCounter[s] > maxAnagramCount:
            maxAnagramCount = anagramCounter[s]


print maxAnagramCount
