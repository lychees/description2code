# include <bits/stdc++.h>
using namespace std;

double root(double a,double b,double c)
{
  double d= b*b -4*a*c;
  d=sqrt(d);
  double r1 = (( -b + d ) / (2*a));
  double r2 = (( -b - d ) / (2*a));
  //cout << r1 << " "<<r2<<endl;
  return min(r1,r2);
}


int main()
{
 ios_base::sync_with_stdio(false);
 double a,b,p;
 cin >> b>>a>>p;
 double x = 200 * a;
 double y = 100*a*a  - a*a*p ;
 printf("%.2lf",root(100.0,-x,y)) ;
}
