#include <bits/stdc++.h>

using namespace std;

int sieve[1000005]={0},prime[1000005]={0};
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    int i,j;
    for(i=2;i<=1000;i++)
    {
        if(sieve[i]==0)
        {
            for(j=2*i;j<=1000000;j+=i)
                sieve[j]=1;
        }
    }
    prime[1]=0;
    for(i=2;i<=1000000;i++)
        if(sieve[i]==0)
            prime[i]=prime[i-1]+1;
        else
            prime[i]=prime[i-1];
    int t;
    cin>>t;
    while(t--)
    {
        int l,r;
        cin>>l>>r;
        cout<<fixed<<setprecision(6)<<((float)prime[r]-(float)prime[l-1])/(r-l+1)<<"\n";
    }
    return 0;
}
