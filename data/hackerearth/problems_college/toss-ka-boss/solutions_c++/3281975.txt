#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long t, i, count, sum;
    string a;
    cin>>t;
    while(t--){
    	count=0;sum=0;
    	cin>>a;
    	for(i=0;i<a.length();i++)
    	{
    		if(a[i]=='H')
    			count++;
    		
    		else
    			count=0;
    		
    		sum+=count;
    		
    	}
    	
    	cout<<sum<<endl;
    	
    }
    return 0;
}
