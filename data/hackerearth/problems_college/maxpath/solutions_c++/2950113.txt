#include <iostream>
using namespace std;

int main()
{
    int N,i,j,s,a[50][50];
    cin>>N;
    for(i=0;i<N;i++)
     {
     	for(j=0;j<N;j++)
     	   {
     	   	cin>>a[i][j];
     	   }
     }
     s=a[0][0];
     i=0;
     j=0;
     while(i<N-1&&j<N-1)
     {
       	   
       	   	if(a[i+1][j]>a[i][j+1]&&(i+1)!=j)
       	   	{
       	   	   s=s+a[i+1][j];
       	   	   i++;
       	   	}
       	   	else
       	   	{
       	   	 if((i)!=(j+1))
       	   	 {
       	   	   s=s+a[i][j+1];
       	   	   j++;
       	   	 }
else 
       	   	    {
       	   	    	s+=a[i+1][j];
       	   	        i++;
       	   	    }
       	   	 
       	   	}
     }
     if(i==N-1)
       {
       	i=N-1;
       	while(j<N)
       	  s=s+a[i][++j];
       }
     if(j==N-1)
       {
       	 j=N-1;
       	 while(i<N)
       	    s+=a[++i][j];
       }
     i=N-1;
     j=N-1;
    while(i>0&&j>0)
     {
       	   
       	   	if(a[i-1][j]>a[i][j-1]&&(i-1)!=j)
       	   	{
       	   	   s+=a[i-1][j];
i--;
       	   	}
       	   	else
       	   	{
       	   	 if(i!=j-1)
       	   	 {
       	   	   s+=a[i][j-1];
       	   	   j--;
       	   	 }
       	   	 else 
       	   	    {
       	   	    	s=s+a[i-1][j];
       	   	        i--;
       	   	    }
       	   	 
       	   	}
     }
     if(i<=0)
       {
       	i=0;
       	while(j>0)
       		s+=a[i][--j];
       }
     if(j<=0)
      {
      	j=0;
      	while(i>0)
      	   s+=a[--i][j];
      }
     
     cout<<s;  
    return 0;
}
