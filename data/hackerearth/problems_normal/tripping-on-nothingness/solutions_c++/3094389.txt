#include <iostream>
using namespace std;
int gcd(int n1,int n2)
{
	if(n2==0)
		return n1;
	return gcd(n2,n1%n2);
}
int main()
{
	int a,b,n;
	cin>>a>>n;
	int ar[n],br[n];
	for(int i=0;i<n;i++)
	{
		cin>>ar[i];
		if(ar[i]>a)
			br[i]=gcd(ar[i],a);
		else
			br[i]=gcd(a,ar[i]);
	}
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<i;j++)
		{
			if(br[i]==br[j])
			{
				br[i]=-1;
				break;
			}
		}
		cout<<br[i]<<endl;
	}
    return 0;
}
