#include <iostream>
#include <cmath>
using namespace std;

long d(long n)
{
	long x = 0;
	while(n)
	{
		n/=2;
		x++;
	}
	return x;
}

long power(int x)
{
	long ans = 1;
	
	for(int i=0;i<x;i++)
	{
		ans *= 2;
	}
	return ans;
}

int main()
{
    int t;
    cin >> t;
    
    while(t--)
    {
    	long n;
    	cin >> n;
    	
    	if(n==1)
    	{
    		cout << 0 << endl;
    		continue;
    	}
    	long x = d(n);
    	
    	cout << power(x)-1 << endl;
    }
}
