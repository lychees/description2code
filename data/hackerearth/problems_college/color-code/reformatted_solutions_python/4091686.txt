v = [i for i in '0123456789UVWXYZ']
for i in range(int(raw_input())):
    r, g, b = map(int, raw_input().split(' '))
    print "Case %d: %s%s %s%s %s%s" % (i+1, v[r/16], v[r % 16], v[g/16], v[g % 16], v[b/16], v[b % 16])
