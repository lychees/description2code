#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
typedef long long ll;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        ll a[n+10];
        for(int i=0;i<n;i++)
                cin>>a[i];
        sort(a,a+n);
        ll b=a[0],ans=0;
        for(int i=1;i<n;i++)
           ans+=(b+a[i]);   
        cout<<ans<<endl;  
        
    }
    return 0;
}
