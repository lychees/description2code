#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

int main() {
	int t;
	scanf("%d",&t);
	long long int a[1000001];
//	long long int b[1000001];
	long long int m = 1000000007;
	a[1] = 0LL;
	for (long long int i = 0; i <= 1000000; i++) {
		a[i] = 987654321LL;
	//	b[i] = 987654321LL;
	}
	for (long long int i = 2; i <= 1000000; i++) {
		if (a[i] == 987654321) {
			for (long long int j = 2; j*i <= 100000; j++) {
				a[i*j] = min(i,a[i*j]);
			//	b[i*j] = min(i,b[i*j]);
			}
			a[i] = 0;
		//	b[i] = 0;
		}	
	}
	for (int i = 2; i <= 1000000; i++) {
		a[i] = (a[i]+a[i-1])%m;
	}
	while (t--) {
		int l,r;
		scanf("%d%d",&l,&r);
		long long int sum = a[r]-a[l];
		long long int sum1 = a[l]-a[l-1];
		sum = (sum+sum1)%m;
		printf("%lld\n",sum);
	}
	return 0;
}
