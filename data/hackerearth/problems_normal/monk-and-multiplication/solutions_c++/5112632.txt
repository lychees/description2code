#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int a[n];
    long long th=0,se=0,fi=0,ans=0;
    for(int i=0;i<n;i++)
    {
    	cin>>a[i];
    	if(i==0)
    	{
    		fi=a[i];
    		cout<<"-1\n";
    	}
    	else if(i==1)
    	{
    		if(a[i]<fi)
    		se=a[i];
    		else
    		{
    			se=fi;
    			fi=a[i];
    		}
    		cout<<"-1\n";
    	}
    	else
    	{
    		if(a[i]<th)
    		{
    			ans=th*se*fi;
    			cout<<ans<<endl;
    		}
    		else if(a[i]<se)
    		{
    			th=a[i];
    			ans=th*se*fi;
    			cout<<ans<<endl;
    		}
    		else if(a[i]<fi)
    		{
    			th=se;
    			se=a[i];
    			ans=th*se*fi;
    			cout<<ans<<endl;
    		}
    		else
    		{
    			th=se;
    			se=fi;
    			fi=a[i];
    			ans=th*se*fi;
    			cout<<ans<<endl;
    		}
    	}
    }
    return 0;
}
