T = int(raw_input())
for k in range(T):
    W = list(raw_input())
    R = list(raw_input())
    R.sort()
    j=0
    for i in range(len(W)):
        if ord(W[i])>ord(R[j]):
            W[i]=R[j]
            j+=1
        if j==len(R):
            break
    print "".join(W)
