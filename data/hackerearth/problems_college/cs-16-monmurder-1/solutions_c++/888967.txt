#include<iostream>
#include<math.h>
#include<vector>
using namespace std;
static int prime[1000001];
vector<int> plist;
int sieve()
{
	plist.push_back(2);
	for(int i=3;i<=1000;i+=2)
	{
		if(prime[i]==0)
		{
			for(int j=i*i;j<=1000000;j+=i)
			{
				prime[j]=1;
			}
		}
	}
	for(int i=3;i<=1000000;i+=2)
	{
		if(prime[i]==0)
		{
			plist.push_back(i);
		}
	}
	return 0;
}
int main()
{
	sieve();
    int t,n;
    cin>>t;
    int sz=plist.size();
    while(t--){

        cin>>n;
        int sum=0;
        int x=sqrt(n);
        for(int i=0;i<sz&&plist[i]<=x;i++){
            int cn=0;
            while(n%plist[i]==0){n/=plist[i];cn++;}
            sum=sum^cn;
        }
        if(n>1)sum^=1;
        if(sum!=0)cout<<"Printf{}"<<endl;
    else cout<<"Pappu Bhujia"<<endl;
    }
	return 0;
}
