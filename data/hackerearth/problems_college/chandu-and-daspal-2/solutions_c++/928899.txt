#include <bits/stdc++.h>
using namespace std;
int gcd (int a,int b )
{
  int c;
  while ( a != 0 ) {
     c = a; a = b%a;  b = c;
  }
  return b;
}
int main() 
{
    int t;cin>>t;
    int x,y,c;
    string str1 = "Chandu";
    string str2 = "Daspal";
    string str;
    while(t--) {
        c = 0;
        cin>>x>>y>>str;
        while(x && y) {
            int k = gcd(x,y);
            x-=k;
            y-=k;
            c++;
        }
        if(str[0] == 'D')
        c++;
        if(c%2)
        cout<<str1;
        else
        cout<<str2;
        cout<<endl;
    }
	return 0;
}
