#include<iostream>
using namespace std;
int main()
{
	int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		int *table=new int[n+1];
		for(int i=0;i<n+1;++i)
		table[i]=0;
		table[0]=1;
		for(int i=1;i<=5;i=i+2)
		{
			for(int j=i;j<=n;++j)
			{
				table[j]+=table[j-i];
			}
		}
		cout<<table[n]<<endl;
		
	}
	return 0;
}
