def main():
    T = input()
    for _ in xrange(T):
        N = input()
        foodDict = {}
        foodCounter = 0
        for __ in xrange(N):
            foodFound, foodNeeded = raw_input().strip().split()
            if foodFound not in foodDict:
                foodDict[foodFound] = 1
            else:
                foodDict[foodFound] += 1

            if foodNeeded not in foodDict:
                foodCounter += 1
            else:
                foodDict[foodNeeded] -= 1
                if foodDict[foodNeeded] <= 0:
                    del foodDict[foodNeeded]
        print foodCounter


if __name__ == "__main__":
    main()
