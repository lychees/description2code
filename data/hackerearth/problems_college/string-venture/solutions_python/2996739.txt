import string
from collections import Counter
import operator
def find_max_letter_count(word):
    cnt = Counter(word)
    return max(cnt.iteritems(), key=operator.itemgetter(1))[0]

st = raw_input() 
ins = map(ord, list(st))
insum = sum(ins)
inlen = len(ins)
weight = insum / inlen
if weight & 1:
    print find_max_letter_count(st)
else:
    print "".join(st[::-1])
