#include<iostream>
using namespace std;
int arr[2000050]={0};
int pri[2000050];
int pos=0;
int ans[2000050]={0};
int syme(int x)
{
	int rev=0,xx=x;
	while(xx>0)
	{
		rev=rev*10+xx%10;
		xx/=10;
	}
	return x==rev;
}
int main()
{
	arr[1]=1;
	for(int i=2;i<=2000000;i++)
	{
		if(!arr[i])
		{
			if(syme(i))
			{
				pri[pos]=i,pos++;
			}
		for(int j=i;j<=2000000;j+=i)
		arr[j]=1;
		}
	}
	int sym=pri[0];
	int pp=0;
	for(int i=1;i<=2000000;i++)
	{
		while(pp<pos && pri[pp]<i)
		pp++,sym=pri[pp];
		ans[i]=sym;
	}
	int test;
	cin>>test;
	while(test--)
	{
		int x;
		cin>>x;
		cout<<ans[x]<<endl;
	}
	return 0;
}
