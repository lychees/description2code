#include<bits/stdc++.h>
using namespace std;

long long n,i,a[200010],j,temp,ans,tc;

int main(){
	ios::sync_with_stdio(0);
	
	cin>>tc;
	
	while(tc--){
		cin>>n;
		for(i=0;i<n;i++)
			cin>>a[i];
		sort(a,a+n);
	
		ans = 0;
		for(i=n-1;i>0;i-=2){
			ans += a[i]*a[i-1];
		}
		cout<<ans<<"\n";
	}
	return 0;
}
