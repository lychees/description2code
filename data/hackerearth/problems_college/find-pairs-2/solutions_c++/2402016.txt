#include <iostream>
using namespace std;
 
int main()
{
    long long int i,n,t,max,d,sum,k;
    cin>>t;
    while(t--)
    {
        sum=0;max=0;
    	cin>>n;
    	long long int a[n+1];
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    		
    	}
    	int *arr=(int *)malloc(sizeof(int)*1000007);
    		int *arrn=(int *)malloc(sizeof(int)*1000007);
    	for(i=0;i<n;i++)
    	{
    		if(a[i]>=0)
    		arr[a[i]]++;
    		else
    		{
    			k=-a[i];
    			arrn[k]++;
    		}
    	}
    	for(i=0;i<1000007;i++)
    	{
    		if(arr[i]!=0)
    		{
    			d=arr[i]-1;
    			sum+=d*(d+1)/2;
    		}
    	}
    	for(i=0;i<1000007;i++)
    	{
    		if(arrn[i]!=0)
    		{
    			d=arrn[i]-1;
    			sum+=d*(d+1)/2;
    		}
    	}
    	sum+=n;
    	cout<<sum<<endl;
    }
    return 0;
}
