#include <bits/stdc++.h>
using namespace std;

long long int dp[1000010];

int main()
{
    int t,n,i,j;

    for (i = 1; i <= 1000000; i++) {
        for (j = i + i; j <= 1000000; j = j + i) {
            dp[j] = dp[j] + 1;
        }
        dp[i] = dp[i] + dp[i-1];
    }

    cin >> t;

    while (t--) {
        cin >> n;
        cout <<dp[n]<<endl;
    }
    return 0;
}
