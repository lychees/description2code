#include <iostream>
using namespace std;

int main()
{
    long n,a,b,x,c,i,y;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	c=0;
    	cin>>a>>b;
    	if(a<b)
    	{
    		x=a;
    		y=b;
    	}
    	else
    	{
    		x=b;
    		y=a;
    	}
    	while(x>0)
    	{
    		if(x%10+y%10>=10)
    			c++;
    		x=x/10;
    		y=y/10;
    	}
    	if(c==0)
    		cout<<"No carry operation"<<endl;
    	else if(c==1)
    		cout<<"1 carry operation"<<endl;
    	else
    		cout<<c<<" carry operations"<<endl;
    }
}
