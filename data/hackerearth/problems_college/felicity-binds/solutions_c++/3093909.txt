#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int a,n;
    	cin>>n;
    	vector<int> m;
    	vector<int> w;
    	for(int i=0;i<n;i++){
    		scanf("%d",&a);
    		m.push_back(a);
    	}
    	for(int i=0;i<n;i++){
    		scanf("%d",&a);
    		w.push_back(a);
    	}
    	sort(m.begin(),m.end());
    	sort(w.begin(),w.end());
    	long long sum=0;
    	for(int i=0;i<n;i++){
    		sum=sum+(long long)(m[i]*w[i]);
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
