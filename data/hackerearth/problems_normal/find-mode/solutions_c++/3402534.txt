#define LL long long
#define mod 1000000007
#define sz size()
#define ln length()
#define FOR(i,a,n) for(int i=a;i<n;i++)
#define FORE(i,a,n) for(int i=a;i<=n;i++)
#define scint(a)    scanf("%d",&a)
#define scllint(l) scanf("%lld",&l)
#define scs(s)  scanf("%s",&s)

int main() {

    int T;
    scint(T);

    int N;

    while(T--)
    {
        scint(N);

        LL int x;

        LL int tempc=0;

        map<LL int,int> m;

        vector<LL int> ans;

        FOR(i,0,N)
        {
            scllint(x);

            m[x]++;

            if(m[x]>tempc) {tempc=m[x];ans.clear();ans.push_back(x);}

            else if(m[x]==tempc) ans.push_back(x);

        }


        int n=ans.size();

        sort(ans.begin(),ans.end(),greater<int>());

        if(n>1)
        {

        FOR(i,0,n-1)
        {
            printf("%lld ",ans[i]);
        }

        }

        printf("%lld\n",ans[n-1]);
    }

    /* Enter your code here. Read input from STDIN. Print output to STDOUT */
    return 0;
}
