#include<iostream>
#include<stdio.h>
#include<stdlib.h>
using namespace std;
void bubble(int *a,int n)
{
    int i,k,temp;
    for(k=0;k<n-1;k++)
        for(i=0;i<n-k-1;i++)
    {
        if(a[i]>a[i+1])
        {
            temp=a[i];
            a[i]=a[i+1];
            a[i+1]=temp;
        }
    }
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,q,*a,i;
        long long sum;
        cin>>n>>q;
        a=(int*)malloc(n*sizeof(int));
        for(i=0;i<n;i++)
            cin>>a[i];
        bubble(a,n);
        sum=0;
        if(n==1)
            sum=0;
        else
        {
           for(i=0;i<n-1;i++)
                sum+=(a[i+1]-a[i]);
        }
        cout<<sum*q<<"\n";

    }
    return 0;
}
