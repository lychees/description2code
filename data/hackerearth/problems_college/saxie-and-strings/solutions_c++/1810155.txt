#include <iostream>
#include<cstring>
using namespace std;

int main()
{int t;
	cin>>t;
	while(t--)
{
   string s;
   int m;
   cin>>s;
   int flag=0;
   m=s.length();
   for(int i=0;i<m;i++)
   {
   	if(s[i]!=s[m-i-1])
   	{
   		flag=1;
   		break;
   	}
   }
   if(flag==1)
   {
   	cout<<"NO"<<endl;
   }
   else
   {
   	cout<<"YES"<<endl;
   }
   
   
}
}
