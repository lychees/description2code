#include<bits/stdc++.h>
using namespace std;
typedef long long lld;

int main()
{
	vector<string > v;
	
	lld p;
	cin>>p;
	
	while(p--)
	{
		string str;
		cin>>str;
		
		v.push_back(str);
		
	}
	lld l;
	cin>>l;
	lld k=0;
	char ch[105];
	while(l--)
	{   int flag=0;
		string s;
		cin>>s;
		
		for(lld i=0;i<v.size();i++)
		{
			if(s==v[i])
			 flag=1;
			
		}
		if(flag==1)
		{   
		 	continue;
		}
		else
		 ch[k++]=toupper(s[0]);
		
		
	}
	ch[k]='\0';
	for(lld i=0;i<strlen(ch);i++)
	{    if(i>0)
	      cout<<".";
	      
	      cout<<ch[i];
		
		
	}
	
	
}
