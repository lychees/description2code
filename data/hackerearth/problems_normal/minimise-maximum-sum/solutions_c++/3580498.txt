#include <iostream>
#include <algorithm>
using namespace std;

int main() {
	// your code goes here
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		n*=2;
		int a[n];
		for(int i=0;i<n;i++)
		{
			cin>>a[i];
		}
		sort(a,a+n);
		int i=0,j=n-1,max=0;
		while(i<n)
		{
			int sum=a[i]+a[j];
			if(max<sum)
			max=sum;
			i++;j--;
		}
		cout<<max<<endl;
	}
	return 0;
}
