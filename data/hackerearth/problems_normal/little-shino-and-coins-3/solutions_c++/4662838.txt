#include <iostream>
using namespace std;
int k;
string st;
int cnt[1000];
int ans;
int main(){
	cin >> k;
	cin >> st;
	for (int i = 0; i < st.size(); i++)
	{
		for (int j = 'a'; j <= 'z'; j++)
		{
			cnt[j] = 0;
		}
		int diff = 0;
		for (int l = i; l < st.size(); l++)
		{
			cnt[st[l]]++;
			if (cnt[st[l]] == 1)
				++diff;
			if (diff == k)
				++ans;
		}
	}
	cout << ans << endl;
	return 0;
}
