#include <iostream>
using namespace std;

int main()
{
	unsigned long long int i,t,n,nDayDistance,distSum;
	signed long long int milestone;
	unsigned long long int * arr;
	int lastSet;
	cin>>t;
	
	while(t--)
	{
		distSum = 0;
		lastSet = -1;
		cin>>n;
		arr = new unsigned long long int[n];
	
		for(i=0;i<n;i++)
		{
			cin>>arr[i];
			if(arr[i] > 0)
				lastSet = i;
			distSum += arr[i];
		}
		
		cin>>milestone;
		milestone = milestone - ((unsigned long long int)milestone/distSum)*distSum;
		
		if(milestone == 0){
			cout<<(lastSet+1)<<endl;
			continue;	
		}
		
		for(i=0;i<n;i++)
		{
			milestone -= arr[i];
			if(milestone<=0)
			{
				cout<<(i+1)<<endl;
				break;	
			}
		}
		
		delete [] arr;
	}
	
    return 0;
}
