	{
		cin >> a >> b;
		c = cc = 0;
		while (a > 0 || b > 0)
		{
			ra = a % 10;
			rb = b % 10;
			if (ra + rb + c > 9)
			{
				cc++;
				c = 1;
			}
			else
				c = 0;
			a /= 10;
			b /= 10;
		}
			ra = a % 10;

			{
				cc++;
				c = 1;
			}
			else
				c = 0;
				
		
		if (cc == 0)
			cout << "No carry operation" << endl;
		else if (cc == 1)
			cout << cc << " carry operation" << endl;
		else
			cout << cc << " carry operations" << endl;
			
			
	}
