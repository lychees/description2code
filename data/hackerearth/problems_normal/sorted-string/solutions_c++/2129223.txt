#include <iostream>
#include <bits/stdc++.h>
#include<string>
#include<cstring>
using namespace std;
int main()
{
    int test,i,j;
    string s;
    cin>>test;
    assert(1<=test<=100);
    while(test--)
    {
    	vector<pair<int,char>> a(26);
    	cin>>s;
    	for(i=0;i<26;i++)
    	{
    		a[i].first=0;
    		a[i].second=(i+'a');
    	}
    	for(i=0;i<s.size();i++)
    	{
    		a[s[i]-'a'].first++;
    	}
    	sort(a.begin(),a.end());
    	for(i=0;i<26;i++)
    	{
    		for(j=0;j<a[i].first;j++)
    			cout<<a[i].second;
    	}
    	cout<<endl;
    }
    return 0;
}
