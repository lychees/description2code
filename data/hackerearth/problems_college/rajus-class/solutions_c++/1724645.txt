#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        int a[n];
        string s[n];
        for(int i=0;i<n;i++)
        {
            cin>>a[i]>>s[i];
        }
        for(int i=0;i<n;i++)
        {
            for(int j=i;j<n;j++)
            {
                if(a[i]>a[j])
                {
                    a[i]^=a[j]^=a[i]^=a[j];
                    string tem=s[i];
                    s[i]=s[j];
                    s[j]=tem;
                }
            }
        }
        int ag;
        cin>>ag;
        cout<<s[ag-1]<<endl;
    }
    return 0;
}
