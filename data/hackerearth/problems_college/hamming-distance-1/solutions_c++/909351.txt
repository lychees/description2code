	int t,a,b,k,i,j;
	int x[100001];
	scanf("%d",&t);

		scanf("%d%d",&a,&b);
		if(a==b)puts("0");

		if(a<b) {
			k=a;
			a=b;
			b=k;
		}
		int count=0;
		while(a>=1) {
			if(a%2) {x[i]=1;count++;}
			else x[i]=0;
			i++;
			a/=2;
		}
		j=i;
		x[i]=a;
		if(a==1)count++;
		if(b==0) printf("%d\n",count);
		else {
		i=0;
		count=0;
		while(b>=1) {
			if(b%2) {
				if(x[i]!=1)count++;
			}
			else {
				if(x[i]!=0)count++;
			}
			i++;
			b/=2;
		}
		if(x[i]!=b)count++;
			while(j!=i) {
				if(x[++i]==1)count++;
			}
		printf("%d\n",count);
		}}
	}
	return 0;
}
