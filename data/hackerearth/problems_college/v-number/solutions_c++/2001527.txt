#include <iostream>
#include<string.h>
using namespace std;
 
int main()
{
    char a[1000000];
    long i,n,p,q,m,t,x;
    cin>>t;
    while(t--)
    {
    	scanf("%s",&a);
    	n=strlen(a);
    	//cout<<n;
    	p=0;q=0;
    	for(i=0;i<n/2;i++)
    	{
    		p+=a[i]-48;
    		q+=a[n-i-1]-48;
    	}
    	//cout<<p<<q;
    	
    	if(p==q)
    	{
    	cout<<"YES\n";
    	}
    	else
    	{
    	cout<<"NO\n";
    	}
    	
    }
    return 0;
}
