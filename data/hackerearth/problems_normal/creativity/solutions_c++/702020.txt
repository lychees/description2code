#include <iostream>
#include <string.h>
using namespace std;
char s1[1005],s2[1005];
int pi[1005],len;

void pre()
{
pi[0]=-1;
int p=-1,i;
len=strlen(s1);



for(i=1;i<len;i++)
{
	
	while(p>-1 && s1[i]!=s1[p+1])
	{
		p=pi[p];
	}
	
	if(s1[i]==s1[p+1])p++;
	
	pi[i]=p;
	
	
}
	
	
}

void find()
{

int p=-1,i;
bool found=false;

for(i=0;s2[i]!='\0';i++)
{

	
	while(p>-1 && s2[i]!=s1[p+1])
	{
		p=pi[p];
	}
	
	if(s2[i]==s1[p+1])p++;
	
	//cout<<p<<" ";

	if(p==len-1){found=true;break;}
}
	
	if(found)cout<<"possible\n"; else cout<<"not possible\n";
}
int main() {
	int t;
	cin>>t;
	cin.ignore();
	while(t--)
	{
	cin.getline(s1,1001);

	cin.getline(s2,1001);
	
	pre();
	
	find();
	
	}
	return 0;
}
