#include <bits/stdc++.h>

using namespace std;

char s[1000005];

int main()
{
    for ( int i = 1; i <= 882; i++ ) {
        int n;
        scanf("%s", s);
        n = (int)strlen(s);
        long long ans = 0;
        int idx = n-1;
        while ( idx >= 0 ) {
            if ( idx == 0 ) break;
            if ( s[idx] == '0' ) {
                while ( idx >= 0 && s[idx] == '0' ) {
                    ans++;
                    idx--;
                }
            }
            else {
                ans++;
                while ( idx >= 0 && s[idx] == '1' ) {
                    idx--;
                    ans++;
                }
                if ( idx == -1 ) break;
                else s[idx] = '1';
            }
        }
        printf("%lld\n", ans);
    }
    return 0;
}
