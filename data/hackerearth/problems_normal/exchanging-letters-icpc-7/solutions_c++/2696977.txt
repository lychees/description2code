#include <bits/stdc++.h>
#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;

int n,k,t;
vector< pair<int,int> >X(2005);

map< pair<int,int>, int> dp;

int solve(int i,int sum)
{
	//printf("%d %d\n",i,sum);
	
	if(sum>t)
	return 0;
	
	if(i==k&&sum==t)
	return 1;
	else if(i==k)
	return 0;

	if(dp.find(make_pair(i,sum))!=dp.end())
	return dp[make_pair(i,sum)];

	dp[make_pair(i,sum)]=(solve(i+1,sum+X[i].first)+solve(i+1,sum+X[i].second))%mod;

	return dp[make_pair(i,sum)];

}

int main()
{
    scanf("%d %d %d",&n,&k,&t);

    for(int i=0;i<k;i++)
    {
    	int num;
    	scanf("%d",&num);

    	X[i].first=num-1;
    	X[i].second=num+1;

    }

    printf("%d\n",solve(0,0));

    return 0;
}
