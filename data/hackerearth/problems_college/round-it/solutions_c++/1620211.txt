#include <iostream>

using namespace std;

int main()
{
	int t, n, b;
	cin >> t;
	while(t--)
	{
		int temp;
		cin >> n >> b;
		temp = n%b;
		if(b%2 == 0)
		{
			if(temp < b/2)
				cout << n - temp << endl;
			if(temp == b/2)
				cout << n + temp << endl;
			if(temp > b/2)
				cout << n + b - temp << endl;
		}
		if(b%2 != 0)
		{
			if(temp <= b/2)
				cout << n - temp << endl;
			if(temp > b/2)
				cout << n + b - temp << endl;
		}
	}
	return 0;
}
