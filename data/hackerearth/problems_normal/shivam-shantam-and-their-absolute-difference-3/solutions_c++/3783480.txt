#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
int main()
{
	int n;
	scanf("%d",&n);
	int a[n];
	int b[n];
	for (int i = 0; i < n; i++)
		scanf("%d",&a[i]);
	for (int i = 0; i < n; i++)
		scanf("%d",&b[i]);
	int q;
	scanf("%d",&q);
	sort(a,a+n);
	sort(b,b+n);
	long long int s1[400001];
	long long int s2[400001];
	memset(s1,0,sizeof(s1));
	memset(s2,0,sizeof(s2));
	s1[0] = 1;
	s2[0] = 1;
 	for (int i = 0; i < n; i++)
 	{
 		for (int j = 400000; j >= 0; j--)
 		{
 			if (s1[j] != 0 && (j+a[i]) <= 400000)
 			{
 				s1[j+a[i]] += s1[j];
 				s1[j+a[i]] %= mod;
 			}
 			if (s2[j] != 0 && (j+b[i]) <= 400000)
 			{
 				s2[j+b[i]] += s2[j];
 				s2[j+b[i]] %= mod;
 			}
 		}
 	}
 	for (int i = 1; i <= 400000; i++)
 	{
 		s2[i] = s2[i]+s2[i-1];
 		s2[i] %= mod;
 	}
 	long long int ans = 0;
 	for (int i = 0; i <= 400000; i++)
 	{
 		if (s1[i] != 0)
 		{
 			long long int k2;
 			if ((i-q-1) >= 0)
 				k2 = s2[min(i+q,400000)]-s2[(i-q-1)];
 			else
 				k2 = s2[min(i+q,400000)];
 			while (k2 < 0)
 			{
 				k2 += mod;
 			}
 			k2 %= mod;
 			ans = ans+(s1[i]*k2)%mod;
 			ans %= mod;
 		//	cout << ans << endl;
 		}
 	}
 	cout << ans << endl;
    return 0;
}
