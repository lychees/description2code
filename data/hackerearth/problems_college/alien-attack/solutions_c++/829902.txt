using namespace std;
const long MOD = 1000000007;
const int MAX = 105;

int main()
{
    //freopen("input.txt", "r", stdin);
    int t, k = 0;
    long a;
    pair <long, int> p[MAX];
    ll x, y, z, B[MAX];
    cin >> t;
    for(int i = 0;i < t;++i)
    {
        cin >> a;
        if(a == 1) B[i] = 2;
        else if(a == 2) B[i] = 7;
        else
            p[k++] = make_pair(a, i);
    }
    sort(p, p + k);
    long j = 2;
    x = 2, y = 7, z = 11;
    for(int i = 0;i < k;)
    {
        while(j < p[i].first)
        {
            z = y + 2*x;
            z %= MOD;
            x = y, y = z;
            //cout << x << ' ' << y << endl;
            j++;
        }
        while(i < k and j == p[i].first)
        {
            B[p[i].second] = z; 
            i++;
            //cout << i << " :P" << endl;
        }
    }
    for(int i = 0;i < t;++i)
        printf("%lld\n", B[i]);
    return 0;
}
