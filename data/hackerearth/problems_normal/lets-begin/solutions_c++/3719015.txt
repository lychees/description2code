#include <bits/stdc++.h>
using namespace std;

int main()
{
   long long t,a[1001000]={0},max=8,in,i;
   cin>>t;
   a[2]=1;
   a[3]=1;
   a[4]=2;
   a[5]=1;
   a[6]=2;
   a[7]=1;
   a[8]=2;
   while(t--)
   {
   	cin>>in;
   	if(in<2)
    cout<<"-1\n";
   	else if(in<=max)
   	cout<<a[in]<<endl;
   	else
   	{
   		for(i=max+1;i<=in;i++)
   		{
   			a[i]=min(min(min(a[i-2],a[i-3]),a[i-5]),a[i-7])+1;
   		}
    max=in;
    cout<<a[in]<<endl;
   	}
   	}
   return 0;
}
