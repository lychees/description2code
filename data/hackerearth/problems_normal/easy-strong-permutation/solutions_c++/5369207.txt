#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    long int n,i,j,k,ans=0;
    scanf("%ld",&n);
    long int a[n],b[n];
    for(i=0;i<n;i++)
    scanf("%ld",&a[i]);
    sort(a,a+n);
    for(i=0;i<n/2;i++)
    b[2*i]=a[i];
    k=1;
    for(i=1;i<n;i+=2)
    {
    	b[i]=a[(n/2)*2-k];
    	k+=1;
    }
    for(i=0;i<n-1;i++)
    {
    	if(b[i+1]>=b[i])
    	ans=ans+b[i+1]-b[i];
    	else
    	ans=ans+b[i]-b[i+1];
    }
    if(b[n-1]>=b[0])
    ans=ans+b[n-1]-b[0];
    else
    ans=ans+b[0]-b[n-1];
    printf("%ld",ans);
    return 0;
}
