using namespace std;
#define in(n) scanf("%lld",&n)

	
long long hcf(long long a,long long b)
{
    for (;;)
    {
        if (a == 0) return b;
        b %= a;
        if (b == 0) return a;
        a %= b;
    }
}

long long abso(long long a)
{
	if(a<0)
		return -a;
	return a;
}

int main()
{
	pair<long long,long long> point;
	
	long long n,x1,y1;
	long long x,y;
	long long prevx,prevy;
	long long ans;
	long long b,i;
	long long sum=0;
	
	in(n);
	in(x);
	in(y);
	
	x1=x;
	y1=y;
	//b=1;
	for(i=0;i<n-1;i++)
	{
		prevx=x;
		prevy=y;
		
		in(x);
		in(y);
		sum+=(prevx*y);
		sum-=(prevy*x);
		
		b+=hcf(abso(x-prevx),abso(y-prevy));
	}
	b+=hcf(abso(x-x1),abso(y-y1));
		
	sum+=(x*y1);
	sum-=(y*x1);
	if(sum<0)
		sum=-sum;
	
	
	ans=(sum+2-b)/2;
	cout<<ans<<endl;
	
	return 0;
}
