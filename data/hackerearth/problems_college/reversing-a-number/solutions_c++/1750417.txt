#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int main()
{
    int t,len,num,j;
    char str[102];
    scanf("%d",&t);
    while(t--)
    {
        scanf("%s",str);
        len=strlen(str);
        if(len==1)
        {
            num=str[0]-48;
            if(num%2)
                printf("ODD\n");
            else
                printf("EVEN\n");
        }
        else
        {
            j=0;
            while(str[j]=='0')
            {
                j++;
            }
            if(j>len)
                printf("EVEN\n");
            else
            {
                num=str[j]-48;
                if(num%2)
                    printf("ODD\n");
                else
                    printf("EVEN\n");
            }
        }
    }
    return 0;
}
