#include <cstdio>
#include <vector>

int main()
{
    int n;
    scanf("%d", &n);

    std::vector<std::vector<int>> dist(n, std::vector<int>(n, 0));
    for (int i = 0; i < n; ++i)
        for (int j = 0; j < n; ++j)
            scanf("%d", &dist[i][j]);

    for (int k = 0; k < n; ++k)
        for (int i = 0; i < n; ++i)
            for (int j = 0; j < n; ++j)
                if (dist[i][j] > dist[i][k] + dist[k][j])
                    dist[i][j] = dist[i][k] + dist[k][j];

    int m;
    scanf("%d", &m);
    while (m--)
    {
        int s, g, d;
        scanf("%d %d %d", &s, &g, &d);

        printf("%d %d\n", dist[s][g] + dist[g][d], dist[s][g] + dist[g][d] - dist[s][d]);
    }

    return 0;
}
