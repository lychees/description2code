#include <stdio.h>
#include<string>
using namespace std;
 
 
int main()
{
    int i,t,j;
    char word[40];
    bool count[26];
    scanf("%d",&t);
    string ans;
    while(t--)
    {
        ans="";
        for(i=0;i<26;i++)
          count[i]=false;
        scanf("%s",word);
        i=0;
        while(word[i]!='\0')
        {
                if(!count[word[i]-'a'])//first time encountered
                        {
                                count[word[i]-'a']=true;
                                ans+=word[i];
                        }
                i++;
        }
        printf("%s\n",ans.c_str());
          
    }
    
    
    return 0;
}
