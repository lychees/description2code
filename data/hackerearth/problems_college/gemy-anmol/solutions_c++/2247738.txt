#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
 //   cout << "Hello World!" << endl;
 int t;
 scanf("%d",&t);
 while(t--)
 {
 	int n;
 	scanf("%d",&n);
 	ll m;
 	scanf("%lld",&m);
 	vector<ll> v;
 	int i;
 	ll x;
 	for(i=0;i<n;i++)
 	{
 		scanf("%lld",&x);
 		v.push_back(x);
 	}
 	sort(v.begin(),v.end());
 	reverse(v.begin(),v.end());
 	ll total = 0;
 	for(i=0;i<n;i++)
 	{
 		total+=v[i];
 		if(total>=m)break;
 	}
 	if(total<m)i = -2;
 	printf("%d\n",i+1);
 }
    return 0;
}
