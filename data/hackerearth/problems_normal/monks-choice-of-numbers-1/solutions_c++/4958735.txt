#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
  	int test, temp;
  	cin >> test;
  	
  	while(test--){
		int n, k;
		cin >> n >> k;
		int numbers[n];
		vector<int> count(n, 0);
		
		for(int i = 0; i < n; i++){
			cin >> numbers[i];
			
			temp = numbers[i];
			while(temp){
				temp = temp&(temp-1);
				count[i]++;
			}
		}
			
		sort(count.begin(), count.end(), greater<int>());
		
		int sum = 0;
		for(int i = 0; i < k; i++)
			sum += count[i];
		
		cout << sum << endl;
  		
  	}
    return 0;
}
