#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cmath>
#include<cstring>
#include<vector>
#include<queue>
#include<stack>
#include<map>
#include<set>
#include<algorithm>
using namespace std;
int cnt[10000001];
int main(){
  int t,n,q;
  set<int> rems;
  scanf("%d",&t);
  while(t--){
    scanf("%d %d",&n,&q);
    if(n==0){
      rems.clear();
      for(int i=0;i<=24;i++){
        rems.insert(q/(1<<i));
      }
      for(set<int>::iterator it=rems.begin();it!=rems.end();it++){
       cnt[*it]++;
      }
    } else if(n==1){
      rems.clear();
      for(int i=0;i<=24;i++){
        rems.insert(q/(1<<i));
      }
      for(set<int>::iterator it=rems.begin();it!=rems.end();it++){
        cnt[*it]--;
      }
    } else {
      printf("%d\n",cnt[q]);
    }
  }
  return 0;
}
