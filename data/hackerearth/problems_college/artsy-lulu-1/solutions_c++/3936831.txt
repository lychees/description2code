#include <iostream>
using namespace std;

int main()
{
    int h,i,j;
    char ch;
    while(cin>>ch>>h){
        if(h<2)  {
        	cout<<"INVALID\n\n";
        }else{
        	if(!(((65<=ch+h-1)&&(ch+h-1<=90))||((97<=ch+h-1)&&(ch+h-1<=122))))
        	   cout<<"INVALID\n\n";
        	else{
        		for(i=0; i<h; i++){
        			for(j=0; j<=i; j++)
        			  cout<<char(ch+i)<<" ";
        			cout<<endl;  
        		}
        		cout<<endl;
        	}   
        }  
        //
    }
    return 0;
}
