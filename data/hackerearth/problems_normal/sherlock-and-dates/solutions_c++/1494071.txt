#include<iostream>
#include<stdio.h>
using namespace std;
int d1,m1,y1,d2,m2,y2;
bool isInRange(int d,int m,int y)
{
  if(y<y1|| y>y2)return false;
  if(y==y1 && m<m1)return false;
  if(y==y2 && m>m2) return false;
  if(y==y1 && m==m1 && d<d1)return false;
  if(y==y2 && m==m2&& d>d2)return false;
  return true;
}
int main()
{
  int t;
  cin>>t;
  
  while(t--)
  {

    int count=0;
    scanf("%02d:%02d:%04d  %02d:%02d:%04d",&d1,&m1,&y1,&d2,&m2,&y2);
    for(int d=1;d<12;d++)
    {
    	int m=d+1;
    	for(int rem=0;rem*100+m+1<=y2;rem++)
    	{
    		if(isInRange(d,m,rem*100+m+1))
    		count++;
    	}
    	
    }
   cout<<count<<"\n";
  }
  return 0;
}
