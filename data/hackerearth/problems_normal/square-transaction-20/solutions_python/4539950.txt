n=int(raw_input())
stock=map(int,raw_input().split())
for i in range(1,n):
	stock[i]+=stock[i-1]

for i in range(int(raw_input())):
	val=int(raw_input())
	s=0
	e=n-1
	if stock[e]<val:
		print -1
		continue
	if stock[s]>val:
		print 1
		continue
	else:
		while s < e:
			if e - s == 1:
				if val == stock[s]:
					print s+1
				else:
					print s+2
				break
			cur = ( s + e ) / 2
			if stock[cur] <= val:
				 s = cur
			else:
				e = cur
