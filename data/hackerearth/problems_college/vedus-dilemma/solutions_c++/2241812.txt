#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
using namespace std;

int gcd(int a, int b)
{
	if (b==0)
		return a;
	else
		return gcd(b, a%b);
}

int main ()
{
	int t;
	cin >> t;
	while (t--)
	{
		int n;
		cin >> n;
		int a[n];
		for (int i=0; i<n; i++)
			cin >> a[i];
		int p, q;
		cin >> p >> q;
		if (q > n)
		{
			cout << "-1" << endl;
			continue;
		}
		else
		{
			vector <int> v;
			int ctr = 0;
			for (int i=0; i<n; i++)
			{
				if (gcd (p, a[i]) > 1)
				{
					v.push_back(a[i]);
					ctr++;
				}
			}
			sort (v.begin(), v.end());
			if (ctr < q)
			{
				cout << "-1" << endl;
				continue;
			}
			else if (ctr == q)
			{
				for (int i=0; i<ctr; i++)
				{
					cout << v[i] << " ";
				}
				cout << endl;
				continue;
			}
			else
			{
				int del = ctr - q;
				while (del--)
				{
					if (fabs(v.front() - p) > fabs(v.back() - p))
					{
						v.erase(v.begin());
					}
					else
					{
						v.erase(v.end()-1);
					}
				}
				for (int i=0; i<v.size(); i++)
				{
					cout << v[i] << " ";
				}
				cout << endl;
				continue;
			}
		}
	}
}
