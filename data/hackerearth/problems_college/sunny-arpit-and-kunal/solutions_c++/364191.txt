#include <iostream>
using namespace std;

int main()
{
	int times;
	cin >> times;

	while(times>0)
	{
		int a, b, c;
		cin >> a >> b >> c;
		if(a>b)
		{
			int temp=a;
			a=b;
			b=temp;
		}
		if(a>c)
		{
			int temp=a;
			a=c;
			c=temp;
		}
		if(b>c)
		{
			int temp=b;
			b=c;
			c=temp;
		}

		int p = b-a;
		int q = c-b;

		if(p>q)
		{
			cout << p-1 << endl;
		}
		else 
			cout << q-1 << endl;;

		times--;
	}
	return 0;
}
