
def hist(mystr):
    myD = dict()
    for i in mystr:
        if i in myD:
            myD[i] += 1
        else:
            myD[i] = 1
    return myD


a = raw_input()
myMap = hist(a)
n = max(myMap.values())
length = [i for i, j in myMap.iteritems() if j == n]
length.sort(key=lambda seq: ord(seq))
value = length[0]
print value, n
