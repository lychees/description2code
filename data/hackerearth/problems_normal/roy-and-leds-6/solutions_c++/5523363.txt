#include <iostream>
using namespace std;

int main()
{
    long long int T,R,G,B;
    long long int r_f=0,g_f=0,b_f=0,r_c=0,g_c=0,b_c=0;
    long long int r=0,g=0,b=0,y=0,c=0,m=0,w=0,bl=0;
    cin >> T >> R >> G >> B;
    while(T!=0){
    	if(r_f==0 && g_f==0 && b_f==0)		bl++;
    	else if(r_f==1 && g_f==0 && b_f==0)	r++;
    	else if(r_f==0 && g_f==1 && b_f==0)	g++;
    	else if(r_f==0 && g_f==0 && b_f==1)	b++;
    	else if(r_f==1 && g_f==1 && b_f==0)	y++;
    	else if(r_f==0 && g_f==1 && b_f==1)	c++;
    	else if(r_f==1 && g_f==0 && b_f==1)	m++;
    	else w++;
    	
    	r_c++;g_c++;b_c++;
    	if(r_c==R){
    		r_c=0;
    		if(r_f==0)	r_f=1;
    		else		r_f=0;
    	}
    	if(g_c==G){
    		g_c=0;
    		if(g_f==0)	g_f=1;
    		else		g_f=0;
    	}
    	if(b_c==B){
    		b_c=0;
    		if(b_f==0)	b_f=1;
    		else		b_f=0;
    	}
    	T--;
    }
    cout<<r<<" "<<g<<" "<<b<<" "<<y<<" "<<c<<" "<<m<<" "<<w<<" "<<bl; 
    return 0;
}
