#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<cstring>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"
int cs,n,i;
int A[1000001];
int main()
{
  scanf("%d",&cs);
  while(cs--)
  {
    scanf("%d",&n);
    for(i=0;i<n;i++)
      scanf("%d",&A[i]);
    if(!prev_permutation(A,A+n))
      next_permutation(A,A+n);
    for(i=0;i<n;i++)
    {
      printf("%d",A[i]);
      if(i+1==n)
        printf("\n");
      else
        printf(" ");
    }
  }
	return 0;
}
