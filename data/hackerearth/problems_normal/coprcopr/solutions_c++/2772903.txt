#include<bits/stdc++.h>
using namespace std;
int ar[10000007];
bool setprime[10000007];
int main() {
    int i,j,t,n,k;
    for(i = 0; i < 10000007; i++)
    ar[i] = i;
    memset(setprime,true,sizeof setprime);
    for(i = 2; i < 10000007; i++) {
    	if(setprime[i]) {
    		for(j = 2*i; j < 10000007; j += i) {
    			setprime[j] = false;
    			ar[j] = (ar[j]/ar[i])*(ar[i]-1);
    		}
    		ar[i] = ar[i]-1;
    	}
    }
    long long int ans;
    scanf("%d",&t);
    while(t--) {
    	ans = 0;
    	scanf("%d",&n);
    	for(i = 1; i <= n; i++) {
    		ans += ar[i];
    	}
    	printf("%lld\n",ans);
    }
    return 0;
}
