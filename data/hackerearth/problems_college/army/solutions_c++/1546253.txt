#include<iostream>
#include<stdlib.h>
#include <stdio.h>
#include<math.h>
#include<string.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	int q;
	cin>>q;
	while(q--)
	{
		int hash[11]={0};
		int m;
		cin>>m;
		for(int i=0;i<m;i++)
		{
			int x;
			cin>>x;
			hash[x]=1;
		}
		int cnt=0;
		for(int i=1;i<=n;i++)
		{
			int temp=i;
			while(temp!=0)
			{
				if(hash[temp%10]==1)
				{
					cnt++;
					break;
				}
				temp/=10;
			}
		}
		cout<<cnt<<endl;
	}
     
}
