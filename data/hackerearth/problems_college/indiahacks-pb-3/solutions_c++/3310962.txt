#define INF 2000000000 // infinity
#define ll long long

ll memo[1000000];
ll solve(ll a[], ll n, ll k,ll i)
{
    if(i>=n)
        return 0;
    if(memo[i]!=-1) return memo[i];
    
    ll ret = -INF;
    ret = max(ret, solve(a,n,k,i+1));
    ret = max(ret, a[i]+solve(a,n,k,i+k+1));
    return memo[i] = ret;
}
int main()
{
	int t;
	si(t);
	while(t--)
    {
        memset (memo, -1, sizeof (memo));
        ll n,k;
        sll(n); sll(k);
        ll a[n];
        FOR(i,0,n) sll(a[i]);
        
        cout<<solve(a,n,k,0)<<endl;
    }
	return 0;
}
