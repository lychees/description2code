test = int(raw_input())
for _ in range(test):
    N = int(raw_input())
    arr = map(int, raw_input().split())
    arr = sorted(arr)
    if N % 2 == 0:
        median = (arr[(N/2)] + arr[(N/2)-1]) / 2
    else:
        median = arr[(N/2)]
    avg = sum(arr) / N
    s1 = 0
    s2 = 0
    for i in arr:
        s1 += abs(i-median)
        s2 += abs(i-avg)
    print min(s1, s2)
