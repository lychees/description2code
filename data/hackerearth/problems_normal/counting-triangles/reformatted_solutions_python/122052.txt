n = int(raw_input())
a = []
for i in xrange(n):
    a.append(raw_input().split(" "))
    a[i][0] = int(a[i][0])
    a[i][1] = int(a[i][1])
    a[i][2] = int(a[i][2])
    a[i].sort()

a.sort(key=lambda x: x[2])
a.sort(key=lambda x: x[1])
a.sort(key=lambda x: x[0])

cnt = 0

if a[0] != a[1]:
    cnt += 1

if a[n-2] != a[n-1]:
    cnt += 1

for i in xrange(n-2):
    if (a[i+1] == a[i+2]) or (a[i] == a[i+1]):
        None
    else:
        cnt += 1

print cnt
