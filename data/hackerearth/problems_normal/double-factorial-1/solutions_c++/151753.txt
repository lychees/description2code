#define MOD 1000000007

typedef unsigned long long LLD;

int main()
{
    int t;
    char s[] = "8401733239439360000";
    cin >> t;
    while(t--)
    {
	LLD n, m;
	cin >> n >> m;
	if(n >= 5)
	{
	    cout << "0\n";
	    continue;
	}
	if(n == 4)
	{
	    int idx = 19-m;
	    if(m == 17)
		cout << 1733239439360000 << endl;
	    else if(m > 4)
		cout << s+idx << endl;
	    else
		cout << 0 << endl;
	    continue;
	}

	LLD p = 1;
	for(int i = 1; i <= m; i ++)
	    p *= 10;

	LLD f = 1;
	for(LLD i = 1; i <= n; i ++)
	    f = f*i;

	LLD ans = 1;
	for(LLD i = 1; i <= f; i ++)
	    ans = (ans * i) % p;
	
	cout << ans << endl;
    }
    return 0;
}
