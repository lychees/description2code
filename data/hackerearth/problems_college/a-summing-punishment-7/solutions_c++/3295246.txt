#include <bits/stdc++.h>
using namespace std;

#define MAX 403

int getMax(int a, int b) {return (a > b ? a : b);}
int getMin(int a, int b) {return (a < b ? a : b);}

int main() {
    int i, n, l, r, q, ans;
    cin>>n;
    int a[n], pref[n];
    for (i=0; i<n; i++) cin>>a[i];
    pref[0] = a[0];
    for (i=1; i<n; i++) pref[i] = a[i] + pref[i-1];
    cin>>q;
    while (q--) {
        cin>>l>>r;
        l--, r--;
        ans = pref[r];
        if (l > 0) ans -= pref[l-1];
        cout<<ans<<endl;
    }
	return 0;
}
