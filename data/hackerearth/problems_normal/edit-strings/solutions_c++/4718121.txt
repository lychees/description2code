#include <iostream>
#include <string>
using namespace std;

int main()
{
    int l,n,output;
    cin>>l>>n;
    int pos;
    char rep;
    string s="";
    cin>>s;
    for(int i=0;i<n;i++)
    {
    	output=0;
    		cin>>pos;
    		cin>>rep;
    		s[pos-1]=rep;
    	
    		for(int j=0;j<l-1;j++)
    		{
    			if(s[j]=='.'&&s[j+1]=='.')
    			{
    				output++;
    			}
    		}
    		cout<<output<<endl;
    }
    
    return 0;
}
