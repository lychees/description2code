#include<bits/stdc++.h>
#define mp make_pair
# define pb push_back
using namespace std;
typedef long long lo;
int main()
{
        lo t,l,l1,i,c0,c1,c01,c10,diff,steps;
        char a[1000000],b[1000000];
    cin>>t;lo v=1;
    while(t--)
    {
        steps=0;
        scanf("%s",a);
        scanf("%s",b);
        c0=c1=c01=c10=0;
        l=strlen(a);l1=strlen(b);
        if(l!=l1)
        { printf("Case %lld: -1\n",v++);continue;}
        for(i=0;i<l;i++)
        {
                if(a[i]=='?'&&b[i]=='0')
                c0++;
                else if(a[i]=='?'&&b[i]=='1')
                c1++;
                else if(a[i]=='0'&&b[i]=='1')
                c01++;
                else if(a[i]=='1'&&b[i]=='0')
                c10++;
        }
        if(c01>=c10)
        {
                diff=c01-c10;
                steps=c01+c1+c0;
               
        }
        else
        {
                diff=c10-c01;
                steps+=c01;
                if(c1<diff)
                {
                        steps=-1;
                }
                else
                {
                        steps+=c0+(2*diff+(c1-diff));
                }
        }
        printf("Case %lld: %lld\n",v++,steps);
    }    
   return 0;    
}
