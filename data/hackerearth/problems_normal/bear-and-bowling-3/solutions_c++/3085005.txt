#include<bits/stdc++.h>
using namespace std;
typedef long long int uli;
const uli mod=1e9+7;

int main(){
   int n;
   scanf("%d",&n);
   uli ans=0,pwr=1,v;
   for(int i=0;i<n-2;i++)pwr=(pwr+pwr)%mod;
   for(uli i=1;i<=n;i++){
      scanf("%lld",&v);
      ans+=pwr*(i+1)%mod*v%mod;
   }
   if(n==1)ans=v;
   ans%=mod;
   printf("%lld\n",ans);
   return 0;
}
