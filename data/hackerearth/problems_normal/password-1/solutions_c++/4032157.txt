#include <bits/stdc++.h>
using namespace std;
string arr [300];

int main()
{

    int n;
    cin >> n;
    for(int i = 0; i < n; i++)
        cin >> arr[i];
    for(int i = 0; i < n; i++){
        for(int j = i + 1; j < n; j++){
            if(arr[i].size() != arr[j].size()) continue;        // check if the 2 words don't have the same length
            bool flag = false;
            for(int k = 0, h = arr[i].size() - 1; k < arr[j].size(); k++, h--){
                if(arr[i][k] != arr[j][h]){                     // check if the 2 words aren't palindrome
                    flag = true;
                    break;
                }
            }
            if(!flag){
                cout << arr[i].size() << " " << arr[i][ arr[i].size() / 2 ];        // print in case of palindrome passwords
                return 0;
            }
        }
    }
    return 0;
}
