#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t , n , m, ans;

    cin >> t;

    while(t--){

        ans = 0;

        cin >> n;

        for(int i = 1; i * i <= n; i++){

            for(int j = i; j <= n; j++){

                if(i * j > n) break;
                ans++;
            }
        }

        cout << ans << "\n";
    }

    return 0;
}
