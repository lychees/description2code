#include <iostream>
using namespace std;
#include<bits/stdc++.h>
int main()
{
	long long n;
	cin>>n;
	string s;
	cin>>s;
	long long ans1=0,ans2=0,cnt=0;
	for(long long i=0;i<n;i++){
		if(s[i]=='v'){
			cnt++;
			continue;
		}
		if(s[i]=='w'){
			cnt+=2;
			continue;
		}
		ans1+=cnt;
		ans2+=cnt/2+cnt%2;
		cnt=0;
		ans1++;
		ans2++;
	}
	ans1+=cnt;
	ans2+=cnt/2+cnt%2;
	
    cout << ans2<<" "<<ans1 << endl;
    return 0;
}
