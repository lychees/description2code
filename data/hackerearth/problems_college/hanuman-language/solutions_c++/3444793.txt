#include <bits/stdc++.h>
using namespace std;

int main()
{
  long int t,i,j,k,l,m,n;
  cin>>t;
  cin.ignore();
  	
  while(t--)
  {
  	string str="",st="";
  	getline(cin,str);
  	//cout<<str;
  	k=str.length();
  	vector <string> v;
  	for(i=0;i<k;i++)
  	{
  		if(str[i]==' ' )
  		{
  	//		cout<<st;
  			v.push_back(st);
  			st="";
  		}
  		else
  		st+=str[i];
  	}
  	
  	if(st.size()>=1)
  	v.push_back(st);
  	
  	for(i=v.size()-1;i>=0;i--)
  	cout<<v[i]<<" ";
  	cout<<"\n";
  	
  }
	
}
