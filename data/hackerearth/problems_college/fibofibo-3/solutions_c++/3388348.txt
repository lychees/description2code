#include <iostream>
using namespace std;

long cux=0,cuy=0;

void foo(int n)
{
	if(n==0)
	{
		cux++;
		return ;
	}
	if(n==1)
	{
		cuy++;
		return ;
	}
	
	foo(n-1);
	foo(n-2);
}


int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	cux=0;
    	cuy=0;
    	int n;
    	cin>>n;
    	
    	foo(n);
    	cout<<cux<<" "<<cuy<<endl;
    }
    return 0;
}
