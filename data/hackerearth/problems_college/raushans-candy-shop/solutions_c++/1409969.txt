#include <cstdio>
#include <cstring>
#include <string>
#include <map>
#include <cassert>

using namespace std;

map<int,int> have;
char s[1000002];

int main() {
int n;
    scanf("%d%s",&n,s);
    assert((n >= 1) && (n <= 1000000));
    assert(strlen(s) == n);
    have[0] = 1;
    int sum = 0;
    long long answer = 0;
    for (int i = 0; i < n; ++i) {
        assert((s[i] == 'C') || (s[i] == 'S'));
        if (s[i] == 'C') {
            --sum;
        }
        else {
            sum += 2;
        }
        map<int,int>::iterator t = have.find(sum);
        if (t != have.end()) {
            answer += t->second;
        }
        ++have[sum];
    }
    printf("%lld\n",answer);
    return 0;
}
