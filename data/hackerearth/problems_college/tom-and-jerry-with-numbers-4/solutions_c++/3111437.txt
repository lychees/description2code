#include<bits/stdc++.h>
int main()
{
   int t;
   scanf("%d",&t);
   long long int n,m1,m2,d,k;
   while(t>0)
   {
      scanf("%lld%lld%lld",&n,&m1,&m2);
      if(n<m1 && n<m2)
         printf("0\n");
      else
         {
            n=n%(m1+m2);
            if(n>=m1 && n>=m2)
            {
                d=n/m1;
                k=n/m2;
                if(d%2!=0 || k%2!=0)
                {
                    printf("1\n");
                }
                else
                {
                    printf("0\n");
                }
            }
            else if(n>=m1)
             {
                d=n/m1;
                if(d%2!=0)
                    printf("1\n");
                else
                    printf("0\n");
             }
             else
             {
                 d=n/m2;
                 if(d%2!=0)
                    printf("1\n");
                 else
                    printf("0\n");
             }
         }
      t--;
   }
   return 0;
}
