#include <bits/stdc++.h>
using namespace std;

void solve(long long n) {
	long long lo = 1, hi = 1e9;
	while(hi - lo > 1) {
		long long mi = lo + ((hi - lo) / 2);
		if((mi*(mi + 1))/ 2 >= n) {
			hi = mi;
		}
		else 
			lo = mi;
	}
	if((lo*(lo + 1))/ 2 >= n) {
		cout << lo << "\n";
		return ;
	}
	cout << hi << "\n";
	
}

int main() {
	long long t;
	cin >> t;
	for(int i = 0;i < t;i++) {
		long long n;
		cin >> n;
		solve(n);
	}
	return 0;
}
