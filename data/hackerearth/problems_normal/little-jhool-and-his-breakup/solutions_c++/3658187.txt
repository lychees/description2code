#include <iostream>
using namespace std;

int main()
{
	char c;
	bool l = false;
	bool o = false;
	bool v = false;
	bool e = false;
    while (cin>>c) {
    	if (c=='l') l = true;
    	if (c=='o' && l) o = true;
    	if (c=='v' && o) v = true;
    	if (c=='e' && v) e = true;
    }
    if (l&&o&&v&&e) cout<<"I love you, too!"; else cout<<"Let us breakup!";
    return 0;
}
