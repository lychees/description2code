#include <iostream>
#include <cstdio>
#include <cstring>
#include <cstdlib>
using namespace std;

long long int knapsack(long long int* litres,long long int* price,long long int L){
	
	long long int mat[L+1];
	mat[0]=0;
	for(int i=1;i<=L;i++){
		long long int maximum=mat[i-1];
		for(int j=0;j<L;j++){
			long long int x=i-litres[j];
			if(x>=0 && (mat[x]+price[j])>maximum){
				maximum=mat[x]+price[j];
			}
			mat[i]=maximum;
		}
	}
	return mat[L];
}
int main()
{
	long long int T,L;
	cin>>T;
	while(T--){
		cin>>L;
		long long int *litres=new long long int[L];
		long long int *price=new long long int[L];
		int i;
		i=0;
		while(i<L){
			litres[i]=i+1;
			cin>>price[i];
			i++;
		}
		cout<<knapsack(litres,price,L)<<endl;
	}
    return 0;
}
