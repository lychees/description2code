#include <bits/stdc++.h>
#define ll long long
using namespace std;
int myfind(int n)
{
	if(n==1||n==2)
	return 1;
	if(n&1)
	return 2*myfind(n/2) +1;
	else
	return 2*myfind(n/2)-1;
}
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int test;
	cin>>test;
	while(test--)
	{
		ll int n;
		cin>>n;
		cout<<myfind(n)<<endl;
	}
	return 0;
}
