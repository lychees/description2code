#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n, i, j;
    
    cin >> n;
    string a[n];
    
    for (i = 0; i < n; i++) {
    	cin >> a[i];
    }
    
    for (i = 0; i < n; i++) {
    	reverse(a[i].begin(), a[i].end());
    	//cout << a[i] << endl;
    	for (j = i + 1; j < n; j++) {
	    	if (a[j] == a[i]) {
	    		cout << a[i].length() << ' ' << a[i][a[i].length() / 2] << endl;
	    	}
	    }
    }
    
    return 0;
}
