#include <bits/stdc++.h>
#define mod 10000007 
using namespace std; 
long long modg(long long int f)
{
    long long yu=1;
    while(f--)
    {
        yu*=2;
        yu=yu%mod;
    }
    return yu;
}
 
long long int runk(long long int M, long long int N, long long int a[20][100001], long long int b[], long long int p) {
        
        long long int ans=0;
     
 
        
        vector<bool> used(M,false);
 
        
        for (int j=0; j<N; j++){
            int i = 0;
            while ( (i < M) && (used[i] || (a[i][j] == 0) ) ) {
 
                i++;
            }
            if(i == M) continue;
            ans++;
            used[i] = true;
           
            for (int k=0; k<M; k++) if(!used[k]){
                
                long long int coef = 0;
                while ( (a[i][j] * coef + a[k][j]) % p != 0) {
                    coef++;
                }
                for (int l=0; l<N; l++) {
                     a[k][l] = (a[k][l] + a[i][l] * coef) % p;
                }
                b[k] = (b[k] + b[i] * coef) % p;
            }
        }
 
        
        for (int i=0; i<M; i++) {
             if(!used[i] && b[i] != 0) {
 
                  return -1;
             }
        }
 
        return ans;
    }

                    if(d==0)
                    dp[j][i]=0;
                    else
                    dp[j][i]=1;
                    t=t/2;
                }
