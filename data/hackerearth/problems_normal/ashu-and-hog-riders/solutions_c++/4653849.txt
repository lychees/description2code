#include<iostream>
using namespace std;

int n, ans, counter;
int grid[1005][1005];

void dfs(int x, int y){
	
	counter++;	
	grid[x][y] = 0;
	if(x+1<n && grid[x+1][y])
		dfs(x+1, y);
	if(x-1>=0 && grid[x-1][y])
		dfs(x-1, y);
	if(y+1<n && grid[x][y+1])
		dfs(x, y+1);
	if(y-1>=0 && grid[x][y-1])
		dfs(x, y-1);
}

int main(){
	
	scanf("%d",&n);
	ans = 0;
	for(int i =0; i<n; i++)
		for(int j =0; j<n; j++)
			scanf("%d", &grid[i][j]);
	
	for(int i =0; i<n; i++)
		for(int j =0; j<n; j++){
			counter = 0;
			if(grid[i][j] == 1)
				dfs(i, j);
			if(counter>ans)
				ans = counter;	
		}			
		
	printf("%d\n",ans);
}
