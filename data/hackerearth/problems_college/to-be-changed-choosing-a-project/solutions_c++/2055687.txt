#include <iostream>
using namespace std;

int main()
{
    long int n,r,p=0,t=0,c=0,j;
    scanf("%ld%ld\n",&n,&r);
    long int a[n],i;
    for(i=0;i<n;i++){
    	a[i]=i+1;
    }
    while(1){t=0;c=0;
    	for(i=p;;i++){
    		t++;
    		if(t==r)
    		{	if(i==n-1)
    				p=0;
    			else
    				p=i;
    			for(j=i;j<n;j++)
    				a[j]=a[j+1];
    			n--;
    			
    			break;
    		}
    		if(i==n-1)
    		i=-1;
    	}
    	
    		if(n==1){
    		printf("%d\n",a[0]);
    		break;}
    	
    	}
    
    return 0;
}
