#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin >> t;
    while(t--)
    {
        int n , cnt = 0;
        cin >> n;
        {
        	if(n==1)
        	{
        		cout << -1<<"\n";
        		continue;
        	}
        }
    cnt += n/7;
    n = n%7;
    cnt += n/5;
    n = n%5;
    cnt += n/3;
    n = n%3;
    cnt += n/2;
    n = n%2;
    if(n == 1)
    ++cnt;

    cout << cnt << "\n";

    }


    return 0;
}
