S = raw_input()
#first look at 3s. if you find a hit, can do all prefixes of it
N = len(S)
ans = 0
for i in xrange(N-2):
	if int(S[i:i+3])%8 == 0: ans += i+1 - S[:i+1].count('0')
#Now look at twos only
for i in xrange(N-1):
	if int(S[i:i+2])%8 == 0 and S[i]!='0': ans += 1
ans += S.count('8')+S.count('0')
print ans
