#include<iostream>
#include<cstdio>
using namespace std;
typedef long int ll;
ll rev(ll n)
{
    ll r=0,d,x;
    x=n;
    while(x!=0)
    {
        d=x%10;
        r=r*10+d;
        x=x/10;
    }
    return r;
}
int main()
{
    ll t,x,y;
    #ifndef ONLINE_JUDGE

#endif

    {
        ll add=0;
        cin >> x;
        cin >> y;
        add= rev(x) + rev(y);
        cout << rev(add) << "\n";
    }
    return 0;
}
