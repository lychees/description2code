#include<iostream>
using namespace std;
main()
{
    unsigned long long a,b,i,t,c,j;
    cin>>t;
    for(i=0;i<t;i++)
    {
        cin>>a;
        cin>>b;
        for(j=0;j<(b-a);j++)
        {
            c=a&b;
            a=b;
            b=b+j;


        }
        cout<<c<<endl;
    }
}
