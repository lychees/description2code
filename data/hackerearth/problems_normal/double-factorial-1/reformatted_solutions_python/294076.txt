import math
t = input()
x = math.factorial(24)
while t > 0:
    t -= 1
    n, m = map(int, raw_input().split())
    if m == 0:
        print "0"
    else:
        if n > 4:
            print 0
        else:
            y = pow(10, m)
            if n == 0:
                print 1
            elif n == 1:
                print 1
            elif n == 2:
                print 2
            elif n == 3:
                print(720 % y)
            elif n == 4:
                print(x % y)
