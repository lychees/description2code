#include<bits/stdc++.h>
using namespace std;

#define lli long long
#define pb push_back
#define mp make_pair

#define lld "%I64d"
#define DEBUG(x) cout<<">value ("<<#x<<") : "<<x<<endl;
#define mod 1000000009

const int lmt = 10;
int a[lmt];
int n;

int solve(int idx,int f,int s){
    if(idx==n){
        if(f<=s) return 1;
        return 0;
    }
    if(f>s) return 0;
    int ans = 0;
    ans += solve(idx+1,f+a[idx],s);
    ans += solve(idx+1,f,s+a[idx]);
    return ans;
}

int permute(int idx){
    if(idx==n)
        return solve(0,0,0);
    int ans = 0;
    for(int i=idx;i<n;i++){
        swap(a[i],a[idx]);
        ans += permute(idx+1);
        swap(a[i],a[idx]);
    }
    return ans;
}

int main(){
    scanf("%d",&n);
    int total = 0;
    for(int i=0;i<n;i++){
        scanf("%d",&a[i]);
        total += a[i];
    }
    int ans = permute(0);
    printf("%d %d\n",ans,total);
    if(total >= ans)
    	printf("Got no way out!\n");
    else
    	printf("We will win!\n");
    return 0;
}
