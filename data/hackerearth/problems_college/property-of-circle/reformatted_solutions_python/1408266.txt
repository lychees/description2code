import math
T = int(input())
for t in range(T):
    x1, y1, r1 = map(int, raw_input().split())
    x2, y2, r2 = map(int, raw_input().split())
    d = math.sqrt((x1-x2)**2 + (y1-y2)**2)
    if d == 0 and r1 == r2:
        print "C2~C1"
    elif d > abs(r1-r2):
        print "NOT SATISFIED"
    elif r1 > r2:
        print "C1CC2"
    else:
        print "C2CC1"
