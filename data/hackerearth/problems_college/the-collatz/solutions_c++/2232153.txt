#include <iostream>
#include <stdio.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long int t,n;
    scanf("%ld\n",&t);
    while(t--){
    	long int se=0,so=1,s=0;
    	scanf("%ld\n",&n);
    	while(n!=1){
    		s++;
    		if(n%2==0){
    			se+=n;
    			n=n/2;
    			
    		}
    		else{
    			so+=n;
    			n=(3*n)+1;
    		}
    		
    	}
    	if((se%s)>(so%s))
    		printf("Even Rules\n");
    	else if((se%s)<(so%s))
    		printf("Odd Rules\n");
    	else
    		printf("Tie\n");
    	}
    return 0;
}
