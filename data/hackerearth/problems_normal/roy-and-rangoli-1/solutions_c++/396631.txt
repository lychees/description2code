#include <iostream>
#include <cmath>
using namespace std;
#define mod 1000000007
int main()
{
	bool isprime[2000009]={false};
	int n;
	cin>>n;
	int temp=sqrt(2*n-2);
    for(int i=2;i<=temp;i++)
    {
    	if(!isprime[i])
    	{
    	for(int j=i*i;j<=2*n-2;j+=i)
    	isprime[j]=true;
    	}
    }
    int res=0;
    for(int i=2;i<=2*n-2;i++)
    {
    if(!isprime[i])
    {
    if(i<=n-1)
    res=(res+i+1)%mod;
    else
    {
    	int temp=(2*n-1)%mod;
    temp=(temp-i)%mod;
    res=(res+temp)%mod;
    }
    }
    }
    cout<<res<<"\n";
	return 0;
}
