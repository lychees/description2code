#include <iostream>
#include <algorithm>
#include <math.h>
using namespace std;

int main()
{
    long long N, temp;
    vector<long long> A;
    cin >> N;
    
    for (int i=0; i<N; i++)
    {
        cin >> temp;
        A.push_back(temp);
    }
    
    sort(A.begin(), A.end());
    
    size_t pos = A.size();
    long long p = A[pos-2] * A[pos-1];
    
    long long p1 = A[0] * A[1];
    
    cout << max(p, p1) << endl;
    
    return 0;
}
