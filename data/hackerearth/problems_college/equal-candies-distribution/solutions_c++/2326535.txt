#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int a[10010];
	int test;
	cin>>test;
	while(test--)
	{
		int n;
		cin>>n;
		int i,j,mymin=INT_MAX;
		for(i=0;i<n;i++)
		{
			cin>>a[i];
			if(a[i]<mymin)
			{
				mymin=a[i];
			}
		}
		int x[5],ans=INT_MAX;
		for(i=0;i<5;i++)
		{
			x[i]=0;
			for(j=0;j<n;j++)
			{
				x[i]+=((abs(a[j]-(mymin-i))/5) + ((abs(a[j]-(mymin-i))%5)/2)  + ((abs(a[j]-(mymin-i))%5)%2));
			}
			ans=min(ans,x[i]);
		}
		cout<<ans<<endl;
	}
	return 0;
}
