#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
   long int n,i,t_s,tw_s,o_s,tot;
	int t;
    cin>>t;
	long int result[t];    
   
    for(i=1;i<=t;i++)
    {
    	cin>>n;
    	if(n==1)
    		result[i]=1;
    	else if(n==2)
    		result[i]=8;
    	else
    	{
    		t_s=8;
    		tw_s=12*(n-2);
    		o_s=6*((n*n)-(2*n+2*(n-2)));
    		tot=t_s+tw_s+o_s;
    		result[i]=tot;
    		
    	}
    	
    }
    for(i=1;i<=t;i++)
    {
    	cout<<result[i]<<endl;
    	
    }
}
