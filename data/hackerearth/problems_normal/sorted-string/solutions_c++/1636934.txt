#include <iostream>
#include<algorithm>
using namespace std;
struct inp{
	int n;
	char c;
};
bool cmp(inp a,inp b){
	return (a.n<b.n)||((a.n==b.n)&&(a.c<b.c));
}
int main()
{
    char s[105];
    int t,i;
    scanf("%d",&t);
    while(t--){
    	inp a[26];
    	for(i=0;i<26;i++){
    	   a[i].c='a'+i;
    	   a[i].n=0;
    	}   
    	scanf("%s",s);
    	for(i=0;s[i]!='\0';i++){
    		a[s[i]-'a'].n++;
    	}
    	sort(a,a+26,cmp);
    	for(i=0;i<26;i++){
    		while(a[i].n>0){
    			printf("%c",a[i].c);
    			a[i].n--;
    		}
    	}
    	printf("\n");
    }
    return 0;
}
