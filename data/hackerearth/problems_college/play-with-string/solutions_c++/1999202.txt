#include <iostream>
#include <algorithm>
#include <cmath>
#include <string.h>
using namespace std;


int main()
{
    int t, i, k, n;
    cin>>t;
    while(t--) {
    	char s[1000];
    	cin>>s;
    	i=0;
    	k=0;
  		n=strlen(s);
    	while(s[i+1]!='\0') {
    		if(abs(s[i+1]-s[i])!=abs(s[n-i-1]-s[n-i-2])) {
    			k=1;
    			break;
    		}
    		i++;
    	}
    	if(k==0)
    		cout<<"Funny";
    	else
    		cout<<"Not Funny";
    	cout<<endl;
    }
    return 0;
}
