#include <iostream>
using namespace std;

int main()
{
	int a;
	cin>>a;
	int m[a][a],pd=0,sd=0;
	for(int i=0;i<a;i++){
		for(int j=0;j<a;j++){
			cin>>m[i][j];
			if(i==j){pd=pd+m[i][j];}
		}
	}
	int j=a-1;
	for(int i=0;i<a;i++){
		sd=sd+m[i][j];j--;
	}
	if(sd>pd)
    cout << sd-pd << endl;
    else
        cout << pd-sd << endl;
    return 0;
}
