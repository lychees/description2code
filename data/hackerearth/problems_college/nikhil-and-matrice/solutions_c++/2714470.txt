#include<iostream>
using namespace std;
#include<math.h>
int main()
{
  int n,i,j,k,flag,val,T;
  cin>>T;
  for(k=0;k<T;k++){
  cin>>n;
  int sq[10][10];
  for(i=0;i<n;i++)
  for(j=0;j<n;j++)
  {
    sq[i][j]=-1;
  }
   
  flag=(n*n);
  val=2;
  i=0;
  j=0;
  sq[0][0]=1;
  while(flag>0)
  {
    if(sq[i][j+1]==-1&&i<n&&j<n-1&&(i==j||j>=i-1))
    {
      j++;
      sq[i][j]=val;
    }
    else if(sq[i+1][j]==-1&&i<n&&j<n&&i<=j)
    {
      i++;
      sq[i][j]=val;
    }
    else if(sq[i][j-1]==-1&&i<n&&j<n&&i>=j&&j>=0)
    {
      j--;
      sq[i][j]=val;
    }
    else if(sq[i-1][j]==-1&&i<n&&j<n&&i>=j&&i>0)
    {
      i--;
      sq[i][j]=val;
    }
    val++;
    flag--;
  }
 for(i=0;i<n;i++)
 {
  for(j=0;j<n;j++)
  {
   cout<<sq[i][j];
   }
   cout<<endl;
  }
  cout<<endl;
  }
  return 0;
 }
