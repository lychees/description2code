#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    char str[1000];
    cin>>str;
    int sum=0;
    for(int i=0;i<strlen(str);i++)
    {
    	int count=1;
    	while(str[i]==str[i+1])
    	{count++;
    	i++;
    	}
    	if(count!=1)
    	sum+=count*(count-1)/2;
    }
    sum+=strlen(str);
    cout<<sum;
    return 0;
}
