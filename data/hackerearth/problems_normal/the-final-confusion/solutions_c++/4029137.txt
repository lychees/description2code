#include<bits/stdc++.h>
using namespace std;
long long int arr[100][100];
long long int fun(long long int b,long long int w,long long cb,long long int cw)
{
	if(b==0&&w==1)
	return 1;
	if(b<=0&&w<=0)
	return 0;
	if(arr[b][w]!=0)
	return arr[b][w];
	if(b==0)
	return arr[b][w]=fun(b,w-1,cb,cw+1);
	
	
	else 
	{
		if(cb>cw)
		{
			return arr[b][w]=(fun(b,w-1,cb,cw+1)+fun(b-1,w,cb+1,cw));
		}
		else
		return arr[b][w]=fun(b-1,w,cb+1,cw);
	}
}


int main()
{
	long long int t,n,ans,i,j,a,b;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
	   ans=fun(n,n,0,0);
	   printf("%lld\n",ans);
	}
	return 0;
}
