#include<iostream>
using namespace std;
long long int sherly()
{
	long long int i,j,odd=0,count=0,n;
	long long int a[100000];
	scanf("%llu",&n);
	for(i=0;i<n;i++)
	{
		scanf("%llu",&a[i]);
	}
	for(i=0;i<n;i++)
	{
		if(a[i]%2!=0)
		{
			odd++;
		}
		else if(a[i]%2==0)
		{
			count++;
		}
	}
		
	long long int e = count * odd;
   return e;
}
int main()
{
   long long int t,i;
   scanf("%llu",&t);
   for(i=0;i<t;i++)
   {
   	printf("%llu\n",sherly());
   }
}
