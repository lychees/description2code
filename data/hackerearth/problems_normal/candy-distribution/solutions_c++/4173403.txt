map<ll,ll> dp;

ll fastpower(ll base,ll power)
{
        long long result=1;
        while (power>0) 
        {
                if (power%2==1)         
                        result = (result*base)%mod;
                base = (base*base)%mod;
                power/=2;
        }
        return result;
}
ll func(ll n)
{
    if(n==1 or n==2)
        return 1;
    if(dp.find(n)!=dp.end())
        return dp[n];
    else 
    {
        ll ans =1;
        for(ll i=2;i*i<=n;++i)
            if((n-1)%i==0 and i!=((n-1)/i))            
            {
                ans = (ans%mod+fastpower((func(i)%mod),(n-1)/i))%mod;
                ans = (ans%mod+fastpower((func((n-1)/i)%mod),i))%mod; 
            }
            else if((n-1)%i==0 and i==((n-1)/i))
            {
                ans = (ans%mod+fastpower((func(i)%mod),i))%mod;
            }
        dp[n] = ans;
        return dp[n];
    }

}
