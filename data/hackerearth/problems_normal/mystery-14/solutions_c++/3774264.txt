#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t>0)
    {
    	int N, count=0;
    	cin>>N;
    	int A[N];
    	for(int i=0; i<N; i++)
    	{
    		cin>>A[i];
    		if(A[i]%2 != 0)
    		{
    			count+=A[i];
    		}
    	}
    	cout<<count<<"\n";
    	t--;
    }
    return 0;
}
