#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t=0;
    scanf("%d",&t);
    while(t--)
    {
    	string s;
    	int count[26]={0},c=0,i=0,f=0;
    	cin>>s;
    	for(i=0;i<s.size();i++)
    	count[s[i]-'a']++;
    	c=count[0];
    	for(i=0;i<26;i++)
    	if(count[i]!=c){f=1;break;}
    	if(f==1)
    	cout<<"No"<<endl;
    	else 
    	cout<<"Yes"<<endl;
    }
    return 0;
}
