#include <bits/stdc++.h>
using namespace std;
map <char,int> a;
int compare(const void* e,const void* b)
{
	char a1=*(char*)e;
	char b1=*(char*)b;
	if(a[a1]>a[b1])
	  return 1;
	else   if(a[a1]<a[b1])
	  return -1;
	else  if(a[a1]==a[b1])
	  return 1; 
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{

		int i;
		char ab;
		for(i=0;i<26;i++)
		   {
		   	  cin>>ab;
		   	  a[ab]=i;
		   }
		char ans[100000];
		cin>>ans;
		qsort(ans,strlen(ans),sizeof(char),compare);
		cout<<ans<<endl;
		
		
	}
	return 0;
}
