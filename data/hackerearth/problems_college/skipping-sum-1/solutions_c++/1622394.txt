#include<bits/stdc++.h>

using namespace std;

long long int ar[100002][12];
int d[100002];

int main(){

    int Q,N;

    scanf("%d %d",&N,&Q);

    for(int i=1;i<=N;i++)
        scanf("%d",&d[i]);

    for(int i=1;i<=10;i++){
        for(int j=1;j<=N;j++){

            ar[j][i]=d[j];

            if(j-i>0)
                ar[j][i]=d[j]+ar[j-i][i];
            else
                ar[j][i]=d[j];

        }
    }

    int l,r,k;

    while(Q--){
        scanf("%d %d %d",&l,&r,&k);

        r=l+((r-l)/k)*k;


        if(l-k>0)
            cout<<ar[r][k]-ar[l-k][k]<<endl;
        else
            cout<<ar[r][k]<<endl;
    }

}
