#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while (t--) {
    	string s;
    	cin >> s;
    	int c = 0;
    	string t = "";
    	int f = 0;
    	for (int i = s.size() - 1; i >= 0; i--) {
    		c++;
    		t += s[i];
    		if (c == 3 && f == 0) {
    			if (i > 0)
    			t += ',';
    			c = 0;
    			f = 1;
    		} else if (c == 2 && f == 1) {
    			if (i > 0)
    			t += ',';
    			c = 0;
    		} 
    	}
    	reverse(t.begin(), t.end());
    	cout << t << endl;
    }
    return 0;
}
