#include<iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
	{
		int t,l,i;
		cin>>t;
		char ch[10007];
		set<string>s;
		string w;
		while(t--)
			{
				w="";
				
				scanf(" %[^\n]s",ch);
				l=strlen(ch);
				for( i=0;i<l;++i)
					{
						if(ch[i]==' '){
							s.insert(w);
							w="";
							}
						else
							w+=ch[i];
					}
				if(w!="")
					s.insert(w);
				cout<<s.size()<<endl;
				s.clear();
			}
		return 0;
	}
