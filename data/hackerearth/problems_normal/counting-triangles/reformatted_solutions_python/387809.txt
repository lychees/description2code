n = input()
d = []
for i in xrange(n):
    a = sorted(map(int, raw_input().split()))
    d.append(a)
d.sort()
f = 1
x = d[0]
el = []
for i in xrange(1, n):
    if d[i] == x:
        f = f+1
    else:
        el.append([x, f])
        x = d[i]
        f = 1
el.append([x, f])
count = 0
for i in xrange(len(el)):
    if el[i][1] == 1:
        count = count+1
print count
