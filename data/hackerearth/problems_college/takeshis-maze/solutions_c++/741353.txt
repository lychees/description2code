#include <iostream>
#include <cstring>

using namespace std;

int n,c;
int mx =0;
int a[200][200];

int dp[200][200];

int dfs(int i,int j){

	if(dp[i][j]!=-1) return dp[i][j];
	
	if(i==n-1 && j==n-1) {
		return a[i][j];
	}
	
	if(i==n || j==n) return -99999;
	
	
	
	dp[i][j] = a[i][j] + max(dfs(i+1,j),dfs(i,j+1));

	return dp[i][j];
}



int main()
{
 int t;
 cin>>t;
  
 //int a[100][2];
 while(t--){
 
 cin>>n>>c;
 	
 	memset(dp,-1,sizeof(dp));
 	
 	for(int i=0;i<n;i++){
 		for(int j=0;j<n;j++){
 			
 			cin>>a[i][j];
 		}
 	}
 	
 	
 	mx = dfs(0,0);
 	
 	if(mx < c) cout<<"-1"<<endl;
 	else cout<<mx-c<<endl;
 	
 }
 
 
  return 0;
}
