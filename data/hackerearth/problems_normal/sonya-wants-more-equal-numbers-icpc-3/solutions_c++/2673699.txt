#include<bits/stdc++.h>

using namespace std;

#define FOR(i,a,b) for(int i=a;i<b;i++)
#define For(i,a,b) for(int i=a-1;i>=b;i--)
#define rep(i,b) FOR(i,0,b)
#define Rep(i,b) For(i,a,0)
#define SORT(x) sort(x.begin(),x.end());
#define ERASE(x) x.erase(x.begin(),x.end());
#define FILL(x,i) memset(x,i,sizeof(x));
#define K 1000000007
#define L 400
#define ll long long
#define s1(a) scanf("%d",&a);
#define s2(a) scanf("%lld",&a);
#define s3(a,b) scanf("%lld%lld",&a,&b);
#define s4(a,b,c) scanf("%lld%lld%lld",&a,&b,&c);
#define pb push_back
#define mp make_pair
#define F first
#define S second 
#define PII pair<int,int>
#define PLL pair<ll,ll>
#define EPS 0.000001

bool cmp(pair<ll,ll> a,pair<ll,ll> b){if(a.S<b.S) return true; return false;}
/*******************************MAIN CODE STARTS*******************************/

map<ll,vector< pair<ll,ll> > > m;
map<ll,vector< pair<ll,ll> > >::iterator it;
ll n,a[110];
vector<ll> res;

void scan()
{
	s2(n)
	FOR(i,0,n)
	{
		s2(a[i]);
	}
}

void out()
{
	FOR(i,0,n)
	{
		ll re=0;
		FOR(j,i,n)
		{
			re+=a[j];
			m[re].pb(mp(i,j));
		}
	}
	ll re=0;
	for(it=m.begin();it!=m.end();it++)
	{
		ll dp[110]={0},dir=0;
		sort((it->S).begin(),(it->S).end(),cmp);
		//cout<<it->F<<'\n';
		FOR(i,0,n)
		{
			dp[i+1]=dp[i];
			while((it->S)[dir].S==i)
			{
				if(dir==(it->S).size())
					break;
			//	cout<<(it->S)[dir].F<<' '<<(it->S)[dir].S<<'\n';
				dp[i+1]=max(dp[i+1],dp[(it->S)[dir].F]+1);
				++dir;
			}
			//cout<<dp[i+1]<<'\n';
		}
		//cout<<dp[n]<<' '<<it->F<<'\n';
		if(dp[n]>re)
		{
			re=dp[n];
			res.erase(res.begin(),res.end());
			res.pb(it->F);
		}
		else if(dp[n]==re)
		{
			res.pb(it->F);
		}
	}
	sort(res.begin(),res.end());
	cout<<re<<'\n';
	FOR(i,0,res.size())
		cout<<res[i]<<' ';
	cout<<'\n';
}

int main()
{
	int t=1;
	//s1(t)
	while(t--)
	{
		scan();
		out();
	}
	return 0;
}
