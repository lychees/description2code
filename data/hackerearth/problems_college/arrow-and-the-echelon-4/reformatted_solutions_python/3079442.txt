for t in range(int(raw_input())):
    x = int(raw_input())
    if x < 3:
        print(-1)
    elif x % 3 == 0:
        print('5'*x)
    else:
        i = 5
        while i < x:
            if i % 5 == 0 and (x-i) % 3 == 0:
                print('5'*(x-i)+'3'*i)
                break
            i += 1
        else:
            if x % 5 == 0:
                print('3'*x)
            else:
                print(-1)
