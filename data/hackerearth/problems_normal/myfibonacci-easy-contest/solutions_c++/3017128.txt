#include <iostream>
using namespace std;

int fib(int A,int B,int N)
{
	int ans;
if(N>0)	
{
	N--;
ans=A+B;
fib(B,ans,N);
}
else
{
	return B;
}
}

int main()
{
    int A,B,N,ans;
    cin>>A>>B>>N;
    if(N==1)
    {
    	cout<<A;
    }
    else if(N==2)
    {
    	cout<<B;
    }
    else
    {
    ans=fib(A,B,N-2);
    cout<<ans;
    }    
}
