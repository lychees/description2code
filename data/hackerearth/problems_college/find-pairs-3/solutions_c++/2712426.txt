#include <bits/stdc++.h>
using namespace std;
int a[1000009];
int main()
{
    int t,n;
    cin>>t;
    while(t--)
    {
    	int i;
    	cin>>n;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	sort(a,a+n);
    	long long c=1;
    	long long sum=0;
    	for(i=1;i<n;i++)
    	{
    		if(a[i]==a[i-1])
    		{
    			c++;
    		}
    		else
    		{	sum+=(c*(c-1))/2;
    			c=1;}
    	}
    	sum+=(c*(c-1))/2;
    	cout<<sum+(n)<<endl;
    }
    return 0;
}
