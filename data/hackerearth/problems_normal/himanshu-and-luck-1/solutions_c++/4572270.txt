#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
#define PII       pair<int,int>
#define all(c)    c.begin(),c.end()
#define sz(c)     (int)c.size()
#define clr(c)    c.clear()
#define pb        push_back
#define mp        make_pair
#define cin(x)    scanf("%d",&x)
#define MOD       1000000007
#define EPS       1E-10
vector<LL> lucky;
void backtrack(LL now)
{
	lucky.pb(now);
	if(now < 0 or now >= 1000000000000000000)
		return ;
	backtrack(now*10 + 4);
	backtrack(now*10 + 7);
}
LL solve(LL x)
{
	int pos = (int)(upper_bound(all(lucky) , x) - lucky.begin());
	return pos;
}
int main()
{
	backtrack(0);
	sort(all(lucky));
	int t;
	cin(t);
	assert(1 <= t && t <= 10000);
	while(t--)
	{
		LL l , r;
		scanf("%lld%lld" , &l , &r);
		assert(1 <= l && l <= r && r <= 1000000000000000000);
		LL ans = solve(r) - solve(l-1);
		printf("%lld\n", ans);
	}
	return 0;
}
