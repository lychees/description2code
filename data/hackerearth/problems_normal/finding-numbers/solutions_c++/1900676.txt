#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t,ans=0;
    long long int z,m,n;
    long int i;
    cin>>t;
    while(t--)
    {
        ans=0;
        cin>>z>>m>>n;
        for(i=1; i<sqrt(z); i++)
        {
            if(z%i == 0)
            {
                long long int num2 = z/i;
                if(i<=m && num2<=n)
                    ans++;
                if(i<=n && num2<=m)
                    ans++;
            }
        }
        if(i*i == z)
            ans++;
        cout<<ans<<endl;
    }
}
