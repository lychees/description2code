#include <iostream>
using namespace std;

int main()
{
    int t,count=0;
    cin>>t;
    string s;
    cin>>s;
    for(int i=0;i<t;i++)
    {
    	if(s[i]=='w')
    	count+=2;
    	else
    	count+=1;
    }
    int chk=0,cal=0,num=0,rest=0;
    for(int i=0;i<t;i++)
    {
    	if(s[i]=='w')
    	chk+=2;
    	else if(s[i]=='v')
    	chk+=1;
    	else
    	{
    		rest+=1;
    		if(chk%2==0)
    		chk=chk/2;
    		else
    		chk=((chk/2)+1);
    		num+=chk;
    		chk=0;
    	}
    }
    if(chk!=0)
    {
    		if(chk%2==0)
    		chk=chk/2;
    		else
    		chk=((chk/2)+1);
    		num+=chk;
    }
    cout<<(num+rest)<<" "<<count;

}
