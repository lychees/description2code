#include<iostream>
#include<cstdio>
using namespace std;

int main()
{
	int n;
	cin >> n;
	int ans[n],arr[n];
	long long int sum=0;
	for(int i=0;i<n;i++)
	{
		cin >> arr[i];
		ans[i]=1;
	}
	for(int i=0;i<n-1;i++)
	{
		if(arr[i+1]>arr[i])
		ans[i+1]=ans[i]+1;
	}
	for(int i=n-1;i>0;i--)
	{
		if(arr[i-1]>arr[i] && ans[i-1]<=ans[i])
		ans[i-1]=ans[i]+1;
	}
	for(int i=0;i<n;i++)
	sum+=ans[i];
	cout << sum << endl;
	return 0;
}
