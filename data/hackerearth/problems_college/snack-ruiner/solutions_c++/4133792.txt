#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	double a[n],b[n];
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	for(int i=0;i<n;i++)
    	cin>>b[i];
    	int index=0;
    	double max=a[0]*b[0];
    	double sum=0,temp;
    	for(int i=0;i<n;i++)
    	{
    		temp=a[i]*b[i];
    		sum+=temp;
    		if(temp>max)
    		{
    			max=temp;
    			index=i;
    		}
    	}
    	max=max/sum;
    	printf("%d %.3f",index,max);
    	printf("\n");
    }
    return 0;
}
