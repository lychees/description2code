#include <iostream>
using namespace std;

int main()
{
    int t,i,n,j,zero,one;
    string s;
    cin>>t;
    while(t--)
    {
    	cin>>s;
    	cin>>n;
    	zero=one=0;
    	for(i=0;i<s.size();i++)
    	{
    		if(s[i]=='0')
    		zero++;
    		else
    		one++;
    	}
    	n=(n+1)/2;
    	if(n&1)
    	cout<<zero<<"\n";
    	else
    	cout<<one<<"\n";
    }
}
