#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
int main()
{
     int t;
	 string s;
	 scanf("%d",&t);
	 
	 while(t--)
	 {
	    cin>>s;
	    string str;
		 
	
		 for( int i=0;i<s.length();i++)
           {
		     if((s[i]=='a')||(s[i]=='e')||(s[i]=='i')
			 ||(s[i]=='o')||(s[i]=='u')||(s[i]=='A')
			 ||(s[i]=='E')||(s[i]=='I')||(s[i]=='O')||(s[i]=='U'))
			 str+=s[i];
		 }
		 if(!str.empty())
		 cout<<str<<endl;
		 else
		 cout<<"No"<<endl;
	 }
	 return 0;
}
