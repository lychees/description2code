#include<iostream>
using namespace std;

int main()
{
	int n,i,j,c1=0,c2=0;
    cin>>n;
    
    int arr[n][2];
    
    for(i=0;i<n;i++)
    for(j=0;j<2;j++)
    cin>>arr[i][j];
    
    for(i=0;i<n;i++)
    {
    	c1 = n-1;
    	c2=0;
    	
    	for(j=0;j<n;j++)
    	if(j!=i)
    	{
    	if(arr[i][1]==arr[j][0])
    	c1++;
    	else
    	c2++;
        }
    	
    	
    	cout<<c1<<" "<<c2<<endl;
    	
    }
}
