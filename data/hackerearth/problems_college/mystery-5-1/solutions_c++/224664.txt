#include <cstdio>
#include <iostream>
#include <string.h>

using namespace std;

int main()
{
    int t, i, ans=0, len, hash[26]={0};
    char str[10001];
    cin>>t;
    while (t--)
    {
        cin>>str;
        len=strlen(str);
        //cout<<len<<endl;
        for (i=0; i<len; i++)
        {
            hash[str[i]-97]+=1;
        }
        ans=0;
        for (i=0; i<26; i++)
        {
            //cout<<hash[i]<<" ";
            if (hash[i]>ans)
            {
                ans=hash[i];
            }
            hash[i]=0;
        }
        cout<<ans<<endl;
    }
    return 0;
}
