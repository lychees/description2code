from sys import stdin
t = int(stdin.readline())
for tt in xrange(t):
	n,k = map(int,stdin.readline().split())
	a = map(int,stdin.readline().split())
	#print a
	ans = 0
	pr = -1
	for i in xrange(0,n):
		if a[i]<=pr:
			dif = pr - a[i]
			req = dif/k
			#print "we need atleast ",req
			while True:
				if a[i] + req*k > pr:
					break
				req+=1
			ans+= req
			pr = a[i] + req*k
		else:
			pr = a[i]
	print ans
