#include <iostream>
using namespace std;

int h(string,int);//returns frequency of most frequent character

int main(){
	int t;
	int n;
	string in;
	cin>>t;
	for(int i=0; i<t; i++){
		cin>>n>>in;
		if(h(in,n)%2==0)
		cout<<"Yes";
		else
		cout<<"No";
		cout<<endl;
	}
	return 0;
}

int h(string s, int len){
	int a[26]={0};
	int i;
	//generate character frequency array
	for(i=0; i<len; i++)
		a[s[i]-97]++;
	
	//find largest in a
	int big=a[0];
	for(i=1; i<26; i++){
		if(a[i]>big)
			big=a[i];
	}
	return big;
}
