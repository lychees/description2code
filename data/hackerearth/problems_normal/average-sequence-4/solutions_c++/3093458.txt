#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    int *p, *q;
    p=new int [t];
    q=new int [t];
    for (int i=0; i<t; i++)
    {
    	cin>>p[i];
    }
    for (int i=0; i<t; i++)
    {
    	q[i]=(i+1)*p[i]-i*p[i-1];
    }
    for (int i=0; i<t; i++)
    {
    	cout<<q[i]<<" ";
    }
}
