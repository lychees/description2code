#include<bits/stdc++.h>
using namespace std;
int gcd(long int a,long int b)
{
    while(b) b ^= a ^= b ^= a %= b;
    return a;
}
int main()
{
    long int n,k;
    cin>>n>>k;
    long int a[n],ans=0,i,j=0,x=0;
    for(i=0;i<n;i++)
        cin>>a[i];
        i=1;
    for(i=1;i<n;i++)
    {
        if(gcd(a[i],a[i-1])>= k)
        {
            long int count1=2;
            for(x=i+1;x<n;x++)
            {
                if(gcd(a[x],a[x-1])>= k)
                count1++;
                else
                    break;
            }
 
            if(count1>ans)
                ans=count1;
            i=x;
        }
 
    }
    cout<<ans<<endl;
 
}
