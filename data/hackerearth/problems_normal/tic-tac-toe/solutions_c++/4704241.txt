#include <iostream>
using namespace std;
int m=1000000007;
long long power(int a,int b)
{
	if(b==0)
	return 1;
	long long p=power(a,b/2);
	if(b&1)
	return (((p*p)%m)*a)%m;
	else
	return (p*p)%m;
}
int main()
{
    long long t,n;
    long long max,min;
    scanf("%lld",&t);
    long long de=power(6,m-2);
    long long ne=power(4,m-2);
    for(long long i=0;i<t;i++)
    {
    	min=1LL;
    	max=1LL;
    	scanf("%lld",&n);
    	max=((n%m)*((n-1)%m))%m;
    	max=((max%m)*((2*n-1)%m))%m;
    	max=((max%m)*(de%m))%m;
    	min=((n%m)*((n-1)%m))%m;
    	min=((min%m)*((n-1)%m))%m;
    	min=((min%m)*(ne%m))%m;
    	printf("%lld ",min);
    	printf("%lld\n",max);
    }
    return 0;
}
