#include <bits/stdc++.h>
using namespace std;
#define pb push_back
#define ll long long int
#define INF 10000000000000000
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int x;
		cin>>x;
		int count=2,tr;
		for(int i=1;i<sqrt(x);i++)
		{
			if(x%i==0)
			{
				tr=x/i;
				if(tr!=x)
					count+=2;
				else count++;
			}
			if(count>=4)
				break;
		}
		if(count<4)
			cout<<"NO\n";
		else cout<<"YES\n";
	}
}
