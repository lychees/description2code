#define F0(i,n) for(ll i=0;i<n;i++)

const ll mod=1e9+7; const ll N=2*1e5+5;
string str;
ll t,n,i,j,x,q,k,l,r,idx,m,arr[N],ans,chk;
char ch;
ll bit[N];

void update(ll idx,ll val)
{
    while(idx<=n)
    {
        bit[idx]+=val;
        idx+=(idx&-idx);
    }
}
ll query(ll idx)
{
    ll ret=0;
    while(idx>0)
    {
        ret+=bit[idx];
        idx-=(idx&-idx);
    }
    return ret;
}
ll round(ll val)
{
    if(val<0)
        return 0;
    ll first=val%10;
    ll last=first;
    while(val)
    {
        last=val%10;
        val/=10;
    }
    if(first==last)
        return 1;
    else
        return 0;
}
int main()
{
    SI(n);SI(q);
    F1(i,n)
    {
        SL(arr[i]);
        if(round(arr[i]))
            update(i,1);
    }
    while(q--)
    {
        SL(chk);
        if(chk==2)
        {
            SL(i);SL(k);
            update(i,round(k)-round(arr[i]));
            arr[i]=k;
        }
        else
        {
            SL(l);SL(r);
            PL(query(r)-query(l-1));
            PE;
        }
    }
    return 0;
}
