#include <bits/stdc++.h>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t;
   cin>>t;
   while(t--){
   	int n;
   	cin>>n;
   	int a[n];
   	for(int i=0;i<n;i++) cin>>a[i];
   	int l=0,r=1,ansl=0,ansr=0;
   	unordered_set<int> s;
   	s.insert(a[0]);
   	while(r<n){
   		if(s.find(a[r])==s.end()){
   			if(r-l > ansr-ansl){
   				ansr=r;
   				ansl=l;
   			}
   		}
   		else{
   			while(a[l]!=a[r]){
   				s.erase(a[l]);
   				l++;
   			}
   			s.erase(a[l]);
   			l++;
   		}
   		s.insert(a[r]);
   		r++;
   	}
   	cout<<(ansl+1)<<" "<<(ansr+1)<<endl;
   }
    return 0;
}
