	for(i='a';i<='z';i++)
		if(a[i]%2==1)
			odd++;
	if(n%2==1&&odd==1)
	{
		for(i='a';i<='z';i++)
		{
			while(a[i]>1)
			{
				a[i]-=2;
				t+=i;
			}
			if(a[i]==1)
				c=i;
		}
		re=t;
		re+=c;
		reverse(t.begin(),t.end());
		re+=t;
	}
	else if(odd==0)
	{
		for(i='a';i<='z';i++)
			while(a[i])
			{
				a[i]-=2;
				t+=i;
			}
		re=t;
		reverse(t.begin(),t.end());
		re+=t;
	}
	else
		re="-1";
	cout<<re<<endl;

}
