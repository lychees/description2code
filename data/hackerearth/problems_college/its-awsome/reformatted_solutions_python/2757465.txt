t = int(raw_input())
while t:
    cnt_4, cnt_7 = 0, 0
    arr = []
    n = int(raw_input())
    while n != 0:
        x = n % 10
        arr.append(x)
        n //= 10
    cnt_4 = arr.count(4)
    cnt_7 = arr.count(7)
    if cnt_4+cnt_7 == 4 or cnt_4+cnt_7 == 7:
        print("YES")
    else:
        print("NO")
    del arr[:]
    t -= 1
