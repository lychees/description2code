#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t,n,i;
	int a[100000][6];
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(i=0;i<n;i++)
		{
			cin>>a[i][0]>>a[i][1]>>a[i][2];
			a[i][3]=a[i][4]=a[i][5]=0;
		}
		a[0][3]=a[0][0];
		a[0][4]=a[0][1];
		a[0][5]=a[0][2];
		for(i=1;i<n;i++)
		{
			a[i][3]+=a[i][0]+min(a[i-1][4],a[i-1][5]);
			a[i][4]+=a[i][1]+min(a[i-1][3],a[i-1][5]);
			a[i][5]+=a[i][2]+min(a[i-1][4],a[i-1][3]);
		}
		cout<<min(a[n-1][3],min(a[n-1][4],a[n-1][5]))<<"\n";
	}
	return 0;
}
