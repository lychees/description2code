#include <bits/stdc++.h>
using namespace std;
long long z[56][100005];

int main()
{
    long long n,q,i,j,k,x,y,p,temp, count,ind,sum;
    cin>>n>>q;
    long long v[n+1];
    for(i=1;i<=n;i++)
    	cin>>v[i];
    memset(z,0,sizeof(z));
    count=1;
    for(p=1;p<=10;p++)
    {
    	for(i=1;i<=p;i++)
    	{
    		temp=0;
    		for(j=i;j<=n;j=j+p)
    		{
    			z[count][j]=temp+v[j];
    			temp=z[count][j];
    		}
    		count++;
    	}
    }
    while(q--)
    {
    	cin>>x>>y>>k;
    	if(x%k==0)
    		ind=(k*(k-1))/2 +k;
    		else
    		ind=(k*(k-1))/2+x%k;
    		while(z[ind][y]==0)
    			y--;
    			cout<<z[ind][y]-z[ind][x]+v[x]<<endl;
    }
    return 0;
}
