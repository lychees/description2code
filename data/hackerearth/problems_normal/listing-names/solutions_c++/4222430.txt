#include <iostream>
#include <map>
using namespace std;

int main()
{
	long long int t,i,j;
    map<string,long long int>M;	
    map<string,long long int>::iterator it;
    cin>>t;
    for(i=0;i<t;i++)
    {
        string s;
        cin>>s;
        if((it=M.find(s))!=M.end())
            it->second++;
        else
            M.insert(pair<string,long long int>(s,1));
    }
    for(it=M.begin();it!=M.end();it++)
        cout<<it->first<<" "<<it->second<<endl;
    return 0;
}
