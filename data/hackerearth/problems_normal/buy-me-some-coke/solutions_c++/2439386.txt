#include <bits/stdc++.h>
using namespace std;

const int MAXN1 = 909,
		  MAXN5 = 409,
		  MAXN10 = 109,
		  INF    = 1e9 + 7;

int tc;
int dp[MAXN1][MAXN5][MAXN10];

inline int go(int coke,int O,int F,int T){
	if(coke == 0)
		return 0;
		
	int &ret = dp[O][F][T];
	
	if(ret != -1)
		return ret;
		
	ret = INF;
	
	if(O >= 8)
		ret = min(ret,8 + go(coke - 1,O - 8,F,T));
	if(O >= 3 && F >= 1)
		ret = min(ret,4 + go(coke - 1,O - 3,F - 1,T));
	if(F >= 2)
		ret = min(ret,2 + go(coke - 1,O + 2,F - 2,T));
	if(T >= 1)
		ret = min(ret,1 + go(coke - 1,O + 2,F,T - 1));
	if(T >= 1 && O >= 3)
		ret = min(ret,4 + go(coke - 1,O - 3,F + 1,T - 1));
	
	return ret;
}

int main(){
	scanf("%d",&tc);
	while(tc--){
		int cokes,five,one,ten;
		scanf("%d %d %d %d",&cokes,&one,&five,&ten);
		int total = cokes * 8;
		int money = one * 1 + five * 5 + ten * 10;
		if(money < total){
			puts("-1");
			continue;
		}
		memset(dp,-1,sizeof(dp));
		printf("%d\n",go(cokes,one,five,ten));
	}
	return 0;
}
