#include<bits/stdc++.h>
using namespace std;
int main()
{
    char ar[110];
    int n,a,b;
    cin>>ar;
    cin>>n;
    while(n--)
    {
        int c=0;
        cin>>a>>b;
        for(int i=a-1;i<b-1;i++)
        {
            if(ar[i]==ar[i+1])
                c++;
        }
        cout<<c<<endl;
    }
}
