#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string a={0},b={0};
    	cin>>a;
    	cin>>b;
    	int i=0,c=0;
    	while(a[i]!=NULL&&b[i]!=NULL)
    	{   
    		if(a[i]==b[i])
    		{
    			c=c+1;
    		}
    		i++;
    	}
    	cout<<c<<endl;
    }
    return 0;
}
