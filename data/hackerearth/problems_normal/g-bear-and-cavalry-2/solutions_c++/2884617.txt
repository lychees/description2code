#include <bits/stdc++.h>

using namespace std;

const int MAX_N = 50;

int H[MAX_N], W[MAX_N];

int main()
{
   ///ifstream cin("data.in");

    int T;
    cin >> T;

    while (T--)
    {
        int N;
        cin >> N;

        for (int i = 1; i <= N; ++i)
            cin >> W[i];

        for (int i = 1; i <= N; ++i)
            cin >> H[i];

        sort(H + 1, H + N + 1);
        reverse(H + 1, H + N + 1);

        sort(W + 2, W + N + 1);

        int val = W[1] * H[1];
        int valid = 1;

        for (int i = 2; i <= N && val; ++i)
        {
            int unit = W[i] * H[i];

            if (unit >= val)
                valid = 0;
        }

        if (valid)
            cout << "YES\n";
        else
            cout << "NO\n";
    }

    return 0;
}
