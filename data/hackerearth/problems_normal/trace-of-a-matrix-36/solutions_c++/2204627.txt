#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int tt;
    cin>>tt;
    long long r,c,x,d;
    long long ans;
    while(tt--){
    	ans=0;
    	cin>>r>>c;
    	x = min(r,c);
    	d = r+c+2;
    	
    	ans = (x*(4+(x-1)*d))/2;
    	
    	cout<<ans<<endl;
    }
    return 0;
}
