#include <iostream>
#include<cstring>
using namespace std;
int a[10000007];
int main()
{
	long long i,j,x,t,k;
    cin>>t;
   
    memset(a,0,sizeof(a));
    
     a[1]=1;
    for(i=2;i*i<=10000006;i++)
        {
            k=i;
            for(j=0;j<=100;j++)
            {
                if(1LL*k*i<10000006*1LL)
                {
                     a[k*i]=1;
                     k = k*i;
                }
                else
                    break;
            }
        }
        
       /*for(i=0;i<=1000;i++)
       if(a[i])
       cout<<i<<'\n';*/

   while(t--)
    {
    	cin>>x;

    	for(i=0;i<=x;i++)
            if(a[i] && a[x - i])
            {
            	cout<<"Yes\n";
            	break;
            }
            
            if(i==x+1)
            cout<<"No\n";


    }
    return 0;
}
