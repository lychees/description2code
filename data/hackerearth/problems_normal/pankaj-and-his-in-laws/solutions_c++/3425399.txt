#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    vector<int> a(t),b(t,1);
    for(int i=0;i<t;++i){
    	scanf("%d",&a[i]);
    }
    int l=0;
    for(int i=0;i<t;++i){
    	for(int j=i+1;j<t;++j){
    		if(a[i]<a[j]){
    			b[j]=max(b[j],b[i]+1);
    		}
    		l=max(l,b[j]);
    	}
    }
    vector<int> res;
    while(l>0){
    	res.push_back(l%2);
    	l/=2;
    }
    for(int i=res.size()-1;i>=0;--i){
    	printf("%d",res[i]);
    }
    return 0;
}
