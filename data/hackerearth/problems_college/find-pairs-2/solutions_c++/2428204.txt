#include <iostream>
#define ll long long int
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	ll arr[n];
    	for(int i=0;i<n;i++){
    		cin>>arr[i];
    	}
    	ll ans = n;
    	for(int i=0;i<n;i++){
    		for(int j=i-1;j>=0;j--){
    			if(arr[i]==arr[j]) ans++;
    		}
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
