def check_prime(number):
    count=0
    for i in xrange(number):
        if i!=0 and i!=1:
            if number%i==0:
                return 0;
    return 1;
a=[]
count=0
for j in xrange(501):
    if j==0:
        a.append(0)
    elif j==1:
        a.append(0)
    else:
        if check_prime(j)==1:
            count=count+1
        a.append(a[j-1]+pow(2,count-1))
