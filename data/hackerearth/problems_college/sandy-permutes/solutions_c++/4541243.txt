#include<bits/stdc++.h>
using namespace std;

int main()
{
    long int n;
    long int k;
    int max;
    int index;
    cin>>n>>k;
    int a[n+1];
    
    for(long int i=1; i<=n; i++){
    	cin>>a[i];
    }
    
    if(k >= n-1){
    	sort(a+1, a+1+n, greater<int>() );
    }
    else{
    	
    	for(int i=1, m=1; i<=k; m++){
    		max = a[m];
    		index = m;
    		for(int j=m+1; j<=n; j++){
    			if(a[j] > max){
    				max = a[j];
    				index = j;
    			}
    		}
    		
    		if(index == m){
    			continue;
    		}
    		else{
    			int temp = a[index];
    			a[index] = a[m];
    			a[m] = temp;
    			i++;
    		}
    	}
    }
    
    for(int i=1; i<=n; i++){
    	cout<<a[i]<<" ";
    }
    return 0;
}
