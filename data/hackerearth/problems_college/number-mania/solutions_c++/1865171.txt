#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
#define Limit 200
int primes[Limit];

void gen()
{
    for(int i = 0; i < Limit; i++) primes[i] = 1;
    primes[0] = primes[1] = 0;
    for(int i = 2; i * i < Limit; i++) {
        if(primes[i]) {
            for(int j = i * i; j < Limit; j += i) {
                primes[j] = 0;
            }
        }
    }
    return;
}

void fact(int num)
{
    int ans[2005], k = 0;
	for(int i = 0; i < 2005; i++) ans[i] = 1;  
    for(int i = 2; i <= num; i++) {
        int carry = 0;
        for(int j = 0; j <= k; j++) {
            ans[j] = (ans[j] * i + carry);
            carry = ans[j] / 10;
            ans[j] = ans[j] % 10;
        }
        while(carry) {
            k++;
            ans[k] = carry % 10;
            carry /= 10;
        }
    }
    for(int i = k; i >= 0; i--) {
        printf("%d", ans[i]);
    }
    printf("\n");
}
 
int main() 
{
	gen();
	int n;
	scanf("%d", &n);
	for(int i = 1; i < n; i++) {
		if(primes[i]) {
			 fact(i);
		}
	}
	return 0;
}
