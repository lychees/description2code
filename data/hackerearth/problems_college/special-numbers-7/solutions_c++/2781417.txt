#include "bits/stdc++.h"

using namespace std;

#define REP(i,n) for(int (i)=0;(i)<(int)(n);(i)++)
#define _REP(i,x,n) for(int (i)=(x);(i)<(int)(n);(i)++)

typedef long long int LL;

vector < LL > ans;

bool is_prime(LL x) {
    if (x < 2) return false;
    for (int i = 2; i <= sqrt(x); i++) {
        if (x % i == 0) return false;
    }
    return true;
}

bool do_factor(LL x) {
    int cnt = 0;
    
    for (int i = 2; i < x; i++) {
        if (x % i == 0) {
            if (is_prime(i)) cnt++;
        }
    }
    
    if (cnt >= 3) return true;
    else return false;
}

void pre() {
    ans.push_back(0);
    int num = 0;
    long long int i = 1;
    while (1) {
        if (do_factor(i)) {
            ans.push_back(i);
            num++;
            //cout << i << endl;
        }
        i++;
        if (num > 1000) break;
    }
}

void solve() {
    int N; cin >> N;
    cout << ans[N] << endl;
}

int main () {
    pre();
    //freopen("in.txt", "r", stdin);
    int t; cin >> t; while(t--) solve();
    return 0;
}
