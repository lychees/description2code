#include<bits/stdc++.h>
using namespace std;

#define ll long long

ll a[2001],n,k,b[2001];


int main()
{
	int i,j;
	ll cnt=0;
	cin>>n>>k;

	for(i=0;i<n;i++){
		cin>>a[i];
	}

	sort(a,a+n);

	for(i=0;i<n;i++)
	{
		for(j=i+1;j<n;j++)
		{
			ll val=k/a[i];
			val/=a[j];
			ll ans=lower_bound(a,a+n,val+1)-a;
			if(ans>j)
				cnt+=(ans-j-1);
		}			
	}
	cout<<cnt<<endl;
	return 0; 

}
