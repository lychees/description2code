#include <iostream>

using namespace std;

int main()
{
    int t;

    cin>>t;

    while(t--)
    {
        int a,b,nr=0;

        cin>>a>>b;

        for(int i=a;i<=b;i++)
        {
            int aux=i,k=0,t=0,a[100];

            if(i<10)
            {
                k=1;
            }else
            {
                while(aux)
                {
                    a[k++]=aux%10;
                    aux=aux/10;
                }
            }

            for(int j=0;j<k/2;j++)
            {
                if(a[j]!=a[k-j-1])
                {
                    t=1;
                    break;
                }
            }

            if(t==0)
            {
                nr++;
            }
        }

        cout<<nr<<"\n";
    }

    return 0;
}
