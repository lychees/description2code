#include<iostream>
#include <vector>
#define fri(i,n) for (int i = 1; i <= n; i++)
using namespace std;

typedef long long ll;
typedef vector<vector<ll> > cell;

cell mul(cell A, cell B)
{
    cell C(3, vector<ll>(3));
    fri(i, 2) fri(j, 2) fri(k, 2)
        C[i][j] = (C[i][j] + A[i][k] * B[k][j]) % 1000000007;
    return C;
}

cell pow(cell A, int p)
{
    if (p == 1)
        return A;
    if (p % 2)
        return mul(A, pow(A, p-1));
    cell X = pow(A, p/2);
    return mul(X, X);
}

int fib(int N)
{
    vector<ll> F1(3);
    F1[1] = 1;
    F1[2] = 1;

    cell T(3, vector<ll>(3));
    T[1][1] = 0, T[1][2] = 1;
    T[2][1] = 1, T[2][2] = 1;

    if (N == 1)
        return 1;
    T = pow(T, N-1);

    ll res = 0;
    fri(i,2)
        res = (res + T[1][i] * F1[i]) % 1000000007;
    return res;
}
