#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	string s;
    	cin>>s;
    	int count1=0,count2=0;
    	for(int i=0;i<2*n;i+=2){
    		if(s[i]!='G') count1++;	
    	}
    	for(int i=0;i<2*n;i+=2){
    		if(s[i]!='B') count2++;	
    	}
    	cout<<min(count1,count2)<<endl;
    }
    return 0;
}
