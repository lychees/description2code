#include <iostream>
using namespace std;

int main()
{
   string s;
   cin>>s;
   int n=s.length();
   int q;
   cin>>q;
   while(q--)
   {
   	long long int a,b;
   	cin>>a>>b;
   	a=(a-1)%n;
   	b=(b-1)%n;
   	if(s[a]==s[b])
   	cout<<"Yes"<<endl;
   	else
   	cout<<"No"<<endl;
   }
    return 0;
}
