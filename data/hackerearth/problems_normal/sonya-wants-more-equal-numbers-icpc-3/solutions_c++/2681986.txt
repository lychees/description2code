#include <bits/stdc++.h>
using namespace std;

const int maxn = 111111;
const int maxm = 111;

int dp[maxn][maxm], A[maxm];
int main() {

    int n; cin >> n;
    for(int i=1; i<=n; i++) {
        cin >> A[i];
        A[i] += A[i-1];
    }
    for(int i=1; i<=n; i++) {
        for(int j=i; j>=1; j--) {
            int sum = A[i] - A[j-1];
            dp[sum][i] = max(dp[sum][i], dp[sum][j-1]+1);
        }
        for(int j=1; j<maxn; j++) dp[j][i] = max(dp[j][i], dp[j][i-1]);
    }
    int cnt = 0;
    for(int i=1; i<maxn; i++) cnt = max(cnt, dp[i][n]);
    cout << cnt << "\n";
    for(int i=1; i<maxn; i++) if(cnt == dp[i][n]) cout << i << " ";
    cout << "\n";
    return 0;
}
