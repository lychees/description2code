#include <stdio.h>
#include<iostream>
#include<string.h>
int main()
{
    int T,a,b,n,mod,i;
    scanf("%d",&T);
    for(;T--;)
    {
    	scanf("%d %d %d %d",&a,&b,&n,&mod);
    	long int ct[mod];
    	memset(ct,0,mod*sizeof(long int));
		ct[a]=1;
		ct[b]+=1;
		for(i=2;i<n;i++)
    	{
    		int temp=a;
    		a=b;
    		b=(temp+a)%mod;
    			ct[b]+=1;
    	}
    	long long int sum=0;
  	 	for(i=0;i<mod;i++)
  	 		sum+=ct[i]*ct[i];
  	 	printf("%lld\n",sum);
    }

    return 0;
}
