#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
     {
     	int m,n,ans;
     	cin>>m>>n;	
     	if(n>=m)
     	{
     		n=2*n;
     		ans=n/m;
     	if(ans*m==n)
     	 cout<<ans*5<<"\n";
     	else cout<<(ans+1)*5<<"\n";
     	}
     	else if(n==0)
     	 cout<<0<<"\n";
     	else
     	 {
     	 	ans=10;
     	 	cout<<ans<<"\n";
     	 }
     }
}
