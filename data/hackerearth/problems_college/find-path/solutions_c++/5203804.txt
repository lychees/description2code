#include<stdio.h>
int n;
int p[100005];
int lis[100005];
int bs(int r,int x)
{
    int l = 0,m,ans = 0;
    while(l<=r)
    {
        m = (l+r)/2;
        if(lis[m]<x)
        {
            ans = m;
            l = m+1;
        }
        else r = m-1;
    }
    return ans;
}
main()
{
    int time,m,i,x;
    scanf("%d",&time);
    while(time--)
    {
        scanf("%d",&n);
        for(i=0;i<n;i++) scanf("%d",&p[i]);
        m = 0; lis[0] = -1;
        for(i=0;i<n;i++)
        {
            x = bs(m,p[i]);
            lis[x+1] = p[i];
            if(x==m) m++;
        }
        printf("%d\n",n-m);
    }
}
