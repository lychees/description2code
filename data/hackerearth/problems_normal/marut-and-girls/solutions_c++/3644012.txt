int
main ()
{

  long long M = 0, N = 0;
  long long marut[101];
  long long hash[10001];


  cin >> M;
  for (long long i = 0; i < M; i++)
    {
      cin >> marut[i];
      hash[marut[i]] = 0;
    }

  cin >> N;
  long long i = 0;
  long long p = 0, count = 0;
  bool isSuitable = true;
  p = getc (stdin);
  int num = 0;
  int numN=0;
  while (1)
    {
      num = 0;
      while (1)
	{
	  p = getc (stdin);
	  if (p == ' ')
	    {
	      hash[num] = 1;
	      //cout << "setting" << num << " to 1\n";
	      num = 0;
	    }
	  else if (p == '\n' || p == EOF)
	    {
	      hash[num] = 1;
	      //cout << "setting" << num << " to 1\n";
	      num = 0;
              numN++;
	      break;
	    }
	  else
	    {
	      num = num * 10 + p - 48;
	      //cout << "num is now " << num << endl;
	    }
	}
      for (i = 0; i < M; i++)
	{
	  if (hash[marut[i]] != 1)
	    {
	      isSuitable = false;
	      //cout << "Girl not good\n";
	    }
	  hash[marut[i]] = 0;
	}
      if (isSuitable)
	count++;
      isSuitable = true;
      //cout<<"numN="<<numN<<"and N="<<N<<endl;
      if(numN == N)
      break;
    }
  cout << count << endl;

}
