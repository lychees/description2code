#include <bits/stdc++.h>
using namespace std;

int main() {
	long int t,n,i,p,count,j;
	cin>>t;
	while(t--)
	{
		cin>>n;
		count=0;
		
	
		vector <long long int> t;
		for(i=0;i<n;i++)
		{
			cin>>p;
			t.push_back(p);
	}
	for(i=0;i<n;i++)
	{
		for(j=i;j<n;j++)
		{
			if(t[i]==t[j])
			count++;
		}
	}
	
	cout<<count<<"\n";
	}
}
