#include<iostream>
#include<vector>
#include<cstdio>
using namespace std;

int main() {

	const int ARRAY_SIZE = 19;
	long long f[ARRAY_SIZE]={0}, g[ARRAY_SIZE]={0};
	f[0] = 0;
	g[0] = 1;
	for (int i = 1; i < ARRAY_SIZE; i++) {
		f[i] = 6 * f[i - 1] + 4 * g[i - 1];
		g[i] = g[i - 1] * 10;
	}

	int fCount[10]={0}, gCount[10]={0};

	gCount[2 + 1] = gCount[3 + 1] = gCount[5 + 1] = gCount[7 + 1] = 1;
	fCount[2 + 1] = fCount[3 + 1] = fCount[5 + 1] = fCount[7 + 1] = -1;
	for (int i = 1; i <= 9; i++) {
		gCount[i] += gCount[i - 1];
		fCount[i] += fCount[i - 1] + 1;
	}

	int t;
	cin >> t;

	while (t--) {
		long long n;
		scanf("%lld", &n);
		long long ans = 0;

		long long remainingNumber = 0;
		for (int digitNumber = 1; n; digitNumber++) {
			int lastDigit = n % 10;

			if (lastDigit == 2 || lastDigit == 3 || lastDigit == 5
					|| lastDigit == 7) {
				ans = remainingNumber + 1;
			}
			ans += fCount[lastDigit] * f[digitNumber - 1]
					+ gCount[lastDigit] * g[digitNumber - 1];
			remainingNumber += lastDigit * g[digitNumber - 1];
			n /= 10;
		}
		printf("%lld\n", ans);
	}
}
