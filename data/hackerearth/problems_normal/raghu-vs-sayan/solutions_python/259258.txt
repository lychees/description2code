T = int(raw_input())
assert 1 <= T <= 100

for i in range(T):
	
	A,B,N = map(int,raw_input().split())
	assert 1 <= N <= 10000
	assert 1 <=A,B <= 1000000000
	
	items = map(int,raw_input().split())
	
	items.sort()
	k = 0;cnta=0;cntb=0
	for j in items:
		k += j
		if k<=A :
			cnta+=1
		if k<=B :
			cntb+=1
	if cnta==cntb:
		print "Tie"
	elif cnta > cntb:
		print "Raghu Won"
	else:
		print "Sayan Won"
