#include <iostream>
#include<vector>
#include<map>
#include<algorithm>
using namespace std;

int main()
{
    int n,a,b;
    cin>>n;
    vector<pair<int, int> > order;
    for(int i=0; i<n; i++)
    {
    	cin>>a>>b;
    	order.push_back(make_pair(a+b,i+1));
    }
    
    sort(order.begin(),order.end());
    for(int i=0; i<n; i++)
    {
    	cout<<order[i].second<<" ";
    }
 
    
    
    return 0;
}
