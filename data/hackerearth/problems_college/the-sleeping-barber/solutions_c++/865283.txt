#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long g,m,s=0;
    	cin>>g>>m;
    	long long arr[g];
    	for(long long i=0;i<g;i++)
    	cin>>arr[i];
    	sort(arr,arr+g);
    	long long i=0;
    	while(s<=m)
    	{
    		s+=arr[i];
    		i++;
    	}
    	cout<<--i<<endl;
    	
    }
    return 0;
}
