#include <bits/stdc++.h>
using namespace std;
string s[505];
int main() {
	int n,i;
	cin >>n;
	for(i=0;i<n;i++){
	    getchar();
	    getline(cin,s[i]);
	}
	/*for(i=0;i<n;i++){
	    cout <<s[i]<<endl;
	}*/
	char ans[1005];
	for(i=0;i<n;i++){
	    int l,ind,j,c=0; 
	    l=s[i].size();
	    //cout <<l<<endl;
	    ind=s[i].find('$');
	    c=0;
	    for(j=ind+1;j<l;j++){
	        if(s[i][j]==' ')
	            continue;
	        if(!(s[i][j]>='0' && s[i][j]<='9'))
	            break;
	        ans[c++]=s[i][j];    
	    }
	    ans[c]='\0';
	    for(j=0;j<c;j++){
	        if(ans[j]!='0')
	            break;
	    }
	    if(j==c){
	        cout <<"$0"<<endl;continue;
	    }
	    ind=j;
	    cout <<"$";
	    for(j=ind;j<c;j++)
	        cout <<ans[j];
	    cout <<endl;
	}
	
	return 0;
}
