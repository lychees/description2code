#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int n;
int arr[1000][1000];

int fun(int i,int j)
{
	if(i==n-1 && j==n-1 && arr[i][j]==1)	return 1;
	if(arr[i][j]==0)	return 0;
	return fun(i+1,j) || fun(i,j+1);
}


int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	cin>>n;
		memset(arr,0,sizeof(arr));
    	for(int i=0;i<n;i++)
    	{
    		for(int j=0;j<n;j++)
    			cin>>arr[i][j];
    	}
    	if(fun(0,0))  cout<<"POSSIBLE"<<endl;
    	else cout<<"NOT POSSIBLE"<<endl;
    }
    return 0;
}
