#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,m;
    	cin>>n>>m;
    	int a[n][m];
		int dp[n][m];
    	for(int i=0;i<n;i++)
    		for(int j=0;j<m;j++)
    		{
    			cin>>a[i][j];
    			dp[i][j]=0;
    		}
    	//Base Cases
    	dp[0][0]=1;
    	
    	for(int i=1;i<n;i++)
    		if(dp[i-1][0]!=0)
    		{
    			if(a[i][0]>a[i-1][0]) 
    				dp[i][0]=dp[i-1][0]+1;
    			else 
    				dp[i][0]=0;
    		}
    	/*for(int i=0;i<n;i++)
    	{	
    		for(int j=0;j<m;j++)
    			cout<<dp[i][j]<<" ";
    		cout<<endl;
    	}*/
    	
    	for(int i=1;i<m;i++)
    		if(dp[0][i-1]!=0)
    		{
    			if(a[0][i]>a[0][i-1])
    				dp[0][i]=dp[0][i-1]+1;
    			else
    				dp[0][i]=0;
    		}
    	//Bases cases end
    	/*for(int i=0;i<n;i++)
    	{	
    		for(int j=0;j<m;j++)
    			cout<<dp[i][j]<<" ";
    		cout<<endl;
    	}*/
    	
    	for(int i=1;i<n;i++)
    		for(int j=1;j<m;j++)
    		{
    			if(dp[i-1][j]==0 && dp[i][j-1]==0)
    				dp[i][j]=0;
    			else
    			{
    				if(a[i][j]>a[i-1][j] && dp[i-1][j]>0)
    					dp[i][j]=dp[i-1][j]+1;
    				if(a[i][j]>a[i][j-1] && dp[i][j-1]>0)
    					dp[i][j]=max(dp[i][j],dp[i][j-1]+1);
    			}
    				
    		}
    	/*<<"T="<<t<<endl;
    	for(int i=0;i<n;i++)
    	{	
    		for(int j=0;j<m;j++)
    			cout<<dp[i][j]<<" ";
    		cout<<endl;
    	}*/
    		
    	int ans=-1;
    	for(int i=0;i<n;i++)
    		for(int j=0;j<m;j++)
    			ans=max(ans,dp[i][j]);
    	cout<<ans<<endl;
    	
    }
    return 0;
}
