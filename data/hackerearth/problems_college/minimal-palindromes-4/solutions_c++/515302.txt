#include<iostream>
#include<cstring>
using namespace std;

void check( char *n)
{

	unsigned int len=strlen(n),max=0,mid;
	unsigned int i;
	for(i=0;i<len;i++)
	n[i]=n[i]-'0';
	
	if(len%2==0)
	mid=len/2;
	else
	mid=((len/2)+1);

	max=n[0];
	for(i=0;i<mid;i++)
	if(n[i]>max)
	max=n[i];
	
	
	for(i=0;i<len&&(max>0);)
	if(n[i]%max==0)
	{
		i++;
	}
	else 
	{
		max--;
		i=0;
	}
	
	if(max!=0)
	for(i=0;i<len;i++)
	n[i]=n[i]/max;
	
	for(i=0;i<len;i++)
	n[i]=n[i]+'0';
	
	cout<<n;
	
}

int main()
{
	 char *a;
	 a=new char[10000];
	 unsigned int tc;
	 cin>>tc;
	 while(tc--)
	 {
	 cin>>a;
	 check(a);
	 cout<<"\n";
	 }
	 
	 
return 0;
}
