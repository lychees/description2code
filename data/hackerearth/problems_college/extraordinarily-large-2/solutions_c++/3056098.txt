#include <bits/stdc++.h>
using namespace std;

int fac(char a[],int al){
	int count=0;
	al--;
	while(1){
		if(a[al]=='!'){
			count++;
			al--;
		}
		else break;
	}
	a[al+1]='\0';
	return count;
}

long long int factorial(long int bn){
	if(bn<=1) return 1;
	else return bn*factorial(bn-1);
}


int main()
{
    int t;
    cin>>t;
    
    while(t--){
    	char a[205],b[205];
    	scanf("%s %s",a,b);
    	
    	int al=strlen(a);
    	int bl=strlen(b);
    	
    	int f1=fac(a,al);
    	int f2=fac(b,bl);
    	
    	long int an=atoi(a);
    	long int bn=atoi(b);
    	
    	
    	if(f1<f2){
    		while(f1!=f2){
    			if(bn>20){
    				cout<<"x<y"<<endl;
    				break;
    			}
    			bn=factorial(bn);
    			f2--;
    		}
    	}
    	else if(f2<f1){
    		while(f2!=f1){
    			if(an>20){
    				cout<<"x>y"<<endl;
    				break;
    			}
    			an=factorial(an);
    			f1--;
    		}
    	}
    	
    	
    	if(f2==f1){
    		if(an==bn)cout<<"x=y"<<endl;
    		else if(an>bn) cout<<"x>y"<<endl;
    		else cout<<"x<y"<<endl;
    	}
    	
    }
    return 0;
}
