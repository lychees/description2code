#include<bits/stdc++.h>
using namespace std;

#define gc getchar_unlocked
inline void scanint(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}
int main()
{
    int n,m,q;
    scanint(n);
    scanint(m);
    scanint(q);
    int nm[n][m];
    int i,j,a,ind;
    for(i=0;i<n;i++)for(j=0;j<m;j++)scanint(nm[i][j]);
    while(q--)
    {ind=m-1;
    	for(j=0;j<n;j++)
    		{
            	scanint(a);
                while(a<nm[j][ind])
                    ind--;
            }
        cout<<ind+1<<"\n";
    }
return 0;
}
