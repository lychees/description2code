#include <iostream>
#include<math.h>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	cin>>t;
	for(int i=0;i<t;i++)
	{
		int n;
		cin>>n;
		
		int man[1000];
		int women[1000];
		for(int i=0;i<n;i++)
		cin>>man[i];
		for(int i=0;i<n;i++)
		cin>>women[i];
		sort(man,man+n);
		sort(women,women+n);
		long long unsigned int pro=0;
		for(int i=0;i<n;i++)
		{
		pro=pro+(man[i]*women[i]);
		}
		
		
		cout<<pro<<endl;
	}
}
