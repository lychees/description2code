#include <iostream>
using namespace std;

int main()
{
    int t,i,j,k,l;
    cin>>t;
    long long int n,ans=0;
    for(i=0;i<t;i++)
    {
    	ans=0;
    	cin>>n;
    	ans+=n;
    	int a[n];
    	for(j=0;j<n;j++)
    	{
    		cin>>a[j];
    	}
    	for(j=0;j<n-1;j++)
    	{
    		for(k=j+1;k<n;k++)
    		{
    			if(a[j]==a[k])
    			ans++;
    		}
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
