#include<stdio.h>
#include<iostream>
#include<string.h>
#include<algorithm>
using namespace std;

#define MAX 1000000
#define LMT 1000

int flag[MAX>>6];

#define ifc(x) (flag[x>>6]&(1<<((x>>1)&31)))
#define isc(x) (flag[x>>6]|=(1<<((x>>1)&31)))

long int a[100006]={0},l=2;

void sieve()
{
        long int i, j, k,x;

        for(i=3; i<LMT; i+=2)
        {
            if(!ifc(i))
                {
                    a[l++]=i;
                        for(j=i*i,k=i<<1; j<MAX; j+=k)
                                isc(j);
                }
        }
        for(i=LMT+1; i<MAX; i+=2)
        {
            if(!ifc(i))
                {
                    a[l++]=i;
                }
        }

}

int main()
{
    long t,m,k,temp;
    long int ans,i;
    sieve();
    scanf("%ld",&t);
    while(t--)
    {
        scanf("%ld%ld",&m,&k);
        ans=0;
        l=2;
        for(i=m;i<=k;i++)
        {
            while(a[l]<i)
                l++;

            if(a[l]==i)
                continue;
            temp=i;
            while(temp>0)
                {
                    if(temp%10==7)
                    {
                        ans++;
                        break;
                    }

                    temp/=10;
                }

           // printf("%ld ",i);
        }
        if(ans==0)
            printf("-1\n");
        else
        printf("%ld\n",ans);

    }

    return 0;
}
