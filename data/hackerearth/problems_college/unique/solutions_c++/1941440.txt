#include<iostream>
#include<stdio.h>
#include<stdlib.h>
using namespace std;
int compare (const void * a, const void * b)
{
  return ( *(int*)a - *(int*)b );
}
int main(){
	long long int t;
	cin>>t;
	while(t--){
		long long int n,i,j;
		cin>>n;
		 int a[n];
	
		for(i=0;i<n;i++){
			cin>>a[i];
		}
		     qsort (a, n, sizeof(int), compare);
		     long long int c=0;
		     for(j=0;j<n;j++){
		     	if(a[j] != a[j+1]){
		     		c++;
				 }
			 }
			 cout<<c<<"\n";
	}
	return 0;
}
