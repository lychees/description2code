#include <iostream>
using namespace std;

int main()
{
    int t,r,g,b,bl=0,R,G,B,W=0,Y=0,M=0,C=0,x=0,y=0,z=0;
    cin>>t>>r>>g>>b;
    for(int i=0;i<t;i++)
    { if((i/r)%2 !=0)
         R=1;
         else
         R=0;
       if((i/g)%2 !=0)
        G=1;
        else
        G=0;
       if((i/b)%2 !=0)
       B=1;
       else
       B=0;

       if(R&&G&&B)
        W++;
        else if(R&&G)
        Y++;
        else if(R&&B)
        M++;
        else if(G&&B)
        C++;
        else if(R)
        x++;
        else if(G)
        y++;
        else if(B)
        z++;
        else
        bl++;
    }
    cout<<x<<" "<<y<<" "<<z<<" "<<Y<<" "<<C<<" "<<M<<" "<<W<<" "<<bl;
    return 0;
}
