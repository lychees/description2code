#include <bits/stdc++.h>
#define ll long long
using namespace std;
struct worker
{
	ll t, x, y;
};
bool mcompare(worker A, worker B)
{
	if(A.t!=B.t)
	return A.t<B.t;
	else if(A.y!=B.y)
	return A.y>B.y;
	else
	return A.x<B.x;
}
worker W[100002];
int main()
{
    ll D, n, i, ans, cur_w, x;
    cin>>n>>D;
    for(i=0;i<n;i++)
    cin>>W[i].t>>W[i].x>>W[i].y;
    sort(W, W+n, mcompare);
    ans=W[0].x;
    cur_w=0;
    for(i=1;i<n;i++)
    {
    	x=W[i].t-W[cur_w].t;
    	if(D<=x*W[cur_w].y)//work completed before new worker's arrival
    	{
    		D=0;
    		break;
    	}
    	if(W[cur_w].y<W[i].y)
	    {
	    	D-=x*W[cur_w].y;
	    	ans+=W[i].x;
	    	cur_w=i;
	    }
    }
    cout<<ans;
    return 0;
}
