def check(a, l):

    for i in range(0, l-1):
        x = len(a[i])
        j = i+1
        if (len(a[j]) >= x and a[i] == a[j][0:(x)]):
            return 1
    return 0


t = int(raw_input())
while (t > 0):
    t = t - 1
    n = int(raw_input())
    a = []
    while (n > 0):
        n -= 1
        a.append(raw_input())
    a.sort()

    if check(a, len(a)) == 1:
        print('NO')
    else:
        print('YES')
