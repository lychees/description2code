#include <cstdio>

#define ll long long int
#define maxN (1<<18)
#define maxVal 1000000
#define minVal -100000000
#define mod 1000000007LL

#define gcd(a,b) __gcd(a,b)

using namespace std;

int a[maxN+5];
vector<int> l,r;

int main()
{
	ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    #ifndef LOCAL_SYS
        //freopen("in.txt","r",stdin);
        //freopen("out.txt","w",stdout);
    #endif
    int t,n,i,j,k;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	for(i=0;i<n;i++)
    		scanf("%d",&a[i]);
    	l.clear();
    	r.clear();
    	for(i=0;i<=(int)log2(n);i++)
    	{
    		j=(1<<i)-1;
    		k=(1<<(i+1))-2;
    		while(j<=k)
    		{
    			if(a[j]==0)
    				j++;
    			if(a[k]==0)
    				k--;
    			if(a[j]!=0&&a[k]!=0)
    				break;
    		}
    		if(a[j]!=0&&a[k]!=0)
    		{
    			r.push_back(a[k]);
    			if(j!=k)
    				l.push_back(a[j]);
    		}
    	}
    	for(i=0;i<(int)r.size();i++)
    		printf("%d\n",r[i]);
    	for(i=0;i<(int)l.size();i++)
    		printf("%d\n",l[i]);
    	printf("\n");
    }
    return 0;
}
