#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    long long m,temp=0;
    cin>>m;
    long long* ar=new long long[m];
    for(long long i=0;i<m;i++)
    {
    	for(long long j=0;j<m;j++)
    	{
    		cin>>ar[j];
    	}
    	temp = temp + ar[i] - ar[m-1-i];
    }
    cout<< abs(temp) <<endl;
}
