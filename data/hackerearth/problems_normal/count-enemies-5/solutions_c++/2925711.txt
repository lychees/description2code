#include <iostream>
#include <string>
using namespace std;

int main()
{
    int T;

    int cnt=0,temp=0;
    bool x_found = false;
    scanf("%d", &T);
    while(T--) {
		string s;
		cin>>s;
    	cnt = 0;
    	temp=0;
    	x_found = false;
    	for(int i=0; i<s.length(); i++) {
    		if(s[i] == '*') {
    			x_found =  false;
    			temp=0;
    		} else if(s[i] == 'O') {
    			if (!x_found) {
    				cnt++;
    				temp++;
    			}
    		} else { //s[i] == 'X';
    			x_found = true;
    			cnt-=temp;
    			temp = 0;
    		}	
    	} 
    	cout<<cnt<<"\n";
    }
    return 0;
}
