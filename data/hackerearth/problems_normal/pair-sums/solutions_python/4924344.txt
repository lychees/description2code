First = raw_input()
First = First.split(" ")
temp = {}
found = 0
N,K = int(First[0]), int(First[1])
Second = raw_input()
Second = Second.split(" ")
for i in range(0,len(Second)):
	x = K - int(Second[i])
	if int(Second[i]) in temp:
		found = 1
		break
	else:
		temp[x]=1
if found == 1:
	print "YES"
else:
	print "NO"
