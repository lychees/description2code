#include <cstdio>

int main(){
 int t;
 scanf("%d",&t);
 while(t--){
 int n;
 scanf("%d", &n);
 int a[n];
 int i;
 for(i=0; i<n; i++){
  scanf("%d", &a[i]);
 }
 int temp2 = 0, temp3 = 0;
 for(i=0; i<n; i++){
  if(a[i] == -1){
   temp2 ++;
  }
  else{
   break;
  }
 }
 for(i=n-1; i>=0; i--){
  if(a[i] == 1){
   temp3++;
  }
  else{
   break;
  }
 }
 
 int ans = 0;
 for(i=temp2; i <= n-temp3-1; i++){
  if(a[i] == -1){
   ans ++;
  }
 }
 if(temp2 >= temp3){
  printf("%d\n", temp2 + ans);
 }
 else{
  printf("%d\n", temp2 + ans + 1);
 }
 }
 return 0;
}
