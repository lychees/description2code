#include<bits/stdc++.h>
using namespace std;
main()
{
	int t;
	cin>>t;
	while(t--)
	{
		char str[100000];
		cin>>str;
		int l=strlen(str);
		int count[27];
		for(int i=0;i<26;i++)
		count[i]=0;
		for(int i=0;i<l;i++)
		count[str[i]-'A']++;
		int flag=0;
		cin.ignore();
		for(int i=0;i<26;i++)
		{
			if(count[i]>0)
			{
				if(count[i]==26-i)
				flag=1;
				else
				{
					flag=0;
					break;
				}
			}
		}
		if(flag==1)
		cout<<"Yes"<<endl;
		else
		cout<<"No"<<endl;
	}
}
