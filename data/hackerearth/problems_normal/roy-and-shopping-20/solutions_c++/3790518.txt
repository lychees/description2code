#include <iostream>
#include <stdio.h>
using namespace std;
typedef long long int ll;
int a[1000001];
int main()
{
    ll  i, j, max = 1000000;
    for(i = 2; i <= max; i++) {
    	if(a[i] == 0) {
    		a[i] = i;
    		for(j = i*i; j <= max; j+=i) {
    			if(a[j] == 0) {
    				a[j] = i;
    			}
    		}
    	}
    }
    int t;
    scanf("%d", &t);
    while(t--) {
    	int val;
    	scanf("%d", &val);
    	val =  val - a[val];
    	printf("%d\n", val);
    }
    return 0;
}
