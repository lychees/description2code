
#include <iostream>

using namespace std;

#define ll long long





int main() {

	ll T;
	cin>>T;

	for (ll j=0 ; j<T ; ++j) {

		ll N,K,M;
		cin>>N>>K>>M;

		ll sum = ( N*(N+1) ) / 2;
		sum *= sum;
		sum *= K;

		ll ans = sum - M;
		ans = ans < 0 ? 0 : ans;
		cout<<ans<<endl;
	}


}
