#include <iostream>
#include <math.h>
#include <algorithm>
using namespace std;

int main()
{
    unsigned short int T;
    cin >> T;
    while(T--)
    {
    	unsigned long a, b;
    	cin >> a >> b;
    	
    	unsigned long aHeight = (unsigned long)(log2(a));
    	unsigned long bHeight = (unsigned long)(log2(b));
    	
    	if(aHeight < bHeight)
    	{
    		swap(aHeight, bHeight);
    		swap(a, b);
    	}
    	
    	while(aHeight != bHeight)
    	{
    		a = (unsigned long)(a/2);
    		aHeight--;
    	}
    	
    	while(a != b)
    	{
    		a = (unsigned long)(a/2);
    		b = (unsigned long)(b/2);
    	}
    	
    	cout << a << endl;
    	
    }
    return 0;
}
