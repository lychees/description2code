#include <bits/stdc++.h>
using namespace std;

#define MAX 401

int main() {
    long long int i, n, t, m, temp, j;
    bool flag;
    cin>>t;
    while (t--) {
        cin>>n;
        flag = true;
        long long int a[n];
        for (i=0; i<n; i++) cin>>a[i];
        cin>>m;
        long long int r[m][n], dist[m][n];
        for (i=0; i<m; i++) {
            for (j=0; j<n; j++) {
                cin>>r[i][j];
            }
        }
        for (i=0; i<m; i++) {
            for (j=0; j<n; j++) {
                cin>>temp;
                dist[i][j] = temp - r[i][j];
                r[i][j] = temp;
            }
        }
        for (i=0; i<m; i++) {
            sort(r[i], r[i]+n);
            for (j=0; j<n; j++) {
                if (dist[i][j] <= a[j]) a[j] += r[i][j];
                else {
                    flag = false;
                    break;
                }
            }
            if (flag == false) break;
        }
        if (flag == true) cout<<"YES"<<endl;
        else cout<<"NO"<<endl;
    }
	return 0;
}
