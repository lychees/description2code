#include <iostream>
#include <map>
#include <cstdio>
using namespace std;

int main()
{
	int t,n;
	
	int a[28], k = 1,ans;
	
	for (int i = 0; i < 28; i++) {
		a[i] = k;
		k = k * 2;
	}
	
	scanf("%d", &t);
	
	for (int i = 0; i < t; i++) {
		scanf("%d", &n);
		
		for (int j = 27; j >= 0; j--) {
			if (n - a[j] >= 0) {
				ans = n - a[j];
				break;
			}
		}
		printf("%d\n", 2 * ans + 1);
		
	}
	
    return 0;
}
