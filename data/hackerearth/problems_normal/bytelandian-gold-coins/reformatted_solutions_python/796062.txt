exchange = {0: 0}


def convert(x):
    if x in exchange:
        return exchange[x]
    else:
        exchange[x] = max(x, convert(x/4)+convert(x/3)+convert(x/2))
        return exchange[x]


while True:
    try:
        num = input()
        print convert(num)
    except BaseException:
        break
