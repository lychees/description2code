#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n,i=0,size,j;
	cin >> n;
	size =n;
	vector <int> v(n);
	while(n--)
	{
		cin>>v[i++];
	}
	i=0;
	while(i<size)
	{
		for(j=0;j<size;j++)
		{
			if(i!=j && v[i]%v[j]==0)
				break;
		}
		if(j==size)
			cout<<v[i]<<" ";
		i++;
	}
    return 0;
}
