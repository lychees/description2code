#include<iostream>
using namespace std;

int stck[5000],top=-1;
void push(int num)
{
    if(top==-1)
    {
        stck[0]=num;
        top=0;
    }
    else
    {
        top++;
        stck[top]=num;
    }
}
void pop()
{
    if(top<0)
        return;
    top=top-1;
}
int main()
{
    int rating,i,n,sum=0;
    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>rating;
        if(rating!=0)
            push(rating);
        else
            pop();
    }
    for(i=top;i>=0;i--)
        sum=sum+stck[i];
    cout<<"\n"<<sum;
    return 0;
}
