#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int a[100020];
int b[100020];
#define max 101000
int main()
{
    int t;ll ans,tem,sum;
    cin>>t;int n;ll l,r;
    while(t--)
    {
    for(int i=0;i<max;i++)b[i]=0;
     cin>>n;
     for(int i=1;i<=n;i++)
     cin>>a[i];
     ans=tem=max;
     sum=0;
    for(int i=1;i<=n;i++)
    {
        sum+=a[i];
        sum%=n;
        if(b[sum]==0)
        {
            if(sum==0)
              tem=i;
              else
              tem=max;
        }
        else
        tem=i-b[sum];
        if(tem<ans)
        {
            ans=tem;
            l=b[sum]+1;
            r=i;
        }
        b[sum]=i;

    }
    cout<<l<<" "<<r<<endl;


    }



    return 0;
}
