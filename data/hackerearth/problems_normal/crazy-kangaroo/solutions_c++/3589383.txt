#include <iostream>
using namespace std;
long solve(long a,long b,long m)
{
	long res = (b/m)-(a/m);
//	long res = 1;
	if(a%m == 0)
	return res+1;
	else
	return res;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long a,b,m;
    	cin>>a>>b>>m;
    	cout<<solve(a,b,m)<<endl;
    }
    return 0;
}
