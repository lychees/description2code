#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--) {
    	int n;
    	cin >> n;
    	vector<int>men(n);
    	vector<int>women(n);
    	for(int i = 0; i < n; i++) {
    		cin >> men[i];
    	}
    	for(int i = 0; i < n; i++) {
    		cin >> women[i];
    	}
    	sort(men.begin(), men.end());
    	sort(women.begin(), women.end());
    	int ans = 0;
    	for(int i = 0; i < n; i++) {
    		ans += (men[i] * women[i]);
    	}
    	cout << ans << endl;
    }
    return 0;
}
