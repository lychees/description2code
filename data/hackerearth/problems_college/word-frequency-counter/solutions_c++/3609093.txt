#include<bits/stdc++.h>
using namespace std;

int main()
{
    char s[100000];
    gets(s);

    string str="";
    map<string,int> m;
    vector<string> v;
    map<string,bool> visited;

    for(int i=0;i<strlen(s);i++)
    {
       if(s[i]==' ')
       {
           m[str]++;
           if(visited[str]==false)
           {
           	v.push_back(str);
           	visited[str]=true;
           }
           str="";

       }
       else if(i==strlen(s)-1)
       {
           str+=toupper(s[i]);
           m[str]++;
           if(visited[str]==false)
           {
           	v.push_back(str);
           	visited[str]=true;
           }
           str="";

       }
       else
       {
           str+=toupper(s[i]);
       }
    }

    for(int i=0;i<v.size();i++)
    {
        cout<<v[i]<<" "<<m[v[i]]<<endl;
    }


}
