#include<iostream>
using namespace std;
int main(){
    int arr[26];
    int t,n,i,flag;
    string city;
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(i=0;i<26;i++)
            arr[i]=0;
            flag=1;
        while(n--)
        {
            cin>>city;
            arr[city[0]-'a']++;
        }
        for(i=0;i<26;i++)
        {
            if(arr[i]>1)
            {
                flag=0;
                break;
            }
        }
        if(flag==0)
        {
            cout<<"NO\n";
        }
        else
        {
            cout<<"YES\n";
        }
    }
}
