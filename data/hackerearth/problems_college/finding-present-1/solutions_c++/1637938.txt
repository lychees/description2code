#include <stdio.h>

using namespace std;

int main()
{
	long long int p,q;
	scanf("%lld", &p);
	int y1[p];
	for(long long int i=0;i<p;i++)
		scanf("%d", &y1[i]);
	scanf("%lld", &q);
	int y2[q];	
	for(long long int i=0;i<q;i++)
		scanf("%d", &y2[i]);
	static int list1[10000], list2[10000];
	for(long long int i=0;i<p;i++)
		list1[y1[i]]++;
	for(long long int i=0;i<q;i++)
		list2[y2[i]]++;
	for(int i = 0;i<10000;i++)
	{
		if(list1[i] - list2[i] != 0)
		{
			printf("%d ", i);
			list1[i] = 0;
			list2[i] = 0;
		}
	}
	printf("\n");
	return 0;
}
