#include <iostream>
#include <vector>
#include <algorithm>
#include <string>
#include <map>
using namespace std;

void func(vector<string> v){
	 map<string, int> c;
	 int n=v.size();
	 for(int i=0;i<n;i++){
	 	c[v[i]]++;
	 }
	for (map<string,int>::iterator it = c.begin(); it != c.end(); it++){
    cout<< it->first <<" "<<it->second<<'\n';
}
return;
}
int main() {
	int n;
	cin>>n;
	vector<string> v;
	for(int i=0;i<n;i++){
		string s;
		cin>>s;
		v.push_back(s);
	}
	func(v);
	return 0;
}
