#include <iostream>
#include<math.h>
using namespace std;

int main()
{
	long long int t,n,i,tib[1000006];
    cin>>t;
    
    tib[0]=1;
    tib[1]=1;
    for(i=2;i<=1000000;i++)
     tib[i] = (tib[i-1] + tib[i-2])%1000000007;
     
    while(t--)
    {
      cin>>n;
     
      cout<<tib[n]<<endl;
    }
    return 0;
}
