#include<stdio.h>
#define gc() getchar()
//#define gc() getchar_unlocked()
inline int fastread()
{
int i=0;
char c=gc();
while(c<48) c=gc();
while(c>47)
{
i=i*10+c-48;
c=gc();
}
return i;
}
int main()
{
    int t,n,i,j,flag,x;
    long int count,counte,counto;
    t=fastread();
    while(t--)
    {
        n=fastread();
        count=counte=counto=0;
        for(i=0;i<n;i++)
        {
            x=fastread();
            if(x%2==0)
            {
                counte++;
                count+=counto;
            }
            else
            {
                counto++;
                count+=counte;
            }
        }


        printf("%ld\n",count);
    }
    return 0;
}
