#include<bits/stdc++.h>
using namespace std;
int main()
{
    cin.sync_with_stdio(false);
    int t,len,ans,i,j;
    string str;
    cin>>t;
    while(t--)
    {
        ans=0;
        cin>>str;
        len=str.size();
        for(i=0;i<len;i++)
            if(str[i]=='-')
                if(str[i+1]!='B' && str[i+2]!='B' && str[i+1]!='S' && str[i-1]!='S')
                        ans++;
        cout<<ans<<endl;
    }
    return 0;
}
