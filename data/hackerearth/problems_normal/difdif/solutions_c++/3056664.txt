#include<bits/stdc++.h>
#include <iostream>
using namespace std;
#define ll long long
int main()
{
	std::ios::sync_with_stdio(false);
    //cout << "Hello World!" << endl;
    string s;
    cin>>s;
    unordered_set <string> myset;
    for(ll i=0;i<s.length();++i)
    {
	    	string v;
	    	v = v+s[i];
	    	myset.insert(v);
	    	for(ll j=i+1;j<s.length();++j)
	    	{
	    		v = v + s[j];
	    		myset.insert(v);
	    	}
	    	//cout<<v<<endl;
    }
    cout<<myset.size()<<endl;
    return 0;
}
