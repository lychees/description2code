def gcd(x, y):
    return x if y == 0 else gcd(y, x % y)


a = ['Chandu', 'Daspal']
for _ in range(int(raw_input())):
    x, y, s = raw_input().split()
    x, y = int(x), int(y)
    z = a.index(s)
    p = gcd(x, y)
    while x != 0 and y != 0:
        x, y = x-p, y-p
        p = gcd(x, y)
        z = z ^ 1
    print a[z ^ 1]
