vii v[SIZE];
bool visit[SIZE];
bool is;
int ans[SIZE];
void dfs(int i,int temp)
{
	visit[i]=true;
	ans[i]=temp;
	for(int j=0;j<v[i].size();j++){
		if(!visit[v[i][j]]){
			if(temp%2==0){
				dfs(v[i][j],temp-1);
			}
			else{
				dfs(v[i][j],temp+1);
			}
		}
		else{
			if(ans[i]==ans[v[i][j]])
			is=false;
		}
	}
	return ;
}
int expo[SIZE];
void func()
{
	expo[0]=1;
	for(int i=1;i<=1e4;i++){
		expo[i]=expo[i-1]*2;
		expo[i]%=MOD;
	}
	return ;
}
