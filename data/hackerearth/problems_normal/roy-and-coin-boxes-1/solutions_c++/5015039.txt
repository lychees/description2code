

#include <iostream>
#include <vector>

std::vector<int> vs;
std::vector<int> tree;
int main()
{
	//std::freopen("coin.txt","r",stdin);
	int n;
	std::cin>>n;
	vs.resize(n+2,0);
	tree.resize(n+2,0);
	int m;
	std::cin>>m;
	while(m--){
		int l,r;
		std::cin>>l>>r;
		vs[l]++;
		vs[r+1]--;
	}
	int block=0;
	for(int i=1 ; i<=n ;++i){
		block+=vs[i];
		tree[block]++;
	}
	for(int i=1 ; i<=n ; ++i) 
		tree[i]+=tree[i-1];
	int q;
	std::cin>>q;
	while(q--){
		int x;
		std::cin>>x;
		std::cout<<n-tree[x-1]<<std::endl;
	}
}
