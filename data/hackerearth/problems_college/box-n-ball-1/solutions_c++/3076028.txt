#include <iostream>
#include <iomanip>
#include <cmath>
using namespace std;

int main()
{
    unsigned long int t,i,l,b,h;
    cin>>t;
    double a[t];
    for(i=0;i<t;i++)
    {
    	cin>>l>>b>>h;
    	a[i]=(double)l/2;
    	if((b/2)<a[i])
        	a[i]=(double)b/2;
    	if((h/2)<a[i])
        	a[i]=(double)h/2;
    }
    for(i=0;i<t;i++)
    {
    	cout.setf(ios::fixed);
        cout << setprecision(1) << a[i] << endl;
   	}
    return 0;
}
