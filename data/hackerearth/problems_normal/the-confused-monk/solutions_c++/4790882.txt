lli power(lli a, lli b){
    lli ans=1LL;
    while(b!=0){
        if(b%2==1){
            ans=(ans*a)%mod;
        }
        a=(a*a)%mod;
        b/=2;
    }
    return ans;
}

lli gcd(lli a, lli b){
    if(a%b==0)
        return b;
    else
        return gcd(b,a%b);
}

int main(){
    opt
    lli n,i,p=1;
    vlli v;
    cin>>n;
    rep(i,0,n){
        lli x;
        cin>>x;
        p=(p*x)%mod;
        v.pb(x);
    }
    sort(all(v));
    reverse(all(v));
    lli x=v[0];
    rep(i,1,n){
        if(x>=v[0])
            x=gcd(x,v[i]);
        else
            x=gcd(v[i],x);
    }
    lli ans;
    ans=power(p,x);
    cout<<ans<<"\n";
    return 0;
}
