#include<bits/stdc++.h>
using namespace std;
long long int be(long long int c[],long long int p,long long int q,long long int x)
{long long int ans=-1;
    while(p<=q)
    {
       long long int mid=(p+q)/2;
       if (c[mid]<=x)
       {
           ans=mid;
           q=mid-1;
       }
       else if (c[mid]>x) p=mid+1;

    }
   return ans;
}
int main()
{
   long long int t;
    cin>>t;
    while(t--)
    {
     long long n;
     cin>>n;
     long long int a[n],b[n];
     for(int i=0;i<n;i++) cin>>a[i];
     for(int i=0;i<n;i++) cin>>b[i];
     long long int maxx=0;
     for(int i=n-1;i>=0;i--)
     {
         long long int val;
         val=be(a,0,n-1,b[i]);
         if(val!=-1)
         {
         maxx=max(maxx,i-val);
         	
         }
     }

     cout<<maxx<<endl;
    }
return 0;
}
