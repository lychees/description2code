#include <iostream>
using namespace std;
int arr[1000001][10];
int func(int n)
{
     static int max = 0;
     int sum = 0;
     if(n == 0) return 0;
     if(n == 1) return 10;
     if(max == 0){
     	for(int i = 0; i < 10; i++)
     	   arr[0][i] = 1;
     }
     if(n > (max + 1)){
     	for(int i = max+1; i < n; i++){
     		for(int j = 0; j < 10; j++){
     			if(j == 0){
     				arr[i][j] = arr[i-1][j+1];
     			} else if(j == 9) {
     				arr[i][j] = arr[i-1][j-1];
     			} else {
     				arr[i][j] = (arr[i-1][j-1] + arr[i-1][j+1])%1000000007;
     			}
     		}
     	}
     	max = n-1;
     }
     for(int i =1; i < 10; i ++){
     	sum = (sum + arr[n-1][i])%1000000007;
     }
     return sum;
}
int main()
{
    int T, N;
    cin >> T;
    for(int i = 0; i < T; i++)
    {
    	cin >> N;
    	cout << func(N) << endl;
    }
    return 0;
}
