#include <iostream>
using namespace std;

long int out[1024][1024];
void setMatrixElements(int i, int j, int n) {
	static long counter = 0;
	if (n==0) {
		counter++;
		out[i][j] = counter;
		return;
	}
	setMatrixElements(i, j, n-1);
	setMatrixElements(i, j+(1<<n-1), n-1);
	setMatrixElements(i+(1<<n-1), j, n-1);
	setMatrixElements(i+(1<<n-1), j+(1<<n-1), n-1);
}

int main() {
	int n;
	cin>>n;

	setMatrixElements(0, 0, n);
	for (int i=0; i<(1<<n); i++) {
		for (int j=0; j<(1<<n); j++)
			cout<<out[i][j]<<" ";
		cout<<endl;
	}
	
	return 0;
}
