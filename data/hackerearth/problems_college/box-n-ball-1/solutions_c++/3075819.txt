#include <cstdlib>
#include <iostream>
#include <algorithm>

using namespace std;
double l,b,h;
long T;
int main(int argc, char *argv[])
{   cin>>T;
    while(T--)
    {
       cin>>l>>b>>h;
       cout.setf(ios::fixed,ios::floatfield);
       cout.precision(1);
       cout<<min(min(l,b),h)/2<<endl;
    }
    
    return EXIT_SUCCESS;
}
