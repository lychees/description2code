#include <iostream>
using namespace std;
int a[100005], b[100005];
int main()
{
	int n, k;
	long long c = 0;
	cin >> n >> k;

    for(int i = 0; i< k; i++) {
       scanf("%d", &a[i]);
    }
    for(int i = 0; i< k; i++) {
       scanf("%d", &b[i]);
    }
    for(int i = 0; i< k; i++) {
        int q = abs(1-a[i]) + abs(1-b[i]);
        int w = abs(n-a[i]) + abs(n-b[i]);
        int e = abs(1-a[i]) + abs(n-b[i]);
        int r = abs(n-a[i]) + abs(1-b[i]);
       c += min(q, min(w, min(e, r)));
    }
    cout << c<< endl;
    return 0;
}
