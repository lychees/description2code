#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define max 100005
const ll mod=1e9+7;
ll fact[max],ifact[max];
ll inverse(ll a,ll n)
{
    ll ret=1;
    while(n)
    {
        if(n&1)
        {
            ret*=a;
            ret%=mod;
        }
        a*=a;
        a%=mod;
        n>>=1;
    }
    return ret;
}
void factCompute()
{
    fact[0]=fact[1]=1;
    for(ll i=2;i<max;i++)
        fact[i]=(fact[i-1]*i)%mod;
    ifact[max-1]=inverse(fact[max-1],mod-2);
    for(ll i=max-2;i>=0;i--)
        ifact[i]=(ifact[i+1]*(i+1))%mod;
}
ll comb(ll n,ll k)
{
    ll ret=fact[n];
    ret*=ifact[k];
    ret%=mod;
    ret*=ifact[n-k];
    ret%=mod;
    return ret;
}
int main()
{
    ll t,n,k;
    cin>>t;
    factCompute();
    while(t--)
    {
        cin>>n>>k;
        ll power[n+5];
        for(int i=0;i<n;i++)
            cin>>power[i];
        ll ans=0;
        sort(power,power+n);
        for(int i=0;i<n-k+1;i++)
        {
            ans+=comb(n-i-1,k-1)*power[i];
            ans%=mod;
        }
        cout<<ans<<endl;
    }
    return 0;
}
