#include<bits/stdc++.h>
using namespace std;
typedef int ll;
int a[26];
int main()
{
 	//freopen("in.txt","r",stdin);
	//freopen("out.txt","w",stdout);
    ll i;
    string s;
    cin>>s;
    bool f=true;
    bool c=true;
    for(i=0;i<s.length();++i)
    {
    	++a[s[i]%97];
	}
	for(i=0;i<26;++i)
    {
    	if(a[i]%2==1&&!f)
    	{
    		c=false;
    		break;
		}
		if(a[i]%2==1)
		{
			f=false;
		}
	}
	if(c)
	cout<<"YES\n";
	else
	cout<<"NO\n";
    return 0;
}
