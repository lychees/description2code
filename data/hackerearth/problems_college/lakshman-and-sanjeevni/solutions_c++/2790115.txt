int arr[MAX],a[MAXX],sieve[MAXX];
void pre(){
	for(int i=2;i*i<MAXX;i++)
		if(!sieve[i]){
			int ps = i*i;
			for(int j=ps;j<MAXX;j+=ps)
				sieve[j] = 1;
		}
}
void factorsieve()
{
        for(int i=1;i<MAXX;i++)
                    a[i]=i;
 
        for(int i=2;i*i<MAXX;i++)
        {
           if(a[i] == i)
           {
                for(int j=i*i;j<MAXX;j+=i)
                {
                      if(a[j] == j)
                      {
                             a[j]=i;
                      }
                 }
           }
 
        }
}
int main()
{_
	factorsieve();

	for(int i=0;i<n;i++){

		if(N == 1){

			continue;
		}

		int pro = 1;
        int temp=a[N];

        N/=a[N];
        int count=1;
        
        while(a[N]!=1)
        {
            //cout<<"temp="<<temp<<" a[n]="<<a[n]<<endl;
            if(temp == a[N])
                count++;
            else
            {
            	pro *= 2; 
                temp = a[N];                
                count=1;
            }
            N/=a[N];
        }
		
        pro *= 2;
        
        ans += pro;

	}

	cout<<ans<<endl;

}
