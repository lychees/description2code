//no need of graphs
#include<bits/stdc++.h>
using namespace std;
int visited[1000002];
int main()
{
    int t,e,x,y,i,cou=0;
    scanf("%d",&t);
    while(t--)
    {
        cou=0;
        for(i=0;i<=1000001;i++)
            visited[i]=0;
        scanf("%d",&e);
        for(i=0;i<e;i++)
        {
            scanf("%d%d",&x,&y);
            if(visited[x]==0)
            {
                visited[x]=1;
                cou++;
            }
            if(visited[y]==0)
            {
                visited[y]=1;
                cou++;
            }
        }
        printf("%d\n",cou);
    }
    return 0;
}
