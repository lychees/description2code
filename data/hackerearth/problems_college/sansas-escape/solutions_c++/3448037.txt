#include <cstring>
#include <iostream>
#include <cstdlib>
#include <math.h>
#include <vector>
using namespace std;
int main(){
int n;
cin>>n;
int a[n];
for(int i=0;i<n;i++)
{
    cin>>a[i];
}
int q;
cin>>q;
int k[n];
k[0]=a[0];
for(int i=1;i<n;i++){
    k[i]=k[i-1]^a[i];
    //cout<<k[i]<<endl;
}
for(int i=0;i<q;i++)
{
    int a,b,h;
    cin>>a>>b;
    if(a==0)
    h=k[b];
    else
     h=k[b]^k[a-1];
    cout<<h<<endl;
}
}
