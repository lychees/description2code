#include <iostream>
#include <cstdio>
#include <algorithm>
#include <queue>
#include <cmath>

using namespace std;

vector<int> primeFactors(int n, vector<int> v)
{
    while (n%2 == 0)
    {
        v.push_back(2);
        n = n/2;
    }

    for (int i = 3; i <= sqrt(n); i = i+2)
    {
        while (n%i == 0)
        {
            //printf("%d ", i);
            v.push_back(i);
            n = n/i;
        }
    }

    if (n > 2) {
        v.push_back(n);
    }
    return v;
}

int main() {
	int t;
	cin >> t;
	while (t--) {
		int a, b;
		priority_queue<int, vector<int>, greater<int> > pq;
		cin >> a >> b;
		for (int i = 0; i < a; i++) {
			pq.push(1);
		}
		vector<int> v;
		v = primeFactors(b, v);
		reverse(v.begin(), v.end());
		for (int i = 0; i < v.size(); i++) {
			int temp = pq.top();
			temp = temp * v[i];
			pq.pop();
			pq.push(temp);
		}
		int ctr = 0;
		while (!pq.empty()) {
			ctr += pq.top();
			pq.pop();
		}
		cout << ctr << endl;
	}
	return 0;
}
