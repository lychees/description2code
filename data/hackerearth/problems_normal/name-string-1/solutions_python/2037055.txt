for _ in range(int(raw_input())):
    r,k,v=map(int,raw_input().split())
    if r>=k:
        a=r
        if k>=v:
            b=k
        else:
            b=v
    else:
        a=k
        if r>=v:
            b=r
        else:
            b=v
    if b>a:
        b,a=a,b
    if b==a:
        print(2*b);
    else:
        print(2*b+1);
