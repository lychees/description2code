import string
smaAlp = list(string.ascii_lowercase)

testCases = raw_input()

while testCases > 0 :
    mult = 1
    palStr = raw_input()
    if str(palStr) == str(palStr)[::-1] :
    	print "Palindrome"
    else :
    	for i in palStr :
    		mult *= int(smaAlp.index(i)) +1
    	print mult
    	
    testCases = int(testCases) - 1
