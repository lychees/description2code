#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int T,N,C;
    cin>>T;
    while(T--) {
    	cin>>N>>C;
    	vector<vector<int> > arr(N,vector<int>(N,-1));
    	for(int i=0;i<N;i++) {
    		for(int j=0;j<N;j++) {
    			cin>>arr[i][j];
    		}
    	}
    	for(int i=0;i<N;i++) {
    		for(int j=0;j<N;j++) {
    			int num=0;
    			if(i>0)
    				num=max(num,arr[i-1][j]);
    			if(j>0)
    				num=max(num,arr[i][j-1]);
    			arr[i][j]+=num;
    		}
    	}
    	int ans=arr[N-1][N-1]-C;
    	if(ans<0) ans=-1;
    	cout<<ans<<endl;
    }
    return 0;
}
