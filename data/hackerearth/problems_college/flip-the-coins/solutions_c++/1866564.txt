#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,q,a[104],i;
    cin>>n>>q;
    memset(a,0,sizeof(a));
    int ans=0,l,r,c;
    while(q--)
    {
    	cin>>c>>l>>r;
    	ans=0;
    	if(!c)
    	{
    		for(i=l;i<=r;i++)
    		{
    			if(!a[i]) a[i]=1;
    			else a[i]=0;
    		}
    	}
    	else
    	{
    		for(i=l;i<=r;i++)
    		{
    			if(a[i]) ans++;
    		}
    		cout<<ans<<"\n";
    	}
    }
    return 0;
}
