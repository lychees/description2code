#include <iostream>
#include <map>

using namespace std;

map<string, int> m;

int main()
{
    int t;
    cin >> t;
    
    while (t--) {
    	string s;
    	cin >> s;
    	m[s]++;
    }
    
    for (auto i = m.begin(); i != m.end(); i++) {
    	cout << (*i).first << " " << (*i).second << "\n";
    }
    
    return 0;
}
