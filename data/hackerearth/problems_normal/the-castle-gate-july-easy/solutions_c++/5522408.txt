#include <iostream>
using namespace std;

int main()
{
    int t,n,pair=0;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	cin>>n;
    	pair=0;
    	for(int j=1;j<n;j++)
    	{
    		for(int k=j+1;k<=n;k++)
    		{
    			if((j^k) <=n)
    			{
    				//cout<<"sa "<<n<<"  "<<(j^k)<<endl;
    				pair++;
    			}
    			else
    				continue;
    		}
    	}
    	cout<<pair<<endl;
    }
    return 0;
}
