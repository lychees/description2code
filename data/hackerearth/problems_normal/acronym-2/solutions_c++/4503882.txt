#include <iostream>
#include <ctype.h>
#include <string.h>

using namespace std;

int main()
{
    int k,n,i,temp=0;
    char dislike[10][100],sent[100][100];
    bool ans=1,first=0;
    cin>>k;
    for (i=0;i<k;i++)
        cin>>dislike[i];
    cin>>n;
    for (i=0;i<n;i++)
    {
        cin>>sent[i];
    }
    for (i=0;i<n;i++)
    {   temp=0;
        for (temp=0;temp<k;temp++)
        {
            if(!strcmp(sent[i],dislike[temp]))
            {
                ans = 1;
                break;
            }
            ans = 0;

        }
        if (ans == 0)
        {
            if(first !=0)
            {
                cout<<".";
            }
            cout <<(char)toupper(sent[i][0]);
            first=1;
            ans = 1;
        }
    }
    return 0;
}
