def isprime(n):
	for i in range(2, int(pow(n,0.5)) + 1):
		if n % i == 0:
			return False
	return True

p = dict()
def factorization(n):
	if isprime(n):
		if n in p:
			p[n] += 1
		else:
			p[n] = 1
	else:
		for i in range(2, int(pow(n, 0.5)) + 1):
			if n % i == 0:
				factorization(i)
				factorization(n/i)
				break
		
t = int(raw_input())
for i in range(t):
	H = int(raw_input())
	p = dict()
	factorization(H)
	print 'Printf{}' if reduce(lambda x,y: x^y, p.values()) else 'Pappu Bhujia'
