#include <bits/stdc++.h>
using namespace std;

#define Lim 1000005
long long etf[1000010];

void cal_etf()
{
    for(int i = 1 ;i <= Lim ; i++ )
    	etf[i]=i;
    for(int i=2 ; i<= Lim ; i++ )
        if ( etf[i] == i)
            for(int j = 2*i ; j <= Lim ; j += i)
            	etf[j] -=  etf[j]/ i ;
    etf[1]=1;
    for(int i=2 ; i<= Lim ; i++)
    {	
        if ( etf[i]==i)//if that number is prime
    		etf[i] = i-1;
    }
    for(int i=2;i<=Lim;i++) {
    	etf[i]+=etf[i-1];
    }
}

int main()
{
    cal_etf();
    int t;cin>>t;while(t--) {
    	long long n;cin>>n;
    	long long ans = (((n*(n-1))/2) + 1) - etf[n];
    	cout<<ans<<"\n";
    }
}
