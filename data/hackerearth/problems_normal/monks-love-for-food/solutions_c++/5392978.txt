int pop(int stack[], int n);
int push(int stack[], int p);

int pop(int stack[], int n)
{
	int popper;
	if(top == -1)
	{
		return (-9);
	}
	if(top >= 0 && top < n)
	{
		popper = stack[top];
		top = top - 1;
		return (popper);
	}
}

int push(int stack[], int p)
{
	top = top + 1;
	stack[top] = p;
}
