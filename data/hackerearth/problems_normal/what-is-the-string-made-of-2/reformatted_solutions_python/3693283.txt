def char2num(s):
    return {
        '0': 6,
        '1': 2,
        '2': 5,
        '3': 5,
        '4': 4,
        '5': 5,
        '6': 6,
        '7': 3,
        '8': 7,
        '9': 6}[s]


def str2int(s):
    return reduce(lambda x, y: x+y, map(char2num, s))


st = raw_input()
print str2int(st)
