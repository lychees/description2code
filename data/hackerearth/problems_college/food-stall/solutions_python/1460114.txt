t=input()
if t>=1 and t<=10:
  for i in range(0,t):
    s=raw_input()
    s1=s.split()
    n=int(s1[0])
    d=int(s1[1])
    A=[]
    c=0
    for j in range(0,n):
      n2=input()
      A.append(n2)
    for j in range(0,n-1):
      for j2 in range(j+1,n):
        if A[j]>=0 and A[j2]>=0:
          if abs(A[j]-A[j2])<=d:
            c+=1
            A[j]=-1
            A[j2]=-1
    print(c)
