#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int n;
    	cin>>n;
    	long long int a[n],i,j;
    	for(i=0;i<n;++i)
    	{
    		cin>>a[i];
    	}
    	long long int sum1[n],sum2[n];
    	for(i=0,j=n-1;i<n;++i,--j)
    	{
    		if(i==0)
    		{
    			sum1[i]=a[i];
    			sum2[j]=a[j];
    		}
    		else
    		{sum1[i]=sum1[i-1]+a[i];
    		sum2[j]=sum2[j+1]+a[j];}
    	}
    	//for(int i=0;i<n;++i)
    	//{cout<<sum1[i]<<" "<<sum2[i];cout<<"\n";}
    	int flag=0;
    	for(i=0;i<n;++i)
    	{
    		if(sum1[i]==sum2[i])
    		flag=1;
    	}
    	if(flag==0)
    	cout<<"NO\n";
    	else cout<<"YES\n";
    }
    return 0;
}
