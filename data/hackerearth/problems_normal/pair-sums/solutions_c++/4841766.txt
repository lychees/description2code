#include<bits/stdc++.h>
using namespace std;
long int a[1000005];
int main()
{
    long int n,k;
    scanf("%ld%ld",&n,&k);
bool sum=false;
    bool m[2000005]={0};
    for(int i=0;i<n;i++)
    {
    	scanf("%ld",&a[i]);
    	int temp=k-a[i];
    	if(temp>=0&&m[temp]==1)
    	{
    		sum=true;
		}
	m[a[i]]=1;
	
    	
	}

if(sum==true)
{
	printf("YES\n");
}
else
{
	printf("NO\n");
}

    return 0;
}
