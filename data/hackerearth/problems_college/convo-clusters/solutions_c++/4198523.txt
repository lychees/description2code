#include <bits/stdc++.h>
using namespace std;

int main()
{
    char a[105];
    while(gets(a)){
    	int n = strlen(a);
    	
    	int f1=1,f2=1;
    	for(int i=0;i<n-1;i++){
    		if((a[i]>'z' || a[i]<'a') && a[i]!=' '){
    			f1=1;f2=1;
    			break;
    		}
    		if((a[i]!='a' && a[i]!='e' && a[i]!='i' && a[i]!='o' && a[i]!='u') && (a[i+1]!='a' && a[i+1]!='e' && a[i+1]!='i' && a[i+1]!='o' && a[i+1]!='u')){
    			f1=0;
    		}
    		else if((a[i]=='a' || a[i]=='e' || a[i]=='i' || a[i]=='o' || a[i]=='u')&&(a[i+1]=='a' || a[i+1]=='e' || a[i+1]=='i' || a[i+1]=='o' || a[i+1]=='u')){
    			f2=0;
    		}
    	}
    	if(f1==0 && f2==0){
    		cout<<"YES"<<endl;
    	}
    	else{
    		cout<<"NO"<<endl;
    	}
    }
    return 0;
}
