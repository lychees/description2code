#include <iostream>
using namespace std;
long long int gcd(long long int a,long long int b)
{
	if(a==0)
	return b;
	return(gcd(b%a,a));
}	

int main()
{
    int d;
    long long int x,y,k,i,num,f,h;
    cin>>d;
    while(d--)
    {
        cin>>x>>y>>k;
        num=0;
        f=0;
        h=gcd(x,y);
        for(i=1;i<=h;i++)
        {
        	if(h%i==0)
        	num++;
        	if(num==k)
        	{
        		f=1;
        		cout<<h/i<<endl;
        		break;
        	}
        }
    	if(f==0)
    	cout<<"No crymeth today"<<endl;
    }
    return 0;
}
