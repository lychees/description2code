#include<iostream>
using namespace std;
int a[100000],b[100000],c[300000];

void merge(int,int,int);
void merge_sort(int low,int high)
{
 int mid;
 if(low<high)
 {
  mid=(low+high)/2;
  merge_sort(low,mid);
  merge_sort(mid+1,high);
  merge(low,mid,high);
 }
}
void merge(int low,int mid,int high)
{
 int h,i,j,d[100000],k;
 h=low;
 i=low;
 j=mid+1;

 while((h<=mid)&&(j<=high))
 {
  if(c[h]>=c[j])
  {
   d[i]=c[h];
   h++;
  }
  else
  {
   d[i]=c[j];
   j++;
  }
  i++;
 }
 if(h>mid)
 {
  for(k=j;k<=high;k++)
  {
   d[i]=c[k];
   i++;
  }
 }
 else
 {
  for(k=h;k<=mid;k++)
  {
   d[i]=c[k];
   i++;
  }
 }
 for(k=low;k<=high;k++) c[k]=d[k];
}


int main()
{
    int t,n,m,i;
    cin>>t;
    while(t>0)
    {
        cin>>n;
        cin>>m;
        for(i=0;i<n;i++)
        {
            cin>>a[i];
            c[i]=a[i];
        }
        for(i=0;i<m;i++)
        {
            cin>>b[i];
            c[i+n]=b[i];
        }
  merge_sort(0,m+n-1);
 for(i=0;i<m+n;i++)
 cout<<c[i]<<" ";
 cout<<endl;
        t--;
    }
    return 0;
}
