#include <iostream>
using namespace std;

int main()
{
    int i,n,j;
    long long int a,x;
    a=0LL;
    cin>>n;
    for(i=0;i<n;i++)
    for(j=0;j<n;j++)
    {
    	cin>>x;
    	if(i==j)
    		a+=x;
    	
    	if(i+j==n-1)
    		a-=x;
    }
    if(a<0)
    a=-a;
    
    cout<<a<<endl;
    return 0;
}
