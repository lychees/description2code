#include<iostream>
using namespace std;
long long int fib[81];
long long int fibb(int n)
{
	fib[0] = 1;
	fib[1] = 1;
	for(int i = 2; i <= n; i++)
		fib[i] = fib[i-1]+fib[i-2];
	return fib[n];
}
int main()
{
	long long int n,m,x;
	long long int sum1 = 0, sum2 = 0;
	cin >> n >> m;
	for(long long int i = 0; i < m; i++) {
		cin >> x;
		sum1 += x;
	}
	fibb(n);
	for(long long int i = 0; i < n; i++)
		sum2 += fib[i];
	cout << sum2-sum1 << endl;
	return 0;
}
