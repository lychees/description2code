def solve(n):
    if n == 1:
        print 0, 1
        return
    ln = (n-1).bit_length()
    ans = int('1'*ln, 2)
    cnt = 0
    l = ln-1
    num = int('1' + '0'*l, 2)+1
    k = abs(num - n) + 1
    cnt = 2*k
    print ans, cnt
    return


for i in xrange(input()):
    n = int(raw_input())
    solve(n)
