#include <iostream>
#include<algorithm>
#include<stdio.h>
using namespace std;
long long compute(int arr[],int n)
{
	int i;
	for(i=0;i<n;i++)
	{
		if(arr[i]>0)
			break;
	}
	if(i==n)
	{
		sort(arr,arr+n);
		return arr[n-1];
	}
	else
	{
		long long sum=0,high=-9999;
		for(;i<n;i++)
		{
			sum+=arr[i];
			if(sum<0)
				sum=0;
			if(high<sum)
				high=sum;
		}
		return high;
	}
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n;
    	scanf("%d",&n);
    	int arr[n],i;
    	for(i=0;i<n;i++)
    		scanf("%d",&arr[i]);
    	long long ans=compute(arr,n);
    	printf("%lld\n",ans);
    }
    return 0;
}
