#include<iostream>
#define M 1000000007
#define ll long long
using namespace std;
ll int pow(int a,int n)
{
    if(n==0)
        return 1;
    ll int x=pow(a,n/2);
    x=(x*x)%M;
    if(n&1)
        x=(x*a)%M;
    return x;
}
int main()
{
    int t,n;
    ll int ans;
    cin>>t;
    while(t--)
    {
        cin>>n;
        ans=pow(3,n);
        ans=(ans+3)%M;
        ans=(ans-(3*pow(2,n))%M)%M;
        if(ans<0)
            ans+=M;
        cout<<ans<<endl;
    }
    return 0;
}
