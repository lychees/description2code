for _ in xrange(input()):
    s = raw_input()
    a = "aeiouy"
    t, x, y = 0, 0, 0
    flag = 0
    for q in s:
        if q in a:
            x += 1
            t = 0
        else:
            y += 1
            t += 1
        if t == 3:
            flag = 1
            break
    if y > x or flag == 1:
        print "hard"
    else:
        print "easy"
