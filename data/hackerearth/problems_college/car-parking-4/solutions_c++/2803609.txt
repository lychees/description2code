#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t,i;
    scanf("%d",&t);

    string s;
    while(t--)
    {
        cin>>s;
        int len = s.length();
        int ans=0;


        for(i=0;i<len;)
        {
            if(s[i]=='-')
            {
                if(i>0 && s[i-1]=='S')
                {
                    i++;
                }
                else if(i+1<len && s[i+1]=='B')
                {
                    i++;
                }
                else if(i+1<len && s[i+1]=='S')
                    i++;
                else if(i+2<len && s[i+2]=='B')
                    i++;
                else
                    ans++,i++;
            }
            else if(s[i]=='B')
            {
                i++;
            }
            else if(s[i]=='S')
            {
                i++;
            }
            else if(s[i]=='D')
            {
                i++;
            }
            //printf("%d ",ans);
        }
        printf("%d\n",ans);

    }
}
