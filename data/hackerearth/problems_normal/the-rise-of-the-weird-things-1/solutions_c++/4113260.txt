#include <iostream>
using namespace std;


void insertion(int ar[],int n)
{
	int i,j;
	
for(i=0;i<n;i++)
{
int temp=ar[i];
j=i;

while(temp<ar[j-1] && j>0)
{
	ar[j]=ar[j-1];
	j=j-1;
}
ar[j]=temp;
}
	
}



int main()
{

  cin>>n;
  int ar[n];
  int i,j;
  
  for(i=0;i<n;i++)

  	cout<<ar[i]<<" ";
