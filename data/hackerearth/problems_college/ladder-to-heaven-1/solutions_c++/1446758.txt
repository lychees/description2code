#include<iostream>
using namespace std;
int main()
{
    long long t,n,temp,sum,temps;
    cin>>t;
    for(;t>0;t--)
    {
                 sum=0;
                 temp=0;
                 temps=0;
                 cin>>n;
                 while(temps<=n)
                 {
                                temps=(3*temp)+(4*sum)+2;
                                if(temps<=n)
                                {
                                            sum=sum+temp;
                                            temp=temps;
                                            }
                                }
                                cout<<(sum+temp)<<"\n";
                 }
    return 0;
    }
