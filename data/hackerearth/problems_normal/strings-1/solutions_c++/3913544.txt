#include <stdio.h>
#include<iostream>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    	string s1,s2;
    	cin>>s1>>s2;
    	if(s1.compare(s2)==0)
    		printf("YES\n");
    	else if((s1.compare("2")==0)&&(s2.compare("4")==0))
    		printf("YES\n");
    	else if((s2.compare("2")==0)&&(s1.compare("4")==0))
    		printf("YES\n");
    	else
    		printf("NO\n");
    }
    return 0;
}
