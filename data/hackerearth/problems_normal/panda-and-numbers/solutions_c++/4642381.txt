#include <bits/stdc++.h>

using namespace std;
 
#define MAX 1000005
#define INF 2ll * (1e18)
#define MOD 1000000007
#define pb push_back
#define mp make_pair
#define ff first
#define ss second
#define ll long long

bool dp[MAX], visited[MAX];

inline int len(int num){
	int l = 0;
	while(num)
		l++, num /= 10;
	return l;
}

bool solve(int num){
	if(num<=0)
		return false;
	if(visited[num])
		return dp[num];
	
	int l = len(num), pow = 10, last = 1;
	bool ans = false;

	for(int i=0 ; i<l ; i++, pow *= 10, last *= 10){
		int digit = (num % pow) / last;
		if(digit == 0)
			continue;
		ans |= solve(num - digit * digit);
	}

	dp[num] = ans;
	visited[num] = true;

	return ans;
}

int main(){
	ios_base::sync_with_stdio(false);
	int t;	
	cin>>t;

	for(int i=1 ; ; i++){
		int temp = 1;
		for(int j=1 ; j<=i ; j++)
			temp = temp * i;
		if(temp>=MAX)
			break;
		visited[temp] = dp[temp] = true;
	}

	for(int i=MAX-1 ; i>0 ; i--){
		if(!visited[i])
			solve(i);
	}

	while(t--){
		int n;
		cin>>n;

		if(dp[n])
			cout<<"Yes\n";
		else
			cout<<"No\n";
	}

	return 0;
}
