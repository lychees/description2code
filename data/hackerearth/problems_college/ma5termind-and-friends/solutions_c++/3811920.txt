#include<bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
 	int t;
 	scanf("%d",&t);
 	while(t--)
 	{
		int n,m,i;
		scanf("%d%d",&n,&m);
		ll a[m+1],b[m+1],c[1+m];
		for(i=1;i<=m;i++)
		{
			scanf("%lld%lld%lld",&a[i],&b[i],&c[i]);
		}
		int q;
		ll arr[100005];
		ll sol[n+5];
		for(i=0;i<=n+2;i++){
			arr[i]=0;	sol[i]=0;
		}
		for(i=0;i<=m+1;i++)
			arr[i]=0;
		scanf("%d",&q);
		while(q--)
		{
			int x,y;
			scanf("%d%d",&x,&y);
			arr[x]++;arr[y+1]--;
		}
		for(i=1;i<=m;i++)
		{
			arr[i]+=arr[i-1];
			//cout<<arr[i]<<"	";
		}
		for(i=1;i<=m;i++)
			c[i]*=arr[i];
		for(i=1;i<=m;i++)
		{
			sol[a[i]]+=c[i];
			sol[b[i]+1]-=c[i];
		}
		for(i=1;i<=n;i++)
		{
			sol[i]+=sol[i-1];
			cout<<sol[i]<<" ";
		}
		cout<<"\n";
 	}
	return 0;
}
