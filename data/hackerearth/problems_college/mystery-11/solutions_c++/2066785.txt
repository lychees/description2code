#include <iostream>
using namespace std;

int main()
{
    int t,x;
    cin>>t;
    string n;
    long long int ans;
    int arr[10]={6,2,5,5,4,5,6,3,7,6};
    while(t--)
    {
    	ans=0;
    	cin>>n;
    	for(int i=0;i<n.length();i++)
    	{
    		x=n[i]-'0';
    		ans+=arr[x];
    	}	
    	
    	cout<<ans<<endl;
    }
}
