#include<iostream>
#include<vector>
#include<algorithm>
#define mag 1000000
using namespace std;
int main(){
bool a[mag+1];
for(long int i=1;i<=mag;i++)
    a[i]=false;
for(long int i=2;i<=mag;i++){
    for(long int j=i*i;j<=mag;j+=i){
        a[j]=true;
    }
}
a[1]=true;
long int sum,t,c,b;
cin>>t;
while(t--){
    cin>>c>>b;
    sum=0;
    for(long int i=c;i<=b;i++){
        if(a[i]==false)
            sum+=i;
    }
    cout<<sum<<endl;
}



}
