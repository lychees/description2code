#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int T, N;
    cin >> T;
    while(T--)
    {
    	cin >> N;
    	int X[N];
    	  for(int i=0;i<N;i++)
         	  cin >> X[i];
         	  
        sort(X,X+N);

         int max, count, temp;
         
         max=1;
         count=1;
         
         for(int j=1;j<N;j++)
            if(X[j-1] == X[j])
              count++;
             else
             {
             	if(max < count) max =count;
             	count =1;
             }
             
         if(max < count) max =count;
       cout << max << endl; 
    }
    return 0;
}
