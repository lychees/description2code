#include<bits/stdc++.h>
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        long long int x,y,count=0;
        scanf("%lld %lld",&x,&y);
        if(x==y)
            printf("-1\n");
        else
        {
            x=x-y;
            for(long long int i=1;i<=sqrt(x);i++)
            {
                if(x%i==0)
                {
                    if(i>y)
                        count++;
                    if(i!=x/i&&(x/i)>y)
                        count++;
                }
            }
            printf("%lld\n",count);
        }
    }
    return 0;
}
