#include<stdio.h>
char a[100];
int check(int i,int j)
{
int count=0;
    while(a[i]==a[j]&&i<=j)
{
i++;
j--;
}
if(a[i]!=a[j]&&i<=j)
{
    count+=1;
int a=check(i,j-1);
int b=check(i+1,j);
count+=(a>b?b:a);
}
return count;
}

int main()
{
int t;
scanf("%d",&t);
while(t--)
{
int n;
scanf("%d",&n);
scanf("%s",a);
int m=check(0,n-1);
printf("%d\n",m);
}
return 0;
}
