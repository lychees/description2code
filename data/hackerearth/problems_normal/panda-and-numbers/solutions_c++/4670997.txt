#define MAX 1000006
ll modpow(ll a,ll n,ll temp){ll res=1,y=a;while(n>0){if(n&1)res=(res*y)%temp;y=(y*y)%temp;n/=2;}return res%temp;} 
set<int> populate;
int mem[1000006];
int findit(int n)
{
	if(n<1)
		return 0;
	if(mem[n]!=-1)
	{
		return mem[n];
	}
	if(populate.find(n)!=populate.end())
		return 1;
	int calc,n1=n,ans=0,val;
	while(n!=0)
	{
		calc=n%10;
		//trace1(calc);
		if(calc!=0)
		{
			val=n1-calc*calc;
			ans^=findit(val);
			if(ans)
				break;
		}
		n/=10;
	}
	mem[n]=ans;
	return mem[n];
}
int main()
{
	int i,t,n,ans,val;
	FOR(i,0,MAX)
		mem[i]=-1;
	FOR(i,1,8)
	{
		val=modpow(i,i,mod);
		populate.insert(val);
	}
	si(t);
	assert(t>=1 && t<=100000);
	while(t--)
	{
		si(n);
		assert(n>=1 && n<=1000000);
		ans=findit(n);
		if(ans)
			printf("Yes\n");
		else
			printf("No\n");
	}
	return 0;
}
