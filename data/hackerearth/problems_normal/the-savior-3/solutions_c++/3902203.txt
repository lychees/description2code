#include <iostream>
using namespace std;

int main()
{
	int tc=0,n=0,count;
	int* a;
	cin>>tc;
	{
		while(tc--)
		{
			count=0;
			cin>>n;
			a=new int[n];
			for(int i=0;i<n;i++)
				cin>>a[i];
			for(int i=0;i<n;i++)
			{
				for (int j=i+1;j<n;j++)
				{
					if(a[i]!=a[j] && (a[i]+a[j])%2==0)
						count++;
				}
			}
			cout<<count<<"\n";
		}
	}
    return 0;
}
