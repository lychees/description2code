#include <bits/stdc++.h>

using namespace std;

#define ll long long int
#define s(x) scanf("%d", &x)
#define sl(x) scanf("%lld", &x)
#define pb push_back
#define mp make_pair
#define mod 1000000007

int main()
{
    int n, i, sum, j, k;
    cin >> n;
    int a[n];
    for (i = 0; i < n; i++)
        cin >> a[i];
    sort(a, a+n);
    cin >> sum;
    for (i = 0; i < n; i++) {
        for (j = i+1; j < n; j++) {
            for (k = j+1; k < n; k++) {
                if (a[i]+a[j]+a[k] == sum) {
                    cout << "True" << endl;
                    return 0;
                }
            }
        }
    }
    cout << "False" << endl;
    return 0;
}
