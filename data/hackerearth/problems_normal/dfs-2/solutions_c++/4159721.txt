#include <iostream>
#include <bits/stdc++.h>
using namespace std;
vector <int> v;
int main()
{
    int q,i,n,l,r,ans;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>q;
    	v.push_back(q);
    }
    cin>>q;
    sort(v.begin(),v.end());
    while(q--)
    {
    	cin>>l>>r;
    	ans=upper_bound(v.begin(),v.end(),r)-v.begin();
    	ans-=lower_bound(v.begin(),v.end(),l)-v.begin();
    	cout<<ans<<endl;
    }
    return 0;
}
