bool exist[51][10001];
double pr[51][10001];

double dp(int n, int m, int k)
{
    //cout << m << " " << k << endl;
    if (m == 1) {
        if (k >= 1 && k <= n)
            return (double)1/n;
        else
            return 0.0;
    }
    if (exist[m][k]) {
        return pr[m][k];
    }
    double p = 0;
    for(int i = 1; i <= n; i++) {
        //cout << "hi";
        p += (double)1/n * dp(n, m - 1, k - i);
    }
    exist[m][k]  = 1;
    pr[m][k] = p;
    return p;
}

int main()
{
    int t;
    cin >> t;
    while (t--) {
        for (int i = 0; i < 51; i++)
            for (int j = 0; j < 10001; j++)
                pr[i][j] = 0, exist[i][j] = 0;
        int n, m, k;
        cin >> m >> n >> k;
        if (k > n*m || k < m) {
            printf("0.000 0\n");
        } else {
        double p = dp(n, m, k);
        int exp = 0;
        while (p < 1.0) {
            p *= 10;
            exp++;
        }
        printf("%.3lf %d\n", p, exp);
        }
    }
}
