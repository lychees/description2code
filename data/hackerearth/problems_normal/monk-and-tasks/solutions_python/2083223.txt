# Due to array lookup in python instead of dict lookup for python global vars.
def bitcount(n):
    
    count = 0
    while n > 0:
        count = count + 1
        n = n & (n-1)
    return count 


def make_dict(l):
	d = {}
	for i in l:
		cnt = bitcount(i)
		if d.get(cnt, 0):
			d[cnt].append(i)
		else:
			d[cnt] = [i]
	return d
