#include<bits/stdc++.h>
using namespace std;
int main() {
	int n;
	cin>>n;
	vector<int> v(n);
	for(int i=0;i<n;i++)
		scanf("%d",&v[i]);
	sort(v.begin(),v.end());
	int ans=1e4+3;
	for(int i=0;i<n;i++) {
		vector<int>::iterator it=upper_bound(v.begin(),v.end(),3*v[i]);
		int val=v.size()-(it-v.begin());
		ans=min(ans,i+val);
	}
	cout<<ans<<endl;
	return 0;
}
