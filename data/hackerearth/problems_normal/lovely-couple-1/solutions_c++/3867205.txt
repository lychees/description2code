#include<bits/stdc++.h>
using namespace std;
int arr[1000001]={0},i,j;
void sieve()
{
    /*for(i=1;i<=1000000;i++)
        a[i]=0;*/
    for(i=2;i<=1000000;i++)
    {
        if(arr[i]==0)
        {
            for(j=i*2;j<=1000000;j+=i)
                arr[j]++;
        }
    }
}
int gcd(int a,int b)
{
    if(a==0)
        return b;
    return (gcd(b%a,a));
}
int main()
{
    int t,a,b,res,g,l;
    sieve();
    cin>>t;
    while(t--)
    {
        cin>>a>>b;
        g=gcd(a,b);
        l=(a*b)/g;
        if(arr[l]>1&&arr[arr[l]]==0)
            cout<<"Yes"<<endl;
        else
            cout<<"No"<<endl;
    }
    return 0;
}
