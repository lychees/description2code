#include <iostream>
using namespace std;

int main()
{
	int n;
    cin>>n;
    int b[n];
    int a[n];
    for (int i=1;i<=n;i++)
    {
    	cin>>b[i];
    }
    a[1]=b[1];
    for(int i=2;i<=n;i++)
    {
    	a[i]=0;
    }
    for (int j=2;j<=n;j++)
    {
    	a[j]=(b[j]*j);
    	for(int k=1;k<j;k++)
    	{
    		a[j]-=a[k];
    	}
    }
    for(int i=1;i<=n;i++)
    {
    	cout<<a[i]<<" ";
    }
}
