#include <bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;
ll int dp[1011][1011];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int i,j;
	dp[0][0]=1;
	for(i=1;i<1011;i++)
	dp[0][i]=1;
	for(i=1;i<1011;i++)
	dp[i][0]=1;
	for(i=1;i<1011;i++)
	{
		for(j=1;j<1011;j++)
		{
			dp[i][j]=(dp[i-1][j]+dp[i][j-1]+dp[i-1][j-1])%mod;
		}
	}
	ll int test;
	cin>>test;
	while(test--)
	{
		ll int a,b,c,d,x,y;
		cin>>a>>b>>c>>d;
		x=abs(c-a);y=abs(d-b);
		//~ cout<<x<<" "<<y<<endl;
		cout<<dp[x][y]<<endl;
	}
	return 0;
}
