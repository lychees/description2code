#include <bits/stdc++.h>
using namespace std;
#define max1 1000001
int a[max1]={0};
int b[max1]={0};
int main() {
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;int x=0; long long counter=0;
		for(int i=0;i<n;i++)
		{cin>>x;
			if(x>0)
			{if(a[x])
			counter+=a[x];
			a[x]++;}
			else
			{
				{if(b[-1*x])
			counter+=b[-1*x];
			b[-1*x]++;}
			}
		
		}
		cout<<counter+n<<endl;
		for(int i=0;i<max1;i++) a[i]=0;
		for(int i=0;i<max1;i++) b[i]={0}; 
		
	}
	return 0;
}
