#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n; cin>>n;
	int arr[n];
	for(int i=0;i<n;i++)cin>>arr[i];
	int index,mx,sum=0;
	int l=n,temp;
	for(int i=1;i<n-1;i++)
	{
		mx = 0;
		for(int j=1;j<l-1;j++)
		{
			temp = arr[j-1] * arr[j+1];
			if(temp>mx)
			{
				mx=temp;
				index = j;
			}
		}
		sum += mx;
		l--;
		for(int j=index;j<n;j++)
		{
			arr[j] = arr[j+1];
		}
	}
	cout<<sum<<endl;
}
