#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
    	int i,n,a[101],k,j,l;
    	cin>>n;
    	cin>>k;
    	for(i=0;i<n;i++)cin>>a[i];
    	sort(a,a+n);
    	int min_cast=10000000;
    	for(i=0;i<=n-k;i++)
    	{
    		for(j=0;j<k;j++)
    		{
    		 int c=0;
    		 for(l=0;l<k;l++)
    		 {
    		 	if(l==j)continue;
    		 	if(a[i+j]<a[i+l])
    		 	 c+=5*(a[i+l]-a[i+j]);
    		 	 else
    		 	 c+=3*(a[i+j]-a[i+l]);
    		 }
    		 if(c<min_cast)min_cast=c;
    		
    		}
    	}
    	cout<<min_cast<<endl;
    }
    return 0;
}
