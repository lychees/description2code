#include<bits/stdc++.h>
using namespace std;
int main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int arr[10004];
		for(int i=0;i<n;i++)
			cin>>arr[i];
		int dp[n+1];
		for(int i=0;i<n;i++)
			dp[i]=arr[i];
		dp[1]=max(dp[1],dp[0]);
		for(int i=2;i<n;i++)
		{
			dp[i]=max(dp[i]+dp[i-2],dp[i-1]);
		}
		cout<<dp[n-1]<<"\n";
	}
	return 0;
}
