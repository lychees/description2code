#include<iostream>
#include<algorithm>
#include<string.h>
#include<math.h>
#include<vector>
using namespace std;
#define lli long long int 
string ss[55];
lli l[55];
char f(lli x,int i){
	if(i==0)return ss[i][x];
	if(x>=2*l[i-1])return ss[i][x-2*l[i-1]];
	if(x<l[i-1])return f(x,i-1);
	return f(2*l[i-1]-x-1,i-1);
}
int main() {
	int t;
	cin>>t;
	while(t--){
		int n,m;
		cin>>n>>m;
		for(int i=0;i<n;i++){
			cin>>ss[i];
		}
		lli x;
		l[0]=ss[0].size();
		for(int i=1;i<n;i++){
			l[i]=2*l[i-1]+ss[i].size();
		}
		for(int i=0;i<m;i++){
			cin>>x;
			cout<<f(x,n-1);
		}
		cout<<endl;
	}
	return 0;
}
