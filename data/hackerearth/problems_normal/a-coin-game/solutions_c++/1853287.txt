#include <iostream>
using namespace std;
#include<cstring>

int main()
{
	int t;
	cin>>t;
	int a[1000001];
   memset(a,-1,sizeof(a));
   int p=1;
   for(int i=1;i<=1000000&&i+p<=1000000;i++)
   {
   	if(a[i]==-1)
   	{
   		a[i]=i+p;
   		a[i+p]=i;
   		p++;
   }
   		
   }
   int x,y;
   for(int i=0;i<t;i++)
   {
   	cin>>x>>y;
   	if(x==0&&y==0)
   	cout<<"Don't Play\n";
   	else if(a[x]==y)
   	cout<<"Don't Play\n";
   	else
   	cout<<"Play\n";
   }
    return 0;
}
