#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	int ans[t];
	int n,k,yes=0,no=0;
	for(int i=0;i<t;i++)
	{
		yes=0;
		no=0;
		cin>>n>>k;
		int a[n];
		for(int j=0;j<n;j++)
		{
			cin>>a[j];
			if(a[j]<=0)
			no++;
			else
			yes++;
		}
		//cout<<yes<<"...."<<no<<endl;
		if(no<k)
		ans[i]=1;
		else
		ans[i]=0;
	}
	for(int i=0;i<t;i++)
	{
		if(ans[i]==1)
		cout<<"YES"<<endl;
		else if(ans[i]==0)
		cout<<"NO"<<endl;
	}
}
