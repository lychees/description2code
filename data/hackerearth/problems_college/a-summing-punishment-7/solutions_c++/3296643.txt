#include<iostream>
using namespace std;
int main()
{
    int n,*a,q,l,r,i,sum;
    cin>>n;
    a=new int [n];
    for(i=1;i<=n;i++)
        cin>>a[i];
    cin>>q;
    while(q--)
    {
        cin>>l>>r;
        if(r>n)
            while(r>n)
            r--;
        sum=0;
        for(i=l;i<=r;i++)

            sum+=a[i];
        cout<<sum<<"\n";
    }
    return 0;
}
