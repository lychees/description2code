#include <bits/stdc++.h>
using namespace std;
int main(int argc, char const *argv[]) {
    int test,d,w;
    int i,j,k;
    int fare;
    scanf("%d",&test);
    while(test--)
    {
        scanf("%d%d",&d,&w);
        if(d>60)
        {
            printf("NA\n");
            continue;
        }
        fare=0;
        if(d>0 && d<2)
        {
            fare=50;
            d=0;
        }
        if(d>=2)
        {
            fare=50;
            d-=2;
        }
        if(d>0)
            fare=fare+12*d;
        if(w>0)
        fare=fare+2*w;
        printf("%d\n",fare);
    }
    return 0;
}
