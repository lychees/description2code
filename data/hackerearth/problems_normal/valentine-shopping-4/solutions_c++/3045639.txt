#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--) {
    	int n, m;
    	cin >> n >> m;
    	vector < int > ans(n);
    	for (int i = 0; i < n; i++) {
    		double val = 100000000.0;
    		int pos = 1;
    		for (int j = 0; j < m; j++) {
    			double a, b, c;
    			cin >> a >> b >> c;
    			double temp = (1.0 - a/100.0);
    			temp = temp*(1.0 - b/100.0);
    			temp = temp *(1.0 - c/100.0);
    			if (temp < val) {
    				val = temp;
    				pos = j + 1;
    			}
    		}
    		ans[i] = pos;
    	}
    	for (int i = 0; i < n; i++) {
    		cout << ans[i] << " ";
    	}
    	cout << endl;
    }
    return 0;
}
