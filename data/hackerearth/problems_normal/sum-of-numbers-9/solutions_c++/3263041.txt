#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t; cin >> t;
    while(t--) {
    	int n;	cin >> n;
    	int arr[15];
    	int flag = 0;
    	for(int i = 0; i <n; i++) {
    		cin >> arr[i];
    	}
    	int get_sum;
    	cin >> get_sum ;
    	
    	for(int j = 0; j < (1 << n);j++) {
    		//cout << i << endl;
    		int i = j;
    		int counter = 0;
    		int sum = 0;
    		while(i) {
    			if(i & 1) {
    				sum = sum + arr[counter];
    			}
    			
    			counter++;
    			i = i / 2;
    			
    		}
    		
    		if(sum == get_sum) {
    			cout << "YES\n";
    			flag = 1;
				break;
    		}
    		
    	}
    	if(flag == 0)
    	cout << "NO\n";
	}
	return 0;
}
