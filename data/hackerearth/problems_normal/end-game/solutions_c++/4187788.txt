#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,a,b,c,d,m;
    	cin>>n>>a>>b>>c>>d>>m;
    	if(m==1)
    	a-=1;
    	int pawn=n-a;
    	int ver=n-c;
    	int hoz=pawn;
    	if(ver<=pawn && hoz>=abs(d-b))
    	cout<<"Draw"<<endl;
    	else
    	cout<<"White Wins"<<endl;
    }
    return 0;
}
