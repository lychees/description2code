#include <iostream>
using namespace std;
static int T,N,EL[1000000],i,j,k,R[100000],temp;
main()
{
	cin>>T;
	for(i=0;i<T;i++)
	{
		cin>>N;
		for(j=0;j<N;j++)
		cin>>EL[j];
		for (j = 0 ; j < ( N - 1 ); j++)
		  {
		    for (k = 0 ; k < N - j - 1; k++)
		    {
		      if (EL[k] > EL[k+1]) /* For decreasing order use < */
		      {
		        temp  = EL[k];
		        EL[k]   = EL[k+1];
		        EL[k+1] = temp;
		      }
		    }
		  }
		  temp=0;
		  for(j=0;j<N-1;j++)
		  temp=temp+(EL[j+1]-EL[j]);
		  R[i]=temp*2;
	}
	for(i=0;i<T;i++)
	{
		cout<<R[i]<<"\n";
	}
}
