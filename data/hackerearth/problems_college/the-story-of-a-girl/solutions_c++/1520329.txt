#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define mod 1000000007
void matmul(ll a[][2],ll b[][2],ll c[][2])
{
    int i,j,k;
    for(i=0;i<2;i++)
    {
        for(j=0;j<2;j++)
        {   c[i][j]=0;
            for(k=0;k<2;k++)
                c[i][j]+=(a[i][k]*b[k][j])%mod;
        }
    }
}
void matexp(ll z[][2],ll n,ll res[][2])
{
    ll temp[2][2];
    int i,j;
    res[0][0]=1; res[0][1]=0; res[1][0]=0; res[1][1]=1;
    while(n>0)
    {
        if(n&1)
        {
            matmul(res,z,temp);
            for(i=0;i<2;i++)
                for(j=0;j<2;j++)
                res[i][j]=temp[i][j]%mod;
        }
        matmul(z,z,temp);
        for(i=0;i<2;i++)
        {
            for(j=0;j<2;j++)
                z[i][j]=temp[i][j]%mod;
        }
      n=n/2;
    }
}
