#include <iostream>
#include <cstdio>
#include <vector>

using namespace std;


void solve() {
    int n, k; cin >> n >> k;
    vector<int> x(n); for (int i = 0; i < n; ++i) cin >> x[i];
    int res = x[n - 1];
    int l = 0;
    int r = x[n - 1];
    while (l <= r) {
        int m = (l + r) / 2;
        int i, t = 0;
        bool ok = true;
        for (i = 0; i < n - 1;) {
             int j = i;
             while (j < n && x[j] - x[i] <= m) ++j;
             if (j - 1 == i) {
                 ok = true;
                 break;
             }
             i = j - 1;
             ++t;
        }
        if (i == n - 1 && ok && t <= k) {
            res = m;
            r = m - 1;
        } else {
            l = m + 1;
        }
    }
    cout << res << endl;
}

int main() {
    int t;
    cin >> t;
    while (t --> 0)
        solve();
    return 0;
}
