#include <bits/stdc++.h>
using namespace std;
 
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long int n,p,c=0,f=0;
    	cin>>n>>p;
    	long int a[n];
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    		if(a[i]>=p)
    		c++;
    		if(a[i]==p)
    		f=1;
    	}
    	if(f==0)
    	cout<<"NO 0"<<"\n";
    	else
    	cout<<"YES "<<c-1<<"\n";
    }
}
