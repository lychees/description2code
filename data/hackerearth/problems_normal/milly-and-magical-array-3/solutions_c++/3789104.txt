#include <iostream>
using namespace std;

int main()
{
    int cases;
    cin>>cases;
    while(cases--)
    {
    	int size;
    	long jump;
    	cin>>size>>jump;
    	long pichhla, curr;
    	cin>>pichhla;
    	size--;
    	long result=0;
    	while(size--)
    	{
    		cin>>curr;
    		long steps=0;
    		if(curr<=pichhla)
    		{
    			long diff = pichhla-curr+1;
    			steps = diff/jump;
    			if(diff%jump>0)
    				steps++;
    		}
    		result+=steps;
    		pichhla=curr+(steps*jump);
    	}
    	cout<<result<<endl;
    }
    return 0;
}
