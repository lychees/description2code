#include <iostream>
#include <cstdio>
#include <algorithm>
#include <vector>
using namespace std;
int main()
{
	int n,l,r,q,i,num;
	vector<int> arr;
	vector<int>::iterator it;
	scanf("%d%d%d%d",&n,&l,&r,&q);
	for(i=0;i<n;i++){
		scanf("%d",&num);
		if(i>=l&&i<=r)
			arr.push_back(num);
	}
	sort(arr.begin(),arr.end());
	while(q--){
		scanf("%d",&num);
		it=lower_bound(arr.begin(),arr.end(),num+1);
		if(it!=arr.end())
			printf("%d\n",*it);
		else
			printf("-1\n");
	}
    return 0;
}
