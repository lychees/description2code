#include <bits/stdc++.h>
using namespace std;
int main()
{
int t;
cin>>t;
while(t--)
{
long long int c=0,i,x,y,z,m,n;
cin>>z>>m>>n;
for(i=1;i<=sqrt(z);i++)
{
	if((z%i)==0)
	{
		if(i<=m&&(z/i)<=n)
			c++;
		if(i<=n&&(z/i)<=m)
			c++;
		if(i==(z/i))
			c--;
	}
}
cout<<c<<"\n";
}	
return 0;
}
