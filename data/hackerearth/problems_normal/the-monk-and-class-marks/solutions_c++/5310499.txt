#include<bits/stdc++.h>
using namespace std;

int main()
{
	int n;
	cin >> n;
	pair<int,string> p[n];
	int marks;
	string stng;
	for(int i=0;i<n;i++)
	{
		cin >> stng >> marks;
		p[i]=make_pair(-marks,stng);
	}
	sort(p,p+n);
	for(int i=0;i<n;i++)
	{
		cout << p[i].second << " " << -1*p[i].first << endl;
	}
	return 0;
}
