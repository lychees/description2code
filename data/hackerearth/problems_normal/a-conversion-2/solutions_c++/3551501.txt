#include <bits/stdc++.h>
#define ll long long
using namespace std;
int l;
string s1,s2;
int myfind(string t,int index)
{
	int temp,ans=INT_MAX;
	if(index==l)
	return 0;
	if(t.at(index)==s2.at(index))
	temp=0;
	else if(t.at(index)>s2.at(index))
	temp=(t.at(index)-s2.at(index));
	else
	temp=(s2.at(index)-t.at(index));
	ans=min(ans,temp+myfind(t,index+1));
	int i;
	char c;
	for(i=index+1;i<l;i++)
	{
		c=t.at(i);
		t.at(i)=t.at(index);
		t.at(index)=c;
		if(t.at(index)==s2.at(index))
		temp=0;
		else if(t.at(index)>s2.at(index))
		temp=(t.at(index)-s2.at(index));
		else
		temp=(s2.at(index)-t.at(index));
		ans=min(ans,1+temp+myfind(t,index+1));
		c=t.at(i);
		t.at(i)=t.at(index);
		t.at(index)=c;
	}
	return ans;
}
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int test;
	cin>>test;
	while(test--)
	{
		cin>>s1>>s2;
		l=s1.length();
		int ans=myfind(s1,0);
		cout<<ans<<endl;
	}
	return 0;
}
