#include <iostream>
using namespace std;
int fun(long long int a,long long int b)
{
	long long int parent,child;
	if(a>b)
	{
		parent=b;
		child=a;
	}
	else
	{
		child=b;
		parent=a;
	}
	if(parent==child)
	return parent;
	while(child>parent)
		{
			if(child%2==0)
			child=child/2;
			else
			child=(child-1)/2;
		}
		return fun(parent,child);
}
int main()
{
	int t;
	long long int a,b;
	cin>>t;
	while(t--)
	{
		cin>>a>>b;
		cout<<fun(a,b)<<endl;
	}
    return 0;
}
