#include <bits/stdc++.h>
using namespace std;
const int MAXT = 500 + 5;
const int MAXN = 1e6 + 6;
const int MOD = 1e9 + 7;

int T;
int N;

int fact[ MAXN ];
int table[ MAXN ];

int pw( int a, int p ){
    int res = 1;
    while( p ){
        if( p & 1 )
            res = 1ULL * res * a % MOD;
        p >>= 1;
        a = 1ULL * a * a % MOD;
    }
    return res;
}

int solve( int n ){
    int a = pw( 2, n - 2 );
    int b = pw( 2, n / 2 - 1 );
    return ( a + b + 1 ) % MOD;
}

int main(){
    cin >> T;
    while( T-- ){
        cin >> N;
        if( N <= 2 ){
            cout << 1 << endl;
            continue;
        }
        if( N == 3 ){
        	cout << 2 << endl;
        	continue;
        }
        N -= 2;
        cout << solve( N ) << endl;
    }
    return 0;
}
