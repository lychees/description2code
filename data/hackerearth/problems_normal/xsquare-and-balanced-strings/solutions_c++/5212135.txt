#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int i=0;
    	string str;
    	cin>>str;
    	int a[26]={0};
    	while(str[i]){
    		++a[str[i]-'a'];
    		i++;
    	}
    	int flag=0;
    	for(int i=0;i<26;i++){
    		if(a[i]%2!=0){
    			flag=1;
    			break;
    		}
    	}
    	if(flag==0)
    		cout<<"1"<<endl;
    	else
    		cout<<"-1"<<endl;
    }
    return 0;
}
