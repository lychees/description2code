#include <iostream>
using namespace std;

int main()
{
	int t, r,c;
	long long odd,even;
	cin>>t;
	while(t--){
		cin>>r>>c;
		odd=0;even=0;
		for(int i=1;i<=r;i++){
			if(i%2){
				if(i>c) odd+=c;
				else odd+=i;
			}
			else{
				if(i>c) even+=c;
				else even+=i;
			}
		}
		cout<<even<<" "<<odd<<endl;
	}
    
    return 0;
}
