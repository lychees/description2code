#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define d1(a)cout<<#a<<": "<<a<<"\n";
#define d2(a,b)cout<<#a<<": "<<a<<" , "<<#b<<": "<<b<<"\n";
#define d3(a,b,c)cout<<#a<<": "<<a<<" , "<<#b<<": "<<b<<" , "<<#c<<": "<<c<<"\n";
#define d4(a,b,c,d)cout<<#a<<": "<<a<<" , "<<#b<<": "<<b<<" , "<<#c<<": "<<c<<" , "<<#d<<": "<<d<<"\n";

int a[105][105];
bool dp[105][105][505];

int main() {
	int t;
	cin>>t;
	while(t--){
		int n,m,o;
		memset(dp,false,sizeof(dp));
		memset(a,0,sizeof(a));
		cin>>n>>m>>o;
		for(int i=0;i<n;i++){
			for(int j=0;j<m;j++){
				cin>>a[i][j];
			}
		}
		for(int i=0;i<n;i++){
			for(int j=0;j<m;j++){
				for(int k=a[i][j];k<=o;k++){
					if(i == 0 && j == 0){
						if(k == a[i][j]){
							dp[i][j][k] = true;
						}
					}
					else if(i == 0){
						dp[i][j][k] = (dp[i][j][k] || dp[i][j-1][k-a[i][j]]);
					}
					else if(j == 0){
						dp[i][j][k] = (dp[i][j][k] || dp[i-1][j][k-a[i][j]]);
					}
					else{
						dp[i][j][k] = (dp[i][j][k] || dp[i][j-1][k-a[i][j]]);
						dp[i][j][k] = (dp[i][j][k] || dp[i-1][j][k-a[i][j]]);
						dp[i][j][k] = (dp[i][j][k] || dp[i-1][j-1][k-a[i][j]]);
					}
				}
			}
		}
		int ans = -1;
		for(int i=o;i>=0;i--){
			if(dp[n-1][m-1][i]){
				ans = i;
				break;
			}
		}
		cout<<ans<<"\n";
	}
	return 0;
}
