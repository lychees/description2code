#include<bits/stdc++.h>
 
using namespace std;
int main()
{
	ios_base::sync_with_stdio(0);
	int n,a,x;
	cin>>n; 
	stack <int>st;
	for(int i=0;i<n;i++)
	{
		cin>>a;
		if(a==1)
		{
		if(!st.empty())
		{
		  cout<<st.top()<<endl;	
		  st.pop();	}
		else
		  cout<<"No Food"<<endl;
	}
		else
		{
			cin>>x;
            st.push(x);
		}
		
		
		
		
	}					
    return 0;
}
