#include<iostream>
using namespace std;
#define MAX 1001

int main(){
	long long int t, n, possibleSteps[MAX];
	
	possibleSteps[0]=0;
	possibleSteps[1]=1;
	possibleSteps[2]=2;
	for(int i=3; i<MAX; i++){
	        possibleSteps[i]=possibleSteps[i-2] + possibleSteps[i-1];
    }
    
	cin>>t;
	while(t--){
		cin>>n;
		cout<<possibleSteps[n]<<endl;
	}
	return 0;
}
