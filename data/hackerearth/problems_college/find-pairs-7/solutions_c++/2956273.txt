#include <iostream>
using namespace std;
long long a[10000000];
int main()
{
    long long t,i,n,co,j;
    cin>>t;
    while(t--)
    {co=0;
    	cin>>n;
    	for(i=0;i<n;++i)
    	{
    		cin>>a[i];
    	}
    	for(i=0;i<n;++i)
    	{
    		for(j=i;j<n;++j)
    		{
    			if(a[i]==a[j])
    			{
    				co++;
    			}
    		}
    	}
    	cout<<co<<"\n";
    }
    return 0;
}
