#include <bits/stdc++.h>

using namespace std;

int a[100005];

int main()
{
	int t,n,k,cnt;
	cin >> t;
	while ( t-- ) {
		cin >> n >> k;
		cnt = 0;
		for ( int i = 0; i < n; i++ ) cin >> a[i];
		for ( int i = 0; i < n; i++ ) {
			int idx = -1;
			int mn;
			for ( int j = i+1; j < n; j++ ) {
				if ( a[i] > a[j] ) {
					if ( (cnt+(j-i)) <= k ) {
						if ( idx == -1 ) {
							idx = j;
							mn = a[j];
						}
						else {
							if ( mn > a[j] ) {
								mn = a[j];
								idx = j;
							}
						}
					}
				}
			}
			if ( idx != -1 ) {
				for ( int j = idx; j >= i+1; j-- ) {
					swap(a[j-1],a[j]);
				}
				cnt += (idx-i);
			}
			if ( cnt == k ) break;
		}
		for ( int i = 0; i < n-1; i++ ) cout << a[i] << " ";
		cout << a[n-1] << endl;
	}
	return 0;
}
