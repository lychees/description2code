#include<iostream>
#include<math.h>
#include<vector>
#define Max 4000000
using namespace std;
vector <int> isPrime(Max,1);
void genprime(int n){
    long long k=0;
    for(int i=2; i<n; i++) if(isPrime[i]){
        k=k+i;
        for(int j=i; j<n; j+=i){
            if(j>i) isPrime[j] = 0;
        }
    }
    cout<<k;
}
int main()
{

	int n;
	cin>>n;
	genprime(n);
	return 0;
}
