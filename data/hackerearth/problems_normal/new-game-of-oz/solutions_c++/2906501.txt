#include<iostream>
#include<algorithm>
using namespace std;
#include<vector>

int main(){
	int i,t,n,s,x,c;
	vector<int> v;
	cin>>t;
	while(t--){
		c=0;
		v.clear();
		cin>>n;
		for(i=0;i<n;i++)
		{
			cin>>s;
			v.push_back(s);
		}	
		std::sort(v.begin(),v.end(),std::greater<int>());
		for(i=0;i<v.size();i++)
		{
			x = v[i];
			c++;
			if(i!=(n-1))
			{
				if(v[i+1]==x-1)
				i++;
			}
			
		}
		cout<<c<<endl;
	}
	
}
