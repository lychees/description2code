#include<bits/stdc++.h>
using namespace std;
int K;

int main()
	{
		int N,K,T;
		cin>>T;
		while(T--)
			{
				cin>>N>>K;
				int x,j;
				vector<int>h;
				vector<pair<int,int>>boys(N);
				for(int i=0;i<N;i++)
					{
						scanf("%d",&boys[i].first);
						boys[i].second=i;
						//orgnl.push_back(x);
					}
				for(int i=0;i<K;i++)
					{
						scanf("%d",&x);
						h.push_back(x);
					}
				sort(boys.begin(),boys.end());
				//sort(h.begin(),h.end());
				int i=0;
				int ans=0;
				for(int j=0;j<K&&i<N;++j)
					{
						int mini=INT_MAX;
						while(i<N && boys[i].first<h[j])
							{
								mini=min(boys[i].second,mini);
								++i;
							}
						ans=mini;
					}
				if(i<N)
					{
						ans=INT_MAX;
						while(i<N)
							{
								ans=min(ans,boys[i].second);
								++i;
							}
					}
				cout<<ans<<endl;
			}
		return 0;
	}
