    #include <bits/stdc++.h>
    using namespace std;
    #define ll long long int 
    int main()
    {
        ll test;
        double a,b,c,ans;
        cin>>test;
        while(test--)
        {
        	ans=0.0;
        	cin>>a>>b>>c;
        	ans+=min(min(a,b),c);
        	printf("%.1lf\n",ans/2);
        }
    }
