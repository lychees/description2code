#include<cstdio>

int main()
{
          long long t,l,b,h,min;
          scanf("%lld\n",&t);
          while(t--)
          {
                    scanf("%lld%lld%lld",&l,&b,&h);
                    min=l;
                    if(b<min) min=b;
                    if(h<min) min=h;
                    if(min%2==0) printf("%lld.0\n",min/2);
                    else         printf("%lld.5\n",min/2);
          }
}
