#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long n;
    scanf("%lld",&n);
    while(n--)
    {
        string a;
        char c;
        scanf("%d",&c);
        getline(cin,a);
        int x=sqrt(a.length());
        char q[x][x];int k=0;
        for(int i=0;i<x;i++)
        {
            for(int j=0;j<x;j++)
            {
                q[i][j]=a[k];
                k++;
            }
        }
        for(int j=0;j<x;j++)
        {
            for(int i=0;i<x;i++)
                printf("%c",q[i][j]);
        }
        printf("\n");
    }
    return 0;
}
