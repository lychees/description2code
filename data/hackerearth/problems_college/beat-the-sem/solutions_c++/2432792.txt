#include <iostream>
using namespace std;

int main()
{
    int t , s , d;
    cin>>t;
    while(t--)
    {
    	cin>>s>>d;
    	if(s%2 == 0)
    	{
    		if(d>=0 && d<=s && d%2 == 0)
    			cout<<(s+d)/2<<" "<<s-((s+d)/2)<<"\n";
    		else
    			cout<<"impossible\n";
    	}
    	else
    	{
    		if(d>0 && d<=s && d%2 != 0)
    			cout<<(s+d)/2<<" "<<s-((s+d)/2)<<"\n";
    		else
    			cout<<"impossible\n";
    	}
    }
    return 0;
}
