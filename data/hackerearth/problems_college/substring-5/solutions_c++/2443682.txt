#include <iostream>
#include <cstdio>
#include <map>
#include <string>
using namespace std;

int main()
{
    string s;

    cin>>s;
    map<char,char> m1;
    int i,count=0,n;
    
    n = s.size();
    for(i=0;i<n;i++){
        m1.insert(pair<char,char>(s[i],s[i]));
    }
    int j,k,l;
    map<char,char>::iterator it=m1.begin();
    while(it!=m1.end()){
        for(i=2;i<=n;i++){
            string match="";
            for(k=0;k<i;k++){
                match = match+it->second;
            } 
            for(j=0;j<=n-i+1;j++){
                string match1="";
                for(l=j;l<j+i;l++){
                    match1 = match1 + s[l];
                }
                if(match1.compare(match)==0){
                    count++;
                }
            }
        }
        ++it;
    }
    count = count + s.size();
    cout<<count<<"\n";
    
    
    return 0;
}
