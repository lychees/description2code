#include<iostream>
#include<string>
 
using namespace std;
 
 
int main()
{
	string input;
	getline(cin, input);
	int length = input.length();
 
	if (islower(input[0]))
		input[0] = toupper(input[0]);
	for (int i=0; i< length; i++)
	{
		while (input[i] != ' ' && (i < length))
			i++;
		
		if (islower(input[i+1]))
				input[i+1] = toupper(input[i+1]);
	}
 
	cout << input;
	return 0;
}
