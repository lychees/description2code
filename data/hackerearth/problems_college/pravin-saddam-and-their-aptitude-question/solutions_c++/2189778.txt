#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t, a, b, z[4], i, k, m;
    z[0]=1;
    z[1]=2;
    z[2]=6;
    z[3]=24;
    cin>>t;
    while(t--) {
    	k=0;
    	cin>>a>>b;
    	if(min(a, b)>=5)
    		cout<<"0\n";
    	else {
    		m=max(a,b);
    		if(m>5)
    			m=4;
    		for(i=min(a, b)-1;i<m;i++) {
    			k+=z[i];    		
    		}
    		cout<<k%10<<endl;
    	}
    }
    return 0;
}
