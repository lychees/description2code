#include<bits/stdc++.h>
 
using namespace std;
 
int main()
{
	int cases;
	cin >> cases;
	while(cases--) {
		int n, a, b, c;
		scanf("%d %d %d %d", &n, &a, &b, &c);
		long long int x = 0;
		for(int i = 0; i <= a; i++) {
			for(int j = 0; j <= min(n-i, b); j++) {
				if(n-i-j < 0)
					break;
				else {
					x += min(n-i-j+1, c+1);
				}
			}
		}
		printf("%lld\n", x);
	}
}
