#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

        bool compare(int a, int b) {
                if (a % 2 == 0 && b % 2 == 1)
                        return true;
                else if ( a % 2 == 1 && b % 2 == 0)
                        return false;
                else
                        return a < b;
        }

int main() {
        int size;
        cin >> size;
        vector<int> v ;
        int val;
        for(int i = 0;i < size; ++i) {
                cin >> val;
                v.push_back(val);
        }
        stable_sort(v.begin(),v.end(),compare);
        bool printOnce = true;
        int oddSum = 0;
        int evenSum = 0;
        for(int i = 0; i < v.size(); ++i) {
                if (v[i] % 2 == 0) {
                        evenSum += v[i];
                        cout<<v[i]<<" ";
                }else{
                        if(printOnce) {
                                cout<<evenSum<<" ";
                                printOnce = false;
                        }
                        oddSum += v[i];
                        cout<<v[i]<<" ";
                }
        }
        cout<<oddSum<<endl;
}
