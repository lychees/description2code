#include <cstdio>
#include <string>
#include <cstring>
#include <iostream>
using namespace std;
int main()
{
	string s;
	int Q, l;
	long long a, b;
	cin >> s;
	l = s.size();
	cin>>Q;
	while (Q--){
		cin>>a>>b;
		
		a--;
		b--;
		a = a%l;
		b = b%l;
		if (s[a]==s[b])
			cout<<"Yes"<<endl;
		else
			cout<<"No"<<endl;
	}
	return 0;
}
