#include <iostream>
using namespace std;

int main()
{
    unsigned long long n,q,d,d1;
    cin>>n;
    long arr[n];
    for(long i=0;i<n;i++) cin>>arr[i];
    cin>>q;
    d1=1;
    while(q--){
    	cin>>d;
    	d1*=d;
    	if(d1>1000000){
    		for(long i=0;i<n;i++) cout<<0<<" ";
    		return 0;
    	}
    }
    for(long i=0;i<n;i++) cout<<arr[i]/d1<<" ";
    return 0;
}
