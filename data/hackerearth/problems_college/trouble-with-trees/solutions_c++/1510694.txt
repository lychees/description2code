#include<iostream>
using namespace std;
unsigned long long int catalanNumber(unsigned int n)
{

    unsigned long long int catalan[n+1];
 
    catalan[0] = catalan[1] = 1;
 

    for (int i=2; i<=n; i++)
    {
        catalan[i] = 0;
        for (int j=0; j<i; j++)
            catalan[i] = (catalan[i]+(catalan[j] * catalan[i-j-1])%1000000007)%1000000007;
    }
 

    return catalan[n];
}
 

int main()
{
    int t;
	cin>>t;
	while(t--)
	{
		unsigned int n;
		cin>>n;
		cout<<catalanNumber(n)<<endl;
			
	}
    return 0;
}
