for t in range(input()):
	n,s = raw_input().split()
	n = int(n)
	
	a = [0]*26
	for x in s:
		a[ ord(x)-97 ] += 1
	
	mxAns = 0
	noOfMem = 1<<30
	idxAns = n
	
	for i in range(n):
		sm = 0
		s = raw_input()
		for x in s:
			sm += a[ ord(x)-97 ]
		
		if mxAns < sm :
			mxAns = sm
			noOfMem = len(s)
			idxAns = i+1
		
		elif mxAns == sm :
			if noOfMem > len(s) :
				noOfMem = len(s)
				idxAns = i+1
	
	print idxAns
