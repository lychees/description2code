#include <iostream>
using namespace std;

int main()
{
	int N = 0;
    cin >> N;
    for(int i=0; i<N; i++)
    {
    	string num1, num2;
    	cin >> num1;
    	cin >> num2;
    	const char *n1, *n2;
    	n1 = num1.c_str();
    	n2 = num2.c_str();
    	for(int j=0; j<num1.length(); j++)
    	{
    		if(n1[j] == n2[j])
    			cout << '0';
    		else
    			cout << '1';
    	}
    	cout << endl;
    }
    return 0;
}
