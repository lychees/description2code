#include<iostream>
    #include<algorithm>
    using namespace std;
    int main(){
    long t,r,s,n,sumr,sums;
    cin>>t;
    while(t--){
    cin>>r>>s>>n;
    long a[n];
    for(int i=0;i<n;i++)
    cin>>a[i];
    sort(a,a+n);
    sumr=0;
    sums=0;
    for(int i=0;i<n;i++){
    if(sumr+a[i]<=r)
    sumr=sumr+a[i];
    else
    break;
    }
    for(int i=0;i<n;i++){
    if(sums+a[i]<=s)
    sums=sums+a[i];
    else
    break;
    }
    if(sumr>sums)
    cout<<"Raghu Won"<<endl;
    else if(sumr<sums)
    cout<<"Sayan Won"<<endl;
    else
    cout<<"Tie"<<endl;
    }
    }
