int gcd(int a,int b)
{
	if(b==0)
	return 0;
	else return (gcd(b,a%b));
}

// function to check that substring (i...len) is palindrome or not.
int check(char *str, int i, int len)
{
    int m,n;
    for(m=i, n=len-1 ; m <= n; m++, n--)
    if(str[m]!=str[n])
    return 0;

    return 1;

}

int caclangle(int hour, int min) 
{ 
if(hour<0||hour>12) 
return -1; 
if(min<0||min>60) 
return -1; 
if(hour==12)

if(min==60) 
{min=0;hour++;} 
int angle=30*hour+0.5*min-6*min; 
if(angle<0&&angle>-180) 
return -angle; 
if(angle<=-180) 
return 360+angle; 
if(angle>180) 
return 360-angle; 
}
