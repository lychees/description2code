#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int a[n],i,s=0,temp,j;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	for(i=n-1;i>0;i--)
    	{
    		if(a[i]<=a[i-1])
    		s++;
    	}
    	cout<<s<<endl;
    }
    return 0;
}
