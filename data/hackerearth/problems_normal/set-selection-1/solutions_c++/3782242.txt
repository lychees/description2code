#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t, n, z;
	cin>>n;
	int a[n];
    for(int i=0; i<n; i++){
    	cin>>a[i];
    }
    bool dp[100001];
    memset(dp, false, sizeof(dp));
    dp[0]=true;
    for(int i=0; i<n; i++){
    	for(int j=100000-a[i]; j>=0; j--){
    		if(dp[j]){
    			dp[j+a[i]]=true;
    		}
    	}
    }
    cin>>t;
    for(int i=0; i<t; i++){
    	cin>>z;
    	if(dp[z]){
    		cout<<"Yes\n";
    	}else{
    		cout<<"No\n";
    	}
    }
    return 0;
}
