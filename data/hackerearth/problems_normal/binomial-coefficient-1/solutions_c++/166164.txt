#include<iostream>
using namespace std;
const long long mod=1000000007;
long long pow(long long a,long long b)
 {if(b==0)
   return 1;
  if(b==1)
    return a;
   long long c=pow(a,b/2);
   c=(c*c)%mod;
   if(b%2)
     c=(c*a)%mod;
    return c;
 }
long long fact[1000002]={0};
long long ifact[1000002]={0};

void preprocess()
{fact[0]=fact[1]=1;
 long long i;
 for(i=2;i<=1000000;i++)
   fact[i]=(i*fact[i-1])%mod;
 ifact[1000000]=pow(fact[1000000],mod-2);
 for(i=999999;i>=0;i--)
   ifact[i]= ((i+1)*ifact[i+1])%mod;
}
int main()
 {long long a,b,n,k,v,x;
 preprocess();
  cin>>a>>b>>n>>k;
  //cout<<fact[n]<<" "<<ifact[k]<<" "<<ifact[n-k]<<" "<<endl;
  v= (((fact[n]*ifact[k])%mod)*ifact[n-k])%mod;
  x=(pow(a,k)*pow(b,n-k))%mod;
  cout<<(v*x)%mod<<endl;
 }
