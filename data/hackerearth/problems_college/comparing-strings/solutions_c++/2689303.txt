#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    getchar();
    while(t--)
    {
    	int i=0,temp=0;
    	string a,b;
    	cin>>a>>b;
    	while(a[i]!='\0' &b[i]!='\0')
    	{
    		if(a[i]==b[i]) {temp=1;break;}
    		i++;
    	}
    	if(temp) cout<<"YES"<<endl;
    	else cout<<"NO"<<endl;
    }
    return 0;
}
