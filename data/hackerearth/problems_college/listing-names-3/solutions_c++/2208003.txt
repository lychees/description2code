#include <bits/stdc++.h>
using namespace std;
int main()
{
	int n, i, j;
    scanf("%d", &n);
    string s[n+1];
    for(i = 0 ; i <n ;i++)
   cin>>s[i];
    for(i = 0 ; i < n; i++) {
    	for(j = i+1; j < n; j++) {
    		if(s[i].size() > s[j].size()) {
    			string q = s[i];
    			s[i]  = s[j];
    			s[j] = q;
    		}
    	}
    }
    string a[n+1];
    int b[n+1],k = 0;
    for(i = 0 ; i < n; i++)
    b[i] = 1;
    a[0] = s[0];
    for(i = 0 ; i <n-1 ;i++) {
    	if(!s[i].compare(s[i+1])) {
    		b[k]++;
    	}
    	else
    	a[++k] = s[i+1];
    }
    for(i = 0; i < k; i++) {
    	for(j = 0 ; j <= k; j++) {
    		if(b[i] < b[j]) {
    			string q = a[i];
    			a[i]  = a[j];
    			a[j] = q;
    			n = b[i];
    			b[i] = b[j];
    			b[j] = n;
    		}
    	}
    }
    for(i = 0 ; i <= k; i++)
    cout<<a[i]<<" "<<b[i]<<"\n";
    return 0;
}
