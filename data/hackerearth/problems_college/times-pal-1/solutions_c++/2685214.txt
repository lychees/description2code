#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstdlib>
#include<vector>
#include<cstring>
using namespace std;
int main()
{
          int t;
          cin>>t;
          while(t--)
          {
          string str1,str2;
          cin>>str1>>str2;
          int l=str1.length(),count=0;
          for(int i=0;i<l;i++)
          {
                    string s = str1.substr(0,i) + str2 + str1.substr(i,l);
                    string ss= s;
                    reverse(s.begin(),s.end());
                    if(ss==s)
                              count++;
                    //cout<<s<<endl;
          }
          string sss=str1+str2;
          string srev = sss;
          reverse(sss.begin(),sss.end());
          if(sss==srev)
                    count++;
          cout<<count<<endl;
          }
          return 0;
}
