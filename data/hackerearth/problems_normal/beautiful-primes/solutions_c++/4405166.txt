#include <iostream>
#include <bits/stdc++.h>
using namespace std;
long int m=1e9+7;
#define lli long long int
lli pw[100005],p[100005];
lli mexp(lli a,lli b)
{
	if(b==0)
	{
		return 1;
	}
	lli t=mexp(a,b/2);
	if(b%2==0)
	{
		return (t%m*t%m)%m;
	}
	else
	{
		return ((t%m*t%m)%m*a%m)%m;
	}
}
int main()
{
    lli t,x,ans,i;
    cin>>t;
    while(t--)
    {
    	cin>>x;
    	for(i=1;i<=x;i++)
    	{
    		cin>>p[i];
    	}
    	for(i=1;i<=x;i++)
    	{
    		cin>>pw[i];
    	}
    	ans=1;
    	for(i=1;i<=x;i++)
    	{
    		ans=(ans%m*((mexp(p[i],pw[i]+1)%m-p[i]+m)%m))%m;
    		ans=(ans%m*(mexp(p[i]-1,m-2)%m))%m;
    		//ans*=mexp(p[i],pw[i]+1)%m;
    	//	ans/=(p[i]-1);
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
