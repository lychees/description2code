#include <iostream>
#define ll long long
using namespace std;

int main()
{
    ll n = 0, nn = 0, tc = 0, sum = 0, c = 0;
    cin >> n;
    
    for (ll i = 0; i < n; i++){
    	cin >> nn;
    	for (ll j = 0; j < nn; j++){
    		cin >> tc;
    		sum+=tc;
    		c++;
    	}
    	cout << sum / c<< endl;
    	sum = 0;
    	c = 0;
    }
    return 0;
}
