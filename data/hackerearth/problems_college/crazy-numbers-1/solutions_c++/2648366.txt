#include <iostream>
#include <vector>
#include <queue>
#include <string>
#include <assert.h>
#define MOD 1000000007
using namespace std;

int main()
{
	
	
	int t,n;
	scanf("%d",&t);
	
	while(t--)
	{
		long long int count[10],answer[10],sum;
		count[0]=1;
		count[1]=1;
		count[2]=2;
		count[3]=2;
		count[4]=2;
		count[5]=2;
		count[6]=2;
		count[7]=2;
		count[8]=2;
		count[9]=1;
		
		sum=0;
		scanf("%d",&n);
		//assert(n>=1);
		if(n==0)
			printf("0\n");
		else if(n==1)
			printf("10\n");
		else if(n==2)
			printf("17\n");
		else
		{
			for(int k=2;k<n;k++)
			{
				for(int i=0;i<10;i++)
				{
					if(i==0)
						answer[i] = count[i+1]%MOD;
					else if(i==9)
						answer[i] = count[i-1]%MOD;
					else	
						answer[i] = count[i-1]%MOD + count[i+1]%MOD;				
				}
				
				//printf("\n");
				for(int i=0;i<10;i++)
				{
					//printf("%lld ",answer[i]);
					count[i]=answer[i];
				}
			}
			
			for(int j=0;j<10;j++)
				sum = ((sum + answer[j]))%MOD;
				
			printf("%lld\n",sum);
		}
	}
	return 0;
}
