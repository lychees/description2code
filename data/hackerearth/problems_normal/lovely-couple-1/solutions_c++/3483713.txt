#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
	int i, j, T, a, b, count; 
	vector<int> primes;
	int primeTrue[1001] = {0};
	for (i = 2; i < 32; i++) {
		if (primeTrue[i] != 1) {
			for (j = i * i; j < 1001; j += i) {
				primeTrue[j] = 1;
			}
		}
	}
	for(i = 2; i < 1001; i++) {
		if(primeTrue[i] != 1) {
			primes.push_back(i);
		}
	}
	cin >> T;
	for(i = 0; i < T; i++) {
		cin >> a >> b;
		count = 0;
		vector<int>::iterator it = primes.begin();
    	while(it != primes.end()) {
        	if(a % *it == 0 || b % *it == 0) {
        		count++;
        	}
        	it++;
		}
		if(primeTrue[count] == 1 || count < 2) {
			cout << "No" << endl;
		} else {
			cout << "Yes" << endl;
		}
    }
	
    return 0;
}
