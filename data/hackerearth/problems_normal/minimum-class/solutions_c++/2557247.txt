#include<bits/stdc++.h>
using namespace std;
long long int s = 1000000001, t, n;
long long int bs(long long int l, long long int h)
{
	long long int mid = (l + h) / 2;
	long long int b = mid / 2;
	long long int a = (mid - 1) / 2;
	b = (b*(b + 1));
	a = (a*(a + 1));
	if (mid % 2 != 0)
		b += (mid / 2 + 1);
	else
		a += ((mid - 1) / 2 + 1);
	if (b < n)
		return bs(mid + 1, h);
	if (b >= n&&a < n)
		return mid;
	return bs(l, mid-1);
}
int main()
{
	cin >> t;
	for (int i = 0; i < t; i++)
	{
		cin >> n;
		cout << bs(0, s) << endl;
	}
	return 0;
}
