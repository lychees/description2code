#include <iostream>
using namespace std;

int main()
{
	
    int t,n;
    cin>>t;
    while(t--)
    {
    	int a[100001]={0},b;
    	cin>>n;
    	for(int i=0;i<n;i++)
    	{
    		cin>>b;
    		a[b]++;
    	}
    	for(int i=0;i<100001;i++)
    	{
    		if(a[i]%2==1)
    		{
    			cout<<i<<endl;
    		}
    		
    	}
    	
    	
    }
    return 0;
}
