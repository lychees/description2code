#include<iostream>
#include<algorithm>
#include<vector>
#include<string>
#include<cstring>
#include<memory>
#include<iterator>
#include<math.h>

using namespace std;

int main()
{
    int t,x,p;
    cin>>t;while(t--)
    {
        cin>>x>>p;
        int c=0;
        while(true)
        {
            if(x%p==0)
            {
                c++;
                x=x/p;
            }
            else{
                break;
            }
        }
        if(x==1)
            cout<<c<<" 0"<<endl;
        else
            cout<<c<<" "<<x<<endl;
    }
return 0;
}
