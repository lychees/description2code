#include <iostream>
using namespace std;

typedef unsigned long long ull;

int main()
{
    ull t;
    cin >> t;
    while (t--)
    {
    	ull x, y, p=1, q=1, sum=1, count = 0;
    	cin >> x >> y;
    	
		while (sum <= y)
		{
			sum = p + q;
			p = q;
			q = sum;
			if (sum >= x && sum <= y)		
				count++;
		}
    	
    	if (count & 1)
    		cout << "INVALID\n";
    	else
    		cout << "VALID\n";
    }
    return 0;
}
