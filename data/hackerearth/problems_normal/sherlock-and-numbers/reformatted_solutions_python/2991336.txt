import sys
t = input()
for j in range(t):
    n, k, p = map(int, raw_input().split())
    num = map(int, raw_input().split())
    num.append(0)
    num.append(sys.maxint)
    num.sort()
    ans = n
    for i in range(1, len(num)):
        ivl = num[i] - num[i-1] - 1
        if (p - ivl) <= 0:
            ans = num[i-1] + p
            break
        else:
            p -= ivl
    if ans < n:
        print ans
    else:
        print -1
