#include <iostream>
#include<string.h>
using namespace std;
char str[1000000];

int main()
{
	unsigned long long int k;
	long long int n=0,i,j;
    int test,l,flag1[26],flag2[26],flag3[10];
    cin>>test;
    while(test)
    {n=0;
    	
    		for(i=0;i<26;i++)
    		{
    			flag1[i]=0;
    			flag2[i]=0;
    			
    		}
    		for(i=0;i<10;i++)
    		{
    			flag3[i]=0;
    		}
    		j=0;
    		cin>>str;
    		while(str[j]!='\0')
    	{
    		
    		l=(int)str[j];
    		j++;
    	//	cout<<l<<endl;
    		if(l>=97&&l<=122)
    		{
    			if(flag1[l-97]==0)
    			{
    				flag1[l-97]=1;//cout<<l;
    			n++;}}
    			else if(l>=48&&l<=57)
    			{
    				if(flag3[l-48]==0)
    				{
    					flag3[l-48]=1;
    					n++;
    				}
    			
    				
    			}
    			else
    			{
    				if(flag2[l-65]==0)
    				{
    					flag2[l-65]=1;
    					n++;
    				}
    			}
    		
    	
    	}
    	
    	
    	k=n*(n+1)/2;
    	cout<<k<<endl;
    	test--;
    }
    return 0;
}
