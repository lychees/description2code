#include<bits/stdc++.h>

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		int a[100000];
		long long int sum =0,leftsum=0;
		scanf("%d",&n);
		for(int i =0;i<n;i++) {
			scanf("%d",&a[i]);
			sum += a[i];
		}
		int flag = 0;
		for(int i =0;i<n;i++) {
			sum -=a[i];

			if(leftsum == sum)
			{
				flag = 1;
				printf("YES\n");break;
			}
			leftsum += a[i];
		}
		if(flag == 0)
			printf("NO\n");
	}
	return 0;
}
