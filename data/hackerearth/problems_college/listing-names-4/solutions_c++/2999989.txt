#include<iostream>
#include<cstring>
using namespace std;
void rec(char s[][35],int st,int ed,int mid)
{
    int i=st,j=mid+1,c=0;
    char b[ed-st+1][35];
    while(i<=mid&&j<=ed)
    {
        if(strcmp(s[i],s[j])<0)
            strcpy(b[c++],s[i++]);
        else
            strcpy(b[c++],s[j++]);
    }
    while(i<=mid)
        strcpy(b[c++],s[i++]);
    while(j<=ed)
        strcpy(b[c++],s[j++]);
    c--;
    while(c>=0)
        strcpy(s[ed--],b[c--]);
}
void mer(char s[][35],int st,int ed)
{
    int c=(st+ed)/2;
    if(c==st)
    {
        if(strcmp(s[st],s[ed])>0)
        {
            char* p;
            strcpy(p,s[st]);
            strcpy(s[st],s[ed]);
            strcpy(s[ed],p);
        }
    }
    else
    {
        mer(s,0,c);
        mer(s,c+1,ed);
        rec(s,0,ed,c);
    }

}
int main()
{
    int t,i,j;
    cin>>t;
    char s[t][35];
    for(i=0;i<t;i++)
        cin>>s[i];
    mer(s,0,t-1);
    j=0;
    int m;
    for(i=0;i<t;i=j)
    {
        m=1;
        for(j=i+1;j<t;j++)
        {
            if(strcmp(s[i],s[j])!=0)
            {
                cout<<s[i]<<" "<<m<<endl;
                break;
            }
            m++;
        }
    }
            cout<<s[i-1]<<" "<<m<<endl;
    return 0;
}
