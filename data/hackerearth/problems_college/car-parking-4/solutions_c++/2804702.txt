#include <iostream>
#include <string.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int i,l,count=0;
		char a[1000];
		scanf("%s",a);
		l=strlen(a);
		for(i=0;i<l;i++)
		{
			if (a[i]=='-')
			{
				
			if(i==0)
			{
				if(a[i+1]!='S'&&a[i+1]!='B'&&a[i+2]!='B')
					count++;
				else
					continue;
				
			}
			else if(i==l-1)
			{
				if(a[i-1]!='S')
					count++;
				else
					continue;
				
			}
			else
			{
				if(a[i-1]!='S'&& a[i+1]!='S'&&a[i+1]!='B'&&a[i+2]!='B')
					count++;
				else
					continue;	
			}
			}
			
		}
		cout<<count<<endl;
	}
}
