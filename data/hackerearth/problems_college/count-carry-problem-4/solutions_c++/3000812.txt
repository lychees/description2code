#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    int n;
    string s,s1;
    cin>>n;
    while(n--)
    {
    	cin>>s>>s1;
    int l=s.length()-1;
    int	l1=s.length()-1;int c=0,cnt=0,x,y;
    	while(l>=0||l1>=0)
    	{ x=0;y=0;
    		if(l>=0)
    		   x=s[l]-48;
    		 if(l1>=0)
    		   y=s1[l1]-48;
    		 c=(x+y+c)/10;
    		 if(c==1)
    		    cnt++;
    		    l1--;l--;
    	}
    	if(cnt==0)
    	   cout<<"No carry operation\n";
    	 else if(cnt==1)
    	     cout<<"1 carry operation\n";
    	  else
    	     cout<<cnt<<" carry operations\n";
    }
    return 0;
}
