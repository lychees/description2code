#include <bits/stdc++.h>
 
#define fi first
#define se second
#define pb push_back
#define mp make_pair
#define pii pair<int,int>
#define pll pair<ll,ll>
#define MOD 1000000007
#define ll long long int
#define MAXINT 2000000000
#define SET(x,y) memset(x,y,sizeof(x));
 
using namespace std;

int ans,q,n,m;
int M[1002][1002],visited[1002][1002];

void dfs(int x,int y) {
	queue<pii>qu;
	qu.push(mp(x,y));
	int dx,dy;
	while(!qu.empty()) {
		pii t=qu.front();
		x = t.fi;
		y=t.se;
		qu.pop();
		if(visited[x][y] || x<0 || x>n || y<0 || y>m || M[x][y] ==0)	continue;
		q++;
		visited[x][y]=1;
		for(dx=-1;dx<=1;dx++) {
			for(dy=-1;dy<=1;dy++) {
				if(dx == 0 && dy==0)	continue;
				qu.push(mp(x+dx,y+dy));
			}
		}
	}
}

void solve(void) {
	int i,j,c;
	ans = q= c= 0;
	cin>>n>>m;
	for(i=1;i<=n;i++) {
		for(j=1;j<=m;j++)
			visited[i][j]=0,cin>>M[i][j];
	}
	for(i=1;i<=n;i++) {
		for(j=1;j<=m;j++) {
			if(!visited[i][j] && M[i][j]){
				q=0;
				dfs(i,j);
				ans = max(ans,q);
				c++;
			}
		}
	}
	cout<<c<<" "<<ans<<endl;
}

int main() {
	int t,i,n,k;
	ll q;
	cin>>t;
	while(t--) {
		solve();
	}
	return 0;
}
