#include<bits/stdc++.h>
using namespace std;
char str1[101], str2[101], res[101];

int main()
    {
        long test, i;
        cin>>test;
        while(test--)
            {
                cin>>str1>>str2;
                for(i = 0; i<strlen(str1); i++)
                    {
                        res[i] = (str1[i]-'0')^(str2[i]-'0')+'0';
                    }
                  res[i] = '\0';
               cout<<res<<endl;
            }
       return 0;     
    }
