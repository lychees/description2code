#include <cstdio>
typedef long long ll;
using namespace std;

int main()
{
    ll t,n;
    scanf("%lld",&t);
    for(ll i=1;i<=t;i++)
    {
    	scanf("%lld",&n);
    	ll current=1;
    	ll diff=2;
    	do
    	{
    		if(n%2==0)
    			current=current;
    		else
    			current=current+diff;
    		n=n/2;
    		diff*=2;
    	}while(n!=1);
    	printf("%lld\n",current);
    }
    return 0;
}
