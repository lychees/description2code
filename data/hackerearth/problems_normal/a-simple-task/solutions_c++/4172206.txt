#include <iostream>
#include <map>
using namespace std;
typedef long long int ll;
int main()
{
    ll n;
    cin >> n;
    ll a[n];
    ll i;
    for(i = 0; i < n; i++) {
    	cin >> a[i];
    }
    ll f, j, mod1, mod2, t;
    map<ll, ll> m;
    map<ll, ll> :: iterator it;
    ll p = abs(a[0] - a[1]);
    for(i = 2; i * i <= p; i++) {
    	if(p%i == 0) {
    		ll mod1 = a[0]%i;
    		ll t = p/i;
    		ll mod2 = a[0]%t;
    		f = 1;
    		for(j = 2; j < n; j++) {
    			if(a[j]%i != mod1) {
    				f = 0;
    				break;
    			}
    		}
    		if(f) {
    			m[i]++;
    		}
    		f = 1;
    		for(j = 2; j < n; j++) {
    			if(a[j]%t != mod2) {
    				f = 0;
    				break;
    			}
    		}
    		if(f) {
    			m[t]++;
    		}
    	}
    }
    mod1 = a[0]%p;
    f = 1;
    for(i = 2; i < n; i++) {
    	if(a[i]%p != mod1) {
    		f = 0;
    		break;
    	}
    }
    if(f) {
    	m[p]++;
    }
    for(it = m.begin(); it != m.end(); it++) {
    	cout << it->first << " ";
    }
    cout << endl;
    return 0;
}
