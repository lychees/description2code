#include<iostream>
#include<string>
#define MAX 26

using namespace std;

int main(){
	int letter_count[MAX];
	string str;
	int t,len,i,weight;
	int n;
	unsigned int temp;
	char ignore;
	
	cin>>t;
	while(t--){
		cin>>str;
		weight=0;
		for(i=0;i<MAX;i++)
			letter_count[i]=0;
		len=str.length();
		for(i=0;i<len;i++){
			temp=str[i]-'a'+1;
			letter_count[temp-1]++;
			weight=weight+temp;
		}
			
		cin>>n;
		for(i=0;i<n;i++){
			cin>>ignore;
			temp=ignore-'a';
			weight=weight-((temp+1)*(letter_count[temp]));
		}
		cout<<weight<<endl;
	}
	return 0;
}
