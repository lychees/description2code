#include <iostream>
using namespace std;
typedef long long ll;
const int MAX_N = 30;
const int BASE = 10000;
ll f[MAX_N+1][MAX_N+1];
ll compute(int F, int H)
{
  if (f[F][H] > 0) {
    return f[F][H];
  }
  if (F == 0) {
    return f[F][H] = 1;
  }
  if (H == 0) {
    compute(F-1, H+1);
    return f[F][H] = compute(F-1, H+1);
  }
  return f[F][H] = compute(F-1, H+1) + compute(F, H-1);
}
int main()
{
  int tc;
	scanf("%d",&tc);
	while (tc--)
	{
		int n;
		cin >> n;
    cout << compute(n, 0) << endl;
  }
  return 0;
}
