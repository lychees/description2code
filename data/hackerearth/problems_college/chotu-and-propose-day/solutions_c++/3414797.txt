#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t,temp,p,q,r,s;
    string str;
    scanf("%d", &t);
    while(t--){
        scanf("%d %d %d %d", &p, &q, &r, &s);
        str = "";
        temp = p/2;
        while(temp--){
            str+= "aa";
        }
        temp = min(r,s);
        while(temp--){
            str+= "ab";
        }
        temp = q/2;
        while(temp--){
            str+="bb";
        }
        if(p%2==1){
            str+= "a";
        }
        else if(q%2==1){
            str+= "b";
        }
        cout<<str;
        for(int i = str.length()-1; i>=0; i--){
            cout<<str[i];
        }
        printf("\n");
    }
    return 0;
}
