#include<bits/stdc++.h>
#include<math.h>
using namespace std;

int main()
{
	int T, N;
	cin >> T;
	while(T--)
	{
	    cin>>N;
	long long int arr[N], i= 0,j=0, num = pow(2, N);
		long long int curr_xor = 0, curr_and = 1, ans = INT_MIN, temp = ans, x  =  0 , y = 0, len1 = 0, temp_len= 0, newnum = 0;
		for(i = 0; i < N ; i++)
		{
			cin >> arr[i];
		}
		for (i = 1; i < num ; i++)
		{
		    curr_xor = 0;
		    curr_and = 0;
			temp_len= 0;
			int k = 0;
			string s="";
			for (int ii = (1<<(N-1)); ii > 0; ii = ii / 2)
				(i & ii)? s+="1": s+="0";
			for(j = 0 ; j < s.length() ; j++)
				if( s[j] == '1')
				{
					curr_and = arr[j];
					break;
				}
			for(j = 0 ; j < s.length() ; j++)
			{
				if( s[j] == '1')
				{
					curr_xor = curr_xor ^ arr[j];
					curr_and = curr_and & arr[j];
					temp_len += 1;
					//cout << arr[j] <<" " <<temp_len << endl;
				}
			}
			temp = (curr_xor | curr_and);
			if(ans < temp )
				len1 = temp_len;
			else if(ans==temp)
                len1=max(len1,temp_len);
            ans = max(ans,temp);
		}
		cout << ans << " " << len1 << endl;
	}
}
