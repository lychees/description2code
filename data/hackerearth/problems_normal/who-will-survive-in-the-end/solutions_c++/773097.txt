#include <iostream>
using namespace std;

int f(int start,int step,int n,int rem)
{
	if(n==1)
	return start;
	start=start+step*rem;
	step*=2;
	int next_n=n/2;
	if(n%2==1)
	{
		next_n=next_n+1-rem;
	}
	rem=rem^(n%2);
	return f(start,step,next_n,rem);
}

int main()
{
    int t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	int ans=f(1,1,n,0);
    	cout<<ans<<endl;
    }
    return 0;
}
