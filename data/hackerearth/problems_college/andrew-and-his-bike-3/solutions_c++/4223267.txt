#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
typedef vector<ll> vll;
typedef pair<ll,ll> pll;
#define MOD 1000000007
#define pb push_back
#define endl '\n'
#define f first
#define s second
#define forn(i,n) for(int i = 0; i < int(n); i++)
vll graph[100005];
int ans[100005];
bool visit[100005];
int dfs(int v)
{
	visit[v] = 1;
	int sum = 0;
	for(int i = 0; i < graph[v].size(); i++)
		if(visit[graph[v][i]] == 1) sum++;
		else sum += dfs(graph[v][i]);
	ans[v] = sum;
	return sum;
}
int main()
{
	ios_base::sync_with_stdio(false);
	int n;
	cin >> n;
	for(int i = 0; i < n-1; i++)
	{
		ll u,v;
		cin >> u >> v;
		u--,v--;
		graph[u].pb(v);
		graph[v].pb(u);
	}
	dfs(0);
	for(int i = 0; i < n; i++) cout << ans[i] << " ";
	return 0;
}
