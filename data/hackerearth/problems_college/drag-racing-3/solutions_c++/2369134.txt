#include<bits/stdc++.h>
using namespace std;

int main()
{
	int n;
	cin >> n;
	int i = 0;
	string w,l;
	map< string,int > m;
	int race = pow(2,n)-1;
	for(i=0;i<race;i++)
	{
		cin >> w >> l;
		m[w]++;
	}
	string winner;
	map<string,int> :: iterator it=m.begin();
	winner=it->first;
	int count=it->second;
	++it;
	while(it!=m.end())
	{
		if(it->second > count)
		{
			count = it->second;
			winner = it->first;
		}
		++it;
	}
	cout << winner << endl;
	return 0;
}
