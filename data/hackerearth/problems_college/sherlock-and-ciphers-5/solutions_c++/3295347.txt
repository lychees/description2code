#include<iostream>
#include<string.h>
#include<math.h>
using namespace std;
int main()
{
	int sentence;
	char arr[120][200];
	cin>>sentence;
	for( int i = 0; i<=sentence; i++)
		cin.getline(arr[i],200);
	char max[15][15];
	for( int i = 1 ; i<= sentence; i++)
	{
		int l=strlen(arr[i]);
		int s = sqrt(l);
		int m=0;
		for( int j= 0 ; j < s ; j++ )
			for( int k = 0; k < s ; k++ )
			{
				max[k][j] = arr[i][m];
				m++;
			}
		for( int j= 0 ; j < s ; j++ )
			for( int k = 0; k < s ; k++ )
				cout<<max[j][k];
		cout<<"\n";
	}
	return 0;	
}
