#include <iostream>
#include <algorithm>
#include <map>
using namespace std;

int main()
{
   int n;
   cin>>n;
   int a[1000];
   int b[1000];
   for(int i=0;i<n;i++){
         cin>>a[i];
         b[i]=a[i];
   }
      
    sort(a,a+n);
    
    map<int,int> mymap;
    map<int,int>::iterator it;
    int rnk = 1;
    
     for(int i=n-1;i>=0;i--)
     {
     	
     	if(i==n-1)
     	  {
     		mymap[a[i]]=1;
     	  }
     	 else if(a[i]!=a[i+1])
     	  {
     	  		rnk = n-i;
     	  		mymap[a[i]]=n-i;
     	  }
     	 else if(a[i]==a[i+1])
     	  {
     	  		mymap[a[i]]=rnk;
     	  }
     	   
     }
     
     for(int i=0;i<n;i++)
     {
     	 it = mymap.find(b[i]);
     	 
        if (it != mymap.end())
        {
        	cout<<it->second<<" ";
        }
      
     }
     
}
