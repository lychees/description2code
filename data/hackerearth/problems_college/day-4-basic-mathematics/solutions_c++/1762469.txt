#include <iostream>
using namespace std;

int main()
{
    int t,i;
    long num;
    cin>>t;
    while(t--)
    {
    	cin>>num;
        i=1;
        while(num>0)
        {
        	num=num-i;
        	i++;
        }
        if((i-1)%3==0)
            cout<<"SL\n";
        else if((i-1)%3==1)
            cout<<"LB\n";
        else
            cout<<"BS\n";
    }
    return 0;
}
