#include <cmath>
#include <iostream>
using namespace std;
int arr[100][100],col[100][100];

int colour(int x,int y,int n,int m, int d, int c){
	col[x][y]=c;
	int count=1;
	//cout<<"query "<<c<<" visited "<<x<<" "<<y<<endl;
	if(y>0&&col[x][y-1]<c&&abs(arr[x][y-1]-arr[x][y])<=d)
		count+=colour(x,y-1,n,m,d,c);
	if(x<n-1&&col[x+1][y]<c&&abs(arr[x+1][y]-arr[x][y])<=d)
		count+=colour(x+1,y,n,m,d,c);
	if(y<m-1&&col[x][y+1]<c&&abs(arr[x][y+1]-arr[x][y])<=d)
		count+=colour(x,y+1,n,m,d,c);
	if(x>0&&col[x-1][y]<c&&abs(arr[x-1][y]-arr[x][y])<=d)
		count+=colour(x-1,y,n,m,d,c);
	return count;
}
	
int main()
{
    int n,m,q;
    cin>>n>>m>>q;
    for(int i=0;i<n;i++)
    	for(int j=0;j<m;j++){
    		cin>>arr[i][j];
    		col[i][j]=0;
    	}
    for(int i=1;i<=q;i++){
    	int x,y,d;
    	cin>>x>>y>>d;
    	cout<<colour(x-1,y-1,n,m,d,i)<<"\n";
    }
    return 0;
}
