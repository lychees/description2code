inline ll scan(){
    char c = getchar_unlocked();
    register ll x = 0;
    bool neg=false;
    if(c=='-'){
        neg=true;
    }
    while(c<'0'||c>'9'){
        c=getchar_unlocked();
    }
    while(c>='0'&&c<='9'){
        x=(x<<1)+(x<<3)+c-'0';
        c=getchar_unlocked();
    }
    if(neg==false) return x;
    else if(neg==true) return -1*x;
}

ll dp[1234],pref[1234],n;

ll solve(ll lim){
	ll i,j,k,p,q=1LL;
	reset(dp,0);
	for(i=1;i<=n;i++){
		dp[pref[i]]++;
	}
	p=0;
	for(i=1;i<=lim;i++){
		p+=dp[i];
		if(p>i) return 0LL;
	}
	p=0;
	for(i=1;i<=lim;i++){
		while(dp[i]){
			q=(q*(i-p))%modu;p++;dp[i]--;
		}
	}
	return q;
}

int main(){
	ll t=scan();
	while(t--){
		register  ll i,j,k=0,p,q,x,y,ans=1;
		n=scan();
		for(i=1;i<=n;i++){
			pref[i]=scan();
			k=max(k,pref[i]);
		}
		printf("%lld\n",solve(k));
	}
	return 0;
}
