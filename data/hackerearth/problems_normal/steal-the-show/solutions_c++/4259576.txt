#include<bits/stdc++.h>
using namespace std;

int main()
	{
		int CEO,COO,CTO;
		cin>>CEO>>COO>>CTO;
		int members,x;
		cin>>members;
		vector<int>v;
		for(int i=0;i<members;i++)
			{
				cin>>x;
				v.push_back(x);
			}
		sort(v.begin(),v.end());
		int a[20]={0};
		int mini=min(COO,CTO);
		if(mini==COO)
			{
				a[0]=COO;
				a[members+2]=CTO;
			}
		else
			{
				a[0]=CTO;
				a[members+2]=COO;
			}
		int total=members+3;
		if(total%2!=0)
			{	
				int index=total/2;
				a[index]=CEO;
				int j=1;
				for(int i=0;i<v.size();)
					{
						if(j!=index)
							{
								a[j]=v[i];
								i++;
							}
						j++;
					}
			}
		else
			{
				int index=members/2;
				if(CEO<=v[index])
					{
						int in=(total/2)-1;
						a[in]=CEO;
						int j=1;
						for(int i=0;i<v.size();)
							{
								if(j!=in)
									{
										a[j]=v[i];
										i++;
									}
								j++;
							}
					}
				else
					{
						int in=(total/2);
						a[in]=CEO;
						int j=1;
						for(int i=0;i<v.size();)
							{
								if(j!=in)
									{
										a[j]=v[i];
										i++;
									}
								j++;
							}
					}
			}
		//for(int i=0;i<total;i++)
		//	cout<<a[i]<<" ";
		//cout<<endl;
		int cum_sum=0;
		for(int i=1;i<total;i++)
			{
				cum_sum=cum_sum+abs(a[i]-a[i-1]);
			}
		cout<<cum_sum<<endl;
		return 0;
	}
