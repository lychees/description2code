#include<bits/stdc++.h>
using namespace std;

const int N=1010;
const int inf=1e9+9;

int minVal;
int dp[N][N];
int a[N][N];
int top[N];
int bottom[N];

int main() {
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++) {
        for(int j=0;j<n;j++) {
            scanf("%d",&a[i][j]);
            dp[i][j]=inf;
        }
    }
    for(int i=0;i<n;i++) {
        dp[i][n-1]=a[i][n-1];
    }
    for(int j=n-2;j>=0;j--) {
        for(int i=0;i<n;i++) {
            dp[i][j]=min(dp[i][j],dp[i][j+1]+a[i][j]);
        }
        for(int i=0;i<n;i++) {
            top[i]=dp[i][j];
            bottom[i]=dp[i][j];
        }
        for(int i=1;i<n;i++) {
            top[i]=min(top[i],top[i-1]+a[i][j]);
        }
        for(int i=n-2;i>=0;i--) {
            bottom[i]=min(bottom[i],bottom[i+1]+a[i][j]);
        }
        for(int i=0;i<n;i++) {
            dp[i][j]=min(top[i],bottom[i]);
        }
    }
    minVal=inf;
    for(int i=0;i<n;i++) {
        minVal=min(minVal,dp[i][0]);
    }
    printf("%d\012",minVal);
    return 0;
}
