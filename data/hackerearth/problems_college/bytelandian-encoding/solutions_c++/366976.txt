using namespace std;
#include<iostream>
int main()
{
    int t; cin>>t;
    while(t>0)
    {
        string s; cin>>s; string s1="";
        int d=s.length();
        if(d%2==0)
        {
            int j=d/2; int i=j-1;
            while(i>=0)
            {
                int a=s[i]; int b=s[j]; if(a<=b) s1=s1+s[i]+s[j]; else s1=s1+s[j]+s[i]; i--; j++;
            }

        }
        else
        {
            int i=(d-1)/2; s1=s1+s[i]; int j=i+1; i= i-1;
             while(i>=0)
            {
                int a=s[i]; int b=s[j]; if(a<=b) s1=s1+s[i]+s[j]; else s1=s1+s[j]+s[i]; i--; j++;
            }
        }
        cout<<s1<<endl;
        t--;

    }
}
