#include <iostream>
#include <algorithm>
using namespace std;
 
int main()
{
   ios::sync_with_stdio(0);
   int t,n,a,b,c,i,j,k;long long x;
   cin >> t;
   while(t--)
   {
   		x=0;
   		cin >> n >> a >> b >> c;
   		for(i=0;i<=min(a,n);++i)
   		{
   			for(j=0;j<=min(n-i,b);++j)
   			{
   				for(k=0;k<=min(n-i-j,c);++k)
   				{
   					x++;
   					
   				}
   			}
   		}
   		cout << x << endl;
   }
   return 0;
}
