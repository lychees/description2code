#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n, ans = 0;
	cin >> n;
	vector<long> t(3);
	map<vector<long>, int> freq;
	
	for (int i = 0; i < n; i++)
	{
		cin >> t[0] >> t[1] >> t[2];
		sort(t.begin(), t.end());
		freq[t]++;
	}

	for (auto i = freq.begin(); i != freq.end(); i++)
	{
		if (i->second == 1L)
		{
			ans++;
		}
	}
	
	cout << ans << endl;
	system("pause");
	return 0;
}
