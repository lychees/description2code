#include <bits/stdc++.h>

#define debug(x) cout << #x << " = " << x << endl
#define fori(i, ini, lim) for(int i = int(ini); i < int(lim); i++)
#define ford(i, ini, lim) for(int i = int(ini); i >= int(lim); i--)

using namespace std;

typedef long long ll;
typedef long double ld;
typedef pair<int, int> ii;

const int bits = 20;
int A[1 << bits], F[1 << bits];

int main() {
	int t;
	scanf("%d", &t);
	while(t--) {
		int N;
		scanf("%d", &N);
		memset(A, 0, sizeof A);
		vector<int> arr;
		fori(i, 1, N + 1) {
			int cur;
			scanf("%d", &cur);
			A[cur]++;
			arr.push_back(cur);
		}
		fori(i, 0, (1 << bits)) {
			F[i] = A[i];
		}
		fori(i, 0, bits) {
			fori(mask, 0, 1 << bits) {
				if(mask & (1 << i)) {
					F[mask] += F[mask & ~(1 << i)];
				}
			}
		}
		ll ans = 0;
		for(auto &each : arr) {
			ans += F[~each & ((1 << bits) - 1)];
		}
		printf("%lld\n", ans);
	}
	return 0;
}
