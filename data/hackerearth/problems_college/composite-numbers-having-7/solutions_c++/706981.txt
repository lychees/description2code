#include <iostream>
using namespace std;
#define sz 1000005
int main()
{
    //cout << "Hello World!" << endl;
    int s[1000005],pr[10001],i,j,t;
    for(i=0;i<sz;i++)
    s[i]=0;
    
    s[0]=1;
    s[1]=1;
    for(i=2;i*i<sz;i++)
    {
    	if(s[i]==0)
    	for(j=2*i;j<sz;j+=i)
    	s[j]=1;
    }
    
    cin>>t;
    while(t--)
    {
    	int n,m,ans=0,p;
    	cin>>n>>m;
    
    	
    	for(i=n;i<=m;i++)
    	{
    		if(s[i]!=0)
    		{
    			p=i;
    			while(p)
    			{
    				if(p%10==7)
    				{ans++;
    			 	break;
    				}
    				p=p/10;
    				
    			}
    		}
    	}
    	if(ans==0)
    	ans=-1;
    	cout<<ans<<"\n";
    }
    return 0;
}
