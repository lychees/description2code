#include<bits/stdc++.h>
using namespace std;
int ar[111],br[111];
int main()
{
    int bb,i,t,n,j;
    cin>>t;
    while(t--)
    {
        cin>>n;
        cin>>bb;
        for(i=0;i<n-1;i++)
            cin>>ar[i];
        for(i=0;i<n;i++)
            cin>>br[i];
        sort(ar,ar+n-1);
        sort(br,br+n);
        bb=bb*br[n-1];
        bool cond=true;
        for(i=0,j=n-2;i<n-1;i++,j--)
            if(bb<=ar[i]*br[j])
            cond=false;
        (cond)?puts("YES"):puts("NO");
    }
}
