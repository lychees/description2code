#include<iostream>
#include<stdio.h>
#include<vector>
#include<algorithm>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    for(int i=0;i<t;i++)
    {
            int n;
            scanf("%d",&n);
            int ans=1;
            for(int k=0;k<=10000;k++)
            {
                    if(n<=ans)
                    {   cout<<k<<"\n";
                        k=100000;
                    }
                    ans*=3;
            }
            
    }
    return 0;
    
}
