#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int a[10000005];
vector <int> v[10];
int maxa;
void sieve()
{
	int i,j;
	maxa=0;
	for(i=2;i<=10000000;i++)
	{
		if(!a[i])
		{
			for(j=i*2;j<=10000000;j+=i)
			{
				a[j]++;
			}
		}
	}
	for(i=2;i<10000005;i++)
	{
		if(!a[i])
		{
			a[i]=1;
		}
	}
	for(i=2;i<=10000000;i++)
	{
		maxa=max(maxa,a[i]);
		v[a[i]].push_back(i);
	}
}
int main()
{
	sieve();
    int t,a,b,c,k;
    scanf("%d",&t);
    //cout<<maxa<<endl;
    while(t--)
    {
    	scanf("%d%d%d",&a,&b,&k);
    	if(k>=9)
    	{
    		printf("0\n");
    		continue;
    	}
    	c=(upper_bound(v[k].begin(),v[k].end(),b)-v[k].begin())-(lower_bound(v[k].begin(),v[k].end(),a)-v[k].begin());
    	printf("%d\n",c);
    }
    return 0;
}
