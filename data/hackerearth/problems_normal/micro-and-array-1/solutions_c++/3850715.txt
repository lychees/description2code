#include <iostream>
using namespace std;

int main()
{
	int N,Q,i,a1,b1,c1,d1,loc,k,flag;
	cin>>N;
	cin>>Q;
	int a[N];
	for(i=0;i<N;i++)
	cin>>a[i];
	for(i=0;i<Q;i++)
	{
		flag=0;
		cin>>a1;
		if(a1==1)
		{
			cin>>b1;
			for(k=0;k<N;k++)
			{
				if(a[k]>=b1)
				{
					loc=k;
					flag=1;
					//cout<<loc+1<<endl;
					break;
				}
				
			}
			if(flag==1)
			cout<<loc+1<<endl;
			else
			cout<<"-1"<<endl;
		}
		else
		{
			cin>>c1;
			cin>>d1;
			a[c1-1]=d1;
		}
	}
    //cout << "Hello World!" << endl;
    return 0;
}
