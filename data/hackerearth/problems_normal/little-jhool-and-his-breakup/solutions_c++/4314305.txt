#include <iostream>
using namespace std;

int main()
{
	ios::sync_with_stdio(false);
    string s, s1 = "love";
    cin >> s;
    int p = 0, l = s.length();
    for(int i = 0; i < l; i++) {
    	if(s[i] == s1[p])
    		p++;
    	if(p == 4)
    		break;
    }
    if(p == 4)
    	cout << "I love you, too!" << endl;
    else
    	cout << "Let us breakup!" << endl;
    return 0;
}
