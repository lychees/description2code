#include <bits/stdc++.h>
using namespace std;

int a[105][105];
int fun(int cg,int N) {
	int dp[N+1][N+1];
	memset(dp,0,sizeof(dp));
	for(int i =1; i<=N;i++){
		for(int j=1;j<=N;j++){
			dp[i][j]=max(dp[i-1][j],dp[i][j-1])+a[i][j];
		}
	}
	if((dp[N][N]-cg) < 0)return -1;
	else
		return dp[N][N]-cg;
}
int main()
{
    int t;
    cin >> t;
    while(t--) {
    	int N,c;
    	cin >> N >> c;
    	memset(a,0,sizeof(a));
    	for (int i = 1; i <= N ; i++)
   		 	for (int j = 1; j <= N ; j++)
    			cin >> a[i][j];
    	cout << fun(c,N) << endl;
    }
    return 0;
}
