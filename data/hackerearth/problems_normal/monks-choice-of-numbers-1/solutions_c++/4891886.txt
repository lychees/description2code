#include <cstdio>
#include <vector>
#include <algorithm>
using namespace std;

// Snippet: bitCount
int bitCount(int n) {
	n = (n & 0x55555555) + ((n >> 1) & 0x55555555);
	n = (n & 0x33333333) + ((n >> 2) & 0x33333333);
	n = (n & 0x0f0f0f0f) + ((n >> 4) & 0x0f0f0f0f);
	n = (n & 0x00ff00ff) + ((n >> 8) & 0x00ff00ff);
	return (n & 0xffff) + (n >> 16);
}

int main() {
    int t;
    scanf("%d", &t);
    while (t--) {
        int n, k, ans = 0;
        scanf("%d%d", &n, &k);
        vector<int> a(n);
        for (int i = 0; i < n; ++i) {
            scanf("%d", &a[i]);
            a[i] = bitCount(a[i]);
        }
        sort(a.begin(), a.end());
        for (int i = n - 1; i >= n - k; --i) ans += a[i];
        printf("%d\n", ans);
    }
    return 0;
}
