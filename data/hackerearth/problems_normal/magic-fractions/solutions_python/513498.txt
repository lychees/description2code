pp = [1]*509
def fun(n):
	if n <4:
		return n-1
	res = 1
	for i in range (2, n+1):
		if pp[i]> 0:
			res = res*2
	res = res/2
	return res

x = int(input(""))
#print x
pp[1] = 0
for i in range(2, 505):
	pp[i] = 1
	for j in range(2, i):
		if i%j == 0:
			pp[i] = 0
#	print pp[i]
ans = 0		
for i in range(1, x+1):
	ans = ans+fun(i)
