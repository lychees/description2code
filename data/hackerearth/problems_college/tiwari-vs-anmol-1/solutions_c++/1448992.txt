#include <bits/stdc++.h>

using namespace std;

#define  MAX 1000001
#define MOD 1000000007

typedef long long ll;

int dat[MAX];

void init()
{
    for(int i=4;i<MAX;i+=2)
        dat[i]=2;
    for(int i=3;i<=sqrt(MAX);i+=2)
    {
        if(!dat[i])
        {
            for(int j=2*i;j<MAX;j+=i)
            {
                if(!dat[j])   
                    dat[j]=i;
            }
        }
    }
    for(int i=1;i<MAX;i++)
    {   
        dat[i]=(dat[i]+dat[i-1])%MOD;
    }
}

int main()
{
    ios_base::sync_with_stdio(0);
    int t, l, r;
    ll sum;
    init();
    cin>>t;
    while(t--)
    {   
        sum=0;
        cin>>l>>r;
        /*for(int i=l;i<=r;i++)
        {
            //cout<<i<<" : "<<dat[i]<<endl;
            sum=(sum+dat[i])%MOD;
        }*/
        cout<<((dat[r]-dat[l-1])+MOD)%MOD<<endl;
    }
    return 0;
}
