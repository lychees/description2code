using namespace std;


#define LL long long int
#define sc1(x) scanf("%d", &x)
#define sc2(x, y) scanf("%d%d", &x, &y)
#define sc3(x, y, z) scanf("%d%d%d", &x, &y, &z)
#define pr1(x) printf("%d %d\n", x.first, x.second)
#define FOR(i, a, n) for(int i=a;i<n;i++)
#define pp pair<int, int>

#define MAX 10001
#define MOD 1000000007LL

int arr[MAX], sortA[MAX];


void reverse(int left, int right) {
	while(left < right) {
		int temp = arr[left];
		arr[left] = arr[right];
		arr[right] = temp;

		left++; right--;
	}
}


pp func(int n) {
	int startS, endS;

	startS = endS = -1;
	FOR(i, 0, n) {
		if(arr[i] != sortA[i]) {
			if(startS == -1) startS = i;
			endS = i;
		}
	}

	if(startS == -1) {
		return {-1, -1};
	}

	reverse(startS, endS);

	FOR(i, 0, n) {
		if(arr[i] != sortA[i]) {
			return {-1, -1};
		}
	}

	return {startS+1, endS+1};
}


int main() {

	int t, n;

	sc1(t);
	while(t--) {
		sc1(n);

		FOR(i, 0, n) {
			sc1(arr[i]);
			sortA[i] = arr[i];
		}

		sort(sortA, sortA+n);

		pp result = func(n);
		pr1(result);
	}
	return 0;
}
