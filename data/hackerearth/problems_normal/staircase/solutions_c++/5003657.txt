#include <bits/stdc++.h>
using namespace std;

int main()
{
   int n;cin>>n;
    int arr1[n],arr2[n];
    arr1[0]=1,arr1[1]=2;
    arr2[0]=1,arr2[1]=2,arr2[2]=4;
    for(int i=2;i<n;i++) arr1[i]=arr1[i-1]+arr1[i-2];
    for(int i=3;i<n;i++) arr2[i]=arr2[i-1]+arr2[i-2]+arr2[i-3];
    if(arr2[n-1]>100000||arr1[n-1]>100000) cout<<"ERROR"<<endl;
    else{
    	cout<<"Jack-"<<arr2[n-1]<<endl;
    	cout<<"Jill-"<<arr1[n-1]<<endl;
    }
    return 0;
}
