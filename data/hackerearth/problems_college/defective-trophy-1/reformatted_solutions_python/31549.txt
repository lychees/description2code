from __future__ import division
i = int(raw_input())

k = []
for j in range(i):
    m = int(raw_input())
    k.append(m)


for s in k:
    if s != 1 and s != 0:
        a = 0
        l = s/3

        while l > 1:
            a = a+1
            l = l/3

        a = a+1
        print a

    else:
        a = 0
        print a
