#include <cstdio>

typedef long long LL;

LL reverse(LL n)
{
	LL res = 0;
	
	while (n != 0)
	{
		res = res * 10 + n % 10;
		n /= 10;
	}
	
	return res;
}

int main()
{
	LL n, a, b;
	scanf("%lld", &n);
	
	while (n--)
	{
		scanf("%lld%lld", &a, &b);
		printf("%lld\n", reverse(reverse(a) + reverse(b)));
	}
	
	return 0;
}
