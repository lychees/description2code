#include<bits/stdc++.h>
using namespace std;
#define lli long long int
#define f(i,a,b) for(i=a;i<b;i++)
#define si(a) scanf("%d",&a);
#define slli(a) scanf("%lld",&a);
#define MOD 1000000007
#define MAX 100005
#define pb(a) push_back(a)
int main()
{
	lli ans,i,j,k,m,n,p,t,b[2005];
	slli(n);
	slli(p);

	f(i,0,n)
	slli(b[i]);
    
    ans=0;

	f(i,0,n)
	{
		if(b[i]%p==0)
		{
			ans=ans+((n-i-1)*(n-i-2))/2;
			continue;
		}
		f(j,i+1,n)
		{
			if((b[i]*b[j])%p==0)
			{
				ans=ans+(n-j-1);
				continue;
			}
			f(k,j+1,n)
			{
				if((((b[i]*b[j])%p)*b[k])%p==0)
					ans++;
			}
		}
	}
	printf("%lld\n",ans);

}
