#include <iostream>
#include <algorithm>
#include <cstdio>
using namespace std;
#define LL long long
LL arr[100005];
int main() {
	int t;
	cin>>t;
	while(t--)
	{
		long long n;
		cin>>n;
		//long long co=1;
		LL maxi=1;
		for(int i=0;i<n;++i)cin>>arr[i];
		sort(arr,arr+n);
		int co=1;
		int ans=1;
		for(int i=1;i<n;++i)
		{
			if(arr[i]==arr[i-1])co++;
			else
			{
				ans=max(ans,co);
				co=1;
			}
		}
		ans=max(ans,co);
		cout<<ans<<endl;
	}
	
}
