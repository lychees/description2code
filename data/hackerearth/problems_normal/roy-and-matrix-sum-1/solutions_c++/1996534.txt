#include <iostream>
#include <cmath>
using namespace std;
#define LL long long
#define U unsigned

int main()
{
    int T;
    cin >> T;
    
    while(T--)
    {
    	U LL n,i,sum=0;
    	
    	cin >> n;
    	
    	for(i=1 ; i<n ; i++)
    	sum+=i*(n-i);
    	
    	cout << 2*sum << endl;
    }
    return 0;
}
