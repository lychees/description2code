#include <bits/stdc++.h>
using namespace std;
#define mod 1000000009
long long int fastpow(long long int a,long long int b)
{
long long int res=1;
while(b)
{
	if(b&1)
	res=(res*a)%mod;
	a=(a*a)%mod;
	b=b/2;
}
	return res%mod;
}	
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	long long int n;cin>>n;
    	n=(n+1)/2;
    //	cout<<n<<endl;
    	cout<<fastpow(26,n)<<endl;
    }
    return 0;
}
