#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long t;
    cin>>t;
    while(t--)
    {
    	int flag=0;
    	string str,str2;
    	cin>>str;
    	str2=str;
    	reverse(str.begin(),str.end());
    	for(int i=1;i<str.size();i++)
    	{
    		if(abs(str2[i]-str2[i-1])!=abs(str[i]-str[i-1]))
    		{
    			flag=1;
    			break;
    		}
    	}
    	if(flag)
    	cout<<"Not Funny"<<endl;
    	else
    	cout<<"Funny"<<endl;
    }
    return 0;
}
