#include<stdio.h>
#include<memory.h>
typedef long long int ll;
ll dp[100005]={0};
#define INF 1000000007
int main()
{
    //memset(dp,-1,sizeof(dp));
    ll n,k,a[105]={0},i,j,ans=0,sum=0,pow[105]={0};
    scanf("%lld%lld",&n,&k);
    for(i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
        sum+=a[i];
    }
    pow[0]=1;
    for(i=1;i<=n;i++)
        pow[i]=(2*pow[i-1])%INF;
    dp[0]=1;
    for(i=0;i<n;i++)
    {
        for(j=k-1;j-a[i]>=0;j--)
        {
            if(dp[j-a[i]])
            {
                dp[j]=(dp[j]+dp[j-a[i]])%INF;
            }
        }
    }
    bool visited[100005]={false};
    for(i=0;i<k;i++)
    {
        if(visited[i])
            continue;
        //printf("i=%lld dp=%lld\n",i,dp[i]);
        ans=(ans+2*dp[i])%INF;
        if(sum>=i && sum-i<k)
            visited[sum-i]=1;
        if(sum%2==0 && i==sum/2)
        	ans=(ans-dp[i]+INF)%INF;
    }
    printf("%lld\n",(pow[n]-ans+INF)%INF);
    return 0;
}
