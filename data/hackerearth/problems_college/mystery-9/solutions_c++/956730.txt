#include <iostream>

using namespace std;

int main()
{
    
    int t,n,arr[12]={31,28,31,30,31,30,31,31,30,31,30,31},i,mon;
    cin>>t;
    while(t--)
    {
        mon=1;
        cin>>n;
        
        for(i=0;i<12;i++)
        {
            if(n-arr[i]>0)
            {
                n=n-arr[i];
                mon++;
            }
            else
            {
                cout<<mon<<endl;
                break;
            }
            
        }
        
    }; 
   
   return 0;
}
