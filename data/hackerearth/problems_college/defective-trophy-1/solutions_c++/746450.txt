#include<iostream>
#include<math.h>
using namespace std;

int main()
{
    long t;
    int i;
    long long n;
    cin>>t;
    while(t--)
    {   i=0;
        cin>>n;
        while(1)
        {
          if(pow(3,i)>=n)
          {
              cout<<i<<endl;
              break;
          }
          i++;
        }
    }
}
