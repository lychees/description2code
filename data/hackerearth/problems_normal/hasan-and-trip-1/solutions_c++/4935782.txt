#include <bits/stdc++.h>

using namespace std;

double dp[3005];
int x[3005], y[3005], f[3005];

double dist(int i, int j)
{
	double d = sqrt((double)(pow(x[i]-x[j],2) + pow(y[i]-y[j],2)));
	return d;
}

int main()
{
	int n;
	cin >> n;
	for (int i = 0; i < n; i++) {
		cin >> x[i] >> y[i] >> f[i];
	}
	dp[0] = 1.0*f[0];
	for (int i = 1; i < n; i++) {
		dp[i] = -1e9;
		for (int j = 0; j < i; j++) {
			double D = dist(i, j);
			dp[i] = max(dp[i], dp[j] - D);
		}
		dp[i] += f[i];
	
	}
	
	
	printf("%lf\n", dp[n-1]);
	
	return 0;
}
