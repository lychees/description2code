#include <iostream>

using namespace std;

int main()
{
    int t;
    cin>>t;
    while (t--){
        int n,k=0;
        cin>>n;
        int a[n];
        for (int i=1;i<=n;i++)
            cin>>a[i];
         for (int i=1;i<=n;i++){
                if(a[a[i]] != i){
                    cout<<"not inverse"<<endl;
                    k=99;
                        break;
                }

         }
         if(k!=99)
            cout<<"inverse"<<endl;
    }
    return 0;
}
