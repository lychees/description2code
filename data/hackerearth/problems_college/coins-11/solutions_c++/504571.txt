#include<iostream>

using namespace std;

bool isPrime(int n)
{
	if(n%2 == 0)
		return false;
	else
	{
		for (int i = 3; i * i <= n; i += 2)
		{
			if(n%i == 0)
				return false;
		}
	}
	return true;
}

int main()
{
	int t, n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		if(n < 4)
			cout<<n<<endl;
		else if(isPrime(n))
			cout<<"2"<<endl;
		else
			cout<<"1"<<endl;
	}
	return 0;
}
