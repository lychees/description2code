# include <iostream>
# include <limits.h>
 
using namespace std;

int max(int a, int b) { return (a > b)? a: b; }
 
/* Driver program to test to pront printDups*/
int main()
{
	int t;
	cin>>t;
	int eggFloor[501][501];
    int res;
    int i, j, x;
    for (i = 1; i <= 500; i++)
    {
        eggFloor[i][1] = 1;
        eggFloor[i][0] = 0;
    }
 
    for (j = 1; j <= 500; j++)
        eggFloor[1][j] = j;
 
    for (i = 2; i <=500; i++)
    {
        for (j = 2; j <=500; j++)
        {
            eggFloor[i][j] = INT_MAX;
            for (x = 1; x <= j; x++)
            {
                res = 1 + max(eggFloor[i-1][x-1], eggFloor[i][j-x]);
                if (res < eggFloor[i][j])
                    eggFloor[i][j] = res;
            }
        }
    }
	while(t--)
	{
		int k,n;
		cin>>n>>k;
	    cout<<eggFloor[k][n]<<endl;
	}
    return 0;
}
