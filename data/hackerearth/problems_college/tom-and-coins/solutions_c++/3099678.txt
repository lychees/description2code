#include <iostream>
using namespace std;

int main()
{
    int test;
    cin>>test;
    for(int i=0;i<test;i++)
    {
    	int a,k,n,sum=0;
    	cin>>a>>k>>n;
    	for(int j=0;j<n;j++)
    	{
    		if(j==0)
    		{
    			sum=sum+a;
    		}
    		else
    		{
    			sum=sum+k;
    		}
    	}
    	cout<<sum<<endl;
    }
}
