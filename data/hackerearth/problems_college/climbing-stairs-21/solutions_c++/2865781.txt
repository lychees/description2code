#include <bits/stdc++.h>
using namespace std;

typedef long long lint;

const lint maxn = 1010;
lint ans[maxn];

int main(){
	
	lint t;
	
	scanf("%lld",&t);
	
	ans[0] = ans[1] = 1;
	
	for( lint i=2 ; i < maxn ; i++ )
		ans[i] = ans[i-1] + ans[i-2];
	
	while( t-- ){
		lint n;
		scanf("%lld",&n);
		printf("%lld\n",ans[n]);
	}
	
	return 0;
}
