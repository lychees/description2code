#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int t;
    cin>>t;
   // cout<<"t "<<t<<endl;
    int count=0;
    for(int i1=0;i1<t;i1++)
    {
    	string str;
    	cin>>str;
    	//cout<<str<<endl;
    	if(str.length()%2 == 0)
    	{ 
    		//cout<<"gracias"<<endl;
    		bool flag=true;
    		int visited[str.length()];
    		memset(visited,0,sizeof(visited));
    		for(int i=0;i<str.length();i++)
    		{
    			//cout<<i<<endl;
    			if(!visited[i])
    			{ 
    				//cout<<"Entering"<<endl;
    				visited[i]=1;
    				bool current=false;
    				for(int j=i+1;j<str.length();j+=2)
	    			{
	    				    
	    			     	if(!visited[j] && str[i] == str[j])
	    			     	{
	    			     		visited[j]=1;
	    			     		current=true;
	    			     		break;
	    			     	}
	    			}
	    			
	    			if(!current)
	    			{
	    				flag=false;
	    				break;
	    			}
	                
    			}
    		}
    		
    		if(flag)
    		{
    			count++;
    		}
    	}
    }
    
    cout<<count<<endl;
    
    
    
    return 0;

}
