//#include "stdafx.h"
#include <iostream>
#include <string.h>
#include <algorithm>
using namespace std;

int matrix[101][101];
int OutMatrix[101][101];
int N, C;


int Traverse(int i, int j)
{
	if(i < 0 || j < 0 || i >= N || j >= N)
		return 0;

	if(OutMatrix[i][j] != -1)
	{
		return OutMatrix[i][j];
	}

	int sum1 = matrix[i][j] + Traverse(i+1, j);
	int sum2 = matrix[i][j] + Traverse(i, j+1);
	
	OutMatrix[i][j] = max(sum1, sum2);
	return OutMatrix[i][j] ;	
}

int main()
{
	//freopen("DPMaze.txt", "r", stdin);

	int test_case;
	cin >> test_case;

	for (int tc = 1; tc <= test_case; ++tc)
	{
		memset(matrix, 0, sizeof(matrix));
		memset(OutMatrix, -1, sizeof(OutMatrix));

		cin >> N >> C;

		for (int i = 0; i < N; ++i)
		{
			for (int j = 0; j < N; ++j)
			{
				cin >> matrix[i][j];
			}
		}

		int out = Traverse(0, 0);

		if(out >= C)
		{
			cout << out - C << endl;
		}
		else
		{
			cout << -1 << endl;
		}

	}
	//int result = CheckMatrixPattern();

	return 0;
}
