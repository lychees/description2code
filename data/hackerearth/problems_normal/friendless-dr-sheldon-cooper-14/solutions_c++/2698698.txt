#include <iostream>
#include<cstdio>
#include<cstdlib>
#include<vector>
#include<algorithm>
#include<list>
using namespace std;

int main()
{
  int t;cin>>t; int N,M,v1,v2;
  while(t>0)
  {
  	cin>>N>>M;
  	vector<list<int>> graph(N+1);
  	for(int i=0;i<M;i++)
  	{
  		cin>>v1>>v2;
  		graph[v1].push_back(v2);
  		graph[v2].push_back(v1);
  	}
  	vector<int> visited(N+1,-1);
  	visited[0]=0; 
  		cout<<N-1<<endl;
  	t--;
  }
  
  	return 0;
}
