#include <vector>
#include <map>
#include <set>
#include <algorithm>
#include <iostream>
#include <cstdio>
#include <cmath>
#include <cstdlib>
#include <string>
#include <cstring>
using namespace std;
int a[105];
int main(){
	int n;
	cin >> n;
	int flag = 0;
	long long ans = 1;
	for (int i = 0; i < n; i++)
	{
		cin >> a[i];
		if (a[i] == 1) flag = 1; 
	}
	if (flag == 0) ans = 0;
	else
	{
		int poi = -1, poj = -1;
		for (int i = 0; i < n; i++)
		{
			if (a[i] == 1) { poi = i;  break; } 
		}
		for (int j = n - 1; j >= 0; j--)
		{
			if (a[j] == 1) { poj = j;  break; }
		}
		if (poi != -1 && poj != -1)
		{
			int zero = 0;
			for (int i = poi + 1; i <= poj; i++)  
			{
				if (a[i] == 0)  zero++;
				else
				{
					ans *= zero + 1; 
					zero = 0;
				}
			}
		}
	}
	cout << ans << endl;
	return 0;
}
