

    #include <iostream>
    using namespace std;
     
    int a1,e1,i1,o1,u1,num;
     
    int main()
    {
    	a1=e1=i1=o1=u1=0;
        string str;
        cin>>num;
        cin>>str;
        for(int i=0;i<num;i++)
        {
        	if(str[i]=='a')
        	{
        		a1++;
        	}
        	else if(str[i]=='e')
        	{
        		e1++;
        	}
        	else if(str[i]=='i')
        	{
        		i1++;
        	}
        	else if(str[i]=='o')
        	{
        		o1++;
        	}
        	else if(str[i]=='u')
        	{
        		u1++;
        	}
        	if(a1>0&&e1>0&&i1>0&&o1>0&&u1>0)
        	{
        		break;
        	}
        }
            if(a1>0&&e1>0&&i1>0&&o1>0&&u1>0)
        	{
        		cout<<"YES";
        	}
        	else
        	{
        		cout<<"NO";
        	}
    }
