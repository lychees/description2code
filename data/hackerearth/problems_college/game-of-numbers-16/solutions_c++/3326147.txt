#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    int a[t],sum=0;
    for(int i=0;i<t;i++)
    cin>>a[i];
    for(int k=0;k<t;k++)
    { sum=0;
       for(int i=1;i<=a[k];i++)
    	for(int j=1;j<i;j++)
    	if((i*j)%(i+j)==0)
    	sum++;
    	cout<<sum<<endl;
    }
    
    return 0;
}
