#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
   
    for(int i=0;i<t;i++)
    {
    	long long a,b,sum=0,flag=0;
    	cin>>a>>b;
    	long long check=a;
    	
    	while(a!=b)
    	{
    		if(a>b)
    		{
    			check=a;
    			a=a/2;
    		}
    		else
    		{
    			check=b;
    			b=b/2;
    		}
    		
    		flag=0;
    		
    		if(check==2)
    			sum=sum-2;
    		else 
    		{
    			if(check%2==0)
    				flag=1;
    				
    			for(int j=3;j<=sqrt(check) && flag==0;j=j+2)
  				{
      				if(check%j==0)
      				{
          				flag=1;
      				}
  				}
  			
  				if (flag==0)
      				sum=sum-check;
  		   	    else
  					sum=sum+check;
    		}
    		check=a;
  		
    	}
    	
    	flag=0;
    	
    	if(check==1)
    		sum=sum+1;
    	else if(check==2)
    		sum=sum-2;
    	else
    	{
    		if(check%2==0)
    			flag=1;
    		for(int j=3;j<=sqrt(check) && flag==0;j=j+2)
  				{
      				if(check%j==0)
      				{
          				flag=1;
      				}
  				}
  			
  				if (flag==0)
      				sum=sum-check;
  		   	    else
  					sum=sum+check;
    	}
    	cout<<sum<<endl;
    }
    
    
    	  
   return 0;
}
