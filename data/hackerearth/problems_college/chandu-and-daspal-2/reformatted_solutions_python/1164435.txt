from fractions import gcd
for _ in xrange(input()):
    x, y, a = raw_input().split()
    if a == "Chandu":
        b = "Daspal"
    else:
        b = "Chandu"

    x = int(x)
    y = int(y)
    steps = 0
    while x > 0 and y > 0:
        t = gcd(x, y)
        x = x-t
        y = y-t
        steps += 1

    if steps % 2 == 0:
        print b
    else:
        print a
