#include<iostream>
using namespace std;
int main()
{
	long int n;
	cin>>n;
	long int arr[100001][3]={0},i;
	for(i=1;i<=n;i++)
	cin>>arr[i][1]>>arr[i][2];
	long int h,a;
	long int brr[100001]={0};
	for(i=1;i<=n;i++)
	{
		brr[arr[i][1]]=brr[arr[i][1]]+1;
	}
	
	for(i=1;i<=n;i++)
	{
		h=n-1;
		a=n-1;
		h=h+brr[arr[i][2]];
		a=a-brr[arr[i][2]];
		cout<<h<<" "<<a<<endl;
	}
	
	return 0;
}
