
#include<bits/stdc++.h>
using namespace std;

int main()
{
	set<string> s;
	string str;
	unsigned int t,i,x,y,j;
    cin>>t;
    for(i=0;i<t;i++)
    {
    	cin>>x>>y;
    	for(j=0;j<x;j++)
    	{
    	cin>>str;
    	s.insert(str);
    	}
    	if(s.size()==x)
    	cout<<"Yes"<<endl;
    	else
    	cout<<"No"<<endl;
    	s.clear();
    }
    return 0;
}
