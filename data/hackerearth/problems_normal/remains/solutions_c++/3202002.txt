#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll euc(int x, int y, int n) {
	assert(x >= 0 && y >= 0 && n >= 0);
	if(x == 0 && y == 0) return 0;
	if(n == 0) return x+y;
	if(n <= 20 || x <= 4LL * y) return x + euc(y, abs(x-y), n-1);
	int steps;
	if(y == 0) steps = n;
	else steps = min(n, (x / y) / 2 * 3);
	steps -= 3;
	while(steps % 3 > 0) --steps;
	assert(steps > 0);
	int k = steps / 3;
	int a = x;
	int b = x - (k-1) * 2 * y;
	assert(b >= 0);
	return euc(x - k * 2 * y, y, n-steps) + (long long) (a + b) * k;
}
void te() {
	int x, y, n;
	scanf("%d%d%d", &x, &y, &n);
	printf("%lld\n", euc(x, y, n-2));
}
int main() {
	int z;
	scanf("%d", &z);
	while(z--) te();
	return 0;
}
