#include <bits/stdc++.h>
using namespace std;

int point_f(char c)
{
    if(c=='T')
        return 150;
    else if(c=='D')
        return -100;
    else
        return 0;
}

int main()
{
    int T;
    scanf("%d",&T);
    for(int i=1; i<=T; i++)
    {
        int N,M,row,col,p;
        char c;
        scanf("%d %d\n",&N,&M);
        char Arr[N][M];
        int point[N][M];
        for(int k=0; k<N; k++)
            for(int l=0; l<M; l++)
                scanf("%c ",&Arr[k][l]);
        for(row=0; row<N; row++)
            for(col=0; col<M; col++)
            {
                c=Arr[row][col];
                p=point_f(c);
                if(row==0&&col==0)
                    point[row][col]=p;
                else if(row==0)
                {
                    point[row][col]=point[row][col-1]+p;
                }
                else if(col==0)
                {
                    point[row][col]=point[row-1][col]+p;
                }
                else
                {
                    point[row][col]=max(point[row-1][col],point[row][col-1])+p;
                }


                if(point[row][col]<0)
                    point[row][col]=-2000000000;
            }
       /* for(int k=0; k<N; k++)
        {
            for(int l=0; l<M; l++)
            {
                printf("%c",Arr[k][l]);
            }
            printf("\n");
        }
        for(int k=0; k<N; k++)
        {
            for(int l=0; l<M; l++)
                printf("%d ",point[k][l]);
            printf("\n");
        }*/
        int result=point[N-1][M-1];
        if(result<0)
            printf("-1\n");
        else
            printf("%d\n",point[N-1][M-1]);
    }
    return 0;
}
