#include<bits/stdc++.h>
using namespace std;
#define ll long long int

int n,dp[25][3],mat[25][3];
int min(int a,int b)
{
    return a>b?b:a;
}
int memo(int idx,int j)
{
    int sum = 99999999;
    if(idx>=n) return 0;
    //if(dp[idx][j]!= -1) return dp[idx][j];
    for(int i=0;i<3;i++)
    {
        if(i!=j)
            sum = min((mat[idx][i]+ memo(idx+1,i)),sum);
    }
    //return (dp[idx][j] = sum);
    return sum;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(int i=0;i<25;i++)
            for(int j=0;j<3;j++) dp[i][j] = -1;
        for(int i=0;i<n;i++)
            for(int j=0;j<3;j++)
                cin>>mat[i][j];

        int ans = memo(0,-1);
        cout<<ans<<endl;
    }
    return 0;
}
