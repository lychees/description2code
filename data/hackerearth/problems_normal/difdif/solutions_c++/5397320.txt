#include<bits/stdc++.h>
using namespace std;

int main()
{
	string s;
	set<string>v;
	int i,j,k;
	cin>>s;
	k=s.length();
	for(i=0;i<k;i++)
	{
		string r;
		for(j=k-i;j>=0;j--)
		{
			r=s.substr(i,j);
			if(r!="")
			{
			//cout<<r<<endl;
			v.insert(r);
			}
		}
	}

	cout<<v.size()<<endl;
	return 0;
}
