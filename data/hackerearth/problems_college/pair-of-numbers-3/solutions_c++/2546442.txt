#include <iostream>
using namespace std;

int main()  
{  
      int t;
      cin>>t;
      while(t--)  
      {  
           int cnt=0;  
           int N;
           cin>>N;
           for(int i=1;i<=N;i++)  
           {  
                for(int j=i+1;j<=N;j++)  
                {  
                     if((i*j)%(i+j)==0)  
                     cnt++;  
                }  
           }  
           cout<<cnt<<"\n" ;
      }  
      return 0;  
}
