#include<iostream>
#include<cmath>
using namespace std;
int main()
{
	unsigned long k,a,b,c,d,t,T;
	cin>>T;
	while(T--)
	{
		cin>>a>>b>>c>>d>>k;
		t=(int)floor(cbrt((float)((k)/a)))+1; 	
		while(t--)								
		if((a*(t*t*t)+b*(t*t)+c*t+d)<=k||t==0) 	
		{
 		cout<<t<<"\n";						
		break;								
 		}
	}
    return 0;
}
