int countp(int n)
{
	unsigned long long int a,b,c,d;
	if(n==1)
	{
	
		return 1;
	}
	if(n==2)
	{
		b=2;
		return 2;
	}
	if(n==3)
	{
		c=4;
		return 4;
	}
	else
	{
	//return (countp(a-1)%1000000007 + countp(a-2)%1000000007 + countp(a-3)%1000000007)%1000000007;
		a=1;
		b=2;
		c=4;
	for(int i=4;i<=n;i++)
		{
			d=(a+b+c)%1000000007;
			a=b;
			b=c;
			c=d;
			
		}
		return d;
	}
}
