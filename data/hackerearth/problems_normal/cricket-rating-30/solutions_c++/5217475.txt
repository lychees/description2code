#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n,num;
    cin >> n;
    int cur_max,max_so_far;
    cin>> cur_max;
    max_so_far = cur_max;
    for(int i = 1; i< n; i++){
    	cin >> num;
    	cur_max = max(num,cur_max+num);
    	max_so_far = max(cur_max,max_so_far);
    }
    cout << max_so_far;
    
    return 0;
}
