#include <iostream>
#include <math.h>

using namespace std;

int main()
{
	int hh,mm;
	float result,angle_h,angle_m;
	char ch;
	
	//cin>>hh>>mm;
	scanf("%d",&hh);
    scanf("%c",&ch);
    scanf("%d",&mm);
	
	 if (hh <0 || mm < 0 || hh >12 || mm > 60)
       cout<< 0;
       
	hh=hh%12;
	
	angle_h = hh*30 + mm*0.5;
	angle_m = mm * 6;
	result = fabs(angle_m-angle_h);
	if(result>180)
	 	result=360-result;
	 
    printf("%0.6f\n",result);
    return 0;
}
