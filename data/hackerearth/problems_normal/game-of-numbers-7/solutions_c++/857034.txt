#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;


int main()
{
    int t=0;
    
    long long int  min=0,max=0,count=0;
    cin>>t;
    while(t--)
    {
        count=0;
        cin>>min>>max;
        vector<bool> mark(max - min + 1, true);
        for (long long int  j = 2; j* j <= max; j++)
        {
            long long int gap=j*j;
             long long int base = min + (gap - min % gap) % gap;
            
            for(long long int i=base;i<=max;i+=gap)
                mark[int(i-min)]=false;
            
        }
        
        for(long long int i=0;i<mark.size();i++)
            if(mark[i])count++;
        
        cout<<count<<endl;;
        
        
    }
    
    return 0;
}
