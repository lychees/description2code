#include <iostream>
using namespace std;

int main()
{
    int t,n,m,i,a;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	if(n%3==0)
    	{
    		a=n/3;
    		for(i=0;i<(a*3);i++)
    			cout<<"5";
    	}
    	else if(n%5==0)
    	{
    		a=n/5;
    		for(i=0;i<(a*5);i++)
    			cout<<"3";
    	}
    	else if((n-5)%3==0)
    	{
    		a=(n-5)/3;
    		for(i=0;i<(a*3);i++)
    			cout<<"5";
    		m=n-(a*3);
    		m/=5;
    		for(i=0;i<(m*5);i++)
    		{
    			cout<<"3";
    		}
    	}
    	else if((n-3)%5==0)
    	{
    		a=(n-3)/5;
    		for(i=0;i<(a*5);i++)
    		{
    			cout<<"3";
    			
    		}
    		m=n-(a*5);
    		m/=3;
    		for(i=0;i<(m*3);i++)
    		{
    			cout<<"5";
    		}
    	}
    	else cout<<"-1";
    	cout<<endl;
    		
    }
    return 0;
}
