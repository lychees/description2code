#include <iostream>
using namespace std;

int main()
{
    long t;
    cin>>t;
    while(t--)
    {
    	long n,k,count=0;
    	cin>>n>>k;
    	long arr[n];
    	for(int i=0;i<n;i++)
    	{
    		
    		cin>>arr[i];
    	}
    	for(int i=0;i<n;i++)
    	{
    		if(arr[i]<=0)
    		{
    			count++;
    			
    		}
    		
    	}
    	if(count<k)
    	{
    		cout<<"YES"<<"\n";
    		//cout<<count;
    		
    	}else{
    	
    		cout<<"NO"<<"\n";
    		
    	}
    	
    }
    
    
    
    
    return 0;
}
