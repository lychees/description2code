	string s; cin >> s;
	while (Q--) {
		int i, j; cin >> i >> j;
		set<int> st;
		while (i && s[i - 1] == '0') {
			st.insert(i);
			i /= 2;
		}
		bool flag = false;
		while (j && s[j - 1] == '0') {
			if (st.count(j)) flag = true;
			j /= 2;
		}
		cout << (flag ? "Yes" : "No") << endl;
	}
}
