#include<bits/stdc++.h>
using namespace std;
void permute(char *a, int i, int n)
{
   int j,temp;
   if (i == n)
     printf("%s\n", a);
   else
   {
        for (j = i; j <= n; j++)
       {
           temp=a[i];
           a[i]=a[j];
           a[j]=temp;

          permute(a, i+1, n);
          temp=a[i];
           a[i]=a[j];
           a[j]=temp;

       }
   }
}
int main()
{

int a,b,i,n;
char str[1000];
cin>>str;
cin>>n;
for(i=0;i<n;i++)
{
cin>>a>>b;
permute(str,a,b);
cout<<endl;
}
return 0;
}
