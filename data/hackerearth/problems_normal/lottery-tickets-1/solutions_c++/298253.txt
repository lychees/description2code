#include <iostream>
#include <vector>
#include <algorithm>
#include <iomanip>
using namespace std;

int main()
{
    int t = 0; 
    cin>>t;
    while(t--)
    {
    	int n = 0;
    	cin>>n;
    	vector<int> st(n), et(n);
    	vector<double> prob(n);
    	for(int i = 0 ;i < n; i++)
    	{
    		cin>>st[i]>>et[i]>>prob[i];
    	}
    	for(int i = 0; i < n; i++)
    	{
    		for(int j = i+1; j < n; j++)
    		{
    			if(et[i] > et[j])
    			{
    				swap(st[i],st[j]);
    				swap(et[i],et[j]);
    				swap(prob[i],prob[j]);
    			}
    		}
    	}
    	vector<double> ans(n);
    	double final = 0.0;
    	for(int i = 0; i < n; i++)
    	{
    		double x = prob[i]/100;
    		ans[i] = x;
    		for(int j = i-1; j >= 0; j--)
    		{
    			if(st[i] >= et[j])
    			{
    				ans[i] = max(ans[i], ans[j] + x);
    			}
    		}
    		if(ans[i] > final) final = ans[i];
    	}
    	cout<<fixed;
    	cout<<setprecision(2)<<final<<endl;
    }
    return 0;
}
