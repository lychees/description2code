long long int dp[100006],p = 23,ha = 23,fir,sec;
string str;
void pre() {
  int i;
  for(i=0;i < str.length();i++) {
    dp[i+1]=(dp[i] + (str[i]*p)%MOD)%MOD;
    p=(p*ha)%MOD;
  }
}

int main()
{
  int q,l1,r1,l2,r2;
  cin >> str;
  pre();
  s(q);
  while(q--) {
    s(l1);s(r1);s(l2);s(r2);
    fir = mulmod(dp[r1] - dp [l1-1] + MOD,modpow(modpow(ha,l1),MOD-2),MOD);
    sec = mulmod(dp[r2] - dp [l2-1] + MOD,modpow(modpow(ha,l2),MOD-2),MOD);
    printf("%s\n",fir == sec ? "Yes" : "No");
  } 
  return 0;
}
