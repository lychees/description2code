#include <stdio.h>
using namespace std;
int main()
{
	long long int testCases;
	scanf("%lld", &testCases);
	for(int c=0;c<testCases;c++)
	{
		long long int n;
		scanf("%lld", &n);
		long long int oddsum = 1, evensum = 0, steps = 0;
		while(n != 1)
		{
			if(n % 2 == 0)
			{
				evensum += n;
				n /= 2;
			}
			else
			{
				oddsum += n;
				n = (3 * n) + 1;
			}
			steps++;	
		}
		long long int oddrem = oddsum % steps;
		long long int evenrem = evensum % steps;
		if(oddrem > evenrem)
			printf("Odd Rules\n");
		else if(evenrem > oddrem)
			printf("Even Rules\n");
		else	
			printf("Tie\n");
	}
	return 0;
}
