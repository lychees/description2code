#include<iostream>
#include<algorithm>
#include<vector>
#include<string>
#include<cstring>
#include<memory>
#include<math.h>
#include<iso646.h>

using namespace std;

int main()
{
    long int n,count=0,max=0;
    cin>>n;
    vector<long int> a(n);
    for(int i=0;i<n;i++)
    {
        cin>>a[i];
    }
    if(n==1)
        cout<<"1"<<endl;
    else{
    for(int i=0;i<n-1;i++)
    {
        if((a[i+1]-a[i])==1 || (a[i+1]-a[i])==-1)
            count++;
        else
        {
            if(count>=max)
                {
                    max=count;
                    count=0;
                }
            else{
                count=0;
            }
        }
    }
    if(count>=max)
        max=count;
    if(max==0)
        cout<<max<<endl;
    else
        cout<<(max+1)<<endl;
    }
return 0;
}
