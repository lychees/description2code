//string without primes//

#include <iostream>
#include <string>
#include <string.h>
using std::cout;
using std::cin;
using std::string;

int main()
{
   int table[123];
   memset(table,0,sizeof(table));
   bool prime[123]={false};
   for(int i=2;i<123;i++)
   {
   	    if(table[i]==0)
   	    {
   	    	prime[i]=true;
   	    	int x=i;
                int j=1;
   	    	while(x<123)
   	    	{
   	    		  table[x]=1;
                  j++;
                  x=i*j;
   	    	}
   	    }
   }
   int T;
   cin>>T;
   while(T)
   {
   string s;
   string n;
   cin>>s;
   for(int i=0;i<s.length();i++)
   {
   	    if(!prime[s[i]])
   	    {
   	    	n+=s[i];  
   	    }
   }
   if(!n.empty())
   	  cout<<n<<"\n";
   	  else
   	  cout<<"NULL"<<"\n";
   T--;
}
}
