#include <iostream>
using namespace std;

int cmp(const void *a,const void *b)
{
	return *(int*)a - *(int*)b ;
}

int main()
{
    int n ;
    cin>>n;
    int arr[n] ;
    for(int i=0;i<n;i++)
    cin>>arr[i] ;
    qsort(arr,n,sizeof(int),cmp) ;
    int min,s=0,count=0;
    cin>>min ;
    for(int i=n-1;i>=0;i--)
    {  s=s+arr[i] ;
       count++ ;
       if(s>min)
          break ;        
    	
    }
    cout<<count ;
    return 0;
}
