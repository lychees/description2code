#include <iostream>
#include <cstdio>

#define MAXN 1000000

using namespace std;

int N;

void read()
{
	scanf("%d", &N);
}

void solve()
{
	int cnt = 0;
	bool can = false;

	for(cnt = N; cnt >= 0; cnt--)
		if(cnt % 3 == 0 && (N - cnt) % 5 == 0)
		{
			can = true;
			break;
		}

	if(!can)
	{
		printf("-1\n");
		return;
	}

	for(int i = 0; i < cnt; i++)
		printf("5");
	for(int j = cnt; j < N; j++)
		printf("3");
	
	printf("\n");
}

int main()
{
	int T;
	scanf("%d", &T);

	for(int i = 0; i < T; i++)
	{
		read();
		solve();
	}
	
	return 0;
}
