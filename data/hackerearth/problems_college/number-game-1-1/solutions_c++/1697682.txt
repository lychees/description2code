#include <iostream>
using namespace std;
int main(){
	ios::sync_with_stdio(false);
	int T=0;
	
	cin >> T;
	
	while(T--)
	{
	 int in=0;
	 cin >> in;
	 int count=0;
	 
	 while(in!=1)
	 {
	  if(in%3!=0 and in%2!=0)
	   in--;
	  else if(in%3==0)
	   in/=3;
	  else if(in%2==0)
	   in/=2;
	  count++;
	 }
	 
	 cout << count << endl;
	}
	
	return 0;
}
