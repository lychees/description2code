#include <bits/stdc++.h>
#define MOD 1000000007
#define ll long long
using namespace std;

ll int a[1010],b[1010],n,k;
ll int dp[1010][1010];

ll int calc(ll int x,ll int y)
{
    if(x+y > n) return 1;
    if(  x < k*y) return dp[x][y]=0;

    if(dp[x][y]!=-1)
        return dp[x][y];

    if(x+y<n)
    return dp[x][y]=( calc(x+1,y) + calc(x,y+1) ) %MOD;
    else if(x!=y)
    return dp[x][y]=1;
    else if(x==y)
    return dp[x][y]=0;
}


int main()
{
    ll int t,i,j;

    cin>>t;

    while(t--)
    {
        cin>>n>>k;

        for(i=0;i<=n;i++)
            for(j=0;j<=n;j++)
                dp[i][j]=-1;
        cout<<calc(1,0)<<endl;
    }
}
