#include <iostream>
using namespace std;
// let see o(n^2) solution first
int main()
{
	
	int n,q;
	cin>>n>>q;
	int arr[100001];
	  for(int i=0;i<n;i++){
	  	cin>>arr[i];
	  	 }
	  	 for(int i=0;i<q;i++){
	  	 	int c;
	  	 	cin>>c;
	  	 	if(c==1){
	  	 		int x;
	  	 		 bool flag=true;
	  	 		cin>>x;
	  	 		for(int i=0;i<n;i++){
	  	 			  
	  	 			  if(arr[i]>=x){
	  	 			  	     cout<<i+1<<endl;
	  	 			  	      flag=false;
	  	 			  	      goto here;
	  	 			  }
	  	 			  
	  	 		}
	  	 		here: ;
	  	 		if(flag==true)
	  	 		 cout<<"-1\n";
	  	 	}
	  	 	else
	  	 	{
	  	 		int a,b;
	  	 		cin>>a>>b;
	  	 		arr[a-1]=b;
	  	 	}
	  	 	
	  	 	
	  	 }
    
}
