#include <iostream>
#include <string>
using namespace std;

char* alpha= new char[26]{'a', 'b','c', 'd','e', 'f','g', 'h','i', 'j','k', 'l','m', 'n','o', 'p','q', 'r','s', 't','u', 'v','w', 'x','y', 'z'};

bool getChar(char c, string& s ){
	for(int i = 0; i < s.length(); i++){
		if(s[i] == c)
		return true;
	}
	return false;
}

int main()
{
    int count = 0;
    cin >> count;
    string* strings = new string[count];
    
    for(int i = 0; i < count; i++){
    	cin >> strings[i];
    }
    
    bool* test = new bool[count];
    for(int e = 0; e < count; e++){
    	test[e] = true;
	 for(int i = 0; i < 26; i++){
    		if(getChar(alpha[i], strings[e]) == false)
    		test[e] = false;
    	}
    }
    
    for(int e = 0; e < count; e++){
    	if(test[e] == true){
    		cout << "YES" << endl;
    	} else {
    		cout << "NO" << endl;
    	}
    }
    
    return 0;
}
