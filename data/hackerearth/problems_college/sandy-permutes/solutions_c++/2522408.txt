#include <iostream>
using namespace std;

int main()
{
  long long	int n,k;
  int max,e;
   cin>>n;
   cin>>k;
   long long int arr[n];
   for(int i=0;i<n;i++)
   {
   	cin>>arr[i];
   }
   for(int j=0;j<n;j++)
   {
   
        max=arr[j];
        int p=j;
        
   for(int i= j+1; i<n;i++)
     {
     	 
          if(arr[i] >=max)
          {
            max = arr[i];
            p=i;
          }
     }
     if(p==j)
     continue;
     else
     {
     	++e;
     	int temp=arr[p];
     	arr[p]=arr[j];
     	arr[j]=temp;
     }
     	if(e==k)
     	break;
     
    
   }
  for(int i=0;i<n;i++)
   {
   cout<<arr[i]<<" ";
   }
   return 0;
 }
