#include<iostream>
#include<vector>

int main() {
        int tot_cases;
        std::cin >> tot_cases;
        std::cin.get();

        int s,i,j;
        long long m[22],d[10001]={0};
        for(i=1;i<=21;i++)
                m[i]=i*i*i;
        d[0]=1;
        for(i=1;i<=21;i++)
                for(j=0;j<10001;j++)
                        if(j>=m[i])
                                d[j]+=d[j-m[i]];


        std::vector<uint64_t> res;

        for (int i = 0; i < tot_cases; i++) {
                int num ;
                std::cin >> num;
                res.push_back(d[num]);
        }
        std::vector<uint64_t>::iterator it = res.begin();
        for (; it != res.end(); it++)
                std::cout << *it << '\n';

        return 0;
}
