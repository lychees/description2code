const ll M = 1e9 + 7;

ll a[N], b[N];
int main() {

	std::ios::sync_with_stdio(false);
	//cout << "hello";
	int n;
	cin >> n;
	for(int i = 0; i < n; i++) {
		int x, y;
		cin >> x >> y;
		a[i] = x, b[i] = y;
	//	a[i] += 1e9;

	//	if(a[i] >= M) a[i] %= M;

	}
	sort(a, a + n);
	sort(b, b + n);
	ll mul1 = n - 1, mul2 = 0;
	ll ans1 = 0, ans2 = 0, ans3 = 0, ans4 = 0;
	for(int i = 0; i < n; i++) {
		ll mid = mul1 * a[i];
		if(mid >= M) mid %= M;
		ans1 += mid;
		if(ans1 >= M) ans1 %= M;
		mid = mul2 * a[i];
		if(mid >= M) mid %= M;
		ans2 += mid;
		if(ans2 >= M) ans2 %= M;
		mid = mul1 * b[i];
		if(mid >= M) mid %= M;
		ans3 += mid;
		if(ans3 >= M) ans3 %= M;
		mid = mul2 * b[i];
		if(mid >= M) mid %= M;
		ans4 += mid;
		if(ans4 >= M) ans4 %= M;
		mul1--, mul2++;
	}
	ll ans = (ans2 - ans1 + M) % M;
	ans += (ans4 - ans3 + M) % M;
	if(ans >= M) ans %= M;

}
