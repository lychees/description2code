#include <iostream>
using namespace std;

const long long mod = 1000000007ll;

int pw(int x) {
	if(x == 0) return 1;
	long long r = pw(x/2);
	r = (r * r) % mod;
	if(x % 2) r = (3 * r) % mod;
	return r;
}
int main()
{
	int t;
	cin >> t;
	while(t--) {
		int n;
		cin >> n;
		cout << pw(n) << "\n";
	}
    return 0;
}
