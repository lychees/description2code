#include <iostream>
#include <math.h>
using namespace std;

int main()
{
	
	int t;
	cin>>t;
	
	while(t-- >0){
		long long b, w;
		cin>> b;
		cin>>w;
		
		long long bC, wC, cC;
		cin>> bC;
		cin>> wC;
		cin>> cC;
		
		long long o = b *bC + w * wC;
		long long t = b *bC + w * (bC + cC);
		long long f = b *(wC +cC) + w * wC;
		
		o= min(o,t);
		o= min(o,f);
		cout<<o<<"\n";
		
		
	}
    //cout << "Hello World!" << endl;
    return 0;
}
