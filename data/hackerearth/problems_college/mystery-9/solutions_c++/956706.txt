#include<iostream>

using namespace std;

int main(){
int t,i;
int days[12]={31,28,31,30,31,30,31,31,30,31,30,31};
for(i=1;i<12;i++) days[i]+=days[i-1];
cin>>t;
while(t--){
        int n;
        cin>>n;
        for(i=0;i<12;i++){
            if(days[i]>=n){
                cout<<(i+1)<<"\n";
                break;
            }
        }
}
return 0;
}
