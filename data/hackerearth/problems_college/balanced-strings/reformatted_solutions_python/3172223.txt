t = int(raw_input())
for i in range(0, t):
    n = list(raw_input())
    l = [0]*26
    for k in range(0, len(n)):
        l[ord(n[k])-ord('a')] = l[ord(n[k])-ord('a')]+1
    cnt = l[0]
    flg = 0
    for m in range(0, 26):
        if (l[m] != cnt):
            flg = 1
            break
    if (flg == 1):
        print "No"
    else:
        print "Yes"
