#include <iostream>
using namespace std;

int main()
{
	int t, n, i, j, total, row, column, flag;
	cin >> t;
	
	for (; t; t--) {
		cin >> n;
		char str[n][n+1];
		for (i = 0; i < n; i++) {
			cin >> str[i];
		}
		
		total = 0;
		for (i = 0; i < n; i++) {
			for (j = 0; j < n; j++) {
				if (str[i][j] == '#') {
					total++;
				}
			}
		}
		
		//cout << total << endl;
		flag = 1; row = 0;
		for (i = 0; i < n; i++) {
			for (j = 0; j < n; j++) {
				if (str[i][j] == '#') {
					row++;
				}
			}
			//cout << row << endl;
			if ((row * 2) == total) {
				break;
			} else if ((row * 2) > total) {
				flag = 0;
				break;
			}
		}
		
		if (flag == 1) {
			cout << "YES" << endl;	
		} else {
			column = 0;
			for (i = 0; i < n; i++) {
				for (j = 0; j < n; j++) {
					if (str[j][i] == '#') {
						column++;
					}
				}
				if ((column * 2) == total) {
					flag = 1;
					break;
				} else if ((column * 2) > total) {
					flag = 0;
					break;
				}
			}
			
			if (flag == 1) {
				cout << "YES" << endl;
			} else {
				cout << "NO" << endl;
			}
		}
	}
}
