#include <stdio.h>
#include<math.h>
int Prime(long int n)
{
     if(n==1)
        return 0;
     for(long int i=2;i<=sqrt(n);i++)
     {
     	if(n%i==0)
            return 0;
     }
     return 1;
}
int main()
{
    int t;
    long int n;
    scanf("%d",&t);
    while(t--)
    {
       	scanf("%ld",&n);
       	if(n==3)
       	{
       	   printf("%d\n",3);
       	}
       	else if(Prime(n))
       	{
       		printf("%d\n",2);
       	}
       	else
       	    printf("%d\n",1);
    }
    return 0;
}
