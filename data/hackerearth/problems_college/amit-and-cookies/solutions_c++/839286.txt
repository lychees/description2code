#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
		long long int N,arr[100005],ans=0,aa=0,K;
		cin>>N>>K;
		for(int i=1;i<=N;i++)
		{
			cin>>arr[i];
		}
		sort(arr+1,arr+N+1);
		for(int i=1;i<=N;i++)
		{
			aa+=arr[i];
			if(aa<=K)
			{
				ans++;
			}
			else
			{
				break;
			}
		}
		cout<<ans<<endl;
		return 0;
}
