from math import ceil
t = input()
for times in range(t):
    pc = 1
    pd = 0
    time = 0
    n, m = raw_input().split()
    n, m = int(n), int(m)
    while pc < n:
        pc, pd = pc+pd, pd+pc
        time += 1
        if (pd >= m) and (pc <= n):
            time += int(ceil((n-pc)/float(m)))
            break
    print time
