#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int a[10000005];
vector <int> pr;
void pre()
{
	int i,j,p,c;
	for(i=2;i<=10000000;i++)
	{
		if(!a[i])
		{
			pr.push_back(i);
			for(j=i*2;j<=10000000;j+=i)
			{
				a[j]=1;
			}
		}
	}
}
int main()
{
	pre();
    long long int n,ans,c;
    cin>>n;
    c=upper_bound(pr.begin(),pr.end(),n)-pr.begin();
    ans=(c*(c-1))/2;
    cout<<ans<<endl;
    return 0;
}
