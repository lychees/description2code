#include <iostream>

using namespace std;

long long int gcd(long long int a, long long int b)
{
    long long int c;
    while (a != 0) {
        c = a;
        a = b % a;
        b = c;
    }

    return b;
}

int main()
{
    int t;

    cin >> t;

    while (t--) {
        long long int a, b;

        cin >> a >> b;

        if (gcd(a, b) == 1)
            cout << "Is a Co-Prime\n";

        else
            cout << "Not a Co-Prime\n";
    }

    return 0;
}
