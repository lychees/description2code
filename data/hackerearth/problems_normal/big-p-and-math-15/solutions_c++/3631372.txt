/*

         Do not go gentle into that good night,
         Old age should burn and rave at close of day;
         Rage, rage against the dying of the light.
         Though wise men at their end know dark is right,
         Because their words had forked no lightning they
         Do not go gentle into that good night.


*/
#include<bits/stdc++.h>
using namespace std;

int main()
{
	int a,b;
	cin>>a>>b;
	int aa=a,bb=b;
	int carry=0,add=0;
	int ans=0,rema=0,remb=0;
	vector<int>large,small;
	while(a or b){
		rema=a%10,remb=b%10;
		if(rema==5){
			rema++;
		}
		if(remb==5){
			remb++;
		}
		add=(rema+remb+carry);
		carry=add/10;
		large.push_back(add%10);
		a/=10,b/=10;
	}
	while(carry){
		large.push_back(carry%10);
		carry/=10;
	}
	while(aa or bb){
		rema=aa%10,remb=bb%10;
		if(rema==6){
			rema--;
		}
		if(remb==6){
			remb--;
		}
		add=rema+remb+carry;
		carry=add/10;
		small.push_back(add%10);
		aa/=10,bb/=10;
	}
	while(carry){
		small.push_back(carry%10);
		carry/=10;
	}
	for(int i=small.size()-1;i>=0;i--)cout<<small[i];
	cout<<" ";
	for(int i=large.size()-1;i>=0;i--)cout<<large[i];
	return 0;
}
