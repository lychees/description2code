#include<bits/stdc++.h>
#define mx 2005
#define ll long long
#define pb push_back
using namespace std;
int n;
int m;
char a[mx][mx];
ll b[mx][mx];
//char xx[]="_#_#_#_#_#_#_#_#_#_#_#_#_#_#_#_#";
//char yy[]="#_#_#_#_#_#_#_#_#_#_#_#_#_#_#_#_";
ll xx=0b10101010101010101010101010101010;
ll yy=0b01010101010101010101010101010101;
bool check(int x,int y)
{
    if(n-x>=32 && m-y>=32)
    {
        for(int i=0;i<32;i++)
        {
            if(i%2==0 && b[x+i][y]==xx)continue;
            else if(i%2==1 && b[x+i][y]==yy)continue;
            else return 0;
        }
        return 1;
    }
    return 0;
}

int main()
{
    //printf("%lld %lld\n",xx,yy);
    scanf("%d %d",&n,&m);
    if(n<32 ||m <32){printf("0\n");return 0;}
    for(int i=0;i<n;i++)scanf("%s",a[i]);
    for(int i=0;i<n;i++)
    {

        for(int j=0;j<m;j++)
        {
            ll x=0;
            for(int k=0;k<32 && j+k<m;k++)
            {
                x=(x*2)+(a[i][j+k]=='_'?1:0);
            }
            b[i][j]=x;
            //if(b[i][j]==xx)printf("%lld\n",b[i][j]);
        }
    }
    int ans=0;
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m;j++)
        {
            //if(taken[i][j])continue;
            if(a[i][j]=='_' && check(i,j))
            {
                ans++;
                //make(i,j);
            }
        }
    }
    printf("%d\n",ans);


}
