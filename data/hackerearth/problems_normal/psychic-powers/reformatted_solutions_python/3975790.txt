inp = raw_input()

sum = 1

for i in range(len(inp)-1):
    if inp[i] == inp[i+1]:
        sum = sum+1
        if sum == 6:
            break
    else:
        sum = 1

if sum != 6:
    print "Good luck!"
else:
    print "Sorry, sorry!"
