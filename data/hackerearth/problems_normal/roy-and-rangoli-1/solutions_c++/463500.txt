#include<iostream>
#include<math.h>
using namespace std;

int isPrime(long int a)
{
	if(a==0 || a==1)
		return 0;

	for(int i = 2 ; i<= sqrt(a) ;i++)
	{
		if(a%i==0)
			return 0;
	}
	return 1;
}

int main()
{
	long n, count=0, max;
	int i;
	cin>>n;
	max=(n-1)*2;
	for(i=2 ; i<n ; i++)
		if(isPrime(i)==1)
			count+=i+1;

	for(i=n ; i<=max ; i++)
		if(isPrime(i)==1)
			count+=max-i+1;
	cout<<count%1000000007;
	return 0;
}
