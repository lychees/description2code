#include<cstdio>
#include<algorithm>
const int N = 21;
int n, mx, wynik, i, t[N], t2[N];
bool w[N];

void rek(int j)
{
  if (j < 0){return;}
  
  if (wynik > mx)
  {
    mx = wynik;    
  }
  
  for(int i = n - 1; i >= 0; i--)
  {
    if (w[i] == 0 && t[j] > t2[i])
    {
      wynik += 100;
      w[i] = 1;
      rek(j - 1);
      w[i] = 0;
      wynik -= 100;
    }
  }
  
  for(int i = n - 1; i >= 0; i--)
  {
    if (w[i] == 0 && t[j] == t2[i])
    {
      wynik += 50;
      w[i] = 1;
      rek(j - 1);
      w[i] = 0;
      wynik -= 50;
    }
  }  
}

int main()
{
  scanf("%d", &n);
  
  for(i = 0; i < n; i++)
  {
    scanf("%d", &t[i]);    
  }
  for(i = 0; i < n; i++)
  {
    scanf("%d", &t2[i]);
  }
  
  std::sort(t, t + n);
  std::sort(t2, t2 + n);
  
  mx = 0;
  wynik = 0;
  rek(n - 1);
  
  printf("%d\n", mx);
  
return 0;
}
