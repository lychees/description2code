class TimeCalculator():
	def __init__(self, calling_order, ideal_order):
		self.calling_order = calling_order
		self.ideal_order = ideal_order
	
	def push_back_process(self):
		item = self.calling_order.pop(0)
		self.calling_order.append(item)
	
	def consume_process(self):
		self.calling_order.pop(0)
		self.ideal_order.pop(0)
	
	def can_be_processed(self):
		return self.calling_order[0] == self.ideal_order[0]
		
	def calculate_time(self):
		time_taken = 0
		while len(self.ideal_order) is not 0:
			if self.can_be_processed():
				self.consume_process()
			else:
				self.push_back_process()
			time_taken +=1
		return time_taken

def main():
	input_line_1 = raw_input()
	input_line_2 = raw_input().split(' ')
	input_line_3 = raw_input().split(' ')
	calling_order = [int(x) for x in input_line_2]
	ideal_order = [int(x) for x in input_line_3]
	tc = TimeCalculator(calling_order, ideal_order)
	print tc.calculate_time()

main()
