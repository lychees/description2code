def digsum(num):
    dsum = 0
    while num >= 0:
        d,r = divmod(num,10)
        dsum+=r
        num = d
        if num==0:
            break
    return dsum

#print digsum(12215)
        
   
noc = int(raw_input())
while noc:
    noc-=1
    num = raw_input()
    lower = int(''.join(num[:len(num)/2]))
    upper = int(''.join(num[len(num)/2:]))
    if digsum(lower)==digsum(upper):
        print "YES"
    else:
        print "NO"
