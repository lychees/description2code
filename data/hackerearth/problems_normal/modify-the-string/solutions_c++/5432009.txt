#include <iostream>

using namespace std;

int main()
{
    string str;
    cin >> str;

    const size_t len = str.length();

    for (size_t i = 0; i < len; i++) {
        char ch = str[i];

        if ('a' <= ch && ch <= 'z') {
            // cerr << ch << " is small" << endl;
            str[i] = ch - ('a' - 'A');
        } else if ('A' <= ch && ch <= 'Z') {
            // cerr << ch << " is capital" << endl;
            str[i] = ch + ('a' - 'A');
        }
    }

    cout << str << endl;

    return 0;
}
