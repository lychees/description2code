list1 = []
import sys
for i in sys.stdin:
	list1.append(i.strip())
for i in range(1,len(list1)-1,2):
	if list1[i]+list1[i+1] == (list1[i]+list1[i+1])[::-1]:
		print "YES"
	elif list1[i+1]+list1[i] == (list1[i+1]+list1[i])[::-1]:
		print "YES"
	else:
		print "NO"
