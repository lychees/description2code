#include<iostream>
#include<cstring>
#include<cstdio>
#include<map>
using namespace std;
int main()
{
    int n, q;
    cin>>n>>q;
    map<int, int> v;
    map<int, int> :: iterator it;
    while(q--) {
        int i, j;
        cin>>i>>j;
        int kk = min(i, j);
        j = max(i, j);
        i = kk;
        if(i == j) cout<<"NO"<<endl;
        else {
            int flag = 0;
            for(it = v.begin(); it!= v.end(); it++) {
                if((i > it -> first && i > it -> second && j > it -> second && j > it -> first) || (i < it -> first && i < it -> second && j < it -> second && j < it -> first)) flag = 0;
                else if(i > it -> first && i < it -> second && j > it -> first && j < it -> second) flag = 0;
                else if(i < it -> first && j > it -> second) flag = 0;
                else if(i > it -> second && j > it -> second) flag = 0;
                else {flag = 1; break;}
            }
            if(flag == 1) cout<<"NO"<<endl;
            else {
                v.insert(pair<int, int> (i, j));
                cout<<"YES"<<endl;
            }
        }

    }
    return 0;
}
