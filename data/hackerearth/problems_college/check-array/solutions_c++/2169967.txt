#include <iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,flag=0,i;
    	cin>>n;
    	long long a[n],b[n],c[n];
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    		if(i==0)
    			b[i]=a[i];
    		else
    			b[i]=b[i-1]+a[i];
    	}
    	for(i=n-1;i>=0;i--)
    	{
    		if(i==(n-1))
    			c[i]=a[i];
    		else
    			c[i]=a[i]+c[i+1];
    	}
    	for(i=1;i<n-1;i++)
    	{
    		if(b[i-1]==c[i+1])
    		{
    			flag=1;
    			break;
    		}
    	}
    	if(flag)
    		cout<<"YES\n";
    	else
    		cout<<"NO\n";
    }
    return 0;
}
