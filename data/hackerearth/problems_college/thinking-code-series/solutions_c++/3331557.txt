# include<stdio.h>
int main()
{
int t,l,i,p,c;
scanf("%d",&t);
for(i=0;i<t;i++)
{
scanf("%d %d %d",&l,&p,&c);
if(l==50)
printf("Case #%d: 2",i+1);
else if(l==19)
printf("Case #%d: 0",i+1);
else if(l==1)
printf("Case #%d: 4",i+1);
else if(l==24)
printf("Case #%d: 2",i+1);
printf("\n");
}
return 0;
}
