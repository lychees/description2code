#include<bits/stdc++.h>
using namespace std;


int main()
{

int tc;

cin>>tc;


while(tc--)
{



string s;

cin>>s;

int len=s.length();
string temp="";

for(int i=4;i<s.length()-4;i++)
{
if(s[i]!='a'&&s[i]!='e'&&s[i]!='i'&&s[i]!='o'&&
s[i]!='u')
	temp+=s[i];

}
int tl;
//cout<<tl;
tl=temp.length()+4;
//cout<<temp;
string res="";
res=res+to_string(tl)+"/"+to_string(len);
cout<<res<<endl;
}

}
