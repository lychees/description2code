#include <iostream>
#include <climits>
using namespace std;
int q[10000];
int fq[10000];
int n;
bool is_sorted(){
		
    	for(int i=0;i<n-1;i++){
    		if(q[i]>q[i+1]){
    			return false;
    		}
    		
    	}
    	return true;
}
int main()
{
    int t;
    cin>>t;
    while(t--){
    	//int n;
    	cin>>n;
    	for(int i=0;i<n;i++)
    		cin>>q[i];
    	if(n==1 || is_sorted()){
    		cout<<"-1 -1\n";
    		continue;
    	}
    
    	int s=0;
    	int e=n-1;
    	int mini=INT_MAX;
    	for(int i=0;i<n;i++){
    	
    		if(q[i]>q[i+1]){
    			s=i;
    			break;
    		}
    	}
    	for(int i=s;i<n-1;i++){
    		if(q[i+1]>q[i]){
    			e=i;
    			break;
    		}
    	}
    	int ans1,ans2;
    	ans1=s+1;
    	ans2=e+1;
    	while(s<e){
    		int temp=q[s];
    		q[s]=q[e];
    		q[e]=temp;
    		s++;
    		e--;
    		
    	}
    	
    
    	if(is_sorted()){
    		cout<<ans1<<" "<<ans2<<"\n";
    		
    	}
    	else
    		cout<<"-1 -1\n";
    }
}
