#include <bits/stdc++.h>
using namespace std;
//3(3^(N-1) - 2 ^N + 1)
#define lli long long int
#define mod 1000000007
lli pw(lli a,lli b,lli c){
	if(b<=0)return 1;
	if(b==1)return a;
	lli ans=pw(a,b/2,c);
	ans=(ans*ans)%c;
	if(b&1)ans=(ans*a)%c;
	return ans;
}
int main(){
    int t;
    cin>>t;
    lli n;
    while(t--){
    	cin>>n;
    	cout<<3*((pw(3,n-1,mod))-pw(2,n,mod)+1+mod)%mod<<"\n";
    }
    return 0;
}
