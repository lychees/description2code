test = int(input())
visited = [False for i in range(10001)]
for t in range(test):
    e = int(input())
    visited = [False for i in range(10001)]
    for i in range(e):
        x, y = map(int, raw_input().split())
        visited[x] = True
        visited[y] = True
    c = 0
    for i in range(10001):
        if (visited[i]):
            visited[i] = False
            c += 1
    print(c)
