#include <iostream>
#include <vector>
#include <stack>
#include <algorithm>
using namespace std;
 
int main()
{
	int i, t, n, k, a, j;
	vector<int> v;
	
    cin>>t;
    while(t--)
    {
    	cin>>n>>k;
    	j = 0;
    	v.resize(0);
    	for(i=0; i<n; i++)
    	{
    		cin>>a;
    		if(v.size() == 0)
    		{
    			v.push_back(a);
    		}
    		else
    		{
    			while(v.size() >0 && a > v.back() && j<k)
    			{
    				v.pop_back();
    				j++;
    			}
    			v.push_back(a);
    		}
    	}
    	while(j < k)
    	{
    		v.pop_back();
    		j++;
    	}
    	for(i=0; i<v.size(); i++)
    	{
    		cout<<v[i]<<" ";
    	}
    	cout<<endl;
    }
    return 0;
}
