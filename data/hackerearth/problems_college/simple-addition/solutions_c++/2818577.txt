#include<bits/stdc++.h>
#define MAX 100000
using namespace std;
int calculate(int a,int b)
{
    if(a==b)
    {
        if(a==1)
            return 1;
        else
            return MAX;
    }
    else
    {
        if(a>b)
        {
            return 1+calculate(a-b,b);
        }
        else
            return 1+calculate(a,b-a);
    }
}
int main()
{
    int t,x,y,l,c,temp;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&l);
        if(l==1)
            printf("%d\n",0);
        else if(l==2)
            printf("%d\n",1);
        else
        {
            x=l-1;
            y=1;
            c=MAX;
            while(x>y)
            {
                c=min(c,calculate(x,y));
                x--;y++;
            }
            printf("%d\n",c);
        }
    }
}
