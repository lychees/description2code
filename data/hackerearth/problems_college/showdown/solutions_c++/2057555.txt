using namespace std;

int cmpfunc (const void * a, const void * b)
{
   return ( *(int*)a - *(int*)b );
}

int cmpfunc1 (const void * a, const void * b)
{
   return ( *(int*)b - *(int*)a );
}

int main(int argc, char *argv[])
{
    int n;
    cin >> n;
    
    int arr1[21];
    int arr2[21];
    
    
    for (int i = 0; i < n; i++)
    {
        cin >> arr1[i];
    }
    
    for (int i = 0; i < n; i++)
    {
        cin >> arr2[i];
    }
    qsort(arr1, n, sizeof(int), cmpfunc1);
    qsort(arr2, n, sizeof(int), cmpfunc1);

    int res = 0;
    
    for (int i = 0; i < n-1; i++)
    {
        if (arr1[i] > arr2[i+1])
        {
            res +=100;
        }
        else{
            if (arr1[i] == arr2[i+1])
            {
                res += 50;
            }
        }
    }
    
    if (arr1[n-1] > arr2[0])
    {
        res += 100;
    }
    else
    {
        if (arr1[n-1] == arr2[0])
        {
            res += 50;
        }
    }
    
    cout << res << endl;

 	return 0;
}
