#include <cstdio>
#include <vector>
#include <stack>
using namespace std;

vector <int> adj[1005];
bool visited[1005];

int dfs(int s){
	stack <int> st;
	int sum=1;
	visited[s]=true;
	st.push(s);
	//printf ("%d ", s);
	while (!st.empty()){
		int v=st.top();
		//printf ("%d ", v);
		st.pop();
		for (int i=0; i<adj[v].size(); i++){
			if (visited[adj[v][i]]==false){
				sum++;
				visited[adj[v][i]]=true;
				st.push(adj[v][i]);
			}
		}
		
	}
	//printf ("\n\n");
	return sum;
	
}


int main(){
	int n, m;
	scanf("%d %d", &n, &m);
	int a, b;
	for (int i=0; i<m; i++){
		scanf ("%d %d", &a, &b);
		adj[a].push_back(b);
	}
	
	
	int mn=1005, tmp;
	for (int i=1; i<=n; i++){
		for (int i=1; i<=n; i++)
			visited[i]=false;
		tmp=dfs(i);
		if (tmp<mn)
			mn=tmp;
	}
	printf ("%d\n", mn);
	
	return 0;
}
