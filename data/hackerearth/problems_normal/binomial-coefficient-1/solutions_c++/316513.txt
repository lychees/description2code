#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
const int mod = 1000000007;
LL pow_mod(LL a, LL n) {
    LL ret = 1 ;
    while(n) {
        if(n&1) ret = ret * a % mod ;
        a = a*a%mod ;
        n>>=1 ;
    }
    return ret ;
}
long long fact[1000001];
long long rev[1000001];
void factorial(int n){
    int i=1;
    while(i<n){
        fact[i] = (fact[i-1]*i)%mod;
        rev[i] = pow_mod(fact[i],mod-2);
        i++;
    }
}
long long getnCk(int a, int b){
    if(b>a) return 0;
    else return (((fact[a]*rev[b])%mod)*rev[a-b])%mod;
}
int main(){
    fact[0]=1;
    rev[0] = pow_mod(fact[0],mod-2);
    int a,b,n,k;
    long long ans;
    cin>>a>>b>>n>>k;
    factorial(1000001);
    //cout<<getnCk(n,k);
    ans = (getnCk(n,k)*pow_mod(a,k))%mod;
    ans = (ans*pow_mod(b,n-k))%mod;
    cout<<ans<<endl;
    return 0;
}
