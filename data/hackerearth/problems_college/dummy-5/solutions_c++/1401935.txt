#include <iostream>
using namespace std;
int p[20],v[20],c[20];
bool subsetsum(int n,int gp,int gv,int gc){
	//cout<<n<<" "<<gp<<" "<<gv<<" "<<gc<<endl;
	if(gp==0 && gv==0 && gc==0)
		return true;
	
	if(n==0 && (gp!=0 || gv!=0 || gc!=0))
		return false;
	
	if((p[n-1]>gp) || (v[n-1]>gv) || (c[n-1]>gc))
		return subsetsum(n-1,gp,gv,gc);
	
	return subsetsum(n-1,gp,gv,gc) || subsetsum(n-1,gp-p[n-1],gv-v[n-1],gc-c[n-1]);
}

int main()
{
    int t,n,gp,gv,gc;
    cin>>t;
    for(int k=1;k<=t;k++){
    	
    	cin>>gp>>gv>>gc;
    	cin>>n;
    	for(int i=0;i<n;i++)
    		cin>>p[i]>>v[i]>>c[i];
    	
    	if(subsetsum(n,gp,gv,gc)==true)
    		cout<<"Case #"<<k<<": yes"<<endl;
    	else
    		cout<<"Case #"<<k<<": no"<<endl;
    }
    return 0;
}
