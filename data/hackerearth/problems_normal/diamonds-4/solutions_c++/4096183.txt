#include <iostream>
using namespace std;

int main()
{
	int T;
	cin>>T;
	while(T--)
	{
		int N,M;
		cin>>N>>M;
		char input[N][M];
		for(int i=0;i<N;i++)
		{
			for(int j=0;j<M;j++)
			{
				cin>>input[i][j];
			}
		}
		int count=0;

		for(int i=0;i<N-1;i++)
		{
			for(int j=0;j<M-1;j++)
			{
				if(input[i][j]=='/' && input[i][j+1]=='\\' && input[i+1][j]=='\\' && input[i+1][j+1]=='/')
				{
					count++;	
				}
			}
		}
		cout<<count<<endl;
	}
	
    return 0;
}
