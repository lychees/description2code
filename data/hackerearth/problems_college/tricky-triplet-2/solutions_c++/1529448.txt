#include<iostream>
#include<stdlib.h>
#include<stdio.h>
#include<math.h>
#define MOD 1000000007
using namespace std;
 
int main()
{
	int n,cnt=0;
	int a[100000];
	cin>>n;
	for(int i=0;i<n;i++)
	cin>>a[i];
	
	for(int i=0;i<n;i++)
	{
		 for(int j=i+1;j<n;j++)
		 {
		 	for(int k=j+1;k<n;k++)
		 	{
		 		if(a[j]-a[i]==a[k]-a[j])
		 		cnt++;
		 	}
		 }
	}
	cout<<cnt<<endl;
}
