#include <iostream>
#include <algorithm>
#include <stdio.h>
using namespace std;

int main()
{
	int t,i,j,k,p,n;
	scanf("%d",&t);
	for(i=0;i<t;i++){
		scanf("%d%d",&n,&k);
		long int arr[n];
		for(j=0;j<n;j++){
			scanf("%li",&arr[j]);
		}
		sort(arr,arr+n);
		long int sum=0;
		for(p=n-1;p>=n-k && p>=0;p--){
			sum =sum + arr[p];
		}
		cout << sum << endl;

	}
    return 0;
}
