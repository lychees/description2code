#include <iostream>
#include<algorithm>

using namespace std;

int main()
{
	int x[1000];
	int y[1000];
int n,t;
cin>>t;
while(t--){
	cin>>n;
	for(int i=0;i<n;i++) cin>>x[i];
	for(int i=0;i<n;i++) cin>>y[i];
	sort(x,x+n);
	sort(y,y+n);
	int sum=0;
	for(int j=0;j<n;j++){
		sum+=x[j]*y[j];
	}
	cout<<sum<<endl;
}
return 0;
}
