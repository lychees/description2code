#include<bits/stdc++.h>
using namespace std;
int main()
{
	ios::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int arr[n],dp[n];
		for(int i=0;i<n;i++)
		cin>>arr[i];
		dp[0]=0;
		for(int i=1;i<n;i++)
		dp[i]=INT_MAX;
		for(int i=1;i<n;i++)
		{
			for(int j=0;j<i;j++)
			{
				if(arr[j]>=i-j)
					{
						if(dp[j]+1<dp[i])
						dp[i]=dp[j]+1;
					}
			}
		}
		cout<<dp[n-1]<<endl;
	}
	return 0;
}
