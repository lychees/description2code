#include<bits/stdc++.h>
using namespace std;
#define ll long long
double area(ll x1, ll y1, ll x2, ll y2, ll x3, ll y3)
{
   return abs((x1*(y2-y3) + x2*(y3-y1)+ x3*(y1-y2))/2.0);
}
int main()
{
    ll t;
    scanf("%lld",&t);
    while(t--)
    {
        ll x1,y1,x2,y2,x3,y3,x4,y4;
        double a,a1,a2,a3;
        scanf("%lld%lld%lld%lld%lld%lld%lld%lld",&x1,&y1,&x2,&y2,&x3,&y3,&x4,&y4);
        a=area(x1,y1,x2,y2,x3,y3);
        a1=area(x1,y1,x2,y2,x4,y4);
        a2=area(x2,y2,x3,y3,x4,y4);
        a3=area(x1,y1,x3,y3,x4,y4);
        if(a==(a1+a2+a3))
            printf("INSIDE\n");
        else
            printf("OUTSIDE\n");
    }
    return 0;
}
