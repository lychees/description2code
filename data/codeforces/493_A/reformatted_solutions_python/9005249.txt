import sys

home = sys.stdin.readline()
away = sys.stdin.readline()
home = home.replace('\n', '')
away = away.replace('\n', '')
l = [home, away]
# Tarjeta Amarilla, con 2 el jugador es expulsado (una roja = 2 amarillas)
players = [[0, 0] for i in range(99)]
n = int(sys.stdin.readline())
for _x in range(n):
    time, team, playerNumber, cardColor = sys.stdin.readline().split()
    # Resto uno porque es su lugar en la lista (indexa de 0)
    playerNumber = int(playerNumber) - 1
    if team == 'h':
        team = 0
    else:
        team = 1
    if cardColor == 'y':
        cardColor = 1
    else:
        cardColor = 2
    if players[playerNumber][team] < 2 and players[playerNumber][team] + \
            cardColor >= 2:
        print l[team] + ' ' + str(playerNumber + 1) + ' ' + time
    players[playerNumber][team] += cardColor
