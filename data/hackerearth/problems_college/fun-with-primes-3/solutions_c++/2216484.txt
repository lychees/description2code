#include <iostream>
using namespace std;

bool isprime(int n){
	if(n<=1)
		return false;
	else if(n<=3)
		return true;
	else if(n%2==0||n%3==0)
		return false;
	
	int i = 5;
	while(i*i<=n){
		if(n%i==0||n%(i+2)==0)
			return false;
		i = i+6;
	}
	return true;
}
int main()
{
    int t;
    cin>>t;
    int n;
    int sum;
    
    while(t--){
    	cin>>n;
    	sum = 0;
    	int z = n;
    	int r=0;
    	while(z!=0){
    		r = z%10;
    		sum = sum+r;
    		z = z/10;
    	}
    	if(isprime(sum))
    		cout<<"YES\n";
    	else cout<<"NO\n";
    }
    return 0;
}
