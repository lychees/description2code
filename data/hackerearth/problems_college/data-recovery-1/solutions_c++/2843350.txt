#include<bits/stdc++.h>
using namespace std;
class cmp{
   public:
   bool operator() (int a,int b){
   	if(a>b)
   	return true;
   	return false;
   }
};
priority_queue<int,vector<int>,cmp > q;
bool mrk[1000001];
int indeg[1000001];
vector<int> adj[1000000];
int main(){
   int n;
   for(int i=0;i<=1000000;i++)
   indeg[i]=0;
   scanf("%d",&n);
   memset(mrk,false,sizeof(mrk));
   while(n--){
   	int k;
    scanf("%d",&k);
    int pr;
    for(int i=0;i<k;i++){
    	int x;
    	scanf("%d",&x);
    	mrk[x]=true;
    	if(i>0)
    	adj[pr].push_back(x),indeg[x]++;
    	pr=x;
    }
   }
   for(int i=0;i<=1000000;i++){
   	if(mrk[i]==true&&indeg[i]==0)
   	q.push(i);
   }
   while(!q.empty()){
   	int temp=q.top();
   	q.pop();
   	printf("%d ",temp);
   	for(int i=0;i<adj[temp].size();i++){
   		indeg[adj[temp][i]]--;
   		if(indeg[adj[temp][i]]==0)
   		q.push(adj[temp][i]);
   	}
   }
   printf("\n");
   return 0;	
}
