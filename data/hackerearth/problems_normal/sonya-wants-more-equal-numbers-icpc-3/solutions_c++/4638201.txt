#include <bits/stdc++.h>

using namespace std;
 
#define MAX 100005
#define INF 2ll * (1e18)
#define MOD 1000000007
#define pb push_back
#define mp make_pair
#define ff first
#define ss second
#define ll long long

int a[MAX], last[MAX], sum[MAX], dp[MAX];
vector<int> ans[MAX];

int main(){
	ios_base::sync_with_stdio(false);
	int n;
	cin>>n;

	for(int i=1 ; i<=n ; i++){
		cin>>a[i];
		sum[i] = sum[i-1] + a[i];
	}

	set<int> Set;

	for(int i=1 ; i<=n ; i++)
		for(int j=i ; j<=n ; j++)
			Set.insert(sum[j] - sum[i-1]);

	vector<int> vec;
	for(set<int>::iterator i = Set.begin() ; i != Set.end() ; i++)
		vec.pb(*i);

	for(int i=0 ; i<vec.size() ; i++){
		int val = vec[i], temp = 0;
		for(int j=1 ; j<=n ; j++){
			if(sum[j] == val){
				dp[j] = max(1, dp[j-1]);
			} else {
				if(sum[j] > val && last[sum[j] - val] != 0)
					dp[j] = max(dp[last[sum[j] - val]] + 1, dp[j-1]);
				else
					dp[j] = dp[j-1];
			}
			last[sum[j]] = j;
			temp = max(temp, dp[j]);
		}

		for(int j=1 ; j<=n ; j++)
			last[sum[j]] = 0, dp[j] = 0;

		ans[temp].pb(val);
	}

	for(int i=MAX-1 ; i>0 ; i--){
		if(ans[i].size()){
			cout<<i<<"\n";
			for(int j=0 ; j<ans[i].size() ; j++)
				cout<<ans[i][j]<<" ";
			return 0;
		}
	}

	return 0;
}
