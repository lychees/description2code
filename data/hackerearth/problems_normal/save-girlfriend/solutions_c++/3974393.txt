#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
#define sn(x) scanf("%lld",&x)

ll arr[10010];

int main()
{
    ll t;
    sn(t);
    while(t--)
    {
        ll a,b;
        sn(a);
        sn(b);
        memset(arr,0,sizeof(arr));

        ll mx=-1;
        ll cnt=0;

        for (ll i=1;i*i<=b;i++)
        {
            ll sq = i*i;
            ll first_div_by_i = (((a-1)/i)+1)*i;

            for (ll j=first_div_by_i;j<=b;j+=i)
            {
                if (j<sq)
                    continue;
                else if (j==sq)
                {
                    arr[j-a]++;
                }
                else arr[j-a] += 2;
            }
        }
        for(ll i=a;i<=b;i++)
        {
            if(arr[i-a]>mx)
            {
                cnt=1;
                mx=arr[i-a];
            }
            else if(arr[i-a]==mx)
            cnt++;
        }
        printf("%lld\n",cnt+1);//<<endl;

    }
}
