#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    long long i,n,k,ans=0,pos,x;
    scanf("%lld%lld",&n,&k);
    long long a[n];
    for(i=0;i<n;i++)
     scanf("%lld",&a[i]);
    sort(a,a+n);
    x=1;
    for(pos=n-1;pos>=0;)
    {
    	for(i=0;i<k&&pos>=0;i++)
    	{
    		ans+=(x*a[pos]);
    		pos--;
    	}
    	x++;
    }
    printf("%lld\n",ans);
    return 0;
}
