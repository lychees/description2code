n = input()
c = map(int, raw_input().split())
i = 1
while(i < n):
	c[i] = c[i]+c[i-1]
	i+=1


def binSearch(indi, low, high):
	global c
	mid =  (low+high)/2
	if mid==0 and c[mid] >= indi:
		return mid
	elif c[mid]>=indi and c[mid-1]<indi:
		return mid
	elif c[mid] < indi:
		return binSearch(indi, mid+1, high)
	else:
		return binSearch(indi, low, mid-1)
	
q = input()
for _ in xrange(q):
	indi = input()
	print binSearch(indi, 0, n-1)+1
