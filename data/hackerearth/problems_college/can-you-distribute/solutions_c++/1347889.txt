#include<iostream>
using namespace std;

typedef long long ll;
#define MOD 10000007

ll c(ll n,ll k)
{
	ll res=1ll;
	if (k>n-k) {
		k=n-k;
	}
	for(ll i=0ll;i<k;i++) {
		res*=(n-i);
		res/=(i+1ll);
	}
	return res%MOD;
}

int main()
{
	ll n,k;
	cin>>n>>k;
	if(n>=k) {
		cout<<c(n-1ll,k-1ll)<<endl;
	} else cout<<"-1\n";
	return 0;
}
