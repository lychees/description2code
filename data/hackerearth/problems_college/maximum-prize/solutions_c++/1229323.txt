#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

const int N=1e5+10;
int a[N];

int main() {
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        int n,k;
        scanf("%d%d",&n,&k);
        for(int i=0;i<n;i++) {
            scanf("%d",&a[i]);
        }
        ll sum=0;
        sort(a,a+n);
        reverse(a,a+n);
        for(int i=0;i<min(n,k);i++) {
            sum+=a[i];
        }
        printf("%lld\012",sum);
    }
    return 0;
}
