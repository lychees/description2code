#include <iostream>
#include<stdio.h>
#include<string.h>
using namespace std;

int main()
{
    int t,y,hash[125],k;
    long int n,f,i;
    long long int j;
    char a[1000000],b[1000000];
    scanf("%d",&t);
    for(y=0;y<t;y++){
    	memset(hash,0,sizeof(hash));
    	scanf("%s",a);
    	i=0;f=0;
    	while(a[i]!='\0'){
    		k=a[i];
    		if(hash[k]==0){
    		b[f]=a[i];
    		hash[k]=1;
    		f++;
    	}
    	i++;
    	}
    	b[f]='\0';
    	n=strlen(b);
    	j=(n*(n+1))/2;
    	printf("%lld\n",j);
    }
    return 0;
}
