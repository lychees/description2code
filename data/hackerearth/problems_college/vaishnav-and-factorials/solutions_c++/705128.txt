#include <iostream>
#include <vector>
#include <stdio.h>

using namespace std;

int main()
{
	int t, n, m, temp, x, i, j, count;
	
	scanf("%d", &t );
	
	while( t-- ) {
		
		scanf("%d", &n );
		
		vector<int> A( 3000 );
		A[ 0 ] = 1;
		m = 1;
		
		for( j = 2;j <= n;j++ ) {
			
			temp = 0;
			
			for( i = 0;i < m;i++ ) {
				
				x = A[ i ]*j + temp;
				A[ i ] = x % 10;
				temp = x / 10;
			}
			
			while( temp > 0 ) {
				A[ m++ ] = temp % 10;
				temp /= 10;
			}
		}
		
		count = 0;
		for( i = 0;i < m;i++ )
			if( A[ i ] == 4 || A[ i ] == 7 )
				count++;
		printf("%d\n", count );		
	}
	return 0;
}
