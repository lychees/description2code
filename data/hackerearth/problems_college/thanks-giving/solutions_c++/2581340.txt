#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define S(x) sort(x.begin(),x.end())
#define INF_MAX 2147483647
#define INF_MIN -2147483647
#define INF_LL 9223372036854775807LL
#define EPS 1e-8
#define mod 1000000007
#define pb push_back
#define f first
#define s second
map<char, long long> m;
int main()
{
	std::ios_base::sync_with_stdio(0);
	string x;
	cin >> x;
	for (long long i = 0; i < x.length(); i++)
		m[x[i]]++;
	long long max = INF_MIN;
	char minc;
	map<char, long long>::iterator i = m.begin();
	for (map<char, long long>::iterator i = m.begin(); i != m.end(); i++)
	{
		if((*i).second > max)
		{
			max = (*i).second;
			minc = (*i).first;
		}
	}
	string y = "head rush", z = "cubing club";
	int flag1 = 0, flag2 = 0;
	for (long long i = 0; i < y.length(); i++)
		if(y[i] == minc)
		{
			flag1 = 1;
			break;
		}
	for (long long i = 0; i < z.length(); i++)
		if(z[i] == minc)
		{
			flag2 = 1;
			break;
		}
	if(flag1 && flag2)
		cout << "Thanks both";
	else if(flag1)
		cout << "Thanks head rush";
	else if(flag2)
		cout << "Thanks cubing club";
	else
		cout << "none" << endl;
}
