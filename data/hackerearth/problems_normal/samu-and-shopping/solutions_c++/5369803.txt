#include <iostream>
using namespace std;

int main()
{
    int t,n,a[100000][3];
    cin>>t;
    while(t--){
    	cin>>n;
    	for(int i=0;i<n;i++){
    		cin>>a[i][0]>>a[i][1]>>a[i][2];
    		if(i>0){
    			a[i][0]+=min(a[i-1][1],a[i-1][2]);
    			a[i][1]+=min(a[i-1][0],a[i-1][2]);
    			a[i][2]+=min(a[i-1][0],a[i-1][1]);
    		}
    	}
    	cout<<min(a[n-1][0],min(a[n-1][1],a[n-1][2]))<<"\n";
    }
    return 0;
}
