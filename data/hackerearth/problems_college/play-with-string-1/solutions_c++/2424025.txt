#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s,rev;
    	long long int i,j,s1,s2,l,flag=1;
    	cin>>s;
    	l=s.length();
    	for(i=l-1,j=0;i>=0;i--)
    	{
    		rev[j]=s[i];
    		j++;
    	}
    	for(i=1;i<l;i++)
    	{
    		s1=abs(s[i]-s[i-1]);
    		s2=abs(rev[i]-rev[i-1]);
 			if(s1!=s2)
    		{
    			flag=0;
    			break;
    		}
    	}
    	if(flag==1)
    	{
    		cout<<"Funny"<<endl;
    	}
    	else
    	{
    		cout<<"Not Funny"<<endl;
    	}
    }
    
    return 0;
}
