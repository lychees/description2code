#include <bits/stdc++.h>
using namespace std;
#define llint long long

int bs(llint k, vector<llint>& maxv){
	int high = maxv.size();
	int low =0;
	int mid = (high+low)/2;
	while (mid!=low){
		if (k>maxv[mid])
			high=mid;
		else
			low=mid;
		mid=(low+high)/2;
	}
	return mid;
}

int main(){
	int n;scanf("%d\n",&n);
	vector<int> input(n);
	for (int i=0;i<n-1;i++)
		scanf("%d ",&input[i]);
	scanf("%d\n",&input[n-1]);
	vector<llint> cum(n);
	cum[0]=input[0];
	for (int i=1;i<n;i++)
		cum[i] = cum[i-1]+input[i];
	vector<llint> maxv(n),minv(n);
	minv[0]=cum[0];
	for (int i=1;i<n;i++)
		minv[i]=min(minv[i-1],cum[i]);
	maxv[n-1]=cum[n-1];
	for (int i=n-2;i>=0;i--)
		maxv[i]=max(maxv[i+1],cum[i]);
	int gap=0;
	for (int i=0;i<n;i++){
		int k = bs(minv[i],maxv);
		if (maxv[k]-minv[i]>=0) gap = max(gap,k-i); 
	}
	int k = bs(0,maxv);
	gap = max(gap,k+1);
	int count =0;
	llint sum=0;
	for (int i=0;i<gap;i++)
		sum += input[i];
	if (sum>=0)count++;
	for (int i=gap;i<n;i++){
		sum = sum+input[i]-input[i-gap];
		if (sum>=0)count++;
	}
	if (count==0) {cout<<-1<<endl;return 0;}
	cout<<gap<<" "<<count<<endl;

	return 0;

}
