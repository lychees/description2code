#include <iostream>
using namespace std;
typedef long long LL;
const int MAX=10000005;
int t,n;
int pr[MAX],phi[MAX];
LL s[MAX];
void init(){
	for(int i=2;2*i<=MAX;++i){
		if(pr[i]==0){
			for(int j=2*i;j<MAX;j+=i)
				pr[j]=i;
		}
	}
	phi[1]=1;
	s[1]=1;
	for(int i=2;i<MAX;++i){
		int q=i;
		if(pr[q]==0)
			phi[q]=q-1;
		else{
			int x=pr[i];
			while(q%x==0)
				q/=x;
			phi[i]=phi[q]*((i/q)/x*(x-1));
		}
		s[i]=s[i-1]+phi[i];
	}
}

int main()
{
    ios::sync_with_stdio(false); cin.tie(NULL);
    init();
    cin>>t;
    while(t--){
    	cin>>n;
    	cout<<s[n]<<"\n";
    }
    return 0;
}
