#include <iostream> 

using namespace std;

int main () {
	int n, j, flag=0;
	cin>>n;
	
	char str[n+1][5000];
	
	for(int i=0;i<n+1;i++)
		cin.getline(str[i],5000);
		
	for(int i=1;i<n+1;i++)
	{
		flag = 0;
		
		for(j=0;j<5000;j++)
		{
			if(str[i][j]=='$')
			{
				cout<<"$";
				break;
			}
		}
		
		for(int k=j+1;k<5000;k++)
		{
			if(str[i][k]=='\0')
			break;
			
			if(str[i][k]=='0' && flag == 0)
			{
				flag=0;
			}
			
			else if(str[i][k]>'0' && str[i][k]<='9')
			{
				cout<<str[i][k];
				flag++;
			}
			
			else if(str[i][k]>='0' && str[i][k]<='9')
			cout<<str[i][k];
			
			else if(str[i][k]>=65 && str[i][k]<=90)
			break;
			
			else if(str[i][k]>=97 && str[i][k]<=122)
			break;
		}
		
		if(flag==0)
		cout<<"0";
		
		cout<<endl;
	}
	
	
  return 0;
}
