#include <bits/stdc++.h>

using namespace std;

int main()
{
    int T, i, n;
    string s;
    cin >> T;
    while(T--)
    {
        cin >> s;
        n = s.length();
        for(i=1;i<n;++i)
        {
            if(abs(s[i]-s[i-1]) != abs(s[n-i]-s[n-i-1]))
                break;
        }
        if(i == n)
            printf("Funny\n");
        else
            printf("Not Funny\n");
    }
    return 0;
}
