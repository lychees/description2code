#include<bits/stdc++.h>
using namespace std;
int arr[10005];
int main(){
	arr[1]=arr[3]=arr[4]=arr[6]=arr[8]=arr[9]=arr[11]=1;
	for( int i=12;i<=10000;i++){
		if( arr[i-1] == 0 ){
			arr[i] = 1;
		
		}
		else if( arr[i-4] == 0){
			arr[i] = 1;
		}
		else{
			arr[i] = 0;
		} 
		
	}
	int t;
	scanf("%d",&t);
	while(t--){
		int n;
		cin>>n;
		if( arr[n] == 1)
			cout<<"Yes\n";
		else
			cout<<"No\n";
	}
	
}
