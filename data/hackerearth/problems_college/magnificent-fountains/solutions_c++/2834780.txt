#include <iostream>
#include <map>
#include <cmath>
using namespace std;

#define mod 1000000007
map<int,int> m;
int countprimes(int x)
{
	map<int,int> mp;
	while(!(x%2))
	{
		mp[2]++;
		if(mp[2]>m[2])
			m[2]=mp[2];
		x/=2;
	}
	for(int i=3;i<=sqrt(x);i+=2)
	{
		while(!(x%i))
		{
			mp[i]++;
			if(mp[i]>m[i])
				m[i]=mp[i];
			x/=i;
		}
	}
	if(x>2)
	{
		mp[x]++;
		if(mp[x]>m[x])
			m[x]=mp[x];
	}
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	long long a[n],result=1;
    	m.clear();
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    		countprimes(a[i]);
    	}
    	map<int,int>::iterator i;
    	for(i=m.begin();i!=m.end();++i)
    		for(int j=0;j<i->second;j++)
    			result=(result*(i->first))%mod;
    	cout<<result<<endl;
    }
    return 0;
}
