int arr[N][N],n;

int fun(int a,vector<int> v){
    if(a == -1){
        rep(i,26) if(v[i] != 0) return 0;
        return 1;
    }
    int k = fun(a-1,v);
    if(k) return 1;
    vector<int> u(26);
    rep(i,26) u[i] = v[i];
    rep(i,26){
        u[i] -= arr[a][i];
        if(u[i] < 0) return 0;
    } 
    k = fun(a-1,u);
    return k;
}
