#include <iostream>
using namespace std;

#include <stdio.h>

int main()
{
	long long t,n,k,i,a[1000009],max,sum;
    scanf("%lld",&t);
    while(t--){
    	scanf("%lld %lld",&n,&k);
    	for(i=0;i<n;i++){
    		scanf("%lld",&a[i]);
    	}
    	max=0;
    	sum=0;
    	for(i=0;i<n;i++){
    		if(a[i]<=k)
    		sum=sum+a[i];
    		else{
    			if(max<sum)
    			max=sum;
    			sum=0;
    		}
    	}
    	if(max<sum)max=sum;
    	printf("%lld\n",max);
    }
    return 0;
}
