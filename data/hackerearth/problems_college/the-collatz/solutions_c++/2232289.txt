#include<bits/stdc++.h>
using namespace std;

typedef long long int ll;

ll t,n,sumx,sumy;

int main()
{
    cin >> t;
    
    while(t--)
    {
        cin >> n;
        
        sumx=0;
        sumy=0;
        ll cnt=0;

        while(1)
        {
            if(n==1)
            {
                sumy+=1;
                break;
            }
            if(n&1 && n!=1)
            {
                sumy+=n;
                n=3*n;
                n++;
                cnt++;
            }
            if(!(n&1) && n!=1)
            {
                sumx+=n;
                n=n/2;
                cnt++;
            }
        }
       // cout<<cnt<<endl;
        //cout<<sumx<<" "<<sumy<<endl;
        /*
        if(cnt==0)
        {
            cout<<"Tie"<<endl;
        }
        else if(sumx==sumy)
        {
            cout<<"Tie"<<endl;
        }
        */
        if((sumx%cnt)>(sumy%cnt))
        {
            cout<<"Even Rules"<<endl;
        }
        else if((sumx%cnt)<(sumy%cnt))
        {
            cout<<"Odd Rules"<<endl;
        }
        else
        {
        	cout<<"Tie"<<endl;
        }

    }
    return 0;
}
