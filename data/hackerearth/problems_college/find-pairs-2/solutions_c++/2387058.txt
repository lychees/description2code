#include<bits/stdc++.h>
#define m 2000007
using namespace std;

long long int a[m],countt[m];
int main() 
{
	long long int t;
	cin>>t;
	while(t--)
	{
		long long int n;
		scanf("%lld",&n);
		long long int i;
		for(i=0;i<=m;i++)
		countt[i] = 0;
		for(i = 0;i<n;i++)
		{
			cin>>a[i];
			countt[a[i]+1000006]++;
		}
		long long int ans = 0;
		for(i=0;i<m;i++)
		{
			if(countt[i]!=0)
			{
				ans +=(countt[i]*(countt[i]-1))/2;
			}
		}
		printf("%lld\n",ans+n);
	}
    return 0;
}
