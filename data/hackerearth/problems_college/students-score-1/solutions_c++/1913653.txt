    #include<stdio.h>
     
    int max(int x, int y)
    { return (y > x)? y : x; }
     
    int fun(int A[], int n)
    {
       int m=A[0],i;
       int c=A[0];
     
       for (int i=1;i<n;i++)
       {
            c=max(A[i],c+A[i]);
            m=max(m,c);
       }
       return m;
    }
     
    int main()
    {
       
       int n,A[1000005];
       scanf("%d",&n);
       for(int i=0;i<n;i++)
        scanf("%d",&A[i]);
       int sum=fun(A,n);
       if(sum>0)
        printf("%d\n",sum);
       else
        printf("0\n");
       return 0;
    }
