#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
long long int n,m,i,j,s,k,c;
cin>>n;
long long int a[n+10],b[n];
j=0;k=0;
s=0;
c=0;
for(i=0;i<n;i++)
{
	cin>>m;
	if(m%2==0)
	{
		a[j++]=m;
		s=s+m;
	}
	
	else
	{
		b[k++]=m;
		c=c+m;
	}
}
sort(a,a+j);
sort(b,b+k);
a[j++]=s;
i=j;
j=0;

while(j<k)
{
	a[i++]=b[j];
	j++;
}

a[i++]=c;
s=i;
for(i=0;i<s;i++)
cout<<a[i]<<" ";

    return 0;
}
