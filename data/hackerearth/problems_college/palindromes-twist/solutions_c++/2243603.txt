#define f(i,n) for(i=0;i<n;i++)
#define fd(i,n) for(i=n-1;i>=0;i--)

bool comp(const int& a, const int& b)
       {
          return a<b?false:true;//min_heap
       }
using namespace std;

int main()
{
	string s;
	int a,b,t,i,j,n;
	s(t);
	w(t)
	{
	    cin>>s;
	    n=s.length();
	    cin>>a>>b;
	    j=n-1;
	    int cost=0;
	    int f=0;
	    for(i=0;2*i<n;i++)
        {
            if(s[i]==s[j])
            {
                if(s[i]=='/')
                    cost+=2*min(a,b);
            }
            else if((s[i]=='a'&&s[j]=='/')||(s[j]=='a'&&s[i]=='/'))
            {
                cost+=a;
            }
            else if((s[i]=='b'&&s[j]=='/')||(s[j]=='b'&&s[i]=='/'))
            {
                cost+=b;
            }
            else
            {
                f=1;
                break;
            }
            j--;
        }
        if(f==1)
            cout<<"-1\n";
        else
            cout<<cost<<endl;
	}
	return 0;
}
