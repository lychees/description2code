using namespace std;
bool pri[1001]={0};
void init()
{
    pri[1]=1;
    for(int i=2;i<1001;i++)
    {
	if(!pri[i])
	{
	    for(int j=i*i;j<1001;j+=i)
		pri[j]=1;
	}
    }
}
int main()
{
    int t;
    init();
    cin>>t;
    while(t--)
    {
	int cur=0,ma=1;
	string s;
	cin>>s;
	for(int i=0;s[i]!='\0';i++)
	{
	    cur=0;
	    int j=i,k=i;
	    while(s[k]==s[j] && j<s.size() && k>=0)
	    {
		cur++;
		j++;
		k--;
	    }
	    ma=max(ma,2*cur-1);
	    j=i,k=i-1;
	    cur=0;
	    while(s[k]==s[j] && j<s.size() && k>=0)
	    {
		cur++;
		j++;
		k--;
	    }
	    ma=max(ma,2*cur);
	}
	if(!pri[ma])
	    cout<<"YES\n";
	else
	    cout<<"NO\n";
    }
    return 0;
}
