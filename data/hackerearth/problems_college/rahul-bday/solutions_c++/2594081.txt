#include <iostream>
using namespace std;

int main() {
	int no_of_test_cases;
	int black_gifts=0;
	int white_gifts=0;
	long long int black_gift_cost=0;
	long long int white_gift_cost=0;
	long long int exchange_cost=0;
	long long int bill=0;
	int i=0;
	cin >> no_of_test_cases;
	while(i < no_of_test_cases)
	{
		cin >>black_gifts>>white_gifts;
		cin >>black_gift_cost>>white_gift_cost>>exchange_cost;
		if(black_gift_cost>white_gift_cost)
		{
			if(black_gift_cost>(white_gift_cost+exchange_cost))
			{
				black_gift_cost=white_gift_cost+exchange_cost;
			}

		}
		else if(black_gift_cost<white_gift_cost)
		{
			if(white_gift_cost>(black_gift_cost+exchange_cost))
			{
				white_gift_cost=black_gift_cost+exchange_cost;
			}
		}
		bill=black_gifts*black_gift_cost+white_gifts*white_gift_cost;
		cout << bill;
		cout <<"\n";
		i++;
	}

	return 0;
}
