#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
typedef vector<ll> vll;
typedef vector<int> vi;
typedef vector< pair<ll, ll> > vpll;

map<ll, ll> mp;

int main() {

	int t;
	int n;
	int lo;
	int hi;
	
	cin >> t;
	for (int i = 0; i < t; i++) {
		cin >> n;
		for (int j = 0; j < n; j++) {
			cin >> lo >> hi;
			mp[lo]++;
			mp[hi]--;
		}
		ll ans = 0;
		ll maxCount = -1;
		for (auto x : mp) {
			ans += x.second;
			maxCount = max(ans, maxCount);
		}
		cout << maxCount << endl;
		mp.clear();
	}
    return 0;
}
