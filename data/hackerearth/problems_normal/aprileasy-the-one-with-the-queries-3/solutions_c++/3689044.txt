#include<bits/stdc++.h>
using namespace std;
set<int>v[30];
set<int>:: iterator it;
vector<int>xx[30];
#define pb push_back
#define mp make_pair
int main()
{
	int i;
	string s1,s2;
	cin>>s1;
	for(i=0;s1[i]!='\0';i++)
		v[s1[i]-'a'].insert(i);
	int q,j;
	cin>>q;
	while(q--)
	{
		cin>>s2;
		int p=0;
		int flag=0;
		
		for(i=0;s2[i]!='\0';i++)
		{
				it=v[s2[i]-'a'].lower_bound(p);
				if(it==v[s2[i]-'a'].end())
				{
					flag=1;
					break;
				}
				p=*it;
				xx[s2[i]-'a'].pb(*it);
				v[s2[i]-'a'].erase(it);
		}
		if(flag==0)
			printf("Yes\n");
		else
			printf("No\n");	
		for(i=0;i<26;i++)
		{
			for(j=0;j<xx[i].size();j++)
				v[i].insert(xx[i][j]);
			xx[i].clear();				
		}		
	}
	return 0;	
}
