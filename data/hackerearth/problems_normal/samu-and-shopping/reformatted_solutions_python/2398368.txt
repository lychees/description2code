for i in range(input()):
    n = input()
    arr = []
    for j in range(n):
        arr.append(map(int, raw_input().split()))
    mini = [[0 for i in range(3)]for j in range(n)]
    mini[0][:] = arr[0][:]
    for i in range(1, n):
        mini[i][0] = min(mini[i-1][1], mini[i-1][2]) + arr[i][0]
        mini[i][1] = min(mini[i-1][0], mini[i-1][2]) + arr[i][1]
        mini[i][2] = min(mini[i-1][1], mini[i-1][0]) + arr[i][2]
    print min(mini[n-1])
