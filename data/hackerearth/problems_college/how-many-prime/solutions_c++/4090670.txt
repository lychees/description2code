#include <iostream>
using namespace std;
#define limit 1000000
int prime[limit];
void sieve()
{	prime[0] =1;
	prime[1] =1;
	for (int p=2; p*p<=limit; p++)
    {	if (prime[p] == 0)
        {   for (int i=p*2; i<=limit; i += p)
                prime[i] = 1;
        }
    }
}	
int main() 
{	int t;
	cin>>t;
	sieve();
	while(t--)
	{	int n;
		cin>>n;
		int l =(n-1)/2;		int r =n-l;
		int flag =0;
		// if(!prime[n])			flag =1;
		while(l>0 && (prime[l] || prime[r]))
		{	l--;
			r++;
		}
		if(l>0 && !prime[l] && !prime[r])
			cout<<l<<" "<<r<<endl;
		else
			cout<<-1<<" "<<-1<<endl;
	}
return 0;
}
