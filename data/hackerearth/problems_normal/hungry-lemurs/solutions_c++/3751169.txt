#include <iostream>
#include <cmath>
using namespace std;

int main()
{
	std::ios::sync_with_stdio(false);
	int n,k,i,j,ans;
	cin>>n>>k;
	ans = 10000000 ;
	for(i=1;i<=100000;i++)
	{
		for(j=0;j<100000;j+=i)
		{
				if( j % i==0)
					ans = min (ans,(int)abs(n-j)+(int)abs(k-i));
		}
	}
	//cout<<"hi";
    cout << ans << endl;
    return 0;
}
