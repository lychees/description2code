#include <iostream>
#include<bits/stdc++.h>
#define ll long long
using namespace std;
char s1[1000004],s2[1000004];
#define mod 1000000007
int main()
{
   // cout << "Hello World!" << endl;
    ll t,n1,n2,f[26],i,j;
    scanf("%lld",&t);
    while(t--)
    {
    	scanf("%s",s1);
    	scanf("%s",s2);
    	n1=strlen(s1);
    	n2=strlen(s2);
    	memset(f,0,sizeof(f));
    	
    	for(i=0;i<n1;i++)
    	{
    	    f[s1[i]-'a']++;	
    	}
    	ll c=1;
    	for(i=0;i<n2;i++)
    	{
    		c*=f[s2[i]-'a'];
    		c= c % mod;
    		if(f[s2[i]-'a'])
    		 f[s2[i]-'a']--;
    	}
    	printf("%lld\n",c);
    }
    return 0;
}
