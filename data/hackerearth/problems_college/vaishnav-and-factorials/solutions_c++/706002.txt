    #define gc getchar

    #define gc getchar_unlocked
    #define pc putchar_unlocked

using namespace  std;
void scanInt(int *num){
    *num=0;
    char c=gc();
    while(c<'0'){
        c=gc();
    }
    while(c>='0' && c<='9'){
        *num = (*num<<3) + (*num<<1) + (c-'0');
        c=gc();
    }
}
int length;
int array[1000];

void factorial(int n){
    for(int i=0;i<1000;++i){
        array[i]=0;
    }
    length=0;
    if (n==1){
        array[0]=1;
        length = 1;
    }
    else {
        int i = 3,
            j,
            temp = 0,
            sum;
        length = 1;
        array[0] = 2;
        while (i != n+1){
            j = 0;
            while(j!=length){
                sum = (array[j]*i) + temp ;
                array[j] = sum % 10;
                temp = sum/10;
                ++j;
            }
            while (temp != 0) {
                array[j] = temp%10;
                ++j;
                temp = temp/10;
                ++length;
            }
        ++i;

        }
    }
}
int main(){
    int count,t,n;
    scanInt(&t);
    while(t--){
        count=0;
        scanInt(&n);
        factorial(n);
        for(int i=0;i<length;++i){
            if(array[i] == 4 || array[i] == 7){
                ++count;
            }
        }
        cout<<count<<"\n";
    }
    return 0;
}
