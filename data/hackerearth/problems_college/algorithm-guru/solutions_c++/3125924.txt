#include<iostream>
#include<vector>

using namespace std;


int main()
{
    int t;
    cin>>t;

    while(t!=0){

    int n,num;
    cin>>n>>num;

    int arr[n];

    for(int i=0;i<n;i++)
    {
        cin>>arr[i];
    }

    vector<int> vec(num+1,0);

    for(int i=1;i<=num;i++)
    {
        int add=0;
        for(int j=0;j<n;j++)
        {
            int diff = i-arr[j];

            if(diff>=0)
            {
                add = max(add,max(vec[diff]+arr[j],vec[i-1]));
            }
        }
        vec[i]+=add;
    }

    cout << vec[num] << endl;

    t--;
}
}
