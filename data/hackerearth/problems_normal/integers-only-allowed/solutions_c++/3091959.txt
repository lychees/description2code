#include<bits/stdc++.h>

using namespace std;
#define ll long long int
#define mq 1000000007
#define pb push_back
#define mk make_pair

//http://codeforces.com/blog/entry/4838
//Pick's Theorem

int gcd(int a, int b) {
	if(b==0)
		return a;

	return gcd(b, a%b);
}
int main() { 

	ll n ;
	cin>>n;

	ll x[n],y[n];

	for(int i=0; i<n; i++) {
		cin>>x[i]>>y[i];
	}

	ll area=0;

	for(int i=0; i<n; i++) {
		area+=x[i%n]*y[(i+1)%n]-x[(i+1)%n]*y[i%n];
	}

	area=abs(area)/2;
	//cout<<area<<endl;

	ll border=0;
	int a=0,b=0;
	for(int i=0; i<n; i++) {
		a=abs(x[(i+1)%n]-x[i%n]);
		b=abs(y[(i+1)%n]-y[i%n]);
		if(a==0) {
			border+=b-1;
		}
		else if (b==0) {
			border+=a-1;
		}
		else {
			border+=gcd(a,b)-1;
		}
	}
	border+=n;
	
	//cout<<border<<endl;
	ll i;
	i=area-border/2+1;
	cout<<i;
	return 0;
}
