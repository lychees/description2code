#include <bits/stdc++.h>
#define ll long long
#define sc(a) scanf("%lld", &a)
#define pr(a) printf("%lld", a)
#define MAX 1005

using namespace std;

ll arr[100005],l[100005],r[100005], flag[100005];

int main()
{
	ll n,i;
	sc(n);
	for (i = 1; i <= n; i++)
		sc(arr[i]);
	l[1] = arr[1];
	for (i = 2; i <= n; i++) {
		l[i] = max(arr[i], l[i-1]);
	}
	
	r[n] = arr[n];
	for (i = n-1; i >= 1; i--) {
		r[i] = max(arr[i], r[i+1]);
	}  
	for (i = 1; i <= n; i++) {
		if (l[i] <= arr[i] || r[i] <= arr[i]) {
			flag[i] = 1;
		}
	}
	for (i = 1; i<= n; i++) {
		if (flag[i])
			cout << i << " ";
	}
	cout << endl;
	
	return 0;
}
