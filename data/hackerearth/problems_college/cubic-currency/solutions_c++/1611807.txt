#include <iostream>
using namespace std;

int arr[22];
long long ways[10001];
int main()
{
    for(int i=1;i<=21;++i) arr[i]=i*i*i;
    ways[0]=1;
    for(int i=1;i<=21;++i)
    {
    	for(int j=arr[i];j<=10000;++j)
    	{
    		ways[j]+=ways[j-arr[i]];
    	}
    }
    int t,n;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	printf("%lld\n",ways[n]);
    }
    return 0;
}
