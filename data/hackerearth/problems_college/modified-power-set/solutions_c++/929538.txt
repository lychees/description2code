#include<stdio.h>
#include<iostream>
#include<algorithm>
#include <string.h>
#include <math.h>


using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while (t--) {
		char str[1000001];
		scanf("%s",str);
		int a[1000] = {0};
		long long int k = 0;
		for(long long i = 0 ;i <strlen(str);i++) {
			if(a[str[i]] == 0) {
				k++;
				a[str[i]]++;
			} 
		}
		long long int ans = 0;
	
		ans = (k*(k-1))/2;
		ans = ans + k;
		printf("%lld\n",ans);
		
		
	}


return 0;
}
