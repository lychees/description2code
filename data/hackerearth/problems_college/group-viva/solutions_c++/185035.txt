#include<cstdio>
#include <iostream>
#include<vector>
using namespace std;
int a[100001];
int b[100001],k;
vector<int>v;
long long gcd(long long a,long long b)
{
	if(b==0)
	return a;
	else
	return gcd(b,a%b);
}
void sieve()
{
	long long int i,j;
	for(i=2;i<=100000;i++)
	a[i]=1;
	for(i=2;i<=100000;i++)
	if(a[i]!=0)
	for(j=i*i;j<=100000;j+=i)
	a[j]=0;
	for(i=2;i<=100000;i++)
	if(a[i]!=0)
	v.push_back(i);
}
int main()
{
	int t,i;
	long long n,r,sum,sum1;
	scanf("%d",&t);
	sieve();
	while(t--)
	{
		sum1=1;
		scanf("%lld%lld",&n,&r);
		r--;
		int x=0;
		for(i=0;i<v.size();i++)
		{			
		//	cout
			if(r%v[i]==0)
			while(r%v[i]==0)
			{
				x++;
				r=r/v[i];
			}
			sum1*=(x+1);
			x=0;
		}
		if(r!=1)
		sum1*=2;
		//cout<<sum1<<" "<<v[0]<<endl;
		sum=n-sum1;
//		cout<<sum<<endl;
		cout<<(sum/gcd(n,sum))<<"/"<<(n/gcd(n,sum))<<endl;
	}
    return 0;
}
