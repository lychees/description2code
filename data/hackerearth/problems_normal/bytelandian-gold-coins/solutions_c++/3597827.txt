#include <bits/stdc++.h>
using namespace std;

map<int,long long> store;
long long answer(long long n){
	if(n == 0 || n == 1) return n;
	if(store.count(n)) return store[n];
	long long temp = max(n,answer(n/2)+answer(n/3)+answer(n/4));
	return store[n] = temp;
}
int main() {
	int n;
	while(cin>>n){
		cout<<answer(n)<<endl;
	}
	return 0;
}
