#include <iostream>
using namespace std;
#include<string.h>

bool comp(int A[],int B[]){
	for(int i=0;i<26;i++){
		if(A[i]!=B[i])
			return false;
	}
	return true;
}

bool fun(char A[],char B[]){
	long la,lb,i;
	int AA[26]={0},BB[26]={0};
	la=strlen(A);
	lb=strlen(B);
	for(i=0;i<la;i++){
		AA[A[i]-'a']++;
		BB[B[i]-'a']++;
	}
	for(i=la;i<lb;i++){
		if(comp(AA,BB))
			return true;
		BB[B[i]-'a']++;
		BB[B[i-la]-'a']--;
	}
	if(comp(AA,BB))
		return true;
	return false;
}

int main()
{
 int t;
 bool f;
 char A[1001],B[100001];
 cin>>t;
 while(t--){
  cin>>A;
  cin>>B;
  f=fun(A,B);
  if(f)
   cout<<"YES\n";
  else 
   cout<<"NO\n";
 }
 return 0;
}
