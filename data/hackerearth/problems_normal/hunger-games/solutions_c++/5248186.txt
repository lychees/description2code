#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n,a[10001],ans=0;
	cin>>n;
	for(int i=1;i<=n;i++) cin>>a[i];
	sort(a+1,a+n+1);
	//sort(a+(n/2+1),a+n+1,std::greater<int>());
	//for(int i=1;i<=n;i++) cout<<a[i]<<" "; cout<<endl;
	for(int i=1;i<=n-2;i++)
	{
		if(abs(a[i+2]-a[i])>ans) ans=abs(a[i+2]-a[i]);
	}
	cout<<ans;
    return 0;
}
