#define pb push_back
#define mp make_pair
#define sz(_vvv) (int)(_vvv.size())
#define all(_v) (_v).begin(),(_v).end()
#define Geti(x) scanf("%d",&x)
#define Getl(x) scanf("%ld",&x)
#define Getll(x) scanf("%64d",&x)

int main() {
	int t; cin >> t;
	while (t --) {
		int n; cin >> n;
		vector<int> v,vv;
		
		for (int i = 0; i < n; i ++) {
			int x; cin >> x;
			v.pb(x);
		}
		sort(all(v));
		int p = 0, q = n - 1;
		while (p < q) {
			int x = v[p ++] + v[q --];
			vv.pb(x);
		}
		sort(all(vv));
		cout << vv.back() - *vv.begin() << endl;
	}
	return 0;
}
