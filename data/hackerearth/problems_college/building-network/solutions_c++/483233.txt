#include <iostream>
#include <cstdio>
#include <cstdlib>
using namespace std;
int partition(long long int a[], int p, int q){
    int x = a[q];
    int i = p-1, tmp;
    for (int j = p; j<q; j++){
        if (a[j]<= x){
                   i = i+1;
                   tmp = a[i];
                   a[i] = a[j];
                   a[j] = tmp;
                   }
    }
        tmp = a[i+1];
        a[i+1] = a[q];
        a[q] = tmp;
        return i+1;
}

void quicks(long long int a[],int p,int q){
     if (p<q){
             int s = partition(a, p, q);
             quicks(a, p, s-1);
             quicks(a, s+1,q );
             }
      return;
}
void solve(){
     int N;
     cin >>N;
     long long int sum = 0;
     long long int a[1000];
     for (int i = 0; i<N; i++){
         cin >> a[i];
         }
     quicks(a,0, N-1);
   //  cout << "sorted array  :";
   //  for (int i = 0; i<N; i++)
    //    cout <<a[i]<<"  ";
     int x = a[0];
     for (int j = 1; j<N; j++){
         sum = sum + x + a[j];
         }
     cout <<sum<<"\n";
}


int main(){
    int tc;
    cin >>tc;
    while (tc--){
          solve ();
}
