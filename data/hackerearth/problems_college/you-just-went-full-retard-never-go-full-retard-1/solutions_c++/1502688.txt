#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    int t,flag;
    cin>>t;
    while(t--)
    {
    	unsigned long long int n,i;
    	cin>>n;
    	flag=n<2;

    	for(i=2;i<=sqrt(n);i++)
    	if(n%i==0)
    	{
    		flag=1;
    		break;
    	}
    	if(!flag)
    	cout<<"YES"<<endl;
    	else
    	cout<<"NO"<<endl;
    	
    	
    }
    return 0;
}
