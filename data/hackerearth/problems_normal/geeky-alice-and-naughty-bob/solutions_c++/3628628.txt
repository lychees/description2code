#include<stdio.h>

int main()
{
	long long int t,i,j,a,b,sum;
	int arr[]={0,1,2,6,6,3};
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld%lld",&a,&b);
		if(a>=6)
		{
			printf("%lld\n",(b-a+1)*9);
		}
		else if(b<6)
		{
			sum=0;
			for(i=a;i<=b;i++)
			sum+=arr[i];
			printf("%lld\n",sum);
		}
		else
		{
			sum=0;
			for(;a<6;a++)
			sum+=arr[a];
			sum+=((b-a+1)*9);
			printf("%lld\n",sum);
		}
	}
	return 0;
}
