#include <iostream>
#include<map>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int choc[n];
    for(int i=0;i<n;i++){
    	cin>>choc[i];
    }
    map<long,int> mymap;
    long counter=1;
    for(int i=0;i<n;i++){
    	int no_of_choc=choc[i];
    	while(no_of_choc){
    		mymap[counter]=i+1;
    		counter++;
    		no_of_choc--;
    	}
    }
    int q;
    cin>>q;
    while(q){
    	int index;
    	cin>>index;
    	cout<<mymap[index]<<endl;
    	
    	--q;
    }
    return 0;
}
