#include <bits/stdc++.h>

using namespace std;

set<int> s;

int main()
{
	int t,n,i;
	scanf("%d",&t);
	while (t --) {
		scanf("%d",&n);
		int a[n];
		for (i = 0; i < n; i++) {
			scanf("%d",&a[i]);
		}
		for (i = 0; i < n; i++) {
			s.insert(a[i]);
		}
		printf("%d\n",s.size());
		s.clear();
	}
	
	return 0;
}
