#include <bits/stdc++.h>

using namespace std;

const int N = 4e6 + 100;

int f[N];

int main() {
  // freopen("in", "r", stdin); freopen("out", "w", stdout);
  int n, k, m;
  scanf("%d %d %d", &n, &k, &m);
  set <int> bad;
  for (int i = 0; i < m; ++i) {
    int foo;
    scanf("%d", &foo);
    bad.insert(foo);
  }
  printf("%d", 1);
  int x = 2;
  for (int it = 1; it < n; ++it) {
    while (true) {
      bool prime = true;
      int mn = 1e9;
      for (int i = 2; i * i <= x; ++i) {
        if (x % i == 0) {
          mn = min(mn, f[i] + f[x / i]);
          prime = false;
        }
      }
      f[x] = mn;
      if (bad.count(x) == 0 && mn > k) {
        f[x] = 1;
        printf(" %d", x);
        break;
      }
      ++x;
    }
    ++x;
  }
  printf("\n");
  return 0;
}
