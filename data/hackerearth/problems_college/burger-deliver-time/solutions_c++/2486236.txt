#include <iostream>
#include <stdio.h>
#include <algorithm>
using namespace std;

struct order {
	int no;
	int deltime;
};

int func(struct order o1, struct order o2) {
	return o1.deltime < o2.deltime? 1: 0;
}

int main()
{
    int N;
    scanf("%d", &N);
    struct order o[N];
    for(int i=0; i<N; i++) {
    	int t, d;
    	scanf("%d %d", &t, &d);
    	o[i].no = i+1;
    	o[i].deltime = t+d;
    }
    
    sort(o, o+N, func);
    for(int i=0; i<N; i++)
    	printf("%d ", o[i].no);
    return 0;
}
