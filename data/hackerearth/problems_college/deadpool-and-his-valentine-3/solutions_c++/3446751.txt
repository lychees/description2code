#include <iostream>
using namespace std;

int main()
{
	int m,k,i,j,count=0,p,a[1001],l;
    cin>>k>>m;
    for(i=0;i<k;i++)
    {
    	cin>>p;
    	for(j=0;j<p;j++)
    	{
    		cin>>l;
    		a[l]++;
    	}
    }
     for(i=1;i<=m;i++)
    {
    	a[i]==0;
    }
    for(i=1;i<=m;i++)
    {
    	if(a[i]==0)
    	count++;
    }
   // cout<<count;
    if(count>0)
    cout<<"NO";
    else
    cout<<"YES";
    return 0;
}
