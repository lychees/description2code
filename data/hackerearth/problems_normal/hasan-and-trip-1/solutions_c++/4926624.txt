#include <stdio.h>
#include <math.h>
using namespace std;
double a[3][3001];
double dp[30001];
double cal(int index,int n)
{
    double x,y;
    int i;
    double pos,ans,max=-1000000000;
    if(dp[index]!=1000000000)
    {
        return dp[index];
    }
    if(index==n)
    {
        dp[index]=a[2][index];
        return a[2][index];
    }
    for(i=index+1;i<=n;i++)
    {
        x=a[0][i]-a[0][index];
        y=a[1][i]-a[1][index];
        pos=sqrtl(x*x+y*y);
      //  printf("in cal (%d ,%d)   and pos between %d and %d is %lf\n",index,n,i,index,pos);
        ans=cal(i,n)-pos+a[2][index];
        if(ans>max)
        {

            max=ans;
        }
    }
    dp[index]=max;
    return max;
}
int main()
{
    int n,i;
    scanf("%d",&n);
    for(i=1;i<=n;i++)
    {
        scanf("%lf %lf %lf",&a[0][i],&a[1][i],&a[2][i]);
        dp[i]=1000000000;
    }
    double ans = cal(1,n); //calculate from 1 to n
    
    printf("%.6lf\n",ans);
    return 0;
}
