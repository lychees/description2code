#include <iostream>
#include <string>
using namespace std;

int main()
{
    string s;
    int t,n,i,j,fg;
    cin>>t;
    while(t--){
    	cin>>s;fg=0;
    	n=s.length();
    	for(i=0,j=n-1;i<n && j>0;i++,j--){
    		if(abs(s[i+1]-s[i])!=abs(s[j-1]-s[j])){fg=1;break;}
    		
    	}
    	if(fg==1)
    		cout<<"Not Funny\n";
    	else
    		cout<<"Funny\n";
    }
    return 0;
}
