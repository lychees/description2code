#include<iostream>
#include<vector>
#include<fstream>
#include<algorithm>
#include<cmath>
double x[3030],y[3030],f[3030];
long int i,j,n;
 double dis(long int a,long int b)
{
    return sqrt((long double)((x[a]-x[b])*(x[a]-x[b])+(y[a]-y[b])*(y[a]-y[b])));
}
using namespace std;
void sol()
{

    long double dp[3030];
    cin>>n;
    for(i=1;i<=n;i++)
    {
        cin>>x[i]>>y[i]>>f[i];
    }
    for(i=1;i<=n;i++) dp[i]=-1e9;
    dp[1]=0;
    for(i=1;i<=n;i++)
    {
        dp[i]+=f[i];
        for(j=i+1;j<=n;j++)
            dp[j]=max(dp[j],dp[i]-dis(i,j));
    }
    cout.precision(6);
    cout<<fixed<<dp[n]<<endl;
}
int main()
{
    sol();
    return 0;
}
