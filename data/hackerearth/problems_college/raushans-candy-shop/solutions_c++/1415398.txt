#include <bits/stdc++.h>

using namespace std;
char s[1000011];
int main()
{
	int n,i,j,k;
	map<int,int> mp;
	map<int,int> :: iterator it;
	
	scanf("%d",&n);
	scanf("%s",s);
	long long ans = 0;
	int sum = 0;
	mp[0] = 1;
	for (i = 0; i < n; i++) {
		if (s[i] == 'S') {
			sum += 2;
		} else {
			sum -= 1;
		}
		it = mp.find(sum);
		if (it != mp.end()) {
			ans += it -> second;
		}
		mp[sum]++;
	}
	printf("%lld\n",ans);
	
	return 0;
}
