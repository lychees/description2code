def div(n):
    c = 0
    for i in xrange(1, int(n**.5)+1):
        if n % i == 0:
            c += 1
            if n/i != i:
                c += 1
    return c


x = [1]
for j in xrange(2, 10**4):
    x.append(x[-1]+j)
# print x
y = [div(i) for i in x]
# print y
for _ in xrange(input()):
    n = input()
    i = 0
    while y[i] <= n:
        i += 1
    print x[i]
