#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    string s[t];
    for(int i=0;i<t;i++)
    cin>>s[i];
    
    sort(s,s+t);
    int i=0;
    while(i<t)
    {
    	string temp=s[i];
    	int count=0;
    	while((s[i]==temp)&&(i<t))
    	{
    		count++;
    		i++;
    	}
    	cout<<temp<<" "<<count<<"\n";
    }
    return 0;
}
