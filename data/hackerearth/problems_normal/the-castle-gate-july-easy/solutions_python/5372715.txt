def count(num):
	res = 0
	for i in range(1, num + 1):
		for j in range(i + 1, num + 1):
			if (i ^ j) <= num:
				res += 1
	return res
testcase = int(input());
for i in range(testcase):
	num = int(input());
	res = count(num);
	print(res)
