#include <bits/stdc++.h>
typedef long long int ll;
using namespace std;
#define MAX 100005
char a[55];
bool dp[55][55];
int main(){
 int t;
 scanf("%d",&t);
 while(t--){
  memset(dp,false,sizeof dp);
  scanf("%s",a);
  reverse(a,a+strlen(a));
  int n = strlen(a);
  for(int i=0;i<n;i++) dp[i][i] = true;
  for(int len=2;len<=n;len++){
    for(int i=0;i<n-len+1;i++){
      int j = i+len-1;
      if(a[i]==a[j] && len==2) dp[i][j] = true;
      else if(a[i]==a[j] && dp[i+1][j-1]) dp[i][j] = true;
    }
  }
  int maxi = 0;
  for(int i=0;i<n;i++){
    if(dp[0][i]==true) maxi = max(maxi,i+1);
  }
  printf("%d\n",2*n-maxi);
 }
 return 0;
}
