#include <bits/stdc++.h>
using namespace std;
struct node
{
	int id;
	int start;
	int finish;
};
node a[1005];
int comp(node x,node y)
{
	if(x.finish==y.finish)
	return x.id<y.id;
	else
	return x.finish<y.finish;
}
int main()
{
	int n;
	int x;
	scanf("%d",&n);
	for(int i=0;i<n;i++)
	{
		a[i].id=i;
		scanf("%d%d",&a[i].start,&x);
		a[i].finish=a[i].start+x;
	}
	sort(a,a+n,comp);
	vector<int>res;
	for(int i=0;i<n;i++)
	{
		res.push_back(a[i].id);
	}
	for(int i=0;i<n;i++)
	{
		cout<<res[i]+1<<" ";
	}
	return 0;
}
