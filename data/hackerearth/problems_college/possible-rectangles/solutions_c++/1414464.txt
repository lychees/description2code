#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main() {
	int test;
	scanf("%d",&test);
	while(test--) {
		ll n,m,a;
		scanf("%lld %lld %lld",&n,&m,&a);
		ll ans = 0;
		for(ll i = 1;i*i <= a;i++) {
			if(a % i == 0) {
				ll d1 = i;
				ll d2 = a/i;
			
				if(d1<n && d2<m) {
					ans +=(n -d1)*(m - d2);
				}
				if(d1!=d2) {
					if(d1 <m && d2 <n) {
						ans +=(m - d1)*(n - d2);
					}	
				}
			}
		} 
		printf("%lld\n",ans);
	}

}
