#include <iostream>
#include <bits/stdc++.h>
#include <stdio.h>
using namespace std;

long sumOfDigits(long n)
{
	long s=0;
	while(n > 0)
	{
		s += n%10;
		n /= 10;
	}

	return s;

}

int main()
{
    int flag=0,k=5;
    long n,t;
	scanf("%ld",&t);
	assert(t <= 1000000 && t >= 1) ;
	while(t--)
	{
		k=5;
		flag=0;
		scanf("%ld",&n);
		assert(n <= 1000000000 && n >= 1) ;
		while(k--)
		{
			flag = ( n == 4 || n == 1 );
			
			if(flag)
			   break;
		    
		    n = sumOfDigits( (n*n) );
			
		}
		(flag==1) ? printf("YES\n") : printf("NO\n");
	}
	
    return 0;
}
