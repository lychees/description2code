
t = int(raw_input())

while t:
    t -= 1
    s = raw_input().strip()

    count = 0
    prev = 0
    curr = 0
    l = len(s)
    s = s.replace("a", "z")
    if "z" in s:
        for i, v in enumerate(s):
            if v == "z":
                count += l - i
                prev += curr
                count += prev

                curr = 0
            else:
                count += prev
                curr += 1

    print count
