#include<bits/stdc++.h>

using namespace std;

 
int main(){
     int t;
     cin>>t;
     while(t--)
     {
     	int n;
     	cin>>n;
     	int x,y,xm=-20000,ym=-20000,xmi=20000,ymi=20000;
     	for(int i=0;i<n;i++)
     	{
     		cin>>x>>y;
     		if(x<xmi)
    		xmi=x;
    		if(x>xm)
    		xm=x;
    		if(y<ymi)
    		ymi=y;
    		if(y>ym)
    		ym=y;
     	}
     	
         int ans=(ym-ymi)*(xm-xmi);
         if(ans==0)
         {
         	ans=xm+ym-xmi-ymi;
         }
         if(ans<0)ans=-ans;
     
         cout<<ans<<endl;
     }
 return 0;
}
