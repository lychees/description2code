#include <iostream>
using namespace std;

int main()
{
    long long int t, n, r;
    cin >> t;
    
    while(t--) {
    	cin >> n;
    	r = 0;
    	while(n != 0) {
    		if(n % 2)
    			r++;
    		n /= 2;
    	}
    	
    	if (r % 2)
    		cout << "odd\n";
    	else
    		cout << "even\n";	
    }
}
