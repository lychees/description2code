#include<stdio.h>
int gp[25],gc[25],gv[25],p,v,c,n;
bool sum(int i,int pp,int pv,int pc)
{
	if(pp==p&&pv==v&&pc==c)
	return true;
	if(pp>p||pv>v||pc>c||i>=n)
	{
		return false;
	}
	
	return sum(i+1,pp+gp[i],pv+gv[i],pc+gc[i])||sum(i+1,pp,pv,pc);
	
}
int main()
{
	int t;
	scanf("%d",&t);
	int cnt=1,i;
	while(cnt<=t)
	{
			scanf("%d",&p);
			scanf("%d",&v);
			scanf("%d",&c);
		scanf("%d",&n);
		for(i=0;i<n;++i)
		{
			scanf("%d",&gp[i]);
			scanf("%d",&gv[i]);
			scanf("%d",&gc[i]);
		}
	
		printf("Case #%d: ",cnt);
		if(sum(0,0,0,0))
		printf("yes\n");
		else
		printf("no\n");
		cnt++;
	}
	return 0;
}
