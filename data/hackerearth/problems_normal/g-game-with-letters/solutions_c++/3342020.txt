#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--){
    	int n;
    	string s;
    	long long score[26]={0};
    	cin>>n>>s;
    	int slen=s.length();
    	for(int i=0;i<slen;i++) score[s[i]-'a']++;
    	long long ans=1,maxp=0,p,len=100000000;
    	for(int i=0;i<n;i++){
    		p=0;
    		cin>>s;
    		slen=s.length();
    		for(int i=0;i<slen;i++) p+=(score[s[i]-'a']);
    		if(p>maxp || (p==maxp && slen<len)){
    			maxp=p;
    			ans=i+1;
    			len=slen;
    		}
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
