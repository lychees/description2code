#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

const ll MOD = 1e9 + 7;

ll modn(ll x)
{
	return x % MOD;
}


int main()
{
    ll t, n, k, ans;
    scanf("%d", &t);
    while(t--)
    {
    	scanf("%lld %lld", &n, &k);
    	ans = 1;
    	for(int i = 0; i < n; i++, k--)
    		ans = modn(ans * k);
    	printf("%lld\n", ans);
    }
    return 0;
}
