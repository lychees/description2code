#include <iostream>
using namespace std;

int main()
{
    int t;cin>>t;
    while(t--) {
    	string a, b;
    	cin>>a>>b;
    	int has[256] = {};
    	for(int i = 0; i < a.size() ; i++) {
    		has[a[i]]++;
    	}
    	int flag = 0;
    	for(int i = 0; i < b.size() ; i++) {
    		if(has[b[i]] ) {
    			flag = 1;
    			break;
    		}
    	}
    	if(flag) {
    		cout<<"YES"<<endl;
    	}else {
    		cout<<"NO"<<endl;
    	}
    }
    return 0;
}
