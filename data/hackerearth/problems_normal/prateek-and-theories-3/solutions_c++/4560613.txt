#include <iostream>
#include <algorithm>
using namespace std;

int main(int argc, char const *argv[]) {
	int T;
	cin >> T;
	for(int i = 0; i < T; i++) {
		double time[20000];
		int N, max = 0, popular = 0;
		cin >> N;
		for(int j = 0; j < N; j++) {
			long int start, end;
			cin >> start >> end;
			time[2 * j] = (double)start;
			time[2 * j + 1] = (double)end - 0.99; 
		}
		sort(time, time + 2 * N);
		for(int j = 0; j < 2 * N; j++) {
			if(time[j] - (long int)time[j] == 0.0)
				popular++;
			else
				popular--;
			if(popular > max)
				max = popular;
		}
		cout << max << endl;
	}
	return 0;
}
