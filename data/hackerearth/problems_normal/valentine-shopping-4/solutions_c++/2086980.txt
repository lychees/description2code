#include <bits/stdc++.h>
using namespace std;
 
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n,m;
    	scanf("%d%d",&n,&m);
    	vector < vector < vector<int> > > tube;
		for(int i=0;i<n;i++)
		{
	   		tube.push_back(vector<vector<int> >());
	 		for(int j=0;j<m;j++)
	   		{
   		 	    tube[i].push_back(vector<int>());
	       		for(int k=0;k<3;k++)
 	  		    {
 	  		    	int temp;
 	  		    	scanf("%d",&temp);
 		       		tube[i][j].push_back(temp);
	       		}
	   		}
		}
		vector<int>res[1000];
		vector<int>::iterator it;
		double min_price;
		int min_price_index;
		for (int i = 0; i < tube.size(); i++)
    	{
    		min_price=1000;
    		min_price_index=10001;
    		for (int j = 0; j < tube[i].size(); j++)
      		{
      			int a,b,c;
      			double total;
      			a=tube[i][j][0];
      			b=tube[i][j][1];
      			c=tube[i][j][2];
      			total=((100-a)*(100-b)*(100-c)*100)/1000000.0;
      			if(total<min_price)
      			{
      				min_price=total;
      				min_price_index=j+1;
      			}
      		}
      		res[i].push_back(min_price_index);
    	}
    	for(int i=0;i<n;i++)
    	{
    		for(it=res[i].begin();it!=res[i].end();it++)
    		{
    			printf("%d ",(*it));
    		}
       	}
       	printf("\n");
        
    }
}
