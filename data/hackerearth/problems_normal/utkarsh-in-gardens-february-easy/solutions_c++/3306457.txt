#include <cstdio>
#include <iostream>
using namespace std;
typedef long long ll;
const int inf=2009;
ll num[inf][35];
int main()
{
	int n,a;
	scanf("%d",&n);
	for (int i=0;i<n;++i)
	{
		int s=0,e=0;
		for (int j=0;j<n;++j)
		{
			scanf("%d",&a);
			if (a==1)
				num[i][s]|=1ll<<e;
			++e;
			if (e==61)
				e=0,s++;
		}
	}
	int mx=n/61;
	long long sol=0;
	for (int i=0;i<n;++i)
		for (int j=i+1;j<n;++j)
		{
			int r=0;
			for (int u=0;u<=mx;++u)
			{
				long long k=num[i][u]&num[j][u];
				while(k)
				{
					r++;
					k-=k&-k;
				}
			}
			sol+=r*(r-1)/2;
		}
	printf("%lld\n",sol/2ll);
}
