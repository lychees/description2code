#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int a[100010];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int n,q;
	cin>>n>>q;
	ll int i,x,y,ms=0,d=0;
	unordered_map<ll int, ll int> mp;
	bool flag=1;
	for(i=0;i<n;i++)
	{
		cin>>x>>y;
		ms+=min(x,y);
		mp[max(x,y)-min(x,y)]++;
		d+=(max(x,y)-min(x,y));
	}
	unordered_map<ll int, ll int>::iterator it;
	for(it=mp.begin();it!=mp.end();it++)
	{
		if(it->second & 1ll)
		{
			flag=0;
			break;
		}
	}
	if(flag)
	{
		ms+=d/2;
	}
	for(i=0;i<q;i++)
	{
		cin>>x;
		if(flag && x==ms)
		cout<<"YES\n";
		else
		cout<<"NO\n";
	}
	return 0;
}
