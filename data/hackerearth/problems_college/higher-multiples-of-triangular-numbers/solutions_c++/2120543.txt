#include <iostream>
#include <cmath>
using namespace std;

int nD(long long x)
{
	int s;
	
	if(x==1)
	return 1;
	else
    s=2;
	
	if((long long)sqrt(x)*(long long)sqrt(x)==x)
	{
		s++;
	}
	for(long long i=2;i<sqrt(x);i++)
	{
		if(x%i==0)
		{
			s+=2;
		}
	}
	
	return s;
}

int main()
{
    int t;
    cin >> t;
    for(int i=0;i<t;i++)
    {
    	int n;
    	cin >> n;
    	
    	long long x=1,d=2;
    	while(1)
    	{
    		if(nD(x)>n)
    		{
    			break;
    		}
    		else
    		{
    			x+=d;
    			d++;
    		}
    	}
    	cout << x << endl;
    }
}
