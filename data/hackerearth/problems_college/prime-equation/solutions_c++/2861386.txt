#include <bits/stdc++.h>
using namespace std;
#define llong long long
#define long llong

const int n_max = 10000010;

vector<int> P;
bool M[10000010];
int n, n_test;

void init ()
{
    fill(M + 2, M + n_max, true);

    int k = 0;
    for (int i = 2; i < n_max; ++i)
    if (M[i])
    {
        if (k > 2) P.emplace_back(i - k);
        k = i;

        for (int j = i + i; j < n_max; j += i) M[j] = false;
    }

    sort(P.begin(), P.end());
    P.resize(unique(P.begin(), P.end()) - P.begin());
}

int main ()
{
    ios::sync_with_stdio(false); cin.tie(nullptr);
	if (ifstream("test.inp")) cin.rdbuf((new ifstream("test.inp"))->rdbuf());

    init();

    cin >> n_test;
    while (n_test--)
    {
        cin >> n;
        if (n > 400) { cout << "NO\n"; continue; }

        bool ok = false;
        for (int i = 0; i < P.size(); ++i)
        for (int j = i; j < P.size(); ++j) ok |= (P[i] + P[j] == n);

        cout << (ok ? "YES" : "NO") << "\n";
    }
}
