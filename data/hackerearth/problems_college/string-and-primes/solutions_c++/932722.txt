#include <bits/stdc++.h>
using namespace std;

char ch[100005];
char ans[100005];
bool prime[150];

void sieve(){
	prime[0]= prime[1]= true;
	for(int i=2;i<(int)sqrt(150);i++){
		if(!prime[i])
		for(int j=i;j*i<150;j++){
			prime[i*j]= true;
		}
	}
}


int main()
{
	sieve();
    int tc;
    scanf("%d", &tc);
    
    while(tc--){
    	int k=0;
    	cin>> ch;
    	for(int i=0;i<(int)strlen(ch);i++){
    		int n= (int)(ch[i]);
    		if(prime[n]){
    			ans[k]= ch[i];
    			k++;
    		}
    	}
    	if((int)strlen(ch)==k){
    		printf("NULL\n");
    	}
    	else{
    		
    		for(int i=0;i<k;i++){
    			printf("%c", ans[i]);
    			
    		}
    		printf("\n");
    	}
    	
    }
    
    return 0;
}
