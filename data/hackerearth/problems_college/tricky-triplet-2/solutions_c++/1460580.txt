#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;cin>>n;
    long long ans=0;
    vector<int>v(n);
    for(int i=0;i<n;i++)cin>>v[i];
    for(int i=0;i<n;i++){
    	for(int j=i+1;j<n;j++){
    		for(int k=j+1;k<n;k++){
    			if(v[j]-v[i]==v[k]-v[j])ans++;
    		}
    	}
    }
    cout<<ans<<endl;
    return 0;
}
