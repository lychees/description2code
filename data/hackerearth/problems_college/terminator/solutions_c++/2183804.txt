#include<bits/stdc++.h>
using namespace std;

int main()
{
	vector<int> a;
	char s[500];
	int t,i,j,n;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%s",s);
		n=strlen(s);
		a.resize(n);
		for(i=0;i<n;i++)
			a[i]=0;
		for(i=0;i<n;i++)
		{
			if(s[i]=='x')
			{
				for(j=i+1;j<n;j++)
				{
					if(s[j]=='o')
						a[j]=1;
					else if(s[j]=='*')
						break;
				}
				for(j=i-1;j>=0;j--)
				{
					if(s[j]=='o')
						a[j]=1;
					else if(s[j]=='*')
						break;
				}
			}
		}
		for(i=0;i<n;i++)
			if(a[i]==0)
				printf("%c",s[i]);
		printf("\n");
	}
	return 0;
}
