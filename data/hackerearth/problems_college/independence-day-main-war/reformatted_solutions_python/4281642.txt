T = int(raw_input().strip())
for i in xrange(T):
    N = int(raw_input().strip())
    X, Y = [], []
    for j in xrange(N):
        x, y = raw_input().strip().split(' ')
        X.append(float(x))
        Y.append(float(y))
    a = sum(X)
    b = sum(Y)
    a = a/float(N)
    b = b/float(N)
    print '(%.2f , %.2f)' % (a, b)
