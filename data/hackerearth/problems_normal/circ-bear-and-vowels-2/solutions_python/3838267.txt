
vowels = ['a','e','i','o','u','y']

test_cases = input()
while(test_cases > 0):
    sentence = raw_input()
    count_cons = 0
    count_vows = 0

    counter = 0
    length = len(sentence)
    flag = False
    while(counter < length):
        if(sentence[counter] in vowels):
            count_vows += 1
        else:
            count_cons += 1

        if(counter + 2 < length):
            if(sentence[counter] not in vowels and sentence[counter + 1] not in vowels and sentence[counter + 2] not in vowels):
                print("hard")
                flag = True
                break
        counter += 1

    if(not flag):
        if(count_cons > count_vows):
            print("hard")
        else:
            print("easy")
    test_cases -= 1
