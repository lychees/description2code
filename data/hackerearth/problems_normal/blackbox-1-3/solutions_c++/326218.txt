#include<iostream>
using namespace std;
int gcd(long long int a,long long int b)
{
	if(a!=b)
	{
		if(a==0)
			return b;
		gcd(b%a,a);
	}
	//return 0;
}
int main()
{
	long long int  t;
	cin>>t;
	while(t--)
	{
		long long int a,b,c,x,y,z,l,m,n;
		cin>>a>>b>>c;
		x=a*a;
		y=b*b;
		z=c*c;
		if(x+y==z || x+z==y || y+z==x)
		{
			l=gcd(a,b);
			m=gcd(a,c);
			n=gcd(b,c);
			if(l==1 && m==1 && n==1)
				cout<<"YES"<<endl;
			else
				cout<<"NO"<<endl;

		}
		else 
			cout<<"NO"<<endl;
		
	}
	return 0;
}
