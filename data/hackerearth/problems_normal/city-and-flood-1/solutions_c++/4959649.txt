#define loop(n) for(int i=0; i<n; i++)
#define initialize(a,s,n) loop(n){a[i] = i;s[i] = 1;}

int root(int *a, int i)
{
    while(a[i]!=i){
        a[i] = a[a[i]];
        i = a[i];
    }
    return i;
}

bool find(int *a, int i, int j)
{
    if(root(a,i) == root(a,j))
        return true;
    return false;
}

void un(int *a, int *s, int i, int j)
{
    int rootA = root(a,i);
    int rootB = root(a,j);
    if(s[rootA] < s[rootB]){
        a[rootA] = a[rootB];
        s[rootB] += s[rootA];
    }
    else{
        a[rootB] = a[rootA];
        s[rootA] += s[rootB];
    }
}

int main()
{
    int n, k, i, j;
    cin >> n >> k;
    int a[n],s[n];
    initialize(a,s,n);
    while(k--){
        cin >> i >> j;
        un(a,s,i-1,j-1);
    }
    loop(n)
    s[i] = 0;
    int count = 0;
    for(int i=0; i<n; i++){
        int r = root(a,a[i]);
        if(s[r] != 1){
            count++;
            s[r] = 1;
        }
    }
    cout << count << endl;
    return 0;
}
