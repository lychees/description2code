#include<iostream>
#include<cstdio>
#include<cstring>
#include<vector>
#include<utility>
#include<climits>
#include<cstdlib>
#include<algorithm>
#include<map>
using namespace std;
int main()
{
	int t,n;
	scanf("%d",&t);
	while(t--)
	{
		char a[100000]={0};
		char b[100000]={0};
		char c[100000]={0};
		int x=0,y=0;
		scanf("%d",&n);
		scanf("%s",a);
		for(int i=0;i<n;i++)
		{
			if(i%2)
			{
				c[i]='N';
				b[i]='Y';
			}
			else
			{
				c[i]='Y';
				b[i]='N';
			}
		}
		for(int i=0;i<n;i++)
		{
			if(a[i]==b[i])
			y++;
			else
			x++;
		}
		printf("%d\n",min(y,x));
	}	
	return 0;
}
