t = input()
while (t > 0):
    n = input()
    count = 0
    flag = 0
    while (n > 0):
        rem = n % 10
        n /= 10
        if (rem == 4 or rem == 7):
            count += 1
    while (count > 0):
        flag = 1
        rem = count % 10
        if (rem == 4 or rem == 7):
            count /= 10
        else:
            flag = 2
            print "NO"
            break
    if (count == 0 and flag == 0):
        print "NO"
    elif (flag != 2):
        print "YES"
    t -= 1
