#include <iostream>
using namespace std;
long N,M,Q;
long unburnt=0;
int mat[1001][1001];
void destroy(int i,int j) {
	if(mat[i][j]==-1)
	return;
	if(mat[i][j]==0)
	return;
	if(unburnt==0)
	return;
	if(mat[i][j]==1)
	{
		mat[i][j]=-1;
		unburnt=unburnt-1;
		if(i+1<=N && j<=M && mat[i+1][j]==1)
		destroy(i+1,j);
		if(i-1>=1 && j<=M && mat[i-1][j]==1)
		destroy(i-1,j);
		if(j+1 <=M && i<=N && mat[i][j+1]==1)
		destroy(i,j+1);
		if(j-1>=1 && i<=N && mat[i][j-1]==1)
		destroy(i,j-1);
	}
	
}
int main()
{
    int qi,qj;
    cin>>N>>M>>Q;
    mat[N][M];
    for(int i=1;i<=N;i++)
    for(int j=1;j<=M;j++)
    {
    	scanf("%d",&mat[i][j]);
    	if(mat[i][j]==1)
    	unburnt+=1;
    }
    for(int i=1;i<=Q;i++) {
    	scanf("%d%d",&qi,&qj);
    	destroy(qi,qj);
    	printf("%ld\n",unburnt);
    }
    return 0;
}
