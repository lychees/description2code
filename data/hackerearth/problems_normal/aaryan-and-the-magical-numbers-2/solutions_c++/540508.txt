#include <iostream>
#include<cstdio>
#include<math.h>
#include<string.h>
using namespace std;
typedef unsigned long long int ll;
int num[10]={0,1,2,2,2,3,3,4,4,5};
int main()
{
    ll t,len,i,j,c,k;
    char n[50];
    scanf("%llu",&t);
    while(t--)
    {
    	scanf("%s",n);
    	len=strlen(n);
    	c=0;
    	k=0;
    	for(i=0;i<len;i++)
    	{
    		k=k*10+(n[i]-'0');
    	}
    	for(j=0;j<len;j++)
    	{
    		c=c+(num[n[j]-'0'])*pow(6,len-j-1);
    		if(n[j]=='2' || n[j]=='3' || n[j]=='5' || n[j]=='7')
    		{
    			c--;
    			break;
    		}
    	}
    	//res=answer(n,len);
    	printf("%llu\n",k-c);
    }
    return 0;
}
