#include <bits/stdc++.h>
using namespace std;
char s[55];
int ind[255];
string a;
int cmp(char a, char b)
{
	if(ind[a] < ind[b])return 1;
	return 0;
	
}
int main()
{
   // cout << "Hello World!" << endl;
   int t;
   scanf("%d",&t);
   while(t--)
   {
   	scanf("%s",s);
   	int i;
   	for(i=0;s[i];i++)ind[s[i]] = i;
   	cin>>a;
   	sort(a.begin(),a.end(),cmp);
   	cout<<a<<endl;
   }
    return 0;
}
