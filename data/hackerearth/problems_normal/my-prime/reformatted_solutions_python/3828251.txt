import sys

res = []
n = int(raw_input())
lst = map(int, raw_input().split())
for num in lst:
    flag = 1
    remove_list = []
    for m in res:
        if not (num % m and m % num):
            if m > num:
                remove_list.append(m)
            else:
                flag = 0
                break
    l3 = [x for x in res if x not in remove_list]
    res = l3
    if flag:
        res.append(num)
for m in res:
    sys.stdout.write(str(m) + " ")
