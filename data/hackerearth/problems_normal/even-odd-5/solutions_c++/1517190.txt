using namespace std;
int main()	{
	int t;
	unsigned long long R, C, evenCount = 0, oddCount = 0, evenN, oddN, temp;
	scanf("%d",&t);
	while(t--)	{
		evenCount = oddCount = evenN = oddN = temp = 0;
		scanf("%llu %llu",&R,&C);
		//printf("%llu %llu\n",r,c);

		if(R <= C)	{
			evenN = (R - 2) / 2 + 1;
			oddN = (R - 1) / 2 + 1;
			evenCount = evenN * (evenN + 1);
			oddCount = oddN * oddN;
		}
		else	{
			evenN = (C - 2) / 2 + 1;
			oddN = (C - 1) / 2 + 1;
			evenCount = evenN * (evenN + 1);
			oddCount = oddN * oddN;

			temp = (R - C) / 2;

			if(R % 2 == 0)	{
				if(C % 2 == 0)	{
					evenN = temp;
					oddN = temp;
				}
				else if(C % 2 == 1)	{
					evenN = temp + 1;
					oddN = temp;
				}
			}
			else	{
				if(C % 2 == 0)	{
					evenN = temp;
					oddN = temp + 1;
				}
				else	{
					evenN = oddN = temp;
				}
			}
			evenCount += (evenN * C);
			oddCount += (oddN * C);

		}
		printf("%llu %llu\n", evenCount, oddCount);
	}
}
