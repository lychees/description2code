#include <bits/stdc++.h>
#define ll long long int
using namespace std;
int main(){
ll n,m,i,p,max=INT_MIN;
string s,s1;
cin>>n>>m;
map<string,ll> a;
for(i=0;i<m;i++){
    cin>>s;
    if (a.find(s)==a.end()) a[s]=0;
    cin>>p;
    a[s]+=p;
    if (max<a[s]) {
        s1=s;
        max=a[s];
    }
}
cout<<s1<<endl;
return 0;
}
