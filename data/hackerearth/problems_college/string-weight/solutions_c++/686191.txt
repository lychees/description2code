#include <bits/stdc++.h>
using namespace std;

int main() 
{
    int t,n,i;
    char s[10005],x,y;
    cin>>t;
    while(t--) {
        scanf("%s",s);
        cin>>n;
        int a[26];
        for(i = 0; i < 26; i++) {
            a[i] = 0;
        }
        for(i = 0; s[i]; i++) {
            a[s[i]-97]++;
        }
        for(i = 0; i < n; i++) {
            cin>>x;
            a[x-97] = 0;
        }
        long long int s = 0;
        for(i = 0; i < 26; i++) {
            if(a[i] > 0)
            s += a[i]*(i+1);
        }
        cout<<s<<"\n";
    }
	return 0;
}
