    #include <bits/stdc++.h>
    using namespace std;
    int N,M;
    int coin[55];
    int count( int S[], int m, int n ){
        long long int dp[n+1];
        memset(dp, 0, sizeof(dp));
        dp[0] = 1;
        for(int i=0; i<m; i++)
            for(int j=S[i]; j<=n; j++)
                dp[j] += 1ll*dp[j-S[i]];
        return dp[n];
    }
    int c;
    int main()
    {
    	scanf("%d%d",&N,&M);
    	memset(coin, false, sizeof(coin));
    	for( int i =0; i< M; i++){
    		scanf("%d",&coin[i]);
    	}
    	printf("%lld\n", count(coin,M,N));
    	return 0;
    }
