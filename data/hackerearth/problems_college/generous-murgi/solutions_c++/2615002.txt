#include <bits/stdc++.h>
using namespace std;

int main()
{
    int i,j,ans,t,x,y,val,val1,val2;
    scanf("%d",&t);
    while(t--)
    {
    	ans=0;
    	scanf("%d%d",&x,&y);
    	val=x-y;
    	if(val>0)
    	for(i=1;i<=sqrt(val);i++)
    	{
    		if(val%i==0)
    		{
    			if(i*i!=val)
    			{
    				if(y!=0)
    				{
    				if(y-i<0)
    					ans++;
    				}
    				else
    					ans++;
    				val2=val/i;
    				if(y!=0)
    				{
    				if(y-val2<0)
    					ans++;
    				}
    				else
    					ans++;
    			}
    			else
    			{
    				if(y!=0)
    				{
    				if(y-i<0)
    					ans++;
    				}
    				else
    					ans++;
    			}
    		}
    	}
    	if(val==0)
    		ans=-1;
    	printf("%d\n",ans);
    }
    return 0;
}
