#include <cstdio>
#include <algorithm>
#define sd(x) scanf("%d",&x)
#define pd(x) printf("%s\n",x)
#define FOR(i,j,k) for(auto i=j;i<k;i++)
#define rep(i,n) FOR(i,0,n)
using namespace std;
bool willsurvive(int *s,int *a){
	if(*s && *a){
		if(*s>*a)
			return willsurvive(s+1,a+1);
		return false;
	}	
	return true;
}
int main(void){
	int t;
	sd(t);
	while(t--){
		int n;
		sd(n);
		int *s = new int[n];
		int *a = new int[n];
		rep(i,n)
			sd(s[i]);
		rep(i,n)
			sd(a[i]);
		sort(s,s+n);
		sort(a,a+n);
		pd(willsurvive(s,a)?"YES":"NO");
	}
}
