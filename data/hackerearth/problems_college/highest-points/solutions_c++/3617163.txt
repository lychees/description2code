#include <iostream>
using namespace std;

int main()
{
	int N,d1,d2,d3;
	cin>>N;
	int temp=0;
	for(int i=0;i<N;i++){
	
			if(i==0){
				cin>>d1>>d2>>d3;
				temp=d1*3+d2-d3;
				continue;
			}
			else{
				cin>>d1>>d2>>d3;
				if(d1*3+d2-d3 > temp){
					temp = d1*3+d2-d3;
				}
			}
	}
	cout<<temp;
    return 0;
}
