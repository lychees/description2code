#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,n;
	string s;
	map<string,int> m;
	set<string> s1;
	set<string>::iterator it;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>s;
		
		if(s1.find(s)!=s1.end())
		m[s]+=1;
		else
		{
			s1.insert(s);
			m[s]=1;
		}
		
	}
	for (it=s1.begin(); it!=s1.end(); ++it)
    cout<< *it<<" "<<m[*it]<<endl;
  
	return 0;
}
