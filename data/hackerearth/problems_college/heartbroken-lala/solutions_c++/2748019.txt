int get(LL len , LL pos)
{
	if(pos == 1) return 0;
	else if(len == 1) return 0;
	LL leftHalf , rightHalf;
	leftHalf = rightHalf = len / 2;
	if(pos <= leftHalf) return get(leftHalf , pos);
	if(pos == leftHalf + 1) return 0;
	return !get(rightHalf , len - pos + 1);
}

int solve(LL k)
{
	LL i = 1;
	while(i <= k)
	{
		i = i * 2 + 1;
	}
	return get(i , k);
}
