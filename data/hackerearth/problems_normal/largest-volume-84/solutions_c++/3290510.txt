#include<iostream>
#include<cmath>
#include<iomanip>
using namespace std;
 
int main()
{
   double r1,r2,s;
   cin>>r1>>s;
   double sine=0.5,cos;
   double a,max=0,height;
 
   while(sine<=1)
   {
        height=s*sine;
        cos=sqrt(1-(sine*sine));
        r2=r1+s*cos;
        a=(M_PI/3)*(r1*r1+r1*r2+r2*r2)*height;
        if(max<a)
        max=a;
        sine+=0.000001;
   }
   cout<<fixed;
    cout<<setprecision(2)<<max;
 
   return 0;
}
