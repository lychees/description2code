#include <iostream>
using namespace std;

int dp(int i,int n,int arr[],int a[])
{
	if (i>=n)
		return 0;
	else if (arr[i]!=-1)
		return arr[i];
	else
	{
		arr[i]=dp(i+1,n,arr,a);
		arr[i]=max(arr[i],a[i]+dp(i+2,n,arr,a));
		return arr[i];
	}
}
int main()
{
	int t,i,a[100001],arr[100001],n;
	cin>>t;
	while (t)
	{
		t-=1;
		cin>>n;
		for (i=0;i<n;i++)
		{
			cin>>a[i];
			arr[i]=-1;
		}
		cout<<dp(0,n,arr,a)<<endl;
	}
    return 0;
}
