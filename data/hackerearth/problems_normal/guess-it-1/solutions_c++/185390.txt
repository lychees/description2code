#include<bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef vector<int> vi;
typedef pair<int, int> ii;
typedef vector<ii> vii;
typedef set<int> si;
typedef map<string, int> msi;
#define MOD 1000000009


template <typename X> X gcd(X a, X b){if(!b)return a; else return gcd(b, a%b);}

int main()
{
int t;
cin>>t;
while(t--)
{
string str;
cin>>str;
int ans=0;
int len=str.length();
for(int i=0;i<len;i++)
{
	if(str[i]!='_')
	{
	ans+=(str[i] - '0');
	if(ans>=9) ans=ans%9;
	}
}
if(ans==0) printf("0 or 9\n");
else printf("%d\n",9-ans);
}
}
