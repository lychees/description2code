#include<iostream>
using namespace std;
int main()
{
	int n;
	cin>>n;
	int arr[n][3],p[n],ctr[n];
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<3;j++)
		{
			cin>>arr[i][j];
		}
	}
	for(int t=0;t<n;t++)
	{
		p[t]=(3*arr[t][0])+(1*arr[t][1])+(-1*arr[t][2]);
		
	}
	for(int i=0;i<n;i++)
	{
     ctr[i]=0;
		for(int j=0;j<n;j++)
		{
			if(p[j]<=p[i])
			{
				ctr[i]++;
			}
		}
		if(ctr[i]==n)
		{
			cout<<p[i];
		}
	}
	return 0;
}
