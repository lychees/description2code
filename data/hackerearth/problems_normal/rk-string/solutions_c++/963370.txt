LL solve(string str,char ch,int cnt)
{
	//deb(ch<<" "<<cnt<<" "<<str)
	
	LL ans=0;
	int n=str.length(),i,k;
	int tt=0;
	i=k=0;
	if(str[0] == ch) tmp[k++]=0;
	while(i<n)
	{
		if(str[i]==ch) ++i;
		int cnt=0;
		while(i<n && str[i]!=ch){++i;++cnt;}
		tmp[k++]=cnt;
	}
	//if(str[n-1] == ch) tmp[k++]=0;
	//if(tt<cnt) return 0;
	//for(int i=0;i<k;++i) deb(tmp[i])
	if(cnt)
	for(int i=cnt;i<k;++i)
	{
		ans+=(tmp[i]+1)*(tmp[i-cnt]+1);
	}
	else{
		
		for(int i=cnt;i<k;++i)
		if(tmp[i])
		{
			ans+=((tmp[i]+1)*(tmp[i]))/2;
		}
	}
	return ans;
}
