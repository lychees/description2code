#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t,n,k,i,j;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	j=n;
    	k=j/8;
    	i=j%8;
    	while((i%6!=0||j%8!=0)&&(j>=0))
    	{
    		j-=1;
    		k=j/8;
    		i=(n-j)%6;
    		//printf("%d\t%d\n",j,i);
    	}
    	if(j!=-1)
    	printf("%d\n",k+((n-j)/6));
    	else
    	printf("-1\n");
    	
    	
    }
    return 0;
}
