#include <iostream>
#include <string.h>
using namespace std;

int main() {
int t;
cin>>t;
while(t--)
{
	char a[10000],b[10000];
	cin>>a>>b;
	int cc1[26]={0},cc2[26]={0},f=0;
	int l1=strlen(a);
	int l2=strlen(b);
	if(l1!=l2)
	{
		cout<<"No\n";
	}
	else
	{
		for(int i=0;i<l1;i++)
		{
			cc1[a[i]-97]++;
			cc2[b[i]-97]++;
			
		}
		for(int i=0;i<26;i++)
		{
			if(cc1[i]!=cc2[i])
			{
				f=1;break;
			}
		}
		if(f==0)
		cout<<"Yes\n";
		else cout<<"No\n";
	}
}
	return 0;
}
