#include <iostream>
#include<vector>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n,j,i;
    cin>>n;
    string s;
    cin>>s;
    int size = s.length();
    vector <int> v(n);
    for(j=0; j<n; j++){
    	v[j] = j+1;
    }
    i=0;
    j=0;
    while(v.size()>1){
    	
    	i = i%size;
    	j = j%(v.size());
    	if(s[i] == 'a'){
    		j++;
    	}
    	if(s[i] == 'b'){
    		v.erase(v.begin()+j);
    		/*for(int i=0;i<v.size();i++)
    			cout<<v[i]<<" ";
    		cout<<endl;*/
    	}
    	i++;
    	
    }
    cout<<v[0]<<endl;
    return 0;
}
