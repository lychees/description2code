using namespace std;
ll MOD = ((ll)1e9) + 7;
long long int power(long long int base,long long int pow)
{
	long long int sum=1;
	if(base==0)
	return 0;
	while(pow)
	{
		if(pow&1)
		{
			sum*=base;
			if(sum>=MOD)
			sum%=MOD;
		}
		base*=base;
		if(base>=MOD)
		base%=MOD;
		pow/=2;
	}
	return sum;
}

		ll ans=power(3,nnn);
		ans-=3*power(2,nnn);
		ans+=3;
		ans%=MOD;
		ans+=MOD;
		ans%=MOD;
		cout<<ans<<endl;
		ans=0;
	}
    return 0;
}
