#include <iostream>
#include <cctype>
#include <string>
#include <map>

using namespace std;

int main()
{
	string str;
	cin >> str;
	unsigned long long len = str.length();
	unsigned long long slen = 0;
	map<char, unsigned long long> smap;

	for (unsigned long long int i = 0; i < len; ) {		
		char ch = str[i];
		i++;
		unsigned long long x = 0;
		do {
			x = x * 10 + (str[i] - '0');
			i++;
		} while (!islower(str[i]) && i != len);
		smap[ch] += x;	
		slen += x;
	}

	int q;
	cin >> q;
	while (q--) {
		unsigned long long k;
		cin >> k;

		if (k > slen) {
			cout << -1 << endl;
		}
		else  {
			unsigned long long sum = 0;

			map<char, unsigned long long>::iterator itr = smap.begin();

			while (sum < k) {
				sum += itr->second;
				if (sum >= k) {
					cout << itr->first << endl;
					break;
				}
				++itr;
			}
		}
	}
}
