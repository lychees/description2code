#include <iostream>
#include<cmath>
#include<string>
using namespace std;
bool pal(string s,int a,int b)
{
	for(int i=a,j=b;i<j;i++,j--)
	if(s[i]!=s[j])return  false;
	return true;
}

int main()
{int t;
cin>>t;
while(t--)
{ int max=0;
	string s;
	cin>>s;
	for(int i=0;i<s.length();i++)
	{
		for(int j=i+1;j<s.length();j++)    //abcdser
		{
			if(pal(s,i,j))
			{
				int l=abs(i-j)+1;
			     if(max<l)
			      {
			      max=l;
			      }
			  }
		}
	}
	cout<<max<<endl;
}
  
   
    return 0;
}
