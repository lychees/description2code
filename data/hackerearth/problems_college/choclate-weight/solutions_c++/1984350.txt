#include <iostream>
#include <algorithm>
using namespace std;

bool check(long long int a[],long long int n,long long int q);
bool docheck(long long int a[],long long int n,long long int z);
int main()
{
    long long int t;
    cin>>t;
    long long int i;
    long long int a[t];
    for(i=0;i<t;i++)
    {
    	cin>>a[i];
    }
    sort(a,a+t);
    long long int q;
    cin>>q;
    while(q--)
    {
    	long long int z;
    	cin>>z;
    	if(check(a,t,z)==true || docheck(a,t,z)==true)
    	{
    		cout<<"I Like You"<<endl;
    	}

    	else{
    		cout<<"I Hate You"<<endl;
    	}
    }
    return 0;
}

bool check(long long int a[],long long int n,long long int q)
{
	long long int lo=0,hi=n-1;

	while(lo<=hi)
	{
		long long int mid=lo + (hi-lo)/2;
		if(a[mid]==q)
		{
			return true;
		}
		else if(a[mid]>q)
		{
			hi=mid-1;
		}
		else{
			lo=mid+1;
		}
	}

	return false;
}


bool docheck(long long int a[],long long int n,long long int z)
{
	long long int i,j;
	for(i=0;i<n;i++)
	{
		for(j=i+1;j<n;j++)
		{
			if(a[i]+a[j]==z)
			{
				return true;
			}
			else if(a[i]+ a[j]>z)
			{
				break;
			}
		}
	}
	return false;
}
