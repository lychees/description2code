#include <iostream>
using namespace std;

int main()
{
    int no;
    cin>>no;
   for(int i=0;i<no;++i)
   {
   	   int n;
   	   cin>>n;
   	   
   	   if(n==1)
   	      cout<<"0"<<endl;
   	   else
   	     {
   	     	int tmp=(n*(n-1))/2;
   	        cout<<tmp<<endl;
   	  }
   	  }
   	     
    return 0;

}
