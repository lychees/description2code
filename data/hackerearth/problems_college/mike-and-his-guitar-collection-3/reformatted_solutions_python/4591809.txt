from itertools import combinations
for _ in range(input()):
    arr = []

    n = input()
    a = map(int, raw_input().split())
    for i in range(n+1):
        comb = combinations(a, i)
        arr.append(comb)
    count = 0
    sumset = set()
    # Print the obtained combinations
    for it in arr:
        for i in list(it):
            # print i
            sumset.add(sum(list(i)))
            count += 1
    # print count
    # print sumset
    print len(sumset)
