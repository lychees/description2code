#include <iostream>
using namespace std;

int main()
{
    string str;
    cin >> str;
    if(str == ""){
    	cout << 0 << endl;
    	return 0;
    }
    char prev = str[0];
    int c = 1, count = 0;
    for(int i=1;i<str.size();i++){
    	if(str[i] == prev)
    		c++;
    	else{
    		for(int k=1;k<=c;k++){
    			count += (c-k+1);
    		}
    		c = 1;
    		prev = str[i];
    	}
    }
    for(int k=1;k<=c;k++){
		count += (c-k+1);
	}
    cout << count << endl;
    return 0;
}
