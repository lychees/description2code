void printMat(int a[100][100],int n){
	
	REP(i,n){
		REP(j,n)
			printf("%d ",a[i][j]);
		printf("\n");
	}
	printf("\n");
}

int main(){

	int q, n;
	int a[100][100];
	int r1[100][100], r2[100][100], r3[100][100], r4[100][100];
	scanf("%d%d", &n,&q);
	REP(i,n)
		REP(j,n)
		scanf("%d",&a[i][j]);
		
	REP(i,n){
		REP(j,n){
			r2[i][j]=a[n-1-j][i];
		}
	}	
	REP(i,n){
		REP(j,n){
			r3[i][j]=r2[n-1-j][i];
		}
	}
	
	REP(i,n){
		REP(j,n){
			r4[i][j]=r3[n-1-j][i];
		}
	}
	
	REP(i,q){
		int x;
		scanf("%d",&x);
		x=x%360;
		if(x==0)
			printMat(a,n);
		else if(x==90)
			printMat(r2,n);
		else if(x==180)
			printMat(r3,n);
		else if(x==270)
			printMat(r4,n);
	}
	
	return 0;
}
