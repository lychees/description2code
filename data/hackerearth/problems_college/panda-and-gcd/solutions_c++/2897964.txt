#define all(a) a.begin(),a.end()
#define bitcnt(x) __builtin_popcountll(x) 
#define MOD 1000000007
#define total 500005
#define MAX 2000002
#define CMAX 26
 
typedef unsigned long long int uint64;
typedef long long int int64;
 
int a[1000005];

int cnt[1000005];
int64 ans[1000006];
int64 subset[1000006];

int main(){
	int n,i;
	ans[0]=1;	
	for(int i=1;i<=1000000;i++){
		ans[i]=(int64)2*ans[i-1];
		if(ans[i]>=MOD)
		ans[i]%=MOD;
	}
	
	cin>>n;
	int mx=0;
	
	for(i=0;i<n;i++){
		scanf("%d",&a[i]);
		cnt[a[i]]++;
		mx=max(mx,a[i]);
	}
	
	for(i=mx;i>=1;i--){
		int64 sub=0;
		int64 add=cnt[i];
		//cout<<i<<" ";
		for(int j=2;j*i<=mx;j++){
			add+=cnt[j*i];
			sub+=subset[j*i];
			if(sub>=MOD)
			sub%=MOD;
		}
		//cout<<add<<" "<<sub<<" ";
		int64 fin=ans[add]-1;
		fin-=sub;
		//cout<<fin<<endl;
		while(fin<0)
		fin+=MOD;
		subset[i]=fin;
	}
	
	for(i=0;i<n;i++)
	printf("%d ",subset[a[i]]);
	
	return 0;
}
