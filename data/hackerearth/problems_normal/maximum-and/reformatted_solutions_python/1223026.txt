t = int(raw_input())


def test(x, i):
    return (x >> i) & 1 == 1


def slow(a, b):
    return max(i & j for i in range(a, b + 1) for j in range(i + 1, b + 1))


def f(a, b):
    if b - a == 1:
        return a & b
    return max(b & (b - 1), (b - 1) & (b - 2))


for _ in range(t):
    a, b = map(int, raw_input().split())
    print(f(a, b))
