#include<iostream>
#include<string.h>
#include<math.h>
using namespace std;
int main()
{
	int t;
    cin>>t;
    while(t--)
    {
    	char str[200001];
    	int len,i=0,c1=0,c0=0;
    	cin>>str;
    	len=strlen(str);
    	while(i<len)
    	{
    		if(str[i]=='1')
    		c1++;
    		i++;
		}
		c0=len-c1;
		cout<<abs(c1-c0)<<endl;
	}
}
