#include <bits/stdc++.h>

using namespace std;

const int MOD=1000000007;
const int DIV2=(MOD+1)/2;
int comp[1000001];
int sum[1000001];

int calc(int n)
{
    if(n<=2)
        return 1;
    n-=2;
    int ans=comp[n];
    int sym=0;
    for(int i=1; i*2<=n; i++)
        sym=(sym+comp[i])%MOD;
    ans=(1LL*(ans-sym+MOD+1)%MOD*DIV2%MOD+sym+1)%MOD;
    return ans;
}

int main()
{
    comp[0]=1;
    sum[0]=1;
    for(int i=1; i<=1000000; i++)
    {
        comp[i]=sum[i-1];
        sum[i]=(sum[i-1]+comp[i])%MOD;
    }
    int T;
    scanf("%d", &T);
    while(T--)
    {
        int N;
        scanf("%d", &N);
        printf("%d\n", calc(N));
    }
    return 0;
}
