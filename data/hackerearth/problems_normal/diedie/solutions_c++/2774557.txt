//Calculating Inverse Modulus 

// When we do (a^b)%m then we can also do ((a%m)^(b%m))%m...Sometimes in some problems we may have large base which can be treated like this

#include<bits/stdc++.h>
using namespace std;

#define mp make_pair
#define pb push_back

typedef long long ll;

const long long MOD=1e9+7;

long long modex(long long base,long long expo){

    if(expo==0)
        return 1;

    long long f=modex(base,expo/2);


    if((expo&1)==0)
         return ((f%MOD)*(f%MOD))%MOD;


    return (((f*f)%MOD)*(base%MOD))%MOD;

}

void solve(){

    ll n;
 
    cin>>n;

    ll ans=modex(2,n-1);

    ans=modex(ans,MOD-2);

    cout<<ans%MOD<<'\n';
}

int main(){

    int test;

    cin>>test;

    while(test--)
        solve();  
 
    return 0;

}
