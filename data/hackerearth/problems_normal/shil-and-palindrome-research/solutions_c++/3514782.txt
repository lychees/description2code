#define foru(i,a,b) for(i=a;i<=b;i++)
#define ford(i,a,b) for(i=a;i>=b;i--)

#define MOD 1000000007

int arr[N][29];
string s;
int n;
void sum(int i){
	char cur = s[i - 1];
	while(i <= n){
		arr[i][cur - 'a']++;
		i = i + (i & (-i));
	}
}

void query(int *a, int index){
	int i;
	while(index > 0){
		foru(i,0,25)
			a[i] = a[i] + arr[index][i];
		index = index - (index & (-index));
	}
}

void update(int index, char c){
	char prev = s[index - 1];
	while(index <= n){
		arr[index][prev - 'a']--;
		arr[index][c - 'a']++;
		index = index + (index &(-index));
	}
}

			s[index-1] = c;

		else if(choice == 2){
			counter++;

			int a[29] = {0};

			query(a,r);
