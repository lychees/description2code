#include <iostream>
#include<cstdio>
using namespace std;
#define ULL unsigned long long 
int main() 
{
	int t;
	ULL n,cnt;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%llu",&n);
		cnt=0;
		while(n)
		{
			if( ((n%10)==4) || ((n%10)==7) )
			++cnt;
			n/=10;
		}
		if( (cnt!=4) && (cnt!=7) ) 
			printf("NO\n");
		else
			printf("YES\n");
	}
	return 0;
}
