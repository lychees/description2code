#include<bits/stdc++.h>
using namespace std;

long long int a[100005];
long long int b[100005];
long long int n;
void update(long long int x,long long int v) {
	for(;x<=n;x+=x&(-x))
	   b[x]+=v;
}

long long int query(long long int x) {
   long long int sum=0;
	for(;x>0;x-=x&(-x))
	  sum+=b[x];
   return sum;
}

int main() {
    long long int i,j,q;
    cin >> n;
    for(i=1;i<=n;i++) {
        cin >> a[i];
        update(i,a[i]);
    }
    cin >> q;
    while(q--) {
      cin >> i >> j;
      cout << query(j) -query(i-1) << endl;
    }
	return 0;
}
