#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s;
    getline(cin,s);
    int n=s.size(),sum=0;
    for(int i=0;i<n;i++)
    {
    	sum=sum+s[i];
    }
    //cout<<sum;
    sum=sum/n;
    if(sum%2==0)
    {
    	reverse(s.begin(),s.end());
    	cout<<s;
    	//cout<<s<<endl;
    }
    else
    {
    	char str[200];
    	strcpy(str,s.c_str());
    	int cnt[200]={0},maxi=-1;
    	for(int i=0;i<n;i++)
    	{
    		cnt[str[i]]++;
    	}char ans;
    	for(int i=0;i<n;i++)
    	{
    		if(cnt[str[i]]>maxi)
    		{	maxi=cnt[str[i]];
    			ans=str[i];
    		}
    	}
    	cout<<ans;
    }
    
}
