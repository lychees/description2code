#include <iostream>
#include<cstring>
#include<stack>
using namespace std;


char c1[1000001],c2[1000001];
int main()
{
     
    int t,i,j,l1,l2;
    pair<char, int> a;
    stack < pair<char, int> > st;
    cin>>t;
    while(t--)
    {	int flag = 0;
    	cin>>c1>>c2;
    	
    	l1 = strlen(c1);
    	int top = -1;
    	for(i = 0 ; i<l1 ; i++)
    	{
    		if(c1[i] == '#')
    			continue;
    		else
    		{	a.first = c1[i];
    			a.second = i;
    			st.push(a);
    		}
    	}
    	
    	for(i = l1-1; i>=0; i--)
    	{
    		if(c2[i] == '#')
    			continue;
    		else
    			if(c2[i] == 'A' && c2[i] == st.top().first)
    			{
    				if(i>st.top().second)
    				{
    					flag=1;
    					break;
    				}
    				else
    				{
    					st.pop();
    				}
    				
    			}
    			else if(c2[i] == 'B' && c2[i] == st.top().first)
    			{
    				if(i < st.top().second)
    				{
    					flag=1;
    					break;
    				}
    				else
    				{
    					st.pop();
    				}
    				
    			}
    			else
    			{
    				flag = 1;
    				break;
    			}
    	}	
    	
    	if(flag == 1)
    		cout<<"No\n";
    	else
    		cout<<"Yes\n";
    	
    }
    
    return 0;
}
