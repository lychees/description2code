#include <iostream>
using namespace std;
long long int left( long long int n ,int t)
{
while(t--)
{
	if(n%2==0)
	{
		n-=n/2;
	}
	else
	{
		n-=(n+1)/2;
	}
	n=n-n/4;
	
}
return n;
}
int main()
{
   long long int n;
   long int t;
   int test;
   cin>>test;
   while(test--)
   {
   cin>>n;
   cin>>t;
   cout<<left(n,t);
   cout<<"\n";
   }
   
    return 0;
}
