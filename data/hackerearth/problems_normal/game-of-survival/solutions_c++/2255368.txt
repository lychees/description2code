#include <bits/stdc++.h>
using namespace std;

int main()
{
    int tc;
    long int N;
    int sharpness[100001];
    int strength[100001];
    cin>>tc;
    while(tc--)
    {
    	int count=0;
    	cin>>N;
    	for(long int i=0;i<N;i++)
    	{
    		cin>>sharpness[i];
    	}
    	for(long int i=0;i<N;i++)
    	{
    		cin>>strength[i];
    	}
    	sort(sharpness,sharpness+N);
    	sort(strength,strength+N);
    	for(int i=0;i<N;i++)
    	{
    		if(strength[i]>=sharpness[i])
    		{
    			count++;
    			break;
    		}
    	}
    	if(count==0)
    	{
    		cout<<"YES"<<endl;
    	}
    	else
    	{
    		cout<<"NO"<<endl;
    	}
    }
}
