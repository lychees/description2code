#include <bits/stdc++.h>
using namespace std;

long a[2005];

int main(){
	long n;
	cin>>n;
	long kk;
	cin>>kk;
	for(int i=0;i<n;i++){
		cin>>a[i];
		a[i]%=kk;
	}
	int c=0;
	for(int i=0;i<n;i++){
		if(a[i]==0){
			c+=((n-i-1)*(n-i-2)/2);
			continue;
		}
		for(int j=i+1;j<n;j++){
			for(int k=j+1;k<n;k++){
				long long xx=(a[i]*a[j]*a[k])%kk;
				if(xx==0){
					c++;
				}
			}
		}
	}
	cout<<c<<endl;
}
