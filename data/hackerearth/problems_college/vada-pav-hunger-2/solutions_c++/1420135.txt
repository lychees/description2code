#include<bits/stdc++.h>
using namespace std;
int dist[100000];
int vis[100000];
vector<int> gra[1001];
int sum1=0;

int dfs(int s) {
	vis[s] = 1;
	int sum = 0;
		for(int j = 0 ; j < gra[s].size(); j++) {
			if(!vis[gra[s][j]]) {
				sum += 1 + dfs(gra[s][j]);
			}
	   }
	   return sum;
}

int dfs2(int s) {
	vis[s] = 1;
	int sum = 0;
		for(int j = 0 ; j < gra[s].size(); j++) {
			if(!vis[gra[s][j]]) {
				sum = max(sum,1 + dfs2(gra[s][j]));
			}
	   }
	   return sum;
}

int main() {
	int n;
	cin>>n;
	int x,y;
	for(int i = 0 ; i  < n-1 ; i++) {
		cin>>x>>y;
		gra[x].push_back(y);
		gra[y].push_back(x);
	}
	int sum = dfs(1);
	memset(vis,0,sizeof(vis));
	int maxu = dfs2(1);
	cout<<2*sum-maxu<<"\n";
	//cout<<2*sum-maxu<<"\n";
	return 0;
}
