#include<iostream>
using namespace std;
int main() {
    int n,m;
    int arr[1000][1000];
    int rSum[1000];
    int cSum[1000];
    cin>>n>>m;
    int t=0;
    for(int i=0; i<n; i++) {
        t=0;
        for(int j=0; j<m; j++) {
            cin>>arr[i][j];
            t += arr[i][j];
        }
        rSum[i] = t;
    }
    for(int j=0;j<m;j++) {
        t=0;
        for(int i=0; i<n; i++) {
            t += arr[i][j];
        }
        cSum[j] = t;       
    }
    int MAX_XOR = 0;
    for(int i=0; i<n; i++) {
        for(int j=0; j<m; j++) {
            int tempXOR = rSum[i]^cSum[j];
            if(tempXOR > MAX_XOR)
                MAX_XOR = tempXOR;
        }
    }
    cout<<MAX_XOR;
    return 0;
}
