#include<iostream>
#include<algorithm>
#include<functional>
#define mod 1000000007
using namespace std;

long long int fun(long long int *a,long long int &n){
	long long int ans=0;
	sort(a,(a+n),greater<int>());
	long long int freq=n-1;
	for(long long int i=0;i<n;i++,freq-=2){
		ans=ans+((freq%mod)*a[i])%mod;
		ans=ans%mod;
	}
	ans=ans%mod;
	return ans;
}

int main(){
	long long int a[200001],b[200001];
	long long int n,ans=0;;
	cin>>n;
	for(long long int i=0;i<n;i++){
		cin>>a[i];
		cin>>b[i];
	}
	ans=fun(a,n);
	ans+=fun(b,n);
	ans=ans%mod;
	cout<<ans;
	return 0;
}
