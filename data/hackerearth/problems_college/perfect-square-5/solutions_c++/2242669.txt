#include<bits/stdc++.h>

using namespace std;

int main(){
	long long t,n;
	cin >> t;
	while(t--){
		scanf("%lld",&n);
		long long a=((long long)sqrt(n)*(long long)sqrt(n));
		long long b=(long long)sqrt(n)+(long long)1;
		long long c=b*b;
		if(n-a==0||c-n==0)
			cout << "YES" << endl;
		else if(min(n-a,c-n)==n-a)
			printf("-%lld\n",n-a);
		else if(min(n-a,c-n)==c-n)
			printf("+%lld\n",c-n);
	}
	return 0;
}
