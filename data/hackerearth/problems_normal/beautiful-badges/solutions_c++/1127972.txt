#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;
int main()
{
    int n,bn=0,yn=0;
    cin>>n;
    int c[n],d[n];
    for(int i=0;i<n;i++){
        cin>>c[i]>>d[i];
        if(c[i]==1)bn++;
        else yn++;
    }
    int b[bn],y[yn],j=0,k=0;
    for(int i=0;i<n;i++)
    {
        if(c[i]==1)
        b[j]=d[i],j++;
        else
        y[k]=d[i],k++;
    }
    sort(b,b+bn);
    sort(y,y+yn);
    int i=bn-1,count=0;j=yn-1;
    while(i>=0 && j>=0)
    {
        if((float)(b[i] * sqrt(2))> y[j] &&(float) (b[i]/sqrt(2))< y[j])
        count++,i--,j--;
        else if(b[i] * sqrt(2) < y[j] )
        j--;
        else i--;
    }
    cout<<count<<endl;
}
