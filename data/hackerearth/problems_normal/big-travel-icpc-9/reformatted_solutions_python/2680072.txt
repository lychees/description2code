N = int(raw_input().strip())
x = []
y = []
sum_x = sum_y = sum_n = 0
for n in range(N):
    [x1, y1] = [int(i) for i in raw_input().strip().split()]
    x.append(x1)
    y.append(y1)
x.sort()
y.sort()
sum_x = sum(x)
sum_y = sum(y)
for i in range(N):
    sum_x -= x[i]
    sum_y -= y[i]
    sum_n += sum_x - (N-i-1)*x[i] + sum_y - (N-i-1)*y[i]
print sum_n % 1000000007
