#include <iostream>
using namespace std;

long long min(long long a, long long b)
{
	if(a<b)
		return a;
	return b;
}

long long GCD(long long a,long long b)
{
	if(b==0)
		return a;
	GCD(b,a%b);
}

int main()
{
    long long a,b,c;
    cin>>a>>b>>c;
    long long l=min(b,c-a);
    if(l<0)
    	l=0;
    long long u=min(b,c);
    long long num=c*(u-l)*2;
    num=num-( (u*u) - (l*l)) + 2*l*a ;
    long long den=2*a*b;
    long long gcd=GCD(num,den);
    num=num/gcd;
    den=den/gcd;
    cout<<num<<"/"<<den<<endl;
    return 0;
}
