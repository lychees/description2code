#include <bits/stdc++.h>
using namespace std;

int main()
{
    int tt; cin >> tt;
    while(tt--) {
    	int n; cin >> n;
    	map<int, int> mp;
    	for(int i = 0;i < n;i++) {
    		int x; cin >> x;
    		mp[x]++; 
    	}
    	long long ans = n;
    	for(map<int, int>::iterator it = mp.begin();it != mp.end();++it) {
    		long long x = (*it).second;
    		ans += (x * (x - 1) / 2);
    	}
    	cout << ans << endl;
    }
    return 0;
}
