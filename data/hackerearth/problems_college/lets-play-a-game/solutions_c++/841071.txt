inline void sc(int &a)
{
    register int c;
    a = 0;
	do c = getchar_unlocked(); while(c < '0');
	do{
		a = (a << 1) + (a << 3) + c - '0';
		c = getchar_unlocked();
	}while(c >= '0');
}
inline void printint(int a)
 {
	char s[11];
	int t = -1;
 
	do{
		s[++t] = a % 10 + '0';
		a /= 10;
	}while(a > 0);
 
	while(t >= 0)putchar_unlocked(s[t--]);
	putchar_unlocked('\n');
 }

int random_long(int digit=8)
{
   int len=digit;
   int x=0;
   for ( int i = 0; i < len; ++i )
   {
    int dig = rand() % 10;
    while ( x == 0 && dig == 0 )
      dig = rand() % 10;
    x = x * 10 + dig;
  }
   return x;
}

int random(int l, int r)
{
    int diff = r - l + 1;

    return l+random_long()%diff;

}
#define MAX 1000001
int B[MAX],A[MAX];
void ETF(){
	int i;
	rep(i,1,MAX)
		B[i]=i;

	rep(i,2,MAX){
		if(B[i]==i){
			int j=i;
			while(j<MAX){
				B[j]-=(B[j]/i);
				A[j]++;
				j+=i;
			}
		}
	}	
}

int F[MAX];
int main(){

	int t,i;
	ETF();
	cin >> t;
	rep(i,11,MAX){
		if(((i-A[i])>=10 && F[i-A[i]]==0)||(i-B[i]>=10&&F[i-B[i]]==0))
			F[i]=1;
	}
	while(t--){
		int n;
		cin >> n;
		F[n] ? cout << "Alice\n" : cout << "Bob\n";
	}
	return 0;
}
