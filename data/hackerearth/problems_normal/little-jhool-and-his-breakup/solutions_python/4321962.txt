# https://www.hackerearth.com/problem/algorithm/little-jhool-and-his-breakup/


def check_string(s):
    expected_string = 'love'
    letter_to_expect = 0
    for ch in s:
        if ch == expected_string[letter_to_expect]:
            letter_to_expect += 1
            if letter_to_expect == len(expected_string):
                return True
    return False

s = raw_input()
if check_string(s):
    print "I love you, too!"
else:
    print "Let us breakup!"
