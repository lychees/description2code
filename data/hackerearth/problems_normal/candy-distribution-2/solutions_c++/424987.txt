#include <iostream>
using namespace std;

#define gc getchar_unlocked

void scanint(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}

void scanlong(long long &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}
int main() {
	int t;
	scanint(t);
	while(t--)
	{
		int n;
		long long k,sum=0;
		scanint(n);
		for(int i=0;i<n;i++)
		{
			scanlong(k);
			sum = sum + k;
		}
		sum = sum - n + 1;
		cout<<sum<<endl;
	}
	return 0;
}
