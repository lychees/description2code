#include <iostream>
using namespace std;

int main()
{
    static long long int i,x,r,g,b,BK,R,G,Y,BE,M,C,W,l[1000000];
    cin >> x >> r >> g >> b;
    while(i<x)
    {
    	if(i/r%2==1)
    	 l[i]=l[i]+1;
    	if(i/g%2==1)
    	 l[i]=l[i]+10;
    	if(i/b%2==1)
    	 l[i]=l[i]+100;
    	i++; 
    }
    i=0;
    while(i<x)
    {
    	switch(l[i])
    	{
    		case 0: BK++; break;
    		case 1: R++; break;
    		case 10: G++; break;
    		case 11: Y++; break;
    		case 100: BE++; break;
    		case 101: M++; break;
    		case 110: C++; break;
    		case 111: W++; break;
    	}
    	i++;
    }
    cout << R << " " << G << " " << BE << " " << Y << " " << C << " " << M << " " << W << " " <<BK;
    return 0;
}
