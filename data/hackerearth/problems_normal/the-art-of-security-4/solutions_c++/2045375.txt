#include<bits/stdc++.h>

using namespace::std;

#define ll long long int


int main()
{
	ll i,j,k,n;

	cin>>n;
	ll a[1000];
	ll sum=0;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
		sum+=a[i];
	}
	sort(a,a+n);

	ll ways=0;
	ll limit = (1<<n) -1 ;
	//cout<<limit<<"\n";
	do
	{
		for(i=0;i<=limit;i++)
		{
			ll val = i;
			ll one = 0;
			ll two = 0;
			ll flag=0;
			for(j=0;j<n;j++)
			{

				if((val>>j)&1)
				one+=a[j];
				else
				two+=a[j];

				if(one<two)
				{
					flag=1;
					break;
				}	
			}
			if(flag==0)
			ways++;	
		}
	}
	while(next_permutation(a,a+n));

	cout<<ways<<" "<<sum<<endl;
	if(sum>=ways)
	cout<<"Got no way out!\n";
	else
	cout<<"We will win!\n";
}
