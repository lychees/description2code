#include<iostream>
#include<string>
using namespace std;
int main()
{
int t;
cin>>t;
if(t<=10&&t>=1)
{
while(t--)
{
string str;
cin>>str;
int lower=0,upper=0;
int n=str.size();
if(n>=1&&n<=100)
{
for(int i=0;i<n;i++)
{
if('A'<=str[i]&&str[i]<='Z')
upper++;
else if('a'<=str[i]&&str[i]<='z')
lower++;
}
}
if(lower==0&&upper==0)
{
cout<<"Invalid Input"<<endl;
}
else
{
if(lower<upper)
cout<<lower<<endl;
else 
cout<<upper<<endl;
}
}
}
else
cout<<"Invalid Test"<<endl;
return 0;
}
