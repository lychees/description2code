#include<bits/stdc++.h>
using namespace std;
vector<int> gph[100009];
 int color[100009];
 int n;
bool bi(int st)
{
    int i,vt;  
    color[st]=1;
    queue<int> q;
    q.push(st);
    while(!q.empty())
    {
        int u=q.front();
        q.pop();
        for(int i=0;i<gph[u].size();i++)
        {
            vt=gph[u][i];
            if(color[vt]==-1)
            {
                color[vt]=1-color[u];
                q.push(vt);
            }
            else if(color[vt]==color[u])
                return false;
        }
    }
    return true;
}
int main()
{
    int t,m,a,b,i;
    cin>>t;
    while(t--)
    {
        cin>>n>>m;
        for(i=0;i<n;i++)
        	gph[i].clear();
        while(m--)
        {
            cin>>a>>b;
            a--;
            b--;
            gph[a].push_back(b);
            gph[b].push_back(a);
        }
        bool ans=true;
        for(i=0;i<n;i++)
        color[i]=-1;
        for(i=0;i<n;i++)
        {
        if(color[i]!=-1)
        continue;
        bool v=bi(i);
        ans=(ans && v);
        }
        if(ans)
        	printf("YES\n");
        else
            printf("NO\n");
    }
    return 0;
}
