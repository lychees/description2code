#include <iostream>
using namespace std;

int main()
{
    long int e,n,q,m,a,b,c,d,i;
    cin>>n>>q;
    long int z[n];
    for(i=0;i<n;i++)
    z[i]=0;
    cin>>c>>a>>b;
    for(i=a;i<=b;i++)
    z[i]=1;
    q=q-1;
    while(q--)
    {
    	e=0;
    	cin>>d>>m>>n;
    	for(i=m;i<=n;i++)
    	{
    		if(z[i]==1)
    		e++;
    	}
    	cout<<e<<"\n";
    }
    return 0;
}
