#include<bits/stdc++.h>
using namespace std;
#define maxn 26
struct pt{
	int x,y;
};
int mat[maxn][maxn];
struct pt *find(int n,int m){
	struct pt *p=(struct pt *)malloc(sizeof(struct pt));
	for(int i=n-1;i>=0;i--){
		for(int j=m-1;j>=0;j--){
			if(mat[i][j]==0){
				p->x=i;
				p->y=j;
				return p;
			}
		}
	}
	p->x=-1;
	p->y=-1;
	return p;
}
void update(int x,int y){
	for(int i=0;i<=x;i++){
		for(int j=0;j<=y;j++){
			if(mat[i][j])mat[i][j]=0;
			else mat[i][j]=1;
		}
	}
}
int main(){
	int t;
	cin>>t;
	while(t--){
		int n,m;
		cin>>n>>m;
		//int mat[n][m];
		memset(mat,0,sizeof(mat));
		string s;
		
		for(int i=0;i<n;i++){
			cin>>s;
			for(int j=0;j<m;j++){
				mat[i][j]=s.at(j)-'0';
			}	
		}
	/*	for(int i=0;i<n;i++){
			for(int j=0;j<m;j++){
				cout<<mat[i][j]<<" ";
			}
			cout<<endl;
		}*/
		int count=0;
		int flag=1;
		while(flag){
			struct pt* p=find(n,m);
			if(p->x==-1 && p->y==-1)break;
			update(p->x,p->y);
			count++;
		//	cout<<p->x<<" "<<p->y<<" "<<count<<endl;
		//	char x;
		//	cin>>x;
		}
		cout<<count<<endl;
	}
	return 0;
}
