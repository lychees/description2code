#include<bits/stdc++.h>
using namespace std;
#define gc getchar
inline long long int read()
{
	register char c = gc();
	while(c<'0' || c>'9') c = gc();
	long long int ret = 0;
	while(c>='0' && c<='9')
	{
		ret = 10 * ret + c - 48;
		c = gc();
	}
	return ret;
}
int prime[201];
int sieve()
{

	{
		prime[i]=0;
	}
    

		for(i=2;i*i<=n;i++)
		{
			if(!prime[i])
			{
				for(j=i*i;j<=n;j=j+i)
				{
					prime[j]=1;
				}
			}
		}
	return nu;
}
