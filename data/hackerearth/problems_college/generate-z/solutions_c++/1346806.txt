#include <iostream>
#include <cstdio>
#include <algorithm>

using namespace std;

int main()
{
	//code goes here...
	int n;
	cin>>n;
	for(int i=0; i<n; i++)
		printf("*");
	printf("\n");
	for(int i=0;i<n-2;i++){
		for(int j=i;j<n-2;++j)
			printf(" ");
		printf("*\n");
	}
    for(int i=0; i<n; i++)
		printf("*");
	return 0;
}
