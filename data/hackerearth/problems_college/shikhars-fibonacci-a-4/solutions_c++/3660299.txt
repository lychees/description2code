#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
#define mod 1000000007
int main(){
	ll a,b,m;
	cin>>a>>b>>m;
	ll fib[m+1];
	memset(fib,0,sizeof(fib));
	fib[1]=a;
	fib[2]=b;
	for(ll i=3;i<=m;i++){
		fib[i]=(fib[i-1]+fib[i-2])%mod;
	}
	cout<<fib[m]<<endl;
	return 0;
}
