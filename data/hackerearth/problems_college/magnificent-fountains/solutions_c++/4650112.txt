#include <bits/stdc++.h>
#define lli long long int
using namespace std;

lli prr[10005];
vector <lli> vect;
void primes()
{
	
	for(int i=2;i<=10000;i++)
	{
		int p=2;
		if(prr[i]==0)
		{
		while(i*p<=10000)
		{
			prr[i*p]=1;
			p++;
		}
		vect.push_back(i);
		}
	}
}

lli pow(lli a,lli p)
{
	lli ret=1;
	for(lli i=1;i<=p;i++)
	{
	ret=(ret*a)%1000000007;	
	}
	return ret;
}
int main()
{
    int t;
    cin>>t;
    memset(prr,0,sizeof prr);
    primes();
    /*
    cout<<"hi "<<vect.size()<<endl;
    for(int i=0;i<vect.size();i++)
    cout<<vect[i]<<" ";
    */
    
    while(t-->0)
    {
    	int n;
    	cin>>n;
    	lli arr[n+5];
    	lli brr[10005];
    	memset(brr,0,sizeof brr);
    	
		for(int i=1;i<=n;i++)
		{
			cin>>arr[i];
			for(int j=0;j<vect.size();j++)
			{
				lli cnt=0;
				while(arr[i]%vect[j]==0)
				{
					arr[i]/=vect[j];
					cnt++;
				}
				brr[ (int)vect[j] ]=max(brr[(int)vect[j]],cnt);
			}
		}
		lli ans=1;
		for(int i=1;i<=10000;i++)
		{
			if(brr[i]!=0)
			{
				ans=(ans*pow((lli)i,brr[i]))%1000000007;
			}
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
