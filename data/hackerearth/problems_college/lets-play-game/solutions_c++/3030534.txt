#include <iostream>

using namespace std;

int main()
{
    int t,*a,*b;
    int **x,**y;
    cin>>t;
    x = new int*[t];
    y = new int*[t];
    a = new int[t];
    b = new int[t];
    for(int i=0; i<t; i++){
        cin>>a[i]>>b[i];
        x[i] = new int[a[i]];
        y[i] = new int[a[i]];
        for(int j=0; j<a[i]; j++){
            cin>>x[i][j];
        }
        for(int j=0; j<a[i]; j++){
            cin>>y[i][j];
        }
    }

    for(int i=0; i<t; i++){
        int max=0;
        for(int j=0; j<a[i]; j++){
                    if(max<y[i][j]*(b[i]/x[i][j]))
                        max = y[i][j]*(b[i]/x[i][j]);
        }
        cout<<max<<endl;
    }


    return 0;
}
