const int MAXN = 100100;
int dig[MAXN];
int prec[110];
int myabs(int x) {
    if (x < 0) {
        x = -x;
    }
    return x;
}
void solve() {
    const int p10[] = {1, 10, 100, 1000, 10000};
    for (int i = 0; i < 110; i++) {
        prec[i] = i % 10;
    }
    int a, b, c, n;
    cin>>a>>b>>c>>n;
    dig[0] = a; dig[1] = b;
    for (int i = 2; i < n; i++) {
        dig[i] = prec[ dig[i - 1] *c + dig[i - 2] ];
    }
    if (n == 1) {
        if (a == 0) {
            cout<<"YES"<<endl;
        }
        else {
            cout<<"NO"<<endl;
        }
        return ;
    }
    long long p = 0;
    for (int i = 0; i < 5; i++) {
        for (int j = n - 1 - i; j >= 0; j -= 5) {
            p += dig[j] * p10[i];
        }
    }
    if (p % 41 == 0) {
        cout<<"YES"<<endl;
    }
    else {
        cout<<"NO"<<endl;
    }
}
int main() {
    int test;
    cin>>test;
    while (test--) {
        solve();
    }
    return 0;
}
