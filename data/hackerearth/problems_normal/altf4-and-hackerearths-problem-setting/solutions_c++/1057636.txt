#include <bits/stdc++.h>

using namespace std;

const int MAXN = 2000 + 10;
const long long INF = (long long)(1e15);

struct ps {
    long long price, task;
};

ps a[MAXN];
long long f[MAXN];
int n, k;

bool cmp(ps a, ps b) {
    return a.price < b.price;
}

int main()
{
    int test;
    cin >> test;
    while (test --) {
        cin >> n >> k;
        for(int i = 1; i <= n; i++) cin >> a[i].price;
        for(int i = 1; i <= n; i++) cin >> a[i].task;
        sort(a + 1, a + 1 + n, cmp);
        f[0] = 0;
        for(int i = 1; i <= n; i++) {
            int s = a[i].task;
            f[i] = INF;
            for(int j = i - 1; j >= 0; j--) {
                f[i] = min(f[i], f[j] + a[i].price * (s + k));
                s += a[j].task;
            }
            //cout << i << " " << f[i] << endl;
        }
        cout << f[n] << endl;
    }
}
