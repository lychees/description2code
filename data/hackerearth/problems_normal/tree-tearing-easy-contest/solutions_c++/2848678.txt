#include <cstdio>
#include <vector>

using namespace std;

const int N = 100010;
int n, k, z;
vector<int> v[N];

int dfs(int c)
{
	int s = 0;
	for (auto i : v[c]) s += dfs(i);	
	if (s + 1 >= k) 
	{
		z++;
		return 0;
	}
	return s + 1;
}

int main()
{
    scanf("%d %d", &k, &n);
    for (int i = 2; i <= n; i++)
    {
    	int x;
    	scanf("%d", &x);
    	v[x].push_back(i);
    }
    dfs(1);
    printf("%d", z);
    return 0;
}
