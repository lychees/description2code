s = raw_input()
n = len(s)
assert(1 <= n and n <= 10**5)
q = int(raw_input())
assert(1 <= q and q <= 10**5)
for i in xrange(q):
    l1,r1,l2,r2=map(int, raw_input().split())
    assert(1 <= l1 and l1 <= 10**5)
    assert(1 <= r1 and r1 <= 10**5)
    assert(1 <= l2 and l2 <= 10**5)
    assert(1 <= r2 and r2 <= 10**5)
    x = s[l1-1:r1]
    y = s[l2-1:r2]
    if x == y:
    	
    	
        print "Yes"
    else:
        print "No"
