#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

unsigned long long int gcd(unsigned long long int a,unsigned long long int b)
{
	if(b==0)
	 return a;
	else
	 return gcd(b,a%b);
}

int main()
{
   	unsigned long long int a,s,q,print;
   	vector<unsigned long long int> ans;
    cin>>s>>q;
	while(q--)
	{
		cin>>a;
		print=gcd(s,a);
		if(find(ans.begin(),ans.end(),print)==ans.end())
		{
			cout<<print<<"\n";
			ans.push_back(print);
		}
		else
		cout<<"-1\n";
	}
    return 0;
}
