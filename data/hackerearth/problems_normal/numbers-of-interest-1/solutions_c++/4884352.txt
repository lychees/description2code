#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   long int t;
 	cin>>t;
   for(;t--;)
   {
   	long long int a,b,c,i,j;
   	cin>>a>>b>>c;
   	long long int ar[c];
   	if(c<=a)
   	cout<<b<<endl;
   	else
   	{
   	for(i=0;i<a;i++)
   	ar[i]=b;
   	for(i=a;i<c;i++)
   	{
   		ar[i]=0;
   		for(j=1;j<=a;j++)
   		{
   			ar[i]=ar[i]+ar[i-j];
   		}
   	}
   	cout<<ar[c-1]<<endl;
   }
   }
    return 0;
}
