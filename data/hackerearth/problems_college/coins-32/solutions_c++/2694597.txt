#include<iostream>
int coins[]={1,2,5,10};
int counts[5];
long long co(int i,int get){
	if(get==0)
		return 1;
	if(i==2 && get%5!=0)
		return 0;
	if(i==3 && get%10!=0)
		return 0;	
	if(i==4)
		return 0;
	int v=coins[i];
	int j=counts[i];
	if(get-v<0)
		return 0;
	long long count=0;
	int x=0;
	while(x<=j){
		if(get-x*v<0)
		{
			break;
		}
		else{
			count+=co(i+1,get-x*v);
		}
		x++;
	}
	return count;

}
int main(){
	int T,get,i,j,k;
	std::cin>>T;
	while(T--){
		std::cin>>get;
		for(i=0;i<4;i++){
			std::cin>>counts[i];
		}
		long long ans;
		ans=co(0,get);
		std::cout<<ans<<std::endl;
	}
}
