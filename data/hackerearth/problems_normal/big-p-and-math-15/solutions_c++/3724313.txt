#include<iostream>
using namespace std;
long long int rmv(long long int n,long long int rep,long long int to)
{
	long long int i,num,rem;
	num=n;
	i=1;
	while(num)
	{
		rem=num%10;
		num=num/10;
		if(rem==rep)
		n=n+(to-rep)*i;
		i=i*10;
	}
	return n;
}
int main()
{
	long long  int a,b,sum1,sum2;
	cin>>a>>b;
	sum2=rmv(a,5,6)+rmv(b,5,6);
	sum1=rmv(a,6,5)+rmv(b,6,5);
	cout<<sum1<<" "<<sum2;
	return 0;
}
