#include <iostream>
using namespace std;
#define ll long long
ll sum(ll n) {
	ll ans = 0;
	while(n > 0) {
		ll k = (n - 1 + (n%2));
		ans += ((k + 1)*(1 + k)/4);
		n >>= 1;
	}
	return ans;
}
int main()
{
    int t;
    ll x;
    cin>>t;
    while(t--) {
    	cin>>x;
    	cout<<sum(x)<<endl;
    }
    return 0;
}
