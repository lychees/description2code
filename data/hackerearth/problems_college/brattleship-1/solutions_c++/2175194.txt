#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t,r,c,w,k=0,sum=0;
    cin>>t;
    while(t>0)
    {
    	sum=0;
    	cin>>r>>c>>w;
    	if(c!=w)
    	{
    		sum=c-w+1;
    	}
    	else sum=w;
    	sum*=r;
    	cout<<"Case #"<<k+1<<": "<<sum<<endl;
    	k++;
    	t--;
    }
    return 0;
}
