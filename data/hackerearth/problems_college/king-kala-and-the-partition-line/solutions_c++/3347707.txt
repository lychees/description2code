#include<bits/stdc++.h>
using namespace std;

int main(){

int t;
cin >> t;
long long a,b,c,n;
long long temp=0;
long long x,y;
long long s1,s2;
while(t--){
    cin >> n;
    cin >> a >> b >> c;
    s1=0;
    s2=0;
    while(n--){
            cin >> x >> y;
        temp = (x*a)+(b*y)+c;
        if(temp<0)
            s1++;
        else if(temp>0)
            s2++;
    }
    if(s1<s2)
        cout << s2 << "\n";
    else
        cout << s1 << "\n";
}

}
