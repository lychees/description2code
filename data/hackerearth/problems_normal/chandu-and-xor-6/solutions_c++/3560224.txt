#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,m;
    scanf("%d %d",&n,&m);
    int A[n][m];
    for(int i=0;i<n;i++)
    	for(int j=0;j<m;j++)
    		scanf("%d",&A[i][j]);
    
    int rowSum[n];
    for(int i=0;i<n;i++){
    	rowSum[i]=0;
    	for(int j=0;j<m;j++)
    		rowSum[i]+=A[i][j];
    }
    
    int columnSum[m];
    for(int i=0;i<m;i++){
    	columnSum[i]=0;
    	for(int j=0;j<n;j++)
    		columnSum[i]+=A[j][i];
    }
    
    int ans=0;
    for(int i=0;i<n;i++)
    	for(int j=0;j<m;j++)
    		ans=max(ans,rowSum[i]^columnSum[j]);
    		
    cout<<ans;
    
}
