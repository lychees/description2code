const int MAX = 1e5 + 10;
using namespace std;
int ti = 0;
vector<int>v[MAX];
int lev[MAX], in[MAX], ot[MAX], vis[MAX];
void dfs(int x)
{
    vis[x] = 1;
    in[x] = ++ti;
    for (int i = 0; i < v[x].size(); i++)
    {
        int cur = v[x][i];
        if (!vis[cur])
        {
            lev[cur] = lev[x] + 1;
            dfs(cur);
        }
    }
    ot[x] = ++ti;
}
int main()
{

    ios_base::sync_with_stdio(0);
    cin.tie(0);
#ifndef ONLINE_JUDGE

#endif
    int t, q, x, n, a, b, root;
    cin >> t;
    while (t--)
    {
        cin >> n;
        MS0(lev);
        MS0(in);
        MS0(ot);
        MS0(vis);
        ti = 0;
        for (int i = 0; i < MAX; i++)
            v[i].clear();
        for (int i = 1; i <= n; i++)
        {
            cin >> x;
            if (x == 0)
            {
                root = i;
                continue;
            }
            v[i].pb(x);
            v[x].pb(i);
        }
        //cout << root << endl;
        dfs(root);
        cin >> q;
        while (q--)
        {
            cin >> a >> b;
            if (in[b] < in[a] && ot[a] < ot[b])
            {
                cout << (lev[a] - lev[b] - 1) << endl;
            }
            else
                cout << "-1" << endl;
        }
    }
    return 0;
}
