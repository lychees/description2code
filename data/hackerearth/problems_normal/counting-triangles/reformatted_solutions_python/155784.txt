from collections import defaultdict

if __name__ == '__main__':
    temp_dict = defaultdict(int)
    cases = int(raw_input())
    results = 0
    temp = []
    for c in xrange(cases):
        inp = sorted(map(int, raw_input().split()))
        temp = ''.join(map(str, inp))
        temp_dict[temp] = temp_dict[temp] + 1
    for t in temp_dict:
        if temp_dict.get(t) == 1:
            results = results + 1

    print results
