#include <stdio.h>
#include<iostream>
using namespace std;

int main()
{
    char str[19];
    int n,c;
    cin>>n;
    while(n--)
    {
    	c=0;
    	cin>>str;
    	for(int i=0;str[i]!='\0';i++)
    		if(str[i]=='4' || str[i]=='7')
    			c++;
    	if(c==4 || c==7)
    		cout<<"YES"<<'\n';
    	else
    		cout<<"NO"<<'\n';
    }
    return 0;
}
