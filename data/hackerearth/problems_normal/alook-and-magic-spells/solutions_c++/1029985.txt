#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using namespace std;

typedef long long ll;

const ll mod = ll(1e9) + 7;

ll fpow(ll a, ll b){
    ll t = 1LL;
    for(;b; b >>= 1){
        if (b & 1) t = (t * a) % mod;
        a = (a * a) % mod;
    }
    return t;
}

struct mat{
    ll a[3][3];
    mat(){memset(a, 0, sizeof a);}
    mat operator * (const mat &b)const{
        mat ans;
        memset(ans.a, 0, sizeof ans.a);
        for(int i = 0; i < 3; ++i)
            for(int j = 0; j < 3; ++j)
                for(int k = 0; k < 3; ++k)
                    ans.a[i][j] = (ans.a[i][j] + a[i][k] * b.a[k][j]) % mod;
        return ans;
    }
    mat operator ^(ll k){
        mat ans;
        for(int i = 0; i < 3; ++i) ans.a[i][i] = 1LL;
        mat cur = *this;
        for(; k ; k >>= 1){
            if (k & 1) ans = ans * cur;
            cur = cur * cur;
        }
        return ans;
    }
    void in(){
        for(int i = 0; i < 3; ++i){
            for(int j = 0; j < 3; ++j) cout << a[i][j] << ' ';
            cout << '\n';
        }
        cout << '\n';
    }
} d1, bas, d2;

int tc;
ll n;

int main(){
    bas.a[1][0] = 1LL;
    bas.a[0][1] = 4LL;
    bas.a[1][1] = 2LL;
    bas.a[0][2] = 4LL;
    bas.a[1][2] = 2LL;
    bas.a[2][2] = 1LL;
    d1.a[0][0] = 2LL;
    d1.a[0][1] = 12LL;
    cin >> tc;
    while (tc--){
        cin >> n;
        if (n == 1){
            cout << "2\n";
            continue;
        }
        if (n == 2){
            cout << "14\n";
            continue;
        }
        d2 = d1 * (bas ^ (n - 2));
        cout << (d2.a[0][2] + 14LL) % mod << '\n';
    }
    return 0;
}
