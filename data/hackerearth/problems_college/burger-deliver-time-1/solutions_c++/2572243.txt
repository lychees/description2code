#include<bits/stdc++.h>
using namespace std;

int main(){
    int n;
    scanf("%d", &n);
    vector<pair<int, int> > vi;
    for(int i=1;i<=n;++i){
        int a, b;
        scanf("%d %d", &a, &b);
        vi.push_back(make_pair(a+b, i));
    }
    sort(vi.begin(), vi.end());
    for(int i=0; i<n; ++i)
        printf("%d ", vi[i].second);
}
