#include <iostream>
using namespace std;

int main()
{
    long long int tc,n,p;
    cin>>tc;
    while(tc--)
    {
    	int ans=0;
    	cin>>n>>p;
    	while(n%p==0)
    	{
    		ans++;
    		n/=p;
    		if(n==1)
    		{
    			n=0;
    			break;
    		}
    	}
    	cout<<ans<<" "<<n<<endl;
    }
    return 0;
}
