#include <iostream>
using namespace std;

int main()
{
    int t,i,j,count;
    string s;
    
    cin>>t;
    
    for(i=0;i<t;i++){
    	cin>>s;
    	count=0;
    	for(j=4;j<=s.length()-4;j++){
    		if(s[j]=='a'||s[j]=='e'||s[j]=='i'||s[j]=='o'||s[j]=='u'){
    			continue;
    		}
    		else{
    			count++;
    		}
    	}
    	cout<<count+3<<"/"<<s.length()<<endl;
    }
    return 0;
}
