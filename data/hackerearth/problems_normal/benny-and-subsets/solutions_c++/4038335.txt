#include<bits/stdc++.h>
#define ll long long
#define mod 10000007

using namespace std;

ll dp[2050][1005];
ll xr[2097156];

int main()
{
	int n, i, x, a[1005], c, l = 1;
	vector<int> b;
	scanf("%d%d", &n, &x);

	for( i = 1 ; i <= n ; i++ )
	{
		scanf("%d", &c);
		if( c > 2048 )
			b.push_back(c);
		else
			a[l++] = c;
	}
	
	dp[0][0] = 1;
	for( i = 1 ; i < 1026 ; i++ )
	{
		dp[i][0] = 0;
	}
	
	int j;
	for( j = 1 ; j < l ; j++ )
	{
		for( i = 0 ; i < 2048 ; i++ )
		{
			dp[i][j] = dp[i][j - 1] + dp[i ^ a[j]][j - 1];
			if(dp[i][j] >= mod)
				dp[i][j] %= mod;
		}
	}

	memset(xr, 0, sizeof xr);
	int k, ex, tr, flag, m = b.size();
	flag = 1 << (int)m;
	for( j = 0 ; j < flag ; j++ )
	{
		tr = 1, ex = 0, k = m - 1;
		while( tr < flag )
		{
			if( tr & j )
			{
				ex ^= b[k];
			}
			tr = tr << 1;
			k--;
		}
		xr[ex]++;
	}

	ll ans = 0;
	for( i = 0 ; i < 2097153 ; i++ )
	{
		if( xr[i] > 0 && (i ^ x) < 2048 )
		{
			ans += xr[i] * dp[i ^ x][l - 1];
			if( ans >= mod )
			{
				ans %= mod;
			}
		}
	}
	printf("%lld\n", ans);
	return 0;
}
