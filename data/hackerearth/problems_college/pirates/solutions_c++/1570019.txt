#include <cstdio>

int A[100100];

int main()
{
    int N,c=1,max=1;
    scanf("%d",&N);
    for(int i = 0; i < N; ++i)
    {
        scanf("%d",&A[i]);
        if( i > 0 )
        {
            if( (A[i] == A[i-1]+1 )|| (A[i] + 1 == A[i-1]) )
            {
                ++c;
                if(max < c)
                    max = c;
            }
            else
             c = 1;
        }
    }
    printf("%d\n",max);
    return 0;
}
