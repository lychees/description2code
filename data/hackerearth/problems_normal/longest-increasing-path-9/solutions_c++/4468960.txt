#include <iostream>
using namespace std;

int main()
{
	int tc,r,c,i,j,max = 0,k;
	cin >> tc;
	while(tc--)
	{
		cin >> r >>c;
		long int val[r][c];
		long int len[r][c];
		max = 1;
		for(i = 0;i<r;i++)
		{
			for(j = 0;j<c;j++)
				cin>>val[i][j];
		}
		len[0][0] = 1;
		for(i = 0;i<r;i++)
		{
			for(j=0;j<c;j++)
			{
				if(i-1>=0 && j-1>=0)
				{
					if(len[i-1][j]!=0 && len[i][j-1]!=0)
					{
						if(val[i][j]>val[i-1][j])
						{
							len[i][j] = len[i-1][j]+1;
						}
						else
						{
							len[i][j] = 0;
						}
						if(val[i][j]>val[i][j-1])
						{
							if(len[i][j-1]+1 > len[i][j])
								len[i][j] = len[i][j-1]+1;
						}
						if(len[i][j]>max)
							max = len[i][j];
					}
					else if(len[i-1][j]!=0)
					{
						if(val[i][j]>val[i-1][j])
						{
							len[i][j] = len[i-1][j]+1;
							if(len[i][j]>max)
								max = len[i][j];
						}
						else
							len[i][j] = 0;
					}
					else if(len[i][j-1]!=0)
					{
						if(val[i][j]>val[i][j-1])
						{
							len[i][j] = len[i][j-1]+1;
							if(len[i][j]>max)
								max = len[i][j];
						}
						else
							len[i][j] = 0;
					}
					else
					{
						len[i][j] = 0;
					}
				}
				else if(i-1>=0)
				{
					if(val[i][j]>val[i-1][j] && len[i-1][j]!=0)
					{
						len[i][j] = len[i-1][j]+1;
						if(len[i][j]>max)
							max = len[i][j];
					}
					else
						len[i][j] = 0;
				}
				else if(j-1>=0)
				{
					if(val[i][j]>val[i][j-1] && len[i][j-1]!=0)
					{
						len[i][j] = len[i][j-1]+1;
						if(len[i][j]>max)
							max = len[i][j];
					}
					else
						len[i][j] = 0;
				}
			}
		}
		cout<<max<<endl;
		
	}
    return 0;
}
