#include<iostream>
#include<stdio.h>
using namespace std;
#define mod 1000000007
#define ll long long
ll power(ll a,ll b)
{
	ll mul=1;
	while(b)
	{
		if(b&1)
			mul=(mul*a)%mod;
		a=(a*a)%mod;
		b>>=1;
	}
	return mul;
}
int main()
{
	ll t,n,ans;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
		if(n<3)
		{
			cout<<"0\n";
			continue;
		}
		ans=power(3,n)-power(2,n)*3%mod+3;
		ans+=mod;
		printf("%lld\n",ans%mod);
	}
}
