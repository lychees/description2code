#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,i;
    cin>>t;
    while(t--)
    {
    	string str;
    	cin>>str;
    	int l=str.length();
    	int a[26];
    	memset(a,0,sizeof(a));
    	a['a'-'a']=1;
    	a['e'-'a']=1;
    	a['i'-'a']=1;
    	a['o'-'a']=1;
    	a['u'-'a']=1;
    	a['y'-'a']=1;
    	int count=0;
    	int flag=0;
    	for(i=0;i<=l-3;i++)
    	{
    		if(a[str[i]-'a']==0 && a[str[i+1]-'a']==0 && a[str[i+2]-'a']==0 )
    		{
    			flag=1;
    			cout<<"hard"<<endl;
    			break;
    		}
    		else if(a[str[i]-'a'] ==1)
    			count--;
    		else
    			count++;

    	}
    	if( flag==0)
    	{
    		for(;i<l;i++)
    		{
    			if(a[str[i]-'a'] ==1)
    				count--;
    			else
    				count++;
    		}
    		if(count>0)
    			cout<<"hard"<<endl;
    		else
    			cout<<"easy"<<endl;
    	}
    }
    return 0;
}
