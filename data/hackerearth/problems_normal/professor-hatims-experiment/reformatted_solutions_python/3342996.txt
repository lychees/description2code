import os
import sys


def dfs(u, depth, visited, adjlist, d):
    visited[u] = True
    depth[u] = d
    for x in adjlist[u]:

        if visited[x]:
            if (depth[x]-depth[u]) % 2 == 0:
                return True
        else:
            out = dfs(x, depth, visited, adjlist, d+1)
            if out:
                return True

    return False


def find_loop(n, adjlist):
    visited = [False]*n
    depth = [0]*n
    for i in range(n):
        if not visited[i]:
            res = dfs(i, depth, visited, adjlist, 0)
            if res:
                return True

    return False
