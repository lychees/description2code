#include<iostream>
using namespace std;
int main()
 {
	int t;
	long long int i,j,n,m,k,p,q;
	cin>>t;
	for(j=0;j<t;j++)
	{
	  cin>>n;
	  cin>>m;
	  long long int arr[n],a[n];
	  for(i=0;i<n;i++)
	  {
	  arr[i]=i;
	  a[i]=i;
	  }
	   k=0;
	   p=n-1;
	    q=0;
	  for(i=0;i<n;i++)
	  {
	   if(i%2==0)
	   
	   	arr[k++]=a[p--];
	   
	   else
	   arr[k++]=a[q++];
	  }
	  for(i=0;i<n;i++)
	  {
	  	if(m==arr[i])
	  	{
	     cout<<i<<"\n";
	     break;
	  	}
	  }
	  
	  
	}
	return 0;
}
