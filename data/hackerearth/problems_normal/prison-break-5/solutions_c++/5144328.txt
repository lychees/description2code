#include <bits/stdc++.h>
using namespace std;
#define scan(x) scanf("%d",&x)
#define print(x) printf("%d",x)
#define MOD 1000000007
#define scanl(x) scanf("%lld",&x)
#define printl(x) printf("%lld",x)
int countp;
bool arr[20][20];
bool vis[20][20];
int xx[4]={0,0,1,-1};
int yy[4]={1,-1,0,0};

void pathcount(int x,int y,int fx,int fy,int n)
{
	if(x==fx && y==fy) 
	{
		countp++;
		return;
	}
	vis[x][y]=1;
	for(int k=0;k<4;k++)
	{
		int a=x+xx[k];
		int b=y+yy[k];
		if(a<0 || b<0 || b>=n ||a>=n) continue;
		if(vis[a][b]) continue;
		if(!vis[a][b] && arr[a][b]==1)
		{
			vis[a][b]=1;
			continue;
		}
		else
		{
			pathcount(a,b,fx,fy,n);
			vis[a][b]=0;
		}
	}
}


int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		countp=0;
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<n;j++) 
			{
				cin>>arr[i][j];
				vis[i][j]=0;
			}
		}
//		if(arr[0][0]==1 || arr[n-1][n-1]==1) cout<<"0\n";
//		else
		{
			pathcount(0,0,n-1,n-1,n);
			cout<<countp<<"\n";
		}
	}
	
    return 0;
}
