#include <iostream>
#include<stdio.h>
#include<math.h>
using namespace std;

int main()
{
    long long int t,i,j,k,l,n,min,max,cmin,cost;
    scanf("%lld",&t);
    while(t--)
    {
    	scanf("%lld",&n);
    	int a[n],c[n];
    	//min=10000;max=1;
    	for(i=0;i<n;i++)
    	{
    		scanf("%lld",&a[i]);
    		//if(a[i]<=min)	min=a[i];
    		//if(a[i]>=max)	max=a[i];
    	}
    	for(i=0;i<n;i++)
    	{
    		scanf("%lld",&c[i]);
    	}
    	cmin=1000000000000000000;cost=0;k=0;
    	for(i=0;i<n;i++)
    	{
    		cost=0;
    		for(j=0;j<n;j++)
    		{
    			k=abs(a[j]-a[i]);
    			cost=cost+k*c[j];
    			if(cost>=cmin)	break;
    		}
    		if(cost<=cmin)
    			cmin=cost;
    	}
    	printf("%lld\n",cmin);
    }
    return 0;
}
