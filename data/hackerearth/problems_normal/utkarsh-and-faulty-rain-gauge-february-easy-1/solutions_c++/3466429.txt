#include <iostream>
#include <cstdio>

using namespace std;

int main()
{
    int t;
    double a,n,r;
    a=1.0;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%lf %lf",&n,&r);
    	a *= (n*(n+1)/2)*(n*(n+1)/2);
    	a /= r;
    	a /= n*(n+1)*(2*n+1)/6;
    }
    printf("%.4lf",a);
    return 0;
}
