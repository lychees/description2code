int main()
{
	int t;
	si(t);
	while(t--)
	{
		ll n;
		sl(n);
		string s;
		cin>>s;
		if(n==1)
		{
			printf("1\n");
			continue;
		}

		ll answer = 0;
		ll i=1;
		char prev=s[0];
		ll sorted;
		while(i<n)
		{
			sorted=1;
			while(i<n && s[i]>=prev)
			{
				prev=s[i];
				i++;
				sorted++;
			}
			answer+=(sorted*(sorted+1)/2);
			if(i==(n-1))
			{
				answer++;
				break;
			}
			prev=s[i];
			i++;
		}
		printf("%lld\n",answer);
	}
	return 0;
}
