#include <iostream>
using namespace std;

int main()
{
	int t;
	cin >> t;
	
	while (t--)
	{
		string str;
		int n;
		cin >> str >> n;
		
		int len = str.length();
		int sum = 0;
		for (int i = 1; i <= n; i++)
		{
			if (i % 2 != 0)
			sum++;
		}
		bool flag = false;
		if (sum % 2 == 0)
			flag = true;
		
		int count = 0;
		for (int i = 0; i < len; ++i)
		{
			if (str[i] == '1')
				count++;
		}
		
		if (flag)
		{
			cout << count << endl;
		}
		else
		{
			cout << len-count << endl;
		}
	}
}
