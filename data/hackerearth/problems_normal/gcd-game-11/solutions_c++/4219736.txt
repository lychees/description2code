#include <iostream>
#include <stdio.h>
#define MAX 1007
using namespace std;

int GCD[MAX][MAX];
int dp[MAX][MAX][2];
bool visited[MAX][MAX][2];

int gcd(int a, int b)
{
	if(b == 0)
	{
		return a;
	}
	else
	{
		return gcd(b, a % b);
	}
}

void pre()
{
	int i, j;
	for(i=1;i<MAX;++i)
	{
		for(j=1;j<=i;++j)
		{
			GCD[i][j] = GCD[j][i] = gcd(i, j);
		}
	}
}

int process(int a, int b, int turn)
{
	if(turn == 0)
	{
		if(a == 1)
		{
			return 0;
		}
	}
	else
	{
		if(b == 1)
		{
			return 0;
		}
	}
	if(visited[a][b][turn])
	{
		return dp[a][b][turn];
	}
	visited[a][b][turn] = true;
	int ans = 0;
	if(turn == 0)
	{
		ans |= (!process(a, b - 1, turn ^ 1));
		if(GCD[a][b] > 1)
		{
			ans |= (!process(a, b / GCD[a][b], turn ^ 1));
		}
	}
	else
	{
		ans |= (!process(a - 1, b, turn ^ 1));
		if(GCD[a][b] > 1)
		{
			ans |= (!process(a / GCD[a][b], b, turn ^ 1));
		}
	}
	dp[a][b][turn] = ans;
	return ans;
}


int main()
{
	pre();
	int T;
	scanf("%d", &T);
	while(T--)
	{
		int A, B;
		scanf("%d %d", &A, &B);
		if(A == 1 && B == 1)
		{
			printf("Draw\n");
		}
		else if(B == 1)
		{
			printf("Arjit\n");
		}
		else if(A == 1)
		{
			printf("Chandu Don\n");
		}
		else
		{
			if(process(A, B, 0))
			{
				printf("Arjit\n");
			}
			else
			{
				printf("Chandu Don\n");
			}
		}
	}
	return 0;
}
