#include <iostream>
using namespace std;

int main()
{
    int t,n,i,f;
    int *a,*b;
    cin>>t;
    while(t--){
    	cin>>n;
    	a = (int *)malloc(sizeof(int)*(n+1));
    	b = (int *)malloc(sizeof(int)*(n+1));
    	for(i=1;i<=n;i++){
    		cin>>a[i];
    		b[a[i]] = i;
    	}
    	f = 1;
    	for(i=1;i<=n;i++){
    		if(a[i]!=b[i]){
    			f=0;
    			break;
    		}	
    	}
    	if(f){
    		cout<<"inverse"<<endl;
    	}else{
    		cout<<"not inverse"<<endl;
    	}
    }
    
    
    return 0;
}
