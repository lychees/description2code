#include<bits/stdc++.h>
using namespace std ;
#define ll long long

ll count(ll n)
{
    ll c =  0 ;
    while(n!=0)
    {
        n = n&(n-1) ;
        c++ ;
    }
    return c ;
}

int main()
{
    int  t;
    scanf("%d",&t);
    while(t--)
    {
        ll n ;
        scanf("%lld",&n);
        printf("%lld\n",count(n)) ;
    }
}
