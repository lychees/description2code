#include<bits/stdc++.h>
using namespace std;
#define maxn 105
#define F(i,p,n) for(i=p;i<n;i++)
#define I(i,p,q) for(i=p;i>=q;i--)
#define Ss(x) scanf("%s",x)
#define S(x) scanf("%d",&x)
#define P(x) printf("%d\n",x)
#define Ps(x) printf("%d ",x)
#define LL long long int
char s[maxn];
main()
{
    int i,j,m,k,n,len,t;
    S(t);
    while(t--)
    {
        k=0;
        Ss(s);
        len=strlen(s);
        F(i,0,len/2)
        {
            if(s[i]!=s[len-1-i])
            {
                k++;
                break;
            }
        }
        if(k)
            printf("NO\n");
        else
            printf("YES\n");
    }
}
