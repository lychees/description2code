#include<iostream>
#include<string>
#define si(n) scanf("%d",&n)
#define pi(n) printf("%d\n",n)
#define sll(n) scanf("%lld",&n)
#define pll(n) printf("%lld\n",n)
#include<math.h>
#include<algorithm>
#define ll long long int 
using namespace std;
int main()
{
	int t;si(t);
	while(t--)
	{
		ll z,x,y;
		sll(z);sll(x);sll(y);
		ll s = sqrt(z);
		//cout<<s<<endl;
		ll ans = 0;
		for(int i=1;i<=s;i++)
		{
			if(z%i==0)
			{
				ll j = z/i;
				if(i<=x && j<=y) ans++;
				if(j<=x && i<=y) ans++;
			}
		}
		if(s*s==z)
		{
			if(s<=x && s<=y) ans--;
		}
		pll(ans);
	}
}
