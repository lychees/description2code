#include<bits/stdc++.h>
using namespace std;
#define pii pair< int, int >
#define pb push_back
#define ft first
#define sd second
#define ll long long

int main() {
	int t;
	cin>>t;
	while ( t-- ) {
		int n;
		cin>>n;
		vector< vector<int> > gr(n, vector<int> (n,0));
		for ( int i=0;i<n;i++ ) {
			for ( int j=0;j<n;j++ ) {
				cin>>gr[i][j];
			}
		}
		bool flag=0;
        for ( int i=0;i<n;i++ ) {
			for ( int j=0;j<n;j++ ) {
				for ( int k=0;k<n;k++ ) {
                    if ( gr[i][j]==1 && gr[j][k]==1 && gr[k][i]==1 ) {
						flag=1;
                    }
				}
			}
        }
        if ( flag==0 ) {
			cout<<"Nae\n";
        }
        else {
			cout<<"Aye\n";
        }
	}
	return 0;
}
