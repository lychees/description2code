#include<iostream>
#include<vector>
#include<set>
using namespace std;
int main()
{
    int t;
    cin>>t;
    long long int e,x,y;
    set<long long int> myset;
    while(t--)
    {
        cin>>e;
        for(int i=0;i<e;i++)
        {
            cin>>x>>y;
            myset.insert(x);
            myset.insert(y);
        }
        cout<<myset.size()<<"\n";
        myset.clear();
    }
    return 0;
}
