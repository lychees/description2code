#include<bits/stdc++.h>
using namespace std;
long long a[100005];
int n;
short t;
priority_queue<pair<int, long long> > q;
int f(long long x)
{
	int cnt=0;
	while(x!=0)
	{
		if(x%2==1)cnt++;
		x/=2;
	}
	
	return cnt;
}
void solve()
{
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
		q.push(make_pair(-f(a[i]),-i));
	}
	while(q.size()!=1)
	{
		cout<<a[-q.top().second]<<" ";
		q.pop();
	}
	cout<<a[-q.top().second]<<endl;
	q.pop();
}
int main()
{
	ios::sync_with_stdio(false);
	cin.tie(NULL);
	cin>>t;
	for(short i=0;i<t;i++)
		solve();
	return 0;
}
