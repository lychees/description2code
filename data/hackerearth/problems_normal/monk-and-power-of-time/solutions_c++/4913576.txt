#include <iostream>
#include <vector>
#include <utility>
#include <algorithm>
#include<queue>
typedef long long int ll;

using namespace std;
int main()
{
	ll i,a,n;
	queue<ll>qq,q;
	cin>>n;
	for(i=0;i<n;i++)
	{
    	scanf("%lld",&a);
    	q.push(a);
	}
	ll c=0;
	for(i=0;i<n;i++)
	{
    	scanf("%lld",&a);
    	qq.push(a);
	}
	while(!qq.empty())
	{
		ll z=qq.front();
		while(z!=q.front())
		{
			q.push(q.front());
			q.pop();
			c++;
			
		}
		qq.pop();
		q.pop();
		c++;
		
	}
	cout<<c<<endl;
}
