#include <bits/stdc++.h>
using namespace std;

char s[100000];
int n;

int func(char x){
	int c = 0;
	for (int i=0;i<n;i++){
		if (s[i]==x) c++;
		else{
			x = s[i];
		}
	}
	return c;
}

int main(){
	int t;
	scanf("%d",&t);
	while (t--){
		cin >> s;
		n = strlen(s);
		char a = s[0];
		char b = NULL;
		int i =1;
		while (i<n){
			if (s[i]!=a){
				b = s[i];
				break;
			}
			i++;
		}
		if (b==NULL){
			cout << n-1 << endl;
			continue;
		}
		int c1 = func(a);
		int c2 = func(b);
		cout << min(c1,c2) << endl;
	}
	return 0;
}
