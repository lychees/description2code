#include <iostream>
using namespace std;
typedef long long ll;



int sumof(ll n)
{
int sum=0;
while(n)
{
	sum+=n%10;
	n/=10;
}
	return sum;
	
}

int func(ll n,int *p)
{


if(n==1||n==4)
return 1;

if(n==9||n==16)
return 0;

n=sumof(n*n);

if(p[n]==1)
return 0;

p[n]=1;
return func(n,p);	
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	ll n;
    	scanf("%lld",&n);
    	int p[163]={0};
    	int f=func(n,p);
    	if(f==1)
    	printf("YES\n");
    	else
    	printf("NO\n");
    }
    
    return 0;
}
