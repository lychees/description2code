#include <iostream>
#include<stack>
using namespace std;

int main()
{
    stack<int>st;
    int type,cost;
    int querry;
    cin>>querry;
    while(querry--){
    	cin>>type;
    	if(type==1){
            if(st.empty()){
                cout<<"No Food"<<endl;
            }
            else{
                    int k=st.top();
                    cout<<k<<endl;
                    st.pop();
            }
    	}
    	if(type==2){
            cin>>cost;
            st.push(cost);
    	}
    }
    return 0;
}
