#include <iostream>
using namespace std;

int main()
{
   int tc;
   cin>>tc;
   while(tc--)
     {
     	string s;
     	cin>>s;
     	int i=0;
     	int count=0;
     	while(s[i]!='\0')
     	{
     	 if(s[i]==s[i+1])
     	 count++;
     	 i++;
     	}
     	cout<<count<<endl;
     }
    return 0;
}
