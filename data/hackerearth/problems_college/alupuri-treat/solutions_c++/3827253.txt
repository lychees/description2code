#include <iostream>
using namespace std;

int main()
{
    int s=0,s1=0,i,m,j;
    cin>>m;
    int a[m][m];
    for(i=0;i<m;i++)
    {
    	for(j=0;j<m;j++)
    	cin>>a[i][j];
    }
    for(i=0;i<m;i++)
    {
    	for(j=0;j<m;j++){
    	if(i==j) s=s+a[i][j];}
    }
    for(i=0;i<m;i++)
    {
    	for(j=0;j<m;j++)
    	{
    		if(i+j==m-1)
    		s1=s1+a[i][j];
    	}
    }
    cout<<abs(s-s1)<<endl;
}
