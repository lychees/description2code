#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    long long int t;
    cin>>t;
    while(t>0){
        t--;
        int n;
        cin>>n;
        long long int k;
        cin>>k;
        long long int p=1;
        for(int i=0;i<n-1;i++){
            p=p*2;
        }
        
        long long int rem=(long long int)(k/p);
        long long int res=k%p;
        if(rem%2==0)
            cout<<"NO"<<endl;
        else if(res==p-1){
            cout<<"YES"<<endl;
        }
        else {
            cout<<"NO"<<endl;
        }
    }
    return 0;
}
