#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

short t, it;
long n, m, i, a;
vector<long> vec;

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    
    cin>>t;
    for(it=0;it<t;it++){
    	cin>>n>>m;
    	for(i=0;i<n+m;i++){
    		cin>>a;
    		vec.push_back(a);
    	}
    	sort(vec.begin(),vec.end());
    	for(i=n+m-1;i>=0;i--){
    		cout<<vec[i]<<" ";
    	}
    	cout<<"\n";
    	vec.clear();
    }
    return 0;
}
