
#include<bits/stdc++.h>
using namespace std;

typedef long long int ll;
#define MOD 1000000007LL

template <typename T>
T modpow(T base, T exp, T modulus) {
  base %= modulus;
  T result = 1;
  while (exp > 0) {
    if (exp & 1) result = (result * base) % modulus;
    base = (base * base) % modulus;
    exp >>= 1;
  }
  return result;
}
int main()
{
    ll t,n;
    cin>>t;
    while(t--)
    {
        cin>>n;
        if(n<3)
        {
            cout<<"0\n"; //3(3^(N-1) - 2 ^N + 1)
            continue;
        }
        ll x=3;
        ll ans=modpow(x,n-1,MOD);
        x=2;
        ans-=modpow(x,n,MOD);
        ans+=MOD;
        ans%=MOD;
        ans++;
        ans%=MOD;
        ans*=3;
        ans%=MOD;
        cout<<ans<<"\n";
    }
    return 0;
}
