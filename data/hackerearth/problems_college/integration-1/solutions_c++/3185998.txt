int find(int p,int n){
	int k,t;
  k=(u-p)/r;
  if(n==0)
  	return p;
  if(k>0&&k<=n){
  	p=p+k*r;
  	n=n-k;
  	return find(p,n);
}
  else if(k>n){
  	p=p+n*r;
  	return p;
  }
  else if(k==0){
  if(p-s>0)
  return find(p-s,n-1);
  else
  return p;
  }

}
