#include <iostream>
#include <vector>
#include <cmath>
using namespace std;

int n, k, t;
vector<long long int>ac;

int main()
{

	cin >> k >> t;
	ac.resize(k);
	for(int i=0;i<k;i++)	
		cin>>ac[i];
		
	for(int r=0;r<t;r++)
	{
	long long int a,b;
	cin>>a>>b;
	a--; 
	long long int ans = 0;
	for(int i=1;i<(1<<k);i++)
	{
		long long int cnt = 0,tm=1;
		for(int j=0;j<k;j++)	
			if((i>>j) &1){
				cnt++; tm*=ac[j];
				//a/=ac[j];
				//b/=ac[j];
				if(tm>b) break;
			}
		
		if(cnt&1)
			ans+=b/tm-a/tm;
		else
			ans-=b/tm-a/tm;
	//cout<<ans<<endl;	
	}
	cout<<ans<<endl;
	}
	return 0;
}
