typedef long long ll;

typedef pair<int, int> pii;

typedef pair<ll, ll> pll;
typedef vector<int> vi;

const int N=2*1e5+5;
ll n, q;
char a[N][20];
ll BIT[N];
ll l, r;
ll i, sum;

const int INF = 0x3f3f3f3f, MOD = 1e9+7;

bool isround(char *s) {
  if(s[0]=='-') return 0;
  return s[0]==s[strlen(s)-1];
}
void update(int pos, int val) {
  for(;pos<N;pos+=pos&-pos) BIT[pos]+=val;
}
ll query(int pos) {
  sum=0;
  for(;pos>0;pos-=pos&-pos)sum+=BIT[pos];
  return sum;
}

int main() {
  scanf("%lld%lld", &n, &q);
  for(i=1;i<=n;i++){
    scanf("%s",a[i]);
    update(i,isround(a[i]));
  }
  for(i=0;i<q;i++) {
    scanf("%lld", &l);
    if(l==1){
      scanf("%lld%lld", &l, &r);
      printf("%lld\n", query(r)-query(l-1));
    }
    else{
      scanf("%lld", &l);
      update(l,-isround(a[l]));
      scanf("%s", a[l]);
      update(l,isround(a[l]));
    }
  }
  return 0;
}
