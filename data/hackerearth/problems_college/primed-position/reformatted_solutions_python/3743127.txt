N, primes = 1367000, [1,]
mod = 1000000007
check = [True]*(N+1)
check[0] = check[1] = False

for i in range(2, 1+N//2):
    for j in range(2, 1+N//i):
        check[i*j] = False

for i in range(2, N+1):
    if check[i]:
        primes.append(i)

for _ in range(int(raw_input())):
    n, m = map(int, raw_input().split())
    print(primes[primes[m]] % mod * primes[primes[n]] % mod) % mod
