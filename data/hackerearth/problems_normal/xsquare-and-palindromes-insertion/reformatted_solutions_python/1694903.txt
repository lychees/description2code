t = int(raw_input())
for i in range(t):
    string = raw_input()
    chars = [0]*26
    for char in string:
        chars[ord(char) - 97] += 1
    even = 0
    odd = 0
    for count in chars:
        if count != 0:
            if count % 2 == 0:
                even += 1
            else:
                odd += 1
    if odd == 0:
        print 0
    else:
        print odd-1
