#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n , ans = 0 , a , b , c, check;

    cin >> n;

    for(int i = 0; i < n; i++){

        cin >> a >> b >> c;

        check = b * b - 4 * a * c;
        if(check >= 0) ans++;

    }

    cout << ans;

    return 0;
}
