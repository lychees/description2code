int main()
{
	vector<string> as;
	map<string, int> mp;
	int n, no;
	string x;
	scanf("%d",&n);
	int out1 = -10000;
	string name;
	for(int i = 0; i < n; i++) {
		cin >> x >> no;
		if(mp[x] == 0) {
			mp[x] = no;
			if(mp[x] > out1) {
				out1 = mp[x];
				name = x;
			}
		}
		else {
			mp[x] += no;
			if(mp[x] > out1) {
				out1 = mp[x];
				name = x;
			}
		}
	}
	cout << name << endl;
	return 0;
}
