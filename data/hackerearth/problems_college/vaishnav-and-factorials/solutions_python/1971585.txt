for i in range(int(input())):
    f=1
    n=int(input())
    for j in range(1,n+1):
        f=f*j
    #print(f)
    count=0
    while int(f)>0:
        temp=int(f%10)
        #print(temp)
        if temp==4:
            count+=1
        if temp==7:
            count+=1
        f/=10
    print(count)
