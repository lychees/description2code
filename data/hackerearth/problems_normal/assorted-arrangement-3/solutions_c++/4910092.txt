#include <iostream>
using namespace std;
int current = 0;

bool check(int x, int c[] , int max)
{
	if(max == x)
	return true;
	else{
		int count = 0;
		for(int i = x ; i <= max ; i++)
		{
			if((current % c[i]) == 0)
			count++;
			if(count > 1)
			break;
		}
		if(count == 1)
		return true;
		else 
		return false;
		
	}
}

int findNum(int x,int c[],int max)
{
	
		int d = current/c[x];
		while(true)
		{
		current = c[x]*(d+1);
		if(check(x,c,max))
		break;
		else
		d++;
		}
	
	return current;
}

int main()
{
    int i,n,maxCol;
    int c[100],x;
    cin>>n>>maxCol;
    for(i = 1 ; i <= maxCol; i++)
    cin>>c[i];
    
    while(n--)
    {
      cin>>x;
      current = findNum(x,c,maxCol);
    }
    cout<<current;
    return 0;
}
