#include <iostream>
using namespace std;

const long long MOD = 1e9 + 7;

int main()
{
    int t;
    cin >> t;
    
    while (t--) {
    	long long n, k;
    	cin >> n >> k;
    	
    	long long ans = 1;
    	while (n--) {
    		ans = ans * k % MOD;
    		k--;
    	}
    	cout << ans << endl;
    }
}
