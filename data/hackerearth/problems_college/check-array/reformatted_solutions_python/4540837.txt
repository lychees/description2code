for _ in xrange(input()):
    pre, flag = [0]*100001, 0
    size, bucket = input(), map(int, raw_input().split())
    for i in xrange(size):
        pre[i+1] = pre[i] + bucket[i]
    for ind in xrange(size):
        if pre[ind] == pre[size]-pre[ind+1]:
            flag = 1
            break
    print "YES" if flag else "NO"
