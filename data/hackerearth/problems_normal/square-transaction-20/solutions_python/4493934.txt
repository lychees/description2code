T = input()
A = map(int,raw_input().split())
for i in range(1,len(A)):
	A[i]=A[i]+A[i-1]
Target=input()

def find(Q,m,M):
	if m==M:
		if Q>A[m]:
			if m<T-1:
				print m+2
			else:
				print -1
		else:
			print m+1
	else:
		mid=(m+M+(m+M)%2)/2
		if Q>A[mid]:
			find(Q,mid,M)
		elif Q<A[mid]:
			find(Q,m,mid-1)
		else:
			print mid+1

for i in range(Target):
	Q=input()
	find(Q,0,T-1)
