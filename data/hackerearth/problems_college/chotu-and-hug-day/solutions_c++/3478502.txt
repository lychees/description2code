#include <iostream>
using namespace std;

int main()
{
   int t ;
   cin>>t;
   int a[1000], i ;
   while(t--) {
   	int n , h;
   	cin>>n>>h;
   	int min = 10000;
   	for(i = 0 ; i < n ;i++) {
   		cin>>a[i];
   		int d = abs(h-a[i]);
   		if(d < min )
   			min = d;
   	}
   	cout<<min<<endl;
 
   }
    return 0;
}
