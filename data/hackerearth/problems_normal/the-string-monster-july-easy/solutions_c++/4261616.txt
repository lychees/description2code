#include<bits/stdc++.h>
using namespace std;
typedef long long int uli;
string d[33];

int main(){
   int t,n;
   string s;
   cin>>t;
   while(t--){
      cin>>n;
      for(int i=0;i<n;i++)
         cin>>d[i];
      cin>>s;
      vector<int>f(26,0);
      set<vector<int> >seto;
      for(char x:s)f[x-'a']++;

      int to=min(n-1,15);
      //0...to
      for(int b=0;b<(1<<(to+1));b++){
         vector<int>fb(26,0);
         for(int i=0;i<=to;i++){
            if(b&(1<<i)){
               for(int ch:d[i])fb[ch-'a']++;
            }
         }
         seto.insert(fb);
      }
      if(to==n-1){
          if(seto.count(f))puts("YES");
          else puts("NO");
          continue;
      }
      //to+1..n-1
      int len=n-1-to;
      bool ok=false;
      for(int b=0;b<(1<<len) && !ok;b++){
//         vector<int>fb(26,0);
         vector<int>rem=f;
         for(int i=0;i<len;i++){
            if(b&(1<<i)){
               for(int ch:d[i+to+1])rem[ch-'a']--;
            }
         }
         if(seto.count(rem))ok=true;
      }
      if(ok)puts("YES");
      else puts("NO");
   }
   return 0;
}
