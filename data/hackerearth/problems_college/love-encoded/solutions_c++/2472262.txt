#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int d;
    	cin>>d;
    	string s;
    	cin>>s;
    	for(int i=0;i<s.length();i++){
    		if(s[i]-d<'A'){
    			s[i]= (char)('Z'+(s[i]-d-'A'+1));
    		}else{
    			s[i]= (char)(s[i]-d);
    		}
    		
    	}
    	cout<<s<<endl;
    }
    return 0;
}
