#include <bits/stdc++.h>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int t,n,a[100000],k;
    cin>>t;
    while(t--) {
    	cin>>n>>k;
    	for(int i=0;i<n;i++)
    		cin>>a[i];
    	sort(a,a+n);
    	long long int cnt = 0, i = n-1, l = 0;
    	while(l<k && i>=0) {
    		cnt+=a[i];
    		i--;
    		l++;
    	}
    	cout<<cnt<<endl;
    }
    return 0;
}
