#include <iostream>
using namespace std;

int main()
{
    int n,i,k,j,x;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	if(i==0||i==(n-1)){for(j=0;j<n;j++)
    						cout<<"*";}
    	else{x=((n)-i);
    		for(j=1;j<=x;j++)
    		{
    			if(j!=x){cout<<" ";}
    			else {cout<<"*";}
    		}}
    		cout<<"\n";
    }
}
