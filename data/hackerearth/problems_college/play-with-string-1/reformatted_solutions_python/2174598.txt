t = input()
for i in range(t):
    stri = raw_input()
    rStri = stri[::-1]
    x = len(stri)
    f = 0
    for i in range(1, x):
        if abs(ord(stri[i]) - ord(stri[i-1])
               ) != abs(ord(rStri[i]) - ord(rStri[i-1])):
            f = 1
            break
    if f == 0:
        print "Funny"
    else:
        print "Not Funny"
