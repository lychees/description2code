#include <iostream>
#include <stdio.h>
#include <stdlib.h>
#include <math.h>

using namespace std;

int main()
{
    int a;
    int b;
    cin >> a >> b;
    
    int diff = a - b;
    int ans[10];
    int i = 0;
    while(diff > 0)
    {
        ans[i++] = diff % 10;
        diff = diff / 10;
    }
    
    for (int j = i-1;j >= 0;j--)
    {
    	if (j == 0)
    		ans[j] = (ans[j] + 1) % 10;
        cout << ans[j];
    }
    
    return 0;
}
