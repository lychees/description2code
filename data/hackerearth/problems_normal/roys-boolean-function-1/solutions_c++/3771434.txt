#include <iostream>
#include <math.h>
using namespace std;
int gcd(int n1,int n2){
	while(n1 != n2) {
        if(n1 > n2)
            n1 -= n2;
        else
            n2 -= n1;
    }
    return n1;
}

int main() {
	
    int n,count,t;
    cin>>t;
    while(t--){
    	count=0;
    cin >> n;
    for(int i=n-1;i>0;i--){
    	if(gcd(n,i)==1)
    	count++;
    }
    if(count==0)
    cout<<"FALSE\n";
    else
    for(int i=sqrt(count);i>=1;i++)
    {
    	if(count%i==0)
    	{
    		if(i==1&&count!=1)
    		cout<<"TRUE\n";
    		else
    		cout<<"FALSE\n";
    		break;
    	}
     }
    }
    
    return 0;
}
