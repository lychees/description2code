#include <iostream>
using namespace std;

const double eps = 1e-9;
struct point {
	int x,y;
};

point segments[110][2];
double slope[110];
bool used[110];
int main()
{
    int n,x,y = 10001;
    scanf("%d %d",&x,&n);
    for(int i=0;i<n;++i)
    {
    	scanf("%d %d %d %d",&segments[i][0].x,&segments[i][0].y,&segments[i][1].x,&segments[i][1].y);
    	slope[i] = (segments[i][1].y - segments[i][0].y) * 1.0 / (segments[i][1].x - segments[i][0].x);
    }
    while(1)
    {
    	int l = -1;
    	double maxy = -1;
    	for(int i=0;i<n;++i)
    	{
    		if(used[i]) continue;
    		if(x<=segments[i][1].x && x>=segments[i][0].x) {
    			double ty = slope[i] * (x - segments[i][0].x) + segments[i][0].y;
    			if(ty > y + eps) continue;
    			if(ty > maxy) {
    				maxy = ty;
    				l = i;
    			}
    		}
    	}
    	if(l == -1) break;
    	
    	if(segments[l][1].y > segments[l][0].y) {
    		x=segments[l][0].x;
    		y=segments[l][0].y;
    	} else {
    		x=segments[l][1].x;
    		y=segments[l][1].y;
    	}
    	used[l]=1;
    }
    printf("%d\n",x);
    return 0;
}
