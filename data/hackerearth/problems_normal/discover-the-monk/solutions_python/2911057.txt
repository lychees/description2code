def bsearch(arr , n , num):
	l , r = 0 , n-1
	while l <= r:
		mid = (l+r)/2
		if arr[mid] == num:
			return 1
		elif arr[mid] > num:
			r = mid - 1
		else:
			l = mid + 1
	return 0
n , q = map(int , raw_input().split())
arr = map(int , raw_input().split())
arr = sorted(arr)
for i in range(q):
	num = int(raw_input())
	if bsearch(arr , n , num):
		print "YES"
	else:
		print "NO"
