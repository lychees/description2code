#include<iostream>
using namespace std;
int main()
{
    long long arr[1000],t,x,y,n,i=0,j;
    cin>>t;
    for(;t>0;t--)
    {
    cin>>x>>y>>n;
        for(i=0;i<n;i++)
        {
            if(i<x)
                arr[i]=y;
            else
            {
            arr[i]=0;
                for(j=i-1;j>=i-x;j--)
                {
                arr[i]+=arr[j];
                }
            }
        }
        cout<<arr[n-1]<<"\n";
    }
}
