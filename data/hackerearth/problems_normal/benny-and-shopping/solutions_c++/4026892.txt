#include <iostream>
using namespace std;

int main()
{
    char ans[1009],inp[1009];
    int N;
    scanf("%d",&N);
    fflush(stdin);
    N++;
    while(N-->0)
    {
    	
    	gets(inp);
    	if(inp[0]=='\0')
    		continue;
    	for(int i=0;inp[i]!='\0';i++)
    	{
    		if(inp[i]=='$')
    		{
    			i++;
    			while(inp[i]=='0' || inp[i]==' ')
    				i++;
    			int l=0;
    			ans[l++] = '$';
    			while(!((inp[i]<=90 && inp[i]>=65) || (inp[i]>=97 && inp[i]<=122)))
    			{
    				if(inp[i]=='\0')
    					break;
    				if(inp[i]!=' ')
    					ans[l++] = inp[i];
    				i++;
    			}
    			//cout<<l;
    			if(l==1)
    				ans[l++] = '0';
    			ans[l] = '\0';
    			break;
    		}
    	}
    	cout<<ans<<endl;
    }
}
