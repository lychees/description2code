#include <bits/stdc++.h>
using namespace std;
#define M 1000001

 
int main()
{
        long long int a1[M];
    
    a1[0]=a1[1]=a1[2]=a1[3]=a1[4]=0;
    for(int i=5;i<=M;i++)
    {
    	long long int count=0;
    	for(int j=5;i/j>=1;j*=5)
    	{
    		count+=(i/j);
    	}
    	a1[i]=count+a1[i-1];
    }
    int t;
    cin>>t;
    while(t--)
    {
    	int a,b;
    	cin>>a>>b;
        cout<<a1[b]-a1[a-1]<<endl;
    }
    return 0;
}
