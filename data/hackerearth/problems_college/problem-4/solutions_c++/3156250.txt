#include<bits/stdc++.h>
using namespace std ;
int main()
{
  int n, c, d ;
  char temp ;
  string array ;
  cin >> array ;
  n = array.size() ;
  for (c = 0 ; c < ( n - 1 ); c++)
  {
    for (d = 0 ; d < n - c - 1; d++)
    {
      if (array[d] > array[d+1])
      {
        temp       = array[d];
        array[d]   = array[d+1];
        array[d+1] = temp ;
      }
    }
  }
 cout << array ;
 cout << endl ;

  return 0;
}
