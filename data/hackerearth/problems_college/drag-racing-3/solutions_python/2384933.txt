map = {}
details = []
nTestCases = int(raw_input())
for i in range(0,(2**nTestCases)-1):
	matches = raw_input()
	details.append(matches)
	racer1,racer2 = ''.join(matches.split(" ")[0]),''.join(matches.split(" ")[1])
	map[racer1],map[racer2] = "true","true"

for y in details:
	loser = ''.join(y.split(" ")[1])
	map[loser] = "false"

for k in map.keys():
    if map[k] == "true":
    	print k
