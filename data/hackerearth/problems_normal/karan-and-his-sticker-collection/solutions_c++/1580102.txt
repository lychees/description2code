#include <iostream>
#include <math.h>
using namespace std;

int main()
{
	std::ios::sync_with_stdio(false);
    int t,count=0;
    cin>>t;
    unsigned long long num;
    for(int i=0;i<t;i++)
    {
    	cin>>num;
    	if(num<3*pow(10,8))
    	{
    		count++;
    	}
    }
    cout<<count<<endl;
    return 0;
}
