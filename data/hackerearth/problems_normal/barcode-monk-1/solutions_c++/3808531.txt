#define ll long long

#define MOD 1000000007
#define INF 0x3f3f3f3f
#define dbg(x) { cout<< #x << ": " << (x) << endl; }
#define dbg2(x,y) { cout<< #x << ": " << (x) << " , " << #y << ": " << (y) << endl; }

#define sz(x) (int)x.size()
ll dp[1001][2];
ll w[1001];
int main()
{
    int i,j;
    int n,k;
    cin >> n >> k;
    assert(1 <= n && n <= 1000);
    assert(1 <= k && k <= 1000);
    w[0] = 1;
    while(k--)
    {
        int ww;
        cin >> ww;
        assert(1 <= ww && ww <= 1000);
        if(ww > n)
            continue;
        for(i=ww;i<=n;i++)
            w[i] |= w[i-ww];
    }
    dp[0][0] = 1;
    for(i=1;i<=n;i++)
    {
        dp[i][0] = (dp[i-1][0] + dp[i-1][1])%MOD;
        for(j=1;j<=i;j++)
        {
            if(w[j])
            {
                dp[i][1] += dp[i-j][0];
                dp[i][1] %= MOD;
            }
        }
    }
    ll ans = (dp[n][0] + dp[n][1])%MOD;
    cout << ans << endl;
    return 0;
}
