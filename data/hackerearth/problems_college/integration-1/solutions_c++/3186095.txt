#include <iostream>
using namespace std;

int main() {
	//code
	long long T,N,P, U, R, S,i=1;
	cin>>T;
	while(T--)
	{
	    cin>>N>>P>>U>>R>>S;
	    while(N--)
	    {
	        if(P+R<=U)
	            P+=R;
	        else if(P-S>0)
	            P-=S;
	    }
	    cout<<"Case #"<<i++<<": "<<P<<endl;
	}
	return 0;
}
