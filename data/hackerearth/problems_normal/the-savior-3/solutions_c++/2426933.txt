#include <iostream>
using namespace std;

int combination(int ar[], int n) {
	int count = 0;
	for (int i = 0; i < n; i++) {
		for (int j = i+1; j < n; j++) {
			if (ar[i] != ar[j]) count++;
		}
	}
	return count;
}

int main()
{
    int tests;
    cin >> tests;
    
    while (tests-- > 0) {
    	int n;
    	cin >> n;
    	
    	int evens[n], odds[n], e = 0, o = 0;
    	
    	for (int i = 0; i < n; i++) {
    		int number;
    		cin >> number;
    		
    		if (number%2 == 0) evens[e++] = number;
    		else               odds[o++] = number;
    	}
    	
    	cout << (combination(evens, e) + combination(odds, o)) << endl;
    }
    
    return 0;
}
