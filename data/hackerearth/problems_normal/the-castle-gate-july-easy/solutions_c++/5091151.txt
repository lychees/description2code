#include <iostream>
using namespace std;

int N, answer;

int main()
{
	int T;
	cin >> T;
    for(int tc = 1; tc <= T; tc++)
    {
    	answer = 0;
    	cin >> N;
    	for(int i = 1; i < N; i++)
    		for(int j = i + 1; j <= N; j++)
    		{
    			if((i ^ j) <= N)
    				answer++;
    		}
    	cout << answer << endl;
    }
    return 0;
}
