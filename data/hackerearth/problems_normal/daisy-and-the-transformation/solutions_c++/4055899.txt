#include <iostream>
#include <string>
using namespace std;

int main()
{
    int t, n, count;
    string s1, s2;
    cin >> t;
    while(t--) {
    	cin >> s1 >> s2;
    	cin >> n;
    	count = 0;
    	if(s1.length() != s2.length()) {
    		cout << "No" << endl;
    		continue;
    	}
    	for(int i=0; i<s1.length(); i++) {
    		if(s1[i] != s2[i]) {
    			count++;
    		}
    	}
    	if(count <= n && (count == n || (n-count)%2 == 0)) {
    		cout << "Yes" << endl;
    	} else {
    		cout << "No" << endl;
    	}
    }
    return 0;
}
