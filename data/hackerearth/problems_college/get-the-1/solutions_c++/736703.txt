#include <iostream>
using namespace std;

int main()
{
    int t;
   cin>>t;
   int a[]={0,1,4,9,16,25,36,49,64,81,100};
   for(int i=1;i<=t;i++)
   {
   	long long n;
   	cin>>n;
   	long long sum =0;
   	while(n>9)
   	{
   		sum=0;
   		while(n!=0)
   		{
   		sum+=a[n%10];
   		n=n/10;
   		}
   		n=sum;
   		
   }
   if(n==1)
   	cout<<"Case #"<<i<<":"<<"YES\n";
   
   
   else
   	cout<<"Case #"<<i<<":"<<"NO\n";
   }
    return 0;
}
