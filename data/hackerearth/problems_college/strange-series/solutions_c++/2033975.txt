#include<bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
	ll n;
	ll t,i,j;
	ll c[199][100];
	cin>>t;
	while(t--)
	{
		
		cin>>n;
		ll k,ans=0;
		for(k=0;k<=n;k++)
		{
			c[199][100]={0};
	for(i=0;i<=n;i++)
	{
		for(j=0;j<=min(i,k);j++)
		{
			if(i==j || j==0)
			{
				c[i][j]=1;
			}
			else
			{
				c[i][j]=c[i-1][j-1]+c[i-1][j];
			}
		}
		
	}
	ans=ans+c[n][k];
		}
	cout<<ans-1;
	cout<<'\n';
	}
	return 0;
}
