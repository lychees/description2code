    #include<bits/stdc++.h>
    using namespace std;
    int main()
    {
    long long t;
    scanf("%lld",&t);
    while(t--)
    {
    long long n,b;
    cin>>n>>b;
    long long x=n/b;
    long long y=x+1;
    if(abs(b*x-n)<abs(b*y-n))
    {cout<<b*x<<endl;
    continue;
    }
    if(abs(b*x-n)==abs(b*y-n))
    {cout<<b*y<<endl;
    continue;
    }
    cout<<b*y<<endl;
    }
    return 0;
    }
