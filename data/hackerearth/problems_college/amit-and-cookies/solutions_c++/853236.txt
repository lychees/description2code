#include <iostream>
using namespace std;

int main()
{
	int N,K,i;
	cin>>N>>K;
	int A[N];
	for(i=0;i<N;++i)
	{
		cin>>A[i];
	}
	int j,k,m=0,sum=0;
	for(j=0;j<N;++j)
	{
		for(k=j+1;k<N;++k)
		{
			if (A[j]>A[k])
			{
				int temp=A[j];
				A[j]=A[k];
				A[k]=temp;
			}
		}
	}
	int x;
	for(x=0;x<N;++x)
	{
		sum+=A[x];
		if(sum>K) {break;}
	}
	cout<<x;
}
