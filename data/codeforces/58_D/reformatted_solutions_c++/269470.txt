using namespace std;

string s[10100];
string s1;
bool was[10100];
int n;

int main() {
  //	freopen("anarc05b.in","r",stdin);
  //	freopen("anarc05b.out","w",stdout);
  scanf("%d", &n);
  int x = 0;
  for (int i = 0; i < n; i++) {
    cin >> s[i];
    x += s[i].size();
    was[i] = false;
  }
  cin >> s1;

  for (int i = 0; i < n; i++)
    s[i] = s[i] + s1;

  sort(s, s + n);

  x = 2 * x / n;

  for (int i = 0; i < n; i++) {
    if (was[i])
      continue;
    for (int j = i + 1; j < n; j++) {
      if (was[j])
        continue;
      if (s[i].size() + s[j].size() == x + 2) {
        was[i] = true;
        was[j] = true;

        string ss1, ss2;

        ss1 = s[i] + s[j];
        ss1.erase(ss1.size() - 1, 1);
        cout << ss1 << endl;
        break;
      }
    }
  }
  //	in.getline(s);

  return 0;
}
