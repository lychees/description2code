#include <iostream>
#include <cmath>
#define EPS 1e-9
using namespace std;

int main() {
	int tc; cin >> tc;
	while(tc--){
		int n; cin >> n;
		double lowerDouble = log(n)/log(2);
		int lower = (int) lowerDouble;
		if(lowerDouble-lower < EPS){
			cout << "1" << endl;
		}
		else{
			int additional = (n-pow(2,lower))*2;
			cout << additional+1 << endl;
		}
	}
	return 0;
}
