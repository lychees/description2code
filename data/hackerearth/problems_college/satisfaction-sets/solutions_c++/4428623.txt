#include<bits/stdc++.h>

using namespace std;

int main()
{
	
	int n,m,no,i,j,flag;
	cin>>n;
	vector<int> v[n];
	
	for(i=0;i<n;i++)
	{
		cin>>m;
		for(j=0;j<m;j++)
		{
			cin>>no;
			v[i].push_back(no);
			
		}
	}
	
	set<int> u,d;
	for(i=0;i<n;i++)
	u.insert(v[i].begin(),v[i].end());
	
	set<int> :: iterator itr;
	
	for(itr =u.begin();itr!=u.end();itr++)
	{
		flag=0;
		for(int k=0;k<n;k++)
		if(find(v[k].begin(),v[k].end(),*itr) == v[k].end())
		flag=true;
		
		if(flag)
		d.insert(*itr);
		
	}
	
	int un=accumulate(u.begin(),u.end(),0);
	int df = accumulate(d.begin(),d.end(),0);
	
	int res=(df-un)%(int)(pow(10.0,9.0));
	printf("%d",res);
	
	return 0;
	
}
