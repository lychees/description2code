#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
   int t,n;
   long x,y,sumx,sumy;
   cin>>t;
   while(t--)
   {
   	sumx=sumy=0;
   	cin>>x>>y>>n;
   	const int N=n;
   	long arr[N];
   	for(int i=0;i<N;i++)cin>>arr[i];
   	sort(arr,arr+N);
   	for(int i=0;i<N;i++)
   	{
   		if(sumx+arr[i]<=x)sumx+=arr[i];
   		if(sumy+arr[i]<=y)sumy+=arr[i];
   	}
   	if(sumx>sumy)cout<<"Chef Won"<<endl;
   	else if(sumx<sumy)cout<<"Jhool Won"<<endl;
   	else cout<<"Tie"<<endl;
   }
    return 0;
}
