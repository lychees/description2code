#include <iostream>
using namespace std;

int main()
{
	int input[100];
	int size;
	cin>>size;
	for (int i=0; i<100; i++)
	{
		cin>>input[i];
	}

	for(int i=0; i< size; i++) {
		bool isPrime = true;
		for(int j=0; j < size; j++) {
			if(i!=j && input[i] != input[j] && (input[i] % input[j]) == 0) {
				isPrime = false;
				break;
			}
		}
		if(isPrime) {
			cout<<input[i]<< " ";
		}
	}
    return 0;
}
