#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    while(t--)
    {
    	int a0, a1;
    	int c, n;
    	cin>>a0>>a1>>c>>n;
    	int flag = 0;
    	int rem = 0, cur;
    	if(n == 1) cur = a0;
    	else cur = (a0 * 10 + a1);
    	for(int i = 2; i<n; i++)
    	{    		
    		rem = cur % 41;
    		int newdigit = ((a1 * c) + a0) % 10;
    		cur = rem * 10 + newdigit;
    		a0 = a1;    		
    		a1 = newdigit;
    	}
    	if(cur % 41) cout<<"NO\n";
    	else cout<<"YES\n";
    }
    return 0;
}
