#include <algorithm>
#include <iostream>
#include <string>

using namespace std;

int main() {
  string s;
  cin >> s;
  int n = s.length(), MAX = 0;
  for (int i = 0; i < n; i++) {
    int t = 1;
    for (int j = i + 1; j < n; j += 2)
      if (s[j] == s[i]) {
        j--;
        t++;
      }
    MAX = max(MAX, t);
  }
  cout << MAX << endl;
  return 0;
}
