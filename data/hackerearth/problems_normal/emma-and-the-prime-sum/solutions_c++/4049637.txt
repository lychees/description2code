#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll arr[1000009],p[1000009];
void siv()
{
    ll i,j;
    for(i=2; i*i<=1000000; i++)
    {
        if(arr[i]==0)
        {
            for(j=i*i; j<=1000000; j+=i)arr[j]=1;
        }
    }
    for(i=2; i<=1000000; i++)
    {
        if(arr[i]==0)
            p[i]=p[i-1]+i;
        else if(arr[i]==1) p[i]=p[i-1];
    }
}
int main()
{
    siv();
    ll t,i,x,y;
    //for(i=2; i<=20; i++)cout<<p[i]<<"  ";
    cin>>t;
    while(t--)
    {
        cin>>x>>y;

        cout<<p[y]-p[x-1]<<endl;
    }

}
