 #include <bits/stdc++.h>
using namespace std ;

long long int M = 1e9+9 ;
long long int power(long long int a,long long int b){
	long long int res = 1 ;
	while(b){
		if(b%2){
			res = res*a ;
			res %= M ;
		}
		b /= 2 ;
		a = a*a ;
		a %= M ;
	}
	
	return res ;
}
int main(){
	int T ;
	long long int N ;
	cin >> T ;
	while(T--){
		cin >> N ;
		cout << power(26,(N+1)/2) << endl ;
	}
    return 0;
}
