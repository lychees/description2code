#include <iostream>
using namespace std;
int findsum(int x)
{
	int sum = 0;
	while(x>0)
	{
		sum = sum + x%10;
		x = x/10;
		
	}
return sum;
	
}

int main()
{
	int n;
	cin>>n;
	int a[60] = {0};
	
	for(int i = 1; i<=n; i++)
	cin>>a[i];
	
	int l,b;
	cin>>l>>b;
	int ans=0;
	for(int i = 1; i<=n; i++)
	{
		int val = a[i]*findsum(a[i])*l*b;
		//1-10 having 25% discount,11-20 having 30% discount ,20-50 having 20% discount
		if(a[i]>=1&&a[i]<=10)
		val=(val*3)/4;
		
		else if(a[i]>=11&&a[i]<=20)
		val=(val*7)/10;
		
		else if(a[i]>=20&&a[i]<=50)
		val=(val*4)/5;
		
		ans = ans + val;
		
		
	}
	
	cout<<ans;
	
    return 0;
}
