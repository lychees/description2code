#include <iostream>
#include <stdio.h>
#include <string.h>
using namespace std;

int min(int a, int b)
{   return a < b ? a : b;  }


int findMinInsertionsForPalindrome(char str[], int n)
{
    
    int table[n][n], l, h, gap;

   
    memset(table, 0, sizeof(table));

    for (gap = 1; gap < n; ++gap)
        for (l = 0, h = gap; h < n; ++l, ++h)
            table[l][h] = (str[l] == str[h])? table[l+1][h-1] :
                          (min(table[l][h-1], table[l+1][h]) + 1);

    return table[0][n-1];
}


int main()
{
    int t;
    cin>>t;
    char c;
    char carr[25];
    while(t--) {
    	int n;
    	cin>>n;
    	for(int i=0; i<n; i++) {
    		
    		cin>>c;
    		carr[i] = c;
    		
    	}
    	int ans = findMinInsertionsForPalindrome(carr, n);
    	cout<<ans<<"\n";
    }
    return 0;
}
