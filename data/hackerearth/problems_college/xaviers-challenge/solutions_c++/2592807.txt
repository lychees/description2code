#include<bits/stdc++.h>

using namespace std;

int main()
{
	
	string str;
	int t,i,l;
	int tot;
	cin>>t;
	cin.ignore();
	while(t--)
	{
	getline(cin,str);
	tot=0;
	l=str.length();
	for(i=0;i<l;i++)
		tot=tot+(int)str[i];
	tot=tot/l;
	if(tot%2)
	reverse(str.begin(),str.end());
	cout<<str<<endl;
	}
	
}
