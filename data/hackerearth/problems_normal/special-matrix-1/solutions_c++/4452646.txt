#include<bits/stdc++.h>

using namespace std;



int main(void)
{
	int t,sti,stj,n;
	char ch;
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(int i=0;i<n;i++)
		{	for(int j=0;j<n;j++)
			{
				cin>>ch;
				if(ch=='*')
				{
					sti=i;
					stj=j;
				}
			}
		}
		cout<<(abs(sti-n/2)+abs(stj-n/2))<<endl;
	}
	return 0;
}
