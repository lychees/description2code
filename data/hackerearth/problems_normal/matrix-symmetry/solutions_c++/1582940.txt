#include<iostream>
using namespace std;

int main()
{
	int t,n,i,j,hor=1,ver=1;
	char a[50][50];
	cin>>t;
	
	while(t--)
	{
		cin>>n;
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
			cin>>a[i][j];
		}
		
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
			{
				
				if(a[i][j]!=a[n-i-1][j])
				hor=0;
				
				if(a[i][j]!=a[i][n-j-1])
				ver=0;
			}
		}
		if(hor!=0 && ver!=0)
		cout<<"BOTH";
		else if(hor!=0)
		cout<<"HORIZONTAL";
		else if(ver!=0)
		cout<<"VERTICAL";
		else
		cout<<"NO";
		
		cout<<endl;
		hor=ver=1;
		
	}
	return 0;
}
