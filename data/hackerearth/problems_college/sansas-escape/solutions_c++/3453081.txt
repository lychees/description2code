int main()
{
   ios::sync_with_stdio(false);
   int n;
   cin >> n;
   vector<int> v(n + 1, 0);
   for (int i = 1; i <= n; i++) cin >> v[i];
   for (int i = 1; i <= n; i++) v[i] ^= v[i - 1];
   int Q;
   cin >> Q;
   while (Q--) {
   	      int a, b;
   	      cin >> a >> b;
   	      a++;
   	      b++;
   	      cout << (v[b] ^ v[a - 1]) << endl;
   }
   return 0;
}
