#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n, i, j, count =0;
    	cin>>n;
    	for(i =1;i<=n;++i){
    		for(j =i;j<=n;++j){
    			if(i == j){
    				continue;
    			}
    			else{
    				int pair =i^j;
    				if(pair<=n){
    					count++;
    				}
    			}
    		}
    	}
    	cout<<count<<endl;
    }
    return 0;
}
