using namespace std;
typedef long long Long;

int main()
{
	int TC;
	cin >> TC;
	while(TC-->0){
		int N;
		cin >> N;
		Long oo = 0, ee = 0, oe = 0, eo = 0;
		for(int i = 0; i < N; ++i){
			Long u,v,c;
			cin >> u >> v >> c;
			u--;v--;
			if((u+v)%2) {
				if((c)%2){
					oo++;
				} else {
					oe++;
				}
			} else {
				if((c)%2){
					eo++;
				} else {
					ee++;
				}
			}
				
		}
		if( (oo && oe) || (eo && ee) || (oe && ee) || (eo && oo) ){
			cout << "No" << endl;
		} else {
			cout << "Yes" << endl;
		}
	}
	
	
}
