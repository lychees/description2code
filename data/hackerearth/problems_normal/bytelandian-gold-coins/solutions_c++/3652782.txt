using namespace std;
map<ll,ll> m;
ll dp(ll n)
{
    if(n==0)
    return 0;
	if(m[n]!=0)
	return m[n];
	m[n]=max(n,dp(n/2)+dp(n/3)+dp(n/4));
	return m[n];
}
