#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    while(n--)
    {
    	int ar[26]={0},min=101;
    	string s;
    	cin>>s;
    	for(int i=0;i<s.length();i++)
    	{
    		ar[s[i]-'a']++;
    	}
    	int i='r'-'a',j='u'-'a',k='b'-'a',l='y'-'a';
    	if(ar[i]<min)
    		min=ar[i];
    	if(ar[j]<min)
    		min=ar[j];
    	if(ar[k]<min)
    		min=ar[k];
    	if(ar[l]<min)
    		min=ar[l];
    	cout<<min<<endl;
    }
    return 0;
}
