#include <bits/stdc++.h>

using namespace std;

int main()
{
    long long int n,tc;
    cin >> tc;

    while(tc--){
    cin >> n;
    int count = 0;

    while(n > 0)
    {
        count++;
        n /= 2;
    }

    long long int ans = pow(2,count) - 1;
    cout << ans << endl;
    }

}
