#include <iostream>
using namespace std;

int main()
{
	string a;
    cin>>a;
    int len = a.length();
    int ch[26],key,max;
    for(int i=0;i<26;i++){
       ch[i]=0; 
    }
    for(int i=0;i<len;i++){
        ch[(a[i]-97)]++;
    }
    key = 0;
    max=ch[0];
    for(int i=1;i<26;i++){
        if(ch[i]>max){
            key = i;
            max = ch[i];
        }
    }
    int x=0,y=0;
    string str="headrush";
    for(int i=0;i<8;i++){
        if(key == (str[i]-97)){
            x=1;
            break;
        }
    }
    str="cubingclub";
    for(int i=0;i<10;i++){
        if(key == (str[i]-97)){
            y=1;
            break;
        }
    }   
    if(x & y){
        cout<<"Thanks both";
    }
    else if(x){
        cout<<"Thanks head rush";
    }
    else if(y){
        cout<<"Thanks cubing club";
    }
    else{
        cout<<"none";
    }
    return 0;
}
