#include<iostream>
#include<algorithm>
#include<vector>
#include<string.h>
using namespace std;
int main()
{
  int t;
  cin>>t;
  while(t--) {
    long long i,j,n,m,sum=0,temp;
    cin>>n;
    if(n==1) { cout<<"0\n"; continue; }

  }

}
