#include <iostream>
using namespace std;

long long gcd(long long a,long long b)
{
	long long r;
	while(b!=0)
	{
		r=a%b;
		a=b;
		b=r;
	}
	return a;
}
int main()
{
    int t;
    long long p,q;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%lld %lld",&p,&q);
    	if(gcd(max(p,q),min(p,q))==1) printf("Is a Co-Prime\n");
    	else printf("Not a Co-Prime\n");
    }
    return 0;
}
