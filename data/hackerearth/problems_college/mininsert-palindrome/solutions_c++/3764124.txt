#include<bits/stdc++.h>
//zone
//time complexity o(n^2)
//space = same as above
using namespace std;
int main()
{

    int t,n;
    cin >> t;
    while(t--)
    {
    cin >> n;
    string s;
    cin >> s;
    int dp[s.length()][s.length()];
    memset(dp,0,sizeof(dp));
    int gap,l,h;
    for(gap = 1 ; gap < s.length() ;gap++)
    {
        for(l = 0 , h = gap; h < s.length(); ++l,++h)
        {
            if(s[l] == s[h])
            {
                dp[l][h] = dp[l + 1][h -1];
            }
            else
                dp[l][h] = min(dp[l][h - 1],dp[l + 1][h]) + 1;
        }
    }
    cout<<dp[0][s.length() -1]<<endl;
    }
    return 0;
}
