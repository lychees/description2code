#include <iostream>
#include <cmath>
using namespace std;
 
int main()
{
   int t;
   cin>>t;
   int j;
   for(j=0;j<t;j++){
   	int d,i,m,n,k;
   	cin>>d>>i>>m>>n;
   	int arr[n];
   	int c=0;
   	for(k=0;k<n;k++){
   		cin>>arr[k];
 
if(k!=0){
	int num=arr[k]-arr[k-1];
	num=abs(num);
	int numbe;
	if(num>m){
		numbe=num-m;
		if(numbe>d && d<i) c+=d;
		else if(numbe>i && d>i){
		c+=num/m;	
		}
		else c+=numbe;
	}
}
   	}
   	cout<<"Case"<<" "<<"#"<<j+1<<":"<<" "<<c<<endl;
   }
    return 0;
}
