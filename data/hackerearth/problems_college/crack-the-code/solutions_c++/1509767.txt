#include <iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    long long int dp[1000];
    dp[0]=0;
	dp[1]=1;
	for(long long int i=2;i<=1000;i++)
	{
		dp[i]=dp[i-1]+dp[i-2];
	}
    while(t--)
    {
    	long long int x,y;
    	cin>>x>>y;
    	long long int c=0;
    	for(long long int i=x;i<=y;i++)
    	{
    		for(int j=0;j<=1000;j++)
    		{
    			if(i==dp[j])
    			c++;
    		}
    	}
    	if(c%2==0)
    	{
    		cout<<"VALID"<<endl;
    	}
    	else
    	cout<<"INVALID"<<endl;
    }//cout << "Hello World!" << endl;
    return 0;
}
