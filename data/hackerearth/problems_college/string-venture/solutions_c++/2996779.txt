#include <bits/stdc++.h>
using namespace std;
int main()
{
        //printf("Hello World!\n");
        char c;
        int start=0;
        string a,b;
        while(cin>>b){
                if(start!=0)
                        a.append(" ");
                a.append(b);
                start++;
        }
        int l = a.length(),max=0;
        char maxc=a[0];
        map<char,int> ch;
        int ascii=0;
        for(int i=0;i<l;i++){
                if(ch.find(a[i])!=ch.end()){
                        ch[a[i]]++;
                }else
                        ch[a[i]]=1;
                if(ch[a[i]]>max){
                        max = ch[a[i]];
                        maxc = a[i];
                }
                ascii += a[i];
        }
        int flr = floor(ascii/l);
        if(flr%2==1){
        //odd
                cout<<maxc<<endl;
        }else{
                reverse(a.begin(),a.begin()+l);
                cout<<a<<endl;
        }

        return 0;
}
