#include <iostream>
using namespace std;

int main()
{
	int t;
	cin >> t;
	for (int j = 0; j < t; j++) {
		string s;
		cin >> s;
		int size = s.size();
		int ammount = 0;
		
		for (int i = 0; i < s.size() / 2; i++) {
			ammount += abs(s[i] - s[size - 1 - i]);
		}
		
		std::cout << ammount << endl;
	}
    return 0;
}
