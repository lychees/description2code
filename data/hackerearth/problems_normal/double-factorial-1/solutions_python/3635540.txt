def fact(n):
    if n==0:
        return 1
    else:
        return(n*fact(n-1))
t=input()
for i in range(t):
    n,m=map(int,raw_input().split())
    if n<5:
        for j in range(2):
            if j==0:
                x=(fact(n))
            else:
                print(fact(x)%10**m)
    else:
        print "0"
