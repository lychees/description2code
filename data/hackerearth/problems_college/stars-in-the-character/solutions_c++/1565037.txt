#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    int t,count,count1,j;
    int d;
    char m;
    char str[100];
    cin>>t;
    while(t--)
    {
    	cin>>str;
    	count=0;
    	count1=0;
    	for(j=0;j<strlen(str);j++)
    	{
    	    if(str[j]=='A'|| str[j]=='D'|| str[j]=='O'|| str[j]=='P'|| str[j]=='R' || str[j]=='Q')
    		{count1=count1+1;}
    		else if(str[j]=='B')
    		{count1=count1+2;}
    	    d=((int)str[j]+32);
    	    m=(char)d;
    	    if(m=='a' || m=='d' || m=='o' || m=='p' || m=='q' || m=='b' || m=='g' || m=='e')
    		{count=count+1;}
    	
    	}
   if(count1-count==-5)
   {cout<<-3<<endl;}
   else
   {cout<<count1-count<<endl;}
    }
    return 0;
}
