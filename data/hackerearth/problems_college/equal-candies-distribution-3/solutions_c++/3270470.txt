#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int t,n,i,j;
	cin>>t;
	while(t--)
	{
		cin>>n;
		int a[n+2];
		for(i=0;i<n;i++)
		cin>>a[i];
		sort(a,a+n);
		int f=0,d,c=0,s1,s2,s3;
		for(i=0;i<n-1;i++)
		{
			s1=0;
			s2=0;
			s3=0;
			d=a[i+1]-a[i];
             f=0;
            	if(d/5 > 0 )
					s1=d/5,c=c+(d/5),d=d%5;
				if(d/2 > 0 )
				s2=d/2,c=c+(d/2),d=d%2;
				if(d/1 > 0)
				s3=d,c=c+(d/1),d=d%1;
					for(j=i+2;j<n;j++)
            			a[j]+= 5*s1+2*s2+s3;
		}
			cout<<c<<endl;
	}
	return 0;
}
