#include<bits/stdc++.h>
 
using namespace std;
 
int main(){
 
    int t;
    long long int r;
    int k;
    cin>>t;
    //cout<<t;
    while(t--){
 
        cin>>r;
        cin>>k;
        //cout<<r<<k;
        long long int no=0;
        for(int i=1;i<=r;i++){
            int x=i,counter=0;
            while(1){
            if(x%8==5){
            counter++;
            }
            x=x>>1;
            if(x==0) break;
            }
            if(counter>=k) { no++;}
        }
        cout<<no<<"\n";
    }
 
 
 
 
 
 
 
    return 0;
}
