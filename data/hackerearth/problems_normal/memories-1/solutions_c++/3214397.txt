#include<bits/stdc++.h>
using namespace std;
int main()
{
int t,n,p,a,b,size,fin;
cin>>t;
while(t--)
{
    fin=0;
    cin>>n>>p;
    size=pow(2,n);
    size--;
    bool ans[32769]={false};
    while(p--)
    {
        cin>>a>>b;
        for(int i=0;i<=size;i++)
        {
            if(ans[i]==false)
            {
            std::bitset<15> foo(i);
            if(foo[a-1]==1 and foo[b-1]==1)
                {
                //cout<<"here";
                ans[i]=true;
                }
            }                          
        }
        /*for(int i=1;i<=size;i++)
        {
            cout<<ans[i];
        }
        cout<<endl;*/
    }
    
   
    for(int i=1;i<=size;i++)
        {
            if(ans[i]==false)
            {
                fin++;
            }
        }
        cout<<fin<<endl;
}
return 0;
}
