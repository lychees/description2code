N = input()
S = [int(x) for x in raw_input().strip().split()]
F = [int(x) for x in raw_input().strip().split()]
T = input()


def truncate(f, n):
    s = '{}'.format(f)
    if 'e' in s or 'E' in s:
        return '{0:.{1}f}'.format(f, n)
    i, p, d = s.partition('.')
    return '.'.join([i, (d+'0'*n)[:n]])


economy = []
for i in range(N):
    economy.append(S[i] / float(F[i]))

economy.sort(reverse=True)

print truncate(economy[0] * T, 3)
