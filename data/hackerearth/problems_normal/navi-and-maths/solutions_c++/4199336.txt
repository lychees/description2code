#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cmath>
#include<cstring>
#include<vector>
#include<cstdlib>
#define ll long long
#define ull unsigned long long
#define e 1000000007
#define rep(i,a,b) for(i=a;i<b;i++)
#define sll(a) scanf("%I64d",&a)
#define si(a) scanf("%d",&a)
#define pll(a) printf("%I64d\n",a)
#define pi(a) printf("%d\n",a)
using namespace std;
int t,n,i,j,x;
ll a[50],b[50],prd,sum,maxi,res,z,xx,ii;

ll pow(ll a, ll b, ll MOD) {
ll x = 1, y = a;
    while(b > 0) {
        if(b%2 == 1) {
            x=(x*y);
            if(x>MOD) x%=MOD;
        }
        y = (y*y);
        if(y>MOD) y%=MOD;
        b /= 2;
    }
    return x;
}
 
ll modInverse(ll a, ll m) {
    return pow(a,m-2,m);
}

int main()
{
	cin>>t;
	x=0;
	while(t--)
	{
		cin>>n;
		rep(i,0,n)
		cin>>a[i];
		maxi=0;
		z=pow(2,n)-1;
		while(z)
		{
			prd=1;
			sum=0;
			xx=z;
			ii=0;
			while(xx)
			{
				if(xx%2==1)
				{
					prd*=a[ii];
					prd%=e;
					sum+=a[ii];
				}
				ii++;
				xx/=2;
			}
			prd%=e;
			res=prd*modInverse(sum,e)%e;
			res%=e;
			//cout<<res<<endl;
			if(res>maxi)
			maxi=res;
			--z;
		}
		printf("Case #%d: %lld\n",++x,maxi);
	}
	return 0;
}
