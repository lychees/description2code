long long int power(long long int n, long long int p)
{
    long long int r = 1;
    for(long long int i = p;i>=1;)
    {
        if(i%2==0)
        {
            n = (n*n)%MOD;
            i = i>>1;

        }
        else
        {
            r = (r*n)%MOD;
            i--;
        }
        //cout<<"value of r"<<r<<endl;
    }
    return r;
}

void sieveforprimes()
{
    for(int i = 0;i*i<=SIZEP;i++)
        prime[i]=1;
    prime[0] = 0;
    prime[1] = 1;
    for(int i = 2;i<SIZEP;i++)
    {
        if(prime[i])
        {
            for(int j = 2;i*j<SIZEP;j++)
            {
                prime[i*j] = 0;
            }
        }
    }
}

void chanduinterns()
{
    sieveforprimes();
    long long int n,x;
    cin>>n;
    while(n--)
    {
        cin>>x;
        if(prime[x])
            cout<<"NO"<<endl;
        else
        {
            int sq = sqrt(x);
            //cout<<sq<<endl;
            if(prime[sq] && sq*sq==x)
                cout<<"NO"<<endl;
            else
                cout<<"YES"<<endl;

        }
    }

}

long long int gcd(long long int a, long long int b)
{
    if(b==0)
        return a;
    else
        return gcd(b,a%b);
}
