#include <iostream>
using namespace std;
int mod(int a)
{
	if(a<0) return 0-a;
	else return a;
}
int main()
{
	int n,max=0,i,k=0;
   cin>>n;int A[n];
   for(i=0;i<n;i++) cin>>A[i];
   for(i=1;i<n;i++)
   {
   	if(mod(A[i]-A[i-1])==1) {
   		while(mod(A[i]-A[i-1])==1) {
   			k++;i++;
   		}
   		if(k!=0) {if(max<k) max=k;k=0;}
   	}
   }
   cout<<max+1;
    return 0;
}
