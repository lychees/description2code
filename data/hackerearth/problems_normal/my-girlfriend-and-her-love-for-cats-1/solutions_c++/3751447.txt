#include <cstdio>
#include <algorithm>
#include <functional>

using namespace std;
using l = unsigned long;

int main() {
  l n;
  scanf("%ld",&n);

  l cats[n], pckts[n];

  for (l i = 0; i < n; ++i)
    scanf("%ld",&cats[i]);
  for (l i = 0; i < n; ++i)
    scanf("%ld",&pckts[i]);

  stable_sort(cats,cats+n,greater<int>());
  stable_sort(pckts,pckts+n,greater<int>());

  unsigned long long sum = 0;

  for (l i = 0; i < n; ++i) 
    sum += (cats[i]*pckts[i]);
  
  printf("%lld\n",sum);
  return 0;
}
