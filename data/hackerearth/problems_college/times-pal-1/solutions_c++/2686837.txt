#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstdlib>
#include<vector>
#include<cstring>
#include<string.h>
using namespace std;
 
int main()
{
    int t,c,i,length;
    string s3,s4;
    cin>>t;
    while(t--)
    {
    	c=0;
    	cin>>s3>>s4;
    	length=s3.length();
    	for(i=0;i<length;i++)
    	{
    		string s=s3.substr(0,i) + s4 + s3.substr(i,length);
    		string s5=s;
    		reverse(s.begin(),s.end());
    		if(s==s5)
    		c++;
    	}
    	string s6=s3+s4;
    	string s4rev=s6;
    	reverse(s6.begin(),s6.end());
    	if(s6==s4rev)
    	c++;
    	
    	cout<<c<<endl;
    }
    return 0;
}
