#include <iostream>
using namespace std;

bool ispal(string s,int start,int l){
	int t=start;
	int i=l-1;
	while(t<=i){
		if(s[t]!=s[i]){
			//cout<<s[i];
			return false;
		}
		t++;
		i--;
	}
	//cout<<endl;
	return true;
}

int main()
{
   // cout << "Hello World!" << endl;
   int t;
   cin>>t;
   while(t--){
   	string s;
   	cin>>s;
   	int l = s.length();
   	int max=0;
   	for(int i=0;i<l;i++){
   		if(ispal(s,i,l)){
   			max = l-i;
   			break;
   		}
   	}
   	//cout<<max<<endl;
   	cout<<l+(l-max)<<endl;
   }
    return 0;
}
