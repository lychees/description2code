#include <iostream>

using namespace std;
long long dp[1005][1005];
void solve()
{
    for(int i=1;i<1005;i++)
    {
        for(int j=1;j<1005;j++)
        {
            if(i==1||j==1)
                dp[i][j]=1;
            else
                dp[i][j]=((dp[i-1][j-1])%1000000007+(dp[i-1][j])%1000000007+(dp[i][j-1])%1000000007)%1000000007;
        }
    }

}
int main()
{
  int t;
  cin>>t;
  solve();
  while(t--)
  {
      int a,b,c,d;
      cin>>a>>b>>c>>d;
      int x=a-1;
      int y=b-1;
      int m=c-x;
      int n=d-y;
      cout<<dp[m][n]<<endl;
  }
}
