#include <iostream>
using namespace std;
typedef unsigned long long int ll;
ll max(ll a,ll b)
{
 if(b-a==1)
  return (b&a);
 else if(!(b&1))
  return ((b-1)&(b-2));
 else
  return (b&(b-1));
  
}
 
int main()
{
    ll t;
    cin>>t;
    ll a,b;
    while(t--)
    {
     cin>>a>>b;
     cout<<max(a,b)<<endl;
    }
    return 0;
}
