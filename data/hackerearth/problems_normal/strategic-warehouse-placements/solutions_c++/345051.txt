int n,m;
vector<vector<int> > adj(1010);
vector<int> v(1010);
vector<int> color(1010);//0-->white[no warehouse],1--> black [warehouse]
int dfs(int i){
	v[i]=1;
	int ret=0;
	int need=0;
	FOR(j,adj[i].size()){
		if(!v[adj[i][j]]){
			ret+=dfs(adj[i][j]);
			if(color[adj[i][j]]==0)need++;
		}
	}
	if(need==0)color[i]=0;//white color 
	else color[i]=1; //black color //i.e. warehouse here
	return ret+color[i];
}

int main()
{
    cin>>n>>m;
    while(m--){
    	int x,y;
    	cin>>x>>y;
    	x--;y--;
    	if(x!=y){
	    	adj[x].pb(y);
	    	adj[y].pb(x);
    	}
    }
    
    int ans=0;
    FOR(i,n)v[i]=0,color[i]=0;
    
    FOR(i,n){
    	if(v[i]==0)
	    	ans+=dfs(i);
    }
    
    if(ans==0)cout<<3;
    else cout<<ans;
    return 0;
}
