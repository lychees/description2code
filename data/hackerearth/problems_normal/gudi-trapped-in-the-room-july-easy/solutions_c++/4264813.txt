#include<bits/stdc++.h>
using namespace std;
map<string,bool> m; 
void rec(string str,int a,int h)
{
	if(m[str])
	{
		return ;
	}
	//cout<<str<<endl;
	m[str]=1;
	string str1=str;
	int i;
	for(i=0;i<str1.length();i++)
	{
		if(i%2!=0)
		{
			int d=str1[i]-'0';
			d=(d+a)%10;
			str1[i]=d+'0';
		}
	}
	rec(str1,a,h);
	char temp;
	string str2;
	str2=str;
	for(i=0;i<str.length();i++)
	{
		str2[(i+h)%(str.length())]=str[i];
	}
	rec(str2,a,h);
}
int main()
{
