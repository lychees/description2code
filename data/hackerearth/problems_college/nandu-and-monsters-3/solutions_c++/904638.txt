#include <iostream>
#include <cmath>
using namespace std;

#define MAX 1000000

int N, Q;
char m[MAX];

bool check(int i, int j)
{
	if (m[i] == '1' || m[j] == '1') return false;
	if (i < j) return check(i, j >> 1);
	else if (i > j) return check(i >> 1, j);
	else return (m[i] == '0');
}

int main() {

	cin >> N >> Q;
	cin >> m + 1;

	int i, j;
	while (Q--) {
		cin >> i >> j;
		if (check(i, j)) cout << "Yes" << endl;
		else cout << "No" << endl;
	}
	return 0;
}
