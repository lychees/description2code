#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
   	long long int n,q,a,b,i;
   	cin>>n>>q;
   	long long int s[n],sum=0;
   	for(i=0;i<n;i++)
   	{
   		cin>>a>>b;
   		s[i]=abs(a-b);
   		sum+=min(a,b);
   	}
   	sort(s,s+n);
   	for(i=0;i<n;i=i+2)
   	{
   		if(s[i]!=s[i+1])
   		{
   			sum=-1;
   			break;
   		}
   		sum=sum+s[i];
   	}
   	//sum=abs(a-b)+sum;
   	while(q--)
   	{
   		cin>>a;
   		if(a==sum)
   		cout<<"YES\n";
   		else
   		cout<<"NO\n";
   	}
    return 0;
}
