#include<iostream>

using namespace std;

long maximumXOR(unsigned long long L, unsigned long long R) {
    unsigned long long W = L ^ R, m = 1;
    while(W){
        W =W/2;
        m = m<<1;
    }
	m=m-1;
    return m;
}

int main()
{
	long t;
	
	cin>>t;
	
	for(long i = 0;i < t; i++)
	{
		unsigned long long n;
		cin>>n;
		
		if(n>1)
		{
			unsigned long long ans = maximumXOR(0, n-1);
			cout<<ans<<" "<< (n-(ans+1)/2)*2<<endl;
		}
		else{
			cout<<"0 1"<<endl;
		}
	}	
	return 0;
}
