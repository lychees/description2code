#include <iostream>
using namespace std;
int check(int *a,int n)
{
	int s=0;
	for(int i=0;i<n;i++)
	if(a[i]>0)
		s++;
	return s;
}

int main()
{
    int n,i,s=5;
    cin>>n;
    int a[n];
    for(i=0;i<n;i++)
    	cin>>a[i];
    s=check(a,n);
   	do
   	{
   		cout<<s<<endl;
   		for(i=0;i<n;i++)
   			a[i]-=2;
   		s = check(a,n);
   		
   	}while(s>0);
}
