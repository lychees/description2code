#include <bits/stdc++.h>
using namespace std;
int fr[27];
void init(int fr[]){
	for(int i=0;i<26;i++)
	fr[i]=0;
}

int main()
{int coun=0;
int ans =0;
    //cout << "Hello World!" << endl;
    int k;
   char s1[5001];
    cin>>k;
    cin>>s1;
    for(int i=0;i<strlen(s1);i++)
    {init(fr);
    	for(int j=i;j<strlen(s1);j++)
    	{
    	fr[s1[j]-'a']++;
    	if(fr[s1[j]-'a']==1)
    	coun++;
		//cout<<count;
	
    		//cout<<coun;
    			if(coun==k)
    			ans++;
    		
    		
    		
    	}
    			
    			coun=0;
    }
   cout<<ans;
    
    
    
    return 0;
}
