#include <iostream>
using namespace std;
int main()
{
    int t;
    char c;
    string str;
    cin>>t;
    getline(cin,str);// to read \n from cin
    while(t--)
    {
    	getline(cin,str);
    	c=str[0];

    	 cout<<"EVEN"<<endl;
    	else
    	 cout<<"ODD"<<endl;
    }
    return 0;
}
