#include <bits/stdc++.h>
using namespace std;
bool order(const pair<int,int> &a,const pair<int,int> &b){
	return a.second<b.second;
}
int main(){
	int n;
	cin >> n;
	vector<int> b(n+1,0);
	for(int i=1;i<=n;i++)
		cin >> b[i];
	for(int i=1;i<=n;i++)
		b[i] += b[i-1];
	vector<vector<pair<int,int> > > a(100001);
	for(int i=1;i<=n;i++){
		for(int j=i;j<=n;j++){
			int sum = b[j]-b[i-1];
			a[sum].push_back(make_pair(i,j));
		}
	}
	int *ma=new int[100001];
	int m=0;
	for(int i=1;i<=100000;i++){
		int st=-1,en=-1;
		sort(a[i].begin(),a[i].end(),order);
		int temp=0;
		for(int j=0;j<a[i].size();j++){
			if(a[i][j].first>en){
				st=a[i][j].first;
				en=a[i][j].second;
				++temp;
			}
		}
		ma[i]=temp;
	}
	for(int i=1;i<=100000;i++)
		m = max(m,ma[i]);
	cout << m << endl;
	for(int i=1;i<=100000;i++){
		if(ma[i]==m)
			cout << i << " ";
	}
	cout << endl;
}
