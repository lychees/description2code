#include <iostream>
#include <cmath>
using namespace std;

typedef long long int ll;

ll fn(ll w,ll x, ll y, ll z,ll p){
	return w*p*p*p + x*p*p + y*p + z;
}

ll vn(ll w,ll x, ll y, ll z,ll p){
	return w*p*p*p + x*p*p + y*p + z;
}

int main()
{
	int T;cin>>T;
	while(T--){
		ll a,b,c,d,k,limit,mid,f,t,u,v,low;
		cin>>a>>b>>c>>d>>k;
		bool no_zero = false;
		limit=(ll)ceil(cbrt(k/a));
		//limit=ceil(limit/a);
		low = 1;
		mid=(limit + low)/2;
		u=mid+1;
		//cout<<k<<endl;
		//cout<<limit<<endl;
	//	if(fn(a,b,c,d,0)<=k && vn(a,b,c,d,1)>k){
	//		no_zero=false;
	//		mid=0;
	//	}
		
		while(low<limit){
			if(fn(a,b,c,d,mid)<=k && vn(a,b,c,d,u)>k){
			//	no_zero=true;
				break;
			}
			if(fn(a,b,c,d,mid)>k){
				limit=mid-1;
				mid=(limit+low)/2;
				u=mid+1;
			}else{
				low=mid+1;
				mid=(limit+low)/2;
				u=mid+1;
			}    
		}
	//	if(no_zero){
