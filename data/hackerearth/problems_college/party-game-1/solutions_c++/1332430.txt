#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
   int t,n,i;
   long int a[10000];
   long long int x,y,sum,m;
   scanf("%d",&t);
   while(t--)
   {
       scanf("%lld %lld %d",&x,&y,&n);
       for(i=0;i<n;i++)
       {
           scanf("%ld",&a[i]);
       }
       sort(a,a+n);sum=0;i=0;
       while(sum<=x)
       {
           sum+=a[i];i+=1;
       }
       m=i;sum=0;i=0;
       while(sum<=y)
       {
           sum+=a[i];
           i+=1;
       }
       if(i>m)
        printf("Jhool Won\n");
       else
        if(i<m)
        printf("Chef Won\n");
       else
        printf("Tie\n");

   }
   return 0;
}
