import itertools

def nCr(listOfPrimes):
	resultSet = set()
	for L in range(0, len(listOfPrimes)+1):
		for subset in itertools.combinations(listOfPrimes, L):
			if len(subset) !=0 :
				resultSet.add(subset)
	return len(resultSet)

def get_primes(num):
	i = 2
	results = []
	while i * i <= num:
		if num%i == 0:
			results.append(i)
			num = num/i
		else:
			i = i + 1
	results.append(num)
	return results

		print nCr(get_primes(num))
