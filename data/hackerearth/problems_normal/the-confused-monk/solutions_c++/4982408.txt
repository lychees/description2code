unsigned long long int mod(unsigned long long int a, unsigned long long int b, unsigned long long int c)
{
	unsigned long long int res = 1;
	a = a % c;
	while(b != 0) {
				if(b % 2 != 0) {
						res = (res * a) % c;
				}
				b = b / 2;
				a = (a * a) % c;
	}
	return res;
}
unsigned long long int gcd1(unsigned long long int a, unsigned long long int b) {
		if(b == 0) {
				return a;
		}else {
				gcd1(b, a % b);
		}
		
		
		
		
}
