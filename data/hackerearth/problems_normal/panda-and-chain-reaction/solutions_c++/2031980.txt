#include <bits/stdc++.h>
using namespace std;
#define ll long long int
#define MOD 1000003
int main()
{
    int t;
    scanf("%d",&t);
    ll res[1000001],i;
     res[0]=1;
    for(i=1;i<1000001;i++)
    {
    	res[i]=((i%MOD)*(res[i-1]%MOD))%MOD;
    }
    while(t--)
    {
    	ll n,x;	
   		cin>>n>>x;
		x=x%MOD;
		if(n>MOD)
		n=MOD;
		cout<<(res[n]*x)%MOD<<endl;
	}
}
