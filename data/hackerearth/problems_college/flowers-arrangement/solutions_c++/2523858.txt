#include <iostream>
using namespace std;

int main()
{
    int t,i;
    cin>>t;
    while(t--){
    	int flag=1;
    	string str;
    	cin>>str;
    	for(i=0;i<str.size();i++){
    		if(str[i]!='R' && str[i]!='Y'){
    			flag=0;
    			break;
    		}
    	}
    	
    	if(flag){
    		if(str[0]!='R'){
    			flag=0;
    		}else{
    			for(i=1;i<str.size()-1;i++){
    				if(str[i]=='Y' && str[i-1]=='Y' && str[i+1]=='Y'){
    					flag=0;
    					break;
    				}
    			}
    		}
    	}
    	
    	if(flag){
    		cout<<"YES"<<endl;
    	}else{
    		cout<<"NO"<<endl;
    	}
    }
    return 0;
}
