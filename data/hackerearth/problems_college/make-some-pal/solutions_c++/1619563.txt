#include <iostream>
using namespace std;

int main()
{
	int t;
	cin >> t;
	
	while (t--) {
		string str;
		cin >> str;
		
		int l = str.size();
		
		for (int i=0; i<l; i++) {
			bool done = true;
			
			int k = i;
			int j = l-1;
			while (j>k) {
				if (str[j]!=str[k]) {
					done = false;
				}
				--j;
				++k;
			}
			if (done) {
				cout << l + i << endl;
				break;
			}
		}
		
	}
	
	
    return 0;
}
