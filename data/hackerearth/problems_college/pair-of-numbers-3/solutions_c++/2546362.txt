#include <iostream>
using namespace std;

int main()
{
    int t,n,i,j,k,c;
    cin>>t;
    for(i=0;i<t;i++)
    {
    	c=0;
    	cin>>n;
    	for(j=1;j<n;j++)
    	{
    		for(k=j+1;k<=n;k++)
    		{
    			if(((j*k)%(j+k))==0)
    			{
    				c++;
    			}
    		}
    	}
    	cout<<c<<"\n";
    	
    }
    return 0;
}
