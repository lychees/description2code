t=input()
while t:
	s=raw_input()
	st = list(s)
	d = {}
	for i in st:
		if d.has_key(i):
			d[i] += 1
		else:
			d[i]=1
	p,q,r,s=0,0,0,0
	if(d.has_key('r')):
		p=d['r']
	if(d.has_key('u')):
		q=d['u']
	if(d.has_key('b')):
		r=d['b']
	if(d.has_key('y')):
		s=d['y']
	ans = min(min(min(p,q),r),s)
	print ans
	t=t-1
