import re

t = int(raw_input())
while(t > 0):
	line = raw_input()
	lst = line.split('$')
	number = re.findall('[0-9 ]+',lst[1])
	new_lst = number[0].split()
	output = ''
	for nos in new_lst:
		output = output + nos
	data = int(output)
	print('$' + str(data))
	t = t-1
