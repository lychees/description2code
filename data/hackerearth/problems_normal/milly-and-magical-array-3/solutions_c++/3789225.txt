#include <cstdlib>
#include <cstdio>
typedef long long int lli;

using namespace std;

/*
 * 
 */
int main(int argc, char** argv) {
    int t;
    lli n, x, i, count = 0,value;
    lli arr[100002];
    scanf("%d", &t);
    for (; t > 0; t--) {
        count = 0;
        scanf("%lld%lld", &n, &x);
        for (i = 0; i < n; i++) {
            scanf("%lld", &arr[i]);
        }
        for (i = 0; i < n - 1; i++) {
            if (arr[i] >= arr[i + 1]) {
                value = (arr[i] - arr[i + 1]) / x + 1;
                arr[i+1] = arr[i+1] + value*x;
                count = count + value;
            }
        }
        printf("%lld\n",count);
    }
    return 0;
}
