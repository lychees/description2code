#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int test;
    cin>>test;
    
    while(test--)
    {
    	int len;
    	//long long powlen;
    	cin>>len;
    	int array[len];
    	long long temp=0;
    	
    	for(int i=0; i<len; i++)
    	{
    		array[i]=i+1;
    	}
    	
    	//cin>>sum;
    	
    	long long blah=pow(2, (len-1));
    	
    	for(int i=0; i<len; i++)
    	{
    		temp=(temp%1000000007)+((blah*array[i])%1000000007);
    	}
    
    cout<<(temp%1000000007)<<"\n";
    
}
return 0;
}
