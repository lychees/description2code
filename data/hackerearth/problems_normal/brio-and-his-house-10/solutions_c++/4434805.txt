#include<bits/stdc++.h>
using namespace std;
int main()
{
    ios::sync_with_stdio(false);
    cout<<fixed;
    int t;
    long double a,b,c,ans,R;
    cin>>t;
    while(t--)
    {
        cin>>a>>b>>c;
        ans=(a+b-c)*(a+c-b)*(b+c-a)*(a+b+c);
        ans=sqrt(ans);
        R=a*b*c;
        R/=ans;
        R*=R;
        R*=3.1415;
        cout<<setprecision(4)<<R<<"\n";
    }
    return 0;
}
