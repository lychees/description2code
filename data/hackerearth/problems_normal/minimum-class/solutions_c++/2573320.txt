#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        long n;
        scanf("%ld",&n);
        long a,b,classes;
        a=sqrt(n);
        classes=(a-1)*2;
        b=ceil(sqrt(n));
       // cout<<a*a+a<<endl;
        if(b==a)
            classes+=1;
        else
        {
            if(n<=(a*a+a))
                classes+=2;
            else
                classes+=3;

        }
        cout<<classes<<endl;
    }
    return 0;
}
