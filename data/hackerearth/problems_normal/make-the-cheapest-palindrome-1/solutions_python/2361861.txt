for T in range(input()):
	string = raw_input().rstrip()
	aC = input()
	bC = input()
	c = min(aC, bC)
	ch = 'a'
	if c == bC:
		ch == 'b'
	cost = 0
	n = len(string)
	new = ''
	for i in range(n):
		if string[i] == '/':
			if string[n-i-1] == 'a':
				cost += aC
				new += 'a'
			elif string[n-i-1] == 'b':
				cost += bC
				new += 'b'
			else:
				cost += c
				new += ch
		else:
			new += string[i]
	if new == new[::-1]:
		print cost
	else:
		print -1
