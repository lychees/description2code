#include<bits/stdc++.h>
#define MOD 1000000007
using namespace std;
int main()
{
	long long int T,a,b,c,i,j,ans;
	vector<long long int> v;
	cin>>T;
	while(T--)
	{
		cin>>a>>b>>c;
		v.push_back(a);
		v.push_back(b);
		v.push_back(c);
		sort(v.begin(),v.end());
		ans=1;
		ans=(ans*v[0])%MOD;
		ans=(ans*(v[1]-1))%MOD;
		ans=(ans*(v[2]-2))%MOD;
		cout<<ans<<"\n";
		v.clear();
	}
}
