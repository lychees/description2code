#include <iostream>
#include<algorithm>
#include<string>
using namespace std;

int main()
{
    int test_cases;
    cin>>test_cases;
    string s,res;
    while(test_cases--)
    {
    	cin>>s;
    	res.clear();
    	for(string::size_type i=0;i<s.size();i++)
    	{
    		if(count(s.begin(),s.begin()+i,s[i])==0)
          res.push_back(s[i]);
    	}
    	cout<<res<<"\n";
    }
    return 0;
}
