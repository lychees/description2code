#include<iostream>
#include<stdio.h>
using namespace std;

long long int sum[100000][10]={0};
long long int A[100000];

int main()
{
    int N,Q;
    scanf("%d %d",&N,&Q);
    
    for(int i=0;i<N;i++)
    scanf("%lld",&A[i]);


    for(int i=0;i<=9;i++)
      for(int j=0;j<10;j++)
        for(int k=i;k<N;k=k+j+1)
              sum[i][j]+=A[k];
              
    for(int i=10;i<N;i++)
       for(int j=0;j<10;j++)
            sum[i][j]=sum[i-(j+1)][j]-A[i-(j+1)];

	while(Q--)
    {   long long int l,r,k;
        scanf("%lld%lld%lld",&l,&r,&k);
        l--;
        r--;
        long long int it=((r-l)/k)+1;
        r=l+(it)*k;
        long long int ans=(r<N)?sum[l][k-1]-sum[r][k-1]:sum[l][k-1];

        printf("%lld\n",ans);
     }

}
