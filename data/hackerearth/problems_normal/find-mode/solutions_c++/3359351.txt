#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
	int T;
	cin>>T;
	while(T--)
	{
		int N;
		cin>>N;
		int array[1000000];
		memset(array, 0 , sizeof(array));
		int a;
		for(int i = 0;i<N ; i++)
		{
			cin>>a;
			array[a]++;
		}
		int max = -1 ;

		for(int i = 0 ; i < 1000000 ; i++)
		{
			if(array[i]>max)
			max = array[i];
		}
		
		for(int i = 999999 ; i >= 0  ;i--)
		{
			if(array[i] == max)
			cout<<i<<" ";
			
		}
		cout<<endl;
		
	}
    return 0;
}
