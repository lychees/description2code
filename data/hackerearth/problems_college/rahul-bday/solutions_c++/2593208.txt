#include <iostream>
using namespace std;
#include<limits.h>

long long int min(long long int t1,long long int t2,long long int t3)
{
	if(t1<t2)
	{
		return t1<t3?t1:t3;
	}
	else return t2<t3?t2:t3;
}
int main()
{
    int t; cin>>t;
	while(t--)
	{
		long long int b,w,x,y,z,tot1,tot2,tot3,j,k;
		cin>>b>>w>>x>>y>>z;
		tot1=(b*x)+(w*y);
			tot2=(b*x);
			j=(w*(z+x));
			tot2+=j;
		
			tot3=(w*y);
			k=(b*(z+y));
			tot3+=k;
		cout<<min(tot1,tot2,tot3)<<endl;
	}
    return 0;
}
