#include <iostream>

using namespace std;

int main() {
        int T;
        cin >> T;
        while(T--) {
                long long int OC = 1, ZC = 0;
                int Touches = 0;
                cin >> Touches;
                while(Touches--) {
                        long long int oldoc = OC;
                        OC = ZC;
                        ZC = ZC+oldoc;
                }
                cout << OC << " " << ZC << endl;
        }
        return 0;
}
