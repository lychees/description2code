#include <bits/stdc++.h>

using namespace std;

typedef long long int ll;
ll a[100005];
int main()
{
	int n;
	ll k;
	scanf("%d %lld", &n,&k);
	
	for(int i = 0; i < n; i++) {
		scanf("%lld",&a[i]);
	}
	if(n <= k) {
		for(int i = n; i >= 1; i--) {
			cout << i << " "; 
		}
		cout << endl;
		return 0;
	}
	int key,swa = 0;
	while(k) {
		for(int i = swa; i < n-1; i++) {
			ll data = a[i];
			key = -1;
			for(int j = n-1; j > i; j--) {
				if(a[i] < a[j]) {
					key = j;
					a[i] = a[j];
				}
			}
			if(key != -1) {
				a[key] = data;
				k--;
				break;
			}
		}
		if(key == -1) {	
			break;
		}
		swa++;
	}
	for(int i = 0; i < n; i++) {
		printf("%lld ", a[i]);
	}
	cout << endl;
	return 0;
}
