#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	scanf("%d", &t);

	while (t--) {
		int n;
		scanf("%d", &n);

		vector<string> v(n);

		for (int i = 0; i < n; i++) {
			cin >> v[i];
		}

		int mxcnt = INT_MIN, cnt = 0;
		char ans = 'a';

		for (char ch = 'a'; ch <= 'z'; ch++) {
			cnt = 0;
			for (int i = 0; i < n; i++) {
				if (v[i].find(ch) != string::npos) {
					cnt++;
				}
			}

			if (cnt > mxcnt) {
				mxcnt = cnt;
				ans = ch;
			}
		}

		printf("%c\n", ans);
	}
}
