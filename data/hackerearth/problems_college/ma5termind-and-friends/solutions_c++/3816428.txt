#include<bits/stdc++.h>
typedef long long ll;
using namespace std;
int x,y,q,n,m,t;
ll c;
 
vector< pair< pair<int ,int> ,ll> >f(100009);

int main()
{
    ios::sync_with_stdio(false);
    cin>>t;
    while(t--)
    {
        cin>>n>>m;
        vector<ll>v(n+10);
        vector<int>qu(m+10);
        
        for(int i=0;i<m;i++)
        {
            cin>>f[i].first.first>>f[i].first.second>>f[i].second;

        }
        cin>>q;
         for(int i=0;i<q;i++)
         {
             cin>>x>>y;
             qu[x-1]++;
             qu[y]--;

         }
         for(int i=1;i<m;i++)
            qu[i]+=qu[i-1];
         
         for(int i=0;i<m;i++)
         {
             v[f[i].first.first-1]+=qu[i]*f[i].second;
             v[f[i].first.second]-=qu[i]*f[i].second;

         }
         for(int i=1;i<n;i++)
            v[i]+=v[i-1];
         for(int i=0;i<n;i++)
            cout<<v[i]<<" ";
         cout<<"\n";
         //v.clear();
         //qu.clear();
		//f.clear();
    }





}
