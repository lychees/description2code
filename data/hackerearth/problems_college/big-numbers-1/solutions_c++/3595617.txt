#include <bits/stdc++.h>
using namespace std;
#define MOD		1000000007
#define LL long long

long long  dp[51][100][2001];
int mod , n , k;

LL solve(int a, int  b , int  c)
{
	if(c > k) return 0;
	if(a == n) return (b == 0 && k == c);
	else if(dp[a][b][c] != -1) return dp[a][b][c];
	LL ret = 0;
	for(int i = 0; i < 10; i++)
		ret = (ret + solve(a + 1 , (b * 10 + i) % mod , c + i*i )) % MOD;
	return (dp[a][b][c] = ret);
}

LL calc()
{
	for(int i=0;i<51;i++)
	{
		for(int j=0;j<100;j++)
		{
			for(int z =0;z<2001;z++)
			{
				dp[i][j][z]=-1;
			}
		}
	}
	LL ret = 0;
	for(int i = 1; i < 10; i++)
		ret = (ret + solve(1 , i%mod , i*i) ) % MOD;
	return ret;
}

int main()
{
	cin >> n >> mod >> k;
	cout << calc() <<endl;;
	return 0;
}
