#include <iostream>
using namespace std;

long int gcd(long int a, long int b)
{
    if (b)
        return gcd(b, a % b);
    else
        return a;
}


int main()
{
   	long int T,a,b,c;
   	cin>>T;
   	while(T--)
   	{
   		cin>>a>>b>>c;
   		if(gcd(a,b)==1&&gcd(b,c)==1&&gcd(c,a)==1)
   		{
   			if((a*a+b*b==c*c)||(a*a+c*c==b*b)||(b*b+c*c==a*a))
   				cout<<"YES\n";
   			else
   				cout<<"NO\n";
   		}
   		else
   			cout<<"NO\n";
   	}
    return 0;
}
