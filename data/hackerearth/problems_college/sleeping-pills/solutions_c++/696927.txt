#include<iostream>
#define m 1000000007
using namespace std;

int fast_pow(long long base, long long n,long long M) 
{
    if(n==0)
       return 1;
    if(n==1)
    return base;
    long long halfn=fast_pow(base,n/2,M);
    if(n%2==0)
        return ( halfn * halfn ) % M;
    else
        return ( ( ( halfn * halfn ) % M ) * base ) % M;
}
int findMMI_fermat(int n,int M)
{
    return fast_pow(n,M-2,M);
}
int main()
{long long int t,n,k,sum=0,a[100001],r,z,r1,z1;

long long int num,den,id,ans;
ios_base::sync_with_stdio(false);
cin>>t;
while(t--)
{cin>>n>>k;sum=0;
for(int i=0;i<n;i++)
{cin>>a[i];
sum+=a[i];
}
if((k-sum+n-1)<=0)
{cout<<"0\n";
}
else
{z=k-sum+n-2;
r=n-2;
z1=k-sum+n-1;
r1=n-1;
while(r>0)
{z1=(z1*z)%m;
r1=(r1*r)%m;
z--;
r--;
}
id=findMMI_fermat(r1,m);
ans=(z1*id)%m;
cout<<ans<<"\n";
}
}
return 0;
}
