#include<bits/stdc++.h>
#define FOR(i,a,b) for(long long int i=(a);i<=(b);++i)
#define NFOR(i,a,b) for(long long int i=(a);i>=(b);--i)
#define pb push_back
#define ll long long
using namespace std;
ll visited[1001][1001];
 ll n,m,q,x,y,total;
 pair<ll,ll>dmh;
ll ab[1001][1001];
void bfs(ll fg,ll gh)
{
queue<pair<ll,ll>>q;
dmh.first=fg;
dmh.second=gh;
q.push({dmh.first,dmh.second});
pair<ll,ll>r,t;
visited[dmh.first][dmh.second]=1;
--total;
while(!q.empty())
{
	r=q.front();
	q.pop();
	if(r.first+1<n&&ab[r.first+1][r.second]==1&&!visited[r.first+1][r.second])
	{
		t.first=r.first+1;
		t.second=r.second;
		q.push({t.first,t.second});
		ab[t.first][t.second]=0;
		visited[r.first+1][r.second]=1;
		--total;
	}
	if(r.second+1<m&&ab[r.first][r.second+1]==1&&!visited[r.first][r.second+1])
	{
		t.first=r.first;
		t.second=r.second+1;
		q.push({t.first,t.second});
		visited[r.first][r.second+1]=1;
		ab[t.first][t.second]=0;
		--total;
	}
	if((r.first-1)>=0&&ab[r.first-1][r.second]==1&&!visited[r.first-1][r.second])
	{
		t.first=r.first-1;
		t.second=r.second;
		q.push({t.first,t.second});
		visited[r.first-1][r.second]=1;
		ab[t.first][t.second]=0;
		--total;
	}
	if(r.second-1>=0&&ab[r.first][r.second-1]==1&&!visited[r.first][r.second-1])
	{
		t.first=r.first;
		t.second=r.second-1;
		q.push({t.first,t.second});
		visited[r.first][r.second-1]=1;
		ab[t.first][t.second]=0;
		--total;
	}
	
}
}
int main()
{
  
   scanf("%lld%lld%lld",&n,&m,&q);
   total=0;
   FOR(i,0,n-1)
   {
   	FOR(j,0,m-1)
   	{scanf("%lld",&ab[i][j]);if(ab[i][j]==1)++total;}
   	
   }
   
   while(q--)
   {
   	ll c=0;
   pair<ll,ll>ax;
   scanf("%lld%lld",&x,&y);
   x--;y--;
   ax.first=x;
   ax.second=y;
   if(ab[x][y]==1&&!visited[x][y])
   bfs(x,y);
   
  printf("%lld\n",(total));
   }
   
   
    return 0;
}
