#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
	int ceo, coo, cto, n, a[101];
	
	cin >> ceo >> coo >> cto;
	
	cin >> n;
	
	for (int i = 0; i < n; i++)
		cin >> a[i];
	
	sort(a, a+n);
	
	if (cto < coo)
		swap(cto, coo);
	
	int ans = 0;
	
	for (int i = 0; i < n/2; i++)
	{
		ans += abs(a[i]-a[i+1]);
	}
	
	if (coo < ceo)
	{
		ans += abs(coo-a[0]);
		ans += abs(ceo-a[n/2]);
	}
	else
	{
		ans += abs(ceo-a[0]);
		ans += abs(coo-a[n/2]);
	}
	
	
	for (int i = n/2+1; i < n-1; i++)
	{
		ans += abs(a[i]-a[i+1]);
	}
	
	if (cto < ceo)
	{
		ans += abs(cto-a[n-1]);
		ans += abs(ceo-a[n/2+1]);
	}
	else
	{
		ans += abs(ceo-a[n-1]);
		ans += abs(cto-a[n/2+1]);
	}
	
	cout << ans << endl;
	
    return 0;
}
