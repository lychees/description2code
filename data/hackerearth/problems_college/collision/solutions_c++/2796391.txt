#include<cstdio>
int z, n, i, x, wynik, w[11];

int main()
{
  scanf("%d", &z);
  
  while(z--)
  {
    for(i = 0; i <= 9; i++)
    {
      w[i] = 0;
    }
    scanf("%d", &n);
    wynik = 0;
    while(n--)
    {
      scanf("%d", &x);
      x %= 10;
      if (w[x] == 1){wynik++;}
      w[x] = 1;
    }
    printf("%d\n", wynik);
  }  
  
return 0;
}
