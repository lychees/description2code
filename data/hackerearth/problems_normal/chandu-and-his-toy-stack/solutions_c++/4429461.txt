#include <bits/stdc++.h>
using namespace std;

typedef long long int ll;

int main()
{
	ll t,a,b,n,x,y,i;
	multiset<ll> s1,s2;
	
	cin>>t;
	
	while(t--)
	{
		cin>>n>>x>>y;
		
		while(n--)
		{
			cin>>a>>b;
			
			s1.insert(a);
			s2.insert(b);
		}
		
		ll ctr=0;
		multiset<ll>::iterator j=s1.begin();
		multiset<ll>::iterator k=s2.begin();
		
		while(j!=s1.end())
		{
			if(*j<*k)
			ctr+=(x*(*k-*j));
			
			else if(*j>*k)
			ctr+=(y*(*j-*k));
			
			j++;
			k++;
		}
		
		cout<<ctr<<endl;
		s1.clear();
		s2.clear();
	}
	
    return 0;
}
