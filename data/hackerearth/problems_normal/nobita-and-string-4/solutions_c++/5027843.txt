#include <iostream>
#include <sstream>
#include <vector>
using namespace std;

int main() {
    
	vector<string> vec;
	int t;
	string str;
	cin >> t;
	cin.ignore(); 
	while(t--) {
		getline(cin,str);
		
		stringstream ss(str);
		ss << str;
		while(ss >> str) 
			vec.push_back(str);
		
		for(int i=vec.size()-1; i>=0; i--)
			cout << vec[i] << ' ';
		cout << endl;
		
		vec.clear();
	}

    return 0;
}
