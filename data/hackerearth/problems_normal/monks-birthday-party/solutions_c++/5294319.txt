#include <iostream>
#include<bits/stdc++.h>
using namespace std;
set<string>st;
set<string>::iterator it; 

int main()
{
    int t,n;
    string s;
    cin>>t;
    while(t--)
    {	cin>>n;
    	while(n--)
    	{
	    	cin>>s;
	    	st.insert(s);
    	}
    	for(it=st.begin();it!=st.end();it++)
    	{
    		cout<<(*it)<<endl;
    	}
    	st.clear();
    }
}
