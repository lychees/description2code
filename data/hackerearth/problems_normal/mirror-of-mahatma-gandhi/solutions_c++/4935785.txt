using namespace std;
int main()
{
	ll t,i,j,flag,len;
	string s;
	cin>>t;
	while(t--)
	{
		cin>>s;
		len = s.length();
		j=len-1;
		flag=0;
		rep(i,0,len)
		{
			if(s[i]=='0' || s[i]=='1' || s[i]=='8')
			{
				if(s[i]==s[j])
				{
					j--;
					continue;
				}
				else
					flag=1;
			}
			else
				flag =1;
			j--;
		}
		if(flag==0)
			cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;
	}
	return 0;
}
ll max(ll a, ll b)
{
	if(a>b)
		return a;
	else
		return b;
}
ll min(ll a, ll b)
{
	if(a>b)
		return b;
	else
		return a;
}
ll pow(ll a, ll b)
{
	ll ans=1;
	while(b)
	{
		if(b&1)
		{
			ans=ans*a;
		}
		a=a*a;
		b=b>>1;
	}
	return ans;
}
