#include<iostream>
#include <cstdio>
#include<cmath>
typedef long long ll;
using namespace std;

int main()
{
    ll t,n,nn;
    scanf("%lld",&t);
    for(ll i=1;i<=t;i++)
    {
    	scanf("%lld",&n);
        nn=pow(2, floor(log(n)/log(2)));
        nn=n-nn;
        cout<<(nn<<1)+1<<endl;
    }
    return 0;
}
