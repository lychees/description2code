#include <bits/stdc++.h>
using namespace std;
typedef unsigned int uint;


int main()
{	
    uint t;
    cin>>t;
    ++t;
    while(--t)
    {
      string s;
      cin>>s;
      
      uint len=s.size();
      len/=3;
         if((s[0]==s[len])||(s[0]==s[len<<1])||(s[len]==s[len<<1]))
      {
	    cout<<"Not OK"<<endl;
	    continue;
      }
        
      string s1(len,s[0]),s2(len,s[len]),s3(len,s[len<<1]);  
      if((s1+s2+s3)==s)
      {
       cout<<"OK"<<endl;
       continue;
      }
      
    
        
       cout<<"Not OK"<<endl;
    }
    
    return 0;
}
