#include <bits/stdc++.h>

using namespace std;

#define pb push_back
#define mp make_pair
#define ll long long int
#define s(x) scanf("%d", &x)
#define sl(x) scanf("%lld", &x)
#define sd(x) scanf("%lf", &x)
#define mod 1000000007
#define get getchar_unlocked

int isP[10000005];
int ans[10000005];

void sieve()
{
    int i, j;
    for (i = 2; i <= 10000000; ++i) {
        if (!isP[i]) {
            for (j = 1; i*j <= 10000000; ++j)
                isP[i*j] = i;
        }
    }
    for (i = 2; i <= 10000000; ++i) {
        ans[i] = ans[i-1];
        j = i;
        while (j > 1) {
            ++ans[i];
            j /= isP[j];
        }
    }
}

int scan()
{
    int n = 0, ch = get();
    while (ch < '0' || ch > '9')
        ch = get();
    while (ch >= '0' && ch <= '9') {
        n = (n << 3) + (n << 1) + ch - '0';
        ch = get();
    }
    return n;
}

int main()
{
    sieve();
    int t, n;
    t = scan();
    while (t--) {
        n = scan();
        printf("%d\n", ans[n]);
    }
    return 0;
}
