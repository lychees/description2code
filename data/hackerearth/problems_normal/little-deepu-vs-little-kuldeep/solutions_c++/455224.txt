#include<iostream>
#include<cstdio>
typedef unsigned long long ull;
using namespace std;
const ull nX2=2000005;
ull fact[nX2];
ull p=1000000007;
void preCal(){
	fact[0]=1;
	for(int i=1;i<nX2;i++){
		fact[i]=fact[i-1]*i;
		fact[i]%=p;	
	}
}
ull power(ull a,ull k,ull n){
	if(a%n==0)return 0;
	ull rem=a%n;
	ull ans=1;
	while(k){
		if(k%2){
			ans*=rem;
			ans%=n;
		}
		rem=(rem*rem)%n;
		k/=2;
	}
	return ans;
}
ull inverseModP(ull n){
	return power(n,p-2,p);
}
int main(){
	preCal();
	int t;
	scanf("%d",&t);
	while(t--){
		int n;
		scanf("%d",&n);
		ull ans=(fact[2*n]*inverseModP(fact[n]))%p;
		ans=(ans*inverseModP(fact[n+1]))%p;
		printf("%llu\n",ans);
	}
	
}
