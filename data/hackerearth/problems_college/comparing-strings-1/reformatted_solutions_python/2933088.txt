for _ in xrange(int(raw_input())):
    A = raw_input()
    B = raw_input()
    P, Q = [0]*29, [0]*29
    for i in A:
        P[ord(i)-97] += 1
    for i in B:
        Q[ord(i)-97] += 1
    for i in xrange(0, 28):
        if P[i] >= 1 and Q[i] >= 1:
            print "YES"
            break
    if i >= 27:
        print "NO"
