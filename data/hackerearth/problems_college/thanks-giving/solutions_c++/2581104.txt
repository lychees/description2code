#include<bits/stdc++.h>
using namespace std;
int main()
{
    string s;
    cin>>s;
    int a[26];
    for(int i=0;i<26;i++)
        a[i]=0;
    int n=s.length();
    for(int i=0;i<n;i++)
    {
        a[s[i]-'a']++;
    }
    int count=0;
    for(int i=0;i<26;i++)
        count=max(count,a[i]);
    int value=0;
    for(int i=0;i<26;i++)
    {
        if(a[i]==count)
        {
            value=i;
            break;
        }
    }
    string first = "head rush";
    string second= "cubing club";
    int flag=0;
    for(int i=0;i<first.length();i++)
    {
        if(first[i] == (value+'a'))
        {
            flag++;
            break;
        }
    }
    for(int i=0;i<second.length();i++)
    {
        if(second[i] == (value+'a'))
        {
            flag++;
            break;
        }
    }
    if(flag==0)
        cout<<"none";
    else if(flag==2)
        cout<<"Thanks both";
    else
    {
        flag =0;
        for(int i=0;i<first.length();i++)
        {
            if(first[i] == (value+'a'))
            {
                flag++;
                break;
            }
        }
        if(flag==1)
            cout<<"Thanks head rush";
        else
            cout<<"Thanks cubing club";
    }
    return 0;
}
