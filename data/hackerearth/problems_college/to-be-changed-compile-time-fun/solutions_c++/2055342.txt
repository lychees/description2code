#include<cstdio>
#include<vector>
long long int n, x, mx, wynik;
int i;
std::vector<long long int>v;

int main()
{
  scanf("%lld", &n);
  
  for(i = 0; i < n; i++)
  {
    scanf("%lld", &x);
    v.push_back(x);
  }
  
  while(v.size() > 2)
  {
    mx = 0;
    for(i = 1; i + 1 < v.size(); i++)
    {
      mx = std::max(mx, v[i - 1] * v[i + 1]);
    }
    for(i = 1; i + 1 < v.size(); i++)
    {
      if (mx == v[i - 1] * v[i + 1])
      {
	wynik += mx;
	v.erase(v.begin() + i);
	break;
      }
    }
  }
  
  printf("%lld\n", wynik);
  
return 0;
}
