#include<iostream>
#include<stdlib.h>
using namespace std;
int seq(long long int a[],int n)
{
	int *lis,*d,i,j,max=0;
	lis=(int *)malloc(sizeof(int)*n);
	d=(int *)malloc(sizeof(int)*n);
	for(i=0;i<n;i++)
	{
	lis[i]=1;
	if(a[i]>0)
	d[i]=1;
	else
	d[i]=0;
}
	for(i=1;i<n;i++)
	for(j=0;j<i;j++)
	if(abs(a[i])>abs(a[j])&&lis[i]<lis[j]+1&&d[i]!=d[j])
	lis[i]=lis[j]+1;
	
	for(i=0;i<n;i++)
	if(max<lis[i])
	max=lis[i];
	
	return max;
}
int main()
{
	long long int a[100000];
	int i,n;
	cin>>n;
	for(i=0;i<n;i++)
	cin>>a[i];
	cout<<seq(a,n);
	return 0;
}
