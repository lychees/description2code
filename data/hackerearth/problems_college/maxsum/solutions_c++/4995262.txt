#include <bits/stdc++.h>
using namespace std;

typedef long long ll ;

ll dp[100000], n, a[100000] ;

ll solve(ll i){
	
	if(i >= n)
	return 0;
	
	if(i == n-1)
	return dp [i] = a[i] ;
	
	if(dp[i] != -1)
	return dp[i] ;

    return dp[i] = max(solve(i+2)+a[i], solve(i+1));
}

int main()
{
    ll t ;
    cin >> t; 
    while(t--){
    	cin >> n;
    	for( ll i = 0; i < n; i++){
    		cin >> a[i] ;
    	}
    	for(ll i = 0; i < n; i++){
    		dp[i] = -1;
    	}
    	
    	cout << solve(0) << endl;
    }
    return 0;
}
