#include <iostream>
#include<math.h>
using namespace std;

bool prime(int n)
{
	int i;
	for(i=2;i<(int)sqrt(n)+1;i++)
		if(n%i == 0)
			return false;
	return true;
}

int main()
{
    int i;
    bool a[1000002]={false};
    for(i=2;i<1000000;i++)
    	if(!prime(i))
    	{
    		int j = i;
    		while(j)
    		{
    			if(j%10 == 7)
    			{
    				a[i]=true;
    				break;
    			}
    			j /= 10;
    		}
    	}
    int t;
    cin>>t;
    while(t--)
    {
    	int m,n,count=0;
    	cin>>m>>n;
    	for(i=m;i<=n;i++)
    		if(a[i] == true)
    			count++;
    	if(count == 0)
    		cout<<"-1\n";
    	else
    		cout<<count<<endl;
    }
    return 0;
}
