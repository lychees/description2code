#include <iostream>
using namespace std;
#include <string>

bool checkPalindrom(string st, int n){
  bool flag = true;
  for(int i = 0; i < (n/2 +1);i++){
    if(st[i] != st[n - i - 1]){
      flag = false;
      return flag;
      break;
    }
  }
}






int main(){

  int t;
  cin >> t;
  while(t--){
    string st1,st2,st12,st21;
    cin >> st1;
    cin >> st2;
    
    st12 = st12.append(st1);
    st12 = st12.append(st2);

    st21 = st21.append(st2);
    st21 = st21.append(st1);

    int n = st12.length();

    bool flag1,flag2;
    flag1 = checkPalindrom(st12,n);
    flag2 = checkPalindrom(st21,n);

    if(flag1 || flag2) cout << "YES" << endl;
    else cout << "NO" << endl;


  
  }

   return 0;



}
