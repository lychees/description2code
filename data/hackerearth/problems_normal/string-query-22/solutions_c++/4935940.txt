#include <stdio.h>
#include <string.h>
#include <vector>
#include <iostream>
using namespace std;

int main() 
{
	char s[200005];
	int q,i;
	char c;
	vector<int> count[26];
	scanf("%s%d",s,&q);
	size_t length=strlen(s);
	for(i=0;i<length;i++)
	{
		count[s[i]-97].push_back(i);
	}
	while(q--)
	{
		scanf("%d%c",&i,&c);
		scanf("%c",&c);
		s[count[c-97][i-1]]='|';
		count[c-97].erase(count[c-97].begin()+i-1);
	}
	for(i=0;i<length;i++)
	{
		if(s[i]=='|')
			;
		else
			printf("%c",s[i]);
	}
	return 0;
}
