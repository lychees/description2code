#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int x,y,n,w,i;
		double p1,p2,x1,x2,x3;
		cin>>x>>y>>n>>w>>p1>>p2;

		double  p[1005][1005];

		for( i=0;i<=n;i++)
	      	p[i][0]=1;

		for( i=1;i<=w;i++)
	     	p[0][i]=0;
		p1*=.01;
		p2*=.01;
		for( i=1;i<=n;i++)
		{
			for(int j=1;j<=w;j++)
			{
				 x1=p[i-1][max(j-x,0)];
				 x2=p[i-1][max(j,0)];
				 x3=p[i-1][max(j-y,0)];
				p[i][j]=max(p1*x1+(1.0-p1)*x2,p2*x3+(1.0-p2)*x2);
			}
		}
		cout<<setprecision(6)<<fixed<<p[n][w]*100<<endl;
	}
	return 0;

}
