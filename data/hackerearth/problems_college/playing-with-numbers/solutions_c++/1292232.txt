#include <bits/stdc++.h>
#define ll long long int
#define SZ 1234567

using namespace std;

int main()
{
	ios::sync_with_stdio(false);
	
	ll t,flag=0;
	scanf("%llu",&t);
	while(t--)
	{
		ll n;
		scanf("%llu",&n);
		ll ans = __builtin_popcountll(n);
		if(flag==0)
		{
			if(ans%2==0)
			{
				printf("Watson\n");
				flag=1;
			}
			else
			{
				printf("Sherlock\n");
				flag=0;
			}
		}
		else
		{
			if(ans%2!=0)
			{
				printf("Watson\n");
				flag=1;
			}
			else
			{
				printf("Sherlock\n");
				flag=0;
			}
		}
	}
	return 0;
}
