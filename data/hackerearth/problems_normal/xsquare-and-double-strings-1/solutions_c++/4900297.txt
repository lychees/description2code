#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin >> t;
	int ar[26];
	while(t--)
	{
		string s;
		cin >> s;
		for(int i=0;i<26;i++)ar[i]=0;
		for(int i=0;i<s.size();i++)
		{
			ar[s[i]-'a']+=1;
		}
		int flag=0;
		for(int i=0;i<26;i++)if(ar[i]>=2){flag=1;break;}
		if(flag)cout<<"Yes"<<endl;
		else cout<<"No"<<endl;
	}
	return 0;
}
