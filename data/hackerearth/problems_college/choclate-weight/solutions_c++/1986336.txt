#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin>>n;
    int a[100000];
    for(int i=0;i<n;i++)
        cin>>a[i];
    int q;
    cin>>q;int s[100000];
    for(int i=0;i<q;i++)
    {
        cin>>s[i];
    }
    for(int i=0;i<q;i++)
    {int c=0;
        for(int j=0;j<n-1;j++)
        {
            for(int k=j+1;k<n;k++)
            {
                if(s[i]==a[j]||s[i]==a[j]+a[k])c++;
            }
        }
        if(c>0)cout<<"I Like You"<<endl;
        else cout<<"I Hate You"<<endl;
    }
    return 0;

}
