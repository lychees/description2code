using namespace std;

map<string,int>mp;

int checkIf(string x)
{
    for(int i=x.size();i>0;i--)
    {
        string s1=x.substr(0,i);
        //cout<<"??"<<s1<<endl;
        if(mp[s1]) return 1;
    }
    return 0;
}

set<string>s;
string x;
int main()
{

    int t,n,i;
    scanf("%d",&t);
    while(t--)
    {
        mp.clear();
        s.clear();
        scanf("%d",&n);
        int err=0;
        for(i=0;i<n;i++)
        {
            cin>>x;
            s.insert(x);
        }
        set<string>::iterator it;
        for(it=s.begin();it!=s.end();it++)
        {
            //cout<<(*it)<<endl;
            if(!err)
            {
                err=checkIf((*it));
                mp[(*it)]=1;
            }
        }
        if(err) printf("NO\n");
        else printf("YES\n");
    }
    return 0;
}
