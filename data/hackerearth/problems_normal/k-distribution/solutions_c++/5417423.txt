#include <bits/stdc++.h>
using namespace std;

long long sum = 0;
vector<int> candidates;
#define sz(a)	(int)(a.size())
int dp[101][100001];
const int MOD = 1000000007;

int main()
{
	int n,k;
	scanf("%d%d", &n, &k);
	for(int i = 1; i <= n; i++)
	{
		int x;
		scanf("%d", &x);
		sum += x;
		if(x >= k) continue;
		else candidates.push_back(x);
	}
	if(sum < 2 * k)
	{
		cout << "0";
		return 0;
	}
	dp[0][0] = 1;
	for(int i = 0; i < sz(candidates); i++)
	{
		for(int sum = 0; sum < k; sum++)
		{
			dp[i + 1][sum] += dp[i][sum];
			while(dp[i + 1][sum] >= MOD)
				dp[i+1][sum] -= MOD;
			if(sum + candidates[i] >= k) continue;
			dp[i + 1][sum + candidates[i]] += dp[i][sum];
			while(dp[i + 1][sum + candidates[i]] >= MOD)
				dp[i + 1][sum + candidates[i]] -= MOD;
		}
	}
	long long ans = 1;
	for(int i = 1; i <= n; i++)
		ans = (ans * 2) % MOD;
	n = sz(candidates);
	long long rem = 0;
	for(int sum = 0; sum < k; sum++)
		rem = (rem + 2 * dp[n][sum]) % MOD;
	ans = (ans - rem + MOD) % MOD;
	cout << ans << "\n";
    return 0;
}
