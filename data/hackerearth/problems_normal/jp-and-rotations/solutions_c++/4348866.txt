#include <iostream>
#include <stdio.h>
#include <string>
using namespace std;

int main()
{
	int n,m;
	int a,t;
	int b;
	int right = 0;
	scanf("%d %d", &n, &m);
	scanf ("%d", &a);
	for (int i=1; i< n; i++) {
		scanf ("%d", &t);
	}
	for (int i=0; i< n; i++) {
		scanf ("%d", &b);
		if (a == b) {
			right = i;
		}
	}
	if (right == 0) {
		cout << "-1"<<endl;
		return 0;
	}
	int left = n-right;
	long lcount=0;
	char op;
	int r;
	int ans = -1;
	getchar();
	for (int i=0; i<m; i++) {
		scanf ("%c %d", &op, &r);
		if (op == 'R') {
			lcount += n-(r%n);
		} else if (op == 'L'){
			lcount += (r%n);
		}
		lcount = lcount%n;
		if ((lcount == left) && ans == -1) {
			ans = i+1;
		}
		getchar();
	}
	if (ans == n) {
		cout<<"-1"<<endl;
	} else
		cout<<ans<<endl;
}
