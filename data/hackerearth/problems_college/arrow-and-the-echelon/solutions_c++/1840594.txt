#include<bits/stdc++.h>
#define ll long long int
#define pb push_back
using namespace std;
int main()
{
	//freopen("input.txt","r",stdin);
	ll test,n,diff;
	bool flag,flaggi;
	cin>>test;
	while(test--)
	{
	    cin>>n;
	    flag=flaggi=0;
	    if(n==1 || n==2)cout<<"-1"<<endl;
	    else if(n%3==0)
        {
            for(int i=0;i<n;++i)
            {
                cout<<"5";
            }
            cout<<endl;
        }
       /* else if(n%5==0)
        {
            for(int i=0;i<n;++i)
            {
                cout<<"3";
            }
            cout<<endl;
        }*/
        else
        {
            int k=1;
            while(flag!=1)
            {
                diff=n-(5*k);
                if(diff%3==0)
                    flag=1;
                else k++;
                if(5*k>n)
                {
                    flaggi=1;
                    cout<<"-1"<<endl;
                    break;
                }
            }
            if(flaggi==0){
            for(int i=0;i<diff;++i)
            {
                cout<<"5";
            }
            for(int i=0;i<k*5;++i)
            {
                cout<<"3";
            }
            cout<<endl;
            }
        }
	}
	return 0;
}
