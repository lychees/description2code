#include<iostream>
using namespace std;
long long arr[1000][1000];
long long calculate(int k,int n){
	if(k>n)
		return 0;
   else	if(k==n)
		return 1;
	else	if(arr[k][n])
		return arr[k][n];
	arr[k][n]= calculate(k+1,n)+calculate(k,n-k);
	return arr[k][n];
}
int main(){
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		cout<<calculate(2,n)<<endl;
	}
}
