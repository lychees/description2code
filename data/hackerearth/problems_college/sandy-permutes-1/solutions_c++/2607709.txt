#include <iostream>
using namespace std;
 
int main()
{
   // cout << "Hello World!" << endl;
   int n,k;
   cin>>n>>k;
   int arr[n];
   for(int i=0;i<n;i++)
   cin>>arr[i];
   int arr1[n];
   for(int i=0;i<n;i++)
   {
   	arr1[i]=n-i;
   }
   if(k>=n)
   {
   	for(int i=0;i<n;i++)
   	cout<<arr1[i]<<" ";
   	cout<<endl;
   }
   else
   {
   while(k--)
   {
   	int i;
   	for( i=0;i<n;i++)
   	{
   		if(arr1[i]!=arr[i])
   		break;
   	}
   	//cout<<i<<endl;
   	if(i==n-1)
   	break;
   	int j=i+1;
   	int index;
   	int min=i;
   	for(int x=j;x<n;x++)
   	{
   		if(arr[x]>arr[min])
   		min=x;
   		
   	}
   //	cout<<min<<endl;
   	int temp=arr[i];
   	arr[i]=arr[min];
   	arr[min]=temp;
   }
   
   for(int i=0;i<n;i++)
   cout<<arr[i]<<" ";
   cout<<endl;
   }
    return 0;
}
