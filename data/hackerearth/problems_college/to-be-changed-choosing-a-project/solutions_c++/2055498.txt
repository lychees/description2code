#include <iostream>
#include <vector>
using namespace std;

int main() {
	int n,r;
	cin >> n >> r;
	vector<int> a;
	a.resize(n);
	for(int i = 0;i < n;i++) a[i] = i+1;
	int in = 0;
	while(a.size() > 1)
	{
		n = a.size();
		in += r%n-1;
		if(in < 0) in += n;
		in %= n;
		a.erase(a.begin()+in);
	}
	cout << a[0] << endl;
	return 0;
}
