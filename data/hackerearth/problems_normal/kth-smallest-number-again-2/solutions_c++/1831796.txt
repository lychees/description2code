#include <iostream>
#include <utility>
#include <algorithm>
#include <vector>
#include <cstdio>
using namespace std;

#define LL long long int 

bool cmp(pair<LL,LL> a, pair <LL,LL> b)
{
	if(a.first<b.first)
	{
		return true;
	}
	else
	{
		if(a.first == b.first) return a.second<b.second;
		else return false;
	}
}
int main()
{
	long long int ig = 9223372036854775807;
	//cout <<ig<<endl;
	vector< pair<LL,LL> > p(110);
	vector<LL> v(110,1);

    LL t,i,n,q,a,b,x,y,idx,sum,val,ans,k;
    
    cin>>t;
    
    while(t--)
    {
    	for(i=0;i<=100;i++) v[i]=1;
    	cin>>n>>q;
    	
    	for(i=0;i<n;i++)
    	{
    		scanf("%ld %ld",&a,&b);
    		
    		p[i].first = a;
    		p[i].second = b;
    	}
    	sort(p.begin(),p.begin()+n,cmp);
    	
    	//for(i=0;i<n;i++) cout<<p[i].first<<" "<<p[i].second<<endl;
		//cout <<"------------------------";
  
    	for(i=0;i<n-1;i++)
    	{
    		x=p[i+1].first;
    		y=p[i+1].second;
    		a=p[i].first;
    		b=p[i].second;
    		
    		if(x>=a && x<=b)
    		{
    			p[i+1].first = a;
    			p[i+1].second = b>y ? b : y;
				v[i]=0;
    		}
    		if(x==b)
    		{
    			if(b-a>=1)
    			{
    				p[i].second--;
    			}
    			else
    			{
    				v[i]=0;
    			}
    		}
    	}
    	//for(i=0;i<n;i++) if(v[i]) cout<<p[i].first<<" "<<p[i].second<<endl;
    	//cout <<"------------------------";

    	while(q--)
    	{
    		cin >> k;
	    	sum = k;
	    	
	    	for(i=0;i<n;i++)
	    	{
	    		if(!v[i]) continue;
	    		
	    		val = p[i].second-p[i].first + 1;
	    		
	    		if(val < sum)
	    		{
	    			sum -= val;
	    		}
	    		else
	    		{
	    			ans = p[i].first + sum - 1;
	    			sum=0;
	    			break;
	    		}
	    	}
	    	if(sum>0) ans=-1;
	    	cout << ans << endl;
    	}
    	
    }
    return 0;
}
