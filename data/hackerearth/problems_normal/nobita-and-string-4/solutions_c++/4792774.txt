#include <iostream>
#include<bits/stdc++.h>
using namespace std;
 
int main(void) {
	int n,i,j,k,t,l,m;
	scanf("%d\n",&t);
	char a[100000];
	for(i=0;i<t;i++){
		gets(a);
		n=strlen(a);
		k=0;
		int b[n/2+1];
		b[k]=n;
		k++;
		for(j=n-1;j>=0;j--){
			if(a[j]==' '){
				b[k]=j;
				k++;
			}
		}
		b[k]=-1;
		m=0;
		for(j=1;j<=k;j++){
			if(m>0) printf(" ");
			for(l=b[j]+1;l<=b[j-1]-1;l++)
			printf("%c",a[l]);
			m++;
		}
		printf("\n");
	}
	return 0;
}
