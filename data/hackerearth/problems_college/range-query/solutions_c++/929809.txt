#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
using namespace std;
int a[10005];
int dp[105][10005];
int main() {
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++) scanf("%d",&a[i]);
    for(int i=0;i<n;i++) {
        for(int j=1;j<=100;j++) {
            if(a[i]==j) dp[j][i]=dp[j][i-1]+1;
            else dp[j][i]=dp[j][i-1];
        }
    }
    int q;
    scanf("%d",&q);
    while(q--) {
        int temp;
        int l,r,val;
        scanf("%d %d %d",&l,&r,&val);
        if(l==0) temp=0;
        else temp=dp[val][l-1];
        printf("%d\012",dp[val][r]-temp);
    }
    return 0;
}
