#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t,x,i,j,f;
	scanf("%d",&t);
	while (t --) {
		scanf("%d",&x);
		int c = x;
		f = 0;
		while (c >= 0) {
			if (c % 3 == 0 && (x - c) % 5 == 0) {
				for (i = 0; i < c; i++) {
					printf("5");
				}
				for (i = 0; i < x - c; i++) {
					printf("3");
				}
				f = 1;
				break;
			}
			c -= 5;
			
		}
		if (f == 0) {
			printf("-1");
		}
		printf("\n");
	}
	
	return 0;
}
