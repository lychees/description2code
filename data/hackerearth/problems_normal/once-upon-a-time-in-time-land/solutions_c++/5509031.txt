#include<bits/stdc++.h>
#define Int long long
using namespace std;

Int dp[100005];

int main(){
	
	int w;
	cin>>w;
	while(w--){
		
		memset(dp,0,sizeof dp);
		int n,k,i;
		cin>>n>>k;
		Int A[n];
		for(i=0;i<n;i++)
			cin>>A[i];
		
		if(A[0]>0)
			dp[0]=A[0];
		else dp[0]=0;
		//dp[0]=max(A[0]);
		for(i=1;i<n;i++){
			
			dp[i]=max(dp[i-1],A[i]);
			if(i-k>0)
				dp[i]=max(dp[i],max(dp[i-k-1]+A[i],dp[i-k-1]));
			
		}
		
		//for(i=0;i<n;i++)
		//	cout<<dp[i]<<' ';
		//cout<<endl;
		cout<<dp[n-1]<<endl;
	}
	
	
}
