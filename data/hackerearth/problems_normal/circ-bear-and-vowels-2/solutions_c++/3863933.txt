#include <iostream>
#include<string.h>
using namespace std;

int main()
{
	int t,i,l, c =0, v =0 ;
	bool t2=0, t1=0, t0=0, flag = 0;
	cin>>t;
	while(t--)
	{
		c=v=0;
		flag = 0;
		string s;
		cin>>s;
		t2= t1 = t0 = 0;
		for(i=0;i<s.length();i++)
		{
			if(s[i] == 'a' || s[i] == 'e' || s[i] == 'i' || s[i] == 'o' || s[i] == 'u' || s[i] == 'y')
			{
				t2 = t1 = t0 = 0;
				v++;
				//cout<<s[i];
			}
			else
			{
				if(t2 == 0)
					t2 = 1;
				else if(t1 == 0)
					t1 = 1;
				else
					{
						flag = 1;
						break;
					}
					c++;
			}
		}
		if(flag == 0 && c <= v)
			cout<<"easy"<<endl;
		else
			cout<<"hard"<<endl;
	}
	return 0;
}
