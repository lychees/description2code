#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	for(int i=1;i<=t;i++)
	{
		long long int x,k;
		cin>>x>>k;
		long long int j=1,c=0;
		while(c<=k)
		{
			if((x+j)==(x^j))
			{
				c++;
			}
			if(c==k)
			{
			 cout<<"Case #"<<i<<": "<<j<<"\n";
				break;
			}
			j++;
		}
	}
}
