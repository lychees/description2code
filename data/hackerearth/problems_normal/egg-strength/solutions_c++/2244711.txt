#include <iostream>
using namespace std;
int out[501][501]={{0}};
int max(int a, int b)
{
	return (a>b)?a:b;
}
void calculate()
{
	int i,j;
	int res;
	int x;
	for(i=1;i<=500;i++)
	{
		out[i][1]=1;
		out[i][0]=0;
	}
	for(i=1;i<=500;i++)
		out[1][i]=i;
	for(i=2;i<=500;i++)
		for(j=2;j<=500;j++)
		{
			out[i][j]=99999;
			for(x=1;x<=j;x++)
			{
				res=1+max(out[i-1][x-1],out[i][j-x]);
				if (res < out[i][j])
				{
					out[i][j]=res;
				}
			}
		}
}
int main()
{
	int T,n,k;
	calculate();
    cin>>T;
    while(T--)
    {
    	cin>>n>>k;
    	cout<<out[k][n]<<"\n";
    }
    return 0;
}
