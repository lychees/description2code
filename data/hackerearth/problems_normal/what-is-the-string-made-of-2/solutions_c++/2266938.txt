#include <iostream>
#include<string.h>
using namespace std;
int dashes[10];
void initialize(){
	dashes[0]=6;
	dashes[1]=2;
	dashes[2]=5;
	dashes[3]=5;
	dashes[4]=4;
	dashes[5]=5;
	dashes[6]=6;
	dashes[7]=3;
	dashes[8]=7;
	dashes[9]=6;
}
void printResult(char c[]){
	int result=0;
	for(int i=0;i<strlen(c);i++){
		result+=dashes[c[i]-'0'];
	}
	cout<<result<<endl;
}
int main()
{
	initialize();
	char c[101];
	cin>>c;
	printResult(c);
    return 0;
}
