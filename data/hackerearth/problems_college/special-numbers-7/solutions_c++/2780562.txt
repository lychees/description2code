#include <iostream>
using namespace std;

int m[3000];
int ans[1001];

void Sieve(){
	for(int i=2;i<3000;++i)
	    if(!m[i])
           for(int j=2;j*i<3000;++j)
               m[i*j]++;
    for(int i=0,j=0;i<3000 && j<=1000;++i)			
        if(m[i]>=3)
           ans[j++]=i;
}

int main()
{
	int t,n; cin>>t;
	Sieve();
	while(t--){
		cin>>n;	
		printf("%d\n",ans[n-1]);
    }
    //cout << "Hello World!" << endl;
    return 0;
}
