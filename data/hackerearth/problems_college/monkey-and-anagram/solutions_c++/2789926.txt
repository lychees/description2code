#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,i;
    cin>>t;
    while(t--)
    {
    	string s1,s2;
    	int a[26]={0};
    
    	cin>>s1;
    	cin>>s2;
    		int count=0,l1=s1.length(),l2=s2.length();
    	
    	//cout<<s1<<" "<<s2<<endl;
    //	int 
    	for(i=0;i<l2;i++)
    	{
    		a[s2[i]-'a']++;
    	}
    	for(i=0;i<l1;i++)
    	{
    		a[s1[i]-'a']--;
    	}
    	for(i=0;i<26;i++)
    	{
    		if(a[i]>0) count++;
    		else if(a[i]<0) {count=-1;break;}
    	}
    	
    printf("%d\n",count);
    }
    return 0;
}
