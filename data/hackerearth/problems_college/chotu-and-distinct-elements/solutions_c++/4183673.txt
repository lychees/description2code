#include <bits/stdc++.h>
#define ll long long int
#define sii(x) scanf("%d",&x)
#define pii(x) printf("%d\n",x)
#define sll(x) scanf("%lld",&x)
#define pll(x) printf("%lld\n",x)
#define present_set_map(container, element) (container.find(element) != container.end()) 
#define present_vec(container, element) (find(all(container),element) != container.end()) 
#define mp make_pair
#define all(x) x.begin(),x.end()
#define all_rev(x) x.rbegin(),x.rend()
#define pb push_back
#define fr first
#define se second
#define mod 1000000007
using namespace std;

ll freq[1000007];
ll arr[1000007];
ll uni[1000007];

int main()
{
    ll tc,distinct,n,i;
    sll(tc);
    while(tc--)
    {
        distinct=0;
        memset(freq,0,sizeof(freq));
        sll(n);
        for(i=0;i<n;i++)
        {
            sll(arr[i]);
            if(freq[arr[i]]==0)
                uni[distinct++]=arr[i];
            freq[arr[i]]++;
        }
        sort(uni,uni+distinct);
        /*printf("distinct=%lld\nuni array = ",distinct);
        for(i=0;i<distinct;i++)
            printf("%lld ",uni[i]);
        printf("\n");*/
        ll terms=1;
        ll p1=distinct-1;
        ll p2=0;
        for(i=0;terms<=distinct;i++)
        {
            if(i%2==0)
                printf("%lld ",uni[p1--]);
            else
                printf("%lld ",uni[p2++]);
            terms++;
        }
        printf("\n");
    }   
}
