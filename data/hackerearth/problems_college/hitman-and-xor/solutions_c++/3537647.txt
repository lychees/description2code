#include <iostream>
using namespace std;

int main()
{
  long int t,i,c,n,a;
  cin>>t;
  while(t--)
  {
  	c=0;
  	cin>>n;
  	for(i=0;i<n;i++)
  		{
  			 cin>>a;
  		     if(a%2==0)
  		     c++;
  		}
  		 if(n-c==c)
         cout<<n<<'\n';
         else
         cout<<(n-c)*c<<'\n';
  }
    return 0;
}
