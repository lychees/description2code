#include <iostream>
#include<map>
using namespace std;

int main() {

string a,b;

int t;
cin>>t;

while(t--)
{
	
	cin>>a>>b;
	
	map<char,int> m;
	map<char,int> m1;
	map<char,int>::iterator it;
	map<char,int>::iterator it1;

	
	long long int i,l1,l2,c=0,f=0;
	
	l1 = a.length();
	l2 = b.length();
	for(i=0;i<l1;i++)
	m[a[i]]++;
	
	for(i=0;i<l2;i++)
	m1[b[i]]++;
	
	for(i=0;i<l1;i++)
	{
        it=m1.find(a[i]);
        
        if(it==m.end() || it->second==0)
        {f=1;
        break;
        }
        
        it->second--;
	}

if(f==1)
cout<<"-1\n";

else
{
	m1.clear();
	for(i=0;i<l2;i++)
	m1[b[i]]++;
	for(it=m1.begin();it!=m1.end();++it)
	{
		
		it1=m.find(it->first);
		
		if(it1==m.end() || it->second>it1->second)
	      c++;
	}
	
	
	cout<<c<<"\n";
}
	


}

	return 0;
}
