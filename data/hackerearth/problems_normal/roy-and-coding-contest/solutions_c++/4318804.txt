#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define MOD 1000000007
#define MAX 100005
ll modexp(ll a, ll b)
{
	ll res = 1;
	while(b)
	{
		res=(res*a)%MOD;
		a=(a*a)%MOD;
		b/=2;
	}
	return res;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,m;
		cin>>n>>m;
		if(n==1) {cout<<0<<endl; continue;}
		if(n==2) {cout<<2<<endl;continue;}
		int p,c,tmp;
		p = c = 1;
		int tim = 1;
		while(1)
		{
			tmp = p+c;
			p = c = tmp;
			tim++;
			if(p > m)
			{
				p = m;
				break;
			}
			if(c >= n) 
			{
				break;
			}
		}
		if(c >= n) cout<<tim<<endl;
		else
		{
			int ans;
			n-=c;
			if(n%m ==0) ans = n/m;
			else ans = n/m+1;
			cout<<ans+tim<<endl;
		}
	}
    return 0;
}
