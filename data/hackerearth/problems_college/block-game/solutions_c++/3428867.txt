#include <iostream>
#include <stdio.h>
using namespace std;
int main()
{
	long long int n,t,i,j,diff;
	long long int mincost,cost;
	scanf("%lld",&t);
	while(t--)
	{
		mincost=1000000000000000000;
	scanf("%lld",&n);
	long long int a[n],c[n];
	for(i=0;i<n;i++)
		scanf("%lld",&a[i]);
	for(i=0;i<n;i++)
		scanf("%lld",&c[i]);
	for(i=0;i<n;i++)
	{
		cost=0;
		for(j=0;j<n;j++)
		{
			diff=a[j]-a[i];
			if(diff<0)
				diff=(-1)*diff;
			cost=cost+(diff*c[j]);
			if(cost>mincost)
				break;
		}
		if(cost<mincost)
			mincost=cost;
	}
	printf("%lld\n",mincost);
	}
	return 0;
}
