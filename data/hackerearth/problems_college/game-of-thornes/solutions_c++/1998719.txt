#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#include <string>
using namespace std;
 
int alp[26];
int main() {
 
    string s;
    cin>>s;
     int a=0;
    int flag = 1;
    // Assign Flag a value of 0 or 1 depending on whether or not you find what you are looking for, in the given string 
    int l=s.length();
 
 
        for(int i=0;i<l;i++)
          {
            alp[s.at(i)-'a']++;            
        }  
    if(l%2==0)
        {
        for(int i=0;i<26;i++)
            {
            if(alp[i]%2==1)
                {
                flag=0;
                break;
            }
        }
    }
    else
        {
        for(int i=0;i<26;i++)
            {
            if(alp[i]%2==1)
                {
                a++;
                if(a>1)
                    {
                    flag=0;
                    break;
                }
            }
        }
 
 
    }
 
 
 
 
    if(flag==0)
        cout<<"NO";
    else
        cout<<"YES";
    return 0;
}
