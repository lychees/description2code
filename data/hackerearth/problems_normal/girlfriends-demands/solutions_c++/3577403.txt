#include <bits/stdc++.h>
using namespace std;

int main()
{
    char str[100002];
    cin>>str;
    int n=strlen(str);
    int q;
    cin>>q;
    while(q--) {
    	long long int a,b;
    	cin>>a>>b;
    	if(a%n!=0)
    	a=a%n;
    	else a=n;
    	if(b%n!=0)
    	b=b%n;
    	else b=n;
    	if(str[a-1]==str[b-1]) cout<<"Yes\n";
    	else cout<<"No\n";
    }
    return 0;
}
