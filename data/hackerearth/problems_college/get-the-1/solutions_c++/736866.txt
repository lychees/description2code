#include<iostream>
using namespace std;
int test(int n){
	int sum=0;
	int b=n;
	int x,c;
	while(b!=0){
		c=b%10;
		sum+=(c*c);
		b/=10;
	}
	if((sum/10)!=0){
		x=test(sum);
		return x;
	}
	else
		return sum;
}
int main(){
	int t,n,i=1,x;
	cin>>t;
	while(t--){
		cin>>n;
		x=test(n);
		if(x==1)
			cout<<"Case #"<<i<<":YES"<<endl;
		else
		 	cout<<"Case #"<<i<<":NO"<<endl;
		i++;	 		
	}	
	return 0;
}
