# print 'Hello World!'
import operator as op
M = 10000007


def ncr(n, r):
    res = 1
    for i in range(1, r+1):
        res *= n-i+1
        # res %= M
        res /= i
        # res %= M
    return res


n, r = map(int, raw_input().split())
if r > n:
    print -1
else:
    print ncr(n-1, r-1) % M
