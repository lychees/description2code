#include <iostream>
#include<algorithm>
using namespace std;
long int arr[10];
long int f(long int cur,long int pos,long int k,long int n,long int size){
	if(size==0)
	return n/cur;
	if(cur>n)
	return 0;
	long int i,ans=0;
	for(i=pos;i<k;i++){
		if(cur*arr[i]>n)
		break;
		ans=ans+f(cur*arr[i],i+1,k,n,size-1);
	}
	return ans;
}
int main()
{
   long int t,k,a,b,i,ans=0;
   cin>>k>>t;
   for(i=0;i<k;i++)
   cin>>arr[i];
   sort(arr,arr+k);
   while(t--){
   	ans=0;
   	cin>>a>>b;
   	for(i=1;i<=k;i++){
   		ans=ans+(2*(i%2)-1)*(f(1,0,k,b,i)-f(1,0,k,a-1,i));
   	}
   	cout<<ans<<"\n";
   }
    return 0;
}
