#include<bits/stdc++.h>

using namespace std;

long long gcd(long long a,long long b)
{
	if(b==0)
	return a;
	else
	return gcd(b,a%b);
}
long long findMax(long long n1,long long n2,long long k)
{
	long long maxValue=INT_MIN;
    long long i,j;    
        for( i=max((long long)1,n1-k);i<=n1+k;i++) {
            for( j=max((long long)1,n2-k);j<=n2+k;j++) {
                if(abs(n1-i)+abs(n2-j)<=k) {
                    maxValue=max(maxValue,gcd(i,j));
                }
            }
        }
      return maxValue;
}
int main() {
    int test;
    cin>>test;
    while(test--) {
        int n1,n2,k;
        cin>>n1>>n2>>k;
        
        cout<<findMax(n1,n2,k)<<endl;
        
    }
    return 0;
}
