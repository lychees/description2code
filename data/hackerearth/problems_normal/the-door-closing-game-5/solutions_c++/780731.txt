//Based on NIM game
#include<iostream>
#include<string.h>
#include<stdio.h>
using namespace std;
char s[105];
int main()
{
	int tc=0;
	cin>>tc;
	while(tc-->0)
	{
		scanf("%s",s);
		int l=strlen(s);
		int cur=0;
		int flag=0;
		int ans=0;
		for(int i=0;i<l;i++)
		{
			if(s[i]=='|')
			{
				flag=1;
				ans=ans^cur;
				cur=0;
			}
			else if(s[i]=='_')
			{
				cur++;
			}
		}
		if(flag&&cur)
			ans=ans^cur;
		if(ans)
			printf("Alice\n");
		else
			printf("Bob\n");
	}
	return 0;
}
