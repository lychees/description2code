#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s1,s2;
    	cin>>s1;
    	int a,b,i=0,j=s1.length()-1,sum=0,flag=0;
    	cin>>a>>b;
    	while(i<j){
    	 if((s1[i]=='a' &&s1[j]=='b')||(s1[j]=='a' &&s1[i]=='b'))
    		{
    			flag=1;
    			break;
    		}
    		else if((s1[i]=='a' &&s1[j]=='/')||(s1[j]=='a' &&s1[i]=='/'))
    		sum+=a;
    		else if((s1[i]=='b' &&s1[j]=='/')||(s1[j]=='b' &&s1[i]=='/'))
    		sum+=b;
    			else if((s1[i]=='/' &&s1[j]=='/'))
    			sum+=2*min(a,b);
    		i++;
    		j--;
    	}
    	if(!flag)
    	cout<<sum<<endl;
    	else cout<<"-1"<<endl;
    }
    return 0;
}
