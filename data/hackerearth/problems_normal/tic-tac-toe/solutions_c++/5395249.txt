#include<bits/stdc++.h>
using namespace std;

#define mod 1000000007
#define W while
#define LL long long int

LL gcds(LL a,LL b)
{
	if(b==0)
		return a;
	else
		return gcds(b,a%b);
}
LL power(LL a,LL b)
{
	if(b==0)
		return 1;
	if(b==1)
        return a;
	LL temp = power(a,b/2);
	if(b%2)	
		return (((a%mod)*(temp%mod))%mod*(temp%mod))%mod;
	else 
		return ((temp%mod)*(temp%mod))%mod;
}
