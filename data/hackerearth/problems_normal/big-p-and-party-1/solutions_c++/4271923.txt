#include <bits/stdc++.h>
using namespace std;

int main()
{
    int N,M,a,b;
    cin>>N>>M;
    vector<int> gr[N+1];
    for(int i=0;i<M;i++)
    {
    	cin>>a>>b;
    	gr[a].push_back(b);
    	gr[b].push_back(a);
    }
    int vis[N];
    for(int i=0;i<N;i++)
    {
    	vis[i] = 1000000009;
    }
    
    queue<int> q;
    q.push(0);
    vis[0] = 0;
    while(!q.empty())
    {
    	a = q.front();
    	q.pop();
   		for(vector<int>:: iterator it = gr[a].begin();it!=gr[a].end();it++)
   		{
   			if(vis[*it]==1000000009)
   			{
   				vis[*it] = vis[a]+1;
   				q.push(*it);
    		}
    	}
    }
    for(int i=1;i<N;i++)
    {
    	if(vis[i]==1000000009)
    		cout<<"INF";
    	else cout<<vis[i];
    	cout<<endl;
    }
}
