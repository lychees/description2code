#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int t,i,f,c;
    string s;
    cin>>t;
    while(t--){
    	cin>>s;
    	f=1;
    	c=0;
    	for(i=0;i<s.size();i++){
    		if(s[i]=='4' || s[i]=='7'){
    			c++;
    		}
    	}
    	
    	string a = to_string(c);
    	
	    for(i=0;i<a.size();i++){
	    	if(a[i]=='4' || a[i]=='7'){
	    		f=1;
	    	}else{
	    		f=0;
	    		break;
	    	}
	    }
    	
    	if(f){
    		cout<<"YES"<<endl;
    	}else{
    		cout<<"NO"<<endl;
    	}
    }
    return 0;
}
