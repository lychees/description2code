#include <iostream>
using namespace std;

int main()
{
   int t;
   long long int a[10000];
   cin>>t;
   while(t--)
   {
   	int n;
   	cin>>n;
   	for(int i=0;i<n;i++)
   	{
   		cin>>a[i];
   	}
   	
   	for(int i=1;i<n;i++)
   	{
   		for(int j=0;j<=(i/2);j++)
   		{
   			long long int temp = a[j]+a[i-j-1];
   			if(a[i] < temp)
   			   a[i]=temp;
   		}
   	}
   	cout<<a[n-1]<<endl;
   	
   }
   return 0;
}
