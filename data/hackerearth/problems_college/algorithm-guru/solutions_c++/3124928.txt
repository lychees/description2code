#include <iostream>
using namespace std;
int closest(int i, int x,int N,int a[])
{
    if(x < 0)
        return 100000000;
	if(i == N)
		return x;
	return min(closest(i,x-a[i],N,a),closest(i+1,x,N,a));
}
int main()
{
    int T=0;
    cin >> T;
    for (int i = 0; i < T; ++i)
    {
    	int N,x;
    	cin >> N;
        cin >> x;
    	int a[N];

    	for (int n = 0; n < N; ++n)
    	{
    		cin >> a[n];
    	}
        int Mem[x+1],Mempre[x+1];
        for (int j = 0; j <= x; ++j)
        {
           Mem[j] = j;
           Mempre[j] = j;
        }
        for(int n=N-1;n>=0;n--)
        {
            for (int j = 0; j <= x; ++j)
            {
                if(j-a[n]>=0)
                    Mem[j] = min(Mem[j-a[n]],Mempre[j]);
                else
                    Mem[j] = Mempre[j];
                //cout << Mem[n][j] <<"\t";
            }
            for (int j = 0; j <= x; ++j)
                Mempre[j] = Mem[j];
            //cout << endl;
        }
        cout <<  x -Mem[x] << endl;

    }

}
