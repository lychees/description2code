#include<iostream>
using namespace std;
main()
{
    int t,n,k,ct;
    cin>>t;
    while(t--)
    {
        int arr[1000];
        cin>>n>>k;
        ct=0;
        for(int i=0;i<n;i++)
            cin>>arr[i];
        for(int i=0;i<n;i++)
        {
            if(arr[i]>0)
                continue;
            else
                ct++;
        }
        if(ct>=k)
            cout<<"NO"<<endl;
        else
            cout<<"YES"<<endl;
    }
}
