#include <bits/stdc++.h>
using namespace std;
char str[100005];
int main()
{
	long long t,n,i,j,ans,rec,num;
    cin >> t;
    while(t--){
    	cin >> n;
    	ans = 0;
    	rec = 0;
    	getchar();
    	for(i = 0;i < n;i++){
    		gets(str);
  //  		printf("%d %s\n",i,str);
    		if(strcmp(str,"New Project") == 0){
    //			cout << "y\n";
    			if(rec <= 0){
    				ans++;
    			}
    			else{
    				rec--;
    			}
    		}
    		else{
    			num = 0;
    			for(j = 0;str[j] != 0;j++){
    				num = num*10 + str[j]-'0';
    			}
    			rec += num;
    //			cout << num << endl;
    		}
    	}
    	cout << ans << endl;
    }
    return 0;
}
