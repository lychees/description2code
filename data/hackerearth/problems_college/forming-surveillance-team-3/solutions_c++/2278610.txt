#include <iostream>
#include <stdio.h>
#include <set>
using namespace std;

set <int> s;

void initialize()
{
	s.clear();
}

int main()
{
    int T;
    scanf("%d", &T);
    while(T--)
    {
    	initialize();
    	int M;
    	scanf("%d", &M);
    	while(M--)
    	{
    		int a;
    		scanf("%d", &a);
    		s.insert(a);
    	}
    	printf("%d\n", s.size());
    }
    return 0;
}
