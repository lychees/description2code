#include <iostream>
#include <stdio.h>
using namespace std;
#define max 1000006

int a[max];

void slve()
{
	int i,j;
	a[0]=0;
	a[1]=0;
	for(i=0;i<max;i++)
	{
		a[i]=0;
	}
	for(i=2;i*i<=max;i++)
	{
		if(a[i]==0)
		{
			for(j=i*i;j<=max;j+=i)
			{
				if(a[j]==0)
					a[j]=i;
			}
			a[i]=i;
		}
	}
}

int main()
{
	int n,r;
	slve();
	scanf("%d",&n);
	while(n--)
	{
		scanf("%d",&r);
		if(a[r]==0)
			printf("0\n");
		else
			printf("%d\n",r-a[r]);
	}
	return 0;
}
