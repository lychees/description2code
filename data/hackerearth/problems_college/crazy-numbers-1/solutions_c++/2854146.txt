#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin >> t;
	
	for(int i=0 ;i<t; ++i) {
		int n;
		cin  >> n;
		if(n == 0) { cout << 0 << endl; continue; }
		vector<long long int> a(10,1);
		a[0] = 1; a[1] = 1; a[9] = 1;
		
		for(int i=2;i<=8;++i) {
			a[i] = 2;
		}
		
		vector<long long int> b(10,1);
		
		for(int j=3; j<=n; ++j) {
			for(int i=0; i<=9; ++i) {
				if(i != 0 && i!=9) { 
					b[i] = (a[i-1] + a[i+1])%1000000007 ;
				} else { 
					if(i == 0) { 
						b[i] = (a[i+1])%1000000007;
					} else { 
						b[i] = (a[i-1])%1000000007;
					}				
				}
			}
			a = b;
		}
	
		
		long long int sum = 0;
		for(int i=0; i<=9; ++i) {
			sum = sum + a[i];
		}
		if(n == 1) {
			cout << 10 << endl;
			continue;
		}
		
		cout << sum%1000000007 << endl;
	}
	
	
    return 0;
}
