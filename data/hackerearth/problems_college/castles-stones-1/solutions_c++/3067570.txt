#include <iostream>
using namespace std;

int main()
{
    int t,n,m,k,i,min,s;
    cin>>t;
    while(t--){cin>>m>>n;
               int p=0;
    	       int a[n];
    	       for(i=0;i<n;i++){cin>>a[i];if(a[i]>m){a[i]=0;}}
    	       for(i=0;i<n;i++){for(s=i+1;s<n;s++){if(a[i]>a[s]){min=a[i];
    	                                                         a[i]=a[s];
    	                                                         a[s]=min;
    	       	
    	       }}}
    	       
    	       for(i=0;i<n;i++){
    	       	               if(a[i]>0){for(k=n-1;k>i;k--){if(a[k]>0 && a[i]>0){if(a[i]+a[k]<=m){p++;
    	       	                	  a[k]=0;a[i]=0;}}
    	       	              }
    	                       } }
    	                        for(i=0;i<n;i++){if(a[i]>0){p++;}}
    	                         cout<<p<<"\n";
    }
    return 0;
}
