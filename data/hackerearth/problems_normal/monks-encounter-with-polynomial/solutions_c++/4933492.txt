#include <cstdio>
using namespace std;

//returns the first int x so that the polynomial is >=k;
int bs(int a, int b, int c, long long k){
	int left=0, right=100000, mid, ans = right;
	while (left <= right){
		mid=(right+left)/2;
		//printf ("%d\n", mid);
		if ((long long)a * mid * mid + b * mid + c >= k) {
			ans = mid;
			right=mid-1;
		} else {
			left=mid+1;
		}
	}
	return ans;
}

int main(){
	int t;
	scanf ("%d", &t);
	while (t--){
		int a, b, c;
		long long k;
		scanf ("%d %d %d %lld", &a, &b, &c, &k);
		printf ("%d\n", bs(a, b, c, k));
	}
	return 0;
}
