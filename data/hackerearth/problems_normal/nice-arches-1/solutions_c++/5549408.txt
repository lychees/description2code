#include <iostream>
#include <stack>

using namespace std;

int main()
{
    int m;
    int output = 0;
    string word = "";
    
    cin >> m;
    
    for(int i = 0; i < m; ++i) {
    	cin >> word;
    	if(word.length() % 2 == 0) {
	    	int numOfAs = 0;
	    	for(unsigned j = 0; j < word.length(); ++j) {
	    		if(word.at(j) == 'A') {
	    			numOfAs++;
	    		}
	    	}
	    	if(numOfAs % 2 == 0) {
	    		std::stack<bool> stk; // A: true, B: false
	    		for(unsigned j = 0; j < word.length(); ++j) {
		    		if(word.at(j) == 'A') {
		    			if(stk.empty() || !stk.top()) {
		    				stk.push(true);
		    			} else {
		    				stk.pop();
		    			}
		    		} else {
		    			if(stk.empty() || stk.top()) {
		    				stk.push(false);
		    			} else {
		    				stk.pop();
		    			}
		    		}
		    	}
		    	if(stk.empty()) {
	    			output++;
		    	}
	    	}
    	}
    }
    
    cout << output;
    return 0;
}
