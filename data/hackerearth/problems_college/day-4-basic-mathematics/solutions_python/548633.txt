import math
t=int(raw_input())
result=[]
for i in range(0,t):
    days=int(raw_input())
    days*=-2
    root=(-1+math.sqrt(1-(4*days)))/2
    root=math.ceil(root)
    result.append(root)

for i in result:
    if i%3==0:
        print 'SL'
    elif i%3==1:
        print 'LB'
    elif i%3==2:
        print 'BS'
