#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;cin>>t;
    while(t--)
    {
        int n;cin>>n;
        int m = n/3;
        if(m*3 == n && m >= 15)
        {
            cout<<"CORRECT\n";
        }
        else
            cout<<"INCORRECT\n";
    }
    return 0;
}
