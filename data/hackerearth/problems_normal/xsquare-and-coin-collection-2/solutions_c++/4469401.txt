int main()
{
	int T;
	cin >> T;
	while (T--)
	{
		i64 n, k, a[100001], dp[100001],i;
		cin >> n >> k;
		for (i = 0; i < n; i++)
		{
			cin >> a[i];
		}
		memset(dp, 0, sizeof(dp));
		if (a[0] <= k)
			dp[0] = a[0];
		for (i = 1; i < n; i++)
		{
			if (a[i] <= k)
				dp[i] = dp[i - 1] + a[i];
		}
		i64 maxx = dp[0];
		for (i = 1; i < n; i++)
		{
			if (dp[i] > maxx)
			{
				maxx = dp[i];
			}
		}
		cout << maxx << endl;
	}
}
