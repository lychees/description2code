#include <iostream>
#include<math.h>
using namespace std;
int gcd(int a,int b)
{
	if(b==0)return a;
	return gcd(b,a%b);
}
int main()
{
    long long t,a,b,c,d;
    cin>>t;
    while(t--){
    	cin>>a>>b>>c>>d;
    	long long gc=gcd(fabs(a-c),fabs(b-d));
    	if(gc>1)cout<<(gc-1)<<endl;
    	else cout<<"0"<<endl;
    }
    return 0;
}
