#include<bits/stdc++.h>
using namespace std;
long long sum[59]={0};
#define M 10e18
long long gcd(long long a,long long b)
{
	if(b<1)
	{
		return a;
	}
	else
	{
		return gcd(b,a%b);
	}
}
long long lcm(long long a)
{
	long long sum=1;
	for(long long i=1;i<=a;i++)
	{
		long long d=gcd(i,sum);
		sum=sum/d;
		if(M/sum<=i)
		{
			return M;
		}
		sum=sum*i;
	}
	return sum;
}
long long func(long long a)
{
	long long total[52]={0},i,j;
	for(i=50;i>=1;i--)
	{
		long long m=lcm(i-1);
		if(a/m>0)
		{
			total[i]+=a/m;
		    for(j=1;j<i;j++)
		    {
			   total[j]-=total[i];
		    }
		}
	}
	long long ans=0;
	for(i=1;i<=50;i++)
	{
		ans+=(sum[i]+1)*total[i];
	}
	if(a>=2)
	{
		ans=ans-2;
	}
	return ans;
}
