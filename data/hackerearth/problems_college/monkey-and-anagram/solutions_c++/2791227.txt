#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin >> t;
	while ( t -- )
	{
		string ch, ch1;
		cin >> ch >> ch1;
		vector<int> count1(27);
		fill(count1.begin(),count1.end(),0);
		for( int i = 0 ; i < ch.length() ; i++ ) {
			count1[ch[i] - 'a']++;
		}
		
		for( int i = 0 ; i < ch1.length() ; i++ ) {
			count1[ ch1[i] - 'a']--;
		}
		
		int c1=0, ok=0 ;
		for( int i = 0 ; i < 26 ; i++){ //cout << count1[i] <<" ";
			if( count1[i] > 0 ) {
				ok=1;
				break;}
			else if( count1[i] < 0 ) c1++;
				
		}
		if( ok ) cout << "-1" << endl;
		else cout << c1 << endl; 
	}
	return 0;
}
