#include<bits/stdc++.h>
using namespace std;

string s[510];

int main() {
    int n;
    scanf("%d",&n);
    int maxLen=-1;
    for(int i=0;i<n;i++) {
        cin>>s[i];
        int len=s[i].length();
        maxLen=max(maxLen,len);
    }
    for(int i=0;i<n;i++) {
        int len=s[i].size();
        while(len<maxLen) {
            s[i]="0"+s[i];
            len++;
        }
    }
    string res;
    int carry=0;
    for(int i=maxLen-1;i>=0;i--) {
        int sum=0;
        for(int j=0;j<n;j++) {
            sum+=s[j][i]-'0';
        }
        sum+=carry;
        res=(char)(sum%10+'0')+res;
        carry=sum/10;
    }
    while(carry) {
        res=(char)(carry%10+'0')+res;
        carry/=10;
    }
    int sum=0;
    for(int i=0;i<res.length();i++) {
        sum+=res[i]-'0';
    }
    while(sum>=10) {
        int temp=0;
        int re=sum;
        while(re) {
            temp+=re%10;
            re/=10;
        }
        sum=temp;
    }
    printf("%d\012",sum);
    return 0;
}
