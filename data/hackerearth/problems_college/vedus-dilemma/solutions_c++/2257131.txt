#include<bits/stdc++.h>
#define scn(a) scanf("%d",&a)
#define prn(a) printf("%d ",a)
using namespace std;
int gcd(int A,int B)
{
	int T;
	while (B!=0){
		T = B;
		B = A%T;
		A = T;
	}
		return A;
}

int comp(const void *pa,const void *pb)
{
	int *a = (int *)pa;
	int *b = (int *)pb;
	return a[1]-b[1];  
}

int main()
{
	int t;
	scn(t);
	while(t--){
		int n; scn(n);
		int a[n],p,q,i,count=0;
		for(i=0;i<n;i++)
			scn(a[i]);
		scanf("%d%d",&p,&q);
		int b[n][2],x=0;
		for(i=0;i<n;i++){
			if(gcd(a[i],p)>1){
				b[x][0]=a[i];
				b[x][1]=abs(a[i]-p);
				count++;
				x++;
			}
		}
		if(count<q)
			printf("-1");
		else{
			qsort(b,x,sizeof(b[0]),comp);
			int ans[q];
			for(i=0;i<q;i++)
				ans[i]=b[i][0];
			sort(ans,ans + q);
			for(i=0;i<q;i++)
				prn(ans[i]);
		}
		printf("\n");
	}
	return 0;
}
