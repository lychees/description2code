#include <stdlib.h>
#include <stdio.h>

int main(int argc, char *argv[])
{
int i,j;
int cnt=0;
long int n;
scanf("%ld",&n);
while(n--) 
{
int from;
cnt =0;
//=atoi(argv[1]);
int to;
//=atoi(argv[2]);
scanf("%d",&from);
scanf("%d",&to);
int *sieve=(int *)calloc(to+1,sizeof(int));
for(i=2;i<=to;i++)
{
if(sieve[i]==0)
{
if(i>=from)
{
cnt++;
}
if(i*i<=to)
{
for(j=i;j<=to;j+=i)
{
sieve[j]=1;
}
}
}
}
printf("%d ",cnt);

printf("\n");

free(sieve);
} 
return 0;
}
