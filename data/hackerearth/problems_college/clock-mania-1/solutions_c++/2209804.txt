#include <iostream>
using namespace std;
 
int main()
{
    int n;
    int d,a,b,c,e,x;
    cin>>n;
    while(n--)
    {
    	cin>>a>>b;
    	if(a==12)
    	c=0;
    	else
    	c=30*a;
    	d=b/2;
    	e=6*b;
    	c=c+d;
    	x=c-e;
    	if(x<0)
    	x=x*(-1);
    	if(x>180)
    	x=360-x;
    	cout<<x<<endl;
    }
    return 0;
}
