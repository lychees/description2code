#include <bits/stdc++.h>
using namespace std;

int main(void)
{
   // cout << "Hello World!" << endl;
   int n,k,k1;
   cin>>n;
   vector<int>v;
   vector<int>v1;
   for(int i=0;i<n;i++)
   {
   	cin>>k;
   	v.push_back(k);
   }
   for(int j=0;j<n;j++)
   {
   	cin>>k1;
   	v1.push_back(k1);
   }
   int maxm1=0;
   int maxm2=0;
   for(int k=0;k<n-1;k++)
   {
   	maxm1=max(maxm1,abs(v[k]-v[k+1]));
   	maxm2=max(maxm2,abs(v1[k]-v1[k+1]));
   }
   if(maxm1>maxm2)
   {
   	cout<<"Dom"<<endl;
   	cout<<maxm1<<endl;
   }
   else if(maxm1<maxm2)
   {
   	cout<<"Brian"<<endl;
   	cout<<maxm2<<endl;
   }
   else
   cout<<"Tie"<<endl;
    return 0;
}
