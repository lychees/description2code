ll power(ll a,ll b)
{
	if(!b)
		return 1;
	if(b==1)
		return a;
	ll temp=power(a,b/2);
	temp=(temp*temp);
	if(b&1)
		temp=(temp*a);
	return temp;
}
int find( int i,vi &root)
{
	while( i != root[i])
		i= root[i];
		return i;
}
void merge(int x,int y,vi& rank,vi & root)
{
	if( rank[x]>rank[y])
	{
		root[y]=x;
	}
	else
		root[x]=y;
		if( rank[x]==rank[y])
			rank[y]++;
}
int main()
    {
		ios_base::sync_with_stdio(false);
		int i, n, m;
		cin >> n >> m;
		vi root(n);
		for( i=0;i<n;i++)
			root[i]=i;
		vi rank(n);
		int a,b,x,y;
		for( i=0;i<m;i++)
		{
			cin >> a >> b;
			a--;
			b--;
			x=find(a,root);
			y=find(b,root);
			root[a]=x;
			root[b]=y;
			merge(x,y,rank,root);
		}
		set<int> s;
		int ans=n;
		for(i=0;i<n;i++)
		{
			s.insert(find(i,root));
		}
		ans =s.size();
		cout << ans;
    return 0;
}
