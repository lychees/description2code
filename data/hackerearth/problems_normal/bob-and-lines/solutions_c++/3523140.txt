#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define ff first
#define ss second
int x[100001];
int y[100001];

int main(){
    int t;
    cin >> t;
    while(t--){
        int n,q;
        scanf("%d%d",&n,&q);
        for(int i=0;i<n;i++){
            int x1,x2,y1,y2;
            scanf("%d%d%d%d",&x1,&y1,&x2,&y2);
            int xm = min(x1,x2);
            int ym = min(y1,y2);
            int xma = max(x1,x2);
            int yma = max(y1,y2);
            x[xm]++,x[xma+1]--;
            y[ym]++,y[yma+1]--;
        }
        for(int i=1;i<=100000;i++)x[i]+=x[i-1],y[i]+=y[i-1];
        for(int i=0;i<q;i++){
            int xx,yy;
            scanf("%d%d",&xx,&yy);
            if(xx==0)printf("%d\n",y[yy]);
            else printf("%d\n",x[xx]);
        }
        memset(x,0,sizeof x);
        memset(y,0,sizeof y);
    }
    return 0;
}
