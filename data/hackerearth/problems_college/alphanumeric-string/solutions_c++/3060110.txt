#include<bits/stdc++.h>
using namespace std;
int convert(string str)
{
	int sum = 0;
	int i,len = str.length();
	for(i=0;i<len;i++)
	{
		sum = sum*10 + str[i]-48;
	}
	return sum;
}
int main()
{
	int t;
	scanf("%d ",&t);
	
	while(t--)
	{ 
	    char str[1000];
	    scanf(" %[^\n]s",str);
		int i;
		int len =strlen(str);
		int sum=0;
		for(i=0;i<len;i++)
		{
			if(isdigit(str[i]))
			{
				sum = sum + str[i]-48;
			}
		} 
		vector<string>vec;
		char ch[1000];
		int j=0;
		for(i=0;i<=len;i++)
		{
			if(!isspace(str[i]) && i!=len)
			{
			if(isalpha(str[i]) )
			{
			ch[j]=str[i];
			j++;	
			}
		}
			else
			{
			ch[j]='\0';
			vec.push_back(ch);
			j=0;
			}
			
		}
		sort(vec.begin(),vec.end());
		for(i=0;i<vec.size();i++)
		{
			sum = sum-vec[i].length();
			cout<<vec[i]<<" "<<abs(sum)<<" ";
		}
		cout<<"\n";
	}
}
