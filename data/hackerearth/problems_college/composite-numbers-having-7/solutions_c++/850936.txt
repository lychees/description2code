
#include<iostream>
#include<math.h>
#include<vector>
#define Max 1000006
#define mod 1000000007
using namespace std;
vector <int> isPrime(Max,1);
void genprime(){
    for(int i=2; i<=Max; i++) if(isPrime[i])
        for(int j=i; j<=Max; j+=i){
            if(j>i){
                int k=j;
                while(k>0){
                    if(k%10==7){ isPrime[j] = 0; break;}
                    k/=10;
                }
            }
        }
}
int main()
{
	int t;
	genprime();
	cin>>t;
	while(t--){
        int m, n, sum=0;
        cin>>m>>n;
        while(m<=n){
            if(!isPrime[m]){
                ++sum;
              //  cout<<m<<endl;
            }
            ++m;
        }
        sum=(sum==0)?-1:sum;
        cout<<sum<<endl;
	}
	/*for(t=0; t<40; t++){
        cout<<t<<"\t"<<isPrime[t]<<endl;
	}*/
	return 0;
}
