#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll power(ll a,ll b,ll MOD)
{
	if(b==0)
	  return 1;
	if(a==0)
	 return 0;
	ll res=power(a,b/2,MOD);
	if(b%2)
	 return ((a*res)%MOD*res)%MOD;
	 else
	 return (res*res)%MOD;
}

int main()
{
  // freopen("in.txt","r",stdin);
    int t;
    cin>>t;
    while(t--)
    {
           set<ll int> s;
       ll n,x;
       cin>>n;
       for(int i=0;i<n;i++) 
	   {
	     cin>>x;
         s.insert(x);
         
	  }
         ll res=power(n,s.size(),1000000000+7);
         cout<<res<<endl;
       s.clear();
       
    }

    return 0;
}
