#include <iostream>
using namespace std;
#include <algorithm>
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long k;
        cin>>k;
        string s;
        cin>>s;
        for(int i=0; i<s.size(); ++i)
        {
            if(s.at(i)<=90) //CAP
            {
                long long  d = s.at(i) - 'A';
                 d = d+k;
                 d = d%26;
                 cout<<char('A' + d);
            }
            else
            {
                long long  d = s.at(i) - 'a';
                 d = d+k;
                 d = d%26;
                 cout<<char('a' + d);

            }
        }
        cout<<endl;
    }
}
