#include <iostream>
using namespace std;
int main()
{
	long int a[100000],n;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
	}
	int max_so_far = 0, max_ending_here = 0;
 
    for (int i = 0; i <n; i++)
    {
        max_ending_here = max_ending_here + a[i];
        if (max_ending_here < 0)
            max_ending_here = 0;
        if (max_so_far < max_ending_here)
            max_so_far = max_ending_here;
    } 
    cout<<max_so_far;
	return 0;
}
