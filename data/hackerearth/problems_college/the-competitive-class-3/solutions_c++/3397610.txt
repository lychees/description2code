#include <iostream>
#include <algorithm>
#define ll long long
using namespace std;

int main()
{
	ll t, pos, x, i, j, n;
    	pos=1;
    	x=0;
    	cin>>n;
    	ll a[n], b[n], p[n];
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    		b[i]=a[i];
    		p[i]=0;
    	}
    	sort(a,a+n);
    	for(i=n-1;i>=0;i--)
    	{
    		x=0;
    		for(j=0;j<n;j++)
    		{
    			if(a[i]==b[j])
    			{
    				p[j]=pos;
    				x++;
    			}
    		}
    		pos+=x;
    		i=i-x+1;
    	}
    	for(i=0;i<n;i++)
    		cout<<p[i]<<' ';
    	cout<<endl;
    return 0;
}
