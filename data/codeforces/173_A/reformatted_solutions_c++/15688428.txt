#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstring>
#include <iostream>
#include <map>
#include <queue>
#include <set>
#include <string>
#include <vector>
using namespace std;
string x, y;
int mk, n, cx, cy, sx, sy, kx, ky, tims;
bool play(char a, char b) {
  if (a == 'R' && b == 'P')
    return 1;
  if (a == 'P' && b == 'S')
    return 1;
  if (a == 'S' && b == 'R')
    return 1;
  return 0;
}
int main() {
  cin >> n >> x >> y;
  sx = x.size();
  sy = y.size();
  mk = sx * sy;
  for (int i = 0; i < mk && i < n; i++) {
    tims = 1 + (n - i - 1) / mk;
    cx += play(x[i % sx], y[i % sy]) * tims;
    cy += play(y[i % sy], x[i % sx]) * tims;
  }
  cout << cx << " " << cy;
  return 0;
}
