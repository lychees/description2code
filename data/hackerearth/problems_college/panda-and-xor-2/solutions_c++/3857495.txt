int main()
{ long int a,b,c,n,i;
   scanf("%ld",&n);
   while(n)
   {
   	i=0;
   	 scanf("%ld%ld",&a,&b);
   	 c=a^b;
   	 while(c)
   	 {
   	 	c=c&(c-1);
   	 	i++;
   	 }
   	 if(i%2)
   	 printf("NO\n");
   	 else
   	printf("YES\n");
   	 n--;
   	
   }
    return 0;
}
