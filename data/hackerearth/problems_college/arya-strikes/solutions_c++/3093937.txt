#include <iostream>
using namespace std;

int main()
{
    int i,j,k,n,m,p,t ;
    cin >> t ;
    while(t--) {
    	cin >> n ;
    	bool a[1001][1001] ;
    	bool dp[1001][1001] = {0} ;
    	for(i=0;i<n;i++) {
    		for(j=0;j<n;j++) cin >> a[i][j] ;
    	}
    	dp[0][0] = 1 ;
    	p = 1 ;
    	for(i=1;i<n;i++) {
    		if(dp[0][i-1] == 1 && p == 1 && a[0][i] == 1) dp[0][i] = 1 ;
    		else {
    			dp[0][i] = 0 ;
    			p = 0 ;
    		}
    	}
    	p = 1 ;
    	for(i=1;i<n;i++) {
    		if(dp[i-1][0] == 1 && p == 1 && a[i][0] ) dp[i][0] = 1 ;
    		else {
    			dp[i][0] = 0 ;
    			p = 0 ;
    		}
    	}
    	for(i=1;i<n;i++) {
    		for(j=1;j<n;j++) {
    			if((dp[i-1][j] || dp[i][j-1]) && a[i][j] == 1 ) dp[i][j] =1 ;
    			else dp[i][j] = 0 ;
    		}
    	}
    	if(dp[n-1][n-1] == 1) cout << "POSSIBLE\n" ;
    	else cout << "NOT POSSIBLE\n" ;
    	
    }
    return 0;
}
