#include <iostream>
using namespace std;
int cubes[1001];
int lsum[1001][1001],final[1001][1001];
int grid[1001][1001];
int search(int n){
	int l=0,r=1000;
	int mid;
	while(l<=r){
		mid=l+(r-l)/2;
		if(cubes[mid]==n) return mid;
		if(cubes[mid]<n) l=mid+1;
		else r=mid-1;
	}
	return -1;
}
int main()
{
    //cout << "Hello World!" << endl;
    for(int i=0;i<1001;i++) cubes[i]=i*i*i;
    int n,k,ans=0;
    cin>>n>>k;
    for(int i=0;i<n;i++){
    	for(int j=0;j<n;j++)
    		cin>>grid[i][j];
    }
    
    for(int i=0;i<n;i++){
    	for(int j=0;j<n;j++){
    		lsum[i][j]=grid[i][j];
    		if(j>0) lsum[i][j]+=lsum[i][j-1];
    		if(j>=k) lsum[i][j]-=grid[i][j-k];
    	}
    }
    for(int i=0;i<n;i++){
    	for(int j=0;j<n;j++){
    		final[i][j]=lsum[i][j];
    		if(i>0) final[i][j]+=final[i-1][j];
    		if(i>=k) final[i][j]-=lsum[i-k][j];
    	}
    }
    for(int i=k-1;i<n;i++){
    	for(int j=k-1;j<n;j++){
    		if(search(final[i][j])>=0) ans++;
    	}
    }
    cout<<ans;
    return 0;
}
