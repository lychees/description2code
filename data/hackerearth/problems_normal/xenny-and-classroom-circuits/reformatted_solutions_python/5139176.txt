for _ in range(input()):
    n = input()
    b = raw_input()
    s1 = ['G']*(2*n)
    s2 = ['G']*(2*n)
    j = count1 = count2 = 0
    for i in xrange(1, 2*n, 2):
        s1[i] = 'B'
        s2[j] = 'B'
        j = j+2
    for i in xrange(2*n):
        if b[i] != s1[i]:
            count1 += 1
        if b[i] != s2[i]:
            count2 += 1
    print min(count1/2, count2/2)
