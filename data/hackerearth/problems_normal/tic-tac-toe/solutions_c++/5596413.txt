#include <iostream>
using namespace std;

int main(){
	int t;
	cin >> t;
	while(t--){
		long long N;
		long long m = 1e9 + 7;
		cin >> N;
		long long minx = N % m;
		minx = (minx * ((N - 1)/2)) % m;
		minx = (minx * ((N - 1)/2)) % m;
		long long a = N;
		long long b = (N - 1)/2;
		long long c = (2 * N - 1);
		if(a % 3 == 0) a /= 3;
		else if(b % 3 == 0) b /= 3;
		else if(c % 3 == 0) c /= 3;
		long long maxx = (((a * b) % m) * c) % m;
		cout << minx << " " << maxx << "\n";
	}
    return 0;
}
