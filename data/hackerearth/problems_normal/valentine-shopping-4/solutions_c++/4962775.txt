#include<bits/stdc++.h>
using namespace std;
#define eps 1e-9
#define ld long double
int main()
{
	int tc;
	cin>>tc;
	while(tc--)
	{
		int n,m,ans,j;
		cin>>n>>m;
		for(int i=0;i<n;i++)
		{
			ld mi=99999999.000;
			for(j=0;j<m;j++)
			{
				ld a,b,c,x,y,z;
				cin>>a>>b>>c;
				x=(ld)(1-(ld)a/100);
				y=(ld)(1-(ld)(b)/100);
				z=(ld)(1-(ld)(c)/100);
				ld anss = 1000*x*y*z;
			//	cout<<x<<" "<<y<<" "<<z<<endl;
				if(mi-anss>eps)
				{mi=anss;
				ans=j+1;
				}
				
			}
			cout<<ans<<" ";
		}
		cout<<endl;
		//v.clear();
	}
return 0;
}
