l = int(raw_input())
if l != 0:
    arr = [int(el) for el in raw_input().split()]
    final_sum = arr[0]
    local_sum = 0
    for a in arr:
        local_sum = local_sum+a
        if local_sum > final_sum:
            final_sum = local_sum
        if local_sum < 0:
            local_sum = 0
    print final_sum
else:
    print l
