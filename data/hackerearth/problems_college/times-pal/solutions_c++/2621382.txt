#include <iostream>
using namespace std;
bool isP(string s)
{
	int size=s.length();
	int l=0,r=size-1;
	while(l<r)
	{
		if(s[l]!=s[r])
		return false;
		l++; r--;
	}
	return true;
}
int main()
{
    int t; cin>> t;
    while(t--)
    {
    	string s,in;
    	cin>>s>>in;
    	string temp;
    	int res=0;
    	temp=s+in;
    	
    	for(int i=0;i<=s.length();i++)
    	{
    		temp=s;
    		temp.insert(i,in);
    		//cout<<temp<<"  ";
    		if(isP(temp))
    		res++;
    	}
    	cout<<res<<endl;
    }
    return 0;
}
