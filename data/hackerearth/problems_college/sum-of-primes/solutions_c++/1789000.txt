#include <iostream>
#include<math.h>
using namespace std;
typedef unsigned long long int ll;
int test_prime(ll numb)
{
	for(ll i=2;i<=sqrt(numb);i++)
	{
		if(numb%i==0)
			return 0;
	}
	return 1;
}
int main()
{
    bool p[1000001];
    p[0] = false;
    p[1] = false;
    for(ll i=2;i<1000001;i++) p[i] = true;
    for(ll i=2;i<1000;i++)
    	if(p[i]==true)
    		for(ll j=i;i*j<1000001;j++)
    			p[i*j] = false;
   	ll t = 0, a = 0 , b = 0 , temp = 0;
   	cin>>t;
   	while(t--)
   	{
   		cin>>a>>b;
   		temp = 0;
   		for(ll i=a;i<=b;i++)
   		{
   			if(p[i]==true)
   				temp+=i;
   		}
   		if(test_prime(temp) && temp!=0)
   			cout<<"YES"<<endl;
   		else
   			cout<<"NO"<<endl;
   	}
    return 0;
}
