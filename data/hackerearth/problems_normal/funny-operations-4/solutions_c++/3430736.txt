#include <iostream>
#include<vector>
using namespace std;

int main()
{
    int t;
	cin>>t;
	for(int i =0;i<t;i++)
	{
		int n;
		cin>>n;
		vector<long long> a;
		for(int j =0; j<n;j++)
		{
			long long temp;
			cin>>temp;
			a.push_back(temp);
		}
		long long x =0;
		long long m =0;
		long long p =0;
		int ne = 0;
		int num = 0;
		for(int j=0; j<(1<<n); j++)
		{
			bool first = true;
			for(int k =0; k<n;k++)
			{
				if(j & (1<<k))
				{
					if(first)
					{
						x = a[k];
						m = a[k];
						first = false;
						ne=1;
					}
					else
					{
						x = x ^ a[k];
						m = m & a[k];
						ne++;
					}
				}
			}
			long long te = x|m;
			//cout<<te<<endl<<endl;
			if((te) > p)
			{
				p = te;
				num = ne;
			}
			if(te == p && ne > num)
			{
				num = ne;
			}
		}
		
		if(i!=t-1)
			cout<<p<<" "<<num<<endl;
		else
			cout<<p<<" "<<num;
	}
    return 0;
}
