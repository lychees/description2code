#include <iostream>
using namespace std;

int main()
{
       long int n,m,sum=0,temp;
       	long int t1=0,t2=1,t3;
       cin>>n>>m;
       
       for(int i=0;i<m;i++){
       		cin>>temp;
       		sum=sum+temp;
       }
       	temp=1;
		for(int i=2;i<=n;i++){
			t3=t1+t2;
			temp+=t3;
			t1=t2;
			t2=t3;
		}
		cout<<(temp-sum);
}
