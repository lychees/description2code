// Example program
#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

int main()
{
    int size;
    cin>>size;    
    unsigned long long maxcount = 0;
    unsigned long long contcount = 0;
    unsigned long long maxcontcount = 0;
    for(int i = 0; i<size; i++)
    {
        string str;
        cin>>str;
        unsigned long long count = 0;
        bool capture = false;
        for(int j = 0; j<str.length(); j++)
        {
            if(str[j] == 'C')
            {                
                    count++;
                    contcount ++;
                    if(!capture)
                        capture = true;
            }
            else
            {
            	
                if(capture)
                {
                	maxcount = max(maxcount, count);
                	maxcontcount = max(maxcontcount, contcount);
                    capture= false;
                }
                count = 0;
                contcount = 0;
                maxcontcount = max(maxcontcount, contcount);
            }
            //cout<<count<<"\n";
        }
        maxcount = max(maxcount, count);
        maxcontcount = max(maxcontcount, contcount);
    }
    cout<<maxcount<<" "<<maxcontcount<<"\n";
    return 0;
}
