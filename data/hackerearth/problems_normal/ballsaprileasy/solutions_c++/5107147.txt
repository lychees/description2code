#include<bits/stdc++.h>
using namespace std;
 
int main()
{
    int n;
    int i;
    scanf("%d",&n);
    int A[n+1];
    for(i=1;i<=n;i++)
        scanf("%d",&A[i]);
    sort(A+1,A+n+1);
    long long int ans=0;
    long long int sq;
    long long int j;
    j=sqrt(A[1]-1);
    long long int t=(j*(j+1)*(2*j+1))/6;
    ans=ans+t;
    j=j+1;
    i=1;
    while(i<=n)
    {
        sq=j*j;
        if(sq>=A[i])
        {
            ans=ans+sq-A[i];
            i++;
        }
        else
            ans=ans+sq;
        j++;
    }
    printf("%lld",ans);
}
