test=int(input())
for i in range(0,test):
	k,l = raw_input().split(' ')
	k=int(k)
	l=int(l)
	d, c, r = {0 : 1}, 0, 0
	e,f,g={0:1},0,0
	for x in raw_input():
		c += x == 'R'
		f += x == 'K'
		r += d.get(c - k, 0)
		g += e.get(f-l,0)
		d[c] = d.get(c, 0) + 1
		e[f]=e.get(f,0)+1
	print (str(r)+" "+str(g))
