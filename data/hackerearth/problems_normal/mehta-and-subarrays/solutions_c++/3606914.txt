using namespace std;
#define MOD 1000000007
long long a[1111111];
int main(void){
	std::ios::sync_with_stdio(false);cin.tie(0);
	int i,n;
	int ll,rr,mm;
	int re=-1,ren=0,num;
	long long sum;
	cin>>n;
	for(i=0;i<n;i++)
		cin>>a[i];
	ll=1;rr=n;
	while(ll<=rr)
	{
		sum=0;
		num=0;
		mm=(ll+rr)/2;
		for(i=0;i<mm;i++)
			sum+=a[i];
		if(sum>=0)
			num++;
		for(;i<n;i++)
		{
			sum+=a[i]-a[i-mm];
			if(sum>=0)
				num++;
		}
		if(num)
		{
			re=mm;
			ll=mm+1;
			ren=num;
		}
		else
		{
			rr=mm-1;
		}
	}
	if(re==-1)
		cout<<re<<endl;
	else
		cout<<re<<' '<<ren<<endl;
	//system("pause");
	return 0;
}
