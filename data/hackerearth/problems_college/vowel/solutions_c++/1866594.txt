#include<iostream>
#include<math.h>
#include<string.h>
using namespace std;
int main()
{
   int t;
   cin>>t;
   while(t--)
   {
       char c[100];
       cin>>c;
       int l=strlen(c);
       int count=0;
       for(int i=0;i<l;i++)
       {

           if(c[i]=='A' || c[i]=='a' || c[i]=='e' || c[i]=='E' ||c[i]=='I' || c[i]=='i' ||c[i]=='O' || c[i]=='o' || c[i]=='u' || c[i]=='U')
           {
                cout<<c[i];
                count++;
           }
       }

       if(count==0)
            cout<<"No";
        cout<<endl;
   }
   return 0;


}
