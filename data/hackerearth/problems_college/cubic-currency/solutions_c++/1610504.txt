using namespace std;

int main()
{
    lli t,n, x, y;
    lli dp[100009]={0};

    int S[25];
    for(int i=1; i<25; i++) S[i-1]=(i*i*i);

    //memset(dp, 0, sizeof(dp));
    dp[0]=1;

    for(int i=0; i<21; i++)
    {
        for(int j=S[i]; j<=10009; j++)
        {
            dp[j] += dp[j-S[i]];
        }
    }
    cin>>t;
    while(t-->0)
    {
        scanf("%d",&n);
        printf("%lld\n",dp[n]);
    }

}
