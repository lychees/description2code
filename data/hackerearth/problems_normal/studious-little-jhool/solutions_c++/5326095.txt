#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    for(int f=0;f<t;f++)
    {
    	int n;
    	cin>>n;
    	int *a=new int[n+1];
    	a[0]=0;
    	for(int i=1;i<=9;i++)
    	{
    		a[i]=-1;
    	}
    	a[10]=1;
    	a[11]=-1;
    	a[12]=1;
    	for(int i=13;i<=n;i++)
    	{
    		if(a[i-10]==-1 && a[i-12]==-1)
    		   a[i]=-1;
    		else if(a[i-10]==-1 && a[i-12]!=-1)
    		   a[i]=a[i-12]+1;
    		else if(a[i-10]!=-1 && a[i-12]==-1)
    		   a[i]=a[i-10]+1;
    		else if(a[i-10]!=-1 && a[i-12]!=-1)
    		   a[i]=min(a[i-12],a[i-10])+1;
    	}
    	cout<<a[n]<<endl;
    }
    return 0;
}
