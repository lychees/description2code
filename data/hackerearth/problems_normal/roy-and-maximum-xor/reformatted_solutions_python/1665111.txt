import math
t = input()
for _ in xrange(t):
    n = input()
    m = 2**int(math.floor(math.log(n, 2)))
    c = 1
    if n != m:
        c = 2*(n-m)
        m = (2*m)-1

    else:
        m -= 1
        c = n
    print m, c
