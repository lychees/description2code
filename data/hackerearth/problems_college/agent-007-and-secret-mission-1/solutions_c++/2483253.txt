#include <iostream>
#include <string>
using namespace std;

int main()
{
	int T;
	cin>>T;
	while(T--)
	{
		string s;
		int N;
		cin>>s;
		cin>>N;
		int c0=0,c1=0;
		bool odd = false;
		for(int i = 0;i < s.length();i++)
		{
			if(s[i] == '0')
				c0++;
			else
				c1++;
		}
		if(N%2 == 0)
		{
			N /= 2;
			if(N%2 == 0)
				odd = false;
			else
				odd = true;
		}
		else
		{
			N = (N+1)/2;
			if(N%2 == 0)
				odd = false;
			else
				odd = true;
		}
		if(odd)
			cout<<c0<<endl;
		else
			cout<<c1<<endl;
	}
    return 0;
}
