#include <bits/stdc++.h>
using namespace std;

int main()
{
	list<int> x;
    int n,a,l=INT_MAX;
    cin>>n;
    
    while(n--)
    {
    	cin>>a;
    	x.push_back(a);
    	if(l>a) l=a;
    }
    int len=x.size();
    while(len)
    {
    	cout<<len<<endl;
    	while(l)
    	{
    		x.remove(l);
    		--l;
    	}
    	
    	len=x.size();
    	l=INT_MAX;
    	for(auto & b:x)
    	 if(b<l) l=b;	
    }
    
    return 0;
}
