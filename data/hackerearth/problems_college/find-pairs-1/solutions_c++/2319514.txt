#include <iostream>
using namespace std;
 
int main()
{
    int tc;
    cin>>tc;
    long int arr[1000001],count;
    while(tc--)
    {
    	count=0;
    	long int num;
    	cin>>num;
    	for(long int i=0;i<num;i++)
    	{
    		cin>>arr[i];
    	}
    	for(long int i=0;i<num;i++)
    	{
    	for(long int j=i;j<num;j++)
    	{
    		if(arr[i]==arr[j])
    		{
    			count++;
    		}
    	}
    	}
    	cout<<count<<"\n";
    }
}
