#include <iostream>
#include<string>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	long int n=s.length(),i;
    	long int count=0,sum=0;
    	for(i=0;i<n;i++)
    	{
    if(s[i]=='a'||s[i]=='z')
    {
    	sum=sum+(count*(count+1))/2;
    	count=0;
    }
    else
    count++;
    	}
    	sum=sum+(count*(count+1))/2;
    	cout<<(n*(n+1))/2-sum<<endl;
    
    }
    return 0;
}
