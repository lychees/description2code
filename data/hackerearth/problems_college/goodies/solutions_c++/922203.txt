#include <bits/stdc++.h>
using namespace std;


int *l;

int main()
{
	int t;
	cin >> t;
	while (t > 0){
		int n;
		cin >> n;
		l = new int[n];
		for (int i = 0; i < n; i++){
			cin >> l[i];
		}
		sort(l,l+n);
		int max = 0;
		int local = 0;
		for (int i = 0; i < n-1; i++){
			if (l[i] == l[i+1]){
				local++;
			}
			else{
				if (local > max)
					max = local;
				local = 0;
			}
		}
		if (local > max)
			max = local;
		cout << max+1 << endl;
		t--;
	}     
	return 0;
}
