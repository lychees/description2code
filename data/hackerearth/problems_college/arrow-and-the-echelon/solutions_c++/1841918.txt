#include <iostream>
using namespace std;

int main()
{
    int t;
    scanf("%d\n",&t);
    while(t--)
    {
    	long long int x,count=0,i,d,flag=0;
    	scanf("%lld\n",&x);
    	if(x%3==0)
    	{
    		for(i=0;i<x;i++)
    		{
    			printf("%d",5);
    			flag=1;
    		}
    		printf("\n");
    	}
    	else
    	{
    		d=x/3;
    		for(i=0;i<d;i++)
    		{
    			if((x-((d-i)*3))%5==0)
    			{
    				count=3*(d-i);
    				flag=1;
    				break;
    			}
    		}
    		if(flag==1)
    		{
    			for(i=0;i<count;i++)
    			printf("%d",5);
    			for(i=0;i<x-count;i++)
    			printf("%d",3);
    			printf("\n");
    		}
    	}
    	if(flag==0)
    	{
    		if(x==5||x==10)
    		{
    			for(i=0;i<x;i++)
    			printf("%d",3);
    			printf("\n");
    		}
    		else
    		printf("%d\n",-1);
    	}
    	
    }
}
