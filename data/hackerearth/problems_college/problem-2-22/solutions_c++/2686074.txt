#include<iostream>
#include<cmath>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int r,xc,yc;
		cin>>r>>xc>>yc;
		int i,a[4],b[4],s;
		int c=0;
		for(i=0;i<4;i++)
		{
			cin>>a[i]>>b[i];
			s=pow((a[i]-xc),2)+pow((b[i]-yc),2)-pow(r,2);
			if(s<=0)
			{
				c++;
			}
		}
		if(c==4)
		{
			cout<<"Yes"<<"\n";
		}
		else
		{
			cout<<"No"<<"\n";
		}
	}
}
