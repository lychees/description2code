#include<bits/stdc++.h>
using namespace std;

map<long long, int> memo;

#define MOD 1000000007

long long pow(long long x,long long y)
{
      if(y==0)
        return 1;

    long long temp = pow(x,y/2);

    if(y&1)
        return (x*((temp*temp)%MOD))%MOD;
    else
        return (temp*temp)%MOD;
}

int main()
{
    long long t;
    cin>>t;

    while(t--){
        long long n;
        cin>>n;
        long long dis = 0;
        long long x;
        memo.clear();
        for(long long i=0;i<n;i++){
            cin>>x;
            if(memo.find(x)==memo.end()){
                memo[x] = 1;
                dis++;
            }
        }


       // cout<<dis<<" "<<n<<endl;
        cout<<pow(n,dis)<<endl;
    }

    return 0;
}
