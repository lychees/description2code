#include<iostream>
#include<stdlib.h>
#include<stdio.h>
using namespace std;
int arr[10000]={0};
int has[10000]={0};
void init()
{
	arr[0]=1;
	int k=2,count=1;
	for(int i=1;i<1000;i++)
	{
		arr[i]=arr[i-1]+k;
		k++;
	}
	for(int i=1;i<1000;i++,count=1)
	{
		for(int j=2;j<=arr[i]/2;j++)
		{
			if(arr[i]%j==0)
				count++;
			if(has[count]==0)
			{
				has[count]=arr[i];
			}
		}
	}
	
}

int main()
{
	init();
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		if(n==1)
		{
			cout<<"3"<<endl;
			continue;
		}
		cout<<has[n]<<endl;
	}
}
