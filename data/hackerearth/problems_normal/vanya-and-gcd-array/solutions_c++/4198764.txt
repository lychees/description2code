const int N = 500+5;
int n;
int a[N];
int dp[N][102][102];
LL rec(int pos,int last,int g) {
  if(pos == n) {
    return (g==1);
  }
  if(dp[pos][last][g]!=-1)
    return dp[pos][last][g];
  LL ans = 0;
  if(a[pos] > last) {
    ans += rec(pos+1,a[pos],__gcd(g,a[pos]));
    ans %= MOD;
  }
  ans += rec(pos+1,last,g);
  ans %=MOD;
  return dp[pos][last][g] = ans;
}
int main() {
  M(dp,-1);
  cin >> n;
  F(i,0,n) {
    sci(a[i]);
  }
  cout<<rec(0,0,0)<<endl;
}
