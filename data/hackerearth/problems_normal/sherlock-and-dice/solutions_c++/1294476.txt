#include<bits/stdc++.h>
using namespace std;

typedef long long ll;

double sum[55][2505];

int main() {
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        int m,n,k;
        scanf("%d%d%d",&m,&n,&k);
        memset(sum,0,sizeof(sum));
        if(k>m*n) {
            puts("0.000 0");
            continue;
        } else {
            for(int i=1;i<=n&&i<=k;i++) {
                sum[1][i]=1.0/n;
            }
            for(int i=2;i<=m;i++) {
                for(int j=1;j<=n*m&&j<=k;j++) {
                    for(int l=1;l<=n&&l<=j;l++) {
                        sum[i][j]+=sum[i-1][j-l];
                    }
                    sum[i][j]/=n;
                }
            }
        }
        int tens=0;
        if(sum[m][k]==0.0) {
            puts("0.000 0");
        } else {
            while(sum[m][k]<1.0) {
                sum[m][k]*=10;
                tens+=1;
            }
            printf("%.3lf %d\012",sum[m][k],tens);
        }
    }
    return 0;
}
