#include<iostream>
#include<cmath>
using namespace std;
long long int val(long long int x);

int main(){
    long long int n,i,j,temp,num;
    long long int prime[501],res=0;
    prime[0]=0;
    prime[1]=0;
    for(i=2;i<501;i++)
        prime[i]=1;
    for(i=2;i<=250;i++){
        if(prime[i]!=0){
                temp=501/i;
            for(j=2;j<=temp;j++){
                prime[i*j]=0;
            }
        }
    }
    num=0;
    for(i=0;i<501;i++){
        if(prime[i]!=0)
            num++;
        prime[i]=num;
    }
    for(i=2;i<501;i++)
        prime[i]=(val(prime[i]))/2;

    cin>>n;
    for(i=0;i<=n;i++)
        res=res+prime[i];
    cout<<res<<endl;
    return 0;
}

long long int val(long long int x){
    long long int m=2,a=1,temp;
    while(1){
        temp=x%2;
        x=x/2;
        if(temp==1)
            a=a*m;
        m=m*m;
        if(x==0)
            break;
    }
    return a;
}
