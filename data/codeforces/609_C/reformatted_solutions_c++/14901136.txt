#include <algorithm>
#include <conio.h>
#include <iostream>
#include <math.h>
#include <string.h>
#include <string>
using namespace std;

long long q, h, s, n, m[100001];
int main() {
  int i;
  cin >> n;
  for (i = 0; i < n; i++) {
    cin >> m[i];
    s += m[i];
  }
  q = s % n;
  s /= n;
  for (i = 0; i < n; i++) {
    if (m[i] - s > 0) {
      if (q == 0)
        h += (m[i] - s);
      else {
        h += m[i] - s - 1;
        q--;
      }
    }
  }

  cout << h;

  return 0;
}
