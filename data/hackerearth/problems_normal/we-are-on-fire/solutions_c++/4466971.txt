#include<bits/stdc++.h>
#include<algorithm>
using namespace std;
#define fr(i,a,b) for(int i=a;i<b;i++)
#define frn(i,a,b) for(int i=a;i>b;i--)
#define si(x) scanf("%d",&x)
#define sll(x)  scanf("%lld",&x)
#define pln printf("\n")
#define pi(x) printf("%d",x)
#define pll(x) printf("%lld",x)
#define mp make_pair
#define pb push_back
#define x first
#define y second
typedef long long ll;
ll MOD=1000000007;
ll pw(ll a,ll b){
  ll r=1;
  while(b>0){
     if(b&1){
       r=(r*a)%MOD;
        }
     b=b>>1;
     a=(a*a)%MOD;
  }
return (r)%MOD;
}
int n,m;
int a[1001][1001];
int vis[1001][1001]={{}};
int tt=0,burnt=0;
void dfs(int r,int c){
  if(r<0||r>=n||c<0||c>=m)return;
  if(!a[r][c]||vis[r][c])return;
   vis[r][c]=1;
   burnt++;
   dfs(r+1,c);
   dfs(r-1,c);
   dfs(r,c+1);
   dfs(r,c-1);
}


int main(){
  int q;
  si(n),si(m),si(q);
  fr(i,0,n){
     fr(j,0,m){
        si(a[i][j]);
        if(a[i][j])tt++;
     }
  }
  int r,c;
  while(q--){
      si(r),si(c);
      dfs(r-1,c-1);
    pi(tt-burnt);pln;
  }

return 0;
}
