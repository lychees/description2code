#include <iostream>
using namespace std;

int main()
{
    int t,n,i,j;
    int A[109];
    cin>>t;
    while(t--){
    	cin>>n;
    	long sum=0;
    	for(i=0;i<n;i++){
    		cin>>A[i];
    		sum+=A[i];
    	}
    	if(sum%2)
    		cout<<"NO\n";
    	else{
    		sum/=2;
    		int S[sum+1][n+1];
    		for(i=0;i<=n;i++)
    			S[0][i]=1;
    		for(i=1;i<=sum;i++)
    			S[i][0]=0;
    		for(i=1;i<=sum;i++){
    			for(j=1;j<=n;j++){
    				S[i][j]=S[i][j-1];
    				if(i>=A[j-1]){
    					S[i][j]=S[i][j]||S[i-A[j-1]][j-1];
    				}
    			}
    		}
    		if(S[sum][n])
    			cout<<"YES\n";
    		else
    			cout<<"NO\n";
    	}
    }
    return 0;
}
