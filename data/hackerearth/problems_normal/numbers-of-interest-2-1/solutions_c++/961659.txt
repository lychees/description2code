#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
 
using namespace std;
long long f[1000002];
long long g[1000002];
long long y[1000002];
 
void multiple(int num){
 
    int i=2;
    while(num*i<=1000002){
            if(f[num*i]==0) 
                f[num*i]=num;
 
        i++;
    }
}
 
void checkForPrime(){
 
    for(int i=2;i<1000002;i++){
 
        if(f[i]==0){    //prime
 
            f[i]=f[i-1]+i;
            multiple(i);
        }
        else{
 
            f[i]+=f[i-1];
        }
    }
}
 
void calcSum(){
 
    for(int i=1;i<=(1000002/2);i++){
 
        for(int j=i+i;j<1000002;j+=i){
                y[j]+=i;
        }
    }
 
    for(int i=1;i<1000002;i++)
        g[i]=g[i-1]+y[i];
 
}
 
int main(){
    int T,n;
 
    checkForPrime();
    calcSum();
 
    scanf("%d",&T);
 
    while(T--){
        //cout<<"enter : ";
        scanf("%d",&n);
        //printf("%lld %lld\n",f[n],g[n]);
        printf("%lld\n",(f[n]+g[n])%n);
    }
 
 
    return 0;
}
