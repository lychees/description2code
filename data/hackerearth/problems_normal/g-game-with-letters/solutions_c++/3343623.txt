#include <bits/stdc++.h>

using namespace std;


int main() {
 	int T;
	cin >> T;
	while (T--) {
		int n;
		string s;
		cin >> n >> s;
		vector<int> cnt(26);
		for (char c : s) {
			cnt[c - 'a']++;
		}
		int maxScore = -1;
		int winner = 0;
		vector<int> teamSizes;
		for (int i = 0; i < n; i++) {
			string t;
			cin >> t;
			teamSizes.push_back(t.size());
			int score = 0;
			for (char c : t) {
				score += cnt[c - 'a'];
			}	
			if (score > maxScore) {
				maxScore = score;
				winner = i;
			} else if (score == maxScore) {
				if (teamSizes[i] < teamSizes[winner]) {
					winner = i;
				}
			}
		}
		cout << winner + 1 << endl;
	}

	return 0;
}
