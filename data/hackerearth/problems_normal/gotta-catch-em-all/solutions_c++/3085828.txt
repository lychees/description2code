#include <iostream>
#include <vector>
#include <cstdio>
#include <algorithm>
 
#define SIZE 1000001
 
using namespace std;
int main() 
{
	long long int n,i,oval,itr,no,max;
	long long int val[SIZE]={0};
	scanf("%lld",&n);
	max=0;
	for(i=0;i<n;i++)
	{
		scanf("%lld",&no);
		val[no]++;
		if(no>max)
			max=no;
	}
	itr=0;
	oval=0;
	for(i=max;i>=1;i--)
	{
		if(val[i]>0)
		{
			itr+=val[i];
			oval=(oval>itr+i)?oval:itr+i;
		}
	}
	printf("%lld",oval+1);
	return 0;
}
