#include <iostream>
using namespace std;
long int count(long int n)
{
  long  int count = 0;
  while(n)
  {
    count += n & 1;
    n >>= 1;
  }
  return count;
}
int main()
{
    int t;
    cin>>t;
    long int a,b,c;
    while(t--)
    {
    	cin>>a>>b;
    	c=a^b;
    	if(count(c)%2==0)
    	cout<<"YES"<<endl;
    	else
    	cout<<"NO"<<endl;
    }
    return 0;
}
