#include <iostream>
using namespace std;

int main()
{
   long long int n,q,d;
   cin>>n;
   int a[n];
   for(int i=0;i<n;i++)
   cin>>a[i];
   cin>>q;
   long long res=1;
   while(q--)
   {
   	cin>>d;
   	res*=d;
   }
   for(int i=0;i<n;i++)
   {
   	
   	if(res==0)
   	a[i]=0;
   	else
   	a[i]/=res;
   	cout<<a[i]<<" ";
   }
    return 0;
}
