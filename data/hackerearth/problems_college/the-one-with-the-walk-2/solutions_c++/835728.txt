#include<iostream>
#include<vector>
#define ll long long
using namespace std;
 
#define MOD 1000000007


ll dp[1007][1007]={0};

int main()
{
    dp[0][0]=1;
	for(int i=1;i<1007;i++) 
	{
	    dp[i][0]=1,dp[0][i]=1;
	}
	for(int i=1;i<1007;i++)
	{
		for(int j=1;j<1007;j++)
		{
			dp[i][j]=dp[i-1][j] + dp[i][j-1] + dp[i-1][j-1];
			dp[i][j]%=MOD;
		}
	}
	/*
	for(int i=0;i<10;i++)
	{
		for(int j=0;j<10;j++)
		{
			cout<<dp[i][j]<<" ";
		}
		cout<<"\n";
	}
	*/
    ll a,b,c,d;
    ll t;
    cin>>t;
    while(t--)
    {
        cin>>a>>b>>c>>d;
        cout<<dp[c-a][d-b]<<"\n";
    }
}
