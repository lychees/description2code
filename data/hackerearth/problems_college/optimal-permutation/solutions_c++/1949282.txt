#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main(void)
{
	ios::sync_with_stdio(false);
	ll tc,n,ans;
	cin >> tc;
	while(tc--)
	{
		cin >> n;
		if(n%2)
			ans = (n-1)*(n-1)/2 + (n-2);
		else
			ans = n*(n-2)/2 + (n-1);
		if(n == 1)
			ans = 0;
		cout << ans << '\n';
	}
	return 0;
}
