#include<iostream>
#include<algorithm>
#include<functional>
using namespace std;
int main()
{
	long long int n,i,j;
	cin>>n;
	long long int arr[n],max=-1,ans=0;
	for(i=0;i<n;i++)
	{
		cin>>arr[i];
		if(max<arr[i])
		max=arr[i];
	}
	long long int brr[max+1];
	for(i=0;i<=max;i++)
	brr[i]=0;
	for(i=0;i<n;i++)
	brr[arr[i]]++;
	for(i=0;i<=max;i++)
	{
		if(brr[i]!=0)
		ans++;
	}	
	cout<<ans*(ans-1)/2;
}
