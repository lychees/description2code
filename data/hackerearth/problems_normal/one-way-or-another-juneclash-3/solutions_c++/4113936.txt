int main()
{
	IOS;
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		ll x[n],y[n],value[n];
		bool flag=true;
		rep(i,n)cin>>x[i]>>y[i]>>value[i];
		ll a=x[0],b=y[0],v=value[0];
		int res;
		FOR(i,1,n-1)
		{
			res=(v&1)?1:0;
			if(a&1 and x[i]&1 or !(a&1) and !(x[i]&1))
			{
				if (b&1 and y[i]&1 or !(b&1) and !(y[i]&1))
				{
					if(value[i]%2!=res)
					{
						flag=false;
						break;
					}
				}
				else
				{
					res=!res;
					if(value[i]%2!=res)
					{
						flag=false;
						break;
					}
				}
			}
			else
			{
				res=!res;
				if (b&1 and y[i]&1 or !(b&1) and !(y[i]&1))
				{
					if(value[i]%2!=res)
					{
						flag=false;
						break;
					}
				}
				else
				{
					res=!res;
					if(value[i]%2!=res)
					{
						flag=false;
						break;
					}
				}
			}
			if(!flag)break;
		}
		cout<<((flag)?"Yes\n":"No\n");
	}
	return 0;
}
