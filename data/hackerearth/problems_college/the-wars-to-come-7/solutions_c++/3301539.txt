#include <bits/stdc++.h>

using namespace std;

int main()
{
	int n;
	scanf("%d", &n);

	vector<int> v(n);
	for (int i=0; i<n; i++) {
		cin >> v[i];
	}

	int m;
	scanf("%d", &m);
	
	sort(v.rbegin(), v.rend());

	int sum=0;
	int i;
	for (i=0; i<n; i++) {
		sum += v[i];
		if (sum >= m) {
			break;
		}
	}

	printf("%d\n", i+1);

	return 0;
}
