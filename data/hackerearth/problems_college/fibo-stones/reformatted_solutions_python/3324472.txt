def fib(n):
    l = []
    n1 = 0
    n2 = 1
    n3 = 0
    l.append(n2)
    for i in range(n-1):
        n3 = n1+n2
        l.append(n3)
        n1 = n2
        n2 = n3
    return l


def mario(n, m, l):
    p = []
    p = fib(n)
    q = []
    for i in p:
        if i not in l:
            q.append(i)
    if l.count(1) == 1:
        q.append(1)

    print sum(q)
