#include<bits/stdc++.h>
using namespace std;
unsigned long long int arr[1000000];

int findnumberofzero(int n)
{
	int count=0;
	for(int i=5;n/i>=1;i=i*5)
	{
		count+=n/i;
	}
	return count;
}

void solve()
{
	arr[2]=0;
	arr[3]=0;
	arr[4]=0;
	for(int i=3;i<=1000000;i++)
	{
	//cout<<findnumberofzero(i)<<" ";
	arr[i]=arr[i-1]+findnumberofzero(i);
	//cout<<arr[i]<<endl;
    }
}
int main()
{
	solve();
	int t;
	long long int a,b;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%lld %lld",&a,&b);
		cout<<arr[b]-arr[a-1]<<endl;
	}
}
