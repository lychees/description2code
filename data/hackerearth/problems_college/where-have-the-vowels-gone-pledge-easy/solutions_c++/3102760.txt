#include <iostream>
#include<string>
using namespace std;
 
int main()
{
    long a,e,t,o,u,N;
    string m;
    a=0;e=0;t=0;o=0;u=0;
    cin>>N;
    cin>>m;
    for(long i=0;i<N;i++)
    {
    	if(m[i]=='a')
    	{
    		a++;
    		break;
    	}
    }
    for(long i=0;i<N;i++)
    {
    	if(m[i]=='e')
    	{
    		e++;
    		break;
    	}
    }
    for(int i=0;i<N;i++)
    {
    	if(m[i]=='i')
    	{
    		t++;
    		break;
    	}
    }
    for(long i=0;i<N;i++)
    {
    	if(m[i]=='o')
    	{
    		o++;
    		break;
    	}
    }
    for(long i=0;i<N;i++)
    {
    	if(m[i]=='u')
    	{
    		u++;
    		break;
    	}
    }
    if((a!=0)&&(e!=0)&&(t!=0)&&(o!=0)&&(u!=0))
    {
    cout<<"YES"<<endl;
    }
    else 
    {
    	cout<<"NO"<<endl;
    	}
    return 0;
}
