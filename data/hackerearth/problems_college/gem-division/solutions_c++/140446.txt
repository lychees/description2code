I dist(I a[],I n,I s)
{
    I i,j,su=s/2;
    //out("hello\n");
    I te[5005][102]={{0}};
    FOR(i,0,n+1)
    	te[0][i]=1;
    FOR(i,1,su+1)
	{
		FOR(j,1,n+1)
		{
			//te[i][j] = (a[j-1] <= i) ? (te[i][j-1] || te[i-a[j-1]][j-1]) : (te[i][j-1]);
			te[i][j]=te[i][j-1];
			if(i>=a[j-1])
				te[i][j]=te[i][j]||te[i-a[j-1]][j-1];
			//out("%d\n",te[i][j]);
		}
		//cout<<"\n";
	}
    //out("hello\n");
	for(i=su;i>=0;i--)
		if(te[i][n])
			return i;
}
     
I main()
{
    I t,i,j;
    in(t);
    for(;t--;)
    {
		I n,s=0;
		I mi=MAX,m=0,mm=MAX;
		in(n);
		I a[n];
		FOR(i,0,n)
	    {
		    in(a[i]);
		    s+=a[i];
	    }
	    sort(a,a+n);
	    FOR(i,0,n)
	    {
    		I b[n],k=0;
    		FOR(j,0,n)
    		{
    			if(j==i)
    				continue;
    			b[k++]=a[j];
    		}
//			out("%d\n",i);
			k=dist(b,n-1,s-a[i]);
//			out("k=%d\n",k);
			if(abs(k-abs(s-a[i]-k))<mm)
			{
				mi=abs(s-a[i]-k);
				mm=abs(k-mi);
				m=i;
			}
    	}
		out("%d %d\n",mm,a[m]);
