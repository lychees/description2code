inline ll power(ll a, ll b, ll m) {
	ll r = 1;
	while(b) {
		if(b & 1) r = r * a % m;
		a = (a * a)% m;
		b >>= 1;
	}
	return r;
}
inline ll power(ll a, ll b) {
	 ll r = 1;
	while(b) {
		if(b & 1) r = r * a ;
		a = a * a;
		b >>= 1;
	}
	return r;
}

int gcd(int a, int b){
	if(!b) return a;
	return gcd(b,a%b);
}
bool isperfect(ll num){
	int counter,gcd1=0;
	bool flag = 0;
	for(ll i =2 ; i*i <= num ; i++){
		counter=0;
		while(num%i == 0){
			num/= i;
			counter++;
		}
		if(flag && counter > 0){
			gcd1= gcd(gcd1,counter); 
		} 
		
		if(!flag && counter > 0){
			flag = 1;
			gcd1=counter;
		}
		
		
	}
	if(num > 1) return 0;
	if(gcd1 > 1) return 1;
	return 0;
}
