#include <iostream>
#include <cstdio>
#include <vector>
#include <cmath>
#include <map>
#include <string>
#include <algorithm>
#define pb push_back

using namespace std;

typedef long long ll;

struct prs
{
    string name;
    int in;
};

int main()
{
    //freopen("test.txt", "r", stdin);
    int n, count = 0, found;
    cin >> n;
    string name;
    prs save[n];

    for(int i=0; i<n; i++){
        cin >> name;
        found = -1;
        for(int j=0; j<count; j++){
            if(save[j].name == name) {found = j; break;}
        }
        if(found == -1){
            save[count].name = name;
            save[count].in = 1;
            count++;
        }

        else{
            save[found].in++;
        }

    }

    for(int i=0; i<count; i++){
        for(int j=i+1; j<count; j++){
            if(save[j].in > save[i].in) swap(save[i], save[j]);
        }
    }
    for(int i=0; i<count; i++) cout << save[i].name <<  ' ' << save[i].in << endl;

    return 0;
}
