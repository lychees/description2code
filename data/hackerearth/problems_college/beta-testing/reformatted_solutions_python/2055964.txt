N = int(raw_input())
c = raw_input().split()
a = []
b = []
for i in range(N):
    t = 0
    for j in range(len(a)):
        if a[j] == c[i]:
            b[j] = b[j]+1
            t = t + 1
    if t == 0:
        a.append(c[i])
        b.append(1)
for l in range(len(a)):
    max = 0
    p = 0
    for m in range(l, len(a)):
        if max < b[m]:
            max = b[m]
            p = m
    temp = b[l]
    b[l] = b[p]
    b[p] = temp
    temp2 = a[l]
    a[l] = a[p]
    a[p] = temp2
for k in range(len(a)):
    print a[k], b[k]
