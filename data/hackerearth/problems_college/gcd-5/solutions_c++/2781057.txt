#include <bits/stdc++.h>
using namespace std;

int main()
{
	int test;
	cin>>test;
    for(int t=0; t<test; t++)
        {
           
            int n;
            cin>>n;
            long long a[n]; 
            long long b[n];
            
            for(int i=0; i<n; i++)
            {
              //  System.out.println(line[i]+":");
              cin>>a[i];
               
                b[i] = a[i];
            }
            sort(b, b + n);
            bool flag = true;
            for(int i=0; i<n; i++)
            {
                if(a[i] != b[i])
                {
                    flag = false;
                    break;
                }
            }
            if(flag)
                cout<<"YES"<<endl;
            else 
                cout<<"NO"<<endl;
            
        }
    return 0;
}
