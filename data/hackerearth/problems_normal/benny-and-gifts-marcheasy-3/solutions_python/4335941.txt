def move(seq,x,y):
	if seq=='L':
		y-=1
	elif seq=='R':
		y+=1
	elif seq=='U':
		x-=1
	else:
		x+=1
	return x,y	

seq=list(raw_input().strip())
x=0
y=0

ice.add((0,0))

for i in range(len(seq)):

	if (x,y) in ice:
		count+=1
