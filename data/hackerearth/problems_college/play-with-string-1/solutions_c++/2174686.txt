#include <iostream>
#include<string.h>
using namespace std;

int main()
{   int t;
    cin>>t;
    while(t--)
    {   char str[100000];
        cin>>str;
        int n=strlen(str);
    	int i=0,j=n-1,flag=1;
    	while(i<n-1 && j>0)
    	{
    		int r1=abs(str[i+1]-str[i]);
    		int r2=abs(str[j]-str[j-1]);
    		if(r1!=r2)
    		{ flag=0;
    		  break;
    		}
    		i++;
    		j--;
    	}
    	if(flag)
    	  cout<<"Funny\n";
    	else
    	  cout<<"Not Funny\n";
    }
    
    return 0;
}
