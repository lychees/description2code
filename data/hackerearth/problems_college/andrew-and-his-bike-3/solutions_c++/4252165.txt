#include<iostream>
#include<vector>
#include<cmath>
#include<algorithm>
#include<stack>
using namespace std;

#define rep(i,a,b) for(int i=(a);i<(b);i++)
#define ll unsigned long long int
#define M 1000000007

int main() {
    std::ios::sync_with_stdio(false);
    int n;
    cin >> n;
    if (n == 1) {
    	cout << 0;
    	return 0;
    }

    vector<vector<int>> v(n+1);
    rep(i, 0, n-1) {
        int a, b;
        cin >> a >> b;
        v[a].push_back(b);
        v[b].push_back(a);
    }
    
    
    int ans[n+1], parent[n+1], visited[n+1];
    rep(i, 1, n+1) ans[i] = -1, parent[i] = -1, visited[i] = 0;
    parent[1] = 0;
    stack<int> s;
    s.push(1);
    while(!s.empty()) {
        int i = s.top();
        s.pop();
        if (ans[i] != -1) continue;
        s.push(i);
        if (!visited[i]) {
            rep(j, 0, v[i].size()) {
                if (parent[v[i][j]] == -1) parent[v[i][j]] = i;
                if (v[i][j] != parent[i]) s.push(v[i][j]);
            }
        }
        visited[i] = 1;
        
        bool notNegative = true;
        int sum = 0;
        rep(j, 0, v[i].size()) {
            if (ans[v[i][j]] == -1) {
                if (v[i][j] == parent[i]) continue;
                notNegative = false;
                break;
            } else {
                sum += ans[v[i][j]];
            }
        }
        
        if (i != 1) sum++;
        if (notNegative) ans[i] = sum;
    }
    
    rep(i, 1, n+1) cout << ans[i] << ' ';
    return 0;
}
