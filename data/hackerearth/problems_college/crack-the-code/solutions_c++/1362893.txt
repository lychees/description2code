#include <iostream>
using namespace std;

int main()
{
    int t;
    long int x,y,s,a,b,temp;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	cin>>x>>y;
    	a=0;
    	b=1;
    	s=0;
    	while(1)
    	{
    		if(b<x)
    		{
    			temp=b;
    			b=a+b;
    			a=temp;
    		}
    		else if(b>=x && b<=y)
    		{
    			s++;
    			temp=b;
    			b=a+b;
    			a=temp;
    		}
    		else 
    		break;
    	}
    	if(s%2==0)
    	cout<<"VALID"<<endl;
    	else
    	cout<<"INVALID"<<endl;
    }
    return 0;
}
