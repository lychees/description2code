#include<bits/stdc++.h>
using namespace std;
long long int s[1000002],c[1000002];
int main()
{
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%lld",&s[i]);
    }
    for(int i=0;i<n;i++){
        scanf("%lld",&c[i]);
    }
    sort(s,s+n);
    sort(c,c+n);
    long long ans =0;
    for(int i=0;i<n;i++){
        ans += s[i]*c[i];
    }
    printf("%lld",ans);
}
