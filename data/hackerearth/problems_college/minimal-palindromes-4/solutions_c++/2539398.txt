int gcd(int m, int n) {
   if ((m % n) == 0)
      return n;
   else
      return gcd(n, m % n);
}

int main()
{
char s[10001];
int T;
scanf("%d\n",&T);
int ans=4*5*6*7*8*9;
while(T--)
{
gets(s);
int g=4*5*6*7*8*9;
int i;
for(i=0;s[i];i++)
	{
	g=gcd(s[i]-'0',g);
	}
for(i=0;s[i];i++)
	{
	s[i]=(s[i]-'0')/g + '0';
	}
printf("%s\n",s);
}
