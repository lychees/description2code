#include <iostream>
#include <set>
#include <stdio.h>
using namespace std;
set<int> a;
int main() {
  int i, j, k, n, m, p, x;
  cin >> n >> m;
  while (n--) {
    cin >> x;
    while (x--) {
      cin >> p;
      a.insert(p);
    }
  }
  if (a.size() == m)
    cout << "YES" << endl;
  else
    cout << "NO" << endl;
  return 0;
}
