#include <iostream>
using namespace std;

int main()
{
    int N,flag,M,Q,lev,i,j;
    cin>>N>>M>>Q;
    int GL[100][5000],Qu[100];
    for( j=0;j<N;j++)
    for( i=0;i<M;i++)
    	cin>>GL[j][i];
    for( i=0;i<Q;i++)
    {
    for( j=0;j<N;j++)
    cin>>Qu[j];
    for(lev=M-1;lev>=0;lev--)
    {flag=1;
    	for( j=0;j<N;j++)
	    	if(Qu[j]<GL[j][lev]) {flag=0;break;}
	    	if(flag) break;
    }
    cout<<lev+1<<endl;
    }
    return 0;
}
