#include <iostream>
using namespace std;

int main()
{
    int tno,i;
    int count[26];
    string s;
    
    cin>>tno;
    
    while(tno--)
    {
    	cin>>s;
    	
  		for(i=0;i<26;i++)
  			count[i]=0;
  			
  		for(i=0;i<s.length();i++)
  		{
  			if(count[s[i]-'a']==1)
  				break;
  			count[s[i]-'a']++;
  		}
  		
  		if(i==s.length())
  			cout<<"No"<<endl;
  		else
  			cout<<"Yes"<<endl;
    }
    return 0;
}
