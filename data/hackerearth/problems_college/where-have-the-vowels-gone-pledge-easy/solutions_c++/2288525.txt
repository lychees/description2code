#include <iostream>
using namespace std;

int main()
{
    int N;
    string garbage;
    char vowels[5];
    int flag = 0;
    cin >> N;
    cin >> garbage;
    for( int i = 0; i < N; i++) {
    	switch ( garbage[i] ) {
    		case 'a': 
    		 vowels[0] = 'Y'; break;
    		case 'e': 
    		  vowels[1] = 'Y'; break;
    		case 'i': 
    		 vowels[2] = 'Y'; break;
    		case 'o': 
    		  vowels[3] = 'Y'; break;
    		case 'u': 
    		 vowels[4] = 'Y'; break;
    		default:
    		continue;
    	}
    }
    for ( int i =0; i < 5; i++) {
    	if(vowels[i] != 'Y') {
    		flag = 1;
    		break;
    	  
    	}	
    }
    if(!flag) {
    cout << "YES";
    }
    else{
    	cout << "NO";
    }
}
