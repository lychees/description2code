#include <iostream>
using namespace std;
int mx=0;

int cal(string s, int *dmax){
	int cur=0, c=0;
	if(s[0]!='C')
		mx=0;
	for(int i=0;i<s.size(); i++ ){
		if(s[i]=='C'){
			c++;
			mx++;
		}
		else {
			c=0;
			mx=0;
		}
		if(cur<c)
			cur=c;
		if(*dmax<mx)
			*dmax=mx;
	}
	if(s[s.size()-1]!='C')
		mx=0;
	return cur;
}

int main()
{
    int n, cur=0, dmax=0, c;
    cin>>n;
    string s[n];
    for(int i=0; i<n; i++){
    	cin>>s[i];
    	c=cal(s[i], &dmax);
    	if(cur<c)
    		cur=c;
    	//if(dmax<max)
    	//	dmax=max;
   		
    }
    cout<<cur<<" "<<dmax<<endl;
    return 0;
}
