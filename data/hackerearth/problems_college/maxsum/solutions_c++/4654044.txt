#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
    ll t;
    cin>>t;
    while(t--)
    {
        ll n;
        cin>>n;
       ll  dp[n],a[n];
        memset(dp,0,sizeof(dp));
        for(ll i=0;i<n;i++)
            cin>>a[i];
            dp[0]=a[0];
            dp[1]=max(a[0],a[1]);
            for(ll i=2;i<n;i++)
            {
                dp[i]=max(dp[i-1],a[i]+dp[i-2]);
            }
            cout<<dp[n-1]<<endl;

    }

}
