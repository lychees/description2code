using namespace std;

	cin>>t;
	
	while(t--){
		string x,y;
		cin>>x>>y;
		int a[27];
		int b[27];
		memset(a,0,sizeof(a));
		memset(b,0,sizeof(b));
		for(int i=0;i<x.length();i+=1)
			a[x[i]-'a']+=1;
		for(int i=0;i<y.length();i+=1)
			b[y[i]-'a']+=1;
			
		for(int i=0;i<26;i+=1){
			if(a[i]!=0 && b[i]!=0){
				cout<<"YES"<<endl;

			}
		}
		cout<<"NO"<<endl;

	}
