#include <iostream>
#include <cmath>
using namespace std;

int main()
{
	int T;
	cin>>T;
	while(T--)
	{
		int N,i;
		cin>>N;
		int arr[10]={0};
		int a=N,count=0;
		while(a)
		{
			arr[a%10]++;
			a/=10;
		}
		for(i=1;i<=sqrt(N);i++)
		{
			if(N%i==0)
			{
				int c,d;
				c=i;
				while(c)
				{
					if(arr[c%10]>0)
					{
						count++;
						break;
					}
					c/=10;
				}
			
			
			if(i*i!=N )
			{
				int c,d;
				c=N/i;
				while(c)
				{
					if(arr[c%10]>0)
					{
						count++;
						break;
					}
					c/=10;
				}
			}
			}
		}
		cout<<count<<endl;
	}
	
    //cout << "Hello World!" << endl;
    return 0;
}
