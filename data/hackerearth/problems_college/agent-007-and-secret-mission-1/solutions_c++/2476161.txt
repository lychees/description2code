#include <bits/stdc++.h>
using namespace std;
char ar[100002];
int main()
{
    int t,n,l,flag,i,ans;
    scanf("%d",&t);
    while(t--){
    	ans=0;
    	flag=0;
    	fflush(stdin);
    	scanf("%s",ar);
    	fflush(stdin);
    	scanf("%d",&n);
    	l=strlen(ar);
    	if(n%2){
    		if(((n/2)%2)==0)
    			flag=1;
    	}
    	else{
    		if(((n/2)%2)==1)
    			flag=1;
    	}
    	//cout<<"flag==="<<flag<<endl;
    	if(flag==0){
    		for(i=0;i<l;i++)
    			if(ar[i]=='1')
    				ans++;
    	}
    	else{
    		for(i=0;i<l;i++){
    			if(ar[i]=='0')
    				ans++;
    		}
    	}
    	printf("%d\n",ans);
    }
    return 0;
}
