#include <iostream>
using namespace std;

int p,q;

int fibo(int n)
{	if(n==0)
	{	++p;
		return 0;
	}
	if(n==1)
	{	++q;
		return 0;
	}
	return fibo(n-1)+fibo(n-2);
}

int main()
{	int t;
    cin>>t;
    while(t--)
    {	int n,x;
    	cin>>n;
    	p=0;
    	q=0;
    	x=fibo(n);
    	cout<<p<<" "<<q<<"\n";
    }
    return 0;
}
