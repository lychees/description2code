#include <bits/stdc++.h>

#define MAX=100000;
#define si(a) scanf("%d",&a)
using namespace std;
long long col[100100]={0};
long long wt[100100]={0};
long long dfe[100100]={0};
long long df[100100]={0};
int dfin=0;
vector <long long > adj[100100];

void bf(int u,int di)
{
    col[u]=1;
    wt[u]=di;
    df[u]=dfin;
    dfin++;
    for(int i=0;i<adj[u].size();i++)
    {

        int y=adj[u][i];
        if(col[y]==0)
        {
            bf(y, di+1);
        }
    }
    dfe[u]=dfin;
    dfin++;
}

int main()
{
    int t;
    cin>>t;
    for(int z=0;z<t;z++)
    {
    int n,root;
    cin>>n;
    //adj[n+1].clear();
    //col[0]=0;
    for(int i=0;i<=n;i++)
    {
    	adj[i].clear();
        
        col[i]=0;
        df[i]=0;
        dfe[i]=0;
        wt[i]=0;
    }
    for(int i=1;i<=n;i++)
    {
    	int a;
    	scanf("%d",&a);
        if(a==0)
        	root=i;
        else
        adj[a].push_back(i);
    }

    bf(root,0);
    int q;
    si(q);
    for(int i=0;i<q;i++)
    {
        int x,y;
        si(x);
         si(y);
        int ans=wt[x]-wt[y];
        if(df[y]<df[x] && dfe[x]<dfe[y])
        cout<<ans-1<<endl;
        else
        cout<<"-1"<<endl;
    }
    }
}
