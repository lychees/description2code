#include<iostream>
using namespace std;
int main(){
	long long int t;
	cin>>t;
	while(t--){
		long long int n,i,j,c=0;
		cin>>n;
		long long int a[n];
		for(i=0;i<n;i++){
			cin>>a[i];
		}
		for(i=0;i<n;i++){
			for(j=i;j<n;j++){
				if(a[i] == a[j]){
					c++;
				}
			}
		}
		cout<<c<<"\n";
	}
	return 0;
}
