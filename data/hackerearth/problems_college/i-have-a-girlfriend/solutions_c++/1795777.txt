const int N = (int) 1e6 + 10;
int n, k, delta[N], z[N];
char s[N];

int main() {
//	assert(freopen(taskname".in", "r", stdin));
//	assert(freopen(taskname".out", "w", stdout));
	scanf("%d%d%s", &n, &k, s);
	z[0] = 0;
	for (int i = 1, l = 0, r = 0; i < n; ++i) {
		z[i] = min(z[i - l], max(0, r - i + 1));
		while (s[z[i]] == s[i + z[i]])
			++z[i], l = i, r = i + z[i] - 1;
	}
	for (int l = 1; l * (int64) k <= n; ++l)
		if (z[l] >= (k - 1) * l)
			delta[k * l - 1]++, delta[k * l + min(l, z[k * l])]--;
	for (int i = 0, sum = 0; i < n; ++i)
		printf("%d", (int) ((sum += delta[i]) > 0));
	printf("\n");
	return 0;
}
