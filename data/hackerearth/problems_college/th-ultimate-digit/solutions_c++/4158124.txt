#include <iostream>
#include <string>
using namespace std;

int power(int x,unsigned long long y)
{
	int res=1;
	while(y>0)
	{
		if(y&1)	res=(res*x)%10;
		y=y >> 1;
		x=(x*x)%10;
	}
	return res;
}

int main()
{
    int t;
    string a;
    unsigned long long b;
    cin>>t;
    while(t--)
	{
		cin>>a>>b;
		cout<<power(a[a.length()-1]-48,b)<<endl;
	}
    return 0;
}
