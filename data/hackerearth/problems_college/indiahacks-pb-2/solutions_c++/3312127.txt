#include <iostream>
#include<vector>
using namespace std;
int is(string A,string B)
{
	if(A.size()!=B.size())
	return 0;
	else
	{
		string s=A+B;
		int i=0,j=2*A.size()-1;
		while(i<j)
		{
			if(s[i]!=s[j])
			return 0;
			i++;
			j--;
		}
	}

	return 1;
}
int main()
{
    int t;
    cin>>t;
    vector<string> s;
    int x;
    char y;
    while(t--)
    {
    	string A;
    	cin>>A;
    	for(int i=0;i<s.size();i++)
    	{
    		int a=is(A,s[i]);
    		if(a==1)
    		{
    			x=A.size();
    			y=A[(A.size()-1)/2];
    		}
    		
    	}
    	s.push_back(A);
    	
    }
    cout<<x<<" "<<y<<endl;
    return 0;
}
