#include <bits/stdc++.h>
using namespace std;

void solve(){
    string s;
    cin>>s;
    if(string(s.rbegin(),s.rend())!=s){
        cout<<"NO"<<endl;
        return;
    }
    for(int i=0;i<s.size();i++){
        if(s[i]!='0'&&s[i]!='1'&&s[i]!='8'){
            cout<<"NO"<<endl;
            return;
        }
    }
    cout<<"YES"<<endl;

}
int main(){
    int t;
    cin>>t;
    while(t--)solve();
}
