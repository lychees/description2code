#include <bits/stdc++.h>
#define M 1000000007
#define ll long long

using namespace std;
int n, a[52][52];

int path(int x, int y) {
    if (x == n-1 && y == n-1) return 1;
    int g, h;
    if (y + 1 <= n - 1 && a[x][y+1] == 1) g = path(x, y + 1);
    else g = 0;
    if (x + 1 <= n - 1 && a[x+1][y] == 1) h = path(x + 1, y);
    else h = 0;

    return g + h;
}
int main() {
    int t;
    cin >> t;
    while(t--) {
        cin >> n;

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                cin >> a[i][j];
            }
        }

        if (path(0, 0)) {
            cout << "POSSIBLE\n";
        } else {
            cout << "NOT POSSIBLE\n";
        }
    }
    return 0;
}
