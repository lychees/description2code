t = int(raw_input())
for i in range(t):
    n = int(raw_input())
    if n < 4:
        print 0
    elif n % 2 == 0:
        if n % 4 == 0:
            print(n/4)**2
        else:
            n = (n-2)/4
            print n*(n+1)
    else:
        if (n-1) % 4 == 0:
            n = (n-1)/4
            print n**2
        else:
            n = (n+1)/4
            print n*(n-1)
