#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
typedef vector<int> vi;
typedef pair<int,int> pii;

const int N = 1010;
int r[N],c[N],a[N][N];
char s[N];
int main()
{
	int m,n;
	cin>>m>>n;
	for(int i=1;i<=m;i++)
	{
		scanf("%s",s);
		for(int j=1;j<=n;j++)
		{
			if(s[j-1]=='B')
			{
				a[i][j]=(a[i][j]+1)%2;
				r[i]=(r[i]+1)%2;
				c[j]=(c[j]+1)%2;
			}
		}
	}
	int ans = 0;
	for(int i=1;i<=m;i++)
	{
		for(int j=1;j<=n;j++)
		{
			a[i][j]=(a[i][j]+r[i]+c[j])%2;
			ans += a[i][j];
		}
	}
	cout<<ans<<endl;
}
