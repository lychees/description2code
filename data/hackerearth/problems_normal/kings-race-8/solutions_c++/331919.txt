#include<iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;
int A[1000003];
int B[1000003];
int main(){
	int i,j,t,n,k,min_val,min_index,p,cnt,tmp;;
	scanf("%d",&t);
	while(t--){
		scanf("%d %d",&n,&k);
	for(i=0;i<n;i++)
		scanf("%d",&A[i]);
	for(i=0;i<k;i++)
		scanf("%d",&B[i]);
	cnt=0;p=0;
	min_index=0;
	for(i=0;i<n;i++){
		for(j=p;j<k;j++){
			if(A[i]>=B[j]){
				min_index=i;
				p++;
			}
			else{
				p=j;
				break;
			}
		}
	}
	
	printf("%d\n",min_index);
	}
	
	return 0;
	
	
}
