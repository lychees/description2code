#!/usr/bin/env python
def fact(x):
    val = 1
    for i in range(2, x+1):
        val *= i
    return val


i = 20
ans = []
for x in range(20):
    ans.append(fact(x))
t = eval(raw_input())
while t != 0:
    t -= 1
    num = map(int, raw_input().split())
    val = 0
    for x in range(num[0], num[1]+1):
        val += ans[x]
    print val % 10
