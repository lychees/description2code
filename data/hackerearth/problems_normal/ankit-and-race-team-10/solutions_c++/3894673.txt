#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define MOD 1000000007
ll fact[100005],ifact[100005];
ll sqr(ll x)
{
    return x*x;
}
ll bmod(ll a,ll b)
{
    if(b==0)
    return 1;
    else if(b%2==0)
    {
        return (sqr(bmod(a,b/2)))%MOD;
    }
    else
    {
        return (a*bmod(a,b-1))%MOD;
    }
}
void fac()
{
    fact[0]=1;
    ll i;
    for(i=1;i<=100000;i++)
    fact[i]=(i*fact[i-1])%MOD;
}
void ifac()
{
    ll i;
    ifact[100000]=bmod(fact[100000],MOD-2);
    for(i=100000-1;i>=0;i--)
    {
        ifact[i]=((i+1)*ifact[i+1])%MOD;
    }
}
ll ncr(ll n,ll r)
{
    ll res;
    res=fact[n];
    res=(res*ifact[r])%MOD;
    res=(res*ifact[n-r])%MOD;
    return res;
}
int main()
{
    fac();
    ifac();
    ll t,n,k,i,j,ans;
    cin>>t;
    while(t--)
    {
        cin>>n>>k;
        ll a[n];
        for(i=0;i<n;i++)
        cin>>a[i];
        sort(a,a+n);
        j=1;
        ans=0;
        for(i=0;i<=n-k;i++)
        {
            j=ncr(n-i-1,k-1);
            ans=(ans+(a[i]*j)%MOD)%MOD;
        }
        cout<<ans<<endl;
    }
    return 0;
}
