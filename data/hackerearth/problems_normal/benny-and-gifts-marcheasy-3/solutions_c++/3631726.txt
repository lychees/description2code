#include <bits/stdc++.h>
using namespace std;
int main()
{
    string in;
    int ans=0;
    cin>>in;
    map<pair<int, int>, bool> m;
    int x=0, y=0;
    m[make_pair(x,y)]=1;
    for(int i=0; i<in.length(); i++){
    	if(in[i]=='R') y+=1;
    	else if(in[i]=='L') y-=1;
    	else if(in[i]=='D') x+=1;
    	else x-=1;
    //	cout<<x<<" "<<y<<endl;
    	if(m[make_pair(x,y)]) ans++;
    	else{
    		m[make_pair(x,y)]=1;
    	}
    }
    cout<<ans<<endl;
    return 0;
}
