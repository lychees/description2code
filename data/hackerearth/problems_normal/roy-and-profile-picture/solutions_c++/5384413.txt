#include <iostream>
using namespace std;

int main()
{
	int i,l,n,w[1001],h[1001];
	cin>>l;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>w[i]>>h[i];
	}
	for(i=0;i<n;i++)
	{
		if((w[i]<l)||(h[i]<l))
		{
			cout<<"UPLOAD ANOTHER"<<endl;
		}
		else 
		if(w[i]==h[i])
		{
			cout<<"ACCEPTED"<<endl;
		}
		else
		{
		cout<<"CROP IT"<<endl;	
		}
		
	}
    return 0;
}
