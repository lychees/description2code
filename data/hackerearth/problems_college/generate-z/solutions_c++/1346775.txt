#include <bits/stdc++.h>
using namespace std;

int main()
{
   	int n;
   	cin >> n;
   	for(int i = 0; i < n; i++) {
   		if(i == 0 || i == n - 1) {
   			for(int i = 0; i < n; i++) {
   				printf("*");
   			}
   			cout << endl;
   		}
   		else {
   			for(int j = 0; j < (n - 1) - i; j++) {
   				printf(" ");
   			}	
   			printf("*\n");
   		}
   	}
    return 0;
}
