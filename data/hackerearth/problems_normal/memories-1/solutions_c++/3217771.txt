#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main()
{
    int t,n,p,y,x,c,i,j,k,f;
    int chk[15][15];
    cin >> t;
    while(t--)
    {
    	memset(chk,0,sizeof(chk));
    	cin >> n;
    	cin >> p;
    	while(p--)
    	{
    		cin >> x >> y;
    		chk[--x][--y]=1;
    		chk[y][x]=1;
    	}
    	c=0;
    	for(i=1;i<(1<<n);++i)
    	{
    		f=0;
    		for(j=0;j<n;++j)
    		{
    			if((1<<j)&i)
    			for(k=0;k<j;++k)
    			{
    				if((1<<k)&i)
    				{
    					if(chk[j][k])
    					{f=1;break;}
    				}
    			}
    			if(f)break;
    		}
    		if(!f)c++;
    	}
    	cout << c << endl;
    }
 return 0;   
}
