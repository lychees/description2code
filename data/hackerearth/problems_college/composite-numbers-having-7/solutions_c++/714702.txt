 #include<iostream>

using namespace std;
int arr[1000001] = { 0 };
int composite[1000001] = { 0 };
int main()
{
	for (int i = 2; i < 1000001; i++)
	{
		if (arr[i] == 1)
			continue;
		for (int j = 2 * i; j < 1000001; j += i)
			arr[j] = 1;
	}
	for (int i = 1; i < 1000001; i++)
	{
		if (arr[i] == 1)
		{
			int x = i,flag=1;
			while (x)
			{
				if (x % 10 == 7)
				{
					flag = 0;
					composite[i]=composite[i-1]+1;
					break;
				}
				x = x / 10;
			}
			if (flag)
				composite[i] = composite[i - 1];
		}
		else
			composite[i] = composite[i - 1];
	}
	int t,n,m;
	scanf("%d",&t);
	while (t--)
	{
		scanf("%d %d",&n,&m);

		if (composite[m] - composite[n-1] != 0)
			printf("%d\n", composite[m] - composite[n-1]);// << endl;
		else
			printf("-1\n");
	}

	return 0;
}
