#include<stdio.h>
#include<stdlib.h>
#define INT_MAX 300000000
int max(int a,int b)
{
    return a>b?a:b;
}
int min(int a,int b)    
{
    return a>b?b:a;
}
int hash[30000000];
int main()
{
    int i,j,k,n,t,x;
    scanf("%d",&t);
    while(t--)
    {
        int f=0;
        int x_max=0,x_min=INT_MAX,flag=0;
        scanf("%d",&n);
        for(i=0;i<n;i++)
        {
            scanf("%d",&x);
            hash[x]++;

            x_max=max(x,x_max);
            x_min=min(x,x_min);

        }

       if(hash[x_min]>1)
         flag=1;
        hash[x_min]=0;

        int tag=x_min,mmin=INT_MAX;
        for(j=x_min+1;j<=x_max;j++)
        {
            if(hash[j]==1)
            {
               mmin=min(mmin,j-tag);
               tag=j;
               hash[j]=0;
            }
            else if(hash[j]>1)
            {
               flag=1;
               hash[j]=0;
            }
        }
        if(flag==1)

            mmin=0;
        printf("%d\n",mmin);

    }
    return 0;
}
