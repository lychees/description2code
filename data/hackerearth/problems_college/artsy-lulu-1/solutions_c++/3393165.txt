#include <iostream>
#include <map>
#include <cstdio>

using namespace std;

int main(){

        char ch; int h; bool flag = false; string alpha = "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
        //map<int,char> data_store;
       // for(int i = 1; i < alpha.length(); i++) data_store[i] = alpha[i-1];

        while(cin >> ch >> h){
            if(isalpha(ch) == false || h <= 1) cout << "INVALID\n"; //TEMP
            else{
                if(isupper(ch)) flag = true;
                else flag = false;

                int counters = int(ch)-1;

                if((flag == true && counters+h <= 90) || (flag == false && counters+h <= 122))
                for(int i = 0; i <= h; i++){
                    for(int j = 0; j < i; j++)
                        if(flag == true) cout << char(counters) << " ";
                        else cout << char(counters) << " ";
                    counters++;
                    cout << endl;
                }
                else cout << "\nINVALID\n";

            }
            //cout << endl;
        }
        return 0;
}
