#include<cstdio>
using namespace std;
long long Ad[1<<20];
int main(){
	int N,Q;
	long long ans=0;
	scanf("%d%d",&N,&Q);
	for(int i=1;i<=N;++i)Ad[i]=0;
	while(Q--){
		int a,b,k;
		
		scanf("%d%d%d",&a,&b,&k);
		for(int i=a;i<=b;++i)
		Ad[i]+=k;
}
	for(int i=1;i<=N;i++)
	ans+=Ad[i];
		printf("%lld",ans/N);
	
}
