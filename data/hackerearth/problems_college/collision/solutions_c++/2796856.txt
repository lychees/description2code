#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int num,i;
		int f[10]={0},sum=0;
		for(i=0;i<n;i++)
		{
			cin>>num;
			f[num%10]++;
		}
		
		for(i=0;i<10;i++)
		{
			if(f[i]!=0)
			{
				sum = sum + f[i]-1;
			}
		}
		cout<<sum<<"\n";
	}
 }
