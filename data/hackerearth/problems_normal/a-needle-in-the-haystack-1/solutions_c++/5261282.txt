#include <bits/stdc++.h>
using namespace std;

int main(){
	int n;
	cin >> n;
	while(n--){
		string foo, bar;
		cin >> foo >> bar;
		if (foo.size() > bar.size()){
			puts("NO"); continue;
		}
		map<char, int> mp;
		for (char i = 'a'; i <= 'z'; i++)
			mp[i] = 0;
		for (int i = 0; i < foo.size(); i++)
			mp[bar[i]]++;
		for (int i = 0; i < foo.size(); i++)
			mp[foo[i]]--;
		bool done = false;
		for (int i = foo.size(); i < bar.size(); i++){
			bool flag = false;
			for (map<char,int>::iterator it=mp.begin();it!=mp.end();it++){
				if (it->second != 0){
					flag = true; break;
				}
			}
			if (!flag){
				done = true;
				break;
			}
			else
				mp[bar[i]]++, mp[bar[i - foo.size()]]--;
		}
		bool flag = false;
		for (map<char,int>::iterator it=mp.begin();!done&&it!=mp.end();it++){
			if (it->second != 0)
				flag = true;
		}
		if (!flag)
			done = true;
		(done)? puts("YES"): puts("NO");
		
	}
}
