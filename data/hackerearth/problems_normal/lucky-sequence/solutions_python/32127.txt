from math import log
t = int(raw_input())
for i in range(t):
    k = int(raw_input())
    n = log(k+2)/log(2)
    if n==int(n):
        n -= 1
    else:
        n = int(n)
    binary = bin(int(k - pow(2, n) + 1)).lstrip('0b')
    l = len(binary)
    if l<n:
        binary = ''.zfill(n-l) + binary
    binary = binary.replace('0', '4')
    binary = binary.replace('1', '7')
