#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int n,num=0,d;
    cin>>n;
    string s,p="";
    
    for(int i=0;i<n;i++)
    {
    	cin>>s;
    	p=p+s[0];
    }
    for(int i=0;i<26;i++)
    {
    	d=count(p.begin(),p.end(),(char)(97+i));
    	num=num+(d/10);
    	if(d%10!=0)
    	num+=1;
    }
    cout<<num<<endl;
    
    
    return 0;
}
