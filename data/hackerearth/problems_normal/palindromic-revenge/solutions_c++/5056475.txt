#include <iostream>
#include <string.h>
using namespace std;

int main()
{
int l,diff,t;
char str[100000];
cin>>t;
int count1[15] = {0};
for (int j=0;j<t;j++)
{
cin>>str;
l = strlen(str);
    for(int i=0;i<l/2;i++)
    {
        diff = *(str+i) - *(str+l-1-i);
        if(diff<0)
            diff= -diff;
        count1[j] = count1[j] + diff;
    }
}
for (int i=0;i<t;i++)
cout<<count1[i]<<"\n";
}
