int main()
{
	unsigned long k,a,b,c,d,t,T;
	cin>>T;
	while(T--)
	{
		cin>>a>>b>>c>>d>>k;
		t=int(cbrt((float)((k)/a)))+1; 	
		while(t--)							
		if((d+(t)*(c+(t)*(b+a*(t))))<=k||t==0) 	
		{
 		cout<<t<<endl;					
		break;									
 		}
	}
    return 0;
}
