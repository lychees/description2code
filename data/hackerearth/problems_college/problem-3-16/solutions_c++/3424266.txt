#include <bits/stdc++.h>

using namespace std;

map<int, int> m;

int main()
{
    int t, n, k, x;

    cin >> t;

    while(t--) {

        int sum = 0;

        scanf("%d", &n);
        scanf("%d", &k);

        for(int i = 0; i < n; i++) {
            scanf("%d", &x);
            m[x]++;
        }

        int rem = 0;

        for(map<int, int>::iterator it = m.begin(); it != m.end(); it++) {
            sum += it->second;
            rem = max(rem, it->second);
        }

        sum -= rem;
        m.clear();
        printf("%d\n", sum);
    }
}
