#include <iostream>
#include <cmath>
using namespace std;

int main()
{
	long long int n;
	cin>>n;
	long long int x[n],y[n],f[n];
	long double dp[n];
	for(int i=0;i<n;i++){
		dp[i]=-1000000;
	}
	for(int i=0;i<n;i++){
		cin>>x[i]>>y[i]>>f[i];
		dp[0] = f[0];
		for(int j=0;j<i;j++){
			dp[i] = max(dp[i], f[i] + dp[j] - sqrt((long double)(pow((x[i]-x[j]),2)+pow((y[i]-y[j]),2))));
		}
	}
	cout.precision(6);
 	cout<<fixed<<dp[n-1]<<endl;
	// printf("%.6f", ((signed long)((dp[n-1] * 1000000)) /1000000.0f));
	// cout<<round(dp[n-1]*1000000)/1000000.0f;
    return 0;
}
