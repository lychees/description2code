#include<iostream>
#include<stdlib.h>
using namespace std;
int main()
{
    string s,m="",n="";
    cin>>s;
    for(int i=0;i<s.length();i++)
    {
        if(i!=s.length()-1){
            m+=s[i];
            m+="-";
        }

        else
            m+=s[i];
    }

    for(int i=s.length()-1;i>=0;i--)
    {
        if(i!=0){
            n+=s[i];
            n+="-";
        }

        else
            n+=s[i];
    }
    cout<<m<<endl<<n;
}
