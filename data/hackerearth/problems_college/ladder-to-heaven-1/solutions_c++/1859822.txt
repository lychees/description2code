#include <iostream>
using namespace std;

int main()
{
	unsigned long long n,i=0,num,temp,ans=2,a,b;
	cin>>n;
	while(i<n){
		cin>>num;
		a=1;
		b=2;
		ans=2;
		while(1){
			temp=a+b;
			if(num<=temp)
				break;
			else if(temp%2 == 0)
				ans+=temp;
			a=b;
			b=temp;
		}
		cout<<ans<<endl;
		i++;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
