#include <bits/stdc++.h>
 
using namespace std;
 

 
const int dx[4] = {-1,0,1,0};
const int dy[4] = {0,1,0,-1};
bool out (int x, int y, int r, int c) {
    if (x < 0 || x >= r) return 1;
    if (y < 0 || y >= c) return 1;
    return 0;
}
 
/*---------------- Main Code begins --------------------*/
 
int a[105][105], vis[105][105], n;
 
long long  solve (int x, int y) {
	if ((x == n - 1) && (y == n - 1)) {
		return !a[x][y];
	}	
	vis[x][y] = 1;
	long long  res = 0;
	for (int i=0; i<4; i++) {
		int nx = x + dx[i];
		int ny = y + dy[i];
		if (!out(nx,ny,n,n) && a[nx][ny] != 1) {
			if (!vis[nx][ny]) {
				res += solve(nx, ny);
			}
		}
	}
	vis[x][y] = 0;
	return res;
}
 
int main() {
	cin>>n;
	for (int i=0; i<n; i++) {
		for (int j=0; j<n; j++) scanf("%d",&a[i][j]);
	}
	memset(vis, 0, sizeof(vis));
	long long  ans = solve (0,0);
	printf("%lld",ans); 
    return 0;
}
