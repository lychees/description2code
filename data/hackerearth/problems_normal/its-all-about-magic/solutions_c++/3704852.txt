#include <iostream>
using namespace std;

int main()
{
    int t=0;
    cin>>t;
    string digit="1";
    for(int i=0; i<t; i++)
    {
    	int m=0;
    	cin>>m;
    	long zerocount=0;
    	long onecount=1;
    	while(m>0)
    	{
    		long k = zerocount;
    		long j = onecount;
    		while(k>0)
    		{
    			onecount++;
    			k--;
    		}
    		while(j>0)
    		{

    			j--;
    		}
    		m--;
    	}
    	cout<<onecount<<" "<<zerocount<<endl;
    	
    }
    return 0;
}
