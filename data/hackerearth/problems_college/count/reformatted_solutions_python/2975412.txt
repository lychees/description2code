#!usr/bin/python
from __future__ import print_function
t = input()
while (t > 0):
    r, c = map(int, raw_input().split())
    e = 0
    o = 0
    while (r > 0):
        if (r % 2 == 0):
            e += min(r, c)
        else:
            o += min(r, c)
        r -= 1
    print(e, o, end='\n', sep=' ')
    t -= 1
