#include <iostream>
#include<algorithm>
using namespace std;

unsigned long long n;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		cin>>n;
		const unsigned long long q=n;
		unsigned long x[q];
		for(int i=0; i<q; ++i)
		{
			cin>>x[i];
		}
		
		sort(x,x+q);
		int count=1;
		
		for(int i=1; i<q; ++i)
		{
			if(x[i]!=x[i-1])
			{
				count++;
			}
		}
		cout<<count<<endl;
		
	}
    return 0;
}
