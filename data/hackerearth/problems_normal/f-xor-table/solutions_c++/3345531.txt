string s;
int a[1000006],b[1000006];
int main(){
  int t;
  cin>>t;
  while(t--){
    LL n,m,p,x;
    cin>>n>>m>>p>>x;
    cin>>s;
    for(int i=0;i<m;i++){
      if(s[i]=='1') a[i] = 1;
      else a[i] = 0;
    }
    int nn = n-1;
    for(int k=30;k>=0;k--){
      if(nn&(1<<k)){
        LL bin = 1<<k;
        for(LL i=0;i<m;i++){
          LL j = (i + (bin*p)%m) % m;
          if(a[i]==a[j]) b[j] = 0 ^ x;
          else b[j] = 1 ^ x;
        }
        for(int i=0;i<m;i++){
          a[i] = b[i];
        }
        nn -= bin;
      }
    }
    for(int i=0;i<m;i++) cout<<a[i]; cout<<endl;
  }
	return 0;
}
