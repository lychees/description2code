#include <iostream>
using namespace std;
long cnt[27];
int main()
{
	int n,i,od;
	string str;
    cin >> str;
    n = str.length();
    for(i = 0;i < n;i++){
    	cnt[str[i] - 'a']++;
    }
    for(i = 0;i < 26;i++){
    	if(cnt[i]%2 == 1){
    		od++;
    	}
    }
    if(n%2 == 0 && od == 0){
    	cout << "YES\n";
    }
    else if(n%2 == 1 && od == 1){
    	cout << "YES\n";
    }
    else{
    	cout << "NO\n";
    }
}
