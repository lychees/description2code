#include<bits/stdc++.h>
using namespace std;
main(){
int t,n,i;
long long normal,boost,a,b;	
long long x[100000],y[100000];
long long ans;
scanf("%d",&t);
while(t--)
{
	ans=0;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	scanf("%lld",&x[i]);
	for(i=0;i<n;i++)
	scanf("%lld",&y[i]);
	scanf("%lld%lld",&a,&b);
	scanf("%lld%lld",&normal,&boost);
ans=abs(a)*normal+abs(b)*normal;
for(i=0;i<n;i++)
ans=min(ans,((abs(x[i])+abs(y[i]))*normal+boost*(abs(a-x[i])+abs(b-y[i]))));
printf("%lld\n",ans);
}
return 0;
}
