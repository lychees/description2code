#include <bits/stdc++.h>
using namespace std;

#define iter(c) __typeof((c).begin())
#define tr(i, c) for (iter(c) i = (c).begin(); i != (c).end(); i++)
#define rep(i, a, b) for (int i = (a); i < (b); i++)
#define read(x) scanf("%lld", &(x))
#define debug(x) cout << #x << "  " << x << endl

#define mp make_pair
#define pb push_back
#define f first
#define s second

typedef long long ll;
typedef pair<int, int> ii;

const int MAXN = int(1e5) + 5;

ll n, M, a[MAXN];

int process(ll t) {
    int cnt(0);
    ll tmp = 0;
    rep(i, 0, n) {
        if (tmp < a[i]) {
            if (t < a[i]) return -1;
            cnt++;
            tmp = t - a[i];
        }
        else tmp -= a[i];
    }
    return cnt;
}

int main() {
//    freopen("input.inp", "r", stdin);

    read(n), read(M);
    rep(i, 0, n) read(a[i]);

    ll res(1e17);
    ll l = 1, h = ll(1e17);
    while (l <= h) {
        ll m = (h-l)/2 + l;
        int x = process(m);

        if (x != -1 && x <= M) res = m, h = m-1;
        else l = m+1;

    }

    printf("%lld", res);
}
