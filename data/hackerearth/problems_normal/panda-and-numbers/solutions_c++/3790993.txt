#include <iostream>
using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while (t--)
   	{
   		int n;
   		scanf("%d",&n);
   		if (n == 1)
   			printf("Yes\n");
   		else if (n == 4)
   			printf("Yes\n");
   		else if (n > 1 && n < 13)
   			printf("No\n");
   		else if (n == 90 || n == 91)
   			printf("No\n");
   		else
   			printf("Yes\n");
   	}
    return 0;
}
