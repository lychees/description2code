#include <iostream>
#include<string.h>
#include<string>
#include<vector>

#define LL long long
using namespace std;

LL dp[10005];
vector<LL> ed[10005];
LL n;

LL func(LL ind) {
	
	if(ind == n)
	return 1;
	
	if(dp[ind] != -1)
	return dp[ind];
	
	LL i,j,k;
	
	k = 0;
	
	for (i = 0; i < ed[ind].size(); i++) {
		k = k + func(ed[ind][i]);
	}
	
	return dp[ind] = k;
}

int main()
{
    LL i,j,k,l,m,t;
    
    cin >> n;
    
    scanf("%lld%lld",&k,&l);
    
    while(k!=0) {
    	ed[k].push_back(l);
    	cin >> k >> l;
    }
    
    memset(dp,-1,sizeof dp);
    
    l = func(1);
    
    cout << l << endl;
    
    return 0;
}
