typedef long long LL;

#define MOD 1000000007
#define MAX 1000001
#define pb push_back
#define yes puts("YES")
#define no puts("NO")
#define sd(x) scanf("%d", &x)
#define PI 3.14159265

using namespace std;

LL pf[MAX];
LL sum[MAX];
LL dv[MAX];
vector<int> arr;

bool prime[MAX];

int main() {
//	cout << "helo";
	for (int i=2; i<MAX; i++) {
		if (!prime[i]) {
			arr.pb(i);
			sum[i] = (i+sum[i-1])%MOD;
			for (int j=2*i; j<MAX; j+=i)
				prime[j]=true;
			pf[i]=pf[i-1]+1;
		} else {
			sum[i]=sum[i-1];
			pf[i]=pf[i-1];
		}
		
	}
	arr.pb(10000000);
	
	int t;
	cin >> t;
	
	while (t--) {
		int n;
		cin >> n;
		cout << pf[n] << ' ' << sum[n] << ' ';
		
		LL z=1;
		for (int j=0; arr[j]<=n; j++) {
			LL num=arr[j];
			LL p=0;
			while (num<=n) {
				p+=n/num;
				num*=arr[j];
			}
			z=(z*(p+1))%MOD;
		}
	    cout << z << endl;
	}
	
	
	return 0;
}
