#include <iostream>
using namespace std;

int main()
{
	int n;
	cin >> n;
	int i, dom[n], brian[n];
	for(i= 0; i< n; i++)
	{
		cin >> dom[i];
	}
	for(i= 0; i< n; i++)
	{
		cin >> brian[i];
	}
	int mod[n - 1], nairb[n - 1];
	for(i = 0; i < n - 1; i++)
	{
		mod[i] = dom[i + 1] - dom[i];
		nairb[i] = brian[i + 1] - brian[i];
		if(mod[i] < 0)
		{
			mod[i] = mod[i] * (-1);
		}
		if(nairb[i] < 0)
		{
			nairb[i] = nairb[i] * (-1);
		}
	}
	int escalade = mod[0], lambo = nairb[0];
	for(i = 1; i < n - 1; i++)
	{
		if(mod[i] > escalade)
		{
			escalade = mod[i];
		}
		if(nairb[i] > lambo)
		{
			lambo = nairb[i];
		}
	}
	if(escalade > lambo)
	{
		cout << "Dom\n" << escalade <<endl;
	}
	if(lambo > escalade)
	{
		cout << "Brian\n" << lambo << endl;
	}
}
