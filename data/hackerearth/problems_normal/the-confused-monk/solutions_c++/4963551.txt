#include <iostream>
#include <vector>
#define MOD 1000000007

using namespace std;

long long int GCD(long long int a, long long int b)
{
	if(b==0) return a;
	else return GCD(b,a%b);
}

int main()
{
    vector<long long> a;
    long long int n,temp,gcd;
    cin>>n;
    for(long long int i=0; i<n ; i++)
    {
    	cin>>temp;
    	a.push_back(temp);
    }
    gcd=a[0];
    for(long long int i=0; i< n; i++)
    {
    	gcd = GCD(gcd,a[i]);
    }
    long long int pro=1;
    for(long long int i=0;i<n;i++)
    {
    	pro*=(a[i]);
    	pro%=MOD;
    }
    long long int ans=1;
  
    while(gcd!=0)
    {
    	
    	if(gcd%2==1)
    	{
    		ans*=pro;
    		ans%=MOD;
    	}
    	pro=((pro%MOD)*(pro%MOD))%MOD;
    	pro%=MOD;
    	gcd/=2;
    }
    ans=ans%MOD;
    cout<<ans;
    return 0;
}
