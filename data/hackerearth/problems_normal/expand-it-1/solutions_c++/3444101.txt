#include<iostream>
#include<string>
using namespace std;
int main()
{
unsigned long long f[26],q,x,num,n,c,j,i;
string s;
cin>>s;
x=s.length();
for(int i=0;i<26;i++)
{
    f[i]=0;
}
for(int i=0;i<x;)
{
	if(s[i]>='a' && s[i]<='z')
	{
		j=i+1;
		num=0;
		while(j<x && (s[j]>='0' && s[j]<='9'))
		{
			num=num*10+(s[j]-'0');
				j++;
		}
		f[s[i]-'a']+=num;
		i=j;
	}
}
	
cin>>q;
while(q--)
{
	cin>>n;

	c=0;
	for(i=0;i<26;i++)
	{
		c+=f[i];
		if(n<=c)
		{
		cout<<(char)(i+'a')<<"\n";
		break;
		}
	}
	if(i==26)
		cout<<"-1\n";
}

	return 0;
}
