#include<bits/stdc++.h>
using namespace std;
int main()
{	int t;
	cin>>t;
	while(t--)
	{	int n;
		map<int,int> m;
		cin>>n;
		for(int i=0;i<n;i++)
		{	int value;
			cin>>value;
			m[value]++;
		}
			
		set<int> s;
		for(map<int,int>::iterator it=m.begin();it!=m.end();++it)
		{	s.insert(it->second);
		}
		if(s.size()==1)
		{	cout<<"YES"<<endl;
		}
		else	if(s.size()>=3)
			{	cout<<"NO"<<endl;
			}
			else
			{
		int c[3];
		int index=0;
		for(set<int>::iterator it=s.begin();it!=s.end();++it)
		{	int value=*it;
			c[index++]=value;
		}
		int sm=min(c[0],c[1]);
		int lar=max(c[0],c[1]);
		int scou=0,lcou=0;
	
		for(map<int,int>::iterator it=m.begin();it!=m.end();++it)
		{	if(it->second==sm)scou++;
			else	if(it->second==lar)lcou++;
		}
		if(scou==1&&lcou==1&&lar-sm==1)cout<<"YES"<<endl;
		else{
		if(scou==1)
		{	if(sm!=1)
			{	cout<<"NO"<<endl;
			}
			else
			{	cout<<"YES"<<endl;
			}
		}
		else	if(lcou==1&&(lar-sm)==1)
			{	cout<<"YES"<<endl;
			}
			else	cout<<"NO"<<endl;
		}}
	}
	return 0;
}
