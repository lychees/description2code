#include <iostream>
#include<vector>
#include<climits>
#include <numeric> 
using namespace std;

int main() {

 int t,i,z;
 
vector<int> v;
int ans=0;

cin>>t;

     for(i=0;i<t;i++)
     {
         cin>>z;
         
         if(z%2)
         v.push_back(z);
         else
         ans+=z;
     }
     
     if(v.size()==0){ cout<<":(";
     return 0;}
     
     if(v.size()%2) ans+=accumulate(v.begin(),v.end(),0);
     else
     {
     	int g=INT_MAX;
     	
     	for(i=0;i<v.size();i++)
     { if(v[i]<g)
      g=v[i];
     }
     //cout<<"___"<<g;
     ans+=accumulate(v.begin(),v.end(),0);
     ans-=g;
     	
     	
     }
     
 
  cout<<ans;
  
	return 0;
}
