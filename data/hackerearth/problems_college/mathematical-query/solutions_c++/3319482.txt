#include <iostream>
using namespace std;
#define mod 10000003
long long int arr[1000001],i;
long long int  f(int num)
{  
    long long a=arr[999998];
    long long b=arr[999999];
    long long c=arr[1000000];
    long long d;
	for(i=1000001;i<=num;i++)
	{
		d=(a+b)%mod;
		d+=i*2;
		d%=mod;
		a=b;
		b=c;
		c=d;
	}
	return d;
}
int main()
    {
    arr[0]=0;
    arr[1]=1;
    arr[2]=1;
    for(i=3;i<1000001;i++)
    {
    	arr[i]=arr[i-2]+arr[i-3];
    	arr[i]%=mod;
    	arr[i]+=(i*2);
    	arr[i]%=mod;
    }
    int tc,num;
    cin>>tc;
    while(tc--)
    {
    	cin>>num;
    	if(num>1000000)
    	{
    	//	cout<<"h";
		cout<<f(num)<<endl;}
    	else
    	cout<<arr[num]<<endl;
    }
    return 0;
}
