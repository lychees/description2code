lli modpow(lli base,lli exp,lli mod)
{   if(mod==1) return 0;
    lli res=1;
    base%=mod;
    while(exp>0)
    {   if(exp%2) res=(res*base)%mod;
        base=(base*base)%mod;
        exp/=2;
    }
    return res;
}
lli modinverse(lli a,lli mod)
{   return modpow(a,mod-2,mod);
}
lli nCrmod(lli n,lli r,lli mod)
{   if(r>n) r=n-r;
    lli res=1;
    for(lli i=r;i>=1;i--)
    {   res=(res*(n-i+1))%mod;
        res=(res*modinverse(i,mod));
    }
    return res;
}
lli totient(lli n)
{   lli res=n,p;
    for(p=2;p*p<=n;p++)
        if(n%p==0)
        {   while(n%p==0)
            n/=p;
            res-=res/p;
        }
    if(n>1) res-=res/n;
    return res;
}
