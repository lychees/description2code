#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int test;
	cin>>test;
	while(test--)
	{
		ll int n;
		cin>>n;
		ll int i,j,mc=0;
		for(i=0;i<n;i++)
		{
			cin>>j;
			if(j&1)
			mc++;
		}
		if((mc%2)==0)
		cout<<"Yes\n";
		else
		cout<<"No\n";
	}
	return 0;
}
