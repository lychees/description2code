#include <iostream>
using namespace std;
long long int gcd(long long int a,long long int b)
{
    if (a%b==0) return b;
    return gcd(b,a%b);
}
int main()
{
    long long int ans,one,num,den,n,i,p,g;
    int t;
    string a;
    cin>>t;
    while (t--)
    {
        cin>>n>>p;
        cin>>a;
        p+=1;
        one=0;
        ans=0;
        for (i=0;i<p;i++) if (a[i]=='1') one++;
        ans=one*one;
        for (i=p;i<n;i++)
        {
            if (a[i-p]=='1') one-=1;
            if (a[i]=='1')
            {
               one+=1;
               ans+=2*one-1;
            }
        }
        num=ans;
        den=n*n;
        g=gcd(num,den);
        cout<<num/g<<"/"<<den/g<<endl;
    }
    return 0;
}
