#include<stdio.h>
#include<algorithm>
using namespace std;
bool compare(int a,int b)
{
	if(a>b)
	return true;
	else
	return false;
}
int main()
{
	int n;
	scanf("%d",&n);
	long long int s[n],c[n];
	long long int mul=0;
	for(int i=0;i<n;i++)-
	scanf("%lld",&s[i]);
	for(int i=0;i<n;i++)
	scanf("%lld",&c[i]);
	sort(s,s+n,compare);
	sort(c,c+n,compare);
	for(int i=0;i<n;i++)
	mul+=s[i]*c[i];
	printf("%lld",mul);
	return 0;
}
