#include <iostream>
using namespace std;

int main()
{
    int l, w, h, n;
    cin >> l;
    cin >> n;
    
    for (int i = 0; i < n; i++) {
    	cin >> w >> h;
    	
    	if (w < l || h < l)
    		cout << "UPLOAD ANOTHER" << endl;
    	else if (w >= l && h >= l && w != h)
    		cout << "CROP IT" << endl;
    	else if (w >= l && h >= l && w == h)
    		cout << "ACCEPTED" << endl;

    }
}
