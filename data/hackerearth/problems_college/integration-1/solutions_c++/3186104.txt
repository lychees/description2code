#include <iostream>

using namespace std;

int main()
{
    int t;
    int i=0;

    cin>>t;

    while(t--){
        long int n,p,u,r,s,counter;


        cin>>n>>p>>u>>r>>s;
        while( n-- ){

            if(p + r <= u){
                p = p + r;

                continue;
            }
            if(p + r > u && p - s >= 0){
                p = p - s;

                continue;
            }
            else{
                break;
            }
        }

        cout<<"Case #"<<++i<<": "<<p<<endl;
    }
    return 0;
}
