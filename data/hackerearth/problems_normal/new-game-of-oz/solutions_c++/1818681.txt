{

    	int n;

    	int a[n];
    	for(int i=0;i<n;i++) cin>>a[i];
    	map<int,int> h;
    	for(int i=0;i<n;i++) h[a[i]]=1;

    	for(int i=0;i<n;)

    		z++;

}
