#include <bits/stdc++.h>
#define mp make_pair
#define all(x) x.begin(),x,end()
using namespace std;

int main()
{
    int n;
    string s,t[1000];
    int j=0;
    //cin>>s;
    getline(cin,s);
    //cout<<s.length()<<endl;
    s+=" ";
    std::locale loc;
  
  for (std::string::size_type i=0; i<s.length(); ++i)
    s[i]=std::toupper(s[i],loc);
    
    map<string,int> m;
    map<string,int>::iterator it;
    for(int i=0;i<s.length();i++){
    	if(s[i]==' '){
    		//cout<<s.length()<<endl;
    		it=m.find(t[j]);
    		if(it!=m.end()){
    			it->second=it->second +1;
    			t[j]="";
    		}
    		else{
    			m.insert(mp(t[j],1));
    		j++;
    	}
    	
    	}
    	else{
    		t[j]+=(s[i]);
    	}
    }
    for(int i=0;i<1000;i++){
    	if(t[i].empty())
    		continue;
    	cout<<t[i]<<" "<<m[t[i]]<<endl;
    }
    return 0;
}
