#include <iostream>
using namespace std;


int main(int argc, char** argv) {
	int t;
	unsigned long long int L,B,H;
	long double r;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%lld",&L);
		scanf("%lld",&B);
		scanf("%lld",&H);
		if(L<=B && L<=H)
		{
		r=L/2.0;
		printf("%.1Lf",r);
		printf("\n");
	}
		else if(B<=L && B<=H)
		{
		r=B/2.0;
		printf("%.1Lf",r);
		printf("\n");
	}
		else if(H<=B && H<=L)
		{
		r=H/2.0;
		printf("%.1Lf",r);
		printf("\n");
	}
	}
	return 0;
}
