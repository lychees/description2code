#include<iostream>
using namespace std;
 
int main()
{
	int res[999999];
	int t,n,carry,prod,res_size;
	long long int ans;
	cin >> t;
	while(t--)
	{
		cin >> n;
		res_size = 1;
		res[0] = 1;
		for(int i=2;i<=n;i++)
		{
			carry = 0;
			for(int j=0;j<res_size;j++)
			{
				prod = res[j]*i+carry;
				res[j] = prod%10;
				carry = prod/10;
			}
			while(carry)
			{
				res[res_size] = carry%10;
				carry = carry/10;
				res_size++;
			}
		}
		ans = 0;
		for(int k=(res_size-1);k>=0;k--)
		{
			ans += res[k];
		}
		cout << ans << "\n";
	}
	return 0;
}
