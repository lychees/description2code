int is_pal(char *str){

    for(int i=0;i<l;i++){
            arr[str[i]-97]++;
        }

    if(l%2==0){

        for(i=0;i<26;i++){
            if(arr[i]%2!=0)
                return 0;
        }


    }
    else{
        j=0;
        for(i=0;i<26;i++){
            if(arr[i]%2!=0)
                j++;
        }
        if(j!=1)
            return 0;
    }

    return 1;
}
