# include <iostream>
# include <stdio.h>
using namespace std;
long long A[1000000];
void zero()
{
    int i;

    for(i=1;i<=1000000;i++)
    {
        int count = 0;
        if(i == 5)
            {
                A[i]=1;

            }

        else
        {
            for ( int j = 5; i/j >= 1; j *= 5 )
            {

                count  +=  i / j;
            }

            A[i]=count+A[i-1];
        }
    }
}



int main()
{
    zero();
    long long t,a,b;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld%lld",&a,&b);

        printf("%lld\n",A[b]-A[a-1]);
    }
    return 0;
}
