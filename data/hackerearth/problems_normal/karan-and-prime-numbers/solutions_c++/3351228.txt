#include <iostream>
#include <math.h>
using namespace std;
int prime(long long N)
{
	       if(N==1)
	      return 0;
	    if(N==2||N==3)
	         return 1;
	for(long long i=2;i<=sqrt(N);i++)
	      if(N%i==0)
	        return 0;
   return 1;	
}
int main()
{
    int T;
    long long a,b,sum;
    cin>>T;
    for(int i=0;i<T;i++){
    	cin>>a>>b; sum=0;
    	for(long long j=a;j<=b;j++)
    		 if(prime(j))
    		   sum+=j;
        cout<<sum<<endl;
    }
    return 0;
}
