#include <bits/stdc++.h>
#define max 1000005
#define ll long long
ll etf[max],mul[max];
using namespace std;
void pre()
{
	ll i,j;
	//we have to find phi(n)=n*m(1-1/p)
	for(i=1;i<=max;i++)
	etf[i]=i;
	for(i=2;i<=max;i++){
		if(etf[i]==i)//that's prime
		{
			etf[i]=i-1;
			for(j=i<<1;j<=max;j+=i)//start from 2*i as phi(prime)=prime-1
				etf[j]-=etf[j]/i;//same as n*(1-1/p)=n-n/p
		}
	}
}
int main() {
	pre();
	// your code goes here
	int t;
	cin>>t;
	while(t--)
	{
		int y,n,x,k;
		cin>>n>>x>>k;
		//cout<<etf[30]<<endl;
		for(y=n;y>=x;y--)
		{
			if(etf[y]==k)
			{
				cout<<y<<endl;
				break;
			}
		}
		if(y<x)
		{
			cout<<"-1"<<endl;
		}
	}

	return 0;
}
