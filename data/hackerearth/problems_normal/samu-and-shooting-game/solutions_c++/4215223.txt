#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

ll t,x,y,n,w,p1,p2;
long double a2[1000][1000];
ll a1[1000][1000];
long double prob1 , prob2;
long double  recurse(int idx , int points){
    if(idx == n || points >= w){
        return (points >= w);
    }
    if(a1[idx][points]!=-1)return a2[idx][points];
    a1[idx][points] = 1;
    double res1 = prob1*recurse(idx+1,points+x) + (1-prob1)*recurse(idx+1,points);
    double res2 = prob2*recurse(idx+1,points+y) + (1-prob2)*recurse(idx+1,points);
    return a2[idx][points] = max(res1,res2);
}
int main()
{

    scanf("%d",&t);
    while(t--)
    {
        scanf("%lld %lld %lld %lld %lld %lld",&x,&y,&n,&w,&p1,&p2);
        memset(a2,0,sizeof(a2));
        memset(a1,-1,sizeof(a1));
        prob1 = (p1/100.0);
        prob2 = (p2/100.0);
        double ans = recurse(0,0)*100.0;
        cout << setprecision(6) << fixed << ans << endl;
    }
	return 0;
}
