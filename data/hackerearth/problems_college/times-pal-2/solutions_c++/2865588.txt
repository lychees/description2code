#include<bits/stdc++.h>
using namespace std;
bool ispal(string a)
{
    string ret=a;
    reverse(a.begin(),a.end());
    if(ret==a)
        return true;
    else
        return false;
}
int main()
{
    string a,b,temp,tt,ty;
    int i,ct,t;
    cin>>t;
    while(t--)
    {
        ct=0;
    cin>>a>>b;
    for(i=0;i<a.size();i++)
    {
        temp=a.substr(0,i);
        temp+=b;
        tt=a.substr(i,a.size()-i+1);
        temp+=tt;
        if(ispal(temp))
            ct++;
    }
    if(ispal(a+b))
        ct++;
    cout<<ct<<endl;
    }
}
