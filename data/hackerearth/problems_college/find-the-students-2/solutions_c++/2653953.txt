#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n;
    int x[1000];
    vector<int> v;
    int prev;
    cin>>n;
    for(int i=0;i<n;i++)
    {
    	cin>>x[i];
    }
    sort(x,x+n);
    prev = x[0];
    int count = n;
    v.push_back(count);
    count--;
    for(int i=1;i<n;i++)
    {
    	x[i]-=prev;
    	if(x[i] == 0)
    	count--;
    	else
    	{
    		//cout<<"here at "<<i<<" for "<<x[i]<<endl;
    		int temp = x[i]+prev;
    		prev = temp;
    		v.push_back(count);
    		count--;
    	}
    }
    for(int i=0;i<v.size();i++)
    cout<<v[i]<<endl;
    return 0;
}
