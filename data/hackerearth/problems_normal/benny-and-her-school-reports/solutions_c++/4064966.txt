#include <iostream>
using namespace std;

int main()
{
    long int t,n,m,x;
    cin>>t;
    while(t--)
    {
    	cin>>n>>m>>x;
    	int a[n];
    	long int sum=0;
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    		sum+=a[i];
    	}
    	long int xx=(n+1)*x-sum;
    	if(xx<0)
    	cout<<"1\n";
    	else if(xx<1 || xx>m)
    	cout<<"Impossible\n";
    	else
    	cout<<xx<<"\n";;
    }
    return 0;
}
