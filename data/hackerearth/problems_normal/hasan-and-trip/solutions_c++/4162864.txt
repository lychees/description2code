double  dist[3001][3001];
double f[3001];
double dp[3001];

int main(){
ll n;cin>>n;
ll i,j;
ll x[n+1],y[n+1];
for(i=1;i<=n;i++){
cin>>x[i]>>y[i]>>f[i];
}
for(i=1;i<=n;i++)
for(j=1;j<=n;j++){
dp[i]=-inf;
dist[i][j]=sqrt((x[i]-x[j])*(x[i]-x[j])+(y[i]-y[j])*(y[i]-y[j]));}
dp[1]=f[1];
for(i=2;i<=n;i++)
for(j=1;j<i;j++){
dp[i]=max(dp[i],f[i]-dist[i][j]+dp[j]);
}
cout.precision(6);
cout<<std::fixed<<dp[n];
}
