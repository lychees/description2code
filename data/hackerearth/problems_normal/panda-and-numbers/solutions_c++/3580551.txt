#include<bits/stdc++.h>
#define ll long long
using namespace std;
vector<ll> v;
bool dig[10];
void digit(ll num)
{
    for(int i=0;i<10;i++)
        dig[i]=false;
    while(num)
    {
        dig[num%10]=true;
        num/=10;
    }

}
ll power(ll base,ll pwr)
{

    if(pwr==0)
        return 1;
    else if(pwr&1)
    {
        return base*power(base,pwr-1);

    }
    else
    {

        ll x= power(base,pwr/2);
        return x*x;
    }
}
bool panda[1000005];
int main()
{

    ll t,y,n,flag;
    for(int i=0;i<1000005;i++)
        panda[i]=false;
    cin>>t;
    for(int i=1;;i++)
    {

        y=(power(i,i));
        if(y>1000005)
            break;
        else
        {
            panda[y]=true;

        }

    }
       for(ll i=0;i<1000005;i++)
       {

           ll k=i;
           while(k)
           {
               int x=k%10;
               k/=10;
               if(i-x*x>=0)
                panda[i]|=panda[i-x*x];
           }
       }
    while(t--)
    {
        cin>>n;
        flag=0;


        if(panda[n])
            cout<<"Yes\n";
        else
            cout<<"No\n";

    }
}
