#include <iostream>
using namespace std;
int arr[1000001];
int main()
{
    int t,n,l,flag;
    long long sum,cost;
    cin>>t;
    while(t--)
    {
    	flag=0;
    	cin>>n>>cost;
    	for(int i=0;i<n;i++)
    	{
    		cin>>arr[i];
    	}
    	sum=l=0;
    	for(int i=0;i<n;i++)
    	{
    		
    		
    		sum+=arr[i];
    		if(sum>cost)
    		{
    			while(sum>cost)
    			{
    				sum-=arr[l];
    				l++;
    			}
    			//sum+=arr[i];
    		}
    //		cout<<sum<<" ";
    		if(cost==sum)
    		{
    			flag=1;
    			break;
    		}
    	}
    //	cout<<sum;
    	if(flag)
    	cout<<"YES\n";
    	else
    	cout<<"NO\n";
    }
    return 0;
}
