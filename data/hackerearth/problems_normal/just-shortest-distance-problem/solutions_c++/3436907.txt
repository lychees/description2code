#include <bits/stdc++.h>

using namespace std;

#define X first
#define Y second
#define INPUT freopen("clash.inp","r",stdin)
#define OUTPUT freopen("clash.out","w",stdout)
#define FOR(i,l,r) for(auto i=(l);i<=(r);i++)
#define REP(i,l,r) for(auto i=(l);i<(r);i++)
#define FORD(i,l,r) for(auto i=(l);i>=(r);i--)
#define REPD(i,l,r) for(auto i=(l);i>(r);i--)
#define ENDL printf("\n")
#define debug 1

typedef long long ll;
typedef pair<int,int> ii;

const int inf=1e9;
const int MOD=1e9+7;
const int N=1e3+10;

int n,q[N],d[N];
vector <int> v[N];

void remake(){
    int top=1,bot=1;
    memset(d,0,sizeof(d));
    d[1]=1;
    q[1]=1;
    while (top<=bot){
        int x=q[top++];
        REP(i,v[x].begin(),v[x].end()) {
        	int y=*i;
        	if (!d[y]){
	            d[y]=d[x]+1;
	            q[++bot]=y;
	        }
        }
    }
}
bool check(int x,int y){
    if (d[x]) return (d[y]==0||d[x]+1<d[y]);
    return 0;
}
int main(){
    int m,type,x,y;
    scanf("%d%d",&n,&m);
    d[1]=1;
    while (m--){
        scanf("%d%d",&type,&x);
        if (type==1){
            printf("%d\n",d[x]-1);
            continue;
        }
        scanf("%d",&y);
        v[x].push_back(y);
        if (check(x,y)) remake();
    }
}
