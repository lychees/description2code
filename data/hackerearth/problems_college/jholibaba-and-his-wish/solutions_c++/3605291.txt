#include <cstdio>
using namespace std;
int n;
long long  a[1000005];
long long total=0;
int hash[26];
int main() {
	int i,j;
	long long sum=0;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%lld",&a[i]);
		total+=a[i];
		sum+=a[i]*(i+1);
		
	}
	long long max=sum;
	for(i=0;i<n-1;i++)
	{
		sum=sum+total-a[n-1-i]*n;
		if(sum>max)
		max=sum;
	}
	printf("%lld\n",max);
	return 0;
}
