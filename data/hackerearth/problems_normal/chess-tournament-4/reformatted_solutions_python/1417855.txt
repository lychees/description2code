arr = [[0], [0]]
t = input()
for i in xrange(1, 2**t):
    new_list = [0]+map(int, raw_input().split())
    arr.append(new_list)
a = [i for i in xrange(1, 2**t+1)]
for i in xrange(1, 2*(2**t-1), 2):
    for j in xrange(i-1, i):
        x, y = a[i], a[j]
        if arr[x][y] == 1:
            a.append(x)
        else:
            a.append(y)
print a.pop()
