#include <iostream>
#include <string>
#include <vector>
using namespace std;

int main()
{
    int p1,p2,k,base;
    cin>>k;
    string str,temp;
    char ch;
    cin>>str;
    temp=str;
    base=0;
    p2=1;
    while(p2<str.size())
    {
   		ch=str[str.size()-1];
   		str.erase(str.size()-1);
   		str.insert(p2,1,ch);
   		p2+=2;
    }
    base=1;
    while(str!=temp)
    {
    	p2=1;
    	while(p2<str.size())
    	{
    		ch=str[str.size()-1];
    		str.erase(str.size()-1);
    		str.insert(p2,1,ch);
    		p2+=2;
    	}
    	base++;
    }
    k=base-k%base;
    for(p1=1;p1<=k;p1++)
    {
    	p2=1;
    	while(p2<str.size())
    	{
    		ch=str[str.size()-1];
    		str.erase(str.size()-1);
    		str.insert(p2,1,ch);
    		p2+=2;
    	}
    	//cout<<p1<<"-"<<str<<endl;
    }
    cout<<str;
    return 0;
}
