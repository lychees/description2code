chkpts = int(raw_input())
dom = map(int,raw_input().split())
brian = map(int,raw_input().split())
max_sp = []

def maxDiff(arr):
	max_sp=[]
	for i in xrange(chkpts):
		if i != chkpts-1:
			max_sp.append(arr[i+1] - arr[i])
	max_sp = map(abs,max_sp)
	return max(max_sp)  

domMax = maxDiff(dom)
brianMax = maxDiff(brian)

if domMax > brianMax:
    print 'Dom'
    print domMax
elif brianMax > domMax:
    print 'Brian'
    print brianMax
elif domMax == brianMax:
    print 'Tie'
    print domMax
