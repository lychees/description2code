#include <iostream>
using namespace std;
long int pow(long int x)
{
	int count=-1;
	while (x!=0)
	{
		x=x/2;
		count++;
	}
	return count;
}
long int number(int count)
{
	long int num=1;
	for (int i=0;i<count;i++)
	{
		num=2*num;
	}
	return num;
}
int main()
{
	int t;
	cin>>t;
	for (int i=0;i<t;i++)
	{
		long int n,m;
		cin>>n>>m;
		while (m>1)
		{
			cout<<"("<<n<<"<<"<<pow(m)<<")";
			m-=number(pow(m));
			if (m!=0)
				cout<<" + ";
		}
		if (m==1)
			cout<<"("<<n<<"<<0)";
		cout<<endl;
	}
}
