#include <bits/stdc++.h>
using namespace std;
int prime[1000002];
map<int,int>mp;
int n;
void sieve()
{
	int i,j;
	prime[0] = prime[1] = 0;
	for(i = 2; i <= n; i++) {
		if(!prime[i]) {
			for(j = 2; j*i <= n; j++) {
				prime[i*j] += 1;
			}
			prime[i] = 1;
		}
	}
	for(i = 2; i <= n; i++) {
		//cout<<prime[i]<<" ";
		mp[prime[i]] += 1;
	}
}
int main()
{
	//sieve();
	int q,k;
	cin>>n>>q;
	sieve();
	while(q--) {
		cin>>k;
		cout<<mp[k]<<endl;
	}
    return 0;
}
