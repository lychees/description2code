
#include<iostream>
#include<malloc.h>
#include<math.h>
using namespace std;
inline unsigned long int max(unsigned long int a,unsigned long int b)
{   if(a>b)
      return a;
     else
      return b;
}
inline int check(unsigned long int t)
{  unsigned long int i;
int f=0;
    for(i=2;i<sqrt(t) && f!=1;i++)
    {    if(t%i==0)
         { f=1;
           break;
         }
         else
         {f=0;
          continue;
         }
    }
  return f;
}
int main()
{ unsigned long int n,*x,i,m=0,d=0;
  int t,count=0,pre;
 cin>>n;
 x=(unsigned long int*)malloc(n*sizeof(unsigned long int));
 for(i=0;i<n;i++)
 { cin>>*(x+i);
 }
 for(i=0;i<n;i++)
 {    d=*(x+i);
      pre=m;
     if(d!=m)
     { t=check(*(x+i));
       //cout<<*(x+i)<<" "<<t<<endl;
       if(t!=1)
       {
        m=max(m,*(x+i));
          if(m!=pre)
            count=1;
            
        }
     }
     else
      count++;
 }
 cout<<m<<endl;
 cout<<count;
return 0;}
