#include <iostream>
#include<stdio.h>
using namespace std;
int a[1000007];
int b[1000007];
char str[1000007];

int main()
{
    int cases;
    scanf("%d",&cases);
    while(cases--)
    {
    	int m,n;
    	scanf("%d %d",&m,&n);
    	scanf("%s",str);
    	for(int i=0;i<1000002;i++)
    	{
    		a[i]=b[i]=0;
    	}
    	a[0]=b[0]=1;
    	int num_of_k=0;
    	int num_of_r=0;
    long long 	int res1=0;
    long long 	int res2=0;
    	for(int i=0;str[i];i++)
    	{
    		if(str[i]=='K')
    		num_of_k++;
    		else 
    		num_of_r++;
    		if(num_of_k>=n)
    		res1=res1+b[num_of_k-n];
    		
    		if(num_of_r>=m)
    		res2=res2+a[num_of_r-m];
    		
    		a[num_of_r]++;
    		b[num_of_k]++;
    		
    	
    		
    		
    		
    		
    		
    	}
    	printf("%lld %lld\n",res2,res1);
    	
    }
    return 0;
}
