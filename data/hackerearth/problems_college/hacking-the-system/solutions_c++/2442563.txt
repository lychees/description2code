#include <iostream>
using namespace std;

int main()
{
    long long int ct = 0;
    string a , b;
    cin >> a >> b;
    for(long long int i = 0; i < a.size(); i++) {
    	if(a[i] != b[i]) {
    		ct++;
    	}
    }
    if(ct == 1 || ct == 0) {
    	cout << "wins" << endl;
    }
    else {
    	cout << "loses" << endl;
    }
    return 0;
}
