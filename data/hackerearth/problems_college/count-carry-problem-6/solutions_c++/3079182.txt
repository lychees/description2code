typedef long long int ll;

using namespace std;

int t, mx;
string a, b;

int get(char c){
return c - '0';
}
int main(){
//    ios_base::sync_with_stdio(0);
//    cin.tie(0);
//    freopen("data.in","r",stdin);
    cin >> t;
    while(t--){
        cin >> a >> b;
        mx = max(a.size(), b.size());
        while(mx != a.size())a = "0" + a;
        while(mx != b.size())b = "0" + b;
        int ans = 0, aux = 0;
        for(int i = mx - 1; i >= 0; i--){
            aux += (get(a[i]) + get(b[i]));
            if(aux > 9)ans++;
            aux /= 10;
        }
        if(ans == 0)cout << "No carry operation" ;
        else if(ans == 1)cout << "1 carry operation";
        else cout << ans << " carry operations";
        cout << endl;
    }
    return 0;
}
