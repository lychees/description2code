#include <iostream>
#include <math.h>
using namespace std;

int main()
{
    int n,a;
    long int d1=0,d2=0;
    cin>>n;
    for (int i=0;i<n;i++)
    {	for (int j=0;j<n;j++)
    	 {
    	 	cin>>a;
    	 	if (i==j)
    	 		d1+=a;
    	 	if (j+i==n-1)
    	 		d2+=a;
    	 	
    	 }
    }
	cout<<fabs(d1-d2);
    return 0;
}
