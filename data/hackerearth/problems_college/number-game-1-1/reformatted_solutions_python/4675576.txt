t = int(raw_input())
i = 0
while i < t:
    n = int(raw_input())
    count = 0
    while n > 0:
        if n % 3 == 0:
            n /= 3
            count += 1
        elif n % 2 == 0:
            n /= 2
            count += 1
        elif n != 0:
            n -= 1
            count += 1

    print count-1
    i += 1
