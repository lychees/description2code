#include <iostream>
using namespace std;
 
int countPairs(long long int arr[], long long int n)
{
    long long int count = 0;
     
    // Pick all elements one by one
    for (long long int i = 0; i < n; i++)
    {       
        // See if there is a pair of this picked element
        for (int j = i+1; j < n; j++)
            if (arr[i] - arr[j] == 0)
                  count++;
    }
    return count;
}
 
int main()
{
	long long T,N,A[1000000],i;
	cin >> T;
	while(T--){
		cin >> N;
		for(i=0;i<N;i++){
			cin >> A[i];
		}
		cout << countPairs(A,N)+N << endl;
	}
    return 0;
}
