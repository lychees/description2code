#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t-->0)
    {
       int i,n;
       cin>>n;
       unsigned long long int a=0,f;
       f = pow(2,n-1);
       if(n==1)
       {
              cout<<"1"<<endl;
              continue;
       }
       for(i=1;i<=n;i++)
       {
              a = a + (i*f);
       }
       cout<<a<<endl;
    }
    return 0;
}
