#include<bits/stdc++.h>
using namespace std;
#define ll long long
pair <ll , pair <int,ll> > p[100000];
int main()
{
int t,n;
cin>>t;
while(t--)
{
    cin>>n;
    for(int i=0;i<n;i++)
     {
         ll temp,f;
         cin>>temp;
         f=temp;
         int ones=0;
         while(temp)
         {
             if(temp%2 ==1 ) ones++;
             temp/=2;
         }
          p[i]=make_pair(ones,make_pair(i,f));
        }
    sort(p,p+n);
    for(int i=0;i<n;i++)
        cout<<p[i].second.second<<" ";
    cout<<endl;
}
 return 0;
}
