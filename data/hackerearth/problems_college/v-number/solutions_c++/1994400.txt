#include <iostream>
#include <string.h>
#include <stdio.h>
#include <stdlib.h>
using namespace std;
int main(int argc,char *argv[]){
		int t;
		cin >> t;
		while(t--){
			char num[100]={0};
			cin >> num;
			int i = strlen(num);
			int half = i/2;
			int j=0;
			int firsthalfsum = 0;
			int secondhalfsum = 0;
			//cout << "half length = " << half <<endl;
			//cout << num[0] << num[1] << num[2] << num[3];
			for(j=0;j<half;j++){
				firsthalfsum += num[j]-48;
			}

			for(j=half;j<i;j++){
				secondhalfsum += num[j]-48;
			}

			//cout << "first " << firsthalfsum << "\n second : " << secondhalfsum << endl;
				if(firsthalfsum == secondhalfsum)
					cout << "YES" <<endl;
				else
					cout << "NO" << endl;
		}
}
