t = input()
while t:
    t -= 1
    n = input()
    sh = []
    sta = []
    # sh=map(int, raw_input().split())
    # sta=map(int, raw_input().split())
    for i in range(0, n):
        sh.append(int(input()))
    for i in range(0, n):
        sta.append(int(input()))

    sh.sort()
    sta.sort()

    survive = True
    for i in range(0, n):
        if sta[i] > sh[i]:
            survive = False
            break
    print 'YES' if survive else 'NO'
