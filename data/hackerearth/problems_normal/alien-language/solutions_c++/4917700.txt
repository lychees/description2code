#include<iostream>
#include<string>
#include<map>
using namespace std;

int main() {
  int t;
  cin >> t;
  string text, pattern;
  map<char, bool> codeMap;
  while (t--) {
    text.clear();
    pattern.clear();
    codeMap.clear();
    cin >> text >> pattern;
    for (int i = 0; i < text.size(); i++) {
      codeMap[text[i]] = true;
    }
    bool can = false;
    for (int i = 0; i < pattern.size(); i++) {

      if (codeMap[pattern[i]]) {
        can = true;
        break;
      }
    }

    if (can) {
      cout << "YES" << endl;
    }
    else {
      cout << "NO" << endl;
    }

  }

  return 0;
}
