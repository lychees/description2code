#include <iostream>
#include<cmath>
#include<math.h>
#include<stdlib.h>
#include<cstdio>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		float x1,y1,x2,y2,radius;
		cin>>x1>>y1>>x2>>y2;
		cin>>radius;
		//int m=(y2-y1)/(x2-x1);
		float res= abs(x2*y1-x1*y2)/sqrt((y2-y1)*(y2-y1) + (x2-x1)*(x2-x1));
		if(res<radius)
		cout<<"REPLANNING"<<endl;
		if(res>radius)
		cout<<"SAFE"<<endl;
		if(res==radius)
		cout<<"JUST MISSED"<<endl;
		
	}
    //cout << "Hello World!" << endl;
    return 0;
}
