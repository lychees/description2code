#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,i,c;
    cin>>t;
    while(t--)
    {
    	c=0;
    	//flag=0;
    	cin>>n;
    	int a[n], f[n];
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    		f[i]=0;
    	}
    	if(a[0]==-1)
    	cout<<"Win\n";
    	else
    	{i=0;
    	//f[0]=1;
    	while((f[i]!=1&&a[i]!=-1)&&c<n)
    	{
    		f[i]=1;
    		i=a[i];
    		c++;
    	}
    	if(a[i]==-1)
    	cout<<"Win\n";
    	else
    	cout<<"Lose\n";
    	}
    }
    return 0;
}
