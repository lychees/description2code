int main()
{
    string str1,str2;
    cin>>str1>>str2;

    lli dp[str1.size()+2][str2.size()+2];

    memset(dp,0,sizeof(dp));
    for(i=0;i<str2.size();i++)dp[0][i]=1;

    for(i=1;i<=str1.size();i++)
    {
        for(j=1;j<=str2.size();j++)
        {
            if(str1[i-1]==str2[j-1]){
                dp[i][j]=dp[i-1][j-1] + dp[i][j-1] ;
            }
            else dp[i][j]=dp[i][j-1];
           // cout<<dp[i][j]<<"  ";
        }
        //cout<<endl;
    }
    lli ans=dp[str1.size()][str2.size()];
