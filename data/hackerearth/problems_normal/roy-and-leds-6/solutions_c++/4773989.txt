#include <iostream>
using namespace std;

int main()
{
    long int t,i,r,g,b;
    cin>>t>>r>>g>>b;
    int red=1,blue=3,green=5;
    int R=0,G=0,B=0,Y=0,C=0,M=0,W=0,Bl=0;
    //color code + R 1 G 5 Y 6 C 8 M 4 W 9 B 0 
    int sum=0;
    for(i=0;i<t;i++)
    {sum=0;
    	if(i%(2*r)>r-1){sum+=red;}
    	if(i%(2*g)>g-1){sum+=green;}
    	if(i%(2*b)>b-1){sum+=blue;}
    	//cout<<sum<<" "<<i<<endl;
    	if(sum==1){R++;}
    	else if(sum==5){G++;}
    	else if(sum==3){B++;}
    	else if(sum==6){Y++;}
    	else if(sum==8){C++;}
    	else if(sum==4){M++;}
    	else if(sum==9){W++;}
    	else{Bl++;}
    	
    }
    cout<<R<<" "<<G<<" "<<B<<" "<<Y<<" "<<C<<" "<<M<<" "<<W<<" "<<Bl;
    
}
