#include <bits/stdc++.h>
using namespace std;

int parity(unsigned x) 
{
   unsigned y;
   y = x ^ (x >> 1);
   y = y ^ (y >> 2);
   y = y ^ (y >> 4);
   y = y ^ (y >> 8);
   y = y ^ (y >>16);
   return y & 1;
}
int main()
{
	int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
       if(parity(n)==0)
         cout<<"even"<<endl;
       else
         cout<<"odd"<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
