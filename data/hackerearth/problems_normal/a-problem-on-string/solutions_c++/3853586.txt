#include <iostream>
#include <cstdio>
#include <cstdlib>

using namespace std;

int main()
{
	long N, Q, i,j;
	char *str;
	scanf("%ld",&N);
	str = (char*)malloc((N+2)*sizeof(char));
	scanf("%s",str);
	scanf("%ld",&Q);
	char X,Y;
	long firstCount =0, secondCount=0, total;
	for(i=0;i<Q;i++) {
	   getchar();
	   scanf("%c",&X);
		//X = getchar();
		getchar();
		scanf("%c",&Y);
		//Y = getchar();
		//getchar();
		firstCount = 0;
		secondCount = 0;
		
		if(X == Y) {
			for(j=0; j< N;j++) {
				if(str[j] == X) {
					firstCount++;
				}
			}
			total = firstCount * (firstCount -1)/2;
		} else if(X != Y) {
				for(j=0; j< N;j++) {
				if(str[j] == X) {
					firstCount++;
				} else if(str[j] == Y) {
					secondCount++;
				}
				}
				total = firstCount*secondCount;
		}
		cout << total << endl;
	//	printf("%d\n",total);
	}
	free(str);
    return 0;
}
