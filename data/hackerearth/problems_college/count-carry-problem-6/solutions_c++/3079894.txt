using namespace std;

typedef long long ll;

const int N = 2000001, md = 1e9 + 7;
const double Eps = 0.01;

using namespace std;

int n, t, s, m;
char A[N], B[N];
int main() {
    //freopen("in","r",stdin);
   scanf ("%d", &t);
   while (t--)
   {
       scanf ("%s%s", &A, &B);
       n = strlen(A);
       m = strlen(B);
       s = 0;
       n --;
       m --;
       int car = 0;
       while (n >= 0 || m >= 0)
       {
            int F = n >= 0 ? A[n] - '0' : 0;
            int S = m >= 0 ? B[m] - '0' : 0;
            int sum = S + F + car;
            car = sum/10;
            if (car)
                s++;
            n--;
            m--;
       }
       if (!s)
           puts("No carry operation");
       else
            if (s == 1)
                puts("1 carry operation");
            else
                printf("%d carry operations\n", s);
   }
}
