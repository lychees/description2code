#include <iostream>
#include<string>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int k;
    	cin>>k;
    	string s;
    	cin>>s;
    	vector<long> cum(s.length());
    	cum[0]=s[0]-96;
    	for(int i=1;i<s.length();i++)
    	{
    		cum[i]=cum[i-1]+s[i]-96;
    	}
    	int start=0;
    	int end=s.length()-1;
    	int mid;
    	int f=0;
    	int ans=0;
    	int x;
    	for(int i=0;i<s.length();i++)
    	{
    		end=s.length()-1;
    		x=k+cum[i];
    	
    	if(cum[i]==k)
    	{
    		++ans;
    		//cout<<"cool";
    	}
   
    //	if(start>=s.length()-1)
    ///	break;
    	
    	while(start<=end)
    	{
    		
    		mid=(start+end)/2;
    		if(cum[mid]==x)
    		{
    			start=mid;f=1;
    		//	cout<<"cool"
    			break;
    		}
    		else if(cum[mid]>x)
    		end=mid-1;
    		else
    		start=mid+1;
    	}
    	if(f==1)
    	{
    		++ans;
    		f=0;
    	}
    	
    
    	}
    //	for(int i=0;i<s.length();i++)
    //	cout<<cum[i]<<" ";
    //	cout<<endl;
    	cout<<ans<<endl;
    }
}
