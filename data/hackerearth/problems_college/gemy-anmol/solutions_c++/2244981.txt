#include<bits/stdc++.h>
using namespace std;

int main() { 
    long long int i,j,k,a[100],t,n,m;
    cin >> t;
    while(t--) {
         cin >> n >> m;
         for(i=0;i<n;i++)
           cin >> a[i];
         sort(a,a+n);
         for(i=n-1;i>=0;i--) {
            if(a[i]<m) {
            	m-=a[i];
            }else {
            	cout << n-i << endl;
            	break;
            }
         }
 
          if(i==-1) 
            cout << i << endl;

    }

	return 0;
}
