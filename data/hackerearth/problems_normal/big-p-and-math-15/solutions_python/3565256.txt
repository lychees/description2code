def convert_five(l):
	x = map(lambda x : '6' if x == '5' else x, l)
	return int(''.join(x))	
	
def convert_six(l):
	x = map(lambda x :'5' if x == '6' else x, l)
	return int(''.join(x))
	
	
a, b = map(list, raw_input().split())

mx = convert_five(a) + convert_five(b)

mn= convert_six(a) + convert_six(b)

print mn, mx
