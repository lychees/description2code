from sys import stdin
from math import sqrt
maxi=1000010
comp=[False]*maxi
primes=[]

def soe():
	for i in range(2,maxi):
		if comp[i]==True:
			continue
		for j in range(i*i,maxi,i):
			comp[j]=True
	for i in range(2,maxi):
		if not comp[i]:
			primes.append(i)

t=int(stdin.readline())
for i in range(0,t):
	soe()
	n=int(stdin.readline())
	ans=1
	for i in primes:
		if i>n:
			break
		k=0
		while n%i==0:
			k+=1
			n/=i
		ans*=(k+1)
	if n>1:
		ans*=2
	print ans-1
