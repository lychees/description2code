#include<iostream>
#include<stdlib.h>
#include<stdio.h>
using namespace std;

int main()
{  // long long int ans=2;
int t;
long long int k;
//cout<<ans;
int ar[150000];
int index=0;

    cin>>t;
    while(t--)
    {

        cin>>k;
        while(k>0)
        {

            int j=k%3;
            k=k/3;
            if(j==0) k--;
            ar[index++]=j;
         }
        for(int j=index-1;j>=0;j--)
        {
            if(ar[j]!=0) cout<<ar[j]+1;
            else cout<<"4";
        }
        cout<<endl;
            index=0;
    }
}
