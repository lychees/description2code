#include <iostream>
#include <cstdio>
#include <string>
using namespace std;

int main()
{
    int t,i,j,x,y,z,n,zero=0,one=0;
    string s;
    scanf("%d",&t);
    for(i=0;i<t;i++){
    	cin>>s>>n;
    	zero = one=0;
    	x = n%2;
        y = (n/2)%2;
        z = x%2;
    	for(j=0;j<s.size();j++){
    		if(s[j]=='0'){
    			zero++;
    		}
    		else
    			one++;
    	}
    	if((x==1 && y==1 && z==1) || (x==0 && y==0 && z==0)){
    		printf("%d\n",one);
    	}
    	else
    		printf("%d\n",zero);
    	
    }
    return 0;
}
