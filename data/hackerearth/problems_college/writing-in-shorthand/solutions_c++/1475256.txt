#include <iostream>
#include <cstring>

using namespace std;

int n,m;
char l1[3001][11],l2[3001][11];

char* check(char *word){
	
	for(int i=0;i<m;i++){
		if(strcmp(l1[i],word)==0 || strcmp(l2[i],word)==0){
			if(strlen(l1[i])<=strlen(l2[i]))
				return l1[i];
			else
				return l2[i];
		}
	}
	
}

int main()
{
  	char word[11];
    cin>>n>>m;
    for(int i=0;i<m;i++){
    	cin>>l1[i]>>l2[i];
    }
    for(int i=0;i<n;i++){
    	cin>>word;
    	cout<<check(word)<<" ";
    }
    return 0;
}
