#include<iostream>
using namespace std;
int main()
{
	int t,n,i;
	cin>>t;
	long long a[1005],sum,p,q;
	while(t--)
	{
		cin>>n;
		sum=0;
		for(i=0;i<n;i++)
		{
		cin>>a[i];
        	}
	for(i=0;i<n;i++)
	{
		sum+=a[i];
	}
	p=sum/n;
	q=sum%n;
	if(q==0)
	{
		cout<<"YES"<<endl;
	     cout<<p<<" "<<q<<endl;
		
	}
	else
	{
		cout<<"NO"<<endl;
		cout<<p<<" "<<q<<endl;
	}
	  
	}
	return 0;
}
