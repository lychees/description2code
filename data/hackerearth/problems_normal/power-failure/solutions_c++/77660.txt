#include <iostream>
#include<algorithm>
#include<cstdio>

using namespace std;

#define FOR(i,n) for(int i=0;i<n;++i)
#define FORS(i,s,n) for(int i=s;i<n;++i)
#define MAX 100000
#define MOD 1000000007 

void print(int a[], int n)
{
        FOR(i,n)
        {
                cout<<a[i]<<",";
        }

         cout<<"\n";
}

void sieve(bool a[],int n) /* 1 to n, bool a[n+1] */
{
   if(n<1) return;

    for(int i=1;i<=n;i++)
        a[i] = true;

   a[1] = false;

   for(int i=2;i<=n;i++)
    {
        if(a[i]) //prime
        {
            int j=2;
            while(j*i<=n)
            {
                    a[j*i] = false;
                    j++;
            }
        }
    }
}

bool cmp(const int& a, const int& b)
{
	return a>b;
}

int main()
{
 	int t,n, m, v[MAX], cache[MAX+2];
	bool hash[MAX+1];
	sieve(hash, MAX);

	scanf("%d", &t);

	FOR(T,t)
	{
		scanf("%d%d", &n, &m);

		FOR(i,n)
		{
			scanf("%d", &v[i]);
		}

		sort(v,v+n, cmp);
		cache[m+1] = 0;
	
		for(int i=m;i>0;--i)
		{
			cache[i] = cache[i+1] + (hash[i]?0:1);
		}

		//print(cache, m+1);
		long long ways = 1;

		FOR(i,n)
		{
			ways = (ways * (cache[v[i]] -i))%MOD;
		}

		printf("%lld\n", ways);
	}

  	return 0;
}
