#include <bits/stdc++.h>
#define ll long long
using namespace std;
int a[1010];
int main()
{
	int test;
	scanf("%d",&test);
	while(test--)
	{
		int n,k;
		scanf("%d%d",&n,&k);
		int i,j,mc;
		for(i=0;i<n;i++)
		{
			scanf("%d",(a+i));
		}
		mc=0;//mycount
		int mymin,index;
		for(i=0;i<n;i++)
		{
			mymin=a[i];
			index=i;
			for(j=(i+1);j<n;j++)
			{
				if((mc+(j-i))>k)
				{
					break;
				}
				else if(a[j]<mymin)
				{
					mymin=a[j];
					index=j;
				}
			}
			if(mymin<a[i])
			{
				for(j=(index);j>i;j--)
				{
					a[j]=a[j-1];
				}
				a[j]=mymin;
				mc+=(index-i);
			}
			if(mc==k)
			{
				break;
			}
		}
		for(i=0;i<n;i++)
		{
			printf("%d ",a[i]);
		}
		printf("\n");
	}
	return 0;
}
