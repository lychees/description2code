#include <bits/stdc++.h>

using namespace std;

const int MAXN = 100000 + 2;

int g[70][70][70];

void init() {
    g[0][0][0] = 0;
    for(int i = 0; i <= 50; i++) {
        for(int j = 0; j <= 50; j++) {
            for(int k = 0; k <= 50; k++) {
                if (i + j + k == 0) continue;
                set<int> a;
                for(int t = 1; t <= i; t++) a.insert(g[i - t][j][k]);
                for(int t = 1; t <= j; t++) a.insert(g[i][j - t][k]);
                for(int t = 1; t <= k; t++) a.insert(g[i][j][k - t]);
                int t = -1;
                g[i][j][k] = -1;
                //cout << a.size() << endl;
                for(set<int>::iterator it = a.begin(); it != a.end(); it++) {
                    t++;
                    if (t != *it) {
                        g[i][j][k] = t;
                        break;
                    }
                }
                if (g[i][j][k] == -1) g[i][j][k] = t + 1;
            }
        }
    }
}

int main()
{
	//re-submit
    int test;
    cin >> test;
    init();
    while (test --) {
        int n;
        cin >> n;
        int x = 0;
        for(int i = 0; i < n; i++) {
            int a, b, c;
            cin >> a >> b >> c;
            x ^= g[a][b][c];
        }
        if (x > 0) cout << "Stark" << endl;
        else cout << "Lannister" << endl;
    }
}
