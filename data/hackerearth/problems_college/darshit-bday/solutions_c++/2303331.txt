#include <iostream>
using namespace std;
int min(int a, int b)
{
	if(a<b)
		return a;
	else
		return b;
}
int main()
{
    unsigned long long int t;
    unsigned long long int b,w,x,y,z;
    unsigned long long int res1=0,res2=0,res3=0,res4=0,res5=0,res = 0;
    cin>>t;
    while(t--){
    	
    	
    	cin>>b>>w;
    	cin>>x>>y>>z;
    
    			res1 = x*b;
    			res1+=x*w;
    			res1+=w*z;
    		
    			res2 = y*w;
    			res2+= y*b;
    			res2+= b*z;
    			
    			res3= x*b + y*w;
    			res = min(res1,min(res2,res3));
   	
    	cout<<res<<endl;
    }
    return 0;
}
