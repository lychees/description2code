#include <iostream>
using namespace std;

#define max 101
int main()
{
	int testCase,stairsCount;
	long noOfWays[max];
	noOfWays[0]=0; noOfWays[1]=1;noOfWays[2]=2;
	cin>>testCase;
	for(int i=3;i<max;i++) noOfWays[i] = noOfWays[i-1] + noOfWays[i-2];

	for(int i=0;i<testCase;i++){
		cin>>stairsCount;
//		cout<<" stairsCount="<<stairsCount;
		cout<<noOfWays[stairsCount]<<"\n";	
	}
    return 0;
}
