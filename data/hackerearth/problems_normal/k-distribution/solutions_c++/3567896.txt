using namespace std;

long long int dp[100005][105]={0};
int n,k;
long long int ar[105];
long long int powe[105];
int vis[100005][105];
int timer=0;
long long int mod=1000000007;

long long int find(long long int s,int i)
{
    if(i==n)
    {
        if(s<k)
            return 1;
        return 0;
    }
    if(s>=k)
        return 0;
    if(vis[s][i]==timer)
        return dp[s][i];
    vis[s][i]=timer;
    return dp[s][i]=((find(s+ar[i],i+1)+find(s,i+1))%mod);
}

int main()
{
    int t;
    powe[0]=1;
    t=1;
    while(t<105)
        powe[t]=(2*powe[t-1])%mod,t++;
    //scanf("%d",&t);
    t=1;
    while(t--)
    {
        long long int s=0;
        timer++;
        cin>>n>>k;
        int i=0;
        while(i<n)
        {
            cin>>ar[i];
            s+=ar[i];
            i++;
        }
        if(s-k<k)
        {
            cout<<"0\n";
           break;
        }
        s=find(0,0);
        s=(s*2)%mod;
        s=powe[n]-s;
        if(s<0)
            s+=mod;
        cout<<s<<endl;
    }
    return 0;
}
