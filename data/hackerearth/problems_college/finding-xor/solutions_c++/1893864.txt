#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	char s[10001];
    	char s1[10001];
    	cin>>s;
    	cin>>s1;
    	int i=0;
    	while(s[i]!='\0')
    	{
    		if(s[i]==s1[i])
    		cout<<0;
    		else
    		cout<<1;
    		i++;
    	}
    	cout<<endl;
    }
    return 0;
}
