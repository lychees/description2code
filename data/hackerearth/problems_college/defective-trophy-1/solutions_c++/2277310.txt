#include <iostream>
#include <cmath>
using namespace std;
int fun(int n)
{
	if(n==0||n==1)
	return 0;
	else 
	return 1+fun((int) ceil (n / 3.0));
}
int main()
{
    //cout << "Hello World!" << endl;
    int t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	int res=fun(n);
    	cout<<res<<"\n";
    }
    return 0;
}
