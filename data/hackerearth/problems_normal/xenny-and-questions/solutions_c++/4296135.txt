#include<bits/stdc++.h>
using namespace std;
void sol()
{
   long long int i,n;
   long long  int k;
   scanf("%lld%lld",&n,&k);
   long long a[n+1];
    for(i=0;i<n;++i)
        {
            scanf("%lld",&a[i]);
        }
   long long int t=0;
   sort(a,a+n);
    for(i=n-1;i>=0&&k>0;--i)
    {
    	t+=a[i];
    	--k;
    }
    
   printf("%lld\n",t);
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--) sol();
    return 0;
}
