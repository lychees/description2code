#include <iostream>
#define read(x) std::cin>>(x)
typedef unsigned long long ull;
using namespace std;

template <typename T>
void print(T x){std::cout<<x<<'\n';}

int main()
{
	int t;
    read(t);
    while(t--)
    {
    	int n;
    	ull l=0;
    	read(n);
    	while(n--)
    	l+=n;
    	print(l);
    }
    return 0;
}
