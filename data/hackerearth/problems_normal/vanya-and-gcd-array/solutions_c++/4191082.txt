using namespace std;


#define LL long long int
#define sc1(x) scanf("%d", &x)
#define sc2(x, y) scanf("%d%d", &x, &y)
#define sc3(x, y, z) scanf("%d%d%d", &x, &y, &z)
#define pr1(x) printf("%d\n", x)
#define FOR(i, a, n) for(int i=a;i<n;i++)
#define pp pair<int, int>
#define pb push_back 

#define MAX 501
#define MOD 1000000007LL


int gcd(int a, int b) { return b ? gcd(b, a%b) : a; }


int arr[MAX];
int cnt[102][102], temp[102][102];


int main() {

	int n, x;

	sc1(n);
	FOR(i, 0, n) {
		sc1(x);

		FOR(j, 1, x) {
			FOR(k, 1, 101) {
				int d = gcd(x, k);
				temp[x][d] = (temp[x][d] + cnt[j][k]) % MOD;
			}
		}

		FOR(j, 1, 101) {
			FOR(k, 1, 101) {
				cnt[j][k] = (cnt[j][k] + temp[j][k]) % MOD;
				temp[j][k] = 0;
			}
		}
		cnt[x][x]++;
	}

	int result = 0;
	FOR(i, 1, 101) result = (result + cnt[i][1]) % MOD;
	pr1(result);

	return 0;
}
