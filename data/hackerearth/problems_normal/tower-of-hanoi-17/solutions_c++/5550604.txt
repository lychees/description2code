#include <iostream>
#include <vector>
#include <algorithm>
#include <functional>
using namespace std;

vector<long long>dp(202);

auto main()->int
{
    long long t,n;
    cin>>t;
    long long x,y;
    while(t--)
    {
        cin>>n;
        vector<pair<long long, long long>>v;
        for(auto i=0;i<n;i++)
        {
            cin>>x>>y;
            v.push_back(make_pair(x,y));
        }
        sort(v.begin(),v.end());
        dp[0] = v[0].second;
        long long ans = dp[0];
        for(auto i=1;i<n;i++)
        {
            dp[i] = v[i].second;
            for(auto j=0;j<i;j++)
            {
                if(v[i].second>v[j].second && v[i].first>v[j].first)
                {
                    dp[i] = max(dp[i],dp[j]+v[i].second);
                }
            }
            ans = max(ans,dp[i]);
        }
        cout<<ans<<endl;
    }
}
