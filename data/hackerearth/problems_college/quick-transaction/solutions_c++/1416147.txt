#include <bits/stdc++.h>
using namespace std;
int main()
{
	std::ios_base::sync_with_stdio(false);
	int i,j,k,l,ans=0;
	int A[10]={1,2,5,10,20,50,100,500,1000};
	cin>>i;
	for(j=0;j<i;j++)
	{
		ans=0;
		cin>>k;
		for(l=8;l>=0;l--)
		{
			if((k/A[l])>0)
			{
				ans+=(k/A[l]);
				k=k%A[l];
			}
		}
		cout<<ans<<endl;
	}
	return 0;
}
