#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main(){
    ll A, B; cin >> A >> B; A--;
    ll num[] = {2, 6, 12, 60, 420, 840, 2520, 27720, 360360, 720720, 12252240, 232792560, 5354228880, 26771144400, 80313433200, 2329089562800, 72201776446800, 144403552893600, 5342931457063200}, cntM[19] = {2,3,2,2,3,2,2,2,3,2,2,2,2,2,2,2,3,2,2};
    ll cnt[19] = {}, cntS[20] = {};
    ll length = B-A, impars = length-(B/2 - A/2),ans = length+impars;
    for(int x = 18; x >= 0; x--){
        cnt[x] = B / num[x] - A / num[x] - cntS[x+1];
        cntS[x] = cnt[x] + cntS[x+1];
        ans += cnt[x] * cntM[x];
    }
    cout << ans - (A==1?2LL:0LL) << '\n';
}
