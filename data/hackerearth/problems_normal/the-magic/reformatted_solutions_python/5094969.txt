def binary(num):  # To count number of 1's in binary equivalent
    ret = 0
    while (num != 0):
        bit = num % 2
        if bit == 1:
            ret = ret+1
        num = num/2
    return ret


T = int(raw_input())
for t in range(0, T):
    i = int(raw_input())
    p = binary(i)
    print p
