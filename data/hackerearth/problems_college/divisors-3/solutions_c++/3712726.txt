#include <bits/stdc++.h>
#define sz(x) ((int)x.size())
using namespace std;
#define F first
#define S second
#define int long long
typedef pair<int,int> pii;
const int N = 100;
inline int in() { int x; scanf("%lld", &x); return x;}

int32_t main() {
    int n = in();
    vector<pii> p;
    for (int i = 2; i * i <= n; i++) {
        if (n % i == 0) {
            p.push_back({i, 0});
            while (n % i == 0)
                p.back().S++, n /= i;
        }
    }
    if (n > 1)
        p.push_back({n, 1});
    int kol = 1;
    for (int i = 0; i < sz(p); i++)
        kol *= (p[i].S + 1);
    int Q = in();
    while (Q--) {
        int t = in(), k = in();
        vector<int> v;
        for (int i = 0; i < sz(p); i++) {
            int x = 0;
            while (k % p[i].F == 0)
                x++, k /= p[i].F;
            v.push_back(x);
        }
        if (t == 1) {
            int ans = 1;
            for (int i = 0; i < sz(v); i++)
                ans *= (min(v[i], p[i].S) + 1);
            printf("%lld\n", ans);
            continue;
        }
        else {
            int ans = (int)(k == 1);
           // cerr << k <<' ' << ans << endl;
            for (int i = 0; i < sz(v); i++) {
                if (v[i] > p[i].S) ans = 0;
                else
                    ans *= (p[i].S - v[i] + 1);
            }
            if (t == 2)
                printf("%lld\n", ans);
            else
                printf("%lld\n", kol - ans);
        }
    }
}
