#include <iostream>
using namespace std;

int main()
{
   int tc,a,i,j,k;
   cin>>tc;
   while(tc--)
   {int c=0;
   	cin>>a;
   	for(k=0;k<=(a/5);k++)
   	{
   		for(j=0;j<=(a/3);j++)
   		{
   			for(i=0;i<=a;i++)
   			{
   				if((i+(3*j)+(5*k))==a)
   				c++;
   			}
   		}
   	}cout<<c<<"\n";
   }
    return 0;
}
