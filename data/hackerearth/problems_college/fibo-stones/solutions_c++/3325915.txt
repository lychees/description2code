#include <bits/stdc++.h>
#define ll long long
using namespace std;


ll dp[81];
ll fib(int n)
{
	if(dp[n]!=-1)
		return dp[n];
	else if(n==1 || n==2)
		return dp[n]=1;
	else
		return dp[n]=(fib(n-1)+fib(n-2));
}

int main()
{
    memset(dp,-1,sizeof(dp));
    //fib(81);
    ll n,m,i,a[81];
    ll sum=0,s=0;
    cin>>n>>m;
    fib(n);
    for(i=1;i<=n;++i) {
    	//cout<<dp[i]<<"\n";
    	sum+=dp[i];
    }
    for(i=1;i<=m;++i) {
    	cin>>a[i];
    	s+=a[i];
    }
    
    cout<<(sum-s)<<"\n";
    
    return 0;
}
