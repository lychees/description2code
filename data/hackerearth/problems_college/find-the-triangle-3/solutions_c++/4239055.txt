#include <bits/stdc++.h>
using namespace std;

int main() {
	int t;
	cin>>t;
	while(t--)
	{
	int n;
	cin>>n;
	int f=0;
	int a[n];
	
	for(int i=0;i<n;i++)
	cin>>a[i];
	if(n<=2)
	{
		cout<<"-1"<<endl;
		continue;
	}
	sort(a,a+n);
	for(int i=n-3;i>=0;i--)
	{
		if((a[i]+a[i+1])>(a[i+2]))
		{
			f=1;
			cout<<a[i]<<" "<<a[i+1]<<" "<<a[i+2]<<endl;
			break;
		}
		
	}
	if(f==0)
	{
		cout<<"-1"<<endl;
	}
	}
	return 0;
}
