#include <iostream>
#include <map>
#include <cmath>
using namespace std;

int main() {
	int n =10000;
	 map<int,bool> number;
       for( int i =2;i<n;i++) 
       { 
       number.insert(pair<int,bool>(i,0));}
       	//cout << "---------"<<"\n";
      
       
       int count;
       cin >> count;
       map<int,bool>::iterator it;
      for (int i =2;i<=pow(n,0.5);i++)
      {
          if (number[i]==0)
          { 
          for(int j =i*i;j<=n;j=j+i)
              {	
                 number[j]=1;
              }
          }
      }
      int no=0;
      int ans;
      for(it = number.begin();it!=number.end();it++)
        {
            if(it->second==0)
            
            {		ans = it->first;
            //cout << it->first<< "\n";
                no = no+1;
                if(no==count)
                	break;
            }
        }
       cout << ans;
       return 0 ;
}
