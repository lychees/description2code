#include <iostream>
#include<string> 
#include<cstring> 
using namespace std;  
   
int main ()
{
   long long l,result,i,j;
   int n;
   cin>>n;
   for(int d=0;d<n;d++)
   {
     cin>>l;
     result=0;
     while(l!=0)
     {
     if(l%2==0)
        j=l/2;
     else
        j=l/2+1;
     result=result+j*j; 
     l=l-j; 
     }
     cout<<result<<endl;
   }
   return 0;
}
