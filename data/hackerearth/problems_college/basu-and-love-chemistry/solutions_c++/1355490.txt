#include <bits/stdc++.h>
using namespace std;
inline long long int fast(long long int a, long long int b,long long int m)
{
    long long int prod=1;
    while(b)
    {
        if(b&1)prod=(prod*a)%m;
        a=(a*a)%m,b/=2;
    }
    return prod;
}
int main() 
{
	
	string s;
	cin>>s;
	long long int multiinverse=111111112,MOD=1000000007,coef,res=0;
	int len=s.size();
	for(int i=0;i<len;i++) 
	{
    coef = (( ( fast(10, len - i, MOD) - 1) * multiinverse ) % MOD) * (i + 1) % MOD;
    res += ( coef * (s[i] - '0') ) % MOD;
}
printf("%lld\n", res % MOD );
	return 0;
}
