#include <iostream>
using namespace std;
char arr[1003][1003];
int check(int i,int f)
{
	if(arr[i][f]=='/'&&arr[i][f+1]=='\\' &&arr[i+1][f]=='\\' && arr[i+1][f+1]=='/')
	return 1;
	return 0;



}


int main()
{int t;
int n,m;
cin>>t;
while(t--)
{cin>>n>>m;
for(int i=0;i<n;i++)
	for(int j=0;j<m;j++)
		cin>>arr[i][j];
int ans=0;
for(int i=0;i<n-1;i++)
	for(int f=0;f<m-1;f++)
		ans+=check(i,f);
	
	
	
	
cout<<ans<<"\n";	
}
   
}
