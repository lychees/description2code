#include <iostream>
using namespace std;
long long int f[100];
int main()
{
    long long int i,n,m,x,ans;
    f[1]=1;
    f[2]=1;
    for(i=3;i<=80;i++)
    {
    	f[i]=f[i-1]+f[i-2];
    }
    cin>>n>>m;
    ans=0;
    for(i=1;i<=n;i++)
    {
    	ans+=f[i];
    }
    for(i=1;i<=m;i++)
    {
    	cin>>x;
    	ans-=x;
    }
    cout<<ans;
    return 0;
}
