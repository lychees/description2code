#include <iostream>
#include <algorithm>
#include <stdio.h>
using namespace std;
#define get getchar
inline int input(){
	int n = 0, s = 1 ;
	char p = get();
	if (p == '-') {
		s = -1 ;
	}
	while ((p < '0' || p > '9') && p != EOF && p != '-') {
		p = get( ) ;
	}
	if (p == '-') {
		s = -1, p = get() ;
	}
	while (p >= '0' && p <= '9') {
		n = (n << 3) + (n << 1) + (p - '0') ;
		p = get() ;
	}
	return n * s ;
}
int a[100000];
int main(){
	cout.sync_with_stdio(0);
	cin.tie(0);
	int t,n,k,i;
	long long ans;
	//cin>>t;
	t=input();
	while(t--){
		ans=0;
		//cin>>n>>k;
		n=input();
		k=input();
		if(k>n)
			k=n;
		for(i=0;i<n;++i)
			a[i]=input();
		//	cin>>a[i];
		sort(a,a+n);
		for(i=n-1;i>=n-k;--i)
			ans+=a[i];
		cout<<ans<<"\n";
	}
	return 0;
}
