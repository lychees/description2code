#include<bits/stdc++.h>
using namespace std;



int main()
{
	int n,m;
	cin>>n>>m;
	long long a[n][m];
	
	vector<long long> row,col;

	long long ans=0;

	for(int i=0;i<n;i++)
	{
		ans=0;
		for(int j=0;j<m;j++)
		{
			cin>>a[i][j];
			ans+=a[i][j];
		}
		row.push_back(ans);
	}

	for(int i=0;i<m;i++)
	{
		ans=0;
		for(int j=0;j<n;j++)
		{
			ans+=a[j][i];
		}
		col.push_back(ans);
	}


	int l = row.size();
	int l1 = col.size();
	
	ans=-1;

	for(int i=0;i<l;i++)
	{
		for(int j=0;j<l1;j++)
		{
			ans = max(ans,(row[i]^col[j]));
		}
	}
	cout<<ans<<endl;

}
