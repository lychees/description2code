#include<bits/stdc++.h>
using namespace std;

int main()
{
   int t;
   cin >> t;
   int zer[50], on[50];
   memset ( zer, 0, sizeof (zer) );
   memset ( on, 0, sizeof (on) );

   zer[0] = 1;
   on[0] = 0;

   zer[1] = 0;
   on[1] = 1;
  
   for ( int i = 2; i <= 40; i++ )
   {
      zer[ i ] = zer[i-1] + zer[ i-2 ];
      on[i] = on[i-1] + on[i-2]; 
   }

   while ( t-- )
   {
      int n;
      cin >> n;
      cout << zer[n] << " " << on[n] << endl;
   }
   return 0;
}
