#include <iostream>
#include <cstdio>
#include <algorithm>
#include <climits>
using namespace std;
const int maxn = 10010;
const int maxp = 1000100;

int c[maxn], phi[maxp], mini[maxp];
bool sieve[maxp];

void prepare()
{
     for (int i = 1; i < maxp; i++)
          phi[i] = i;
     for (int i = 2; i < maxp; i+=2)
          phi[i] -= phi[i]/2;
     for (int i = 3; i < maxp; i+=2)
          if (phi[i] == i) {
              for (int j = i; j < maxp; j += i) {
                   phi[j] -= phi[j] / i;
              }
          }
     for (int i = 0; i < maxp; i++)  mini[i] = 123456789;
     for (int i = 1; i < maxp; i++) {
          //cout << mini[phi[i]] << " " << phi[i] << " " << i << endl;
          if (mini[phi[i]] == 123456789)
              mini[phi[i]] = i;
     }
     for (int i = maxp - 2; i >= 0; i--)
          mini[i] = min(mini[i], mini[i + 1]);
     return;
}

int main() {
	prepare();
	int cases;
	cin >> cases;
	while (cases--) {
	            int n;
	            cin >> n;
	            long ans = 0;
	            for (int i = 0; i < n; i++) {
	            	 int c;
	            	 cin >> c;
	            	 ans += mini[c];
	            }
	            cout << ans << endl;
	}
	return 0;
}
