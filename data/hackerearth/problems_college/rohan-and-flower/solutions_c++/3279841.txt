#include<stdio.h>
#define ll long long int
ll ar[2005];
int main()
{
	ll n,m,i,j,k,ans,temp;
	scanf("%lld%lld",&n,&m);
	for(i=1;i<=n;i++)
	{
		scanf("%lld",&ar[i]);
	}
	ans=0;
	for(i=1;i<=n;i++)
	{
		if((ar[i]%m)!=0)
		{
			for(j=i+1;j<=n;j++)
			{
				temp=ar[i]*ar[j];
				if(temp%m!=0)
				{
					for(k=j+1;k<=n;k++)
					{
						if((temp*ar[k])%m==0) ans++;
					}
				}
				else
				{
					ans+=(n-j);
				}
			}
		}
		else
		{
			if(i!=n&&i!=n-1) ans+=(((n-i-1)*(n-i))/2);
		}
	}
	printf("%lld\n",ans);
	return 0;
}
