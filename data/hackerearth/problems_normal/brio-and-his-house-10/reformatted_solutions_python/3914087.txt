def brio():
    T = int(raw_input())

    pi = 3.1415

    for i in range(T):
        tc_dict = dict()
        a, b, c = [float(x) for x in raw_input().split(' ')]
        radius = (a*b*c)/((a+b+c)*(b+c-a)*(c+a-b)*(a+b-c))**(0.5)
        area = pi*radius**2
        print("%.4f" % area)
