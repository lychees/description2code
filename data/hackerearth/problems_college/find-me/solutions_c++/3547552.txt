#include<bits/stdc++.h>
using namespace std;
int main()
{
	int tc;
	cin>>tc;
	while(tc--)
	{
		int n,p;
		cin>>n>>p;
		int ans=0;
		bool present_flag=false;
		for(int i=1;i<=n;++i)
		{
			int temp;
			cin>>temp;
			if(temp==p)
			{
				present_flag=true;
			}
			if(temp>p)ans++;
			
		}
		(present_flag)?printf("YES %d\n",ans):printf("NO 0\n");
	
	}
	return 0;
}
