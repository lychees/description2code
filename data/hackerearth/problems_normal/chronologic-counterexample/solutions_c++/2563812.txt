#include <bits/stdc++.h>

using namespace std;

int N, M;
int A[51];
int B[51];
int seq[51];
int seq2[51];
bool seen[51];

int main()
{
    scanf("%d%d", &N, &M);
    for(int i=1; i<=M; i++)
        scanf("%d%d", A+i, B+i);
    for(int i=1; i<=N; i++)
    {
        bool ok=false;
        for(int j=1; j<=N; j++) if(!seen[j])
        {
            seq[i]=j;
            int n=i;
            for(int k=N; k>=1; k--) if(!seen[k] && j!=k)
                seq[++n]=k;
            memcpy(seq2, seq, sizeof seq2);
            for(int k=1; k<=M; k++)
                sort(seq2+A[k], seq2+B[k]+1);
            for(int k=2; k<=N; k++) if(seq2[k]!=seq2[k-1]+1)
                ok=true;
            if(ok)
            {
                seen[j]=true;
                break;
            }
        }
        if(!ok)
        {
            printf("All ok!\n");
            return 0;
        }
    }
    for(int i=1; i<=N; i++)
        printf("%d ", seq[i]);
    printf("\n");
    return 0;
}
