#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		vector<int> stock;
		for(int i=0;i<n;i++){
			int x;
			cin>>x;
			stock.push_back(x);
		}
		long long int profit=0;
		int m=0;
		for(int i=n-1;i>=0;i--){
			int ai=stock[i];
			if(m<=ai){
				m=ai;
			}
			profit+=m-ai;
		}
		cout<<profit<<"\n";
		
	}
}
