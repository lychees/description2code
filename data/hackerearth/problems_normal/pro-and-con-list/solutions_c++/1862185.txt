#include<bits/stdc++.h>
using namespace std;
long long int ans[100001];

int main()
{

	long long int t,n,i,j,k,a,b,rem;
	cin>>t;
	while(t--)
	{
		cin>>n;
		rem=0;
		for(i=0;i<n;i++)
		{
			cin>>a>>b;
			ans[i]=a+b;
			rem=rem+b;
		}
		sort(ans,ans+n);
		cout<<ans[n-1]+ans[n-2]-rem<<endl;
	}	
	return 0;
}
