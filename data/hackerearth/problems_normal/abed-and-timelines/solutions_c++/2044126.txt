#include <iostream>
#include <math.h>
using namespace std;

int main()
{
    long long int t,x1,y1,x2,y2,m,n,p,ans,a,b,c,d;
    cin>>t;
    while(t--)
    {
    	ans=0;
    	cin>>n>>m;
    	cin>>x1>>y1>>x2>>y2;
    	cin>>p;
    	a=abs(x1-x2);
    	b=n-abs(x1-x2);
    	c=abs(y1-y2);
    	d=m-abs(y1-y2);
    	ans=(a>b?b:a)+(c>d?d:c);
    	ans=ans*p;
    	cout<<ans<<endl;
    	if(ans<=1000)
    	cout<<"You saved the group.\n";
    	else
    	cout<<"Let go of the group.\n";
    }
    return 0;
}
