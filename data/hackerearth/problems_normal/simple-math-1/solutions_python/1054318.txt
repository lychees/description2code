N, R = [int(i) for i in raw_input().split()]
grid = [[0]*N for i in range(R)]

for i in range(N):
	grid[0][i] = 1

for i in range(1, R):
	if N > i:
		sum_ = grid[i-1][i-1]
		for j in range(i, N):
			grid[i][j] = sum_
			sum_ = (sum_ + grid[i-1][j]) % 10000007
if grid[R-1][N-1] == 0:
	print "-1"
else:
	print grid[R-1][N-1]
