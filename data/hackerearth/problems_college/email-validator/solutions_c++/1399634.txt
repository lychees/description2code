#include<stdio.h>
#include<string.h>
int work()
{
    int a,b,d,i,ii,j,p;
    char c;
    char str[200];
    scanf("%s",&str);
    char *ch=strstr(str,"@");
    if(ch==NULL)
        return 0;
    p=ch-str;
    if(p<1||p>16)
        return 0;
    a=b=d=0;
    for(i=0;i<p;i++)
    {
        c=str[i];
        if((c>='a'&&c<='z')||(c>='A'&&c<='Z'))
            a++;
        if(c>='0'&&c<='9')
            b++;
        if(c=='_')
            d++;
    }

    if(a+b+d!=p)
        return 0;

    char *ch1=strstr(str,".");
    if(ch1==NULL)
        return 0;
    ii=ch1-str;

    if(ii-p-1<1||ii-p-1>16)
        return 0;

    if(ii-p<2)
    return 0;
        a=b=d=0;
    for(i=p+1;i<ii;i++)
    {
        c=str[i];
        if((c>='a'&&c<='z')||(c>='A'&&c<='Z'))
            a++;
        if(c>='0'&&c<='9')
            b++;
        if(c=='-')
            d++;
    }
if(a<1)
    return 0;
    if(a+b+d!=ii-p-1)
        return 0;
    j=strlen(str)-ii-1;
    if(j<1||j>10)
    return 0;
    a=0;
    for(i=ii+1;i<strlen(str);i++)
    {
        c=str[i];
        if(c>='a'&&c<='z')
            a++;
    }
    if(a!=j)
        return 0;

    return 1;
}
int main()
{
    int ntc;
    scanf("%d",&ntc);
    while(ntc--)
    {
        int k;
        k=work();
        if(k==0)
            printf("NO\n");
        else
            printf("YES\n");
    }
}
