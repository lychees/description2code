#include <bits/stdc++.h>
using namespace std;
 
#define ll long long
 
ll a[1000005];
ll dp[1000005];
 
int main(){
    ll t,n,mod;
    cin>>t;
    while(t--){
    	memset(a,0,sizeof(a));
    	memset(dp,0,sizeof(dp));
    	cin>>a[0]>>a[1]>>n>>mod;
    	dp[a[0]]++;
    	dp[a[1]]++;
    	for(ll i=2;i<n;i++){
    		a[i]=(a[i-1]+a[i-2])%mod;
    		dp[a[i]]++;
    	}
    	ll ans=0;
    	for(ll i=0;i<mod;i++){
    		ans+=(dp[i]*dp[i]);
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
