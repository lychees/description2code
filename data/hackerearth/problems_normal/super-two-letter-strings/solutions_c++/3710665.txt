#include <bits/stdc++.h>
//war declare
using namespace std;
int dp[100005];
int main()
{
	int t;
	cin >> t;
	int n,p;
	int i,j;
	while(t--)
	{
		memset(dp,0,sizeof(dp));
		dp[0] = 1;
		cin >> n >> p;
		for(i = 1; i <= n ; i++)
		{
			for(j = i - 1; j >= 0 && (i - j <= p);j--)
			{
				dp[i] = dp[i] + dp[j];
				dp[i] = dp[i] % 1000000007;
			}
		}
		cout<<dp[n]<<endl;
	}
	return 0;
}
