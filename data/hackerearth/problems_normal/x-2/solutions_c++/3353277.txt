#include<bits/stdc++.h>

using namespace std;

typedef long long ll;

const int max_n= 1000010;

ll n, m, a[max_n], v[max_n];

int main()
{
    ll t, i, j, k, len, sum, ans;

    scanf("%lld", &t);

    while(t--)
    {
        memset(v, 0, sizeof(v));

        scanf("%lld %lld", &n, &k);

        for(i=0;i<n;++i)
        {
            scanf("%lld", &a[i]);
        }

        i=0;j=-1;

        sum= 0;

        while(i<n)
        {
            if((sum>k) || (j==n))
            {
                //cout<<"i= "<<i<<"    j= "<<j<<"   sum= "<<sum<<endl;

                v[j-i]++;

                sum-= a[i];
                ++i;
            }
            else
            {
                ++j;
                sum+= a[j];
            }
        }

        for(i=n;i>0;--i)
        {
            v[i]+= v[i+1];
        }

        for(i=1;i<=n;++i)
        printf("%lld ", v[i]);

        printf("\n");
    }

    return 0;
}
