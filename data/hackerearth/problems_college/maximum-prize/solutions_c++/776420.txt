#include <bits/stdc++.h>
using namespace std;
bool myfunction (int i,int j) 
{ return (i>j); }
int main()
{
    long long int cases,n,k;
    cin>>cases;
    long long int* ar=new long long int[100000];
    while(cases--)
    {
    cin>>n>>k;
    for(int i=0;i<n;i++)
    cin>>ar[i];
	if(k>n)
		k=n;
    sort(ar,ar+n,myfunction);
        long long int sum=0;
        for(int i=0;i<k;i++)
            sum+=ar[i];

        cout<<sum<<endl;
    }
    return 0;
}
