#include <iostream>
#include <queue>
#include <utility>
#include <map>
using namespace std;

int main()
{
	int i,j,k;
    int n,m;
    priority_queue< pair <int,int> > p;
    map <int,int> mp;
    cin>>n>>m;
    //int k;
    for(i=0;i<n;i++)
    {
    	cin>>k;
    	if(mp.count(k)==0)
    	mp[k]=1;
    	else
    	mp[k]++;
    	p.push(make_pair(mp[k],k));
    	cout<<p.top().second<<" "<<p.top().first<<endl;
    }
    return 0;
}
