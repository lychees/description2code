#include<cstdio>
#include<iostream>
using namespace std;
bool mat[1030][1030];
int winner[1030];
int main(){
    int n;
    cin>>n;
    int len=(1<<n);
    for(int i=2;i<=len;i++)
    	for(int j=1;j<i;j++)
    		cin>>mat[i][j];  
    for(int i=1;i<=len;i++)
    	winner[i]=i;
    while(len>1) {	
   		for(int i=2;i<=len;i+=2){
    		if(mat[winner[i]][winner[i-1]])
    			winner[i>>1]=winner[i];
    		else 
    			winner[i>>1]=winner[i-1];
    	}
    	len/=2;   	
    }
    printf("%d\n",winner[1]);
}
