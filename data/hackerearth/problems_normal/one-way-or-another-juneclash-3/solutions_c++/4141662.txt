#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long t,n,x1,y1,x2,y2,v1,v2,temp,f;
    cin>>t;
    while(t--)
    {
    	f=1;
    	cin>>n;
    	cin>>x1>>y1>>v1;
    	n--;
    	while(n--)
    	{
    		cin>>x2>>y2>>v2;
    		if(!f)
    		{
    			continue;
    		}
    		temp=abs(x2-x1)+abs(y2-y1);
    		if(temp%2==0)
    		{
    			if((v1%2==0 && v2%2==0) || (v1%2!=0 && v2%2!=0))
    			{
    				f=1;
    			}
    			else
    			{
    				f=0;
    			}
    		}
    		else
    		{
    			if((v1%2==0 && v2%2!=0) || (v1%2!=0 && v2%2==0))
    			{
    				f=1;
    			}
    			else
    			{
    				f=0;
    			}
    		}
    	}
    	if(!f)
    	{
    		cout<<"No"<<endl;
    	}
    	else
    	{
    		cout<<"Yes"<<endl;
    	}
    }
    return 0;
}
