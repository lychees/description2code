#include <iostream>
#include <cstring>
#include <string>
using namespace std;
int main(int argc, char** argv) {
	string str[100];
	int t,i,j,sl;
	bool palindrome;
	long long val;
	cin>>t;
	for(j=0;j<t;j++)
	{
		cin>>str[j];
	}
	for(j=0;j<t;j++)
	{	
	sl=str[j].length();
	palindrome=1;
	for(i=0;i<sl;i++)
	{
		if(str[j][i]!=str[j][sl-i-1])
		{
			palindrome=0;
			break;
		}
	}
	if(palindrome)
	{
		cout<<"Palindrome"<<endl;
	}
	else
	{
	val=1;
	for(i=0;i<sl;i++)
	{
		val=(str[j][i]-96)*val;
	}
	cout<<val<<endl;
	}
	}
	return 0;
}
