#include<stdio.h>
int lengh(long int num)
{
	if(num<10)
	return 10;
	else if (num<100)
	return 100;
		else if (num<1000)
	return 1000;
		else if (num<10000)
	return 10000;
		else if (num<100000)
	return 100000;
		else if (num<1000000)
	return 1000000;
}
int prime[1000001]={0};
int p2[1000001]={0};
long int sum[1000001]={0};

int main()
{

	int t;
	for(long int i=2;i<1000001;i++)
	{
		if(prime[i]==0)
		{ 
			for(long int j=2*i;j<1000001;j=j+i)
			{
				prime[j]=1;
			}
		}
	}
	
	long int suffix,prefix,num;
	for(long int i=2;i<1000001;i++)
	{
		if(prime[i]==0)
		{
			long int length=lengh(i);
			suffix=i;
			prefix=1;
			long int temp=suffix;
			
			while(temp<1000001)
			{
				p2[temp]=1;
				temp=prefix*length+suffix;
				prefix++;
			}
		}
	}
	
  	for(int i=1;i<1000001;i++)
	{
		sum[i]+=sum[i-1]+p2[i];
	}
	 
	scanf("%d",&t);
	long int a,b;
	while(t)
	{   
		scanf("%d%d",&a,&b);
		int k=a-1;
	
	printf("%d\n",sum[b]-sum[k]);
	t--;
	}

}
