#include<bits/stdc++.h>
using namespace std;
//someday I'll look back and smile
char s[1000000+10];
long long  start[1000000+10];
long long ending[1000000+10];
int main()
{ scanf("%s",s);
    int len=strlen(s);
 long long ans=0;
    for(int i=0;i<len;i++)
    {start[i+1]=start[i]; if(s[i]=='1') start[i+1]++;}
    for(int i=len-1;i>=0;i--)
    {ending[i+1]=ending[i+2]; if(s[i]=='1') ending[i+1]++;}
    for(int i=0;i<len;i++)
        if(s[i]=='0')
        ans=ans+start[i+1]*ending[i+1];
    printf("%lld\n",ans);


}
