#include<bits/stdc++.h>

using namespace std ;

int main()
{
	int t , k ;
	string str ;
	cin>>t ;
	int i , j , len ;
	long int sum = 0 ;
	int count = 0 ;
	while(t--)
	{
		sum = 0 ;
		count = 0 ;
		str = "" ;
		cin>>k ;
		cin>>str ;
		len = str.length() ;
		j = 0 ;
		for(i = 0 ; i < len ; i++)
		{
			sum += (str[i]-'a' +1) ;
			if(sum == k)
			{
				sum -= (str[j]-'a' + 1) ;
				j++ ;
				count++ ;
			}
			else
			{
				while(sum >= k)
				{
					sum -= (str[j]-'a' + 1 ) ;
					if(sum == k)
						count++ ;
					j++ ;
				}
			}
		}
		cout<<count<<endl ;
	}
	return 0 ;
}
