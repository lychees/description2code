import copy

def main():
	f=input()
	keyf=[int(s) for s in raw_input().split()]
	nk=input()
	keys=[int(s) for s in raw_input().split()]
	assign=[0]*nk
	sm=0
	for x in keys:
		sm=sm+x
	if f>sm:
		print -1
		return 0
	keyf=sorted(keyf,reverse=True)
	total=0
	DEATH=max(keys)+1
	for k in keyf:
		v=min(assign)
		i=assign.index(v)
		assign[i]=assign[i]+1
		total=total+(k*assign[i])
		if assign[i] == keys[i]:
			assign[i]=DEATH
	print total

main()
