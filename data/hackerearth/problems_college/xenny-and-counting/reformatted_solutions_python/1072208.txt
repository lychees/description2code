import math


def combinations(n, r):
    z = math.factorial(n)
    x = math.factorial(n-r)
    y = math.factorial(r)
    r = z/x
    r = r/y
    return r


x = input()
o = []
for i in range(0, x):
    a, b = raw_input().split()
    a = int(a)
    b = int(b)
    o.append(combinations(a+b-1, a-1))

for i in range(0, x):
    print int(o[i])
