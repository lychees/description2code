import sys
from re import findall

answer = []


def pre_Compute():
    for i in range(100001):
        lists = "".join(str(j) for j in bin(i)[2:])
        answer.append(len(findall('(?=101)', lists)))


pre_Compute()
for __ in range(input()):
    n, t = map(int, sys.stdin.readline().split())
    print len([i for i in range(1, n+1) if answer[i] >= t])
