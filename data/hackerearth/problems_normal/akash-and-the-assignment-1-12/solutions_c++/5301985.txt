#include <iostream>
using namespace std;
char a;
int count[500000][26];
int main()
{
    int n,d,k,l,r,x,i;
    cin>>n>>d;
    for(int i=1;i<=n;i++)
    {cin>>a;
      count[i][a-'a']=1;
      for(int j=0;j<26;j++)
      count[i][j]+=count[i-1][j];
    }
    
    while(d--)
    {cin>>l>>r>>k;
    x=0;
 
      for( i=0;i<26;i++)
      {x+=(count[r][i]-count[l-1][i]);
        if(x>=k)
        break;
      }
      if(x>=k)
      cout<<char('a'+i)<<endl;
      else 
      cout<<"Out of range"<<endl;
    
   
   
  
    }
    
    return 0;
}
