#include <iostream>

using namespace std;

int main()
{
    int x1, y1, x2, y2, t;
    cin >> t;
    for(int i = 0;i < t;++i)
    {
        cin >> x1 >> y1 >> x2 >> y2;
        if(x1 > y1) swap(x1, y1);
        if(x2 > y2) swap(x2, y2);
        if(x1 == y2 or x2 == y1) cout << "Point" << endl;
        else if(y2 < x1 or y1 < x2) cout << "Line" << endl;
        else if(x1 == y1 or x2 == y2) cout << "Point" << endl;
        else cout << "Nothing" << endl;
    }
    return 0;
}
