#include <iostream>
using namespace std;

int main()
{
  	int t;
  	scanf("%d",&t);
  	while(t--)
  	{
  		int n,b,g;
  		scanf("%d",&n);
  		scanf("%d%d",&b,&g);
  		int flag=0;
  		if(b==g)
  		flag=2;
  		else if(b-g>=2 || g-b>=2)
  		flag=1;
  		if(flag==1)
  		printf("Little Jhool wins!\n");
  		else printf("The teacher wins!\n");
  	}
    return 0;
}
