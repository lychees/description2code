#include<bits/stdc++.h>
using namespace std;
int main()
{
	long int t, pos;
	unsigned long long int n, ans, check, times;
	scanf("%ld", &t);
	while(t--)
	{
		scanf("%llu", &n);
		
		if(n == 1)
			cout << "0 1\n";
		else
		{
		check = n - 1;
		
		pos = 0;
		while(check)
		{
			check >>= 1;
			pos++;
		}
		
		ans = 1;
		for(int i = 1; i < pos; i++)
		{
			ans = (ans << 1) + 1;
		}
		
		times = ((n - 1) - (ans >> 1));
		times = times * 2;
		
		cout << ans << ' ' << times << endl;
		}
	}
	return 0;
}
