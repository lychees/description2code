#include <iostream>
#define ll long long
using namespace std;
ll bin(ll n)
{
	ll prev=0, current=0, f=0;
	while(n!=0)
	{
		current=n&1;
		if(current==prev && prev==1)
		{
			f=1;break;
		}
		prev=current;
		n=n>>1;
	}
	if(f==1) return 0;
	return 1;
}
int main()
{
	ll t, n, f;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	f=0;
    	while(f!=1)
    	{
    		if(bin(n)==1)
    			f=1;
    		n++;
    	}
    	cout<<n-1<<endl;
    }
    return 0;
}
