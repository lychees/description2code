#include <iostream>
using namespace std;

int main()
{
    int a[100000];
    int n;
    cin>>n;
    for(int i=0;i<n;i++)
    	cin>>a[i];
    int k;
    cin>>k;
    int m=1;
    while(k--)
    {
    	int d;
    	cin>>d;
    	if(m*d>1000000)
    	{
    		m=1000001;
    		break;
    	}
    	m*=d;
    }
    for(int i=0;i<n;i++)
    	cout<<(a[i]/m)<<" ";
    return 0;
}
