#include<bits/stdc++.h>
using namespace std;
#define gc getchar_unlocked

void scanint(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc())   
        {x = (x<<1) + (x<<3) + c - 48;}
}

int main()
{
	int t,l,r;
	scanint(t);
	for(int i=1;i<=t;i++)
	{
		scanint(l);
		scanint(r);
		int c=0;
		int a;
		int s=0;
		if(l==1)
		l++;
		long long count=0;
		for(int j=l;j<=r;j++)
		{
			int f=0;
			for(int k=2;k<=sqrt(j);k++)
			{
				if(j%k==0)
				{
				f=1;
				break;
				}
			}
			if(f==0)
			{
				count++;
			}
			
		}
		cout<<count<<endl;
	}
	return 0;
	
}
