#include <iostream>
#include <queue>
using namespace std;

int main(){
    int n;
    string names[ 5 ] ={ "Sheldon", "Leonard", "Penny", "Rajesh", "Howard"};
    cin>>n;
    n--;
    int que = 5;
    int cuenta=1;
    for(;n>=que;)
    {
       n-=que;
       que*=2;
       cuenta*=2; 
        }
    cout<<names[n/cuenta]<<endl;
    return 0;
    }
