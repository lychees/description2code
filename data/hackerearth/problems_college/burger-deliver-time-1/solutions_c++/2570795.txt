#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int n;
   pair<int ,int> p[100008];
   cin>>n;
   for(int i=0;i<n;i++)
   {
   	int a,b;
   	 cin>>a>>b;
   	 p[i].first = a+b;
   	 p[i].second = i+1;
   }
   	sort(p,p+n);
   	for(int i=0;i<n;i++)
   	cout<<p[i].second<<" ";
   	cout<<endl;
   
    return 0;
}
