#include<iostream>
#include<stdio.h>
//#include<bits/stdc++.h>
#include<string.h>
#include<math.h>

using namespace std;

int main()
{
    long long t,n,w,a[1000],i,count;
    char str[10000];
    cin>>t;
    for(w=0;w<t;w++)
    {
            memset(a,0,1000);

        count=0;
        fflush(stdin);                                      //<stdio.h>
        scanf("%s",str);
        for(i=0;i<strlen(str);i++)
        {
            if(str[i]=='D')
                a[i]=-1;
            else if(str[i]=='B')
            {
                a[i]=-1;
                if(i!=0)
                  a[i-1]=-1;
                if(i!=1)
                a[i-2]=-1;

            }
            else if(str[i]=='S')
            {
                a[i]=-1;
                if(i!=0)
                  a[i-1]=-1;
                a[i+1]=-1;
            }
        }
        for(i=0;i<strlen(str);i++)
        {
            if(a[i]==0)
                count++;
        }
        cout<<count<<"\n";
    }
    return 0;
}
