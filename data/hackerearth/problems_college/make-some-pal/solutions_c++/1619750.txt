bool pal(string &s){
    int n = s.length();
    for(int i=0;i*2 <= n;++i){
        if(s[i] != s[n-i-1]) return 0;
    }
    return 1;
}

int main(){
    int t;cin>>t;while(t--){
        string s;cin>>s;
        int n = s.length();
        if(pal(s)){cout<<n;el;continue;}
        string t;
        for(int i=n-2;i>=0;--i) t += s[i];
        s += t;
        int ans = 2*(n-1)+1;
        int k = n;
        while(s.length()>n){
            s.erase(s.begin()+k,s.begin()+k+1);
            if(pal(s)) ans = s.length(); //, cout<<s,el;
        }
        cout<<ans;el;
    }
    return 0;
}
