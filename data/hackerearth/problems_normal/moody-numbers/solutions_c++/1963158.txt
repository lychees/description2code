#include<bits/stdc++.h>
using namespace std;
long long calculate(long long a)
{
 
    long long  n = a*a;
    long long sum = 0;
    while(n)
    {
        sum = sum + n%10;
        n = n /10;
    }
    return sum;
 
}
int arr[300]= {0};
int main()
{
int t;
int f = 1;
for(int  i = 1 ; i < 200;i++)
{
    long long a   = i,k = 10000;
    f = 1;
    while(k--)
    {
        long long sum =   calculate(a);
        //cout<<sum<<endl;
        if(sum == 1 || sum == 4)
        {
         //   cout<<"yes "<<i<<endl;
            f = 0;
            break;
        }
        a = sum;
    }
    if(f==0)
        arr[i] = 1;
    else
        arr[i] = 0;
 
 
}
arr[4]=1;
scanf("%d",&t);
while(t--)
{
long long n;
scanf("%lld",&n);
if(n<200)
{
    if(arr[n] == 1)
        printf("YES\n");
    else
        printf("NO\n");
}
else
{
    long long sum = calculate(n);
    if(arr[sum] == 1)
    {
        printf("YES\n");
    }
    else
    {
        printf("NO\n");
    }
 
}
 
 
}
}
