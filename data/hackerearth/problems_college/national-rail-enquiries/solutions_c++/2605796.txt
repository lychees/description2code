#include <bits/stdc++.h>

using namespace std;
typedef long long ll;

ll gcd(ll a, ll b)
{
	while(a>0 && b>0)
	{
		if (a>b)
			a%=b;
		else
			b%=a;
	}
	return (a+b);
}

ll fact(ll x)
{
	if (x==0)
	return 1;
	else if (x==1)
	return 1;
	else return ((x*fact(x-1)));
}

int main()
{
	ll t, n, p;
	scanf("%lld", &t);
	for(ll i=1; i<=t; i++)
	{
		scanf("%lld",&n);
		if (n>=110)
		p=0;
		else
		p=fact(n-1);
		if (gcd(p%110, n) == 1 && p>0)
		printf("Case %lld:Yes\n", i);
		else
		printf("Case %lld:No\n", i);
	}
	return 0;
}
