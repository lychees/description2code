#include<bits/stdc++.h>

#define slong(a) scanf("%lld",&a)
#define all(c) c.begin(),c.end()
#define r_all(c) (c).rbegin(),(c).rend()
#define iter(c,i) for(typeof((c).begin()) it=(c).begin();it!=(c).end();it++)
#define pb(c) push_back(c)
typedef long long int ll;
using namespace std;
vector<long long int> vec;
int main()
{
    ll t,n,m,tmp,i,anss;
    cin>>t;
    while(t--)
    {
        cin>>n>>m;
        vec.clear();
        vec.resize(n);
        for(i=0;i<n;i++)
        {
            cin>>vec[i];
        }
        sort(vec.begin(),vec.end());
        for(i=n-1;i>=0;i--)
        {
            m=m-vec[i];
            if(m<=0)    break;
        }
        anss=n-i;
        if(anss>n)  anss=-1;
        cout<<anss<<endl;
    }
    return 0;
}
