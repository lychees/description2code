#include <bits/stdc++.h>
using namespace std ;

#define sc(x) scanf("%d",&x)
#define pr(x) printf("%d ",x) 
#define prn(x) printf("%d\n",x)
#define LL long long int 

int cnt(LL X){

	int c = 0 ;
	while(X){
		c ++ ;
		X -= (X&-X) ;
	}
	return (c%2) ;
}
 
int main(){

	int t ;
	sc(t) ;
	bool ok = 1 ;
	while(t--){
		LL N ;
		scanf("%lld",&N) ;
		if(ok){		
			ok = cnt(N) ;
			puts( ok ? "Sherlock" : "Watson" ) ;
					
		}else{
			ok = (cnt(N) == 0) ;
			puts( ok ? "Sherlock" : "Watson" ) ;
	 	}
	}
	return 0 ;
}
