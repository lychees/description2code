#include<bits/stdc++.h>
using namespace std;
# define ll long long
int main()
{
    cin.sync_with_stdio(false);
    ll i,t,temp,ar[30],t1,t2;
    string str,y="YES",n="NO";
    char ch;
    cin>>str;
    memset(ar,0,sizeof(ar));
    for(i=0;i<str.size();i++)
    {
        ch=str[i];
        temp=ch-96;
        ar[temp]++;
    }
    t2=0;
    t1=str.size();
    for(i=1;i<=26;i++)
    {
        if(ar[i]%2==1)
            t2++;
    }
    if(t1%2==0)
    {
        if(t2==0)
            cout<<y<<endl;
        else
            cout<<n<<endl;
    }
    else
    {
        if(t2==1)
            cout<<y<<endl;
        else
            cout<<n<<endl;
    }
    return 0;
}
