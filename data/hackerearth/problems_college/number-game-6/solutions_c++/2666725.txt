#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
ll sum[25150],k=1;
int main()
{
	sum[0]=0;
    for(ll i=1;i<=25155;i++)
    {
    	sum[k++] = i*i*i + i*i*2 + i + sum[i-1];
    	if(sum[i]>100000000000000000)
    		break;
    }
    int t; scanf("%d",&t);
    while(t--)
    {
    	ll n; scanf("%lld",&n);
    	if(n<=3)
    	{
    		printf("0\nDraw\n"); continue;
    	
    	}
    	int loc = lower_bound(sum,sum+25151,n)-sum;
    	//cout << "loc : " << loc << endl;
    	if(sum[loc]>n)
    		loc--;
    	printf("%d\n",loc);
    	if(loc%2)
    		printf("Dhruv\n");
    	else
    		printf("Ketan\n");
    }
    return 0;
}
