#include<bits/stdc++.h>
using namespace std;
long long int dp[1005][1005];
main()
{
	int i,j;
	
	for(i=0;i<=1004;i++)
	{
		for(j=0;j<=1004;j++)
		{
			if(i==0 || j==0)
			 dp[i][j]=0;
			 else
			  dp[i][j]=1;
		}
	}
	long long int mod=1000000007;
	for(i=1;i<=1004;i++)
	{
		for(j=2;j<=1004;j++)
		{
			dp[i][j]=(dp[i-1][j]+dp[i][j-1]+dp[i-1][j-1])%mod;
		}
	}
	int t;
	cin>>t;
	int sx,sy,dx,dy;
	while(t--)
	{
		cin>>sx>>sy>>dx>>dy;
		cout<<dp[dx-sx+1][dy-sy+1]<<endl;
	}
	
}
