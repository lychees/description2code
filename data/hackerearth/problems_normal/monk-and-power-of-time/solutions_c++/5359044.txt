#include <iostream>
#include<queue>
using namespace std;

int main()
{
	int y=0;
	queue <int> q;
	int n,z;
    cin>>n;
    z=n;
    int array[n];
    for(int i=0;i<n;i++){
    	int a;
    	cin>>a;
    	q.push(a);
    }
    for(int j=0;j<n;j++){
    	int b;
    	cin>>b;
    	array[j] = b;
    	
    }
    for(int k=0;k<n;k++){
    	for(int l=0;l<n;l++){
    if(q.front()!=array[k]){
    	q.push(q.front());
    	q.pop();
    	y++;
    	
    }else{
    	q.pop();
    	break;
    }	
    	}
    }
    cout<<y+z;
    return 0;
}
