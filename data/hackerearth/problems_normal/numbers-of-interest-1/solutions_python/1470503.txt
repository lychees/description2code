test=int(raw_input())
def nthNum(x,y,n):
    ''' Returns nth number of the sequence'''
    if n<=x:
        return y
    else:
        temp_list=[int(a) for a in x*str(y).split()]
        for i in range(n-x):
            temp_list.append(sum(temp_list))
            temp_list.remove(temp_list[0])
        return temp_list[-1]
for i in xrange(test):
    x,y,n=[int(j) for j in raw_input().split()]
    print nthNum(x,y,n)
