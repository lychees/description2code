#include <iostream>
using namespace std;

void sort(int k[],int n)
{
	for(int i=0;i<n;i++)
	{
		for(int j=i+1;j<n;j++)
		{
			if(k[i]>k[j])
			{
				int temp=k[i];
				k[i]=k[j];
				k[j]=temp;
			}
		}
	}
}

int main()
{
    int t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	int k[n];
    	for(int i=0;i<n;i++)
    	cin>>k[i];
    	sort(k,n);
    	int tp[n/2];
    	for(int a=0;a<n/2;a++)
    	{
    		tp[a]=k[a]+k[n-1-a];
    	}
    	sort(tp,n/2);
    	cout<<(tp[(n/2)-1]-tp[0])<<endl;
    }
    return 0;
}
