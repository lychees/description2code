#include <iostream>
using namespace std;

int main()
{
	long long sum,start,t,n,i,ans,price[50002];
	cin>>t;
	while(t--)
	{
		cin>>n;
		ans=0;
		for(i=0;i<n;i++)
		cin>>price[i];
		sum=price[n-1];
		for(i=n-2;i>=0;i--)
		{
			if(price[i]<sum)
			ans+=sum-price[i];
			else
			sum=price[i];
		}
		cout<<ans<<"\n";
	}
    return 0;
}
