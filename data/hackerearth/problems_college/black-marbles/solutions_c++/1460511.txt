#include <bits/stdc++.h>
using namespace std;

int main()
{
    int m,n,ans,f,ff,x,y;cin>>m>>n;
    int dx[4]={-2,-2,2,2};
    int dy[4]={-2,2,-2,2};
    while(m!=0 and n!=0){
    	vector<string>v(m);
    	for(int i=0;i<m;i++)cin>>v[i];
    	ans=0;
    	f=1;
    	while(f){
    		f=0;
    		for(int i=0;i<m;i++){
    			for(int j=0;j<n;j++){
    				if(v[i][j]=='F'){
	    				ff=1;
	    				for(int k=0;k<4;k++){
	    					x=i+dx[k];
	    					y=j+dy[k];
	    					if(x>=0&&x<m&&y>=0&&y<n){
	    						if(v[x][y]=='G'){
	    							if(v[i+dx[k]/2][j+dy[k]/2]!='P'){
	    								ff=0;
	    							}
	    						}
	    					}
	    				}
	    				if(ff)v[i][j]='G',ans++,f=1;
    				}
    			}
    		}
    	}
    	cout<<ans<<endl;
    	cin>>m>>n;
    }
    return 0;
}
