#include <iostream>
using namespace std;

#define MAX 1000000

int main()
{
    int a[MAX]={0};
    int t,n;
    cin>>t;
    while(t--){
    	cin>>n;
    	//a={0};
    	int max  = 0;
    	while(n--){
    		int i;
    		cin>>i;
    		a[i]++;
    		if(a[i] > max){
    			max = a[i];
    		}
    	}
    	for(int i=1000000-1; i>=0; i--){
    		if(a[i] == max){
    			cout<<i<<" ";
    		}
    		a[i] = 0;
    	}
    	cout<<endl;
    }
    return 0;
}
