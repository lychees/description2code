#include<iostream>
#include<string>
using namespace std;
int b[100]={0};
void preprocess(string pat)
{
	int j=-1,i=0;
	b[0]=-1;
	while(i<pat.size())
	{
		while(j>=0 && pat[i]!=pat[j])
			j=b[j];
			i++;
			j++;
			b[i]=j;
		
	}
	
}
int kmp(string text,string pat)
{
	int i=0,j=0,count=0;
	preprocess(pat);
	while(i<text.size())
	{
		while(j>=0 && text[i]!=pat[j])
			j=b[j];
			i++;
			j++;
			if(j==pat.size())
			{
				count++;
				j=b[j];
				
			}
		
	}
	return count;
	
}
int main()
{
	string text;
	string pat="..";
	int l,n,index;
	char c;
	cin>>l>>n;
	cin>>text;
	for(int i=0;i<n;i++)
		{
			cin>>index>>c;
			text[index-1]=c;
			cout<<kmp(text,pat)<<endl;
			
		}
	return 0;
}
