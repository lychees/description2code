#include <bits/stdc++.h>
#define fi first
#define se second
#define mp make_pair
#define pb push_back
#define pf push_front
#define endl '\n'
#define INF 1000000000000000LL
#define PI 3.14159265358979323846
typedef long long LL;
using namespace std;

LL a[111111];

int main(){
        ios_base::sync_with_stdio(false);

        int t;
        cin >> t;

        while(t--){
                int n, k;
                cin >> n >> k;

                for(int i = 0; i < n; i++) cin >> a[i];

                LL i = 0, mx = 0, ans = 0;
                while(i < n){
                        if(a[i] > k){
                                i++;
                                ans = 0;
                        }else while(a[i] <= k && i < n){
                                ans += a[i];
                                i++;
                        }
                        mx = max(mx, ans);
                }
                cout << mx << endl;
        }

        return 0;
}
