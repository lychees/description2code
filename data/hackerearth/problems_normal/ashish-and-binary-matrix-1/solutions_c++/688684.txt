#include <iostream>
#include<cstdio>
#include<set>
using namespace std;

int main()
{
	string s;
    set <string> se;
    int t,n,m,i;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d %d",&n,&m);
    	for(i=0;i<n;i++)
    	{
    		cin>>s;
    		se.insert(s);
    	}
    	if(se.size()==n)
    	printf("Yes\n");
    	else
    	printf("No\n");
    	se.clear();
    }
    return 0;
}
