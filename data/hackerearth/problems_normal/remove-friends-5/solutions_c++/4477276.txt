#include <iostream>
#include <stack>
using namespace std;

stack<int> S;
int arr[100005];
int T, N, K, n;

int main()
{
    cin >> T;
    
    while(T--) {
    	cin >> N >> K;
    	
    	for(int i = 0;i < N;i++) {
    		cin >> arr[i];
    	}
    	
    	for(int i = 0;i < N;i++) {
    		if(S.empty()) {
    			S.push(arr[i]);
    		} else {
    			while(S.size() && S.top() < arr[i] && K) {
    				S.pop();
    				K--;
    			}
    			S.push(arr[i]);
    		}
    	}
    	
    	while(K--) {
    		S.pop();
    	}
    	
    	N = n = S.size();
    	while(S.size()) {
    		arr[--n] = S.top();
    		S.pop();
    	}
    	
    	for(int i = 0;i < N;i++) {
    		if(i) cout << " ";
    		cout << arr[i];
    	}
    	cout << endl;
    }
    
    return 0;
}
