#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;


int main()
{
   // cout << "Hello World!" << endl;
   int t,i,x,y;
   cin>>t;
   pair<int,int>p[t] ;
   for(i=0;i<t;i++)
   {
   	cin>>x>>y;
   	p[i]=make_pair(x+y,i+1);
   }
   sort(p,p+t);
   for(i=0;i<t;i++)
   {
   	cout<<p[i].second<<" ";
   }
    return 0;
}
