#include<bits/stdc++.h>

using namespace std;

int main()
{
    long long t;
    cin >> t;

    while(t--)
    {
        long long n;
        cin >> n;
        long long ara[5]={0},no=0;
        while(n--)
        {
            long long a,b,c;
            cin >> a >> b >> c;
            long long x = a+b;
            x%=2;
            if(c%2==1)
            {
                if(ara[x]==2)no=1;
                ara[x]=1;
            }
            else
            {
                if(ara[x]==1)no=1;
                ara[x]=2;
            }
            if(ara[0]==ara[1])no=1;
        }
        if(no)cout << "No\n";
        else cout << "Yes\n";

    }

}
