#include <bits/stdc++.h>
#define ll long long
#define pb push_back
using namespace std;

vector<ll> prime;

void prime_sieve()
{
	prime.clear();
	
	bool vst[10001];
	
	memset(vst,0,sizeof(vst));
	
	for(int i=2;i<=10000;i++)
	{
		if(!vst[i])
			prime.pb(i);
		for(int j=2*i;j<=10000;j+=i)
			vst[j]=1;
	}

}
int main()
{
   
   prime_sieve();
   
  
   ll t,g,n;
   
   cin>>t;
   
   while(t--)
   {
   		cin>>g>>n;
   
   		for(int i=0;i<n-1;i++)cout<<g*2<<" ";cout<<g*3<<"\n";
   			
   
   }
   
   return 0;
}
