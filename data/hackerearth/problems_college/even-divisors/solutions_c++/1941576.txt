#include <iostream>
#include<cmath>
using namespace std;

int main()
{
   int t; cin >> t;
   while(t--)
   {
   		int n; cin >> n;
   		int i,ans=0;
   		if(n%2==0)
   		{
   			for(i=1;i*i<=n;i++)
   			{
   				if(i*i==n)
   				{
   					ans++;break;
   				}
   				if(n%i==0 && i%2==0)
   					ans++;
   				if((n/i)%2==0 && n%i==0)
   					ans++;
              //  cout << i << " " << ans << endl;
   			}
   		}
   		cout << ans << endl;
   }
    return 0;
}
