#include <iostream>
using namespace std;

int main()
{
	int t,no,nx;string s;
	cin>>t;
	while(t--)
	{   no=nx=0;
		cin>>s;
		int sl=s.length(),i;
		for(i=0;i<sl;)
		{
			if(s[i]=='x')
			{
				no=0;cout<<"x";for(;s[i+1]=='o';)i++;i++;
			}
			else if(s[i]=='*')
			{  
				for(int j=0;j<no;j++)
				cout<<"o";
				no=0;cout<<"*";
				i++;
			}
			else
			no++,i++;
		}if(i==sl)
		for(int j=0;j<no;j++)cout<<"o";
		cout<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
