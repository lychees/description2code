using namespace std;

int t;
string a, b;
long long ans, MOD = 1e9 + 7, f[1001000], cnt1[26], cnt2[26];

long long inv(long long a){
	long long ret = 1, b = MOD-2;
	while(b){
		if(b%2) ret = (ret*a)%MOD;
		a = (a*a)%MOD;
		b >>= 1;
	}
	return ret;
}

int main(){ _
	cin >> t;
	f[0] = 1;
	for(int i = 1; i <= 1000000; i++)
		f[i] = (f[i-1]*i)%MOD;
	
	while(t--){
		cin >> a >> b;
		ans = 1;
		int l1 = a.length();
		int l2 = b.length();
		
		memset(cnt1, 0, sizeof cnt1);
		memset(cnt2, 0, sizeof cnt2);
		
		for(int i = 0; i < l1; i++){
			cnt1[a[i]-'a']++;
		}
		for(int i = 0; i < l2; i++){
			cnt2[b[i]-'a']++;
		}
		
		for(int i = 0; i < 26; i++){
			if(cnt1[i] < cnt2[i]){
				ans = 0;
				break;
			}
			
			ans = (ans * ((f[cnt1[i]] * inv(f[cnt1[i]-cnt2[i]])) % MOD)) %MOD;
		}
		
		printf("%lld\n", ans);
	}
	return 0;
}
