#include <iostream>
using namespace std;

int main()
{
    int num, i, count_child, count = 0;
    cin>>num>>count_child;
    int child[count_child];
    for(i = 0; i < count_child; i++) {
    	cin>>child[i];
    	int min = (num < count_child) ? num : count_child;
    	for(int j = 2; j <= min; j++) {
    		if( (num % j == 0) && (child[i] % j == 0) ) {
    			count++;
    			break;
    		}
    	}
    }
    cout<<count;
    return 0;
}
