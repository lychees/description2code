#include<iostream>
#include<algorithm>
#include<string>
#include<cstring>
#include<cmath>
#include<climits>

using namespace std;
typedef long long int ll;

int main()
{
	int t, n;
	cin >> t ;
	while(t--)
	{
		cin >> n;
			ll arr[n], i= 0,j=0, num = pow(2, n);
			ll curr_xor = 0, curr_and = 1, ans = INT_MIN, temp = ans, x=0, y=0, len=0, temp_len= 0, newnum = 0;
		
		for(i = 0; i < n ; i++)
			cin >> arr[i];
		
		for (i = 1; i <= num ; i++)
		{
		    curr_xor = 0;
		    curr_and = 0;
			temp_len = 0;
			int k = 0;
			string s = "";
			
			//converting into binary			
			for (int ii = (1<<(n-1)); ii > 0; ii = ii / 2)
				(i & ii)? s+="1": s+="0";
			
			for(j = 0 ; j < s.length() ; j++)
			{			
				if( s[j] == '1')
				{
					curr_and = arr[j];
					break;
				}
			
			}			
			for(j = 0 ; j < s.length() ; j++)
			{
				if( s[j] == '1')
				{
					curr_xor = curr_xor ^ arr[j];
					curr_and = curr_and & arr[j];
					temp_len++;
				}
			}
			temp = (curr_xor | curr_and);
			if(ans < temp )
				 len= temp_len;
			else if(ans == temp)
				 len= max(len, temp_len);
			ans = max(ans, temp);
		}
		cout << ans << " " << len << endl;
	}
}
