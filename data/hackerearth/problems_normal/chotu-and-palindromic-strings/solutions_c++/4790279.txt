#include <bits/stdc++.h>
using namespace std;

int check(string temp)
{
    int i,l,k,flag=0;
    l=temp.size();
    for(i=0,k=l-1;i<l/2;i++,k--)
   	{
    	if(temp[i]!=temp[k])
    	{
    	    flag=1;
    	    break;
    	}
   	}
   	return flag;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s1,s2;
    	cin>>s1;
    	cin>>s2;
    	string s3=s1+s2;
    	string s4=s2+s1;
    	int i,j;
    	i=check(s3);
    	j=check(s4);
    	//cout<<i<<" "<<j<<endl;
    	if(i==0||j==0)
    	{
    		cout<<"YES"<<endl;
    	}
    	else
    	{
    		cout<<"NO"<<endl;
    	}
    }
    return 0;
}
