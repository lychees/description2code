#include <iostream>
using namespace std;

int min(int a,int b,int c){return a<b?(a<c?a:c):(b<c?b:c);}
int max(int a,int b,int c){return a>b?(a>c?a:c):(b>c?b:c);}
int n;

int getMin(int s1,int s2,int s3,int arr[],int count)
{
	if(count==n){
		int w=max(s1,s2,s3);
		return w;
	}
	int x=getMin(s1+arr[count],s2,s3,arr,count+1);
	int y=getMin(s1,s2+arr[count],s3,arr,count+1);
	int z=getMin(s1,s2,s3+arr[count],arr,count+1);
	
	return min(x,y,z);
}

int main()
{
    cin>>n;
    int arr[n];
    for(int i=0;i<n;i++)
    	cin>>arr[i];
    	
    int result=getMin(0,0,0,arr,0);
    cout<<result<<endl;
    
    return 0;
}
