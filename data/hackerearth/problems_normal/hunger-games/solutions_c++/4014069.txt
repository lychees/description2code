#include <bits/stdc++.h>
using namespace std;

int main()
{
    // cout << "Hello World!" << endl;
    int n, i;
    cin>>n;
    int a[n];
    for(i = 0; i < n; i++) cin>>a[i];
    sort(a, a+n);
    int mx, p1, p2;
    mx = max(a[n-1]-a[n-2], a[n-1]-a[n-3]);
    p1 = n-2; p2 = n-4;
    while(p2 >= 0) {
    	mx = max(mx, a[p1]-a[p2]);
    	p1--; p2--;
    }
    mx = max(mx, a[1]-a[0]);
    cout<<mx<<endl;
    return 0;
}
