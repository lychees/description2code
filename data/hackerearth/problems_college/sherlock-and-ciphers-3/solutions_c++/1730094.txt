#include <iostream>
#include<stdio.h>
#include<string.h>
#include<math.h>
using namespace std;
int main()
{
    char code[30][30],s[120][200];
    int i,j,k,l=0,len=0,N=0,sq;
    cin>>N;
 	cin.ignore();
 	for(i=0;i<N;i++)	
 	cin.getline(s[i],200);
 		
    for(i=0;i<N;i++)
    {	
    	l=len=sq=0;
    	len=strlen(s[i]);
    	sq=sqrt(len);
    	for(j=0;j<sq;j++)
    		for(k=0;k<sq;k++)
    			{
    				code[k][j]=(*(*(s+i)+l));
    				l++;
    			}
    	for(j=0;j<sq;j++)
    	for(k=0;k<sq;k++)
    	cout<<code[j][k];
    
    cout<<"\n";
    }
   }
