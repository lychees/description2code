#include <iostream>
using namespace std;

string convertToGood(string s){
	string nextString;
	int j = 0; 
	for(int i=0; i<s.size(); i++){
		if(nextString.size() == 0){
			nextString += s[i];
		}
		else if(s[i] == nextString[j]){
			continue;
		}
		else{
			nextString += s[i];
			j += 1;
		}
	}
	return nextString; 
}


int main()
{
    int t;
    cin >> t;
    for(int i=0; i<t; i++){
    	string s;
    	cin >> s;
    	cout << convertToGood(s) << "\n";
    }
    return 0;
}
