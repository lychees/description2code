#include <bits/stdc++.h>

#define ll long long

using namespace std;

ll mod = 1000000007;

ll modexpo(ll x,ll y)
{
    ll val1,ty,my;
    if( y == 0) return 1;
    val1 = modexpo(x, y/2);
    ty=(val1%mod)*(val1%mod);
    if (y%2 == 0) return ty%mod;
    else {
        my=(x%mod)*(ty%mod);
        return my%mod;
    }
}

ll fast_power(ll base, ll exponent, ll modulus)
{
    ll result = 1;
    while (exponent > 0) {
        if (exponent % 2 == 1)
            result = (result * base) % modulus;
        exponent = exponent >> 1;
        base = (base * base) % modulus;
    }
    return result;
}

void navi(int tt)
{
    int n;
    scanf("%d", &n);
    ll vall = INT_MIN;
    vector <ll > arr(n + 1);
    for (int i = 0; i < n; i++) scanf("%lld", &arr[i]);
    int xy = 1 << n;
    for (int i = 1; i < xy; i++) {
        ll prod = 1;
        ll sum = 0;
        int count = 0;
        for (int j = 0; j < n; j++) {
            if (i&1<<j) {
                sum = (sum + arr[j])%mod;
                prod = (prod * arr[j])%mod;
                count++;
            }
        }
        if (count - 2 >= 0 ) {
            ll val1 = fast_power(sum, mod-2, mod);//modexpo(sum, mod - 2);
            val1 = (val1*prod)%mod;
            vall = max(vall, val1);
        }
    }
    cout << "Case #" << tt << ": " << vall << endl;
}

int main()
