#include <iostream>
#include <map>
using namespace std;

int main()
{
	int n;
	cin >> n;
	int a[n];
	for (int i = 0; i < n; ++i) 
		cin >> a[i];
	
	int m = 0;
	int c = 0;
	
	for (int i = 0; i < n; ++i) {
		if (a[m] == a[i]) {
			++c;
		}
		else {
			--c;
		}
		if (c == 0) {
			m = i + 1;
			c = 0;
		}
	}
	
	
	map<int, int> mp;
	int mx = 0;
	c = 0;
	int t = 0;
	if (m >= n) m = 0;
		
	for (int i = 0; i < n; ++i) {
		if (i < n - 1 && a[i] == a[i + 1]) {
			mp[a[i]]++;
			mx = max(mx, mp[a[i]]);
			++t;
		}
		
		if (a[m] == a[i]) ++c;
	}
	
		
	if (c > (n + 1)/ 2) cout << -1 << '\n';
	else {
		cout << max((t + 1) / 2, mx) << '\n';	
	}

	
    return 0;
}
