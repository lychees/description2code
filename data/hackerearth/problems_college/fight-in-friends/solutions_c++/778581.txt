bool *mat[1001];
int dist[1001];

typedef pair<int, int> ii;
typedef vector<ii> vii;
typedef vector<vii> vvii;

int n, a, b;

int main()
{
#ifdef OFFLINE_VS_EXECUTION
	freopen("..\\input.txt", "r", stdin);
	freopen("..\\output.txt", "w", stdout);
#endif

	int i, j, m;

	FOR(i, 0, 1000) {
		mat[i] = MEM(bool, 1000);
		dist[i] = 10000;
		FOR(j, 0, i) {
			mat[i][j] = mat[j][i] = true;
		}
	}

	scanf("%d%d%d%d", &n, &m, &a, &b);

	while (m--) {
		scanf("%d%d", &i, &j);
		mat[i][j] = mat[j][i] = false;
	}

	priority_queue<ii,vector<ii>, greater<ii> > pq;
	dist[a] = 0;
	pq.push(ii(0,a));

	while (!pq.empty()) {
		ii temp = pq.top();
		pq.pop();
		int vertex = temp.second;
		int distance = temp.first;

		if (distance <= dist[vertex]) {
			FOR(i, 1, n) {
				if (mat[vertex][i]) {
					if (distance + 1 < dist[i]) {
						dist[i] = distance + 1;
						pq.push(ii(dist[i], i));
					}
				}
			}
		}
	}

	if (dist[b] == 10000)
		cout << -1 << endl;
	else
		cout << dist[b] << endl;

	return 0;
}
