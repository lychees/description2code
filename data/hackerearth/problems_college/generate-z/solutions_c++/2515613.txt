#include <iostream>
using namespace std;

int main()
{
    int n,i,j;
    
    cin>>n;
    
    int a[20][20];
    
    for(i=0;i<n;i++){
    	for(j=0;j<n;j++){
    		if(i==0 || i==n-1 || i+j==n-1){
    			a[i][j]=1;
    		}else{
    			a[i][j]=0;
    		}
    	}
    }
    
    for(i=0;i<n;i++){
    	for(j=0;j<n;j++){
    		if(a[i][j]==1){
    			cout<<"*";
    		}else if(i+j<n-1){
    			cout<<" ";
    		}
    	}
    	cout<<endl;
    }
    
    return 0;
}
