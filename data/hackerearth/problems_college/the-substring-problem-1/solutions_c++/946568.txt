#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;cin>>t;while(t--){
    	int n;cin>>n;
    	int hash[n][26];
    	memset(hash,0,sizeof(hash));
    	string s;
    	for(int i=0;i<n;i++)
    	{
    		cin>>s;
    		int len=s.length();
    		for(int k=0;k<len;k++)
    		hash[i][s[k]-'a']++;
    	}
    	
    	int k_hash[26];
    	memset(k_hash,0,sizeof(k_hash));
    	for(int i=0;i<n;i++)
    	for(int j=0;j<26;j++)
    	if(hash[i][j]>0)k_hash[j]++;
    	
    	int c;int count=0;
    	for(int i=0;i<26;i++)
    	if(k_hash[i]>count){count=k_hash[i];c=i;}
		
		
		cout<<(char)('a'+c)<<endl;
    	
    	
    }
    
    return 0;
}
