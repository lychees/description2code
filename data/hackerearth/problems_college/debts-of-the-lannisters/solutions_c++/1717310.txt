#include <iostream>
using namespace std;

void delete_from_array(char a[], char c) {
	for (int i = 0; a[i]!='\0'; ++i)
	{
		if (a[i] == c) {
			for (int j = i; a[j] != '\0'; ++j)
			{
				a[j] = a[j+1];
			}
			return;
		}
	}
}

int count(char a[], char c) {
	int x = 0;
	for (int i = 0; a[i]!='\0'; ++i)
	{
		if (a[i] == c)
			x++;
	}
	return x;
}

bool search(char a[], char c) {
	for (int i = 0; a[i]!='\0'; ++i)
	{
		if (a[i] == c)
			return true;
	}
	return false;
}

int main() {
	int t;
	
	char gold[100], silver[100], sequence[100], guess[100];
	int n, r, g = 0, s = 0;
	cin >> sequence;
	cin >> n;
	while (n--) {
		cin >> guess;
		g = 0;
		s = 0;
		gold[0] = '\0';
		silver[0] = '\0';
		for (int i = 0; i < 10; ++i)
		{
			if (guess[i] == sequence[i]) {
				gold[g++] = guess[i];
				gold[g] = '\0';
				if (count(gold, guess[i]) + count(silver, guess[i]) > count(sequence, guess[i])) {
					delete_from_array(silver, guess[i]);
					s--;
				}

			}
			else if (search(sequence, guess[i])) {
				if (count(gold, guess[i]) + count(silver, guess[i]) < count(sequence, guess[i])) {
					silver[s++] = guess[i];
					silver[s] = '\0';
				}
			}
		}
		cout << g << " " << s << endl;
	}
}
