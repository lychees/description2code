#include<bits/stdc++.h>
using namespace std;
int check_prime(int );
int main()
{
	int count=2,prime=3,p;
	cin>>p;
	if(p==1)
	cout<<2;
	else if(p==2)
	cout<<3;
	else
	{
	while(count!=p)
	{
		prime=prime+2;
		if(check_prime(prime))
		{
			count++;
	    }
	}
	cout<<prime;
	}
	return 0;
}
int check_prime(int prime)
{
		int n=(prime+1)/6,rt;
		if(prime==((6*n)+1)||prime==((6*n)-1))
		{
			rt=sqrt(prime);
			int i=3;
			while(i<=rt)
			{
				if(prime%i==0)
				return 0;
				i+=2;
			}
			return 1;
		}
		else return 0;
}
