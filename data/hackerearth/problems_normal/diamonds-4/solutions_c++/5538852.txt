#include <iostream>
using namespace std;

int main()
{
    int t,count=0,n,m,i,j;
    char a[1001][1001];
    cin>>t;
    while(t--)
    {
    	cin>>n>>m;
    	count=0;
    	for(i=0;i<n;i++)
    	for(j=0;j<m;j++)
    	cin>>a[i][j];
    	for(i=0;i<n-1;i++)
    	for(j=0;j<m-1;j++)
    	{
    	    if(a[i][j]=='/')
    	    if(a[i][j+1]==92)
    	    if(a[i+1][j]==92 )
    	    if(a[i+1][j+1]=='/')
    		count++;
    	}
    	cout<<count<<endl;
    }
    return 0;
}
