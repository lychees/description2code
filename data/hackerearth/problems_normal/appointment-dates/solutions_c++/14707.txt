#include<cstdio>
int abs(int x)
    {
        return x<0?-x:x;
    }
int nly(int y)
    {
        int t=0;
        for(int i=2000;i<=y;i+=4)
            {
                if( i%400==0) ++t;
                
                    else if(i%4==0 && i%100!=0) ++t;
            }
            return t;
    }
    
int ly(int y)
    {
        if(y%400==0) return 1;
        else if(y%100!=0 && y%4==0) return 1;
        else return 0;
    }
    
main(void)
{
    int t,mn,y,dg,sun;
    scanf("%d",&t);
    while(t--)
            {
                    scanf("%d%d",&mn,&y);
                    int m[12]={31,28+ly(y),31,30,31,30,31,31,30,31,30,31};
                
                    int d[7]={0,1,2,3,4,5,6};
                    dg=(y-2000)+nly(y);
                    if(ly(y)==1) --dg;
                    int j1;
                    j1=d[(5+dg%7)%7];
                    sun=7-j1;
                    
                    int i=0;
                    for(;sun<=y;sun+=7)
                        {
                            if(sun>m[i]){sun-=m[i++];}
                            if(i==mn-1) printf("%d ",sun);
                        }
                    printf("\n");
                
            }
}
