#include <iostream>
#include <stdio.h>
using namespace std;

	int rev(int no)
	{
		int x=0;
		while(no)
		{
			x=x*10+no%10;
			no/=10;
		}
	return x;
	}
	

int main() {
	
	int tc;
	scanf("%d",&tc);
	while(tc)
	{
		int x,y;
		scanf("%d %d",&x,&y);
		printf("%d\n",rev(rev(x)+rev(y)));
		tc--;
	}

	return 0;
}
