{
	long long int n,a,b;
	float val1,val2;

	{
		cin >> a >> b;
		val1 = (a+b)/2.0;
		val2 = (a-b)/2.0;
		if(val2>=0)
		{
			float check1,check2;
			check1 = val1-(int)(val1);
			check2 = val2-(int)(val2);
			if((check1==0) && (check2==0))
			{
				cout << (int)(val1) << " " << (int)(val2) << endl;
			}
			else
			{
				cout << "impossible" << endl;
			}
		}
		else
		{
			cout << "impossible" << endl;
		}
	}
	return 0;
}
