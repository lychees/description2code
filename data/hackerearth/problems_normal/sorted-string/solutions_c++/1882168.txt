#include<iostream>
#include<stdio.h>
#include<vector>
#include<string.h>
#include<algorithm>
using namespace std;
int main()
{
                       int i,j,count,t;
                         string str;
                           char ch;
                             scanf("%d",&t);
                               while(t--)
                               {

                                        cin>>str;
                                           vector<pair<int,int> >v;
                                               for(i=0;i<26;i++)
                                              {
                                                //cout<<"0";
                                                      v.push_back(make_pair(0,i));
                                                   //cout<<v[i].first<<" "<<v[i].second<<endl;
                                               }
                                                           for(i=0;i<str.length();i++)
                                                          {
                                                                v[str[i]-'a'].first++;
                                                             // cout<<v[str[i]-'a'].first<<endl;
                                                          }
                                                                    sort(v.begin(),v.end());


                                                                      for(i=0;i<26;i++)
                                                                     {
                                                                         count=v[i].first;
                                                                             ch=v[i].second+'a';
                                                                                for(j=0;j<count;j++)
                                                                                  cout<<ch;

                                                                     }
                                                                                      cout<<endl;
                               }
                               return 0;

}
