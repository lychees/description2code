#include <iostream>
#include <string>
using namespace std;

int main()
{
    int t;
    string s1,s2;
    cin>>t;
    while(t--){
    	cin>>s1>>s2;
    	int f1,f2,i;
    	int l1,l2;
    	l1=s1.length();
    	l2=s2.length();
    	f1=f2=0;
    	i=0;
    	while(i<l1){
    	//	cout<<(1<<(char(s1[i])-'a'))<<endl;
    		f1=f1|(1<<(char(s1[i])-'a'));
    	//	cout<<"f1"<<f1<<endl;
    		i++;
    	}
    	//
    	i=0;
        while(i<l2){
        	
    		f2=f2|(1<<(char(s2[i])-'a'));
    		i++;
    	}
    //	cout<<"f2"<<f2<<endl;
    	if(f1&f2){
    		cout<<"Yes"<<endl;
    	}
    	else{
    		cout<<"No"<<endl;
    	}
    }
    return 0;
}
