#include <iostream>
#include<cstring>
#include<vector>
using namespace std;

int main()
{
	int n,m;
	cin>>n>>m;
	
	string player;
	long long int point=0;
	long int max_point=0;
	
	int len=0;
	typedef struct p{
		string name;
		long long int point=0;
	}p;
	
	p players[100];
	
	string winner;
	long long int max_points=-1000000;
	
	while(m--)
	{
		
	   cin>>player>>point;
	 	
	 	bool flag=false;
	 	int index;
	 	
	 	for(int i=0;i<len;i++)
	 	{
	 		if(players[i].name==player)
	 		{
	 			players[i].point+=point;
	 			index=i;
	 			flag=true;
	 			break;
	 		} 
	 	}
	 	
	 	if(!flag)
	 	{
	 		int l=len;
	 		players[l].name=player;
	 		players[l].point=point;
	 		index=l;
	 		len++;
	 	}
	 	
	 	
	 	if(players[index].point>max_points)
	 	{
	 		max_points=players[index].point;
	 		winner=players[index].name;
	 	}
	 	
	 
	}
	
    cout << winner << endl;
    return 0;
}
