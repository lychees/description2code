#!/bin/env python
import sys
stdin = sys.stdin


def check_inverse(n, numbers):
    # new_numbers = list(xrange(1,n+1))
    # a  =''.join(str(x) for x in new_numbers)
    # b  =''.join(str(x) for x in numbers)
    a = {}
    j = 1
    new_numbers = []
    for i in numbers:
        a[i] = j
        j += 1
    for i in a:
        new_numbers.append(a[i])
    # print new_numbers
    # print numbers
    if new_numbers == numbers:
        print "inverse"
    else:
        print "not inverse"


t = int(stdin.readline().strip())
for i in range(t):
    n = int(stdin.readline().strip())
    numbers = map(int, raw_input("").strip().split())
    check_inverse(n, numbers)
