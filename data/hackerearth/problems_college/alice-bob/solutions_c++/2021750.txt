#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int a[2000010];
ll int myabs(ll int x,ll int y)
{
	if(x<y)
	{
		return y-x;
	}
	return x-y;
}
int main()
{
	ll int n;
	scanf("%lld",&n);
	ll int i,j,k;
	map<ll int,int> m;
	map<ll int,int> used;
	for(i=0;i<n;i++)
	{
		scanf("%lld",(a+i));
		m[a[i]]=1;
	}
	ll int t=0;
	int flag=0;
	for(i=0;i<n;i++)
	{
		if(used[a[i]]==1)
		continue;
		for(j=(i+1);j<n;j++)
		{
			if(used[a[j]]==1)
			continue;
			k=myabs(a[i],a[j]);
			if(m[k]==0)
			{
				used[a[i]]=1;
				used[a[j]]=1;
				m[k]=1;
				a[n]=k;
				n++;
				t++;
			}
			//~ cout<<i<<" "<<j<<endl;
		}
	}
	if((t%2)!=0)
	{
		printf("Alice\n");
	}
	else
	{
		printf("Bob\n");
	}
	return 0;
}
