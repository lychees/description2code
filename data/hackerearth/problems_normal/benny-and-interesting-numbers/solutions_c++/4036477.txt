#include<iostream>

using namespace std;

bool prime[400000] = {false};
long int primelist[100000],size = 0;

int main()
{
	for(long int i = 2; i < 632; i++)
	{
		if(!prime[i])
		{
			for(long int j = 2; j*i <= 400000; j++)
				prime[j*i] = true;
		}
	}
	for(long int i = 2; i < 400000; i++)
	{
		if(!prime[i])
			primelist[size++] = i;
	}
	int t;
	cin>>t;
	while(t--)
	{
		long long int n,i;
		cin>>n;
		if(n <= 3)
			cout<<"Yes"<<endl;
		else	
		{
			for(i = 0; primelist[i]*primelist[i] <= n; i++)
				if(n % (primelist[i] * primelist[i]) == 0)
				{
					cout<<"No"<<endl;
					break;
				}
			if(primelist[i] * primelist[i] > n)
				cout<< "Yes"<<endl;
		}
	}
	return 0;
}
