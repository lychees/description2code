    #include <iostream>
    #include <cstdio>
    #include <vector>
    using namespace std;
     
     unsigned long long int power( unsigned long long int a)
     {
     	 unsigned long long int MOD = 1000000007, prod=1, n=1000000005;
     	 
     	 while(n > 0)
     	 {
     	 	if(n & 1)
     	 	 prod = (prod * a)%MOD;
     	 	a = (a * a) %MOD;
     	 	n >>= 1;
     	 }
     	 
     	return prod; 
     }
     
    int main()
    {
        int T, N;
        unsigned long long int arr[16], p, sum, MOD=1000000007, max, val;
        scanf("%d",&T);
        for(int j=1;j<=T;j++)
        {
          scanf("%d",&N);
          
          for(int i=0;i<N;i++)
            scanf("%llu",&arr[i]);
        
         max=0;
         vector<unsigned long long int> temp1, prod, sum;
         temp1.clear();
         prod.clear();
         sum.clear();
         prod.push_back(1);
         sum.push_back(0);
         vector<unsigned long long int>::iterator it, it1, it2;
         
         for(int i=0;i<N;i++)
         {
            temp1.clear();
            
           for(it=prod.begin();it != prod.end();it++)
             	temp1.push_back( ( (*it) * arr[i] ) % MOD );
             	
           for(it=temp1.begin();it != temp1.end();++it)
              prod.push_back((*it));
             
           temp1.clear();
           
           for(it=sum.begin();it != sum.end();it++)
             	temp1.push_back( ( (*it) + arr[i] ) % MOD ); 
             	
              for(it=temp1.begin();it != temp1.end();++it)
              sum.push_back((*it)); 
            
           temp1.clear();   
         }
        	
         for(it1=prod.begin(),it2=sum.begin();it1 != prod.end() && it2 != sum.end();++it1,++it2)
         {
           	val = ( (*it1) * power((*it2)) ) % MOD;
           	if (val > max)
           	 max = val;
         }
         
           prod.clear();
           sum.clear();
         printf("Case #%d: %llu\n",j,max);  
        }
        return 0;
    }
