#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
    int t,n,num=0;
    cin>>t;
    vector<int> a,b;
    for(int i=0;i<t;i++)
    {
        cin>>n;
        for(int j=0;j<n;j++)
        {
            cin>>num;
            a.push_back(num);
        }
        for(int j=0;j<n;j++)
        {
            cin>>num;
            b.push_back(num);
        }
        sort(a.begin(),a.end());
        sort(b.begin(),b.end());
        int sum=0;
        for(int j=0;j<n;j++)
            sum+=(a[j]*b[j]);
        cout<<sum<<endl;
        a.clear();
        b.clear();
    }
    return 0;
}
