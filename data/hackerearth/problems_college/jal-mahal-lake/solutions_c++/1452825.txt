#include<bits/stdc++.h>

using namespace std;

int main()
{
    int t,n,i,j,k;
    long long sum,num;
    long long arr[50][2];
    vector<long long> leng,pos,diff,sol;
    map<long long,long long> m;
    map<long long,long long>::iterator it;
    cin>>t;
    while(t--)
    {
        cin>>n;
        sum=0;
        memset(arr,0,sizeof(arr));
        for(i=0;i<n;i++)
        {
            cin>>num;
            pos.push_back(num);
        }
        for(i=0;i<n;i++)
        {
            cin>>num;
            m[pos[i]]=num;
        }
        i=0;
        for(it=m.begin();it!=m.end();it++)
        {
            arr[i][0]=it->first;
            arr[i][1]=it->second;
            i++;
        }
        for(i=0;i<n-1;i++)
        {
            diff.push_back(arr[i+1][0]-(arr[i][0]+arr[i][1]));
        }
        for(i=0;i<n;i++)
        {

            sum=0;
            for(j=i;j<diff.size();j++)
            {
                sum+=diff[j]*(diff.size()-j);
            }
            for(k=i-1;k>=0;k--)
            {
                sum+=diff[k]*(k+1);
            }
            sol.push_back(sum);
        }
        sort(sol.begin(),sol.end());
        cout<<sol[0]<<endl;
        sol.clear();
        leng.clear();
        pos.clear();
        diff.clear();
        m.clear();
    }

    return 0;
}
