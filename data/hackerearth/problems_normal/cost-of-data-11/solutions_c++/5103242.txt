#include <iostream>
using namespace std;
int ans = 0;
struct node{
	node* next[26];
};
struct node* newNode()
{
	struct node* nn = new node;
	for(int i=0;i<26;i++)
	   nn->next[i] = NULL;
	ans++;
    return nn;
}
struct node* insert(struct node* root, string s)
{
	node* temp = root;
	for(int i=0;i<s.length();i++)
	{
		if(temp->next[s[i]-'a']==NULL)
		{
			//cout<<s<<" "<<s[i]<<endl;
			temp->next[s[i]-'a'] = newNode();
		}
		temp = temp->next[s[i]-'a'];
	}
	return root;
}
int main()
{
    int t;
    cin>>t;
    string s;
    struct node* root = newNode();
    while(t--)
    {
    	cin>>s;
    	insert(root,s);
    }
    cout<<ans<<endl;
    return 0;
}
