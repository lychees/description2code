#include <iostream>
using namespace std;
int l,r,k;

void readInp()
{
	cin>>l>>r>>k;
}

int calc()
{
	int z = 0;
	int i = 0, j = 1;
	int cnt = 1;
	j = l;
	while(j<=r){
		int z = j*k+1;
		if(z<=r){
			cnt++;
			j = z;
		}else
		break;
	}
	return cnt;
}
int main()
{
	int t;
	cin>>t;
	while(t){
	readInp();
	cout<<calc()<<endl;
	t--;
	}
}
