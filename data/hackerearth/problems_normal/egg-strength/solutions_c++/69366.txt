#include <iostream>
using namespace std;

int mat[501][501];

int max(int x,int y)
{
    if(x>y)
		return x;
	return y;
}

void dp()
{
	int i,j,k,x;
	for(i=1;i<=500;i++)
	{
		mat[i][1]=i;
	}
	
    for(i=1;i<=500;i++)
    {
		mat[1][i]=1;
	}
    
	for(i=2;i<=500;i++)
	{
		
		for(j=2;j<=500;j++)
		{
			mat[i][j]=max(mat[1][j],mat[i-2][j-1])+1;
			for(k=2;k<i;k++)
			{
				x=max(mat[k][j],mat[i-k-1][j-1])+1;
				
				if(x<mat[i][j])
					mat[i][j]=x;
				
			}
			
			
		}
		
		
	}
	
}

int main()
{
   int test,n,e;
   
   dp();
   
   cin>>test;
   while(test--)
   {
   		cin>>n>>e;
   		cout<<mat[n][e]<<endl;
   	
   	
   }
    return 0;
}
