using namespace std;
long long int f(long long int x)
{
	if(x<3)
		return 0;
	return x*(x-1)*(x-2)/6;
}
int main()
{
	int n;
	cin>>n;
	vector <int> A(n,0);
	for (int i = 0; i < n; ++i)
	{
		long long int x;
		cin>>x;
		while(x)
		{
			switch(x%10)
			{
				case 2:A[i] |= (1<<0);
						break;
				case 3:A[i] |= (1<<1);
						break;
				case 5:A[i] |= (1<<2);
						break;
				case 7:A[i] |= (1<<3);
						break;
			}
			x/=10;
		}
	}
	long long int ans=0;
	for (int mask = 1; mask < (1<<4); ++mask)
	{
		int c=0;
		long long int nc=0;
		for (int i = 0; i < 4; ++i)
		{
			if(mask & (1<<i))
				c++;
		}
		for (int i = 0; i < n; ++i)
		{
			if((A[i] & mask)==mask)
				nc++;
		}
		if(c%2)
			ans+=f(nc);
		else
			ans-=f(nc);
	}
	cout<<ans<<"\n";
	return 0;
}
