#include <iostream>
using namespace std;

int main()
{
   int t,n,i,j,m[1001],x,ma;
   cin>>t;
   while(t--)
   {
   	cin>>n;
   	for(i=0;i<n;i++)
   	{
   		for(j=0,ma=0;j<n;j++)
   		{
   			cin>>x;
   			if(x>ma)
   			ma=x;
   		}
   		m[i]=ma;
   	}
   	for(i=0,j=0;i<n;i++)
   {
   	if(j==1&&m[i]==n-1)
   	cout<<n<<" ";
   	else
   	{
   		cout<<m[i]<<" ";
   		if(m[i]==n-1)
   		j=1;
   	}
   }
   cout<<endl;
   }
   
    return 0;
}
