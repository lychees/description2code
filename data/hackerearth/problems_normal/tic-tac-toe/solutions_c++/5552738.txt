#include <bits/stdc++.h>
#define ll long long
#define m 1000000007
using namespace std;
ll modInverse(ll a)
{
    a = a%m;
    for (int x=1; x<m; x++)
       if ((a*x) % m == 1)
          return x;
}
int main()
{
    ll t,n;
    cin>>t;
    ll z=modInverse(6);
    while(t--){
    	cin>>n;
    	ll d=(n*(n-1))/2;
    	ll min=((((d/n)*(d/n))%m)*n)%m;
    	ll x=((n-1)%m*n%m)%m;
    	ll max=((((x%m)*((2*n-1)%m)%m)%m)*(z%m))%m;
    	cout<<min<<" "<<max<<endl;
    }
    return 0;
}
