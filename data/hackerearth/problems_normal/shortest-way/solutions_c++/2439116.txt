#include<bits/stdc++.h>
#define all(c) c.begin(),c.end()
#define si(x) scanf("%d",&x)
#define sii(x,y) scanf("%d%d",&x,&y)
#define sl(x) scanf("%lld",&x)
#define sll(x,y) scanf("%lld%lld",&x,&y)
using namespace std;
int arr[105][105];
int N,M;
int dp[100009];
typedef long long ll;
ll a,b;
int main(){
	int t;
	si(t);
	while(t--){
		sll(a,b);
		if(a==1&&b==1){
			cout<<"0"<<endl;
		}
		else if(a==1&&b!=1){
			cout<<(b-1)<<endl;
		}
		else if(b==1&&a>1){
			cout<<(a-1)<<endl;
		}
		else{
		ll count=0;
			while(a>1||b>1){
				if(a>b){
					count+=a/b;
					a=a%b;
				}
				else{
					count+=b/a;
					b=b%a;
				}
			}
			if(a==1&&b==1){
				cout<<count<<endl;
			}
			else{
				cout<<(count-1)<<endl;
			}
		}
	}
}
