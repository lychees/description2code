#include <bits/stdc++.h>
#define MAX 10005
using namespace std;
void seive(int n)
{
    bool prime[n+5];
    int i,j,k;
    for(i=0;i<=n;i++)
        prime[i]=true;
    prime[0]=false;
    prime[1]=false;
    for(i=2;i*i<=n;i++)
    {
        if(prime[i]==true)
        {
            for(j=i*i;j<=n;j+=i)
            {
                prime[j]=false;
            }
        }
    }
    for(i=4;i<=n;i++)
    {
        if(prime[i]==false)
            printf("%d\n",i);
    }
}
int main(int argc, char const *argv[]) {
    int n;
    scanf("%d",&n);
    seive(n);
    return 0;
}
