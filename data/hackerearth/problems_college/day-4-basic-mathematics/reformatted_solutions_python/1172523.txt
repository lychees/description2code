def findday(stark):
    n = 0
    i = 1
    while True:
        if i+n >= stark:
            return i
        n = i+n
        i = i+1


t = int(raw_input())
for i in range(t):
    stark = int(raw_input())
    battle = findday(stark)
    if battle % 3 == 0:
        print('SL')
    elif battle % 3 == 1:
        print('LB')
    else:
        print('BS')
