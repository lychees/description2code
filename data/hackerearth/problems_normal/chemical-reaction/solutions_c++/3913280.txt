int BIT[MX+5];
void add(int p,int v) {
    for(int i=p; i<=MX; i+=i&-i)BIT[i]+=v;
}

int read(int p) {
    int ret=0;
    for(int i=p; i>0; i-=i&-i)ret+=BIT[i];
    return ret;
}

string s[MX+2];
int ar[MX+2];
int main() {
    int tc,m,n,q,i,j,k,x,y,op;
    S(tc);
    while(tc--) {
        S(n);
        fr(i,1,n)cin>>s[i];
        fr(i,1,n)S(ar[i]);
        CLR(BIT);
        fr(i,1,n)add(i,+1);
        int ans=0;
        for(i=1; i<=n; i++) {
            if(i<=n-1)S(x);
            else x=1;
            int lo=1,hi=n,ret=n;
            while(lo<=hi) {
                int md=(lo+hi)>>1;
                if( read(md)>=x ) {
                    ret=min(ret,md);
                    hi=md-1;
                } else lo=md+1;
            }
            add(ret,-1);
            ans=max(ans,ar[ret]);
            printf("%s %d\n",s[ret].c_str(),ans);
        }
    }
    return 0;
}
