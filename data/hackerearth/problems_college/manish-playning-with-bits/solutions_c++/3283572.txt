int n, m;;
ll dp[55][55];


ll fnc(int x, int y)
{
    if(dp[x][y] != -1)
        return dp[x][y];
   // cout << " "<<x <<" "<<y << endl;

    if(x == n)
        return 1;
    ll ret = 0;
    ret = fnc(x + 1, 0);
    if(y != m - 1)
        ret += fnc(x + 1, y + 1);
    dp[x][y] = ret;
   // cout << x <<" "<<y <<" "<<ret << endl;
    return dp[x][y];

}
