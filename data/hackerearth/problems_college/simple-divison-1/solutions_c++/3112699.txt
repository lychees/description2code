#include<iostream>
using namespace std;
int main()
{
    long long a,b,c,t,n,i,j,m,siz;
    cin>>t;
    while(t--)
    {
        i=0;                    //Representing size of array
        cin>>a>>b>>c;
        n=a/b;
        while(n>0)
        {
            n=n/10;
            i++;
        }
        if(i!=0)
                siz=i+c+1;
        else
            siz=i+c+2;
        int s[siz];
        n=a/b;
        j=i-1;
        if(j==-1)
        {
            i=1;
            s[0]=0;
        }
        while(n>0)
        {
            s[j--]=n%10;
            n=n/10;
        }
        s[i]=-1;     //Storing the decimal
        n=a%b;
        j=i-1;
        if(c!=0)
        j=i;            //starting point of storing digits after decimal
        while(c>0)
        {
            j++;
            m=n*10;
            s[j]=m/b;
            n=m%b;
            c--;
        }
        for(i=0;i<=j;i++)
        {
            if(s[i]==-1)
                cout<<".";
            else
                cout<<s[i];
        }
        cout<<endl;
    }
    return 0;
}
