#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
const int mod = 1e9+7;
ll total(ll n){
    ll ans = ((n*(n+1))/2)%mod;
    return ans;
}
int main(){
    int t;
    cin >> t;
    while(t--){
        ll n;
        cin >> n;
        ll ans = 0;
        ll sqrtn = sqrt(n);
        for(ll i=1;i<=sqrtn;i++){
            ll rem = n/i;
            ans += (rem*i);
            ans = ans%mod;
        }
        ll m = n/sqrtn;
        for(ll i=2;i<=m;i++){
            ll fx = (n/i);
            ll fy = (n/(i-1));
            if(fx < fy){
                ll rem = (total(fy)-total(fx)+mod)%mod;
                rem = (rem*(i-1))%mod;
                ans = (ans+rem)%mod;
            }
        }
        cout << ans << endl;
    }
    return 0;
}
