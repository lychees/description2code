#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
	int N, p, evenCount = 0, oddCount = 0;
	cin >> N;
	vector<int> zo, va;
	for (int i = 0; i < N; i++) {
		cin >> p;
		if (p % 2 == 0) {
			zo.push_back(p); evenCount += p;
		} else {
			va.push_back(p); oddCount += p;
		}
	}
	sort(zo.begin(), zo.end());
	sort(va.begin(), va.end());
	int lzo = zo.size(), lva = N-lzo;
	for (int i = 0; i < lzo; i++)
		cout << zo[i] << " ";
	cout << evenCount << " ";
	for (int i = 0; i < lva; i++)
		cout << va[i] << " ";
	cout << oddCount << endl;
	return 0;
}
