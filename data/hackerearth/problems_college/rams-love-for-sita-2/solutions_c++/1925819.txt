#include<cstdio>

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,c,m,ans;
		scanf("%d %d %d",&n,&c,&m);
		ans=n/c;
		int temp=ans/m;
		int temp2=ans%m+temp;
		ans+=temp;
		while(temp2>=m)
		{
			temp=temp2/m;
			temp2=temp2%m+temp;
			ans+=temp;
		}
		printf("%d\n",ans);
	}
	return 0;
}
