#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    long long int *arr1,*arr2;
    arr1=(long long int *)malloc(sizeof(long long int)*n);
    arr2=(long long int *)malloc(sizeof(long long int)*n);
    
    for(int i=0;i<n;i++)
    {
    	int k;
    	cin>>k;
    	arr1[i]=k;
    	
    }
    for(int i=0;i<n;i++)
    {
    	int k;
    	cin>>k;
    	arr2[i]=k;
    	
    }
    
    long long int max1=0,max2=0;
    
    for(int i=0;i<n-1;i++)
    {
    	long long int k=arr1[i+1]-arr1[i];
    	long long int l=arr2[i+1]-arr2[i];
    	if(k<0) k=k*(-1);
    	if(l<0) l=l*(-1);
    	if(k>max1) max1=k;
    	if(l>max2) max2=l;
    	
    }
    if(max1==max2)
    {
    	cout<<"Tie"<<endl<<max1;
    }
     else if(max1>max2)
    {
    	cout<<"Dom"<<endl<<max1;
    }
    else if(max1<max2)
    {
    	cout<<"Brian"<<endl<<max2;
    }
    
   
    return 0;
}
