#include <iostream>
#include <cmath>

using namespace std;

int main()
{
    int t,n;
    int dp[1001];
    for(int i=0;i<1001;i++)
        dp[i]=-1;
    cin>>t;
    while(t--)
    {
        cin>>n;
        if(dp[n]==-1)
        {
            double sq=sqrt(n);
            int s=(int)sq;
            int ans=n;
            for(int i=2;i<=s;i++)
                ans+=(n/i)-(i-1);
            dp[n]=ans;
        }
        cout<<dp[n]<<endl;
    }
    return 0;
}
