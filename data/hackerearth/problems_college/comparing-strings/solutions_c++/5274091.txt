#include <iostream>
#include<map>
using namespace std;

int main()
{
    int t; cin>>t;
    while(t--){
    	bool flag=0;
    	string s1,s2; cin>>s1>>s2;
    	map<char,int>mymap;
    	for(int i=0;i<s1.length();i++) mymap[s1[i]]=1;
    	for(int i=0;i<s2.length();i++){
    		if(mymap[s2[i]]==1){
    			flag=1;
    			break;
    		}
    	}
    	if(flag) cout<<"YES"<<endl;
    	else cout<<"NO"<<endl;
    }
    return 0;
}
