typedef long long ll;
#include <iostream>
#include <stdio.h>
#include <cmath>
#include <algorithm>
#include <vector>
#include <string>
#include <string.h>
#include <map>
#include <queue>
#include <stack>
#include <sstream>
#include <fstream>
#include <stdlib.h>
#include <utility>

using namespace std;
int main()
{
	ll t,i,j,sum,n,s,a[107],flag;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
		for(i=0;i<n;i++)
		scanf("%lld",&a[i]);
		scanf("%lld",&s);
		flag=0;
		for(i=0;i<(1<<n);i++)
		{
			sum=0;
			for(j=0;j<n;j++)
			{
				if(i&(1<<j))
				sum+=a[j];
			}
			if(sum==s)
			{
				flag=1;
				break;
			}
		}
		if(flag)
		printf("YES\n");
		else
		printf("NO\n");
	}
	return 0;
}
