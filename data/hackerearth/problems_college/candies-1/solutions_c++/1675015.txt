#include <iostream>
#include <vector>
using namespace std;

int main() 
{
	long long t, n, i, temp, sum, quotient, Remainder;
	cin >> t;
	while (t--) {
		vector<long long> v;
		cin >> n;
		sum = 0;
		for (i = 0; i < n; i++) {
			cin >> temp;
			v.push_back(temp);
			sum += temp;
		}
		
		quotient = sum / n;
		Remainder = sum % n;
		
		if (!Remainder) 
			cout << "YES\n" << quotient << ' ' << Remainder << endl;
			
		else
			cout << "NO\n" << quotient << ' ' << Remainder << endl;
		
		v.erase(v.begin(), v.end());
	}
}
