s = raw_input()
sz = len(s)
l = [0]*sz
r = [0]*sz
if s[0] == '1':
    l[0] = 1
for i in xrange(1, sz):
    l[i] = l[i-1]
    if s[i] == '1':
        l[i] += 1
if s[sz-1] == '1':
    r[sz-1] = 1
for i in xrange(sz-2, -1, -1):
    r[i] = r[i+1]
    if s[i] == '1':
        r[i] += 1
ans = 0
for i in xrange(sz):
    if s[i] == '0':
        ans += l[i]*r[i]
print ans
