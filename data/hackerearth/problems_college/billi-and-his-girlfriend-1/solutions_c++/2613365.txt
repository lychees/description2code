#include <bits/stdc++.h>
using namespace std;

long long Modulo;

long fast(long long m,long long n) 
{
    long long temp=1;
       while(n>0) 
        {
          if(n%2==1) 
          temp=(temp*m)%Modulo;
          m=(m*m)%Modulo;
          n/=2;
        }
    return temp%Modulo;
}

int main()
{
    long long t;
    cin>>t;
    while(t--)
    {
    	long long a,b,k;
    	cin>>a>>b>>k;
    	long long result=0;
    	long long temp1;
    	
    	Modulo=pow(10,k);
    	temp1=fast(a,b);
    	
    	while(temp1>0)
    	{
    		result=result+(temp1%10);
    		temp1=temp1/10;
    	}
        
        cout<<result<<"\n";
    }
}
