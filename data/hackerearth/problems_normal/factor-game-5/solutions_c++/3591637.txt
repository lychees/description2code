int main()
{
	int n,i,j;
	sc(n);
	int ans=0;
	for (i=0;i<n;i++)		
	{
		int d;
		sc(d);
		int fac=0;
		for (j=2;j*j<=d;j++)
		{
			int c=0;
			while  (d%j==0)
			{
				c++;d/=j;
			}
			fac+=(c);
		}
		if (d>1)
			fac+=1;
		//printf("factors::%d\n",fac);
		ans^=(fac);
	}
	if (ans)
	{
		printf("ALICE\n");
	}

}
