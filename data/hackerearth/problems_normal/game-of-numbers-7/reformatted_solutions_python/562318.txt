test = int(raw_input().strip())
while test:
    l, h = map(int, raw_input().strip().split())
    mark = [1]*(h-l+1)
    i = 2
    while i*i <= h:
        gap = i*i
        st = l + (gap-l % gap) % gap
        while st <= h:
            mark[st-l] = 0
            st += gap
        i += 1
    print mark.count(1)
    test -= 1
