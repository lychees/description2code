#include <iostream>
using namespace std;
#include<algorithm>
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int arr[n];
    	for(int i=0;i<n;i++)
    	cin>>arr[i];
    	sort(arr,arr+n);
    	int count=1,max=1;
    	int element=arr[0];
    	for(int i=1;i<n;i++)
    	{if(arr[i]==element)
    		count++;
    		else 
    		{
    			max=(max>count)?max:count;
    			count=1;
    			element=arr[i];
    		}
    		
    	}
    	max=(max>count)?max:count;
    	cout<<max<<endl;
    }
}
