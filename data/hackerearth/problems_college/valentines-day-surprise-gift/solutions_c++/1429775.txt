#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define S(x) scanf("%lld", &x)

ll r[20] = {1,1,1,3,3,15,15,105,105,105,105};

ll mod = 1000000007ll;

ll modulo(ll a,ll b){
	ll c = mod;
    long long x=1,y=a; // long long is taken to avoid overflow of intermediate results
    while(b > 0){
        if(b%2 == 1){
            x=(x*y)%c;
        }
        y = (y*y)%c; // squaring the base
        b /= 2;
    }
    return x%c;
}

ll finde(ll n) {
	if(n <= 10) {
		return r[n];
	}
	else {
		ll ind = n/ 5;
		if(n % 5 != 0) 
			ind++;
		ll ind1 = n/ 10;
		if(n % 10 != 0) 
			ind1++;
		ll ind2 = n/ 4;
		if(n % 4 != 0) 
			ind2++;
	//	cout << n << "  " <<ind << " " << ind1 << " " << ind2 << "\n";
		ll val = (finde(ind)*finde(ind1)) % mod;
		val = (val*modulo(2, ind2)) % mod;
		return val;
	}
}

int main() {
	ll t;
	S(t);
	ll i,j;
	while(t--) {
		ll n;
		S(n);
		ll val = finde(n);
	//	cout << val  << "\n";
		val %= mod;
	//	cout << (4*val ) << " " << 5*val << "\n";
		ll ma = 0;
		for(i = 1;i <= 5;i++) {
			for(j = 1;j <= 5;j++) {
				if(__gcd(i,j) == 1) 
					ma = max(ma, modulo(i*val,j*val));
			}
		}
		
		cout << ma << "\n";
 	}
}
