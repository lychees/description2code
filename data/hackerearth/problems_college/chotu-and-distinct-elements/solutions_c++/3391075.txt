#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int n,x;
		cin>>n;
		set <int> st;
		set <int> :: iterator it1,it2;
		long long int i;
		for(i=0;i<n;i++)
		{
			cin>>x;
			st.insert(x);
		}
		for(it1=st.begin(),it2=--st.end();*it2>=*it1;it1++,it2--)
		{
			if(*it1==*it2)
			{
				cout<<*it1<<' ';
				break;
			}
			else
			{
				cout<<*it2<<' ';
				cout<<*it1<<' ';
			}
		}
		cout<<endl;
	}
	return 0;
}
