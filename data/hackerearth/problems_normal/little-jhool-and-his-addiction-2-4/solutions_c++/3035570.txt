#include <iostream>
#include <vector>
#include<algorithm>
#include <cmath>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--){
		int n,k;
		cin>>n>>k;
		vector<int> v(n);
		for(int i=0;i<n;i++){
			cin>>v[i];
		}
		sort(v.begin(),v.end(),greater<int>());
		vector<int> j;
		for(int i=0;i<v.size()/2;i++){
			j.push_back(v[i]+v[v.size()-1-i]);
		}
		sort(j.begin(),j.end());
		int ans=abs(*j.begin()-*j.rbegin());
		cout<<ans<<"\n";
		if(ans>k){
			cout<<"No more girlfriends!\n";
		}
		else if(ans==k){
			cout<<"Lucky chap!\n";
		}
		else{
			cout<<"Chick magnet Jhool!\n";
		}
	}
    return 0;
}
