#include <iostream>
using namespace std;
void solve(){
	int a,b;
	cin >> a >> b;
	int res = 1;
	int sum =0;
	for (int i = a; i <= b; i++){
		res = 1;
		for (int j = 1; j <= i; j++){
		//	cout<<i<<" ";
			res = res * j;
			res = res % 10;
		}
	//	cout <<"res"<<res<<endl;
		sum+=res;
	}
	cout << sum%10 << endl;
}
int main() {
	ios_base::sync_with_stdio(0);
	int t;
	cin >> t;
	while(t--)
		solve();
    return 0;
}
