int binarySearch(int *a , int left ,int right , int ele)
{
	int p,mid = (left+right)/2;
	while(left<right)
	{
		if(a[mid]==ele)
			left = mid + 1;
		else if(a[mid] > ele)		
			right = mid-1;
		else
			left = mid + 1;
	}
	return right;
}
