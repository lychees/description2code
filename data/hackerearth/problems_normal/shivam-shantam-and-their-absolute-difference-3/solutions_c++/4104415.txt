#include<bits/stdc++.h>
#define max_sum 400001
#define mod 1000000007
using namespace std;
 long suma[max_sum],sumb[max_sum],new_suma[max_sum],new_sumb[max_sum];
int main()
{
    long long i,j,n;
    scanf("%lld",&n);
    long long a[n],b[n];
    for(i=0;i<n;i++)
    {
        scanf("%lld",&a[i]);
    }
    for(i=0;i<n;i++)
    {
        scanf("%lld",&b[i]);
    }
    long long q;
    scanf("%lld",&q);
    for(i=0;i<max_sum;i++)
    {
        new_suma[i]=new_sumb[i]=suma[i]=sumb[i]=0;
    }
    suma[0]=1;
    sumb[0]=1;
    for(i=0;i<n;i++)
    {
        for(j=0;j<max_sum;j++)
        {
            if(suma[j]>0)
            {
                new_suma[j]=(new_suma[j]+suma[j])%mod;
                new_suma[j+a[i]]=(new_suma[j+a[i]]+suma[j])%mod;
            }
            if(sumb[j]>0)
            {
                new_sumb[j]=(new_sumb[j]+sumb[j])%mod;
                new_sumb[j+b[i]]=(new_sumb[j+b[i]]+sumb[j])%mod;
            }
        }
        for(j=0;j<max_sum;j++)
        {
            suma[j]=new_suma[j];
            sumb[j]=new_sumb[j];
            new_suma[j]=0;
            new_sumb[j]=0;
        }
    }
    /*for(j=0;j<max_sum;j++)
    {
        if(suma[j]>0)
        {
            cout<<j<<":"<<suma[j]<<"\n";
        }
    }*/
    for(j=1;j<max_sum;j++)
    {
        /*if(sumb[j]>0)
        {
            cout<<j<<":"<<sumb[j]<<"\n";
        }*/
        sumb[j]=(sumb[j]+sumb[j-1])%mod;
    }
    long long ans=0;
    for(j=0;j<max_sum;j++)
    {
        long long tmp1,tmp2,minm,maxm;
        if(suma[j]>0)
        {
            minm=(j-q);
            maxm=(j+q);
            if(minm<0)
                minm=0;
            if(maxm>=max_sum)
            {
                maxm=max_sum-1;
            }
            tmp1=sumb[maxm];
            if(minm>0)
            {
                tmp1=(tmp1-sumb[minm-1])%mod;
                if(tmp1<0)
                    tmp1=(tmp1+mod)%mod;
            }
            tmp2=(suma[j]*tmp1)%mod;
            ans=(ans+tmp2)%mod;
        }
    }
    cout<<ans;
    return 0;
}
