using namespace std;
int check(char str[])
{
    int i=0,l=0,flag=1;
    l=strlen(str);
    //cout<<str;
    while(str[i]!=0 && i<l)
   {
   	if(str[i]!=str[l-1])
   	{
   	 cout<<"-1\n";
   	 flag=0;
   	 break;
   	}
   	  else
   	{
   	   i++;
   	   l--;   	 
   	}
   }
   if(flag==0)
   return 0;
   else
   return 1;
    
}
