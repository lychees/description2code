using namespace std;
int main()
{
    /*freopen("1in.txt", "r", stdin);
    freopen("1out.txt", "w", stdout);*/
    char str[100003];
    gets(str);
    li i = 0;
    li len = strlen(str);
    li freq[26];
    fill(freq,freq+26,0);
    iter(i,0,len,1) freq[str[i]-'a']++;
    int noOdd = 0,letO = -1;
    iter(i,0,26,1)
    {
        //printf("%d ",freq[i]);
        if(freq[i]&1)
        {
            noOdd++;
            letO = i;
        }
    }
    //printf("\n");
    if((noOdd == 0 && len % 2 == 0) || (noOdd == 1 && len % 2 != 0))
    {
        char ans[len];
        int cur = 0;
        li mid = len/2;
        ans[len] = '\0';
        if(noOdd)
        {
            ans[mid] = letO + 'a';
            freq[letO]--;

        }
        iter(i,0,mid,1)
        {
            while(freq[cur] == 0){ cur++; if(cur >= 26) break;}
            if(cur >= 26) break;
            ans[i] = cur + 'a';
            ans[len - i - 1] = cur + 'a';
            //printf("%d %d %c %c\n",i,cur,ans[i],ans[len-i-1]);
            freq[cur] -= 2;
        }
        printf("%s",ans);
    }
    else printf("-1");
    return 0;
}
