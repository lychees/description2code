const int MOD=1e7+1;
long long ncr[5005][5005];
int vis[1234567];
void precompute()
{
	int k;
	for(int i = 0; i < 5005; i++)
	{
		ncr[i][0] = ncr[i][i] = 1;
		k = i>>1;
		for (int j = 1;j < k + 1;j++)
			ncr[i][j] = ncr[i][i - j] = (ncr[i - 1][j] + ncr[i - 1][j - 1]) % MOD;
	}
}
int mx;
void mark(int n)
{
	for(int j=1;j*n<=mx;j++)
	vis[j*n]=1;
}

int main()
{
	precompute();
	int a, b;
	si(a);si(b);
	if(a==0 && b==0)
	{
		cout << 0 << endl;
		return 0;
	}
	else if((a==0 && b!=0) || (a!=0 and b==0))
	{
		int cnt=0;
		cnt=1;
		cout << cnt << endl;
		return 0;
	}
	long long ans = 0;
	mx=max(a,b);
	for (int i = 2;i <= max(a,b);i++)
	{
		if ((a%i == 0) && (b%i == 0) && !vis[i])
		{
			ans = (ans + ncr[(a/i)+(b/i)][(a/i)]) % MOD;
			mark(i);
		}
	}
	pl(ans);
	return 0;
}
