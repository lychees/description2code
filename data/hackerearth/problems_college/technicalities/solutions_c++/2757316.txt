#include<bits/stdc++.h>

using namespace std;

int main(){
    int t;
	scanf("%d", &t);
	while(t--){
		long long int n, i;
		scanf("%lld", &n);
		char s[n+1];
        scanf("%s", s);
		long long int ans =0;
		i =0;
		for( ; s[i] != '\0'; ++i){
			ans =(ans*10 +s[i]-'0')%1000000007;
		}
		printf("%lld\n", ans);
	}
	return 0;
}
