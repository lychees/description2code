#include<bits/stdc++.h>
using namespace std;
int main ()
{
       int t;
       long long n,i,j,sum,res;
        cin >> t;
       while(t--)
       {
             cin>> n;
             sum=0;
             while(n)
             {
             if(n%2==1)
                res=n/2+1;
             else
                res=n/2;
             sum+=res*res;
             n=n-res;
            }
         cout << sum << endl;
       }
}
