#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define MOD 1000000007
int main()
{
    ll n,t,i;
    ll a[100001];
    a[0]=0;
    a[1]=1;
    a[2]=2;
    a[3]=4;
    for(i=4;i<=100000;i++)
    a[i]=(a[i-1]+a[i-2]+a[i-3])%MOD;
    cin>>t;
    while(t--)
    {
        cin>>n;
        cout<<a[n]<<endl;
    }
    return 0;
}
