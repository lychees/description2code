#include <iostream>
using namespace std;
#include<cstring>
#include<set>
int a[1000006];

int main()
{
	int t,i,j,k,l,m,n;
	cin>>t;
	char s[1000006];
	while(t--){
		
		cin>>k;
		cin>>s;
		n=strlen(s);
		int c=0;
		for(i=1;i<=n;i++){
			a[i]=s[i-1]-'a'+1;
			if(a[i]==k){
				c++;
			}
		}
		a[0]=0;
		int f=0;
		for(l=1;l<=n;l++){
			f=0;
			for(i=1;i<=n-l;i++){
				a[i]=a[i]+(s[i+l-1]-'a'+1);
	 			//cout<<a[i]<<endl;
 				if(a[i]==k){
 					//cout<<k<<" "<<i<< endl;
					c++;
				}
				if(a[i]>k){
					f++;
				}
				
			}
			if(f==n-l){
				break;
			}
		}
		cout<<c<<endl;
	}
	return 0;
}
