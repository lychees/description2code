#include <iostream>
using namespace std;

int A[100010];
int inc_right[100010];
int inc_left[100010];
int a,b;

int main() {
    int n, m;
    cin >> n >> m;
    for (int i = 0 ; i < n; i++) cin >> A[i];
    for (int i = 1 ; i < n; i++) {
        inc_right[i] = i;
        inc_left[i] = i;
        if (A[i] <= A[i-1]) inc_left[i] = inc_left[i-1];
    }
    for (int i = n-2; i >= 0; i--) {
        if (A[i] <= A[i+1]) inc_right[i] = inc_right[i+1];
    }
    for (int i = 0; i < m; i++) {
        cin >> a >> b;
        a--,b--;
        if (inc_right[a] >= inc_left[b]) cout << "Yes\n";
        else cout << "No\n";
    }
    return 0;
}
