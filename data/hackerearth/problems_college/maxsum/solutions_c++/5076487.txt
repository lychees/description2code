#include <iostream>
using namespace std;
long long int max(long long int a,long long int b)
{
	return (a>=b)?a:b;
}
int main()
{int t;
    long long int n,i,j;
    cin>>t;
    	long long int a[10001];
    for(i=0;i<t;i++)
    {
    	cin>>n;
    
    	long long maxx;
    		long long int dp[n];
    	for(j=0;j<n;j++)
    	   cin>>a[j];
    	  if(n==1)
    	  maxx=a[0];
    	   else if(n==2)
    	    maxx=max(a[0],a[1]);
    	    else
    	   {
    
    	dp[0]=a[0];
    	dp[1]=max(a[0],a[1]);
    	maxx=max(a[0],a[1]);
    	for(j=2;j<n;j++)
    	  { dp[j]=max(dp[j-1],dp[j-2]+a[j]);
    	  if(maxx<dp[j])
    	  maxx=dp[j];
    	  }

    }

    	  cout<<maxx<<endl;
    	
    }
    return 0;
}
