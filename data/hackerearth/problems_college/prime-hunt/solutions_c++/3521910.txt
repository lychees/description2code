#include<bits/stdc++.h>
using namespace std;


#define FOR(i,a,b) for(long long i=a;i<b;i++)
#define cin(a) scanf("%lld",%a);
#define cout(a)scanf("%lld\n",a);
#define DRT() int t;cin>>t;while(t--)


short int a[9999999];
int main()
{
	int i=0 ;
    FOR(i,0,9999999) a[i]=1;
    a[0] = 0;
    a[1] = 0;
    FOR(i,2,sqrt(9999999)+1)
    {
     if(a[i]==1)
     {
        for(int j=2*i ; j < 9999999 ;j+=i)
            a[j] = 0;
     }
    }
    DRT()
    {
    	long long ax,bx;
    	cin>>ax>>bx;
    	
    	long long cu=0;

    	FOR(i,ax,bx+1)
    	{
    		if(a[i]==1)cu++;
    	}
    	cout<<cu<<endl;
    	
    }














	return 0;
}
