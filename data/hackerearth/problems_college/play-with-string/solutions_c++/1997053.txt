#include<iostream>
#include<algorithm>
using namespace std;
#include<stdio.h>
#include<stdlib.h>
#include<string.h>


int main()
{
    int n,t,f,j,i;
    char a[100000];
    cin>>t;
    while(t--)
    {
        f=1;
        scanf("%s",&a);
        n=strlen(a);
        for( i=1,j=n-2;i<n;i++,j--)
        {
            
            if(abs(a[i]-a[i-1])!=abs(a[j]-a[j+1]))
            {
                f=0;
                break;

            }
        }
        if(f==0)
            cout<<"Not Funny\n";
        else
                     cout<<"Funny\n";

    }
}
