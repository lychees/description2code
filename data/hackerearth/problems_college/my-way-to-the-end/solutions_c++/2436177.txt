#include <iostream>
using namespace std;
string str[105];
int k;
bool bfs(int i,int j)
{
	if(i==k-1 && j==k-1) return true;
	if(str[i][j]=='1')
	{
	 if(i==k-1) return bfs(i,j+1);
	else if(j==k-1) return bfs(i+1,j);
	else return (bfs(i+1,j) || bfs(i,j+1));
	}
	else return false;
}
int main()
{
    int t,i,j;
    //int a[105][105];
    scanf("%d",&t);
    while(t--)
    {
    	int n,count=0;
    	scanf("%d%d",&n,&k);
    	
    	for(i=0;i<n;i++) cin>>str[i];
    	if(str[k-1][k-1]=='0') printf("%d\n",-1);
    	else
    	{	
    		 bool fun=false;
    		 fun=bfs(0,0);
    	
    	if(fun) printf("%d\n",2*(k-1));
    	else printf("%d\n",-1);
    	}
    	
    }
    return 0;
}
