int main() {
	int T; cin >> T;
	while (T--) {
		string s, t; cin >> s >> t;
		int n = s.length(), m = t.length();
		int mini = INT_MAX;
		for (int i = 0; i + m <= n; i++) {
			int cnt = 0;
			for (int j = 0; j < m; j++)
				if (s[i + j] != t[j])
					cnt++;
			mini = min(mini, cnt);
		}
		cout << mini << endl;
	}
}
