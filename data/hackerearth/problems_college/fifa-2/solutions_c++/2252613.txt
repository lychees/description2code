#include<bits/stdc++.h>

using namespace std;

int a[200005],b[200005];
int c[200005],d[200005];
int main(){
	int n;
	cin >> n;
	for(int i=0;i<n;i++){
		scanf("%d %d",&c[i],&d[i]);
		a[c[i]]++;
		b[d[i]]++;
	}
	for(int i=0;i<n;i++){
		int home=(n-1)+a[d[i]];
		int away=2*(n-1)-home;
		printf("%d %d\n",home,away);
	}
	return 0;
}
