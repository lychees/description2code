#include <bits/stdc++.h>
using namespace std;

int main()
{
   vector<long long int> spd;
   vector<long long int> dist;
   long long int t;
   long long int total_fuel;
   double max = 0;
   double distance;
   long long int q;
   long long int max_int;
   cin >>t;
   for(int i=0 ; i<t ; i++)
   {
   	cin >> q;
   	spd.push_back(q);
   }
   for(int i=0 ; i<t ; i++)
   {
   	cin>>q;
   	dist.push_back(q);
   }
   cin >> total_fuel;
   for(int i=0 ; i< t; i++)
   {
   	distance = (double)total_fuel*((double)spd[i]/(double)dist[i]);
   	if(max < distance)
   	{
   		max = distance;
   		max_int = max*1000;
   		max = (double)max_int/1000;
   	}
   }
   printf("%.3f" , max);
    return 0;
}
