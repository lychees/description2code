#include<iostream>
#include<stdio.h>

#define MAX 100000

int main()
{
	int n,i,max1=0,ct=0;
	int a[MAX];
	scanf("%d",&n);
	for(i=0; i<n; i++)
	{
		scanf("%d", &a[i]);
		if(i>0)
		{
			if(a[i]-a[i-1]==1 || a[i-1] - a[i]==1)
				ct++;
				else
				  ct=0;

            if(max1<=ct)
			{
				max1=ct;

			}

		}
	//	printf("%d ", max1);
	//	printf("%d ", ct);
	}
	printf("%d",max1+1);
	return 0;
}
