#include <iostream>

using namespace std;

int main()
{
    long long int test;
    cin >> test;
    while(test--) {
        long long int a;
        cin >> a;
        if(a == 3) {
            cout << 555 << endl;
        }
        else if(a == 5) {
            cout << 33333 << endl;
        }
        else {
            if((a / 3 ) < 1)
                cout << -1 << endl;
            else {
                long long int d = a / 3, maxv = 0;
                while(d) {
                    if((a - 3*d) % 5 == 0 || (a-3*d) == 0) {
                        if(maxv < d) {
                            maxv = d*3;
                        }
                    }
                    d--;
                }
                long long int f = a - maxv;
                if(maxv != 0) {
                    while(maxv--)
                        cout << 5;
                    while(f--)
                        cout << 3;
                    cout << endl;
                }
                else
                    cout << -1 << endl;
            }
        }
    }
    return 0;
}
