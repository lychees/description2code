#include <iostream>
#include <algorithm>
using namespace std;

long long int mod_inv(long long int a, long long int b)
{
	long long int b0 = b, t, q;
	long long int x0 = 0, x1 = 1;
	if (b == 1)
		return 1;
	while (a > 1)
	{
		q = a / b;
		t = b, b = a % b, a = t;
		t = x0, x0 = x1 - q * x0, x1 = t;
	}
	if (x1 < 0)
		x1 += b0;
	return x1;
}

int main()
{
	ios_base::sync_with_stdio(false);
	long long int T,N,R,i,a,b,c,m=1000000007,temp;
	cin>>T;
	while(T--)
	{
		cin>>N>>R;
		for(i=0;i<N;i++)
		{
			cin>>temp;
			R-=temp;
		}
		N--;
		long long int numr=1,denr=1;
		for(i=R+1;i<=R+N;i++)
		{
			numr = (numr*i)%m; 
		}
		for(i=1;i<=N;i++)
		{
			denr=(denr*mod_inv(i,m))%m;
		}
		cout<<(numr*denr)%m<<"\n";
	}
	return 0;
}
