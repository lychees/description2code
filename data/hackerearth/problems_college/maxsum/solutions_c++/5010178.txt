#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        int a[10005];
        for(int i=1;i<=n;i++)
            cin>>a[i];
        long int dp[10005];
        dp[0] = 0;
        dp[1] = a[1];
        for(int i=2;i<=n;i++)
            dp[i] = max(dp[i-1],a[i]+dp[i-2]);
        cout<<dp[n]<<endl;
    }
    return 0;
}
