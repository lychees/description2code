#include <iostream>
using namespace std;

int main()
{
    char ch[26]={'a','b','c','d','e','f','g','h','i','j','k','l','m','n','o','p','q','r','s','t','u','v','w','x','y','z'},str[10000],err[26];
    int wt[10000],n,k,sum,flag=2,h,l,a;
   
   	cin>>n;
   	for(int i=0;i<n;i++)
   	{
   		cin>>str;
   		cin>>k;
   		for(int f=0;f<k;f++)
   		{
   			cin>>err[f];
   		}
   		sum=0;
   		
   		for(l=0;str[l]!='\0';l++)
   		{
   			flag=0;
   			for(int a=0;a<k;a++)
   			{
   				if(str[l]==err[a])
   				{
   					flag=1;
   					break;
   				}
   			}
   			
   			if(flag==0)
   			{
   				for(int h=0;h<26;h++)
   				{
   					if(str[l]==ch[h])
   					{
   						sum=sum+(h+1);
   						break;
   					}
   				}
   			}
   			
   		}
   		
   	
   		
   		wt[i]=sum;
   		
   	}
   	
   	for(int d=0;d<n;d++)
   	{
   		cout<<wt[d]<<"\n";
   	}
    
    return 0;
}
