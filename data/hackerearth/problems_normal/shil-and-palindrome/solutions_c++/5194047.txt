#include <iostream> 
using namespace std ; 
#include <bits/stdc++.h>
int main()

{
	string s ; 
	cin >> s ; 
	string s2 ; 
	string s3 ; 
	int  a[26] ={0} ;
	int i ;  
	for ( i = 0 ; i< s.length() ; i++)
	{
		a[s[i]-'a']++ ; 
	}
	
 
	int cnt = 0 ; 
	for ( i = 0 ; i< 26 ; i++)
	{ 
	  if ( a[i]%2)
	  {
	  	s2+=i+'a' ; 
	  	cnt++ ; 
	  	if ( cnt>1)
	  	{
	  		cout << "-1" ; 
	  		return 0 ; 
		}
		  }	
		while(a[i])
		{
			if ( a[i] == 1)
			{
				break ; 
			}
			s3+=i+'a' ; 
			a[i] = a[i] -2 ; 
		  }  
	}
	s = s3 ; 
	reverse(s3.begin() , s3.end()) ; 
	s = s+s2+s3 ; 
	cout <<s <<endl; 
	return 0 ; 
}
