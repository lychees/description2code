#include <iostream>
#include <math.h>
using namespace std;
int gcd(int a,int b)
{
	if(b!=0)
	return gcd(b,a%b);
	else
	return a;
}
int main()
{
	int t,n,k=0;
    cin>>t;
    int a[100001];
    int b[100000];
    b[k++]=2;
    a[2]=1;
    for(int i=3;i<=100000;i+=2)
    {
    	int flag=0;
    	for(int j=0;j<sqrt(i) && j<k && b[j]<=i ;j++)
    	{
    			if(i%b[j]==0)
    			flag=1;
    	}
    	if(flag==0)
    	{b[k++]=i;
    	a[i]=1;
    	}
    }
    while(t--)
    {
    	cin>>n;
    	int count=1;
    	for(int i=2;i<n;i++)
    	{
    		if(gcd(i,n)==1)
    		{
    			count++;
    		}
    	}
    	if(a[count]==1)
    	cout<<"TRUE"<<endl;
    	else
    	cout<<"FALSE"<<endl;
    	
    }
    return 0;
}
