LL k;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--) {
        scanf("%lld",&k);
        int ct = 1;
        while(k>0) {
            k -= (LL)(1LL<<ct);
            ++ct;
        }
        if(k == 0) {
            for(int i=1;i<ct;++i) printf("7");
            //printf("\n");
        }
        else {
            --ct;
            k += (1LL<<ct);
            --ct;
            while(ct>=0) {
                if(k>(1LL<<ct)) {
                    printf("7");
                    k -= (1LL<<ct);
                    --ct;
                }
                else {
                    printf("4");
                    --ct;
                }    
            }        
        }
        printf("\n");
    }
    return 0;
}
