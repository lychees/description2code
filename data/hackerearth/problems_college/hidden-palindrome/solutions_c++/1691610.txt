#include <iostream>
using namespace std;
int n, i, j, k, l;
string str;
int f(int x, int y, int l){
	if(str[x] != str[y])
		return 0;
	while(x >= 0 && y < l){
		if(str[x] == str[y]){
			x --; y++;	
		} else {
			return (min(y - 1, l) - max(x  +1, 0) + 1);
		}
	}
	return (min(y - 1, l) - max(x + 1, 0) + 1);
}
int main()
{
	//string str;
	cin >> n;
	while(n--){
		cin >> str;	
		l = str.size();
		for(i = 0; i < l-1; i++){
			j = max(j, f(i, i, l));
			j = max(j, f(i, i+1, l));
		}
		j = max(1, j);
		cout << j << endl;
		j = 0;
	}
    return 0;
}
