test=int(raw_input())
def find(x,low,high):
	if high==low:
   		print low
   	else:
   		mid=(low+high+(low+high)%2)/2
   		val=x[0]*(mid**3)+x[1]*(mid**2)+x[2]*mid+x[3]
   		if val>x[-1]:
   			find(x,low,mid-1)
   		elif val<x[-1]:
   			find(x,mid,high)
   		else:
   			print mid
for t in range(test):

   	high=1000000
   	low=0
   	find(x,low,high)
