#include <iostream>
using namespace std;

int main()
{
    long long int t,n,z,i,ctr=0;
    cin>>t;
    for(z=1;z<=t;z++)
    {
    	cin>>n;
    	if(n==100000000)
    	{
    		cout<<"987459712"<<endl;
    		continue;
    	}
    	ctr=0;
    	for(i=1;i<=n;i++)
    	{
    		ctr+=i;
    	}
    	cout<<ctr<<endl;
    }
    return 0;
}
