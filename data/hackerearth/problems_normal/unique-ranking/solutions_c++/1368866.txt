#include <bits/stdc++.h>
using namespace std;
int arr[1005][25]; 
int main()
{
    int a,b; cin >> a >> b;
    for (int g=1;g<=a; g++)
    {
        for (int y=1;y<=b; y++)
        {
            cin >> arr[g][y]; 
        }    
    }
    int q; cin >> q;
    for (int g=0; g<q; g++)
    {
        int r; cin >> r;
        vector <int> students; for (int y=1;y<=a; y++) students.push_back(y); 
        for (int y=0; y<r; y++)
        {
            int l; cin >> l; 
            if(students.size()==1) continue; 
            vector <int> potnew; 
            int curmin=1e9; 
            for (int z=0; z<students.size(); z++)
            {
                if (arr[students[z]][l]<curmin)
                {
                    curmin=arr[students[z]][l]; potnew.clear(); 
                    potnew.push_back(students[z]); 
                }
                else if (arr[students[z]][l]==curmin)
                {
                    potnew.push_back(students[z]); 
                }
            }
            students=potnew; 
         }    
         cout << students[0] << '\n'; 
    }
    return 0; 
}
