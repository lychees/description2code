#include <iostream>
using namespace std;

int gcd ( int a, int b )
{
  if ( a==0 ){
  	return b;
  }else{
  	gcd ( b%a, a );
  }
}

int main()
{
    int S, N;
    cin>>S>>N;
    int A[N];
    for(int i = 0; i < N; i++){
    	cin>>A[i];
    	A[i] = gcd(S, A[i]);
    }
    cout<<A[0]<<endl;
    for(int i = 1; i < N; i++){
    	for(int j = 0; j < i; j++){
    		if(A[i] == A[j]){
    			A[i] = -1;
    			break;
    		}
    	}
    	cout<<A[i]<<endl;
    }
}
