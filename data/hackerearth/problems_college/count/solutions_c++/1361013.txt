#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,r,c,ans1,ans,i,j;
	scanf("%d",&t);
	while(t--)
	{
		ans=0;
		ans1=0;
		scanf("%d %d",&r, &c);
		for(i=1;i<=r;i++)
		{
			if(i<=c)
			{
				if(i%2==0){
					ans=ans+i;}
				else{
					ans1=ans1+i;}
			}
			if(i>c)
			{
				if(i%2==0){
					ans=ans+c;}
				else{
					ans1=ans1+c;}
			}
		}
		printf("%d %d\n",ans ,ans1);
	}
	return 0;
}
