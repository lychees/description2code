#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int x, y; 
    vector <int> chu;
    
    while(cin >> x){
    	while(x--){
	    	cin >> y;
	    	if(y % 2 == 0)
	    		chu.push_back(y);
	   	}
	   	
	   	if(chu.size() != 0){
	    	for(int i = chu.size()-1; i >= 0; i--){
	    			cout << chu[i] << " ";
	    	}
	   	}
	   	else
	   		cout << "None to display";
	    	
	    	cout << endl;
	    	chu.clear();
    }
}
