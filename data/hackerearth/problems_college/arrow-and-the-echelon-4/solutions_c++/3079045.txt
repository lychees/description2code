#include<stdio.h>
using namespace std;

int main()
{
    long long int n,fv,th,a;
    scanf("%lld",&n);
    for(int j=1;j<=n;j++)
    {
        scanf("%lld",&fv);
        th=0;
        while((fv%3)!=0)
        {
            th=th+5;
            fv=fv-5;
            if(fv<5) break;
        }
        if(th%5==0 && fv%3==0)
        {
            for(int i=1;i<=fv;i++)
                printf("5");
            for(int i=1;i<=th;i++)
                printf("3");
            printf("\n");
        }
        else printf("-1\n");
    }

}
