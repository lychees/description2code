LL a[200005],b[200005];
LL pa[200005],pb[200005];
const LL mod = 1e9+7;

int main(){
	int n;
	si(n);
	for(int i=0;i<n;i++){
		sll(a[i]);
		sll(b[i]);
	}
	sort(a,a+n);
	sort(b,b+n);
	pa[n-1]=0;
	pb[n-1]=0;
	for(int i=n-2;i>=0;i--){
		pa[i] = (pa[i+1] + a[i+1]) % mod; 
		pb[i] = (pb[i+1] + b[i+1]) % mod; 
	}
	LL ans=0;
	for(int i=0;i<n;i++){
		ans = (ans + pa[i] - ((n-i-1)*a[i]) % mod ) % mod;
		ans = (ans + pb[i] - ((n-i-1)*b[i]) % mod ) % mod;
	}
	lldout(ans);
	return 0;
}
