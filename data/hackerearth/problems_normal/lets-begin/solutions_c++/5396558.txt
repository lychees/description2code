#include <bits/stdc++.h>
using namespace std;
int dp[4][1000005];
int main()
{
    int p[4]={2,3,5,7};
    for(int i=0;i<4;i++)
     dp[i][0]=dp[i][1]=1000005;
    for(int i=0;i<4;i++)
    {
    	for(int j=0;j<1000005;j++)
    	{
    		dp[i][j] = 1000005;
    		if(i>0 && j<p[i]) dp[i][j]=dp[i-1][j];
    		else if(j==p[i]) dp[i][j]=1;
    		//else if(dp[i][j-p[i]]>1000005) continue;
    		else dp[i][j]=min(1+dp[i][j-p[i]],dp[i][j]);
    	}
    }
    int t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	if(n==1) cout<<"-1"<<endl;
    	else cout<<dp[3][n]<<endl;
    }
    return 0;
}
