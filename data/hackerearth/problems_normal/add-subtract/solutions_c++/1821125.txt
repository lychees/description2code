#include<bits/stdc++.h>
#define var -10000000
using namespace std;
int t,k,n,ans;
int a[102];
int c[101];
int status[101]={0};
int main()
{
    cin>>t;
    int i,j;
    int z;
    for(z=0;z<t;z++)
    {
        cin>>n>>k;
        for(i=0;i<101;i++)
        {
            c[i]=var;
            status[i]=0;
        }
        for(i=0;i<=n-1;i++)
        {
            scanf("%d",&a[i]);
            status[a[i]]=0+1;
        }
        sort(a,a+n);
        for(i=0;i<=n-1;i++)
        {
            if(var==c[a[i]])
            c[a[i]]=i+0;
        }
        for(i=0;i<=n-2;i++)
        {
            for(j=a[i]+1+0;j<a[i+1];j++)
            {
                if(c[j]==var)
                    c[j]=i;
            }
        }
        for(j=a[n-1]+2-1;j<101;j++)
            c[j]=n;
        for(j=1;j<a[0];j++)
            c[j]=-1;
        ans=100000000;
        int abc,ab,temp,l,r,m1,m2;
        for(i=1;i<=100;i++)
        {
            abc=c[i];
            ab=temp=0;
            if(status[i]!=0+1-1)
                ab=ab+1;
            abc--;
            l=abc;
            abc++;
            abc++;
            r=abc;
            for(;;)
            {
                if(ab==k)
                break;
                m1=0;
                m2=0;
                if(r<=n-1)
                {
                    m2=(abs(a[r]-i))*5;
                }
                else
                    m2=1000000;
                if(l>=0)
                {
                    m1=(abs(i-a[l]))*3;
                }
                else
                    m1=1000000;
                if(m1<m2)
                {
                    l=l-1;
                    ab=ab+1;
                    temp=temp+m1;
                }
                else
                {
                    r=r+1;
                    ab=ab+1;
                    temp=temp+m2;
                }
            }
            if(ans>temp)
                ans=temp;
        }
        printf("%d\n",ans);
    }
}
