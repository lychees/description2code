#include <iostream>
#define SIZE 1000100
using namespace std;

bool arr[SIZE];

bool check(int num) {
	int last=0;
	while(num>0) {
		if(num%2==1) {
			if(last==1)
				return false;
			last=1;
		}
		else last = 0;
		num=num/2;
	}
	return true;
}

void pre() {
	
	for(int i=1; i<SIZE; i++) {
		arr[i] = 0;
		if(check(i)) arr[i] = 1;
	}
}	
int main()
{
    int test;
    cin>>test;
    pre();
    while(test--) {
    	int num;
    	cin>>num;
    	for(int i=num; i<SIZE; i++)
    		if(arr[i]) {cout<<i<<endl;break;}
    }
    return 0;
}
