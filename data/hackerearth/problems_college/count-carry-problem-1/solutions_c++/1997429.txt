using namespace std;
typedef pair<ll,ll> ii;
typedef pair<ii,ii> iii;
typedef vector<ll> vv;
char a[50],b[50];
int main()
{
	ll t;
	scanf("%lld\n",&t);
	while(t--)
	{
	   ll as,bs,aptr,bptr,carry=0,ctr=0;
       scanf("%s %s",a,b);
       as=strlen(a);
       bs=strlen(b);
       aptr=as-1,bptr=bs-1;
       while(aptr>=0&&bptr>=0)
       {
       	 ll x=a[aptr]-'0'+b[bptr]-'0'+carry;
         carry=x/10;
         if(carry)
         	ctr++;
         aptr--,bptr--;
       }
       while(aptr>=0)
       {
         ll x=a[aptr]-'0'+carry;
         carry=x/10;
         if(carry)
         	ctr++;
         aptr--;
       }
       while(bptr>=0)
       {
       	ll x=b[bptr]-'0'+carry;
         carry=x/10;
         if(carry)
         	ctr++;
         bptr--;
       }
     if(!ctr)
     	cout<<"No carry operation\n";
     else if(ctr==1) cout<<"1 carry operation\n";
     else cout<<ctr<<" carry operations\n";
	}
}
