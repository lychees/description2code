#include <iostream>
using namespace std;

int main()
{
    int t,flag;
    cin>>t;
    string s1,s2,s3;
    while(t--)
    {
    	flag=0;
    	cin>>s1>>s2;
    	if(s1.length()==s2.length())
    	{
    		for(int i=0,j=s1.length()-1;i<s1.length();i++,j--)
    		{
    			if(s1[i]!=s2[j])
    			{
    				flag=1;
    				break;
    			}
    		}
    		if(flag==0)
    			cout<<"YES"<<endl;
    		else
    			cout<<"NO"<<endl;
    	}
    	else
    	{
    		for(int i=0,j=s2.length()-1;i<s1.length()&&j>=0;i++,j--)
    		{
    			if(s1[i]!=s2[j])
    			{
    				flag=1;
    				break;
    			}
    		}
    		if(flag==0)
    			cout<<"YES"<<endl;
    		else
    		{
    			for(int i=0,j=s1.length()-1;i<s2.length()&&j>=0;i++,j--)
    			{
    				if(s2[i]!=s1[j])
    				{
    					flag=1;
    					break;
    				}
    			}
    			if(flag==0)
    				cout<<"YES"<<endl;
    			else
    				cout<<"NO"<<endl;
    		}
    	}
    }
    return 0;
}
