#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    scanf("%d",&t);
    while(t--){
    	double p;
    	double tt;
    	scanf("%lf%lf",&p,&tt);
    	if(p==1.0){
    		return 0;
    	}
    	printf("%.4lf\n",1-pow(1-p,tt/1440.0));
    }
    return 0;
}
