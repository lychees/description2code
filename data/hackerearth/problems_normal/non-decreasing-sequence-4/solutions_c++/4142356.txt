int main()
{
    int test;
    cin>>test;
    while(test--)
    {
        int n;
        cin>>n;
        int arr[n];
        int stack[n+1];
        memset(stack,0,sizeof(stack));
        stack[0] = -1;
        int top = 0;
        for(int i=0;i<n;++i)
        {
            int temp;
            cin>>temp;
            if(temp>=stack[top])
                stack[++top] = temp;
            else
            {
                int low = 1, high = top;
                while(low<=high)
                {
                    int mid = (low+high)>>1;
                    if(temp>stack[mid])
                        low = mid+1;
                    else
                        high = mid-1;
                }
                stack[low] = temp;
            }
        }
        // for(int i=1;i<=n;++i)
        //     cout<<stack[i]<<" ";
        // cout<<endl;
        int len;
        cin>>len;
        if(top>=len)
            cout<<stack[len]<<endl;
        else
            cout<<"-1"<<endl;
    }
    
}
