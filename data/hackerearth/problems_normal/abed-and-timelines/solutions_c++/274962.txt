#include <iostream>
using namespace std;

int main()
{
    int tc,N,M,x1,x2,y1,y2,p;
    int x , y, ans;
    cin>>tc;
    while(tc--)
    {
    	cin >> N >> M;
    	cin >> x1 >> y1;
    	cin >> x2 >> y2;
    	cin >> p;
    	
    	
    	if( (x2-x1) < 0)
    	x = x1 - x2;
    	else x = x2 - x1;
    	
    	if((y2-y1) < 0)
    	y = y1 - y2;
    	else y = y2-y1;
    	
    	if(x > ( N - x)) x = N-x;
    	if(y > (M-y)) y = M-y;
    	
    	ans = (x+y) * p;
    	
    	cout << ans << endl;
    	if(ans > 1000)
    	cout << "Let go of the group." << endl;
    	else
    	cout << "You saved the group." << endl;
    	
    	
    	
    	
    	
    }
    return 0;
}
