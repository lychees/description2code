#include <cstdio>
#include <string>
#include <iostream>
#include <vector>
#include <cmath>
#include <cstring>
#define MOD 1000000007
using namespace std;
vector<int> v;
long long npowm(long long a, long long b)
{
	long long ans=1;
	while(b)
	{
		if(b&1)
			ans=(ans*a)%MOD;
		b/=2;
		a=(a*a)%MOD;
	}
	return ans;
}
long long fact[1000010];
long long ifact[1000010];
void pre(){
	fact[0]=1;
	for(int i=1;i<=1000005;i++)
		fact[i]=(fact[i-1] * i) % MOD;
	ifact[0]=1;
	for(int i=1;i<=1000005;i++)
		ifact[i]=npowm(fact[i],MOD-2);
}
int arr[8];
int brr[28];
int main(){
pre();
int t,n,cnt,vow;
string s;
scanf("%d",&t);
while(t--){
cin>>s;
cnt=0;
vow=0;
n=s.size();
memset(arr,0,sizeof arr);
memset(brr,0,sizeof brr);
for(int i=0;i<n;i++){
    if(s[i]=='a'){
        arr[0]++;
        vow++;}
    else if(s[i]=='e'){
        arr[1]++;
        vow++;}
    else if(s[i]=='i'){
        arr[2]++;
        vow++;}
    else if(s[i]=='o'){
        arr[3]++;
        vow++;}
    else if(s[i]=='u'){
        arr[4]++;
        vow++;}
    else{
        cnt++;
        brr[s[i]-'a']++;
        }

}
if(cnt<(n/2)){
    printf("-1\n");
    continue;
}
long long ans1=1,ans2=1;
ans2=(ans2*fact[cnt])%MOD;
for(int i=0;i<28;i++){
    ans2=(ans2*ifact[brr[i]])%MOD;
}
long long left = cnt+1;
ans1=(fact[left])%MOD;
ans1=(ans1*ifact[left-vow])%MOD;
for(int i=0;i<6;i++){
    ans1=(ans1*ifact[arr[i]])%MOD;
}
printf("%lld\n",(ans1*ans2)%MOD);
}
return 0;
}
