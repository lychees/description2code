#include<stdio.h>
long int arr[1000005];

void fun()
{
	long int max=1000003;
	for(long int i=1;i<max;i++)
		arr[i]=i;
	for(long int i=2;i<max;i++)
	{
		if(arr[i]==i)
		{
			for(long int j=2*i;j<max;j+=i)
				arr[j]=(arr[j]/i)*(i-1);
		}
	}
	for(long int i=2;i<max;i++)
	{
		if(arr[i]==i)
			arr[i]=i-1;
		arr[i]+=arr[i-1];
	}
}
int main()
{
	long int t,n,m;
	scanf("%ld",&t);
	fun();
	while(t--)
	{
		scanf("%ld %ld",&n,&m);
		printf("%ld\n",arr[n/m]);
	}
	getchar();
	getchar();
	return 0;
}
