#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int aa,bb,ab,ba;
    	cin>>aa>>bb>>ab>>ba;
    	ab=min(ab,ba);
    		for(int i=0;i<aa/2;i++)
    			cout<<"aa";
    		for(int i=0;i<ab;i++)
    			cout<<"ab";
    		if(aa%2==0)
    		{
    			for(int i=0;i<bb;i++)
    			     cout<<"bb";
    		}
    		else
    		{
    		    for(int i=0;i<bb/2;i++)
    		    	cout<<"bb";
    		    cout<<"aa";
    	    	for(int i=0;i<bb/2;i++)
    			    cout<<"bb";
    		}
    		for(int i=0;i<ab;i++)
    			cout<<"ba";
    		for(int i=0;i<aa/2;i++)
    			cout<<"aa";

    	cout<<endl;
    }
    return 0;
}
