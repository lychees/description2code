#include <iostream>
#include<cmath>
using namespace std;

int main()
{
	int t;cin>>t;
	while(t--)
	{
		long n;
		cin>>n;
		long k=sqrt(n);
		if(k*k == n)
		cout<<"YES\n";
		else
		{
			if(abs(k*k - n)<abs((k+1)*(k+1) - n))
			      cout<<(k*k - n)<<endl;
			      else
			      cout<<"+"<<((k+1)*(k+1) - n)<<endl;
		}
		
	}
   // cout << "Hello World!" << endl;
    return 0;
}
