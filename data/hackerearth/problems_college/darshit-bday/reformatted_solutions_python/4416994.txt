t = input()
while (t > 0):
    b, w = map(int, raw_input().split())
    x, y, z = map(int, raw_input().split())
    if (x == y):
        print(b * x) + (w * y)
    elif (x < y):
        if (x + z <= y):
            print(b * x) + (w * (x + z))
        else:
            print(b * x) + (w * y)
    else:
        if (y + z <= x):
            print(b * (y + z)) + (w * y)
        else:
            print(b * x) + (w * y)
    t -= 1
