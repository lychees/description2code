long double  C[1001][1001]={0};
void fun()
{
int  i,k;
 
	for (i=0; i<1001; i++)
	{
		for (k=0;k<=i;k++)
			if (k==0 || k==i)
				C[i][k] = 1;
			else
				C[i][k] = (C[i-1][k-1]+C[i-1][k]);
	}
 
}
int main()
 {
fun();
int  t,s,n,m,k,i;
long double res,res1,res2;
    cin>>t;
      while(t--)
        {  res=0;
           cin>>s>>n>>m>>k;
              if(s==n)
              {
            cout<<"1.000000"<<endl;
            continue;
            }
             else if(n<k)
             {
            cout<<"0.000000"<<endl;
             continue;
             }
             else
                {
                  for(i=k;i<=min(m,n);i++)
                      {
                      res1= ((C[m][i])*(C[s-m][n-i]));
                      res+=res1;
                        }
               res2=C[s][n];
               
		printf("%0.6Lf\n",res/res2);
                 }
}
