#include <iostream>
#include<string.h>
using namespace std;
char s[109];
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		cin>>s;
		int l=strlen(s);
		int i;
		for(i=0;i<l/2;i++)//cout<<l<<" "<<s[i]<<" "<<s[l-1-i]<<endl;
		if(s[i]!=s[l-1-i])break;
		//cout<<i<<endl;
		if(i==l/2)cout<<"YES"<<endl;
		else cout<<"NO"<<endl;
	}
    return 0;
}
