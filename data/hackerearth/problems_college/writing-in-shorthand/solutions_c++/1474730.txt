#include <iostream>
#include<string.h>
using namespace std;
int lena,lenb,m,n;
char a[3001][11],b[3001][11],c[3001][11];
int main()
{
	int t,i,j,k;
	cin>>n>>m;
	for(i=0;i<m;i++)
	{
		cin>>a[i]>>b[i];
	}
	
	for(i=0;i<n;i++)
	{
		cin>>c[i];
	}
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			if(strcmp(c[i],a[j])==0)
			{
				//cout<<"x";

				break;
			}
			
		}
		
	}

}
