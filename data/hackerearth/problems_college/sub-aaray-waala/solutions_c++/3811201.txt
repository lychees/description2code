#include <bits/stdc++.h>
using namespace std;
#define ll long long
ll mod=1000000007;
ll Power(ll a,ll b){
	if(b==0)
		return 1;
	ll temp=Power(a,b/2);
	temp=temp*temp;
	if(temp>=mod)
		temp%=mod;
	if(b%2==1){
		temp*=a;
	if(temp>=mod)
		temp%=mod;
	}
	return temp;
}

int main()
{
	ll i,j,k,n,m,T,ans;
	scanf("%lld",&T);
	while(T--){
		set<ll> st;
		set<ll>::iterator it;
		scanf("%lld",&n);
		for(i=0;i<n;i++){
			scanf("%lld",&k);
			st.insert(k);
		}
		ans=Power(n,st.size());
		cout<<ans<<endl;
	}
    return 0;
}
