#include <bits/stdc++.h>

using namespace std;



int main(){
	long long int T,a[100003],b[100003],N,lar=0;
	cin >> T;
	for (int i = 0; i < T; ++i)
	{
		/* code */

		cin >> N;
		for (int i = 0; i < N; ++i)
		{
			/* code */
			cin >> a[i];
		}
		for (int i = 0; i < N; ++i)
		{
			/* code */
			cin >> b[i];
		}
		for (int i = 0; i < N; ++i)
		{
			/* code */
			int min = i-1;
			int max = N;
			int mid;
			while(max - min > 1){
				mid = (min + max)/2;
				if(b[mid] >= a[i]){
					min = mid;
				}
				else max = mid;
			}
			if(min - i > lar) lar = min - i;

		}
		cout << lar << endl;
		lar = 0;

	}



	return 0;
}
