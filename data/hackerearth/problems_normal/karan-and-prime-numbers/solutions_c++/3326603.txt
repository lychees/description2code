#include<iostream>
#include<vector>
#include<math.h>
#include<algorithm>
using namespace std;
int main()
{
	vector<long> v;
	v.push_back(2);
	v.push_back(3);
	int c;
	for(long i=5;i<=1000000;i+=2)
	{
		long i1=sqrt(i);
		c=0;
		for(long j=0;v[j]<=i1 && j!=v.size();j++)
		{
			if(i%v[j]==0)
			{
				c=1;
				break;
			}
		}
		if(c==0)
		v.push_back(i);
	}
	long t;
	cin>>t;
	while(t--)
	{
		long long n,m;
		cin>>n>>m;
		c=0;
		long long s=0;
		for(long i=0;i!=v.size();i++)
		{
			
			if(c==1)
			{
				if(v[i]<=m)
				s+=v[i];
				else
				break;
			}
			if(c==0 && v[i]>=n)
			{
				if(v[i]>m)
				break;
				s+=v[i];
				c=1;
			}
		}
		cout<<s<<endl;
	}
	return 0;
}
