#include <iostream>
#include <algorithm>
using namespace std;
bool comparator(int a, int b)
{
	return a>b;
}


int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,k;
    	cin>>n>>k;
    	int a[n];
    	int dp[n];
    	for(int i=0; i<n; i++)
    	{
    		cin>>a[i];
    		dp[i]=0;
    	}
    	int j = n-1;
    	if(a[n-1]>=0)
    		dp[n-1]=a[n-1];
    	//cout<<dp[n-1]
    	for(int j=n-2;j>=0;--j)
    	{
    		if(j+k+1<n)
    			dp[j] = max((a[j]+dp[j+k+1]),dp[j+1]);
    		else
    			dp[j]=max(a[j],dp[j+1]);
    		//cout<<dp[j]<<" ";
    	}
    	cout<<dp[0]<<endl;
    	
    }
    return 0;
}
