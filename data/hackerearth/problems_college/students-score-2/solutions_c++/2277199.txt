#include<iostream>
using namespace std;
int main()
{
    int n,i,count=0,m=0,j;
    cin>>n;
    int A[n];
    for(i=0;i<n;i++)
    {
        cin>>A[i];
    }
    for(i=0;i<n;i++)
    {
        count=0;
        if(A[i]>=0)
        {
            for(j=i;j<n;j++)
            {
                count+=A[j];
                if(count>m)
                    m=count;

            }
        }
    }
    cout<<m<<endl;
}
