#include <iostream>
using namespace std;

int main()
{   
	int i,j,l,r,q,sum[10001]={0},a[10001]={0},n;
    cin>>n;
    for(i=1;i<n+1;i++)
        cin>>a[i];
    cin>>q;
    for(i=0;i<q;i++)
    {   cin>>l>>r;
    	for(j=l;j<=r;j++)
    	{
    		sum[i]=sum[i]+a[j];
    	}
    }
    for(i=0;i<q;i++)
       cout<<sum[i]<<endl;
    return 0;
}
