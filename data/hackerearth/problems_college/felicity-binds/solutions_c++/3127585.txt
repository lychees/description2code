#include <bits/stdc++.h>
using namespace std;

vector<long long>m, f;

int main()
{
    int t;
    cin >> t;
    while(t--){
    	int n;
    	cin >> n;
    	for(int i=0; i<n; ++i){
    		long long a;
    		cin >> a;
    		m.push_back(a);
    	}
    	for(int i=0; i<n; ++i){
    		long long a;
    		cin >> a;
    		f.push_back(a);
    	}
    	sort(m.begin(), m.end());
    	sort(f.begin(), f.end());
    	long long sum=0;
    	for(int i=m.size()-1; i>=0; --i){
    		sum += m[i]*f[i];
    	}
    	cout << sum << endl;
    	m.clear(); f.clear();
    }
    return 0;
}
