int max(int a,int b)
{
	if(a>b)
	return a;
	else
	return b;
}

dp[1]=arr[1];

 {
 	for(j=2*i;j<=n;j+=i)
 	{
 		dp[j]=max(dp[j],dp[i]+arr[j]);
 		
 		
 		
 	}
 }

    return 0;
}
