#include<iostream>
#include<math.h>
using namespace std;
long int fa(long int a)
{long int  c=0,i;
    for(i=1;i<sqrt(a);i++)
        if(a%i==0)c++;
    if(i==sqrt(a)) return ( 2*c+1 );
    else return ( 2*c );
}
main()
{
 long int  n,i,t,j;
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(i=0,j=1;fa(i)<=n;i+=j,j++);
        cout<<i<<endl;
    }


}
