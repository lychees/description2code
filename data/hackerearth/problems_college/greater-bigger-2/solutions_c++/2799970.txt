#include <iostream>
using namespace std;
#include<algorithm>
int main()
{
    int t,j;
    string s;
    cin>>t;
    while(t--)
    {
    	cin>>s;
    	int n=s.length();
    	int i=n-1;
    
    		while(s[i-1]>=s[i])
    		{ i--;
    	    if(i==0) break;
          	}
          	
    	i--;
    //	cout<<i;
    	if(i==-1)cout<<"no answer"<<endl;
    	else {
    		j=i;
    	while(s[j]>=s[i]){j++;
    	if(j==n)break;
    	}
        j--;
       // cout<<j;
    	swap(s[i],s[j]);
    	sort(s.begin()+i+1,s.end());
    	cout<<s<<endl;
    	
    	
    	}
    	
    	}
    
    
    return 0;
}
