#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t,n,temp;
	cin>>t;
	while(t--)
	{
		cin>>n;
		int a[10001];
		vector < pair<int,int> > vect;
		for(int i=0;i<n;i++)
		{
			cin>>temp;
			vect.push_back(make_pair(temp,i+1));
		}
		sort(vect.rbegin(),vect.rend());
		for(int i=1,j=1;i<=n;)
		{
			a[vect[i-1].second]=i;
			while(vect[j-1].first==vect[j].first&&j<=n)
			{
				a[vect[j].second]=i;
				j++;
			}
			j++;
			i=j;
		}
		for(int i=1;i<=n;i++)
			cout<<a[i]<<" ";
		cout<<endl;
	}
	return 0;
}
