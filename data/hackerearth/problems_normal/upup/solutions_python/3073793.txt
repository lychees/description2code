ans = ''
s = raw_input()
ls = 1
# len(s)
ans = ans+s[0].upper()
while ls < len(s):
	if s[ls] == ' ' and s[ls+1] != ' ':
		or_d = ord(s[ls+1])
		if or_d >= 97 and or_d <= 122:
			ans = ans + ' '+s[ls+1].upper()
		else:
			ans = ans+' '+s[ls+1]
		ls = ls+2
	else:
		ans = ans + s[ls]
		ls = ls + 1
print ans
