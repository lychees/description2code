#include <bits/stdc++.h>
using namespace std;
const int maxn=(int)(3*1e5+1e1);
int x[maxn],y[maxn],cnt1[maxn],cnt2[maxn],n;
 int main()
{  //int n;
	cin >> n;
	for(int i = 1; i <= n; ++i)
	{
		cin >> x[i] >> y[i];
		cnt1[x[i]]++;
		cnt2[y[i]]++;
	}
	for(int i = 1; i <= n; ++i)
	{
		int b = y[i];
		b = cnt1[b];
		int ans1 = n - 1, ans2 = n - 1;
		ans1 += b;
		ans2 -= b;
		cout << ans1 << " " << ans2 << endl;
	}
	return 0;
}
