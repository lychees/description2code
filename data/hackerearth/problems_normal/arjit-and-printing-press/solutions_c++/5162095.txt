#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long test,i,j,k,len_reserve,len_input;
	string input,reserve;
	cin >> test;
	while(test--)
	{
		cin >> input;
		cin >> reserve;
		k=0;
		len_input=input.size();len_reserve=reserve.size();
		long long arr[len_reserve];
		for(i=0;i<len_reserve;i++)
			arr[i]=int(reserve[i]);
		sort(arr,arr+len_reserve);
		for(i=0;i<len_input;i++)
		{
			for(j=k;j<len_reserve;j++)
			{
				if(arr[j]<int(input[i]))
				{
					input[i]=char(arr[j]);k++;break;
				}
			}
		}
		cout << input << endl;
	}
	return 0;
}
