#include <iostream>
#include<string>
using namespace std;

int main()
{
	int n,p;
    string s;
    cin >> n;
    for(int i=0;i<n;i++)
    {
    	if(i==0)
    	cin >> s;
    	else
    	{
    		string l,s1;
    		cin >> p >> s1;
    		for(int i=0;i<p;i++)
    		{
    			l=l+s.at(i);
    		}
    		l=l+s1;
    		s=l;
    	}
    }
    cout << s;
    return 0;
}
