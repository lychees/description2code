#include<bits/stdc++.h>
using namespace std;
int main()
{
    ios_base::sync_with_stdio(false);
    long long t, sum1, sum2, len, i;
    cin>>t;
    while(t--){
        string s;
        cin>>s;
        sum1=sum2=0;
        len=s.size();
        for(i=0;i<len/2;i++){
            sum1+=s[i]-'0';
        }
        for(;i<len;i++){
            sum2+=s[i]-'0';
        }
        if(sum1==sum2){
            cout<<"YES\n";
        }
        else
            cout<<"NO\n";
    }
    return 0;
}
