#include <bits/stdc++.h>
using namespace std;

int main()
{
	cin.sync_with_stdio(false);
	
    int t;
    cin>>t;
    
    while(t--){
    	int n; cin>>n;
    	vector<int> v(n);
    	
    	for(int i=0;i<n;i++) cin>>v[i];
    	sort(v.begin(),v.end());
    	
    	int l = v.size();
    	double val1, val2;
    	if(l%2==0){
    		val1 = (v[(l-1)/2] + v[(l-1)/2 + 1])/2;
    		val2 = val1+1;
    	}
    	else{
    		val1 = v[(l-1)/2];
    		val2 = val1 + 1;
    	}
    	
    	int64_t sum1=0, sum2 = 0;
    	
    	for(int i=0;i<l;i++){
    		sum1+=abs(val1  - v[i]);
    		sum2 += abs(val2 - v[i]);
    	}
    	
    	cout<<min(sum1,sum2)<<endl;
    }
    return 0;
}
