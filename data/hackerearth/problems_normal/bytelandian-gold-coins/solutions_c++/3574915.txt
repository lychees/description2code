#include <iostream>
#include<string>
#include<vector>
#include <map>
using namespace std;




map <int , long long> h;

long long f(int n) //precondition -> h.clear()
{
  if (n == 0) return 0; //base

  long long r = h[n];

  if (r == 0)
  {
    r =  n > f(n/2)+f(n/3)+f(n/4)?n:f(n/2)+f(n/3)+f(n/4) ;
    h[n] = r;
  }

  return r;
}
int main(){

long long int n;
while(cin >> n){



	cout<<f(n)<<endl;
}


}
