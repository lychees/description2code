#include <iostream>
using namespace std;

int main()
{
   int t,l,le,ay=0,sy=0;
   string str;
   char r;
   cin>>str;
   le=str.length();
   r=str[0];
   r=r+3;
   cin>>t;
   while(t--)
   {
   	string s;
   	cin>>s;
   	l=s.length();
   	if(le==l && s[0]==str[0])
   	sy++;
   	else if(le==l && s[0]==r)
   	ay++;
   	
   }
   cout<<sy<<" "<<ay;
    return 0;
}
