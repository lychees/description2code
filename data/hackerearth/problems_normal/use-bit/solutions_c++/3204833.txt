#include <iostream>
#include<string.h>
#include<stdio.h>
#include<algorithm>
using namespace std;

void swap(char *s,int i,int j){
    char temp=s[i];
    s[i]=s[j];
    s[j]=temp;
}

void reverse(char *s,int index,int n){
    int i,j;
    for(i=index+1,j=n-1;i<j;i++,j--){
        swap(s,i,j);
        
    }
}

int main() {
    char s[11],smallest;
    int i,flag=0,pos,index;
	scanf("%s",s);
	int n=strlen(s);
	sort(s,s+n);
	printf("%s\n",s);
	while(flag!=1){
	for(i=n-1;i>=0;i--){
	   if(i==0)
	    {
	        flag=1;
	        break;
	    }
	   
	   
	    if(s[i-1]<s[i])
	      break;
	    
	}
	if(flag!=1){
	   
	     index=i-1;
	     smallest=123;
	    for(i=index+1;i<n;i++){
	        if(s[i]>s[index]&&s[i]<smallest){
	            smallest=s[i];
	            pos=i;
	            
	            
	        }
	          
	    }
	    
	    
	    swap(s,index,pos);
	    
	    sort(s+index+1,s+n);
	    
	    printf("%s\n",s);
	    
	}
	
	
}
	return 0;
}
