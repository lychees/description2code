#include<stdio.h>
#include<iostream>
#include<vector>
#include<map>
#include<math.h>
#include<algorithm>
#include<string.h>
#define mod 1000000007
#define ll long long
using namespace std;
ll ar[100005],cum[100005];
int main()
{
	int t,n;
	ll sum;
	scanf("%d",&t);
	while(t--)
	{
		sum=0;
		scanf("%d",&n);
		for(int i=1;i<=n;i++)
		scanf("%lld",&ar[i]);
		sort(ar+1,ar+n+1);
		cum[0]=0;
		for(int i=1;i<=n;i++)
		{
			cum[i]=(cum[i-1]+ar[i])%mod;
		}
		for(ll i=n;i>=1;i--)
		{
			sum=((((ar[i]*(i-1))%mod)-cum[i-1]+mod)%mod+sum)%mod;
		}
		printf("%lld\n",(sum*ar[n])%mod);
	}
}
