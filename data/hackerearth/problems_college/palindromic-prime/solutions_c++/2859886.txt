#include <iostream>
using namespace std;
int palprime[115][2];
int prime[100000];
int p;
int pp;

int ispal(int n)
{
	int rev = 0, d, prod = 1, m = n;
	while(n > 0)
	{
		rev = rev * 10 + n % 10;
		n /= 10;
	}
	
	if(m != rev)
		return 0;
	else
	{
		while(m > 0)
		{
			if(m % 10 != 0)
				prod = prod * (m % 10);
			m /= 10;
		}
		
		return prod;
	}
	
}
void sieve()
{
	int a[100000];
	prime[p++] = 2;
	palprime[pp][1] = 2;
	palprime[pp++][0] = 2;
	
	
	
	for(int i = 3; i < 100000; i+=2)
	{
		if(a[i])
			continue;
		prime[p++] = i;
		
		int prod = ispal(i);
		if(prod != 0)
		{
			palprime[pp][1] = prod;
			palprime[pp++][0] = i;
			
			
		}		
		for(int j = i + i; j < 100000; j+=i)
			a[j] = 1;
	}
	
	/*for(int i = 0; i < 10; i++)
		cout << prime[i];*/
//	cout << endl << palprime[pp-1][0] << ' ' <<palprime[pp-1][1];
}

int factors(int n)
{
	int expr = 1, k = 0, ans = 1;
	while(n > 1)
	{
		expr = 1;
		while(n % prime[k] == 0)
		{	expr++;
			n /= prime[k];
		}
		ans *= expr;
		k++;
	}
	return ans;
}

int main()
{
	sieve();
	int t, n, m;
	scanf("%d", &t);
	while(t--)
	{
		scanf("%d%d", &n, &m);
		printf("%lld\n", factors(palprime[n-1][1] * m));
	}
	
    //cout << "Hello World!" << endl;
    return 0;
}
