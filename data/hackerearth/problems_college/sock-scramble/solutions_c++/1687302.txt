#include <iostream>
#include <algorithm>
using namespace std;

int main(){
	int t,n,i,j,missing,a[50];
	cin>>t;
	for(i=0;i<t;i++){
		cin>>n;
		for(j=0;j<n;j++)
			cin>>a[j];
		sort(a,a+n);
		for(j=0;j<n;){
			if(a[j]==a[j+1]){
				j+=2;
			}
			else{
				missing=a[j];
				break;
			}
		}
		cout<<missing<<"\n";
	}
	return 0;
}
