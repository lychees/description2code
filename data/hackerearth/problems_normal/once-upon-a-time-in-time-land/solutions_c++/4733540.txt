#include<iostream>
#include<bits/stdc++.h>

using namespace std;
int a[100007];
long long dp[100007];
int k;
long long solve(long long int i)
	{
		if(i<0)
			return 0;
		 if(dp[i]!=-1)
			return dp[i];
		
		long long int ans=max(solve(i-1),solve(i-k-1)+a[i]);
			dp[i]=ans;
			
		return dp[i];
	}
int main()
	{
		int t;
		cin>>t;
		int n;
		
		while(t--)
			{
				cin>>n>>k;
				for(int i=0;i<n;++i)
					scanf("%d",&a[i]);
				for(int i=0;i<=n;++i)
					dp[i]=-1;
				
				long long int ans=solve(n-1);
				cout<<ans<<endl;
			}
		return 0;
	}
