#include <iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int n,k;
		cin>>n>>k;
		long long int h[n];
		long long presum[n];
		for(int i=0;i<n;i++)
		{
			cin>>h[i];
			if(i>0)
			presum[i]=presum[i-1]+h[i];
			else presum[i]=h[i];
		}
		int prev_idx=-1;
		long long int sum=0;
		if(h[0]>k) prev_idx=0;
		for(int i=1;i<n;i++)
		{
			if(h[i]>k)
			{
				if(prev_idx==-1)
				{
					sum=presum[i-1];
				}
				else
				sum=max(sum,presum[i-1]-presum[prev_idx]);
				prev_idx=i;
			}
		}
		sum=max(sum,presum[n-1]-presum[prev_idx]);
		cout<<sum<<endl;
	}
    return 0;
}
