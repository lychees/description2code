#include <bits/stdc++.h>

using namespace std;

int main()
{
    long long t;
    scanf("%lld",&t);
    while(t--){
    string a;
    cin>>a;
    long long ans=0;
    for(int i=0;i<a.length();i++){
        if(a[i]!='D'){
            if(a[i]!='B'&&a[i+1]!='B'&&a[i+2]!='B'){
                if(a[i]!='S'&&a[i+1]!='S'&&a[i-1]!='S'){
                    ans++;
                    //cout<<i<<" ";
                }
            }
        }
    }
    printf("%lld\n",ans);
    }
    return 0;
}
