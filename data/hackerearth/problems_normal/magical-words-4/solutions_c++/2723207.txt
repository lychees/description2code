

#include<iostream>
#include<string>
#include<algorithm>

using namespace std;


int main(){

	long long int i,j,k,l,n,m,t,vl,fl;
	cin>>t;
	string s;
	while(t--){
    cin>>s;  vl=0;
    n = s.length();
    for(i=0;i<n;i++){
    	for(j=i,k=i;j>=0&&k<n;j--,k++){
    		if(s[j]==s[k])vl+=(k-j+1)*(k-j+1);
    		else break;
    	}
    }
    for(i=0;i<n-1;i++){
    	for(j=i,k=i+1;j>=0&&k<n;j--,k++){
    		if(s[j]==s[k])vl+=(k-j+1)*(k-j+1);
    		else break;
    	}
    }
    cout<<vl<<endl;
    }
    return 0;
}
