#include <bits/stdc++.h>
#define mod 1000000007
using namespace std;
unsigned long long int power(unsigned long long int x,unsigned long long int n)
{    if (n==0)
     return 1;
     if (n==1)
     return x%mod;
     if (n%2==0)
     return (power((x*x)%mod,n/2))%mod;
     if (n%2==1)
     return (x%mod*power((x*x)%mod,(n-1)/2))%mod;
}

		int n;
		int t,j;
