#include <iostream>
#include <algorithm>

using namespace std;

string str;

int main()
{
	ios::sync_with_stdio(false);
	int t;
	cin >> t;
	while(t--) {
		cin >> str;
		int l = str.length();
		int U = 0, S = 0;
		for(int i = 0;i < l;i++) {
			U += (str[i] == 'U');
			S += (str[i] == 'S');
		}
		if(abs(U-S) > 1) {
			cout << -1 << endl;
			continue;
		}
		if(U == S) {
			int ans1 = 0, ans2 = 0;
			for(int i = 0;i < l;i++) {
				ans1 += (str[i] != (i%2 ? 'U' : 'S'));
				ans2 += (str[i] != (i%2 ? 'S' : 'U'));
			}
			ans1 /= 2;
			ans2 /= 2;
			cout << min(ans1, ans2) << endl;
		}else {
			char ch = (U > S ? 'U' : 'S');
			char ch2 = (U < S ? 'U' : 'S');
			int ans = 0;
			for(int i = 0;i < l;i++) {
				ans += (str[i] != (i%2 ? ch2 : ch));
			}
			ans /= 2;
			cout << ans << endl;
		}
	}
    return 0;
}
