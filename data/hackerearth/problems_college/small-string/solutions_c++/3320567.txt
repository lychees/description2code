int test, n;
char a[1010], temp[1010], b[1010], res[1010];
 
int main(){
    int i, j, k;
 
    scanf("%d", &test);
    while (test--){
        scanf("%d %s", &n, a);
        strcpy(res, a);
 
        for (i = 0; i < n; i++){
            int d = 0;
            for (j = 0; j < n; j++){
                if (i != j) temp[d++] = a[j];
            }
            temp[d] = 0;
 
            for (j = 0; j <= d; j++){
                int c = 0;
                for (k = 0; k < j; k++) b[c++] = temp[k];
                b[c++] = a[i];
                for (k = j; k < d; k++) b[c++] = temp[k];
                 b[c] = 0;
 
                if (strcmp(b, res) < 0) strcpy(res, b);
            }
        }
 
        puts(res);
    }
    return 0;
}
