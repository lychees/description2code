#include<iostream>
#include<stdio.h>
using namespace std;
#define ll long long int 
ll  po(ll b,ll p,ll mod)   // compute r^(mod-2) 
{
ll half;
if(p==0)
return 1;
if(p==1)
return b;
half=po(b,p/2,mod);
if(p%2==0)
return ((half)%mod*(half)%mod)%mod;
else
return ((((half)%mod*(half)%mod)%mod)*b)%mod;
}




ll nCr(int a,int b,int n,int r,int mod)
{
int s,j=n;
s=r;
ll ans,x=(ll)1,y=(ll)1;
while(r>0)
{   
	y=(y*n)%mod;
	x=(x*r)%mod;
	n--;r--;
}
ans=(y*po(a,s,mod)%mod)%mod;
ans=(ans*po(b,j-s,mod)%mod)%mod;
ans=(ans*po(x,mod-2,mod)%mod)%mod;
return ans;
}


int main()
{
int n,r,mod=1000000007,a,b;
scanf("%d %d %d %d",&a,&b,&n,&r);
printf("%lld\n",nCr(a,b,n,r,mod));

return 0;
}
