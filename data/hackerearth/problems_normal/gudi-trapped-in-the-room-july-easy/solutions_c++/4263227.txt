#include <iostream>
#include <string>
#include <set>
#include <queue>

using namespace std;

int main()
{
	ios::sync_with_stdio(false);
	cin.tie(nullptr);
	cout.tie(nullptr);

	int T;
	cin >> T;
	while (T--)
	{
		string s;
		cin >> s;

		int A, H;
		cin >> A >> H;

		queue<string> q;
		set<string> possibleNumbers;
		q.push(s);
		while (!q.empty())
		{
			auto cur = q.front();
			q.pop();
			if (possibleNumbers.count(cur)) continue;
			possibleNumbers.insert(cur);

			auto tmp1 = cur;
			for (int i = 1; i < s.size(); i += 2) tmp1[i] = (A + tmp1[i] - '0') % 10 + '0';
			q.push(tmp1);

			auto tmp2 = cur;
			for (int i = 0; i < s.size(); i++) tmp2[(i + H) % s.size()] = cur[i];
			q.push(tmp2);
		}

		cout << *possibleNumbers.begin() << "\n";
	}
}
