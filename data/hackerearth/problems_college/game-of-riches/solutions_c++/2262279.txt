#include <iostream>

using namespace std;

int main()
{
	class game{
	public:
	int id;
	int wealth;
	int group;
	};
   int T;
   cin>>T;
   for(int i=0;i<T;i++)
   {
   	int nR,nG,Id=0;
   	cin>>nR>>nG;
   	game *n=new game[nR];
   	for(int j=0;j<nR;j++)
   	{
   		cin>>n[j].wealth;
   		n[j].id=Id++;
   	}
   	game temp;
   for(int j=0;j<nR-1;j++)
   {
   	for(int k=j;k<nR;k++)
   	{
   		if(n[j].wealth>n[k].wealth)
   		{
   			temp=n[k];
   			n[k]=n[j];
   			n[j]=temp;
   		}
   	}
   }
   Id=1;
   for(int j=0;j<nR;j++)
   {
   	n[j].group=(Id++)%nG;
   }
   int l;
   for(l=0;l<nR;l++)
   {
   	if(n[l].id==0)
   	{
   		break;
   	}
   }
   int count=0;
   for(int j=0;j<nR;j++)
   {
   	if(n[l].group==n[j].group&&n[l].wealth<n[j].wealth)
   	{
   		count++;
   	}
   }
   cout<<count<<endl;
   }
    return 0;
}
