#include <iostream>
#include <string.h>
#include <stdio.h>
using namespace std;

 
int main()
{
	int t;
	cin>>t;
	for(int m=0;m<t;m++)
	{
		int a[26]; 
		for(int j=0;j<27;j++) a[j] = 0;
	    char s[10000];
	    memset(s,'\0',10000);
		cin>>s;
		int len = 0;
		for(len = 0; s[len] != '\0' ; ++len)
		len = len+1;
		for(int i=0;i<len;i++)
		{
			if(int(s[i])>=97 && int(s[i])<= 122)
			{
				a[int(s[i])-96]++;
			}			
		}
		int cnt=0;
		for(int i=0;i<27;i++)
		{
			if(a[i] >= 1) cnt++;
		}
		if(cnt==26) cout<<"YES"<<endl;
		else cout<<"NO"<<endl;
	}
    return 0; 
}
