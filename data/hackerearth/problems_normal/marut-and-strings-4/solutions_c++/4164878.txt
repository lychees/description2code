#include<bits/stdc++.h>
using namespace std;
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    if(t > 10 || t < 1) puts("Invalid Test");
    else
    {
    	while(t--)
    	{
    		string str;
    		cin>>str;
    		if(str.size() > 100) puts("Invalid Input");
    		else
    		{
    			int low,up;
    			low = up =0;
    			for(int i=0;i<str.size();++i)
    			{
    				if(str[i] >= 65 && str[i] <= 90) up++;
    				else if(str[i] >= 97 && str[i] <=122) low++;
    			}
    			if(up == 0 && low == 0) puts("Invalid Input");
    			else
    			{
    				cout<<min(low,up)<<endl;
    			}
    		}
    	}
    }
    return 0;
}
