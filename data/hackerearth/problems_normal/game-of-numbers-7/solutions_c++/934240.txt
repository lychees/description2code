//Game of Numbers
#include<stdio.h>
#include<iostream>
#include<math.h>
using namespace std;

int arr[1000005] = {0};

int main()
{
	int test, start=0, end=0, i, index, temp, cnt, j, flag;
    scanf("%d", &test);
    while(test--)
    {
		for(i=0;i <= end-start;i++)
			arr[i] = 0;
    	scanf("%d %d", &start, &end);
    	for(i=2;i <= sqrt(end);i++)
    	{
			for(j=2, flag=0;j <= sqrt(i);j++)
			{
				if(i % j == 0)
				{
					flag = 1;
					break;
				}
			}
			if(flag == 0)
    		{
				temp = i*i;
				index = start/temp;
				if(index*temp < start)
					index++;
	    		while(index*temp <= end)
					arr[(index++*temp)-start] = 1;
			}
    	}
    	for(i=0, cnt=0;i <= (end-start);i++)
			if(arr[i] == 0)
				cnt++;
    	printf("%d\n", cnt);
    }
    return 0;
}
