#include<bits/stdc++.h>
using namespace std;
#define int long long
const int MAXN = 500;
long long dp[20][MAXN];
int ar[MAXN];
long long sum[MAXN];
const int INF=1e9+3;
#undef int
int main() {
	#define int long long
	int m;
	cin>>m;
	int p;
	cin>>p;
	sum[0]=0;
	for(int i=1;i<=p;i++) {
		scanf("%lld",&ar[i]);
		dp[0][i]=INF;
		sum[i]=sum[i-1]+ar[i];
		dp[1][i]=sum[i];
	}
	for(int i=0;i<=m;i++)
		dp[i][0]=0;
	for(int i=2;i<=m;i++) {
		for(int j=0;j<=p;j++) {
			dp[i][j]=dp[i-1][j];
			for(int k=j;k>=1;k--)
				dp[i][j]=min(dp[i][j],max(sum[j]-sum[k-1],dp[i-1][k-1]));
		}
	}
	cout<<dp[m][p]<<endl;
	return 0;
}
