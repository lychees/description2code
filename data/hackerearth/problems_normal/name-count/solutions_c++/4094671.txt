#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define mod 1000000007
ll fact[100004];
ll ifact[100004];
ll power(ll a,ll b)
{
    if(b==0)
        return 1;
    if(b==1)
        return a%mod;
        ll temp=power(a,b/2);
        if(b%2==0)
            return ((temp%mod)*(temp%mod))%mod;
        return ((((a%mod)*(temp%mod))%mod)*(temp%mod))%mod;


}
int main()
{
    int t;
    int n,k;
    cin>>t;
     fact[0]=1;
    int i;
    for(i=1;i<=100001;i++)
    {
        fact[i]=(i*(fact[i-1]%mod))%mod;
    }
    ifact[0]=1;
    for(i=1;i<=100001;i++)
    {
        ifact[i]=power(fact[i],mod-2)%mod;
    }
    while(t--)
    {
        cin>>n>>k;
        ll ans=(fact[k]%mod*ifact[k-n]%mod)%mod;
        cout<<ans<<endl;
    }




}
