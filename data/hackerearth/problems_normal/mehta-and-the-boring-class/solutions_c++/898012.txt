    #include <bits/stdc++.h>
    using namespace std;
    int dp[1000005];
  
    void pre()
    {
    dp[0] = dp[1] = 0;
    for ( int i = 2; i <= 1000000; i++ ) {
    if ( dp[i] > 0 ) continue;
    for ( int j = i; j <= 1000000; j += i ) {
    int k = j;
    while ( k%i == 0 ) dp[j]++, k /= i;
    }
    }
    for ( int i = 2; i <= 1000000; i++ ) dp[i] += dp[i-1];
    return;
    }
    int main()
    {
    pre();
    int t,a,b;
  //  fi(&t);
  scanf("%d",&t);
  //  assert(t <= 1000000);
    while ( t-- ) {
    	scanf("%d%d",&a,&b);
    //fi(&a), fi(&b);
   // assert(a <= b && a >= 1 && b <= 1000000);
    printf("%d\n", dp[b]-dp[a-1]);
    }
    return 0;
    }
