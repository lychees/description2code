#include <bits/stdc++.h>
using namespace std;
#define pii pair <int, int>
#define mp make_pair
#define pb push_back

bool cmp(const pii &a, const pii &b)
{
	if(a.second == b.second)
		return a.first < b.first;
	return a.second < b.second;
}

int main()
{
	ios_base::sync_with_stdio(0);
	cin.tie(0);
	cout.tie(0);
	int n;
	cin>>n;

	int a[111];
	for (int i = 1; i <= n; ++i)
		cin>>a[i];

	int sums[111];
	sums[0] = 0;
	for (int i = 1; i <= n; ++i)
		sums[i] = sums[i-1] + a[i];
	
	map <int, vector <pii> > m;
	for (int i = 1; i <= n; ++i)
		for (int j = i; j <= n; ++j)
			m[ sums[j] - sums[i-1] ].pb(mp(i,j));


	map <int, vector <int> > res;
	for (map<int, vector<pii> >::iterator i = m.begin(); i != m.end(); ++i)
	{
		vector< pii > &v = (i->second);
		if(v.size() == 0)
			continue;
		sort(v.begin(), v.end(), cmp);

		int c = 1;
		bool flg = 0;
		res[c].pb(i->first);
		int f = v[0].second;

		for (int j = 1; j < v.size(); ++j)
		{
			if(f < v[j].first)
			{
				++c;
				f = v[j].second;
				flg = 1;
			}
			else
				flg = 0;
			
			if(flg)
				res[c].pb(i->first);
		}
	}

	cout<<res.rbegin()->first<<endl;
	vector<int> &v = res.rbegin()->second;
	sort(v.begin(),v.end());
	for (vector<int>::iterator i = v.begin(); i != v.end(); ++i)
		cout<<*i<<" ";
	cout<<endl;
	return 0;
}
