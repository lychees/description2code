ll power(ll a,ll b)
{
	if(!b)
		return 1;
	if(b==1)
		return a;
	ll temp=power(a,b/2);
	temp=(temp*temp)%mod;
	if(b&1)
		temp=(temp*a)%mod;
	return temp;
}

int main()

	ll n, x;
	cin >> t;
	vl v(1000003);
	v[1]=1;
	for( i=2;i<=1000002;i++ )
	{
		v[i] =(i*v[i-1])%mod;
	}
	ll ans;
	while( t--)

		cin >> n >> x;
		ans = x;
		ans %= mod;
		if( n >=1000003)
		{
			ans = 0;
		}
		else
		{
			ans =(ans*v[n])%mod;
		}
		cout << ans << "\n";
	}
}
