#include<iostream>
#include<cstring>
#include<cstdio>
#include<cstdlib>
#include<math.h> 
using namespace std;
int main()
{
     int t,i,j,sv[10004];
     scanf("%d",&t);
     while (t--)
     {
     	int a,b;
     	scanf("%d%d",&a,&b);
     	for(i=0;i<=b-a+1;i++)
     		sv[i] = 0;
     	for (i=1;i*i<=b;i++)
     	{
     		int sq = i*i;
     		for (j = ( (a-1) / i + 1) * i; j <= b; j += i)
     		{
     			if (j<sq)
     				continue;
     			else if (j==sq)
     			{
     				sv[j-a]++;
     			}
     			else sv[j-a] += 2;
     		}
     	}

     	int mx,ans=0;
     	for(i=a;i<=b;i++)
     	{
     		if (i==a)
     			mx = sv[i-a];
     		else if (mx<sv[i-a])
     			mx = sv[i-a];
     	}
     	for(i=a;i<=b;i++)
     	if (sv[i-a]==mx) ans++;
     	cout<<ans+1<<endl;
     }
return 0;
}
