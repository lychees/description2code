#include <iostream>
using namespace std;
int main()
{
	int t,k,i;
	long n,m,a,b,x,y,tx,ty,jump;
    cin>>t;
    while(t--)
    {
    	x=1;y=1;
    	jump=0;
    	cin>>n>>m>>k;
    	for(i=0;i<k;i++)
    	{
    		cin>>a>>b;
    		tx=x+a;
    		ty=y+b;
    		while(tx>=1 && tx<=n && ty>=1 && ty<=m)
    		{
    			jump++;
    			x=tx;y=ty;
    			tx=x+a;
    			ty=y+b;
    		}
    	}
    	cout<<jump<<"\n";
    }
    return 0;
}
