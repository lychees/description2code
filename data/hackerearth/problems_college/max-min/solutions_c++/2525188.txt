#include <iostream>
#include<algorithm>
#include<climits>
using namespace std;
#define ll long long
int main()
{
    ll n,k;
    cin>>n>>k;
    ll a[n];
    for(ll i=0;i<n;i++)
    cin>>a[i];
    sort(a,a+n);
    ll m=INT_MAX;
    for(ll i=0;k-1+i<n;i++)
    m=min(m,a[k-1+i]-a[i]);
    cout<<m<<endl;
    return 0;
}
