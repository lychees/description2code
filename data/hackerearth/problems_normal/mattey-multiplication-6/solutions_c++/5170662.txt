#include<iostream>
using namespace std;
int main()
{
		long int T;
		cin>>T;
		for(long i=0; i<T; i++)
		{
			long int N,M;
			cin>>N;
			cin>>M;
			long int rem,l=0;
			long int Arr[256];
			while(M!=0)
			{
				rem=M%2;
				Arr[l]=rem;
				M=M/2;
				l++;
			}
			long int check=0,j;
			for( j=l-1; j>=0; j--)
			{
				if(Arr[j]==1 && check==0)
				{
					cout<<"("<<N<<"<<"<<j<<")";
					check++;
				}
				else if(Arr[j]==1 && check!=0)
				{
					cout<<" + "<<"("<<N<<"<<"<<j<<")";
				}
			}
			cout<<endl;
		}
	}
