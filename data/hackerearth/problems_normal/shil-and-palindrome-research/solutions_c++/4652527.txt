#include <iostream>
using namespace std;
string s;
void update(char c,int x)
{
	s[x-1]=c;
//	cout<<s<<endl;
}
int query(int l,int r)
{
	int a[256]={0},c=0;
	for(int i=l-1;i<r;i++)
	{
		a[(int)s[i]]++;
	}
	for(int i=0;i<256;i++)
	{
		if(a[i]%2)
		{
		c++;
	//	printf("%d %c\n",a[i],i);
		}
	  //  printf("%d %c %d\n",c,i,i);
		if(c>1)
		{
	//		cout<<"->"<<c<<endl;
		return 0;
		}
	}
	return 1;
}
int main()
{
    int n,q;
    cin>>n>>q;
    cin>>s;
    while(q--)
    {
    	int x;
    	cin>>x;
    	if(x==1)
    	{
    		int l;
    		char c;
    		cin>>l>>c;
    		update(c,l);
    	}
    	else
    	{
    		int l,r;
    		cin>>l>>r;
    		if(query(l,r))
    		cout<<"yes"<<endl;

    	}
    }

}
