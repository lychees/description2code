#include <bits/stdc++.h>
using namespace std;
#define ll long long
int solve(string& s,int n,int m)
{
	int res=0,i;
	ll ans=0;
	for(i=0;i<n;i++)
	{
		ans=(ans*10+s[i]-'0')%m;
	}
	res=ans;
	return ans;
}
int main()
{
    ios::sync_with_stdio(false);
    int i,n,t,m,ans;
    string s;
    cin>>t;
    while(t--)
    {
    	cin>>s>>m;
    	n=s.length();
    	ans=solve(s,n,m);
    	cout<<ans<<"\n";
    }
    return 0;
}
