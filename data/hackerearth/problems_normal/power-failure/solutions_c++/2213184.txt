#include <bits/stdc++.h>
#define _ ios_base::sync_with_stdio(false);cin.tie(0);
using namespace std;
#define pb push_back
#define pob pop_back
#define pf push_front
#define pof pop_front
#define mp make_pair
#define all(a) a.begin(),a.end()
#define bitcnt(x) __builtin_popcountll(x)
#define MOD 1000000007
#define MAXN 100005
#define eps  1e-9
typedef unsigned long long int uint64;
typedef long long int int64;


int prime[100005];
int primelt[100005];
vector<int>v;
int main(){
	int val,t,i,j,m,n;
	for(i=2;i<=100000;i++){
		if(prime[i])
		continue;
		for(j=2*i;j<=100000;j+=i)
		prime[j]=1;
	}
	for(i=2;i<=100000;i++)
	primelt[i]=primelt[i-1]+(prime[i]==0);
	cin>>t;
	while(t--){
		cin>>n>>m;
		v.clear();
		for(i=0;i<n;i++){
			cin>>val;
			v.pb(val);
		}
		sort(all(v));
		reverse(all(v));
		int64 ans=1;
		for(i=0;i<v.size();i++){
			int64 actv=m-v[i]+1;
			actv-=i;
			actv-=(primelt[m]-primelt[v[i]-1]);
			if(actv<=0)
			break;
			ans=ans*actv;
			if(ans>=MOD)
			ans%=MOD;
		}
		if(i!=v.size())
		cout<<0<<endl;
		else
		cout<<ans<<endl;
	}
	return 0;
}
