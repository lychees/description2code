#include<bits/stdc++.h>
using namespace std;
int dp[1009];
int rock_num[1009];
int main()
{
	int tc;
	cin>>tc;
	while(tc--)
	{
		int maxi=9999;
		int n;
		cin>>n;
		memset(rock_num,0,sizeof rock_num);
		memset(dp,0,sizeof dp);
		for(int i=1;i<=n;++i)
			cin>>rock_num[i];
		dp[1]=0;
		dp[2]=1;
		for(int i=3;i<=n;++i)
		{
			dp[i]=maxi;
			for(int j=1;j<i;++j)
			{
				if(rock_num[j]>=(i-j))
				{
					if(dp[j]+1<dp[i])
						dp[i]=dp[j]+1;
				}
			}
		}
		cout<<dp[n]<<endl;
	}
	return 0;
}
