#include <bits/stdc++.h>
using namespace std;

bool mycomp(const pair<int,char>&a,const pair<int,char>&b)
{
	if(a.first<b.first || (a.first==b.first && a.second>b.second))
	return true;
	return false;
}
int main()
{
    std::ios::sync_with_stdio(0);
    int t;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	int n=s.length();
    	vector<pair <int,char> >v1(26);
    	for(int i=0;i<26;i++)
    	{
    		v1[i].first=0;
    		v1[i].second=char(i+97);
    	}
    	for(int i=0;i<n;i++)
    	{
    		v1[s[i]-97].first++;
    	}
    	sort(v1.begin(),v1.end(),mycomp);
    	for(int i=0;i<26;i++)
    	cout<<v1[i].second<< " ";
    	cout<<endl;
    }
    
    return 0;
}
