#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    // sync_with_stdio(0);
    while(t--){
    	int n;
    	cin>>n;
    	long long arr[n];
    	long long p,l,c=0,d=0,res=0,len=0;
    	
    	for(long long i=0;i<n;i++)
    		cin>>arr[i];
    		
    	
    	
    	long long int pow_set_size = pow(2, n);
    	long long counter, j;
 
    	for(counter = 1; counter < pow_set_size; counter++)
		{
			bool first = true;
			l=0;
		  for(j = 0; j < n; j++)
		   {
		      if(counter & (1<<j))
		      {
		      	l++;
		      	if(first){
		      		c=arr[j];
		      		d=c;
		      		first=false;
		      	}else{
		      		c ^= arr[j];
		      		d &= arr[j];
		      	}
		      }
		        
		   }
		   p = c | d;
		   
		   
		   if(p >= res)
		   {
		   	if(p == res)
		   	len = max(len,l);
		   	else
		   	len = l;
		   	res = p;
		   	
		   	// cout<<'b';
		   }
		}
    	
    	cout<<res<<" "<<len<<endl;
    	res=0;len=0;
    	
    }
    return 0;
}
