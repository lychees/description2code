#include <iostream>
using namespace std;

int main()
{
    int t,n,d,i,check=0;
    char result[1000];
    scanf("%d",&t);
    for(i=0;i<t;i++)
    {
    	check=0;
    	scanf("%d %d",&n,&d);
    	if(n*2>=d)
    	{
    	result[i]='y';continue;
    	}
    	while(n>=3)
    	{
    		check=check+(3*2);
    		n=n-3+2;
    		if(n<3)
    		check=check+(n*2);
    	}
    	if(check>=d)
    	result[i]='y';
    	else
    	result[i]='n';
    }
    for(i=0;i<t;i++)
    {
    	if(result[i]=='y')
    	printf("Yes\n");
    	else
    	printf("No\n");
    }
    return 0;
}
