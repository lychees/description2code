#include<bits/stdc++.h>
using  namespace std;
int arr[1000001];
int main()
{
    int n1,n2;
    int counter=1;
    for(int i=1;i<=1000000;i++)
    {
        if(arr[i]==0)
        {
            arr[i]=i+counter;
            if(i+counter<=1000000)
                arr[i+counter]=i;
            counter++;
        }
    }
    int T=0;
    scanf("%d",&T);
    while(T--)
    {
        scanf("%d %d",&n1,&n2);
        if(n1>n2)
            swap(n1,n2);
        if(arr[n1]==n2)
        {
            printf("Don't Play\n");
        }
        else
            printf("Play\n");
    }
}
