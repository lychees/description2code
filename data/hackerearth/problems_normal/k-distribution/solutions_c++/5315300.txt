#include <bits/stdc++.h>
using namespace std;

int n, k;
long long dp[100][100001], a[100], z=0;

long long rec (int i, long long s1, long long s2)
{
	if(i==n)
		return !s1 && !s2;

	long long &ret=dp[i][max(s1,s2)];

	if(ret!=-1)
		return ret;

	return ret=(rec(i+1,max(s1-a[i],z),s2)+rec(i+1,s1,max(s2-a[i],z)))%1000000007;
}

int main()
{
	ios::sync_with_stdio(false);cin.tie(NULL);cout.tie(NULL);

	memset(dp,-1,sizeof dp);

	cin >> n >> k;

	for(int i=0 ; i<n ; i++)
		cin >> a[i];

	cout << rec(0,k,k) ;

	return 0;
}
