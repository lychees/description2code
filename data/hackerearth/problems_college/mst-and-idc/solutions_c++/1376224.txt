#include<bits/stdc++.h>
using namespace std;

int main(){
	long long int a[100001],t,n,m,i,ans,c,ans1,aa,b;
	cin>>t;
	while(t--){
		cin>>n;
		for(i=0;i<n;i++){
			cin>>a[i];
		}
		sort(a,a+n);
		if(n%2==0){
			aa = n/2;
			b = aa-1;
			ans = 2*((a[n-1]-a[aa])+(a[aa]-a[0]));
			
			ans1 = 2*((a[n-1]-a[b])+(a[b]-a[0]));
			if(ans1<ans){
				cout<<ans1<<endl;
			}
			else{
				cout<<ans<<endl;
			}
		}
		else{
			aa = n/2;
			ans = 2*((a[n-1]-a[aa])+(a[aa]-a[0]));
			cout<<ans<<endl;
		}
	}
	
	return 0;
}
