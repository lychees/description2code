#include <iostream>
using namespace std;
#define ll long long int
int main()
{
	ll t;
    cin>>t;
    while(t--)
    {
    	ll n;
    	cin>>n;
    	ll a[n];
    	for(ll i=0;i<n;i++)
    	cin>>a[i];
    	ll m1=a[0],m2=a[0];
    		for(ll i=0;i<n;i++)
    		{
        	if(a[i]<m1)
    	    m1=a[i];
    	    else if(m2<a[i])
    	    m2=a[i];
    	    }
    	ll d=(m2-m1)*2;
    	cout<<d<<endl;
    }
    return 0;
}
