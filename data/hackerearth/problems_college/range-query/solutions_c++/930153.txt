int BIT[110][20500], n;

inline void Update(int index, int pos, int val){
    while(pos <= n){
        BIT[index][pos] += val;
        pos += pos & -pos;
    }
    return;
}

inline int Read(int index, int pos){
    int ans = 0;
    while(pos > 0){
        ans += BIT[index][pos];
        pos -= pos & -pos;
    }
    return ans;
}

inline void Solve(){
    int i, x;
    memset(BIT, 0, sizeof(BIT));
    sd(n);
    for(i = 1; i <= n; i++){
        sd(x);
        Update(x, i, 1);
    }
    int q, y, val, ans = 0;
    sd(q);
    while(q--){
        sd(x); sd(y); sd(val);
        x++; y++;
        ans = Read(val, y) - Read(val, x - 1);
        printf("%d\n", ans);
    }
    return;
}

int main(){
    //freopen("input.txt", "r", stdin);
    int t = 1;
    //sd(t);
    while(t--){
        Solve();
    }
    return 0;
}
