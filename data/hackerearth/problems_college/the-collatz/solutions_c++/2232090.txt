#include <iostream>
using namespace std;

int main()
{
    int t;
    long long n;
    cin>>t;
    while(t--){
    	long long a=0,b=1;
    	cin>>n;
    	long long stp=0;
    	while(n!=1){
    		stp++;
    		if(n%2==0){
    			a=(a+n);
    			n/=2;
    		}else {
    			b=(b+n);
    			n=3*n+1;
    		}
    	}
    	a=a%stp;
    	b=b%stp;
    	if(a==b) cout<<"Tie"<<endl;
    	else if(a<b) cout<<"Odd Rules"<<endl;
    	else cout<<"Even Rules"<<endl;
    }
    return 0;
}
