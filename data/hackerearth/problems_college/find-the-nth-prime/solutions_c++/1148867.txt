#include <iostream>
#include <string.h>
using namespace std;

 
void nthPrime(int n)
{
int count = 0;
if(n==1)
{
	cout<<"2"<<endl;
}
else
{
count++;
int i;
bool prime=false;
for ( i=3; count<n;i++)
{
	prime=false;
	int k=2;
	int j=i;
	while(((j%k)!=0)&&(j>k))
	{
		if((j<k)&&(prime==false))
		{
			count++;
		}
		k++;
	}
	if(j==k)
	{
		count++;
	}
}
cout<<i-1<<endl;
}

}
 
// Driver Program to test above function
int main()
{
    int n;
cin>>n;
    nthPrime(n);

    return 0;
}
