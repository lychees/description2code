#include <iostream>
#include<algorithm>
#include<vector>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	vector<unsigned long long int>A(n);
    	unsigned long long ans=0;
    	for(int i=0; i<n; i++)
    	{
    		cin>>A[i];
    	}
    	
    	sort(A.begin() , A.end());
    	
    	for(int i=A.size()-1; i>=1; i=i-2)
    	{
    		ans+=(A[i]*A[i-1]);
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
