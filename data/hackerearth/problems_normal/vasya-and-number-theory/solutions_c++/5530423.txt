# include <bits/stdc++.h>
using namespace std;
int a[1000];
int dp[1000];
int main(){
	int n;
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>a[i];
	}
	sort(a,a+n);
	for(int i=0;i<n;i++){
		dp[i]=1;
	}
	for(int i=1;i<n;i++){
		int temp=INT_MIN;
		for(int j=i-1;j>=0;j--){
			if(a[i]%a[j]==0){
				temp=max(temp,1+dp[j]);
			}
		}
		dp[i]=max(dp[i],temp);
	}
	int ans=-1;
	for(int i=0;i<n;i++){
		ans=max(ans,dp[i]);
	}
	if(ans==1){
		cout<<"-1"<<endl;
	}
	else{
		cout<<ans<<endl;
	}
}
