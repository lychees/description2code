#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n,i;
	string str,str2,str3;
	cin >> n;
	vector <string> s;
	for(i=0;i<n;i++){
		cin >> str;
		s.push_back(str);
	}
	int j,l=0;
	char c=' ';
	for(i=0;i<n;i++){
		str = s[i];
		str2= string(str.rbegin(),str.rend());
		//cout << str2 << endl;
		for(j=0;j<n;j++){
			str3=s[j];
			if(str3==str2){
				l=str2.size();
				c=str2[l/2];
			}
		}
	}
	cout << l << " " << c;
    return 0;
}
