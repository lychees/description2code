#include<cstdio>
#include<cmath>
#include<iostream>
#include<cstring>
using namespace std;
long long int coin(long long int a[],long long int m,long long int n)
{
long long 	int tbl[n+1],i,j;
	memset(tbl,0,sizeof(tbl));
	tbl[0]=1;
	for(i=0;i<m;i++)
	{
		for(j = a[i];j<=n;j++)
		{
			tbl[j]=tbl[j] + tbl[j-a[i]];
		}
	}
	return tbl[n];
}
int main()
{
	long long int n,i;
	cin>>n;
	long long int a[4];
	for(i=0;i<4;i++)
	{
		cin>>a[i];
	}
	long long int p = coin(a,4,n);
	cout<<p<<"\n";
}
