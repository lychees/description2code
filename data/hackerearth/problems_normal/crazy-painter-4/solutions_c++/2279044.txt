#include <iostream>
#include <algorithm>
#include <vector>
#include <stack>
#include <queue>
#include <cstring>
#include <stdio.h>
using namespace std;

int main(int argc, char const *argv[])
{
	long costOfLetter[26]={0};
	long testCases = 0;
	long sizePainting;

	long totalCost = 0;

	scanf("%lld",&testCases);

	while (testCases>0)
	{
		totalCost = 0;
		scanf("%lld",&sizePainting);

		for (int i = 0; i < 26; ++i)
			scanf("%lld",&costOfLetter[i]);
		
		long totalPoints = 12*sizePainting;

		long turns = totalPoints/26;
		int extraturns = totalPoints%26;

		long addFactor = ((turns - 1)*(turns))/2;

		for (int i = 0; i < 26; ++i)
			totalCost+=costOfLetter[i]*turns + addFactor;

		for (int i = 0; i < extraturns; ++i)
			totalCost+=costOfLetter[i] + turns;

		printf("%lld\n", totalCost);

		testCases--;
	}

	return 0;
}
