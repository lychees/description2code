#include<iostream>
#include<math.h>
#include<stdlib.h>
#include<algorithm>
#include<cstring>
#define ll long long
#define MOD 1000000007
using namespace std;
void myfunc()
{
	string a,b;
	cin>>a>>b;
	ll meriarraykamukablanhi[26]={0},doosriarray[26]={0};
	ll finalanswer=1;
	ll akilength=a.length()-1;
	ll bkilength=b.length()-1;
	for(ll i=bkilength;i>=0;i--)
	doosriarray[b[i]-'a']++;
	for(ll i=akilength;i>=0;i--)
	meriarraykamukablanhi[a[i]-'a']++;
	for(ll i=0;i<26;i++)
	{
		if(meriarraykamukablanhi[i]-doosriarray[i]>=0)
			for(ll j=meriarraykamukablanhi[i];doosriarray[i]>0;j--,doosriarray[i]--)
				finalanswer=(finalanswer*j)%MOD;
		else if(meriarraykamukablanhi[i]-doosriarray[i]<0)
		{
			cout<<0;
			return ;
		}
	}
	cout<<finalanswer;
}
int main()
{
	ll n;
	cin>>n;
	while(n--)
	{
		myfunc();
		cout<<"\n";
	}
	return 0;
}
