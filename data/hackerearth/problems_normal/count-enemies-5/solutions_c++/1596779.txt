#include <iostream>
#include<string.h> 
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int x=0,o=0,ans=0;
    	char s[100001];
    	cin>>s;
    	for(int i=0;i<strlen(s);i++)
    	{
    		if(s[i]=='X')
    		x++;
    		else if(s[i]=='O')
    		o++;
    		else if(s[i]=='*')
    		{
    			if(x==0)
    			ans+=o;
    			o=0;
    			x=0;
    		}
    	}
    	if(x==0)
    	ans+=o;
    	cout<<ans<<endl;
    }
    return 0;
}
