#include<bits/stdc++.h>

using namespace std;
vector<int>v[26];
int main()
{
    string s;
    cin >> s;
    //int n;
    //cin >> n;
    int q;
    cin >> q;
    int i ;
    for(i = 0 ; i < s.length(); i++)
    {
        v[s[i]-'a'].push_back(i + 1);
    }

    for(i = 0; i < 26; i++)
        v[i].push_back(100000000);

    string f;
    int x = -1;
    int low = -1;
    int flag = 0;
    while(q--)
    {
        flag = 0;
        cin >> f;
        int idx = 0;

        for(i = 0 ; i < f.length() ; i++)
        {
            low = lower_bound(v[f[i] - 'a'].begin(),v[f[i] - 'a'].end(),idx) - v[f[i] - 'a'].begin();
            idx = v[f[i] - 'a'][low];
            //cout<<idx<<endl;
            if(idx == 100000000)
            {
                flag = 1;
                break;
            }
            //cout<<low<<endl;

        }
        if(flag == 1)
        {
            cout<<"No\n";
        }
        else
            cout<<"Yes\n";
    }
    return 0;
}
