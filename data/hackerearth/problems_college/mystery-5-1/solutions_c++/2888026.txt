#include <bits/stdc++.h>
using namespace std;
 
int main(){
    int tc;
    cin>>tc;
    while(tc--){
		int a[26]={0};
		int n;
		string str;
		cin>>str;
		n=str.size();
		
		for(int i=0;i<n;i++)
			a[str[i]-'a']+=1;
		int max = -1;
		
		for(int i=0;i<26;i++)
			if(a[i]>max)
				max=a[i];
    	
    	cout << max << endl;
    }
    
}
