#include <iostream>
#include <vector>
#include <cstdio>
using namespace std;
bool check[100001];
void dfs(int s, vector <int> adj[]){
    check[s] = true;
    for(int i = 0; i<adj[s].size(); i++){
        if(check[adj[s][i]] == false){
            dfs(adj[s][i], adj);
        }
    }
}
int main(){
    int tc;
    scanf("%d", &tc);
    while(tc-->0){
        int n, e;
        scanf("%d%d", &n, &e);
        for(int i = 0; i<n; i++){
            check[i] = false;
        }
        vector <int> adj[n];
        int a, b, c=0;
        for(int i = 0; i<e; i++){
            scanf("%d%d", &a, &b);
            adj[a].push_back(b);
            adj[b].push_back(a);
        }
        for(int i = 0; i<n; i++){
            if(check[i]==false){
                c++;
                dfs(i, adj);
            }
        }
        printf("%d\n", c);
    }
    return 0;
}
