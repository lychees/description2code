#include <iostream>
using namespace std;

int main()
{
    string s;
    bool comment;
    while (getline(cin, s)) {
    	int i = 0;
    	s = s + 'q';
    	while (i < s.length()) {
    		if (s[i] == '/' && s[i+1] == '/')
    			break;
    			
    		if (s[i] == '-' && s[i+1] == '>') {
    				cout << '.';
    				i += 2;
    		}
    		else {
    				if (i < s.length()-1)
    					cout << s[i];
    				i++;
    		}
    	}
    	
    	for(; i < s.length()-1; i++)
    		cout << s[i];
    		
    	cout << endl;	
    }
    return 0;
}
