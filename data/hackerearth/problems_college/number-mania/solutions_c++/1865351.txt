int length= 0;
int fact[10000];
void yo(int num)
{
    long i,pr=0,r=0;
    for(i=0;i<=length;i++)
    {
        r = (fact[i]*num + pr)/10;
        fact[i] = (fact[i]*num + pr)%10;
        pr=r;
    }
    if(pr!=0)
    {
         while(r!=0)
         {
             fact[i]=r%10;
             r= r/10;
             i++;
         }
    }
    length = i-1;
}

void hybrid(int num)
{
    int i;
    for(i=2;i<=num;i++)
    {
         yo(i);
    }
}

{

        }
    }
}
