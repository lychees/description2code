import math
t=input()
while(t>0):
	t=t-1
	m,n=map(int,raw_input().split())
	if n <= m+1:
		gls=math.factorial(m)
		bys=gls*(m+1)
		print ((bys*gls)/math.factorial(m+1-n))%1000000007
	else:
		print -1
