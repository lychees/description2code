#include <bits/stdc++.h>

using namespace std;

char a [20 + 1][30 + 1];

int main()
{
    int t , n , m , ans = 0;

    cin >> t;

    while(t--){

        ans = 0;
        cin >> n >> m;

        for(int i = 0; i < n; i++){
            for(int j = 0; j< m; j++){

                cin >> a[i][j];
            }
        }

        for(int i = n - 1; i >= 0; i--){
            for(int j = m - 1; j >= 0; j--){

                if(a[i][j] == '0'){                   

                    ans++;                              

                    for(int k = 0; k <= i; k++){        
                        for(int h = 0; h <= j; h++){

                            if(a[k][h] == '1') a[k][h] = '0';       
                            else a[k][h] = '1';
                        }
                    }
                }
            }
        }

        cout << ans << "\n";
    }

    return 0;
}
