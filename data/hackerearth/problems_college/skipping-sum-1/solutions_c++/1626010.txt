#include <bits/stdc++.h>
using namespace std;
long long int dp[11][100010];
int a[100000];
int main()
{
    int n,q,i,j,l,r,k;
    cin>>n>>q;
    for(i=1;i<=n;i++)
    {
        cin>>a[i];
    }
    for(i=1;i<=10;i++)
    {
        for(j=1;j<=i;j++)
        {
            dp[i][j]=a[j];
        }
        for(j=i+1;j<=n;j++)
        {
            dp[i][j]=dp[i][j-i]+a[j];
        }
    }
    while(q--)
    {
        cin>>l>>r>>k;
        cout<<dp[k][r-((r-l)%k)]-dp[k][l]+a[l]<<endl;
    }
}
