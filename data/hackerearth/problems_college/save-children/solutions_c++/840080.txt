#include<bits/stdc++.h>

using namespace std;

long long ch[100001],tree[100001],diff[100001];
int main()
{
int t;

cin>>t;

while(t--)
{long long i,j,n,ans1,ans2;

cin>>n;

for(i=0;i<n;i++)
    cin>>ch[i];

for(i=0;i<n;i++)
    cin>>tree[i];

sort(ch,ch+n);
sort(tree,tree+n);

if(n%2==0)
{
 ans2=min(ch[n/2],ch[n/2-1]);
}
else
    ans2=ch[n/2];

    for(i=0;i<n;i++)
        diff[i]=abs(tree[i]-ch[i]);

    ans1=INT_MIN;

    for(i=0;i<n;i++)
        if(ans1<diff[i])
        ans1=diff[i];

    cout<<ans1<<endl;
    cout<<ans2<<endl;




}
}
