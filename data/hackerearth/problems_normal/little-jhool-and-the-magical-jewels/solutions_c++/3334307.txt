#include <iostream>
#include<string>
#include<algorithm>
using namespace std;

int main()
{
    string str;
    int arr[4];
    int t;
    cin>>t;
    while(t--)
    {
    	cin>>str;
    	for(int i=0;i<4;i++)arr[i]=0;
    	for(int i=0;i<str.length();i++)
    	{
    		if(str[i]=='r')arr[0]++;
    		if(str[i]=='u')arr[1]++;
    		if(str[i]=='b')arr[2]++;
    		if(str[i]=='y')arr[3]++;
    	}
    	cout<<*min_element(arr,arr+4)<<endl;
    }
    return 0;
}
