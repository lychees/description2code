#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
#define F(i,a,b) for(int i = (int)(a); i <= (int)(b); i++)
#define RF(i,a,b) for(int i = (int)(a); i >= (int)(b); i--)
#define MOD 1000000007
ll A[1000005];
ll B[1000005];
int main()
{
    #ifndef ONLINE_JUDGE
        freopen("input.txt","r",stdin);
        freopen("output.txt","w",stdout);
    #endif
    int N,M,_i,_j;
    ll K;
    scanf("%d",&N);
    F(i,1,N) scanf("%lld",A+i);
    scanf("%d",&M);
    F(i,1,M)
    {
        scanf("%d%d%lld",&_i,&_j,&K);
        B[_i] += K;
        B[_j+1] -= K;
    }
    F(i,1,N) B[i] += B[i-1];
    F(i,1,N) A[i] += B[i];
    F(i,1,N) printf("%lld\n",A[i]);
    return 0;
}
