#include <bits/stdc++.h>
using namespace std;
vector < bool > v(1002,false);
void pre(){
	v[0]=v[1]=true;
	for(int i=2;i*i<=1000;i++){
		if(v[i]==false){
			for(int j=2*i;j<=1000;j+=i)
			  v[j]=true;
		}
	}
}
int main()
{
    int t;
    cin >> t;
    pre();
    while(t--){
    	int n,ans=0;
    	cin >> n;
    	vector < int > nums(n);
    	for(int i=0;i<n;i++){
    	 cin >> nums[i];
    	 if(v[nums[i]]==false)
    	  ans++;
    	}
    	cout << ans <<endl;
    }
    return 0;
}
