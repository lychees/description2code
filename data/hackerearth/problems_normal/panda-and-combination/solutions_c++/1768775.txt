#include <iostream>
#include <math.h>
using namespace std;

int main()
{
    long long int t,n,m,count,sum,x;
    cin>>t;
    
    while(t--)
    {
    	cin>>n>>m;
    	count =0;
    	
    	if(m==1)
    	{
    		cout<<n<<endl;
    		continue;
    	}
    	
    	while(n!=0)
    	{
    		if(n<m)
    		{
    			count = count +n;
    			break;
    		}
    		x = log(n)/log(m);
    		n = n- pow(m,x);
    		count++;
    	}
    	
    	cout<<count<<endl;;
    	
    }
    
    
    return 0;
}
