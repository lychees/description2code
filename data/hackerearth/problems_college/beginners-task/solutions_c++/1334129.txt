#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	int n;
    	cin >> n;
    	int count = 0, a;
    	while(n--)
    	{
    		cin >> a;
    		if(a % 2 == 0)
    		{
    			count ++;
    		}
    	}
    	cout << count << endl;
    }
    return 0;
}
