#include <iostream>
using namespace std;
int main()
{
	bool RED=0,GRN=0,BLU=0;
	long t,r,g,b,i;
	int R=0,G=0,Blu=0,Y=0,C=0,M=0,W=0,B=1,n;
    cin>>t>>r>>g>>b;
    for(i=1;i<t;i++)
    {
    	if(i%r==0)RED=!RED;
    	if(i%g==0)GRN=!GRN;
    	if(i%b==0)BLU=!BLU;
    	n=(BLU)+(GRN*10)+(RED*100);
    	switch(n)
    	{
    		case 0:B++;break;
    		case 100:R++;break;
    		case 10:G++;break;
    		case 1:Blu++;break;
    		case 110:Y++;break;
    		case 11:C++;break;
    		case 101:M++;break;
    		case 111:W++;
    	}
    }
    cout<<R<<" "<<G<<" "<<Blu<<" "<<Y<<" "<<C<<" "<<M<<" "<<W<<" "<<B;
    return 0;
}
