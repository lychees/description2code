#include <iostream>
using namespace std;
 
int main()
{
	int t;
	long long a[101];
	a[0]=0;
	a[1]=1;
	for(int i=2;i<101;i++)
		a[i]=a[i-1]+a[i-2];
	scanf("%d",&t);
	while(t-->0)
	{
		int n;
		scanf("%d",&n);
		for(int i=n-1;i>=0;i--)
			printf("%lld ",a[i] );
		printf("\n");
	}
	return 0;
}
