#include<bits/stdc++.h>
using namespace std;
#define MOD 1000000007
int main() {
	int tc;
	cin >> tc;
	while(tc--) {
		string s;
		cin >> s;
		int n=s.size(),ans=0;
		for(int i=0;i<n;i++) {
			for(int j=i;j<n;j++) {
				int st=i,en=j,flag=0;
				while(st<=j) {
					if(s[st]!=s[en]) {
						flag=1;
						break;
					}
					st++;en--;
				}
				if(flag==0)
					ans=max(ans,j-i+1);
			}
		}
		cout << ans << endl;
	}
	return 0;
}
