#include <iostream>
using namespace std;

int main()
{
	int n,q,i,l,r;
    cin>>n;
    int a[n];
    cin>>q;
    for(i=0;i<n;i++)
    cin>>a[i];
    
    for(i=0;i<q;i++)
    {
    cin>>l;
    l--;
    cin>>r;
    r--;
    int sum=a[r],dif=a[r];
    
    for(int j=l;j<r;j++)
    {
    sum=sum+a[j];
    dif=dif-a[j];
    }
    
    cout<<sum<<" "<<dif<<"\n";
    }
    return 0;
}
