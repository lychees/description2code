bool prime(lli n)
{
	if(n%2==0)
	return false;	
	 
	for(lli i=3;i<=sqrt(n);i+=2)
	{
		if((n%i)==0)
			return false;
	}
	return true;
}
int main() {
	li n;
	li a[100001];
	cin>>n;
	for(li i=0;i<n;i++)
		cin>>a[i];
		lli max=0;
	for(li i=0;i<n;i++)
	{
		if(prime(a[i]))
		{
			if(a[i]>max)
			max=a[i];
		}
		
	}
	li counter=0;
	for(li i=0;i<n;i++)
		if(a[i]==max)
		counter++;
		
	cout<<max<<endl<<counter;
	return 0;
}
