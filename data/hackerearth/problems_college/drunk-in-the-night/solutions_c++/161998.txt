vi *v;
int E,W,N,S,a[30][30];

double dfs(int x,int y,int n)
{
	if(!n)		return 1.0;
	a[x][y]=1;
	double j=0;
	if(!a[x][y+1])		j+=(E/100.0  *  dfs(x,y+1,n-1));		
	if(!a[x][y-1])		j+=(W/100.0  *  dfs(x,y-1,n-1));
	if(!a[x-1][y])		j+=(N/100.0  *  dfs(x-1,y,n-1));
	if(!a[x+1][y])		j+=(S/100.0  *  dfs(x+1,y,n-1));
	a[x][y]=0;		
	return j;
}

int main()
{
	int t,i,j,k,n,m;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d%d%d%d",&n,&E,&W,&N,&S);
		for(i=0;i<30;i++)	for(j=0;j<30;j++)
			a[i][j]=0;
		printf("%lf\n",dfs(15,15,n));
	}
	return 0;
}
