#include<iostream>
using namespace std;
int main(){
    int t;
    cin>>t;
    while(t--){
        string s;
        cin>>s;
        int az=0,sum=0;
        int a[26];
        for(int i=0;i<26;i++)
            a[i]=0;

        for(int i=0;i<s.length();i++){
            az=s[i];
            a[az-97]=1;
        }
        for(int i=0;i<26;i++){
            sum=sum+a[i];
        }
        if(sum==26)
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
    }


return 0;
}
