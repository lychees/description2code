#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define mod 1000000007
ll b[105];
ll g[105];
int main()
{
	std::ios_base::sync_with_stdio(false);
	ll t;
	cin>>t;
	while(t--)
	{
		ll n, k, m;
		cin>>n>>k>>m;
		vector<string> v;
		vector<pair<string, ll> > temp_v;
		for(int i=0;i<n;i++)
		{
			string temp;
			cin>>temp;
			v.push_back(temp);

			ll sz = temp.size();
			sz = min(sz, m);

			temp_v.push_back(make_pair(temp.substr(0, sz), i));
			//cout<<temp.substr(0, sz)<<"\n";
		}
		sort(temp_v.begin(), temp_v.end());
		ll ans = temp_v[k-1].second;
		cout<<v[ans]<<"\n";
	}
	return 0;
}
