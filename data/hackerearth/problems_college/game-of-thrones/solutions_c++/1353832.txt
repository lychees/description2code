#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */   
    int n;
    int t;
    cin>>t;
    long long int x;
    
    while(t--)
    {
    	cin>>n;
    	int i;
    	vector<long long int> v;
    	for(i=0;i<n;i++)
    	{
    		cin>>x;
    		v.push_back(x);
    	}
    	sort(v.begin(),v.end());
    	int count=0;
    	for(i=0;i<n-1;i++)
    	{
    		if(v[i]==v[i+1])
    		count++;
    		//cout<<v[i]<<" ";
    	}
    	cout<<count<<"\n";	
    	
    	
    	
    }
    
    return 0;
}
