#include<stdio.h>
using namespace std;

int coolNess(int k)
{
 int f = 5,p=0,c=0,t=2;
 
 while(f<=k)
 {
  if(5==((k&f)>>p)  &&  0==((k&t)>>p))
  {
   c++;
  }
  f=f<<1;
  t=t<<1;
  p++;
 };
 return c;
}
int main()
{
 int nT,nC,nN,nR,i=0,k;  
 scanf("%d",&nT);
 while(nT--)
 {
  scanf("%d%d",&nN,&nR);
  k=5;
  nC=0;
  while(k<=nN)
  { 
    if(nR<=coolNess(k++))
    {
     nC++;
    }
  } 
  printf("%d\n",nC);
 }    

 return 0;
}
