#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
    int t,p;
    long long int a,max;
    cin>>p;
    while(p>0){
    cin>>t;max=-1;
    vector<long long int>vec(1000000);
    for(int i=0;i<=1000000;i++)
    vec[i]=0;
    vector<long long int>vec1;
    for(int i=0;i<t;i++)
    {
    	cin>>a;
    	vec[a]++;
    	if(vec[a]>max)
    	{
    		vec1.clear();
    		vec1.push_back(a);
    		max=vec[a];
    	}
    	else if(vec[a]==max)
    	vec1.push_back(a);
    }
    sort(vec1.begin(),vec1.end());
    for(int i=vec1.size()-1;i>=0;i--)
    cout<<vec1[i]<<" ";
    cout<<endl;
    	p--;
    }
    return 0;
}
