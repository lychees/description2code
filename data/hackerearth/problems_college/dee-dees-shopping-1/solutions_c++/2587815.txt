typedef long long int ll;

int n , t;
int main(){
//    ios_base::sync_with_stdio(0);

    cin >> t;
    while(t--){
        cin >> n;
        int ans = mx;
        for(int i = 0; i * 8 <= n; i ++){
            if((n - i * 8) % 6 == 0){
                ans = min(ans, (n - i * 8) / 6 + i);
            }
        }
        for(int i = 0; i * 6 <= n; i ++){
            if((n - i * 6) % 8 == 0){
                ans = min(ans, (n - i * 6) / 8 + i);
            }
        }
        ans = ans == mx ? -1 : ans;
        cout << ans << endl;
    }
    return 0;
}
