#include<iostream>
#include<string>

int main()
{

string sa,sb;

    {
    int flag=0;

   int map[26]={0};
    for(i=0;sa[i];i++)
        {
        if(!isalpha(sa[i]))
        continue;
        sa[i]=toupper(sa[i]);
        map[sa[i]-65]++;
        }
    for(i=0;sb[i];i++)
        {
        if(!isalpha(sb[i]))
        continue;
        sb[i]=toupper(sb[i]);

        }

    char cent;
    int pali=0;
    string a;
    for(i=0;i<26;i++)
        {
        if(map[i]>0)
            {
            if(flag==0 || flag==1)
                {
                flag=1;
                if(map[i]%2==0)
                    {
                    for(j=0;j<map[i]/2;j++)
                    a[pali++]=i+'a';
                    }
                else if(cflag==0)
                    {
                    cflag=1;
                    cent=i+'a'; 
                    for(j=0;j<(map[i]-1)/2;j++)
                    a[pali++]=i+'a';  
                    }
                else
                    {
                    flag=3;    
                    break;
                    }
                }
            else
                {
                flag=3;    
                break;   
                }
            }
        else if(map[i]<0)
            {
            if(flag==0 || flag==2)
                {
                flag=2;
                map[i]*=-1;
                if(map[i]%2==0)
                    {
                    for(j=0;j<map[i]/2;j++)
                    a[pali++]=i+'a';
                    }
                else if(cflag==0)
                    {
                    cflag=1;
                    cent=i+'a';   
                    for(j=0;j<(map[i]-1)/2;j++)
                    a[pali++]=i+'a';
                    }
                else
                    {
                    flag=3;    
                    break;
                    }
                }
            else
                {
                flag=3;    
                break;   
                }
            }
        }
    if(flag==0)
        {
        cout<<"YES"<<endl; 
        }
    else if(flag==2 || flag==1)
        {
        for(i=0;i<pali;i++)
            {
            cout<<a[i];   
            }    
        if(cflag==1)
            {
            cout<<cent;
            }
        for(i--;i>=0;i--)
            {
            cout<<a[i];
