#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
bool cmp(pair<ll,ll> a, pair<ll,ll> b){
	
}
ll arr[1000005];
int main(){
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		ll a;
		memset(arr,0,sizeof(arr));
		ll max1=-999;
		for(int i=0;i<n;i++){
			scanf("%lld",&a);
			arr[a]++;
			max1=max(max1,a);
		}
		priority_queue<pair<ll,ll> ,vector<pair<ll,ll> > > pq;
		for(int i=0;i<=max1;i++){
			if(arr[i]!=0){
				pq.push(make_pair(arr[i],i));
			}
		}
		pair<ll,ll> p=pq.top();
		ll x=p.first;
		pq.pop();
		vector<ll> ans;
		ans.push_back(p.second);
		while(!pq.empty()){
			pair<ll,ll> p1=pq.top();
			if(x==p1.first){
				ans.push_back(p1.second);
				pq.pop();
			}else{
				break;
			}
		}
		for(int i=0;i<ans.size();i++){
			cout<<ans[i]<<" ";
		}
		cout<<endl;
	}
	return 0;
}
