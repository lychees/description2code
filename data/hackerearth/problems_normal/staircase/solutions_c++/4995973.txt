#include<bits/stdc++.h>
using namespace std;
int dp[1000001];
int n;
int jill(int start)
{
	if(start==n)
	  return 1;
	if(start>n-1)
	 return 0;
	if(dp[start])
	  return dp[start];
	dp[start]=jill(start+1);
	if(start<=n-2)
	  dp[start]+=jill(start+2);
	return dp[start];
}
int jack(int start)
{
	if(start==n)
	  return 1;
	if(start>n-1)
	  return 0;
	if(dp[start])
	  return dp[start];
	dp[start]=jack(start+1);
	if(start<=n-2)
	 dp[start]+=jack(start+2);
	if(start<=n-3)
	  dp[start]+=jack(start+3);
	return dp[start];
}
main()
{
	cin>>n;
	if(n<=0)
	{
		cout<<"ERROR";
		return 0;
	}
	memset(dp,0,sizeof(dp));
	cout<<"Jack-"<<jack(0)<<endl;
	memset(dp,0,sizeof(dp));
	cout<<"Jill-"<<jill(0)<<endl;
}
