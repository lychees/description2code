def ans(l):
    if len(l) == 2:
        return 0
    if len(l) == 3:
        return l[0]*l[2]
    a = 0
    for i in range(1, len(l)-1):
        m = l[i-1]*l[i+1]
        p = l[:]
        # print(i)
        del p[i]
        k = ans(p)+m
        if (k > a):
            a = k
    return a


n = int(raw_input())
l = list(map(int, raw_input().split()))
print ans(l)
