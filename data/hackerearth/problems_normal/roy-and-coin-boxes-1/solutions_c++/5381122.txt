#include <bits/stdc++.h>
using namespace std;
#define ll long long
int has[1000001],ans[1000001];

int main()
{
	int n;
	ios_base::sync_with_stdio(false);
	cin>>n;
	int a[n+2];
	memset(a,0,sizeof(a));
	int m;
	cin>>m;
	while(m--)
	{
		int l,r;
		cin>>l>>r;
		a[l-1]++;
		a[r]--;
	}
	has[a[0]]++;
	//cout<<a[0]<<endl;
	for(int i=1;i<n;i++)
	{
		a[i] = a[i-1]+a[i];
		has[a[i]]++;
	}
	ans[0] = has[0];
	for(int i=1;i<n;i++)
	{
		ans[i] = ans[i-1]+has[i];
		//cout<<has[i]<<" "<<ans[i]<<endl;
	}
	int q;
	cin>>q;
	while(q--)
	{
		int x;
		cin>>x;
		int answer = n-ans[x-1];
		cout<<answer<<endl;
	}
	return 0;
}
