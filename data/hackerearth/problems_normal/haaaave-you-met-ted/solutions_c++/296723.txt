using namespace std;
int getDist(long long a) {
	int ret = 0;
	while (a) {
		++ret;
		a &= (a-1);
	}
	return ret;
}
int main() {
	int T;
	for (cin >> T; T; T--) {
		int N;
		int mini = 32;
		cin >> N;
		for (int i = 0; i < N; i++) {
			long long a; cin >> a;
			mini = min(getDist(a), mini);
		}
		if (N == 1) mini = 1;
		cout << mini <<"\n";
	}
    return 0;    
}
