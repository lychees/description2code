#include<bits/stdc++.h>

using namespace std;

vector < int > v;
set < int > ss;

int main()
{

    int t , n , m, tem1, j;

    scanf("%d", &t);

    int i = 1, last = 1, maxi = -1, compare;

    pair < int , int > ans;

    while(t--){

        ss.clear();
        v.clear();
        maxi = -1, last = 1, i = 1, j = 0;

        scanf("%d", &n);

        for(i = 1; i <= n; i++){

            scanf("%d", &tem1);

            if(ss.find(tem1) != ss.end()){


                compare = i - last;
                if(compare > maxi){

                    maxi = compare;
                    ans = {last , i - 1};
                }

                int pos = 1;

                while(v[j] != tem1){

                    ss.erase(ss.find(v[j]));
                    j++;
                    pos++;
                }

                ss.erase( ss.find(v[j]));
                j++;
                last += pos;
            }

             ss.insert(tem1);
             v.push_back(tem1);
        }

        compare = i - last;
        if(compare > maxi){
            ans = {last , i - 1};
        }

        printf("%d %d\n", ans.first , ans.second);
    }

    return 0;
}
