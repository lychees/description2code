int get(int n, int k) {
	int half = (n + 1) / 2;
	if (k <= half) return (k - 1) * 2 + 1;
	int z = get(n - half, k - half);
	if (n & 1) {
		if (z == n - half) z = 1;
		else z++;
	}
	return z * 2;
}

int main() {
	int tc;
	scanf("%d", &tc);
	while (tc--) {
		int n, k;
		scanf("%d %d", &n, &k);
		printf("%d\n", get(n, k));
	}
	return 0;
}
