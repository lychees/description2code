#include <iostream>
#include <math.h>

using namespace std;

int main()
{
    unsigned long long int i,j,k,l,f,h,x,n,m,a,b,c,d,t;
    cin>>t;
    while(t--)
    {
        cin>>a>>b>>c>>d>>k;
        h=cbrt(k)+1;l=1;
        x=0;
        while(l<=h)
        {
            m=(l+h)/2;
            f=a*m*m*m+b*m*m+c*m+d;
            if(f==k)
            {
                x=m;
                break;
            }
            else
            {
                if(f<k)
                {x=m;
                    l=m+1;
                }
                else
                    h=m-1;

            }
        }

        cout<<x<<"\n";

    }
    return 0;
}
