for t in range(input()):
 s=raw_input()
 l=len(s)
 flag=1
 hashh=[0]*26
 check=26
 for i in range(l):
  hashh[ord(s[i])-65]+=1
 for i in range(26):
  if hashh[i]==0:
   check=check-1	
   continue
  if hashh[i]!=check:
   flag=0
   break
  check=check-1 
 if flag==1:
  print "Yes"
 else:
  print "No"
