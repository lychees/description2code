#include <bits/stdc++.h>
using namespace std;
 
long long int dp[1000][1000];
long long int count( int S[], int m, int n )
{
	if(dp[m][n]!=0)
	return dp[m][n];
    if (n == 0)
        return 1;
    
    if (n < 0)
        return 0;
    if (m <=0 && n >= 1)
        return 0;
 
    
    dp[m][n]=count( S, m - 1, n ) + count( S, m, n-S[m-1] );
    return dp[m][n];
}
 
int main()
{
	memset(dp,0,sizeof dp);
    int i, j;
    int m,n;
    cin>>m>>n;
    int arr[n];
    for(int i=0;i<n;i++)
    cin>>arr[i];
    
    
    printf("%lld ", count(arr, n, m));
    
    return 0;
}
