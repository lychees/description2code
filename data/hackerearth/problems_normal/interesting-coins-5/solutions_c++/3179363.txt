    #include<bits/stdc++.h>
    using namespace std;
    int n;
    long long int a[1007];
    long long int b[1007];
    bool check(long long int x){
    for(int i=0;i<n;i++)
    b[i]=a[i];
    long long int temp=0;
    for(int i=0;i<n;i++){
    if(i==0){
    if((b[i]-x)>0){
    b[i]-=x;
    }
    else{
    b[i]=1;
    }
    }
    else{
    long long int y=b[i-1]+1;
    if(b[i-1]>=b[i]){
	if(abs(y-b[i])>x)	return false;
	b[i]=y;
	}
	else{
		if((b[i]-y)<=x)	b[i]=y;
		else	b[i]-=x;
	}
    }
    }
    return true;
    }
     
    void solve() {
    long long int l=0,r=2000000000;
    long long int ans=0;
    while(l<=r){
    long long int mid=(l+r)/2;
    if(check(mid)==true){
    ans=mid;
    r=mid-1;
    }
    else{
    l=mid+1;
    }
    }
    cout<<ans<<"\n";
    }
    int main(){
    int T;
    cin>>T;
    while(T--){
    cin>>n;
    bool flag=true;
    for(int i=0;i<n;i++){
    cin>>a[i];
    if(i!=0 && a[i-1]>=a[i]){
    flag=false;
    }
    }
    if(flag==true){
    cout<<"0\n";
    }
    else{
    solve();
    }
    }
    }
