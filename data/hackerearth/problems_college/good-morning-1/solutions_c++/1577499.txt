#include<iostream>

using namespace std;

int main()
{
	int t, n, d;
	cin >> t;
	while (t--)
	{
		int x = 0;
		cin >> n >> d;
		while (1)
		{
			if (x == 3)
			{
				x = 0;
				n += 2;
			}
			if (n >= d)
			{
				cout << "Yes\n";
				break;
			}
			if (n == 0)
			{
				cout << "No\n";
				break;
			}
			x++;
			n--;
			d = d - 2;

		}
	}
	return 0;
}
