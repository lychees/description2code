#include <bits/stdc++.h>
using namespace std ;

int DP[101][26] , N , cnt ;
int main(){

    cin >> N ;
    for(int i=1;i<=N;i++){
        string str ;
        cin >> str ;
        for(int j=0;j<str.length();j++)
            DP[i][str[j]-'a'] ++ ;
    }
    for(int i=1;i<=26;i++){
        bool ok = 1 ;
        for(int j=1;j<=N;j++)
              if(!DP[j][i-1]){
                ok = 0 ;
              }
        if(ok) cnt ++ ;
    }
    cout << cnt << endl ;
    return  0 ;
}
