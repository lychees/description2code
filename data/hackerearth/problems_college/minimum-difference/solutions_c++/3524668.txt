#include <bits/stdc++.h>
using namespace std;
long hsh[10000000] = {0};
int main() {
   int t, n, f, i;
   long a;
   scanf("%d", &t);
   while(t--) {
      scanf("%ld", &n);
      int min = INT_MAX, max = INT_MIN, f = 0;
      for(i = 0 ; i < n ; i++) {
         scanf("%d", &a);
         ++hsh[a];
         if(min > a)
            min = a;
         if(max < a)
            max = a; 
      } 
      if(hsh[min] > 1) 
         f = 1;
         hsh[min] = 0;
    
      int ex = min, ans = INT_MAX;
      for(i = min+1 ; i <= max ; i++) {
         if(hsh[i] == 1) {
            if(ans > i-ex)
               ans = i-ex;
            ex = i;
            hsh[i] = 0; 
         }
         if(hsh[i] > 1) {
            f = 1;
            hsh[i] = 0; 
         }
      }      
      if(f)
         ans = 0;
         printf("%d\n", ans);   
   }
   return 0;
}
