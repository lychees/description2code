T = long(raw_input(''))
while (T > 0):
    [A, B] = map(long, raw_input().split(' '))
    C = A ^ B
    count = 0
    while C > 0:
        C -= C & (-1*C)
        count += 1
    if (count & 1):
        print "NO"
    else:
        print "YES"
    T -= 1
