
#include <bits/stdc++.h>

using namespace std;

int main()
{
	//freopen("inp.txt","r",stdin);

	int n;
	cin>>n;

	int max=-1000000, total;
	int w,t,l;

	while(n>0)
	{
		cin>>w>>t>>l;
		total=(3*w)+(t)+(-1*l);

		if(total>max)
			max=total;

		n--;
	}

	cout<<max<<endl;

	//fclose(stdin);
	return 0;

}
