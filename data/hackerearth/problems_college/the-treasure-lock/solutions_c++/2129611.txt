#include<iostream>
#include<stdlib.h>
#include<stdio.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		if(n==1)
		{
			cout<<"-1"<<endl;
			continue;
		}
		if(n%5==0&&n%3==0)
		{
			for(int i=0;i<n;i++)
				cout<<"5";
		}
		else if(n%3==0)
		{
			for(int i=0;i<n;i++)
				cout<<"5";
		}
		else if(n%5==0)
		{
			for(int i=0;i<n;i++)
				cout<<"3";
		}
		else
		{
			int flag=0;
			while(flag==0)
			{
				int x=n-5;
				if(x%3==0)
				{
					flag=1;
					int y=x/3;
					for(int i=1;i<=y;i++)
						cout<<"555";
					int z=n-y*3;
					z=z/5;
					for(int i=1;i<=z;i++)	
						cout<<"33333";
						break;
				}
				
			}
		}
		cout<<endl;
	}
}
