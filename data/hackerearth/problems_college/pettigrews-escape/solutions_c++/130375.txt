#include<iostream>

using namespace std;

int mat[1003][1003],flag[1003][1003];

int search(int i,int j,int n)
{
	
	
	if((i==n-1)&&(j==n-1)&&(!mat[i][j]))
		return 1;

	int x=0;

	if(flag[i][j])
		return 0;

	if((!mat[i][j]))
	{
		flag[i][j]=1;

		if((i-1)>=0)
			x+=search(i-1,j,n);
		if((j-1)>=0)
			x+=search(i,j-1,n);
		if((i+1)<n)
			x+=search(i+1,j,n);
		if((j+1)<n)
			x+=search(i,j+1,n);

		flag[i][j]=0;
	}

	return x;
}

int main()
{
	int n,i,j;

	cin>>n;
	for(i=0;i<n;i++)
		for(j=0;j<n;j++)
			cin>>mat[i][j];

	cout<<search(0,0,n)<<endl;

	return 0;
}
