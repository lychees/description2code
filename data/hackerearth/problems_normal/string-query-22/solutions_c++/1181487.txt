#include<cstdio>
#include<vector>
#include<cstring>
using namespace std;


char s[212345];
bool ok[212345];
vector<int>p[30],ft[30];

void upd(int t, int idx, int v){
    for(;idx<int(ft[t].size());idx+=(idx&-idx)){
        ft[t][idx]+=v;
    }
}
int qry(int t ,int idx){
    int ans = 0;
    for(;idx>0;idx-=(idx&-idx)){
        ans+=ft[t][idx];
    }
    return ans;
}
int main(){
//    freopen("data.in","r",stdin);
    scanf("%s",s);
    int l = strlen(s);
    for(int i=0;i<l;i++){
        p[s[i]-'a'].push_back(i);
        ok[i] = true;
    }
    for(int i=0;i<30;i++){
        int sz = p[i].size();
        if(sz>0){
            ft[i] = vector<int>(sz+1,0);
            for(int j=1;j<sz+1;j++){
                upd(i,j,1);
            }
        }
    }
    int q,k;
    char chr[5];
    scanf("%d",&q);
    while(q--){
        scanf("%d %s",&k,chr);
        char ch = chr[0];
        int li = 1, ls = ft[ch-'a'].size()-1,mid,ans;
        while(li<=ls){
            mid = (li+ls)>>1;
            int cnt = qry(ch-'a',mid);
            if(cnt<k) li = mid+1;
            else if(cnt>=k){
                ls=mid-1;
                ans = mid;
            }
        }
        upd(ch-'a',ans,-1);
        ok[p[ch-'a'][ans-1] ] = false;
    }
    for(int i=0;i<l;i++){
        if(ok[i]){
            printf("%c",s[i]);
        }
    }
    return 0;
}
