#include<iostream>
#include<cstdio>
#define gc getchar
using namespace std;
typedef long long ll;
ll read_int() {
char c = gc();
while(c<'0' || c>'9') c = gc();
ll ret = 0;
while(c>='0' && c<='9') {
ret = 10 * ret + c - 48;
c = gc();
}
return ret;
}
int main()
{
    ll n,c=0,x;
    n=read_int();
    while(n!=0)
    {
        x=read_int();
        if(x<300000000)
            c++;
        n--;
    }
    printf("%lld",c);
    return 0;
}
