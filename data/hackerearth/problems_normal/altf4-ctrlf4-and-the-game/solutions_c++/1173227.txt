#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
const int MAXN = 100 + 10;
LL val[MAXN], A[MAXN][MAXN];
int id[MAXN], n;

bool cmp(int a, int b) {
    return val[a] > val[b];
}

int main() {
    int T; scanf("%d", &T);
    for (int _ = 0; _ < T; ++ _) {
        scanf("%d", &n);
        for (int i = 0; i < n; ++ i) val[i] = 0, id[i] = i;
        for (int i = 0; i < n; ++ i) {
            for (int j = 0; j < n; ++ j) {
                scanf("%lld", &A[i][j]);
                val[i] += A[i][j];
                val[j] += A[i][j];
            }
        }
        sort(id, id + n, cmp);
        LL sa = 0, sb = 0;
        for (int i = 0; i < n; i += 2) {
            for (int j = 0; j < n; j += 2) {
                sa += A[id[i]][id[j]];
            }
        }
        for (int i = 1; i < n; i += 2) {
            for (int j = 1; j < n; j += 2) {
                sb += A[id[i]][id[j]];
            }
        }
        printf("%lld %lld\n", sa, sb);
    }
    return 0;
}
