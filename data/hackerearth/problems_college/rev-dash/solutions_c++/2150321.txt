#include<iostream>
#include<bits/stdc++.h>

using namespace std;

int main ()
{
	ios::sync_with_stdio(false);

		 
	string a;
	cin>>a;
	int i,l;
	l=a.length();
	for(i=0;i<l-1;i++)
	{
		cout<<a[i]<<"-";
	}
	cout<<a[l-1]<<endl;
	for(i=l-1;i>=1;i--)
	{
		cout<<a[i]<<"-";
	}
	cout<<a[0]<<endl;
	



	
	return 0;
}
