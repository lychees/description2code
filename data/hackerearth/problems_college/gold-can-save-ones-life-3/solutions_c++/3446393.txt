#include<bits/stdc++.h>
using namespace std;

int n, k;
long long int arr[10010];
long long int maxi, ans, maximum;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        maxi=0;
        ans=0;
        maximum=0;
        cin>>n>>k;
        for(int i=0; i<n; i++)
        {
            cin>>arr[i];
            if(arr[i]>maximum)
                maximum=arr[i];
            if(arr[i]<0)
                arr[i] = 0;
        }

        maxi = arr[n-1];
        if(maxi<0)
            maxi=0;

        for(int i=n-1-k-1; i>=0; i--)
        {
            if(arr[i+k+1]>maxi)
                maxi=arr[i+k+1];
            arr[i]+=maxi;
            if(arr[i]>ans)
                ans=arr[i];
        }
        if(maximum>ans)
            ans=maximum;
        cout<<ans<<endl;
    }
    return 0;
}
