#include <iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;

int scanint(){
	register int c = getchar();
	int x=0;
	for(;(c<48 || c>57);c=getchar());
	for(;c>47 && c<58;c=getchar())
	{
		x=(x<<1)+(x<<3)+c-48;
	}
	return x;
}


int main()
{
    int t,n,a[100],z,i,j;
    bool dp[1000001]={false};
    n=scanint();
    for(i=0;i<n;i++){
    	a[i]=scanint();
    }
    dp[0]=true;
    for(i=0;i<n;i++){
    	for(j=1000000-a[i];j>=0;j--){
    		if(dp[j])
    			dp[j+a[i]]=true;
    	}
    }
    t=scanint();
    while(t--){
    	z=scanint();
    	if(dp[z])
    		cout<<"Yes\n";
    	else
    		cout<<"No\n";
    }
}
