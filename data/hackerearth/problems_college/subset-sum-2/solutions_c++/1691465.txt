#include <iostream>
using namespace std;
long long n, ans, i;
#define mod 1000000007
int main()
{
    int t;
    cin >> t;
    while(t--){
    	cin >> n;
    	ans = 1;
    	for(i = 1; i <= n-1; i++) {
    		ans *= 2;
    		ans %= mod;
    	}
    	long long ans2 = 0;
    	for(int i = 1; i <= n; i++)
    		ans2 += i;
    	ans2 %= mod;
    	ans *= ans2;
    	ans %= mod;
    	cout << ans << endl;
    }
    return 0;
}
