#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
	int n,k,x;
	bool b;
    cin >> n >> k;
    vector<int> v;
    vector<int>::iterator it;
    for(int i=0;i<n;++i)
    {
    	cin >> x;
    	v.push_back(x);
    }
    sort(v.begin(),v.end());
    int count=0;
    it=v.begin();
    
    while(count<n)
    {
    	x=v[count];
    	++it;
    	b=binary_search(it,v.end(),k-x);
    	if(b==true)
    	{
    		//cout << "x" <<x;
    		break;
    	}
    	++count;
    }
    if(it==v.end())
    	cout << "NO";
    else
    	cout << "YES";
    return 0;
}
