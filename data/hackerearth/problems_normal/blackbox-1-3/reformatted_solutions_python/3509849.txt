from fractions import gcd
for _ in range(input()):
    alls = sorted(map(int, raw_input().split()))
    a = alls[0]
    b = alls[1]
    c = alls[2]
    if a * a + b * b == c * c:
        if gcd(a, b) == 1 and gcd(b, c) == 1 and gcd(a, c) == 1:
            print "YES"
            continue
    print "NO"
