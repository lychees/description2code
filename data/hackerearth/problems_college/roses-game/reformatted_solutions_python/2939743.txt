t = int(raw_input())
while t:
    n = int(raw_input())
    a = map(int, raw_input().split())
    i = 0
    flg = 0
    vis = [0]*n
    while vis[i] == 0:
        if (a[i] == -1):
            flg = 1
            break
        else:
            vis[i] = 1
            i = a[i]
    if (flg):
        print "Win"
    else:
        print "Lose"
    t = t-1
