#include <bits/stdc++.h>
using namespace std;
const int MAX = 1000 + 7;
long long a[MAX];
int N;
bool check (long long value )
{
	long long temp[MAX];
  long long min = a[0] - value;
   
   if(min <= 0)
    min = 1;
    temp[0] = min;
 // cout << temp[0] << endl;
  for(int i = 1; i < N; i++)
  {
  	 long long curr = a[i] - value;
  	  
  	    if(curr > temp[i-1])
		temp[i] = curr;
		
		
		else if(curr  <= temp[i-1])
		{
			
			long long want = temp[i-1] + 1;
			 
			  if(a[i] + value  >= want)
			  temp[i] = want;
			else
		 return 0;
		
		}
	
}
return 1;
}	
int main()
{
    int T;
    cin >> T;
    while (T--)
    {
    	long long maxi = 0;
    	cin >> N ;
    	for(int i =0 ; i < N ; i++){
    	cin >> a[i];
    	//maxi = maxi + a[i];
    	}
    	long long low = 0;
    	long long high = 1000000000000 + 7;
    	maxi =0;
    	while (low <=high)
    	{
    		long long mid = low + (high - low)/2;
    		
    		if(check(mid))
    		{
    		 maxi = mid;
    		high = mid-1;
    		}
    		else
             {
             
             	low = mid+1;
             }
    	}
   
  // cout << check(1) << endl;
    	cout << maxi << endl;
    }
    return 0;
}
