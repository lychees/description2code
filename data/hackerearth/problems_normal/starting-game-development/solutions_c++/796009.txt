#include<stdint.h>
#include<stdio.h>
#include<limits.h>
int findlevel(int a[],int b,int m)
{
int i;
int l=0,r=m-1;
int mid=(l+r)/2;
int n;
if(m>15){
if(a[mid]<b)
l=mid-3;
else
r=mid+3;
}
for(i=l;i<=r;i++)
{
    if(b<a[i])
    return i;
}
return i;
}
int main()
{
int n,m,q;
int level[101][5001];
int strth[101];
//cin>>n>>m>>q;
scanf("%d%d%d",&n,&m,&q);
for(int i=0;i<n;i++)
{
    for(int j=0;j<m;j++)
    {
        //cin>>level[i][j];
        scanf("%d",&level[i][j]);
    }

}
for(int i=0;i<q;i++)
{
int min=INT_MAX;
    for(int j=0;j<n;j++)
    {
       // cin>>strth[j];
       scanf("%d",&strth[j]);
       int lvl= findlevel(level[j],strth[j],m);
        if(lvl<min)
        min=lvl;
    }
    //cout<<min<<endl;
    printf("%d\n",min);
}

}
