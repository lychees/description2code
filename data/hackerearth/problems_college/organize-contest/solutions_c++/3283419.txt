#include"bits/stdc++.h"
using namespace std;
#define ll long long int
#define mod 1000000007
int main()
{
	map<string,vector<int> > m;
	int n,i;cin>>n;
	for(i=1;i<=n;i++)
	{
		string s;
		cin>>s;
		int c;
		cin>>c;
		if(m.count(s)==0)
		{
			m.insert(pair <string,vector<int> > ( s , vector <int> () ) );
			m[s].push_back(c);
			m[s].push_back(i);
		}
		else
		{
			m[s][0]+=c;
			m[s][1]=i;
		}
	}
	map<string,vector<int> > :: iterator p;
	int max=0;
	for(p=m.begin();p!=m.end();p++)
	{
		if(m[p->first][0]>max) max=m[p->first][0];
		//cout<<m[p->first][0]<<" "<<m[p->first][1]<<endl;
	}
	string g;
	int r=1009;
	for(p=m.begin();p!=m.end();p++)
	{
		//if(m[p->first][0]>max) max=m[p->first][0];
		//cout<<m[p->first][0]<<" "<<m[p->first][1]<<endl;
		if(m[p->first][0]==max)
		{
			if(m[p->first][1]<r)
			{
				r=m[p->first][1];
				g=p->first;
			}
		}
		
	}
	cout<<g<<endl;
}
