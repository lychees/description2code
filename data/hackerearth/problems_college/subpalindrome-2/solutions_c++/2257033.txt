#include <iostream>
#include<cstring>
using namespace std;
int dp[100][100];
int ways(string s, int i, int j)
{
	if(dp[i][j]!=-1)
		return dp[i][j];
	if(j<i)
		return 0;
	if(i==j)
		dp[i][j]=1;
	else
	{
		int ans = ways(s,i+1,j) + ways(s,i,j-1) - ways(s,i+1,j-1);
		if(s[i] == s[j])
			ans += 1 + ways(s,i+1,j-1);
		dp[i][j] = ans;
	}
	return dp[i][j];
}
int main()
{
    int t;cin  >> t;
    while(t--)
    {
    	string s; cin >> s; memset(dp,-1,sizeof(dp));
    	cout << ways(s,0,s.length()-1) << endl;
    }
    return 0;
}
