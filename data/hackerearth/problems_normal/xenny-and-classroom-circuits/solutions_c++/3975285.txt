#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define sz 1000000
int main()
{
    //cout << "Hello World!" << endl;
    ll t;
    cin >> t;
    char s1[sz + 1], s2[sz + 1];
    for(ll i = 0 ; i <= sz ; i+=2) {
    	s1[i] = 'B';
    	s1[i + 1] = 'G';
    	s2[i] = 'G';
    	s2[i + 1] = 'B';
    }
    /*for(ll i = 1 ; i <= sz ; i+=2) {
    	s1[i] = 'G';
    	s2[i] = 'B';
    }*/
    //cout << s1;
    while(t--) {
    	ll n , c1 = 0 , c2 = 0;
    	cin >> n;
    	string s;
    	cin >> s;
    	for(ll i = 0 ; i  < 2*n ; i++) {
    		//cout << s1[i];
    		if(s[i] != s1[i]) c1++;
    		if(s[i] != s2[i]) c2++;
    	}
    	/*for(ll i = 0 ; i < n ; i++) cout << s1[i];
    	cout << endl;
    	for(ll i = 0 ; i < n ; i++) cout << s2[i];*/
    	cout << min(c1, c2)/2 << endl;
    }
    return 0;
}
