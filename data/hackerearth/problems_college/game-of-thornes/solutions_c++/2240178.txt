#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
int main(){
	char arr[100001];
	scanf("%s",arr);
	char *p=arr;
	int l,i=0,j;
	l=strlen(arr);
    
	sort(p,p+l);
	//printf("%s",arr);	
    
    while(i!=l)
     {
     	j=i;
        
		if(arr[j]==arr[j+1])
        {
          arr[j]=arr[j+1]='#';
          i+=2;
        }
        
		else
        i++;
	 }
	 int count=0;
	 for(int k=0;arr[k]!=NULL;k++)
	  {
	  	if(arr[k]=='#')
	  	 count++;
	  }
	 
	 if(count==l||count==l-1)
	  cout<<"YES\n";
	  else
	   cout<<"NO\n";
}
