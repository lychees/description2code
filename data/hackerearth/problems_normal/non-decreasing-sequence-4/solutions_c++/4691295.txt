#include <bits/stdc++.h>
using namespace std;

const int N = 1000010;
int st[N], a[N];

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    int t, n, L;
    cin >> t;
    while (t--) {
    	cin >> n;
    	for (int i = 0; i < n; ++i)
    		cin >> a[i];
    	cin >> L;
    	int top = 1;
    	st[1] = a[0];
    	for (int i = 1; i < n; ++i) {
    		if (a[i] > st[top])
    			st[++top] = a[i];
    		else {
    			int s = 1, e = top;
    			while (s <= e) {
    				int m = s + (e - s)/2;
    				if (a[i] > st[m])
    					s = m + 1;
    				else
    					e = m - 1;
    			}
    			st[s] = a[i];
    		}
    	}
    	if (top >= L) cout << st[L] << endl;
    	else cout << -1 << endl;
    }
    return 0;
}
