#include<iostream>
using namespace std;
int main()
{
    long int p,q,i,cnt1[10001],cnt2[10001];
    int a[1000010],b[1000010];
    cin>>p;
    for(i=0;i<p;i++)
      {
        cin>>a[i];
        cnt1[a[i]]++;
      }
    cin>>q;
    for(i=0;i<q;i++)
       {
           cin>>b[i];
           cnt2[b[i]]++;
       }
    for(i=0;i<10001;i++)
    {
        if(cnt1[i]!=cnt2[i])
            cout<<i<<" ";
    }
    return 0;

    }
