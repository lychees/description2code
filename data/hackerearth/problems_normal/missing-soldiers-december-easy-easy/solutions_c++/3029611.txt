#include <bits/stdc++.h>
using namespace std;
#define ll long long
void solve(){
	ll int n,y;
    cin>>n;
    pair<ll int,ll int>p[n];
    for(int i = 0;i<n;i++){
    	cin>>p[i].first;
    	cin>>y;
    	cin>>p[i].second;
    }
    sort(p,p+n);
    ll int ans = 0;
    ll int maxi = 0;
    maxi = p[0].first + p[0].second;
    ans  += maxi - p[0].first + 1;
    for(int i = 1;i<n;i++){
    	if(p[i].first > maxi){
    		ans += p[i].second +1;
    		maxi = p[i].first + p[i].second;
    	}
    	else{
    		if(p[i].first + p[i].second  >=  maxi){
    			ll int d = p[i].first + p[i].second - maxi;
    			ans += d;
    			maxi = p[i].first + p[i].second;
    		}
    		else{
    			continue;
    		}
    	}
    }
    cout<<ans<<endl;
}
int main(){
 	int t = 1;
 	//cin>>t;
 	while(t--){
 		solve();
 	}
}
