#include <bits/stdc++.h>
using namespace std;

long long int m[1000001];

long long int f(long long int n)
{
	if( m[n] != -1)
		return m[n];
	
	
	m[n]  = ( f(n-1) % 1000000007 + ( f(n-2)  % 1000000007 * (n-1) % 1000000007 ) % 1000000007 ) % 1000000007;
	return m[n];
}

int main()
{
	for(int i = 0;i<1000001;i++)
		m[i] = -1;
		
	m[0] = m[1] = 1;
		
	long long int t;
	cin>>t;
	while(t--)
	{
		long long int n;
		cin>>n;
		cout<<f(n)<<endl;
	}
    
    return 0;
}
