#include<bits/stdc++.h>

#define mod 1000000007 // 10**9 + 7

using namespace std;

typedef long long int ll;
typedef long int l;

//defining the global variables
ll q[1005];
 
int main(){
    int T;
	cin>>T;
	memset(q,0,sizeof(q));
    q[0] = 1;
    for (int i=1; i < 1005; i++){
        for (int j=1; j <= 4; j++){
            if (i-j >= 0){
                q[i] += q[i - j];//adding previous values to the current value
                q[i] %= mod;
            }
        }
    }
    while(T--){
        int M,N;
        cin>>N>>M;
		vector<ll> dp_1;
		vector<ll> dp_2;
        //memset(d1, 0, sizeof(d1));
        //memset(d2, 0, sizeof(d2));
        dp_1.push_back(1);
        dp_2.push_back(1);
        for (int i=1; i<=M; i++)
        {
            ll v = 1;
            l x = q[i];

            for (int j = 1; j<=N; j<<=1){
                if(j & N){ 
					v *= x;
					v %= mod;
				}
                x *= x;
				x %= mod;
            }

            dp_2.push_back(v);

            for (int j = 1; j<i; j++){
                v -= dp_1[j] * dp_2[i - j];
                v %= mod;
				v += mod;
				v %= mod;
            }
            dp_1.push_back(v);
        }
        cout<<dp_1[M]<<endl;
    }
    return 0;
}
