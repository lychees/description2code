#include<iostream>
#include<cstring>
#include<stdio.h>
#include<vector>

using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int k=0;k<t;k++){
        char s1[1000000],s2[1000000];
        cin>>s1>>s2;
        int l1=strlen(s1);
        int l2=strlen(s2);
        int i,j;
        vector<int> veca,vecb;
        i=0;
        while(i<l1){
            if(s1[i]=='A')
                veca.push_back(i);
            else if(s1[i]=='B')
                vecb.push_back(i);
            i++;
        }
        i=0;
        vector<int>::iterator it1=veca.begin();
        vector<int>::iterator it2=vecb.begin();

        int m=*it2;
        int n=*it1;
        int flag=0;
        while(i<l2){
            int p=0,q=0;
            if(it1==veca.end())
                p=1;
            if(it2==vecb.end())
                q=1;
            if(s2[i]=='A'){
                if(i>*it1){
                    flag=1;
                    break;
                }
                if(q==0 && n>m && i<=m){
                    flag=1;
                    break;
                }
                it1++;
                n=*it1;
            }
            else if(s2[i]=='B'){
                if(i<*it2){
                    flag=1;
                    break;
                }
                if(p==0 && m>n && i>=n){
                    flag=1;
                    break;
                }
                it2++;
                m=*it2;
            }
            i++;
        }
        if(flag==0)
            cout<<"Yes"<<endl;
        else
            cout<<"No"<<endl;
    }
    return 0;
}
