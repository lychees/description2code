#include<iostream>
#include<vector>
#include<stdio.h>
using namespace std;
long long int fun(long long int n)
{
	long long int temp=n,sum=0;
	while(temp>0)
	{
		sum=(sum*10)+(temp%10);
		temp/=10;
	}
	return sum;
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long int a,b,c;
		scanf("%lld%lld",&a,&b);
		a=fun(a);
		b=fun(b);
		c=a+b;
		c=fun(c);
		printf("%lld\n",c);
		
	}
	return 0;
}
