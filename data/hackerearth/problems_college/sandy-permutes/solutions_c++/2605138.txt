using namespace std;

int main ()
{
	int i,n,k;
	si2(n,k);
	int a[n+1];
	map<int,int>m;
	fr(i,1,n+1)	{
		si(a[i]);
		m[a[i]]=i;
	}
	i=1;
    while(i<n&&k)
    {
        if(a[i]==n-i+1)
        {
            i++;
            continue;
        }
        int t=m[n-i+1];
        m[a[i]]=m[n-i+1];
        m[n-i+1]=i;
        swap(a[t],a[i]);
        k--;
        i++;
    }
	fr(i,1,n+1)
		pi(a[i]);
	nl;
return 0;
}
