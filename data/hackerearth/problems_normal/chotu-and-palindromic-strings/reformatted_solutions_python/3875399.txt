t = input()
o = []
for i in range(0, t):
    s1 = raw_input()
    s2 = raw_input()

    l1 = list(s1+s2)
    rev = l1[::-1]

    if l1 == rev:
        o.append("YES")
    else:
        l2 = list(s2+s1)
        rev = l2[::-1]
        if l2 == rev:
            o.append("YES")
        else:
            o.append("NO")

for e in o:
    print e
