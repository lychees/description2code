#include <bits/stdc++.h>
using namespace std;
int make_reverse(int a){
	int b=0,c;
	while(a!=0){
		c=a%10;
		b+=c;
		b*=10;
		a/=10;
	}
	b/=10;
	return b;}
	
int check_reverse(int a){
	int b=make_reverse(a);
	if(b==a)
	return 0;
	int c = pow(b,2);
	c=make_reverse(c);
	if(c==pow(a,2))
	return b;
	return 0;
}

int main()
{
   int t,n,b;
   cin>>t;
   while(t--){
   	cin>>n;
   	b=check_reverse(n);
   	if(b)
   	cout<<b<<endl;
   	else
   	cout<<"No"<<endl;
   }
    return 0;
}
