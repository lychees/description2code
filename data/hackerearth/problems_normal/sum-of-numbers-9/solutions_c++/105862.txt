#include <iostream>
#include<vector>
using namespace std;


int canGetSum(vector<int>& arr, long long sum, int index)
{
if(sum == 0)	
return 1;

if(index > arr.size())
return 0;


return canGetSum(arr,sum-arr[index], index+1) + canGetSum(arr,sum, index+1);
	
	
}
	
	
	
int main()
{
    
    int t;
    cin>>t;
    
    for(int i=0; i<t; i++)
    {
    	int n;
    	cin>>n;
    	vector<int> input(n);
    	int temp;
    	for(int j=0; j<n; j++)
    	{
    		
    		cin>>temp;
    		input[j] = temp;
    		
    	}
    	long long sum;
    	cin>>sum;
    	if(canGetSum(input,sum,0) == 0)
    	cout<<"NO\n";
    	else
    	cout<<"YES\n";
    	
    	
    }
    
    return 0;
}
