#include <bits/stdc++.h>

using namespace std;

int main()
{
    int tc;
    cin >> tc;
	int count = 1;
    while(tc--){
    	cout << "Case " <<count << ": "; 
        int num[3];
        cin >> num[0] >> num[1] >> num[2];

        for (int i = 0;i < 3;i++){

            int re[2] = {0};
            int k = 0;
            while(num[i] > 0)
            {
                re[k++] = num[i] % 16;
                num[i] /= 16;
                //cout << re << "  ";
            }


            for (int i = 1;i >= 0;i--)
            {
                //cout << re[i] << "  ";
                if (re[i] == 15)
                    cout << 'Z';
                else if (re[i] == 14)
                    cout << 'Y';
                else if (re[i] == 13)
                    cout << 'X';
                else if (re[i] == 12)
                    cout << 'W';
                else if (re[i] == 11)
                    cout << 'V';
                else if (re[i] == 10)
                    cout << 'U';
                else
                    cout << re[i];
            }

            cout << " ";
        }
		count++;
        cout << endl;
    }

    return 0;
}
