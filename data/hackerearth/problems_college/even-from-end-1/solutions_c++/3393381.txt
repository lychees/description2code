#include <iostream>
using namespace std;

int main()
{
    int n;
    int arr[1000000],i,j;
    while(cin>>n){
    	for(i=0;i<n;i++){
    		cin>>arr[i];
    	}
    	int flag=0;
    	for(i=n-1;i>=0;i--){
    		if((arr[i]%2) == 0){
    			flag=1;
    			cout<<arr[i]<<" ";
    		}
    	}
    	if(!flag){
    		cout<<"None to display";
    	}
    	cout<<endl;
    }
    return 0;
}
