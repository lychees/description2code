    #include <iostream>
    #include <string>
    #include <math.h>
    using namespace std;

    int main() {
     
     
    	long long test;
    	cin>>test;
    	long long a;
    	long long result=0;
    	long long p,q;
    	while(test--){
    		cin>>a;
     
    		p=1;q=1;
    		if(a==1)
    			cout<<"1"<<endl;
    		else{
    			a=a-1;
    			while(a--){
    				result=p+q;
    				p=q;
    				q=result;
    			}
    			cout<<result<<endl;
     
    		}
     
    	}
     
    	return 0;
     
    }
