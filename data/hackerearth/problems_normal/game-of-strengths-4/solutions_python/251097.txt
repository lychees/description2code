import sys
inp = sys.stdin.readline

def absDiff(arr, n):
	if  n < 2:
		return 0;
	
	arr.sort()
	
	summ = 0
	i = n-1
	while i >= 0:
		summ += (arr[i]*(i) - arr[i]*(n-i-1));
		i -= 1

	return summ*arr[-1];
	

def main():
	t = int(inp())
	for test in range(t):
		n = int(inp())
		a = map(int, inp().strip().split())
		print absDiff(a, n)%1000000007

main()
