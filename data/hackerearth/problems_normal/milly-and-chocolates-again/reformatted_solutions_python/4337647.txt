def b_search(C, query, low, high):
    while low < high:
        m = low+(high-low)/2
        if C[m] < query:
            low = m+1
        else:
            high = m
    return low+1


N = int(raw_input())
C = map(int, raw_input().split())
for i in xrange(1, N):
    C[i] += C[i-1]

Q = int(raw_input())
for i in xrange(Q):
    query = int(raw_input())
    print b_search(C, query, 0, len(C)-1)
