#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
	{
		long long int a[3],i,min;
		min = LLONG_MAX;
		for(i=0;i<3;i++)
		{
			scanf("%lld",&a[i]);
			if(a[i]<min)
			{
				min = a[i];
			}
		}
		double d = (1.0*min)/2;
		printf("%.1lf\n",d);
	}
    return 0;
}
