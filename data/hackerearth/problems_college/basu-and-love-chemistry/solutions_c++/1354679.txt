#include <iostream>
#include <string>
using namespace std;
static long SumSubtring(string s)
{
    long long sum = 0, mult = 1;
    for (int i = s.size(); i > 0; i--, mult = (mult * 10 + 1)%1000000007)
        sum = (sum + (s[i - 1] - '0') * mult * i)%1000000007;
    return sum;
}
int main()
{
    
        string s;
        cin>>s;
        cout<<SumSubtring(s)<<endl;
    return 0;
}
