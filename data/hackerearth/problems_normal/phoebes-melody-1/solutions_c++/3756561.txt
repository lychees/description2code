#include <bits/stdc++.h>
using namespace std;
int pos[200100];
set<int>Q;
int main()
{
    ios_base::sync_with_stdio(0);
    int t,n,k,i,num,tk;
    cin>>t;
    while(t--)
    {
    	cin>>n>>k;
		int Q[n];
    	for(i=0;i<n;i++)
    	{
    		cin>>Q[i];
    	}
    	long long sum=0;
    	for(i=0;i<n;i++)
    	{
    		tk=k;
    		while(i-tk>=0||i+tk<n)
    		{
    			if(i-tk>=0&&Q[i-tk]<Q[i])
    			{
    				sum+=tk;
    				break;
    			}
    			if(i+tk<n&&Q[i+tk]<Q[i])
    			{
    				sum+=tk;
    				break;
    			}
    			tk++;
    		}
    	}
    	cout<<sum<<"\n";
    }
    return 0;
}
