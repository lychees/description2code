#include <bits/stdc++.h>

#define REP(i, a) for (int i = 0; i < (int) (a); i ++)
#define REPP(i, a, b) for (int i = (int) (a); i <= (int) (b); i ++)
#define MST(a, b) memset(a, b, sizeof(a))
#define gc getchar_unlocked
using namespace std;

void scanint(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}

const int N = 205;
bool vis[N];

int main(){
    int k,m;
    scanint(k);
    scanint(m);
    //assert(k<=100);
    //assert(m<=100);
    int cnt = 0;
    while(k--){
        int x;
        scanint(x);
        //assert(x<=m);
        int y;
        while(x--){
            scanint(y);
            if(!vis[y]){
            	vis[y] = true;
            	cnt++;
            	if(cnt==m){
            		printf("YES\n");
            		return 0;
            	}
            }
        }
    }
    bool ans = true;
    for(int i=1;i<=m;i++){
        if(!vis[i]){
        	printf("NO\n");
        	return 0;
        }
    }
    printf("YES\n");
    return 0;
}
