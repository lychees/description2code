from math import ceil

def formatInput():
    return map(int, filter(lambda a: a, raw_input().split(" ")))

def main():
    numPc, numPd = map(int, raw_input().split(" "))
    pc = 1
    pd = 0
    totalTime = 0
    while pc<numPc: # all pc's don't have pdf's
        pc, pd = pc+pd, pd+pc
        totalTime += 1
        if (pd>= numPd) and (pc<=numPc):
            totalTime += int(ceil((numPc-pc)/float(numPd)))
            break
    return totalTime


if __name__ == '__main__':
    numTest = int(raw_input())
    for t in range(numTest):
        print main()
