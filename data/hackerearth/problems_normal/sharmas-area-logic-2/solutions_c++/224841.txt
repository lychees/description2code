#include<cstdio>
#include<cmath>
#include<iostream>
#include<vector>
#include<map>
#include<stack>
#include<algorithm>
using namespace std;
int main()
{
int T;
int minx,maxx,miny,maxy;
scanf("%d",&T);
int N;
int a,b;
while(T--)
	{
	scanf("%d",&N);
	maxy=maxx=-1000000;
	miny=minx=100000;
	while(N--)
		{
			scanf("%d %d",&a,&b);
			minx=min(minx,a);
			maxx=max(maxx,a);
			miny=min(miny,b);
			maxy=max(maxy,b);
		}
	int ans=(maxx-minx)*(maxy-miny);
	if(minx==maxx)
		{
		ans=maxy-miny;
		}
	else if(maxy==miny)
		{
		ans=maxx-minx;
		}
	printf("%d\n",ans);
	}
return 0;
}
