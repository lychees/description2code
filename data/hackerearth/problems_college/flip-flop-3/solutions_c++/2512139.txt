#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    int t; cin >> t;
    while(t--)
    {
    	string s; cin >> s;
    	int j=0,ans=0;
    	for(int i=1;i<s.length();i++)
    	{
    		if(s[i]!=s[j])
    			j=i;
    		else
    			ans++;
    	}
    	cout << ans << endl;
    }
    return 0;
}
