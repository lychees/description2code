#include <iostream>
#include<string.h>
int main()
{ int t,i,n;
char a[10000],b[10000],g[10000];
scanf("%d",&t);
while(t>0)
{ scanf("%s",&a);
scanf("%s",&b);
n=strlen(a);
for(i=0;i<n;i++)
{ if(a[i]==b[i])
  {  g[i]=0;
  }
  else
  { g[i]=1;
}}
for(i=0;i<n;i++)
{  printf("%d",g[i]);
}
printf("\n");
t--;
}
    return 0;
}
