#include <iostream>
using namespace std;

int main()
{
	long int tc,x,y,n,i,sum,a[1000000],temp;
    cin>>tc;
    while(tc--)
    {
    	a[0]=0;
    	a[1]=1;
    	x=0;
    	y=1;
    	i=2;
    	cin>>n;
     	temp=n;
    	while(temp>2)
    	{
    		sum=x+y;
    		a[i]=sum;
    		x=y;
    		y=sum;
    		i++;
    		temp--;
    	}
    	for(i=n-1;i>=0;i--)
    	cout<<a[i]<<" ";
    	cout<<"\n";
    	
    }
    return 0;
}
