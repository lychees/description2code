# print 'Hello World!'

dec = {}
count = 0

n = int(raw_input())
for i in range(0, n):
    t = raw_input()
    a, b, c = t.split()
    sos = (int(a)**2) + (int(b)**2) + (int(c)**2)
    if sos not in dec:
        dec[sos] = 1
    else:
        dec[sos] = dec[sos] + 1

for key in dec:
    if dec[key] == 1:
        count = count + 1

print count
