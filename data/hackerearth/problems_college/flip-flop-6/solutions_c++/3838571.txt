#include <iostream>
#include <string>
using namespace std;
int main()
{
	int i,n,j,size,count=0;
	  cin>>n;
	  string str[n];
	  i=0;
	  while(i<n){
	  cin>>str[i]; 
	  i++;
	  }	
	  i=0;
	  j=0;
	  for(i=0;i<n;i++)
	       {
			   count=0;
		 size=str[i].length();
		for(j=0;j<size;j++){
	  if(str[i][j]==str[i][j+1])
	  {
		  count++;
	  }
  }
	  cout<<count<<endl;
   }
	  return 0;
  }
