#include <iostream>
#include<math.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    long int n,count;
    while(t--)
    {
      	count=0;
       	scanf("%ld",&n);
  		for(int i=1;i<=sqrt(n);i++)
   		{
   			if(n%i==0)
   			{
   				if(i%2==0)
   				    count++;
      			if((n/i)!=i && (n/i)%2==0)
                    count++;
      		}

        }
        printf("%d\n",count);
    }
}
