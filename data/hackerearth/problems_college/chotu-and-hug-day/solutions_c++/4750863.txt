#include <bits/stdc++.h>
using namespace std;
int main(){
    int t;
    int n;
    int h;
    int g,w;
    int mn;
    int i;
    cin>>t;
    while(t--)
    {
        cin>>n>>h;
        mn=10000;
        for(i=0;i<n;i++)
        {
            cin>>g;
            w=abs(g-h);
            mn=min(mn,w);
        }
        cout<<mn<<endl;
    }
    return 0;
}
