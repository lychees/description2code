#include<iostream>
#include<bits/stdc++.h>
using namespace std;
main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long n;
		cin>>n;
		long long a[n],b[n];
		for(int i=0;i<n;i++)
		{
			cin>>a[i];
			b[i]=a[i];
		}
		sort(b,b+n);
		int flag=0;
		for(int i=0;i<n;i++)
		{
			if(a[i]==b[i])
				continue;
			flag=1;
			break;
		}
		if(flag)
			cout<<"NO\n";
		else
			cout<<"YES\n";
	}	
}
