def maxsum(c, i, w):
    if (i <= 0 or w <= 0):
        return 0

    if (c[i-1] > w):
        return maxsum(c, i-1, w)
    if (d[i-1][w] != -1):
        return d[i-1][w]
    d[i-1][w] = max(maxsum(c, i-1, w), c[i-1] + maxsum(c, i-1, w-c[i-1]))
    return d[i-1][w]


t = int(raw_input())

    t -= 1

    s, W = map(int, raw_input().strip().split())

    d = [[-1]*(W+1) for i in range(len(arr) + 1)]
