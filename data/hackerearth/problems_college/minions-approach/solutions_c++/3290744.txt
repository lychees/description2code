#include <iostream>
#include <math.h>
#include <cmath>
#include <cstdio>

using namespace std;

int t, n;

int gcd(int a , int b)
{
   if(b==0) return a;
   a%=b;
   return gcd(b,a);
}

int main() {
	ios_base::sync_with_stdio(0);
	cin >> t;
	while (t--) {
		cin >> n;
		int count = 0;
		for (int i = 1; i <= n; i++) {
			if (gcd(i, n) == 1) {
				count++;
			}
		}
		if (count <= 1) {
			cout << "Incorrect" << endl;
		}
		else {
			int ccount = 0;
			for (int i = 2; i < count; i++) {
				if (count % i == 0) {
					ccount++;
				}
			}
			if (ccount > 0) {
				cout << "Incorrect" << endl;
			}
			else {
				cout << "Correct" << endl;
			}
		}
		
	}
	return 0;
}
