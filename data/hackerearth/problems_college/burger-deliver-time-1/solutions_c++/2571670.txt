#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int n, t1, t2, t3;
    vector< pair<int, int> > v;
    cin >> n;
    for(int i=0; i<n; i++) {
    	pair <int,int> z;
    	cin >> t1 >> t2;
    	t3 = t1+t2;
    	z = make_pair(t3, i+1);
    	v.push_back(z);
    }
    sort(v.begin(), v.end());
    for(int i=0; i<n; i++) {
    	cout << v[i].second << " ";
    }
    return 0;
}
