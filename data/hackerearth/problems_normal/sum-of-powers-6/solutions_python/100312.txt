divi=[]
for i in range(1,1001):
    for j in range(2,21):
        m=pow(i,j)
        if(m>1000000):
            break;
        divi.append(m)
divi=set(divi)
divi=list(divi)
divi.sort()
t=int(raw_input())
for i in range(t):
    num=int(raw_input())
    count=0
    for i in divi:
        if i>num:
            break;
        check=num-i
        if check in divi:
            count=1
            break;
    if count==0:
        print "No"
    else:
        print "Yes"
