#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int p,f;
    	cin>>p>>f;
    	
    	int p1 = (p+f)/2;
    	int p2 =  (p-f)/2;
        if(p1<0 || p2<0 || (p+f)%2!=0 || (p-f)%2!=0)
        cout<<"impossible"<<"\n";
        else
        {
        	if(p1>p2)
        	cout<<p1<<" "<<p2<<"\n";
        	else
        	cout<<p2<<" "<<p1<<"\n";
        }
        
}
    return 0;
}
