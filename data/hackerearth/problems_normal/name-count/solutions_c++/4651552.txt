#include <iostream>
using namespace std;
typedef unsigned long long ull;
const ull MOD = 1000000007;

ull factorial(ull x)
{
	if(!x) return 1;
	return factorial(x-1)*x%MOD;
}

ull binary_pow(ull x, ull e)
{
	ull total = 1;
	while(e)
	{
		if(e&1) total=total*x%MOD;
		x = x*x%MOD;
		e>>=1;
	}
	return total;
}

int main()
{
    int t;
    cin>>t;
    ull n,m;
    while(t--)
    {
    	cin>>n>>m;
    	cout<<factorial(m)*binary_pow(factorial(m-n),MOD-2)%MOD<<'\n';
    }
    return 0;
}
