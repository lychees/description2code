#include <iostream>
#define uint64 unsigned long long int
#define mod 1000003
using namespace std;
long arr[1000003];
int main()
{
    uint64 x, n, i;
    int t;
    cin>>t;
    arr[0] = 1;
    for(i=1; i<mod; i++) {
    	arr[i] = (i * (arr[i-1]%mod))%mod;
    }
    while(t--) {
    	cin>>n>>x;
    	if(n>mod) {
    		cout<<0<<endl;
    	} else {
    		cout<<(x%mod * arr[n])%mod<<endl;
    	}
    }
    return 0;
}
