#include <stdio.h>
#include <algorithm>

using namespace std;

struct node {
	int a, b, c;
};

int min (int i, int j) {
	if (i < j) {
		return i;
	}
	return j;
}

node w [1001];
int dp [1001];

bool compare (node i, node j) {
	return i.a < j.a;
}

int main ()
{
	int t, i, j, k, l, n;
	
	scanf ("%d", &t);
	
	while (t--) {
		scanf ("%d%d", &l, &k);
		for (i = 0; i < k; i++) {
			scanf ("%d%d%d", & w [i].a, & w [i].b, & w [i].c);
			dp [i] = 1000001;
		}
	
	
	for (i = 0; i <= l; i++) {
		dp [i] = 1000001;
	}
	
	sort (w, w + k, compare);
	
	dp [0] = 0;
	
	for (i = 0; i < k; i++) {
		for (j = w [i].a; j <= w [i].b; j++) {
			dp [w [i].b] = min (dp [w [i].b], dp [j] + w [i].c);
		}
	}
	
	if (dp [l] == 1000001) {
		dp [l] = -1;
	}
	
	printf ("%d\n", dp [l]);
	}
	
	return 0;
}
