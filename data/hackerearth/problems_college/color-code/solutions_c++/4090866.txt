#include <bits/stdc++.h>
#define ll long long int
#define ff fflush(stdout)
#define fb(i,n) for(i=0;i<n;i++)
#define sd(a) scanf("%d",&a)
#define sld(a) scanf("%lld",&a)
#define ss(a) scanf("%s",a)
#define pd(a) printf("%d",a)
#define pld(a) printf("%lld ",a)
#define ps(a) printf("%s",a)
#define pn() printf("\n")
#define w(t) while(t--)
#define mod 1000000007
#define PI 3.141592653589793

using namespace std;

int main()
{

    ll t,it;sld(t);
    fb(it,t)
    {
        ll a,b,c,i;
        cin>>a>>b>>c;
        vector<ll>va,vb,vc;
        while(a)
        {
            va.push_back(a%16);
            a/=16;
        }
        while(b)
        {
            vb.push_back(b%16);
            b/=16;
        }
        while(c)
        {
            vc.push_back(c%16);
            c/=16;
        }
        cout<<"Case "<<it+1<<": ";
        if((va.size()==1))
        	cout<<"0";
        else if(va.size()==0)
        	cout<<"00";
        for(i=(ll)va.size()-1;i>-1;i--)
        {
            if(va[i]<10)
                cout<<va[i];
            else
            {
                cout<<(char)(va[i]-10+'U');
            }
        }
        cout<<" ";
        if(vb.size()==1)
        		cout<<"0";
        else if(vb.size()==0)
        	cout<<"00";
        for(i=(ll)vb.size()-1;i>-1;i--)
        {
            if(vb[i]<10)
                cout<<vb[i];
            else
            {
                cout<<(char)(vb[i]-10+'U');
            }
        }
        cout<<" ";
        if((vc.size()==1))
        	cout<<"0";
        else if(vc.size()==0)
        	cout<<"00";
        for(i=(ll)vc.size()-1;i>-1;i--)
        {
            if(vc[i]<10)
                cout<<vc[i];
            else
            {
                cout<<(char)(vc[i]-10+'U');
            }
        }
        cout<<endl;
    }
    return 0;
}
