#include<iostream>
#include<algorithm>
using namespace std;
#include<stdio.h>
#include<string.h>

long long a[10000009],b[10000009];
int main()
{


int t;
cin>>t;
while(t--)
{
    long long n,s;long long x,mapo=0,mane=0;
    s=0;

    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>x;
        if(x>=0)
        {
            a[x]++;
            if(x>mapo)
                mapo=x;

        }
        else
        {
            b[abs(x)]++;
            if(abs(x)>mane)
                mane=abs(x);
        }
        }
       
        for(int i=0;i<=mapo;i++)
        {
            s+=a[i];
            s+=(((a[i])*(a[i]-1))/2);
            a[i]=0;
        }
           for(int i=0;i<=mane;i++)
        {
            s+=b[i];
            s+=(((b[i])*(b[i]-1))/2);
            b[i]=0;
        }
        cout<<s<<endl;
}


}
