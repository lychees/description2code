#include <bits/stdc++.h>

using namespace std;

int a[100];

int sumofdigits(int x)
{
	int t = 0;
	while(x > 0) {
		t += x % 10;
		x = x/10;
	}
	return t;
}

int main()
{
	int n;
	scanf("%d",&n);
	for(int i = 0; i < n; i++) {
		scanf("%d",&a[i]);
	}	
	int l,b;
	scanf("%d %d",&l,&b);
	int area = l*b;
	int total = 0;
	double val;
	for(int i = 0; i < n; i++) {
		val = a[i] * sumofdigits(a[i]) * area;
		if(a[i] >= 1 && a[i] <= 10) {
			val = val * (0.75);
		} else if(a[i] >= 11 && a[i] <= 20) {
			val = val * (0.7);
		} else {
			val = val * (0.8);
		}
//		cout << val << endl;
		total += (int)floor(val);
	}
	printf("%d\n",total);
	return 0;
}
