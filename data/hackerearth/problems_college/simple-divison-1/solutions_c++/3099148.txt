#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long a,b,c;
		scanf("%lld%lld%lld",&a,&b,&c);
		printf("%lld",a/b);
		a %= b;
		if(c)
			printf(".");
		while(c--)
		{
			a *= 10;
			printf("%lld",a/b);
			a %= b; 
		}
		printf("\n");
	}
	return 0;
}
