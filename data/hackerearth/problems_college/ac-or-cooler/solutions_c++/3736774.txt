#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
typedef vector<int> vi;
typedef pair<int,int> pii;

int main()
{
	ll n,m;
	cin>>n>>m;
	vector<ll> v;
	ll x;
	for(int i=1;i<=2*n;i++)
	{
		scanf("%lld",&x);
		v.push_back(x);
	}
	sort(v.begin(), v.end());
	ll ans = 0;
	for(int i=0;i<2*n;i++)
	{
		if(m>=v[i])
		{
			m-=v[i];
			ans++;
		}
		else break;
	}
	cout<<min(ans,n)<<endl;
}
