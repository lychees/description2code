#include <bits/stdc++.h>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false); 
	cin.tie(NULL);
    int m;
    cin>>m;
    
    map<int,int> orig_map;
    map<int,int> copy_map;
    
    for(int i=0;i<m;i++)
    {
    	int x;
    	cin>>x;
    	orig_map[x] = 1;
    }
    
    
    
    int n;
    cin>>n;
    
    int ans=0;
    
    while(n--)
    {
    	copy_map = orig_map;
    	string the_string;
		getline(cin,the_string);
		istringstream iss(the_string);
		for (int s; iss >> n; )
		{
			  copy_map.erase(n);
		}
		
		if(copy_map.size()==0)
			ans++;
    }
    cout<<ans;
}
