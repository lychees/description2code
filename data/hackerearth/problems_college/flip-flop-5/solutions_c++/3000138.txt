#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    char s[100000];
    stack<char>st;
    cin>>n;
    while(n--)
    {
    	cin>>s;
    	int i=1,ans,l;
    	l=strlen(s);
    	st.push(s[0]);
    	while(s[i])
    	{
    		if(st.top()==s[i])
    		{
    			i++;
    		}
    		else
    		{
    			st.push(s[i]);
    			i++;
    		}
    	}
    ans=l-st.size();
    cout << ans;
    printf("\n");
    while(!st.empty())
    {
        st.pop();
    }
    }
    return 0;
}
