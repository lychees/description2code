#include <iostream>
using namespace std;

void flip(string &s)
{
	for(int i=0, j=s.size()-1; i<j; i++,j--)
    {
    	char c = s[i];
    	s[i] = s[j];
    	s[j] = c;
    }	
}

void remove_trailing_zeros(long int &x)
{
	while(x%10 == 0)
	{
		x = x/10;
	}
}

int main()
{
    long int t;
    cin>>t;
    
    while(t)
    {
    	
    	string s1,s2;
    	cin>>s1>>s2;
    	flip(s1);
    	flip(s2);
    	long int n1 = stoi(s1);
    	long int n2 = stoi(s2);
    	long int sum = n1+n2;
    	remove_trailing_zeros(sum);
    	string res = to_string(sum);
    	flip(res);
    	cout<<res<<endl;
    	t--;
    }
    return 0;
}
