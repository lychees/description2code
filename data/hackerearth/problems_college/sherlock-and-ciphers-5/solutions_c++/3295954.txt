#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<math.h>
int main()
{
	long long int n,i,l,j,k,si=0;
	scanf("%lld",&n);
	getchar();
	for(i=0;i<n;i++)
	{
		char s[201],a[201][201];
		scanf("%[^\n]s",s);
		getchar();
		l=(int)sqrt(strlen(s));
		for(j=0;j<l;j++)
		{
			for(k=0;k<l;k++)
			{
				a[j][k]=s[si];
				si+=1;
			}
		}
		for(j=0;j<l;j++)
		{
			for(k=0;k<l;k++)
			{
				printf("%c",a[k][j]);
			}
		}
		si=0;
		printf("\n");
	}
	return 0;
}
