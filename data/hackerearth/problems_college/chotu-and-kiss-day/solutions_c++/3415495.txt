#include <stdio.h>
#include <iostream>
using namespace std;
#include <algorithm>
long long int bs(long long int a[],long long int l,long long int u,long long int f)
{
	long long int mid;
	if(l<=u)
	{
		mid=(l+u)/2;
		if(a[mid]==f)
		return f;
		else if(a[mid]>f)
		return bs(a,l,mid-1,f);
		else
		return bs(a,mid+1,u,f);
	}
	return -1;
}
int main()
{
	long long int op=0,x,l,u,f,p,c=0,k,n,i,j,s;
	scanf("%lld",&n);
	long long int a[n];
	for(i=0;i<n;i++)
	{
		scanf("%lld",&a[i]);
	}
	l=0;u=n-1;
	scanf("%lld",&p);
	sort(a,a+n);
	for(i=0;i<n-2;i++)
	{
		for(j=i+1;j<n-1;j++)
		{
				s=0;
				s+=a[i]+a[j];
				f=p-s;
				x=bs(a,l,u,f);
				if(f==x)
				{
					op=1;
					break;
				}
		}
	}
	if(op==1)
	printf("YES\n");
	else
	printf("NO\n");
}
