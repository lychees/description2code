#include<iostream>
#include<vector>
#include<string>
#include<cstdio>
#include<queue>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string s;
		long long int c=0,d=0,i;
		cin>>s;
		for(i=0;i<s.size();i++)
		{
			if(s[i]=='1')
			c++;
			else
			d++;
		}
		if(c>d)
		cout<<c-d<<endl;
		else
		cout<<d-c<<endl;
	}
	return 0;
}
