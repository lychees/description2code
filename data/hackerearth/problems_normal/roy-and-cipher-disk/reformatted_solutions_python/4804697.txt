n = input()


for i in xrange(0, n):
    value = list(raw_input())
    value.insert(0, 'a')
    for i in xrange(1, len(value)):
        diff = ord(value[i]) - ord(value[i - 1])
        if abs(diff) > 13:
            if diff > 0:
                print diff - 26,
            else:
                print 26 + diff,
        elif diff == -13:
            print 13,
        else:
            print diff,
    print
