#include <iostream>
using namespace std;

int main()
{
    int t,prime[1000000]={0};
    long i=2,j,a,b,s=1;
    prime[0]=prime[1]=1;
    while(i<=1000)
    {
    	if(prime[i]==0)
    	{
    		j=2*i;
    		while(j<=1000000)
    		{
    			prime[j]=1;
    			j=j+i;
    		}
    	}
    	i++;
    }
   	cin>>t;
    while(t>0)
    {
    	s=1;
    	cin>>a;
    	if(prime[a]==1)
    		cout<<1;
    	else
    	{
    		
			for(i=2;i<=a;i++)
			{
				if(prime[i]==1&&prime[a-i]==1)
				{
					s=s+1;
					break;
				}
			}
			if(s==1)
				s=3;
			cout<<s;
    	}
    	cout<<endl;
    	t--;
    }
}
