#include <iostream>
#include <algorithm>
 
using namespace std;
 
bool check(long *q, long N, long M, long jump)
{
	long m = 0;
	long sum = 0;
	long index = 0;
	
	while(true)
	{
		sum += q[index];
		
		if(sum > jump)
		{
			sum = 0;
			m++;
			if(m >= M)
				return false;
		}
		else
		{
			index++;
			if(index >= N)
				return true;
		}
	}
}
 
long sumAll(long *q, long N)
{
	long ans = 0;
	for(long i = 0; i < N; i++)
		ans += q[i];
		
	return ans;
}
 
int main()
{
    long N, M;
    cin >> N >> M;
    
    long *q = new long[100000];
    
    for(long i = 0; i < N; i++)
  		cin >> q[i];
    
    long start = 1;
    long end = sumAll(q, N);
    long ans;
    
    while(start <= end)
    {
    	long mid = start + end;
  		mid >>= 1;
  		if(check(q, N, M, mid))
  		{
  			ans = mid;
  			end = mid - 1;
  		}
  		else
  		{
  			start = mid + 1;
  		}
    }
    
    cout << ans;
    
    return 0;
}
