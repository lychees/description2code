#include <iostream>
using namespace std;

int arr[1005][3005];

void PRE()
{
	arr[0][3000]=1;
	for(int i=1;i<=1000;i++)
	{
		int c=0;
		for(int j=3000;j>=0;j--)
		{
			int res =arr[i-1][j]*i + c;
			arr[i][j]=res%10;
			c=res/10;
		}
	}
}

int main()
{
	PRE();
	int T;
	cin>>T;
	while(T--)
	{
		int N;
		cin>>N;
		int s=0;
		for(int l=0;l<=3000;l++)
		s+=arr[N][l];
		cout<<s<<endl;
	}
}
