#include <iostream>
#include <cstdio>

using namespace std;
long long N[8];

long long sum (long long i) {
    return (((i+1)*(i+2)*(2*i+3)+3*(i+1)*(i+2))/12);
}

int main () {
    long long t, n, p, q, r, i, s;
    long long count=0;
    scanf ("%lld", &t);
    while (t--) {
        scanf ("%lld %lld %lld %lld",&n ,&p ,&q ,&r);
        for (i = 0; i < 8; i++) {
            N[i] = 0;
        }
        N[0] = sum (n);
        s = n-p-1;
        if (s >= 0) {
            N[1] = sum (s);
        }
        s = n-q-1;
        if (s >= 0) {
            N[2] = sum (s);
        }
        s = n-r-1;
        if (s >= 0) {
            N[3] = sum (s);
        }
        s = n-p-q-2;
        if (s >= 0) {
            N[4] = sum (s);
        }
        s = n-q-r-2;
        if (s >= 0) {
            N[5] = sum (s);
        }
        s = n-r-p-2;
        if (s >= 0) {
            N[6] = sum (s);
        }
        s = n-p-q-r-3;
        if (s >= 0) {
            N[7] = sum (s);
        }
        
        count = (N[0]+N[4]+N[5]+N[6])-(N[1]+N[2]+N[3]+N[7]);    
        printf ("%lld\n", count);
    }
    
    return 0;
}
