#include <iostream>
using namespace std;
 
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,i,j,o;
    	cin>>n>>o;
    	int a[n],p=0;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    		if(a[i]<=0)
				p++;
		
		}
    	if(p>=o)
    	cout<<"NO"<<endl;
    	else
    	cout<<"YES"<<endl;
    }
}
