#include <iostream>
#include <string>
#include <cmath>
#include <map>
#include <vector>
using namespace std;

int main()
{
	int n;
	cin>>n;
	map<string,int> party;
	map<string,int> tme;
	for(int i=0;i<n;i++)
	{
		string s;
		cin>>s;
		
		if(party.count(s)==0)
		{
			party[s]=1;
		//	continue;
		}
		else party[s]++;
		//party[s]++;
		if(tme.count(s)==0)
		{
			tme[s]=i;
		}
	}
	map<string,int>::iterator it;
	map<string,int>:: iterator pos;
	int prev=-1;
	int now=0;
	while(prev!=now)
	{
		prev=now;
		int max=-1;
		int tim=99999999;
		for(it=party.begin();it!=party.end();it++)
		{
			if(max<it->second || (max==it->second && tme[it->first]<tim))
			{


				now=prev+1;
				max=it->second;
				pos=it;
				tim=tme[it->first];
			}

		}
		if(now==prev)break;
		cout<<pos->first<<' '<<pos->second<<endl;
		pos->second=-2;
	}
	return 0;
}
