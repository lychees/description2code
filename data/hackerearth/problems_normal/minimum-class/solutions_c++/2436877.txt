#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    long long int n,ans,i,x;
    cin>>t;
    while(t--)
    {
        cin>>n;
        ans=0;
        i=1;
        x=0;
        while(true)
        {
            x=x+i;
            ans++;
            if(x>=n)
                break;
            x=x+i;
            ans++;
            if(x>=n)
                break;
            i++;
        }
        cout<<ans<<endl;
    }
    return 0;
}
