#include <iostream>
using namespace std;
int find(int a[],int b[],int c[],int sump,int sumv,int sumc,int i,int n)
{
	if(sump==0&&sumv==0&&sumc==0)
			return 1;
	if(i==n)	
	{
		return 0;
	}
	else
	{
		if(sump==0&&sumv==0&&sumc==0)
			return 1;
		else if(sump==0||sumv==0||sumc==0)
			return 0;
		else
		{
			if(find(a,b,c,sump-a[i],sumv-b[i],sumc-c[i],i+1,n)||find(a,b,c,sump,sumv,sumc,i+1,n))
			{
				return 1;
			}
			else return 0;
		}
		
	}
}

int main()
{
    int t,i,sump,sumv,sumc,n,j;
    cin>>t;
    for(i=1;i<=t;i++)
    {
    	cin>>sump>>sumv>>sumc;
    	cin>>n;
    	int a[n],b[n],c[n];
    	for(j=0;j<n;j++)
    		cin>>a[j]>>b[j]>>c[j];
    	if(find(a,b,c,sump,sumv,sumc,0,n))
    	{
    		cout<<"Case #"<<i<<": yes"<<endl;
    	}
    	else
    	{
    		cout<<"Case #"<<i<<": no"<<endl;
    	}
    	    
    }
    return 0;
}
