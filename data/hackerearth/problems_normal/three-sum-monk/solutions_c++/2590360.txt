#include <bits/stdc++.h>
#define ll long long
using namespace std;
int h[20];
ll int dp[2010][2010];
int a[2010];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int i,j,k,e;
	for(i=1;i<=18;i++)
	{
		h[i-1]=(i*i*i);
	}
	int test;
	cin>>test;
	while(test--)
	{
		int n;
		cin>>n;
		//Prefix sum
		for(i=0;i<2001;i++)
		{
			for(j=0;j<n;j++)
			{
				dp[i][j]=0;
			}
		}
		for(i=0;i<n;i++)
		{
			cin>>a[i];
			dp[a[i]][i]=1;
		}
		for(i=0;i<2001;i++)
		{
			for(j=0;j<n;j++)
			{
				if(j==0)
				continue;
				else
				dp[i][j]+=dp[i][j-1];
			}
		}
		//~ for(i=0;i<10;i++)
		//~ {
			//~ for(j=0;j<10;j++)
			//~ cout<<dp[i][j]<<" ";
			//~ cout<<endl;
		//~ }
		int p=0,t=0;
		for(i=0;i<n;i++)
		{
			for(j=(i+1);j<n;j++)
			{
				for(k=17;k>=0;k--)
				{
					e=(h[k]-(a[i]+a[j]));
					if(e<0)
					break;
					else if(e<=2000 && dp[e][n-1]>=1)
					{
						//~ cout<<e<<" "<<a[i]<<" "<<a[j]<<endl;
						//~ cout<<dp[e][2009]<<" "<<dp[e][j]<<endl;
						p+=(dp[e][n-1]-dp[e][j]);
						//~ cout<<"p is "<<p<<endl;
					}
				}
			}
		}
		//~ cout<<p<<endl;
		t=(n*(n-1)*(n-2))/6;
		double ans= (p*1.0)/(t*1.0);
		printf("%.9lf\n",ans);
	}
	return 0;
}
