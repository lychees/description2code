#include <iostream>
using namespace std;

int main()
{
    long N,Q,i,arr[200000],v,y,x;
    
    int op;
    
    cin>>N>>Q;
    
    for( i=0 ;  i < N ;  i++ )
    cin>>arr[i];
    
    for( ; Q ; Q--)
    {
    	cin >> op;
    	
    	switch(op)
    	{
    		case 1: cin>>v;
    		        for( i=0 ;  i < N , arr[i] < v ;  i++ );
    		        if(i >= N)
    		        cout<< -1 << endl;
    		        else
    		        cout<< ++i << endl;
    		        break;
    		
    		case 0: cin>>x>>y;
    		        x--;
    		        arr[x]=y;
    		        break;
    		
    	}
    }
    
    return 0;
}
