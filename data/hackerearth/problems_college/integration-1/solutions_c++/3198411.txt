#include <iostream>
using namespace std;
 
int main()
{
    int t,j=1;
    cin>>t;
    while(j<=t)
    {
    	int n,p,u,r,s,i;
    	cin>>n>>p>>u>>r>>s;
    	while(n--)
    	{
    		if(p+r<=u)
    		p=p+r;
    		else if(p-s>=0)
    		p=p-s;
    		else
    		break;
    	}
    	cout<<"Case #"<<j<<": "<<p<<endl;j++;
    }
    return 0;
}
