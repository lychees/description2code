#include<bits/stdc++.h>
using namespace std;
int main(){
	ios_base::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--){
		int a[3],i,p=0;
		for(i=0;i<3;i++){
			cin>>a[i];
		}
		sort(a,a+3);
		for(i=0;i<3;i++){
			if(!a[i]){
				p++;
			}
		}
		if(p==3){
		cout<<"0"<<endl;
		}
		else{
				if(a[1]==a[2])
			{
				cout<<2*a[1]<<endl;
			}
			else
			cout<<2*a[1]+1<<endl;
		}
	}
}
