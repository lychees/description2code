

    #include <bits/stdc++.h>
    using namespace std;
    int main()
    {
    	int t;
    	scanf(" %d",&t);
    	while(t--)
    	{
    		long long sum = 0;
    		long long n;
    		scanf(" %lld",&n);
    		n--;
    		sum = n*(n+1)*(2*n + 1);
    		sum /= 6;
    		sum += (n*(n+1))/2;
    		printf("%lld\n",sum);
    	}
    	return 0;
    }
