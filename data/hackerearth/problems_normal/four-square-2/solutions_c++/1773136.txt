#include<iostream>
#include<cmath>
using namespace std;

typedef long long ll_t;
#define PRE_CPP do { ios_base::sync_with_stdio(0);	cin.tie(0); } while(0)

ll_t find_all_mod_4_divisors(ll_t& input) {
		ll_t total, i;
		ll_t sqt = sqrt(input);
		for (total = 0, i = 1; i <= sqt; i++) {
			if (input % i == 0) {
				if (i % 4 != 0) total += i;
				if (i * i != input) {
				  if ((input / i) % 4 != 0) total += input / i;
				}
			}
		}
	 	if (i * i == input && i % 4 != 0) total += i;
	 	return total;
}

int main() {
	PRE_CPP;
	int testcase;
	ll_t input;
	cin >> testcase;
	while (testcase--) {
		cin >> input;
	    cout << find_all_mod_4_divisors(input) * 8 << "\n"; 
	}
 	return 0;
}
