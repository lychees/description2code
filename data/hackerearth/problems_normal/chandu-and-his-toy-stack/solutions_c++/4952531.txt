#include <bits/stdc++.h>
#define ll long long 
using namespace std;

int main()
{
	
	
	ll t,n,x,y,i,cnt,ans;
	cin>>t;
	while(t--)
	{
		cin>>n>>x>>y;
		ll a[n],b[n];
		for(i=0;i<n;i++)
		{
			cin>>a[i]>>b[i];
		}
		sort(a,a+n);
		sort(b,b+n);
		ans=0;
		for(i=0;i<n;i++)
		{
			if(b[i]>a[i])
			 ans+=(b[i]-a[i])*x;
			else if(a[i]>b[i])
			 ans+=(a[i]-b[i])*y;
		}
		cout<<ans<<endl;
	} 
    
    return 0;
}
