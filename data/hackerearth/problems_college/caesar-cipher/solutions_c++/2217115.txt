#include<stdio.h>
#include<string.h>
#include<iostream>
#include<math.h>
#include<stdlib.h>
#include<ctype.h>
using namespace std; 

int main() {
	int t;
	cin>>t;
	
	while(t--)
	{   char str[1100];
		int k,i,rem,j;
		cin>>k;
		cin.ignore(254,'\n');
		rem = k%26;
		gets(str);
		
		for(i=0;str[i]!='\0';i++)
		{
			if(isupper(str[i]))
			{
				j=str[i];
			   
			   if(str[i]+rem>90)
			   {
			   	str[i]='A'+rem-1-90+j;
			   }
			   else
			   {
			   str[i]=str[i]+rem;
			   }
			}
		}
		
	//	puts(str);
		for(i=0;str[i]!='\0';i++)
		{
			if(islower(str[i]))
			{
				j=str[i];
			   
			   if(str[i]+rem>122)
			   {
			      str[i]='a'+rem-1-122+j;
			   }
			   else
			   {
			   	str[i]=str[i]+rem;
			   }
			}
		}
		puts(str);
	}
	return 0;
}
