
#include<bits/stdc++.h>

using namespace std;

#define si(x) scanf("%d",&x)
#define sll(x) scanf("%lld",&x)
#define pri(x) printf("%d",x)
#define pll(x) printf("%lld",x)
#define sstr(s) scanf("%s",s)
#define nl printf("\n")
#define ll long long int
#define pii pair<int,int>
#define pLL pair<ll,ll>
#define vi vector<int>
#define pb push_back
#define mp make_pair
#define fr first
#define se second
#define mod 1000000007

ll x[200005];
ll y[200005];
ll px[200005];
ll py[200005];
ll sx=0,sy=0;

int main()
{
    int n; si(n);
    for(int i=0;i<n;++i) {
	sll(x[i]); sll(y[i]);
	sx+=x[i]; sy+=y[i];
    }
    sort(x,x+n);
    sort(y,y+n);
    for(int i=0;i<n;++i) {
	if(i==0) {
	    px[i] = x[i]; py[i] = y[i];
	}
	else {
	    px[i] = px[i-1]+x[i];
	    py[i] = py[i-1]+y[i];
	}
    }
    ll ans = 0;
    for(int i=n-1;i>0;--i) {
	ans += 1ll*(i)*x[i] - px[i-1];
	ans %= mod;
	ans += 1ll*(i)*y[i] - py[i-1];
	ans %= mod;
    }
    pll(ans); nl;
    return 0;
}
