N = int(raw_input())
result = []
str_input = raw_input()
S = str_input.split(' ')
for ci in range(0, N):
    for cj in range(0, N):
        if ci != cj and int(S[ci]) % int(S[cj]) == 0:
            result.append(S[ci])
            break

for ci in range(0, len(result)):
    S.remove(result[ci])
print " ".join(S)
