#include<iostream>

using namespace std;

int main()
{   int T, stack[5000], sum=0, top=-1;
    cin>>T;
    while(T--)
    {
        cin>>stack[++top];
        sum+=stack[top];
        if(stack[top]==0  )
        { top--;
          sum-=stack[top];
          top--;
        }
        
    }
    cout<<sum;
    return 0;
}
