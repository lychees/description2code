#include <iostream>
using namespace std;

int main()
{
    int t,n,k,rem,sum,i,srem,j;
    cin>>t;
    while(t--)
    {
        srem=2000;
        cin>>n;
        cin>>k;
        int a[n+1];
        if(k==0)
        {
            cout<<0<<"\n";
        }
        else
        {
            for(i=0;i<n;i++)
        {
            cin>>a[i];
        }
        for(i=0;i<n;i++)
        {
            for(j=i;j<n;j++)
            {
                if(i==j)
                {
                    sum=a[i];
                }
                else
                {
                    sum=a[i]+a[j];
                }
                rem=k%sum;

                if(srem>rem)
                srem=rem;
            }
        }
        cout<<k-srem<<"\n";
        }
    }

    return 0;
}
