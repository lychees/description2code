#include<bits/stdc++.h>
#define MOD 1000000007
using namespace std;
long long fib[15005],fs[15005],bs[15005];
int main()
{
    //freopen("in.txt","r",stdin);
    //freopen("outwa.txt","w",stdout);
    long long i,t,kb;
    long long l,r,sm,LOOP=15000;
    fib[0]=0;
    fib[1]=1;
    for(i=2;i<LOOP;i++)
        fib[i]=(fib[i-1]+fib[i-2])%10000;
    for(i=0,sm=0;i<LOOP;i++)
    {
        sm=(sm+fib[i])%MOD;
        fs[i]=sm;
    }
    for(i=LOOP-1,sm=0;i>=0;i--)
    {
        sm=(sm+fib[i])%MOD;
        bs[i]=sm;
    }
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld%lld",&l,&r);
        l--;r--;
        sm=0;
        //printf("%lld %lld\n",l%LOOP,r%LOOP);
        if(l/LOOP==r/LOOP)
        {
            if(l%LOOP) sm=fs[r%LOOP]-fs[(l%LOOP)-1];
            else sm=fs[r%LOOP];
            if(sm<0) sm+=MOD;
            printf("%lld\n",sm);
            continue;
        }
        if(l%LOOP)
        {
            sm=bs[l%LOOP];
            l=(l/LOOP+1)*LOOP;
        }
        sm=(sm+bs[0]*((r-l)/LOOP%MOD)%MOD)%MOD;
        sm=(sm+fs[r%LOOP])%MOD;
        printf("%lld\n",sm);
    }

}
