#include <iostream>
using namespace std;

using ull = unsigned long long;
constexpr ull mod = 1000000007;

int strips[1002] = {};
ull count[1002];

int main() {
	ios_base::sync_with_stdio(0);
    cin.tie(0);

	int n, k;
	cin >> n >> k;
	
	strips[0] = 1;
	for (int i = 0; i < k; ++i) {
		int b;
		cin >> b;
		for (int j = b; j <= n; ++j) 
			strips[j] |= strips[j - b];
	}
	
	count[0] = 1;
	for (int i = 1; i <= n + 1; ++i) {
		count[i] = count[i - 1];
		for (int j = 1; j <= i; ++j)
			if (strips[j])
				count[i] += count[i - j - 1];
		count[i] %= mod;
	}
	
    cout << count[n + 1] << '\n';
    return 0;
}
