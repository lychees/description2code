#include <iostream>
 
using namespace std;
 
int main()
{
	long long t;
	cin>>t;
	while(t--)
	{
		long long n;
		cin>>n;
		long long shop[n+5][3];
		for(long long i=0;i<n;i++)
			{
				for(long long j=0;j<3;j++)
				cin>>shop[i][j];
			}
 
		long long dp[n+5][3];
 
		for(long long i=0;i<3;i++)
		{
			dp[0][i] = shop[0][i];
		}
 
		for(long long i=1;i<n;i++)
		{
			dp[i][0] = shop[i][0] + min(dp[i-1][1],dp[i-1][2]);
			dp[i][1] = shop[i][1] + min(dp[i-1][0],dp[i-1][2]);
			dp[i][2] = shop[i][2] + min(dp[i-1][1],dp[i-1][0]);
		}
 
		cout<<min(dp[n-1][0],min(dp[n-1][1],dp[n-1][2]))<<endl;
	}
	return 0;
}
