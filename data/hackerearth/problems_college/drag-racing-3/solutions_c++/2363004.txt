#include <iostream>
#include <set>
using namespace std;

int main()
{
	long long n, k, i;
	string a;
	char aa[15], bb[15];
	multiset<string> occ;
	cin >> n;
	k = (1 << n) - 1;
	for (i = 0; i < k; i++) {
		scanf("%s%s", &aa, &bb);
		a = aa;
		occ.insert(a);
		if (occ.count(a) == n) {
			cout << a << endl;
			return 0;
		}
	}
	return 0;
}
