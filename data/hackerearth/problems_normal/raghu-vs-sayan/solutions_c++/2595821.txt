#include <iostream>
#include <algorithm>
#include <cstdio>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    long int a,b,sa=0,sb=0,counta=0,countb=0;
    int n;
    scanf("%ld%ld%d",&a,&b,&n);
    int i,s[n];
    for(i=0;i<n;i++)
    {
        scanf("%d",&s[i]);
    }
    sort (s,s+n);
    for(i=0;i<n;i++)
    {   if(a-sa<s[i] && b-sb<s[i])
        break;
        else
        {
            if(sa+s[i]<=a)
            {
            sa+=s[i];
            counta++;
            }
            if(sb+s[i]<=b)
            {
            sb+=s[i];
            countb++;
            }
        }
    }
    if(counta<countb)
    printf("Sayan Won\n");
    else
    if(counta>countb)
    printf("Raghu Won\n");
    else
    printf("Tie\n");
    }
    return 0;
}
