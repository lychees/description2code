#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,k;
    cin >> n >> k;
    vector<long long> A(n);
    for(int i=0;i<n;++i)
    	cin >> A[i];
    sort(A.begin(),A.end());
   	for(int i=1;i<A.size();++i)
   		A[i] += A[i-1];
   	int pos = upper_bound(A.begin(),A.end(),k)-A.begin();
   	cout << pos;
    return 0;
}
