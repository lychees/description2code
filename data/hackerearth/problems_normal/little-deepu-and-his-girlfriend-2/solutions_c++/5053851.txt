#include <bits/stdc++.h>

using namespace std;
bool dp[10005];

int main()
{
	int t, m, n;
	vector <int> v;
	cin >> t;
	while (t--) {
		v.clear();
		cin >> m >> n;
		for (int i = 0; i <= m + 5; i++) {
			dp[i] = false;
		}
		for (int i = 0; i < n; i++) {
			int x;
			scanf("%d", &x);
			v.push_back(x);
			dp[x] = true;
		}
		// 	for (int i = 1; i <= m; i++) {
		// 	cout << dp[i] << " ";
		// }
		sort(v.begin(), v.end());
		bool f;
		for (int i = 1; i <= m; i++) {
			if (find(v.begin(), v.end(), i) != v.end()) {
				continue;
			}
			f = false;
			for (int j = 0; v[j] <= i && j < v.size(); j++) {
				if (dp[i-v[j]] == false) {
					f = true;
					break;
				}
			}
			if (f == true) {
				dp[i] = true;
			}
		}
		if (dp[m] == true) {
			printf("Little Deepu\n");
		} else {
			printf("Kate\n");
		}
	}
	return 0;
}
