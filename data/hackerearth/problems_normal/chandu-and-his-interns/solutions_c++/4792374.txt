#include <iostream>
#include <cmath>


using namespace std;

#define ll long long int


int main()
{
    ll x,n,c,i,j;
    cin >> x;
    while(x--)
    {
        cin>>n;
        c = 2;
        i = 2;
        ll flag = 1;
        while(i<=sqrt(n))
        {
            if(n%i==0)
             c+=2;
             if(i*i==n)
               c--;
             if(c>=4)
             {
                 flag = 0;
                 break;
             }
             i++;
        }
        /*if(n==6||n==8)
          c = 4;*/
        if(flag==0||c>=4)
          cout<<"YES"<<endl;
        else
          cout<<"NO"<<endl;

    }
}
