#include <iostream>
using namespace std;

typedef unsigned long long ull;

int main()
{
    int t;
    cin >> t;
    while (t--)
    {
    	ull a, b;
    	cin >> a >> b;
    	while (b^=a^=b^=a%=b);
    	if (a == 1) cout << "Is a Co-Prime" << endl;
    	else cout << "Not a Co-Prime" << endl;
    }
    return 0;
}
