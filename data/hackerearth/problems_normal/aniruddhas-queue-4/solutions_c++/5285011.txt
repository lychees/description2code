#include<bits/stdc++.h>
  
using namespace std;         

#define ll	long long int
int main()
{
	int t;
cin >> t;
	while(t--)
	{
		int n;
		cin >> n;
		ll ar[n],cnt=0;
		for(int i=0;i<n;++i)
		{
			scanf("%lld",&ar[i]);
			cnt+=ar[i];
		}
		ll m;
		cin >> m;
		m%=cnt;
		int ans;
		if(m==0)
		{
			for(int i=n-1;i>=0;--i)
			{
				if(ar[i]!=0)
				{
					ans=i;
					break;
				}
			}
		}
		else
		{
			for(int i=0;i<n;++i)
			{
				m-=ar[i];
				if(m<=0)
				{
					ans=i;
					break;
				}
			}
		}
		cout << ans + 1 << "\n";
	}
	return 0;
}
