#include<bits/stdc++.h>
using namespace std;
long long n,q,arr[100010],size[100010],a,b,mx,mn;
set<pair<long long,long long> > s;
set<pair<long long,long long> >::iterator it;
void init()
{
	for(int i=1;i<=n;i++)
	{
		arr[i]=i;
		size[i]=1;
		s.insert(make_pair(1,i));
	}
}
int root(int x)
{
	while(x!=arr[x])
	{
		arr[x]=arr[arr[x]];
		x=arr[x];
	}
	return x;
}
void unn(int a, int b)
{
	int root_a=root(a);
	int root_b=root(b);
	if(root_a!=root_b)
	{
		s.erase(make_pair(size[root_a],root_a));
		s.erase(make_pair(size[root_b],root_b));
		
		size[root_a]+=size[root_b];
		arr[root_b]=root_a;
		s.insert(make_pair(size[root_a],root_a));
	}
}
int main()
{
	ios::sync_with_stdio(false);
	//cin.tie(NULL);
	cin>>n>>q;
	init();
	while(q--)
	{
		cin>>a>>b;
		unn(a,b);
		/*
		cout<<"----------------------------------\n";
		for(it=s.begin();it!=s.end();it++)
		{
			cout<<(*it).first<<" ";
		}
		cout<<endl;
		cout<<"----------------------------------\n";
		*/
		it=s.begin();
		mn=(*it).first;
		it=s.end();it--;
		mx=(*it).first;
		cout<<mx-mn<<endl;
	}
	return 0;
}
