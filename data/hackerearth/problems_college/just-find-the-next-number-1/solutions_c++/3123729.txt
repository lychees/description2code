#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,flag,pb,j;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	int i=n;
    	j=n;
    	while(1)
    	{
    		flag=0;
    		pb=0;
    		while(i>0)
    		{
    			if(i&1)
    			{
    				if(pb)
    				{
    					flag=1;
    					break;
    				}
    				else pb=1;
    			}
    			else 
    			pb=0;
    			i>>=1;
    			
    		}
    		if(!flag)
    		{
    			cout<<j<<"\n";
    			break;
    		}
    		else
    		{
    			j++;
    			i=j;
    		}
    	}
    }
    return 0;
}
