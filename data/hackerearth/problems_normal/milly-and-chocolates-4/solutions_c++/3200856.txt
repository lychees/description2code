#include <iostream>
#include<iomanip>
using namespace std;

int main()
{
    int N;
    cin>>N;
    
    for(int i=0;i<N;i++){
    	int M;
    	cin>>M;
    	int data[M];// = new int(M);
    	double res = 0;
    	for(int j=0;j<M;j++){
    		cin>>data[j];
    		res += data[j];
    	}
    	res = res/M;
    	//printf("%.8lf",res);
    	cout<<setiosflags(ios::fixed)<<setiosflags(ios::showpoint)<<setprecision(8)<<res<<endl;
    	//delete[] data;
    }
    
    return 0;
}
