def area(x1, y1, x2, y2, x3, y3):

   return abs((x1*(y2-y3) + x2*(y3-y1)+ x3*(y1-y2))/2.0)

t = int(raw_input())
for i in range(t):
	coord = map(int, raw_input().split())
	x1 = coord[0]
	y1 = coord[1]
	x2 = coord[2]
	y2 = coord[3]
	x3 = coord[4]
	y3 = coord[5]
	x4 = coord[6]
	y4 = coord[7]
	A = area(x1, y1, x2, y2, x3, y3)
	A1 = area(x4, y4, x2, y2, x3, y3)
	A2 = area(x1, y1, x4, y4, x3, y3)
	A3 = area(x1, y1, x2, y2, x4, y4)
	if (A == A1 + A2 + A3):
		print 'INSIDE'
	else:
		print 'OUTSIDE'
