def countRow(totRow, initRow, destRow):
    possList = [abs(initRow-destRow), abs(abs(initRow -0)+abs(totRow-destRow)), abs(abs(initRow-totRow)+destRow)]
    return min(possList)
     
def countColumn(totColumn, initColumn, destColumn):
    possList = [abs(initColumn-destColumn), abs(abs(initColumn -0)+abs(totColumn-destColumn)), abs(abs(initColumn-totColumn)+destColumn)]
    return min(possList)
