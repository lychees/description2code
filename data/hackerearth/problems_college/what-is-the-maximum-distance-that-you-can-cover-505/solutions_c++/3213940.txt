#include<bits/stdc++.h>
using namespace std;
int arr[100005];
int fuel[100005];
int main()
{
    int n,k,i;
    cin>>n;
    for(i=0;i<n;i++)
      cin>>arr[i];
    for(i=0;i<n;i++)
        cin>>fuel[i];
    cin>>k;

    double maxx=0.0,ans=0.0;

    for(i=0;i<n;i++)
      {
          ans=((k*1.0)/(fuel[i]*1.0))*arr[i]*1.0;
          if(ans>maxx)
            maxx=ans;
      }
   printf("%.3lf", ((signed long)(maxx * 1000) * 0.001f));
    return 0;
}
