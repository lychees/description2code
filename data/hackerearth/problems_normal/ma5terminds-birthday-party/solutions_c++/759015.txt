#include<iostream>
#include<cstdio>
using namespace std;
	int arr[1000001];
	int prime[1000001]={0};
int main()
{
	 int i,j;
	 prime[1]=0;
	 for(i=2;i<=1000;i++)
	 {
 		if(!prime[i])
 		 for(j=i*i;j<=1000000;j=j+i)
 		 prime[j]=1;
 	}
 
 	arr[0]=0;
 	arr[1]=0;
 	  for(i=2;i<=1000000;i++)
 	  {
                 if(!prime[i])
                 arr[i]=1+arr[i-1];
                 else
                 arr[i]=arr[i-1];
                 if(i==4)
                 	arr[i]+=1;
  	 	  
  	 }
 	int t,l,r;
 	scanf("%d",&t);
 	while(t--)
 	{
	 	scanf("%d%d",&l,&r);
	 	printf("%d\n",((r-l+1)-(arr[r]-arr[l-1])));
	 }
	 return 0;
	 
}
