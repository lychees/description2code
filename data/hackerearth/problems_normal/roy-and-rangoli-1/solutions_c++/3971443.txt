#include <iostream>
using namespace std;
typedef long int li;
const li SIZE=10000001;
bool primes[SIZE];	

void prima()
{
	primes[4]=true;
	primes[1]=true;
	for(li i=2;i<SIZE;i++)
	{
		if(!primes[i])
		{
		for(li j=3;j<SIZE/3 && (i*j)<SIZE;j++)
		{
		primes[i*j]=true;	
		}
	}
	}
}

int main()
{
    prima();
    li N;
    cin>>N;
    li total=0;
    li max_sum=(N-1)*2; //prime sum cannot be greater than this cus in that case the 2 numbers making the sum will be                 // greater than (N-1).
    if(N==1)
    {
    cout<<0<<endl;
    }
    else if(N==2)
    {
    cout<<1<<endl;
    }
    else
    {
    	li i;
    	total=3;
        for( i=1;i<N;i+=2)
        {
    	if(!primes[i])
    	{
    	    total+=i+1;
    	}
        }
        for(li j=i;j<=max_sum;j+=2)
        {
        	li t=0;
        	if(!primes[j])
        	{
        		t=j-(N-1);
        		total = (total + (j+1)) - (t*2);
        	}
        }
        cout<<total%1000000007<<endl;
    }
    return 0;
}
