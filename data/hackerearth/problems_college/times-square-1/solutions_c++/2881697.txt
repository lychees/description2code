int F[N] , cnt[N];

void sieve(int n) {
	frep(i , 1 , n) {
		for(int j = i;j <= n;j += i) {
			dp[j] += j/i;
			cnt[j]++;
		}
	}
}

ll brute(int n) {
	ll ans = 0;
	frep(i , 1 , n) {
		cout << gcd(n,i) << " " << n/gcd(n,i) << "\n";
		ans+= n/gcd(n,i);
	}
	return ans;
}
