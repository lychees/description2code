using namespace std;

int t, n, m, cnt, visited[100010];
vector<int>adj[100010];

void dfs(int x)
{
    visited[x] = 1;
    for(int i = 0; i < adj[x].size(); i++)
        if(!visited[adj[x][i]])
            dfs(adj[x][i]);
}

int main(){
//freopen("input.txt", "r", stdin);
ios_base::sync_with_stdio(0);
cin.tie(0);

cin>>t;
while(t--)
{
    cin>>n>>m;
    cnt = 0;
    for(int i = 0; i <= n; i++)
        adj[i].clear();
    memset(visited, 0, sizeof visited);
    while(m--)
    {
        int u, v;
        cin>>u>>v;
        adj[u].pb(v);
        adj[v].pb(u);
    }
    for(int i = 0; i < n; i++)
        if(!visited[i])
        {
            ++cnt;
            dfs(i);
        }
    cout<<cnt<<endl;
}

return 0;}
