#include <iostream>
using namespace std;
#include<vector>
#include<limits.h>
int main()
{
    int t,i,j,n,m,x,y,c;
    cin>>t;
    while(t--){
    	cin>>n>>m;
    	int d[n+1];
    	vector< pair< int , pair<int,int> > > g;
    	for(i=0;i<m;i++){
    		cin>>x>>y>>c;
    		g.push_back(make_pair(x,make_pair(y,c*-1)));
    	}
    	d[1]=0;
    	for(i=2;i<=n;i++){
    		d[i]=INT_MAX;
    	}
    	for(j=0;j<n-1;j++){
    	for(i=0;i<m;i++){
    		int u=g[i].first;
    		int v=g[i].second.first;
    		int val=g[i].second.second;
    		if(d[u]!=INT_MAX)
    		d[v]=min(d[v],d[u]+val);
    	}
    	}
    	int f=0;
    	for(i=0;i<m;i++){
    		int u=g[i].first;
    		int v=g[i].second.first;
    		int val=g[i].second.second;
    		if(d[u]!=INT_MAX && d[v]>d[u]+val){
    			f=1;
    			break;
    		}
    	}
    	if(f){
    		cout<<"Yes"<<endl;
    	}
    	else{
    		cout<<"No"<<endl;
    	}
    }
    return 0;
}
