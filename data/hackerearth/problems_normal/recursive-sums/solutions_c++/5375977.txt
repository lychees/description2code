#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int d,t,m;
    long long l,sum;
    cin>>t;
    while(t--){sum=0;
    	cin>>m;
    	while(m--){
    		cin>>l>>d;
    		l*=d;
    		if(l/10!=0){
    			while(l){
    				sum+=l%10;
    				if(sum/10!=0){
    					int p=sum;sum=0;
    					while(p){
    						sum+=p%10;
    						p/=10;
    					}
    				}
    				l/=10;
    			}
    			
    		}
    		else{
    			sum+=l;
    			if(sum/10!=0){
    					int p=sum;sum=0;
    					while(p){
    						sum+=p%10;
    						p/=10;
    					}
    		}
    	}
    	}
    	cout<<sum<<endl;
    }
    
    return 0;
}
