#include <iostream>
using namespace std;

int main()
{
	long long int x, y;
    cin >> x;
    long long int a[3];
    a[0] = 0;
    a[1] = 0;
    a[2] = 0;
    
    for (long long int i = 0 ; i< x ;i++){
    	cin >> y;
    	a[i% 3] += y;
    }
    cout << a[0] << ' ' << a[1] << ' ' << a[2] << endl;
    return 0;
}
