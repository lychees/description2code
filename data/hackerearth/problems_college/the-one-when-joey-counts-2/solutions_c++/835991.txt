#include <iostream>
#include <math.h>
using namespace std;

void find_sring(long long int n)
{  
	long long int str[10000];
	long long int i,j;
	long long int no_of_digits=1;
	long long int past_max;
	long long int temp=0;

	while(temp < n)
	{
		past_max=temp;
		temp+=pow(3,no_of_digits);
		no_of_digits++;
	}
    temp=past_max;
    --no_of_digits;

    for(i=0;i<no_of_digits;i++)
    {
    	j=0;
    	temp=past_max;

    	while(temp < n)
    	{
    		past_max=temp;
    		temp+=pow(3,no_of_digits-i-1);
    		++j;
    	}
    	str[i]=j+1;
    }
    for(i=0;i<no_of_digits;i++)
    cout<<str[i];
}

int main()
