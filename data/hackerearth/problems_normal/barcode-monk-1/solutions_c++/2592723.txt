#include <bits/stdc++.h>
#define MOD 1000000007
using namespace std;
typedef long long ll;
#define MOD 1000000007

ll a[1005];
ll ans[1005];

ll bsh[2005];
ll n, k;

vector<ll> brsh;
ll bsz;

ll dp[1005][2];

void pre() {
	bsh[0] = 1;;
	
	for (ll  i = 0; i < k; i++) {
		for (ll j = 0; j < 1005; j++)
			if (j -a[i] >= 0)
				bsh[j] |= bsh[j-a[i]];
	}
	for (ll i = 1; i <= n; i++)
		if (bsh[i])
			brsh.push_back(i);
	bsz = brsh.size();

}

void calc() {
	dp[0][0] = 1;
	dp[0][1] = 0;

	for (ll  i = 1; i <= n; i++) {
		dp[i][0] = (dp[i-1][0] + dp[i-1][1]) %MOD;
		
		ll ans = 0;

		for (ll j = 0; j <  bsz; j ++) {
			if (i - brsh[j] < 0)
				break;
			dp[i][1] = (dp[i][1] + dp[i-brsh[j]][0]) % MOD;
		}
	}
}

int main()
{
	scanf("%lld%lld",  &n, &k);
	for (int i = 0; i < k; i++) {
		scanf("%lld", &a[i]);
	}
	pre();

	calc();
	cout <<(dp[n][0] + dp[n][1])%MOD << endl;

	return 0;

}
