#include <iostream>
using namespace std;
int fact(int );
int main()
{
    int n,a,b,i,sum;
    cin>>n;
    while(n--)
    {
    	cin>>a>>b;
    	sum=0;
    	if(a>=5&&b>=5)
    	cout<<"0"<<endl;
    	else
    	{
    		if(a<5&&b>=5)
    		{
    			for(i=a;i<=4;i++)
    		    {
    				sum=sum+fact(i);
    		    }
    		    cout<<sum%10<<endl;
    		}
    		else
    		{
    			for(i=a;i<=b;i++)
    			{
    				sum=sum+fact(i);
    			}
    			cout<<sum%10<<endl;
    		}
    	}
    }
    return 0;
}
int fact(int y)
{
	if(y==1)
	return 1;
	else
	return y*fact(y-1);
}
