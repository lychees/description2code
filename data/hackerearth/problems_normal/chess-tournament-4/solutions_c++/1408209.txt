#include <iostream>
#define SIZE 1025

using namespace std;

int m[SIZE][SIZE];

int main() {
    std::ios_base::sync_with_stdio(false);
    int n;
    cin>>n;
    
    for (int i=2 ; i<=(1<<n) ; i++) {
        for(int j=1 ; j<i ; j++) {
            cin>>m[i][j];
        }
    }
    int count = 1<<n;
    int players[(1<<n) + 5];
    for (int i=1 ; i<=(1<<n) ; i++) {
        players[i-1] = i;
        // cout<<players[i-1]<<" ";
    }
    // cout<<"\nHello\n";
    while (count > 1) {
        int index = 0;
        for (int i=0 ; i<count ; i+=2) {
            players[index++] = m[players[i+1]][players[i]]==1 ? players[i+1] : players[i];
        }
        count/=2;
    }
    cout<<players[0]<<"\n";
    
    return 0;
}
