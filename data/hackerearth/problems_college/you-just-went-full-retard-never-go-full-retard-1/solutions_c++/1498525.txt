#include <iostream>
#include<cmath>
using namespace std;
int prime(long long int n)
{
	if(n<2) return 0;
	long long int i,	l=sqrt(n);
	for(i=2;i<=l;i++) if(n%i==0) return 0;
//	for(i=l;i>sqrt(l);i--) if(n%i==0) return 0;
	return 1;
	
}
int main()
{
 	int T;
	long long int n;
    cin>>T;
    while(T--)
    {
    	cin>>n;
    	if(prime(n)==1) cout<<"YES"<<endl;
    	else cout<<"NO"<<endl;
    	}
    return 0;
}
