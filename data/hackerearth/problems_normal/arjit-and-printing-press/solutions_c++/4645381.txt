#include <iostream>
#include <cstring>
#include <algorithm>
using namespace std;

int main()
{
    string w, r;
    int t,i,x,lr,lw;
    cin >> t;
    while (t--)
    {
    	cin >> w;
    	cin >> r;
    	x = 0;
    	lr = r.size();
    	lw = w.size();
    	sort(r.begin(),r.end());
    	for (i = 0; i < lw; i++)
    	{
    		if (x < lr &&  w[i] > r[x])
    		{
    			w[i] = r[x];
    			x++;
    		}
    		if (x == lr) break;
    	}
    	cout << w << endl;
    }
    return 0;
}
