testca = int(raw_input())
for x in range(testca):
    a, b, c, d, k = map(int, raw_input().split())
    def f(t): return a * (t ** 3) + b * (t ** 2) + c * t + d
    t = 0
    r = k
    l = 0
    while l < r:
        mid = (l + r) // 2
        if f(mid) == k:
            t = mid
            break
        elif f(mid) < k:
            t = mid
            l = mid + 1
        elif f(mid) > k:
            r = mid

    print t
