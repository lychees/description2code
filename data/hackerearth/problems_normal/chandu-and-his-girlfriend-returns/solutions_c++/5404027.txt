#include <bits/stdc++.h>
using namespace std;

int main()
{
    long t,n,m,i,j;
    cin>>t;
    while(t--)
    {
    	cin>>n>>m;
    	long long c[n+m+1];
    	for(i=0;i<n+m;i++)
    	cin>>c[i];
    	n=n+m;
    	sort(c,c+n);
    	reverse(c,c+n);
    	for(i=0;i<n;i++)
    	{
    	cout<<c[i]<<" ";
    	}
    	cout<<endl;
    }

	return 0;
}
