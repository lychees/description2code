inp=raw_input()
l=len(inp)
dp = [[0 for x in xrange(26)]for y in xrange(l+1)]

for i in xrange(26):
	dp[l][i]=l+1
for i in xrange(l-1,-1,-1):
	for j in xrange(26):
		dp[i][j]=dp[i+1][j]
	dp[i][ord(inp[i])-ord('a')]=i

t=int(raw_input())
while t:
	t-=1
	s=raw_input()
	m,pos,flag=len(s),0,True
	for i in xrange(m):
		if dp[pos][ord(s[i])-ord('a')] >=l:
			flag=False
			break
		else:
			pos=dp[pos][ord(s[i])-ord('a')]
	if flag:
		print "Yes"
	else:
		print "No"
