#include<iostream>
using namespace std;
main()

{
   long int a,b[1000],i,j,t,cnt=0,k;
    cin>>t;
    for(i=0;i<t;i++)
    {
        cin>>a;
        for(j=0;j<a;j++)
        {
            cin>>b[j];
        }
        for(j=0;j<a;j++)
        {
            for(k=j+1;k<a;k++)
            {
               
                if((b[j]+b[k])%3==0)
                {
                    
                    cnt++;
                }
            }
        }
        cout<<cnt<<endl;
        cnt=0;
    }
}
