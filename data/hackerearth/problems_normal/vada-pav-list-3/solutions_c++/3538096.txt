#include <iostream>
#include <set>
#include <string>

using namespace std;

void solve() {
	int N;
	scanf("%d\n", &N);
	set<string> st;
	char buff[10];
	while(N--) {
		string temp(gets(buff));
		st.insert(temp);
	}
	printf("%d\n", st.size());
	for (const string& str : st) {
		puts(str.c_str());
	}
	
}

int main() {
	int T = 1;
	// scanf("%d", &T);
	while(T--) {
		solve();
	}
	return 0;
}
