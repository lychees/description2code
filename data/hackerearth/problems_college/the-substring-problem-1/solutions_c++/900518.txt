#include <iostream>
#include <string.h>
using namespace std;

int main()
{
    int test; cin>>test;
    
    while(test>0)
    {
    	
    	--test;
    	
    	int n; cin>>n;
    	
    string  s[100];
    
    for(int i=0;i<n;++i)
    cin>>s[i];
    
    char ansc;
    int max1=0;
    
    
    for(int i=97;i<=122;++i)
    {
    	
    	char c = i;
    	int maxc=0;
    	
    	for(int j=0;j<n;++j)
    	{
    		if(s[j].find(c)!=string::npos)
    		{
    			maxc++;
    		}
    		
    	}
    	
    	if(maxc>max1)
    	{
    		ansc = c;
    		max1=maxc;
    	}
    	
    	
    }
    
    	cout<<ansc<<endl;
    	
    	
    }
    
    
    return 0;
}
