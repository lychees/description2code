#include <iostream>
#include <stdio.h>
#include <string.h>
using namespace std;
#define REP(i,a,b)  for (int i = int(a); i < int(b); i++)
int dp[101][257];
int D, I, M, N;
int a[100];

int sol() {
    memset(dp, 0, sizeof(dp));
    int ret = N * D;
    for (int i = N-1; i >= 0; i--)
    REP(j, -1, 256) {
        int& x = dp[i][j+1];
        x = dp[i+1][j+1] + D;
        if (j == -1) {
            REP(t, 0, 256)
                x = min(x, dp[i+1][t+1] + min(abs(t-a[i]), I + D));
        } else {
            REP(t, 0, 256) {
                if (!M) {
                    x = min(x, dp[i+1][j+1] + min(abs(a[i]-j), I + D));
                } else {
                    int ic = max(0, (abs(t-j) - 1) )/ M *I; //max((abs(t-j) + M - 1) / M -1, 0) * I;
                    x = min(x, dp[i+1][t+1] + ic + abs(t-a[i]));
                    x = min(x, dp[i+1][t+1] + ic + I + D);
                }
            }
        }
    }
    ret = min(ret, dp[0][0]);
    return ret;
}

int main() {
    int T;
    cin >> T;
    REP(t, 1, T+1) {
        cin >> D >> I >> M >> N;
        REP(i, 0, N)
            cin >> a[i];
        cout << "Case #" << t << ": " << sol() << endl;
    }
    return 0;
}
