#include <iostream>
#include<map>
#include<vector>
using namespace std;
int power(int n)
{
	if(n==0)
		return 1;
	return 2 * power(n-1);
}
int main()
{
    int n;
    cin>>n;
    vector<string> winner;
    map<string, int> win_map;
    map<string, int> loser;
    int till = power(n);
    for(int i=0;i<till;i++)
    {
    	string s1, s2;
    	cin>>s1>>s2;
    	if(loser.find(s2)==loser.end())
    	{
    		//cout<<"not found"<<s2;
    		loser[s2] = 1;
    	}
    	if(win_map.find(s1)==win_map.end())
    	{
    		
    		//cout<<"Winner"<<s1;
    		win_map[s1] = 1;
    		winner.push_back(s1);
    	}
    }
   // cout<<winner.size();
    string out = "";
    for(int j=0;j<winner.size();j++)
    {
    	if(loser.find(winner[j])==loser.end())
    	{
    		out = winner[j];
    		break;
    	}
    }
    cout<<out<<endl;
    return 0;
}
