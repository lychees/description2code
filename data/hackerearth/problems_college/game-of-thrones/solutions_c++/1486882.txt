#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() 
{
    int t,n,ar[100000],i,count;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(i=0;i<n;i++)
    	{cin>>ar[i];}
    	sort(ar,ar+n);
    	count=0;
    	for(i=0;i<n;i++)
    	{
    		if(ar[i]==ar[i+1])
    		{count=count+1;}
    	}
    cout<<count<<endl;
    }
    return 0;
}
