using namespace std;

int main(){
	int sz,n,i,ans,flag,count;
	int dp[100011];
	string s;
	cin >> n;
	while(n--){
		cin >> s;
		sz = s.size();
		memset(dp,0,sizeof(dp));
		dp[sz] = 1;
		count = 1;
		if(s[sz-1] == 'R'){
			dp[sz-1] = 1;
			count = 0;
		}
		if(!(s[sz-1] == 'Y' || s[sz-1] == 'R')){
			cout << "NO" << endl;
			continue;
		}
		for(i=sz-2;i>=0;i--){
			if(!(s[i] == 'R' || s[i] == 'Y') || count == 3){
				dp[0] = 0;
				break;
			}
			flag = 0;
			if(s[i] == 'R' && count  == 0){
				flag = 1;
			}
			else if(i + 1 < sz && dp[i+2] && s[i] == 'R' && s[i+1] == 'Y' && count <= 1){
				flag = 1;
			}
			else if(i + 2 < sz && dp[i+3] && s[i] == 'R' && s[i+1] == 'Y' && s[i+2] == 'Y' && count <= 2){
				flag = 1;
			}
			if(flag){
				dp[i] = 1;
				count = 0;
			}
			else
				count++;
		}
		if(dp[0])
			cout << "YES" << endl;
		else
			cout << "NO" << endl;
	}
	return 0;
}
