#include<bits/stdc++.h>
using namespace std;
int main ()
{	
	long long int T,comp=1,n,m,count,left;cin>>T;	
	while(T--)
	{count=0;comp=1;
	cin>>n>>m;
	if(n!=1)
	{
	while(comp<n)
	{
	if(comp<=m)
		{count++;comp*=2;}
	else
	{
	if(m<comp)
		{
		left=n-comp;
		if(left%m==0)
		{count+=left/m;}
		if(left%m!=0)
		{count+=(left/m)+1;}
		break;	
		}
	}
	
	}
	cout<<count+1<<endl;}
	if(n==1)
	cout<<"0\n";
	}
return 0;
}
