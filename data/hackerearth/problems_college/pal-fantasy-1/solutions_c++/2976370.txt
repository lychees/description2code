#include<bits/stdc++.h>
using namespace std;
int  ispalin(string str5)
{
	string str6;
	str6 = str5;
	reverse(str5.begin(),str5.end());
	if(str5 == str6)
	{
		return 0;
	}
}


int main()
{
	int t;
	cin>>t;
	while(t--)
	{
	string str;
	cin>>str;
	int n = ispalin(str);
	if(n == 0)
	{
		cout<<str.length()<<"\n";
		continue;
	}
	int i;
	int l = str.length();
	for(i=1;i<l;i++)
	{
		string  str1;
		str1 = str.substr(0,i);
		string str2;
		reverse(str1.begin(),str1.end());
		str2 = str + str1;
		int t=ispalin(str2);
		if(t==0)
		{
			cout<<str2.length()<<"\n";
			break;
		}
	}
}
	
}
