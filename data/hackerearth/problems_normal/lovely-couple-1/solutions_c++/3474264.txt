#include <iostream>
#include <cstring>
#include <cmath>
using namespace std;

int cntprime[1005];
bool prime[1005];


int gcd(int a ,int b)
{
	if(b==0) return a;
	else return gcd(b,a%b);
}

void calPrime()
{
	memset(prime,true,sizeof(prime));
	prime[1]=false;
	for(int i=2;i*i<=1000;i++)
	{
		if(prime[i])
		{
		 for(int j=i*2;j<=1000;j+=i)
		   prime[j]=false;
		}
	}
  	for(int i=1;i<=1000;i++)
  	{
  		int n=i,cnt=0;
  		if(n%2==0) cnt++;
  		while(n%2==0) n=n/2;
  		for(int j=3;j<=sqrt(i);j+=2)
  		{
  			if(n%j==0) cnt++;
  			while(n%j==0) n=n/j;
  		}
  		if(n>2) cnt++;
  		cntprime[i]=cnt;
  	}
}

int main()
{
   int t;
   cin>>t;
   calPrime();
   for(;t;t--)
   {
   	int a,b;
   	cin>>a>>b;
   	int g=gcd(a,b);
   	if(prime[cntprime[a]+cntprime[b]-cntprime[g]]==true) cout<<"Yes"<<endl; else cout<<"No"<<endl;
   }
    return 0;
}
