#include <iostream>
using namespace std;
bool isprime[10000001];
 
int main()
{
	int t,n,i,j;
	for (i = 2; i <= 10000000; i++) {
		isprime[i] = true;
	}
	for (i = 2; i*i <= 10000000; i++) {
		if (isprime[i]) {
			for (j = i*i; j <= 10000000; j+=i) {
				isprime[j] = false;
			}
		}
	}
	isprime[0]=false;
	isprime[1]=false;
	scanf("%d",&n);
	int count = 0;
	int index = -1;
	for(int i=2;i<=10000000;i++){
		if(isprime[i])
			count++;
		if(count==n){
			index = i;
			break;
		}
	}
	if(index!=-1)
		printf("%d\n",index);
}
