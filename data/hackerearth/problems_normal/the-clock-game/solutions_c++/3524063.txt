#include <bits/stdc++.h>

using namespace std;

int main()
{
		int n;
		cin>>n;
		int period[n];
		for(int i=0; i<n;i++)
		{
			cin>>period[i];
		}

		int p;
		cin>>p;
		vector<pair<int,int> > v;
		for(int i=0;i<p;i++)
		{
			int b,ii;
			cin>>b>>ii;
			v.push_back(make_pair(b,ii));

		}
		

		int beep[100005];
		int intensity[100005];
		for(int j=0;j<100005 ;j++)
		{
			intensity[j] = 0;
			beep[j] = 0;
		}

		for(int i=0; i<n; i++)
		{
			for(int j=period[i];j<100005 ;j = j+ period[i])
			{
				intensity[j]++;
				beep[j] = 1;
			}
		}
		int maxi[100005];
		maxi[0] = intensity[0];
		for(int j=1;j<100005 ;j++)
		{
			
			beep[j] = beep[j-1] + beep[j];
			maxi[j] = max(intensity[j],maxi[j-1]);
		}


		int q;
		cin>>q;
		while(q--)
		{	
			int i,ti;
			cin>>i>>ti;
			i--;
			//cout<<v[i].first<<" "<<v[i].second<<endl;
			//cout<<beep[ti]<<" "<<maxi[ti]<<endl;

			if((v[i].second < maxi[ti]) || (v[i].first < beep[ti]))
			{
				cout<<"Awake"<<"\n";
			}
			else
				cout<<"Asleep"<<"\n";
		}




	return 0;
}
