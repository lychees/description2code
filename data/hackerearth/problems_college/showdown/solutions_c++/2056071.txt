#include <bits/stdc++.h>
using namespace std;
const int N=20;
int n;
int a[N],b[N];
int f(int pt) {
	sort(a,a+n,greater<int>());
	sort(b,b+n);
	sort(b,b+pt+1,greater<int>());
	int r=0,rv=INT_MIN;
	for(int i=0;i<n;++i) {
		if(a[i]>b[i]) r+=100;
		else if(a[i]<b[i]) r-=0;
		else r+=50;
		rv=max(rv,r);
	}
	return rv;
}
int main() {
	ios_base::sync_with_stdio(0);
	cin.tie(0);
	cin>>n;
	for(int i=0;i<n;++i) cin>>a[i];
	for(int i=0;i<n;++i) cin>>b[i];
	int ans=INT_MIN;
	for(int i=0;i<n;++i) {
		ans=max(ans,f(i));
	}
	cout<<ans<<"\n";
}
