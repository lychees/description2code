#include <bits/stdc++.h>
using namespace std;


    
    

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
   int t;
	cin>>t;
	while(t--)
	{

		long long int a, b;
		cin>>a>>b;
		long long int ans = a;
		for(long long int i=a+1; i<=b; i++)
		{

			ans &=i; 
		}
		cout<<ans<<endl;
	}

	
    return 0;
}
