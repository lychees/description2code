#include <bits/stdc++.h>
using namespace std;
#define pb push_back
#define mp make_pair
#define ll long long
#define vi vector<int>
#define all(x) x.begin(),x.end()
#define ii pair<int,int>

ll mod = 1e9 + 7;
const int maxn = 1e6 + 7;

ll dp[maxn][10];


int main() {
	ios::sync_with_stdio(0);
	cin.tie(0);

	dp[1][0] = 0;
	for (int i = 1; i < 10; ++i) dp[1][i] = 1;
	for (int i = 2; i < maxn; ++i) {
		for (int j = 0; j < 10; ++j) {
			ll ans = 0;
			if (j > 0) ans += dp[i - 1][j - 1];
			if (j < 9) ans += dp[i - 1][j + 1];
			dp[i][j] = ans%mod;
		}
	}

	int t; cin >> t;
	while (t--) {
		int n; cin >> n;
		ll ans = 0;
		if (n == 1) ans = 10;
		else {
			for (int i = 0; i < 10; ++i) {
				// ans += solve(n, i);
				ans += dp[n][i];
				ans %= mod;
			}
		}
		cout << ans << endl;
	}


	return 0;
}
