#include <bits/stdc++.h>
using namespace std;

vector<int> edges[210];
int color[210];
bool visited[201];
int main()
{
    int t,n,e,a,b;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d %d",&n,&e);
    	for(int i=0;i<n;++i){ edges[i].clear(); color[i]=0; visited[i]=0;}
    	for(int i=1;i<=e;++i)
    	{
    		scanf("%d %d",&a,&b);
    		edges[a].push_back(b);
    		edges[b].push_back(a);
    	}
    	color[0]=1;
    	queue<int> q;
    	visited[0]=1;
    	q.push(0);
    	bool flag=false;
    	while(!q.empty())
    	{
    		int p=q.front();
    		q.pop();
    		for(int i=0;i<edges[p].size();++i)
    		{
    			if(visited[edges[p][i]] && color[edges[p][i]]==color[p]){
    				flag=true;
    				break;
    			}
    			if(!visited[edges[p][i]])
    			{
    				visited[edges[p][i]]=1;
    				q.push(edges[p][i]);
    				color[edges[p][i]]=3-color[p];
    			}
    		}
    		if(flag) break;
    	}
    	if(flag) printf("NOT BICOLORABLE.\n");
    	else printf("BICOLORABLE.\n");
    }
    return 0;
}
