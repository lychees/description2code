#include <bits/stdc++.h>
using namespace std;

int main()
{
	ostringstream oss;
	string ss, rs;
	int t;
	cin >> t;
	
	while(t--) {
		cin >> rs;
		ss = rs;
		reverse(rs.begin(), rs.end());
		int sz = ss.size();
		bool isFunny = true;
		for (int i = 1; i < sz; i++) {
			if (abs(ss[i] - ss[i-1]) != abs(rs[i] - rs[i-1])) {
				isFunny = false;
				break;
			}
		}
		oss << (isFunny ? "Funny" : "Not Funny") << "\n";
	}
	
	cout << oss.str();
    return 0;
}
