#include <iostream>
using namespace std;

int main()
{
	int t; cin>>t;
    for(int i=1;i<=t;i++){
    	long long l,p,c; cin>>l>>p>>c;
    	long long cnt = 0;
    	long long j=l;
    	while(j<p){
            j*=c;
            ++cnt;
        }
        long long ans=0;
        while((1<<ans)<cnt)
            ++ans; 
            
        printf("Case #%d: %d\n",i,ans);
    }
    //cout << "Hello World!" << endl;
    return 0;
}
