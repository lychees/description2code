#include<bits/stdc++.h>
using namespace std;
char a[501],b[501];
main()
{
	int n,la,lb,k,i,j,count;
	scanf("%d",&n);
	while(n--)
	{
		count=0;
		scanf("%s%s",a,b);
		la=strlen(a);
		lb=strlen(b);
	//	k=la<lb?la:lb;
		if(la<=lb)
		{
			for(i=0;i<la;i++)
			{
				if(a[i]==b[i])
				count++;
			}
			printf("%d\n",count);
		}
		else
		{
			for(i=0;i<lb;i++)
			{
				if(a[i]==b[i])
				count++;
			}
			printf("%d\n",count);
		}
		for(i=0;i<lb;i++)
		b[i] = '\0';
		for(i=0;i<la;i++)
		a[i] = '\0';
	}
}
