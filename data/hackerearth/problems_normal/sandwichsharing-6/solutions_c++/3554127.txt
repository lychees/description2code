#include <iostream>
#include <cmath>
using namespace std;

int main()
{
	int t, p;
	double n;
	cin>>t;
	int *result=new int[t];
	int i=0;
	while(i<t){
		cin>>n>>p;
		while(p>0){
			if(n==3){
				break;
			}
			n = ceil(n/2)+1;
			p--;
		}
		result[i]=n;
		i++;
	}
	for(i=0; i<t; i++) cout<<result[i]<<endl;
    return 0;
}
