#include <iostream>
using namespace std;
int main()
{
	long long int n,m,i;
	cin >> n >> m;
	int *a = new int[n];
	int *b = new int[m];
	long long int sa=0,sb=0;
	for (i = 0; i < n; i++)
	{

		cin >> a[i];
		sa+=a[i];
	}
	for (int i = 0; i < m; ++i)
	{
		/* code */cin >> b[i];
		sb+=b[i];
	}
	cout << sa * sb;
	return 0;
}
