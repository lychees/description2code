#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>
#include <algorithm>
#include <vector>
#include <map>
#include <set>
#include <cctype>
#include <utility>
#include <queue>
#include <cmath>
using namespace std;

#define MOD 1000000007
#define MX 200005

int s[MX], Dec[MX];
char str[MX];

int main() {
	int n, i, ans;
	scanf("%s", str);
	n = strlen(str);
	
	for (Dec[0] = 1, i = 1; i < n; i++) Dec[i] = 1LL * Dec[i - 1] * 10 % MOD;
	for (s[0] = 1, i = 1; i < n; i++) s[i] = (s[i - 1] + Dec[i]) % MOD;
	for (ans = i = 0; i < n; i++) {
		ans += 1LL * (str[i] - '0') * (i + 1) * s[n - i - 1] % MOD;
		ans %= MOD;
	}
	printf("%d\n", ans);
	return 0;
}
