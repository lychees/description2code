#include <iostream>
using namespace std;

long long tot[1000001];
int arr[1000001];
bool pr[1000001];
void init()
{
	for(int i=1;i<=1000000;++i){ tot[i]=i; arr[i]=i; }
	for(int i=2;i<=1000;++i)
	{
		if(pr[i]==0)
		{
			tot[i]=i-1;
			arr[i]=1;
			for(int j=2*i;j<=1000000;j+=i)
			{
				pr[j]=1;
				tot[j]-=tot[j]/i;
				while(arr[j]%i==0){
					arr[j]/=i;
				}
			}
		}
	}
	for(int i=2;i<=1000000;++i)
	{
		if(arr[i]>1){
			tot[i]-=tot[i]/arr[i];
		}
	}
	for(int i=3;i<=1000000;++i)
	{
		//tot[i]--;
		tot[i]+=tot[i-1];
	}
}
int main()
{
    init();
    int t;
    long long n,total;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%lld",&n);
    	if(n==1){
    		printf("0\n");
    		continue;
    	}
    	total=(n*(n-1))/2;
    	long long ans=total-tot[n];
    	printf("%lld\n",ans);
    }
    return 0;
}
