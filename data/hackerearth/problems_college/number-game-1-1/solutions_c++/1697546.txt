#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	int n;
    	cin>>n;
    	int count = 0;
    	while(n!=1)
    	{
			if(n%3==0)
			{
				n=n/3;
			}
			else if(n%2==0)
			{
				n=n/2;
			}
			else
			{
				n = n-1;
			}
			count++;
    	}
    	cout<<count<<endl;
    }
    return 0;
}
