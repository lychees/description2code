#include <bits/stdc++.h>
using namespace std;

typedef long long LL;
const int MAXN = 1e5 + 10;
const LL MOD = 1e9 + 7;

int main(){
	int t;
	cin >> t;
	while(t--){
		long long a0, a1, c, n;
		cin >> a0 >> a1 >> c >> n;
		long long ans = 0;
		if(n >= 1){
			ans = a0 % 41;
		}
		if(n >= 2){
			ans = (ans * 10 + a1) % 41;
		}
		if(n >= 3){
			for(int i = 2;i < n;++i){
				long long newa = (a1 * c + a0) % 10;
				ans = (ans * 10 + newa) % 41;
				a0 = a1;
				a1 = newa;
			}
		}
		if(ans == 0){
			cout << "YES\n";
		} else cout << "NO\n";
	}
    return 0;
}
