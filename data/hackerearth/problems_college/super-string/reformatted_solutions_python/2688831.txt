import string
d = {}
j = 26
for x in string.ascii_uppercase:
    d[x] = j
    j -= 1

t = int(raw_input())
for i in range(t):
    d1 = {}
    count = 0
    s = raw_input()
    for x in s:
        d1[x] = d1.get(x, 0)+1
    for x in d1.keys():
        if d1[x] == d[x]:
            count += 1
    if count == len(d1.keys()):
        print 'Yes'
    else:
        print 'No'
