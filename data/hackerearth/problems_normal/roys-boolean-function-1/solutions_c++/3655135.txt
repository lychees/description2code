#include <iostream>
#include<math.h>
using namespace std;
int gcd(int a,int b)
{
	int r;
	while(b)
	{
		r=a%b;
		a=b;
		b=r;
	}
	return a;
}
int isprime(int n)
{
	int i,j=sqrt(n);
	if(n==1)
	 return 0;
	for(i=2;i<=j;i++)
	{
		if(n%i==0)
		 return 0;
	}
	return 1;
	
}
int main()
{
    int i,t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
      int c=1,k=2;
      while(k<n)
      {
      	if(gcd(n,k)==1)
      	  c++;
      	  k++;
      }
      if(isprime(c))
         cout<<"TRUE"<<endl;
       else
         cout<<"FALSE"<<endl;
    }
    return 0;
}
