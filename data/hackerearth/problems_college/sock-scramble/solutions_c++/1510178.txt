#include <cstdio>

int main()
{
    int t;
    scanf("%d", &t);
    while (t--)
    {
        int n;
        scanf("%d", &n);

        int ans = 0;
        while (n--)
        {
            int x;
            scanf("%d", &x);

            ans ^= x;
        }

        printf("%d\n", ans);
    }

    return 0;
}
