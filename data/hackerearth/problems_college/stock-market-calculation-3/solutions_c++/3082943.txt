#include<iostream>
    using namespace std;
    int main()
    {
    	int t,i,j;    	cin>>t;
    	while(t--)
    	{long int n,s=0;		cin>>n;
    		long int a[n];
    		for(i=0;i<n;i++)
    		cin>>a[i];
    		long int temp[n];
    		temp[n-1]=a[n-1];
    		for(i=n-2;i>=0;i--)
    		{
    			if(a[i]>temp[i+1])
    				temp[i]=a[i];
				else
					temp[i]=temp[i+1];
    		}
    		    		for(i=0;i<n;i++)
    		    			s=s+temp[i]-a[i];
    		    		cout<<s<<"\n";
    	}
    }
