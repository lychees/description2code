for t in xrange(input()):
    x, y = map(int, raw_input().split())
    c = 0
    while (x > 0 and y > 0):
        d1 = x % 10
        d2 = y % 10
        if d1+d2 >= 10:
            c += 1
            x = x/10+((d1+d2) % 10)
            y = y/10+((d1+d2) % 10)
        else:
            x /= 10
            y /= 10
    if c == 0:
        print "No carry operation"
    elif c == 1:
        print "1 carry operation"
    else:
        print c, "carry operations"
