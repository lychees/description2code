#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	int a[n];
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
	}
	int x;
	cin>>x;
	sort(a,a+n);
	int flag = 0;
	for(int j=n-1;j>=0;j--)
	{
		int i=0;
		int k = j-1;
		int sum ;
		while(i<j)
		{
			sum = a[i] + a[j] + a[k];
			if(sum == x)
			{
				cout<<"YES"<<"\n";
				flag = 1;
				break;
			}
			if(sum > x)
			{
				k--;
			}
			if(sum<x)
			{
				i++;
			}
		}
		if(flag == 1)
		{
			break;
		}
	}
	if(flag == 0)
	{
		cout<<"NO"<<"\n";
	}
}
