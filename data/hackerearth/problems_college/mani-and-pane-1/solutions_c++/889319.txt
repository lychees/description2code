typedef long long LL;

#define MOD 1000000007
#define MAX 1000000
#define sd(x) scanf("%d", &x)

using namespace std;

int sub[101];

int main() {
	int gcd[101][101];
	
	for (int i=2; i<101; i++) gcd[1][i]=1;
	
	for (int i=2; i<101; i++) {
		for (int j=i+1; j<101; j++) {
			int a=i;
			int b=j;
			while (a) {
				int tmp=a;
				a=b%a;
				b=tmp;
			}
			gcd[i][j]=b;
		//	printf("%d %d %d\n", i, j, b);
		}
	}
	
	int tot=100*33*49;
	
	sub[1]=99*49;
	for (int i=2; i<101; i++) {
		for (int j=i+1; j<101; j++) {
			int g=gcd[i][j];
			if (g==1) {
				sub[1]+=(100-j);
			} else {
				for (int k=j+1; k<101; k++) {
					++sub[gcd[g][k]];
				}
			}
		}
	}
	
	int t;
	cin >> t;
	
	while (t--) {
		long long x, y;
		cin >> x >> y;
		x=abs(x-y);
		int ans=sub[1];
		for (int i=2; i<101; i++) {
			if (x%i==0) {
				ans+=sub[i];
			}
		}
		
		printf("%.6lf\n", (double)ans/tot);
	}
	
	return 0;
}
