#include<iostream>
#include<set>
#include<bitset>
using namespace std;
int main(){
int t,n;
cin>>t;
while(t--){
    cin>>n;
    int d[n];
    set<int> uni;
    for(int i=0;i<n;i++){
        cin>>d[i];
    }
    int up=1<<n;
    for(int i=0;i<up;i++){
        bitset<20> val(i);
        int sum=0;
        for(int j=0;j<n;j++){
            if(val[j])sum+=d[j];
        }
        uni.insert(sum);
    }
    cout<<uni.size()<<endl;


}
return 0;
}
