#include <iostream>
#include<string.h>
#include<math.h>
using namespace std;
 
int main()
{
	int t,flag=0,len,i;
	cin>>t;
	char a[100];
	while(t--)
	{
		flag=0;
		cin>>a;
		len=strlen(a);
		for(i=1;i<len;i++)
		{
			if(abs(a[i]-a[i-1])!=abs(a[len-i]-a[len-i-1]))
			{
				flag=1;
				cout<<"Not Funny\n";
				break;
			}
		}
		if(flag==0)
		cout<<"Funny\n";
		
	}
}
