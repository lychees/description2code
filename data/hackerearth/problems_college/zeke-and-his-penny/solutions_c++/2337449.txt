#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
#define ll long long
#define mod 1000000007
int main()
{
	ll ar[3],i;
	int t;
	ll sum;
	scanf("%d",&t);
	while(t--)
	{
		sum=1;
		for(i=0;i<3;i++)
		scanf("%lld",&ar[i]);
 		sort(ar,ar+3);
        for(i=0;i<3;i++)
        {
        	sum*=ar[i]-i;
        	sum%=mod;
        }
        printf("%lld\n",sum);
	}	
	return 0;
}
