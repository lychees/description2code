#include <iostream>
#define ll long long 
using namespace std;

ll fast_pow(ll base,ll pow,ll mod){
	if(pow==0)
	return 1;
	if(pow==1)
	return base;
	ll halfn=fast_pow(base,pow/2,mod);
	if(pow%2==0)
	return (halfn%mod*halfn%mod)%mod;
	else
	return (((halfn%mod*halfn%mod)%mod)*base)%mod;
}
int main()
{
	ll t;
	cin>>t;
	while(t--)
	{
		ll n;
		cin>>n;
		ll res=fast_pow(2,n,1000000007);
		res=(res-1)%1000000007;
		cout<<res<<endl;
	}
    }
