#include <bits/stdc++.h>
using namespace std;
#define INF 1000000007
long dp[1000001];
int main()
{
    long T,N;
    cin>>T;
    dp[1]=2;dp[2]=5;dp[3]=10;
    for(long i=4;i<=1000000;++i)
    {
    	dp[i]=dp[i-1]+dp[i-1]-dp[i-3];
    	while(dp[i]<0)dp[i]+=INF;
    	dp[i]=dp[i]%INF;
    }
    while(T--)
    {
    	cin>>N;
    	cout<<dp[N]<<endl;
    	
    }
    return 0;
}
