#include <bits/stdc++.h>

using namespace std;

int main()
{

	int n;
	cin>>n;
    string s1;
    getline(cin,s1);
	for(int i=0; i<n; i++)
	{
		string s;
		getline(cin,s);
        //cout<<s<<endl;
		int s_ind = 0;
		for(int i=0; i<s.length(); i++)
		{
			if(s[i]=='$')
			{
				s_ind = i+1;
				break;
			}
		}
		int e_ind = -1;
		string ans = "";
		for(int i=s_ind; i<s.length(); i++)
		{
			int val = (int)s[i];
			if(val>=65 && val<=90)
			{
				e_ind = i-1;
				break;
			}
			if(val>=97 && val<=122)
			{
				e_ind = i-1;
				break;
			}
			ans += s[i];
		}
        if(e_ind==-1)
        {
            e_ind = s.length()-1;
        }
		string a1 = "";
        //cout<<s_ind<<" "<<e_ind<<endl;
		for(int i=s_ind; i<=e_ind; i++)
		{
			if(s[i]==' ')
			{
				continue;
			}
			a1 += s[i];
		}
        string a2 = "";
        int z = 0;
        int p = 0;
        //cout<<a1<<endl;
        while(z<a1.length())
        {
            int val = a1[z] - '0';
            //cout<<val<<endl;
            if(val==0)
            {
                z++;
                continue;
            }
            else
            {
                p = z;
                break;
            }
            z++;
        }
        while(p<a1.length())
        {
            a2 += a1[p];
            p++;
        }
        int flag_z = 0;
        z = 0;
        for(z=0; z<a2.length(); z++)
        {
            if(a2[z]!='0')
            {
                flag_z = 1;
                break;
            }
        }
        if(flag_z==0)
        {
            cout<<"$0"<<endl;
        }
        else
        {
            cout<<"$"<<a2<<endl;   
        }
	}
	return 0;
}
