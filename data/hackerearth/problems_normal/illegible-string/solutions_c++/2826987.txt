#define MOD 1000000009
int main(void){
	std::ios::sync_with_stdio(false);cin.tie(0);
	int i,n;
	int x=0,y=0;
	string s;
	cin>>n;
	cin>>s;
	for(i=0;i<n-1;i++)
	{
		if(s[i]=='w')
			y++;
		else if(s[i]=='v'&&s[i+1]=='v')
		{
			y++;
			i++;
		}
		else if(s[i]=='v'&&s[i+1]=='w')
		{
			y++;
			s[i+1]='v';
		}
		else
			x++;
	}
	if(i==n-1)
	{
		if(s[i]=='w')
			y++;
		else
			x++;
	}
	cout<<x+y<<' '<<x+y+y<<endl;
//	system("pause");
	return 0;
}
