#include<iostream>
#include<cstdio>
#include<cstring>
#include<vector>
using namespace std;

int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		long long n;
		scanf("%lld",&n);
		long long a=2,b=1;
		vector<char> v;
		while(n>a*b){
			n-=a*b;
			a*=2;
			b++;
		}
		a/=2;
		while(a){
			if(n>a*b){
				v.push_back('5');
				n-=a*b;
			}
			else
				v.push_back('4');
			a/=2;
		}
		if(v[n-1]=='5')
			printf("Earth\n");
		else
			printf("Hacker\n");
	}
	return 0;
}
