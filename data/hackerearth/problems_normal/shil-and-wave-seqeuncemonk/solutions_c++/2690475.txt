void inc(int id,int i,int delta)
{
	for (;i<=100002;i=(i|(i+1)))
		t[id][i]+=delta,
		t[id][i]%=bs;
}
int sum(int id,int r)
{
	int res=0;
	for (;r>=0;r=(r&(r+1))-1)
		res+=t[id][r],
		res%=bs;
	return res;
}

int sum(int id,int l,int r)
{
//	cout<<sum(id,r)<<"%"<<sum(id,l-1)<<endl;
	
	return sum(id,r)-sum(id,l-1);
}
