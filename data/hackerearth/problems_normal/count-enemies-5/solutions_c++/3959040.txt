#include <bits/stdc++.h>
using namespace std;
 
int main()
{
	int t,i,n,k,j,l;
	cin >> t;
	while(t--){
		string s;
		cin >> s;
		l=s.size();
		int max=0;
		int gmax=0;
		for(i=0;i<l;i++){
			max=0;
			while(s[i]!='*' && i<l){
				if(s[i]=='X'){
					max=0;
					while(s[i]!='*' && i<l) i++;
					break;
				}
				max++;
				i++;
			}
			gmax+=max;
		}
		cout << gmax << endl;
	}
    return 0;
}
