try:
    while (True):
        a = raw_input()
        b = a.find('//')
        s = ""
        s1 = ""
        l = a.split('//')
        l[0] = l[0].replace("->", ".")
        if (len(l) == 1):
            s += l[0]
        else:
            for i in range(1, len(l)):
                s1 += l[i]
            s = l[0]+"//"+a[b+2:]
        print(s)
except EOFError:
    pass
