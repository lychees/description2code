#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define INF 1000000007

int main()
{
	 ll a,b,c,t,ans;
	 cin>>t;
	 while(t--)
	 {
	 	cin>>a>>b>>c;
	 	ans=__gcd(a,__gcd(b,c));
	 	cout<<ans<<" ";
	 	ans=(((((a/ans)%INF)*((b/ans)%INF))%INF)*((c/ans)%INF))%INF;
	 	cout<<ans<<endl;
	 }
	
	return 0;
}
