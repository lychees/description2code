#include <iostream>
using namespace std;

int main()
{
    int noOfTcases;
    long long int *noOfCoins;
    cin>>noOfTcases;
    noOfCoins = new long long int[noOfTcases];
    for(int i = 0 ; i < noOfTcases ; i++)
    {
        cin>>noOfCoins[i];
    }
    for(int i = 0 ; i < noOfTcases ; i++)
    {
        if(noOfCoins[i] == 1 || noOfCoins[i] == 2)
            cout<<"Alice\n";
        else if(noOfCoins[i]%2 == 0)
            cout<<"Alice\n";
        else
            cout<<"Alice\n";

    }

    return 0;
}
