#include<bits/stdc++.h>
#define ll long long int
#define md 1000000007
using namespace std;
bool prime[10000007];
ll arr[10000007],p;
void seive(ll n)
{
	p=0;
	ll i,j;
	prime[0]=prime[1]=1;
	for(i=2;i*i<=n;i++)
	{
		if(!prime[i])
		{
			j=2;
			while(j*i<=n)
			{
				prime[j*i]=1;
				j++;
			}
			arr[p++]=i;
		}
	}
	for(j=i;j<=n;j++)
	{
		if(!prime[j])
		{
			arr[p++]=j;
		}
	}
}
int main()
{
	ll num,i,j,sum,temp,maxi;
	scanf("%lld",&num);
	num--;
	seive(num+num+10);
	sum=0;
	temp=(2*num)+1;
	for(i=0;i<p;i++)
	{
		if(arr[i]<=num+num)
		{
			ll val=arr[i];
			if(val>num)
			{
				sum=sum+(temp-val);
			}
			else
			{
				sum=sum+val+1;
			}
			sum=(sum%md);
		}
	}
	printf("%lld\n",sum);
}
