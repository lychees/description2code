#include <cstdio>
#include <algorithm>
#include <set>
using namespace std;
const int inf =100009;
typedef pair<long long,int> ii;
int f[inf];
int s[inf];
int sz[inf];
set<ii>mp;
int main()
{
	int c,p,n,a;
	scanf("%d%d%d",&c,&p,&n);
	for (int i=0;i<n;++i)
	{
		scanf("%d",&f[i]);
		sz[i]=1;
		mp.insert(ii(f[i],i));
	}
	for (int i=n;i<c;++i)
		mp.insert(ii(0,i));
	while(p--)
	{
		if (scanf("%d",&a)==EOF)
			a=0;
		ii k=*mp.begin();
		mp.erase(mp.begin());
		printf("%d%c",k.second+1,p?' ':'\n');
		s[k.second]=f[k.second];
		f[k.second]=a;
		sz[k.second]++;
		k.first=sz[k.second]*1ll*(f[k.second]+s[k.second]);
		mp.insert(k);
	}

}
