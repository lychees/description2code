#include <iostream>
#include <algorithm>
using namespace std;


long long val[2000002],in[2000002];

int main() 
{
	long long l,r,i,j,n,el,t,x,y,sum=0;
	in[1]=val[1]=1;
	for(i=2;i<=2000000;i++)
	{
		in[i]=in[i-1]+i-1;
		val[i]=val[i-1]+i;
		if((i-1)%2==0)
		val[i]-=(i-1);
	}
	val[0]=0;
	cin>>t;
	while(t--)
	{
		cin>>l>>r;
		x=upper_bound(in+1,in+2000001,l)-in-1;
		y=upper_bound(in+1,in+2000001,r)-in-1;
		sum=val[y];
		if(y && (r-in[y])%2)
		{sum-=y;}
		sum-=val[x];
		if(x && (l-in[x])%2==0)
		sum+=x;
		cout<<sum<<"\n";
	}
	return 0;
}
