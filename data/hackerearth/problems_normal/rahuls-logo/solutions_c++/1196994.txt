int GCD (int a,int b) {
	if (b == 0) return a;
	return(a % b == 0 ? b : GCD(b,a % b));
}

long long int POW (long long int base,long long int exp) {
	long long int val;
	val = 1;
	while (exp > 0) {
		if (exp % 2 == 1) {
			val = (val * base) % MOD;
		}
		base = (base * base) % MOD;
		exp = exp / 2;
	}
	return val;
}

int a[205][205];
int mx[205];

void create (int n) {
	int i,j,r,c;
	memset(a,0,sizeof(a));
	memset(mx,0,sizeof(mx));
	for (i = n; i > 0; i -= 2) {
		r = 1;
		c = n - i + 1;
		for (j = 0; j < i; j++) {
			a[r][c] = 1;
			mx[r] = max(mx[r],c);
			r += 1;
			c += 1;
		}
	}
	for (i = 1; i <= n; i++) {
		for (j = 1; j <= n; j++) {
			if (a[i][j] == 1) {
				a[i][2 * n + 1 - j] = 1;
				mx[i] = max(mx[i],2 * n + 1 - j);
			}
		}
	}
	for (i = n; i >= 1; i--) {
		for (j = 1; j <= 2 * n; j++) {
			if (a[i][j] == 1) {
				if (j <= n) {
					printf("/");
				} else {
					printf("\\");
				}
				if (j == mx[i]) {
					printf("\n");
					break;
				}
			} else {
				printf(" ");
			}
		}
	}
	for (i = 1; i <= n; i++) {
		printf("  ");
		for (j = 1; j <= 2 * n; j++) {
			if (a[i][j] == 1) {
				if (j <= n) {
					printf("\\");
				} else {
					printf("/");
				}
				if (j == mx[i]) {
					printf("\n");
					break;
				}
			} else {
				printf(" ");
			}
		}
	}
}

int main()
