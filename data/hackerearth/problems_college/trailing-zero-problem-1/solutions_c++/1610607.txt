#include<iostream>
#include<math.h>
using namespace std;
#include<cstdio>
#define gc getchar_unlocked
void scanf(int &x)
{
	register int c=gc();
	x=0;
	for(;(c<48 || c>57);c=gc());
	for(;c>47 && c<58; c=gc()) {x=(x<<1)+(x<<3)+c-48;}
}
int main()
{
	int t,n,k,c,x=1,s=0,nod;
	double p=0.0;
	scanf(t);
	while(t--)
	{
		scanf(n);
         k=n;
	    p=0.0;
		while(n>1)
		{
			p=p+log10((double)n);
			n--;
		}
		nod=(int)p+1;
		x=1;
		c=1;
		s=0;
		while(x>0)
		{
		   x=(int)(k/pow(5,c));
		   s=s+x;
		   
           c++;
		   			
		}
		printf("%d\n",nod-s);
	}
	return 0;
}
