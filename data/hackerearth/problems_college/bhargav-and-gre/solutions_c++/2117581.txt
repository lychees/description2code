#include <bits/stdc++.h>
using namespace std;

int main()
{
	string s,s1;
	int lens,q,ant=0,syn=0,lens1;
	cin>>s;
	lens=s.length();
	cin>>q;
	while(q--){
	//	ant=0;syn=0;
	s1.clear();
	cin>>s1;
		lens1=s1.length();
		if(lens!=lens1){
			continue;
		}else{
			if(s[0]==s1[0]){
				syn++;
			}else {
				char c=s[0]+3;
				if(c==s1[0]){
			ant++;
			}
		}
		}
	}
	cout<<syn<<" "<<ant<<"\n";
    return 0;
}
