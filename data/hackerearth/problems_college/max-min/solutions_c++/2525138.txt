
#include <iostream>
using namespace std;
long int a[100000],c[100000];
int i,j,k;
void merge(int low,int mid,int high)
{
i=low;
j=mid+1;
k=low;

while((i<=mid)&&(j<=high))
{
if(a[i]<a[j])
c[k++]=a[i++];
else
c[k++]=a[j++];
}
while(i<=mid)
{
c[k++]=a[i++];
}
while(j<=high)
{
c[k++]=a[j++];
}
for(i=low;i<=high;i++)
    a[i]=c[i];
}

void mergesort(int low,int high)
{
int mid;
if(low<high)
{
mid=(low+high)/2;
mergesort(low,mid);
mergesort(mid+1,high);
merge(low,mid,high);
}
}

	for(p=0;p<=n-k1;p++)
