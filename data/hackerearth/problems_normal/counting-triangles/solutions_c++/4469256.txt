#include <iostream>
#include <bits/stdc++.h>
using namespace std;
#define lli long long int
map < pair <  pair <lli,lli> ,lli > ,int  > mp;
map < pair <  pair <lli,lli> ,lli > ,int  >  :: iterator it;
int main()
{
    lli a,b,c,x,y,z,mn;
    int n,ans=0;
    cin>>n;
    while(n--)
    {
    	cin>>a>>b>>c;
    	mn=min(min(a,b),c);
    	if(mn==a&&b<=c)
    	{
    		x=a;y=b;z=c;
    	}
    	else if(mn==a&&b>c)
    	{
    		x=a;y=c;z=b;
    	}
    	else if(mn==b&&a<=c)
    	{
    		x=b;y=a;z=c;
    	}
    	else if(mn==b&&a>c)
    	{
    		x=b;y=c;z=a;
    	}
    	else if(mn==c&&a<=b)
    	{
    		x=c;y=a;z=b;
    	}
    	else
    	{
    		x=c;y=b;z=a;
    	}
    	mp[(make_pair(make_pair(x,y),z))]++;
    }
    for(it=mp.begin();it!=mp.end();it++)
    {
    	if(it->second==1)
    	{
    		ans++;
    	}
    }
    cout<<ans<<endl;
    return 0;
}
