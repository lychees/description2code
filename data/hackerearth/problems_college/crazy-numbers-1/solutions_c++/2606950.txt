#include<iostream>
#define mod 1000000007
using namespace std;
int calc(string str,int n){

    if(n == 0)
        return 1;
    else{
        if(str[str.length() - 1] > '0' && str[str.length() - 1] < '9'){
            string x = str + (char)(str[str.length() - 1] + 1);
            string y = str + (char)(str[str.length() - 1] - 1);
            return calc(x, n - 1) + calc(y, n - 1);
        }
        else if(str[str.length() - 1] == '0'){
            string x = str + '1';
            return calc(x, n - 1);
        }
        else if(str[str.length() - 1] == '9'){
            string x = str + '8';
            return calc(x, n - 1);
        }
    }

}
long long a[1000001];
int main(){
    int t;
    a[1] = 9;
    a[2] = 17;
    a[3] = 32;
    a[4] = 61;
    a[5] = 116;
    for(int i = 6;i <= 1e6; ++i){
        a[i] = a[i - 1] + 4*a[i - 2] - 3*a[i - 3] - 3*a[i - 4] + a[i - 5];
        while(a[i] < 0)
            a[i] += mod;
        a[i] %= mod;
    }
    cin >> t;
    while(t--){
        int n;
        cin >> n;
        if(n == 1)
            cout << "10\n";
        else
            cout << a[n] << "\n";
    }

}
