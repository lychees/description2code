#include<bits/stdc++.h>
using namespace std;

string s1;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        cin>>s1;
        int len=s1.length(),i;
        for(i=0;i<len;i++)
        {
            if(s1[i]==s1[len-1-i])
                continue;
            else
                break;
        }
        if(i==len)
        {
            printf("-1\n");
            continue;
        }
        string s2;
        int flag=0,res=-1;
        for(i=0;i<len;i++)
        {
            flag=0;
            s2=s1;
            s2.erase(i,1);
            int len1=len-1,j;
            for(j=0;j<len1;j++)
            {
                if(s2[j]==s2[len1-1-j])
                    continue;
                else
                    break;
            }
            if(j==len1)
            {
                flag=1;
                res=i;
                break;
            }
        }
        if(flag==0)
            res=-1;
        printf("%d\n",res);
    }
return 0;
    }
