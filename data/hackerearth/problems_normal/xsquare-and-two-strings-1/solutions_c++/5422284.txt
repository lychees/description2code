#include <bits/stdc++.h>
using namespace std;
 
int main()
{
    int T;
    cin>>T;
    while(T--){
    	string s1,s2;
    	cin>>s1>>s2;
    	int n1=s1.length(),n2=s2.length();
    	
    	int mask1=0,mask2=0;
    	
    	for(int i=0;i<n1;i++){
    		mask1= mask1 | 1<<(s1[i] - 'a');
    	}
    	
    	for(int i=0;i<n2;i++){
    		mask2= mask2 | 1<<(s2[i] - 'a');
    	}
    	
    	mask1 = mask1 & mask2;
    	
    	if(mask1!=0)
    		cout<<"Yes\n";
    	else
    		cout<<"No\n";
    }
}
