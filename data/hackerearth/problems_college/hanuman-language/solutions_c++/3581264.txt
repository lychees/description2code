#include <iostream>
using namespace std;
#include<string.h>
int main()
{
    std::string input;
    int l,s,i,t,j,k;
    char opo[100][100];
    char flag[100][100];
    int length[100];
    int T,P;
    int u;
    cin>>T;
    P=T+1;

    while(P--)
    {
        for(i=0;i<100;i++)
        {

            for(j=0;j<100;j++)
            {
                opo[i][j]='\0';
                flag[i][j]='\0';
            }
        }
    std::getline(std::cin,input);
    l=0,s=0,i=0;
    while(input[s])
    {
        l++;
        s++;
    }
    t=l;
    j=0;
    k=0;
    	while(t)
    	{
    		if(input[i]!=' ')
    		{
    		flag[j][k]=input[i];
    		i++;
    		k++;
    		}
    		else if(input[i]==' ')
            {
                j++;
                k=0;
                i++;
            }
            t--;
    	}
    	for(i=0;i<=j;i++)
    	{
    		length[i]=strlen(flag[i]);
    	}
    	for(i=j;i>=0;i--)
        {
            for(u=0;u<=length[j-i];u++)
            {
                opo[i][u]=flag[j-i][u];
            }
        }
        for(i=0;i<=j;i++)
        {
            cout<<opo[i]<<" ";
        }
        cout<<endl;
}
    return 0;

}
