long int eular(long int n)
{
	long int ans=n,i;
	for(i=2;i*i<=n;i++)
	{
		if(n%i==0)
			ans-=ans/i;
		while(n%i==0)
			n/=i;
	}
	if(n!=1)
		ans-=ans/n;
	return ans;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long int n,d;
		cin>>n>>d;
		if(n%d==0)
		{
			long int c=eular(n/d);
			cout<<c<<endl;
		}
		else
			cout<<0<<endl;
	}
	return 0;
}
