n = int(input())
while n > 0:
    s = raw_input()
    t = ''
    tm = 0
    for i in range(len(s)):
        if s[i] != '*':
            t = t+s[i]
        else:
            if 'X' not in t:
                tm += t.count('O')
            t = ''
    if len(t) != 0:
        if 'X' not in t:
            tm += t.count('O')
    print tm

    n = n-1
