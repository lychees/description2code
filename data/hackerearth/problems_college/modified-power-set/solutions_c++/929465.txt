#include<iostream>
#include<vector>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<cstdio>
#include<stack>
#include<queue>
#include<set>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
         string s;
         set<char> sets;
         cin>>s;
         int l=s.size();
         for(int i=0;i<l;i++)
         {
              char x=s[i];
              sets.insert(x);
         }
         int num=sets.size();
         long long int res=(num*(num+1))/2;     
         cout<<res<<endl;
     }
     return 0;
}
