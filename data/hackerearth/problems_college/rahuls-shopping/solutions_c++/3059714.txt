#include <bits/stdc++.h>

using namespace std;

int sumd(int i)
{
	int sum = 0;
	while(i)
	{
		int r = i%10;
		sum += r;
		i /= 10;
	}
	return sum;
}

int main()
{
	int n;
	cin>>n;
	int arr[n];
	int len,wid;
	for(int i = 0;i<n;i++)
		cin>>arr[i];
	cin>>len>>wid;
	int area = len*wid;
	int ans = 0;
	for(int i = 0;i<n;i++)
	{
		int cost = area*arr[i]*sumd(arr[i]);
		if(arr[i] >= 1 && arr[i] <= 10)
			cost = floor(1.0*cost - 0.25*cost); 
		else if(arr[i] >10 && arr[i] <= 20)
			cost = floor(1.0*cost - 0.3*cost);
		else
			cost = floor(1.0*cost - 0.20*cost);
			
		ans += cost;
	}
	cout<<ans<<'\n';
	
}
