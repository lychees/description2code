#include <bits/stdc++.h>
using namespace std;
#define ll long long

bool solve(ll i,bool sw,ll k){
	if(k==i/2+1){
		if(sw) return 1;
		return 0;
	}

	if(k<=i/2){
		return solve(i/2,sw,k);
	}
	bool cur=solve(i/2,sw^1,i-k+1);
	return cur;
}
int main(){
	ll t,k;
	cin >> t;
	int cnt=0;
	while(t--){
		cin >> k;
		ll len=0;
		while(len<=k){
			len=2*len+1;
		}
		cnt++;
		cout<<solve(len,0,k)<<endl;
	}
}
