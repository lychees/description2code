#include <bits/stdc++.h>
using namespace std;

int arr[100008];

int main() {
	int n;
	scanf("%d", &n);
	int k;
	for (k = 0 ; k < n ; k++) {
		scanf("%d", k + arr);
	}
	int m;
	scanf("%d", &m);
	int x;
	while (m--) {
		scanf("%d", &x);
		for (k = 0 ; k < n ; k++) {
			if (k[arr] > x) {
				k[arr]--;
			}
		}
	}
	for (k = 0 ; k < n ; k++) {
		printf("%d ", k[arr]);
	}
	printf("\n");
	return 0;
}
