#include <bits/stdc++.h>

using namespace std;

const int MOD=252;
long long N;
unsigned int dp[1001][MOD];

struct matrix
{
    unsigned int data[MOD][MOD];
    matrix()
    {
        memset(data, 0, sizeof data);
    }
    matrix operator* (const matrix& other) const
    {
        matrix ret;
        for(int i=0; i<MOD; i++)
            for(int k=0; k<MOD; k++)
                for(int j=0; j<MOD; j++)
                    ret.data[i][j]+=data[i][k]*other.data[k][j];
        return ret;
    }
} A, T, I;

int main()
{
    scanf("%lld", &N);
    N--;
    for(int i=1; i<10; i++)
        A.data[i][0]=1;
    for(int j=0; j<MOD; j++)
        for(int k=1; k<10; k++) if(j*10%k==0)
            T.data[(j*10+k)%MOD][j]++;
    for(; N>0; N/=2)
    {
        if(N&1)
            A=T*A;
        T=T*T;
    }
    unsigned int ans=0;
    for(int i=0; i<MOD; i++)
        ans+=A.data[i][0];
    printf("%u\n", ans);
    return 0;
}
