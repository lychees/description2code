#include <bits/stdc++.h>
#include <stack>
using namespace std;
stack<char>S;
void insert(char ch){
	if(S.empty()){
		S.push(ch);
		return;
	}
	if(ch=='x'){
		while(!S.empty()){
			if(S.top()=='o')S.pop();
			else break;
		}
		S.push(ch);
	}
	else if(ch=='o'){
		if(S.top()!='x'){
			S.push(ch);
		}
	}
	else S.push(ch);

}
int main()
{
    int t;
    cin>>t;
    while(t--){
    	string ss;
    	cin>>ss;
    	int l=ss.size();
    	for(int i=0;i<l;i++){
    		insert(ss[i]);
    	}
    	string tt;
    	while(!S.empty()){
    		tt+=S.top();
    		S.pop();
    	}
    	for(int i=tt.size()-1;i>=0;i--)
    	cout<<tt[i];
    	cout<<"\n";
    }
    return 0;
}
