#include <stdio.h>
using namespace std;

void subt(int array1[], int array2[]){
		int temp;
		for(int i=0;i<100001;i++){
			temp = array1[i];
			array1[i]=(array1[i]>=array2[i])?(array1[i]-array2[i]):((10+array1[i])-array2[i]);
			array1[1+i]=(temp>=array2[i])?(array1[i+1]):(array1[i+1]-1);
		}
}

int main()
{
    int t;
    scanf("%d",&t);
    for(int i=0;i<t;i++){
    	int n,k;
    	scanf("%d %d",&n,&k);
		int initial[100001]={0};
		initial[n]=1;
		int inter1[100001]={0}; int inter2[100001]={0};
		inter1[k]=1; inter2[n-k]=1;
		subt(initial,inter1); subt(initial,inter2);
		int start=0;
		for(int j=100001;j>0;j--){ if(initial[j]!=0){ start = j; break; } }
		for(int j=start;j>0;j--){ printf("%d",initial[j]); }
		printf("%d\n",initial[0]+1);
    }
}
