#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main()
{
	int t;
	ll int n,k,i;
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		ll int arr[k];
		arr[0]=n/k;
		for(i=1;i<k;++i)
		arr[i]= (int)(n-i)/k + 1;
		if(k%2!=0)
		{
			ll int sum=0;
			sum=sum+ (arr[0]*(arr[0]-1))/2;
			for(i=1;i< (float)k/2;++i)
			sum=sum+ (arr[i]*arr[k-i]);
			cout<<sum<<endl;
		}
	else
	{
		ll int sum=0;
		sum=sum+(arr[0]*(arr[0]-1))/2;
		for(i=1;i< k/2;++i)
		sum=sum+ arr[i]*arr[k-i];
		sum=sum+(arr[k/2]*(arr[k/2] -1))/2;
		cout<<sum<<endl;
	}
	}
	return 0;
}
