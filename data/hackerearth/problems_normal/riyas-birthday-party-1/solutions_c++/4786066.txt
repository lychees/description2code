
#include <bits/stdc++.h>
#define mod 1000000007
using namespace std;

long long mulmod(long long a,long long b,long long c)
{
    long long x = 0,y=a%c;
    while(b > 0){
        if(b%2 == 1){
            x = (x+y)%c;
        }
        y = (y*2)%c;
        b /= 2;
    }
    return x%c;
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long n,ans;
		scanf("%lld",&n);
		long long k = mulmod(n,n-1,mod);
		k = (2*k)%mod;
		n%=mod;
		ans=(n+k)%mod;
		printf("%lld\n",ans);
	}
	return 0;
}
