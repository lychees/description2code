#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,i,j,diff1=-1,diff2=-1;
    cin>>n;
    int a[n];
    for(i=0;i<n;i++)
    cin>>a[i];
    sort(a,a+n);
    int k=0;
    for(i=1;i<n;i+=2)
    {
    	j=a[i]-a[k];
    	if(j>diff1)
    	diff1=j;
    	k=i;
    }
    for(i=2;i<n;i+=2)
    {
    	j=a[i]-a[k];
    	if(j>diff2)
    	diff2=j;
    	k=i;
    }
    if(diff1>diff2)
    cout<<diff1<<endl;
    else
    cout<<diff2<<endl;
    return 0;
}
