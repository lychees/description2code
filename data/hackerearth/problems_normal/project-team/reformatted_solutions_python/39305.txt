x = input()
for i in range(x):
    y = raw_input().split()
    n = int(y[0])
    ls = []
    for j in range(1, n+1):
        ls.append(int(y[j]))
    ls.sort()
    max = ls[0]+ls[-1]
    min = ls[0]+ls[-1]
    for k in range(n/2):
        temp = ls[k]+ls[n-k-1]
        if temp > max:
            max = temp
        if temp < min:
            min = temp
    print max-min
