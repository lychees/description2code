

ht = {}
def floors(e, s):
    if e == 0 or s == 0:
        return 0
    elif e == 1:
        return s
    elif s == 1:
        return 1
    elif e > s:
        return floors(s, s)
    if not ht.has_key((e, s)):
        ht[(e, s)] = floors(e-1, s-1) + floors(e, s-1) + 1
    return ht[(e, s)]

noc = input()
while noc:
    noc-=1
    n,k = map(int,raw_input().split())
    s = 0
    while floors(k, s) < n:
        s+=1
    print s
