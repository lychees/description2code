#include <iostream>
using namespace std;

int main()
{
	int t;

    cin>>t;
    int a[t];
    for(int i=0;i<t;i++)
    {
    	cin>>a[i];
    }
    int i,j,m=0;
    for(i=0;i<t;i++)
    {
    	int sum=0;
    	for(j=i;j<t;j++)
    	{
    		sum+=a[j];
    		if(m<sum)
    		{
    			m=sum;
    		}

    	}
    }
    cout<<m<<endl;
    return 0;
}
