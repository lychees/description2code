int vis[1000005];
vector<int> v[1000000];
void dfs(int s)
{
	vector<int>::iterator i;
	for(i=v[s].begin();i!=v[s].end();++i)
	{
		if(vis[*i]==0)
		{
			vis[*i]=1;
			dfs(*i);
		}
	}
}
int main()
{
	int t,i,j,flag=0;
	int m,n;
	int e,l,r;
	scanf("%d",&t);
	int cnt;
	
	while(t--)
	{
		
		
		scanf("%d",&n);
		for(i=0;i<n;++i)
		{
			vis[i]=0;
			v[i].clear();
		}
		cnt=0;
		scanf("%d",&e);
		while(e--)
		{
			scanf("%d",&l);
			scanf("%d",&r);
			v[l].push_back(r);
			v[r].push_back(l);
		}
		for(i=0;i<n;++i)
		{
			if(vis[i]==0)
			{
				//printf("%d ",i);
				vis[i]=1;
				cnt++;
				dfs(i);
			}
		}
			
		printf("%d\n",cnt);
		
		
	}
	return 0;
}
