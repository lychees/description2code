#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t,n,m;
	scanf("%d",&t);
	while (t --) {
		scanf("%d %d",&n,&m);
		if (m <= n) {
			if (m & (n - m)) {
			   printf("even\n");
		    } else {
			   printf("odd\n");
		    }
		}
		else {
			printf("odd\n");
		}
	}
	
	return 0;
}
