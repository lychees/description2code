#include<bits/stdc++.h>

using namespace std;

int main()
	{
		int T;
		cin>>T;
		while(T--)
			{
				int N,i,j,x;
				cin>>N;
				vector<int>v,res;
				for(i=0;i<2*N;i++)
					{
						cin>>x;
						v.push_back(x);
					}
				sort(v.begin(),v.end());
				for(i=0,j=v.size()-1;i<N;i++,j--)
					{
						int sum=v[i]+v[j];
						res.push_back(sum);
					}
				sort(res.begin(),res.end());
				cout<<res[N-1]<<endl;
			}
		return 0;
	}
