from sys import stdin
hi = 10**9 + 7
fa = [1]
for i in xrange(1, 10**6+1):
    fa.append((fa[-1]*i) % hi)
n, m = map(int, stdin.readline().split())
di = {}
ans = 1
for i in xrange(1, n+1):
    for j in xrange(1, m+1):
        di[i ^ j] = di.get(i ^ j, 0)+1

for i in di:
    ans = (ans*fa[di[i]]) % hi
print ans
