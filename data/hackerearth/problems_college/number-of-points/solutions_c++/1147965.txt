#include <iostream>
#include <cmath>
using namespace std;

int gcd(int a, int b) {
	if(a==0&&b==0)return 1;
	if (a<0) a = -a;
	if (b<0) b = -b;
	int t;
	while(a>0 && b>0) {
		t = a;
		a = min(a,b);
		b = max(t,b)%min(t,b);
	}
	return max(a,b);
}

int main() {
	int n,a,b,c,d;
	cin>>n;
	while(n--){
		cin>>a>>b>>c>>d;
		cout<<gcd(c-a,d-b)-1<<endl;
	}
}
