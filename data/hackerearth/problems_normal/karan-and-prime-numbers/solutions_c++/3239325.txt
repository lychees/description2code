#include <iostream>

using namespace std;
typedef long long int lld;
int isPrime(lld inp)
{
    if(inp==1)
        return 0;
    else if(inp==2)
        return 1;
    else if(inp%2==0)
        return 0;
    else{
        for(int i=3;i*i<=inp;i+=2)
        {
            if(inp%i==0)
                return 0;
        }
        return 1;
    }
}

int main()
{
    int t;
    cin>>t;
    while(t--){
        int a,b;
        cin>>a>>b;
        int ans=0;
        for(int i=a;i<=b;i++){
            if(isPrime(i)){
                ans=ans+i;
            }
            }
        cout<<ans<<endl;
    }
   // cout << "Hello world!" << endl;
    return 0;
}
