//subset ssum problem
#include<bits/stdc++.h>
#include<algorithm>
using namespace std;
#define fr(i,a,b) for(int i=a;i<b;i++)
#define frn(i,a,b) for(int i=a;i>b;i--)
#define si(x) scanf("%d",&x)
#define sll(x)  scanf("%lld",&x)
#define pln printf("\n")
#define pi(x) printf("%d",x)
#define pll(x) printf("%lld",x)
#define mp make_pair
#define pb push_back
#define x first
#define y second
typedef long long ll;
ll MOD=1000000007;
ll pw(ll a,ll b){
  ll r=1;
  while(b>0){
     if(b&1){
       r=(r*a)%MOD;
        }
     b=b>>1;
     a=(a*a)%MOD;
  }
return (r)%MOD;
}
 int dp[10101]={};
int main(){
  int t;
  si(t);
  while(t--){
    int n;
    si(n);
    int a[111];
    int sum=0;
    fr(i,0,n){
      si(a[i]);
      sum+=a[i];
    }
  fr(i,0,10101){
     dp[i]=0;
  }
   dp[0]=1;
   fr(i,0,n){
      frn(j,10101,-1){
         if(j>=a[i]){
            dp[j]|=dp[j-a[i]];
         }
      }
   }

    int c=0;
    fr(i,0,10101){
       if(dp[i])c++;
    }
    pi(c);pln;
  }



return 0;
}
