#include <iostream>
#include<string.h>
using namespace std;

int main()

{   int t;
string s;
int flag;
	cin>>t;
while(t--)
{
	cin>>s;
	int len=s.length();
	//cout<<len;
	flag=0;
	if(len>=10)
	{cout<<"YES"<<"\n";
	   continue;
	}//if(len%4==0||len%4==2)
	//cout<<"YES"<<"\n

		for(int i=1;i<len;i++)
		{
			for(int j=i+1;j<len;j++)
			{
				for(int k=j+1;k<len;k++)
				{
					string s1=s.substr(0,i);
					string s2=s.substr(i,j-i);
					string s3=s.substr(j,k-j);
					string s4=s.substr(k,len-k);
					if((s1!=s2)&&(s1!=s3)&&(s1!=s4)&&(s2!=s3)&&(s2!=s4)&&(s3!=s4))
					{
						flag=1;
					}
				
				
				}
			}
		}
		if(flag)
				cout<<"YES"<<"\n";
				else
				cout<<"NO"<<"\n";
	}
	
    return 0;
}
