#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t;

    cin >> t;

    while (t--) {
        int n, c, m, t1, t2, t3, ans = 0;

        cin >> n >> c >> m;

        t1 = n / c;
        ans = t1;
        t2 = t1 / m;
        t3 = t1 % m;
        while (t2 != 0) {
            ans += t2;

            t1 = t2 + t3;
            t2 = t1 / m;
            t3 = t1 % m;
        }

        cout << ans << endl;
    }

    return 0;
}
