#include<cstdio>
using namespace std;
int main()
{
    long long t,a,b,c,ans;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld%lld",&a,&b);
        c=a xor b;
        ans=0;
        while(c!=0)
        {
            ans+=c%2;
            c/=2;
        }
        if(ans%2==0)
        {
            printf("YES\n");
        }
        else
        {
            printf("NO\n");
        }
    }
    return 0;
}
