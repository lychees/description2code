bool check(ll mid ,ll rem) {
	mid = mid*(mid + 1);
	if(mid > rem)
		return true;
	return false;
}

ll search(ll rem , ll lo , ll hi) {
	ll mid = (lo + hi) / 2;
	while(hi > lo + 1) {
		if(check(mid , rem))
			hi = mid;
		else
			lo = mid + 1;
		mid = (lo + hi) / 2;
	}
	if(check(lo , rem))
		return lo;
	else
		return lo + 1;
}
