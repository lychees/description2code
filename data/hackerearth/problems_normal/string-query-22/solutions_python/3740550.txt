import sys
s = raw_input()
m = int(raw_input())

dic = {}
for i in range(len(s)):
	if s[i] not in dic:
		dic[s[i]] = []
	dic[s[i]].append(i)

for i in range(m):
	temp = raw_input().split()
	loc = int(temp[0]) - 1
	key = temp[1]
	strLoc = dic[key][loc]
	del dic[key][loc]
	if not dic[key]:
		del dic[key]

testAns = ['' for x in range(len(s))]

for key, value in dic.iteritems():
	for item in value:
		testAns[item] = key

for test in testAns:
	if test != '':
		sys.stdout.write(test)
