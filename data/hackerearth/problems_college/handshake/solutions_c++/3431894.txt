#include <iostream>
using namespace std;  
 int main(){  
   int t;
   int o;
   int ty;  
   cin>>t; 
   while(t--){  
     o=0;  
     cin>>ty;  
     while(ty--)  
     {  
       if(ty>0)  
       {  
         o+=ty;  
       }  
     }  
    cout<<o; cout<<endl;  
   }  
 }
