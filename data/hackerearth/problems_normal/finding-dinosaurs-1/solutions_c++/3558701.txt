//#define DEBUG ON

#define dout(...) cout<<__VA_ARGS__
#else
#define dout(...) do_nothing()
#endif

void do_nothing() {}

char* name[16][16][16][16][16] = {0};

int main()
{
    int n, k, l, q;
    cin>>n>>k>>l>>q;
    int i;

    for(i=1; i<=n; i++)
    {
        char *name1;
        name1 = new char[11];
        memset(name1, 0, 11);
        cin>>name1;
        int k1, k2, k3, k4, k5;
        k1=k2=k3=k4=k5=0;
        for(int j=1; j<=k; j++)
        {
            switch(j)
            {
                case 1: cin>>k1; break;
                case 2: cin>>k2; break;
                case 3: cin>>k3; break;
                case 4: cin>>k4; break;
                case 5: cin>>k5; break;
            }
            
        }
        name[k1][k2][k3][k4][k5] = name1;

    }
    dout("proceeding with the queries now\n");
    for(i=1; i<=q; i++)
    {
        int k1, k2, k3, k4, k5;
        k1=k2=k3=k4=k5=0;
        dout("query no: "<<i<<endl);
        for(int j=1; j<=k; j++)
        {
            switch(j)
            {
                case 1: cin>>k1; break;
                case 2: cin>>k2; break;
                case 3: cin>>k3; break;
                case 4: cin>>k4; break;
                case 5: cin>>k5; break;
            }

        }
        if(name[k1][k2][k3][k4][k5] != 0)
            cout<<name[k1][k2][k3][k4][k5]<<endl;
        else
            cout<<"You cant fool me :P"<<endl;

    }

    return 0;
}
