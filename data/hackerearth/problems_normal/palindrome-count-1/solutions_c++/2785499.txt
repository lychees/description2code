#include <iostream>
using namespace std;

bool isPalindrome(string s){
	int n=s.length();
	for(int i=0;i<n/2;i++){
		if(s[i]!=s[n-i-1])
			return false;
	}
	return true;
}

int main()
{
    string s;
    cin>>s;
    int result=0;
    for(int i=0;i<s.length();i++){
    	for(int j=i;j<s.length();j++){
    		bool isit=isPalindrome(s.substr(i,j-i+1));
    		if(isit)
    			result++;
    	}
    }
    cout<<result;
    return 0;
}
