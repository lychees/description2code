char A [ MAX ] ;
char B [ MAX ] ;

int fv [ 27 ] ;

int main ( void )
{
    int t ;
    cin >> t ;
    cin.getline ( A , 100001 ) ;
    for ( ; t ; -- t)
    {
        memset ( fv , 0 , sizeof ( fv ) ) ;
        cin.getline ( A , 100001 ) ;
        cin.getline ( B , 100001 ) ;
        int l = strlen ( A ) ;
        FORN ( i , 0 , l - 1 )
            if ( A [ i ] != ' ' )
            fv [ A [ i ] - 'a' ] ++ ;
        l = strlen ( B ) ;
        FORN ( i , 0 , l - 1 )
            if ( B [ i ] != ' ' )
            fv [ B [ i ] - 'a' ] -- ;
        int unu , doi ;
        unu = doi = 0 ;
        FORN ( i , 0 , 26 )
            if ( fv [ i ] > 0 ) unu = 1 ;
            else if ( fv [ i ] < 0 ) doi = 1 ;
        if ( unu and doi )
            cout << "You draw some.\n" ;
        else if ( unu )
            cout << "You win some.\n" ;
        else cout << "You lose some.\n" ;
    }
    return 0;
}
