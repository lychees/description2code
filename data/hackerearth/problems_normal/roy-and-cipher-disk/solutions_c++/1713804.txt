#include<iostream>
#include<string.h>
using namespace std;
int main(){
int t;

cin>>t;
char str[100];
while(t--){
    
    cin>>str;
    int l=strlen(str);
    int n=0,m,j,x,k;
     for(int i=0;i<l;i++){
          x=str[i];
          k=x-97;
       if(k>=n){
       	m=k-n;
       	 j=26-m;
       	 if(m<=j)
       	 cout<<m<<" ";
       	 else
       	 cout<<"-"<<j<<" ";
       }
       else{
       	m=n-k;
       	j=26-m;
       	if(m<j)
       	cout<<"-"<<m<<" ";
       	else
       	cout<<j<<" ";
       }
                
      n=k;
     }
     cout<<"\n";
}
}
