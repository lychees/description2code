#include <iostream>
#include <cstdio>

using namespace std;

const int maxn = 100000 + 6;
long long f[maxn * 2][12];
int a[maxn];

int main()
{
	int n = 0;
	int q = 0;
	scanf("%d%d", &n, &q);
	for (int i = 1; i <= n; i++)
	{
		scanf("%d", &a[i]);
	}
	for (int i = n; i >= 1; i--)
	{
		for (int k = 1; k <= 10; k++)
			f[i][k] = f[i + k][k] + a[i];
	}
	while (q --> 0)
	{
		int l, r, k;
		scanf("%d%d%d", &l, &r, &k);
		long long ans = f[l][k];
		int dist = (r - l + k) / k;
		l += k * dist;
		if (l <= n)
			ans -= f[l][k];
		cout << ans << "\n";
	}
	return 0;
}
