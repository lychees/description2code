#include<iostream>
using namespace std;

int sol(int a[], int size)
{
   int msf = 0, meh = 0;
   int i;
   for(i = 0; i < size; i++)
   {
     meh = meh + a[i];
     if(meh < 0)
        meh = 0;
     if(msf < meh)
        msf = meh;
    }
    return msf;
} 

int main(){
	int n;
	cin>>n;
	int a[n];
	for(int i=0;i<n;i++)
	cin>>a[i];
	cout<<sol(a,n)<<endl;
}
