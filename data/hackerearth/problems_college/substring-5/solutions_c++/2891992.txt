#include <iostream>
using namespace std;

int main()
{
    string str;
    cin>>str;
    int val=str.length();
    int m=1,i;
    int c=val;
    for(i=0;i<val;i++)
    {
    	if(str[i]==str[i+1])
    	{
    		c=c+m;
    		m++;
    	}
    	else
    	{
    		m=1;
    	}
    }
    cout<<c<<endl;
    
    return 0;
}
