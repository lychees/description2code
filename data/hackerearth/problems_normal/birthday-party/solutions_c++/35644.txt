int subsetSum ( int a[] , int n , int sum ){
	int *s = new int[sum+1];

	for(int i=0; i<=sum; i++) s[i]=0;
	s[0]=1;


	for(int i=0; i<n; i++)
		for(int j=sum; j>=a[i]; j--)
			s[j] |= s[j-a[i]];

	return s[sum];
}
