s = raw_input()
a = []
for i in xrange(26):
    a += [[]]
for i in range(len(s)):
    k = ord(s[i])-97
    a[k] += [i]
b = []
for i in s:
    b += [i]
q = int(raw_input())
while (q):
    q -= 1
    o, l = raw_input().split()
    o = int(o)-1
    lk = ord(l)-97
    i = a[lk][o]
    b[i] = ''
    a[lk].pop(o)
print ''.join(b)
