#include <iostream>
using namespace std;

int main()
{
    int n,t;
    cin >>n>>t;
    int a[n][n];
    for(int i=0;i<n;i++)
    for(int j=0;j<n;j++)
    {
    	cin >> a[i][j];
    }
    while(t--)
    {
    	int p;
    	cin >> p;
    	if((p%360)==0)
    	for(int i=0;i<n;i++)
	    for(int j=0;j<n;j++)
	    {
	    	cout << a[i][j]<<" ";
	    	if(j==n-1)
	    	{
	    		cout << endl;
	    	}
	   
	    }
    	else if(p%360==90)
    	{
    		for(int i=0;i<n;i++)
    		for(int j=n-1;j>=0;j--)
    		{
    			cout << a[j][i] << " ";
	    		if(j==0)
		    	{
		    		cout << endl;
		    	}
		   
    		}
    		
    		
    	}
    	else if(p%360==180)
    	{
    		for(int i=n-1;i>=0;i--)
    		for(int j=n-1;j>=0;j--)
    		{
    			cout << a[i][j] << " ";
	    		if(j==0)
		    	{
		    		cout << endl;
		    	}
		   
    		}
    	}
    	else if(p%360==270)
    	{
    		for(int i=n-1;i>=0;i--)
    		for(int j=0;j<n;j++)
    		{
    			cout << a[j][i] << " ";
	    		if(j==n-1)
		    	{
		    		cout << endl;
		    	}
		   
    		}
    		
    		
    	}
    	cout<<endl;
    }
    return 0;
}
