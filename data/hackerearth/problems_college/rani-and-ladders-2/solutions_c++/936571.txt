#include<iostream>
#include<stdlib.h>
#include <string.h>
#include<algorithm>
using namespace std;


int main()
{
    int T;
    cin>>T;
    while(T--)
    {
        int n,l;
        cin>>n>>l;

        int *a=new int[n];
        int *b=new int[n];
        int *t=new int[n];
        int i,j;
        for(i=0;i<n;i++)
            b[i]=0;

        int temp;
        for(i=0;i<l;i++)
        {
            cin>>temp;
            temp--;
            b[temp]=1;
            cin>>t[temp];
            t[temp]--;
        }

        a[n-1]=0;
        a[n-2]=9999;
        int mi;
        for(i=n-3;i>=0;i--)
        {
            if(!b[i])
            {
                mi=a[i+2]+1;
                for(j=3;j<13;j++)
                {
                    if(a[i+j]+1<mi && i+j<n)
                        mi=a[i+j]+1;
                }
                a[i]=mi;
            }
            else
                a[i]=a[t[i]];
        }

        cout<<a[0]<<endl;
    }
}
