#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	char str[100000];
	cin>>str;
	int i=1;
	int temp=n;
	int arr[10000];
	for(i=1;i<n+1;i++)
		arr[i]=1;
	while(n>1)
	{
		for(int j=0;str[j]!='\0';j++,i++)
		{
			if(i>temp)
				i=1;
			for(i;;i++)
			{
				if(i>temp)
					i=1;
				if(!arr[i])
					continue;
				break;
			}
			if(str[j]=='b')
			{
				n--;
				arr[i]=0;
				if(n==1)
					break;				
			}
		}
	}	
	for(int k=1;k<=temp;k++)
	{
		if(arr[k]==1)
		{
			cout<<k<<endl;
			break;
		}
	}
	
}
