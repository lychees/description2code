#include <cstdio>
#include <cstdio>
int m[105][105];
int visited[105][105];
int n;
int flag;

void zero()
{
    for(int i=0 ; i<105 ; i++)
        for(int j=0 ; j<105 ; j++)
            visited[i][j] = 0;
}

void path_col(int i, int j, int x)
{
    if(j==(n)) flag = 1;
    if(i>=0 && i<n && j>=0 && j<n && visited[i][j]==0)
    {
        visited[i][j] = 1;
        if(m[i][j]==x)
        {
            path_col(i-1,j-1,x);
            path_col(i-1,  j,x);
            path_col(i-1,j+1,x);
            path_col(  i,j-1,x);
            path_col(  i,j+1,x);
            path_col(i+1,j-1,x);
            path_col(i+1,  j,x);
            path_col(i+1,j+1,x);
        }
    }
}

void path_row(int i, int j, int x)
{
    if(i==(n)) flag = 1;
    if(i>=0 && i<n && j>=0 && j<n && visited[i][j]==0)
    {
        visited[i][j] = 1;
        if(m[i][j]==x)
        {
            path_row(i-1,j-1,x);
            path_row(i-1,  j,x);
            path_row(i-1,j+1,x);
            path_row(  i,j-1,x);
            path_row(  i,j+1,x);
            path_row(i+1,j-1,x);
            path_row(i+1,  j,x);
            path_row(i+1,j+1,x);
        }
    }
}

int main()
{
	scanf("%d",&n);
	for(int i=0 ; i<n ; i++)
        for(int j=0 ; j<n ; j++)
            scanf("%d",&m[i][j]);
            
    if(n==1)
    {
        if(m[0][0] == 0) printf("0");
        else if(m[0][0]==1) printf("1");
        else printf("2");
        return 0;
    }
    
    bool row=false,col=false;
    for(int i=0 ; i<n ; i++)
    {
        zero();
        flag=0;
        path_col(i,0,2);
        
            if(flag==1)
            {
                col=true;
                break;
            }
    }
    for(int i=0 ; i<n ; i++)
    {
        zero();
        flag=0;
        path_row(0,i,1);
        
            if(flag==1)
            {
                row=true;
                break;
            }
    }
    if(row && col) printf("AMBIGUOUS");
    else if(row) printf("1");
    else if(col) printf("2");
    else printf("0");
}
