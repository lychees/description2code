#include <iostream>
#include<stdlib.h>
using namespace std;
int cmp(const void*a,const  void*b)
{
	return *(int*)a-*(int*)b;
	
}
int l[100009],r[100009];
int arr[100009];
int ans=10000000;
int n;
void dp(int i,int j,int count)
{
	
	if(i>=n||j<0)
	return;

	if(arr[j]-3*arr[i]<=0)
	{ans=min(ans,count);
	return ;
	}
		if(l[i]==1)
	return;
	l[i]=1;
	
	dp(i+1,j,count+1);
	dp(i,j-1,count+1);
	
	
}
int main()
{
 	
 	cin >> n;
 	
 	int i;
 	for(i=0;i<n;i++)
 	{
 		cin >> arr[i];
 		l[i]=-1;
 		r[i]=-1;
 	}
 	qsort(arr,n,4,cmp);
 	
 	
	 dp(0,n-1,0);
 	cout << ans << endl;
    return 0;
}
