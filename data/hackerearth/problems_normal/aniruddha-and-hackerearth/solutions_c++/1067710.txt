#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
int fib[1000001];
int dp[1000001];
inline int scan(){
    char c = getchar_unlocked();
    int x = 0;
    while(c<'0'||c>'9'){
        c=getchar_unlocked();
    }
    while(c>='0'&&c<='9'){
        x=(x<<1)+(x<<3)+c-'0';
        c=getchar_unlocked();
    }
    return x;
}
int main(){
    fib[0]=1;
    fib[1]=2;
    dp[0]=0;
    dp[1]=2;
    for(int i=2;i<1000001;i++){
        fib[i]=fib[i-1]+fib[i-2];
        fib[i]%=mod;
        dp[i]=dp[i-1]+fib[i];
        dp[i]%=mod;
    }
    int t=scan();
    while(t--){
        int n=scan();
        printf("%d\n",dp[n]);
    }
}
