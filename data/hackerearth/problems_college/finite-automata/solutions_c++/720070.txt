#include <iostream>

using namespace std;

int main()
{
   int n;
   cin >> n;
   int arr[n][3];
   int k=0;
   for(int i=0;i<n;i++)
   {
       arr[i][0]=i;
       for(int j=1;j<3;j++)
       {
           if(k<n)
           arr[i][j]=k;
           else
           {
               k=0;
               arr[i][j]=k;
           }
           k++;
       }
   }
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<3;j++)
        cout << arr[i][j] << " ";
        cout << "\n";
    }
   
   return 0;
}
