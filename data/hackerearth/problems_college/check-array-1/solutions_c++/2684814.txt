#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	cin >> t;
	while (t--)
	{
		int n, flag = 0;;
		cin >> n;
		int a[n];
		long long int sum = 0, sumr = 0, suml = 0;
		
		for (int i = 0; i < n; i++)
		{
			cin >> a[i];
			sum = sum + a[i];
		}
		
		sumr = sum;
		
		for (int i = 0; i < n; i++)
		{
			if (sumr == suml)
			{	
				cout << "YES" << endl;
				flag = 1;
			}
			sumr = sumr - a[i];
			if (i >= 1)
				suml = suml + a[i - 1];
		}
		if (flag != 1)
			cout << "NO" << endl;
	}
	return 0;
}
