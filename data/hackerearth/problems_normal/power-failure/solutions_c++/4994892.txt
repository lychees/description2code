#include <bits/stdc++.h>
#define MOD 1000000007
using namespace std;
bool nprime[100005];
long int p[100005]={0};
long int nums[100005];

void seive(long int n){
	for(long long int i=2;i*i<=n;i++){
		if(nprime[i]) continue;
		for(int j=i*i;j<n;j=j+i){
			nprime[j]=true;
		}
	}
	for(long long int i = 2; i <= n; i++){            
        p[i] += p[i - 1] + !nprime[i];
    }
}

int main()
{
	
	seive(100009);
	
    int t;
    cin>>t;
    while(t--){
    	long long int ans=1;
    	long long int n,m;
    	cin>>n>>m;
    	for(int i=0;i<n;i++){
    		cin>>nums[i];
    	}
    	sort(nums,nums+n);
    	reverse(nums,nums+n);
    	
    	for(long long int i=0;i<n;i++){ 
    		
    		long long ways = m-nums[i]+1-(p[m]-p[nums[i]-1])-i;
    		if(ways<0){ans=0;break;}
    		
    	ans=ans=(ans%MOD*ways%MOD)%MOD;	
    	}
    	ans%=MOD;
    	cout<<ans<<endl;
    	
    }
    
    return 0;
}
