#include<bits/stdc++.h>
using namespace std;
long long mod_inv(long long a,long long b,long long mod)
{
    if(b==0) return 1;
    if(b==1) return a%mod;
    if(b&1)
    {
        long long temp=mod_inv(a,b/2,mod)%mod;
        temp=(temp*temp)%mod;
        return (a*temp)%mod;
    }
    else
    {
        long long temp=mod_inv(a,b/2,mod)%mod;
        return (temp*temp)%mod;
    }
}
void sol()
{
    long long a,d,n,i;
    cin>>a>>d;
    cin>>n;
    a%=n;
    d%=n;
    if(a==0) i=0;
    else if(d==0) i=-1;
    else
    {
        long long x=mod_inv(d,n-2,n);
        i=(x*(n-a))%n;
    }
    cout<<i<<endl;
}
int main()
{
    int t;
    cin>>t;
    while(t--) sol();
    return 0;
}
