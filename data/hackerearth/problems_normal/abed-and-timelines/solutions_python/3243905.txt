for _ in range(input()):
	n,m=map(int,raw_input().split())
	#print n,m
	x1,y1=map(int,raw_input().split())
	x2,y2=map(int,raw_input().split())
	pr=input()
	x=min(abs(x2-x1),abs(n-abs(x2-x1)))
	y=min(abs(y2-y1),abs(m-abs(y2-y1)))
	pr=(x+y)*pr
	if(pr>1000):
		print pr
		print"Let go of the group."
	else:
		print pr
