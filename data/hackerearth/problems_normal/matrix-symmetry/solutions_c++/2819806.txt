#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		char a[n][n];
		for(int i=0;i<n;i++){
			for(int j=0;j<n;j++){
				cin>>a[i][j];
			}
		}int ch=0,cv=0;
		//for(int k=0;k<=n/2;k++){int m=-1;m++;
			for(int i=0,j=n-1;i<n/2;i++,j--){
				for(int k=0;k<n;k++){
					if(a[i][k]!=a[j][k]){cv++;}
					if(a[k][i]!=a[k][j]){ch++;}
				}
			}
		//}
		// cout<<ch<<" "<<cv<<endl;
		if(ch>0 && cv>0){cout<<"NO\n";}
		else if(ch==0 && cv==0){cout<<"BOTH\n";}
		else if(ch>0){cout<<"HORIZONTAL\n";}
		else{cout<<"VERTICAL\n";}
	}
   // cout << "Hello World!" << endl;
    return 0;
}
