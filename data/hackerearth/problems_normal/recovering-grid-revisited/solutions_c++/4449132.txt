#include <bits/stdc++.h>
#define MOD 1000000007
#define ll long long

using namespace std;

ll power(ll x,ll n)
{
	ll ans=1;
	
	while(n)
	{
		if(n&1)
		ans = (ans*x)%MOD;
		
		n = n>>1;
		
		x = (x*x)%MOD;
	}
	
	return ans;
}



int main()
{
    int t;
    
   	cin >> t;
   	
   	while(t--)
   	{
   		ll int n,m;
   		
   		cin >> n >> m;
   		if(n==1)
   		{
   			cout << 0 << endl;
   			continue;
   		}
   		n = (n*n)-1;
   		ll int num=1;
   		ll int denom=1;
   		
   		for(ll int i=m+n-1;i>n-1;i--)
   		num = (num * i)%MOD;
   		
   		for(ll int i=1;i<=m;i++)
   		denom = (denom*i)%MOD;
   		
   		denom = power(denom,MOD-2);
   		
   		cout << (num*denom)%MOD << endl;
   	}
    return 0;
}
