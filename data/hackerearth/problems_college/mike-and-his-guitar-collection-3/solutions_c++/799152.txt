#include<cstdio>
#include<algorithm>

inline int Read()
{
	register int c=getchar();
	int x=0;
	for(;(c<48 || c>57);c=getchar());
	for(;c>47 && c<58;c=getchar())
		x=(x<<1)+(x<<3)+c-48;
	return x;
}

int notin(int *a,int sum,int n)
{
	for(int i=0;i<n;i++)
	{
		if(a[i]==sum)
			return 0;
	}
	return 1;
}

int main()
{
	int t=Read();
	while(t--)
	{
		int n=Read(),ans=0;
		int a[n];
		for(int i=0;i<n;i++)
			a[i]=Read();
		int *s=new int,cnt=0;
		for(int i=0;i<=n;i++)
		{
			int x[n];
			for(int j=0;j<n;j++)
				x[j]=0;
			for(int j=0;j<i;j++)
				x[j]=1;
			do
			{
				int sum=0;
				for(int i=0;i<n;i++)
					if(x[i])
						sum+=a[i];
				if(notin(s,sum,cnt))
					s[cnt++]=sum,ans++;
			}while(std::prev_permutation(x,x+n));
		}
		printf("%d\n",ans);
	}
	return 0;
}
