using namespace std;
typedef long long ll;
string s;
bool checkPalindrome(int i,int j)
{
    while(s[i]==s[j] && i<j)
    {
        i++;j--;
    }
    if(i==j || i==j+1)
        return true;
    else
        return false;
}
int main()
{
    int t; get(t);
    while(t--)
    {
        cin >> s;
        int flag=0,ans=1,mx=1;
        for(int i=0;i<s.length();i++)
        {
            for(int j=s.length()-1;j>i;j--)
            {
                if(checkPalindrome(i,j))
                {
                    ans=j-i+1;
                    mx=max(mx,ans);
                }
            }
        }
        cout << mx << endl;
    }
    return 0;
}
