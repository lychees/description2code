#include <iostream>
using namespace std;

int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
    	int a,b;
    	cin>>a>>b;
    	int arr[26];
    	int i=0;
    	while(a!=0)
    	{
    		arr[i]=a%b;
    		a=a/b;
    		i++;
    	}
    	for(int j=i-1;j>=0;j--)
    	  cout<<arr[j];
    	 cout<<endl;
    }
    return 0;
}
