#include<iostream>
using namespace std;
class A
{
    int n, q;
public:
    void call()
    {
        int i, j;
        cin>>n;
        cin>>q;
        int arr[n];
        int a[q][3];
        for(i=0;i<n;i++)
        {
            cin>>arr[i];
        }
        for(i=0;i<q;i++)
        {
            for(j=0;j<3;j++)
                cin>>a[i][j];
        }
        for(i=0;i<q;i++)
        {
            if(a[i][0] == 1)//Firewall
            {
                int x, y;
                x=a[i][1];
                y=a[i][2];
                --x;--y;
                int c=0;
                long int sum=0;
                for(j=x;j<=y;j++)
                {
                    c++;
                    sum=sum+arr[j];
                }
                cout<<sum<<"/"<<c<<endl;
            }
            else//Filter
            {
                int x, y;
                x=a[i][1];
                y=a[i][2];
                --x;
                --y;
                int c=1;
                for(j=x;j<=y;j++)
                {
                    if(c%5==1)
                        arr[j]=arr[j]+16;
                    else if(c%5==2)
                        arr[j]=arr[j]+9;
                    else if(c%5==3 || c%5==4)
                        arr[j]=arr[j]+26;
                    else if(c%5==0)
                        arr[j]=arr[j]+1;
                    c++;
                }
            }
        }
    }
};
int main()
{
    A obj;
    obj.call();
    return 0;
}
