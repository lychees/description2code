#include<cstdio>
#include<cstdlib>
#include<algorithm>
#include<iostream>
using namespace std;
long long values[100010];
#define MODULUS 1000000007

int main()
{
    
  long long tc,num,result;  
 
  cin>>tc;
  while(tc--)
  {
               cin>>num;
               result=0;
               for(int i=0;i<num;i++)
               cin>>values[i];
               sort(values,values+num);  
               for(int i=num-1;i>=0;i--)
               result =(result%MODULUS + ((((values[i]*i)%MODULUS ) - (values[i] * (num-i-1))%MODULUS)+MODULUS) %MODULUS )%MODULUS;
               result=(result*values[num-1]%MODULUS)%MODULUS;
 
               cout<<result<<endl;

  }
 
  return 0;
  
  }
