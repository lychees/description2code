#include<stdio.h>
#include<algorithm>
#include<iostream>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,k;
        scanf("%d %d",&n,&k);
        int i,j,a[n];
        long long int s=0;
        for(i=0;i<n;i++)
            scanf("%d",&a[i]);
        sort(a,a+n);
        i=n-1;
        while(k>0 && i>=0)
        {
            s=s+a[i];
            i--;
            k--;
        }
        printf("%lld\n",s);
    }
    return 0;
}
