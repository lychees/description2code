#include <stdio.h>

int main()
{
    int k,i,j,q,pair=0,count,flag=0;
    char s[5000];
    scanf("%d%s",&q,&s);
    
    for(i=0;s[i]!='\0';i++){
    	
    	count=0;
    	
    	for(j=i;s[j]!='\0';j++){
    		
    		if(count>q){
    			break;
    		}
    		
    		for(k=i;k<j;k++){
    			
    			if(s[j]==s[k]){
    					break;
    			}
    		}
    		
    		if(k>=j){
    			count++;
    		}
    		
    		if(count==q){
    			  pair++;
    	
    		}
    	
    	}
    }
    printf("%d\n",pair);
}
