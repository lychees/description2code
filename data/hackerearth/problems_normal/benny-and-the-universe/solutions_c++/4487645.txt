#include <bits/stdc++.h>

using namespace std;

#define pb push_back
#define mp make_pair
#define REP(i, n) for (int i = 0; i < (int)(n); ++i)
typedef long long LL;
typedef pair<int, int> PII;

int n, q;
int a[1000];
int mod;
const int INF = 1e9 + 111;
int dist[10000];
bool used[10000] = {};

int main() {
    scanf("%d%d", &n, &q);
    REP(i, n) scanf("%d", a + i);
    sort(a, a + n);
    mod = a[0];
    REP(i, mod) dist[i] = INF;
    dist[0] = 0;
    while (true) {
        int v = -1;
        REP(i, mod) if (!used[i] && (v == -1 || dist[v] > dist[i])) {
            v = i;
        }
        if (v == -1 || dist[v] == INF) break;
        used[v] = true;
        REP(i, n) {
            int nv = (v + a[i]) % mod;
            dist[nv] = min(dist[nv], dist[v] + a[i]);
        }
    }
    REP(i, q) {
        int x;
        scanf("%d", &x);
        if (dist[x % mod] > x) {
            printf("NO\n");
        } else {
            printf("YES\n");
        }
    }
    return 0;
}
