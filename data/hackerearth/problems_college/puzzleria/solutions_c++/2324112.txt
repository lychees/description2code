#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define F(i, j, k) for(long long i=j; i>k; i--)
#define db(x) cout << x <<" "

ll fact(ll n)
{
	ll res =1;
	F(i, n, 1)
	{
		res*=i;
	}
	return res;
}
ll comb(ll n, ll i)
{
	ll res = 1;
	ll x=max(i, n-i);
	ll denom = min(i, n-i);
	F(i, n, x)
	{
		res*=i;
	}
	F(i, denom, 1)
	{
		res/=i;
	}
	//db(res);
	return res;
}
int main()
{
    int t; 
    scanf("%d", &t);
    while(t--)
    {
    	ll c, v;
    	scanf("%lld%lld", &c, &v);
    	ll ans = comb(7, c)*comb(5, v)*fact(c+v);
    	printf("%lld\n", ans);
    }
    return 0;
}
