#include <iostream>
#include <cstdio>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;

void solve() {
	int n;
	string s;
	cin >> n >> s;
	vector<int> c(256);
	for (int i = 0; i < c.size(); ++i)
		c[i] = count(s.begin(), s.end(), i);

	string name = "";
	int res = 0;
	int score = -1;
	for (int i = 0; i < n; ++i) {
		string t; cin >> t;
		int current = 0;
		
		for (int j = 0; j < t.length(); ++j) {
			current += c[t[j]];
		}

		if (current > score || (current == score && name.length() > t.length())) {
			score = current;
			res = i;
			name = t;
		}
	}
	cout << res + 1 << endl;
}

int main() {
	int t; cin >> t;
	while (t --> 0)
		solve();
	return 0;
}
