from heapq import heappush, heappop

n = int(raw_input())

line = raw_input()
words = line.split()
a = [int(word) for word in words]

heap = []

for i in range(n):
    heappush(heap, -a[i])

    # print(heap)

    if len(heap) < 3:
        print(-1)
        continue

    first = heappop(heap)
    second = heappop(heap)
    third = heappop(heap)

    print(-1 * first * second * third)

    heappush(heap, first)
    heappush(heap, second)
    heappush(heap, third)
