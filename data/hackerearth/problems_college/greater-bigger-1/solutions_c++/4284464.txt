#include <bits/stdc++.h>
using namespace std;
int minInd(string a, int i)
{
	int ret=-1;
	int cnt=0;
	for(int j=i+1;j<a.length();++j)
	{
		if(a.at(j)>a.at(i)&&cnt==0)
			{
					cnt=1;
					ret=j;
			}
			else if(a.at(j)>a.at(i)&&cnt==1)
			{
				if(a.at(j)<a.at(ret))
					ret=j;
			}
	}
	return ret;
}

int main()
{
    int test;
    cin >> test;
    while(test--)
    {
    	string w;
    	cin >> w;
    	int pos=0;
    	for(int i=w.length()-1;i>=0;--i)
    	{
    		int t=minInd(w,i);
    		if(t!=-1)
    		{
    			pos=1;
    			char temp=w.at(t);
    			w.at(t)=w.at(i);
    			w.at(i)=temp;
    			string g="";
    			for(int p=i+1;p<w.length();++p)
    				g+=w.at(p);
    			//cout << g << "\n";
    		
    			sort(g.begin(),g.end());
    			for(int p=i+1;p<w.length();++p )
    					w.at(p)=g.at(p-i-1);
    			break;
    		}
    	}
    	if(pos)
    		cout << w << "\n";
    	else
    		cout << "no answer\n";
    }
    return 0;
}
