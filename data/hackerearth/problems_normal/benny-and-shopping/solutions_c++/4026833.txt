#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    string s;
    char ch;
    ch=getchar();
    while(t--)
    {
        string res;
        getline(cin,s);
        bool flag=false;
        for(int i=0;i<s.length();++i)
        {
            if(((s[i]>='0') && (s[i]<='9')))
            {
                if(flag)
                {
                    res+=s[i];
                }
            }
            else
            {
                if(flag && s[i]!=' ')
                    break;
                else if(!flag && s[i]=='$')
                    flag=true;
            }
        }
        int hu=res.find_first_not_of('0');
        if(hu<res.length())
        {

        	cout<<"$"<<res.substr(hu,s.length()-hu+1)<<"\n";
        }

        else
            cout<<"$0\n";
    }
return 0;
}
