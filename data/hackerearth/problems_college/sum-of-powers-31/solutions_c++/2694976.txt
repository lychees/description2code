#include<bits/stdc++.h>
using namespace std;
set <int> pre() {
    set <int> s;
    s.insert(1);
    for(int i=2; i<=1000; i++) {
        int temp=i*i;
        while(temp<=1000000) {
            s.insert(temp);
            temp*=i;
        }
    }
 
    return s;
}
 
int main() {
    set <int> s=pre();
    int T;
    scanf("%d", &T);
    while(T--) {
        int N;
        scanf("%d", &N);
        set <int> ::iterator it;
        bool chk=false;
        for(it=s.begin(); it!=s.end(); it++) {
            int x=*it;
            if(s.find(N-x)!=s.end()) {
                chk=true;
                break;
            }
        }
 
        if(chk) {
            printf("Yes\n");
        }
 
        else {
            printf("No\n");
        }
    }
    return 0;
}
