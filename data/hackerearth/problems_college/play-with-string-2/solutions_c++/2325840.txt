#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin>>n;
    while(n--)
    {
    char s1[103],s2[103];
    cin>>s1;

    int l=strlen(s1);
    int z=l;
    for(int i=0;i<l;i++)
    {
       s2[i]=s1[z-1];
       z--;
    }int c=0;
    for(int i=0;i<l-1;i++)
    {
        if(abs(s1[i+1]-s1[i])==abs(s2[i+1]-s2[i]))c++;
    }
    if(c==l-1)cout<<"Funny"<<endl;
    else cout<<"Not Funny"<<endl;

    }
    return 0;
}
