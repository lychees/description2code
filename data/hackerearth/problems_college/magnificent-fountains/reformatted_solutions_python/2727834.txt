def gcd(a, b):
    if (a < b):
        return gcd(b, a)
    if (a % b == 0):
        return b
    else:
        return gcd(b, a % b)


def lcm(a, b):
    return a*b/(gcd(a, b))


test = input()
for i in range(test):

    n = input()
    arr = map(int, raw_input().split())
    lcm_result = arr[0]
    for i in range(1, n):
        lcm_result = lcm(lcm_result, arr[i])
    print lcm_result % 1000000007
