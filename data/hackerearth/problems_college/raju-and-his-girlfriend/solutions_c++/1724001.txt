#include <cstdio>
#include <cstring>

int a[255], b[255];
char s[1010];

void init(){
	for (int i = 'a'; i <= 'o'; i++){
		a[i] = (i - 97) % 3 + 1;
		b[i] = (i - 97) / 3 + 2;
	}
	a['p'] = 1; a['q'] = 2; a['r'] = 3; a['s'] = 4;
	b['p'] = b['q'] = b['r'] = b['s'] = 7;
	a['w'] = 1; a['x'] = 2; a['y'] = 3; a['z'] = 4;
	b['w'] = b['x'] = b['y'] = b['z'] = 9;
	a['t'] = 1; a['u'] = 2; a['v'] = 3;
	b['t'] = b['u'] = b['v'] = 8;
	a['_'] = a['.'] = 1; a[','] = 2; a['?'] = 3; a['!'] = 4;
	b['_'] = 0;
	b['.'] = b[','] = b['?'] = b['!'] = 1;
	for (int i = '0'; i <= '9'; i++)
		b[i] = i - '0';
	a['1'] = a['7'] = a['9'] = 5;
	a['0'] = 2;
	a['2'] = a['3'] = a['4'] = a['5'] = a['6'] = a['8'] = 4;
}

int main(){
	init();
	int t, pos, cur, len;
	int x, y, n;
	scanf("%d", &t);
	while (t--){
        scanf("%d%d%d", &x, &y, &n);
		scanf("%s", s);
		pos = 1; cur = 0;
		len = strlen(s);
        while (cur < len && n > 0){
            if (a[s[cur]]*y + (b[s[cur]] != pos)*x > n)
                break;
            printf("%c", s[cur]);
            n -= a[s[cur]]*y + (b[s[cur]] != pos)*x;
            pos = b[s[cur]];
            cur++;
        }
		printf("\n");
	}
	return 0;
}
