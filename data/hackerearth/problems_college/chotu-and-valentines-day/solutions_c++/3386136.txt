#include <stdio.h>

int main()
{
	int N, K, Q, L, R, x;
	long long sum;
    scanf("%d %d",&N,&Q);
    int A[N+1];
    A[0] = 0;
    for(int i=1;i<=N;i++){
    	scanf("%d",&x);
    	A[i] = A[i-1] + x;
    }
    for(int i=0;i<Q;i++)
    {
    	scanf("%d %d",&L,&R);
    	printf("%d\n",A[R]-A[L-1]);
    }
    return 0;
}
