mInt dec(mInt a, mInt b){
	if(a == 1 && b == 1) return 0;
	if(a == 1) return b-1;
	if(b == 1) return a-1;
	if(a > b) return (a/b) + dec(a%b, b);
	else return (b/a) + dec(a, b%a);
}

void decrement(mInt* a, mInt* b, int* counter){
	if(*a > *b){
		*a = *a - *b;
		(*counter)++;
		decrement(a, b, counter);
	}
	else if(*a < *b){
		*b = *b - *a;
		(*counter)++;
		decrement(a, b, counter);
	}
}
