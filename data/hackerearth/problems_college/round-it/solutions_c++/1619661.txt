#include<bits/stdc++.h>
#define ll long long
#define f(a,b,c) for(ll int i=a;i<b;i+=c)
using namespace std;

int main() {
	cin.tie(0);
	ios_base::sync_with_stdio(false);
	ll int T,n,b;
	cin >> T;
	while(T--) {
		cin >> n >> b;
		ll int near = (ll int)(n/b);
		near*=b;
		ll int far = near + b;
		if(n-near>=far-n)
			cout << far << endl;
		else
			cout << near << endl;
	}
	return 0;
}
