#include<iostream>
#include<cstdio>
#include<vector>
#include<cstring>
#include<cmath>
#include<algorithm>
using namespace std;
typedef unsigned long long int ull;
typedef long long int ll;

int main()
{
	int t;
	cin >> t;
	int n,m,k;
	while(t--){
		cin  >> m>>n>>k;
		double arr[m+1][k+1];//= {{0.0}};
		memset(arr,0.0,sizeof(arr));
		for(int i = 1;i <= k && i <= n;i++){
			arr[1][i] = (double)1/n;
		}
		for(int i = 2;i <= m;i++){
			
			for(int j = 1; j <= k;j++){
				for(int kk = 1;kk < j && kk <= n;kk++){
					arr[i][j] = arr[i][j]+arr[i-1][j-kk];
				}
				arr[i][j]/=n;
			}
		}
		double rest = (double)arr[m][k];
		int des = 0;
		 if(rest==0){
      printf("0.000 0\n");
    }else{
		while(rest < 1.0){
			rest *= 10.0;
			des++;
		}
		printf("%.3lf %d\n",rest,des);
		//print(res,an);
	}}
	return 0;
}
