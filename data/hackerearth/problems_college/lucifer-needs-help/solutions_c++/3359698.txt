#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,s;
	cin>>n>>s;
	int a[n];
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
	}
	int c=0;
	sort(a,a+n);
	for(int j=n-1;j>=2;j--)
	{
		int l = 0;
		int r = j-1;
		while(l<r)
		{
			int sum = a[l] + a[r] + a[j];
			if(sum == s)
			{
				cout<<"YES"<<"\n";
				c=1;
				break;
			}
			else
			{
				if(sum > s)
				{
					r--;
				}
				else
				{
					l++;
				}
			}
			
		}
		if(c==1)
		{
			break;
		}
	}
	if(c==0)
	{
		cout<<"NO"<<"\n";
	}
	
}
