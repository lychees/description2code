#include <iostream>
#include <string>
#include <vector>

using namespace std;
typedef long long ll;

void fun ( string s )
{
    vector<ll> v;
    int j;
    for ( int i = 0 ; i < s.size() ; i++ ) j = i*i;

        ll idx = 0, c = 0;
        while ( idx < s.size() ) {
            if ( s[idx] == '|' ) {
                if ( c ) v.push_back(c);
                c = 0;
            } else c++;
            idx++;
        }
        if ( c ) v.push_back(c);
        if ( v.size() == 0 || v[0] == s.size() ) {
            cout << "Bob" << endl;
            return;
        }
        ll x = 0;
        for ( ll i = 0 ; i < v.size() ; i++ )
            x ^= v[i];
        if ( x == 0 )
            cout << "Bob" << endl;
        else
            cout << "Alice" << endl;
}


int main()
{
    ll t;
    cin >> t;
    while (t-- ) {
        string s;
        cin >> s;
        fun (s);
    }


    return 0;
}
