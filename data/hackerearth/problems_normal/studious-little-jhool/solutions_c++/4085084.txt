#include <bits/stdc++.h>
using namespace std;

int main() {
   	int t,n;
   	int sol[114];
   	memset(sol,-1,sizeof(sol));
   	sol[10] = 1;
   	sol[12] = 1;
   	for(int i = 20; i < 114; i++) {
   		if(sol[i-10] != -1) {
   			sol[i] = sol[i-10]+1;
   		}
   		if(sol[i-12] != -1) {
   			if(sol[i] == -1) {
   				sol[i] = sol[i-12]+1;
   			} else {
   				sol[i] = min(sol[i],sol[i-12]+1);
   			}
   		}
   	}
   	scanf("%d", &t);
   	for(int i = 0; i < t; i++) {
   		scanf("%d", &n);
   		printf("%d\n", sol[n]);
   	}
    return 0;
}
