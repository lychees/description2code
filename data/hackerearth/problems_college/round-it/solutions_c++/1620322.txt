#include <algorithm>
#include <cstdio>
#include<iostream>
#include <array>
#include<bits/stdc++.h>
#define ll long long unsigned int
using namespace std;
int main()
{
    ll t=0,a,b,n=0;
cin>>t;
while(t--!=0)
{
cin>>a>>b;
if(a<b)
{cout<<b;
cout<<endl;
continue;
}
n=a%b;
ll a1=a-n;
ll b1=a1+b;
ll a2=a-a1;
ll b2=b1-a;
if(a2<b2)
{cout<<a1;
cout<<endl;
continue;
}
if(b2<a2)
{cout<<b1;
cout<<endl;
continue;
}
if(b2==a2)
{cout<<b1;
cout<<endl;
continue;
}
}
    return 0;
}
