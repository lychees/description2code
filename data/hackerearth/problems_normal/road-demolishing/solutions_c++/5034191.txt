#include<iostream>
using namespace std;
main()
{

	int t;
	cin>>t;
	for(int z=0;z<t;z++)
	{
		long long int n,q;
		cin>>n>>q;
		q--;
		long long int ans;
		ans=n*n-(n%q)*((n+q-1)/q)*((n+q-1)/q);
		ans=ans-(q-n%q)*(n/q)*(n/q);
		ans=ans/2;
		ans=(n*(n-1))/2-ans;
		cout<<ans<<endl;



	}


}
