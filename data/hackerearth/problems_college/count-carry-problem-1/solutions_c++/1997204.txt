#include <iostream>
using namespace std;

int main()
{
    int t=0;
    scanf("%d",&t);
    while(t--)
    {
    	long long a=0,b=0,c=0,carry=0,r=0,s=0;
    	scanf("%lld%lld",&a,&b);
    	if(a==0&&b==0){return 0;}
		c=0;carry=0;
		while(a!=0||b!=0){
			if((a%10+b%10+carry)>9){c++;carry=1;}
			else	carry=0;
			a/=10;b/=10;
		}
    	if(c==0)
    	printf("No carry operation\n");
    	else if(c==1)
    	printf("1 carry operation\n");
    	else
    	printf("%lld carry operations\n",c);
    }
    return 0;
}
