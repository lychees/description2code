#include <iostream>
using namespace std;

int main()
{
	typedef unsigned long int uli;
	uli T=0, R=0, C=0, sum=0, l=0, n=0;
	cin >> T;
	while(T--)
	{
		sum=0;
		cin >> R >> C;
		if(R<C)
			n=R;
		else
			n=C;
		l=((n-1)*C)+n;
		sum+=((n*(1+l))/2);
		l=((n-1)*R)+n;
		sum+=((n*(1+l))/2);
		cout << sum << endl;
	}
    return 0;
}
