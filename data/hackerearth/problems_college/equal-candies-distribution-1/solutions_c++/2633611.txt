	 int arr[10005];
	 int dif[10005];
	 
    int main()
    {
    	int i,j,k,n,m,t,mini;
    	scd(t);
		while(t--)
		{
			scd(n);
			mini = INT_MAX;
			For(i,0,n)
			{
				scd(arr[i]);
				if(arr[i] < mini)
				{
					mini = arr[i];
				}
			}
			For(i,0,n)
			{
				dif[i] = arr[i] - mini;
			}
			li c =0;
			For(i,0,n)
			{
				if(dif[i] >= 5)
				{
					c += dif[i]/5;
					dif[i] %= 5;
				}
				if(dif[i] >= 2)
				{
					c += dif[i]/2;
					dif[i] %= 2;
				}
				if(dif[i] == 1)
				{
					c++;
					dif[i] --;
				}
			}
			cout<<c<<endl;
		}
    	return 0;
    }
