#include<iostream>
using namespace std;
int main()
{
	int test;
	cin>>test;
	if(test<1 || test>10) return 0;
	while(test--){
		int i,j,temp,n;
		cin>>n;
		int a[n];
		for(i=0;i<n;i++)
		cin>>a[i];
		for(i=0;i<n;i++)
		for(j=0;j<n-i-1;j++)
		if(a[j]>a[j+1]){
			temp=a[j];
			a[j]=a[j+1];
			a[j+1]=temp;
			
		}
		int k=0,c=0;
		while(k!=n){
			if(a[k]!=a[k+1]){
				c++;
				k++;
			}else k++;
		}
		cout<<c<<"\n";
	}
}
