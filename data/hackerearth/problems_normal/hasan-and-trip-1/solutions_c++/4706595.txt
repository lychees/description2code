#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define ld long double
#define MAxpos 3100

ll xpos[MAxpos],ypos[MAxpos],f[MAxpos];
ld cost[MAxpos];
int main()
{
    int n;
    cin>>n;

    for(int i=1;i<=n;i++)
        cin>>xpos[i]>>ypos[i]>>f[i];

    cost[1]=f[1];

    for(int i=2;i<=n;i++)
    {
        cost[i]=INT_MIN;
        for(int j=1;j<i;j++)
            cost[i]=max(cost[i],cost[j]-sqrt((long double) ((xpos[i]-xpos[j])*(xpos[i]-xpos[j]) + (ypos[i]-ypos[j])*(ypos[i]-ypos[j]))));
        cost[i]+=f[i];
    }
    cout<<fixed;
    cout.precision(6);
    cout<<cost[n];
    return 0;
}
