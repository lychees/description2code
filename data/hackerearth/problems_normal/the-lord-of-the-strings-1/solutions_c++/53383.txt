#include <iostream>
#include <string.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    int mymap[26];
    for(int i=0;i<=t;i++){
        memset(mymap,0,sizeof(mymap));
    	string s;
    	cin>>s;
    	for(int j=0;j<s.size();j++){
    		if(mymap[s[j]-'a']==0) {
    			cout<<s[j];
    			mymap[s[j]-'a']=1;
    		}
    	}
    	cout<<endl;
    }
    return 0;
}
