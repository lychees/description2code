#include <iostream>
#include <algorithm>
#include <cstdio>
using namespace std;
#define lim 100000
long long cum[105];
long long F[105];
int main()
{
    int i,T;
    long long N,fib;
    F[1]=1;
    F[2]=2;
    F[3]=3;
    for(i=4;i<=100;i++)
    {
    	F[i]=F[i-1]+F[i-2];
    	cum[i-3]=cum[i-4]+F[i]-F[i-1]-1;
    }
   
    //now binary_search
    scanf("%d",&T);
    int pos;
    while(T--)
    {
    	scanf("%lld",&N);
     	pos=lower_bound(cum+1,cum+100,N)-(cum);
     	//you will get the pos
        
        
     	//now we are at some pos and now we find closest fibonacci no
     	fib=F[pos+2];
     	
     	N=N-cum[pos-1];
     	
     	printf("%lld\n",fib+N);
     	
     	
    }
    return 0;
}
