lll get_pow(lll a, lll b)
{
    lll x=1,y=a; 
    while(b > 0)
    {
        if(b%2 == 1)
        {
            x=(x*y);
            if(x>MOD) x%=MOD;
        }
        y = (y*y);
        if(y>MOD) y%=MOD; 
        b /= 2;
    }
    return x;
}
lll gcd(lll a,lll b)
{
   if(b==0)
   return a;
   return gcd(b,a%b);
}
