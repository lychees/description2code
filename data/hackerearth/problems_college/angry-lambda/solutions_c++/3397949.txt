#include <iostream>
using namespace std;

int main()
{
    int i,j,k,l,p,m,t;
    cin>>t;
    while(t--)
    { k=0;
    	cin>>p>>m;
    	for(i=0;i<p;i++)
    	   {
    	   	cin>>j;
    	   	if(j<=0)
    	   	   k++;
    	   }
    	   if(k>=m)
    	      cout<<"NO\n";
    	   else
    	      cout<<"YES\n";
    }
    return 0;
}
