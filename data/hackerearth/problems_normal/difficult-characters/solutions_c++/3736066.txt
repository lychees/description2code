#include<iostream>
#include<cstdio>
using namespace std;

int frequency[26];

int main()
{
	int t;
	cin>>t;
	string str;
	while(t--)
	{
		for(int i=0;i<=25;i++)
			frequency[i]=0;
		cin>>str;
		for(int i=0;i<str.length();i++)
			frequency[str[i]-'a']++;
		int frequentChar[26];
		for(int i=0;i<26;i++)
			frequentChar[i]=i;
		for(int i=0;i<25;i++)
		{
			for(int j=0;j<25-i;j++)
			{
				if(frequency[j]<frequency[j+1])
				{
					int temp1=frequency[j];
					frequency[j]=frequency[j+1];
					frequency[j+1]=temp1;
					int temp2=frequentChar[j];
					frequentChar[j]=frequentChar[j+1];
					frequentChar[j+1]=temp2;
				}
			}
		}
		for(int i=25;i>=0;i--)
			printf("%c ", frequentChar[i]+'a');
		cout<<endl;
	}
	return 0;
}
