#include<string.h>
#include <iostream>
#include<stdlib.h>
using namespace std;
char upper(char s)
{
	if(s>91)
	return s-32;
	else
	return s;
}
int main()
{
	int k,temp=0,i,j,w;
	char s[100];
	cin>>k;
	char p[k][100];
	for(i=0;i<k;i++)
	cin>>p[i];
	cin>>w;
	for(i=0;i<w;i++)
	{
		temp=0;
		cin>>s;
		for(j=0;j<k;j++)
		{
			if(!strcmp(s,p[j]))
			{
				temp=1;
				break;
			}
		}
			if(!temp)
			{
			cout<<upper(s[0]);
			break;
			}
	}
	for(++i;i<w;i++)
	{
		temp=0;
		cin>>s;
		for(j=0;j<k;j++)
		{
			if(!strcmp(s,p[j]))
			{
				temp=1;
				break;
			}
		}
			if(!temp)
			{
			cout<<"."<<upper(s[0]);
			}
	}
}
