#include<bits/stdc++.h>
using namespace std;
int main()
{
             int n,a,d,i;
             scanf("%d",&n);
             vector<pair<int,int> > v(n+1);
             for(i=1;i<n+1;i++)
             {
                          scanf("%d%d",&a,&d);
                          v[i].first=a+d;
                          v[i].second=i;
             }
             sort(v.begin()+1,v.end());
             for(i=1;i<n+1;i++)
             {
                          printf("%d ",v[i].second);
             }
             return 0;
}
