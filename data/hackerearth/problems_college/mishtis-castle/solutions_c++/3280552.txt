#include<bits/stdc++.h>
using namespace std;
int dp[103][103]={0};
int a[103][103]={0};
int c;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d %d",&n,&c);
		for(int i=1;i<=n;++i)
		{
			for(int j=1;j<=n;++j)
			{
				cin>>a[i][j];
			}
		}
		//reset the dp
		int i,j;
		dp[1][1]=0;
		for( j=2;j<=n;++j)
			dp[1][j]=dp[1][j-1]+a[1][j];//top row dp finish
		for(i=2;i<=n;++i)//asdfasdf
			dp[i][1]=dp[i-1][1]+a[i][1];
		for(i=2;i<=n;++i)
			for(j=2;j<=n;++j)
			{
				dp[i][j]=a[i][j]+max(dp[i-1][j],dp[i][j-1]);
			}
		/*for(int i=1;i<=n;++i)
		{
			for(int j=1;j<=n;++j)
			{
				printf("%d ",dp[i][j]);
			}
			printf("\n");
		}*/
		if(dp[n][n]-c>=0)printf("%d\n",dp[n][n]-c);
		else printf("-1\n");
	}


	return 0;
}
