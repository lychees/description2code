#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<cstring>
using namespace std;
#define lim 20
int i, cs, x, y, p, j;
int C[lim][lim];
int main()
{
  for (i = 0; i < lim; i++)
    for (C[i][0] = C[i][i] = j = 1; j < i; j++)
      C[i][j] = C[i - 1][j - 1] + C[i - 1][j];
  scanf("%d",&cs);
  while (cs--)
  {
    scanf("%d%d",&x,&y);
    p = C[x+y][x];
    for (i = 0; i < x; i++)
      p *= 7 - i;
    for (i = 0; i < y; i++)
      p *= 5 - i;
    printf("%d\n",p);
  }
  return 0;
}
