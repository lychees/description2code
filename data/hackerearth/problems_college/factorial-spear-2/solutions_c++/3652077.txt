#include <stdio.h>

int sp[10000001];
int facPrime[10000001];

void sieve() 
{
	long long int i, j;
	for(i = 2; i < 10000001; i++) {
		if(!sp[i]) {
			for(j = i; j * i < 10000001; j++) {
				sp[j * i] = i;
			}
		}
	}
}

int main()
{
	sieve();

	facPrime[2] = 1;
	for(int i = 3; i < 10000001; i++) {
		
		facPrime[i] = facPrime[i - 1];
		if(!sp[i]) {
			facPrime[i]++;
		} else {
			int f = i;
			int count = 0;
			while(f && sp[f]) {
				f /= sp[f];
				count++;
			}
			if(f) count++;
			facPrime[i] += count;
		}
	}

	int cases;
    scanf("%d", &cases);
    while(cases--) {
    	int n;
    	scanf("%lld", &n);
    	printf("%d\n", facPrime[n]);
    }
    return 0;
}
