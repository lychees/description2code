#include <bits/stdc++.h>

using namespace std;

#define M 1000000007

long long tab[1000005];

long long VAM(int n)
{
	//cout<<n<<endl;
	if(n==0)
		return 1;
	
	if(n<0)
		return 0;
	
	if(tab[n-1]==-1)
		tab[n-1] = VAM(n-1)%M;
		
	if(tab[n-2]==-1)
		tab[n-2] = VAM(n-2)%M;
		
	tab[n] = (tab[n-1] + (n-1)*tab[n-2])%M;
	
	return tab[n];
	
	/*
	
	
	
	
	//cout<<n<<" "<<tab[n-2]<<endl;
	
	long long x = 1;
	
	if(tab[n-2]>0)
		x = (n-1+tab[n-2])%M;
	
	tab[n] = (tab[n-1]+x)%M;
	
	return tab[n];
	*/
}

int main()
{
    int n, t;
    memset(tab, -1, sizeof(tab));
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	cout<<VAM(n)<<endl;
    }
    return 0;
}
