#include <iostream>
#include <stdio.h>
#include <string.h>
using namespace std;

typedef long long LL;


int main()
{
    int n,q;
    cin>>n>>q;

    string s;
     cin>>s;


    while(q>0)
    {
        --q;
        int i,j; cin>>i>>j;

        int p1=i; int p2=j;
        int flag=0;
        while(true)
        {




            if(s[p1-1]== '1' || s[p2-1]== '1')
            {
                flag=1;
                break;
            }

            if(p1>p2)
            {
                p1 = p1/2;
            }
            else if(p2>p1)
            {
                p2 =p2/2;
            }
            else if(p1==p2)
            {
                break;
            }

        }

        if(flag==0)
        {
            cout<<"Yes"<<endl;
        }
        else
        {
            cout<<"No"<<endl;
        }


    }



}
