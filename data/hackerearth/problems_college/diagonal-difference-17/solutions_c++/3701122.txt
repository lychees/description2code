#include <iostream>
using namespace std;

int main()
{
    long long n,sum1=0,sum2=0;
    cin>>n;
    long long A[n][n];
    for(int i=0;i<n;i++)
    {
    	for(int j=0;j<n;j++)
    	{
    		cin>>A[i][j];
    	}
    }
    for(int i=0;i<n;i++)
    {
     sum1+=A[i][i];	
    }
    for(int i=0;i<n;i++)
    {
    	for(int j=0;j<n;j++)
    	{
    		if(i+j==n-1)
    		sum2+=A[i][j];
    	}
    }
    if(sum1-sum2<0)
    cout<<-(sum1-sum2);
    else
    cout<<sum1-sum2;
    
    return 0;
}
