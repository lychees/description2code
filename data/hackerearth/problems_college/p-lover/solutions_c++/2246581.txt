#include <iostream>
#include <cmath>
using namespace std;
#define ll long long
int IS_prime(ll num)
{
	int isprime = 1;
	for(ll i = 2; i <= sqrt(num); i++)
	{
		if(num % i == 0)
		{
			isprime = 0;
			break;
		}
	}

	return isprime;
}
int checkpal(ll n)
{
   ll reverse = 0,temp = n;
 
   while (temp != 0)
   {
      reverse = reverse * 10;
      reverse = reverse + temp%10;
      temp    = temp/10;
   }
 
   if (n == reverse)
      return 1;
   return 0;
}	
int main()
{
    int t;
    cin >> t;
    long long n,i;
    while(t--)
    {
    	cin >> n;
    	for(i = n;1;i++)
    		if(checkpal(i) && IS_prime(i))
    		  break;
    	cout << i << '\n';	  
    }
    return 0;
}
