#include<bits/stdc++.h>
#define p1 32452843
#define p2 10000007
using namespace std;
typedef long long ll;
char s[1000000];
ll len;
inline ll mod(ll a,ll b)
{
    return (a%b+b)%b;
}
bool isAvail(int m)
{
    unordered_multimap<ll,ll> mp;
    unordered_multimap<ll,ll>::iterator it;
    ll t1=0,t2=0,e1=1,e2=1;
    for(int i=0;i<m;i++)
    {
        t1=(t1*27+s[i])%p1;
        t2=(t2*27+s[i])%p2;
    }
    mp.insert({t1,t2});
    for(int i=0;i<m-1;i++)
    {
        e1=(e1*27)%p1;
        e2=(e2*27)%p2;
    }
    for(int i=0;i<=len-m;i++)
    {
        t1 = (mod((27*(t1 - (s[i]*e1)%p1)),p1) + s[i+m])%p1;
        t2 = (mod((27*(t2 - (s[i]*e2)%p2)),p2) + s[i+m])%p2;
        if((it=mp.find(t1))==mp.end())
            mp.insert({t1,t2});
        else
        {
            int cnt=0,cn=mp.count(t1);
            while(cn--)
            {
                if(it->second==t2)
                    cnt++;
                it++;
            }
            if(cnt==0)
                mp.insert({t1,t2});
            else
                return true;
        }
    }
    return false;
}
int main()
{
    scanf("%s",s);
    int l=0,r=strlen(s)-1,m;
    len=r;
    while(l<r)
    {
        m=(l+r+1)/2;
        if(isAvail(m))
            l=m;
        else
            r=m-1;
    }
    m=(l+r+1)/2;
    printf("%d",m);
}
