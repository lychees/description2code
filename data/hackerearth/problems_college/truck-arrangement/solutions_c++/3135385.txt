#include <stdio.h>
#include<iostream>
#include<cmath>
#include <algorithm>
using namespace std;
pair<int,int> a[10000];
int main()
{
        int test_cases;
        scanf("%d",&test_cases);
        while(test_cases > 0)
        {
                int N,i=0;
                scanf("%d",&N);
                while(i<N)
                {
                        scanf("%d",&a[i].first);
                        a[i].second=i;
                        i = i+1;
                }
                sort(a,a+N);
               int maxx=0;
               for(i=0;i<N;i++)
               {
              // 	cout<<a[i].first;
               		if(abs(a[i].second-i)>maxx)
               			maxx=abs(a[i].second-i);
               }
             cout<<maxx<<endl;
                test_cases--;
        }
        return 0;
}
