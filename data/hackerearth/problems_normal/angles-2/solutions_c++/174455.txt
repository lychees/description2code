#include <iostream>
#include <cstdio>
#include <cstdlib>
using namespace std;
int findAngle (int theta , int phi);
int main()
{
    int N, K;
    cin >> N >> K;
    int i, j;
    int NArr[N] ,KArr[K];
    for(i = 0; i < N; i++)
        cin >> NArr[i];
    for(i = 0; i < K; i++)
        cin >> KArr[i];
    int result = 0;
    for( j = 0; j < K; j++)
    {
        result = 0;
        for (i = 0; i < N; i++)
        {
            result = findAngle(NArr[i], KArr[j]);
            if(result == 1)
                break;
        }
        if (result == 1)
            cout << "YES" <<endl;
        else
            cout << "NO" <<endl;
    }
    return 0;
}
int findAngle (int theta , int phi)
{
    int i, res = 0;
    for ( i = 0; i <= 360; i++)
    {
        if ( (theta * i) %360 == phi)
        {
            res = 1;
            break;
        }
    }
    return res;
}
