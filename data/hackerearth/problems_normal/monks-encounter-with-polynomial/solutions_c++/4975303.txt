#include<stdio.h>
#include<iostream>
#include<algorithm>
#include<math.h>
using namespace std;
long long int fun(long long int a, long long int b, long long int c, long long int k, int start, int en)
{

    if(start<=en)
    {
        long long int mid;
        mid = (start+en)/2;
        long long int sum1;
        sum1 = (a*(mid*mid)+(b*mid)+c);
        if(sum1>k)
        {
            return fun(a,b,c,k,start, mid-1);
        }else if(sum1<k)
        {
            return fun(a,b,c,k,mid+1, en);
        }
        else if(sum1 == k)
            return mid;
    }
    return start;
}
