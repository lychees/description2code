#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;
//int array[50030];
int main() {
    int t;
    cin>>t;
    while(t--){
        long long int n;
        cin>>n;
        int array[n+1];
        for(long long int i=1;i<=n;i++){
            cin>>array[i];
        }
        long long int j = array[n];
        long long int sum = 0;
        long long int ans = 0;
        long long int count = 0;
        for(long long int i=n-1;i>=1;i--){
            if(array[i] >= j){ ans = ans - sum + count*j; count  = 0; sum = 0; j = array[i];}
            else{
                sum += array[i];
                count++;
            }
        }
        //cout<<count<<" "<<j<<" "<<sum<<endl;
        if(count > 0) ans = ans - sum + count*j;
        cout<<ans<<endl;
    }
    return 0;
}
