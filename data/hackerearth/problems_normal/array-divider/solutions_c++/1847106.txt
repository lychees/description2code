typedef long long ll;
typedef pair<int,int> ii;

const ll inf=1e17;
const int MOD=1e9+7;
const int N=5e2+10;

int n,m,a[N];
ll f[305][N];
int main(){
    scanf("%d%d",&n,&m);
    FOR(i,1,n) scanf("%d",a+i);
    FOR(i,1,n) a[i]+=a[i-1];
    FOR(i,1,n) f[0][i]=inf;
    FOR(i,1,m)
        FOR(j,i,n){
            f[i][j]=inf;
            FORD(k,j-1,i-1) f[i][j]=min(f[i][j],f[i-1][k]+1LL*(m-1)*(a[j]-a[k])*(a[j]-a[k])-2LL*a[k]*(a[j]-a[k]));
        }
    //FOR(i,1,m)
        //FOR(j,0,n) cout<<f[i][j]<<" \n"[j==n];
    cout<<f[m][n];
}
