#include <bits/stdc++.h>
using namespace std;

int main() {
	long long int t,i,j,k,l,m,n,a,b,c;
	cin>>t;
	while(t--)
	{
		cin>>a>>b>>c;
		k=pow(a,b);
		
	//	cout<<k<<" ";
		
		if(k<1)
		{
			cout<<"0\n";
			continue;
		}
		
		else
		{
			i=1;
			
			while(i*c<k)
			{
				i++;
			}
			
			
			//cout<<i<<" ";
		
		
		m=c*i;
		n=c*(i-1);
		
		i=(m-k);
		j=(k-n);
		
		if(i<j)
		cout<<m<<"\n";
		else
		cout<<n<<"\n";
		}
	}
}
