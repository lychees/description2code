#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll mat[1001][1001] ;
int main()
{
	int t;
	string s;

	scanf("%d",&t) ;

	while(t--)
	{
		cin>>s;

		int j,n=s.size() ;

		for(int i=1;i<=n;i++)mat[i][i]=1;


		for(int ln=1;ln<n;ln++)
		{
			for(int i1=0;i1+ln<n;i1++)
			{
				int i=i1+1;
				j=i1+ln+1;

			//	cout<<"["<<i<<"-"<<j<<"]"<<" ";

				mat[i][j] =  mat[i][j-1] - mat[i+1][j-1] ;  //staring from i to before j

				mat[i][j]+= mat[i+1][j] - mat[i+1][j-1] ;  //starting after i and ending on j

				mat[i][j]+= mat[i+1][j-1] ;             //staring after i and ending before j

				if(s[i-1]==s[j-1])
					mat[i][j]+= mat[i+1][j-1] + 1 ;

			}
		//cout<<"\n" ;
		}

		printf("%lld\n",mat[1][n]) ;

	}

}
