#include <iostream>
#include <string>
using namespace std;
const int m=1000000007;
int main()
{
	long long t,sum,n;
	string a;
   cin>>t;
   while(t--){
   	cin>>n;
     cin>>a;
     sum=0;
   	for(int i=0;i<n;i++){
   		sum=(((sum%m)*(10%m))%m+(a[i]-'0')%m)%m;
  
   	}
   	cout<<sum<<endl;
   }
    return 0;
}
