fib = [1, 1]
c = 10**15


def fill_fibo():
    l = 1
    while (fib[l] < c):
        fib.append(fib[l]+fib[l-1])
        l += 1
    return


fill_fibo()
fib.pop(0)
for _ in xrange(input()):
    res = input()
    for i in fib:
        if (i <= res):
            res += 1
        else:
            break
    print res
