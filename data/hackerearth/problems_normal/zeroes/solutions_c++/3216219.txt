#include <bits/stdc++.h>
using namespace std;
#define PB push_back
#define MP make_pair

typedef pair<int, int> II;
typedef vector<int> VI;
typedef vector<II> VII;

VI primos;

void getprimos( int lim ){
	primos.PB( 2 );
	vector<bool>bs(lim, 1);

	for( int i=4; i<lim; i+=2 ){
		bs[i] = 0;
	}

	for( int i=3; i<1000; i+=2){
		if( bs[i] ){
			primos.PB(i);
			for( int j=i*i; j<lim; j+=i){
				bs[j]=0;
			}
		}
	}

}

VII facto( int x ){
	VII fact;
	for(int idx=0, cnt; idx<primos.size() && 1<x ; idx++ ){
		cnt = 0;
		while( x%primos[idx]==0 ){
			cnt++;
			x/=primos[idx];
		}
		if( cnt>0 )
			fact.PB( MP(primos[idx], cnt) );
	}
	if( x>1 )
        fact.PB( MP(x, 1) );
    return fact;
}

int main(){
	int lim = 100010;
    getprimos( lim );
    for( int N, Q, aux, x; cin>>N; ){
    	map<int, int>cnt;

    	for( int i=0; i< N; i++) {
    		cin>>aux;

    		for(int idx=0; idx<primos.size() && aux>1; idx++ ){
    			while( aux%primos[idx]==0 ){
    				cnt[ primos[idx] ]++;
    				aux/=primos[idx];
    			}
    		}
    		if( aux>1 )
                cnt[ aux ]++;
    	}

    	cin>>Q;
    	for( int i=0, res; i<Q;i++){
    		cin>>x;
    		res = 1000000;
    		VII fac = facto(x);
    		for( int idx=0; idx<fac.size(); idx++){
    			res = min(res, cnt[ fac[idx].first ]/fac[idx].second );
    		}

    		cout<<res<<"\n";
    	}
    }
    return 0;
}
