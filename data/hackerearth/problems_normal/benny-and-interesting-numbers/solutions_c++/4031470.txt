#include <bits/stdc++.h>
using namespace std;

const int N = 320000;
vector < int > primes;
bool isPrime[N];
int t;
long long n;

inline void solve(long long x){
	for(int i = 0; i < (int) primes.size(); i++){
		long long pr = primes[i];
		int count = 0;
		while(x % pr == 0){
			x /= pr;
			count = count + 1;
		}
		if(count > 1){
			cout << "No\n";
			return;
		}
	}
	cout << "Yes\n";
}

int main(){
	for(int i = 1; i < N; i++) isPrime[i] = true;
	isPrime[1] = false;
	for(int i = 2; i < N; i++){
		if(isPrime[i]){
			primes.push_back(i);
			for(int j = i + i; j < N; j += i)
				isPrime[j] = false;
		}
	}
	cin >> t;
	while(t--){
		cin >> n;
		solve(n);
	}
}
