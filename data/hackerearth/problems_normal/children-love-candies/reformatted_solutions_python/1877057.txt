import sys


def main():
    I = sys.stdin.readlines()
    x = int(I[0])
    for i in xrange(x):
        n, t = map(int, I[i+1].split())
        for j in xrange(t):
            if (n % 2 == 0):
                n = n/2
            else:
                n = n-((n+1)/2)
            n = n-(n//4)
        print n


main()
