
import sys

coin_val_map = {}


def recurse(inp):
    if inp in coin_val_map:
        return coin_val_map[inp]

    if (inp/2 + inp/3 + inp/4) < inp:
        return inp

    coin_val = recurse(inp/2) + recurse(inp/3) + recurse(inp/4)
    coin_val_map[inp] = coin_val
    return coin_val


if __name__ == '__main__':
    data = (sys.stdin.readlines())
    for inp in data:
        max_coin_val = recurse(int(inp))
        print max_coin_val
