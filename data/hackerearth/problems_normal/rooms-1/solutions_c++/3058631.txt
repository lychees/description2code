#include <bits/stdc++.h>
using namespace std;
int arr[100005];
int dep[100005];
int main()
{
	int t,n,x;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		for(int i=0;i<n;i++)
		scanf("%d",&arr[i]);
		for(int i=0;i<n;i++)
		{
			cin>>x;
			dep[i]=x+arr[i];
		}
		sort(arr,arr+n);
		sort(dep,dep+n);
		int ans=1;
		int rooms=1;
		int i=1;
		int j=0;
		while(i<n&&j<n)
		{
			if(arr[i]<dep[j])
			{
				rooms++;
				if(rooms>ans)
				ans=rooms;
				i++;
			}
			else
			{
				rooms--;
				j++;
			}
		}
		cout<<ans<<"\n";
	}
	return 0;
}
