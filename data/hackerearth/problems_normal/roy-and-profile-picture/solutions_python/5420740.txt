l=input()
n=input()
details=[]

for i in range(1, n+1):
    details.append(raw_input())

for profile in details:
    val = profile.split(" ")
    w = int(val[0])
    h = int(val[1])
    if(w<l) or (h<l) :
        print "UPLOAD ANOTHER"
    elif (w >=l) and (h >= l):
        if w==h:
            print "ACCEPTED"
        else:
            print "CROP IT"
