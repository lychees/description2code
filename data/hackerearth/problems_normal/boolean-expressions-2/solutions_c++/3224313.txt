#include<bits/stdc++.h>
using namespace std;
const int MAXN = 305;
long long dp[2][MAXN][MAXN];
char str[MAXN];
char op[MAXN];
const int MOD = 1e9+9;
inline void check(long long &x) {
	if(x>=MOD)
		x%=MOD;
}
int main() {
	scanf("%s%s",str+1,op+1);
	int len=strlen(str+1);
	for(int i=1;i<=len;i++) {
		if(str[i]=='1')
			dp[1][i][i]=1,dp[0][i][i]=0;
		else
			dp[0][i][i]=1,dp[1][i][i]=0;
	}
	for(int i=2;i<=len;i++) {
		for(int j=1;j+i<=len+1;j++) {
			int st=j,end=j+i-1;
			dp[0][st][end]=dp[1][st][end]=0;
			for(int k=j;k<j+i-1;k++) {
				if(op[k]=='a') {
					long long temp=dp[1][st][k]*dp[1][k+1][end];
					check(temp);
					dp[1][st][end]+=temp;
					check(dp[1][st][end]);
					temp=dp[0][st][k]*dp[1][k+1][end];
					check(temp);
					dp[0][st][end]+=temp;
					check(dp[0][st][end]);
					temp=dp[1][st][k]*dp[0][k+1][end];
					check(temp);
					dp[0][st][end]+=temp;
					check(dp[0][st][end]);
					temp=dp[0][st][k]*dp[0][k+1][end];
					check(temp);
					dp[0][st][end]+=temp;
					check(dp[0][st][end]);
				}
				if(op[k]=='o') {
					long long temp=dp[1][st][k]*dp[1][k+1][end];
					check(temp);
					dp[1][st][end]+=temp;
					check(dp[1][st][end]);
					temp=dp[0][st][k]*dp[1][k+1][end];
					check(temp);
					dp[1][st][end]+=temp;
					check(dp[1][st][end]);
					temp=dp[1][st][k]*dp[0][k+1][end];
					check(temp);
					dp[1][st][end]+=temp;
					check(dp[1][st][end]);
					temp=dp[0][st][k]*dp[0][k+1][end];
					check(temp);
					dp[0][st][end]+=temp;
					check(dp[0][st][end]);
				}
				if(op[k]=='x') {
					long long temp=dp[1][st][k]*dp[1][k+1][end];
					check(temp);
					dp[0][st][end]+=temp;
					check(dp[0][st][end]);
					temp=dp[0][st][k]*dp[1][k+1][end];
					check(temp);
					dp[1][st][end]+=temp;
					check(dp[1][st][end]);
					temp=dp[1][st][k]*dp[0][k+1][end];
					check(temp);
					dp[1][st][end]+=temp;
					check(dp[1][st][end]);
					temp=dp[0][st][k]*dp[0][k+1][end];
					check(temp);
					dp[0][st][end]+=temp;
					check(dp[0][st][end]);
				}
			}
		}
	}
	int q;
	cin>>q;
	while(q--) {
		int a,b;
		char tt[10];
		scanf("%d%d%s",&a,&b,&tt);
		if(strcmp(tt,"true")==0)
			printf("%lld\n",dp[1][a][b]);
		else
			printf("%lld\n",dp[0][a][b]);
	}
	return 0;
}
