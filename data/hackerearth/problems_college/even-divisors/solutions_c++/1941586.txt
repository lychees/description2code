#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int t;
    cin >> t;
    
    for(int i=0;i<t;i++)
    {
    	long n;
    	cin >> n;
    	
    	int c=0;
    
    	if((int)sqrt(n)*(int)sqrt(n)==n && (int)sqrt(n)%2==0)
    	    c++;
    	
    	for(long j=1;j<sqrt(n);j++)
    	{
    		if(n%j==0)
    		{
    			if(j%2==0)
    		        c++;
    		    if((n/j)%2==0)
    		        c++;
    		}
    	}
    	cout << c << endl;
    }
}
