#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int test;
	cin>>test;
	while(test--)
	{
		set<char> st;
		string s;
		cin>>s;
		ll int l=s.length();
		ll int i;
		for(i=0;i<l;i++)
		{
			st.insert(s.at(i));
		}
		l=st.size();
		ll int ans=(l*(l+1))>>1;
		cout<<ans<<endl;
	}
	return 0;
}
