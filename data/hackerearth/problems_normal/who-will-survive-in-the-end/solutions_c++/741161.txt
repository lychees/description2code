#include <iostream>
using namespace std;

int power(int n)
{
	int a=1;
	while(n--)
		a=a<<1;
	return a;
}

int largest(int n)
{
	int k=0;
	while(n>0)
	{
		n=n>>1;
		k++;
	}
	return k;
}

int main()
{
    int t;
   	cin>>t;
   	while(t--)
   	{
   		int n;
   		cin>>n;
   		int greater2=power(largest(n));
   		int ans=greater2-1;
   		int p = greater2 - n;
   		ans-=2*(p-1);
   		cout<<ans<<endl;
   	}
    return 0;
}
