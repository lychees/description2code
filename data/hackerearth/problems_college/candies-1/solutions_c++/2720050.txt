#include <iostream>
using namespace std;

int main()
{
    int t,n,i;
    long long int *a,s;
    
    scanf("%d",&t);
    while(t--){
    	scanf("%d",&n);
    	a = (long long int *)malloc(sizeof(long long int)*n);
    	s=0;
    	for(i=0;i<n;i++){
    		scanf("%lld",&a[i]);
    		s+=a[i];
    	}
    	
    	if(s%n==0){
    		cout<<"YES"<<endl;
    		cout<<(s/n)<<" "<<0<<endl;
    	}else{
    		cout<<"NO"<<endl;
    		cout<<(s/n)<<" "<<(s%n)<<endl;
    	}
    }
    
    
    return 0;
}
