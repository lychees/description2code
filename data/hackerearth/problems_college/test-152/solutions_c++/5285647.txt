#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        map<char,int> my_map;
        int n;
        string s;
        cin>>n>>s;
        for(int i=0;i<=n;++i)
        {
            my_map[s[i]]++;
        }
        int max=0;
        for(int i=0;i<=n;i++)
        {
            int temp=my_map[s[i]];
            if(temp>max)
            {
                max=temp;
            }
        }
        if(max%2==0)
        {
            cout<<"Yes\n";
        }
        else
        {
            cout<<"No\n";
        }
    }
    return 0;
}
