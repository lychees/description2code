int is_palindrome(string str)
{
    int i=0,j=str.size()-1;int flag=1;
    while(str[i])
    {
        if(str[i]!=str[j])
        {
            flag=0;
            break;
        }
        i++;j--;
    }
    if(flag){return 1;}
    else {return 0;}
          }
