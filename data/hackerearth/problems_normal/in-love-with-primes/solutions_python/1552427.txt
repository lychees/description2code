def isprime(a):
    a=int(a)
    i=2
    flag=1
    while i<=int(a/2) and flag==1:
        if a%i==0:
            flag=0
        i=i+1
    return flag
 
T=int(raw_input())
for t in xrange(T):
    num=int(raw_input())
    a=2
    b=num-2
    f=0
    while b>=a and f==0:
        if isprime(a) and isprime(b):
            f=1
        a=a+1
        b=b-1
    if f==0:
        print "Arjit"
    else:
