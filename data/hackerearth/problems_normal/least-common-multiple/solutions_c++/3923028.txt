#include <bits/stdc++.h>

#define ll long long

#define mod 1000000007

using namespace std;

int main(){
	ll n;
	cin>>n;
	
	ll m = n/2 + 1;
	while(m<=n){
		ll x = m;
		bool f = false;
		for(ll i = 2;i*i<=x;i++){
			ll p =1;
			while(x % i == 0){
				p*=i;
				x/=i;
			}
			if(p == 1)   // not a factor
				continue;
		
			if(p*2 >n ){
				f = true;
				break;
			}
			if(2*x <= n)
				break;
		}
		if(f || 2*x > n)
			break;
		m++;
	}
	cout<<m<<endl;
	
	return 0;
}
