#include<bits/stdc++.h>
#include <iostream>
#define FOR(i,j,k) for(int i=j;i<=k;i++)
using namespace std;

int main()
{ int t;
  int k,n,p,i;
	scanf("%d",&t);
	while(t--)
	{ int rm[1000004];
		scanf("%d %d %d", &n, &k, &p);
	  for(i=1;i<=k;i++)
		{scanf("%d",&rm[i]);}
		
		sort(rm+1,rm+1+k);
		FOR(i,1,k)
		{
			if(rm[i]<=p) {
				p++;
			}
		}
		printf("%d\n",p>n?-1:p);	
	}
    
}
