#include <bits/stdc++.h>
using namespace std;
#define max 1000005
long long x[max],y[max],f[max],g[max];
 
void fx()
{
	long long i,j;
	memset(x,0,sizeof(x));
	x[0]=0;x[1]=0;
	for(i=2;i<max;i++)
	{
		if(x[i]==0)
		{
			for(j=1;j*i<max;j++)
			{
				if(x[i*j]==0)
					x[i*j]=i;
			}
		}
	}
}
 
void ff()
{
	long long i;
	memset(f,0,sizeof(f));
	f[0]=0;
	for(i=1;i<max;i++)
	{
		f[i]=f[i-1]+x[i];
	}
}
 
void fy()
{
	long long i,j;
	memset(y,0,sizeof(y));
	y[0]=0;y[1]=0;
	for(i=1;i<max;i++)
	{
		for(j=2;j*i<max;j++)
		{
				y[i*j]=y[i*j]+i;
		}
	}
}
 
void fg()
{
	long long i;
	memset(g,0,sizeof(g));
	g[0]=0;
	for(i=1;i<max;i++)
	{
		g[i]=g[i-1]+y[i];
	}
}
 
int main()
{
	fx();
	ff();
	fy();
	fg();
 
	long long t,n,i,ans;
	/*/

	for(i=0;i<10;i++)
	{
		cout<<y[i]<<" ";
	}cout<<endl;
	for(i=0;i<10;i++)
	{
		cout<<g[i]<<" ";
	}cout<<endl;
	*/
	cin>>t;
	while(t--)
	{
		cin>>n;
		ans=(f[n]+g[n])%n;
		cout<<ans<<endl;
	}
    return 0;
}
