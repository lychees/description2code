#include <iostream>
using namespace std;
int a[1000000]={0};
int main()
{
	a[0]=0;a[1]=0;a[2]=1;a[3]=1;a[4]=2;a[5]=1;a[6]=2;a[7]=1;
    for(int i=8;i<1000000;i++)
    {
    	if(a[i]==0){
    	if(a[i-7]!=0)
    	a[i]=a[i-7]+1;
    	else if(a[i-5]!=0 )
    	a[i]=a[i-5]+1;
    	else if(a[i-3]!=0)
    	a[i]=a[i-3]+1;
    	else if(a[i-2]!=0)
    	a[i]=a[i-2]+1;
    	}
    }
    int t; cin >> t;
    while(t--)
    {
    	int n; cin >> n;
    	cout<< a[n]<<endl;
    }
    return 0;
}
