#include <bits/stdc++.h>
using namespace std;

int main()
{
	//priority_queue<long int> p;
    long long int n,k,a[100001],i,j,temp,max,c=0;
    cin>>n>>k;
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    	//p.push_back(a[i]);
    }
    for(i=0;i<n-1;i++)
    {
    	max=i;
    	for(j=i+1;j<n;j++)
    	{
    		if(a[j]>a[max])
    		max=j;
    	}
    	//e=p.top();
    	if(i!=max)
    	{
    		temp=a[i];
    		a[i]=a[max];
    		a[max]=temp;
    		c++;
    		if(c==k)
    		break;
    		
    	}
    	
    }
    for(i=0;i<n;i++)
    cout<<a[i]<<" ";
    return 0;
}
