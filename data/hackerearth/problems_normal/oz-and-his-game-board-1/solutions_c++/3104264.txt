#include <iostream>
#include <cmath>

int main(int argc, char **argv) {
	int T;
	std::cin >> T;

	while (T > 0) {
		long long N, M, A, B, C;
		std::cin >> N >> M >> A >> B >> C;

		long long score = 0;
		if (2 * C > A + B) {
			int x = std::min(N, M);
			score += 2 * x * C;
			score += (N - x) * A;
			score += (M - x) * B;
		} else {
			score = A * N + B * M;
		}

		std::cout << score << std::endl;

		T--;
	}
}
