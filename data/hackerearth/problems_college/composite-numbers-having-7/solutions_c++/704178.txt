int arr[1000001] = {0};
// limit <=10^7
void sieve(int limit = 1000000)
{
      FOR(i,2,limit+1) arr[i] = 1;
      int i = 2;
      while(i<=limit)
      {
            int j = i;
            while(1LL*i*j<=limit)
            {
                  arr[i*j] = 0;
                  j++;
            }
            i++;
            while(i<=limit && !arr[i])
                  i++;
      }
}

int yes(int a)
{
      while(a)
      {
            if(a%10 == 7) return 1;
            a/=10;
      }
      return 0;
}

int main()
{
      sieve();
      int dp[1000001] = {0};
      dp[0] = 0;
      FOR(i,1,1000001)
      {
            dp[i] = dp[i-1];
            if(arr[i])
                  continue;
            else if(yes(i))
                  dp[i]++;
      }
      inp_s;
      DRT()
      {
            int n,m;
            cin>>n>>m;
            int ans = dp[m] - dp[n-1];
            if(!ans) ans = -1;
            cout<<ans<<endl;
      }
      return 0;
}
