#include<iostream>
#include<cstdio>

using namespace std;

int main()
{
	bool temp=0;
	string s;
	int t,count;
	cin>>t;
	while(t!=0)
	{
		temp = 0;
		count=0;
		cin>>s;
		for(int i=0;i<s.size()-1;i++)
		{
			if(s[i]!=s[i+1])
			{
				count = count + 1;
			}
			else
			{
				cout<<"SLAP\n";
				temp = 1;
				break;
			}
		}
		if(count == (s.size()-1))
		{
			cout<<"KISS\n";
		}
		else if(temp==0)
		{
			cout<<"SLAP\n";
		}
		t = t-1;
	}
}
