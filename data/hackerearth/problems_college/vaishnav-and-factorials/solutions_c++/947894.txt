#include <iostream>
using namespace std;
 
int main()
{
    int T, N, a[2000], i, num, index, x, temp, count;
    cin>>T;
    while(T--){
    	count=0;
    	cin>>N;
    	num=N;
    	i=0;
    	a[0]=1;
    	while(num)
    	{
    		a[i]=num%10;
    		num=num/10;
    		i++;
    	}
    	while(N-1)
    	{
    		index=0;
    		temp=0;
    		while(index<i)
    		{
    			x=(a[index]*(N-1))+temp;
    			a[index]=x%10;
    			temp=x/10;
    			index++;
    		}
    		while(temp)
    		{
    			a[index++]=temp%10;
    			temp=temp/10;
    			i++;
    		}
    		N--;
    	}
    	for(int j=i-1; j>=0; j--)
    	{
    		if(a[j]==4 || a[j]==7)
    		count++;
    	}
    	cout<<count<<endl;
    }
    return 0;
}
