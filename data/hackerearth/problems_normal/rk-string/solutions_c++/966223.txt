#include <bits/stdc++.h>
using namespace std;

const int N=1e6+10;
long long int m[N][2]={{0}},freq[N]={0};

//function for solving and finding the value

long long int finding(int N,string s,char ch)
{
	long long int ans=0;
	int d;
	for(int i=0;s[i];i++)
	{
		if(i==0)
		{
			if(s[i]==ch)
			freq[i]=1;
			else
			freq[i]=0;
			if(freq[i]>=N)
			{
				d=freq[i]-N;
				if(s[i]!=ch)
				{
					ans+=m[d][0];
					ans++;
					
				}
				else if(s[i]==ch && N!=0)
				{
					ans+=m[d][0];
					ans++;
					
				}
				
			}
			if(s[i]==ch)
			m[freq[i]][1]++;
			else
			m[freq[i]][0]++;
			
		}
		else
		{
			if(s[i]==ch)
			freq[i]=freq[i-1]+1;
			else
			freq[i]=freq[i-1];
			if(freq[i]>=N)
			{
				d=freq[i]-N;
				if(s[i]!=ch)
				{
					ans+=m[d][0];
					ans++;
				}
				else if(s[i]==ch && N!=0)
				{
					ans+=m[d][0];
					ans++;
				}
				
			}
			if(s[i]==ch)
			m[freq[i]][1]++;
			else
			m[freq[i]][0]++;
			
			
		}
	}
	//the value in ans containing the answer is returned
	
	return ans;
}

//Start of the main function

int  main()
{
	//variable declaration
   int T,M,N;
   cin>>T;
   while(T--)
   {
   	memset(m,0,sizeof(m));
   	cin>>M;
   	cin>>N;
   	string s;
   	cin>>s;
   	char ch='R';
   	printf("%lld ",finding(M,s,ch));
   	memset(m,0,sizeof(m));
   	ch='K';
   	printf ("%lld",finding(N,s,ch));
   	printf("\n");
   	
   }
}
