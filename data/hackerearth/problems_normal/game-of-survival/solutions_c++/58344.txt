#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
#define pb push_back
int main(){
	int tst;
	cin>>tst;
	while(tst--){
		int N;
		cin>>N;
		vector<int>sw,str;
		for(int i=0;i<N;i++){
			int val;
			cin>>val;
			str.pb(val);
		}
		for(int i=0;i<N;i++){
			int val;
			cin>>val;
			sw.pb(val);
		}
		sort(str.begin(),str.end());
		sort(sw.begin(),sw.end());
		bool flag=false;
		for(int i=0;i<N;i++){
			if(str[i]<=sw[i]){
			//	cout<<i<<endl;
				flag=true;
			}
		}
		if(!flag){
			cout<<"YES\n";
		}else{
			cout<<"NO\n";
		}
	}
}
