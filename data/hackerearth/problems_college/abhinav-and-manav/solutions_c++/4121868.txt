#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
LL digit( LL x)
{
	LL ans = 0;
	while(x>0)
	{
		x/=10;
		ans++;
	}
	return ans;
}
int main()
{
	LL t,l,r,digitsL,digitsR,ans,num,maxi;
	cin>>t;
	while(t--)
	{
		cin>>l>>r;
		digitsL = digit(l);
		digitsR = digit(r);
		digitsR = max(digitsL,digitsR);
		num = pow(10 ,digitsR) - 1;
		maxi = num/2;
		if(maxi>= l && maxi <= r)
		cout<<maxi*(num-maxi)<<endl;
		else if(maxi>=r)
		cout<<r*(num-r)<<endl;
		else if(maxi <= l)
			cout<<l*(num-l)<<endl;
	}
}
