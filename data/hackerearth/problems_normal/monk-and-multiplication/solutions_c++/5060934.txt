#include<iostream>
using namespace std;
int main()
{
	long n,f=0,s=-1,t=-2,i;
	cin>>n;
	long a[n];
	long long b[n];
	b[0]=-1;
	b[1]=-1;
	for(i=0;i<n;i++)
	{ 
		cin>>a[i];
		if(a[i]>t)
		{
			if(a[i]>s)
			{
				if(a[i]>f)
				{
					t=s;s=f;f=a[i];
				}
				else
				{
					t=s;s=a[i];
				}
			}
			else
				t=a[i];
		}

		if(i>=2)
			b[i] = s*f*t;
	}
	for(i=0;i<n;i++)
		cout<<b[i]<<endl;
}
