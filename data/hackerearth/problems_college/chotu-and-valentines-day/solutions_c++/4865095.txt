#include <iostream>
#include <bits/stdc++.h>
using namespace std;
#define nt long long int
int a[100005],bit[100005];
void add(int ind,int v,int n)
{
	while(ind<=n)
	{
		bit[ind]+=v;
		ind+=(ind&(-ind));
	}
}
int que(int ind)
{
	int s=0;
	while(ind>0)
	{
		s+=bit[ind];
		ind-=(ind&(-ind));
	}
	return s;
}
int main()
{
    int i,n,l,r,q;
    cin>>n>>q;
    for(i=1;i<=n;i++)
    {
    	cin>>a[i];
    	add(i,a[i],n);
    }
    //cin>>q;
    while(q--)
    {
    	cin>>l>>r;
    	cout<<que(r)-que(l-1)<<endl;
    }
    return 0;
}
