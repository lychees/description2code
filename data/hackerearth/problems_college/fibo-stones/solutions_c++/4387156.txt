#include<bits/stdc++.h>
using namespace std;

int main()
{
	long long n,m,arr[100],a,b,c,temp,sum=2;

	cin>>n>>m;
	arr[0]=a=1;
	arr[1]=b=1;
	for(int i=2;i<n;i++)
	{
		arr[i]=a+b;
		a=b;
		b=arr[i];
		sum=sum+arr[i];	
	}
	long long temp1=0;
	for(int i=0;i<m;i++)
	{
		
		cin>>temp;
		temp1+=temp;
	}
	cout<<sum-temp1<<endl;	

	return 0;
}
