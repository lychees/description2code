#include <iostream>
#include <math.h>
using namespace std;

//int x;

int calculateMoves(int r, int c, int w){
	int missRows = r-1;
	int missGuess = floor(c/w);	
	int totalMiss = missRows*missGuess;
	int missGuess2, i, hits, remBoxes;
	
	for(i=1, missGuess2=0; i<=c-w; i+=w, missGuess2++);
			
	remBoxes = (c-w)%w + w;
	
	if(remBoxes<=w)
	  hits = w;
    
    else
	  hits = 2 + (w-1);
	
	return (totalMiss + missGuess2 + hits);
	
}

int main()
{
    int i=0,t,r,c,w;
    cin>>t;
    
    while(i<t){
    	cin>>r>>c>>w;
    	cout<<"Case #"<<i+1<<": "<<calculateMoves(r,c,w)<<"\n";
    	++i;
    }
          
    return 0;
}
