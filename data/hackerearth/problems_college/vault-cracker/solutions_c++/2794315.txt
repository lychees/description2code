#include<bits/stdc++.h>
#define gc getchar
using namespace std;
int mem[2502][2502];
 string a,b;
int l1,l2;
int cal(int m ,int n)
{
    if (m == l1) return 1;
    if (n == l2) return 0;
    if(mem[m][n])
        return mem[m][n];
    if(a[m]==b[n])
    {
        mem[m][n]+=cal(m+1,n+1);
    }
    mem[m][n]+=cal(m,n+1);
    return mem[m][n];
}
int main()
{

    cin>>a>>b;
    l1=a.length();
    l2=b.length();
     cout<<cal(0,0)<<endl;
