def checkAndChangetoPrime(n):
    if n == 9 or n == 8:
        return 7
    for p in range(n, 2*n):
        for i in range(2, p):
            if p % i == 0:
                break
        else:
            return p
    return None


t = raw_input()

for i in range(int(t)):
    p = raw_input()
    sp = p.split(' ')
    count = 0
    for item in sp[1]:
        if int(item) != 0:
            if int(item) == 1:
                count += 1
            else:
                count += abs(int(item) - checkAndChangetoPrime(int(item)))
    print count
