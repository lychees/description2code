#include <iostream>
using namespace std;

int main() {
	int n, m;
	cin >> n >> m;
	int* g = new int[m];

	for (int i = 0; i < m; i++) {
		cin >> g[i];
	}

	for (int i = 0; i < m - 1; i++) {
		for (int j = i + 1; j < m; j++) {
			if (g[i] != -1 && g[i] == g[j]) {
				g[j] = -1;
			}
		}
	}

	for (int i = 1; i < n; i++) {
		int temp_m;
		cin >> temp_m;
		int* temp_g = new int[temp_m];
		for (int j = 0; j < temp_m; j++) {
			cin >> temp_g[j];
		}

		for (int j = 0; j < m; j++) {
			if (g[j] != -1) {
				bool flag = false;
				for (int k = 0; k < temp_m; k++) {
					if (g[j] == temp_g[k]) {
						flag = true;
					}
				}
				if (!flag) {
					g[j] = -1;
				}
			}
		}
	}

	int sum = 0;
	for (int i = 0; i < m; i++) {
		if (g[i] != -1) {
			sum += g[i];
		}
	}
	cout << -sum;
}
