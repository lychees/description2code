#include<bits/stdc++.h>
using namespace std;
int main()
{
long long	int n,k,a[100000],j,i;
	cin>>n>>k;
	for(i=0;i<n;++i)
	cin>>a[i];
if(k<n)
{
		for(i=0;i<k&&i<n;++i)
	{
long long	int	c=a[i];
long long int	temp=-1;
		for(j=i+1;j<n;++j)
		{
			if(a[j]>=c&&a[j]>=temp)
			{
				temp=a[j];
				
			}
			
		}
		if(temp!=-1&&temp!=c)
	{
			a[i]=temp;
			for(j=n-1;j>i;--j)
			{
				if(a[j]==temp)
			a[j]=c;
			}
	}
	else if(temp==c)
	++k;
	else
	++k;
	

		
	}
	for(i=0;i<n;i++)
	cout<<a[i]<<" ";
	cout<<endl;
}
else
{
	sort(a,a+n);
	for(i=n-1;i>=0;--i)
	cout<<a[i]<<" ";
	cout<<endl;
}
	
	return 0;
}
