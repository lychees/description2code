#include <stdio.h>
#include <math.h>
#include <algorithm>
#include <queue>
using namespace std;

int dig(long long x){
	int sum=0;
	while(x>0){
		sum+=x%10;
		x/=10;
	}
	return sum;
}
bool check(long long N,long long i){
	//printf("%lld %d %lld\n",i*i,dig(i),N );
	return i*i==N-dig(i)*i;
}
int main(int argc, char const *argv[])
{
	int t;
	scanf("%d",&t);
	while(t-->0){
		long long N;
		scanf("%lld",&N);
		bool flag=false;
		for(long long i=sqrt(N)+1;(i*i>N-200*i)&&(i>0);i--){
			if(flag) break;
			if(check(N,i)) {flag=true;printf("%lld\n",i );}
		}
		if(!flag) printf("-1\n");
	}
}
