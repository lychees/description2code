#include"bits/stdc++.h"
using namespace std;
#define ll long long int
int main()
{
	ll t;
	cin>>t;
	while(t--)
	{
		ll n,s=0;
		cin>>n;
		ll arr[n];
		for(ll i=0;i<n;i++)
		{
			cin>>arr[i];
		}
		sort(arr,arr+n);
		if(n%2==0)
		{
			for(ll i=0;i<n-1;i+=2)
			{
				s+=arr[i]*arr[i+1];
			}
		}
		else
		{
			s=arr[0];
			for(ll i=1;i<n-1;i+=2)
			{
				s+=arr[i]*arr[i+1];
			}
		}
		cout<<s<<endl;
	}
}
