#include<iostream>
using namespace std;

inline int marking(int p,int range,int mark){
	for(int j=1;j<=range;j++){
		if(p%j==0){
			mark++;
		}
	}
	return mark;
}

int main(){
int range=100000;
int input,count=0;
cin>>input;
for(int i=2;i<=range;i++){
	int mark=0;
	mark = marking(i,range,mark);
	if(mark==2){	
		count++;
	}
	if(count==input){
		cout<<i<<"\n";
		break;
	}
}
return 0;
}
