#include <iostream>
#include <string>
using namespace std;

int rev(int a)
{
	while(a%10==0) a=a/10;
	int b = 0;
	while(a)
	{
		b*=10;
		b+=a%10;
		a/=10;
	}
	return b;
}

int main()
{
    int t; cin>>t;
    while(t--)
    {
    	int a,b;
    	cin>>a>>b;
    	cout<<rev(rev(a)+rev(b))<<endl;
    }
}
