import itertools
from sets import Set

n = input()
p = raw_input()
l = p.split()
x = input()

z = Set()
for i in l:
    z.add(i)


def findsubsets(S, m):
    return map(set, itertools.combinations(S, m))


d = findsubsets(z, 3)
flag = 0

for i in d:
    sum = 0
    for k in i:
        sum += int(k)
    if sum == x:
        flag = 1

if flag == 1:
    print "True"
else:
    print "False"
