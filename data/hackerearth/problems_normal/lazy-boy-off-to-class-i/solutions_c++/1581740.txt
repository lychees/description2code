#include <bits/stdc++.h>
using namespace std;
 
int main()
{
   
    long long t;
    cin>>t;
   
    while(t--)
    {
    	 long long arr[50000];
    	long long i,j,n,a,b;
    	long long sum=0,k=0;
    	float cont=0;
    	for(i=0;i<7;i++)
    	{
    		cin>>n;
    		cont+=n;
    		for(j=0;j<n;j++)
    		{
    			cin>>a>>b;
    		arr[k++]=abs(a-b);
    		}
    	}
    	sort(arr,arr+k);
    	 int  m=ceil((cont*3)/4);
    	for(i=0;i<m;i++)
    	sum+=arr[i];
    	cout<<86400*7-sum<<endl;
    }
    return 0;
}
