#include <stdio.h>

int main()
{
	int total_test_cases;
	int current_input,current_input_length;
	scanf("%d",&total_test_cases);
	while(total_test_cases--!=0){
			scanf("%d",&current_input_length);
			if(current_input_length==1) {
				scanf("%d",&current_input);
				printf("%d\n",current_input);
			} else {
				while(current_input_length--!=0) {
				scanf("%d",&current_input);
				}
				printf("0\n");
			}
			
	}
    return 0;
}
