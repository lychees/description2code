#include <iostream>
using namespace std;
long long arr[100000];
long long seg[300000][3];
int n;
void update(int ind,int a,int b,int x,int r){
	//cout<<seg[ind][2]<<" "<<a<<" "<<b<<" "<<r<<endl;
	seg[ind][0] = seg[ind][0]-r;
	seg[ind][1] = seg[ind][1]-r;
	seg[ind][2] = seg[ind][2]+r;
	if(seg[ind][1]<=x)
		return;
	if(seg[ind][0]>x){
		seg[ind][0] = seg[ind][0]-1;
		seg[ind][1] = seg[ind][1]-1;
		seg[ind][2] = seg[ind][2]+1;
		return;
	}
	if(a<b){
		int mid = (a+b)/2;
		//cout<<seg[ind]<<" "<<a<<" "<<b<<" "<<x<<endl;
		update(2*ind+1,a,mid,x,seg[ind][2]);
		update(2*ind+2,mid+1,b,x,seg[ind][2]);
		if(seg[ind][1]<seg[2*ind+2][1])
			seg[ind][1] = seg[2*ind+2][1];
		seg[ind][0] = seg[2*ind+1][0];
		if(seg[ind][0]>seg[2*ind+2][0])
			seg[ind][0] = seg[2*ind+2][0];
		seg[ind][2] = 0;
		
		//seg[ind]--;
		//cout<<seg[ind]<<" "<<a<<" "<<b<<" "<<x<<endl;
		
	}else{
		seg[ind][0] = seg[ind][0]-1;
		seg[ind][1] = seg[ind][1]-1;
		arr[a] -= 1;
		seg[ind][2] = 0;
		return;
	}
}
void fin_update(int ind,int a,int b,int r){
	//cout<<a<<" "<<b<<" "<<r<<endl; 
	if(a<b){
		int mid = (a+b)/2;
		fin_update(2*ind+1,a,mid,r+seg[ind][2]);
		fin_update(2*ind+2,mid+1,b,r+seg[ind][2]);
	}else{
		arr[a] -= (r+seg[ind][2]);
	}
}
void fill(int ind,int a,int b){
	
	if(a<b){
		int mid = (a+b)/2;
		fill(2*ind+1,a,mid);
		fill(2*ind+2,mid+1,b);
		seg[ind][1] = seg[2*ind+1][1];
		if(seg[ind][1]<seg[2*ind+2][1])
			seg[ind][1] = seg[2*ind+2][1];
		seg[ind][0] = seg[2*ind+1][0];
		if(seg[ind][0]>seg[2*ind+2][0])
			seg[ind][0] = seg[2*ind+2][0];
		seg[ind][2] = 0;
		//cout<<seg[ind][0]<<" "<<seg[ind][1]<<" "<<a<<" "<<b<<endl;
	}else{ 
		seg[ind][1] = arr[a];
		seg[ind][0] = arr[a];
		seg[ind][2] = 0;
	}
}
int main()
{
    
    cin>>n;
    for(int i=0;i<n;i++){
    	cin>>arr[i];
    }
    fill(0,0,n-1);
    int m;
    cin>>m;
    for(int i=0;i<m;i++){
    	int X;
    	cin>>X;
    	//cout<<X<<" ?? ";
    	update(0,0,n-1,X,0);
    }
    fin_update(0,0,n-1,0);
    for(int i=0;i<n;i++){
    	cout<<arr[i]<<" ";
    }
    return 0;
}
