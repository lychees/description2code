import math
t = int(input())
while (t > 0):
    x1, y1, x2, y2 = map(int, raw_input().split())
    r = int(input())
    D = x1*y2-x2*y1
    d = math.sqrt((x2-x1)**2 + (y2-y1)**2)

    De = (r**2)*(d**2)-D**2
    if (De == 0):
        print "JUST MISSED"
    elif (De > 0):
        print "REPLANNING"
    else:
        print "SAFE"
    t -= 1
