noc = int(raw_input())
while noc:
    noc -= 1

    ins = map(str, raw_input().split())
    a = list(ins[0])
    b = list(ins[1])
    carry = 0
    ccnt = 0
    for x, y in zip(a, b):
        if int(x) + int(y) + carry >= 10:
            ccnt += 1
            carry = 1
        else:
            carry = 0
    if ccnt == 0:
        print "No carry operation"
    else:
        if ccnt > 1:
            print("%d carry operations" % (ccnt))
        else:
            print "1 carry operation"
