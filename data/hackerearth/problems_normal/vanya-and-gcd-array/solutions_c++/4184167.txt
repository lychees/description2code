#include <iostream>
using namespace std;
#define mod 1000000007
int gcd(int a, int b){
	if(a == 0 || b == 0)
		return a^b;
	return gcd(b%a, a);
}
int Mem[501][501][101];
long long int LGIS(int i, int large, int GCD, int N, int A[]){
	// cout << i << endl;
	if(i == N){
		// cout << GCD << endl;
		Mem[i][large][GCD] = (GCD == 1?1:0);
		return Mem[i][large][GCD];
	}
	if(Mem[i][large][GCD] >= 0)	
		return Mem[i][large][GCD];
	long long int t1 = 0, t2 = 0 ;
	if(GCD == 0 || A[large] < A[i])
		t1 = LGIS(i + 1, i, gcd(GCD, A[i]), N, A);
	t2 = LGIS(i + 1, large, GCD, N, A);
	Mem[i][large][GCD] = (t1 + t2)%mod;
	return Mem[i][large][GCD];
}
int main()
{
    int N = 0;
    cin >> N;
    int A[N];
    for(int n = 0; n < N; n++){
    	scanf("%d", &A[n]);
    }
    
    for(int n = 0; n <= N; n++){
    	for(int i = 0; i <= N; i++){
    		for(int j = 0; j <= 100; j++){
    			Mem[n][i][j] = -1;
    		}
    	}
    }
   
    cout << LGIS(0, 0, 0, N, A) << endl;
    return 0;
}
