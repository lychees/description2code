long long cnr[1004][1004];
long long fact[1005];

void shark_s()
{
	fact[0]=1;
	fact[1]=1;
	for(int i=2;i<=1000;i++)
		fact[i] = (fact[i-1]*i)%mod;
	for(int i=0;i<=1000;i++)
		for(int j=0;j<=i;j++)
		{
			if(i==j || j==0)
				cnr[i][j]=1;
			else 
				cnr[i][j] = (cnr[i-1][j]+cnr[i-1][j-1])%mod;
		}
	int n,k;
	cin>>n>>k;
	dp[0]=1;
	for(int i=1;i<=n;i++)
		for(int j=0;j<i;j++)
		{
			ways = (cnr[i-1][j]*fact[j+1])%mod;
			ways = (ways*dp[i-j-1])%mod;
			dp[i] = (dp[i]+ways)%mod;
		}
		printf("%lld\n",(dp[n]*cnr[n][k])%mod);
	return ;
}
