#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int t,n,k,i,max;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	cin>>k;
    	int arr[n];
    	int count[11];
    	memset(count,0,sizeof(count));
    	for(i=0;i<n;i++)
    		{
    		cin>>arr[i];
    		count[arr[i]]++;
    		}
        max=0;
        for(i=1;i<11;i++)
        {
        	if(count[i]>max)
        		max=count[i];
        }
        cout<<n-max<<endl;
    }
    return 0;
}
