#include <bits/stdc++.h>
using namespace std;
#define MAX 11111
int prime[MAX];


void sieve()
{
    for(int i=2;i<=sqrt(MAX);i++)
    {
        if(!prime[i])
        {
            for(int j=i+i;j<=MAX;j+=i)
            {
                prime[j]=1;
            }
        }
    }
}
int main()
{
    sieve();
    int n;
    cin>>n;
    for(int i=4;i<=n;i++)
    {
        if(prime[i])
            cout<<i<< " ";
    }

    return 0;
}
