#include <bits/stdc++.h>
using namespace std;
int main(){
 int t,x,y;
 string s;
 scanf("%d",&t);
 while(t--){
    scanf("%d %d",&x,&y);
    cin >> s;
    int flag = 0;
    while(x && y){
        int n  = __gcd(x,y);
        x -= n;
        y -= n;
        flag = flag^1;
    }
      //cout << flag << endl;
      if(flag == 0 && s=="Chandu"){
        cout << "Daspal" << endl;
      }
      else if(flag == 0 && s=="Daspal"){
         cout << "Chandu" << endl;
      }
      else if(flag == 1 && s=="Daspal"){
        cout << "Daspal" << endl;
      }
      else if(flag == 1 && s=="Chandu"){
        cout << "Chandu" << endl;
      }
 }
return 0;
}
