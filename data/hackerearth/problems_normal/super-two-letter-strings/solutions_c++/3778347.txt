#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll long long
ll dpx[10004];
ll dpy[10004];
int n,p;
int main()
{
int t;
cin>>t;
while(t--)
    {
	    cin>>n>>p;
		memset(dpx,0,sizeof(dpx));
		memset(dpy,0,sizeof(dpy));
		dpx[1]=0;
		dpy[1]=1;
		
		for(int i=2;i<=n;i++){
		dpy[i]=dpx[i-1]+dpy[i-1];
		dpy[i]%=mod;
		
		dpx[i]=dpy[i];
		
		if(i-p >=1)
		dpx[i]-=(dpy[i-p]);
		dpx[i]=(dpx[i]+mod)%mod;
		}
		printf("%lld\n",(dpx[n]+dpy[n])%mod);
	}
    return 0;
}
