import itertools
n = int(raw_input())
lyrics = list(raw_input())
kids = list(range(1, n+1))
pointer = 1
for letter in itertools.cycle(lyrics):
    if len(kids) == 1:
        break
    else:
        if letter == 'a':
            if pointer == kids[-1]:
                pointer = kids[0]
            else:
                pointer = kids[kids.index(pointer)+1]

        else:
            pointer1 = pointer
            if pointer == kids[-1]:
                pointer = kids[0]
            else:
                pointer = kids[kids.index(pointer)+1]

            kids.remove(pointer1)


print kids[0]
