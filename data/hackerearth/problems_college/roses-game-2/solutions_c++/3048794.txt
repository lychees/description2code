#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,a[100001],i;
    bool flag,mark[100001];
    cin>>t;
    while(t--) {
    	cin>>n;
    	memset(mark,false,sizeof(mark));
    	flag=false;
    	for(i=0;i<n;++i) {
    		cin>>a[i];
    		if(a[i]==-1 && flag==false)
    			flag=true;
    	}
    	if(flag==false)
    		cout<<"Lose\n";
		else {
			i=0;
			while(1) {
				if(a[i]==-1) {
					cout<<"Win\n";
					break;
				}
				else if(i==a[i] || mark[i]==true) {
					cout<<"Lose\n";
					break;
				}
				else if(mark[i]==false) {
					mark[i]=true;
					i=a[i];
				}
			}
		}   	
    	
    }
    
    return 0;
}
