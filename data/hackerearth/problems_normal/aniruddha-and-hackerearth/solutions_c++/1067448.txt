#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll hi = 1000000007;
ll gcd(ll a,ll b)
    {
    ll t;
    while(a%b){
        t=a%b; a=b ;b = t;
    }
    return b;
}
ll a[1000005];
ll b[1000005];
int main()
    {
    int t,m,n;
    ll ans ;
    scanf("%d",&t);
    a[0] = b[0] = 0;
    a[1] = b[1] = 1;
    for(m=2;m<=1000001;m++)
        {
        a[m] = (a[m-1] + b[m-1])%hi;
        b[m] = a[m-1];
    }
    for(m=1;m<=1000001;m++)
        {
        a[m] = (a[m] + a[m-1])%hi;
        b[m] = (b[m] + b[m-1])%hi;
    }
    while(t--)
        {
      scanf("%d",&n);
       printf("%lld\n",(a[n] + b[n])%hi);
    }
    return 0;
}
