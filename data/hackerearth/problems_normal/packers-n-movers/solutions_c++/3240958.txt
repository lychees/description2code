#include <iostream>
#include <cstring>

using namespace std;

int tsum[300], p[300];
int dp[300][300][20];
int pac;

void pre() {
	memset(dp, -1, sizeof(dp));
	tsum[0] = p[0];
	for(int i = 0; i < pac; ++i)
		if(i > 0)
			tsum[i] = tsum[i-1] + p[i];
}

int sum(int i, int j) {
	int ans = 0;
	ans = tsum[j] - (i-1 >= 0 ? tsum[i-1] : 0);
	
	//cout<<i<<" "<<j<<' '<<ans<<endl;
	return ans;
}

int f(int i, int j, int m) {
	if(i >= j)	return 0;
	if(m == 1) {
		//cout<<"val "<<i<<" "<<j<<endl;
		return dp[i][j][m] = sum(i, j-1);
	}
	if(dp[i][j][m] != -1)	return dp[i][j][m];
	int ans = 1000000001;
    for(int ind = i; ind < j; ++ind) {
    	//cout<<"val "<<i<<" "<<j<<" "<<ind<<endl;
    	ans = min(ans, max(sum(i, ind), f(ind+1, j, m-1)));
    }
    return dp[i][j][m] = ans;
}

int main()
{
	int m;
    cin>>m;
    cin>>pac;
    for(int i = 0; i < pac; ++i)
    	cin>>p[i];
    
    pre();
    cout<<f(0, pac, m);
    
    return 0;
}
