Input = raw_input().split(' ')
IN = []
for i in Input :
	IN.append(int(i))
max_win = 0
consitency_trophy = 0
d = dict() 
for j in IN :
	if d.has_key(j) == True :
		d[j] = d[j] + 1
	else :
		d.update({j : 1})
	if max_win < d[j] :
		max_win = d[j]
		consitency_trophy = j
	elif max_win == d[j] :
		if j > consitency_trophy :
			consitency_trophy = j
	print("%d %d" % (consitency_trophy,max_win))
