#include <iostream>
using namespace std;

int main()
{
    long long t,ln,rn,root,flag;
    cin>>t;
    while(t--)
    {
    	cin>>ln>>rn;
    	flag=0;
    	while(ln!=rn)
    	{
    		while(ln>rn)
    		{
    			ln=ln/2;
    			if(ln==rn)
    			{
    				root=ln;
    				flag=1;
    				break;
    			}
    		}
    		while(rn>ln)
    		{
    			rn=rn/2;
    			if(ln==rn)
    			{
    				root=rn;
    				flag=1;
    				break;
    			}
    		}
    		if(flag==1)
    		{
    			break;
    		}
    	}
    	cout<<root<<endl;
    }
    return 0;
}
