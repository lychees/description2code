#include<bits/stdc++.h>
#define ll long long int
using namespace std;
main()
{
    ll n,i,q,x,y,a,b;
    cin>>n;
    vector<ll> v;
    for(i=0;i<n;i++)
    {
        cin>>x;
        v.push_back(x);
    }
    sort(v.begin(),v.end());
    cin>>q;
    while(q--)
    {
        cin>>x>>y;
        cout<<upper_bound(v.begin(),v.end(),y)-lower_bound(v.begin(),v.end(),x)<<endl;
    }
    return 0;
}
