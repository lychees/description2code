#include <iostream>
using namespace std;

bool flag;
long long int n,t,i,tot,sum[100005],a[100005];
int main()
{
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	tot=0;
    	for(i=1;i<=n;i++)
    	{
    		cin>>a[i];
    		sum[i]=sum[i-1]+a[i];
    		tot+=a[i];
    	}
    	flag=0;
    	for(i=1;i<=n;i++)
    	{
    		if(sum[i-1]==tot-sum[i])
    		flag=1;
    	}
    	if(flag)
    	cout<<"YES\n";
    	else
    	cout<<"NO\n";
    }
    return 0;
}
