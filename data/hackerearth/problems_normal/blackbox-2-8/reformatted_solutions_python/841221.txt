import sys
inp = sys.stdin.readline


def factorial(n):
    fact = 1
    for i in range(1, n+1):
        fact *= i

    return fact


def main():
    t = int(inp())
    for test in range(t):
        n = int(inp())
        fact = factorial(n)
        sFact = str(fact)
        factList = map(int, list(sFact))
        print sum(factList)


if __name__ == "__main__":
    main()
