#include <bits/stdc++.h>
using namespace std;

vector <bool> prime;

void seive(int n) {    // sieve of eratosthenes
    prime.resize(n + 1);

    for(int i = 0; i < n; i++) {
        prime[i] = 1;                   //Assume all numbers to be prime
    }

    prime[0] = 0;                       //Zero is not prime
    prime[1] = 0;                       //1 is not prime

    for(int i = 2; i * i <= n; i++) {    //EX for n = 100 lopp will run till sqrt(n) i = 10         10*10 <= 100
        int pp = i * i;
        if(prime[i]) {                    //If a no is prime then its multiple cannot be prime
           // for(int j = 2; i * j <= n; j++)
                            for(int j = 0; (pp + j) <= n; j+=i)

                prime[pp  + j] = 0;
        }
    }
}
