void solve()
{
   int i,j,k,m,n;
   string s1,s2;
   cin>>s1>>s2;
   n=s1.size();
   m=s2.size();
   rep(i,0,n)
   {
     rep(j,0,m)
     {
       if(!j||!i)
       {
        dp[1][j]=0;
        continue;
       }
       else
       if(s1[i-1]==s2[j-1])
       {
         if(binary_search(s.begin(),s.end(),s1[i-1])||binary_search(s.begin(),s.end(),s2[j-1]))
          dp[1][j]=max(dp[1][j-1],dp[0][j]);
         else
          dp[1][j]=max(dp[0][j-1]+1,max(dp[1][j-1],dp[0][j]));
       }
       else 
          dp[1][j]=max(dp[1][j-1],dp[0][j]);
     }
     rep(j,0,m)
       dp[0][j]=dp[1][j]; 
   }
   cout<<dp[1][m];
}
