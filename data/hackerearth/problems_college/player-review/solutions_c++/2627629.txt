#include<iostream>
using namespace std;
int main()
{
    long int n,n1;
    cin>>n;
    long int s=0;
    long int max;
    while(n--)
    {
        cin>>n1;
        s+=n1;
       	if(s>max)
        max=s;
        if(s<=0)
        {
        	s=0;
        }
    }
    cout<<max<<endl;
    return 0;
}
