
#include <iomanip>
#include <math.h>
#include <cstdlib>
#include <string>
#include <iostream>
#include <map>

using namespace std;

int main() {
    int n;
    cin >> n;
    int arr[n][n], primDigSum = 0, secDigSum = 0;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            cin >> arr[i][j];
        }
    }
    for (int i = 0; i < n; i++) {
        primDigSum += arr[i][i];
    }
    for (int i = n - 1, j = 0 ; i >= 0 ; i--, j++) {
        secDigSum += arr[i][j];
    }
    
    cout << abs(primDigSum - secDigSum) << endl;
    return 0;
}
