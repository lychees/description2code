#include<bits/stdc++.h>
#define z 1000000007
using namespace std;
long long po(long long a,long long b){
	long long x=1,y=a;
	//printf("%lld\n",a);
	while(b){
		if((b%2)){
			//printf("%lld\n",b);
			x=x*y,x%=z;
		}
		y=y*y;
		y%=z;b/=2;
	}
	return x;
}
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		long long int n;
		scanf("%lld",&n);
		long long ans=po(3,n);
		ans%=z;
		printf("%lld\n",ans);
	}
	return 0;
}
