#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
	int n;
	cin >> n;
	
	long long a[n];
	
	for (int i = 0; i < n; i++) {
		cin >> a[i];
	}
	
	sort(a, a + n);
	
	int q;
	cin >> q;
	
	while (q--) {
		long long m;
		cin >> m;
		
		if (binary_search(a, a + n, m)) {
			cout << "I Like You\n";
		} else {
			bool f = false;
			
			for (int i = 0; i < n - 1 && a[i] < m; i++) {
				if (binary_search(a + i + 1, a + n, m - a[i])) {
					f = true;
					
					break;
					
				}
			}
			
			if (f) {
				cout << "I Like You\n";
			} else {
				cout << "I Hate You\n";
			}
		}
	}
	
    return 0;
}
