#include<bits/stdc++.h>
using namespace std;
int main()
{
    ios_base::sync_with_stdio(false);
    long long int a,b,n;
    cin>>a>>b>>n;
    if(n==1)
    {
    cout<<a<<endl;
    return 0;
    }
    if(n==2)
    {
    cout<<b<<endl;
    return 0;
    }
    n=n-2;
    while(n--)
    {
        long long int c=a+b;
        swap(a,b);
        b=c;
    }
    cout<<b<<endl;
    return 0;
}
