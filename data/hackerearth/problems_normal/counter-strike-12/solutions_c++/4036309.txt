#include <iostream>
#include<cmath>
using namespace std;

struct point
{
	int x,y;
};
double distance(struct point p1,struct point p2)
{
	return (abs(p1.x-p2.x)+abs(p1.y-p2.y));
}
int main()
{
	int t;
	scanf("%i",&t);
	while(t--)
	{
		int m,n,d;
		scanf("%i%i%i",&m,&n,&d);
		struct point shoot[1000];
		struct point tar[1000];
		bool is_shoot[1000]={0};
		for(int i=0;i<m;i++)
		{
			scanf("%i%i",&shoot[i].x,&shoot[i].y);
		}
		for(int i=0;i<n;i++)
		scanf("%i%i",&tar[i].x,&tar[i].y);
		int count=0;
		bool is_complte=0;
		double dd=(double)d;
		int complete=n>>1;
		for(int i=0;i<m;i++)
		{
			//count=0;
			for(int j=0;j<n;j++)
			{
				if(is_shoot[j])
				 continue;
				 if(distance(shoot[i],tar[j])<=dd){
				  count++;
				  is_shoot[j]=1;}
				  if(count>complete)
				   {
				   	is_complte=1;
				   	break;
				   }
			}
			if(is_complte)
			 break;
		}
		if(is_complte)
		printf("YES\n");
		else
		printf("NO\n");
		
	}
    //cout << "Hello World!" << endl;
    return 0;
}
