#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;
typedef long long int LL;

int unit[] = {
    		  0, 0, 0, 0,
    		  1, 1, 1, 1,
    		  6, 2, 4, 8,
    		  1, 3, 9, 7,
    		  6, 4, 6, 4, 
    		  5, 5, 5, 5,
    		  6, 6, 6, 6,
    		  1, 7, 9, 3,
    		  6, 8, 4, 2,
    		  1, 9, 1, 9
};

LL FIND(LL n, LL a)
{
    return unit[((n%10)<<2)+(a&3)];
}

int main()
{
    LL T;
    scanf("%lld",&T);
    while(T--){
    	LL B,E;
    	scanf("%lld %lld",&B,&E);
		printf("%lld\n",FIND(B,E));
    }
    return 0;
}
