cases = int(raw_input())
for i in range(cases):
    n = int(raw_input())
    swords = []
    for k in range(n):
        value = int(raw_input())
        swords.append(value)

    animals = []
    for a in range(n):
        value = int(raw_input())
        animals.append(value)

    swords.sort()
    animals.sort()
    killedAll = 1

    for j in range(n):
        if animals[j] >= swords[j]:
            print "NO"
            killedAll = 0
            break

    if killedAll == 1:
        print "YES"
