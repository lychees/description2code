#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int bt1,bt2,mt1,mt2;
    	cin>>bt1>>bt2>>mt1>>mt2;
    	
    	if(bt1>bt2)
    	{
    		swap(bt1,bt2);
    	}
    	if(mt1>mt2)
    	{
    		swap(mt1,mt2);
    	}
    	
    	if((bt1<mt1 && bt2<mt1) || (bt1>mt2 && bt2>mt2))
    		cout<<"Line"<<endl;
    	else if((bt1==mt2 && bt2>mt2) || (bt2==mt1 && bt1<mt1) || (mt1==mt2 && bt1<mt1 && bt2>mt2) || (bt1==bt2 && bt1>mt1 && bt2<mt2) )
    		cout<<"Point"<<endl;
    	else
    		cout<<"Nothing"<<endl;
    }
    return 0;
}
