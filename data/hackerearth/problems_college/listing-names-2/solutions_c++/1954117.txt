#include <iostream>
#include <cstring>
#include <algorithm>
using namespace std;

int main()
{
    vector<string> a;
    string s;
    int T, i, j, count;
    cin>>T;
    while(T--)
    {
    	cin>>s;
    	a.push_back(s);
    }
    sort(a.begin(),a.end());
    a.push_back("$");
    for(i=0; i<a.size()-1; ++i)
    {
    	count=1;
    	cout<<a[i]<<" ";
    	while(a[i]==a[i+1])
    	{
    		count++;
    		i++;
    	}
    	cout<<count<<endl;
    }
    return 0;
}
