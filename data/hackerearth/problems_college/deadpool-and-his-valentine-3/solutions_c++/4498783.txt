#include <iostream>
using namespace std;

int main()
{

	int k,m,f=1,t,n;
	cin>>k>>m;
	int a[m];
	for(int i=0;i<m;i++)
		{
			a[i]=0;	
		}
	for(int z=0;z<k;z++)
	{
		cin>>n;
		for(int i=0;i<n;i++)
		{
			cin>>t;
			a[t-1]=1;
		}
	}
	for(int i=0;i<m;i++)
		{
			if(a[i]==0)
				f=0;
		}
		if(f==1)
			cout<<"YES";
		else
			cout<<"NO";
}
