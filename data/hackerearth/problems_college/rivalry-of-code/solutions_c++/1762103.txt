#include<string>
#include<iostream>
using namespace std;
 
inline int min(int a, int b) {
   return a < b ? a : b;
}
 
inline void swap(string& a, string& b) {
	string c = a;
	a = b;
	b = c;
}

int main()
{
	int testcase;
	cin >> testcase;
	while (testcase--) {
		string pat, text;
		cin >> pat >> text;
		int change_count = 0, min_change_count = 9999;
		int len_pat = pat.length(), len_text = text.length();
		
		if (len_pat > len_text) { 
			swap(pat, text);
			len_pat = pat.length();
			len_text = text.length();
		}
		
	 	for (int j = 0; j + len_pat <= len_text; j++) {
		 	for (int i = 0; pat[i]; i++) {
		 		if (pat[i] != text[j + i]) {
		 			change_count++;
				}
			}	
			min_change_count = min(min_change_count, change_count);
			change_count = 0;
		}
	 	cout << min_change_count << "\n";
	}

    return 0;
}
