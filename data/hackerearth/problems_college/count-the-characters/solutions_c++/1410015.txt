#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,i=0;
    int a[26]={0};
    cin>>n;
    while(i<n)
    {
      string s;
      cin>>s;
      int j=0;
      while(j<s.length())
      {
           if(a[s[j]%'a']==i)
              a[s[j]%'a']++;
           j++;
      }
      i++;
    }
    int j=0;
    int c=0;
    while(j<26)
      {
           if(a[j]==n)
              ++c;
              j++;
      }
      cout<<c<<endl;
    return 0;

}
