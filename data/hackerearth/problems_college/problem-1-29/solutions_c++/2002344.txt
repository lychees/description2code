int arr[100][100], n;
bool solve(){
	int x, y;
	queue<pair<int,int> > Q;
	Q.push(mp(0,0));
	while(!Q.empty()){
		x = Q.front().F; y = Q.front().S;
		Q.pop();
		if(x < n - 1 && arr[x + 1][y]){
			if(x + 1 == n - 1 && y == n - 1) return true;
			Q.push(mp(x + 1, y));
			arr[x + 1][y] = 0;
		}
		if(y < n - 1 && arr[x][y + 1]){
			if(y + 1 == n - 1 && x == n - 1) return true;
			Q.push(mp(x, y + 1));
			arr[x][y + 1] = 0;
		}
	}
	return false;
}
int main(){

	ios_base::sync_with_stdio(false);

	int t;
	cin >> t;
	while(t--){
		cin >> n;
		f(i,n) f(j,n) cin >> arr[i][j];
		if(solve()) printf("POSSIBLE\n");
		else printf("NOT POSSIBLE\n");
	}
	
	return 0;
}
