#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t; cin >> t;
	while(t--)
	{
		int n, k, p; 
		cin >> n >> k >> p;
		int count = 0; bool flag = true;
		vector<int> arr;
		arr.push_back(0); arr.push_back(n+1);
		for(int i=0;i<k;i++)
		{
			int a;
			cin >> a; arr.push_back(a);
		}
		sort(arr.begin(), arr.end());
		for(int i=1;i<k+2;i++)
		{
			int x = arr[i] - arr[i-1]-1;
			if(p<=x)
			{
				cout << arr[i-1]+p << endl;
				flag = false; break;
			}
			else p = p-x;
		}
		if(flag) cout << "-1" << endl;
	}
}
