#include<iostream>
using namespace std;
int main()
{
	long long int n,petals[200],i,min=200,sum=0,nodds=0;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>petals[i];
		sum+=petals[i];
		if(petals[i]%2!=0)
		{
			nodds++;
			if(min>petals[i])
			min=petals[i];
		}
	}
	if(nodds==0)
	cout<<":(\n";
	else if(nodds%2==0)
	cout<<sum-min<<"\n";
	else
	cout<<sum<<"\n";
	return 0;
}
