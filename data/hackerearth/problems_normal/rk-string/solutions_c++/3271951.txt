const int cs = 0x186A1;

const int fixx = 1;
const int me = 1000005;
const ll mod = 1.e9 + 7;

int t, n, m, k, M, K;
string s;
int f[2][me];
ll ans[2];

ll c(char x){
    ll re = 0, w = 0;

    for(int i = 1 ; i <= n ; i ++){
        if(s[i] != x)
            re += 1LL * w * (w + 1) / 2, w = 0;
        else w ++;
    }
    re += 1LL * w * (w + 1) / 2;
    return re;
}
int main()
{
    ios_base:: sync_with_stdio(false);
    cin.tie(false);

    cin >> t;

    while(t --){
        cin >> m >> k;
        cin >> s, n = s.size(), s = "$" + s;
        ans[0] = ans[1] = 0LL;
        M = 0, K = 0;
        for(int i = 1 ; i <= n ; i ++){
            if(s[i] == 'R')
                f[0][++M] = i;
            else f[1][++K] = i;

            if(M >= m)
                ans[0] += f[0][M - m + 1] - f[0][M - m];
            if(K >= k)
                ans[1] += f[1][K - k + 1] - f[1][K - k];
        }
        if(m == 0)
            ans[0] = c('K');
        if(k == 0)
            ans[1] = c('R');
        cout << ans[0] << " " << ans[1] << endl;
    }

    ///cout << "Elapsed time is: " << clock() / CLOCKS_PER_SEC << endl;

    return 0;
}
