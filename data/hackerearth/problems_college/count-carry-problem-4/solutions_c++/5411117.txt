#include<bits/stdc++.h>
using namespace std;

int main()
{
	
	long long int  x,y,ans,c,t;
	cin>>t;
	while(t--)
	{
		cin>>x>>y;
		ans=0;
		c=0;
		while(x!=0||y!=0)
		{
			if((x%10+y%10+c)>=10)
			{
				ans++;
				c++;
			}
			x/=10;
			y/=10;
			//cout<<x<<endl;
		}
		
		if(ans==0){cout<<"No carry operation\n";}
		else if(ans==1)
		cout<<ans<<" carry operation\n";
		else
		cout<<ans<<" carry operations"<<endl;
		
	}
	return 0;
}
