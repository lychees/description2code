#include <bits/stdc++.h>
#define ll long long int
#define MOD 1000000007
using namespace std;

int main()
{
    map<vector<ll>,ll> m;
    ll n,a;
    vector<ll> v(3);
    cin>>n;
    while(n--)
    {
    	cin>>v[0]>>v[1]>>v[2];
    	sort(v.begin(),v.end());
    	m[v]++;
    }
    a=0;
    for(auto i=m.begin();i!=m.end();i++)
    {
    	if(i->second==1)
    		a++;
    }
    cout<<a;
    return 0;
}
