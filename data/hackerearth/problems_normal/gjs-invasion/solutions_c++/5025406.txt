#include<bits/stdc++.h>
#define ll long long
using namespace std;



int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    
    int t;
    cin>>t;
	
	while(t--)
	{
	   	int n;
	    cin>>n;
	    
	    ll int dp[16][65540]={0};
		ll int ways[16][65540]={0};
	
	    int mv=(1<<n)-1;
	    for(int j=0;j<n-1;j++)
	    {
	        ways[n-1][mv-(1<<j)]=1;
	        dp[n-1][mv-(1<<j)]=n-1-j;
	    }
	  //  cout<<"k";
	    for(int i=n-2;i>=0;i--)
	    {
	        //cout<<i<<endl;
	        for(int j=0;j<=mv;j++)
	        {
	            bitset<16> b(j);
	
	            int s[16];
	            s[n-1]=b[n-1];
	            for(int k=n-2;k>=0;k--)
	                s[k]=s[k+1]+b[k];
	
	            for(int k=0;k<n;k++)
	            {
	                if(b[k]==0 && k!=i)
	                {
	                    ways[i][j]+=ways[i+1][j+(1<<k)];
	                    dp[i][j]+=(s[k])*ways[i+1][j+(1<<k)]+dp[i+1][j+(1<<k)];
	                }
	            }
	        }
	    }
	
	    cout<<dp[0][0]<<"\n";
	}
}
