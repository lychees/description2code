#include <iostream>
using namespace std;

int main()
{
    unsigned long long int T,count,N;
    cin>>T;
    while(T--)
    {
    	count=0;
    	cin>>N;
    	while(N!=1)
    	{
    		count++;
    		if(N%3==0)
    			N=N/3;
    		else if(N%2==0)
    			N=N/2;
    		else
    			N--;
    	}
    	cout<<count<<"\n";
    }
    return 0;
}
