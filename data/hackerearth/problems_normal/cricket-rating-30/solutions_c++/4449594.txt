#include<iostream>
using namespace std;
int main()
{
    long long n,maxs=0,maxe=0,i;
    cin>>n;
    long long a[n];
    for(i=0;i<n;i++)
    {
        cin>>a[i];
    }
    for(i=0;i<n;i++)
    {
        maxe+=a[i];
        if(maxe<0)
            maxe=0;
        if(maxs < maxe)
            maxs=maxe;
    }
    cout<<maxs;
    return 0;
}
