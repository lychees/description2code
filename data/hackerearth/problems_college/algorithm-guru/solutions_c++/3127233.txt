#define fast ios_base::sync_with_stdio(false);cin.tie(0);cout.tie(0)

#define openout freopen("output.txt","w",stdout)
vector<int> v;
int dp[4000];
int main()
{
    fast;
    int t;
    cin>>t;
    while(t--)
    {
        int n,x;
        cin>>n>>x;
        v.clear();
        rep(i,1,n){int xx;cin>>xx;v.pb(xx);}
        rep(i,0,x)dp[i]=0;
        sort(v.begin(),v.end());
        rep(i,1,x)
        {
        	for(int j=0;j<v.size();++j)
        	{
        		if(v[j]>i)continue;
        		dp[i]=max(dp[i],v[j]+dp[i-v[j]]);
        	}
        }
        cout<<dp[x]<<endl;
    }

    return 0;
}
