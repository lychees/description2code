#include <iostream>
using namespace std;

int main()
{
	int n,p,t,i,j;
	cin >>t;
	for( i=1;i<=t;i++){
		
		cin>>n>>p;
		
		for(j=1;j<=p;j++){
			if(n==3)
				break;
			
			if(n%2==0){
				
				n=n/2 +1;
			}else{
				n=n/2 +2;
			}
		}
		
		cout << n<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
