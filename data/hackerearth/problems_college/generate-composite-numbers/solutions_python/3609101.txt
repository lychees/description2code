import math
no = int(raw_input())


def shouldPrint(n):
	shouldPrint = False
	
	if(n == 0):
		shouldPrint = False;
	if(n == 1):
		shouldPrint =  False;
	
	upperLimit = int(math.sqrt(n))
	start = 2
	
	while(start <= upperLimit):
		if(n % start == 0):
		  shouldPrint = True
		start += 1
	return shouldPrint	  

s = ""

for i in range(0, no + 1):
	if(shouldPrint(i) == True):
		s += " " + str(i)
print s
