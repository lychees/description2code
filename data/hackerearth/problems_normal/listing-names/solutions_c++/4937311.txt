#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int t,n,count,i;
	cin>>t;
	string s[t];
	for(i=0;i<t;i++)
	{
		cin>>s[i];
	}
	sort(s,s+t);
	count=1;
	for(i=0;i<t;i++)
	{
		
		if(s[i]==s[i+1])
		{
			count++;
		}
		else
		{
			cout<<s[i]<<" "<<count<<endl;
			count=1;
		}
	}
}
