void fs(int &x)
{
    register int c = gc();
    x = 0;
    int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}

int n;
ll mod=1e9+7;
int a[1000][1000];
ll dp[1000][1000];


ll cal(int x, int y)
{
    if(dp[x][y]!=0) return dp[x][y];
    ll ans=1;
    if(x>0 && a[x-1][y]<a[x][y]) ans=(ans+cal(x-1,y))%mod;
    if(y>0 && a[x][y-1]<a[x][y]) ans=(ans+cal(x,y-1))%mod;
    if(x<n-1 && a[x+1][y]<a[x][y]) ans=(ans+cal(x+1,y))%mod;
    if(y<n-1 && a[x][y+1]<a[x][y]) ans=(ans+cal(x,y+1))%mod;
    dp[x][y]=ans;
    return ans;
}


int main()
{
    cin>>n;
    rep(i,n) rep(j,n) cin>>a[i][j];
    ll ans=0;
    rep(i,n) rep(j,n) ans=(ans+cal(i,j))%mod;
    cout<<ans<<endl;
    return 0;
}
