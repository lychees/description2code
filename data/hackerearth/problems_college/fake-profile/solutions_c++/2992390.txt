#include <iostream>
using namespace std;

int main()
{
    int arr[26]={0},nDistinct=0;
    string str;
    cin>>str;
    
    for(int i=0;i<str.size();i++)
    {
    	if(arr[str[i]-'a'] == 0)nDistinct++;
    	arr[str[i]-'a']++;
    }
    
    if(!(nDistinct%2))cout<<"CHAT WITH HER!";
    else cout<<"IGNORE HIM!";
    return 0;
}
