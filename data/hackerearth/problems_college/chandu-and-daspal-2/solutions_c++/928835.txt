#include<stdio.h>
int gcd(int a, int b)
{
    int min=((a<b)?a:b);
    while(1)
    {
    	if(a%min==0&&b%min==0)
    	return min;
    	min--;
    }
    return -9999;
}
int main()
{
    int ntc,a,b,h,i;
    scanf("%d",&ntc);
    char str[20];
    while(ntc--)
    {

        scanf("%d%d",&a,&b);
        scanf("%s",&str);
//printf("a=%db=%dstr=%s
        i=0;
        while(a>0&&b>0)
        {
            h=gcd(a,b);
            a=a-h;
            b=b-h;
            i++;
            i=i%2;
        }
        if(i==0)
        {
            if(str[0]=='C')
                printf("Daspal\n");
            else
                printf("Chandu\n");

        }
        else
        {
            if(str[0]=='C')
                printf("Chandu\n");
            else
                printf("Daspal\n");
        }
    }
}
