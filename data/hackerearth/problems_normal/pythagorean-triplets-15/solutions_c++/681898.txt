#include<bits/stdc++.h>
int gcd(int m, int p);
using namespace std;

int main()
{
	int i,j,k,counter,a,b,c,t,n;
	counter=0;
	 cin>>t;
	 while(t--)
	 {
	 	
	 	cin>>n;
	 	counter=0;
	 	for (i= 2; i<=(sqrt(n-1) + 1);i++)
          for (j = 1; j < i; j++)
              if ((gcd(i,j) == 1) && ((i-j) % 2) && ((i*i + j*j) <= n))
              {
                    k = 0;
                    a = i * i - j * j;
                    b = 2 * i * j;
                    c = i * i + j * j;
                    while ((++k) * c <= n)
                      counter++;
              }
              
              cout<<counter<<endl;
	 }
	
	
	
	
	return 0;
}
 int gcd(int m, int p)
{
   if (p==0) return m;
   return gcd(p,m%p);
}
