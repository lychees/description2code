#include<bits/stdc++.h>
using namespace std;

int main()
{
	int n,m;
    cin>>n>>m;
    vector<int> A;
	vector<int> B;
    while(n--){
    	int temp;
        cin>>temp;
        A.push_back(temp);
    }

    while(m--){
    	int temp;
        cin>>temp;
        B.push_back(temp);
	}
	sort(A.begin(),A.end());
	sort(B.begin(),B.end());	
	int mini=max(2*A[0],A[A.size()-1]);
	
	if (B[0]<=mini)
		cout<<"-1\n";
	else
		cout<<mini<<"\n";
}
