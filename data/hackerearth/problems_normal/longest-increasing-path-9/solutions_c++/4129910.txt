#include<bits/stdc++.h>
using namespace std;
int arr[100][100],dp[100][100];
int khalsa(int b,int c)
{
    memset(dp,0,sizeof(dp));
    int d,e,f,g,i,j;
    dp[0][0]=1;
    g=1;
    for(i=1;i<c;i++)
    {
        if(arr[0][i]>arr[0][i-1])
        {
            dp[0][i]=i+1;
            g=max(dp[0][i],g);
        }
        else
            break;
    }
    for(i=1;i<b;i++)
    {
        if(arr[i][0]>arr[i-1][0])
        {
            dp[i][0]=i+1;
            g=max(dp[i][0],g);
        }
        else
            break;
    }
    for(i=1;i<b;i++)
    {
        for(j=1;j<c;j++)
        {
            if(arr[i-1][j]<arr[i][j] && dp[i-1][j]!=0)
                d=dp[i-1][j]+1;
            else
                d=0;
            if(arr[i][j-1]<arr[i][j] && dp[i][j-1]!=0)
                e=dp[i][j-1]+1;
            else
                e=0;
            dp[i][j]=max(d,e);
            if(dp[i][j]>g)
                g=dp[i][j];
        }

    }
    return g;
}
int main()
{

    int a,b,c,d,e,f,g,h,i,j,k,l;
    cin>>a;
    while(a--)
    {
        cin>>b>>c;
        for(i=0;i<b;i++)
        {
            for(j=0;j<c;j++)
                cin>>arr[i][j];
        }
        cout<<khalsa(b,c)<<"\n";
    }
    return 0;
}
