#include<iostream>
#include<algorithm>
#include<vector>
#include<string.h>
using namespace std;
string s;
int main()
{
  int t;
  cin>>t;
  while(t--) {
    cin>>s;
    vector< char > ans;
    int i,j,n,m,k,temp;
    n=s.length();
    for(i=0;i<n;i++) {
      if(s[i]-'0'>=0 && s[i]-'0'<=9) {
        temp=s[i]-'0';
        m=ans.size();
        for(j=1;j<temp;j++) {
          for(k=0;k<m;k++) ans.push_back(ans[k]);
        }
      }
      else ans.push_back(s[i]);
    }
    for(i=0;i<ans.size();i++) cout<<ans[i];
    cout<<"\n";
  }
//  system("pause");
  return 0;
}
