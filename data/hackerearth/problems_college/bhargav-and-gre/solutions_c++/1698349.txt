#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;

int main()
{
	int t,n,m,l,lw;
	char w[1000];
	char wr[1000];
	scanf("%s",w);
	cin>>t;
	m=0;
	n=0;
	l = strlen(w);
	while(t--){
		
		scanf("%s",wr);
		lw = strlen(wr);
		if(lw==l && (wr[0]-3)==(w[0]-0))
			m++;
		if(lw==l && wr[0]==w[0])
			n++;
			
   
	}
	 cout <<n<< " " <<m<< endl;
    return 0;
}
