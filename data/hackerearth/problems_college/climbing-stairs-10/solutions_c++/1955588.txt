#include<iostream>

using namespace std;

long long fibTable [ 10001 ];

long long fib( long long n )
{
   if(fibTable[n] == -1)
   {
    if ( n <= 1 )
      fibTable[n] = n;
    else
      fibTable[n] = fib(n-1) + fib(n-2);
   }
 
   return fibTable[n];
}

long long countWays( long long s)
{
    return fib(s + 1);
}
 
// Driver program to test above functions
int main ()
{
  int t;
  long long ans, n;
  cin >> t;
  for ( int i = 0; i < 10001; i++ )
  	fibTable [ i ] = -1;
  while ( t-- )
  {
  		cin >> n;
  		ans = countWays( n );
  		cout << ans << endl;
  }
  return 0;
}
