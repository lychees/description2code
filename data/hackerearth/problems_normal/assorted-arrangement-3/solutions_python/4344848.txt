import sys

L=map(int, sys.stdin.read().split())
n, m=L[0], L[1]

c=L[2: 2+m]
L=L[2+m:]

res=1
for x in xrange(len(L)):
	if x==0:
		res*=c[L[x]-1]
	else:
		
		temp=(res/c[L[x]-1])+1
		while True:
			flag=0
			res=c[L[x]-1]*temp
			for y in xrange(L[x], len(c)):
				if res%c[y]==0:
					temp+=1
					flag=1
					break
			
			if flag==0:
				break
	
print res
