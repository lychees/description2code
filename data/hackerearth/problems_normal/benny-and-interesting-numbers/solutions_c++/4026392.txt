#include <iostream>
#include <vector>
#include <cmath>
using namespace std;

const long long int LEN = 320000;

bool arr[LEN];

vector<long long int> prime, v;

void init() {
	for(int i = 0; i < LEN; ++i) {
		arr[i] = true;
	}
	
	for(long long int i = 2; i < LEN; ++i) {
		if(arr[i] == true) {
			prime.push_back(i);
			long long int k = i * (i - 1);
			for(long long int j = i + k; j < LEN; j += k) {
				arr[j] = false;
			}
		}
	}
	
	for(auto iter = prime.begin(); iter != prime.end(); ++iter) {
		v.push_back((*iter) * (*iter));
	}
}

int main()
{
    int test;
    cin >> test;
    
    init();
   
    
    while(test--) {
    	long long int n;
    	
    	cin >> n;
    	
    	bool ans = true;
    	
    	for(auto iter = v.begin(); iter != v.end() && (*iter <= n); ++iter) {
    		if((n % (*iter)) == 0) {
    			ans = false;
    			break;
    		}
    	}
    	
    	if(ans) cout << "Yes" << endl;
    	else cout << "No" << endl;
    }
    return 0;
}
