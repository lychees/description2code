int main()
{

    for(int i = 1; i<= n; i++)

        cin >> a[i];
    }

    {
        int c;
        cin >> c;
        switch(c)
        {
        case 0:
            int i,b;
            cin >> i >> b;
            a[i] = b;
            break;
        case 1:
            int v;
            bool flag = true;
            cin >> v;
            for(int i = 1 ; i <=n; i++)
            {
                if(a[i] >= v){ cout << i <<endl; flag = false;break;}

            }
            if(flag)
            cout << "-1" << endl;
            break;
        }
    }
    return 0;
}
