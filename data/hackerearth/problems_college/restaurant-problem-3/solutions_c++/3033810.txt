#include <iostream>
#include<cmath>
using namespace std;

int main()
{
	string a;
	cin>>a;
	
	int n = a.length();
	
	long int num,count = 0;
	for(int i =1; i<=n; i++){
		for(int j=0; j<n-i+1;j++){
			num=0;
			int flag = 0;
			for(int k=j; k<j+i; k++){
				if(a[j] == '0' && i > 1){
					flag = 1;
					break;
				}
				num *=10;
				num += ((int)a[k] - 48);					
			}
			if(num%8 == 0 && flag == 0){
				count++;
			}
		}
	}
	cout<<count<<endl;
	return 0;
	}
