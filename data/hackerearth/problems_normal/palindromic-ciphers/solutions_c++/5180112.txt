#include<iostream>
using namespace std;
int main(){
    int t;
    cin>>t;
    while(t--){
        string s;
        cin>>s;
        int count =0;
        for(int i=0;i<s.length()/2;i++){
            if(s[i] == s[s.length()-i-1]){
                count = 1;
            }
            else {
                count=0;
                break;
            }
        }
        if(count == 1){
            cout<<"Palindrome"<<endl;
        }
        else{
            unsigned long long mul=1;
            for(int i=0;i<s.length();i++){
                mul*=(s[i]-96);
            }
            cout<<mul<<endl;
        }
    }
}
