def fact(n):
	if n == 0 or n == 1:
		return 1
	else:
		return n * fact(n-1)
		
t = input()
for i in range(t):
	n = input()
	result = fact(n)
	result_list = list(str(result))
	if len(str(result)) < 5:
		print ''.join((['0']*(5-len(result_list)) + result_list))
	else:
		print ''.join(result_list[len(result_list)-5:])
