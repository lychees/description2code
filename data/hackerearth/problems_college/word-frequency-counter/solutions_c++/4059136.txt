#include<iostream>
#include<map>
#include<vector>
#include<string>
using namespace std;
int main()
{
    string st;
    vector<string>v;
    map<string,int>mp;
    while(cin>>st)
    {
        for(int i=0;i<st.size();++i)
        {
            if(st[i]>='a' && st[i]<='z')
                st[i]=st[i]-'a'+'A';
        }
        v.push_back(st);
        mp[st]++;
    }
    for(auto c : v)
    {
        if(mp[c]!=0)
        {
            cout<<c<<" "<<mp[c]<<endl;
            mp[c]=0;
        }
    }
    return 0;
}
