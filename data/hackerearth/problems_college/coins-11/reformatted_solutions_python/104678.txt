S = []


def pre():
    for i in range(0, 100005):
        S.append(0)
    S[0] = 1
    S[1] = 1
    for i in range(2, 1000):
        if (S[i] == 0):
            j = i*i
            while (j <= 100002):
                S[j] = 1
                j = j+i


pre()
t = int(raw_input())
while (t != 0):
    n = int(raw_input())
    if (n <= 3):
        print n
    else:
        if (S[n] == 1):
            print "1"
        else:
            print "2"
    t = t-1
