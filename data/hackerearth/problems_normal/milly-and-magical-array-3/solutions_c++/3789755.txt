#include <iostream>
#include <vector>
using namespace std;

int main() {
int t;
cin >> t;

for (int i = 0; i < t; i++) {
long long n, x, l, s = 0;
cin >> n >> x >> l;

for (int j = 1; j < n; j++) {
long long c, d;
cin >> c;
d = (l - c + x) / x;
if (d < 0)d = 0;
c += d*x;
s += d;
l = c;
}

cout << s << endl;
}

return 0;
}
