#include <bits/stdc++.h>

using namespace std;

#define pb push_back
#define mp make_pair
#define ll long long int

int main()
{
    ll n, t, x, i, k, cnt;
    cin >> t;
    while (t--) {
        cin >> n >> k;
        cnt = 0;
        for(i = 0; i < n; i++) {
            cin >> x;
            if(x > 0) {
                cnt++;
            }
        }
        if(n - cnt < k) {
            cout << "YES\n";
        } else {
            cout << "NO\n";
        }

    }

    return 0;
}
