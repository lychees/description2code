int n;
int a[502];
int ans=0;
int intrance1,intrance2;
void solve(int num)
{
         if(num==n)
         {
             ans++;
             return ;
         }

         if(intrance1+a[num]<=intrance2)
         {
              intrance1+=a[num];
              solve(num+1);
              intrance1-=a[num];
         }
         intrance2+=a[num];
         solve(num+1);
         intrance2-=a[num];
}
