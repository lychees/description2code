#include<cstdio>
#include<cstdlib>
#include<algorithm>
using namespace std;
int a[2000],A[2000];
int check(int num,int n){
    for(int i=0;i<=n;i++){
        A[i]=a[i];
    }    
    A[1]=max(A[1]-num,1);
    for(int i=2;i<=n;i++){
        if(A[i-1]<A[i]){
            A[i]=max(A[i-1]+1,A[i]-num);
        }    
        else{
            A[i]=min(A[i-1]+1,A[i]+num);
        }    
        if(A[i]<=A[i-1]){return 1;}
    }    
    return 0;
}    

int bin_search(int low,int high,int n){
    int mid;
    while(low<=high){
        mid=(low+high)/2;
        //printf("mid=%d\n",mid);
        if(check(mid,n)==0 && (mid==low||check(mid-1,n)))break;
        else if(check(mid,n)==0){high=mid-1;}
        else{low=mid+1;}
    }    
    return mid;
}    

int main(){
    int t;scanf("%d",&t);
    while(t--){
        int n;
        scanf("%d",&n);
        int flag=1;
        a[0]=0;
        for(int i=1;i<=n;i++){
            scanf("%d",&a[i]);
            if(a[i]<=a[i-1]){
                flag=0;
            }
        }    
        if(flag){printf("0\n");}
        else{
            int temp=1;
            while(check(temp,n)){
                temp*=2;
            }
            printf("%d\n",bin_search(temp/2+1,temp,n));  
        }    
    }    
    return 0;
}
