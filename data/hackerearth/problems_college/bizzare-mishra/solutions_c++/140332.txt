#include<iostream>
#include<cstdio>
using namespace std;
int main() {
    int t;
    scanf("%d",&t);
    while(t--){
        int l,d1=0,d2=0;
        scanf("%d",&l);
        char s[10005],v1[10005],v2[10005];
        scanf("%s",s);
        for(int i=0;i<l;i++){
            if(i%2){
                v1[i]='Y';
                v2[i]='N';
            }
            else {
                v1[i]='N';
                v2[i]='Y';
            }
        }
        for(int i=0;i<l;i++){
            if(v1[i]!=s[i])
                d1++;
            if(v2[i]!=s[i])
                d2++;
        }
        printf("%d\n",(d1<d2)?d1:d2);
    }
    return 0;
}
