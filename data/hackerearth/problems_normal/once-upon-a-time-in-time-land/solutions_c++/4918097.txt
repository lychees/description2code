
#include <iostream>
#include<cstring>
using namespace std;

int main()
{
//cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int k;
    	cin>>k;
    	long long int mx=0;
    	long long int a[n];
    	long long int dp[n];
    	memset(dp,0,sizeof(dp));
    	for(int i=0;i<n;i++)
    		cin>>a[i];
    	dp[0]=max(a[0],(long long int)0);
    	for(int i=1;i<=k;i++)
    		dp[i]=max(dp[i-1],a[i]);
    	for(int i=k+1;i<n;i++)
    	{
    		dp[i]=max(dp[i-1],a[i]+dp[i-k-1]);
    		mx=max(mx,dp[i]);
    	}
		cout<<dp[n-1]<<endl;
    }
    
    return 0;
}
