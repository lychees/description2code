#include <iostream>
using namespace std;

int main()
{
	int n;
	cin>>n;
	int arr[n];
	for(int i=0; i<n; i++){
		cin>>arr[i];
	}
	int q;
	cin>>q;
	int d=1;
	while(q--){
		int t;
		cin>>t;
		d = d*t;
	}
	if(d==0){
		for(int i=0; i<n; i++){
			cout<<0<<" ";
		}
	} else {
		for(int i=0; i<n; i++){
			cout<<arr[i]/d<<" ";
		}
	}
    return 0;
}
