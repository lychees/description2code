#include<iostream>
#include<map>
#include<stdio.h>
using namespace std;
bool prime[1000001];
long long int primeCount[1000001];


void preComp(){
	primeCount[0]=0;
	primeCount[1]=0;
	primeCount[2]=1;
	prime[0]=true;
	prime[1]=true;
	prime[2]=false;
	for(long long int i=2;i<=1000000;i++){
		if(!prime[i]){
			primeCount[i]=primeCount[i-1]+1;
			for(long long int j=i*i;j<=1000000;j+=i){
				prime[j]=true;
			}
		}
		else{
			primeCount[i]=primeCount[i-1];
		}
	}
}

int main(){
	preComp();
	int t;
	cin>>t;
	while(t--){
		long int l,r;
		cin>>l>>r;
		printf("%.6f\n",((float)primeCount[r]-primeCount[l-1])/((float)r-l+1));
	}	
	return 0;
}
