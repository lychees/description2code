#include <bits/stdc++.h>
typedef long long ll;
using namespace std;

int main()
{
    int n;
    cin>>n;
    int* arr=new int[n];
    for(int i=0;i<n;i++)
    	cin>>arr[i];
    int q;
    cin>>q;ll z=1;
    while(q--)
    {
    	ll a;cin>>a;
    	z*=a;
    }
    for(int i=0;i<n;i++)
    {
    	if(z!=0)
    		arr[i]/=z;
    	else
    		arr[i]=0;
    }
    for(int i=0;i<n;i++)
    	cout<<arr[i]<<" ";
    return 0;
}
