n=int(raw_input())
assert n>2 and n<=1000
h=raw_input().split()
for i in xrange(0,len(h)):
	h[i]=int(h[i])
	assert h[i]>0 and h[i]<=1000
h=sorted(h)
m=h[1]-h[0]
for i in xrange(0,len(h)-2):
	d=h[i+2]-h[i]
	if d>m:
		m=d
d=h[n-1]-h[n-2]
if d>m:
	m=d
