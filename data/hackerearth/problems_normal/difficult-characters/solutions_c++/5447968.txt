#include <iostream>
#include <algorithm>
#include <vector>
#include <string>
using namespace std;

struct charLevel {
	char c;
	int frequency;
};

bool cmp(const charLevel &a, const charLevel &b) {
	if (a.frequency == b.frequency) {
		return a.c > b.c;
	}
	
	return a.frequency < b.frequency;
}

int main()
{
	vector<charLevel> charsTemplate;
	for (int i = 0; i < 26; i++) {
		charLevel temp;
		temp.c = 'a' + i;
		temp.frequency = 0;
		charsTemplate.push_back(temp);
	}
	
	int t;
	string s;
	vector<charLevel> chars;
	
	cin >> t;
	for (int i = 0; i < t; i++) {
		cin >> s;
		chars = charsTemplate;
		
		for (int j = 0; j < s.length(); j++) {
			chars[ s[j] - 'a' ].frequency++;
		}
	
		sort(chars.begin(), chars.end(), cmp);
		
		for (int j = 0; j < 26; j++) {
			cout << chars[j].c << " ";
		}
		
		cout << endl;
	}
	
    return 0;
}
