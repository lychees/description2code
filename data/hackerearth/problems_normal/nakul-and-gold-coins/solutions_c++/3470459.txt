#include <iostream>

using namespace std;
const int MAX = 1000001;
int p[MAX], np;
void ciur()
{
    int i;
    np = 1;
    p[1] = 2;
    for(i=3; i*i<MAX; i+=2)
        if(p[i]==0){
            p[++np] = i;
            for(int j=i*i; j<MAX; j=j+i+i)
                p[j] = 1;
        }

    for(; i<MAX; i+=2)
        if(p[i]==0)
            p[++np] = i;
}
int main()
{
    ciur();
    int t; cin>>t;
    while(t--)
    {
        unsigned l, r, ans = 0, mic = np, mare = np;
        cin>>l>>r;
        for(int i=1; i<=np; i++){
            while(mic>0 and p[i]*p[mic]>=l)
                mic--;
            while(mare<=np and p[i]*p[mare]>r)
                mare--;
            int a = mic+1, b = mare;
            if(a<=i)
                a = i+1;
            if(a<=b)
                ans += b-a+1;
        }
        cout<<ans<<'\n';
    }
    return 0;
}
