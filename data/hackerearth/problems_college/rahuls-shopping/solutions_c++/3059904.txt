#include <bits/stdc++.h>
using namespace std;
int fn(int a) {
	int i=0;
	while(a!=0) {
		i+=(a%10);
		a/=10;
	}
	return i;
}

int main()
{
    int x, n, i, l, w, s;
    cin>>n;
    float z;
    vector <int> a;
    for(i=0;i<n;i++) {
    	cin>>x;
    	a.push_back(x);
    }
    cin>>l>>w;
    l=l*w;
    s=0;
    for(i=0;i<n;i++) {
    	w=fn(a[i]);
    	z=a[i]*w*l;

    		z=z*0.75;
    	else if(a[i]>10&&a[i]<21)
    		z=z*0.70;
    	else
    		z=z*0.80;
   	 	s+=z;
    }
    cout<<s;
    return 0;
}
