#include<iostream>
using namespace std;
int main()
{
int size;
cin>>size;
int a[size];
   for(int i=0;i<size;i++)
   {
	cin>>a[i];
    }
int queries,l,m;
cin>>queries;
for(int j=0;j<queries;j++)
    {
    	int sum=0;
    	cin>>l;
    	cin>>m;
    	for(int p=l-1;p<m;p++)
    	{
    		sum=sum+a[p];
    	}
    	cout<<sum<<"\n";
    	
    }
}
