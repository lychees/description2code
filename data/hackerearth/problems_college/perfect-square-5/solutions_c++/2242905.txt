#include <bits/stdc++.h>
using namespace std;
#define ll long long

int main(){
	int t;cin>>t;
	while(t--){
		ll n,p,q,r; cin>>n;
	    ll k=sqrt(n);
	    if(k*k==n){
	    	cout<<"YES"<<endl;
	    }
	    else{
		    p=n-k*k;
		    q=k+1;
		    r=q*q-n;
		    if(abs(p)<r){
		    	cout<<-p<<endl;
		    }
		    else{
		    	cout<<"+"<<r<<endl;
		    }	
	    }
	}	
	
    //cout << "Hello World!" << endl;
    return 0;
}
