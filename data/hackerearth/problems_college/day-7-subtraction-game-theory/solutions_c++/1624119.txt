#include<bits/stdc++.h>
#define mod 1000000007
#define max 1000000
using namespace std;
 
typedef long long ll;
 
int main()
{
	//ios::sync_with_stdio(false);	
    ll x,y, arr[max+1];    
    //cin >> x >> y;
    scanf("%lld%lld", &x, &y);
    
    arr[0]=1;
    for (int i = 1; i <= max; ++i)
        arr[i]=(x*arr[i-1] - y) % mod;
        
    int q;
    //cin >> q;
    scanf("%d", &q);
    while(q--)
    {
        int a;
        //cin >> a;
        scanf("%d", &a);
        //cout << arr[a] << "\n";
        printf("%d\n", arr[a]);
    }
    
    return 0;
 }
