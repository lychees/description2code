#include<bits/stdc++.h>

using namespace std;

int prime( long long int n)
{
	long long int i,count = 0;
	
	for( i = 1; i <= n/2; i++ )
	{
		if( (n % i) == 0 )
		{
			count++;
		}
	}
	
	if ( count > 2)
	return 0;
	else
	return 1;
}


int main()
{
	int t;
	cin>>t;
	
	while(t--)
	{
		long long int n,k;
		cin>>n;
		
		if (n == 3)
		cout<< n<<endl;
		
		else if ( n == 1)
		cout << n<<endl;
		
		else
		{
			k = prime(n);
		
			if(k)
			cout<<"2"<<endl;
			else
			cout<<"1"<<endl;
		}
	}
	
	return 0;
	
}
