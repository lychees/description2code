import sys
ip=[]
for line in sys.stdin:
    ip.append(line)

for k in range(0,len(ip)):
    ip[k]=ip[k].rstrip('\n')
ip[0]=int(ip[0])
for i in range(1,ip[0]+1):
    temp_ip=ip[i].split()
    temp=[]
    l=int(temp_ip[0])
    r=int(temp_ip[1])
    for i in range(l,r+1):
        for j in range(i+1,r+1):
            temp.append(i^j)
    print max(temp)
