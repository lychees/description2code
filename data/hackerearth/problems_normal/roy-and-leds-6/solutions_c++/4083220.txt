#include<bits/stdc++.h>
using namespace std;
void initialise(bool *arr,int t,int r)
{
    int s=0;
    bool v=0;
    while(s<t)
    {
        for(int i=0;i+s<t && i<r;++i)arr[i+s]=v;
        v=1-v;
        s+=r;
    }
}
int main()
{
    int t,r,g,b;
    cin >> t >> r >> g >> b;
    bool red[t],green[t],blue[t];
    initialise(red,t,r);
    initialise(green,t,g);
    initialise(blue,t,b);
    int BK=0,W=0,M=0,C=0,Y=0,BL=0,G=0,R=0;
    for(int i=0;i<t;i++)
    {
        if(!red[i] && !green[i] && !blue[i])BK++;
        else if(red[i] && green[i] && blue[i]) W++;
        else if(red[i] && !green[i] && blue[i]) M++;
        else if(!red[i] && green[i] && blue[i]) C++;
        else if(red[i] && green[i] && !blue[i]) Y++;
        else if(!red[i] && !green[i] && blue[i])BL++;
        else if(!red[i] && green[i] && !blue[i]) G++;
        else R++;
    }
    cout<<R<<" "<<G<<" "<<BL<<" "<<Y<<" "<<C<<" "<<M<<" "<<W<<" "<<BK;
    return 0;
}
