#include<stdio.h>
int ch;
int max,i,j,index;
int b[21],c[21];
void read(int *p)
{
 *p=0;
 ch=getchar_unlocked();
 while(ch>='0' && ch<='9')
      {
        *p=(*p<<3)+(*p<<1)+ch-48;
         ch=getchar_unlocked();
      }
}
void sort(int *p,int n)
 {
      for(i=0;i<n-1;++i)
       {
          max=p[i];index=i;
          for(j=i+1;j<n;++j)if(max<p[j]) {max=p[j];index=j;}
          if(i!=index)
            {
              p[i]=p[i]^p[index];
              p[index]=p[i]^p[index];
              p[i]=p[i]^p[index];
            }
       }
 }
int main()
{
int t,n,pos,cnt=1;
read(&t);
 while(t--)
  {
    read(&n);
    for(i=0;i<n;++i) read(&pos);
    for(i=0;i<n;++i) {read(&b[i]);c[i]=b[i];}
    sort(b,n);
    printf("Case #%d: ",cnt);
    for(i=0;i<n;++i)
     {
       ch=b[i],pos=i;
       for(j=0;j<n;++j)
       {
        if(c[j]==ch)
        {
         pos=j;break;
        }
       }
       c[pos]=-1;
       printf("%d ",pos);
     }
    putchar_unlocked(10);
    cnt++;
  }
}
