#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n,q;
    cin>>n>>q;
    int a[n];
    int i,j;
    for(i=0;i<n;i++)
    cin>>a[i];
    while(q--)
    {
    	int x,y;
    	cin>>x>>y;
    	if(x==1)
    	{
    		for(i=0;i<n;i++)
    		{
    			if(a[i]>=y)
    			{
    				cout<<i+1<<endl;
    				break;
    			}
    		}

    		cout<<"-1"<<endl;
    	}
    	else if(x==0)
    	{
    		int c;
    		cin>>c;
    		a[y-1]=c;
    		
    	}
    }
    return 0;
}
