#include <iostream>
#include <cstring>
#include <vector>
using namespace std;
struct TrieNode {
	int next[26];
	TrieNode() {
		memset(next, 0, sizeof next);
	}
};
struct Trie {
	int allocated;
	vector<TrieNode> nodes;
	Trie() { allocated = 1;
		nodes.push_back(TrieNode());
	}

	void clear() {
		allocated = 1;
	}

	void insert(char * s) {
		int curr = 0;
		for (int i=0;s[i];i++) {
			int successor = nodes[curr].next[s[i] - 'a'];
			if (successor) {
				curr = successor;
			} else {
				nodes.push_back(TrieNode());
				nodes[curr].next[s[i] - 'a'] = allocated++;
				curr = allocated - 1;
			}
		}
	}
};

int main(){
	Trie trie;
	int t;
	cin >> t;
	while(t--){
		char s[40];
		cin >> s;
		trie.insert(s);
	}
	cout << trie.allocated << endl;
	return 0;
}
