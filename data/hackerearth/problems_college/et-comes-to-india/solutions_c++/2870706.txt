

    #include<bits/stdc++.h>
    #define sc(n) scanf("%d",&n)
    #define scl(n) scanf("%lld",&n)
    #define ll long long
    using namespace std;
    bool solve(int n,ll int k,bool flag)
    {
        ll int l=((ll)1<<n)-1;
        if(k==(l+1)/2)
            return flag;
        else if(k<(l+1)/2)
            return solve(n-1,k,flag);
        else
            return solve(n-1,l+1-k,!flag);
    }
    int main()
    {
        int t,n;
        ll int k;
        sc(t);
        while(t--)
        {
            sc(n);
            scl(k);
            if(k>((ll)1<<n)-1)
                cout<<-1<<endl;
            else
                solve(n,k,0)==1?cout<<"C\n":cout<<"G\n";
        }
        return 0;
    }
