#include <iostream>
#include <algorithm>

using namespace std;

bool arjit[1001][1001], chandan[1001][1001];

int findGCD(int a, int b){
	if(b == 0)
		return a;
	return findGCD(b, a%b);
}

void preProcess()
{
	for(int i =1; i<1001; i++)
	{
		arjit[1][i] = chandan[i][1] = 0;
		arjit[i][1] = chandan[1][i] = 1;		
	}

	int gcd;
	for(int i = 2; i<1001; i++)
	{
		for(int j = 2; j<1001; j++)
		{
			gcd = findGCD(i,j);
			arjit[i][j] = !chandan[i][j-1] || (gcd > 1 && !chandan[i][j/gcd]);
			chandan[i][j] = !arjit[i-1][j] || (gcd > 1 && !arjit[i/gcd][j]);
		}
	}
}


int main()
{
	int t, a, b;
	preProcess();
	cin>>t;

	while(t--){
		cin>>a>>b;
		if(a==1 && b==1)
			cout<<"Draw\n";
		else if(arjit[a][b])
			cout<<"Arjit\n";
		else cout<<"Chandu Don\n";
	}
	return 0;
}
