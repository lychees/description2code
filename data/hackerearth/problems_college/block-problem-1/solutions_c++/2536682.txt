	long long int power( long long int n, int p)
	{	 
		  if(p==0)
			return 1;
		  if(p==1)
		  	return n;
		 int i;
		   long long int nu=n;
		  fr(i,2,p+1) {
		    n*=nu;
		    n%=MOD;
		  }
		  return n;
		}
	int main()
		{
		  int N, M;
		  long long int T[1001];
		  long long int S[1001];
		  long long int P[1001];
		 int i;
		  T[0] = T[1] = 1;
		  T[2] = 2 ;
		  T[3] =  4;
		  T[4] = 8 ;
		  P[0] = P[1] = 1;
		  fr(i,5,1001)
		    T[i] = (T[i-1] + T[i-2] + T[i-3] + T[i-4])%MOD;
		 
		  S[0] = 1;
		  S[1] = 1;
		 
		  long long int s;
		  int Tt,t,j;
		  si(Tt);
		  fr(t,0,Tt) {
               si(N);
               si(M);
		    fr(i,0,M+1)
		      P[i]=power(T[i],N);
		 
		    fr(i,2,M+1) {
		      s=0;
		      fr(j,1,i) {
			s+=(S[j]*P[i-j])%MOD;
			s%=MOD;
		      }
		      S[i]=(P[i]-s);
		      S[i]=S[i]%MOD;
		    }
		    while(S[M]<0)
		      S[M]+=MOD;

		  }
