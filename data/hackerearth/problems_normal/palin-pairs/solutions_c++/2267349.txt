#include<iostream>
#include<cstring>
#include<sstream>
#include<map>

using namespace std;
typedef long long ll;

int main()
{
	string str, rstr;
	map<string, int> m;
	map<string, int>::iterator it, itf;
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
	{
		cin >> str;
		m[str]++;
	}
	ll c = 0;
	for (it = m.begin(); it != m.end(); it++)
	{
		rstr = it->first;
		int len = rstr.length();
		for (int i = 0; i < len / 2; i++)
		{
			rstr[i] = rstr[i] ^ rstr[len - 1 - i];
			rstr[len - 1 - i] = rstr[i] ^ rstr[len - 1 - i];
			rstr[i] = rstr[i] ^ rstr[len - 1 - i];
		}
		itf = m.find(rstr);
		if (itf != m.end())
		{
			c += ((ll)it->second*(ll)itf->second);
			m.erase(itf);
		}
	}
	cout << c << endl;
}
