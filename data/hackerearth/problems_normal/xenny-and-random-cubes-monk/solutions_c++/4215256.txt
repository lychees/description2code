#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define pii pair<int,int>
#define pll pair<ll,ll>
#define mp make_pair

#define ss second

#define IOS ios::sync_with_stdio(false)
#define MOD 1000000007
int n , k ;
char c[20][20];
string s;
int alpha[20][200];
int dp[10000][20];
int solve(int mask , int ind){
    if(ind == k)return 1;
    if(dp[mask][ind]!=-1)return dp[mask][ind];
    dp[mask][ind] = 0;
    for(int i = 0 ; i < n ; i++){
        if((mask & (1<<i))==0){
            dp[mask][ind] = (dp[mask][ind] + 1ll*alpha[i][s[ind]-'a']*solve(mask | 1 << i,ind+1))%MOD;
        }
    }
    return dp[mask][ind];
}
int main(){

	int i , j ;
    cin >> n >> k;
    memset(dp,-1,sizeof(dp));
    for(i = 0 ; i < n ; i++){
        for(j = 0 ; j < 6 ; j++){
            cin >> c[i][j];
            alpha[i][c[i][j]-'a']++;
        }
    }
    cin >> s;
    ll siz = s.size();
    ll tot = 0;
    tot = solve(0,0);
    cout << tot << endl;
	return 0;
}
