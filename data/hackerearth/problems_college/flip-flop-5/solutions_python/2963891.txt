
t = raw_input()
t = int(t)
for i in range(1,t+1):
	str_to_flip = raw_input()
	list_to_flip = list(str_to_flip)
	
	no_of_del = 0
	for j in range(1,len(list_to_flip)):
		
		prev = list_to_flip[(j - 1)]
		
		curr = list_to_flip[j]
		
#		next_one = list_to_flip[(idx + 1) % len(list_to_flip)]
		if prev != curr:
			continue
		else:
			no_of_del = no_of_del + 1
			
	print no_of_del
