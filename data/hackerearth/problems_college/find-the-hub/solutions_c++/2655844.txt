#include <iostream>
using namespace std;

int isprime(int n)
{
	for(int i=2;i*i<n;i++)
	{
		if(n%i==0)
			return 0;
	}
	return 1;
}

int main()
{
    int n,t,count=1,i,j,x=0;
    cin>>n;
    int ar[n],far[n];
    for(i=0;i<n;i++)
    {
    	cin>>ar[i];
    	if(isprime(ar[i]))
    	{
    		far[x]=ar[i];
    		x++;
    	}
    }
    for(i=0;i<x;i++)
    {
    	for(j=i+1;j<x;j++)
    	{
    		if(far[i]<far[j])
    		{
    			t=far[i];
    			far[i]=far[j];
    			far[j]=t;
    		}
    	}
    }
    for(i=0;i<x;i++)
    {
 		if(far[i]==far[i+1])
    		count++;
    	else
    		break;
    }
    cout<<far[i]<<"\n"<<count;
    return 0;
}
