#include <bits/stdc++.h>
using namespace std;
int main()
{
	std::ios_base::sync_with_stdio(false);
	int t; cin>>t;
	while(t--)
	{
		int n,k; cin>>n>>k;
		vector<vector<int> > v(k,vector<int>(26,0));
		for(int i = 0; i < n; ++i)
		{
			string s; cin>>s;
			for(int j = 0; j < k; ++j)
				v[j][s[j]-'a']++;
		}
		for(int j = 0; j < k; ++j)
		{
			int max=1000000000,index;
			for(int i = 0; i < 26; ++i)
			{
				int res = 0;
				for(int m = 0; m < 26; ++m)
					res += abs(i-m)*v[j][m];
				if(res<max)
				{
					max = res;
					index = i;
				}
			}
			cout<<(char)('a'+index);
		}
		cout<<endl;
	}	
	return 0;
}
