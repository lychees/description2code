#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	
    	long long int max=-1;
    	int n;
    	long long int k;
    	cin>>n>>k;
    	long long int a[n];
    	for(int i=0;i<n;i++)
    		cin>>a[i];
    	for(int i=0;i<n;i++){
    		long long int p;
    		cin>>p;
    		long long int temp=k/a[i];
    		if(max<(temp*p))
    			max=temp*p;
    	}
    	cout<<max<<endl;
    }
    return 0;
}
