#include<iostream>
using namespace std;
struct colour{
    int a;
    int b;
    int c;
};
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        colour v[n];
        for(int i=0;i<n;i++)
        {
            int x,y,z;
            cin>>x>>y>>z;
            v[i].a=x;
            v[i].b=y;
            v[i].c=z;
        }
        int s=0,d=0,f=0,t=0,r=0,e=0;
        for(int i=n-1;i>=0;i--)
        {
            s=min(v[i].a+r,v[i].a+e);
            d=min(v[i].b+t,v[i].b+e);
            f=min(v[i].c+t,v[i].c+r);
            t=s;
            r=d;
            e=f;
        }
        int j=min(s,d);
        cout<<min(j,f)<<endl;
    }
}
