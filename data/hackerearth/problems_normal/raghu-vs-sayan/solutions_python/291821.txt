t=input()
for _ in xrange(t):
	a,b,n=[int(x) for x in raw_input().split()]
	k=sorted([int(x) for x in raw_input().split()])
	rs,ss,rc,sc=0,0,0,0
	rd,sd=False,False
	for i in k:
		rs+=i
		if rs<=a:
			rc+=1
		else:
			rd=True
			
		ss+=i
		if ss<=b:
			sc+=1
		else:
			sd=True
		if (rs>=a and ss>=b) or (rd and sc>rc) or (sd and rc>sc):
			break
	if rc>sc:
		print "Raghu Won"
	elif rc<sc:
		print "Sayan Won"
	else:
		print "Tie"
