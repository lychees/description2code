#include <iostream>
#include <stdio.h>
using namespace std;
typedef long long int ll;
ll prime[1000005];

void sieve()
{
	ll n = 1000005;
	ll p, i;
	for(p =2; p * p < n; p++){
		if(prime[p] == 0){
			for(i= p *p ; i < n; i += p){
				if(prime[i] == 0){
					prime[i] = p;
				}
			}
		}
	}
}

int main() {
	sieve();
	ll t, a, y;
	scanf("%lld", &t);
	while(t--){
		scanf("%lld", &a);
		if(prime[a] == 0){
			printf("%d\n", prime[a]);
		}
		else{
			y = a - prime[a];
			printf("%d\n", y);
		}
	}
	return 0;
}
