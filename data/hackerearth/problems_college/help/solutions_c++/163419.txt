#define MaxVal 1000009
int tree[MaxVal]={};
int read(int idx){
    int sum = 0;
    while (idx > 0){
	sum += tree[idx];
	idx -= (idx & -idx);
    }
    return sum;
}
void update(int idx ,int val){
    while (idx <= MaxVal){
	tree[idx] += val;
	idx += (idx & -idx);
    }
}
