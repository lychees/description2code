#include<bits/stdc++.h>
using namespace std ;
vector<vector<int> > graph(10005);
long long int dp[10005];
long long int f(int node)
{
    if(dp[node]!=0)
        return dp[node];
    for(int i=0;i<graph[node].size();i++)
        dp[node]+=f(graph[node][i]);
    return dp[node];
}
int main()
{
    int N,a,b;
    cin>>N;
    cin>>a>>b;
    while(a!=0 && b!=0)
    {
        graph[b].push_back(a);
        cin>>a>>b;
    }
    /*for(int i=1;i<=N;i++)
    {
        cout<<i<<"->";
        for(int j=0;j<graph[i].size();j++)
            cout<<graph[i][j]<<" ";
        cout<<endl;
    }*/
    for(int i=0;i<=N;i++)
        dp[i]=0;
    dp[1]=1;
    f(N);
    cout<<dp[N]<<endl;
    return 0;
}
