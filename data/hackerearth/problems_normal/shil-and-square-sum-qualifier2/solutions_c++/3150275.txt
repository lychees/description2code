#include<bits/stdc++.h>
using namespace std;
#define int long long
const int MOD=1e9+7;
const int MAXN = 1e6+3;
long long suf[MAXN];
int ar[MAXN];
#undef int
int main() {
	#define int long long
	int n,k;
	cin>>n>>k;
	long long sub=0,ans=0;
	for(int i=1;i<=n;i++)
		scanf("%lld",&ar[i]);
	suf[n]=ar[n];
	suf[n+1]=0;
	for(int i=n-1;i>=1;i--) {
		suf[i]=suf[i+1]+ar[i];
		suf[i]%=MOD;
	}
	for(int i=1;i<=k;i++) {
		ans+=ar[i]*(i*i%MOD)%MOD;
		ans%=MOD;
		sub+=(2*i-1)*ar[i]%MOD;
		sub%=MOD;
	}
	cout<<ans<<" ";
	for(int i=k+1;i<=n;i++) {
		ans+=1LL*k*(k*ar[i]%MOD)%MOD;
		ans%=MOD;
		ans-=sub;
		if(ans<0)
			ans+=MOD;
		cout<<ans<<" ";
		sub-=ar[i-k];
		if(sub<0)
			sub+=MOD;
		sub-=2*((suf[i-k+1]-suf[i]+MOD)%MOD)%MOD;
		if(sub<0)
			sub+=MOD;
		sub+=(2*k-1)*ar[i]%MOD;
		sub%=MOD;
	}
	cout<<endl;
	return 0;
}
