#include <iostream>
using namespace std;

int main()
{
	int t,n,i,des;
	cin >> t;
	while(t--)
	{
		cin >> n;
		int arr[n];
		int count[n];
		for(i=0;i<n;i++)
		{
		cin >> arr[i];
		count[i]=0;
		}
		des=0;
		while(1)
		{
			count[arr[des]]++;
			des=arr[des];
			if(count[arr[des]]>1)
			{
				cout << "Lose"<<endl;
				break;
			}
			if(arr[des]==-1)
			{
				cout << "Win"<<endl;
				break;
			}
		}
	}
    return 0;
}
