#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define ld long double
#define mod 1000000007

void solve()
{
    int n;
    cin>>n;
    vector<int> arr(n);
    for(int i=0;i<n;++i)
        cin>>arr[i];
    int pos_count = 1;
    int neg_count = 1;
    for(int i = 1; i < n; ++i)
    {
        int temppos=pos_count,tempneg=neg_count;
        if (arr[i-1] <= arr[i])
          temppos = neg_count + 1;
        if (arr[i-1] >= arr[i])
          tempneg = pos_count+1;
        pos_count=temppos;
        neg_count=tempneg;
    }
    cout<<max(pos_count,neg_count)<<'\n';
}
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    int t;
    cin>>t;
    while(t--)
        solve();
    return 0;
}
