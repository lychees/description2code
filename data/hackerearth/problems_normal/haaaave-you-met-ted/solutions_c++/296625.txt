#include <iostream>
#include <cinttypes>

using namespace std;

int set_bits(uint32_t a) {
	int count;

	count = 0;
	
	while (a) {
		count += (a & 1);
		a >>= 1;
	}

	return count;
}

int main()
{
	int t;
	int n;
	int min;
	int count;
	uint32_t a[100000];

	cin >> t;

	while (t--) {
		cin >> n;

		for (int i = 0; i < n; i++) {
			cin >> a[i];
		}
		
		min = 32;

		for (int i = 0; i < n; i++) {
			count = set_bits(a[i]);

			if (count < min) {
				min = count;
			}
			if (min == 1) {
				break;
			}
		}

		cout << min << endl;
	}

	return 0;
}
