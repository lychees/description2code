t = int(raw_input())

for i in range(t):
	tx = raw_input().split()
	kx = []
	for s in tx:
		k = ""
		while s:
			l = len(s)
			if l > 1:
				if l % 2 == 0:
					m2 = l / 2
					m1 = m2 - 1
					if s[m1] < s[m2]:
						m = m1
					else:
						m = m2
				else:
					m = (l / 2)
			else:
				m = 0
			
			k = k + s[m]
			s = s[:m] + s[m+1:]
		kx.append(k)
		
	print " ".join(kx)
