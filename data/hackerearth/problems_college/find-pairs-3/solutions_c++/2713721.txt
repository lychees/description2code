#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,x;
		vector<int>v;
		cin>>n;
		for(int i=0;i<n;i++)
		{
			cin>>x;
			v.push_back(x);
		}
		int cnt=0;
		for(int i=0;i<n;i++)
		{
			for(int j=i;j<n;j++)
			{
				if(v[i]==v[j])
					cnt++;
			}
		}
		cout<<cnt<<endl;
	}
}
