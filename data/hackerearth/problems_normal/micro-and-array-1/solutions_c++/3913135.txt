#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   long n,q;
   cin>>n>>q;
   int i;
   long a[n];
   for(i=1;i<=n;i++)
   {
   	cin>>a[i];
   }
   while(q--)
   {
   		int oper;
   		cin>>oper;
   		long val1;
   		int pos=-1;
   		if(oper==1)
   		{
   			cin>>val1;
   			for(i=1;i<=n;i++)
   			{
   				if(a[i]>=val1)
   				{
   					pos=i;
   					break;
   				}
   			}
   			cout<<pos<<endl;
   		}
   		else
   		{
   			long data1,data2;
   			cin>>data1>>data2;
   			a[data1]=data2;
   		}
   }

}
