#include<bits/stdc++.h>
using namespace std;
#define ll long long int
ll a[1000006];
ll b[1000006];
ll ab(ll r)
{
    if(r>0)return r;
    return -r;
}
int main()
{
    ll n,i,j;
    cin>>n;
    for(i=0;i<n;i++)
        cin>>a[i];
    sort(a,a+n);
    ll k=0;
    for(i=0,j=n-1;i<=j;i++,j--)
    {
        if(i==j)
        {
            b[k++]=a[i];
        }
        else
        {
            b[k++]=a[i];
            b[k++]=a[j];
        }
    }
    b[k]=a[0];
    ll sol=0;
    for(i=0;i<n;i++)
    {
        sol+=ab(b[i]-b[i+1]);
    }
    cout<<sol;
    return 0;
}
