#include<bits/stdc++.h>
using namespace std;
bool prime[10005];
void primeno(){
	memset(prime,true,sizeof(prime));
	prime[0]=false;
	prime[1]=false;
	for(int i=2;i<10005;i++){
		if(prime[i]==true){
			for(int j=i*i;j<10005;j+=i){
				prime[j]=false;
			}
		}
	}
}
int main(){
	primeno();
	int n;
	cin>>n;
	for(int i=4;i<=n;i++){
		if(prime[i]==false)cout<<i<<" ";
	}
	cout<<endl;
}
