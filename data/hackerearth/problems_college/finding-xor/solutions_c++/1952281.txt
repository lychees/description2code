#include<stdio.h>
int main()
{
	char a[100000],b[100000],sol[100000];
	int t,i;
	scanf("%d",&t);
	while(t--)
	{
			scanf("%s",&a);
			scanf("%s",&b);
			for(i=0;a[i]!='\0';i++)
			{
				if(a[i]==b[i])
				sol[i]='0';
				else
				sol[i]='1';
				
			}
			sol[i]='\0';
			printf("%s\n",sol);
		
	}
	    return 0;
}
