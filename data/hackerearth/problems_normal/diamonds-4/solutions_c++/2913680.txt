#include <stdio.h>
#include <vector>
#include <iostream>

using namespace std;

int main()
{
	int t;
	cin >> t;
	for(int i=0;i < t;i++)
	{
	    int n,m;
	    cin >> n >> m;
	    int d=0;
	    vector< vector<char> > arr(n,vector<char>(m,'0'));
	    for(int i=0;i < n;i++)
	    {
	    	for(int j=0;j < m;j++)
	    	{
	    		cin >> arr[i][j];
	    	}
	    }
	    for(int i=0;i < n;i=i+1)
	    {
	    	for(int j=0;j < m;j=j+1)
	    	{
	    		if(i+1 < n)
	    		{
	    			if( (arr[i][j]==('/')) && (arr[i+1][j]==('\\')) )
	    			{
	    				if(j+1 < m)
	    				{
	    					if( (arr[i][j+1]==('\\')) && (arr[i+1][j+1]==('/')) )
	    					{ 
	    						d++;
	    					}
	    				}
	    			}
	    		}
	    	}
	    }
	    cout << d << "\n";
	}
    return 0;
}
