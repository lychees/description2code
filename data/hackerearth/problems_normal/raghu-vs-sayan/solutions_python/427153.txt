t = int(input())
for i in xrange(t):
	r,s,n = map(int,raw_input().split())
	cal = map(int,raw_input().split())
	cal = sorted(cal)
	sd = 0
	rd = 0
	for j in cal:
		if j<=r:
			r-=j
			rd+=1
		if j<=s:
			s-=j
			sd+=1
	if rd>sd:
		print("Raghu Won")
	elif rd<sd:
		print("Sayan Won")
	else:
		print("Tie")
