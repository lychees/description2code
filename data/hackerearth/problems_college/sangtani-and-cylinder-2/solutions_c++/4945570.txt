#include <iostream>
#include<stdlib.h>
#include<cstdio>
#include<math.h>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while(t--)
	{
   double n;
   cin >> n;
  double pi=2*acos(0.0);
   double temp=pow(n/(2*pi),1.0/3);
   double ans=((2*n)/temp)+(2*pi*temp*temp);
   printf("%.9lf\n",ans);
	}
    return 0;
}
