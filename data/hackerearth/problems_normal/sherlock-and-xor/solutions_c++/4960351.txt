#include <iostream>
using namespace std;

int main()
{
	int T;
    cin>>T;
    while(T--){
    	long int n;
    	cin>>n;
    	int odd=0,even=0;
    	while(n--){
    		int num;
    		cin>>num;
    		num&1?odd++:even++;
    	}
    	cout<<(long int)odd*(long int)even<<"\n";
    }
    cout<<endl;
    return 0;
}
