#include <bits/stdc++.h>
using namespace std;

int dp[1010][1010];

int solve(int arr[], int n)
{
    for(int i=n-1;i>=0;i--)
    {
        for(int j=i;j<=n-1;j++)
        {
            if(i==j)
            {
                dp[i][j] = arr[i];
            }
            else if(j-i==1)
            {
                dp[i][j] = max(arr[i], arr[j]);
            }
            else
            {
                dp[i][j] = max(arr[i]+min(dp[i+2][j], dp[i+1][j-1]), arr[j]+min(dp[i][j-2],dp[i+1][j-1]));
            }
        }
    }
    return dp[0][n-1];
}

int main()
{
    int t;
    scanf(" %d",&t);
    while(t--)
    {
        int n;
        scanf(" %d",&n);
        int a[1010];
        int sum = 0;
        for(int i=0;i<n;i++)
        {
            scanf(" %d",&a[i]);
            sum += a[i];
        }
        int res = solve(a,n);
        if(sum%2 == 0)
        {
            if(res == sum/2)
            {
                printf("Tie\n");
            }
            else if(res > sum/2)
            {
                printf("Ravi %d\n",res);
            }
            else
            {
                printf("Abhinav %d\n",sum-res);
            }
        }
        else
        {
            if(res > sum/2)
            {
                printf("Ravi %d\n",res);
            }
            else
            {
                printf("Abhinav %d\n",sum-res);
            }
        }
    }
    return 0;
}
