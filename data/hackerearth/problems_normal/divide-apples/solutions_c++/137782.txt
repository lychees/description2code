#include <bits/stdc++.h>

using namespace std;

int main()
{
     long long n;
    cin>>n;
     long long arr[n];
     long long sum=0;
    for( long long i=0;i<n;i++)
    {
        cin>>arr[i];
        sum+=arr[i];
    }

    sum=sum/n;

     long long a[n];

    for( long long i=0;i<n;i++)
    {
        a[i]=sum-arr[i];
    }

    for( long long i=n-2;i>0;i--)
    {
        a[i]=a[i+1]+a[i];
    }
    a[0]=0;

    sort(a,a+n);

     long long median=(a[n/2]+a[(n-1)/2])*0.5;

     long long ans=0;
    for( long long i=0;i<n;i++)
    {
        a[i]-=median;
        ans+=abs(a[i]);
    }
    cout<<ans<<endl;
    //sort(arr,arr+n);


    return 0;
}
