#include <bits/stdc++.h>
using namespace std;
int cost[1001], weight[1001];
int main(){
int test_case,n, W;
scanf("%d", &test_case);
while(test_case--){
	scanf("%d %d", &n, &W);
	for(int i=0; i<n; i++){
		scanf("%d", &weight[i]);
	}
	for(int i=0; i<n; i++){
		scanf("%d", &cost[i]);
	}
	int dp[n+1][W+1];
	for(int j=0; j<=W; j++){
		dp[0][j]=0;
	}
	for(int j=0; j<=n; j++){
		dp[j][0]=0;
	}
	int p1=INT_MAX, p2=-1;
	for(int i=1; i<=n; i++){
		for(int j=0; j<=W; j++){
			dp[i][j] = dp[i-1][j];
			if(j-weight[i-1] >= 0){
				dp[i][j] = max(dp[i-1][j], dp[i-1][j-weight[i-1]]+cost[i-1]);

			}

		}
	}
	cout<<dp[n][W]<<endl;
}
}
