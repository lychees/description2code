#include <bits/stdc++.h>
using namespace std;
#define ll long long

bool check(ll step,ll arr[],ll N,ll K)
{
	ll stepsize=0,k=0,i=0;
	while(true)
	{
		stepsize += arr[i+1]-arr[i];
		if(stepsize<=step)
		{
			i++;
			if(i == N-1) return true;
		}
		else
		{
			k++;
			stepsize = 0;
			if(k==K) return false;
		}
	}
}

int main(int argc, char const *argv[])
{
	ios::sync_with_stdio(false);

	ll t,n,k;
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		ll arr[n];
		for(ll i=0;i<n;++i) cin>>arr[i];

		ll start=0,end=arr[n-1]-arr[0],ans;
		while(start<=end)
		{
			ll mid = (start+end)>>1;

			if(check(mid,arr,n,k))
			{
				ans = mid;
				end = mid-1;
			}

			else
				start = mid+1;
		}
		cout<<ans<<endl;

	}	
	return 0;
}
