#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <string>
#include <vector>
#include <math.h>

using namespace std;

int main() {
    ios::sync_with_stdio(0);
    int t, n, k, x;
    cin >> t;
    while (t --> 0) {
        cin >> n >> k;
        while (n --> 0) {
            cin >> x;
            k -= (x <= 0);
        }
        if (k > 0)
            cout << "YES\n";
        else
            cout << "NO\n";
    }
    return 0;
}
