#include <iostream>
using namespace std;
typedef long long int ll;
ll a[2000004];
int main()
{
	
    ll max = 2000000;
    ll i, j;
    for(i = 2; i*i <= max; i++) {
    	if(a[i] == 0) {
    		for(j = i * i; j <= max; j+=i) {
    			a[j] = 1;
    		}
    	}
    }
    for(i = 2; i <= max; i++) {
    	a[i] = 1 - a[i];
    }
    for(i = 2; i <= max; i++) {
    	a[i] += a[i -1];
    }
    ll n;
    cin >> n;
    ll s = 0;
    ll mod = 1000000007;
    s = s + a[n - 1];
    for(i = 1 ; i < n; i++) {
    	s = s + a[n - 1 + i] - a[i - 1];
    	s = s%mod;
    }
    cout << s << endl;
    return 0;
}
