#include<iostream>
#include<stdio.h>
#define si(t) scanf("%d",&t)
#define sii(t,t2) scanf("%d %d",&t,&t2)

using namespace std;

int main()
{
    int t;
    si(t);
    while(t--)
    {
        int n,k;
        long long int tot,ev,od,tmp,evsm=0,odsm=0;
        ev=od=0;
        si(n);
        tot=((n+1)*n)/2;
        while(n--)
        {
            si(k);
            if(k&1)
            {
                tmp=ev;
                ev=od;
                od=tmp+1;
            }
            else
                ev=ev+1;
            evsm+=ev;
            odsm+=od;
        }
        if(evsm>odsm)
            printf("ANDI\n");
        else
            printf("BOB\n");
    }
    return 0;
}
