#include <iostream>
#include <utility>
using namespace std;


pair<unsigned long long, unsigned long long> answers[91];

int main()
{
	answers[0] = make_pair(1LL, 0LL);
	
	for (int i = 1; i <= 90; i++) {
		auto ones = answers[i - 1].first;
		auto zeros = answers[i - 1].second;
		answers[i] = make_pair(zeros, zeros + ones);
	}	
	
	int T;
	cin >> T;
	while (T--) {
		int M;
		cin >> M;
		cout << answers[M].first << " " << answers[M].second << endl;
	}
    return 0;
}
