#include <bits/stdc++.h>
using namespace std;
void isPossible(long long int *a,long long int *b,long long int n,long long int l)
{
	
	for(long long int i=0;i<n;i++)
	{
		for(long long int j=0;j+1<n;j++)
		{
			if(a[j]>a[j+1])
			{
				swap(a[j],a[j+1]);
				swap(b[j],b[j+1]);
			}
		}
	}
/*	for(long long int i=0;i<n;i++)
	{
		cout<<a[i]<<"  "<<b[i]<<"  \n";
	}
*/	long long int initial=0,end=0;
	for(long long int i=0;i<n;i++)
	{
		initial=a[i],end=b[i];
		if(end-initial>l)
		{
			
		}
		for(long long int j=i;j<n;j++)
		{
			if(end>=a[j] && end<=b[j])
			{
				if((b[j]-initial)<l)
				{
					end=b[j];
				}
				else if((b[j]-initial)==l)
				{
					end=b[j];
					cout<<"Yes\n";
					return;
				}
				
			}
		}
	}
	
	if((end-initial)==l)
	{
		cout<<"Yes\n";
	}
	else
	{
		cout<<"No\n";
	}
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int n,l;
		cin>>n>>l;
		long long int a[n],b[n],i=0;
		while(i<n)
		{
			cin>>a[i]>>b[i];
			i++;
		}
		isPossible(a,b,n,l);

//		cout<<n<<"   "<<l<<"   \n";
	}
}
