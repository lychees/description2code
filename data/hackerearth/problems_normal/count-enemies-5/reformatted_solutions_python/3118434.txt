def enemies(string):
    string = string.split('*')
    count = 0
    for check in range(len(string)):
        if (string[check].count('O') > 0 and string[check].count('X') == 0):
            count = count+string[check].count('O')
        else:
            continue
    return count


result_set = []
no_of_testcases = int(input())
for testcases in range(no_of_testcases):
    string = raw_input()
    enemy_count = enemies(string)
    result_set.append(enemy_count)
for result in result_set:
    print(result)
