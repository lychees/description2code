#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
string str;
cin>>str;
ll len = str.size();
int t;
cin>>t;
while(t--){
    ll a,b;
    cin>>a>>b;
    a = (a-1)%len;
    b = (b-1)%len;
   cout<<(str[a] == str[b] ? "Yes" : "No")<<endl;
  }
  return 0;
}
