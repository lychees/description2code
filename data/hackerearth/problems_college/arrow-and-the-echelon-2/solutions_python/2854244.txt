n=int(raw_input())
for j in range(n):
	x=int(raw_input())
	a=[]
	if x%3==0:
		s=[5]*x
		print "".join(map(str,s))
	elif x%5==0:
		s=[3]*x
		print "".join(map(str,s))
	else:
		for i in range(x):
			if i%3==0:
				a.append(i)
		a.sort()
		f=0
		l=len(a)-1
		while l>=0:
			if (x-a[l])%5==0:
				d=[5]*a[l]+[3]*(x-a[l])
				print "".join(map(str,d))
				f=1
				break
			l-=1
		if f!=1:
			print -1
