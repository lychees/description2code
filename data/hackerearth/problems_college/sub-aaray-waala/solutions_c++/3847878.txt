#include<cstdio>
#include<iostream>
#include<set>
using namespace std;
#define km 1000000007
long long int power(long long int a,long long int b)
{
    long long int temp;
    if(b==0)
        return 1;
    temp=power(a,b/2);
    if(b%2==0)
        return (temp*temp)%km;
    else
        return (((temp*temp)%km)*a)%km;


}
int main()
{
    int t;
    long long int temp,n,i;

    scanf("%d",&t);
    while(t--)
    {
        set<long long int> ss;
        scanf("%lld",&n);
        //long long int a[n];
        for(i=0;i<n;i++)
            {
                scanf("%lld",&temp);
                ss.insert(temp);

            }
            printf("%lld\n",power(n,ss.size()));

    }
}
