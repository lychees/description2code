#include <bits/stdc++.h>
using namespace std;

long a[50005];

int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		int n;
		scanf("%d",&n);
		unsigned long long v=0;
		for(int i=0;i<n;i++){
			cin>>a[i];
			v+=a[i];
		}
		v/=(n-1);
		for(int i=0;i<n;i++){
			long long x=v-a[i];
			cout<<x<<" ";
		}
		cout<<endl;
	}


}
