#include <bits/stdc++.h>
using namespace std;
void print_binary(int);
int main(){
int n;
cin>>n;
int a[n];
for(int i=0; i<n; i++){
	cin>>a[i];
}
int max_=0;
int L[n];
for(int i=0; i<n; i++){
	L[i]=1;
	for(int j=i-1; j>=0; j--){
		if(a[i]>a[j] && L[i]<L[j]+1){
			L[i]=L[j]+1;
		}
	}
	max_=max(max_, L[i]);
}
print_binary(max_);
}
void print_binary(int max_){
	string s=bitset<8>(max_).to_string();
	int pos=s.find("1");
	if(pos!=-1){
		s = s.substr(pos, s.size()-pos);
		cout<<s<<endl;
	}else{
		cout<<0<<endl;
	}
}
