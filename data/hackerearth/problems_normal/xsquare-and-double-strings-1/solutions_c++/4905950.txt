#include <iostream>
#include<string.h>
//include <bits/stdc++.h>
using namespace std;

int cnt[32];

int main()
{
	int t;
	string s;
	cin >> t;
	while ( t-- ) {
		int check=1;
		cin >> s;
		memset(cnt, 0, sizeof(cnt));
		for ( int i = 0; i < s.size(); i++ ) 
			cnt[s[i]-'a']++;
		
		for ( int i = 0; i < 26; i++ ) {
			if ( cnt[i] >= 2 ) {
				printf("Yes\n");
				check=0;
				break;
			}
		}
		if(check)
			printf("No\n");
	}
	return 0;
}
