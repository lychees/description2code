#include<stdio.h>
#include<string.h>

int min(int a, int b){ return (a<b) ? a : b; }

int minCut(char *s) {
	int len = strlen(s);
	int dp[len]; 
	bool palidrome[len][len];
	for (int i = 0; i <= len; ++i)
		dp[i] = len - i - 1;
	for (int i = 0; i<len; ++i)
	{
		for (int j = 0; j<len; ++j)
		{
			if (i == j)
				palidrome[i][j] = true;
			else
				palidrome[i][j] = false;
		}
	}

	for (int i = len - 2; i >= 0; --i)
		for (int j = i; j<len; ++j)
			if ((s[i] == s[j]) && (j - i<2 || palidrome[i + 1][j - 1])) {
				palidrome[i][j] = true;
				dp[i] = min(dp[i], dp[j + 1] + 1);
			}
	int cnt = 0, i, j;

	for (i = 0; i<len; i++)
	{
		for (j = 0; j<len; j++)
		{
			if (palidrome[i][j] == true)
				cnt++;
		}
	}
	return cnt;
}

int main()
{
	char s[100];
	scanf("%s", &s);
	printf("%d\n", minCut(s));
}
