#include <bits/stdc++.h>

using namespace std;

int set_bit(int n)
{
	int ans = 0;
	while (n != 0) {
		ans += (n & 1);
		n >>= 1;
	}
	
	return ans;
}

int main()
{
	int t,n,i,j,ans;
	scanf("%d",&t);
	while (t --) {
		scanf("%d",&n);
		ans = set_bit(n);
		if (ans % 2 == 0) {
			printf("even\n");
		} else {
			printf("odd\n");
		}
	}
	
	return 0;
}
