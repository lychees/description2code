#include <bits/stdc++.h>
using namespace std;

#define MX 100005
typedef pair<long long, int> pli;

priority_queue< pli , vector<pli>, greater<pli> > q;

int c,p,n, x[MX], y[MX], v[MX][2], cou[MX];

long long calc(int id){
	return 1ll * cou[id] * (v[id][0]+v[id][1]);
}

void proc(int mid, int cid){
	cou[cid]++;
	v[cid][1] = v[cid][0]; v[cid][0] = x[mid];
	q.push(make_pair(calc(cid), cid));
}

int main(){
	cin>>c>>p>>n;
	for(int i=0; i<n; i++) {cin>>y[i]; v[i][0]=y[i]; cou[i]=1;}
	for(int i=0; i<p; i++) cin>>x[i];
	for(int i=0; i<c; i++) q.push(make_pair(calc(i), i));
	for(int i=0; i<p; i++){
		pli t = q.top(); q.pop();
		cout<<t.second+1<<" ";
		proc(i, t.second);
	}
}
