#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int res[500];
		for(int i=0;i<500;i++)
			res[i]=0;
		res[0]=1;
		int len=1;
		for(int i=1;i<=2*n-1;i+=2)
		{
			int q=0,r=0;
			for(int j=0;j<len;j++)
			{
				q=res[j]*i+r;
				res[j]=q%10;
				r=q/10;
 
			}
			while(r!=0)
			{
				res[len++]=r%10;
				r=r/10;
			}
 
 
		}
		for(int j=len-1;j>=0;j--)
			cout<<res[j];
		cout<<endl;
	}
 
	return 0;
}
