winner_list = set()
looser_list = set()
n = int( raw_input() )
for _ in range( (2**n)-1 ):
	d1, d2 = raw_input().split()

	if d2 in winner_list :
		winner_list.remove( d2 )

	looser_list.add( d2 )

	if d1 not in looser_list:
		winner_list.add( d1 )


for entry in (winner_list-looser_list):
	print entry
