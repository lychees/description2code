#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
 
 unsigned long long int Z,M,N;
 int t;
 cin>>t;
 while(t--){
 	scanf("%llu",&Z);
 	scanf("%llu",&M);
 	scanf("%llu",&N);
 	unsigned long long i=1;
 	    		unsigned long long count = 0;
        	for(i=1;i*i<Z;i++){
        		unsigned long long num1 = i;
        		if(Z%num1==0){
        			unsigned long long num2 = Z/num1;
        			if(num1<=M && num2<=N)
        			count++;
        			if(num2<=M && num1<=N)
        			count++;
        		}
        	}
        	if(i*i==Z){
        		if(i<=M && i<=N)
        			count++;
        	}
      cout<<count<<endl;
 }
    return 0;
}
