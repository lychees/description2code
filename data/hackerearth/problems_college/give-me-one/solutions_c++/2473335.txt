#include <stdio.h>
#include <stdlib.h>
#define MIN(a,b) ((a) < (b)) ? (a) : (b)
int main(void) {
        int num=20000005;
        int i;
        int *out;
        
        out = (int*)malloc(sizeof(int) * (num+1));
        out[1] = 0;
        for(i=2; i <= num; i++) {
                out[i] = out[i-1] + 1;
                if (i%2 == 0) {
                        out[i] = MIN(1 + out[i/2], out[i]);
                }
                if (i%3 == 0) {
                        out[i] = MIN(1 + out[i/3], out[i]);
                }
        }
        int t,n;
        scanf("%d",&t);while(t--){scanf("%d",&n);
        printf("%d\n", out[n]);}
        return 0;
}
