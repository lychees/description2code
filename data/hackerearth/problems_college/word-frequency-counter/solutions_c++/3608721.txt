#include <iostream>
using namespace std;

int main()
{
	pair<string,int> has[105];
	string str;
	int i,j=0;
	while(cin>>str) 
	{
		int t=str.length();
		for(i=0;i<t;i++) if(str[i]>=97) str[i]-=32;
		i=0;
		while(i<=j)
		{
			if(str==has[i].first) {has[i].second++;break;}
			i++;
		}
		if(i==j+1) {j++;has[j].first=str;
		has[j].second=1;}
		
	}
	for(i=1;i<=j;i++) cout<<has[i].first<<" "<<has[i].second<<endl;
    return 0;
}
