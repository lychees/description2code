#include <iostream>
#include <algorithm>
using namespace std;

typedef struct{
	int amount;
	char c;
}node;

bool myFunc(node i,node j){
	return i.amount<j.amount;
}

{

    	cin>>x;

    		alpha[i].c=(char)(97+i);
    		alpha[i].amount=0;
    	}
    	for(int i=0;i < x.size();i++){
    		alpha[x[i] - 97].amount++;
    	}
    	
    	for(int i=24;i>=0;i--){
    		node key=alpha[i];
    		int j = i+1;
    		//char k=alpha[i].c;
    		while(j <= 25 && key.amount < alpha[j].amount){
    			//cout<<"swapping "<<alpha[i].c<<" "<<alpha[j].c<<endl;
    			swap(alpha[j-1],alpha[j]);
    			j++;
    			//cout<<"Now "<<alpha[j].c<<" "<<key.c<<endl;

    	}

}
