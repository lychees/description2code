#include <iostream>
#include <math.h>

using namespace std;


int main()
{
	int t,r,xc,yc;
	int x1,y1,x2,y2,x3,y3,x4,y4,res;
	cin>>t;
	while(t!=0)
	{
		cin>>r>>xc>>yc;
		cin>>x1>>y1;
		cin>>x2>>y2;
		cin>>x3>>y3;
		cin>>x4>>y4;

		res=ceil(sqrt(((x1-xc)*(x1-xc))+((y1-yc)*(y1-yc))));
		if(res<=r)
		{
			res=ceil(sqrt(((x2-xc)*(x2-xc))+((y2-yc)*(y2-yc))));
			if(res<=r)
			{
				res=ceil(sqrt(((x3-xc)*(x3-xc))+((y3-yc)*(y3-yc))));
				if(res<=r)
				{
					res=ceil(sqrt(((x4-xc)*(x4-xc))+((y4-yc)*(y4-yc))));
					if(res<=r)
					{
						cout<<"Yes"<<endl;
					}
					else
					{
						cout<<"No"<<endl;
					}



				}
				else
				{
					cout<<"No"<<endl;
				}


			}
			else
			{
				cout<<"No"<<endl;
			}


		}
		else
		{
			cout<<"No"<<endl;
		}














		t--;
	}


	return 0;
}
