#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    
    long long int n;
    cin>>n;
    long long int arr[n];
    for(long long int i=0;i<n;i++)
    {
       cin>>arr[i];
    }
   long long  int sum[n+1];
    long long int max=0;
    for(int i=0;i<n+1;i++)
    {
    	sum[i]=0;
    }
    for(long long int i=0;i<=n;i++)
    {
    	for(long long int j=0;j<n;j++)
    	{
    		if(i!=j)
    		{
    			sum[i]+=arr[j];
    		}
    	}
    }
    
  
    
    for(long long int i=0;i<=n;i++)
    {
    	if(sum[i]%2!=0)
    	{
    		if(sum[i]>max)
    		  {
    		  max=sum[i];
    		  }
    	}
    }
    if(max==0)
    {
    	cout<<":(";
    }
    else
    cout<<max;
    
    return 0;
}
