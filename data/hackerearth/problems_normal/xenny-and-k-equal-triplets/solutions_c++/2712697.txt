long long int binomialCoeff(long long int n,int k)
{
    long long int res = 1;
 
    if ( k > n - k )
        k = n - k;

    for (long long int i = 0; i < k; ++i)
    {
        res *= (n - i);
        res /= (i + 1);
    }
 
    return res;
}

long long int gcd(long long int a,long long int b) {
    return b == 0 ? a : gcd(b, a % b);
}
