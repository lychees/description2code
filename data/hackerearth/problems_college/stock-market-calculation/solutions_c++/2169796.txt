#include <iostream>
#define MAX 100005
using namespace std;
typedef long long ll;
int a[MAX],b[MAX];
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
    int n;
    cin>>n;
    for(int i=0;i<n;i++)
    {
    	cin>>a[i];
    }
   b[n-1]=a[n-1];
  for(int i=n-2;i>=0;i--)
   {
  b[i]=max(b[i+1],a[i]);
   } 
   
   ll maxprofit=0;
   for(int i=0;i<n;i++)
   {
   if(b[i]>=a[i]) 	
   	maxprofit+=b[i]-a[i];
   }
   cout<<maxprofit<<endl;
}
return 0;
}
