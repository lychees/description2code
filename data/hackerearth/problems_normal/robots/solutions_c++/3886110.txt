#include <bits/stdc++.h>
using namespace std;

#define MOD 1000000007

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n=0;
        cin>>n;
        int a[n];
        for(int i=0;i<n;i++)cin>>a[i];
        sort(a,a+n);

        long long int count=1;
        for(int i=0;i<n;i++)
        {
            if((a[i]-i)==0)
            {
                count=0;
                break;
            }
            count=(count*(a[i]-i))%MOD;
        }

        cout<<count<<endl;
    }
    return 0;
}
