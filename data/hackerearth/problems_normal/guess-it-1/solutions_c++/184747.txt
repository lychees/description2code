void solve()
{
    string s;
    cin>>s;
    int ans = 0;
    for(int i=0;i<s.length();i++)
    {
    	if(s[i]=='_')continue;
        ans += s[i]-'0';
    }
    if(ans%9==0)
    {
        cout<<"0 or 9"<<endl;
    }
    else{
        ans = 9 - ans%9;
        cout<<ans<<endl;
    }
}

int main()
{
    int t;
    cin>>t;
    while(t--){
   solve();
    }
}
