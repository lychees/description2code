#include<cstdio>
using namespace std;
int mine[1000001];
int main()
{
    long long int n,k,l,u,i,j,r1,r2,sum;
    scanf("%lld%lld",&n,&k);
    for(i=0;i<n;i++)scanf("%lld",&mine[i]);
    while(k--)
    {
        scanf("%lld%lld%lld%lld",&l,&u,&r1,&r2);
        sum=0;
        for(i=r1-1;i<r2;i++)
        {
            if(mine[i]>=l&&mine[i]<=u)sum+=mine[i];
        }
        printf("%lld\n",sum);
    }

}
