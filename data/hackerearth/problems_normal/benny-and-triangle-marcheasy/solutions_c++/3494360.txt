#include "bits/stdc++.h"
using namespace std;
double a , b , p;
double area;
double solve(double val){
    double x = b - val;
    double y = a * (b - val) / b;
    return x * y / 2.0;
}
int main(){
    cin >> a >> b >> p;
    area = a * b / 2.0;
    double l = 0.0;
    double r = b;
    for(int i = 0 ; i < 100 ; ++i){
        double mid = l + r;
        mid /= 2.0;
        if(solve(mid) <= area * p / 100.0){
            r = mid;
        }
        else{
            l = mid;
        }
    }
    printf("%.2lf" , l);
}
