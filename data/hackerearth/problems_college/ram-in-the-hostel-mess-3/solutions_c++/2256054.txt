#include<bits/stdc++.h>
using namespace std;
long long int fibn[1005];
long long int pw(long long int p,long long int q)
{
    long long int x=1,y=p; 
    while(q > 0)
    {
        if(q%2 == 1)
        {
            x=(x*y);
        }
        y = (y*y);
        q /= 2;
    }
    return x;
}
void func()
{
	int i;
	fibn[0]=0;
	fibn[1]=2;
	fibn[2]=3;
	for(i=3;i<=1000;++i)
	fibn[i]=fibn[i-1]+fibn[i-2];
}
long long int fib( int n)
{
	return fibn[n];
}
int main()
{
	int i,t;
	scanf("%d",&t);
	func();
	while(t--)
	{
		long long int res;
		int n;
		scanf("%d",&n);
		res=fib(n);
		res=pw(res,2);
		printf("%lld\n",res);
	}
	return 0;
}
