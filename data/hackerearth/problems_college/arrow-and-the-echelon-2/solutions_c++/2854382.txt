#include <iostream>
using namespace std;

int main() {
	int t;
	cin>>t;
	while(t--){
		int x;
		cin>>x;
		int ct =0, cf = 0;
		while(x%3!=0 && x>=0)
		{
			x -= 5;
			ct++;
		}
		if(x<0)
		cout<<-1<<endl;
		else{
			cf = x/3;
			for(int i=0;i<cf*3;i++)
			cout<<5;
			for(int i=0;i<ct*5;i++)
			cout<<3;
			cout<<endl;
		}
	}
	return 0;
}
