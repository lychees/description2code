#include<iostream>
#include<malloc.h>
using namespace std;
bool a[10000][10000];
int main(){
    int n;
int x=1,y=1,i=0,c=0;
cin>>n;

int r[n];

 cin>>x;
 cin>>y;
while(x!=0 && y!=0){
a[x-1][y-1]=1;
cin>>x;
cin>>y;
}


r[0]=1;
for(int i=1;i<n;i++)
{
c=0;
for(int j=0;j<i;j++){
 if(a[j][i]==1)
        c=c+r[j];

}
r[i]=c;


}
    cout<<r[n-1]<<endl;




}
