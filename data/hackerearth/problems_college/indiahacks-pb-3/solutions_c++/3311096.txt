typedef long long       ll;

typedef pair<int, int>  ii;

ll dp[10005];
ll maxsum(vector<ll> &as, int index, ll n, ll k)
{
	if(index >= n) {
		return 0;
	}
	if(dp[index] != -1) {
		return dp[index];
	}	
	ll ans = INT_MIN;
	ans = max(ans, maxsum(as, index+1, n, k));
	ans = max(ans, as[index]+maxsum(as, index+k+1, n, k));
	return (dp[index] = ans);
}
int main()
{
	vector<ll> as;
	ll k,t,n,x;
	Sl(t);
	while(t--) {
		Sl(n);Sl(k);
		as.clear();
		for(int i = 0; i < n; i++) {
			Sl(x);
			as.pb(x);
		}
		memset(dp, -1, sizeof(dp));
		cout << maxsum(as, 0, as.size(), k) << endl;
	}
	return 0;
}
