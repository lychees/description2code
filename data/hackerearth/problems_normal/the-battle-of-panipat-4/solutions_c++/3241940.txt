#include <bits/stdc++.h>
#define MOD 1000000007
using namespace std;
vector<set<int > > adj;
bool visit[100005];
long func(int x, long& total);
int main(){
int n, x, y;
scanf("%d", &n);
adj.resize(n+1);
fill(visit, visit+n+1, false);
for(int i=1; i<n; i++){
	scanf("%d %d", &x, &y);
	adj[x].insert(y);
	adj[y].insert(x);
}
long total=0;
visit[0]=true;
func(0, total);
cout<<(total+1)<<endl;
}

long func(int x, long& total){
long c=1;
for(set<int >::iterator j=adj[x].begin(); j!=adj[x].end(); j++){
	if(!visit[*j]){
		visit[*j]=true;
		c = (c*(func(*j, total)+1))%MOD;
	}
}
total = (total + c)%MOD;
return c;
}
