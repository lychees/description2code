#include <iostream>
#include <cmath>
#include <cstdio>
#include <cstring>
#include <functional>
#include <algorithm>

using namespace std;

int main()
{
	int t;
	cin >> t;

	while(t--)
	{
		long n,m, key ;
		cin >> n ;
		cin >> m;
		key = n/2;


		if(m > key)
		{
			cout << 2 * ((n-1)-m) << endl;
		}
		else 
		{
			if( m < key)
			{
				cout << (2*m)+1 << endl;
			}
			else
			{
				//m==key
				if(n% 2 == 0) { cout << (n-1)-1 << endl; }
				else { cout << (n-1) << endl; }
			}
		}	
	}
}
