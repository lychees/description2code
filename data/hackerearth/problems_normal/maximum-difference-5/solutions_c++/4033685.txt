#include<bits/stdc++.h>

using namespace std;
const int MAXN = 1e5+5;
int ar[MAXN];
int main() {
	// freopen("TASK.in","r",stdin);	
	// freopen("TASK.out","w",stdout);
	int n;
	cin>>n;
	for(int i=1;i<=n;i++) scanf("%d",&ar[i]);
	sort(ar+1,ar+1+n);
	int p1=1,p2=n;
	long long ans1=0;
	int counter=0;
	while(p1<p2) {
		++counter;
		ans1+=1LL*(abs(ar[p1]-ar[p2]));
		if(counter&1) p1++;
		else p2--;
	}
	p1=1,p2=n;
	long long ans2=0;
	counter=0;
	while(p1<p2) {
		++counter;
		ans2+=1LL*(abs(ar[p1]-ar[p2]));
		if(counter&1) p2--;
		else p1++;
	}
	cout<<max(ans1,ans2)<<endl;
	return 0;
}
