#define  _CRT_SECURE_NO_WARNINGS
#include <bits/stdc++.h>
#define EPS 1e-9
#define INF 1000000000
#define ll long long 
using namespace std;
int n;
string s;
ll start, s1, s2;
bool vis[(1 << 20) + 5];
int main()
{

	cin >> n >> s;
	for (int i = 0; i < n; ++i)
		if (s[i] == 'B')
			start ^= (1LL << i);
	queue<pair<int, int> > q;
	q.push({ start,0 });
	s1 = 0; s2 = (1LL << n) - 1;
	while (!q.empty())
	{
		auto front = q.front();
		q.pop();
		if (front.first == s1 || front.first== s2)
		{
			cout << front.second << endl;
			return 0;
		}
		if (vis[front.first])continue;
		vis[front.first] = true;
		ll curMask, mask = front.first;
		for (int i = 0; i < n - 1; ++i)
		{
			curMask = (mask ^ (1LL << i) ^ (1LL << (i + 1)));
		//	cout << bitset<7>(curMask) << " ";
			if (!vis[curMask])q.push({ curMask ,front.second + 1 });
		}
		cout << endl;
		for (int i = 0; i < n - 2; ++i)
		{
			curMask = (mask ^ (1LL << i) ^ (1LL << (i + 1)) ^ (1LL << (i + 2)));
			if (!vis[curMask])q.push({ curMask , front.second + 1 });
	
		}
	}
	//system("pause");
}
