#include<iostream>
#include<cmath>
using namespace std;
long long int fn(long long int num)
	{
    long long int ct=0, n;
    while(num > 0)
		{
        n=0;
        while(num >= 1<<(n+1))
            n++;
        num -= 1<<n;
        ct += (num + 1 + (1<<(n-1))*n);
    }
    return ct;
}
int main()
{
	   long long int t,a,b;
	   cin>>t;
	   while(t--)
	   {
   		cin>>a>>b;
   		cout<<fn(b)-fn(a-1)<<endl;
   	}
   	return 0;
	   
}
