#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    if(n<0)
     cout<<"ERROR"<<endl;
     else{
     int jack[n+1],jill[n+1];
     jack[0]=0;
     jack[1]=1;
     jack[2]=2;
     jack[3]=4;
     jill[0]=0;
     jill[1]=1;
     jill[2]=2;
      for(int i=4;i<=n;i++)
      {
      	 jack[i]=jack[i-1]+jack[i-2]+jack[i-3];
      	 
      }
    for(int i=3;i<=n;i++)
    {
    	jill[i]=jill[i-1]+jill[i-2];
    }
    
    cout<<"Jack-"<<jack[n]<<endl;
    cout<<"Jill-"<<jill[n]<<endl;}
    return 0;
}
