#include <iostream>
using namespace std;

int main()
{
    int T,N,A[100000],i,j,ans,num1,num0;
    cin>>T;
    for(i=0;i<T;i++)
    {
    	cin>>N;
    	num1=0;
    	num0=0;
    	for(j=0;j<N;j++)
    	{
    		cin>>A[j];
    		if(A[j]==1)
    		{
    			num1++;
    		}
    		else
    		num0++;
    	}
    	if(num1%2!=0)
    	{
    		ans = num1;
    	}
    	else
    	ans = num0;
    	
    	cout<<ans<<endl;
    }
    return 0;
}
