#include <bits/stdc++.h>

using namespace std;

#define LL long long int 
#define sd(x) scanf("%d", &x)
#define F first
#define S second
#define mp make_pair
#define pb push_back
#define N 2000000
#define pii pair<int,int>

int h[N];

int main(){
	int t, n, i, ans;
	sd(t);
	while(t--){
		sd(n);
		for(i = 1; i <= n; i++){
			sd(h[i]);
		}
		h[0] = 0;
		ans = -10000;
		for(i = 1; i <= n; i++){
			if(h[i - 1] > 0){
				h[i] += h[i - 1];
			}
			if(h[i] > ans){
				ans = h[i];
			}
		}
		cout<<ans<<endl;
	}
	return 0;
}
