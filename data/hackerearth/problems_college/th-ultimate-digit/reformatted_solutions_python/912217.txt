def main():
    number_of_tc = input()
    s = list()
    res = list()
    for e in xrange(number_of_tc):
        elements = raw_input()
        s.append(elements.split())
    for a, b in s:
        a = int(a) % 10
        b = int(b)
        res.append((a**b) % 10)
    for each in res:
        print each


if __name__ == '__main__':
    main()
