#include <bits/stdc++.h>
using namespace std;
int a[100000];
int main()
{
    int t,n,k;
    cin>>t;
    for(int k1=0;k1<t;k1++)
    {	
    cin>>n>>k;
    long long sum=0,temp=0;
    for(int i=0;i<n;i++)
    {
    cin>>a[i];
    if(a[i]<=k)
    {temp+=a[i];
    sum=max(sum,temp);}
    else
    temp=0;
    }
    cout<<sum<<"\n";
    }
    return 0;
}
