ll res[MAX], pr[MAX];
void sieve()
{
    pr[0] = pr[1] = 1;
    for (int i = 2; i <= sqrt(500); i++)
    {
        if (!pr[i])
        {
            for (int j = i * i; j <= 500; j += i)
                pr[j] = 1;
        }
    }
}
ll powr(ll a , ll b)
{
    if (b == 0)
        return 1;
    ll x = powr(a, b / 2);
    if (b % 2 == 1)
        return (a * x * x);
    else
        return x * x;
}
