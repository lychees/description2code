#include <iostream>
using namespace std;
int getPower(int n){
	int sum = 1;
	for(int i = 0;i<n;i++){
		sum*=(5);
	}
	return sum;
}
void compute(int n){
	int index = 1;
	int answer = 0;
	while(n/(getPower(index)) >0){
		answer+=(n/(getPower(index)));
		index++;
	}
	cout<<answer<<endl;
}
int main()
{
    int t,n;
    cin>>t;
    while(t--){
    	cin>>n;
    	compute(n);
    }
    return 0;
}
