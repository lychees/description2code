#include<iostream>
#include<vector>
#include<algorithm>
#include<cstring>
#define mx 1000000
using namespace std;
long int prime[1000001],num[1000001],k;
void pre()
{
    long int i,j,t;
    prime[0]=prime[1]=1;
    for(i=2;i<1000001;i++)
        {
        	if(prime[i]==0)
        	{
        		t=1;
        	for(j=i*2;j<1000001;j+=i)
        {
        	if(prime[j]==0) t++;
        	prime[j]=1;
        }
        num[i]=t;
       }
        }
        
}
void sol()
{
    long int x;
    cin>>x;
    cout<<num[x]<<endl;
}
int main()
{
    int t;
    pre();
    cin>>t;
    while(t--) sol();
    return 0;
}
