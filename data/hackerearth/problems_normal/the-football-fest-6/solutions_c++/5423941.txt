#include<bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    int n,id;
    cin>>n>>id;
    stack<int> player;
    player.push(id);
    while(n--)
    {
    	char c;
    	int id1;
    	cin>>c;
    	if(c=='P')
    	{   cin>>id1;
    		player.push(id1);
    	//	cout<<player.top()<<endl;
    	}
    	else if(c=='B')
    	     {
    	     	if(!player.empty())
    	     		{
    	     			int first=player.top();
    	     			player.pop();
    	     		//	cout<<first<<" ";
    	     			int second=player.top();
    	     		//	cout<<second<<" ";
    	     			player.pop();
    	     			player.push(first);
    	     			player.push(second);
    	     		}
    	     }
    }
    int ans=player.top();
    cout<<"Player "<<ans<<endl;
    }
    return 0;
}
