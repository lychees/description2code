from bisect import bisect_left


def make_array(N, A):
    B, C = [0] * N, [0] * N
    B[0], C[0] = 1, A[0]
    for i in xrange(1, N):
        B[i] = C[i - 1] + 1
        C[i] = B[i] + A[i] - 1
    return B, C


if __name__ == '__main__':
    N = input()
    A = map(int, raw_input().split())
    B, C = make_array(N, A)

    for _ in xrange(input()):
        I = input()
        li = bisect_left(C, I)
        print li + 1
