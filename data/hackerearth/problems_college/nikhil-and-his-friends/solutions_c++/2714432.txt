#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,s;
    	cin>>n>>s;
    	
    	int a[n];
    	
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	int cur_sum=0,f=0,j=0;
    	for(int i=0;i<n;i++)
    	{
    		
    		cur_sum+=a[i];
    		while( cur_sum>s && j<i ) cur_sum-=a[j++];
    		
    		if(cur_sum==s){ f=1; break; }
    		
    		
    		
    	}
    	if(f) cout<<"YES"<<endl;
    	else cout<<"NO"<<endl;
    	
    }
    return 0;
}
