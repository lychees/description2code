#include <iostream>
#include <stack>
#include <bitset>
#include <cmath>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>
#include <map>
#include <climits>

#define MOD 4294967296
#define pb push_back
#define mp make_pair

typedef long long int ll;

using namespace std;

int main()
{
    ios::sync_with_stdio(0);
    ll n,x,i,cans,k,ans,j,loc;
    cin>>n;
    vector <ll> v1,v2;
    for(i=0;i<n;i++) {
        cin>>x;
        v1.pb(x);
    }
    for(i=0;i<n;i++) {
        cin>>x;
        v2.pb(x);
    }
    sort(v1.rbegin(),v1.rend());
    sort(v2.rbegin(),v2.rend());
    ans = 0;
    for(i=0;i<n;i++) {
        j=i;
        cans = 0;
        for(k=0;k<n;k++) {
            if(v1[k] > v2[j]) {
                cans+=100;
            } else if(v1[k] == v2[j]) {
                cans+=50;
            }
            j=(j+1)%n;
        }
        ans = max(ans,cans);
    }
    cout<<ans<<endl;
    return 0;
}
