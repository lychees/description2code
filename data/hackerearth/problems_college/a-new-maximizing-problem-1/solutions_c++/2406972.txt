#include <bits/stdc++.h>
using namespace std;
#define forall(i,a,b)               for(int i=a;i<b;i++)
long long wi[200000];

int main()
{
    long long test,size;
    cin >>test;
    while(test--)
    {
       cin >> size;
       forall(i,0,size)cin >> wi[i];
       long long b=0,a=0;
       forall(i,0,size){if(wi[i]<0)b-=wi[i];else b+=wi[i];}
       sort(wi,wi+size);
       if(size%2==1)
            { forall(i,size/2-1,size)a+=wi[i];
              forall(i,0,size/2-1)a-=wi[i];}
        else 
        { forall(i,size/2,size)a+=wi[i];
              forall(i,0,size/2)a-=wi[i];}
       
       if(a<0)a=-a;
       long long c=((a%1000000007)*(b%1000000007))%1000000007;
       cout <<  c<<"\n" ;
    }
}
