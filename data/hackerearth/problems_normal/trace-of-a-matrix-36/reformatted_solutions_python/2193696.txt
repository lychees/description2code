tc = int(raw_input())


def solve():
    R, C = [int(i) for i in raw_input().split()]

    n = min(R, C) - 1

    ans = 2*(n+1) + (R+C+2)*(n*(n+1))/2

    return ans


for zz in range(tc):
    print(solve())
