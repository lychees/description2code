#include<iostream>
#include<cstring>
#include<map>
#include<cstdio>
using namespace std;
int main()
{
    int i,n,m;
    long int maxi=0,points;
    char name[16];
    char str[16];
    map<string,long int> s;
    scanf("%d%d",&n,&m);
    for(i=0;i<m;i++)
    {
       scanf("%s",str);
        cin>>points;
        s[str]+=points;
        if(s[str]>maxi)
        {
            maxi=s[str];
            strcpy(name,str);
        }
    }
    cout<<name;
    return 0;
}
