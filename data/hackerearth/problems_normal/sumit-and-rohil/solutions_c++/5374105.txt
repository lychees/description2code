#include <iostream>
#include<bits/stdc++.h>
using namespace std;

map<string,int> m;


int main()
{
    int n;
    cin>>n;
    vector<string> temp(n);
    int i=0;
    for(i=0;i<n;i++){
    	cin>>temp[i];
    	sort(temp[i].begin()+1,temp[i].end()-1);
    	m[temp[i]]++;
    	}
    	cout<<m.size();
    return 0;
}
