#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        int a[n];
        for(int i=0;i<n;i++)
        {
            cin>>a[i];
        }
        int flag=0;
        int i=0;
        int count=2*n;
        while(count)
        {
            if(a[i]==-1)
            {
                flag=1;
                break;
            }
            else{
                    i=a[i];
            }
            count--;
        }
        if(flag==1)
            printf("Win\n");
        else
            printf("Lose\n");
    }
    return 0;
}
