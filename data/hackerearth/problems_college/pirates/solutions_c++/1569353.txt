#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    int N,i=0,c=0,j=0,m;
    cin>>N;
    int A[N],t[N];
    for(i=0;i<N;i++)
    {
    	cin>>A[i];
    }
    for(i=1;i<N;i++)
    {
    	if(fabs(A[i]-A[i-1])==1) c++;
    	else {
    		t[j]=c+1;
             j++;
               c=0;
    	}
    }
    t[j]=c+1;
j++;
    m=t[0];
    for(i=1;i<j;i++)
    {
    	if(t[i]>m)
    	m=t[i];
    }
    cout<<m;
}
