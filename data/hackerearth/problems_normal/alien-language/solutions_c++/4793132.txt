#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin >>t;
    while(t--){
    	string txt,pat;
    	cin >>txt;
    	cin >>pat;
    	int f[26]={0};
   	for(int i=0;i<txt.length();i++){
   		f[txt[i]-'a']++;
   	}
   		int p[26]={0};
   	for(int i=0;i<pat.length();i++){
   		p[pat[i]-'a']++;
   	}
   	int k=0;
   	for(int i=0;i<26;i++){
   		if(f[i] && p[i]){
   			cout << "YES" << endl;
   			k=1;
   			break;
   		}
   		
   	}
   	if(k==0)
   	cout <<"NO" << endl;
 
    }
    return 0;
}
