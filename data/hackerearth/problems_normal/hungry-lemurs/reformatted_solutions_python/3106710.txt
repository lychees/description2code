n, k = map(int, raw_input().split())
if n % k == 0:
    print 0
elif k > n:
    print min(n, k-n)
else:
    c = k-1
    for i in range(2, n+1):
        temp = abs(i-k)
        prev = n-(n/i)*i
        nxt = (n/i+1)*i-n
        # print temp,prev,nxt
        temp += min(nxt, prev)
        c = min(c, temp)
    print c
