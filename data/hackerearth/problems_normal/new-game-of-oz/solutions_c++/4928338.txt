#include<iostream>
#include<algorithm>
using namespace std;

int main()
{
	int t, n, *a, i, c=0;
	cin>>t;
	while(t--)
	{
		cin>>n;
		a=new int[n];
		for(i=0;i<n;i++)
			cin>>a[i];
		sort(a,a+n);
		for(i=0;i<n;i++)
		{
			if(a[i+1]==a[i]+1)
				i++;
			c++;
		}
		cout<<c<<endl;
		c=0;
	}
	return 0;
}
