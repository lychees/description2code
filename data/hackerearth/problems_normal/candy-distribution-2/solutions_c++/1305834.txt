#include <iostream>
#include <vector>
#include <limits.h>
#include <algorithm>
using namespace std;

typedef unsigned long long ull;

int main()
{
    int t;
    cin >> t;
    while(t--) {
    	int n;
    	cin >> n;
    	if(n < 1) continue;
    	vector<ull> arr(n);
    	ull sum = 0;
    	for(auto &i : arr) {    		
    		cin >> i;
    		sum += i;
    	}
        cout << sum - (n - 1) << endl;
    }
    return 0;
}
