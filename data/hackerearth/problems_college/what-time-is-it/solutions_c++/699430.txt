#include <iostream>
#include<iomanip>
using namespace std;

int main()
{
 char ch;
  int i, hrs, min;
  float hangle,mangle;
  float angle;

  cin >> hrs;
  cin >> ch;
  cin >> min;
  if( hrs == 12 && min == 60)
   {
   hrs = 1;
   min = 0;
   }

  else if( min == 60)
  {
   min = 0;
   hrs++;
  }
  else if(hrs == 12)
   hrs  = 0;

  mangle = min*6;

  hangle = (hrs*30) + (mangle/12);

  if(hangle > mangle)
   angle = hangle - mangle;

   else if( hangle < mangle)
   angle = mangle - hangle;

  if(angle > 180)
   angle-= 180;
   
    cout << setprecision(6) << fixed;
    cout << angle;
    return 0;
}
