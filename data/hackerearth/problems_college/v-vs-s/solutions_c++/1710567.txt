#include <iostream>
#include <cstring>
#include <cmath>
#include <cstdio>
using namespace std;
int main()
{
   // cout << "Hello World!" << endl;
   long long int t,n1,ans,i,len;
   scanf("%lld\n",&t);
   while(t--)
   {
   ans=0; n1=0;
   	char a[1002];
   	scanf("%s",a);
   	len=strlen(a);
   	ans=a[len-1]-48;
   	for(i=len-2;i>=0;i--)
   	{
   		if(i==len-2) ans+=(a[i]-48)*3;
   		ans+=(a[i]-48)*9;
   			while(ans/10 !=0)
	{
	while(ans!=0)
	{
	n1+=ans%10;
	ans/=10;
	}
	ans=n1;
	n1=0;
	}
   	}
   	cout<<ans<<endl;
   }
}
