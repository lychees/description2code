#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int a[1000],b[1000];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int test;
	cin>>test;
	while(test--)
	{
		ll int n;
		cin>>n;
		ll int i,j;
		for(i=0;i<n;i++)
		{
			cin>>a[i];
		}
		sort(a,a+n);
		j=0;
		for(i=0;i<n/2;i++)
		{
			b[j++]=a[i]+a[n-1-i];
		}
		sort(b,b+n/2);
		cout<<(b[n/2-1]-b[0])<<endl;
	}
	return 0;
}
