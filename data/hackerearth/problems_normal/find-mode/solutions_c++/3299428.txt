#include<bits/stdc++.h>
using namespace std;
long long int arr[1000005];

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		memset(arr,0,sizeof(arr));
		int n;
		cin>>n;
	//	int a[n];
		long long l = -1,max = -1;
		for(int i = 0; i <n; i++)
		{	
			long long int k; 
			cin>>k;
			arr[k]++;
			if(k > l)
			l = k;
			if(arr[k] > max)
			max = arr[k];
		}
		for(int i = l; i>=0; i--)
		{
			if(arr[i]==max)
			cout<<i<<" ";
		}
		cout<<endl;
	}
	return 0;
}
