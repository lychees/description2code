#include <bits/stdc++.h>

using namespace std;

map<string, int> cnt;

int main() {
    string S;
    cin >> S;
    for (int i = 0; i < S.length(); i++) {
    	for (int j = 0; j < S.length()-i; j++) {
    		cnt[S.substr(i, j+1)]++;
    	}
    }
    
    cout << cnt.size() << endl;
    
    return 0;
}
