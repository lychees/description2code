#include<iostream>

using namespace std;

int main()
{
    int T;
    cin >> T;
    for (int k = 0; k < T; k++)
    {
        int N;
        int a[10] = {0};

        cin >> N;

        int cnt = 0;
        for (int i = 0; i < N; i++)
        {
            int temp;
            cin >> temp;
            int rem = temp%10;
            if (a[rem] == 0)
                a[rem] = 1;
            else
                cnt++;
        }
        cout << cnt << endl;
    }
    return 0;
}
