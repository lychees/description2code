#include<iostream>
using namespace std;
int main()
{
    int t,n,i;
    long int a,sum;
    cin>>t;
    while(t--)
    {
        sum=0;
        cin>>n;
        for(i=0;i<n;i++)
        {
            scanf("%ld",&a);
            sum+=a;
        }
        printf("%ld\n",sum/n);
    }
    return 0;
}
