#include <iostream>
using namespace std;

	int main()
	{
		int t;
		cin>>t;
		string s,dum;
		for(int i=0;i<t;i++)
		{
			s.clear();
			dum.clear();
			cin>>s>>dum;
			int count=0;
			int temp=0;
			if(s.size()>dum.size())
				temp=dum.size();
			else
				temp=s.size();
				for(int j=0;j<temp;j++)
				{
					if(s[j]==dum[j])
						count++;
				}
				cout<<count<<endl;
		}
}
