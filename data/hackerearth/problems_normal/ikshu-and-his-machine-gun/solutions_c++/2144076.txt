#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
int main()
{
    ios::sync_with_stdio(false);
    long long int l,S=0,x,i;
    vector<long long int> st;
    cin>>l;
    while(l--){
    cin>>x;
    st.push_back(x);
    }
    sort(st.begin(),st.end());
    for(i=0;i<st.size();++i){
            cout<<st[i]-S<<" ";
    ++S;

    }

}
