#include <iostream>
using namespace std;

int  main()
{
    long long  t;
    cin>>t;
    for(long long  z=0;z<t;z++)
    {
    	long long  n;
    	cin>>n;
    	long long  a[n];
    	for(long long  i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	long long  an[n];
    	an[0]=a[0];
    	an[1]=max(a[0],a[1]);
    	for(long long  i=2;i<n;i++)
    	{
    		an[i]=max(an[i-2]+a[i],an[i-1]);
    	}
    	cout<<"Case "<<z+1<<": "<<an[n-1]<<endl;
    }
    return 0;
}
