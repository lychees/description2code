#include <iostream>
using namespace std;

int main()
{
    int test;
    cin>>test;
    while(test>0)
    {
    	int num;
    	cin>>num;
    	int count=0;
    	for(int i=2;i<=num;i++)
    		for(int j=i+1;j<=num;j++)
    			if((i*j)%(i+j)==0 )
    				{
    				
    					count++;
    				}
    	cout<<count<<"\n";
    	test--;
    }
    return 0;
}
