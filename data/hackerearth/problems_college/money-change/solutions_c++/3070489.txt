#include <iostream>
#include <string.h>
#define ll long long
using namespace std;
ll dp[100][10000], change[100], n, m;
ll solve(ll i, ll s)
{
    ll ans=0;
    if(i==m)
    {
        if(s==n) return 1;
        else return 0;
    }
    if(dp[i][s]!=-1) return dp[i][s];
    else
    {
        if((s+change[i])<=n)
            ans=solve(i,s+change[i]);
        ans+=solve(i+1,s);
        dp[i][s]=ans;
    }
    return ans;
}
int main()
{
    cin>>n>>m;
    for(ll i=0;i<m;i++)
        cin>>change[i];
    memset(dp,-1,sizeof(dp));
    ll ans=solve(0,0);
    cout<<ans<<endl;
    return 0;
}
