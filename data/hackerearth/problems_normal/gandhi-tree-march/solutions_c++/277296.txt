int tc_poss=0;
std::string testcase;
bool comma=false;
char next(){
    
    if(tc_poss == testcase.size())return -1;
    else{
        char c = testcase[tc_poss];
        char c2 = tc_poss<testcase.size()?testcase[tc_poss+1]:-1;
        ++tc_poss;
        if(c!='(' and c!=')'){ //character
            if(comma){
                comma=false;
                --tc_poss;
                return ',';
            }else { 
                comma = (c2!='(' and c2!=')'); //c1 is character
                return c;
            }
        }else { //special character
            if(c==')'){
                comma = (c2!='(' and c2!=')'); //c1 is character
            }
            return c;
        }
    }
}

void solve(int line){
    std::string result;
    int column = 0;
    char current=0;
    while((current=next()) != -1){ 
        if(current == '(')
            --column;
        if(current == ',')
            column+=2;
        if(current == ')')
            --column;
        
        if(line == column and current !='.' and 
              current != '(' and current != ')'
                and current != ','){
           result.push_back(current);
        }
        //cout << current;
    }
    //cout << endl;

    if(result.empty()){
        cout << "Common Gandhijee!" << endl;
    } else {
        std::sort(result.begin(), result.end());
        cout << result << endl;
    } 

}

int main(int argc, char** argv) {
    
    unsigned int t=0;
    
    cin >> t;
    int column_number=0;
    for(int i=0;i<t;++i){
        int line;
        cin >> line;
        testcase.clear();
        tc_poss=0;
        comma=false;
        cin >>testcase; //read the whole line
        solve(line);
    }
    return 0;
}
