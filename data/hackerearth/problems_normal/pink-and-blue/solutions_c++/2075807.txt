#include<bits/stdc++.h>
using namespace std;
int main(){
	int n,m; cin>>n>>m;
	int a[n+1];
	for(int i=1; i<=n; i++){
		char x; cin>>x;
		a[i]=0;
		if(x=='G') a[i]++;
	}  
	vector<int> v[n+1];
	while(m--){
		int x,y; cin>>x>>y;
		v[x].push_back(y); v[y].push_back(x);
	}
	int visited[n+1]; memset(visited,0,sizeof visited);
	//girl=1, boy=0
	queue<pair<int,int> >q;
	if(a[1]==0) q.push(make_pair(1,0));
	else q.push(make_pair(1,1));
	visited[n+1]; memset(visited,0,sizeof visited);
	visited[1]=1;
	int ans=0;
	while(!q.empty()){
		int node=q.front().first; int col=q.front().second;
		q.pop();
		if(col!=a[node]) ans++;
		for(int i=0; i<v[node].size(); i++){
			if(!visited[v[node][i]]){
				visited[v[node][i]]=1;
				q.push(make_pair(v[node][i],col^1));
			}
		}
	}
	cout<<min(ans,n-ans)<<endl;
}
