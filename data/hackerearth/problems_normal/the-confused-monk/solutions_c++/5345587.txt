#include<iostream>
#include<stdio.h>
using namespace std;
#define m 1000000007
long long int GCD(long long int A, long long int B) {
    if(B==0)
        return A;
    else
        return GCD(B, A % B);
}
long long int modRecursion(long long int a, long long int b, long long int c)
{
    if(b == 0)
    return 1;
    if(b == 1)
    return a%c;
    else if( b%2 == 0)                                           //if b is even
    {
        return modRecursion((a *a)%c,b/2,c);
    }
    else                                                                    // if b is odd.
    {
        return (a*modRecursion((a*a%c),b/2,c))%c;
    }
}

int main()
{ int n;
  cin>>n;
  long long int a[n];
  int i;
  for(i=0;i<n;i++)
    cin>>a[i];
  long long int val=a[0];
  long long int mul=a[0];
  for(i=1;i<n;i++)
   {  val=GCD(val,a[i]);
      mul=((mul%m)*(a[i]%m))%m;
   }
  long long int f=modRecursion(mul,val,m);
  cout<<f;
  return 0;
}
