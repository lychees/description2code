#include<iostream>
#include<stdlib.h>
#include<string.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int count=0;
		long long int n;
		cin>>n;
		while(n!=1)
		{
			if(n%3==0)
				n/=3;
			else if(n%2==0)
				n/=2;
			else
				n-=1;
				
			count++;
		}
		cout<<count<<endl;
	}



}
