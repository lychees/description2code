#include <iostream>
#include <stdio.h>
#include <vector>

using namespace std;

int main() {
	
	int t, i, n, j, flag;

	vector<bool> A( 1000, true );
	cin >> t;
	A[ 0 ]  = A[ 1 ] = false;

	for( i = 2;i * i< 1000;i++ )
		for( j = i * i;j < 1000;j += i )
			if( A[ j ] )
				A[ j ] = false;

	while( t-- ) {

		string s, str = "";

		cin >> s;
		n  = s.length();

		flag = 0;
		for( i = 0;i < n;i++ ) {
			if( !A[ s[ i ] ] )
				str +=  s[ i ];
			else
				flag = 1;
		}
		
		if( flag == 0 )
			cout << "NULL";
		else {
			if( str == ""){}
			else
				cout << str;
		}
		cout << endl;

	}	
	return 0;
}
