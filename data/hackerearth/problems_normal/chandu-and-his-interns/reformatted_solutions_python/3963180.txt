N = int(raw_input())


def divisors(n):
    if n == 1:
        return "NO"
    i = 2
    count = 2
    while i*i <= n:
        if n % i == 0:
            if i*i == n:
                count += 1
            else:
                count += 2
        if count >= 4:
            return "YES"
        i += 1
    if count < 4:
        return "NO"


for i in range(N):
    x = int(raw_input())
    print divisors(x)
