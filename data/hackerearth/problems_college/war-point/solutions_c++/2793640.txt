#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		long long arr[n+10],arr2[n+10],dp[n+10];
		memset(dp,0,sizeof(dp));
		memset(arr,0,sizeof(arr));
		memset(arr2,0,sizeof(arr2));
		for(int i=1;i<=n;i++)
		{
			cin>>arr[i];
			dp[i]=dp[i-1]+arr[i];
		}
		for(int i=n;i>=1;i--)
		arr2[i]=arr2[i+1]+arr[i];
		long long ans=INT_MAX;
		for(int i=1;i<=n;i++)
		{
			long long t1;
			t1=abs(dp[i]-arr2[i+1]);
			ans=min(ans,t1);
		}
		cout<<ans<<endl;
	}
	
	return 0;
}
