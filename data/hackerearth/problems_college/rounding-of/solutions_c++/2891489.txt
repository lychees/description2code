#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
   int a[t][2];
    for(int i=0;i<t;i++){
    	cin>>a[i][0]>>a[i][1];
    	if(a[i][1]%2==0){
    	if((a[i][0]%a[i][1])<((a[i][1])/2)){
    		a[i][0]-=(a[i][0]%a[i][1]);
    	}
    	else
    	a[i][0]+=(a[i][1]-a[i][0]%a[i][1]);
    }else{
    	if((a[i][0]%a[i][1])<=((a[i][1])/2)){
    		a[i][0]-=(a[i][0]%a[i][1]);
    	}
    	else
    	a[i][0]+=(a[i][1]-a[i][0]%a[i][1]);
    }}
    
    for(int i=0;i<t;i++)cout<<a[i][0]<<endl;
    return 0;
}
