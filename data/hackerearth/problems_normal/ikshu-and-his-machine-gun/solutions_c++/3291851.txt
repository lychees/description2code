#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long int s;
    cin>>s;
    vector<int> val(s);
    for(int i=0;i<s;i++)
    cin>>val[i];
    int sub=0;
    sort(val.begin(),val.end());
    for(int i=0;i<s;i++)
    {
    	val[i]-=sub;
    	cout<<val[i]<<" ";
    	sub+=1;
    }
    return 0;
}
