#include <iostream>
using namespace std;
#define M 1000000007
#define ll long long
#define mod 1000000007
ll modexp(ll a, ll b)
{
		if(b==1) return a;
		ll res=1;
		while(b)
		{
				if(b%2==1) res=((res%mod)*(a%mod))%mod;
				a=((a%mod)*(a%mod))%mod;
				b/=2;
		}
		return (res%mod);
}
int main()
{
	long long n,t,min,max,h;
    cin >> t;
    while(t--){
    	cin >> n;
    		min = ((((n-1)/2)*((n-1)/2))%M)*n;
    		min = min %M;
    		max = (((n*(n-1))%M)*(2*n - 1))%M;
    		max = (max * modexp(6,M-2))%M;
    	cout << min << " " << max << endl;
    }
}
