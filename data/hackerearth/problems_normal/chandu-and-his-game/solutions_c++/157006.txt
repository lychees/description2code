#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <math.h>
using namespace std;
long n,k,i,x,y,t,cem=0,a,b,c,d;
int main(int argc, char *argv[]) 
{   cin>>n>>k;
    for(i=1;i<=k;i++)
{   cin>>x>>y;
    a=x-1;
    b=y-1;
    c=n-x;
    d=n-y;
    if(a<=c)
    cem=cem+a;
    if(a>c)
    cem=cem+c;
    if(b>=d)
    cem=cem+d;
    if(b<d)
    cem=cem+b;
}   cout<<cem<<endl;
	return 0;
}
