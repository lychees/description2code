#include<bits/stdc++.h>
using namespace std;
#define MOD 1000000007
long long power(long long n, long long m)
{
    if(m==0) return 1;
    long long x=power(n,m/2);
    if(m%2==0) return (x*x)%MOD;
    else return (((x*x)%MOD)*n)%MOD;
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long n;
		scanf("%lld",&n);
		long long ans2=power(10,n);
		long long ans1=power(5,n);
		ans2=power(ans2,MOD-2);
		ans1=(ans1*ans2)%MOD;
		ans1=(2*ans1)%MOD;
		printf("%lld\n",ans1);
	}
	return 0;
}
