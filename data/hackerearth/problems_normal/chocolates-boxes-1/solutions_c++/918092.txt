#include<bits/stdc++.h>
using namespace std;
#define MAX 100005
int arr[MAX];
int mul[MAX];
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,i;
        scanf("%d",&n);
        for(i=0;i<n;i++){
            scanf("%d",&arr[i]);
        }
        int k;
        scanf("%d",&k);
        for(int i=0;i<k;i++){
            mul[i]=0;
        }
        long long int ways=0;
        for(int i=0;i<n;i++)
        {
            if(arr[i]<k)
                mul[arr[i]%k]++;
        }
        for(int i=1;i<=(k/2);i++)
        {
            if(2*i == k && mul[i]>=2)
                ways+=(mul[i]*(mul[i]-1)/2);
            else
                ways+=mul[i]*mul[k-i];
        }
        printf("%lld\n",ways);
    }
    return 0;
}
