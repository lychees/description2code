#include <bits/stdc++.h>
using namespace std;

int main()
{
	int T;
	scanf("%d",&T);
	while(T--)
	{	
		string s;
		cin>>s;
		
		// Used pair vectors to bind frequency & corresponding alphabet because sorting frequencies will change the arrangement making it impossible to locate alphabet. 
		
		vector<pair<int, int> > V(26);    
		
		/* 
		   first is to count frequency
		   second is to note the alphabet index
		*/   
		
		for(int i=0;i<26;i++)
		{
			V[i].first=0;
			V[i].second=i;
		}
		for(int i=0;i<s.size();i++)
			V[s[i]-97].first++;
		sort(V.begin(),V.end());
		for(int i=0;i<26;i++)
		{
			int count=V[i].first;
			char ch=V[i].second+'a';                      // for type casting
			for(int j=0;j<count;j++)
				cout<<ch;
		}
		cout<<endl;
	}
	return 0;
}
