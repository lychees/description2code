#include<bits/stdc++.h>

using namespace std;

int main()
	{
		int T,t;
		cin >> T;
		getchar();
		for(t=0;t<T;t++)
			{
				string s;
				getline(cin,s);
				int i;
				int leng=s.length();
				
				stack<char> S;
				S.push(s[0]);
				i=1;
				while(i<leng)
					{
						if(S.top()=='O')
							{
								if(s[i]=='X')
									{
										while(S.size()!=0 && S.top()=='O')
											{
												S.pop();
											}
										S.push(s[i]);
										i++;
									}
								else
									{
										S.push(s[i]);
										i++;
									}
							}
						else if(S.top()=='*')
							{
								S.push(s[i]);
								i++;
							}
						else if(S.top()=='X')
							{
								if(s[i]=='O')
									{
										while(s[i]=='O')
											{
												i++;
											}
									}
								else if(s[i]=='X')
									{
										S.push(s[i]);
										i++;
									}
								else if(s[i]=='*')
									{
										S.push(s[i]);
										i++;
									}
							}
					}
				int cnt=0;
				while(S.size()!=0)
					{
						if(S.top()=='O')
							cnt++;
						S.pop();
					}
				printf("%d\n",cnt);
			}
		return 0;
	}
