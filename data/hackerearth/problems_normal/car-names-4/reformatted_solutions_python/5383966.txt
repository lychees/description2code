n = int(raw_input())
while n:
    a = raw_input()
    b = list(set(list(a)))
    if (len(b) == 3):
        if (a.count(b[0]) == a.count(b[1]) and a.count(b[0]) == a.count(b[2])):
            if (b[0] *
                a.count(b[0]) in a and b[1] *
                a.count(b[0]) in a and b[2] *
                    a.count(b[0]) in a):
                print "OK"
            else:
                print "Not OK"
        else:
            print "Not OK"
    else:
        print "Not OK"
    n = n-1
