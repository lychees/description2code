t = input()
while t:
	x, y, n = map(int, raw_input().split())
	cal = map(int, raw_input().split())
	cal.sort()
	
	tot = c = j = 0
	
	for i in cal:
		tot += i
		if tot <= x:
			c += 1
		if tot <= y:
			j += 1			
	
	if c > j:
		print "Chef Won"
	elif c < j:
		print "Jhool Won"
	else:
		print "Tie"
		
	t -= 1
