#include<stdio.h>
#define gc() getchar()
//#define gc() getchar_unlocked()
inline int fastread()
{
int i=0;
char c=gc();
while(c<48) c=gc();
while(c>47)
{
i=i*10+c-48;
c=gc();
}
return i;
}
int main()
{
    int t,i,n;
    t=fastread();
    unsigned long long sum,x;
    while(t--)
    {
        n=fastread();
        sum=0;
        for(i=0;i<n;i++)
        {
            scanf("%llu",&x);
            sum+=x;
        }
        if(sum%n==0)
        {
            printf("YES\n");
            printf("%llu 0\n",sum/n);
        }
        else
        {
            printf("NO\n");
            printf("%llu %llu\n",sum/n,sum%n);
        }
    }
    return 0;
}
