#include <stdio.h>
#include <math.h>
#include <stdlib.h>
#include <algorithm>
using namespace std;



bool isprime(int n){
	for(int i=2;i<=sqrt(n);i++){
		if(n%i==0) return false;
	}
	return true;
}
int main(){
	int t;
	scanf("%d",&t);
	while(t-->0){
		int n;
		scanf("%d",&n);
		if(n==1) printf("1\n");
		else if(n==3) printf("3\n");
		else{
			if(isprime(n)){
				printf("2\n");
			}else {
				printf("1\n");
			}
		}
	}		
}
