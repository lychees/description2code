#include<cstdio>
#include<algorithm>
#include<vector>
using namespace std;

bool cmp(long i,long j)
{
	return i>j;
}

int main()
{
	long t;
	long n,k;
	long a[100000];
	scanf("%ld",&t);
	for(long i=0;i<t;i++)
	{
		scanf("%ld %ld",&n,&k);
		for(long j=0;j<n;j++)
		{
			scanf("%ld",&a[j]);
		}
		sort(a,a+n,cmp);
		long long sum = 0;
		for(long j=0;j<k && j<n;j++)
		{
			sum += a[j];
		}
		printf("%lld\n",sum);
	}
	return 0;
}
