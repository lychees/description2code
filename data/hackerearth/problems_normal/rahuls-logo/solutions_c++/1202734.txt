#include<iostream>
using namespace std;

void Top(int N)
{
	for(int i=1;i<=N;i++)
	{
		int j;
		for(j=1;j<=N-i;j++)
			cout<<" ";
			
			if(N%2==1)
			{
			
				for(int k=j-1;k<N;k++)
				if((k-(j-1))%(2)==1)
					cout<<" ";
				else
					cout<<"/";
				
				
				for(int k=j-1;k<=N;k++)
				if(k%2==1)
					cout<<" ";
				else
					cout<<"\\";	
			}
			else
			{
				for(int k=j-1;k<N;k++)
					if((k-(j-1))%2==1)
						cout<<" ";
					else
						cout<<"/";	
						
						
				for(int k=j-1;k<=N;k++)
					if(k%2==1)
						cout<<"\\";
					else
						cout<<" ";	
			}
					
				cout<<"\n";
		
						
	
	}
}

void Bottom(int N)
{
	for(int i=1;i<=N;i++)
	{
		int j;
	
			if(N%2==1)
			{
			
				for( j=1;j<=i+1;j++)
			cout<<" ";
			
				for(int k=j-1;k<=N+1;k++)
				if((k-(j-1))%2==1)
					cout<<" ";
				else
					cout<<"\\";	
				
				
				for(int k=j-1;k<=N+1;k++)
				if((k%2==1))
					cout<<" ";
				else
					cout<<"/";	
			}
			else
			{
				
				for( j=1;j<=i+1;j++)
			cout<<" ";
				for(int k=j-1;k<=N+1;k++)
				if((k-(j-1))%2==1)
					cout<<" ";
				else
					cout<<"\\";	
				
				
				for(int k=j-1;k<=N+2;k++)
				if((k%2==1))
					cout<<"/";
				else
					cout<<" ";
			}
			
		
			cout<<"\n";
	}
}




int main()
{
	int N;
	cin>>N;
	Top(N);											
	Bottom(N);
	N++;
//	cout<<"\n";
	Top(N);
	Bottom(N);
	
		
}
