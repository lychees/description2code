def gcd(a, b):
    if b == 0:
        return a
    return gcd(b, a % b)


t = input()
for _ in range(t):
    n = input()
    ar = map(int, raw_input().split())
    m = max(ar)
    su = sum(ar)
    g = 0
    for i in ar:
        g = gcd(i, g)
    c = (m / g) - n
    l = m/g
    su1 = (l*(2*g+g*(l-1)))/2
    if c % 2 == 0:
        print "Chandler", (su1-su)
    else:
        print "Joey", (su1-su)
