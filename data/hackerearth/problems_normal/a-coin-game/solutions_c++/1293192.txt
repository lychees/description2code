#include <bits/stdc++.h>
using namespace std;
static int t,d,n1,n2,n,a[1000005],k1,k2;
int main ()
{
int m=1;
for(int i=1;i<=1000000;i++)
{
if(a[i])continue;
a[i]=m;
if((i+m)<=1000000)
a[i+m]=-m;
m++;
}
scanf("%d",&t);
while(t--)
{
scanf("%d%d",&n1,&n2);
k1=max(n1,n2);
k2=min(n1,n2);
d=k1-k2;
if(a[k2]==d)printf("Don't Play\n");
else printf("Play\n");
}
return 0;}
