from operator import itemgetter
names = []


def compare(a, b):
    # print a, b
    if a[1] == b[1]:
        return 1 if a[0] > b[0] else -1
    return b[1] - a[1]


for tc in range(input()):
    a, b = raw_input().split()
    names.append((a, int(b)))

for name, marks in sorted(names, cmp=compare):
    print name, marks
