T = int(raw_input())
l = []
h = []
a = []
n = T
while T > 0:
    l.append(tuple(raw_input().split()))
    T = T-1
for i in l:
    h.append(int(i[0]))
    a.append(int(i[1]))

for i in range(len(l)):
    home = n-1
    away = n-1
    temp = h.count(a[i])
    home = home+temp
    away = away-temp
    print str(home)+' '+str(away)
