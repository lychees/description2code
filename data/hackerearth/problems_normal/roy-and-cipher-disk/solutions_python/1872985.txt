t=int(raw_input())
while t>0:
	text=raw_input()
	current=0
	for letter in text:
		character=ord(letter)-97
		diff=character-current
		if diff>=0 and diff<=13:
			print diff,
			current=character
			continue
		elif diff>13:
			diff=diff-26
			print diff,
			current=character
			continue
		elif diff<-12:
			diff=diff+26
			print diff,
			current=character
			continue
		elif diff<0 and diff>=-12:
			print diff,
			current=character
			continue
	t=t-1
	print ''
