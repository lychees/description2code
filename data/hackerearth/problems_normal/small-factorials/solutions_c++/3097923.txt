#include <iostream>
#include<vector>


using namespace std;

void fact(int n)
{
	
        int a[200];
        int t=1,temp=0,idx=0,i,j,x;
        a[0]=1;
        for(i=1;i<=n;i++){
                temp=0;idx=0;
                for(j=0;j<t;j++){
                        x=temp+a[idx]*i;
                        temp=x/10;
                        a[idx]=x%10;
                        idx++;
                }
                while(temp!=0){
                        a[idx]=temp%10;
                        temp=temp/10;
                        idx++;t++;
                }                       
        }
        for(i=idx-1;i>=0;i--)
                cout<<a[i];
        cout<<endl;
}

int main()
{
    int i=0;
    cin>>i;
    
    int no;
    while(i-->0){
    	cin>>no;
    	fact(no);
    }
    return 0;
}
