using namespace std;

int n;
ll int a[maxm];
void read()
{
    cin >> n;
    rep(i,n)
        cin >> a[i];
}

void solve()
{
    sort(a,a+n);
    ll int mx = a[n-1];

    ll int sum = a[1]-a[0];
    ll int prev = a[1]-a[0];

    for(int i = 2;i<n;++i)
    {
        prev = ((((a[i]-a[i-1])%mod)*(i%mod))%mod + prev%mod)%mod;
        sum = (sum%mod + prev%mod)%mod;
    }
    sum = ((sum%mod)*(mx%mod))%mod;
    cout << sum << endl;
}

int main()
{
    ios_base::sync_with_stdio(false);
    int t;
    cin >> t;
    while(t--)
    {
        read();
        solve();
    }
    return 0;
}
