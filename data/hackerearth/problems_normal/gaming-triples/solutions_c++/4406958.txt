#include<bits/stdc++.h>
using namespace std;
typedef long long int lli;
const int N=100*1000+3;
lli bit[N+9];
int  arr[N],cop[N];
int   L1[N],L2[N];
int  R1[N],R2[N];
inline void upd(int at,lli by){
    for(;at<=N;at+=at&-at)
        bit[at]+=by;
    return;
}
inline lli qry(int at){
    lli sum=0;
    for(;at>=1;at-=at&-at)
        sum+=bit[at];
    return sum;
}
int main(){
    int tc;scanf("%d",&tc);
    while(tc--){
        int n;scanf("%d",&n);
        memset(arr,0,sizeof arr);
        memset(L1,0,sizeof L1);
        memset(R1,0,sizeof R1);
        memset(L2,0,sizeof L2);
        memset(R2,0,sizeof R2);
        for(int i=1;i<=n;++i){
            int   temp;scanf("%d",&temp);
            arr[i]=temp;
            cop[i]=temp;
        }
        sort(cop+1,cop+n+1);
        for(int i=1;i<=n;++i){
            arr[i]=(lower_bound(cop+1,cop+n+1,arr[i])-(cop+1))+1;
        }
        //   for(int i=1;i<=n;++i)
        //      printf("%d ",arr[i]);
        //  printf("\n");
        memset(bit,0,sizeof bit);
        for(int i=1;i<=n;++i){
            upd(arr[i],1);
            L1[i]=(int)qry(arr[i]-1);
        }
        memset(bit,0,sizeof bit);
        for(int i=n;i>=1;--i){
            upd(arr[i],1);
            R1[i]=(int)qry(arr[i]-1);
        }
        lli ans1=0;
        for(int i=1;i<=n;++i){
            ans1+=(lli)L1[i]*R1[i];
        }
        // printf("ans1 is %lld\n",ans1);
        memset(bit,0,sizeof bit);
        for(int i=1;i<=n;++i){
            upd(arr[i],1);
            L2[i]=((i-1+1)-(int)qry(arr[i]));
        }
        memset(bit,0,sizeof bit);
        for(int i=n;i>=1;--i){
            upd(arr[i],1);
            R2[i]=((n-i+1)-(int)qry(arr[i]));
        }
        lli ans2=0;
        for(int i=1;i<=n;++i){
            //   printf("typ2 %d %d\n",L2[i],R2[i]);
            ans2+=(lli)L2[i]*R2[i];
        }
        // printf("ans2 is %lld\n",ans2);
        int plyr=-1;scanf("%d",&plyr);
        lli  sarika=-1,ananya=-1;
        if(plyr==0){
            ananya=ans1;
            sarika=ans2;
           // printf("ananya %lld sarika %lld\n",ananya,sarika);
            if(ananya<sarika){
                puts("Sarika");
                printf("%lld\n",2*min(sarika,ananya));
            }else if(ananya==sarika){
                puts("Sarika");
                printf("%lld\n",2*min(sarika,ananya));
            }else if(ananya>sarika){
                puts("Ananya");
                printf("%lld\n",2*min(sarika,ananya)+1);
            }
        }else if(plyr==1){
            sarika=ans1;
            ananya=ans2;
          //  printf("sarika  %lld ananya %lld\n",sarika,ananya);
            if(sarika<ananya){
                puts("Ananya");
                printf("%lld\n",2*min(sarika,ananya));
            }else if(sarika==ananya){
                puts("Ananya");
                printf("%lld\n",2*min(sarika,ananya));
            }else if(sarika>ananya){
                puts("Sarika");
                printf("%lld\n",2*min(sarika,ananya)+1);
            }//ads
        }
    }
    return 0;
}
