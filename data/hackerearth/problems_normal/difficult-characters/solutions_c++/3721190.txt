int cmp(pair<int,char> a, pair<int,char> b) {  
    if(a.first == b.first) return a.second > b.second;  
    else return a.first < b.first;  
}
int main(){

	ll t,n;
	in(t);

	{
		string s;
		in(s);
		n=sz(s);
		pair<int,char> a[26];
		
		for (int j = 25; j >-1; j--)
		{
			a[j].first=0;
			a[j].second=97+j;
		}
		for (int j = 0; j <n; ++j)

		}
