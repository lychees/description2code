#include <iostream>
#include <queue>
using namespace std;

int main()
{
	queue <int> st;
	queue <int> ideal;
    int n, i, j, t;
    cin>>n;
    for( i =0; i < n; i++){
    	cin>>t;
    	st.push(t);
    }
    for( i =0; i < n; i++){
    	cin>>t;
    	ideal.push(t);
    }
    int count = 0;
    for( i = 0; i < n; i++){
    	int temp = ideal.front();
    	ideal.pop();
    	//cout<<"ideal: "<<temp<<endl;
    	for( j = 0; j < n; j++){
    		if( temp == st.front() ){
    			st.pop();
    			count++;
    			break;
    		}
    		else{
    			int shift = st.front();
    			st.pop();
    			st.push(shift);
    			count++;
    		}
    		
    	}
    }
    cout<<count<<endl;
    return 0;
}
