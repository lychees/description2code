#include<bits/stdc++.h>
using namespace std;
typedef long long int uli;
const int mx=1e3+10;
const int mxor=(1<<11);
const int mod=1e7+7;
int f[2][mxor+10];
int a[mx],b[25];
int main(){
   int n,x,v;
   scanf("%d %d",&n,&x);
   int ca=0,cb=0;
   for(int i=0;i<n;i++){
      scanf("%d",&v);
      if(v<=(1<<10))a[ca++]=v;
      else b[cb++]=v;
   }
   int rw=0;
   f[0][0]=1;
   for(int i=0;i<ca;i++){
      rw^=1;
      v=a[i];
      for(int x=0;x<mxor;x++){
         f[rw][x]=f[rw^1][x]+f[rw^1][x^v];
         if(f[rw][x]>=mod)f[rw][x]-=mod;
      }
   }
   int ans=0;
//   cout<<"ca="<<ca<<" cb="<<cb<<endl;
   for(int m=0;m<(1<<cb);m++){
      int y=0;
      for(int i=0;i<cb;i++){
         if(m&(1<<i))y^=b[i];
      }
      if((x^y)<mxor){ 
         ans+=f[rw][x^y];
         if(ans>=mod)ans-=mod;
      }
   }
   printf("%d\n",ans);
   return 0;
}
