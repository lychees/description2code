#include <bits/stdc++.h>
#define m 1000000007LL
using namespace std;

long long ex(long long a, long long b)
{
	if(!b) return 1;
	if(b%2)
		return ( a*ex((a*a)%m,(b-1)/2) ) % m;
	else
		return ex((a*a) % m, b/2);
}

int main()
{
    ios::sync_with_stdio(0);
    int t; cin>>t;
    while(t--)
    {
    	long long n; cin>>n;
    //	cout<<ex(3,n)<<" "<<ex(2,n)<<endl;
    	cout<<((ex(3,n) - (ex(2,n)*3LL)%m + 3 + m) % m)<<endl;
    }
    return 0;
}
