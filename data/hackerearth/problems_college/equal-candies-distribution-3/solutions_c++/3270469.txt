#include <iostream>
#include<bits/stdc++.h>
using namespace std;


int main()
{
    int t,n,i,j,k;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	int a[n],b[n];
    	
    	for(i=0;i<n;i++)
    		cin>>a[i];
    		
    	sort(a,a+n);
    		
    	for(i=0;i<n;i++)
    		b[i]=a[i]-a[0];
    		
    	int sum=0;	
    	for(i=0;i<n;i++)
    	{
    		sum+=b[i]/5;
    		b[i]%=5;
    		sum+=b[i]/2;
    		b[i]%=2;
    		sum+=b[i];
    	}
    	
    	cout<<sum<<endl;
    }
    
    return 0;
}
