#include <bits/stdc++.h>
#define ll long long int
#define pb push_back
#define reset(a,b) memset(a,b,sizeof a)
#define _pause system("pause")

using namespace std;

bool pr[1000005];
const int MAX = 1000000;
void sieve()
{
	reset(pr,0);pr[1]=1;
	for(int i=2;i<=sqrt(MAX);i++)
	{
		if(pr[i]) continue;
		for(int j=i*i;j<=MAX;j+=i)
		{
			pr[j]=1;
		}
	}
}
int main()
{
	//ios_base::sync_with_stdio(0);
	//cin.tie(NULL);

	sieve();
	vector<int> p;
	for(int i=2;i<=sqrt(MAX)+10;i++)
		if(!pr[i]) p.pb(i);

	int n,k;cin>>n>>k;
	int l=p.size();
	int ans=0;
	for(int i=2;i<=n;i++)
	{
		if(!pr[i])
		{
			if(i==k)ans++;
			continue;
		}
		int sum=0;
		int x=i;
		for(int j=0;j<l;j++)
		{
			if(p[j]>x) break;
			if(x%p[j]==0) sum+=p[j];
			while(x%p[j]==0)x/=p[j];
		}
		if(!pr[x])sum+=x;
		if(sum==k)ans++;
	}
	cout<<ans<<endl;
	return 0;
}
