#include <iostream>
using namespace std;
int check(char val,string arr)
{
    int i,flag=0;
    for(i=0;i<arr.size();i++)
    if(val==arr[i])
    {
    flag=1;
    break;
    }
    if(flag==1)
    return 1;
    else
    return 0;
}

int main()
{
    int T,i,j;
    string W,output[101];
    cin>>T;
    for(i=0;i<T;i++)
    {
    cin>>W;
    int pos=0;
    string temp;
    for(j=0;j<W.size();j++)
    if(!check(W[j],temp))
    temp.insert(pos++,1,W[j]);
    output[i]=temp;
    }
    for(i=0;i<T;i++)
    cout<<output[i]<<endl;
    return 0;
}
