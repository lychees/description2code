#include<bits/stdc++.h>
using namespace std;
#define in(n) scanf("%lld",&n)
int main()
{
    long long int t,n,i,tm,tr,j;
    in(t);
    while(t--)
    {
        in(n);
        vector<long long int> v(n);
        for(i=0;i<n;i++)
            in(v[i]);
        i=0;j=n-1;
        tm=0,tr=0;
        while(j-i>-1 )
        {
            if(tm>tr)
                tr=tr+2*v[j],j--;
            else if(tm<=tr)
                tm=tm+v[i],i++;
        }
        cout << i << " " << n-1-j << endl;
    }
    return 0;
}
