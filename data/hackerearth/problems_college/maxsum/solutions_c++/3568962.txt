#include <bits/stdc++.h>
using namespace std;
long long max_sum(long arr[], int n, long long dp[]) {
   if(dp[n])
      return dp[n]; 
   long long a = max_sum(arr, n-1, dp);
   long long b = max_sum(arr, n-2, dp);  
   return (dp[n] = max(arr[n]+b, a));
}

int main() {
   int n, t, j, i;
   long arr[10000];
   long long ans;
   scanf("%d", &t);
   for(i = 1 ; i <= t ; i++) {
      scanf("%d", &n);
      long long dp[10005] = {0};  
      for(j = 0 ; j < n ; j++)
         scanf("%ld", &arr[j]);
      dp[0] = arr[0];
      dp[1] = max(arr[0], arr[1]);
      ans = max_sum(arr, n-1, dp);
      printf("%lld\n", ans);
   }
   return 0;
}
