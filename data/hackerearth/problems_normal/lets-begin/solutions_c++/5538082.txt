#include<bits/stdc++.h>
using namespace std;

#define mod 1000000007
#define W while
#define LL long long int

int main()
{
	LL  t,n,i;
	LL dp[1000001];
	dp[1]=100000000;
	dp[2]=1;dp[3]=1;dp[5]=1;dp[7]=1;
	dp[4]=2;dp[6]=2;
	
	for(i=8;i<=1000000;i++)
		dp[i]=1+min(dp[i-2],min(dp[i-3],min(dp[i-5],dp[i-7])));
	
	
	cin>>t;
	W(t--)
	{
		cin>>n;
		if(n==1)
			cout<<-1<<endl;
			else
			
		cout<<dp[n]<<endl;
	}
	return 0;
}
