#include<bits/stdc++.h>
#define ll long long int
using namespace std;
ll md=1e9+7;
ll fact[1000006];
void factfun()
{
	fact[0]=1;
	for(int i=1;i<1000006;i++)
	{
		fact[i]=(fact[i-1]*i)%md;
	}
}

ll power(ll a,ll b)
{
	ll x=1,y=a;
	while(b>0)
	{
		if(b%2==1)
		{
			x=(x*y)%md;
		}
		y=(y*y)%md;
		b=b/2;
	}
	return x%md;
}

ll inv(ll c)
{
	return(power(c,md-2));
}

int main()
{
	factfun();
	ll t,n,i,j,len,vow=0,cons=0,vowans=1,consans=1,freq[30],temp=0;
	string str;
	for(cin>>t;t;t--)
	{
		memset(freq,0,sizeof(freq));
		cin>>str;
		len=str.size();
		vow=0,cons=0;
		for(i=0;i<len;i++)
		{
			freq[str[i]-'a']++;
			if(str[i]=='a' || str[i]=='e' || str[i]=='i' || str[i]=='o' || str[i]=='u')
			{
				vow++;
			}
			else
			{
				cons++;
			}
		}
		temp=cons+1-vow;
		if(temp<0)
		{
			cout<<"-1"<<endl;
		}
		else
		{
			consans=fact[cons];
			vowans=(fact[cons+1]*inv(fact[temp]))%md;
			for(i=0;i<26;i++)
			{
				if(str[i]=='a' || str[i]=='e' || str[i]=='i' || str[i]=='o' || str[i]=='u')
				{
					vowans=(vowans*inv(fact[freq[i]]))%md;
				}
				else
				{
					consans=(consans*inv(fact[freq[i]]))%md;
				}
			}
			cout<<(vowans*consans)%md<<endl;
		}
	}
	return 0;
}
