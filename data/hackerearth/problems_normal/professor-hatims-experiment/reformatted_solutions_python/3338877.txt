from collections import defaultdict as ddic
BAD = "Suspicious lizards found!"
GOOD = "No suspicious lizards found!"


def solve(N, M, E):
    # lizards, interactions, edges
    G = ddic(list)
    for i, j in E:
        G[i].append(j)
        G[j].append(i)
    color = {}
    curcolor = 0

    for liz in xrange(1, 1+N):
        if liz in color:
            continue
        newstack = [liz]
        while newstack:
            stack = newstack[:]
            newstack = []
            while stack:
                e = stack.pop(0)
                color[e] = curcolor
                for f in G[e]:
                    if f in color and color[f] == curcolor:
                        return BAD
                    if f not in color:
                        newstack.append(f)
            curcolor ^= 1
    return GOOD


rr = raw_input
for tc in xrange(int(rr())):
    N, M = map(int, rr().split())
    E = [map(int, rr().split()) for i in xrange(M)]
    print solve(N, M, E)
