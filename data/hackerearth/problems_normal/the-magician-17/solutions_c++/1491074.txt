#include <iostream>
#include <cmath>

int main()
{
	using std::cin;
	using std::cout;
	using std::endl;
	using std::log;
	using std::pow;
	// total number of inputs
	int n;
	cin >> n;
	for (int ii = 0; ii < n; ii++)
	{	
		long N, K;
		int ff, pp;
		cin >> N >> K;
		if (N % 2 == 0 && K <= N/2)
		{
			cout << K*2 -1 << endl;
		}
		else if (N % 2 == 1 && K <= N/2+1)
		{
			cout << K*2 -1 << endl;	
		}
		else
		{
			ff = 3+2*(N-K-1);
			pp = int(log((N - 1e-5)/ff)/log(2))+1;
			cout << long(2*N-ff*pow(2,pp)) << endl;
		}
	}
	return 0;
}
