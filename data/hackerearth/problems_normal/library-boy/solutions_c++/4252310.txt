using namespace std;

int main()
{
    ll freq[26];
    memset(freq,0,sizeof(freq));
    ll n,ans=0,i;
    scanf("%lld\n",&n);
    while(n--)
    {
        string str;
        getline(cin,str);
        freq[str[0]-'a']++;
    }
    for(i=0;i<26;i++)
    {
        if(freq[i]!=0)
        {
            if(freq[i]%10==0)
                ans+=freq[i]/10;
            else
                ans+=freq[i]/10+1;
        }
    }
    pll(ans);
}
