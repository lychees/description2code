#include <iostream>
using namespace std;

int F(int x)
{
	int s=0;
	while(x!=0)
	{
		if(x%10==4 || x%10==7)
		s++;
		
		x/=10;
	}
	return s;
}

int main()
{
    int a,l;
    cin >> a >> l;
    
    bool b=false;
    
    for(int i=a+1;i<100000000;i++)
    {	
    	int j;
    	for(j=0;j<l;j++)
    	{
    		if(F(a+j)!=F(i+j))
    		{
    			break;
    		}
    	}
    	if(j==l)
    	{
    		b=true;
    		cout << i;
    		break;
    	}
    }
    if(b==false)
    {
    	cout << -1;
    }
}
