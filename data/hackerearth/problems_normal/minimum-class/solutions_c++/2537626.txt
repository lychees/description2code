#include <iostream>
using namespace std;

int main()
{
   long long t,n,i,j,k,l,m,count=2;
   cin>>t;
   while(t--)
   {
   	cin>>n;
   	if(n==1)
   	count=1;
   	else
   	if(n==2)
   	count=2;
   	else
   	{
   	m=n-2;
   	j=2;
   	while(true)
   	{
   	 if(m==0)
   	 {
   	 	break;
   	 }
   	 if(m<=j)
		{
			++count;
			break;
		}
		else	
	if(m>=2*j)
	{
	   m=m-2*j;
	   ++j;
	   count=count+2;	
	}	
	else
	 if(m>j)	
	 {
	 	count=count+2;
	 	break;
	 }
   	
    }
   
   }
   cout<<count<<endl;
   	count=2;
}
    return 0;
}
