t = input()
for i in range(t):
    r, c, w = map(int, raw_input().split())
    res = 0
    if w == 1:
        res = r*c
    elif c == w:
        res = r+c-1
    else:
        cnt = 0
        while c >= 2*w:
            c = c-w
            cnt += 1
        if c == w:
            res = cnt*r+r+w-1
        else:
            res = (cnt+1)*r+w
    print "Case #"+str(i+1)+": "+str(res)
