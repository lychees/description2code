#include <bits/stdc++.h>
#define ll long long
#define mod 1000000007
#define lim 10000000
using namespace std;
vector<ll>v;
bool a[lim+1]={false};
int main() {
ll n,t;
for(ll i=2;i*i<=lim;i++)
{
	if(!a[i])
	for(ll j=2*i;j<=lim;j+=i)
	a[j]=true;
}
for(ll i=2;i<=lim;i++)
if(!a[i])
v.push_back(i);
ll m;
cin>>t;
while(t--)
{
	scanf("%lld%lld",&n,&m);
	ll ans=v[v[n-1]-1];
	ans%=mod;
	ll we=v[v[m-1]-1];
	we%=mod;
	ans*=we;
	ans%=mod;
	printf("%lld\n",ans);
}
}
