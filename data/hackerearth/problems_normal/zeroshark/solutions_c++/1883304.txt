#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
int main()
{
    long long int fib[100001],ans[100001],i,j,n,t;
    fib[1]=0;
    fib[2]=1;
    fib[3]=1;
    for(i=4;i<100001;i++)
    {
        fib[i]=(fib[i-1]+fib[i-2])%mod;
    }
    ans[0]=0;
    ans[1]=0;
    ans[2]=1;
    ans[3]=2;
    for(i=4;i<100001;i++)
    {
        ans[i]=(ans[i-1]+ans[i-2]+fib[i])%mod;
    }
    cin>>t;
    while(t--)
    {
        cin>>n;
        cout<<ans[n]<<"\n";
    }
    return 0;
}
