int powr(int a)
{
    if(a==0) return 1;
    else return 2*powr(a-1);
}
int main()
{
    int t,n;
	scanf("%d",&t);
	while(t--)
    {
        cin>>n;
        int r=1,cnt=1;
        while(n>1)
        {
            if(n%2==1) r+=powr(cnt);
            n/=2;
            cnt++;
        }
        cout<<r<<endl;
    }
