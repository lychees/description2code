#include <bits/stdc++.h>

using namespace std;

int main(void) {
    int t;
    cin >> t;
    while (t--) {
        int m, k,x;
        cin >> m >> k;
        vector<int>nums;
        for (int i = 0; i < m; ++i) {
            scanf("%d",&x);
            nums.push_back(x);
        }
        sort(nums.begin(),nums.end());
        vector<int>sum;
        int last = nums.size()-1;
        for (int j = 0; j < nums.size() / 2; ++j) {
            sum.push_back(nums[j]+nums[last]);
            last--;
        }
        sort(sum.begin(),sum.end());
        int mini = 0;
        if(sum.size()>=2) mini = sum[sum.size()-1]-sum[0];
        printf("%d\n",mini);
        if(mini==k){
            puts("Lucky chap!");
        }else if (mini<k){
            puts("Chick magnet Jhool!");
        }else{
            puts("No more girlfriends!");
        }
    }
}
