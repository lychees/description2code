    #include <iostream>
    #include<math.h>
    using namespace std;
     
    int calc(long int N,long int D,long int R)
    {
    	if(N<=0)
    			return 0;
    	long int temp;
    	temp=N/D;
    	N=N-temp*D;
    	while(R)
    	{
    		int flag=0;
    		if(N<D)
    		{
    			if(flag)
    				goto l;
    			N=N*10;
    			flag=1;
    		}
    		temp=N/D;
    		N=N-temp*D;
     
    		l:
    		R--;
    	}
    	return temp;
    }
     
     
    int main()
    {
        int T;
        long int R;
        long int N,D;
        int i;
        cin>>T;
        for(i=0;i<T;i++)
        {
        	cin>>N>>D>>R;
        	cout<<calc(N,D,R)<<endl;
     
        }
        return 0;
    }
