#include <bits/stdc++.h>
using namespace std;
int z[300005];
string a,b;
int main()
{
    cin.sync_with_stdio(false);cin.tie();
    int t;
    cin>>t;
    while(t--)
    {
        memset(z,0,sizeof(z));
        int n=0,sv;
        cin>>a;
        sv=a.size();int d=0;
        cin>>b;  d=sv+1+b.size();   b=b.append(b);a=a.append("$");a=a.append(b);
        n=a.size();
        int ans=0;
        for(int i=1,l=0,r=0;i<n;i++)
        {
            if(i<=r)
                z[i]=min(z[i-l],r-i+1);
            while(z[i]+i<n and a[z[i]]==a[i+z[i]])z[i]++;
            if(i+z[i]-1>r)
                l=i,r=i+z[i]-1;
            if(i<d and z[i]==sv)ans++;
        }
        cout<<ans<<"\n";
    }
    return 0;
}
