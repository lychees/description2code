#include <iostream>
#include<string>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t;
   cin>>t;
   string s;
   for(int i=0;i<t;i++){
   	cin>>s;
   	int len=s.size();
   	int x=0;
   	int cnt=0;
   	for(int j=0;j<len;j++){
   		if(s[j]=='a'||s[j]=='A'||s[j]=='b'||s[j]=='B'||s[j]=='c'||s[j]=='C'){
   			
   			x=x+1;
   			if(j==len-1){
   					cnt=cnt+(x*(x+1))/2;
   			}
   		}else{
   			
   			cnt=cnt+(x*(x+1))/2;
   			x=0;
   		}
   		
   		
   	}
   	cout<<cnt<<endl;
   	
   }
   
   
    return 0;
}
