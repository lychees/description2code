#include<bits/stdc++.h>
using namespace std;

long long int a[100010];

int main()
{
	long long int i,j,t,y,n,x,count;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld%lld%lld",&n,&x,&a[0]);
		count=0;
		for(i=1;i<n;i++)
		{
			scanf("%lld",&a[i]);
			if(a[i]<=a[i-1])
			{
				y=a[i-1]-a[i];
				if(y==0)
				{
					a[i]+=x;
					count++;
				}
				else
				{
					if(y%x!=0)
					{
						a[i]+=((y/x)*x+x);
						count+=((y/x)+1);
					}
					else
					{
						a[i]+=((y/x)*x);
						count+=(y/x);
					}
				}
				
			}
			
		}
		printf("%lld\n",count);
	}
	
}
