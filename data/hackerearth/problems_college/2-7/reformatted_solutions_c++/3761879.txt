#include <bits/stdc++.h>
using namespace std;

int main() {
  int tc;
  cin >> tc;
  while (tc--) {
    int n;
    cin >> n;
    int f[21] = {0};
    for (int i = 2; i <= 20; i++) {
      int j = i;
      while (j <= n) {
        f[i] += n / j;
        j = j * j;
      }
    }
    int maxm = -1;
    int var = 0;
    for (int i = 2; i <= 20; i++) {
      if (f[i] >= maxm) {
        maxm = f[i];
        var = i;
      }
    }
    cout << var << endl;
  }
  return 0;
}
