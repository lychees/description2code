for _ in xrange(input()):
	n=input()
	pos=map(int,raw_input().split())
	sz=map(int,raw_input().split())
	a=[]
	for i in xrange(n):
		a.append([pos[i],sz[i]])
	a.sort()
	tmp=a[0][0]+a[0][1]
	ans=0
	for i in xrange(1,n):
		ans+=a[i][0]-tmp
		tmp+=a[i][1]
	for z in xrange(1,n):
		t=0
		tmp=a[z][0]+a[z][1]
		for i in xrange(z+1,n):
			t+=a[i][0]-tmp
			tmp+=a[i][1]
		tmp=a[z][0]
		for i in xrange(z-1,-1,-1):
			t+=tmp-(a[i][0]+a[i][1])
			tmp-=a[i][1]
		ans=min(ans,t)
	print ans
