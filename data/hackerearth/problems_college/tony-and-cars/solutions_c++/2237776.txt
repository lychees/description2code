#include <bits/stdc++.h>
#define ll long long
#define pi pair<int,int>
#define pl pair<long long,long long>
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	int t;
	cin>>t;
	ll a[14];
	a[0]=0;
	a[1]=9;
	ll x=9;
	for(int i=2;i<=13;i++)
	{
		a[i]=(x*10*i);
		x=x*10;
	}
	for(int i=2;i<=13;i++)
	{
		a[i]=a[i]+a[i-1];
		//cout<<a[i]<<"\n";
	}
	while(t--)
	{
		ll n,m;
		cin>>n;
		m=n;
		int c=0;
		while(m)
		{
			m=m/10;
			c++;
		}
		ll ans=a[c-1];
		ll x=1;
		for(int i=0;i<c-1;i++)
		{
			x=(x*10);
		}
		ans=ans+((n-x+1)*c);
		cout<<ans<<"\n";
	}
}
