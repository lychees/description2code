#include <algorithm>
#include <cstdio>
#include <iostream>
#include <vector>

#include <cstring>
#include <string>

#include <functional>
#include <numeric>

using namespace std;

int A[2000007];
int B[2000007];

int main()
{
    int n; cin >> n;
    for (int i = 0; i < n; ++i) cin >> A[i];
    for (int i = 0; i < n; ++i) cin >> B[i];
    int maxa = 0;
    int maxb = 0;
    for (int i = 0; i < n - 1; ++i) {
    	maxa = max(maxa, abs(A[i] - A[i+1]));
    }
	for (int i = 0; i < n - 1; ++i) {
    	maxb = max(maxb, abs(B[i] - B[i+1]));
    }
    
    if (maxa < maxb) cout << "Brian\n" << maxb << endl;
    else if (maxa == maxb) cout << "Tie\n" << maxa << endl;
    else cout << "Dom\n" << maxa << endl;
    return 0;
}
