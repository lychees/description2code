#include <iostream>
#include<algorithm>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        string s1,s2;
        cin>>s1>>s2;
        string s3=s1+s2,s3_rev=s3;
        string s4=s2+s1,s4_rev=s4;
        reverse(s3_rev.begin(),s3_rev.end());
        reverse(s4_rev.begin(),s4_rev.end());
        if(s3==s3_rev||s4==s4_rev)
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
    }
    return 0;
}
