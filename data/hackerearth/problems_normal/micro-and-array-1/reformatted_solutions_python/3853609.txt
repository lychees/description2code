N, Q = map(int, raw_input().split())
array = map(int, raw_input().split())

for i in xrange(Q):
    op = map(int, raw_input().split())
    if len(op) == 2:
        try:
            out = next(n[0] for n in enumerate(array) if n[1] >= op[1])
            print(out+1)
        except BaseException:
            print(-1)
    else:
        array[op[1]-1] = op[2]
