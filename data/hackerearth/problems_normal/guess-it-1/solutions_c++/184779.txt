#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int t,sum,i;
    string str;
    cin>>t;
    while(t--)
    {
    	cin>>str;
    	sum=0;
    	for(i=0;str[i]!='\0';i++)
    	{
    		if(str[i]>=48 && str[i]<58)
    		{
    			sum+=str[i]-48;
    		}
    	}
    	sum=sum%9;
    	if(sum==0 || sum==9)
    	cout<<"0 or 9\n";
    	else
    	cout<<9-sum<<endl;
    }
    return 0;
}
