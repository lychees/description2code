t = int(raw_input())
count = 0
for i in range(t):
    s = raw_input()
    stack = []
    stack.append(s[0])
    for j in range(1, len(s)):
        if len(stack) != 0:
            if stack[-1] == s[j]:
                del stack[-1]
            else:
                stack.append(s[j])
        else:
            stack.append(s[j])
    if len(stack) == 0:
        count = count+1
print count
