int pow(long long int a, long long int b)
{
   long long int ret = 1;
   while(b)
   {
	  if(b % 2 == 1)
	     ret = (ret*a)%M;
 	  a = (a*a)%M;
	  b /= 2;
   }
 return ret%M;
}
int sum(long long int c, long long int k)
{
   long long int r;

   while(c > 0 && k--)
   {
	  r = c % 10;
 	  c = c / 10;
      ret = ret + (r%M); 
   }

}
