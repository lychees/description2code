from sys import stdin
mod = 10**9 + 7
import math
fa = [1]
for i in xrange(1,3000):
	fa.append((i*fa[-1])%mod)
def ncr(n,r):
	if r>n:
		return 0
	num = fa[n]
	den = (fa[r] * fa[n-r])%mod
	den = pow(den, mod-2, mod)
	return (num * den)%mod
t = int(stdin.readline())
for _ in xrange(t):
	n,m = map(int,stdin.readline().split())
	ans = 0
	if m%2==0:
		x = 18*n
		y = 6*n
		for i in xrange(m/2+1):
			ans += (ncr(x,i)*ncr(x,m/2-i)*ncr(y,i)*ncr(y,m/2-i))
	print ans%mod
