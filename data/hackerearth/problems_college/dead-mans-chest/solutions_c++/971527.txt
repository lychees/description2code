#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
using namespace std;
string str;
int table[26];
int main() {
    int t;
    scanf("%d",&t);
    while(t--) {
        int count=0;
        memset(table,0,sizeof(table));
        cin>>str;
        for(int i=0;i<str.size();i++) table[str[i]-'a']++;
        for(int i=0;i<26;i++) if(table[i]&1) count++;
        if(count!=0&&(count%2==0)) printf("DAVY JONES\012");
        else printf("JACK\012");
    }
    return 0;
}
