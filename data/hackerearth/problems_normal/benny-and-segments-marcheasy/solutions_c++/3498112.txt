#include<iostream>
#include<algorithm>
#include<string.h>
#include<vector>
using namespace std;

vector< pair<pair<long long,long long>,long long> >v;

int main()
{
    int t;
    cin>>t;
    while(t--) {
        v.clear();
        long long i,j,n,m,temp,x,y,cnt,strt,l;
        bool ans=false;
        cin>>n>>l;
        for(i=1;i<=n;i++) {
            cin>>x>>y;
            v.push_back(make_pair(make_pair(x,-1),y));
            v.push_back(make_pair(make_pair(y,1),x));
        }
        sort(v.begin(),v.end());
        for(i=0;i<v.size() && !ans;i++) {
            if(v[i].first.second==-1 && v[i].second-v[i].first.first<=l) {
                cnt=1; strt=v[i].first.first;
                for(j=i+1;j<v.size() && !ans;j++) {
                    if(v[j].first.second==1 && v[j].second>=strt) {
                        if(v[j].first.first-strt==l) {
                            ans=true; break;
                        } else cnt--;
                    } else if(v[j].first.second==-1 && v[j].second<=strt+l) cnt++;
                    if(cnt<=0) break;
                }
            }
        }
        if(ans) cout<<"Yes\n";
        else cout<<"No\n";
    }
    return 0;
}
