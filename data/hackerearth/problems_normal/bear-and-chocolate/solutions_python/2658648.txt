for testCases in range(input()):
    flag = False
    n = input()
    horizontalArr = []
    require = 0
    for i in range(n):
        r = raw_input()
        require += r.count('#')
        r = map(str,r)
        horizontalArr.append(r)
    verticalArr = [['.' for i in range(n)]for i in range(n)]
    for i in range(n):
        for j in range(n):
            verticalArr[j][i] = horizontalArr[i][j]
    horzCnt, vertCnt = 0,0
    for i in horizontalArr:
        horzCnt += i.count('#')
        if horzCnt*2 == require:
            flag = True
            break
    for i in verticalArr:
        vertCnt += i.count('#')
        if flag or vertCnt*2 == require:
            flag = True
            break
    if flag:
        print 'YES'
    else:
        print 'NO'
