#include <iostream>
#include <vector>
using namespace std;

int main()
{
	int M;
	cin >> M;
	typedef std::vector< std::vector<long long> > matrix;
	matrix a(M, std::vector<long long>(M));

	for(int i = 0; i < M; ++i)
		for(int j = 0; j < M; ++j)
			cin >> a[i][j];

	long long sum1 = 0, sum2 = 0;
	for(int i = 0; i < M; ++i)
		sum1 += a[i][M - i - 1];
	for(int i = 0; i < M; ++i)
		sum2 += a[i][i];

	long long sum = sum2 - sum1;
	cout << abs(sum) << endl;
}
