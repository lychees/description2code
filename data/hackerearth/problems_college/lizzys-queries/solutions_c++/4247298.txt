#include<bits/stdc++.h>
#define ll long long
#define ull unsigned long long
#define e 1000000007
#define MAX 1000006
#define rep(i,a,b) for(i=a;i<b;i++)
#define pb push_back
using namespace std;

char s[MAX],c;
ll n,q,l,r,ch,x,pos;

int main()
{
	ll i,j,flag;
	//ios_base::sync_with_stdio(false);cin.tie(0);
	scanf("%lld%lld",&n,&q);
	scanf("%s",s);
	while(q--)
	{
		//cout<<q<<endl;
		scanf("%lld",&ch);
		if(ch==0)
		{
				scanf("%d",&pos);
				cin>>c;
				s[pos-1]=c;
		}
		else
		{
				scanf("%lld%lld",&l,&r);
				flag=1;
				for(j=0;j<(r-l+1)/2;j++)
				{
					if(s[l-1+j]!=s[r-1-j])
					{
						flag=0;
						break;
					}
				}
				if(flag)
				cout<<"Yes"<<endl;
				else
				cout<<"No"<<endl;
		}
	}
	return 0;
}
