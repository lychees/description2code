#include<iostream>
using namespace std;
bool is_sum(int s,int arr[],int n){
	if(s==0)
		return true;
	if(n<0)
		return false;
	return (is_sum(s-arr[n],arr,n-1)||is_sum(s,arr,n-1));
}
int main(){
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		int arr[n];
		for(int i=0;i<n;i++)
			cin>>arr[i];
		int s;
		cin>>s;
		if(is_sum(s,arr,n-1))
			cout<<"YES\n";
		else
			cout<<"NO\n";
	}
}
