# print 'Hello World!'
t = input()
while t > 0:
    t -= 1
    n, k, p = map(int, raw_input().split())
    l = sorted([int(i) for i in raw_input().split()])
    c = 0
    for i in range(n-k+1):
        if l[k+i-1]-l[i] <= 2*p:
            c += 1
    if c == 0:
        print"YES"
    else:
        print "NO"
