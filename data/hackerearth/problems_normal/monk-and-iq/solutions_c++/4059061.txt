#include<bits/stdc++.h>
using namespace std;
vector<pair<int,int> > v;
typedef pair<int,int> P;
priority_queue<P,vector<P>,greater<P > > pq;
pair<int,int> pr;
int p,n,c,f[100000],en[100000],iq1[100000],iq2[100000],no[100000];
int main()
{
    int i,j,in,mn,l,r,m;
    scanf("%d %d %d",&c,&p,&n);
    for(i=0;i<n;i++)
    {
        no[i]=1;
        scanf("%d",en+i);
        iq2[i]=en[i];
    }
    for(i=0;i<p;i++)
        scanf("%d",f+i);
    for(i=0;i<c;i++)
        pq.push({en[i],i});
    for(i=0;i<p;i++)
    {
        pr=pq.top();
        v.push_back(pr);
        mn=pr.second;
        in=0;
        pq.pop();
        for(j=1;j<c;j++)
        {
            pr=pq.top();
            if(pr.first==v[0].first)
            {
                if(pr.second<mn)
                {
                    mn=pr.second;
                    in=j;
                }
            }
            else
                break;
            v.push_back(pr);
            pq.pop();
        }
        printf("%d ",mn+1);
        no[mn]++;
        iq1[mn]=iq2[mn];
        iq2[mn]=f[i];
        v[in].first=no[mn]*(iq1[mn]+iq2[mn]);
        for(j=0;j<v.size();j++)
            pq.push(v[j]);
        v.clear();
    }
}
