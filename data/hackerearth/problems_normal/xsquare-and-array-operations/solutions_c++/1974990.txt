#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    int n;
    vector <long long int> v;
    long long int sum,inp;
    while(t--)
    {
    	sum=0;
    	scanf("%d",&n);
    	v.resize(n);
    	for(int i=0;i<n;i++)
    	{
    		scanf("%lld",&v[i]);
    	}
    	for(int i=0;i<n-1;i++)
    	{
    		sum+=max(v[i],v[i+1]);
    	}
    	cout<<sum<<endl;
    	v.clear();
    }
    
}
