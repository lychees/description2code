#include <iostream>
using namespace std;
#include <bits/stdc++.h>
#include <string.h>

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string a, b;
    	cin>>a>>b;
    	sort(a.begin(),a.end());
    	sort(b.begin(),b.end());
    	int i=0, j=0, flag=0, k=0;;
    	while(i<a.size())
    	{
    		j=0; k=0;
    		while(j<b.size())
    		{
    			if(a[i]==b[j])
    			{
    				k=1; flag=1;
    				break;
    			}
    			j++;
    		}
    		if(k==1)
    		{
    			break;
    		}
    		i++;
    	}
    	if(flag==0)
    	cout<<"NO"<<endl;
    	else
    	cout<<"YES"<<endl;
    }
    return 0;
}
