def fracnum(N, D, R):
    N = N % D

    v, d = abs(N)*10, abs(D)
    while R:
        (v, r, R) = ((v % d)*10, v//d, R-1)

    return r


name = int(raw_input())
count = 0
while count < name:
    count += 1
    strings = raw_input()
    words = strings.split()
    N = int(words[0])
    D = int(words[1])
    R = int(words[2])
    print fracnum(N, D, R)
