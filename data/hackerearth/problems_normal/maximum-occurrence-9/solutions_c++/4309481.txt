#include<iostream>
#include<string.h>
#include<cstring>
using namespace std;
int main()
{
	int arr[256];
	char s[1000];
	char str[1000];
	memset(arr,0,256*sizeof(arr[0]));
	
	gets(s);
	
	int k=0;
	for(int i=0;i<strlen(s);i++)
	{
		//cout<<arr[s[i]]<<endl;
		if(!arr[s[i]])
		{
			str[k]=s[i];
			//cout<<str[k]<<endl;
			k++;
		}
		arr[s[i]]++;
	}
	int max;
	char maxchar;
	max=arr[str[0]];
	maxchar=str[0];
	for(int i=1;i<=k-1;i++)
	{
		if(arr[str[i]]>max)
		{
			max=arr[str[i]];
			maxchar=str[i];
		}
	    else if(arr[str[i]]==max)
		{
		   if((int(str[i]))<(int(maxchar)))
		   {
		   	max=arr[str[i]];
			maxchar=str[i];
		   	
		   }
		}
	}
	cout<<maxchar<<" "<<max<<endl;
	
	
	return 0;
}
