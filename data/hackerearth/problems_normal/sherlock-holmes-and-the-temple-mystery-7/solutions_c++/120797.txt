typedef unsigned long long uint64;

bool check(int n,int cost[],int sum){
     bool **subset;
	 subset=new bool *[sum+1];
	 for(int i=0;i<sum+1;++i)	subset[i]=new bool[n+1];
	 for (int i = 0; i <= n; i++)      subset[0][i] = true;
     for (int i = 1; i <= sum; i++)      subset[i][0] = false;
     for (int i = 1; i <= sum; i++)
     {
       for (int j = 1; j <= n; j++)
       {
         subset[i][j] = subset[i][j-1];
         if (i >= cost[j-1])
           subset[i][j] = subset[i][j] || subset[i - cost[j-1]][j-1];
       }
     }
     return subset[sum][n];
}

int main()
{
	int testcases;
	for(scanf("%d",&testcases);testcases--;){
		int n;
		int cost[101];
		int sum=0;
		scanf("%d",&n);
		for(int i=0;i<n;++i){
			scanf("%d",&cost[i]);
			sum=sum+cost[i];
		}
		if(sum%2){
			printf("NO\n");
			continue;
		}
		if(check(n,cost,sum/2)==true)	printf("YES\n");
		else printf("NO\n");
	}
	return 0;
}
