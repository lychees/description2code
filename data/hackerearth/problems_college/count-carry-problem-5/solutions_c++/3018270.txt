#include <iostream>
using namespace std;

int main()
{
    long num1;
    long num2;
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%ld",&num1);
    	scanf("%ld",&num2);
    	int count = 0;
    	int carry = 0;
    	while(num1>0&&num2>0)
    	{
    		int i = num1%10;
    		int j= num2%10;
    		if((i+j+carry)>9)
    		{
    			++count;
    			carry = (i+j+carry)%10;
    		}
    		num1 = num1/10;
    		num2 = num2/10;
    	}
    	if(count>0)
    	{
    		
    		if(count>1)
    			printf("%d carry operations\n",count);
    		else
    			printf("%d carry operation\n",count);
    	}
    	else
    		printf("No carry operation\n");
    	
    }
    return 0;
}
