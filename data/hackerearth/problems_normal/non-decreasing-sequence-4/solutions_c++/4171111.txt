using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        int arr[n];
        for(int i=0;i<n;i++)
            cin>>arr[i];
        int brr[n+1];
        memset(brr,0,sizeof(brr));
        brr[0]=-1;
        int index=0;
        for(int i=0;i<n;i++)
        {
            if(arr[i]>brr[index])
            {
                brr[++index]=arr[i];
            }
            else
            {
                int x=lower_bound(brr,brr+index,arr[i])-brr;
                brr[x]=arr[i];
            }
        }
        int l;
        cin>>l;
        if(l>index)
        {
            cout<<-1<<"\n";
        }
        else
            cout<<brr[l]<<"\n";
    }
}
