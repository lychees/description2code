#include <stdio.h>
int main(){
	int n,q,l,i,j,x,y,max,min,lca,f;
	scanf("%d %d",&n,&q);
	char a[n+1],k1,b;
	getchar();
	for(i=1;i<=n;i++) {
	 scanf("%c",&k1);
	 a[i]=k1;
	}
	a[i+1]='\0';
	int z[100001];
	while(q--){
		scanf("%d %d",&x,&y);
		if(a[x]=='1' || a[y]=='1') printf("No\n");
		else if(x==y) {
			if(a[x]=='1')puts("No");
			else puts("Yes");
		}
		else {
			if(x>y) {
				max=x;min=y;
			}
			else {
				max=y;min=x;
			}
			x=max;y=min;
			for(i=0;i<=max+1;i++)z[i]=0;
			while(min>=1) {
				z[min]=1;
				min/=2;
			}
			while(max>=1&&z[max]!=1) {
				max/=2;
			}
			lca=max;
			f=0;
			max=x;min=y;
			while(y>=lca) {
				if(a[y]=='1') {f=1;break;}
				y/=2;
			}
			if(f) puts("No");
			else {
				f=0;
				while(x>=lca) {
					if(a[x]=='1') {f=1;break;}
					//if(x<min) {f=1;break;}
					x/=2;
				}
				if(f) puts("No");
				else puts("Yes");
			}
		}
	}
	return 0;
}
