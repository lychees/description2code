#include<bits/stdc++.h>

using namespace std;

typedef long long ll;

const int max_n= 2*(1e4) + 10;
const int max_b= (1e6) + 10;

ll k, n, m, best[max_b];
bool b[max_b];

bool possible(ll j)
{
	ll i, sq, mini= k+1;
	
	sq= sqrt(j)+1;
	
	sq= min(sq, j-1);
	
	for(i=2;i<=sq;++i)
	{
		if(j%i==0)
		mini= min(mini, best[i]+best[j/i]);
	}
	
	if(mini<=k || b[j])
	{
		best[j]= mini;
		return false;
	}
	else
	{
		best[j]= 1;
		printf("%lld ", j);
		return true;
	}
}

int main()
{
	ll i, j, mul, it, low, a;
	
	scanf("%lld %lld %lld", &n, &k, &m);
	
	for(i=0;i<m;++i)
	{
		scanf("%lld", &a);
		
		if(a<max_b)
		{
			b[a]= true;
		}	
	}
	
	j=1;
	i=0;
	while(i<n)
	{
		if(possible(j))
		{
			++i;
		}
		
		++j;
	}
	
	return 0;
}
