#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,arr[100005],freq[1000005]={0},j=0,i;
		scanf("%d",&n);//cout<<n<<"\n";
		for(i=0;i<n;i++)
		{
			int temp;
			scanf("%d",&temp);
			if(!freq[temp]){
				freq[temp]++;arr[j++]=temp;}
		}
		sort(arr,arr+j);
		for(i=0;i<=j/2-1;i++)
		{
			printf("%d ",arr[j-i-1]);
			//if(i!=n-i-1)
				printf("%d ",arr[i]);
		}
		if(j&1)printf("%d ",arr[i]);
		printf("\n");
	}
    //cout << "Hello World!" << endl;
    return 0;
}
