void swap(int *a, int *b) {
    int t = *a;
    *a = *b;
    *b = t;
}

void subset(int A[], int n) {
    int cnt[n];

    for (int i = 0; i < n; i++) {
        cnt[i] = 1;
    }
    sort(A, A + n);
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < i; j++) {
            if (A[i] % A[j] == 0 && cnt[j] + 1 > cnt[i]) {
                //cout << A[i] << "  " << A[j] << endl;
                cnt[i] = cnt[j] + 1;
            }
        }
    }
    int mx = 1;
    for (int i = 0; i < n; i++) {
        if (mx < cnt[i]) {
            mx = cnt[i];
        }
    }
    if (mx < 2) {
        cout << "-1\n";
        return;
    }
    cout << mx << endl;
}
