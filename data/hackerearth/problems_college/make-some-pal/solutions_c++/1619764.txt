#include<bits/stdc++.h>
using namespace std;

bool ispalin(string s) {
	for(int j = 0 ; j< s.size() - 1 ; ++j) {
		//printf("j=%c  size-j = %c \n",s[j],s[s.size()-1-j]);
		if(s[j] != s[s.size() - 1 - j]){
			//cout << endl;
			return false;
		}
	}
	return true;
}

int main() {
	int t;
	cin >> t;

	while(t--) {
		string s;
		int ans = 0 ,len;
		cin >> s;
		len = s.size();
		reverse(s.begin(),s.end());
		//cout << s << endl;
		for(int i = 0 ; s.size() > 0 ; ++i) {
			if(ispalin(s)){
				//printf("i = %d\n ",i);
				break;
			}
			s.resize (s.size () - 1);
			ans++;
		}
		cout << ans + len << endl;
	}
	return 0;
}
