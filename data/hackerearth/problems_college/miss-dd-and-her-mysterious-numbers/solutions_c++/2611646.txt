#include <bits/stdc++.h>
#define ll long long
using namespace std;

ll hash_arr[10000005];
ll arr[10000005];


int main()
{
    ll n;
    cin>>n;
    ll q;
    cin>>q;
    ll pos;
     
   for(ll i=2; i <= n; i+=2 )  
   { 
   	arr[i]=1; 
   }

  for(ll i=3;i<=n;i+=2) 
  {
      if( arr[i]==0 )  
      {
          for(ll j=i; j <= n; j+=i ) 
          { 
          	arr[j]+=1; 
          }
      }
  }
  ll temp;
  for(ll i=2; i <= n; ++i )  
  {   
  	  temp=arr[i];
      hash_arr[temp]+=1;
  }
    while(q--)
    {
    	cin>>pos;
    	if(pos>n)
    	   cout<<"0\n";
    	else
    	   cout<<hash_arr[pos]<<"\n";
    }
  

}
