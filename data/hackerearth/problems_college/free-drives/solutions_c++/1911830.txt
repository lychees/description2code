#include<iostream>
using namespace std;
int main()
{
	std::ios::sync_with_stdio(false);
unsigned long	int t;
	
 std::cin>>t;
	while(t--)
	{
	
unsigned long	int n,sum=0;
	std::cin>>n;
	
	for(unsigned long int i=0;i<n;i++)
{
	sum=sum+n-i-1;
	
	}	
	
	std::cout<<sum<<endl;
	
}
	return 0;
}
