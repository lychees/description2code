#include <bits/stdc++.h>
using namespace std;
int gcd(int a,int b){
	   if(b==0){
	   	  return a;
	   }
	   else{
	   	return gcd(b,a%b);
	   }
	}
int main()
{
    int t;
    cin>>t;
    while(t--){
    	int x,y,c;
    	cin>>x>>y>>c;
    	if((c-x)>0&&(c-y)>0&&c<(x+y)){
    		c=c*c-(c-x)*(c-x)-(c-y)*(c-y);
    		x=2*x*y;
    		int g=gcd(c,x);
    		c/=g;x/=g;
    		cout<<c<<"/"<<x<<endl;
    	}
    	else if((c-x)>=0&&(c-y)<=0){
    	  c=c*c-(c-x)*(c-x);
    		x=2*x*y;
    		int g=gcd(c,x);
    		c/=g;x/=g;
    		cout<<c<<"/"<<x<<endl;	
    	}
    	else if((c-y)>=0&&(c-x)<=0){
    		c=c*c-(c-y)*(c-y);
    		x=2*x*y;
    		int g=gcd(c,x);
    		c/=g;x/=g;
    		cout<<c<<"/"<<x<<endl;
    	}
    	else if((c-x)<=0&&(c-y)<=0){
    		c=c*c;
    		x=2*x*y;
    		int g=gcd(c,x);
    		c/=g;x/=g;
    		cout<<c<<"/"<<x<<endl;
    	}else{
    		cout<<"1/1"<<endl;
    	}
    }
    return 0;
}
