#include<stdio.h>

void gcd(int p,int q);
int nofdiv(int n);
int main()
{
    int cases;
    scanf("%d",&cases);
    while(cases--)
    {
        int studs,pos,divsr,prob=0;
        scanf("%d%d",&studs,&pos);
        prob += studs-pos+1;       

        divsr = nofdiv(pos-1);
        prob += (pos-1)-divsr;
        gcd(prob,studs);
    }
    return 0;
}
void gcd(int p,int q)
{
    int p1=p;
    int q1=q;
    int rem=q%p;
    while(rem)
    {
        q = p;
        p = rem;
        rem = q%p;
    }
    printf("%d/%d\n",p1/p,q1/p);
}

int nofdiv(int n)
{
    int i,count=0;
    for(i=1; i*i<=n; i++)
        if(n%i == 0)
            count += 2;
    if((i-1)*(i-1) == n)
        count--;
    return count;
}
