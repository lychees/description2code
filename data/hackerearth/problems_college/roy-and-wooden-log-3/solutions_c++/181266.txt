#include <cstdio>


int main(int argc, char** argv){
	int cases;
	scanf("%d",&cases);

	for(int i=0; i<cases; i++){
		long long n;
		scanf("%lld ", &n);

		if(n%2==0){ // n is evena
			//Total  possible cases: n/2+1
			//Min cuts: n/2-1
			printf("%lld %lld\n",n/2+1,n/2-1);

		} else { // n is odd
			//Total possible cases: (n+1)/2
			//Min cuts: (n-1)/2
			printf("%lld %lld\n",(n+1)/2,(n-1)/2);
		}

	}


}
