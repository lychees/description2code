#include<iostream>
using namespace std;
int main()
{
    int T;
    cin>>T;
    while(T--)
    {
        int N,M,i;
        cin>>N>>M;
        for(i=1;i<=N;i++)
        {
            int min=1000000,index;
            //THIS CODE BELONGS TO anurag_MANIT
            for(int j=1;j<=M;j++)
            {
                int d[3];
                cin>>d[0];
                cin>>d[1];
                cin>>d[2];
                int basePrice=1000000;
                for(int k=0;k<3;k++)
                {
                    basePrice = (basePrice/100)*(100-d[k]);
                }
                //cout<<basePrice<<endl;
                if(min>basePrice)
                {
                    index=j;
                    min=basePrice;
                }
            }
            cout<<index<<" ";
        }
        cout<<endl;
    }
}
