#include<iostream>
int gcd(int a,int b)
{
	if(b==0)
        return  a;
        return gcd(b,a%b);
 }
using namespace std;
int main()
{
	int t,x,mx,n;
	cin>>t;
	while(t--)
	{
		mx=0;
		cin>>n;
		while(n--)
		{
			 cin>>x;
			mx=gcd(mx,x);
			 
		}
		cout<<mx<<endl;
	}
	return 0;
}
