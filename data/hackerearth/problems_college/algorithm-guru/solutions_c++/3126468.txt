#include<bits/stdc++.h>
#define pb push_back
#define mp make_pair
#define f first
#define maxn 10100
#define s second
#define ll long long int
#define inf 1000000014
#define infl (ll)(1e18+1)
#define mod 1000000007
#define sz(x) (int) x.size()
using namespace  std;
int arr[maxn];
int dp[maxn];
int main(int argc, char const *argv[])
{
   int t,n,sum;
   cin>>t;
   while(t--)
    {
      memset(dp,0,sizeof dp);
    cin>>n>>sum;
  for(int i=1;i<=n;i++)
    cin>>arr[i];
  for(int i=1;i<=n;i++)
  {
    int cur=arr[i];
    dp[0]=1;
    for(int j=0;j<=3000;j++)
    {
        if(dp[j]==1)
        {
          for(int x=1;x*cur+j<=3000;x++)
          {
              dp[x*cur+j]=1;
          }
        }
    }
  }

  for(int i=sum;i>=0;i--)
  {
    if(dp[i]==1)
      {cout<<i<<"\n";break;}
    }
  }
  return 0;
}
