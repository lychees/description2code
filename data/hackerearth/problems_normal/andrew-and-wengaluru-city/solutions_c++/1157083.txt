#include <iostream>

#include <stdio.h>
using namespace std;
int main()
{
    int n,i,j,t;
    scanf("%d",&t);
    while(t--){
    	scanf("%d",&n);
    	int a[n];
    	long long int sum=0;
    	for(i=0;i<n;i++)
    		scanf("%d",&a[i]);
    	int maxl[n];
    	
    	int maxr[n];
    	maxl[0]=a[0];
    	maxr[n-1]=a[n-1];
    	
    	for(i=1;i<n;i++){
    		maxl[i]=max(maxl[i-1],a[i]);
    	}
    	
    	for(i=n-2;i>=0;i--){
    		
    		maxr[i]=max(maxr[i+1],a[i]);
    	}
    	
    	for(i=1;i<n-1;i++){
    		if(a[i]<maxl[i] && a[i]<maxr[i]){
    			
    		 	sum = sum + (min(maxl[i],maxr[i])-a[i]);
    		 	sum%=1000000007;
    		
    		}
    	}
    	
    
    	printf("%d\n",sum);
    }
    return 0;
}
