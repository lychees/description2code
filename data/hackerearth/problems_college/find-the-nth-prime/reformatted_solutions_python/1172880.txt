import math
prime = []
n = 8000
b = int(math.sqrt(n))
flag = [True]*n
i = 2
j = 4
prime.append(i)
while j < n:
    flag[j] = False
    j += i

i = 3
while i <= b:
    if flag[i]:
        j = i*i
        prime.append(i)
        while j < n:
            flag[j] = False
            j += i
    i += 2


for i in range(b+1, n):
    if flag[i]:
        prime.append(i)

n = input()
print prime[n-1]
