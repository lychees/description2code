    #include <iostream>
    using namespace std;
     
    int main()
    {
        int t,flag=0;cin>>t;
        while(flag<t)
        {  long long int r,k;
            long long int i,j,x,y,z,ans=0;
            cin>>r>>k;long long int arr[r];
            for(i=1;i<=r;i++)
            {   
            	x=i;y=0;z=0;
            	while(x!=0)
            	{
            		arr[y]=x%2;
            		x=x/2;
            		y++;
            		
            	}
             for(j=0;j<y-2;j++)
             {
             	if((arr[j]+arr[j+1]*10+arr[j+2]*100)==101)
             	 z++;
             	
             }
            	if(z>=k)ans++;
          	 
            	
            }
            cout<<ans<<endl;
        	
        	flag++;
        }
        return 0;
    }
