}

}
ll cntDis ( ll a, ll b, ll q ) {
    ll w= ( b+1-a ) /q;
    ll r= ( b+1-a ) %q;
    if ( b%q<r ) w++;
    return w;
}

int main() {
    ios_base::sync_with_stdio ( false );

        int k,t;
        cin>>k>>t;
        vi pr ( k );
        cin>>pr;
//         cout<<pr<<endl;
        vpii tests;
        reu ( i,1,1<<k ) {
            int prod=1;
            int j=i;
            bool f=0;
            int l=0;
            int cnt=0;
            while ( j ) {
                if ( j&1 ) {
//                     cout<<l<<" "<<pr[l]<<endl;
                    if ( INF/pr[l]<prod ) {
                        f=1;
                       
                        break;
                    }
                     cnt++;
                    prod*=pr[l];
                }
                j>>=1;
                l++;
            }
            int par=1;
            if ( cnt%2==0 ) par=-1;
            if ( !f ) tests.push_back(mp(prod,par));
        }
        rep(i,t){
        int a,b ; cin>>a>>b;
        int divs=0;
//         cout<<tests<<endl;
        for(pii p: tests){
        int no=cntDis(a,b,p.first);
        divs+=p.second*no;
            
        }
        cout<<divs<<'\n';    
        }
        
    
}
