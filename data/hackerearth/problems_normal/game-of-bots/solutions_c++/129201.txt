#include <iostream>
#include <string>
using namespace std;

bool isPossible(string first, string second) {
	bool returnVal = true;
	int firstIndex = 0;
	//int secondIndex = 0;
	int startSecond = -1;
	while(firstIndex < first.length()) {
		while(firstIndex < first.length() && first.at(firstIndex) == '#')
			firstIndex++;

		if(firstIndex >= first.length())
			break;

		if(first.at(firstIndex) == 'A') {
			startSecond = second.find('A', startSecond+1);
			if(startSecond > firstIndex){
				returnVal = false;
				break;
			}
		}
		else if(first.at(firstIndex) == 'B') {
			startSecond = second.find('B', startSecond+1);
			if(startSecond < firstIndex){
				returnVal = false;
				break;
			}
		}
		firstIndex++;
	}
	return returnVal;
}

int main()
{
	int t;
	cin>>t;
	string first, second;
	while(t--)
	{
		cin>>first>>second;
		//cout<<first<<second;
		if(isPossible(first, second))
			cout<<"Yes"<<endl;
		else
			cout<<"No"<<endl;
	}
	return 0;
}
