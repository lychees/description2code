#include <iostream>
using namespace std;

int main()
{
    
    
    
    int n,t,s;
    int a[100005];
    cin>>t;
    while(t--)
    {
        cin>>n;
        
            for(int i=0;i<n;i++)
            {
                cin>>a[i];
            }
    
    cin>>s;
    int flag;
    for(int i=0;i<(1<<n);i++)
         {
           
            int sum=0;
             flag=0;
            for(int j=0;j<n;j++)
            {
                if(i&1<<j)
                {
                    sum+=a[j];
                }
            }
            
            if(sum==s)
            {
                    flag=1;
                
                break;
            }
            else
            {
                 flag=0;
                
            }

            
         } 

         if(flag)
                cout<<"YES"<<endl;
            else
                cout<<"NO"<<endl;
            
    }
    return 0;
}
