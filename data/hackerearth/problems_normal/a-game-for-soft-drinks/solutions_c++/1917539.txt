#include <iostream>
#define MAX 101
using namespace std;
char mat[MAX][MAX][MAX];
void init()
{
	for(int i=0;i<MAX;i++){
		for(int j=0;j<MAX;j++){
			for(int k=0;k<MAX;k++){
				if((i>0 && mat[i-1][j][k]=='L') || (j>0 && mat[i][j-1][k]=='L') || (k>0 && mat[i][j][k-1]=='L') ||(i>0 && j>0 && mat[i-1][j-1][k]=='L')||(k>0 && j>0 && mat[i][j-1][k-1]=='L')||(i>0 && k>0 && mat[i-1][j][k-1]=='L') || (i>0 && j>0 && k>0 && mat[i-1][j-1][k-1] == 'L')){
					mat[i][j][k]='W';
				}
				else{
					mat[i][j][k] = 'L';
				}
			}
		}
	}
}

int main()
{
    init();
    int T;
    cin>>T;
    int a ,  b ,c;
    while(T--){
    	cin>>a>>b>>c;
    	if(mat[a][b][c]=='W'){
    		cout<<"Ashima"<<endl;
    	}
    	else cout<<"Aishwarya"<<endl;
    }
    return 0;
}
