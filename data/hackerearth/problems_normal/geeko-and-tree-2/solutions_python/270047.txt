# Compute a**b
def power(a, b):
 result = a
 for i in range(1, b):
  result *= a
 return(result)

t = raw_input()
for i in range(0, int(t)):
 input0 = raw_input()
 k, n = input0.split()
 k1 = int(k)
 n1 = int(n)
 nodes0 = power(k1, n1 + 1) - 1
 nodes0 = nodes0 / (k1 - 1) 

 ans = 0
 while(nodes0 > 0):
  ans = ans + nodes0 % 10
  nodes0 = nodes0 / 10

 #print("%d %d %d" %(k1, n1, nodes0))
 print("%d" %(ans))
