#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll dp[100][100];
int main(){
	dp[0][0]=0;
	dp[0][1]=1;
	dp[1][0]=1;
	dp[1][1]=0;
	for(int i=2;i<=90;i++){
		dp[i][1]=dp[i-1][0];
		dp[i][0]=dp[i-1][1]+dp[i-1][0];
	}
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		cout<<dp[n][1]<<" "<<dp[n][0]<<endl;
	}
	return 0;
}
