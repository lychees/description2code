#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;
int x1=0,x2=0,b[100][100],n,a[100][100];

int dfs(int a1,int a2)
{
    if(a1==n-1 && a2==n-1){x1=1;}
    if(a1+1<n && a[a1+1][a2]==1 && b[a1+1][a2]==0){b[a1+1][a2]=1;x2=0;dfs(a1+1,a2);}else x2--;
    if(a2+1<n && a[a1][a2+1]==1 && b[a1][a2+1]==0){b[a1][a2+1]=1;x2=0;dfs(a1,a2+1);} else x2--;
    if(x1==1){return 1;}
    if(x2==-2){return -1;}
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int i,j,x=0,y=0,z=0;
        cin>>n;
        for(i=0;i<n;i++)
            for(j=0;j<n;j++){cin>>a[i][j];b[i][j]=0;}
         z=dfs(0,0);
        if(z==1){cout<<"POSSIBLE"<<endl;}
        else {cout<<"NOT POSSIBLE"<<endl;}
        x1=0;x2=0;for(i=0;i<n;i++)for(j=0;j<n;j++){a[i][j]=0;b[i][j]=0;}
    }
    return 0;
}
