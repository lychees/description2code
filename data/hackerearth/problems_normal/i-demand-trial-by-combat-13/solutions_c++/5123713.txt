#include <iostream>
using namespace std;

int main()
{
    long int t,n,A[10000],l,m,i,j,k,B[10000],count,count2,C[10000]={0},p=0;;
    
    cin>>t;
    for(i=0;i<t;i++)
    {   count2=0;
    	cin>>n;
    	cin>>m;
    	for(j=0;j<n;j++)
    	{
    		cin>>A[j];
    	}
    	
    	for(j=0;j<n;j++)
    	{    
    		if(p==n)
    		if(C[j]==A[j])
    		count2++;
    		else 
    		break;
    		else
    		break;
    	}
    	
    	
    	
    	
    	if(count2==n)
    	{
    		for(j=0;j<n;j++)
    		cout<<B[j]<<" ";
    		cout<<endl;
    		continue;
    	}
    	else
    	{
    		for(j=0;j<n;j++)
    		C[j]=A[j];
    		p=n;
    	}
    	
    	for(k=0;k<m;k++)
    	{
    		for(j=0;j<n;j++)
    	   {
    	   	count=0;
    	   	
    	   	for(l=0;l<n;l++)
    	   	if(A[l]==0)
    	   	count++;
    	   	else
    	   	break;
    	   	
    	   	if(count==n)
    	   	{
    	   		break;
    	   	}
    	   	
    		if(j==0)
    		{
    			if(A[j+1]==1)
    			{
    				B[j]=1;
    			}
    			else
    			B[j]=0;
    			
    		}
    		
    		else
    		if(j==n-1)
    		{
    			if(A[j-1]==1)
    			{
    				B[j]=1;
    			}
    			else
    			B[j]=0;
    			
    		}
    		
    		else
    		{
    			
    			if(A[j-1]==1&&A[j+1]==1)
    			{
    				B[j]=1;
    			}
    			else
    			B[j]=0;
    			
    		}
    		
    		
    	}
    	
    	if(count==n)
    	   	break;
    	   	
    	for(j=0;j<n;j++)
    	A[j]=B[j];
    		
    	}
    	
    	
    	for(j=0;j<n;j++)
    	{
    		cout<<A[j]<<" ";
    	}
    	cout<<endl;
    }
}
