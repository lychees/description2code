#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,m,tmp;
		scanf("%d %d",&n,&m);
		vector<int> v;
		for(int i=0;i<n;++i)
		{
			scanf("%d",&tmp);
			v.push_back(tmp);
		}
		vector<int> prev,prev2;
		for(int i=1;i<=m;++i)
		{
			vector<int> ans;
			ans.push_back(v[1]);
			for(int j=1;j<n-1;++j)
			{
				if(v[j-1] && v[j+1])
					ans.push_back(1);
				else
					ans.push_back(0);
			}
			ans.push_back(v[n-2]);
			v=ans;
			if(ans==prev)
				break;
			else if(ans==prev2)
			{
				int y=m-i;
				if(y%2==0)
					break;
				else
				{
					v=prev;
					break;
				}
			}
			prev2=prev;
			prev=ans;
		}
		for(int i=0;i<n;++i)
			cout<<v[i]<<" ";
		cout<<endl;
	}
	return 0;
}
