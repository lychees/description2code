#include <iostream>
using namespace std;

int main()
{
    int t,h,n,i;
 	cin>>t;
 	while (t--)
 	{
 		cin>>n;
 		int *a = new int[n];
 		cin>>h;
 		for (i=0;i<n;i++)
 			cin>>a[i];
 		int min=10000;
 		int x=0,y=0;
 		for (i=0;i<n;i++)
 		{
 			x=a[i]-h;
 			if (x>=0)
 				y=x;
 			else
 				y=h-a[i];
 			if (y<min)
 				min=y;
 		}
 		cout<<min<<endl;
 	}
    return 0;
}
