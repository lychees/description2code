int main()
{
    //freopen("input.txt","r",stdin);
    //freopen("output.txt","w",stdout);
    //std::ios::sync_with_stdio(false);
    int T,x1,x2,y1,y2,X1,X2,Y1,Y2;
    bool rec,line,point,cek[1005][1005];
    scanf("%d",&T);
    rep (t,1,T) {
        scanf("%d %d %d %d",&x1,&y1,&x2,&y2);
        scanf("%d %d %d %d",&X1,&Y1,&X2,&Y2);
        rec=line=point=false;
        Set(cek,false);
        x1++, y1++;
        X1++, Y1++;
        rep (i,x1,x2) rep (j,y1,y2) cek[i][j]=true;
        rep (i,X1,X2) rep (j,Y1,Y2) {
            if (cek[i][j]) rec=true;
            else if (cek[i-1][j] || cek[i+1][j] || cek[i][j-1] || cek[i][j+1]) line=true;
            else if (cek[i-1][j-1] || cek[i+1][j-1] || cek[i-1][j+1] || cek[i+1][j+1]) point=true;
        }
        if (rec) printf("Rectangle\n");
        else if (line) printf("Line\n");
        else if (point) printf("Point\n");
        else printf("Apart\n");
    }
    return 0;
}
