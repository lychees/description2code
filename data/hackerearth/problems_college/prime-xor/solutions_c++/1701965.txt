#include <bits/stdc++.h>
using namespace std;
int prime[1000012];
#define ll long long
void sieve()
{
    int i,j;
    for(i = 2; i <= 1000010; i++) {
        if(prime[i] == 1)
            for(j = 2; j*i <= 1000010; j++) {
                prime[i*j] = 0;
            }
    }
    for(i = 1; i <= 1000010; i++) {
        prime[i] += prime[i-1];
    }
}
int main() 
{
    int t,l,r,i;
    for(i = 0; i <= 1000010; i++) {
        prime[i] = 1;
    }
    prime[0] = prime[1] = 0;
    sieve();
    cin>>t;
    while(t--) {
        cin>>l>>r;
        if(l > r) swap(l,r);
        if(l > 2) {
            puts("0");
            continue;
        }
        if(l == r && l == 1) {
        	puts("0");
        	continue;
        }
        cout<<prime[r]-prime[l-1]-1<<endl;
    }
	return 0;
}
