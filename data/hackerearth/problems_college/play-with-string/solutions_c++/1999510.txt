#include <iostream>
using namespace std;

int main(){
    int t;
    cin>>t;
    while(t--){
    	string a;
    	cin>>a;
    	int l=a.size();
    	int i=0,j=l-1,f=0;
    	while(i<=j){
    		int t1=abs((int)(a[i]-a[i+1]));
    		int t2=abs((int)(a[j]-a[j-1]));
    		if(t1!=t2){
    			f=1;
    			break;
    		}
    		i++;j--;
    	}
    	if(f)cout<<"Not Funny\n";
    	else cout<<"Funny\n";
    	
    }
    return 0;
}
