f = [0]*1000
dp = [0]*1000
def pre():
	f[0] = 1
	f[1] = 1
	for i in range(998):
		f[i+2] = f[i]+f[i+1]
def count():
	dp[0] = 1
	for i in range(999):
		dp[i+1] = dp[i]
		for j in range(i+1):
			if(f[i+1]%f[j]==0):
				dp[i+1]+=1
			if(f[j]%f[i+1]==0):
				dp[i+1]+=1
		dp[i+1] += 1
pre()
count()
t = int(raw_input())
for i in range(t):
	n = int(raw_input())
	print dp[n-1]
