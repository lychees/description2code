#include<iostream>
using namespace std;
int main()
{
	long int n;
	cin>>n;
	int num[n];
	for(int i=0;i<n;i++)
	{
		cin>>num[i];
		}
		int mul=2;
		int curr=num[0];
		cout<<curr<<" ";
		for(int i=1;i<n;i++)
		{
			int temp=num[i]*mul;
			cout<<temp-curr<<" ";
			int res=(temp-curr);
			curr+=res;
			mul++;
			}
			return 0;
}
