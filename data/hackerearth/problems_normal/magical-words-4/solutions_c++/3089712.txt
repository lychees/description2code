#include<bits/stdc++.h>
using namespace std;
long long int palin(string str,int l,int r)
{
long long 	int n = str.length();
	long long int sum =0;
	while(l>=0 && r<n && str[l] == str[r])
	{
		sum = sum + (r-l+1)*(r-l+1);
		l--;
		r++;
	}
	return sum;
}
int main()
{
	int t;
    scanf ("%d",&t);
	while(t--)
	{
		string str;
		cin>>str;
		long long int sum=0;
		for(int i = 0;i<str.length();i++)
		{
			sum += palin(str,i,i+1) +  palin(str,i,i);
		}
		cout<<sum<<"\n";
	}
}
