#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,i,j,ans;
    	cin>>n;
    	int a[n][3];
    	for(i=0;i<n;i++)
    	{
    		for(j=0;j<3;j++)
    		{
    			cin>>a[i][j];
    		}
    	}
    	int d[n][3];
    	for(i=0;i<3;i++)
    	{
    		d[0][i]=a[0][i];
    	}
    	for(i=1;i<n;i++)
    	{
    		d[i][0]=min(d[i-1][1],d[i-1][2])+a[i][0];
    		d[i][1]=min(d[i-1][2],d[i-1][0])+a[i][1];
    		d[i][2]=min(d[i-1][0],d[i-1][1])+a[i][2];
    		//cout<<d[i][0]<<" "<<d[i][1]<<" "<<d[i][2]<<endl;
    	}
    	ans=min(min(d[n-1][0],d[n-1][1]),d[n-1][2]);
    	cout<<ans<<endl;
    }
}
