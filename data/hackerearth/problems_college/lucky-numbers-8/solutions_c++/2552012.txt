long long max1(long long c,long long d)
{return (c>d?c:d);}
 
struct node{
	int s,e;
};
 
 
bool comp(struct node less,struct node more)
{
	return less.e<more.e;	
}
 
long long int gcd(long long int a,long long int b)
{
	if(a==0)
	return b;
	else
	return gcd(b%a,b);
}
 
int main()
{
	long long int t;
	scanf("%lld",&t);
	for(;t>0;t--)
	{
		long long int n;
		scanf("%lld",&n);
		
		long long int a[n];
		for(int i=0;i<n;i++)
		scanf("%lld",&a[i]);
		
		long long int max=INT_MIN;
		int c=0;
		for(int i=n-1;i>=0;i--)
		{
			if(a[i]>max)
			{
				max=a[i];
				c++;
			}
		}		
		
		printf("%lld\n",c);
	}
		
	return 0;
}
