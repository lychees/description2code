#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    long long int t,n,r,h,i,j,k,l,l1;
    cin>>t;
    while(t--){
    	cin>>n;
    	long long int dp[n];
    	vector<pair<long long int,long long int> > v;
    	for(i=0;i<n;i++){
    		cin>>r>>h;
    		v.push_back(make_pair(r,h));
    		dp[i]=0;
    	}
    	sort(v.begin(),v.end());
    	dp[0]=v[0].second;
    	for(i=1;i<v.size();i++){
    		k=0;
    		set<long long int>s;
    		s.insert(v[i].first);
    		l=v[i].second;
    		for(j=i-1;j>=0;j--){
    			if(l>v[j].second ){
    				l1=s.size();
    				s.insert(v[j].first);
    				l1=s.size()-l1;
    				if(dp[j]>k && l1>0){
    					k=dp[j];
    				}
    			}
    		}
    		dp[i]=v[i].second+k;
    	}
    	k=0;
    	for(i=0;i<n;i++){
    		if(dp[i]>k){
    			k=dp[i];
    		}
    	}
    	cout<<k<<endl;
    }
    return 0;
}
