#include <bits/stdc++.h>
using namespace std;
#define MOD 1000000007;
long long pow_fast(long long base, long long ex){
	//long long ans=1;
	long long res=1;
	while(ex>0){
		if(ex%2==1)res=(base*res)%MOD;
		base=(base*base)%MOD;
		ex>>=1;
	}
	return res%MOD;
}
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--){
    	long long n;
    	cin>>n;
    	long long ans=pow_fast(2,n)-1;
    	cout<<ans<<endl;
    }
    return 0;
}
