

    #include <bits/stdc++.h>
    using namespace std;
    # define MOD 1000000007
    char message[100005];
    int main()
    {
    	scanf("%s", message);
    	int semicolons = 0, temp = 0, ans = 0;
    	for (int i = 0; message[i] != '\0'; ++i)
    	{
    		if(message[i] == ';')
    		{
    			semicolons++;
    			ans = (ans + temp)%MOD;
    		}
    		else
    			temp = (2ll*temp + semicolons)%MOD;
    	}
    	printf("%d\n", ans);
    	return 0;
    }
