#include <bits/stdc++.h>
using namespace std;

int a[1000006];
int main()
{
    int T,n,k;
    cin >> T;
    while(T--){
    	cin >> n >> k;
    	for(int i=0;i<n;i++){
    		cin >> a[i];
    	}
    	sort(a,a+n);
    	int mn =100000054;
    	int mx =-1;
    	int first=0,last = n-1;
    	while(first<last){
    		int k = a[first++]+a[last--];
    		mn=min(k,mn);
    		mx=max(k,mx);
    	}
    	cout << mx-mn << endl;
    	if(mx-mn > k)
    		cout << "No more girlfriends!" << endl;
    	else if(mx-mn == k)
    		cout << "Lucky chap!" << endl;
    	else
    		cout << "Chick magnet Jhool!" << endl;
    }
    return 0;
}
