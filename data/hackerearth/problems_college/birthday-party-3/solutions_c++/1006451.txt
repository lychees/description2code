#include <iostream>
#include <bits/stdc++.h>
#define MOD 179424673
using namespace std;

int dp[1001][1001][2];
int main()
{
    int t,p,q,mx,my;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d %d %d %d",&p,&q,&mx,&my);
    	dp[0][0][0]=1;
    	dp[0][0][1]=1;
    	for(int i=0;i<=p;i++)
    	{
    		for(int j=0;j<=q;++j)
    		{
    			for(int k=1;k<=i && k<=mx;++k) dp[i][j][0]=(dp[i][j][0]+dp[i-k][j][1])%MOD;
    			for(int k=1;k<=j && k<=my;++k) dp[i][j][1]=(dp[i][j][1]+dp[i][j-k][0])%MOD;
    		}
    	}
    	int ans=(dp[p][q][0]+dp[p][q][1])%MOD;
    	printf("%d\n",ans);
    	memset(dp,0,sizeof(dp));
    }
    return 0;
}
