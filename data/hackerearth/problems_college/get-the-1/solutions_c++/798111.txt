#include <iostream>
using namespace std;

int main()
{
    int T;
    cin>>T;
    for(int i=0; i<T; i++)
    {
    	int a ;
    	cin>>a;
    	int c=0;
    	for(int j=0; j<1000; j++)
    	{
    		int b=0;
    		while(a>0)
    		{
    			b+=(a%10)*(a%10);
    			a=a/10;
    		}
    		a=b;
    		if(a==1){c++; break;}
    	}
    	if(c==0)
    	{cout<<"Case #"<<i+1<<":NO\n";}
    	else
    	{cout<<"Case #"<<i+1<<":YES\n";}
    	
    }
    return 0;
}
