#!/bin/env python
import sys
stdin = sys.stdin


def find_flip_sum(numbers):
    n1 = list(numbers[0])
    n2 = list(numbers[1])
    n1.reverse()
    n2.reverse()
    # print n1,n2
    a = ''.join(str(x) for x in n1)
    b = ''.join(str(x) for x in n2)
    # print a,b
    x = int(a) + int(b)
    y = str(x)
    y = y[::-1]
    return int(y)+0


t = int(stdin.readline().strip())
for i in range(t):

    numbers = [str(x) for x in raw_input().split()]
    print find_flip_sum(numbers)
