#include <bits/stdc++.h>

using namespace std;

#define gu getchar_unlocked()
#define ga getchar()
#define ll long long int
#define MOD 1000003

inline int scan()
{
    int n = 0,ch = ga,s = 0;
    while (ch < 48 || ch > 57) {
        if (ch == '-') s = 1;
        ch = ga;
    }
    while (ch >= 48 && ch <= 57)
    {
        n = (n<<3) + (n<<1) + ch - 48;
        ch = ga;
    }
    if (s == 0)
        return n;
    else
        return -1*n;
}

int main()
{
    int n, m;
    n = scan();
    m = scan();
    ll s1 = 0, s2 = 0;
    while (n--) {
        s1 += scan();
    }
    while (m--)
        s2 += scan();
    printf("%lld\n", (s1*s2));

    return 0;
}
