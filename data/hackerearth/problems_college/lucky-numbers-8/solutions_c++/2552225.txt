#include <iostream>
using namespace std;

int main()
{
    int tc,n,i,c,t,j;
    cin>>tc;
    while(tc--)
    {
    	c=1;
    	cin>>n;
    	int a[n];
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	for(i=0;i<n-1;i++)
    	{
    		t=1;
    		for(j=i+1;j<n;j++)
    		{
    			if(a[i]<=a[j])
    			{
    				t=0;
    				break;
    			}
    		}
    		if(t==1)
    		c++;
    	}
    	cout<<c<<endl;
    }
    return 0;
}
