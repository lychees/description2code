#include <iostream>
#include <string.h>
using namespace std;

int main()
{
    int t,n,ans[26],i,j,c,max,maxj;
    cin >> t;
    char s[105],ans1;
    while (t--) {
        cin >> n;
        int a[n][26];
        memset(ans,0,sizeof(ans));
        memset(a,0,sizeof(a));
        for (i = 0; i < n; i++) {
            cin >> s;
            for (j = 0; j < strlen(s); j++) {
                c = s[j] - 'a';
                if (a[i][c] == 0) {
                    a[i][c] = 1;
                }
            }
            for (j = 0; j < 26; j++) {
                ans[j] = ans[j] + a[i][j];
              //  cout << ans[j] << " ";
            }
        }
        max = 0;
        maxj = 0;
        for (j = 0; j < 26; j++) {
            if (ans[j] > max) {
                max = ans[j];
                maxj = j;
            }
        }
        ans1 = (char)(maxj + 97);
        cout << ans1 <<  endl;
    }

    return 0;
}
