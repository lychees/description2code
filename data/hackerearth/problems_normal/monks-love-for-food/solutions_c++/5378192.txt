#include <stdio.h>
#include<stack>

inline void perform_action(int q)
{
	std::stack<int> mystack;
	int x;
	register int c=getchar_unlocked();
	while(c>0)
	{
		x=0;
		for(;c<=47 || c>=58;c=getchar_unlocked());
		for(;c>47 && c<58;c=getchar_unlocked())
		{
			x=(x<<3)+(x<<1)+c-48;
		}
		if(x==1)
		{
			if(mystack.empty())
				printf("No Food\n");
				else
				{
					printf("%d\n",mystack.top());
					mystack.pop();
				}
				
		}
		else
		if(x==2)
		{
			x=0;
			for(;c<=47 || c>=58;c=getchar_unlocked());
			for(;c>47 && c<58;c=getchar_unlocked())
			{
				x=(x<<3)+(x<<1)+c-48;
			}
			mystack.push(x);
		}
	}
}


int main()
{
	
	int q;
	scanf("%d",&q);
	perform_action(q);
    return 0;
}
