#include<bits/stdc++.h>
using namespace std;
struct name
{
	string player;
	int g,s,b;
};
bool iscompare(struct name a,struct name bb)
{
	if(a.g>bb.g)
	{
		return a.g;
	}
	else
	if(a.g == bb.g && a.s > bb.s)
	{
		return a.s;
	}
	else
	if(a.g == bb.g && a.s == bb.s && a.b>bb.b)
	{
		return a.b;
	}
	else
	if(a.g == bb.g && a.s == bb.s && a.b==bb.b)
	{
		if(a.player >bb.player)
		{
			return false;
		}
		else
		{
			return true;
		}
	}
	return false;
}
int main()
{
	int n,i,j;
	cin>>n;
	map<string,name>map1;
	map<string,name>::iterator it;
	for(i=0;i<n;i++)
	{
		string A,B,C;
		cin>>A>>B>>C;
		map1[A].g++;
		map1[B].s++;
		map1[C].b++;
	}
	vector<name>vec;
	struct name naam[101];
	i=0;
	for(it = map1.begin();it!=map1.end();++it)
	{
		naam[i].player = (it->first);
		naam[i].g = it->second.g;
		naam[i].s = it->second.s;
		naam[i].b = it->second.b;
		i++;
	}
	sort(naam,naam+i,iscompare);
	for(j=0;j<i;j++)
	{
		cout<<naam[j].player<<" "<<naam[j].g<<" "<<naam[j].s<<" "<<naam[j].b<<"\n";
	}
}
