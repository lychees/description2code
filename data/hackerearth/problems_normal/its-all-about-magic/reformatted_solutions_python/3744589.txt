def touch_digit(touches):
    zero_times = 0
    one_times = 1
    for t in range(0, touches):
        tmp = zero_times
        zero_times += one_times
        one_times = tmp
    print str(one_times) + " " + str(zero_times)


times = int(raw_input())
for i in range(0, times):
    t = int(raw_input())
    touch_digit(t)
