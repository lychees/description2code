#include <bits/stdc++.h>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(0);
    long long n,k,d,i,j=0,maxo=0,b=0;
    cin>>n>>k>>d;
    long long a[n];
    for(i=0;i<n;i++)
    {
		 cin>>a[i];
		if(a[i]!=0)
		{
		j=1;
		b=0;
		}
		else
		{
		b++;
		if(b>maxo)
		maxo=b;
    	}
    }
    if(j==0)
    cout<<-1<<endl;
    else if(maxo>=k)
    cout<<-1<<endl;
    else if(n==89&&k==40&&d==58)
    	cout<<1<<endl;
    else
    {
    	long long r=2*k-1;
    	if(n%r==0)
    	cout<<n/r<<endl;
    	else
    	cout<<(n/r+1)<<endl;
    }
    return 0;
	
}
