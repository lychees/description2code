#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n,t,w;
    cin>>t;
    while(t--){
        cin>>n>>w;
        if(n>60){
            cout<<"NA\n";
            continue;
        }
        int fare=50;
        n-=2;
        if(n>0){
            fare+=n*12;
        }
        fare+=2*w;
        cout<<fare<<endl;
    }
    return 0;
}
