#include <stdio.h>
#include <iostream>
#include <string>
#include <map>
using namespace std;
string str,str1;
map<string,int> mp;
int main()
{
	int cnt=0,i,j;
    cin>>str;
    cin>>str1;
    for(i=0;i<str.size();i++)
    {
    	for(j=i;j<str.size();j++)
    	{
    		//substring from i to j
    		if(str.substr (i,j-i+1).find(str1)!=string::npos && mp.find(str.substr(i,j-i+1))==mp.end())
    		{
    		
    			cnt+=1;
    			mp[str.substr(i,j-i+1)]=1;
    		}
    		
    	}
    	cout<<endl;
    }
    
    printf("%d\n",cnt);
    return 0;
}
