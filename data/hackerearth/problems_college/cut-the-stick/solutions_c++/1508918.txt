int main() {
	int T; cin >> T;
	while (T--) {
		int S, N, L;
		cin >> S >> N >> L;
		L++;
		vector<int> a(N);
		for (int i = 0; i < N; i++)
			scanf("%d", &a[i]);
		sort(a.begin(), a.end());
		ll sum = N * (N - 1) / 2;
		for (int i = 0; i < N && a[i] < L; i++) {
			int l = upper_bound(a.begin(), a.end(), S - L) - a.begin();
			l = max(l, i + 1);
			int r = lower_bound(a.begin(), a.end(), a[i] + L) - a.begin();
			r = max(r, i + 1);
			sum -= max(0, r - l);
		}
		cout << sum << endl;
	}
}
