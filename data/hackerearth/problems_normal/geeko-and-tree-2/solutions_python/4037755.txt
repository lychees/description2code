import sys
def getAns(n, k):
  cur = k
  ans = 1
  ret = 0
  while (n>0):
    ans = ans+cur
#print ans,cur
    cur = cur*k
    n -= 1
#print ans
  while ans:
    ret = ret + ans%10
    ans = ans/10
  return ret
 
def main():
  data = sys.stdin
  t = int(data.readline())
  while t > 0:
    ln = data.readline()
    x = ln.split()
    print getAns(int(x[1]), int(x[0]))
    t -= 1
 
if __name__ ==  "__main__":
  main()
