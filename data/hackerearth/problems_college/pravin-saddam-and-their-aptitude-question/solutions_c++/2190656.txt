#include<iostream>
using namespace std;

long long int factorial(long long int);

int main(void) {
	long long int number,n,m=0,nn,i;
	cin>>n;
	while(n)
	{
		n--;
		m=0;
	cin >> number>>nn;
	for(i=number;i<=nn;i++)
	   m+=factorial(i);
       
		cout <<m%10<<endl;
	}
	return 0;
}

long long int factorial(long long int number) {
	long long int temp;

	if(number <= 1) return 1;

	temp = number * factorial(number - 1);
	return temp;
}
