{
	j=0;

	{

		{
			a[j++]=i;
			a[j++]=num/i;
		}
    	//cout << i << num/i << endl;
		if(num % i == 0&&i*i==num)
		a[j++]=i;
   		//cout << i<<"\n";
   	}
}
