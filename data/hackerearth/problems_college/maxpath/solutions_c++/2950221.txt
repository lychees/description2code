#include<iostream>
#include<cstdio>
#include<cmath>

#define FOR(i,j,k) for(i=j;i<k;i++)
#define MOD 1000000007
#define s(x) scanf("%d",&x)

typedef long long ll;
using namespace std;

int max(int a,int b)
{
	if(a>=b)
	return a;
	else 
	return b;
}

int main()
{
	int n,a[100][100],i,j,flag=0;
	cin>>n;
	
	for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
		cin>>a[i][j];
	}
	
	int sum=a[0][0],x=0,y=0;
	
	sum+=a[0][1];
	x=0,y=1;
	
	while(flag!=1)
	{
	if(x!=y+1&&y+1>=n&&x+1<n)
	{
	sum+=a[x+1][y];
	x=x+1;
	}
	
	else if(y!=x+1&&x+1>=n&&y+1<n)
	{
	sum+=a[x][y+1];
	y=y+1;
	}
	
	else
	{
	sum+=max(a[x+1][y],a[x][y+1]);
	if (max(a[x+1][y],a[x][y+1])==a[x+1][y])
	x=x+1;
	else
	y=y+1;
	}
	
	if(x==y)
	flag=1;
	}
	
	while(flag!=0)
	{
	if(x!=y-1&&y-1>=n&&x-1<n)
	{
	sum+=a[x-1][y];
	x=x-1;
	}
	
	else if(y!=x-1&&x-1>=n&&y-1<n)
	{
	sum+=a[x][y-1];
	y=y-1;
	}
	
	else
	{
	sum+=max(a[x-1][y],a[x][y-1]);
	if (max(a[x-1][y],a[x][y-1])==a[x-1][y])
	x=x-1;
	else
	y=y-1;
	}
	
	if(x==y&&x==0)
	flag=0;
	}
	
	cout<<sum;
}
