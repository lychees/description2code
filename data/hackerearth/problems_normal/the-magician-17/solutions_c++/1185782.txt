#include <bits/stdc++.h>
using namespace std;

int main() {
	int T;
	cin >> T;
	for(int t =0; t < T; t++) {
		int N,K,ans =-1,r =1,off =0, o =0;
		cin >> N >> K;
		K--;
		while(ans < 0) {
			int p =(o == 0)?(N-N/2):(N/2);
			if(K < p) ans =(1<<r)*K+off+o*(1<<(r-1));
			else K -=p;
			if(o == 0) off +=1<<(r-1);
			if(N&1) o =1-o;
			N -=p;
			r++;}
		cout << ans+1 << endl;}
	}
