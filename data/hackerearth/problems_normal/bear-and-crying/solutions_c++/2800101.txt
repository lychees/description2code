#include<bits/stdc++.h>
#include<stdio.h>
#include<string.h>
using namespace std;
#define mod 1000000007
int main()
{
    char s[100002];
    scanf("%s",s);
   long long int n,i,semi=0,count=0,finalvalue=0;
    n=strlen(s);
    for(i=0;i<n;i++)
    {
        if(s[i]==';')
           {
               semi++;
            finalvalue=(finalvalue+count)%mod;}
        else
        {
            count=(2*count+semi)%mod;
        }

    }
    printf("%d",finalvalue);
}
