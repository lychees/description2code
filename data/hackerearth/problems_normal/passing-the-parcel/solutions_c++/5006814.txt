#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    string s;
    cin>>s;
    int i,j=0;
    vector <int> v;
    vector <int> :: iterator it;
    for(i=0;i<n;i++)
    {
    	v.push_back(i+1);
    }
    i=0;
    int l=n;
    int len=s.length();
    while(l!=1)
    {
    	if(j>=len)
    	{
    		j=0;
    	}
    	if(i>=l)
    	{
    		i=0;
    	}
    	if(s[j]=='a')
    	{
    	    i++;	
    	}
    	else if(s[j]=='b')
    	{
    		v.erase(v.begin()+i);
    	   // i++;
    	    l--;
    	}
    	j++;
    	
    }
   // for(int k=0;k<v.size();k++)
   // {cout<<v[k]<<endl;}
    cout<<v[0]<<endl;
    return 0;
}
