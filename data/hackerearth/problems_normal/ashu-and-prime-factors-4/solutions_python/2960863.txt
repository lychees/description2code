from __future__ import division
from bisect import bisect


C = 1000000
lst = [0]*(C + 1)
lst[0] = 0
lst[1] = 1

ans = {}

for i in range(2, C + 1):
	if lst[i] == 0:
		ans[i] = 1
		lst[i] = i
		j = i*i
		while j <= C:
			if lst[j] == 0:
				ans[i] += 1
				lst[j] = i
			j += i
