#include<bits/stdc++.h>
using namespace std;
#define lld long long int
int main()
{
	
    int n;
    cin>>n;
    int i;
    lld B[101];
    for(i=0;i<n;i++)
    	cin>>B[i];
    
    lld A[101];
    A[0]=B[0];
   
    lld sum=A[0];
	for(i=2;i<n;i++)
     {A[i-1]=(lld)i * B[i-1]-sum;sum=sum+A[i-1];}
	A[i-1]=(lld)i*B[i-1]-sum;
	for(i=0;i<n;i++)
		cout<<A[i]<<" ";

	return 0;
}
