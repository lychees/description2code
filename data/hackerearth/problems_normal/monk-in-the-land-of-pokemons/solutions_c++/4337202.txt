#include<bits/stdc++.h>
using namespace std;

int item[1000001];

void initialise()
{
	for(int i=0;i<1000001;i++)
		item[i]=0;
}

int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		initialise();
		int n,food,pokemon,cnt=0;
		cin >> n;
		for(int i=0;i<n;i++)
		{
			cin >> food >> pokemon;
			item[food]++;
			if(item[pokemon]==0)
			{
				cnt++;
			}
			else
			{
				item[pokemon]--;
			}
		}
		cout << cnt << endl;
	}
	return 0;
}
