#include <bits/stdc++.h>
using namespace std;
int a[110][110],dp[110][110];
int main()
{
    int t,n,m;
    cin>>t;
    while(t--){
    	cin>>n>>m;
    	
    	for(int i=0;i<n;i++){
    		for(int j=0;j<m;j++){
    			cin>>a[i][j];
    		}
    	}
    	
    	
    	for(int i=0;i<n;i++){
    		for(int j=0;j<m;j++){
    			dp[i][j]=-1;
    		}
    	}
    	dp[0][0]=1;
    	for(int i=1;i<n;i++){
    		if(dp[i-1][0]!=-1){
    				if((a[i-1][0]<a[i][0])){
    					dp[i][0]=1+(dp[i-1][0]);
    				}
    				else{
    					dp[i][0]=-1;
    				}
    		}
    		
    	}
    	for(int i=1;i<m;i++){
    		if(dp[0][i-1]!=-1){
    				if((a[0][i-1]<a[0][i])){
    					dp[0][i]=1+(dp[0][i-1]);
    				}
    				else{
    					dp[0][i]=-1;//(d[i][j-1]);
    				}
    		}
    	}
    	for(int i=1;i<n;i++){
    		for(int j=1;j<m;j++){
    				if(dp[i][j-1]==-1 && dp[i-1][j]==-1){
    					dp[i][j]=-1;
    				}
    				else if(dp[i][j-1]!=-1 && dp[i-1][j]==-1){
    					if((a[i][j-1]<a[i][j]))
	    				  	dp[i][j]=1+dp[i][j-1];
	    				else
	    					dp[i][j]=-1;
    				}
    				else if(dp[i][j-1]==-1 && dp[i-1][j]!=-1){
    					if((a[i-1][j]<a[i][j]))
	    				  	dp[i][j]=1+dp[i-1][j];
	    				else
	    					dp[i][j]=-1;
    				}
    				else {
	    				  /*if((a[i][j-1]<a[i][j])&&(a[i-1][j]<a[i][j]))
	    					dp[i][j]=1+max(dp[i][j-1],dp[i-1][j]);
	    				  else if((a[i][j-1]<a[i][j]))
	    				  	dp[i][j]=1+dp[i][j-1];
	    				  else
	    				  	dp[i][j]=1+dp[i-1][j];*/
	    				  	if(a[i][j-1]<a[i][j])dp[i][j]=dp[i][j-1]+1;
						    else dp[i][j]=-1;
						    if(a[i-1][j]<a[i][j])dp[i][j]=max(dp[i][j],dp[i-1][j]+1);
    				}
    		}
    	}//
    	int mx=0;
    	for(int i=0;i<n;i++){
    		for(int j=0;j<m;j++){
    			mx=max(mx,dp[i][j]);
    			//cout<<dp[i][j]<<" \n"[j==m-1];
    		}
    	}
    	cout<<mx<<"\n";
    }
    return 0;
}
