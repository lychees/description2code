#include <iostream>
#include<algorithm>
using namespace std;


int max(int a,int b)
{
	if(a>b)return a;
	else return b;
}
int main()
{
    
    int t,m=-10000;
    cin>>t;
    
    while(t--)
    {
    	int a,b,c;
    	cin>>a>>b>>c;
    	
    	m=max(m,3*a+1*b+c*(-1));
    }
    cout<<m<<endl;
    return 0;
}
