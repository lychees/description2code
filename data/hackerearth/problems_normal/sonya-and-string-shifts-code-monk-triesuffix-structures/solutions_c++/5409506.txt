typedef long long ll;
typedef pair<int,int> pii;
typedef vector<int> vi;
typedef vector<pii> vpii;

int n,q;
int a[1123123],dp[1123123];
char s[2123123];

bool cmp(const int&x,const int&y){
  for(int i=0;i<n;i++){
    if(s[x+i]<s[y+i])
      return 1;
    if(s[x+i]>s[y+i])
      return 0;
  }
  return 1;
}

int main(){
  ios::sync_with_stdio(0);
  cin.tie(0),cout.tie(0);

  cin>>n;
  cin>>(1+s);
  for(int i=1;i<=n;i++)
    s[i+n]=s[i];s[1+2*n]=0;
  
  for(int i=1;i<=n;i++)
    a[i]=i;

  sort(a+1,a+n+1,cmp);

  for(int i=1;i<=n;i++)
    dp[a[i]-1]=i-1;

  for(int i=1;i<n;i++)
    dp[i]=min(dp[i],dp[i-1]);

  cin>>q;

  while(q--){
    int x;cin>>x;
    cout<<a[dp[x]+1]-1<<endl;
  }
  
  return 0;
};
