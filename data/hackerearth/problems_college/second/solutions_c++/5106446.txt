#include <bits/stdc++.h>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int n,m,i,x;
   scanf("%d%d",&n,&m);
   set<int> s;
   for(i=0;i<n;i++)
   {
   	scanf("%d",&x);
   	s.insert(x);
   }
   int ans = 0;
   while(1)
   {
   	if(s.find(ans+m)==s.end())break;
   	ans = ans+m;
   }
   printf("%d\n",ans);
    return 0;
}
