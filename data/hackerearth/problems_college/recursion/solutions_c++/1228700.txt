#include<iostream>
#include<stdio.h>
#include<vector>
#include<algorithm>
using namespace std;

vector<int>ans;

int a[20],b[20];
int n;

void swap(int *a,int *b)
{
    int temp=*a; *a=*b; *b=temp;
}

void permute(int i)
{
    int j;
    if(i>n)
    {
        ans.push_back(b[n]);
        return;
    }
    for(j=i;j<=n;j++)
    {
        b[i]=(b[i-1]*10)+a[j];
        swap(&a[i],&a[j]);
        permute(i+1);
        swap(&a[i],&a[j]);
    }
}


int main()
{
    int t,i,j;
    scanf("%d",&t);
    while(t--)
    {
        ans.resize(0);

        scanf("%d",&n);
        for(i=1;i<=n;i++)
            a[i]=i;
        b[0]=0;
        permute(1);

        sort(ans.begin(),ans.end());

        for(i=0;i<ans.size();i++)
            cout<<ans[i]<<" ";

        printf("\n");
    }



    return 0;


}
