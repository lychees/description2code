#include <bits/stdc++.h>

using namespace std;

const int me = 1000025;

int n, sz = 1, ans;
string s;
int trie[26][me];
int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);

    cin >> n;
    for(int i = 0; i < n; i ++){
        cin >> s;
        int c = 0;
        for(int j = 0; j < (int)s.size(); j ++){
            int x = (int)s[j] - 'a';
            if(trie[x][c] == 0){
                ans ++;
                trie[x][c] = sz ++;
            }
            c = trie[x][c];
        }
    }
    cout << ans + 1<< endl;


    return 0;
}
