#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
int main()
{
    long long int dp[10002][12];
    memset(dp,0,sizeof(dp));
    for (int i = 1; i < 10002; i++)
    {
    	dp[i][1] = 1;
    }
    for (int i = 1; i < 12; i++)
    {
    	dp[1][i] = 1;
    }
    for (int i = 2; i < 10002; i++)
    {
    	for (int j = 2; j < i && j < 12; j++)
    	{
    		int k = i-1;
    		int ctr = j;
    		while (ctr--)
    		{
    			dp[i][j] += dp[k][j];
    			dp[i][j] %= mod;
    			k--;
    		}
    	}
    	long long int ans = dp[i][min(i-1,11)]+1;
    	for (int j = i; j < 12; j++)
    	{
    		dp[i][j] = ans;
    	}
    }
  /*	for(int i=0;i<20;i++) {
		for(int j=0;j<10;j++) {
			cout<<dp[i][j]<<" ";
		}
		cout<<"\n";
	}*/
    int t;
    scanf("%d",&t);
    while (t--)
    {
    	int n,p;
    	cin >> n >> p;
    	cout << dp[n][p] << endl;
    }
    return 0;
}
