#include <iostream>
#include <string>
#include <vector>

using namespace std;

int main(){
	int T,check;
	string s1,s2;
	cin >> T;
	while(T--){
		cin >> s1 >> s2;
		vector<int> a(26),b(26);
		for(int i=0;i<s1.size();i++){
			a[s1[i]-'a']++;
		}
		for(int i=0;i<s2.size();i++){
			b[s2[i]-'a']++;
		}
		check=0;
		for(int i=0;i<26;i++){
			if(a[i]==b[i])	check++;
		}
		if(check==26)	cout<<"Yes"<<endl;
		else	cout<<"No"<<endl;

	}
	return 0;
}
