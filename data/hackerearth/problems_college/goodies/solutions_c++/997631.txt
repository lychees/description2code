#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    long t,n,i,count;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	count=0;
    	long *arr= new long[n];
    	for(i=0;i<n;i++)
    	{
    		cin>>arr[i];
    	}
    	sort(arr,&arr[n]);
    /*		for(i=0;i<n;i++)
    	{
    		cout<<arr[i]<<" ";
    	}*/
    	int *ans= new int[n];
//    	ans[0]=10001;
    	int k=-1,flag;
    	for(i=n-1;i>=0;i--)
    	{
    		flag=1;
    		for(int j=0;j<=k;j++)
    		{
    			if(arr[i]<ans[j])
    			{
    				ans[j]=arr[i];    				
    				flag=0;
    				break;
    			}
    		}
    		if(flag==1)
    		{
    			k++;
    			ans[k]=arr[i];
    		}
    	}
    	cout<<k+1<<endl;
    }
    return 0;
}
