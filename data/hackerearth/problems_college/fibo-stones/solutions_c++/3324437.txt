#include<iostream>
#include<stdlib.h>
using namespace std;
int main(){
 long long int n,k,sum=0,sum2=0,i,x;
 cin>>n>>k;
  long long int a[n];
  a[1]=1;
  a[2]=1;
  
  for(i=3;i<=n;i++){
  	a[i]=a[i-1]+a[i-2];
  	sum=sum+a[i];
  }
  sum=sum+2;
  for(i=0;i<k;i++){
  	cin>>x;
  	sum2=sum2+x;
  }
  cout<<sum-sum2<<"\n";
}
