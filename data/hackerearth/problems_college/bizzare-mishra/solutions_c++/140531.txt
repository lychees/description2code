#include<cstdio>
#include<cstring>
int main()
{
	int t,n;
	char s[10005];
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		scanf("%s",s);
		//int len=strlen(s);
		int i,j;
		int arr[10005];
		for(i=0;i<n;i++)
		{
			if(s[i]=='Y')
			{
				arr[i]=1;
			}
			else
				arr[i]=0;
		}
		int min1=0;
		int min2=0;
		for(i=0;i<n;i++)
		{
			if(i%2==0)
			{
				min1=min1+(arr[i]^0);
			}
			else
				min1=min1+(arr[i]^1);
		}
		
		for(i=0;i<n;i++)
		{
			if(i%2==0)
			{
				min2=min2+(arr[i]^1);
			}
			else
				min2=min2+(arr[i]^0);
		}
		printf("%d\n",(min1<min2)?min1:min2);
	}
	return 0;
}
