#include<iostream>
using namespace std;
struct node
{
	int c,p,f;
};
int func(int gp,int gc,int gf,int visit[],int N,node food[])
{
	if(gc==0&&gp==0&&gf==0)
	{
		return 1;
	}
	if(gc<0||gp<0||gf<0)return 0;
	int ans,i;	
	for(i=1;i<=N;i++)
	{
		if(visit[i]==0)
		{
			visit[i]=1;
			
			ans=func(gp-food[i].p,gc-food[i].c,gf-food[i].f,visit,N,food);
			
			if(ans==1)return 1;
			
			visit[i]=0;
		}
	}
	
	return 0;
}
int main()
{
//	freopen("new_years_resolution.txt","r",stdin);
//	freopen("output.txt","w",stdout);
	int t,gc,gp,gf,ans;
	node food[30];
	int visit[30],N,i,T;
	scanf("%d",&T);
	
	for(t=1;t<=T;t++)
	{
		scanf("%d%d%d",&gp,&gc,&gf);
		
		scanf("%d",&N);
		
		for(i=1;i<=N;i++)
		{
			visit[i]=0;
			scanf("%d%d%d",&food[i].p,&food[i].c,&food[i].f);
		}
		
		ans=func(gp,gc,gf,visit,N,food);
		
		if(ans==1)
		printf("Case #%d: yes\n",t);
		else printf("Case #%d: no\n",t);
	}
//	fclose(stdout);
//	fclose(stdin);
	return 0;
}
