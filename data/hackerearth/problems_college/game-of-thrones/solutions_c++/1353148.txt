#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
     /* Enter your code here. Read input from STDIN. Print output to STDOUT */   
     int t;
     cin >>t;
     while(t--)
     {
     	long long int n,a[100005],i,count=0;
     	cin >>n;
     	for(i=0;i<n;i++)
     	{
     		cin >>a[i];
     	}
     	sort(a,a+n);
     	for(i=0;i<n-1;i++)
     	{
     		if(a[i]==a[i+1])
     		count++;
     	}
     	cout << count <<endl;
     }
    return 0;
}
