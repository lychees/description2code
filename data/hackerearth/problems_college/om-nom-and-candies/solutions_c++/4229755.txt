#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main()
{
	ios::sync_with_stdio(false);
	cin.tie(NULL);

	int n, a[100005];
	cin >> n;
	long long ans[100005] = { };
	for(int i = 1; i <= n; i++)
	{
		cin >> a[i];
		ans[i] += a[i];
		cout << ans[i] << ' ';

		for(int j = 2*i; j <= n; j += i) ans[j] = max(ans[j], ans[i]);
	}

	return 0;
}
