#include <iostream>
#include <vector>
#include <math.h>
#include <limits.h>
#include <iomanip>
using namespace std;
double dist(int x1,int y1, int x2,int y2)
{
	double dif = pow(x1 - x2 , 2) + pow(y1 - y2, 2);
	return sqrt(dif);
}


int main()
{
	int n=0;
	cin>>n;
	
	vector<vector<int>> data(n,vector<int>(3,0));
	vector<double> dp(n,INT_MIN);
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<3;j++)
		{
			cin>>data[i][j];
			// cout<<data[i][j]<<" ";
		}
		// cout<<endl;
	}
	if(n==0)
	{
		cout<<0<<endl;
		return 0;
	}
	dp[0] = data[0][2];
	for(int i=1;i<n;i++)
	{
		// cout<<"for i="<<i<<endl;
		for(int j=0;j<i;j++)
		{
			// cout<<"for j="<<j<<endl;
			double a = dist(data[i][0],data[i][1],data[j][0],data[j][1]);
			// cout<<a<<endl;
			
			dp[i] = max(dp[i], dp[j] + data[i][2] - a );
			// cout<<dp[i]<<endl;
		}
	}
	cout << fixed << setprecision(6);
	cout<<dp[n-1]<<endl;
    // cout << "Hello World!" << endl;
    return 0;
}
