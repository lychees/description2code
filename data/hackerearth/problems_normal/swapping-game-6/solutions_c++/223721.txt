#include <iostream>
#include<cstdio>
#include<cstring>
#include<string>
using namespace std;

string swap(string s)
{
	int i;
	string s1="",s2="";
	for( i = 0 ; i < s.length(); i+=2)
	{
		s1 = s1 + s[i];
		if(i+1 < s.length())
			s2 = s[i+1] + s2;
	}
	s = s1+s2;
	return s;
}

int main()
{
    long long int k;
    string st;
    cin>>k;
    cin>>st;
    int lp=0;
    string tst = st;
    while(1)
    {
    	lp++;
    	tst = swap(tst);
    	if(st == tst)
    		break;
    	//cout<<tst<<endl;
    }
    lp = k%lp;
    for(int i = 0 ; i < lp; i++)
    	st = swap(st);
    cout<<st;
    return 0;
}
