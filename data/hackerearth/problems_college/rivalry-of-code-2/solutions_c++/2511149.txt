#include<bits/stdc++.h>
using namespace std;
int min(int i,int j)
{
	return (i<j?i:j);
}
int main()
{
   	int t;
   	cin>>t;
   	while(t--)
   	{
   		string str1,str2;
   		cin>>str1>>str2;
   		int l1=str1.size();
   		int l2=str2.size();
   		int mini=INT_MAX,count;
   		if(l1<=l2)
   		{
   			for(int i=0;i<=l2-l1;i++)
   			{
   				count=0;
   				for(int j=0;j<l1;j++)
   				{
   					if(str1[j]!=str2[j+i])
   					{
   						count++;
   					}
   				}
   				mini=min(mini,count);
   				
   			}
   		}
   		else
   		{
   			for(int i=0;i<=l1-l2;i++)
   			{
   				count=0;
   				for(int j=0;j<l2;j++)
   				{
   					if(str2[j]!=str1[j+i])
   					{
   						count++;
   					}
   				}
   				mini=min(mini,count);
   			}
   		}
   		cout<<mini<<endl;
   	}
}
