for _ in xrange(input()):
    a, b = map(int, raw_input().split())
    cnt = 0
    carry = 0
    while a and b:
        x = a % 10
        a /= 10

        y = b % 10
        b /= 10

        s = x+y
        if carry != 0:
            s += carry

        if s > 9:
            cnt += 1
            carry += (s)/10

    if cnt == 0:
        print "No carry operation"
    elif cnt == 1:
        print "1 carry operation"
    else:
        print "%d carry operations" % cnt
