
#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n;
    cin >> n;
    int total = 0;
    int arr[10000];
    for (int i = 0; i < n; ++i)
    {
        cin >> arr[i];
        total += arr[i];
    }
    if (total == 2*(n-1)) cout << "Yes" << endl; else cout << "No" << endl;
    return 0;
}
