#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int a[1000010];
unordered_map<int,int> mp;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	memset(a,0,sizeof(a));
	mp.clear();
	a[0]=1;a[1]=1;//Not primes
	int n;
	cin>>n;
	//Find all primes till n
	int i,j;
	for(i=2;i<=n;i++)
	{
		if(a[i]==0)
		{
			for(j=2;(j*i)<=n;j++)
			{
				a[j*i]++;
			}
			a[i]++; //Since every prime number is divisible by itself
		}
	}
	for(i=2;i<=n;i++)
	{
		//~ cout<<i<<" "<<a[i]<<endl;
		if(mp.find(a[i])!=mp.end())
		mp[a[i]]++;
		else
		mp[a[i]]=1;
	}
	int q;
	cin>>q;
	for(i=0;i<q;i++)
	{
		cin>>j;
		if(mp.find(j)==mp.end())
		{
			cout<<0<<endl;
		}
		else
		{
			cout<<mp[j]<<endl;
		}
	}
	return 0;
}
