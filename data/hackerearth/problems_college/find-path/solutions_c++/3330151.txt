#include <bits/stdc++.h>
#define ll long long int
using namespace std;

int t, n;
ll A[100005];

int main()
{
	cin >> t;
	while (t--) {
		cin >> n;
		for (int i = 0; i < n; i++) {
			cin >> A[i];
		}
		int cnt = 0;
		for (int i = 1; i < n; i++) {
			if (A[i - 1] >= A[i]) {
				cnt++;
			}
		}
		cout << cnt << endl;
	}	
	return 0;
}
