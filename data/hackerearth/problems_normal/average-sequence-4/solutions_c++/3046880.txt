#include<iostream>
using namespace std;
int main()
{
	int n;
	cin>>n;
	long long a[n],b[n];
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
	}
	for(int i=0;i<n;i++)
	{
		if(i==0)
		{
		
		cout<<a[i]<<" ";
		b[i]=a[i];
		}
		else
		{
		b[i]=(i+1)*a[i];
		for(int j=0;j<i;j++)
		{
			b[i]-=b[j];
		}
		cout<<b[i]<<" ";
		}
	
	}
	
	return 0;
}
