#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int n,k;

ll memo[61][61][61][2];

ll solve(int idx,bool ant,int act,int maxi){
    if(idx == n && maxi >= k) return 1;
    if(idx == n) return 0;
    
    if(memo[idx][act][maxi][ant] != -1) return memo[idx][act][maxi][ant];
    
    ll sol;
    if(ant){
        int maxii = max(maxi,act+1);
        sol  = solve(idx+1,1,act+1,maxii);
        sol += solve(idx+1,0,0,maxi);
    }else{
        int maxii = max(maxi,1);
        sol  = solve(idx+1,1,1,maxii);
        sol += solve(idx+1,0,0,maxi);
    }
    return memo[idx][act][maxi][ant] = sol;
}


int main(){

    while(cin>>n>>k){
        memset(memo,-1,sizeof memo);
        ll den = 1ll<<n;
        ll num = solve(0,0,0,0);
        while(num%2==0){
            den>>= 1;
            num>>= 1;
        }
        cout<<num<<"/"<<den<<endl;

    }
}
