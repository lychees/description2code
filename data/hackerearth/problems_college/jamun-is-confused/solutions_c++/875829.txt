#include <iostream>
using namespace std;

int main()
{
	int t,n,a[1000],s,i,temp;
    cin>>t;
    while(t)
    {
    	cin>>n;
    	cin>>a[0];
    	for(i=1;i<n;i++)
    	{
    		cin>>a[i];
    		while(a[i-1])
    		{
    			temp = a[i-1];
    			a[i-1]=a[i]%a[i-1];
    			a[i]=temp;
    		}
    	}
    	cout<<a[n-1]<<endl;
    	t--;
    }
    return 0;
}
