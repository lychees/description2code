#include <iostream>
#include <math.h>
#include <string.h>
using namespace std;

int main()
{
   int *a = new int[501],c=0;
   int count[501];
   memset(count,0,sizeof(count));
 
 for(int i=0;i<=500;i++)
      a[i]=-i;
      
   for(int i=2;i<=500;i++)
   {
   	 if(a[i]<0)
   	 {
    	for(int j=2*i;j<=500;j+=i)
   	      if(a[j]<0) a[j]=j;
   	   
   	    c++;
   	 }
   	 
   	  count[i]=c;
  }
  
   
   int n; cin>>n;
   
   long long int ans=0;
   int i=2;
   
   while(i<=n)
   	 { ans += pow(2,count[i]-1);  
   	    i++;
   	  }
   	  
 cout<<ans<<endl;  
    
    return 0;
}
