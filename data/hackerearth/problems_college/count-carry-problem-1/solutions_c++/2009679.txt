int main()
{
    int t;
    cin>>t;
    while(t--){
    string a,b;int c=0;
    cin>>a>>b;
for(int i=0;i<a.length();i++){
    if((int(a[i])+int(b[i])-96)>9){
        c++;
    }
}
if(c==0)
cout<<"No carry operation"<<endl;
else if(c==1)
cout<<c<<" carry operation"<<endl;
else

    }

}
