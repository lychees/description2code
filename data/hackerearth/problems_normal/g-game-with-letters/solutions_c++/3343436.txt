template<typename T> ostream& operator , (ostream& os, T& val){ return os << " " << val;}
template<typename T> ostream& operator >> (ostream& os, T& val){ return os << " " << val;}

using namespace std;

int main(){
	int T;
	cin >> T;
	while(T--){
		int n;
		string s;
		cin >> n >> s;
		vector<int> z(26, 0);

		for(char c: s){
			z[c-'a']++;
		}

		long long max_score = -1;
		int member = 0;
		int ans = -1;

		for(int i=0; i<n; i++){
			string tmp;
			cin >> tmp;

			long long score = 0;

			for(char c:tmp){
				score += z[c-'a'];
			}

			if(score > max_score){
				max_score = score;
				ans = i;
				member = tmp.size();
			}else if(score == max_score && member > tmp.size()){
				max_score = score;
				ans = i;
				member = tmp.size();
			}

		}

		cout << ans+1 << endl;

	}
	return 0;
}
