#include<stdio.h>
int ind;
int work(int arr[])
{
    int sum,wina,winb,i,j,k,t;
    sum=wina=winb=0;
    for(i=0;i<ind;i++)
        sum+=arr[i];
    if(sum==1)
        return 0;

    for(i=0;i<ind;i++)
    {
        t=arr[i];
        for(j=arr[i]-1;j>=0;j--)
        {
            arr[i]=j;
            k=work(arr);
            if(k==0)
                winb=1;
            else
                wina=1;
        }
        arr[i]=t;
    }
    if(wina==1&&winb==1)
        return 0;
    else if(wina==1)
        return 0;
    else
        return 1;
}
int main()
{
    int ntc,n,a,f,i,k;
    scanf("%d",&ntc);
    while(ntc--)
    {
        int arr[100001]={};
        ind=0;
        scanf("%d",&n);
        a=2;
        f=0;
        while(n%a==0)
        {
            f++;
            n=n/a;
        }
        if(f!=0)
        arr[ind++]=f;
        f=0;
        for(i=3;i*i<=n;i=i+2,f=0)
        {
            while(n%i==0)
            {
                f++;
                n=n/i;
            }
            if(f!=0)
            arr[ind++]=f;


        }

        if(n>1)
        {
        arr[ind++]=1;
        }


        k=work(arr);
        if(k==0)
            printf("Printf{}\n");
        else
            printf("Pappu Bhujia\n");
    }
}
