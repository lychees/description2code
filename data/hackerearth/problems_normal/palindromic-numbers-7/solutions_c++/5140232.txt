#include <iostream>
using namespace std;
bool palindrome(long int n)
{
	long int sum=0,no=n;
	while(no)
	{
		sum=sum*10+no%10;
		no=no/10;
	}
	if(sum==n)
	return true;
	else
	return false;
}
int main()
{
   long int a,b,no,count,i;
   cin>>no;
   while(no--)
   {
   	cin>>a>>b;count=0;
   	for(i=a;i<=b;i++)
   	{
   		if(palindrome(i))
   		count++;
   	}
   	cout<<count<<"\n";
   }
    return 0;
}
