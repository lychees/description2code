#include <iostream>
using namespace std;

int arr[1000001] = {0};
int saved[1000001] = {0};

int main()
{

    int n,m,l,r,i,j,q,x,count;
    cin>>n>>m;
    for(i = 0; i <m ; i++ )
    {
    	cin>>l>>r;
    	arr[l-1] ++;
    	arr[r]--;
    }

    for(i = 1; i <n ; i ++) arr[i] += arr[i-1];
    for(i = 0; i <n; i++ ) saved[arr[i]]++;
    for(i = 999999;i>=0;i--) saved[i] += saved[i+1];

    cin>>q;

    while(q--)
    {
    	cin>>x;
    	cout<<saved[x]<<endl;
    }
    return 0;
}
