#include <iostream>
using namespace std;
int primes[105]={0};
void sieve()
{
	primes[0]=primes[1]=1;
	for(int i=2;i*i<=100;i++)
	{
		if(!primes[i])
		{
			for(int j=i+i;j<=100;j+=i) primes[j]=1;
		}
	}
		
}
void compute(int n)
{
	int size=1;
	int a[1000]={0};
	a[0]=1;
	for(int i=2;i<=n;i++)
	{
		int carry=0;
		for(int j=0;j<size;j++)
		{
			int data=a[j]*i + carry;
			carry = data/10;
			data%=10;
			a[j]=data;
		}
		while(carry)
		{
			a[size++]=carry%10;
			carry/=10;
		}
	}
	for(int i=size-1;i>=0;i--)
		cout<<a[i];
		cout<<endl;
}
int main()
{
	sieve();
	int n;
	cin>>n;
	for(int i=2;i<=n;i++)
		if(primes[i]==0)
			compute(i);
	return 0;
}
