#include <iostream>
#include <algorithm>
#include <vector>
#include <utility>
#include <map>
using namespace std;

bool cmp(pair<string, int> a, pair<string, int> b)
{
	return a.second > b.second;
}

int main()
{
    map<string, int> mp;
    string name;
    int t;
    
    mp.clear();
    
    cin >> t;
    while (t--)
    {
    	cin >> name;
    	mp[name]++;
    }
    vector<pair<string, int> > vt(mp.begin(), mp.end());
    sort(vt.begin(), vt.end(), cmp);
    int max = vt[0].second;
    
    int ans = 0;
    for (int i = 0; i < vt.size(); ++i)
    {
    	if (vt[i].second < max)
    		break;
    	ans += vt[i].second;
    }
    cout << ans << endl;
    
    return 0;
}
