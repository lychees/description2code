#include <iostream>
using namespace std;

int main()
{
    int i;
    long long a[1000];
    a[0]=1;
    a[1]=2;
    a[2]=3;
    for(i=3;i<1000;i++)
    {
    	a[i]=a[i-1]+a[i-2];
    }
    int t;
    cin >> t;
    while(t--)
    {
    	int n;
    	cin >> n;
    	cout << a[n-1] << endl;
    }
    return 0;
}
