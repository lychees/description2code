#include <bits/stdc++.h>
using namespace std;

typedef long long LL;

#define PII       pair<int,int>
#define all(c)    c.begin(),c.end()
#define sz(c)     (int)c.size()
#define clr(c)    c.clear()
#define pb        push_back
#define mp        make_pair
#define cin(x)    scanf("%d",&x)
#define MOD       1000000007
#define EPS       1E-10

int m1 , m2;

int dp[1000000] = {0};

int solve(int i)
{
	if(i == 0) return 0;
	else if(dp[i] != -1) return dp[i];
	int f = 0;
	if(i >= m1)
		f |= (!solve(i-m1));
	if(i >= m2)
		f |= (!solve(i-m2));
	return (dp[i] = f);
}

void solve()
{
	int n;
	cin(n);
	cin(m1);
	cin(m2);
	for(int i = 0; i <= n; i++)
		dp[i] = -1;
	printf("%d\n", solve(n));
}

int main()
{
	int t;
	cin(t);
	while(t--)
		solve();
	return 0;
}
