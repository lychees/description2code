#include <algorithm>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>

using namespace std;
int A[10005], N, K;
int main() {
  cin >> N >> K;
  fill_n(A + 1, N - 1, 1);
  for (int i = 1; i <= K; ++i) {
    for (int j = 1; j <= N; A[j] += A[j + A[j]], ++j)
      printf("%d ", j + A[j]);
    printf("\n");
  }
  return 0;
}
