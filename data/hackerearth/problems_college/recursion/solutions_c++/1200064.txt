
#include <iostream>
#include <string.h>
using namespace std;


void shift(char *s, int i, int j){
	char a = s[j];
	while (i<j){
		s[j] = s[j - 1];
		j--;
	}
	s[j] = a;
}

void printCombinations(char *s, int i, int n)
{
	if (i == n)
		cout << s<<" ";
	char st[10];
	int j = 0;
	for (j = i; j <= n; j++){
		strcpy(st, s);
		//strcpy(st, s);
		shift(st, i, j);
		printCombinations(st, i + 1, n);
	}

}


int main()
{
	char s[] = { '1', '2', '3', '4', '5', '6', '7', '8', '9', '\0' };
	int t;
	cin >> t;
	int n;
	char c[10];
	while (t--){
		strcpy(c, s);
		cin >> n;
		c[n] = '\0';
		printCombinations(c, 0, n - 1);
		cout << "\n";
	}

	
}
