#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s1,s2;
    	cin>>s2;
    	int len=s2.length();
    	s1=s2;
    	reverse(s2.begin(),s2.end());
    	int flag=0;
    	for(int i=1;i<len;i++)
    	{
    		if(abs(s1[i]-s1[i-1])!=abs(s2[i]-s2[i-1]))
    		{
    			flag=1;
    			break;
    		}
    	}
    	if(flag==1)
    	{
    		cout<<"Not Funny"<<endl;
    	}
    	else
    	{
    		cout<<"Funny"<<endl;
    	}
    }
}
