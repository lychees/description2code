#include<bits/stdc++.h>
using namespace std;
#define ll long long int
ll BIT[1000005][26]={0};
ll n;
void update(ll x,int val,int alpha)
{
	for(;x<=n; x += x&-x)
	{
		BIT[x][alpha]+=val;
	}
}
ll query(ll x,int alpha)
{
	ll ans=0;
	for(;x>0;x -= x&-x)
	{
		ans+=BIT[x][alpha];
	}
	return ans;
}
int main()
{
	ll t;
	cin>>n>>t;
	string s;
	cin>>s;
	for(int i=1;i<=n;i++)
	{
		update(i,1,s[i-1]-'a');
	}
	while(t--)
	{
		int choice;
		cin>>choice;
		if(choice == 1)
		{
			ll x;
			char val;
			cin>>x>>val;
			update(x,1,val-'a');
			update(x,-1,s[x-1]-'a');
			s[x-1]=val;
		}
		if(choice == 2)
		{
			ll l,r,odd=0,pal;
			cin>>l>>r;
			for(int i=0;i<26;i++)
			{
				pal=query(r,i)-query(l-1,i);
				if(pal%2)
				{
					odd++;
				}
			}
			if(odd>1)
			{
				cout<<"no"<<endl;
			}
			else
			{
				cout<<"yes"<<endl;
			}
		}
	}
	return 0;
}
