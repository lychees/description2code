#include<iostream>
using namespace std;
int main()
{
    int t;
    ios::sync_with_stdio(false);
    cin>>t;
    int k;string str;
    while(t--)
    {
        cin>>k>>str;
        int n = str.length();
        int count =0;
        int i=0;int j;
        //cout<<n<<endl;
        while(i<n)
        {
            //cout<<"ASD"<<i<<' '<<str[i]-'a'+1<<endl;
            j=i;int sum = 0;
            while(j<n&&sum+(str[j]-'a'+1)<k)
            {
              //  cout<<"QWE"<<i<<' '<<j <<' '<<str[j]-'a'+1<<endl;
                sum+=(str[j]-'a'+1);
                j++;
            }
            //cout<<"QWEQWEQWE"<<' '<<j<<endl;
            if(j==n)
                break;
            sum+=(str[j]-'a'+1);

            //cout<<"SUM"<<' '<<sum<<endl;

            if(sum==k)
            {
                count++;
                sum-=str[i]-'a'+1;
                i++;
            }
            while(sum>k)
            {
                sum-=(str[i]-'a'+1);
                i++;
            }
            //cout<<"COUNT"<<' '<<count<<endl;
            if(sum==-92)
                break;
        }
        cout<<count<<endl;
    }
}
