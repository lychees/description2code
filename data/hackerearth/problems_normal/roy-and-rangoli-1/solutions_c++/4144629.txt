#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <cstring>
#include <iterator>
#include <stdio.h>
#define lld long long int
#define m 1000000007
#define pb(x) push_back(x)
#define mp(x,y) make_pair(x,y)
using namespace std;

int main()
{
	int n;
	lld ans=0;
	cin>>n;
	lld sum=2*(n-1);
	bool mark[sum+1];
	//int arr[sum+1];
	for(int i=0;i<=sum;i++)
		mark[i]=1;
	mark[0]=0;
	mark[1]=0;
	for(int i=2;i*i<=sum;i++)
	{
		if(mark[i])
		{
			for(int j=i*i;j<=sum;j=j+i)
			{
				mark[j]=0;
			}
		}
	}
    //cout << "Hello World!" << endl;
    for(int i=0;i<sum/2;i++)
    {
    	ans=(ans%m+(mark[i]==1?i+1:0)%m)%m;
    	ans=(ans%m+(mark[sum-i]==1?i+1:0)%m)%m;
    	
    	/*arr[i]=i+1;
    	arr[sum-i]=i+1;*/
    }
    ans=(ans%m+(mark[sum/2]==1?n:0)%m)%m;
    cout<<ans;
    return 0;
}
