#include<bits/stdc++.h>
using namespace std;
 vector<vector<int> > factors(1000001);

vector<int> b(1000001);
int  a[1000001]={0};
int prime[1000005];
int p=1;
void generate(int y)
    {int i,j;
     prime[0]=2;
	 for(i=3;i*i<=y;i+=2)
        {if(a[i])
          continue;
         prime[p++]=i;
		for(j=i*i;j<=y;j+=2*i)
           {a[j]=1;
            factors[j].push_back(i);
           }
        }
        for(i;i<=y;i+=2)
          if(!a[i])
            prime[p++]=i;
     //   for(i=0;i<p;i++)
             ;//cout<<prime[i]<<" ";
    }
    

  long long etof[1000005]={0};
 	
  long long ans[1000005]={0};
void etofgen()
 {
 int i,j,copy;
  ans[0]=0;
  ans[1]=0;
  prime[0]=2;
  etof[0]=etof[1]=1;
  int n,t;
 
  for(i=2;i<=1000000;i++)
     {copy=i;
      etof[i]=1;
     
   	   for(j=0;j<1000 && prime[j]<=min(copy,1000);j++)
	     {if(copy%prime[j]==0)
	         {int m=1;
			  while(copy%prime[j]==0)
	            {
	             copy/=prime[j];
	             m*=prime[j];
	            }
	            m/=prime[j];
	          etof[i]*=m*(prime[j]-1);
	          etof[i]*=etof[copy];
	          copy=1;
			         }
	     }
	    if(copy!=1)
	      {etof[i]=i-1;
	      }
	 
     }
   
 	
 }
int main()
 {
 	
 	int n,i,t;
 	generate(1000000);
 	etofgen();
    for(i=0;i<=10;i++)
      b[i]=0;
    
   // for(i=11;i<=100;i+=2)
     //cout<<i<<" "<<factors[i].size()<<endl;
    
	for(i=11;i<=1000000;i++)
     {
     	int d=etof[i];
     	int c=1-i%2,copy=i;
     	while(copy%2==0)
     	  copy/=2;
     	int copy2=copy;
     	for(int j=0;j<factors[copy2].size();j++)
     	 {
     	 	while(copy%factors[copy2][j]==0)
     	 	  copy/=factors[copy2][j];
     	 	c++;
     	 }
     	if(copy!=1)
     	 c++;
     	if(b[i-d]==0 && i-d>=10 || b[i-c]==0 && i-c>=10)
     	  b[i]=1;
     	//cout<<i<<" "<<c<<" "<<d<<endl;
     }
    cin>>t;
    while(t--)
      {
      	
      	cin>>n;
      	if(b[n]==0)
      	  cout<<"Bob"<<endl;
      	else
      	  cout<<"Alice"<<endl;
      }
 	
 }
