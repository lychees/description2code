#include <bits/stdc++.h>
using namespace std;
#define N 1010

int x, y, n, w, t;
double p, q, f[N][N];

int main() {
//	freopen("samu.in", "r", stdin);
	scanf("%d", &t);
	while (t --) {
		scanf("%d %d %d %d %lf %lf", &x, &y, &n, &w, &p, &q);
		p /= 100, q /= 100;
		memset(f, 0, sizeof f);
		f[0][0] = 1;
		for (int d = 1; d <= n; d ++) {
			for (int j = 0; j <= w; j ++)
				f[d][j] = max(f[d-1][max(0,j-x)] * p + f[d-1][j] * (1 - p),	f[d-1][max(0,j-y)] * q + f[d-1][j] * (1 - q));
		}
		printf("%.6lf\n", f[n][w] * 100);
	}
	return 0;
}
