#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,k,i,count=0;
    	cin>>n>>k;
    	int a[n];
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	for(i=0;i<n;i++)
    	{
    		if(a[i]<=0)
    		count++;
    	}
    	if(count<k)
    	cout<<"YES"<<endl;
    	else
    	cout<<"NO"<<endl;
    }
    return 0;
}
