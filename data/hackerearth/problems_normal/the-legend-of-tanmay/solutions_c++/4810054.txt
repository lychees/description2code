#include <stdio.h>
#include <vector>
#include <algorithm>
using namespace std;
long long proizvod(vector<int> niz){
	long long p=1;
	for(unsigned int i=0;i<niz.size();i++)p*=niz[i];
	return p;
}

int main()
{
    int t;
    scanf("%d", &t);
    for(int lolo=0;lolo<t;lolo++){
    	int n;
    	scanf("%d", &n);
    	vector<int> m,v;
    	int nule=0;
    	
    	for(int i=0;i<n;i++){
    		int temp;
    		scanf("%d", &temp);
    		if(temp<0)m.push_back(temp);
    		else if(temp>0)v.push_back(temp);
    		else nule++;
    	}
    	sort(m.begin(),m.end());
    	sort(v.begin(),v.end());
    	long long mini, maxi;
    	if(nule==n){
    		maxi=0;
    		mini=0;
    	}else if(m.size()==0){
    		if(nule!=0)mini=0;
    		else mini=v[0];
    		maxi=proizvod(v);
    	}else if(v.size()==0 && m.size()>1){
    		mini=proizvod(m);
    		maxi=proizvod(m);
    		if(mini>0)mini/=m.back();
    		if(maxi<0)maxi/=m.back();
    	}else if(v.size()==0 && m.size()==1){
    		mini=m[0];
    		if(nule!=0){
    			maxi=0;
    		}else maxi=mini;
		}else{
			mini=proizvod(m)*proizvod(v);
    		maxi=proizvod(m)*proizvod(v);
    		if(mini>0)mini/=m.back();
    		if(maxi<0)maxi/=m.back();
		}
		printf("%lli %lli\n",maxi,mini);
    }
    
    return 0;
}
