#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,z;
    long double x,y,s,s1;
    cin >> t;
    while(t--)
    {
    	cin >> n;
    	z=n;
    	s=0.000000,s1=0.000000;
    	while(n--)
    	{
    		cin >> x >> y;
    		s+=x;
    		s1+=y;
    	}
    	s=s/z;
    	s1=s1/z;
    	cout << "(" ;
    	cout << fixed << setprecision(2) << s;
    	cout << " , ";
    	cout << fixed << setprecision(2) << s1;
    	cout << ")\n";
    }
    return 0;
}
