#define REP(i,n) for(int i=0;i<n;i++)

#define sll(n) scanf("%lld",&n)

typedef long long LL;
#define PII pair<int,int>
using namespace std;
int main()
{
    int n;
    si(n);
    LL a[n];
    REP(i,n) sll(a[i]);
    LL ans=a[0];
    LL m=a[0];
    for(int i=1;i<n;i++)
    {
        m=max(m+a[i],a[i]);
        ans=max(ans,m);
    }
    if(ans>0) cout<<ans<<endl;
    else cout<<0<<endl;
	return 0;
}
