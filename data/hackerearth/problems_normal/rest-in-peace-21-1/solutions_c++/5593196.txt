#include <iostream>
#include <string>

using namespace std;

int main()
{
	int T, num;
	string str;
	bool flag;
	
	cin>>T;
    while(T>0){
    	cin>>str;
    	
    	flag = false;
    	if(str.find("21") != -1){
    		flag = true;
    	}else{
    		num = stoi(str);
    		if(num%21 == 0){
    			flag = true;
    		}
    	}
    	
    	if(flag){
    		cout<<"The streak is broken!"<<endl;
    	}else{
    		cout<<"The streak lives still in our heart!"<<endl;
    	}
    	T--;
    }
    return 0;
}
