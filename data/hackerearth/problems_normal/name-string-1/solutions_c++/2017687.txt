#include <iostream>
using namespace std;
#include<algorithm>
int main()
{
    long int t,a[3];
    cin>>t;
    while(t--)
    {
    	cin>>a[0]>>a[1]>>a[2];
    	sort(a,a+3);
        if(a[0]==0&&a[1]==0&&a[2]==0)
        cout<<"0\n";
        else if(a[2]>a[1])
        cout<<2*a[1]+1<<"\n";
        else if(a[2]>=a[1])
        cout<<2*a[1]<<"\n";
        else if(a[0]==a[1]&&a[1]==a[2])
        cout<<2*a[0]<<"\n";
        else continue;
    }
    return 0;
}
