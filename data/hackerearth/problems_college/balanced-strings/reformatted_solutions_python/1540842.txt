for _ in xrange(int(input())):
    s = raw_input()
    danger = 0
    d = set(s)
    c = set('abcdefghijklmnopqrstuvwxyz')
    new = sorted(c)
    p = s.count('a')
    if c.issubset(d):
        for i in new:
            if s.count(i) == p:
                danger += 1
    print("Yes" if (danger) == 26 else "No")
