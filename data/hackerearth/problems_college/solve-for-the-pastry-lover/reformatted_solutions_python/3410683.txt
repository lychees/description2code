# print 'Hello World!'
t = input()
while t > 0:
    t -= 1
    n = input()
    s = raw_input()
    m = []
    for i in range(n-1):
        d = 1
        for j in range(i+1, n):
            if s[i] == s[j]:
                m.append(d)
            else:
                d += 1

    if (len(m) == 0):
        print "-1"
    else:
        print min(m)
