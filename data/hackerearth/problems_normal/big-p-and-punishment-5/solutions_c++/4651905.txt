#include<bits/stdc++.h>
using namespace std;
vector<int>v[10005];
int visit[100005];
int ans;
void dfs(int k)
{
	visit[k]=1;
	ans++;
	for(int i=0;i<v[k].size();i++)
	{
		if(!visit[v[k][i]])
		{
			dfs(v[k][i]);
		}
	}
}
int main()
{
	int t;
	cin>>t;
	while(t--)
    {
    	ans=0;
    	int n,f,s;
    	cin>>n>>f>>s;
    	for(int i=1;i<=n;i++)
    	{
    		v[i].clear();
    	}
    	for(int i=0;i<f;i++)
    	{
    		int x,y;
    		cin>>x>>y;
    		v[x].push_back(y);
    	}
    	memset(visit,0,sizeof(visit));
    	for(int i=1;i<=s;i++)
    	{
    		int k;
    		cin>>k;
    		if(!visit[k])dfs(k);
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
