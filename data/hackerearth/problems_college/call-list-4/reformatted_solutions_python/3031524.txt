from sys import stdin
t = int(stdin.readline())
for tt in xrange(t):
    n = int(stdin.readline())
    a = []
    for i in xrange(n):
        x = stdin.readline().strip()
        a.append(x)
    valid = 1
    a.sort()
    for i in xrange(n-1):
        for j in xrange(i+1, i+2):
            if a[j].startswith(a[i]):
                valid = 0
                break
        if valid == 0:
            break
    if valid:
        print "YES"
    else:
        print "NO"
