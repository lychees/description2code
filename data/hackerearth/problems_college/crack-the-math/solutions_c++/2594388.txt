#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int t,i;
    
    cin>>t;
    while(t--){
    	string a,b,c;
    	cin>>a>>b;
    	for(i=0;i<a.size();i++){
    		if(a[i]==b[i]){
    			c[i]='0';
    		}else{
    			c[i]='1';
    		}
    	}
    	for(i=0;i<a.size();i++){
    		cout<<c[i];	
    	}
    	cout<<endl;
    }
    return 0;
}
