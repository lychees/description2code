//constants
const int mxn=1e5+1;
const int MOD=1e9+7;
using namespace std;
int par[mxn];
set<pair<int,int> > s;
void makeset()
{
	memset(par,-1,sizeof(par));
}
int find(int child)
{
	if(par[child]<0)
		return child;
	return par[child]=find(par[child]);
}
void join(int first,int second)
{
	int x=find(first);
	int y=find(second);
	if(x==y)
		return ;
	s.erase(mp(-par[x],x));
	s.erase(mp(-par[y],y));
	if(par[x]>par[y])
	{
		par[y]+=par[x];
		par[x]=y;
		s.insert(mp(-par[y],y));
	}
	else
	{
		par[x]+=par[y];
		par[y]=x;
		s.insert(mp(-par[x],x));
	}
}
int main()
{
	int t=1;
//	sc1(t);
	while(t--)
	{
		int n,q;
		cin>>n>>q;
		makeset();
		for(int i=1;i<=n;++i)
			s.insert(mp(1,i));
		while(q--)
		{
			int x,y;
			cin>>x>>y;
			join(x,y);
			cout<<(*s.rbegin()).first-(*(s.begin())).first<<"\n";
		}
		
	}
	return 0;
}
