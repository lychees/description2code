#include <bits/stdc++.h>
using namespace std;

long long int DP[10004];
int mod = 1000000007;
int main()
{
    int t; cin>>t;
    while(t--) {
    	int n, p, i;
    	cin>>n>>p;
    	memset(DP, 0, sizeof(DP));
    	int j;
    	DP[0] = 1;
    	for(i = 1; i <= n; i++) {
    		for(j = i-1; (j>=0)&&(i-j<= p); j--) {
    			DP[i] += DP[j];
    			DP[i] = (DP[i])%mod;
    		}
    	}
    	cout<<DP[n]<<endl;
    }
    return 0;
}
