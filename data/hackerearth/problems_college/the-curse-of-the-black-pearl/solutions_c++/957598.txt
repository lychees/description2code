#include<iostream>
#include<cstdio>
#include<string.h>

using namespace std;

int main(){
int t=882;
char a[32810];
while(t--){
        scanf("%s",a);
        int i,len=strlen(a),count=0;
        for(i=(len-1);i>=1;){
                while(a[i]=='0'&&i>=1){
                    count++;
                    i--;
                }
                if(i==0) break;
                while(a[i]=='1'&&i>=1){
                    count++;
                    i--;
                }
                count++;
                if(i==0){
                        count++;
                        break;
                }
                a[i]='1';
        }
        cout<<count<<"\n";
}
return 0;
}
