#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    for(int k=0;k<t;k++)
    {
    	long long int n;
    	int m=0;
    	cin >> n;
    	long long int a[n],b[n];
    	for(int h=0;h<n;h++)
    		cin >> a[h];
		for(int h=0;h<n;h++)
    		cin >> b[h];    	
    	for(int i=n-1;i>=0;i--)
    	{	
    		int l;
    		for(int j=i;j>=0;j--)
    		{
    			if(b[i]>=a[j])
    				{
    					l=i-j;
    				}
    			else
    				break;
    		}
    		if(l>m)
    			m=l;
    	}
    	cout <<  m <<endl;
    }
}
