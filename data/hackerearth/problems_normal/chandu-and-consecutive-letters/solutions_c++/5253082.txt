#include <iostream>
#include <string.h>
using namespace std;

int main()
{
   	int t;
   	cin>>t;
   	while(t--){
   		string s;
   		cin>>s;
   		char arr[31];
   		int count = 0;
   		arr[0] = s[0];
   		for(int i = 1; i < s.length(); i++){
   			if(s[i] == arr[count]){
   				
   			}
   			else{
   				arr[++count] = s[i];
   			}
   		}
   		for(int i = 0; i <= count; i++){
   			cout<<arr[i];
   		}
   		cout<<endl;
   	}
}
