#include <iostream>
#include<algorithm>
using namespace std;
int min(int a,int b)
{
	return (a>b)?b:a;
}
int f(int n)
{
	if(n==0)
	return 0;
	else if(n<0)
	return 100000000;
	else
	{
	 int r=1+min(f(n-10),f(n-12));
	return r;}
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int k=f(n);
		if(k>n)
		cout<<"-1"<<endl;
		else
		cout<<k<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
