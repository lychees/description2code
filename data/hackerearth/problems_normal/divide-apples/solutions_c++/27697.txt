typedef pair<int,int> PI;
typedef vector<int> VI;
typedef long long LL;
int arr[10001],n;
LL s=0;
LL sol(LL val)
{
   LL ans=0,curr=val;
   int i;
   for(i=1;i<=n;i++)
   {
      ans=ans+abs(curr);
      if(i<n)
      {
	 curr=s/n+curr-arr[i+1];
      }
   }
   return ans;
}
int main()
{
   int i;
   LL lo,hi,mid,ans;
   cin>>n;
   for(i=1;i<=n;i++)cin>>arr[i];
   for(i=1;i<=n;i++)s+=arr[i];
   lo=s*-1;hi=s;ans=-1;
   while(lo<=hi)
   {
      mid=(hi+lo)/2;
      if(sol(mid)>sol(mid+1))
      {
	 lo=mid+1;
      }
      else
      {
	 if(ans==-1||ans>sol(mid))ans=sol(mid);
	 hi=mid-1;
      }
   }
   cout<<ans<<endl;
   return 0;
}
