#include<bits/stdc++.h>
using namespace std;
bool fun(int a[],int n,int m,int c)
{
	 if(m==0)
	      return 1;
	 if(c==0&&m==0)
	     return 1;;
	  if((n<0||c==0)&&(m!=0))
	       return 0;
	       else return fun(a,n-1,m,c)||fun(a,n-1,m-a[n-1],c-1);
}
int main()
{
	int n,m;
	cin>>n;
	int a[n];
	int i;
	 for(i=0;i<n;i++)
	   cin>>a[i];
	   cin>>m;
	  if(fun(a,n,m,3))
	     cout<<"True";
	  else cout<<"False";
	  return 0;
}
