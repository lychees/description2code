#include<iostream>
#include<string.h>

using namespace std;

int main()
{
	int h,m,i;
	char a[][10]={"one","two","three","four","five","six","seven","eight","nine","ten","eleven","twelve","thirteen","fourteen","fifteen","sixteen","seventeen","eighteen","nineteen"};

	cin>>h>>m;
	
	if(m==0)
	{
		for(i=0;a[h-1][i]!='\0';i++)
			{
				cout<<a[h-1][i];
			}
		cout<<" o' clock";
	}
	
	else if(m==15)
	{
		cout<<"quarter past "<<a[h-1];
	}
	
	else if(m==30)
	{
		cout<<"half past "<<a[h-1];
	}
	else if(m==45)
	{
		cout<<"quarter to "<<a[h%12];
	}
	else
	{
		if(m<30)
		{
		
		if((m>0)&&(m<20))
		{
			for(i=0;a[m-1][i]!='\0';i++)
			{
				cout<<a[m-1][i];
			}
			cout<<" minutes past "<<a[h-1];
		}
		else 
		{
			cout<<"twenty ";
			if(m%10!=0)
			{
				for(i=0;a[m%10-1][i]!='\0';i++)
			   {
				cout<<a[m%10-1][i];
		       }
			   cout<<" minutes past "<<a[h-1];
		   }
		   else 
		   {
			   cout<<"minutes past "<<a[h-1];
		   }
	    
	    }
        }
        
        else
        {
			m=60-m;
			h=h%12+1;
			if((m>0)&&(m<20))
		{
			for(i=0;a[m-1][i]!='\0';i++)
			{
				cout<<a[m-1][i];
			}
			cout<<" minutes to "<<a[h-1];
		}
		else 
		{
			cout<<"twenty ";
			if(m%10!=0)
			{
				for(i=0;a[m%10-1][i]!='\0';i++)
			   {
				cout<<a[m%10-1][i];
		       }
			   cout<<" minutes to "<<a[h-1];
		   }
		   else
		   {
		   cout<<"minutes to "<<a[h-1];
	   }
	   }
      }
   }
   
   return 0;
}
