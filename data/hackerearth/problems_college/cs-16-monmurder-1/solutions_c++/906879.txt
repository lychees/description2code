#include<bits/stdc++.h>
using namespace std;
int pn[1000000],arr[1000000];
int main()
{
	int i,j,t,cnt,n,p,nsum;
	memset(pn,0,sizeof(pn));
	pn[0]=pn[1]=1;
	for(i=2;i<sqrt(1000000);++i)
	{
		cnt=2;
		for(j=i*cnt;j<1000000;j=i*cnt)
		{
			pn[j]=1;
			++cnt;
		}
	}
	cin>>t;
	while(t--)
	{
		cin>>n;
		p=0;
		for(i=2;i<=n;++i)
		{
			cnt=0;
			if(pn[i]==0)
			{
				while(n%i==0)
				{
					++cnt;
					n=n/i;
				}
				if(cnt)
				{
					arr[p]=cnt;
					++p;
				}
			}
		}
		nsum=arr[0];
		for(i=1;i<p;++i)
			nsum^=arr[i];
		nsum?cout<<"Printf{}"<<endl:cout<<"Pappu Bhujia"<<endl;
	}
	return 0;
}
