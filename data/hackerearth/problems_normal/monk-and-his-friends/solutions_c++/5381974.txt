#include <iostream>
#include <map>
using namespace std;
int main()
{
   int t;scanf("%d",&t);
   map<long long int,bool> s;
   string out="";
   while(t--){
   	int n,m;scanf("%d%d",&n,&m);
   		while(n--){
   			long long int a;scanf("%lld",&a);
   			//s.insert(a);
   			s[a]=1;
   		}
   		while(m--){
   			long long int a;scanf("%lld",&a);
   			if(s.find(a)!=s.end()){
   				printf("%s","YES\n");
   			}
   			else{
   				printf("%s","NO\n");
   				//s.insert(a);
   				s[a]=1;
   			}
   		}		
   		s.clear();
   	}
   //	printf("%s",out.c_str());
   
    return 0;
}
