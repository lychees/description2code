def print_reverse(var):
	print var[::-1]

def main():
	test_cases = int(raw_input())
	input_str = []
	for i in range(test_cases):
		var = raw_input()
		input_str.append(var)
	
	for i in range(test_cases):
		print_reverse(input_str[i])
		
if __name__ == "__main__":
	main()
