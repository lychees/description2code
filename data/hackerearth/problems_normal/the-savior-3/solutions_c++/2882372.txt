#include <iostream>
using namespace std;
#define max 1000
int main()
{
    int n,no,i,j,count=0;
    int a[max];
    cin>>n;
    while(n--)
    {
    	count=0;
    	cin>>no;
    	for(i=0;i<no;i++)
    		cin>>a[i];
    	for(i=0;i<no-1;i++)
    	{
    		for(j=i+1;j<no;j++)
    		{
    			if(a[i]==a[j])
    				continue;
    			else if((a[i]+a[j])%2==0)
    			{
    				count++;
    			}
    		}
    	}
    	cout<<count<<endl;
    	
    }
    return 0;
}
