typedef long long int ll;

ll a[1000001];

int fi(int n) 
 { 
   int result = n; 
   for(int i=2;i*i <= n;i++) 
   { 
     if (n % i == 0) result -= result / i; 
     while (n % i == 0) n /= i; 
   } 
   if (n > 1) result -= result / n; 
   return result; 
 } 

void generate()
{
	a[0] = 0;
	a[1] = 0;
	for(int i=2;i<1000001;i++)
	{
		a[i] = fi(i);
		a[i] += a[i-1];
	}
