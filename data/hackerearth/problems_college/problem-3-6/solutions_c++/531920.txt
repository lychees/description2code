#include <iostream>
using namespace std;
long int x[100002],y[100002],cnt[100002],away[100002],home[100002];
int main()

{
	long int n;
	long int j;
	cin>>n;
	
		for( j = 0; j<n; j++)
    {   cin>>x[j]>>y[j]; 
    	cnt[ x[j] ]++;
}
for(long int i = 0; i < n; i++)
{
    home[i] = n - 1;
    home[i] += cnt[ y[i] ];

    away[i] = 2 * (n - 1) - home[i];
    cout<<home[i]<<" "<<away[i]<<"\n";
}

return 0;
}
