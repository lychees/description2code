template<typename T, typename U> inline void amin(T &x, U y) { if(y < x) x = y; }

vector<int> smallestPrimeFactor;
void getSmallestPrimeFactors(int n) {
	n = max(n, 1);
	if((int)smallestPrimeFactor.size() >= n+1) return;
	smallestPrimeFactor.assign(n+1, 0);
	smallestPrimeFactor[1] = 1;
	rer(i, 2, n) if(!smallestPrimeFactor[i]) {
		smallestPrimeFactor[i] = i;
		if((ll)i * i <= n)
		for(int j = i * i; j <= n; j += i)
			if(!smallestPrimeFactor[j]) smallestPrimeFactor[j] = i;
	}
}

int main() {
	const int X = 10000000;
	getSmallestPrimeFactors(X);
	vector<int> digitSum(X+1), factorDigitSum(X+1);
	vector<bool> v(X+1);
	rer(i, 1, X) {
		digitSum[i] = i < 10 ? i : digitSum[i / 10] + i % 10;
		int p = smallestPrimeFactor[i];
		if(p == i)
			factorDigitSum[i] = digitSum[p];
		else
			factorDigitSum[i] = digitSum[p] + factorDigitSum[i / p];
		v[i] = digitSum[i] == factorDigitSum[i];
	}
	vector<int> cnt(X+2);
	rep(i, X+1)
		cnt[i+1] = cnt[i] + v[i];
	int T;
	scanf("%d", &T);
	rep(ii, T) {
		int L, R;
		scanf("%d%d", &L, &R), ++ R;
		int ans = cnt[R] - cnt[L];
		printf("%d\n", ans);
	}
	return 0;
}
