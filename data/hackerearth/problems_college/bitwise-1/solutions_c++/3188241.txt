#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	scanf("%d",&t);
	for(int te=1;te<=t;te++){
		long long x,k;
		scanf("%lld %lld",&x,&k);
		long long ans=0;
		int c=0;
		for(int i=0;i<64;i++){
			if((x>>i)&1){
				continue;
			}
			else {
				if((k>>c)&1){
					ans=ans|(1<<i);
				}
				c++;
			}
		}
		printf("Case #%d: %lld\n",te,ans);
	}
	return 0;
}
