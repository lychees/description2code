#include <bits/stdc++.h>
using namespace std;


int main()
{
	long long int n,m,i,j,k;
	cin >> n >> m;
	long long int val[m];
	for(i=0;i<m;i++){
		cin >> val[i];
	}
	long long int dp[n+1][m+1];
	for(i=0;i<=n;i++){
		for(j=0;j<=m;j++){
			if(i==0) dp[i][j]=1;
			else if(j==0) dp[i][j]=0;
			else if(val[j-1]<=i) dp[i][j]=dp[i-val[j-1]][j]+dp[i][j-1];
			else dp[i][j]=dp[i][j-1];
		}
	}
	cout << dp[n][m];
    return 0;
}
