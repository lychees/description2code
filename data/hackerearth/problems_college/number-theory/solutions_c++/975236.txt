#include <bits/stdc++.h>
using namespace std;
bool findcomm (int b, int a)
{map <int, int> first; 
	while (b)
	{
		first[b%10]=1; b/=10; 
	}
	map <int,int> second;
	while (a)
	{
		second[a%10]=1; a/=10; 
	}
	for (int g=0;g<=9; g++)
	{
		if (second[g] && first[g]) return true; 
	}
	return false;
}
int main()
{int a; cin >> a;
	for (int g=0; g<a; g++)
	{
		int b; cin >> b;int ans=0; 
		for (int y=1;y<=sqrt(b); y++)
		{
			if (b%y) continue;
			if (y==sqrt(b))
			{
				if (findcomm(y,b))
				{
					ans++;
				}
			}
			else
			{
				if (findcomm(y,b)) ans++;
				if (findcomm(b,b/y)) ans++; 
			}
		}cout << ans << '\n';
	}
	return 0;
}
