#include <iostream>
using namespace std;

int main()
{
	int t=0,n,count=0;
    cin>>t;
    for(int i=0;i<t;i++){
    	count=0;
    	cin>>n;
    	int a[n];
    	for(int j=0;j<n;j++){
    		cin>>a[j];
    	}
    	for(int j=0;j<n;j++){
    		int out =1;
    		for(int k=0;k<n;k++){
    			if(k!=j){
    				out= out^a[k];
    			}
    		}
    		if(out == 0){
    			count++;
    		}
    	}
    	cout<<n-count<<endl;
    }
    return 0;
}
