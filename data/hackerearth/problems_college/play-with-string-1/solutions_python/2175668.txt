from __future__ import print_function

for i in range(int(raw_input())):
	word = raw_input()
	rev =word[::-1]
	for i in range(1,int(len(rev))):
		rev_ord = abs(ord(rev[i]) -ord(rev[i-1]))
		word_ord= abs(ord(word[i])-ord(word[i-1]))
		if rev_ord != word_ord:
			print("Not Funny")
			break
	else:
		print("Funny")
