#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--){
    	int n,i,x,y,c=0,h[1000005]={0};
    	cin>>n;
    	for(i=0;i<n;i++){
    		cin>>x>>y;
    		h[x]++;
    		if(h[y]==0)
    		c++;
    		else
    		h[y]-=1;
    	}
    	cout<<c<<endl;
    }
    return 0;
}
