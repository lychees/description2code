using namespace std;

bool prime(long int n){
long int x,t = sqrt(n);
if(n%2==0){
    return false;
}
for(x=3;x<=t;x+=2)
{
    if(n%x==0){
        return false;}
}
if(x>=t){
return true;
}
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    long int m,n,count = 0;
    scanf("%ld %ld",&m,&n);
    for(long int i=m;i<=n;i++)
    {
        if(i==1 || i%2==0 && i!=2){
            continue;}
        else if(i==2){
            count++;
            continue;}
        else if(prime(i))
            {
               count++;
            }
    }
    printf("%ld\n",count);
}
}
