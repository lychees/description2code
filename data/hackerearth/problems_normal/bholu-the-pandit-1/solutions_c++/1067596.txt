#include<bits/stdc++.h>
#define ll long long int
#define mod 1000000007
using namespace std;
ll gcd(ll a,ll b)
{
    if(b==0)
        return a;
    return gcd(b,a%b);
}
int main()
{
    ll t,a,b,c,s,n;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld%lld%lld",&a,&b,&c);
        ll s=gcd(a,gcd(b,c));
        a=a/s;b=b/s;c=c/s;
        n=(((a*b)%mod)*c)%mod;
        printf("%lld %lld\n",s,n);
    }
    return 0;
}
