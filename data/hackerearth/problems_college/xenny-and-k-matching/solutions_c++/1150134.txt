#include <bits/stdc++.h>

using namespace std;

int main()
{
	int n,i,j,k,ans,t1,t2;
	scanf("%d %d",&n,&k);
	int a[n];
	for (i = 0; i < n; i++) {
		scanf("%d",&a[i]);
	}
	ans = 0;
	for (i = 0; i < n; i++) {
		j = i + k;
		if (j >= n) {
			break;
		}
		t1 = abs(a[j]-a[i]);
		t2 = abs(a[j]+a[i]);
		if (t1 % k == 0 || t2 % k == 0) {
			ans++;
		}
	}
	printf("%d\n",ans);
	
	return 0;
}
