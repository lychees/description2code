    #include<iostream>
    #include<math.h>
     
    using namespace std;
    int cnt=0;
     
    int main()
    {
        double c,b;
     
        long long t,n,i,j,f=0,total,ans;
        cin >> t;
     
        total = pow(2, 32) - 1;
     
        while(t--)
        {
            f=0;
     
            cin >> n;
            long long a[n];
     
            for(i=0; i<n; i++)
                cin >> a[i];
     
            if(n==1)
            {
                if((log2(a[0])-(int)log2(a[0]))==0)
                    cout << "YES" <<endl;
                else
                    cout << "NO" <<endl;
            }
     
            else
        {
            for(i=0; i<32 ;i++)
            {
                ans= total;
                for(j=0; j<n; j++)
                {
                    if((a[j] & (1<<i))!=0)
                        ans= ans&a[j];
                }
     
                c= log2(ans);
                b=(int)log2(ans);
     
     
                if(c-b==0)
                {
                    f=1;
                    break;
                }
     
            }
     
            if(f==1)
                cout << "YES" <<endl;
     
            else
                cout << "NO" <<endl;
        }
     
        }
     
        return 0;
    }
