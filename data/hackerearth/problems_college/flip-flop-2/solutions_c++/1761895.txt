#include <bits/stdc++.h>
using namespace std;
int main()
{
    long long int tc;
    cin>>tc;
    while(tc--)
    {
        long long int coun=0,ans=0;
        string s;
        cin>>s;
        for(int i=0;i<(s.length()-1);i++)
        {
             if(s[i] == s[i+1])
                    coun++;
              else ans++;
        }
        cout<<coun<<endl;
    }
}
