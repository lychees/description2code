#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t; cin >> t;
    while(t--)
    {
    	string s; cin >> s; int i;
    	string r=s; reverse(s.begin(),s.end());
    	for( i=1;i<s.length();i++)
    	{
    		if(abs(s[i]-s[i-1]) != abs(r[i]-r[i-1]))
    			break;
    	}
    	cout << (i==s.length()?"Funny":"Not Funny") << endl;
    }
    return 0;
}
