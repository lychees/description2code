long long int pow(long long int x, long long int i)
{
        long long int t;
        if(i<=0)
                return 1;

        t = pow(x,i/2);
        if(i%2 == 0) {
                return t*t;
        }
        else {
                return x*t*t;
        }
}
long long int MAX(long long int N)
{
        long long int i,x;
        x = 1;
        while(1) {
                if( x > N) {
                        return x/2;
                }
                x=x*2;
                i++;
        }
}
