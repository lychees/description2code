#include <bits/stdc++.h>
using namespace std;

int main() {
	int tt; cin >> tt;	
	while (tt--) {
		int n; cin >> n;
		if (n == 2 ) {
			cout << "Yes\n";
			continue;
		}
		set<long> myset;
		myset.insert(1);
		bool ok = true;
		for (int i = 2; i * i <= n && ok; i++) {
			int lg = ceil(log10(n) / double(log10(i)));			
			for (int j = 2; j <= lg; j++) {
				long num = (long) pow(i, j);
				if (myset.count(n - num) || num + num == n) {
					ok = false;
					break;
				}
				myset.insert(num);
			}
		}
		if (!ok)
			cout << "Yes\n";
	    else 
			cout << "No\n";		
	}
}
