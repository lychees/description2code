#include <iostream>
using namespace std;
#define MAX 100001
#define M 1000000007
int main()
{
    	long long  int t,n;
	long long int table[MAX];
	table[0]=0;
	table[1]=1;
	table[2]=2;
	table[3]=4;
	for(int i=4;i<MAX;i++){
		table[i]=table[i-1]+table[i-2]+table[i-3];
		if(table[i]>=M)
        table[i]=table[i]%M;
		//table[i]=table[i]%M;
	}
	cin>>t;
	while(t--){
		cin>>n;
		cout<<table[n]<<endl;
		
	}
	return 0;
}
