#include <iostream>
#include <vector>
#include <algorithm>
#define mod 1000000007
using namespace std;
typedef long long int64;

int64 modexp(int x,int p) {
	if(p==0)
		return 1;
	int64 res = modexp(x,p/2);
	res = (res*res)%mod;
	if(p&1)
		return (res*x)%mod;
	return res;
}

int main()
{
    int T,c;
    cin>>T;
    int N;
    while(T) {
    	vector<int> items;
    	cin>>N;
    	for(int i=0;i<N;i++) {
    		cin>>c;
    		items.push_back(c);
    	}
    	sort(items.begin(),items.end());
    	int64 max = 0;
    	int arr[20],cnt=0;
		for(int i=1;i<1<<N;i++) {
			int k = i,t=0;
			cnt = 0;
			while(k>0) {
				if(k&1)
					arr[cnt++] = items[t];
				t++;
				k=k>>1;
			}
			if(cnt>0 && cnt%2==0) {
				int64 num=1,den=1;
				for(int j=0;j<cnt/2;j++) {
					den=(den*arr[j])%mod;
				}
				for(int j=cnt/2;j<cnt;j++) {
					num=(num*arr[j])%mod;
				}
				den = modexp(den,mod-2);
				int64 res = (num*den)%mod;
				if(res>max)
					max = res;
			}
		}
		cout<<max<<endl;
    	T--; 
    }
    return 0;
}
