for _ in range(input()):
    noOfCandies, noOfDays = map(int, raw_input().split())
    for _ in range(noOfDays):
        if noOfCandies % 2 > 0:
        	noOfCandies -= (noOfCandies+1)/2
        else:
        	noOfCandies -= noOfCandies/2
       
        noOfCandies -= noOfCandies//4
    print noOfCandies
