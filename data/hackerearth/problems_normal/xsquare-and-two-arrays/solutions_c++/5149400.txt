#include <bits/stdc++.h>
#define lli long long
#define MAX 100005
using namespace std;
lli A[2][MAX];
lli dp[2][MAX];
int main()
{
    int n,q,type,x,y;
    cin >> n >> q;
	assert(n>=1 && n <= 100000);
	assert(q>=1 && q <= 100000);
    for ( int i = 0; i < 2; i++ ) {
        for ( int j = 1; j <= n; j++ ) {
        	cin >> A[i][j];
        	assert(A[i][j] <= 1000000000);
        }
    }
    for ( int i = 1; i <= n; i++ ) {
        if ( i%2 == 0 ) dp[0][i] = dp[0][i-1] + A[0][i], dp[1][i] = dp[1][i-1] + A[1][i];
        else dp[0][i] = dp[0][i-1] + A[1][i], dp[1][i] = dp[1][i-1] + A[0][i];
    }
    while ( q-- ) {
        cin >> type >> x >> y;
        if ( type == 2 ) {
            if ( x%2 == 1 ) cout << dp[0][y] - dp[0][x-1] << endl;
            else cout << dp[1][y] - dp[1][x-1] << endl;
        }
        else {
            if ( x%2 == 1 ) cout << dp[1][y] - dp[1][x-1] << endl;
            else cout << dp[0][y] - dp[0][x-1] << endl;
        }
    }
    return 0;
}
