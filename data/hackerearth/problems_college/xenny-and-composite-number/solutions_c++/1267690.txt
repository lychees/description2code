#include <iostream>
#include<math.h>
using namespace std;

int isPrime(int n){
    
    int i;

    if (n==2)
        return 1;

    if (n%2==0)
        return 0;

    for (i=3;i<=sqrt(n);i+=2)
        if (n%i==0)
            return 0;

    return 1;
}



int main()
{
    int t,n,i;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	int status=isPrime(n);
    	if(n==2 || n==1 || n==0)
    		cout<<"4"<<endl;
    	
    	else if(status==1)
    		cout<<n+1<<endl;
    	
    	else
    	{
    		status=isPrime(n+1);
    		if(status==1)
    			cout<<n+2<<endl;
    		else
    			cout<<n+1<<endl;
    	}
    	
    	
    }
    return 0;
}
