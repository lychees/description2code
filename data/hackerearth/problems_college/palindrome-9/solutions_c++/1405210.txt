#include<cstdio>
long long palin(long long n)
{
	long long temp=n,rev=0;
	while(temp>0)
	{
		rev=rev*10+temp%10;
		temp=temp/10;
	}
	if(rev==n)
	return 1;
	else return 0;
}
long long ans[1000000];
int main()
{
	long long int t,i,j,k=0,a,b;
	for(i=1;i*i<=100000000;i++)
	{
		long long temp=i*i;
		for(j=i+1;j*j<=100000000;j++)
		{
			temp+=j*j;
			if(palin(temp)==1)
			{
				long long y,f=0;
				for(y=0;y<k;y++)
				{
					if(ans[y]==temp)
					{f=1;break;}
				}
				if(f==0)
			     ans[k++]=temp;	
			}
			if(temp>100000000)break;
			
		}
	}
	scanf("%lld%lld",&a,&b);
	long long s=0;
	for(i=0;i<k;i++)
	{
		if(ans[i]<=b&&ans[i]>=a)
		s+=ans[i];
	}
	printf("%lld\n",s);
	return 0;
	
}
