#include<iostream>
using namespace std ;
int main()
{
    int t,n,i,count,m ;
    scanf("%d",&t);
    while(t--)
    {
        count=0 ;
        i=0;
        scanf("%d",&n);
        if(n<6)
            count=0;
       else if(n&1==1)
            count=0;
            
            else
            {
            	i=n/8 ;
                m=n-i*8 ;
                if(m==0)
                count=i ;
              
                while(m>0)
                {
                	if(m%6==0){
                		count=i+m/6 ;
                		break ;	
                	}
                	else
                	{
                		i--;
                		 m=n-i*8 ;
                		
                	}
                	if(m<=0)
                	count=0;  	
                }
                
                
            }
               if(count==0)
                printf("-1\n");
                else
                 cout<<count<<endl ;
             
    }
    
    
    
    return 0;
    
}
