#include<bits/stdc++.h>
using namespace std;

const int N=1e5+10;

int fib[N];
int a[N];

void preprocess() {
    int a=0;
    int b=1;
    int c=1;
    while(c<N) {
        c=a+b;
        if(c<N) {
        	fib[c]++;
        }
        a=b;
        b=c;
    }
    fib[1]=0;
    for(int i=1;i<N;i++) {
        fib[i]+=fib[i-1];
    }
    return;
}
int main() {
    preprocess();
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        int a,b;
        scanf("%d%d",&a,&b);
        if(a>b) {
        	swap(a,b);
        }
        int ans=fib[b]-fib[a-1];
        if(ans%2) {
            puts("INVALID");
        } else {
            puts("VALID");
        }
    }
    return 0;
}
