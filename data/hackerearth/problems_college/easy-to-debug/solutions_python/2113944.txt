n = int(raw_input())


k=1
s=0
while(k<n):
	if(n%k==0):
		s+=k
	k+=1	
if s==n:
	arr = [[0 for _ in range (n+1)] for _ in range (n+1)]
	for line in range (0,n+1):
		s=[]
		for i in range (0,line):
			if line==i or i==0:
				arr[line][i]=1
			else:
				arr[line][i] = arr[line-1][i-1]+arr[line-1][i]
			s.append(str(arr[line][i]))
		print "".join(s)
else:
	print "error"
