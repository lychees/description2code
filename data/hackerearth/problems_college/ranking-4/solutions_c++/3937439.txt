#include <bits/stdc++.h>
//zone
using namespace std;
long int a[10003];
int main()
{
	int t;
	cin >> t;
	while(t--)
	{
	int n;
	vector<long int> v;
	cin >> n;
	int i;
	for(i = 0 ; i < n ; i++)
	{cin >> a[i];
	v.push_back(a[i]);
	}
	int w = 1;
	map<long int, long int> m;
	sort(a,a+n,greater<long int>());
	m[a[0]] = 1;
	for(i = 1; i< n ;i++)
	{
		if(a[i] == a[i - 1])
		{
			m[a[i]] = m[a[i - 1]];
			w++;
		}
		else
		{
			w++;
		   m[a[i]] = w;
		}
	}
	for(i = 0 ; i < v.size() ; i++)
	{
		cout<<m[v[i]] << " ";
	}
	cout<<endl;
	}
	return 0;
}
