#include <bits/stdc++.h>
using namespace std;

typedef long long LL;
#define PII       pair<int,int>
#define all(c)    c.begin(),c.end()
#define sz(c)     (int)c.size()
#define clr(c)    c.clear()
#define pb        push_back
#define mp        make_pair
#define cin(x)    scanf("%d",&x)
#define MOD       1000000007

#define MAXN	10000001

int mark[MAXN] = {0};
vector<int> primes;
void sieve(int lmt = MAXN)
{
	for(int i = 2; i < lmt; i++)
		mark[i] = 1;
	for(int i = 2; i < lmt; i++)
	{
		if(mark[i] == 0) continue;
		if(i > 2)
		primes.pb(i);
		int j = 2 * i;
		while(j < MAXN)
		{
			mark[j] = 0;
			j += i;
		}
	}
}

int diffPoss[MAXN] = {0};
set<int> possDiff;

vector<int> arr;

int main()
{
	sieve();
	for(int i = 1; i < sz(primes); i++)
		possDiff.insert(primes[i] - primes[i-1]);
	for(set<int>::iterator ii = possDiff.begin(); ii != possDiff.end(); ii++)
		arr.pb(*ii);
	int t;
	cin(t);
	while(t--)
	{
		int n;
		cin(n);
		int yes = 0;

		for(int i = 0; i < sz(arr); i++)
		{
			int left = n - arr[i];
			if(binary_search(all(arr) , left))
			{
				yes = 1;
				break;
			}
		}

		if(yes)
			printf("YES\n");
		else
			printf("NO\n");
	}
	return 0;
}
