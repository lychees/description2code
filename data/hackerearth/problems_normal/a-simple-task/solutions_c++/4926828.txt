#include <bits/stdc++.h>

using namespace std;

#define ll long long int
#define pb push_back

vector<ll> ans;
ll a[105];
ll m;

bool all(ll x)
{
    ll i;
    ll rem=a[0]%x;
    for(i=0;i<m;i++)
        if( (a[i]%x)!=rem )
            return false;
    return true;
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    ll i,diff,x;
    cin>>m;
    for(i=0;i<m;i++)
        cin>>a[i];
    diff=abs(a[0]-a[1]);
    for(x=1;(x*x)<=diff;x++)
    {
        if( (diff%x)==0 )
        {
           if( all(x) )
                ans.pb(x);
           if( all(diff/x) )
                ans.pb(diff/x);
        }
    }
    sort(ans.begin(),ans.end());
    for(i=1;i<ans.size();i++)
        cout<<ans[i]<<" ";
    cout<<"\n";

    return 0;
}
