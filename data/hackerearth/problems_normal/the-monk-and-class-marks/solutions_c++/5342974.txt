#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

void marksMain()
{
    vector< vector<string> > grades(100);
    int N;
    cin >> N;
    for(int i = 0; i < N; i++)
    {
        string name;
        cin >> name;
        int grade;
        cin >> grade;
        grades[grade-1].push_back(name);
    }

    for(int i = 99; i >= 0; i--)
    {
        vector<string> &current = grades[i];
        sort(current.begin(), current.end());
        for(vector<string>::iterator it = current.begin(); it != current.end(); it++)
            cout << *it << " " << i+1 << "\n";
    }
}



int main()
{
    marksMain();
    return 0;
}
