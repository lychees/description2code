#include<bits/stdc++.h>
using namespace std;
struct trienode {
	struct trienode *child[26];
	int wt;
};
struct trienode *getnode()
{
	struct trienode *p;
	p=(struct trienode *)malloc(sizeof(struct trienode));
	int i;
	for(i=0;i<26;i++)
		p->child[i]=NULL;
	p->wt=-1;
	return p;
}
void insert(string s,int w,struct trienode *root)
{
	struct trienode *q=root;
	int t,i;
	for(i=0;s[i]!='\0';i++) {
		if(q->child[s[i]-97]==NULL)
			q->child[s[i]-97]=getnode();
		q=q->child[s[i]-97];
		t=q->wt;
		q->wt=max(t,w);
	}
}
int search(string s, struct trienode *root)
{
	int ind,i,ans;
	struct trienode *p=root;
	for(i=0;s[i]!='\0';i++) {
		ind=s[i]-97;
		if(p->child[ind]==NULL) {
			ans=-1;
			break;
		}
		p=p->child[ind];
		ans=p->wt;
	}
	return ans;
}
int main()
{
	int n,q;
	scanf("%d%d",&n,&q);
	struct trienode *root=getnode();
	while(n--) {
		string s;
		int w;
		cin>>s;
		scanf("%d",&w);
		insert(s,w,root);
	}
	int a;
	while(q--) {
		string s1;
		cin>>s1;
		a=search(s1,root);
		printf("%d\n",a);
	}
	return 0;
}
