#include <bits/stdc++.h>
using namespace std;

bool isTransformationPossible(string s1, string s2, int N)
{
	if(s1.length() != s2.length())
		return false;
		
	int cost=0;
	for(int i=0;i<s1.length();i++)
	{
		if(s1[i] != s2[i])
			cost++;
		if(cost>N)
			return false;
	}
	if((cost <= N) && ((N-cost)%2 == 0)) 
		return true;
	return false;
}

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string s1, s2;
		int N;
	    cin>>s1>>s2>>N;
	    cout<<(isTransformationPossible(s1,s2,N) ? "Yes" : "No")<<endl;
	}
    return 0;
}
