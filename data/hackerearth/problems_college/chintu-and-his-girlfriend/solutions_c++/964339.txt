#include <bits/stdc++.h>
using namespace std;

int main()
{
	int x,y;
	cin >> x >> y;
	int t = x;
	map<char,int> mymap;
	while (x > 0){
		string s;
		cin >> s;
		bool l[26];
		for (int i= 0; i < 26; i++){
			l[i] = false;
		}
		for (int i = 0; i < s.length(); i++){
			if (l[s[i] - 97] == false){
				if (mymap.find(s[i]) == mymap.end()){
					mymap[s[i]] = 1;
				}	
				else{
					mymap[s[i]] += 1;
				}
				l[s[i] - 97] = true;
			}
		}
		x -= 1;
	}
	map<char,int>::iterator it;
	for (std::map<char,int>::iterator it=mymap.begin(); it!=mymap.end(); ++it){
   		if (it->second == t)
   			y--;
   	}
   	if (y == 0)
   		cout << ":-)" << endl;
   	else
   		cout << ":-(" << endl;
    return 0;
}
