#include <iostream>
#include<math.h>
using namespace std;
long long int m=1000000007;

int main()
{
    int t,n,c;
    long long int a,b,r,sum=0;
    scanf("%d",&t);
    while(t--)
    {	sum=0;
    
    		scanf("%d",&n);
    		a=(n*(n+1))/2;
    		c=n-1;
    		b=pow(2,c);
    		
    		sum=a*b;
    		r=sum%m;
    		printf("%lld\n",r);
    		
    }
    return 0;
}
