#include<iostream>
#include<cstdio>
#include<cstring>
#include<vector>
#include<utility>
#include<climits>
#include<cstdlib>
#include<algorithm>
#include<map>
using namespace std;
int n;
int a[100];
int main()
{
	int t,i,j;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		int min=0;
		int x=-1;
		for(i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
			min+=a[i];
			x=max(x,a[i]);
		}	
		int c[n];
		for(i=0;i<n;i++)
		{
			int k=0;
			int sum1=0;
			for(j=0;j<n;j++)
			if(j!=i)
			{
				c[k]=a[j];
				sum1+=a[j];
				k++;
			}
		//	cout<<min<<" "<<x<<" "<<sum1<<endl;
			int sum=sum1/2;
			int cal[n][sum+1];
			memset(cal,0,sizeof(cal));
			for(k=1;k<=n-1;k++)
			{
				for(j=1;j<=sum;j++)
				if(j>=c[k-1])
				cal[k][j]=max(cal[k-1][j],c[k-1]+cal[k-1][j-c[k-1]]);
				else
				cal[k][j]=cal[k-1][j];
			}
		//	cout<<cal[n-1][sum]<<"  ";
			if((sum1-2*cal[n-1][sum])<min)
			{
				min=(sum1-2*cal[n-1][sum]);
				x=a[i];
			}
			else if((sum1-2*cal[n-1][sum])==min)
			{
				if(x>a[i])
				x=a[i];
			}
			//cout<<x<<" "<<min<<endl;

		}
		cout<<min<<" "<<x<<endl;
	}
	return 0;
}
