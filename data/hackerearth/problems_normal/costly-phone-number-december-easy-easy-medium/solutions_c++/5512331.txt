#include <iostream>
using namespace std;

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    
    unsigned int t,a[10],c[1000],i,j,p,k,ans,l;
    char f;
    bool ch;
    cin >> t;
    while (t--)
    {
    	ans = 0;
    	ch = 1;
    	for (i = 0; i < 10; i++)
    	  cin >> a[i];
    	cin >> l;
    	for (i = 0; i < l; i++)
    	{
    		cin >> f;
    		c[i] = f - 48;
    	}
    	while (ch)
    	{
    		ch = 0;
    		for (i = 0; i < 10; i++)
    		 for (j = 0; j < 10; j++)
    		 {
    	 		p = (i + j) % 10;
    	 		if (a[p] > a[i] + a[j])
    	 	    {
    	 	    	a[p] = a[i] + a[j];
    	 	    	ch = 1;
    	 	    }
    	     }
    	 }
    	 for (i = 0; i < l; i++)
    	   ans += a[c[i]];
    	 cout << ans << "\n";
    }
    return 0;
}
