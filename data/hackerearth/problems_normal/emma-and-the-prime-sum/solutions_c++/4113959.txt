#include <bits/stdc++.h>
using namespace std;
long long int sieve[1000008];
long long int sum[1000008];
int main()
{
	long long int k, j;
	for (k = 2 ; k <= 1008 ; k++) {
		if (sieve[k] == 0) {
			for (j = 2 ; j <= 1000000 ; j++) {
				if (k * j > 1000000) {
					break;
				}
				sieve[k * j] = 1;
			}
		}
	}
	sieve[1] = 1;
	long long int sums = 0;
	for (k = 1 ; k <= 1000000 ; k++) {
		if (sieve[k] == 0) {
			sums += k;
		}
		sum[k] = sums;
	}
	int tc;
	scanf("%d", &tc);
	while (tc--) {
		long long int x, y;
		scanf("%lld %lld", &x, &y);
		printf("%lld\n", sum[y] - sum[x - 1]);
	}
	return 0;
}
