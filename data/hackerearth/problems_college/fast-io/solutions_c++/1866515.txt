#include <bits/stdc++.h>
#define max 300000000
using namespace std; 
long long int scan()
{
    long long int n = 0, ch = getchar_unlocked();
    while (ch < '0' || ch > '9')
        ch = getchar_unlocked();
    while (ch >= '0' && ch <= '9') {
        n = (n << 3) + (n << 1) + ch - '0';
        ch = getchar_unlocked();
    }
    return n;
}
int main()
{
	long  n,c=0;
	n=scan();
	long long l;
	while(n--)
	{
		l=scan();
		c+=(l<max?1:0);
	}	
	printf("%ld",c);
	return 0;
}
