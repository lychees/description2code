#include <iostream>
using namespace std;


int main()
{
    int t,i=1;
    cin>>t;
    while(i<=t){
    	int n,j;
    	cin>>n;
    	int a[n],b[n];
    	for(j=0;j<n;j++)
    		cin>>a[j];
    		    	for(j=0;j<n;j++)
    		cin>>b[j];
    		int c,d,p;
    		for(c=0;c<n-1;c++){
    			for(d=c+1;d<n;d++){
    				if(a[c]>a[d]){
    					p=a[d];
    					a[d]=a[c];
    					a[c]=p;
    				}
    			}
    		}
    		int g;
    		for(c=0;c<n-1;c++){
    			for(d=c+1;d<n;d++){
    				if(b[c]<b[d]){
    					g=b[c];
    					b[c]=b[d];
    					b[d]=g;
    				}
    			}
    		}
    		int l=0;
    		for(c=0;c<n;c++){
    			if(b[c]%a[c]==0||a[c]%b[c]==0)
    			l++;
    		}
    		cout<<l<<"\n";
    		i++;
    		
    	
    }
    return 0;
}
