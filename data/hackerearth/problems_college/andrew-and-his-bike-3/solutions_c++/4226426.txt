#include <bits/stdc++.h>
using namespace std;
int visited[100007];
vector<int>gra[100007];
int ans[100007];
int dfs(int u)
{
	visited[u]=1;
	int temp=0;
	for(int i=0;i<gra[u].size();i++)
	{
		if(!visited[gra[u][i]])
		{
			temp=temp+dfs(gra[u][i]);
		}
	}
	ans[u]=temp+1;
	return ans[u];
}
int main()
{
	int n,a,b;
	scanf("%d",&n);
	for(int i=0;i<n-1;i++)
	{
		scanf("%d %d",&a,&b);
		gra[a].push_back(b);
		gra[b].push_back(a);
	}
	dfs(1);
	printf("%d ",ans[1]-1);
	for(int i=2;i<=n;i++)
	{
		printf("%d ",ans[i]);
	}
	printf("\n");
    return 0;
}
