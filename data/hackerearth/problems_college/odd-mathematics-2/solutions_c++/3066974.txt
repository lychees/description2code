//JUST LIKE ANIMALS !!!!

#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
 
int main()
{
    
    int t;
    cin>>t;
    while(t--)
    {
    	ll n,sum=0;
    	cin>>n;
    	while(n)
    	{
    		long long int x=(n+1)/2;
    		x=x*x;
    		sum=sum+x;
    		n=n/2;
    		
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
