#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t; cin >> t;
    while(t--)
    {
    	int n; cin >> n;
    	int a[n],i;
    	for(i=0;i<n;i++)
    		cin >> a[i];
    	int count_left = count(a,a+n,-1);
    	int count_right = count(a,a+n,1);
    	cout << (count_left<count_right?count_left+1:count_left) << endl;
    }
    return 0;
}
