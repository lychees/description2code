typedef unsigned long ul;

int sumofd(ul num)
{
	int sum =0;
	while(num>0)
	{
		sum = sum + num%10;
		num/=10;
	}
	return sum;

}

int main()
{

	int t=0;
	scanf("%d",&t);
	ul n;

	while(t--)
	{
		scanf("%ld",&n);
		ul x = n;
		bool bno = false;
		if(x==1||x==4)
			printf("YES\n");
		else
		{
			bno = false;
				set<ul> hold;

			while(1)
			{
				x=x*x;
				x = sumofd(x);
				if(x==1||x==4)
				{
					cout<<"YES\n";
					break;
				}
				if(hold.find(x)!=hold.end())
				{
					bno = true;
					break;
				}
				else
					hold.insert(x);
			}
		}
			if(bno)
				printf("NO\n");
	}
	return 0;
}
