#include<stdio.h>

#define lli long long int

#define in(t) scanf("%d",&t)

#define inll(t) scanf("%lld",&t)
#define inlu(t) scanf("%llu",&t)
#define MOD 1000000007
int main()
{
    int cases;
    scanf("%d",&cases);
    while(cases--)
    {
    	string a;
    	string b;
    	cin>>a;
    	cin>>b;
    	int alen= a.length();
    	int blen  = b.length();
    
    	int arr[alen];
    	int brr[alen];
    	int arr2[blen];
    	int brr2[blen];
    	for(int i=0;i<alen;i++)
    	{
    		arr[i] = 0;
    		brr[i] = 0;
    	}
    	for(int j=0;j<blen;j++)
    	{
    		arr2[j]  =0;
    		brr2[j]  =0;
    	}
    	int cta = 0;
    	int ctb = 0;
    	for(int i=0;i<alen;i++)
    	{  
    		
    	         if(a[i]=='A')
    	         cta++;
    	         else if(a[i]=='B')
    	         ctb++;
    	    arr[i] =  cta;
    	         brr[i]  =  ctb;     
    	         
     	        
    	}
    	cta = 0;
    	ctb = 0;
    		for(int j=0;j<blen;j++)
    		{
    			
    			
    			if(b[j]=='A')
    	         cta++;
    	         else if(b[j]=='B')
    	         ctb++;
    			arr2[j] =  cta;
    	         brr2[j]  =  ctb;
    		}
    		
    		int no = 0;
    		for(int i=0;i<alen;i++)
    		{
    			if(arr[i]>arr2[i])
    			no = 1;
    		}
    		for(int i=0;i<blen;i++)
    		{
    			if(brr[i]<brr2[i])
    			no = 1;
    		}
    		if(no==0)
    		cout<<"Yes"<<endl;
    		else
    		cout<<"No"<<endl;
    		
    }
    return 0;
}
