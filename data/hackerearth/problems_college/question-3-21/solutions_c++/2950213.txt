#include<iostream>
#include<math.h>

using namespace std;

int isPrime(int a)
{
    if (a == 0 || a == 1)
        return 0;
    for (int i = 2; i <= sqrt(a); i++)
        if (a%i == 0)
            return 0;
    return 1;

}

int main()
{
    int i, j, n, p, count = 0, x;
    int ans[1000];
    cin >> n >> p;
    for (i = 0; i < p; i++)
    {
        count = 0;
        cin >> x;
        for (j = 1; j <= x; j++)
            if (isPrime(j))
                count++;
        ans[i] = count;
    }
    for (i = 0; i < p; i++)
    {
        cout << ans[i] << '\n';
    }
    return 0;
}
