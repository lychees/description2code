#include<bits/stdc++.h>
using namespace std;

#define mod 1000000007
#define W while
#define LL long long int

int main()
{
    LL n,sum=0,i;
    cin>>n;         
    LL a[n];
    for(i=0;i<n;i++)
    	cin>>a[i];     
    LL t[n];
    t[0]=1; 
    for(i=1;i<n;i++)       
    {
    	if(a[i]>a[i-1])   
    		t[i]=t[i-1]+1;
    	else
    		t[i]=1;
    }	
    for(i=n-2;i>=0;i--)
	{
		if(a[i]>a[i+1]&&t[i]<=t[i+1])
			t[i]=t[i+1]+1;
	}
	for(i=0;i<n;i++)
		sum=sum+t[i];
	
	cout<<sum<<endl;
	return 0;
}
