#include <iostream>
using namespace std;

int main()
{
    int t,r;
    long int n,a,m;
    cin>>t;
    while(t--)
    {
    	a=0;
    	m=1;
    	cin>>n;
    	while(n)
    	{
    	         	r=n%10;
    	             if(r>3)
    	            {
    	                   	a+=((r-1)*m);
                   	}
               	else
                 	{
    		           a+=(r)*m;	
                  	}
                    	m*=9;
    	            n=n/10;
    	}
    	cout<<a<<endl;
          
    }
        return 0;
}
