#include <bits/stdc++.h>

using namespace std;

#define maxN 10000

vector<int> g[maxN+5];
int n;
int dp[maxN+5];

int dfs(int u)
{
	if(u==n)
		return 1;
	if(dp[u]!=-1)
		return dp[u];
	int i,v;
	dp[u]=0;
	for(i=0;i<(int)g[u].size();i++)
	{
		v=g[u][i];
		dp[u]+=dfs(v);
	}
	return dp[u];
}

int main()
{
    int i,u,v;
    scanf("%d",&n);
    while(true)
    {
    	scanf("%d%d",&u,&v);
    	if(u==0||v==0)
    		break;
    	g[u].push_back(v);
    }
    for(i=1;i<=n;i++)
    	dp[i]=-1;
    printf("%d\n",dfs(1));
    return 0;
}
