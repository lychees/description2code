inline LL power(LL x,LL y)
{
    LL ans=1;
    while(y>0){
        if(y&1)
            ans=(ans*x)%M;
        x=(x*x)%M;
        y/=2;
    }
    return ans;
}
inline int read()
{
	int ret = 0;
	int c = getchar_unlocked();

	while(c<'0' || c>'9')
		c = getchar_unlocked();
	while(c>='0' && c<='9')
	{
		ret = (ret<<3) + (ret<<1) + c - '0';
		c = getchar_unlocked();
	}
        return ret;
}
const int N=11111;
vector<int> V[N];
int ans,cnt[N];
int k,n,x;
void dfs(int cur,int prev)
{
    cnt[cur]=1;
    for(int i=0;i<V[cur].size();i++){
        if(V[cur][i]!=prev){
            dfs(V[cur][i],cur);
            cnt[cur]+=cnt[V[cur][i]];
        }
    }
    if(cnt[cur]>=k){
        ans++;
        cnt[cur]=0;
    }
}
int main()
{
    scanf("%d %d",&k,&n);
    for(int i=2;i<=n;i++){
        scanf("%d",&x);
        V[i].pb(x);
        V[x].pb(i);
    }
    dfs(1,-1);
    printf("%d\n",ans);
    return 0;
}
