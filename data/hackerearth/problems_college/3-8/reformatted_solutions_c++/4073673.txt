#include <bits/stdc++.h>
#include <iostream>
using namespace std;

int main() {
  int t;
  string s, ls;
  int i, l, nls, j;
  cin >> t;
  while (t--) {
    cin >> s;
    ls = "";
    ls[0] = s[0];
    nls = 1;
    l = s.size();
    for (i = 1; i < l; i++) {
      if (s[i] >= ls[0]) {
        for (j = nls; j >= 1; j--) {
          ls[j] = ls[j - 1];
        }
        ls[0] = s[i];
      } else {
        ls[nls] = s[i];
      }
      nls++;
    }
    for (i = 0; i < nls; i++) {
      cout << ls[i];
    }
    cout << endl;
  }
  return 0;
}
