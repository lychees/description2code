#include <iostream>
using namespace std;

int main()
{
    //printf("Hello World!\n");
    int t,n;
    cin>>t;
    long long arr[1000000];
    arr[0]=0;
    arr[1] = 1;
    arr[2] = 2;
    arr[3] = 4;
    for(int i=0;i<t;i++)
    {
    	cin>>n;
    	if(n==1||n==2||n==3)	cout<<arr[n]<<endl;
    	else
    	{
    		for(int j=4;j<=n;j++)
    		{
    			arr[j] = (arr[j-1] + arr[j-2] + arr[j-3])%1000000007;
    		}
    		cout<<arr[n]<<endl;
    	}
    }
    return 0;
}
