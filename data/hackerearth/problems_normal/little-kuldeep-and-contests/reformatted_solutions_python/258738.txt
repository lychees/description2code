N = int(raw_input())
a, b, c = [], [], []
for i in range(N):
    a.append(raw_input().split('-'))
for value in a:
    for val in value:
        h, m = map(int, val.split(':'))
        b.append(h+m/60.0)
for i in range(N):
    d = (b[2*i], b[2*i+1])
    c.append(d)
c.sort()
for i in range(N-1):
    if c[i][1] > c[i+1][0]:
        print "Will need a moderator!"
        break
else:
    print "Who needs a moderator?"
