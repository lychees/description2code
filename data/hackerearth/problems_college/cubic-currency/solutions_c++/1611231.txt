#include <iostream>
#include <stdio.h>
#include <vector>

using namespace std;

typedef long long int Lint;

Lint Ways( vector<Lint>& A, Lint m, Lint n ) {
	
	Lint i, j, x, y;
	vector< vector<Lint> > B( n + 1, vector<Lint> ( m, 0 ) );
	
	for( i = 0;i < m;i++ )
		B[ 0 ][ i ] = 1;
	
	for( i = 1;i < n + 1;i++ ) {
		
		for( j = 0;j < m;j++ ) {
			
			x = ( i - A[ j ] >= 0 ) ? B[ i - A[ j ] ][ j ] : 0;
			y = ( j >= 1 ) ? B[ i ][ j - 1 ] : 0;
			B[ i ][ j ] = x + y;
		} 
	}
	
	return B[ n ][ m - 1 ];
}

int main()
{
	
	Lint i, t, n;
	vector<Lint> A( 22 );
	
	for( i = 0;i < 22;i++ )
		A[ i ] = i*i*i;
		
	scanf("%lld", &t );
	
	while( t-- ) {
		
		scanf("%lld", &n );
		printf("%lld\n", Ways( A, 22, n ) );
	}
	
	return 0;
}
