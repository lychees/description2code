#include <iostream>
#include<vector>
#include<algorithm>
#include<cstring>
using namespace std;
#define pb push_back 
vector<int>v[100005];
bool visited[100005];
void dfs(int i){
	visited[i]=1;
for(int j=0;j<v[i].size();j++){
	if(!visited[v[i][j]])dfs(v[i][j]);
}
}

int main()
{
	int T,a,b;
    cin>>T;
    while(T--){
    	int N,E;
    	cin>>N;
    	cin>>E;
    	memset(v,0,sizeof(v));
    	memset(visited,0,sizeof(visited));
    	while(E--){
    		cin>>a>>b;
    		v[a].pb(b);
    		v[b].pb(a);
    	}
    	int ans=0;
    	for(int i=0;i<N;i++){
    		if(!visited[i]){
    			ans++;
    			dfs(i);
    		}
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
