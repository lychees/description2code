#include <iostream>
using namespace std;

int main()
{
    int i,n,q;
    long long cnt[27]={0};
    char str[1000002],c,d;
    scanf("%d %s",&n,str);
    for(i=0;str[i]!='\0';i++)
    	cnt[str[i]&31]++;
    scanf("%d",&q);
    while(q--){
    	c=getchar();
    	while(c<'a' || c>'z')
    		c=getchar();
    	d=getchar();
    	while(d<'a' || d>'z')
    		d=getchar();
    	if(c==d)
    		printf("%lld\n",(cnt[c&31]*(cnt[c&31]-1))/2);
    	else
    		printf("%lld\n",cnt[c&31]*cnt[d&31]);
    }
    
    return 0;
}
