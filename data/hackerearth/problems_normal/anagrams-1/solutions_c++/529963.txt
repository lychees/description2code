#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,i,j,fre[26],odd,oddv,ag,bg;
    char a[105],b[105];
    cin>>t;
    while(t--)
    {
    	getchar();
    	scanf("%[^\n]s",a);
    	getchar();
    	scanf("%[^\n]s",b);
    	ag=bg=odd=0;
    	memset(fre,0,sizeof(fre));
    	for(i=0;a[i];i++)
    	{
    		if(a[i]>='a' && a[i]<='z')
    			fre[a[i]-'a']++;
    		else if(a[i]>='A' && a[i]<='Z')
    			fre[a[i]-'A']++;
    	}
    	for(i=0;b[i];i++)
    	{
    		if(b[i]>='a' && b[i]<='z')
    			fre[b[i]-'a']--;
    		else if(b[i]>='A' && b[i]<='Z')
    			fre[b[i]-'A']--;
    	}
    	for(i=0;i<26;i++)
    	{
    		if(fre[i]>0)
    			ag=1;
    		else if(fre[i]<0)
    			bg=1;
    		if(fre[i]&1)
    				odd++;
    	}
    	if(!ag && !bg)
    	puts("YES");
    	else if(ag&&bg || odd>1)
    	puts("NO LUCK");
    	else 
    	{
    		
    		for(i=0;i<26;i++)
    		{
    			if(!fre[i])
    				continue;
    			if(fre[i]<0)
    				fre[i]=-fre[i];
    			if(fre[i]&1)
    				oddv=i;
				for(j=0;j<fre[i]/2;j++)
					printf("%c",i+'a');
    		}
    		if(odd)
    				printf("%c",oddv+'a');
    		for(i=25;i>=0;i--)
    		{
    			if(!fre[i])
    				continue;
				for(j=0;j<fre[i]/2;j++)
					printf("%c",i+'a');		
    		}
    		puts("");
    	}
    }
    return 0;
}
