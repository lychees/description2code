#include <bits/stdc++.h>
using namespace std;

int A[100005];

int main()
{
    int N;
    scanf("%d", &N);
    
    for(int i=1; i<=N; i++)
    {
    	int x;
    	scanf("%d", &x);
    	
    	for(int j=2; j<=sqrt(x); j++)
    	{
    		while(x%j==0)
    		{
    			A[j]++;
    			x/= j;
    		}
    	}
    	
    	if(x>1)
    		A[x]++;
    }
    
    int Q;
    scanf("%d", &Q);
    
    while(Q--)
    {
    	int x;
    	scanf("%d", &x);
    	
    	long long ans = 1000001;
    	
    	for(int i=2; i<=sqrt(x); i++)
    	{
    		int countx = 0;
    		
    		while(x%i==0)
    		{
    			countx++;
    			x/= i;
    		}
    		
    		if(countx)
	    		ans = min(ans, (long long)A[i]/countx);
    	}
    	
    	if(x>1)
	    	ans = min(ans, (long long)A[x]);
    	
    	if(ans>N)
    		ans = 0;
    	
    	printf("%lld\n", ans);
    }
    
    return 0;
}
