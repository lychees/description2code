#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		long long int n;
		cin >> n;
		int fl=0;
		for(int i=2;i*i<=n && n>0;i++)
		{
			int ct=0;
			while(n%i==0)
			{
				ct++;
				n/=i;
			}
			if(ct>1)
			{
				fl=1;
				break;
			}
			
		}
		if(fl)
		cout << "No" << endl;
		else
		cout << "Yes" << endl;
	}
    return 0;
}
