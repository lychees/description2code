#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,p1,p2,q1,q2,tmp,pos;cin>>t;
    long long ans;
    vector<int>v(1000001);
    for(int i=0;i<1000001;i++)v[i]=i;
    pos=tmp=2;
    while(pos<1000001){
    	for(int i=tmp+1;i<=min(v[pos]+pos-1,1000000);i++)v[i]=pos;
    	tmp=v[pos]+pos-1;
    	pos++;
    }
    while(t--){
    	cin>>p1>>p2>>q1>>q2;
    	ans=0;
    	for(int i=p1;i<=p2;i++){
    		tmp=max(q1,v[i]);
    		pos=min(q2,v[i]+i-1);
    		ans+=(q2-q1+1)-max(0,pos-tmp+1);
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
