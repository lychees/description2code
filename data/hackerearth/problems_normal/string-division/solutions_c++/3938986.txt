#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t,n,i,j,k,l;
	cin >> t;
	while(t--){
		string s;
		cin >> s;
		int l=s.size();
		int temp=0;
		if(l>=10){
			temp=1;
		}
		else{
			string s1,s2,s3,s4;
			for(i=1;i<l;i++){
				for(j=i+1;j<l;j++){
					for(k=j+1;k<l;k++){
						s1=s.substr(0,i);
						s2=s.substr(i,j-i);
						s3=s.substr(j,k-j);
						s4=s.substr(k,l-k);
						if(s1!=s2 && s1!=s3 && s1!=s4 && s2!=s3 && s2!=s4 && s3!=s4 ){
							temp=1;
						}
					}
				}
			}
		}
		if(temp==0){
			cout << "NO" << endl;
		}
		else{
			cout << "YES" << endl;
		}
	}
    return 0;
}
