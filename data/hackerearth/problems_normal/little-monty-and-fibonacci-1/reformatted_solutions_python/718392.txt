import sys
t = int(sys.stdin.readline())
while (t > 0):
    n = int(sys.stdin.readline())
    sum = 0
    if (n == 1):
        sum = 1
    elif (n == 2):
        sum = 2*n
    elif (n > 2):
        sum = 2*n
        i = 3
        while (i <= n):
            tmp = n/i
            sum += tmp
            i = i+1
    print sum
    t = t-1
