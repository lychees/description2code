#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define mod 1000000007
ll arr[1000006] ;
int main(){
    arr[1] = 3 ;
    for(ll i = 2 ; i <= 1000000 ; i++){
        arr[i] = (arr[i-1]*i)%mod ;
    }
    int T , n ;
    cin>>T ;
    while(T--){
        cin >> n ;
        cout << (arr[n] + 3)%mod << endl ;
    }
}
