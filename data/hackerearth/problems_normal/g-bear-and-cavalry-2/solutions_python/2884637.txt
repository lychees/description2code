import fileinput


def solve(w, h):
    h.sort()
    bb = w.pop(0)*h.pop()
    w.sort()
    while len(w) > 0:
        if w.pop(0)*h.pop() >= bb:
            return "NO"
    return "YES"

first = True
getn = False
getw = False
geth = False
for line in fileinput.input():
    if first:

    elif getn:
        getn = False
        b = int(line)
        getw = True
    elif getw:
        getw = False
        w = map(int, line.split())
        geth = True
    elif geth:

        h = map(int, line.split())
