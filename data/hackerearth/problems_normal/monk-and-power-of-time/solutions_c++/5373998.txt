#include <iostream>
using namespace std;

int main()
{
    int cnt,i,j,temp,n,ttime=0;
    cin>>n;
    cnt=n;
    int call[100],ex[100];
    for(i=0;i<n;i++)
    	cin>>call[i];
    for(i=0;i<n;i++)
    	cin>>ex[i];
 	while(cnt>0)
    {
    	if(call[0]==ex[0])
    	{
    		cnt--;
    		ttime++;
    		for(i=0;i<n-1;i++)
    		{
    			call[i]=call[i+1];
    			ex[i]=ex[i+1];
    		}
    		n=n-1;
    		continue;
    	}
    	temp=call[0];
    	j=0;
    	while(call[n-2]!=call[n-1])
  		{
    		call[j]=call[j+1];
    		j++;
    	}
    	call[n-1]=temp;
    	ttime++;
    }
    cout<<ttime;
    return 0;
}
