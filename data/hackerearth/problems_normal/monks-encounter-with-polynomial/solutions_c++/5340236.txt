#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <vector>

using namespace std;

int n;
vector <int> a;

int main()
{
    int a,b,c,n,k;
    scanf("%d",&n);
    for(int i=1;i<=n;++i)
    {
       int x=0;
       scanf("%d%d%d%d",&a,&b,&c,&k);
       while(a*x*x+b*x+c<k){
        x++;
       }
       printf("%d\n",x);
    }

    return 0;
}
