#include<bits/stdc++.h>
using namespace std;

int prime[100012];

long long int phi(long long x)
 {
   long long int ret = 1,i,pow;
   for (i = 2; x != 1; i++)
   {
     pow = 1;
     if(i>sqrt(x))break;
     while (!(x%i))
     {
       x /= i;
       pow *= i;
     }
     ret *= (pow - (pow/i));
    }
    if(x!=1)ret*=(x-1);
    return ret;
}
void euler()
{
      for(int i=1;i<=100000;i++)
        prime[i]=1;
    prime[1]=0;
    for(int i=2;i<=sqrt(100000);i++)
    {
        for(int j=i*i;j<=100000;j+=i)
        {
            prime[j] = 0;
        }
    }
}
int main()
{
    int t;
    euler();

    scanf("%d",&t);
    for(int k=1;k<=t;k++)
    {

           int x = phi(n);
           //cout<<x<<endl;

    }

}
