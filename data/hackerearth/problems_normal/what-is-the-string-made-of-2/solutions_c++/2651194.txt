    #include <iostream>
    using namespace std;
     
    int main()
    {
        string s;
        cin>>s;
        int num=0;
        for(int i=0;i<s.size();i++)
        {
        	if(s[i]=='8')
        	num=num+7;
        	else if(s[i]=='0'||s[i]=='6'||s[i]=='9')
        	num=num+6;
        	else if(s[i]=='2'||s[i]=='3'||s[i]=='5')
        	num=num+5;
        	else if(s[i]=='4')
        	num=num+4;
        	else if(s[i]=='7')
        	num=num+3;
        	else
        	num=num+2;
        	
        }
        cout<<num<<endl;
        
        return 0;
    }
