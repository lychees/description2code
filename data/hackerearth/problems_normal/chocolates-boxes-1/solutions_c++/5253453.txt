#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;


int main()
{
    int t,i,j;
    cin>>t;
    while(t--)
    {
    	long int n,k,ans=0;
    	cin>>n;
    	int arr[n],arr1[101];
    	for(i=0;i<n;i++)
    	cin>>arr[i];
    	cin>>k;
    	//sort(arr,arr+n);
    	for(i=0;i<=100;i++)
    	arr1[i]=0;
    	
    	for(i=0;i<n;i++)
    	{
    		arr1[ arr[i] ]++;
    	}
    	if(k<=200)
    	{
    		for(i=100;2*i>=k;i--)
    		{
    			if(i!=k-i){
    			if(k-i>0)
    			ans += arr1[i]*arr1[k-i];
    			}
    			else ans += (arr1[i]*(arr1[i]-1))/2;
    		}
    	}
    	
    	cout<<ans<<endl;
    }
    return 0;
}
