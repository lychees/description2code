
#include <bits/stdc++.h>
using namespace std ;
int main()
{
  char a[100000] ;
  cin >> a ;
  int l ;
  l = strlen(a) ;
  int r = 0;
  int i ;
  for ( i = 0 ; i< l ; i++)
  {
   if ( a[i] == '<')
   {
     i = i+1;
     if ( a[i] == '3')
     {
       r = r+1;
     }
   }
  }
  cout << r <<endl;
  return 0 ;
}
