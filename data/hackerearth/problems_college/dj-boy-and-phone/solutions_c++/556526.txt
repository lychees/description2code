#include <iostream>
#include<vector>
using namespace std;
#define LL long long int
#define mod 1000000009
int main()
{
	vector<int> v[10];
	v[1].push_back(2);
    v[1].push_back(4);
    v[2].push_back(1);
    v[2].push_back(5);
    v[2].push_back(3);
    v[3].push_back(2);
    v[3].push_back(6);
    v[4].push_back(1);
    v[4].push_back(5);
    v[4].push_back(7);
    v[5].push_back(2);
    v[5].push_back(4);
    v[5].push_back(6);
    v[5].push_back(8);
    v[6].push_back(3);
    v[6].push_back(5);
    v[6].push_back(9);
    v[7].push_back(4);
    v[7].push_back(8);
    v[8].push_back(5);
    v[8].push_back(7);
    v[8].push_back(9);
    v[8].push_back(0);
    v[9].push_back(6);
    v[9].push_back(8);
    v[0].push_back(8);
    int t;
    cin>>t;
    while(t--)
    {
    	LL n,i,j,k;
    	cin>>n;
    	LL d[10];
    	if(n==0){cout<<0<<endl;continue;}
    	for(i=0;i<=9;i++)d[i]=1;
    	for(i=2;i<=n;i++)
    	{
    		//for(j=0;j<=9;j++)cout<<"d[i]="<<d[i]<<endl;
    		LL e[10]={0};
    		for(j=0;j<=9;j++)
    		{
    			for(k=0;k<v[j].size();k++)
    			{
    				e[j]+=d[v[j][k]];
    				if(e[j]>mod)e[j]%=mod;
    			}
    		}
    		for(j=0;j<=9;j++)
    		{
    			d[j]+=e[j];
    			if(d[j]>mod)d[j]%=mod;
    		}
    	}
    	LL ans=0;
    	for(i=0;i<=9;i++)ans=(ans+d[i])%mod;
    	cout<<ans<<endl;
    }
    return 0;
}
