#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    int t, i;
    cin>>t;
    while(t>0)
    {
    	char str[100000];
    	scanf("%s", str);
    	int j, flag=0;
    	j=strlen(str)-1;
    	for(i=0; i<strlen(str)/2; i++)
    	{
    		if(str[i]!=str[j])
    		{
    			flag= 1;
    			break;
    		}
    		else
    		j--;
    	}
    	if(flag==1)
    	{
    		cout<<"-1\n";
    	}
    	else
    	printf("%d\n", str[i]);
    	t--;
    }
    return 0;
}
