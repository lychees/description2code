#include <iostream>
using namespace std;

struct rect {
	int bottomleftx, bottomlefty, toprightx, toprighty;
};

int max(int a, int b) {
	return (a>b) ? a : b;
}

int min(int a, int b) {
	return (a>b) ? b : a;
}

int main()
{
	int t;
	cin>>t;
	struct rect rect1, rect2, inter;
	while(t--) {
		
		cin>>rect1.bottomleftx>>rect1.bottomlefty>>rect1.toprightx>>rect1.toprighty;
		cin>>rect2.bottomleftx>>rect2.bottomlefty>>rect2.toprightx>>rect2.toprighty;
		inter.bottomleftx = max(rect1.bottomleftx, rect2.bottomleftx);
		inter.bottomlefty = max(rect1.bottomlefty, rect2.bottomlefty);
		inter.toprightx = min(rect1.toprightx, rect2.toprightx);
		inter.toprighty = min(rect1.toprighty, rect2.toprighty);
		
		if(inter.bottomleftx == inter.toprightx && inter.bottomlefty == inter.toprighty) {
			cout<<"Point\n";
		}
		
		else if(inter.bottomleftx > inter.toprightx || inter.bottomlefty > inter.toprighty) {
			cout<<"Apart\n";
		}
		
		else if(inter.bottomleftx == inter.toprightx || inter.bottomlefty == inter.toprighty) {
			cout<<"Line\n";
		}
		else
			cout<<"Rectangle\n";
		
	}
    return 0;
}
