#include <iostream>
using namespace std;

int main()
{
	long long int A,B,i,j,max,temp;
	 int T,k;
    cin>>T;
    for(k=0;k<T;k++)
    {
    cin>>A;
    cin>>B;
    max=0;
    if(B-A > 2)	A = B-2; 
    for(i=A;i<=B;i++)
    {
    	for(j=i+1;j<=B;j++)
    	{
    		temp = i&j;
    		if(temp > max)
    			max = temp;
    	}
    }
    cout<< max << endl;
    }
}
