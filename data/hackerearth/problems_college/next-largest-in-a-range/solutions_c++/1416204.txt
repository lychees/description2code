#include <iostream>
using namespace std;

#define sd(x) scanf("%d", &x)

int a[1123456];

int main()
{
    int n, l, r, q, i;
    cin>>n>>l>>r>>q;
    for(i = 0; i < n; i++){
    	sd( a[i] );
    }
    int x, ans;
    bool flag;
    while(q--){
    	sd(x);
    	ans = 1123456;
    	flag = 0;
    	for(i = l; i <= r; i++){
    		if( a[i] > x  ){
    			flag = 1;
    			ans = min(a[i], ans);
    		}
    	}
    	if(flag == 0) ans = -1;
    	printf("%d\n", ans);
    }
    return 0;
}
