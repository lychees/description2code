#include<bits/stdc++.h>

int main()
{
	int num,h[100],m[100],h1,h2,m1,m2,i,flag = 0;
	cin >> num;
	for(i = 0;i < num;i++)
	{
		scanf("%d:%d-%d:%d",&h1,&m1,&h2,&m2);
		h[i] = h1*60 + m1;
		m[i] = h2*60 + m2;
	}
	for(i = 0;i < num-1;i++)
	{
		if((h[i+1] >=h[i] && h[i+1] < m[i])||(h[i+1] <=h[i] && m[i+1] >= h[i]))
		{
			printf("Will need a moderator!\n");
			flag =1 ;
			break;
		}
	}
	if(flag == 0)
	{
		printf("Who needs a moderator?\n");
	}
	return 0;
}
