cases = int(raw_input())

for case in range(cases):
    string = raw_input()
    x = set(list(string))
    a = {}
    for i in x:
        if string.count(i) in a:
            a[string.count(i)].append(i)
        else:
            a[string.count(i)] = [i]
    s = ''
    for i in sorted(a.keys()):
        for j in sorted(a[i]):
            s += j * i
    print s
