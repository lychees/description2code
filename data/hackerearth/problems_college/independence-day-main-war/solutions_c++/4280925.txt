#include <iostream>
#include <iomanip>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		float n,x,y;
		float sx=0.0,sy=0.0,ansx,ansy;
		cin>>n;
		int n1=n;
		while(n--)
		{
		cin>>x>>y;
		sx+=x;
		sy+=y;
		}
		ansx=sx/n1;
		ansy=sy/n1;
		cout<<"("<<setprecision(2)<<fixed<<ansx<<" , "<<setprecision(2)<<fixed<<ansy<<")"<<endl;
	}
}
