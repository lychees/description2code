#include<iostream>
#include<math.h>
using namespace std;

long long powers[100009], elems[100009], pow2[40];

long long myfunc (long m, long n)
{
	long long l,r,mid;
	if (m==n)
	return powers[m];
	
	mid = (m+n)/2;
	l = myfunc(m,mid);
	r = myfunc(mid+1,n);
	
	return (l+r+l*r)%1000000007;
}

int main()
{
	long long n,i,j,k,m,sum;
		
	cin>>n;
	for(i=0;i<n;i++)
	cin>>elems[i];
	
	pow2[0]=1;
	pow2[1]=2;
	
	for(i=2;i<=35;i++)
	pow2[i]=(pow2[i-1]*pow2[i-1])%1000000007;
	
	for(i=0;i<n;i++)
	{
		powers[i]=1;
		m=elems[i];
		while(m>0)
		{
			k=1;
			j=1;
			while(j*2<=m)
			{
				j*=2;
				k++;
			}
			powers[i]=(powers[i]*pow2[k])%1000000007;
			m-=j;
		}
	}
	
	cout<<myfunc(0,n-1)<<endl;
	
	return 0;	
}
