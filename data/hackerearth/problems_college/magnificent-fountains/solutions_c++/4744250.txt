#include <bits/stdc++.h>
#include <boost/multiprecision/cpp_int.hpp>
namespace mp = boost::multiprecision;
using namespace mp;

using namespace std;


cpp_int hcf(cpp_int  a, cpp_int b)
{
	if(b==0)
		return a;
	return hcf(b,a%b);
}

void lcm(cpp_int *a,int n)
{
	mp::cpp_int l=1;
	int i;
	for(i=0;i<n;i++)
	{
		l=l*a[i]/hcf(l,a[i]);
	}
	cout<< l%1000000007 <<endl;
}

int main()
{
	int t,n;
	cpp_int a[10000];
	cin>>t;

	while(t--)
	{
		cin>>n;
		int i;
		for(i=0;i<n;i++)
			cin>>a[i];
		lcm(a,n);
		
	}
	return 0;
}
