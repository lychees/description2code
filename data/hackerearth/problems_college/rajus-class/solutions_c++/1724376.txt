#include<bits/stdc++.h>
using namespace std;
typedef struct node
{
int age;
char name[100];
}NODE;
NODE entry[1000];
bool fn(NODE x,NODE y)
{
     return x.age<y.age;
}
int main()
{
int t,n,i,r;
cin>>t;
while(t--)
{
cin>>n;
for(i=0;i<n;i++)
cin>>entry[i].age>>entry[i].name;
sort(entry,entry+n,fn);

cin>>r;
cout<<entry[r-1].name<<endl;
}
return 0;
}
