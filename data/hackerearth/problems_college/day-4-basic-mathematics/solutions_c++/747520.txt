#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        long long int d;
        long long int ans=0;
        scanf("%lld",&d);
        double det=sqrt(1+8*d);
        det=det-1;
        det=det/2;
        ans=ceil(det);
        if(ans%3==0)
        {
            printf("SL\n");
        }
        else if(ans%3==1)
        {     
            printf("LB\n");
        }
        else if(ans%3==2)
        {
            printf("BS\n");
        }
    }
return 0;
}
