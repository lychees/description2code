	
	#include <bits/stdc++.h>
	   
	using namespace std;
		
	map<int,int> m;
		
	int main(){
		
			//if(m.find(1)!=m.end())
			int t;
			cin>>t;
			while(t--){
				m.clear();
				int cnt=0;
				int N;
				cin>>N;
				for(int i=0;i<N;i++){
				int a,b;
				cin>>a>>b;
				if(m[a])
				m[a]++;
				else
				m[a]=1;
				if(m[b])
				m[b]--;
				else
				cnt++;
			}
			cout<<cnt<<endl;
			}
			
		
	}
