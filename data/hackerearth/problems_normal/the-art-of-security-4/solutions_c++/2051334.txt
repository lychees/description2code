int n,arr[10],e1,e2,ans;

void backtrack(int idx)
{
	if(idx==n)
	{
		ans++;
		return ;
	}
	if(e1+arr[idx]<=e2)
	{
		e1+=arr[idx];
		backtrack(idx+1);
		e1-=arr[idx];
	}
	e2+=arr[idx];
	backtrack(idx+1);
	e2-=arr[idx];
}

	}

}
