#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	int n;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		int a[n];
		for(int i=0;i<n;i++)
			scanf("%d",&a[i]);
		sort(a,a+n);
		int i=0,j=n-1;
		while(i<=j)
		{
			while(a[i]==a[i+1])
				i++;
			while(a[j]==a[j-1])
				j--;	
			if(a[i]!=a[j])	
				printf("%d %d ",a[j],a[i]);
			else
				printf("%d ",a[i]);
			i++;
			j--;
		}
		printf("\n");	
	}
	return 0;
}
