#include <iostream>
#include <cmath>
using namespace std;

long long findTrailingZeros(long long  n)
{
    long long count = 0LL;
    long long i;
    for (i=5; n/i>=1; i *= 5)
          count += n/i;
    return count;
}

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long n;
		cin>>n;
		if(n==0||n==1||n==2||n==3) 
		printf("1\n");
		else{
			long long ans = (long long int)floor((log(2*acos(-1)*n)/2+n*(log(n)-1))/log(10))+1;
			ans=ans-findTrailingZeros(n);
			printf("%lld\n",ans);
		} 
	}
	return 0;
}
