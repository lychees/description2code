t = int(raw_input())
while t:
    n = int(raw_input())
    arr = [int(i) for i in raw_input().split()]
    ans = 1
    if n == 1:
        print '1'
        t -= 1
        continue
    m = arr[0]
    for i in range(1, n):
        if arr[i] <= m:
            m = arr[i]
            ans += 1
    print ans
    t -= 1
