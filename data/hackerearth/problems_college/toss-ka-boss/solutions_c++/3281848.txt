#include <iostream>
using namespace std;

int main()
{
   long int t,i,l,count,p;
   string s;
   cin>>t;
   while(t--)
   {
   	count=0;
   	cin>>s;
   	l=s.size();
   	for(i=0,p=1;i<l;i++,p++)
   	{
   		if(s[i]=='H')
   		{
   		 count=count+p;	
   		}
   		else
   		p=0;
   	}
   	cout<<count<<endl;
   }
    return 0;
}
