#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int p[2001],n[2001];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int i;
	p[1]=1;n[1]=1;
	for(i=2;i<2001;i++)
	{
		n[i]=n[i-1]+p[i-1];
		p[i]=n[i-1];
	}
	ll int test;
	cin>>test;
	while(test--)
	{
		ll int num,ans;
		cin>>num;
		ans=p[num]+n[num];
		ans=ans*ans;
		cout<<ans<<endl;
	}
	return 0;
}
