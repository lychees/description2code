#include <stdio.h>
int gcd(int a,int b)
{
	if(b!=0)
	return gcd(b,a%b);
	else
	return a;
}
int main()
{
    //printf("Hello World!\n");
    int ans1,ans2,ans3,a,b,c,t,flag=1;
    scanf("%d",&t);
    while(t--)
    {
    	flag=0;
    	scanf("%d %d %d",&a,&b,&c);
    	
    	if((a*a+b*b)==c*c || (a*a+c*c)==b*b || (c*c+b*b)==a*a)
    		flag=1;
    	ans1=gcd(a,b);
    //	printf("%d\n",ans1);
    	if(ans1==1)
    	ans2=gcd(b,c);
    //	printf("%d\n",ans2);
    	if(ans1==1 && ans2==1)
    	ans3=gcd(a,c);
    //	printf("%d\n",ans3);
    	if(ans1==1 && ans2==1 && ans3==1 && flag)
    	printf("YES\n");
    	else
    	printf("NO\n");
    	
    }
    return 0;
}
