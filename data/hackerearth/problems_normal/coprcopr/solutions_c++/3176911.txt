#include<iostream>
#include<cmath>
using namespace std;

int main()
{
    static bool pr[10000009];
    static int lp[10000009];
    static long long int phi[10000009];

    int t;
    cin>>t;

    for(int i=2;i<=sqrt(10000000);i++)
    {
        if(pr[i]==1)
            continue;
        lp[i]=i;
        for(int j=2;i*j<=10000000;j++)
        {
            pr[i*j]=1;

            if(lp[i*j]==0)
                lp[i*j]=i;
        }
    }

    phi[1]=1;
    for(int i=2;i<=10000000;i++)
    {
        if(pr[i]==0)
            phi[i]=i-1;

        else
        {
            if(lp[i]==lp[i/lp[i]])
                phi[i]=phi[i/lp[i]]*lp[i];

            else
                phi[i]=phi[i/lp[i]]*(lp[i]-1);
        }
    }

    for(int i=2;i<=10000000;i++)
        phi[i]+=phi[i-1];

   while(t--)
    {
        int n;
        cin>>n;
        cout<<phi[n]<<"\n";
    }
}
