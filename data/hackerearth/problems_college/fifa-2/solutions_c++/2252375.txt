#include<bits/stdc++.h>
using namespace std;
int main(){

     int n;
     cin >> n;

     int hsh[100001]={0};

     int ar[n+1][2];
     for(int i = 1; i <= n;  ++i )
      {
       cin >> ar[i][0] >> ar[i][1];

        hsh[ar[i][0]]++;
      }

       for(int i = 1 ; i <= n ; ++i ){

          int b = hsh[ar[i][1]];

          int diff = ( n - 1) - b;
          int a =  (n-1) + b;

          cout << a <<" "<<diff << endl;

       }



}
