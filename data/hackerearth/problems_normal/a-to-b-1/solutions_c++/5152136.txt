#include <iostream>
using namespace std;
typedef long long ll;
ll hi = 1000000007;
ll po(ll a,ll b){
if(b==0)return 1;
ll x = po(a,b/2);
x=(x*x)%hi;
if(b%2)x=(x*a)%hi;
return x;
}
ll a[100005];
int main()
{
    int n;
    scanf("%d",&n);
    int i;
    ll ans = 1;
     int total = 0;
    for(i=1;i<=n;i++)
    {
    	scanf("%lld",&a[i]);
    	if(a[i])
    	ans = (ans*a[i])%hi;
    	else total++;
    }
    int m;
   
    scanf("%d",&m);
    while(m--){
    	int u;
    	scanf("%d",&u);
    	if(u==0){
    		int x;
    		ll y;
    		scanf("%d%lld",&x,&y);
    		if(a[x]){
    			if(y==0)total++;
    			ll rev;
    			
    			rev = po(a[x],hi-2);
    			ans = (ans*rev)%hi;
    		}
    		else if(y)total--;
    		a[x] = y;
    		if(y)
    		ans = (ans*y)%hi;
    	
    	}
    	else {
    		int x;
    		scanf("%d",&x);
    		if(a[x]){
    	ll res;
    	res = po(a[x],hi-2);
    	res = (res*ans)%hi;
    	if(total)res = 0;
    		printf("%lld\n",res);
    		}
    		else if(total==1)printf("%lld\n",ans);
    		else printf("0\n");
    	}
    }
    return 0;
}
