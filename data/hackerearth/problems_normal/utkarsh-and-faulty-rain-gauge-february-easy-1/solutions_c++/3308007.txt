int main()
{
	ll t;
	double res,n,r;
	scanf("%lld",&t);
	res=1;
	while(t--)
	{
		scanf("%lf %lf",&n,&r);
		res*=(1.5*(n+1)*n)/(2*n+1)/r;
	}
	printf("%.4lf\n",res);
