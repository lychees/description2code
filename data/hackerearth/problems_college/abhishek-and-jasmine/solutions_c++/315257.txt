int main()
{
     char n;
   
    scanf("%d",&n);

    for(int i=0;i<n;i++)
    {
            scanf("%d",&a[i]);
            x[i]=0;
    }
    int r=0;
    for(int i=0;i<n-1;i++)
    {
            for(int j=i+1;j<n;j++)
            {
                    x[i]|=a[i]^a[j];
            }
            r=r|x[i];
    }
