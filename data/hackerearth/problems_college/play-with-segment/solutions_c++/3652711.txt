#include <iostream>
using namespace std;

int main()
{
	int t;
	scanf("%d", &t);
	while(t--) {
		int n;
		scanf("%d", &n);
		
		int ar[500001];
		
		for(int i = 0; i < n; i++) {
			scanf("%d", &ar[i]);
		}
		
		int l = 0;
		int r = 0;
		int maxl = 1, min = ar[l], max = ar[l];
		
		while(r < n - 1) {
			
			if(ar[r + 1] < min) {
				min = ar[r + 1];
			}
			if(ar[r + 1] > max) {
				max = ar[r + 1];
			}
			
			if(abs(max - min) < 2) {
				r++;
			} else {
				l++;
				r = l;
				min = ar[l];
				max = ar[l];
				continue;
			}
			if (r - l + 1 > maxl) {
				maxl = r - l + 1;
			}
		}
		printf("%d\n", maxl);
	}
    return 0;
}
