# include <bits/stdc++.h>
using namespace std;
int main()
{

	int t;
	cin>>t;
	while(t--)
	{
		char a[10000], b[10000];
		scanf("%s %s",a, b);
		int l=strlen(a);
		int m=strlen(b);
		int hasha[26]={0}, hashb[26]={0};
		for(int i=0;i<l;i++)
		{
			hasha[a[i]-'a']++;
		
		}
		for(int i=0;i<m;i++)
			hashb[b[i]-'a']++;
		int i;
		for(i=0;i<26;i++)
		{
			if(hasha[i]!=hashb[i])
			{
				break;
			}
		}
		if(i==26)
			cout<<"Yes"<<endl;
		else 
			cout<<"No"<<endl;
	}
	return 0;
}
