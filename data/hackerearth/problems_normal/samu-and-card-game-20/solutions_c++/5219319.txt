#include <iostream>
using namespace std;

int main()
{
   int n ; cin>>n;
   for (int i=1;i<=n;i++)
   {
   	int n,m;cin>>n>>m;
   	int k;cin>>k;
   	int x=1,y=1,jump=0;
   	for (int j=1;j<=k;j++)
   	{
   		int a,b;cin>>a>>b;
   		for (int i=0;i<=(m+n);i++)
   		{
   			if(((x+a)>=1&&(x+a)<=n)&&((y+b)>=1&&(y+b)<=m))
   			{
   				x+=a;y+=b;jump++;
   			}
   			else break;
   		}   
   	}
   	cout<<jump<<endl;
   }
}
