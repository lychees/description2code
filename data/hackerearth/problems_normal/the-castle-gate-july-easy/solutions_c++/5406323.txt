#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int k=0;k<t;k++){
    	int count=0;
    	int n;
    	cin>>n;
    	for(int i=1;i<n;i++){
    		for(int j=i+1;j<=n;j++){
    			if((i^j)<=n){
    				count++;
    			}
    		}
    	}
    	cout<<count<<endl;
    }
    return 0;
}
