#include<bits/stdc++.h>
//zone
using namespace std;
int mrk[1005][1005];
int a[1005][1005];
int s = 0;
 
int n,m;
int bfs(int x, int y)
{
	int i, j;
	mrk[x][y] = 1;
 	queue< pair <int, int> > q;
 	q.push(make_pair(x,y));
	 int w,l;
 	while(!q.empty()){
		 w = q.front().first;
		 l = q.front().second;
	 	 q.pop();
	 	for(i = -1; i <= 1; i++) {
	 		for(j = -1 ; j <= 1; j++) {
				int op = w + i;
				int po = l + j;
				if(op < 0 || po < 0 || op > n || po > m) 		
					continue;
	  			if(!mrk[w+i][l+j] && a[w+i][l+j] == 1) {
	 				 s++;
	  				q.push(make_pair(w+i,l+j));
	  				mrk[w+i][l+j] = 1;
	  			}
	  		}
  	}
//  	cout<<s<<endl;
  }
  return s;
}
 
int main()
{
	int t;
	cin >> t;
	int no = 0;
	int ans = 0;
	while(t--) {
		memset(mrk,0,sizeof(mrk));
		//memset(a,0,sizeof(a));
		ans = 0;
		cin >> n >> m;
		s = 0,no = 0;
		int i, j;
 
		for(i = 1; i <=n ; i++) 
			for(j = 1; j <= m ; j++) 
			 	cin >> a[i][j];
			
		
		for(i = 1 ; i <= n  ; i++) {
			for(j = 1; j <= m ; j++) {
				 if(!mrk[i][j] && a[i][j] == 1) {
					 s = 1;
					  ans = max(ans,bfs(i,j));
					  no++;
				  }
			 }
		}
	       cout<<no<<" " <<ans<<endl;
	}
  return 0;
 }
