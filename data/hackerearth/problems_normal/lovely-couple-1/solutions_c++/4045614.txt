#include <iostream>
#include <vector>
#include <map>
using namespace std;

int main()
{
    int p1,p2,t,a,b;
    vector<int> sieve(10000,1),prime;
    for(p1=2;p1<10000;p1++)
    {
    	if(sieve[p1]==0)
    		continue;
    	prime.push_back(p1);
    	for(p2=2*p1;p2<10000;p2+=p1)
    		sieve[p2]=0;
    } 
    sieve[1]=0;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b;
    	map<int,int> mp1;
    	p1=0;
    	while(a>1)
    	{
    		if(a%prime[p1]==0)
    			mp1[prime[p1]]++;
    		while(a%prime[p1]==0)
    			a/=prime[p1];
    		p1++;
    	}
    	a=b;
    	p1=0;
    	while(a>1)
    	{
    		if(a%prime[p1]==0)
    			mp1[prime[p1]]++;
    		while(a%prime[p1]==0)
    			a/=prime[p1];
    		p1++;
    	}
    	if(sieve[mp1.size()]==1)
    		cout<<"Yes"<<endl;
    	else
    		cout<<"No"<<endl;
    }
    return 0;
}
