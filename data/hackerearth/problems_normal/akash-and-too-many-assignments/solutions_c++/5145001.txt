using namespace std;

#define ll long long int
#define MOD 1000000007

int BIT[26][1000001];

void insert(int a, int x, int val)
{
	for (; x <= 1000000; x += (x&(-x))) 
		BIT[a][x] += val;
}

int query(int a, int x)
{
	int sum = 0;
	for (; x > 0; x -= (x&(-x)))
		sum += BIT[a][x];
	return sum;
}

char str[1000000], ch[2];

int main()
{
	int n, q, i, j, k, a, b, c;
	cin >> n >> q;
	scanf("%s", &str[1]);
	for (i = 1; i <= n; ++i) {
		k = str[i] - 'a';
		insert(k, i, 1);
	}
	while (q--) {
		scanf("%d", &k);
		if (k == 0) {
			scanf("%d %s", &a, ch);
			k = str[a] - 'a';
			insert(k, a, -1);
			str[a] = ch[0];
			k = str[a] - 'a';
			insert(k, a, 1);
		} else {
			scanf("%d %d %d", &a, &b, &c);
			k = 0;
			for (i = 0; i < 26; ++i) {
				k += (query(i, b) - query(i, a-1));
				if (k >= c) break;
				//cout << k << endl;
			}
			if (i < 26) printf("%c\n", i+'a');
			else printf("Out of range\n");
		}
	}
	
	return 0;
}
