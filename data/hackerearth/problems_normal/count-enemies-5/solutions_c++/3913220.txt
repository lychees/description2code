#include <iostream>
#include <string>
#include <cmath>
using namespace std;

int main()
{
    int n;
    cin >> n;
    for(int k=0; k<n;k++)
    {
        string s;
        cin >> s;
        string X ("XO*D");
        int i = 0;
        while(i<s.length())
        {
            if(s[i]==X[0])
            {
                int cond = 0;
                int j=i-1;
                while(cond==0 && j>=0)
                {
                    if(s[j]==X[1])
                    {
                        s[j]=X[3];
                    }
                    else if(s[j]==X[2])
                    {
                        break;
                    }
                    else{break;}
                    j--;
                }
                cond=0;
                j=i+1;
                while(cond==0 && j<s.length())
                {
                    if(s[j]==X[1])
                    {
                        s[j]=X[3];
                    }
                    else if(s[j]==X[2])
                    {
                        break;
                    }
                    else{break;}
                    j++;
                }
            }
            i++;
        }
        int ans=0;
        for(int l=0; l<s.length();l++)
        {
            if(s[l]==X[1])
            {
                ans++;
            }
        }
        cout << ans<<endl;
    }
    return 0;
}
