#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    char str[51];
    cin>>str;
    int i,vow_count=0,con_count=0,flag1=0,flag0=0,flag2=0,flag3=0;
    int len=strlen(str);
    
    for(i=0;i<len;i++)
    {
     if(str[i]=='a'||str[i]=='e'||str[i]=='i'||str[i]=='o'||str[i]=='u'||str[i]=='y')
     {
     vow_count++;
     flag2=0;
     }
     else 
     {con_count++;
     flag2++;
     if(flag2>=3)
     {flag3=1;
     break;}
     
     } 
    }
 	 
    
    if(vow_count<con_count)
    flag0=1;
    else
    flag0=0;
   
    if((flag0==1)||(flag1==1)||(flag3==1))
    cout<<"hard"<<endl;
    else
    cout<<"easy"<<endl;
    }
    return 0;
}
