#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
int main()
{
	//freopen("input.txt","r",stdin);
    //freopen("output.txt","w",stdout);
	long long int a[100007],i,t;
    a[0]=0;a[1]=1;a[2]=2;a[3]=4;
    for(i=4;i<=100000;i++)
    {
    	a[i]=((a[i-1])%mod+(a[i-2])%mod+(a[i-3])%mod)%mod;
	}
	cin>>t;
	while(t--)
	{
		int step;
		cin>>step;
		cout<<a[step]<<endl;
	}
	
    	
    
    return 0;
}
