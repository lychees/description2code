t = int(raw_input())
a = map(int, raw_input().split())
for i in xrange(1, t):
    a[i] += a[i-1]


def binary_search(key, imin=0, imax=None):
    global a
    if imax is None:
        # if max amount not set, get the total
        imax = len(a) - 1

    while imin <= imax:
        # calculate the midpoint
        mid = (imin + imax)//2
        midval = a[mid]

        # determine which subarray to search
        if midval < key:
            # change min index to search upper subarray
            imin = mid + 1
        elif midval > key:
            # change max index to search lower subarray
            imax = mid - 1
        else:
            # return index number
            return mid
    return imin


q = int(raw_input())
while q != 0:
    x = int(raw_input())
    if x > a[-1]:
        print -1
    else:
        print binary_search(x)+1
    q -= 1
