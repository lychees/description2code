#include<bits/stdc++.h>
#define ll long long
using namespace std;
long long int divisors(ll num)
 {
        ll count=0;
        ll sum=0;
   for(ll i=1;i*i<=num;i++)
    {
    if(num%i==0)
      {
     	if(i%4!=0)
     	sum=sum+i;
       if(i!=1)
       {
         if(((i*i)!=num)&&(num/i)!=0)
       	{
       		if((num/i)%4!=0)
			   sum=sum+(num/i);	
		}
       }
     }
    }
  if(num!=1)
    {
    	if(num%4!=0)
    	sum=sum+num;
	}
	return sum;
 }
int main()
{
	ll t,n,val;
	cin>>t;
	while(t--)
	{
		cin>>n;
		val=divisors(n);
		cout<<8*val<<endl; //jacobi's four square theorem
	}
}
