#include <iostream>
using namespace std;

int main()
{
    int n,m;
    cin>>n;
    cin>>m;
    int c[m];
    for(int i=0;i<m;i++)
    	cin>>c[i];
    
    unsigned long int maxVal;
    
    int colorIndex;
    cin>>colorIndex;
    --colorIndex;
    maxVal=c[colorIndex];
    
    int col;
    int found;
    int index;
    
    for(int i=1;i<n;i++)
    {
    	cin>>colorIndex;
    	--colorIndex;
    	col = c[colorIndex];
    	
    	maxVal = maxVal - maxVal%col + col; 
    	found = 1;
    
       	while(found)
    	{
    		index = colorIndex;
    		found = 0;
    		while(++index < m)
    		{
    			if(maxVal%c[index] == 0)
    			{
    				maxVal += col;
    				found =1;
    				break;
    			}
    		}
    	}
    		
    }
    
    cout<<maxVal;
    return 0;
}
