#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	int a[n];
    	for(int i=0;i<n;i++){
    		int x;
    		cin>>x;
    		a[i]=x;
    	}
    	int b[n];
    	for(int i=0;i<n;i++){
    		b[a[i]-1]=i+1;
    	}
    	bool f=true;
    	for(int i=0;i<n;i++){
    		if(a[i]!=b[i]){
    			f=false;
    			break;
    		}
    	}
    	if(f)cout<<"inverse"<<endl;
    	else cout<<"not inverse"<<endl;
    }
    return 0;
}
