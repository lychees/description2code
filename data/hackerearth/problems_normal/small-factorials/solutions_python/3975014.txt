def fact(number):
    if number == 1:
        return 1
    else:
        return number * fact(number-1)

no_of_input = int(input())
inputs = []
for i in range(1, no_of_input+1):
    number = int(input())
    inputs.append(number)

for i in inputs:
    print(fact(i))
