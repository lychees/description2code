#include <bits/stdc++.h>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   	int t;
   	scanf("%d",&t);
   	while(t--){
   		int i,n,ec=0,oc=0;
   		scanf("%d",&n);
   		for(i=0;i<n;i++)
   		{
   			int x;
   			scanf("%d",&x);
   			if(x%2==0)
   			ec++;
   			else
   			oc++;
   		}
   		if((oc%2==0))
   			printf("Yes\n");
   		else
   			printf("No\n");
   		
   		
   	}
   	
   
   
    return 0;
}
