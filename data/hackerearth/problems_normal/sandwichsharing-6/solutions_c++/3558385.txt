#include <iostream>
using namespace std;
int main()
{
    int32_t testCase;
    cin>>testCase;

    if((testCase<=0)||(testCase>=100001))
    {
        return 0;
    }
    else
    {
        while(testCase>0)
        {
            int32_t noOfSides,noOfFriends;
            cin>>noOfSides;
            if((noOfSides<=2)||(noOfSides>=1000000001))
            {
                return 0;
            }
            cin>>noOfFriends;
            if((noOfFriends<0)||(noOfFriends>=1000000001))
            {
                return 0;
            }

            for(int32_t i =0;i<noOfFriends;i++)
            {
                if(noOfSides%2 == 0)
                {
                    noOfSides = (noOfSides/2);
                    ++noOfSides;
                }
                else
                {
                    noOfSides = (noOfSides/2);
                    ++noOfSides;
                    ++noOfSides;
                }
                if(noOfSides == 3)
                {
                	break;
                }
            }
            cout<<noOfSides<<endl;

            --testCase;
        }
    }
}
