#include <bits/stdc++.h>
using namespace std;
int ar[500005];
int main()
{
	int n,k;
	cin>>n>>k;
	for (int i=0; i<n; i++) {
		cin>>ar[i];
	}
	
	int ans = ar[0], maxLen = 0;
	if (ans >= k) maxLen = 1;
	int tempLen = maxLen;
	for (int i=1; i<n; i++) {
		ans = __gcd(ans, ar[i]);
		if (ans >= k) {
			tempLen++;
			maxLen = max(maxLen, tempLen);
		} else {
			tempLen = 1;
			ans = ar[i];
		}
	}
	cout<<maxLen;
	
    return 0;
}
