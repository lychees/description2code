#include <iostream>
#include <bits/stdc++.h>

using namespace std;

void compute()
{
	string s;
	cin>>s;
	int l = s.length();
	int i;
	int count = 0;
	for(i=0;i<l;i++)
	{
		if( s[i]=='-'  && !((i+1<l && s[i+1]=='S') || (i+1<l && s[i+1]=='B') || (i-1>=0 && s[i-1]=='S')|| (i+2<l && s[i+2]=='B')))
		{
		
			count++;							
		}
	}
	cout<<count<<"\n";
}

int main()
{
    int t;
    cin>>t;
    while(t>0)
    {
    	compute();
    	t--;
    }
}
