#include <iostream>
#define modulo 1000000007
using namespace std;

unsigned int a,b,c,d;
unsigned long long int ways[1004][1004],t;

int main()
{
	//Pre Process
	for(int i=0;i<1004;i++)
	{
		ways[0][i]=1;
		ways[i][0]=1;
	}
	for(int i=1;i<1004;i++)
	{
		for(int j=1;j<1004;j++)
		{
			ways[i][j]=(ways[i-1][j-1]+ways[i-1][j]+ways[i][j-1])%modulo;
		}
	}

	//Now accept Test Cases
	cin>>t;
	while(t--)
	{
		cin>>a>>b>>c>>d;
		c=c-a;
		d=d-b;
		cout<<ways[c][d]<<"\n";
	}
	return 0;
}
