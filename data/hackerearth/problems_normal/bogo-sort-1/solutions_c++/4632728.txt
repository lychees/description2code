#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
    int n,temp,i;
    cin>>n;
    vector<long> v;
    for(i=0;i<n;++i)
    {
    	cin>>temp;
    	v.push_back(temp);
    }
    sort(v.begin(),v.end());
    for(i=0;i<n;++i)
    {
    	cout<<v[i]<<endl;
    }
    return 0;
}
