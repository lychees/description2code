#include<iostream>
#include<algorithm>
#include<cstdio>
using namespace std;

int main()
{
	int n;
	scanf("%d", &n);
	int array[n];
	for(int i=0;i<n;i++)
		cin>>array[i];
	// for(int i=0;i<n;i++)cout<<array[i]<<" ";
	// cout<<endl;
	sort(array, array+n);
	// for(int i=0;i<n;i++)cout<<array[i]<<" ";
	// cout<<endl;
	int ii=0,jj=n-1;
	int temp1[n], temp2[n];
	
	int k=0;
	int k1=0;
	while(ii<jj){
		temp1[k]=array[jj];
		temp2[k1]=array[ii];
		k++;k1++;
		temp1[k]=array[ii];
		temp2[k1]=array[jj];
		k1++;k++;ii++;jj--;
	}
	if(k<=n-1)
		temp1[k]=array[ii];
	if(k1<=n-1)
		temp2[k1]=array[ii];
	long sum1=0;
	long sum2=0;
	 /*for(int i=0;i<n;i++)cout<<temp1[i]<<" ";
	 cout<<endl;
	  for(int i=0;i<n;i++)cout<<temp2[i]<<" ";
	 cout<<endl;*/
	for(int i=0;i<n-1;i++)
		sum1=sum1+abs(temp1[i]-temp1[i+1]);	
	for(int i=0;i<n-1;i++)
		sum2=sum2+abs(temp2[i]-temp2[i+1]);
	// cout<<sum1<<" "<<sum2<<endl;
	cout<<max(sum2, sum1)<<endl;
    return 0;
}
