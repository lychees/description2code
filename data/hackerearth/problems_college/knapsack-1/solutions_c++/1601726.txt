#include <bits/stdc++.h>
using namespace std;
long long s=0;
int main()
{
    //cout << "Hello World!" << endl;
    long long t;
    cin>>t;
    while(t--)
    {
    	long long n,k,i,j;
    	cin>>n>>k;
    	long long a[n+1],V[1000001];
    	for(i=1;i<=n;i++)
    	cin>>a[i];
    for( i=1;i<=k;i++) V[i] = 0 ;
        V[0] = 1 ;
        int ans = 0 ;
        for(int i=1;i<=k;i++){
            for(int j=1;j<=n;j++){
                if(i-a[j] >= 0 && V[i-a[j]]){
                    V[i] = 1;
                    ans = i ;
                    break ;
                }
            }
        }
        cout<<ans<<endl;
    }
    return 0;
}
