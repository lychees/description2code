VI v;
int m[1000];
int main(){
	int t;
	si(t);
	while(t--){
		SET(m,0);
		v.clear();
		int a,b,c;
		int flag=0;
		si(a);si(b);
		int sum=0;
		v.PB(a);
		if(a<1000)
			m[a]=1;
		for(int i=1;i<=b;i++){
			sum=0;
			while(a!=0){
				int tp=a%10;
				sum+=tp*tp;
				a/=10;
			}
			if(m[sum]==0){
				v.PB(sum);
				m[sum]=1;
			}
			else{
				flag=1;
				c=sum;
				break;
			}
			a=sum;
		}
		if(flag==0){
			cout << v[SZ(v)-1] << endl;
			continue;
		}
		b-=(int)SZ(v);
		int id;
		for(int i=0;i<SZ(v);i++){
			if(v[i]==c){
				id=i;
				break;
			}
		}
		int len=SZ(v)-id;
		cout<<v[id+b%len]<<endl;
	}
	return 0;
}
