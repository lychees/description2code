#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    long long int a,b,sqa,sqb;
    cin>>t;
    while(t--)
    {
        cin>>a>>b;
        sqa=sqrt(a);
        sqb=sqrt(b);
        if(sqa*sqa>a) sqa--;
        if(sqb*sqb>b) sqb--;
        cout<<sqb-sqa+((sqa*sqa)==a)<<endl;
    }
    return 0;
}
