#include<bits/stdc++.h>
using namespace std;
int arr[1000002];
int dp[1000002];
int main()
{
    int t,n,i;
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(i=1;i<=n;i++)
            cin>>arr[i];
            dp[n]=arr[n];
        for(i=n-1;i>=1;i--)
        {
            dp[i]=max(dp[i+1],arr[i]);
        }
        int cnt=0;
        for(i=1;i<=n-1;i++)
        {
            if(arr[i]==dp[i]&&dp[i]!=dp[i+1])
                cnt++;
        }
        cout<<cnt+1<<endl;

    }

}
