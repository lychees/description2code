#include <bits/stdc++.h>
using namespace std;

int main() {
    int t, pos, x;
    bool f;
    scanf("%d", &t);
    
    while (t--) {
        scanf("%d", &x);
        f = false;
        for (int i = x; i >= 0; i--) {
            if (i%3==0 and (x-i)%5==0) {
                f = true;
                pos = i;
                break;
            }
        }
        if (!f) {
            puts("-1");
            continue;
        }
        
        for (int i = 0; i < pos; i++) printf("5");
        for (int i = 0; i < x-pos; i++) printf("3");
        puts("");
    }
    return 0;
}
