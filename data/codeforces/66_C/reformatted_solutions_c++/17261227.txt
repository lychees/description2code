using namespace std;
map<string, int> m1, m2;
string s;
int ans1 = 0, ans2 = 0;
int main() {
  while (cin >> s) {
    int ff = 0;
    while (true) {
      int pos = s.find_last_of('\\');
      if (pos == 2)
        break;
      s = s.substr(0, pos);
      bool f = (m2[s] == 0);
      m1[s] += ff;
      m2[s] += 1;
      ans1 = max(ans1, m1[s]);
      ans2 = max(ans2, m2[s]);
      if (f)
        ff++;
    }
  }
  cout << ans1 << " " << ans2 << endl;
  return 0;
}
