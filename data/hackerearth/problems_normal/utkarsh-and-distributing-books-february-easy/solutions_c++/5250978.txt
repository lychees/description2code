#include<iostream>
using namespace std;
int main()
{
	int i,t;
	cin>>t;
	for(i=0;i<t;i++)
	{
		int j,n,min=-1,sum=0;
		cin>>n;
		for(j=0;j<n;j++)
		{
			int temp;
			cin>>temp;
			sum+=temp-1;
			if(min==-1)
			{
				min=temp;
			}
			else
			{
				if(temp<min)
				{
					min=temp;
				}
			}
		}
		cout<<min-1<<" "<<sum<<endl;
	}
	return 0;
}
