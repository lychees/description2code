ll dp[100][100]={0};
ll func(ll d, ll h)
{
  if(d<0||h<0)
  return 0;
  if(d<=0&&h==1) return 1;
  if(d==0&&h==0) return 0;
  if(dp[d][h]!=0)
  return dp[d][h];
  return dp[d][h]=func(d-1,h+1)+func(d,h-1);
}
