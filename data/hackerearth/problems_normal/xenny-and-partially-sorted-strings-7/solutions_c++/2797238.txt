//conversion functions
string atob(const int a) { 
  stringstream os; 
  string b; 
  os<<a; 
  os>>b; 
  return b; 
} 

long long atob(const string a) { 
  stringstream os; 
  long long b; 
  os<<a; 
  os>>b; 
  return b; 
} 
string getSubstringofLengthM(string s,int m,int i){
	string tt = s.substr(0,m);
	string num ="";
	if(i<10){
		num = "00"+atob(i);
	}else if(i<100){
		num = "0"+atob(i);
	}else{
		num = atob(i);
	}
	return tt+num;
}
