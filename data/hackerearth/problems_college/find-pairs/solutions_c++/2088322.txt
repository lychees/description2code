#include <bits/stdc++.h>
using namespace std;

const int N = 10000010;
int cnt[2*N];
void solve() {
	int n;
	cin >> n;
	fill(cnt, cnt + 2*N, 0);
	for (int i = 0; i < n; i++) {
		int t;
		cin >> t;
		cnt[t + N]++;
	}
	
	long long ans = 0;
	for (int i = 0; i < 2 * N; i++) {
		long long val = cnt[i];
		ans += (val + 1) * val / 2;
	}
	
	cout << ans << "\n";
}

int main()
{
	ios_base::sync_with_stdio(false);
    int t;
    cin >> t;
    for (int i = 0; i < t; i++)
    	solve();
    return 0;
}
