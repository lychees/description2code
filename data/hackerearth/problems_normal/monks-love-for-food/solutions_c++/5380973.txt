#include <iostream>
#include<stack>
using namespace std;
int main(){
	stack<long long> s;
	long long q,x,c;
	cin>>q;
	while(q--){
		cin>>x;
		if(x==1){
			if(s.empty()){
				cout<<"No Food"<<endl;
			}
			else{
				cout<<s.top()<<endl;
				s.pop();
			}
		}
		else{
			cin>>c;
			s.push(c);
		}
	}
	return 0;
}
