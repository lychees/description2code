#include <iostream>
#include <vector>
#define c cin
#define o cout
using namespace std;

string reverse(string s){
	string buf = "";
	
	for(int i = s.length()-1; i >= 0; i--){
		buf += s[i];
	}
	
	return buf;
}
int main(){
	int t;
	vector <string> v;
	c>>t;
	while(t--){
		string buffer;
		c>>buffer;
		v.push_back(buffer);
	}
	
	for(int i = 0; i < v.size()-1; i++){
		for(int j = i+1; j < v.size(); j++){
			if(v[i] == reverse(v[j])){
				string buffer;
				buffer = v[i];
				o << buffer.length() << " " << buffer[buffer.length()/2];
				break;
			}
		}
	}
	return 0;
}
