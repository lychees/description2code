#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    string s[t];
    int min[t][2];
    int flag=0;
    for(int i=0;i<t;i++)
    {
    	cin>>s[i];
    	min[i][0]=60*((s[i][0]-'0')*10 + (s[i][1]-'0'))+((s[i][3]-'0')*10 + (s[i][4]-'0'));
    	min[i][1]=60*((s[i][6]-'0')*10 + (s[i][7]-'0'))+((s[i][9]-'0')*10 + (s[i][10]-'0'));
    	for(int j=0;j<i;j++)
    	{
    		if((min[i][0]<=min[j][0] && min[i][1]<=min[j][0] ) || (min[i][0]>=min[j][1] && min[i][1]>=min[j][1] ))
    			continue;
    		else
    		 {  flag=1;
    			break;
    	     }
    	}
    	if(flag==1) break;
    }
   if(flag==1)
   	cout<<"Will need a moderator!";
   	else
   	cout<<"Who needs a moderator?";
}
