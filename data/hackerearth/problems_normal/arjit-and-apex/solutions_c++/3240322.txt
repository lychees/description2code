#include <bits/stdc++.h>

#define si(x) scanf("%d",&x)
#define sl(x) scanf("%lld",&x)
#define ss(x) scanf("%s",(x))
#define pi(x) printf("%d",x)
#define pl(x) printf("%lld",x)
#define ps(x) printf("%s",(x))
#define pn printf("\n")
#define rep(i,n) for(i=0;i<n;i++)
#define PII pair<int,int>
#define pb push_back
#define mk make_pair
#define loop(i,a,n) for(i=a;i<n;i++)
#define si2(a,b) scanf("%d %d",&a,&b)
#define si3(a,b,c) scanf("%d %d %d",&a,&b,&c)
#define ll long long int

using namespace std;

map<int,multiset<int> > student,master;

int main(){
	int T,M,N,G,H,temp1,temp2,good,great,i;si(T);
	map<int,multiset<int> >::iterator itr1;
	multiset<int>::iterator itr2,itr3,itr4;
	while(T--){
		si(M);si(N);
		rep(i,M){
			si2(temp1,temp2);
			student[temp1].insert(temp2);
		}
		rep(i,N){
			si2(temp1,temp2);
			master[temp1].insert(temp2);
		}
		si2(G,H);
		itr1=student.begin();
		great=good=0;
		while(itr1!=student.end()){
			if(master[itr1->first].size()==0){itr1++;continue;}
			itr4 = master[itr1->first].begin();
			itr3 = itr1->second.begin();
			while(itr3 != itr1->second.end()){
				while(itr4 != master[itr1->first].end() && *itr3>*itr4){itr4++;}
				if(itr4 != master[itr1->first].end()){
					good++;
					if(*itr4==*itr3){great++;}
				}
				if(good>=G && great >= H){break;}
				itr3++;
			}
			if(good>=G && great >= H){break;}
			itr1++;
		}
		if(good>=G && great >= H){ps("Great\n");}
		else if(good>=G){ps("Good\n");}
		else{ps(":(\n");}
		master.clear();student.clear();
	}
}
