#include <iostream>
#include <math.h>
using namespace std;

int main()
{
    long long int n;
    int t;
    cin>>t;
    while(t--)
    {
    	long long int i,p,m,count=0,flag=0,check=0,b=0;
    	cin>>n;
    	for(i=2;i<=sqrt(n);i++)
    	{
    		if(n%i==0)
    		{
    		p=n/i;
    		if(p!=i)
    		{
			
    			if(p%2==0)
    			count++;
    			if(i%2==0)
    			flag++;
    			
				}
				if(p==i && p%2==0)
				b++;
    		}
    	}
    	if(n%2==0)
    	check++;
    	m=count+flag+check+b;
    	cout<<m<<endl;
    }
    return 0;
}
