#include <iostream>
#include<vector>
using namespace std;

vector<long long int> v;
long long int n,t,maxx,len,a[50];
void recurse(long long int pos)
{  long long int xx,yy,zz;
	if(pos>=n)
{  // xx=v[0]
    if(v.size()>0)
    {  xx=v[0];
       yy=v[0];
	for(int i=1;i<v.size();i++)
	{
		xx=xx ^ v[i];
		yy=yy & v[i];
	}
//	cout<<v.size()<<endl;
  /*for(int i=0;i<v.size();i++)
    {
    	cout<<v[i]<<" ";
    }
    cout<<endl;
    */
   
    
   zz=xx|yy;
 
   if(zz>maxx)
   {  //cout<<"vivek\n";
      //cout<<zz<<endl;
   	  maxx=zz;
   	  len=v.size();
   }
   else if(zz==maxx)
   {
   	  if(v.size()>len)
   	  {
   	  	len=v.size();
   	  }
   }
}
return ;
}
	
	v.push_back(a[pos]);
//	cout<<a[pos]<<endl;
	recurse(pos+1);
	v.pop_back();
	recurse(pos+1);
}


int main()
{ 
    cin>>t;
    while(t--)
    {   maxx=0;
        len=0;
        v.clear();
    	 cin>>n;
    	 for(int i=0;i<n;i++)
    	 {
    	 	cin>>a[i];
    	 }
    	 
    	 recurse(0);
    	 cout<<maxx<<" "<<len<<endl;
    }
    return 0;
}
