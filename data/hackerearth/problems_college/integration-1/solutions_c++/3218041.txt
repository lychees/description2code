#include <stdio.h>
int main()
{
    //printf("Hello World!\n");
    int t,i=0,j,k,n,m,p,u,r,s,cur=0,max=0;
    scanf("%d",&t);
    while(t--)
    {
    	i++;
    	scanf(" %d %d %d %d %d",&n,&p,&u,&r,&s);
    	cur=p;
    	while(n--)
    	{
    		if(u-cur>=r)
    		{
    			cur=cur+r;
    		}
    		else
    		{
    			if(cur<s)
    			{
    				break;
    			}
    			cur=cur-s;
    		}
    	}
    	printf("Case #%d: %d\n",i,cur);
    }
    
    return 0;
}
