#include <iostream>
using namespace std;

int main()
{
	long long i,n0[500],n1[500],n2[500],n;
    cin>>n;
    n0[0]=1;
    n1[0]=0;
    n2[0]=0;
    
    for(i=1;i<=n;i++)
    {
    	n0[i] = n1[i-1] + n2[i-1];
    	n1[i] = n0[i-1];
    	n2[i] = n1[i-1];
    }
    
    cout<<n0[n]+n1[n]+n2[n];
    return 0;
}
