#include <bits/stdc++.h>
typedef long long ll;
using namespace std;



int main()
{
	int t;
	ll n;
	char a[1000005];
	cin>>t;
	while(t--)
	{
	//cout<<t;
	ll c=1,res=0;
	cin>>n>>a;
	for(ll i=1;i<=n;i++)
	{
		if(a[i]>=a[i-1])
		c++;
		else
		{
			res+=((c+1)*c)/2;
			c=1;
		}
		
	}
	cout<<res<<endl;
	}
    return 0;
}
