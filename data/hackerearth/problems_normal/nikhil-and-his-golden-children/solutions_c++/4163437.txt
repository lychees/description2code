#include <iostream>
using namespace std;

using ull = unsigned long long;
constexpr ull mod = 1000000007;

ull powm(ull a, ull n) {
	ull res = 1;
	while (n) {
		if (n % 2 == 1)
			res = (res * a) % mod;
		a = (a * a) % mod;
		n /= 2;
	}
	return res;
}

ull inversem(ull n) {
	return powm(n, mod - 2);
}

ull test() {
	int k;
	int m;
	cin >> k >> m;
	for (int i = 0; i < k; ++i) {
		int x;
		cin >> x;
		m -= (x - 1);
	}
	if (m < k)
		return 0ull;
	k -= 1;
	m -= 1;
	if (2 * k > m) 
		k = m - k;
	ull num = 1;
	ull den = 1;
	for (int i = 1; i <= k; ++i, --m) {
		num = (num * (ull)m) % mod;
		den = (den * (ull)i) % mod;
	}
	return (inversem(den) * num) % mod;
}

int main() {
	int t;
	cin >> t;
	for (int i = 0; i < t; ++i) 
		cout << test() << '\n';
    return 0;
}
