#include <iostream>
#include <stdio.h>
using namespace std;

long long min(long long a, long long b)
{
	if(a > b)
		return b;
	return a;
}

int main()
{
    int T;
    scanf("%d", &T);
    while(T--)
    {
    	long long R, C;
    	scanf("%lld %lld", &R, &C);
    	int i;
    	long long p = min(R, C);
    	long long trace1, trace2;
    	trace1 = (p * (2 + (p - 1) * (C + 1))) / 2;
    	trace2 = (p * (2 + (p - 1) * (R + 1))) / 2;
    	printf("%lld\n", trace1 + trace2);
    }
    return 0;
}
