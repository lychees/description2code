{

    int dp1[n],dp2[n];

    dp1[0]=1;
    dp1[1] = 2;
    dp1[2] = 4;
    dp2[0] = 1;
    dp2[1] = 2;
    for(i=3;i<n;i++){
    	dp1[i] = dp1[i-1] + dp1[i-2] + dp1[i-3];
    }
    for(i=2;i<n;i++){
    	dp2[i] = dp2[i-1] + dp2[i-2];
    }
    if(n<0) cout << "ERROR" << endl;
    else{
