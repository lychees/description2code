t = int(raw_input())
A = []
for i in range(t):
    N, M = raw_input().split(' ')
    M = int(M)
    A.append([N, M])

A = sorted(A, key=lambda x: x[1], reverse=True)
# print A
l = len(A)
# print l
i = 0
while i < l:
    j = i+1
    temp = [A[i]]
    while j < l and A[i][1] == A[j][1]:
        temp.append(A[j])
        j = j+1
    temp = sorted(temp, key=lambda x: x[0])
    # pr/int temp
    if i == j:
        print A[i][0], A[i][1]
    for k in temp:
        print k[0], k[1]
    i = j
