#include<iostream>
using namespace std;
int hcf_calculator(long [],int);
int main()
{ int t,n;
  long  a[100];
  cin>>t;
  while(t--)
  { cin>>n;
   for(int i=0;i<n;i++)
   { cin>>a[i];}
   int p=hcf_calculator(a,n);
   if(p==1)
   cout<<"YES"<<endl;
   else
   cout<<"NO"<<endl;
}
return 0;
}
    
  int hcf_calculator(long  weight[], int asize)
{
	int i; 
    long a, t;

	a = weight[0];

	for(i=0;i<(asize-1);i++) {
		/* function gcd(a, b) */
		t = 0;
		while(weight[i+1] != 0) {
			t = weight[i+1];
			weight[i+1] = a % weight[i+1];
			a = t;
		}
	}
	
    return a;
}
