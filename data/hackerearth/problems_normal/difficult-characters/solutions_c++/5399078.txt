#include<bits/stdc++.h>
using namespace std;

int main(){
	int n;
	string s;
	cin >> n;
	for(int i=0;i<n;i++){
		vector<pair<int, char> > vp;
		for(int i=0;i<26;i++){
			vp.push_back(make_pair(0,'a'+i));
		}
		cin >> s;
		for(int i=0;i<s.size();i++){
			vp[s[i]-'a'].first++;
		}
		
		sort(vp.begin(),vp.end());
		
		
		for(int i=0,j;i<26;i++){
			j=i+1;
			int x=i;
			for(;j<26;j++,i++){
				if(vp[i].first!=vp[j].first) break;
			}
			reverse(vp.begin()+x,vp.begin()+j);
		}
		
		
		for(int i=0;i<26;i++){
			if(i>0) cout << " ";
			cout << vp[i].second;
		}
		cout << endl;
	}
	return 0;
}
