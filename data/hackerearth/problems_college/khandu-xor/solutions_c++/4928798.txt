#include <iostream>
#include<string.h>
using namespace std;
int chk(string str)
{
	int i,l;
	l=str.size();
	for(i=0;i<l/2; i++){
		if(str[i]!= str[l-i-1]){
			return 0;
		}
	}
	return 1;
}
int main()
{
	int t;
    cin >> t;
    string s1,s2,s3,s4;
    while(t--){
    int t;
    	cin >>s1 ;
		cin >>s2;
    	s3 = s1+ s2;
    	s4 = s2 + s1;
    	if(chk(s3) || chk(s4))
    	cout << "YES\n";
    	else

    }

}
