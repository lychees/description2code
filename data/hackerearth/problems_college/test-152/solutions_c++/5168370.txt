#include <bits/stdc++.h>
using namespace std;
int f[26];
int hashFun(char c){
	return (c-'a');
}
int main()
{
    int t;
    long long int n,i;
    string s;
    cin>>t;
    while(t--){
    	int max=0;
    	scanf("%lld",&n);
    	cin>>s;
    	for(i=0;i<26;i++){
    		f[i]=0;
    	}
    	for(i=0;i<s.length();i++){
    		int index =hashFun(s[i]);
    		
    		f[index]++;
    	
    		if(f[index]>max){
    			max=f[index];
    		}	
    	}
    	//cout<<max;
    	if(max%2==0)
    	cout<<"Yes"<<endl;
    	else
    	cout<<"No"<<endl;
    }
    
    return 0;
}
