#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int i,n,q,j;
    cin>>n>>q;
    long long A[n],x;
    for(i=0;i<n;i++)
    	cin>>A[i];
    sort(A,A+n);
    for(i=0;i<q;i++)
    {
    	cin>>x;
    	if( binary_search(A,A+n,x) )
    		cout<<"YES\n";
    	else
    		cout<<"NO\n";
    }
    return 0;
}
