#include<iostream>
#include<cstdio>
#include<algorithm>
#define MAX 31650
#define IND 3404
using namespace std;

bool arr[MAX];
long long prime[IND];

inline void sieve()
{
int i,j;
for(i=4;i<MAX;i+=2)
arr[i]=true ;

for(i=3;i<MAX;i+=2)
for(j=i;j*i<MAX;j+=2)
arr[i*j]=true;

int k=0;
prime[k++]=2;
for(i=3;i<MAX;i+=2)
if(!arr[i])
prime[k++]=i;
}



int main()
{
    sieve();   
    int t,ct,ct1,i;
    long long a,b,c,d;
    bool flag;
    cin>>t;
    
    while(t--)
    {
              cin>>a>>b>>c>>d;
              if(d==0)
              {
                      cout<<"Divisible\n";
                      continue;
                      }
                      
              flag=false;
              for(i=0;i<3404;++i)
              {
                                                                 ct=0;
                                                                 while(a%prime[i]==0)
                                                                 {
                                                                                     ++ct;
                                                                                     a/=prime[i];
                                                                             }
                                                    ct1=0;     
                                               while(c%prime[i]==0)
                                               {
                                                                ++ct1;
                                                                c/=prime[i];
                                                                }
                                               if((ct==0 && ct1!=0) || ((ct*b)<(ct1*d)))
                                               {
                                                             flag=true;
                                                             break;
                                                             }
                                                             }
                                               if((c>1 && a>1 && c!=a) || (a%c))
                                               flag=true;
                                               
                                               if(flag)
                                               cout<<"Not divisible\n";
                                               else
                                               cout<<"Divisible\n";
                                               }
                                               return 0;
                                               }
