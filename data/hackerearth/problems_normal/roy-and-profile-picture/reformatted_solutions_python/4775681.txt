l = input()
n = input()
i = 0
while i < n:
    a = map(int, raw_input().split())
    w = a[0]
    h = a[1]
    if (w >= h and w >= l) and w == h:
        print "ACCEPTED"
    elif (w > l and h == l) or (w == l and h > l) or (w > l and h > l and w != h):
        print "CROP IT"
    elif w < l or h < l:
        print "UPLOAD ANOTHER"
    i += 1
