#include<iostream>

using namespace std;

int main()
{
    int t;
    cin>>t;

    while(t!=0)
    {
        int n;
        cin>>n;

         long long sum=0;

         long long arr[n];

        for(int i=0;i<n;i++)
        {
            cin>>arr[i];
            sum+=arr[i];
        }

        sum=sum/(n-1);

        for(int i=0;i<n;i++)
        {
            cout << sum-arr[i] << " ";
        }

        cout << endl;

        t--;
    }
}
