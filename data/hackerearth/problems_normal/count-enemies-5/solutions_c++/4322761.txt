#include <iostream>

using namespace std;

int main(){

int t;
cin>>t;
while(t--){
string a;
cin>>a;
int n= a.size(),count = 0,i;
for(i=0;i<n;i++){
    if(a[i]=='O') count = count + 1;
}
int j;
for(i=0;i<n;i++){

if(a[i]=='X'){

for(j=i-1;j>=0 ;j--){
if(a[j]=='O'){
    a[j]='-';
    count = count -1;
}
else if(a[j]!='-' || a[j]!='*' || a[j]=='X') break;
}

for(j=i+1;j<n ;j++){
if(a[j]=='O'){
    a[j]='-';
    count = count -1;
}
else if(a[j]=='-' || a[j]=='*' || a[j]=='X') break;

}
}

}

cout<<count<<endl;
}
return 0;
}
