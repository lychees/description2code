#include <bits/stdc++.h>
using namespace std;
vector <pair<char,long long int> > v;
long long int m;

long long int cal(long long int n,long long int idx){
	if(v[idx].first=='+'){
		n+=v[idx].second;
	}
	if(v[idx].first=='-'){
		n-=v[idx].second;
	}
	if(v[idx].first=='*'){
		n*=v[idx].second;
	}
	if(v[idx].first=='/'){
		n/=v[idx].second;
	}
	if(v[idx].first=='N'){
		n=-n;
	}
	return n;
}

int main()
{
	long long int a,b,t,i,c,d,w;
	char q;
	cin >> t;
	while(t--){
		cin >> m;
		for(i=0;i<m;i++){
			cin >> q;
			if(q!='N'){
				cin >> w;
			}
			v.push_back(make_pair(q,w));
		}
		c=1;
		d=cal(1,0);
		a=min(c,d);
		b=max(c,d);
		for(i=1;i<m;i++){
			c=cal(a,i);
			d=cal(b,i);
			//cout << c << " " << d << endl;
			a=min(min(a,c),d);
			b=max(max(b,c),d);
		}
		cout << b << endl;
		v.clear();
	}
    return 0;
}
