#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int k;
    	cin>>k;
    	char s[1000004];
    	cin>>s;
    	int l = strlen(s);
    	int c=0;
    	for(int i=0;i<l;i++){
    		int sum = 0;
    		for(int j=i;j<l;j++){
    			sum = (s[j] - 'a') + 1 + sum;
    			if(sum == k)
    				c++;
    			else if(sum > k)
    				break;
    		}
    	}
    	cout<<c<<endl;
    }
    return 0;
}
