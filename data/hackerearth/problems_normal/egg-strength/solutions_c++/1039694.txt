#include <iostream>
#include <algorithm>
using namespace std;

int output[501][501]={{0}};

void calculate()
{
	int result;
	for(int i=1;i<501;i++)
	{
		output[i][0]=0;
		output[i][1]=1;
	}
	for(int i=1;i<501;i++)
		output[1][i]=i;
	for(int i=2;i<501;i++)				//eggs
	{
		for(int j=2;j<501;j++)			//floors
		{
			output[i][j]=2147483647;
			for(int x=1;x<=j;x++)
			{
				result=1+max(output[i-1][x-1],output[i][j-x]);
				if(result<output[i][j])
					output[i][j]=result;
			}
		}
	}
}

int main()
{
	calculate();
    int t;
    cin>>t;
    while(t--)
    {
    	int flors, eggs;
    	cin>>flors>>eggs;
    	cout<<output[eggs][flors]<<endl;
    }
    return 0;
}
