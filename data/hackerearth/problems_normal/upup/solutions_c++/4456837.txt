#include <iostream>
#include <string>
using namespace std;

int main()
{
	char str[30003];
	gets (str);
	int i = 0;
	
	if (str[0] >= 97 && str[0] <= 122)
	{
		str[0] = str[0] - 32;
	}
	
	while(str[i])
	{
		if (str[i] == ' ' && str[i+1] != ' ' && (str[i+1] >= 97 && str[i+1] <= 122))
		{
			str[i+1] = str[i+1] - 32;
			i++;
		}
		i++;
	}
    cout <<str<< endl;
    return 0;
}
