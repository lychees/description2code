#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

bool myfunction (int i,int j) { return (i>j); }
int main() {
	// your code goes here
	int n,a,n1,l=0,k=0,ans=0,ans1=0;
	vector<int> v,v1,v2,v3;

	cin>>n;
	n1=n;
	while(n1--)
	{
		cin>>a;
		v.push_back(a);
		v1.push_back(a);
	}
	sort (v.begin(), v.end());
	sort (v1.begin(), v1.end(),myfunction);
	for(int i=0;i<n;i++)
	{
		if(i<n)
		v2.push_back(v[l++]);

		i++;
		if(i<n)
		v2.push_back(v1[k++]);
	}
	
	l=0;k=0;
	for(int i=0;i<n;i++)
	{
		if(i<n)
		v3.push_back(v1[l++]);

		i++;
		if(i<n)
		v3.push_back(v[k++]);
	}
	for(int i=1;i<n;i++)
        {
            if(v2[i] > v2[i-1])
               ans=ans + (v2[i] - v2[i-1]);
            if(v2[i] < v2[i-1])
                ans=ans + (v2[i-1] - v2[i]);

        }
    	for(int i=1;i<n;i++)
        {
            if(v3[i] > v3[i-1])
               ans1=ans1 + (v3[i] - v3[i-1]);
            if(v3[i] < v3[i-1])
                ans1=ans1 + (v3[i-1] - v3[i]);

        }
        if(ans>ans1)
    cout<<ans;
    else
    cout<<ans1;
	return 0;
}
