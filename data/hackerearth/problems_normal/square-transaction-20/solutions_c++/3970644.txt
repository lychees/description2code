#include<bits/stdc++.h>
using namespace std;
int main()
{
    {
        int n;
        cin>>n;
        int a[n];
        int sum[n];
        cin>>a[0];
        sum[0]=a[0];
        for(int i=1;i<n;i++)
        {
            cin>>a[i];
            sum[i]=a[i]+sum[i-1];
        }
        int q;
        cin>>q;
        while(q--)
        {
            int x;
            cin>>x;
            if(sum[n-1]<x)
                cout<<-1<<endl;
            else
            {
                int rk=lower_bound(sum,sum+n,x)-sum;
                cout<<rk+1<<endl;
            }
        }
    }
}
