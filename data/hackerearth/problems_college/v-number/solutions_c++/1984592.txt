#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t,i;
	char s[100];
    cin>>t;
    while(t--)
    {
    	cin>>s;
    	int l=strlen(s);
    	if(l%2==0) {
    		int sum=0;
    		for(i=0;i<l/2;i++) sum+=s[i]-48;
    		for(i=l/2;i<l;i++) sum-=s[i]-48;
    		(sum==0)?cout<<"YES\n":cout<<"NO\n";
    	}
    	else cout<<"NO\n";
    	
    }
    return 0;
}
