#include<iostream>
#include<cstdio>
#include<cmath>
using namespace std;

double dp[3500];
double x[3500],y[3500],arr[3500];

double dist(int i,int j)
{
    double ans=(x[i]-x[j])*(x[i]-x[j])+(y[i]-y[j])*(y[i]-y[j])*1.0;
    ans=sqrt(ans)*1.000000;
    return ans;
}

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);

    int n;
    cin>>n;

    for(int i=0;i<n;i++){
        cin>>x[i]>>y[i];
        cin>>arr[i];
    }

    dp[0]=arr[0]*1.0;

    for(int i=1;i<n;i++)
    {
        double maxi=arr[0]+arr[i]-dist(0,i)*1.000000;
        for(int j=1;j<i;j++)
            maxi=max(maxi,(dp[j]+arr[i]-dist(i,j)*1.000000));


        dp[i]=maxi;
    }
    printf("%.6f",dp[n-1]);
}
