#include <iostream>
#include<algorithm>
using namespace std;

struct stu
{
	string s;
	int n;
};
int main()
{
	stu student[100];
	string stemp;
   int no,flag,found,nooccur=0;
   cin>>no;
   cin.ignore(32767,'\n');
   int i,j;
   for(i=0;i<no;i++)
   {
    cin>>stemp;//cin.ignore(32767,' ');//cout<<stemp;
   	flag=nooccur;
   	found=0;
   	for(j=0;j<nooccur;j++)
   	{
   		if(stemp==student[j].s)
   		{
   			student[j].n++;
   			found=1;
   		}
   	}
   	if(found==0)
   	{
   		student[nooccur].s=stemp;
   		student[nooccur].n=0;
   		student[nooccur].n++;
   		nooccur++;
   	}
   	
   }
  /* for(i=0;i<nooccur;i++)
   {
   	cout<<student[i].s<<" "<<student[i].n<<"\n";
   }
   check statement */
   int arrange[100];
   for(i=0;i<nooccur;i++)
   {
   	arrange[i]=student[i].n;
   	
   }
   sort(arrange,arrange+nooccur);
   int arrangefinal[100];
     arrangefinal[0]=arrange[0];
	for (int i = 1; i < nooccur; i++)
	{
		bool matching = false;
		for (int j = 0; (j < i) && (matching == false); j++)
		if (arrange[i] == arrange[j])
		matching = true;
		if (!matching) arrangefinal[i]=arrange[i];
	}
   for(i=nooccur;i>=0;i--)
   {
   	for(j=0;j<nooccur;j++)
   	{
   		if(student[j].n==arrangefinal[i])
   		{
   			cout<<student[j].s<<" "<<student[j].n<<"\n";
   		}
   	}
   }
    return 0;
}
