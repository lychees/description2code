#include<iostream>
#include<string>
using namespace std;

int main()
{
    int k;
    int cont=0;
    cin>>k;
    string s,s1,s2;
    cin>>s;
    string temp = s;
    while(1)
    {
        s1="";
        s2="";
        for(int i = 0;i<s.size();i++)
        {
            if(i%2 == 0)
            {
                s1 = s1 + s[i];
            }
            else
            {
                s2 = s2 + s[i];
            }
        }
        s = s1 + string(s2.rbegin(),s2.rend());
        cont++;
        if(s == temp)
        {
            break;
        }
       // cout<<s<<endl;
        //k--;
    }
    cont = k%cont;
    while(cont)
    {
        s1="";
        s2="";
        for(int i = 0;i<s.size();i++)
        {
            if(i%2 == 0)
            {
                s1 = s1 + s[i];
            }
            else
            {
                s2 = s2 + s[i];
            }
        }
        s = s1 + string(s2.rbegin(),s2.rend());

        cont--;
    }
    //cout<<endl;

    cout<<s;

}
