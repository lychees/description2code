#include <bits/stdc++.h>
using namespace std;
int a[100005];
int sum[100005];
int main()
{
   int n,q,i,l,r;
   cin>>n>>q;
   int sum1=0;
 
   for(i=0;i<n;i++){
   		cin>>a[i];
   		
   	}
   	sum[0]=a[0];
   	for(i=1;i<n;i++){
   		sum[i]=sum[i-1]+a[i];
   	}
   	while(q--){
   		cin>>l>>r;
   		if(l==r){
   			cout<<a[l-1]<<endl;
   		}
   		else if(l==1)
   		{
   			cout<<sum[r-1]<<endl;
   		}
   		else{
   		 sum1 = sum[r-1]-sum[l-2];
   		 
   		cout<<sum1<<endl;
   		}
   		
   	}
   	
   	
   
}
