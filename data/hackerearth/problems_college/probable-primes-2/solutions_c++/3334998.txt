#include<bits/stdc++.h>
using namespace std;
int a[1000001];
int c[1000001];
int main()
{
	int i,j;
	for(i=2;i<=1000000;i++)
	{
		a[i]=1;
	}
	for(i=2;i*i<=1000000;i++)
	{
		if(a[i]==1)
		{
			for(j=2*i;j<=1000000;j+=i)
			{
				a[j]=0;
			}
		}
	}
	/*for(i=2;i<=10;i++)
	{
		cout<<a[i]<<" ";
	}*/
	for(i=2;i<=1000000;i++)
	{
		if(a[i]==1)
		c[i]=c[i-1]+1;
		
		else
		c[i]=c[i-1];
	}
/*	for(i=2;i<=10;i++)
	cout<<c[i]<<" ";*/
	int t,l,r;
	float total,ans;
	cin>>t;
	while(t--)
	{
		cin>>l>>r;
		total=r-l+1;
		//cout<<total<<"\n";
		ans=c[r]-c[l-1];
		//cout<<ans<<"\n";
		cout<<fixed<<setprecision(6)<<ans/total<<"\n";
	}
	return 0;
}
