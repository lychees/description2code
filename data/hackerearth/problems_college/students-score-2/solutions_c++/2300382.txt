#include <iostream>
using namespace std;
#include<bits/stdc++.h>
int main()
{
	int t,n,a[100000];


		int finish;
		int start;
		int sum;
		int maxsum;
		cin>>n;
		int i=0,n1=n;
		while(n1--)
		{
			cin>>a[i++];
		}
		sum=maxsum=a[0];
		for(int i=1;i<n;i++)
		{
			if(sum+a[i]<a[i])
			{
				sum=a[i];
				start=i;
			}
			else
			sum=sum+a[i];
			if(sum>maxsum)
			{
				maxsum=sum;
				//printf("the maximum sum is %d\n",maxsum);
				finish=i;
			}
			
			
		}
		if(maxsum<sum)
		maxsum=sum;
		if(maxsum>0)
		cout<<maxsum<<endl;
		else cout<<"0"<<endl;
	
	
    //cout << "Hello World!" << endl;
    return 0;
}
