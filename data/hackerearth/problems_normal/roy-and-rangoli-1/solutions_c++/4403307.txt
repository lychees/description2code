#include<iostream>
using namespace std;
long long int arr[2000005];
#define m 2000005
void sieve()
{
    arr[0]=arr[1]=1;
    for(long long int i=2*2;i<m;i+=2)
    {
        arr[i]=1;
    }
    //for(long long int p=0;p<100;p++)
      //  cout<<arr[p]<<' ';
    //cout<<endl;
    for(long long int i=3;i*i<=m;i++)
    {
        if(arr[i]==0)
        {
            for(long long int j=i*2;j<m;j+=i)
                arr[j]=1;
        }
    }
}
int main()
{
    sieve();
    long long int x;ios::sync_with_stdio(false);
    cin>>x;
    long long int count=0,tempcount=0,c=0;
    for(int i=0;i<x;i++)
    {
        if(arr[i]==0)
            (count++)%1000000007;
    }c=count;
    //cout<<count<<endl;
    for(int i=1;i<x;i++)
    {
        tempcount = c;//%1000000007;
        if(arr[i-1]==0)
            tempcount = (tempcount-1)%1000000007;
        if(arr[x-1+i]==0)
            tempcount=(tempcount+1)%1000000007;
        count=(count+tempcount)%1000000007;
        c=tempcount;
    }
    cout<<count<<endl;
}
