#include <iostream>
#include <cstdio>
#include<algorithm>
#include<limits.h>

using namespace std;

struct node
{
	int index;
	int val;
};


bool comp(node x,node y)
{
	int i,j;
	i=x.val;
	j=y.val;
	if(i==j)
	{
      if(x.index<y.index)
	   return 1;
	  else 
	   return 0;
	}
	else if(i<j)
	  return 1;
	else
	  return 0;
	
}

int main()
{
        int test_cases;
        scanf("%d",&test_cases);
        while(test_cases > 0)
        {
                int N,arr[10000],i=0,ans;
                scanf("%d",&N);
                while(i<N)
                {
                        scanf("%d",&arr[i]);
                        i = i+1;
                }
                node a[N];
                for(int j=0;j<N;j++)
                {
                	a[j].val=arr[j];
                	a[j].index=j;
                }
                sort(a,a+N,comp);
                int val,max=INT_MIN;
                for(int j=0;j<N;j++)
                {
                	val=abs(j-a[j].index);
                	if(val>max)
                	  max=val;
                //	cout<<a[j].val<<" ";
                }
                cout<<max<<"\n";
                test_cases--;
        }
        return 0;
}
