#include <bits/stdc++.h>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   	int t;
   	scanf("%d",&t);
   	while(t--){
   		long long a[1000000],b[1000000],c[1000000],n;
   		scanf("%d",&n);
   		int i,c1=0,c2=0;
   		for(i=0;i<n;i++)
   		{
   			scanf("%lld",&a[i]);
   		}
   		b[0]=0;
   		for(i=1;i<n;i++)
   		{
   			b[i]=b[i-1]+a[i-1];
   		}
   		c[n-1]=0;
   		for(i=n-2;i>=0;i--)
   		{
   			c[i]=c[i+1]+(a[i+1]*2);
   		}
   		for(i=0;i<n;i++){
   			if(b[i]<=c[i])
   				c1++;
   			else
   				c2++;
   		}
   		cout<<c1<<" "<<c2<<endl;
   		
   	}
    return 0;
}
