#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,d;
		cin>>n>>d;
		if(n>=d)
		{
			cout<<"Yes"<<"\n";
			continue;
		}
		if(n==0 && d!=0)
		{
			cout<<"No"<<"\n";
			continue;
			
		}
		if(n==1 && d<=2)
		{
			cout<<"Yes"<<"\n";
			continue;
			
		}
		else if(n==1 && d>2)
		{
			cout<<"No"<<"\n";
			continue;
		}
	
		int sum=(n-2)*6+4;
		if(sum>=d)
		{
			cout<<"Yes"<<"\n";
			
		}
		else
		{
			cout<<"No"<<"\n";
		}
	}
}
