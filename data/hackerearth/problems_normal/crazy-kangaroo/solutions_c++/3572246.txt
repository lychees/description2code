#include <iostream>

using namespace std;

int main()
{
    long t;
    long long A,B,M,countt=0,j=0;
    cin>>t;
    for(long i=0;i<t;i++)
    {
        cin>>A>>B>>M;
        if(M>=A)
            j=M;
        else if(A%M==0)
            j=A;
        else
        {
            j=A/M;
            j=M*j+M;
        }
        A=B%M;
        A=B-A;
        A=A-j;
        countt=A/M + 1;
        cout<<countt<<endl;
    }
    return 0;
}
