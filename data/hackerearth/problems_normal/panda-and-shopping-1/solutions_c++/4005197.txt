#include<bits/stdc++.h>

using namespace std;
#define fre 	freopen("0.in","r",stdin);freopen("0.out","w",stdout)
#define abs(x) ((x)>0?(x):-(x))
#define MOD 104729
#define mod MOD
#define LL signed long long int
#define pii pair<int,int>
#define scan(x) scanf("%d",&x)
#define print(x) printf("%d\n",x)
#define scanll(x) scanf("%lld",&x)
#define printll(x) printf("%lld\n",x)
#define rep(i,from,to) for(int i=(from);i <= (to); ++i)
#define VI vector<long long int>
#define MAX_N 100000
LL tree[MAX_N+5];
LL read(LL *bit,int idx){
    LL sum = 0;
    if(idx==0)
		return 0;
    while (idx > 0){
    	sum = max(bit[idx],sum);
        idx -= (idx & -idx);
    }
    return sum;
}
void update(LL *bit,int idx ,LL val){
    while (idx <= MAX_N){
        bit[idx] = max(bit[idx], 1LL*val);
        idx += (idx & -idx);
    }
}
int H[100000+5];
LL L[100000+5];
int main(){
	int N;
	memset(tree,-MOD*1LL*MOD,sizeof tree);
	vector<pair<LL,int> > V;
	cin>>N;
	rep(i,1,N)cin>>H[i]>>L[i];
	rep(i,1,N){
		V.push_back({L[i],i});
	}
	sort(V.begin(),V.end());
	rep(i,1,N){
		int x = V[i-1].second;
		update(tree,x,max(H[x]*1LL, H[x]+read(tree,x)));
	}
	printll(read(tree,N));
}
