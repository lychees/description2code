bool arr[1000001];

{

    for(int i=0;i<1000001;++i)
        arr[i]=false;

        {
           j=i*i;
            while(j<1000001)
            {
                arr[j]=true;

            }
        }
    }
    bool flag;

    while(t--)
    {
        flag=false;

        {
            if((arr[i])&&(arr[x-i]))

        }

}
