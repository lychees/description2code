#include<iostream>
#include<string>
#include<cstring>
using namespace std;
int arr[27];
int main()
{
    int k;string str;
    cin>>k>>str;
    int ans = 0;
    int n = str.length();
    for(int i=0;i<n;i++)
    {
        memset(arr,0,sizeof(arr));
        int count = 0;
        for(int j=i;j<n;j++)
        {
            arr[str[j]-'a']++;
            if(arr[str[j]-'a']==1)
                count++;
            if(count == k)
                ans++;
        }
    }
    cout<<ans<<endl;
}
