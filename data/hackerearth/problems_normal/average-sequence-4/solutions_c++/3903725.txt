#include<iostream>
#include<algorithm>
using namespace std;

int main()
{
    int n;
    cin >> n;
    int a[n];
    int i = 0;
    int x;
    while(i < n)
    {
        cin >> x;
        if(i == 0)
            a[i] = x;
        else
        {
            int sum = 0;
    		for(int j=0; j<i; j++)
    		{
    			sum += a[j];
    		}
           // cout << "sum = " << sum << " x = " << x << " i = " << i << endl;
            a[i] = (x * (i+1)) - sum;
           // cout << "a[i] = " << a[i] << endl;
        }
        cout << a[i] << " ";
        ++i;
    }
    return 0;
}
