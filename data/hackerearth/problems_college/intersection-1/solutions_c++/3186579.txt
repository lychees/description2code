#include <bits/stdc++.h>
#define ll long long int
#define s(a) scanf("%lld",&a)
#define sd(a) scanf("%lf",&a)
#define f first
#define sc second
#define pb push_back
#define mp make_pair
#define inf 10e16

using namespace std;

int main()
{
    ll t,n,i,j,k,w,p,q,ww;
    double x,y,z;
    s(t);
    for(i=1;i<=t;i++) {
        s(p);s(q);s(k);
        ll ans=0;
        for(j=0;j<=10000000;j++) {
            w=k-j-pow(j,p);
            if(w<0) break;
            double x = log((double)w);
            x = x/(double)q;
            x=exp(x);
            ll xx = round(x);
            ww=j+pow(j,p)-k+pow(xx,q);
            if(ww==0) ans++;
        }
        cout<<"Case #"<<i<<": "<<ans<<endl;
    }
    return 0;
}
