#include <bits/stdc++.h>
#define ll long long
#define mylimit 1000001
using namespace std;
int main()
{
	set<ll int> s;
	ll int a[1000010];
	memset(a,0,sizeof(a));
	ll int i,j,t;
	//~ a[0]=1;
	a[1]=1;
	//~ s.insert(0);
	s.insert(1);
	for(i=2;i<mylimit;i++)//hash all perfect powers for all possible i^j
	{
		t=i;
		for(j=2;;j++)
		{
			t*=i;
			if(t<mylimit)
			{
				//~ cout<<t<<endl;
				a[t]=1;
				s.insert(t);
			}
			else
			break;
		}
	}
	ll int test;
	scanf("%lld",&test);
	while(test--)
	{
		ll int x;
		scanf("%lld",&x);
		int flag=0;
		set<ll int>::iterator it;
		for(it=s.begin();it!=s.end();it++)
		{
			i=*it;
			if((x-i)>=0 && a[x-i]==1)
			{
				//~ cout<<i<<" "<<(x-i)<<endl;
				flag=1;
				break;
			}
		}
		if(flag)
		{
			printf("Yes\n");
		}
		else
		{
			printf("No\n");
		}
	}
	return 0;
}
