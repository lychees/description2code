#include <bits/stdc++.h>
using namespace std;
char a[52];
bool dp[52][52];
int main(){
 int t;
 scanf("%d",&t);
 while(t--){
  scanf("%s",a);
  int n = strlen(a);
  for(int i=0;i<n;i++){
  	for(int j=0;j<n;j++){
  	 if(i==j) dp[i][j] = true;
  	 else dp[i][j] = false;
  	}
  }
  reverse(a,a+n);
  for(int len=2;len<=n;len++){
    for(int i=0;i<n-len+1;i++){
      int j = i+len-1;
      if(a[i]==a[j] && len==2) dp[i][j] = true;
      else if(a[i]==a[j] && dp[i+1][j-1]) dp[i][j] = true;
    }
  }
  int maxi = 0;
  for(int i=0;i<n;i++){
    if(dp[0][i]==true) maxi = max(maxi,i+1);
  }
  printf("%d\n",n+n-maxi);
 }
 return 0;
}
