#include "bits/stdc++.h"
using namespace std;
bool palindrome(long long number)
{
	int arr[10];
	int arrC=0;
	while(number)
	{
		arr[arrC++]=number%10;
		number=number/10;
	}
	for(int i=0,j=arrC-1;i<j;i++,j--)
		if(arr[i]!=arr[j])
			return 0;
	return 1;
}
int main()
{
	long long arr[100001];
	long long arrC=0;
	
	for(long long i=0;i<=100000;i++)
		if(palindrome(i))
			arr[arrC++]=i;
	
//	cout << arrC << endl;
	
	long int tc;
	cin >> tc;
	while(tc--)
	{
		long long a,b;
		cin >> a >> b;
		
		long long ans=0;
		for(long long i=0;i<arrC;i++)
			if(arr[i] >=a && arr[i] <= b)
				ans++;
		cout << ans << endl;
	}
	return 0;
}
