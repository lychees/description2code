#include<bits/stdc++.h>
using namespace std;
inline int scan(){
    char c = getchar_unlocked();
    int x = 0;
    while(c<'0'||c>'9'){
        c=getchar_unlocked();
    }
    while(c>='0'&&c<='9'){
        x=(x<<1)+(x<<3)+c-'0';
        c=getchar_unlocked();
    }
    return x;
}
long long a[2001];
int main(){
    long long n=scan(),K=scan();
    for(int i=1;i<=n;i++){
        a[i]=scan();
    }
    sort(a+1,a+n+1);
    long long ans=0;
    for(int i=1;i<n-1;i++){
        for(int j=i+1;j<n;j++){
        	long long temp=a[i]*a[j];
        	if(temp*a[n]<=K){
            		ans+= n-j;
            		continue;
            }
            for(int k=j+1;k<=n;k++){
            	long long tempo=temp * a[k];
               if(tempo>K){
            		break;
            	}
            	else{
            		ans++;
            	}
            }
        }
    }
    printf("%lld\n",ans);
}
