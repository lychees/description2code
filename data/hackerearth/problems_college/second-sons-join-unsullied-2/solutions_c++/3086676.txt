int main() {
	int i, j, k, n, m;
	int t;
	cin>>t;
	string s;
	while(t--) {
		cin>>s;
		int l = s.length();
		j = 0;k = 0;
		for(i=0;i<l;i++) {
			if(s[i] =='S')j++;
			else k++;
		}
		if(l % 2 == 0) {
			if(abs(j-k) != 0) {
				printf("-1\n");
				continue;
			}
		}
		else {
			if(abs(j-k) != 1) {
				cout<<"-1"<<endl;
				continue;
			}
		}
		
		int a[l], b[l];
		for(i=0;i<l;i++) {
			a[i] = i%2;
			b[i] = 1-a[i];
		}
		int c1 =0;int c2 = 0;
		for(i=0;i<l;i++) {
			int x = (s[i] == 'S')? 1: 0;
			if(x!=a[i])c1++;
			if(x!=b[i])c2++;
		}
		int mini = min(c1, c2);
		if((l&1) == 0)
			cout<<mini/2<<endl;
		else {
			if(j>k)cout<<(c2+1)/2<<endl;
			else cout<<(c1+1)/2<<endl;
		}
	}
	return 0;
}
