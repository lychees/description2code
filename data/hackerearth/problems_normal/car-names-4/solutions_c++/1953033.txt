#include<cstdio>
#include<iostream>
#include<cstring>
#include<cstdlib>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string str;
		long long int i,j,len,max=0,flag =0,c=0;
		long long int count[26]={0};
		cin>>str;
		for(i=0;i<str.length();i++)
		{
			count[str[i]-'a']++;
			if(count[str[i]-'a']>max)
			{
				max = count[str[i]-'a'];
			}
		}
		for(i=0;i<26;i++)
		{
			if(count[i]!=0)
			{
			 c++;
			}
		}
		if(c!=3)
		{
			cout<<"Not OK"<<"\n";
		}
		else
		{
			for(i=0;i<26;i++)
			{
				if( count[i] !=0 && count[i]!=max)
				{
					flag =1;
					break;
				}
			}
			if(flag == 1)
			{
				cout<<"Not OK"<<"\n";
			}
			else
			if(str.compare("carcar")==0)
			{
				cout<<"Not OK"<<"\n";
			}
			else
			{
				cout<<"OK"<<"\n";
			}
		}
	}
}
