#include<iostream>

using namespace std;

int main()
{
	int i,n,t,h;
		cin>>t;

		while(t)
		{
			cin>>n>>h;
			int min=(n>=h)?h:n;
			bool a[n],b[n],c[n];

				for(i=0;i<n;i++)
					{
                     	cin>>a[i];
                         c[i]=a[i];
                      }
                    
					b[0]=a[1];
					b[n-1]=a[n-2];
                 if(h!=min)
                 {
				for(int j=0;j<n-2;j++)
				{
					b[0]=a[1];
					b[n-1]=a[n-2];
					for(int k=1;k<n-1;k++)
						b[k]=a[k-1]&&a[k+1];

					for(i=0;i<n;i++)
						{
                             a[i]=b[i];
                              c[i]=a[i];
                         }
				}
                  b[0]=c[1];
                  b[n-1]=c[n-2];  
                for(i=1;i<n-1;i++)
                   b[i]=c[i-1]&&c[i+1];
                }
               else
                 {
                    for(i=1;i<=h;i++)
                     {
                           b[0]=a[1];
					      b[n-1]=a[n-2];
					for(int k=1;k<n-1;k++)
						b[k]=a[k-1]&&a[k+1];


					for(int t=0;t<n;t++)
                             a[t]=b[t];
                     }
                  }  
			for(i=0;i<n;i++)
			{
				if(min!=h)
                  {
					  if(n%2==1&&h%2==1||n%2==0&&h%2==0)
                     
                          cout<<a[i]<<" ";
                       else
                         cout<<b[i]<<" ";
                      
                   }


				else
					cout<<a[i]<<" ";
			}
			cout<<endl;
		t--;
		}
return 0;
}
