#include <iostream>
#include <string>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int alpha[26]={0};
    	string s;
    	cin>>s;
    	for(int i=0;i<s.size();i++)
    	{
    		++alpha[s[i]-97];
    	}
    	int count=0;
    	for(int j=0;j<26;j++)
    	{
    		if(alpha[j]==1)
    		{
    			++count;
    		}
    	}
    	if(count==s.size())
    	{
    		cout<<"No\n";
    	}
    	else
    	{
    		cout<<"Yes\n";
    	}
    }
    return 0;
}
