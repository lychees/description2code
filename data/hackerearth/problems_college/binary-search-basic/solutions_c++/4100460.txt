int bsearch(int arr[],int x,int n)
 {
 
     int up=0,lp=n-1,mid;
 
     while(up<=lp)
      {
           //cout<<"IN"<<endl;
          mid= (up+lp)/2;
          if(arr[mid]==x)
              {
                 cout<<"YES ";
                 return mid;
              }
          else if(arr[mid]<x)
                    up=mid+1;
          else
                   lp=mid-1;
 
      }
     cout<<"NO ";
     return mid;
 }
