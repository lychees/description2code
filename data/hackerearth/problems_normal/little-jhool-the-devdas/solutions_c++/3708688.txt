#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n, x;
    cin>>n;
    vector<int > odd;
    int sum=0;
    for(int i=0; i<n; i++){
    	cin>>x;
    	if(x%2==0){
    		sum+=x;
    	}else{
    		odd.push_back(x);
    	}
    }
    sort(odd.begin(), odd.end());
    int s = odd.size();
    if(s==0){
    	cout<<":(\n";
    }else{
    	s--;
    	if(s>0){
	    	int y=0;
	    	if(s%2 == 1){
	    		y++;
	    	}
	    	for(int i=y; i<s; i++){
				sum+=odd[i];
			}
    	}
    	sum += odd[odd.size()-1];
	    cout<<sum<<endl;
    }
    
    return 0;
}
