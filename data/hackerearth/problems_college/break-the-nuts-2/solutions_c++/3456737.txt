#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,l,i,a=-1;
    long long result=0;
    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>l;
        if(l==1)
        {
            if(a==-1)
            {
                result=1;
            }
            else
            {
                result*=i-a;
            }
            a=i;
        }
    }
    cout<<result<<endl;
}
