#include<iostream>
#include<vector>
#define lli long long int
using namespace std;
int main()
{
    lli t;
    cin>>t;
    vector<lli> res;
    for(lli i=0;i<t;i++)
    {
        lli n;
        cin>>n;
        res.push_back(14*n-6);
    }
    for(lli i=0;i<t;i++)
        cout<<res[i]<<endl;
    return 0;
}
