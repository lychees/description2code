#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    char s[100000];
    int n,i,l;
    //scanf("%[^\n]s",s);
    gets(s);
    l=strlen(s);
    scanf("%d",&n);
    if(n<=20&&l<=10000)
    {
    for(i=0;i<l;i++)
    {
    	if((i)%n==0&&i!=0)
    	{
    		printf("\n");
    		printf("%c",s[i]);
    		
    	}
    	else printf("%c",s[i]);
    }
    }
    return 0;
}
