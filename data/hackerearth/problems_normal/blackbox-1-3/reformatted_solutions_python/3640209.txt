def gcd(a, b):
    if a < b:
        c = a
        a = b
        b = c
    if b == 0:
        return a
    else:
        if a % b == 0:
            return b
        else:
            return gcd(b, a % b)


t = input()
for i in range(t):
    s = sorted(map(int, raw_input().split()))
    if s[0]*s[0]+s[1]*s[1] == s[2] * \
            s[2] and gcd(s[1], s[0]) == 1 and gcd(s[2], s[0]) == 1:
        print "YES"
    else:
        print "NO"
