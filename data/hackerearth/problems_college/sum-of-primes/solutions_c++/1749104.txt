using namespace std;
typedef long long ll;
//#define DEBUG
#define mod 1000000007
#define pb push_back

int is_prime[1000001]={0};

void markMultiples(int a){
    int i=2, num;
    while((num=i*a)<=1000000){
        i++;
        is_prime[num] = 1;
    }
}

void precalc(){
    ll num;
    is_prime[1] = 1;
    for(int i=2;i<=1000000;i++){
        if(is_prime[i]==0){
            markMultiples(i);
        }
    }
}

bool isPrime(ll n){
    if(n==2 || n==3)
        return true;
    if(n%2==0 || n%3==0)
        return false;

    ll i=5, w=2;
    while(i*i <= n){
        if(n%i==0)
            return false;
        i+=w;
        w = 6-w;
    }
    return true;
}

int main(){
    //ios::sync_with_stdio(false);
    #ifdef DEBUG
    freopen("input.txt", "r", stdin);
    #endif // DEBUG

    int T,N,M;
    precalc();
    scanf("%d", &T);
    while(T--){
        scanf("%d %d", &M, &N);
        ll sum = 0;
        for(int i=M;i<=N;i++){
            if(is_prime[i]==0)
                sum += i;
        }
        if(sum==0)
            printf("NO\n");
        else if(sum>1000000){
            printf("%s\n", isPrime(sum)?"YES":"NO");
        }
        else if(is_prime[sum]==0)
            printf("YES\n");
        else
            printf("NO\n");
    }

    return 0;
}
