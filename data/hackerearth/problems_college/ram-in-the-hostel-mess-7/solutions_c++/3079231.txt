#include<bits/stdc++.h>
using namespace std;
main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		long long int ans;
		if(n<=3)
		{
			ans=pow(2,n-1)+1;
			cout<<ans*ans<<endl;
		}
		else
		{
			int array[n+1];
			array[1]=2;
			array[2]=3;
			for(int i=3;i<=n;i++)
			{
				array[i]=array[i-1]+array[i-2];
			}
			cout<<array[n]*array[n]<<endl;
			
		}
	}
}
