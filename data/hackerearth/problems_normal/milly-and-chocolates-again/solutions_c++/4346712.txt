#include<iostream>
using namespace std;

int N,Q,*box,**sol;

int BinarySearch(int q,int l,int h){
    int mid=(l+h)/2;

    if(l+1==h){
        if(sol[l][1]>q || sol[l][1]==q){
            return l+1;
        }
        else return h+1;
    }

    if(sol[mid][1]<q){
        return BinarySearch(q,mid,h);
    }
    else if(sol[mid][1]>q){
        return BinarySearch(q,l,mid);
    }
    else return mid+1;
}

int main(){
    cin>>N;
    box=new int[N];
    sol=new int*[N];
    for(int i=0;i<N;i++){
        cin>>box[i];
        sol[i]=new int[2];
    }
    sol[0][0]=1;
    sol[0][1]=box[0];
    for(int i=1;i<N;i++){
        sol[i][0]=sol[i-1][1]+1;
        sol[i][1]=sol[i-1][1]+box[i];
    }

    cin>>Q;
    for(int i=0;i<Q;i++){
        int a;
        cin>>a;
        cout<<BinarySearch(a,0,N-1)<<endl;
    }
    return 0;
}
