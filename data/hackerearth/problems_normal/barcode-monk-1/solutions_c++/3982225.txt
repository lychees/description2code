#include <bits/stdc++.h>
using namespace std;
#define md 1000000007ll
long long int n,m,a;
long long sv[1001][3];
bool ok[11111111];
	
int main()
{	
	ok[0]=1;
	cin>>n>>m;
	for(int i=0;i<m;i++){
		cin>>a;
		ok[a]=1;
		if(a>n) continue;
		for(int j=a;j<=n;j++){
			ok[j]|=ok[j-a];
		}
	}
	sv[0][0]=0;
	sv[0][1]=1;
	for(int i=1;i<=n;i++){
		sv[i][1]=sv[i-1][0]+sv[i-1][1];
		sv[i][1]%=md;
		sv[i][0]=0;
		for(int j=1;j<=i;j++){
			if(ok[j]){
				sv[i][0]+=sv[i-j][1];
				sv[i][0]%=md;
			}
		}
		sv[i][0]%=md;
	}
	cout<<(sv[n][0]+sv[n][1])%md<<endl;
	return 0;
}
