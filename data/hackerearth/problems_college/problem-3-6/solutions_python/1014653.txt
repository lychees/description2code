n = long(input())
team_kits = []
for i in xrange(n) :
	x,y = [long(a) for a in raw_input().split(' ')]
	team_kits.append([x,y])
home = [0]*n
away = [0]*n
for i in xrange(n) :
	for j in xrange(n) :
		if i == j :
			continue
		elif team_kits[i][0] == team_kits[j][1] :
			home[j] += 1
		else :
			away[j] += 1
		home[i] += 1
for k in xrange(n) :
	print home[k],away[k]
