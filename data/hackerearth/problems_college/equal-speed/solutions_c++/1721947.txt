#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

vector<int> speed;

int main() {
	int n,k; cin >> n >> k;
	for(int i=0;i<n;i++){
		int temp; cin >> temp;
		speed.push_back(temp);
	}

	sort(speed.begin(),speed.end());

	cout << speed[speed.size()-k] << endl;

	return 0;
}
