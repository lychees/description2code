#include <iostream>
#include<bits/stdc++.h>
using namespace std;
#define ll long long

int at[26];
int as[10010][26];
ll sum[10010];
vector<pair<ll,int> >vp;
vector<pair<ll,int> >v;
vector<int>len;

int main()
{
    int t,n,m,i,j,k;
    string a,s;
    cin>>t;
    while(t--)
    {
    	cin>>n>>a;
    	memset(at,0,sizeof(at));
    	memset(as,0,sizeof(as));
    	memset(sum,0,sizeof(sum));
    
    	for(i=0;i<a.length();i++)
    	{
    		at[a[i]-'a']++;
    	}
    	
    	for(i=0;i<n;i++)
    	{
    		cin>>s;
    		len.push_back(s.length());
    		for(j=0;j<s.length();j++)
    		{
    			as[i][s[j]-'a']++;
    		}
    	}
    	
    	for(i=0;i<n;i++)
    	{
    		for(j=0;j<26;j++)
    		{
    			if(at[j] && as[i][j])
    				sum[i]+=(1ll*(at[j])*(as[i][j]));
    		}
    	}
    	
    /*	for(i=0;i<n;i++)
    	{
    		cout<<sum[i]<<" ";
    	}*/
    	
    	for(i=0;i<n;i++)
    	{
    		vp.push_back(make_pair(sum[i],i));
    	}
    	
    	sort(vp.rbegin(),vp.rend());
    	
    	//if(!vp.empty())
    	v.push_back(make_pair(vp[0].first,vp[0].second));
    	
    	for(i=1;i<vp.size();i++)
    	{
    		if(vp[i].first==vp[i-1].first)
    		{
    			v.push_back(make_pair(vp[i].first,vp[i].second));
    		}
    		else
    			break;
    	}
    	
    	int minx=1000010,maxl=1000100;
    	if(v.size()==1)
    	{
    		cout<<v[0].second+1<<endl;
    	}
    	else
    	{
    		for(i=0;i<v.size();i++)
    		{
    			if(len[v[i].second]<maxl)
    			{
    				minx=v[i].second;
    				maxl=len[v[i].second];
    			}
    			else if(len[v[i].second]==maxl)
    			{
    				minx=min(minx,v[i].second);
    			}
    		}
    		cout<<minx+1<<endl;
    	}
    	
    	vp.clear();
    	v.clear();
    	len.clear();
    }
    return 0;
}
