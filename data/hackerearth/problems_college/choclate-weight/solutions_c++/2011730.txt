#include <iostream>
#include <map>
using namespace std;

int main()
{
    int N;
    cin>>N;
    map<int,int>stuff;
    for(int i=0;i<N;i++)
    {
    	int n;
    	cin>>n;
    	stuff[n]=1;
    }
    int Q;
    cin>>Q;
    for(int i=0;i<Q;i++)
    {
    	int n;
    	cin>>n;
    	if(stuff[n]>0)
    	cout<<"I Like You"<<endl;
    	else
    	{   
    		int g=0;
    		map<int,int>::iterator it;
    		for(it=stuff.begin();it!=stuff.end();it++)
    		{
    			if(n/2>(*it).first)
    			{
    				if(stuff[n-(*it).first]>0)
    				{
    					g=1;
    					break;
    				}
    			}
    		}
    		if(g==1)
    		cout<<"I Like You"<<endl;
    		else
    		cout<<"I Hate You"<<endl;
    	}
    }
    return 0;
}
