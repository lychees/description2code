#include<bits/stdc++.h>
using namespace std;
#define MOD 1000000007
long long int power(long long int n,long long int m)
{
   if(m==0) return 1%MOD;
   long long int x=power(n,m/2);
   if(m%2==0) 
   return (x*x)%MOD;
   else
   return (((x*x)%MOD)*n)%MOD;
}


int main(){
	
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		n--;
		long long int ans=3*power(2,n)-power(-1,n);
		cout<<ans%MOD<<"\n";
		
	}
}
