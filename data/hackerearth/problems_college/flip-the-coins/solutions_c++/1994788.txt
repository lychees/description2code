#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int q;
    cin>>q;
    bool A[n];
    for(int i=0;i<n;i++)
    A[i]=0;
    for(int i=0;i<q;i++)
    {
    	int Q,a,b;
    	cin>>Q>>a>>b;
    	if(Q==0)
    		for(int i=a;i<=b;i++)
    			A[i]=1-A[i];
    	else
    		{
    			int ans=0;
    			for(int i=a;i<=b;i++)
    				if(A[i]==1)
    				ans++;
    			cout<<ans<<endl;
    		}
    }
    return 0;
}
