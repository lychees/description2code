using namespace std;

int main()
{
    long long p,q;
    long long i,j,k,tot,flag=0;
    scanf("%lld%lld",&p,&q);
    for(i=p;i<=q;i++)
     {
     	long long num=i*i,num1=0,num2=0,tmp;
     	long long l,r,d=0;
     	tot=0;
     	tmp=i;
     	while(tmp>0)
     	{
     		d++;
     		tmp=tmp/10;
     	}
     	long long f=1;
     	for(j=d;j>0;j--)
     	{
     		num1=num1+(num%10)*f;
     		num=num/10;
     		f=f*10;
     	}
     	f=1;
     	while(num>0)
     	{
     		num2=num2+(num%10)*f;
     		num=num/10;
     		f=f*10;
     	}
     	//cout<<num1<<" "<<num2<<endl;
     	if((num1+num2)==i)
        {
     	  printf("%lld ",i);
     	  flag=1;
        }

     }
     if(flag==0)
        cout<<"INVALID RANGE";
    return 0;
}
