#include <iostream>
#include <cstdio>
using namespace std;
 
int t,n;
long long arr[50005];
long long sum;
 
int main()
{
    scanf("%d",&t);
    
    while(t--){
    	sum = 0;
    	scanf("%d",&n);
    	for(int i=0;i<n;i++){
    		scanf("%llu",arr+i);
    		sum += arr[i];
    	}
    	
    	sum /= (n-1);
    	//cout<<"c"<<sum-arr[0]<<endl;
    	for(int i=0;i<n;i++){
    		cout << sum-arr[i]<<" ";
    	}
    	printf("\n");
    }
    
    return 0;
}
