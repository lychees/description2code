#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int a,b,n,i,value;
		cin>>a>>b>>n;
		vector<long long> v;
		for(i=0;i<n;i++)
		{
			cin>>value;
			v.push_back(value);
		}
		sort(v.begin(),v.end());
		long long int sum1=0,sum2=0,k=0;
		while(sum1<=a && sum2<=b && k<v.size())
		{
			sum1=sum1+v[k];
			sum2=sum2+v[k];
			k++;
		}
		if(k==v.size())
		{
			if((sum1<=a && sum2<=b )||(sum1>a && sum2>b) )
			cout<<"Tie"<<endl;
			else if(sum1>a && sum2<=b)
			cout<<"Sayan Won"<<endl;
			else if(sum2>b && sum1<=a)
			cout<<"Raghu Won"<<endl;	
		}
		else if(sum1>=a && sum2>=b)
			cout<<"Tie"<<endl;
			
		else if(sum1>a && sum2<=b)
			cout<<"Sayan Won"<<endl;
			
		else if(sum2>b && sum1<=a)
			cout<<"Raghu Won"<<endl;	
		v.clear();
	}
}
