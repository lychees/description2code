#include <bits/stdc++.h>
using namespace std;

int solve(unsigned long long z,unsigned long long m,unsigned long long n)
{
	unsigned long long n2,c=0;
    int sqrt_val=0,n1;
    sqrt_val=sqrt(z);
    for(n1=1;n1<=sqrt_val;n1++)
    {
    	n2=z/n1;
    	if(n1*n2==z)
    	{
    		if(n2<=m)
    		{
    			c++;
    		}
    		if(n2<=n && n2!=n1)
    		{
    			c++;
    		}
    	}
    }
    return c;
}
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	unsigned long long z,m,n;
    	cin>>z>>m>>n;
    	int d=solve(z,m,n);
    	cout<<d;
    	cout<<endl;
    }
    
    return 0;
}
