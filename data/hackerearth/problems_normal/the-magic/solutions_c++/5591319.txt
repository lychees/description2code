#include <iostream>
#include <math.h>
#include <cmath>
using namespace std;

int main()
{
	long t;
    cin>>t;
    
    while(t--)
    {
    	long n;
    	cin>>n;
    	
    	long ans(0);
    	while(n > 0)
    	{
    		long rem = n%2;
    		n /= 2;
    		if(rem == 1)
    			ans++;
    	}
    	
    	cout<<ans<<endl;
    }
    return 0;
}
