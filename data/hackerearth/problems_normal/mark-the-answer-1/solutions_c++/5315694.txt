#include <iostream>

using namespace std;

int main()
{
    int n,x;

    cin>>n>>x;

    int v[n];

    for(int i=0;i<n;i++)
    {
        cin>>v[i];
    }

    int rez=0,nr=0,s=0;

    for(int i=0;i<n;i++)
    {
        if(v[i]<=x)
        {
            nr++;
        }else if(s==0)
        {
            s=1;
        }else if(nr>rez)
        {
            rez=nr;
            nr=0;
            s=0;
        }else if(v[i]>x && s==1)
        {
            nr=0;
            break;
        }
    }
    
    if(rez==0)
    {
        cout<<nr;
    }else
    {
        cout<<rez;
    }

    return 0;
}
