#include <iostream>
long fun(long);
using namespace std;

int main()
{
    int t,n;
    cin>>t;
    while(t--){
    	cin>>n;
    	cout<<n-fun(n)<<endl;
    }
    return 0;
}
   long fun(long a){
   	long d=0;
   	if(a<3)
   	return 0;
   	else{
   		if(a%10>=3)
   		d=1;
   		a=a/10;
   		d=d+a;
   		return (d-fun(a)+10*fun(a));
   	}
   }
