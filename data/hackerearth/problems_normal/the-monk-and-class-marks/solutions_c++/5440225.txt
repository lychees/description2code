#include <iostream>
#include<map>
#include<algorithm>
using namespace std;

int main()
{
    int n;
    scanf("%d",&n);
    int marks; string name;
    vector<pair<int,string>> student;
    for(int i=0;i<n;i++){
    	cin>>name>>marks;
    	student.push_back(make_pair(-1*marks,name));
    }
    sort(student.begin(),student.end());
    for(int i=0;i<n;i++){
    	cout<<student[i].second<<" "<<-student[i].first<<endl;
    }
    return 0;
}
