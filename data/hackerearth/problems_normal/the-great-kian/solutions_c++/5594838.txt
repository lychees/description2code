#include <cstdio>
using namespace std;

int main(){
	int n, temp, count = 1;
	long long ans1 = 0, ans2 = 0, ans3 = 0;
	scanf("%d", &n);
	for (int i = 0; i < n; i++){
		scanf("%d", &temp);
		if (count == 0)
			ans1 += temp;
		else if (count == 1)
			ans2 += temp;
		else
			ans3 += temp;
		count = (count+1)%3;
	}
	printf("%lld %lld %lld\n", ans2, ans3, ans1);
    return 0;
}
