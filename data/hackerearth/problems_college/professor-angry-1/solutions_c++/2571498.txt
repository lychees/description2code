

    #include <cmath>
    #include <cstdio>
    #include <vector>
    #include <iostream>
    #include <algorithm>
    using namespace std;
     
     
    int main() {
        int T,n,r,k,a[1000],i,m,c;
        cin>>T;
        while(T){
            c=0;
            i=0;
            cin>>n;
            m=n;
            cin>>r;
            while(n){
                cin>>k;
                a[i]=k;
                i++;
                n--;
            }
    for(i=0;i<m;i++){
      if(a[i]<=0)
          c++;
    }
            //cout<<c;
            if(c>=r)
                cout<<"NO"<<endl;
            else
                cout<<"YES"<<endl;
            
            T--;
        }
           
        return 0;
    }
