#include <iostream>
using namespace std;

int main()
{
    int n1;
    cin>>n1;
    for(int i=1;i<=n1;i++)
    {
    	long int m,k,sum=0,term=0;
    	cin>>m>>k;cin>>term;
    	for(long int j=2;j<=m;j++)
    	{
    		long int l;
    		cin>>l;
    		
    		if(term-l==k||term-l==-k)
    		sum++;
    		term=l;
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
