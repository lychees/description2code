bool arr[1000000];
bool has7(int x)
{
	//int count=0;
	do
	{
		if(x%10==7)return true;
		x/=10;
	}
	while(x);
	return false;
}

void SieveOfEratosthenes(int n)
{
   arr[0]=arr[1]=true;
	for(int i=2;i*i<1000000;i++)
		if(arr[i]==false)
		{
			for(int j=i*i;j<1000000;j=j+i)
				arr[j]=true;
		}
}  


     
int main()
{
	SieveOfEratosthenes(1000000);
	ll t,n,m;
	cin>>t;	
	while(t--)
	{
		scanf("%lld %lld",&m,&n);
			ll count=0;
			for(int i=m;i<=n;i++)
			{
				if(arr[i]==true && has7(i)==true)count++;				
			}
			if(count==0)printf("-1\n");
			else printf("%lld\n",count);
	}
}
