#include <bits/stdc++.h>
using namespace std;
set<int> s;
vector<int> v(1000001);
int palin(int n)
{
    int r,temp,rev=0;
    temp=n;
    while(temp>0)
    {
        r=temp%10;
        rev=(rev*10)+r;
        temp/=10;
    }
    if(rev==n)
        return 1;
    return 0;
}
int main()
{
    int i,j,t,x;
    for(i=2;i<=1000000;i++)
    {
        if(v[i]==0)
        {
            if(palin(i))
            {
                s.insert(i);
            }
            for(j=2*i;j<=1000000;j=j+i)
            {
                v[j]=1;
            }
        }
    }
    s.insert(1003001);
    set<int>::iterator it;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&x);
        it=s.lower_bound(x);
        printf("%d\n",*it);
    }
    return 0;
}
