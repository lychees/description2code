#include <iostream>
#include<math.h>
using namespace std;

int main()
{
   int x,a,b,y;
   long int t;
   cin>>t;
   while(t--)
   {
   	cin>>x>>y>>a>>b;
   	if(abs(x-a)==abs(y-b))
   	cout<<"0"<<endl;
   	
   	else if((abs(x-a)%2==1||abs(y-b)%2==1)&&!((abs(x-a)%2==1&&abs(y-b)%2==1)))
   	cout<<"-1"<<endl;
   	else //(((x-a)%2==1&&(y-b)%2==1)||((x-a)%2==0||(y-b)==0))
   	{
   		cout<<"1"<<endl;
   	}
		   	
   }
    return 0;
}
