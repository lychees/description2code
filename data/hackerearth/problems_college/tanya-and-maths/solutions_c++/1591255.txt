#include <bits/stdc++.h>
using namespace std;
int phi (int n) {
    int result = n;
    for (int i=2; i*i<=n; ++i)
        if (n % i == 0) {
            while (n % i == 0)
                n /= i;
            result -= result / i;
        }
    if (n > 1)
        result -= result / n;
    return result;
}

int main()
{
    //cout << "Hello World!" << endl;
    int t,a,b,c=0;
    cin>>t;
    while(t--)
    {
    c=0;
    cin>>a>>b;
    for(int i=1;i<=min(a,b);i++)
    {
    	if(a%i==0&&b%i==0)
    	{
    		c=c+phi(i);
    	}
    }
    cout<<c<<endl;
    }
    
    return 0;
}
