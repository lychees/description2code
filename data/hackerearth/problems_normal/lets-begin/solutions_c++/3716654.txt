#include <bits/stdc++.h>
#define ll long long int
using namespace std;

int main()
{
	int t,ans;
	ll x;
	cin>>t;
	while(t--){
		cin>>x;
		if(x<8){
			if(x==2 || x==3 || x==5 || x==7){
				ans =1;
			}
			else{
				ans=-1;
			}
		}
		else if(x%7==0){
			ans = x/7;
		}
		else if(x%7==1 || x%7==2 || x%7==3 || x%7 == 5){
			ans = x/7 +1;
		}
		else if(x%7==4 || x%7==6) {
			ans = x/7 +2;
		}
		else{
			ans = -1;
		}
		cout<<ans<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
