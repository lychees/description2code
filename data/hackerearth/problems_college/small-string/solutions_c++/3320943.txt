string find(int n,string s){
  string p;
  int k;
  k=1;
  p.push_back(s[1]);
  int i;
  for( i=2;i<n;i++){
     
    if(s[i]>s[0]&&k){
      p.push_back(s[0]);
      p.push_back(s[i]);
      k=0;
    }
      else
      p.push_back(s[i]);
    }
    if(k)
    p.push_back(s[i]);
    
  int j;
  for(j=1;j<n;j++){
    k=1;
    string f;
    for( i=0;i<n;i++){
      if(i==j)
        continue;
      if(i==j+1)
      {
        f.push_back(s[i]);
        continue;
      }
     
    if(s[i]>s[j]&&k){
      f.push_back(s[j]);
      f.push_back(s[i]);
      k=0;
    }
      else
      f.push_back(s[i]);
    }
    if(k)
    f.push_back(s[j]);
    if(f<p)
      p=f;
  }
    
  return p;
  
}
