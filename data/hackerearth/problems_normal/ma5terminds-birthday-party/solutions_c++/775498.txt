#include<iostream>
#include<cmath>
#include<stdbool.h>
#include<cstdlib>
#include<cstring>
#define MAX 1000001
using namespace std;

int P[MAX];
void fun()
{
	for(int i=2;i<MAX;i++)
	{
		if(!P[i])
		{
			int j=2*i;
			while(j<MAX)
			{
				P[j]=1;
				j+=i;
			}
	}
		}
	P[4]=0;
	P[1]=1;
	for(int i=1;i<MAX;i++)
	  P[i]+=P[i-1];
}
int main()
{
    int t;
    cin>>t;
    fun();
    while(t--)
    {
    	int l,r;
    	scanf("%d%d",&l,&r);
    	printf("%d\n",P[r]-P[l-1]);
    }
    return 0;
}
