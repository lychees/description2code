#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
     long long int n,m,k;
     cin>>n>>k>>m;
     long long int a[n],i,sum=0,min=1e18,pos=0;
     for(i=1;i<=n;i++)
        cin>>a[i];
     long long int temp,j=0;
     for(i=1;i<=n;i++)
     {
         j=1;
         sum+=(k*a[i]);
         while(j<=a[i])
         {
             cin>>temp;
             sum+=temp;
             j++;

         }
         sum+=((a[i]-1)*m);
         if(sum<=min)
         {
             min=sum;
             pos=i;
         }
         sum=0;
     }
     cout<<pos<<" "<<min<<endl;
     sum=0;
     min=1e18;

    }
}
