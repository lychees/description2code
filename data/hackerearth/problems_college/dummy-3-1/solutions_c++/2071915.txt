#include <iostream>
using namespace std;

int main()
{
  //  cout << "Hello World!" << endl;
   // return 0;
   
   int t;
   cin>>t;
   while(t--)
   {
   	int n,i,j=1,g=0;
   	cin>>n;
   	int a[n],b[n];
   	
   	for(i=0;i<n;i++)
   	cin>>a[i];
   	
   	for(i=0;i<n;i++)
   	{
   	int k=a[i];
   	
   	b[k-1]=j++; 		
   	}
   	
   	for(i=0;i<n;i++)
   	if(a[i]!=b[i])
   	{
   		g=1;
   		break;
   	}
   	
   	if(g==1)
   	cout<<"not inverse"<<"\n";
   	else
   	cout<<"inverse"<<"\n";
   	
   }
}
