# define base complex<double>
# define vb vector<base>

vb omega;
int fft_n;
//int n;
void initFFT(int n)
{
 	fft_n=n;
 	omega.resize(fft_n);
 	double angle=2*PI/n;
 	for(int i=0;i<n;i++)
      	omega[i]=base(cos(i*angle),sin(i*angle));
}
void fft(vb &a)
{
 	int n=a.size();
 	if(n==1)return;
 	int half=n>>1;
 	vb even(half),odd(half);

 	for(int i=0,j=0;i<n;i+=2,++j)
 	{
      	even[j]=a[i]; odd[j]=a[i+1];
 	}
 	fft(even);fft(odd);

 	base tmp;
 	for (int i=0, fact = fft_n/n; i < half; i++)
 	{
      	tmp =  odd[i] * omega[i * fact] ;
      	a[i] =  even[i] + tmp;
      	a[i+half] = even[i] - tmp;
 	}
}
void multiply (const vector<long long> & a, const vector<long long> & b, vector<long long> & res)
{
 	vb fa (a.begin(), a.end()),  fb (b.begin(), b.end());
 	int n = 1;
 	while (n < 2*max (a.size(), b.size()))  n <<= 1;
 	fa.resize (n),  fb.resize (n);
        	initFFT(n);
 	fft (fa),  fft (fb);
 	for (size_t i=0; i<n; ++i)
      	fa[i] = conj( fa[i] * fb[i]);
 	fft (fa);
        	res.resize (n);
 	for (size_t i=0; i<n; ++i)
      	res[i] = (long long) (fa[i].real() / n + 0.5);
}

#define N 210000
#define B 100010
int vis[N];
vector<LL> a,b,res;

int mod(int a)
{
    if(a<0)a=-a;
    return a;
}

void solve()
{
    a.resize(N);
    b.resize(N);
    for(int i=0;i<N;i++)
    {
        a[i] = 0;
        b[i] = 0;
    }
    int n,tmp,ans = 0;
    sd(n);
    for(int i=0;i<n;i++)
    {
        sd(tmp);
        if(a[tmp] == 1)ans = 1;
        a[tmp] = 1;
        b[B-tmp] = 1;
    }
    multiply(a,b,res);
    for(int i=0;i<res.size();i++)
    {
        if(i>=B)break;
        if((res[i] > 0) && (vis[B-i] == 0))
        {//cout<<B-i<<endl;
            ans++;
            vis[B-i] = 1;
        }
    }
    cout<<ans<<endl;
}

int main()
{
    //freopen("1.in","r",stdin);
	int t=1;
	//scanf("%d",&t);
	for(int i=1;i<=t;i++)
	{
		solve();
	}
}
