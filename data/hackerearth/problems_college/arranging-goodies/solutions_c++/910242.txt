#include<bits/stdc++.h>
using namespace std;

struct element{
	double floatvalue;
	int integervalue;
};

inline bool comp(const struct element &a, const struct element &b) {
	if(a.floatvalue < b.floatvalue) {
		return true;
	} 
	else if(a.floatvalue == b.floatvalue) {
		if(a.integervalue < b.integervalue) {
			return true;
		} 
		else {
			return false;
		}
	}
	return false;
}

int main()
{
	int n,m;
	scanf("%d%d", &n, &m);
	double ans = (double)(m + 1)/(double)2;
	struct element a[m + 1];
	for(int i = 1 ; i <= m; i++) {
		a[i - 1].integervalue = i;
		a[i - 1].floatvalue = abs(ans - double(i));
	}
	sort(a, a + m, comp);
    for(int i = 0 ; i < n; i++) {
        printf("%d\n", a[i%m].integervalue);
    }
	return 0;
}
