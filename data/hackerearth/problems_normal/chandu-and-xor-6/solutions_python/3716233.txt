n,m = map(int, raw_input().split())
arr = []
check = n
row, col = [0]*n, [0]*m
while check>0:
	arr.append([int(x) for x in raw_input().split()])
	check -= 1
for i in range(n):
	for j in range(m):
		row[i] += arr[i][j]
		col[j] += arr[i][j]
result = 0
for i in range(n):
	for j in range(m):
		result = max(result, row[i]^col[j])
print result
