#include <iostream>
#include <algorithm>

using namespace std;

#define L long
int main() {
	L n;
	L k;
	
	scanf ("%ld %ld", &n, &k);
	
	L x[k];
	L y[k];
	
	L cnt = 0;
	
	for (int i = 0; i < k; i++) {
		scanf ("%ld", &x[i]);
	}
	
	for (int i = 0; i < k; i++) {
		scanf ("%ld", &y[i]);
	}
	
	L steps = 0;
	
	for (int i = 0; i < k; i++) {
		L min = 2 * n;
		
		L c1 = abs(x[i] - 1) + abs(y[i] - 1);
		
		if (min > c1) {
			min = c1;
		}
		
		L c2 = abs(x[i] - 1) + abs(y[i] - n);
		
		if (min > c2) {
			min = c2;
		}
		
		L c3 = abs(x[i] - n) + abs(y[i] - 1);
		
		if (min > c3) {
			min = c3;
		}
		
		L c4 = abs(x[i] - n) + abs(y[i] - n);
		
		if (min > c4) {
			min = c4;
		}
		
		steps += min;
	}
	
	printf ("%ld\n", steps);
	
	return 0;
}
