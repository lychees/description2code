#include<iostream>
 
using namespace std;
 
long distance(long x1, long y1, long x2, long y2)
{
	long dx = x2-x1;
	long dy = y2-y1;
	
	if(dx < 0)
		dx*=-1;
	if(dy < 0)
		dy*=-1;
		
	return dx+dy;
}
 
long minDistance(long N, long x, long y)
{
	long cx1,cx2,cx3,cx4;
	long cy1, cy2, cy3, cy4;
	
 
	cx1 = 1;
	cy1 = 1;
	
	cx2 = 1;
	cy2 = N;
	
	cx3 = N;
	cy3 = 1;
	
	cx4 = N;
	cy4 = N;
	
	
	
	long d1 = distance(cx1,cy1,x,y);
	long d2 = distance(cx2,cy2,x,y);
	long d3 = distance(cx3,cy3,x,y);
	long d4 = distance(cx4,cy4,x,y);
	
	if((d1<d2) && (d1<d3) && (d1<d4))
		return d1;
	else if((d2<d1) && (d2<d3) && (d2<d4))
		return d2;
	else if((d3<d1) && (d3<d2) && (d3<d4))
		return d3;
	else return d4;
	
}
 
int main()
{
	long n, k;
	
	long x[100000];
	long y[100000];
	
	cin >> n;
	cin >> k;
 
	
	long total=0;
	
	for(long i=0;i<k;i++)
	{
		cin >> x[i];
	}
	
	for(long i=0;i<k;i++)
	{
		cin >> y[i];
	}
	
	for(long i=0;i<k;i++)
	{
		total+=minDistance(n,x[i],y[i]);
	}
	cout<< total<< endl;
	return 0;
	
}
