#include <stdio.h>
#define mod 1000000007
int main()
{
	int t;
	scanf("%d",&t);
	getchar();
	while(t--)
	{
		int A[26],B[26];
		char c;
		int i, j;
		long long int p=1;
		for(i=0;i<26;i++)
			A[i]=B[i]=0;
		while(1)
		{
			scanf("%c",&c);
			if(c!='\n')
				A[c-97]++;
			else 
				break;
			///printf("%c",c);
		}
		while(1)
		{
			if(scanf("%c",&c)==EOF)
				break;
			if(c=='\n')
				break;
			B[c-97]++;
			//printf("%c",c);
		}
		//Permutations
		for(i=0;i<26;i++)
		{
			//printf("A[i]=%d B[i]= %d\n",A[i],B[i]);
			int q=A[i]-B[i];
			if(q<0)
			{
				p=p*0;
				continue;
			}
			for(j=q+1;j<=A[i];j++)
			{
				p=p*j;
				p=p%mod;
			}
		}
		printf("%lld\n",p);
	}
	return 0;
}
