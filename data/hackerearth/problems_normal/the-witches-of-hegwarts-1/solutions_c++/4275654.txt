#include <bits/stdc++.h>
using namespace std;
map<int,int> vis;

int main()
{
    int T;
    cin>>T;
    while(T--)
    {
    	long N;
    	int a;
    	cin>>N;
    	queue<int> q;
    	q.push(N);
    	vis.clear();
    	vis[N] = 0;
    	
    	while(!q.empty())
    	{
    		a = q.front();
    		q.pop();
    		if(a==1)
    			break;
    		if(a%3==0)
    		{
    			if(!vis[a/3])
    			{
    				vis[a/3] = vis[a]+1;
    				q.push(a/3);
    			}
    		}
    		if(a%2==0)
    		{
    			if(!vis[a/2])
    			{
    				vis[a/2] = vis[a]+1;
    				q.push(a/2);
    			}
    		}
    		if(a-1>0)
    		{
    			if(!vis[a-1])
    			{
    				vis[a-1] = vis[a]+1;
    				q.push(a-1);
    			}
    		}
    	}
    	cout<<vis[1]<<endl;
    }
}
