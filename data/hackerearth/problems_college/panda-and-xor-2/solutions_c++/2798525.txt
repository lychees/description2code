#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    long int a,b,c;
    while(t--)
    {
    	int count=0;
    	cin>>a>>b;
    	c=a^b;
    	while(c!=0)
    	{
    		if(c&1==1)
    		count++;
    		c=c>>1;
    	}
    if(count%2)
    cout<<"NO"<<endl;
    else
    cout<<"YES"<<endl;
    }
    return 0;
}
