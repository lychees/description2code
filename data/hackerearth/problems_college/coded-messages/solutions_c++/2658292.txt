#include<iostream>
using namespace std;
int main()
{
	int t,l,i,j,k,sum;
	char s[23];
	int a[]={1,5,10,50,100,500,1000};
	cin>>t;
	while(t--)
	{
		cin>>s;sum=0;
		
	for (int i = 0; s[i]; ++i)
	{
		if(i>0 && s[i]>s[i-1])
			sum+= (a[s[i]-'A']) - 2*a[s[i-1]-'A'];
		else
			sum+=a[s[i]-'A'];
	}

		while(sum>0)
		{
			if(sum>=1000)
			{
				cout<<"G";
				sum=sum-1000;
			}
			else if(sum>=900)
			{
				cout<<"EG";
				sum-=900;
			}
			else if(sum>=500)
			{
				cout<<"F";
				sum-=500;
			}
			else if(sum>=400)
			{
				cout<<"EF";
				sum-=400;
			}
			else if(sum>=100)
			{
				cout<<"E";
				sum-=100;
			}
			else if(sum>=90)
			{
				cout<<"CE";
				sum-=90;
			}
			else if(sum>=50)
			{
				cout<<"D";
				sum-=50;
			}
			else if(sum>=40)
			{
				cout<<"CD";
				sum-=40;
			}
			else if(sum>=10)
			{
				cout<<"C";
				sum-=10;
			}
			else if(sum>=9)
			{
				cout<<"AC";
				sum-=9;
			}
			else if(sum>=5)
			{
				cout<<"B";
				sum-=5;
			}
			else if(sum>=4)
			{
				cout<<"AB";
				sum-=5;
			}
			else if(sum>=1)
			{
				cout<<"A";
				sum-=1;
			}
			
		}
		 
		cout<<endl;
	}
	
}
