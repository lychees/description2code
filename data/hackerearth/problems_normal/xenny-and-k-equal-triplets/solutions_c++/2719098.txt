#include <iostream>
using namespace std;
long long gcd (long long r, long long s) {
  if (!r) return s;
  return gcd ( s%r, r);
}
int main()
{
	long long t, n, c, k, x;
    cin >> t;
    while(t--) {
    	cin >> n >> k;
    	c = 0;
    	for (int i = 0; i < n; i++) {
    		scanf("%lld", &x);
    		if (x == k)
    			c++;
    	}
    	if (c < 3 && n < 3) {
    		cout << "0/0" << endl;
    		continue;
    	}
    	
    	long long num, den;
    	den = n * (n - 1) * (n - 2) / 6;
    	num = c * (c - 1) * (c - 2) / 6;

		if (c < 3) {
    		cout << "0/" << 1 << endl;
    		continue;
    	}
    	
    	long long g = gcd (num, den);
    	num /= g;
    	den /= g;
    	cout << num << "/" << den << endl;
    }
    return 0;
}
