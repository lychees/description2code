for _ in range(input()):
	x = input()
	li = list(map(int,raw_input().split()))
	dist = input()
	remaining = dist % sum(li)
	t = 0
	if remaining == 0:
		for x in range(0,len(li)):
			if li[x] != 0:
				t = x + 1
		print t
	else:
		for x in li:
			remaining = remaining - x
			t = t + 1
			if remaining == 0 or remaining < 0:
				print t
				break
