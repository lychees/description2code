#include<bits/stdc++.h>
using namespace std;
#define max 10000002
int a[max];
vector <int> v;
void seive(){
	for(int i=2;i<max;i++)
		a[i]=i;
	for(int i=2;i<max;i++)
		if(a[i]==i){
			a[i]=i-1;
			for(int j=i*2;j<max;j+=i){
				a[j]*=(i-1);
				a[j]/=i;
			}
			
		}
}
int main(){
	seive();
	int t;
	scanf("%d",&t);
	while(t--){
		int n;
		scanf("%d",&n);
		int x=a[n];
		printf("%d\n",x);
	}
}
