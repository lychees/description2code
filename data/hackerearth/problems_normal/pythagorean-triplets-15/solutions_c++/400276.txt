#include <iostream>
#include <cmath>
using namespace std;
long output;

long gcd(long a, long b)
{
	long temp;
	while(b>0)
	{
		temp = b;
		b= a%b;
		a = temp;		
	}
	return a;
}
long no_triplets(long N)
{
	long c;
	long a;
	long b;
	long answer=0;
	long multiple;
	for(int i=2; i<=sqrt(N);i++)
	{
		for(int j=1;j<i;j++)
		{
			multiple=0;
			c=i*i +j*j;
			if(c <= N && gcd(i,j)==1 && (i-j)%2)
			{
				while(++multiple * c <= N )
				answer++;
			}
		}
	}
	return answer;
}
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    long n;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	cin>>n;
    	output= no_triplets(n);
    	cout <<output<<endl;
    }
        
    //cout << "gcd (15,6):" << gcd(15,6);

}
