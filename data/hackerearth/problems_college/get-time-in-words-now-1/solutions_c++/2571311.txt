#include <iostream>
#include<string>
using namespace std;

string see(int m)
{
	string hour;
	switch(m)
	{
		case 1: hour="one"; return hour; break;
		case 2: hour="two"; return hour; break;
		case 3: hour="three"; return hour; break;
		case 4: hour="four"; return hour; break;
		case 5: hour="five"; return hour; break;
		case 6: hour="six"; return hour; break;
		case 7: hour="seven"; return hour; break;
		case 8: hour="eight"; return hour; break;
		case 9: hour="nine"; return hour; break;
		case 10: hour="ten"; return hour; break;
		case 11: hour="eleven"; return hour; break;
		case 12: hour="twelve"; return hour; break;
		case 13: hour="thirteen"; return hour; break;
		case 14: hour="fourteen"; return hour; break;
		case 16: hour="sixteen"; return hour; break;
		case 17: hour="seventeen"; return hour; break;
		case 18: hour="eightteen"; return hour; break;
		case 19: hour="nineteen"; return hour; break;
		case 20: hour="twenty"; return hour; break;
		case 21: hour="twenty one"; return hour; break;
		case 22: hour="twenty two"; return hour; break;
		case 23: hour="twenty three"; return hour; break;
		case 24: hour="twenty four"; return hour; break;
		case 25: hour="twenty five"; return hour; break;
		case 26: hour="twenty six"; return hour; break;
		case 27: hour="twenty seven"; return hour; break;
		case 28: hour="twenty eight"; return hour; break;		
		case 29: hour="twenty nine"; return hour; break;		
	}
}


int main()
{
 	int h,m;
 	cin>>h;
	cin>>m;
	if(m==00) cout<<see(h)<<" o\' clock";
	else if(m==30) cout<<"half past "<<see(h);
	else if(m==45) cout<<"quarter to "<<see(h+1);
	else if(m==15) cout<<"quarter past "<<see(h);
	else if(m==1) cout<<see(m)<<" minute past "<<see(h);
	else if(m==59) cout<<see(60-m)<<" minute to "<<see(h+1);
	else if(m<30 && m>=0)
	{
		cout<<see(m)<<" minutes past "<<see(h);
	}
	else if(m>30 && m<=60)
	{
		cout<<see(60-m)<<" minutes to "<<see(h+1);
	}
    return 0;
}
