#include <iostream>
using namespace std;

int main()
{
    int n,m,i,x,max,min,mid;
    cin>>n;
    int a[n];
    for(i=0;i<n;i++)
    	cin>>a[i];
    cin>>m;
    for(i=0;i<m;i++)
    {
    	cin>>x;
    	max=n-1;min=0;
    	mid=(min+max)/2;
    	while(min<=max)
    	{
    		if(x<a[mid])
    		{
    			max=mid-1;
    			mid=(min+max)/2;
    		}
    		else if(x>a[mid])
    		{
    				min=mid+1;
    				mid=(min+max)/2;
    				
    		}
    		else
    		{
    			cout<<mid<<endl;
    			break;
    		}
    	}
    	if(min>max)
    		cout<<-1<<endl;
    }	
    return 0;
}
