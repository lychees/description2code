#include <bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;
ll arr[1000006];
ll pre[1000006];
int main()
{
    ll n,k,i,j;
    scanf("%lld%lld",&n,&k);
    for(i=1;i<=n;i++)
        scanf("%lld",&arr[i]);
    pre[n]=arr[n];
    for(i=n-1;i>0;i--)
        pre[i]=(arr[i]+pre[i+1])%mod;
    ll sum1=0,sum2=0;
    for(i=1;i<=k;i++){
        sum1=(sum1+( (i*i)%mod * arr[i] )%mod)%mod;
        if(i==1){
            sum2=arr[i];
        }
        else{
        j=(((i) * (i))%mod) - (((i-1)*(i-1))%mod);
        if(j<0)
            j+=mod;
        sum2=(sum2 + ( j*arr[i] )%mod)%mod;
        //cout<<( j*arr[i] )<<endl;
        }
    }
    ll x,y,z;
    cout<<sum1<<" ";
    for(i=2;i<=n-k+1;i++){
        //cout<<"sum1 and sum2 "<<sum1<<"  "<<sum2<<endl;
        x=( (k*k)%mod * arr[i+k-1])%mod;
        j=(k*k)%mod - ((k-1)*(k-1))%mod;
        if(j<0) j+=mod;
        j=(j*arr[i+k-1]);
        z=sum1-sum2;
        if(z<0)
            z+=mod;
        z=(z+x)%mod;
        cout<<z<<" ";
        y=pre[i]-pre[i+k-1];
        if(y<0)
            y+=mod;
        y=(y*2)%mod;
        sum1=z;
        sum2=sum2-arr[i-1]-y;
        if(sum2<0)
            sum2+=mod;
        sum2=(sum2+j)%mod;
    }
    cout<<endl;
    return 0;
}
