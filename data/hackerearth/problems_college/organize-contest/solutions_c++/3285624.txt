int main()
{
    map<string,int>m;
    ll int n,v,i,maxv;
    string s,maxs;
    cin >> n;
    maxv = INT_MIN;

    for (i = 0; i < n; i++) {
        cin >> s >> v;
        if (m[s]+v > maxv) {
            maxv = m[s]+v;
            maxs = s;
            
        }
        m[s] += v;
    }
    cout << maxs;
    return 0;
}
