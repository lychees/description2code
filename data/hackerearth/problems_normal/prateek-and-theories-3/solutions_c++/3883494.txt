#include <iostream>
using namespace std;
#include<vector>
#include<algorithm>
#define MOD 1000000007
int main() 
{
	int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		vector<int> in(n);
		vector<int> out(n);
		for(int i=0;i<n;i++)
		{cin>>in[i]>>out[i];}
		std::sort(in.begin(),in.end());
		std::sort(out.begin(),out.end());
		long long ans=0,tempans=0;
		int i=0,j=0;
		while(i<n && j<n)
		{
			if(in[i]<out[j])
			{
				tempans++;
				ans=max(ans,tempans);
				i++;
			}
			else
			{tempans--;j++;}
		}
		cout<<ans<<endl;
	}
	// your code goes here
	return 0;
}
