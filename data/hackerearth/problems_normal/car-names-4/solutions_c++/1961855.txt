#include <iostream>
#include <string>
using namespace std;

int main()
{
    int n;
    cin >> n;
    while(n--) {
    	int a[3] = {0};
    	int i = 0; int cnt = 1;
    	string s;
    	cin >> s;
    	int len = s.size();
    	//if(len%3 != 0) 
    	//	cout << "Not OK" << endl;
    	//else
    	//{
    		int j;
    		for(j = 1; j < len; j++) {
    			if(s[j-1] == s[j]) {
    				cnt = cnt + 1;
    			}
    			else
    			{
    				if(i == 3) {
    					//cout << "Not OK" << endl;
    					break;
    				}
    				a[i++] = cnt;
    				cnt = 1;
    			}
    		}
    		if(cnt >= 1)
    			a[i++] = cnt;
    		if(i == 3 && a[0] == a[1] && a[1] == a[2] && j == len && s[0] != s[len-1])
    			cout << "OK" << endl;
    		else
    			cout << "Not OK" << endl;
    	//}
    }
    return 0;
}
