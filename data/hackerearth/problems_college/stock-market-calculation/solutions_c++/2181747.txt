#include <bits/stdc++.h>
using namespace std;
long long int A[50005];
int main()
{
	long long int n,i,val,sum;	int T;
	scanf("%d",&T);
	while(T--)
	{
		sum=val=0;
		scanf("%lld",&n);
		for(i=0;i<n;i++)
		{
			scanf("%lld",&A[i]);
		}
		for(i=n-1;i>=0;i--)
		{
			val=max(val,A[i]);
			sum+=val-A[i];
		}
		printf("%lld\n",sum);
	}
	return 0;
}
