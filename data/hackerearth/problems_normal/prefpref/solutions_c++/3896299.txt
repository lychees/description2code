#include<iostream>
using namespace std;
long a[1000004];
int main(){
string s;
cin>>s;
string t;
cin>>t;

if(s[0]==t[0])a[0]=1;
else a[0]=0;
for(long i=1;i<t.size();i++){
    if(a[i-1]<s.size()){
        if(t[i]==s[a[i-1]]){a[i]=a[i-1]+1;}
        else a[i]=a[i-1];
    }
}
cout<<a[t.size()-1]<<endl;


}
