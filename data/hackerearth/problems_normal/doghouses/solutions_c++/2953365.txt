#include<bits/stdc++.h>
using namespace std;
int const maxn = 1e5+10;
int n, A[maxn], m[maxn];
bool mrk[maxn];
int main(){
  cin >> n;
  for(int i = 1; i <= n; i++){
    cin >> A[i];
    m[A[i]]++;
    mrk[A[i]] = 1;
  }
  int ans = 0;
  for(int i = 0; i <= 1000; i++){
    for(int j = 0; j <= 1000; j++){
      if((!m[i] && !m[j]) || !mrk[i] || !mrk[j]) continue;
      int c = -1, cmp;
      if(i == j) cmp = m[i];
      else cmp = m[i] + m[j];
      for(int k = 1; k <= n; k++){
        if(A[k] == i || A[k] == j) break;
        if(min(i,j) < A[k] && A[k] < max(i,j)){
          cmp++, c = A[k]; break;
        }
      }
      for(int k = n; k >= 1; k--){
        if(A[k] == i || A[k] == j || A[k] == c) break;
        if(min(i,j) < A[k] && A[k] < max(i,j)){
          cmp++; break;
        }
      }
      ans = max( ans, cmp);
    }
  }
  cout << ans;
}
