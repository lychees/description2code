#include <iostream>
#include<vector>
#include<algorithm>
#include<cstdio>
using namespace std;

vector<long int> num(2100);

int main()
{
    int n,i,j;
    long int k,key;
    long long int res=0;
    scanf("%d%ld",&n,&k);
    for(i=0;i<n;i++)
        scanf("%ld",&num[i]);
    if(n < 3)
        cout<<"0"<<endl;
    else
    {
        sort(num.begin(),num.begin()+n);
        for(i=0;i<n-2;i++)
            for(j=i+1;j<n-1;j++)
            {
                key = k/num[i];
                key = key/num[j];
                auto up = upper_bound(num.begin(),num.begin()+n,key);
                if(up - num.begin() > j)
                    res += (up-(num.begin()+j)-1);
            }
        printf("%lld\n",res);
    }
    return 0;
}
