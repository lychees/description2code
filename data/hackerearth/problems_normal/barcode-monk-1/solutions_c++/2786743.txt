#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef pair<int,int> ii;
typedef vector<int> vi;
typedef vector<ii> vii;

const int  MOD=1000000007;
const int  INF= int(1e9);
const int  MAX=1005;
int n,k,d;
bool superbrush[MAX];
int dp[MAX][2];
inline void add(int &a, int b) {
    a += b;
    if (a >= MOD) a -= MOD;
}
int main()
{
	ios_base::sync_with_stdio(false);
    cin>>n>>k;
    superbrush[0]=true;
    for(int i=0;i<k;i++) {
        cin>>d;
        if(d > n) {
            continue;
        }
        for(int j=d;j<=n;j++) {
            superbrush[j] |= superbrush[j-d];
        }
    }
    dp[0][0]=1;
    for(int i=1;i<=n;i++) {
        dp[i][0]=(dp[i-1][1]+dp[i-1][0])%MOD;
        for(int j=1;j<=i;j++) {
            if(superbrush[j]) {
                add(dp[i][1],dp[i-j][0]);
            }
        }
    }
    ll res=(dp[n][0]+dp[n][1])%MOD;
    cout<<res<<"\n";


	return 0;

}
