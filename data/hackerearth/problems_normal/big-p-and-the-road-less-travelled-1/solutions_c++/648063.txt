#include <bits/stdc++.h>
using namespace std;
#define s(n) scanf("%lld",&(n))
#define ll long long
vector<ll>vec[10001];
ll n;
long long dp[100005];
bool vis[100005];
long long solve(int idx)
{
	if ( idx == n ) return 1;
	if ( vis[idx] ) return dp[idx];
	vis[idx] = true;
	long long ans = 0;
	for ( int i = 0; i < vec[idx].size(); i++ ) 
		ans += solve(vec[idx][i]);
	dp[idx] = ans;
	return ans;
}
int main()
{
  
    s(n);
    ll x,y;
    while(1){
    	s(x);s(y);
    	if(x==0&&y==0)
    	break;
    	vec[x].push_back(y);
    }
    cout<<solve(1);
    return 0;
}
