import sys
n=int(raw_input())

while n!=0:
	s=str(raw_input())
	m=str(raw_input())
	k = len(s)
	for i in range (0,k):
		if s[i] is m[i]:
			sys.stdout.write('0'),
		elif s[i] is not m[i]:
			sys.stdout.write('1'),
	print ""
	n=n-1
