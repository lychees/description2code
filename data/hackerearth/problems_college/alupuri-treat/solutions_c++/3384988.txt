#include <iostream>
#include <bits/stdc++.h>
#define ll long long
using namespace std;
int arr[555][555];
int main()
{
    int i,j;
    ll answer;
    int m;
    cin>>m;
    for(i=0;i<m;i++)for(j=0;j<m;j++)cin>>arr[i][j];
    ll p1=0,p2=0;
    for(i=0;i<m;i++)p1+=arr[i][i];
    for(i=0;i<m;i++)p2+=arr[i][m-1-i];
    answer=p1-p2;
    if(answer<0)answer*=-1;
    cout<<answer;
    return 0;
}
