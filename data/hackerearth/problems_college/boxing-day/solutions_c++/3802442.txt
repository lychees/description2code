#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	int count[n];
	for(int i=1;i<=n;i++)
	count[i]=0;
	int i;
	
	int q;
	cin>>q;
	while(q--)
	{
		int a,b;
		cin>>a>>b;
		count[b] =a;
	}
	int c=0;
	for(i=1;i<=n;i++)
	{
		if(count[i]==0)
		{
			c++;
		}
	}
	cout<<c<<"\n";
}
