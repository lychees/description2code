#include <iostream>
#include <string>
using namespace std;
int main()
{
    int n,i,ans=0,ar[109][26]={},j,flag,l,k;
    cin >> n>>k;
    string s;
    for(i=0; i<n; i++)
    {
    cin >> s;
    l=s.size();
    for(j=0; j<l; j++)
        ar[i][s[j]-'a']++;
    }
    for(i=0; i<26; i++)
    {
    flag=0;
    for(j=0; j<n; j++)
        if(ar[j][i]==0)flag=1;
    if(flag==0)ans++;
    }
    if(ans==k) cout<<":-)"<<endl;
    else cout<<":-("<<endl;
    return 0;
}
