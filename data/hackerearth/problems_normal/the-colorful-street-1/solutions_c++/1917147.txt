#include<bits/stdc++.h>
using namespace std;
int main()
{
   int ts;
   cin>>ts;
   while(ts -- )
   {
   	int n;
       cin>>n;
       int dp[n][3];
       int pink[n];
       int yellow[n];
       int orange[n];
       for(int i=0;i<n;i++)
       {
         cin>>pink[i]>>yellow[i]>>orange[i];
       }
       dp[0][0]=pink[0];
       dp[0][1]=yellow[0];
       dp[0][2]=orange[0];
       for(int i=1;i<n;i++)
       {
         dp[i][0]=min(dp[i-1][1],dp[i-1][2])+pink[i];
         dp[i][1]=min(dp[i-1][0],dp[i-1][2])+yellow[i];
         dp[i][2]=min(dp[i-1][0],dp[i-1][1])+orange[i];
       }
  //     int ans=min(dp[n-1][0],dp[n-1][1],dp[n-1][2]);
    //   cout<<ans<<endl;
    printf("%d\n",min(dp[n-1][0],min(dp[n-1][1],dp[n-1][2])));
   }
   
}
