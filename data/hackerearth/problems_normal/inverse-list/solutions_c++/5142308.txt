#include<iostream>
using namespace std;
int main()
{
int tc;
cin>>tc;
 
while(tc>0)
{
int a[1001];
int n;
cin>>n;
int i;
for( i=0;i<n;i++)
cin>>a[i];
int j=0,b[1000];
for(i=0;i<n;i++)
{
 j=a[i];
 
 b[j]=i+1;
}
int flag=0;
for(i=0;i<n;i++)
{
  if(a[i]==b[i+1])
 flag=1;
  else
 {
  flag=2;
  break;
 }
}
 
if(flag==2)
cout<<"not inverse"<<endl;
else if(flag==1)

}

}
