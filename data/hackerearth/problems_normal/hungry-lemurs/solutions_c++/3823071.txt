using namespace std;

#define ll long long

vector<ll> factors(ll n) {
	vector<ll> fact;
	for (ll i = 2; i < sqrt(n) + 1; i++) {
		if (!(n % i)) {
			fact.push_back(i);
			if (n != i*i) {
				fact.push_back(n / i);
			}
		}
	}
	fact.push_back(n);
	return fact;
}

int main()
{
	ll N, K; cin >> N >> K;
	ll m = 2 * N * K;
	ll new_n = -1;
	ll new_k = -1;

	if (N < K) {
		cout << min(N, K - N) << endl;
	}
	else if (!(N % K)) {
		cout << 0 << endl;
	}
	else {
		for (ll i = 1; i < 2 * N + 1; i++) {
			vector<ll> f = factors(i);
			for (ll k : f) {
				ll new_m = abs(N - i) + abs(K - k);
				if (new_m < m) {
					m = new_m;
					new_n = i;
					new_k = k;
				}
			}
		}
		cout << m << endl;
	}
	return 0;
}
