#include<iostream>
using namespace std;
int main()
{
	long int a[10000],i,j,n;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
	}
	for(j=0;j<n;j++)
	{
		for(int k=j+1;k<n;k++)
		{
			if(a[k]<a[j])
			{
				swap(a[k],a[j]);
			}
		}
	}
	for(i=0;i<n;i++)
	{
		cout<<a[i]<<'\n';
	}
return 0;
}
