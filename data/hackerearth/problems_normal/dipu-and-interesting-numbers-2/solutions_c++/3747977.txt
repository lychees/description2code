#include <iostream>
#include <cmath>
using namespace std;

int bin_search(long long n)
{
	long long low=1, high=1e9+1, mid, m1, m2;
	while(low<=high)
	{
		mid = (low + high)/2LL;
		m1 = mid*mid;
		m2 = (mid+1LL)*(mid+1LL);
		if(m1<=n)
		{
			if(m2>n)return (int)mid;
			else low = mid+1LL;
		}else 
			high = mid - 1LL;
	}
	return 0;
}

int main()
{
    int t;
    long long l, r;
    cin>>t;
    while(t--)
    {
    	cin>>l>>r;
    	printf("%d\n", (bin_search(r)-bin_search(l-1)));
    }
    return 0;
}
