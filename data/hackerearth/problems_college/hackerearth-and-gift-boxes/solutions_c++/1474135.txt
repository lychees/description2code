#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
	int t,n,m,a;
	cin>>t;
	while(t--)
	{
		cin>>n>>m>>a;
		long long int q=a*a;
		long long int w=q*a;
		long long int v=n+m*w;
		long long int ans=200000000000;
		for(long long int i=a;i*i*i<=ans;++i)
		{
			long long int r=i/a;
			for(long long int j=i;j*i*j<=ans;++j)
			{
				long long int p=j/a;
				long long int k=(m+r*p-1)/(r*p)*a;
				long long int v1=i*j*k;
				long long int v2=max(0ll,v-v1);
				k+=(v2+i*j-1)/(i*j);
				ans=min(ans,i*j*k);
			}
		}
		cout<<ans<<endl;
	}
    return 0;
}
