#include<iostream>
#include<algorithm>
#include<climits>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long n;
        cin>>n;
        long a[2*n],max;
        for(long i=0;i<(2*n);i++)
            cin>>a[i];
        sort(a,a+(2*n));
        max=INT_MIN;
        long i=0,j=2*n-1;
        while(i<j)
        {
            if(a[i]+a[j]>max)
                max=a[i]+a[j];
            i++;
            j--;
        }
        cout<<max<<endl;
    }
}
