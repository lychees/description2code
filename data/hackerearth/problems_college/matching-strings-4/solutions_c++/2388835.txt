#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

int main()
{
    int testCases;
    cin>>testCases;
    for(int c=0;c<testCases;c++)
    {
    	string a, b;
    	cin>>a;
    	cin>>b;
		int size = (int) min(a.length(), b.length());
		int count = 0;
		for(int i=0;i<size;i++)
			if(a[i] == b[i])
				count++;
		cout<<count<<"\n";		
    }
    return 0;
}
