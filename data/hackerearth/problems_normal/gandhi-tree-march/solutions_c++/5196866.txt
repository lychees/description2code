#include <iostream>
#include <algorithm>
#include <vector>
#include <stack>

#define MAX 20005
using namespace std;

int main()
{
    int t, c, col;
    string s;
    vector<char> vt[MAX];
    stack<int> prev;
    
    cin >> t;
    while (t--)
    {
    	cin >> c;
    	cin >> s;
    	int len = s.length();
    	int range = len * 2;
    	for (int i = 0; i <= range; ++i)
    		vt[i].clear();
    	
    	col = len;
    	vt[col].push_back(s[0]);
    	prev.push(col);
    	for (int i = 1; i < len; ++i)
    	{
    		if (s[i] == '(')
    			col--;
    		else if (s[i] == ')')
    		{
    			col = prev.top();
    			prev.pop();
    		}
    		else
    		{
    			if (s[i - 1] == '(')
    			{
    				if (s[i] != '.')
    				{
    					vt[col].push_back(s[i]);
    					prev.push(col);
    				}
    			}
    			else
    			{
    				col = prev.top() + 1;
    				if (s[i] != '.')
    				{
    					vt[col].push_back(s[i]);
    					prev.push(col);
    				}
    			}
    		}
    	}
    	
    	c += len;
    	if (vt[c].size())
    	{
			sort(vt[c].begin(), vt[c].end());
	    	for (int i = 0; i < vt[c].size(); ++i)
	    		cout << vt[c][i];
	    	cout << endl;
    	}
    	else
    		cout << "Common Gandhijee!" << endl;
    }
    
    return 0;
}
