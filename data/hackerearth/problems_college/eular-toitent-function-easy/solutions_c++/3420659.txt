#include<bits/stdc++.h>
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,r,ans,i;
        scanf("%d",&n);
        r=n;
        ans=n;
        for(i=2;i*i<=r;i++)
        {
            if(n%i ==0)
            {
                ans -= ans/i;
            }
            while(n%i==0)
            {
                n=n/i;
            }
        }
        if(n>1)
        {
            ans -= ans/n;
        }
        printf("%d\n",ans);
    }
    return 0;
}
