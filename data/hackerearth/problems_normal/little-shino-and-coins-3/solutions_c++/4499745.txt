#include <iostream>
#include <algorithm>
#include <vector>
#include <cstring>
using namespace std;
int letter[5001][26];
int dp[5001][5001];
int main()
{
    int K;
    string s;
    cin>>K>>s;
    int n = s.length();
    int cnt = 0;
    letter[0][s[0]-'a'] = 1;
    for(int i=1;i<n;i++){
    	for(int j=0;j<26;j++){
    		letter[i][j] = letter[i-1][j];
    	}
    	letter[i][s[i]-'a']++;
    }
    int m=0;
    for(int i=0;i<26;i++) m+=(letter[K-1][i]>0);
    dp[0][K-1] = (m==K);
    for(int i=K;i<n;i++){
    	m = 0;
    	for(int j=0;j<26;j++) m+=(letter[i][j]>0);
    	dp[0][i] = (m==K);
    }
    for(int i=1;i<n;i++){
    	for(int j=i+K-1;j<n;j++){
    		m = 0;
    		for(int k=0;k<26;k++){
    			m += ((letter[j][k]-letter[i-1][k])>0);
    		}
    		dp[i][j] = (m==K);
    	}
    }
    for(int i=0;i<n;i++)
    for(int j=i+K-1;j<n;j++) cnt+= (dp[i][j]==1);
    cout << cnt;
    return 0;
}
