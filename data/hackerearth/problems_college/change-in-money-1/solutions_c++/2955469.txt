#include<bits/stdc++.h>
using namespace std;
int main(){
	int n,m,coin[52];
	scanf("%d %d",&n,&m);
	long dp[m][n+1];
	for(int i=0;i<m;i++){
		scanf("%d",&coin[i]);
	}
	for(int i=0;i<m;i++){
		dp[i][0]=1;
	}
	for(int i=0;i<m;i++){
		for(int j=1;j<=n;j++){
			long x=(i>0)?dp[i-1][j]:0;
			long y=( (j-coin[i])>=0)?dp[i][j-coin[i]]:0;
			dp[i][j]=x+y;
		}
	}
	printf("%ld\n",dp[m-1][n]);
	
	return 0;
}
