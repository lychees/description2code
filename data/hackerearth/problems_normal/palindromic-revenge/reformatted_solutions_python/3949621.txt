n = int(raw_input())

while (n > 0):
    s = raw_input()
    s = list(s)
    c = 0
    for i in range(len(s)/2):
        if s[i] == s[len(s)-1-i]:
            continue
        else:
            c = c + abs(ord(s[i])-ord(s[len(s)-1-i]))
    print c
