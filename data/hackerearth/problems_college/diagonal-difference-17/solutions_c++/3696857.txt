#include<bits/stdc++.h>
using namespace std;

int main()
{
	long long n,i,j,sum1=0,sum2=0;
	cin>>n;
    long long int a[n][n];
    for(i=0;i<n;i++)
     {
     	for(j=0;j<n;j++)
     	  cin>>a[i][j];
     }
     for(i=0;i<n;i++)
       {
       	sum1=sum1+a[i][i];
       }
     for(i=0;i<n;i++)
     {
     	sum2=sum2+a[i][(n-1)-i];
     	
     }
     cout<<abs(sum1-sum2)<<endl;
     
    return 0;
}
