#include <stdio.h>
#include <iostream>
#include <string>

using namespace std;

int main()
{
  int t;
  string p, q;

  cin >> t;
  getline(cin, p);
  while(t--)
  {
    getline(cin, p);
    getline(cin, q);

    int position = q.find(p);
    if(position >= 0 && position < q.length()) cout << "possible" << endl;
    else cout << "not possible" << endl;
  }

  return 0;
}
