#include <bits/stdc++.h>
using namespace std;
#define int long long
#define PII pair <int, int> 
vector <PII> d;
main()
{
	int a; cin >> a;
	for (int g=0; g<a; g++)
	{ 	
		d.clear(); 
		vector <int> e, f; 
		int b; cin >> b;
		for (int y=0; y<b; y++)
		{
			int r; cin >> r; e.push_back(r); 
		}
		for (int y=0;y<b; y++)
		{
			int r; cin >> r; f.push_back(r); 
		}
		for (int g=0; g<e.size(); g++) d.push_back(PII(e[g], f[g])); 
		sort(d.begin(), d.end()); 
		int cur=1; 
		int ans=0; 
		for (int g=1; g<d.size(); g++)
		{
			if ((int)d.size()-g>=cur)
			{
				ans+=cur*(d[g].first-d[g-1].first-d[g-1].second); 
			}
			else
			{
				ans+=((int)d.size()-g)*(d[g].first-d[g-1].first-d[g-1].second); 
			}
			cur++; 
		}
		cout << ans << '\n'; 
	}	
	return 0; 
}
