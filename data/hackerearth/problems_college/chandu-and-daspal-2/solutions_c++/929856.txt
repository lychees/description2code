#include <bits/stdc++.h>

using namespace std;

int gcdr ( int a, int b )
{
  if ( a==0 ) return b;
  return gcdr ( b%a, a );
}

int main()
{
	int t;
	scanf(" %d",&t);
	while(t--)
	{
		int a,b;
		char s[10];
		scanf("%d %d %s",&a,&b,s);
		int no = 0; //odd -> chandu, even -> daspal
		while(1)
		{
			if(a!=0 && b!=0)
			{
				int gcd = gcdr(a,b);
				a -= gcd;
				b -= gcd;
				no++;
			}
			else
				break;
		}
		if(s[0] == 'C')
		{
			if(no%2 == 1)
				printf("Chandu\n");
			else
				printf("Daspal\n");
		}
		else
		{
			if(no%2 == 1)
				printf("Daspal\n");
			else
				printf("Chandu\n");
		}
	}
	return 0;
}
