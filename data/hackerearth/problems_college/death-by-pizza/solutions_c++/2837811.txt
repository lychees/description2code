#include <bits/stdc++.h>
using namespace std;

int main()
{
	int st[5]={16,9,26,26,1};
    int n,q;cin>>n>>q;
    int arr[n];
    for(int i=0;i<n;i++) cin>>arr[i];
    while(q--){
    	//for(int i=0;i<n;i++) cout<<arr[i]<<" ";
    //	cout<<endl;
    	int type,l,r;cin>>type>>l>>r;
    	if(type==1){
    		long long int sum=arr[l-1];
    		for(long int i=l;i<r;i++) sum+=arr[i];
    		cout<<sum<<"/"<<(r-l+1)<<endl;
    	}
    	else{
    		int i=0;
    		for(long int j=l-1;j<r;j++){
    			arr[j]+=st[i];
    			i++;
    			if(i==5) i=0;
    		}
    	}
    }
    return 0;
}
