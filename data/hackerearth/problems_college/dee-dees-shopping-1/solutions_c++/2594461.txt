int min(int a,int b)
{
	return a<b?a:b;
}

int deedee(int arr[],int n,int k)
{
	if(k==0) return 0;
	if(n<0 && k>0) return 10000;
	if(arr[n]>k) return deedee(arr,n-1,k);
	return min(deedee(arr,n-1,k),1+deedee(arr,n,k-arr[n]));
}

main()
{
	int arr[2]={6,8};
	int t; cin>>t;
	while(t--)
	{
		int n; cin>>n;
		int h=deedee(arr,1,n);
		if(h==10000) cout<<-1<<endl;
		else cout<<h<<endl;
	}
}
