#include <bits/stdc++.h>
using namespace std;
int min_steps[100005]={-1,-1,1,1,2,1,2,1,2,2,2};
int main()
{
    
    for(int i=11;i<100005;i++)
    min_steps[i] = 1 + min(min(min_steps[i-2],min_steps[i-3]),min(min_steps[i-5],min_steps[i-7]));
    int t; cin >> t;
    while(t--)
    {
    	int n; cin >> n;
    	cout << min_steps[n] << endl;
    }
    return 0;
}
