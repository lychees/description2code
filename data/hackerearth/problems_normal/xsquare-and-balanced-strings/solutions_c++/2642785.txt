#include <iostream>
#include <map>
using namespace std;

int main()
{
    int t,f,i;
    cin>>t;
    while(t--){
    	string s;
    	map <char,int> m;
    	cin>>s;
    	for(i=0;i<s.size();i++){
    		m[s[i]]++;
    	}
    	f=1;
    	map <char,int> :: iterator it;
    	for(it=m.begin();it!=m.end();it++){
    		if(it->second%2==1){
    			f=-1;
    			break;
    		}
    	}
    	cout<<f<<endl;
    }
    return 0;
}
