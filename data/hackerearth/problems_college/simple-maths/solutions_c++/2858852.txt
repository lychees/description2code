#include <bits/stdc++.h>

using namespace std;

#define icin(x) scanf("%d",&x)
#define lcin(x) scanf("%lld",&x)

typedef long long LL;
typedef vector<int> vi;
typedef vector<vi> vvi; 


LL gcd(LL a,LL b){
  LL tmp;
  while(b){
    tmp=a%b;a=b;b=tmp;
  }
  return a;
}

int main(){
  int t;
  icin(t);
  while(t--){
    LL n,a,b;
    lcin(n);
    lcin(a);
    lcin(b);
    if(a==0 || b==0){
      cout<<"-1"<<endl;
    }
    else{
      LL g=gcd(a,b);
      LL lcm=a*(b/g);
      cout<<((n/a)+(n/b)-(n/lcm))<<endl;
    }
  }
  return 0;
}
