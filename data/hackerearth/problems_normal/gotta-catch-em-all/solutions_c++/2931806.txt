#include <iostream>
#include<vector>
#include <algorithm>
using namespace std;

int main()
{
	int t;
	cin>>t;
	vector<int>pok;
	vector<int>tm;
	int l=t;
	while(t--){
		int day;
		cin>>day;
		pok.push_back(day);
		
	}
	sort(pok.begin(),pok.end());
	int mx=-1;
	int curday=1;
	for(int i=l-1;i>=0;--i){
		int takes=curday+pok[i];
		if(takes>mx)mx=takes;
		curday++;
	}
	
    cout << mx+1 << "\n";
    return 0;
}
