#include<bits/stdc++.h>
using namespace std;
struct abc
{
   long int a;
   long int b;
};
int main()
{
    int n;
    cin>>n;
    struct abc s[1003];
    struct abc s1[1003];
    for(int i=0;i<n;i++)
    {
        cin>>s[i].a>>s[i].b;
    }
    for(int i=0;i<n;i++)
    {
        s1[i].a=s[i].a+s[i].b;
        s1[i].b=i;
    }
 
 
for(int j=0;j<n-1;j++)
{
    for(int i=0;i<n-1-j;i++)
    {
        if(s1[i].a>s1[i+1].a)
        {
            long int k=s1[i].a;
            s1[i].a=s1[i+1].a;
            s1[i+1].a=k;
 
           long int m=s1[i].b;
            s1[i].b=s1[i+1].b;
            s1[i+1].b=m;
 
        }
 
    }
}
 
for(int i=0;i<n;i++)cout<<s1[i].b+1<<" ";
 
}
