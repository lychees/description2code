#include<bits/stdc++.h>

using namespace std;

#define all(v) v.begin() , v.end()
using ll = long long;

int cnt[1 << 11];
int parity[11];
char s[(int) 1e5 + 5];
int main() {
	//ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
	int n;
	scanf("%d", &n);
	scanf("%s", s);
	++cnt[0];
	ll res = 0;
	for (int i = 0; i < n; ++i) {
		parity[s[i] - '0'] ^= 1;
		int mask = 0;
		for (int j = 0; j < 10; ++j)
			if (parity[j])
				mask |= (1 << j);
		res += cnt[mask];
		for (int j = 0; j < 10; ++j)
			res += cnt[mask ^ (1 << j)];
		++cnt[mask];
	}
	cout << res;
	return 0;

}
