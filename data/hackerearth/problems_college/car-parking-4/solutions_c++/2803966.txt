#include <bits/stdc++.h>
using namespace std;


int main()
{

	int t;
	cin>>t;
	while(t--)
	{
       string str;
		cin>>str;
		int len = str.size();int count = 0;
		for(int i=0; i<len; i++)
		{

			if(str[i]=='-')
			{

				int flag = 1;
				if(i<len-1 && str[i+1]=='B')flag=0;
				if(i<len-2 && str[i+2]=='B')flag=0;
				if(i>0 && str[i-1]=='S')flag=0;
				if(i<len-1 && str[i+1]=='S')flag=0;
				if(flag)count++;
			}
		}
		cout<<count<<endl;
	}
return 0;
}
