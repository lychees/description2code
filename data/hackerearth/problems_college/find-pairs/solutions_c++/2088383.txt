#include<iostream>
#include<stdlib.h>
#include<string.h>
using namespace std;
int main(){
	long long int t,i,num;
	cin>>t;
	while(t--){

	cin>>num;
	unsigned long long int a[num],c=0,i,j;
	for(i=0;i<num;i++){
		cin>>a[i];
	}
	for(i=0;i<num;i++){
		for(j=i+1;j<num;j++){
			if(a[i]==a[j]){
				c++;
			}
		}
	}
	cout<<num+c<<"\n";
}
return 0;
}
