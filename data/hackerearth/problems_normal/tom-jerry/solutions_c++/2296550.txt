#include<iostream>
using namespace std;
int M[101][101],v[101][101];
int count100(int i,int j,int n)
{int cnt=0;
    if(i<0||i>=n)
        return 0;
    if(j<0||j>=n)
        return 0;
    if(i==n-1&&j==n-1)
    {
        return 1;
    }
    if(v[i][j]==1||M[i][j])
        return 0;
        v[i][j]=1;
     cnt=count100(i+1,j,n)+count100(i-1,j,n)+count100(i,j+1,n)+count100(i,j-1,n);
v[i][j]=0;
return cnt;

}

int main()
{
int n;
cin>>n;
 for(int i=0;i<n;i++)
    for(int j=0;j<n;j++)
      cin>>M[i][j];

cout<<count100(0,0,n);
}
