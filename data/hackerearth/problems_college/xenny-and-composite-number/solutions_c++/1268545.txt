#include <bits/stdc++.h>
using namespace std;

bool izprm(int n){
	int i,x=sqrt(n);
	for(i=2;i<=x;i++)
		if(n%i==0)
			return false;
	return true;
}

int main()
{
    int t,n;
    scanf("%d",&t);
    while(t--){
    	scanf("%d",&n);
    	if(n<=3){
    		printf("4\n");
    	}
    	else{
    		if(n&1)
    			printf("%d\n",n+1);
    		else{
    				if(izprm(n+1))	printf("%d\n",n+2);
    				else	printf("%d\n",n+1);
    		}
    	}
    }
    return 0;
}
