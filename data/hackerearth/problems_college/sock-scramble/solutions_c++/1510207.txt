#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
		int sum=0,n,t,val;
		scanf("%d",&t);
		while(t--)
		{
			scanf("%d",&n);
			for(int i=0;i<n;i++)
			{
				scanf("%d",&val);
				if(i!=0)
				{
					sum=sum^val;
				}
				else 
				 sum=val;
			}
			printf("%d\n",sum);
		}
		return 0;
}
