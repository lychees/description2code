#include<bits/stdc++.h>
using namespace std;
 
int main()
{
	int t;
	cin>>t;
	
	while(t--)
	{
		int l;
		cin>>l;
		string str,ans,r,ans1 ;
		
		cin>>str ;
		char ch;
		ans = ans1=str ;
		for(int i=0;i<l;i++)
		{
			r=str;ch = str[i] ;
			
			for(int j=i+1;j<l;j++)
			{
				r[j-1]=r[j] ;
				r[j]=ch;
				if(r<ans)
				ans=r ;
			}
			r=str;
			for(int k=i-1;k>=0;k--)
			{
				r[k+1]=r[k] ;
				r[k]=ch;
				if(r<ans1)
				ans1=r ;
				
			}
 			
		}
		
		if(ans<ans1)	cout<<ans<<"\n";
		else
		cout<<ans1<<"\n" ;
		
		
	}
	
	
	
	return 0;
}
