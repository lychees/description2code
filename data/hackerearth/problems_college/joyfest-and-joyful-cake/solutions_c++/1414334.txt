LL dp[2000010];

int main()
{

    int i,j,k,T;
    LL val;

    dp[0]=1;


    for(i=1;i<=2000010;i++)
    {
      dp[i]=dp[i-1]+i;
    }

    scanf("%d",&T);

    while(T--)
    {
      scanf("%lld",&val);
      int lo=0,hi=2000010,mid;

      }

    }
