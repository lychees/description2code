#include <iostream>
using namespace std;
long long arr[10000],dp[10000];

int main()
{
    int N;
    cin>>N;
    for(int i=0;i<N;i++)
    {
        cin>>arr[i];
        dp[i]=1;
    }

    int s1,s2;
    int m=1;

    for(int i=1;i<N;i++)
    {
        s1=arr[i]>0?1:-1;

        for(int j=i-1;j>=0;j--)
        {
            s2=arr[j]>0?1:-1;
            if(s1!=s2 && (abs(arr[i])>abs(arr[j])))
                dp[i]=max(dp[i],1+dp[j]);

        }
        if(dp[i]>m)
            m=dp[i];
    }
    cout<<m;
    return 0;
}
