#include <iostream>
#include<stdio.h>
#include<string.h>
using namespace std;


char c1[1000001],c2[1000001];

int main()
{
	int t,j;
	cin>>t;
	
	
	while(t--)
	{
		scanf("%s",c1);
		scanf("%s",c2);
		int flag = 0;
		int l1 = strlen(c1),l2 = strlen(c2);
		if(l1!=l2)
		{	cout<<"No\n";
			break;
		}
		
		else
		{	j= 0;
			for(int i = 0 ; i<l1 ; i++)
			{
				if(c1[i] == '#')
					continue;
				else if(c2[j] == '#')
					 {   j++;i--;
					 }
					 else
					 	if (c1[i] == c2[j] && c1[i] == 'A')
					 	{
					 		if(j<=i)
					 			{
					 				j++;
					 				continue;
					 			}
					 		else
					 		{	cout<<"No\n";
					 			flag = 1;
					 			break;
					 		}
					 	}
					 	else
					 	if (c1[i] == c2[j] && c1[i] == 'B')
					 	{
					 		if(j>=i)
					 			{
					 				j++;
					 				continue;
					 			}
					 		else
					 		{	cout<<"No\n";
					 			flag = 1;
					 			break;
					 		}
					 	}
					 	else
					 	{	cout<<"No\n";
					 		flag = 1;
							break;
					 	}
			}
		}
		
		if(!flag )
			cout<<"Yes\n";
	//	else cout<<"No\n";
		
	}


	return 0;
}
