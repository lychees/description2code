#include <bits/stdc++.h>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    int q;
    long long freq, k, ct[27] = {0};
    char letter;
    string line;
    cin >> line;
    istringstream input(line);
    while (!input.eof()) {
    	input >> letter >> freq;
    	ct[letter - 'a'] += freq;
    }
    for (int i = 1; i < 26; ++i)
    	ct[i] += ct[i - 1];
    cin >> q;
    while (q--) {
    	cin >> k;
    	auto idx = lower_bound(ct, ct + 26, k) - ct;
    	if (idx > 25)
    		cout << -1 << endl;
    	else
    		cout << (char)('a' + (int)idx) << endl;
    }
    return 0;
}
