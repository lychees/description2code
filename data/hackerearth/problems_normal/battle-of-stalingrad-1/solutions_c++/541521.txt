#include <iostream>
using namespace std;

double area(int x1,int x2,int x3,int y1,int y2,int y3)
{
	double res;
	res = ((x1 *(y2-y3)) +(x2*(y3-y1))+(x3*(y1-y2)));
	if(res <0)
		res = res * -1;
	return res;
}

int main() {
	int t;
	cin >> t;
	while(t--)
	{
		int x1,x2,x3,x4,y1,y2,y3,y4;
		cin >> x1>>y1>>x2>>y2>>x3>>y3>>x4>>y4;
		double a1,a2,a3,a4;
		a1 = area(x1,x2,x3,y1,y2,y3);
		a2 = area(x1,x2,x4,y1,y2,y4);
		a3 = area(x1,x4,x3,y1,y4,y3);
		a4 = area(x4,x2,x3,y4,y2,y3);
		
		if(a1 == (a2+a3+a4))
			cout <<"INSIDE"<<endl;
		else
			cout << "OUTSIDE"<<endl;
	}
	return 0;
}
