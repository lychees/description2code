#include<bits/stdc++.h>
using namespace std;
typedef long long int lli;
const lli MOD=1000*1000*1000LL+7LL;
lli mpow(lli a,lli b){
    lli ret=1;
    a%=MOD;
    while(b>0){
        if(b&1){
            ret=((ret%MOD)*(a%MOD))%MOD;
        }
        b>>=1;
        a=((a%MOD)*(a%MOD))%MOD;
    }
    return ret%MOD;
}
lli  arr[20];
int main(){
    int tc;cin>>tc;
    for(int t=1;t<=tc;++t){
        int n;cin>>n;
        for(int i=0;i<n;++i)
            cin>>arr[i];
        lli  ans=-23;
        for(int i=0;i<(1<<n);++i){
            int j=i;
            int pos=0;
            lli num=1;
            lli den=0;
            while(j>0 ){
                if((j&1)){
                    num=num*(arr[pos]%MOD);
                      num%=MOD;
                    den=den+arr[pos];
                    // den%=MOD;
                }
                j>>=1;
                pos++;
            }
            ans=max(ans,((num%MOD)*(mpow(den,MOD-2)%MOD))%MOD);
        }
        printf("Case #%d: %lld\n",t,ans);
    }
    return 0;
}
