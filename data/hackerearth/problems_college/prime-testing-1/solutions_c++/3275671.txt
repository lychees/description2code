#include <stdio.h>

int main()
{
    long int t,n,i,a[100000],count,flag,j;
    scanf("%ld",&t);
    while(t--)
    {
    	flag=0;
        count=0;
    	scanf("%ld",&n);
    	for(i=0;i<n;i++)
    	{
    		scanf("%ld",&a[i]);
    		if(a[i]==0||a[i]==1)
    		continue;
    		for(j=2;j<a[i];j++)
    		{
    			if(a[i]%j==0)
    			{
    				flag=1;
    				break;
    			}
    		}
    		if(flag==0)
    		count++;
    		else
    		flag=0;
    	}
    	printf("%ld\n",count);
    }
    return 0;
}
