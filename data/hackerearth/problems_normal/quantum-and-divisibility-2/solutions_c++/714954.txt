#include <cstdio>
#include<string.h>
#include<map>
#define LL long long
using namespace std;
map<int,LL> factors(int n,int m)
{
     int i;
     LL ctr=0;
     map<int,LL> k;
     while(n%2==0)
     {
     	ctr++;
     	n/=2;
     }
     k[2]=ctr*m;
    for (i=3; (i*i)<=n; i=i+2)
    {
    	ctr=0;
        while (n%i == 0)
        {
            ctr++;
            n = n/i;
        }
        k[i]=ctr*m;
    }
    if(n>1)
    k[n]=m;
    return k;
}
int main()
{
    int t,flg,i,a,b,c,d;
    scanf("%d",&t);
    while(t--)
    {
    	flg=0;
    	scanf("%d%d%d%d",&a,&b,&c,&d);
    	map<int,LL> x=factors(a,b);
    	map<int,LL> y=factors(c,d);
    	map<int,LL>::iterator it;
    	for(it=y.begin();it!=y.end();it++){
    	if(it->second>x[it->first])
    	{flg=1;break;}
    	}
    	if(flg==0 || d==0 )
    	printf("Divisible\n");
    	else printf("Not divisible\n");
    }
}
