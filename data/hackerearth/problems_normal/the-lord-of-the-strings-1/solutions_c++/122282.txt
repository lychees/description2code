#include <iostream>
using namespace std;

int main()
{
    int test,i,count[256];
    char str[100];
    
    cin>>test;
    while(test--)
    {
    	for(i=0;i<256;i++)
    		count[i]=0;
    		
    	cin>>str;
    	for(i=0;str[i];i++)
    		count[str[i]]++;
    		
    	for(i=0;str[i];i++)
    	{
    		if(count[str[i]])
    		{
    			cout<<str[i];
    			count[str[i]]=0;
    		}
    	}
    	
    	cout<<endl;
    	
    }
    return 0;
}
