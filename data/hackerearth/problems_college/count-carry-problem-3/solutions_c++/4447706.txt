#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    long int t,a,b;
    int c=0;
    int flag = 0;
    cin>>t;
    while(t--){
    	c=0;
    	flag = 0;
    	cin>>a>>b;
    	while(a!=0){
    		if((a%10+b%10+flag) > 9) {
    			c =c+1;
    		    flag=1;
    		}else {
    			flag=0;
    		}
    		b=b/10;
    		a=a/10;
    	//	printf(" \tc= %d ",c);

    	}
    	if(c>1)  cout<<c<<" carry operations\n";
    	else if (c==1)  cout <<"1 carry operation\n";
    	else cout <<"No carry operation\n";
    }
    
    
    return 0;
}
