#include <bits/stdc++.h>

using namespace std;

int main()
{
    string a;
    int sum = 0,max1 = -1;
    int prev_sum = -1;
	string ans;
    while(cin >> a)
    {
        for (int i = 0;i < a.size();i++)
        {
            sum += (a[i]-'0');
        }

        //max1 = max(prev_sum,sum);
        
        if (prev_sum <= sum)
        {
        	ans = a;
        	max1 = sum;
        } else 
        {
        	max1 = prev_sum;
        }
        
        prev_sum = max1;
        sum = 0;
    }

    cout << ans << endl;

    return 0;

}
