def mysieve():
    temp = list()
    temp.append(2)
    for i in xrange(3, 1000001, 2):
        if prime[i]:
            temp.append(i)
        if i*i <= 1000001:
            for j in xrange(i*i, 1000001, i):
                prime[j] = 0
    return temp


prime = [1]*1000001
final = mysieve()


for _ in xrange(input()):
    a, b = map(int, raw_input().split())
    ans = 0
    for val in final:
        if val >= a and val <= b:
            ans += val
    print ans
