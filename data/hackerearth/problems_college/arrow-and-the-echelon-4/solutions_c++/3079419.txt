    #include<iostream>
    using namespace std;
     
    int main()
    	{
    		bool b;
    		int t,n,x;
    		cin>>t;
    		while ( t-- )
    			{
    				x = 0;
    				cin>>n;
    				b = false;
    				while( n - 5*x >= 0)
    					{
    						if ( ( n - 5 * x ) % 3 == 0 )
    						{
    							b = true;
    							break;
    						}
    						x++;
    					}
    				if ( b )
    					{
    						for (int  i = 1 ; i <= n - 5 * x ; i++ )
    							cout<<5;
    						for (  int i = 1 ; i <= 5*x ; i++ )
    							cout<<3;
    					}
    				else
    					cout<<-1;
    				cout<<endl;
    				
    				
    			}
    	}
