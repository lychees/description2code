#include<iostream>
using namespace std;

typedef long long int uli;
using namespace std;
const uli mod=1e9+7;
const uli iv6=166666668;
const uli iv2=500000004;
uli n,m;
inline uli ndiag(uli i,uli j){ 
   return n+j-i;
}
int main(){
   uli p,q;
   cin>>n>>m>>p>>q;
   uli ans=0,inc;
   for(int i=0;i<2;i++){   
      uli w=p+1,h=q+1;
      if(w<=n && h<=m){
         inc=(n-w+1)*(m-h+1)%mod;
         inc=(inc<<2)%mod;
         ans=(ans+inc)%mod;
      }
      swap(p,q);
   }
   uli d1=ndiag(0,0);
   uli d2=ndiag(n-1,m-1);
   if(d1>d2)swap(d1,d2);

   //1(0)+2*(1)+3*(2)+...+(d1-1)*(d1-2)=sum(i*i-i)
   uli sq=(d1-1)*d1%mod*(d1+d1-1)%mod*iv6%mod;
   uli ln=d1*(d1-1)%mod*iv2%mod;
   uli sm=(sq-ln+mod)%mod;
   sm=(sm<<1)%mod;
   if(d1<=d2) sm+=d1*(d1-1)%mod*(d2-d1+1)%mod;
   sm%=mod;
   ans=(ans+sm+sm)%mod;

   uli tot=n*m%mod;
   tot=tot*(tot-1)%mod;
   ans=(tot-ans+mod)%mod;
   cout<<ans<<endl;
   return 0;
}
