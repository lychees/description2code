#include <iostream>
using namespace std;
int main()
{
    long int n,m,t,cnt=0,i=-1;
    int a[100];
    cin>>t;
    while(t-- >0){
    	cin>>n>>m;
    	while(m!=0){
    		if(m%2==1)
    			a[++i]=cnt;
    		cnt++;
    		m>>=1;
    	}
    	for(;i>=1;i--)
    		cout<<"("<<n<<"<<"<<a[i]<<") + ";
    	cout<<"("<<n<<"<<"<<a[i]<<")"<<endl;
    	i=-1;
    	cnt=0;
    }
    return 0;
}
