#include<iostream>
using namespace std;
int main()
{  
	int n;
	cin>>n;
	while(n--)
	{int flag=0;
		string str;
		cin>>str;
		int cnt[26]={0};
		for(int i=0;str[i];i++)
		{
			cnt[str[i]-'a']++;
			if(cnt[str[i]-'a']>1)
			{
				flag=1;
				break;
			}
		}
		if(flag)
		cout<<"Yes"<<"\n";
		else
		cout<<"No"<<"\n";
	}
}
