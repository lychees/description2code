#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n;
    cin >> n;
    int a[n];

    for (int i = 0;i <n;i++)
        cin >> a[i];

    sort(a,a+n);
    int k;
    cin >> k;
    int count = 0,sum = 0;

    for (int i = n-1;i >= 0;i--)
    {
        if (sum < k)
        {
            sum = sum + a[i];
            count++;
        } else
            break;
    }

    cout << count << endl;

    return 0;
}
