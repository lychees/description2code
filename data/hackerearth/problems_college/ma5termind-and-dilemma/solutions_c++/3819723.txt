#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t,n,i,len,z,o,temp1,temp2,ans;
	cin>>t;
	char str[100005];
	char a[100005],b[100005];
	while(t--)
	{
		cin>>n;
		cin>>str;
		len=strlen(str);
		z=0;o=0;temp1=0;temp2=0;
		
		for(i=0;i<len;i++)
		{
			if(str[i]=='0')
				z++;
			else
				o++;
		}
		for(i=0;i<z;i++)
			a[i]='0';
		for(i=z;i<len;i++)
			a[i]='1';
		
		for(i=0;i<o;i++)
			b[i]='1';
		for(i=o;i<len;i++)
			b[i]='0';
			
		for(i=0;i<len;i++)
			if(str[i]!=a[i])
				temp1++;
				
		for(i=0;i<len;i++)
			if(str[i]!=b[i])
				temp2++;
		
		temp1=temp1/2;
		temp2=temp2/2;
		ans=min(temp1,temp2);
		cout<<ans<<endl;
	}
	return 0;
}
