#include <bits/stdc++.h>
using namespace std;

int main()
{
    int a,b,t;
    cin>>t;

    string str;
    while(t--){
        cin>>str;
        cin>>a>>b;
        int minimum = min(a,b);
        int count=0;
        int len = str.length();
        int first = 0, last = len-1;
        int flag=0;

        while(first<last){

            if(str[first]==str[last]){
                if(str[first]!='/')
                {first++;last--;continue;}
            }
            else if(str[first]!=str[last] && str[first]!='/' && str[last]!='/')
            {flag=1;break;}

            if(str[first]=='/'){

                if(str[last]=='/'){
                    first++;last--;
                    count+=minimum*2;
                }
                else if(str[last]=='a'){
                    first++;last--;
                    count+=a;
                }
                else{
                    first++;last--;
                    count+=b;
                }
            }

            else if(str[last]=='/'){

                if(str[first]=='/'){
                    first++;last--;
                    count+=minimum*2;
                }
                else if(str[first]=='a'){
                    first++;last--;
                    count+=a;
                }
                else{
                    first++;last--;
                    count+=b;
                }
            }
        }

        if(flag)    cout<<"-1"<<endl;
        else cout<<count<<endl;
    }
    return 0;
}
