#include <iostream>
using namespace std;

int main(){
    int n,R,i,j,k=0;
    cin>>n;
    cin>>R;
    int project[n];
    for(i=0;i<n;i++)
        project[i]=i+1;
    for(i=0;i<n-1;i++){
        k=((((R%(n-i))+k)-1)+(n-i))%(n-i);
        for(j=k;j<n-i-1;j++)
            project[j]=project[j+1];
    }
    cout<<project[0]<<endl;
    return (0);
}
