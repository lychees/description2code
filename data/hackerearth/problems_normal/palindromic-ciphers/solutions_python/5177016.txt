def checkPalindrome(string):
	s1  = string[::-1]
	if s1 == string:
		return True
	return False
def getCipher(element):
	return ord(element) - 96

n = int(raw_input())
for i in range(0,n):
	string = raw_input()
	if checkPalindrome(string):
		print "Palindrome"
	else:
		product = 1
		for elm in string:
			product *= getCipher(elm)
		print product
