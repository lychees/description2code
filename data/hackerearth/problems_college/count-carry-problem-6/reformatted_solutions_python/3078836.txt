n = int(raw_input())
for i in range(0, n):
    string = raw_input().split(" ")
    str1 = string[0]
    str2 = string[1]
    cnt = 0
    for j in range(0, len(str1)):
        if ((int(str1[j])+int(str2[j])) >= 10):
            cnt = cnt+1
    if (cnt == 0):
        print"No carry operation"
    elif (cnt == 1):
        print cnt, "carry operation"
    else:
        print cnt, "carry operations"
