#define in_T int t;for(scanf("%d",&t);t--;)

#define MIN(a,b) a<b?a:b

#define pb push_back

using namespace std;
int level[100010];
vector<int> adj[100010];
void dfs(int s)
{

    for(int i=0;i<adj[s].size();i++)
    {

        if(level[adj[s][i]]==-1)
        {
            level[adj[s][i]]=level[s]+1;
            dfs(adj[s][i]);
        }
    }


}
int main()
{
    in_T
    {

        int root;
        long long n;

        cin>>n;
         long long total=(n*(n-1))/2;
         long long hs=0,fist=0;
        for(int i=1;i<=n;i++)
            level[i]=-1;
        for(int i=1;i<=n;i++)
        {

            int a;
            cin>>a;
            if(a==0)
            {
                root=i;

            }
            else
            {
            adj[i].pb(a);
            adj[a].pb(i);
            }
        }
        level[root]=0;
        dfs(root);
        for(int i=1;i<=n;i++)
        {
            hs+=level[i];
        }
        fist=total-hs;
        cout<<hs<<" "<<fist<<endl;

        for(int i=0;i<100000;i++)
            adj[i].clear();


    }
}
