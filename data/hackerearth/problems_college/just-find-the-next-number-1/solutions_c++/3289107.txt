#include <bits/stdc++.h>
#define ll long long int
#define mod (int)(1e9 + 7)

using namespace std;

int t, n;

bool check(int n) {
	vector<int> v;
	while (n > 0) {
		if (n % 2 == 0) {
			v.push_back(0);
		} else {
			v.push_back(1);
		}
		n /= 2;
	}
	bool flag = true;
	for (int i = 0; i < v.size() - 1; i++) {
		if (v[i] == 1 && v[i + 1] == 1)	return false;
	}
	return true;
}

int main()
{
	cin >> t;	
	for (int tc = 1; tc <= t; tc++) {
		cin >> n;
		while (true) {
			if (check(n)) {
				cout << n << endl;
				break;
			}
			n++;
		}
	}
	return 0;
}
