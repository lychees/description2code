N, Q = map(int, raw_input().split())
arr = []
for x in xrange(Q):
    i, j = map(int, raw_input().split())
    if i > j:
        i, j = j, i
    if i == j:
        print('NO')
        continue
    for r in arr:
        if (r[0] <= j and r[1] >= j and r[0] >= i) or (
                r[0] <= i and r[1] >= i and r[1] <= j):
            # print(arr)
            print 'NO'
            break
    else:
        # print(arr)
        arr.append((i, j))
        print 'YES'
