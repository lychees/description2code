#include <bits/stdc++.h>
using namespace std;
#define MP(x,y) make_pair(x,y)
int main()
{
    string s;
    cin>>s;
    map<pair<int,int>,int > mp;
    int n = s.size();
    int x=0, y=0;
    int ans = 0;
    for(int i=0; i<n; i++){
    	if(mp[MP(x,y)] == 1)ans++;
    	mp[MP(x,y)]=1;
    	switch(s[i]){
    		case 'U': x--;break;
    		case 'D': x++; break;
    		case 'L': y--; break;
    		case 'R': y++; 
    	}
    }
    if(mp[MP(x,y)] == 1)ans++;
    cout<<ans;
    return 0;
}
