#include <iostream>
using namespace std;
long long int tra(long long int n,long long int k){
	int ans = 0;
	for(int i=1;i<=n;i++)
		ans = (ans+k)%i;
	return ans;
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
    	long long int n;
    	cin>>n;
    	cout<<tra(n,2)+1<<endl;
    }
    return 0;
}
