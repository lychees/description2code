#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;
int main()
{
    int t,n,x,y,l1,l2,r1,r2;
    scanf("%d",&t);
    while(t--)
    {
     vector<int> X,Y;
     scanf("%d",&n);
     while(n--)
     {
      scanf("%d%d",&x,&y);
      X.push_back(x);
      Y.push_back(y);
     }
      sort(X.begin(),X.end());
      sort(Y.begin(),Y.end());
      l1=X[0];l2=X[X.size()-1];
      r1=Y[0];r2=Y[Y.size()-1];
      int ans=(l2-l1)*(r2-r1);
      int h=0;
      int v=0;
      int sum=0;
      if(ans==0)
      {
        for(int i=0;i<X.size()-1;i++)
            if(X[i]==X[i+1])
             v++;
        for(int i=0;i<Y.size()-1;i++)
            if(Y[i]==Y[i+1])
             h++;
        if(h==X.size()-1)
        {
          for(int i=1;i<X.size();i++)
            sum+=(X[i]-X[i-1]);
        }
        else
        {
          for(int i=1;i<Y.size();i++)
            sum+=(Y[i]-Y[i-1]);
        }
        printf("%d\n",sum);
      }
      else
        printf("%d\n",ans);
    }
    return 0;
}
