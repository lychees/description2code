#include <iostream>
using namespace std;

bool checkMirror(string s)
{
	bool flag = true;
	for(int i = 0, j = s.length()-1 ; i <= s.length()/2 ;i++,j-- )
	{
	  if(s[i] != s[j])
	   {
		 flag = false;
		 break;
	   }
	   else if ((s[i] > '1') && (s[i] != '8'))
	   {
         flag = false;
         break;
	   } 
	}
	return flag;
	
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	if(checkMirror(s))
    	cout<<"YES"<<endl;
    	else cout<<"NO"<<endl;
    }
    return 0;
}
