#include <bits/stdc++.h>

using namespace std;

typedef long long ll;

vector<int> A[27];

int main() {
#ifndef ONLINE_JUDGE
    freopen("input.txt", "r", stdin);
#endif

    string S;
    cin >> S;

    for(int i=0; i<S.size(); ++i) {
        A[S[i]-'a'].push_back(i);
    }

    int Q;
    cin >> Q;
    while(Q--) {
        string s;
        cin >> s;
        int p = -1;
        bool ok = true;
        for(int i=0; i<s.size(); ++i) {
            int c = s[i] - 'a';
            auto it = upper_bound(A[c].begin(), A[c].end(), p);
            if(it == A[c].end()) {
                cout << "No" << endl;
                ok = false;
                break;
            }
            p = *it;
        }
        if(ok) cout << "Yes" << endl;
    }

    return 0;
}
