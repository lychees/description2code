#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
long long modpow(long long a,long long b)
{
long long res=1;
while(b)
{
	if(b%2)
	res=(res*a)%mod;
	a=(a*a)%mod;
	b>>=1;
}
return res;
}
int main()
{
   long long t,n,ans;
   cin>>t;
   while(t--)
   {
	   ans=0;
	   cin>>n;
	   ans=modpow(2,n)-1;
	   cout<<ans<<endl;
	   
	   }	
}
