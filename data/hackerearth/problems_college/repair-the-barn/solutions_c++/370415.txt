#include<iostream>
#include<cstdio>
#include<algorithm>

using namespace std;

int main()
{
    int m,n,c,i,j;
    scanf("%d %d %d",&m,&n,&c);
    int a[c],d[c-1];
    for(i=0;i<c;i++)
        scanf("%d",&a[i]);
    sort(a,a+c);
    for(i=1;i<c;i++)
        d[i-1]=a[i]-a[i-1]-1;
    sort(d,d+c-1);
    int sum=a[c-1]-a[0]+1;
    for(i=c-2,j=1;i>=0 && j<m;j++,i--)
        sum-=d[i];
    printf("%d\n",sum);

    return 0;
}
