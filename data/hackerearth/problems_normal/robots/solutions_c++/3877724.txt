#include <stdio.h>
#include <algorithm>
#define mod 1000000007
int main()
{
    int i,j,k,l,m,n,flag,t;
    
    long long int mxm[10000] , ans;
    scanf("%d",&t);
    while(t--)
    {
    	flag =1;
    	scanf("%d",&n);
    	for(i=0;i<n;i++)
    	scanf("%lld",&mxm[i]);
    	ans =1;
    	std::sort(mxm,mxm + n);
    	for(i=0;i<n;i++)
    	{
    		if(mxm[i]-i>0)
    		ans = (ans*(mxm[i]-i))%mod;
    		else
    		{
    			flag =0;
    			break;
    		}
    	}
    	if(flag == 0)
    	printf("0\n");
    	else
    	printf("%d\n",ans);
    	
    }
    return 0;
}
