using namespace std;

struct user {
	int idx,s,j;
}arr[100001];

bool compare(user a,user b)
{
	return a.s<b.s;
}

int main(int argc, char const *argv[])
{
	int n,m;sc2(n,m);
	iloop(i,0,n)
	{
		arr[i].idx=i;
		sc2(arr[i].s,arr[i].j);
	}
	ll busytime[m];
	set(busytime,-1);
	ll ans[n];
	set(ans,-1);
	sort(arr,arr+n,compare);
	// iloop(i,0,n)
	// 	cout << i << ' ' << arr[i].idx << endl;
		// ' ' << arr[i].s << ' ' << arr[i].j << endl;
	iloop(i,0,n)
	{
		iloop(x,0,m)
		{
			if(busytime[x] <= arr[i].s)
			{
				busytime[x]= arr[i].s + arr[i].j;
				ans[i] = x+1;
				break;
			}
		}
	}
	int d[n];
	iloop(i,0,n)
		d[arr[i].idx] = i;
	iloop(i,0,n)
		cout << ans[d[i]] << ' ';
	cout << endl;
	return 0;
}
