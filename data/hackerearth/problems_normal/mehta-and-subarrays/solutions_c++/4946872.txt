using namespace std;

ll N;
ll arr[1000006];
bool check(ll K)
{
	ll i, j;
	bool foo = 0;
	for (i = 1; i <= N - K + 1; i++)
	{
		j = i + K - 1;
		ll val = arr[j] - arr[i - 1];
		if (val >= 0)
		{
			foo = true;
			break;
		}
	}
	if (foo == 0)
		return false;
	else
		return true;
}
int main()
{
	ll i, j;
	cin >> N;
	ll initial = 0;
	for (i = 1; i <= N; i++)
	{
		ll X;
		cin >> X;
		arr[i] = X + initial;
		initial = arr[i];
	}

	ll L = 1, R = N + 1;
	ll ans = -1;
	while (L <= R)
	{
		ll mid = L + (R - L) / 2;
		bool v = check(mid);
		if (v == true)
		{
			ans = mid;
			L = mid + 1;
		}
		else
			R = mid - 1;
	}
	if (ans == -1)
		cout << "-1";
	else
	{
		cout << ans << " ";
		ll count = 0;
		for (i = 1; i <= N - ans + 1; i++)
		{
			j = i + ans - 1;
			ll val = arr[j] - arr[i - 1];
			if (val >= 0)
				count++;
		}
		cout << count;
	}
	return 0;
}
