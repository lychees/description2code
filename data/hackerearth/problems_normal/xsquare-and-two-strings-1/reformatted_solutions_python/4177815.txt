for i in range(input()):
    a = raw_input()
    b = raw_input()
    s = a
    s1 = ""
    for m in range(0, len(b)):
        if (b[m] not in s1):
            s1 += b[m]
    for j in range(0, len(b)):
        if (b[j] not in s):
            s += b[j]

    if (len(s) == len(a)+len(s1)):
        print('No')
    else:
        print('Yes')
