LL mod = 1000000007;

LL BigMod(LL base,LL exp)
{
      if(exp == 0) return 1;
      LL x = BigMod(base,exp/2);
      x = (x*x)%mod;
      if(exp%2) x = (x*base)%mod;
      return x;
}

bool cmp(const PII &a, const PII &b)
{
      return (a.first<b.first || (a.first == b.first && a.second<b.second));
}

int main()
{
	inp_s;
	int t;
	cin>>t;
	char ch;
	cin.get(ch);
	while(t--)
      {
            int n;
            cin>>n;
            queue<int> pos[10001];
            int arr[n];
            FOR(i,0,n)
            {
                  int x;
                  cin>>x;
                  arr[i] = x;
                  pos[x].push(i);
            }
            sort(arr,arr+n);
            int ans = 0;
            FOR(i,0,n)
            {
                  int num = arr[i];
                  int position = pos[num].front();
                  pos[num].pop();
                  ans = max(ans,abs(i-position));
            }
            cout<<ans<<endl;
      }
      return 0;
}
