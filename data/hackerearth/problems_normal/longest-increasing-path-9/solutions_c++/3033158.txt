#include<bits/stdc++.h>
using namespace std;

#define check(i,j) (i<n and j<m)
int main(){
    static int dp[101][101],m,a[101][101],n,t;
    cin>>t;
    while(t--){
        cin>>n>>m;
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++)
                cin>>a[i][j];
        }
        for(int i=n-1;i>=0;i--){
            for(int j=m-1;j>=0;j--){
                int p=check((i+1),j)?dp[i+1][j]:0;
                int q=check(i,(j+1))?dp[i][j+1]:0;
                if(a[i+1][j]<=a[i][j]){
                    p=0;
                }
                if(a[i][j+1]<=a[i][j])
                    q=0;
                dp[i][j]=1+max(p,q);
            }
        }
        cout<<dp[0][0]<<endl;
    }
    return 0;
}
