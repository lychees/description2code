#include <stdio.h>

#define MAXN 100005

using namespace std;

int n,k;
int pos[MAXN];
int maxGap;

bool can(int dist) {
	int cur=1, jump=1;
	for (int i = 1; i <= n && jump <= k; i++) {
		if ( pos[i] - pos[cur] > dist ) {
			cur = i - 1;
			jump++;
		}
	}
	return (jump <= k);
}

int main () {
	int t;
	scanf("%d",&t);
	for (int i=1; i<=t; i++) {
		maxGap = 0;
		scanf("%d %d",&n,&k);
		for (int i = 1; i <= n; i++) {
			scanf("%d",&pos[i]);
			if ( i > 1 ) maxGap = (maxGap<pos[i]-pos[i-1])?pos[i]-pos[i-1]:maxGap;
		}
		int lo = maxGap, hi = pos[n] - pos[1];
		while ( lo < hi ) {
			int mid = (lo+hi)/2;
			if ( can(mid) ) hi = mid;
			else lo = mid + 1;
		}
		printf("%d\n", lo);
	}
}
