def reverse(x):
    rev = 0
    while x > 0:
        digit = x % 10
        rev = (rev * 10) + digit
        x /= 10
    return rev


test = int(raw_input())
for i in range(test):
    ip = raw_input()
    ip = map(int, ip.split())
    a = ip[0]
    b = ip[1]
    ans = reverse(reverse(a) + reverse(b))
    print ans
