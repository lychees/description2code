#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long test,i,len,j,ans;
	cin >> test;
	string input;
	while(test--)
	{
		cin >> input;
		len=input.size();
		if(len%2)
		{
			if(len==1)
				cout << 0 << endl;
			else if(len==3)
				cout << abs(int(input[0])-int(input[2]))<<endl;
			else
			{
				i=0;j=len-1;ans=0;
				while(i<(len/2) && j>(len/2))
					{ans+=abs(int(input[i])-int(input[j]));i++;j--;}
				cout << ans << endl;
			}
		}
		else
		{
			if(len==2)
				cout << abs(int(input[1])-int(input[0])) << endl;
			else
			{
				i=0;j=len-1;ans=0;
				while(i<(len/2) && j>=(len/2))
					{ans+=abs(int(input[i])-int(input[j]));i++;j--;}
				cout << ans << endl;	
			}
		}
	}
	return 0;
}
