#include<iostream>
using namespace std;
int main()
{
	int tc;
	long n,a[100001],i,j,sum,t;
	cin>>tc;
	while(tc--)
	{
		cin>>n;sum=0;
		for(i=0;i<n;i++)
		{
			cin>>a[i];
			sum+=a[i];
		}
		
		
		
		
		t=0;
		for(i=0;i<n;i++)
		{
		  if(sum-t-a[i]==t)
		     break;
		  t+=a[i];
		}
		if(i!=n)
		cout<<"YES"<<endl;
		else
		cout<<"NO"<<endl;
		
	    
	}
}
