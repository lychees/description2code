#include <cstdio>
#include<iostream>
using namespace std;
#include<algorithm>
#include<cmath>
#include<cstdlib>
long long s;
void check(long long a[],long long n) {
	long long sum=0;
	int i,f=0;
	for(i=n-1;i>=0;i--)  {
		sum+=a[i];
		if(sum>=s) {
			printf("%d\n",(n-i));
			f=1;
			break;
		}
	}
	if(f==0)
		printf("-1\n");
}
int main()
{
	int t;
	long long a[100001],ans,n,i;
    scanf("%d",&t);
    while(t--) {
    	scanf("%lld %lld",&n,&s);
    	for(i=0;i<n;i++)  {
    		scanf("%lld",&a[i]);
    	}
    	sort(a,a+n);
    	check(a,n);
    }
    return 0;
}
