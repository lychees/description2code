#include <iostream>
using namespace std;

int rev(int num)
{
	int rev=0;
	while(num)
	{
		rev=rev*10+num%10;
		num/=10;
	}
	return rev;
}
int main()
{
	int t;
	cin>>t;
	int *ans=new int[t];
	for(int ind=0;ind<t;ind++)
	{
			int num1,num2;
			cin>>num1>>num2;
			int sum=rev(rev(num1)+rev(num2));
			ans[ind]=sum;
	}
	for(int ind=0;ind<t;ind++)
		cout<<ans[ind]<<endl;
}
