#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int s[100],n,temp;
 		cin>>n;
 		for(int i=0;i<n;i++){
 			cin>>s[i];
 		}
 		for(int i=0;i<n-1;i++){
 			for(int j=i+1;j<n;j++){
 				if(s[i]>s[j]){
 					temp=s[i];
 					s[i]=s[j];
 					s[j]=temp;
 				}
 			}
 		}
 		
 		for(int i=0;i<n/2;i++){
 			s[i]=s[i]+s[n-1-i];
 		}
 		sort(s,s+n/2);
		if(n/2==1){
			cout<<s<<endl;
		}
		else
		cout<<s[n/2-1]-s[0]<<endl;
 		}
    return 0;
}
