for _ in range(0, input()):
    case = list(raw_input())
    case2 = list(reversed(case))
    ans = True
    for x in range(1, len(case)):
        a = abs(ord(case[x])-ord(case[x-1]))
       # print a
        b = abs(ord(case2[x])-ord(case2[x-1]))
      #  print b
        if (a != b):
            ans = False
            break
    if (ans):
        print "Funny"
    else:
        print "Not Funny"
