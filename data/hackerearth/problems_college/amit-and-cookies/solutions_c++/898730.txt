#include <iostream>
#include<algorithm>
#include<vector>
using namespace std;

int main()
{
 int n,k,i;
 int count=0;
 cin>>n>>k;
 vector<int> arr(n);
 for(i=0;i<n;i++)
 {
 cin>>arr[i];	
 }
 std::sort(arr.begin(),arr.begin()+n);
 int sum=0;
 for(i=0;i<n;i++)
 {
  sum=sum+arr[i];
  if(sum<=k){
  count=count+1;
  }
  else
  break;
 }
 cout<<count<<endl;
 return 0;
}
