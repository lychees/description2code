#include <iostream>
using namespace std;

int main()
{
    int t;
    char a[100000];
    long int i,cnt;
    cin>>t;
    while(t--)
        {
            cnt=0;
           cin>>a;
           for(i=0;a[i]!='\0';i++)
               {
               if(a[i]=='X' && a[i+1]=='Y')
                   continue;
               else
                   {
               if(a[i]=='Y' && a[i+1]=='X')
                   continue;
               else
                   cnt++;
               }
           }
        cout<<(cnt-1)<<"\n";
    }
    return 0;
}
