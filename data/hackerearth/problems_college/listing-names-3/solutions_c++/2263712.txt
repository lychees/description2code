#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    //priority_queue<string> pq;
    string s;
    vector<string> pq;
    while(t--)
    {
        cin>>s;
        pq.push_back(s);
    }
    sort(pq.begin(),pq.end());
    vector<string>::iterator it=pq.begin();
    vector<string>::iterator  it1=pq.begin();
    while(it!=pq.end())
    {
        int count=0;
        while(it1!=pq.end()&&*it==*it1)
        {
            it1++;
            count++;
        }
        cout<<*it<<" "<<count<<endl;
        it=it1;
    }
    return 0;
}
