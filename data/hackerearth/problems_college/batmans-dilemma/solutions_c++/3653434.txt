#include<bits/stdc++.h>
using namespace std;
#define MOD 1000000007
#define ll long long int
int arr[3600][3600];
//ll fac[10000000];
int fac[200000];
void fact(void)
{
		int i;
		ll res=1;
		for(i=1;i<200000;++i)
		{
				res=(res*i)%MOD;
				fac[i]=res;
		}

}
int ck(int n, int m)
{
		map<int,int > mp;
		ll i,j;
		for(i=1;i<=n;++i)
		{
				for(j=1;j<=m;++j)
				{
						arr[i][j]=i|j;
						mp[arr[i][j]]++;
				}
		}
		map<int,int>::iterator it;
		ll res=1;
		ll max=-1;
		for(it=mp.begin();it!=mp.end();++it)
		{
				res=(res * (ll)fac[it->second])%MOD;		
				//cout<<it->second<<endl;
				//if(it->second>max) max=it->second;
		}
		//cout<<"max="<<max<<endl;

		return res%MOD;

}
int main()
{
		int t;
		fact();
		cin>>t;
		while(t--)
		{
				int n,m;
				cin>>n>>m;
				cout<<ck(n,m)<<endl;
		}
}
