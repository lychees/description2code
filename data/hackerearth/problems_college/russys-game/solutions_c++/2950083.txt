#include<stdio.h>
int main()
{
    int t,i,temp,l,j,val=0;
    scanf("%d",&t);
    int o[t];
    for(i=0;i<t;i++)
    {
        scanf("%d",&l);
        int a[l];
        for(j=0;j<l;j++)
        {
            scanf("%d",&a[j]);
        }
        for(j=0;j>-100;)
        {
            if(a[j]==-99)
            {
                o[i]=1;
                break;
            }
            if(a[j]==-1)
            {
                o[i]=2;
                break;
            }
            val=a[j];
            a[j]=-99;
            j=val;
        }
    }
    for(i=0;i<t;i++)
    {
        if(o[i]==1)
        {
            printf("Lose\n");
        }
        else if(o[i]==2)
        {
            printf("Win\n");
        }
    }
    return 0;
}
