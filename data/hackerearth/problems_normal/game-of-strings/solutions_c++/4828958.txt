#include <bits/stdc++.h>
using namespace std;

int main()
{
    char s1[5005],s2[5005];
    cin>>s1;
    cin>>s2;
    int l1=strlen(s1);
    int l2=strlen(s2);
    int a[l1][l2];
    for(int i=0;i<l1;i++)
    {
    	for(int j=0;j<l2;j++)
    	{
    		if(s1[i]==s2[j]&&(s1[i]!='a'&&s1[i]!='e'&&s1[i]!='i'&&s1[i]!='o'&&s1[i]!='u'))
    		{
    			if(i==0||j==0)
    			a[i][j]=1;
    			else
    			a[i][j]=a[i-1][j-1]+1;
    		}
    		else
    		{
    			if(i==0&&j==0)
    			{
    				a[i][j]=0;
    			}
    			else
    			if(i==0&&j>0)
    			{
    				a[i][j]=a[i][j-1];
    			}
    			else
    			if(j==0&&i>0)
    			{
    				a[i][j]=a[i-1][j];
    			}
    			else
    			{
    				a[i][j]=max(a[i-1][j],a[i][j-1]);
    			}
    		}
    	}
    }
    cout<<a[l1-1][l2-1];
    return 0;
}
