#include <iostream>
using namespace std;

int main()
{

   int t,c;
   cin>>t;
   while(t--)
   {c=0;
   	int n,x,y;
   	cin>>n;

   	 int i=0;
   	int a[n];
   	while(i!=n)
   {cin>>a[i];i++;}
   i=0;x=a[0];
   while(i!=n-1)
   {y=a[i+1];
   	if(y<=x)
   {c++;
   y=x+1;
   x=y;
   }
   i++;
   }
   //for(int i=0;i<n;i++)
    //cout<<a[i]<<" ";
   int d=0;
   i=n-1;x=a[n-1];
   while(i!=0)
   {y=a[i-1];
   	if(y>=x)
   {d++;
   y=x-1;
   }

   x=y;
   i--;
   }//cout<<c<<d;

   if(c==0 || d==0)cout<<0<<endl;
   else  cout<<min(c,d)<<endl;

   }
return 0;
}
