#include <bits/stdc++.h>
using namespace std;
int main (){
	string alphabetCap = "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
	string alphabetlow = "abcdefghijklmnopqrstuvwxyz";
	char input;
	int base;
	while(cin >> input >> base){
		if(base <=1){
			cout << "INVALID"<< endl;
		}else{
			int pos;
			if(isupper(input)){
				pos = alphabetCap.find(input);
				if(alphabetCap.size()-pos < base){
					cout << "INVALID" << endl;
				}else{
					for(int i = 0; i < base; i++){
						for(int j = 0; j < i + 1; j++){
							cout << alphabetCap[pos] << " "; 
						}
						pos++;
						cout << endl;
					}
				}
			}else{
					pos = alphabetlow.find(input);
				if(alphabetlow.size()-pos < base){
					cout << "INVALID" << endl;
				}else{
					for(int i = 0; i < base; i++){
						for(int j = 0; j < i + 1; j++){
							cout << alphabetlow[pos] << " "; 
						}
						pos++;
						cout << endl;
					}
				}
			}
		}
		cout << endl;
	}
	return 0;
}
