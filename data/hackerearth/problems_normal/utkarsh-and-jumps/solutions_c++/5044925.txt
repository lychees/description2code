#include <iostream>
#include <vector>
#include <iomanip>
using namespace std;

int main()
{
	int n;
	double p;
	cin>>n>>p;
	vector<double> arr(n+1,0);
	arr[0] = 1.0;
	arr[2] = ((double)p/100)*arr[0];
	arr[3] = (1-(double)p/100)*arr[0];
	for(int i=4;i<=n;i++){
		arr[i] = (p/100)*arr[i-2]+(1-p/100)*arr[i-3];
	}
	cout<<setprecision(6)<<fixed;
	cout<<arr[n];
    return 0;
}
