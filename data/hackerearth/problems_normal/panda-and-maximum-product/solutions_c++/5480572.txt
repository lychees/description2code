#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n; cin >> n;
    vector < long long > vec;
    for (int i = 0; i < n; i ++) {
    	long long x; cin >> x;
    	vec.push_back(x);
    }
    
    sort(vec.rbegin(),vec.rend());
    
    cout << max(vec[0] * vec[1], vec.back() * vec[vec.size() - 2]) << '\n';
    return 0;
}
