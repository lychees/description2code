#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define pb push_back
#define mp make_pair

vector<pair<ll,ll> >v[1000003];
ll vis[1000003],dis[1000003],n;
stack<ll>st;

void topo(int x)
{
    vis[x]=1;
    for(ll j=0;j<v[x].size();j++)
    {
        ll k=v[x][j].first;
        ll l=v[x][j].second;
        if(vis[k]==0)
            topo(k);
    }
    st.push(x);
}

void dfs(ll z)
{
   for(ll i=0;i<v[z].size();i++)
   {
       ll j=v[z][i].first;
       ll k=v[z][i].second;
       if(dis[j]<dis[z]+k+2)
        dis[j]=dis[z]+k+2;
   }
}

int main()
{
    ll m,j,i,k,l,x,y,z,c=0,maxe,ans=0;
    cin>>n>>m;
    for(i=1;i<=n;i++)
    {
        dis[i]=0;
        vis[i]=0;
    }
    for(i=0;i<m;i++)
    {
        cin>>x>>y>>z;
        v[x].pb(mp(y,z));
    }
    for(i=1;i<=n;i++)
    {
        if(vis[i]==0)
            topo(i);
    }

    maxe=0;
    while(!st.empty())
    {
        k=st.top();
        dfs(k);
        st.pop();
    }

    for(i=1;i<=n;i++)
        {
            ans=max(ans,dis[i]);
            //cout<<dis[i]<<" ";
        }
    cout<<ans;

}
