#include <iostream>
#include <cstdio>
#include <set>
using namespace std;

int n;
int a[100];
int ok(int M) { //kinda of brute force
    int z = a[0]%M;
    for(int i=1; i<n; ++i ) {
        if(a[i]%M!=z)return 0;
        }
    return 1;
    }
int main() {

    scanf("%d",&n);
    set<int>v;
    for(int i=0; i<n; ++i) scanf("%d",&a[i]);
    int p = a[0]-a[1];//gcd(a-b,b-c,c-d,..) cannot be larger than (a-b)/(b-c)/(c-d)...could have taken any value but the author chose to take the first difference, a more optimal approach would be to take the smallest difference
    if(p<0) p=-p;
    for(int d=2; d <= p; ++d) {
        if(p%d!= 0)continue;
        if(ok(d) && d!=1) v.insert(d);
        }
    set<int>::iterator it;
    for(it = v.begin(); it != v.end(); ++it) printf("%d ",*it);
    printf("\n");
    return 0;
    }
