#include<bits/stdc++.h>
#define N 1024
#define MOD 1000000007
using namespace std;
short arr[N],mas[N],size[N],siz[N];
short n,k,a,b;
long long ans=1,_ans=1;
int find_root(int x)
{
	while(arr[x]!=x)
	{
		arr[x]=arr[arr[x]];
		x=arr[x];
	}
	return x;
}
void _union(int x,int y)
{
	int root_x=find_root(x);
	int root_y=find_root(y);
	if(root_x!=root_y)
	{
		arr[root_y]=root_x;
	}
}
int main()
{
	ios::sync_with_stdio(false);
	cin.tie(NULL);
	cin>>n;
	for(short i=1;i<=n;i++)
	{
		cin>>mas[i];
		arr[i]=i;
		size[i]=1;
	}
	cin>>k;
	for(short i=0;i<k;i++)
	{
		cin>>a>>b;
		_union(a,b);
	}
	//cout<<"---------"<<endl;
	for(short i=1;i<=n;i++)
	{
		arr[i]=find_root(i);
		//cout<<i<<" arr[i]="<<arr[i]<<endl;
		siz[arr[i]]++;
		//if(arr[i]==5)cout<<"ATtention i="<<i<<endl;siz[arr[i]]++;
	}
	for(short i=1;i<=n;i++)
	{
		//cout<<i<<" "<<arr[i]<<endl;
		if(siz[i]>1)
		{
			//cout<<"i="<<i<<" "<<size[i]<<endl;
			short cnt=0;
			int minn=100000;
			for(short j=1;j<=n;j++)
			{
				if(arr[j]==i)
				{
					if(minn>mas[j])
					{
						cnt=1;
						minn=mas[j];
					}
					else
					if(minn==mas[j])
					{
						cnt++;
					}
				}
			}
			//cout<<minn<<endl;
			//cout<<i<<" "<<cnt<<endl;
			ans=(ans*cnt)%MOD;
			//_ans=(_ans*size[i])%MOD;
		}
	}
	cout<<ans<<endl;
	return 0;
}
