#include <bits/stdc++.h>
using namespace std;
int ans;
int n;
int a[21][3];
void foo(int idx,int cnt,int sum){
	int p,q;
	if(cnt==n){ ans= min(ans,sum); return ;}
	else {
		if(idx==0) {
			foo(1,cnt+1,sum+a[cnt][1]);
			foo(2,cnt+1,sum+a[cnt][2]);
		}
		else if(idx==1)
		{
			foo(0,cnt+1,sum+a[cnt][0]);
			foo(2,cnt+1,sum+a[cnt][2]);
		}
		else 
		{
			foo(0,cnt+1,sum+a[cnt][0]);
			foo(1,cnt+1,sum+a[cnt][1]);
		}
	}
}
int main()
{
    int t;
    cin>>t;
    ans=0;
    while(t--)
    {
    	ans=INT_MAX;
    	cin>>n;
    	
    	for(int i=0;i<n;i++)
    	{
    		for(int j=0;j<3;j++)
    		{
    			cin>>a[i][j];
    		}
    	}
    			
    	foo(0,1,a[0][0]);
    	foo(1,1,a[0][1]);
    	foo(2,1,a[0][2]);
    	cout<<ans<<endl;
    }
    return 0;
}
