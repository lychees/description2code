#include<iostream>
#include<queue>
#include<vector>
using namespace std;
int main(){
	long long int N,temp,f,s,t;
	long long int r;
	priority_queue<int> q;
	scanf("%d",&N);
	printf("-1\n-1\n");
	scanf("%d",&temp);
	q.push(temp);
	scanf("%d",&temp);
	q.push(temp);
	for(int i=1;i<=N-2;i++){
		scanf("%d",&temp);
		q.push(temp);
		f=q.top();
		q.pop();
		s=q.top();
		q.pop();
		t=q.top();
		r=f*s*t;
		cout<<r<<"\n";
		q.push(f);
		q.push(s);
		}
	return 0;
}
