#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define mod 1000000007
int vis[100005]={0};
int arr[100005];
vector<int> gh[100005];
map<ll,ll> mp;
void dfs(int u)
{
	if(vis[u] == 0)
	{
		mp[arr[u]]++;	
		vis[u] = 1;
		vector<int>::iterator it;
		it = gh[u].begin();
		while(it!=gh[u].end())
		{
			//mp[arr[*it]]++;
			dfs(*it);
			it++;
		}
	}
}
int main()
{
	ios_base::sync_with_stdio(false);
	int n,m,i;
	cin>>n>>m;
	for(i=0;i<n;i++)
		cin>>arr[i];
	ll ans = 1;
	for(i=0;i<m;i++)
	{
		int x,y;
		cin>>x>>y;
		x--;y--;
		gh[x].push_back(y);
		gh[y].push_back(x);
	}
	for(i=0;i<n;i++)
	{
		mp.clear();
		if(vis[i]==0)
		{
			mp[arr[i]]++;
			vector<int>::iterator it;
			it = gh[i].begin();
			vis[i] = 1;
			while(it!=gh[i].end())
			{
				if(vis[*it]==0)
				{
					dfs(*it);
				}
				it++;
			}
			ans *= (mp.rbegin())->second;
			if(ans >= mod)
				ans = ans%mod;
		}
	}
	cout<<ans<<"\n";
	return 0;
}
