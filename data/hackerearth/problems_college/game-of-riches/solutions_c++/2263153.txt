#include <stdio.h>
#include <stdlib.h>

int cmpfunc(const void * a, const void *b)
{
	return (*(int*)b-*(int*)a);
}

int main()
{
    int t, nr, ng, w[100], i, j, c, myw, gp_size, pos, q;
    scanf("%d",&t);
    
    for(i=0;i<t;i++)
    {
    	scanf("%d %d",&nr, &ng);
    	
    	for(j=0;j<100;j++)
    		w[j]=0;
    	
    	for(j=0;j<nr;j++)
    		scanf("%d",&w[j]);
    	
    	myw=w[0];
    	qsort(w,nr,sizeof(int),cmpfunc);
    	
    
    	for(j=0;j<nr;j++)
    	{ 
    		if(w[j]==myw)
    		{
    			pos=j;
    			break;
    		}
    	}
    	
    	gp_size=ng;
    	q=pos%gp_size;	// group id
    	
    
    	c=0;
    	for(j=0;j<nr;j++)
    	{    		
    		if(j%gp_size==q && w[j]>myw)	//if gp_id==same elements id
    			c++;
    	}
    	printf("%d\n",c);
    }
}
