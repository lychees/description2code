#include <bits/stdc++.h>
#include <math.h>
 
using namespace std;
typedef long long int ll;
typedef long double ld;
#define pb push_back
#define mod 1000000007
 
int main()
{
	ios_base::sync_with_stdio(0);
	string str="";
	cin>>str;
	int l=str.length();
	int dp[l+1][26];
	for(int i=0;i<26;i++)
		dp[l][i]=l+1;
	for(int i=l-1;i>=0;i--)
	{
		for(int j=0;j<26;j++)
			dp[i][j]=dp[i+1][j];
		dp[i][str[i]-'a']=i;
	}
	int t;
	cin>>t;
	while(t--)
	{
		string inp;
		cin>>inp;
		int pos=0;
		bool flag=true;
		for(int i=0;i<inp.length();i++)
		{
			if(dp[pos][inp[i]-'a']>l)
			{
				flag=false;
				break;
			}
			else
				pos=dp[pos][inp[i]-'a'];
		}
		if(flag)
			cout<<"Yes\n";
		else
			cout<<"No\n";
	}
	return 0;
}
