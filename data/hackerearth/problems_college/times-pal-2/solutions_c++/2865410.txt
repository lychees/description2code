#include<bits/stdc++.h>
using namespace std;

bool isPalin(string s)
{

	int i=0,j=s.size()-1;
	while(i<j)
	{

		if(s[i]!=s[j])return false;
		i++;j--;
	}
	return true;
}

int main()
{
  int t;
	cin>>t;
	while(t--)
	{

		string a, b;
		cin>>a>>b;
		int count = 0;int n = a.size();
		for(int i=0; i<=n; i++)
		{

			string x, y, z;
			x = a.substr(0,i);
			y = a.substr(i,n-i);
			z = x+b+y;
			if(isPalin(z))count++;
		}
		cout<<count<<endl;
	}
	return 0;
}
