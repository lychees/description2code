#include <iostream>
using namespace std;

int main()
{int t;
cin>>t;
while(t--){
	int n;
	cin>>n;
	long long int ar[n];
	for(int i=0;i<n;++i)
		cin>>ar[i];
	long long int count=0;
	long long int count1=0;
	for(int i=0;i<n;++i)
	{
		if(ar[i]%2==0)
			count++;
		else
			count1++;
			
	}
	
	cout<<count*count1<<endl;
	
	
}
   // cout << "Hello World!" << endl;
    return 0;
}
