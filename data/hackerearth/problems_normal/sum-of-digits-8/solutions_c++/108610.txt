void solve();
int min(long long int a, long long int b)
{
	if(a<b)
	return a;
	else return b;
}
int main()
{
	long long int x,y,sum;
	scanf("%lld %lld %lld",&x,&y,&sum);
	long long int a[17];
	for(long long int i=0;i<17;i++)
	a[i]=0;
	long long int val=16;
	while(x>0)
	{
		a[val]=x%10;
		x=x/10;
		val--;
	}
	long long int val1=val+1;
	long long int b[17];
	for(long long int i=0;i<17;i++)
	b[i]=0;
	val=16;

	while(y>0)
	{
		b[val]=y%10;
		y=y/10;
		val--;
	}
	val=val+1;
	long long int dp[17][136];
	for(long long int j=0;j<136;j++)
	dp[16][j]=0;
	for(long long int j=0;j<=9;j++)
	dp[16][j]=1;
	for(long long int i=15;i>=0;i--)
	{
		for(long long int j=0;j<136;j++)
		{
			dp[i][j]=0;
			for(long long int k=0;k<=min(9,j);k++)
			{
				dp[i][j]=dp[i][j]+dp[i+1][j-k];
			}
		}
	}
	long long int flag=1;
	long long int s;
	long long int e;
	long long int tcnt=0;
	long long int tcnt1=0;
	long long int ts=sum;
	long long int c[17];
	for(long long int i=0;i<17;i++)
	{
		c[i]=a[i];
		a[i]=0;
	}
	for(long long int it=val;it<17;it++)
	{
		if(flag)
		s=a[it];
		else
		s=0;
		e=b[it];
		for(long long int i=s;i<e;i++)
		{
			flag=0;
			if(sum >= i)
			{
				if(it!=16)
				{
					tcnt=tcnt+dp[it+1][sum-i];

				}
				else{
					if(sum == i)
					{
//						cout<<"here"<<endl;
						tcnt++;
					}
				}
			}
		}
		if(it==16)
		{
				if(sum == e)
				tcnt++;
		}				
		sum=sum-b[it];
		if(sum<0)
		break;
	}	
	tcnt1=0;
	for(long long int i=0;i<17;i++)
	b[i]=c[i];
	val=val1;
	sum=ts;
	flag=1;
	for(long long int it=val;it<17;it++)
	{
		if(flag)
		s=a[it];
		else
		s=0;
		e=b[it];
		for(long long int i=s;i<e;i++)
		{
			flag=0;
			if(sum >= i)
			{
				if(it!=16)
				{
					tcnt1=tcnt1+dp[it+1][sum-i];
	//				cout<<dp[it+1][sum-i]<<" "<<sum-i<<endl;
				}
				else{
					if(sum == i)
					{
		//				cout<<"here"<<endl;
						tcnt1++;
					}
				}
			}
		}
		if(it==16)
		{
				if(sum == e)
				tcnt1++;
		}				
		sum=sum-b[it];
		if(sum<0)
		break;
	}
	val=0;
	for(long long int i=0;i<17;i++)
		val=val+b[i];
	if(val == ts)
	tcnt++;	
	long long int final1=tcnt-tcnt1;
	long long int final2=0;
	flag=1;
	long long int diff=ts-val;
	if(val <= ts)
	{
		for(long long int i=16;i>=0;i--)
		{
			while(1){
			if(diff == 0)
			{
				flag=0;
				break;
			}
			if(b[i] == 9)
			break;
			b[i]++;
			diff--;
			}
			if(flag==0)
			break;
		}
	}
	else
	{
		long long int ind;
		for(long long int i=16;i>=0;i--)
		{
			while(1)
			{	
		
				if(diff == 1)
				{
					ind=i;
					flag=0;
					break;
				}
				if(b[i] == 0)
				break;
				b[i]--;
				diff++;
			}
			if(flag == 0)
			break;
		}
		for(long long int j=ind-1;j>=0;j--)
		{
			if(b[j] == 9)
			continue;
			b[j]++;
			break;
		}
	}					
	for(long long int i=0;i<17;i++)
	{
		final2=final2*10+b[i];
	}
	cout<<final1<<endl;
	cout<<final2<<endl;
}
