def search_max(arr):
	n,mx,x = len(arr),0,1
	for i in range(1,n-1):
		if arr[i-1]*arr[i+1] > mx:
			x = i
			mx = arr[i-1]*arr[i+1]
	return [mx,x]

n = int(raw_input())
p = map(int, raw_input().split())

ans = 0
for i in range(n-2):
	ret = search_max(p)
	ans += ret[0]
	p = p[:ret[1]]+p[ret[1]+1:]
print ans
