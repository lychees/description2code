#include<bits/stdc++.h>
using namespace std;
main()
{
	long long int n,f,l,s,m,flag=0;
	scanf("%lld",&n);
	long long int a[n+5],i,j,k;
	for(i=0;i<n;i++)
	{
		scanf("%lld",&a[i]);
	}
	scanf("%lld",&k);
	sort(a,a+n);
	for(i=0;i<n;i++)
	{
		for(j=n-1;j>=0;j--)
		{
		    if(i<j)
			{
				s=k-(a[i]+a[j]);
				f=0;l=n-1; 
				m=(f+l)/2;
				while(f<=l)
				{
					if(a[m]==s&&m!=i&&m!=j)
					{
						printf("YES") ;
						exit(0) ;
					}
					else if(a[m]<s)
					{
						f=m+1;
					}
					else if(a[m]>s)
					{
						l=m-1;
					}
					m=(f+l)/2;
				}
		  }
		}
	}
	printf("NO") ;
}
