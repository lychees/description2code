def foo(a,b,c,d,t):
	return a*(t**3)+b*(t**2)+c*t+d
	
test_cases=int(raw_input())
for t in xrange(test_cases):
	items=raw_input().split(" ")
	a,b,c,d,k=[int(item) for item in items]
	t_current=0
	gap=100
	t_prev=0
	while True:
		if k <= foo(a,b,c,d,t_current):
			break
		else:
			t_prev=t_current
			t_current+=gap
	for pointer in xrange(t_prev,t_current+1):
		if k <= foo(a,b,c,d,pointer):
			break
		pointer+=1
	if pointer==0:
		print pointer
	else:
		print pointer-1
