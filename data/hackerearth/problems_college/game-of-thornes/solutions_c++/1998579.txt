#include<bits/stdc++.h>

using namespace std;


map<char,int> memo;
int main()
{
	string a;
    int flag=0;
    //scanf("%s",a);
	cin>>a;
    for(int i=0;a[i]!='\0';i++)
        memo[a[i]]++;

    for(map<char,int>::iterator it = memo.begin();it!=memo.end();it++)
    {
        if((it->second)%2!=0)
        {
            flag++;
            if(flag>1){
        		printf("NO\n");        
              	return 0;
            }
        }
    }

    printf("YES\n");

	return 0;
}
