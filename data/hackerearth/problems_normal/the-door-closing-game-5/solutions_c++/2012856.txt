#include <bits/stdc++.h>
using namespace std;

void win(char *str,int l)
{
int c=0,x=0,i=0,flag=0;

while(i<l)
	{
		
	if(str[i]=='|')
	{
		x=x^c;
		flag=1;
		c=0;	
	}
	else
	c++;
	
	i++;
}
if(flag&&c)
x=x^c;


if(x)
cout<<"Alice\n";

else
cout<<"Bob\n";


}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	char str[101];
    	cin>>str;
    //	cout<<str;
    	int l=strlen(str);
        win(str,l);
    }
    return 0;
}
