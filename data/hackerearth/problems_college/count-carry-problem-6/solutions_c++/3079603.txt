#include <iostream>
using namespace std;
int main()
{
	int t,count;
	cin>>t;
	for(int i=0;i<t;i++)
	{
		long long int a,b;
		cin>>a>>b;
		count=0;
		while(a>0 || b>0)
		{
			int p=a%10;
			int q=b%10;
			if((p+q)>=10)
			count++;
			a=a/10;
			b=b/10;
		}
		if(count>1)
		cout<<count<<" carry operations"<<endl;
		else if(count==1)
		cout<<"1 carry operation"<<endl;
		else
		cout<<"No carry operation"<<endl;
	}
    return 0;
}
