#include <bits/stdc++.h>
#include <iostream>
#include <algorithm>
#include <set>
#include <string>
using namespace std;
#define MAX 100005
int f[MAX];

void preproccess()
{
	memset(f, 0, sizeof(f));
	int a = 0;
	int b = 1;
	f[a] = 1;
	f[b] = 1;
	int c = 0;
	while ((a + b) < MAX - 2) {
		c = a + b;
		f[c]++;
		a = b;
		b = c;
	}
	for (int i = 1; i < MAX; i++) {
		f[i] += f[i - 1];
	}
}

int main()
{
	preproccess();
	int t;
	scanf("%d", &t);
	while (t--) {
		int l, r;
		scanf("%d%d", &l, &r);
		if ((f[r] - f[l - 1]) & 1) {
			puts("INVALID");
		}
		else {
			puts("VALID");
		}
	}
	return 0;
}
