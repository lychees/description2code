#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
	ll t;
	cin>>t;
	while(t--){
   string s;
   ll l,i,j;
   cin>>s;
   l=s.length();
   int f=0;
   for(i=0,j=l-1;i<j;i++,j--)
   {
   	if(abs(s[i+1]-s[i])!=abs(s[j]-s[j-1]))
   	{
   		f=1;
   		break;
   	}
   }
   if(f==1)cout<<"Not Funny"<<endl;
   else
   cout<<"Funny"<<endl;
	}
    return 0;
}
