#include<iostream>
#include<stdio.h>
using namespace std;


int main()
{
	long long int t,i,count,n;
	cin>>t;
	while(t--)
	{
		
		cin>>n;
		count = 0;
		long long int hash[1000020] = {0};
		for(i=0;i<n;i++)
		{
			long long int a,b;
			cin>>a;
			hash[a]++;
			cin>>b;
			if(hash[b])
			{
				hash[b]--;
			}
			else
			{
				count++;
			}
		}
		
		cout<<count<<endl;
	}
	
	return 0;
}
