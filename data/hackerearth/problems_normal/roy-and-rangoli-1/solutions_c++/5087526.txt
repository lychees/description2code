#include<bits/stdc++.h>
using namespace std;
typedef long long int lu;
#define de 2000001
#define re 1000000007
int main()
{
	
	lu n,j,r,c,i,k,ans;
	int pr[de]={0};
	pr[0]=pr[1]=1;
	for(i=2;i*i<de;i++)
	{
		k=2;
		if(pr[i]==0)
		for(j=i*2;j<de;)
		{
			if(j%i==0)
			pr[j]=1;
			k++;
			j=i*k;
		}
	}
	scanf("%lld",&n);
	r=n-1;
	c=n-1;
	ans=0;
	for(i=2;i<=c;i++)
	{
		if(pr[i]==0)
		{
			ans=(ans+i+1)%re;
		}
	}
	j=r;
	
	if(pr[n]==0)
	{
		ans=(ans+n-1)%re;
	}
	for(i=2;i<=r;i++)
	{
		if(pr[j+i]==0)
		ans=(ans+n-i)%re;
	}
	printf("%lld",ans);
	
	return 0;
}
