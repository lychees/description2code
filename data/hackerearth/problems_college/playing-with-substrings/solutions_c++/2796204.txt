#include <bits/stdc++.h>
using namespace std;
int main()
{
    int i,j;
string s,t;
cin>>s>>t;
int c=0;
set<string> myset;
set <string> ::iterator it;
for(i=0;i<s.length();i++)
{
    for(j=1;j+i<=s.length();j++)
    {
        string tok=s.substr(i,j);
        myset.insert(tok);

    }


}
for(it=myset.begin();it!=myset.end();it++)
{
    string str=*it;
    if(str.find(t)!=string::npos)
        c++;
}

cout<<c;
return 0;
}
