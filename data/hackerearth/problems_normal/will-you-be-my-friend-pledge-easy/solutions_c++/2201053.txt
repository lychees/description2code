#include<bits/stdc++.h>
using namespace std;
long int gcd(long long int a , long long int b)
{
    long long int r=a%b;
    if(r==0)
        return b;
    else
        return gcd(b,r);


}
int main()
{

    long long int a,n,c=0,temp;

    cin>>a>>n;

    for(long long int i=0;i<n;i++)
    {
        cin>>temp;
        if(gcd(temp,a)!=1)
            c++;
    }
    cout<<c<<endl;


}
