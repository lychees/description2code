   int t;

   while(t--)
   {
	 map<ll,bool> mp;
	 memset(ans,0,SIZE);
	 ll n,m,i,val;
	 cin>>n>>m;
	 for(i=1;i<=(n+m);i++)
	 {
		scanll(val);
		if(i>n)
		{
	 	  	auto it=mp.find(val);
 	 		ans[i]=(it!=mp.end())?true:false;
		}
	  mp.insert({val,true});
	}
	for(i=n+1;i<=(n+m);i++)
	{
		ans[i]?cout<<"YES\n":cout<<"NO\n";
	}

	}
