#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main(){
	int t;
	cin>>t;
	while(t--){
		ll n,m;
		cin>>n>>m;
		vector<ll> v;
		for(ll i=0;i<n;i++){
			ll temp;
			cin>>temp;
			v.push_back(temp);
		}
		sort(v.rbegin(),v.rend());
		long long sum=0;
		ll count=0;
		bool done=false;
		for(ll i=0;i<n;i++){
			sum+=v[(int)i];
			count++;
			if(sum>=m){
				done=true;
				break;
			}
		}
		if(!done){
			cout<<"-1"<<endl;
		}
		else{
			cout<<count<<endl;
		}
	}
}
