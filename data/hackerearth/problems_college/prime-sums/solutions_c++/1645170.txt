#include <bits/stdc++.h>
using namespace std;
const int maxi=4000001;
vector<long long> a(maxi,0);
long long pre(long long n)
{
	a[1]=a[0]=0;
	long long i,j,s=0;
	for(i=2;i<=sqrt(n);i++)
	{
		if(a[i]==0)
		{
			for(j=i*2;j<n;j+=i)
			a[j]=1;
		}
	}
	for(i=2;i<n;i++)
	{
		if(a[i]==0)
		s+=i;
	}
	return s;
}
int main()
{
	   
    //cout << "Hello World!" << endl;
    long long n;
    cin>>n;
   
 if(n<2)
 cout<<"0"<<endl;
 else
 cout<< pre(n)<<endl;
    return 0;
}
