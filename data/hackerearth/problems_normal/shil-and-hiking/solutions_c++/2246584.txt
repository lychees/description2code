#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n;
	int x[1000], y[1000];
	int i, j, ans, max_x, max_y;
	
	cin >> n;
	
	for (i=0; i<n; i++) {
		cin >> x[i] >> y[i];
	}
	
	ans = n-1;
	for (i=0; i<n-1; i++) {
		max_x = x[i+1] - x[i];
		max_y = y[i+1] - y[i];
		for (j=i+2; j<n; j++) {
			if (1.0*(y[j]-y[i])/(x[j]-x[i]) >= 1.0*max_y/max_x) {
				ans++;
				max_x = x[j] - x[i];
				max_y = y[j] - y[i];
			}
		}
	}
	
	cout << ans << endl;
	
    return 0;
}
