for _ in xrange(input()):
    n = input()
    a = map(int, raw_input().split())
    if n == 1:
        print 0
        continue
    ans = 0
    mini = abs(a[1]-a[0])
    for i in xrange(1, n-1):
        tmp = abs(a[i]-a[i-1])+abs(a[i]-a[i+1])-abs(a[i+1]-a[i-1])
        if tmp > mini:
            mini = tmp
            ans = i
    if abs(a[n-1]-a[n-2]) > mini:
        ans = n-1
    print ans
