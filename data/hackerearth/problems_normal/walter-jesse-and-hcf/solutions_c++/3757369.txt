#include<cstdio>


long long gcd(long long a, long long b){
    if(b == 0) return a;
    return gcd(b, a%b);
}

void solve(long long g, long long k){
    long long temp = 0;
    long long check1 = 0;
    long long check2 = 0;
    for(long long i=1;i*i<=g;i++){
        if(g%i == 0){
            temp++;
            if(g/i != i){
                temp++;
            }
        }
    }
    if(temp >= k){
        for(long long i=1;i*i<=g;i++){
            if(g%i == 0){
                check1++;
                check2 = temp-check1+1;
                if(check2 == k){
                    printf("%lld\n", i);
                }
                if(g/i != i){
                    check2 = check1;
                    if(check2 == k){
                        printf("%lld\n", g/i);
                    }
                }
            }
        }
    }
    else{
        printf("No crymeth today\n");
    }
}

int main(){
    int d;
    scanf("%d", &d);
    while(d--){
        long long x, y, k;
        scanf("%lld %lld %lld", &x, &y, &k);
        long long g = gcd(x, y);
        solve(g, k);
    }
}
