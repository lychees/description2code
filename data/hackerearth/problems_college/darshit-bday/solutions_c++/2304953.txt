#include<cstdio>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		unsigned long long int b,w,x,y,z,cst;
		scanf("%llu %llu %llu %llu %llu",&b,&w,&x,&y,&z);
		if(x+z<=y)
			cst=x*b+w*(x+z);
		else if(y+z<=x)
			cst=b*(y+z)+w*y;
		else
			cst=x*b+y*w;
		printf("%llu\n",cst);	
	}
	return 0;
}
