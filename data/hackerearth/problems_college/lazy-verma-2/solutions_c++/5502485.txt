#include <iostream>
using namespace std;

int main()
{
    long long int n,m,sum1=0,sum2=0;
    int narr[100000],marr[100000];
    cin >> n;
    cin >> m;
    for(int i = 0;i < n;i++)
    {
    	cin >> narr[i];
    	sum1 = sum1 + narr[i];
    }
    for(int j = 0;j < m;j++)
    {
    	cin >> marr[j];
    	sum2 = sum2 + marr[j];
    }
    cout << sum1*sum2 << endl;
    return 0;
}
