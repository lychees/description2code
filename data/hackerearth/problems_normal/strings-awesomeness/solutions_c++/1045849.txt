#include <bits/stdc++.h>

using namespace std;

int get_val(char c) {
    return ((c >= 'a') && (c <= 'z')) ? c - 'a' : c - 'A';
}

bool check(vector<int> &a, vector<int> &b) {
    for(int i = 0; i < 26; i++) {
        if (a[i] != b[i]) return false;
    }
    return true;
}

int main()
{
    string s;
    cin >> s;
    int n = s.length();
    vector<int> S(n + 1, 0), a(30, 0), b(30, 0);
    string pat = "hackerearth";
    for(int i = 0; i < pat.length(); i++) a[get_val(pat[i])]++;
    for(int i = 0; i < pat.length() - 1; i++) b[get_val(s[i])]++;
    for(int i = pat.length() - 1; i < n; i++) {
        if (i >= pat.length()) b[get_val(s[i - pat.length()])]--;
        b[get_val(s[i])]++;
        S[i - pat.length() + 2] = S[i - pat.length() + 1] + (check(a, b));
    }
    //for(int i = 0; i <= n - pat.length() + 1; i++) cout << S[i] << " "; cout << endl;
    int q;
    cin >> q;
    for(int i = 1; i <= q; i++) {
        int l, r;
        scanf("%d %d", &l, &r); //cin >> l >> r;
        if (r - l + 1 < pat.length()) cout << 0 << endl;
        else printf("%d\n", S[r - pat.length() + 1] - S[l + pat.length() - 1 - pat.length()]);
    }
}
