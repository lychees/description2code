#include <bits/stdc++.h>
using namespace std;

int main()
{
	// your code goes here
	int test; cin>>test;
	while(test--)
	{
		string s; cin>>s;
		int i,j,c=0;
		for(i=0,j=s.size()-1;i<=j;i++,j--)
		{
			if(s[i]!=s[j])
			{
				c=1;
				break;
			}
		}
		if(c==0)cout<<-1<<endl;
		else
		{
			string ss="";
			for(int k=0;k<s.size();k++)if(k!=i)ss+=s[k];
			string sss = ss;
			reverse(ss.begin(),ss.end());
			//cout<<sss<<' '<<ss<<endl;
			if(ss==sss)cout<<i<<endl;
			else cout<<j<<endl;
		}
	}
	return 0;
}
