#include <iostream>
using namespace std;

int main()
{
    long long int t,n,m,c,p,ans,x,temp;
   
   cin>>t;
   
   while(t--)
   {
   	cin>>n>>m;
   	ans=0;
   	c=1;
   	p=0;
   	if(n==1)
   		p=m;
   		
   	while(p<m&&c<n)
   	{
   		ans++;
   		temp=c;
   		c+=p;
   		if(p<m)
   		p=min(m,p+temp);
   		
   	}
   	
   	x=(n-c)/p;
   	if(x>=0)
   	{
   	c+=(p*x);
   	ans+=x;
   	}
   	
   	if(n-c>0)
   	ans++;
   	
   cout<<ans<<endl;
   }
    return 0;
}
