#include <bits/stdc++.h>
using namespace std;
#define lim 1000003
int main()
{
	long long i,j,A[lim],T,t,p,count,n,k;
	//for(i=0;i<p;i++) cout<<A[i]<<" ";
	cin>>T;
	while(T--){
		cin>>n;
		if(n%2==0) k=1;
		else k=0;
		for(i=2;i<=sqrt(n);i++){
			if(n%i==0){
				if(i%2==0) k++;
				if((n/i)%2==0 && i!=n/i) k++;
			}
		}
		cout<<k<<endl;
	}
    return 0;
}
