#include <iostream>
using namespace std;
int main()
{
    long long t,M,N,s;
    cin>>t;
    while(t--)
    {
      cin>>N>>M>>s;
      if(N%M==0)
      {
        if(s==1)
        cout<<N<<endl;
        else
        cout<<s-1<<endl;
      }
      else
      {
        int k=(M%N)-1;
        k=k+s;
        k=k%N;
        if(k==0)
        cout<<N<<endl;
        else
        cout<<k<<endl;
      }
   }
   return 0;
}
