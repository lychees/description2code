using namespace std;

bool nt[200000];
long nto[200000];
long n, snt, cnt, z;
long a[200000];


void sang()
{
     long i, j;
     memset(nt, true, sizeof(nt));
     nt[0] = false;
     nt[1] = false;
     for (i = 2; i <= 200000; i++)
         if (nt[i])
         {
            j = i;
            while (j + i <= 200000)
            {
                  j = j + i;
                  nt[j] = false;
            }
         }
     snt = 0;
     for (i = 2; i <= 400; i++)
         if (nt[i])
         {
            snt += 1;
            nto[snt] = i;
         }
}

int main()
{
    long x;
    sang();
    cin >> n;
    for (int i = 1; i <= n; i++)
    {
        cin >> x;
        if (x == 1)
        {
           a[i] == 0;
           continue;
        }
        
        for (int j = 1; j <= snt; j++)
        {
            while (x % nto[j] == 0)
            {
                  a[i] += 1;
                  x = x / nto[j];
                  if (x == 1) break;
            }
            if (x == 1) break;
        }
        if (x > 1) a[i] += 1;
    }
    
  //  for (int i = 1; i <= n; i++) cout << a[i] << " ";
    
    for (int i = 1; i <= n; i++) 
    {
        x = a[i];
        if (x == 1) cnt++;
        z = z ^ x;
    }
    if (cnt == n) 
    {
       if (n % 2 == 0) cout << "BOB";
       else cout << "ALICE";
    }
    else 
    {    
         if (z > 0) cout << "ALICE";
         else cout << "BOB";
    }
  //  system("PAUSE");
    return 0;
}
