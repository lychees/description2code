import sys

def adda(val):
	nums = val.split(" ")
	sums = 0
	for val in nums:
		sums += int(val) 
	return sums
list1 = []

while True:
	s = sys.stdin.readline()
	if not s:
		break
	list1.append(s)
	
limit = int(list1[0])
i = 1
while (i<=limit):
	n = 2*i
	if(adda(list1[n])%2==0):
		print 'YES'
	else:
		print 'NO'
	i=i+1
