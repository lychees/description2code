#include <iostream>
#include<math.h>
#include<vector>
using namespace std;

void solve(vector<vector<int> >& matrix, int row, int col, int val, int n){
	if(n==1){
		matrix[row][col]=val++;
		matrix[row][col+1]=val++;
		matrix[row+1][col]=val++;
		matrix[row+1][col+1]=val;
	}
	else{
		int p=pow(2, n-1);
		int v=p*p;
		solve(matrix, row, col, val, n-1);
		solve(matrix, row, col+p, val+v, n-1);
		solve(matrix, row+p, col, val+2*v, n-1);
		solve(matrix, row+p, col+p, val+3*v, n-1);
	}
}


int main()
{
	int n;
	cin>>n;
	int size=pow(2, n);
	vector<int> row(size);
	vector<vector<int> > matrix(size, row);
	solve(matrix, 0, 0, 1, n);
	for(int i=0; i<size;i++){
		for(int j=0; j<size; j++)
			cout<<matrix[i][j]<<" "; 
		cout<<endl;
	}
    return 0;
}
