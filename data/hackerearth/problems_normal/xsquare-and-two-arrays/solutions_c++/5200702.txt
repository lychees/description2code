#include <bits/stdc++.h>
using namespace std;

typedef long long  ll ;
#define end return 0;


int main()
{
    /*freopen("input.txt", "r", stdin);
     freopen("output.txt", "w", stdout);*/
    ios::sync_with_stdio(false), cin.tie(NULL);



    int n , q;
    cin >> n >> q ;
    ll a[n+1] , b[n+1] ;
    for(int i=1 ; i<=n ; i++)cin >> a[i] ;
    for(int i=1 ; i<=n ; i++)cin >> b[i] ;
    for(int i=1 ; i<=q ; i++)
    {
        int type  , x , y;
        cin >> type >> x >> y;
        ll ans=0;
        int h=1;
        if(type == 1)
        {
            for(int j=x ; j<=y ; j++)
            {
                if(h==1)
                {
                    ans+=a[j];
                    h=2;
                }
                else
                {
                    ans+=b[j];
                    h=1;
                }
            }

            cout << ans<<endl;

        }

        if(type == 2)
        {
            for(int j=x ; j<=y ; j++)
            {
                if(h==1)
                {
                    ans+=b[j];
                    h=2;
                }
                else
                {
                    ans+=a[j];
                    h=1;
                }
            }
            cout << ans<<endl;
        }
    }






    return 0;
}
