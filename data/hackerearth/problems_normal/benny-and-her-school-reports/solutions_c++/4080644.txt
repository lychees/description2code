#include <iostream>
using namespace std;
typedef long long int rt;
int main()
{
    rt t,n,m,x,*a,sum,res;
    cin>>t;
    for(rt tc=0;tc<t;tc++)
    {
    	sum=0;
    	res=0;
    	cin>>n>>m>>x;
    	a=new rt[n];
    	for(rt i=0;i<n;i++)
    	{
    		cin>>a[i];
    		sum+=a[i];
    	}
    	res=(x*(n+1))-sum;
    	if(res>m)
    	{
    		cout<<"Impossible"<<endl;
    		continue;
    	}
    	else
    	{
    		if(res<1)
    			res=1;
    	}
    	cout<<res<<endl;
    	delete a;
    }
    return 0;
}
