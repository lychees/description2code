#include <iostream>
using namespace std;

int main()
{
	int t,i,j;
	cin>>t;
	for(i=0;i<t;i++)
	{
		int a[26]={'0'};
		a[0]=0;
	string unique;
	cin>>unique;
	int l=unique.size();
	for(j=0;j<l;j++)
	{
		if(a[(int)unique[j]-97]==0)
		{
			cout<<unique[j];
			a[(int)unique[j]-97]++;
		}
	}
	cout<<endl;
	
	}
    return 0;
}
