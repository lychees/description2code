#include <iostream>
using namespace std;

int main()
{
    int m,i,j,p=0,q=0,a[100][100];
    cin>>m;
    for(i=0;i<m;i++)
    {
    	for(j=0;j<m;j++)
    	{
    		cin>>a[i][j];
    		if(i==j)
    		p+=a[i][j];
    		if(i+j==m-1)
    		q+=a[i][j];
    	}
    }
    int ans=abs(p-q);
    cout<<ans;
    return 0;
}
