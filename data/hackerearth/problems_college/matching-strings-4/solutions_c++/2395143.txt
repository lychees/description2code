#include <bits/stdc++.h>
using namespace std;

int main()
{
   int t,i,c,l1,l2;
   cin>>t;
   while(t--){
   	c=0;
   	string s1,s2;
   	cin>>s1>>s2;
   	l1=s1.length();
   	l2=s2.length();
   	if(l1>=l2)
   	{
   		for(i=0;i<l2;i++){
   			if(s1[i]==s2[i])
   			c++;
   		}
   	}
   	else
   	{
   			for(i=0;i<l1;i++){
   			if(s1[i]==s2[i])
   			c++;
   		}
   	}
   	cout<<c<<endl;
   }
    return 0;
}
