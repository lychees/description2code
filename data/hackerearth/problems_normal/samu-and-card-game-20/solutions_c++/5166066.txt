#include <iostream>
using namespace std;

int main()
{
    long long t,a,b,x,y,sum,n,m,k;
    cin>>t;
    
    while(t--)
    {cin>>n>>m>>k;
     x=1;
     y=1;
     sum=0;
    for(int i=0;i<k;i++)
    {cin>>a>>b;
    
    while(1)
    {x+=a;
    y+=b;
    	if(x>=1&&x<=n&&y>=1&&y<=m)
     sum++;
     else
     {x-=a;
     y-=b;
     break;
    }
    }
    }
    cout<<sum<<endl;
    }

    
    return 0;
}
