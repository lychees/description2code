#!/usr/bin/env python

def interest():
    x, y, n = map(int, raw_input().split())
    arr = []

    i = 0
    while i < x:
        arr.append(y)
        i += 1
    j = i
    while j < n:
        sm = 0
        i = j - x
        while i < j:
            sm = sm + arr[i]
            i += 1
        arr.append(sm)
        j += 1

    print(arr[-1])


if __name__ == '__main__':
    t = int(raw_input())
    for i in xrange(t):
        interest()
