#include <iostream>
using namespace std;

int count_ways(int n) {
	if(n == 1) return 2;
	if(n <= 0) return 1;
	return (count_ways(n-1)+count_ways(n-2));
}

int total_ways(int n) {
	int ans = count_ways(n);
	ans *= ans;
	return ans;
}

int main() {
	long long int t, n;
	cin>>t;
	while(t--) {
		cin>>n;
		cout<<total_ways(n)<<endl;
	}
	return 0;
}
