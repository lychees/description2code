#include <iostream>
using namespace std;

int main()
{
	int T;
	cin >> T;
	while (T--){
		int w, n;
	
	cin >> w >> n;
	int a[n], x[n]; 
	long long l = 0;
	x[0] = 0;
	for (int i = 0; i <n; i++)
	{
		cin >> a[i];
	}
	for (int i = 1; i <=n; i++)
	{
		
		if (a[i-1] == 1)
		{
			x[i] = 2*i;
			int h = x[i] - x[i-1];
			l = h + w + l;
		}
		else
		{
			x[i] = x [i-1]+1;
			int h = x[i] - x[i-1];
			l = h +w +l;
		}
	}
	cout <<l<<endl;
}
return 0;
}
