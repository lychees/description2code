#include<bits/stdc++.h>
#include <iostream>
using namespace std;

int main()
{
 int t;
 scanf("%d",&t);
 while(t--){
 	int x1,y1,x2,y2;
 	scanf("%d%d%d%d",&x1,&y1,&x2,&y2);
 	if(((x1+y1)+(x2+y2))%2==1)
 	printf("-1\n");
 	//(y1-y2)/(x1-x2) is 1 or -1 if x1,y1 and x2,y2 is diagonal
 	else if((int)abs(x1-x2) == (int)abs(y1-y2))
 			printf("0\n");
 			else
 			printf("1\n");
 }
 //   cout << "Hello World!" << endl;
    return 0;
}
