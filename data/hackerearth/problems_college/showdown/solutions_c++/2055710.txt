#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll long long int
#define pb push_back
#define mk make_pair
ll power(ll a, ll b) {
ll x = 1, y = a;
    while(b > 0) {
        if(b%2 == 1) {
            x=(x*y);
            if(x>mod) x%=mod;
        }
        y = (y*y);
        if(y>mod) y%=mod;
        b /= 2;
    }
    return x;
}
int n,a[1001],b[1001];
int play(int i, int j)
{
	int s = 0;
	while(i < n && j < n) {
		if(a[i] > b[j]) {
			s += 100;
			i++;
			j++;
		}
		else if(a[i] < b[j]) {
			j++;
		}
		else {
			int x = s+50+play(i+1,j+1);
			int y = s+play(i,j+1);
			if(x >= y) {
				s += 50;
				i++;
				j++;
			}
			else {
				j++;
			}
		}
	}
	return s;
}
int main()
{
	int i;
	cin>>n;
	for(i = 0; i < n; i++) cin>>a[i];
	for(i = 0; i < n; i++) cin>>b[i];
	sort(a,a+n);
	reverse(a,a+n);
	sort(b,b+n);
	reverse(b,b+n);
	int s = play(0,0);
	cout<<s<<endl;
	return 0;
}
