#include <iostream>
using namespace std;
long long memory[1000][1000];
void search(int M, int N, long long x) {
    int i=0;
   	int j = N-1;
   	while(i<M && j>=0) {
		if(memory[i][j] == x) {
   			cout<<i<<" "<<j<<endl;
   			return;
    	} else {
	    	if(memory[i][j] > x) {
    			j--;
    		} else {
	    		i++;
    		}
    	}
   	}
    cout<<"-1 -1\n";
}

int main() 
{
    int M, N, Q, i, j;
    long long x;
    cin>>M>>N; 
    for(i=0; i<M; i++) {
    	for(j=0; j<N; j++) {
    		cin>>memory[i][j];
    	}
    }
    cin>>Q;
    while(Q--) {
    	cin>>x;
    	search(M, N, x);
    }
    return 0;
}
