#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    string s;
    cin>>s;
    int count,i,j,l;
    l=s.length();
    for(i=0;i<l;i++)
    {
    	s[i]=s[i]-48;
    }
    for(i=0;i<l;i++)
    {
    	count=0;
    	for(j=i;j<l;j++)
    	{
    		if(s[j]%2==0)
    		count++;
    	}
    	cout<<(count)<<" ";
    }
    return 0;
}
