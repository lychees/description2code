void print(long long unsigned k)
{
	if(k==0){
		cout<<endl;
	}
	else {
		if(k%3==0){
			print(--k/3);
			cout<<4;
		}
		else {	
			print(k/3);
			cout<<k%3+1;
		}
	}
}
