t = int(raw_input())
for i in range(t):
    x, y = raw_input().split()
    x = int(x)
    l = []
    for j in range(x):
        l.append(raw_input())
    l.sort()
    flag = True
    for j in range(x-1):
        if (l[j] == l[j+1]):
            flag = False
            break
    if flag:
        print("Yes")
    else:
        print("No")
