#include<cstdio>
#include<algorithm>
using namespace std;

int main(){
    int t;
    scanf("%d", &t);
    while(t--){
        int n, k, p;
        scanf("%d %d %d", &n, &k, &p);
        int ans = p;
        int a[100001] = {0};
        for(int i=0;i<k;i++){
            scanf("%d", &a[i]);
        }
        sort(a, a+k);
        for(int i=0;i<k;i++){
            if(a[i] <= p){
                p++;
            }
        }
        printf("%d\n", p<=n? p:-1);
    }
}
