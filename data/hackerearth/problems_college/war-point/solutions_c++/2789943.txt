long long int mod_pow(long long int a, long long int b,long long int mod)
{
	long long int ans=1;
	while(b)
	{
		int r=b%2;
	
		if(r!=0)
		ans=(ans*a)%mod;
		
		a=(a*a)%mod;
		
		b/=2;
	}	
	return ans;
}

long long gcdf(long long a,long long b)
{
	if(a==0)
	return b;
	return gcdf(b%a,a);	
}

long long min1(long long a,long long b)
{
	return a>b?b:a; 	
}

	long long int t;

	for(;t>0;t--)
	{
		long long int n;
		scanf("%lld",&n);
		
		long long int a[n],sum[n],val=0;
		for(int i=0;i<n;i++)
		{
			scanf("%lld",&a[i]);
			
			val+=a[i];
			if(i==0)
			sum[i]=a[i];
			else
			sum[i] =sum[i-1]+a[i];
		}
		
		long long int min=INT_MAX,m1,m2;
		for(int i=0;i<n;i++)
		{
			m1=sum[i];

			//printf("%lld %lld\n",m1,m2);

}
