#include <iostream>
#include <vector>
#include <limits.h>
#include <math.h>

using namespace std;

int main()
{
    double N, H;
    cin >> N >> H;
    vector<double> A(N);
    double sum = 0;
    for(int i=0; i<N; ++i) {
        cin >> A[i];
        sum += A[i];
    }
    for(int i=1; i<N; ++i) A[i] += A[i-1];

    double pA = (double)N*(double)N;
    double res = 1e300;
    for(double i=0; i<N; ++i) {
        double pB = double(N) * (2*(i+1) + 1) + 2 * sum;
        double pC = (i+1)*(i+2) + 2*A[i] - (double)H*2.0;
        double pD = pB*pB - 4*pA*pC;
        if(pD >= 0) {
            double r = (-pB - sqrt(pD)) / (2*pA);
            if(r >= 0) res = min(res, i+1);
            else {
                r = (-pB + sqrt(pD) ) / (2*pA);
                if(r < 0) r = 0;
                r = ceil(r);
                res = min(res, r*N + i+1);
            }
        }
        else res = min(res, i+1);

    }
    cout << (unsigned long long)res << endl;
    
    return 0;
}
