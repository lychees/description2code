#include <bits/stdc++.h>
using namespace std;
int main()
{
    string s[10];
    s[0]="_0";
    s[1]=".,?!1";
	s[2]="abc2";
	s[3]="def3";
	s[4]="ghi4";
	s[5]="jkl5";
	s[6]="mno6";
	s[7]="pqrs7";
	s[8]="tuv8";
	s[9]="wxyz9";
	int count;
	int t;
	string str;
	cin>>t;
	getchar();
	while(t--)
	{
		getline(cin,str);
		count=0;
		//cout<<str<<endl;
		int curr=1;
		int len=str.length();
		for(int i=0;i<len;i++)
		{
			size_t found=s[curr].find(str[i]);
			if(found==string::npos)
			{
				count++;
				for(int j=0;j<10;j++)
				{
					size_t find=s[j].find(str[i]);
					if(find!=string::npos)
					{
						count=count+find+1;
						curr=j;
						break;
					}
				}
			}
			else
			{
				count=count+s[curr].find(str[i])+1;
			}
		}
		cout<<count<<endl;
	}
}
