#include<bits/stdc++.h>
using namespace std;
#define FOR(i,N) for(int i=0;i<(N);i++)
const int maxn = 1000001;
bool  arr[maxn] = {false};
vector<int> sol;

void siv(){
	for(int i=2;i<=sqrt(maxn);i++){
		if(arr[i] == false){
			for(int j=i*i;j<maxn;j +=i)
				arr[j] = true;
		}
	}
		

}

int main(){
	siv();
	for(int i=2;i<maxn;i++){
		if(arr[i] == false){
			sol.push_back(i);
		}
	}
	int t; cin>>t;
	while(t--){
		int n; cin>>n;
		cout<<"2^";
		int m = n;
		for(int i=0; n > 1 && sol[i] < m ; i++){
			int count = 0;
			while( n>=sol[i] && n%sol[i] == 0){
				count++;
				n = n/sol[i];	
			}
			if(i ==0 ) cout<<count;
			else if(count>0) cout<<"*"<<sol[i]<<"^"<<count;
			
		}
		cout<<endl;
	

}

return 0;
}
