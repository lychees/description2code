
#include <iostream>
using namespace std ;
int main()
{

   long long  int t ;
     cin >> t ;
    while ( t--)
    {
       long long  int n ;
       cin >> n ;
        long long int i ;
       long long  int j ;
      long long   int a[10000] ;
        for ( i = 0; i< n ; i++)
        {
            cin >> a[i] ;
        }
        long long int  r = 0 ;
        for ( i = 0; i < n ; i++)
        {
            for ( j = 0; j < n ; j++)
            {
                if ( (j >=i) && ( a[i]== a[j]))
                {
                    r = r+1 ;
                }
            }
        }
        cout << r << endl;
    }
    return 0 ;
}
