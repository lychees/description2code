#include<iostream>
#include<algorithm>
#include<string.h>
using namespace std;
int a[200005];
int b[200005];
int main()
{
	
	int n,i,j;
	long long int tot=0;
	scanf("%d",&n);
	for(i=0;i<n;i++) {
		scanf("%d",&a[i]);
	}
	for(i=0;i<n;i++) {
		scanf("%d",&b[i]);
	}
	//sort(a,a+n);
	//sort(b,b+n);
	i=0;
	j=0;
	while(i<n&&j<n) {
		if(a[i]>b[j]&&i<j) {
			tot=tot+(j-i);
			j++;
		}
		else if(a[i]>b[j]&&i>=j) {
			j=i+1;
		}
		else if(a[i]<=b[j]) {
			i++;
		}
	}
	int flag=0;
	for(i=2;i*i<=tot;i++) {
		if(tot%i==0) {
			flag=1;
			break;
		}
	}
	if(tot==0||tot==1) {
		flag=1;
	}
	if(flag==0) {
		printf("MAGIC INVERSION\n");
	}
	else {
		printf("SIMPLE INVERSION\n");
	}
	return 0;
}
