#include<iostream>
using namespace std;
int find(string,string,int,int,int);
int main()
{
	int testcase;
	cin>>testcase;
	int i,vfind=0,moves,len1,len2;
	string S1,S2;
	for(i=1;i<=testcase;i++)
	{
		cin>>S1;
		cin>>S2;
		cin>>moves;
		len1=S1.length();
		len2=S2.length();
		vfind=find( S1 , S2 , len1, len2,0);
		if(moves<=vfind)
		{
	    (vfind==moves)?cout<<"Yes\n":cout<<"No\n";
		}
		if(moves>vfind)
		{
			int a;
			a=moves-vfind;
			(a%2==0)?cout<<"Yes\n":cout<<"No\n";
			
		}
	}
	return 0;
}
int find(string S1 , string S2 , int len1 ,int len2,int total)
	{
  	if(len1!=len2)
		{	
			return -1;
		}
	if (S1[len1] == S2[len2] && len1>=0 && len2>=0)
		{
	   		return find(S1,S2, len1-1, len2-1,total);
		}
	if (S1[len1]!= S2[len2] && len1>=0 && len2>=0)
		{
	    total++;
	  	return find(S1, S2, len1-1, len2-1,total);
		}
	return total; 
	}
