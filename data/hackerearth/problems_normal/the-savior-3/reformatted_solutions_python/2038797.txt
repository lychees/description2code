n = int(raw_input())
i = 0
while (i < n):
    m = []
    k = int(raw_input())
    l = raw_input().split()
    for j in l:
        m.append(int(j))
    count = 0
    for on in range(0, k):
        for lm in range(on, k):
            if ((m[on] + m[lm]) % 2 == 0 and m[on] != m[lm]):
                count += 1

    print count
    i += 1
