t = input()
while t > 0:
    v = 0
    str = raw_input()
    rev = str[::-1]
    i = 0
    j = 0
    while (j < (len(rev)-1)) and (i < (len(str)-1)):
        if (abs((ord(str[i+1])-ord(str[i]))) !=
                abs((ord(rev[j+1])-ord(rev[j])))):
            v = 1
            break
        i += 1
        j += 1
    if v == 0:
        print "Funny"
    else:
        print "Not Funny"

    t -= 1
