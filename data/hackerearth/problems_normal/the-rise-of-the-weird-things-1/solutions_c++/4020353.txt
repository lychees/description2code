#include <iostream>
#define MAX 1000
using namespace std;

int partition(int a[],int p,int r)

{

    
    
    
    int pivot=a[r];

                   int k=p-1;

    for(int i=p;i<r;i++)

    {
             if(pivot>a[i])

        {
                k=k+1;
                   int l=a[k];
                   a[k]=a[i];
                   a[i]=l;
   
     } 

        }
                 int l=a[k+1];
                 a[k+1]=a[r];
                   a[r]=l;
                   return k+1;

 }



 



void quicksort(int a[],int l,int m)

{
    
    if(l<m)
   
          {

                    int q;
     
                    q=partition(a,l,m);
 
                    quicksort(a,l,q-1);

                    quicksort(a,q+1,m);

                  }
 
}



int main()
{
    int n;
    cin>>n;
    int a[MAX];
    
    for(int i=0;i<n;i++)
    cin>>a[i];
    
    int k=0;
    for(int i=0;i<n;i++)
    {
        if(a[i]%2==0)
        {
            int t=a[i];
            a[i]=a[k];
            a[k]=t;
            k++;
        }
    }

quicksort(a,0,k-1);
quicksort(a,k,n-1);

int sum=0;
    
    for(int i=0;i<=k-1;i++)
    {
    cout<<a[i]<<" ";
    sum+=a[i];
    }
    cout<<sum<<" ";
    sum=0;
    for(int i=k;i<=n-1;i++)
    {
        cout<<a[i]<<" ";
        sum+=a[i];
    }
    cout<<sum;
    
    return 0;
    
}
