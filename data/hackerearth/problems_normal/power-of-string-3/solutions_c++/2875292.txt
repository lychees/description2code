#include <bits/stdc++.h>

using namespace std;

#define fpos adla
#define link fcytkrjf
const int inf = 1e9;
const int maxn = 2e6;
char s[maxn];
int ans = 0 ; 
int to[maxn][27];
int len[maxn], fpos[maxn], link[maxn];
int node, pos;
int sz = 1, n = 0;
long long K , N; 

int make_node(int _pos, int _len)
{
    fpos[sz] = _pos;
    len [sz] = _len;
    return sz++;
}

void go_edge()
{
    while(pos > len[to[node][s[n - pos]]])
    {
        node = to[node][s[n - pos]];
        pos -= len[node];
    }
}

void add_letter(int c)
{
    c -= 'a' ;
    s[n++] = c;
    pos++;
    int last = 0;
    while(pos > 0)
    {
        go_edge();
        int edge = s[n - pos];
        int &v = to[node][edge];
        int t = s[fpos[v] + pos - 1];
        if(v == 0)
        {
            v = make_node(n - pos, inf);
            link[last] = node;
            last = 0;
        }
        else if(t == c)
        {
            link[last] = node;
            return;
        }
        else
        {
            int u = make_node(fpos[v], pos - 1);
            to[u][c] = make_node(n - 1, inf);
            to[u][t] = v;
            fpos[v] += pos - 1;
            len [v] -= pos - 1;
            v = u;
            link[last] = u;
            last = u;
        }
        if(node == 0)
            pos--;
        else
            node = link[node];
    }
}




int dfs(int r , int dep){
    bool isleaf = true ;
    int cnt = 0;
    for(int i = 0 ; i < 27 ; ++i){
        if(to[r][i]){
            isleaf = false; 
            cnt += dfs(to[r][i],dep+min(len[to[r][i]] , n - 1 - fpos[to[r][i]]));
        }
    }
    if(isleaf)
        ++cnt;
    if(cnt >= K){
        ans=max(ans,dep);
    }
   // cout << r << " " << dep << " " << cnt << "\n" ; 
    return cnt;
}






int main()
{
    //freopen("input.txt","r",stdin);
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    len[0] = inf;
    string s;
    cin >> K >> N ;
    cin >> s; 
    s += ('z'+1) ; 
    for(int i = 0; i < s.size(); i++)
        add_letter(s[i]);
    dfs(0,0);
    cout << ans << "\n" ; 
    return 0 ;
}
