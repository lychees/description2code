#include <iostream>
#include<algorithm>
using namespace std;
long long sumval = 0;
void countr(int a[],int n,int sum1,int sum2,int i)
{
 if(i == n)
 	sumval++;
 else
 	{
 		if((a[i] + sum1) <= sum2)
 		{
 			countr(a,n,sum1+a[i],sum2 ,i+1);
 		}
 		countr(a,n,sum1,sum2+a[i],i+1);
 			
 	} 	
}

void permutateandcnt(int a[],int l,int r)
{
	if(l == r) 
		{
			countr(a,r,0,0,0);
		}
	else
	{
		for(int i =l ;i<r;i++)
		{
		 swap(a[l], a[i]);
		 permutateandcnt(a,l+1,r);
		 swap(a[l],a[i]);
		}
	}
}
int main()
{
    int n;
    cin>>n;
    int a[n];
    int sum = 0;
    for(int i =0;i<n;i++)
    	{ cin>>a[i]; sum += a[i];}
    permutateandcnt(a,0,n);
    cout<<"\n"<<sumval<<" "<<sum;
    if(sumval<=sum) cout<<"\n"<<"Got no way out!";
    else
    cout<<"\nWe will win!";
}
