def solve(a,turn):
	flag=0
	if a[0] == 0 and a[1] == 0 and a[2] == 0:
		return abs(1-turn)
	for i in xrange(3):
		if a[i]%2 !=0:
			a[i]-=1
			flag=1
	if flag:
		return solve(a,abs(1-turn))
	for i in xrange(3):
		if a[i] != 0:
			a[i]-=1
	return solve(a,abs(1-turn))	
		
	
	
	
for _ in xrange(input()):
	a=map(int,raw_input().split(" "))
	turn=0
	ans=solve(a,turn)
	if(ans == 0):
		print "Ashima"
	else:
		print "Aishwarya"
