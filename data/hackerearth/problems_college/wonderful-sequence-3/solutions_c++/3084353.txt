#include <bits/stdc++.h>
using namespace std;
long long v[100001];
int main()
{
    long long n,a,prev=-1,ans=0,t,r,x;
    cin>>t;
    while(t--)
    {
    	
    	cin>>n;
    	r=0;
    	for(int i=0;i<n;i++)
    	{
    		cin>>x;
    		x -= i;
	        if(x <= 0)
	        {
	            continue;
	        }
        	long long loc = upper_bound(v, v+r, x) - v;
        	v[loc] = x;
        	r = max(loc + 1, r);
    	}
    	cout<<n-r<<endl;
    }
    return 0;
}
