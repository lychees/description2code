#include <bits/stdc++.h>

using namespace std;

int main(){
	int t;
	cin >> t;
	while(t--){
	int n,k;
	cin >> k >> n;
	int table[n+1][k+1];
	int i,j,x;
	int res;
	for(i=1;i<=n;i++){
		table[i][0] = 0;
		table[i][1] = 1; 
	}

	for(j=1;j<=k;j++){
		table[1][j] = j;
	}

	for(i=2;i<=n;i++){
		for(j=2;j<=k;j++){
			table[i][j] =100000;
			for(x = 1; x<=j;x++){
				res = 1 + max(table[i-1][x-1],table[i][j-x]);
				if(res< table[i][j])
					table[i][j] = res;
			}

		}
	}
	cout << table[n][k] << endl;
}
}
