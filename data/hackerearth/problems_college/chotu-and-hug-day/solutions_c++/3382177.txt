#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int N, H, A, t, min;
    cin>>t;
    while(t--)
    {
    	cin>>N>>H;
    	min = 10000;
    	for(int i=0; i<N; i++){
    		cin>>A;
    		if(abs(A-H) < min){
    			min = abs(A-H);
    		}
    	}
    	cout<<min<<endl;
    }
    return 0;
}
