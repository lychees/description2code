    #include <iostream>
    using namespace std;
    int main()
    {
    int t;
    cin>>t;
    while(t--)
    {
    int n;
    cin>>n;
    int temp,cnt=0;
    for(int i=0;i<n;i++)
    {
    cin>>temp;
    if(temp%2==1)
    cnt+=temp;
    }
    cout<<cnt<<endl;
   
    }
    }
