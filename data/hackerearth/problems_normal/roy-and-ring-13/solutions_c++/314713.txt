#include<cstdio>
#include<algorithm>
#include<map>
using namespace std;
const int MOD=1000000007;
map<long long,long long> M;
long long f(long long n)
{
    if(n==1) return 1;
    if(n<=0) return 0;
    if(M.count(n)) return M[n];
    if((n%2)==0){
        long long ans=(2*f((n/2)-1)+f(n/2))%MOD;
        ans=(ans*f(n/2))%MOD;
        return M[n]=ans;
    } else
    {
        long long z=(n+1)/2;
        long long ans=f(z)*f(z);
        ans%=MOD;
        long long ans1=f(n/2)*f(n/2);
        ans1%=MOD;
        return M[n]=(ans+ans1)%MOD;
    }
}

int main()
{
    int Q;
    scanf(" %d",&Q);
    while(Q--){
        int N;
        scanf(" %d",&N);
        N=(N+1)/2;
        long long ans=f(N);
        ans=(ans*ans)%MOD;
        printf("%lld\n",ans);
    }
    return 0;
}
