#include <bits/stdc++.h>
using namespace std;
long long a[1000001];
int f[9]={1,5,25,125,625,3125,15625,78125,390625};
int main()
{
    for(int i=1;i<=1000000;++i)
    {
    	for(int j=1;j<=8;++j)
    	{
    		a[i]+=i/f[j];
    		
    	}a[i]+=a[i-1];
    	
    }
    ios_base::sync_with_stdio(false);cin.tie(0);
    int t;
    cin>>t;
    while(t--)
    {
    	int x,y;
    	cin>>x>>y;
    	cout<<a[y]-a[x-1]<<'\n';
    }
    
    return 0;
}
