#include <iostream>
#include<ctype.h>
#include<string>
using namespace std;

int main()
{
	string s;
	int t,count=-1;
	long long int a[1000];
	cin>>t;
	while(t--)
	{
		cin>>s;
		for(int i=0;i<s.size();i++)
		{
			for(char j='a';j<='z';j++)
			{
				count++;
				if(s[i]==j)
				{
					if(i==0)
					{
						a[0]=count;
						if(a[0]>13)
						cout<<a[0]-26;
						else
						cout<<a[0];
					}
					else
					{
						a[i]=count;
						if((a[i]-a[i-1])>13 || (a[i]-a[i-1])<-13)
						{
							if((a[i]-a[i-1])>13)
							cout<<' '<<a[i]-a[i-1]-26;
							else
							cout<<' '<<a[i]-a[i-1]+26;
						}
						else
						{
							if((a[i]-a[i-1])==-13)
							{
								cout<<' '<<13;
							}
							else if((a[i]-a[i-1])<=13)
							{
								cout<<' '<<a[i]-a[i-1];
							}
						}
					}
				}
			}
			count=-1;
		}
		cout<<endl;
	}
    return 0;
}
