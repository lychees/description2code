#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
  int t;
  cin>>t;
  while(t--)
  {
  	
  	int n ;
  	cin>>n;
  	vector< pair <pair<int,int>,double > > v(n);
  	double dp[n];
  	
  	for(int i=0;i<n;i++)
  	{
  		
  		cin>>v[i].first.second>>v[i].first.first>>v[i].second;
  		
  	}
  	sort(v.begin(),v.end());
  	dp[0]=v[0].second;

  	for(int i=1;i<n;i++)
  	{
  		int s=v[i].first.second;
  		double value=0;
  		for(int j=i-1;j>=0;j--)
  		{
  			
  			if(v[j].first.first<=s)
  			 {
  			 	value=dp[j];break;
  			 }
  			
  		}
  		dp[i]=max(dp[i-1],value+v[i].second);
  	
  		
  	}
  	
  	printf("%.2lf\n",dp[n-1]/100);
  	
  }
    return 0;
}
