#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    int t,a,b,j;
    cin>>t;
    while(t--)
    {
    	int p,sum=0;
      cin>>a>>b;
      if(a==1)
      a=a+1;
      
     
      
      for(int i=a;i<=b;i++)
      {
      	p=1;
   
      	for( j=2;j<=sqrt(i);j++)
      	{
      	 if(i%j==0)
      	 {
      	 	p=0;
      	 	break;
      	 }
      		
      		
      	}
      	if(p==1)
      	sum=sum+i;
      	
      }
      cout<<sum<<endl;
    }
    return 0;
}
