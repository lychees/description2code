#include <cstdio>
#include <set>
#include <algorithm>
using namespace std;
long long int binsearch(long long int* a, long long int l, long long int h, long long int v){
	if(l<=h){
		long long int mid = (l+h)/2;
		if(a[mid]==v) return 1;
		else if(a[mid]>v) binsearch(a, l, mid-1, v);
		else binsearch(a,mid+1, h,v);
			
	}
	else return 0;
}
int main() {
	long long int n,k;
	scanf("%lld %lld",&n,&k);
	long long int s[n];
	long long int temp;
	for(long long int i=0; i<n; ++i){
		scanf("%lld",&s[i]);
	}
	sort(s, s+n);
	long long int flag = 0;
	for (long long int i = 0; i < n - 2; ++i) {
		for (long long int j = i+1; j < n - 1; ++j) {
			long long int key = k - s[i] - s[j];
			if (binsearch(s, j+1, n-1, key) == 1) {
				flag = 1;
				break;
			}
		}
		if (flag == 1) {
			break;
		}
	}
	if (flag == 1) {
		printf("YES\n");
	} else {
		printf("NO\n");
	}


	return 0;
}
