inline int inpi()
{
    char c = gc();
    int ret = 0, sign = 1;
    while((c<'0' || c>'9') && c!='-') c = gc();
    if(c=='-')
    {
        sign = -1;
        c = gc();
    }
    while(c>='0' && c<='9')
    {
        ret = 10 * ret + c - 48;
        c = gc();
    }
    if(sign<0) ret = -ret;
    return ret;
}
