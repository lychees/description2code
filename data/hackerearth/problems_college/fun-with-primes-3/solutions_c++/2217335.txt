#include <iostream>
using namespace std;

int main()
{
	int check(int);
    int a,n;
    cin>>n;
    for(int i=0;i<n;i++)
    {
    	cin>>a;
    	if(check(a)==1)
    		cout<<"YES"<<endl;
    		else
    		cout<<"NO"<<endl;
    }
    return 0;
}
int check(int b)
{
	int sum=0,found=0;
	while(b>0)
	{
		sum+=b%10;
		b=b/10;
	}
	for(int i=2;i<sum;i++)
	{
		if(sum%i==0)
		{
			found=1;
			break;
		}
	}
	if(sum==2)
	 return 1;
	 if(sum==1)
	  return 0;
	if(found==0)
	 return 1;
	 else
	 return 0;
}
