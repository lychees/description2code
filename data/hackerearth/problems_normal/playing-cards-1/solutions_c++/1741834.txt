#include <bits/stdc++.h>
using namespace std;

int C[1002][1002];
int MOD = 1e9 + 7;

int main(){


    for(int i = 0; i < 1002; i ++){
        C[i][0] = 1;
        for(int j = 1; j <= i; j++) C[i][j] = (C[i-1][j] + C[i-1][j-1]) % MOD;
    }

    int test;
    cin >> test;
    while(test--){

        int N, M;
        cin >> N >> M;


        if(M % 2 != 0){

            cout << 0 << endl;
            continue;
        }

        int rf = 6 * N, bf = rf;
        int rr = 18 * N, br = rr;
        
        M = M / 2;

        long long ans = 0;

        for(int x = 0; x <= M; x++){
            int y = M - x;
            long long ff = C[rf][x];// rf x 
            ff *= C[br][x]; ff %= MOD;
            ff *= C[bf][y]; ff %= MOD;
            ff *= C[rr][y]; ff %= MOD;

            ans = ans + ff;
            ans %= MOD;
        }
        
        cout << ans << endl;

    }



    return 0;
}
