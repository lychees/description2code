#include <iostream>
using namespace std;

int main()
{
	int t;
	long long int n,j;
	cin>>t;
    for(int i=1;i<=t;i++){
    	cin>>n;
    	long long int a[n],b[n];
    	int flag=0;
    	for(j=1;j<=n;j++){
    		cin>>a[j];
    		b[a[j]]=j;
    	}
    	for(int j=1;j<=n;j++){
    		if(a[j]!=b[j]){
    			flag=1;
    			cout<<"not inverse\n";
    			break;
    		}
    	}
    	if(flag==0){cout<<"inverse\n";}
    }
    return 0;
}
