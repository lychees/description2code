#include <bits/stdc++.h>
using namespace std;
vector<pair<long long int,pair<long long int,long long int>> >P;
long double D(pair<long long int,long long int>A,pair<long long int,long long int>B)
{
	return sqrt(((long double)(A.first-B.first)*(A.first-B.first))+((long double)(A.second-B.second)*(A.second-B.second)));
}
long double DP[3001];
int main()
{
    long long int n,x,y,f,i,j;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>x>>y>>f;
    	P.push_back({f,{x,y}});
    }
    DP[0]=P[0].first;
    for(i=1;i<n;i++)
    {
    	DP[i]=-1e9;
    	for(j=0;j<i;j++)
    	DP[i]=max(DP[i],DP[j]-D(P[j].second,P[i].second));
    	DP[i]+=P[i].first;
    }
    cout<<fixed<<setprecision(6)<<DP[n-1]<<"\n";
    return 0;
}
