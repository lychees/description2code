def main():
    test = int(raw_input())
    for i in range(test):
        n, id = map(int, raw_input().split())
        a = []
        a.append(id)
        for j in range(n):
            q = raw_input().split()
            if q[0] == 'P':
                a.append(int(q[1]))
            else:
                a.append(a[-2])
        print('Player '+str(a[-1]))


if __name__ == '__main__':
    main()
