#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    int n;
    cin>>t;
    
    while(t--){
    	cin>>n;
    	long long int a[n+1],k;
    	cin>>k;
    	for(int i = 0; i < n; i++){
    		cin>>a[i];
    	}	
    	long long sum = 0;
    	long long int ans = 0;
    	for(int i = 0; i < n; i++){
    		if(a[i] <= k){
    			sum =sum + a[i];
    		}
    		else{
    			ans = max(ans,sum);
    			sum = 0;
    		}
    	}
    	ans = max(ans,sum);
    	cout<<ans<<endl;
    }
    return 0;
}
