typedef long long ll;

const int mod = 1000000007;
const int N = 3339;

int W[N][N],skos[N][N],pion[N][N];
int main() {
	int nn,L,K,T;
	scanf("%d%d%d%d",&nn,&L,&T,&K);
	W[0][0]=pion[0][0]=skos[0][0]=1;
	for(int n=1;n<nn+1;++n) fru(k,n+1){
		LL s=pion[n-1][k];
		if(n>=L){
			s+=mod-pion[n-L][k];
			if(k) s+=skos[n-L][k-1];
		}
		s=1LL*s%mod*(K-1)%mod;
		if(k==max(n-L+1,0)) ++s;
		s%=mod;

	if(0){	LL brut=0;
		for(int i=1;i<=n;++i) if(k>=max(i-L+1,0))
			brut+=W[n-i][k-max(i-L+1,0)];
		brut%=mod;
		brut=1LL*(K-1)*brut%mod;
		if(k==max(n-L+1,0)) ++brut;
		brut%=mod;
		assert(s==brut); }

		W[n][k]=s;
		pion[n][k]=(s+pion[n-1][k])%mod;
		skos[n][k]=(s+(k?skos[n-1][k-1]:0))%mod;
	}
	printf("%d\n",W[nn][T]);
	return 0;
}
