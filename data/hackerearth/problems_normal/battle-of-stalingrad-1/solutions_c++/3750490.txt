#include <iostream>
#include <stdio.h>
#include <stdlib.h>
#include <math.h>
using namespace std;
#define ll long long

float area(int x1, int y1, int x2, int y2, int x3, int y3)
{
    return fabs((x1*(y2-y3) + x2*(y3-y1)+ x3*(y1-y2))/2.0);
}

bool Inside(int x1,int y1,int x2,int y2,int x3,int y3,int x4,int y4)
{
    float A=area(x1,y1,x2,y2,x3,y3);
    float A1=area(x1,y1,x2,y2,x4,y4);
    float A2=area(x1,y1,x4,y4,x3,y3);
    float A3=area(x4,y4,x2,y2,x3,y3);
    if(A==A1+A2+A3)
        return 1;
    return 0;
}

int main(int argc, const char * argv[])
{
    int t,x1,y1,x2,y2,x3,y3,x4,y4;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d %d %d %d %d %d %d %d",&x1,&y1,&x2,&y2,&x3,&y3,&x4,&y4);
        if(Inside(x1,y1,x2,y2,x3,y3,x4,y4))
            printf("INSIDE\n");
        else
            printf("OUTSIDE\n");
    }
    return 0;
}
