#include <iostream>
#include <string>
#include <vector>
#include <cstdio>
using namespace std;

int main() {
	// your code goes here
	string s;
	char c;
	vector <string> a;
	bool flag=false,yo=false;
	int n,no,array[31],max,maxi;
	cin>>n;
	scanf(" ");
	for(int i=0;i<31;i++)
	{
		array[i]=0;
	}
	for(int i=0;i<n;i++)
	{
		getline(cin,s);
		a.push_back(s);
	}
	for(int i=0;i<a.size();i++)
	{
		//cout<<a.at(i)<<endl;
		flag=false;
        s=a.at(i);
        if(s.at(0)=='G')
            flag=true;
        no=0;
        for(int j=0;j<s.size();j++)
        {
            yo=false;
            if(s.at(j)-'0'>=0 && s.at(j)-'0'<=9)
            {
                yo=true;
                no*=10;
                no=no+(s.at(j)-'0');
            }
            if(no>0 && !yo)
            {
            	//cout<<no<<endl;
                if(flag)
                    array[no]+=2;
                else
                    array[no]++;
                no=0;
            }
        }
        if(no>0)
        {
        	//cout<<"yo"<<no<<endl;
        	if(flag)
                    array[no]+=2;
                else
                    array[no]++;
        }
	}
	max=array[1];
        maxi=1;
        int index=1;
        for(int i=2;i<31;i++)
        {
            if(array[i]>max)
            {
                max=array[i];
                maxi=1;
                index=i;
            }
            else if(array[i]==max)
            {
                maxi++;
            }
        }
        if(maxi>1)
        {
            cout<<"No Date"<<endl;
        }
        else if(index==19 || index==20)
            cout<<"Date"<<endl;
        else
            cout<<"No Date"<<endl;
	return 0;
}
