using namespace std;
typedef long long int ll;

ll n,a[1111],g,f,ans = 1;

#define MOD 1000000007

ll gcd(ll a, ll b)
{
	if(b == 0) return a;
	return gcd(b,a%b);
}

int main()
{
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
	}
	g = a[0];
	for (int i = 1; i < n; i++)
	{
		g = gcd(g,a[i]);
	}
	f = a[0];
	for (int i = 1; i < n; i++)
	{
		f = (f*a[i])%MOD;
	}
	for (int i = 0; i < g; i++)
	{
		ans = (ans*f)%MOD;
	}
	cout<<ans<<" ";
}
