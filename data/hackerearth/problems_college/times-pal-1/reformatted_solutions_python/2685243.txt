t = input()
for tests in range(0, t):
    s = raw_input()
    l = raw_input()
    length = len(s)
    count = 0
    temp = l+s
    if temp == temp[::-1]:
        count += 1
    for i in range(1, length+1):
        temp = s[0:i]+l+s[i:length]
        # print(temp)
        if temp == temp[::-1]:
            count += 1
    print(count)
