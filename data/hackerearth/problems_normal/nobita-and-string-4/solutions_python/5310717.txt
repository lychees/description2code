def reverse(s):
	s = s.split()
	l = len(s)
	for i in range(l/2):
		temp = s[i]
		s[i] = s[l - i - 1]
		s[l - i - 1] = temp
	return ' '.join(s)

n = int(raw_input())
for i in range(n):
	print reverse(raw_input())
