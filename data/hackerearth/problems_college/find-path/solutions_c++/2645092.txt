#include<bits/stdc++.h>
using namespace std;

int main()
{
   long long t;
    cin>>t;
    while(t--)
    {
       long long n;
        cin>>n;
        long long a[n];
        for(int i=0;i<n;i++)cin>>a[i];
        int cu=0;
        for(int i=1;i<n;i++)
        {

            if(a[i] <= a[i-1])cu++;
        }
        cout<<cu<<endl;
    }



    return 0;
}
