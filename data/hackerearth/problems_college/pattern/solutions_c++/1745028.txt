#include <iostream>
#define ll int
using namespace std;

int main()
{
	ll n,i,j;
	cin>>n;
	for(i=n;i>=1;i--)
    {
    	if(i==1||i==n)
    	{
    		for(j=1;j<=n;j++)
    			cout<<"*";
    		cout<<endl;
    	}
    	else
    	{
    		for(j=1;j<=i;j++)
    		{
    			if(j==i)
    				cout<<"*";
    			else
    				cout<<" ";
    		}
    		cout<<endl;
    	}
    }
	return 0;
}
