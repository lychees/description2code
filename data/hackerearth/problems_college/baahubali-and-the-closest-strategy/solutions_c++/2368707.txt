#include <iostream>
#include<algorithm>
#include<limits.h>
using namespace std;

int main()
{
	int n=0,q=0;
	cin>>n>>q;
	int arr[n];
	for(int i=0;i<n;i++)
	{
		cin>>arr[i];
	}
	sort(arr,arr+n);
	while(q-->0)
	{
		int num=0;
		cin>>num;
		int left=0;
		int right=n-1;
		int min=INT_MAX;
		int min_l=0,min_r=0;
		while(left<right)
		{
			int sum=arr[left]+arr[right];
			if(abs(num-sum)<min)
			{
				min=abs(num-sum);
				min_l=left;
				min_r=right;
			}
			if(arr[left]+arr[right]>num)
				right--;
			else
				left++;
		}
		cout<<arr[min_l]+arr[min_r]<<endl;
		
	}
    return 0;
}
