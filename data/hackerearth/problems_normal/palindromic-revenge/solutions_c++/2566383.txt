#include <iostream>
#include<math.h>
#include<cstring>
using namespace std;

int main()
{
    long int t,i,j,l;
    char a[100000];
    cin>>t;
    while(t--)
    {   int count=0;
    	cin>>a;
    	l=strlen(a);
    	for(i=0,j=l-1;i<l/2;i++,j--)
    	{
    		if(a[i]!=a[j])
    		{
    			count+=fabs(a[j]-a[i]);
    		}
    		
    	}
    	cout<<count<<endl;
    }
    return 0;
}
