#include<iostream>
#include<string>
using namespace std;
int main()
{
    int t,i,j,acost,bcost,total;
    string s;
    scanf("%d",&t);
    while(t--)
    {
        cin>>s;
        scanf("%d",&acost);
        scanf("%d",&bcost);
        total=0;
        for(i=0,j=s.length()-1;i<j;i++,j--)
        {            
            if(s[i]=='/' && s[j]=='/')
                {
                         if(acost<bcost)
                            total+=2*acost;
                         else
                             total+=2*bcost;
                }
            else if(s[i]=='/')
            {
                if(s[j]=='a')
                    total+=acost;
                else if (s[j]=='b')
                    total+=bcost;
            }
            else if (s[j]=='/')
            {
                if(s[i]=='a')
                    total+=acost;
                else if (s[i]=='b')
                    total+=bcost;
            }
            else if(s[i]!=s[j])
            {
                 total=-1;
                 break;
            }
        }
        printf("%d\n",total);
    }
    return 0;
}
