#include<bits/stdc++.h>
typedef long long int ll;
using namespace std;
int main()
{
ll n,q,i,a[1000000],l,r,j,addi,subt;
cin>>n>>q;
for(i=0;i<n;i++)
cin>>a[i];
for(i=0;i<q;i++)
{
addi=subt=0;
cin>>l>>r;
for(j=l-1;j<=r-1;j++)
addi+=a[j];
subt=a[r-1];
for(j=r-2;j>=l-1;j--)
subt-=a[j];
cout<<addi<<" "<<subt<<endl;
}
return 0;
}
