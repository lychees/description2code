n = int(raw_input())
v = [int(x) for x in raw_input().split()]

a1, a2, a3 = (0, 0, 0)
lim = n / 3
if 0 != n % 3:
	lim += 1
for i in xrange(0, lim):
	if i * 3 < n:
		a1 += v[i * 3]
	if i * 3 + 1 < n:
		a2 += v[i * 3 + 1]
	if i * 3 + 2 < n:
		a3 += v[i * 3 + 2]
print a1, a2, a3
