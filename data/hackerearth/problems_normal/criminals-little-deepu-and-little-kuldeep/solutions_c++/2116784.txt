#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		long int n;
		cin >> n;
		long int a[n];
		for(int i=0;i<n;i++)
		{
			cin >> a[i];
		}
		sort(a,a+n);
		int ans=1,prev_ans=1;
		for(int i=0;i<n-1;i++)
		{
			if(a[i]==a[i+1])
				ans++;
			else
			{
				if(ans>prev_ans)
					prev_ans=ans;
				ans=1;
			}
		}
		if(ans>prev_ans)
			prev_ans=ans;
		cout << prev_ans << endl;
	}
	return 0;
}
