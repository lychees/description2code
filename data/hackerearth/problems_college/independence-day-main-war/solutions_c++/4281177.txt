#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		float s1=0,s2=0;
		float x[n],y[n];
		for(int i=0;i<n;i++)
		{
			cin>>x[i]>>y[i];
			s1+=x[i];
			s2+=y[i];
		}
		float a=s1/float(n);
		float b=s2/float(n);
			cout<<fixed<<setprecision(2)<<"("<<a<<" "<<","<<" "<<b<<")"<<endl;
	}
}
