import math
number_of_bottles = int(raw_input())
l = [1, 2, 3]
for i in xrange(number_of_bottles):
    bottle_number = int(raw_input())
    if bottle_number in l:
        print "NO"
        continue
    val = True
    for j in xrange(2, int(math.ceil(math.sqrt(bottle_number))) + 1):
        if bottle_number % j == 0:
            val = False
            if (j * j) == bottle_number:
                print "NO"
            else:
                print "YES"
            break
    if val:
        print "NO"
