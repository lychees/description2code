#include<stdio.h>
#define ll long long int
int main(){
	ll a, b, i; scanf("%lld%lld", &a, &b);
	for(i = a + 1; i <= b - 2; i++){
		if(!(i % 5) && !(i % 3))	printf("%lld,", i);
		if(!(i % 5))	continue;
		if(!(i % 3))	continue;
		printf("%lld,", i);
	}
	if(!(i % 5) && !(i % 3)){printf("%lld\n", i); return 0;}
	if(!(i % 5))	return 0;
	if(!(i % 3))	return 0;
	printf("%lld\n", i);	
	return 0;
}
