#include <iostream>
using namespace std;

int main()
{
	long t,a,b,n,j,x,sum,sum1,sum2,c,i;
    cin>>t;
    while(t--){
    	n=0;sum=0;
    	cin>>a;
    	cin>>b;
    	c=a;
    	for(i=c;!(i/10==0&&i%10==0);i/=10)
    	{
    		n++;
		}
		for(x=1,i=1;x<n;x++,i*=10);
		for(x=0;x<n;x++)
    	{
    		sum+=((c%10)*i);
			c=c/10;
			i=i/10;
		}
    	sum1=sum;
    	n=0;sum=0;
    	c=b;
    	for(i=c;!(i/10==0&&i%10==0);i/=10)
    	{
    		n++;
		}
		for(x=1,i=1;x<n;x++,i*=10);
		for(x=0;x<n;x++)
    	{
    		sum+=((c%10)*i);
			c=c/10;
			i=i/10;
		}
    	sum2=sum;
    	n=0;sum=0;
    	c=sum1+sum2;
    	for(i=c;!(i/10==0&&i%10==0);i/=10)
    	{
    		n++;
		}
		for(x=1,i=1;x<n;x++,i*=10);
		for(x=0;x<n;x++)
    	{
    		sum+=((c%10)*i);
			c=c/10;
			i=i/10;
		}
    	cout<<sum<<endl;
    }
    return 0;
}
