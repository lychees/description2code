#define CLEAR(arr) memset(arr,0,sizeof(arr))
#define ini(x) scanf("%d",&x)
#define inll(x) scanf("%I64d",&x)
#define inf(x) scanf("%f",&x)
#define inc(x) scanf("%c",&x)
#define ind(x) scanf("%lf",&x)

using namespace std;
typedef long long ll;
int main()
{
	int t,n,len;
	char str[10100],arr[11000];
	ini(t);
	while(t--)
	{
		ini(n);
		scanf("%s",str);
		len=0;
		for(int i=0;i<n;i++)
		{if(i%2)
			arr[i]='N';
			else
			arr[i]='Y';
		}
		for(int i=0;i<n;i++)
		{
			if(str[i]!=arr[i])
			{
				len++;
			}
		}
		printf("%d\n",min(len,n-len));
	}
	return 0;
}
