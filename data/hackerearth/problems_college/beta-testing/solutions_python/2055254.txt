from collections import OrderedDict

def main():
    memo={}
    x=input()
    a=raw_input().split()
    for i in a:
        if i not in memo:
            memo[i]=1
        else:
            memo[i]+=1
    for w in sorted(memo, key=memo.get, reverse=True):
        print w, memo[w]
    
if __name__=="__main__":
    main()
