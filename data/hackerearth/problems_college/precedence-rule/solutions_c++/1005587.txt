#include<iostream>
#include<algorithm>
#include<vector>

using namespace std;
int main()
{
    int T;
    cin>>T;
    while(T--)
    {
        string name;
        cin>>name;

       long long  int sum=name[name.size()-1]-'0';
        //cout<<" "<<sum<<endl;
        int n=name.size();

        for(int i=name.size();i!=0;i--)
        {


                if(name[i]=='+')
                   { sum=sum+(name[i-1]-'0');
                        //cout<<sum<<endl;
                   }

                   else if(name[i]=='-')
                   {
                     sum=sum-(name[i-1]-'0');
                    // cout<<" after min"<<sum<<endl;
                   }


                  else  if(name[i]=='*')
                   {
                       //cout<<name[i-1]-'0'<<endl;
                       sum=sum*(name[i-1]-'0');
                       //cout<<"after mul"<<sum<<endl;
                   }




        }
        cout<<sum<<endl;






            }}
