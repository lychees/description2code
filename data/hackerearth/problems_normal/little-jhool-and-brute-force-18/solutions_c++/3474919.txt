#include<iostream>

#define ll long long

using namespace std;
int main()
{
  ll a[]={1729,4104,13832,39312,46683,216027,32832,110656,314496,216125,439101,110808,373464,593047,149389,262656,40033,195841,20683,513000,65728,134379,515375,64232,171288,443889,320264,165464,525824,327763,558441,513856,402597,684019};
   ll t;
   cin>>t;
   while(t--)
   {
       ll n,n2=0,flag=0;
       cin>>n;
       for(int i=0;i<34;i++)
        {
            if(a[i]<n)
            {
                if(n2==0)
                    n2=a[i];
                else
                {
                    if(n2<a[i])
                        n2=a[i];
                }
                flag=1;
            }
        }
        if(flag==1)
            cout<<n2<<endl;
        else
            cout<<-1<<endl;
   }

    return 0;
}
