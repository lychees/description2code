#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define pb push_back
#define mod 1000000007

ll pow_mod(ll a, ll b)
{
    ll r = 1;
	while (b > 0)
	{
		if(b % 2 == 1)
			r = (r * a) % mod;
		b = b/2;
		a = (a*a) % mod;
	}
	return r;
}
