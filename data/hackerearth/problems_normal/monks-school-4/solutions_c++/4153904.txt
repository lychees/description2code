#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int main()
{
	long long int n,m,i,a,j;
	cin>>n>>m;
	string str,str1;
	vector<string> t;
	for(i=0;i<n;i++)
	{
		cin>>str;
		t.push_back(str);
	}
	sort(t.begin(),t.end());
	vector<pair<pair<string,long int>,string> > v;
	for(i=0;i<m;i++)
	{
		cin>>str>>str1>>a;
		v.push_back(make_pair(make_pair(str,a),str1));
	}
	sort(v.begin(),v.end());
	i=0;
	for(j=0;j<t.size();j++)
	{
		cout<<t[j]<<endl;;
		while(i<v.size())
		{
			if(v[i].first.first==t[j])
			cout<<v[i].second<<" "<<v[i].first.second<<endl;
			else break;
			i++;
		}
	}
}
