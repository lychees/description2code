#include<bits/stdc++.h>
#include<algorithm>
using namespace std;

#define gc getchar_unlocked

void scanint(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}

int arr[100005];

int main()
{
	int n,i,j,k,l;
	scanint(n);
	for(i=0;i<n;i++) scanint(arr[i]);
	sort(arr,arr+n);
	j=0;
	for(i=0;i<n;i++)
	{
	    printf("%d ",arr[i]-j);
	    j++;
	}
	printf("\n");
}
