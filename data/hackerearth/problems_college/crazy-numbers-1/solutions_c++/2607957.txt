#include <bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;
ll int dp[1000010][10];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int i,j;
	for(i=0;i<10;i++)
	dp[1][i]=1;
	for(i=2;i<1000005;i++)
	{
		for(j=0;j<10;j++)
		{
			
			if(j==0)
			{
				dp[i][j]=0;
			}
			else if(j==1 && i==2)
			{
				dp[i][j]=(dp[i-1][j+1]+1)%mod;
			}
			else if(j==1)
			{
				dp[i][j]=(dp[i-1][j+1] +dp[i-2][1])%mod;
			}
			else
			{
				dp[i][j]=(dp[i-1][j-1] +dp[i-1][j+1])%mod;
			}
		}
	}
	ll int test;
	cin>>test;
	while(test--)
	{
		ll int n;
		cin>>n;
		ll int i,ans=0;
		for(i=0;i<10;i++)
		ans=(ans+dp[n][i])%mod;
		cout<<ans<<endl;
	}
	return 0;
}
