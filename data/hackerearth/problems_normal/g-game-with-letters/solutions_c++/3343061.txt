#include<bits/stdc++.h>
using namespace std;
int  arr[256];
int main()
{
	
	int t,n,ind,ans,len,i,j,k,l,cnt;
	string str,str1;
	cin>>t;
	
	
	while(t--)
	{
		cin>>n;
		cin>>str;
		memset(arr,false,sizeof(arr));
		l=str.length();
		for(i=0;i<l;i++)
		arr[str[i]]++;
		ans=ind=0;
		len=100000000;
		for(i=1;i<=n;i++)
		{
			cin>>str;
			l=str.length();
			cnt=0;
			for(j=0;j<l;j++)
			{
				cnt+=(arr[str[j]]);
			}
			if(ans<cnt)
			{
				ans=cnt;
				ind=i;
				len=l;
				
			}
			else if(ans==cnt)
			{
				if(len>l)
				{
					ind=i;
					len=l;
				}
				
			}
			
		}
		cout<<ind<<endl;
	}
}
