#include<iostream>
using namespace std;
int main()
{
    long long int a[100005],l,r,n,q;
    cin>>n;
    for(long long int i=1;i<=n;i++)
        cin>>a[i];
    cin>>q;
    while(q--)
    {
        cin>>l>>r;
        long long int sum=0;
        for(long long int j=l;j<=r;j++)
            sum += a[j];
        cout<<sum<<endl;
    }
    return 0;
}
