#include<algorithm>
#include<iostream>
#include<cstring>

using namespace std;

int main(int argc, char const *argv[]) {

  int T, ins, length;
  int freq[26];
  string str;

  cin >> T;

  while(T--) {
    std::cin >> str;

    length = str.size();

    memset(freq, 0, sizeof(freq));

    for (int i = 0; i < length; i++)
      freq[str[i] - 'a']++;

      ins = 0;

    for (int i = 0; i < 26; i++)
      ins += freq[i] % 2;
      
  --ins;
  ins = max(ins, 0);

  cout << ins << endl;

  }

  return 0;
}
