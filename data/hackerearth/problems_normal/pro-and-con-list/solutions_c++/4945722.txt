#include <iostream>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while (t--)
	{
		int n;
		cin >> n;
		long maxab = 0, max2ab = 0, result = 0;
		while (n--)
		{
			long a, b;
			cin >> a >> b;
			if (a + b>maxab)
			{
				result -= max2ab;
				max2ab = maxab;
				maxab = a + b;
				result += a;
			}
			else if (a + b>max2ab)
			{
				result -= max2ab;
				max2ab = a + b;
				result += a;
			}
			else
				result -= b;
		}
		cout << result << endl;
	}
	return 0;
}
