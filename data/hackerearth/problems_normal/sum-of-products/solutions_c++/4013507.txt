#include<iostream>
#include<algorithm>
#define MOD 1000000007

using namespace std;

long long power(long long a,long long b) {

    if(b == 0) return 1;
    if(b == 1) return a;
    long long temp;
    temp = power(a,b/2);
    if(b % 2 == 0) {
        return ((temp % MOD) * (temp % MOD)) % MOD;
    }
    else {
        return (((a % MOD) * (temp % MOD) % MOD) * (temp % MOD)) % MOD;
    }

}


int main(){

    int t;
    cin >> t;
    while(t--){
        long long n,i,x=0,ans=0,y=0,cnt=0,x1,x2,temp;
        cin >> n;
        temp = power(2,MOD - 2);
        for(i = 1 ; i * i <= n ; i++) {
            x = n / i;
            ans = (ans + ((x * i) % MOD)) % MOD;
            if(n / i == i) continue;
            y = n / (i + 1);
            x1 = ((((y % MOD) * ((y + 1) % MOD)) % MOD) * (temp)) % MOD;
            x2 = ((((x % MOD) * ((x + 1) % MOD)) % MOD) * (temp)) % MOD;
            cnt = ((x2 - x1 + MOD) * i) % MOD;
            ans = (ans + cnt) % MOD;
        }
        cout << ans << "\n";
    }
    return 0;







}
