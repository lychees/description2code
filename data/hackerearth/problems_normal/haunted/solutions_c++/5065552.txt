#include <iostream>
#include <map>
using namespace std;

int main()
{
    int n,m,i,max=0,val,age,temp;
    map <int,int> mp;
    cin>>n>>m;
    for(i=0;i<n;i++)
    {
    	cin>>temp;
    	mp[temp]++;
    	val=mp[temp];
    	if(val>max)
    	{
    		age=temp;
    		max=val;
    		cout<<age<<" "<<max<<endl;
    	}
    	else if(val==max)
    	{
    		if(temp>age)
    		age=temp;
    		cout<<age<<" "<<max<<endl;
    	}
    	else
    	{
    		cout<<age<<" "<<max<<endl;
    	}
    	
    }
    return 0;
}
