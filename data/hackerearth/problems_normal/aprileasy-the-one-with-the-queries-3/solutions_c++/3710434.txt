#include <iostream>
#include <vector>

using namespace std;

vector <int> a[30];

int main()
{
	int q, lo, hi;
	string s;
    cin >> s;
    
    for (int i = 0; i < s.length(); i++) {
    	a[s[i] - 'a'].push_back(i);
    }
    
    cin >> q;
    
    int last = -1;
    
    while (q--) {
    	cin >> s;
    	last = -1;
    	
    	for (int i = 0; i < s.length(); i++) {
    		int j = s[i] - 'a';
    		
    		lo = 0;
    		hi = a[j].size() - 1;
    		
    		int ans = -1;
    		
    		while (lo <= hi) {
    			int m = (lo + hi) / 2;
    			
    			if (a[j][m] > last ) {
    				ans = a[j][m];		
    				hi =  m - 1;
    			}
    			else {
    				lo = m + 1;
    			}
    		}
    		
    		if (ans == -1) {
    			cout << "No\n";
    			goto out;
    		}
    		else 
    			last = ans;
    	}
    	
    	cout << "Yes\n";
    	
    	out:
    	;
    	
    }
   
    
    
    return 0;
}
