#include <iostream>
#include <vector>
#include<cstdio>
#define MAX 1001
 
using namespace std;
 
vector <int> adjL[MAX];
bool vis[MAX];
int L[MAX];
int Queue[MAX];
 
void bfs(int s)
{
    int f=0; int r = 0;
    vis[s] = true;
    L[s] = 0;
    Queue[r++] = s;
 
 
    while(f!=r)
    {
        int u = Queue[f];
        f = (f+1)%MAX;
        int l = L[u];
 
        for(int i=0; i<adjL[u].size(); i++)
        {
            if(!vis[adjL[u][i]])
            {
                vis[adjL[u][i]] = true;
                L[adjL[u][i]] = l+1;
                Queue[r] = adjL[u][i];
                r = (r+1)%MAX;
            }
        }
 
    }
 
}
 
int main()
{
    int N; int M;
    scanf("%d",&N);
    scanf("%d",&M);
 
    int u,v;
    for(int i=1; i<=M; i++)
    {
        scanf("%d",&u);
        scanf("%d",&v);
        adjL[u].push_back(v);
        adjL[v].push_back(u);
    }
    for(int i=0; i<N; i++)
    {
        vis[i] = false;
        L[i] = -1;
    }
 
    //for(int i=0; i<N; i++) if(!vis[i]) 
	bfs(0);
 
    for(int i=1; i<= N-1; i++)
    {
        printf("%d\n",L[i]);
    }
 
    return 0;
}
