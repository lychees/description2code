#include <bits/stdc++.h>
using namespace std;

#define MAX 1000

vector < pair<int, int> > v(MAX);
vector <int> ans(MAX);

bool comp(pair<int,int> a,pair<int,int> b){
    if(a.first == b.first) a.second<b.second;
    return a.first<b.first;
}

int main(){
    int t,n,i,temp,j,_max;

    scanf("%d",&t);

    while(t--){
        scanf("%d",&n);

        for(i=0;i<n;++i){
            scanf("%d",&temp);
            v[i] = make_pair(temp,0);
        }

        for(i=0;i<n;++i){
            scanf("%d",&temp);
            v[i].second = temp;
        }

        sort(v.begin(),v.begin()+n,comp);

        for(i=0;i<n;++i)
            ans[i] = 1;

        for(i=1;i<n;++i){
            _max = 0;
            for(j=0;j<i;++j){
                if(v[j].first<v[i].first && v[j].second<v[i].second){
                    if(_max<ans[j])
                        _max = ans[j];
                }
            }
            ans[i]+=_max;
        }

        _max = 0;
        for(i=1;i<n;++i){
            if(ans[_max]<ans[i])
                _max = i;
        }
        printf("%d\n",ans[_max]);
    }

    return 0;
}
