#include <iostream>
#include <bits/stdc++.h>
#define INF 1000000007

using namespace std;

int main()
{
    int t;
    cin >> t;
    
    while(t--)
    {
    	int n;
    	cin >> n;
    	
    	long long int ans = 1;
    	vector<int> nums(n);
    	for(int i=0;i<n;i++)
    		cin >> nums[i];
    	sort(nums.begin(),nums.end());
    	for(int i=0;i<n;i++)
    	{
    		ans = (ans * (nums[i]-i)%INF)%INF;
    	}
    	cout << ans << endl;
    }
    return 0;
}
