'''
# Read input from stdin and provide input before running code

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
n = int(raw_input())
ar = map(int, raw_input().split())
x = y = z = 0
for i in range(n):
	a = ar[i]
	if i % 3 == 0:
		x += a
	elif i % 3 == 1:
		y += a
	else:
		z += a
print x, y, z
