#include <bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
     int t;
	 ios::sync_with_stdio(0);
	 cin>>t;
	 while(t--)
	 {
	     int m,n,a,b,g,h,i;
		 
		 cin>>m>>n;
		 int arr1[m],arr2[m];
		 map<int ,int> w,x;
		 for(i=0;i<m;i++)
		 {
		     cin>>a>>b;
			 arr1[i]=a;
			 arr2[i]=b;
		 }
		 for(i=0;i<n;i++)
		 {
		     cin>>a>>b;
			 w[a]++;
			 x[b]++;
		 }
		 cin>>g>>h;
		 int good=0,great=0;
		 for(i=0;i<m;i++)
		 {
		     
			 if(w[arr1[i]]>0&&x[arr2[i]]>0)
			 {
			     
			      //cout<<i<<" "<<arr2[i]<<" "<<x[arr2[i]]<<endl;
			      x[arr2[i]]--;
				 great++;
			 }
			 if(w[arr1[i]]>0)
			 {
			     //cout<<i<<" "<<arr1[i]<<" "<<w[arr1[i]]<<endl;
			     w[arr1[i]]--;
				 good++;
			 }
			 
			 
		 }
		 if(great>=h&&good>=g)
		 cout<<"Great"<<endl;
		 else if(good>=g)
		 cout<<"Good"<<endl;
		 else
		 cout<<":("<<endl;
	 }
	 return 0;
}
