#include<iostream>
#include<cmath>
#include<cstdio>
#include<algorithm>
#include<stdlib.h>

using namespace std;

main()
{
	long long n,r,number=0,i,x,a=0,b=0;
	cin>>n>>r;
	long long arr[n],sum=0;
	for(i=0;i<n;i++)
	{
		cin>>x;
		{	
			arr[i]=x+1;
		}
		
		sum=sum+arr[i];
		//cout<<"length "<<sum<<endl;
		//cout<<"number "<<number<<endl;
	}
	//cout<<"sum "<<sum<<endl;
	
	sort(arr,arr+n);
	
	if(sum%n==0)
	{
		i=n-1;
		while(arr[i]>(sum/n))
		{
			//cout<<"number "<<arr[i]<<endl;
			a++;
			i--;
			
		}
		//cout<<"a "<<a<<endl;
		i=0;
		while(arr[i]==0)
		i++;
		while(arr[i]<(sum/n))
		{
			b++;
			i++;
		}
		//cout<<"b "<<b<<endl;
		if(a>=b)
		cout<<a*r<<endl;
		else
		cout<<b*r<<endl;
	}
	else
	cout<<-1<<endl;
}
