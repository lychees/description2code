int main() {

	int test;
	cin >> test;

	while (test--) {
		int n;
		cin >> n;
		int num[MAX];

		for (int i = 0; i < n; ++i) {
			cin >> num[i];
		}

		int ans;
		if (n == 1) {
			cout << 1 << endl;
		}
		else {
			ans = 40;

			for (int i = 0; i < n; ++i) {
				int temp = 0;
				for (int j = 0; j < 31; ++j) {
					if (num[i] & (1 << j)) {
						temp++;
					}
				}
				ans = min(ans,temp);
			}
		}

		cout << ans << endl;
	}	
	
	return 0;
}
