#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
	ios_base::sync_with_stdio(0);
	ll tc;cin>>tc;
	ll ans[202];
	ans[0]=0;
	ans[1]=1;
	ans[2]=2;
	for(int i=3;i<=201;i++)
		ans[i]=ans[i-1]+ans[i-2];
	for(ll i=0;i<tc;i++)
	{
		ll n;cin>>n;
		cout<<ans[n]<<endl;
	}
	return 0;
}
