N, M = map(int, raw_input().strip().split())
req = {}
car = [0]*M
for i in xrange(1, N+1):
    S, J = map(int, raw_input().strip().split())
    req[S] = [i, J]
res = ['-1']*N
k = sorted(req.keys())
for i in k:
    temp = req[i]
    for j in xrange(M):

        if i >= car[j]:
            car[j] = i+temp[1]
            res[temp[0]-1] = str(j+1)
            # print i,temp,car,res
            break
print ' '.join(res)
