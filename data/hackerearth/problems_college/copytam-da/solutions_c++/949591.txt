#include<iostream>
using namespace std;

int main()
{
	int t,a[1000],b[1000],i,j,count=0,k=0;
	long long int n;
	cin>>t;
	
	while(t--)
	{
		cin>>n;
		for(i=0;i<n;i++)
		cin>>a[i];
		
		
		for(i=0;i<n;i++)
		{
			for(j=0;j<i;j++)
			{
				if(a[i]==b[j])
				count=1;
			}
			if(count==0)
			{
			 b[k++]=a[i];
		    }
		    count=0;
		}
	cout<<k<<endl;
	k=0;count=0;
   }
   return 0;
}
