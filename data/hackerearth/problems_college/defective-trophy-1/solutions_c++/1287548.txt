#include<bits/stdc++.h>
using namespace std;

int solve(int n) {
    if(n==0||n==1) {
        return 0;
    }
    return 1+solve(int(ceil(n/3.0)));
}
int main() {
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        int n;
        scanf("%d",&n);
        int res=solve(n);
        printf("%d\012",res);
    }
    return 0;
}
