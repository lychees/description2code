#include<iostream>
#include<stdio.h>
#include<bits/stdc++.h>
using namespace std;
long long dp[10000000];
long long bs(long long l,long long h,long long n)
{
	long long ans=-1;
	while(l<=h)
	{
		long long mid=(l+h)/2;
		if(dp[mid]<=n)
		{
			ans=mid;
			l=mid+1;
		}
		else 
			h=mid-1;
	}
	return ans;
}
main()
{
	long long t,sum=0,l=1,h;
	scanf("%lld",&t);
	memset(dp,0,sizeof dp);
	for(long long i=1;i<=50000;i++)
	{
		sum+=(i*(i+1)*(i+1));
		dp[i]=sum;
		if(sum>=100000000000000000LL)
		{
			h=i;
			break;
		}
	}
	while(t--)
	{
		long long n;
		cin>>n;
	//	for(long long i=l;i<=n;i++)
	//		cout<<dp[i]<<" ";
		long long ans;
		ans=bs(l,h,n);
		if(ans==-1)
			cout<<"0\nDraw\n";
		else if(!(ans&1))
			cout<<ans<<"\nKetan\n";
		else
			cout<<ans<<"\nDhruv\n";
		
		
	}

	
}
