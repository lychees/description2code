import sys
from collections import defaultdict
c = 0
d = defaultdict(lambda: 0)
n = int(sys.stdin.readline().strip('\n'))
for i in xrange(n):
    S = ' '.join(
        map(str, sorted([int(x) for x in sys.stdin.readline().strip('\n').split()])))
    d[S] += 1
for T in d:
    if d[T] == 1:
        c += 1
print c
