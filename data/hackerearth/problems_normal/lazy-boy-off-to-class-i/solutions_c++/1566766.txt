#include<bits/stdc++.h>

using namespace std;

long long a[10000000];

int main(){

    long long t,n,x,y,z,k,sum;

    scanf("%lld",&t);

    while(t--)
    {
        k=0;
        n=0;
        sum=0;
        for(int i=0;i<7;i++)
        {
            scanf("%lld",&x);
            n+=x;

            for(int j=0;j<x;j++){

            scanf("%lld%lld",&z,&y);

            a[k++]=y-z;

            }

        }

        sort(a,a+k);

        y= n*3;

        if(y%4==0)
            y=y/4;
        else
            y=y/4+1;

        for(int i=0;i<y;i++)
            sum+=a[i];


        x=7*24*60*60-sum;

        printf("%lld\n",x);

    }
}
