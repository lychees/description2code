#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	long long int arr[n];
    	for(int i=0;i<n;i++)
    	{
    		cin>>arr[i];
    	}
		long long int sum1=0,sum2=0,count1=0,count2=0;
		int i=0,j=n-1;
		while(i<j)
		{
			if(sum1>sum2)
			{
				sum2+=2*arr[j];
				count2++;
				j--;
			}
			else if(sum1<sum2)
			{
				sum1+=arr[i];
				count1++;
				i++;
			}
			else
			{
				sum1+=arr[i];
				count1++;
				i++;
				sum2+=2*arr[j];
				count2++;
				j--;
			}
		}
		if(i==j && sum1<=sum2) count1++;
		else if(i==j && sum1>sum2) count2++;
		cout<<count1<<" "<<count2<<"\n";
    }
    return 0;
}
