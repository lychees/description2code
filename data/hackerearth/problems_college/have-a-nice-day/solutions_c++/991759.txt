#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int n;
    cin>>n;
    while(n--)
    {
    	int no;
    	cin>>no;
    	int *a= new int[no];
    	for(int i=0;i<no;i++)
    	cin>>a[i];
    	
    	int *b= new int[no];
    	for(int i=0;i<no;i++)
    	cin>>b[i];
    	
    	sort(a,(a+no));
    	sort(b,(b+no));
    	long long sum=0;
    	for(int i=0;i<no;i++)
    	{
    		sum+= a[i]*b[i];
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
