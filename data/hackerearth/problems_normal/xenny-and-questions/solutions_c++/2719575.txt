#include <iostream>
#include <algorithm>
#define MAX 100001
using namespace std;

int main()
{
    long int t,n,k,i;
    long int marks;
    long int b[MAX];
    long int a[MAX];    
        
    scanf("%ld",&t);
    while(t--){
    	scanf("%ld%ld",&n,&k);
    	int c[100001]={0};	    			
    		
	    for(i=1;i<=n;i++){
	    	scanf("%ld",&a[i]);
	    }
	    	
	    for(i=1;i<=n;i++){
	    	c[a[i]]++;
	    }
	    	
	    for(i=1;i<MAX;i++){
	    	c[i]+=c[i-1];
	    }
	    	
	    for(i=n;i>0;i--){
	    	b[c[a[i]]] = a[i];
	    	c[a[i]]--;
	    }
    	
    	marks=0;
    	for(i=n;i>0 && k>0;i--){
    		marks+=b[i];
    		k--;
    	}	
    	    	
    	printf("%ld\n",marks);
    }
    return 0;
}
