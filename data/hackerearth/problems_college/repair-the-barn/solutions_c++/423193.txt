int fil[205];
int dp[205][55];
int go(int st,int en,int wood){
	if(wood<0)
	return 1e5;
	if(fil[st]==1e9)
	return 0;
	if(wood==0)
	return 1e5;	
	if(dp[st][wood]!=-1)
	return dp[st][wood];
	int ret=1e5;
	for(int i=st;i<en;i++){
		ret=min(ret,fil[i]-fil[st]+1+go(i+1,en,wood-1));
	}
	dp[st][wood]=ret;
	return ret;
}
int main(){
	int m,s,c,i,j;
	cin>>m>>s>>c;
	memset(dp,-1,sizeof(dp));
	for(i=0;i<c;i++){
		cin>>j;
		fil[i]=j;
	}
	sort(fil,fil+c);
	fil[c]=1e9;
	cout<<go(0,c,m);
	return 0;
}
