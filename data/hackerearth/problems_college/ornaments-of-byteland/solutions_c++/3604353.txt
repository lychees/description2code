#include<iostream>
using namespace std;
int main()
{
	int a[26]={0},n,i,k,c,j;
	cin>>n;
	string s;
	cin>>s;
	for(i=0;i<s.length();i++)
	{
		a[s[i]-'a'] = 1;
	}
	for(k=1;k<n;k++)
	{
		int b[26]={0};
		cin>>s;
		for(j=0;j<s.length();j++)
		{
			b[s[j]-'a'] = 1;
		}
		for(i=0;i<26;i++)
		{
			a[i] = a[i] && b[i];
		}
	}
	c = 0;
	for(i=0;i<26;i++)
	{
		if(a[i]!=0)
		{
			c++;
		}
	}
	cout<<c<<endl;
}
