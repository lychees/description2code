#include <iostream>
#include <vector>
typedef long long ll;
using namespace std;

int main()
{
    int t = 0;
    cin>>t;
    vector<ll>sq;
    sq.push_back(0LL);
    sq.push_back(1LL);
    ll i=0,j=1;
    while(j<1e18){
    	sq.push_back(j*2);
    	j*=2;
    }
    int l = sq.size();
    
    while(t--){
    	ll n,prev=-1;
    	cin>>n;
    	if(n==1){cout<<"0 1"<<endl;continue;}
    	for(i=0;i<l;i++){
    		if(sq[i]>=n){
    			prev = sq[i-1];
    			i=sq[i];
    			break;
    		}
    	}
    	cout<<i-1<<" "<<2*(n-prev)<<endl;
    	
    }
    return 0;
}
