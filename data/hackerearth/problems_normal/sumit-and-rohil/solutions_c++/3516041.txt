#include <bits/stdc++.h>
using namespace std;

int main()
{
    // cout << "Hello World!" << endl;
    int n, i, j, k;
    cin>>n;
    string s[n];
    for(i = 0; i< n; i++) cin>>s[i];
    int l1, l2, c1, c2, c_1, c_2, t, grp = 0;
    for(i = 0; i < n; i++) {
    	l1 = s[i].length();
    	if(l1 == 0) continue;
    	c1 = s[i][0], c2 = s[i][l1-1];
    	for(j = i+1; j <n; j++) {
    		l2 = s[j].length();
	    	if(l2 == 0) continue;
	    	c_1 = s[j][0], c_2 = s[j][l2-1];
	    	if(l1 == l2 && c1 == c_1 && c2 == c_2) {
	    		sort(s[i].begin(), s[i].end()); sort(s[j].begin(), s[j].end());
	    		t = 0;
	    		for(k = 0; k < l1; k++) {
	    			if(s[i][k] != s[j][k]) {
	    				t = 1; break;
	    			}
	    		}
	    		if(t == 0) s[j] = "";
	    	}
    	}
    	grp++;
    }
    cout<<grp;
    return 0;
}
