#include<bits/stdc++.h>
#define ll long long int
using namespace std;
main()
{
    ll n,i,ans,j;
    cin>>n;
    ll arr[n];
    for(i=0;i<n;i++)
        cin>>arr[i];
    ll dp[n];
    for(i=0;i<n;i++)
        dp[i]=1;
    sort(arr,arr+n);
    ans=0;
    for(i=0;i<n;i++)
    {
        for(j=i+1;j<n;j++)
        {
            if(arr[j]%arr[i]==0)
            {
                dp[j]=max(dp[j],dp[i]+1);
                if(ans<dp[j])
                    ans=dp[j];
            }
        }
    }
    if(ans>1)
        cout<<ans<<endl;
    else
        cout<<-1<<endl;
    return 0;
}
