#include <iostream>
#include<climits>
using namespace std;

int main()
{
    int t;
    cin >> t;
    	int T[501][501];
    	for(int i=0;i<=500;i++)
    	T[0][i]=0;
    	for(int i=0;i<=500;i++)
    	T[i][0]=0;
    	for(int i=1;i<=500;i++)
    	T[1][i]=i;
    	for(int i=2;i<=500;i++)
    	{
    		for(int j=1;j<=500;j++)
    		{
    			T[i][j]=INT_MAX;
    			for(int k=1;k<=j;k++)
    			{
    				int c = max(1+T[i-1][k-1],1+T[i][j-k]);
    				if(c<T[i][j])
    				T[i][j]=c;
    			}
    		}
    	}
    while(t--)
    {
    	int floors,eggs;
    	cin >> floors >> eggs;
    
    	cout<<T[eggs][floors]<<endl;
    }
    return 0;
}
