#include<bits/stdc++.h>
#define maximum 100000

using namespace std;

int ARR[maximum],siz[maximum];

void set_Initialize(int n)
{
    for(int i=0;i<=n;++i)
    {
        ARR[i]=i;
        siz[i]=i;
    }
}

int root(int x)
{
    if(ARR[x]!=x) return ARR[x]=root(ARR[x]);
    return ARR[x];
}

void Union(int x,int y)
{
    int root_A=root(x);
    int root_B=root(y);
    if(root_A!=root_B)
        ARR[root_A]=root_B;
}

int main()
{
    int soldier,query,x,y;
    //#ifndef ONLINE_JUDGE
    //freopen("int.txt","r",stdin);
    //freopen("int.txt","w",stdout);
    //#endif // ONLINE_JUDGE
    scanf("%d %d",&soldier,&query);
    set_Initialize(soldier);

    while(query--)
    {
        int query_operation_number;
        scanf("%d",&query_operation_number);

        if(query_operation_number==1)
        {
            scanf("%d %d",&x,&y);
            Union(x,y);
        }
        else if(query_operation_number==2)
        {
            scanf("%d",&x);
            ARR[root(x)]=x;
            ARR[x]=x;
        }
        else
        {
            scanf("%d",&x);
            cout<<root(x)<<endl;
        }
    }
 return 0;
}
