#include <iostream>
#include <vector>
using namespace std;
#define MOD 1000000007
//This is for my Modulo operation 
int getWays(vector<int> x, vector<int> d, int M) {
    int N = d.size();
 //long longg better
    long long multid[M + 1][N + 1];
    for (int i = 0; i < N; i++) {
        long long m[d[i] + 1][M + 1];
        int D = d[i];
        for (int j = 1; j <= D; j++)
            m[j][0] = 1;
        for (int n = 1; n <= M; n++) {
            for (int j = 1; j <= D; j++) {
                m[j][n] = 0;
                if (j - 1 > 0)
                    m[j][n] = (m[j][n] + m[j - 1][n - 1]) % MOD;
                if (j + 1 <= D)
                    m[j][n] = (m[j][n] + m[j + 1][n - 1]) % MOD;
            }
        }
 
        multid[0][i + 1] = 1;
        for (int j = 1; j <= M; j++)
            multid[j][i + 1] = m[x[i]][j];
    }
 
    long long c[M + 1][M + 1];
    for (int i = 0; i <= M; i++) {
        c[i][0] = 1;
        c[i][i] = 1;
    }
    for (int i = 1; i <= M; i++)
        for (int j = 1; j < i; j++)
        c[i][j] = (c[i - 1][j - 1] + c[i - 1][j]) % MOD;
 
    long long multidt[M + 1][N + 1];
    for (int i = 0; i <= M; i++)
        multidt[i][1] = multid[i][1];
    for (int i = 0; i <= N; i++)
        multidt[0][i] = 1;
 
    for (int i = 2; i <= N; i++) 
    {
    for (int j = 1; j <= M; j++) 
    {
   	multidt[j][i] = 0;
    for (int k = 0; k <= j; k++)
    multidt[j][i] =(multidt[j][i] +((c[j][j - k] *((multidt[k][i - 1] * multid[j - k][i]) % MOD)) % MOD)) % MOD;
        }}
    return multidt[M][N];
}
 
int main() {
    int T, N, M, temporary;
    cin >> T;
 
    for (int i = 0; i < T; i++) {
        cin >> N >> M;
        vector<int> x;
        vector<int> d;
        for (int j = 0; j < N; j++) {
            cin >> temporary;
            x.push_back(temporary);
        }
        for (int j = 0; j < N; j++) {
            cin >> temporary;
            d.push_back(temporary);
        }
        cout << getWays(x, d, M) << endl;
    }
    return 0;
}
