#include<iostream>
using namespace std;
int main()
{
    long int n,m,x,p;
    cin>>n>>m;
    long int a[n];
    for(x=0;x<n;x++)
        a[x]=0;
    for(long int i=0;i<n;i++)
    {
        cin>>x;
        if(x%m==0)
        {
            p=x/m;
            if(p<n)
            a[p]=x;
        }
    }
    for(long int i=1;i<n;i++)
    {
       // cout<<a[i];
        if(a[i]!=0);
        else
        {cout<<a[i-1];
        break;}
    }
}
