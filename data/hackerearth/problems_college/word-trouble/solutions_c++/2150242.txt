#include <iostream>
#include <cstring>
using namespace std;

int checkstr(string s1,string s2){

	int length=0,sz;

sz = min(s1.size(),s2.size());

	for(int i=0;i<sz;i++){

		if(s1[i] != s2[i])
		length++;

	}

	if(s1.size() > s2.size())
	length += s1.size() - s2.size();
	else length += s2.size() - s1.size();


return length;
}

int main()
{
    int t;
    string str1,str2;

    cin>>t;

   while(t--){

   		cin>>str1>>str2;

   		if(checkstr(str1,str2)==0)
   		   cout<<"Exact\n";
   		else cout<<"Non Exact "<<checkstr(str1,str2)<<"\n";

   }

   return 0;
}
