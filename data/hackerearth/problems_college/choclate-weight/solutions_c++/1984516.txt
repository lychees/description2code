#include <stdio.h>
#include <iostream>
#include <algorithm>
using namespace std;

int main(void) 
{
	int t,a[101],n,k,i,j,sum,flag=0;
	scanf("%d",&t);
	for(i=0;i<t;i++)
	{
		scanf("%d",&a[i]);
	}
	sort(a,a+t);
	scanf("%d",&n);
	while(n--)
	{
		scanf("%d",&k);
		for(i=0;i<t-1;i++)
		{
			flag=0;
			for(j=i;j<t;j++)
			{
				if(a[j]==k)
				{
					printf("I Like You\n");
					flag=1;
					break;
				}
				if(a[j]+a[i]==k)
				{
					printf("I Like You\n");
					flag=1;
					break;
				}
				if(flag==1)
				break;
			}
			if(flag==1)
			break;
		}
		if(flag==0)
		printf("I Hate You\n");
	}
	return 0;
}
