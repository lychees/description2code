#include<bits/stdc++.h>
using namespace std;
int A[200]={0};
int main()
{
    int n,q,x,a,b;
    scanf("%d %d",&n,&q);
    while(q--)
    {
        scanf("%d %d %d",&x,&a,&b);
       if(!x)
       {
           for(int i=a;i<=b;i++)
            A[i]=A[i]^1;
       }
      else
      { int ans=0;
          for(int i=a;i<=b;i++)
          {
              if(A[i])
                ans++;
          }
          printf("%d\n",ans);
      }
    }
    return 0;
}
