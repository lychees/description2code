#include<stdio.h>

const int SZ = (1e+7);

int prime[SZ+5],phi[SZ+5];
long long phisum[SZ+5];
int n,t;

void calc()
{
    phi[1]=1;
    prime[1]=1;

    for(int i=2;i<=(1e+7);++i)
    {
        if(prime[i]==0)
        {
            for(int j=i*2;j<=(1e+7);j+=i)
                prime[j] = i;
        }
    }

    for(int i=2;i<=(1e+7);++i)
    {
        if(prime[i]==0)
            phi[i]=i-1;
        else
        {
            int x=i;
            int max_prime = prime[i];

            while(x%max_prime==0)
                x/=max_prime;

            phi[i] = phi[x]*(i/x)/max_prime*(max_prime-1);
        }
    }
    for(int i=1;i<=(1e+7);++i)
        phisum[i] = phisum[i-1] + phi[i];
}

int main()
{
    
    calc();
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        printf("%lld\n",phisum[n]);
    }
    return 0;
}
