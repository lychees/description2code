#include<bits/stdc++.h>
using namespace std;

#define ll long long

template <class X>

void print_num(X a)
{
    cout<<a;
}

int main()
{
    int n;
    string s, t;
    int s1, t1;
    int i,j,k;
    cin>>n;
    for(k=0;k<n;k++)
    {
        cin>>s>>t;
        s1 = s.length();
        t1 = t.length();
        i = 0;
        j = 0;

        sort(t.begin(), t.end());

        for(i=0,j=0;i<s1;i++)
        {
            if(j>=t1)
                break;
            if(s[i]>t[j])
            {
                s[i] = t[j];
                j++;
            }
        }
        cout<<s<<endl;
    }
    return 0;
}
