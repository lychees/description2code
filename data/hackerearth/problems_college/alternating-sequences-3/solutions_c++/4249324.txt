#include<iostream>
#include<bits/stdc++.h>
#define max(i,j) i>=j?i:j;
using namespace std;
int main()
{
	int n,i,j,k=1;
	long ar[5005];
	int dp[5005];
	scanf("%d",&n);
	for(i=1;i<=n;++i)
	{
		scanf("%ld",&ar[i]);
		dp[i]=1;
	}
	for(i=2;i<=n;++i)
	{
		for(j=1;j<i;++j)
		{
			if(ar[i]*ar[j]<0 && abs(ar[i])>abs(ar[j]))
			    dp[i]=max(dp[i],dp[j]+1);
			if(dp[i]>k)
			k=dp[i];
		}
	}
	printf("%d\n",k);
}
