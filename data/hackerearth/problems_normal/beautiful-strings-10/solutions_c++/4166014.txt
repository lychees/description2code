#include<iostream>
#include<algorithm>
using namespace std;
pair<int,int> p[100001];
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        string s;
        cin>>s;
        int n=(int)(s.size());
        p[0].first=0;
        p[0].second=0;
        for(int i=0;i<n;i++)
        {
            p[i+1]=p[i];
            if(s[i]=='a')
            {
                p[i+1].first--;
                p[i+1].second--;
            }
            if(s[i]=='b')
            {
                p[i+1].first++;
            }
             if(s[i]=='c')
            {
               p[i+1].second++;
            }
        }
        sort(p,p+n+1);
        int c=1;
        int ans=0;
        for(int i=1;i<=n;i++)
        {
            if(p[i]==p[i-1])
                c++;
            else
            {
                ans+=((c*(c-1))/2);
                c=1;
            }
        }
        ans+=((c*(c-1))/2);
        cout<<ans<<endl;
    }
}
