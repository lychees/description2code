#include<iostream>
#include<cstdio>

using namespace std;

int main()
{
        int n;

        int dp[100001]={0};

        cin>>n;
        char s[n];

        cin>>s;
        dp[0]=s[0]-48;
        for(int i=1;i<n;i++)
        {

                dp[i]=dp[i-1]+s[i]-48;
        }

        int q;
        cin>>q;

        while(q--)
        {
                int l,r;

                cin>>l>>r;
                int k=dp[r-1]-dp[l-1]+s[l-1]-48;
                if(k%9==0)
                {
                        cout<<"SAMEER"<<endl;
                }
                else
                {
                        cout<<"NO"<<endl;
                }

        }

        return 0;
}
