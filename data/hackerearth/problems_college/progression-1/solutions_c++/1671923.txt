#include <iostream>
using namespace std;
#define ll long long int

int main()
{
    int t;
    cin>>t;
    
    while(t--)
    {
    	ll n,n1=1;
    	cin>>n;
    	n1=n;
       if(n>=90)
       {
       	if(n%3==0)
     	cout<<"CORRECT"<<endl;
    	else
    	cout<<"INCORRECT"<<endl;
        continue;
       }
       else if(n<45)
       cout<<"INCORRECT"<<endl;
       else
       {
       	n=n-45;
       	if(n%3==0)
       	cout<<"CORRECT"<<endl;
    	else
    	cout<<"INCORRECT"<<endl;
       }
       
    }
}
