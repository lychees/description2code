#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int a[n][n];
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
        {
            cin>>a[i][j];
        }
    vector<int> f(n);
    for(int i=0;i<n;i++)
    {
        f[i]=a[i][n-1];
    }

    for(int i=n-2;i>=0;i--)
    {
        f[0]+=a[0][i];
        for(int j=1;j<n;j++)
        {
            f[j]=min(f[j-1]+a[j][i],f[j]+a[j][i]);
        }

        for(int j=n-2;j>=0;j--)
            f[j]=min(f[j],f[j+1]+a[j][i]);
    }
    cout<<*min_element(f.begin(), f.end());

}
