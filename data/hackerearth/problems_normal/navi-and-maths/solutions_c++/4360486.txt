#include <iostream>
#define M 1000000007
typedef long long int ll;
using namespace std;
    ll ex(ll sum,ll Q)
    {
    	ll temp;
    	if(Q==0) return 1;
    	temp = ex(sum,Q/2);
    	if(Q%2==0) return((temp%M)*temp)%M;
    	else return (((sum*temp)%M)*temp)%M;
    }
    int main()
    {
       // cout << "Hello World!" << endl;
        int t,k;
        cin>>t;
        for(k=1;k<=t;k++)
        {
        	int n;ll i,j,X,N,sum,Q,res,pro,maxi=0;
        	ll A[20];
        	cin>>n;
        	for(i=0;i<n;i++) cin>>A[i];
        	N = 1<<n;
        	for(i=1;i<N;i++)
        	{
        		res=0;pro=1;sum=0;
        		for(j=0;j<=n-1;j++)
        		{
        			X = i&(1<<j);
        			if(X!=0)
        			{sum=(sum+A[j]);
        			 pro=(pro*A[j])%M;}
        		}
        		if(sum!=0)
        		{Q=M;
        		sum=ex(sum,Q-2)%M;
        		res=(pro*sum)%M;}
        		if(res>maxi) maxi=res;
        	}
        	cout<<"Case #"<<k<<": "<<maxi<<endl;
        }
        return 0;
    }
