#include <iostream>
using namespace std;

int main()
{
	int n,i;
	
	cin>>n;
	
	char str[1000006];
	int ms[1000006];
	
    cin>>str;
    
    ms[0]=0;
    
    
    for(i=1;i<n;i++)
    {
    	if(str[i]<str[ms[i-1]])
    	  ms[i] = i;
    	else if(str[i]==str[ms[i-1]])
    	{
    	   int j=1;
    	   while(j<n)
    	   {
    	   	  if(str[i+j]<str[ms[i-1]+j])
    	   	    {
    	   	    	ms[i]=i;
    	   	    	break;
    	   	    }
    	   	  if(str[i+j]>str[ms[i-1]+j])
    	   	    {
    	   	    	ms[i]=ms[i-1];
    	   	    	break;
    	   	    } 
    	   	    j++;
    	   }
    	   if(j==n)
    	     ms[i]=0;
          }
    	else
    	 ms[i]=ms[i-1];
     }
    
   int t,ml;
   
   cin>>t;
   
   while(t--)
   {
   	cin>>ml;
    cout<<ms[ml]<<endl;
   }
    
    
    return 0;
}
