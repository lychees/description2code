#include <iostream>
using namespace std;

int main()
{
	int t ;
	cin >> t;
	while(t>0)
	{
		t--;
		
		int m;
		cin >> m;
		
		long long countOne=1;
		long long countZero=0;
		
		while(m>0)
		{
			m--;
			long long zero = countOne;
			countOne = 0;
			
			if(countZero>0)
			{
				countOne += countZero;
			}
			
			countZero+=zero;
		}
		
    	cout << countOne << " " << countZero << endl;
	}
    return 0;
}
