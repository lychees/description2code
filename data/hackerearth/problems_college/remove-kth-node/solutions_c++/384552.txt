#include<stdio.h>
#include<iostream>
using namespace std;
long fi(long n, long k)
{
    if(n==1)
     return 1;
    else
      return ((fi(n-1,k)+k-1)%n)+1;   
}

int main()
{
 long n,t;
 long k;
 
 cin>>t;
 while(t--)
 {
          cin>>n>>k; k=k+1;
        //  cout<<n<<","<<k<<endl;
          cout<<fi(n,k)<<endl;
           
 }
    
    return 0;
}
