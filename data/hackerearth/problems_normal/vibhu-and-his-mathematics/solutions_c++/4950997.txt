#include <bits/stdc++.h>
 
#define LLI long long int
#define LD long double
#define PB push_back
#define MP make_pair
#define FORi(i, a, b) for(int i = a; i < b ; ++i)
#define FORd(i, a, b) for(int i = a; i > b ; --i)

using namespace std;

const int MOD = 1e9+7;
vector<int> cache(1234567, -1);

int solve(int n){
	if (cache[n] < 0){
		LLI tmp = solve(n-1) + 1ll*(n-1)*solve(n-2);
		cache[n] = tmp % MOD;
	}
	return cache[n];
}

int main()
{
	int t,n;
	cache[1] = 1;
	cache[2] = 2;
	cin >> t;
	while(t--){
		cin >> n;
		cout << solve(n) << endl;
	}
    return 0;
}
