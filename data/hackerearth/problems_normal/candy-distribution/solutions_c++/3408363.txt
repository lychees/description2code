#define MOD 1000000007
using namespace std;
map<ll,ll> dp;
vector<int> pr;
ll power(ll a,ll b)
{
    ll x=a%MOD,res=1;
    while(b)
    {
        if(b&1)
            res=(res*x)%MOD;
        x=(x*x)%MOD;
        b>>=1;
    }
    return res;
}
ll fn(ll n)
{
    if(dp[n])
        return dp[n];
    ll i;
    ll sum=1;
    fr(i,2,sqrt(n))
    {
        if(i>=(n-1))
            break;
        if((n-1)%i==0)
        {
            sum=(sum+power(fn(i),(n-1)/i))%MOD;
            if((n-1)/i>sqrt(n))
                sum=(sum+power(fn((n-1)/i),i))%MOD;
        }
    }
    return dp[n]=sum;
}
