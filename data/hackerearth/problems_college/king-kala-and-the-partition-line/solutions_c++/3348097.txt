#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int tc;int n,a,b,c;
    cin>>tc ;
    while(tc--)
    {
    	cin>>n;
    	cin>>a>>b>>c ;
    	int c1=0,c2=0 ;
    	int i; int l,m;
    	for(i=0;i<n;i++)
    	{  cin>>l>>m;
    	   if(a*l+b*m+c==0)
    	    continue;
    	    else
    	    if(a*l+b*m+c>0)
    	    c1++;
    	    else
    	    c2++;
    	    
    		
    	}
    	
    	cout<<max(c1,c2)<<endl ;
    	
    }
    
    return 0;
}
