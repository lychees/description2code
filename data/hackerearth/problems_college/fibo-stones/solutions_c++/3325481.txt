#include<iostream>
using namespace std;

long long int fab[100];

void fibonacci()
{
	fab[0]=1;  fab[1]=1;
	
	for(int i=2; i<=97; ++i )
	  fab[i]=fab[i-1]+fab[i-2];
	  
}

int main()
{
	long long int n,m,i,p,count=0,count2=0;
	
	cin>>n>>m;
	
	fibonacci();
	
	while(m--)
	   {
	   	  cin>>p;
	   	  
	   	  count+=p;
	   }
	   
	
	for(i=0; i<n; ++i)
	{
		
		count2+=fab[i];
	}
	
	cout<<count2-count<<endl;	
	
return 0;	
}
