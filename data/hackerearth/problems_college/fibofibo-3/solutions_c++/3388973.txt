#include <bits/stdc++.h>
using namespace std;

int a[100];
int fibo(int n)
{
	int ctr=0,ctr2=0;
	
	if(n==0)
	{
	ctr++;
	a[0]=ctr;
	return 0;
	}
	
	else if(n==1)
	{
	ctr2++;
	a[1]=ctr2;
	return 1;
	}
	
	else
	return fibo(n-1)+fibo(n-2);
	
	

}

int main()
{
	int t,n,dp[10000],dp2[10000],i;
	
	cin>>t;


	while(t--)
	{
		dp[0]=1;
		dp[1]=0;
		dp2[0]=0;
		dp2[1]=1;
		cin>>n;
		
		if(n==0)
		cout<<"1 0"<<endl;
		
		else if(n==1)
		cout<<"0 1"<<endl;
		
		else
		{
			for(i=2;i<=n;i++)
			{
			dp[i]=dp[i-1]+dp[i-2];
			
			dp2[i]=dp2[i-1]+dp2[i-2];
			}
		
		
		
		cout<<dp[n]<<" "<<dp2[n]<<endl;
		}
	}

    return 0;
}
