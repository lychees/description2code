#include <iostream>
using namespace std;

int main()
{
	int T, N;
	cin >> T;
	while (T > 0) {
		cin >> N;
		int *arr = new int[N];
		for (int i = 0; i < N; i++)
			cin >> arr[i];
		int start = 0, end = N-1;
		while (start < N-1 && arr[start] < arr[start+1])
			start++;
		while (end > 0 && arr[end] > arr[end-1])
			end--;
		int i = start;
		while (i < end) {
			if (arr[i] < arr[i+1])
				break;
			i++;
		}
		if (i == end && (start == 0 || arr[end] > arr[start-1]) && (end == N-1 || arr[start] < arr[end+1])) {
			cout << start+1 << " " << end+1 << endl;
		} else {
			cout << -1 << " " << -1 << endl;
		}
		delete[] arr;
		T--;
	}
    return 0;
}
