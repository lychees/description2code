#include<bits/stdc++.h>
int  n;
using namespace std;
char s[10000],s1[10000];
int main()
{	
	int t,a,b;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%s%s",s,s1);
		int i,j,k;
		int l1=strlen(s);
		int l2=strlen(s1);
		i=l1-1;
		j=l2-1;
		int carry=0,c=0;
		while (i>=0&&j>=0)
		{
			if ((carry+s[i]+s1[j]-'0'-'0')>9)
			{
				c++;
				carry=(carry+s[i]+s1[j]-'0'-'0')/10;
			}
			i--;j--;
		}
		while (carry>0&&(i>0||j>0))
		{
			if (i>0)
				carry=(carry+s[i])/10;
			else if (j>0)
				carry=(carry+s1[j])/10;
			if (carry>0)
				c++;
			i--;j--;
		}
		if (c==0)
			printf("No carry operation");
		else
			printf("%d carry operation",c);
		if (c>1)
			printf("s");
		printf("\n");
	}
	return 0;
}
