#include <iostream>
using namespace std;
int search(long long my_numbers[], int start, int size, long long key)
{
	int left = start, right = size -1, mid, index=-1;
	while (left <= right)
	{
		mid = (int) ((left + right) / 2);
		if (key <= my_numbers[mid])
		{
			left = mid + 1;
			index = mid;
		}
		else
		{
			right = mid - 1;
		}
	}
	return index;
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int t;
    cin>>t;
    while(t--)
    {
    	int n,i,ind;
    	cin>>n;
    	long long A[n],B[n];
    	for(i=0;i<n;i++)
    		cin>>A[i];
    	for(i=0;i<n;i++)
    		cin>>B[i];
    	ind = search(B,0,n,A[0]);
		for(i=1;i<n;i++)
		{
			if((search(B,i,n,A[i])-i)>ind)
				ind=(search(B,i,n,A[i])-i);
		}
		if(ind==-1)
			cout<<0<<endl;
		else
			cout<<ind<<endl;
    }
    return 0;
}
