#include <iostream>
using namespace std;

int main()
{
    int t=0;
    long long k=0;
    long long n=0;
    long long m=0;
    
    cin >> t;
    while(t--){
    	cin >> n;
    	cin >> k;
    	cin >> m;
    	
    	long long totalAmount=0;
    	for(int i=1; i<=n; i++){
    		totalAmount += (i*i*i);
    	}
    	
    	totalAmount = totalAmount*k-m;
    	if(totalAmount<=0){
    		std::cout << 0 << std::endl;
    	}else{
    		std::cout << totalAmount << std::endl;
    	}
    }
    return 0;
}
