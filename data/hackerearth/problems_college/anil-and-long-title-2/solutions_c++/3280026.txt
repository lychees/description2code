#include <iostream>
using namespace std;
#include<cctype>
int main()
{
    char str[100];
    int i=0;
    cin >> str;
    if(!i)
      str[i]=toupper(str[i]);
    for(i=0;str[i]!='\0';i++)
    {
    
       if(str[i]=='_' && str[i+1]=='_')
           continue;
       if(str[i]=='_')
          str[i+1]=toupper(str[i+1]);
    }
    cout << str<<endl;
    return 0;
}
