    #include <iostream>
    #include<string>
    #include<cmath>
    using namespace std;
     
    int main()
    {
    	int t;
        cin>>t;
        cin.ignore();
        while(t--)
        {
        	string input;
        	char output[200];
        	getline(cin,input);
        	int k=0;
        	int len=input.length();
        	int side;
        	side=sqrt(len);
        	for(int i=0;i<side;i++)
        	for(int j=i;j<len;j+=side)
        	{
        		output[k]=input[j];
        		k++;
        	}
        	for(int i=0;i<k;i++)
        	cout<<output[i];
        	cout<<endl;
        	
        	
        	
        }
    }
