#include <iostream>
#include<algorithm>
#include<cstdio>
using namespace std;

int main()
{
    int n, *a, q;
    cin >> n;
    a = new int[n];
    for(int i = 0; i < n; i++)
    	scanf("%d", &a[i]);
    sort(a, a + n);
    cin >> q;
    while(q--)
    {
    	int l, r;
    	int *low, *up;
    	scanf("%d%d", &l, &r);
    	low = lower_bound(a, a + n, l);
    	up = upper_bound(a, a + n, r);
//    	cout << low - a << endl << up - a << endl;
    	cout << up - low << endl;
    }
    return 0;
}
