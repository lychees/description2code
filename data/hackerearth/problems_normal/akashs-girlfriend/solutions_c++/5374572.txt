#include<cstdio>
#define MOD 1000000009

long long POW(long long a,long long n){
    long long ans=1;
    while(n){
        if(n&1)
            ans=(ans*a)%MOD;
            a=(a*a)%MOD;
            n>>=1;
    }
    return ans;
}
int main(){
    int cs;
    scanf("%d",&cs);
    while(cs--){
        long long n;
        scanf("%lld",&n);
        long long ans=(POW(10,n)-POW(8,n))*POW(2,MOD-2)%MOD;
        if(ans<0)
            ans+=MOD;
        printf("%lld\n",ans);
    }
   return 0;
}
