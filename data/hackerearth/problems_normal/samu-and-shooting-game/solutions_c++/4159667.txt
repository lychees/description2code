#include <iostream>
#include <stdio.h>
#include <string.h>
#include <algorithm>
#include <queue>
#include <math.h>
#include <map>
#include <set>
#include <unordered_map>
#include <vector>
#include <iomanip>

using namespace std;

int x,y,n,w;
double p1,p2;

double dp[1010][1010];
double f(int N , int W )
{
    if ( W < 0 ) W = 0 ;
    if ( !N ) return W == 0 ;
    if ( dp[N][W] >=0 ) return dp[N][W] ;
    dp[N][W] = p1*f(N-1,W-x) + (1-p1)*f(N-1,W) ;
    dp[N][W] =  max(dp[N][W],p2*f(N-1,W-y) + (1-p2)*f(N-1,W)) ;
    return dp[N][W];
}
int main()
{
    int t ;
    cin >> t ;
    while ( t-- )
    {
        memset(dp,-1,sizeof dp) ;
        cin >> x >> y >> n >> w >> p1>> p2 ;
        p1/=100.;
        p2/=100.;
        cout << setprecision(6) << fixed << f(n,w)*100 << endl;
    }
    return 0;
}
