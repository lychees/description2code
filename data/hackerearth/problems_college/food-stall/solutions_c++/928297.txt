#include <iostream>
#include<algorithm>
 
using namespace std;
#define inp(t) scanf("%lld",&t)
int main()
{
	long long t,n,i,d,a[100000];
    inp(t);
    while(t--)
    {
    	inp(n);
    	inp(d);
    	long c=0;
    	for(i=0;i<n;i++)
    	inp(a[i]);
    	sort(a,a+n);
    	for(i=n-1;i>0;i--)
    	{
    		if(a[i]-a[i-1]<=d)
    		{c++;i--;}
    		
    	}
    	
    	printf("%ld\n",c);
    	
    }
    return 0;
}
