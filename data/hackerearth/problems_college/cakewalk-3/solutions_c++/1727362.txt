#include <bits/stdc++.h>
#define ll long long int

using namespace std;

int main()
{
	int n,x,l,j;
	char s[100010],s1[100010];
	cin>>n;
	scanf("%s",s);
	for(int i=0;i<n-1;i++)
	{
		scanf("%d %s",&x,s1);
		l=strlen(s1);
		for(j=0;j<l;j++)
			s[x+j]=s1[j];
	}
	s[x+j]='\0';
	printf("%s\n",s);
	return 0;
}
