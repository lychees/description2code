for testcases in xrange(int(raw_input())):
    # problem solution
    answer = False
    pattern = raw_input()
    text = raw_input()
    pattern_length = len(pattern)
    text_length = len(text)
    if text_length < pattern_length:
        print 'NO'
        continue
    pats = [0]*26
    txts = [0]*26
    for x in pattern:
        pats[ord(x) - 97] += 1
    for x in xrange(pattern_length):
        txts[ord(text[x]) - 97] += 1
    for x in xrange(pattern_length, text_length):
        if pats == txts:
            answer = True
            break
        txts[ord(text[x]) - 97] += 1
        txts[ord(text[x - pattern_length]) - 97] -= 1
    if pats == txts:
        answer = True
    if answer:
        print 'YES'
    else:
        print 'NO'
