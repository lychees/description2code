#include <iostream>
using namespace std;
#include <bits/stdc++.h>

int num=1;

void mat(int i, int j, int size, vector<vector<int>> &arr, int n)
{
	if(i>=n || j>=n)
		return;
	if(size==2)
	{
		arr[i][j]=num++;
		arr[i][j+1]=num++;
		arr[i+1][j]=num++;
		arr[i+1][j+1]=num++;
	}
	else
	{
		mat(i, j, size/2, arr, n);
		mat(i, j+size/2, size/2, arr, n);
		mat(i+size/2, j, size/2, arr, n);
		mat(i+size/2, j+size/2, size/2, arr, n);
	}
    return;
}

int main()
{
    int n;
    cin>>n;
    if(n<1||n>10)
    	return 0;
    int size=pow(2,n);
    vector<vector<int>> arr(size, vector<int>(size));
    mat(0, 0, size, arr, size);
    for(int i=0; i<size; i++)
    {
    	for(int j=0; j<size; j++)
    	{
    		cout<<arr[i][j]<<" ";
    	}
    	cout<<endl;
    }
    return 0;
}
