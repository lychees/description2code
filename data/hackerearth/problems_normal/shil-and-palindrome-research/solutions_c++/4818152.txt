#include<iostream>
#include<cstring>

using namespace std;
int bit[100005][26],n;

void update(int x,int v,int p){
	for(;x<=n;x += x&-x)
		bit[x][p] += v;
}
int query(int x,int p){
	int sum=0;
	for(;x>0;x -= x&-x)
		sum += bit[x][p];
	return sum;
}
int main()
{
	string s;
	char a;
	int i,l,r,q,o;
	cin>>n>>q;
	cin>>s;
	for(i=0;i<n;i++){
		update(i+1,1,s[i]-'a');
	}
	while(q--){
		cin>>o;
		if(o==1){
			cin>>l>>a;
			update(l,1,a-'a');
			update(l,-1,s[l-1]-'a');
			s[l-1]=a;
		}
		if(o==2){
			cin>>l>>r;
			int odd=0,c;
			for(i=0;i<26;i++){
				c= query(r,i)-query(l-1,i);
				if(c%2)
					odd++;
			}
			if(odd<=1)
				cout<<"yes\n";
			else
				cout<<"no\n";
		}
	}
    return 0;
}
