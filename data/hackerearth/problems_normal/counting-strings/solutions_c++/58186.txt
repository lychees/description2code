typedef pair<int,int> PI;

typedef vector<VI> VVI;
typedef long long LL;

int main()
{
	int i,j,n,t;
	si(n);
	rep(i,n)
	{
		string q;
		cin >> q;
		VI s;
		LL l,r;
		rep(j,sz(q))
			if(q[j]=='a' || q[j]=='z')
				s.pb(j);
		int x = sz(q);
		LL ans = 0;
		rep(j,sz(s))
		{
			r = x-1-s[j];
			if(j==0)
			{
				l = s[j];
			}
			else
			{
				l = s[j]-s[j-1]-1;
			}
			t = l + r + 1 + l*r;
			ans += t;
			//printf("%d %d %d\n",s[j],t,ans);
		}
		cout << ans << endl;
	}
	return 0;
}
