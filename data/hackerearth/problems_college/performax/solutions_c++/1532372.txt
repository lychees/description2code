#include<iostream>
using namespace std;

int main()
{
    int i,j,n,d,a[20005],m=0,c;
    cin>>n>>d;
    for(i=0;i<n;i++)
      cin>>a[i];

    for(i=1;i<n;i++)
    {
        if(a[i]<a[i-1])
          {
             c=(a[i-1]-a[i])/d +1;
             a[i] = a[i]+(c*d);
             //cout << c << endl;
             m=m+c;
          }

    }

    cout<<m;

return 0;
}
