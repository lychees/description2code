#include<bits/stdc++.h>
using namespace std;
int main(){
    int t;
    cin>>t;
    while(t--){
    	string s; cin>>s;
    	string r=s;
    	reverse(r.begin(),r.end());
    	int l=s.size();
    	int f=1;
    	for(int i=1; i<l; i++){
    		if(abs(s[i]-s[i-1])!=abs(r[i]-r[i-1])){
    			f=0; break;
    		}
    	}
    	if(f) cout<<"Funny\n";
    	else cout<<"Not Funny\n";
    }
}
