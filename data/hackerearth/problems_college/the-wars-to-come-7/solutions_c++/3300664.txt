#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++)
    	cin>>a[i];
    sort(a,a+n);
    int sum=0;
    int k;
    cin>>k;
    int i;
    for(i=n-1;i>=0;i--)
    {
    	sum+=a[i];
    	if(sum>=k)
    		break;
    }
    cout<<n-i;
    return 0;
}
