#include <iostream>
using namespace std;

int main()
{
    int m,n,h[1001] = {0},i,sum=0,max=0;
    cin >> m >> n;
    int a[m],c[m];
    for(i=0;i<m;i++)
    {
    	cin >> a[i];
    	h[a[i]]++;
    }
    for(i=0;i<m;i++)
    {
    	cin >> c[i];
    	h[c[i]]++;
    }
    for(i=0;i<1001;i++)
    {
   		max = max + i*h[i];
    	if(max <= n)
    	{
    		sum = sum + h[i];
    	//	max = max + i*h[i];
    	}
    	else
    	{
    		max = max - i*h[i];
    		break;
    	}
    		
    //	cout << max << endl;
    }
  // cout << i << endl;
   if( n > max)
    {
    	sum = sum + (n-max)/i;
  //sum = 0;
    	
    }
    if(sum > m)
    {
    	cout << m;
    }
    else
    {
    	cout << sum ;
    }
    return 0;
}
