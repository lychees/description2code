from math import sqrt; from itertools import count, islice

def prime(n):
    if n < 2: return False
    return all(n%i for i in islice(count(2), int(sqrt(n)-1)))


t = int(raw_input())
while t:
	t-=1
	y,z= map(int,raw_input().split())
	summ = 0
	while y<=z:
		if prime(y):
			summ+=y
		y+=1
	print summ
