tc=input()
lr=[]
while tc>0:
	num=input()
	st=raw_input()
	ll=st.split()
	ln=[]
	d={}
	for i in ll:
		ln.append(int(i))
	for i in ln:
		if d.has_key(i)==True:
			d[i]=d[i]+1
		else:
			d.setdefault(i,1)
	ld=d.values()
	val=max(ld)
	lans=[]
	for i in d.keys():
		if d[i]==val:
		   lans.append(i)
	for i in range(0,len(lans)):
		for j in range(i+1,len(lans)):
			if lans[i] <lans[j]:
				lans[i],lans[j]=lans[j],lans[i]
	lr.append(lans)	
	tc=tc-1
