#include<string.h>
#include <iostream>
using namespace std;
int mod(int x)
{
	if(x<0)return -x;
	return x;
}
int main()
{
    char a[1000],b[1000];
    int n,t,i,j;
    cin>>t;
    while(t--)
    {
    	j=0;
    	scanf("%s",&a);
    	int k=0;
    	n=strlen(a);
    	for(i=n-1;i>=0;i--){b[k]=a[i];k++;}
    	
    	
    	//strrev(b);
    	for(i=1;i<n;i++)
    	{if(mod(a[i]-a[i-1])!=mod(b[i]-b[i-1])){j=1;break;}
    	
    	}
    	if(j)cout<<"Not Funny\n";
    	else cout<<"Funny\n";
    }
    return 0;
}
