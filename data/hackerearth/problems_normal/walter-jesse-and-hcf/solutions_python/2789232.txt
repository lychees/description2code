d = input()

def gcd(a,b):
	if(a>b):
		return gcd(b,a)
	else:
		r = b%a
		if(r==0):
			return a
		else:
			return gcd(r,a)

while(d>0):
	x,y,k = map(int,raw_input().split())
	g = gcd(x,y)
	t = 0
	r = 0
	for i in range(1,g+1):
		if(g%i == 0):
			t = t + 1
		if(t == k):
			r = i
			break
	if(t == k):
		print(g/r)
	else:
		print("No crymeth today")
	d = d - 1
