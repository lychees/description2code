#include<iostream>
#include<cmath>
#include<vector>
#define LLD long long int; 
using namespace std;
int main()
{
    int n,t,i,j,k,temp=0;
    scanf("%d",&t);
    while(t--)
    {        int a[3000];
             scanf("%d",&n);
             int m,x=n;
             i=0;
             while(x)
             {
                     a[i]=x%10;
                     x=x/10;
                     i++;
             }
            for(k=n-1;k>=1;k--)
             {
                              for(j=0;j<i;j++)
                              {
                                  m=a[j]*k+temp;
                                  a[j]=m%10;
                                  temp=m/10;
                              }
                              
                              while(temp)
                              {
                                         a[j]=temp%10;
                                         temp=temp/10;
                                         j++;
                              }
                              i=j;
             }
            int ans=0;
            for(int z=(i-1);z>=0;z--)
              {
                     if(a[z]==4 || a[z]==7)
                     ans++;
              }
              cout<<ans<<endl;
    }
    //system("pause");
    return 0;
}
