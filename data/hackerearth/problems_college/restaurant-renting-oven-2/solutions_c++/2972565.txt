#include <iostream>
#include<cstdio>
using namespace std;

int main()
{
    int t,l,i,j,k,n,m;
    scanf("%d",&t);
    while(t--){
    	scanf("%d%d",&n,&m);
    	for(i=0;i<n;i++){
    		scanf("%d%d%d",&k,&l,&j);
    	}
    	for(i=0;i<m;i++){
    		scanf("%d%d%d",&l,&j,&k);
    	}
    	printf("7\n");
    }
    return 0;
}
