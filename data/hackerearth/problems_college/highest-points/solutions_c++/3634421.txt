#include <iostream>
using namespace std;

int main()
{
    int n,i,big;
    cin>>n;
    int a[n],w[n],t[n],l[n];
    for(i=0;i<n;i++)
      cin>>w[i]>>t[i]>>l[i];
    for(i=0;i<n;i++)
    {
    	a[i]=3*w[i]+t[i]-l[i];
    }
    big=a[0];
    for(i=1;i<n;i++)
    {
    	if(a[i]>=big)
    	  big=a[i];
    }
    cout<<big<<endl;
    return 0;
}
