
for _ in xrange(input()):
	stk = []
	n,p = map(int,raw_input().split())
	prev = -1
	stk.append(p)
	for i in xrange(n):
		inp = raw_input().split(' ')
		if inp[0] == "P":
			stk.append(int(inp[1]))
			prev = -1
		else:
			if prev == -1:
				prev = stk.pop()
			else:
				stk.append(prev)
				prev = -1
		#print stk
	print "Player",stk[-1]
