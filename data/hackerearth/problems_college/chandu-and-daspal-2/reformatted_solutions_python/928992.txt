from sys import stdin
from fractions import gcd
for i in range(0, int(stdin.readline())):
    a = stdin.readline().split()
    c = ''
    if a[2] == 'Chandu':
        c = 'Daspal'
    else:
        c = 'Chandu'
    a[0] = int(a[0])
    a[1] = int(a[1])
    win = False
    while (a[0] != 0 and a[1] != 0):
        b = gcd(a[0], a[1])
        a[0] -= b
        a[1] -= b
        win = (not win)
    if not win:
        print c

        print a[2]
