#include <iostream>
#include <cstdio>
#include <cstring>
#define SIZE 1000005
using namespace std;
double DP[SIZE];
int main(){
	long int n,i;
	double p;
	memset(DP,0.0,sizeof(DP));
	cin>>n>>p;
	DP[0] = 1;
	DP[1] = 0;
	DP[2] = p/100.0;
	DP[3] = 1-p/100.0;

	for(i=4;i<=n;i++){

		DP[i] = (p/100.0)*DP[i-2] + (1- p/100.0)*DP[i-3]; 

	}	
	
	//cout.setf(ios::showpoint);
	//cout<<DP[n]<<endl;
	printf("%f\n",DP[n]);
return 0;
}
