for tc in xrange(1, input() + 1):
    N = input()
    if N == 0:
        continue
    A = list(map(int, raw_input().split()))
    dp = [0] * N
    if N < 3:
        print "Case %d: %d" % (tc, max(A))
        continue

    dp[0] = A[0]
    dp[1] = max(A[0], A[1])
    for i in xrange(2, N):
        dp[i] = max(dp[i - 1], A[i] + dp[i - 2])

    print "Case %d: %d" % (tc, dp[-1])
