#include <bits/stdc++.h>

using namespace std;

#define BASE 37
#define MOD 10000007

long long b[100000];
long long hashVal[100000];

long long getHash(int l, int r) {
	return (hashVal[r] - (hashVal[l-1]*b[r-l+1]) % MOD + MOD) % MOD;
}

int main() {
    string S;
    cin >> S;
    
    b[0] = 1;
    for (int i = 1; i <= S.length(); i++) {
    	b[i] = (b[i-1]*BASE) % MOD;
    }
    hashVal[0] = 0;
    for (int i = 1; i <= S.length(); i++) {
    	hashVal[i] = (hashVal[i-1]*BASE + (S[i-1]-'a')) % MOD;
    }
    
    
    int Q;
    cin >> Q;
    for (int q = 0; q < Q; q++) {
    	int l1, r1, l2, r2;
    	cin >> l1 >> r1 >> l2 >> r2;
		
    	if (getHash(l1, r1) == getHash(l2, r2))
    		cout << "Yes" << endl;
    	else
    		cout << "No" << endl;
    }
    return 0;
}
