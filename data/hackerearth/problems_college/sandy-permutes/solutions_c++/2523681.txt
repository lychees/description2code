#include<bits/stdc++.h>
using namespace std;
int a[100002];
int b[100002];
int main()
{
	int i,j,k,l,n;
	scanf("%d%d",&n,&k);
	for(i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
	}
	{
		for(i=0;i<k;i++)
		{
			int s=i,t;
			t=a[i];
			for(j=i+1;j<n;j++)
			{
				if(t<a[j])
				{
					s=j;
					t=a[j];
				}
			}
			if(t==a[i])
			{
				k++;
				if(k>n)
				{
					k=n-1;
				}
			}
			else
			{
			l=a[i];
			a[i]=a[s];
			a[s]=l;
			}
		}
		for(i=0;i<n;i++)
		{
			printf("%d ",a[i]);
		}
	}
}
