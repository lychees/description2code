#include <iostream>
#include <vector>
using namespace std;

int main()
{
	int T;
	cin >> T;
	cin.ignore(256,'\n');
	for(int i = 0; i < T; ++i)
	{
		string s1;
		string s2;
		getline(cin,s1,'\n');
		getline(cin,s2,'\n');
		//cout << s1.size() << endl;
		//cout << s2.size() << endl;
		int list1[26];
		int list2[26];
		for(int j = 0; j < 26; ++j)
		{
			list1[j] = list2[j] = 0;
		}
		for(int j = 0; j < s1.size(); ++j)
		{
			if (s1[j] >= 'a' && s1[j] <= 'z')
			{
				++list1[s1[j]-'a'];
			}else if (s1[j] >= 'A' && s1[j] <= 'Z')
			{
				++list1[s1[j]-'A'];
			}
		}
		for(int j = 0; j < s2.size(); ++j)
		{
			if (s2[j] >= 'a' && s2[j] <= 'z')
			{
				++list2[s2[j]-'a'];
			}else if (s2[j] >= 'A' && s2[j] <= 'Z')
			{
				++list2[s2[j]-'A'];
			}
		}
		vector<char> v;
		vector<char> v2;
		int countodd = 0;
		for(int j = 0; j < 26; ++j)
		{
			int tmp = list2[j] - list1[j];
			if (tmp < 0)
			{
				tmp = -tmp;
			}
			if (tmp%2 == 1)
			{
				++countodd;
				v2.push_back('a'+j);
				--tmp;
			}
			while(tmp > 0)
			{
				v.push_back('a'+j);
				--tmp;
			}
		}
		if (v.empty() && v2.empty())
		{
			cout << "YES" <<endl;
		}else if (countodd > 1)
		{
			cout << "NO LUCK" << endl;
		}else if (countodd == 0)
		{
			for(int j = 0; j < v.size()/2; ++j)
			{
				cout << v[2*j];
			}
			for(int j = v.size()/2-1; j >= 0; --j)
			{
				cout << v[2*j+1];
			}
			cout << endl;
		}else if (countodd == 1)
		{
			for(int j = 0; j < v.size()/2; ++j)
			{
				cout << v[2*j];
			}
			for(int j = 0; j <v2.size();++j)
			{
				cout << v2[j];
			}
			for(int j = v.size()/2-1; j >= 0; --j)
			{
				cout << v[2*j+1];
			}
			cout << endl;
		}
	}
    //cout << "Hello World!" << endl;
    return 0;
}
