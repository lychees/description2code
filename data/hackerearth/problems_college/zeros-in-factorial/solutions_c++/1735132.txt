#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define max 1000000
ll res[max+1];
ll ans[max+1];

long long  calcc(long  n)
{
    // Initialize result
    int count = 0;

    // Keep dividing n by powers of 5 and update count
    for (int i=5; n/i>=1; i *= 5)
          count += n/i;

    return count;
}
ll calc(ll n)
{
    ll cnt=0,i=1;
    while(n/(5*i)>=1)
    {
        cnt=cnt+n/(5*i);
        i=i*5;
    }
    return cnt;

}
int main()
{
    ll i;
    res[0]=ans[0]=0;
    for(i=1;i<=max;i++)
    {
        res[i]=calc(i);
        ans[i]=ans[i-1]+res[i];
    }
   // for(i=0;i<13;i++)
        //printf("i=%d %d %d\n",i,res[i],ans[i]);
     //   cout<<"i="<<i<<" "<<res[i]<<"  "<<ans[i]<<endl;
    ll t,l,r;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld %lld",&l,&r);
        printf("%lld\n",ans[r]-ans[l-1]);
    }
    return 0;

}
