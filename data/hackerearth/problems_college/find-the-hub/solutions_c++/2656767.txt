#include <bits/stdc++.h>
using namespace std;
int arr[1000001];
int main()
{
	arr[1]=1;
	int x,n,j,i;
	for(i=4;i<=1000000;i+=2)
	arr[i]=1;
	for(j=3;j<=1000;j+=2)
	{
		i=j*j;
		while(i<=1000000)
		{
			arr[i]=1;
			i+=j;
		}
		
	}
	
    scanf("%d",&n);
    int largest=0;
    int a[n];
    for(i=0;i<n;i++)
    {
    	scanf("%d",&x);
    	a[i]=x;
    	if(arr[x]==0&&x>largest)
    	largest=x;
    }
    int c=0;
    for(i=0;i<n;i++)
    if(a[i]==largest)
    c++;
     printf("%d\n%d",largest,c);
    return 0;
}
