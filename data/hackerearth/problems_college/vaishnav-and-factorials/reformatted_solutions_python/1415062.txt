def find_lucky(input_number):
    result = 1
    for i in range(2, input_number+1):
        result = result * i
    result = str(result)
    count = 0
    for i in result:
        if i == '4':
            count = count + 1
        if i == '7':
            count = count + 1
    print count


if __name__ == '__main__':
    cases = int(raw_input())
    start = 1
    while start <= cases:
        input_number = int(raw_input().strip())
        find_lucky(input_number)
        start = start + 1
