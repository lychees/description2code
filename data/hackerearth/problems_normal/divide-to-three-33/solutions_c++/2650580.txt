#include <iostream>
using namespace std;
typedef long long int lint;
lint a[40],n,ans,finalans=1000000000;
void function(lint a1,lint a2,lint a3,lint index){
	if(index==n){
		ans = max(a1,max(a2,a3));
		if(finalans>ans)
		finalans =ans;
	}
	else{
		function(a1+a[index],a2,a3,index+1);
		function(a1,a2+a[index],a3,index+1);
		function(a1,a2,a3+a[index],index+1);
	}
	}
int main()
{
    cin>>n;
    for(int i=0;i<n;i++){
    	cin>>a[i];
    }
    function(0,0,0,0);
    cout<<finalans;
    return 0;
}
