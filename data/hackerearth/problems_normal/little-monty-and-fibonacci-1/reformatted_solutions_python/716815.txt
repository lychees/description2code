f = [0]*1006
f[0] = 0
f[1] = 1
x = {}
for i in xrange(2, 1006):
    f[i] = f[i-1]+f[i-2]
for i in xrange(1, 1006):
    ans = 0
    for j in xrange(1, 1006):
        if (f[j] % f[i]) == 0:
            ans += 1
        x[(i, j)] = ans
t = input()
while t:
    t -= 1
    n = input()
    ans = 0
    for i in range(1, n+1):
        ans += x[(i, n)]
    print ans
