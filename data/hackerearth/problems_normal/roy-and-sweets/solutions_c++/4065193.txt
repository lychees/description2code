#include <bits/stdc++.h>
using namespace std;
#define M 1000001

int main()
{
	int* arr = new int [M]();
    //cout << "Hello World!" << endl;
    int N; cin>>N; 
    register int S,i;
    int ma=0; 
    int mi=M+1;
    for (i = 0; i < N;++i) {
    	cin>>S;
    	cin>>arr[S];
    	if ( S > ma) ma = S;
    	if ( S < mi) mi = S;
    }
    //not needed
    for ( i=M-1;i > ma; --i ) {
    	arr[i]=0;
    }
    int mv = arr[ma];
    for (i=ma-1;i >=mi; --i) {
    	if (arr[i] > mv) 
    		mv = arr[i];
    	arr[i] = mv;
    }
    for (i=0;i < mi; ++i ) {
    	arr[i]=mv;
    }
    long int R = 0;
    int G; cin >>G;
    register int m;
    for ( i = 0; i < G; ++i) {
    	cin >>m;
    	R += arr[m]/m;
    }
    
    cout<<R*100;
    delete [] arr;
    return 0;
}
