#include<iostream>
#include<strings.h>
using namespace std;
int main()
{
    int t;
    string s1;
    cin>>t;
    while(t--)
    {
        cin>>s1;
        string ans="";
        ans=ans+s1[0];
        int len=s1.size();
        for(int i=0;i<len-1;i++)
        {
            if(s1[i]==s1[i+1])
                continue;
            else
                ans+=s1[i+1];
        }
        cout<<ans<<"\n";
    }
    return 0;
}
