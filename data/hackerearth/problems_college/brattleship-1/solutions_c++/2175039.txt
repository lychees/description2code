#include <stdio.h>
#include <iostream>

using namespace std;

void func()
{
	int r, c, w;
    cin>>r>>c>>w;
	int a;
	if(w==1) a=r*c;
	else
	{
		a=(r-1)*(c/w);
		if(c%w==0) a+=c/w+w-1;
		else a+=c/w+w;
	}
    cout<<a<<endl;
}
int main()
{
	int t,i;
    cin>>t;
	for(i=1; i<=t; i++)
	{
		printf("Case #%d: ", i);
		func();
	}
	return 0;
}
