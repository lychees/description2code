#include <algorithm>
#include <iostream>
#include <math.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
using namespace std;

int main() {
  int i, j, n, m, v, k;
  while (~scanf("%d%d%d", &n, &m, &v)) {
    if (m < n - 1 || (n - 1) * (n - 2) / 2 + 1 < m) {
      printf("-1\n");
      continue;
    }
    if (v == 1)
      k = 2;
    else
      k = 1;
    printf("%d %d\n", k, v);
    m--;
    for (i = 1; i <= n; i++) {
      if (i == k)
        continue;
      for (j = i + 1; j <= n; j++) {
        if (i != k && j != k) {
          printf("%d %d\n", i, j);
          m--;
        }
        if (!m)
          break;
      }
      if (!m)
        break;
    }
  }
}
