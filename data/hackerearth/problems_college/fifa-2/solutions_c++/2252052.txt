#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int n;
    cin >> n;
    
    int *A=new int[n];
    int *B=new int[n];
    int *C=new int[100000];
    
    for(int i=0;i<100000;i++)
    {
    	C[i]=0;
    }
    for(int i=0;i<n;i++)
    {
    	cin >> A[i] >> B[i];
    	C[A[i]-1]++;
    }
    
    long ans=(n-1);
    for(int i=0;i<n;i++)
    {
    	long c=C[B[i]-1];
    	cout << ans+c << " " << ans-c << endl;
    }
}
