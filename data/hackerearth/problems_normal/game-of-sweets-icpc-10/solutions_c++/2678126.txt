using namespace std;          
const int N = int(1e5);
int a[N+5];
int n;
bitset <N+5> d,dp;
int b[N+5]; 

int main () {
	//freopen("in","r",stdin);
	//freopen("ans","w",stdout);
	scanf("%d",&n);
	for(int i=1;i<=n;i++){
	    scanf("%d",&a[i]);
	}
	sort(a+1,a+n+1);
	for(int i=1;i<n;i++){
		b[i] = a[i+1] - a[i];
	}
	n--;
	for(int i=1;i<=n;i++){
		d = d << b[i];
		dp |= d;
		dp[b[i]] = 1;
		d[b[i]] = 1;
	}
	int ans = 0;
	for(int i=0;i<=N;i++){
		if(dp[i]) {
			ans++;
		}
	}
	cout << ans;

return 0;
}
