#include<iostream>
#include<string.h>
using namespace std;
main()
{
	string ar;

	while (getline(cin, ar))
	{
		string s = ""; int p = 0;
		for (int i = 0; i<ar.size(); i++)
		{
			if (p == 1)
			{
				s = s + ar[i];
				continue;
			}
			if (i<ar.size() - 1 && ar[i] == '/'&&ar[i + 1] == '/')
			{
				s = s + "//";
				i++;
				p = 1; continue;
			}
			if (i<ar.size() - 1 && ar[i] == '-'&&ar[i + 1] == '>')
			{
				s = s + ".";
				i++;
				continue;
			}
			s = s + ar[i];
		}
		cout << s << endl;
	}
}
