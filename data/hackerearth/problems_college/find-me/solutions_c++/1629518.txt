#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,a[10000],sam,flag=0,cnt=0;
    	cin>>n>>sam;
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    		if(a[i]==sam)
    		{
    			flag=1;
    		}
    		if(a[i]>sam)
    		cnt++;
    	}
    	if(flag==1)
    	cout<<"YES "<<cnt<<endl;
    	else
    	cout<<"NO 0"<<endl;
    }
    return 0;
}
