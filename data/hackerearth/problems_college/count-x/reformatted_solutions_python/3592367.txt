t = int(raw_input())
for _ in range(t):
    lst = []
    s = raw_input()
    lst = s.split("-")
    counter = []
    for i in lst:
        count = 0
        for j in i:
            count += 1
        counter.append(count)
    a = ''.join(map(str, counter))
    b = ''
    for i in range(1, len(a)-1):
        b = b+a[i]
    print b
