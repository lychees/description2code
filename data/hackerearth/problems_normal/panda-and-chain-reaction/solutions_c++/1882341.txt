#include <iostream>
#include <cassert>
using namespace std;
#define MOD 1000003
#define MAXN 100000
#define MAXX 1000000000000000000LL
long long a[MOD+1];
int main()
{
	a[0]=1;
	for(int i=1;i<=MOD;i++)
		a[i] = (a[i-1]*(long long)i)%MOD;
	int Tc;
	long long N, A;
	cin >> Tc;
	assert(Tc>0 and Tc<=MAXN);
	while(Tc-- > 0){
		cin >> N >> A;
		assert(N>0 and N<=MAXX);
		assert(A>0 and A<=MAXX);
		A = A % MOD;
		if(N>MOD)
			N = MOD;
		cout << (a[N]*A)%MOD << endl;
	}
    return 0;
}
