#include <bits/stdc++.h>
#define ll long long
using namespace std;
char str[1000010];
ll int a[1000010];
int main()
{
	ll int test;
	scanf("%lld",&test);
	while(test--)
	{
		ll int k;
		scanf("%lld",&k);
		scanf("%s",str);
		a[0]=0; //Because index 0 has no previous,appending a 0 at the beginning
		ll int i,l=0;
		for(i=0;str[i]!='\0';i++)
		{
			a[i+1]=(str[i]-'a' +1);
			//Prefix sum
			a[i+1]+=a[i];
			l++;
		}
		ll int n=l+1;
		//Prefix sum
		//~ for(i=1;i<n;i++)
		//~ {
			//~ a[i]+=a[i-1];
		//~ }
		ll int ans=0;
		for(i=0;i<(n-1);i++)
		{
			ll int *p=lower_bound(a+i+1,a+n,a[i]+k);
			//~ cout<<"here "<<*p<<endl;
			if(p==a+n)
			continue;
			else if((*p-a[i])==k)
			{
				ans++;
			}
		}
		printf("%lld\n",ans);
	}
	return 0;
}
