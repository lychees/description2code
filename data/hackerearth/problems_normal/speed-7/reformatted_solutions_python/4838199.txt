result = []
for tc in range(int(raw_input())):
    total_top_speed_cars = 0
    least_speed = None
    no_of_cars = int(raw_input())

    for car_speed in map(int, raw_input().split()):

        if least_speed is None or no_of_cars == 1:
            least_speed = car_speed
            total_top_speed_cars = 1
            continue

        if car_speed <= least_speed:
            total_top_speed_cars += 1
            least_speed = car_speed

    result.append(total_top_speed_cars)

for i in result:
    print i
