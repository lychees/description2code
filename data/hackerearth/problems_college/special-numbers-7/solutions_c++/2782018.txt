#include <bits/stdc++.h>
using namespace std;
bool prime(long long int n)
{
   long long  int coun=0;

    if(n == 1)
    {
        return false;
    }
    else if(n == 2)
    {
        return true;
    }
    else if(n%2 == 0)
    {
        return false;
    }
    else
    {
        for(int i=3;i<=sqrt(n);i+=2) // here we start checking from 3 and then 5,7,9.... becoz we know that 2 is the only even prime
            // and rest all the even numbers are not prime numbers.
        {
            if(n%i == 0)
            {
                coun++;
            }

        }
        if(coun>0)
        {
            return false;
        }
        else
        {
            return true;
        }
    }
}
int ans(int n)
{
    int coun=0;
    for(int i=2;i<n;i++)
    {
        if(n%i==0)
        {
            if(prime(i)) coun++;
        }
    }
    if(coun>=3) return true;
    else return false;
}
int main()
{
    vector<int> v;
