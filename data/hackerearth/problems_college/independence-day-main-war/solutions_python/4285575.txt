T = input()
for _ in xrange(T):
	N = input()
	lx,ly = [],[]
	for _ in xrange(N):
		l = map(float,raw_input().split())
		lx.append(l[0])
		ly.append(l[1])
	print '({:.2f} , {:.2f})'.format(sum(lx)/N,sum(ly)/N)
