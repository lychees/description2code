#include <iostream>
using namespace std;

int main()
{	
	int t;
	cin>>t;
	int n;
	while(t--){
		cin>>n;
		int count=0;
	    for(int i=0;i<10;i++){
	    	for(int j=0;j<10;j++){
	    		if(i*10+j*12==n){
	    			cout<<i+j<<endl;
	    			count=1;
	    			break;
	    		}
	    	}
	    	if(count==1){
	    		break;
	    	}
	    }
	    if(count==0){
	    	cout<<-1<<endl;
	    }
	}
    return 0;
}
