#include<bits/stdc++.h>
using namespace std;
int main()
{
    int i,e,t,x,y;
    set <int> s;
    cin>>t;
    while(t>0)
    {
        cin>>e;
        for(i=0;i<e;i++)
        {
            cin>>x>>y;
            s.insert(x);
            s.insert(y);
        }
        cout<<s.size()<<"\n";
        s.erase(s.begin(),s.end());
        t--;
    }
    return 0;
}
