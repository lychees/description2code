#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int a,b,c;
    	cin>>a>>b>>c;
    	int max1,max2;
    	if(a<b)
    		max2=b;
    	else
    		max2=a;
    	int temp=a+b-max2;
    	if(temp<c)
    		max1=c;
    	else
    		max1=temp;
    	int minofmax,res;
    	if(max1<max2)
    	{
    		minofmax=max1;
    		res=2*minofmax+1;
    	}
    	else if(max1>max2)
    	{
    		minofmax=max2;
    		res=2*minofmax+1;
    	}
    	else
    	{
    		res=2*max1;
    	}
    	if(a==0&&b==0&&c==0)
    	{
    		cout<<0<<endl;
    	}
    	else
    	{
    		cout<<res<<endl;
    	}
    }
}
