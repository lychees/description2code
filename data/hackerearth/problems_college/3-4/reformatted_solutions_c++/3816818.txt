#include <iostream>
#include <string.h>
using namespace std;
int main() {
  long long int t, n, k, a[100000] = {0}, i, j;
  cin >> t;
  while (t--) {
    cin >> n >> k;
    for (i = 0; i < k; i++) {
      cin >> j;
      a[j]++;
    }
    for (i = 1; i <= n; i++) {
      if (a[i] % 2 != 0) {
        cout << i << "\n";
      }
    }
    memset(a, 0, sizeof(a));
  }
  return 0;
}
