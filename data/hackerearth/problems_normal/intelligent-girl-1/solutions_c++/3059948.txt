#include <iostream>
#include<string.h>
using namespace std;
 
int main()
{
    string str;
    int even=0;
   cin>>str;
   
   int l=str.length();
   
  
    for(int i=0;i<l;i++)
    {
    	even=0;
	    for(int j=i;j<l;j++)
	    {
	    if((str[j]%2)==0)
	    even++;
	    }
    cout<<even<<" ";
    
    } 
    
  return 0;  
    
}
