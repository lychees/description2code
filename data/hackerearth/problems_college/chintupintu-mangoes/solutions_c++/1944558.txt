#include<bits/stdc++.h>
using namespace std;
int main()
{
	int test;
	cin>>test;
	while(test--)
	{
		long long x,h,d,tmp;
		int flag=1; //flag=1 ans-pintu ...else ans chintu
		cin>>x>>h>>d;
		tmp=h;
		if(!(x % 2))
		flag=1;
		else
		{
			for(int i=2;i<=sqrt(h);i++)
			{
				if(h % i==0)
				{
					tmp=i;
					break;
				}
			}
			long long height_div=h/tmp;
			if(height_div>=d)
			flag=0;
		}
		if(flag)
		cout<<"Pintu"<<endl;
		else
		cout<<"Chintu"<<endl;
	}
			
}
