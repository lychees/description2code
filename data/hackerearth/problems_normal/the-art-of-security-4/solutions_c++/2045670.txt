#include <bits/stdc++.h>
using namespace std;
int arr[13], n, ans = 0, entrance1, entrance2;
void solve(int num) {
	if(num == n){
		ans++;
		return;
	}
	
	if (entrance1 + arr[num] <= entrance2) {
		entrance1 += arr[num];
		solve(num + 1);
		entrance1 -= arr[num];
	}
	entrance2 += arr[num];
	solve(num + 1);
	entrance2 -= arr[num];
}
int main() {
	scanf("%d",&n);
	int sum = 0;
	for(int i=0; i<n; i++) {
		scanf("%d",&arr[i]);
		sum += arr[i];
	}
	sort(arr, arr+n);
	do{
		entrance1=0; entrance2=0;
		solve(0);
	}while(next_permutation(arr,arr+n));
	printf("%d %d\n",ans, sum);
	if (ans>sum)
		printf("We will win!\n");
	else
		printf("Got no way out!\n");
	return 0;
}
