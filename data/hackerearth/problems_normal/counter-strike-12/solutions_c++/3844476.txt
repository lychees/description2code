#include <algorithm>
#include <cstdio>
#include <iostream>
#include <vector>
#include <map>
#include <cstring>
#include <string>
#include <set>
#include <cmath>
#include <memory>
#include <functional>
#include <numeric>
#include <queue>
using namespace std;
int Nx[1007], Ny[1007], Mx[1007], My[1007];
int main()
{
    int t; cin >> t;
    while (t--) {
    	int cnt = 0;
    	int n, m, d; cin >> n >> m >> d;
    	for (int i = 0;i < n; ++i) {
    		cin >> Nx[i] >> Ny[i];
    	}
    	for (int i = 0;i < m; ++i) {
    		cin >> Mx[i] >> My[i];
    	}
    	for (int i = 0;i < m; ++i) {
    		for (int j = 0; j < n; ++j) {
    			int dd = abs(Nx[j] - Mx[i]) + abs(Ny[j] - My[i]);
    			if (dd <= d) {
    				++cnt;
    				break;
    			}
    		}
    	}
    	if (cnt > m/2) cout << "YES\n";
    	else cout << "NO\n";
    }
    return 0;
}
