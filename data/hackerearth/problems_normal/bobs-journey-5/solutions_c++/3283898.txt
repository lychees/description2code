#include <bits/stdc++.h>
using namespace::std;

int main()
{
//    ios_base::sync_with_stdio(false);
//    cin.tie(NULL);

    int t;
    cin>>t;

    while (t--)
    {
    	int n;
    	cin>>n;
    	set< char > S;
    	bool ok = true;
    	while (n--)
    	{
    		string ss;
    		cin>>ss;
    		char ch = ss[0];
    		//fflush(stdin);
            //putchar(ch);
    		if (S.find(ch) != S.end())
    			ok = false;
    		else
    			S.insert(ch);
    	}
    	puts( (ok ? "YES" : "NO") );
    }
    return 0;
}
