def process(base, s):
    s = list(s)
    for i in base:
        for j in range(2):
            if i not in s:
                return "Impossible"
            del s[s.index(i)]
    if len(s) == 0:
        return "Possible"
    return "Impossible"


T = input()
for i in xrange(T):
    base, s = raw_input().strip(" \n").split(" ")
    print process(base, s)
