#include <iostream>
using namespace std;

int processData(int data)
{
    int count = 0;
    int temp = data;
    
    while(temp > 0)
    {
        temp = temp & (temp - 1);
        count++;
    }
    
    return count;
}
int main(int argc, const char * argv[])
{
    int tcCount;
    int stonesTocollect;
    
    cin >> tcCount;
    
    int days[tcCount];
    
    for (int i = 0; i < tcCount; i++)
    {
        cin >> stonesTocollect;
        days[i] = processData(stonesTocollect);
    }
    
    for (int j = 0; j < tcCount; j++)
    {
        cout << days[j] << endl;
    }
    return 0;
}
