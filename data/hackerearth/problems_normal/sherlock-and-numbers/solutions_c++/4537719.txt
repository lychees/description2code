#include <iostream>
using namespace std;

int main()
{
	int T;
	cin >> T;
	while(T--) {
	    int N, K, P;
	    cin >> N >> K >> P;
	    int arr[K];
	    for(int i = 0; i < K; i++)
	        cin >> arr[i];
	    if(N - K < P) {
	        cout << -1 << endl;
	        continue;
	    }
        long long count = 0;
        for(int i = 0; i < K; i++) {
        	if(arr[i] <= P + count) count++;
        }
        cout << P + count << endl;
	}
    return 0;
}
