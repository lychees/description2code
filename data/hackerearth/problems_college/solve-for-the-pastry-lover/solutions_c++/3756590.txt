#include <iostream>
#include<cstring>
using namespace std;

int main()
{  int t,i,j;
   cin>>t;
   string s;
   long int n;
   
   while(t--){
   	   cin>>n;
   	   cin>>s;
   	   long int min=100000000;
   	   i=0;
   	   j=0;
   	   while(s[i+1]!='\0'){
   	   	j=i+1;
   	   	 while(s[j]!='\0'){
   	   	 	if(s[j]==s[i]){
                if(j-i<min)
                min=j-i;
   	   	 		break;
   	   	 	}
   	   	 	j++;
   	   	 }
   	   	 i++;
       }
        if(min==100000000)
        cout<<"-1"<<endl;
        else
        cout<<min<<endl;
   }
    return 0;
}
