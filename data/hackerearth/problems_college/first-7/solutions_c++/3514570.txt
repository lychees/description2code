#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    int t;
    ll n;
    cin>>t;
    while(t--)
    {
              cin>>n;
              double ans=0.0;
              ans+=(2*n-1.0)*n*n;
              ans+=(4.0*(n-2)*(n-1)/3+2.0*(n-1))*n;
              ans/=(n*n);
              //ans+=1;
              printf("%.6lf\n",ans);
    }
     return 0;         
}
