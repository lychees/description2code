#include<bits/stdc++.h>
using namespace std;
int n;
string st;
char s[12];
unordered_map < string, vector<int> > m;
unordered_map < string, vector<int> >::iterator it;
vector<int>::iterator ii;
int main()
{
	int L;
	scanf("%d",&n);
	int i,j,k;
	for(i=0;i<n;i++)
	{
		scanf("%s",s);
		L=strlen(s);
		for(j=0;j<L;j++)
		{
			string st("");
			for(k=j;k<L;k++)
			{
				st.push_back(s[k]);
				m[st].push_back(i);
			
			}
		}
	}
	for(it=m.begin();it!=m.end();it++)
	{
		ii=unique((it->second).begin(),(it->second).end());
		(it->second).resize(distance((it->second).begin(),ii));
	}
	int q;
	scanf("%d",&q);
	while(q--)
	{
		int l,r;
		scanf("%d %d",&l,&r);
		scanf("%s",s);l-=1;r-=1;
		int ub=upper_bound(m[s].begin(),m[s].end(),r)-m[s].begin();
		int lb=lower_bound(m[s].begin(),m[s].end(),l)-m[s].begin();
		printf("%d\n",ub-lb);
	}
	return 0;
}
