#include<bits/stdc++.h>
using namespace std;
#define MOD 1000000007
int n,m;
int fans;
int dp[1005][1005];
int fun(int dwin,int mwin)
{
	if(dwin+mwin == n) {
		if(dwin > mwin) return fans++;
		else return 0;
	}
	int ans = 0,ans2 = 0;
	ans += fun(dwin+1, mwin);
	if(dwin >= m*(mwin+1)) ans2 += fun(dwin, mwin+1);
	return max(ans,ans2);
	
}
int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		//int n,m;
		cin >> n >> m;
	//	fun(1, 0 ); //)fun(dwin,mwin);
	//	cout << fans<<endl;
	//	fans = 0;
		dp[0][0] = 1;
		for(int i = 1; i <= n; i++){
			dp[i][0] = 1;
		}
		for(int i = 1; i <= n; i++){
			for(int j = 1; j < n; j++){
				if(i < j*m) continue;
				dp[i][j] = dp[i-1][j] + dp[i][j-1];
				dp[i][j] %= MOD;
			}
		}
		long long int ans = 0 ;
		for(int i = 0; i <= n && n-i >= i*m; i++){
			if(n-i > i)
			ans += dp[n-i][i];
			ans %= MOD;
		}
		memset(dp,0,sizeof dp);
		cout << ans << endl;
	}
	return 0;
}
