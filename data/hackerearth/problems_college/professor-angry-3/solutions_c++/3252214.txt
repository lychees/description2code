#include <iostream>
using namespace std;

int main()
{int t;
   // cout << "Hello World!" << endl;
   cin>>t;
   while(t--){
   	int n,k,a[1000],count=0;
   	cin>>n>>k;
   	for(int i=1;i<=n;i++){
   		cin>>a[i];
   	}
   	for(int i=1;i<=n;i++){
   		if(a[i]<=0){
   			count++;
   		}
   	}
   	if(count<k){
   		cout<<"YES"<<endl;
   	}
   	else{
   		cout<<"NO"<<endl;
   	}
   }
    return 0;
}
