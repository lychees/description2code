#include<iostream>
using namespace std;
int R,G,B,Y,C,M,W,BL;
int main()
{
    int t,r,g,b;
    ios::sync_with_stdio(false);
    cin>>t>>r>>g>>b;
    int r1,g1,b1;
    for(int i=0;i<t;i++)
    {
        r1=(i/r)%2;
        g1=(i/g)%2;
        b1=(i/b)%2;
        if(r1==0&&g1==0&&b1==0)
            BL++;
        else if(r1==1&&g1==0&&b1==0)
            R++;
        else if(r1==0&&g1==0&&b1==1)
            B++;
        else if(r1==0&&g1==1&&b1==0)
            G++;
        else if(r1==1&&g1==1&&b1==0)
            Y++;
        else if(r1==0&&g1==1&&b1==1)
            C++;
        else if(r1==1&&g1==0&&b1==1)
            M++;
        else if(r1==1&&g1==1&&b1==1)
            W++;
    }
    cout<<R<<' '<<G<<' '<<B<<' '<<Y<<' '<<C<<' '<<M<<' '<<W<<' '<<BL<<endl;

}
