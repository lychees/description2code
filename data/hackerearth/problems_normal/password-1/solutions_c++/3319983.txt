#include<iostream>
#include<cstring>
using namespace std;
main()
{
	int n,i,j,b,l,c;
	string s[2000];
	string k;
	cin>>n;
	for(i=0;i<n;i++)
	cin>>s[i];
	for(i=0;i<n;i++)
	{
		
		l=s[i].length();
			for(j=0,b=l-1;j<s[i].length();j++,b--)
			{
				k[j]=s[i][b];
			}
		
		for(c=i+1;c<n;c++)
		{
			for(j=0;j<l;j++)
			{
				if(k[j]!=s[c][j])
				break;
			}
			if(j==l)
			break;
		}
		if(c!=n)
		{
			cout<<l<<" ";
			cout<<s[c][l/2];
			
			break;
		}
		else
		continue;
	}
	
	return 0;
}
