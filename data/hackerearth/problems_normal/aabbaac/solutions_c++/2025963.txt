typedef long long ll;
using namespace std;
const int MAXN = 55;
int n, m;
string s[MAXN];
char rec(ll pos, int cur, ll len) {
    len -= s[cur].length();
    if (pos >= len) {
        return s[cur][pos - len];
    }
    len /= 2;
    if (pos >= len) {
        return rec(len + len - pos - 1, cur - 1, len);
    }
    else {
        return rec(pos, cur - 1, len);
    }
}
void solve() {
    cin>>n>>m;
    for (int i = 0; i < n; i++) {
        cin>>s[i];
    }
    for (int i = 0; i < m; i++) {
        ll len = 0, pos;
        cin>>pos;
        for (int j = 0; j < n; j++) {
            len *= 2;
            len += s[j].length();
            if (len > pos) {
                cout<<rec(pos, j, len);
                break;
            }
        }
    }
    cout<<"\n";
}
int main() {
    int test;
    cin>>test;
    while (test--) {
        solve();
    }
    return 0;
}
