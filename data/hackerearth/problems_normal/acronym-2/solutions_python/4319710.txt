from __future__ import print_function
k = input(); ak = [];
while(k > 0):
	s = raw_input(); ak.append(s); k -= 1;
n = input(); an = raw_input().split(" "); ll = [];
for i in range(0, n):
	if(an[i] not in ak):	ll.append(str(an[i])[0].swapcase());
	
lllen = len(ll);
for i in range(0, lllen - 1):
	print(ll[i], '.', sep = "", end = "");
print(ll[lllen - 1], end = "\n");
