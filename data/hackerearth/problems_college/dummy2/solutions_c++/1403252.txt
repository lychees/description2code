
#include <string>
#include <iostream>
#include <vector>

using namespace std;
int main()
{
	int t;
	string str,temp,Result;

	cin>>t;
	while(t--)
	{
		cin>>str;
		Result.clear();
		for(int i=0;i<str.size();i++)
		{
			if(str[i]>='0' && str[i]<='9')
			{
				int x=str[i]-'0';
				temp.clear();
				x--;
				for(int j=0;j<x;j++)
					temp.append(Result.substr(0,Result.size()));
				Result.append(temp);
			}
			else
				Result.push_back(str[i]);
		}
		cout<<Result<<endl;		
	}
	return 0;
}
