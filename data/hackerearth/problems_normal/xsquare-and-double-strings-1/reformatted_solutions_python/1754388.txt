
def double_string(S):
    l = len(S)
    A = {}
    flag = False
    for i in range(0, 26):
        A[i] = 0
    for i in range(0, len(S)):
        A[ord(S[i])-97] += 1
    for i in range(0, 26):
        if A[i] >= 2:
            flag = True
            break
    if not flag:

    else:
        return "Yes"
