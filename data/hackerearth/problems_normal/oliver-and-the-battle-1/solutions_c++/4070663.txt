#include <iostream>
using namespace std;
#include<queue>
#include<utility>
int a[1003][1003];
int v[1003][1003];
int main()
{
	int i,j,t,n,m,k,mx,mt,mm,l;
    cin>>t;
    int tx[]={-1,-1,-1,0,0,1,1,1};
    int ty[]={-1,0,1,-1,1,-1,0,1};
    while(t--){
    	queue<pair<int, int> > q;
    	cin>>n>>m;
    	for(i=1;i<=n;i++){
    		for(j=1;j<=m;j++){
    			cin>>a[i][j];
    			v[i][j]=0;
    		}
    	}
    	mx=0;
    	mm=0;
    	mt=0;
    	for(i=1;i<=n;i++){
    		for(j=1;j<=m;j++){
    			if(v[i][j]==0 && a[i][j]==1){
    				q.push(make_pair(i,j));
    				mm++;
    				mt=0;
    				v[i][j]=1;
    				while(!q.empty()){
    					mt++;
    					int ii=q.front().first;
    					int jj=q.front().second;
    					q.pop();
    					for(l=0;l<8;l++){
    						int ni=ii+tx[l];
    						int nj=jj+ty[l];
    						if(nj>0 && nj<=m && ni>0 && ni<=n ){
    							if(a[ni][nj]==1 && v[ni][nj]==0	){
    								q.push(make_pair(ni,nj));
    								v[ni][nj]=1;
    							}
    						}
    					}
    				}
    				if(mt>mx){
    					mx=mt;
    				}
    			}
    		}
    	}
    	cout<<mm <<" "<<mx<<endl;
    }
    return 0;
}
