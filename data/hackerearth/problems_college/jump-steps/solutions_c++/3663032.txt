#include <bits/stdc++.h>
#define ll long long int
using namespace std;

int main()
{
    ll tc,n,i,c;
    scanf("%lld",&tc);
    while(tc--)
    {
        c=0;
        scanf("%lld",&n);
        while(n)
        {
            n=(n&(n-1));
            c++;
        }
        printf("%lld\n",c);
    }
    return 0;
}
