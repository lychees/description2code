n = int(raw_input())
a = raw_input().split(' ')
s1 = 0
s2 = 0
s3 = 0
for i in range(1, n+1):
    if i % 3 == 1:
        s1 = s1 + int(a[i-1])
    if i % 3 == 2:
        s2 = s2 + int(a[i-1])
    if i % 3 == 0:
        s3 = s3 + int(a[i-1])
print s1, s2, s3
