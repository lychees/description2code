line=[]
f=int(raw_input())
i=0
while(i<2*f):
	m=raw_input()
	line.append(m)
	i+=1
	
for i in range(int(f)):
	sr_arr=[int(x) for x in line[2*i].split()]
	tr_arr=[int(x) for x in line[2*i+1].split()]
	k=(sr_arr[0]+1)*sr_arr[2]-sum(tr_arr)
	if(k<=1):k=1
	elif(k>sr_arr[1]):k='Impossible'
	print(k)
