#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t,i,j,ans,n,b;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d%d",&n,&b);
    	i=0;
    	while(n>=(i*b))
    	{
    		i++;
    	}
    	
    	//printf("%d\n",i);
    	if(n==(i*b))
    	ans=i*b;
    	else
    	{
    		int y=(i*b)+(b*(i-1));
    		if(n==y/2&&y%2==0)
    		ans=i*b;
    		else
    		{
    			//printf("%d  %d\n",i*b,b*(i-1));
    			if((i*b)-n<n-((i-1)*b))
    			ans=i*b;
    			else
    			ans=b*(i-1);
    		}
    	}
    	printf("%d\n",ans);
    	
    	
    	
    }    
    return 0;
}
