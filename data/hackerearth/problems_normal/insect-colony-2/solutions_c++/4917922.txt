#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
    
   int t;
   cin>>t;
   while(t--){
   	int n;
   	cin>>n;
   	int x=0;
   	while(n--){
   		int y;
   		cin>>y;
   		x=x^y;
   		if(x%2==0)
   		x=0;
   		
   	}
   	if(!x)
   	cout<<"Yes\n";
   	else
   	cout<<"No\n";
   }
    return 0;
}
