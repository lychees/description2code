#include<stdio.h>
#define mod 1000000007
int main(){
	long long int t, f[100001], i, no; scanf("%lld", &t); f[0] = 0, f[1] = 1, f[2] = 2, f[3] = 4;
	for(i = 4; i <= 100000; i++)	f[i] = ((f[i - 1] % mod) + (f[i - 2] % mod) + (f[i - 3] % mod)) % mod;
	while(t-- && scanf("%lld", &no))	printf("%lld\n", f[no]);
	return 0;
}
