#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ll int t[400]; //Triangular series
	ll int i,j,x,y,mc,d;
	t[0]=1;
	d=2;//difference to generate the series
	for(i=1;i<400;i++)
	{
		t[i]=t[i-1]+d;
		d++;
	}
	//~ for(i=0;i<10;i++)
	//~ cout<<t[i]<<" ";
	//~ cout<<endl;
	ll int div[400];//Count of divisors
	for(i=0;i<400;i++)
	{
		mc=0;//Mycount for number of divisors
		for(j=1;(j*j)<=t[i];j++)
		{
			if((t[i]%j)==0)
			{
				x=j;
				y=t[i]/j;
				if(x==y)
				{
					mc++;
				}
				else
				{
					mc+=2;
				}
			}
		}
		div[i]=mc;
		//~ cout<<t[i]<<" "<<mc<<endl;
		//~ if(mc>=100) //This tells me at index 383 value = 73920  has divisors 112 so I got magical value 400
		//~ {
			//~ cout<<i<<" "<<t[i]<<" "<<mc<<endl;
			//~ break;
		//~ }
	}
	//~ cout<<t[382]<<" "<<div[383]<<endl;
	ll int test;
	scanf("%lld",&test);
	while(test--)
	{
		ll int n;
		scanf("%lld",&n);
		for(i=0;;i++)
		{
			if(div[i]>n)
			{
				break;
			}
		}
		printf("%lld\n",t[i]);
	}
	return 0;
}
