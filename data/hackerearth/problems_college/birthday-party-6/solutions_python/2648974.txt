import math
T=input()
for i in xrange(T):
    M,N=map(int,raw_input().split())
    if (M+1)<N:
        print -1
    else:
        girls=math.factorial(M)
        num=math.factorial(M+1)
        den=math.factorial((M-N)+1)
        boys=math.factorial(N)
        ans=(girls * (num/(den*boys))*boys)
        print ans%1000000007
