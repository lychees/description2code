#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
    int T;
    scanf("%d", &T);
    char S1[51], S2[52];
    for(int i =0; i<T;i++){
    	char S1[51], S2[52];
    	int N;
    	scanf("%s",S1);
    	scanf("%s",S2);
    	scanf("%d",&N);
    	int j =0;
    	int MAX = 0;
    	while(S1[j] != NULL){
    		if(S1[j] != S2[j]){
    			MAX++;
    		}
    		j++;
    	}
    	int y = N - MAX;
    	if(y > 0){
    		if((y%2) == 0){
    			printf("Yes\n");
    		}
    		else{
    			printf("No\n");
    		}
    	}
    	else if(y == 0){
    		printf("Yes\n");
    	}
    	else{
    		printf("No\n");
    	}
    	
    }
    return 0;
}
