#include <bits/stdc++.h>
using namespace std;
typedef vector<int> vi;

vi z_function(string s){
  vi z;
  int n = s.length();
  z.assign(n, 0);
  for(int i = 1, l = 0, r = 0; i < n; i++){
      if(i <= r) z[i] = min(r - i + 1, z[i - l]);
      while(i + z[i] < n and s[z[i]] == s[i + z[i]]) ++z[i];
      if(i + z[i] - 1  > r) l = i, r = i + z[i] - 1;
  }
  return z;
}

int main(int argc, char const *argv[]) {
  int t;
  cin>>t;
  while(t--){
    string s;
    cin>>s;
    int k; cin>>k;
    k--;
    vi z = z_function(s);
    int n = s.length();
    int mx = 0;
    bool flg = 1;
    if(k >= n - 1) mx = n;
    for(int i = 1; i < n; i++){
      if(z[i] == n - i and mx >= n - i){
        cout<<s.substr(i)<<endl;
        flg = 0;
        break;
      }
      if(z[i] + i - 1 <= k) mx = max(mx, z[i]);
      else mx = max(mx, k - i + 1);
    }
    if(flg) cout<<"Puchi is a cheat!"<<endl;
  }
  return 0;
}
