int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n1;
        scanf("%d",&n1);
        int x1=sqrt(n1);
        int temp=x1*x1;
        if(temp==n1) printf("%d\n",2*x1-1);
        else if(n1<=temp+x1) printf("%d\n",2*x1);
        else printf("%d\n",2*x1+1);
    }

    return 0;
}
