#include <iostream>
using namespace std;

int main()
{
	int t;
	int n, k;
	int count;
	int a[10000];
	cin >> t;
	while(t--) {
		cin >> n >> k;
		count = 0;
		for (int i = 0; i < n; i++) {
			cin >> a[i];
			if (a[i] < 1)
				count++;
		}
		if (count >= k)
			cout << "NO" << endl;
		else cout << "YES" << endl;
	}
    
    return 0;
}
