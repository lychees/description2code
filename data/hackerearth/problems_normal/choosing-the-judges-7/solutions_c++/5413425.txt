#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int t, n, i, temp;
    cin >> t;
    temp = 0;
    
    while(t--){
    	temp++;
    	cin >> n;
    	vector<int> scores(n);
    	vector<long long> dp(n+2);
    	for(i = 0; i < n; i++)
    		cin >> scores[i];
    		
    	dp[0] = 0;
    	dp[1] = 0;
    	for(i = 2; i <= n+1; i++)
    		dp[i] = max(scores[i-2]+dp[i-2], dp[i-1]);
    		
    	cout << "Case " << temp << ": " << dp[n+1] << endl;
    		
    }
    return 0;
}
