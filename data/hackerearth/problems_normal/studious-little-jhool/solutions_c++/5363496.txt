#include <iostream>
using namespace std;

int main()
{
    int t, n, i, x, y;
    cin >> t;

    int dp[114];
    for ( i = 0; i < 10; i++ )
        dp[i] = -1;
    dp[10] = 1;
    dp[11] = -1;
    dp[12] = 1;
    for ( i = 13; i < 114; i++ ) {
        x = dp[i - 10];
        y = dp[i - 12];
        if ( x == -1 && y == -1 ) {
            dp[i] = -1;
        }
        else if ( x == -1 ) {
            dp[i] = dp[i - 12] + 1;
        }
        else if ( y == -1 ) {
            dp[i] = dp[i - 10] + 1;
        }
        else {
            dp[i] = min( x, y ) + 1;
        }

    }

    while ( t-- ) {
        cin >> n;
        cout << dp[n] << '\n';
    }
}
