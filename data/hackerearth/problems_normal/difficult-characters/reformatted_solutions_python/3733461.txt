from string import ascii_lowercase
from collections import OrderedDict
import operator
num_test_cases = int(raw_input())

for word in range((num_test_cases)):
    word = str(raw_input())

    count = 0
    dic = {}
    for c in ascii_lowercase:
        dic[c] = [count, 0]
        count = count+1

    for letter in word:
        dic[letter] = [dic[letter][0], dic[letter][1]+1]

    dic = sorted(dic.items(), key=lambda key: key[0], reverse=True)
    # dic=sorted(dic.values(), key=operator.itemgetter(0), reverse=False)
    dic.sort(key=lambda x: x[1][1])
    result = ''
    for i, j in dic:
        result = result + i + ' '

    print result
