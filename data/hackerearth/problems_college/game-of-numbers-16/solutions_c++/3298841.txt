#include <iostream>
using namespace std;
long long int solve(long long int n)
{long long int i,j,count=0;
 for(i=1;i<=n;i++)
     {
     for(j=i+1;j<=n;j++)
         {
         if((i*j)%(i+j)==0)count++;
     }
 }
 return count;
}
int main()
{
    long long int t;
    cin>>t;
    while(t--)
    {
    	long long int n;
    	cin>>n;
    	
    	cout<<solve(n)<<endl;
    }
    return 0;
}
