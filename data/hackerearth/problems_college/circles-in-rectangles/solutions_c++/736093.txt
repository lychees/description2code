#include <iostream>
#include <algorithm>
using namespace std;
typedef long long LL;

int main()
{
	LL t;
	cin>>t;
	while(t--)
	{
			LL a,b,c;
      		cin>>a>>b>>c;
      		a/=c;
      		b/=c;
      		LL mod = 1000000007;
      		a%=mod;
      		b%=mod;
      		cout<<(a*b)%mod<<endl;
	}
    return 0;
}
