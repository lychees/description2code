#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,i,a[1000000],c,large;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(i=0;i<n;i++)
    		cin>>a[i];
    	c=1;
    	large=a[n-1];
    	for(i=n-2;i>=0;i--)
    	{	
    		
    		if(a[i]>large)
    		{	
    			large=a[i];
    			c++;
    		}
    	}
    	cout<<c<<endl;
    	
    	
    }
    return 0;
}
