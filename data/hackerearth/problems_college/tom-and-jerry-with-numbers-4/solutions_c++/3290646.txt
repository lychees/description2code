#include <bits/stdc++.h>
#define ll long long int
#define mod (int)(1e9 + 7)

using namespace std;

int t, n, m1, m2;
int f[1000005];

bool jerryWin(int n) {
	if (n < m1) return false;
	if (f[n] != -1) return f[n];
	f[n] = ((n >= m1 && !jerryWin(n - m1)) || (n >= m2 && !jerryWin(n - m2)));
	return f[n];
}

int main()
{
	cin >> t;
	
	while (t--) {
		cin >> n >> m1 >> m2;
		memset(f, -1, sizeof(f));
		if (jerryWin(n)) {
			cout << "1\n";
		} else {
			cout << "0\n";
		}
		
	}	
	return 0;
}
