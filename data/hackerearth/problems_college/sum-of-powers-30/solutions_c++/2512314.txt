#include<bits/stdc++.h>
#define mx 1000005
#define mod 1000000007
#define ll long long
#define pb push_back
using namespace std;
bool a[mx];
vector<int>x;
void make()
{
	//x.pb(0);
    x.pb(1);
    for(int i=2;i<=1000;i++)
    {
        int j=i;
        while(j*i<mx){x.pb(j*i);j*=i;}
    }
    sort(x.begin(),x.end());
    for(int i=0;i<x.size()-1;i++)
        for(int j=i;j<x.size();j++)if(x[i]+x[j]<mx)a[x[i]+x[j]]=1;

}
int main()
{
    make();
    int  t;
    scanf("%d",&t);
    for(int i=1;i<=t;i++)
    {
        int n;
        scanf("%d",&n);
        if(a[n])printf("Yes\n");else printf("No\n");
    }


}
