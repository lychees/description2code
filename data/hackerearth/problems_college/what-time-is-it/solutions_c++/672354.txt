#include <cstdio>
#include <iostream>
#include <cmath>

using namespace std;

int main()
{
	float x,y,ans,res;
	char s[6];
	char c;
	scanf("%s",s);
	x = (int)s[0]-48;
	x = x * 10;
	y = (int)s[1]-48;
	x = x + y;
	if (x == 12) {
		x = 0;
	}
	//x = (x * 360)/12;
	ans = (int)s[3]-48;
	ans = ans * 10;
	y = (int)s[4]-48;
	ans = ans + y;
	if (ans == 60) {
		ans = 0;
	}
	//ans = (ans * 360)/60;
	x = 0.5 * (x * 60 + ans);
	ans = 6 * ans;
	ans = abs(ans - x);
	if (ans < 360-ans) {
		res = ans;
	} else {
		res = 360-ans;
	}
	printf("%.6f\n",res);
	
	return 0;
}
