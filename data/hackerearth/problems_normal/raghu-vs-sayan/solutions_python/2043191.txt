t=int(raw_input())
while t!=0:
	t-=1
	r,s,n=map(int,raw_input().split())
	a=map(int,raw_input().split())
	a.sort()
	rc=0
	sc=0
	for item in a:
		if item<=r:
			r-=item
			rc+=1
		if item<=s:
			s-=item
			sc+=1
		if item>r and item >s:
			break
	if rc>sc:
		print "Raghu Won"
	elif sc>rc:
		print "Sayan Won"
	else:
		print "Tie"
