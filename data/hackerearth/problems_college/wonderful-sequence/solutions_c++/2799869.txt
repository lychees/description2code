#include <bits/stdc++.h>
using namespace std;
long long int arr[1000002];
int main()
{
    int t;
    cin >> t;
    while(t){
    	int n,c=0,i;
    	cin >> n;
    	for(i=0;i<n;i++)
    	{
    		cin >> arr[i];
    	}
    	for(i=0;i<n-1;i++)
    	{
    		if(arr[i]>=arr[i+1])
    		 c++;
    	}
    	cout << c << endl;
    	t--;
    }
    return 0;
}
