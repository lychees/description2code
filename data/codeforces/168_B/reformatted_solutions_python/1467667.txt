s = []
while True:
    try:
        S = raw_input()
    except BaseException:
        break
    tt = S.split()
    if len(tt) == 0 or tt[0][0] != '#':
        if len(tt) == 0:
            s += [""]
        else:
            s += tt
        continue
    if len(s) > 0:
        print("".join(s))
    print(S)
    s = []
if len(s):
    print("".join(s))
