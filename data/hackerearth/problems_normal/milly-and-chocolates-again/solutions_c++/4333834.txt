#include <iostream>
using namespace std;
 
int main()
{
	int n;
	cin>>n;
	int arr[n];
	int prev;
	int cur;
	for(int i=0;i<n;i++)
	{
		cin>>cur;
		if(i>1)
		{
			arr[i] = arr[i-1] + prev;
		}
		else if(i==0)
		{
			arr[i] = 1;
		}
		else
		{
			arr[i] = prev + 1;
		}
		prev = cur;
	}
	int q;
	cin>>q;
	int l,r,m;
	int index;
	while(q-->0)
	{
		cin>>index;
		l = 0;
		r = n-1;
		if(arr[n-1]<=index)
		{
			cout<<n<<"\n";
			continue;
		}
		while(l<=r)
		{
			 m = l + (r-l)/2;
			 if(arr[m]<=index&&arr[m+1]>index)
			 {
			 	cout<<m+1<<"\n";
			 	break;
			 }
			 else if(arr[m]>index)
			 {
			 	r = m-1;
			 }
			 else
			 {
			 	l = m+1;
			 }
		}
	}
    return 0;
}
