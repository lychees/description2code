#include <iostream>
using namespace std;

int main()
{
	int n;
	cin>>n;
	
	int x[n], y[n];
	int ans[n][2];
	
	for(int i=0 ; i<n ; i++){
		cin>>x[i]>>y[i];
		ans[i][0] = 0;
		ans[i][1] = 0;
	}
	
	for(int i=0 ; i<n ; i++){
		for(int j=0 ; j<n ; j++){
			if(j==i){
				continue;
			} else {
				ans[i][0] += 1;
				if(x[i]==y[j]){
					ans[j][0] += 1;
				} else {
					ans[j][1] += 1;	
				}
			}
		}
	}
	
	for(int i=0 ; i<n ; i++){
		cout<<ans[i][0]<<" "<<ans[i][1]<<endl;
	}
	
	return 0;
}
