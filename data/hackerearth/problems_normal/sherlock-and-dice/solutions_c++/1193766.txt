#include <bits/stdc++.h>
#define LLD long long
double sum[55][2505];
int main()
{
	int T,M,N,K,i,j,k,tens;
	LLD total,possible;
	double ans;
	scanf("%d",&T);
	while(T--)
	{
		tens=total=0;
		scanf("%d %d %d",&M,&N,&K);
		sum[1][0]=0.000;
		memset(sum,0.0,sizeof(sum));
	    if(K>N*M)
	    printf("0.000 0\n");
	    else
	    {
	    for(i=1;i<=N && i<=K;i++)
	    sum[1][i]=(double)(1.0/N);
	    for(i=2;i<=M;i++)
	    {
	    	for(j=1;j<=N*M && j<=K;j++)
	    	{
	    		
	    		for(k=1;k<=N && k<=j;k++)
	    		{
	    			sum[i][j]=sum[i][j]+sum[i-1][j-k];
	    		}
	    		sum[i][j]=sum[i][j]/N;

	    	}
	    }
	    if(sum[M][K]==0.0)
	    printf("0.000 0\n");
	    else
	    {
	    while(sum[M][K]<1.0)
	    {
	    	sum[M][K]*=10;
	    	tens+=1;
	    }
	    printf("%.3lf %d\n",sum[M][K],tens);
	    }
	    }
		
	}
	return 0;
	
}
