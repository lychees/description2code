#include<bits/stdc++.h>
using namespace std;
int coun=0,sc[1001];
string s[1001],st;
int getsw()
{
	for(int i=0;i<coun;i++)
	{
		if(s[i]==st)
		return i;
	}
	return -1;
}
int main()
{
int n;
//freopen("input.txt","r",stdin);
	cin>>n;
	
	int temp,maxx=0,ind=0;
	
	for(int i=1;i<=n;i++)
	{
		cin>>st>>temp;
		if(getsw()==-1)
		 {
		 	s[coun]=st;
		 	coun++;		 	
		 }
		if(temp>0)
		{ int ins=getsw();
		//cout<<sc[ins]<<" op "<<temp<<endl;
			sc[ins]+=temp;
			  //cout<<sc[ins]<<" lk "<<temp<<endl;
			if(sc[ins]>maxx)
			{  
				maxx=sc[ins];
				ind=ins;
			}
		}
	}
	cout<<s[ind]<<endl;
	
}
