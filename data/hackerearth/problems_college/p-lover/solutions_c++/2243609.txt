#include<bits/stdc++.h>
using namespace std;

bool a[10000000];
int b[10];

vector <int> s;
void sieve() {
   int i,j;
   a[1]=true;
   
   for(i=2;i<=10000;i++) {
   	  if(!a[i]) {
   	  	for(j=i*i;j<10000000;j=j+i) {
   	  		a[j]=true;
   	  	}
   
   	  }
   }

   for(i=2;i<10000000;i++) {
      if(!a[i])
         s.push_back(i);
   }

}

int ispalin(int n) {
   int i=0,j,k;
   while(n!=0) {
       b[i]=n%10;
       i++;
       n=n/10;
   }

   for(j=0;j<=i/2;j++) {
      if(b[j]!=b[i-j-1]) {
         return 0;
      }
   }
   
   return 1;
}

int main() {

   int i,j,k,t;
   sieve();
   cin >> t;
   vector<int>::iterator it;
   while(t--) {
     cin >> k;
     for(it=s.begin();it!=s.end();it++) {
         if(*it>=k) {
            if(ispalin(*it)) {
               cout << *it << endl;
               break;
            }
         }
     }
   }
   return 0;
}
