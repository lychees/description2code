int ans = 0;

void brute(vector<int> v, int n, int s, int num) {
	 if (s == n) {
	 	ans++;
	 }
	 int sum = 0;
	 for (int i = s; i < n; i++) {
	 	  sum += v[i];
	 	  if (sum >= num) {
	 	  	  brute(v, n, i + 1, sum);
	 	  }
	 }
}

int main()
{
   ios::sync_with_stdio(false);
   int t;
   cin >> t;
   while (t--) {
   	      string s;
   	      cin >> s;
   	      vector<int> v((int)s.length());
   	      for (int i = 0; i < (int)v.size(); i++) {
   	      	   v[i] = s[i] - '0';
   	      }
   	      ans = 0;
   	      brute(v, (int)v.size(), 0, 0);
   	      cout << ans << endl;
   }
   return 0;
}
