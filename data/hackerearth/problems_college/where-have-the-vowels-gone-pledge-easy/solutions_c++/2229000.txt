#include <iostream>
using namespace std;

int main()
{
    int N,i=0;
    char str[10000];
    int count[5]={0};
    
    cin>>N;
    if(N<=10000)
    {
    	cin>>str;
    	
    	while(str[i]!='\0')
    	{
    		if(str[i]=='a')
    			count[0]++;
    		else if(str[i]=='e')
    			count[1]++;
    		else if(str[i]=='i')
    			count[2]++;
    		else if(str[i]=='o')
    			count[3]++;
    		else if(str[i]=='u')
    			count[4]++;
    		
    			
    		i++;
    	}
    	
    	for(i=0;i<5;i++)
    	{
    		if(count[i]==0)
    		{
    			cout<<"NO";
    			exit(0);
    		}
    	}
    	
    	cout<<"YES";
    }
    return 0;
}
