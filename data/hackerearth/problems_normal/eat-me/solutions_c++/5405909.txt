#include <iostream>
#include <vector>
using namespace std;
long long dp[1001][1001];

int main() {
	int i, j;

	dp[0][1] = 1;

	for (i = 1; i <= 1000; i++) {
		dp[0][i] = i;
		dp[i][1] = i + 1;
	}
	for (i = 1; i <= 1000; i++) {
		for (j = 2; j <= 1000; j++) {
			dp[i][j] = dp[i - 1][j];
			dp[i][j] += (dp[i][j - 1] - dp[i][j - 2]) % (long long)(1e+9 + 7);
			dp[i][j] %= (long long)(1e+9 + 7);
			dp[i][j] += (dp[i][j - 1] - dp[i - 1][j - 1]) % (long long)(1e+9 + 7);
			dp[i][j] %= (long long)(1e+9 + 7);

			if (dp[i][j] < 0)
				dp[i][j] += 1e+9 + 7;
		}
	}

	cin >> i >> j;

	cout << dp[i - 1][j] << endl;
	return 0;
}
