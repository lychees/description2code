#include <iostream>
using namespace std;

int main()
{
    int i=0;
    string s;
    cin>>s;
    while(s[i]!='?')
     i++;
    i++;
    while(s[i]!='=')
     {
     	cout<<s[i];
     	i++;
     }
    cout << ": " ;
    i++;
    
    while(s[i]!='&'||s[i+1]!='p')
     {
     	cout<<s[i];
     	i++;
     }
    cout << "\n"; 
    i++;
    
    while(s[i]!='=')
     {
     	cout<<s[i];
     	i++;
     }
    cout << ": " ;
    i++;
    
    while(s[i]!='&'||s[i+1]!='p')
     {
     	cout<<s[i];
     	i++;
     }
    cout << "\n"; 
    i++;
    
    while(s[i]!='=')
     {
     	cout<<s[i];
     	i++;
     }
    cout << ": " ;
    i++;
    
    while(s[i]!='&'||s[i+1]!='r')
     {
     	cout<<s[i];
     	i++;
     }
    cout << "\n"; 
    i++;
    
    while(s[i]!='=')
     {
     	cout<<s[i];
     	i++;
     }
    cout << ": " ;
    i++;
    
    while(s[i]!='&'||s[i+1]!='k')
     {
     	cout<<s[i];

     }

     {
     	cout<<s[i];
     	i++;

     	i++;
     }
