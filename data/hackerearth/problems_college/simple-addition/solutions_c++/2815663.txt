#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    long long n,x,y,b,a,i;
    while(t--)
    {
    	cin>>n;
    	if(n==1)
    	cout<<"0\n";
    	else
    	{
    		a=4294967295;
    		for(i=n/2;i<n;i++)
    		{
    			x=i;
    			y=n-i;
    		b=1;
    		while(x!=y)
    		{
    			if(y>x)
    			y=y-x;
    			else
    			x=x-y;
    			b++;
    		}
    		if(x==1)
    		a=(a>=b)?b:a;
    	}
    	cout<<a<<"\n";
    	}
    }
    return 0;
}
