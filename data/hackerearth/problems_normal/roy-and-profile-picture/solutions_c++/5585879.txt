#include <iostream>
using namespace std;

int main() {
	// your code goes here
	int l;
	cin>>l;
	int n;
	cin>>n;
	for(int i=1;i<=n;i++){
		int w,h;
		cin>>w>>h;
	    
	    if(w<l || h<l)

	    else

	}
	return 0;
}
