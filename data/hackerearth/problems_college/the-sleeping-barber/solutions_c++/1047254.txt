#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int n,p,t,i;
    cin>>t;
    while(t>0)
    {
    	cin>>n>>p;
    	int a[n],sum=0,count=0;
    	for(i=0;i<n;i++)
    	{
    	    cin>>a[i];
    	}
    	sort(a,a+n);
    	for(i=0;i<n;i++)
    	{
    		sum=sum+a[i];
    		count++;
    		if(sum>p)
    		{
    			sum=sum-a[i];
    			count--;
    			break;
    		}
    	}
    	printf("%d\n",count);
    	t--;
    }
    return 0;
}
