#include <iostream>
using namespace std;

int main()
{
    int n,i,size,k,l,cnt=0,j;
    long int arr[100000];
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>size;
    	cnt=0;
    	for(j=0;j<size;j++)
    	{
    		cin>>arr[j];
    	}
    	for(k=0;k<size;k++)
    	{
    		for(l=0;l<size-1;l++)
    		{
    			if(arr[l]>arr[l+1])
    			{
    				cnt++;
    			}
    		}
    		if(cnt!=0)
    		{
    			break;
    		}
    	}
    	if(cnt==0)
    	{
    		cout<<"YES"<<"\n";
    	}
    	else
    	{
    		cout<<"NO"<<"\n";
    	}
    }
    return 0;
}
