#include <iostream>
using namespace std;

int main()
{
	long long t,q,count,i,j;
    cin >> t;
    while(t--){
    	cin >> q;
    	count = 0;
    	for(i = 1;i <= q;i++){
    		for(j = i+1;j <= q;j++){
    			if((i*j) % (i+j) == 0){
    				count++;
    			}
    		}
    	}
    	cout << count << endl;
    }
    return 0;
}
