#include <iostream>
#include<cmath>
using namespace std;

int ar2[101]={0};


void Sive2()
{
    int i,j;

ar2[0]=ar2[1]=1;
for(i=2;i<=sqrt(101);i++)
{
    if(ar2[i]!=0){continue;}
    for(j=2;j*i<101;j++)
    {
        ar2[i*j]=1;
    }

}
}


int main()
{
    int T,N;
    cin>>T;
    Sive2();
    while(T--)
    {
        int i;
        cin>>N;
        int pa1[N],pa2[N];
        long long int sum1=0,sum2=0;
        for(i=0;i<N;i++)
        {
            
            cin>>pa1[i];
        }
        for(i=0;i<N;i++)
        {
            
            cin>>pa2[i];
        }
        for(i=0;i<N;i++)
        {
            //cout<<pa1[i]<<pa2[i]<<endl;
        
                if(ar2[pa1[i]]==1 && ar2[pa2[i]]==0)
                {
                sum1+=pa1[i];
                sum2+=pa2[i];
                }
                else if(ar2[pa1[i]]==1)
                {
                    sum1+=pa1[i];
                }
            else if(ar2[pa2[i]]==0)
            {
                sum2+=pa2[i];
            }
            else
            {
                sum1+=pa1[i];
                sum2+=pa2[i];
            }
        }
        if(sum1>sum2)
        {
            cout<<"The Composites\n";
        }
        else if(sum2>sum1)
        {
            cout<<"The Primes\n";
        }
        else
        {
            cout<<"Tie\n";
        }
    }
    return 0;
}
