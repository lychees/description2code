#include <iostream>
using namespace std;

long long int arr[1000005];
long long int a[1000005];
void prime()
{
	long long int p,i;
	for( p = 2; p <= 1000000; p++ ) {
		if( arr[p] == 0) {
			arr[p] = p;
			for( i = p*p; i <= 1000000; i+=p ) {
				if ( arr[i] == 0)
					arr[i] = p;
			}
		}
	}
	for(i = 2; i <= 1000000; i++) {
		a[arr[i]]++;
	}
}

int main()
{
	prime();
	int t;
	cin >> t;
	while(t--) {
		long long int n;
		cin >> n;
		cout << a[n] << endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
