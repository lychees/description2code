{
	if(l==n)
	{
		v=max(max(i,j),k);
		if(v<ans)
		  ans=v;
	}
	else
	{
		fun(i+a[l],j,k,l+1);
		fun(i,j+a[l],k,l+1);
		fun(i,j,k+a[l],l+1);		
		
	}
}
int main()
