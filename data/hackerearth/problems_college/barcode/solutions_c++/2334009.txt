#include<stdio.h>
#include<string.h>
#define min(a,b) ((a)<(b)?(a):(b))
int n,m,x,y;
int d[1010];
char map[1010][1010];
int dp[1010][2];
void pretreat()
{
	int i,j;
	memset(d,0,(m+1)*sizeof(d[0]));
	dp[0][0]=dp[0][1]=0;
	for(i=1;i<=m;i++)
		dp[i][0]=dp[i][1]=0xffffff;
	for(i=0;i<n;i++)
		for(j=0;j<m;j++)
			d[j+1]+=map[i][j]=='#'?0:1;
	for(i=1;i<=m;i++)
			d[i]+=d[i-1];
}
void solve()
{
	int i,j;
	for(int i=0;i<m;i++){
		for(j=x;j<=y;j++){
			if(i+j>m) break;
			dp[i+j][1]=min(dp[i+j][1],dp[i][0]+d[i+j]-d[i]);
			dp[i+j][0]=min(dp[i+j][0],dp[i][1]+n*j-(d[i+j]-d[i]));
		}
	}
	printf("%d\n",min(dp[m][1],dp[m][0]));
}
int main()
{
	int i,j;
	while(scanf("%d%d%d%d",&n,&m,&x,&y)!=EOF){
		for(i=0;i<n;i++)
			for(j=0;j<m;j++)
				scanf("%1s",&map[i][j]);
		pretreat();
		solve();
	}
	return 0;
}
