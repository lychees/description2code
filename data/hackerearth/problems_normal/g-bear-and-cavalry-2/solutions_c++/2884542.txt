const int MAXN = 1000006;
int A[MAXN],B[MAXN];
bool solve() {
	int n;
	scanf("%d",&n);
	fru(i,n) scanf("%d",&A[i]);
	fru(i,n) scanf("%d",&B[i]);
	int ilo=0;
	fru(i,n) ilo=max(ilo,B[i]);
	fru(i,n) if(B[i]==ilo) {
		swap(B[i],B[n-1]);
		--n;
		break;
	}
	ilo*=A[0];
	fru(i,n) A[i]=A[i+1];
	sort(A,A+n);
	sort(B,B+n);
	reverse(B,B+n);
	fru(i,n) if(A[i]*B[i]>=ilo) return 0;
	return 1;
}

int main() {
	int te = 1;
	scanf("%d",&te);
	fru(ti,te) printf("%s\n",solve()?"YES":"NO");
	return 0;
}
