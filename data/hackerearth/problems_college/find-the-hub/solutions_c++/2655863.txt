#include <bits/stdc++.h>
using namespace std;

int main()
{
  int n;
  cin>>n;
  	int arr[n];
  	int i,j,max=0,fs=0;
  	for(i=0;i<n;i++)
  	cin>>arr[i];
  	
  	for(i=0;i<n;i++){
  		
  		int x=sqrt(arr[i]);
	    int temp=0;
	    for(j=2;j<=x;j++)
	    {
	        if(arr[i]%j==0)
	        {
	            temp=1;
	            break;
	        }
	    }
	    if(temp==0){
	    	if(fs==0){
	    		max=arr[i];
	    		fs=1;
	    	}
	    	else{
	    		if(arr[i]>max)
	    		max=arr[i];
	    	}
	    }
  	}
  	int count=0;
  	for(i=0;i<n;i++){
  		if(arr[i]==max)
  		count++;
  	}
  	cout<<max<<endl;
  	cout<<count<<endl;
    return 0;
}
