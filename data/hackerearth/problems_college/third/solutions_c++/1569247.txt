using namespace std;
int n;
long long int dp[100001];
long long int a[100001];
int main(){
	int t;
	scanf("%d", &t);
	while (t--){
		scanf("%d", &n);
		for (int i = 0; i < n; i++){
			scanf("%lld", &a[i]);
		}
		dp[0] = a[0];
		dp[1] = a[1];
		for (int i = 2; i < n; i++){
			dp[i] = dp[i - 2];
			if(i-3>=0){
				dp[i]=max(dp[i],dp[i-3]);
			}
			dp[i] += a[i];
		}
		long long int ans = 0;
		for (int i = 0; i < n; i++){
			ans = max(ans, dp[i]);
		}
		printf("%lld\n", ans);
	}
	return 0;
}
