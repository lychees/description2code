#include<bits/stdc++.h>
using namespace std;
int main(){
    long long int a, b;
    cin >> a >> b;
    string s;
    cin >> s;
    long long int A = a, res = 0;
    if(s[0] == '1') res += a;
    for(int i = 1; i < (int)s.length(); i++) {
        A = (A*A)%b;
        if(s[i] == '1') res = (res + A)%b;
    }
    cout << res%b << endl;
}
