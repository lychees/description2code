int main(int argc, char const *argv[])
{
	long int n, ind = 0, k, arr[BUF];
	scanf("%ld %ld", &n, &k);

	for (int i = 0; i < n; ++i)
		scanf("%ld", &arr[i]);
	sort(&arr[0], &arr[n]);

	for (int i = 1; i < n; ++i)
	{
		if (arr[i] - arr[i - 1] > k)
			break;
		ind = i;
	}

	printf("%ld\n", arr[ind]);
	return 0;
}
