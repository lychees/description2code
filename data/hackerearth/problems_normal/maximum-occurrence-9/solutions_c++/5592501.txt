#include <bits/stdc++.h>
using namespace std;
#define max 10000
char str[max];
int a[300];

int main()
{
    gets(str);
    int x=strlen(str);
    int i;
    int y;
    for(i=0;i<x;i++)
    {
    	y=str[i];
    a[y]++;	
    }
    	
    int j=0;	
    int m=0;
    for(i=0;i<300;i++)
    {
    	if(a[i]>m)
    	{
    	m=a[i];
    	j=i;
    	}
    }

    printf("%c %d\n",j,m);
    
    return 0;
}
