#include<bits/stdc++.h>
using namespace std;
typedef long long int uli;
const int mxn=212;
const int mxk=512;
char s[mxn],t[mxn];
int f[mxn][mxn][mxk];
int d[26][26];
int p[mxn];
int main(){
   for(int i=0;i<26;i++)
      for(int j=i;j<26;j++)
         d[i][j]=d[j][i]=min(j-i,i+26-j);
   int n,m,k;
   scanf("%d %d %d %s %s",&n,&m,&k,s,t);
   int pi=0;   
   for(int i=1;i<n;i++){
      while(pi>0 && s[pi]!=s[i])pi=p[pi-1];
      if(s[pi]==s[i])pi++;
      p[i]=pi;
   }
//   for(int i=0;i<n;i++)cout<<p[i]<<" ";cout<<endl;
   if(n>m){
      puts("0");
      return 0;
   }
   for(int i=m-1;i>=0;i--){
      if(i+n-1>=m)continue;
      for(int x=0;x<n;x++){
         int h=0;
         for(int j=0;j<n-x;j++)h+=d[t[i+x+j]-'a'][s[x+j]-'a'];
         for(int q=0;q<=k;q++){
            int ni=i+1;
            int nx=0;
            if(x!=0)ni=(i+x-p[x-1]),nx=p[x-1];
            f[i][x][q]=f[ni][nx][q];
            if(h<=q){
               ni=i+n-p[n-1];
               nx=p[n-1];
               f[i][x][q]=max(f[i][x][q], f[ni][nx][q-h]+1 );
            }
         }
      }
   }
   printf("%d\n",f[0][0][k]);
   return 0;
}
