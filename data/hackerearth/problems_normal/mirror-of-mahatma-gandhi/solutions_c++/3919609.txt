#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	string s;
    	cin>>s;
    	bool flag = true;
    	for(int i=0;i<s.length()/2;i++){
    		if(s[i]!=s[s.length()-1-i]){
    			flag = false;
    			break;
    		}
    	}
    	if(flag){
    		bool flag2 = true;
    		for(int i=0;s[i]!='\0';i++){
    			if(s[i]!='0'&&s[i]!='1'&&s[i]!='8'){
    				flag2 = false;
    				break;
    			}
    		}
    		if(!flag2) cout<<"NO\n";
    		else cout<<"YES\n";
    	}
    	else cout<<"NO\n";
    }
    return 0;
}
