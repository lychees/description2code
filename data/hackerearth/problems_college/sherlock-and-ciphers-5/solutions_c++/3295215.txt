#include <bits/stdc++.h>
using namespace std;

int main() {
	int i, j, l, s, n;
	string c;
	scanf("%d", &n);
	getline(cin, c);
	while (n--) {
		getline(cin, c);
		l = c.size();
		for (s=1; s<=l; s++) {
			if (s*s == l) break;
		}
		for (i=0; i<s; i++) {
			for (j=0; j<s; j++) printf("%c", c[i+(s*j)]);
		}
		printf("\n");
	}
}
