#include <bits/stdc++.h>
using namespace std;

vector<int> v;
int main(){
  int t,n;
  cin>>t;
  while(t--){
    cin>>n;
    v.clear();
    
    for (int i=1;i<=n;i++){
      int x;
      cin>>x;
      v.push_back(x);
    }    
    int res=0;
    for (int i=0;i<v.size();i++){
      int x;
      cin>>x;
      res^=x-1;
      res^=v[i]-x;
    }
    if(res)
      cout<<"First"<<endl;
    else
      cout<<"Second"<<endl;
  }
  return 0;
  
}
