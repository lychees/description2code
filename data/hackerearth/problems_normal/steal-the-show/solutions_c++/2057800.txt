#define MOD 4294967296
#define pb push_back
#define mp make_pair

typedef long long int ll;

using namespace std;

int main()
{
    ios::sync_with_stdio(0);
    ll n,x,ceo,coo,cto,ansr,i;
    cin>>ceo>>coo>>cto;
    cin>>n;
    vector <ll> ans1,ans,v;
    for(i=0;i<n;i++) {
        cin>>x;
        v.pb(x);
    }
    sort(v.begin(),v.end());
    if(v.size()%2 == 0) {
        ans.pb(min(coo,cto));
        for(i=0;i<v.size()/2;i++) {
            ans.pb(v[i]);
        }
        ans.pb(ceo);
        for(;i<v.size();i++) {
            ans.pb(v[i]);
        }
        ans.pb(max(cto,coo));
        ll tmpans = 0;
        for(i=1;i<ans.size();i++) {
            tmpans+=abs(ans[i] - ans[i-1]);
        }
        ansr = tmpans;
    } else {
        ans1.pb(min(coo,cto));
        for(i=0;i<v.size()/2;i++) {
            ans1.pb(v[i]);
        }
        ans1.pb(ceo);
        for(;i<v.size();i++) {
            ans1.pb(v[i]);
        }
        ans1.pb(max(cto,coo));
        ll tmpans1 = 0;
        for(i=1;i<ans1.size();i++) {
            tmpans1+=abs(ans1[i] - ans1[i-1]);
        }
        ans.pb(min(coo,cto));
        for(i=0;i<(v.size()+1)/2;i++) {
            ans.pb(v[i]);
        }
        ans.pb(ceo);
        for(;i<v.size();i++) {
            ans.pb(v[i]);
        }
        ans.pb(max(cto,coo));
        ll tmpans = 0;
        for(i=1;i<ans.size();i++) {
            tmpans+=abs(ans[i] - ans[i-1]);
        }
        ansr = min(tmpans,tmpans1);
    }
    cout<<ansr<<endl;
    return 0;
}
