N = int(raw_input())

for i in range(N):
	s = raw_input()
	
	c = [0,0,0]
	m = [' ', ' ', ' ', ' ']
	k = 0
	
	for j in range(len(s)):
		
		if not (j == 0 or s[j] == s[j-1]):
			k += 1
			m[k] = s[j]
		else:
			m[k] = s[j]

		if k > 2:
			c[0] = -1
			break
		
		c[k] += 1
		
	if c[0] == c[1] and c[0] == c[2] and m[0] != m[1] and m[1] != m[2] and m[0] != m[2]:
		print 'OK'
	else:
		print 'Not OK'
