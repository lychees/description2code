#include <bits/stdc++.h>

using namespace std;

long long arr[105];	

int main()
{
    int n, k, cnt = 1;
    long long ans = 0LL;
    scanf("%d%d", &n, &k);
    for (int i = 0; i < n; i++) scanf("%lld", &arr[i]);
    
	sort(arr, arr+n, greater<long long>());
	for (int i = 0; i < n; i++) {
		if (i > (k*cnt-1)) cnt++;
		ans += arr[i]*cnt;
	}

    printf("%lld\n", ans);
    return 0;
}
