#include <cstdio>
#define sd(x) scanf("%d",&x)
#define ps(x) printf("%s\n",x )
#define ll long long int
using namespace std;
ll a[100000];
int size;
int binary_search(int n){
	int low=1 , high=size , mid;
	while(1){
		mid = (low+high)/2;
		if(a[mid]==n){
			return mid;
		}else if (a[mid]<=n && a[mid+1]>=n){
			return mid+1;
		}else if(a[mid]<n){
			low = mid+1;
		}else{
			high=mid-1;
		}
	}
}
int main(void){
	size = 1;
	int i,t,n;
	a[1]=1;
	for(i=2;;i++){
		a[++size]=a[i-1]+i;
		if(a[i]>1000000000){
			break;
		}
	}
	sd(t);
	while(t--){
		sd(n);
		int val = binary_search(n);
		if(val%3==0)
			ps("SL");
		else if(val%3==1)
			ps("LB");
		else
			ps("BS");
	}
	return 0;
}
