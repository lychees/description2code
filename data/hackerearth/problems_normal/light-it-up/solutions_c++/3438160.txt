            A.push_back(MP(MP(l, r), h));
        }
        sort(ALL(A));

        FOR(i,0,A.size() - 1)
        {
            int x = A[i + 1].first.first - A[i].first.second;
            int y1 = A[i].second;
            bool ok1 = 1;
            FOR(j,0,i)
            {
                if (A[j].second * x >= y1 * (A[i + 1].first.first - A[j].first.second))
                {
                    ok1 = 0;
                    break;
                }
            }
            int y2 = A[i + 1].second;
            bool ok2 = 1;
            FOR(j,i + 2,A.size())
            {
                if (A[j].second * x >= y2 * (A[j].first.first - A[i].first.second))
                {
                    ok2 = 0;
                    break;
                }
            }
            //cout << i << ' ' << ok1 << ' ' << ok2 << endl;
            if (ok1 || ok2) ++ res;
        }

        cout << res << endl;
    }

    return 0;
