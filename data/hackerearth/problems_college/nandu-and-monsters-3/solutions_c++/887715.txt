#include<iostream>
#include<queue>
using namespace std;
int visited[1000000];
int ct=1,n;
string m;
void bfs(int s) {
    queue<int> q;
    q.push(s);
    visited[s]=ct;
    while(!q.empty()) {
        int tp=q.front();
        q.pop();
        if(2*tp+1<n&&m[2*tp+1]=='0'&&!visited[2*tp+1]) {
            visited[2*tp+1]=ct; 
            q.push(2*tp+1);
        }
        if(2*tp+2<n&&m[2*tp+2]=='0'&&!visited[2*tp+2]) {
            visited[2*tp+2]=ct;
            q.push(2*tp+2);
        }
        if(m[(tp-1)/2]=='0'&&!visited[(tp-1)/2]) {
            visited[(tp-1)/2]=ct;
            q.push((tp-1)/2);
        }   
    }     
}
main() {
    int q;
    cin>>n>>q;
    cin>>m;
    for(int i=0;i<n;++i)
        visited[i]=0;
    for(int i=0;i<n;++i) {
        if(!visited[i]&&m[i]=='0') {
            bfs(i);
//            for(int i=0;i<n;++i)
  //              cout<<visited[i]<<" ";
    //        cout<<endl;
            ++ct;
        }
    }
//    for(int i=0;i<n;++i)
  //      cout<<visited[i]<<" ";
    //cout<<endl;
    for(int i=0;i<q;++i) {
        int s,d;
        cin>>s>>d;
        --s;
        --d;
        if(m[d]=='1'||m[s]=='1') {
            cout<<"No"<<endl;
            continue;
        }
        if(visited[s]==visited[d])
            cout<<"Yes"<<endl;
        else
            cout<<"No"<<endl;
    }
    return 0;
}
