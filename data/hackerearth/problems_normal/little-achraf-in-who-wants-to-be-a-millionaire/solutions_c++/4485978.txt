#include<iostream>
#define lim 10000000000000000
#define maxn 200000
using namespace std;
int l,crit;
unsigned long long stor[maxn];
int seq[200];


int f(int a,unsigned long long p)
{
    if(a<=l)
        return a;
    unsigned long long term=0;
    for(int i=0;i<l;i++)
    {
        if(term+stor[a-l+i]>=p)
        {
            return f(a-l+i,p-term);
        }
        term+=stor[a-l+i];
    }
    cout<<5/0;
}

int ff(unsigned long long a,unsigned long long p)
{
    if(a<crit+l)
    {
        return f(a,p);
    }
    a-=crit;
    return f(crit+l+a%l,p);
}

int cc(char ch)
{
    if(ch>='a'&&ch<='z')
    {
        return ch-'a'+1;
    }
    if(ch>='A'&&ch<='Z')
    {
        return ch-'A'+27;
    }
    if(ch>='0'&&ch<='9')
    {
        return ch-'0'+53;
    }
    return 5/0;

}

unsigned long long power(int n,unsigned long long k ,int m=1000000007)  //return n^k % m
{
 if(k==0) return 1;
 if(k==1) return n;
 unsigned long long f=power(n,k/2,m);
 //cout<<"#"<<f<<"#";
 unsigned long long p=(f*f)%m;
 //t=f*f;
 //cout<<t;
 //p=(t)%m;
  if(k%2!=0)
  p=(p*n)%m;
  return p;
}

             //   cout<<i;
