#include <iostream>
#include <stdio.h>
#include <string.h>
#include <algorithm>
#include <queue>
#include <math.h>
#include <map>
#include <set>
#include <unordered_map>

using namespace std;

int n , p , c ;
int y[100001][3] ;
int x[100001] ;
int a ;

int main()
{
    cin >> c >> p >> n ;
    priority_queue < pair < int , int > > q ;
    for ( int i = 0 ; i < n ; i++ )
    {
        cin >> a ;
        y[i][2] = a;
        y[i][0] = 1;
    }
    for ( int i = 0 ; i < c ; i++ )
        q.push({-(y[i][1]+y[i][2])*y[i][0],-i}) ;
    for ( int i = 0 ; i < p ; i++ )
        cin >> x[i] ;
    for ( int i = 0 ; i < p ; i++ )
    {
        pair < int , int > pp = q.top() ;
        q.pop() ;
        cout << -pp.second + 1 << " " ;
        y[-pp.second][1] = y[-pp.second][2] ;
        y[-pp.second][2] = x[i] ;
        y[-pp.second][0]++ ;
        q.push({-(y[-pp.second][1]+y[-pp.second][2])*y[-pp.second][0],pp.second}) ;
    }
    return 0;
}
