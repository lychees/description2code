#include<bits/stdc++.h>
using namespace std;

typedef long long ll;

map<int,int> factors(int n) {
    map<int,int>r;
    int cnt=0;
    while(n%2==0) {
        n/=2;
        cnt++;
    }
    r[2]=cnt;
    for(int i=3;i*i<=n;i+=2) {
        cnt=0;
        while(n%i==0) {
            n/=i;
            cnt++;
        }
        r[i]=cnt;
    }
    if(n>2) {
        r[n]=1;
    }
    return r;
}
int main() {
    //freopen("a.cpp","r",stdin);
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        int a,b,c,d;
        scanf("%d%d%d%d",&a,&b,&c,&d);
        map<int,int> x = factors(a);
        map<int,int> y = factors(c);
        map<int,int> :: iterator it;
        bool flag=true;
        for(it=y.begin();it!=y.end();it++) {
            if(ll(d)*it->second>ll(b)*x[it->first]) {
                flag=false;
            }
        }
        if(flag==false) {
            puts("Not divisible");
        } else {
            puts("Divisible");
        }
    }
    return 0;
}
