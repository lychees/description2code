#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
    int t;
    scanf("%d", &t);
    while(t--)
    {
    	long long n;
    	scanf("%lld", &n);
    	int count = 0;
    	
    	while(n>0)
    	{
    		if(n%2==1)
    			count++;
    		n/=2;
    	}
    	printf("%d\n",count);
    }
    return 0;
}
