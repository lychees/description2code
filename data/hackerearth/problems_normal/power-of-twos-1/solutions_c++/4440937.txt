#include<bits/stdc++.h>
using namespace std ;
map < int , int > mp ;
int sp[1000002] , dp[1000002] ;
void sieve ( ) {
    int i , j ;
    for ( i = 2 ; i < 1000001 ; i++ ) {
        
	    for ( j = i+i ; j < 1000001 ; j+=i )
            sp[j]++ ;
    }
}
int main () {
    sp[1] = 0 ;
    sp[2] = 1 ;
    sp[3] = 2 ;
    sieve() ;
    for ( int i = 4 ; i <= 1000000 ; i++ ) {
        sp[i] += sp[i-1]+1 ;
    }
    int t ;
    scanf ( "%d" , &t ) ;
    while ( t-- ) {
        int n ;
        scanf ( "%d" , &n ) ;
        printf ( "%d\n" , sp[n] ) ;
    }
    return 0 ;
}
