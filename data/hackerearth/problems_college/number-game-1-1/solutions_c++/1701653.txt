#include <iostream>
#define scan(t) scanf("%lld",&t)
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   long long int t,n,temp,ans,i,j,k;
   scan(t);
   while(t--)
   {
   	ans=0;
   	scan(n);
   	while(n!=1)
   	{
   		if(n%3==0) { n/=3; ans++; } 
   		else if(n%2==0) { n/=2; ans++; }
   		else { n--; ans++; }
   	}
   	cout<<ans<<endl;
   }
    return 0;
}
