#include <bits/stdc++.h>
using namespace std;
int n,c,p;
typedef pair <int,int> P;
pair <int,int> pr;
priority_queue<P,vector<P>,greater<P>> q;
int last[100005];
int last1[100005];
int count1[100005];
int l[100005];
int main()
{
	int i,j,v,x;
    cin>>c>>p>>n;
    for(i=0;i<n;i++)
    {
    	cin>>x;
    	count1[i]=1;
    	last[i]=x;
    }
    for(i=0;i<p;i++)
    	cin>>l[i];
    for(i=0;i<c;i++)
    {
    	q.push({last[i],i});
    }
    for(i=0;i<p;i++)
    {
    	cin>>x;
    	pr=q.top();
    	q.pop();
    	int v;
    	v=pr.second;
    	cout<<v+1<<" ";
    	count1[v]++;
    	last1[v]=last[v];
    	last[v]=l[i];
    	pr.first=(last[v]+last1[v])*count1[v];
    	q.push(pr);
    } 
    return 0;
}
