#include<iostream>
#include<vector>
#include<string>
#include<algorithm>
#include<cstdio>
#include<numeric>
#include<cstring>
#include<ctime>
#include<cstdlib>
#include<set>
#include<map>
#include<unordered_map>
#include<unordered_set>
#include<list>
#include<cmath>
#include<bitset>
#include<cassert>
using namespace std;
pair<int, int>a[100007];
bool used[100007];
int main()
{
	//freopen("input.txt", "r", stdin);
	//freopen("output.txt", "w", stdout);
	int n, q;
	scanf("%d%d", &n, &q);
	for (int i = 1; i <= n; i++)
	{
		scanf("%d%d", &a[i].first, &a[i].second);
		if (a[i].first > a[i].second) swap(a[i].first, a[i].second);
	}
	long long total = 0;
	bool ok = true;
	set<pair<long long, int> >go;
	for (int i = 1; i <= n; i++)
	{
		set<pair<long long, int> >::iterator it = go.lower_bound(make_pair(a[i].first - a[i].second, -1));
		if (it == go.end())
		{
			go.insert(make_pair(a[i].first - a[i].second, i));
			continue;
		}
		pair<long long, int> cur = *it;
		if (cur.first == a[i].first - a[i].second)
		{
			used[cur.second] = true;
			used[i] = true;
			total += a[i].first;
			total += a[cur.second].second;
			go.erase(cur);
		}
		else
		{
			go.insert(make_pair(a[i].first - a[i].second, i));
		}
	}
	for (int i = 1; i <= n; i++)
	{
		if (!used[i]) ok = false;
	}
	for (int i = 1; i <= q; i++)
	{
		long long x;
		scanf("%lld", &x);
		if (ok && total == x)
		{
			printf("YES\n");
		}
		else
		{
			printf("NO\n");
		}
	}
}
