ll dp[MXN][11];

int main(){
    scanf("%d %d",&n,&q);    
    rep(i,n)  scanf("%d",&A[i]);

    rep(j,11) dp[0][j]=0;

    forup(i,1,n) forup(j,1,10)
        dp[i][j] = A[i-1] + ((i-j) < 0 ? 0 : dp[i-j][j]);

    while(q--){
        scanf("%d %d %d",&l,&r,&k);
        printf("%lld\n",dp[l + k * ((r-l)/k)][k]-(l-k < 0 ? 0 : dp[l-k][k]));
    }
}
