n = int(raw_input())
freq = map(int, raw_input().split())
k = int(raw_input())
keys = map(int, raw_input().split())

freq = sorted(freq, reverse=True)
pos = {}

for key in keys:
    for i in range(key):
        if i+1 not in pos:
            pos[i+1] = 0
        pos[i+1] += 1

cur = 0
s = 0

for key in sorted(pos.keys()):
    val = pos[key]
    s += key*sum(freq[cur:cur+val])
    cur += val

if sum(keys) >= n:
    print(s)
else:
    print(-1)
