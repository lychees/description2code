#include <iostream>
using namespace std;

int main()
{
	int tt;
	cin>>tt;
	for(int t=0 ; t<tt; t++){
		long long int n; 
		cin>>n;
		long long int ans = n*(n+1);
		ans /= 2;
		long long int prev = ans;
		ans = 0;
		for(long long int i=0; i<n; i++){
			prev = prev-(n-i);
			ans += prev;
		}
		ans *= (long long)2;
		cout<<ans<<"\n";
	}
    return 0;
}
