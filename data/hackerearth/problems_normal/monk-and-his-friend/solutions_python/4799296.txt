for _ in range(input()):
    m,n=map(int,raw_input().split())
    a=bin(m)
    a=a[2:]
    b=bin(n)
    b=b[2:]
    count=0
    if len(a)>len(b):
        b=(len(a)-len(b))*'0'+b
    else:
        a=(len(b)-len(a))*'0'+a
    #print a,b
    for i in range(len(a)):
        if a[i]!=b[i]:
            count+=1
    print count
