#include <iostream>
#include <string>
using namespace std;

int main()
{
	string inp;
	getline(cin, inp);
	int len = inp.length();
	long long int T, a, b;
	cin >> T;

	for(int i = 0; i < T; i++)
	{
		cin >> a >> b;
		a--; b--;
		a = (a > len) ? a % len : a;
		b = (b > len) ? b % len : b;
		if(inp.at(a) == inp.at(b))
			cout << "Yes" << endl;
		else
			cout << "No" << endl;
	}
    return 0;
}
