#include <iostream>
#include<cstring>
using namespace std;
void removeconsecutive(char a[])
{
    int i=strlen(a),k=0;

    for(int j=0;j<i-1;j++)
    {
        if(a[j]==a[j+1])
        {
            for( k=j;k<i-1;k++)
            {
                a[k]=a[k+1];
            }
            a[k]='\0';
            j--;
            i--;

        }

    }
cout<<a<<endl;
}
int main()
{
   int T;
cin>>T;
 while(T>=0)

{
     char a[30];
    cin.getline(a,30);
     removeconsecutive(a);
T--;
}

    return 0;
}
