#include <iostream>
#include<string.h>
using namespace std;

int modulo(int ch)
{
	if(ch<0)
	return -ch;
	else
	return ch;
}

int main() {
	int t;
	cin>>t;
	while(t--)
	{
		char str[10000],temp[10000];
		cin>>str;
		
		int l=strlen(str);
		
		int q=0;
		for(int i=l-1;i>=0;i--)
		temp[q++]=str[i];
		temp[q]='\0';
		int ct=0;
		
		
		for(int i=1;i<l;i++)
		{
			if(modulo(str[i]-str[i-1])==modulo(temp[i]-temp[i-1]))
			ct++;
		}
		
		if(ct+1==l)
		cout<<"Funny\n";
		else
		cout<<"Not Funny\n";
	}
	return 0;
}
