cache = {}
def is_fibonacci(n):
	if n in cache:
		return cache[n]
	
	a = 0
	b = 1
	
	while a < n:
		a, b = b, a+b
		cache[a] = True
	
	if a == n:
		return True
	else:
		cache[n] = False
		return False

for _ in xrange(int(raw_input())):
	x, y = map(int, raw_input().strip().split())
	
	even = True
	if x == 1:
		x = 2
	
	for i in xrange(x, y+1):
		if is_fibonacci(i):
			even = not even
	
	if even:
		print "VALID"
	else:
		print "INVALID"
