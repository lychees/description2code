#include<bits/stdc++.h>
using namespace std;
int main() {
	int t,i,a,b,c;
	char str1[10000],str2[10000];
	cin>>t;
	while(t--)
	{
		c=0;
		cin>>str1;
		cin>>str2;
		a = strlen(str1);
		b = strlen(str2);
		if(a>=b)
		{
			for(int i=0;i<b;i++)
			{
				if(str1[i]==str2[i])
				c++;
			}
		}
		else if(a<b)
		{
			for(int i=0;i<a;i++)
			{
				if(str1[i]==str2[i])
				c++;
			}

		}
		cout<<c<<endl;
		for(int i=0;i<b;i++)
		str2[i] = '\0';
	}

}
