#include <iostream>
using namespace std;
#define IOS 		ios_base::sync_with_stdio(false); cin.tie(NULL);
#define IN          int t; scanf("%d",&t); while(t--)
int main()
{
   int i,n,project,ans;
   
   char s[25];
   IN
   {
   		scanf("%d",&n);
   		getc(stdin);
   		project=ans=0;
   		while(n--)
   		{
   			gets(s);
   			if(!project and s[0]=='N')
   			ans++;
   			else
   			{
   			if(s[0]=='1')
   			project++;
   			else if(s[0]=='2')
   			project+=2;
   			else if(s[0]=='3')
   			project+=3;
   			else if(s[0]=='4')
   			project+=4;
   			else if(s[0]=='5')
   			project+=5;
   			else if(s[0]=='N')
   			project--;
   			}
   		}
   		cout << ans << endl;
   }
    return 0;
}
