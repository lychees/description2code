
#include <bits/stdc++.h>
using namespace std;
#define ll long long
ll CombinationsIter(ll iN, ll iR) {
    if (iR < 0 || iR > iN) {
        return 0;
    }
    ll iComb = 1;
    ll i = 0;
    while (i < iR) {
        ++i;
        iComb *= iN - i + 1;
        iComb /= i;
    }
    return iComb;
}

int main()
{
    ll n,r;
    cin>>n>>r;
    if(n<r)
        cout<<-1<<endl;
    else
        cout<<(CombinationsIter(n-1,r-1))%10000007<<endl;

    return 0;
}
