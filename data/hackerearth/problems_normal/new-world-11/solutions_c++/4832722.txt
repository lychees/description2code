#include<bits/stdc++.h>
using namespace std;
#define fast ios_base::sync_with_stdio(false);

int arr[100005];

bool check(int distance,int n,int k)
{
	int curr=0,prev=0,req=0;
	for(int i=0;i<n;i++)
	{
		while((curr!=n)&&(arr[curr]-arr[prev]<=distance))
		 {
			   curr++;}
			req++;
			if(curr==n)
			break;
			prev=curr-1;
		
	}
		if(curr!=n)
		return false;
		if(req<=k) 
		return true;
	    return false;
	
}


int main()
{
	fast
	int t;
	cin>>t;
	while(t--)
	{
		int n,k;
		cin>>n>>k;
	    for(int i=0;i<n;i++) cin>>arr[i];
	    
	    int low=1,high=(1e9),mid,ans=0;
	    while(low<=high)
	    {
	    	mid = (low+high)/2;
	    	if(check(mid,n,k))
	    	{
	    		ans= mid;
	    		high=mid-1;
			}
			
			else low = mid+1;
		}
		cout<<ans<<endl;
	    
	}
}
