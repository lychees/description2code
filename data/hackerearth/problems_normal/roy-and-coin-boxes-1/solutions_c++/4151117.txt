#include<cstdio>

int temp[1000001] = {0};
int dp[1000001] = {0};

int main(){
    int n, m;
    scanf("%d %d", &n, &m);
    for(int i=0;i<m;i++){
        int l, r;
        scanf("%d %d", &l, &r);
        temp[l]++;
        temp[r+1]--;
    }
    int sum = 0;
    for(int i=0;i<=n;i++){
        sum += temp[i];
        dp[sum]++;
    }
    for(int i=n-1;i>=1;i--){
        dp[i] = dp[i] + dp[i+1];
    }
    int q;
    scanf("%d", &q);
    for(int i=0;i<q;i++){
        int x;
        scanf("%d", &x);
        printf("%d\n", dp[x]);
    }
    return 0;
}
