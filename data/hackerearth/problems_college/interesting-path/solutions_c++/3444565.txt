#include <iostream>
using namespace std;
#include <limits.h>
 

int minJumps(int arr[], int l, int h)
{
   if (h == l)
     return 0;
   if (arr[l] == 0)
     return INT_MAX;
 
   int min = INT_MAX;
   for (int i = l+1; i <= h && i <= l + arr[l]; i++)
   {
       int jumps = minJumps(arr, i, h);
       if(jumps != INT_MAX && jumps + 1 < min)
           min = jumps + 1;
   }
 
   return min;
}

int main()
{
	int t,n,p,i,j,arr[1001];
	cin>>t;
	for(i=0;i<t;i++)
	{
		cin>>n;
		for(j=0;j<n;j++)
		cin>>arr[j];
    cout<<minJumps(arr, 0, n-1)<<endl;
	}
  return 0;
}
