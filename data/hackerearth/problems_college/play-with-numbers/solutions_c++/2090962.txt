#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int t;
    cin >> t;
    
    for(int i=0;i<t;i++)
    {
    	long m,n,z,ans=0;
    	cin >> z >> m >> n;
    	
    	if(z<=m)
    	{
    		ans++;
    	}
    	if(z<=n)
    	{
    		ans++;
    	}
    	
    	long r=(long)sqrt(z);
    	
    	if(r*r==z)
    	{
    		if(r<=m && r<=n)
    		{
    			ans++;
    		}
    	}
    	
    	for(long j=2;j<r;j++)
    	{
    		if(z%j==0)
    		{
    			long x=j;
    			long y=z/j;
    			
    			if(x<=n && y<=m)
    			{
    				ans++;
    			}
    			if(x<=m && y<=n)
    			{
    				ans++;
    			}
    		}
    	}
    	
    	cout << ans << endl;
    }
}
