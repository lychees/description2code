#include<bits/stdc++.h>
using namespace std;
bool pre[100001][101];
int arr[101],n;
void pre_compute()
{
	for(int i=0;i<=n;i++)
		pre[0][i]=true;
	
	for(int i=1;i<=100000;i++)
		pre[i][0]=false;
	
	for(int i=1;i<=100000;i++)
	{
		for(int j=1;j<=n;j++)
		{
			pre[i][j]=pre[i][j-1];
			if(i>=arr[j])
				pre[i][j]=pre[i][j]||pre[i-arr[j]][j-1];
		}
	}
}
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&arr[i]);
	}
	pre_compute();
	int tc;
	scanf("%d",&tc);
	while(tc-->0)
	{
		int z=0;
		scanf("%d",&z);
		if(z>100000)
			printf("No\n");
		else if(pre[z][n])
			printf("Yes\n");
		else
			printf("No\n");
	}
	return 0;
}
