    for(int i=1;i<=n;i++)

   	}

   	while(m--)
   	{
   		long long int l,j,k;
   		scanf("%lld%lld%lld",&l,&j,&k);
   		b[l]+=k;
   		b[j+1]-=k;
   	}

   	for(int i=1;i<=n;i++)
