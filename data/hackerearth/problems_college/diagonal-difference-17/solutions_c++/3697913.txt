#include <iostream>
using namespace std;

int main()
{
    int n,i,j,k,l;
    k=0;l=0;
    cin>>n;
    int a[n][n];
    for(i=0;i<n;i++)
    for(j=0;j<n;j++)
    {
    	cin>>a[i][j];
    	if(i==j)
    	k+=a[i][j];
    	if((i+j)==n-1)
    	l+=a[i][j];
    }
    if(k>=l)
    k-=l;
    else
    k=l-k;
    
    cout<<k;
    return 0;
}
