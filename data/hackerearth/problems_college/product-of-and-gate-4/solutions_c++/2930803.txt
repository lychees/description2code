#include <iostream>
using namespace std;

int main()
{
    long long int t,a,b,ans,i;
    scanf("%d",&t);
    while(t--)
    {
     scanf("%lld%lld",&a,&b);
     
     if(a==0&&b==0){ans=0;}
     else if(a==0&&b==1){ans=1;}
     else
     {
     	if(a==b)ans=a;
     	else 
     	{
     	if(a==0){a=1;}	
     	ans=a;
        for(i=a+1;i<=b;i++)
        ans=ans&i;
     
        }
     }
     printf("%lld\n",ans);
     
     	
     
     
    }
    return 0;
}
