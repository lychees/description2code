#include <iostream>
#include<stack>
using namespace std;

int main()
{
    stack <int> el;
    int n;
    cin>>n;
    int x;
    for(int i=0; i<n; ++i)
    {
    	cin>>x;
    	if(x != 0)
    		el.push(x);
    	else
    		if(!el.empty())
    			el.pop();
    			
    }
    
    int sum=0;
    while(!el.empty())
    {
    	sum = sum + el.top();
    	el.pop();
    }
    
    cout<<sum<<endl;
    return 0;
}
