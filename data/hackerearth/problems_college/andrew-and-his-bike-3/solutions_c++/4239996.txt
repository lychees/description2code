#include <bits/stdc++.h>
using namespace std;

vector<int> graph[100001];
int n;
int sz[100001];
bool vis[100001];

int dfs(int node){
	sz[node]=1;
	vis[node]=1;
	for(int i=0;i<(int)graph[node].size();i++){
		if(!vis[graph[node][i]])
		{
			sz[node]+=dfs(graph[node][i]);
		}
	}
	return sz[node];
}

int main()
{
	cin>>n;
	int x,y;
	for(int i=0;i<=n;i++)
	graph[i].clear();
	memset(vis,0,sizeof(vis));
	memset(sz,0,sizeof(sz));
	for(int i=1;i<n;i++){
		cin>>x>>y;
		graph[x].push_back(y);
		graph[y].push_back(x);
	}
	cout<<dfs(1)-1;
	for(int i=2;i<=n;i++)
	cout<<" "<<sz[i];
    return 0;
}
