#include <iostream>
#include <bits/stdc++.h>
#include <string>
using namespace std;

int main()
{
    int T,size, i;
    string s1;
    char Alph[26];
    scanf("%d", &T);
    while (T--) {
    	cin>>s1;
    	memset(Alph, 0, sizeof(Alph));
    	for(i=0, size=s1.length(); i<size; i++) {
    		Alph[s1[i]-'a']++;
    	}
    	for (int j=0; j<26; j++) {
    		int min = size+1;
    		char letter = 'a';
    		for (i=0; i<26; i++) {
    			if (Alph[i] && Alph[i] < min) {
    				min = Alph[i];
    				letter = 'a'+i;
    			}
    		}
    		if (min == size+1)
    			break;
    		for (i=0; i<min; i++) {
    			printf("%c",letter);
    		}
			Alph[letter-'a'] = 0;
    	}
    	printf("\n");
    }
    return 0;
}
