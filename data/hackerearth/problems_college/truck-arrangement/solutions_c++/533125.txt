#include<cstdio>
#include<iostream>
#include<vector>
#include<algorithm>
#include<cstring>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	int i,j,k;
	
	int c[10001];
	while(t--)
	{
		int n;
		scanf("%d",&n);
		vector<pair<int,int> > v;
		for(i=0;i<n;i++)
		{
		
		scanf("%d",&j);
		
		v.push_back(make_pair(j,i));
		}
		
		
		sort(v.begin(),v.end());
		
			
		for(i=0;i<n;i++)
		{
			c[i] = abs(v[i].second - i);
		}
		
		int ans= 0;
		for(i=0;i<n;i++)
		if(c[i]>ans) ans = c[i];
		
		
		printf("%d\n",ans);
		
		
		
		
	}
	
	return 0;
}
