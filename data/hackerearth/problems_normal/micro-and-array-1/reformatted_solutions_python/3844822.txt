n, q = raw_input().split()
nlist = [int(x) for x in raw_input().split()]
for _ in xrange(int(q)):
    query = raw_input().split()
    if len(query) == 2:
        # assert query[0] == '1'
        try:
            print next(i for i, x in enumerate(nlist) if x >= int(query[1])) + 1
        except BaseException:
            print -1
    elif len(query) == 3:
        # assert query[0] == '0'
        nlist[int(query[1]) - 1] = int(query[2])
