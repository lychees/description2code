#include <bits/stdc++.h>

using namespace std;

int main()
{
    //ios::sync_with_stdio(false);cin.tie(0);

    long long n, m , t;

    cin >>       t         ;

    while(t--){

        cin >> n >> m;

        long long st1 = n, st2 = m, ans1 = 0, ans2 = 0;

        m = min(n , m);
        n = min(n , m);

        n /= 2;

        ans1 =  n * (n + 1);

        if(m % 2 != 0) m++;
        long long temp = m;

        m /= 2;

        ans2 = (m * (m - 1) ) + temp / 2;

        if(st2 >= st1) cout << ans1 << " " << ans2 << "\n";
        else {

            n = st1, m = st2;
            ans1 += (n - m) / 2 * m;
            ans2 += (n - m) / 2 * m;

            if( (n - m) % 2 != 0){

                if(m % 2 != 0) ans1 += m;
                else ans2 += m;
            }

            cout << ans1 << " " << ans2 << "\n";
        }
    }

    return 0;
}
