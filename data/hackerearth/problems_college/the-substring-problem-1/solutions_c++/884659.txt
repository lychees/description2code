#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<string>
using namespace std;
string str[105];
int hashTable[26];
bool visited[26];
int main() {
    int t;
    cin>>t;
    while(t--) {
        memset(hashTable,0,sizeof(hashTable));
        int n;
        cin>>n;
        getchar();
        for(int i=0;i<n;i++) cin>>str[i];
        for(int i=0;i<n;i++) {
            memset(visited,0,sizeof(visited));
            for(int j=0;j<str[i].length();j++) {
                if(visited[str[i][j]-'a']==false)
                hashTable[str[i][j]-'a']++;
                visited[str[i][j]-'a']=true;
            }
        }
        int idx=0;
        int mx=-1;
        for(int i=0;i<26;i++) {
            if(hashTable[i]>mx) {
                mx=hashTable[i];
                idx=i;
            }
        }
        cout<<char('a'+idx)<<endl;
    }
    return 0;
}
