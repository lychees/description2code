#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	 int n;
    	cin>>n;
    	int array[n];
    	int check[10];
    	for(int i=0;i<10;i++)
    	check[i]=0;
    	for(int i=0;i<n;i++)
    	{  
    		int x;
    		cin>>x;
    		array[i]=x%10;
    	}
    	int count =0;
    	for(int i=0;i<n;i++)
    	check[array[i]]++;
    	for(int i=0;i<10;i++)
    	{
    		if(check[i]>1)
    		count=count+check[i]-1;
    	}
    	cout<<count<<"\n";
    }
}
