pii X[4000];
ll DP[2001][2001];
int main()
{

  scanf("%d",&t);

  while(t--)
  {
      int n,k;
    scanf("%d%d",&n,&k);
    for(int i=1;i<=n;++i)
    {
      scanf("%d",&X[i].first);
    }
    for(int i=1;i<=n;++i)
    {
      scanf("%d",&X[i].second);
    }
    sort(X+1,X+n+1);
    X[n+1].first=1e9;
    for(int i=0;i<=n;++i)
      DP[0][i]=0;

    for(int i=1;i<=n;++i)
      {
            for(int next=i+1;next<=n+1;++next)
            {

                  ll o1= (ll)(X[i].second+k)*X[i].first + DP[i-1][i];

                  ll o2= (ll)(X[i].second)*X[next].first + DP[i-1][next];

                  DP[i][next]=min(o1,o2);
            }
      }

    printf("%lld\n",DP[n][n+1]);

  }
