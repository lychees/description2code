from sys import stdin
t = int(stdin.readline())
for _ in xrange(t):
	a = map(int,stdin.readline().split())
	a.sort()
	s = set(a)
	ans = 'NO'
	if a[0] + a[1] > a[2]:
		if len(s)==1:
			ans = 'YES. It is a equilateral triangle.'
		elif len(s)==2:
			ans = 'YES. It is a isosceles triangle.'
		else:
			ans = 'YES. It is a scalene triangle.'
	print ans
