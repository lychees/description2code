#define FOD(x,a,b) for (int x=a;x>=b;x--)
#define REP(x,a,b) for (int x=a;x<b;x++)
#define RED(x,a,b) for (int x=a;x>b;x--)

const int MAX = 1e6 + 10;
const int MAXN = 1e6 + 10;
const int MOD = 1e9 + 7;
const int inf = 1e9;
const double pi = acos(-1.0);
const double eps = 1e-6;

int Z[MAX];
int f[MAX];

void Zfunction(string S){
	int L, R, N;
	N = S.length();
	L = R = 0;
	for (int i = 1; i < N; i++){
		if (i > R){
			L = R = i;
			while (R < N && S[R - L] == S[R])
				R++;
			Z[i] = R - L;
			R--;
		}
		else{
			int k = i - L;
			if (Z[k] < R - i + 1)
				Z[i] = Z[k];
			else{
				L = i;
				while (R < N && S[R - L] == S[R])
					R++;
				Z[i] = R - L;
				R--;
			}
		}
	}
}

int main(){
	// freopen("test.in", "rt", stdin);
	ios::sync_with_stdio(false);

	string s, t;
	int n, m;

	cin >> s;
	
	t = s;
	reverse(t.begin(), t.end());
	t = s + t;
	Zfunction(t);
	REP(i,s.length(),t.length())
		if (Z[i] + i == t.length()){
			n = Z[i];
			break;
		}
	
	t = s;
	reverse(t.begin(), t.end());
	t += s;
	Zfunction(t);
	REP(i,s.length(),t.length())
		if (Z[i] + i == t.length()){
			m = Z[i];
			break;
		}
	
	t = "";
	REP(i,s.length()-m,s.length()) t += s[i];
	REP(i,0,n) t += s[i];

	Zfunction(t);
	REP(i,1,t.length())
		f[Z[i]]++;
	f[t.length()]++;
	FOD(i,t.length(),0) f[i] += f[i+1];

	cout << t << "\n";
	FOR(i,1,t.length())
		cout << f[i] << " ";

	return 0;
}
