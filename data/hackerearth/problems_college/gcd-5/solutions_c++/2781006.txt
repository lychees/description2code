#include<iostream>
#define ll long long
using namespace std;
int main()
{
	ll t,n,a[1000005];
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
		for(int i=0;i<n;i++)
		scanf("%lld",&a[i]);
		int flag=1;
		for(int i=0;i<n-1;i++)
		{
			if(a[i+1]<a[i])
			{
				flag=0;
				break;
			}
			
		}
		if(flag==1)
		 printf("YES\n");
		 else
		 printf("NO\n");
		
	}
	return 0;
}
