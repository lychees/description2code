using namespace std;

#define sd(a) scanf("%lld",&a)
#define pd(a) prlong longf("%d\n",(a))
#define LL long long
#define F first
#define S second
#define MP make_pair
#define PB push_back
typedef long long ll;
int main()
{
	ll t,n,so,se,cnt;
	sd(t);
	while(t--)
	{
		sd(n);
		so=se=cnt=0;
		while(n!=1)
		{
			cnt++;
			if(n&1)
			{
				so+=n;
				n=n*3+1;
			}
			else
			{
				se+=n;
				n/=2;
			}
		}
		so+=1;
		so%=cnt;
		se%=cnt;
		if(se>so)
			printf("Even Rules\n");
		else if(so>se)
			printf("Odd Rules\n");
		else
			printf("Tie\n");
	}
}
