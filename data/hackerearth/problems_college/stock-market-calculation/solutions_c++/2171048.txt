#include<stdio.h>
typedef long long int ll;
int main()
{
    ll t,n,i,j;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld",&n);
        ll a[n];
        ll maxi=0,pos;
        for(i=0;i<n;i++)
        {
            scanf("%lld",&a[i]);
            if(a[i]>=maxi)
            {
                maxi=a[i];
                pos=i;
            }
        }
        ll c=0,sell=0,buy=0,k,p=-1;
        while(true)
        {
           for(i=p+1;i<pos;i++)
           {
               if(a[i]!=-1)
               {
                   c++;
                   buy+=a[i];
               }
           }
           sell+=c*a[pos]-buy;
           //printf("%lld\n",sell);
           c=0;buy=0;
           a[pos]=-1;
           maxi=-1;
           p=pos;
           if(p==n-1)
            break;
           for(i=p+1;i<n;i++)
           {
               if(a[i]>=maxi)
               {
                   maxi=a[i];
                   pos=i;
               }
           }
           if(maxi==-1)
            break;
        }
        i=0;
        if(sell>=0)
            printf("%lld\n",sell);
        else
            printf("%lld\n",i);
    }
    return 0;
}
