#include<iostream>
using namespace std;
int gcd(int a,int b)
{
    while( 1 )
    {
        a = a % b;
    	if( a == 0 )
			return b;
		b = b % a;
        if( b == 0 )
			return a;
    }
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int a,b=180;
        cin>>a;
        if(a>=360)
        a%=360;
        int i=gcd(a,b);
        a/=i;
        b/=i;
        if(a==1 && b==1)
        cout<<"$"<<endl;
        else
        if(a==1)
        cout<<"$/"<<b<<endl;
        else
        if(b==1)
        cout<<a<<"$"<<endl;
        else
        cout<<a<<"$/"<<b<<endl;
         
    }
    return 0;
}
