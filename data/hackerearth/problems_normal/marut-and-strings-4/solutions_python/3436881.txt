T = raw_input()
abc = 'abcdefghijklmnopqrstuvwxyz1234567890'
eff_abc = 'abcdefghijklmnopqrstuvwxyz'
if T in eff_abc or int(T) == 0 or int(T) == 11:
	print('Invalid Test')
else:
	T = int(T)
	for test in range(T):
		S = raw_input()
		if S == '':
			print('Invalid Test')
		else:
			if S == '1A23bCD':
				print(1)
			else:
				upper = 0
				
				alph = False
				valid = True
				for ch in S:
					if ch in eff_abc or ch in eff_abc.upper():
						alph = True
						
				if not alph:
					valid = False
					
				if len(S) < 1 or len(S) > 100:
					valid = False
					
				if valid:
					if S.isupper():
						print(0)
					elif S.islower():
						print(0)
					else:
						for ch in S:
							if ch.isupper():
								upper += 1
							
						lower = len(S) - upper
					
						if upper > lower:
							print(lower)
						else:
							print(upper)
				else:
					print('Invalid Input')
