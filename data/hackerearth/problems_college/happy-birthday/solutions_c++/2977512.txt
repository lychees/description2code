#include <iostream>
#include <cstdio>
using namespace std;

const int Maxd = 5;

int n;
int f[Maxd];
int res;

int main()
{
    scanf("%d", &n);
    for (int i = 0; i < n; i++) {
    	int a; scanf("%d", &a);
    	f[a]++;
    }
    res = f[4];
    res += f[3];
    f[1] = max(0, f[1] - f[3]);
    res += f[2] / 2; f[2] %= 2;
    if (f[2]) { res++; f[1] = max(0, f[1] - 2); }
    res += f[1] / 4; f[1] %= 4;
    if (f[1]) res++;
    printf("%d\n", res);
    return 0;
}
