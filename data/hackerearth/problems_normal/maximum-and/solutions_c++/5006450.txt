#include <bits/stdc++.h>
 
using namespace std;

#define IN          int t; cin >> t; while(t--)
 
int main()
{
    IN
    {
    	unsigned long long int a,b,d;
    	cin >> a >> b;
    	if(b&1)//odd
    	d = (b-1);
   		else
   		d = (b-2);
    	
    	if(d<a)
    	cout << (a&b) << endl;
    	else
    	cout << d << endl;
    	
    }
}
