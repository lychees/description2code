T = int(raw_input())
while T > 0:
    s = raw_input()
    n = int(raw_input())
    zero, one, toggle = 0, 0, 0
    for i in s:
        if i == '0':
            zero += 1
        else:
            one += 1
    for i in range(1, n+1):
        toggle += i

    if toggle % 2 == 0:
        print one
    else:
        print zero

    T -= 1
