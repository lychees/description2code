#include<iostream>
using namespace std;
typedef long long ll;
ll CountDigit(ll a)
{
    ll co=0,prev=0,a1,a2;
    while(a!=0)
    {
     a1=a%2;
     if((prev==1)&&(a1==0))
     {
         if(a/2!=0)
         {
             a2=a/2;
             if(a2%2==1)
             co++;

         }
     }
     if(a1==1)
     prev=1;
     else
     prev=0;
     a=a/2;
    }
    return co;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        ll r,k,i,c=0,cou;
        cin>>r>>k;
        for(i=1;i<=r;i++)
        {
            cou=CountDigit(i);
            if(cou>=k)
            {
               // cout<<"i= "<<i<<endl;
              c++;
            }

        }
        cout<<c<<endl;
    }
    return 0;
}
