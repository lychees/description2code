vector<int> g[mx];
int dfs_low[mx],dfs_num[mx],par[mx],dfs_time=1;
bool vis[mx];
int store[mx],det[mx],mar[mx];
void ae(int u,int v)
{
  g[u].push_back(v);
  g[v].push_back(u);
}
int P,cur,ans;
void dfs(int u)
{
   dfs_num[u]=dfs_time++;
   dfs_low[u]=dfs_num[u];
   vis[u]=true;
   int ct=0;
   mar[u]=1;
   for(int i=0;i<g[u].size();i++)
   {
     int w=g[u][i];
     if(!vis[w])
     {
       ct++;
       par[w]=u;
       dfs(w);
       dfs_low[u]=min(dfs_low[u],dfs_low[w]);
       mar[u]+=mar[w];
       if(dfs_num[u]<dfs_low[w])
         if(abs(mar[w]-(store[det[u]]-mar[w]))<P)
           ans++;
     }
     else
      if(vis[w]&&par[u]!=w)
        dfs_low[u]=min(dfs_low[u],dfs_num[w]);
   }
}
void DFS(int u)
{
  vis[u]=true;
  store[cur]++;
  det[u]=cur;
  for(int i=0;i<g[u].size();i++)
   {
      int w=g[u][i];
      if(!vis[w])
        DFS(w);
   }
}
void solve()
{
  int i,j,x,y,n,m;

    scanf("%d%d%d",&n,&m,&P);
    for(i=1;i<=n;i++)
     g[i].clear();
     memset(par,0,sizeof par);
     memset(vis,false,sizeof vis);
     memset(dfs_low,0,sizeof dfs_low);
     memset(dfs_num,0,sizeof dfs_num);
     memset(store,0,sizeof store);
     memset(det,0,sizeof det);
     memset(mar,0,sizeof mar); 
     while(m--)
     {
      cin>>x>>y;
      ae(x,y);
     }
     rep(i,1,n)
       if(!vis[i])
          {
            DFS(i);
            cur++;
          }
     memset(vis,false,sizeof vis);
     rep(i,1,n)
     {
      if(!vis[i])
      {
        par[i]=-1;
        dfs(i);
      }
     } 
     cout<<ans;
  return ;
}
