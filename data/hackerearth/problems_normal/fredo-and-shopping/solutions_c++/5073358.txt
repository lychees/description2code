#include <bits/stdc++.h>

using namespace std;
const int N=1000005;
int ft[N];
int n;
int a[N];
int b[N];
void add(int p){
    for(int i=p;i<N;i+=(i&(-i))){
        ft[i]++;
    }
    
}
int get(int p){
    int ret=0;
    for(int i=p;i;i-=(i&(-i))){
        ret+=ft[i];
    }
    return ret;
}

int main(){
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%d",&a[i]);
        b[i]=a[i];
    }
    unordered_map<int,int> idx;
    idx.clear();
    sort(b,b+n);
    int pos=1;
    for(int i=0;i<n;i++){
        if(!idx.count(b[i])){
            idx[b[i]]=pos++;
        }
    }
    
    
    for(int i=0;i<n;i++){
        if(i){cout<<" ";}
        pos=idx[a[i]];
        cout<<get(pos-1);
        add(pos);
        
        
    }
    
    
    
    return 0;
}
