#include <iostream>
#include <algorithm>
#include <vector>
#include <limits.h>
using namespace std;

#define pii pair<int,int>
#define ppi pair<pii,int>
#define MP make_pair
#define ff first
#define ss second

int dp[258];
bool cmp(ppi a, ppi b){
	return ((a.ff > b.ff)	|| (a.ff == b.ff && a.ss > b.ss));
}
int main()
{
    int t;
    cin>>t;
    while(t--){
    	vector<ppi>v;
    	int l,k,a,b,c;
    	cin>>l>>k;
    	for(int i=0;i<k;i++){
    		cin>>a>>b>>c;
    		v.push_back(MP(MP(a,b),c));
    		sort(v.begin(),v.end());
    	}
    	for(int i=0; i<=l; i++) dp[i]=90000000;
    	dp[0] = 0;
    	
    	for(int i=0;i<v.size();i++){
    		
    		a = v[i].ff.ff;
    		b = v[i].ff.ss;
    		c = v[i].ss;
    		//cout<<a<<" "<<b<<" "<<c<<"\n";
    		for(int j = a; j < b; j++){
    			if(dp[j] + c < dp[b]) dp[b] = dp[j] + c;
    		}
    		for(int j = a; j < b; j++){
    			if(dp[j] > dp[b]) dp[j] = dp[b];
    		}
    	}
    	if(dp[l] == 90000000) cout<<"-1\n";
    	else cout<<dp[l]<<"\n";
    }
    return 0;
}
