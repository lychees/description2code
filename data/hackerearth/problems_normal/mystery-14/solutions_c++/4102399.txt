#include <iostream>
using namespace std;

int main()
{long long int t,n[10],sum=0,j=0,i,values[10][100000],k;
 cin>>t;
 while(t>0)
  {cin>>n[j];
   for(i=0;i<n[j];i++)
    cin>>values[j][i];
   j++;
   t--;
  }
 for(i=0;i<j;i++)
  {sum=0;
   for(k=0;k<n[i];k++)
    {if(values[i][k]%2==1)
      sum+=values[i][k];
    	
    }
   cout<<sum<<endl; 
  }
 return 0; 
}
