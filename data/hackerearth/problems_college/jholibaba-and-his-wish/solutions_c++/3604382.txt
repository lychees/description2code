#include <iostream>
using namespace std;

int main()
{
    long long int n,i,j,temp,k,l,mx=0,sm=0;
    cin>>n;
    long long int a[n];
    for(i=0;i<n;i++){
    	cin>>a[i];
    	sm=sm+a[i];
    	mx=mx+(i+1)*a[i];
    }
    temp=mx;
    for(i=0;i<n;i++){
    	temp=temp-sm+n*a[i];
    	if(temp>mx)
    		mx=temp;
    }
    cout<<mx<<endl;
    return 0;
}
