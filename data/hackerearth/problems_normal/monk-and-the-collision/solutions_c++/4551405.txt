
#include<iostream>
#include<bits/stdc++.h>

int main(){
int t;
 cin>>t;
  while(t--)
  {   long long  int a[100000],i,c=0,con=0;
      int N;
      cin>>N;
       int b[100];
       for(  i=0;i<N;i++)
       { cin>>b[i];
    
        }
        for( i=0;i<100000;i++)
        {
        	a[i]=0;
        }
        for( i=0;i<N;i++)
        {
        	 c=(b[i])%10;
        	a[c]=a[c]+1;
        	if(a[c]>1)
        	{
        		con++;
        	}
        	
        	
        } 
       
        cout<<con<<endl;;
        
   
  }

return 0;

  
  }
