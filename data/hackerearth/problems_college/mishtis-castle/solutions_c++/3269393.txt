#include <bits/stdc++.h>

using namespace std;

int main()
{
	int i,j,k,l,m,n,t,c;
	
	cin >> t;
	
	while(t--)
	{
		cin >> n >> c;
		
		int a[n+1][n+1];
		//int b[n+1][n+1];
		
		for(i=1;i<=n;i++)
		{
			for(j=1;j<=n;j++)
				cin >> a[i][j];
		}
		
		
		for(i=0;i<=n;i++)
		{
			a[0][i] = 0;
		}
		
		for(i=0;i<=n;i++)
		{
			a[i][0] = 0;
		}
	
		
	/*	for(i=0;i<=n;i++)
		{
			for(j=0;j<=n;j++)
				cout << a[i][j] << " ";
				
			cout << endl;
		}*/
		
	
		for(i=1;i<=n;i++)
		{
			for(j=1;j<=n;j++)
			{
				a[i][j] += max(a[i-1][j], a[i][j-1]);
			}
		}
		
		if(a[n][n] >= c)
			cout <<a[n][n]-c <<endl;
		else
			cout << -1 <<endl;
	}
}
