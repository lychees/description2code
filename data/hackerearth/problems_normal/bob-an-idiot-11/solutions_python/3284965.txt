N=int(raw_input())
alpha='abcdefghijklmnopqrstuvwxyz'
alphaDict={}
for char in alpha:
    alphaDict[char]=char

for _ in xrange(0,N):
    a,b=[x.lower() for x in raw_input().split()][:2]
    for k,v in alphaDict.iteritems():
        if(v==a):
            alphaDict[k]=b
        elif(v==b):
            alphaDict[k]=a

cf=raw_input()
resultList=[]
for char in cf:
    if char.isalpha():
        if char.isupper():
            resultList.append((alphaDict[char.lower()]).upper())
        else:
            resultList.append(alphaDict[char])
    else:
        resultList.append(char)

print "".join(resultList)
