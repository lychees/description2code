#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    char a[110];
    cin >> a;
    int m,q,w,i,j,count;
    cin >> m;
    while(m--)
    {
    	count = 0;
    	cin >> q >> w;
    	for(i=q-1;i<w-1;i++)
    	{
    		if(a[i] == a[i+1])
    			count++;
    	}
    	cout << count << endl;
    	count = 0;
    }
    return 0;
}
