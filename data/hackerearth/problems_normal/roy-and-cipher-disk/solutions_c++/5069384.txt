#include <iostream>
#include <cmath>
using namespace std;
int cipher(char c) {
	return c-'a';
}

int main()
{
	int t;
	cin>>t;
	while(t--) {
    string s;
    cin>>s;
    int pos=0,n=s.length(),dis;
    for(int i=0;i<n;i++) {
    	int diff=cipher(s[i])-pos;
    	if(abs(diff)<13)
    	dis=diff;
    	else{
    	dis=26-abs(diff);
    	if(diff>0 && dis!=13)
    		dis*=-1;
    	}
    	cout<<dis<<" ";
    	pos=cipher(s[i]);
    }
    cout<<endl;
	}
    return 0;
}
