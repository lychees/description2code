#include <iostream>

using namespace std;

int r[31];
const int M = 1000000007;

int mult(long long a, long long b) {
  return (int)((a * b) % M);
}

int gcd(int x, int y) {
  if (y == 0) return x;
  return gcd(y, x % y);
}

int power(int x, int y) {
  if (y == 0) return 1;
  if (y & 1) return mult(x, power(x, y-1));
  int h = power(x, y / 2);
  return mult(h, h);
}

int go(int rr) {
  int best = 0;
  int hi = rr * 5;
  for (int x = rr; x <= hi; x += rr) {
    for (int y = 0; y <= hi; y += rr) if (gcd(x, y) == rr) {
      best = max(best, power(x, y));
    }
  }
  return best;
}

int main() {
  r[0] = r[1] = r[2] = 1;
  r[3] = r[4] = 3;
  r[5] = r[6] = 15;
  r[7] = r[8] = r[9] = r[10] = 105;
  for (int i = 11; i <= 30; i++)
    r[i] = mult(mult(1ll << ((i+3)/4), r[(i+4)/5]), r[(i+9)/10]);

  int t; cin >> t;
  while (t--) {
    int n; cin >> n;
    cout << go(r[n]) << endl;
  }

  return 0;
}
