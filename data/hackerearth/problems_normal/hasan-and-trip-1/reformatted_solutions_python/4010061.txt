def d(x, y):
    return ((x[0]-y[0])**2+(x[1]-y[1])**2)**0.5


n = int(raw_input())
l = []
for i in xrange(n):
    i, j, k = map(int, raw_input().split())
    l.append((i, j, k))
dp = [-2494444444444]*(n+1)
# F=0
# L=n-1
dp[0] = l[0][2]
for i in xrange(1, n):
    for j in xrange(0, i):
        dp[i] = max(dp[i], dp[j]-d(l[i], l[j]))
    # print l[i][2]
    dp[i] += l[i][2]
print "%.6f" % dp[n-1]
