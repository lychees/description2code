#include <bits/stdc++.h>

using namespace std;

int a[100010];
map <int,int> mp;
int tot[100010];
int main()
{
    int t,n,i,j,cnt=1,ans=0,s=0,e=0;

  //freopen("input.txt","r",stdin);
    cin>>t;

    while(t--)
    {
        s=0;
        e=0;
        cin>>n;
        ans=0;
        mp.clear();
        cnt=1;
        for(i=0;i<100005;i++)
            tot[i]=0;
        for(i=0;i<n;i++)
        {
            cin>>a[i];
            if(mp.find(a[i])==mp.end())
                {
                mp[a[i]]=cnt++;
                //cout<<mp[a[i]];
                a[i]=mp[a[i]];
                }
            else
            {
                a[i]=mp[a[i]];
            }
        }

    /*for(i=0;i<n;i++)
        cout<<a[i]<<" ";
    cout<<endl;*/
    j=0;
    for(i=0;i<n;)
    {
        if(tot[a[i]]==0)
        {
            tot[a[i]]=1;
            i++;
        }
        else
        {
            while(a[i]!=a[j])
            {
                tot[a[j]]=0;
                j++;
            }
            tot[a[j]]=0;
            j++;
        }
        //cout<<"i"<<i<<"j"<<j<<endl;
        if(i-j>ans)
        {
            ans=i-j;
            s=j+1;
            e=i;
        }
    }
    cout<<s<<" "<<e<<endl;
    }
}
