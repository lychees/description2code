#include <bits/stdc++.h>

using namespace std;

#define LL long long int 
#define sd(x) scanf("%d", &x)
#define F first
#define S second
#define mp make_pair
#define pb push_back
#define N 2000000
#define pii pair<int,int>

int main(){
	int t, n, i, j, x, y;
	sd(t);
	while(t--){
		sd(n);
		int minx = 100000;
		int miny = 100000;
		int maxx = -100000;
		int maxy = -100000;
		for(i = 0; i < n; i++){
			sd(x); sd(y);
			minx = min(minx, x);
			maxx = max(maxx, x);
			miny = min(miny, y);
			maxy = max(maxy, y);
		}
		if(minx == maxx){
			cout<<maxy - miny<<endl;
		}
		else if(miny == maxy){
			cout<<maxx - minx<<endl;
		}
		else{
			cout<<(maxx - minx) * (maxy - miny)<<endl;
		}
	}
	return 0;
}
