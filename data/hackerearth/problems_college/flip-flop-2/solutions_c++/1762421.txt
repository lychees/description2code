#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;

int main()
{
   	int t,i;
   	scanf("%d",&t);
   	while(t--)
   	{
   		string s;
   		cin>>s;
   		int l=s.length();
   		int st=s[0]-'X';
   		int ans=0;
   		for(i=1;i<l;i++)
   		{
   			if(st)
   			{
   				while(i<l&&s[i]=='Y') {
   					i++;
   					ans++;
   				}
   				st=0;
   			}
   			else
   			{
   				while(i<l&&s[i]=='X') {
   					i++;
   					ans++;
   				}
   				st=1;
   			}
   		}
   		cout<<ans<<endl;
   	}
   	return 0;
}
