import sys, math
M=[]

for i in range(3):
  M.append(map(float, raw_input().split()))
  
mxs=mys=0
#print M
for i in M:
  mxs=mxs+i[0]
  mys=mys+i[1]
#print mxs,mys
res=[]
for i in range(3): 
  res.append([round((mxs-(2*M[i][0])),4),round((mys-(2*M[i][1])),4)])
  
final=sorted(res)
for i in range(3):
  print "%.4f %.4f" %(final[i][0], final[i][1])
