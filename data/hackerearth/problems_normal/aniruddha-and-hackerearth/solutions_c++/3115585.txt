#include "bits/stdc++.h"
using namespace std;

typedef long long ll;
typedef vector<int> vi;
typedef vector< vi > vii;
typedef pair<int, int> pii;
#define FF first
#define SS second
#define pb(v) push_back(v)
#define mp(x,y) make_pair(x, y)
#define swap(x,y) x=x+y-(y=x)

#define s(n) scanf("%d",&n)
#define rep(i,start,end) for(int i = start;i<end;i++)

const int MAX = 1e6, M = 1e9 + 7;
ll DP[MAX+5] = {0};
ll ans[MAX+5] = {0};

void build(){
    DP[0] = 0;
    DP[1] = 2;
    DP[2] = 3;
    ans[1] = 2;ans[2] = 5;
    rep(i,3,MAX+1){
        DP[i] = (DP[i-1] + DP[i-2])%M;
        ans[i] = (ans[i-1] + DP[i])%M;
    }
}

int main(){
    int tt; s(tt);
    build();
    int n;
    while(tt--){
        cin>>n;
        cout<<ans[n]<<endl;
    }
}
