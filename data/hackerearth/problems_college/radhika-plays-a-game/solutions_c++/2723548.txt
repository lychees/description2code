#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
    long long int t;
    scanf("%lld",&t);
    while(t--)
    {
        long long int n,k;
        scanf("%lld%lld",&n,&k);
        if((n%(k+1))==1)
        {
            printf("Nandu\n");
        }
        else
        {
            printf("Rani\n");
        }
    }
return 0;
}
