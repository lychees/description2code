from fractions import gcd
for i in range(input()):
    a, b, c = map(int, raw_input().split())
    m = a*b*c
    ans = gcd(int(a), gcd(int(b), int(c)))
    fin = ((int(m))/(ans*ans*ans)) % 1000000007
    print ans, fin
