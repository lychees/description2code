#include <iostream>
#include<iomanip>
using namespace std;
 
int main()
{
   long long int t,max,l,b,h;
   double r;
   cin>>t;
   while(t--)
   {
   	cin>>l>>b>>h;
   	if((l<=b) && (l<=h))
   	max=l;
   	else if(b<=l&&b<=h)
   	max=b;
   	else
   	max=h;
   	r=(double)max/2;
   	cout<<setprecision(1)<<std::fixed;
   	cout<<r<<endl;
   }
    return 0;
    
}
