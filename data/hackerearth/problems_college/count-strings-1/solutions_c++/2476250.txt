#include <bits/stdc++.h>
using namespace std;
#define Size 1000005
#define Mod 1000000007
#define MAX 200005

int cnt[27];
char s1[1000005];
char s2[1000005];
int len1,len2;

long long solve(){
	long long res = 1;
	for(int i = 0;i<len2;i++){
		int pos = (int)(s2[i]-'a');
		if(cnt[pos] == 0) return 0;
		res = (res * cnt[pos])%Mod;
		cnt[pos]--;
	}
	return res;
}

int main() {
	int nCase;
	scanf("%d",&nCase);
	for(int cs = 1;cs<=nCase;cs++){
		scanf("%s %s",s1,s2);
		memset(cnt,0,sizeof(cnt));
		len1 = strlen(s1);
		len2 = strlen(s2);
		for(int i = 0;i<len1;i++){
			int pos = (int)(s1[i]-'a');
			cnt[pos]++;
		}
		long long res = solve();
		printf("%lld\n",res);
	}
}
