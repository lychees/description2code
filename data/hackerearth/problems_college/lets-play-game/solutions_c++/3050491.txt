#include <iostream>
using namespace std;
typedef long long ll;
int main()
{
	ll tc;
	cin>>tc;
	while(tc--)
	{
		ll n,k,a[100005],b[100005],v,mul,max=0;
		cin>>n>>k;
		for(int i=0;i<n;i++)
			cin>>a[i];
		for(int i=0;i<n;i++)
			cin>>b[i];
		for(int i=0;i<n;i++)
		{
			v=k/a[i];
			mul=v*b[i];
			if(max<mul) max=mul;
		}
    	cout << max << endl;
	}
    return 0;
}
