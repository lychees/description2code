#include <iostream>
#define M 1000000007
using namespace std;
long long modulo(long long a,long long b,long long c){
long long x=1,y=a; // long long is taken to avoid overflow of intermediate results
while(b > 0){
if(b%2 == 1){
x=(x*y)%c;
}
y = (y*y)%c; // squaring the base
b /= 2;
}
return x%c;
}
int main()
{
	int t,n;
	long long q,x;
	
	cin>>t;
	
	while(t--)
	{
		cin>>n;
		q=0;
		for(int i=0;i<n;i++)
		{
			cin>>x;
			q |= x;
		}
		cout<<(modulo(2,n-1,M)*q)%M<<'\n';
	}
  
    return 0;
}
