#include <bits/stdc++.h>
#define MOD 1000000007
#define ll long long
using namespace std;

ll power(ll a, ll p){
	ll ret = 1;
	while (p > 0){
		if (p & 1){
			ret *= a;
		}
		a *= a;
		p >>= 1;
	}
	return ret;
}

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	map<long long, long long>st;
	ll mxx = 0;
	for (int i = 0; i<63; i++)
	{
		for (int j = 0; j<63; j++)
		{
			if (i == j) continue;
			long long x = power(2, i) + power(2, j);
			mxx = max(mxx, x);
			st[x] = x;

		}
	}
	map<ll, ll>::iterator it = st.begin();
	long long tot = it->second;
	it++;
	while (it != st.end())
	{
		tot += it->second;
		tot %= MOD;
		it->second = tot;
		it++;
	}
	int tc;
	cin >> tc;
	while (tc--)
	{
		long long n;
		cin >> n;
		if (n < 3)
		{
			cout << "0\n";
			continue;
		}
		it = st.lower_bound(n);
		if (it->first == n && it != st.end())
		{
			cout << it->second << "\n";
		}
		else
		{
			it--;
			cout << it->second << "\n";
		}
	}
}
