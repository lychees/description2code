#include<bits/stdc++.h>
using namespace std;
#define ll long long
int arr[100004];
int chk[100005];
int main()
{
  ll sum=0;
  int i,t,n;
  cin>>t;
  while(t--)
  {
      cin>>n;
      sum=0;
      ll range=INT_MAX;
      int sti,edi;
      for(i=1;i<=n;i++)
      {
          scanf("%d",&arr[i]);
          chk[i]=0;
      }
      for(i=1;i<=n;i++)
      {
          sum=sum+arr[i];
          sum=sum%n;
          if(sum==0)
          {
              if(i<range)
              {
                range=i;
                sti=1;
                edi=i;
              }
          }
          else if(chk[sum]==0)
          {
            chk[sum]=i;
          }
          else
          {
              if(i-chk[sum]<range)
              {
                  sti=chk[sum]+1;
                  edi=i;
                  range=i-chk[sum];
              }
              chk[sum]=i;
          }
      }
      cout<<sti<<" "<<edi<<endl;

  }

}
