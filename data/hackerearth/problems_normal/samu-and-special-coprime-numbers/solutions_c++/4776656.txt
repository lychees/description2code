#include <bits/stdc++.h>
using namespace std;

#define pb push_back
#define ff first
#define ss second
#define rz resize
#define len length
#define nl '\n'

typedef long long ll;
typedef pair<int,int> pii;
typedef vector<int> vi;
typedef vector<pii> vpii;


ll dp[20][200][1500];
int dig[20];

ll f(int idx,int tight,int s,int sq){

  if(idx==-1)
    return (__gcd(s,sq)==1);

  if(!tight and dp[idx][s][sq]>=0)
    return dp[idx][s][sq];

  ll ans = 0;

  int ub = (tight?dig[idx]:9);

  for(int last=0;last<=ub;last++)
    ans = (ans + f(idx-1,tight and last==dig[idx],s + last, sq + last*last) );

  if(!tight)
    dp[idx][s][sq]=ans;
  return ans;
}

ll get(ll x){
  if(x==0)
    return 0;

  int len=0;
  while(x){
    dig[len++]=x%10;
    x/=10;
  }
  return f(len-1,1,0,0);
}

int main(){
  ios::sync_with_stdio(0);
  cin.tie(0),cout.tie(0);

  memset(dp,-1,sizeof(dp));

  int t;
  ll l,r;
  cin>>t;
  while(t--){
    cin>>l>>r;
    cout<<get(r)-get(l-1)<<nl;
  }

  return 0;
};
