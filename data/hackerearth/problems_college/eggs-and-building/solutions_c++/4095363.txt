using namespace std;
int eggp(int k, int n)
{

    int egg[n + 1][k + 1];
    int i,j,x;
    for(i = 1;  i <= n ; i++)
    {
        egg[i][1] = 1;
        egg[i][0] = 0;
    }
    for(j = 1 ; j <= k; j++)
    {
        egg[1][j] = j;
    }
    int res = 0;
    for(i = 2 ; i <= n; i++)
    {
        for(j = 2 ; j <= k ; j++)
        {
            egg[i][j] = INT_MAX;
            for(x = 1 ; x <= j; x++)
            {
            res = 1 + max(egg[i - 1][x - 1],egg[i][j - x]);
            if(res < egg[i][j])
            {
            egg[i][j] = res;
            }
        }
        }
    }
    return egg[n][k];
}
int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		int k,n;
		cin >> k >> n;
		cout<< eggp(k,n)<<endl;
	}
	return 0;
}
