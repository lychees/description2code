#include<iostream>
#include<list>
#include<stack>
#include<vector>
using namespace std;
 
#define IOS ios_base::sync_with_stdio(false);cin.tie(0)
 
int main() {
  IOS;
  int t, n, k;
  cin >> t;
  stack<int> result;
  stack<int> ret;
  int popularity;
  while (t--) {
    while (!result.empty()) {
      result.pop();
    }
    while (!ret.empty()) {
      ret.pop();
    }
    cin >> n >> k;
    int delCount = 0;
    for (int i = 0; i < n; i++) {
      cin >> popularity;
      if (result.empty()) {
        result.push(popularity);
      }
      else {
        while (!result.empty() && delCount < k) {
          if (result.top() < popularity) {
            result.pop();
            delCount++;
          }
          else {
            result.push(popularity);
            break;
          }
        }
        if (result.empty()||delCount==k) {
          result.push(popularity);
        }
 
      }
    }
    while (delCount <k) {
      result.pop();
    }
    while (!result.empty()) {
      ret.push(result.top());
      result.pop();
    }
    while (!ret.empty()) {
      cout << ret.top() << " ";
      ret.pop();
    }
 
    cout << endl;
 
  }
  return 0;
}
