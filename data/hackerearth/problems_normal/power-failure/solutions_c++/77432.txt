#include <stdio.h>
#include <algorithm>
#define mod 1000000007
using namespace std;
long long int arr[100006];
long long int prime[100006];
long long int prime_no[100006];
int main()
{
int t,i,j;
long long int n,V,x,num,f;
prime[1]=0;
for(i=2;i*i<=100005;i++)
{
if(prime[i]==0)
{
for(j=i*i;j<=100005;j+=i)
prime[j]=1;
}
}
for(i=2;i<=100005;i++)
{
if(prime[i]==0)
prime_no[i]=prime_no[i-1]+1;
else
prime_no[i]=prime_no[i-1];
}
scanf("%d",&t);
while(t--)
{
f=1;
scanf("%lld %lld",&n,&V);
for(i=0;i<n;i++)
scanf("%lld",&arr[i]);
sort(arr,arr+n);
num=0;
for(i=n-1;i>=0;i--)
{
x=V-arr[i]+1-(prime_no[V]-prime_no[arr[i]-1]);
x=x-num;
if(x<0)
x=0;
f=(f*x)%mod;
num+=1;
}
printf("%lld\n",f%mod);
}
system("pause");
return 0;
}
