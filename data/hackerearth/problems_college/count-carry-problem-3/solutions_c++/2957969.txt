#include<iostream>
#include<vector>
typedef long long int ll;
using namespace std;
int main()
{
	ll t;
	cin>>t;
	while(t--)
	{
		ll a,b,r1,r2,c=0;
		cin>>a>>b;
		vector<ll>aa,bb;
		while(a>0)
		{
			aa.push_back(a%10);
			a/=10;
		}
		while(b>0)
		{
			bb.push_back(b%10);
			b/=10;
		}
		ll mn=min(aa.size(),bb.size()),cry=0;
		for(ll i=0;i<mn;i++)
		{
			if(aa[i]+bb[i]>9)
			{c++;
			cry=1;}
			else
			{
				if(aa[i]+bb[i]+cry>9)
				c++;
			}
		}
		if(c==0)
		cout<<"No carry operation\n";
		else if(c==1)
		cout<<c<<" carry operation\n";
		else cout<<c<<" carry operations\n";
	}
}
