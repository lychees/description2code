T = int(raw_input())
S = []
V = ['a','e','i','o','u','A','E','I','O','U']

	S.append(raw_input())
for i in range(T):
	X = list(S[i][4:len(S[i])-4])
	Z = [m for m in X if m not in V]
	print str(len(Z) + 4) + "/" + str(len(S[i]))
