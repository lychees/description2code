#include<bits/stdc++.h>
using namespace std;
long int count ( long int S[], long int m, long int n )
{
    if (n == 0)
        return 1;
    if (n < 0)
        return 0;
    if (m <=0 && n >= 1)
        return 0;
    return count( S, m - 1, n ) + count( S, m, n-S[m-1] );
}

	long int m=i;
    printf("%ld ", count(arr, m, n));
    return 0;
}
