#include <bits/stdc++.h>

using namespace std;

int ar[21]={1,9,45,55,99,297,703,999,2223,2728,4950,5050,7272,7777,9999,17344,22222,77778,82656,95121,99999};

int main()
{
	int p,q;
	cin>>p>>q;
	bool output=false;
	for(int i=0;i<21;++i)
		if(ar[i]>=p && ar[i]<=q)
		{
			output=true;
			cout<<ar[i]<<" ";
		}
	if(!output)
		cout<<"INVALID RANGE";
}
