#include<stdio.h>
#include<iostream>
using namespace std;

int main()
{
	long int t;
	cin>>t;
	while(t)
	{
		long int n , m , x;
		cin>>n>>m>>x;
		long int check = (n+1) * x;
		long int sum = 0;
		for(long int i = 0 ; i < n ; i++)
		{
			long int num;
			cin>>num;
			sum = sum + num; 
		}
		if(sum >= check)
		{
			cout<<"1"<<endl;
		}
		else
		{
			if((check-sum) > m)
				cout<<"Impossible"<<endl;
			else
				cout<<(check-sum)<<endl;
		}
		t--;
	}
	return 0;
}
