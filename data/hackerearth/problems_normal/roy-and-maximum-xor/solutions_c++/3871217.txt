#include <iostream>
using namespace std;

int main()
{
	long long int t,z,n,k;
    cin>>t;
    while(t--){
    	z=1;
    	cin>>n;
    	while(n>z)z=z<<1;
    	//cout<<z<<endl;
    	//z=z>>1;
    	if(n==z)cout<<z-1<<" "<<n<<endl;
    	else
    	{
    		k=z>>1;
    		cout<<z-1<<" "<<(n-k)*2<<endl;
    	}
    }
    return 0;
}
