#include <cstdlib>

using namespace std;

#define ll long long
#define MOD 1000000007
#define gc getchar_unlocked


void scanint(ll int &x){

	char c = gc();
	x = 0;
	while (c<'0' || c>'9')c = gc();
	while (c >= '0' && c <= '9'){
		x = x * 10 + c - '0';
		c = gc();
	}

}



int main(){

	ll int t, n,c=0;
	scanf("%lld", &t);
	while (t--){ scanint(n); if (n < 300000000)c++;}
	cout << c;
	return 0;
}
