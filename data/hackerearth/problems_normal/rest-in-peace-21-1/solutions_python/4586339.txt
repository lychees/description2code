n = int(raw_input())

while n>0:
	s = int(raw_input())
	if s % 21 == 0:
		print "The streak is broken!"
	elif str(s).find(str(21)) >= 0:
		print "The streak is broken!"
	else:
		print "The streak lives still in our heart!"
	n=n-1
