#include <iostream>
#include<stdio.h>
#include<string.h>
#include<math.h>
using namespace std;
int main()
{
	 char s[1000000];
	 int hash1[26]={0},n;
	 int cnt=0;
	 cin>>n;
	 int t=n;
	 while(n--)
	 {
	 	cin>>s;
	 	int hash[26]={0};
	 	for(int i=0;s[i]!='\0';i++)
	 	{
	 		int x=s[i]-'a';
	 		hash[x]++;
	 	}
	 	for(int i=0;i<26;i++)
	 	{
	 		if(hash[i]>0)
	 		{
	 			hash1[i]++;
	 		}
	 	}
	 }
	 for(int i=0;i<26;i++)
	 {
	 	if(hash1[i]==t)
	 	cnt++;
	 }
	 cout<<cnt<<endl;

}
