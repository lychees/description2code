#include<cstdio>
#include<map>
#include<string>
std::map<std::string,int> m;
long int cost[1004][1004];
int arr[1004];
int main()
{
	long long ans;
	char str[50];
	int i,j,n,p;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%s",str);
		m[str]=i;
	}
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=n;j++)
		{
			scanf("%ld",&cost[i][j]);
		}
	}
	scanf("%d",&p);
	for(i=0;i<p;i++)
	{
		scanf("%s",str);
		arr[i]=m[str];
	}
	ans=cost[1][arr[0]];
	for(i=1;i<p;i++)
	{
		ans+=cost[arr[i]][arr[i-1]];
	}
	printf("%lld",ans);
	return 0;
}
