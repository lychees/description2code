#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int i,j,n,k,x,l=-1;
        cin>>n>>k;
        int a[n];
        for(i=0;i<n;i++)
            cin>>a[i];
        i=0;j=0;
        //cout<<n<<" "<<k<<" "<<i<<"\n";
        while(i<n && k>0)
        {
            l=i;x=a[i];
            for(j=i;j<=min(i+k,n-1);j++)
                if(x>a[j])
                    x=a[j],l=j;
            //cout<<i<<" "<<j<<" "<<l<<" "<<x<<"\n";
            if(l!=i)
            {
                k-=l-i;
                for(j=l;j>i;j--)
                    swap(a[j],a[j-1]);
            }
            i++;
        }
        for(i=0;i<n;i++)
            cout<<a[i]<<" ";
        cout<<"\n";
    }
}
