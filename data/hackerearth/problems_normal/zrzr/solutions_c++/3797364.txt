#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    long long n,a,sum;
    cin>>t;
    while(t--)
    {
    	a=5;
    	sum=0;
    	cin>>n;
    	while(1)
    	{
    		sum+=floor(n/a);
    		a*=5;
    		if(n/a==0)
    		{
    			break;
    		}
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
