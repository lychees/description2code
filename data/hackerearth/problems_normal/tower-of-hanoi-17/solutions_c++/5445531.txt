#include <bits/stdc++.h>
using namespace std;
struct interval{
	long long int rad;
	long long int hei;
};
interval a[200];
long long int dp[200];
bool comp(interval x,interval y){
	if(x.rad<y.rad){
		return true;
	}
	if(x.rad==y.rad && x.hei<y.hei){
		return true;
	}
	return false;
}
int main() {
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		for(int i=0;i<n;i++){
			long long int x,y;
			cin>>x>>y;
			interval temp;temp.rad=x;temp.hei=y;
			a[i]=temp;
		}
		sort(a,a+n,comp);
		memset(dp,0,sizeof(dp));
		dp[0]=a[0].hei;
		for(int i=1;i<n;i++){
			for(int j=i-1;j>=0;j--){
				if(a[i].rad>a[j].rad && a[i].hei>a[j].hei){
					dp[i]=max(dp[i],a[i].hei+dp[j]);
				}
			}
			dp[i]=max(dp[i],a[i].hei);
		}
		long long int ans=0;
		for(int i=0;i<n;i++){
			ans=max(ans,dp[i]);
		}
		cout<<ans<<endl;
	}
	return 0;
}
