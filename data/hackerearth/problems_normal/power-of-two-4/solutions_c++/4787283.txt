#include<iostream>
using namespace std;


int main() {

	long long int Total = 0;
	int testCase;
	cin >> testCase;


	for (int j = 0; j < 32; j++)
	{
		Total += (1<<j);
	}

	while (testCase--)
	{
		bool flag = false;
		int Number;
		cin >> Number;

		long long int myArray[100001];
		long long localSum;
		
		for(int i = 0; i < Number; i++)
		{
			cin >> myArray[i];
		}

		for (int l = 0; l < 32; l++)
		{
			localSum = Total;

			for (int m = 0; m < Number; m++)
			{
				if (myArray[m] & (1 << l))
					localSum &= myArray[m];
			}
			if (!(localSum&(localSum - 1)))
			{
				flag = true;
				break;
			}
		}

		if (flag)
			cout << "YES" << endl;
		else
			cout << "NO" << endl;
	}


	return 0;
}
