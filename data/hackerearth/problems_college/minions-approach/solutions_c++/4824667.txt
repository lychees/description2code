bool sieve[100010];

void pre() {
	memset(sieve, true, sizeof(sieve));
	sieve[0] = sieve[1] = false;
	for (int i = 2; i <= 100000; i++) {
		if (sieve[i]) {
			for (int j = i+i; j <= 100000; j += i)
				sieve[j] = false;
		}
	}
}

int etf(int n) {
	int ans = n;
	for (int i = 2; i*i <= n; i++) {
		if (n % i == 0) {
			while (n % i == 0)
				n /= i;
			ans -= ans / i;
		}
	}
	if (n > 1) ans -= ans / n;
	return ans;
}

int main() {
	pre();
	int t, n;
	scanf("%d", &t);
	while (t--) {
		scanf("%d", &n);
		printf("%s\n", sieve[etf(n)] ? "Correct" : "Incorrect");
	}
	return 0;
}
