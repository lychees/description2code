#include<bits/stdc++.h>
using namespace std;

int main(){
    unsigned long long int no, n, q, i, j, ans;
    cin>>no;
    for(i = 0 ; i < no; i++){
        cin>>n>>q;
        int h[n];
        ans = 0;
        for(j = 0; j < n; j++)
            cin>>h[j];

        sort(h,h+n);

        if(n == 1){
            cout<<"0"<<endl;
            continue;
        }

        for(j = 1; j < n; j++){
            ans+=h[j] - h[j-1];
        }

        ans*=q;
        cout<<ans<<endl;
    }
    return 0;
}
