

    #include <iostream>
    #include <stack>
    using namespace std;
     
    int main()
    {
        int q;
        int input;
        stack <int> pile;
        int cost;
        cin>>q;
         
        while(q--)
        {
         cin>>input;
         if(input==1)
         {
          if(pile.empty())
           cout<<"No Food"<<endl;
          else
          {
           cout<<pile.top()<<endl;
           pile.pop();
          }
         }
         else if(input==2)
         {
          cin>>cost;
          pile.push(cost);
         }
        }
        return 0;
    }
