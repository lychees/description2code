#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long n, sum, q, x, y, i; 
    cin>>n;
    long long a[n];
    for(i=0;i<n;i++)
    	cin>>a[i];
    cin>>q;
    while(q--){
    	sum=0;
    	cin>>x>>y;
    	for(i=x-1;i<y;i++)
    		sum+=a[i];
    	cout<<sum<<endl;
    	
    }
    return 0;
}
