#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
 ll t;
 cin>>t;
 while(t--)
 {
 	ll n;
 	cin>>n;
 	 int a[3]={0};
 	 ll temp;
 	for(ll i=0;i<n;i++)
 	{
 		cin>>temp;
 		a[temp%3]++;
 	}
 	//cout<<a[0]<<" "<<a[1]<<a[2]<<endl;
 	ll u=a[0];
 	u=u*(u-1);
 	u/=2;
 	cout<<a[1]*a[2]+u<<endl;
 }
}
