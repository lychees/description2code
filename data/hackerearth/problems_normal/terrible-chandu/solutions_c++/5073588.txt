#include <iostream>
#include <algorithm>

using namespace std;
int main()
{
    int Test;
    cin >> Test;
    for(int t=0; t<Test; t++){
        string n;
        //getline(cin,n);
        cin >> n;
        for(int i=n.length()-1; i>=0; i--){
            cout << n[i];
        }
        cout << endl;
    }
}
