#include <bits/stdc++.h>
using namespace std;


int main() {
    int n;
    cin >> n;

    while(n--){
        int divisors = 0, flag = 0, x;
        cin >> x;
        
        for(int i = 1; i <= sqrt(x); i++){
            if(x % i == 0){
                divisors += 1;
                if(i != sqrt(x)){
                    divisors += 1;
                }
            }
            if(divisors >= 4){
                flag = 1;
                break;
            }
        }

        if(flag)
            cout << "YES" << endl;
        else
            cout << "NO" << endl;
    }


    return 0;
}
