from collections import defaultdict

def go(arr, k):
    count = 0
    occurences = {}

    i = 0
    for n in arr:
        if n < k:
            occurences[n] = occurences.get(n, 0) + 1

    while occurences:
        n, n_count = occurences.popitem()
        m = k - n


        if m in occurences:
            count += n_count * occurences[m]
            del occurences[m]
        elif n * 2 == k:
            count += sum(range(n_count))

    print count

	k = int(raw_input())
	go(arr, k)
