    #include<stdio.h>
    #include<string.h>
    int main()
    {
    	int test;
    	scanf("%d",&test);
    	char d[6000],s[6000];
    	while(test--)
    	{
    		scanf("%s",&d[0]);
    		scanf("%s",&s[0]);
    		int i,j,ld,ls,t,ans;
    		ld=strlen(&d[0]);
    		ls=strlen(&s[0]);
    		if(ls<=ld)
    		ans=ls;
    		else
    		ans=ld;
    		if(ld<=ls)
    		{
    			for(i=0;i<(ls-ld+1);i++)
    			{
    				t=0;
    				for(j=0;j<ld;j++)
    				{
    					if(s[i+j]!=d[j])
    					{
    						t++;
    					}
    				}
    				if(t<ans)
    				ans=t;
    			}
    			printf("%d\n",ans);
    		}
    		else
    		{
    			for(i=0;i<ld-ls+1;i++)
    			{
    				t=0;
    				for(j=0;j<ls;j++)
    				{
    					if(s[j]!=d[i+j])
    					{
    						t++;
    					}
    				}
    				if(t<ans)
    				ans=t;
    			}
    			printf("%d\n",ans);
    		}
    	}
    	return 0;
    }
