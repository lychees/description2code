#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		int a[n],b[n];
		for(int i=0;i<n;i++){cin>>a[i];}
		for(int i=0;i<n;i++){cin>>b[i];}
		sort(a,a+n);
		sort(b,b+n);
		int s=0;
		for(int i=0;i<n;i++){s=s+(a[i]*b[i]);}
		cout <<s<< endl;
	}
    
    return 0;
}
