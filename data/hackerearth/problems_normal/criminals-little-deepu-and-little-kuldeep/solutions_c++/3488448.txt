#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;
 
int main()
{
    int t;
    cin>>t;
    while(t--)
    {   int n;
    	cin>>n;
    	vector <int> vec;
    	int x;
    	for(int i=0;i<n;i++)
    	{
    		cin>>x;
    		vec.push_back(x);
    	}
    	sort(vec.begin(),vec.end());
    	int bc=1,mc=1;
    	for(int i=0;i<vec.size()-1;i++)
    	{
    		if(vec[i]==vec[i+1])
    		{
    			bc++;
    		}
    		else
    		{
    			if(bc>mc)
    			mc=bc;
    			
    			bc=1;
    		}
    	}
    	if(bc>mc)
    	mc=bc;
    	cout<<mc<<endl;
    	
    }
    return 0;
}
