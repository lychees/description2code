inline void read(int &x){
    x=0;
    bool neg=false;
    register char c=gc();
    for(;c<'0' || c>'9';c=gc()){
        if(c=='-'){
            neg=true;
            c=gc();
            break;
        }
    }
        
    for(;c>='0' && c<='9';c=gc())
        x=(x<<3)+(x<<1)+(c-'0');
    if(neg)
        x*=-1;
}
inline void write(int x){

    register char buffor[35];
    register int i=0;
    do{
        buffor[i++]=(x%10)+'0';
        x/=10;
    } while(x);
    i--;
    while(i>=0) putchar_unlocked(buffor[i--]);
    putchar_unlocked('\n');
}
