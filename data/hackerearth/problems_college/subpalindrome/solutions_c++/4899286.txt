#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	char s[55];
    	cin>>s;
    	int l=strlen(s);
    	int a[55][55];
    	for(int i=l-1;i>=0;i--)
    	{
    		a[i][i]=1;
    		if(i+1<l)
    		{
    			if(s[i]==s[i+1])
    			a[i][i+1]=3;
    			else
    			a[i][i+1]=2;
    		}
    		for(int j=i+2;j<l;j++)
    		{
    			if(s[i]==s[j])
    			a[i][j]=(a[i][j-1])+(a[i+1][j])+1;
    			else
    			a[i][j]=(a[i][j-1])+(a[i+1][j])-(a[i+1][j-1]);
    		}
    	}
    	cout<<a[0][l-1]<<endl;
    	
    }
    return 0;
}
