	int T;
	int i,n,j;
	int re;
	double x;
	cin>>T;
	while(T--)
	{
		re=0;
		for(i=0;i<=6000;i++)
			b[i]=c[i]=0;
		cin>>n;
		for(i=0;i<n;i++)
			cin>>a[i];
		for(i=1;i<n;i++)

			for(j=0;j<=4000;j++)
			{
				c[j+a[i]]+=b[j];
			}
			for(j=0;j<i;j++)
			{
				b[a[i]+a[j]]++;
			}
		}
		for(i=1;i<=19;i++)
		{
			re+=c[i*i*i];
		}
		x=re;
		x/=n*(n-1)*(n-2)/6;
		printf("%.9lf\n",x);
	}

}
