main()
{
	int n,max=INT_MIN;
	cin>>n;
	int check=0;
	int x[n];
	for(int i=1;i<=n;i++)
	{
		cin>>x[i];		
		int k=x[i];
		if(k%2!=0){
		
		for(int j=2;j<=k/2;j++)
		{
			if(k%j==0) {check++; break;}
		}
		if(check==0 && k>max) max=k;}
		check=0;
	}
	cout<<max<<endl; check=0;
	for(int i=1;i<=n;i++)
	{
		if(x[i]==max) check++;
	}
	cout<<check;
}
