#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

ll c[1000005];

int main()
{
	ll n, m, ans, i, j, x;
	memset(c, 0, sizeof(c));
	
	cin>>n>>m;
	for(i = 0; i<n; i++){
		cin>>x;
		c[x]++;
	}
	ans = 0;
	i = 1000000;
	while(m){
		if(m>c[i]){
			ans = ans + c[i] * i;
			m = m - c[i];
			c[i-1] += c[i];
			i--;
		}
		else{
			ans = ans + m * i;
			m = 0;
			i--;
		}
	}
	cout<<ans<<endl;
    return 0;
}
