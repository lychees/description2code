#include <iostream>
#include<string>
using namespace std;

int main()
{
    string input;
    cin>>input;
    int arr[26];
    for(int i=0;i<26;i++)
    {
    	arr[i]=0;
    }
    int len=input.length();
    for(int i=0;i<len;i++)
    {
    	arr[input[i]-'a']++;
    }
    int count=0;
    for(int i=0;i<26;i++)
    {
    	if(arr[i]%2==1)count++;
    }
    char out[len];
    if(count>1)cout<<-1<<endl;
    else{
    	int j=-1;
    	int k=0;
    	for(int i=0;i<26;i++)
    	{
    		int cunt=arr[i];
    		if(cunt%2==0)
    		{
    			int c=cunt/2;
    			while(c--)
    			{
    				out[k]=(char)i+97;
    				k++;
    			}
    		}
    		else{
    			j=i;
    			int c=cunt/2;
    			while(c--)
    			{
    				out[k]=(char)i+97;
    				k++;
    			}
    			
    		}
    	}
    	if(j>=0)out[k++]=(char)j+97;
    	for(int i=0;i<=len/2;i++)
    	{
    		out[len-1-i]=out[i];
    	}
    	
    	for(int i=0;i<len;i++)
    	{
    		cout<<out[i];
		 }
    }
    return 0;
}
