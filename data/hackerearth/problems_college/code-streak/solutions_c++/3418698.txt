#include <iostream>
#include <vector>
#include <math.h>
#include <algorithm>



using namespace std;

int main(){


	int counter;

	cin >> counter;

	int problems, result, prob;
	int c, max_c;

	vector <int> prob_solved;
	for(int i = 0; i < counter; i++)
	{
		cin >> problems;
		c = 0;
		max_c = 0;
		for(int j = 0; j < problems; j++)
		{
			cin >> prob >> result;
			int found = 0;
			for(int u = 0; u < prob_solved.size(); u++)
			{
				if(prob == prob_solved[u])
				{
					found = 1;
					//cout << "solved ->" << prob << "already" << endl;
				}
					
			}

			if(result == 1 && found == 0)
			{
				c++;
				//cout << "Solved " << prob<<endl;
				prob_solved.push_back(prob);
				if(c > max_c)
					max_c = c;
			}
			if(result == 0)
				c = 0;

		}
		prob_solved.clear();
		cout << max_c << endl;

	}

	return 0;
}
