#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    int row,col,price,startx,starty,endx,endy,cost;

    cin>>t;
    while(t--)
    {
        cin>>row>>col;
        cin>>startx>>starty;
        cin>>endx>>endy;
        cin>>price;

        int y,z;
        int a,b;
        a=abs(startx-endx);
        b=abs(row-abs(startx-endx));
        
        y=min(a,b);
        a=abs(starty-endy);
        b=abs(col-abs(starty-endy));
        
        z=min(a,b);

        cost=(y+z)*price;
        cout<<cost<<endl;
        if(cost<=1000)
        {
            cout<<"You saved the group.\n";
        }
        else
        {
            cout<<"Let go of the group.\n";
        }
    }

    return 0;
}
