#include <bits/stdc++.h>
using namespace std;

int main() {
    int t,m,n,temp,count,t1,t2,t3;
    cin>>t;
    while(t--)
    {
        count=0;
        t3=0;
        cin>>m;
        cin>>n;
        vector<int>v;
        
        for(int i=0;i<n;i++)
        {
            cin>>temp;
            v.push_back(temp);
        }
        sort(v.begin(),v.end());
        
        	int k=0;
        	int j=v.size()-1;
        	while(v[j]>m)
        		j--;
            while(k<=j)
            {
            	if((v[k]+v[j])<=m)
            	{
            		k++;
            		j--;
            	}
            	else
            		j--;
            		
            	count++;
            }
            cout<<count<<"\n";
    }
	// your code goes here
	return 0;
}
