t=int(raw_input())
while t>0:
	nm=str(raw_input()).split()
	n=int(nm[0])
	m=int(nm[1])
	k=int(raw_input())
	i=1
	x=1
	y=1
	jumps=0
	while i<=k:
		ab=str(raw_input()).split()
		a=int(ab[0])
		b=int(ab[1])
		while True:
			if (x+a)>n or (y+b)>m or (x+a)<1 or (y+b)<1:
				break
			else:
				x=x+a
				y=y+b
				jumps=jumps+1
		i=i+1
	t=t-1
	print jumps
