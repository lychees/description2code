#include <iostream>
#include <algorithm>
#include <string>
using namespace std;

int sum(string s) {
  int cnt = 0;
  for (int i = 0; i < s.length(); i++) {
    cnt += (s[i] - '0');
  }
  if (cnt>9) return sum(to_string(cnt));
  else return cnt;
}

int main() {
  int n;
  cin >> n;
  string s = string(1000,'0');
  for (int i = 0; i < n; i++) {
    string t; cin >> t;
    int carry = 0;
    for (int j = 0; j < t.length(); j++) {
      int p = s[j] - '0';
      int q = t[j] - '0';
      int r = p + q + carry;
      if (r>9) {
        carry = r / 10;
        r %= 10;
        s[j] = (r + '0');
      }
      else {
        s[j] = ('0' + r);
        carry = 0;
      }
    }
    int pos = t.length();
    while (carry > 0) {
      int p = s[pos] - '0';
      int r = carry + p;
      if (r > 9) {
        carry = r / 10;
        r %= 10;
        s[pos] = (r + '0');
      }
      else {
        s[pos] = (r + '0');
        carry = 0;
      }
      pos++;
    }
  }
  cout << sum(s) << endl;
}
