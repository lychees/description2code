#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,j,k,n,t;
	cin>>t;
	while(t--){
	cin>>n;
	int a[50];
	while(1)
	{
		int p;
		p=n;
		int c1=0,c2=0;
		int size;
		i=0;
		while(p!=0)
		{
			int r;
			r=p%2;
			if(r==0)
			{
				a[i++]=0;
				c1++;
				p=p/2;
			}
			else
			{
				a[i++]=1;
				c2++;
				p=p/2;
				
			}
		}
		size=c1+c2;
		int flag=0;
		for(i=0;i<size-1;i++)
		{
			if(a[i]==1&&a[i+1]==1)
			{
				flag=1;
				break;
			}
		}
		if(flag==1)
		{
			n++;
			continue;
		}
		cout<<n<<endl;
		break;
	}
}
	return 0;
}
