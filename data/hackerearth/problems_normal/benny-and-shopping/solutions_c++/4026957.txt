#include<bits/stdc++.h>
using namespace std;
string s,res;
int main(){
int t;
ios_base::sync_with_stdio(0);
cin>>t;
cin.ignore();
while(t--){
getline(cin,s);
res="$";
int k=0;
int di=10000;
for(int i=0;i<s.size();i++){
    if(s[i]=='$')
        di=i;
    if(s[i]>='1'&&s[i]<='9'&&i>di){
        res+=s[i];
        k=1;
    }
    if(s[i]=='0'&&k==1)
        res+=s[i];
    if(isalpha(s[i])&&i>di)
            break;
}


    if(res.size()==1)
        res+='0';
cout<<res<<endl;
res.clear();
cin.clear();
}

return 0;
}
