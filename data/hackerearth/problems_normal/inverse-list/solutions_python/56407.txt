import sys

def main(list, n):
	for i in range(n):
		if int(list[i]) - 1 != list.index(str((i + 1))):
			return 'not inverse'
	return 'inverse'
		
if __name__ == '__main__':
	t = int(sys.stdin.readline())
	for _ in range(t):
		n = int(sys.stdin.readline())
		sys.stdout.write('%s\n' % main(sys.stdin.readline().strip().split(), n))
