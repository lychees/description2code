#include <iostream>
using namespace std;
#define ll long long 
int main()
{
    //cout << "Hello World!" << endl;
    ll t,n,cnt,a[1000000],i,j;
    cin>>t;
    while(t--)
    {
    	cnt=1;
    	cin>>n;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	i=0,j=n-1;
    	while(i<=j)
    	{
    		if(a[i]>a[j])
    		{
    			j--;
    			if(i==j)
    			cnt++;
    		}
    		else{
    		i++;
    		j=n-1;
    		}
    		
    	}
    	cout<<cnt<<endl;
    }
    return 0;
}
