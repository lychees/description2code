#include<bits/stdc++.h>
using namespace std;
int main()
{   int t,n,x,i,j;
	cin>>t;
	while(t--)
	{
		cin>>n>>x;
		int in[n],sum=0;
		for(i=0;i<n;i++)
		{cin>>in[i];
		sum+=in[i];}
		bool a[sum+1][n+1];
		for(i=0;i<=n;i++)
		a[0][i]=1;
		for(i=1;i<=sum;i++)
		a[i][0]=0;
		for(i=1;i<=sum;i++)
		{
			for(j=1;j<=n;j++)
			{
				a[i][j]=a[i][j-1];
				if(i>=in[j-1])
				a[i][j]=a[i][j-1]||a[i-in[j-1]][j-1];
			}
		}
		for(int i=sum/2;i>=0;i--)
		{
			if(a[i][n]==1)
			{
				if((sum-i)<=x)
				cout<<"YES\n";
				else
				cout<<"NO\n";
				break;
			}
		}
		
	}
	return 0;
}
