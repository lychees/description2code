#include<bits/stdc++.h>
using namespace std;
int main()
{
        long int n;
        long int i;
        scanf("%ld",&n);
        char str[n+1];
        scanf("%s",str);
        int a[n];
        for(i=0;i<n;i++)
        {
                a[i]=str[i]-48;
        }
        for(i=1;i<n;i++)
                a[i]+=a[i-1];
        long int q;
        scanf("%ld",&q);
        long int d;
        while(q--)
        {
                int l,r;
                scanf("%d%d",&l,&r);
                if(l==1)
                        d=a[r-1]%9;
                else
                        d=(a[r-1]-a[l-2])%9;
                if(d==0)
                        printf("SAMEER\n");
                else
                        printf("NO\n");
        }
        return 0;

}
