#include<iostream>
#include<algorithm>
#include<stdio.h>
#include<string.h>
#include<math.h>
#include<stdlib.h>
#include<ctype.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	long long int ans=0;
	while(n>0)
	{
		long long int i,k=0;
		char a[100001]={0};
		cin>>a;
		k=strlen(a);
		for(i=0;i<k;i++)
		{
			if(a[i]=='X' && a[i+1]=='X')
			{
				ans=ans+1;
			//	cout<<ans;
			}
			else if(a[i]=='Y' && a[i+1]=='Y')
			{
				ans=ans+1;
			}
		}
		cout<<ans<<endl;
		ans=0;
		n--;

	}
}
