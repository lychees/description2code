#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int main()
{
	int i,j,sum=0,count=0,max=0;
    string s;
    char ch;
    getline(cin, s);
    for(i=0;i<s.size();i++)
    	sum+=s[i];
    sum/=s.size();
    if(sum%2)
    {
    	for(i=0;i<s.size();i++)
    	{
    		count=0;
    		for(j=i;j<s.size();j++)
    		{
    			if(s[i]==s[j])
    			count++;
    			if(max<count)
    			{
    			max=count;
    			ch=s[i];
    			}
    		}
    	}
    	cout<<ch;    	
    }
    else
    {
    for(i=s.size()-1;i>=0;i--)
     	cout<<s[i];
    }
    return 0;
}
