#include <bits/stdc++.h>

using namespace std;

#define ll long long

const ll mod = 1000000007;

ll cnt[1000010], subsets[1000010];
int arr[1000010];

ll modpow(ll a, ll b)
{
	ll x=1ll, y=a;
	while(b){
		if(b%2)
			x=(x*y)%mod;
		y=(y*y)%mod;
		b/=2;
	}
	return x;
}

int main()
{
	int n, i, j, mx=0;
	ll x, y, z;
	scanf("%d", &n);
	for(i=0; i<n; i++){
		scanf("%d", &arr[i]);
		cnt[arr[i]]++;
		mx = max(mx, arr[i]);
	}
	for(i=mx; i>=1; i--){
		x = 0;
		y = 0;
		for(j=i; j<=mx; j+=i){
			x+=cnt[j];
			y = (y+subsets[j])%mod;
		}
		z = (modpow(2ll, x)-1+mod)%mod;
		subsets[i] = (z-y+mod)%mod;
	}
	for(i=0; i<n; i++)
		printf("%lld ", subsets[arr[i]]);
	printf("\n");
	return 0;
}
