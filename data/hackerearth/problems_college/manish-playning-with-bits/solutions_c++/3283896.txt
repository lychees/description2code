#include <bits/stdc++.h>

using namespace std;

#define ll long long int

int n, k;
ll nu, d;
ll dp[1010][1001] ;

void solve(){
    cin >> k >> n ;
    memset(dp,0,sizeof dp) ;
    dp[0][0] = 1 ;
    for(int i = 1; i <= n; i++){
        dp[0][i] += dp[0][i - 1] ;
        for(int j = 1; j < k; j++){
            dp[0][i] += dp[j][i - 1] ;
            dp[j][i] += dp[j - 1][i - 1] ;
        }
    }
    nu = 0;
    for(int i = 0; i < k; i++)
        nu += dp[i][n] ;
    cout << nu << endl ;
}

int main(){
    int t ;
    cin >> t ;
    while(t --)
        solve();

    return 0 ;
}
