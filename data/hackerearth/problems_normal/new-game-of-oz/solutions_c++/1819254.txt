#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	int t,n,a[105],ans,f;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(int i=0;i<n;i++)
    	cin>>a[i];
    	sort(a,a+n);
    	ans=1;
    	f=0;
    	for(int i=1;i<n;i++)
    	{
    		if(a[i]-a[i-1]>1)
    		{
    			ans++;
    			f=0;
    		}
    		else if(a[i]-a[i-1]==0)
    		{
				f=0;
    		}
    		else if(a[i]-a[i-1]==1&&f==0)
    		{
    			f=1;
    		}
    		else if(f==1)
    		{
    			ans++;
    			f=0;
    		}
    		
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
