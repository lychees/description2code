t = int(raw_input())
for i in range(t):
    inj = raw_input().split()
    a = int(inj[0])
    b = int(inj[1])
    ans = 0
    while ((a > 1) and (b > 1)):
        if a > b:
            ans = ans+a/b
            a = a % b
        elif b > a:
            ans = ans+b/a
            b = b % a
    print ans+max(a-1, b-1)
