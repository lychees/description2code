#include <iostream>

using namespace std;

int main() {
 
	char ch[10010];
	int t;
    cin>>t;
    
	while(t--)	{
    	long long int sum =0;
    	int a,i=0;
    	cin>>ch;
         while(isdigit(ch[i])||ch[i]=='_') {
            
	        if(ch[i] =='_');
         	else {
       		 sum+=ch[i]-'0';
        	 //cout<<ch-'0'<<" ";
        	}i++;
	 	}
     	a = sum%9;
     	if(a)	{
       		cout<<9-a<<"\n";
     	}
     	else
       		cout<<"0 or 9"<<"\n";
    }
   	return 0;
}
