#include <iostream>
#include <string>
#include <cmath> 
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	string s;
    	cin >> s;
    	int i,len = s.size();
    	int flag = 0;
    	for(i=1;i<len;i++)
    	{
    		if(abs(s[i]-s[i-1])!=abs(s[len-i-1]-s[len-i]))
    		{
    			cout << "Not Funny" << endl;
    			flag = 1;
    			break;
    		}
    	}
    	if(!flag)
    	{
    		cout << "Funny" << endl;
    	}
    }
    return 0;
}
