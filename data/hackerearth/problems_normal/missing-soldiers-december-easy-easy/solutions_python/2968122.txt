def solve(n,l):
    l.sort()
    c = 0
    o = 0
    for x,g in l:
        if g == 0:
            o += 1
            if o == 1: p = x
        else:
            o -= 1
            if o == 0: c += (x - p) + 1
    return c

n = int(raw_input())
l = []
for i in xrange(n):
    x,y,d = map(int,raw_input().strip().split())
    l.append((x,0))
    l.append((x+d,1))
print solve(n,l)
