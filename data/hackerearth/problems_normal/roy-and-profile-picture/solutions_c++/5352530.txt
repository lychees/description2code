#include <iostream>

int check(int a,int b,int c)
{if(a<c||b<c)
return 2;
else if(a>=c&&b>=c)
{if(a==b)
return 1;
else
return 3;
}	
}

int main()
{int l,b,t,n,no;

cin>>t>>no;

while(no!=0)
{cin>>l>>b;
n=check(l,b,t);
	
	switch(n)
	{
		case 2:cout<<"UPLOAD ANOTHER"<<endl;
			   break;
		case 3:cout<<"CROP IT"<<endl;
			   break;
		case 1:cout<<"ACCEPTED"<<endl;
			   break;
	}

}
