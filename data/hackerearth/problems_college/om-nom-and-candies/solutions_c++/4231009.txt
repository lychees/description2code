int main()
{
    int n;
    cin>>n>>arr[1];
    ans[1]=arr[1];
    for(int i=2;i<=n;i++)
    {
        cin>>arr[i];
        ans[i]=arr[i];
        for(int j=1;j*j<=i;j++)
        {
            if(i%j==0)
            {

                ans[i]=max(ans[i],arr[i]+ans[j]);
                int rem=i/j;
                if(rem<i)
                ans[i]=max(ans[i],arr[i]+ans[rem]);
            }
        }
    }
    for(int i=1;i<=n;i++)
        cout<<ans[i]<<" ";
    return 0;
}
