#include <bits/stdc++.h>
using namespace std;
int gcd(int a, int b)
{
    while(b) b ^= a ^= b ^= a %= b;
    return a;
}
long arr1[100001],arr2[100001];
int main(){
    long n,i,x,y;
    cin>>n;
    int c[n];
    for(i=0;i<n;i++){
    	c[i]=0;
    }
    for(i=0;i<100001;i++){
    	arr1[i]=0;
    	arr2[i]=0;
    }
    long arr3[n][2];
    for(i=0;i<n;i++){
    	cin>>x>>y;
    	arr1[x]++;
    	arr2[y]++;
    	arr3[i][0]=x;
    	arr3[i][1]=y;
    	if(x==y){
    		c[i]=1;
    	}
    }
    for(i=0;i<n;i++){
    	cout<<arr1[arr3[i][1]]+n-1-c[i]<<" "<<2*(n-1)-arr1[arr3[i][1]]-(n-1)+c[i]<<endl;
    }
	return 0;
}
