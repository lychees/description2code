#include <stdio.h>
#include <algorithm>
#include <iostream>
using namespace std;
//Brute Forcing
int main()
{
	int i,n,A[200000],B[200000],t,swap;
    scanf("%d",&n);
    for(i=0;i<n;++i)
    	scanf("%d",&A[i]);
    for(i=0;i<n;++i)
    	B[i]=A[i];
    sort(B,B+n);
    for(i=0;i<n;++i)
    {
    	;
    }
    /*if((sum/n)<(n/2)){
    	printf(":(");return 0;
    }*/
    do{swap=0;
    	for(i=0;i<n-1;++i)
    	{
    		if(A[i]>A[i+1]+1)
    		{
    			t=A[i+1]+1;
    			A[i+1]=A[i]-1;
    			A[i]=t;
    			swap=1;
    			if(A[i+1]<0){
    				printf(":(");return 0;
    			}
    		}
    	}
    }while(swap);
    for(i=0;i<n-1;++i)
    {
    	if(A[i]>A[i+1])
    	{
    		printf(":(");return 0;
    	}
    }
    for(i=0;i<n;++i)
    	printf("%d ",A[i]);
    return 0;
}
