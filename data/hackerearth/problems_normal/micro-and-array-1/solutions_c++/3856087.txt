#include <iostream>
using namespace std;

int main()
{
	int n=0,q=0,x=0,y=0,u=0,v=0;
	cin >> n >> q;
	int a[n],b[q];
    int i=0,j=0,k=0,r=0;
    
    for(i=0;i<n;i++)
    {
    	cin >> a[i];
    }
    
    for(i=0;i<q;i++)
    {
    	cin >> u;
    	
    	if(u==0)
    	{
    		cin >> x >> y;
    		
    		if(x<n)
    		{
    			a[x-1] = y;
    		}
    	}
    	else
    	{
    		cin >> v;
    		
    		for(j=0; j<n;j++)
    		{
    			if(a[j] >= v)
    			{
    				r=1;
    				b[k++] = j+1;
    				break;
    			}
    		}
    		if(r!=1)
    		{
    			b[k++] = -1;
    		}
    	}
    	
    	r = 0;
    }
    
    for(i=0;i<k;i++)
    {
    	cout << b[i] <<endl;
    }
    return 0;
}
