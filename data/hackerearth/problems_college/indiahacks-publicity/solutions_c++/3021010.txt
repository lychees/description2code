#include<bits/stdc++.h>
#define pb push_back
#define mp make_pair
#define f first
#define maxn 101000
#define s second
#define ll long long int
#define inf 1000000014

using namespace std;
std::vector< int>  v[maxn];
int vis[maxn];
void dfs(int s)
{
        vis[s]=1;
        for(int i=0;i<v[s].size();i++)
        {
                if(!vis[v[s][i]])
                dfs(v[s][i]);
        }
}
int main(int argc, char const *argv[])
{
        int t;
        cin>>t;
        int n,m;
        int a,b;
   while(t--)
   {
      cin>>n>>m;
      for(int i=0;i<n;i++)
      {
        vis[i]=0;
        v[i].clear();
      }
      for (int i = 0; i < m; ++i)
      {
          cin>>a>>b;
          v[a].pb(b);
          v[b].pb(a);
      }
      int ctr=0;
      for(int i=0;i<n;i++)
      {
        if(!vis[i])
        {
                ctr++;
                dfs(i);
        }
      }
      cout<<ctr<<"\n";

   }
        return 0;
}
