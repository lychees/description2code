#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
int main()
{
	
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int flag=0;
		string s;
		cin>>s;
		int arr[26];
		memset(arr,0,sizeof(arr));
	
		for(int i=0;i<s.size();i++)
		{
			arr[s[i]-'a']=arr[s[i]-'a']+1;	
			
		}
		for(int j=0;j<26;j++)
		{
			if(arr[j]>=2)
			{
				printf("Yes\n");
				flag=1;
				break;
				
			}
		}
		if(flag==0)
		printf("No\n");
				
	}
}
