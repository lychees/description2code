import operator

t = int(raw_input())
for _ in xrange(t):
    n = input()
    arr = map(int, raw_input().split())
    di = dict()
    for i in arr:
        if i in di:
            di[i] += 1
        else:
            di[i] = 1
    sx = sorted(di.items(), key=operator.itemgetter(1))
    pi = [x for (x, y) in sx if y == sx[-1][1]]
    pi = sorted(pi, reverse=True)
    s = ""
    for i in pi:
        s += str(i)+' '
    print(s)
