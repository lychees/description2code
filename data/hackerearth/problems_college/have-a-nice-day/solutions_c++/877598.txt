#include <iostream>
#include<algorithm>
using namespace std;
int man[1000];
int woman[1000];
int main()
{
	int t,n;
	long long tot;
    //cout << "Hello World!" << endl;
    cin>>t;
    while(t--)
    {
    	tot=0;
    	cin>>n;
    	for(int i=0;i<n;i++)
    		cin>>man[i];
    		for(int i=0;i<n;i++)
    		cin>>woman[i];
    		sort(man,man+n);
    		sort(woman,woman+n);
    		for(int i=0;i<n;i++)
    		 tot+=man[i]*woman[i];
    		 cout<<tot<<endl;
    }
    
    return 0;
}
