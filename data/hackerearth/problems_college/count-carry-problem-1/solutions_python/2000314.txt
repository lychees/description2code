n=input()
for i in range(0,n):
	x,y=(i for i in raw_input().split())
	l=len(x)
	carry=0
	count=0
	for j in range(0,l):
		if(int(x[j])+int(y[j])+carry>=10):
			count+=1
			carry=int((int(x[j])+int(y[j])+carry)/10)
	if count==0:
		print "No carry operation"
	elif count==1:
		print "1 carry operation"
	else:
		print count,"carry operations"
