def readNextWord():
    data = []
    cur = 0
    while True:
        if len(data) == cur:
            st = raw_input()
            data = st.split()
            cur = 0
        yield data[cur]
        cur += 1

def getNextInt(reader):
    return int(float(next(reader)))

def getNthNonFibo(n):
    lfib = 1
    cfib = 1
    curi = 1
    while cfib - curi < n:
        lfib, cfib = cfib, cfib + lfib
        curi += 1
    nonFibs = cfib - curi
    cnfib = cfib + n - 1 - nonFibs
    return cnfib

def main():
    reader = readNextWord()
    t = getNextInt(reader)
    while t > 0:
        n = getNextInt(reader)
        print getNthNonFibo(n)
        t -= 1



if __name__ == "__main__":
    main()
