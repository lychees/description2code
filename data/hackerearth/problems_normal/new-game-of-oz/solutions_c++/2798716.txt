#include <bits/stdc++.h>
using namespace std;
int func(vector<int>ar){
	int ans=0;
	sort(ar.begin(),ar.end());
	int sz=0;
	for(int i=0;i<ar.size();i++){
		int ff=0;
		if(i==0 || ar[i-1]+1==ar[i]) sz++;
		else{
			ff=sz;
			sz=1;
		}
		ans+=(ff+1)/2;
		if(i==ar.size()-1) ans+=(sz+1)/2;
		
	}
	return ans;
}
int main()
{
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		vector<int> ar(n,0);
		for(int i=0;i<n;i++){
			cin>>ar[i];
		}
		cout<<func(ar)<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
