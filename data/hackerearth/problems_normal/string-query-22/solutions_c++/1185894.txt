#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int main(){
string s;
vector<int> pos[26];
static int flag[200010];
int m;
cin>>s>>m;
int l=s.length(),p;
for(int i=0;i<l;i++){
    pos[s[i]-'a'].push_back(i);
}
char c;
for(int i=0;i<m;i++){
    cin>>p>>c;
    flag[pos[c-'a'][p-1]]=1;

    pos[c-'a'].erase(pos[c-'a'].begin()+p-1);
   // cout<<"y"<<endl;
}
for(int i=0;i<l;i++){
    if(!flag[i])cout<<s[i];
}
cout<<endl;
return 0;
}
