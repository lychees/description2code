'''

name = raw_input('What is your name?\n')

n = input()
for i in range(n):
    x = raw_input()
    x = x.split()
    num = int(x[0])
    k = int(x[1])
    l = str(k+1)
    for i in range(num-1):
        l = l + " " + str((i+2) * (k+1))
    print l
