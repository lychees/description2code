#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,a,b,count;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b;
    	count=0;
    	a=a^b;
    	while(a>0)
    	{
    		count++;
    		a&=(a-1);
    	}
    	if(count%2==0)
    	cout<<"YES"<<endl;
    	else
    	cout<<"NO"<<endl;
    }
}
