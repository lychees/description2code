#include <bits/stdc++.h>
#define ll long long int

using namespace std;

//int brr[10000007];
int dp[70005][20];

ll func(ll arr[105], ll n, ll index, ll num, ll pichla);

int main()
{
	ll t;
	
	cin >> t;
	
	while (t--) {
		
		memset(dp, -1, sizeof(dp));
		ll n;
		cin >> n;
		
		ll arr[105];
		
		for (ll i = 0; i < n; i++) {
			cin >> arr[i];
		}
		
		cout << func(arr, n, 0, 0, 0) << endl;
	}
	
	return 0;
}

ll func(ll arr[105], ll n, ll index, ll num, ll pichla)
{
	ll temp = 0;
	
	if (index == n) {
		return 0;
	}
	
	
	if (dp[num][pichla] != -1) {
		return dp[num][pichla];
	}
	
	if (index != 0) {
		
		for (ll i = 0; i < n; i++) {
			
			ll ai = 1 << i;
			ai = num & ai;
			
			if (ai == 0) {
				
				ll ct = __builtin_popcount(num);
				ct += 1;
				
				ll tempo1 = num;
				ll tempo2 = 1 << i;
				
				tempo1 = tempo1 | tempo2;
				
				//cout << ct << endl;
				
				ll an = abs(arr[pichla] - arr[i]) * ct + func(arr, n, index + 1, tempo1, i);
				//cout << an << endl;
				
				temp = max(temp, an);
				
			}
			
		}
		
		
		
	}
	
	else {
		
		for (ll i = 0; i < n; i++) {
			ll tempo1 = num;
			ll tempo2 = 1 << i;
			
			tempo1 = tempo1 | tempo2;
			//cout << "aya " << tempo1 << endl;
			temp = max(temp, func(arr, n, index + 1, tempo1, i));	
		}
		
	}
	
	dp[num][pichla] = temp;
	return dp[num][pichla];
}
