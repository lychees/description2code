#include <iostream>
using namespace std;

int main()
{
    int n,ptr=0,i,f=0;
    cin>>n;
    long long int a[n];
    for(i=0;i<n;i++)
        cin>>a[i];

        for(i=ptr;i<n-1;i++)
        {
            //while(a[i]!=0&&a[i+1]!=0)
            {
                if(a[i]>a[i+1])
                {
                    a[i]=a[i]-a[i+1];
                    a[i+1]=0;
                }
                else{
                    a[i+1]=a[i+1]-a[i];
                    a[i]=0;
                }

            }
            if(a[i]==0)
            {
                ptr++;
            }
            else if(a[i+1]==0)
            {   f=1;
                break;
            }
            else if(a[i]==0&&a[i+1]==0)
            {   f=0;
                ptr+=2;
            }
        }
    for(i=0;i<n;i++)
    {
        if(a[i]!=0)
        {
            f=1;
            break;
        }
        else f=0;
    }
    if(f==1)    {
        cout<<"NO";

    }
    else cout<<"YES";
    return 0;
}
