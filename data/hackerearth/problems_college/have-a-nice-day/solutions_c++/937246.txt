#include <iostream>
#include <stdio.h>
#include <vector>
#include <algorithm>
using namespace std;
#define Max 1001
vector <int> male(Max);
vector <int> female(Max);
//bool desc (int i,int j) { return (i>j); }
int main()
{
    //freopen("DATA.dat", "r", stdin);
    int t;
    cin>>t;
    while(t--){
    	int n, i, c=0;
    	cin>>n;
    	for(i=0; i<n; i++)
    		cin>>male[i];
    	for(i=0; i<n; i++)
    		cin>>female[i];
    	sort(male.begin(), male.begin()+n);
    	sort(female.begin(), female.begin()+n);
    	for(i=0; i<n; i++){
    		c+=male[i]*female[i];
    	}
    	cout<<c<<endl;
    }
    return 0;
}
