#include <bits/stdc++.h>
using namespace std;

const long int N=1e7+3;
vector<long long int>a(N);

void pre(){
	long int i,j;
	long long int sum=0;
	for(i=2;i<N-1;i++){
		if(a[i]==0){
			a[i]=sum+i;
			sum=a[i];
			for(j=2*i;j<=N-1;j+=i){
				a[j]=1;
			}
		}
	}
}

int main() {
	long int t;
	scanf("%ld",&t);
	pre();
	while(t--){
		long long int aa,bb;
		scanf("%lld%lld",&aa,&bb);
		long long int ans=0;
		for(long int i=bb;i>=2;i--){
			if(a[i]!=1){
				ans+=a[i];
				break;
			}
		}
		for(long int i=aa-1;i>=2;i--){
			if(a[i]!=1){
				ans-=a[i];
				break;
			}
		}
		printf("%lld\n",ans);
	}
	return 0;
}
