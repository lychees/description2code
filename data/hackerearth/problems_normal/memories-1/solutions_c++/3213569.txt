using namespace std;
typedef long long int LL;
int nt[22][22] , u , v;

void init()
{
	memset(nt , 0 , sizeof(nt));
}


int main()
{
int  test , V , E , flag;
scanf("%d",&test);
LL ans = 0;
while(test--)
{
	init();
	ans = 0;
	scanf("%d %d",&V,&E);
	for(int i=1;i<=E;i++)
	{
		scanf("%d %d",&u,&v);
		nt[u][v] = 1;
		nt[v][u] = 1;
	}

	for(int i=1;i<pow(2 , V);i++)
	{
		flag = 0;
		for(int j=1;j<=V;j++)
		{

			for(int k=j+1;k<=V;k++)
			{
				if((i&(1<<(j-1)))!=0 && (i&(1<<(k-1)))!=0 && nt[k][j]!=0){
					flag = 1;
				}
			}
		}
		if(flag==0)
			ans++;
	}
	cout << ans << endl;
}

return 0;
}
