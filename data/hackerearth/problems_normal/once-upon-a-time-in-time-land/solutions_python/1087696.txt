
_ = input()

for __ in range (_):
	n,k = map(int,raw_input().split())
	arr = map(int, raw_input().split())	
	ans = []	
	ans.append(max(0,arr[0]))	
	for i in range (1,k+1):
		ans.append(max(ans[i-1],arr[i]))	
	for i in range (k+1,n):
		ans.append(max(ans[i-1],ans[i-k-1]+arr[i]))		
	print ans[n-1]
