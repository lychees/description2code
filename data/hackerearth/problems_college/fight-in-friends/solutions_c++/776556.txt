#include<bits/stdc++.h>
using namespace std;
#define S(n) scanf("%d",&n)
int graph[1002][1002];
int v[100002];

int ans(int n,int a,int b)
{
    queue<pair<int,int> >q;
    q.push(make_pair(a,0));
    pair<int,int> p;

    v[a] = 1;

    while(!q.empty())
    {   p = q.front();
        q.pop();

        for(int i = 1; i<=n ; i++)
        {
            if(  graph[p.first][i] == 0 && v[i] == 0 )
            {
                v[i] = 1;
                if(i == b)
                    return p.second+1;
                q.push(make_pair(i,p.second+1));
            }
        }
    }
    return -1;
}


int main()
{
    int n,m,a,b;

    cin>>n>>m>>a>>b;
    int x,y;
    for(int i = 0 ; i<m ; i++)
    {

        cin>>x>>y;
        graph[x][y] = 1;
        graph[y][x] = 1;
    }
    cout<<ans(n,a,b)<<endl;

    return 0;
}
