N = input()

ar = sorted(map(int, raw_input().split(" ")))


ne = []
ne1 = []
L = 0

H = len(ar) - 1


def compute(ar):
    su = 0
    for i in range(len(ar)-1):
        su += abs(ar[i+1] - ar[i])
    return su


while (L <= H):
    if L == H:
        ne.append(ar[L])
        ne1.append(ar[H])
        break
    ne.append(ar[L])
    ne1.append(ar[H])

    ne.append(ar[H])
    ne1.append(ar[L])
    L += 1
    H -= 1
print max(compute(ne), compute(ne1))
