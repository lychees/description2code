#include <iostream>
using namespace std;
const int INT_MAX=1000000000;
int main()
{
    //cout << "Hello World!" << endl;
    long long int N,T,X,credit,debit,miniT,miniX;
    cin>>N;
    miniT=miniX=INT_MAX;
    while(N--){
    	cin>>T>>X;
    	if(T<=X){
    		credit=credit+X-T;
    		miniT=min(T,miniT);
    	}
    	else{
    		miniX=min(miniX,X);
    		debit=debit+T-X;
    	}
    }
    if(miniX==INT_MAX) miniX=0;
    if(miniT==INT_MAX) miniT=0;
    if(credit+miniT>debit+miniX) cout<<miniT<<endl;
    else cout<<debit+miniX-credit<<endl;
    return 0;
}
