#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;
int main()
{
	int n,k,i;
	cin>>n>>k;
	int ct=0,a[n+1];
	for( i = 0; i < n; ++i){
   		cin>>a[i];
   	}
   	   for( i = 0; i < n; ++i)
   	   {
   	   	     if(i+k < n)
   	   	       ct=ct+ ( (int)fabs(a[i]+a[i+k])%k==0 || (int)fabs(a[i]-a[i+k])%k==0 );
   	   }
   	   cout<<ct<<endl;
}
