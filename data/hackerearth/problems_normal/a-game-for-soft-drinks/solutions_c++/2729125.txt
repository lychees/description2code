#include<bits/stdc++.h>
using namespace std;
bool dp[101][101][101],visited[101][101][101];
bool win(int n1,int n2,int n3)
{
   bool flag = false;
   if(n1<0||n2<0||n3<0)
    return false;
   if(n1==0&&n2==0&&n3==0)
        return false;
   if(visited[n1][n2][n3])
            return dp[n1][n2][n3];
   // possibilities
   if(n1>0&&n2>0&&n3>0)
   if(!win(n1-1,n2-1,n3-1))
   flag = true;

   if(n1>0)
   if(!win(n1-1,n2,n3))
   flag = true;

   if(n2>0)
   if(!win(n1,n2-1,n3))
   flag = true;

   if(n3>0)
   if(!win(n1,n2,n3-1))
   flag = true;

   if(n1>0&&n2>0)
   if(!win(n1-1,n2-1,n3))
   flag = true;

   if(n2>0&&n3>0)
   if(!win(n1,n2-1,n3-1))
   flag = true;

   if(n1>0&&n3>0)
   if(!win(n1-1,n2,n3-1))
   flag = true;

   visited[n1][n2][n3] = true;
   dp[n1][n2][n3] = flag;
   return flag;
}

int main()
{
    int i,j,k;
    for(i=0;i<=100;i++)
        for(j=0;j<=100;j++)
           for(k=0;k<=100;k++)
           visited[i][j][k] = false;
    bool f = win(100,100,100);
    int q;
    cin>>q;
    int n1,n2,n3;
    while(q--)
    {
        cin>>n1>>n2>>n3;
        if(win(n1,n2,n3))
            cout<<"Ashima\n";
        else
            cout<<"Aishwarya\n";
    }
    return 0;
}
