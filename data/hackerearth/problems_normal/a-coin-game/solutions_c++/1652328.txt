

    #include <bits/stdc++.h>
    using namespace std;
    int l[1000006];
    int pro()
    {
    l[0]=0;
    int d=1,i;
    for(i=1;i<=1000000;i++)
    {
    if(l[i]!=0)
    continue;
    l[i]=i+d;
    if(i+d<=1000000)
    l[i+d]=i;
    d++;
    }
    }
    int main()
    {
    pro();
    int t;
    cin>>t;
    while(t--)
    {
    int a,b;
    cin>>a>>b;
    if(a>b)
    {
    int tmp=a;
    a=b;
    b=tmp;
    }
    if(l[a]==b || l[b]==a)
    cout<<"Don't Play";
    else
    cout<<"Play";
    cout<<endl;
    }
    return 0;
    }
