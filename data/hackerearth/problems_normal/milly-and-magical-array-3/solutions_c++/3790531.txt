#include <bits/stdc++.h>
using namespace std;

int main()
{
		long long int t;
	   cin>>t;
	   while(t--)
	   {
		   	long long int n,x;
		   	cin>>n>>x;
	   		long long int a[n];
	   		for(long long int i=0;i<n;i++)
	   		{
	   			cin>>a[i];
	   		}
	   		long long int count=0;
	   		for(long long int i=1;i<n;i++)
	   		{
	   			
	   			if(a[i-1]>=a[i])
	   			{
	   				long long int temp = a[i-1]+1;
	   				long long int f =temp-a[i];
	   				long long int val;
	   				if(f%x!=0)
	   				{
	   					val= (temp-a[i])/x;
	   					val+=1;
	   				}
	   				else
	   				{
	   					val= (temp-a[i])/x;
	   				}
		   			count=count+val;
		   			a[i]=a[i]+val*x;
	   			}
	   			
	   		}
	   		cout<<count<<"\n";
	   }
}
