#include<iostream>
using namespace std;
long conv(long a1,int x)
{	int r,t=1;
	long rev=0;
	if(a1%10==0)
	t=0;
	if(x==6)
	{	
		while(a1)
		{	r=a1%10;
			if(r==6)
			  r=5;
			rev=rev*10+r;
			a1/=10;
		}
		a1=0;
		while(rev)
		{	r=rev%10;
			a1=a1*10+r;
			rev/=10;
		}
		if(t==0)
		a1*=10;
		return a1;
	}
	else
	{	while(a1)
		{	r=a1%10;
			if(r==5)
			  r=6;
			rev=rev*10+r;
			a1/=10;
		}
		a1=0;
		while(rev)
		{	r=rev%10;
			a1=a1*10+r;
			rev/=10;
		}
		if(t==0)
		a1*=10;
		return a1;
	}
}
int main()
{	long a,b,a1,b1,max,min;
	cin>>a>>b;
	a1=a;
	b1=b;
	a1=conv(a1,6);
	b1=conv(b1,6);
	min=a1+b1;
	a1=a;
	b1=b;
	a1=conv(a1,5);
	b1=conv(b1,5);
	max=a1+b1;
	cout<<min<<" "<<max;
	return 0;
}
