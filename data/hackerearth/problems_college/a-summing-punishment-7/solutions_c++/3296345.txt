#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int n,c,i; vector<long long int>arr; cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>c; arr.push_back(c);
	}
	long long int t; cin>>t;
	while(t--)
	{	int sum=0;
		long long int a,b; cin>>a>>b;
		for(i=a-1;i<b;i++)
		{
			sum+=arr[i];
		}
		cout<<sum<<endl;
	}
}
