#include <bits/stdc++.h>
using namespace std;
const int MAX = 1000000;
int A[MAX];
int main()
{
	int n;
	cin>>n;
	assert(n>=1 && n<=1000000);
	for (int i = 0; i < n; ++i)
		{cin>>A[i];
	   // assert(A[i]>=1 && A[i]<=1000000000);
		}
	sort(A, A+n);
	long long int ans = 0, L = 1, P = 0;
	while(P < n)
	{
		if(L*L < A[P])
			ans+=L*L;
		else
		{
			ans+=(L*L)-A[P];
			P++;
		}
		L++;
	}
	cout<<ans<<"\n";
	return 0;
}
