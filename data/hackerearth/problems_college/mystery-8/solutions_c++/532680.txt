#include<iostream>
#include<string>
#include<cstring>
#include<cstdio>
using namespace std;
int main()
{
    int test_cases;
    cin>>test_cases;
    string s="";
    int v,i;
    getchar();
    while(test_cases>0)
    {
        char str[101];
        gets(str);
        // your code here
        s="";
        for(i=0;i<strlen(str);i++)
        {
        	if((str[i]>='a' && str[i]<='z') || str[i]=='*')s+=str[i];
        }
        
        for(i=0;i<s.length();i++)
        {
        	if(s[i]=='*')
        	{
        		v = i;
        		break;
        	}
        }
        cout<<s[s.length()-v-1]<<"\n";
        test_cases--;
    }
    return 0;
}
