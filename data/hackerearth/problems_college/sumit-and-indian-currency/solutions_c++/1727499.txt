#include <bits/stdc++.h>
using namespace std;

int main()
{
    int ite;
    cin>>ite;
    while(ite--){
    	string s;
    	cin>>s;
    	string ans="";
		int j=s.length()-1, k=3;
		while(k-- && j>=0){
			ans+=s[j];
			j--;
		}
		while(j>=0){
			ans+=',';
			int fl=0;
			k=2;
			while(k-- && j>=0){
			fl=1;
			ans+=s[j];
			j--;
		}
			if(!fl) break;
		}
		reverse(ans.begin(), ans.end());
		cout<<ans<<endl;
    }
    return 0;
}
