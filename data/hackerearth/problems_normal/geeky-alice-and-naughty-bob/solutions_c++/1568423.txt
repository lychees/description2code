#include<iostream>
using namespace std;
int main()
{
	long long a,b,i,rsum=0,t;
	int c[6]={0,1,2,6,6,3};
	cin>>t;
	while(t--)
	{
		cin>>a>>b;
		if(a>=6&&b>=6)
		rsum=(b-a+1)*9;
		else if(a<6&&b>=6)
		{
			for(i=a;i<6;i++)
			rsum+=c[i];
			rsum+=(b-6+1)*9;
		}
		else if(a<6&&b<6)
		for(i=a;i<=b;i++)
		rsum+=c[i];
		cout<<rsum<<"\n";
		rsum=0;
	}
	return 0;
}
