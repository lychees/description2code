t = int(raw_input())
while t > 0:
    n = int(raw_input())
    mst = []
    sum = 0
    for x in raw_input().split():
        sum += int(x)
        mst.append(int(x))
    m = int(raw_input())
    if m == 0:
        print 1
    elif m == 0 or m % sum == 0:
        i = n-1
        while mst[i] == 0:
            i -= 1
        print i+1
    else:
        m = m % sum
        i = 0
        while m > 0:
            m -= mst[i]
            i += 1
        print i
    t -= 1
