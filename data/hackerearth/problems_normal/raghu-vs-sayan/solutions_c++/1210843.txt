#include <iostream>
#include <algorithm>

using namespace std;
typedef long long ll;
int dishes[10000];

int main()
{
    int N;
    cin >> N;
    ll A;
    ll B;
    int numDish;
    for(int i = 0; i < N; i++)
    {
    	cin >> A >> B >> numDish;
    	for(int k = 0; k < numDish; k++)
    	{
    		cin >> dishes[k];
    	}
    	sort(dishes, dishes+numDish);
    	int Raghu = 0;
    	int Sayan = 0;
    	for(int k = 0; k < numDish; k++)
    	{
    		if(A >= dishes[k])
    		{
    			A -= dishes[k];
    			Raghu++;
    		}
    		else
    		{
    			break;
    		}
    	}
    	for(int k = 0; k < numDish; k++)
    	{
    		if(B >= dishes[k])
    		{
    			B -= dishes[k];
    			Sayan++;
    		}
    		else
    		{
    			break;
    		}
    	}
    	
    	if(Raghu > Sayan)
    	{
    		cout << "Raghu Won" << "\n";
    	}
    	else if (Sayan > Raghu)
    	{
    		cout << "Sayan Won" << "\n";
    	}
    	else
    	{
    		cout << "Tie" << "\n";
    	}
    }
    
}
