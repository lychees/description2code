#include <bits/stdc++.h>
#define ll long long
char g[100010],p[100010];
int hg[26],hp[26];
using namespace std;
int main()
{
ios_base::sync_with_stdio(false);
cin.tie(NULL);
int test;
cin>>test;
while(test--)
{
	cin>>g>>p;
	int lg=strlen(g),lp=strlen(p);
	int flag=0;
	int i,j,t;
	if(lp<=lg)
	{
		memset(hg,0,sizeof(hg));
		memset(hp,0,sizeof(hp));
		//~ cout<<g<<" "<<p<<endl;
		for(i=0;i<lp;i++)
		{
			hg[g[i]-'a']++;
			hp[p[i]-'a']++;
		}
		flag=1;
		for(t=0;t<26;t++)
		{
			if(hg[t]!=hp[t])
			{
				flag=0;
				break;
			}
		}
		if(!flag)
		{
			j=0;
			for(i=lp;i<lg;i++)
			{
				hg[g[j]-'a']--;
				hg[g[i]-'a']++;
				j++;
				flag=1;
				for(t=0;t<26;t++)
				{
					if(hg[t]!=hp[t])
					{
						flag=0;
						break;
					}
				}
				if(flag)
				break;
			}
		}
	}
	if(flag)
	cout<<"YES\n";
	else
	cout<<"NO\n";
}
return 0;
}
