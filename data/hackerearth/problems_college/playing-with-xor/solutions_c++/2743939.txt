#include<bits/stdc++.h>
#define ll long long
using namespace std;


bool isprime[1025];

void sieve(ll limit)
{
    int  k=sqrt(limit)+1;
    isprime[1]=false;
    for(ll i=2;i<limit;i++)
        isprime[i]=true;
    for(ll i=2;i<k;i++)
    {

      if(isprime[i])
      {


          for(int j=i+i;j<limit;j+=i)
            isprime[j]=false;

      }


    }
}

int main()
{
    sieve(1025);

    int t,n,nos;
    cin>>t;
    while(t--)
    {
        int arr[1025];
        for(int i=1;i<1025;i++)
            arr[i]=0;

        cin>>n;
        for(int i=0;i<n;i++)
        {
            cin>>nos;
            arr[nos]=1;
        }
        for(int i=1;i<1024;i++)
        {
            for(int j=1;j<1024;j++)
            {
                if(arr[i]==1 && arr[j]==1)


                arr[i^j]=1;
            }

        }

        int cnt=0;
        for(int i=1;i<1024;i++)
        {
            if(isprime[i]&&arr[i]==1)
                cnt++;

        }
        cout<<cnt<<"\n";
    }
}
