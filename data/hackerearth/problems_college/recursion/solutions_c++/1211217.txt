#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
void p(int a)
{
    cout<<a;
}
int main()
{
    int a;
    cin>>a;
    vector<int>v1;
    while(a--)
    {
        int b;
        cin>>b;
        for(int i=1;i<=b;i++)
            v1.push_back(i);
        do{
            for_each(v1.begin(),v1.end(),p);
            cout<<" ";
        }
        while(next_permutation(v1.begin(),v1.end()));
        cout<<endl;
        v1.clear();
    }
}
