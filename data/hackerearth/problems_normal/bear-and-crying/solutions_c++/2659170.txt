#include <iostream>
#define MOD 1000000007
using namespace std;

int main()
{
    string s;
    cin>>s;
    int i,ans=0,temp=0,count=0;
    for(i=0;i<s.size();i++)
    	if(s[i]==';')
    	{
    	ans=(ans+temp)%MOD;	
    	count++;
    	}
    	else
    	temp=(2*temp+count)%MOD;
    cout<<ans<<endl;
    return 0;
}
