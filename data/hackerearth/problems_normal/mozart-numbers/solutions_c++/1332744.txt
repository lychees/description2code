#include <bits/stdc++.h>

using namespace std;

int A, B;
int special[]={7, 13};
int val[]={8, 144};

void solve()
{
    scanf("%d%d", &A, &B);
    int ans=0;
    for(int i=0; i<2; i++)
        if(A<=special[i] && special[i]<=B)
            ans+=val[i];
    printf("%d\n", ans);
}

int main()
{
    int T;
    scanf("%d", &T);
    while(T--)
        solve();
    return 0;
}
