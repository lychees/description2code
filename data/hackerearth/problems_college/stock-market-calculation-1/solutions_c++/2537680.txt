#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int n,profit=0;
		long long int arr[100000];
		cin>>n;
		for(int i=0;i<n;i++)
			cin>>arr[i];
		long long int max;
		for(int i=0;i<n;i++)
		{
			max=arr[i];
			//profit=0;
			for(int j=i+1;j<n;j++)
			{
				if(arr[j]>max)
					max=arr[j];
			}
			if(max!=arr[i])
				profit=profit+(max-arr[i]);
		}
		cout<<profit<<endl;
		
	}

}
