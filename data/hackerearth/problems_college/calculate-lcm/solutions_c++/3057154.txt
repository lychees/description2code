ll gcd(ll a, ll b) {
	if(b == 0)
		return a;
	return gcd(b, a%b);
}

ll power(ll a, ll b) {
	ll res = 1;
	while(b) {
		if(b & 1) {
			res = (res * a) % MOD;
		}
		a = (a*a) % MOD;
		b >>= 1;
	}
	return res;
}

ll expo(ll a, ll b) {
	ll res = 1;
	while(b) {
		if(b & 1) {
			res = (res * a);
		}
		a = (a*a);
		b >>= 1;
	}
	return res;
}
