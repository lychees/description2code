#include<bits/stdc++.h>
using namespace std;
int a[1000000];
int main()
{
    int T;
    long int N,i,j,k;
    a[1]=1;
    for(i=2;i<=1000;i++)
    {
        for(j=2;pow(i,j)<=1000000;j++)
        {
            k=pow(i,j);
            a[k]=1;
        }
    }
    cin>>T;
    while(T--)
    {
        cin>>N;
        int flag=0;
        for(i=1;i<=N/2;i++)
        {
            if(a[i]==1 && a[N-i]==1)
            {
                flag=1;
                break;
            }
        }
        if(flag==1)
            cout<<"Yes\n";
        else
            cout<<"No\n";

    }
}
