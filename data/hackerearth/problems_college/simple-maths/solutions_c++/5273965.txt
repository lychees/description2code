#include <bits/stdc++.h>
using namespace std;
int gcd(int a,int b){
	
	if(a==0) return b;
	return gcd(b%a,a);
}
int main()
{
    int t; cin>>t;
    while(t--){
    	long long int n,a,b,ans=0; cin>>n>>a>>b;
    	if(a==0||b==0) cout<<"-1"<<endl;
    	else{
    		//a=min(a,b);
    		//b=max(a,b);
    		//int lcm=a*b/gcd(a,b);
    		//cout<<lcm<<endl;
	    	ans=n/a+n/b-n/ ((a*b)/gcd(a,b));
	    	cout<<ans<<endl;
    	}
    }
    return 0;
}
