#include <iostream>
#include<cmath>
using namespace std;
 
//typedef long long int ll;
 
int gcd(int x, int y)
{
while(x!=y)
{
	if(x>y)
	return gcd(x-y,y);
	else return gcd(y,y-x);
}
return x;
}
 
int main()
{
	int t,a,b,c,count,i,j,k,n;
	//ll n;
	cin>>t;
	
	while(t--)
	{
		cin>>n;
		count=0;
		for(i=2;i<=(sqrt(n-1)+1); i++)
		{
			for(j=1;j<i;j++)
			{
				if((gcd(i,j)==1) && ((i-j)%2) && ((i*i + j*j)<=n))
				{
					k=0;
					c=i*i+j*j;
					while((++k)*c<=n)
					count++;
				}
			}
		}
			
		
		cout<<count<<endl;
	}
	return 0;
}
