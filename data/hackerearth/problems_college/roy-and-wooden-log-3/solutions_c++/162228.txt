#include <iostream>
using namespace std;

int main()
{
 long t,n,w,m,x;
 cin>>t;
 while(t)
 {
  cin>>n;
  if(n==0)
   cout<<"0"<<" "<<"0\n";
  else if(n==1)
   cout<<"1"<<" "<<"0\n";
  else if(n==2)
   cout<<"2"<<" "<<"0\n";
  else
  {
   if(n%2==0)
   {
  	w=(n/2)+1;
  	m=(n/2)-1;
   }
   else
   {
   	w=(n/2)+1;
   	m=n/2;
   }
   cout<<w<<" "<<m<<"\n";
  }
  t--;	
 }
 return 0;
}
