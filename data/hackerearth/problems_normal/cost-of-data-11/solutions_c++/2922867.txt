#include<bits/stdc++.h>
#define mod 10000000
#define mx 1999999
#define prin(n) printf("%lld\n",n)
#define scal(n) scanf("%lld",&n)
typedef long long LongL;

struct que
{
    LongL isEnd;
    struct que *children[26];
    que()
    {
        isEnd=0;
        for(LongL i=0;i<26;i++)
            children[i]=NULL;
    }
};
 
LongL count1;
 
void add(struct que *node,char *str)
{
    if(str[0]=='\0'){
        return ;
    }
    else
    {
        int ch=str[0]-'a';
        if(node->children[ch]==NULL){
            count1++;
            node->children[ch]=new que;
        }
        add(node->children[ch],str+1);
    }
}
 
char p[55];
 
int main()
{
    LongL n,i,j,k,m;
    count1=0;
    struct que *root=NULL;
    root=new que;
    scal(m);
    LongL flag=0;
    while(m--)
    {
        scanf("%s",p);
        add(root,p);
    }
    prin(count1+1);
    return 0;
}
