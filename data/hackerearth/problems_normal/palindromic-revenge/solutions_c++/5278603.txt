#include <iostream>
#include<string>
#include<algorithm>
using namespace std;

int main()
{
    int t,l,c,i,j;
    string str,str1;
    cin>>t;
    while(t--)
    {
    	c=0;
    	cin>>str;
    	str1=str;
    	l=str.length();
    std::reverse(str.begin(), str.end());
   // cout<<str<<" "<<str1<<endl;
    if(str==str1)
    {
    	cout<<"0"<<endl;
    }
    else
    {
    	i=0;
    	j=l-1;
    	while(i<j)
    	{
    	 int e=(int)str[i];
    	// cout<<e<<endl;
    	 int f=(int)str[j];
    	 if(e>f)
    	 {
    	 	while(e>f)
    	 	{
    	 	e--;
    	 	c++;
    	 	}
    	 }
    	 else
    	 {
    	 	while(f>e)
    	 	{
    	 		f--;
    	 		c++;
    	 	}
    	 }
    	 i++;
    	 j--;
    	}
    	  cout<<c<<endl;
    }
   //cout<<c<<endl;
    }
    return 0;
}
