#include <iostream>

using namespace std;

int main()
{
        int test_cases;
        cin>>test_cases;
        while(test_cases > 0)
        {
                int X1,X2,Y1,Y2, s = 0;
                cin>>X1>>Y1>>X2>>Y2;
                
                if (X1 == Y2) s++;
                if (X2 == Y1) s++;
                
                cout << s << std::endl;
                
                test_cases --;
        }
        return 0;
}
