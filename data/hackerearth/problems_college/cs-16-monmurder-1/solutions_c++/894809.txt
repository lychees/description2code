#include <bits/stdc++.h>
using namespace std;
#define MAX 1000006
vector<int>v;


void factor(int n)
{
    v.clear();
    int i;
    for(i = 2; i <= (int)sqrt(n); i++)
    {
        int cou = 0;
        while(n % i == 0)
        {
           n /= i;
           cou++;
        }
        if(cou != 0) {
            v.push_back(cou);
        }
    }
    if (n > 1) {
        v.push_back(1);
    }
    int ans = v[0];
    for(int i = 1; i < v.size(); i++) {
        ans ^= v[i];
    }
    (ans == 0) ? puts("Pappu Bhujia") : puts("Printf{}");
}

int main()
{
    int t;
    cin >> t;
    while(t--) {
        int n;
        cin >> n;
        if(n == 1) {
            puts("Pappu Bhujia");
            continue;
        }
        factor(n);
    }
    return 0;
}
