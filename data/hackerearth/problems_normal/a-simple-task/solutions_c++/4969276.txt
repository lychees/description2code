#include <bits/stdc++.h>

using namespace std;

void prime_factor(int g)
{
	int i;
	for (i = 2; i <= g/2; i++) {
		if (g % i == 0) {
			printf("%d ",i);
		}
	}
	printf("%d",g);
}

int main()
{
	int m,i;
	cin >> m;
	int g = 0;
	int a,b;
	cin >> a;
	for (i = 1; i < m; i++) {
		cin >> b;
		g = __gcd(g,abs(b-a));
		a = b;
	}
	prime_factor(g);
}
