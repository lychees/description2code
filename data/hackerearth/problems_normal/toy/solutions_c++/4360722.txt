#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int i,x,n,a[1000000]={},maxi=INT_MIN,c=0;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
    	scanf("%d",&x);
    	a[x]=1;
    	maxi=max(maxi,x);
    }
    for(i=0;i<=maxi;i++)
    {
    	if(a[i]==1)
    	{
    		c++;
    		for(int j=i;j<=i+4;j++)
    		{
    			if(a[j]==1)
    				a[j]=0;
    		}
    	}
    }
    cout<<c<<endl;
    
    return 0;
}
