#include <bits/stdc++.h>
using namespace std;

int main()
{
   int t=0;
   scanf("%d",&t);
   t++;
   while(t--)
   {
   	string str,s;
   	vector<string> v;
   	int i=0;
   	getline(cin,str);
   	stringstream ss  (str);
   	while(ss >> s)
   	v.push_back(s);
   	
   	reverse(v.begin(),v.end());
   	
   	for(i=0;i<v.size();i++)
   		cout<<v[i]<<" ";
   		cout<<endl;
   }
    return 0;
}
