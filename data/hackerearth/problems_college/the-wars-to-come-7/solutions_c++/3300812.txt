#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n, i, s;

    cin >> n;

    int a[n];

    for (i = 0; i < n; i++)
        cin >> a[i];

    cin >> s;

    sort(a, a + n);

    int sum = 0;
    int c = 0;
    for ( i = n - 1; i >= 0 && sum < s; i--) {
        sum += a[i];
        c++;
    }

    cout << c << endl;
}
