#include <iostream>
using namespace std;

int main()
{
	long long t, m, zeros, ones, tempOnes, tempZeros;
	cin>>t;
	while(t--) {
		zeros = 0;
		ones = 1;
		tempZeros = 0;
		tempOnes = 1;
		cin>>m;
		while(m--) {
			tempOnes -= ones;
			tempZeros += ones;
			tempOnes += zeros;
			ones = tempOnes;
			zeros = tempZeros;
		}
		cout<<ones<<" "<<zeros<<endl;
	}
    return 0;
}
