#include <iostream>
#include<bits/stdc++.h>
#define ll long long
using namespace std;

ll euclid(ll a,ll b){
	ll c=1;
	while(c!=0){
		c=a%b;
		a=b;
		b=c;
	}
	return a;
}
int main()
{
    ll t,a,b,c,l;
    cin>>t;
    while(t--){
    	ll count=0;
    	cin>>a>>b>>c;
    	l=euclid(euclid(a,b),c);
    	count=(((a/l*b/l)%1000000007)*c/l%1000000007)%1000000007;
    	cout<<l<<" "<<count<<endl;
    }
    return 0;
}
