#include <iostream>
using namespace std;

int64_t a[100005];

int main()
{
   int t;
   cin>>t;
   a[1]=1;
   a[2]=2;
   a[3]=4;
   for(int i=4;i<=100000;i++)
     {
     	a[i]=a[i-1]+a[i-2]+a[i-3];
     	a[i]%=1000000007;
     }
   while(t--)
    {
    	int n;
    	cin>>n;
    	cout<<a[n]<<"\n";
    }
}
