#include <stdio.h>
#include <bits/stdc++.h>
using namespace std ; 
int main()
{
    int n ; 
    cin >> n ; 
    int a ; 
    int b ; 
    cin >> a ; 
    cin >> b ; 
    int sum = 0 ; 
    for (int i = 1 ; i <n ; i++)
    {
    	if( i%a == 0  || i %b ==0 )
    	{
    		sum = sum + i ; 
    	}
    }
    cout << sum << endl ;
    return 0;
}
