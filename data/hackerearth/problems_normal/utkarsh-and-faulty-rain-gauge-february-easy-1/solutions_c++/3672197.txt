#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,k=0;
    double a=1;
    cin>>t;
    
    
    while(t--){
    	int i,n,r;
    	double sq,sum=0;
    	cin>>n>>r;
    	
    	sq=n*(n+1)*(2*n+1)*1.0/6.0;
    //	cout<<"sq="<<sq<<endl;
    	
    	
    	for(i=1;i<=n;i++){
    	//	cout<<(double)i/r<<" "<<(double)i*i/sq<<" ";
    	//	cout<<(double)i*i*i*1.0/(r*sq*1.0)<<endl;
    		sum=sum+ i*i*i*1.0/(r*sq*1.0);
    	}
    	
    	a*=sum;
    }
    printf("%.4lf\n",a);
    
    return 0;
}
