#include<iostream>
using namespace std;

int main()
 {
 	long long n,a,b,c,d,e;
 	long long sum=0;
 	cin>>n>>a>>b;
 	c=(n-1)/a;
 	d=(n-1)/b;
 	e=(n-1)/(a*b);
 	sum+=((2*a + (c-1)*a)*c)/2;
 	sum+=((2*b + (d-1)*b)*d)/2;
 	if(e>0)
 	sum-=((2*(a*b) + (e-1)*a*b)*e)/2;
 	cout<<sum<<endl;
 }
