def solve(input_str):
	# to handle end of string
	SENTINEL = '*'
	input_str += SENTINEL
	
	n_possible_sbstrs = 0
	n_consec_syms = 1
	
	for i in range(len(input_str)):
		if input_str[i] == SENTINEL:
			break
		elif input_str[i] == input_str[i+1]:
			n_consec_syms += 1
		else:
			n_possible_sbstrs += (n_consec_syms * (n_consec_syms + 1)) / 2
			n_consec_syms = 1
			
	return n_possible_sbstrs


if __name__ == '__main__':
	input_str = raw_input()
	print solve(input_str)
