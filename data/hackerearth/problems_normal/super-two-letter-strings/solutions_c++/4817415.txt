#include <iostream>
using namespace std;
long long super(int n,int p)
{
	long long dp[n+1];
	dp[0]=1;
	int i,j;
	for(i=1;i<=n;i++)
	{
		dp[i]=0;
		for(j=i-1;(j>=0)&&(i-j<=p);j--)
		{
			dp[i]+=dp[j];
			dp[i]=dp[i]%1000000007;
		}
	}
	return dp[n];
}
int main()
{
    int t,n,p,i,j;
    cin>>t;
    long long ans[t];
    for(i=0;i<t;i++)
    {
    	cin>>n>>p;
    	if(p==1) ans[i]=1;
    	else ans[i]=super(n,p);
    }
    for(i=0;i<t;i++) cout<<ans[i]<<endl;
    return 0;
}
