#include <iostream>
#include <vector>

using namespace std;

int main()
{
    int t =0;
    
    cin >> t;
    while(t--){
    	
    	int n;
    	cin >> n;
    	std::vector<int> a(n);
    	for(int i=0; i<n; i++){
    		cin >> a[i];
    	}
    	
    	int startIndex = -1;
    	int endIndex = -1;
    	
    	for(int i=1; i<n; i++) {
    		if(a[i-1]>a[i]){
    			startIndex = i-1;
    			break;
    		}
    	}
    	
    	if(startIndex==-1){
    		std::cout << startIndex << " " << endIndex << std::endl;
    		continue;
    	}
    	
    	for(int i=startIndex+1; i<n; i++){
    		endIndex = i;
    		if(a[i-1]<a[i]){
    			endIndex = i-1;
    			break;
    		}
    	}
    	
    	bool isSorted = true;
    	for(int i=endIndex+2; i<n; i++){
    		if(a[i-1]>a[i]){
    			isSorted=false;
    			break;
    		}
    	}
    	
    	if(!isSorted) {
    		std::cout << -1 << " " << -1 << std::endl;
    		continue;
    	}
    	
    	if(startIndex==0&&endIndex==n-1){
    		std::cout << startIndex+1 << " " << endIndex+1 << std::endl;
    	}else if(startIndex==0){
    		if(a[startIndex]<a[endIndex+1]){
    			std::cout << startIndex+1 << " " << endIndex+1 << std::endl;
    		}else{
    			std::cout << -1 << " " << -1 << std::endl;
    		}
    	}else if(endIndex==n-1){
    		if(a[endIndex]>a[startIndex-1]) {
    			std::cout << startIndex+1 << " " << endIndex+1 << std::endl;
    		}else{
    			std::cout << -1 << " " << -1 << std::endl;
    		}
    	}else{
    		if(a[startIndex]<a[endIndex+1]&&a[endIndex]>a[startIndex-1]){
    			std::cout << startIndex+1 << " " << endIndex+1 << std::endl;
    		}else{
    			std::cout << -1 << " " << -1 << std::endl;
    		}
    	}
    	
    }
}
