#include<bits/stdc++.h>
using namespace std;
int main(){

  cin.sync_with_stdio(false);

    int t;
    cin >> t;
    while(t--){

       string x , y;
       cin >> x >> y;

         int hsh[26]={0};
         for( int i = 0 ; i < x.size() ; ++i )
            hsh[x[i]-'a']=1;
            int flag = 0;
            for(int i = 0; i < y.size(); ++i ){
              if(hsh[y[i]-'a']==1)
              flag = 1;
            }
            if(flag)
            cout <<"YES"<<endl;
            else
            cout <<"NO"<<endl;

    }

     return 0;


}
