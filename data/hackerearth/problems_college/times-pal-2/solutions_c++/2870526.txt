#include <bits/stdc++.h>
using namespace std;
int check(char s[])
{
	long long int l=strlen(s);
	long long int flag=1,i,j;
	for(i=0,j=l-1;i<l/2;i++,j--)
	{
		if(s[i]!=s[j])
		{
			flag=0;
			break;
		}
	}
	return flag;
}

    	long long int x,ctr=0,y;

    	{
    		strcpy(s3,s4);
    		x=0;
    		for(j=i;j<i+m;j++)
    		{
    			s3[j]=s5[x++];

    		{
    			s3[j+m]=s4[j];
