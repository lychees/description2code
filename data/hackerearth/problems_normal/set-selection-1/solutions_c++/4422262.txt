#include<bits/stdc++.h>
using namespace std;
#define ll long long int
ll arr[200];
bool mark[110000][110];
ll n;
void cal()
{
	ll i,j;
	for(i=0; i<=n; i++)
	{
		mark[0][i]=true;
	}
	for(i=1; i<=100000; i++)
	{
		mark[i][0]=false;
	}
	for(i=1; i<=100000; i++)
	{
		for(j=1; j<=n; j++)
		{
			mark[i][j]=mark[i][j-1];
			if(i>=arr[j])
			{
				mark[i][j]=(mark[i][j])||(mark[i-arr[j]][j-1]);
			}
		}
	}
}
int main()
{
	
	scanf("%lld",&n);
	ll i,j,z;
	memset(mark,0,sizeof(mark));
	for(i=1; i<=n; i++)
	{
		scanf("%lld",&arr[i]);
	}
	cal();
	ll t;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&z);
		if(mark[z][n])
		{
			printf("Yes\n");
		}
		else
		{
			printf("No\n");
		}
	}
	return 0;
}
