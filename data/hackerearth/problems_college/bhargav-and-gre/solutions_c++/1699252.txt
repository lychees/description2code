#include <iostream>
using namespace std;

int main()
{
    string W;
    cin>>W;
    int b=W[0];
    int N;
    cin>>N;
    int s,a;
    s=a=0;
    while(N--)
    {
    	string str;
    	cin>>str;
    	int c=str[0];
    	if(c==b && str.length()==W.length())
    	{
    		s++;
    	}
    	else if((c-b)==3 && str.length()==W.length())
    	{
    		a++;
    	}
    }
    cout<<s<<" "<<a;
    return 0;
}
