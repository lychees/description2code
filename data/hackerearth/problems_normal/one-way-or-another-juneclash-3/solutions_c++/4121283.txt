#include<iostream>

using namespace std;

int main(){

    int t;
    cin >> t;
    while(t--) {
    unsigned long long n,x,y,val,i,f1=1,f2=1;
    cin >> n;
    for(i=1;i<=n;i++) {
        cin >> x >> y >> val;
        if(f1) {
            if(x % 2 != 0 && y % 2 != 0 && val % 2 == 0) f1 = 1;
            else if(x % 2 == 0 && y % 2 == 0 && val % 2 == 0) f1 = 1;
            else if(x % 2 != 0 && y % 2 == 0 && val % 2 != 0) f1 = 1;
            else if(x % 2 == 0 && y % 2 != 0 && val % 2 != 0) f1 = 1;
            else f1 = 0;
        }
        if(f2) {
            if(x % 2 != 0 && y % 2 != 0 && val % 2 != 0) f2 = 1;
            else if(x % 2 == 0 && y % 2 == 0 && val % 2 != 0) f2 = 1;
            else if(x % 2 != 0 && y % 2 == 0 && val % 2 == 0) f2 = 1;
            else if(x % 2 == 0 && y % 2 != 0 && val % 2 == 0) f2 = 1;
            else f2 = 0;
        }
    }
    if(f1 || f2) cout << "Yes\n";
    else cout << "No\n";
    }
    return 0;







}
