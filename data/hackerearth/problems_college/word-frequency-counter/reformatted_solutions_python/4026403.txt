line = raw_input()
line = line.split()
line = map(lambda x: x.upper(), line)
d = {}
repeat = []
seq = 0
for i in line:
    a = d.get(i)
    if a is None:
        repeat.append(False)
        a = 1
    else:
        repeat.append(True)
        a = a+1
    d.update({i: a})
    seq = seq + 1

seq = 0
for i in line:
    if not repeat[seq]:
        print i + " "+str(d.get(i))
    seq = seq + 1
