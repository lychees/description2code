//balls
#include<bits/stdc++.h>
using namespace std;

int main()
{
	long int n,layer=1,buy=0,rec=0;
	cin>>n;
	vector<int > v(n);
	for(int i=0;i<n;++i)
		cin>>v[i];
    sort(v.begin(),v.end());
    while(rec<n)
    {
        if(layer*layer>=v[rec])
        {
            buy+=layer*layer-v[rec];
            rec++;
        }
        else
        {
            buy+=layer*layer;
        }
        layer++;
    }
     cout<<buy<<'\n';

	return 0;


}
