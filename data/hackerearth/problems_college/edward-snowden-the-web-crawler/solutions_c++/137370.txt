#include <iostream>
#include <string.h>
#include <stdio.h>
using namespace std;

int main() {
	char a[100],b[1000][1000];
	int i,j=0,n,f,k;
	gets(a);
	while(1)
	{
		gets(b[j]);
		j++;
		if(strcmp(b[j-1],"exit")==0)
		break;
	}
	n=strlen(a);
	for(i=0;i<j;i++)
	{
		for(int m=0;m<strlen(b[i]);m++)
		{
			if(b[i][m]==34)
			{
				f=0;
				for(k=0;k<n;k++)
				{
					if(a[k]==b[i][m+k+1])
					f++;
				}
				if(f==n)
				{
				for(k=m+1;b[i][k]!=34;k++)
				cout<<b[i][k];
				cout<<endl;
				}
			}
		}
	}
	return 0;
}
