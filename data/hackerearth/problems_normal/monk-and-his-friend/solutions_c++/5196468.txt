#include <iostream>
using namespace std;
int checkbits(long long int m,long long int n){
	int count = 0;
	long long int xo = m^n;
	while(xo){
		count++;
		xo = xo & (xo - 1);
	}
	return count;
}
int main()
{
	long long int n,m,t;
	cin >> t;
	while(t--){
		cin >> m >> n;
		cout << checkbits(m,n) << endl;
	}
	return 0;
}
