#include <bits/stdc++.h>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	int n;
	string s;
	unordered_map<string, int> hist;
	cin >> n;
	while (n--) {
		cin >> s;
		sort(s.begin(), s.end());
		hist[s]++;
	}
	int ans = -1;
    for (auto &iter: hist) {
    	ans = max(ans, iter.second);
    }
    cout << ans << endl;
    return 0;
}
