#include <bits/stdc++.h>
using namespace std;
#define ll long long
ll t,i,j,k,l,c,su,n;
int main()
{
	string s,l;
    cin>>t;
    while(t--)
    {
    	cin>>s>>l;k=0;
    	for(i=0;i<l.length();i++)
    	{
    		for(j=0;j<s.length();j++)
    		{
    			if(l[i]==s[j])
    			{
    				i++;k++;
    			}
    		}
    	}
    	if(k==l.length())
    	{
    		cout<<"YES"<<endl;
    	}
    	else
    		cout<<"NO"<<endl;
    }
}
