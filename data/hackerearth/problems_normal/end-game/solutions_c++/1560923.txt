#include <iostream>
using namespace std;

int main()
{
    int test = 0;

  	cin>>test;
  	while(test--){
  		int n , pcol , prow , krow , kcol , turn;
  		cin>>n>>prow>>pcol>>krow>>kcol>>turn;
  		
  		int king_step_req = abs(kcol - pcol);
  		//cout<<king_step_req<<endl;
  		int king_row = krow + king_step_req ;
  		//cout<<king_row<<endl;
  		int pawn_row = prow + king_step_req ;
  		//cout<<pawn_row<<endl;
  		if(turn == 1 )
  			pawn_row--;
  		//else
  		//	pawn_row++;
  			
  		if(pawn_row > king_row || pawn_row > n)
  			cout<<"White Wins\n";
  		else
  			cout<<"Draw\n";
  	}
    return 0;
}
