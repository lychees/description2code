	string s;

	for(i=0;i<s.length();i++)
	  {
	      if(s[i]=='w')
	        x++;
	  }

	  while(i<s.length())
	  {      y=0;count=0;
	  	if(s[i]=='v' || s[i]=='w')
	  	 {  count++;
	  	 	if(s[i]=='v')
	  	 	  y++;
	  	 	 else
	  	 	  y=y+2;
	  	 	j=i+1;
	  	 	  while((s[j]=='v' || s[j]=='w')&& j<s.length())
	  	 	   {   count++;
	  	 	   	  if(s[j]=='v')
	  	 	   	    y++;
	  	 	   	   else
	  	 	   	    y=y+2;
	  	 	   	   j++;
	  	 	   }
	  	 	   	  min=min-count+y/2+y%2;
	  	 	  i=j;
	  	 }
	  	 else
	  	   i++;
	  }
	 // min=s.length()-y+y/2+y%2;
	  max=s.length()-x+x*2;
	cout<<min<<" "<<max;

}
