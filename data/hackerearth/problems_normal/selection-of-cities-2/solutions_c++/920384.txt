#include <bits/stdc++.h>
using namespace std;

int func(long long int a,long long  int b,long long int MOD) {
long long int m = 1, y = a;
    while(b > 0) {
        if(b%2 == 1) {
            m=(m*y);
            if(m>MOD) m%=MOD;
        }
        y = (y*y);
        if(y>MOD) y%=MOD;
        b /= 2;
    }
    return m;
}

int main(int argc, char const *argv[])
{
	long long int testcase;

	cin >> testcase;
	for (long long int i = 0; i < testcase; ++i)
	{
		long long N;
		cin >> N;
		cout << (func(2, N, 1000000007)-1)%1000000007 << endl;
	}
	
	return 0;

}
