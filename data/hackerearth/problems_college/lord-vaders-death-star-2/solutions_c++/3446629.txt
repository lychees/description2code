#include <bits/stdc++.h>
#define ll long long
using namespace std;

int A[1000005];

int main()
{
    int n, x = 0;
    map<int, int> mp;

    cin >> n;

    for (int i = 0; i < n; i++) {
        cin >> A[i];
    }
    sort(A, A + n);
    for (int i = n - 1; i >= 0; i--) {
        int u = upper_bound(A, A + n, A[i]) - A;
        int l = lower_bound(A, A + n, A[i]) - A;
        if (u - l == 1) {
            cout << A[i] << endl;
            return 0;
        }
    }
    cout << "-1\n";
    return 0;
    return 0;
}
