#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<cstring>
using namespace std;
int cs,i,s;
int A[10];
int main()
{
  scanf("%d",&cs);
  while(cs--)
  {
    for(i=0;i<3;i++)
      scanf("%d",&A[i]);
    s=0;
    for(i=0;i<3;i++)
      if(A[i]==9)
      {
        s+=10-3*i;
        break;
      }
      else if(A[i]>6)
        s+=2;
      else if(A[i])
        s++;
    if(i==3)
      s=0;
    printf("%d\n",s);
  }
  return 0;
}
