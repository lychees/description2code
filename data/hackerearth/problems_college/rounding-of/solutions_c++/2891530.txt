#include <iostream>
#include <cstring>
#include <cmath>
#include <functional>
#include <algorithm>

using namespace std;

int main()
{
	int tc;
	cin >> tc;

	int n,b,key,temp;
	
	while(tc--)
	{
	
		cin >> n >> b;
		key = b/2;
		temp = n % b;

		if(temp > key)
		{
			cout << b*((n/b)+1) << endl;
		}
		else if(temp < key)
		{
			cout << b * (n/b) << endl;
		}
		else
		{
			if( b%2 == 0)
			{
				cout << b*((n/b)+1) << endl;
			}
			else
			{
				cout << b*(n/b) << endl;
			}
		}
		
	}
	return 0;
}
