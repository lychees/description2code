using namespace std;
#define run(n) std::ios::sync_with_stdio((n));

inline void solve(){

	register long int N = 0, K = 0, prev = 0;
	register int counter = 1; cin >> N >> K; 
	register int coord[N];

	for(int i = 0; i < N; i++) cin >> coord[i];

	sort(coord,coord+N); register int ans = coord[0];

	for(int i = 1; i < N; i++)
		if(coord[i]-coord[i-1] <= K) ans = coord[i];
			else break;
		cout << ans << endl;

}

int main(){

run(0) solve();
