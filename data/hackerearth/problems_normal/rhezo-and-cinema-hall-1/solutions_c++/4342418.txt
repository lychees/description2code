#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
long long n,m,k;
static long long dp[501][501*10],ans2[501];
long long solve(long long int row,int sel,long long int k){
	if(dp[row][sel]!=-1)
		return dp[row][sel];
	if(row==n)
		return (sel>=k);
	long long ans = 0;
	for(long long int i=2;i<=m;i++){
		ans += (ans2[i] * solve(row+1,sel+i,k))%mod;
		ans%=mod;
	}
	dp[row][sel] = ans;
	return ans;
}
void solve2(int cur,int sel,bool pr){
	if(cur==m){
		ans2[sel]++;
		ans2[sel]%=mod;
		return;
	}
	if(pr==false)
	 	solve2(cur+1,sel+1,true);
	solve2(cur+1,sel,false);
}
int main(){
	for(int i=0;i<501;i++)
		for(int j=0;j<(10*501);j++)
			dp[i][j]=-1;
	scanf("%lld %lld %lld",&n,&m,&k);
	solve2(0,0,false);
	printf("%lld\n",solve(0,0,k));
	return 0;
}
