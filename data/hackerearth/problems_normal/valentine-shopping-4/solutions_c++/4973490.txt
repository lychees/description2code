#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,m,i,j;
    double p,q,r;
    double temp;
    
    scanf("%d",&t);
    
    while(t--)
    {
     scanf("%d %d",&n,&m);
    
     for(i=0;i<n;i++)
     	{
     		int max=-1; double maxval=INT_MAX;
     		for(j=0;j<m;j++)
     			{
     				scanf("%lf %lf %lf",&p,&q,&r);
     				temp=( ((100-p)/100) * ((100-q)/100) * ((100-r)/100) - 1 );
     				
     				if(temp<maxval)
     				{
     					max=j;
     					maxval=temp;
     				}
     				
     			}
     			printf("%d ", max+1);
     			
     	}
     	printf("\n");
    }
    	
    
    return 0;
}
