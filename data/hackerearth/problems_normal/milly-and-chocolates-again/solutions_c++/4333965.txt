#include <iostream>
using namespace std;

int binarySearch(int n,int *ce, int *cs, int low,int high) {
	int mid = (low+high)/2;
	if(n > cs[mid] && n <= ce[mid]) {
		return mid;
	} else if(n>ce[mid]) {
		return binarySearch(n,ce,cs,mid+1,high);
	} else if(n<=cs[mid]) {
		return binarySearch(n,ce,cs,0,mid-1);
	}
	return -1;
}

int main()
{
	int n;
	cin>>n;
	int *c = new int[n];
	for(int i=0;i<n;i++){
		cin>>c[i];
	}
	int qn;
	cin>>qn;
	int *q = new int[qn];
	for(int i=0;i<qn;i++){
		cin>>q[i];
	}
	int *ce = new int[n];
	int *cs = new int[n];
	int sum = 0;
	for(int i=0;i<n;i++){
		cs[i] = sum;
		sum += c[i];
		ce[i] = sum;
	}
	for(int i=0;i<qn;i++){
		printf("%d\n",binarySearch(q[i],ce,cs,0,n-1)+1);
	}
    return 0;
}
