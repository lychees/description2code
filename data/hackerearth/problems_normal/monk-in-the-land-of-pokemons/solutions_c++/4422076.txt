#include <bits/stdc++.h>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
    int t;
    cin>>t;
    int n,a,b;
    long cnt;
    while(t--)
    {
    	vector<int> v(1000006,0);
    	cin>>n;
    	cnt=0;
		for(int i=0;i<n;i++)
		{
			cin>>a>>b;
			v[a]++;
			if(v[b]!=0){
				v[b]--;
			}
			else cnt++;
		}
		cout<<cnt<<endl;

    }
    return 0;
}
