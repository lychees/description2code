    #include <bits/stdc++.h>
    #include <vector>
    #include <algorithm>
    using namespace std;

    int main()
    {
        int n,i,ti,di;
        vector <pair <int,int> > vp;

        cin>>n;
        for(i=1;i<=n;i++){
        	cin>>ti>>di;
        	vp.push_back(make_pair(ti+di,i));
        }

        sort(vp.begin(),vp.end());
        for(i=0;i<vp.size();i++){
        	cout<<vp[i].second<<" ";
        }
        return 0;
    }
