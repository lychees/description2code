#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t,x;
    cin>>t;
    while(t--)
    {
    	cin>>x;
    	int xx=-1;
    	for(int i=0;i<=x;i+=5) {
    		if((x-i)%3==0) {
    			xx=i;
    			break;
    		}
    	}
    	if(xx==-1) {
    		cout<<"-1\n";
    		continue;
    	}
    	for(int i=1;i<=x-xx;i++) cout<<'5';
    	for(int i=1;i<=xx;i++) cout<<'3';
    	cout<<endl;
    }
    return 0;
}
