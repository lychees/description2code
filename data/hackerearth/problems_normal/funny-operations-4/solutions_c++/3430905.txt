#include <iostream>
using namespace std;

int main()
{
	int tt;
	cin>>tt;
	while(tt--){
		int n;
		cin>>n;
		int a[n];
		for(int i=0;i<n;i++)cin>>a[i];
		long long int ans = 0, maxl = 0 ;
		for(int i=1; i< 1<<n ; i++){
			long long int And = 0 ,Xor = 0, result = 0,len =0;
			int c = 0;
			for(int j = 0 ; 1<<j <= i ; j++){
				if(i>>j & 1 ){
					if(c>0){
						And &= a[j];
						Xor ^= a[j];
					}else {
						c++;
						And = a[j];
						Xor = a[j];
					}
					len++;
				}
			}
			result = And | Xor;
			if(result > ans) ans = result, maxl = len;
			else if(result == ans && len > maxl) maxl = len;
		}
		cout<<ans<<" "<<maxl<<"\n";
	}
    return 0;
}
