#include<bits/stdc++.h>

#define lli long long
#define pb push_back
#define mp make_pair

#define lld "%I64d"

#define mod 1000000007

const int lmt = 1000005;
int tmp = 0;

void solve(int x,int y){
    if(x<y) swap(x,y);
    if(y==0){
        tmp = INT_MAX;
        return;
    }
    if(y==1){
        tmp += x-1;
        return;
    }
    tmp += x/y;
    solve(x%y,y);
}

int main(){
    int t,n;
    scanf("%d",&t);
    while(t--){
        int ans = INT_MAX;
        scanf("%d",&n);
        if(n==1){
            printf("0\n");
            continue;
        }
        if(n&1){
            for(int i=1;i<n;i++){
                tmp = 0;
                solve(n,i);
                ans = min(ans,tmp);
            }
        }else{
            for(int i=1;i<n;i+=2){
                tmp = 0;
                solve(n,i);
                ans = min(ans,tmp);
            }
        }
        printf("%d\n",ans);
    }
    return 0;
}
