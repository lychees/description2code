#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
int t,n,m;
scanf("%d",&t);
while(t--)
{
	scanf("%d",&n);
	vector<int> v[205];
	int i,x,y;
	scanf("%d",&m);
	for(i=0;i<m;i++)
	{
		scanf("%d%d",&x,&y);
		v[x].push_back(y);
		v[y].push_back(x);
	}
	int fl[205];
	for(i=0;i<n;i++)
	fl[i] = -1;
	
	queue<int> q;
	int valid = 1;
	
for(i=0;i<n;i++)
{
	if(fl[i]==-1)
	{
	fl[i] = 0;
	q.push(i);
	
	while(!q.empty())
	{
		int cur,x,l,j;
		cur = q.front();
		q.pop();
		l = v[cur].size();
		for(j=0;j<l;j++)
		{
			x = v[cur][j];
			if(fl[x]==-1)
			{
				fl[x] = 1-fl[cur];
				q.push(x);
			}
			else if(fl[x]==fl[cur])
			{
				valid = 0;
				break;
			}
		}
	}
	
	}
}

if(valid)printf("BICOLORABLE.\n");
else printf("NOT BICOLORABLE.\n");
}
return 0;
}
