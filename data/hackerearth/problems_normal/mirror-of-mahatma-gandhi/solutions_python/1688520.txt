t = input()
for i in range(t):
	N = raw_input()
	l = 0
	is_mirror = True
	r = len(N)-1
	while(l<=r):
		if(l<r):
			if(N[l]!=N[r] or N[l] not in ['0','1','8'] or N[r] not in ['0','1','8']):
				print "NO"
				is_mirror=False
				break
			else:
				l+=1
				r-=1
		else:
			if(N[l] not in ['0','1','8']):
				print "NO"
				is_mirror=False
				break
			else:
				l+=1
				r-=1
	if(is_mirror):
		print "YES"
