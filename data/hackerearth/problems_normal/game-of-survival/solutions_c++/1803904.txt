#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    int n;
    int flag;
    while(t--)
    {
    	flag=0;
    	cin>>n;
    	vector <int> s;
    	vector <int> a;
    	s.resize(n);
    	a.resize(n);
    	for(int i=1;i<=n;i++)
    	{
    		cin>>s[i];
    	}
	   	for(int i=1;i<=n;i++)
    	{
    		cin>>a[i];
    	}    	
    	sort(s.begin(),s.end());
    	sort(a.begin(),a.end());
    	for(int i=1;i<=n;i++)
    	{
    		if(s[i]<=a[i])
    		{
    			flag=1;
    			break;
    		}
    	}
    	if(flag==1)
    	{
    		cout<<"NO"<<endl;
    	}
    	else
    	{
    		cout<<"YES"<<endl;
    	}    	
    }
    return 0;
}
