#include <iostream>
#include <cstdio>
#include <map>
using namespace std;

int main()
{
    int n;
    scanf ("%d", &n);
    map<int, int> mp;
    while (n--) {
    	int x;
    	scanf ("%d", &x);
    	mp[x]--;
    }
    int m;
    scanf ("%d", &m);
    while (m--) {
    	int x;
    	scanf ("%d", &x);
    	mp[x]++;
    }
    
    for (map<int, int>::iterator it = mp.begin (); it != mp.end (); it++) {
    	if (it->second > 0)
    		printf ("%d ", it->first);
    }
    return 0;
}
