#include<map>
#include<cstdio>
#include<vector>
#include<queue>
using namespace std;
#define gc getchar//_unlocked
inline void scanint(int &x)
{
    register int c = gc();
    x = 0;
    int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}
double ar[20][20];  // To Store inputs
double ans[300400];  // to store answers
int f[300040];
queue<int> q;
int main()
{
	int N; scanint(N);
	for(int i=0;i<N;i++)
	for(int j=0;j<N;j++)
	scanf("%lf",&ar[i][j]);
	int s= 1<<N; s--;
	f[s]=1; q.push(s);
	ans[s]=1.0;
	while(q.size())
	{
		int t=q.front(); q.pop();
		int cc=0;
		for(int i=0;i<N;i++)
		{
			int tt= t & (1<<i); if(tt) cc++;
		}
		
		int mp = cc * (cc -1); mp/=2;
		double pp= 1.0/mp;
		for(int i=0;i<N;i++)
		{
			int tt= t & (1<<i); if(tt==0) continue;
			for(int j=0;j<i;j++)
			{
				tt= t & (1<<j); if(tt==0) continue;
				int x=(1<<i) ^ t ;
				if(f[x]==0)
				q.push(x);
				f[x]=1;
				ans[x] += ar[j][i]*ans[t]*pp;
			//	printf("setting %d %lf %lf %lf %d  %lf\n",x,ar[i][j],ans[t],pp,t,ans[x]);
				
				x=(1<<j) ^ t ;
				if(f[x]==0)
				q.push(x);
				f[x]=1;
				ans[x]+=ar[i][j]*ans[t]*pp;
			//	printf("setting %d %lf %lf  %d  %lf\n",x,ar[j][i],ans[t],t,ans);
				
			}
		}
		
		
	}
		for(int i=0;i<N;i++)
		{
			int t= 1<<i;
			printf("%0.5lf ",ans[t]);
		}
	
}
