#include <cstdio>
#include <iostream>
using namespace std;

#define MAX 2001

int main(){
    int n,i,ans = 0,sum;
    char str[MAX];

    scanf("%d",&n);

    while(n--){
        scanf("%s",str);

        for(i=sum=0;str[i];++i)
            sum+=(str[i]-'0');
        sum%=9;
        ans = (ans + sum)%9;
    }

    if(ans) printf("%d\n",ans);
    else printf("9\n");

    return 0;
}
