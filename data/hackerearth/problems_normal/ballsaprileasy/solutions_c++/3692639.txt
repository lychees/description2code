#include <bits/stdc++.h>

using namespace std;

#define DEBUG ON

#else

#endif

void do_nothing() {}



int main()
{
    int n;
    long long req;
    cin>>n;

    vector<long long> balls(n+1,0);
    req = 0;
    for(int i=1;i<=n;i++)
    {
        cin>>balls[i];
    }
    sort(balls.begin(), balls.end());
    long long j,i=1;
    for(j=1;j<=n;i++)
    {
        long long temp = (long long)i * (long long)i;
        if(balls[j] <= temp)
        {
            req += temp - balls[j];
            j++;
        }
        else
        {
            req += temp;
        
        }

    }
    cout<<req<<endl;
    return 0;
}
