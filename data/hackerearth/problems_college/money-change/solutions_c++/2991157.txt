#include <iostream>
#include <algorithm>

using namespace std;

typedef long long ll;
#define flx(i,a,b) \
	for(ll i = ll(a); i < ll(b); i++)

int main()
{
    ll n = 0, tc = 0, _ = 0;
    ll d[1000] = {0};
    ll a[1000][1000];
   
   		
    	cin >> tc;
    	cin >> n;
    	for (ll i = 0; i < n; i++){
    		cin >> _;
    		d[i] = _;
    	}
    	
    	sort(d,d+n);
    	
    	for (ll i = 0; i < n; i++){
    		for (ll j = 0; j <= tc; j++){
    			//cout << d[i] << ":" << j << " ";
    			
    			if (j == 0)
    				a[i][j] = 1;
    			else if (i == 0)
    				a[i][j] = 1;
    			else if (d[i] > j)
    				a[i][j] = a[i-1][j];
    			else
    				a[i][j] = a[i-1][j] + a[i][j-d[i]];
    				
    		}
    		 //cout << endl;
    	}
    	/*
    	flx(i,0,n){
    		flx(j,0,tc+1){
    			cout << a[i][j] << " ";
    		}
    		cout << endl;
    	}
    	*/
    	cout << a[n-1][tc] << endl;
    
    return 0;
}
