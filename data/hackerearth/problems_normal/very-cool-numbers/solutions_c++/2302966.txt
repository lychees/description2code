#include<iostream>
using namespace std;
typedef unsigned int uint;

uint coolness(uint x)
{
	uint f=0,s=0,t,cool=0;
	while(x!=0)
	{
	  t=(x&1);
	  if(f==1&&s==0&&t==1)
      {
      	++cool;
      }
      f=s;
      s=t;
      x=x>>1;
	}
  return cool;
}

int main()
{
     uint t;
     cin>>t;
     ++t;
     
     while(--t)
     {
     	uint r,k,ans=0,cool;
     	cin>>r>>k;
     	++r;
     	for(uint i=1;i!=r;++i)
     	{
     		cool=coolness(i);
     		if(cool>=k)
     		  ++ans;
     	}
     	cout<<ans<<endl;
     }
    
    return 0;
}
