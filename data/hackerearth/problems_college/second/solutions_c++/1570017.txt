#include <iostream>
#include<algorithm>
using namespace std;
typedef long long ll;
int main()
{
    ll i,j,m,n,a[100001];
    scanf("%lld%lld",&n,&m);
    for(i=0;i<n;i++)
       scanf("%lld",&a[i]);
    sort(a,a+n);   
    for(i=0;i<n;i++){
    	if(a[i]==m)
    	  break;
    }   
    int s=0;
    if(i<n){
    	int count=2; 
    	s=a[i]; 
    	for(j=i+1;j<n;j++){
    		if(count*m==a[j]){
    			s=a[j];
    			count++; 
    		}
    	}
    }
    printf("%lld\n",s);
    return 0;
}
