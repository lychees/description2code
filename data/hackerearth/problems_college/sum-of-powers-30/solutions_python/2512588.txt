from math import log, sqrt
a = set([1])
for i in xrange(1, 1000):
	for p in xrange(2, 20):
		if i**p > 1000000:
			break
		a.add(i**p)
a = sorted(a)
for _ in xrange(int(raw_input())):
	x = int(raw_input())
	ans = "No"
	for i in a:
		if i > x:
			break
		if x - i in a:
			ans = "Yes"
			break
	print ans
