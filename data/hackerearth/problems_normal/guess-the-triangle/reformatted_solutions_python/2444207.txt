from operator import itemgetter
a1, b1 = map(float, raw_input().split())
a2, b2 = map(float, raw_input().split())
a3, b3 = map(float, raw_input().split())
x1 = a2+a3-a1
x2 = a1+a3-a2
x3 = a1+a2-a3
y1 = b2+b3-b1
y2 = b1+b3-b2
y3 = b1+b2-b3
ans = [[x1, y1], [x2, y2], [x3, y3]]
ans.sort(key=itemgetter(0, 1))
print "%.4f %.4f" % (ans[0][0], ans[0][1])
print "%.4f %.4f" % (ans[1][0], ans[1][1])
print "%.4f %.4f" % (ans[2][0], ans[2][1])
