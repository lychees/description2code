#include <bits/stdc++.h>
#define ll long long
#define read(tc) scanf("%d",&tc)
#define out(ans) printf("%d ",ans)
#define rep(from,till) for(int i=from;i<=till;i++)
#define repdec(l,m) for(int k=l;k>=m;k--)
#define line_change() printf("\n")
using namespace std;

int main()
{
    int tc,n,val,max;
    read(tc);
    rep(1,tc){
    	vector<int>A(1000001,0);
    	read(n);
    	rep(0,n-1){
    		read(val);
    		A[val]+=1;
    	}
    	max=A[0];
    	rep(1,1000000){
    		if (max<A[i]){
    			max=A[i];
    		}
    	}
    	repdec(1000000,0){
    		if (max==A[k]){
    			out(k);
    		}
    	}
    	line_change();
    }
    
    return 0;
}
