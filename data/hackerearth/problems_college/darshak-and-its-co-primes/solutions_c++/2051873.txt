#include <bits/stdc++.h>
using namespace std;
int gcd(long long int p, long long int q)
{
	if(q==0)
		return p;
	else
		return gcd(q,p%q);
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int p,q;
    	cin>>p>>q;
    	int res=gcd(p,q);
    	if(res==1)
    	cout<<"Is a Co-Prime"<<endl;
    	else
    	cout<<"Not a Co-Prime"<<endl;
    }
}
