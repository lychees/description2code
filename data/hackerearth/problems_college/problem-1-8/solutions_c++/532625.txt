#include<iostream>
#include<string>

using namespace std;

int main()
{
	string s,s1;
	cin>>s;
	string ans = "zzzzz" + s;
	int c=0;
	int n;
	cin>>n;
	int t=n;
	while(n--)
	{
		string s1;
		cin>>s1;
		int found = s1.find(s);
		
		
		
		if(found==0)
		{
			ans = min(ans,s1);
		}		
		else
		c++;
		
	}

	
	if(c==t)
	cout<<s<<endl;
	else 
	cout<<ans;
}
