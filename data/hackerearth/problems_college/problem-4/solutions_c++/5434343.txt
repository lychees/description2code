#include <iostream>
#include<string>
using namespace std;

int main()
{
    string s;
    cin>>s;
    char b;
    int hash[26]={0},i,j;
    for(i=0;i<s.length();i++)
    {
    	hash[s[i]-'a']++;
    }
    for(i=0;i<26;i++)
    {
    	if(hash[i])
    	{
    		b=i+'a';
    		for(j=0;j<hash[i];j++)
    		cout<<b;
    	}
    }
    return 0;
}
