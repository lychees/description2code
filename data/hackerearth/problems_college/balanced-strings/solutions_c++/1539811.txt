inline int scan(){
    char c = getchar_unlocked();
    int x = 0;
    while(c<'0'||c>'9'){
        c=getchar_unlocked();
    }
    while(c>='0'&&c<='9'){
        x=(x<<1)+(x<<3)+c-'0';
        c=getchar_unlocked();
    }
    return x;
}
int main(){
    int t=scan();
    while(t--){
        char a[1001]={NULL};
        scanf("%s",a);
        int n=strlen(a);
        if((n%26)){
            printf("No\n");
            continue;
        }
        int count[26]={0};
        for(int i=0;i<n;i++){
            count[a[i]-'a']++;
        }
        bool f=1;
        for(int i=1;i<26;i++){
            if(count[i]!=count[i-1]){
                f=0;
                break;
            }
        }
        if(f){
            printf("Yes\n");
        }
        else{
            printf("No\n");
        }
    }
}
