#include <iostream>
#include<string.h>
#include<stdio.h>
#include<algorithm>
using namespace std;

int main()
{
	char str[10005];
	int n;
	gets(str);
	cin>>n;
	int len=strlen(str);
	int line=0;
	for(int i=0;i<len;i++)
	{
		cout<<str[i];
		line++;
		if(line==n)
		{
			cout<<endl;
			line=0;
		}
	}
    return 0;
}
