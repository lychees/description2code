#include<bits/stdc++.h>

using namespace std;

#define ll long long int
#define print(a) cout<<a
#define T ll t; cin>>t ; while(t--)
#define fin freopen("input.txt","r",stdin)

int main() {

    ll n;
    scanf("%lld",&n);

    ll a[n+1];

    for(ll i=0;i<n;i++) {
        scanf("%lld",&a[i]);
    }

    ll sum;
    scanf("%lld",&sum);
    
    sort(a,a+n);

    for(ll i=0;i<n;i++) {
        ll j=i+1,k=n-1;
        
        if(j>=n || k<=0)
        	continue;
        
        while(k!=j){
            if(a[i] + a[j] + a[k] == sum) {
                printf("True\n");
                return 0;
            }
            else if(a[i] + a[j] + a[k] < sum) {
                j++;
            }
            else {
                k--;
            }
        }
    }

    printf("False\n");

    return 0;
}
