#include<iostream>
using namespace std;
long long int f[100004][3];
long long int r[100004];
int main()
{
	long long int t,n,m,i,q,l,e;
	long long int a[100001];
	cin>>t;
	while(t--)
	{
		cin>>n>>m;
		for(i=0;i<m;i++)
		{
			cin>>f[i][0]>>f[i][1]>>f[i][2];
		}
		cin>>q;
		for(i=0;i<=m+1;i++)
		a[i]=0;
		for(i=0;i<q;i++)
		{
			cin>>l>>e;
			a[l]++;
			a[e+1]--;
		}
		for(i=2;i<=m;i++)
		{
			a[i]=a[i-1]+a[i];
		}
		for(i=0;i<=n+1;i++)
		{
			r[i]=0;
		}
		for(i=0;i<m;i++)
		{
			r[f[i][0]]=r[f[i][0]]+f[i][2]*a[i+1];
			r[f[i][1]+1]=r[f[i][1]+1]-f[i][2]*a[i+1];
		//	cout<<"\n"<<r[f[i][0]]<<"  "<<r[f[i][1]+1]<<"\n";
		}
		for(i=2;i<=n;i++)
		{
			r[i]+=r[i-1];
		}
		for(i=1;i<=n;i++)
		{
			cout<<r[i]<<" ";
		}
		cout<<endl;
	}
	return 0;
}
