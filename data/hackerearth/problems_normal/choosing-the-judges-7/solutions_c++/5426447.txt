#include<iostream>
using namespace std;
long long int arr[10005];
int main(){
int t;
cin>>t;
for(int j=1;j<=t;j++){
 long long int n;
 cin>>n;
  for(int i=0;i<n;i++)cin>>arr[i];
  arr[1]=max(arr[0],arr[1]);
  for(int i=2;i<n;i++){
    arr[i]=max(arr[i-2]+arr[i],arr[i-1]);
  }
  cout<<"Case "<<j<<": "<<arr[n-1]<<"\n";
}
}
