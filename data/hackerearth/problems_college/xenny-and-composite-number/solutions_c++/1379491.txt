#include <iostream>
#include<math.h>
using namespace std;

int  isPrime(int num)
{
    if (num == 1) return 0;
    if (num == 2) return 1;

    int newnum = floor(sqrt(num));

    for (int i = 2; i <= newnum; i++) 
        if (num % i == 0) return 0;

    return 1;
}
int main(void)
{
   int t,n;
   cin >> t;
   while(t--)
   {
   	cin >> n;
   	while(1)
   	{
   		n++;
   		if(isPrime(n))
   			continue;
   		else
   			break;
   	}
   	if(n==1)
   		cout << "4" << endl;
   		else
   	cout << n << endl;
   }
    return 0;
}
