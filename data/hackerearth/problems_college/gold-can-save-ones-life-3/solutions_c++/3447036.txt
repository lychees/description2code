using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,k;
        cin>>n>>k;
        int arr[n];
        for(int i=0;i<n;i++)
            cin>>arr[i];
        int dp[n];
        int ma[n];
        dp[0]=max(arr[0] , 0);
        ma[0]=dp[0];
        for(int i=1;i<n;i++)
        {
            int j = i-k-1;
            if(j<0)
            {
                dp[i]=max(arr[i] , 0);
                ma[i]=max(ma[i-1],dp[i]);
            }
            else
            {
                dp[i] =max(ma[j]+arr[i] , ma[j]);
                ma[i] = max(ma[i-1] , dp[i]);
            }
        }

        cout<<ma[n-1]<<endl;
    }
    return 0;
}
