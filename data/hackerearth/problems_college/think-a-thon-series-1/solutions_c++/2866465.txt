#include<bits/stdc++.h>
using namespace std;
int main()
{
    	long long int t,l,p,c,k,x,steps;
    	cin>>t;
    	
    	for(int i=1; i<=t; i++)
    	{
    		cin>>l>>p>>c;
    		steps=0;
			x=l;
    		k=0;
    		
    		while(x<p)
    		{
    			x=x*c;
    			k++;
    		}
    		while((1<<steps)<k)
    			steps++;
    		cout<<"Case #"<<i<<": "<<steps<<endl;
    	}
    	return 0;
}
