#include<stdio.h>
#include<math.h>
int main()
{
	int t,n,a[12],b[12],i;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		for(i=0;i<n;i++)
		scanf("%d",&a[i]);
		for(i=0;i<n;i++)
		scanf("%d",&b[i]);
		int xo=0;
		for(i=0;i<n;i++)
		{
			if(a[i]-b[i]>0)
			xo=xo^(a[i]-b[i]);
			//printf("1 %d ",xo);
			if(b[i]-1>0)
			xo=xo^(b[i]-1);
		}
		if(xo==0)
		printf("Second\n");
		else
		printf("First\n");
	}
}
