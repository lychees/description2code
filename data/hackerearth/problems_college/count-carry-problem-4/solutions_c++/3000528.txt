#include <iostream>
using namespace std;

int main()
{
    int n,i,count,c[1000];
    cin>>n;
    long first,second;
    for(i=0;i<n;i++)
    {
    	count=0;
    	cin>>first>>second;
    	long temp1=first,temp2=second,carry=0;
    	while(temp1!=0||temp2!=0)
    	{
    		int temp=(temp1%10+temp2%10+carry);
    		if(temp>9)
    		{
    			count++;
    			carry=1;
    		}
    		else
    		carry=0;
    		temp1/=10;
    		temp2/=10;
    		
    	}
    	c[i]=count;
    	
    }
    for(i=0;i<n;i++)
    {
    	if(c[i]==0)
    	cout<<"No carry operation\n";
    	else if(c[i]==1)
    		cout<<c[i]<<" carry operation\n";
    		else
    		cout<<c[i]<<" carry operations\n";
    }
    
    
    return 0;
}
