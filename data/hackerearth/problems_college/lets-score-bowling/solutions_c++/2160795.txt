#include<bits/stdc++.h>
using namespace std;
# define start cin.sync_with_stdio(false);
# define done return 0;
# define mod 1000000007
int main()
{
    start
    int t,a,b,c,ans,p;
    cin>>t;
    while(t--)
    {
    	p=0;
        ans=0;
        cin>>a>>b>>c;
            if(a<9 && a>6)
                ans+=2;
            else if(a<7 && a>0)
                ans+=1;
            else if(a==9)
                {
                	p++;
                	goto gg;
                }
            if(b<9 && b>6)
                ans+=2;
            else if(b<7 && b>0)
                ans+=1;
            else if(b==9)
                {
                	p++;
                	goto gg;
                }
            if(c<9 && c>6)
                ans+=2;
            else if(c<7 && c>0)
                ans+=1;
            else if(c==9)
                {
                	p++;
                	goto gg;
                }
        if(p==0)
        ans=0;
      gg:  {
        if(a==9)
            ans+=10;
        else if(b==9)
            ans+=7;
        else if(c==9)
            ans+=4;
        }
        cout<<ans<<endl;
    }
}
