#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
struct pz
{
    ll a,b;
};
bool cond(pz z1,pz z2)
{
    return (z1.a>z2.a);
}
int main()
{
    ll t,nG,nR,i,j,k,x;
    cin>>t;
    while(t--)
    {
        cin>>nR>>nG;
        ll grp[nR],c=0,mygrp;
        pz w[nR];
        for(i=0,k=1;i<nR;i++)
        {
            cin>>x;
            w[i].a=x;
            w[i].b=i;
            if(k>nG)
                k=1;
            grp[i]=k;
            k++;
        }
        sort(w,w+nR,cond);
        for(i=0;i<nR;i++)
        {
            x=w[i].b;
            if(x==0)
            {
                mygrp=grp[i];
                j=i;
                break;
            }
        }
        for(i=0;i<j;i++)
        {
            if(grp[i]==mygrp)
                c++;
        }
        cout<<c<<'\n';
    }
    return 0;
}
