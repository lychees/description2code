def get_frequency(compressed_string):
	frequency = [0]*27
	i = 0 
	for i in xrange(len(compressed_string)):
		character = compressed_string[i]
		if character.isalpha():
			freq = ""
			j = i+1
			for j in xrange(i+1, len(compressed_string)):
				if compressed_string[j].isalpha():
					i = j-1
					break
				else:
					freq += compressed_string[j]
			index = ord(character) - 97
			frequency[index] += int(freq)
	return frequency
	
compressed_string = raw_input()
no_of_questions = input()
frequency = get_frequency(compressed_string)
for i in  xrange(no_of_questions):
	index = input()

		for j in xrange(len(frequency)):
			if index <= temp_index:
				print chr(j+96)
