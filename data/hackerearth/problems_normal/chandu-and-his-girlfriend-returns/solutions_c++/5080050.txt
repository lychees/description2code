#include <iostream>
using namespace std;

int main()
{
 	int t,n,m,*ar1,*ar2,*ar3,i,j,k;
	cin>>t;
	while(t)
	{
		cin>>n>>m;
		ar1=new int[n];
		ar2=new int[m];
		ar3=new int[n+m];
		
		for(i=0;i<n;i++)
			cin>>ar1[i];
			
		for(i=0;i<m;i++)
			cin>>ar2[i];
			
		i=0;
		j=0;
		k=0;
		while(i<n && j<m)
		{
			if(ar1[i]>ar2[j])
				ar3[k++]=ar1[i++];
			else 
				ar3[k++]=ar2[j++];
		}
		while(i<n)
			ar3[k++]=ar1[i++];
			
		while(j<m)
			ar3[k++]=ar2[j++];
			
		for(i=0;i<n+m;i++)
			cout<<ar3[i]<<" ";
			cout<<endl;
		
		t--;
	}
	return 0;
}
