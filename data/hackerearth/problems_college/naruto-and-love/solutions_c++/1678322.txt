#include <bits/stdc++.h>
using namespace std;


const int N = 131072, BS = 2e8;
int t, n, m, hpos, mpos;
long long H[N], M[N], ans;

bool f(long long pos) {
	hpos = mpos = 0;
	for (int i = 0; i < n; ++i) {
		while (abs(H[hpos] - M[i])>pos) {
			++hpos;
			if (hpos == m) break;
		}
		if (hpos == m) break;
		++mpos, ++hpos;
		if (hpos == m) break;
	}
	return (mpos == n);
}

long long bs(long long lo, long long hi) {
	if (lo == hi) return lo;
	else {
		long long mid = (lo + hi) / 2;
		if (f(mid)) {
			return bs(lo, mid);
		}
		else {
			return bs(mid + 1, hi);
		}
	}
}

int main() 
{
	cin >> t;
	while (t--) {
		cin >> n >> m;
		for (int i = 0; i < n; ++i) {
			cin >> M[i];
		}
		for (int i = 0; i < m; ++i) {
			cin >> H[i];
		}
		sort(H, H + m);
		sort(M, M + n);
		ans = bs(0, BS);
		cout << ans << "\n";
	}
	return 0;
}
