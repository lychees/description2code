#include <iostream>
#include<string.h>
using namespace std;

int main()
{
	int tc,l,i;
	char a[100005];
	cin>>tc;
	while(tc--)
	{
		cin>>a;
		l=strlen(a);
		a[l]='w';
		a[l+1]='w';
		for(i=0;i<l;)
		{
			if(a[i]=='R'&&a[i+1]=='Y'&&a[i+2]=='Y')
			i=i+3;
			else if(a[i]=='R'&&a[i+1]=='Y')
			i=i+2;
			else if(a[i]=='R')
			i++;
			else
			{
				cout<<"\nNO";
				break;
			}
		}
		if(i==l)
		cout<<"\nYES";
	//	else 
	//	cout<<"\n"<<i;
	}
    
    return 0;
}
