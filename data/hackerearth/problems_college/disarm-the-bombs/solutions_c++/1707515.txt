#define LL long long
#define FIT(i,t) for(i=0;i<t;i++)
#define FIN(i,n) for(i=0;i<n;i++)
#define FJT(j,t) for(j=0;j<t;j++)

#define MAX2(a,b) a>b?a:b
#define MIN2(a,b) a>b?b:a
#define REP(i,a,b) for(i=a;i<=b;i++)

using namespace std;

LL dp[105][105][105];
LL e[105];
LL t[105];
LL d[105];


LL func(LL ind, LL se, LL st, LL E , LL T , LL n  ) {
            
           if(ind == n)
           return 0;
           
           if(dp[ind][se][st] != -1)
           return dp[ind][se][st];
           LL l = 0;
           if(se + e[ind] <= E && st + t[ind] <= T)
           l = max(l,d[ind] + func(ind+1, se+ e[ind] , st+t[ind], E, T, n));
           
           l = max(l,func(ind+1,se,st,E ,T,n));
           
           return dp[ind][se][st] = l;
}

int main()
{
    LL n,T,E;
    LL i,j,k,l;
    cin >> n >> E >> T;
    memset(dp,-1,sizeof dp);
    LL sm = 0;
    for(i=0;i<n;i++)
    cin >> e[i];
    
    for(i=0;i<n;i++)
    cin >> t[i];
    
    for(i=0;i<n;i++) {
    cin >> d[i];
    sm += d[i];
    }
    
    l = func(0,0,0, E,T,n);
    l = sm - l;
    cout << l << endl;
    
     
    return 0;
}
