#include<bits/stdc++.h>
using namespace std;
struct node{
	int a,b;
};
int n,m,q;
int a[1001][1001];
int main()
{
	scanf("%d%d%d",&n,&m,&q);
	int total=0;
	memset(a,0,sizeof(a));
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=m;j++)
		{
			scanf("%d",&a[i][j]);
			if(a[i][j]==1)
			total++;
		}
	}
	while(q--)
	{
		int x,y;
		scanf("%d%d",&x,&y);
		int count=0;
		if(a[x][y]==1)
		{
			node temp;
			temp.a=x,temp.b=y;
			a[x][y]=2;
			stack <node> s;
			s.push(temp);
			while(!s.empty())
			{
				node temp1;
				temp1=s.top();
				s.pop();
				count++;
				int u=temp1.a,v=temp1.b;
				node temp2;
				if(u+1<=n)
				{
					temp2.a=u+1,temp2.b=v;
					if(a[u+1][v]==1)
					a[u+1][v]=2,s.push(temp2);
				}
				if(u-1>=1&&a[u-1][v]==1)
					{
						a[u-1][v]=2;
						temp2.a=u-1,temp2.b=v;
						s.push(temp2);
					}
					if(v+1<=m&&a[u][v+1]==1)
					{
						a[u][v+1]=2;
						temp2.a=u,temp2.b=v+1;
						s.push(temp2);
					}
					if(v-1>=1&&a[u][v-1]==1)
					{
						a[u][v-1]=2;
						temp2.b=v-1,temp2.a=u;
						s.push(temp2);
					}
			}
		}
		total=total-count;
		printf("%d\n",total);
	}
}
