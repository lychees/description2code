#include <iostream>
#include<bits/stdc++.h>
using namespace std;


int main()
{
    int n,m;
    cin>>n>>m;
    long long int a[n+1][m+1];
    map<int,pair<int,int>>mp;
    for(int i=1;i<=n;i++)
    {
    	for(int j=1;j<=m;j++)
    	{
    		cin>>a[i][j];
    		mp[a[i][j]]=make_pair(i,j);
    	}
    }
   int q;
   cin>>q;
   while(q--)
   {
   	int x;
   	cin>>x;
   	
   	if(mp[x].first==0)
   	cout<<"-1 -1"<<endl;
   	else
   cout<<mp[x].first-1<<" "<<mp[x].second-1<<endl;
   }
    
}
