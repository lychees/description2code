#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int maxx=-1000,maxy=-1000,minx=1000,miny=1000,x,y;
		for(int i=0;i<n;++i)
		{
			cin>>x>>y;
			if(maxx<x)
				maxx=x;
			if(minx>x)
				minx=x;
			if(maxy<y)
				maxy=y;
			if(miny>y)
				miny=y;
			//maxx=max(maxx,x);
			//minx=min(minx,x);
			//maxy=max(maxy,y);
			//miny=min(minx,y);
		}
		int ans=(maxx-minx)*(maxy-miny);
        if(ans==0)
            ans=(maxx-minx)+(maxy-miny);
        cout<<ans<<endl;
	}
	return 0;
}
