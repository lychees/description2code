#include <iostream>
using namespace std;
 
int main()
{
   int n,even,odd,temp,j;;
   cin>>n;
   long long int even_sum,odd_sum;
   even=0;
   even_sum=0;
   odd=0;
   odd_sum=0;
   int a[n],b[n],c[n];     //array b is for even digit and array c is for odd disgits
   for(int i=0;i<n;i++)
   {
   	cin>>a[i];
   }
   for(int i=1;i<n;i++)    //insertion sort for array a
   {
   	if(a[i]<a[i-1])
   	{
   		temp=a[i];
   		j=i-1;
   		while(j>=0 && temp<a[j])
   		{
   			a[j+1]=a[j];
   			a[j]=temp;
   			j--;
   		}
   	}
   }
   for(int i=0;i<n;i++)   //out of sorted array 'a', now i am segregating even and odd terms in                                                                                   //  respective  arrays 
   {
   	if(a[i]%2==0)
   	{
   	b[even]=a[i];
   	even++;
    }
    else
    {
    	c[odd]=a[i];
    	odd++;
    }
   }
   
   for(int i=0;i<even;i++)
   {
   	cout<<b[i]<<" ";           //printing sorted even array & also taking their sum 
   	even_sum=even_sum+b[i];
   }
   cout<<even_sum<<" ";
   for(int i=0;i<odd;i++)
   {
   	cout<<c[i]<<" ";
    odd_sum=odd_sum+c[i];
   }
   cout<<odd_sum;
   
}
