{
	int t;
	cin>>t;
	while(t--)
	{
		string s;
		cin>>s;
		int flag=0;
		for(int i=0;i<s.length();i++)
		{
			if(s[i]=='a'||s[i]=='e'||s[i]=='i'||s[i]=='o'||s[i]=='u' ||s[i]=='A'||s[i]=='E'||s[i]=='I'||s[i]=='O'||s[i]=='U')
			{
				flag=1;
				cout<<s[i];
			}
		}
		if(flag==0)
		cout<<"No";
		cout<<endl;
	}
	return 0;
}
