#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
	long n,i;
	scanf("%ld",&n);
	long arr[n+50];
	long long ans,tmp;
	for(i=0; i<n; i++)
		scanf("%ld",&arr[i]);
		sort(arr,arr+n);
		ans=0; tmp=0;
		for(i=1; i<n; i++)
		{
			if(arr[i]!=arr[i-1])
			{
				ans+=i;
				ans-=tmp;
			}
			else
				tmp++;
		}
		printf("%lld\n",ans);
		return 0;
}
