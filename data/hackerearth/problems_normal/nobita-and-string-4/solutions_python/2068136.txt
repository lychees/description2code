t=int(input())
while t>0:
    s=raw_input()
    s=s.split()
    j=len(s)-1
    i=0
    while i<j:
        k=s[i]
        s[i]=s[j]
        s[j]=k
        j=j-1
        i=i+1
    s=str(s)
    s=s.replace(']','')
    s=s.replace('[','')
    s=s.replace(',','')
    s=s.replace("'",'')
    print(s)
    t=t-1
