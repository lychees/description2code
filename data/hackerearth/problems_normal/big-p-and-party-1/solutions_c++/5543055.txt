#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define li long int
#define vi vector<int>
vi graph[100005];
bool visited[100005];
int cost[100005];
int bfs(){
	queue<int> q;
	q.push(0);
	visited[0]=1;
	int src;
	while(!q.empty()){
		src=q.front();
		q.pop();
		for(int i=0;i<graph[src].size();i++){
			if(!visited[graph[src][i]]){
				cost[graph[src][i]]=cost[src]+1;
				q.push(graph[src][i]);
				visited[graph[src][i]]=1;
			}
		}
	}
}
int main(){
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int m,n,a,b;
	cin>>n>>m;
	while(m--){
		cin>>a>>b;
		graph[a].push_back(b);
		graph[b].push_back(a);
	}
	memset(cost,INT_MAX,sizeof(cost));
	cost[0]=0;
	bfs();
	for(int i=1;i<n;i++){
		if(cost[i]!=INT_MAX)
			cout<<cost[i]<<endl;
		else cout<<"-1"<<endl;
	}
}
