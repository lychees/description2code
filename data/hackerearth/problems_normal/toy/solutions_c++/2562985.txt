#include <iostream>
#include<algorithm>
using namespace std;
int marble_array[100001];
int main() {
	// your code goes here
	
	int n,units_of_marble,max_element;;
	
	cin>>n;
	for(int i=0;i<n;i++)
	{
	  cin>>marble_array[i];
	 
	}
	units_of_marble=1;
	sort(marble_array,marble_array+n);
max_element=marble_array[0]+4;

	for(int i=0;i<n;i++)
	{
	 if(marble_array[i]<=max_element)
	 {
	 
	   continue;
	 }
	 else
	 {
	  max_element=marble_array[i]+4;
	  units_of_marble++;
	 }
	
	}
	cout<<units_of_marble;
	return 0;
}
