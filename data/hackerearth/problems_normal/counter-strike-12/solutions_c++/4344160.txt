#include <iostream>
using namespace std;

int posx[1007], posy[1007], targetsx[1007], targetsy[1007];

int main()
{
	int t, n, m, d;
	scanf("%d", &t);
	for (int turns = 0; turns < t; turns++) {
		scanf("%d %d %d", &n, &m, &d);
		for (int i = 0; i < n; i++) {
			int x, y;
			scanf("%d %d", &x, &y);
			posx[i] = x;
			posy[i] = y;
		}
		for (int i = 0; i < m; i++) {
			int x, y;
			scanf("%d %d", &x, &y);
			targetsx[i] = x;
			targetsy[i] = y;
		}
		int count = 0;
		for (int i = 0; i < m && count <= m/2; i++) {
			for (int j = 0; j < n; j++) {
				if (abs(targetsy[i]-posy[j]) + abs(targetsx[i]-posx[j]) <= d) {
					count++;
					break;
				}
			}
		}
		if (count > m/2) {
			printf("YES\n");
		}
		else{
			printf("NO\n");
		}
	}
    return 0;
}
