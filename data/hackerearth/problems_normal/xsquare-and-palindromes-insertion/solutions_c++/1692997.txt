#include<cstdio>
#include<cstring>
int main()
{
	int t,l,i;
	char str[10000];
	scanf("%d",&t);
	while(t--)
	{
		scanf("%s",str);
		int l=strlen(str);
		int h[400]={0};
		for(i=0;i<l;i++)
		{
			h[str[i]]++;
		}
		int o=0;
		for(i='a';i<='z';i++)
		{
			if(h[i]%2!=0)
			o++;
		}
		if(o>=1)
		printf("%d\n",o-1);
		else
		printf("0\n");
	}
	return 0;
}
