#include <algorithm>
#include <climits>
#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>
#include <string>

using namespace std;

char a[10];
int n, f[10];

int main() {
  scanf("%s", a);
  n = strlen(a);
  for (int i = 1; i < n; i++)
    f[i] = int(a[i]) - 48;
  int ans = 1;
  for (int i = 2; i <= 6; i++)
    if (f[i] == 0)
      f[i - 1] *= 10;
  for (int i = 1; i <= 6; i++)
    ans += f[i];
  cout << ans << endl;
  return 0;
}
