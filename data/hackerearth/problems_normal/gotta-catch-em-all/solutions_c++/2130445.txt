

    #include <iostream>
    using namespace std;
    int MAX = 100000+7;
     
    int partition(int ar[], int p, int r){
    	int pivot = ar[r];
    	int index = p-1;
    	int temp;
    	for(int j = p; j < r; j++){
    		if(ar[j] > pivot){
    			index++;
    			temp = ar[index];
    			ar[index] = ar[j];
    			ar[j] = temp;
    		}
    	}
    	temp = ar[index+1];
    	ar[index+1] = ar[r];
    	ar[r] = temp;
    	return index+1;
    }
     
    void quicksort(int ar[], int p, int r){
    	if(p < r){
    		int q = partition(ar, p, r);
    		quicksort(ar,p,q-1);
    		quicksort(ar,q+1,r);
    	}
    }
     
    int main()
    {
        int k;
        int ar[MAX];
        cin>>k;
        
        for(int i = 1; i <= k; i++)
        	cin>>ar[i];
        
        quicksort(ar,1,k);
        
        /*for(int i = 0; i < k; i++)
        {
        	cout<<ar[i]<<" ";		
        }*/
        
        int max = 0;
        
        for(int i = 1; i <= k; i++){
        	int num = ar[i]+i;
        	if(num > max)
        		max = num;
        }
        cout<<max+1<<endl;
        return 0;
    }
