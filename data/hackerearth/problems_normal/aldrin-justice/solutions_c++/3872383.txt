#include<iostream>
using namespace std;
int max(int a,int b){
if(a > b){
    return a;
}else{
return b;
}
}
int min(int a,int b){
if(a < b){
 return a;
}
else {
    return b;
}
}
int main(){
int t;
cin>>t;
while(t--){
    int a,b,c,d,l,m;
    cin>>a>>b>>c>>d;
    l = max(min(a,b),min(c,d));
    m = min(max(a,b),max(c,d));
    if(l > m){
            cout<<"Line"<<endl;

    }else if(l == m){
    cout<<"Point"<<endl;
    }else {
    cout<<"Nothing"<<endl;
    }
}
return 0;
}
