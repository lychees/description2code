#include <iostream>
using namespace std;
long long int a[101],b[101];
int main()
{
	int n;
	cin >> n;
	for(int i=0;i<n;i++){
		cin >> a[i];
	}
	long long int temp,sum = 0;
    for(int i=0;i<n;i++){
    	temp = (a[i]*(i+1))-sum;
    	sum += temp;
    	b[i] = temp;
    }
    for(int i=0;i<n;i++){
    	cout << b[i] << " ";
    }
    return 0;
}
