#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int t, n, i;
    vector<int> input;
    cin >> t;
    while(t--)
    {
    	cin >> n;
    	input.clear();
    	input.resize(n);
    	for(int i = 0; i<n; i++)
    		cin >> input[i];
    	sort(input.begin(), input.end());
    	
    	for(i = n-1; i>=2; i--)
    	{
    		if(input[i-1] + input[i-2] > input[i])
    		{
    			cout << input[i-2] << ' ' << input[i-1] << ' ' << input[i] << endl;
    			break;
    		}
    	}
    	if(i == 1)
    		cout << -1 << endl;
    }
    return 0;
}
