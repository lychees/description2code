#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int tc=0;
    cin>>tc;
    while(tc-->0)
    {
    	int n=0;
    	cin>>n;
    	int a[n];
    	for(int i=0;i<n;i++)
    		cin>>a[i];
    	
    	sort(a,a+n);
    	int b[n];
    	for(int i=0;i<n;i++)
    		b[i]=a[i]-a[0];
    	
    	int s=0,c=0;
    	for(int i=0;i<n;i++)
    	{
    		s=b[i];
    		c=c+s/5;
    		s=s%5;
    		c=c+s/2;
    		s=s%2;
    		c=c+s/1;
    	}
    	cout<<c<<endl;
    }
    return 0;
}
