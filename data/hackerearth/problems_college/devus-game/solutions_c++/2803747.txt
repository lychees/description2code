#include<iostream>
using namespace std;
int main()
{ 
   int n,tc,a[100000],i,flag=0,temp;
  cin>>tc;

while(tc--)
{  
   cin>>n;
  
for(i=0;i<n;i++)
cin>>a[i];
flag=0;
for(i=0;1;)
{ 
        if(a[i]==-1)
           {  cout<<"Win"<<endl;
            break;
           }

            else if(a[i]>=0&&a[i]<=n-1)
            {  temp=a[i];
               a[i]=-2;

            i=temp;
       
              }
             else if(a[i]==-2)
            { flag=1;

            break;
               }

  } 
         if (flag==1)
  cout<<"Lose"<<endl;

}

return 0;
}
