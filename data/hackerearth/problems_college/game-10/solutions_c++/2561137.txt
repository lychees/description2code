#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	long long int m,n;
    	cin>>m>>n;
    	if(n<=(m-1)/2)
    	cout<<n*2+1<<endl;
    	else
    	{
    		n=m-1-n;
    		cout<<2*n<<endl;
    	}
    }
    return 0;
}
