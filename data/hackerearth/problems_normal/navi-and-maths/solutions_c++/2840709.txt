#include<bits/stdc++.h>
#define bug(x) cerr<<#x<<" = "<<x<<endl;
#define ll long long int
#define mod 1000000007
using namespace std;
ll fast_exp( int base, int expo )
{
        ll x = 1 , y = base , b = expo ;
        
        while( b )
        {
                if( b & 1 )
                        x = ( x * y ) % mod ;
                y = ( y * y ) % mod ;
                b >>= 1 ;
        }
        
        return x % mod;
} 
int main() {
	int t,i,j,k=0,n;
	cin>>t;
	while(t--) {
		k++;
		cin>>n;
		ll arr[n];
		for(i=0;i<n;i++) {
			cin>>arr[i];
		}
		ll num=1,den=0,max1=-1;
		for(i=1;i<pow(2,n);i++) {
			ll inc=n-1;
			num=1;den=0;
			for(j=0;j<n;j++) {
				if(1<<j & i) {
					num=(num%mod*arr[inc]%mod)%mod;
					den+=arr[inc];
		//			cout<<j<<" ";
				}
				inc-=1;
			}
			//if(i==10)
			//bug(den);
		//	cout<<"\n";
		//	bug(num)
		//	bug(den)
		ll ans=(num%mod*fast_exp(den,mod-2)%mod)%mod;
			if(ans>max1) {
				max1=ans%mod;
			}
		}
		cout<<"Case #"<<k<<": "<<max1<<"\n";
	}
	return 0;
}
