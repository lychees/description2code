#include <iostream>
#include <stdio.h>
#include <string.h>
#define ll long long
using namespace std;
double two, three;
ll n, p;
double dp[1000001];
float solve()
{
	ll i;
	dp[0]=1;dp[2]=two;dp[3]=three;
	for(i=4;i<=n;i++)
	{
		dp[i]=two*dp[i-2]+three*dp[i-3];
	}
	return dp[n];
}
int main()
{
	double ans;
    cin>>n>>p;
    memset(dp,0,sizeof(dp));
    two=p/100.0;
    three=1-p/100.0;
    solve();
    printf("%.6lf",dp[n]);
    return 0;
}
