#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
     long int n;
    cin>>n;
    long int d[n],t[n];
    vector<pair<int,int>> a(n);
    for(long int i=0;i<n;i++)
    {cin>>t[i] >>d[i];
     a[i].first=t[i]+d[i];
    a[i].second=i+1;}
    sort(a.begin(),a.end());
    for(long int i=0;i<n;i++)
        cout<<a[i].second<<" ";
        
    
    
    
    return 0;
}
