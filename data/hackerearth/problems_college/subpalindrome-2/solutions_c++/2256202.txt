template < typename T > T abs(T x)
{
	return x > 0 ? x : -x;
}

template < typename T > T sqr(T x)
{
	return x * x;
}

ull dp[5005][5005];

int main()
{	
	int t;
	
	cin >> t;
	
	string s;
	
	while (t--) {
		cin >> s;
		
		int n = s.size();
		
		for (int i = 0; i <= n; i++) {
			for (int j = 0; j <= n; j++)
				dp[i][j] = 0;
		}
		
		for (int i = 0; i < n; i++) {
			dp[i][i] = 1;
			if (i < n - 1) {
				dp[i][i + 1] = 2 + (s[i] == s[i + 1]);
			}
		}
		
		for (int i = 3; i <= n; i++) {
			for (int j = 0; j < n - i + 1; j++) {
				int st = j, f = j + i - 1;
				dp[st][f] = dp[st][f - 1] + dp[st + 1][f] - (s[st] == s[f] ? -1 : dp[st + 1][f - 1]);
			}
		}
		
		cout << dp[0][n - 1] << "\n";
	}
	
	return 0;
}
