int isprime(int x)
{
	int i;
	if(x%2==0)
	return 0;
	for(i=3;i*i<=x;i++)
	{
		if(x%i==0)
		return 0;
	}
	return 1;
}
int ispal(int x)
{
	int a=x;
	int rev=0;
	while(a!=0)
	{
		int b=a%10;
		rev=rev*10 + b;
		a=a/10;
	}
	if(x==rev)
	return 1;
	else
	return 0;
}

int main()

    int t;
