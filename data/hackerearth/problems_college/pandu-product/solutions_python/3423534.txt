import time
n = input()
a = [int(x) for x in raw_input().split()]

def naive(a):
    result = -99999999
    for i in range(0, n):
        for j in range(i+1, n):
            if a[i]*a[j] > result:
                result = a[i]*a[j]
    return result

def sorted_sol(a):
    a.sort()
    return max(a[-1] * a[-2],a[0]*a[1])
start = time.time()
print(sorted_sol(a))
