#include<iostream>
using namespace std;
int main()
{
    int n,p,q,t;
    cin>>t;
    while(t--){
        cin>>n;
        if(n%2!=0) cout<<"-1"<<endl;
        else{
            n = n/2;
            p = n/4;
            q = n/3;
            int d;bool flag=0;
            for(int i=0;i<=p;i++){
                for(int j=0;j<=q;j++){
                    if(((4*i)+(3*j))== n){
                        d=i+j;
                        flag=1;
                        break;
                    }
                }
            }
            if(flag)
                cout<<d<<endl;
            else cout<<"-1"<<endl;
        }
    }


}
