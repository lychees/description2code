#include <iostream>
using namespace std;
long long array[1000];
int main()
{
    int t;long long int n,point,n1,n2;int a;int array[27];
    cin>>t;
    while(t--){
        cin>>n;
        long long int sum=0,ans=0;
        point = n*12;
        n1 = point/26;
        n2 = point%26;
        for(int i=1;i<=26;i++){
            cin>>array[i];
            sum+=array[i];
        }
