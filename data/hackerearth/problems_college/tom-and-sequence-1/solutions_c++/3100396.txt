#include <bits/stdc++.h>
#define ll long long
#define pb push_back

using namespace std;

const long long MOD = 1000000007;

ll prime[1500007];
vector<ll> a;

void pre() {
	for(int i = 0 ; i < 1500007;i++) prime[i] = 1;
	prime[0]=0, prime[1]=0;
	for(ll i = 2 ; i*i <= 1500000 ; i++) {
		if(prime[i]) {
			for(ll j = i*i ; j <= 1500000 ; j += i) {
				prime[j] = 0;
			}
		}
	}
	for(int i = 2 ; i<=1500000 ; i++) {
		if(prime[i]) {
			a.pb(i);
		}
	}
	for(ll i = 1 ; i < a.size() ; i++) {
		a[i] = (a[i]%MOD * a[i - 1]%MOD) % MOD;
	}
}

int main() {
	pre();
	int t; cin >> t; while(t--) {
		int n; cin >> n; cout << a[n - 1] % MOD << "\n";
	}
}
