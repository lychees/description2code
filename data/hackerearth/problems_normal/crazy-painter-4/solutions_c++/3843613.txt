#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main()
{
    int t;
    cin >> t;
    ll cost[26];
    ll freq[26];
    while (t--) {
    	ll n;
    	cin >> n;
    	for (int i = 0; i < 26; i++)
    		cin >> cost[i];
    	ll t_p = 12 * n;
    	ll f = t_p / 26;
    	ll ad = t_p - f * 26;
    	for (int i = 0; i < 26; i++)
    		freq[i] = f;
    	for (int i = 0; i < ad; i++)
    		freq[i]++;
    	ll ans = 0;
    	for (int i = 0; i < 26; i++) {
    		ans += freq[i] * cost[i];
    		ans += freq[i] * (freq[i] - 1) / 2;
    	}
    	
    	cout << ans << endl;
    	
    }
    return 0;
}
