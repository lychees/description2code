import sys
t = int(sys.stdin.readline())
while t > 0:
    n = int(sys.stdin.readline())
    a = map(int, sys.stdin.readline().split())
    s = sum(a)
    ps = [0]*(s+1)
    ps[0] = 1
    for i in range(n):
        for j in range(s, a[i]-1, -1):
            ps[j] = ps[j] | ps[j-a[i]]
    print sum(ps)
    t = t - 1
