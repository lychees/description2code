#include <iostream>
#include <cstring>

using namespace std;

#define forl(i,l,u) for(int i=(l);i<(u);i++)
#define forn(i,n) forl(i,0,n)
#define clr(a) memset(a,0,sizeof(a));
typedef long long ll;

const int maxn=1000000+11;
const int mod=1000000000+7;

int npi(int n, int i){
	//n! / n-i!
	ll ans=1;
	forl(x,n-i+1,n+1)
		ans=(ans*x)%mod;
	return int(ans);
}

int main(){
	char s1[maxn],s2[maxn];
	int c1[1<<8],c2[1<<8];
	
	int t;
	cin>>t;
	while(t--){
		clr(c1);
		clr(c2);
		cin>>s1>>s2;
		int l1=strlen(s1);
		int l2=strlen(s2);
		forn(i,l1){
			c1[s1[i]-'a']++;
		}
		forn(i,l2){
			c2[s2[i]-'a']++;
		}
		ll ans=1;
		forn(i,'z'-'a'+1){
			if(c1[i]<c2[i]){
				ans=0; break;
			}
			ans*=npi(c1[i], c2[i]);
			ans%=mod;
		}
		cout<<ans<<endl;
	}
	
}
