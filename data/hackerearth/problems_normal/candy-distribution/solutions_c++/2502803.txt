const ll mod=ll(1e9)+7;

ll powmod(ll a,ll b)
{
	if(a==0)
		return 0;
	if(b==0)
		return 1;
	ll temp=powmod(a,b/2);
	temp*=temp;
	temp%=mod;
	if(b%2==1)
	{
		temp*=a;
		temp%=mod;
	}
	return temp;

}
map<ll,ll>my;
ll ans(ll a)
{
	if(a==1||a==2)
		return 1;
	if(my.find(a)!=my.end())
	{
//		cout<<a<<" ";
		return my[a];
	}
	a--;
	ll ret=0;
	ll x=sqrt(a);
	for(int i=1;i<=x;i++)
	{
		if(a%i==0)
		{
			ll u=a/i;
			ret+=powmod(ans(i),u);
			if(u*u!=a&&u!=a)
				ret+=powmod(ans(u),i);
			ret%=mod;
		}
	}
	my[a+1]=ret;
	return ret;
}
