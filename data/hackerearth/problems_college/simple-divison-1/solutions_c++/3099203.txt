#include <bits/stdc++.h>
using namespace std;

int main() 
{
	int t,tmp,b,a,rem,c;
	cin>>t;
	
	while(t--)
	{
		scanf("%d %d %d",&b,&a,&c);
		tmp = b/a;
		rem = b%a;
		printf("%d",tmp);
		if(c!=0)
		{
			printf(".");
			for(int i=0; i<c; i++)
			{
				rem = rem*10;
				tmp = rem/a;
				printf("%d",tmp);
				rem = rem % a;
			}
		}
		printf("\n");
	}
	return 0;
}
