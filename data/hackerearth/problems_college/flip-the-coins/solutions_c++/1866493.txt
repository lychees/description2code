#include <iostream>
using namespace std;
int main()
{
	int n,q;
	cin>>n>>q;
	int* ar=new int[n];
	for(int i=0;i<n;i++)
	{
		ar[i]=0;
	}
	int count=0;
	for(int i=0;i<q;i++)
	{
		int a,b,c;
		cin>>a>>b>>c;
		count=0;
		if(a==0)
		for(int j=b;j<=c;j++)
		{
			if(ar[j]==0)
			ar[j]=1;
			else
			ar[j]=0;
		}
		else
		{
			for(int j=b;j<=c;j++)
			{
				if(ar[j]==1)
				count++;
			}
			cout<<count<<endl;
		}
	}
}
