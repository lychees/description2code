#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
     {   int p,x,y,u,v;
     	 cin>>p>>x>>y>>u>>v;
     	 int a=abs(x-u);
     	 int b=abs(y-v);
     	 
     	 if((a==b&&p>1)||((a+b)%2==0&&p>2)||(a==0&&b==0))
     	 cout<<"Counter-Terrorists Win !\n";
     	 else
     	 cout<<"Terrorists Win !\n";
     	   
     }
    return 0;
}
