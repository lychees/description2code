#include<cstdio>
#include<cmath>
using namespace std;
int main(){
    int t;
    scanf("%d",&t);
    while(t--){
        long long x,c=0;
        int j;
        scanf("%lld",&x);
        while(x!=0){
        j=(int)(log(x)/log(2));
        long long z=pow(2,j);
        c++;
        x-=z;
        }
        printf("%lld\n",c);
    }
    return 0;
}
