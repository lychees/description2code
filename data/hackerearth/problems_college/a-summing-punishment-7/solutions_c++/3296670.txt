#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
int n,q,i,j,sum=0,a,b;
cin>>n;

int *arr = new int [n];
for(i=0;i<n;i++)
{
	cin>>arr[i];
}

cin>>q;
int *o=new int[q];
for(j=0;j<q;j++)
{
	sum=0;
	cin>>a>>b;
	for(i=a-1;i<b;i++)
	{
		sum = sum + arr[i];
	}
	o[j]=sum;
}
for(int i=0;i<q;i++)
cout<<o[i]<<"\n";




return 0;
}
