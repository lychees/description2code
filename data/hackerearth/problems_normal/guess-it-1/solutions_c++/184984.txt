#include<stdio.h>
#include<string.h>
#define max 201
int main()
{
    int cases;
    scanf("%d",&cases);
    while(cases--)
    {
        int sum = 0;
        int res;
        char num[max];
        scanf("%s",num);

        for(int i=0; i<strlen(num); i++)
            if(num[i] != '_')
                sum += (num[i] - '0');
        res = 9-(sum%9);
        if(res == 9)
            printf("0 or 9\n");
        else
            printf("%d\n",res);
    }
    return 0;
}
