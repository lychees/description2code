for i in range(int(raw_input())):
	m = int(raw_input())
	sharpness = [];
	strength = [];
	for i in xrange(m):
		sharpness.append(int(raw_input()))
	for i in xrange(m):
		strength.append(int(raw_input()))
	sharpness.sort()
	strength.sort()
	flag = 0
	for i in range(m):
		if sharpness[i] <= strength[i]:
			flag = 1
			break
	if flag == 1:
		print "NO"
	else:
		print "YES"
