#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
#define MOD 1000000007
using namespace std;
int main()
{
        long long int arr[3],a,b,c,i,t,ans;
        scanf("%lld",&t);
        while(t--)
        {
                for(i=0;i<3;i++)
                        scanf("%lld",&arr[i]);
                sort(arr,arr+3);
                a=arr[0]%MOD;
                b=(arr[1]-1)%MOD;
                c=(arr[2]-2)%MOD;
                ans=((((a%MOD)*b)%MOD)*c)%MOD;
                printf("%lld\n",ans);
        }
}
