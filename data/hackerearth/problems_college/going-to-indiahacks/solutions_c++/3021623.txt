    #include <bits/stdc++.h>
    using namespace std;
    #define ll long long
    ll co=LLONG_MAX;
    ll mo=1000000007;
    unsigned long long g[1005][1005];
     
    int main() { 
        for (int i = 0; i < 1005; ++i) {
            g[i][0] = 1;
            g[0][i] = 1;
        }
        for (int i = 1; i < 1005; ++i) {
            for (int j = 1; j < 1005; ++j) {
                g[i][j] = g[i-1][j] + g[i][j-1]+g[i-1][j-1];
                g[i][j]=(g[i][j]+mo)%mo;
            }
        }
       int n;
       scanf("%d",&n);
       while(n--)
       {
       	
       	int x,y,a,b;
       	scanf("%d%d%d%d",&x,&y,&a,&b);
       	int c=a-x,d=b-y;
       	 printf("%lld\n", g[c][d]%mo);
       }
       
        return 0;
    }
