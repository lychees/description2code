#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    long int n,i,j;
    scanf("%d",&t);
    while(t--){
    	scanf("%ld",&n);
    	long long int a[n];
    	for(i=0;i<n;i++)
    	{
    		scanf("%lld",&a[i]);
    	}
    	if(n==1)
    	{
    		printf("0\n");
    		continue;
    	}
    	j=0;
    	for(i=0;i<n-1;i++)
    	{
    		if(a[i]>=a[i+1])
    		j++;
    	}
    	printf("%ld\n",j);
    }
    
    
    
    
    return 0;
}
