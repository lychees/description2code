using namespace std;
#include<iostream>
int ar[101];
int main()
{
    int t;
    cin>>t;
    while(t>0)
    {
        int n;
        cin>>n;
         int y=0;
        while(y<101)
        {
            ar[y]=0;
            y++;
        }
        while(n>0)
        {
            int k;
            cin>>k;
            ar[k]++;
            n--;
        }


        int k;
        cin>>k;
        int i=0,j=100;
        int sum=0;
        while(i<j)
        {
            //cout<<i<<" "<<j<<endl;
            if(ar[i]!=0&&ar[j]!=0)
            {
                //cout<<"hmm"<<endl;
                if(i+j>k)
                  j--;
            else if(i+j<k)
                  i++;
            else if(i+j==k)
            {
                sum=sum+(ar[i]*ar[j]);
                i++; j--;
            }
            }
            else
            {
                if(ar[i]==0)
                    i++;
                if(ar[j]==0)
                    j--;
            }
        }
        if(ar[i]>=2&&2*i==k)
        sum=sum+(ar[i]*(ar[i]-1)/2);
        cout<<sum<<endl;
         t--;
    }
}
