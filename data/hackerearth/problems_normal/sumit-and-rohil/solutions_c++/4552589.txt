#include <bits/stdc++.h>
using namespace std;

int main()
{
    vector<string> v;
    int T;
    cin>>T;
    int count =1;
    while(T--)
    {
        string us;
        cin>>us;
        if(v.empty())
            v.push_back(us);
        else
        {
            int flag = 0;
            for(auto i : v)
            {
                if(multiset<char> (i.begin(),i.end()) == multiset<char> (us.begin(),us.end()) && i[0]==us[0] && *i.rbegin()== *us.rbegin())
    			{

    				flag = 1;
    				break;
    			}
            }
            if(flag == 0)
    		{
    		    count++;
    			v.push_back(us);
    		}
        }
    }
    cout<<count;
    return 0;
}
