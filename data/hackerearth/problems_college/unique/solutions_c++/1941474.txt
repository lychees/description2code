#include <iostream>
#include<map>
using namespace std;
#define ll long long
int main()
{
    ll t,n;
    cin>>t;
    while(t--)
    {
    	string s;
    	ll count=0;
    	map<string,long> arr;
    	cin>>n;
    	for(ll i=0;i<n;i++)
    	{
    		cin>>s;
    		arr[s]++;
    	}
    	
      map <string,long>::iterator it;
    	for(it=arr.begin();it!=arr.end();it++)
    	{
    		if(it->second>=1)
    		count++;
    	}
    	cout<<count<<endl;
    }
    return 0;
}
