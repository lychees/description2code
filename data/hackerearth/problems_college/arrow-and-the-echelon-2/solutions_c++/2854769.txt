#include <iostream>
using namespace std;

int main()
{
    int t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	int temp=n,i=0,fi=0,th=0;
    	while(5*i<=temp)
    	{
    		double ans=((double)(temp-5*i)/3);
    		int p=ans;
    		if(ans-p!=0)
    		 {
    		 	i++;
    		 }
    		 else
    		 {
    		 	fi=i;
    		 	th=p;
    		 	break;
    		 }
    	}
    	string str="";
    	for(int i=0;i<th*3;i++)
    	{
    	    str+="5";
    	}
    	for(int i=0;i<fi*5;i++)
    	{
    	    str+="3";
    	}
    	if(fi==0 && th==0)
    	 cout<<"-1\n";
    	else
    	 cout<<str<<"\n";
    }
    return 0;
}
