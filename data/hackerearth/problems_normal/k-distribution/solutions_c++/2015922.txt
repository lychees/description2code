inline void print(int a) { char s[20]; int i=0; do { s[i++]=a%10+'0'; a/=10; } while(a); i--; while(i>=0) pc(s[i--]); pc('\n'); }
inline void prlong(ll a) { char s[20]; int i=0; do { s[i++]=a%10+'0'; a/=10; } while(a); i--; while(i>=0) pc(s[i--]); pc('\n'); }

ll a[101], dp[101][MAX];
int main() {
	int n;
	ll k,tot=0,sum1, sum2;
	rdi(n); rdl(k);
	rep(i,0,n) rdl(a[i]), tot+=a[i];
	if(tot<(2*k)) {
		cout<<"0\n";
		return 0;
	}
	rep(i,0,n+1) dp[i][0]=1;
	rep(i,1,n+1) rep(j,0,k) {
		if(tot-j>=k) {
			if(j-a[i-1]>=0) dp[i][j]=dp[i-1][j]+dp[i-1][j-a[i-1]];
			else dp[i][j]=dp[i-1][j];
		}
	}
	sum1=0;
	rep(i,0,k) sum1+=dp[n][i], sum1%=mod;
	sum2=1;
	rep(i,0,n) sum2=(2*sum2)%mod;
	ll ans=(sum2-(2*sum1)+mod)%mod;
	prlong(ans);
}
