inline int go(ll n, int k = 0)
{
    if (n == 1) return 0;
    if (k & 1)
        return 1 + go((n + a - 1) / a, k + 1);
    return 1 + go((n + b - 1) / b, k + 1);
}

inline void solve()
{
    scanf("%d%d"I64, &a, &b, &c);
    puts(go(c) & 1 ? "Adam" : "Bob");
}
