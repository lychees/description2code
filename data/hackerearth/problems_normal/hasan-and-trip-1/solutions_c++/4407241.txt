#include <bits/stdc++.h>
using namespace std;
double x[3009];
double  y[3009];
double f[3009];
double arr[3009];
double distance(double, double, double, double);
int main()
{
	int n;
	scanf("%d", &n);
	int k, j;
	for (k = 0 ; k < n ; k++) {
		scanf("%lf %lf %lf", k + x, k + y, k + f);
	}
	arr[0] = f[0];
	double maxi;
	double trial;
	for (k = 1 ; k < n ; k++) {
		maxi = (double)INT_MIN;
		for (j = 0 ; j < k ; j++) {
			trial = arr[j] + f[k] - distance(x[k], y[k], x[j], y[j]);
			if (trial > maxi) {
				maxi = trial;
			}
		}
		arr[k] = maxi;
	}
	printf("%.6lf\n", arr[n - 1]);
	return 0;
}
double distance(double xd, double yd, double xs, double ys) {
	double ans = sqrt((xd - xs) * (xd - xs) + (yd - ys) * (yd - ys));
	return ans;
}
