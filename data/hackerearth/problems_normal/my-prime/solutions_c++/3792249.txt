#include <iostream>
using namespace std;

bool isdiv(int x, int y)
{
	int r = x/y;
	if(r*y-x == 0)
	return true;
	else
	return false;
}

int main()
{
    int x;
    cin >> x;
    int *v = new int[x];
    for(int i=0; i<x;i ++)
    	cin >> v[i];
    
    for(int i=0;i<x;i++)
    {
    	int c = 0;
    	for(int j=0;j<x;j++)
    	{
    		if(isdiv(v[i],v[j]))
    		c++;
    	}
    	if(c==1)
    	cout << v[i] << " ";
    }
    
    return 0;
}
