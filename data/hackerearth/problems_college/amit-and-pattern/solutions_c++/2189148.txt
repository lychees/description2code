#include <iostream>
using namespace std;

int main()
{
  int n,m,a[200][200];
  cin>>n>>m;
  for(int i=0;i<n;i++)
  for(int j=0;j<m;j++)
  cin>>a[i][j];
  int i=0,j=0;
	for(int k=0;k<((2*n)+(2*m));k++)  
	{
		j=i;
		i=0;
		while(j>=0&&i>=0)
		{
			if(j<m&&i<n)
			cout<<a[i][j]<<" ";
		i++;
		j--;
		}
		cout<<"\n";
		
	}
}
