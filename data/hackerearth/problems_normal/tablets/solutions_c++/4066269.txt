#include <iostream>
#include <algorithm>
#include <string.h>
#include <stdio.h>
#include <vector>
#include <math.h>

using namespace std;

int main()
{
    long int n;

    cin >> n;

    long int arr[n];

    for(long int i=0;i<n;i++)
    {
        cin >> arr[i];
    }

    if(n==1)
    {
        cout << 1;

        return 0;
    }

    long int ans[n];

    ans[0]=arr[0]<=arr[1]?1:-1;

    ans[n-1]=arr[n-1]<=arr[n-2]?1:-1;

    for(long int i=1;i<n-1;i++)
    {
        if((arr[i]<arr[i-1]&&arr[i]<arr[i+1]) || (arr[i]==arr[i-1]&&arr[i]==arr[i+1]) || (arr[i]<arr[i-1]&&arr[i]==arr[i+1]) || (arr[i]==arr[i-1]&&arr[i]<arr[i+1]))
        {
            ans[i]=1;
        }
        else
        {
            ans[i]=-1;
        }
    }

    for(long int i=1;i<n;i++)
    {
        if(arr[i]>arr[i-1])
        {
            ans[i]=ans[i-1]+1;
        }
    }

    for(long int i=n-2;i>=0;i--)
    {
        if(arr[i]>arr[i+1])
        {
            ans[i]=max(ans[i], ans[i+1]+1);
        }
    }

    long long int sum=0;

    for(long int i=0;i<n;i++)
    {
        sum+=(long long int)ans[i];
    }

    cout << sum;

    return 0;
}
