#include<iostream>
using namespace std;
main()
{
	int l,n,i;
	struct images
	{
		int w;
		int h;
	}img[999];
	scanf("%d",&l);
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
	scanf("%d",&img[i].w);
	scanf("%d",&img[i].h);
	}
    for(i=0;i<n;i++)
	{
		if(img[i].w>=l && img[i].h>=l && img[i].w!=img[i].h)
		printf("CROP IT\n");
		else if(img[i].w<l || img[i].h<l)
		printf("UPLOAD ANOTHER\n");
		else if(img[i].w>=l && img[i].h>=l && img[i].w==img[i].h)
		printf("ACCEPTED\n");
	}
   
    return 0;
}
