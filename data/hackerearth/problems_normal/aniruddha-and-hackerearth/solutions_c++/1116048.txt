#include <iostream>
#define N 1000001
#define MOD 1000000007
using namespace std;

long a[N], b[N];

int main()
{

	a[1]=1; b[1]=1;
	//cout<<"hi";
	for(long i=2;i<N;i++)
	{
		a[i]=(a[i-1]+b[i-1])%MOD;
		b[i]=a[i-1];
	}
	for(long i=1;i<N;i++)
		a[i]=(a[i]+b[i])%MOD;
	for(long i=2;i<N;i++){
		a[i]=(a[i]+a[i-1])%MOD;
		//cout<<a[i]<<endl;
	}
	long test, x;
	cin>>test;
	while(test--)
	{
		cin>>x;
		cout<<(a[x])%MOD<<" "<<endl;
	}
}
