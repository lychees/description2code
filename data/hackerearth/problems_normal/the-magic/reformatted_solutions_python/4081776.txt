def ConvertToBinary(decimal):
    mylist = list()
    while decimal > 1:
        rem = decimal % 2
        decimal = decimal/2
        mylist.append(rem)
    mylist.append(decimal)
    return sum(mylist)


T = int(input())
dayslist = list()
for i in range(0, T):
    N = int(input())
    days = ConvertToBinary(N)
    dayslist.append(days)
for i in dayslist:
    print i
