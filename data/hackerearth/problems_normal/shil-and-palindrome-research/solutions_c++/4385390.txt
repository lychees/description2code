void update(int i,char c,int val){
	while(i<=n){
		bit[i][c-'a']+=val;
		i+=i&-i;
	}
}

int get(int l,int pos){
	int sum=0;
	while(l>0){
		sum+=bit[l][pos];
		l-=l&-l;
	}
	return sum;
}
