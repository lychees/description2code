#include <iostream>
using namespace std;



void merge(int arr[],int beg,int mid,int end){
	int n=end-beg+1;
	int sol[n];
	int k=0;
	int i=beg;
	int j=mid+1;
	while(i<=mid && j<=end){
		if(arr[i]<=arr[j]){
			sol[k]=arr[i];
			k++;
			i++;
		}
		else if(arr[i]>arr[j]){
			sol[k]=arr[j];
			k++;
			j++;
		}
	}
	
	while(i<=mid){
		sol[k++]=arr[i++];
	}
	while(j<=end){
		sol[k++]=arr[j++];
	}
	k=0;
	for(int i=beg;i<=end;i++){
		arr[i]=sol[k++];
	}
}

void mergesort(int arr[],int beg,int end){
	if(beg<end){
	int mid=(beg+end)/2;
	mergesort(arr,beg,mid);
	mergesort(arr,mid+1,end);
	merge(arr,beg,mid,end);
	}
}

int main()
{
    int t;
    cin>>t;
    while(t>0){
    	int n;
    	cin>>n;
    	int arr[n];
    	for(int i=0;i<n;i++)
    	cin>>arr[i];
    	mergesort(arr,0,n-1);
    	for(int i=n-1;i>=0;i--)
    	cout<<arr[i]<<" ";
    	cout<<endl;
    	t--;
    }
    return 0;
}
