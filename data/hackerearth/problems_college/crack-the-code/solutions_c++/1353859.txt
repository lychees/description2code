#include <iostream>
using namespace std;
int fb[1000010];

void fibo()
{
	fb[0]=0;
	fb[1]=1;
	for(int i=2;i<=100001;i++)
	{
		fb[i]=fb[i-1]+fb[i-2];
	}
}
int main()
{
	int ts,a,b,cnt;
	fibo();
    cin>>ts;
    while(ts--)
    {
    	cin>>a>>b;
    	cnt=0;
    	for(int i=1;fb[i]<=b;i++)
    	{
    		if(fb[i]>=a) cnt++;
    	}
    	if(cnt%2) cout<<"INVALID"<<endl;
    	else cout<<"VALID\n";
    }
    return 0;
}
