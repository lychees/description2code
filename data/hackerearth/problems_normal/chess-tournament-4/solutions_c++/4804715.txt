int processArray(int **arr, int i, int j)
{
    //cout<<"1"<<"\n";
	if(i > j)
	    return arr[i - 2][j - 1] == 1 ? i : j;
    else
        return arr[j - 2][i - 1] == 1 ? j : i;
}

int getResult(int **arr, int round, int matchnum)
{
	if(round != 1)
	{
	    //cout<<"2"<<"\n";	    
	    int res1 = getResult(arr, round - 1, matchnum * 2 - 1);
	    //cout<<"R : "<<(round - 1)<<" matchnum : "<<( matchnum * 2 - 1)<<" RES : "<<res1<<"\n";
	    int res2 = getResult(arr, round - 1, matchnum * 2);
	    //cout<<"R : "<<(round - 1)<<" matchnum : "<<( matchnum * 2)<<" RES : "<<res2<<"\n";
		return processArray(arr, res1, res2);
	}
	else
	{
	    //cout<<"3"<<"\n";	    
	    int res = processArray(arr, matchnum * 2 - 1, matchnum * 2);
	    //cout<<"Num1 : "<<matchnum * 2 - 1<<" Num2 : "<<( matchnum * 2)<<" RES : "<<res<<"\n";
		return res;
	}
}
