#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
   int T,N,no,turns=0;
   vector <int> nos;
   scanf("%d", &T);
   while (T--) {
   		turns = 0;
   		scanf("%d", &N);
   		for (int i=0; i<N; i++) {
   			scanf("%d", &no);
   			nos.push_back(no);
   		}
   		sort(nos.begin(), nos.end());
   		for (int i=0; i<N; i++) {
   			turns++;
   			if (i==N-1)
   				continue;
   			if (nos[i]+1 == nos[i+1])
   				i++;
   		}
   		printf("%d\n",turns);
   		nos.clear();
   }
   return 0;
}
