#include<iostream>
#include<algorithm>
using namespace std;

int main()
{ int t,s[3];
 cin>>t;

 while(t--)
 {
     cin>>s[0]>>s[1]>>s[2];


     sort(s,s+3);
     
     if(s[2]-s[1]>s[1]-s[0])
     {
        cout<<s[2]-s[1]-1<<endl;
     }
     else
     {
        cout<<s[1]-s[0]-1<<endl;
     }

 }return 0;
}
