#include<bits/stdc++.h>
using namespace std;
/***********************************************/
/*      ____________
 *     /            \
 *    /  /\      /\  \
 *   /  /  \    /  \  \
 *   \                /

 *     \____________/
 */
const long long mod = 1000000007;
int cnt[1010];
pair<int,int> to[1010];
int main(){
	ios_base::sync_with_stdio(false);
	cin.tie(nullptr);

	int n,x;
	cin>>n;
	for(int i = 1;i <= n;i++){
		cin>>x,cnt[x]++;
		if(to[x].first == 0)to[x].first = INT_MAX;
		to[x].first = min(to[x].first,i);
		to[x].second = max(to[x].second,i);
	}
	int res = 0;
	for(int i = 0;i <= 1000;i++){
		if(cnt[i] == 0)continue;
		res = max(res,cnt[i]);
		int mn = INT_MAX,mx = -1;
		for(int j = i+1;j <= 1000;j++){
			if(cnt[j] == 0)continue;
			int cur = cnt[i] + cnt[j];
			if(mn <= min(to[i].first,to[j].first))cur++;
			if(mx >= max(to[i].second,to[j].second))cur++;
			res = max(res,cur);
			mn = min(mn,to[j].first);
			mx = max(mx,to[j].second);
		}
	}
	cout<<res<<'\n';
	return 0;
}
