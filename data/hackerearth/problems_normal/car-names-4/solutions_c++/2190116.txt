#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    while(n--)
    {
    	string s;
    	cin>>s;
    	int length=s.length();
    	char a,b,c;
    	int ca=0,cb=0,cc=0;
    	int i=1;
    	a=s[0];
    	ca++;
    	while(s[i]==a)
    	{
    		ca++;
    		i++;
    	}
    	b=s[i];
    	cb++;
    	i++;
    	while(s[i]==b)
    	{
    		cb++;
    		i++;
    	}
    	c=s[i];
    	cc++;
    	i++;
    	while(s[i]==c)
    	{
    		cc++;
    		i++;
    	}
    	if(a!=b && b!=c && a!=c && ca==cb && cb==cc && i==length)
    		cout<<"OK"<<endl;
    	else
    		cout<<"Not OK"<<endl;
    }
}
