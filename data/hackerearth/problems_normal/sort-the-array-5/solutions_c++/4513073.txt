#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
vector<int> v[10000000];
vector<int> k;
int main()
{
    int n,b;
    int ma=-1;
    cin>>n>>b;
    int a[n];
    for(int i=0;i<n;i++)
    {
        cin>>a[i];
        int m=a[i]%b;
        v[m].push_back(a[i]);
        if(ma<m){
        	ma=m;
		}
    }
/*	for(int i=0;i<=ma;i++){
		for(int j=0;j<v[i].size();j++){
			cout<<v[i][j]<<" ";
		}
		cout<<endl;
	}
	*/
    for(int i=b-1;i>=0;i--)
    {
        k.clear();
        for(int j=0;j<v[i].size();j++)
        {
            k.push_back(v[i][j]);
        }
        sort(k.begin(),k.end());
        for(int j=0;j<v[i].size();j++)
        {
            cout<<k[j]<<" ";
        }
    }
}
