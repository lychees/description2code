#include <iostream>
using namespace std;
 
 
int main()
{
	long arr[]={0,1,2,6,6,3};
	long n,a,b,sum;
	cin>>n;
	
	for(long i=0;i<n;i++){
		cin>>a>>b;
		sum=0;
		if(b<6){
			for(long j=a;j<=b;j++){
				sum=sum+arr[j];
			}
		}
		else{
			if(a<6){
				for(long j=a;j<6;j++){
					sum=sum+arr[j];
				}
			a=6;
			}
				for(long j=a;j<=b;j++){
					sum=sum+9;
				}
		}
		cout<<sum<<'\n';
	}
    return 0;
}
