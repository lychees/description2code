#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int a[n];
    	int ctr0 = 0, ctr1 = 0;
    	
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    		if(a[i]==0)
    		ctr0++;
    	}
    	ctr1 = n-ctr0;
    	if(ctr1%2==0)
    	cout<<ctr0<<endl;
    	else
    	cout<<ctr1<<endl;
    }
    return 0;
}
