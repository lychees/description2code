#include<iostream>
#include<cstring>
using namespace std;
	
 int dlt(string s)
	{ 
	int count=0;
		for(int i=0;i<s.length()-1;i++)
		{
		
		if(s[i]==s[i+1])
		{
			count++;
		}	
		}
		return count;
		
		
	}


int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string s;
		cin>>s;
		
	cout<<dlt(s)<<endl;	
		
		
	}
}
