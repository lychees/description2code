#include<iostream>
#include<cstdio>
#include<cmath>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        double h,d,v,temp,ans;
        cin>>h>>d>>v;
        temp=sqrtl(d*d-h*h);
        ans=(temp/h)*v;
        printf("%.9lf\n",ans);
    }
    return 0;
}
