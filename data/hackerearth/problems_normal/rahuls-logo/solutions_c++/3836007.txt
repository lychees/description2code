#include<iostream>
using namespace std;
void patternA(int value)
{
	for(int i=0;i<value;i++)
	{
		for(int j=0;j<2*value;j++)
		{
			if((j-i)>value) 
			break;
			if((i+j)%2!=(value%2) && (i+j)>=(value-1)&&j<value) 
			cout<<"/";
			else if((j-i)%2==(value%2) && (j-i)<=(value)&&j>=value) 
			cout<<"\\";
			else 
			cout<<" ";
		}
		cout<<endl;
	}
}
void patternV(int value)
{
	for(int i=0;i<value;i++)
	{
		cout<<"  ";
		for(int j=0;j<(2*value);j++)
		{
			 if(i+j>(2*value)-1) 
			 break;
			 if((j-i)%2==0 && (j-i)>=0 && j<value) 
			 cout<<"\\";
			 else if((i+j)%2==1&&(i+j)<(2*value)&&j>=value) 
			 cout<<"/";
			 else 
			 cout<<" ";
		}
		cout<<endl;
	}
}

int main()
{
	int n;
	cin>>n;
	patternA(n);
	patternV(n);
	patternA(n+1);
	patternV(n+1);
	return 0;
}
