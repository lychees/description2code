long long dict[26]= { 0 };

bool checkIfAlpha(char c)
{
	if( c >= 'a' && c <= 'z'  )
		return true;
	return false;
}

{
string temp="";
unsigned long long x=0;

for(long long i=1;i<strInput.size();i++)
{

	{

		 dict[p-'a'] += x;
