#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
   int test;
long int M,cap[100000],count;

cin>>test;

while(test)
{
count=0;

cin>>M;


for(int i=0;i<M;i++)
cin>>cap[i];

sort(cap,cap+M);

for(int i=0;i<M;i++)
{
while(i<M-1 && cap[i]==cap[i+1])
i++;

count++;
}
cout<<count<<"\n";



test--;
}
    return 0;
}
