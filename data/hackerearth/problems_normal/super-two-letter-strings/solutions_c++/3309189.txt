#include <iostream>
using namespace std;
#define MOD 1000000007
long long int dp[20000];
int n,p;

long long int solve(int len,int i){
	
//	cout<<s<<" ";
	if(len>=n)
		return 1;
	long long int ans=0;
	if(i<p-1){
		ans+=solve(len+1,i+1);
		ans=ans%MOD;
	}	
	long long int temp=0;
	if(dp[n-len]!=-1)
	  ans+=dp[n-len];
	else{
	     temp=solve(len+1,0);
		 dp[n-len]=temp%MOD;
		 ans+=temp;
	}
	ans=ans%MOD;
	return ans;
}

int main()
{
    int test;
    cin>>test;
    while(test--){
        
        
        dp[1]=1;
    	cin>>n>>p;
    	for(int i=0;i<n;i++)
        	dp[i]=-1;
    	cout<<solve(1,0)<<endl;
    }
    return 0;
}
