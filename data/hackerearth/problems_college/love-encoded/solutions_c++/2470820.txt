#include <iostream>
#include<string>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	int n;
    	cin>>n;
    	string x;
    	cin>>x;
    	for(int j=0;j<x.length();j++)
    	{
    		int y=(int)(x[j]) - n;
    		if(y<65)
    		y=y+26;
    		x[j]=(char)y;
    	}
    	cout<<x<<'\n';
    }
    return 0;
}
