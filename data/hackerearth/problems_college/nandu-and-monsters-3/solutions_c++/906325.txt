#include <stdio.h>
int main(){
	int n,q,l,i,j,x,y,max,min,lca,f;
	scanf("%d %d",&n,&q);
	char a[n+1],k1,b;
	getchar();
	for(i=1;i<=n;i++) {
	 scanf("%c",&k1);
	 a[i]=k1;
	}
	a[i+1]='\0';
	while(q--){
		scanf("%d %d",&x,&y);
		if(a[x]=='1' || a[y]=='1') printf("No\n");
		else if(x==y) {
			if(a[x]=='1')puts("No");
			else puts("Yes");
		}
		else {
			if(x>y) {
				max=x;min=y;
			}
			else {
				max=y;min=x;
			}
			x=max;y=min;
			int z[max+5];
			for(i=0;i<=max+1;i++)z[i]=0;
			while(min>=1) {
				z[min]=1;
				min/=2;
			}
			while(max>=1&&z[max]!=1) {
				max/=2;
			}
			lca=max;
			f=0;
			max=x;min=y;
			while(y>=lca&&a[y]!='1') {
				min=y;
				y/=2;
			}
			y=min;
			while(x>=lca&&a[x]!='1') {
				max=x;
				x/=2;
			}
			x=max;
			if(x==y)puts("Yes");
			else puts("No");
		}
	}
	return 0;
}
