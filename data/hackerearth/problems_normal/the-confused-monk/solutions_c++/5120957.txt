#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
long long int a[55];
long long int gcd(long long int a,long long int b){

        if(b==0 ) return a;
        else
        return gcd(b,a%b);

}
long long int modular_expo(long long int e,long long int b){
    long long int ans=1;
    while(b>0){
        if(b&1)
        {
            ans*=e;
            ans=ans%mod;

        }
        e=(e*e)%mod;
        b>>=1;

    }

return (ans%mod);
}

int main(){
ios_base::sync_with_stdio(0);
int n,i;
long long int mul,gcds;
cin>>n;
mul=1;
for(i=0;i<n;i++)
{
    cin>>a[i];
    mul*=a[i];
    mul%=mod;

}
gcds=a[0];
for(i=1;i<n;i++)
    gcds=gcd(gcds,a[i]);
cout<<modular_expo(mul,gcds)<<endl;
return 0;
}
