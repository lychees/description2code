#include<iostream>
using namespace std;
#define M -1000000003

int dp[203][203];
int main()
{
	int t;
	ios_base::sync_with_stdio(false);
	cin >> t;
	
	while(t--)
	{
		int n,m;
		cin >> n >> m;
		int a[n+1][m+1];
		for(int i=1;i<=n;i++){
			for(int j=1;j<=m;j++){
				char c;
				cin >> c;
				if(c=='T')
					a[i][j]=150;
				else if(c=='D')
					a[i][j]=-100;
				else
					a[i][j]=0;		
			}
		}
		
		for(int i=0;i<=n;i++)
			dp[i][0]=M;
		for(int i=0;i<=m;i++)
			dp[0][i]=M;
		dp[0][1]=0;
		
		for(int i=1;i<=n;i++){
			for(int j=1;j<=m;j++){
				dp[i][j]=a[i][j] + max(dp[i-1][j],dp[i][j-1]);
				if(dp[i][j] < 0)
					dp[i][j]=M;
			}
		}
		if(dp[n][m] < 0)
			dp[n][m]=-1;
			
		cout << dp[n][m] << endl;				
	}
	return 0;
}
