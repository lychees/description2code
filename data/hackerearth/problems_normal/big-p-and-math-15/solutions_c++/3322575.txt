#include<iostream>
#include<cmath>
using namespace std;

long long int convert_f(long long int a)
{
	long long int temp=0, i=0, x=a;
	while(a)
	{
		if(a%10==6)
			temp-=pow(10, i);
		a/=10;
		i++;
	}
	return x+temp;
}

long long int convert_s(long long int a)
{
	long long int temp=0, i=0, x=a;
	while(a)
	{
		if(a%10==5)
			temp+=pow(10, i);
		a/=10;
		i++;
	}
	return x+temp;
}

int main()
{
	long long int a , b;
	cin>>a>>b;
	cout<<convert_f(a)+convert_f(b)<<" "<<convert_s(a)+convert_s(b)<<endl;
	return 0;
}
