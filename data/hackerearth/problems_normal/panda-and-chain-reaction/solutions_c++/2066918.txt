#include<bits/stdc++.h>
#define ull unsigned long long
#define ld long double
#define ll long long
#define M 1000003
#define N 6e6+10
#define pll pair<ll,ll>
#define f(i,n) for(i=0;i<n;i++)
#define mll map<ll,ll>
#define mset(a) memset(a,0,sizeof(a));
#define foe(i,j,k) for(i=j;i<=k;i++)
#define foi(i,j,k) for(i=j;i<k;i++)
#define dbg(x) cout<<#x<<"="<<x<<endl;
 
 using namespace std;

   int main()
   {
    ull dp[100007],n,x,ans,i,t;
    dp[1]=1;dp[0]=1;
    foe(i,2,100006)
    {
      dp[i]=((i%M)*(dp[i-1]%M))%M;
    }
    cin>>t;
    while(t--)
    {cin>>n>>x;
    if(n>M)cout<<"0"<<endl;
    else
    {
    ans=(dp[n]*x)%M;
    cout<<ans<<endl;
    }
   }
  return 0;
   }
