using namespace std;
int max(int a,int b)
{
    return(a>b)?
    a:b;
}
 
int gcd2 (int n1,int n2) 
{
    return (n2 == 0) ? n1 : gcd2 (n2, n1 % n2);
}
 
int gcd(int a,int b,int c)
{
    return gcd2(a,gcd2(b,c));
}
 
//static int arr[100][100][100];
int main()
{//20 digits limit
    int co[101]={0};
    for (int i=1;i<=100;i++)
    {
        for(int j=1;j<=100;j++)
        {   if(i==j)continue;
            for (int k=1;k<=100;k++)
            {
                if(i==k||j==k)continue;
                co[gcd(i,j,k)]++;
               // cout<<arr[i][j][k]<<endl;
            }
    }
    }
 
    int t;
  //  unsigned long long l;
 
    cin>>t;
    while(t--)
    {
    unsigned long long x,y,z;
    cin>>x>>z;
    if(x>z)y=x-z;
    else y=z-x;
    //cout<<y;
    double ca=0,cb=0;
 
        for(int i=1;i<=100;i++)
        {
            if(y%i==0)ca+=co[i];
            else cb+=co[i];
        }
 
  //  cout<<y;

    double r=ca/(ca+cb);
  // cout<<ca<<endl<<cb<<endl<<count;
    printf("%.6f\n",r);
    }
 return 0;
}
