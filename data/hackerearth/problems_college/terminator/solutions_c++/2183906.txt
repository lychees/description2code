#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	char c[105];
    	scanf("%s",c);
    	int i,j,len=strlen(c);
    	for(i=0;i<len;i++)
    	{
    		if(c[i]=='x')
    		{
    			for(j=i+1;j<len;j++)
    			{
    				if(c[j]=='*')
    				{
    					break;
    				}
    				else if(c[j]=='o')
    				{
    					c[j]='a';
    				}
    			}
    			for(j=i-1;j>=0;j--)
    			{
    				if(c[j]=='*')
    				{
    					break;
    				}
    				else if(c[j]=='o')
    				{
    					c[j]='a';
    				}
    			}
    		}
    	}
    	for(i=0;i<len;i++)
    	{
    		if(c[i]!='a')
    		{
    			printf("%c",c[i]);
    		}
    	}
    	printf("\n");
    }
    return 0;
}
