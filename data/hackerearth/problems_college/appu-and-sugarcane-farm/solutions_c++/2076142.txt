#include <iostream>
#include<vector>
using namespace std;

int main()
{
    int n;
    cin>>n;
    vector<long long int > a(n);
    for(int i=0;i<n;i++)
    cin>>a[i];
    
    long long int final=a[0];
    while(final%2==0)
    final/=2;
    bool flag=true;
    for(int i=1;i<n;i++)
    {
    	long long int cur=a[i];
    	while(cur%2==0)
    	cur/=2;
    	
    	if(final !=cur)
    	{
    		flag=false;
    		break;
    	}
    	
    	
    }
    
    if(flag)
    cout<<"YES\n";
    else
    cout<<"NO\n";
    return 0;
}
