#include<iostream>


using namespace std;
pair<pair<int, int>, int> g[1002];


int MIN(int a, int b){
    if(a < b) return a;
    return b;
}

int dp[102] = {0};

void init(){
    for(int i=0;i<=100;i++){
        dp[i] = 1000000000;
    }
}

int main(){
    int t;
    cin >> t;
    while(t--){
        int n, m;
        cin >> n >> m;
        for(int i=0;i<m;i++){
            int a, b, c;
            cin >> a >> b >> c;
            g[i] = make_pair(make_pair(a, b), -c);
        }
        init();
        dp[1] = 0;
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                int a = g[j].first.first, b = g[j].first.second, c = g[j].second;
                if(dp[a] != 1000000000)
                dp[b] = MIN(dp[b], dp[a]+c);
            }
        }
        bool check = true;
        for(int i=0;i<m;i++){
            int a = g[i].first.first, b = g[i].first.second, c = g[i].second;
            if(dp[a] != 1000000000)
            if(dp[b] > dp[a]+c){
                cout << "Yes" << endl; check = false;
                break;
            }
        }
        if(check) cout << "No" << endl;
    }
}
