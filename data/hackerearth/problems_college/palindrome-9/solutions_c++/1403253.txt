vector<string> split(const string &s, char c) {
	vector<string> v;
	stringstream ss(s);
	string x;
	while (getline(ss, x, c)) v.push_back(x);
	return v;
}

#define error(args...) { vector<string> _v = split(#args, ','); err(_v.begin(), args); }

	cerr << it -> substr((*it)[0] == ' ', it -> length()) << " = " << a << '\n';
	err(++it, args...);
}

int a, b;
map<ll, int> mp;

inline ll calc(int x) {
	return (x * 1LL * (x + 1) * (x + x + 1)) / 6;
}

bool check(ll x) {
	vector<int> d;
	while (x) {
		d.push_back(x % 10);
		x /= 10;
	}
	int n = d.size() - 1;
	rep2(i, 0, n) if (d[i] != d[n - i]) return false;
	return true;
}

int main(int argc, char *argv[]) {
    ios_base::sync_with_stdio(false);
    cin >> a >> b;
    int sb = (int) sqrt(b);
    int sa = (int) sqrt(a);
    ll ans = 0;
    rep2(x, 2, sb) rep1(y, 0, x - 1) {
    	ll tmp = calc(x) - calc(y);
    	if (tmp < a || tmp > b) continue;
    	if (check(tmp) && !mp[tmp]) {
    		ans += tmp;
    		mp[tmp] = 1;
    	}
    }
    printf("%lld\n", ans);
    return 0;
}
