#include<bits/stdc++.h>
#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	for(int j = 0; j < t; j++){
		int n;
		cin>>n;
		long int a[n];
		long int d[n];
		for(int i = 0;i < n; i++){
			cin>>a[i];
		}
		for(int i = 0; i < n; i++){
			long int x;
			cin>>x;
			d[i] = x + a[i];
		}
		vector<pair<long int , int> > v;
		for(int i = 0; i < n; i++){
			v.push_back(make_pair(a[i],1));
		}
		for(int i = 0; i < n; i++){
			v.push_back(make_pair(d[i],0));
		}
		sort(v.begin(),v.end());
		int m  = v.size();
		int count = 0,tot = INT_MIN;
		for(int i = 0; i < m; i++){
			if(v[i].second == 1){
				count++;
			}
			if(v[i].second == 0){
				count--;
			}
			if(count > tot){
				tot = count;
			}
		}
		cout<<tot<<endl;
	}
    return 0;
}
