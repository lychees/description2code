#include <bits/stdc++.h>
using namespace std;

bool isPalindrom(string str){
	int len = str.length();
	int l = 0, r = len-1;
	while(l<r){
		if(str[l]!=str[r]) return false;
		l++;
		r--;
	}
	return true;
}

int main(){
	string str;
	cin>>str;
	int len = str.length(), ans=0;
	for(int i=0; i<len; i++){
		for(int j=i; j<len; j++){
			if(isPalindrom(str.substr(i, j-i+1))) ans++;
		}
	}
	cout<<ans<<endl;
	return 0;
}
