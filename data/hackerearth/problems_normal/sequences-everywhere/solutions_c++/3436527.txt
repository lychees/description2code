long long gcd(long long a, long long b) {
	if(a > b) swap(a,b);
	return (a == 0)?b:gcd(b%a,a);}

int main() {
	cin.sync_with_stdio(0);
	cin.tie(0);
	cout << fixed << setprecision(10);
	long long A,B;
	cin >> A >> B;
	vector<int> len(1000,0);
	for(int i =1; i < 1000; i++) {
		int n =0;
		for(int j =1; j < i; j++) if(i%j != 0) {
			n =j;
			break;}
		if(n == 0) len[i] =1;
		else len[i] =len[n]+1;}
	long long ans =0, lcm =1;
	while(A < 1000 && A <= B) {
		ans +=len[A];
		A++;}
	if(A > B) {
		cout << ans << "\n";
		return 0;}
	ans +=B-A+1;
	for(int i =2; i < 1000; i++) {
		long long p =B/lcm-(A-1)/lcm;
		long long lcm_nw =lcm/gcd(lcm,i);
		if(lcm_nw <= B/i) lcm_nw *=i;
		else lcm_nw =B+1;
		p -=B/lcm_nw-(A-1)/lcm_nw;
		ans +=p*len[i];
		lcm =lcm_nw;
		if(lcm > B) break;}
	cout << ans << "\n";
	return 0;}
