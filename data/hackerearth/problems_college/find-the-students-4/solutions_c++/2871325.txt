#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n;
	cin >> n;
	vector<int>v(n);
	map<int, int>mp;
	for(int i=0;i<n;i++) cin >> v[i], mp[v[i]]++;
	sort(v.begin(),v.end());
	cout << n << "\n";
	
	for(map<int, int>::iterator it = mp.begin();it!=mp.end();it++)
	{
		n -= it->second;
		if(n!=0)
		cout << n << "\n";
	}
    return 0;
}
