#include <iostream>
using namespace std;

int main() {
	int sampleCount;
	cin >> sampleCount;
	int outputCount;
	int tempSample;
	for(int i = 0; i < sampleCount; i++) {
		outputCount = 0;
		cin >> tempSample;
		for(int j = 1; j < tempSample; j++) {
			for(int k = j + 1; k <= tempSample; k++) {
				if((j ^ k) <= tempSample) {
					outputCount++;
				}
			}
		}
		cout << outputCount << endl;
	}
	return 0;
}
