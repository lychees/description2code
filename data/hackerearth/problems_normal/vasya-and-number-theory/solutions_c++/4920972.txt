void fs(int &x)
{
    register int c = gc();
    x = 0;
    int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}

int a[1000], dp[1000];
int n;

int cal(int idx)
{
    if(idx==n-1) return dp[idx]=1;
    if(dp[idx]!=-1) return dp[idx];
    int ans=1;
    for(int i=idx+1; i<n; i++)
    {
        int tmp=1;
        if(a[i]%a[idx]==0) tmp+=cal(i);
        ans=max(ans,tmp);
    }
    return dp[idx]=ans;
}

int main()
{
    ios_base::sync_with_stdio(0);
    cin>>n;
    rep(i,n) cin>>a[i];
    sort(a,a+n);
    ms1(dp);
    rep(i,n) cal(i);
    int ans=1;
    rep(i,n) ans=max(ans, dp[i]);
    if(ans<2) cout<<-1<<endl;
    else cout<<ans<<endl;
    return 0;
}
