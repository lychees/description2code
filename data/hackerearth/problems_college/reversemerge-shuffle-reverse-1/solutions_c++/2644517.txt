#include<stdio.h>
#include<algorithm>
using namespace std;
int b[265];

char a[100005];
int m,n,pos;
int valid(int m){
    int has = 1;
    int ne [26];
    int i;
    for(i=0;i<26;i++)ne[i] = 0;
   if(b[m]==0)return 0;
    b[m] --;
    int k;
    for(k=pos;a[k];k++)if(a[k] == 'a' + m)break;
        for(i=k+1;i<n;i++)ne[a[i]-'a']++;
        for(i=0;i<26;i++)if(ne[i]<b[i])has = 0;
        if(has==0){
        b[m]++;
        return 0;
    }
    pos = k+1;
    return 1;
}
int main()
    {
    while(scanf("%s",&a)==1){
    int i,j;
        pos = 0;
    for(i=0;i<26;i++)b[i] = 0;
    for(i=0;a[i];i++)b[a[i]-'a'] ++;
        n = i;
        reverse(a,a+n);
      m = n/2;
    for(i=0;i<26;i++)
        b[i] /=2;
        for(i=0;i<m;i++)
            {
            for(j=0;j<26;j++){
                if(valid(j)){
                    printf("%c",j+'a');
                    break;
                }
            }
            
        }
        printf("\n");
        
    }
}
