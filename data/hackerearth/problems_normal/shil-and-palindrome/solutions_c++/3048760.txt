#include <iostream>
#include<string>
#include<cstring>
using namespace std;

int main()
{
    string in;
    cin>>in;
    int len = in.length();
	int * freq = new int[26];
	for(int i=0;i<26;i++)
		freq[i] = 0;
	for(int i=0;i<len;i++){
		freq[in[i]-'a']++;
	}
	char* out1 = new char[100000];
	char*out2 = new char[100000];
	int mid;
	bool flag=0;
	int idx = 0;
	for(int i=0;i<26;i++){
		if(freq[i]%2){
			if(len%2 == 0){
				cout<<"-1\n";
				return 0;
			}
			else{	
				if(flag){
					cout<<"-1\n";
					return 0;
				}
				else{
					flag = 1;
					mid = i;
				}
			}
		}
		for(int j=0;j<freq[i]/2;j++){
    			out1[idx++] = 'a'+i;
				out2[len/2 - idx] = 'a'+i;
    			
    	}
	}
	if(len%2)
		out1[idx] = 'a'+mid;
	strcat(out1,out2);
	cout<<out1<<endl;
}
