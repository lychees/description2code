    #include <iostream>

    bool checkPalin(string s){
    int n = s.length();
    bool flag = true;
    for(int i = 0 ; i < n/2 ; i++)
    {
    if(s[i] != s[n-i-1])
    {
    flag = false;
    break;
    }
    }
    return flag;
    }
     
    int main()
    {
    long int t,m,n;

    bool flag = true;

    for(int i = 0; i <= m; i++)
    {
    string s1="",s="",s3="";
    s1 = a.substr(0,i);
    if(i!=m)
    s3 = a.substr(i);
    s = s1+b+s3;
