#include <bits/stdc++.h>
#define ll long long int
 
using namespace std;
 
ll fb(ll n)
{
	if(n == 1) 
		return 2;
	else if(n == 2)
		return 3;
	else 
		return fb(n-1) + fb(n-2);
}
int main()
{
	ll test;
	cin >> test;
	while(test--) {
		ll n, i;
		cin >> n;
		i = fb(n);
		cout << i * i << endl;
	}
    return 0;
}
