#include <iostream>
#include<limits.h>
#define MAX 501
using namespace std;
 
int T;
int F;
int E;
int DP[MAX][MAX];

int main(){
	int a, b;
    cin >> T;
    
    for(int j=1; j<=MAX; j++){
	    	DP[1][j]=j;
	    }
	    
	    for(int i=1; i<=MAX; i++){
	    	DP[i][0]=0;
	    	DP[i][1]=1;
	    }
	    
	    int maxTrial;
	    for(int e=2; e<=MAX-1; e++){
	    	for(int f=2; f<=MAX-1; f++){
	    		
	    		DP[e][f]=INT_MAX;
	    		for(int k=1; k<=f; k++){
	    			a = DP[e-1][k-1];
	    			b = DP[e][f-k];
	    			maxTrial = a > b ? a : b;
	    			
	    			DP[e][f] = maxTrial < DP[e][f] ? 1 + maxTrial : DP[e][f];
	    		}
	    	}
	    }
	    
    for(int ct=1; ct<=T; ct++){
	    cin >> F >> E;
	    cout<<DP[E][F]<<endl;
    }
    return 0;
}
