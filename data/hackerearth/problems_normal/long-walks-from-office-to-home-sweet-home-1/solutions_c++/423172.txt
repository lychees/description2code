#include <iostream>
using namespace std;
#define gc getchar_unlocked
#define mod  1000000007
void scanint(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}
void scanlong(long long &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}


long long ar[51][51];


void pow(int n,long long k)
{
	if(k==1)
	{
		return ;
	}
	long long br[51][51]={0},c[51][51]={0};
	if(k&1)
	{
		for(int i=0;i<n;i++)
			for( int j=0;j<n;j++)
				br[i][j] = ar[i][j];
	}
	for(int i=0;i<n;i++)
		{
			for( int j=0;j<n;j++)
			{
				c[i][j] = 0;
				for(int l =0;l < n ;l++)
				{
					c[i][j] = (c[i][j] + ar[i][l] * ar[l][j]) % mod;
				}
			}
		}
	for(int i=0;i<n;i++)
		{
			for( int j=0;j<n;j++)
				ar[i][j] = c[i][j];
		}
	pow(n,k/2);
	if(k&1)
	{
		for(int i=0;i<n;i++)
			for( int j=0;j<n;j++)
			{
				c[i][j] = 0;
				for(int l =0;l < n ;l++)
				{
					c[i][j] = (c[i][j] + ar[i][l] * br[l][j]) % mod;
				}
			}
		for(int i=0;i<n;i++)
			for( int j=0;j<n;j++)
				ar[i][j] = c[i][j];
	}
	
}
int main() {
	int t,n,u,v;
	long long k;
	scanint(t);
	
	while(t--)
	{
		scanint(n);
		scanlong(k);
		scanint(u);
		scanint(v);
		for(int i = 0;i < n;i++)
		{
			for(int j = 0;j < n;j++)
			{
				scanlong(ar[i][j]);
			}
		}
		pow(n,k);
		cout<<ar[u][v]<<endl;
	}
	
	return 0;
}
