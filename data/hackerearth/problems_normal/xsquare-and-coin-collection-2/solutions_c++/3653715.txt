#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t, n, p;
	cin>>t;
	while(t--)
	{
		cin>>n>>p;
		long long int a[n+1], dp[n+1], i, ma;
		for(i=0;i<n;i++)
		{
			cin>>a[i];
		}
		dp[0]=(a[0]<=p)?a[0]:0;
		ma=dp[0];
		for(i=1;i<n;i++)
		{
			if(a[i]<=p)
				dp[i]=a[i]+dp[i-1];
			else
				dp[i]=0;
			ma=max(ma, dp[i]);
		}
		cout<<ma<<endl;
	}
	return 0;
}
