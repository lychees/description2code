#include <bits/stdc++.h>
using namespace std;
const int mod=1000000007;
int dp[11000];
int main()
{
    int t,n,p,i,j;
    cin>>t;
    while(t--)
    {
    	cin>>n>>p;
    	memset(dp,0,sizeof(dp));
    	dp[0]=1;
    	for(i=1;i<=n;i++)
    	{
    		for(j=i-1;j>=0&&(i-j)<=p;j--)
    		{
    			dp[i]+=dp[j];
    			dp[i]%=mod;
    		}
    	}
    	cout<<dp[n]<<"\n";
    }
    return 0;
}
