#include <bits/stdc++.h>
using namespace std;

#define MOD 1000000007LL
#define ll long long

#define MAXN 1010
int visit[MAXN][MAXN], A[MAXN][MAXN];

int main()
{
	cin.tie(0);
	ios::sync_with_stdio(false);
	int t,n,m;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d",&n,&m);
		for(int i=1;i<=n;++i)
			for(int j=1;j<=m;++j)
				scanf("%d",&A[i][j]);
		memset(visit,0,sizeof visit);
		vector<int> comp;
		for(int i=1;i<=n;++i)
			for(int j=1;j<=m;++j)
				if(!visit[i][j] and A[i][j]==1)
				{
					visit[i][j] = true;
					int cnt = 1;
					queue<pair<int,int> > q;
					q.push({i,j});
					while(!q.empty())
					{
						pair<int,int> p = q.front();
						q.pop();
						int x = p.first,
							y = p.second;
						for(int dx=-1;dx<=1;++dx)
							for(int dy=-1;dy<=1;++dy)
								if((abs(dx)+abs(dy))!=0)	
								{
									int nx = x+dx,
										ny = y+dy;
									if(1<=nx and nx<=n and 1<=ny and ny<=m and !visit[nx][ny] and (A[nx][ny]==1))
									{
										visit[nx][ny] = 1;
										++cnt;
										q.push({nx,ny});
									}
								}
					}
					comp.push_back(cnt);
				}
		if(comp.size())
			cout << comp.size() << " " << *max_element(comp.begin(),comp.end()) << endl;
		else
			cout << "0 0" << endl;
	}
	return 0;
}
