#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,n,i,length,zeros,ones,temp1,temp2,ans;
	cin>>t;
	while(t--)
	{
		cin>>n;
		char str[n+1];
		char a[n+1],b[n+1];
		cin>>str;
		length=strlen(str);
		zeros=0;
		ones=0;
		temp1=0;
		temp2=0;
		for(i=0;i<length;i++)
		{
			if(str[i]=='0')
				zeros++;
			else
				ones++;
		}
		for(i=0;i<zeros;i++)
			a[i]='0';
			
			
		for(i=zeros;i<length;i++)
			a[i]='1';
			
			
		for(i=0;i<ones;i++)
			b[i]='1';
			
			
		for(i=ones;i<length;i++)
			b[i]='0';


		for(i=0;i<length;i++)
			if(str[i]!=a[i])
				temp1++;
				
				
		for(i=0;i<length;i++)
			if(str[i]!=b[i])
				temp2++;
				
				
		temp1=temp1/2;
		temp2=temp2/2;
		ans=min(temp1,temp2);
		cout<<ans<<endl;
	}
	return 0;
}
