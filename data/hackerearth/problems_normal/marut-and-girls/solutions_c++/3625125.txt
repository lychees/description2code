#include<stdio.h>
#include<bits/stdc++.h>
int arr[1000001];
int n;
 
int main()
{
	int i, girls, data, milgayi, accept, c;
	accept=0;
	memset(arr, 0, 1000001*sizeof(int));
	
    scanf("%d", &n);
    for(i=0; i<n; i++)
    {
    	scanf("%d", &data);
    	arr[data]=1;
    }
 
    scanf("%d", &girls);
    for(i=0; i<girls; i++)
    {
    	data=0;
    	milgayi=0;
 
		do
		{
			scanf("%d", &data);
			milgayi += arr[data];
			c = getchar();
			if(c == 10 || c == EOF)
				break;
		}while(1);
 
    	if(milgayi>=n)
    		accept++;
    }
    printf("%d\n", accept);
    
    return 0;
}
