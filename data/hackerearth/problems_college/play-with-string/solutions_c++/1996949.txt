#include <cstdio>
#include <cstring>
#include <string>
#include <iostream>
#include <algorithm>
using namespace std;

char s[10000000];
int t, n, flag;

int main(){
    scanf("%d", &t);
    while (t--){
    	scanf("%s", s);
    	n = strlen(s);
    	flag = 1;
    	for (int i = 1, j = n - 2; i <= j; i++, j--){
    		if (abs(s[i] - s[i - 1]) != abs(s[j] - s[j + 1])){
    			flag = 0;
    			break;
    		}
    	}
    	printf("%s\n", flag ? "Funny" : "Not Funny");
    }
    return 0;
}
