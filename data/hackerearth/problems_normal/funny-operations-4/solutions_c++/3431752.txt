#include<bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ll t;
	cin>>t;
	while(t--) {
		ll n;
		cin>>n;
		ll a[n];
		for(int i=0;i<n;i++)
			cin>>a[i];

		ll ans = -1, len = 0;
		for(int i=0;i<(1<<n);i++) {
			ll maxXor = 0;
			ll maxAnd = (1LL<<40) - 1;
			for(int j=0;j<n;j++) {
				if(i&(1<<j)) {
					maxXor ^= a[j];
					maxAnd &= a[j];
				}
			}
			if(maxAnd != (1LL<<40) - 1)	{
				if(ans < (maxXor|maxAnd)) {
					ans = (maxXor|maxAnd);
					len = 1LL*__builtin_popcount(i);
				}
				else if(ans == (maxXor|maxAnd)) {
					len = max(len, 1LL*__builtin_popcount(i));
				}
				
			}
		}
		cout<<ans<<" "<<len<<"\n";
	}
    return 0;
}
