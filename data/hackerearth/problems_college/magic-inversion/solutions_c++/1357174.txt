using namespace std;

#define MAX 1000010
#define ll long long
const ll oo = 1000000;

int is_prime(int res)
{
    if(res<2){return 0;}
    else if(res==2){return 1;}
    else if(res%2==0){return 0;}
    else
    {
        for(int i=3;i<=sqrt(res);i++)
        {
            if(res%i==0)
            {
                return 0;
            }
        }
        return 1;
    }
}

int a[MAX], b[MAX], f[MAX];
int n;

void set(int idx){
	while (idx <= oo){
		f[idx]++;
		idx += idx & (-idx);
	}
}

ll get(int idx){
	ll res = 0;

	while (idx > 0){
		res += f[idx];
		idx -= idx & (-idx);
	}

	return res;
}

int main(){
	scanf("%d", &n);
	for (int i = 0; i < n; i++)
		scanf("%d", &a[i]);
	for (int i = 0; i < n; i++)
		scanf("%d", &b[i]);

	ll res = 0;
	set(b[n - 1]);
	for (int i = n - 2; i >= 0; i--){
		res += get(a[i] - 1);
		set(b[i]);
	}

	if(is_prime(res)){cout<<"MAGIC INVERSION";}
	else {cout<<"SIMPLE INVERSION";}

	return 0;
}
