#include"bits/stdc++.h"
using namespace std;
#define ll long long int
#define mod 1000000007
ll poww(ll n,ll k)
{
	
	if(k==0) return 1;
	ll temp=poww(n,k/2);
	if(k%2==0)
	{
		return ((temp%mod)*(temp%mod))%mod;
	}
	else
	{
		return ((((temp%mod)*(temp%mod))%mod)*3)%mod;
	}
}
int main()
{
    ll t;
    cin>>t;
    while(t--)
    {
    ll k=1,n;
    cin>>n;
    k=poww(3,n);
    cout<<k<<"\n";
    }
}
