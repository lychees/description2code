#include <iostream>
using namespace std;

int max(int mx,int cur)
{
	if(mx>cur) return mx;
	else return cur;
}

int main()
{
    int i,n,cur,mx;
    mx=0;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	int x;
    	cin>>x;
    	if(cur+x>x)
    	{
    		cur=cur+x;
    	}
    	else
    	{
    		cur=x;
    	}
    	mx=max(mx,cur);
    }
    cout<<mx<<endl;
    return 0;
}
