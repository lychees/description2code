#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
	    string s1,s2;
	    int k;
	    cin>>s1>>s2>>k;
	    int n=s1.length();
	    int i;
	    int count=0;
	    for(i=0;i<n;i++)
	    {
	    	if(s1[i]!=s2[i]) count++;
	    }
	    int temp=k-count;
	    if(temp<0)
	    {
	    	cout << "No\n";
	    }
	    else if(temp==0)
	    {
	    	cout << "Yes\n";
	    }
	    else
	    {
	    	if(temp%2==0) cout << "Yes\n";
	    	else cout << "No\n";
	    }
	}
    return 0;
}
