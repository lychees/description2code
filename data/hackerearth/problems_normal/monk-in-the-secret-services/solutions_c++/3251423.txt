#include <bits/stdc++.h>
#define INF INT_MAX/10
using namespace std;
int a[101][101], dis[101][101];
int main(){
int test_case, n, x, y, z, m;
int S, A, H;
scanf("%d", &test_case);
while(test_case--){
	scanf("%d %d", &n, &m);
	for(int i=1; i<=n; i++){
		for(int j=1; j<=n; j++){
			dis[i][j]=INF;
		}
		dis[i][i]=0;
	}

	for(int i=0; i<m; i++){
        scanf("%d %d %d", &x, &y, &z);
        dis[x][y]=min(dis[x][y], z);
        dis[y][x]=min(dis[y][x], z);
	}
	scanf("%d %d %d", &S, &A, &H);
	for(int k=1; k<=n; k++){
		for(int i=1; i<=n; i++){
			for(int j=1; j<=n; j++){
				if(dis[i][k]+dis[k][j]<dis[i][j]){
					dis[i][j]=dis[i][k]+dis[k][j];
				}
			}
		}
	}
	int max_=0;
	for(int i=1; i<=n; i++){
		if(i==S || i==A || i==H || dis[i][S]==INF || dis[i][A]==INF || dis[i][H]==INF){
			continue;
		}
		max_ = max(dis[i][S]+2*dis[i][A]+dis[i][H], max_);
	}
	cout<<max_<<endl;
}
}
