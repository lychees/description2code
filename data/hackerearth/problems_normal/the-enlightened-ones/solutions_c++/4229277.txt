#include <iostream>
#include<algorithm>
#define ll long long
using namespace std;
ll pos[100005];


ll chek(ll dist,ll n,ll num)
{
	int k=pos[0]+dist;
	num--;
	for(int i=1;i<n;i++)
	{
		if (pos[i]<=(k+dist))
			continue;
		if(num==0)
			return 0;
		k=pos[i]+dist;
		num--;
		
	}
	return 1;	
}
int main()
{
	ll n,ub,ans,lb,mid,tmp,k,i;
	cin>>n>>k;
	for(i=0;i<n;i++)
	{
		cin>>pos[i];
	}
	sort(pos,pos+n);
	lb=0;
	ub=10000000;
	while(lb<=ub)
	{
		mid=(ub+lb)/2;
			ans=mid;
		
		if (chek(mid,n,k)==1 && chek(mid-1,n,k)==0)
			break;
		if (chek(mid,n,k)==0)
			lb=mid+1;
			
		else
			ub=mid-1;
	}
    cout << ans << endl;
    return 0;
}
