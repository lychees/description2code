#include <bits/stdc++.h>

using namespace std;
int fi(int n)
{
    int result = n, i;
    for(i=2;i*i <= n;i++)
    {
     if (n % i == 0) result -= result / i;
     while (n % i == 0) n /= i;
    }
    if (n > 1) result -= result / n;
    return result;
}

//long long c2(long long n){ return (long long)(((long long)n)*((long long)(n-1)))/2;}

long long pre[1000005];

int main()
{

    int t, i;
    scanf("%d", &t);
    pre[1] = 0;
    for(i = 2; i <= 1000004; ++i){
        int x = fi(i);

        pre[i] = pre[i-1] + (i-x- 1);
    }
    //cout<<fi(3) << " " <<fi(4)<<" " <<fi(5)<<endl;
    int n;
    while(t--)
    {
        scanf("%d", &n);
        printf("%lld\n", pre[n]);
    }
}
