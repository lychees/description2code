test = int(raw_input())
while test:
    N = int(raw_input())
    array = map(int, raw_input().split())
    color = list(set(array))
    counter = []
    for i in range(len(color)):
        counter.append(array.count(color[i]))
    while (len(counter) > 1):
        small = min(counter)
        for i in range(len(counter)):
            counter[i] -= small
        counter.pop(counter.index(0))
    if counter[0] > 1:
        print "bad luck"
    else:
        print "can do"
    test -= 1
