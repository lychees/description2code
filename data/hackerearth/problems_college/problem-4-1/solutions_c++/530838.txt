#include<bits/stdc++.h>
using namespace std;
long long dp[100020];
int main()
{
    int n,a;
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
    {
        scanf("%d",&a);
        dp[a]++;
    }
    for(int i=2;i<=100000;i++)
        dp[i]=max(dp[i-1],dp[i-2]+dp[i]*i);
        cout<<dp[100000];
    return 0;
}
