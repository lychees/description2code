#include <iostream>
#include <cstring>
#include <algorithm>
 
using namespace std;
string s[10001];
int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	for(int i=0;i<n;i++) cin>>s[i];
    	sort(s,s+n);
    	bool ok = false;
    	for(int i=0;i<n-1;i++){
    		for(int j=i+1;j<i+2;j++){
    			if (s[i].length()>s[j].length()) continue;
    			string foo = s[i], foobar = s[j];
    			auto res = std::mismatch(foo.begin(), foo.end(), foobar.begin());
 
				if (res.first == foo.end())
				{
  					ok = true;
  					break;
				}
    		}
    		if (ok) break;
    	}
    	if (ok) puts("NO");
    	else puts("YES");
    }
    return 0;
}
