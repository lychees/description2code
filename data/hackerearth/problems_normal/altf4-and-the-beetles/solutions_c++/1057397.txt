void jebaj()
{
	LL n, a, b, c;
	VII v;
	cin >> n >> a >> b >> c;
	REP(i, n)
	{
		int x, y;
		cin >> x >> y;
		v.PB(MP(x, 0));
		v.PB(MP(y, 1));
	}
	sort(ALL(v));
	LL ile_zyje = 0;
	LL ile_spi = 0;
	LL wyn = 0;
	REP(i, SIZE(v))
	{
		PII x = v[i];
		if(x.ND == 0)
		{
			ile_zyje++;
		}
		else
		{
			ile_zyje--;
			ile_spi++;
		}
		wyn = max(wyn, ile_zyje * b + ile_spi * c + (n - ile_zyje - ile_spi) * a);
	}
	cout << wyn << "\n";
}
