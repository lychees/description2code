#include <iostream>
using namespace std;
int main(int argc, char const *argv[])
{
	int t, n, k, A[100005];
	cin >> t;
	while (t != 0) {
		cin >> n >> k;
		for (int i = 0; i < n; i++) {
			if (i < n - k) A[i] = 9; else A[i] = 0;
		}
		if (A[k - 1] == 0) {
			A[k - 1] = 9;
			int j = k - 2;
			while (A[j] != 9) {
				A[j] = 9;
				j--;
			}
			A[j] = 8;
		} else {
			A[k - 1] = 8;
		}
		A[n - 1] = 1;
		for (int i = 0; i < n; i++)
			cout << A[i];
		cout << endl;
		t--;
	}
	return 0;
}
