#include <iostream>
typedef unsigned long long int ll;
ll pre;
using namespace std;

ll fib(ll N)
{
	ll fi=0,ne=1,res,i;
       	if(N==0)
       	return fi;
       	for(i=2;i<=N;i++)
       	{
       		res=fi+ne;
       		fi=ne;
       		ne=res;
       		if((i-1)==N)
       		pre=ne;
       	}
       	return ne;
}	

int main()
{
    ll T,N;
    cin>>T;
    while(T--)
    {
    	cin>>N;
    	cout<<fib(N-1)<<" "<<fib(N)<<"\n";
    }
    return 0;
}
