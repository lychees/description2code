#define sc(x) scanf("%d",&x)
#define scl(x) scanf("%lld",&x)
using namespace std;

LL GCD(LL a, LL b){
    LL r = 1;
    while(1){
        r = a % b;
        if(r == 0)
            return b;
        a = b;
        b = r;
    }
}

int main(){
    int n, t, i, j;
    LL a, b;
    cin>>t;
    while(t--){
        scl(a);scl(b);
        if(GCD(a, b) == 1){
            printf("Is a Co-Prime\n");
        }
        else{
            printf("Not a Co-Prime\n");
        }
    }
}
