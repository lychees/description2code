def f(s):
    vowels = ['a', 'e', 'i', 'o', 'u', 'y']
    count = 0
    countC = 0
    for i in range(len(s)):
        if s[i] not in vowels:
            count += 1
            countC += 1
            if count == 3:
                return 'hard'
        else:
            count = 0
    if countC > len(s) - countC:
        return 'hard'
    return 'easy'


t = int(raw_input())
for i in range(t):
    s = raw_input()
    print f(s)
