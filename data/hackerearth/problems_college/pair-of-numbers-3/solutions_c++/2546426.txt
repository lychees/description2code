#include <iostream>
using namespace std;
int main()
{
    int t,i,n,j,count;
    cin>>t;
    while(t--)
    {
    	count=0;
    	cin>>n;
    	for(i=1;i<n;i++)
    	{
    		for(j=i+1;j<=n;j++)
    		{
    			if((i*j)%(i+j)==0)
    			   count++;
    		}
    	}
    	cout<<count<<"\n";
    }
}
