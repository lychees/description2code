#include <iostream>
#include <string>
#include <vector>

using namespace std;

int main()
{
    string s;
    int t;
    cin >> t;
    int c;
    vector<int> v;
    while (t--) {
    	cin >> s;
    	c = 0;
    	for (int i = 0; i < s.length(); i++) {
    		if (s[i] == '_') c++;
    		if (s[i] == '|' && c) {
    			v.push_back(c);
    			c = 0;
    		}
    	}
    	if (c) v.push_back(c);
    	if (v.size() == 1 && v[0] == s.length()) {
    		cout << "Bob\n";
    		v.clear();
    		continue;
    	}
    	int r = 0;
    	int sz = v.size();
    	for (int i = 0; i < sz; i++) r ^= v[i];

    	if (r) cout << "Alice\n";
    	else cout << "Bob\n";
    	v.clear();
    }

    return 0;
}
