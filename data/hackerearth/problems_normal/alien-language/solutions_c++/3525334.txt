#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string s,t;
		cin>>s>>t;
		int i;
		bool hash[26];
		for( i=0;i<26;i++)
		 hash[i]=0;
		 for( i=0;i<s.length();i++)
		 hash[s[i]-97]=1;
		 for( i=0;i<t.length();i++)
		 {
		 	if(hash[t[i]-97]==1)
		 	{cout<<"YES"<<"\n";break;}
		 }
		 if(i==t.length())
		 cout<<"NO\n";
	}
}
