#include <bits/stdc++.h>
#define ll long long int
ll dp[100005];
ll arr[100005];
ll max(ll a,ll b)
{
	return a>b?a:b;
}
ll func(ll arr[],ll dp[],ll i,ll n)
{
	if(i>=n)
	return 0;
	if(dp[i]!=-1)
	return dp[i];
	if(i==n-1)
	return dp[i]=arr[i];
	dp[i]=max(func(arr,dp,i+1,n),arr[i]+func(arr,dp,i+2,n));
	return dp[i];
	
}
int main()
{
   // printf("Hello World!\n");
   ll t;
   scanf("%lld",&t);
   while(t--)
   {
   		ll i,n;
   		scanf("%lld",&n);
   		memset(dp,-1,sizeof(dp));
	 	for(i=0;i<n;i++)
   		{
   			scanf("%lld",&arr[i]);
   		}
   		ll res=func(arr,dp,0,n);
   		printf("%lld\n",res);
   }
    return 0;
}
