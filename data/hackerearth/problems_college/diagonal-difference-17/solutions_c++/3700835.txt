#include <bits/stdc++.h>
using namespace std;

int main()
{
	long long int t,n,i,j,k,l=0,p=0,u;
	cin>>n;
	long long int a[n][n];
	for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
		{
			cin>>a[i][j];
		}
	}
	
	
	for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
		{
	
			if(i==j)
			{
				l+=a[i][j];
		//		cout<<a[i][j]<<" ";
			}
		 if((i+j)==(n-1))
			{
				p+=a[i][j];
			//	cout<<a[i][j]<<" ";
			}
		}
	}
	
//	cout<<l<<" "<<p;
	
	u=abs(l-p);
	cout<<u;
}
