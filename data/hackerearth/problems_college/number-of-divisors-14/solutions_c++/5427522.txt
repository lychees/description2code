void sieve()
{
    v.pb(2);
    for(i=3; i<=limit; i+=2)
    {
        if(a[i]==0)
        {
            v.pb(i);
            for(j=3*i; j<=limit; j+=2*i ) a[j]=1;
        }
    }

}

void prime_factor(ll n)
{
    sum=1;
    for(i=0; v[i]*v[i]<=n; ++i)
    {
        ll p=1;
        if(n%v[i]==0)
        {

            while(n%v[i]==0)
            {
               ++p;
                n/=v[i];
            }
        }
        sum*=p;
        if(n==1) break;
    }
    if(n>1)
    {

        ll p=2;
        sum*=p;
    }
}
