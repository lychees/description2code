#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define rep(i,n) for(long long int i=0;i<n;i++)
#define repp(i,a,b) for(long long int i=a;i<=b;i++)
#define repp2(i,b,a) for(long long int i=b;i>=a;i--)

map<string,int> myMap;
string s;
ll a,h;
void fun(string k){
    string s1,s2;
    s1=k,s2=k;
    
    rotate(s1.begin(), s1.begin() + h, s1.end());
    for(ll i=1;i<s2.size();i+=2){
        s2[i] = ((s2[i]-'0') + a)%10 + '0';
    }
    if(myMap.find(s1)==myMap.end()){
        myMap[s1]=1;
        fun(s1);
    }
        
    if(myMap.find(s2)==myMap.end()){
        myMap[s2]=1;
        fun(s2);
    }
    
}

int main()
{   
    
    ll t;
    cin>>t;
    while(t--){
        cin>>s>>a>>h;
        myMap[s]=1;
        h%=s.size();
        fun(s);
        cout<<myMap.begin()->first<<endl;
        myMap.clear();
    }
    
    return 0;
}
