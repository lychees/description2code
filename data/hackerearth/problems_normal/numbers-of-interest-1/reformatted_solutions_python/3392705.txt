def nextNumber(arr, X):
    sum = 0
    for n in reversed(arr):
        if X > 0:
            sum += n
            X -= 1
    return sum


def calculateNnumber(X, Y, N):
    arr = []
    for j in range(1, N+1):
        if j <= X:
            arr.append(Y)
        else:
            arr.append(nextNumber(arr, X))
    return arr[-1]


T = int(raw_input())
sol = []
for i in range(0, T):
    X, Y, N = map(int, raw_input().split())
    sol.append(calculateNnumber(X, Y, N))

for s in sol:
    print s
