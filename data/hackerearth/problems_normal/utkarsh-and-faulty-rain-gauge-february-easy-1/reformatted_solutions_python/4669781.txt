from fractions import Fraction
v = Fraction(1)
for t in xrange(int(raw_input())):
    n, r = map(int, raw_input().split())
    a = Fraction(6, n * (n + 1) * (2 * n + 1))
    v *= sum(a * i ** 3 for i in range(1, n + 1)) / r
print '%.4f' % float(v)
