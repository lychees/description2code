#include <cstdio>
using namespace std;

const long long MAX = 100000000000000;
int main()
{
	int T;
	scanf("%d", &T);
	for(int i = 1; i <= T; i++)
	{
		long long X, K;
		scanf("%lld %lld", &X, &K);
		for(long long Y = 1; Y <= MAX; Y++)
		{
			if((X + Y) == (X ^ Y)) { K--; }
			if(K == 0) { printf("Case #%d: %lld\n", i, Y); break; }
		}
	}
	return 0;
}
