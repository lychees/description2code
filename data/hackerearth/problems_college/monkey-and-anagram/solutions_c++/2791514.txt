#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s,n;
    	cin>>s>>n;
    	int b[26]={0};
    	int flag=0;
    	for(int i=0;i<n.length();i++)
    	{
    		b[n[i]-'a']++;
    	}
		
    	int cnt=0;
    	for(int i=0;i<s.length();i++)
    	{
    		b[s[i]-'a']--;
    		if(b[s[i]-'a']<0)
    		{
    			cout<<-1<<endl;
    			flag=1;
    			break;
    		}
    	
    		
    	}
 	
    	if(!flag)
    	{
    	
    	for(int i=0;i<26;i++)
    	if(b[i]>0)
    	{
  
    		cnt++;
    	}
    	cout<<cnt<<endl;
    		
    	}
    	
    	
    	
    }
    return 0;
}
