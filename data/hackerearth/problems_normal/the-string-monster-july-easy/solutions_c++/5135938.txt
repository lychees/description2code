#include <bits/stdc++.h>
using namespace std;
set<vector<int> > st;

int main()
{
    int t,n;
    cin>>t;
    while(t--){
        st.clear();
    	cin>>n;
    	string t,s[n];
    	vector<int> vec[n],s_vec;
    	for(int i=0;i<n;++i){
    		cin>>s[i];
    		vec[i].resize(26);
    		for(int j=0;j<26;++j)
    			vec[i][j]=0;
    		for(int j=0;j<s[i].length();++j)
    			vec[i][s[i][j]-'a']++;

    	}
    	cin>>t;
    	s_vec.resize(26);
    	for(int j=0;j<26;++j)
    		s_vec[j]=0;
    	for(int j=0;j<t.length();++j)
    		s_vec[t[j]-'a']++;
    	bool flag=false;

    	vector<int> arr(26);
    	for(int i=1;i<(1<<(n/2));++i){
    		for(int j=0;j<26;++j)
    			arr[j]=0;
    		for(int j=0;(1<<j)<=i;++j){
    			if(i&(1<<j)){
    				for(int k=0;k<26;++k){
    					arr[k]+=vec[j][k];
    				}
    			}
    		}
    		/*for(int j=0;j<26;++j)
                cout<<arr[j];
            cout<<endl;*/
    		if(arr==s_vec){
    			flag=true;
    			break;
    		}
   			st.insert(arr);
    		if(flag)
    			break;

    	}

    	if(flag){
    		cout<<"YES\n";
    		continue;
    	}

    	vector<int> minu(26);
    	for(int i=1;i<(1<<(n-n/2));++i){
    		for(int j=0;j<26;++j)
    			arr[j]=0;
    		for(int j=0;(1<<j)<=i;++j){
    			if(i&1<<j){
    				for(int k=0;k<26;++k){
    					arr[k]+=vec[j+n/2][k];
    				}
    			}
    		}
    		if(arr==s_vec){
    			flag=true;
    			break;
   			}
    		for(int k=0;k<26;++k){
    			minu[k]=s_vec[k]-arr[k];
    		}
    		if(st.find(minu)!=st.end()){
                /*for(int p=0;p<26;++p)
                    cout<<s_vec[p];
                cout<<endl;
                for(int p=0;p<26;++p)
                    cout<<arr[p];
                cout<<endl;
                for(int p=0;p<26;++p)
                    cout<<minu[p];
                cout<<endl;*/
    			flag=true;
    			break;
    		}
    	}

    	if(flag)
    		cout<<"YES\n";
    	else
    		cout<<"NO\n";
    }
    return 0;
}
