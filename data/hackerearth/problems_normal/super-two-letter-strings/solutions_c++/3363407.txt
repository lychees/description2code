#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,k;
		cin>>n>>k;
		vector<long long> v;
		v.push_back(1);
		for(int i=1;i<=n;i++)
		{
			v.push_back(0);
			for(int j=(i-1);j>=0 && (i-j)<=k;j--)
			{
				v[i]+=v[j];
				v[i]%=(1000000007);
			}
		}
		cout<<v[n]<<endl;
	}
	return 0;
}
