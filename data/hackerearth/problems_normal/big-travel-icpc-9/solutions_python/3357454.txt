#print 'Hello World!'
def fun(a):
	t = 0
	ans = 0
	for i in range(1, len(a)):
		t += i * (a[i] - a[i-1])
		ans += t
	return ans	
X, Y =[], []
for _ in range(int(raw_input())):
	x, y = map(int, raw_input().split())
	X.append(x)
	Y.append(y)
X.sort()
Y.sort()
print (fun(X) +fun(Y)) % 1000000007
