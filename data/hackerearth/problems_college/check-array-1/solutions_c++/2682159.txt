#include <iostream>
using namespace std;
typedef long long ll;
ll arr[100005],pre[100005],post[100005];
int main()
{
//    cout << "Hello World!" << endl;
  	ll t;
  	cin>>t;
  	while(t--){
  		ll n;
  		cin>>n;
  		for(int i=1;i<=n;i++){
  			cin>>arr[i];
  			pre[i]=arr[i]+pre[i-1];
  		}
  		for(int i=n;i>0;i--)
  			post[i]=post[i+1]+arr[i];
  			int i;
  		for( i=1;i<=n-1;i++){
  			if(pre[i]==post[i+2])
  				break;
  		}
  		if(i==n)
  			cout<<"NO"<<endl;
  		else
  			cout<<"YES"<<endl;
  	}
  return 0;
}
