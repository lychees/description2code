#include <iostream>
using namespace std;

int main()
{
    long long int t,n,m,odd,even,i=1;
    cin >> t;
    while(t > 0)
    {
    	even=0;
    	odd=0;
    	i=1;
    	cin >> n >> m;
    	while(i <= n)
    	{
    		if(i % 2 != 0)
    		{
    		if(i == m)
    		odd+=m;
    		
    		else if(i < m)
    		odd+=i;
    		
    		else
    		odd+=m;
    		}
    		
    		else
    		{
    		if(i == m)	
    		even+=m;
    		
    		else if(i < m)
    		even+=i;
    		
    		else
    		even+=m;
    		}
    	i++;
    	if(i == n+1)
    	cout << even <<  " " << odd << endl;
    	}
      t--;		
    }
    return 0;
}
