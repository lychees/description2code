#include <bits/stdc++.h>


using namespace std;

int main()
{
    //freopen("inp.txt","r",stdin);
    //freopen("out.txt","w",stdout);
    int T;
    cin >> T;
    while (T--)
    {
        long long int n, a, b, c, p, q;
        cin >> n >> a >> b >> c;
        multiset<pair<long long int, int> > time;
        multiset<pair<long long int,int> >::iterator it;
        for (int i = 0; i < n; ++i)
        {
            cin >> p >> q;
            time.insert( make_pair(p,1) );
            time.insert( make_pair(q,2) );
        }
        long long int ans = n*a, rs = n*a;
        for (it = time.begin(); it != time.end(); ++it)
        {
            if (it->second == 1)
            {
                ans += b-a;
            }
            else ans += c-b;
            rs = max(ans,rs);
        }
        cout << rs << endl;
    }
    return 0;
}
