#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    int A[1000];
    
    A[0]=1;
    A[1]=7;
    for(int i=2;i<1000;i++){
    	
    	A[i]=(A[i-1]*2)+7;
    	
	}
	

    
    string player[7];
    player[0]="Rohit";
    player[1]="Dhawan";
    player[2]="Kohli";
    player[3]="Yuvraj";
    player[4]="Raina";
    player[5]="Dhoni";
    player[6]="Sir Jadeja";
    
    
    int t;
    cin >> t;
    
    while(t--){
    	int n;
    	cin >> n;
    	int k = 0;
        if(n<=7){
        	
        	cout << player[n-1] << endl;    	
        }    	
    	
    	else{
    		
    		int ace=0;
    		for(int i=0;i<1000;i++){
    			
    			if(A[i]<n){
    				ace = A[i];
    				k = i;
    			}
    			else if(A[i]>n){
    				break;
    			}
    			
    		}
    		
    		int ans = n - ace;
    		int poww = pow(2,k);
    		
    		if(ans%poww==0){
    			
    		int nit = (ans/poww);
    		//cout << nit;
    		cout << player[nit-1] << endl;
    			
    			
			}
    		else{
    		
			int nit = (ans/poww);
    		//cout << nit;
    		cout << player[nit] << endl;
    			
			}
    		
    		
    	}
    	
    	
    }
    
    
}
