#include <bits/stdc++.h>
using namespace std;
# define ll long long
int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	long p=0,a,b,i,j;
    	for(i=1;i<=n;i++){
    		for(j=1;j<=n&&i!=j;j++){
    			a=i*j;
    			b=i+j;
    			if(a%b==0)
    			p++;
    		}
    	}
    	cout<<p<<endl;
    }
}
