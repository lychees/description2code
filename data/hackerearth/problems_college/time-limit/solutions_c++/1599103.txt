#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int n ,m, i, j, k, l, k1, k2;
	long long int a[105], b[105];
	scanf("%lld%lld", &n, &m);
	for(i = 0; i < n; i++) {
		scanf("%lld", &a[i]);	
	}
	for(i = 0; i < n; i++) {
		scanf("%lld", &b[i]);	
	}	
	sort(a, a+n);
	k = a[n-1];
	i = a[0];
	i = i*2;
	j = max(k, i);
	sort(b, b+m);
 	if(j < b[0]) {
		printf("%lld\n", j);
	} else 
		printf("-1\n");
	return 0;
}
