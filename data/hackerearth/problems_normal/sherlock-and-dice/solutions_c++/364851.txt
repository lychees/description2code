#include <iostream>
#include <algorithm>
#include <cstdio>
#include <cmath>
#include <string>
#include <cstring>

#define sf scanf
#define sfll1(a) scanf("%lld", &a)
#define sfll2(a,b) scanf("%lld %lld", &a, &b)
#define sfll3(a,b,c) scanf("%lld %lld %lld", &a, &b, &c)
#define pf printf
#define loop(i,n) for(int i=0; i<n; i++)
#define rloop(i,n) for(int i=n-1; i>=0; i--)


#define MAXN 55
#define MAXSUM 10005

using namespace std;
typedef long long LL;

//variable
LL t,n,m,k, cnt;
double ans, dp[MAXN][MAXSUM];

int main(){
    //freopen("in_04.txt", "r", stdin);
    sfll1(t);
    while(t--){
        sfll3(m,n,k);
        memset(dp, 0, sizeof dp);
       // cout << m << " " << n << " " << k << endl;
        for(int j=1; j<=n; j++) dp[1][j] = 1.0/n;

        for(int i=2; i<=m; i++)
            for(int j=1; j<=k; j++)
                if(j<i || (j > i * n) ) dp[i][j] = 0;
                else for(int k=1; k<=j-i+1 && k<=n; k++) dp[i][j] += (1.0/n) * dp[i-1][j-k];

        cnt = 0; ans = dp[m][k];
        if(ans != 0 )
            while(!floor(ans)){ans *= 10; cnt++;}
        pf("%.3lf %lld\n", ans, cnt);
    }
    //while(1);
    return 0;
}
