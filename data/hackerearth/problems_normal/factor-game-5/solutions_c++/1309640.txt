#include<bits/stdc++.h>
using namespace std;

const int N=1e5+10;

int cnt[N];
bool isPrime[N];
int fact[N];

int main() {
    int n;
    scanf("%d",&n);
    for(int i=2;i<N;i++) {
        if(isPrime[i]==false) {
            fact[i]=i;
            for(int j=2*i;j<N;j+=i) {
                isPrime[j]=true;
                fact[j]=i;
            }
        }
    }
    for(int i=2;i<N;i++) {
        cnt[i]=1+cnt[i/fact[i]];
    }
    int ans=0;
    for(int i=0;i<n;i++) {
        int val;
        scanf("%d",&val);
        ans=ans xor cnt[val];
    }
    if(ans) {
        puts("ALICE");
    } else {
        puts("BOB");
    }
    return 0;
}
