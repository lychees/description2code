#include<bits/stdc++.h>
using namespace std;
int root(int i);
void unions (int a,int b);
int size [100000]={1};
int arr[100000];
int main()
{
	set <int> s;
	int n,k,i,j;
	cin>>n;
	cin>>k;
	for(int y=1;y<=n;y++)
	arr[y]=y;
	for(int x=1;x<=k;x++)
	{
		cin>>i>>j;
		unions(i,j);
	}
	for(int k=1;k<=n;k++)
	{
		int tmp=root(k);
		s.insert(tmp);
	}
	cout<<s.size();
	return 0;
}
void unions (int a, int b)
{
	int root_a=root(a);
	int root_b=root(b);
	if(size[root_a]<size[root_b])
	{
		arr[root_a]=arr[root_b];
		size[root_b]+=size[root_a];
	}
	else
	arr[root_b]=arr[root_a];
	size[root_a]+=size[root_b];
}
int root(int i)
{
	while(arr[i]!=i)
	{
		i=arr[i];
	}
	return i;
}
