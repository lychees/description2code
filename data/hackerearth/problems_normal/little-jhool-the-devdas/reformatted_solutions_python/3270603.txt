li = list(map(int, raw_input().split()))
li = sorted(li)
even = []
odd = []
count = 0
for y in li:
    if y % 2 == 0:
        even.append(y)
    else:
        count += 1
        odd.append(y)
if count == 0:
    print":("
else:
    if sum(odd) % 2 == 0:
        print sum(odd[1:])+sum(even)
    if sum(odd) % 2 != 0:
        print sum(odd)+sum(even)
