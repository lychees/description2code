#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>
#include<vector>

using namespace std;

#define ll long long
#define llu unsigned long long
#define MOD 1000000007
#define INF 2000000000

ll fact[2000001];
ll invfact[2000001];

ll modpow(ll a,ll b)
{
   ll res = 1;
   while(b)
   {
           if(b&1)
           res = (res*a)%MOD;
           
           b >>= 1;
           a = (a*a)%MOD;
   }
   return res;
}

void precalc()
{
     int i;
     fact[0] = fact[1] = 1;
     for(i=2;i<=2000000;i++)
     {
        fact[i] = (i*fact[i-1])%MOD;
     }
     
     invfact[2000000] = modpow(fact[2000000],MOD-2);
     
     for(i=1999999;i>=0;i--)
     {
        invfact[i] = ((i+1)*invfact[i+1])%MOD;
     }
}

ll nCr(int n,int r)
{
   ll res = fact[n];
   res = (res*invfact[r])%MOD;
   res = (res*invfact[n-r])%MOD;
   return res;
}

int main()
{
    int t,n,m;
    precalc();
    scanf("%d",&t);
    while(t--)
    {
              scanf("%d%d",&n,&m);
              printf("%lld\n",nCr(n+m-2,n-1));
    }
    //system("pause");
    return 0;
}
