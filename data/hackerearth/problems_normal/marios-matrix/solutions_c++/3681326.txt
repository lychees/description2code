#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int a,b;
    	int flag=1;
    	cin>>a>>b;
    	map<string,int> m;
    	string str;
    	for(int i=0;i<a;i++){
    		cin>>str;
    		if(m.find(str)==m.end())
    			m.insert(make_pair(str,0));
    		else
    			flag=0;
    	}
    	if(flag==0)
    		cout<<"No";
    	else 
    		cout<<"Yes";
    	cout<<endl;
    }
    return 0;
}
