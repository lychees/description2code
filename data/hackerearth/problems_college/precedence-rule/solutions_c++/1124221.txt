#include<iostream>
#include<algorithm>

using namespace std;

int main()
{
    ios::sync_with_stdio(false);

    long long t,i,x,y;
    string s;

    cin>>t;
    while(t--)
    {
        cin>>s;
        y=s[s.size()-1]-'0';
        for(i=s.size()-2;i>-1;i-=2)
        {
            switch(s[i])
            {
                case '+':
                    x=s[i-1]-'0';
                    y=y+x;
                    break;

                case '-':
                    x=s[i-1]-'0';
                    y=y-x;
                    break;

                case '*':
                    x=s[i-1]-'0';
                    y=x*y;
                    break;
            }
        }
        cout<<y<<endl;
    }
    return 0;
}
