#include<bits/stdc++.h>
using namespace std;
int main()
{
int t;
cin>>t;
string a[1001];
while(t--)
{
	int n;
	int flag=0;
	int c=0;
	cin>>n;
	for(int i=0;i<n;i++)
	{

			cin>>a[i];
		}
		for(int  i=0;i<n;i++)
		{
			for(int j=0;j<n;j++)
			{
				
			
		
			if(a[i][j]=='#')
			c++;
		}
	}
	if(c%2!=0)
	{
		cout<<"NO"<<"\n";
		
	}
	else
	{
		int count=0;
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<n;j++)
			{
				if(a[i][j]=='#')
				count++;
				if(count==(c/2))
				{
						flag=1;
					for(int k=j+1;k<n;k++)
					{
				
					if(a[i][k]=='#')
					{
						flag=2;
						break;
					
					}
					}
					break;
				}
			}
			if(flag==2 || flag==1)
			break;
		}
		if(flag==1)
		cout<<"YES"<<"\n";
		else
		{
			int count=0;
			flag=0;
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<n;j++)
			{
				if(a[j][i]=='#')
				count++;
				if(count==(c/2))
				{
					flag=1;
					for(int k=j+1;k<n;k++)
					{
					
					if(a[k][i]=='#')
					{
						flag=2;
						break;
					
					}
					}
					break;
				}
			}
			if(flag==2 || flag==1)
			break;
		}
		if(flag==1)
		cout<<"YES"<<"\n";	
		else 
		cout<<"NO"<<"\n";
		}
	}
	
}
}
