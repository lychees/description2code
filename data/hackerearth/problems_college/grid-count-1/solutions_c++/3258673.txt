#include <iostream>
#include <math.h>
using namespace std;
long long dp[1001][1001];
int main()
{
    int n,k,i,j;
    long long cnt=0;
    long long x;
    float y;
    cin>>n>>k;
    int a[n][n];
    
    for(i=0;i<n;i++){
    	for(j=0;j<n;j++)
    	  cin>>a[i][j];
    }
    dp[0][0]=a[0][0];
    for(i=1;i<n;i++){
    	dp[0][i]=dp[0][i-1]+a[0][i];
    	dp[i][0]=dp[i-1][0]+a[i][0];
    }
    for(i=1;i<n;i++){
    	for(j=1;j<n;j++){
    		dp[i][j]=dp[i-1][j]+dp[i][j-1]+a[i][j]-dp[i-1][j-1];
    	//	cout<<dp[i][j]<<" ";
    	}
    	//cout<<endl;
    }
    
    for(int row=k-1;row<n;row++)
    {
    	for(int col=k-1;col<n;col++)
    	{
    		x=dp[row][col]-((row-k)>=0?dp[row-k][col]:0)-((col-k)>=0?dp[row][col-k]:0)+(((row-k>=0)&&(col-k>=0))?dp[row-k][col-k]:0);
    	//	cout<<row<<" "<<col<<": "<<x<<endl;
    		y=cbrt(x);
    		long p=(long)y;
    	//cout<<"x= "<< x<<endl;
    	if(p*p*p==x)
    	cnt++;
    	}
    	
    }
    cout<<cnt;
    
    return 0;
}
