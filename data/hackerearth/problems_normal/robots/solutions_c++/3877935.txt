#include <bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;
ll int a[1010];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int test;
	cin>>test;
	while(test--)
	{
		ll int n,i,t,ans=1;
		cin>>n;
		for(i=0;i<n;i++)
		{
			cin>>a[i];
		}
		sort(a,a+n);
		t=0;
		for(i=0;i<n;i++)
		{
			ans=(ans * max(0ll,a[i]-t))%mod;
			t++;
		}
		cout<<ans<<endl;
	}
	return 0;
}
