import sys
mod = 10**9 + 7
for cases in xrange(int(sys.stdin.readline())):
    n = int(sys.stdin.readline())
    arr = list(map(int,sys.stdin.readline().split()))
    A = B = 0
    pos = neg = 0
    for i in arr:   B += abs(i)
    B %= mod
    arr.sort()
    index = 0
    max_neg = n/2 + n%2
    while max_neg > 0:
        A -= arr[index]
        A %= mod
        index += 1
        max_neg -= 1
    for i in xrange(index,n):
        A += arr[i]
        A %= mod
    print (A*B)%mod
