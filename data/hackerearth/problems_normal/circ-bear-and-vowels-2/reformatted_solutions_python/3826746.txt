import re
noOfTest = input()
inputList = []
testNo = 0
for i in range(0, int(noOfTest)):
    inputList.append(raw_input())


def isHard_noOfVowels():
    length = len(inputList[testNo])
    noOfVowels = inputList[testNo].count("a",
                                         0,
                                         length)+inputList[testNo].count("e",
                                                                         0,
                                                                         length)+inputList[testNo].count("i",
                                                                                                         0,
                                                                                                         length)+inputList[testNo].count("o",
                                                                                                                                         0,
                                                                                                                                         length)+inputList[testNo].count("u",
                                                                                                                                                                         0,
                                                                                                                                                                         length)+inputList[testNo].count("y",
                                                                                                                                                                                                         0,
                                                                                                                                                                                                         length)
    if (noOfVowels < (length - noOfVowels)):
        return True
    else:
        return False


def isHard_consecutive():
    p = re.compile('.*[^aeiouy][^aeiouy][^aeiouy].*')
    if (p.match(inputList[testNo])):
        return True
    else:
        return False


for i in range(0, int(noOfTest)):
    if (isHard_consecutive() or isHard_noOfVowels()):
        print('hard')
    else:
        print('easy')
    testNo = testNo + 1
