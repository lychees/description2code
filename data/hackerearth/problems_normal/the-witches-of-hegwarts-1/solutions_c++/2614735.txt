#include <bits/stdc++.h>
#define INF 1000000000
#define MOD 1000000007
#define MAXN 10000005

using namespace std;

typedef long long ll;

int memo[MAXN];
int rec(int n)
{
	if (n < MAXN && memo[n] != -1)
		return memo[n];
	if (n == 1)
		return memo[n] = 0;
	int q = n - 1;
	if (n % 3 == 0)
		q = min(q, 1 + rec(n / 3));
	else
		q = min(q, 1 + rec(n - 1));
	if (n % 2 == 0)
	    q = min(q, 1 + rec(n / 2));
	else
		q = min(q, 1 + rec(n - 1));
    if (n < MAXN)
		memo[n] = q;
	return q;
}
int main()
{
    int t, n;
	memset(memo, -1, sizeof memo);
    for (int i = 1; i < MAXN; i++)
		rec(i);
	scanf("%d", &t);
	while (t--)
	{
		scanf("%d", &n);
		printf("%d\n", rec(n));
	}
	return 0;
}
