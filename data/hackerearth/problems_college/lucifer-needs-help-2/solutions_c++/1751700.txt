#include <iostream>
#include <cstdio>
#include <cstring>
#define size 1000003

using namespace std;
#define get getchar_unlocked()


inline long long scan()
{
	long long n = 0, ch = get;

	while (ch < 48 || ch > 57)
		ch = get;
	while (ch >= 48 && ch <= 57) {
		n = (n<<3) + (n<<1) + ch -48;
		ch = get;
	}

	return n;
}


 
long long a[size],b[size];
int main()
{
    long long n,m,i,j,k,l;
    n = scan();
    for(l=0;l<n;l++)
        a[l] = scan();
    m = scan();
    while(m--)
    {
        i = scan();
        j = scan();
        k = scan();
        i-=1;
        b[i]+=k;
        b[j]-=k;
    }
    printf("%lld\n",a[0]+b[0]);
    for(l=1;l<n;l++)
    {
        b[l]+=b[l-1];
        printf("%lld\n",a[l]+b[l]);
    }
    return 0;
}
