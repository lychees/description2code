#include <iostream>
#include <algorithm>
using namespace std;
 
int main()
{
    long long n;
    cin>>n;
    string s[n];
    int i;
    for(i=0;i<n;i++)
    	cin>>s[i];
    sort(s,s+n);
    int j;
    string m=s[0];
    for(i=1;i<n;i++)
    {
    	j=i;
    	while(s[i]==m)
    		i++;
    	cout<<m<<" "<<(i-j+1)<<endl;
    	m=s[i];
    }
    cout<<m<<" 1"<<endl;
    return 0;
}
