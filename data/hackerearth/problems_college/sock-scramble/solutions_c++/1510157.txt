#include <iostream>
using namespace std;

int main()
{
        int t,n,x,y;
    cin>>t;
    while(t--)
    {
              cin>>n;
              y=0;
              for(int i=0;i<n;i++)
              {
                      cin>>x;
                      y=y^x;
                      }
              cout<<y<<"\n";
              }

    return 0;
}
