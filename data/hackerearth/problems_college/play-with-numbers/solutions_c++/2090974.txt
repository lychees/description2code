#include <iostream>
#include<math.h>
using namespace std;

int main()
{
	long long test,x,y,z,i,count,m;
    cin>>test;
    while(test--)
    {
    	cin >>z>>x>>y;
    	count=0;
        for(i=1;i<=sqrt(z);i++)
        {
            if(z%i==0&&i<=x)
            {
                if((z/i)<=y)
                {
                    count++;
                }
            }
            if(z%i==0&&i<=y&&i*i!=z)
            {
                if((z/i)<=x)
                {
                    count++;
                }
            }
            //cout<<n<<" "<<i<<endl;
        }
    	cout<<count<<endl;
    }
    return 0;
}
