#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,i,j;
	cin>>n;
	string str[n];
	for(i=0;i<n;i++)
	{
		cin>>str[i];
	}
	int flag=0;
	for(i=0;i<n;i++)
	{
		string s;
		for(j=0;j<n;j++)
		{
			s = str[j];
			reverse(s.begin(),s.end());
			if(str[i] == s)
			{
				cout<<str[i].length()<<" "<<s[s.length()/2]<<"\n";
				flag =1;
				break;
			}
		}
		if(flag == 1)
		{
			break;
		}
	}
}
