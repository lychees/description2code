#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	int max=0;
	while(t--){
		max=0;
		int a ,b,c;
		cin>>a>>b;
		for(int i=a;i<=b;i++){
			for(int j=i;j<=b;j++){
				c=i^j;
				if(max<c)
				max=c;
			}
		}
		cout<<max<<endl;
	/*	for(int i=a;i<=b;i++){

	}
}
