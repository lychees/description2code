#include <iostream>
using namespace std;
int min(int d[],int m);
int no(int g[],int o);

int main()
{ int n,a[10000],x,c,l=0;
    cin>>n;
    for(x=0;x<n;x++)
    {
    	cin>>a[x];
	}
	cout<<n<<endl;
no(a,n);
    return 0;
}
int no(int g[],int o)
{int c,x,l=0;
		c=min(g,o);
	for(x=0;x<o;x++)
	{
		g[x]=g[x]-c;
		if(g[x]>0)
		{
			g[l]=g[x];
			l++;
		}
	}	
	if(l!=0)
	{
		cout<<l<<endl;
		no(g,l);
	}
	
}
int min(int d[],int m)
{
	int x,b=d[0];
	for(x=0;x<m-1;x++)
	{
		if(b>d[x+1])
		{
			b=d[x+1];
		}
	
	}
	return b;
}
