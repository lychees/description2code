template <typename T> inline void read(T &x){
	char ch = getchar();x = 0;bool fl = 0;
	for(; ch<'0' || ch>'9';ch = getchar()) if (ch=='-') fl = 1;
	for(;ch>='0' &&ch<='9';ch = getchar()) x = x*10+ch-'0'; if (fl) x = -x;
}
#define MAX 1010101
#define TRUE 1
#define FALSE 0

bool prime[MAX];
void init(){
	ll i,j;
	for(i=0;i<MAX;i++)
        prime[i]=true;
    for(i=0;i<MAX;i+=2)
        prime[i]=false;
    prime[1]=false;prime[2]=true;
    for(i=3;i<MAX;i++)
    {
        if(prime[i]==true)
            for(j=i*i;j<MAX;j+=i)
                prime[j]=false;
	}
}

int isprime(ll n){
	if(n==2)
		return 1;
	for(int i=2;i<sqrt(n);i++){
		if(n%i==0){
			return 0;
		}
	}
	return 1;
}

int main(){
	init();
	ll n,t,i,j,a[MAX];
	a[0]=0;
	a[1]=1;
	for(i=2;i<MAX;i++){
		a[i]=a[i-1]+a[i-2];
	}
	cin>>t;
	while(t--){
		cin>>n;
		ll sum=0;i=0;
		while(a[i]<=n) sum+=a[i],i++;
		//cout<<endl;
		ll meta=((n*(n-1))/2)-sum;
		//cout<<meta<<endl;
		meta++;
		if(isprime(meta)){
			cout<<"Tic"<<endl;
		}else{
			if(!(meta&1)){
				cout<<"Tac"<<endl;
			}
			else{
				cout<<"Toe"<<endl;
			}
		}
	}
    return 0;
}
