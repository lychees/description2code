#include <iostream>
using namespace std;

int main()
{
    short int tc,i,cnt,l;
    int n,j;
    cin>>tc;
    for(i=0;i<tc;i++)
    {
    	l=0;
    	cnt=0;
    	cin>>n;
    	long int p[n],sp;
    	cin>>sp;
    	for(j=0;j<n;j++)
    	{
    		cin>>p[j];
    		if(p[j]==sp)
    		l++;
    		if(p[j]>sp)
    		cnt++;
    		
    	}
    	if(l!=0)
    	cout<<"YES "<<cnt<<endl;
    	else
    	cout<<"NO 0"<<endl;
    	
    }
    return 0;
}
