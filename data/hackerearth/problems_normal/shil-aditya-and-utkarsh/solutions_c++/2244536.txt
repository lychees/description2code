#include<iostream>
#include<cstdio>
void scanint (int &);
#define gc getchar_unlocked
using namespace std;
int main()
{
 int n;
   scanint(n);
 long long int ans=0;
    for(int i=n-1;i>=2;i--)
           ans=ans+(i*(i-1))/2;
           printf("%lld \n",2*ans);
	
	
}
 void scanint(int &m)
 { m=0;

 	 register int c =gc();
 	for(;c<48||c>57;c=gc());
 	for(;c>47&&c<58;c=gc())
 	{
 	 m=(m<<3)+(m<<1)+ (c-48);
 		
	 }
}
