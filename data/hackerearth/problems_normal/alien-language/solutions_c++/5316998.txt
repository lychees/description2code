#include <iostream>
#include <stdio.h>
#include <string>
#include <algorithm>
#include <string.h>

using namespace std;

int h2[27];

int main()
{
    int t;
    scanf("%d",&t);
    for(int tt=1;tt<=t;++tt)
    {
        string a,b;
        cin>>a>>b;
        int ln=a.size();
        int ln1=b.size();
        for(int i=0;i<ln1;++i)
            h2[b[i]-'a']++;
        bool ok=true;
        for(int i=0;i<ln && ok;++i)
            if(h2[a[i]-'a'])printf("YES\n"),ok=false;
        if(ok)printf("NO\n");
        memset(h2,0,sizeof(h2));
    }
    return 0;
}
