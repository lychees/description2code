#include <iostream>
using namespace std;

int main()
{
    int t,i,l,j,k,ok;
    string s;
    cin>>t;
    for(i=0;i<t;i++)
    {
    	ok=0;
    	cin>>s;
    	if(s.length()>=10)
    	{
    		cout<<"YES"<<endl;
    		continue;
    	}
    	for(j=1;j<s.length();j++)
    	{
    	  for(l=j+1;l<s.length();l++)
    	  {
    	  	for(k=l+1;k<s.length();k++)
    	  	{
    	  	    string s1=s.substr(0,j);
    	  		string s2=s.substr(j,l-j);
    	  		string s3=s.substr(l,k-l);
    	  		string s4=s.substr(k,s.length()-k);
    	  		
    	  		if((s1!=s2)&&(s1!=s3)&&(s1!=s4)&&(s2!=s3)&&(s2!=s4)&&(s3!=s4))
    	  		{
    	  			ok=1;
    	  		}
    	  	}
    	  
    	  }
    	}
    	if(ok)
    	cout<<"YES"<<endl;
    	else
    	cout<<"NO"<<endl;
    	
    	
    }
    return 0;
}
