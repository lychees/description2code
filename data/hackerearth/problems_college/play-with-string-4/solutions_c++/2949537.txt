#include <iostream>
#include<string>
#include<math.h>
using namespace std;

int main()
{
	int t,i,j, len;
	cin>>t;
	string S[t];
	for(i=0; i<t; i++)
	cin>>S[i];
	int flag[t];
	
	for(i=0; i<t; i++)
	{
		len = S[i].size();
//		cout<<"Length is : "<<len<<endl;
		flag[i] = 1;
		for(j=0; j<len-1; j++)
		{
			if(abs(((int)S[i][j+1] - (int)S[i][j])) != abs(((int)S[i][len-1-j] - (int)S[i][len-j-2])))
			{
				flag[i] = 0;
				break;
			}

//			cout<<"Fisrt is : "<<abs(((int)S[i][j+1] - (int)S[i][j]))<<endl;
//			cout<<"Second is : "<<abs((int)S[i][len-1-j] - (int)S[i][len - j-2])<<endl;
		}
	}
	
	for(i=0; i<t; i++)
	{
		if(flag[i])
		cout<<"Funny"<<endl;
		else cout<<"Not Funny"<<endl;

//		cout<<S[i]<<endl;
	}

}
