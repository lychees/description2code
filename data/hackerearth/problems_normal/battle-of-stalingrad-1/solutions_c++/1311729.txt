#include <iostream>
#include <math.h>
#include <cmath>
using namespace std;

int main()
{
	long long t;
	cin>>t;
	long int x1,y1,x2,y2,x3,y3,x,y;
	while(t--){
		cin>>x1>>y1>>x2>>y2>>x3>>y3>>x>>y;
		long double A=abs((x1*(y2-y3)+x2*(y3-y1)+x3*(y1-y2))/2.0);
		
			long double a1=abs((x*(y2-y3)+x2*(y3-y)+x3*(y-y2))/2.0);
			long double a2=abs((x1*(y-y3)+x*(y3-y1)+x3*(y1-y))/2.0);
			long double a3=abs((x1*(y2-y)+x2*(y-y1)+x*(y1-y2))/2.0);
			
			//cout<<(A)<< " " <<(a1+a2+a3)<<endl;
			if(a1+a2+a3 == A) cout<<"INSIDE\n";
			else cout<<"OUTSIDE\n";
		
	}
	return 0;
}
