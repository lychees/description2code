#include <iostream>
using namespace std;

int main()
{
   int t,n,b,ans;
   scanf("%d",&t);
   while(t--)
   {
   	scanf("%d%d",&n,&b);
   	if(n%b==0) ans = n;
   	else if(b%2==1) {
   		ans = n - n%b;
   		if(n%b > b/2) ans+=b;
   	}
   	else {
   		ans = n - n%b ;
   		if(n%b >= b/2)  ans+=b;
   	}
   	printf("%d\n",ans);
   }
}
