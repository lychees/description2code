#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll sq_root(ll a)
{
	ll l=0,r=1000000001,mid;
	while(r>l)
	{
		mid = l + ((r - l)>>1);
		if( mid*mid < a)
			l = mid + 1;
		else
			r = mid;
	}
	if( l*l == a )
		return l;
	else
		return l - 1;
}
int main()
{
	ll l,r,ans,t;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld%lld",&l,&r);
		ans = sq_root(r) - sq_root(l - 1);
		printf("%lld\n",ans);
	}
	return 0;
}
