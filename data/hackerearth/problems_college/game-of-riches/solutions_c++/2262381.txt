#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    
    int t;
    cin>>t;
    int n,g;
    int m,k,c;
    while(t--)
    {
    	
    	cin>>n>>g;
    	int *arr=new int[n];
    	for(int i=0;i<n;i++)cin>>arr[i];
    	m=arr[0];
    	sort(arr,arr+n);
    	
    	int i=0;int j=n-1;int t;
    	while(i<j)
    	{
    		 t=arr[i];arr[i]=arr[j];arr[j]=t;
    		 i++;j--;
    	}
    	c=0;
    	for( j=0;j<n;j++)
    	{
    		
    	if(arr[j]==m)
    	{
    		k=j;
    	t=k%g;
    	for(i=t;i<n;i+=g)
    	if(arr[i]>m)c++;
    	break;
    	}
    	
    	}
    	cout<<c<<endl;
    	
    	
    }
    return 0;
}
