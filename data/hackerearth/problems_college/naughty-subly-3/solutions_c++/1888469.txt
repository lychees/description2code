#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
    int n,coun=0;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++)
    {
        cin>>a[i];
    }
    sort(a,a+n);
    for(int i=0;i<(n-1);i++)
    {
        if(a[i] == a[i+1])
            coun++;
    }
    cout<<coun<<endl;
}
