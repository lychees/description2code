#include <bits/stdc++.h>
using namespace std;
#define fast ios::sync_with_stdio(false); cin.tie(0); cout.tie(0)

int main(){
	fast;
	int t,a[50050],rem[50050],c,n; cin>>t;
	while(t--){
		cin>>a[0]>>a[1]>>c>>n;
		for(int i=2;i<n;i++) a[i]=(a[i-1]*c+a[i-2])%10;
		
		rem[0]=a[0];
		for(int i=0;i<n-1;i++){
			rem[i+1]=(rem[i]*10+a[i+1])%41;
		}
		if(!rem[n-1]) cout<<"YES"<<endl;
		else cout<<"NO"<<endl;
	}
	return 0;
}
