#include <bits/stdc++.h>
using namespace std;

int main()
{
    int N;
    cin>>N;
    while(--N>=0)
        {
        int T;cin>>T;
        vector <long long int> ar(T,0);
        for(int i=0;i<T;i++) 
            cin>>ar[i] ;   
        long int pos=0,posb=0,count=0,max=0;
        long long int tprofit=0,cost=0;
        while(ar[max])
        {   count=0,cost=0;
            for(int i=max;i<T;i++)
                 if(ar[i]>=ar[max])
                   max=i;
            for(int i=pos;i<max;i++) 
                count++,cost+=ar[i];
                tprofit+=((count*ar[max])-cost);
                pos=++max;
        }
        cout<<tprofit<<endl;
        }
    return 0;
}
