#include <bits/stdc++.h>
using namespace std;
int N,Cell[12][12],Ans;

void Dfs(int x,int y)
{
        if(x==N-1 && y==N-1)
            { Ans++; }
        Cell[x][y]=0;
	if(Cell[x+1][y]==1 && x+1<=N-1)  Dfs(x+1,y);
	if(Cell[x][y+1]==1 && y+1<=N-1)  Dfs(x,y+1);
	if(Cell[x-1][y]==1 && x-1>0)     Dfs(x-1,y);
	if(Cell[x][y-1]==1 && y-1>0)     Dfs(x,y-1);
	Cell[x][y]=1;
}

int main()
{
	int i,j,t;
	cin>>t;
	while(t--)
    {
	cin>>N;
	for(i=0;i<N;i++)
    {
	  for(j=0;j<N;j++)
      {
        cin>>Cell[i][j];
      }
    }
    Ans=0;
	Dfs(1,1);
	if(Ans>=1)
    {
        cout<<"POSSIBLE"<<endl;
    }
    else{
        cout<<"NOT POSSIBLE"<<endl;
    }
	//printf("%d",Ans);
    }
}
