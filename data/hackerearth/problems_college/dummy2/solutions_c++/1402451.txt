#include <iostream>
using namespace std;
string s1,s;
int main()
{
    long long i,j,k,t,n;
    scanf("%lld",&t);
    while(t--)
    {
    	cin >> s;
    	j=0;
    	s1="";
    	n=s.length();
    	for(i=0;i<n;i++)
    	{
    		if(s[i]>='0'&&s[i]<='9')
    		{
    			j=s1.length();
    			k=s[i]-'0';
    			k--;
    			for(;k>0;k--)
    			s1.append(s1.begin(),s1.begin()+j);	
    		}
    		else
    		{
    			s1+=s[i];
    		}
    	}
    	cout << s1 << "\n";
    }
    return 0;
}
