#include <bits/stdc++.h>
using namespace std;
#define ll long long int
map<int,int>mp;
map<int,int>::iterator it;
int main()
{
	ios::sync_with_stdio(false);
	
	int t;
	cin >> t;
	while(t--)
	{
		int n,i;
		ll sum=0,ans=0;
		cin >> n;
		mp.clear();
		for(i=0;i<n;i++)
		{
			int x,y;
			cin >> x >> y;
			mp[x]++;
			mp[y]--;
		}
		for(it=mp.begin();it!=mp.end();it++)
		{
			sum+=it->second;
			ans = max(ans,sum);
		}
		cout << ans << endl;
	}
	
    return 0;
}
