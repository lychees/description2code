#include<bits/stdc++.h>
using namespace std;
int n,m;
pair<int,pair<int,int> > p[100005];
int par[100005];
int size[100005];
void initialise()
{   int i;
   for(i=1;i<=n;i++)
   {  par[i]=i;
      size[i]=1;
   }
}
int find(int x)
{   if(par[x]!=x)
    return find(par[par[x]]);
    return x;
}
void join(int x,int y)
{   int x1=find(x);
    int y1=find(y);
    if(x1==y1)
    return;
    if(size[x1]>size[y1])
    {   par[y1]=x1;
         size[x1]+=size[y1];
	}
	else
	{  par[x1]=y1;
	   size[y1]+=size[x1];
	}
}
int main()
{    scanf("%d %d",&n,&m);
    int i;
    for(i=1;i<=m;i++)
    {  scanf("%d %d %d",&p[i].second.first,&p[i].second.second,&p[i].first);
	}
	sort(p+1,p+m+1);
	initialise();
	long long int megh=0;
	for(i=1;i<=m;i++)
	{   int x=p[i].second.first;
	   int y=p[i].second.second;
	   if(find(x)!=find(y))
	   {  megh+=p[i].first;
	      join(x,y);
	   }
	   
	}
	printf("%lld\n",megh);
	return 0;
}
