#include <stdio.h>

int main() {
	long long t, n, m, a, b, c, res;
	
	scanf("%lli", &t);
	
	while (t--) {
		scanf("%lli %lli %lli %lli %lli", &n, &m, &a, &b, &c);
		
		if ((a + b) < (2 * c)) {
			if (n > m) {
				res = m * c * 2 + (n - m) * a;
			} else if (m > n) {
				res = n * c * 2 + (m - n) * b;
			} else {
				res = n * c * 2;
			}
		} else {
			res = n * a + m * b;
		}
		
		printf("%lli\n", res);
	}
    return 0;
}
