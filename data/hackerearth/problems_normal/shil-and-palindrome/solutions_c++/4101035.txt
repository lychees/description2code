#include <iostream>
using namespace std;

int main()
{
    int i,h[26],ctr=0,k,idx=-1;
    string s;
    cin >> s;
    for(i=0;i<26;i++)
    {
    	h[i] = 0;
    }
    for(i=0;i<s.size();i++)
    {
    	h[s[i]-97]++;
    }
    for(i=0;i<26;i++)
    {
    	if(h[i]%2==1)
    	{
    		ctr++;
    		idx = i;
    	}
    }
   	if(ctr > 1)
   	{
   		cout << -1 ;
   	}
   	else
   	{
   		if(s.size()%2 == 0)
   		{
   			for(i=0;i<26;i++)
   			{
   				k = h[i];
   				k = k / 2;
   				while(h[i] != k)
   				{
   					printf("%c",i+97);
   					h[i]--;
   				}
   			}
   			for(i=25;i>=0;i--)
   			{
   				while(h[i] != 0)
   				{
   					printf("%c",i+97);
   					h[i]--;
   				}
   			}
   		}
   		else
   		{
   			for(i=0;i<26;i++)
   			{
   				if(h[i] > 0)
   				{
   				k = h[i];
   				k = k/2;
   				if(h[i]%2 ==1 && h[i] != 1)
   				{
   					k++;
   				}
   				while(h[i] > k && k > 0)
   				{
   					printf("%c",i+97);
   				//	cout << "chut";
   					h[i]--;
   				}
   				}
   			}
   			printf("%c",idx+97);
   			h[idx]--;
   			for(i=25;i>=0;i--)
   			{
   				while(h[i] != 0)
   				{
   					printf("%c",i+97);
   					h[i]--;
   				}
   			}
   			
   		}
   	}
    return 0;
}
