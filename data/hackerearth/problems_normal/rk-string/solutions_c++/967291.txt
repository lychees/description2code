#include<bits/stdc++.h>
using namespace std;
typedef long long int int64;
string s;
int t,n,k,pre;
int64 cnt[1111111],sum[1111111];
int64 ans=0;
int64 go(int idx){
	if(idx>=s.length())
	return 0;
	if(s[idx]=='R')
	sum[idx+1]=sum[idx]+1;
	else
	sum[idx+1]=sum[idx];
	int64 ret=0;
	if(sum[idx+1]>=n)
	ret=cnt[sum[idx+1]-n];
	cnt[sum[idx+1]]++;
	return ret+go(idx+1);
}
int64 go1(int idx){
	if(idx>=s.length())
	return 0;
	if(s[idx]=='K')
	sum[idx+1]=sum[idx]+1;
	else
	sum[idx+1]=sum[idx];
	int64 ret=0;
	if(sum[idx+1]>=k)
	ret=cnt[sum[idx+1]-k];
	cnt[sum[idx+1]]++;
	return ret+go1(idx+1);
}	
int main(){
	cin>>t;
	while(t--){
		cin>>n>>k;
		cin>>s;
		for(int i=0;i<=s.length();i++){
			cnt[i]=0;
			sum[i]=0;
		}
		cnt[0]=1;
		cout<<go(0)<<" ";
		for(int i=0;i<=s.length();i++){
			cnt[i]=0;
			sum[i]=0;
		}
		cnt[0]=1;
		cout<<go1(0)<<endl;
	}
	return 0;}
