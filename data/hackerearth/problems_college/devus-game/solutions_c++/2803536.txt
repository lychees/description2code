#include<iostream>
using namespace std;

int main()
{
int t,n,i,a[100000],nxt,countt;
cin>>t;

while(t--)
{
    nxt=0;
    countt=0;
cin>>n;
for(i=0;i<n;i++)
    cin>>a[i];

while(countt<n)
{
    if(a[nxt]==-1)
    break;

 nxt=a[nxt];
countt++;
}

if(a[nxt]==-1)
    cout<<"Win\n";
else
    cout<<"Lose\n";

}

}
