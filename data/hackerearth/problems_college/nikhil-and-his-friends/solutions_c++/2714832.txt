#include<iostream>
#include<bits/stdc++.h>

using namespace std;


int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int n,k;
		cin>>n>>k;
		long long a[n];
		int i,j,flag=0;
	
		for(i=0;i<n;i++)
		{
			cin>>a[i];
		}
		unsigned long long sum=0;
		for(i=0;i<n;i++)
		{
			sum =0;
			for(j=i;j<n;j++)
			{
				sum = sum+a[i];
				
				if(sum ==k)
				{
					flag=1;
					break;
				}
				else
				if(sum >k)
				{
					sum =0;
					break;
				}
			}
			if(flag ==1)
			{
				break;
				
			}
		}
		if(flag ==1)
		{
			cout<<"YES"<<"\n";
			continue;
		}
		for(i=0;i<n;i++)
		{
			sum=0;
			for(j=i;j<n;j++)
			{
				sum = sum+a[j];
				
				if(sum ==k)
				{
					flag=1;
					break;
				}
				else
				if(sum >k)
				{
					sum =0;
					break;
				}
			}
			if(flag ==1)
			{
				break;
			}
		}
		if(flag ==1)
		{
			cout<<"YES"<<"\n";
		}
		else
		cout<<"NO"<<"\n";
	}
}
