for _ in range(int(raw_input())):
    dic = {}
    st = raw_input()
    for i in range(len(st)):
        if dic.get(st[i]):
            dic[st[i]] += 1
        else:
            dic[st[i]] = 1
    for i in dic.values():
        if i % 2 != 0:
            print("-1")
            break
    else:
        print("1")
