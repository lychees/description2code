#include <iostream>
#include <bits/stdc++.h>
using namespace std;

string words[101];
bool edge[26][26];
int out[26];
bool check[26];

string order[26];

void topological()
{
	int left=0,rank=0;
	for(int i=0;i<26;++i)
	{
		out[i]=0;
		for(int j=0;j<26;++j)
			out[i]+=edge[i][j];
		
		if(check[i]) left++;
	}
	
	while(left>0)
	{
		string curr="";
		for(int i=0;i<26;++i)
		{
			if(out[i]==0 && check[i])
			{
				curr+=char('a'+i);
				left--;
				out[i]=-1;
			}
		}
		order[rank]=curr;
		
		for(int i=0;i<curr.length();++i)
		{
			for(int j=0;j<26;++j)
			{
				if(out[j]>0 && edge[j][curr[i]-'a']) out[j]--;
			}
		}
		rank++;
	}
}
int main()
{
    int n;
    cin>>n;
    for(int i=0;i<n;++i)
    	cin>>words[i];
    
    for(int i=0;i<n;++i)
    	for(int j=0;j<words[i].length();++j)
    		check[words[i][j]-'a']=1;
    
    for(int i=0;i<n-1;++i)
    {
    	for(int j=0;j<words[i].length() && j<words[i+1].length();++j)
    	{
    		if(words[i][j]!=words[i+1][j])
    		{
    			edge[words[i+1][j]-'a'][words[i][j]-'a']=1;
    			break;
    		}
    	}
    }
    topological();
    
    for(int i=0;i<26;++i)
    {
    	if(order[i]=="") break;
    	cout<<order[i]<<endl;
    }
    
    return 0;
}
