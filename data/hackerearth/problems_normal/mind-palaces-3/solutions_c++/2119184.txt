#include<bits/stdc++.h>

using namespace std;
int main()
{
    int n,m;
    cin>>n>>m;
    int a[n][m],i,j;
    for(i=0;i<n;i++){
    	for(j=0;j<m;j++){
    		cin>>a[i][j];
    	}
    }
    int t;
    cin>>t;
    while(t--){
    	int b;
    	cin>>b;
    	i=0,j=m-1;
    	int ind1=-2,ind2=-2;
   		 while(i<n&&j>=0){
    		
    	    if(a[i][j]==b){
    			ind1=i,ind2=j;
    			break;
    		}	
    		if(a[i][j]>b){
    			j--;
    		}
    		else
    		i++;
    	}
    	if(ind1==-2&&ind2==-2)
    	cout<<"-1 -1\n";
    	else
    	cout<<ind1<<" "<<ind2<<"\n";
    	
    }
    return 0;
}
