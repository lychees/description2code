#include <iostream>
using namespace std;

int main()
{
    int T,N;
    cin>>T;
    for(int i=0;i<T;i++)
    {
    	cin>>N;
    	int c=0;
    	for(int j=1;j<=N;j++)
    	{
    		for(int k=1;k<=N;k++)
    		{	if(j!=k)
    			{	int sum=j+k,pro=j*k;
    				if(pro%sum==0)
    				{	
    					++c;
    				}
    			}
    		}
    	}
    	cout<<c/2<<endl;
    }
    return 0;
}
