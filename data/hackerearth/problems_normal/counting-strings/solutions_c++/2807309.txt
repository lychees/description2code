#include<bits/stdc++.h>
using namespace std;
int main()
{
	//freopen("in.txt","r",stdin);
	int t;
	cin>>t;
	while(t--)
	{
		string s;
		cin>>s;
		int i=0;
		while(s[i])
		{
		  if(s[i]=='z')
		   s[i]='a';	
           i++;			
		}
		int len=s.length();
		int stopper=-1;
		i=0;long long c=0;
		while(s[i])
		{
			if(s[i]=='a')
			{
			    c+=(len-i)*(i-stopper);
				stopper=i;	
			}
	        i++;	
		}
		cout<<c<<"\n";
	}
	return 0;
	
}
