#include <iostream>
#include <typeinfo>
using namespace std;

int check(const int a[], int n)
{   
    while(--n>0 && a[n]==a[0]);
    return n!=0;
}

bool transform(int* arr, int n)
{
	int prev = 1;
	for(int i=0; i<n-1; i++)
	{
		if(prev == 1 && arr[i+1] == 1)
		{
			prev = arr[i];
			arr[i] = 1;
		}
		else
		{
			prev = arr[i];
			arr[i] = 0;
		}
	}
	arr[n-1] = prev;
	return check(arr,n);
}

int main()
{
    int t=0;
    cin>>t;
    while(t--)
    {
    	int n,m;
    	cin>>n>>m;
    	int arr[n];
    	for(int i=0; i<n; i++)
    		scanf("%d", &arr[i]);

   		while(m-- && transform(arr, n));
   			
   			
   		for(int i=0; i<n; i++)
   			printf("%d ", arr[i]);
   			
   		cout<<"\n";
    }
    return 0;
}
