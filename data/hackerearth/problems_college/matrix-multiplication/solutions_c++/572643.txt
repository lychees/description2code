#include <iostream>

using namespace std;

int main()
{
        int test_cases,COUNT;
        cin>>test_cases;
        while(test_cases > 0)
        {
                int X1,X2,Y1,Y2;
                cin>>X1>>Y1>>X2>>Y2;
                COUNT=0;
                if(X2==Y1)
                COUNT++;
                if(X1==Y2)
                COUNT++;
                cout<<COUNT<<endl;
                test_cases --;
        }
        return 0;
}
