#include <iostream>
#include<bits/stdc++.h>
using namespace std;

vector <pair<int,int> > vec;

int main()
{
    int n,x,y,d,ans=0,i,max1;
    cin>>n;
    
    while(n--){
    	cin>>x>>y>>d;
    	vec.push_back(make_pair(x,x+d));
    }
    sort(vec.begin(),vec.end());
    
    max1=1;
    vector<pair<int,int> >::iterator it;
  
    cout<<endl;
    it=vec.begin();
    for(it=vec.begin();it!=vec.end();it++){
    	max1=max(max1,it->first);
    	
    	if(it->second >= max1){
    		ans=ans+it->second-max1+1;
    		max1=it->second+1;
    	}
    }
    cout<<ans<<endl;
    return 0;
}
