#include <iostream>
#include <string>
using namespace std;

int check_str(string str){
	long size = str.size();
	int arr[26] = {0};
	for(int j = 0 , i = size -1; j < size/2 ; ++j, --i)
	{
		arr[str[j] - 'a']++;
		arr[str[i] - 'a']++;
	}
	for(int j = 0 ; j < 26 ; ++j)
	{
		if(arr[j] % 2 != 0)
			return -1;
	}
	return 1;
}
int main()
{
 	long input;
 	string str;
	cin>>input;
	
	for(int i = 0 ; i < input ; ++i){
		cin>>str;
		cout<<check_str(str)<<endl;
	}
	
}
