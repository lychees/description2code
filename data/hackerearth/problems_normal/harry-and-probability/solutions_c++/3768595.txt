#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		double a,b,c,d;
		scanf("%lf%lf%lf%lf",&a, &b, &c, &d);
		double AB=b-a;
		double CD=d-c;
		double BC=c-b;
		double x=BC/2;
		double alpha=acos(x/AB);
		double totel=acos(-1);
		if(BC<=(AB+CD))
			printf("%lf\n",alpha/totel);
		else
			printf("0.000000\n");	
	}
	return 0;
	
}
