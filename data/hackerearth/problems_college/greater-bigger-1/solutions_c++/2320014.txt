#include <iostream>
#include <stdio.h>
#include <algorithm>

using namespace std;

int main()
{
	#ifndef ONLINE_JUDGE
		freopen("I.in","r",stdin);
	#endif
		
	int t,lol;
	cin>>t;
	while(t--)
	{
		string s;
		cin>>s;
		int n=s.length(),pos=-1;
		for(int i=n-1;i>0;i--)
		{
			char ch=123;
			if(s[i]>s[i-1])
			{
				lol=i-1;
				for(int j=i;j<n;j++)
				{
					if(s[j]>s[i-1])
					{
						ch=min(ch,s[j]);
						pos = j;
					}
				}
				break;
			}
		}
		if(pos==-1) puts("no answer");
		else
		{
			swap(s[lol],s[pos]);
			sort(s.begin()+lol+1,s.end());
			cout<<s<<endl;
		}
	}
	return 0;
}
