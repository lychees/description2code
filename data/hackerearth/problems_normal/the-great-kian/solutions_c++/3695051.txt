#include <iostream>
using namespace std;

int main()
{
	unsigned long n = 0;
	cin >> n;
	
	// unsigned long long array[n];
	unsigned long long value1 = 0, value2 = 0, value3 = 0;
	
	int percentile = -1;
	
	for(unsigned long i = 0; i < n; i++){
		unsigned long long temp = 0;
		cin >> temp;
	
		percentile = i % 3;
		
		if(percentile == 0){
			value1 += temp;
		}else if(percentile == 1){
			value2 += temp;
		}else{
			value3 += temp;
		}
	}
	
	cout << value1 << " " << value2 << " " << value3;
	
	return 0;
}
