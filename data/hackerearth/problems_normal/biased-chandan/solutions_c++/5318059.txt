#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int main()
{
    stack <ll> s;
    ll n;
    cin >> n;
    while(n--){
    	ll elem ;
    	cin >> elem ;
    	if(elem == 0){
    		if(!s.empty())
    		s.pop();
    		else
    		continue ;
    	}
    	else
    	s.push(elem) ;
    }
    
    ll sum = 0; 
    
    while(!s.empty()){
    	sum += s.top() ;
    	s.pop() ;
    }
    
    cout << sum << endl ;
    
    return 0;
}
