#include <iostream>
#include<utility>
#include<algorithm>
#include<vector>
#include<cstdio>
using namespace std;

int main()
{
	int n,i,k,q,ans;
	vector<pair<int,int>>v(100006);
	scanf("%d %d",&n,&q);
	for(i=0;i<n;i++)
	{scanf("%d",&v[i].second);}
		for(i=0;i<n;i++)
	{scanf("%d",&v[i].first);}
    sort(v.begin(),v.begin()+n);
    reverse(v.begin(),v.begin()+n);
    while(q--)
    {
    	scanf("%d",&k);
    	ans=0;
    	for(i=0;i<k;i++)
    	ans+=v[i].second;
    	printf("%d\n",ans);
    }
    
    return 0;
}
