int pos[25];
int arr[25];
int k;
vector<int > v;
int main()
{
	scanf("%d", &k);
	for (int i = 0;i<k;i++)scanf("%d", arr + i);
	deque<int > dq;
	for (int i = 1;i <= k;i++)
		dq.push_back(i);
	for (int i = 0;i<k;i++)
	{
		while (arr[i]--)
		{
			int t = dq.front();
			dq.pop_front();
			dq.push_back(t);
		}
		int temp = dq.back();
		dq.pop_back();

	}
	for (int i = 0;i < v.size();i++)cout << v[i] << endl;

}
