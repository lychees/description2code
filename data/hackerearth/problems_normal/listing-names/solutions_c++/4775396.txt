#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    vector<string> s;
    string temp;
    for(int i=0;i<t;i++)
    {
    	cin>>temp;
    	s.push_back(temp);
    }
    sort(s.begin(),s.end());
    int count=1;
    for(int i=0;i<t-1;i++)
    {
    	if(s[i]==s[i+1]) count++;
    	else
    	{
    		cout<<s[i]<<" "<<count<<"\n";
    		count=1;
    	}
    }
    cout<<s[t-1]<<" "<<count<<"\n";
    return 0;
}
