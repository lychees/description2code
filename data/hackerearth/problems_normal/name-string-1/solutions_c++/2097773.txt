#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ll int test;
	scanf("%lld",&test);
	while(test--)
	{
		ll int a[3];
		ll int i,ans;
		for(i=0;i<3;i++)
		{
			scanf("%lld",(a+i));
		}
		sort(a,a+3);
		if(a[1]==a[2])
		{
			ans=a[1]+a[2];
		}
		else
		{
			ans=2*a[1]+1;
		}
		printf("%lld\n",ans);
	}
	return 0;
}
