#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
#define mod 1000000007

ll combi(int n,int k)
{
	if(k>n)
	{
		return 0;
	}
    ll ans=1;
    k=k>n-k?n-k:k;
    int j=1;
    for(;j<=k;j++,n--)
    {
        if(n%j==0)
        {
            ans*=n/j;
        }else
        if(ans%j==0)
        {
            ans=ans/j*n;
        }else
        {
            ans=(ans*n)/j;
        }
        if(ans>mod)
        {
        	ans=ans%mod;
        }
    }
    return ans;
}

{
	ios::sync_with_stdio(false);
	int t;

	{
		string a,b;
		cin>>a>>b;
		int l1[150]={0},l2[150]={0};
		for(int i=0;i<a.length();i++)
		{
			++l1[a[i]];
		}
		ll ans=1;
		for(int i=0;i<b.length();i++)
		{
			ans=(ans*l1[b[i]])%mod;
			--l1[b[i]];
		}
		cout<<ans<<"\n";
	}

}
