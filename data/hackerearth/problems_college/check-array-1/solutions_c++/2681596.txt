#include <bits/stdc++.h>
using namespace std;
#define MAX 100005
int main()
{
    int t,n,a[MAX];
    cin>>t;
    while(t--){
    	cin>>n;
    	long long sum = 0;
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    		sum+=a[i];
    	}
    	long long left=0,right=sum;
    	if(left==(sum-a[0])){
    		cout<<"YES"<<endl;
    		continue;
    	}
    
    	left=a[0];
    	right = sum - a[0];
    	int flag = 0;
    	for(int i=1;i<n;i++){
    		if(left==(right-a[i]))
    		{
    		//	cout<<i<<endl;
    			flag = 1;break;
    			
    		}
    		else {left+=a[i];right-=a[i];}
    	}
    	if(flag==1)cout<<"YES"<<endl;
    	else cout<<"NO"<<endl;
    	}
    return 0;
}
