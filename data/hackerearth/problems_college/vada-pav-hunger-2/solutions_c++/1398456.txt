#include <bits/stdc++.h>
using namespace std ;
#define LL long long int
#define ft first
#define sd second
#define PII pair<int,int>
#define MAXN 1001
#define mp make_pair
#define f_in(st) freopen(st,"r",stdin)
#define f_out(st) freopen(st,"w",stdout)
#define sc(x) scanf("%d",&x)
#define pr(x) printf("%lld\n",x)
int N , ans ;
vector<int> adj[MAXN] ;

int dfs1(int u,int p = -1){
    int sum = 0 ;
    for(vector<int> :: iterator it = adj[u].begin() ; it != adj[u].end() ; ++it){
        if(p != *it){
            sum += dfs1(*it,u) ;
            sum += 2 ;
        }
    }
    return sum ;
}
int dfs2(int u,int p = -1,int lev = 0){
    ans = max(ans,lev) ;
    for(vector<int> :: iterator it = adj[u].begin() ; it != adj[u].end() ; ++it){
        if(p != *it){
            dfs2(*it,u,lev+1) ;
        }
    }
}
int main(){
    sc(N) ;
    for(int i=1;i<N;i++){
        int u , v ;
        sc(u) ;
        sc(v) ;
        adj[u].push_back(v) ;
        adj[v].push_back(u) ;
    }
    dfs2(1) ;
    cout << 2*(N-1) - ans << endl ;
    return 0 ;
}
