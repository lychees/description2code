#include<bits/stdc++.h>
using namespace std;
struct ghost{
	int titles;
	int age;
};
int main(){
	cout.sync_with_stdio(0);
	cin.tie(0);
	int n,m,i,a,maxT=0,maxA=1;
	map<int,int> myMap;
	ghost tmp;
	cin>>n>>m;
	for(i=0;i<n;++i){
		cin>>a;
		if(myMap.count(a))
			++myMap[a];
		else
			myMap[a]=1;
		if(myMap[a]>maxT){
			maxT=myMap[a];
			maxA=a;
		}else if(myMap[a]==maxT)
			maxA=maxA>a?maxA:a;
		cout<<maxA<<" "<<maxT<<"\n";
	}
	return 0;
}
