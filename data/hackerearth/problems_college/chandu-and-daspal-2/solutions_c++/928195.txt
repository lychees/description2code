ll gcd(ll a, ll b) {
	if (b == 0) return abs(a);
	else return gcd(b, a % b);
}

int main()
{
	int T; cin >> T;
	while (T--) {
		int x, y; string s; cin >> x >> y >> s;
		for (;;) {
			int d = gcd(x, y);
			x -= d; y -= d;
			if (x == 0 || y == 0) break;
			if (s == "Chandu") s = "Daspal";
			else if (s == "Daspal") s = "Chandu";
		}
		cout << s << endl;
	}
}
