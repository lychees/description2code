#include<iostream>
#include<cmath>
#include<cstring>
#include<stdio.h>
using namespace std;
int arr[50];
int binary(int n)
{
	int k=0;
	while(n)
	{
		if(n&1)
			arr[k++]=1;
		else
			arr[k++]=0;
		n>>=1;
	}
	return k;
}
void swap(int *a,int *b)
{
	*a=*a^*b;
	*b=*a^*b;
	*a=*a^*b;
}
void reverse(int h)
{
	int l=0;
	while(l<h)
	{
		swap(&arr[l],&arr[h]);
		l++;h--;	
	}
}
int main()
{
	int t,n,l,i,sum,j;
	scanf("%d",&t);
	while(t--)
	{
		memset(arr,0,sizeof(arr));
		sum=0;
		scanf("%d",&n);
		if(!(n&(n>>1)))
		{
			cout<<n<<endl;
			continue;
		}
		l=binary(n);
		reverse(l-1);
		for(i=0;i<l;i++)
		{
			if(arr[i]==1&&arr[i+1]==1)
				break;
		}
		if(i==0)
			cout<<pow(2,l)<<endl;
		else
		{
			for(j=i-1;j>0;j--)
			{
				if(arr[j]==0&&arr[j-1]==0)
					break;
			}
			if(j==0)
				cout<<pow(2,l)<<endl;
			else
			{
				arr[j]=1;
				for(i=j+1;i<l;i++)
					arr[i]=0;
				reverse(l-1);
				for(i=0;i<l;i++)
				{
					if(arr[i]==1)
						sum+=pow(2,i);
				}
				cout<<sum<<endl;
			}
		}
		
		/*for(i=0;i<l;i++)
			cout<<arr[i];
		cout<<endl;
		*/
	}
	return 0;
}
