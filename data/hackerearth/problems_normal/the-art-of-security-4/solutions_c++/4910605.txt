#include <bits/stdc++.h>
using namespace std;
int e1=0,e2=0,ways=0,n;
void countNo(int num,int *a)
{
	if(num==n)
	{
		ways++;
//		cout<<ways<<"\n";
		return;
	}
	if(e1+a[num]<=e2)
	{
		e1=e1+a[num];
		countNo(num+1,a);
		e1=e1-a[num];
	}							///if one can go in first entrance he can surely go in second entrance;
	e2=e2+a[num];
	countNo(num+1,a);
	e2=e2-a[num];
}
int main()
{
	cin>>n;
	int i=0,a[n],sum=0;
	while(i<n)
	{
		cin>>a[i];
		sum=sum+a[i];
		i++;
	}
	sort(a,a+n);
	do
	{
		e1=0;
		e2=0;
		countNo(0,a);
	}while(next_permutation(a,a+n));
	cout<<ways<<" "<<sum<<"\n";
	if(ways>sum)
	{
		cout<<"We will win!\n";
	}
	else
	{
		cout<<"Got no way out!\n";
	}
}
