#include<iostream>
#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define ull unsigned long long
int main()
{
	
    int n;
   cin>>n;
  //  inp(n);
    //arr.resize(n);
    vector<ll> arr;
    arr.resize(n);
    for(int i=0;i<n;i++){ cin>>arr[i];}
    int bit[62],bit1[62];
    memset(bit,0,sizeof(bit));
    memset(bit1,0,sizeof(bit1));
    unsigned long long temp;
    for(int i=0;i<n;i++){
     for(int j=0;j<62;j++){
        temp=(ull)pow(2,j);
        if(arr[i]&temp)bit[j]++;
     }
    }
   // int bit1[62];
    int OR[62];
    memset(OR,0,sizeof(OR));
    for(int i=0;i<n;i++){
      memset(bit1,0,sizeof(bit1));
      for(int j=0;j<62;j++){
        temp=(ull)pow(2,j);
        if(arr[i]&temp)bit1[j]++;
      }
     for(int j=0;j<62;j++) bit[j]-=bit1[j];

     for(int j=0;j<62;j++){
      if((bit1[j]&&bit[j]==n-1)||(bit1[j]==0&&bit[j]==0)){
      }else
       OR[j]=true;
     }
     for(int j=0;j<62;j++) bit[j]+=bit1[j];
    }
    ull ans=0;
    for(int i=0;i<62;i++){
     if(OR[i]) ans+=(ull)pow(2,i);
    }
    cout<<ans<<endl;
    return 0;
}
