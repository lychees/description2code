#include<bits/stdc++.h>
using namespace std;

void solve1(int n)
{
    char a[2*n+1][2*n+3];
    int i,j;
    for(i=1;i<=2*n;i++)
        for(j=0;j<2*n+3;j++)
            a[i][j]=' ';
    for(i=0;i<n;i+=2)
    {
        for(j=0;j<n-i;j++)
        {
            a[n-j][j+i]='/';
        //  cout<<n-j<<" "<<j+i<<" /\n";
            a[n+j+1][j+i+2]='\\';
        //  cout<<n+j+1<<" "<<j+i+2<<" \\ \n";
        }
    }
    for(i=1;i<=n;i+=2)
    {
        for(j=0;j<i;j++)
        {
            a[n-j][n-1-j+i]='\\';
        //  cout<<n-j<<" "<<n-1-j+i<<" \\ \n";
            a[n+j+1][n-1-j+i+2]='/';
        //   cout<<n+j+1<<" "<<n-1-j+i+2<<" / \n";
        }
    }
    for(i=1;i<=2*n;i++)
    {
        for(j=2*n+2;a[i][j]==' ';j--);
        a[i][j+1]=0;
    }
    for(i=1;i<=2*n;i++)
        puts(a[i]);
}

int main()
{
    int n;
    cin>>n;
    if(n%2)
    {
        solve1(n);
        solve0(n+1);
    }
    else
    {
        solve0(n);
        solve1(n+1);
    }
