#include<cstdio>
#include<cstring>
#include<bits/stdc++.h>
using namespace std;
#define MAX 1000010
bool prime[MAX];
int p[1000001];
inline void gen_primes() 
    { 
      int i,j,c=0; 
      for(i=2;i<=MAX;i++)prime[i]=true;
       for(i=2;i<=(int)sqrt(MAX);i++) 
         if (prime[i]) 
           for(j=i;j*i < MAX;j++) prime[i*j] = false; 
   }
   
   int main()
   {
   	gen_primes() ;
   	int a,b,tc;
   	cin >> tc;
   	while(tc--)
   	{
   		cin >>a>>b;
   		long long sum = 0;
   		for(int i=a;i<=b;i++)
   		{
   			if(prime[i])sum+=i;
   		}
   		cout<<sum<<endl;
   	}
   }
