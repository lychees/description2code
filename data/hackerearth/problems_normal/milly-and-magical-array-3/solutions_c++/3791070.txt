#include <iostream>
using namespace std;

int main()
{
    unsigned long cases,n,k,num,ans,a,tot;
    cin>>cases;
    while(cases--){
    	ans=0;
    	cin>>n>>k;
    	cin>>a;
    	n--;
    	while(n--){
    		cin>>num;
    		if(a<num){
    			a=num;
    			continue;
    		}
    		tot=(a-num)/k+1;
    		ans+=tot;
    		num=num+tot*k;
    		a=num;
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
