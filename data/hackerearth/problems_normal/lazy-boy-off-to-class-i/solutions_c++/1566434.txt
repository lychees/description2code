#include<iostream>
#include<stdlib.h>
#include<math.h>
using namespace std;
int tim[2004];
int compare (const void * a, const void * b)
{
  return ( *(int*)a - *(int*)b );
}
int main()
{
    int i,t,n,j,k=0,x,y;
    cin>>t;
    while(t--)
    {
        k=0;
        for(i=0;i<7;i++)
        {
            cin>>n;
            for(j=0;j<n;j++)
            {
                cin>>x>>y;
                tim[k++]=y-x;
            }
            //cout<<n<<endl;
        }
        qsort(tim,k,sizeof(int),compare);
        float x=(0.75)*k;
        //cout<<k<<" "<<x<<endl;
        int tot=ceil(x),sum=0;
        //cout<<tot<<endl;
        for(i=0;i<tot;i++)
            sum+=tim[i];
        cout<<604800-sum<<endl;
    }
    return 0;
}
