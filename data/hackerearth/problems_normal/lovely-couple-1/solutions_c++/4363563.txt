#include <iostream>
using namespace std;


int arr[1000004] = {0};

void sieve()
{
	int i,j;
	
	for( i = 2; i < 1000004; i++)
	{
		if(arr[i] == 0)
		{
			for(j = i*2; j < 1000004; j += i)
			{
				arr[j]++;
			}
		}
	}
}


int gcd(int a, int b)
{
	if(a == 0)
	{
		return b;
	}
	else
		return gcd(b%a, a);
}


int main()
{
    int t,a,b;
    sieve();
    cin >> t;
    while(t--)
    {
    	cin >> a >> b;
    	
    	int l = a*b/(gcd(a,b));
    	
    	if(arr[l] > 1 && arr[arr[l]] == 0)
    		cout << "Yes" << endl;
    	else
    		cout << "No" << endl;
    }
    
    return 0;
}
