#include<stdlib.h>
#include<stdint.h>
#include<string.h>
#include<math.h>
#include<stdio.h>
int a[100002],p[100002];
void merger(int l,int h,int m)
{
	int i=l,j=m+1,k=l;
	while(i<=m&&j<=h){
		if(a[i]>a[j])
		{
			p[k++]=a[i++];
		}
		else
		{
			p[k++]=a[j++];
		}
	}
	while(i<m+1)
	{
		p[k++]=a[i++];
	}
	while(j<h+1)
	{
		p[k++]=a[j++];
	}
	for(i=l;i<=h;i++)
	{
		a[i]=p[i];
	}
return ;
}
void parting(int l,int h)
{
	if(l<h)
	{
		parting(l,(l+h)/2);
		parting((l+h)/2+1,h);
		merger(l,h,(l+h)/2);
	}
}
int main()
{
	int k,i,max=0;
	scanf("%d",&k);
	for(i=0;i<k;i++)
	{
		scanf("%d",&a[i]);
	}
	parting(0,k-1);
	for(i=0;i<k;i++)
	{
		if(i+2+a[i]>max)
		max=a[i]+i+2;
	}
	printf("%d\n",max);
	return 0;
}
