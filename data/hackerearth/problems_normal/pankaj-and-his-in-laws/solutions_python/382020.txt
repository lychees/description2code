def lis(n):
	arr = []
	for i in n:
		low,high = 0,len(arr)-1
		while (low <= high):
			mid = (low+high)/2
			if(arr[mid] >= i):
				high = mid-1
			else:
				low = mid+1
		if(len(arr)<=low):
			arr.append(i)
		else:
			arr[low]=i
	print "{0:b}".format(len(arr))

n = input()
a = map(int,raw_input().split())
lis(a)
