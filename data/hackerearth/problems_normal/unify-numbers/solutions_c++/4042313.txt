#include<iostream>
#include<stdlib.h>
#include<algorithm>
using namespace std;
int main()
{
    int t,n,i,temp;
    long long int arr[100002],avg=0,sum=0;
    cin>>t;
    while(t-->0)
    {
                cin>>n;
                avg=sum=0;
                for(i=0;i<n;i++)
                cin>>arr[i];
                sort(arr,arr+n);
                if(n%2)
                {
                       temp=n/2;
                       avg=arr[temp];
                       for(i=0;i<n;i++)
                       sum+=abs(arr[i]-avg);
                }
                else
                {
                       temp=n/2;
                       avg=(arr[temp]+arr[temp-1])/2;
                       for(i=0;i<n;i++)
                       sum+=abs(arr[i]-avg);
                }
                cout<<sum<<endl;
    }
    
}
