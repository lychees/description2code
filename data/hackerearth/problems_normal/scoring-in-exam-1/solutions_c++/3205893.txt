#include <iostream>
#include <set>
using namespace std;

int main()
{
    int n, q;
    multiset<long>times;
    long temp;
    cin>>n>>q;
    for(int i=0; i<n; i++){
    	cin>>temp;
    	times.insert(temp);
    }
    for(int i=0; i<n; i++){
    	cin>>temp;
    }
    long ans [n+1];
    for(int i=0; i<=n; i++)ans[i]=0;
    long sum = 0;
    multiset<long>::reverse_iterator rit = times.rbegin();
    int i = 1;
    for(; rit != times.rend() ; rit++){
    	sum += (*rit);
    	ans[i++] = sum;
    }
    int query;
    for(i = 0; i<q; i++){
    	cin>>query;
    	cout<<ans[query]<<endl;
    }
    return 0;
}
