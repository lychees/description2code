#include <iostream>
#include <sstream>
using namespace std;

int main()
{
	ostringstream os;
	int n;
	cin>>n;
	while(n--){
		int x;
		cin>>x;
		int cnt = 0;
		bool flag = 1;
		for(int i=1;i*i<=x;i++){
			if(x%i == 0){
				if(i*i == x) cnt++;
				else cnt += 2;
			}
			if(cnt == 4){
				flag = 0;
				break;
			}
		}
		if(flag) os<<"NO\n";
		else os<<"YES\n";
	}
	cout<<os.str();
    return 0;
}
