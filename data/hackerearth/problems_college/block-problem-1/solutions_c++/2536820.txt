#include <iostream>
#define MOD  1000000007
using namespace std;
long powerfn( long long n, long p) 
{
		 
		  if(p == 0) return 1;
		  if(p == 1) return n;
		 
		   long long n1 = n;
		  for(int i=2;i<=p;i++) 
		  {
		    n *= n1;
		    n %= MOD;
		  }
		  return n;
}
int main()
{
    long long int N,M,i,T[10000],K[10000],P[10000],t,j,sum=0;
    T[0] = T[1] = 1;
		  T[2] = 2 ;
		  T[3] =  4;
		  T[4] = 8 ;
		 
		  P[0] = P[1] = 1;
		 
		  
		 
		  for(i=5;i<=1000;i++)
		    T[i] = (T[i-1] + T[i-2] + T[i-3] + T[i-4])%MOD;
		 
		  K[0] = 1;
		  K[1] = 1;
		  
		  cin>>t;
		 
		  while(t--)
		  {
			 cin>>N>>M;
		    for(int i=0;i<=M;i++)
		      P[i] =powerfn(T[i],N);
		 
		    for(int i=2;i<=M;i++)
		    {
		      sum = 0;
		      for(j=1;j<i;j++)
		      {
			sum += (K[j]*P[i-j])%MOD;
			sum %= MOD;
		      }
		      K[i] = (P[i] - sum);
		      K[i] = K[i]%MOD;
		    }
		    while(K[M] < 0)
		      K[M] += MOD;
		   cout<<K[M]<<endl;
		  }
		 
		 
		
		 
    return 0;
}
