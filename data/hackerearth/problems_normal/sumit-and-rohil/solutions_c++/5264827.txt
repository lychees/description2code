#include <iostream>
using namespace std;
# define NO_OF_CHARS 256
bool Anagram(string str1,string str2);
bool Anagram(string str1, string str2)
{
   
    int count1[NO_OF_CHARS] = {0};
    int count2[NO_OF_CHARS] = {0};
    int i;
    for (i = 0; str1[i] && str2[i];  i++)
    {
        count1[str1[i]]++;
        count2[str2[i]]++;
    }
    if (str1[i] || str2[i])
      return false;
    for (i = 0; i < NO_OF_CHARS; i++)
        if (count1[i] != count2[i])
            return false;
 
    return true;
}
int main()
{
	int count=0;
	int n;
	cin>>n;
	string s[n+1];
    for(int i=0;i<n+1;i++)
    {
    	getline(cin,s[i]);
    }
    
    
    
    
   for(int i=0;i<n+1;i++)
   {
   	if(s[i].size()!=0)
   	{
   for(int j=i+1;j<n+1;j++)
   {
   	if(s[j].size()!=0)
   	{
   	if (Anagram(s[i],s[j]))
   	{
   		if(s[i][0]==s[j][0] && s[i][s[i].length()-1]==s[j][s[j].length()-1])
   		{
   			s[j].clear();
  	
   			}
   			}
   	}
   	 }
   	 s[i].clear();
   	 count++;
   }
   
   }
   
   
   
   
     
    cout<<count<<endl;
    
    
    return 0;
}
