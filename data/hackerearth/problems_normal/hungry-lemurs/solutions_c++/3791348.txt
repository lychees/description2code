#include<iostream>
#include<algorithm>
using namespace std;

int main(){
	
	int m,b;
	cin>>b>>m;
	int a=0,r1,r2,mi;
	
	r1=a+b%m;
	r2=(b/m+1)*m-b;
	r1=min(r1,r2);
	mi=r1;
	int m2=m,m1=m;
	
	if(b>m){
	while(b%m1!=0){
		a++;
		m1--;
		r1=b%m1;
		r2=(b/m1+1)*m1-b;
		r1=min(r1,r2);
		if(mi>r1+a)
			mi=r1+a;
	}
	a=0;
	while(b%m2!=0){
		a++;
		m2++;
		r1=b%m2;
		r2=(b/m2+1)*m2-b;
		r1=min(r1,r2);
		if(mi>r1+a)
			mi=r1+a;
	}
	}
	else{
		mi=m-b;
		if(mi>b)
		mi=b;
	}
	
	cout<<mi;
	
	
	return 0;
}
