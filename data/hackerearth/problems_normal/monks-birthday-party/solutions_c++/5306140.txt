#include <iostream>
#include <set>
#include <string>

using namespace std;

int main()
{
    set <string> invites;
    string name;
    int T, N;
    
    ios_base::sync_with_stdio(false);
    
    cin >> T;
    while (T--)
    {
    	cin >> N;
    	invites.clear();
    	
    	for (int i = 0; i < N; i++)
    	{
    		cin >> name;
    		invites.insert(name);
    	}
    	
    	for (set <string>::iterator it = invites.begin(); it != invites.end(); it++)
    		cout << *it << "\n";
    }
    
    return 0;
}
