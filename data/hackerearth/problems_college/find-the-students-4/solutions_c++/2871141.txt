#include <iostream>
using namespace std;
class student
{
	private:
	int *arr;
	int n,counter=-1;
	public:
	void minimise()
	{	
		if(counter==0)
			return;
		else if(counter!=0)
		{
			counter=0;
			for(int i=0;i<n;i++)
			{
				if(arr[i]==0);
				else if(arr[i]==1)
				{
					arr[i]=0;
					counter++;
				}
				else if(arr[i]>1)
				{
					arr[i]=arr[i]-2;
					counter++;
				}
			}
		}
		if(counter!=0)
			cout<<counter<<endl;
		minimise();
	}
	void input()
	{
		cin>>n;
		arr=new int[n];
		for(int i=0;i<n;i++)
		{
			cin>>arr[i];
		}
	}
};
int main()
{
	student a;
	a.input();
	a.minimise();
}
