#include <iostream>
#include<bits/stdc++.h>
#include<string.h>
typedef long long int ll;
using namespace std;

ll t,i,j,k=1;
string s,s1;
int main()
{
    cin>>t;
    while(t--)
    {
    	cin>>s;
    	s1=s;
    	reverse(s.begin(),s.end());
    	if(s==s1)
    	cout<<"Palindrome"<<endl;
    	else
    	{
    		for(ll i=0;i<s.length();i++)
    		{
    			k=k*(s[i]-96);
    		}
    		cout<<k<<endl;
    	}
    	
    	
    	
    	k=1;
    }
    return 0;
}
