#include <bits/stdc++.h>

using namespace std;

bool isPrime[10000001];

void sieve(int N) {
    for (int i = 0; i <= N; i++) {
        isPrime[i] = true;
    }
    isPrime[0] = false;
    isPrime[1] = false;
    for (int i = 2; i*i <= N; i++) {
         if (isPrime[i] == true) {
             for (int j = i*i; j <= N; j += i) {
                 isPrime[j] = false;
             }
        }
    }
}

int main()
{
	int N;
	cin >> N;
	sieve(10000000);
	for (int i = 0; i < N; i++) {
		int x;
		cin >> x;
		int s = sqrt(x);
		if (x == 1) 
			cout << "NO" << endl;
		else if (isPrime[x] || (isPrime[s] && s*s == x))
			cout << "NO" << endl;
		else
			cout << "YES" << endl;
	}
    return 0;
}
