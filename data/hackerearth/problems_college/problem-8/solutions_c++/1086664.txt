#include <iostream>
#define $ while
using namespace std;

int main()
{
    long long t,f=1,s=1,a;
    cin>>t;
    if(t<=2)
    $(t--)
    {
    	cout<<1<<endl;
    }
    else
    {
    	cout<<1<<endl<<1<<endl;
    	t-=2;
	    $(t--)
	    {
	    	a=f+s;
	    	cout<<a<<endl;
	    	f=s;
	    	s=a;
	    }
    }
    return 0;
}
