#include<bits/stdc++.h>
using namespace std;
#define rep(i,n) for(i=0;i<n;i++)


int main()
{
  int t;
  cin>>t;
  while(t--)
  {
     int n,d,m,i,j;
  
     cin>>n>>m;
  
     double mat[n][m];
     rep(i,n)
     {
      rep(j,m)
      {
        double a,b,c;
        int ta,tb,tc;
       cin>>ta>>tb>>tc;
      a= (double)ta/100.0;
      b= (double)tb/100.0;
      c= (double)tc/100.0;
        mat[i][j]= (1-a)*(1-b)*(1-c);
      }
     }
     for(i=0;i<n;i++)
       {
       double mini=mat[i][0];
       int minpos=0;
         for(j=1;j<m;j++)
         {
          if(mat[i][j]<mini)
          {
            minpos=j;
            mini=mat[i][j];
          }
         }
       printf("%d ",minpos+1);
       }
       printf("\n");
  }
  return 0;
}
