#include <iostream>
#include <cstdio>
#include <algorithm>
#include <string>
using namespace std;
int j(int n, int k) {
    if(n==1) {
        return 1;
    }
    if(n%2 == 0) {
        if(k == 1) {
            return 2*j(n/2, 1);
        } else {
            return 2*j(n/2, 2) - 1;
        }
    } else {
        if(k == 1) {
            return 2*j(n/2, 2);
        } else {
            return 2*j(n/2+1, 1) - 1;
        }
    }
}

int main()
{
        int t , n;
        scanf("%d",&t);
        while(t--)
        {
        scanf("%d",&n);
        printf("%d\n", j(n, 2));
    }
    return 0;
}
