vector< vector<ll> > mul(vector< vector<ll> >& A, vector< vector<ll> >& B) {
	vector< vector<ll> > C(2, vector<ll>(2));
	for (int i = 0; i < 2; i++)
		for (int j = 0; j < 2; j++)
			for (int k = 0; k < 2; k++)
				C[i][j] = (C[i][j] + A[i][k] * B[k][j]) % 1000000007;
	return C;
}

int main() {
	int T; cin >> T;
	while (T--) {
		int N; cin >> N;
		vector< vector<ll> > X(2, vector<ll>(2)), Y(2, vector<ll>(2));
		X[0][0] = X[1][0] = 1; X[0][1] = 2;
		Y[0][0] = Y[1][1] = 1;
		for (N--; N > 0; N /= 2) {
			if (N & 1) Y = mul(Y, X);
			X = mul(X, X);
		}
		cout << (Y[1][0] * 7 + Y[1][1] * 2) % 1000000007 << endl;
	}
}
