#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    string word, rword;
    cin >> t;
    while (t--) {
    	cin >> rword;
    	word = rword;
    	reverse(rword.begin(), rword.end());
    	if (word == rword)
    		cout << "YES" << endl;
    	else
    		cout << "NO" << endl;
    }
    return 0;
}
