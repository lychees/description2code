#include <iostream>
using namespace std;
template< class T > T _max(T a, T b) { return (!(a < b) ? a : b); }

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    int n,a[100010],i,dp[100011];
    cin>>n;
    for(i=0;i<n;i++)
    cin>>a[i];
    
    dp[0]=a[0];
    dp[1]=a[1];
    for(i=2;i<n;i++)
    dp[i]=_max(dp[i-1],dp[i-2]+a[i]);
    cout<<dp[n-1]<<"\n";
    }
    
    return 0;
}
