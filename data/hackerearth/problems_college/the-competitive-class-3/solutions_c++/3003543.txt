#include <bits/stdc++.h>

using namespace std;

bool compare(int x, int y)
{
	return (x>y);
}

int main()
{
	int n, i;
	cin>>n;
	int a[n], b[n];
	
	for (i = 0; i < n; i++) {
		cin>>a[i];
		b[i] = a[i];
	}
	
	sort(a, a+n, compare);
	map<int, int> m;
	m[a[0]] = 1;
	
	for (i = 1; i < n; i++) {
		if (a[i-1] != a[i]) m[a[i]] = i+1;
	}
	
	for (i = 0; i < n; i++) cout<<m[b[i]]<<" ";
	cout<<endl;
	
	return 0;
}
