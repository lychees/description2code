import time


vals = {}
completed = set()            
for i in xrange(2, 10**6+1):
    tmp = set(j for j in xrange(i, 10**6+1, i) if j not in completed)
    vals[i] = len(tmp)
    completed.update(tmp)



T = input()
for i in xrange(T):
	N = input()
	if N in vals:
		print vals[N]
	else:
		print 0
