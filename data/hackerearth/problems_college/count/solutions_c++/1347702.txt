#include <iostream>
using namespace std;

int main()
{
    int t=0;
    cin>>t;
    while(t--){
    	int r,c;
    	cin>>r>>c;
    	int even=0,odd=0;
    	for(int i=1;i<=r;i++){
    		if(i%2==0){
    			if(i<c){
    				even+=i;
    			}
    			else{
    				even +=c;
    			}
    		}
    		else{
    			if(i<c){
    				odd+=i;
    			}
    			else{
    				odd +=c;
    			}    			
    		}
    	}
    	cout<<even<<" "<<odd<<endl;
    }
    return 0;
}
