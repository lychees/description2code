#include <iostream>
#include <algorithm>
using namespace std;
int dif(int a,int b)
{
	if(a>b)
	{
		return a-b;
	}
	else
	{
		return b-a;
	}
}
int gcd(int a,int b)
{
	if(b==0)
	{
		return a;
	}
	else
	{
		return gcd(b,a%b);
	}
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n;
    	scanf("%d",&n);
    	int i,a[n],b[n],c[n],min;
    	for(i=0;i<n;i++)
    	{
    		scanf("%d",&a[i]);
			b[i] = 0;    	
    	}
    	sort(a,a+n);
    	int p,q,k,j,x,y,z;
    	scanf("%d %d",&p,&q);
    	j = 0;
    	min = 100000;
    	for(i=0;i<n;i++)
    	{
    		if(gcd(p,a[i])!=1)
    		{
    			b[j] =  a[i];
    			j++;
    		}	
    	}
    	if(j<q)
    	{
    		printf("-1");
    	}
    	else
    	{
    		z = 0;
    		for(y=0;y<q;y++)
    		{
    			min = 100000;
    			for(i=0;i<j;i++)
    			{
    				if(b[i]!=-1)
    				{
    					k = dif(b[i],p); 
    					if(k<min)
    					{
    						min = k;
    						x = i;
    					}
    				}
    			}
    			c[y] = b[x];
    			b[x] = -1;
    		}
    		sort(c,c+y);
    		for(i=0;i<y;i++)
    		{
    			printf("%d ",c[i]);
    		}
    	}
    	printf("\n");
    }
    return 0;
}
