#include <cstdio>
#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

const int N = 100001;
int a[N], n;

int main(){
    cin >> n;
    for(int i = 2; i < N; ++i) if (a[i] == 0){
        a[i] = i;
        for(int j = i + i; j < N; j += i) a[j] = i;
    }
    int sum = 0;
    for(int i = 0; i < n; ++i){
        int x, here = 0;
        cin >> x;
        while (x > 1){
            x /= a[x];
            ++here;
        }
        sum ^= here;
    }
    cout << (sum ? "ALICE\n" : "BOB\n");
    return 0;
}
