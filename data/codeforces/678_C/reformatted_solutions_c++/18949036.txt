#include <algorithm>
#include <cmath>
#include <cstdio>
#include <iostream>
using namespace std;
int main() {
  long long n, a, b, p, q, sum1, sum2;
  cin >> n >> a >> b >> p >> q;
  sum1 = (n / a) * p + (n / b) * q;
  sum2 = n / ((a * b) / __gcd(a, b)) * min(p, q);
  cout << sum1 - sum2 << endl;
}
