#include <iostream>
using namespace std;

int comb(int n,int k){
	int L[n+1][k+1];
	L[0][0]=1;
	for(int i=1;i<=n;i++)
	L[i][0]=1;
	for(int i=0;i<=n;i++)
		for(int j=1;j<=k;j++){
		if(i==j)
		L[i][j]=1;
		else if(i<j)
		L[i][j]=0;
		else
		L[i][j]=L[i-1][j-1]+L[i-1][j];}
	return L[n][k];
}

int main()
{
    int n;
    cin>>n;
    int count=0;
    for(int i=3;i<=n;i++){
    	count+=(comb(i-1,2)*2);
    }
    cout<<count<<endl;
    return 0;
}
