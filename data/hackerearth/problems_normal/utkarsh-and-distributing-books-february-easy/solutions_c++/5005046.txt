

    #include <iostream>
    #include <cassert>
    using namespace std;
    int n;
    int a[1010];
    void solve() {
    	cin>>n;
    	for (int i = 1; i <= n; i++) {
    		cin>>a[i];
    		assert(2 <= a[i] && a[i] <= 1000);
    	}
    	int minA = 1E9, sum;
    	for (int i = 1; i <= n; i++) {
    		minA = min(minA, a[i] - 1);
    	    sum += a[i];
    	}
    	cout<<minA<<" "<<sum - n <<endl;
    }
    int main()
    {
        int test;
        cin>>test;
        while (test--) {
        	solve();
        }
        return 0;
    }
