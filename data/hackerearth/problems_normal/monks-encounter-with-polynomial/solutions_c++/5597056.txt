#include <iostream>
#include <math.h>

using namespace std;

int main()
{
    long long int i,j,k,l,n,a,b,c,t,h,m,x;
    cin>>t;
    while(t--)
    {
        cin>>a>>b>>c>>k;
        n=sqrt(k)+1;
        l=0;h=n;
        while(l<=h)
        {
            m=(l+h)/2;
            x=a*m*m+b*m+c;
            if(x==k)
            {
                j=m;
                break;
            }
            else
            {
                if(x>k)
                {
                    j=m;
                    h=m-1;
                }
                else
                    l=m+1;
            }
        }
        cout<<j<<"\n";
    }

    return 0;
}
