#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
  {
  	long int n;
  	cin>>n;
  	long int *a=new long int[n];
  	for(int i=0;i<n;i++)
  	  cin>>a[i];
  	  
  	long int i,j;
  	cin>>i>>j;
  	
  	sort(a,a+n);
  	long int sum=a[n-i]+a[j-1];
  	cout<<sum<<endl;
	
  }  
    return 0;
}
