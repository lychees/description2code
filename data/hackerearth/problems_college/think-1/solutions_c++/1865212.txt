#include <bits/stdc++.h>
using namespace std;

int check(int num)
{
    int rev = 0, temp = num;
    while(temp) {
        rev = rev * 10 + (temp % 10);
        temp /= 10;
    }
    return rev;
}

int main()
{
	int t;
	cin >> t;
	while(t--) {
		int num;
		cin >> num;
		int rev = check(num);
		int snum = num * num;
		int pnum = rev * rev;
		if(snum == pnum) {
			puts("No");
		}
		else if(snum != check(pnum)) {
			printf("No\n");
		}
		else {
			 printf("%d\n", rev);
		}
	}
	return 0;
}
