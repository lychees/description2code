#include <bits/stdc++.h>
using namespace std;

int main(){
    int n;
    string s;
    cin >> n;
    map<string, int> st;
    while(n--){
    	cin >> s;
    	st[s]++;
    }
    vector<pair<string, int> > v(st.begin(), st.end());
    for(int i=0; i<v.size()-1; i++){
    	for(int j=i+1; j<v.size(); j++){
    		if(v[i].first.size() < v[j].first.size())
    			swap(v[i], v[j]);
    	}
    }
    for(int i=0; i<v.size(); i++)
    	cout << v[i].first << " " << v[i].second << endl;
    return 0;
}
