#include<stdio.h>
#include<iostream>
using namespace std;
int main(void)
{
	int i,j,t,n;
	float xs,ys,temp;
	cin>>t;
	
	for(i=0;i<t;i++)
	{
		cin>>n;
		xs=0;ys=0;
		for(j=0;j<n;j++)
		{
			cin>>temp;
			xs=xs+temp;
			cin>>temp;
			ys=ys+temp;
		}
		xs=xs/n;
		ys=ys/n;
		cout.precision(2);
		cout<<fixed<<"("<<xs<<" , "<<ys<<")"<<endl;
	}
	return 0;
}
