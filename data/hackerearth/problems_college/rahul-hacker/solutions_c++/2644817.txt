#include <iostream>
#include<string.h>
using namespace std;
int reverse(int n)
{int rev=0;
	while(n!=0)
	{
		rev=rev*10+(n%10);
		n/=10;
	}
	return rev;
	
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int s1,s2;
    	cin>>s1>>s2;
    	int n1=reverse(s1);
    	int n2=reverse(s2);
    	int n3=n1+n2;
    
    	cout<<reverse(n3)<<endl;
    	
    }
    return 0;
}
