#include <iostream>
#define ll long long
using namespace std;
ll power(ll a, ll b)
{
	ll temp;
	if(b==0) return 1;
	temp=power(a,b/2);
	if(b%2==0)
		return temp*temp;
	else
	{
		if(b>0)
		return a*temp*temp;
		else
		return (temp*temp)/a;
	}
}
int main()
{
	ll t, a, b, c, num, rem, x;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b>>c;
    	num=power(a,b);
    	rem=num%c;
    	x=c-rem;
    	if(rem<=x)
    		cout<<(num-rem)<<endl;
    	else
    		cout<<(num+x)<<endl;
    }
    return 0;
}
