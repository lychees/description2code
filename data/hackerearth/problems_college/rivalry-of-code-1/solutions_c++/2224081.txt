#include <iostream>
#include<string>
using namespace std;

int main()
{
    int t;
    string s1,s2;
    cin>>t;
    while(t--){
    	cin>>s1>>s2;
    	//cout<<s1<<" "<<s2;
    	if(s1.length() == s2.length()){
    		int different = 0;
    		for(int i = 0;i<s1.length();i++){
    			if(s1[i]!=s2[i]){
    				different++;
    			}
    		}
    		cout<<different<<endl;
    	}
    	else{
    		int min =9999999;
    		if(s1.length()<s2.length()){
    			for(int i = 0;i+s1.length()<=s2.length();i++){
    				int different = 0;
    				for(int j = i;j<i+s1.length();j++){
    					if(s1[j-i]!=s2[j]){
    						different++;
    					}
    				}
    				if(min>different){
    					min = different;
    				}
    			}
    			cout<<min<<endl;
    		}
    		else{
    			string xx = s1;
    			s1 = s2;
    			s2 = xx;
    			for(int i = 0;i+s1.length()<=s2.length();i++){
    				int different = 0;
    				for(int j = i;j<i+s1.length();j++){
    					if(s1[j-i]!=s2[j]){
    						different++;
    					}
    				}
    				if(min>different){
    					min = different;
    				}
    			}
    			cout<<min<<endl;
    		}
    	}
    }
    return 0;
}
