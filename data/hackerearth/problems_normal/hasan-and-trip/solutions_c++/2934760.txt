typedef long long LL;

#define REP(i,x,n) for(int (i)=(int)x;(i)<(int)(n);++(i))
#define endl "\n"
#define ff first
#define ss second
#define pb push_back
#define mp make_pair

LD d[3005][3005];
LL x[3005],y[3005],f[3005];
LD dp[3005];
int n;
 
int main(int argc, char const *argv[])
{
	ios::sync_with_stdio(false); cin.tie(NULL);
	cin>>n;
	REP(i,0,n)
		cin>>x[i]>>y[i]>>f[i];
	REP(i,0,n){
		REP(j,i+1,n){
			d[i][j] = sqrt((LD)((x[i]-x[j])*(x[i]-x[j])+(y[i]-y[j])*(y[i]-y[j])));
		}
	}
	fill(dp,dp+n,-1ll<<60);
	dp[n-1]=f[n-1];
	for(int i=n-2;i>=0;--i){
		for(int j=i+1;j<n;++j){
			dp[i]=max(dp[i],f[i]+dp[j]-d[i][j]);
		}
	}
	cout.precision(6);
	cout<<fixed<<dp[0];
	return 0;
}
