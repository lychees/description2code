#include <iostream>
#include <math.h>
using namespace std;

int main()
{
	
	int t;
	cin>>t;
	while(t--)
	{
unsigned	long long 	int n;
		cin>>n;
		
	unsigned	long long int sum=0;

unsigned	long long 		int h=n;

unsigned	long long 		int numdig=0;
		while(h>0)
		{
			h=h/10;
			numdig++;
		}
	unsigned	long long int last_num=0;
	
unsigned	long long 	int x=numdig-1;
		while(x>0)
		{
			last_num=last_num*10+9;
	
	x--;
		}
		x=numdig;
	
unsigned	long long 	int u=1;
			while(u<x)
		{
			sum=sum+u*9*pow(10,u-1);
	
	u++;
		}
		sum=sum+(n-last_num)*numdig;
	cout<<sum<<endl;
		
	}
   // cout << "Hello World!" << endl;
    return 0;
}
