#include <iostream>
#define N 100005
using namespace std;

int main()
{
    int t,n,m,i,x,y;
    long long a[N],l[N],r[N];
    cin>>n>>m;
    for(i=1;i<=n;i++)
     cin>>a[i];
     l[1]=1;
    	r[n]=n;
    	for(i=2;i<=n;i++)
    	 if(a[i]<=a[i-1])
    	  l[i]=l[i-1];
    	 else
    	  l[i]=i;
    	
    	for(i=n-1;i>0;i--)
    	 if(a[i]<=a[i+1])
    	  r[i]=r[i+1];
    	 else
    	  r[i]=i;
    while(m--)
    {
    	cin>>x>>y;
       if(r[x]>=l[y])
        cout<<"Yes\n";
        else
         cout<<"No\n";
    }
    return 0;
}
