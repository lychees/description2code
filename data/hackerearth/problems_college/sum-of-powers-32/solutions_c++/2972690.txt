#include <stdio.h>
#include<stdlib.h>
#include<string.h>
int arr[1000001];

int main()
{
	memset(arr,0,1000001);
	arr[1] = 1;
	for(int i=2; i*i<=1000001;i++)
		for(int j=i*i;j<1000001;j*=i)
			arr[j] = 1;
	int T, n;
	scanf("%d", &T);
	while(T-->0) {
		scanf("%d", &n);
		int count=0;
		for(int i=1;i<n;i++)
			if(arr[i]==1 && arr[n-i]==1) {
				count++;
				break;
			}
		if(count!=0)
		printf("Yes\n");
		else
		printf("No\n");
	}
 
	return 0;
}
