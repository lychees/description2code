#include <iostream>
#include <string>
#include <set>
using namespace std;

int main()
{
	string a,b;
	int count=0;
	set < string > xx;
	cin >> a >> b;
//	cout << a  << " " << b <<endl;
	set<string>::iterator it;
    for(int i=0;i<a.length();i++)
    {
    	
    	
    	for(int j=i;j<a.length();j++)
    	{
    		string x=a.substr(i,j-i+1);
    	//	 cout << x << endl;
    	    xx.insert(x);
    	}
    }
    for(it=xx.begin();it!=xx.end();it++)
    {
    	string p=*it;
    	if(p.find(b)!=-1)
    	 count++;
    }
    cout << count <<endl;
    return 0;
}
