#include <iostream>
using namespace std;

int main()
{
    int n,i,idx,t;
    cin >> t;
    while(t--)
    {
    cin >> n;
    double d[n],f[n],sum=0,max = 0;
    for(i=0;i<n;i++)
    {
    	cin >> d[i];
    }
    for(i=0;i<n;i++)
    {
    	cin >> f[i];
    }
    for(i=0;i<n;i++)
    {
    	if(d[i]*f[i] > max)
    	{
    		idx = i;
    		max = d[i]*f[i];
    	}
    	sum = sum + d[i]*f[i];
    //	cout << sum;
    }
    printf("%d %.3f\n",idx,max/sum);
    }
    return 0;
}
