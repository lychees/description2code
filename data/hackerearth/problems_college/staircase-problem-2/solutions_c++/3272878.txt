#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

int main()
{
 ll t;
 cin>>t;
 while(t--)
 {
 	ll s;
 	cin>>s;
 	if(s==1)
 	cout<<s<<endl;
 	else
 	{
 		float b= (-3+ sqrt(9+8*s))/2.0;
 		//cout<<b<<endl;
 		if(b==floor(b))
 		cout<<b<<endl;
 		else
 		cout<<(ll)b+1<<endl;
 	}
 }
	return 0;
}
