#include <iostream>
#include<bits/stdc++.h>
#define ll long long
#define pb push_back
using namespace std;
int main()
{
    //cout << "Hello World!" << endl;
    ll t,i,j,sum,x,y,n,k;
    cin>>t;
    vector<ll> num;
    while(t--)
    {
    	cin>>x>>y>>n;
    	sum=0;
    	for(i=0;i<x;i++)
    	{num.pb(y);sum+=y;}
    	k=0;
    	for(i=0;i<n-x;i++)
    	{
    		num.pb(sum);
    		sum+=sum;
    		sum-=num[k];k++;
    	}
    	/*for(i=0;i<num.size();i++)
    		cout<<num[i]<<" ";*/
    	cout<<num[num.size()-1];
    	cout<<endl;
    	num.clear();
    }
    return 0;
}
