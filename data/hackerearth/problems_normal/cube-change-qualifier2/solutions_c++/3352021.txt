#include <iostream>
using namespace std;
void fun(long int n)
{
	if(n==1)
	cout<<1;
	else
	{
		long int vol = 2*n*n + 2*(n-2)*n + 2*(n-2)*(n-2);
		cout<<vol;
	}
}
int main()
{
	int t;
	cin>>t;
	long int *a= (long int *)malloc(sizeof(int));
	int i=0;
	for(i;i<t;i++)
	{
		cin>>a[i];
	}
	for(i=0;i<t;i++)
	{
		fun(a[i]);
		cout<<'\n';
	}
	
    return 0;
}
