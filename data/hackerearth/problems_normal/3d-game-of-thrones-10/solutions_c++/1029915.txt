#include <bits/stdc++.h>

using namespace std;

int main(int argc, char* argv[]) {
	ios_base::sync_with_stdio(false);
	int t, n, x, res;
	cin >> t;
	while (t--) {
		res = 0;
		cin >> n;
		n *= 3;
		while (n--) {
			cin >> x;
			res ^= x;
		}
		if (res == 0) cout << "Lannister\n";
		else cout << "Stark\n";
	}
	return 0;
}
