#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	int viv[100][1000],i,j;
	string str;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>str;
		for(j=0;j<str.length();j++)
		{
			viv[i][(int)str[j]]++;
		}
	}
	int cnt=0;
	int ans=0;
	for(i=97;i<=122;i++)
	{
	    cnt=0;
		for(j=0;j<n;j++)
		{
			if(viv[j][i]>=1)
			{
				cnt++;
			}
		}
		if(cnt==n)
			ans++;
	}
	cout<<ans<<endl;
}
