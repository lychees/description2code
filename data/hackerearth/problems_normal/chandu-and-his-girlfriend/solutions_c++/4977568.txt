#include <iostream>
using namespace std;
void merge(unsigned int a[],unsigned int l,unsigned int m,unsigned int r)
{
	unsigned int p=l,q=m+1;
	unsigned int arr[r-l+1],i,k=0;
	for(i=l;i<=r;i++)
	{
		if(p>m)
		 arr[k++]=a[q++];
		else if(q>r)
		 arr[k++]=a[p++];
		else if(a[p]>a[q])
		 arr[k++]=a[p++];
		else
		 arr[k++]=a[q++];
	}
	for(i=0;i<k;i++)
	 a[l++]=arr[i];
}
void mergesort(unsigned int a[],unsigned int l,unsigned int r)
{
	if(l<r)
	{
		unsigned int m=(l+r)/2;
		mergesort(a,l,m);
		mergesort(a,m+1,r);
		merge(a,l,m,r);
	}
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	unsigned int n;
    	cin>>n;
    	unsigned int a[n],i;
    	for(i=0;i<n;i++)
    	 cin>>a[i];
    	mergesort(a,0,n-1);
    	for(i=0;i<n;i++)
    	 cout<<a[i]<<" ";
    	cout<<"\n";
    }
    return 0;
}
