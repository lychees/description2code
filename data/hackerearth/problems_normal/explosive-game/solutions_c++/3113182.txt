#include <iostream>
#include <algorithm>
#include <cstdio>
#include <cstdlib>
#include <vector>

using namespace std;

int main()
{
    vector<long long> sum;
    long long total = 0;
    long long n, q,x;
    scanf("%lld %lld",&n,&q);
    
    for(int i = 0; i < n; i++){
        long long a,b;
        scanf("%lld %lld",&a,&b);
        total += min(a,b);
        sum.push_back(abs(a-b));
    }
    
    sort(sum.begin(),sum.end());
    for(int i = 0;i < n; i+=2){
        if(sum.at(i) != sum.at(i+1)){
            total = -1;
            break;
        }
        total += sum.at(i);
    }
    
    while(q--){
        scanf("%lld",&x);
        if(total == x)
            printf("YES\n");
        else
            printf("NO\n");
    }
    
    return 0;
}
