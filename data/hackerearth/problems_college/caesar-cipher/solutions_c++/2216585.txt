#include "iostream"
using namespace std;
string s;
int t,k,x;
int main()
{
	ios_base::sync_with_stdio(0);
	cin>>t;
	
	while (t--)
	{
		cin>>k>>s;
		k%=26;
		for (int i=0; i<s.size(); i++)
		{
			if (s[i]>96)
			{
				x=s[i];
				x+=k;
				if (x>122) x-=26;
				s[i]=x;
			}
			else if (s[i]>64)
			{
				x=s[i];
				x+=k;
				if (x>90) x-=26;
				s[i]=x;
			}
			else
			{
				s[i]+=k;
				if (s[i]>57) s[i]-=10;
			}
		}
		for (int i=0; i<s.size(); i++) cout<<s[i];
		cout<<"\n";
	}
	
	return 0;
}
