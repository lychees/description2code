ulli count_fibnum(ulli x)
{
	if(x==1)return 4;
	if(x==2)return 6;
	ulli a = 1;
	ulli b = 2;
	ulli sum = a + b;
	ulli lim = 2*x;
	ulli count=2;
	ulli lsum=sum;
	//while(sum<lim)
	while(sum-count<=x)
	{
		++count;
		a = b;
		b = sum;
		lsum=sum;
		sum = a + b;
	}
	if(sum==lim)
	{
		lsum = sum;
		++count;
	}
	//cout<<"lsum : "<<lsum<<endl;
	//cout<<"count: "<<count<<endl;
	ulli lastthnum = lsum-1;
	ulli nxtnum = lsum +1;
	ulli lastcnt = lsum-count;
	if(x<=lastcnt)
			return lastthnum - (lastcnt-x);
	else
			return lsum + x-lastcnt;
	
}
