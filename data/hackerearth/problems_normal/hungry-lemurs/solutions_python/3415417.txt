import math
import sys

N, K = map(int, raw_input().split())

if N == K or N % K == 0:
	print(0)
	sys.exit(0)

	
min_time = 2*K
for k in range(1, 2*K+1):
	n = N - (N % k)
	min_time = min([min_time, abs(N - n) + abs(K - k)])
	n += k
	min_time = min([min_time, abs(N - n) + abs(K - k)])

print(min_time)
