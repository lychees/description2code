#include<iostream>
#include<cstdio>
using namespace std;

typedef unsigned long long ull;
typedef long long lol;
#define loops(i,p,n) for(int i=p; i<n; i++)
#define loopr(i,p,n) for(int i=p; i>=n; i--)
const int maxn = 1010;
int grid[maxn][maxn];
int sameAs[maxn], cnt[maxn];
bool taken[maxn];

int main()
{
   ios::sync_with_stdio(false);
   int n, m;
   cin >> m >> n;
   loops(i, 0, m) loops(j, 0, n) cin >> grid[i][j];
   loops(i, 0, m) taken[i] = false;
   loops(i, 0, m) 
   {
   	     if(taken[i]) continue;
   	     cnt[i] = 0;
   	     loops(j, 0, n) if (grid[i][j] == 0) cnt[i]++;
   	     loops(j, 0, m)
         {
   	           bool flag = true;
   	           loops(k, 0, n) if(grid[i][k] != grid[j][k]) flag = false;
   	           taken[j] = flag;
   	           sameAs[i] += flag;
   	           if (flag) cnt[j] = cnt[i];
         }
   }
   int maxi = 0;
   loops(i, 1, m) {
   	     if(sameAs[i] > sameAs[maxi])
   	        maxi = i;
   	     else if (sameAs[i] == sameAs[maxi] && cnt[i] < cnt[maxi])
   	              maxi = i;
   }
   cout << cnt[maxi] << " " << sameAs[maxi];
   return 0;
}
