#include <iostream>
using namespace std;

int main()
{
 	unsigned long long a,b,sum;
 	int n;
 	cin>>a;
 	cin>>b;
 	cin>>n;
 	sum=a+b;
 	if(n==1){
 		cout<<a<<endl;
 		return 0;
 	}
 	else if(n==2){
 		cout<<b<<endl;
 		return 0;
 	}
 	else{
 		for(int i=2;i<n;i++){
 		sum=a+b;
 		a=b;
 		b=sum;
 		}
 		cout<<sum;
 	}
 	
 	
    return 0;
}
