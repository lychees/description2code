#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int t,c=0,n,max=-1,sum=0;
	map <string,int> mp;
	map <string,int> :: iterator it;
	cin >> n;
	while(n--)
	{
		string q;
		cin >> q;
		if(mp.find(q) == mp.end())
			mp[q] = 1;
		else
		{
			mp[q]++;
		}
	}
	for(it=mp.begin();it!=mp.end();it++)
	{
		if(max < it->second)
		{
			max = it->second;
		}
		//cout << it->first << endl;
	}
	for(it=mp.begin();it!=mp.end();it++)
	{
		if(max == it->second)
		{
			sum = sum + max;
		}
		//cout << it->first << endl;
	}
	cout << sum << endl;
}
