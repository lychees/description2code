#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<cstring>
using namespace std;
int cs, n, m, s, i;
char x[100001], y[100001];
int C[257];
int main()
{
  scanf("%d", &cs);
  while (cs--)
  {
    scanf("%s%s", &x, &y);
    n = strlen(x);
    m = strlen(y);
    s = 0;
    memset(C, 0, sizeof C);
    for (i = 0; i < m; i++)
    {
      C[y[i]]--;
      C[x[i]]++;
    }
    for (i = 'a'; i <= 'z'; i++)
      s += C[i] != 0;
    if(s)
      for (i = m; i < n; i++)
      {
        if (!C[x[i]])
          s++;
        C[x[i]]++;
        if (!C[x[i]])
          s--;
        if (!C[x[i - m]])
          s++;
        C[x[i-m]]--;
        if (!C[x[i - m]])
          s--;
        if (!s)
          break;
      }
    printf("%s\n", s ? "NO" : "YES");
  }
  return 0;
}
