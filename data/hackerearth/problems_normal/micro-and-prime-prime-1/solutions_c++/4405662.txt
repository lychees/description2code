#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int a[1000005],s[1000005],pp[1000005];
void sieve()
{
	a[0]=1;
	a[1]=1;
	int i,j;
	for(i=2;i*i<=1000000;i++)
	{
		if(!a[i])
		{
			for(j=i*i;j<=1000000;j+=i)
			{
				a[j]=1;
			}
		}
	}
	s[1]=0;
	for(i=2;i<=1000000;i++)
	{
		s[i]=s[i-1]+!a[i];
	}
	pp[1]=0;
	for(i=2;i<=1000000;i++)
	{
		pp[i]=pp[i-1];
		if(!a[s[i]])
		{
			pp[i]++;
		}
	}
}
int main()
{
	sieve();
    int t,l,r;
    int ans;
    cin>>t;
    while(t--)
    {
    	cin>>l>>r;
    	if(l>1)
    	{
    		ans=pp[r]-pp[l-1];
    	}
    	else
    	{
    		ans=pp[r];
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
