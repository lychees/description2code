#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll A[100005];
int main()
{
    int t,n,i;
    ll maxi,a,b,c;
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(i=0;i<n;i++)
            cin>>A[i];
        if(n<3)
            cout<<"-1"<<endl;
        else
        {
            sort(A,A+n);
            maxi=0;
            for(i=2;i<n;i++)
                if(A[i-2]+A[i-1]>A[i])
            {
                if(A[i]+A[i-1]+A[i-2]>=maxi)
                {
                    maxi=A[i]+A[i-1]+A[i-2];
                    a=A[i-2],b=A[i-1],c=A[i];
                }
            }
            if(maxi==0)
                cout<<"-1"<<endl;
            else
                cout<<a<<" "<<b<<" "<<c<<endl;
        }
    }
    return 0;
}
