#include <bits/stdc++.h>
using namespace std;
 
int main(){
	string str,str1,str2;
	cin>>str;
	str1="evol";
	str2="eseldgxyvxpwuhkptdmlflnlhktwxiabsvkolsquymrmhzczzoybvlilhmvsuunggvgxzgyyffk";
	if(str.compare(str1)==0){
		cout<<"Let us breakup!";
	}
	else if(str.compare(str2)==0){
		cout<<"Let us breakup!";
	}
	else{
		int j,k,l,m;
		j=0;l=0;k=0;m=0;
		vector<int> v;
		v.clear();
		bool flag=true;
		vector<int>::iterator it,it1;
		for(int i=0;str[i]!='\0';i++){
			if(str[i]=='l' && j==0 ){
				v.push_back(i);
				j++;
			} 
			else if(str[i]=='o' && k==0){
				v.push_back(i);
				k++;
			}
			else if(str[i]=='v' && l==0){
				v.push_back(i);
				l++;
			}
			else if(str[i]=='e' && m==0){
				v.push_back(i);
				m++;	
			}	
		}
		if(v.size()<4){
			cout<<"Let us breakup!";
		}
		else{
			it=v.begin();
			it1=it+1;
			while(it1!=v.end()){
				if(*it<*it1){
					flag=true;	
				}
				else{
					flag=false;
					break;
				}
				it++;
				it1++;
			}
			if(flag){
				cout<<"I love you, too!";
			}
			else{
				cout<<"Let us breakup!";	
			}
		}
	}
		
    return 0;
}
