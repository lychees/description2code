#include <bits/stdc++.h>
using namespace std;
vector<int> a[100001];
int dfsin[100001],dfsout[100001],height[100001];
int countv=0;
void dfs(int value,int h){
	dfsin[value]=countv++;
	height[value]=h;
	for(vector<int>::iterator it=a[value].begin(); it!=a[value].end();it++){
		dfs(*it,h+1);
	//	cout<<*it<<endl;
	}
	dfsout[value]=countv++;
}
int main()
{
    
    int t,n,q,temp,root;
    cin>>t;
    while(t--){
    	cin>>n;
    	for(int i=0;i<=n;i++){
    		dfsin[i]=dfsout[i]=0,height[i]=0;
    		a[i].clear();
    	}
    	for(int i=1;i<=n;i++){
    		scanf("%d",&temp);
    		if(temp!=0)
    			a[temp].push_back(i);
    		else
    			root=i;
    	}
    	//cout<<root<<endl;
    	dfs(root,0);
    	//for(int i=1;i<=n;i++)
    	//	cout<<i<<" "<<dfsin[i]<<" "<<dfsout[i]<<" "<<height[i]<<endl;
    	cin>>q;
    	int a,b;
    	while(q--){
    		scanf("%d%d",&a,&b);
    		if(dfsin[a]>dfsin[b]&&dfsout[b]>dfsout[a])
    			printf("%d\n",height[a]-height[b]-1);
    		else
    			printf("-1\n");
    	}
    }
    return 0;
}
