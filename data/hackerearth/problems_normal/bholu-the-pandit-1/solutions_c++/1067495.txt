#include <bits/stdc++.h>
using namespace std;

#define M 1000000007

long long gcd(long long a, long long b)
{
	if(b==0)
		return a;
		
	return gcd(b, a%b);
}

int main()
{
    int T;
    scanf("%d", &T);
    
    while(T--)
    {
    	long long a, b, c, x;
    	
    	scanf("%lld %lld %lld", &a, &b, &c);
    	
    	x = gcd(a, gcd(b,c));
    	
    	printf("%lld %lld\n", x, ((a/x*b/x)%M*c/x)%M);
    }
    
    return 0;
}
