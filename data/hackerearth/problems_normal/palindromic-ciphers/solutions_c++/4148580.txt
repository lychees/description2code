#include <iostream>
#include<string>
#include<vector>
#include<algorithm>

int main()
{

  int t;

  while(t--)
  {
  	vector<char> v1,v2;
  	string s;
  	cin>>s;
  	for(int i=0;i<s.length();i++)
  	{
  		v1.push_back(s[i]);
  	}
   v2=v1;
   	unsigned long long ans=1;
   reverse(v2.begin(),v2.end());

   cout<<"Palindrome"<<endl;
   else
   {
   
   	for(int i=0;i<s.length();i++)
   	ans*=(s[i]-96);
   	cout<<ans<<endl;
   }

  }
