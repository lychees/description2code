#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n, no, maxi=0, sum=0;
    cin>>n;
    for(int i=0; i<n; i++)
    {
    	scanf("%d", &no);
    	if((sum+no)>0)
    	{
    		sum+=no;
    		if(sum>maxi)
    			maxi=sum;
    	}
    	else
    		sum=0;
    }
    cout<<maxi<<endl;
    return 0;
}
