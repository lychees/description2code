#include<cstdio>
#include<bits/stdc++.h>
int main()
{
    int i,n,j,factor;
    //printf("Enter the last number of the sequence:");
    scanf("%d",&j);
    for(n=2;n<=j;n++)
    {
         for(i=1;i<n;i++)
         {
              if(n%i==0) 
              {
                   factor=i;/*finds the largest proper divisor*/
              }
         }
         if(factor>1) 
         {
              printf ("%d ",n);
         }              
     }
     return 0;
}
