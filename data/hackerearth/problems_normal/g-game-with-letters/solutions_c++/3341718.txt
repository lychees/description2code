#include<bits/stdc++.h>
using namespace std;

#define ll long long 
#define mod 1000000007

vector < pair< ll,pair< ll,ll > > > v;

bool compare(const pair< ll,pair< ll,ll > > &a,const pair< ll,pair< ll,ll > > &b)
{
	if(a.first!=b.first)
	{
		return a.first>b.first;
	}
	if(a.second.first!=b.second.first)
	{
		return a.second.first<b.second.first;
	}
	else
		return a.second.second<b.second.second;
}


ll t,n,l,h[30],idx,mx,i,j,score[10005];
string s1[10005],s;
int main()
{
	ios_base::sync_with_stdio(false);cin.tie(0);
	cin>>t;
	while(t--)
	{
		memset(score,0,sizeof(score));
		memset(h,0,sizeof(h));
		cin>>n;
		cin>>s;
		l=s.length();
		for(i=0;i<l;i++)
			h[s[i]-97]++;
		for(j=1;j<=n;j++)
		{
			cin>>s1[j];
			for(i=0;i<s1[j].length();i++)
			{
				if(!h[s1[j][i]])
				score[j]+=h[s1[j][i]-97];
			}
		}
		//mx=-1;
		for(i=1;i<=n;i++)
		{
			v.push_back({ score[i], {s1[i].length(),i} });
		}
		sort(v.begin(),v.end(),compare);
		cout<<v[0].second.second<<endl;
		v.clear();
	}
	return 0;
}
