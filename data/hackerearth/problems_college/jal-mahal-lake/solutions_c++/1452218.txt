#define lli long long int
#define ulli unsigned long long int
#define F first
#define S second
#define pii pair<int,int>
#define pip pair<int,pii>
#define pis pair<int,string>
#define pll pair<lli,lli>

using namespace std;

vector<pii>v;
int dist[51],pos[51];
int main()
{
    int t;
    int n,i,x;
    lli INF=1e9;
    INF*=INF;
    scanf("%d",&t);
    while(t--)
    {
        v.clear();
        scanf("%d",&n);
        for(i=0;i<n;i++)
        {
            scanf("%d",&pos[i]);
        }
        for(i=0;i<n;i++)
        {
            scanf("%d",&x);
            v.push_back(pii(pos[i],pos[i]+x));
        }
        sort(v.begin(),v.end());
        for(i=0;i<n-1;i++) dist[i]=v[i+1].F-v[i].S;
        lli ans=INF;
        for(i=0;i<n-1;i++)
        {
            lli val=0,sm=0;
            int j=i;
            while(j>=0)
            {
                sm+=dist[j];
                val+=sm;
                j--;
            }
            j=i+1;
            sm=0;
            while(j<n-1)
            {
                sm+=dist[j];
                val+=sm;
                j++;
            }
            ans=min(ans,val);
            //printf("outer loop\n");
        }
        printf("%lld\n",ans);
    }
    return 0;
}
