#include <iostream>
#include <string>
using namespace std;

int main()
{
	string numOfLines;
	string givLine = "";
	string currency;
	int amp;
    getline(cin,numOfLines);
    for(int a = 0;a < stoi(numOfLines); a++){
    	getline(cin,givLine);
    	amp = givLine.find('$') + 1;
    	while((isdigit(givLine[amp]) || givLine[amp] == ' ') && amp < givLine.length()){
    	if(isdigit(givLine[amp])){
    		if(currency == "" && givLine[amp] == '0'){
    			
    		}
    		else{
    			currency += givLine[amp];
    		}
    	}
    	amp++;
    	}
    	if(currency == ""){
    		cout << "$0" << endl;
    	}
    	else{
    		cout << '$' << currency << endl;
    	}
    	currency = "";
    }
    return 0;
}
