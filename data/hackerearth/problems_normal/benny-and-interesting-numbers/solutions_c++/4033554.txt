#include<bits/stdc++.h>
using namespace std;
typedef long long int uli;
const int mx=4e5;
bool criba[mx];
int primes[mx],np=0;
int main(){
// freopen("data.in","r",stdin);
// freopen("data.out","w",stdout);
   memset(criba,true,sizeof criba);
   criba[0]=criba[1]=false;
   for(int i=0;i<mx;i++){
      if(criba[i]){
         primes[np++]=i;
         for(int j=i+i;j<mx;j+=i){
            criba[j]=false;
         }
      }
   }
   int t;
   scanf("%d",&t);
   while(t--){
      uli x;
      scanf("%lld",&x);
      if(x==1){
         puts("Yes");
         continue;
      }
      bool ok=true;
      for(int i=0;i<np;i++){
         uli p=primes[i];
         int e=0;
         while(x%p==0)x/=p,e++;
         if(e>1){
            ok=false;
            break;
         }
      }      
      if(ok)puts("Yes");
      else puts("No");
   }
   return 0;
}
