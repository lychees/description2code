n = input()


def Palindrome_Number(n):
    m = n
    a = 0
    while (m != 0):
        a = m % 10 + a * 10
        m = m / 10

    if (n == a):
        return 'True'
    else:
        return 'False'


for each in range(0, n):
    c = 0
    a, b = map(int, raw_input().split())
    for each in range(a, b+1):
        if Palindrome_Number(each) == 'True':
            c = c + 1
    print c
