using namespace std;
string str;

int sing[30];
int dob[30][30];

int dp[1000009];
int n,m;
int recur(int pos)
{
    if(pos>=n)return 0;

    if(dp[pos]!=-1)return dp[pos];
    int ret=0;
    int x= int(str[pos]-'a');

    ret =max(ret, sing[x] + recur(pos+1));

    if(pos < n-1)
    {
        int y= int(str[pos+1]-'a');
        ret =max(ret, dob[x][y] + recur(pos+2));
    }

    dp[pos]=ret;
    return ret;
}
int main()
{
    int i,j,k,a,b;
    string temp;


    memset(dp,-1,sizeof(dp));
    for(i=0; i<30; i++)sing[i]=1;
    for(i=0; i<30; i++)
    {
        for(j=0; j<30; j++)
        {
            dob[i][j]=1;
        }
    }
    cin>>n>>m;
    cin>>str;
    for(i=0; i<m; i++)
    {
        cin>>temp;

        scanf("%d",&a);
        char ch=temp[0];
        j=int(ch -'a');
        if(temp.size()==1)
        {
            sing[j]=a;
        }
        else
        {
            j= int(temp[0]-'a');
            k=int(temp[1]-'a');
            dob[j][k]=a;
        }
    }
    int ans = recur(0);
    cout<<ans;

    return 0;
}
