#include <iostream>
#include<cmath>
#include<cstdlib>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   unsigned long long int t,n,m,k,count;
   
   cin>>t;
   
   while(t--)
   {
   	cin>>n>>m;
   	count=0;
   	if(m==1) {cout<<n<<"\n";
   	continue;}
   	while(n!=0)
   	{
   		if(n<m) {count+=n;n=0;continue;}
   		k=log(n)/log(m);
   		k=pow(m,k);
   		n-=k;
   		count++;
   	}
   	cout<<count<<"\n";
   }
    return 0;
}
