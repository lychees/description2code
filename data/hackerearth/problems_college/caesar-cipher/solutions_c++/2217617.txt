#include <iostream>
using namespace std;

int main()
{
   int t;
   cin>>t;
   string S;
   while(t--)
   {int k;
   	 cin>>k;
   	 cin>>S;
   	 
   	 for(int i=0;S[i]!='\0';i++)
   	  {
   	  	if(S[i]<='Z')
   	  	{
   	  		int temp=int(S[i]-'A');
   	  		temp=(temp+k)%26;
   	  		cout<<char(temp+'A');
   	  	}
   	  	else 
   	  	{
   	  		int temp=int(S[i]-'a');
   	  		temp=(temp+k)%26;
   	  		cout<<char(temp+'a');
   	  	}
   	  }
   	  cout<<"\n";
   }
}
