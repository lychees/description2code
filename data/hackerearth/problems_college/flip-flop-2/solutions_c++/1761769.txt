#include <iostream>
#define ll long long

using namespace std;

int main()
{
    ll n = 0, ctr = 0;
    string s = "";
    char xn = ' ';

	cin >> n;
    while (n--){
    	cin >> s;
    	for (ll i = 0; i < s.length(); i++){
    		if (i == 0){
    			if (s[i] == 'X')
    				xn = 'Y';
    			else
    				xn = 'X';
    		}
    		else{
    			if (s[i] != xn)
    				ctr++;
    			else{
    				if (s[i] == 'X')
    					xn = 'Y';
    				else
    					xn = 'X';
    			}
    		}
    	}
    	
    	cout << ctr << endl;
    	ctr = 0;
    }
    return 0;
}
