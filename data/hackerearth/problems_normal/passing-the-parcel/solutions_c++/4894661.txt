#include <iostream>
using namespace std;

void cir_queue(int q[],int a,int n)
{
	for(int i=a+1;i<n;i++)
		q[i-1]=q[i];
}

int main()
{
	int n;
	cin>>n;
	int q[n];
	for(int i=0;i<n;i++)
		q[i]=i+1;
	char s[1000];
	cin>>s;
	int i=0,j=0,k;
	for(k=0;s[k]!=NULL;k++);
	while(n!=1)
	{
		if(s[i]=='a')
			j=(j+1)%n;
		else if(s[i]=='b')
		{
			cir_queue(q,j,n);
			if(j==--n)
				j=0;
		}
		i=(i+1)%k;
	}
	cout<<q[0];
    return 0;
}
