#include<bits/stdc++.h>
#define ull unsigned long long
using namespace std;
int main()
{
    ull tc,n,i;
    scanf("%llu",&tc);
    while(tc--)
    {
        ull count=0;
        scanf("%llu",&n);
        ull a[n];
        for(i=0;i<n;i++)
            scanf("%llu",&a[i]);
        sort(a,a+n);
        for(i=0;i<n;i++)
        {
            if(a[i]==a[i+1])
                count++;// 1 1 1 2 3 3
        }
        printf("%llu\n",count);
    }
    return 0;
}
