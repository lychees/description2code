#include <bits/stdc++.h>
using namespace std;

int main()
{
    int i,n,j;
    long long int c=0,k;
    scanf("%d",&n);
    int * a=(int *)malloc(n*sizeof(int));
    for(i=0;i<n;i++){
    	scanf("%d",&a[i]);
    }
    sort(a,a+n);j=0;i=0;
    while(j<n){
        //printf("%d %d %d\n",a[i],i+1,max((i+1)*(i+1)-a[i],0));
        i++;
        k=(long long int)i*i;
    	if(a[j]>k){
            c+=k;
    	}else{
            c+=(k-a[j]);j++;
    	}
    }
    printf("%lld",c);
    return 0;
}
