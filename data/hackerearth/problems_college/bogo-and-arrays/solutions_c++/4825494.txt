#include<iostream>
using namespace std;
int main()
{
	int t,n,k;
	cin>>t;
	while(t--)
	{
		int count=0;
		long long total=0l;
		cin>>n>>k;
		int arr[n];
		for(int i=0;i<n;i++)
			cin>>arr[i];
		total=n*(n+1)/2;
		for(int i=0;i<n;i++)
		{
			if(arr[i]<=k)
			{
				count++;
			}
			else
		 	{
		 		total -= (count*(count+1))/2;
		 		count=0;
		 	}
		}
			total-=count*(count+1)/2;
			cout<<total<<endl;
	}
	return 0;
}
