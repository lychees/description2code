#include<stdio.h>

#include<string.h>

using namespace std;
#define llu long long unsigned int
#define pb push_back
#define mp make_pair
#define F first
#define S second
#define rep(i,b) for(i=0;i<b;i++)
#define mod 1000000007
#define pin(n) printf("%llu\n",n)
typedef pair<llu,llu> PII;
typedef vector<llu> VI;
VI degree[1000000];
int visited[1000000]={};
PII f(llu v)
{
    llu i;
    visited[v]=1;
    PII ret = mp(1,0);
    rep(i,degree[v].size())
    {
        if(visited[degree[v][i]]==0)
        {
            PII pr = f(degree[v][i]);
            ret.F = (ret.F*(pr.F + 1))%mod;
            ret.S = (ret.S+(pr.F + pr.S))%mod;
        }
    }
    return ret;
}
int main()
{
    llu n,i,x,y;
    scanf("%llu",&n);
    rep(i,n-1)
    {
        scanf("%llu%llu",&x,&y);
        degree[x].pb(y);
        degree[y].pb(x);
    }
    PII pr = f(0);
    pin((pr.F + pr.S + 1)%mod);
    return 0;
}
