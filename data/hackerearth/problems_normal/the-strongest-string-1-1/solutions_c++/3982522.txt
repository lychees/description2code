#include <bits/stdc++.h>

using namespace std;

bool vis[26+1];

int main()
	{
		int n,i;
		string s;
		cin >> n >> s;
		int st=0;
		int en=s.length();
		while(st<en)
			{
				char maxi='A';
				int max_index=-1;
				for(i=st;i<en;i++)
					{
						if(s[i]>maxi && !(vis[s[i]-'a']))
							{
								maxi=s[i];
								max_index=i;
							}
					}
				if(max_index==-1)
					break;
				cout << maxi;
				st=max_index+1;
				vis[maxi-'a']=true;
			}
		return 0;
	}
