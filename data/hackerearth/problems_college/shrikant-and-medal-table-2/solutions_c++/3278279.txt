#include <bits/stdc++.h>
#define ll long long
#define pb push_back
#define mk make_pair
#define pi pair<int, int>
#define fi first
#define se second
#define duf pair < int, pair < int , pair < int, string > > >
using namespace std;
bool cmp(duf a, duf b) {
	if(a.fi != b.fi) return a.fi < b.fi;
	if(a.se.fi != b.se.fi) return a.se.fi < b.se.fi;
	if(a.se.se.fi != b.se.se.fi) return a.se.se.fi < b.se.se.fi;
	return a.se.se.se > b.se.se.se;
}
int main() {
	int n;
	cin >> n;
	map < string, int > m1, m2, m3, m4;
	for(int i = 0; i < n; i++) {
		string s1, s2, s3;
		cin >> s1 >> s2 >> s3;
		m1[s1] ++;
		m2[s2] ++;
		m3[s3] ++;
	}
	vector < pair < int, pair < int , pair < int, string > > > > v;
	for(auto it : m1) {
		if(!m4.count(it.fi)) {
			m4[it.fi] = 1;
			v.pb({m1[it.fi], {m2[it.fi], {m3[it.fi], it.fi}}});
		}
	}
	for(auto it : m2) {
		if(!m4.count(it.fi)) {
			m4[it.fi] = 1;
			v.pb({m1[it.fi], {m2[it.fi], {m3[it.fi], it.fi}}});
		}
	}
	for(auto it : m3) {
		if(!m4.count(it.fi)) {
			m4[it.fi] = 1;
			v.pb({m1[it.fi], {m2[it.fi], {m3[it.fi], it.fi}}});
		}
	}
	sort(v.begin(), v.end(), cmp);
	reverse(v.begin(), v.end());
	for(int i = 0; i < v.size(); i++) {
		cout << v[i].se.se.se << " " << v[i].fi << " " << v[i].se.fi << " " << v[i].se.se.fi << endl;
	}
	return 0;
}
