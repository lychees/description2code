#include <iostream>
using namespace std;

int main()
{
    long long int t,i,j,flag;
    char str1[100001],str2[100001];
    cin>>t;
    while(t--)
    {
    	cin>>str1>>str2;
    	int p[26]={0};
    	flag=0;
    	for(i=0;str1[i]!='\0';++i)
    	{
    		++p[str1[i]-'a'];
    	}
    	for(i=0;str2[i]!='\0';++i)
    	{
    		if(p[str2[i]-'a']>0)
    		{
    			flag=1;
    			break;
    		}
    	}
    	if(flag==0)
    	cout<<"NO\n";
    	else cout<<"YES\n";
    }
    return 0;
}
