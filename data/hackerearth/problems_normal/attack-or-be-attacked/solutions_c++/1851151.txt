#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<cstring>
using namespace std;
#define mod 1000000007
long long i,d,n,m,p,q,k,z;
int main()
{
  //for(i=0;i<mod && 3*i%mod!=1;i++);printf("%lld\n",i);
  cin>>n>>m>>p>>q;
  if(n>m)
    swap(n,m);
  d=n*(n-1)%mod*(2*n-1)%mod*333333336%mod+n*(n-1)%mod*(m-n)%mod;
  d=d*2%mod;
  k=0;
  if(n-p>0 && m-q>0)
    k+=(n-p)*(m-q);
  if(n-q>0 && m-p>0)
    k+=(n-q)*(m-p);
  k=4*k%mod;
  z=n*m%mod;
  cout<<(z*(z+mod-1)%mod+mod-d+mod-k)%mod<<"\n";
	return 0;
}
