#include<iostream>
 
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		  int n,a[10000],i=0,s;
          scanf("%d",&n);
          while(i<n)
          {
          	scanf("%d",&a[i]);
          	s=255-a[i];
          	printf("%c",s);
          	i=i+1; 	
          } 
		printf("\n");
		
	}
	return 0;
}
