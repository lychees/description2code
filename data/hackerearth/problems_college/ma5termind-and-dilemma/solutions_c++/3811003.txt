#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,c0,c1,ans,l,r,n;
	string s;
	cin>>t;
	while(t--)
	{
		cin>>n;
		c0=0; c1=0;
		cin>>s;
		for(int i=0;s[i]!='\0';i++)
		{
			if(s[i]=='0') c0++;
			else c1++;
		}
		if(c0<c1)
		{
			l=0; r=0;
			for(int i=0;i<c0;i++)
			if(s[i]=='0') l++;
			for(int i=s.length()-1;i>=s.length()-c0;i--)
			if(s[i]=='0') r++;
			if(l>r) ans=c0-l;
			else ans=c0-r;
		}
		else
		{
			l=0; r=0;
			for(int i=0;i<c1;i++)
			if(s[i]=='1') l++;
			for(int i=s.length()-1;i>=s.length()-c1;i--)
			if(s[i]=='1') r++;
			if(l>r) ans=c1-l;
			else ans=c1-r;
		}
		cout<<ans<<"\n";
	}
}
