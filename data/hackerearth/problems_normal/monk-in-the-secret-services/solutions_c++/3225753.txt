#include<bits/stdc++.h>
using namespace std;
#define INF 1000000000
long long temp[200][200];
int main()
{
    long long n,m,x,y,w,t,am,s,h,ans,i,j,k;
    cin>>t;
    while(t--)
    {
        ans=0;
        cin>>n>>m;
        for(i=1;i<=n;i++)
        {
            for(j=1;j<=n;j++)
            {
                temp[i][j]=INF;
            }
            temp[i][i]=0;
        }
        for(long long i=0;i<m;i++)
        {
            cin>>x>>y>>w;
            temp[x][y]=min(temp[x][y],w);
            temp[y][x]=temp[x][y];
        }
        for(k=1;k<=n;k++)
        {
            for(i=1;i<=n;i++)
            {
                for(j=1;j<=n;j++)
                {
                    temp[i][j]=min(temp[i][k]+temp[k][j],temp[i][j]);
                }
            }
        }
        cin>>s>>am>>h;
        for(i=1;i<=n;i++)
        {
            if(i==s||i==am||i==h)
                continue;
            ans=max(temp[s][i]+2*temp[i][am]+temp[i][h],ans);
        }
        cout<<ans<<"\n";
    }
    return 0;
}
