#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    while(t--)
    {
    	string str;
    	cin>>str;
    	int n;
    	n=str[0]-'a';
    	//cout<<n<<endl;
    	for(int i=0; i<str.size(); i++)
    	{
    		char c;
    		c=str[i]-n;
    		if(c<'a') c=c+26;
    		cout<<c;
    	}
    	cout<<endl;
    }
    
    return 0;
}
