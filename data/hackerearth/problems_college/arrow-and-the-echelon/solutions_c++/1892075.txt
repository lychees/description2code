#include <iostream>
using namespace std;

int main()
{  int t;
     cin>>t;
     while(t--)
	{
	     int x,i,j,flag=0;
	     cin>>x;
	     for(i=x;i>=0;i--)
	    if(i%3==0 && (x-i)%5==0) 
	    {flag=1;break;}
	     
	     if(!flag) cout<<"-1"<<endl;
	     else
	     {
	          for(j=0;j<i;j++)
	            cout<<"5";
	          for(j=i;j<x;j++) 
	            cout<<"3";
	            cout<<endl;
	     }
	     
	     
	}

return 0;
}
