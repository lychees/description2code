tc = int(raw_input())

st = []
for i in range(tc):
	st.append(raw_input())
	
for web in st:	
	su = 4
	for letter in range(4,len(web)-4):
		if web[letter] not in "aeiou":
			su += 1
	print str(su)+"/"+str(len(web))
