#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
	int t, n, x, y, a, b, effort;
	vector<int> initials(100000), finals(100000);
	
	cin >> t;
	
	for (int i = 0; i < t; i++)
	{
		cin >> n >> x >> y;
		initials.clear();
		finals.clear();
		effort = 0;
		
		for (int j = 0; j < n; j++)
		{
			
			cin >> a >> b;
			initials.push_back(a);
			finals.push_back(b);
		}
		
		sort(initials.begin(), initials.end());
		sort(finals.begin(), finals.end());
		
		for (int j = 0; j < n; j++)
		{
			if (initials[j] > finals[j])
			{
				effort += y * (initials[j] - finals[j]);
			}
			else if (initials[j] < finals[j])
			{
				effort += x * (finals[j] - initials[j]);
			}
		}
				
		cout << effort << endl;
	}
	
	return 0;
}
