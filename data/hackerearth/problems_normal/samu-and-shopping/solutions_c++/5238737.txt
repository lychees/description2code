#include<bits/stdc++.h>
using namespace std;


int main()
{
    int t,n,i,j,ans;
    cin>>t;
    while(t--)
	{
        cin>>n;
        vector<vector<int> > shop_cost(n,vector<int>(3));
        vector<vector<int> > min_cost(n,vector<int>(3));
        for(i=0;i<n;i++)
		{
			for(j=0;j<3;j++)
				cin>>shop_cost[i][j];
		}
		//initializing the first row
		for(j=0;j<3;j++)
			min_cost[0][j]=shop_cost[0][j];

		//calculate the rest
		for(i=1;i<n;i++)
		{
			min_cost[i][0]=min(min_cost[i-1][1],min_cost[i-1][2])+shop_cost[i][0];
			min_cost[i][1]=min(min_cost[i-1][0],min_cost[i-1][2])+shop_cost[i][1];
			min_cost[i][2]=min(min_cost[i-1][0],min_cost[i-1][1])+shop_cost[i][2];
		}
		//calculate the answer
		ans=min(min(min_cost[n-1][0],min_cost[n-1][1]),min_cost[n-1][2]);
		cout<<ans<<endl;
	}
	return 0;
}
