tc = int(raw_input())
cr = raw_input()
cr = cr.split()
e = 0
o = 0
even = []
odd = []
for num in range(0, tc):
	cr[num] = int(cr[num])
for n in cr:
	if n%2 == 0:
		e = e + n
		even.append(n)
	else:
		o = o + n
		odd.append(n)
even.sort()
odd.sort()
even.append(e)
odd.append(o)
even = str(even)
even = even.translate(None, '[],')
odd = str(odd)
odd = odd.translate(None, '[],')
print even, odd
