#include <bits/stdc++.h>

#define MOD 1000000007

using namespace std;

long long modExp(long long a, long long b, long long c) {
        long long ans = 1;
        while (b != 0) {
                if (b%2 == 1)
                        ans = (ans * a) % c;

                a = (a*a) % c;
                b /= 2;
        }
        return ans;
}

int main()
{
    long long N;
    cin >> N;
    long long arr[N];
    for (int i = 0; i < N; i++) {
    	cin >> arr[i];
    }
    
    long long f = 1;
    for (int i = 0; i < N; i++) {
    	f *= arr[i];
    	f %= MOD;
    }
    long long g = arr[0];
    for (int i = 1; i < N; i++) {
    	g = __gcd(g, arr[i]);
    }
    
    cout << modExp(f, g, MOD);
    
    return 0;
}
