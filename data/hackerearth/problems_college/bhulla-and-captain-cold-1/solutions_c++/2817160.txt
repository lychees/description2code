#include <iostream>
using namespace std;

int main()
{
	int t, n;
	int a, b, c;
	int dp[3];
	cin>>t;
	while(t--) {
		dp[0] = dp[1] = dp[2] = 0;
		cin>>n;
		while(n--) {
		cin>>a>>b>>c;		
		a = a + min(dp[1], dp[2]);
		b = b + min(dp[0], dp[2]);
		c = c + min(dp[0], dp[1]);
		dp[0] = a;
		dp[1] = b;
		dp[2] = c;
		}
		cout<<min(min(dp[0], dp[1]), dp[2])<<endl;
	}
    return 0;
}
