const int N=1e6+2;
int cnt[N];
int ans[N];
inline void sieve(int n) {
    for(int i=2;i<=n;i++) {
        if(!cnt[i]) {
            for(int j=i;j<=n;j+=i) {
                cnt[j]++;
            }
        }
    }
    for(int i=2;i<=n;i++) {
        ans[cnt[i]]++;
    }
}
int main() {
    int n,q;sci(n);sci(q);
    sieve(n);
