#include<bits/stdc++.h>
using namespace std;
set<long long>s;
int main()
{
    int test; scanf("%d",&test);
    while(test--)
    {
    	long n; scanf("%ld",&n);
    	long long x;
    	while(n--)
    	{
    		scanf("%lld",&x);
    		s.insert(x);
		}
		printf("%ld\n",s.size());
		s.clear();
	}
}
