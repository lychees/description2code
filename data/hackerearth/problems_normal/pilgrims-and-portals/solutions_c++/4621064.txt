#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define pb push_back
#define mp make_pair
#define ff first
#define ss second
#define MAX 10000000000

ll graph[200][200];
ll dp[200][200];
int visit[200];
int main()
{
  int test;cin>>test;int t;int i,j,k1;
  int n,m,k;int a,b,c;
  for(t=0;t<test;t++)
  {
    cin>>n>>m>>k;
    for(i=0;i<200;i++)
    {
      visit[i]=0;
      for(j=0;j<200;j++)
      {
        dp[i][j]=MAX;
      }
    }
    for(i=0;i<m;i++)
    {
      cin>>a>>b>>c;
      graph[a][b]=c;
      graph[b][a]=c;
      dp[a][b]=c;
      dp[b][a]=c;
    }

    for(k1=1;k1<=n;k1++)
    {
      for(i=1;i<=n;i++)
      {
        for(j=1;j<=n;j++)
        {
          dp[i][j]=min(dp[i][j],dp[i][k1]+dp[k1][j]);
        }
      }
    }
    visit[1]=1;
    int count1=1;
    vector<int>shrine;
    shrine.pb(1);int sh;
    ll val=MAX;ll result=0;
    while(count1<k)
    {
      val=MAX;
      for(i=1;i<=k;i++)
      {
        if (visit[i]==0)
        {
          for(j=0;j<shrine.size();j++)
          {
            if (dp[shrine[j]][i]<val)
            {
              sh=i;
              val=dp[shrine[j]][i];
            }
          }

        }
      }
      visit[sh]=1;
      shrine.pb(sh);
      result+=val;
      count1++;

    }

    cout<<result<<"\n";
  }//test


  return 0;
}//main
