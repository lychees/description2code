#include <bits/stdc++.h>

using namespace std;

const int M = 200200;

int n, m, p[M], a[M], b[M], r[M];

void ini() {
    for (int i = 0; i < M; ++i) {
        p[i] = i;
        r[i] = 0;
    }
}

int get(int x) {
    return p[x] == x ? x : get(p[x]);
}

void uni(int x, int y) {
    x = get(x);
    y = get(y);
    if (r[x] < r[y])
    	swap(x, y);
    p[y] = x;
    r[x]++;
}

void read() {
    cin >> n >> m;
    for (int i = 1; i <= m; ++i)
        cin >> a[i] >> b[i];
}

void kill() {
    vector<int> ans;

    ini();

    for (int i = m; i >= 1; --i) {
        if (get(a[i]) == get(b[i]))
            ans.push_back(i);
        else
            uni(a[i], b[i]);
    }

    reverse(ans.begin(), ans.end());
    cout << ans.size() << "\n";
    for (int x : ans)
        cout << x << "\n";
}

int main() {
	ios_base::sync_with_stdio(false);
    read();
    kill();
    return 0;
}
