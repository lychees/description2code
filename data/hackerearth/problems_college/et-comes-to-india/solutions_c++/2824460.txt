#include<bits/stdc++.h>
using namespace std;
# define ll long long
ll siz[55];
int main(){

    int i;
    siz[1]=1;
    for(i=2;i<50;i++)
        siz[i]=(2*siz[i-1])+1;
    int t;
    cin >> t;
    for(int T = 1;T <= t; ++T){
            long long n,k;
            cin >> k>>n;
            if(siz[k]<n)
                    cout<<-1<<endl;
            else
            {
            bool flag = false;
            while(n){
                if(n%4 == 1){
                    flag = true;
                    break;
                }
                if(!(n&1))
                    n /= 2;
                else
                    break;
            }
            if(flag)
                cout << "G\n";
            else
                cout << "C\n";
            }
    }
    return 0;

}
