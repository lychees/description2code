#include <iostream>
#include<algorithm>
using namespace std;
 
int main()
{
    int n,i;
    long int k,sum=0;
    int *arr;
    cin>>n>>k;
    arr=new int[n];
    
    sum=0;int c=0;
    
    for(i=0;i<n;i++)
    cin>>arr[i];
    i=0;
    sort(arr,arr+n);
    while(i<n)
    {
    	if(sum+arr[i]<=k)
    	{
    		sum=sum+arr[i];c++;//cout<<arr[i]<<" "<<sum<<endl;
    	}
    	else
    	break;
    	i++;
   }
   cout<<c<<endl;
    
    return 0;
}
