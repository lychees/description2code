#include <iostream>
#include <stdio.h>
using namespace std;

typedef long long ll;

int main()
{
    int T;
    scanf("%d", &T);
    while(T--) {
    	ll B, W, X, Y, Z;
    	scanf("%lld %lld", &B, &W);
    	scanf("%lld %lld %lld", &X, &Y, &Z);
    	
    	ll val = 0;
    	if(X < Y+Z)
    		val += B*X;
    	else
    		val += B*(Y+Z);
    	
    	if(Y < X+Z)
    		val += W*Y;
    	else
    		val += W*(X+Z);
    	
    	printf("%lld\n", val);
    }
    return 0;
}
