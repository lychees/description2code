#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n,i,j,t;
    cin>>n;
    int a[n],b[n];
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    	b[i]=1;
    }
    for(i=0;i<n;i++)
    {
    	t=a[i];
    	for(j=0;j<n;j++)
    	{
    		if(i==j)
    		continue;
    		else if(a[j]%t==0)
    		b[j]=0;
    	}
    }
    for(i=0;i<n;i++)
    {
    	if(b[i]==1)
    	cout<<a[i]<<" ";
    }
    return 0;
}
