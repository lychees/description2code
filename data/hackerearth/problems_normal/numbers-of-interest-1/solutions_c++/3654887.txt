#include <iostream>
using namespace std;


int main()
{
	int t = 0;
	cin >> t;
	for (int i = 0; i<t; i++)
	{
		int x, y, n;
		cin >> x >> y >> n;
		long sum = 0;
		long * a = new long[x];
		for (int j = 0; j<n; j++)
		{
			if (j<x)
			{
				sum=y;
				a[j] = y;
			}
			else
			{
				int temp = 0;
				int k = 0;
				sum = 0;
				do
				{
					sum += a[k];
					if ((k + 1) < x)
					{
						a[k] = a[k + 1];
					}
					else
					{
						a[k] = sum;
					}
					k++;
				} while (k<x);
			}
		}
		cout << sum << endl;
		delete[] a;
	}
	return 0;
}
