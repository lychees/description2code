#include <bits/stdc++.h>
#define ll  long long int
using namespace std;


int main()
{
	int tc; cin>>tc;
	while(tc--){
		ll a,b,c,d,k;
		cin>>a>>b>>c>>d>>k;
		ll t=(int)floor(cbrt((float)k/a))+1;
		while(t--){
			if((d+(t)*(c+(t)*(b+a*(t))))<=k || t==0)
				{cout<<t<<"\n"; break;}
		}
		
	}
    //cout << "Hello World!" << endl;
    return 0;
}
