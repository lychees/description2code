#include <iostream>
#include <algorithm>
#include <map>
using namespace std;
int cs[10];
map<int,int> m;
int gcd(int a,int b)
{
	if(a==0)
	return 0;
	if(a>b)
	return gcd(b,a);
	if(b%a==0)
	return a;
	else 
	return gcd(b%a,a);
}
int main()
{
    int n,k,i,up,v,t,j;
    cin>>n>>k;
    for(i=0;i<n;i++)
    {
    	cin>>cs[i];
    }
    for(i=0;i<n;i++)
    {
    	v=cs[i];
    	up=v*360/gcd(v,360);
    //	cout<<up;
    	for(j=1;j<360&&v<=up;j++)
    	{
    		m[v%360]++;
    		v=(cs[i]*(j+1));
    	//	cout<<v<<"\n";
    	}
    }
    while(k--)
    {
    	cin>>t;
    	if(m[t]>0)
    	cout<<"YES\n";
    	else
    	cout<<"NO\n";
    }
    return 0;
}
