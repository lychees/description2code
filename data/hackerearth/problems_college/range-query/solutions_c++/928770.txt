#include <iostream>
#include <bits/stdc++.h>



using namespace std;
vector<int>  arr[101];
int main()
{
    int n;
    
    cin >> n;
    int val;
    for ( int i = 0; i<n; i++) {
    	cin >> val;
    	arr[val].push_back(i);
    }
    
    int q,x,y;
    cin >> q;
    
    for ( int i = 0; i < q; i++) {
    	
    	vector<int> :: iterator it1,it2;
    	cin >>x >> y >> val;
    	it1 = lower_bound( arr[val].begin(), arr[val].end(), x);
    	it2 = upper_bound( arr[val].begin(),arr[val].end(),y);
    	
    	int cnt = 0;
    	//cout << cnt <<endl;
    
    	
    	while (it1!= arr[val].end() && it1 != it2) {
    		cnt++;
    		it1++;
    	}
    	cout << cnt << endl;
    }
    return 0;
}
