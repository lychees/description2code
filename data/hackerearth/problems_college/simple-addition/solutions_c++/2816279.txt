int find(int x,int y)
{
    int ret=0;
    while(x!=1 ||  y!=1)
    {
        if(x < y)swap(x,y);
        if(x==1)
        {
            ret+= (y-1);
            break;
        }
        else if(y==1)
        {
            ret+=(x-1);
            break;
        }
        if(x ==0 || y==0)break;

        //cout<<x<<" "<<y<<endl;
        int a=x/y;
        x=x-(a*y);
        ret+=a;
    }
    if(x==0 || y==0)ret=mod;
    return ret;
}

int main()
{
    int t,i,j,k,a,b,n;
    scanf("%d",&t);
    while(t--)
    {
        int ans=mod;
        scanf("%d",&n);
        if(n<=1)
        {
             printf("0\n");
             continue;
        }
        for(i=1; i<=n/2; i++)
        {
            ans=min(ans,find(n,i));
        }
        printf("%d\n",ans);
    }


    return 0;
}
