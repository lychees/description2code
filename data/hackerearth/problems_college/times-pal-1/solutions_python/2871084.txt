for i in range(int(raw_input())):
	A = raw_input()
	B = raw_input()
	N = len(A) + 1
	num = 0
	for i in range(len(A)+1):
		if (i == 0 ):
			newString = B + A
			if (newString[::-1] == newString):
				num += 1
		elif ( i == N-1):
			newString = A + B
			if (newString[::-1] == newString):
				num += 1
		else:
			newString = A[0:i] + B + A[i:]
			if (newString[::-1] == newString):
		 		num += 1
	print (num)
