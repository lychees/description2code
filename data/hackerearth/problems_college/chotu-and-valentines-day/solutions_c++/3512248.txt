#include <iostream>
using namespace std;

int main()
{
    int n,l,r,i,q,a[100001];
    long long int s;
    cin>>n>>q;
    a[0]=0;
    for(i=1;i<=n;i++)
    {
    	cin>>a[i];
    	a[i]+=a[i-1];
    }
    while(q--)
    {
    cin>>l>>r;
       s=a[r]-a[l-1];
    cout<<s<<endl;
    }
    return 0;
}
