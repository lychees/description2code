
#include<bits/stdc++.h>

using namespace std;

#define si(x) scanf("%d",&x)
#define pri(x) printf("%d",x)
#define sll(x) scanf("%lld",&x)
#define pll(x) printf("%lld",x)
#define sstr(s) scanf("%s",s)
#define pstr(s) printf("%s",s)
#define nl printf("\n")
#define ll long long int
#define pb push_back
#define mp make_pair
#define mod 1000000007

int main()
{
    int n,k;
    si(n); si(k);
    int w[k];
    for(int i=0;i<k;++i) {
	si(w[i]);
    }
    bool brush[n+1];
    memset(brush,false,sizeof(brush));
    brush[0]=true;
    for(int i=1;i<=n;++i) {
	for(int j=0;j<k;++j) {
	    if(i-w[j]>=0) {
		if(brush[i-w[j]]) {
		    brush[i]=true;
//		    cout<<i<<endl;
		    break;
		}
	    }
	}
    }
    ll c[n+1], nc[n+1];
    memset(c,0,sizeof(c));
    memset(nc,0,sizeof(nc));
    nc[0]=1;
    for(int i=1;i<=n;++i) {
	nc[i] = (nc[i-1]+c[i-1])%mod;
	for(int j=1;j<=i;++j) {
	    if(brush[j]) {
		c[i] += nc[i-j];
		c[i] %= mod;
	    }
	}
    }
    pll((c[n]+nc[n])%mod); nl;
    return 0;
}
