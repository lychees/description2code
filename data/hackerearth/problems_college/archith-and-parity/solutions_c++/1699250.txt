#include <iostream>
using namespace std;
void count(long int n)
{
	int c=0;
	while(n>0)
	{
		if(n&1==1)
		  c++;
	  n=n>>1;	  
	}
  if(c&1)
    cout<<"odd";
  else
    cout<<"even";
}  
int main()
{
    long int t,n;
    cin>>t;
    while(t--)
   {
   	 cin>>n;
   	 count(n);
   	 cout<<endl;
   } 
    return 0;
}
