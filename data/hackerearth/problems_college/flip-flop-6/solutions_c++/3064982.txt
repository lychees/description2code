#include <iostream>
using namespace std;

int main()
{
    string s;
    int t,del=0;
    cin>>t;
    while(t--)
    {
    	int i=0,j=1;
    	del=0;
    	cin>>s;
    	i=s.length()-1;
    	j=s.length()-2;
    	for(int k=s.length()-1;k>0;k--)
    	{
    	if(s[i]==s[j])
    	{
    	 del++;
    	}
    	i--;
    	j--;
       }
       cout<<del<<"\n";
    }
    return 0;
}
