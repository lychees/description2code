string = raw_input()

sum_of_chars = sum([ord(x) for x in string])
avg = sum_of_chars / len(string)

if avg%2:
    chars = [x for x in string]
    counts = {x:chars.count(x) for x in chars}
    max_val = max(counts.values())
    for key in counts.keys():
        if counts[key] == max_val:
            print key
else:
    print string[-1::-1]
