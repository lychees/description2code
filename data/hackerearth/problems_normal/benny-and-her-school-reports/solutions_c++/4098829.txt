#include <iostream>
using namespace std;

int t;
long long int n,m,x;

int main()
{
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%lld %lld %lld",&n,&m,&x);
    	register int i;
    	long long int sum=0;
    	long long int p;
    	for(i=0;i<n;i++)
    	{
    		scanf("%lld",&p);
    		sum+=p;
    	}
    	long long int k=1;
    	k=x*(n+1)-sum;
    	if(k<=0)
    	k=1;
    	if(k<=m)
    	printf("%lld\n",k);
    	else printf("Impossible\n");
    }
    return 0;
}
