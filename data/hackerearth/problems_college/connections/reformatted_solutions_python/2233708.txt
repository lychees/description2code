import collections
list1 = []
list2 = []
n = int(raw_input())
con = int(raw_input())
while con != 0:
    l, m = map(int, raw_input().split())
    list1.append(l)
    list1.append(m)
    con = con-1

# print list1
z = collections.OrderedDict.fromkeys(list1)
nlist1 = list(z)
# print nlist1
q = int(raw_input())
while q != 0:
    h, g = map(int, raw_input().split())
    if h not in nlist1 or g not in list1:
        print "No Connection"
    else:
        intd = nlist1.index(h)
        if nlist1[intd+1] == g:
            print "Direct Connection"
        else:
            print "Indirect Connection"
    q = q-1
