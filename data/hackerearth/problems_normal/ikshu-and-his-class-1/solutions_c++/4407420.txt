#include<iostream>
#include<stdio.h>
#include<vector>
#define MOD 1000000007

using namespace std;

int comp = 0;
bool visited[100000];
long long int components[100000];
long long int fact[100000];
vector <int> adj[100000];
long long int n,k;

void fact_cal()
{
	fact[0] = 1;
	fact[1] = 1;
	for(int i = 2;i<100000;i++)
	{
		fact[i] = i * fact[i-1];
		fact[i] = fact[i] % MOD;
	}
}

void dfs(long long int s)
{
	components[comp]++;
	visited[s] = true;
	for(long long int i = 0; i<adj[s].size();i++)
	{
		if(visited[adj[s][i]] == false)
		{
			dfs(adj[s][i]);
		}
	}
}

int main()
{
	fact_cal();
	for(int i = 0; i< 100000;i++)
	{
		visited[i] = false;
	}
	cin>>n>>k;
	for(long long int i=0;i<k;i++)
	{
		int x, y;
		cin>>x>>y;
		adj[x].push_back(y);
		adj[y].push_back(x);
	}
	for(long long int  i=0;i<n;i++)
	{
		if(!visited[i])
		{
			dfs(i);
			comp++;
		}
	}
	long long int ans = 1;
	for(int i=0;i<comp;i++)
	{
		ans = ans * fact[components[i]];
		ans = ans % MOD;
	}
	
	cout<<ans;
	
	return 0;
}
