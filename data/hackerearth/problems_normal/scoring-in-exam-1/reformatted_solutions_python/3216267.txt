import numpy as np


def main():
    N, Q = map(int, raw_input().split(" "))
    score_time = sorted(map(lambda a, b: (b, a), map(
        int, raw_input().split(" ")), map(int, raw_input().split(" "))))
    score_time = score_time[::-1]
    added_time = np.cumsum([i[1] for i in score_time])
    for i in range(Q):
        K = int(raw_input())
        print added_time[K-1]


main()
