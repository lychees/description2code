#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int n,i;
	cin>>n;
	long long int a[n],sum=0,flag=0;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
		sum=sum+a[i];
    }
    sort(a,a+n);
    if(sum%2!=0)
    {
    	cout<<sum;
    	flag=1;
    }
    else
    {
    	for(i=0;i<n;i++)
    	{
    		if(a[i]%2!=0)
    		{
    			cout<<sum-a[i];
    			flag=1;
    			break;
    		}
    	}
    }
    if(flag==0)
    cout<<":(";
    return 0;
}
