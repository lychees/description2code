#include <iostream>
using namespace std;
int arr[300005]={0};
int brr[300005]={0};
int main()
{
    //cout << "Hello World!" << endl;
    int n,count=0;
    scanf("%d",&n);
    int a[n],i;
    for(i=0;i<n;i++)
    {
    	scanf("%d",&a[i]);
    }
    int j;
    for(i=0;i<n;i++)
    {
    	int flag=0;
    	for(j=0;j<i;j++)
    	{
    		if(a[i]-a[j]<0)
    		if(arr[-1*(a[i]-a[j])])
    		flag=1;
    		if(a[i]-a[j]>0)
    		if(brr[a[i]-a[j]])
    		flag=1;
    	}
    	for(j=0;j<=i;j++)
    	{
    		if(a[i]+a[j]<0)
    		arr[-1*(a[i]+a[j])]=1;
    		if(a[i]+a[j]>0)
    		brr[a[i]+a[j]]=1;
    	}
    	if(flag)
    	count++;
    }
    printf("%d\n",count);
    return 0;
}
