#include <bits/stdc++.h>
using namespace std;
bool prime[1000003];

bool foo(long int ans)
{
    for(int i=2;i<=sqrt(ans);i++)
    {
        if(ans%i==0) return false;
    }
    return true;
}
void seive()
{
	memset(prime, true, sizeof(prime));

	for(int i=2;i*i<1000003;i++)
	{
	    if(prime[i]==true)
        {

        for(int j=i+i;j<1000003;j+=i)
            {
                prime[j]=false;
            }
        }
	}
	prime[0]=prime[1]=false;
}

int main()
{
    seive();
    int t;
    cin>>t;
    while(t--)
    {
    	int a,b;
    	long int ans=0;
    	cin>>a>>b;
    	for(int i=a;i<=b;i++) if(prime[i]==true) ans+=i;
    	if(ans>100000)
        {
            if(foo(ans)) cout<<"YES\n";
            else cout<<"NO\n";
        }
        else
        {
            if(prime[ans]==true) cout<<"YES\n";
            else cout<<"NO\n";
        }
    }
    return 0;
}
