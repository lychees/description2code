#include <algorithm>
#include <cassert>
#include <cstring>
#include <iostream>

using namespace std;

#define FOR(i, a, b) for (int i = (a); i < (b); ++i)
#define REP(i, n) FOR(i, 0, n)
#define TRACE(x) cout << #x << " = " << x << endl
#define _ << " _ " <<

typedef long long llint;

const int inf = 1e9;
const int MAX = 1111;

char a[MAX][MAX];
int b[MAX][MAX];
int r[MAX], c[MAX];

int main(void) {
  int n, m;
  scanf("%d %d", &n, &m);
  REP(i, n) scanf("%s", a[i]);
  REP(i, n) REP(j, m) a[i][j] = a[i][j] == 'B';
  REP(i, n) REP(j, m)
    if (a[i][j]) {
      r[i] ^= 1, c[j] ^= 1;
      b[i][j] ^= 1;
    }


  int cnt = 0;
  REP(i, n) REP(j, m) cnt += b[i][j] ^ r[i] ^ c[j];
  printf("%d\n", cnt);
  return 0;
}
