#include <iostream>
using namespace std;

int main()
{
  int n,a[1000][1000],m,i,j,k=0,c,l=0;
  cin>>n>>m;
  c=n*m;
  for(i=0;i<n;i++)
  {
  	for(j=0;j<m;j++)
  	{
  		cin>>a[i][j];
  	}
  }
  
  while(1)
  {
  	
  	for(i=0;i<n;i++)
  {
  	for(j=0;j<m;j++)
  	{
  	if(i+j==k)
  	{
  		cout<<a[i][j]<<" ";
  	}
  	}
  }
  k++;
  l++;
  cout<<endl;
  if(l==c)
  {
  	break;
  }
  	
  	
  }
  
    return 0;
}
