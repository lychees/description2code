#include <iostream>
using namespace std;

int gcd(int x,int y){
	
	int m,i;
	if(x>y)
         m=y;
    else
         m=x;
	
for(i=m;i>=1;i--){
         if(x%i==0&&y%i==0){
           return i;
         }
    }
}
int main()
{
	int a,b,c;
	
	cin>>a>>b>>c;
	
	int x = c;
	int y = c;
	int n;
	int d = a*b*2;
	if(x<=a && y<=b){
		n = (x*y);
		int g = gcd(n,d);
		cout<<n/g<<"/"<<d/g;
		exit(0);
	}
	
	else if(x>a && y>b){
		
		int i1 = x-b;
		int i2 = y-a;
		if(i1<=a && i2<=b){
			
			n = ((2*a*b)-((a-i1)*(b-i2)));
			int g = gcd(n,d);
			cout<<n/g<<"/"<<d/g;
			exit(0);
		}
		else{
			cout<<"1/1";
			
		}
		
		
	}
	
	else{
		
		
		if(y>b){
			
			int l = y-b;
			n = ((l+x)*b);
			int g = gcd(n,d);
	
			cout<<n/g<<"/"<<d/g;
			exit(0);
			
		}
		
		else{
			
			
			int l = x-a;
			n = ((l+y)*a);
			int g = gcd(n,d);
			cout<<n/g<<"/"<<d/g;
			exit(0);
			
		}
		
		
		
	}
	
	return 0;	
}
