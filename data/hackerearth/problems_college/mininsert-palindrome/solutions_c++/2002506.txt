#include<stdio.h>
#include<algorithm>
#include<iostream>
#include<cstring>

using namespace std;
int main()
{
	int t=0;
	cin>>t;
	while(t>0)
	{
		int len=0;
		cin>>len;
char s1[100001];
char s2[100001];
cin>>s1;
int count=0;
for(int i=strlen(s1)-1;i>=0;i--)
s2[count++]=s1[i];
int l1=strlen(s1);
int l2=strlen(s2);
int lcs[l1+1][l2+1];
for(int i=0;i<=l1;i++)
{
lcs[i][0]=0;
lcs[0][i]=0;
}
//here is the logic
for(int i=1;i<=l1;i++)
{
for(int j=1;j<=l2;j++)
{
if(s1[i-1]==s2[j-1])
{
lcs[i][j]=1+lcs[i-1][j-1];
}
else
{
lcs[i][j]=max(lcs[i-1][j],lcs[i][j-1]);
}

}

}

cout<<len-lcs[l1][l2]<<"\n";

t--;
}
return 0;
}//end of the main
