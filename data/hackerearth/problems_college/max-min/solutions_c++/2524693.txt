#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int i,j,k,n,T;
    int arr[100005];
    scanf("%d%d",&n,&k);
    long long min=1000000007;
    for(i=0;i<n;i++)
    {
    	scanf("%d  ",&arr[i]);
    }
    sort(arr,arr+n);
    for(i=0;i<n-k+1;i++)
    {
    	//printf("%d",i);
    	if(arr[i+k-1]-arr[i]<min)
    		min=arr[i+k-1]-arr[i];
    }
    printf("%lld\n",min);
    return 0;
}
