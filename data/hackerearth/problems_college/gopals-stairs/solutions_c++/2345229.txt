#include<iostream>
# define MAX 100000
# define MOD 1000000007
using namespace std;
long long fibanacci(long long a, long long b,long long c,long long arr[],int index)
{
   if(index<=MAX)
   {
              arr[index]=(((a+b)%MOD)+c)%MOD;
              fibanacci(b,c,arr[index],arr,index+1);


   }



}
int main()
{
   int TC,N;
   long long arr[100001];
    cin>>TC;
    arr[1]=1;
    arr[2]=2;
    arr[3]=4;
    fibanacci(arr[1],arr[2],arr[3],arr,4);
    while(TC--)
    {
       cin>>N;
       cout<<arr[N]<<endl;


    }



}
