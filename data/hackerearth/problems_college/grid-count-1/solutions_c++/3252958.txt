#include<iostream>
#include<cstdlib>
#include<cmath>

using namespace std;

bool is_cube(long long  n) {
        int root(round(cbrt(n)));
            return n == root * root * root;
}

int main(){
    int n, k, i, j, ai, aj;
    long long count=0, count_cube=0;
    cin>>n;
    cin>>k;
    long long int arr[n][n];
    for(i=0;i<n; i++)
        for(j=0;j<n; j++)
            cin>>arr[i][j];

    for(i=0;i<n; i++)
        for(j=1;j<n; j++)
             arr[i][j] += arr[i][j-1];

    for(i=0;i<n; i++)
        for(j=1;j<n; j++)
            arr[j][i] += arr[j-1][i];
    /*
    for(i=0;i<n; i++){
        for(j=0;j<n; j++)
            cout<<arr[i][j]<<"  ";
        cout<<endl;
    }
    */
    for(i=k-1;i<n; i++){
        long long int r=0, c=0, kor=0;
        for(j=k-1;j<n; j++){
            if(i-k>=0){
                r = arr[i-k][j];
            }
            if(j-k>=0){
                c = arr[i][j-k];
            }
            if(i-k >=0 && j-k >=0)
                kor = arr[i-k][j-k];
            long long int  res = arr[i][j] - r -c +kor;
            if(is_cube(res))
                count++;
			kor=r=c=0;

        }
            //cout<<arr[i][j]<<"  ";
        //cout<<endl;
    }



    cout<<count<<endl;
    return 0;
}
