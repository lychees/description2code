#include<bits/stdc++.h>

using namespace std;

bool isPali(string &s) {
    int len=s.length();
    for(int i=0;i<len/2;i++) {
        if(s[i]!=s[len-1-i]) return false;
    }
    return true;
}

int main() {
    int test;
    scanf("%d", &test);
    while(test--) {
        string s, in;
        cin>>s>>in;
        int len=s.length();
        int cnt=0;
        for(int i=0;i<=len;i++) {
            string pre=s.substr(0, i);
            string post=s.substr(i);
            string z=pre+in+post;
            if(isPali(z)) cnt++;
        }
        printf("%d\n", cnt);
    }
}
