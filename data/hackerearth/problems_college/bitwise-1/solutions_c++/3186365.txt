#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t,c=1;
	scanf("%d",&t);
	while(t--) {
		long long int x,k,y=0;
		scanf("%lld%lld",&x,&k);
		long long int i=0;
		long long int cnt=0;
		while(k>0) {
			while(x & (1 << cnt)) {
					cnt++;
					
			}
			if(k & 1) {
				
				y += 1 << cnt;
				cnt++;
			}
			else {
				//y += ( 1 << cnt);
				cnt++;
				 
			}
			k = k >> 1;
		}
		printf("Case #%d: %lld\n",c++,y);
	}
}
