#include<bits/stdc++.h>
using namespace std;
typedef unsigned long long int ll;
ll t,a,b;

int main()
{
scanf("%d",&t);
while(t--)
{
	ll ans=0;
	scanf("%llu%llu",&a,&b);
	while(a>0 && b>0)
	{
		if(a>b)
		{
			ans+=a/b;
			a=a%b;
		}
	else
		{
			ans+=b/a;
			b=b%a;
		}
	}
	ans--;
	cout<<ans<<endl;
}
return 0;
}
