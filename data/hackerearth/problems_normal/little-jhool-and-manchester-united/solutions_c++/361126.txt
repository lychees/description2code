    #define LL long long
    #define PI 2*acos(0.0)
    #define pi pair<int,int>
    #define MAXM 2147483647

    #define MOD 1000000007
    int N;
    int B[20];
    int P[20];
    double mem[17][1<<16];
    double prob(int to_buy, int footballers, int money)
    {
    if (to_buy == 0)
    return 1;
    if (mem[to_buy][footballers] >= 0)
    return mem[to_buy][footballers];
    double mx = 0;
    for (int i = 0; i < N; i++)
    {
    if (!(footballers & (1 << i)) && (money >= B[i]))
    {
    double p = P[i] / 100.;
    double success = p * prob(to_buy - 1, footballers | (1 << i), money - B[i]);
    double failed = (1 - p) * prob(to_buy, footballers | (1 << i), money - B[i]);
    mx = max(mx, (success + failed));
    }
    }
    mem[to_buy][footballers] = mx;
    return mx;
    }
    void solve()
    {
    int c,m;
    cin >> N >> c >> m;
    for (int i = 0; i < N; i++)
    cin >> B[i] >> P[i];
    for (int i = 0; i <= c; i++)
    for (int j = 0; j < 1<<N; j++)
    mem[i][j] = -1;
    printf("%.6f\n", prob(c, 0, m));
    }
    int main()
    {
    int tc;
    scanf("%d",&tc);
    while (tc--)
    solve();
    return 0;
