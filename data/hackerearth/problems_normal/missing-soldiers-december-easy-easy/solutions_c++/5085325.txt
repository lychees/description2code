#include <iostream>
#define ll long long
using namespace std;

int main()
{
    ll t;
    cin>>t;
    ll l=1000000000;
    ll h=0;
    
  for(int i=0;i<t;i++)
    {
   
    ll x, y, d;
    cin>>x>>y>>d;
    ll x1=x+d;
    if(x1>h) 
       h=x1;
    if(x<l)
       l=x;
    }
    if(h-l==11)cout<<h-l;
   else  cout<<h-l+1;
    return 0;
}
