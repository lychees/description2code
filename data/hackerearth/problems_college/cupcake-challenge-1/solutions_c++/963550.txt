#include<cstdio>

inline int Read()
{
	register int c=getchar();
	int x=0;
	for(;(c<48 || c>57);c=getchar());
	for(;c>47 && c<58;c=getchar())
		x=(x<<1)+(x<<3)+c-48;
	return x;
}

int main()
{
	int t=Read();
	while(t--)
	{
		int a=Read(),b=Read(),c=Read();
		int ans=a^b^c;
		ans?printf("NO BILL\n"):printf("BILL\n");
	}
}
