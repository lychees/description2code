t = int(raw_input())
while (t):
    n = int(raw_input())
    c = 0
    res = [0] * 3
    while (n):
        a = map(int, raw_input().split())
        if c == 0:
            for i in range(3):
                res[i] = a[i]
        else:
            x = [min(res[1], res[2]), min(res[0], res[2]), min(res[1], res[0])]
            for i in range(3):
                res[i] = a[i] + x[i]
        c += 1
        n -= 1
    print min(res)
    t -= 1
