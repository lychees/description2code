#include<cstdio>
#include<vector>
#include<algorithm>
#include<iostream>
using namespace std;

int main()
{	int t, i;
	string str;
	char ch;
	cin>>t;
	scanf("%c",&ch);
	while ( t-- > 0 )
	{	cin>>str;
		if ( str.length() == 1 )
		{	if ( str[0] != 'R' )
				cout<<"NO"<<endl;
			else
				cout<<"YES"<<endl;
			continue;
		}
		if ( str.length() == 2 )
		{	if ( str[0] == 'R' && ( str[1]=='R' || str[1] == 'Y' ) )
				cout<<"YES"<<endl;
			else
				cout<<"NO"<<endl;
			continue;
		}
		if ( str[0] == 'Y' )
		{	cout<<"NO"<<endl;
			continue;
		}
		for ( i=0; i<str.length()-2; i++ )
		{	if ( str[i] != 'R' && str[i] != 'Y' )
				break;
			if ( str[i] == 'Y' && str[i+1] == 'Y' && str[i+2] == 'Y' )
				break;
		}
		if ( i != str.length()-2 )
			cout<<"NO"<<endl;
		else if ( ( str[i] != 'Y' && str[i] != 'R') || ( str[i+1] != 'R' &&  str[i+1] !='Y' ) )
			cout<<"NO"<<endl;
		else
			cout<<"YES"<<endl;
	}
	return 0;
}
