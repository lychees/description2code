#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n,m;
    cin>>n>>m;
    vector<vector<int> > table(n,vector<int>(m));
    for (int i=0;i<n;i++){
        for (int j=0;j<m;j++){
            cin>>table[i][j];
        }
    }
    for (int j=0;j<m;j++){
        for (int i=0,y=j;i<n && y>=0; i++,y--){
            cout<<table[i][y]<<" ";
        }
        cout<<endl;
    }

    for (int i=1;i<n;i++){
        for (int x=i,j=m-1;x<n && j>=0;x++,j--){
            cout<<table[x][j]<<" ";
        }
        cout<<endl;
    }

    return 0;
}
