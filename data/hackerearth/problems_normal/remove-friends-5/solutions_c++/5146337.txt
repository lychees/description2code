#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while(t--) {
		long n, k;
		cin >> n >> k;
		list<int> l;
		list<int>::iterator it1, it2;
		for(int i = 0; i < n; ++i) {
			int temp;
			cin >> temp;
			l.push_back(temp);
		}
		it2 = l.begin();
		it2++;
		for(it1 = l.begin(); it1 != l.end() && k && it2 != l.end();) {
			if(*it1 < *it2) {
				it1 = l.erase(it1);
				if(it1 == l.begin()) ++it2;
				else --it1;
				--k;
			} else {
				++it2;
				++it1;
			}
		}
		for(it1 = l.begin(); it1 != l.end(); ++it1) 
			cout << *it1 << " ";
		cout << "\n";
	}
    return 0;
}
