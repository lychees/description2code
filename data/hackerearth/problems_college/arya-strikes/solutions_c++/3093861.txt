#include<bits/stdc++.h>
#define pi pair<int,int>
using namespace std;

int a[100][100];
int n;

int in(int i,int j)
{
	return (i>=0 && j>=0 && i<n && j<n);
}

void bfs(int x,int y)
{
	queue<pair<int,int> > q;
	q.push(make_pair(x,y));
	while(!q.empty())
	{
		pi p = q.front();
		q.pop();
		if(in(p.first,p.second) &&a[p.first][p.second])
		{
			a[p.first][p.second] = 2;
			q.push(pi(p.first+1,p.second));
			q.push(pi(p.first,p.second+1));
		}
	}
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	
    	cin>>n;
    	for(int i=0;i<n;i++)
    		for(int j=0;j<n;j++)
    			cin>>a[i][j];
    	bfs(0,0);
    	if(a[n-1][n-1] == 2)
    		cout<<"POSSIBLE";
    	else cout<<"NOT POSSIBLE";
    	cout<<endl;
    }
}
