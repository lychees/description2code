#include <iostream>
using namespace std;

int main()
{
	int t,n,i,j,k,a[100001],count=0;
    cin>>t;
    for(i=0;i<t;i++)
    {
    	count=0;
    cin>>n;
    for(j=0;j<n;j++)
    {
    	cin>>a[j];
    }
    for(j=0;j<n-1;j++)
    {
    	for(k=j+1;k<n;k++)
    	{
    		if(((a[j]+a[k])%3)==0)
    		count++;
    	}
    }
    cout<<count<<endl;
    }
    
    return 0;
}
