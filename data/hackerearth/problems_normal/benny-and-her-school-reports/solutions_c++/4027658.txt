#include <iostream>
using namespace std;

int main() 
{
	ios_base::sync_with_stdio(false);
	int t,m,i,n;
	long long int temp,x,sum;
	cin>>t;
	while(t--)
	{
			sum=0;
			cin>>n>>m>>x;
			for(i=0;i<n;i++)
			{
					cin>>temp;
					sum+=temp;
			}
		//	cout<<sum<<" ";
			temp=((n+1)*x)-sum;
			if(temp>m)
				cout<<"Impossible\n";
			else if(temp<=1)
				cout<<"1\n";
			else
				cout<<temp<<endl;
	}
	
	return 0;
}
