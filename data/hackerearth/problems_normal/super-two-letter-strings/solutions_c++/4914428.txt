#include <bits/stdc++.h>
const long long MOD =  1000000007;
using namespace std;
long long dp[10001][11];
int n, p ;
long long solve(int ind,int cons)
{
    if(ind > n ) return 1;
    if(dp[ind][cons] !=-1) return dp[ind][cons];
    long long ans = 0;
    if(cons + 1 < p ) ans = solve(ind + 1, cons + 1);
    ans = (ans + solve(ind+1 , 0 )) % MOD;
    return dp[ind][cons] = ans;
}

int main()
{
    int t;
    cin>>t;
    while(t--) {
        cin >> n >> p;
        memset(dp,-1,sizeof dp);
        printf("%lld\n", solve(2,0) );
    }
}
