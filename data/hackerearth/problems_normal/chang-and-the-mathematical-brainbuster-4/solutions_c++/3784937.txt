#include <bits/stdc++.h>
using namespace std;
int arr[10010];
long long int dp[30010];
int n;
int main()
{
    cin>>n;
    for(int i=1;i<=n;i++)
    {
    	cin>>arr[i];
    	dp[arr[i]]++;
    }
	sort(arr+1,arr+n+1);
	dp[0]=0;
	for(int i=1;i<=30000;i++)
    dp[i]=dp[i]+dp[i-1];
    long long int ans=1e9;
    for(int i=1;i<=arr[n];i++)
    ans=min(ans,n-(dp[3*i]-dp[i-1]));
    cout<<ans<<"\n";
    return 0;
}
