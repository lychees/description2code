#include<bits/stdc++.h>
using namespace std;
int a[1000001];
#define MOD 10000007;
int main()
{
    a[1]=1;
    a[2]=1;
    a[3]=2;
    a[4]=4;
    int i,j,k,t,n;
    for(i=5;i<=1000000;i++)
    {
    	a[i]=(a[i]+a[i-1]+a[i-3]+a[i-4])%MOD;
    }
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	printf("%d\n",a[n]);
    }
	return 0;
}
