#include<iostream>
using namespace std;
int main()
{
    short t;
    cin>>t;
    while(t--)
    {
        short h,n,i,min=10000;
        cin>>n>>h;
        int s[n];
        for(i=0;i<n;i++)
        cin>>s[i];

        for(i=0;i<n;i++)
        {
            if(h>s[i])
            {
                if(min>h-s[i])
                    min=h-s[i];
            }
            else
                {
                    if(min>s[i]-h)
                    min=s[i]-h;
                }
        }
        cout<<min<<endl;
    }
    return 0;
}
