#include <iostream>
#include <bits/stdc++.h>

using namespace std;

int main()
{
	int n;
	cin >> n;
	
	while(n--) {
		long long int a;
		long long int b;
		int count = 0;
		
		cin >> a >> b;
		
		int cc = 0;
		
		while(a != 0 && b != 0) {
			int aa = a % 10;
			int bb = b % 10;
			
			if(aa + bb + cc > 9) {
				count++;
				cc = 0;
			} else {
				cc = 0;
			}
			
			a = a / 10;
			b = b / 10;


		}
		
		if(count == 0) {
			cout << "No carry operation" << endl;
		} else if(count == 1) {
			cout << count << " carry operation" << endl;
		} else {
			cout << count << " carry operations" << endl;
		}
	}
	
    //cout << "Hello World!" << endl;
    return 0;
}
