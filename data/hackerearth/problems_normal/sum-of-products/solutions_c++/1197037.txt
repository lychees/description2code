#include <iostream>
using namespace std;
#define MOD 1000000007
int main()
{
    ios::sync_with_stdio(0);
    int t;
    cin>>t;
    long long n;
    while(t--){
    	cin>>n;
    	long long r = 0,ans = 0;
    	long long prev = 1;
    	long long st, en,su;
    	for(int i=1;i*i<=n;i++){
    		ans += (n/i)*i;
    		ans %= MOD;
    		if (i>1){
    			st = n/i+1,en = n/prev;
    			//cout << st << " " << en << " " << prev << endl;
    			su = (st+en)*(en-st+1)/2;
    			su%= MOD;
    			ans += prev*su;
    			ans %= MOD;
    			prev = i;
    		}
    		if ((i+1)*(i+1)>n){
    			i++;
    			st = n/i+1,en=n/prev;
    			//cout << st << " " << en << endl;
    			if (st<i) break;
    			su = (st+en)*(en-st+1)/2;
    			su %= MOD;
    			ans += prev*su;
    			ans %= MOD;
    		}
    	}
    /*	long long ans1=0;
    	for(int i=1;i<=n;i++) ans1+= i*(n/i),ans1%= MOD;
    	cout << ans1 << "\n";*/
    	cout << ans << "\n";
    }
    return 0;
}
