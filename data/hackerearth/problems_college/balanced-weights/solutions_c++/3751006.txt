#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

ll hcf(ll a,ll b)
{
	//a=max(a,b);
	//b=min(a,b);

	if(b==0)
		return a;
	else
		return hcf(b,a%b);
}

int main()
{
	ll t;
	cin>>t;

	ll n;

	while(t--)
	{
		cin>>n;
		ll arr[n];

		for(ll i=0;i<n;i++)
		{
			cin>>arr[i];
		}

		ll h=hcf(max(arr[0],arr[1]),min(arr[0],arr[1]));
		//cout<<h;
		for(int i=2;i<n;i++)
		{
			h=hcf(max(h,arr[i]),min(h,arr[i]));
		}
		cout<<h*n<<"\n";
	}


	return 0;
}
