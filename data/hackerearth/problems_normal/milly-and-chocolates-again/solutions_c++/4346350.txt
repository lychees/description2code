#include <iostream>
#include <algorithm>
using namespace std;
typedef long int l;
l bs(l *a,l n, l k)
{
	l low=0,high=n-1,mid;
	while(low<high)
	{
		mid=low+(high-low)/2;
		if(a[mid]<k)low=mid+1;
		else high=mid;
	}
	return low;
}
int main() {
	l n;
	cin>>n;
	l *a=new l[n];
	l cf=0;
	for(l i=0;i<n;i++)
	{
		cin>>a[i];
		a[i]+=cf;
		cf=a[i];
	}
	l q;
	cin>>q;
	while(q--)
	{
		l val;
		cin>>val;
		cout<<bs(a,n,val)+1<<endl;
	}
	delete [] a;
	return 0;
}
