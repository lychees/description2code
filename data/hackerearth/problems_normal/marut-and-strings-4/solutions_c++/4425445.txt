#include <iostream>
using namespace std;

int main()
{
   int t;
   cin>>t;
   if(t<1 || t>10)
   cout<<"Invalid Test"<<endl;
   else
   {
   while(t--)
   {
   	string s;
   	cin>>s;
   	int i,l,c1=0,c2=0;
   	l=s.length();
   	if(l<1 || l>100)
   	cout<<"Invalid Input"<<endl;
   	else
   	{
 	for(i=0;i<l;i++)
 	{
 		
 		if(s[i]>=97 && s[i]<=122)
 		c1++;
 		
 		if(s[i]>=65 && s[i]<=90)
 		c2++;
 		
 		
 	}
   	if(c1==0 && c2==0)
   	cout<<"Invalid Input"<<endl;
   	else
   	{
   	if(c1>c2)
   	cout<<c2<<endl;
   	else
   	cout<<c1<<endl;
   	
   }
   }
   }
   }
    return 0;
}
