#include <iostream>
using namespace std;

long count_one(long);

int main() {
    int t; long a,b,c;
    cin >> t;
    while (t--) {
        cin >> a >> b;
        c = a ^ b;
        cout << count_one(c) << endl;
    }
    return 0;
}

long count_one(long n) {
    long count=0;
    while(n) {
        n = n & (n-1);
        count++;
    }
    return count;
}
