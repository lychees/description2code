#define LL unsigned long long
inline LL getn(){
	LL n=0,c=getchar();
	while(c< '0'||c>'9')c= getchar();
	while(c>='0'&& c<='9')
		n = (n<<3)+(n<<1)+c-'0',c= getchar();
	return n;
}
