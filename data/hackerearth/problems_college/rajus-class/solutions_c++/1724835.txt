#include <iostream>
#include<string>
#include<algorithm>
using namespace std;
struct data
{
	int a;
	string name;
};

bool comp(data a,data b)
{
	//if(a.a==b.a)return a.name<b.name;
	return a.a<b.a;
	
}
int main()
{
    //cout << "Hello World!" << endl;
    
    int t;
    cin>>t;
    while(t--)
    {
    
    data o[1000];
    int n,i,r,a;
    string name;
    cin>>n;
    for(i=0;i<n;i++)
    {cin>>a>>name;
      o[i].a=a;
      o[i].name=name;
    }
    sort(o,o+n,comp);
    cin>>r;
    cout<<o[r-1].name<<endl;
     }
    return 0;
}
