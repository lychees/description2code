#include <bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d", &x)
#define LL long long
#define MOD 1000000009

LL vis[10000000], ans[10000000];

LL arr[2][1011] = {0};

LL get(LL x){

    LL an = 0, t;
    while(x){
        t = x % 10;
        an += t * t;
        x /= 10;
    }
    return an;
}

LL make(LL x){
    vis[x] = 1;

    LL sum = get(x);

    if(vis[sum] == 0){
        make(sum);
    }
    ans[x] = ans[sum];

}

int main(){
    LL i, j, k;
    vis[89] = 1;
    vis[1] = 1;
    ans[89] = 89;
    ans[1] = 1;
    for(i = 1; i <= 1000; i++){
        if(vis[i] == 0){
            make(i);
        }
    }
    LL n;
    LL ans89 = 0, ans1 = 0;
    cin>>n;
    arr[0][0] = 1;
    int f = 0;

    for(i = 1; i <= n; i++){

        for(k = 0; k <= 1000; k++){
            arr[1-f][k] = 0;
        }

        for(k = 0; k <= 1000; k++){
            for(j = 0; j <= 9; j++){
                if(k + j*j <= 1000){
                    arr[1-f][k + j * j] += arr[f][k];
                }
            }
        }
        f = 1 - f;
    }

    for(k = 1; k <= 1000; k++){
            if(arr[f][k] > 0 && ans[ k ] == 89){
                ans89 += (arr[f][k]);
            }
            else if(arr[f][k] > 0 && ans[ k ] == 1){
                ans1 += (arr[f][k]);
            }
        }
    cout<<ans89<<" "<<ans1 + 1<<endl;

    return 0;
}
