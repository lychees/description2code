#include <iostream>
#include <algorithm>
using namespace std;

char dp[101][101][101];

void solve(){
	for(int i=0; i<101; i++){
		for(int j=0; j<101; j++){
			for(int k=0; k<101; k++){
				if((i>0 and dp[i-1][j][k]=='L') or (j>0 and dp[i][j-1][k]=='L') or (k>0 and dp[i][j][k-1]=='L')
				   or(i>0 and j>0 and dp[i-1][j-1][k]=='L') or(i>0 and k>0 and dp[i-1][j][k-1]=='L') or(k>0 and j>0 and dp[i][j-1][k-1]=='L') or (i>0 and j>0 and k>0 and dp[i-1][j-1][k-1]=='L'))
					dp[i][j][k]='W';
				else
					dp[i][j][k]='L';
			}
		}
	}
}

void ans(){
	int a, b, c;
	cin>>a>>b>>c;
	if(dp[a][b][c]=='W')
		cout<<"Ashima\n";
	else cout<<"Aishwarya\n";
}

int main(int argc, const char * argv[]) {
	// insert code here...
	int tc;
	cin>>tc;
	solve();
	while(tc--)
		ans();
	return 0;
}
