#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    string s1,s2;
    int a[27]={0};
    int i,n;
    cin>>n;
    while(n--)
    {
    	cin>>s1>>s2;
    	for(i=0;i<27;i++)
    	{
    		a[i]=0;
    	}
    	for(i=0;i<s2.length();i++)
    	{
    		a[s2[i]-'a']++;
    	}
    	for(i=0;i<s1.length();i++){
    	while(a[s1[i]-'a']>0)
    	{
    	cout<<s1[i];
    	a[s1[i]-'a']--;
    	}
    	}
    	cout<<endl;
    }
    return 0;
}
