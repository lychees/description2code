#include <bits/stdc++.h>

using namespace std;

int n, a, b, c;

long long ans = 0;

void call(int limit, char turn) {

    if(limit < 0) {
        return;
    }


    if(turn == 'a') {
        for(int i = 0; i <= a && i <= limit; i++) {
            call(limit - i, 'b');
        }
    }
    if(turn == 'b') {
        for(int i = 0; i <= b && i <= limit; i++) {
            call(limit - i, 'c');
        }
    }
    if(turn == 'c') {
        ans += (min(limit, c) + 1);
    }

}

int main()
{
    int t;

    cin >> t;

    while(t--) {
        cin >> n >> a >> b >> c;
        ans = 0;

        call(n, 'a');

        cout << ans << endl;

    }
}
