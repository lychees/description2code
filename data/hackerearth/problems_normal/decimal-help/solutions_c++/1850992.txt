#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	long long int N,D,R,C;
	vector<long long int > a;
	while(t--)
	{
		cin>>N>>D>>R;
		if(N%D==0)
		{
			cout<<"0";
		}
		else
		{
			long long int i=0;
			while(N%D!=0 && i<R)
			{
				N=N%D;
				N=N*10;
				C=N/D;
				a.push_back(C);
				i++;
			}
			if(i>=R)
			cout<<a[R-1]<<endl;
			else
			cout<<"0"<<endl;
		}
		a.clear();
	}
}
