#include <bits/stdc++.h>
#define ll long long
using namespace std;
void swap(int a,int b)
{
	int temp=a;
	a=b;
	b=temp;
}
int main()
{
	int T;
	cin >> T;
	while(T--)
	{
		int n,i,j,start=-1,end=-1;
		ll a[100001];
		cin >> n;
		for(i=1;i<=n;i++)
		{
			cin >> a[i];
		}
		for(i=2;i<=n;i++)
		{
			if(a[i]<a[i-1])
			{
				start=i-1;
				break;
			}
		}
		for(j=n;j>=1;j--)
		{
			if(a[j-1]>a[j])
			{
				end=j;
				break;
			}
		}
		int x=start,y=end,flag=1;
	    while(start<end)
	    {
	    	swap(a[start],a[end]);
	    	start++;
	    	end--;
	    }
	    for(i=2;i<n;i++)
	    {
	    	if(a[i]>a[i-1])
	    	continue;
	    	else
	    	{
	    		flag=0;
	    		break;
	    	}
	    }
	flag?cout<<x << " "<<y<<endl:cout<<"-1"<<" "<<"-1"<<endl;
	}
    
    return 0;
}
