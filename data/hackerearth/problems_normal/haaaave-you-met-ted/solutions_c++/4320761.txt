#include<bits/stdc++.h>
using namespace std;
unsigned int hm(unsigned int x)
{
    unsigned int z=x;
   unsigned int t=0;
    while(z)
    {
        z&=z-1;
        ++t;
    }
    return t;
}
void sol()
{
    unsigned int x,i,n;
    unsigned int s[100001],ans=100000;
    cin>>n;
    for(i=0;i<n;++i) cin>>s[i];
    for(i=0;i<n;++i)
        ans=min(ans,hm(s[i]));
        cout<<ans<<endl;
}
int main()
{
    int t;
    cin>>t;
    while(t--) sol();
    return 0;
}
