#include <iostream>
using namespace std;


int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */   
    int test;
    cin>>test;
    while(test--){
        int n,l,r;
        cin>>n;
        l=n;r=0;
        while((l%3>0 || r%5>0)&& l>=0){ --l;++r;}
        if(l<0)cout<<"-1"<<endl;
        else{
            for(int i=0;i<l;i++)cout<<"5";
            for(int i=0;i<r;i++)cout<<"3";
            cout<<endl;
        }
    }
    return 0;
}
