#include<bits/stdc++.h>
using namespace std;

typedef long long ll;
const int N=83;

ll fibo[N];
ll dp[N];

void preprocess() {
    fibo[0]=1;
    fibo[1]=2;
    int idx=-1;
    for(int i=2;i<N;i++) {
        fibo[i]=fibo[i-1]+fibo[i-2];
    }
    dp[0]=0;
    for(int i=1;i<N;i++) {
        if(fibo[i]%2==0) {
            dp[i]+=fibo[i]+dp[i-1];
        } else {
            dp[i]=dp[i-1];
        }
    }
    return;
}
int main() {
    preprocess();
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        ll n;
        scanf("%lld",&n);
        int res=lower_bound(fibo,fibo+N,n)-fibo;
        if(fibo[res]>n) {
            res-=1;
        }
        printf("%lld\012",dp[res]);
    }
    return 0;
}
