#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
#define F(i,a,b) for(int i = (int)(a); i <= (int)(b); i++)
#define RF(i,a,b) for(int i = (int)(a); i >= (int)(b); i--)
#define MOD 1000000007
ll dp[101][101][101],Time[101],D[101],Energy[101];
int main()
{
    #ifndef ONLINE_JUDGE
        freopen("input.txt","r",stdin);
        freopen("output.txt","w",stdout);
    #endif
    int N,E,T;
    scanf("%d%d%d",&N,&E,&T);
    F(i,1,N) scanf("%lld",&Energy[i]);
    F(i,1,N) scanf("%lld",&Time[i]);
    F(i,1,N) scanf("%lld",&D[i]);
    F(i,1,N) F(t,0,T) F(e,0,E) dp[i][t][e] = LONG_LONG_MAX;
    F(i,1,N)
    {
        F(t,0,T)
        {
            F(e,0,E)
            {
                dp[i][t][e] = min(dp[i-1][t][e] + D[i],dp[i][t][e]);
                if(Energy[i] <= e && Time[i] <= t)
                    dp[i][t][e] = min(dp[i-1][t][e]+D[i],dp[i-1][t-Time[i]][e-Energy[i]]);
            }
        }
    }
    printf("%lld",dp[N][T][E]);
    return 0;
}
