#include<bits/stdc++.h>
using namespace std;
main()
{
int t;
char w[10002],r[10002];
cin>>t;
for(int z=0;z<t;z++)
{
scanf("%s %s",w,r);
sort(r,r+strlen(r));
for(int i=0,j=0;i<strlen(w);i++)
{
  if(j>=strlen(r))
    break;
   if(w[i]>r[j])
   {
     w[i]=r[j];
     j++;
   }
}
printf("%s\n",w);
}
}
