#define SORTF(x) sort(x.begin(),x.end(),func);
#define SORT(x) sort(x.begin(),x.end())
#define pb(x) push_back(x)

#define RSORT(v) { SORT(v) ; REVERSE(v) ; }
#define ALL(v) v.begin(),v.end()
#define REVERSE(v) { reverse(ALL(v)) ; }

ll a[10000000] = {0};

int main() {
	ll i,j;
	for(i = 1;i <= 840;i++) {
		for(j = i;j <= 840;j++) {
			ll val = (i*i*i) + (j*j*j);
			if(val < 10000000) {
				a[val]++;
			} 
		}
	}
	
	ll t;
	vector<ll> v;
	for(i = 0;i <= 704977;i++) {
		if(a[i] > 1) {
			v.pb(i);
		//	cout << i << "\n";
		}
	}
//	cout << v.size() << "\n";
	S(t);
	while(t--) {
		ll n;
		ll val = -1;
		S(n);
		for(i = 0;v[i] <= n && i < v.size();i++) {
			val = v[i];
		}
		cout << val << "\n";
	}
	
}
