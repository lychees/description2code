#include <iostream>

using namespace std;

int main()
{
	int t,n,i;
	cin>>t;
	while(t--)
	{
		cin>>n;
		int arr[n];
		bool chk[n];
		for(i=0;i<n;i++)
		{
			cin>>arr[i];
			chk[i]=false;
		}
		//chk[0]=true;
		i=0;
		while(arr[i]!=-1 && chk[i]==false)
		{
			chk[i]=true;	i = arr[i];	
		}
		if(arr[i]==-1)
			cout<<"Win\n";
		else
			cout<<"Lose\n";
	}
	return 0;
}
