    #include <iostream>
    #include <algorithm>
    #include <cassert>
    using namespace std;
    int Positive[300005]={0};
    int Negative[300005]={0};
    int main()
    {
    int N,Count=0;
    cin>>N;
    assert(N>=1 && N<=5000);
    int a[N];
    for(int i=0;i<N;i++){
    cin>>a[i];
    assert(a[i]>=-100000 && a[i]<=100000);
    }
    for(int i=0;i<N;i++)
    {
    bool flag=false;
    for(int j=0;j<i;j++)
    {
    if(a[i]-a[j]<0)
    if(Negative[-1*(a[i]-a[j])])
    flag=true;
    if(a[i]-a[j]>0)
    if(Positive[a[i]-a[j]])
    flag=true;
    }
    for(int j=0;j<=i;j++){
    if(a[i]+a[j]<0)
    Negative[-1*(a[i]+a[j])]=1;
    if(a[i]+a[j]>0)
    Positive[a[i]+a[j]]=1;
    }
    if(flag)
    Count++;
    }
    cout<<Count<<endl;
    return 0;
    }
