#include <iostream>
using namespace std;

int main()
{
    int T;
    scanf("%d", &T);
    
    while(T--) {
    	long long G;
    	int N;
    	scanf("%lld%d", &G, &N);
    	if(N == 1) {
    		printf("%lld\n", 2 * G);
    		continue;
    	}
    	for(int i = 0; i < N - 1; ++i) {
    		printf("%lld ", 2 * G);
    	}
    	printf("%lld\n", 3 * G);
    }
    return 0;
}
