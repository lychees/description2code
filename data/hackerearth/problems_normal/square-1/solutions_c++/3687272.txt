#include<bits/stdc++.h>
#define ff first
#define ss second

using namespace std;

int n;

pair <double,double> a[1005],pt[1005];

double check(double ang)
{
	double minx, maxx, miny, maxy;
	
	for (int i = 0; i < n; i++)
	{
		pt[i].ff = a[i].ff*cos(ang) - a[i].ss*sin(ang);
		pt[i].ss = a[i].ff*sin(ang) + a[i].ss*cos(ang);
	}
	
	minx = miny = 1e9;
	maxx = maxy = -1e9;
	
	for (int i = 0; i < n; i++)
	{
		minx = min(minx, pt[i].ff);
		miny = min(miny, pt[i].ss);
		maxx = max(maxx, pt[i].ff);
		maxy = max(maxy, pt[i].ss);
	}
	
	return max(maxy - miny, maxx - minx);
}

int main()
{
	double l,r,mid1,mid2,ans=1e9,val1,val2;
	
	cin >> n;
	
	for (int i=0;i<n;i++)
		cin >> a[i].ff >> a[i].ss;
	
	for (int i=0;i<201;i++)
	{	
		l = i*M_PI/201;
		r = (i+1)*M_PI/201;
		
		for (int j=0;j<65;j++)
		{	
			mid1=(2*l+r)/3;
			mid2=(l+2*r)/3;
			
			val1=check(mid1);
			val2=check(mid2);
			
			if (val1<val2)
				r=mid2;
			else
				l=mid1;
				
			ans=min(ans,val1);
		}
	}
	
	printf("%.4lf\n",ans);
	
	return 0;
}
