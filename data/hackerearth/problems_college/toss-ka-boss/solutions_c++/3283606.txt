#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin >>t;
    while(t--)
    {
    	string s;
    	cin >> s;
    	int l= s.length();
    	long long int cnt=0;
    	long long int sum=0;
    	for(int i = 0;i < l;i++)
    	{
    		if(s[i] == 'H')
    		{
    			cnt++;
    			sum += cnt;
    		}
    		else
    		{
    			cnt=0;
    		}
    	}
    	cout << sum << endl;
    }
    return 0;
}
