    ll x = 0 ;
    int n , q ;
    s(n) ; s(q) ;
    map < int , int > mymap;
    REP( i , 0 , n - 1 )
    {
        s(a[i].first);
        s(a[i].second);
        if( a[i].second < a[i].first )
            swap( a[i].first , a[i].second );
        x += a[i].first ;
        mymap[a[i].second - a[i].first]++;
    }
    map < int , int >::iterator it = mymap.begin();
    while( it != mymap.end() )
    {
        if( it -> second % 2 == 1 )
        {
            x = -1;
            break;
        }
        else
            x += ( it -> first ) * 1ll * ( ( it -> second ) / 2 );
        it++;
    }
    REP( i , 0 , q - 1 )
    {
        ll no;
        sll(no);
        if( x == no )
            printf("YES\n");
        else
            printf("NO\n");
    }
    return 0;
}
