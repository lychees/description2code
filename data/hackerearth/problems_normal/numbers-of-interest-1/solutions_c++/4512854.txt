#include<iostream>
using namespace std;
int main()
{
int n,x,y,j,i;
long long int a[50],sum;
	long int no;
	cin>>no;
	while(no--)
	{
		cin>>x>>y>>n;
		for(i=1;i<=x;i++)
		a[i]=y;
		for(i=x+1;i<=50;i++)
		{
			sum=0;
			for(j=i-1;j>=i-x;j--)
			sum=sum+a[j];
			a[i]=sum;
		}
		cout<<a[n]<<"\n";
	}
	return 0;
}
