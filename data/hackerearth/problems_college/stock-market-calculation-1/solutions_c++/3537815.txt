#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll  a[1<<20];
ll maxi ;
int main()
{
	int tc;
	cin>>tc;
	while(tc--)
	{
		ll n;
		cin>>n;
		for(int i=1;i<=n;++i)
		{
			cin>>a[i];
		}
		ll ans=0;
		maxi=0;
		for(int i=n;i>=1;--i)
		{
			maxi=max(maxi,a[i]);
			ans+=maxi-a[i];
		}
		cout<<ans<<endl;

	}

	return 0;
}
