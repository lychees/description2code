import sys

t=input()

while t>0:
    s=sys.stdin.readline().strip().split()
    s=map(int,s)
    x1,y1,x2,y2=s
    r=input()
    a=x2-x1
    b=y2-y1
    u=abs(b*x1-a*y1)
    l=(a*a+b*b)**0.5
    if r==u/l:
        print "JUST MISSED" 
    elif r<u/l:
        print  "SAFE"
    else:
        print "REPLANNING"
    t=t-1
