#include <iostream>

int main()
{

	cin>>q;
	
	unsigned long long int ans = 1;
	while(q--)
	{
		cin>>d;
		ans = ans * d;
		/*for(long int i = 0;i<n;i++)
		{

		}*/
	}
	
	
	
	for(long int i = 0;i<n;i++)
		if( ans == 0)	cout<<"0 ";
		else cout<<a[i]/ans<<" ";
	cout<<endl;
    //cout << "Hello World!" << endl;
    return 0;
}
