#include <iostream>
#include <bits/stdc++.h>
using namespace std;
struct node
{
	int val;
	struct node* arr[26];
};
node* getnode()
{
	node* ptr=(node*)malloc(sizeof(node));
	ptr->val=0;
	for(int i=0;i<26;i++)
	ptr->arr[i]=NULL;
	
	return ptr;
}
void insert(node* root,string s,int p)
{
	int l=s.size();
	node* ptr=root;
	for(int i=0;i<l;i++)
	{
		int ch=((int)s[i])-97;
		if(ptr->arr[ch]==NULL)
		ptr->arr[ch]=getnode();
		
		ptr->arr[ch]->val=max(ptr->arr[ch]->val,p);
		
		ptr=ptr->arr[ch];
	}
	ptr->val=max(ptr->val,p);
}
int search(node* root, string s)
{
	int l=s.size();
	node* ptr=root;
	int ans=-1;
	for(int i=0;i<l;i++)
	{
		int ch=((int)s[i])-97;
		if(ptr->arr[ch]==NULL)
		return -1;
		ans=ptr->arr[ch]->val;
		ptr=ptr->arr[ch];
	}
	return ans;
}
int main()
{
    int n,q,p;
    cin>>n>>q;
    string s;
    struct node* root=getnode();
    while(n--)
    {
    	cin>>s>>p;
    	insert(root,s,p);
    }
    while(q--)
    {
    	cin>>s;
    	cout<<search(root,s)<<endl;
    }
    return 0;
}
