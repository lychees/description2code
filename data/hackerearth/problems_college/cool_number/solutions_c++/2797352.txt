#include<iostream>
using namespace std;


int find_coolnes(int n)
{
    int state=3;
    int coolnes=0;

    while(n)
    {
        int temp=n&1;

        if(state==1)
        {
            if(temp==0)
                state=2;
        }
        else if(state==2)
        {
            if(temp==1)
            {
                state=1;
                coolnes++;
            }
            else
                state=3;
        }
        else if(state==3)
        {
            if(temp==1)
                state=1;
        }

        n=n>>1;

    }

    return coolnes;
}


int main()
{
    int t;
    cin>>t;

    int coolnes[100000];
    for(int i=0;i<100000;i++)
        coolnes[i]=find_coolnes(i);
        
    while(t--)
    {
        int r,k;
        cin>>r>>k;
        
        int res=0;
        
        for(int i=0;i<=r;i++)
        if(coolnes[i]>=k)
            res++;
        
        cout<<res<<endl;
            
    }
}
