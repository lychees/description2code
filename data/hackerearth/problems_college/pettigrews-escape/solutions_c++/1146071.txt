#include <iostream>
#include <bitset>
using namespace std;

int visited[85][85]={0};
bitset<128> a[85];

int t,ways=0;

void move(int i,int j){
	if(i==t-1 && j==t-1) {ways++;return;}
	if (a[i][j]) return;
	a[i][j]=1;
	if(i+1<=t-1) move(i+1,j);
	if(i-1>=0) move(i-1,j);
	if(j+1<=t-1) move(i,j+1);
	if(j-1>=0) move(i,j-1);
	a[i][j]=0;
}

int main(){
	cin>>t;
	char c;
	int i;
	for(int j=0;j<t;j++) {
		for(i=0;i<t;i++) {
			cin>>c;
			a[j][i]=(c=='1');
		}
	}
	move(0,0);
	cout<<ways;
}
