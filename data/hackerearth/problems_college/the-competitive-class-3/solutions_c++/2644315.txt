#include <iostream>
using namespace std;

int main()
{
    int T;
    cin >> T;
    int *p = new int[T];
    for(int i = 0; i < T; i++) {
    	cin >> p[i];
    }
    
    for(int i = 0; i < T; i++) {
    	int c = 1;
    	for(int j = 0; j < T; j++) {
    		if(p[j] > p[i]) c++;
    	}
    	cout << c << " ";
    }
    return 0;
}
