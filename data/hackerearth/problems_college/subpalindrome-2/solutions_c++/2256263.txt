#include <iostream>
#include<cstring>
#include<cstdio>
using namespace std;
int arr[100][100];
long long palindrome(string s,int len)
{
	for(int i=0;i<len;i++)
	for(int j=0;j<len;j++)
	if(i==j) arr[i][j]=1;
 
	for(int i=2;i<=len;i++)
	{
		for(int j=0;j<len-i+1;j++)
		{
			int k=j+i-1;
 
			if(s[j]==s[k])
			{arr[j][k]=arr[j+1][k]+arr[j][k-1]+1;
			}
			else
			{arr[j][k]=arr[j+1][k]+arr[j][k-1]-arr[j+1][k-1];}
 
		}
	}
	return arr[0][len-1];
}
int main()
{
    int t;
    string s;
    scanf("%d",&t);
    cin.ignore();
    while(t--)
    {
    		memset(arr,0,sizeof arr);
    	getline(cin,s);
    	int len=s.length();
 
 
 
 
    	long long  l=palindrome(s,len);
    	
 
    	printf("%lld\n",l);
    }
    return 0;
}
