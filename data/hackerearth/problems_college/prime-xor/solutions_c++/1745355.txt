#include<bits/stdc++.h>
using namespace std;
#define maxn 32000
int prime[maxn],primes[maxn];
void sieve()
{
	int i,j;
	for(i=2;i<maxn;i++)
	{
		if(prime[i]==0)
		{
			for(j=i*i;j<maxn;j=j+i)
			{
				prime[j]=1;
			}
		}
	}
   j = 0;
    for(int i = 2;i<maxn;++i)
    {
        if(prime[i] == 0)
            primes[j++] = i;
    }
}
bool is_prime(int n)
{
    int j = 0;
    for(int i = primes[j];i*i<=n;i=primes[++j])
    {
        if(n%i == 0)
            return false;
    }
    return true;
}
int main()
{
	sieve();
int t;
cin>>t;
while(t--)
{
	int a,b;
	cin>>a>>b;
	if(a>2)
	{
		cout<<0<<endl;
	}
	else
	{
		
		int ans = 0;
    for(int i = 3;i<=b;++i)
    {
        if(is_prime(i))
            ans++;
    }
    cout << ans << endl;
	}
}	
}
