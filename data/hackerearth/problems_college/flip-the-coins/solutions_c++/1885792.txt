#include <iostream>
using namespace std;

int main()
{
    int n;
    cin >> n ;
    
    int t;
    cin >> t ;
    
    int *A=new int[n] ;
    
    for(int i=0;i<t;i++)
    {
    	int q,a,b;
    	cin >> q >> a >> b ;
    	
    	if(q==0)
    	{
    		for(int j=a;j<=b;j++)
    		{
    			if(A[j]==0)
    			    A[j]=1;
    			else
    			    A[j]=0;
    		}
    	}
    	if(q==1)
    	{
    		int c=0;
    		for(int j=a;j<=b;j++)
    		{
    			if(A[j]==1)
    			    c++;
    		}
    		cout << c << endl ;
    	}
    }
}
