import sys


def duplicacy(n):
    i = 0

    b = {}
    diff_personality = 0
    while i < len(n):
        if n[i] not in b:
            b[n[i]] = i
            diff_personality = diff_personality + 1
        i = i + 1
    print diff_personality


for place, line in enumerate(sys.stdin):
    if place:
        if place % 2 == 0:
            duplicacy(line.strip().split(' '))
