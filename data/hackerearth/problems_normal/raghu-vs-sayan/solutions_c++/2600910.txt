typedef long long ll;
#include <iostream>
#include <stdio.h>
#include <cmath>
#include <algorithm>
#include <vector>
#include <string>
#include <string.h>
#include <map>
#include <queue>

#include <fstream>
#include <stdlib.h>
#include <utility>

using namespace std;
ll a[100007];
int main()
{
	ll t,ar,b,c,i,in2,in1,sum;
	cin>>t;
	while(t--)
	{
		cin>>ar>>b>>c;
		for(i=0;i<c;i++)
		{
			cin>>a[i];
		}
		sort(a,a+c);
		in1=in2=0;
		sum=0;
		for(i=0;i<c;i++)
		{
			sum+=a[i];
			if(sum<=ar)
			in1++;
			if(sum<=b)
			in2++;
		}
		if(in1==in2)
		cout<<"Tie"<<endl;
		else if(in1>in2)
		cout<<"Raghu Won"<<endl;
		else 
		cout<<"Sayan Won"<<endl;
	}
	return 0;
}
