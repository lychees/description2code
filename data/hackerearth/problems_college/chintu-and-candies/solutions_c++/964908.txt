#include<bits/stdc++.h>
#define mp(a,b) make_pair(a,b)
#define pb(a) push_back(a)
#define scan(a) scanf("%d",&a)
#define scanl(a) scanf("%lld",&a)
#define print(a) printf("%d",a)
#define printl(a) printf("%lld",a)
#define MAX 10004
using namespace std;

typedef pair<int,int>pii;
typedef long long LL;

int rel[MAX+2];
int p[MAX];
void prime()
{
	for(int i=1;i<=MAX;i++)
	{
		p[i]=i;
	}
	for(int i=2;i*i<MAX;i++)
	{
		if(p[i]==i)
		{
			for(int j=i+i;j<=MAX;j+=i)
			{
				if(p[j]==j)
				{
					p[j]=i;
				}
			}
		}
	}
}
int gcd(int r,int a)
{
	if(a==0)
		return r;
	else
		return gcd(a,r%a);
}
void rel_prime()
{
	rel[1]=0;
	rel[0]=-1;
	prime();
	for(int i=2;i<=MAX;i++)
	{
		int temp=i;
		int den=1;
		int num=1;
		int cnst=1;
		while(temp>1)
		{
			if(p[temp]!=cnst)
			{
				den*=p[temp];
				num*=(p[temp]-1);
				cnst=p[temp];
			}
			temp/=p[temp];
		}
		rel[i]=(num*i)/den;
	}
}

int main()
{
	int t;
	scan(t);
	rel_prime();
	while(t--)
	{
		int n,x,k;
		scanf("%d %d %d",&n,&x,&k);
		int f=0;
		for(int i=n;i>x;i--)
		{
			if(rel[i]==k)
			{
				f=1;
				cout<<i<<endl;
				break;
			}
		}
		if(!f)
		{
			cout<<"-1\n";
		}
	}
}
