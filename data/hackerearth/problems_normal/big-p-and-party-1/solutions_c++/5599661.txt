#include <iostream>
#include<bits/stdc++.h>
using namespace std;
bool visited[1001];
int arr[1001];
void bfs(vector<vector<int> >& res,int v){
	memset(arr,-1,sizeof(arr));
	memset(visited,false,sizeof(visited));
	queue<int> q;
	q.push(0);
	arr[0] = 0;
	visited[0] = 1;
	while(!q.empty()){
		int curr = q.front();
		q.pop();
		for(int i=0; i<res[curr].size(); i++){
			if(!visited[res[curr][i]]){
				visited[res[curr][i]] = true;
				//if(arr[curr]==0)
				arr[res[curr][i]] = arr[curr]+1;
				q.push(res[curr][i]);
			}
		}
	}
}
int main()
{
   int v,e;
   cin>>v>>e;
   vector<vector<int> > res(v+1,vector<int>());
   int v1,v2;
   for(int i=0; i<e; i++){
   	cin>>v1>>v2;
   	res[v1].push_back(v2);
   	res[v2].push_back(v1);
   }
   bfs(res,v);
   for(int i=1; i<v; i++){
   	cout<<arr[i]<<endl;
   }
    return 0;
}
