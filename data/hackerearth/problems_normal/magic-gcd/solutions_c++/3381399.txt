#include <iostream>
#include <bits/stdc++.h>
using namespace std;
#define ll long long

int main()
{
    //cout << "Hello World!" << endl;
    ll t,i,j,k,l,m,n,o,p;
    cin>>t;
    while(t--){
    	ll n1,n2,k;
    	cin>>n1>>n2>>k;
    	ll ans=0;
    	for(i=n1-k;i<=n1+k;i++){
    		for(j=n2-k;j<=n2+k;j++){
    			if(abs(i-n1)+abs(j-n2)<=k) ans=max(ans,(__gcd(i,j)));
    		}
       }
       cout<<ans<<endl;
   }
    
    
    return 0;
}
