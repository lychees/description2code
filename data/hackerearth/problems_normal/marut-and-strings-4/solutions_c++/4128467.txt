#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    if(t>=1 && t<=10)
    {
    		while(t--)
		    {
		    	string s;
		    	cin>>s;
		    	int up=0,low=0;
		    	int flag=0;
		    	if(s.size()>=1&&s.size()<=100)
		    	{
				    	for(int i=0;i<s.size();i++)
				    	{
				    		char c=s[i];
				    		if(c>='A' && c<='Z')
				    		{
				    			flag=1;
				    			up++;
				    		}
				    		else if(c>='a' && c<='z')
				    		{
				    			flag=1;
				    			low++;
				    		}
				    	}
				    	if(flag==1)
				    	{
				    		if(low<up)
				    		cout<<low<<endl;
				    		else
				    		cout<<up<<endl;
				    	}
				    	else
				    	cout<<"Invalid Input"<<endl;
		    	}
		    	else
		    	cout<<"Invalid Input"<<endl;
		    }
		    
    }
    else
    cout<<"Invalid Test"<<endl;
    
}
