#include<bits/stdc++.h>
using namespace std;
struct node{
	int value,sign;
};
int gcd(int a,int b)
{
	if(b==0)
		return a;
	else
		return gcd(b,a%b);
}
int fun(struct node a,struct node b)
{
	if(a.value<b.value)
		return 1;

	else if(a.value==b.value)
	{
		if(a.sign<b.sign)
			return 1;
		else
			return 0;
	}
	else
		return 0;
}
int main()
{
	int c,i,j,k,m,n,t,p,q;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		int arr[n],brr[n];
		for(i=0;i<n;i++)
		{
			scanf("%d",&arr[i]);
		}
		scanf("%d%d",&p,&q);
		c=0;
		for(i=0;i<n;i++)
		{
			if(gcd(arr[i],p)!=1)
				brr[c++]=arr[i]-p;
		}
		struct node crr[c];
		if(c<q)
			printf("-1\n");
		else
		{
			for(i=0;i<c;i++)
			{
				if(brr[i]<=0)
				{
					crr[i].value=-brr[i];
					crr[i].sign=-1;
				}
				else
				{
					crr[i].value=brr[i];
					crr[i].sign=1;
				}
			}
			sort(crr,crr+c,fun);
			int drr[q];
			for(i=0;i<q;i++)
				drr[i]=crr[i].value*crr[i].sign+p;
			sort(drr,drr+q);
			for(i=0;i<q;i++)
				printf("%d ",drr[i]);printf("\n");
		}
	}
}
