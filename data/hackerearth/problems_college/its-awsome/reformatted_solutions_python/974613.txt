t = int(raw_input())
result = []
while 0 < t <= 100:
    n = raw_input()
    count1 = 0
    for i in n:
        if '4' == i or '7' == i:
            count1 = count1+1

    if count1 == 4 or count1 == 7:
        result.append("YES")
    else:
        result.append("NO")
    t = t-1
for i in result:
    print i
