#include<bits/stdc++.h>
using namespace std;

#define PRE_CPP do { ios_base::sync_with_stdio(0);	cin.tie(0); } while(0)

int main() {
	PRE_CPP;
	ostringstream oss;
	int testcase;
	cin >> testcase;
	while (testcase--) {
		int ph = 0, num = 0;
		while(cin>>num && num != -1) {
			ph^=num;
		}
		oss << ph << "\n";
	}
	cout << oss.str();
 	return 0;
}
