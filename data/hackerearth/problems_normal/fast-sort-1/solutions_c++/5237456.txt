#include <bits/stdc++.h>
using namespace std;
typedef struct foo{
	char c;
	int n;
}sats;
bool cmp(sats a, sats b)
{
	return a.n<b.n;
}
int main()
{
    int t,flag=0;
    cin>>t;
    while(t--)
    {
    	if(flag==0)
    	{
    		scanf("\n");
    		flag=1;
    	}
    	int a[100001];
    	sats b[100001];
    	string str1,str2;
    	cin>>str1>>str2;
    	int l1=str1.length();
    	int l2=str2.length();
    	for(int i=0;i<l1;i++)
    	{
    		a[str1[i]-'a']=i; // save the location where the character is present
    	}
    	
    	for(int i=0;i<l2;i++)
    	{
    		b[i].n=a[str2[i]-'a'];
    		b[i].c=str2[i];
    	}
    	sort(b,b+l2,cmp);
    	for(int i=0;i<l2;i++)
    	{
    		cout<<b[i].c;
    	}
    	cout<<"\n";
    }
    return 0;
}
