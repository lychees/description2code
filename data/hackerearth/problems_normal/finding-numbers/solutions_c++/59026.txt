#include <iostream>
#include <math.h>
using namespace std;

int main()
{
    int T, count;
    long long int Z, M, N, i, j;
    cin>>T;
    while(T--)
    {
    	cin>>Z>>M>>N;
    	count = 0;
    	for(i = 1; i <= (long long int)sqrt(Z); i++)
    	{
    		if(Z%i == 0)
    		{
    			j = Z/i;
    			if(i <= M && j <= N)
    				count++;
    			if(i <= N && j <= M)
    				count++;
    			if(i == j && i <= M && i <= N)
    				count--;
    		}
    	}
    	cout<<count<<endl;
    }
    return 0;
}
