int n, m, k;
int a[N], d[N];
set<int> b;

inline bool ok(int x) {
    d[x] = INF;
    if(x == 1) return true;
    for(int i=2; i * i <= x; ++i) if(x % i == 0) {
        d[x] = min(d[x], d[i] + d[x/i]); 
    }
    return b.count(x) == 0 && d[x] > k;
}

void smain() {
    for(; cin >> n >> k >> m; ) {
        forn(i, m) { int x; cin >> x; b.insert(x); }
        int x = 1;
        forn(i, n) {
            for(; !ok(x); ++x);
            d[x] = 1;
            a[i] = x++;
        }
        forn(i, n) cout << a[i] << ' ';
    }
}
