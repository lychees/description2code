#include <iostream>
using namespace std;

int main()
{
    unsigned int n,i,j;
    bool key=0;
    cin>>n;
    unsigned int S[n];
    for(i=0;i<n;i++)
    	cin>>S[i];
    for(i=0;i<n;i++)
    {
    	key=0;
    	for(j=0;j<n;j++)
    	{
    		if(i!=j)
    			if(S[i]%S[j]==0)
    			{
    				key =1;
    				break;
    			}
    	}
    	if(!key)
    		cout<<S[i]<<" ";
    }
    return 0;
}
