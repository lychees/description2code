#include <iostream>
#include <cmath>
#define ll long long
using namespace std;
int main()
{
	ll test,a,b,sa,sb;
    cin>>test;
    while(test--)
    {
    	cin>>a>>b;
    	sa=sqrt(a);
    	sb=sqrt(b);
    	if(sa*sa>a){sa--;}
    	if(sb*sb>b){sb--;}
    	ll res=sb-sa+(sa*sa==a);
    	cout<<res<<endl;	
    }
    return 0;
}
