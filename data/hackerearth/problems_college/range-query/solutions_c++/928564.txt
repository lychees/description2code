#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n, q, x, y, i, val;
	int hash[102][10005];
	memset(hash, 0 , sizeof(hash));
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d",&x);
		hash[x][i] = 1;
	}
	for(i=0;i<102;i++)
	{
		for(int j=1;j<10005;j++)
			hash[i][j] += hash[i][j-1];
	}
	scanf("%d",&q);
	while(q--)
	{
		scanf("%d%d%d",&x,&y,&val);
		int temp;
		if(x == 0)
			temp = hash[val][y];
		else
			temp = hash[val][y] - hash[val][x-1];
		printf("%d\n", temp);
	}
	return 0;
}
