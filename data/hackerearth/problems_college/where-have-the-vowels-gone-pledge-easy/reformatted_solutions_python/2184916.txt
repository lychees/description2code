t = int(raw_input())
s = str(raw_input())
i = 0
l = [0, 0, 0, 0, 0]
while i < t:
    if s[i] == 'a':
        l[0] = 1
    if s[i] == 'e':
        l[1] = 1
    if s[i] == 'i':
        l[2] = 1
    if s[i] == 'o':
        l[3] = 1
    if s[i] == 'u':
        l[4] = 1
    i = i+1
if l[0] == 1 and l[1] == 1 and l[2] == 1 and l[3] == 1 and l[4] == 1:
    print "YES"
else:
    print "NO"
