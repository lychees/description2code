#include <bits/stdc++.h>

using namespace std;

int main(){
	int t;
	cin >> t;
	while(t--){
		int n,c3=0,c5=0,r,flag;
		cin >> n;
		r = n%3;
		c3 = r;
		c5 = n-r;
		flag = 0;
		while(1){
			if(c3%5 == 0  && c5 >= 0 && c3 >= 0){
				for(int i=0;i<c5;i++)
					cout << 5;
				for(int i=0;i<c3;i++)
					cout << 3;
				cout << "\n";
				flag = 1;
				break;
			}
			else if(c3 < 0 || c5 < 0){
				cout << -1 << endl;
				break;
			}
			else{
				c5 = c5-3;
				c3 = c3+3;
			}
		}
	}
	return 0;
}
