#include<iostream>
#include<cstdio>
#define ll long long
#include<cstring>
#include<algorithm>
#include<cmath>
using namespace std;
ll sum(ll n)
{
	ll tmp=0;
	while(n)
	{
		tmp+=n%10;
		n/=10;
	}
	return tmp;
}
int main()
{
	ll t,n,it,flag;
	scanf("%lld",&t);
	while(t--)
	{
		flag=0;
		it=10;
		scanf("%lld",&n);
		while(it--)
		{
			if(n==1||n==4)
			{
				flag=1;
				break;
			}
			n=n*n;
			n=sum(n);
		}
		if(flag)
			printf("YES\n");
		else
			printf("NO\n");
	}
	return 0;
}
