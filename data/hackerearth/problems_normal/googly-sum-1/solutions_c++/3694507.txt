int main()
{

	cin.tie(0);

	res=0;
	for(int i=1;i<=n;i++)

		cin>>a[i];
		for(int j=1;j<i;j++)
		{
			if(s.find(a[i]-a[j])!=s.end())
			{
				res++;

			}
		}
		for(int j=1;j<=i;j++)
		{
			s.insert(a[i]+a[j]);
		}
	}
	cout<<res<<"\n";
	return 0;
}
