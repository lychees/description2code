#include<stdio.h>
#include<string.h>
#include <bits/stdc++.h>
#define k 1300001
#define MOD 1000000007
using namespace std;

int a[k];
vector<long long> prime;
long long term[100001];

void seive()
{
    int j;
    memset(a,0,sizeof a);
    for(int i=2;i<k;i++)
    {
        if(!a[i])
        {
            a[i] = i;
            j = i;
            prime.push_back(i);
            while(j<k)
            {
                a[j] = i;
                j += i;
            }
        }
    }
}
void precompute()
{
    term[0] = 1;
    for(int i=1;i<=100000;i++)
    {
        term[i] = (term[i-1]*prime[i-1])%MOD;
    }
}

int main()
{
    int t,n;
    seive();
    precompute();
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        printf("%d\n",term[n]);
    }
}
