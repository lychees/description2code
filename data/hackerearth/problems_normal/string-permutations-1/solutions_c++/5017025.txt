#include<iostream>
using namespace std;
#include<map>
#include<string>
#include<limits.h>

int main() {
	long t, i, max = LONG_MIN, result = 0;
	string s;
	std::map<string, long> myMap;
	std::map<string, long>::iterator itr;
	
	cin>>t;
	for(i=0;i<t;i++) {
		cin>>s;
		myMap[s]++;
	}
	for(itr=myMap.begin() ; itr!=myMap.end() ; itr++)
		if(itr->second > max)
			max = itr->second;
	
	for(itr=myMap.begin() ; itr!=myMap.end() ; itr++)
		if(itr->second == max)
			result += max;
	cout<<result<<endl;
	
	return 0;
}
