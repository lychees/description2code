#include <iostream>
 using namespace std;
int main()
{
	int t,i,j;
	string s;
    cin>>t;
    for(i=0;i<t;i++)
    {
    	cin>>s;
    	for(j=1;j<(s.length());j++)
    	{
    		if(abs(s[j]-s[j-1])!=abs(s[(s.length())-j]-s[(s.length())-j-1]))
    		break;
    	}
    	if(j>=s.length())
    	cout<<"Funny\n";
    	else
    	cout<<"Not Funny\n";
    }
    return 0;
}
