#include <iostream>
#include<string.h>
using namespace std;


int N,alpha[26]={0};
int main()
{
    cin>>N;
    char *str=new char[101];
    int n,flag,all,j;
    while(N--)
    {
    	all=0;
    	cin>>str;
    	n=strlen(str);
    	if(n<26)
    	{
    		cout<<"NO"<<endl;
    		continue;
    	}
    	for(int i=0;i<26;i++)
    	{
    		alpha[i]=0;
    	}
    	for(int i=0;i<n;i++)
    	{
    	   j=(int)str[i]-'a';
    	   if(alpha[j]!=1)
    	   {
    	   alpha[j]=1;	
    	   all++;
    	   if(all==26)
    	   {
    	   	cout<<"YES"<<endl;
    	   	break;
    	   }
    	   }
    	}
    	if(all!=26)
    	{
    		cout<<"NO"<<endl;
    	} 
    	
    	
    }
    
    return 0;
}
