import collections
m=int(raw_input())
for i in range(0,m):
    s=0
    smax=0
    n=int(raw_input())
    d=collections.defaultdict(int)
    for j in range(0,n):
        l=raw_input().split()
        if int(l[1])==0:
            if smax <= s:
                smax=s
            s=0
        else:
            if d[l[0]]==0:
                s+=1
                d[l[0]]=1
    if smax <= s:
        smax=s
    print smax
