#include<iostream>
#include<cmath>
using namespace std;
#define ull unsigned long long 
#define mod 1000000007
ull rangesum(ull a,ull b)
{
	a = a*(a+1)/2;
	b = b*(b+1)/2;
	return (b-a);
}
int maxy(int a,int b)
{
	if(a>b)
	return a;
	else
	return b;
}
int main()
{
	int t,n,sq,sum,i;
	int max,min;
	cin>>t;
	while(t--)
	{
		cin>>n;
		sq = sqrt(n);
		sum=0;
		for(i=1;i<=sq;i++)
		{
			max = n/i;
			min = maxy(n/(i+1),sq);
			sum = (sum + (rangesum(min,max))*i)%mod;
			if(sum>=mod) sum-=mod;
            sum+=(i*(n/i))%mod;
            if(sum>=mod) sum-=mod;
		}
		cout<<sum<<"\n";
	}
}
