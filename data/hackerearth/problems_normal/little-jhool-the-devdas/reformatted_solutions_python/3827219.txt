n = input()
A = map(int, raw_input().split())
O, E, val = [], [], 0
for i in A:
    if i % 2:
        O.append(i)
    else:
        E.append(i)
O.sort()
E.sort()
val += sum(E)
if len(O):
    if len(O) % 2:
        val += sum(O)
    else:
        val += sum(O)-O[0]
    print val
else:
    print ":("
