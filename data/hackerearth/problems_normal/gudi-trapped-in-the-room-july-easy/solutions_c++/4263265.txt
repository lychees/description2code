#include <bits/stdc++.h>
using namespace std;
string c;
set<string>ans;
int a,h;
void dfs(string s)
{
	ans.insert(s);
	int n=s.length();
	string temp=s;
	for(int i=0;i<n;i++)
	temp[(i+h)%n]=s[i];
	if(ans.find(temp)==ans.end())
	dfs(temp);
	temp=s;
	for(int i=0;i<n;i++)
	{
		if(i%2)
		temp[i]='0'+(temp[i]-'0'+a)%10;
	}
	
	if(ans.find(temp)==ans.end())
	dfs(temp);
	return ;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	ans.clear();
    	cin>>c>>a>>h;
		dfs(c);
		cout<<*(ans.begin())<<"\n";
    }
    return 0;
}
