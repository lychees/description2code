#include<bits/stdc++.h>
#define MOD 1000000007
#define ll long long
#define pb push_back
#define mk make_pair
#define N 100005
using namespace std;
vector<int> v;
map<int,int> mp;
set<int> st;
int main()
{
	int n,arr[N],a[N],ans=0;
	cin>>n;
	for(int i=0;i<n;i++)
		cin>>arr[i];
	a[0]=1;
	for(int i=1;i<n;i++)
	{
		a[i]=1;
		if(arr[i]>arr[i-1])
			a[i]+=a[i-1];
	}
	for(int i=n-2;i>=0;i--)
	{
		if(arr[i]>arr[i+1]&&a[i]<=a[i+1])
			a[i]=a[i+1]+1;
	}
	for(int i=0;i<n;i++)
		ans+=a[i];
	cout<<ans<<endl;
	return 0;
}
