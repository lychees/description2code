#include<iostream>
using namespace std;
#define k 1000000007
int dp[1000004][11]={0};
int main()
{
    for(int i=1;i<=9;i++)
        dp[1][i]=1;
    for(int i=2;i<100002;i++)
    {
    	for(int j=0;j<=9;j++)
    	{
    		dp[i][j]=(dp[i-1][j-1]%k + dp[i-1][j+1]%k)%k;
    	}
    }
    int t;
    cin>>t;
    while(t--)
    {
        int n,ans=0;
        cin>>n;
        if(n==1)
            ans=10;
        else{
            for(int i=0;i<=9;i++)
            ans= (ans%k + dp[n][i]%k)%k;
            
        }
        cout<<ans<<endl;
    }
}
