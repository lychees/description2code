#include <iostream>
using namespace std;

int gcd (int a, int b)
{
	return (b == 0) ? a : gcd (b, a%b);
}

int main(){

    int n;
    scanf("%d\n", &n);
    long Gcd=0; 
    long Max=0;

    for(int k = 0; k < n; k++){
        long temp;
        scanf("%ld", &temp);
        Gcd = gcd(Gcd, temp);
        if(temp > Max)
        {
        	Max = temp;
        }
    }

    if((Max/Gcd - n) % 2)
    {
    	cout<<"Alice"<<endl;
    	}
    else{
    	cout<<"Bob";}
    return 0;
}
