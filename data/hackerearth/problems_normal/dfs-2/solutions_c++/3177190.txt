#include <iostream>
#include<algorithm>
using namespace std;
#define ll long long int

ll a[100001];
ll lbin(ll val , ll lo,ll hi)
{
	while(hi - lo > 1)
	{
		ll mid = (lo+hi)/2;
		if(a[mid] >= val)
			hi = mid;
		else
			lo = mid;
	}
	return hi;
}

ll rbin(ll val , ll lo , ll hi)
{
	while(hi-lo >1)
	{
		ll mid = (lo+hi)/2;
		if(a[mid] <= val)
		{
			lo = mid;
		}
		else
			hi = mid;
	}
	return lo;
}

int main()
{
	ll n,i,q,x,b,pos1,pos2;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    sort(a,a+n);
    cin>>q;
    while(q--)
    {
    	cin>>x>>b;
    	if(x > a[n-1])
    	{
    		cout<<"0\n";
    		continue;
    	}
    
	    pos1 = lbin(x,-1,n);
    	pos2 = rbin(b,0,n);
    	
    //	cout<<pos1<<" "<<pos2<<endl;
	    
    	cout<<pos2-pos1+1<<endl;
    }
    return 0;
}
