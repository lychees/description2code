#include <iostream>
#include <math.h>
using namespace std;
int main(int argc, char *argv[])
{
  int t;
  long long i, n, c;
  cin>>t;
  while (t--) {
    c = 0;
    cin>>n;
    for (i = 1; i <= sqrt(n); i++) {
      if ((n % i) == 0) {
	//std::cout << i << std::endl;
	if ((i % 4) != 0)
	  c += i;
	if ( ((i*i) != n) && (((n/i) % 4) != 0) ) {
	  c += (n/i);
	}
      }
    }
    std::cout << 8*c << std::endl;
  }
  return 0;
}
