#include<iostream>
#include<cstdlib>
#include<set>
#include<cstring>
using namespace std;

int main()
{
	int T;
	cin >> T;
	for(int i =0;i < T;i++)
	{
		string a, b;
		set<char>b1;
		
		b1.clear();
		cin >> a >> b;
		bool q = false;
		if(a.length() > b.length())
		{
			string c = a;
			a = b;
			b = c;
		}
	
		for(int j = 0 ; j < b.length();j++)
		{
			b1.insert(b[j]);
		}
		for(int j = 0 ; j < a.length();j++)
		{
			if(b1.count(a[j]) == 1)
			{
					q = true;
					break;	
			}
		}
		
		if(q)
			cout << "YES" << endl;
		else
			cout << "NO" << endl;
	}
	system("pause");
	return 0;
}
