using namespace std;

#include <bits/stdc++.h>
#define ll long long int
#define s(a) scanf("%lld",&a)

ll dp[2500][2500];

ll mxi(ll a,ll b,ll c)
{
    ll max=a;
    if(b>max) max=b;
    if(c>max) max=c;
    return max;
}

int main()
{
    ll t,f,g,p,q,r,already,a,b,c,i,j,flag,mx;
    s(t);
    while(t--) {
        s(f);
        s(g);
        f=f/25;
        g=g/25;
        s(p);s(q);s(r);s(already);
        for(i=0;i<=f;i++) {
            for(j=0;j<=g;j++) dp[i][j]=0;
        }
      //  memset(dp,0,dp[0][0]*sizeof(dp));
        for(i=0;i<=f;i++) {
            mx=0;
            for(j=0;j<=g;j++) {
                flag=a=b=c=0;
                if(i>=4) {a=dp[i-4][j]+p;flag=1;}
                if(i>=5&&j>=2) {b=dp[i-5][j-2]+q;flag=1;}
                if(i>=2&&j>=4) {c=dp[i-2][j-4]+r;flag=1;}
                if(flag==1)
                dp[i][j]=mxi(a,b,c);
                if(dp[i][j]>mx) mx=dp[i][j];
                else dp[i][j]=mx;
            }
        }
        mx=0;
        for(i=0;i<=f;i++) {
        	for(j=0;j<=g;j++) if(dp[i][j]>mx) mx=dp[i][j];
        }
        if(mx<already) cout<<"-1\n";
        else
        cout<<mx-already<<endl;
    }
    return 0;
}
