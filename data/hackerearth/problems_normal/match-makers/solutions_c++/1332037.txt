#include<iostream>
#include<algorithm>
#define max 100001
using namespace std;
bool wayToSort(long int i, long int j) { return i > j; }
int main()
{
	long int boy[max],girl[max],n,i,j;
int t;
cin>>t;
while(t--)
{
	long int cnt=0;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>girl[i];
		}
		for(j=0;j<n;j++)
		{cin>>boy[j];}
		sort(girl,girl+n);
		sort(boy,boy+n,wayToSort);
		for(i=0;i<n;i++)
		{
			if(girl[i]%boy[i]==0 || boy[i]%girl[i]==0)
			{cnt++;}
			}

cout<<cnt<<endl;
}
return 0;
}
