#include<bits/stdc++.h>
//zone
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>> n;
        int arr[n + 1];
       // set<int> mrk;
        int w;
        for(w = 0 ; w < n ;w++)
        {
            cin >> arr[w];
        }
        int madhav = 0;
        int riya = 0;
        int sum1 = 0;
        int sum2 = 0;
        int i=0;
		int j=n-1;
		
		while(i<j)
		{
		
			if(sum1>sum2)
			{
				sum2+=(2*arr[j]);
				riya++;
				j--;
			}
			else if(sum2>sum1)
			{
				sum1+=arr[i];
				i++;
				madhav++;
			}
			else
			{
			
				sum2+=(2*arr[j]);
				riya++;	
				j--;
				sum1+=arr[i];
				i++;
				madhav++;
			}
		}
		if(i==j && sum1<=sum2)
		madhav++;
		if(i==j && sum1>sum2)
		riya++;

        cout<<madhav << " "<<riya<<endl;
    }
    return 0;

}
