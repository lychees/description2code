int main()
{  
	boost;
	ll t,count1,count2,n,k,x,i;
	cin>>t;
	while(t--)
	{
		count1=0;
		count2=0;
		cin>>n>>k;
		for(i=0;i<n;i++)
		{
			cin>>x;
			if(x==1)
			count1++;
			else
			count2++;
		}
		if(count2>=k)
		{
			cout<<count1+k<<endl;
		}

		{
			k=k-count2;
			count1 += count2;
			if(k%2==0)
			{
				cout<<count1<<endl;
			}
			else
			{
				cout<<count1-1<<endl;
			}
		}
	}
	return 0;
}
