def distance(target, location):
	return abs(target[0] - location[0]) + abs(target[1] - location[1])

for _ in range(int(input())):
	n, m, d = (int(inp) for inp in raw_input().split())
	locations = []
	targets = []
	shot = 0
	for _ in range(n):
		x, y = (int(inp) for inp in raw_input().split())
		locations.append((x, y))
	for _ in range(m):
		x, y = (int(inp) for inp in raw_input().split())
		targets.append((x, y))
	for target in targets:
		for location in locations:
			if distance(target, location) <= d:
				shot += 1
				break
	if shot > m/2:
		print "YES"
	else:
		print "NO"
