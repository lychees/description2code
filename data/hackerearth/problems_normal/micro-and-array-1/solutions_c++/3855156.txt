#include <iostream>
#include<bits/stdc++.h>
using namespace std;


int main()
{
	int n;
	cin>>n;
    int arr[n];
    int q;
    cin>>q;
    for(int i=0;i<n;i++)
    {
    	cin>>arr[i];
    }
    int type,x,y,v,res;
    for(int i=0;i<q;i++)
    {
    	res=0;
    	cin>>type;
    	if(type==1)
    	{
    		cin>>v;
    		for(int i=0;i<n;i++)
    		{
    		if(arr[i]>=v)
    	     	{
    			res = 1;
    			printf("%d\n",i+1);
    			break;
    		  }
    		}
    		if(res==0)printf("%d\n",-1);
    	}
    
    	if(type==0)
    	{
    		cin>>x>>y;
    		arr[x-1]=y;
    	}
    	
    	
    	
    }
    
}
