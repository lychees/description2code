#include<bits/stdc++.h>
using namespace std;
bool A[10001];
void precompute(){
	A[0] = A[1] = false;
	for(int i = 2;i<10001;i++){
		A[i] = false;
		for(int j = 1;j<i;j++)
		if(!A[j]  && !A[i-j]){
			A[i] = true;
			break;
		}
	}
}
int main(){
	int t;
	scanf("%d",&t);
	precompute();
	while(t--){
		int a,b;
		scanf("%d %d",&a,&b);
		if(A[a]||A[b])
		cout<<"Shekhar"<<endl;
		
		else cout<<"Nilesh"<<endl;
	}
	return 0;
}
