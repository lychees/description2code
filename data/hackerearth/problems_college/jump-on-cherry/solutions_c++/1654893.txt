#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t;

    cin >> t;

    while (t--) {
        int x, y, t1, cur, ans;

        cin >> x >> y;

        queue <int> q;
        map <int, int> m;

        t1 = y;
        q.push(x);
        ans = 1;
        while (!q.empty()) {
            cur = q.front();
            q.pop();

            if (m[cur] == 1)
                break;

            ans++;
            m[cur+t1] = 1;
            t1 += y;
            q.push(cur+x);
        }

        cout << ans << endl;
    }

    return 0;
}
