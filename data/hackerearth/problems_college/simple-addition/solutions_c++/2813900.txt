int send(int a,int b)
{
    if(a==0||b==0)
        return -INF;
    if(a==1||b==1)
        return (a==1)?(b-1):(a-1);
    if(a<b)
        return b/a+send(a,b%a);
    else
        return a/b+send(a%b,b);
    return 0;
}

{
     int t,i,n,ans,ans1;

     }

}
