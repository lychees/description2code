#include<bits/stdc++.h>
using namespace std;

//dp
int countWays(int n)
{
    int res[n];
    res[0] = 1; res[1] = 1;
    for (int i=2; i<n; i++)
    {
       res[i] = 0;
       for (int j=1; j<=3 && j<=i; j++)
       {
         res[i] += res[i-j];
         res[i] = res[i] % 1000000007;
       }
    }
    return res[n-1];
}

int main ()
{
    int t;
    cin >> t;
    while(t--)
    {
    	int s;
    	cin >> s;
    cout<<countWays(s + 1)%1000000007<<endl;
    }
    return 0;
}
