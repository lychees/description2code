#include <bits/stdc++.h>
using namespace std;




long long modpow(int a, int b, int MOD)
{
    long long x=1,y=a; 
    while(b > 0)
    {
        if(b%2 == 1)
        {
            x=(x*y);
            if(x>MOD) x%=MOD;
        }
        y = (y*y);
        if(y>MOD) y%=MOD; 
        b /= 2;
    }
    x%=MOD;
    return x;
}

int main(){
	int t;
	scanf("%d",&t);
	int  mod =1000000007;
	while(t--){
		int n;
		scanf("%d",&n);
		int a=3;
		long long ans = modpow(a,n,mod);
		cout<<ans<<endl;
	}
}
