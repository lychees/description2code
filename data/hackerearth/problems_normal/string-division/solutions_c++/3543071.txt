#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string a;
    	cin>>a;
    	if(a.length()>=10) {cout<<"YES"<<endl;continue;}
    	if(a.length()<4) {cout<<"NO"<<endl;continue;}
    	int arr[26]={0};
    	for(int i=0;i<a.length();i++) arr[a[i]-'a']++;
    	int t=0;
    	for(int i=0;i<26;i++) {if(arr[i]!=0) t++;}
    	if(t==1) {cout<<"NO"<<endl;continue;}
    	if(t>3) {cout<<"YES"<<endl;continue;}
    	if(t==3) 
    	{
    		if(a.length()>4) {cout<<"YES"<<endl;continue;}
    		else {cout<<"NO"<<endl;continue;}
    	}
    	
    	// t=2 , 4 <= len <= 9
    	
    	if(a.length()>=7) {cout<<"YES"<<endl;continue;}
    	
    	// 4<= len < 7
    	
    	if(a.length() < 6) {cout<<"NO"<<endl;continue;}
        
        // len = 6
        
        int p = 1;
        for(int i=0;i<26;i++) { if(arr[i]!=0) p*=arr[i];}
        if(p<6) {cout<<"NO"<<endl;continue;}
        else cout<<"YES"<<endl;
    }
}
