#include <bits/stdc++.h>

#define ll long long
#define pb push_back
#define mp make_pair
#define ld long double
#define line pair<ll, ll>

using namespace std;

const int nm = 100002;

int n, a[nm], b[nm], vt[nm];
int tree[nm];

int dem(int i) {
	int res = 0;
	while (i > 0) {
		res += tree[i];
		i -= i & (-i);
	}
	return res;
}

void them(int i) {
	while (i <= n) {
		tree[i]++;
		i += i & (-i);
	}
}

int main() {
#ifdef LOCAL
	freopen("input.txt", "r", stdin);
	cout << "asdf\n";
#endif
	int t;
	scanf("%d", &t);
	while (t--) {
		scanf("%d", &n);
		for (int i = 1; i <= n; ++i)
			scanf("%d", &a[i]);
		for (int i = 1; i <= n; ++i) {
			scanf("%d", &b[i]);
			vt[b[i]] = i;
		}
		for (int i = 1; i <= n; ++i)
			a[i] = vt[a[i]];
		ll res = 0;
		memset(tree, 0, sizeof(tree));
		for (int i = n; i >= 1; --i) {
			res += (ll) dem(a[i] - 1);
			them(a[i]);
		}
		printf("%lld\n", res);
	}
}
