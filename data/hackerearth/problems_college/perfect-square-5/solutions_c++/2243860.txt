#include<bits/stdc++.h>
using namespace std;
long long n,t,i,j,k,l;
int main()
{
	long long q,w,e,r;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
		q=sqrt(n);
		w=q*q;
		if(n==w) printf("YES\n");
		else 
		{
			e=(q+1)*(q+1);
			if((n-w)<=(e-n))
			{
				printf("-%lld\n",(n-w));
			}
			else printf("+%lld\n",(e-n));
		}
	}
	return 0;
}
