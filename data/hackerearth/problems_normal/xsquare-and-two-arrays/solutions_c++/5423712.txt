#include <bits/stdc++.h>
#define  ll long long int
using namespace std;

int main()
{
     ios_base::sync_with_stdio(false);
    int n,q;
    cin>>n>>q;
    ll a[n];
    ll b[n];
    ll dp1[n];
    ll dp2[n];
    memset(dp1,0,sizeof(dp1));
    memset(dp2,0,sizeof(dp2));
    for(int i=0;i<n;i++)
    {
    	cin>>a[i];
    }
     for(int i=0;i<n;i++)
    {
    	cin>>b[i];
    }
    int flag=0;
    dp1[0]=a[0];
    dp1[1]=a[0]+b[1];
    dp2[0]=b[0];
    dp2[1]=a[1]+b[0];
    for(int i=2;i<n;i++)
    {
    	if(flag==0)
    	{
    		dp1[i]=dp1[i-1]+a[i];
    		dp2[i]=dp2[i-1]+ b[i];
    	}
    	else
    	{
    		dp1[i]=dp1[i-1]+b[i];
    		dp2[i]=dp2[i-1]+a[i];
    	}
    	flag=~flag;
    }

    while(q--)
    {
    	int l,r,g;
    	cin>>g>>l>>r;
    	l--;
    	r--;
    	int p;
    	p=0;
    	if(g==1)
    	    p=0;
    	else
            p=1;


    	    if(l%2==p)
            {
    	    if(l-1>=0)
            { ll temp=dp1[r]-dp1[l-1];
                cout<<temp<<endl;
            }
            else
                cout<<dp1[r]<<endl;

            }
            else
        {
            if(l-1>=0)
            {   ll temp=dp2[r]-dp2[l-1];
                cout<<temp<<endl;
            }
            else
            {
                cout<<dp2[r]<<endl;
            }


    	}
    	}

    }
