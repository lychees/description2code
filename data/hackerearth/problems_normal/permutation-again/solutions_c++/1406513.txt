#include <iostream>
#define ll long long
using namespace std;

ll sum_n(ll n){
	return (n*(n+1))/2;
}

ll sum_odd_start_3(ll n){
	if(n<3)
		return 0;
	ll t=(n-1)/2;
	return t+2*sum_n(t);
}

int main()
{
	int t;
	cin>>t;
	while(t--){
		ll n;
		cin>>n;
		if(n==1){
			cout<<1<<endl;
			continue;
		}
		if(n%2==0){
			cout<<(n-1)+4*sum_n(n/2-1)<<endl;
		}else{
			cout<<n+2*sum_odd_start_3(n-2)<<endl;
		}
	}
    return 0; 
}
