#include<stdio.h>
#include<string.h>
int main(){
	int t, i, l; scanf("%d", &t); char a[1001];
	while(t-- && scanf("%s", a)){
		l = strlen(a); int h[26] = {0}, m, f = 0;
		for(i = 0; i < l; i++)	h[a[i] - 97]++;
		m = h[0];
		for(i = 1; i <= 25; i++)	if(h[i] != m){f = 1, printf("No\n"); break;}
		if(f == 0)	printf("Yes\n");
	}		
	return 0;
}
