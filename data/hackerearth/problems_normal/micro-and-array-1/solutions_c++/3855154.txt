#include <iostream>
using namespace std;

int main()
{
	int n,q,op,first_index,update_value;
   cin>>n>>q;
   int a[n];
   for(int i=0;i<n;i++){
   		cin>>a[i];
   }
   for(int i=0;i<q;i++){
   		cin>>op;
   		cin>>first_index;
   		if(op==0){
   			cin>>update_value;
   		}
   		if(op==1){
   			int k;
   			for( k=0;k<n;k++){
   				if(a[k]>=first_index){
   					cout<<k+1<<endl;
   					break;
   				}
   			}
   			if(k==n)
   				cout<<-1<<endl;
   		}
   		if(op==0){
   			a[first_index-1]=update_value;
   		}
   }
   return 0;
}
