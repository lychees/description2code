#include <iostream>
#include <cmath>
using namespace std;

int main() 
{
	int t;
	cin>>t;
	while(t--)
	{
		long long x,y,z;
		cin>>z>>x>>y;
		long long count1=0;
		for(long long i=1;i<=sqrt(z);i++)
		{
			if(z%i==0)
			{
				if(i==z/i)
				{
					if(i<=x&&i<=y)
					count1++;
				}
				else
				{
					if(i<=x&&(z/i)<=y)
					count1++;
					if(i<=y&&(z/i)<=x)
					count1++;
				}
			}
		}
		cout<<count1<<endl;
	}
}
