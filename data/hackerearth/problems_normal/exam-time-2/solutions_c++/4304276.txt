const int N=1e6+9;
bool prime[N];

void sieve()
{
    for(int i=1;i<=N;++i)
        prime[i]=true;
    int lim=sqrt(N);
    for(int i=2;i<=lim;++i)
    {
        if(prime[i]==true)
        {
            for(int j=i*i;j<=N;j+=i)
                prime[j]=false;
        }
    }
    for(int i=2;i<=(N-9);++i)
        if(prime[i])p.push_back(i);
    return;
}
