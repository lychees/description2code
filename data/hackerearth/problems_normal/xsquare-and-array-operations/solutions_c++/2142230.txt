#include <iostream>
using namespace std;
#define lli long long int
int main()
{
    int t;
    cin>>t;
    while(t--){
    	lli n;
    	cin>>n;
    	lli a[n],ans=0;
    	for(int i=0;i<n;i++)cin>>a[i];
    	for(int i=0;i<n-1;i++){
    		ans+=max(a[i],a[i+1]);
    	}
    	cout<<ans<<"\n";
    }
    return 0;
}
