#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <vector>
#include <queue>
#include <string>
#include <iterator>

using namespace std;

int main(){

int t;
scanf("%d",&t);
while(t--){
    string w,r;
    cin>>w>>r;
    sort(r.begin(),r.end());
    for(int i =0,j=0;j<r.length() && i<w.length();i++){
        if(w[i]>r[j]){
          w[i]=r[j];
          j++;
        }
    }
cout<<w<<endl;
}
}
