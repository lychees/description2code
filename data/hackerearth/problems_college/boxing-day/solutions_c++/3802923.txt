#include <iostream>
using namespace std;

int main()
{
    int n,m,i,cnt=0;
    cin>>n>>m;
    int a[n];
    for(i=0;i<n;i++)
    	a[i]=1;
    while(m--)
    {
    	int p,q;
    	cin>>p>>q;
    	a[p-1]=1;
    	a[q-1]=0;
    }
    for(i=0;i<n;i++)
    {
    	if(a[i]==1)
    		cnt++;
    }
    cout<<cnt<<endl;	
    return 0;
}
