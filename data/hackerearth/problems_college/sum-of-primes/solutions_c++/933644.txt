#include<iostream>
#include<cstdio>
#include<cmath>
#include<vector>
#define ll long long int
using namespace std;
bool b[1000005];
ll dp[1000005];
vector<ll> p;
int check(ll n)
{
	for(ll i=0;p[i]*p[i]<=n;i++)
	{
		if(n%p[i]==0)
			return 0;	
	}
	return 1;
}
int main()
{
    ll t,i,k,m,n,x,flag,s;
    for(i=0;i<=1000000;i++)
    {
        b[i]=true;
    }
    b[0]=false;
    b[1]=false;
    dp[0]=0;
    dp[1]=0;
    for(i=2;i<=1000000;i++)
    {
        if(b[i]==true)
        {
        	p.push_back(i);
        	dp[i]=dp[i-1]+i;
            for(k=2*i;k<=1000000;k+=i)
            {
                b[k]=false;
            }
        }
        else{
        	dp[i]=dp[i-1];	
        }
    }
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%lld %lld",&n,&m);
    	s=dp[m]-dp[n-1];
    	if(s<=1000000){
    	if(b[s]==true)
    		printf("YES\n");
    	else
    		printf("NO\n");
    	}
    	else
    	{
    		s=check(s);
    		if(s==0)
    			printf("NO\n");
    		else
    			printf("YES\n");
    	}
    }
}
