#include<bits/stdc++.h>
using namespace std;

long double dp[3002][2];
long int x[10000],y[10000],z[10000];

long double dist(int i,int j) {
	long long d=(x[j]-x[i])*(x[j]-x[i])+(y[j]-y[i])*(y[j]-y[i]);
	return (long double)((long double)sqrt(d));
}

int main() {
	long int i,j,n,value;
	long double health,distance;
	cin >> n;
	for(i=1;i<=n;i++) {
		cin >> x[i] >> y[i] >> z[i];
	}

	dp[0][0]=dp[0][1]=0;
	dp[1][0]=0;
	dp[1][1]=z[1];
	for(i=2;i<=n;i++) {
		dp[i][1]=-1ll<<60;
		for(j=i-1;j>0;j--) {
			distance=dist(i,j);
			dp[i][1]=max(dp[i][1],z[i]+dp[j][1]-distance);
		
		}
	}

	cout << fixed;
	cout.precision(6);
	cout << dp[n][1] << endl;
	return 0;
}
