#include <iostream>
#include<string.h>
using namespace std;


int compare(char a,char b)
{
	if(a==b)
	return 1;
	else
	return 0;
}

int main()
{
	int t,l,i,temp=0;
	cin>>t;
	
	string word;
	
	while(t--)
	{
		temp=0;
		
		cin>>word;
		l=word.size();
		
		for(i=0;i<l-1;i++)
		if(compare(word[i],word[i+1]))
			temp=1;
		
		
			if(temp>0)
			cout<<"SLAP"<<endl;
			else
			cout<<"KISS"<<endl;
		
		
	}
    return 0;
}
