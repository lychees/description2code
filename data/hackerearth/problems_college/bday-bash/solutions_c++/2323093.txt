#include <iostream>
using namespace std;
typedef long long ll;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	ll tt,sf,bf,f,flag=0;
    	cin>>tt>>sf>>bf>>f;
    	ll sfs=100*sf,bfs=50*bf,fs=25*f;
    	ll sum=sfs+bfs+fs;
    	ll sfsp=sfs,bfsp=bfs,fsp=fs;
     while(sum<tt)
    	{
    		ll leftover=tt-sum;
    		if(leftover>=sfs)
    		{
    			leftover-=sfs;
    			sfsp+=sfs;
    			sum+=sfs;
    			flag=1;
    		}
   	if(leftover>=bfs)
    		{
    			leftover-=bfs;
    			bfsp+=bfs;
    			sum+=bfs;
    			flag=1;
    			
    		}
     if(leftover>=fs)
    		{
    	
    			leftover-=fs;
    			fsp+=fs;
    			sum+=fs;
    			
    			flag=1;
    		}
    	else
    		{
    			flag=0;
    			goto a ;
    		}
    	
    	}
    	a:
    	if(flag==0)
    	cout<<"No"<<endl;
    	else
    	cout<<sfsp<<" "<<bfsp<<" "<<fsp<<" "<<endl;
    }
}
