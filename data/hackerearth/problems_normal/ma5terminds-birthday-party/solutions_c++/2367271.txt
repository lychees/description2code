#include <bits/stdc++.h>
using namespace std;
int  a[1000001]={0};
int  cf[1000001]={0};

int prime[1000005];
int p=1;
void generate(int y)
    {int i,j;
     prime[0]=2;
	 for(i=3;i*i<=y;i+=2)
        {if(a[i])
          continue;
         prime[p++]=i;
		for(j=i*i;j<=y;j+=2*i)
           {a[j]=1;
           }
        }
        for(i;i<=y;i+=2)
          if(!a[i])
            prime[p++]=i;
     //   for(i=0;i<p;i++)
             ;//cout<<prime[i]<<" ";
    }
  int main()
    {
     //cout<<"in";
     generate(1000000);
     //cout<<p<<endl;
     cf[1]=0;
     cf[2]=1;
     cf[3]=2;
     cf[4]=3;
     
     for(int i=5;i<=1000000;i+=2)
     {
     	cf[i]=(1-a[i])+cf[i-1];
     	cf[i+1]=cf[i];
     //	cout<<i<<" "<<cf[i]<<endl;
	 }
	 int t;
	 scanf("%d",&t);
	 int l,r;
	 while(t--)
	 {
	 	scanf("%d%d",&l,&r);
	 	int c1=r-l+1;
		c1-=cf[r]-cf[l-1];
	 	printf("%d\n",c1);
	 	
	 }
	 return 0;
}
