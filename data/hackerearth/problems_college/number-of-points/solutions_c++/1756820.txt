#include<bits/stdc++.h>
using namespace std;
long long gcd(long long a,long long b){
	if(b==0){
		return a;
	}
		
	else
		return gcd(b,a%b);
}
int main(){
	long long t;
	cin>>t;
	while(t--){
	long long x1,x2,y1,y2;
	cin>>x1>>y1>>x2>>y2;
	long long res=gcd(abs(x1-x2),abs(y1-y2));
	if(res==0)
	cout<<res<<endl;
	else
	cout<<res-1<<endl;
	}
	return 0;
}
