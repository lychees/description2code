#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,x;
    scanf("%d",&t);
   int a[1000];
    while(t--)
    {
             cin>>n;
             
             for(int i=0;i<n;i++)
             cin>>a[i];
             int ans=0;
             for(int i=0;i<n-1;i++)
             {
                     for(int j=i+1;j<n;j++)
                     {
                            int k=(a[i]+a[j])%3;
                            if(k==0)
                            ans++;
                     }
             }
             cout<<ans<<endl;
    }
    return 0;
}
