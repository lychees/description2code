#include<stdio.h>
#include<iostream>
#include<string.h>
#include<math.h>
#include<algorithm>
#include<stdlib.h>
#include<vector>
#include<map>
#include<set>

using namespace std;

main()
{
	int t;
	cin>>t;
	
	while(t--)
	{
		long int n,a[100000],ct=0;
		cin>>n;
		
		for(long int i=0;i<n;i++)
		cin>>a[i];
		
		for(long int i=0;i<n;i++)
		{
			for(long int j=i;j<n;j++)
			{
				if(a[i]==a[j])
				ct++;
			}
		}
		cout<<ct<<endl;
	}
}
