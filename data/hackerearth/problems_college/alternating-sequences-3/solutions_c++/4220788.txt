#include <bits/stdc++.h>
#define ll long long
#define MY_LONG_MAX (((1ll<<62))|((1ll<<62)-1))
#define MY_LONG_MIN (1ll<<63)
ll int n;
ll int a[5010],dp[5010];
using namespace std;
bool opp(ll int x,ll int y)
{
	if(x>0 && y<0)
	return 1;
	if(x<0 && y>0)
	return 1;
	return 0;
} 
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	cin>>n;
	ll int i,j,ans=0;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
	}
	for(i=0;i<n;i++)
	{
		dp[i]=1;
		for(j=i-1;j>=0;j--)
		{
			if((abs(a[j])<abs(a[i])) && opp(a[i],a[j]))
			dp[i]=max(dp[i],dp[j]+1);
		}
		ans=max(ans,dp[i]);
	}
	cout<<ans<<endl;
	return 0;
}
