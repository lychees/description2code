#include <bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	while(cin>>n)
	{
		getchar();
		int A[n+1];
		memset(A,0,sizeof(A));
		int val=0;
		stringstream ss;
		string s;
		getline(cin,s);
		vector<int>v;
		ss<<s;
		int x;
		while(ss>>x)
		{
			v.push_back(x);
		}
		/*if(v.size()==0)
		{
			cout<<"-1\n";
			continue;
		}*/
		vector<int>vv;
		int flag=0;
		string ans="";
		for(int i=0;i<v.size();i++)
		{
			int ele=v[i];
			while(val<ele)
			{
				val++;
				vv.push_back(val);
				ans+="C";
			}
			if(A[ele]==0)
			{
				while(1)
				{
					int xx=vv[vv.size()-1];
					A[xx]=1;
					vv.pop_back();
					ans+="K";
					if(xx==ele) break;
				}
			}
			else
			{
				flag=1;
				break;
			}
		}
		if(flag==1) cout<<"-1\n";
		else cout<<ans<<"\n";
	}
	return 0;
}
