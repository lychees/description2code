#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstring>
#include <ctime>
#include <fstream>
#include <iomanip>
#include <iostream>
#include <map>
#include <set>
#include <vector>
using namespace std;
char s[1003];
int main() {
  scanf("%s", s);
  int i, j, jj;
  int maxi = 1;
  int res;
  for (i = 0; i < 26; i++) {
    res = 1;
    for (j = 0; s[j] && s[j] - 'A' != i; j++)
      ;
    for (jj = j; s[jj]; jj++) {
      if (s[jj] - 'A' == i && ((jj - j) & 1)) {
        j = jj;
        res++;
      }
    }
    maxi = max(maxi, res);
  }
  printf("%d\n", maxi);
  return 0;
}
