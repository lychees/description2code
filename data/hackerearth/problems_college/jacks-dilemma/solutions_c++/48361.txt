#include<iostream>
using namespace std;

int main()
{
        int t,n,j,l;
        int stime,ltime;
        char dir;
        cin>>t;
        while(t--)
        {
                cin>>n>>j>>l>>dir;
                stime = (j*10);

                if(dir == 'U')
                {
                        //if(l <= j)
                        ltime = ((2*n - l)*5) + 30;
                        if(j == n)
                                ltime -= 10;
                }
                else
                {
                        if(l >= j)
                                ltime = (l*5) + 20;
                        else
                                ltime = (l+(2*n))*5 + 40;
                }
                if(ltime <= stime)
                        cout<<"Lift "<<ltime<<endl;
                else
                        cout<<"Stairs "<<stime<<endl;
        }
        return 0;
}
