#define ul unsigned long
#define ll long long
#define pii std::pair<int, int>

#define all(x) x.begin(), x.end()

const int MOD = 1e9 + 7;

struct myCompare {
	bool operator() (const pii &l, const pii &r) const {
		return l.first < r.first;
	}
};


const int MAXN = 2e3 + 3;
char s[MAXN][MAXN];
pii cnt[MAXN][MAXN];
int m, n;
void solve() {
	scanf("%d%d", &m, &n);
	fore(i, 1, m) scanf("%s", s[i] + 1);
	int res = 0;
	fore(i, 1, m) fore(j, 1, n) {
		if (s[i][j] != s[i][j - 1]) cnt[i][j].first = cnt[i][j - 1].first + 1; else cnt[i][j].first = 1;
		if (cnt[i][j].first >= 32)
			if (s[i][j] != s[i - 1][j]) cnt[i][j].second = cnt[i - 1][j].second + 1; else cnt[i][j].second = 1;
		else cnt[i][j].second = 0;
		if (s[i][j] == '_' && cnt[i][j].first >= 32 && cnt[i][j].second >= 32) res++;
	}
	cout << res;
}
