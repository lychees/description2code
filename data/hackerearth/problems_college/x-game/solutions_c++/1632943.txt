#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
const int mod = 1e9+7;
ll power(ll a,ll b){
 ll res = 1;
 while(b){
  if(b&1) res = (res*a)%mod;
  a = (a*a)%mod;
  b >>= 1;
 }
 return res;
}
int main(){
 int t,n;
 ll x;
 scanf("%d",&t);
 while(t--){
  scanf("%d",&n);
  ll ans = 0;
  for(int i=0;i<n;i++){
    scanf("%lld",&x);
    ans |= x;
  }
  ll res = (ans*power(2,n-1))%mod;
  printf("%lld\n",res);
 }
 return 0;
}
