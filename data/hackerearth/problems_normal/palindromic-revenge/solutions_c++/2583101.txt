#include<bits/stdc++.h>
using namespace std;
int main()
{
    int i,l,t,j,c;
    char s[100001];
    cin>>t;
    while(t--)
    {
        cin>>s;
        l=strlen(s);
        c=0;
        for(i=0,j=l-1;i<j;i++,j--)
        {
            if(s[i]!=s[j])
            {
                if(s[i]>s[j])
                {
                    c+=s[i]-s[j];
                }
                else
                {
                    c+=s[j]-s[i];
                }
            }
        }
        printf("%d\n",c);
    }
    return 0;
}
