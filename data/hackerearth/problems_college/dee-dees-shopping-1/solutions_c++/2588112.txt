#include <cstdlib>
#include <iostream>

using namespace std;
long I,T,i,j,n,k;
int main(int argc, char *argv[])
{   cin>>T;
    for(I=1;I<=T;I++)
    {   cin>>n;
        k=0;
        for(i=0;i<=100;i++)
           for(j=0;j<=100;j++)
              if(6*i+8*j==n)
              {   k=1;
                  goto mm;
              }
        mm:
        if(k==1)
            cout<<(i+j)<<endl;
        else
            cout<<(-1)<<endl;
                     
    }
    
    return EXIT_SUCCESS;
}
