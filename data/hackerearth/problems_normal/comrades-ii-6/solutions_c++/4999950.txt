#include <iostream>
#include<vector>
#include<algorithm>
#include<stdio.h>
using namespace std;
int a[100005];
vector<int> v[100005];
long long int hs;

int dfs(int cur)
{
	int i,c=0;
	
	if(v[cur].size()==0)
	{
		return 1;
	}
	
	for(i=0;i<v[cur].size();i++)
	{
		c+=dfs(v[cur][i]);
	}

//	printf("%d --- %d\n",cur,c);
	
	if(cur!=0)
		hs+=c;
	
	return c+1;
}

int main()
{
	int t,n,i,j;
	long long int ans,x;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		for(i=0;i<=n;i++)
		{
			a[i]=0;
			v[i].clear();
		}
		
		ans=n;
		ans=(ans*(ans-1))/2;
//		printf("here\n");
		
		for(i=0;i<n;i++)
		{
			scanf("%lld",&x);
			v[x].push_back(i+1);
		}
		// for(i=0;i<=n;i++)
		// {
		// 	printf("%d  --->> ",i);
		// 	for(j=0;j<v[i].size();j++)
		// 		printf("%d ",v[i][j]);
		// 	printf("\n");
		// }
		
		hs=0;
		dfs(0);

		printf("%lld %lld\n",hs,ans-hs);
	}

    return 0;
}
