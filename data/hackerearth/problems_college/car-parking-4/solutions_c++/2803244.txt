
#include<iostream>
using namespace std;

	 int freeParks(string street){
        int parkingSpace = 0;
        for(int i = 0; i < street.size(); i++){
            if(street[i] == 'D') continue;
            if(street[i]== 'B') continue;
            if(street[i] == 'S') continue;
            if(i < street.size()-1 && street[i+1] == 'B') continue;
            if(i < street.size()-2 && street[i+2] == 'B') continue;
            if(i < street.size()-1 && street[i+1] == 'S') continue;
                        if(i > 0 && street[i-1] == 'S') continue;
            parkingSpace++;
        }
        return parkingSpace;
    }
    int main( ) {
       int t;
       string str;
cin>>t;
       
       while(t!=0)
       {
       	cin>>str;
       	int d = freeParks(str);
       	cout<<d<<endl;
       	t--;
       }
       
       }
