#include <bits/stdc++.h>
//#include<string.h>
using namespace std;

int main()
{
	string b,a;
	int k;
	cin >>b;
	cin >>k;
	int m = b.length();
	int i,j,c = 0; 
	int t;
	cin >>t;
	
	while(t--) {
		cin >> a;
		int n = a.length();
		j=0;
		for(i=0;i<m && j<n;i++) {
			if(a[j]==b[i]){
				j++;
			}
		}
		if(j==n)
			c++;
		}
		cout<<c<<" ";
		if(c>=k){
			cout<<"LUCKY";
		}
		else
			cout<<"UNLUCKY";
    //cout << "Hello World!" << endl;
    return 0;
}
