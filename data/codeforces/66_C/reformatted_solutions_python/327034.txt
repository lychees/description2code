#!/usr/bin/python
import math
import random
import re
import sys
u = dict([])
v = dict([])
while True:
    s = sys.stdin.readline().strip()
    if s == '':
        break
    t = s.split('\\')
    for k in range(2, len(t) - 1):
        r = '\\'.join(t[:k])
        if r not in u:
            u[r] = []
        for l in range(k + 1, len(t)):
            u[r] += ['\\'.join(t[k:l])]
    for k in range(2, len(t)):
        r = '\\'.join(t[:k])
        if r not in v:
            v[r] = []
        v[r] += ['\\'.join(t[k:])]
# print u
# print v
mu, mv = 0, 0
for key, value in u.iteritems():
    mu = max(mu, len(set(value)))
for key, value in v.iteritems():
    mv = max(mv, len(set(value)))
print mu, mv
