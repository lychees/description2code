#include <iostream>
using namespace std;

int main()
{
    long long ntest;
    cin>>ntest;
    while(ntest--){
    	long long n;
    	cin>>n;
    	long long arr[26];
    	for(int i=0; i<26; i++)
    		cin>>arr[i];
    	long long p = 12*n;
    	long long l = p%26;
    	long long t = p/26;
    	long long ans = 0;
    	for(int i=0; i<26; i++){
    		ans += t*arr[i];
    	}
    	ans += t*(t-1)*13;
    	for(int i=0; i<l; i++){
    		ans += arr[i];
    	}
    	ans += l*t;
    	cout<<ans<<endl;
    }
    return 0;
}
