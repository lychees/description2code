#include<bits/stdc++.h>

using namespace std;
bool x[100];
long long dp[2][20];
long long f ( int fl, int len )
{
	if ( len == 0 )
	{
		if ( fl == 1 )
		return 1;
		else
		return 0;
	}
	long long &ans = dp[fl][len];
	if (ans != -1 )
	return ans;
	if ( fl )
	{
		ans = 10*f(1,len-1);
	}
	else
	{
		ans = 4*f(1,len-1);
		ans += 6*f(0,len-1);
	}
	return ans;
}
char s[25];
int main()
    {
    	for ( int i = 0; i < 2; i++ )
    	for ( int j = 0; j < 20; j++ )
    	dp[i][j]=-1;
    int t;
    cin >> t;
    x['2']=true;
	x['3']=true;
	x['5']=true;
	x['7']=true;
    while ( t-- )
    {
        scanf("%s",s);
        
    	//cout << s << endl;
        int l = strlen(s),i,j;
        long long ans = 0;
        bool fl = false;
        for ( j = 0; j < l; j++ )
    	{
        for (  i = '0'; i < s[j]; i++ )
    	{
    		if (fl || x[i] )
    		ans += f(1,l-j-1);
    		else
    		ans += f(0,l-j-1);
    	}
    	if ( x[s[j]] )
    	fl = true;
    	}
    	if( fl )
    	ans++;
    	printf ("%lld\n", ans);
    }
}
