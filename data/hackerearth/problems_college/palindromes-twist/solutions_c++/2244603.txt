#include<bits/stdc++.h>
using namespace std;

int main() {
    int i,j,k,t,m,p,q;
    string a;
    cin >> t;
    while(t--) {
      cin >> a;
      cin >> p >> q;
      k=a.size();
     
      int ans=0;
      for(i=0;i<k/2;i++) {
      	       	if(a[i]!=a[k-i-1] || (a[i]=='/' && a[k-i-1]=='/')) {
      		if(a[i]=='/') {
      			if(a[k-i-1]=='/') {
      				m=min(p,q);
      				ans+=2*m;
      			}else {
      				if(a[k-i-1]=='a')
      					ans+=p;
      				else
      					ans+=q;
      			}
       		}

       		else if(a[k-i-1]=='/') {
       			if(a[i]=='/') {
      				m=min(p,q);
      				ans+=2*m;
      			}else {
      				if(a[i]=='a') {
      					ans+=p;
      					
      				}

      				else
      					ans+=q;
      			}

       		}

       		else {
       			cout << "-1" << endl;
       			break;
       		}
      	}
      
      }

      if(i==k/2)
       	cout << ans << endl;

    }

	return 0;
}
