#include <iostream>
#include <vector>
using namespace std;

vector<long long>& operator / (vector<long long>&, long long&);

int main()
{
	int N, Q;
    long long temp, D=1;
    vector<long long> A;
    
    cin >> N;
    for (int i=0; i<N; i++)
    {
        cin >> temp;
        A.push_back(temp);
    }
    
    cin >> Q;
    for (int i=0; i<Q; i++)
    {
        cin >> temp;
        D *= temp;
    }
    
    A = A / D;

    for (int i=0; i<N; i++)
    {
        cout << A[i] << " ";
    }
    cout << endl;
    
    return 0;
}

vector<long long>& operator / (vector<long long>& V, long long& D)
{
	if (D != 0)
	{
    	for (vector<long long>::iterator iter=V.begin(); iter != V.end(); iter++)
    	{
        	*iter = *iter / D;
    	}
	}
	else
	{
		for (vector<long long>::iterator iter=V.begin(); iter != V.end(); iter++)
    	{
        	*iter = 0;
    	}
	}
    
    return V;
}
