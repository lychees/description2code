#define all(x) x.begin(),x.end()
#define eps 1e-9
ll a,b,c,d,k;
ll eval(ll t)
{
   double T = t;
   double tmp =  a*T*T*T + b*T*T + c*T + d;
   if(tmp + eps > 1e18 )
   return k+1;
   return a*t*t*t + b*t*t + c*t + d;
}
int main()
{
    int t;
    scanf("%d",&t);
    assert(1<=t && t<=10000);
    while(t--)
    {
         scanf("%lld%lld%lld%lld%lld",&a,&b,&c,&d,&k);
         assert(2<=a && a<=(ll)1e18);
         assert(2<=b && b<=(ll)1e18);
         assert(2<=c && c<=(ll)1e18);
         assert(2<=d && d<=(ll)1e18);
         assert(2<=k && k<=(ll)1e18);
         
         ll lo = 0, hi = (ll)1e6 , mid;
         ll ans = 0;
         while(lo<=hi)
         {
             mid = (lo+hi)>>1;
             if(eval(mid)<=k)
             {
                 ans = mid;
                 lo = mid+1;
             }
             else
             {
                 hi = mid-1;
             }
         }         
         assert(ans>=0);
         printf("%lld\n",ans);
    }
    //system("pause");
    return 0;
}
