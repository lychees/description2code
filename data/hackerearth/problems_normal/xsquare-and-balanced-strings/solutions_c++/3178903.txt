
#include <bits/stdc++.h>
//zone
using namespace std;

int main(){
    int T,N,ans;
	cin >> T ;
	string str;
	while(T--){
		cin >> str ;
		N = str.length() ;
		int ans = 0 ;
		for(int i=0;i<N;i++){
			ans = ans ^ (1 << (str[i]-'a')) ;
		}
		puts(ans == 0 ? "1" : "-1") ;
	}
	return 0 ;
}
