#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string st;
    	cin>>st;
    	int len,flag=0,c=0,i,j;
    	len = st.length();
    	if(len>=4)
    	{
    	map<string,int> m;
    	string prev;
    	for(i=0;i<len;i++)
    	{
    		j=1;
    		string s = st.substr(i,j); 
    		if(m[s]==0)
    		{	
    			c++;
    		//	cout<<s<<endl;
    			m[s]++;
    		}
			else
			{
				s= prev;
				s+=st[i];
				while( m[s]&&(++i)<len)
				{
					s+=st[i];
				}
				if(i<len)
				{
					m[prev]=0;
					//cout<<"REM:"<<prev<<endl;
					m[s]=1;
				//	cout<<s<<endl;
				}
				
			
				
			}
    		if(c==4)
    		{
    			flag=1;
    			break;
    		}
    		prev = s;
    	}
    	if(c!=4)
    	{
    		c=0;
    		map<string,int> m;
    	for(i=0;i<len;i+=j)
    	{
    		j=1;
    		if(m[st.substr(i,j)]==0)
    		{	
    			c++;
    			//cout<<st.substr(i,j)<<endl;
    			m[st.substr(i,j)]++;
    		}
    		else
    		{
    			while((i+j)<=len && m[st.substr(i,j)] )
    			{
    				j++;
    			}
    			if(i+j <= len)
    			{
    			//	cout<<st.substr(i,j)<<endl;
    				m[st.substr(i,j)]++;
					c++;
    			}
    		}
    		if(c==4)
    		{
    			flag=1;
    			break;
    		}
    	}
    	
    	}
    	}
    	puts(flag?"YES":"NO");
    }
    return 0;
}
