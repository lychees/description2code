#include<iostream>
#include<algorithm>
using namespace std;

int main()
{
	int t, c1, c2, m, n, ok, ans;
    cin>>t;
    for(int c=0; c<t; c++)
    {
    	ans=100000000;
    	cin>>n>>m;
    	int a[n], s[n];
    	for(int i=0;i<n;i++) cin>>a[i];
    	sort(a,a+n);
    	s[0]=a[0];
    	for(int i=1;i<n;i++) s[i]=s[i-1]+a[i];
    	for(int i=0;i<n;i++)
    	{
    		int b[n], c=0;
    		for(int j=0;j<n;j++)
    		{
    			if(a[j]>a[i]) b[j] = (a[j]-a[i])*5;
    			else b[j]=(a[i]-a[j])*3;
    		}
    		sort(b,b+n);
    		for(int j=0;j<m;j++) c+=b[j];
    		ans = min(ans,c);
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
