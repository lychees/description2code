lld modpow(lld a,lld n,lld temp){lld res=1,y=a;while(n>0){if(n&1)res=(res*y)%temp;y=(y*y)%temp;n/=2;}return res%temp;}

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int i,n;
		cin>>n;
		vi d1(n),f(n);
		vector<pii>p2(n);
		rep(i,n){cin>>d1[i]>>p2[i].first;p2[i].second=i;}
		sort(all(p2));
		rep(i,n)
			f[i]=d1[p2[i].second];
		int ans[1001]={0},res=0;
		rep(i,n){
			int max1=0;
			for(int j=0;j<i;j++)
			{
				if(f[j]<f[i]&&ans[j]>max1)
					max1=ans[j];
			}
			ans[i]=max1+1;
			res=max(res,ans[i]);
		}
		cout<<res<<endl;

	}
	return 0;
}
