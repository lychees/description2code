#include<bits/stdc++.h>
using namespace std;
void sol()
{
    long long int x,y,k,temp,i;
    vector<long long > v;
    cin>>x>>y>>k;
    temp=__gcd(x,y);
    for(i=1;i*i<=temp;i++)
    {
        if(temp%i==0)
           { v.push_back(i);
            if(i*i!=temp) v.push_back(temp/i);
           }
    }
    sort(v.begin(),v.end());
    if(k>v.size()) cout<<"No crymeth today"<<endl;
    else cout<<v[v.size()-k]<<endl;
}
int main()
{
    int d;
    cin>>d;
    while(d--) sol();
    return 0;
}
