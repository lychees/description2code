#include <bits/stdc++.h>
using namespace std;
#define ll long long
bool prime[1000005];
ll sum[1000005];
void primeNumbers()
{
	memset(prime,true,sizeof(prime));
	prime[0] = false;
	prime[1] = false;
	sum[0] = 0;
	sum[1] = 0;
	for( int i = 2 ;i <= 1000 ; i++)
	{
		if ( prime[i] ==  true)
		{
			for ( int j = 2*i ; j<1000001; j += i)
			prime[j] = false;
		}
	}
	for ( int i = 2; i < 1000001; i ++ )
	{
		
		if ( prime[i] == true)
		sum[i] = sum[i-1] + i;
		else
		sum[i] = sum[i-1];
	}
}
int main()
{
    //cout << "Hello World!" << endl;
    primeNumbers();
   
    int t;
    cin >> t;
    while(t -- )
    {
    	int x,y;
    	cin >> x >> y;
    	ll ans = sum[y] - sum[x-1];
    	cout << ans  << endl;
    }
    return 0;
}
