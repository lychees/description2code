from collections import defaultdict as D
s = raw_input()
n = input()
S = list(set(s))
if len(S) > n:
    print -1
    exit()
C = D(int)
for c in s:
    C[c] += 1
i = 1
while True:
    s = 0
    r = []
    for k, v in C.items():
        c = -(-v/i)
        r += k*c
        s += c
    if s <= n:
        break
    i += 1
print i
print ''.join(r)+'a'*(n-len(r))
