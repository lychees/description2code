#include <iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--) {
    	
    	int n;
    	cin >> n;
    	float ar[n];
    	
    	for(int i=0; i<n;i++){
    		cin>>ar[i];
    	}
    	int i=0;
    	int j=n-1;
    	int ctri=0;
    	int ctrj=1;
    	float x;
    	int flag=0;
    	if(n==1) {
    		ctri=1;
    		ctrj=0;
    	} else {
			while(i<=j) {
				x=ar[i]/2;
				ctri++;
				while(x>0) {
					if(i==j) {
						j--;
						break;
					} else if(ar[j] > x){
						ar[j]=ar[j]-x;
						x=0;
						if (flag==1) {
							ctrj++;
							flag=0;
						}
						
					} else if(ar[j] == x) {
						ar[j]=0;
						j--;
						if (flag==1) {
							ctrj++;
							flag=0;
						}
						flag=1;
						x=0;
					}
					else {
						x-=ar[j];
						ar[j]=0;
						j--;
						if (flag==1) {
							ctrj++;
							flag=0;
						}
						flag=1;
					} 
				}
				i++;
				if(i==j){
					break;
				}
			}
    	}
    	cout<<ctri<<" "<<ctrj<<endl;
    }
    return 0;
}
