#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;
int main()
{
    string s;
    vector<string> v;
    int n,c=1,ans=0;
    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>s;
        sort(s.begin(),s.end());
        v.push_back(s);
    }
    sort(v.begin(),v.end());
    string t=v[0];
    for(int i=1;i<n;i++)
    {
        if(t==v[i])c++;
        else c=1;
        t=v[i];
        if(c>ans)ans=c;
    }
    cout<<ans<<endl;
    return 0;
}
