#include<bits/stdc++.h>

#define ll long long
int main()
{
	ll n,k;
	cin>>n>>k;ll f, i,j, a[28];
	f=n;
	for(i=1;i<=26;i++) a[i]=0;
	while(n--)
	{
		string s;
		cin>>s;
		set<int> se;
		set<int>::iterator it;
		for(i=0;i<s.length();i++)se.insert(s[i]-96);
		for(it=se.begin();it!=se.end();it++)
		{
			a[*it]++;
		}
	}
	cout<<endl;

	for(i=1;i<=26;i++)if(a[i]==f)sum++;

}
