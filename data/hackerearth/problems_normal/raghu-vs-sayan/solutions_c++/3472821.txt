

    #include <cstdio>
    #include <vector>
    #include <algorithm>
    #include <cassert>
    using namespace std;
    #define MAXN 10000
    #define MCAL 100000
    #define MAXL 1000000000
    vector<int> dish;
    int main(){
    	int T, N, A, B, cal;
    	scanf("%d", &T);
    	assert(T>0 and T<=100);
    	while(T--){
    		dish.clear();
    		scanf("%d %d %d", &A, &B, &N);
    		assert(N>0 and N<=MAXN);
    		assert(A>0 and A<=MAXL);
    		assert(B>0 and B<=MAXL);
    		for(int i=0;i<N;i++){
    			scanf("%d", &cal);
    			assert(cal>0 and cal<=MCAL);
    			dish.push_back(cal);
    		}
    		sort(dish.begin(), dish.begin() + N);
    		int cntA = 0, cntB = 0;
    		for(int i=0;i<N;i++){
    			if(A>=dish[i]){
    				A-=dish[i];
    				cntA++;
    			}
    			if(B>=dish[i]){
    				B-=dish[i];
    				cntB++;
    			}
    		}
    		if(cntA>cntB){
    			printf("Raghu Won\n");
    		}else if(cntA<cntB)
    			printf("Sayan Won\n");
    		else
    			printf("Tie\n");
    	}
    	return 0;
    }
