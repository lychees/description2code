#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
	    int i=4;
	    int a[i];
	    a[0]=2;
	    a[1]=3;
	    a[2]=5;
	    a[3]=7;
		int sum=0;
		for(i=3;i>=0;i--)
		{
			if(n/a[i]>0)
			{	
		       sum=sum+n/a[i];			
			   n=n-(n/a[i])*a[i];
			   if(n==1)
			   {
			   	sum=sum+1;
			   }
			   if(n<=0)
			   {
			   	break;
			   }
		    }
		}  
		cout<<sum<<"\n";
	}
}
