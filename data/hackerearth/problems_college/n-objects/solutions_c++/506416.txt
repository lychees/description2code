const LL mod = 1000000007;
LL binom[1000][1000], F[1000];

void prepro()
{
	int N = 500;
	FOR(i, 0, 500)
	{
		binom[i][0] = 1;
		binom[i][i] = 1;
		FOR(j, 1, i - 1)
		{
			binom[i][j] = (binom[i - 1][j] + binom[i - 1][j - 1]) % mod;
		}
	}
	
	F[0] = 1;
	FOR(n, 1, N)
	{
		FOR(k, 1, n)
		{
			F[n] = (F[n] + binom[n - 1][k - 1] * F[n - k]) % mod;
		}
	}
}

	cin >> n;
	cout << F[n] << "\n";
}
