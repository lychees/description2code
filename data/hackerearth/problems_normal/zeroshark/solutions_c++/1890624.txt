#include <bits/stdc++.h>
using namespace std;

#define ll long long int
#define sc(n) scanf("%lld",&n)
#define mod 1000000007

int main()
{
	ll n,t,a[100005],b[100005],i;
	a[1]=0;
	a[2]=1;

	for(i=3;i<=100000;i++)
	{
		a[i]=((a[i-1]%mod)+(a[i-2]%mod))%mod;
	}
	b[1]=0;
	b[2]=1;
	for(i=3;i<=100000;i++)
	{
		b[i]=((b[i-1]%mod)+(b[i-2]%mod)+a[i])%mod;
	}

	sc(t);
	while(t--)
	{
		sc(n);
		printf("%lld\n",b[n]);		
	}
	return 0;
}
