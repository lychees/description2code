#include <iostream>
#include <math.h>

using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    
    int test;
    cin >> test;
    
    while(test > 0)
    {
    	long long int n;
    	cin >> n;
    	
    	if(n == 1)
    		cout << 1 << endl;
    	else
    	{
    		if(n == 2)
    			cout << 8 << endl;
    		else
    		{
    			long long int dirty = 2*n*n + n *(n-2)*2  + 2*n*n-8*n+8;
    			cout << dirty << endl;
    		}
    	}
    	test--;
    }
    return 0;
}
