import os


def file_text_stats(filename):
    num_lines = 0
    num_words = 0
    num_chars = 0

    with open(filename, 'r') as f:
        for line in f:
            words = line.split()

            num_lines += 1
            num_words += len(words)
            num_chars += len(line)

    st = os.stat(filename)

    return num_lines, num_words, num_chars, st[4], st[5], st[8]


if __name__ == "__main__":
    filename = raw_input()
    result = file_text_stats(filename)

    result_len = len(result)
    for i in range(result_len):
        print result[i]
