#include <iostream>
#include <bits/stdc++.h>
using namespace std;
long long int lis(long long int arr[],long long int n )
{
    long long int *lis, i, j, max = 0;
    lis = (long long int*) malloc ( sizeof( int ) * n );
 
    for (i = 0; i < n; i++ )
        lis[i] = 1;
 
    for (i = 1; i < n; i++ )
    for (j = 0; j < i; j++ )
    if(abs(arr[i]) > abs(arr[j]) && ((arr[i]<0 && arr[j]>0) || (arr[i]>0 && arr[j]<0)) && (lis[i] < (lis[j] + 1)))
        lis[i] = lis[j] + 1;

    for (i = 0; i < n; i++ ){
        if (max < lis[i])
            max = lis[i];
    }
    return max;
}
int main()
{
	long long int n,a[100005],i;
	cin>>n;
	for(i=0;i<n;++i)
	cin>>a[i];
	cout<<lis(a,n);
    //cout << "Hello World!" << endl;
    return 0;
}
