#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main (){ 

int day;
cin>>day;
while(day--){ 
	ll gr,max;
	int i=0;
	cin>>gr>>max;
	vector<ll>ar;
	for(i=0;i<gr;i++)
	{ 
		ll x;
		cin>>x;
		ar.push_back(x);
		
		
	}
	sort(ar.begin(),ar.end());
	i=0;
	ll sum=0,count=0;
	while(count<=gr && sum<=max)
	{ 
		sum+=ar[i];
		i++;count++;
	}
	
	cout<<count-1<<"\n";
	}
	
	
	
return 0;
}
