#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	int n;
	while(t--) {
		cin>>n;
		int array[n];
		int x = 0;
		for(int i=0;i<n;i++) {
			cin>>array[i];
			x = x^array[i];
		}
		int ans = 0;
		for(int i=0;i<n;i++) {
			if(x^array[i] == 0) ans++;
		}
		cout<<ans<<endl;
	}
    return 0;
}
