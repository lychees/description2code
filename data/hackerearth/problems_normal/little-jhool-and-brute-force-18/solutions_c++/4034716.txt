    #include<bits/stdc++.h>
    using namespace std;
    int main(void)
    {
    int t;
    cin>>t;
    int dp[101]={0};
    for(int i=0;i<=100;i++)
    dp[i]=(i*i*i);
    int* QW=new int[(int)1e6];
    for(int i=2;i<=100;i++)
    {
    for(int j=1;j<i;j++)
    {
    long long z=dp[i]+dp[j];
    if(z<((int)1e6))
    QW[z]++;
    }
    }
    while(t--)
    {
    int n;
    cin>>n;
    int res=-1;
    for(int i=0;i<=n;i++)
    {
    if(QW[i]>=2)
    res=i;
    }
    cout<<res<<endl;
    }return 0;
    }
