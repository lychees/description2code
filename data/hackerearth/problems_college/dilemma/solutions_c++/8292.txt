#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
    int t,N,A,B,C;
    long long answer = 0;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d %d %d %d",&N,&A,&B,&C);
        for(int a=0;a<=A;++a)
        {
            for(int b=0;b<=B;++b)
            {
                int c = N-b-a;
                if(c>=0)
                answer += min(c,C)+1;
            }
        }
        printf("%lld\n",answer);
        answer = 0;
    }
    return 0;
}
