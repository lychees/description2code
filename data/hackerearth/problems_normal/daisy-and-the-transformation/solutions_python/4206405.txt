
n = int(raw_input().strip())

for i in range(0, n):
    s=[]
    s = (raw_input().strip()).split()
    count = 0

    for j in range(0, len(s[0])):
        if s[0][j] != s[1][j]:
            count +=1
        else:
            continue

    #print count
    if count <= int(s[2]) and (int(s[2]) - count)%2 == 0:
        print "Yes"
    else:
        print "No"
