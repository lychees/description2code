#include <bits/stdc++.h>
using namespace std;

const int MAX = 1004;
const int MOD = 1000000007;

int main()
{
    int t;
    scanf("%d", &t);
    vector <vector <int> > d(MAX, vector <int> (MAX, 1));
    for (int i = 1; i < MAX; i++) {
        for (int j = 1; j < MAX; j++) {
            d[i][j] = ((long long)d[i-1][j] + d[i][j-1] + d[i-1][j-1]) % MOD;
        }
    }

    while (t--) {
        int x[2], y[2];
        scanf("%d %d %d %d", &x[0], &y[0], &x[1], &y[1]);
        printf("%d\n", d[x[1] - x[0]][y[1] - y[0]]);
    }

    return 0;
}
