#include <iostream>
using namespace std;

int main() {
long long int t,i,j,k,n;
cin>>t;
while(t--)
{
	cin>>n;
	int g=0;
	for(i=n;i>=0;i--)
	{
		j=n-i;
		if(j%5==0 && i%3==0)
		{
			for(k=0;k<i;k++)
			cout<<"5";
			for(k=0;k<j;k++)
			cout<<"3";
			cout<<"\n";
			g=1;
		}
		
	}
	
	if(g==0)
	cout<<"-1\n";
	
	
}
}
