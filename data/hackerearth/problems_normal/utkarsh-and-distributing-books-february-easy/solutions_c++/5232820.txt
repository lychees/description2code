#include<iostream>

using namespace std;



int main()
{
    int test;
    cin>>test;
    while(test--)
    {
        int n;
        cin>>n;
        int arr[n];
        int count=0;
        int min=1000;
        for(int i =0;i<n;i++)
        {
            cin>>arr[i];
            if(arr[i]<=min)
            {
                min=arr[i];
            }
        }

        cout<<min-1<<" ";
        for(int i=0;i<n;i++)
        {
            count=count + arr[i]-1;
        }

        cout<<count<<endl;
    }
}
