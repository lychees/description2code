def fixer(n):
    m = str(n)
    j = -9
    try:
        while m[j] == '0':
            j -= 1
    except IndexError:
        return n
    return m[j:]


for _ in xrange(int(raw_input())):
    a, b, k = map(int, raw_input().split())
    a1 = fixer(a)
    b1 = fixer(b)
    value = pow(int(a1), int(b1), 10**k)
    print sum(map(int, `value`))
