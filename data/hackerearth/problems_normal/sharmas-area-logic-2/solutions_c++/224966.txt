#include <bits/stdc++.h>
using namespace std;

#define sld(x) scanf("%lld", &x)
#define sd(x) scanf("%d", &x)
#define LL long long int
#define MAXN 1000
#define INF 1000000007
#define MOD 1000000007

inline void Solve(){
    int n, i, x, y;
    sd(n);
    int  max_x = -1 * INF, max_y = -1 * INF, min_x = INF, min_y = INF;
    for(i = 0; i < n; i++){
        sd(x); sd(y);
        if(x > max_x){
            max_x = x;
        }
        if(y > max_y){
            max_y = y;
        }
        if(x < min_x){
            min_x = x;
        }
        if(y < min_y){
            min_y = y;
        }
    }
    LL a, b;
    a = max_x - min_x;
    b = max_y - min_y;
    if(a == 0){
        printf("%lld\n", b);
        return;
    }
    if(b == 0){
        printf("%lld\n", a);
        return;
    }
    printf("%lld\n", (a * b));
    return;
}

int main(){
    int t;
    cin>>t;
    while(t--){
        Solve();
    }
    return 0;
}
