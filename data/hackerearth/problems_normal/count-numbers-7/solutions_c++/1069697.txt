#include <iostream>
#include<vector>
#include<algorithm>
#include<string.h>
using namespace std;

typedef long long LL ;
LL pt[11] ;

int cntbit(int x){
	int cnt = 0 ;
	while(x){
		
		x=x&(x-1) ;
		++cnt ;
	}
	return cnt ;
}
int sump[11] ;
LL  find_sol(int k,int n)
{

	for(int i=1;i<(1<<k);i++){
		int x = cntbit(i) ;
		LL prod = 1;
		for(int j=0;j<k;j++ ){
			 
			if(i&(1<<j)){
				prod*=pt[j] ;
				if(prod>n)
					break ;
			}
		}
		sump[x]+=(n/prod) ;
	} 
	LL ans = 0 ;
	for(int i=1;i<=k;i++){
		if(i&1){
			ans += sump[i] ; 
		}else{
			ans-=sump[i] ;
		}
	}
	return ans ;
}

LL read_inp(){
	
	int k,t ;
	cin>>k>>t ;
	for(int i=0;i<k;i++)
	{
		cin>>pt[i] ;
	}
	LL r1 ;
	for(int i=1;i<=t;i++)
	{
		int a,b ;
		
		cin>>a>>b ;
		memset(sump,0,sizeof(sump)) ;
		r1 = find_sol(k, b) ;
		memset(sump,0,sizeof(sump)) ;
		LL r2 = find_sol(k, a-1) ;
		r1 = r1-r2 ;
		cout<<r1<<endl ;
	}
	
}


int main(){
	read_inp()  ;
}
