#include<iostream>
#include<math.h>
using namespace std;
 
bool isPrime(int t){
	int root = sqrt(t);
//	cout << root << "*"<<endl;
	if(t==1)
		return false;
	else if(t==2)
		return true;
	else if(t%2==0)
		return false;
	else if(t==3)
		return true;
	else if(t==5)
		return true;
	else if(t==7)
		return true;
	else if(t==9)
		return true;
	else {
		for(int i=3 ; i<=root ; i=i+2){
			if(t%i == 0){
				return false;
			}
		}
		return true;
	}
}
bool isPalindrome(int n){
	int x,y,i;
	x=n;
	i=0;
	y=0;
	while(x>0){
//		cout << n << " "<<y << ": " << pow(10,i) << ":: "<< x <<endl;
		y=y*10+x%10;
		x=x/10;
		i++;
//		cout << n << " "<<y << ": " << pow(10,i) << ":: "<< x <<endl;
 
	}
	if(y==n)
		return true;
	return false;
}
 
int main(){
	int t;
	int n;
	cin>>n;
	while(n--){
	cin >>t;
	int i=t;
	while(!isPrime(i) || !isPalindrome(i)){
		i++;
 
//		cout <<i <<" : " <<isPrime(i) << " "<< isPalindrome(i) << endl;
		
	}
	
	
	cout << i << endl; }
	
}
