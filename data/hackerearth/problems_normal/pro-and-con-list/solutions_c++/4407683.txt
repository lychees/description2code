#include<bits/stdc++.h>
using namespace std;
vector<long long int>ans;
int main()
	{
		int t;
		cin>>t;
		while(t--)
			{
				ans.clear();
				long long int n;
				scanf("%lld",&n);
				long long int a,b,sum;
				sum=0;
				for(int i=0;i<n;i++)
					{
						scanf("%lld %lld",&a,&b);
						a=a+b;
						ans.push_back(a);
						sum=sum+b;
					}
				sort(ans.rbegin(),ans.rend());
				cout<<ans[0]+ans[1]-sum<<endl;
			}
		return 0;
	}
