#include <stdio.h>
#include <bits/stdc++.h>

long long int max (long long int i, long long int j) {
	if (i > j) {
		return i;
	}
	return j;
}

int main ()
{
	long long int t, i, n, x, y, ans, j;
	
	scanf ("%lld", &t);
	
	while (t--) {
		scanf ("%lld", & n);
		ans = 0;
		for (i = 1; i <= sqrt (n); i++) {
			x = n / i;
			j = sqrt (n);
			if (i == j) {
				y = sqrt (n);
			} else
			y = n / (i + 1);
		//	printf ("%lld ", y);
			ans = ans + ((x * (x + 1) / 2) - (y * (y + 1) / 2)) * i;
			ans = ans % 1000000007;
			
			//break;
			ans = ans + (n / i) * i;
			ans = ans % 1000000007;
		}
		if (sqrt (n) * sqrt (n) == n) {
			//ans = ans - n;
		}
		printf ("%lld\n", ans);
	}
}
