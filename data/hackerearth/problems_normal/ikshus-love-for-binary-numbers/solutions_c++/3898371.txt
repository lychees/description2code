#include <bits/stdc++.h>
using namespace std;

long long int a[64] = {0};

void fun(int k){
	a[2] = a[1] = 1;
	for(int i=3;i<65;i++){
		for(int j=i-1;j>i-1-k && j >= 0;j--){
			a[i] += a[j];
		}
	}
}

int main()
{
    int n,k;
    cin>>n>>k;
    fun(k);
    long long int g = __gcd((1L << n) - a[n+2],(long long int)(1<< n));
    cout<<((1L<<n) - a[n+2])/g<<"/"<<(1L<<n)/g;
    return 0;
}
