#define ll long long int
#define maxN 3000
#define maxVal 100000000

#define gcd(a,b) __gcd(a,b)

using namespace std;

struct data
{
	long double x,y;
	long double f;
};
int n;
data a[maxN+5];
long double dp[maxN+5];

long double dist(data p,data q)
{
	return sqrt((long double)((p.x-q.x)*(p.x-q.x)+(p.y-q.y)*(p.y-q.y)));
}

int main()
{
    int i,j;
    cin>>n;
    for(i=1;i<=n;i++)
    	cin>>a[i].x>>a[i].y>>a[i].f;
    fill(dp,dp+n+2,-1ll<<60);
    dp[1]=a[1].f;
    for(i=2;i<=n;i++)
    	for(j=1;j<i;j++)
    	{
    		//cout<<dp[i]<<" "<<dp[j]<<"\n";
    		dp[i]=max(dp[i],a[i].f+dp[j]-dist(a[j],a[i]));
		}
	cout<<fixed;
	cout.precision(6);
	cout<<dp[n]<<"\n";
    return 0;
}
