#include <iostream>
#include <algorithm>

using namespace std;

#define DEBUG(x) cout << "> " << #x << " : "<< x << endl
#define REP(i,n) for (int i = 0; i < (n); i++)
typedef long long ll;

int main() {
	int T;
	ll N[2];
	ll count;
	ll quot;
	cin >> T;
	REP(i, T) {
		cin >> N[0] >> N[1];
		count = 0;
		while(N[0] > 1 || N[1] > 1) {
			if (N[0] == 1) {
				count += (N[1] - 1);
				break;
			} else if (N[1] == 1) {
				count += (N[0] - 1);
				break;
			} else if (N[0] > N[1]) {
				quot = N[0] / N[1];
				N[0] -= quot * N[1];
				count += quot;
			} else {
				quot = N[1] / N[0];
				N[1] -= quot * N[0];
				count += quot;
			}
		}
		cout << count << endl;
	}
    return 0;
}
