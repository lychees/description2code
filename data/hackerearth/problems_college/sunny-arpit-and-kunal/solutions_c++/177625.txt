#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int a,b,c;
        scanf("%d%d%d",&a,&b,&c);
        int ans = 0;
        int arr[3];
        arr[0] = a;
        arr[1] = b;
        arr[2] = c;
        sort(arr,arr+3);
        a = arr[0];
        b = arr[1];
        c = arr[2];
        int diff1 = b-a;
        int diff2 = c-b;
        ans = max(diff1,diff2) - 1;
        printf("%d\n",ans);
    }
    return 0;
}
