#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		char str[1000];
		scanf("%s",str);
		int n=strlen(str);
		int a[n+5];
		for(int i=0;i<n;i++)
		{
			a[i]=1;
		}
		for(int i=0;i<n;i++)
		{
			if(str[i]=='x')
			{
				for(int j=i+1;j<n;j++)
				{
					if(str[j]=='*'||str[j]=='x')
					break;
					else if(str[j]=='o')
					a[j]=-1;
				}
				for(int j=i-1;j>=0;j--)
				{
					if(str[j]=='*'||str[j]=='x')
					break;
					else if(str[j]=='o')
					a[j]=-1;
				}
			}
		}
		for(int i=0;i<n;i++)
		{
//			cout<<a[i]<<endl;
			if(a[i]==1)
			printf("%c",str[i]);
		}
		printf("\n");
	}
    return 0;
}
