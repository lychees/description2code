#include<iostream>
#include<algorithm>
using namespace std;
int arr[105];
template< class T > T gcd(T a, T b) { return (b != 0 ? gcd<T>(b, a%b) : a); }

int main()
{
    int t, n;
    cin>>t;
    while(t--){
        cin>>n;
        for(int i = 0 ; i < n; i++){
            cin>>arr[i];
        }
        sort(arr, arr+n);

        int g = arr[0];
        for(int i = 1; i < n; i++){
            g = gcd(arr[i], g);
        }

        if(g == 1) cout<<"YES\n";
        else cout<<"NO\n";
    }
    return 0;
}
