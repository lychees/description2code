#include <bits/stdc++.h>
using namespace std;

int calc(double h, double m)
{
   
    if (h == 12) 
    h = 0;
    
    if (m == 60) 
    m = 0;
 
    
    int hangle = 0.5 * (h*60 + m);
    int mangle = 6*m;
 
    
    int ang = abs(hangle - mangle);
 

    ang = min(360-ang, ang);
 
    return ang;
}


int main()
{
	int t,a,b,i,j,k,ax,bx;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b;
    	printf("%d \n", calc(a, b));
    }
    return 0;
}
