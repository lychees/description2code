#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
using namespace std;

inline void scanint(int* x)
{
	register char c = getchar_unlocked();
	*x = 0;
	for(; (c<48)||(c>57);c = getchar_unlocked());
	for(; (c>47)&&(c<58);c = getchar_unlocked())
		*x = (int)((((*x)<<1) + ((*x)<<3)) + c - 48);
}

vector<string> v;
string s,t;

int main() 
{
	int n,i;
	cin>>s;
	scanint(&n);
	while(n--)
	{
		cin>>t;
		
		if(s.length()<= t.length())
		{
			
		for(i=0;i<s.length();i++){if(t[i]!=s[i])break;}	
		
		if(i==s.length())v.push_back(t);
		
		}
	}
	

	if(v.size()==0)cout<<s;
	
	else
	{
		sort(v.begin(),v.end());
		cout<<v[0];
	}
	
	
	return 0;
}
