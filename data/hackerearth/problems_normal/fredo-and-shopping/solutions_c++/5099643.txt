
#include<cstdio>
#include<iostream>
#include<algorithm>
#define MAX 1000001

using namespace std;
int ft[MAX];

void update(int n, int idx, int val)
{
	for(;idx <= n; idx += (idx&-idx))
		ft[idx] += val;
}

int getSum(int idx)
{
	int sum = 0;
	for(;idx; idx -= (idx&-idx))
		sum += ft[idx];
	
	return sum;
}

int main()
{
	int n;
	
	scanf("%d", &n);
	int a[n],b[n];
	for(int i=0; i<n; ++i)
	{
		scanf("%d", &a[i]);
		b[i] = a[i];
	}
	
	sort(b, b+n);
	for(int i=0; i<n; ++i)
		a[i] = lower_bound(b,b+n,a[i])-b + 1;
	
	for(int i=0; i<n; ++i)
	{
		printf("%d ",getSum(a[i]-1));
		update(n,a[i],1);
	}
	
	return 0;
}
