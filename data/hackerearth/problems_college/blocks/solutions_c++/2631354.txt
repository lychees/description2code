lli power(li num,li p)
{
	if(p==0)return 1;
	if(p==1)return num;
	li number=num;
	for(li i=2;i<=p;i++)
	{
		num*=number;
		num%=MOD;
	}
	return num;
}
int main() {	
	li N,M;
	li T[1001],S[1001],P[1001];
	T[0]=T[1]=1;
	T[2]=2;
	T[3]=4;
	T[4]=8;
	P[0]=P[1]=1;
	for(li i=5;i<=1000;i++)
	T[i]=(T[i-1]+T[i-2]+T[i-3]+T[i-4])%MOD;
	
	S[0]=1;
	S[1]=1;
	li sum,tr;
	cin>>tr;
	while(tr--)
	{
		cin>>N>>M;
		for(li i=0;i<=M;i++)
		P[i]=power(T[i],N);
		
		for(li i=2;i<=M;i++)
		{
			sum=0;
			for(li j=1;j<i;j++)
			{
				sum+=(S[j]*P[i-j])%MOD;
				sum%=MOD;
			}
			S[i]=(P[i]-sum);
			S[i]=S[i]%MOD;
		}
		while(S[M]<0)
		S[M]+=MOD;
		
		cout<<S[M]<<endl;
	}
	return 0;
}
