#include<iostream>
#include<algorithm>
using namespace std;
int bsearch(long int *a , int s, int e ,long int k)
{
	if(s<=e){
	int mid= (s+e)/2;
	if(a[mid]==k)
		return 1;
	else if(a[mid]>k)
		return bsearch(a,s,mid-1,k);
	else
		return bsearch(a,mid+1,e,k);
	}
	return 0;
}
int main()
{
	int n;
	long int a[3001],i,k,j;
	cin >>n;
	for(i=0; i<n ; i++)
		cin >> a[i];
	cin >>k;
	sort(a, a+n-1);
	for(i=0; i<n-2 ; i++)
		for(j=i+1; j<n-1; j++)
			if (bsearch(a,j+1,n-1,k-a[i]-a[j])){
				cout <<"YES\n";
				return 0;
			}
	cout <<"NO\n";
	return 0;
}
