#include<iostream>
using namespace std;

int main()
{
int t,n,b,temp,b1;
cin>>t;
while(t--)
{

cin>>n>>b;

 if(n==b)
    cout<<n;

else if(n>b)
{   temp=n/b;
    b1=b*temp;
    temp=b1+b;

 if((temp-n)<=(n-b1))
    cout<<temp;
 else
    cout<<b1;

}


else
{
    if((b-n)<=n)
    cout<<b;
    else
    cout<<0;
}


 cout<<endl;
}
return 0;
}
