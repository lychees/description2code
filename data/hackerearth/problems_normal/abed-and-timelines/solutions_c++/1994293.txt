#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t,n,m,a,b,c,d,p;
	scanf("%d",&t);
	while(t--)
	{
		cin>>n>>m;
		cin>>a>>b;
		cin>>c>>d;
		cin>>p;
		// First traverse minimum cells row-wise and do the same column wise.
		unsigned R=abs(a-c);
		unsigned C=abs(b-d);
		unsigned d=min(R,n-R)+min(C,m-C);
		d=d*p;
		if(d>1000)
		{
			printf("%u\nLet go of the group.\n",d);
		}
		else
		{
			printf("%u\nYou saved the group.\n",d);
		}   
	}
    return 0;
}
