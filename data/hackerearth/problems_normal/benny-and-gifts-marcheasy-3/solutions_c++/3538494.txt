#include<stdio.h>
//#include<conio.h>
#include<string.h>
#include<limits.h>
#include<map>
using namespace std;
int main(){
    char S[100001];
    map< pair<int,int>,int> vis;
    pair<int,int> p;
    int i,x=0,y=0,j,count=0,last=-1;
    scanf("%s",S);
    fflush(stdin);
    int n=strlen(S);
    //printf("%d\n",n);
    p=make_pair(0,0);
    vis[p]=1;
    for(i=0;i<n;i++){
        if(S[i]=='L')
            y--;
        if(S[i]=='R')
            y++;
        if(S[i]=='U')
            x--;
        if(S[i]=='D')
            x++;
        p=make_pair(x,y);
        if(vis[p]==1)   count++;
        vis[p]=1;
        //printf("%d\n",last);
    }
    printf("%d\n",count); 
    //getch();
    return 0;
}
