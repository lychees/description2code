#include <iostream>
#include <string>
using namespace std;

int main()
{
  int n, q;
  long int count1 = 0L, count2 = 0L;
  string s;
  cin >> n;
  cin >> s;
  cin >> q;
  for(int i = 0; i < q; i++) {
    char first_char, second_char;
    cin >> first_char >> second_char;
    count1 = 0L;
    count2 = 0L;
    for(char & c : s) {
        if ( c == first_char ) {
          count1++;
        } else if ( c == second_char ) {
          count2++;
        }
    }
    long int result = 0L;
    if ( first_char == second_char ) {
      result = (count1 * (count1 - 1))/ 2;
    } else {
      result = count1 * count2;
    }
    cout << result << endl;
  }
    return 0;
}
