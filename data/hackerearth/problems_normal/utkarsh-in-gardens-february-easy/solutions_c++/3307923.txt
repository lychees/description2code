#include <bits/stdc++.h>
using namespace std;
#define ll long long int
#define pb push_back
#define pii pair<int,int>
#define maxn 2005
#define mod 1000000007
 
ll n,ans;
bitset<maxn> g[maxn];
int main(){
    cin>>n;
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++){
        	ll x;
        	cin>>x;
            g[i][j]=x;
        }
    }
 
    for(int i=0;i<n;i++){
        for(int j=i+1;j<n;j++){
        	ll cnt=0;
        	cnt=(g[i]&g[j]).count();
        	ans+=cnt*(cnt-1)/2;
        }
    }
    
    
 
    cout<<ans/2;
 
    return 0;
}
