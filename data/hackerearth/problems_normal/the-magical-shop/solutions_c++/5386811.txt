#include <iostream>
using namespace std;

int main()
{
    long long a, b;
    cin >> a >> b;
    string s;
    cin >> s;
    long long ans = 0;
    for (int i = 0; i < s.length(); i++)
    {
    	if (s[i] == '1')
    	{
    		ans += a;
    		ans %= b;
    	}
    	a = (a * a) % b;
    }
    cout << ans << endl;
    return 0;
}
