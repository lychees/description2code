#include <stdio.h>
int main()
{
    int t,n,flag,count;
    long s,p;
    scanf("%d", &t);
    while(t--)
    {
    	flag=count=0;
    	scanf("%d%ld",&n,&s);
    	while(n--)
    	{
    		scanf("%ld",&p);
    		if(p>s)
	    		count++;
    		else if(p==s)
	    		flag=1;
	    	else
	    		continue;
    	}
    	if(flag==1)
    		printf("YES %d\n",count);
    	else
    		printf("NO %d\n",0);
    }
    return 0;
}
