t=int(raw_input())
for i in range(t):
    s=raw_input()
    l=len(s)
    if l%3!=0:
        print("Not OK");
    elif len(set(s))!=3:
        print("Not OK");
    else:
        an=list(s)
        
        if an[0]==an[l//3-1] and an[l//3]==an[l//3*2-1] and an[l//3*2]==an[-1]:
            print("OK");
        else:
            print("Not OK");
