#include<bits/stdc++.h>
using namespace std;

#define lli long long
#define pb push_back
#define mp make_pair
#define gc getchar_unlocked
#define lld "%I64d"
#define DEBUG(x) cout<<">value ("<<#x<<") : "<<x<<endl;
#define mod 1000000007

const int lmt = 1000005;
int in[lmt];
lli freq[lmt];
lli ans[lmt];
lli po[lmt];

void solve(int n){

    po[0] = 1;
    for(int i=1;i<lmt;i++){
        po[i] = po[i-1]*2LL;
        po[i] %= mod;
    }

    for(int i=n;i>=1;i--){
        lli cnt = 0;
        for(int j=i;j<=n;j+=i)
            cnt += freq[j];
        lli res = ((po[cnt]-1)%mod+mod)%mod;
        for(int j=2;j*i<=n;j++)
            res = ((res-ans[j*i])%mod+mod)%mod;
        ans[i] = res;
    }
}

int main(){
    int n,x;
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%d",&in[i]);
        freq[in[i]]++;
    }
    solve(n);
    for(int i=0;i<n;i++)
        printf("%lld ",ans[in[i]]);
    printf("\n");
    return 0;
}
