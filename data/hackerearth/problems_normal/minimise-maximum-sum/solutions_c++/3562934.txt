#include <iostream>
#include <algorithm>
#include <cstdio>
#include <climits>
using namespace std;

int arr[200005];
int main(){
    int T,N,maximum,value;
    scanf("%d",&T);
    while(T--){
        scanf("%d",&N);
         for(int i=0;i<2*N;i++){
            scanf("%d",&arr[i]);
        }
        sort(arr,arr+2*N);
        maximum=0;
        int i;
        int j;
        for(i=0,j=2*N-1;i<j;i++,j--){
            value=arr[i]+arr[j];
            maximum=max(maximum,value);
        }
       printf("%d\n",maximum);
    }
}
