#include <iostream>
#include<stdio.h>
#include<string.h>
using namespace std;

string reverse(char s[])
{
	string p="";
	long long int i;
	
	for( i=0;s[i]!='\0';i++);
	
	for(int j=i-1;j>=0;j--)
	{
		p+=s[j];
	}
	return p;
}
int main()
{
    long long int t,sum=0,l,f,i;
    char s[110];
    
    cin>>t;
    
    while(t--)
    {
    	sum=0;
    	
    	scanf(" %[^\n]s",s);
    	
    	l=strlen(s);
    	for(i=0;i<l;i++)
    	{
   // 		cout<<sum;
    	//	if(s[i]!=' ')
    		sum+=(int)s[i];
    	}
  
    	f=sum/l;
    //	cout<<sum<<" "<<f;	
    	if(f%2==0)
    	{
    		cout<<s<<endl;
    	}
    	else
    	{
    		cout<<reverse(s)<<endl;
    	}
    }
    return 0;
}
