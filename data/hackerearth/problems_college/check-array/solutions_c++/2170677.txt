#include"bits/stdc++.h"
using namespace std;
int main()
{
	int t;cin>>t;
	while(t--)
	{
		int i,n;cin>>n;
		int arr[n];
		cin>>arr[0];
		long long int ans[n];
		ans[0]=arr[0];
		for(i=1;i<n;i++)
		{
			cin>>arr[i];
			ans[i]=ans[i-1]+arr[i];
			
		}
		for(i=1;i<n-1;i++)
		{
			long long int sum1=ans[i]-arr[i],sum2=ans[n-1]-ans[i];
			if(sum1==sum2)
			{
				cout<<"YES"<<endl;
				break;
			}
		}
		if(i==(n-1)) cout<<"NO"<<endl;
	}
	return 0;
}
