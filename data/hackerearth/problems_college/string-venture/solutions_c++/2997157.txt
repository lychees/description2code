#include <bits/stdc++.h>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<cstdio>
#include<string>
using namespace std;
int arr[256];
int main()
{
	string str;
	getline(cin,str);
    int sum=0,i=0;
    int n=str.length();
    memset(arr,0,sizeof(arr));
    for(i=0;i<n;i++)
    {
    	sum+=int(str[i]);
    	arr[str[i]]++;
    }
    
    int weight=floor(sum/n);
    int max=arr[0],index=0;
    for(i=0;i<256;i++)
    {
    	if(max<arr[i])
    	{
    		max=arr[i];
    		index=i;
    	}
    }
    if(weight%2!=0)
    {
    	printf("%c\n",index);
    }
    else
    {
    
    	for(i=n-1;i>=0;i--)
    	{
    		printf("%c",str[i]);
    	}
    	printf("\n");
    }
    return 0;
}
