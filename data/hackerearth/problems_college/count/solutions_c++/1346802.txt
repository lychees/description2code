#include <iostream>
using namespace std;

int main()
{
    long long int test,r,c,i;
    cin>>test;
    while(test--)
    {
    	long long int odd=0,even=0;
    	cin>>r>>c;
    	for(i=0;i<r;i++)
    	{
    		if(i%2==0)
    		{
    			if((i+1)<c)
    			odd+=i+1;
    			else
    			odd+=c;
    		}
    		else
    		{
    			if((i+1)<c)
    			even+=i+1;
    			else
    			even+=c;
    		}
    	}
    	cout<<even<<" "<<odd<<endl;
    }
    
    return 0;
}
