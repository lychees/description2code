#july -easy Sherlock XOR

import sys
line = sys.stdin.readline
T = int(line())
for _ in range(T):
	N = input()
	S = [int(x) for x in line().split()]
	even = []
	odd = []
	for x in S:
		if x%2 == 0:
			even.append(x)
		else:
			odd.append(x)
	el = len(even)
	ol = len(odd)
	e = 0
	o = 0
	pair = 0
	while True:
		if e == el or o == ol:
			break
		elif even[e] < odd[o]:
			pair = pair + ol-o
			e = e + 1
		else:
			pair = pair + el -e
			o = o + 1
	print pair
