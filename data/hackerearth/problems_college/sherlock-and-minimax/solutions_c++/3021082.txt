#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int a[110];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int n,p,q,m,d,dp=LONG_MAX,dq=LONG_MAX;
	cin>>n;
	ll int i,j;
	for(i=0;i<n;i++)
	cin>>a[i];
	sort(a,a+n);
	cin>>p>>q;
	ll int ans=LONG_MIN,v=LONG_MIN;
	for(i=1;i<n;i++)
	{
		m=(a[i]+a[i-1])>>1;
		if(m>=p && m<=q)
		{
			d=min(a[i]-m,m-a[i-1]);
			if(d>ans)
			{
				ans=d;
				v=m;
			}
		}
	}
	//if chosen point is p
	for(i=0;i<n;i++)
	{
		dp=min(dp,abs(a[i]-p));dq=min(dq,abs(a[i]-q));
	}
	if(dp>ans)
	{
		ans=dp;
		v=p;
	}
	else if(dp==ans)
	{
		v=min(v,p);
	}
	if(dq>ans)
	{
		ans=dq;
		v=q;
	}
	else if(dq==ans)
	{
		v=min(v,q);
	}
	cout<<v<<endl;
	return 0;
}
