import fileinput

'''
# Read input from stdin and provide input before running code

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
to_return = ""

for s in fileinput.input():
	for i in list(s):
		if(i.istitle()):
			to_return = to_return + i.lower()
		else:
			to_return = to_return + i.upper()
		
	print(to_return)
