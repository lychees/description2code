#include<bits/stdc++.h>
using namespace std;

int main(){
	int t;cin>>t;
	long double ans=1;
	for(int i=1; i<=t; i++){
		int n, r;
		long double sum=0, sumsq=0;
		cin>>n>>r;
		for(int j=1; j<=n; j++){
			long double val = ((long double) j/(long double) r);
			sum += val*val*val;
			sumsq += val*val;
		}
		sum *= (1/sumsq);
		ans *= sum;
	}
	printf("%0.4Lf\n", ans);
	return 0;
}
