#include <iostream>
using namespace std;
int t,n;
int gcd(int p,int q){
	int du;
	while (q){
		du=p%q;
		p=q; q=du;
	}
	return(p);
}
int main()
{
	scanf("%d",&t);
	while (t){
		scanf("%d",&n);
		int tu=n;
		int mau=180;
		int ucln=gcd(tu,mau);
		tu/=ucln;
		mau/=ucln;
		while(tu/mau>=2) tu-=mau*2;
		if (mau==1){
			if (tu==1) printf("$\n");
			else printf("%d$\n",tu);
		}
		else{
			if (tu==1) printf("$/%d\n",mau);
			else printf("%d$/%d\n",tu,mau);
		}
		t--;
	}
    return 0;
}
