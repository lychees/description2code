#include <iostream>
using namespace std;

int countD(char arr[][1000],int n,int m)
{
	int c =0 ;
	for(int i=0;i<n-1;i++)
	for(int j=0;j<m-1;j++)
	{
		if (arr[i][j] == '/' && arr[i+1][j] == '\\' && (arr[i][j+1] == '\\' && arr[i+1][j+1] =='/'))
		c++;
	}
	return c;
}
int main()
{
    int t;
    cin>>t;
    for(int z=0;z<t;z++)
    {
    	int n,m;
    	cin>>n>>m;
    	char arr[n][1000];
    	for(int i = 0;i<n;i++)
    	for(int j=0;j<m;j++)
    	cin>>arr[i][j];
    	
    	cout<<countD(arr,n,m)<<endl;;
    }
}
