#include<bits/stdc++.h>
using namespace std;
void call(char ch)
{
	switch(ch)
	{
		case '2':cout<<"cde";
			break;
		case '6':cout<<"b3";
			break;
		case '7':cout<<"acf";
			break;
		case '9':cout<<"c6a";
			break;
	}
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string str;
    	cin>>str;
    	int len=str.length();
    	for(int i=0;i<len;i++)
    	{
    		call(str[i]);
    	}
    	cout<<endl;
    }
}
