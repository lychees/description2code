#include<bits/stdc++.h>
using namespace std;

int main()
{
	string w[32770],s;
	long int i,j,n,c,ans=0;
	cin>>n;
	n=1<<n;
	for(i=0;i<n-1;i++)
		cin>>w[i]>>s;
	sort(w,w+n-1);
	for(i=0;i<n-1;i++){
		c=1;
		while(i<n-2 && w[i]==w[i+1]){
			c++;
			i++;
		}
		if(c>ans){
			ans=c;
			s=w[i];
		}
	}
	cout<<s;
}
