#include<bits/stdc++.h>
using namespace std;
int main()
{
	ios::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		char str[35];
		cin>>str;
		long long int res=0,a=0;
		for(int i=strlen(str)-1;i>=0;i--)
		{
			if(str[i]=='+')
			{
				a=str[i-1]-'0';
				i--;
				res=a+res;
			}
			else if(str[i]=='-')
			{
				a=str[i-1]-'0';
				i--;
				res=res-a;
			} 
			else if(str[i]=='*')
			 {
			 	a=str[i-1]-'0';
			 	i--;
				res=res*a;
		     }
		     else
		     {
		     	res=str[i]-'0';
		     }
		 }
		cout<<res<<endl;
	}
	return 0;
}
