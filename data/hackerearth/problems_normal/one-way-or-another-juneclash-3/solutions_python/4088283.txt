import sys

inp=sys.stdin.read()

L=map(int, inp.split())

ctr=0

T=L[ctr]
ctr+=1

for x in xrange(T):
	odd=[]
	even=[]
	flag1=[]
	
	n=L[ctr]
	ctr+=1
	
	for y in xrange(n):
		
		res=0
		res+=L[ctr]
		res+=L[ctr+1]
		ctr+=2
		flag=True
		
		try:
			if res%2==1:
				
				if not odd[0]==(L[ctr]%2):
					flag=False
					ctr+=1
				else:
					ctr+=1
					
			else:
				
				if not even[0]==(L[ctr]%2):
					flag=False
					ctr+=1
				else:
					ctr+=1
				 	
		except:
			
			if res%2==1:
				odd.append(L[ctr]%2)
				even.append((L[ctr]+1)%2)
				ctr+=1
			else:
				even.append(L[ctr]%2)
				odd.append((L[ctr]+1)%2)
				ctr+=1
				
		if flag==False:
			flag1.append(False)
		else: 
			flag1.append(True)
	
	if False in flag1:
