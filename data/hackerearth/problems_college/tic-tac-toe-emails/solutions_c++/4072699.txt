#include <iostream>
#include <cstring>

using namespace std;


void rem_dup (char str[]) {
	int len = strlen (str);
	int i, j, wc, k, freq;;
	char  * sub[len], temp[len];

	i = j = k = wc = 0;
	while (i <= len) {

		if (str[i] == ' ' || str[i] == '\0') {
			temp[k] = '\0';
			freq = j = k = 0;
			while (j < wc) {
				if ( strcmp(sub[j], temp) == 0 ) {
					freq++;
					break;
				}
				j++;
			}
			if ( !freq ) {
				
				sub[wc] = new char [sizeof (temp)];
				strcpy (sub[wc], temp);
				cout << temp << " " ;
				wc++;
				memset (temp , 0, strlen(temp));
			}	
		} else {
			temp[k] = str[i];
			k++;
		}	
		i++;	
	}
	cout << endl;
}

int main() {
	int t, i;
	cin >> t;
	cin.ignore();
	char str[1000], * sub[t];

	i = 0;
	while (i < t) {
		cin.getline(str, 1000);
		//cout << str << endl;
		rem_dup (str);

		i++;
	}


	return 0;
}
