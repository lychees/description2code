#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	ios_base::sync_with_stdio(false);
	cin>>n;
	while(n--)
	{
		long long int a,b;
		cin>>a>>b;
		bitset<64> x(a);
		bitset<64> y(b);
		int ct=0;
		for(int i=0;i<64;i++)
		{
			if(x[i]!=y[i])
			ct++;
		}
		cout<<ct<<endl;
	}
}
