#include <iostream>
#include <cstring>
using namespace std;

bool binSearch(long int* arr,int& index,long int val,int low,int high)
{
    if(arr[high]<val)
    {
    	index=-1;
    	return false;
    }
   
   
   while(low<=high)
   {
      int mid=(low+high)/2;
        
      if(arr[mid]>=val)
      {
      	high=mid-1;
      	index=mid;
      }
      else
      {
      	low=mid+1;
      }
   }
   
   return true;
    	
}

int main()
{
	ios_base::sync_with_stdio(false);
    int t;
    cin>>t;
    int* arr=new int[t];
    long int* sum=new long int[t];
    long int val=0;
    memset(sum,0,sizeof(arr));
     
    for(int i=0;i<t;i++)
    {
    	cin>>arr[i];
    	val+=arr[i];
    	sum[i]=val;
    }
    
    long int q;
    cin>>q;
    long int val1;
    bool flag;
    int index;
    while(q--)
    {
    	cin>>val1;
    	
    	flag=binSearch(sum,index,val1,0,t-1);
    	if(flag)
    	{
    		cout<<index+1<<endl;
    	}
    	else
    	{
    		cout<<"-1"<<endl;
    	}
    }
    
    
    
    
    
    return 0;
}
