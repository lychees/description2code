#include<bits/stdc++.h>
using namespace std;
#define ll long long int
int main(){
	int n;
	cin>>n;
	while(n){
		n--;
		ll c,d;
		cin>>c>>d;
		vector<ll>v;
		for(ll i=0;i<c;i++){
			ll a;
			cin>>a;
			v.push_back(a);
		}
		
		for(ll i=0;i<d;i++){
			ll a;
			cin>>a;
			v.push_back(a);
		}
		sort(v.begin(),v.end());
		reverse(v.begin(),v.end());
		for(ll i=0;i<c+d;i++){
			cout<<v[i]<<" ";
		}
		cout<<endl;
	}
}
