#include <iostream>
#include <string>
#include <cctype>
using namespace std;
int main(){
	int T=0;
	string in;
	string key;
	
	cin >> T;
	cin.ignore();
	
	while(T--)
	{
	 in=key="";
	 getline(cin,in);
	 int len = in.length();
	 
	 for(int i=0;i<len;i++)
	  if(isalpha(in[i]) || in[i] == '*')
	   key+=in[i];
	   
	 int last=key.length()-1;
	 len=last;
	 
	 for(int i=0;i<len;i++,last--)
	 {
	  if(key[i] == '*')
	  {
	   cout << key[last] << endl;
	   break;
	  }
	  else if(key[last] == '*')
	  {
	   cout << key[i] << endl;
	   break;
	  }
	 }
	}
	
	return 0;
}
