#include <bits/stdc++.h>
using namespace std;
map<int,int> dp1,dp2;
int solve1(int n)
{
	if(n<0)return 0;
	if(n==0)return 1;
	if(n==0)return 1;
	if(n==1)return 1;
	if(dp1.count(n))return dp1[n];
	return dp1[n]= solve1(n-1) + solve1(n-2);
}
int solve2(int n)
{
	if(n<0)return 0;
	if(n==0)return 1;
	if(n==1)return 1;
	if(n==2)return 2;
	if(dp2.count(n))return dp2[n];
	return dp2[n] = solve2(n-1) + solve2(n-2) + solve2(n-3);
}
int main()
{
    //cout << "Hello World!" << endl;
    int n;
    scanf("%d",&n);
    if(n<0)return printf("ERROR")*0;
    printf("Jack-%d\n",solve2(n));
    printf("Jill-%d\n",solve1(n));
    return 0;
}
