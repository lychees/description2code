#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s;
    cin>>s;
    int len=s.length(),flag=0;
    int a[26];
    memset(a,0,sizeof(a));
    for(int i=0;i<len;i++)a[s[i]-'a']++;
    for(int i=0;i<26;i++){
    	if(a[i]&1)flag++;
    }
    
    if(flag){
    	if(len&1 && flag==1)cout<<"YES\n";
    	else cout<<"NO\n";
    }
    else cout<<"YES\n";
    return 0;
}
