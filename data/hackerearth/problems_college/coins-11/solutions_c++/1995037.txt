#include <iostream>
using namespace std;

char check(int X)
{
	if(X%2==0||X%3==0)
	 return 'f';
	 int i=5;
	for(;i*i<=X;i+=6)
	 if(X%i==0||X%(i+2)==0)
	  return 'f';
	return 't';
}	

int main()
{
    long long T,N;
    scanf("%lld",&T);
    while(T--)
    {
    	scanf("%lld",&N);
    	if(N<=3)
    	{
    		printf("%d\n",N);
    		continue;
    	} 
    	if(check(N)=='t')
    	 printf("2\n");
    	else
    	 printf("1\n");
    }
    return 0;
}
