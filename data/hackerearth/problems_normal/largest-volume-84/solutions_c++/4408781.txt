/*Main code begins now */
double PI=4*atan(1.0);
double r,l;
 
double getV(double x){
    double csx=sqrt(1-x*x);
	return (PI/3.0)*l*(csx)*(3*r*r+3*r*l*x+l*l*x*x);
}
double incr=1e-5;
 
int main(){
	int ri,li;
	scanf("%d%d",&ri,&li);
	r=ri;
	l=li;
	double ans=0;
	for(double x=-1.0;x<=1.0;x+=incr){
		ans=max(ans,getV(x));	
	}
	printf("%.2lf",ans);
}
