#include<iostream>
using namespace std;
#define ll long long int
ll demo(ll n)
{
	ll n1=0;
	do
	{
		ll r=n%10;
		n1=n1*10+r;
		n=n/10;
	}while(n!=0);
	return n1;
}
ll digits(ll n)
{
ll c=0;
do
{
	c++;
	n=n/10;
}while(n!=0);
return c;
}
void display(ll n)
{
	ll a=demo(n);
	if(a==n)
	cout<<"No"<<endl;
	else
	{
	ll u=n*n;
	ll v=a*a;
	ll g=digits(u);
	ll h=digits(v);
	ll p1=demo(u);
	if(p1==v && g==h)
	cout<<a<<endl;
	else
	cout<<"No"<<endl;
	}
}
int main()
{
	ll t;
	cin>>t;
	while(t--)
	{
		ll n;
		cin>>n;
		display(n);
	}
	return 0;
}
