#include <bits/stdc++.h>

using namespace std;

int main() {
	string s;
	set < pair <int, int> > pts;
	int x = 0, y = 0;
	long long res = 0;
	cin >> s;
	pts.insert(make_pair(0, 0));
	for (int i = 0; i < s.length(); i++) {
		if (s[i] == 'L') y--;
		if (s[i] == 'R') y++;
		if (s[i] == 'U') x--;
		if (s[i] == 'D') x++;
		if (pts.find(make_pair(x, y)) != pts.end()) {
			res++;
		} 
		pts.insert(make_pair(x, y));
	}
	printf("%lld\n", res);
	return 0;
}
