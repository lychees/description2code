#include <bits/stdc++.h>

#define ld long double

using namespace std;

int main() {
    cout<<fixed;
    cout.precision(2);
    int t;
    int ships;
    ld x, y, tempX, tempY;

    cin>>t;
    for (int i = 0; i < t; ++i) {
        x = 0.0;
        y = 0.0;
        cin>>ships;
        for (int j = 0; j < ships; ++j) {
            cin>>tempX;
            cin>>tempY;
            x+=tempX;
            y+=tempY;
        }
        x /= ships;
        y /= ships;
        cout<<"("<<x<<" , "<<y<<")"<<endl;
    }
    return 0;
}
