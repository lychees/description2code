#include<iostream>
#include<algorithm>
using namespace std;
int arr[100001];
int main()
{
	int t,n,s;
	cin>>t;
	while(t--)
	{
	cin>>n;
	n=2*n;
	for(int j=0;j<n;j++)	
		cin>>arr[j];
	sort(arr,arr+n);
	int sum=0;
	int j,k;
	for(j=0,k=n-1;j<n/2;j++,k--)
		if(sum<arr[j]+arr[k])
			sum=arr[j]+arr[k];
	cout<<sum<<endl;
	}
	return 0;
}
