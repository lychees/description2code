#include <bits/stdc++.h>
using namespace std;
char a[1005];
int main()
{
	int n;
	cin>>n;
	string s;
	for(int i=1;i<=n;i++)
	{
	cin>>s;
	a[i]=s[0];
	}	
	sort(a+1,a+n+1);
	int ans=0;
	int cnt=1;
	for(int i=2;i<=n;i++)
	{
	if(a[i]==a[i-1])
	{
	cnt++;
	continue;
	}
	if(cnt%10==0)
		ans+=(cnt/10);
	else ans+=((cnt+10)/10);
	cnt=1;
	}
	if(cnt%10==0)
		ans+=(cnt/10);
	else ans+=((cnt+10)/10);
	cout<<ans<<endl;
	return 0;	
}
