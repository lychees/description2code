from sys import stdin
def gcd(a,b):
	while a%b:
		t=a%b;a=b;b=t
	return b
t = int(stdin.readline())
for _ in xrange(t):
	n,k = map(int,stdin.readline().split())
	a = map(int,stdin.readline().split())
	if n<3:
		print "0/1"
	else:
		di = {}
		for i in a:
			di[i] = di.get(i,0)+1
		if di.get(k,0) < 3:
			print "0/1"
		else :
			den = (n*(n-1)*(n-2))/6
			m = di[k]
			num = (m*(m-1)*(m-2))/6
			g = gcd(num,den)
			num/=g
			den/=g
			print "%d/%d"%(num,den)
