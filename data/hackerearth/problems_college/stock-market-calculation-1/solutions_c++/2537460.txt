#include<bits/stdc++.h>
using namespace std ;
int maxarray(int *a,int start,int end)
{
    int i,pos,big;
    big=a[start];
    pos=start;
    for(i=start;i<=end;i++)
    {
        if(a[i]>big)
        {
        pos=i;
        big=a[i];
        }
    }
    return pos;
}
int main()
{
    int t,n,i;
    scanf("%d",&t);
    while(t--)
    {
        long long pro=0;
        int pos=0,start=0;
        scanf("%d",&n);
        int a[n];
        for(i=0;i<n;i++)
        {
            scanf("%d",&a[i]);
        }
        while(pos!=(n-1))
        {
        pos=maxarray(a,start,n-1);
        for(i=start;i<pos;i++)
        {
            pro=pro+(a[pos]-a[i]);
        }
        start=pos+1;
        }
        printf("%lld\n",pro);
    }
    return 0;
}
