#include <iostream>
#include <algorithm>
using namespace std;
int n[3];
int main()
{
   int t ;
   cin>>t ;
   while( t-- ){
      //int a , b , c ;
      cin>>n[0]>>n[1]>>n[2] ;
      sort(n, n+3 );
      int max = 0 ;
      if(  (n[1]-n[0]-1) > (n[2]-n[1]-1) )
         max = n[1]-n[0]-1 ;
      else max = n[2]-n[1]-1 ;
      cout<<max<<endl ;
   }
   return 0 ;
}
