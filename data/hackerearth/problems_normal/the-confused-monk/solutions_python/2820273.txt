from fractions import gcd
gc=0
ans =1
num = int(raw_input())	
arr = map(int,raw_input().split())

mq=1000000007
for i in arr:
	gc = gcd(gc,i)
	ans = (ans*(i%mq))%mq

ans = pow(ans,gc,mq)
print ans
