#include<bits/stdc++.h>
using namespace std;

int main()
{
   int t;
   cin >> t;
   while (t--)
   {
      int n;
      cin >> n;
      int arr[ n+1 ], m;
      for ( int i = 0; i < n; i++ )
      {
         cin >> arr[i];
      }
      int k;
      cin >> k;
      int brr[k+1];
      memset ( brr, 0, sizeof (brr) );
      for ( int i = 0; i < n; i++ )
         if ( arr[i] < k )
            brr[arr[i]]++;
      
      long long int cnt = 0;
      for ( int i = 1; i <= k/2; i++ )
      {
         if( 2*i == k && brr[i]>1 )
             cnt += (brr[i]*(brr[i]-1)/2);
         else
             cnt += brr[i]*brr[k-i];
      }
      cout << cnt << endl;
   }
   return 0;
}
