#include <bits/stdc++.h>
using namespace std;
bool vis[1000001];
void dfs(vector<int> v, int i){
if(v[i]==-1) {
	cout << "Win" <<endl;
	return;
}
if(vis[i]) {
	cout << "Lose" <<endl;
	return;
}
vis[i] = true;
dfs(v,v[i]);
}	
int main()
{
   // cout << "Hello World!" << endl;
   int t; cin >> t;
   while(t--){
   	int n;
   	cin >> n;
   	vector<int> v(n);
   	memset(vis,false,sizeof(vis));
   	for(int i=0;i<n;i++) cin >> v[i];
   	if(v[0]==-1) cout << "Win" << endl;
   	else {
   		dfs(v,0);
   	}
   }
    return 0;
}
