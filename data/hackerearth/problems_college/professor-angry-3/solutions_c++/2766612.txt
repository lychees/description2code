#include <iostream>
using namespace std;
void pre()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);
}
int main()
{
    //out << "Hello World!" << endl;
    pre();
    int t;
    cin>>t;
    while(t--)
    {
    	int n,k,x,count=0;
    	cin>>n>>k;
    	while(n--)
    	{
    		cin>>x;
    		if(x<=0)
    			count++;
    	}
    	k>count?cout<<"YES\n":cout<<"NO\n";
    }
    return 0;
}
