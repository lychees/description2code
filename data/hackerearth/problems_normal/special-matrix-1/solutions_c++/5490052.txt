#include <iostream>
#include<cmath>
using namespace std;
int main()
{
   int T;
   cin>>T;
   while(T--){
   	int N,char_I,char_J,mid;
   	cin>>N;
   	mid=N/2;
   	char Arr[N][N];
   	for(int i=0;i<N;i++){
   		for(int j=0;j<N;j++){
   			cin>>Arr[i][j];
   			if(Arr[i][j]=='*'){
   				char_I=i;char_J=j;}
   		}
   	}
   //	cout<<char_I<<" "<<char_J<<endl;
   	cout<<abs(mid-char_I)+abs(mid-char_J)<<endl;
   }
    return 0;
}
