#include<bits/stdc++.h>
using namespace std;

int a[100100]={0};
int main()
{map<int,int> mp;
    priority_queue<pair<int,int> >q;
    int n,i,x,m;
    int k=0;
    scanf("%d %d",&n,&m);
    for(i=0;i<n;i++)
    {
        scanf("%d",&x);
        if(mp.count(x)==0)
        {
            mp[x]=k;
            a[k]++;
            k++;
        }
        else
        {
            a[mp[x]]++;
        }
        q.push(make_pair(a[mp[x]],x));
        printf("%d %d\n",q.top().second,q.top().first);
    }
    return(0);
}
