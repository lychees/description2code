int T;
int main()
{
	cin >> T;
	while (T--)
	{
		long long n;
		long long ans = 0;
		cin >> n;
		for (int i = 1; i <= sqrt(n); i++)
		{
			if (n % i != 0)
				continue;
			if (i * 1LL * i == n)
			{
				if (i % 4 != 0)
					ans += i;
			}
			else
			{
				if (i % 4 != 0)
					ans += i;
				if ((n / i) % 4 != 0)
					ans += n / i;
			}
		}
		ans *= 8;
		cout << ans << endl;
	}
}
