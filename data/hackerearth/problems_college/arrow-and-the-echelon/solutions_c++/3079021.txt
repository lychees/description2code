#include <iostream>
using namespace std;
 
int main()
{
    int te,ne,reme;
    cin>>te;
    while(te--)
    {
     reme=-1;
     cin>>ne;
     for(int ie=0;ie<=ne;ie=ie+5)
     {
        if((ne-ie)%3==0)
       {
      	reme=ie;
      	break;
        }
     }
    if(reme==-1)
    {
     cout<<"-1"<<endl;
     continue;
    }
    for(int i=0;i<ne-reme;i++)
    cout<<"5";
    for(int i=0;i<reme;i++)
    cout<<"3";
    cout<<endl;
   }
    return 0;
}
