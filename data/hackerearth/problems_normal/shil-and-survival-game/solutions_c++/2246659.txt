#include <bits/stdc++.h>

using namespace std;

typedef long long ll;

int a[100010];

set <int> ans; 

int main () {
	int n;
	cin >> n;
	int l = 0;
	int r = n - 1;
	int curmax = 0;
	for (int i = 0; i < n; i++) {
		cin >> a[i];
		if (a[i] > curmax) {
			ans.insert (i + 1);
			curmax = a[i];
		}
	}
	curmax = 0;
	for (int i = n - 1 ; i >= 0; i--) {
		if (a[i] > curmax) {
			ans.insert (i + 1);
			curmax = a[i];
		}
	}
	for (set<int>::iterator it = ans.begin(); it != ans.end(); it++)
			cout << *it << " ";
}
