#include <iostream>
using namespace std;

int main()
{
    int t,n,m,max,i,dis,pos,a,b,c;
    cin>>t;
    while(t--){
    	cin>>n>>m;
    	while(n--){
    		max=-1;
    		for(i=0;i<m;++i){
    			cin>>a>>b>>c;
    			dis=10000*(a+b+c)-100*(a*b + b*c + c*a) + a*b*c;
    			if(dis>max){
    				max=dis;
    				pos=i+1;
    			}
    		}
    		cout<<pos<<" ";
    	}
    	cout<<"\n";
    }
    return 0;
}
