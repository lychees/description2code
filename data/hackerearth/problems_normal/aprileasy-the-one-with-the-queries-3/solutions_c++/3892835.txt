#include <iostream>
#include <cmath>
#include <algorithm>
#include <vector>
#include <set>
using namespace std;
#define num long long int
#define pb push_back

vector<num> v[26];

int main(){
	string s,qs;
	char c;
	cin>>s;
	for(num i=0;i<s.length();i++)
	{
		c=s[i];
		v[(int)(c-'a')].pb(i+1);
	}
	num q;
	cin>>q;
	while(q--)
	{
		cin>>qs;
		int flag=0,mflag=0;
		num prev=0;
		for(num i=0;i<qs.length();i++)
		{
			int loc=qs[i]-'a';
			for(num j=0;j<v[loc].size();j++)
			{
				if(v[loc][j]>prev)
				{
					prev=v[loc][j];
					flag=1;
					break;
				}
			}
			if(flag)
			{
				flag=0;
			}
			else
			{
				cout<<"No"<<endl;
				mflag=1;
				break;
			}
		}
		if(!mflag)
			cout<<"Yes"<<endl;
	}
	return 0;
}
