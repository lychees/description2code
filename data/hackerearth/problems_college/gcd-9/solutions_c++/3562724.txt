#include <bits/stdc++.h>
using namespace std;
int gcd(int a,int b)
{
    if(b==0)
        return a;
    else
        return gcd(b,a%b);
}
int arr[500006];
int main()
{
    int i,j,k,n,T,ans,g,p;
    scanf("%d%d",&n,&k);
    for(i=0;i<n;i++){
    	scanf("%d",&arr[i]);
    }
    ans=0;
    for(i=0;i<n;i++){
    	p=0;
    	if(arr[i]>k){
	    	g=arr[i];
	    	p=1;
	    	for(j=i+1;(g>k)&&(j<n);j++){
	    		g=gcd(g,arr[j]);
	    		if(g<k)
	    			break;
	    		p++;
	    	}
    		if(ans<p)
    			ans=p;
    	}
    }
    printf("%d\n",ans);
    return 0;
}
