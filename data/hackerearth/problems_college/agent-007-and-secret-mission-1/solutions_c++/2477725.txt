#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	int num;
    	cin>>num;
    	if(((num*(num+1))/2)%2==0)
    	{
    		int count=0;
    		for(int i=0;i<s.size();i++)
    		{
    			if(s[i]=='1')
    			{
    				count++;
    			}
    		}
    		cout<<count<<endl;
    	}
    	else
    	{
    		int count=0;
    		for(int i=0;i<s.size();i++)
    		{
    			if(s[i]=='0')
    			{
    				count++;
    			}
    		}
    		cout<<count<<endl;
    	}
    }
    return 0;
}
