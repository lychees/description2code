#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cstring>
using namespace std;
int main() {
    long long t;
    cin>>t;
    for(int i=0;i<t;++i) {
        long long a0,a1,n,mod;
        cin>>a0>>a1>>n>>mod;
        long long A[n];
        A[0]=a0;
        A[1]=a1;
        for(int j=2;j<n;++j) {
            A[j]=(A[j-1]+A[j-2])%mod;
        }
        long long *arrays=new long long[mod];
        for(int j=0;j<mod;++j) {
            arrays[j]=0;
        }
        for(int j=0;j<n;++j) {
            arrays[A[j]]++;
        }
        long long force=0;
        for(int j=0;j<mod;++j) {
            force=force+arrays[j]*arrays[j];
        }
        delete(arrays);
        cout<<force<<endl;
    }
    return 0;
}
