for __ in range(input()) :
	org,dist,answer,ans= raw_input(),raw_input(),[],0
	if len(dist)==len(org):
		for i in range(0,len(dist)):
			if org[i]!=dist[i]:
				ans+=1
		print ans
	else:
		if len(dist)>len(org):
			dist,org = org,dist
		for i in range(0,len(org)-len(dist)+1):
			ans=0
			for j in range(0,len(dist)):
				if dist[j]!=org[i+j]: 
					ans+=1
			answer.append(ans)
		print min(answer)
