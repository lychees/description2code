#include <iostream>
using namespace std;
#include<algorithm>
int main()
{
	int t;
	string a;
	cin>>t;
	while(t--) {
		cin>>a;
		if(a.size()<3) {
			cout<<"Not possible!"<<endl;
			continue;
		}
		sort(a.begin(),a.end());
		next_permutation(a.begin(),a.end());
		next_permutation(a.begin(),a.end());
		cout<<a<<" ";
		sort(a.begin(),a.end());
		reverse(a.begin(),a.end());
		prev_permutation(a.begin(),a.end());
		prev_permutation(a.begin(),a.end());
		cout<<a<<" ";
		cout<<endl;
	}
    return 0;
}
