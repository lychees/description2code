#include <iostream>
#define ll long long
using namespace std;
bool p[1000000]={false};
void seive()
{
	
	for(ll int i=2; i<=1000; i++)
	{
			for(ll int j=i*i; j<=1000000; j=j+i)
			{	
				p[j]=true;
			}
		
		
	}

}

int main()
{	ll int t;
	cin>>t;
	
	while(t--)
	{
		ll int a,b;
		cin>>a;
		cin>>b;
		seive();
		p[0]=p[1]=true;
		ll int sum=0;
		for(ll int i=a; i<=b; i++)
		{	if(p[i]== false)
				sum=sum+i;
			
		}
		cout<<sum<<endl;
		
	}
    
    
}
