#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n ; 
    long long c = 0;
    
    cin>>n;
    string st,rev;
   
    map<string,int> arr;
   
    while(n--)
    {
    	cin>>rev;
    	st = rev; 
    	reverse(rev.begin(),rev.end());
    	
    	if(arr[rev] > 0)
    	{
    		c += arr[rev];
    		arr[st]++;
        }
    	else
    	{
    		arr[st]++;
    	}
    }
    cout<<c;
    return 0;
}
