N = int(raw_input())
scores = [int(raw_input()) for x in xrange(0, N)]
scoresTemp = []

for i in xrange(0, N):
    current = scores.pop(0)

    if current != 0:
        scoresTemp.append(current)

    elif current == 0:
        if (len(scoresTemp) > 0):
            scoresTemp.pop()

if (len(scoresTemp) > 1):
    print reduce(lambda x, y: x+y, scoresTemp)
elif len(scoresTemp) == 1:
    print scoresTemp[0]
else:
    print 0
