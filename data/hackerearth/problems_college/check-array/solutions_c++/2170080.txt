#include <iostream>
#include <cstring>
#define MAX 100005
using namespace std;
int a[MAX];
int main()
{
	int t;
	cin>>t;
	while(t--){
    int n;
    cin>>n;
   memset(a,0,sizeof a);
   int sum=0;
    for(int i=1;i<=n;i++)
   {
   	cin>>a[i];
    sum+=a[i];
   }
   
   int sumleft=0,sumright=sum,f=0;
   if(sum==0)
   {cout<<"YES\n";continue;}
   
   for(int i=1;i<=n;i++)
   {
   sumright-=a[i];
   sumleft+=a[i-1];
   if(sumleft==sumright)
   {
   	cout<<"YES\n";
   	f=1;
   	break;
   }
   }
    if(f==0)
    cout<<"NO\n";
}
    return 0;
}
