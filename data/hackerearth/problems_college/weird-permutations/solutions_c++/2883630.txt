#include<bits/stdc++.h>
using namespace std;
#define ll long long int
ll dp[18][(1<<16)];
ll fun(ll mask,ll pre,ll idx,ll n,ll k);
int main()
{
        int t;
        scanf("%d",&t);;
        while(t--)
        {
            ll n,k;
            scanf("%lld%lld",&n,&k);
            for(ll k1=0;k1<=n;k1++)
                {
                    for(ll j=0;j<=(1<<(n));j++)
                    dp[k1][j]=-1LL;
                }
            ll mask=0,lol=2,i;
            ll ans=0;
            for(i=1;i<=n;i++)
            {
                mask=mask|(1<<(i-1));
                ans+=fun(mask,i,lol,n,k);
                mask=mask&(~(1<<(i-1)));
            }
            printf("%lld\n",ans);
        }
}
ll fun(ll mask,ll pre,ll idx,ll n,ll k)
{
    if(idx==n+1)
        {
            return 1;
        }
    if(dp[pre][mask]!=-1)
    {
        return dp[pre][mask];
    }

        long long int v=0;
    for(ll i=1;i<=n;i++)
    {
        if((mask&(1<<(i-1)))==0)
        {
            if(abs(3*pre -2*i)<=k)
            {
                mask=mask|(1<<(i-1));
                v+=fun(mask,i,idx+1,n,k);
                mask=mask&(~(1<<(i-1)));
            }
        }
    }
    return (dp[pre][mask]=v);
}
