#include<iostream>
#include<vector>
using namespace std;
int main(){
    int t;
    cin>>t;
    while(t--){

        int n,x;
        cin>>n>>x;
        int arr[n+1];
        long sum=0;
        for(int i=1;i<=n;i++)
        {
            cin>>arr[i];
            sum+=arr[i];
        }
        if(sum<=x)
            cout<<"YES"<<endl;
        else{
            int dp[2][x+1];
            for(int i=0;i<=x;i++)
                dp[0][i]=0;
                for(int i=1;i<=n;i++){

                    for(int j=0;j<=x;j++){
                        dp[i%2][j]=dp[(i-1)%2][j];
                        if(j>=arr[i]){

                            dp[i%2][j]=max(dp[(i-1)%2][j-arr[i]]+arr[i],dp[i%2][j]);
                            }


                    }

                }
               



            if(sum-max(dp[0][x],dp[1][x])<=x)
                cout<<"YES"<<endl;
            else
                cout<<"NO"<<endl;




        }

    }





}
