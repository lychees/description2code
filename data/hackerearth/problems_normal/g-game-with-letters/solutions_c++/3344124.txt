#include <iostream>
#include <cstdlib>
#include <cmath>
#include <algorithm>
#include <map>
#include <utility>
#include <vector>
using namespace std;

int main(){
	int t, n;
	string s;
	cin >>t;
	for(int i = 0; i < t; i++){
		cin >> n >> s;
		long long int teams[n][26];
		long long int scores[n][26];
		long long int scores_sum[n];
		long long int str[26];
		long long int teamsize[n];
		for(int j = 0; j < n; j++){
			for(int k = 0; k < 26; k++){
				teams[j][k] = 0;
				scores[j][k] = 0;
			}
			scores_sum[j] = 0;
			teamsize[j] = 0;
		}
		for(int j = 0; j < 26; j++) str[j] = 0;
		
		string tmp_str;
		for(int j = 0; j < n; j++){
			cin >> tmp_str;
			for(long long int k =0; k < tmp_str.length(); k++){
				teams[j][tmp_str[k]-97]++;
			}
			teamsize[j] = tmp_str.length();
		}
		for(int j =0; j < s.length(); j++){
			str[s[j]-97]++;
		}
		for(int j = 0; j < 26; j++){
			for(int k = 0; k < n; k++){
				scores[k][j] = teams[k][j] * str[j];
			}
		}
		for(int j = 0; j < n; j++){
			for(int k = 0; k < 26; k++){
				scores_sum[j] += scores[j][k];
			}
		}
		long long int max_score = -1;
		long long int max_score_f_ind = -1;
		long long int winn_team = -1;
		long long int min_len = 0;
		for(int j = 0; j < n; j++){
			if(scores_sum[j] > max_score){
				max_score = scores_sum[j];
				max_score_f_ind = j;
				winn_team = j;
				min_len = teamsize[j];
			}
			else if(scores_sum[j] == max_score){
				if(teamsize[j] < teamsize[max_score_f_ind]){
					max_score_f_ind = j;
					winn_team = j;
				}
				else if(teamsize[j] == teamsize[max_score_f_ind]);
			}
		}
		//cout << "Winning Team: " << winn_team+1 << endl;
		cout << winn_team+1 << endl;
	}
	return 0;
}
