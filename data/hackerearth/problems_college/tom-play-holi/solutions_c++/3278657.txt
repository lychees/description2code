typedef long long int LL;
using namespace std;

vector<int>power2;

void pow2(LL n)
{
  int i,j,k;
  vector<int>vr,v;
  vr.push_back(1);
  for(i=1;i<=n;i++)
  {
	 int c=0;
	 for(j=0;j<vr.size();j++)
	 {
		  k=vr[j]*2;
		  v.push_back(k%10+c);
		  c=0;
		  c=k/10;
	 }
	 if(c!=0)
	 v.push_back(c);
	 vr.clear();
	for(j=0;j<v.size();j++)
	vr.push_back(v[j]);
	 v.clear();
  }
  i=vr.size();
  i--;
  for(;i>=0;i--)
	power2.push_back(vr[i]);
}


int main()

   TC{
	power2.clear();
	   LL n;
	   cin>>n;
	   pow2(n);
	  for(int i=0;i<power2.size();i++)
	  cout<<power2[i];
	  cout<<endl;
   }
	return 0;
}
