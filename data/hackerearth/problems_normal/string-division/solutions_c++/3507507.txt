#include<iostream>
#include<cstdio>
#include<string>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		string str,a,b,c,d;
		int i,j,k;
		bool outer=false;
		cin>>str;
		int lim=str.length();
		if(lim>=10){
			cout<<"YES\n";
			continue;
		}
		else if(lim<4){
			cout<<"NO\n";
			continue;
		}
		else{
			for(i=1;i<lim;i++){
				if(outer)
					break;
				a=str.substr(0,i);
				//cout<<"a is "<<a<<endl;
				for( j=i+1;j<lim;j++){
					if(outer)
						break;
					b=str.substr(i,(j-i));
				//	cout<<"b is "<<b<<endl;
					if(a==b){
					//	cout<<a<<" "<<b<<endl;
						continue;
					}
				for( k=j+1;k<lim;k++){
					 c=str.substr(j,(k-j));
					 d=str.substr(k);
				//	 cout<<a<<" "<<b<<" "<<c<<" "<<d<<endl;
					 if(a==b||a==c||a==d)
						 continue;
					 if(b==c||b==d)
						 continue;
					 if(c==d)
						 continue;
					 else{
					 	 
						 outer=true;
						 cout<<"YES\n";
						 break;
					 }
				}
				}
			}
		}
		if(!outer)
			cout<<"NO\n";
	}
}
