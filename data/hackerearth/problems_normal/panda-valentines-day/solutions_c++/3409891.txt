#include <bits/stdc++.h>
using namespace std;

int prime[1000001];
int arr[1000001];
vector<int> p;

void findPrimes(){
	for(int i = 0; i < 1000001; i++) prime[i] = 1;
	prime[0] = prime[1] = 0;
	int count = 0;
	for(int i = 2; i < 1000001; i++){
		if (prime[i]){
			p.push_back(i);
			for(int j = 2*i; j < 1000001; j+=i){
				prime[j] = 0;
			}
		}
	}
}

int countPF(int n){
	int idx = 0;
	while (idx < p.size() && n > p[idx]){
		if (n % p[idx] == 0) return 1 + arr[n/p[idx]];
		idx++;
	}
	
	if (n != 1) return 1;
	return 0;
}



int main()
{
	findPrimes();
	arr[1] = 0;
	arr[0] = 0;
	for(int i = 2; i < 1000001; i++){
		if (prime[i]) arr[i] = 1;
		else arr[i] = countPF(i);
	}
	
	for(int i = 1; i < 1000001; i++) arr[i] += arr[i - 1]; 
	
	int t;
	cin>>t;
	while (t--){
		int n;
		cin>>n;
		cout<<arr[n]<<endl;
	}
	
    return 0;
}
