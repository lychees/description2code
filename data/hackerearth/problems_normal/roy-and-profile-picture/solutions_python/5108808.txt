ideal=input()
N=input()
for i in range(N):
	L=raw_input().split(' ')
	W=int(L[0])
	H=int(L[1])
	if W<ideal or H<ideal:
		print "UPLOAD ANOTHER"
	else:
		if W!=H:
			print "CROP IT"
		else:
			print "ACCEPTED"
