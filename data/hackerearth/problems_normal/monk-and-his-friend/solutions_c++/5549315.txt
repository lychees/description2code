#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	long long x,y;
    	cin>>x>>y;
    	long long z = x^y;
    	long long count = 0;
    	while(z) {
    		z = z & (z-1);
    		count++;
    	}
    	cout<<count<<"\n";
    }
    return 0;
}
