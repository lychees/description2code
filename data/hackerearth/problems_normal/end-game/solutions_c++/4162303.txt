#include <iostream>
#include <cmath>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    //	cout<<t<<" :";
    	int n,a,b,c,d,move;
    	cin>>n>>a>>b>>c>>d>>move;
    	  int l,r,t,bo;
    	  if(move==0)
    	  {
    	  	int k=n-a;
    	  	l=b-k;
    	  	r=b+k;
    	  	t=n;
    	  	bo=a;
    	  }
    	  else if(move==1)
    	  {
    	  	int k=n-a;
    	  	l=b-k-1;
    	  	r=b+k+1;
    	  	t=n;
    	  	bo=a-1;
    	  }
    	  //cout<<t<<bo<<l<<r<<endl;
    	  if(c<=t && c>=bo && d<=r && d>=l)cout<<"Draw"<<endl;
    	  else cout<<"White Wins"<<endl;
    	  
    }
    return 0;
}
