#include<iostream>
#include<iomanip>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long int n;
        cin>>n;
        long int a[n];
        for(long int i=0;i<n;i++)
        {
            cin>>a[i];
        }
        double sum=0;
        for(long int i=0;i<n;i++)
        {
            sum=sum+a[i];
        }
        sum=sum/n;
        //res = floor(1000 * res) / 1000;
        cout<<fixed<<setprecision(8)<<sum<<endl;
    }
    return 0;
}
