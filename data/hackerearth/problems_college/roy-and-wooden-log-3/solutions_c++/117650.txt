#include<iostream>
#include<cstdio>

using namespace std;

int main()
{
	int t,l,c;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&l);
		if(l%2 == 0)
		{
			c=(l/2)-1;
			printf("%d %d\n",c+2,c);
		}
		else
		{
			c=(l/2);
			printf("%d %d\n",c+1,c);
		}
		
			
	}
	return 0;
}
