#include "cstdio"
#include "vector"
#include "deque"
#include "algorithm"
using namespace std;
deque <int> kolejka;
vector <int> G[100005];
bool odw[100005];
int n,x,ile[100005],w[100005],maks;
int main()
{
	scanf ("%d", &n);
	for (int i=1; i<=n; i++)
	{
		scanf ("%d", &x);
		if (x!=-1) G[i].push_back(x),ile[x]++;
	}
	for (int i=1; i<=n; i++)
	{
		if (ile[i]==0) w[i]=1,kolejka.push_back(i);
	}
	for (int i=0; i<kolejka.size(); i++)
	{
		int v=kolejka[i];
		for (int j=0; j<G[v].size(); j++)
		{
			w[G[v][j]]=max(w[G[v][j]],w[v]+1);
			if (w[G[v][j]]>maks) maks=w[G[v][j]];
			kolejka.push_back(G[v][j]);
		}
	}
	printf ("%d\n", maks);

	return 0;
}
