/*Given 'n', print the symbol 'Z' spanning n rows using '*' symbol. Value of n>2 and n<=20.

Example:
1) Input : n=3
Output:
***
 *
***

2) Input n=4
Output:
****
  *
 *
****

*/

#include <iostream>
using namespace std;
 void pat(int n){

    cin >> n;
    for(int i=1; i <= n; i++){
    	cout<<"*";
    }
    cout << endl;
    
    int input;
    for(int i =2; i<n; i++){
    	input=0;
    	while(input!=(n-i)){
    		cout<<" ";
    		input++;
    	}
    	cout<<"*"<<endl;
    }
    for(int i=1;i<=n;i++){
    	cout<<"*";
    }
    }
int main()
{
	 int n;
  	cin >> n;
    pat(n);
    return 0;
}
