def weight(string):
    w = 0
    for i in string:
        w += ord(i) - 96
    return w


t = int(input())
for j in range(t):
    string = raw_input()
    num = int(input())
    chars = list(map(str, raw_input().split(' ')))
    for i in range(num):
        string = string.replace(chars[i], '')
    print(weight(string))
