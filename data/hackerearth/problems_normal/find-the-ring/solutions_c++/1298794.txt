#include <iostream>
using namespace std;

int main()
{
    int T, index, N;
    cin >> T;
    bool isOdd;
    while(T--)
   	{
   		cin >> index >> N;
   		if (N == 0)
   			cout << index;
		else
		{
		   	isOdd = N & 1;
		   	if (isOdd)
		   	{
		   		if(index == 1)
		   			cout << 0;
		   		else
		   			cout << 1;
		   	}
		   	else
		   	{
		   		if(index == 1)
		   			cout << 1;
		   		else
		   			cout << 0;
		   	}
		}
   		
   		cout << endl;
   	}
    
    return 0;
}
