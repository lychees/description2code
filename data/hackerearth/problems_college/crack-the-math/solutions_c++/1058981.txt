#include <iostream>
using namespace std;

int main()
{
    string a,b;
    
    int t,i,len;
    cin>>t;
    while (t--) {
    	cin>>a>>b;
    	len = a.length();
    	char c[len];
    	for (i=0;i<len;i++) {
    		if (a[i]==b[i]) {
    			c[i]='0';
    		}
    		else c[i]='1';
    	}
    	c[i]='\0';
    	cout<<c<<endl;
    }
    return 0;
}
