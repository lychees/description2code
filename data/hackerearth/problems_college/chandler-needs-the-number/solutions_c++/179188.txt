#include<stdio.h>
#include<iostream>
#include<algorithm>
using namespace std;
long long int gcd(long long int a,long long int b)
{
    if(b==0)
        return a;
    else
        gcd(b,a%b);
}
int main()
{
    int N,i;
    scanf("%d",&N);
    long long int a[N];
    for(i=0;i<N;i++)
        scanf("%lld",&a[i]);
    sort(a,a+N);
    printf("%lld\n",a[0]*a[N-1]);
    return 0;
}
