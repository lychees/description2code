#include <iostream>
#include <cstring>
#include <cmath>
#include <cstdio>
#include <functional>
#include <algorithm>

using namespace std;

int nlist[100005];

int main()
{
	int n,mult ;
	cin >> n;
	memset(nlist,0,100005);
	

	for(int i=2;i< n; i++)
	{	
		for(int j = 2 ; j < (n/i +1) ; j++ )
		{
			if(j*i <= n)
			{
				nlist[j*i] = 1;
			}
		}
	}

	for(int i = 0 ; i < n+1; i++)
	{
		if(nlist[i] == 1) { cout << i << endl; }
	}

	return 0;
}
