#include <iostream>
using namespace std;

int main()
{
    int T, N, count;
    cin >> T;
    while(T--)
    {
    	count=0;
    	cin >> N;
    	while(N != 1)
    	{
    		count++;
    		if(N%3 == 0)
    		N /=3;
    		else
    		{
    			if(N&1)
    			  N--;
    			 else
    			  N =(N >> 1);
    		}
    	}
    cout << count << endl;	
    }
    return 0;
}
