#include <bits/stdc++.h>
using namespace std;

//
int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
   int t;
	cin>>t;
	while(t--)
	{

		int n;
		cin>>n;
		int *arr = new int[n];
		for(int i=0; i<n; i++)cin>>arr[i];
		long long int total = 0;
		long long int max_so_far = arr[n-1];
		for(int i=n-2; i>=0; i--)
		{

			if(arr[i]<max_so_far)
				total+=(max_so_far - arr[i]);
			else 
				max_so_far = arr[i];
				
			
		}
		cout<<total<<endl;
		delete arr;
	}

	
    return 0;
}
