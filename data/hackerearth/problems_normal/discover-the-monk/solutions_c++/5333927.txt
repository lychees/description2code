#include <algorithm>
#include <iostream>
using namespace std ;
int main()
{
int t,a,key;
scanf("%d",&a);
scanf("%d",&t);
int arr[a];
for(int i=0;i<a;i++) scanf("%d",&arr[i]);
sort (arr,arr+a);
while(t--){
scanf("%d",&key);
if(binary_search(arr,arr+a,key)){cout<<"YES \n";}
else {cout<<"NO \n";}
}
return 0;
}
