#include <bits/stdc++.h>
using namespace std;
long long int pre[5555556];
#define mod 1000000000000
long long int compute(long long int l, long long int r,long long int c) {
	while(r - l > 1)
	{
		long long int m = (l + r) / 2;
		if(pre[m] <= c) 			
			l = m;
		else
			r = m;
	}
	if(pre[r] <= c) {
		return r;
	}
	if(pre[l] <= c) {
		return l;
	}
	return 0;
}

int main()
{
    long long int i1,i2,cnt = 0;
    for(i1 = 1;i1<=5555555;i1++){
    	i2 = ((i1+1)*i1)/2;
    	if(i2>mod){
    		cnt = i1;
    		break;
    	}
    	pre[i1]=i2;
    }
    long long int test;
    cin>>test;
    while(test--){
    	long long int l,r,ans=0,i,j;
    	cin>>l>>r;
    	l--;
    	i=0;
    	long long int ans1=0;
    	for(i = 0; pre[i] <= l;i++) {
			long long int fin = compute(0, i,l - pre[i]);
			ans1 += fin;
		}
		long long int to1 = 0;
		for(i = 0; pre[i] <= r;i++) {
			long long int fin = compute(0, i,r - pre[i] );
			ans += fin;
		}
		cout << ans - ans1 <<endl;
    }
    return 0;
}
