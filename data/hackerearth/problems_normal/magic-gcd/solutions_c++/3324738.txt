#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
typedef vector<int> vi;
typedef pair<pair<int,int>,int> piii;
typedef map<int,int> mi;

#define si(a) scanf("%d",&a)
#define sii(a,b) scanf("%d %d",&a,&b)
#define siii(a,b,c) scanf("%d %d %d",&a,&b,&c)
#define pi(a) printf("%d\n",a)
#define nl printf("\n");
#define pb push_back
#define mp make_pair
#define all(c) (c).begin(),(c).end()
#define f(i,a,b) for(i=a;i<b;i++)
#define rf(i,a,b) for(i=a;i>=b;i--)
#define clr(x,a) memset(x,a,sizeof(x))
#define MAX 1000100
#define MOD 1000000007
#define F first
#define S second

int n,m;
map<pair<int,int>,int > mm;

int main(){
    int r,k,i,c=0,x=0,y=0,j,t,l,z,x1=0,y1=0;
    int ans=0;string p;

    si(t);
    while(t--){
    	mm.clear();
        ans=0;
        sii(x,y);
        si(k);
        queue<piii > q;
        q.push(mp(mp(x,y),0));
        while(!q.empty()){
            piii pp=q.front();
            q.pop();
            x1=pp.F.F;y1=pp.F.S;z=pp.S;
            if(mm[mp(x1,y1)]!=0)
            	continue;
            mm[mp(x1,y1)]=1;
            if(x1+k<=ans && y1+k<=ans)
            	break;
            ans=max(ans,__gcd(x1,y1));
            if(z>=k)
                continue;
            q.push(mp(mp(x1+1,y1),z+1));
            if(x1>=1)
                q.push(mp(mp(x1-1,y1),z+1));
            q.push(mp(mp(x1,y1+1),z+1));
            if(y1>=1)
                q.push(mp(mp(x1,y1-1),z+1));
        }
        cout<<ans<<endl;
    }

    return 0;
}
