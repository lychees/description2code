noOfTestCases = int(raw_input())

def num(totalDigitCount):
	maxPossiblefiveCount = totalDigitCount / 3
	
	while(maxPossiblefiveCount != 0 and  (totalDigitCount - maxPossiblefiveCount * 3) % 5!= 0):
		maxPossiblefiveCount -= 1
		
	if(maxPossiblefiveCount == 0 and (totalDigitCount - maxPossiblefiveCount * 3)%5 != 0):
		return -1
	else:
		returnString = ""
		u = 3*maxPossiblefiveCount + 1
		for i in range(1, u):
			returnString += "5"
		u = totalDigitCount - maxPossiblefiveCount * 3 + 1
		for i in range(1, u):
			returnString += "3"
	
	return returnString

for i in range(1, noOfTestCases + 1):
	x = raw_input()
	print num(int(x))
