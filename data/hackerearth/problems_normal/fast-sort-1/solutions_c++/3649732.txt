#include<bits/stdc++.h>
using namespace std;
int main()
{
    //printf("Hello World!\n");
    int t;
    cin>>t;
    while(t--)
    {
    	string s1,s2,ans;
    	ans="";
    	cin>>s1>>s2;char c;
    	map<char,int> m;
    	for(int i=0;i<s1.size();i++)
    	{
    		m[s1[i]]=i;
    	}
    	vector<pair<int,char> > vp;
    	for(int i=0;i<s2.size();i++)
    	{
    		vp.push_back(make_pair(m[s2[i]],s2[i]));
    	}
    	sort(vp.begin(),vp.end());
    
    	for(int i=0;i<vp.size();i++)
    	{
    		c=vp[i].second;
    	//	cout<<"hi";
    		ans=ans+c;
    	}
    	
    	cout<<ans<<endl;
    }
    return 0;
}
