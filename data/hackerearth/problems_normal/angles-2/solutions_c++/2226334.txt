int gcd(int x,int y)
{
	if(y==0)
		return x;
	return gcd(y,x%y);
}
int main(void){
	std::ios::sync_with_stdio(false);cin.tie(0);
	int T;
	int i,n,m;
	int g,x;
	T=1;
	while(T--)
	{
		cin>>n>>m;
		g=360;
		while(n--)
		{
			cin>>x;
			if(x==0)
				continue;
			g=gcd(g,x);
		}
		while(m--)
		{
			cin>>x;
			if(x%g==0)
				cout<<"YES"<<endl;
			else
				cout<<"NO"<<endl;
		}
	}
//	system("pause");
	return 0;
}
