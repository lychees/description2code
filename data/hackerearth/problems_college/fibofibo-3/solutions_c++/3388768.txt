#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		if(n == 0)
		{
			cout<<"1"<<" "<<"0"<<"\n";
			continue;
		}
		if(n == 1)
		{
			cout<<"0"<<" "<<"1"<<"\n";
			continue;
		}
		if(n == 2)
		{
			cout<<"1"<<" "<<"1"<<"\n";
			continue;
		}
		int x = 1;
		int y = 1;
		int c;
		for(int i=2;i<n;i++)
		{
			c = x+y;
			x = y;
			y = c;
		}
		cout<<x<<" "<<c<<"\n";
	}
}
