for i in xrange(input()):
	p,x,y,u,v=[int(x) for x in raw_input().split()]
	if (abs(x-u)+abs(y-v))%2!=0:
		print 'Terrorists Win !'
		continue
	if p>2:
		print 'Counter-Terrorists Win !'
		continue
	elif p==2:
		if abs(x-u)==abs(y-v):
			print 'Counter-Terrorists Win !'
			continue
		else:
			print 'Terrorists Win !'
			continue
	elif x-u==0:
		print 'Counter-Terrorists Win !'
		continue
	else:
		print 'Terrorists Win !'
		continue
