#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
long long power(long long a,long long b)
{
	if(b==0)
	return 1;
	if(b==1)
	return (a%mod);
	long long temp=power(a,b/2);
	if(b%2)
	return(((temp%mod)*(temp%mod)%mod)*(a%mod))%mod;
	else
	return ((temp%mod)*(temp%mod)%mod);
}
int main()
{
	long long t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		cout<<(power(3,n)%mod)<<endl;
	}
    return 0;
}
