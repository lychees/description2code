using namespace std;

long long int dp[1 << 16];
int buil[1 << 16];

long long int w[1<<16];
int main(){
	for (int i = 0; i < (1 << 16); i++){
		int cnt = 0;
		for (int j = 0; j < 16; j++){
			if ((i >> j) & 1){
				cnt++;
			}
		}
		buil[i] = cnt;
	}
	int t;
	cin >> t;

	while (t--){
		int n;
		cin >> n;
		memset(dp, 0, sizeof(dp));
		memset(w,0,sizeof(w));
		dp[0] = 1;
		long long int ans = 0;
		for (int i = 0; i < (1 << n); i++){
			int with = buil[i];
			int no = with;
			for (int j = 0; j < n; j++){
				if ((i >> j) & 1){
					no--;
					continue;
				}
				if (j == with){
					continue;
				}
				dp[i | (1 << j)] += dp[i];
				w[i|(1<<j)]+=dp[i] * (long long int)(no)+w[i];
			}
		}
		ans=w[(1<<n)-1];
		printf("%lld\n", ans);
	}
	return 0;
}
