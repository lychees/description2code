#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long int n,b;
    	cin>>n>>b;
    	int temp=n/b;
        long int dif1=(n-b*temp);
        long int dif2=(temp+1)*b-n;
    	if(dif1>=dif2)
    	  {
    	  	cout<<(temp+1)*b<<endl;
    	  }
    	  else
    
    	  {
    	  	cout<<(temp)*b<<endl;
    	  }
    
    }
    return 0;
}
