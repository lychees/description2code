from datetime import datetime
format = '%H:%M'
n=int(raw_input().strip())
time = {}
for i in range (0,n):
	num=raw_input().strip().split('-')
	time[datetime.strptime(num[0],format)] = datetime.strptime(num[1],format)
flag = True
items = time.items()
items.sort()	
for p in range (0,len(items)):
	for q in range(p+1,len(items)):
		diff =  items[q][0] - items[p][1] 
		#print diff.days
		if diff.days < 0:
			flag=False 
			break
