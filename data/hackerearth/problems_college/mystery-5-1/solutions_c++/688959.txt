//Mystery 5
#include <iostream>
#include <algorithm>
#include <vector>

int main()
{	
	int t;
	std::cin >> t;
	
	while (t--)
	{
		std::vector<int> v(256, 0);
		std::string s;
		
		std::cin >> s;
		
		for (int i = 0; i < s.size(); ++i)
		{
			v.at(s.at(i))++;
		}
		
		std::cout << *std::max_element(v.begin(), v.end());
		std::cout << std::endl;
	}
	
	return 0;	
}
