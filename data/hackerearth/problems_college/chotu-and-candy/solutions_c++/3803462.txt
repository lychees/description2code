#include <iostream>
#include <cstdlib>

using namespace std;

struct p {
	int x, s, c;
};

int comp(const void *a, const void *b)
{
	struct p *i, *j;
	i = (p *)a;
	j = (p *)b;
	if (i->s > j->s) {
		return 1;
	} else if (i->s == j->s) {
		if (i->c < j->c) {
			return 1;
		} else if (i->c == j->c) {
			return (i->x < j->x);
		} else {
			return -1;
		}
	} else {
		return -1;
	}
}

int main()
{
	int n, q, i, j, k, x, s, c;
	cin >> n >> q;
	
	struct p arr[n];
	for (i = 0; i < n; ++i) {
		cin >> arr[i].x >> arr[i].s >> arr[i].c;	
	}
	
	qsort(arr, n, sizeof(p), comp);
	
	for (i = 0; i < q; ++i) {
		cin >> x;
		cout << arr[n - i - 1].x << endl;
	}
	
	return 0;
}
