#include<iostream>
#include<string>
using namespace std;
int main()
{
	int i,test,flag;
	
	string str;
	cin>>test;
	
	while(test--)
	{
		int count[30];
		for(i=0;i<30;i++)
	{
		count[i]=0;
	}
		cin>>str;
		flag=0;
		for(i=0;i<str.size();i++)
		{
			count[str[i]-97]++;
			//cout<<"A"<<count[str[i]-97]<<endl;
			if(count[str[i]-97]>1)
			{
				flag=1;
				//cout<<count[str[i]-97]<<endl;
				break;
			}
		}
		if(flag==1)
		{
			cout<<"Yes"<<endl;
		}
		else
		{
			cout<<"No"<<endl;
		}
		
	}
}
