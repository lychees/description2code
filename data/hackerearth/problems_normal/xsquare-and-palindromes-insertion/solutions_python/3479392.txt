def palindrome():
	testcases=input()
	for cases in xrange(testcases):
		string_input=raw_input()
		dict_hash=dict()
		length_string=len(string_input)
		length_stringinputset=len(set(string_input))
		if(length_string==2 and length_stringinputset==1):
			print 0
		elif(length_string==3 and length_stringinputset==2):
			print 0
		elif(length_string==2 and length_stringinputset==2):
			print 1
		else:
			for ch in list(string_input):
				if(dict_hash.has_key(ch)):
					dict_hash[ch]+=1
				else:
					dict_hash[ch]=1
			counter=0
			for key in dict_hash.keys():
				if(dict_hash[key]%2==1):
					counter+=1
			if(counter==0):
				print 0
			elif(counter>0):
				print counter-1
			
				
if __name__=='__main__':
	palindrome()
