#include <iostream>
using namespace std;

int main()
{
    int n,s=0,m=0;
    long int x;
    cin>>n>>x;
    long int a[n];
    for(int i=0;i<n;i++)
     {
     	cin>>a[i];
     	if(s<=1)
     	{
     		if(a[i]<=x)
     		m++;
     		else
     		s++;
     	}
     	else
     	  break;
     }
     cout<<m;
    return 0;
}
