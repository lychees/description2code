#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,k;
    	cin>>n>>k;
    	int a[n];
    	for(int i=0;i<n;i++)
    	cin>>a[i];
    	long long ans=0,sum=0;
    	int j=0;
    	for(int i=0;i<n;i++)
    	{
    		if(a[i]<=k)
    		{
    			if(i-j<=1)
    			{
    				sum+=a[i];
    			}
    			else
    			{
    				sum=0;
    				sum+=a[i];
    			}
    			if(sum>ans)
    			ans=sum;
    			j=i;
    		}
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
