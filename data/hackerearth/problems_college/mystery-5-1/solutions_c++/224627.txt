#include <bits/stdc++.h>
using namespace std;

inline void Solve(){
    int n, i, a[27] = {0};
    for(i = 0; i < 27; i++){
        a[i] = 0;
    }
    string s;
    cin>>s;
    n = s.length();
    for(i = 0; i < n; i++){
        a[int(s[i]) - int('a')]++;
    }
    int ans = 0;
    for(i = 0; i < 27; i++){
        ans = max(ans, a[i]);
    }
    cout<<ans<<endl;
    return;
}

int main(){
    int t;
    cin>>t;
    while(t--){
        Solve();
    }
    return 0;
}
