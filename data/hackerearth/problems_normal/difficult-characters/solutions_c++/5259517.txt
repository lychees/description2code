#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int T;
    cin>>T;
    while(T--)
    {
    	string s;
    	cin>>s;
    	map<char,int,greater<int>> m;
    	for(auto i : s)
    	{
    		if(m.find(i)==m.end())
    		{
    			m.insert(make_pair(i,1));
    		}
    		else
    			m[i]+=1;
    	}
    	string temp = "abcdefghijklmnopqrstuvwxyz";
    	for(auto i : temp)
    	{
    		if(m.find(i)==m.end())
    		{
    			m.insert(make_pair(i,1));
    		}
    		else
    			m[i]+=1;
    	}
    	multimap<int,char> mm;
    	for(auto i : m)
    	{
    		mm.insert(make_pair(i.second,i.first));
    	}
    	
    	for(auto i : mm)
    	{
    		cout<<i.second<<" ";
    	}
    	cout<<"\n";
    }
    
    return 0;
}
