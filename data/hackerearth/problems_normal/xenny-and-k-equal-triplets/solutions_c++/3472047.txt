#include <bits/stdc++.h>
#define ll long long
using namespace std;
vector<ll> a;
ll ncr(ll n, ll r) {
	if(n <= 2) return 0;
	if(r == 3) {
		ll tmp = (((n * (n - 1))/2)*(n-2));
		tmp/=3;
		return tmp;
	}
}
void calc() {
	ll n, k,count=0;
	cin >> n >> k;
	a.clear(); 
	a.resize(n);
	for(int i=0;i<n;i++) {
		cin>>a[i];
		if(a[i]==k)
			count++;
	}
	ll num = ncr(count, 3);
	ll den = ncr(n, 3);
	ll g = __gcd(num, den);
	num /= g;
	den /= g;
	cout << num << "/" << den << "\n";
}
int main() {
	int t; cin>>t; 
	while(t--) 
		calc();
}
