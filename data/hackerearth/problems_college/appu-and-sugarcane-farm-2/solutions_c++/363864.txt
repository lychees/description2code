#include <iostream>
using namespace std;

int main() {
	int N, R;
	scanf("%d%d",&N,&R);
	int a[N];
	for(int i = 0; i<N; i++){
		scanf("%d",&a[i]);
	}
	int sum = 0;
	for(int i = 0; i<N; i++){
		sum+=a[i];
	}
	
	float n = N;
	
	int NUM = sum/N;
	
	float num = sum/n;
	
	int Rout = 0;
	
	// cout<<"sum "<<sum<<endl;
	// cout<<"n "<<n<<endl;
	// cout<<"N "<<N<<endl;
	// cout<<"num "<<num<<endl;
	// cout<<"NUM "<<NUM<<endl;
	
	if(NUM==num){
	
		for(int i = 0; i<N; i++){
			if(NUM-a[i]>0){
				Rout ++;
				
			}
		}
		printf("%d",Rout*R);
	}
	else{
		printf("-1");
	}
	return 0;
}
