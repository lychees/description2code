import math

t = int(raw_input())
for _ in range(t):
	z = list(map(int , raw_input().split()))
	a = z[0]
	b = z[1]
	if b == a*2 or a == b*2 or b % a == 0 or a % b == 0:
		print b if b > a else a
	else:
		if b > a :
			if b-a == 1:
				print(b)
			else:
				print("0")
		elif a > b:
			if a - b == 1:
				print(b)
			else:
				temp = int(math.floor(a/b))
				if a - (b*temp) < (b*(temp+1)) - a:
					print(b*temp)
				else:
					print(b*(temp+1))
