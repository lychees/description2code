#include<bits/stdc++.h>
using namespace std;
#define in(n) scanf("%ld",&n)
int main()
{
    long int i,n,x,t;
    vector<int> v(1000001,0);
    v[1]=1,v[4]=1,v[27]=1,v[256]=1;
    v[3125]=1,v[46656]=1,v[823543]=1;
    for(i=11;i<=1000000;i++)
    {
        if(v[i]==0)
        {
            n=i;
            while(n>0)
            {
                x=n%10,n=n/10;
                if(x>0 && i>x*x && v[i-x*x]==1)
                {
                    v[i]=1;
                    break;
                }
            }
        }
    }
    in(t);
    while(t--)
    {
        in(n);
        if(v[n]==0)
            cout << "No" << endl;
        else if(v[n]==1)
            cout << "Yes" << endl;
    }
    return 0;
}
