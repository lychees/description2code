#include <iostream>

using namespace std;

int main() {

	int T;
	cin >> T;
	while(T--) {

		long long int A, B, M;
		cin >> A >> B >> M;
		long long int n = 0;

		if((B-A) <= M) {
			if(A%M == 0) n++;
			if(B%M == 0) n++;
		} else {

			if(A%M != 0) {
				A = A + (M - A%M);
			}

			B = B - B%M;
			n = (B-A)/M + 1;

		}
		cout << n << endl;

	}

}
