n = input()
dom = map(int, raw_input().strip().split())
brain = map(int, raw_input().strip().split())

max_dom = -(10**9)
max_brian = -(10**9)

for index in xrange(1, n):
    max_dom = max(max_dom, abs(dom[index] - dom[index-1]))
    max_brian = max(max_brian, abs(brain[index] - brain[index-1]))

if max_dom > max_brian:
    print "Dom"
    print max_dom
elif max_dom < max_brian:
    print "Brian"
    print max_brian
else:
    print "Tie"
