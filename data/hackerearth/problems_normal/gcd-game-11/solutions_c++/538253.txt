    #include <bits/stdc++.h>
    using namespace std;
    int GCD[1002][1002];
    bool vis[1002][1002][2];
    int dp[1002][1002][2];
    void pre()
    {
    for ( int i = 1; i <= 1000; i++ ) {
    for ( int j = 1; j <= i; j++ ) {
    GCD[i][j] = GCD[j][i] = __gcd(i,j);
    }
    }
    return;
    }
    int f(int a, int b, int turn)
    {
    if ( turn == 0 ) {
    if ( a == 1 ) return 0;
    }	
    if ( turn == 1 ) {
    if ( b == 1 ) return 0;
    }
    if ( vis[a][b][turn] ) return dp[a][b][turn];
    vis[a][b][turn] = true;
    int ans = 0;
    if ( turn == 0 ) {
    ans |= (!f(a,b-1,turn^1));
    if ( GCD[a][b] != 1 ) ans |= (!f(a,b/GCD[a][b],turn^1));
    }
    else {
    ans |= (!f(a-1,b,turn^1));
    if ( GCD[a][b] != 1 ) ans |= (!f(a/GCD[a][b],b,turn^1));
    }
    dp[a][b][turn] = ans;
    return ans;
    }
    int main() {
    int t,A,B;
    pre();
    cin >> t;
    while ( t-- ) {
    cin >> A >> B;
    if(A==1 && B==1)
    cout<<"Draw"<<endl;
    else if(A==1)
    cout<<"Chandu Don"<<endl;
    else if(B==1)
    cout<<"Arjit"<<endl;
    else{
    if ( f(A,B,0) ) cout << "Arjit"<<endl ;
    else cout << "Chandu Don"<<endl;
    }
    }
    return 0;
    }
