#include <iostream>
using namespace std;

int main()
{
    int t,n,k,i,j;
    cin>>t;
    long long ans[t];
    for(i=0;i<t;i++)
    {
    	cin>>n>>k;
    	long long ar[n],dp[n];
    	for(j=0;j<n;j++) cin>>ar[j];
    	dp[0]=ar[0];
    	for(j=1;j<=k;j++)
    	{
    		dp[j]=max(dp[j-1],ar[j]);
    	}
    	for(;j<n;j++)
    	{
    		dp[j]=max(dp[j-1],ar[j]+dp[j-k-1]);
    		dp[j]=max(dp[j],ar[j]);
    	}
    	if(dp[n-1]>0)ans[i]=dp[n-1];
    	else ans[i]=0;
    }
    for(i=0;i<t;i++) cout<<ans[i]<<endl;
    return 0;
}
