#include <bits/stdc++.h>
using namespace std;
int inf=99999999;
int l, k;
vector<pair<pair<int, int> , int > > p;
int comp(pair<pair<int, int> , int > p1, pair<pair<int, int> , int > p2){
	if(p1.first.first != p2.first.first){
		return (p1.first.first < p2.first.first);
	}else{
		return (p1.first.second < p2.first.second);
	}
}
int dp[515][300][300];
int func(int x, int s, int e){
	
	if(e==l){
		return 0;
	}
	if(x==k){
		return inf;
	}
	if(dp[x][s][e] != -1){
		return dp[x][s][e];
	}
	int ans=inf;
	if(p[x].first.first <=e && p[x].first.second > e){
		ans = min(ans,p[x].second+ func(x+1, p[x].first.first, p[x].first.second));
	}
	ans = min(ans, func(x+1, s, e));
	dp[x][s][e]=ans;
	return ans;
}
int main()
{
    int t,a,b,c;
    cin>>t;
    while(t--){
    	cin>>l>>k;
    	memset(dp, -1, sizeof(dp));
    	p.clear();
    	for(int i=0; i<k; i++){
    		cin>>a>>b>>c;
    		p.push_back(make_pair(make_pair(a, b), c));
    	}
    	sort(p.begin(), p.end(), comp);
    	int ans = func(0, 0, 0);
    	if(ans==inf){
    		cout<<"-1"<<endl;
    	}else{
    		cout<<ans<<endl;
    	}
    }
    return 0;
}
