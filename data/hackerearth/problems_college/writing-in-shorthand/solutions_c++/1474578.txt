#include<iostream>
#include<string.h>
#define MAX 3001
using namespace std;
int main()
{
	int n,m,l1[MAX],l2[MAX];
	cin>>n>>m;
	char a[MAX][11],b[MAX][11],*c[MAX];
	for(int i=0;i<m;i++)
	{
		cin>>a[i]>>b[i];
		
		l1[i]=strlen(a[i]);
		l2[i]=strlen(b[i]);
		
		if(l1[i]<=l2[i])
			c[i]=a[i];
		else
			c[i]=b[i];
	}
	char t[MAX];
	for(int i=0;i<n;i++)
	{
		cin>>t;
		for(int j=0;j<m;j++)
		{
			if(strcmp(a[j],t)==0 || strcmp(b[j],t)==0)
			{	cout<<c[j]<<' ';
				break;
			}
		}
	}
}
