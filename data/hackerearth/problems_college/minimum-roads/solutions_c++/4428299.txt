 #include <iostream>
 #include <vector>
using namespace std;

vector <int> adj[1000001];
int visited[1000001];

void dfs(int s)
{
    visited[s]=1;
    for(int i = 0;i < adj[s].size();++i)    
    {
     if(visited[adj[s][i]] ==0)
         dfs(adj[s][i]);
    }
}

void clear() 
{
    for(int i = 0;i < 1000001;++i)
    {
     visited[i]=0;
     adj[i].clear();
    }
}

int main()
{ clear(); 
    int nodes,edges,x,y,cC=0;
    cin >> nodes;                     
    cin >> edges;                       
    for(int i = 0;i < edges;++i)
    {
     cin >> x >> y;     
     adj[x].push_back(y);                  
     adj[y].push_back(x);          
    }
                            
    for(int i = 1;i <= nodes;++i) 
    {
     if(visited[i] == 0)    
	  {
	      dfs(i);
	      cC++;
	  }
    }
    cout <<(cC-1) << endl;
    return 0;
}
