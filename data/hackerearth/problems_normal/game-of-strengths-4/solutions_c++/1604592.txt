#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

#define ll long long
#define MAX 100010
#define MOD 1000000007

int a[MAX];
int t, n;
ll res, tot;

int main()
{
    scanf("%d", &t);
    while (t--){
    	scanf("%d", &n);
    	for (int i = 0; i < n; i++)
    		scanf("%d", &a[i]);
    	sort(a, a + n);
    	res = tot = 0;
    	for (int i = 1; i < n; i++){
    		tot = (tot + 1LL*i*(a[i] - a[i - 1])) % MOD;
    		res = (res + tot) % MOD;
    	}
    	printf("%lld\n", (res * a[n - 1]) % MOD);
    }
    return 0;
}
