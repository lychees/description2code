#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;
struct Point{
	string name;
	int p;
};
bool compare(const struct Point &a, const struct Point &b)
{
	return a.p > b.p;
}
int main()
{
    //cout << "Hello World!" << endl;
    int n;
    cin>>n;
    vector<Point> v;
    int k=0;
    for(int i=0;i<n;i++)
    {
    	//string a;
    	//cin>>a;
    	//int x;
    	//cin>>x;
    	Point y;
    	cin>>y.name;
    	cin>>y.p;
    	int j;
    	for( j=0;j<k;j++)
    	{
    		if(v.at(j).name == y.name)
    		{
    			v.at(j).p = v.at(j).p + y.p;
    			break;
    		}
    	}
    	if(j==k)
    	{
    		v.push_back(y);
    		k++;
    	}
    	sort(v.begin(),v.end(),compare);
    }
    cout<<v.at(0).name;
    
    
    return 0;
}
