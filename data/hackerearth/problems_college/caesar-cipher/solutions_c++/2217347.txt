#include<bits/stdc++.h>
using namespace std;
int main()
{
    string cap="ABCDEFGHIJKLMNOPQRSTUVWXYZ",smal="abcdefghijklmnopqrstuvwxyz",inp;
    int t,i,k,temp;
    char ch;
    cin>>t;
    while(t--)
    {
        cin>>k;
        cin>>inp;
        for(i=0;i<inp.size();i++)
        {
            ch=inp[i];
            if(ch>96 && ch<123)
            {
                temp=ch-97;
                cout<<smal[(temp+k)%26];
            }
            else
            {
                temp=ch-65;
                cout<<cap[(temp+k)%26];
            }
        }
        cout<<endl;
    }
}
