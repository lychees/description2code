#include <iostream>
using namespace std;
int comb(int  n,int k )
{
    if (k > n) return 0;
    if (k * 2 > n) k = n-k;
    if (k == 0) return 1;

   int result = n;
    for( int i = 2; i <= k; ++i ) {
        result = (result*(n-i+1));
        result /=i;
    }
    return result;
}

int main()
{
   // cout << "Hello World!" << endl;
    int tc ;
    int n ;
    cin>>tc ;
    while(tc--)
    {
    	cin>>n ;
    	cout<<comb(n,2)<<endl ; ;
    }
    
    return 0;
}
