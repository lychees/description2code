#include<iostream>
using namespace std;
int main()
{
    int t,n,i,k,c,d,e;
    cin>>t;
    while(t--)
    {
        cin>>n>>k;
        if(k<=81)
        {
            e=0;
            c=(k-1)/9;
            d=(k-1)%9;
          for(i=1;i<=d;i++)
            {
                if(i==c+1)
                {
                    e=e+2;
                }
                else
                {
                    e=e+1;
                }
            }
            for(i=0;i<n;i++)
            {
                if(i%2==0)
                cout<<c+1;
                else
                cout<<e;
            }
            cout<<endl;
        }
        else
        {
            cout<<-1<<endl;
        }
    }
}
