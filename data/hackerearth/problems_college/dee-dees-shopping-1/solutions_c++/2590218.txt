#include <iostream>
using namespace std;

int main()
{
    int i,j,T,N;
    cin>>T;
    
    while(T!=0)
    {
    	int f = -1;
    	cin>>N;
    	 j = N%8;
    	int k = N/8;
    	i=0;
    	while(i!=k+1)
    	{
    		if(j==0)
    		{
    			f = 0;
    			break;
    		}
    		j = N%8;
    		j = (j+8*i)%6;   		
    		i++;
    		
    		
    	}
    	int p = 0;
    	p = N/8-(i-1);
    	p+= ((N%8 )+(i-1)*8)/6; 
    	if(f==0)
    	cout<<p<<endl;
    	else
    	cout<<"-1"<<endl;
    	T--;
    }
    return 0;
}
