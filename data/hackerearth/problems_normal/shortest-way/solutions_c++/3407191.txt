#include <bits/stdc++.h>

using namespace std;

int main()
{
    long long int t,a,b,x,y,c;
    cin>>t;
    while(t--){
        cin>>a>>b;
        c = 0;
        while(a>1 || b>1){
            if(a>b){
                x = a/b;
                a = a - x*b;
                c = c+x;
            }
            else{
                y = b/a;
                b = b - y*a;
                c = c+y;
            }
        }
        if(c>0){
            c--;
        }
        cout<<c<<endl;
    }
    //cout << "Hello world!" << endl;
    return 0;
}
