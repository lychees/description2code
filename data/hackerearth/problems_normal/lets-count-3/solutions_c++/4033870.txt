#include<bits/stdc++.h>
//zone
using namespace std;
int main()
{
    int t;
   	cin>>t;
   	while(t--) {
   		long long int a[3];
   		long long int counti= 0;
   		long long int divisor = 0;
   		cin>>a[0]>>a[1]>>a[2];
   		sort(a,a+3);
   		long long int i;
   		for(i = 1; i*i <= a[0]; i++) {
   			if(a[0]%i == 0) {
   				if(a[1]%i == 0 && a[2]%i == 0)
   					counti++;
   				if(a[0]/i != i) {
   					long long int divisor = a[0]/i;
   					if(a[1]%divisor == 0 && a[2]%divisor == 0)
   						counti++;
   				}
   			}
   		}
   		cout<<counti<<endl;
   	}
    return 0;
}
