#include<iostream>
using namespace std;
#define mod 1000000007
int  gcd ( int a, int b )
{
  int c;
  while ( a != 0 ) {
     c = a; a = b%a;  b = c;
  }
  return b;
}
int main()
{
    int T,a,b,c;
    cin>>T;
    while(T--)
    {
        cin>>a>>b>>c;
        long long int len = gcd(a,gcd(b,c));
        long long int ans = ((((a/len)%mod) * ((b/len)%mod))%mod);
        ans =  ((((ans)%mod) * ((c/len)%mod))%mod);
        cout<<len<<" "<<ans<<endl;

    }
    return 0;
}
