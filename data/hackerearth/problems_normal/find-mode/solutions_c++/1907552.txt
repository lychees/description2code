    #include <iostream>
    #include <vector>
    using namespace std;
    int a[1000007]={0}; 
     
    int main ()
    {
      int t,i,j,k,maxf,max,n,temp;
      
      scanf("%d",&t);
      while(t--)
      {
      	 scanf("%d",&n);
      	maxf=0;
      	max=0;
      	for(i=0;i<n;i++)
      	{
      		  scanf("%d",&temp);
      		a[temp]++;
      		
      		if(maxf<a[temp])
      		maxf=a[temp];
      		
      		if(max<temp)
      		max=temp;
      		
      		
      	}
      	
      	
      	for(i=max;i>=0;i--)
      	{
      		if(a[i]==maxf)
      		printf("%d ",i);
      		
      		a[i]=0;
      		
      		
      	}
      	
      	
      	cout<<endl;
      	
      	
      }
      
      
     
      return 0;
    }
