# Get input
T = int(raw_input())
Ts = []
for t in xrange(T):
    Ts.append(int(raw_input()))

primes = [2, 3, 5, 7]
minimals = {}
for prime in primes:
    minimals[prime] = 1

# Compute minimal number of single digit primes


def getMinimals(number):
    if number < 2:
        return -1
    if number in minimals:
        return minimals[number]

    minNum = -1
    for prime in primes:
        result = getMinimals(number - prime)
        if result == 0:
            result = -1

        if result != -1 and minNum != -1:
            minNum = min(result, minNum)
        elif result != -1 and minNum == -1:
            minNum = result

    minimals[number] = minNum + 1 if minNum != -1 else -1
    return minimals[number]

# Precompute


def preCompute():
    for n in xrange(max(Ts) + 1):
        getMinimals(n)


preCompute()
for t in Ts:
    print getMinimals(t)
