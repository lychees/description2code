#include <iostream>
using namespace std;

int main()
{
	int t,n,i;
	string str;
	
	cin>>t;
	while(t--){
		cin>>str;
		cin>>n;
		
		int zc=0;
		n = n*(n+1)/2;
		for(i=0;i<str.size();i++){
			if(str[i]=='0'){
				zc++;
			}
		}
		
		if(n%2==0){
			cout<<(str.size()-zc)<<endl;
		}else{
			cout<<zc<<endl;
		}
	}
	return 0;
}
