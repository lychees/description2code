#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
	long n,sum=0;
	cin>>n;
    while(n!=0){
        long ld=n%10;
    sum+=ld;
    n/=10;
    }
    n=sum;
    int count=0;
    for(int a=1;a<=n;a++)
               {
                if(n%a==0)
                   {
                  count++;
                   }
               }
       if(count==2)
      cout << "YES"<<endl;
  else
      cout << "NO"<<endl;
    }return 0;
}
