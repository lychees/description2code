#include<stdio.h>
int main()
{
	int t,k;

	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&k);
		char arr[50];
		arr[0] = '\0';
		scanf("%s",&arr[0]);
		for(int i=0;arr[i]!='\0';++i)
		{
            int x=(int)arr[i];
			x -= 65;
			x = (x-k);
			if(x<0)
                x+=26;
			x += 65;
            arr[i]=x;
		}
		printf("%s\n",arr);
	}
	return 0;
}
