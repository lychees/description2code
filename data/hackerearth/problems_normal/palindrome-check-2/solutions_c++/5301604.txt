#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    char s[101];
    cin.getline(s,101);
    int l=strlen(s)-1;
    for(int i=0;i<=l/2;i++){
    	if(s[i]!=s[l-i]){
    		l=200;
    		break;
    	}
    }
    if(l!=200)
    	cout<<"YES";
    else
    	cout<<"NO";
    return 0;
}
