#include <iostream>
#include <map>
#include <stdio.h>
#include <string.h>
using namespace std;
int main() 
{
	long long n;
	scanf("%lld",&n);
	map< int, map<string,int> > mp2;
	map<int, map<string,int> >::reverse_iterator it2;
	while(n--)
	{
		
		string s1;
		cin>>s1;
		int x;
		cin>>x;
		mp2[x][s1]++;
	}
	long long i=0;
	for (it2=mp2.rbegin(); it2!=mp2.rend(); it2++)
	{
		map<string, int> &internal_map = it2->second;
    	for (map<string, int>::iterator it = internal_map.begin(); it != internal_map.end(); ++it)
        {
        	for(i=0;i<it->second;i++)
        		cout << it->first<<" " << it2->first<<"\n";
        }
   	}
    
	return 0;
}
