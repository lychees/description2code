const int N = 111;

vi g[N];
int vis[N], match[N];
int n;
int h[N], w[N];


void add( int a, int b ){
	g[a].pb(b);
}


int dfs( int x ){
	if( vis[x] ) return 0;
	vis[x] = 1;
	for(int i=0;i<g[x].size();++i)
	{
		int y = g[x][i];
		if( match[y] == -1 || dfs(match[y]) ){
			match[y] = x;
			return 1;
		}
	}
	return 0;
}

int main()
{
	//ios::sync_with_stdio(0);
	int t; scanf("%d",&t);
	while(t--){
		scanf("%d",&n);
		for(int i=0;i<2*n;++i) g[i].clear();
		
		for(int i=0;i<n;++i) scanf("%d",w+i);
		for(int i=0;i<n;++i) scanf("%d",h+i);
		sort(h,h+n); reverse(h,h+n);
		ll pot = h[0]*w[0];
		for(int i=1;i<n;++i) for(int j=1;j<n;++j) if( h[i]*w[j] < pot ) add(i,j+n);
		int ans = 0;
		memset(match,-1,sizeof match);
		for(int i=1;i<n;++i){
			memset(vis,0,sizeof vis);
			ans+=dfs(i);
		}
	//	printf("%d\n",ans);
		
		if( ans == n-1 ) puts("YES");
		else puts("NO");
	}
	return 0;
}
