#include<bits/stdc++.h>
using namespace std;
int is_prime(long long int sum)
{
	int flag=1;
	for(int i=2;i<sqrt(sum);i++)
	{
		if(sum%i==0)
		{
			flag=0;
			break;
		}
	}
	return flag;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int a=0,b=1,c=0;
		int n;
		cin>>n;
		long long int sum=0;
		while(c<=n)
		{
			c=a+b;
			a=b;
			b=c;
			int d=a+b;
			for(int i = c+1;i<d;i++)
			{
				if(i<n)
				sum+=i;
				else break;
			}
		}
		if(sum%2==0)
		{
			cout<<"Tac\n";
		}
		else if(is_prime(sum))
		{
			cout<<"Tic\n";
		}
		else
		{
			cout<<"Toe\n";
		}
	}
	
}
