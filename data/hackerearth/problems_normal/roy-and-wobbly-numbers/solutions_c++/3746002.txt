#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    int t,n,k;
    cin>>t;
    while(t--)
    {
    	cin>>n>>k;
    	int first=ceil(k/9.0);
    	int sec=k%9;
    	if(sec==0)
    	sec=9;
    	int second=sec;
    	if(first>=second)
    	second--;
    	if(k>81)
    	cout<<"-1";
    	else
    	{
    		for(int i=1,j=0;i<=n;i++)
    		{
    			if(j==0)
    			{
    				cout<<first;
    				j=1;
    			}
    			else if(j==1)
    			{
    				cout<<second;
    				j=0;
    			}
    		}
    	}
    	cout<<endl;
    }
    return 0;
}
