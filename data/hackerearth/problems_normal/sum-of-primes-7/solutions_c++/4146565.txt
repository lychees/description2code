#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <cstring>
#include <iterator>
#include <stdio.h>
#define lld long long int
#define max_int 1000000
#define pb(x) push_back(x)
#define mp(x,y) make_pair(x,y)
using namespace std;

int main()
{
	lld n,i,j,l,r;
	cin>>n;
	bool mark[max_int+2];
	lld sum[max_int+1];
	sum[0]=0;sum[1]=0;
	memset(mark,1,sizeof(mark));
	mark[0]=0;mark[1]=0;mark[max_int]=0;
	for(i=2;i*i<max_int;i++)
	{
		if(mark[i])
		{
			for(j=i*i;j<max_int;j=j+i)
			{
				mark[j]=0;
			}
		}
	}
	for(i=2;i<max_int;i++)
	{
		sum[i]=mark[i]==1?sum[i-1]+i:sum[i-1];
	}
	sum[max_int]=sum[max_int-1];
	while(n--)
	{
		cin>>l>>r;
		cout << sum[r]-sum[l-1] << endl;
	}
    
    return 0;
}
