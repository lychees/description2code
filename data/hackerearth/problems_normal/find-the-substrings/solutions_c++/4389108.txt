struct myCompare {
	bool operator() (const pii &l, const pii &r) const {
		return l.first < r.first;
	}
};


const int MAXN = 1e6 + 3;
char s[MAXN];
int n;
char a = 'a', b = 'b', c = 'c';
int wo[MAXN][3];
void solve() {
	int T;
	cin >> T; cin.ignore();
	while (T--) {
		ll res = 1;
		gets(s);
		n = strlen(s);
		wo[0][0] = wo[0][1] = wo[0][2] = 1;
		wo[0][s[0] - 'a'] = 0;
		forl(i, 1, n) {
			switch (s[i]) {
				case 'a':
					res += max(wo[i - 1][1], wo[i - 1][2]) + 1;
					wo[i][0] = 0; wo[i][1] = wo[i - 1][1] + 1; wo[i][2] = wo[i - 1][2] + 1;
					break;
				case 'b':
					res += max(wo[i - 1][0], wo[i - 1][2]) + 1;
					wo[i][1] = 0; wo[i][0] = wo[i - 1][0] + 1; wo[i][2] = wo[i - 1][2] + 1;
					break;
				case 'c':
					res += max(wo[i - 1][0], wo[i - 1][1]) + 1;
					wo[i][2] = 0; wo[i][0] = wo[i - 1][0] + 1; wo[i][1] = wo[i - 1][1] + 1;
			}
		}
		cout << res << endl;
	}
}
