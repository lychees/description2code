    #include <bits/stdc++.h>
     
    using namespace std;
     
    int main()	{
    	int t;
    	long long int m;
     
    	scanf("%d", &t);
    	while(t--)	{
    		scanf("%lld", &m);
    		long long int count = 0;
    		long long int five = 5;
    		
    		while(m >= five)	{
    			count += m/five;
    			five *= 5;
    		}
     
    		printf("%lld\n", count);
    	}
     
    	return 0;	
    }
