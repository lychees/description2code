#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long int c=0,i,j,n;
    	cin>>n;
    	long int arr[n];
    	for(i=0;i<n;i++)cin>>arr[i];
    	for(i=0;i<n;i++)
    	for(j=i;j<n;j++)
    	{
    		if(arr[i]==arr[j])c++;
    	}
    	cout<<c<<endl;
    }
    return 0;
}
