#include <stdio.h>

int a [15];
int n;
int max;

void bi (int index, int s1, int s2, int s3) {
	
	if (index > n){
		return;
	}
	if (s1 < s2 || s3 > s2) {
		//return;
	}
	
	if (index == n && s1 >= s2 && s2 >= s3 && s1 > 0 && s2 > 0 && s3 > 0) {
		//printf ("%d %d %d\n", s1, s2, s3);
		if (s1 < max) {
			max = s1;
		}
	}
	bi (index + 1, s1 + a [index], s2, s3);
	bi (index + 1, s1 , s2 + a [index], s3);
	bi (index + 1, s1 , s2, s3 + a [index]);
}

int main ()
{
	int i;
	
	max = 100000001;
	scanf ("%d", &n);
	
	for (i = 0; i < n; i++) {
		scanf ("%d", & a [i]);
	}
	
	bi (0, 0, 0, 0);
	
	printf ("%d\n", max);
	
	return 0;
}
