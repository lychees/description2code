#include<bits/stdc++.h>
#define mp make_pair
#define endl '\n'
#define pi pair<int,int>
#define int long long
#define mod (1000*1000*1000 +7)
using namespace std;
vector<pi> v;
int dp[100005][5],a[100005],b[100005],c[100005];
 main()
{
    int t,n;
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(int i=0;i<n;++i)
            cin>>a[i]>>b[i]>>c[i];

        for(int i=n-1;i>=0;--i)
        {
            dp[i][1]=a[i]+min(dp[i+1][0],dp[i+1][2]);
            dp[i][2]=b[i]+min(dp[i+1][0],dp[i+1][1]);
            dp[i][0]=c[i]+min(dp[i+1][1],dp[i+1][2]);
        }
        cout<<min(min(dp[0][0],dp[0][1]),dp[0][2])<<endl;

        for(int i=n-1;i>=0;--i)
        {
            dp[i][1]=0;
            dp[i][2]=0;
            dp[i][0]=0;
        }
    }
    return 0;
}
