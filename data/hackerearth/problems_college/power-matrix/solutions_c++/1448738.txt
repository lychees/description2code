#include <iostream>
using namespace std;
#define ll long long
#define mod 1000000007
#define SIZE 51
ll x[51][51];
ll m;
void one(ll x[][SIZE])
{
    for (int i = 0; i < m; i++)
        for (int j = 0; j < m; j++)
            x[i][j] = (i == j);
}
void mul(ll x[SIZE][SIZE], ll z[SIZE][SIZE])
{
    ll res[SIZE][SIZE] = {{0}};
 
    for (int i = 0; i < m; i++)
        for (int j = 0; j < m; j++)
            for (int k = 0; k < m; k++)
            {
                res[i][j] = (res[i][j]+((x[i][k])%mod)*((z[k][j])%mod))%mod;
            }
 
    for (int i = 0; i < m; i++)
        for (int j = 0; j < m; j++)
            x[i][j] = res[i][j];
}
void power(ll x[SIZE][SIZE], ll n, ll res[SIZE][SIZE])
{
    one(res);
 
    while (n) {
        if (n % 2 == 0)
        {
            mul(x, x);
            n /= 2;
        }
        else {
            mul(res, x);
            n--;
        }
    }
}
int main() 
{
	ll t,n;
	cin>>t;
	while(t--) {
		cin>>m>>n;
		ll i,j;
		ll a[SIZE][SIZE],b[SIZE][SIZE];
		for(i = 0; i < m; i++) {
			for(j = 0; j < m; j++) {
				cin>>a[i][j];
			}
		}
		power(a,n,b);
		for(i = 0; i < m; i++) {
			for(j = 0; j < m; j++) {
				cout<<b[i][j]<<" ";
			}
			cout<<endl;
		}
	}
	return 0;
}
