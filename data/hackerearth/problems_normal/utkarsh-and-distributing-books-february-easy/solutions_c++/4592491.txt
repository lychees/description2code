#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	int n,utkarsh,saharsh;
	while(t--)
	{
		utkarsh=1000,saharsh=0;
		cin>>n;
		int *a = new int[n];
		for(int i=0;i<n;i++)
		{
			cin>>a[i];
			if(a[i]<utkarsh)
				utkarsh=a[i];
			saharsh=saharsh+a[i]-1;
		}
		utkarsh--;
		cout<<utkarsh<<" "<<saharsh<<endl;
	}
    return 0;
}
