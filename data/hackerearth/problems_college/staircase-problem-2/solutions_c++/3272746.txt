#include <iostream>
using namespace std;
int arr[100001];
void pre()
{
	arr[1]=1;
	arr[2]=1;
	int ans=2;
	int cnt=0;
	for(int i=3;i<=100000;)
	 if(cnt<=ans){
	 arr[i]=ans;
	 cnt++;
	 i++;
	 }
	 else{
	 	ans+=1;
	 	cnt=0;
	 }
}
int main()
{
    int t;
    cin >> t;
    pre();
    while(t--){
    	int n;
    	cin >> n;
    	cout << arr[n] << endl;
    }
    return 0;
}
