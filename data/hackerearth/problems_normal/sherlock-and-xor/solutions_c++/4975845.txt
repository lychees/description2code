#include <iostream>
using namespace std;
#include<cassert>
int main()
{
    int t;
    cin>>t;
    assert(t>=1&&t<=10);
    while(t--)
    {
    	long long int n,a[100000],i,j,evn=0,odd=0;
    	scanf("%llu",&n);
    	assert(n>=1&&n<=100000);
    	for(i=0;i<n;i++)
    		scanf("%llu",&a[i]);
    	for(i=0;i<n;i++)
    	{
    		if(a[i]%2==0)
    			evn++;
    		else	
    			odd++;
    	}
    	printf("%llu\n",evn*odd);
    }
}
