#include <bits/stdc++.h>
using namespace std;

int n;
int TEST_NO;
int n1, n2;
const int MAXN = 1000006;
int loser[MAXN];
void precompute() {
	int d = 0;
	for (int i = 0; i < MAXN; ++i) {
		if(loser[i]) continue;
		loser[i] = i + d;
                if(i + d < MAXN) loser[i + d] = i;
		d++;
	}
}
void preprocess() {
	if(n1 > n2) swap(n1, n2);
}
void solve() {
	if(n2 == loser[n1]) puts("Don't Play");
	else puts("Play");
}
int main() {
	precompute();
	int t;
	cin>>t;
	while(t--) {
		cin>>n1>>n2;
		if( n1 > n2)
        {
            int temp = n2;
            n2 = n1;
            n1 = temp;
        }
		solve();
	}
	return 0;
}
