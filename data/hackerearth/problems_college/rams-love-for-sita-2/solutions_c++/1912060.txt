#include<bits/stdc++.h>
using namespace std;
int main()
{
int t;
cin>>t;
for(int i=0;i<t;i++)
{
unsigned n,c,m,sum=0,x=0,new_choc=1,wrap;
cin>>n>>c>>m;
sum=n/c;
x=sum;
if(sum<m)
cout<<sum<<endl;
else
{
while(new_choc>0)
{
new_choc=x/m;
wrap=x/m+x%m;
sum=sum+new_choc;
x=wrap;
}
cout<<sum<<endl;
}
}
return 0;
}
