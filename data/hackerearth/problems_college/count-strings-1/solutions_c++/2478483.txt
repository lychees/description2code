#include <iostream>
#include<string>
using namespace std;
long long mod=1000000007;
int main()
{
	int t;
	cin>>t;
	string s1,s2;
	long long ans=1;
	int s11[27]={0};
	while(t--){
		ans=1;
		cin>>s1>>s2;
		for(int i=0;i<26;i++){
			s11[i]=0;
		}
		for(int i=0;i<s1.length();i++)
			s11[s1[i]-'a']++;
		for(int i=0;i<s2.length();i++){
			ans=(ans*s11[s2[i]-'a'])%mod;
			s11[s2[i]-'a']--;
		}
		cout<<ans<<endl;
	}
	
    //cout << "Hello World!" << endl;
    return 0;
}
