#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n; cin>>n;
    map<string, int> cities;
    string city;
    for(int i=0; i<n; i++) {
    	cin>>city;
    	cities.insert(make_pair(city, i));
    }
    int arr[n][n];
    for(int i=0; i<n; i++) {
    	for(int j=0; j<n; j++) {
    		cin>>arr[i][j];
    	}
    }
    int p; cin>>p;
    long cost = 0;
    int curr = 0, i;
    for(int x=0; x<p; x++) {
    	i = curr;
    	string dest; cin>>dest;
    	int j = cities[dest];
    	cost += arr[i][j];
    	curr = j;
    }
    cout << cost << "\n";
    
    return 0;
}
