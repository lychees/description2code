void solve(){
    int N;
    cin>>N;
    vector<ll>x(N);
    rep(i,0,N)cin>>x[i];
    vector<ll>ans(N);
    rep(i,0,N){
        ll maxpos=1;
        rep(j,0,i){
            if(abs(x[j])<abs(x[i]) && ((x[i]>0 && x[j]<0)||(x[i]<0 && x[j]>0))){
                maxpos=max(maxpos,ans[j]+1);
            }
        }
        ans[i]=maxpos;
    }
    debug(ans);
    cout<<*max_element(all(ans));
}

int main(){
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);
	int t=1;
//	cin>>t;
	while(t--){
		solve();
	}
	return 0;
}
