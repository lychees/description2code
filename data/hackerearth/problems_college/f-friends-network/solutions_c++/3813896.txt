#include<bits/stdc++.h>
using namespace std;
long long int t,n,i,j,k,l,s,a[1001][1001];
int main()
{
    cin>>t;
    while(t--)
    {
    	scanf("%lld",&n);
    	for(i=1;i<=n;i++)
    	{
    		for(j=1;j<=n;j++)
    		{
    			scanf("%lld",&a[i][j]);
    		}
    	}
    	s=0;
    	for(i=1;i<=n;i++)
    	{
    		for(j=1;j<=n;j++)
    		{
    			if(a[i][j]==1)
    			{
    				for(k=1;k<=n;k++)
    				{
    					if(a[j][k]==1)
    					{
    						if(a[k][i]==1)
    						s=1;
    					}
    					if(s==1)
    					break;
    				}
    			}
    			if(s==1)
    					break;
    		}
    		if(s==1)
    					break;
    	}
    	if(s==1)
    	cout<<"Aye\n";
    	else 
    	cout<<"Nae\n";
    }
    return 0;
}
