#include <bits/stdc++.h>
using namespace std;
int main()
{
    int i,t,A[91];
    char S[1000000];
    cin>>t;
    while(t>0)
    {
     for(i=0;i<91;i++)
       A[i]=0;
     cin>>S;	
     for(i=0;i<strlen(S);i++)
        A[S[i]]++;
     int flag=1;
     //cout<<"\n";
     //for(i=65;i<=90;i++)
     //   cout<<A[i]<<" ";
    for(i=65;i<91;i++)
    {
     if(A[i]!=(91-i) && A[i]!=0 )
       flag=0;
    }
    if(flag==1)
      	cout<<"Yes\n";
    else
    	cout<<"No\n";
    t--;	
    }
    return 0;
}
