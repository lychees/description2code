#include<iostream>
#include<bits/stdc++.h>
using namespace std;

string calculate_minute(int m) {
	string s2;
	switch(m) {
		case 0: s2 = ""; break; 
		case 1: s2 = "one"; break;
		case 2: s2 = "two"; break;
		case 3: s2 = "three"; break;
		case 4: s2 = "four"; break;
		case 5: s2 = "five"; break;
		case 6: s2 = "six"; break;
		case 7: s2 = "seven"; break;
		case 8: s2 = "eight"; break;
		case 9: s2 = "nine"; break;
		case 10: s2 = "ten"; break;
		case 11: s2 = "eleven"; break;
		case 12: s2 = "twelve"; break;
		case 13: s2 = "thirteen"; break;
		case 14: s2 = "fourteen"; break;
		case 15: s2 = "quarter"; break;
		case 16: s2 = "sixteen"; break;
		case 17: s2 = "seventeen"; break;
		case 18: s2 = "eighteen"; break;
		case 19: s2 = "ninteen"; break;
		case 20: s2 = "twenty"; break;
		case 21: s2 = "twenty one"; break;
		case 22: s2 = "twenty two"; break;
		case 23: s2 = "twenty three"; break;
		case 24: s2 = "twenty four"; break;
		case 25: s2 = "twenty five"; break;
		case 26: s2 = "twenty six"; break;
		case 27: s2 = "twenty seven"; break;
		case 28: s2 = "twenty eight"; break;
		case 29: s2 = "twenty nine"; break;
		case 30: s2 = "half"; break;
	}
	return s2;
}
string calculate_hour(int h) {
	string s1;
	switch(h) {
		case 1: s1 = "one"; break;
		case 2: s1 = "two"; break;
		case 3: s1 = "three"; break;
		case 4: s1 = "four"; break;
		case 5: s1 = "five"; break;
		case 6: s1 = "six"; break;
		case 7: s1 = "seven"; break;
		case 8: s1 = "eight"; break;
		case 9: s1 = "nine"; break;
		case 10: s1 = "ten"; break;
		case 11: s1 = "eleven"; break;
		case 12: s1 = "twelve"; break;	
	}
	return s1;		
}
int main() {
	int h,m;
	cin >> h >> m;
	string s1,s2;
	if(m <= 30) {
		s1 = calculate_hour(h);
		s2 = calculate_minute(m);
		if(m == 0) {
			cout << s1 << " o' clock" << endl;
		}
		else if(m == 1) {
			cout << s2 << " minute past " << s1 << endl;
		}
		else if(m == 15) {
			cout << s2 << " past " << s1 << endl;
		}
		else if(m == 30) {
			cout << s2 << " past " << s1 << endl;
		}
		else {
			cout << s2 << " minutes past " << s1 << endl;
		}
	}
	if(m > 30) {
		s1 = calculate_hour(h+1);
		s2 = calculate_minute(60-m);
		if(m == 45) {
			cout << s2 << " to " << s1 << endl;
		}
		else if(m == 59) {
			cout << s2 << " minute to " << s1 << endl;
		}
		else {
			cout << s2 << " minutes to " << s1 << endl;
		}
	}
	return 0;
}
