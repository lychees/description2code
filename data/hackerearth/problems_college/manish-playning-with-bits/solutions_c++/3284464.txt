#include <iostream>
#include <vector>
#include <cmath>

using namespace std;

int main(){
	int T;
	cin>>T;
	int n,r;
	vector<long long int> a(70);
	int i,j;
	while(T--){
		cin >> r >> n;
		
		if(r==1)
			cout<<n<<endl;
		
		else{
			for(i=0;i<r+n+1;i++)
				a[i]=0;

			
			a[r-1]=1;
			// for(int k=0;k<r+n+1;k++)
			// 		cout << a[k]<<" ";
			// 	cout << endl;
			for(i=r;i<n+r+1;i++){
				for(j=1;j<=r;j++){
					a[i]+=a[i-j];
				}
				// for(int k=0;k<r+n+1;k++)
				// 	cout << a[k]<<" ";
				// cout << endl;
			}

			cout<<a[r+n]<<endl;
		}
		
			
	}

	return 0;
}
