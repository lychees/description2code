number_of_input = raw_input()
for input in range(int(number_of_input)):
	price = "$"
	product_desc = raw_input()
	splitted_on_dollar = str.split(product_desc, "$", 1)
	length = len(splitted_on_dollar[1])
	desc_with_price = splitted_on_dollar[1]
	for char in range(length):
		if desc_with_price[char] == ' ':
			continue
		elif str.isdigit(desc_with_price[char]):
			if int(desc_with_price[char]) is 0 and len(price) is not 1:
				price += desc_with_price[char]
			elif int(desc_with_price[char]) is not 0:
				price += desc_with_price[char]
		else:
			break
	if len(price) is 1:
		price += "0"
	print price
