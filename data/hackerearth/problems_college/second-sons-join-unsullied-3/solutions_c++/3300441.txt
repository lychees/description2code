#define rf(i,a,b) for(i=a;i>=b;i--)
#define clr(x,a) memset(x,a,sizeof(x))
#define MAX 1000100
#define MOD 1000000007

int n,m;
char a[MAX],b[MAX];

int main(){
    int r,k,i,c=0,x=0,y=0,j,t,l,z,x1=0,y1=0;
    ll ans=0;string p;

    f(i,0,MAX){
        if(i&1)
            a[i]='U';
        else
            a[i]='S';
    }
    f(i,0,MAX){
        if(i&1)
            b[i]='S';
        else
            b[i]='U';
    }
    si(t);
    while(t--){
        x=0;y=0;l=0;r=0;
        cin>>p;
        f(i,0,p.size()){
            if(p[i]!=a[i])
                x++;
            if(p[i]!=b[i])
                y++;
            if(p[i]=='U')
                l++;
            else
                r++;
        }
        if(p.size()%2==0){
            if(l!=r)
                cout<<"-1"<<endl;
            else
                cout<<min(x/2,y/2)<<endl;
        }
        else{
            if(l!=r+1 && l!=r-1)
                cout<<"-1"<<endl;
            else if(l==r-1){
                cout<<x/2<<endl;
            }
            else
                cout<<y/2<<endl;
        }
    }

}
