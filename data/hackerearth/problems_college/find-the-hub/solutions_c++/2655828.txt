#include <iostream>
#include<algorithm>
#include<math.h>
using namespace std;
long long n,j,i,kd[1000000],a[1000000],x=0,c=0,q;

int main()
{
	
	
	int flag;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    for(i=0;i<n;i++)
    {
    	for(j=2;j<=sqrt(a[i]);j++)
    	{
    		if(a[i]%j==0)
    		{
    			flag=1;
    			break;
    		}
    		flag=0;
    	}
    	
    	if(flag==0)
    	{
    		kd[x]=a[i];
    		x++;
    	}
    	}
  x--;
    sort(kd,kd+x);
    q=kd[x-1];
    cout<<q;
    for(i=0;i<x;i++)
    	if(q==kd[i])
    		c++;
    cout<<endl<<c;
    return 0;
}
