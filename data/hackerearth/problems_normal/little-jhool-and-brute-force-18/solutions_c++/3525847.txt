#include <bits/stdc++.h>

using namespace std;

long int ans[105][105] = {0};

void fun()
{
    for (int i = 1;i < 105;i++)
    {
        for (int j = 1;j < 105;j++)
        {
            ans[i][j] = pow(i,3) + pow(j,3);
        }
    }
}

int main()
{
    fun();
    int tc;
    cin >> tc;

    while(tc--)
    {
        int num;
        map<int,int> pr1;
        //map<int,int> :: iterator it;
        vector<int> pr;
        vector<int> :: iterator it;

        cin >> num;

        for (int i = 1;i < 100;i++)
        {
            for (int j = 1;j < 100;j++)
            {
                if (ans[i][j] <= num)
                {
                    pr.push_back(ans[i][j]);
                    if (pr1[ans[i][j]] == 0)
                        pr1[ans[i][j]] = 1;
                    else if (pr1[ans[i][j]] == 1)
                        pr1[ans[i][j]] = 2;
                    else if (pr1[ans[i][j]] == 2)
                        pr1[ans[i][j]] = 3;
                }
            }
        }

        sort(pr.begin(),pr.end());

        long int max1 = -INFINITY;
        int sol = -1;

        for (int i = pr.size()-1;i >= 0;i--)
        {
            //cout << pr[i] << "  ";
            if (pr1[pr[i]] == 3)
            {
                sol = pr[i];
                break;
            }
        }

        cout << sol << endl;
    }

    return 0;
}
