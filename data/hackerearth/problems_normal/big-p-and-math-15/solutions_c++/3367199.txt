#include <iostream>
using namespace std;

int replace_5(int A) {
	int factor = 1;
	int mod_A=0;
	while(A) {
		if(A%10 == 6)
		mod_A += factor*5;
		else
		mod_A += factor*(A%10);
		factor*=10;
		A = A/10;
	}
	return mod_A;
}

int replace_6(int A) {
	int factor = 1;
	int mod_A=0;
	while(A) {
		if(A%10 == 5)
		mod_A += factor*6;
		else
		mod_A += factor*(A%10);
		factor*=10;
		A = A/10;
	}
	return mod_A;
}

int main()
{
	int A, B, min_A, max_A, min_B, max_B;
	cin >> A >> B;
	min_A = replace_5(A);
	min_B = replace_5(B);
	max_A = replace_6(A);
	max_B = replace_6(B);
	cout << min_A+min_B << " " << max_A+max_B;
    return 0;
}
