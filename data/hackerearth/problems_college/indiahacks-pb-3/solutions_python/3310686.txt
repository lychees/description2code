for _ in range(input()):
    n, k = map(int, raw_input().split())
    slips= map(int, raw_input().split())
    sums = [0] * n
    sums[0] = max(slips[0], 0)
    for i in range(1, n):
        if slips[i] < 0:
            sums[i] = sums[i-1]
        elif (i - k -1 < 0):
            sums[i] = max(sums[i-1], slips[i])
        else:
            sums[i] = max(sums[i-1], slips[i] + sums[i-k-1])
    print sums[n-1]
