#include <iostream>
#include <string.h>
using namespace std;

int main()
{
    int t,i,flag;
    char ch[100001];
    //bool flag;
    cin>>t;
    while(t--)
    {
    	cin>>ch;
    	flag=0;
    	if(ch[0]!='R') {cout<<"NO"<<endl;flag=1;}
    	else
    	{
	    	for(i=0;i<strlen(ch);i++)
	    	{
	    		if(ch[i]!='R' && ch[i]!='Y')
	    		{cout<<"NO"<<endl;flag=1;break;}
	    		else if(ch[i]=='Y' && ch[i+1]=='Y' && ch[i+2]=='Y' && i<strlen(ch)-2)
	    		{cout<<"NO"<<endl;flag=1;break;}
	    	}
	    }
    	if(flag==0)cout<<"YES"<<endl;
    }
    return 0;
}
