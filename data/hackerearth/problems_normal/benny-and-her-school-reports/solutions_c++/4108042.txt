#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	long int n,m,x,a[100000],sum=0,req,i;
    	cin>>n>>m>>x;
    	for(i=0;i<n;i++){
    		cin>>a[i];
    		sum=sum+a[i];
    	}
    	req=(n+1)*x-sum;
    	if(req<=1){
    		cout<<"1\n";
    	}
    	else{
    		if(req>m)
    			cout<<"Impossible\n";
    		else
    			cout<<req<<"\n";
    	}
    	
    }
    return 0;
}
