#include <iostream>
#include <set>
using namespace std;
#define ll long long
int main()
{
	ll t;
	cin>>t;
	while(t--){
		ll n;
		cin>>n;
		ll arr[n];
		set<ll> visited;
		for(ll i = 0; i < n; i++)
			cin>>arr[i];
		visited.insert(0);
		ll ext = arr[0];
		while(ext != -1 && visited.find(ext) == visited.end()){
			visited.insert(ext);
			ext = arr[ext];
		}
		if(ext == -1){
			cout<<"Win\n";
		}
		else
		cout<<"Lose\n";
	}
    return 0;
}
