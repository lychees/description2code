#include<iostream>
#define ll long long
using namespace std;
int main()
{
//	freopen("in.txt","r",stdin);
//	freopen("out.txt","w",stdout);
	ios_base::sync_with_stdio(false);
	ll t,i,count,temp,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		if(n==0)
		cout<<"0\n";
		if(n==1)
		cout<<"1\n";
		else if(n==2)
		cout<<"2\n";
		else
		{
			for(i=2;;i++)
			{	
				temp=(i*(i+1))/2;
				temp*=2;
				temp-=i;
				if(temp==n)
				{
					cout<<(2*i)-1<<"\n";
					break;
				}
				if(n<temp)
				{
					if(n<=(temp-i))
						cout<<(2*i)-2<<"\n";
					else
					cout<<(2*i)-1<<"\n";
					break;
				}	
			}
		}
	}
return 0;
}
