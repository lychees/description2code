#include <bits/stdc++.h>
using namespace std;
#define scan(x) scanf("%d",&x)
#define pb push_back
int main()
{
    int t;
    scan(t);
    while(t--)
    {
    	int n;
    	scan(n);
    	vector<vector<int>> v(n);
    	int t1,t2,temp;
    	for(int i=0;i<n;i++)
    	{
    		scan(t1);
    		scan(t2);
    		temp=t1+t2;
    		v[i].pb(temp);
    		v[i].pb(t1);
    		v[i].pb(t2);
    	}
    	sort(v.rbegin(),v.rend());
    	long long int ans=0;
    	for(int i=0;i<n;i++)
    	{
    		if(i==0 || i==1) ans+=v[i][1];
    		else ans-=v[i][2];
    	}
    	cout<<ans<<"\n";
    }
    return 0;
}
