#include <bits/stdc++.h>
using namespace std;

int main()
{
   int t,n,i,x;
   scanf("%d",&t);
   while(t--)
   {
   	scanf("%d",&n);
   	double ans=n;
   	for(i=2;i<=sqrt(n);i++)
   	{
   		if(n%i==0)
   		{
   			while(n%i==0)
   			n/=i;
   			ans=ans-ans/i;
   		}
   	}
   	if(n>1)
   	ans=ans-ans/n;
   	
   	int anss=ans;
 //  	printf("%d ",anss);
   	if(anss==1)
   	printf("Incorrect\n");
   	else if(anss==2)
   	printf("Correct\n");
   	
   	else if(anss%2==0)
   	printf("Incorrect\n");
   	else
   	{
   		for(i=3;i<=sqrt(anss);i+=2)
   		{
   			if(anss%i==0)
   			break;
   		}
   		if(i>sqrt(anss))
   		printf("Correct\n");
   		else
   	    printf("Incorrect");
   	}
   }
    return 0;
}
