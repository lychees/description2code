#include <bits/stdc++.h>

using namespace std;

bool prime[1000010];

void sieve()
{
    for(int i=4;i<=1000000;i+=2)
    prime[i]=true;
    for(int i=3;i<=1000;i+=2)
    {
        if(prime[i]==false)
        for(int j=i*i;j<=1000000;j+=2*i)
        prime[j]=true;
    }
}

int a[100010];

int main() {
  int n;
  sieve();
  scanf("%d",&n);
  for(int i=0;i<n;i++)
  {
      scanf("%d",&a[i]);
  }
 int ind,ans=-111;
 for(int i=0;i<n;i++)
 if(prime[a[i]]==false and a[i]>ans)
 {
     ans=a[i];
     ind=i;
 }
 int cnt=0;
 for(int i=0;i<n;i++)
 if(a[i]==a[ind])cnt++;
 printf("%d\n%d\n",ans,cnt);
    return 0;
}
