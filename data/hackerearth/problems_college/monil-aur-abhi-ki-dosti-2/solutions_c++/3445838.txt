#include <bits/stdc++.h>
 
#define fi first
#define se second
#define pb push_back
#define mp make_pair
#define pii pair<int,int>
#define pll pair<ll,ll>
#define MOD 1000000007
#define ll long long int
#define MAXINT 2000000000
#define SET(x,y) memset(x,y,sizeof(x));
 
using namespace std;

int AC[1000],PC[1000];
char p[100005],s[100005];

int main() {
	int t,i,l,ld,j;
	cin>>t;
	while(t--) {
		cin>>p;
		cin>>s;
		l=strlen(p);
		for(i='a';i<='z';i++)	AC[i]=PC[i]=0;
		for(i=0;i<l;i++)		AC[p[i]]++;
		ld=strlen(s);
		for(i=0;i<ld;i++) {
			if(i>=l)
				PC[s[i-l]]--;
			PC[s[i]]++;
			for(j='a';j<='z';j++)	if(PC[j]!=AC[j])	break;
			if(j>'z')	break;
		}
		if(i<ld)	cout<<"YES\n";
		else		cout<<"NO\n";
	}
	return 0;
}
