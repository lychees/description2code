#include<iostream>
#include<cmath>

using namespace std;

int main()
{
    int testcases;
    cin>>testcases;
    while(testcases>0)
    {
        long long n;
        long long m;
        long long x1;
        long long y1;
        long long x2;
        long long y2;
        long long p;
        cin>>n;
        cin>>m;
        cin>>x1;
        cin>>y1;
        cin>>x2;
        cin>>y2;
        cin>>p;
        long total;
        total=min(abs(x1-x2),n-abs(x1-x2))+min(abs(y1-y2),m-abs(y1-y2));
        total*=p;
        if(total<=1000)
        {
            cout<<total<<"\n";
            cout<<"You saved the group.\n";
        }
        else
         {
             cout<<total<<"\n";
             cout<<"Let go of the group.\n";
         }           
         testcases--;
    }
}
