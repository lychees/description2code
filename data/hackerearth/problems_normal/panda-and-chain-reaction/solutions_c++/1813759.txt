#include <iostream>
using namespace std;

unsigned long long n,x;

int main()
{
	unsigned long long t,n,x,i,j,ans[1000003];
	ans[0]=1;
	for(j=1;j<1000003;j++)
	{
		ans[j]=(ans[j-1]*j)%1000003;
	}
	cin>>t;
	for(i=0;i<t;i++)
	{
		cin>>n>>x;
		x%=1000003;
		if(n>=1000003)
		{
			cout<<"0"<<endl;
		}
		else
		{
			cout<<(ans[n]*x)%1000003<<endl;
		}
	}
    return 0;
}
