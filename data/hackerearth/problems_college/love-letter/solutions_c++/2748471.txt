#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while (t--)
    {
    	int n;
    	long long int k;
    	cin >> n >> k;
    	long long int num = pow(2, n) - 1;
    	if ((k | num) == k)
    	{
    		cout << "YES" << endl;
    	}
    	else
    	{
    		cout << "NO" << endl;
    	}
    }
    return 0;
}
