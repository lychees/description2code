#include<iostream>
#include<cstdio>
#include<vector>
#include<stack>
#include<algorithm>
using namespace std;

vector<int> v[10001];
long long int a[10001],vs[10001],s[10001],low[10001],ind[10001];
int st[1000],top;
int index;



long long int maxan,p;

void dfs(int i)
{
	int cur,j,f,w;
	vs[i]=1;
	ind[i]=index;
	low[i]=index;
	index++;
	
	st[top++]=i;
	
	for(j=0;j<v[i].size();j++)
	{
		w=v[i][j];
		for(f=top-1;f>=0;f--)
		{
			if(st[f]==w)
				break;
		}
		
		if(vs[v[i][j]]!=1)
		{
			dfs(v[i][j]);
			low[i]=min(low[i],low[v[i][j]]);
		}
		else if(f!=-1)
		{
			low[i]=min(low[i],ind[v[i][j]]);
		}
	}
	
	if(low[i]==ind[i])
	{
		j=st[--top];
		p=a[j];
		while(j!=i)
		{
			j=st[--top];
			p+=a[j];
		}
		
		if(p>maxan)
			maxan=p;
	}
}

int main()
{
	int t,n,m,u,b,i;
	scanf("%d",&t);
	while(t--)
	{
		index=0;
		maxan=0;
		scanf("%d %d",&n,&m);
//		while(st.size()!=0)
//			st.pop();
		top=0;
		for(i=0;i<n;i++)
		{
			scanf("%lld",&a[i]);
			v[i].clear();
			vs[i]=0;
			ind[i]=101;
			low[i]=101;
		}
		
		
		while(m--)
		{
			scanf("%d %d",&u,&b);
			v[u].push_back(b);
		}
		
		for(i=0;i<n;i++)
		{
			if(vs[i]==0)
			{
				dfs(i);
			}
		}
		printf("%lld\n",maxan);
	}
	return 0;
}
