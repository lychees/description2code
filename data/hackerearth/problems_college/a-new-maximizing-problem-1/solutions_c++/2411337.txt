#define MIN(a,b) a<b?a:b
using namespace std;

int main()
{
	CASE
	{
		ll n,i,a[100005],sum=0,sum1=0;
		cin>>n;
		for(i=1;i<=n;i++)
		cin>>a[i];
		sort(a+1,a+n+1);
		for(i=1;i<=n;i++)
		{
			if(i<=(n+1)/2)
			{
				sum+=(-1*a[i]);
			}
			else
			{
				sum+=a[i];
			}
			sum1+=abs(a[i]);
			sum%=1000000007;
			sum1%=1000000007;
		}
		int temp=0;
		if(n%2)
		{
			temp=2*(a[n/2+1]);
			temp+=sum;
			temp*=sum1;
			cout<<max((ll)abs(temp),sum1*sum)%1000000007<<endl;
			continue;
		}
		cout<<(sum1*sum)%1000000007<<endl;
	}
    return 0;
}
