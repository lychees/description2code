#include <bits/stdc++.h>
using namespace std;
bool comp(pair<long long int, char> a , pair<long long int , char> b){
    return a.first<b.first;
}

int main() {
    string s;
    cin>>s;
    int n = s.size();
    long long int a[26]= {0};
    vector< pair<long long int ,char> > v;
    for(long long int i=0 ; i<n; ){
        long long num = 0; 
        char ch; 
        ch = s[i];
        i++;
        while(i<n && '0'<= s[i] && '9'>= s[i]){
            num = num*(long long)10 +(long long)(s[i]-'0');
            i++;
        }
        a[ch -'a']+=num;
    }
    long long int st = 0;
    for (int i=0 ; i<26;i++){
        // cout<<a[i]<<" ";
        st += a[i];
        if(a[i]>0)
        v.push_back(make_pair(st, (char)(i+'a')));
    }
    int q;
    cin>>q;
    for(int i=0 ; i<q; i++){
        long long int x; 
        cin>>x;
        x -=1;
        if(x>st){
            cout<<-1<<"\n";
            continue;
        };
        vector<pair<long long int, char> >::iterator it;
        it = lower_bound(v.begin(), v.end(), make_pair(x,'0'),comp);
        pair<long long int , char> p;
        p = *it;
        cout<<p.second<<"\n";
    }
	return 0;
}
