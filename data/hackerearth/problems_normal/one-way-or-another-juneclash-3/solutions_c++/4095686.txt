#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
       long long int n;
        cin>>n;
        long long int r[n],x[n],y[n],i;
        for(i=0;i<n;i++)
            cin>>r[i]>>x[i]>>y[i];
        for(i=1;i<n;i++)
        {
            if((r[i]-r[0])%2==0)
            {
                if((abs(x[i]-x[0])+abs(y[i]-y[0]))%2==1)
                   {cout<<"No"<<endl; break;}
            }
            else
                {
                if((abs(x[i]-x[0])+abs(y[i]-y[0]))%2==0)
                   {cout<<"No"<<endl; break;}
            }
        }
        if(i==n)
            cout<<"Yes"<<endl;
    }
}
