#include <bits/stdc++.h>
using namespace std;
int value[1002][1002];
bool done[1002][1002][2];
int FINAL[1002][1002][2];
inline void fi(int *a)
{
 register char c=0;
 while (c<33) c=getchar_unlocked();
 *a=0;
 int tmp = 0;
 while (c>33)
 {
     if ( c == 45 ) tmp = 1;
     else *a=*a*10+c-'0';
     c=getchar_unlocked();
 }
 if ( tmp == 1 ) *a = 0-(*a);
}
int cal(int n1, int n2, int chance)
{
	if ( (chance == 0 && n1 == 1) || (chance == 1 && n2 == 1) ) return 0;
	if ( done[n1][n2][chance] ) return FINAL[n1][n2][chance];
	done[n1][n2][chance] = true;
	int ans = 0;
	if ( chance == 0 ) {
		ans |= (!cal(n1,n2-1,!chance));
		if ( value[n1][n2] != 1 ) ans |= (!cal(n1,n2/value[n1][n2],!chance));
	}
	else {
		ans |= (!cal(n1-1,n2,!chance));
		if ( value[n1][n2] != 1 ) ans |= (!cal(n1/value[n1][n2],n2,!chance));
	}
	FINAL[n1][n2][chance] = ans;
	return ans;
}
int main() {
	int t,n1,n2;	
    for ( int i = 1; i <= 1000; i++ ) {
        for ( int j = 1; j <= i; j++ ) {
        	value[i][j] = value[j][i] = __gcd(i,j);
        }
    }
	fi(&t);
	while ( t-- ) {
		fi(&n1); fi(&n2);
		if ( n1 == 1 && n2 == 1 ) printf("Draw\n");
		else if ( n2 == 1 ) printf("Arjit\n");
		else if ( n1 == 1 ) printf("Chandu Don\n");
		else if ( cal(n1,n2,0) ) printf("Arjit\n");
		else printf("Chandu Don\n");
	}
	return 0;
}
