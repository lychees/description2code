#include <iostream>
using namespace std;
void isCover(int px,int py,int bx,int by)
{
    if(((bx <= px) && (by <= py)) || ((bx <= py) && (by <= px)))
    cout<<"Possible"<<endl;
    else
    cout<<"Not Possible"<<endl;
}
int main()
{
    int bx,by,px,py;
    int T;
    cin>>T;
    while(T--)
    {
    	cin>>px>>py>>bx>>by;
    	isCover(px,py,bx,by);
    }
    return 0;
}
