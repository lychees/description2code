#include<bits/stdc++.h>
using namespace std;
#define ll long long int
 
ll dp[100005] = {0};
 
int main() {
	ll n;
	scanf("%lld",&n);
	ll i,j,k,sum=0,a[n+1],t;
	dp[0]=1;
	for(i=0;i<n;i++) 
	{
		scanf("%lld",&a[i]);
		sum+=a[i];
	}
	for(i=0;i<n;i++)
		for(j=sum;j>=a[i];j--)
		{
			if(dp[j-a[i]]>0) 
				dp[j] = 1;
		}
	scanf("%lld",&t);
	while(t--) {
		ll z;
		scanf("%lld",&z);;
		dp[z]?printf("Yes\n"):printf("No\n");
	}
}
