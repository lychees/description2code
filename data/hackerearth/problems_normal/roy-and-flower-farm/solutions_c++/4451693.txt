#include <bits/stdc++.h>
using namespace std;
int dp[101][10003];
int main()
{
    int t,n,p,i,j;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d%d",&n,&p);
    	int profit[n],cost[n];
    	for(i=0;i<n;i++)
    	{
    		scanf("%d%d",&profit[i],&cost[i]);
    		profit[i]-=cost[i];
    	}
    	for(i=0;i<=n;i++)
    	dp[i][0]=0;
    	for(j=0;j<=p;j++)
    	dp[0][j]=0;
    	int wt=0,ans=0;
    	for(i=1;i<=n;i++)
    	{
    		for(j=1;j<=p;j++)
    		{
    			dp[i][j]=dp[i-1][j];
    			if(j>=cost[i-1])
    			{
    				dp[i][j]=max(dp[i][j],dp[i-1][j-cost[i-1]]+profit[i-1]);
    			}
    			if(dp[i][j]>ans)
    			{
    				ans=dp[i][j];
    				wt=j;
    				
    			}
    			else if(dp[i][j]==ans)
    			{
    				wt=min(j,wt);
    			}
    		}
    	}
    	
    	printf("%d %d\n",wt,ans+p);
    }
    return 0;
}
