#include<iostream>
using namespace std;
#include<string.h>
int gcd(int n,int m)
{
	int r;
	while(m!=0)
	{
		r=n%m;
		n=m;
		m=r;
	}
	return n;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int i,x,k;
		char a[255],b[255],c[255],d[255];
		int e;
		cin>>a>>b>>c>>d>>e;
		int res=0;
		for(i=0;i<strlen(a);i++)
		{
			x=res*10+a[i]-48;
			res=x%e;
		}
	//	cout<<e<<" "<<res<<endl;
		k=gcd(e,res);
	//	cout<<k<<endl;
		res=0;
		for(i=0;i<strlen(b);i++)
		{
			x=res*10+b[i]-48;
			res=x%k;
		}
		k=gcd(k,res);
	//	cout<<k<<endl;
		res=0;
		for(i=0;i<strlen(c);i++)
		{
			x=res*10+c[i]-48;
			res=x%k;
		}
		k=gcd(k,res);
	//	cout<<k<<endl;
		res=0;
		for(i=0;i<strlen(d);i++)
		{
			x=res*10+d[i]-48;
			res=x%k;
		}
		k=gcd(k,res);
		cout<<k<<endl;
	}
	return 0;
}
