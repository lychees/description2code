# include <bits/stdc++.h>
using namespace std;
int flipped (int num) {
	int temp = 0;
	while (num != 0) {
		temp = 10*temp + num%10;
		num /= 10;
	}
	return temp;
}
int main () {
	int t; cin >> t;
	while (t--) {
		int a, b; cin >> a >> b;
		a = flipped(a);
		b = flipped(b);
		cout << flipped(a+b) << endl;
	}
}
