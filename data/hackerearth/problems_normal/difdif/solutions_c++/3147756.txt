#include<string>
#include<unordered_set>
#include<iostream>

using namespace std;

typedef long long ll;

unordered_set<ll> ans;
string s;

int main()
{
    cin>>s;
    for(int i=0;i<s.size();++i)
    {
        ll a=0;
        for(int j=i;j<s.size();++j)
        {
            a = a*179 + s[j];
            ans.insert(a);
        }
    }
    cout<<ans.size();
    return 0;
}
