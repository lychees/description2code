
#include <iostream>

using namespace std;



int main()
{
    int cc=0,a[100000]={0};

    for(int i=1;i<100000;i++)
    {
        int n=i,fl=0,ch=1;

        while(n>0)
        {
            if(n%2==1 && fl==1)
            {
                ch=0;
            }
            else if(n%2==1 && fl==0)
                fl=1;
            else if(n%2==0)
                fl=0;

            n/=2;
        }

        if(ch==1)
        {
            a[i-1]=1;
        }

    }

    int test,n;

    cin >> test;

    while(test--)
    {
        cin >> n;

        for(int i=n-1;i<100000;i++)
        {
            if(a[i]==1)
            {
                cout << i+1 << "\n";
                break;
            }
        }
    }


}
