#include <iostream>
using namespace std;
long long int gcd(long long int a,long long int b)
{
	if(b==0)
	return a;
	else
	return gcd(b,a%b);
}
int main() {
	// your code goes here
	long long int t,p,q;
	cin>>t;
	while(t--)
	{
		cin>>p>>q;
		if(gcd(p,q)==1)
		cout<<"Is a Co-Prime";
		else
		cout<<"Not a Co-Prime";
		cout<<endl;
 
	}
	return 0;
}
