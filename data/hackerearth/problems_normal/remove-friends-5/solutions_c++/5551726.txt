#include<iostream>
#include<vector>
#include<stack>
using namespace std;

int main()
{
    int t,n,k,temp,count;
    cin>>t;
    while(t--){
    	cin>>n>>k;
    	stack<int> mystack;
    	count=0;
    	for(int i=0;i<n;i++){
    		cin>>temp;
    		if(mystack.empty())
    		mystack.push(temp);
    		else{
    			while(!mystack.empty() && mystack.top()<temp && count<k){
    				mystack.pop();
    				count++;
    			}
    			mystack.push(temp);
    		}
    	}
    	if(count<k){
    		while(count<k){
    			mystack.pop();
    			count++;
    		}
    	}
    	stack<int> ans;
    	while(!mystack.empty()){
    		ans.push(mystack.top());
    		mystack.pop();
    	}
    	while(!ans.empty()){
    		cout<<ans.top()<<" ";
    		ans.pop();
    	}
    	cout<<endl;
    }
    return 0;
}
