#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
using namespace std;
int winner(int N)
{
    if(N==0)return 1;
      int p=pow(2,double(N));
    int** arrr=new int*[p+1];
    for(int i=0;i<p+1;i++)
      arrr[i]=new int[p+1];
      memset(arrr,0,sizeof(arrr));

            for(int i=1;i<p;i++)
            {
                for(int j=1;j<=i;j++)
                {
                    cin>>arrr[i+1][j];
                }
            }

          int inp[513];
          memset(inp,0,sizeof(inp));
           int l=0;
       for(int k=1;k<=pow(2,double(N-1));k++)
       {
           if(arrr[2*k][2*k-1]==1)
           {
               inp[l++]=2*k;
           }
           else{
                arrr[2*k][2*k-1]=0;
                inp[l++]=2*k-1;
           }
       }
         int m=l;
         while(m>1)
         {
             l=0;
             for(int i=0;i<m-1;i+=2)
             {
                 if(arrr[inp[i+1]][inp[i]]==1)
                 {
                     inp[l++]=inp[i+1];
                 }
                 else{
                        inp[l++]=inp[i];
                      arrr[inp[i+1]][inp[i]]=0;
                 }
             }
             m=l;

         }
         return inp[0];
}
int main()
{
    int in;
    cin>>in;
    cout<<winner(in)<<endl;
    return 0;

}
