#include<bits/stdc++.h>
using namespace std;

const int N=1e6+10;

int val[N];

void preprocess() {
    for(int i=2;i<N;i++) {
        if(!val[i]) {
            for(int j=2*i;j<N;j+=i) {
                val[j]=1;
            }
        }
    }
    val[4]=0;
    val[1]=1;
    for(int i=1;i<N;i++) {
        val[i]+=val[i-1];
    }
    return;
}
int main() {
    preprocess();
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        int l,r;
        scanf("%d%d",&l,&r);
        printf("%d\012",val[r]-val[l-1]);
    }
    return 0;
}
