#include<bits/stdc++.h>
using namespace std;
int main ()
{
	int N,i,f1,f2;
	int ar1[27],ar2[27];
	cin>>N;
	string str1,str2,str;
	getline(std::cin,str);
	
while(N--)
	{f1=0;f2=0;
	str1.clear();
	str2.clear();
	memset(ar1,0,sizeof(ar1));
	memset(ar2,0,sizeof(ar2));
		
	getline(std::cin,str1);
	//cout<<str1<<endl;	
	for(i=0;i<str1.length();++i)
	{if(str1[i]!=' ')
	ar1[str1[i]-'a']++;}
	getline(std::cin,str2);
	//cout<<str2<<endl;
	for(i=0;i<str2.length();++i)
	{if(str2[i]!=' ')
	ar2[str2[i]-'a']++;}
	for(i=0;i<26;++i)
	{	
	if(ar1[i]>ar2[i])
	f1=1;
	if(ar2[i]>ar1[i])
	f2=1;
	}
//	cout<<f1<<"   "<<f2<<endl;
	if(f1&f2==1)
	cout<<"You draw some. \n";
	if(f1>f2)
	cout<<"You win some. \n";
	if(f2>f1)
	cout<<"You lose some. \n";
	}
return 0;
}
