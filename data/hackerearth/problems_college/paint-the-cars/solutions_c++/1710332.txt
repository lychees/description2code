#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

#define ll long long
#define MAX 1000010

int a[MAX][3];
ll f[MAX][3];
int t, n;

int main(){
    scanf("%d", &t);
    while (t--){
        scanf("%d", &n);
        for (int i = 0; i < n; i++)
            for (int j = 0; j < 3; j++)
                scanf("%d", &a[i][j]);
        for (int i = 0; i < 3; i++) f[0][i] = a[0][i];
        for (int i = 1; i < n; i++){
            f[i][0] = min(f[i - 1][1], f[i - 1][2]) + a[i][0];
            f[i][1] = min(f[i - 1][0], f[i - 1][2]) + a[i][1];
            f[i][2] = min(f[i - 1][0], f[i - 1][1]) + a[i][2];
        }
        printf("%lld\n", min(f[n - 1][0], min(f[n - 1][1], f[n - 1][2])));
    }
    return 0;
}
