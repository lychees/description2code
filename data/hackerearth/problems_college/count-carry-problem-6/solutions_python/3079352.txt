x=input()
for j in range(x):
	a,b=raw_input().split()
	a=[int(i) for i in a]
	b=[int(i) for i in b]
	while len(a)<len(b):
		a=[0]+a
	while len(b)<len(a):
		b=[0]+b
	co=0
	c=0
	for i in range(len(a)-1,-1,-1):
		if a[i]+b[i]+c>=10:
			c=1
			co+=1
		else:
			c=0
	if co==0:
		print "No carry operation"
	elif co==1:
		print "1 carry operation"
	else:
		print str(co)+" carry operations"
