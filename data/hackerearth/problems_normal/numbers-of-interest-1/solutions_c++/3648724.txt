#include <iostream>
using namespace std;

int main()
{
	int T;
	cin>>T;
	long X, Y, N;
	long arr[100000];
	while(T--){
		cin>>X>>Y>>N;
		for(int i=0; i<X; i++){
			arr[i] = Y;
		}
		for(int i=X; i<N; i++){
			int count = X;
			int k = i-1;
			long sum = 0;
			while(count--){
				sum = sum + arr[k];
				k--;
			}
			arr[i] = sum;
		}
		cout<<arr[N-1]<<endl;
	}
    return 0;
}
