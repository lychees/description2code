#include <cstdio>
#include <vector>
#include <map>
#include <algorithm>
#include <iostream>
using namespace std;

int main()
{
    int N;
    cin >> N;
    vector<int> input;
    map<int,int> hash;
    
    input.resize(N);
    for(int i = 0; i < N; ++i)
    {
	cin >> input[i];
	++hash[ input[i] ];
    }
    
    sort(input.begin(),input.end());
    
    int ans = -1;
    
    for(int i = N-1; i>=0; --i)
    {
        if( hash[ input[i] ] == 1 )
	{
	   ans = input[i];
	   break;
	}
    }
    
    cout << ans << endl;
    
    return 0;
}
