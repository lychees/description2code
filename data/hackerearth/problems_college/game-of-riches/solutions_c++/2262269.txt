#include <bits/stdc++.h>
using namespace std;

int t,G,R;
int wealth[110];
int binarySearch(int arr[], int l, int r, int x){
   if (l <= r){
        int mid = l + (r - l)/2;
        if (arr[mid] == x)  return mid;
        if (arr[mid] < x) return binarySearch(arr, l, mid-1, x);
        return binarySearch(arr, mid+1, r, x);
   }
   return -1;
}


int main()
{
    scanf("%d",&t);
    while(t--){
    	scanf("%d%d",&R,&G);
    	for(int i =0; i <R; i++){ scanf("%d",&wealth[i]);}
    	int myW = wealth[0];
    	sort(wealth, wealth+R, greater<int>());
        
        int index= binarySearch(wealth,0,R-1,myW);
   
    	int group = index%G;
    	int count = 0;
    	for(;group< R; group+=G){
    		if( wealth[group] == myW ){ break;}
    		count++;
    	}
    	printf("%d\n",count);
    	
    	
    }
    return 0;
}
