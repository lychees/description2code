#include <iostream>
#include <map>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(;t;t--)
    {
    	long long n;
    	long long co=0;
    	map<char,long long> c;
    	map<char,long long> m;
    	cin>>n;
    	for(int i=65;i<=90;i++)
    	{
    		cin>>c[i];
    	}
    	long long notr=n*12;
    	long long rep=notr/26;
    	long long rem=notr%26;
    	for(int i=65;i<=90;i++)
    	{
    		if(i<rem+65)
    		m[i]=rep+1;
    		else m[i]=rep;
    	}
    	for(int i=65;i<=90;i++)
    	{
    		co+=((c[i]+m[i]-1)*(c[i]+m[i]))/2 - ((c[i]-1)*(c[i]))/2;
    	}
    	cout<<co<<endl;
    }
}
