n, l, r, q = map(int, raw_input().strip().split())
arr = map(int, raw_input().strip().split())
queries = map(int, raw_input().strip().split())

new_arr = sorted(arr[l:r+1])
for i in range(q):
    c = 0
    for k in new_arr:
        if k > queries[i]:
            print k
            c = 0
            break
        else:
            c = 1
    if c == 1:
        print "-1"
