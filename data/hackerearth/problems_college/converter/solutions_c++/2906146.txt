#include <bits/stdc++.h>
using namespace std;
map<string , char> ma;

int main()
{
    string s , ss , c;
    cin >> s;
    ma["000"] = '0';ma["001"] = '1';ma["010"] = '2';ma["011"] = '3';ma["100"] = '4';
    ma["101"] = '5';ma["110"] = '6';ma["111"] = '7';
    int flag = 2;
    for (int i  = 0 ; i < s.size() ; i++){
    	if (s[i] == '.'){flag = 1;continue;}
    	if (flag == 1){c+=s[i];}
    	else {ss+=s[i];}
    }
    s = "";
    int m = ss.size() % 3 ;
    for (int i = m ; i < 3 && m != 0 ; i++){
        s+='0';
    }
    s+=ss;
    m = c.size() % 3;
    for (int i = m ; i < 3 && m != 0 ; i++){
        c+='0';
    }
    m = s.size() / 3;
    int i = 0;
    string res1 ="" ;
   while (m--){
   	string temp = "";
   	for (int j = 0 ; j < 3 ; j++,i++){
   		temp+= s[i];
   	}
   	res1 +=ma[temp];
   }
   if (flag == 1){
   	res1 += '.';
   	m = c.size()/3;
   	i = 0;
   while(m--){
   	string temp = "";
   	for (int j =0 ; j < 3 ; j++,i++){
   		temp += c[i];
   	}
   	res1+=ma[temp];
   }
   }
   cout << res1 << ' ';
   for (int i = 0 ; i < res1.size(); i++){
   	if (res1[i] == '-'){continue;}
   	if (res1[i] == '.'){cout << " ";continue;}
   	printf("%c",((res1[i] - '0') - 1)+ 'A');
   }
    return 0;
}
