#include<bits/stdc++.h>
using namespace std;
#define M 100005
int main()
{
  int t;
  cin>>t;
  while(t--)
  {
    bool f=true;
    int x;
    cin>>x;
    int a,b;
    int temp=x/3;
    if(temp<=0)
     f=false;
    else
    {
      int i;
      for( i=temp;i>=0;i--)
      {
         if((x-3*i)%5==0)
         {
           a=3*i;
           b=x-3*i;
           break;
         }
      }
    }
    if(f)
    {
      for(int i=0;i<a;i++)
       cout<<"5";
      for(int i=0;i<b;i++)
       cout<<"3";
      cout<<endl;
    }
    else
     cout<<"-1"<<endl;

  }
  return 0;
}
