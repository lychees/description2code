import string
t = raw_input()
t = int(t)
ans = []
for i in range(t):
    s = raw_input()
    s = str(s)
    while ('ox' in s) or ('xo' in s):
        if 'xo' in s:
            s = string.replace(s, 'xo', 'x')
        if 'ox' in s:
            s = string.replace(s, 'ox', 'x')
    ans.append(s)
for i in ans:
    print i
