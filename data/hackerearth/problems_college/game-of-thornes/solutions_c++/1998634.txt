#include <iostream>
using namespace std;

int main()
{
	char str[100001];
	scanf("%s",str);
	int c_odd=0,a[26]={0},i=0;
	while(str[i]!='\0')
	{
		a[str[i]-97]++;
		i++;
	}
	for(i=0;i<26;i++)
	{
		if(a[i]%2!=0)
			c_odd++;
	}
	if(c_odd==0 || c_odd==1)
	{
		printf("YES");
	}
	else
		printf("NO");
    return 0;
}
