#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define mod 1000000007
ll f[1025]={0};
ll fact[1000011];
int main(){
    
    ll n,m;
    cin>>n>>m;
    fact[0]=1;
    for(ll i=1;i<=n*m;i++){
        fact[i]=((i%mod)*(fact[i-1]%mod))%mod;
    }
    for(int i=1;i<=n;i++){
        for(int j=1;j<=m;j++){
            f[i^j]++;
        }
    }
    ll ans=1;
    for(int i=0;i<=1024;i++){
        ans=((ans%mod)*(fact[f[i]])%mod)%mod;
    }
    cout<<ans;
}
