#include <algorithm>
#include <cstdio>
#include <cstdlib>
#include <cstring>
using namespace std;
bool work(long long a, long long b) {
  if (a > b)
    swap(a, b);
  if (!a)
    return 0;
  if (!work(b % a, a))
    return 1;
  if ((b / a) % (a + 1) & 1)
    return 0;
  return 1;
}
int main() {
  int t, i;
  long long a, b;
  scanf("%d", &t);
  for (i = 1; i <= t; ++i) {
    scanf("%I64d%I64d", &a, &b);
    puts(work(a, b) ? "First" : "Second");
  }
}
