int gcd(int a, int b)
{
  if(a<b) swap(a,b);
  while(b!=0) {
    a = a%b;
    swap(a,b);
  }
  return a;
}

int main()
{
  int n,k;
  cin>>n>>k;
  int gd = 360;
  for(int i=0;i<n;i++)
  {
    int x;
    cin>>x;
    gd = gcd(gd,x);
  }
  for(int i=0;i<k;i++)
  {
    int x;
    cin>>x;
    if(x%gd==0) cout<<"YES\n";
    else cout<<"NO\n";
  }
}
