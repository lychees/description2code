
primes={}
def gcd(a,b):
	if b==0:
		return a
	return gcd(b,a%b)
def rbf(n):
	i=1
	count=0
	while i<n:
		if gcd(n,i)==1:
			count=count+1
		i=i+1


	if count>1 and primes[count]==0:
		return True
	return False


for i in range(100000):
	primes[i+1]=0

i=2
while i<100000:
	j=2
	while i*j<=100000:
		primes[i*j]=primes[i*j]+1
		j=j+1
	i=i+1
t=input()
while t>0:
	n=input()
	if(rbf(n)):
		print 'TRUE'
	else:
		print 'FALSE'
	t=t-1
