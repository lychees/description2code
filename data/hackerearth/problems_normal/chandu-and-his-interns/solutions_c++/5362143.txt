#include <iostream>
#include <math.h>
using namespace std;
int check( int n)
{
	if( n < 6)
		return 0;
	int count = 2;
	for( int i = 2; i <= sqrt(n); i++)
	{
		if( n % i == 0)
		{
			count++;
			if( n != i*i || count == 4)
				return 1;
		}
	}
	return 0;
}
int main()
{
    int n, x;
    cin>>n;
    while( n-- ){
    	cin>>x;
    	if(check(x))
    		cout<<"YES\n";
    	else
    		cout<<"NO\n";
    }
    return 0;
}
