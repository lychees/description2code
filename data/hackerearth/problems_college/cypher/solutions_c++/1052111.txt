#include <iostream>
using namespace std;
int main()
{
	int t,a=1,k;
	char ch[100];
	cin>>t;
	for(int i=0;i<t;i++)
	{
		cin>>ch;
		for(int j=0;ch[j]!='\0';j++)
		{
			if(ch[j]+j>'z')
			{
			       k=j;
			       while(ch[j]+a<='z')
			       {
			       ch[j]=ch[j]+1;
			       k--;
			       }
			       ch[j]='a'+k-1;
			}
			else
			{
				ch[j]=ch[j]+j;
			}
		}
		cout<<"\n"<<ch;
	}
	return 0;
}
