#include <bits/stdc++.h>

using namespace std;

#define lb      lower_bound
#define ub      upper_bound
#define pb      push_back
#define pf      push_front
#define mp      make_pair
#define gcd     __gcd
#define all(x)  x.begin(), x.end()
#define S(x)    scanf("%d", &x)
#define Sl(x)   scanf("%lld", &x)
#define P(x)    printf("%d", x)
#define Pl(x)   printf("%lld", x)
#define PS(x)   printf("%s", x)
#define MOD     1000000007LL

typedef long long       ll;
typedef vector<int>     vi;
typedef vector<vi>      vvi;
typedef pair<int, int>  ii;
typedef vector<ii>      vii;

int main()
{
	int t,n,x,length;
	S(t);
	while(t--) {
		S(n);
		vi as;
		for(int i = 0; i < n; i++) {
			S(x);
			as.pb(x);
		}
		S(length);
		vi min1(n+1);
		min1[0] = as[0];
		int len = 1;
		for(int i = 1; i < n; i++) {
			if(as[i] < min1[0]) {
				min1[0] = as[i];
			}
			else if(as[i] > min1[len-1]) {
				if(len < length) {
					min1[len] = as[i];
					len++;
				}	
			} 
			else {
				int low = -1, high = len-1;
				while(low < high) {
					int mid = low + (high - low)/2;
					if(min1[mid] < as[i]) {
						low = mid+1;
					}
					else {
						high = mid;
					}
				}
				min1[high] = as[i];
			}
		}
		if(len < length) {
			cout << "-1" << endl;
		}
		else {
			cout << min1[length-1] << endl;
		}
	}
	return 0;
}
