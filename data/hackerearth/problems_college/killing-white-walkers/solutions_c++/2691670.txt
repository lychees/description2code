#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   long long int t,n,res=0,sum=0;
   cin>>t;
   while(t--)
   {
   	cin>>n;
   	int p=1;
   	if(n==0)
   	cout<<"0"<<endl;
   	else if( (n & (n-1)) == 0 )
   	cout<<"1"<<endl;
   	else
   	{
   	 while(p<n)
   	{
   	   p=p*2;
   	} 
   	//p=p/2;
   	res=p-n;
   	sum=n-res;
   	cout<<sum+1<<endl;
   	} 
   
   }
    return 0;
}
