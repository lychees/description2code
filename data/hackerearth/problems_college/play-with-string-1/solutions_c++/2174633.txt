#include<bits/stdc++.h>
using namespace std;
int main()
{
	string s1;
	string s;
	int t;
	cin>>t;
	int l;
	int i;
	int k=0;
	while(t--)
	{
		k=0;
		int flag=0;
		cin>>s;
		l=s.length();
		for(i=l-1;i>=0;i--)
		{
			s1[k]=s[i];
		//cout<<s1[k];
			k++;
		}
	for(i=1;i<k;i++)
	{
		//cout<<abs(s[i]%400-s[i-1]%400);
		//cout<<abs(s1[i]%400-s1[i-1]%400);
		if(abs(s[i]%400-s[i-1]%400)!=abs(s1[i]%400-s1[i-1]%400))
		{
			flag=1;
			break;
		}
	}
	if(flag==1)
	{
		cout<<"Not Funny"<<'\n';
	}
	else{
		cout<<"Funny"<<'\n';
	}
	}
	return 0;
}
