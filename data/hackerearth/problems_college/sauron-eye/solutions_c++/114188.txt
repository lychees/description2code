#include <iostream>
#include <stdio.h>
#define MOD 1000000007

using namespace std;
void multiply(long int F[2][2], long int M[2][2]);
void power(long int F[2][2], long int n);
long int fib(long int n)
{
 long int F[2][2] = {{1,1},{1,0}};
  if(n == 0)
    return 0;
    
  power(F, n-1);
  return F[0][0];
}
 
void power(long int F[2][2], long int n)
{
  if( n == 0 || n == 1)
      return;
 long int M[2][2] = {{1,1},{1,0}};
 
  power(F, n/2);
  multiply(F, F);
 
  if( n%2 != 0 )
     multiply(F, M);
}
 
void multiply(long int F[2][2], long int M[2][2])
{
long int x =  F[0][0]%MOD *M[0][0]%MOD + F[0][1]%MOD*M[1][0]%MOD;
long int y =  F[0][0]%MOD*M[0][1]%MOD + F[0][1]%MOD*M[1][1]%MOD;
long int z =  F[1][0]%MOD*M[0][0]%MOD + F[1][1]%MOD*M[1][0]%MOD;
long int w =  F[1][0]%MOD*M[0][1]%MOD + F[1][1]%MOD*M[1][1]%MOD;
 
  F[0][0] = x%MOD;
  F[0][1] = y%MOD;
  F[1][0] = z%MOD;
  F[1][1] = w%MOD;
}
 

int main()
{

  int t;
  long int n;
  cin>>t;
  while(t--)
  {
      cin>>n;
      printf("%d\n", fib(2*n));
  }  
  getchar();
  return 0;
}
