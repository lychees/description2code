#include<bits/stdc++.h>
using namespace std;
int xmove[8] = {  2, 1, -1, -2, -2, -1,  1,  2 };
int ymove[8] = {  1, 2,  2,  1, -1, -2, -2, -1 };
int check_qbr(int b_x,int b_y,int w_x,int w_y,int x,int y,char enemy)
{
	if(enemy!='b')
	{
	  if(b_x==x )
    	{
    		if(w_x==x)
    		{
    			if((w_y>y && w_y>b_y) || (w_y<y && w_y<b_y))
    			return 1;
    		}
    		else
    		return 1;
    	}
      if(b_y==y )
    	{
    		if(w_y==y)
    		{
    			if((w_x>x && w_x>b_x) || (w_x<x && w_x<b_x))
    			return 1;
    		}
    		else
    		return 1;
    	}
	}
	if(enemy!='r')
	{
		if(x<b_x && y<b_y)
    	{
		  while(x<8 && y<8)
		    {
			 if(x==w_x && y==w_y)
			    break;
			 if(x==b_x && y==b_y)
		    	return 1;
		    	x+=1;
			    y+=1;
	    	}
     	}
	else if(x>b_x && y<b_y)
	{
		while(x>=0 && y<8)
		{
			if(x==w_x && y==w_y)
			    break;
			if(x==b_x && y==b_y)
			    return 1;
			x-=1;
			y+=1;
		}
	}
	else if(x<b_x && y>b_y)
	{
		while(x<8 && y>=0)
		{
			if(x==w_x && y==w_y)
			    break;
			if(x==b_x && y==b_y)
		    	return 1;
			x+=1;
			y-=1;
		}
	}
	else if(x>b_x && y>b_y)
	{
		while(x>=0 && y>=0)
		{
			if(x==w_x && y==w_y)
			    break;
			if(x==b_x && y==b_y)
			return 1;
			x-=1;
			y-=1;
		}
	}
 }
	
	return 0;
}
int check_k(int b_x,int b_y,int x,int y)
{
	for(int i=0;i<8;i++)
	{
		int newx=x+xmove[i];
		int newy=y+ymove[i];
		if(newx>=0 && newx<8 && newy>=0 && newy<8)
		{
			if(newx==b_x && newy==b_y)
			return 1;
		}
	}
	return 0;
}
