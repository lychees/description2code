n=input()
s=raw_input()
lst=s.split(' ')
ar=[0]*n
for i in range(0,n):
	ar[i]=int(lst[i])
ar.sort()

cnt=1
mn=ar[0]
print len(ar)
while(cnt!=0):
	cnt=0
	for i in range(0,n):
		ar[i]=ar[i]-mn
		if(ar[i]>0):
			cnt=cnt+1
	for i in range(0,n):
		if(ar[i]>0):
			mn=ar[i]
			break
	if(cnt>0):
		print cnt
