#include<bits/stdc++.h>
using namespace std;
#define max 10000
int a[max];
int b[max];
int c[max];
int main()
{
    int n;
    scanf("%d",&n);
    int i;
    int j=0;
    int k=0;
    for(i=0;i<n;i++)
    {
    	scanf("%d",&c[i]);
    	if(c[i]%2==0)
    	a[j++]=c[i];
    	else
    	b[k++]=c[i];
    }
    
    int sum1=0;
    int sum2=0;
    for(i=0;i<j;i++)
    sum1+=a[i];
    for(i=0;i<k;i++)
    sum2+=b[i];
    
    sort(a,a+j);
    sort(b,b+k);
    
    for(i=0;i<j;i++)
    printf("%d ",a[i]);
    printf("%d ",sum1);
    for(i=0;i<k;i++)
    printf("%d ",b[i]);
    printf("%d ",sum2);
    
    return 0;
}
