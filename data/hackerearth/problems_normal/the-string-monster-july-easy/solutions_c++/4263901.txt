#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
#define sci(fd) scanf("%d",&fd)
#define scll(fd) scanf("%lld",&fd)
#define pb push_back
#define mp make_pair
#define MOD 1000000007
#define PI 3.1415926535897932
#define pii pair < int,int > 
#define fi first
#define se second
const ll infi=1000000000000000009;
int h[35][35];
char str[35];
char st[909];
int ar[35];
int sum[910][910][29];
int sum1[910][910][29];
int len[35];
int main()
{
    int t;
    sci(t);
    while(t--)
    {
        int n,i,j,l,k;
        sci(n);
        for(i=0;i<34;i++)
        for(j=0;j<34;j++)
        {
            h[i][j]=0;
            ar[j]=0;
        }
        
        for(i=0;i<n;i++)
        {
            scanf("%s",str);
            l=strlen(str);
            len[i]=l;
            for(j=0;j<l;j++)
            h[i][str[j]-'a']++;
        }
        scanf("%s",st);
        int llll=strlen(st);
        for(i=0;i<llll;i++)
        {
            ar[st[i]-'a']++;
        }
        for(i=0;i<909;i++)
        for(j=0;j<34;j++)
        for(l=0;l<ar[j]+4;l++)
        {
            sum[i][l][j]=0;
            sum1[i][l][j]=0;
        }
        for(i=0;i<26;i++)
        sum[0][0][i]=1;
        for(i=0;i<n;i++)
        {
            for(j=0;j<26;j++)
            {
                if(h[i][j]>ar[j])
                break;
            }
            if(j!=26)
            continue;
            for(j=0;j<=903;j++)
            {
                for(l=0;l<26;l++)
                {
                    for(k=0;k<=ar[l]+1;k++)
                    {
                        if(sum[j][k][l]==1)
                        sum1[j+len[i]][k+h[i][l]][l]=1;
                    }
                }
            }
            for(j=0;j<=903;j++)
            {
                for(l=0;l<26;l++)
                {
                    for(k=0;k<=ar[l]+1;k++)
                    {
                        sum[j][k][l]=sum[j][k][l]|sum1[j][k][l];
                        sum1[j][k][l]=0;
                    }
                }
            }
        }
        for(i=0;i<26;i++)
        {
            if(sum[llll][ar[i]][i]==0)
            break;
        }
        if(i==26)
        printf("YES\n");
        else
        printf("NO\n");
    }    
    return 0;
}
