n = int(raw_input())
c = map(int, raw_input().strip().split(' '))
for i in range(1, n):
    c[i] += c[i - 1]
box = {c[i]: i + 1 for i in range(n)}
c.sort()
for qi in range(int(raw_input())):
    q = int(raw_input())
    if q in box:
        print box[q]
    else:
        m = l = 0
        h = n - 1
        while l <= h:
            m = (l + h) // 2
            if q > c[m]:
                l = m + 1
            else:
                h = m - 1
        print box[c[l]]
