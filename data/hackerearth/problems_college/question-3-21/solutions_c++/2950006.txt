#include<iostream>
#include<math.h>

using namespace std;

bool primecheck(int b)
{
    if (b == 0 || b == 1)
        return false;
    for (int j = 2; j <= sqrt(b); j++)
        if (b%j == 0)
            return false;
    return true;

}

int main()
{
    int i, j, n, k, cnt = 0, y;
    int fin[100];
    cin >> n >> k;
    for (i = 0; i < k; i++)
    {
        cnt = 0;
        cin >> y;
        for (j = 1; j <= y; j++)
            if (primecheck(j))
                cnt++;
        fin[i] = cnt;
    }
    for (i = 0; i < k; i++)
    {
        cout <<fin[i]<<"\n";
    }

}
