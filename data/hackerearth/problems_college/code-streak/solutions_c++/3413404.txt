#include<iostream>
#include<cstdio>
#include<set>
#include<algorithm>
using namespace std;
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		int n;
		scanf("%d",&n);
		int id[n];
		int solved[n];
		for(int i=0;i<n;i++){
			scanf("%d%d",&id[i],&solved[i]);
		}
		set<int> s;
		int maxi=0;
		for(int i=0;i<n;i++){
			int count=0;
			if(solved[i]==1){
				while(i<n&&solved[i]==1){
					auto it=s.insert(id[i]);
					if(it.second==true){
						count++;
					}
					i++;
				}
				maxi=max(maxi,count);
			}
		}
		printf("%d\n",maxi);
	}
}
