#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

int main()
{
    int t,n,k,i;
    scanf("%d",&t);
    while(t--){
    	scanf("%d%d",&n,&k);
    	long long int *a = (long long int *)malloc(sizeof(long long int)*n);
    	for(i=0;i<n;i++){
    		scanf("%lld",&a[i]);
    	}
    	
    	sort(a,a+n);
    	printf("%lld\n",a[k-1]);
    }
    return 0;
}
