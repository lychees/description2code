#include <iostream>
#include<string.h>
using namespace std;
 
int main()
{
	int t,k,i,a[32];
	long n,count,j,count2;
 
	cin>>t;
	while(t--)
	{
		count2=0;
	
		cin>>n>>k;
		for(i=1;i<=n;i++)
		{
				count=0;
			for(j=31;j>=0;j--)
			{
				int x=1<<j;
				int show=x&i;
				if(show==0)
				a[j]=0;
				else
				a[j]=1;
			}
			for(j=0;j<32;j++)
			{
				
				if(a[j]==1&&a[j+1]==0&&a[j+1]==1)
				count++;
				else if(a[j-1]==1&&a[j]==0&&a[j+1]==1)
				count++;
			}
			if(count>=k)
			count2++;
		}
		cout<<count2<<endl;
		
	}
}
