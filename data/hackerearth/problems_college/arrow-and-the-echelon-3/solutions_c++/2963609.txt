#include<iostream>
#include<stdio.h>
using namespace std;
 
int main(){
 
 int t;
 
 scanf("%d",&t);
 
 int digit;
 
 while(t--){
 
  scanf("%d",&digit);
  
  if(digit%5==0 && digit%3==0){
   for(int i=0;i<digit;i++) printf("5");
   printf("\n");
  }
  
  else if(digit%3==0){
   for(int i=0;i<digit;i++) printf("5");
   printf("\n");
  }
  
  
  else{
  
   int x=0,y=0;
   
   int finalx,finaly;
   int flag=1;
  
   for(;5*x+3*y<=digit;x++){
    for(;;y++){
       //printf("%d\n",5*x+3*y);
       if(5*x+3*y==digit){
       	finalx=x;
       	flag=0;
       	finaly=y;
       	break;
       	//printf("yes");
       	}
       	
       	else if(5*x+3*y>digit){
       		y=0;
       		break;
       	}
    }   
   }
   
   if(5*x+3*y>digit && flag){
   	printf("-1\n");
   }
   
   else{
   
    for(int i=0;i<3*finaly;i++) printf("5");
    for(int i=0;i<5*finalx;i++) printf("3");
    
    printf("\n");
    
   }
   
   }
  
  } 
 
 
return 0;
}
