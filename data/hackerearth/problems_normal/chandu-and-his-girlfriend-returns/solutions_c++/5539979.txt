#include <bits/stdc++.h>
using namespace std;
#define max 100000
long long int a[max];
long long int b[max];
long long int c[max];
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n,m;
    	scanf("%d%d",&n,&m);
    	int i,j;
    	for(i=0;i<n;i++)
    	scanf("%lld",&a[i]);
    	for(i=0;i<m;i++)
    	scanf("%lld",&b[i]);
    	int k=0;
    	j=0;i=0;
    	while(i<n && j<m)
    	{
    		if(a[i]>b[j])
    		c[k++]=a[i++];
    		else
    		c[k++]=b[j++];
    	}
    	
    	while(i<n)
    	c[k++]=a[i++];
    	
    	while(j<m)
    	c[k++]=b[j++];
    	
    	for(i=0;i<n+m;i++)
    	printf("%lld ",c[i]);
    	
    	printf("\n");
    }
   
    return 0;
}
