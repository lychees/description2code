#include <bits/stdc++.h>
using namespace std;

long long dp[1010][1010];

char s[1010];
long long int dfs(int i, int j) {

    if(i > j) return dp[i][j] = 0;
    if(i == j) return dp[i][j] = 1;
    if(dp[i][j] >= 0) return dp[i][j];
    if(s[i] != s[j]) {
        return dp[i][j] = (dfs(i + 1, j) + dfs(i, j - 1) - dfs(i + 1, j - 1));
    }
    else return dp[i][j] = (dfs(i + 1, j) + dfs(i, j - 1) + 1);
}
int main() {
   
    int t, cas = 1;
    scanf("%d", &t);
    while(t--) {
        s[0] = '$';
        scanf("%s", s + 1);
        memset(dp, -1, sizeof dp);
        printf("%lld\n", dfs(1, strlen(s + 1)));
    }
    return 0;
}
