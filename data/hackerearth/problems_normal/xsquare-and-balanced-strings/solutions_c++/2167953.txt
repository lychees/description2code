#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	int test;
	scanf("%d",&test);
	while(test--)
	{
		char str[100010];
		scanf("%s",str);
		int i;
		int h[26];
		for(i=0;i<26;i++)
		{
			h[i]=0;
		}
		for(i=0;str[i]!='\0';i++)
		{
			h[str[i]-'a']++;
		}
		int flag=1;
		for(i=0;i<26;i++)
		{
			if(h[i]==0)//No problem as character is not present at all
			continue;
			else if((h[i]%2) ==0)//its balanced
			continue;
			else //not balanced
			{
				flag=0;
				break;
			}
		}
		if(flag)
		{
			printf("1\n");
		}
		else
		{
			printf("-1\n");
		}
	}
	return 0;
}
