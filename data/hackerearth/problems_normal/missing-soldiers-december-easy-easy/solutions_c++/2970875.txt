#include <iostream>
#include <set>
#include <algorithm>
using namespace std;
struct point
{
	int x,y,d;
};
bool my_fun(const struct point l,const struct point r)
{
	return l.x < r.x ;
}
int main() {
	// your code goes here
	int N,x,y,i,total=0,prev_last,cur_last;
	point p;
	vector<point> points;
	scanf("%d",&N);
	for(i=0;i<N;i++){
	scanf("%d %d %d",&p.x,&p.y,&p.d);
	points.push_back(p);
	}
	sort(points.begin(),points.end(),my_fun);
	total+=points[0].d+1;
	int last_inc = points[0].x + points[0].d;
	for(i=1;i<N;i++)
	{
		cur_last = points[i].x + points[i].d;
		if(points[i].x  <= last_inc && cur_last > last_inc){
		total += cur_last - last_inc;
		last_inc = cur_last;
		}
		else if(points[i].x > last_inc){
		total += points[i].d + 1;
		last_inc = cur_last;
		}
	}
	printf("%d",total);
	return 0;
}
