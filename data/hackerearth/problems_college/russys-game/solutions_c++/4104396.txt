#include <iostream>
using namespace std;

int main()
{
    int t,n,i,j,c;
    cin>>t;
    while(t--){
    	cin>>n;
    	int a[n];
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	i=0;c=0;
    	while(true){j=i;
    		if(a[i]==-1)
    		{cout<<"Win\n";break;}
    		else
    		{i=a[i];c++;}
    		if(c>=n)
    		{cout<<"Lose\n";break;}
    	}
    }
    return 0;
}
