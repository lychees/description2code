#include <iostream>
#include <stdio.h>
#include <algorithm>

using namespace std;

int main()
{
    int n;
    
    scanf("%d", &n);
    
    int x[n];
    for(int i = 0; i<n; i++)
    	scanf("%d", &x[i]);
    
    sort(x, x+n);
    
    int i = 0; 
    int min = x[0];
    
    while(i < n)
    {
    	int ans = n-i;
    	if(ans > 0)
    		printf("%d\n", ans);
    		
    	bool flag = false;
    	
    	for(int j = i; j<n; j++)
    	{
    		x[j] -= min;
    		 
    		if(x[j] == 0)
    			i++;
    		else
    		{
    			if(!flag)
    			{
    				min = x[j];
    				flag = true;
    			}
    				
    		}
    	}
 
    }
    
    return 0;
}
