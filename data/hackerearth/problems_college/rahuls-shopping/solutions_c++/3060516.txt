#include <bits/stdc++.h>
using namespace std;
 
int main() {
	float n,l,g,u,v;
	float d,sum=0,q;
	cin>>n;
	float a[51];
 	int j,k,b[900],i;
	for(i=1;i<51;i++)
	{
		j=i;
		k=0;
		while(j!=0)
		{
			k+=j%10;
			j/=10;
		}
 
		a[i]=i*k;
	}
 
	for(i=0;i<n;i++)
	cin>>b[i];
	cin>>u>>v;
 
	for(i=0;i<n;i++)
	{
		if(b[i]<=10)
		d=0.75;
 
		else if(b[i]<20)
		d=0.70;
 
		else
		d=0.80;
 
		q=a[b[i]]*u*v*d;
 
		long int h=q;
		
 
		sum+=h;
 
		//cout<<sum<<" ";	
	}
 
 
 
	cout<<sum;
 
 
 
 
 
}
