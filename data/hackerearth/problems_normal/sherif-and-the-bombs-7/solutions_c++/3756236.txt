#include <iostream>
using namespace std;

int main()
{
    int n,c[1001][1001]={0};
    long int i,j;
    cin>>n;
    while(n--)
    {
    	int x,y,r;
    	cin>>x>>y>>r;
    	r*=r;
    	for(i=1;i<=1000;i++)
    	{
    		for(j=1;j<=1000;j++)
    		{
    			if((i-x)*(i-x)+(j-y)*(j-y)<=r)
    			{
    				c[i][j]++;
    			}
    		}
    	}
    }
    long int count=0;
    for(i=1;i<=1000;i++)
    {
    	for(j=1;j<=1000;j++)
    	{
    		if(c[i][j]>1)
    		{
    			count++;
    		}
    	}
    }
    cout<<count;
}
