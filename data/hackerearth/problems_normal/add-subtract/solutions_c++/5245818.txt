#include <bits/stdc++.h>
using namespace std;
int a[10001];
int foo(int l, int r ,int mid)
{
    //cout<<"l = "<<l<<" r = "<<r<<" mid = "<<mid<<endl;
	int sum=0;
	for(int i=l;i<mid;i++)
	{
		sum+=3*(a[mid]-a[i]);
		//cout<<"sum1 = "<<sum<<endl;
	}
	for(int i=mid+1;i<=r;i++)
	{
		sum+=5*(a[i]-a[mid]);
		//cout<<"sum2 = "<<sum<<endl;
	}
	return sum;
}

int main()
{
    int t,n,k;
    cin>>t;
    while(t--)
    {
    	cin>>n>>k;
    	for(int i=1;i<=n;i++) cin>>a[i];
    	sort(a+1,a+n+1);
    	int ans=10020,u;
    	for(int i=1;i<=n-k+1;i++)
        {
            for(int j=i;j<=k+i-1;j++)
            {
                u=foo(i,i+k-1,j);
                //cout<<"u = "<<u<<endl;
                if(u<ans) ans=u;
            }
        }
        cout<<ans<<endl;
    }
    return 0;
}
