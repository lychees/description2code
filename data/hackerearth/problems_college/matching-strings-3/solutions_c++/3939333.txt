#include <iostream>
#include<string>
using namespace std;

int main()
{
  int t;
  cin>>t;
  for(int k=0;k<t;k++)
  {
  	string s1,s2;
  	cin>>s1>>s2;
  	int count=0;
  	int n1=s1.length(),n2=s2.length();
  	int min=n1;
  	if(min>n2) min=n2;                    
  	
  	for(int i=0;i<min;i++)          // we can use i<(n1 or n2 or min).in all case it will work.
  	{
  		if(s1[i]==s2[i]) count++;
  	}
  	cout<<count<<endl;
  }
    return 0;
}
