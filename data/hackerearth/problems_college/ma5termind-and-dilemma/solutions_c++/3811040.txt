#include <bits/stdc++.h>
using namespace std;

int main()
{
	int i,j,k,n,T,ans,cou;
	string str;
	scanf("%d",&T);
	while(T--){
		scanf("%d",&n);
		cin>>str;
		i=0;j=n-1;
		cou=0;
		while(i<j){
			while((str[i]=='1')&&(i<j)){
				i++;
			}
			while((str[j]=='0')&&(j>i)){
				j--;
			}
			
			if(i<j)
				cou++;
			i++;
			j--;
		}
		ans=cou;
		i=0;j=n-1;
		cou=0;
		while(i<j){
			while((str[i]=='0')&&(i<j)){
				i++;
			}
			while((str[j]=='1')&&(j>i)){
				j--;
			}
			if(i<j)
				cou++;
			i++;
			j--;
		}
		if(ans>cou)
			ans=cou;
		cout<<ans<<endl;
	}
    return 0;
}
