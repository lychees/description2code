#include<iostream>
#include<algorithm>
using namespace std;
#define get getchar_unlocked
inline int fastinput( )
{
       int n = 0 , s = 1 ; char p = get( ) ;
       if( p == '-' ) s=-1;

       while( ( p < '0' || p > '9' ) && p != EOF && p != '-' )
          p = get( ) ;
       if( p == '-' )
       s = -1 , p = get( ) ;
       while( p >= '0' && p <= '9' )
       {
              n = ( n << 3 ) + ( n << 1 ) + ( p - '0' ) ;
              p = get( ) ;
       }
       return n * s ;
}


int main()
{
    int t,n,i,c;
    t=fastinput();
    while(t--)
    {
        n=fastinput();
        long long int a[n];
        i=0;
        while(i<n)
        {
            a[i]=fastinput();
            i++;
        }
        sort(a,a+n);
        i=0;c=0;
        while(i<n-1)
        {
            if(a[i]==a[i+1])
                {++i;continue;}
            else
                ++c;
                i++;
        }
        cout<<"\n"<<(n-c-1);
    }
}
