
#include<malloc.h>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

int main()
{
	int n,m,x,y;
	char arr[1010][1010];
	cin>>n>>m>>x>>y;
	rep(i,n)
		scanf("%s",arr[i]);

	int b[1010]={0};

	rep(i,m)
	{
		rep(j,n)
		{
			if(arr[j][i]=='#')
				b[i]+=1;
		}
	}

	int dp[1010][2]={{0}};


	dp[0][1]=1e8;
	dp[0][0]=1e8;
	if(x==1)
	{
		dp[0][1] = b[0];
		dp[0][0] = n - b[0];
	}

	for(int i=1;i<m;i++)
	{
		int cost = 0;
		int costb=0;
		dp[i][0]=1e8;
		dp[i][1]=1e8;
		for(int j = i; j >i-y && j>=0; j--)
		{
			cost+= b[j];
			costb+=n-b[j];
			if(j-1 <= i-x)
			{
				if(j>0)
				{
					dp[i][1]=min(dp[i][1],cost+dp[j-1][0]);
					dp[i][0]=min(dp[i][0],costb+dp[j-1][1]);
				}
				else
				{
					dp[i][1]=min(dp[i][1],cost);
					dp[i][0]=min(dp[i][0],costb);
				}
			}
		}
	//	cout<<dp[i][0]<<" "<<dp[i][1]<<endl;
	}
	cout<<min(dp[m-1][0],dp[m-1][1])<<endl;

	return 0;
}
