#include <iostream>
using namespace std;

int main()
{
    int t,k,l;
    cin>>t;
    string c;
    
    while(t--){
    	cin>>k;
    	int x;
    	cin>>c;
    	l = c.length();
    	if(k>26)
    		k = k%26;
    	for(int i =0;i<l;i++){
    		
    		x = c[i]+k;
    	//	cout<<x<<" ";
    		if(x>90&&x<117&&(int)c[i]<=90){
    			x = x-26;
    			c[i] = (char)x;
    		}else if(x>122){
    			x=x-26;
    			c[i]=(char)x; 
    		}else {
    			c[i] = (char)x;
    		}
    		
    			
    		
    	}
    	cout<<c<<endl;
    }
    return 0;
}
