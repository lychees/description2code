#include <iostream>
#include <map>
using namespace std;

int main()
{
    int t;
    string strIn;
    map<string, int> smap;
    map<string, int>::iterator it;
    
    cin >> t;
    
    if ( !cin.fail()) {
    	for (int i=0; i < t; ++i ) {
   		 	cin >> strIn;
    		
    		if ( smap.empty()) {
    			smap[strIn]=1;
    		} else {
    			it=smap.find(strIn);
    			
   		 		if ( it != smap.end()) {
    				it->second += 1;
    			} else {
    				smap[strIn]=1;
    			}
    		}
    	}
    	
    	for ( it=smap.begin(); it != smap.end(); ++it ) {
    		cout << it->first << " " << it->second << endl;
    	}
    }
    
    return 0;
}
