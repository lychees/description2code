#include <bits/stdc++.h>

using namespace std;

int main()
{
    string s;

    cin >> s;

    for(int i = 0; i < s.size() - 1; i++) {
        if(i == 0 && s[i] != '_') {
            if(s[i] <= 'z' && s[i] >= 'a') {
                s[i] += ('A' - 'a');
            }
        }
        else if( s[i] == '_' && i + 1 < s.size()) {
            if(s[i + 1] <= 'z' && s[i + 1] >= 'a') {
                s[i + 1] += ('A' - 'a');
            }
        }
    }

    cout << s <<endl;

}
