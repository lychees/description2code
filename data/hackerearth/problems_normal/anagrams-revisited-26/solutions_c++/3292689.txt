#include<bits/stdc++.h>
using namespace std;

int main()
{
    map<string,int> mp;
    string s;
    int n,i;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>s;
    	sort(s.begin(),s.end());
    	mp[s]++;
    }
    int max=-1;
    map<string,int> :: iterator it;
    for(it=mp.begin();it!=mp.end();it++){
    	if(it->second>max){
    		max=it->second;
    	}
    }
    cout<<max<<endl;
    return 0;
}
