#include <cstdio>
#include <cstring>
#include <cmath>
#include <vector>
#include <algorithm>
using namespace std;

#define MAX 1000010
#define pb push_back

int a[MAX];

void prepare(){
	int x, y;
	double c = (1 + sqrt(5))/2;
	
	memset(a, -1, sizeof(a));
	for (int i = 0; i < MAX; i++){
		x = (int)(c*i);
		y = (int)(c*c*i);
		if (x < MAX)
			a[x] = y;
	}
}

int main(){
	int n, m, t, flag;
	
	prepare();
	scanf("%d", &t);
	while (t--){
		scanf("%d%d", &n, &m);
		if (n > m){
			n ^= m;
			m ^= n;
			n ^= m;
		}
		printf("%s\n", a[n] != m ? "Play" : "Don't Play");
	}
	return 0;
}
