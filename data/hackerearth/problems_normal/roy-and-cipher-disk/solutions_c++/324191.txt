#include<stdio.h>
#include<stdlib.h>
#include<iostream>

using namespace std;

int main()
{
	int x,i,j,t;
	string s;
	cin>>t;
	char c='a';
	for(i=0;i<t;i++)
	{
		cin>>s;
		for(j=0;j<s.length();j++)
		{
			if(j==0)
			{
				x=(int)s[j]-(int)c;
				if(abs(x)<(26-abs(x)))
					cout<<x<<" ";
				else if(abs(x)>(26-abs(x)))
				{
					if(x<0)
						cout<<26-abs(x)<<" ";
					else
						cout<<-1*(26-abs(x))<<" ";
				}
				else
				{
					cout<<abs(x)<<" ";
				}
			}
			else
			{
				x=(int)s[j]-(int)s[j-1];
				if(abs(x)<(26-abs(x)))
					cout<<x<<" ";
				else if(abs(x)>(26-abs(x)))
				{
					if(x<0)
						cout<<26-abs(x)<<" ";
					else
						cout<<-1*(26-abs(x))<<" ";
				}
				else
				{
					cout<<abs(x)<<" ";
				}
			}
		}
		cout<<"\n";
	}
	return 0;
}
