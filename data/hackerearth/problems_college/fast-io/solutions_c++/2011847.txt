#include <bits/stdc++.h>
using namespace std;
int main()
{
    long long int n,count=0;
    scanf("%lld",&n);
    long long int a[n];
    for(long long int i=0;i<n;i++)
    {
    	scanf("%lld",&a[i]);
    }
    for(long long int i=0;i<n;i++)
    {
    	if(a[i]<300000000)
    	{
    		count++;
    	}
    }
    cout<<count<<endl;
}
