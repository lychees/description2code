#include <iostream>
using namespace std;
int min(string str,string str1)
{
	if(str.length()>str1.length())
	 return str1.length();
	 else
	      return str.length();
}
int max(string str,string str1)
{
	if(str.length()>str1.length())
	 return str.length();
	 else
	      return str1.length();
}
int main()
{
   int n,mm;
   string str,str1;
   cin>>n;
   
   while(n)
   {
   	int m, count=0;
   	n--;
   	cin>>str;
   	cin>>str1;
   	m=min(str,str1);
   	mm=max(str,str1);
   	for(int i=0;i<m;i++)
   	  {
   	  	if(str[i]==str1[i])
   	  	{
   	  		count++;
   	  	}
   	  }
   	  if(mm==count)
   	      cout<<"Exact"<<endl;
   	  else
   	  cout<<"Non Exact"<<" "<<mm-count<<endl;
   	  
   	  count=0;
   	
   }
    return 0;
}
