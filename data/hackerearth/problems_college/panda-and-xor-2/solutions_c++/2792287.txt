#include <iostream>
using namespace std;
int countSetBits(long long int n)
{
    unsigned int count = 0;
    while (n)
    {
      n &= (n-1) ;
      count++;
    }
    return count;
}
int main()
{
    long long int a,b,t;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b;
    	a=a^b;
    	b=countSetBits(a);
    	if(b%2==0)
    		cout<<"YES"<<endl;
    	else
    		cout<<"NO"<<endl;
    }
    return 0;
}
