#include<bits/stdc++.h>
using namespace std;

const int N=1e6+10;
typedef long long ll;

bool isPrime[N];
int eulerPhi[N];
ll res[N];

int cnt(int a,int b) {
  int r=0;
  while(a%b==0) {
    r++;
    a/=b;
  }
  return r;
}
int main() {
  isPrime[1]=true;
  for(int i=2;i<N;i++) {
    if(isPrime[i]==false) {
      for(int j=i+i;j<N;j+=i) {
        isPrime[j]=true;
      }
    }
  }
  for(int i=2;i<N;i++) {
    if(isPrime[i]==false) {
      for(int j=i;j<N;j+=i) {
        res[j]+=cnt(j,i);
      }
    }
  }
  for(int i=1;i<N;i++) {
    res[i]+=res[i-1];
  }
  int tt;
  cin>>tt;
  while(tt--) {
    int n;
    cin>>n;
    cout<<res[n]<<endl;
  }
  return 0;
}
