
#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,k,x,cnt=0;
        cin>>n>>k;
        for(int i=0;i<n;i++)
        {
            cin>>x;
            if(x<=0)
            {
                cnt++;
            }
        }
        if(cnt>=k)
            cout<<"NO"<<endl;
        else
            cout<<"YES"<<endl;

    }
    return 0;
}
