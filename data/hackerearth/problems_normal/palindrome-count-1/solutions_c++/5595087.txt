#include <iostream>
using namespace std;

int main()
{
    string s;
    cin>>s;
    int n=s.size();
    bool a[n][n];
    for(int i=0;i<n;i++)
     for(int j=0;j<n;j++)
      if(i>=j)
       a[i][j]=1;
      else
      a[i][j]=0;
    for(int i=n-1;i>=0;i--)
     for(int j=n-1;j>i;j--)
        if(s.at(i)==s.at(j))
        {
        	if(i+1<=j-1)
        	 a[i][j] = a[i+1][j-1];
            else
             a[i][j]=1;
        }
    int ans=0;
    for(int i=0;i<n;i++)
     for(int j=i;j<n;j++)
      {ans+=a[i][j];
       //cout<<a[i][j]<<' ';
      }
    cout<<ans;
    return 0;
}
