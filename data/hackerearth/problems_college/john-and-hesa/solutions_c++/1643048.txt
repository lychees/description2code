#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin >> t;
    
    for (int i=0; i<t; ++i) {
    	int x,p;
    	cin >> x >> p;
    	//if (p == 1) {cout << x << }
    	int count = 0;
    	while (x%p == 0) {
    		x /= p;
    		++count;
    	}
    	if (x == 1) x = 0;
    	cout << count << " " << x << "\n";
    }
    return 0;
}
