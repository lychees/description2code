#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        int arr[n];
        for(int i=0;i<n;i++)
        {
            cin>>arr[i];
        }
        int ans=0;
        for(int i=0;i<32;i++)
        {
            ans+=(1<<i);
        }
        bool flag=false;
        for(int i=0;i<32;i++)
        {
            int x=ans;
            for(int j=0;j<n;j++)
            {
                if(arr[j]&(1<<i))
                   x=x&arr[j];
            }
            if((x&(x-1))==0)
            {
                flag=true;
                break;
            }
        }
        if(flag==true)
            cout<<"YES\n";
        else
            cout<<"NO\n";
    }
}
