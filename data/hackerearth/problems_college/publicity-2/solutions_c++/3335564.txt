#include<bits/stdc++.h>
using namespace std;
vector<int>ar[100000];
bool visited[100000];
void dfs(int n)
{
	if(visited[n])
	{
		return ;
	}
	visited[n] =1;
	for(vector<int>::iterator it = ar[n].begin();it!=ar[n].end();it++)
	{
		if(!visited[*it])
		{
			dfs(*it);
		}
	}
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		
		int n,e,i,j;
		cin>>n>>e;
		memset(visited,0,sizeof visited);
		for(int i=0;i<n;i++)
		   ar[i].clear();
		for(i=0;i<e;i++)
		{
			int x,y;
			cin>>x>>y;
			ar[x].push_back(y);
			ar[y].push_back(x);
			
		}
		int ans=0;
		for(i=0;i<n;i++)
		{
			if(!visited[i])
			{
				dfs(i);
				ans++;
			}
		}
		cout<<ans<<"\n";
	}
}
