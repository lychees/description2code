#include <iostream>
using namespace std;

int main()
{
	int T;
	cin>>T;
	while(T--)
	{
		char s[501];
		cin>>s;
		char ch1=s[0];
		int l=-1;
		int l1=-1;
		while(s[++l1]!='\0');
		while(s[++l]==ch1);
		if(s[l]=='\0')
		{
			cout<<"Not OK\n";
			continue;
		}
		char ch2=s[l];
		int flag=1;
		for(int i=l;i<l+l;i++)
			if(s[i]!=ch2)
			{
				flag=0;
				break;
			}
		char ch3=s[l+l];
		if(!flag || ch3==ch2 || ch3==ch1)
			cout<<"Not OK\n";
		else
		{
			for(int i=l+l;i<l+l+l;i++)
				if(s[i]!=ch3)
				{
					flag=0;
					break;
				}
			if(flag && l+l+l==l1)
				cout<<"OK\n";
			else
				cout<<"Not OK\n";
		}
	}
    return 0;
}
