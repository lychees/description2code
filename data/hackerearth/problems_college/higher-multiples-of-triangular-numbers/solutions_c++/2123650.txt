#include<iostream>
using namespace std;
int main()
{
	int t,a[1001],n[1000],i,j,k;
	cin>>t;
	for(int i=1;i<1001;i++)
	{
		a[i]=0;
		j=0;
		while(j<=i)
		{
			a[i]+=j;
			j++;
		}
	}
	i=0;
	while(i<t)
	{
		cin>>n[i];
		i++;
	}
	for(int i=0;i<t;i++)
	{
		int c=0;
		for(int j=1;j<1000;j++)
		{
			k=1;
			c=1;
			while(k<a[j])
			{
				if(a[j]%k==0)
				c++;
				if(c>n[i])
				{
				cout<<a[j]<<endl;	
				break;
				}
				
			k++;	
			}
			if(c>n[i])
			break;	
		
		}
	}
 }
