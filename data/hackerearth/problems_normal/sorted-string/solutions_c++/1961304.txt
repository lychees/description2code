#include <iostream>
#include <string>
#include <algorithm>
#include <vector>

using namespace std;

int main()
{
   int t;
   cin>>t;
   string s;
   while(t--){
   	cin>>s;
   	
   	vector<pair<int,int>> v(26);
   	
   	for(int i=0;i<26;i++){
   		v[i].first=0;
   		v[i].second=i;
   	}
   	
   	for(int i=0;i<s.length();i++){
   		v[s[i]-'a'].first++;
   	}
   	
   	sort(v.begin(),v.end());
   	
   	for(int i=0;i<26;i++)
		{
			int count=v[i].first;
			char ch=v[i].second+ 'a';
			for(int j=0;j<count;j++)
				cout<<ch;
		}
		
		cout<<endl;
   	
   }
   		
   	
    return 0;
}
