#include <iostream>
#include<cmath>
using namespace std;

int isnotprime(int n)
{
	int i;
	if(n==2)
		return 0;
	if(n%2==0)
		return 1;
	for(i=3;i<=sqrt(n);i+=2)
	{
		if(n%i==0)
			return 1;
	}
	
	return 0;
}
int main()
{
int i;
int n;
cin>>n;
for(i=4;i<=n;i++)
{
	if(isnotprime(i)==1)
		cout<<i<<endl;
}
return 0;
}
