def mapper(string):
    buff = dict()
    for i in string:
        if i in buff:
            buff[i] = buff[i] + 1
        else:
            buff[i] = 1
    return buff


s = raw_input()
counted = mapper(s)
value = max(counted.values())
listt = [key for key, val in counted.iteritems() if val == value]
listt.sort(key=lambda x: ord(x))
print listt[0], value
