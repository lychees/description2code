#include<iostream>
using namespace::std;
#define M 1000000007

long long N,A,B,K,dp[5020][5020];

int main()
{
	int i,j,k,l,r,x,y;
	cin>>N>>A>>B>>K;
	if(B<A) 
	{
	 A=N+1-A;
	 B=N+1-B;
	}
	
	for(x=A;x<B;x++)
	 dp[0][x]=1;
	
	for(i=0;i<K;i++) 
	{
		for(x=1;x<B;x++) 
		{
			int up=x+(B-x-1)/2;
			dp[i+1][x]=dp[i][up]-(dp[i][x]-dp[i][x-1]);
			dp[i+1][x]=(((dp[i+1][x]+dp[i+1][x-1])%M)+M)%M;
		}
	}
	cout<<dp[K][B-1]<<endl;
}
