#include<iostream>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		int *a,n;
		cin>>n;
		a=new int[n];
		for(int i=0;i<n;i++)
			cin>>a[i];
	
	int c=0;
		for(int i=0;i<n;i++){
			for(int j=i;j<n;j++)
				if((a[i]==a[j]) && j>=i)
					c++;
		}
cout<<c<<endl;
delete a;
	}
}
