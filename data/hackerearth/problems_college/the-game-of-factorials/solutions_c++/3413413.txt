#include <bits/stdc++.h>

using namespace std;

int main(){
	int a[20000];
	int t, temp=0, p, I, n, in, j, h;
	scanf("%d",&t);
	while (t--){
		temp=0; p=1;
		a[0]=1;
		for (I=1; I<20000; I++)
		a[I]=0;
		scanf("%d",&n);
		for (I=2; I<=n; I++){
			in=0;
		 while(in<p||temp){
		 	h=(a[in]*I+temp);
		 	temp=h/10;
		 	a[in]=h%10;
		 	in++;}
		 	p=in;
		}
		int sum=0;
		 	for (j=p-1; j>=0; j--)
		          sum+=a[j];
		      printf("%d\n",sum);
	}
	return 0; 
}
