#include<stdio.h>
#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
int main()
{
    std::vector<int>vec;
    long t,p;
    scanf("%ld",&t);
    for(long k=0;k<t;k++)
    {
        scanf("%ld",&p);
        vec.push_back(p);
    }

    vector<int>::iterator it;
    std::sort(vec.begin(),vec.end());
    for(it=vec.begin() ;it!=vec.end();it++)
        cout<<*it<<"\n";
    return 0;
}
