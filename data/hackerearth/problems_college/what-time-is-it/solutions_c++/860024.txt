# include <iomanip>
# include <iostream>

using namespace std;

int main () {
	double a, h, m;

	cin >> h;
	getchar ();
	cin >> m;

	h = (h * 5) + (m / 12);
	a = (h - m) * 6;
	if (a < 0) a *= -1;
	if (a == 360) a = 0;
	else if (a > 180) a -= 180;

	cout << setiosflags(ios :: fixed) << setprecision(6) << a;

	return 0;
}
