#include <iostream>
#include <algorithm>
#include <cmath>

using namespace std;

int main(void) {
	int n;
	cin >> n;
	int a[n];
	int sum = 0;
	for(int i = 0; i < n; i++) {
		cin >> a[i];
		sum += a[i];
	}
	sort(a, a+n);
	
	if(n <= 3) {
		cout << a[n-1] << endl;
	}
	else {
		cout << ceil(sum/3.0) << endl;
	}
	return 0;
}
