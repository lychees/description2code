#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;

int main()
{
	int t,n,l,i,maxRight,curRight,j,flag=0;
	pair <int,int> roads[2005];
	cin>>t;
	while(t--){
		flag=0;
		cin>>n>>l;
		for(i=0;i<n;i++){
			cin>>roads[i].first>>roads[i].second;
		}
		sort(roads,roads+n);
		for(i=0;i<n;i++){
			curRight=roads[i].second;
			maxRight=roads[i].first+l;
			for(j=i+1;j<n;j++){
				if(roads[j].first<=curRight&&roads[j].first>roads[i].first&&roads[j].second<=maxRight){
					curRight=max(curRight,roads[j].second);
				}
			}
			if(curRight==maxRight){
				flag=1;
				break;
			}
		}
		if(flag == 1)
			cout<<"Yes"<<'\n';
		else 
			cout<<"No"<<'\n';
	}
    return 0;
}
