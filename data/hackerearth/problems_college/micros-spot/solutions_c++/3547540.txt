#include<bits/stdc++.h>
#include<stdio.h>

using namespace std;

int main(){
	int t;
	cin>>t;
	while(t--){
		int n,m;
		scanf("%d %d",&n,&m);
		int a[n][m],b[n][m];
		int amax[2000] = {0},bmax[2000] = {0};
		for(int i=0;i<n;i++){
			for(int j=0;j<m;j++){
				scanf("%d",&a[i][j]);
				//cin>>a[i][j];
				if(a[i][j] > amax[j] )
					amax[j] = a[i][j];
				
			}
		}
		int counter = 0;
		
		for(int i=0;i<n;i++){
			int maxr = 0;
			int cnt = 0;
			for(int j=0;j<m;j++){
				scanf("%d",&b[i][j]);
				//cin>>b[i][j];
				
				
			
				if(a[i][j] == amax[j]){
					if(maxr < b[i][j]){
						maxr = b[i][j];
						cnt = 1;
					}
					else if(maxr == b[i][j]){
						cnt++;
					}
				}
				
				if(maxr < b[i][j]){
					maxr = b[i][j];
					cnt = 0;
				}
			}
			counter += cnt;
		}
		printf("%d\n",counter);
		//cout<<counter<<endl;
		
	}
}
