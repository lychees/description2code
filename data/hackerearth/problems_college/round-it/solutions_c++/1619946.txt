#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long t,n,b;
    scanf("%ld",&t);
    while(t--)
    {
    	scanf("%ld%ld",&n,&b);
    	long a=n%b;
    	a=n-a;
    	long a1=a+b;
    	//printf("%d%d\n",a,a1);
    	if(n-a<a1-n)
    	printf("%ld\n",a);
    	else
    	printf("%ld\n",a1);
    }
    return 0;
}
