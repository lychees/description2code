import string
testcases = int(raw_input())
alphabet_dict = dict()
while (testcases > 0):
    name = raw_input()
    alphabet_dict = dict.fromkeys(string.ascii_lowercase, 0)
    for c in name:
        alphabet_dict[c] = alphabet_dict[c] + 1
    # print([(k,v) for v,k in sorted([(v,k) for k,v in alphabet_dict.items()])])
    # print(sorted(alphabet_dict,key=f))
    x = sorted(alphabet_dict,
               key=lambda k: (-alphabet_dict[k], k), reverse=True)
    print ' '.join(str(p) for p in x)
    testcases = testcases - 1
