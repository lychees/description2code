#include <bits/stdc++.h>
int main()
{
    long long int t,a,b,ab,ac,mul;
    scanf("%lld",&t);
    while(t--)
    {
    	scanf("%lld%lld",&a,&b);
    	mul=a/b;
    	ac=mul*b;
    	ab=(mul+1)*b;
    	if(abs(ab-a)==abs(ac-a))
    		printf("%lld\n",ab);
    	else if(abs(ab-a)<abs(ac-a))
    		printf("%lld\n",ab);
    	else
    		printf("%lld\n",ac);
    }
    return 0;
}
