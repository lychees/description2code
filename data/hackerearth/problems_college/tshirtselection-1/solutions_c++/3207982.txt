#include <bits/stdc++.h>
using namespace std;

int main(){
	int t;
	cin >> t;
	while(t--){
	vector<int > v(1000001,0);
	map<int,int> m1;	
	int n;
	cin >> n;
	int i,j,st=0,ed=0;
	int max1=0;
	for(i=0;i<=n-1;i++) cin >> v[i];
	for(i=0,j=0;j<n;j++){
		m1[v[j]]++;
		if(m1[v[j]]>1){
			while(m1[v[j]]>1 && i<=j){
				m1[v[i]]--;
				i++;
			}
		}
			if(max1<j-i+1){
				st = i;
				ed = j;
				max1 = j-i+1;
			}
	}
	cout << st+1 << " " << ed+1 <<  endl;
}
return 0;
}
