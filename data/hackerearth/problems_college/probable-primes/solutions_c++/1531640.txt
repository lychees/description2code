using namespace std;

int primes[1000004], len=0;
int isPrime(int x)
{
    int last = sqrt(x);
    for(int i=2;i<=last;i++)
    {
        if(x%i==0)
            return 0;
    }
    return 1;
}

void generate_prime()
{
    for(int i=2;i<1000001;i++) // 32000=sqrt(10^9) all possible primes which are factors for numbers less than 10^9
    {
        if(isPrime(i))
            primes[i]=1;
        else
            primes[i]=0;
    }
}
