#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int n;
    int tab[1000];
    cin>>n;
    for(int i=0; i<n; ++i)
    	cin>>tab[i];
    sort(tab, tab+n);
    int res = max(tab[n-1]-tab[n-2], tab[n-1]-tab[n-3]);
    for(int i=n-2; i>=2; --i){
    	res= max(res, tab[i]-tab[i-2]);
    }
    cout<<max(res, tab[1]-tab[0])<<'\n';
    return 0;
}
