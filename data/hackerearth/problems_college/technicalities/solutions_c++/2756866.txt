#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    long long t,n,ans,temp;
    char arr[1000000];
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	cin>>arr;
    	
    	ans=0;
    	for(int i=0;i<n;i++)
    	{
    		temp =(int)arr[i]-48;
    		ans*=10;
    		ans+=temp;
    		ans%=1000000007;
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
