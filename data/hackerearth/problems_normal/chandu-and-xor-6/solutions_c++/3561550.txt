#include <iostream>
using namespace std;

int main()
{
    int n, m, i, j, row_sum = 0, col_sum = 0;
    cin>>n>>m;
    int a[n][m], row[n], col[m];
    for(i =0; i<n; i++) {
    	row_sum = 0;
    	for(j =0; j < m; j++) {
    		cin>>a[i][j];
    		row_sum += a[i][j];
    	}
    	row[i] = row_sum;
    }
    
    for(j = 0; j<m; j++) {
    	col_sum = 0;
    	for(i = 0; i<n; i++) {
    		col_sum += a[i][j];
    	}
    	col[j] = col_sum;
    }
    int ans = 0;
    for(i= 0; i<n; i++) {
    	for(j = 0; j<m; j++) {
    		ans = max(ans, row[i]^col[j]);
    	}
    }
    cout<<ans<<endl;
    
    return 0;
}
