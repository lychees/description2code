#include<iostream>
#include<limits.h>
using namespace std;
long long int sol(long long int n,long long int random)
{
    if(n==1)
    {
        return 1;
    }
    if(n<=random+1)
    return 1;
    if(n<=2*(random+1))
    return 2;
    else
    return(2+sol(n-2*(random+1),random+1));
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long int n;
        cin>>n;
        cout<<sol(n,0)<<"\n";

    }
}
