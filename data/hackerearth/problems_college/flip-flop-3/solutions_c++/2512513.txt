#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while(t--){
		char s[100000],temp;
		long long int i,counts,n;
		cin >> s;
		temp = s[0];
		counts = 0;
		n = strlen(s);
		for( i = 1; i < n; i++)
		{
			if( temp == s[i])
				counts++;
			else
			{
				temp = s[i];
			}
		}
		cout << counts << "\n";
		
	}
	
    return 0;
}
