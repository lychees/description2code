#include<bits/stdc++.h>
using namespace std;
//Edit-Distance Algorithm :D
int min(int x, int y, int z) 
{
    return min(min(x, y), z);
}
int editDistDP(string str1, string str2, int m, int n)
{
    int dp[m+1][n+1];
    for (int i=0; i<=m; i++)
    {
        for (int j=0; j<=n; j++)
        {
            if (i==0)
                dp[i][j] = j;  // Min. operations = j
            else if (j==0)
                dp[i][j] = i; // Min. operations = i
            else if (str1[i-1] == str2[j-1])
                dp[i][j] = dp[i-1][j-1];
            else
                dp[i][j] = 1 + min(dp[i][j-1],  // Insert
                                   dp[i-1][j],  // Remove
                                   dp[i-1][j-1]); // Replace
        }
    }
    return dp[m][n];
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n1,n2;
        cin>>n1>>n2;
        int a,b;
        string str1,str2;
        cin>>a>>str1;
        cin>>b>>str2;
        int sw,fr;
        int d=editDistDP(str1, str2, str1.length(), str2.length());
        sw=(d*5)+n1;
        if(d==0)
            fr=n2+(10*a);
        else
            fr=n2;
        cout<<sw<<endl<<fr<<endl;
        if(sw>fr)
            cout<<"Swapnil\n";
        else if(fr>sw)
            cout<<"Friend\n";
        else
            cout<<"Tie\n";
    }
    return 0;
}
