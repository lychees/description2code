// Example program
#include <iostream>
#include <cstring>
using namespace std;

void strReverse(char* str)
    {
    int length = strlen(str);
    char temp;
    for(int x = 0, y = length-1;x <= (length-1)/2; x++, y--)
    {
        temp = str[x];
        str[x]=str[y];
        str[y] = temp;
    }
    return;
}


int main()
{
    int test,a,i,j,k,res[1000];
    char st1[1000],st2[1000],st3[2000],st4[2000];
    cin>>test;
    for(i=0;i<test;i++)
    {
        res[i]=0;
        cin>>st1>>st2;
        for(j=0;j<=strlen(st1);j++)
        {
            k=0;
            while(k!=j)
            {
                st3[k]=st1[k];
                k++;
            }
            for(a=0;a<strlen(st2);a++)
            {
                st3[k]=st2[a];
                k++;
            }
            for(a=0;a<=(strlen(st1)-j);a++)
            {
                st3[k]=st1[(j+a)];
                k++;
            }
            strcpy(st4,st3);
            //cout<<st3<<" ";
            strReverse(st4);
            //cout<<st4<<"  1 ";
            if(strcmp(st3,st4)==0)
                res[i]++;
        }
    }
    for(i=0;i<test;i++)
    {
        cout<<res[i]<<endl;
    }
    return 0;
}
