#include<iostream>
#include<string>
#include<cstdio>
using namespace std;
int main()
{
	long long i,j,k,l,m,t,x,y,n;
	double ans;
	cin>>t;
	while(t--){
		ans=0.0;
		x=y=0;
		cin>>k>>n;
		if(k>=n)ans=0.0;
		else{
			x=y=1;
			ans=1.0;
			m=n;
			for(i=0;i<k;i++,n--)
			{
				ans*=n;
				ans/=m;
			}
		}
		printf("%.9f\n",1.0-ans);
	}
	return 0;
}
