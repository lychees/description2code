    #include <iostream>
    using namespace std;
     
    int main()
    {
    	int t;
    	cin>>t;
    	while(t--){
    		int h,m;
    		cin>>h>>m;
    		int ah,am;
    		ah=0.5*(60*h+m);
    		am=6*m;
    		int a=abs(am-ah);
    		int s=360-a;
    		int ans=(a>s)?s:a;
    		cout<<ans<<endl;
    	}
        //cout << "Hello World!" << endl;
        return 0;
    }
