#include<iostream>
#include<string>
#include<algorithm>
using namespace std;
int main()
{
    string res;
    int n,ln,i;
    cin>>res;
    cin>>n;
    string s[n];
    for(i=0;i<n;i++)
                    cin>>s[i];
    sort(s,s+n);
    ln=res.size();
    for(i=0;i<n;i++)
    {
                    if(res.compare(s[i].substr(0,ln)) == 0)
                    {
                                                    cout<<s[i];
                                                    break;
                    }
    }
    if(i==n)
    cout<<res;
    return 0;
}
