import sys

t=int(sys.stdin.readline())
while t>0:
	n,m=map(int,sys.stdin.readline().split())
	x0,y0=map(int,sys.stdin.readline().split())
	x1,y1=map(int,sys.stdin.readline().split())
	p=int(sys.stdin.readline())
	x=min(abs(x1-x0),n-abs(x1-x0))
	y=min(abs(y1-y0),m-abs(y1-y0))
	cost=(x+y)*p
	sys.stdout.write(str(cost)+"\n")
	if cost>1000:
		sys.stdout.write("Let go of the group.\n")
	else:
	    sys.stdout.write("You saved the group.\n")
	t-=1
