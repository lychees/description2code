#!/usr/bin/env python

from operator import mul
n, m, t = map(long, raw_input().strip().split())


def nCk(n, k):
    return int(round(
        reduce(mul, (float(n-i)/(i+1) for i in range(k)), 1)
    ))


res = 0
for x in xrange(4, n+1):
    for y in xrange(1, m+1):
        if y+x == t:
            res += nCk(n, x)*nCk(m, y)

print res
