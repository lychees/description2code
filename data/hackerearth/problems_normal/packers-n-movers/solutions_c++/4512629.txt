ll a[MAX];
ll n, m;
bool check(ll mid)
{
    ll cnt = 0;
   for(int i=0;i<n;i++)
   {
      if(a[i]>mid)
        return 0;
   }
    for (int i = 0; i < n;)
    {
        ll s = 0;
        while (s <= mid && i < n)
        {
            if (s + a[i] > mid)
                break;
            s += a[i];
            ++i;
        }
        //cout << i << endl;
        //cout << s << " **** " << endl;
        cnt++;
    }
    //cout << cnt << endl;
    //cout << "x" << endl;
    if (cnt <= m)
        return 1;
    else
        return 0;
}

    ll lo = 1, hi = 1e17, ans = 0;
    //check(2);
    while (lo <= hi)
    {
        ll mid = lo + (hi - lo) / 2;
        //cout << mid << endl;
        if (check(mid))
        {
            ans = mid;
            hi = mid - 1;
        }
        else
            lo = mid + 1;
    }

    return 0;
