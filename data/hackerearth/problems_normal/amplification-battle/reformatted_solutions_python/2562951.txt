import math
t = input()
for i in range(0, t):
    str = raw_input()
    ls = map(long, str.split())
    a = ls[0]
    b = ls[1]
    c = ls[2]
    base = a*b
    r = math.log(c, base)
    r = long(r)
   # print r
    rem = float(c)/float((base ** r))
   # print rem
    p = math.ceil(rem)  # give atleast p
  #  print p
    if p <= 1:
        print "Bob"
        continue
    n = math.ceil(p/float(b))  # get atleast n
   #  print n
    if n <= 1:
        print "Adam"
    else:
        print "Bob"
