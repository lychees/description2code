#include <bits/stdc++.h>
using namespace std;

int main()
{
    int x;
    int y;
    cin >> x;
    int a[x + 1];
    int m1[100002];
    int m2[100002];
    int ans[100002];
    int w = 0;
    memset(m1,0,sizeof(m1));
    memset(m2,0,sizeof(m2));
    int i;
    for(i = 0 ; i < x; i++)
    {
    	cin >> a[i];
    }
    for(i = 0 ; i < x ;i++)
    {
    	m1[a[i]]++;
    }
    cin >> y;
    int b[ y + 1];
    for(i = 0 ; i < y; i++ )
    {
    	cin >> b[i];
    }
    sort(b , b + y);
    int len;
    len = b[y - 1];
    for(i = 0 ; i < y ; i++)
    {
    	m2[b[i]]++;
    }
    for(i = 0 ; i <= len; i++ )
    {
    	if(m1[i] < m2[i])
    	{
    		ans[w] = i;
    		w++;
    	}
    }
    sort(ans,ans + w);
    for(i = 0 ; i < w; i++)
    {cout << ans[i] << " ";
    }
    cout << endl;
    return 0;
}
