#include <cstring>
#include <cassert>
#include <iostream>
#include <iomanip>
#include <cstdio>
using namespace std;
#define FOR(i,n) for(int i=0; i<(int)n ;i++)
char s[105];
int main()
{
    int t;
    scanf("%d",&t);
    assert(t <= 20000 && t >= 1);
    while(t--)
    {
        scanf("%s",s);
        int l=strlen(s);
        assert(l <= 100);
        int cur = 0;
        int flag = 0;
        int ans = 0;
        ans = 0;
        for(int i=0;i<l;i++)
        {
            if(s[i]=='|')
            {
                flag = 1;
                ans = ans^cur;
                cur = 0;
            }
            else if(s[i] == '_')
            {
                cur++;
            }
            else
                assert(0);
        }
        if(flag && cur)
            ans = ans^cur;
        if(ans)
            printf("Alice\n");
        else
            printf("Bob\n");
    }
return 0;
}
