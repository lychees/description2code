#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    char ar[10000],ar1[10000],ar2[10000];
    int t,n;
    cin>>t;
    while(t--)
    {
    	cin>>ar;
    	cin>>ar1;
    	n=strlen(ar);
    	for(int i=0;i<n;i++)
    	{
    		if(ar[i]==ar1[i])
    		ar2[i]='0';
    		else
    		ar2[i]='1';
    	}
    	for(int i=0;i<n;i++)
    	{
    	cout<<ar2[i];
    	}
    	cout<<"\n";
    }
    return 0;
}
