#include<iostream>
#include<vector>
#include<string>
#include<cmath>
#include<algorithm>
using namespace std;
int solve(vector<int>&have)
{
	int ansd = 1000000007, ans = 1;
	for (int i = 1; i <= 26; i++)
	{
		int curans = 0;
		for (int j = 1; j <= 26; j++)
		{
			curans += have[j] * abs(i - j);
		}
		if (curans < ansd)
		{
			ansd = curans;
			ans = i;
		}
	}
	return ans;
}
int main()
{
	int t;
	cin >> t;
	for (int tt = 1; tt <= t; tt++)
	{
		int n, k;
		cin >> n >> k;
		vector<vector<int> >have;
		vector<int>tmp(35);
		have.resize(35, tmp);
		for (int i = 1; i <= n; i++)
		{
			string s;
			cin >> s;
			for (int j = 0; j < k; j++)
			{
				have[j][s[j] - 'a' + 1]++;
			}
		}
		string ans;
		for (int i = 0; i < k; i++)
		{
			char cur = 'a' + solve(have[i]) - 1;
			ans += cur;
		}
		cout << ans << endl;
	}
}
