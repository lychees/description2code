#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    int  a,b,c,r;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b>>c;
    	cout<<a/b;
    	if(c>0)
    	{
    		cout<<".";
    	}
    		r=a%b;
    		r=r*10;
    	while(c--)
    	{
    	
    		cout<<r/b;
    		r=r%b;
    		r=r*10;
    	}
    	cout<<"\n";
    	
    }
    return 0;
}
