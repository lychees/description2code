from operator import itemgetter

def countOnes(work):
	count = 0
	while(work):
		work = (work&(work-1))
		count +=1
	return count
	
def monkTasks(workDays):
	workDays.sort(key=itemgetter(1))
	return workDays
	

try:
	testCase = int(raw_input())
	while(testCase):
		workDays = []
		days = int(raw_input())
		works = raw_input().split()
		for work in works:
			workDays.append((int(work),countOnes(int(work))))
			days -= 1
		result = monkTasks(workDays)
		print (" ".join([str(item[0]) for item in result]))
		# for item in result:
		# 	print("%d " % item[0])
		# print '\n'
		testCase -= 1
except Exception as e:
	print e.message
