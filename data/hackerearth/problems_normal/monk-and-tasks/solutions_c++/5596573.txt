#include <iostream>
using namespace std;
string output[61];
int main() {
    int t, n;
    long int number;
    scanf("%d", &t);
    while (t--){
        scanf("%d", &n);
        int j = 61;
        while (j--){
            output[j]="";
        }
        while (n--) {
            scanf("%ld", &number);
            int no_ones = __builtin_popcount (number);
            output[no_ones] = output[no_ones].append(to_string(number)+ " ");
        }
        for (int i = 1; i < 61; ++i) {
            cout<<output[i];
        }
        cout<<endl;
    }
    return 0;
}
