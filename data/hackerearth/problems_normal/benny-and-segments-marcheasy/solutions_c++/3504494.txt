#include<bits/stdc++.h>
using namespace std;
#define pb push_back
#define mp make_pair 
vector<pair<int,int> >v;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,L,a,b,i,j;
		v.clear();
		scanf("%d%d",&n, &L);
		for(i=0;i<n;i++)
		{
			scanf("%d%d",&a, &b);
			v.pb(mp(a,b));
		}
		int f=0;
		sort(v.begin(),v.end());
	//	for(i=0;i<n;i++)
		   // printf("%d %d\n",v[i].first,v[i].second);
	//	printf("\n");
		
		for(i=0;i<n;i++)
		{
			if((v[i].second-v[i].first)>L)
				continue;
			int y=v[i].second;
			f=0;		
			for(j=i;j<n;j++)
			{
				if(v[i].first<=v[j].first && v[j].first<=y)
				{
					y=max(y,v[j].second);
				//	printf("%d    %d\n",(v[j].second-x),y);
					if((v[j].second-v[i].first)==L)
					{
						printf("Yes\n");
						f=1;
						break;		
					}
				}
			}
			if(f==1)
				break;
		}
		if(f==0)
			printf("No\n");
	}
	return 0;
}
