#include <iostream>
#include<stack>
using namespace std;

int main()
{
    int i, itr, t_cases;
    cin >> t_cases;
    int count = 0;
    for(itr = 0; itr < t_cases; itr++){
    	string str;
    	cin >> str;
    	stack <char> s;
    	s.push(str[0]);
    	
    	for(i = 1; i < str.length(); i++){
    		if(s.empty()){
    			s.push(str[i]);
    		}
    		else if(str[i] == s.top()){
    			s.pop();
    		}
    		else{
    			s.push(str[i]);
    		}
    		
    	}
    	if(s.empty()){
    	//	cout << "test" << "\n";
    		count++;
    	}
    }
    cout << count;
}
