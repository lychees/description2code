#include <iostream>
#include<cmath>
using namespace std;

#define PI 3.14159265
#define e 2.71828183

int main()
{
   int t;
   cin>>t;
   while(t--)
    {
    	int64_t n;
    	cin>>n;
    	int64_t ans=floor((log(2*PI*n)/2 + n*(log(n) - 1) + 1/(12*n))/log(10))+1;
    	int64_t x=n,c=0,i=5;
    	while((x/i)>0)
    	 {
    	 	c+=(x/i);
    	 	i=i*5;
    	 }
    	 ans=ans-c;
    	cout<<ans<<"\n";
    }
}
