#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,m,val,pos;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d%d",&n,&m);
    	if(n==1)
    	{
    		printf("0\n");
    	}
    	else
    	{
    		val=n-1;
    		if(m==val/2)
    		{
    			pos=val;
    		}
    		else if(m>val/2)
    		{
    			pos=2*(val-m);
    		}
    		else
    		{
    			pos=1+2*(m);
    		}
    	}
    	printf("%d\n",pos);
    }
    return 0;
}
