#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    int n;
    while(t--)
    {
        cin>>n;int x,ans = 0;
        cin>>x;
        ans = x%5;
        for(int i=1;i<n;i++)
        {
            cin>>x;
            ans = ans^(x%5);
        }
        if(ans==0)
            cout<<"Andrew"<<endl;
        else
            cout<<"Aniruddha"<<endl;
    }
}
