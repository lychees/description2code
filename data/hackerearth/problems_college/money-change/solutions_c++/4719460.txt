#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long m,n;
   	long long i,j;
    cin >> n >> m;
    long long arr[m];
    for(i=0;i<m;i++)
    cin >> arr[i];
    long long x,y;
    long long table[n+1][m];
    
    for(i=0;i<m;i++){
    	table[0][i] = 1;
    }
    
    for(i=1;i<=n;i++){
    	for(j=0;j<m;j++){
    		if(i-arr[j]>=0)
    			x = table[i-arr[j]][j];
    		else
    			x= 0;
    		if(j>=1)
    			y = table[i][j-1];
    		else
    			y = 0;	
    	
    		table[i][j] = x+y;
    		
    	}
    	
    	
    }
    cout << table[n][m-1] << endl; 
    return 0;
}
