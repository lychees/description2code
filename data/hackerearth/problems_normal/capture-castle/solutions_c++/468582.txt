#include <iostream>
#include <stdio.h>

void fillcostmap(int map[101][101],int cost[101][101],int i,int j,int M,int N,int curcost,int x,int y)
{
    if(i<0||j<0||i>=M||j>=N)
        return;
    if(i>x && j>y)
        return;
    curcost = curcost+map[i][j];
    if(cost[i][j]  == -1// not filled
             || cost[i][j] > curcost // we need minimum cost
             )
    {
        cost[i][j] = curcost;
        fillcostmap(map,cost,i,j+1,M,N,curcost,x,y);
        fillcostmap(map,cost,i+1,j,M,N,curcost,x,y);
        fillcostmap(map,cost,i,j-1,M,N,curcost,x,y);
        fillcostmap(map,cost,i-1,j,M,N,curcost,x,y);
    }
}

int main()
{
    int cost[101][101],map[101][101],T,M,N,x,y,t;

    scanf("%d",&T);
    while(T--)
    {
        scanf("%d %d",&M,&N);
        for(int i=0;i<M;i++)
        {
            for(int j=0;j<N;j++)
            {
                scanf("%d",&map[i][j]);
                cost[i][j] = -1;
            }
        }

        scanf("%d %d %d",&x,&y,&t);
        cost[0][0] = map[0][0] + 1;
        fillcostmap(map,cost,0,0,M,N,0,x-1,y-1);

        if(t >= cost[x-1][y-1])
        {
            printf("YES\n");
            printf("%d\n",t-cost[x-1][y-1]);
        }
        else
            printf("NO\n");

    }

}
