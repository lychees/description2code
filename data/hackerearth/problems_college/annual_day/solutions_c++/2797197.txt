#include <iostream>
#include<algorithm>
using namespace std;
int countSetBits(int n)
{
    unsigned int count = 0;
    while (n)
    {
      n &= (n-1) ;
      count++;
    }
    return count;
}
int main()
{
    int t,n,k,a[1001],i,b,s;
    cin>>t;
    while(t--)
    {
    	s=0;
    	cin>>n>>k;
    	for(i=0;i<n;i++)
    	{
    		cin>>b;
    		a[i]=countSetBits(b);
    	}
    	sort(a,a+n);
    	for(i=n-1;i>n-k-1;i--)
    		s=s+a[i];
    	cout<<s<<endl;
    }
    return 0;
}
