#include <iostream>
#include <cstdio>
using namespace std;



void initializeHash(int* hash){
	for(int i = 0; i < 256; i++){
		hash[i] = 0;
	}
}

int main()
{
	int T, i, j;
	int hash[256];
	cin >> T;
	string str;
	for(i = 0; i < T; i++){
		initializeHash(hash);
		cin >> str;
		for(j = 0; j < str.length(); j++){
			hash[str[j]] = 1;
		
		}
		
		for(j = 0; j < str.length(); j++){
			if(hash[str[j]] == 1){
				cout << str[j];
				hash[str[j]] = 0;
			}
		}
		cout << endl;
	}
    
    return 0;
}
