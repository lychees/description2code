#include <iostream>
#include<cmath>
#include<iomanip>
using namespace std;

int main()
{
    long long t,n,tmp;
    long double x=0.00,y=0.00,x1=0.00,y1=0.00,ax=0.00,ay=0.00;
    cin >> t;
    //std::cout<<fixed<<std::setprecision(2);
    //cout.setf(ios::showpoint);
    while(t--)
    {
    //	std::cout<<fixed<<std::setprecision(2);
    	cin >> n;
    	tmp = n;
    	while(n--)
    	{
    		cin >> x >> y;
    		x1=x1+x;
    		y1+=y;
    	}
    ax = x1/(double)tmp;
    ay = y1/(double)tmp;
    std::cout<<fixed<<std::setprecision(2);
    cout.setf(ios::showpoint);
    cout << "("<<ax<<" "<<","<<" "<<ay << ")"<<endl;
    x1=0.00;
    y1=0.00;
    }
    return 0;
}
