    #include <iostream>
    #include<algorithm>
    using namespace std;
     
    int main()
    { 
    	long long int temp;
    	cin>>temp;
    	
    while(temp--)
    { 
      long long int num;int s=0,c=0;
      cin>>num;
      int arr[num],b[num];
      for(int i=0;i<num;i++) cin>>arr[i];
      sort(arr,arr+num);
    for(int j=0;j<num;j++) b[j]=arr[j]-arr[0];
    for(int k=0;k<num;k++)
    { s=b[k];
      c=c+s/5;
      s=s%5;
       c=c+s/2;
      s=s%2;
       c=c+s/1;
    }
    cout<<c<<endl;
    }
        return 0;
    }
