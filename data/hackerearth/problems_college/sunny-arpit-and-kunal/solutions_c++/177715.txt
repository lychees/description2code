#include<iostream>
using namespace std;


void ret(int a,int b,int c)
{
    int p,q;

    p=b-a;
    q=c-b;


    if(p>q)
        cout<<p-1<<endl;
    else
        cout<<q-1<<endl;
}
int main()
{
    int t,arr[3],i,j,temp;
    cin>>t;

    while(t--)
    {
    for(i=0;i<3;i++)
            cin>>arr[i];

        for(i=0;i<3;i++)
            for(j=i+1;j<3;j++)
            if(arr[i]>arr[j])
            {
            temp= arr[j];
            arr[j]=arr[i];
            arr[i]=temp;
            }


            ret(arr[0],arr[1],arr[2]);

    }

    return 0;

}
