#include <iostream>
using namespace std;
void mid(int i,int j,int n){
	int ans = 0;
	ans = ans+ abs(n/2-i)+abs(n/2-j);
	cout<<ans<<endl;
}
int main()
{
	int t,n,j,k;
	string line;
	cin >> t;
	while(t > 0){
		cin>>n;
		for(int i=0;i<n;i++){
			cin>>line;
			int found = line.find('*');
			if(found >=0){
				j=i;k=found;	
			}
		}
		mid(j,k,n);
		t--;
	}
    return 0;
}
