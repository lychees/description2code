#include <iostream>
using namespace std;

int main()
{
    int threes[10],t,n;
    threes[0]=0;
    threes[1]=1;
    int base=10;
    for(int i=2;i<10;i++){
    	threes[i]=9*threes[i-1]+base;
    	base*=10;
    }
    cin>>t;
    for(int i=0;i<t;i++){
    	cin>>n;
    	int org=n;
    	base=1;
    	int cnt=0;
    	int ans=0;
    	while(n!=0){
    		int digit=n%10;
    		if(digit>3){
    			ans+=base;
    			ans+=(digit-1)*threes[cnt];
    		}
    		else{
    			ans+=digit*threes[cnt];
    		}
    	
    	//	cout<<ans;
    		n=n/10;
    		base*=10;
    		cnt++;
    	}
    	//cout<<ans<<" ";
    	cout<<(org-ans)<<endl;
    }
    return 0;
}
