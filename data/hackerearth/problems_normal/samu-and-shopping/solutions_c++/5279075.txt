#define sf scanf
#define sfd(z) scanf("%d",&z)
#define sfdd(z,y) scanf("%d%d",&z,&y)
#define sflld(z) scanf("%lld",&z)
#define sff(z) scanf("%f",&z)
#define fr(init,stop) for(int z=init;z<stop;z++)
#define pf printf
#define pfd(vari) printf("%d",vari)
#define pff(vari) printf("%f",vari)
#define pfdd(vari) printf("%lld",vari)
#define vrd(vari) vector<int>vari
#define qed(vari) queue<int>vari
#define pb push_back
#define pft push_front
#include<bits/stdc++.h>
using namespace std;
int main()
{
	int  t;
	cin>>t;
	while(t--)
	{
	    int n;
		sfd(n);
		vector<vector<int> > ans(n,vector<int>(3));
		int x,y,z;
		sf("%d%d%d",&x,&y,&z);
		ans[0][0]=x;
		ans[0][1]=y;
		ans[0][2]=z;
		for(int i=1;i<n;i++)
		{
			sf("%d%d%d",&x,&y,&z);
			ans[i][0]=x+min(ans[i-1][1],ans[i-1][2]);
			ans[i][1]=y+min(ans[i-1][0],ans[i-1][2]);
			ans[i][2]=z+min(ans[i-1][0],ans[i-1][1]);
		}
     
        int w=min(ans[n-1][0],min(ans[n-1][1],ans[n-1][2]));
        printf("%d\n",w);
	}
	return 0;
}
