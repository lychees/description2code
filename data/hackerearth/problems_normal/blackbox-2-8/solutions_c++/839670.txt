#include<iostream>
using namespace std;

int fact[1001][3000] = {0};

void factorial(int n)
{
     int i;
     int sum=0;
     fact[1][0] = 1;
     fact[1][1] = 1;
     if (fact[n][0] == 0) {
       for (i=n-1; i > 0 ; i--) {
             if(fact[i][0] != 0)
              break;
       }
    
    for ( ; i<n; i++) {
        int j = 1;
        int carry = 0;
        int len = fact[i][0];
        while (len--){
            int temp = (i+1)*fact[i][j] + carry;
            fact[i+1][j] = temp%10;
            carry = temp/10;
            j++;

        }
        
        while (carry > 0) {
            fact[i+1][j] = carry%10;
            carry /= 10;
            j++;
        }
        fact[i+1][0] = j-1;
       
    }

}
for ( i=fact[n][0]; i > 0; i--) {
    sum+=fact[n][i];
}
cout<<sum<<endl;
}

int main() {
   int i,T,N;
   cin>>T;
   for(i=0;i<T;i++)
   		{
   			cin>>N;
   			if(N==0)
   				cout<<1<<endl;
   			else
   				factorial(N);
   		}
   return 0;
}
