#include <bits/stdc++.h>
using namespace std;

int main()
{
    
    int n,x;
    cin>>n>>x;
    int a[n];
    
    for(int i=0;i<n;i++){
    	cin>>a[i];
    }
    
 
int count = 0;
int i = 0;
int val = 0;

while(i<n){
	if(a[i]<=x){
		val++;
	}else{
		if(count==0){
			count++;
		}else{
			break;
		}
	}
	i++;
}
cout<<val;
    
    return 0;
}
