#include<bits/stdc++.h>
using namespace std;
int near(int a)
{
    int i=2;
    while(i<=a)
    {
        i=i*2;
    }
    return i/2;
}
int main()
{
    int n;
    cin >> n;
    while(n--)
    {
        int a;
        cin >> a;
        cout << 2*(a-near(a))+1 << endl;
    }
    return 0;
}
