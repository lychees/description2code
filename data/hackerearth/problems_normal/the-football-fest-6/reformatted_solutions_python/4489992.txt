st = []
for n in range(T):
    N_P = raw_input().split()
    N = N_P[0]
    P = []
    P.append(int(N_P[1]))
    for i in range(int(N)):
        c = raw_input().split()
        if c[0] == 'B':
            P.append(P[len(P)-2])
        else:
            P.append(int(c[1]))
    st.append(P.pop())

for i in st:
