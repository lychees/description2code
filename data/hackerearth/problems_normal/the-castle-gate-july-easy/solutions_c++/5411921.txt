#include <iostream>
using namespace std;

int main()
{
    int T;
    cin>>T;
    int *N= NULL;
    N = new int[T];
    int i,j,k;
    int count=0;
    int x;
    
    for(i=0;i<T;i++)
    {
    	cin>>N[i];
    }
    for(i=0;i<T;i++)
    {
    	count=0;
    	for(j=1;j<=N[i];j++)
    	{
    		for(k=j+1;k<=N[i];k++)
    		{
    			if(k==j)
    			continue;
    			else
    			{
    				x= k^j;
    				if(x <= N[i])
    				count++;
    			}
    		}
    	}
    	cout<<count<<endl;
    }
    
    return 0;
}
