#include <bits/stdc++.h>

#define ll long long
#define pb push_back
#define mp make_pair
#define ld long double
#define line pair<ll, ll>

using namespace std;

const int nm = 1000002;

int n, n2;
char s[nm];
int a[nm], b[nm];

int main() {
	int t;
	scanf("%d", &t);
	while (t--) {
		scanf("%s", s);
		n = strlen(s);
		memset(a, 0, sizeof(a));
		for (int i = 0; i < n; ++i)
			a[n - 1 - i] = s[i] - 48;
		scanf("%s", s);
		n2 = strlen(s);
		memset(b, 0, sizeof(b));
		for (int i = 0; i < n2; ++i)
			b[n2 - 1 - i] = s[i] - 48;
		int res = 0;
		int nho = 0;
		for (int i = 0; i < max(n, n2); ++i) {
			if (nho)
				res++;
			nho = (a[i] + b[i] + nho) / 10;
		}
		if (nho)
			res++;
		if (!res)
			printf("No carry operation\n");
		else if (res == 1)
			printf("1 carry operation\n");
		else
			printf("%d carry operations\n", res);
	}
}
