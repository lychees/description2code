#include <iostream>


using namespace std;


int main() {
	int array[1000005] = {-1};
	int t,n1,n2,i;
	cin>>t;
	fill(array, array + 1000005, -1);
	array[0]=0;
	int max_val = 0, max_ind = 0;
	for(i=1; i<1000005; i++) {
		if(array[i] == -1) {
			array[i] = max_val + 1 + i - max_ind;
			max_ind = i;
			max_val = array[i];
			if(max_val < 1000005)
				array[max_val] = i;
		}
	}
	int a, b;
	while(t--) {
		cin>>a>>b;
		if(array[a] == b)
			cout<<"Don't Play"<<endl;
		else
			cout<<"Play"<<endl;
	}
}
