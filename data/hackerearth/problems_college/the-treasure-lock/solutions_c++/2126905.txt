#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<cstring>
using namespace std;
#define p(x) cout<<x<<":"<<x<<"\n"
int cs,i,n,j;
int main()
{
  scanf("%d",&cs);
  while(cs--)
  {
    scanf("%d",&n);
    if(!(n%5))
    {
      for(i=0;i<n;i++)
        printf("3");
      printf("\n");
      continue;
    }
    for(i=0;i<=n;i+=5)
      if(!((n-i)%3))
      {
        for(j=0;j<n-i;j++)
          printf("5");
        for(j=0;j<i;j++)
          printf("3");
        break;
      }  
    if(i>n)
      printf("-1");
    printf("\n");
  }
	return 0;
}
