#include <iostream>
using namespace std;

int main()
{
	unsigned long numtest;
	unsigned long long numcards;
    cin>>numtest;
    
    for(int i=0;i<numtest;i++)
    {
    	cin>>numcards;
    	if(numcards==1 || numcards==2)
    	cout<<"1"<<"\n";
    	else if(numcards%3==0)
    	cout<<numcards/3<<"\n";
    	else
    	cout<<numcards<<"\n";
    }
    return 0;
}
