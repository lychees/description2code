#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define MAX 17
int n;
ll M[17][17],DP[1<<17][17],LV[17],D[17];
bool V[1<<17][17];
int main(){
	
	cin >> n;
	
	for(int i=0;i<n;i++) cin >> LV[i];

	for(int i=0;i<n;i++) cin >> D[i];

	for(int i=0;i<n;i++)
		for(int j=0;j<n;j++)
			M[i][j] = n*LV[i] - ((j*(j+1)/2)*(i*D[i]) + 1LL*(n-j-1)*(j*(i*D[i])));
 	int sz = 1<<n;

	for(int i=0;i<sz;i++)
		for(int j=0;j<n;j++)
			DP[i][j] = -1000000000000LL;

	for(int i=0;i<n;i++)
		V[1<<i][0] = 1,DP[1<<i][0] = M[i][0];

		
	for(int i=1;i<n;i++){

		for(int j=1;j<sz;j++){

			if(V[j][i-1]){
				for(int k=0;k<n;k++){
					if(!(j&(1<<k))){
						DP[j|(1<<k)][i] = max(DP[j|(1<<k)][i] , DP[j][i-1] + M[k][i] );
						V[j|(1<<k)][i] = 1;
					}															
				}
			}
		}
	}
	cout << DP[(1<<n)-1][n-1] << endl;
	return 0;
}
