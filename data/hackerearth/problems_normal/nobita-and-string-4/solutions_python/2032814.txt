cases = int(raw_input())
i = 1
output = []
while i <= cases :
    text = raw_input().strip().split()
    l = len(text)
    text = list(text)
    i+=1
    for j in range(int(l/2)):
        temp = text[j]
        text[j] = text[l-1-j]
        text[l-1-j] = temp
    s = " ".join(text)
    output.append(s)
for item in output:
    print(item)
