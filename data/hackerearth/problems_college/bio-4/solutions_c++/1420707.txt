#include <bits/stdc++.h>
#define PB push_back
#define MP make_pair
#define LL long long
#define MAXN 100001
#define M 1000000007
#define F first
#define S second
#define sc(x) scanf("%d",&x)
#define scl(x) scanf("%lld",&x)
using namespace std;

int main(){
    int n, t, i, j, ans, cnt, pos, len1, len2;
    string s1, s2;
    cin>>t;
    while(t--){
        cin>>s1>>s2;
        len1 = s1.length();
        len2 = s2.length();
        ans = -1;
        for(i=0;i<len1;i++){
            cnt = 0;
            for(j=0;j<len2;j++){
                if((j + i >= len1))
                break;
                if(s2[j] == s1[j + i])
                    cnt++;
            }
            if(cnt > ans){
                ans = cnt;
                pos = i;
            }
        }
        cout<<(len2 - ans)<<endl;
    }
}
