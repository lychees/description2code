#include <iostream>

using namespace std;
long long int first[1000007], second[1000007];
int main(){
    int n;
    int q;
    
    cin >> n >>q;
    
    first[0] = 0;
    second[0] = 0;
    
    for(int i = 1; i <= n; i++) 
    	cin >> first[i];
    
    
    for(int i=1;i<=n;i++)
    	second[i] = second[i-1] + first[i];
    	
    int l;
    int r;
    while(q--)
    {
        cin >> l >> r;
        cout << second[r] - second[l-1] << " " << first[r] - (second[r-1] - second[l-1]) <<endl;
    }
}
