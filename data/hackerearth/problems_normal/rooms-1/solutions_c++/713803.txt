#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
int main(){
	std::ios::sync_with_stdio(false);
	int t,n;
	vector<long long> start;
	vector<long long> en;
	long long a,b;
	cin>>t;
	while(t--){
		cin>>n;
		start.resize(n);
		en.resize(n);
		for(int i=0;i<n;i++){
			cin>>start[i];
		}
		for(int i=0;i<n;i++){
			cin>>en[i];
		}
		for(int i=0;i<n;i++){
			en[i] += (start[i]-1);
		}
		
		sort(start.begin(),start.end());
		sort(en.begin(),en.end());
		int maxv=0,c=0,p1=0,p2=0;
		while(p1<n && p2<n){
			if(start[p1] < en[p2]){
				c++;
				if(c>maxv) maxv = c;
				p1++;
			}
			else if(start[p1] == en[p2]){
				while(p1<n-1 && start[p1+1] == start[p1]){
					p1++;
					c++;
				}
				c++;
				if(c>maxv) maxv = c;
				while(p2<n-1 && en[p2+1] == en[p2]){
					p2++;
					c--;
				}
				c--;
				p2++; p1++;
			}
			else{
				c--; p2++;
			}
		}
		cout<<maxv<<"\n";
	}
	return 0;	
}
