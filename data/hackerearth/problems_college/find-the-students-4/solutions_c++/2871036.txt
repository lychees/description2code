#include <iostream>
using namespace std;
int main()
{
	int n,i,yo=0;
	cin>>n;
	int a[1000];
	int min=1000,max=0;
	int len=n;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
		if(a[i]>max)
		{
			max=a[i];
		}
	}
	for(i=0;i<n;i++)
	{
		if(a[i]==max)
		{
			yo++;
		}
	}
	while(len>yo)
	{
		
		
	//cout<<endl;
	cout<<len<<endl;
	for(i=0;i<n;i++)
	{
		if(min>a[i])
		{
			if(a[i]>0)
			{
				min=a[i];
			}		
		}
		//cout<<"min"<<min<<endl;
	}
	for(i=0;i<n;i++)
	{
		a[i]=a[i]-min;
		if(a[i]==0)
		{
			len--;
		}
	}
	}	
	cout<<len;

	return 0;
}
