import math
t=int(raw_input())
while t > 0 :
	n1 = raw_input()
	n2 = n1.split( )
	n=long(n2[0])
	x = long(n2[1])
	p = raw_input()
	p1 = p.split()
	ans=0
	pre = long(p1[0])
	for i in range(1,n):
		cur=long(p1[i])
		if cur <= pre :
			z = float(pre-cur+1)
			z = z/float(x)
			ans += long(math.ceil(z))
			y = math.ceil(z) 
			cur = cur + (x*y)
		pre = cur
	print ans 
	t= t-1
