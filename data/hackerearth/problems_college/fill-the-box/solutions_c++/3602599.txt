#include<iostream>
#include<cmath>
using namespace std;
int main()
{
	long long int m,n,a,b,k,z[10000],sum=0,i;
	cin>>m;
	cin>>n;
		for(i=0;i<m;i++)
	z[i]=0;
	while(n--)
	{

	cin>>a>>b>>k;
	for(i=a-1;i<b;i++)
	{
	z[i]=z[i]+k;		
	}
		
	}
	for(i=0;i<m;i++)
	sum=sum+z[i];
	cout<<sum/m<<endl;
}
