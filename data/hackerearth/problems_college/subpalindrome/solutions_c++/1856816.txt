#define ll long long int
#define IN_M 1061109567

#define maxi 100000005

using namespace std;

ll dp[105][105];

int main()
{
   ll t;
   
   cin >> t;
	getchar();
   while (t--) {
   		string str;
   		cin >> str;
   		ll n = str.length();
   		
   		for (ll i = 0; i < n; i++) {
   			dp[i][i] = 1;
   		}
   		
   		
      	for (ll l = 2; l < n + 1; l++) {
      		
      		for (ll i = 0; i < n - l + 1; i++) {
      			ll j = i + l - 1;
      			
            	if (str[i] == str[j]) {
            		dp[i][j] = (dp[i + 1][j] + dp[i][j - 1] + 1);		
            	}
            	else {
            		dp[i][j] = (dp[i + 1][j] + dp[i][j - 1] - dp[i + 1][j - 1]);		
            	}
               
      		
      		}
      	}
      	
      	
      	cout << dp[0][n - 1] << endl;
   }
   return 0;
}
