
# Read input from stdin and provide input before running code

def fact(a):
    i = 1
    fact1 = 1
    while i <= a:
        fact1 = fact1*i
        i = i+1
    return fact1


t = int(raw_input())
for i in range(t):
    m = int(raw_input())
    k = fact(m)
    if len(str(k)) <= 5:
        print '{:05}'.format(k)
    elif k % 100000 == 0:
        print '{:05}'.format(0)
    else:
        print k % 100000
