#include<bits/stdc++.h>
using namespace std;
#define ll long long int 
int main()
{
	ll t,n,m;
	cin>>t;
	while(t--)
	{
		cin>>n>>m;
		ll count=0;
		vector<int>rs;
		while(m>0)
		{
			if(m&1)
			rs.push_back(count);
			count++;
			m>>=1;
		}
		for(ll i=rs.size()-1;i>=0;i--)
		{
			cout<<"("<<n<<"<<"<<rs[i]<<")";
			if(i!=0)
			cout<<" + ";
		}
		cout<<endl;
	}
	return(0);
}
