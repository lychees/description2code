#include <iostream>
#include <limits.h>
using namespace std;

int main()
{
    int t;
    string so,sd;
    cin>>t;
    while(t--){
    	cin>>so;
    	cin>>sd;
    	int ans = INT_MAX;
    	for(int i=0;i<so.length()-sd.length()+1;i++){
    		int count = 0;
    		for(int j=0;j<sd.length() && j+i<so.length();j++){
    			if(sd[j]!=so[j+i]) count++;
    		}
    		//cout<<count<<endl;
    		ans=min(ans,count);
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
