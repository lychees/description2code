#include <cstring>
#include <cassert>
#include <algorithm>
#include <iostream>
#include <cstdio>

using namespace std;

typedef long long llint;

#define FOR(i, a, b) for (int i = (a); i < (b); ++i)
#define REP(i, n) FOR (i, 0, n)
#define TRACE(x) cout << #x << " = " << x << endl;
#define _ << " _ " <<

void solve(void) {
  int n;
  scanf("%d", &n);
  vector<int> w(n), h(n);
  REP (i, n) scanf("%d", &w[i]);
  REP (i, n) scanf("%d", &h[i]);
  sort(h.begin(), h.end(), greater<int>());
  int val = w[0]*h[0];
  w.erase(w.begin());
  h.erase(h.begin());
  sort(w.begin(), w.end());
  REP (i, n-1)
    if (w[i]*h[i] >= val) {
      printf("NO\n");
      return;
    }
  printf("YES\n");
}

int main(void) {
  int t;
  scanf("%d", &t);
  while (t--) solve();

  return 0;
}
