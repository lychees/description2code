#define ll long long int
#define maxN 1000
#define maxVal 100000000
#define minVal -100000000
#define mod 1000000007LL

#define gcd(a,b) __gcd(a,b)

using namespace std;

ll n,m;
ll dp[maxN+5][maxN+5];

ll compute(ll i,ll j)
{
	if((i+j)==n)
	{
		if(i>j)
			return 1;
		return 0;
	}	
	if(dp[i][j]!=-1)
		return dp[i][j];
	ll ans;
	ans=compute(i+1,j);
	if(i>=(j+1)*m)
		ans=(ans+compute(i,j+1))%mod;
	return dp[i][j]=ans;
}

int main()
{
    ll t;
    scanf("%lld",&t);
    while(t--)
    {
    	scanf("%lld%lld",&n,&m);
    	memset(dp,-1,sizeof(dp));
    	printf("%lld\n",compute(0,0));
    }
    return 0;
}
