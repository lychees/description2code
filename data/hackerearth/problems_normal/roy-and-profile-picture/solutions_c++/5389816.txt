#include <iostream>
using namespace std;
int main()
{
    int i,l,n;
	cin>>l;
	cin>>n;
	int w[n],h[n];
	for(i=0;i<n;i++)
	{
		cin>>w[i]>>h[i];
	}
	for(i=0;i<n;i++)
	{
		if(w[i]==h[i] && w[i]>=l && h[i]>=l)
			cout<<"ACCEPTED\n";
		else if(w[i]<l || h[i]<l)
			cout<<"UPLOAD ANOTHER\n";
		else
			cout<<"CROP IT\n";
	}
    return 0;
}
