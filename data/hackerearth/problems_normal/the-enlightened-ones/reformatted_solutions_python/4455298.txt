(n, k) = map(int, raw_input().split(' '))

pos = sorted(map(int, raw_input().split(' ')))
low = 0
high = 10000000

while low < high:
    # mid denotes the enlightening value
    mid = (low + high) >> 1

    # check if everything can be done with it
    possible = True
    monkCount = 1
    monk = pos[0] + mid  # first monk pos
    for p in range(1, n):
        if pos[p] > monk + mid:
            monk = pos[p] + mid
            monkCount += 1
            if monkCount > k:
                possible = False
                break

    if possible:
        high = mid
    else:
        low = mid + 1

print high
