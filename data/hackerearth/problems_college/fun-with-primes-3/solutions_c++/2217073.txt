#include <iostream>
using namespace std;
#include <algorithm>
#include <vector>
int main()
{
    vector<long long> a(1000001, 0);
    for(long long i=2; i<1000001; ++i)
    {
        for(long long j=2; i*j<1000001; ++j)
        {
            a[i*j]=1;
        }
    }
    a[1]=1;
    long long t;
    cin>>t;
    while(t--)
    {
        long long x;
        cin>>x;
        if(x==1)
        {
            cout<<"NO\n";
            continue;
        }
        long long sum=0;
        while(x>0)
        {
            sum += x%10;
            x/=10;
        }

        if(!a[sum])
        {
            cout<<"YES";
        }
        else
        {
            cout<<"NO";
        }
        cout<<endl;

    }
}
