def print_result(str_arr):
    letter_count_dic = {}

    for i in range(len(str_arr)):
        letter = str_arr[i]

        if letter_count_dic.has_key(letter):
            count = letter_count_dic[letter]
            letter_count_dic[letter] = count + 1
        else:
            letter_count_dic[letter] = 1

    count_letters_dic = {}
    for letter in letter_count_dic.keys():
        count = letter_count_dic[letter]

        if count_letters_dic.has_key(count):
            letters = count_letters_dic[count]
            letters.append(letter)
        else:
            count_letters_dic[count] = [letter]

    counts = sorted(count_letters_dic.keys())

    result_str = ''

    for count in counts:
        letters = sorted(count_letters_dic[count])

        for letter in letters:
            result_str += letter*count

    print result_str


def run_test():
    n = int(raw_input())

    for i in range(n):
        string_to_sort = str(raw_input())
        print_result(string_to_sort)


run_test()
