#include <iostream>
#include <stdlib.h>
#include <string.h>
using namespace std;
#define MAX 1000000

int main()
{
    //cout << "Hello World!" << endl;
    long long int n,k,flag = 0;
    cin>>n>>k;
    long long int arr[MAX]; 
    for(int i=0;i<n;i++)
    	cin>>arr[i];
    int *tmp = (int *)calloc(MAX, sizeof(long long int));
    for(int i=0;i<n;i++){
    	if( k-arr[i] > 0 && tmp[k-arr[i]] == 1){
    		flag = 1;
    		break;
    	}
    	tmp[arr[i]] = 1;
    }
    if(flag)
    	cout<<"YES";
    else
    	cout<<"NO";
    	free(tmp);
    return 0;
}
