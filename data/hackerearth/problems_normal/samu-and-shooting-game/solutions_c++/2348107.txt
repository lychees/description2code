double p1,p2;

int was[2000][2000];
double dp[2000][2000];

double solve(int n,int w)
{
	if (w<=0)
		return 1;
	if (was[n][w])
		return dp[n][w];
	was[n][w]=1;
	double res=0;
	if (n>=1)
	{
		res=max(res,solve(n-1,w-x)*p1+solve(n-1,w)*(1.0-p1));
	}
	if (n>=1)
		res=max(res,solve(n-1,w-y)*p2+solve(n-1,w)*(1.0-p2));
	dp[n][w]=res;
	return res;
}
