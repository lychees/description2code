#include<stdio.h>
#include<iostream>
#include<string.h>
using namespace std;
int main()
{
    int t,j,i;
    scanf("%d",&t);
    for(j=1;j<=t;j++)
    {
        char s[105];
        scanf("%s",s);
        int l=strlen(s);
        int flag=0;
        l--;
        i=0;
        while(i<l)
        {
            if(s[i]!=s[l])
            {flag=1;
            break;}
            i++;
            l--;
        }
        if(flag==0)
        cout<<"YES\n";
        else
        cout<<"NO\n";

    }
    return 0;
}
