typedef long long int ll;
const int mx = 1e6 + 10;
using namespace std;

int n, v[mx];
int maxSum = 0, curSum = 0;
int main(){

    scanf("%d", &n);
    for(int i = 0; i < n; i++){
        scanf("%d", &v[i]);
        curSum += v[i];
        if(curSum > maxSum)maxSum = curSum;
        else if(curSum <= 0)curSum = 0;
    }
    
    printf("%d\n", maxSum);
    return 0;
}
