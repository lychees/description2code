for _ in xrange(input()):
    s = raw_input()
    l = len(s)
    d = dict()
    for x in s:
        if d.has_key(x):
            d[x] = d[x] + 1
        else:
            d[x] = 1
    count = 0
    for x in d:
        if d[x] % 2 != 0:
            count = count + 1
    if count < 2:
        print '0'
    else:
        print count - 1
