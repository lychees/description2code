#include<bits/stdc++.h>
using namespace std;
int main(){
	int n;
	cin>>n;
	int graph[n+1][n+1];
	for(int i=1;i<=n;i++){
		for(int j=1;j<=n;j++){
			graph[i][j]=0;
		}
	}
	int q;
	cin>>q;
	while(q--){
		int a,b;
		cin>>a>>b;
		graph[a][b]=1;
		graph[b][a]=1;
	}
	cin>>q;
	while(q--){
		int a,b;
		cin>>a>>b;
		if(graph[a][b]==1){
			cout<<"Direct Connection"<<endl;
			continue;
		}
		else{
			queue<int> q;
			int visited[n+1];
			int done=0;
			memset(visited,0,sizeof(int)*(n+1));
			q.push(a);
			while(!q.empty()){
				int top=q.front();
				visited[top]=1;
				if(graph[top][b]){
					done=1;
					cout<<"Indirect Connection"<<endl;
					break;
				}
				else{
					q.pop();
					for(int i=1;i<=n;i++){
						if(graph[top][i]==1&&visited[i]==0){
							q.push(i);
						}
					}
				}
			}
			if(!done){
				cout<<"No Connection"<<endl;
			}
		}
	}
}
