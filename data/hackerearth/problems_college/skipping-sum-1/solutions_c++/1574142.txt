#include <iostream>
#include <vector>
#include <stdio.h>
using namespace std;

long a[100005], s[100005];
vector<long> v2[2];
vector<long> v3[3];
vector<long> v4[4];
vector<long> v5[5];
vector<long> v6[6];
vector<long> v7[7];
vector<long> v8[8];
vector<long> v9[9];
vector<long> v10[10];
int n,q;

void fill(vector<long> v[], int k)
{
	long sum = 0;
	for(int start=0; start<k; start++) {
		sum = 0;
		for(int l=start; l<n; l+=k) {
			if(l==start)
				v[start].push_back(a[l]);
			else
				v[start].push_back(a[l] + sum);
			sum += a[l];
		}
	}
}

int main()
{
	cin>>n>>q;
	
	for(int i=0;i<n;i++) {
		scanf("%ld",&a[i]);
		if(i == 0) s[i] = a[i];
		else s[i] = s[i-1] + a[i];
	}

	fill(v2, 2);
	fill(v3, 3);
	fill(v4, 4);
	fill(v5, 5);
	fill(v6, 6);
	fill(v7, 7);
	fill(v8, 8);
	fill(v9, 9);
	fill(v10, 10);
	
	for(int i=0;i<q;i++) {
		int l,r,k;
		//cin>>l>>r>>k;
		scanf("%d%d%d",&l,&r,&k);
		l--; r--;
		long ans = 0;
		int c,start,end;
		
		c = l%k;
		start = l/k;
		end = start + (r-l)/k;
	
		//cout<<c<<" "<<start<<" "<<end<<endl;

		switch(k) {
			case 1:
				if(l == 0) ans = s[r];
				else ans = s[r] - s[l-1];
				break;
			case 2: 
			{
				if(start == 0) ans = v2[c][end];
				else ans = v2[c][end] - v2[c][start-1];
				break;	
			}
			case 3:
			{
				if(start == 0) ans = v3[c][end];
				else ans = v3[c][end] - v3[c][start-1];
				break;	
			}
			case 4:
			{
				if(start == 0) ans = v4[c][end];
				else ans = v4[c][end] - v4[c][start-1];
				break;	
			}
			case 5:
			{
				if(start == 0) ans = v5[c][end];
				else ans = v5[c][end] - v5[c][start-1];
				break;	
			}
			case 6:
			{
				if(start == 0) ans = v6[c][end];
				else ans = v6[c][end] - v6[c][start-1];
				break;	
			}
			case 7:
			{
				if(start == 0) ans = v7[c][end];
				else ans = v7[c][end] - v7[c][start-1];
				break;	
			}
			case 8:
			{
				if(start == 0) ans = v8[c][end];
				else ans = v8[c][end] - v8[c][start-1];
				break;	
			}
			case 9:
			{
				if(start == 0) ans = v9[c][end];
				else ans = v9[c][end] - v9[c][start-1];
				break;	
			}
			case 10:
			{
				if(start == 0) ans = v10[c][end];
				else ans = v10[c][end] - v10[c][start-1];
				break;	
			}
		}
		
		printf("%ld\n",ans);
	}
    return 0;
}
