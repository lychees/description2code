#include <bits/stdc++.h>
using namespace std;

int main()
{
   	int t;
   	cin >> t;
   	while(t--) {
   		string s, t;
   		cin >> s >> t;
   		int n = s.size();
   		int m = t.size();
   		int i = 0, cnt = 0;
   		while(i < n && i < m) {
   			if(s[i] == t[i]) cnt++;
   			i++;
   		}
   		cout << cnt << endl;
   	}
    return 0;
}
