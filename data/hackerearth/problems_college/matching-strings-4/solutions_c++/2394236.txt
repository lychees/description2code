#include<bits/stdc++.h>
using namespace std;
int main()
{
    int test,sum;
    scanf("%d",&test);
    while(test--)
    {
        sum=0;
        string str1,str2;
        cin>>str1>>str2;
        int i=0,j=0;
        while(str1[i]&&str2[j])
        {
            if(str1[i]==str2[j])
                sum++;
            i++,j++;
        }
        cout<<sum<<endl;
    }
}
