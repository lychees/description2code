#include <iostream>
using namespace std;

bool isPrime(int x)
{
	for(int i=2; i<x; i++)
	{
		if(x%i==0)
			return false;
	}
	return true;
}

int main()
{
	long long int n;
	cin >> n;
	if(n < 2)
	{
		cout << 0;
		return 0;
	}
	long long int tot = 1;
	long long int cur = 1;
	
	for(int i = 3; i <= n ; i++)
	{
		if(isPrime(i))
		{
			cur*=2;
		}
		tot += cur;
	}
	cout << tot;
	return 0;
}
