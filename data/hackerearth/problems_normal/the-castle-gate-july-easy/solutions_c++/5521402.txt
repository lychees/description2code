#include <iostream>
using namespace std;

int main()
{
    int t, n, num;
    cin >> t;
    while(t--){
    	cin >> n;
    	num = 0;
    	for (int i = 1; i <= n-1 ; i++){
    		for(int j = i+1; j <= n; j++){
    			if ( (i^j) <= n){
    				num++;
    			}
    		}
    	}
    	cout << num << endl;
    }
    return 0;
}
