// week limits so brute force :P
#include<bits/stdc++.h>
using namespace std;
static int boxes[50001];
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		int n,m;
		scanf("%d %d",&n,&m);
		for(int i=0;i<n;i++){
			boxes[i]=0;
		}
		int start=0,end=n-1;
		for(int i=0;i<m;i++){
			int mi=100000,ind=0;
			if(i%2==0)
			for(int j=0;j<=n/2;j++){
				if(boxes[j]<mi){
					mi=boxes[j],ind=j;
				}
			}
			else
			for(int j=n-1;j>=n/2;j--){
				if(boxes[j]<mi){
					mi=boxes[j],ind=j;
				}
			}
			if(i==m-1){
				printf("%d %d\n",ind+1,boxes[ind]);
			}
			boxes[ind]++;
		}
	}
	return 0;
}
