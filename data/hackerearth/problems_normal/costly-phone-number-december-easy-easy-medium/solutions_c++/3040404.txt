#include <bits/stdc++.h>
using namespace std;

int arr1[10];
string str;
int main()
{
	long long int t,i,j,n,k,s;
	scanf("%d",&t);
	while(t--)
	{
		for(i=0;i<10;i++){
			scanf("%lld",&arr1[i]);
		}
		scanf("%lld",&s);
		cin>>str;
		while(1){
		int seen=0;	
		for(i=0;i<10;i++){
			for(j=0;j<10;j++){
				k=(j+i)%10;
				if(arr1[j]+arr1[i]<arr1[k]) {
					arr1[k]=arr1[i]+arr1[j];
					seen=1;
			}
		}
		}
		if(seen==0) break;
		}
		k=0;
		for(i=0;i<s;i++){
			k+=arr1[str[i]-'0'];
		}
		printf("%lld\n",k);
	}
    
}
