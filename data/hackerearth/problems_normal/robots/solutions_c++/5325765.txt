#include <bits/stdc++.h>
#define ll long long int
#define sc1(x) scanf("%d",&x)
#define sc2(x,y) scanf("%d%d",&x,&y)
#define scll(x) scanf("%lld",&x)
#define pint(c) printf("%d",c)
#define pll(c) printf("%lld",c)
#define ps() printf(" ")
#define pn() printf("\n")

#define vi vector<int>
#define vii vector<pair<int,int> >
#define mp make_pair
#define pb push_back

//loops
#define ff(i,n,a) for(i=a;i<n;++i)
#define fb(i,n,a) for(i=n,i>=a;--i)

//constants
const int mxn=1e5+1;
const int MOD=1e9+7;
using namespace std;
int main()
{
	int t=1;
	sc1(t);
	while(t--)
	{
		int n;
		cin>>n;
		int A[n];
		int mx=0;
		for(int i=0;i<n;++i)
		{
			cin>>A[i];
			mx=max(mx,A[i]);
		}	
		if(mx<n)
		{
			cout<<0<<endl;
			continue;
		}
		sort(A,A+n);
		ll prod=A[0];
		for(int i=1;i<n;++i)
		{
			prod=(prod*(A[i]-i))%MOD;
		}
		cout<<prod<<endl;
	}
	return 0;
}
