#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    long long int sum1,sum2;
    sum1=0;
    sum2=0;
    int arr[100][100];
    for(int i=0;i<n;i++)
       for(int j=0;j<n;j++){
           cin>>arr[i][j];
          if(i==j)
              sum1+=arr[i][j];
            if(i+j==n-1){
            	sum2+=arr[i][j];
            }    
       }
         // cout<<sum1<endl<<sum2<<endl;
       cout<<abs(sum1-sum2)<<endl;
       
}
