#include <iostream>
#include<algorithm>
#include <bitset>
#define MAX 1000000000
using namespace std;
bitset<1000000004> a;
int main()
{
  int t;
  long long k;
  cin>>t;
  a[1]=1;
  for(int i=2;i<=sqrt(MAX);i++)
  {
  	
  	k=i*i;
  	if(a[k]==0)
  	{
	  	a[k]=1;
	  	for(int j=2;k*j<=MAX;j++)
	  	{
	  		if(a[k*j]==0)
	  		a[k*j]=1;
	  	}
  		
  	}
  }
  long long x,y,count,temp;
  while(t--)
  {
  	
  	cin>>x>>y;
  	count=(x-y);
  	if(count<0)
  	count=-1*count;
  	count+=1;
  	if(x>y)
  	{
  		temp=x;
  		x=y;
  		y=temp;
  	}
  	for(long long int i=x;i<=y;i++)
  		if(a[i]==1)
  		count--;
  	cout<<count<<endl;
  }
    
}
