#include <iostream>
#include<algorithm>
using namespace std;
int cmp(int a,int b)
{
	return a>b;
}
int main()
{
   int t;
   long n,a[100000],c[100000],j,i;
   long s,k;
   cin>>t;
   while(t--)
   {
   	  cin>>n;s=0;k=0;
   	  for(i=0;i<n;i++)
   	  {
   	  	cin>>a[i];
		 }
		 for(i=0;i<n;i++)
   	  {
   	  	cin>>c[i];
		 }
		sort(c,c+n,cmp);
		for(i=0;i<n;i++)
		{
			s=s-a[i];
			if(s<0)
			{
				s=s+c[k++];
			    // cout<<"get include"<<c[k-1];
			}
			
		}
		
	  
      cout<<k<<endl;
    }
}
