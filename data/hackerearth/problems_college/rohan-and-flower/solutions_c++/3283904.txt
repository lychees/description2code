#include <iostream>
using namespace std;

long long int a[4000005];

int main()
{
    long long int n, p, k, i, j, ans, b[4005], a, tmp;
    
    scanf("%lld", &n);
    scanf("%lld", &p);
    
    for (i = 0; i < n; i++) {
    	cin>>b[i];
    }
    
    a = 0;
    for (i = 0; i < n; i++) {
    	for (j = i + 1; j < n; j++) {
    		tmp=b[i]*b[j];
    	if(tmp%p==0)
    	{
    		a+=(n-j-1);
    		continue;
    	}
    		for (k = j + 1; k < n; k++) {
    			ans = (((b[i] * b[j]) % p) * b[k]) % p;
    			if (ans % p == 0) {
    				a++;
    			}
    		}
    	}
    }
    
    printf("%lld\n", a);
    
    return 0;
}
