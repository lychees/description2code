dp = [0]*102
for i in range(8):
    dp[i] = -1
dp[6] = 1
dp[8] = 1

for i in range(9, 101):
    a = i-8
    b = i-6

    if (dp[a] == -1 and dp[b] == -1):
        dp[i] = -1
    elif (dp[a] != -1 and dp[b] != -1):
        dp[i] = 1+min(dp[a], dp[b])
    else:
        if (dp[a] == -1):
            dp[i] = 1+dp[b]
        else:
            dp[i] = 1+dp[a]

t = int(raw_input())
while t:
    t -= 1
    n = int(raw_input())
    print dp[n]
