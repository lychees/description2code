#include<bits/stdc++.h>
using namespace std;

typedef struct sats
{
int age;
string name;
}sats;

int compare(sats a , sats b)
 {
       return a.age<b.age;
 }

int main()
{
int t; cin>>t; while(t--)
{
int n,roll,i;
cin>>n;
sats c[n];
for(i=0;i<n;i++)
 {
       cin>>c[i].age>>c[i].name;
 }
 sort(c,c+n,compare);
//qsort(c,n,sizeof(sats),compare);
cin>>roll;
cout<<c[roll-1].name<<"\n";
}
return 0;
}
