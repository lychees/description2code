bool isPrime(int no){
    int flag=0;
    if(no==0||no==1)
        flag=1;
    for(int i=2;i<=sqrt(no);i++){
        if(no%i==0){
                flag=1;
            break;
        }
        else{
            flag=0;
        }
    }
   //cout<<no<<" "<<flag<<endl;
    if(flag==1) return false;
    else return true;
}
