#include <iostream>
#include<string>
using namespace std;
int check(string s)
{
	int temp=s.length(),i;
	for(i=0;i<temp/2;i++)
    {
    	if(s[i]!=s[temp-i-1])
    	break;
    }
    if(i==temp/2)
    return 1;
    else
    return 0;
}
int main()
{
    string s1,s2;
    int t;
    cin>>t;
    while(t--)
    {
    	cin>>s1>>s2;
    	if(check(s1+s2) || check(s2+s1))
    	cout<<"YES\n";
    	else
    	cout<<"NO\n";
    }
    return 0;
}
