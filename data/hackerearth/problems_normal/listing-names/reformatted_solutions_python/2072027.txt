names = []
for i in range(input()):
    names.append(raw_input())
names = sorted(names)
count = 1
for i in range(len(names)-1):
    if names[i] == names[i+1]:
        count = count+1
    else:
        print names[i], count
        count = 1
l = len(names)-1
if names[l] != names[l-1]:
    print names[l], "1"
