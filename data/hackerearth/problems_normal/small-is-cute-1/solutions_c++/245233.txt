ll mpow(ll a, ll n)
{
    ll ret=1;
    ll b=a;
    while(n)
    {
        if(n&1)
        ret=(ret*b)%MOD;
        b=(b*b)%MOD;
        n>>=1;
    }
    return (ll)ret;
}

ll sum(int a,int b,int n)
{
    vector<int> v;
    int i;
    for(i=a;i<n;i+=a)
    v.push_back(i);

    for(i=b;i<n;i+=b)
    v.push_back(i);

    set<int> s(v.begin(),v.end());
    v.assign(s.begin(),s.end());

    ll cnt=0;
    for(i=0;i<v.size();i++)
    {
        //printf("%d ",v[i]);
        cnt+=v[i];
    }
    return cnt;
}
