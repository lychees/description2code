#include <iostream>
#include<algorithm>
//#include<conio.h>
using namespace std;

typedef unsigned long long int A;

struct node
{
   public:
          
   A time;
   A score;
      
};

int mycomp(const node& a ,const  node& b)
{
     return a.score > b.score;
}

int main()
{
  int n ,q,x;
  cin>>n>>q;
  
  node arr[n];

  for(int i = 0 ; i <n; ++i)
  {
    cin>>arr[i].time;
  }

  for(int i = 0 ; i <n; ++i)
  {
    cin>>arr[i].score;
  }
  
  sort(arr,arr+n,mycomp);
  

  int max = 0;
  int karr[q];
  
  

  for(int i =0 ; i< q;++i)
  {
        cin>>karr[i];
        if(karr[i] > max)
        {
             max = karr[i];
        }
  }
  
  A cum[max+1];
  
  cum[0] = arr[0].time;
  for(int i = 1 ; i<=max ; ++i)
  {
          cum[i] = cum[i-1] + arr[i].time;
  }


 for(int i =0 ; i< q;++i)
  {
         
         cout<<cum[karr[i]-1]<<"\n";
  }
  
//getch();

          return 0;
}
