#include <iostream>
#include<algorithm>
#include<stdio.h>
using namespace std;

long long int combination(long long int n)
{
	long long int r;
	if(n==3)
		return 1;
	
	r=(n)*(n-1)*(n-2);
	r=r/6;
	
	return r;
}

int main()
{
	int t;
	cin>>t;
	while(t--)
    {
    	long long int n,k,count=0,num,i;
    	scanf("%lld%lld",&n,&k);
    	
    	for(i=0;i<n;i++)
    	{
    		scanf("%lld",&num);
    		if(num==k)
    			count++;
    	}
    	
    	long long int numerator=combination(count),denominator=combination(n);
    	
    	if(numerator==denominator)
    	{
    		printf("1\n");
    		continue;
    	}
    	
    	long long int gcd= __gcd(numerator,denominator);
    	
    	if(gcd==1)
    	{
    		printf("%lld/%lld\n",numerator,denominator);
    		continue;
    	}
    	
    	printf("%lld/%lld\n",(numerator/gcd),(denominator/gcd));
    	
    }
    
    return 0;
}
