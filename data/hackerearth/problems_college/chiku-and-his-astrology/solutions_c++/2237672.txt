#include <iostream>
#include<string.h>
#include<math.h>
using namespace std;

int main()
{
    int t,c,c1;
    string s;
    cin>>t;
    while(t--)
    {
    	cin>>s;
    	int c=0;
    	for(int i=0;i<s.length();i++)
    	{
    		if(s[i]=='1')
    		c++;
    	}
    	c1=s.length()-c;
    	cout<<abs(c-c1)<<"\n";
    }
    return 0;
}
