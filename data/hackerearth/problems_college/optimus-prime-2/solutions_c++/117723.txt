#include<stdio.h>
#define ULL unsigned long long
ULL Range[1000141];
using namespace std;
ULL fastread()
{
 ULL num=0;
 char c;
 while((c=getchar())<48);
  num+=c-'0';
 while((c=getchar())>='0')
 {
 	num*=10;
 	num+=c-'0';
 } 	
 return num;
}
int main()
{
//	freopen("Output_Sample.txt","w",stdout);
//	freopen("Input_Sample.txt","r",stdin);
    int T;
    scanf("%d",&T);
	  //cin>>T;
    while(T--)
    {
        ULL i,j,m,n,Start;
	       m=fastread();
	       n=fastread();
        ULL z;
        for(i=m,z=0;i<=n;i++,z++)
        {
            Range[z] = i;
        }
        for(i=2;i*i<=n;i++)
        {
            Start=m/i;
            Start*=i;
            for(j=Start;j<=n;j+=i)
            {
                if(j!=i && j>=m)
                Range[j-m] = 0;
            }
        }
        if(m==1)
        {
            Range[0] = 0;
        }
        ULL cnt=0,cnt2=0;
        for(i=0;i<z;i++)
        {
            if(Range[i])
            {
                cnt=0;
                j=Range[i];
                while(j!=0)
                {
                    n=j%10;
                    if(n==2 || n==3 || n==5 || n==7)
                    {
                        cnt++;
                    }
                    j/=10;
                }

                if(cnt==2 || cnt==3 || cnt==5 || cnt==7 || cnt==11)
                {
//                    printf("%d\n",Range[i]);
//					  cout<<Range[i]<<endl;
                    cnt2++;
                }
            }
        }
        printf("%d\n",cnt2);
    }
    return 0;
}
