#include <iostream>
#include <cstdio>
#include <vector>
#include <string>

#define MAX(x,y) x>y?x:y
#define MIN(x,y) x<y?x:y
#define pb push_back
#define mp make_pair
#define rep(i,n) for(int i=0;i<n;i++)
#define repr(i,j,n) for(int i=j;i<=n;i++)
#define endl '\n'
using namespace std;
typedef long long ll;
typedef vector<ll> vll;
const ll maxn = 10000000;
//ll a[maxn];
const ll mod = 1e9+7;
ll dp[1024][1024];

int main()
{
	std::ios::sync_with_stdio(0);
	//freopen("input.in","r",stdin);
	ll i,j,k,t,q,m,l,r,n;
	dp[0][0]=1;
	for(i=1;i<=1000;i++)
	{
		for(j=0;j<=i;j++)
		{
			if(j==0)dp[i][j]=1;
			else
			{
				if(j>i/2)dp[i][j]=dp[i][i-j];
				else
				{
					dp[i][j]=((dp[i-1][j-1]%mod)+(dp[i-1][j]%mod))%mod;
				}
			}
		}
	}
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		ll sum=0;
		for(i=0;i<=k;i++)
		{
			sum+=dp[n][i];
			if(sum>=mod)sum%=mod;
		}
		cout<<sum<<endl;
	}
}
