#define LOCALTEST 0 // change this to 1 to perform localtest on "in.txt"
int n;
int cur[SIZE], tar[SIZE], pos[SIZE];
int fen[SIZE];
int add(int ind, int v) {
    ind++;
    for (;ind<SIZE;ind+=ind&-ind) fen[ind]+=v;
}
int getval(int ind) {
    ind++;
    int ans=0;
    for (;ind>0;ind-=ind&-ind) ans+=fen[ind];
    return ans;
}
int getrange(int a, int b) {
    return getval(b)-getval(a-1);
}
int main(){
    if (LOCALTEST) {
        freopen("in.txt", "r", stdin);
        freopen("out.txt", "w", stdout);
    }
    CASET {
        RI(n);
        REP(i,n) {
            RI(cur[i]);
            pos[cur[i]]=i;
        }
        REP(i,n) RI(tar[i]);
        MS0(fen);
        int ans=0;
        REP(i,n) {
            int p=pos[tar[i]];
            int s=getval(p);
            ans+=abs(p+s-i);
            //printf("i %d added %d, cuz p %d s %d i %d ",i,abs(p+s-i),p,s,i);
            //REP(i,n) printf("%d%c",getval(i),i==n-1?'\n':' ');
            add(0,1);
            add(p+1,-1);
        }
        printf("%d\n",ans);
    }
    return 0;
}
