#include <iostream>
using namespace std;

void merge(long int a[],long int b[],int n,int m)
{
	int l=n+m;
	long int c[l];
	int k=0,i=0,j=0;
	while(i<n && j<m)
	{
		if(a[i]>b[j])
		c[k++]=a[i++];
		else
		c[k++]=b[j++];
	}

		while(j<m)
		c[k++]=b[j++];

	
		while(i<n)
		c[k++]=a[i++];

	for(i=0;i<k;i++)
	cout<<c[i]<<" ";
	
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,m;
    	cin>>n>>m;
    	long int a[n],b[m];
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    		for(int i=0;i<m;i++)
    	{
    		cin>>b[i];
    	}
    	merge(a,b,n,m);
    	cout<<endl;
    }
    return 0;
}
