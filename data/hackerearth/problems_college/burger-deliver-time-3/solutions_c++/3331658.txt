#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n, a, b, i;
    // cout << "Hello World!" << endl;
    cin >> n;
    vector<pair<int, int> > v;
    for (i = 1; i <= n; i++) {
    	cin >> a >> b;
    	v.push_back(make_pair(a + b, i));
    }
    
    sort(v.begin(), v.end());
    
    for (i = 0; i != v.size(); i++) {
    	cout << v[i].second << ' ';
    }
    
    
    return 0;
}
