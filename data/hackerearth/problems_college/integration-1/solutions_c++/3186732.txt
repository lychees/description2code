#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll long long int
#define pb push_back
#define mk make_pair
ll power(ll a, ll b) {
ll x = 1, y = a;
    while(b > 0) {
        if(b%2 == 1) {
            x=(x*y);
            if(x>mod) x%=mod;
        }
        y = (y*y);
        if(y>mod) y%=mod;
        b /= 2;
    }
    return x;
}
int main()
{
	int t,tc = 1;
	ll n,p,u,r,s,i;
	cin>>t;
	while(t--) {
		cin>>n>>p>>u>>r>>s;
		for(i = 0; i < n; i++) {
			if(p+r <= u) {
				p+=r;
			}
			else if(p-s >= 0) {
				p -= s;
			}
			else break;
		}
		cout<<"Case #"<<tc<<": ";
		cout<<p<<endl;
		tc++;
	}
	return 0;
}
