from fractions import gcd
import sys
t = int(sys.stdin.readline())

while (t > 0):
    t -= 1
    N, K = sys.stdin.readline().split()
    k = 0
    for i in sys.stdin.readline().split():
        if (i == K):
            k += 1
    n = int(N)
    r1 = k*(k-1)*(k-2)
    r2 = n*(n-1)*(n-2)
    g = gcd(r1, r2)
    r1 /= g
    r2 /= g
    print "{0}/{1}".format(r1, r2)
