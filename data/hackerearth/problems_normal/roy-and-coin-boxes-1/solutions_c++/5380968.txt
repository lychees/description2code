#include <bits/stdc++.h>
using namespace std;
#define ll long long

int main()
{
	int n;
	ios_base::sync_with_stdio(false);
	cin>>n;
	int a[n+2];
	memset(a,0,sizeof(a));
	int m;
	cin>>m;
	while(m--)
	{
		int l,r;
		cin>>l>>r;
		a[l-1]++;
		a[r]--;
	}
	for(int i=1;i<n;i++)
	{
		a[i] = a[i-1]+a[i];
		//cout<<a[i]<<" ";
	}
	sort(a,a+n);
	int q;
	cin>>q;
	while(q--)
	{
		int x;
		cin>>x;
		int ans = lower_bound(a,a+n,x)-a;
		cout<<n-ans<<endl;
	}
	return 0;
}
