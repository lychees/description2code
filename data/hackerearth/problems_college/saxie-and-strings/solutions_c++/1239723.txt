#include <iostream>
#include<algorithm>
#include<cstring>
using namespace std;

int main()
{
    int n;
    cin>>n;
    while(n--)
    {
        string s;
        cin>>s;string t;
        t=s;
        reverse(s.begin(),s.end());
        if(t.compare(s)==0)
        cout<<"YES"<<endl;
        else
        cout<<"NO"<<endl;
    }
    return 0;
}
