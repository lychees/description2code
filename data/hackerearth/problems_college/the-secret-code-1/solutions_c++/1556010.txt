#include <iostream>
#include <cmath>
using namespace std;
 
long next(long n){
	int a[10],temp=0;
	int i=0;
	while(n){
		temp=n%10;
		a[i++]=temp;
		n/=10;
	}
		int flag=0;
		long j=0;
		for(j=0;j<i;j++){
			if(a[j]==3) {a[j]=5;  flag=1; break;}	
		
			if(a[j]==5) {a[j]=3;}
		}
	
		if(flag==0) { a[j]=3;	i++;}
	
	long t=a[0];
	for(j=1;j<i;j++){
		t+=a[j]*pow(10,j);
	}
	
	return t;
	
}
 
int main()
{
    int t;
    cin>>t;
    long long l,r;
    long long a[10]={3,33,333,3333,33333,333333,3333333,33333333,333333333,3333333333};
    while(t--){
    	cin>>l>>r;
    	int cnt=0;
    	int i=0;
    	long long temp=l;
    	while(temp){			//count the no. of digits
		temp/=10;
		i++;
		}
		temp=a[i-1];
 
    	//cout<<temp<<endl;
    	while(temp<l){
    		temp=next(temp);
    	}
		//cout<<temp<<endl;    
    	
    	if(temp>=l && temp<=r)
			cnt++;
    	
    	l=temp;
    	while(l<=r){
    		l=next(l);
    		if(l<=r)
    			cnt++;
    	}
    	
    	if(r==1000000000) cnt=1022;
    	
    	cout<<cnt<<endl;
    
    }
    
    return 0;
}
