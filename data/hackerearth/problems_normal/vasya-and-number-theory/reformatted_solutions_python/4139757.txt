def largest(ints, sz):
    # dp = [[0 for i in xrange(sz)] for j in xrange(sz)]
    dp = [1 for i in xrange(sz)]
    for i in xrange(sz):
        for j in xrange(i+1, sz):
            if ints[j] % ints[i] == 0:
                dp[j] = max(dp[j], dp[i] + 1)
    # print 'A = ', ints
    # print 'DP =', dp
    return max(dp)


N = int(raw_input())
A = sorted(map(int, raw_input().split()))
res = largest(A, N)
print res if res != 1 else -1
