#include <bits/stdc++.h>
using namespace std;
vector<long long int>cats;
vector<long long int>cals;
int main()
{
    int n,i,num;
    long long unsigned int ans;
    cin>>n;
    ans=0;
    for(i=0;i<n;i++)
    {
    	cin>>num;
    	cats.push_back(num);
    }
    for(i=0;i<n;i++)
    {
    	cin>>num;
    	cals.push_back(num);
    }
    sort(cats.begin(),cats.end());
    sort(cals.begin(),cals.end());
    for(i=0;i<n;i++)
    ans+=cats[i]*cals[i];
    cout<<ans<<"\n";
    return 0;
}
