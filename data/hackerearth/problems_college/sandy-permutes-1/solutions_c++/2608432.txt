#include <iostream>
using namespace std;

int main() {
	long long int n,k,i,j,l,m;
	long long int arr[100001];
	cin>>n>>k;
	for(i=0;i<n;i++)
		cin>>arr[i];
		
	if(k>=n){
		for(i=0;i<n;i++)
			cout<<n-i<<" ";
		return 0;	
	}
		
	for(i=0;i<k;i++){
		m=arr[i];
		l=i;
		for(j=i;j<n;j++){
			if(arr[j]>m){
				m= arr[j];
				l=j;
			}
		}
		if(arr[i]==m && k+1<=n){
			k++;
		}
		else{
			m= arr[l];
			arr[l]= arr[i];
			arr[i]= m;
		}
	}
	// cout<<endl;
	
	for(i=0;i<n;i++)
		cout<<arr[i]<<" ";
	
	
	return 0;
}
