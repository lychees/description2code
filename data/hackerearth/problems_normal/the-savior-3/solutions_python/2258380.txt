def check(li=[]):
    ll = len(li)
    count = 0
    for l in range(ll):
        for m in range(l + 1, ll):
            if li[l] == li[m]:
                continue
            if (li[l] + li[m]) % 2 == 0:
                count += 1
    return count

tn = input()

for _ in range(tn):
    m = input()
    mm = raw_input().strip().split(' ')
    mm = [int(ii) for ii in mm]
    print check(mm)
