#include <bits/stdc++.h>
using namespace std;
inline int Fn(unsigned int x, unsigned int y)
{
    int ans = 0, temp;
    while(x != 1 || y != 1)
    {
        temp = x;
        x = y-x;
        y = temp;
        if(x > y)
            swap(x,y);
        ans++;
    }
    return ans;
}
int main()
{
    ios::sync_with_stdio(false);
    int T, L, min0;
    cin >> T;
    while(T--)
    {
        cin >> L;
        if(L == 1)
            cout << 0 << endl;
        else
        {
            min0 = INT_MAX;
            for(int i=1;i<=L/2;++i)
                if(__gcd(i,L-i) == 1)
                    min0 = min(min0, Fn(i,L-i));
            cout << 1 + min0 << endl;
        }
    }
    return 0;
}
