#include<iostream>
#include<string.h>
using namespace std;

int main()
{
    long long int size,i;
    string str;
    cin>>str;
    size=str.length();
    for(i=0;i<size-1;i++)
        cout<<str[i]<<"-";
    cout<<str[size-1]<<endl;
    for(i=size-1;i>0;i--)
    {
        cout<<str[i]<<"-";
    }
    cout<<str[0];
    return 0;
}
