#include<bits/stdc++.h>
using namespace std;

typedef long long ll;
const int md=1e9+7;
const int N=1e5+10;

int a[N];

int powmod(int a,int b) {
  if(b==0) {
    return 1;
  }
  int p=powmod(a,b/2);
  p=1ll*p*p%md;
  if(b&1) {
    p=1ll*p*a%md;
  }
  return p;
}
int main() {
  int tt;
  scanf("%d",&tt);
  while(tt--) {
    int n;
    scanf("%d",&n);
    ll temp=0;
    for(int i=0;i<n;i++) {
      scanf("%d",a+i);
      temp|=a[i];
    }
    temp%=md;
    temp=temp*powmod(2,n-1)%md;
    printf("%lld\012",temp);
  }
  return 0;
}
