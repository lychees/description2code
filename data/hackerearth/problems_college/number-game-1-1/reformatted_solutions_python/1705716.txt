def fun(n):
    c1, c2, c3 = 1111111111111110, 1111111111111111111111110, 11111111110
    if n == 1:
        return 0
    else:
        if n % 3 == 0:
            c1 = 1+fun(n/3)
        elif n % 2 == 0:
            c2 = 1+fun(n/2)
        else:
            c3 = 1+fun(n-1)
    return min(c1, c2, c3)


t = input()
for i in range(t):
    n = input()
    c = fun(n)
    print c
