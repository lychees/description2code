#include<stdio.h>
#include<algorithm>

using namespace std;

bool func(int a,int b){
	return a>b;
}


int main(){
	int t,n,k,i,j,a[100050];
	long long int sum;
	scanf("%d",&t);
	for(i=0;i<t;++i){
		scanf("%d %d",&n,&k);
		for(j=0;j<n;++j){
			scanf("%d",&a[j]);
		}
		sort(a,a+n,func);
		sum = 0;
		for(j=0;j<k && j<n;++j){
			sum+=a[j];
		}
		printf("%lld\n",sum);
	}
	return 0;
}
