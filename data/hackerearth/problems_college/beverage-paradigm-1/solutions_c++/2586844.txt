#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    ll n,M,x,y,z,i,j,k,ans;
    vector<ll>v;
    cin>>n;
    for(i=0;i<n;i++) cin>>x , v.push_back(x);
    cin>>M;
    for(i=0;i<n-2;i++)
    {
        x = v[i];
        for(j=i+1;j<n-1;j++)
        {
            y = x + v[j];
            for(k=j+1;k<n;k++)
            {
                if(y+v[k]==M) { cout << "True" << endl; return 0; }
            }
        }
    }
    cout << "False" << endl;
    return 0;
}
