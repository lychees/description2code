arr = []


def xor(x, y):
    arr1 = [x-1, 1, x, 0]
    arr2 = [y, 1, y+1, 0]
    x1 = arr1[int((x-1) % 4)]
    x2 = arr2[int(y % 4)]
    arr.append(x1 ^ x2)


T = input()
for i in range(1, T+1):
    str = raw_input()
    # X=input()
    N, X = str.split()
    xor(int(N), int(X))


for i in arr:
    print i
