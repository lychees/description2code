#include <iostream>
using namespace std;
long long int fact(long long int n)
{
	long long int i,res=1;
	for(i=2;i<=n;i++)
	{
		res = res * i;
	}
	//cout << res << " ";
	return res;
}
long long int ncr(long long int n, long long int r)
{
	long long int i,res = n;
	for(i=2;i<=r;++i)
	{
		res *= (n-i+1);
		res /= i;
	}
	//cout << res << " ";
	return res;
}
int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	long long int c,v;
    	cin >> c;
    	cin >> v;
    	long long int res;
    	res = ncr(7,c)*ncr(5,v);
    	res = res * fact(c+v);
    	cout << res << "\n";
    }
    return 0;
}
