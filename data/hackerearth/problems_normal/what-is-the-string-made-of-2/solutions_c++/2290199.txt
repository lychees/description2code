#include<string>
#include<string.h>
using namespace std;

int main()
{
	char s[100];
	cin>>s;
	int arr[100];
	int sum=0;
	for(int i=0;i<strlen(s);i++)
	{
    if(s[i]==48||s[i]==54||s[i]==57)
    {
    	arr[i]=6;
    	
    	
    }
    
    else if(s[i]==50||s[i]==53||s[i]==51)
    {
    	arr[i]=5;
    
    	
    }

    else if(s[i]==52)
    {
    	arr[i]=4;
    	
    	
    }
    
    else if(s[i]==55)
    {
    	arr[i]=3;
    
    	
    }
    
    else if(s[i]==56)
    {
    	arr[i]=7;


    }
    
    else if(s[i]==49)
    {
    	arr[i]=2;
	}
    sum=sum+arr[i];
    
	}
	cout<<sum;
}
