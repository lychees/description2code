#include <iostream>
#include<cstring>
using namespace std;

int main() {
	int t;
	cin>>t;
//	cout<<t<<endl;
	while(t)
	{
		char s[100];
		cin>>s;
		int l=strlen(s);
	//	cout<<l<<endl;
		for(int j=0;j<l;j++)
		{
			
			if((s[j]+j)<=122)
			s[j]=(s[j]+j);
			
			else
			{
				s[j]=96+s[j]+j-122;
			}
		}
		cout<<s<<endl;
		--t;
	}
	return 0;
}
