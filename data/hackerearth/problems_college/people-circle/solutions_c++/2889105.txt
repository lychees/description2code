#include <cstdio>
#include <iostream>
#define DEBUG if(0)

using namespace std;

void Process(int M,int F,int K)
{
    int total = M + F;
    int ans[total];

    for(int i = 0; i < total; ++i ) 
        ans[i] = 0;

    int start = 0;
    while( F>0 )
    {
        for(int i=0;i<K-1;)
        {

            if(start == -1)
            {
               start = (start + 1)%total;
               ++i;
            }
            else
            {
                if(ans[start] == 0)
                    ++i;
                start = (start + 1)%total;
            }
        }

        
        while(ans[start] == 1)
            start = (start + 1)%total;
        ans[start] = 1;
        
        --F;

        DEBUG
        {
            for(int i = 0; i < total; ++i)
                cout << ans[i] << " ";
            cout << endl;
        }
    }

    for(int i = 0; i < total; ++i )
    {
        if(ans[i] == 1) cout << "F";
        else            cout << "M";
    }
    cout << endl;
}


int main()
{
    int T,M,F,K;
    cin >> T;
    while(T--)
    {
        cin >> M >> F >> K;
        Process(M,F,K);
    }
    return 0;
}
