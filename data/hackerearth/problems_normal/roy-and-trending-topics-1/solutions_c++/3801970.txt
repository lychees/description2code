#include <iostream>
#include <queue>
#include <utility>
using namespace std;
typedef long long int ll;
typedef pair<ll, ll> ii;
typedef pair<ll, ii> ii_i;

class compare{
	public:
	bool operator()(ii_i l, ii_i r){
		if(l.second.first == r.second.first){
			return l.first < r.first;
		}
		return l.second.first < r.second.first;
	}
};

ll minimum(ll a, ll b){
	if(a<b){
		return a;
	}
	return b;
}

int main()
{
    ll n;
    priority_queue<ii_i,vector<ii_i>,compare> pq;
    ll id, oldScore, post, like, share, comment;
    cin>>n;
    ll nn = n;
    while(nn--){
    	cin>>id>>oldScore>>post>>like>>share>>comment;
    	ll newScore = 50LL * post + 5LL * like + 10LL * share + 20LL * comment;
    	ii_i p;
    	p.first = id;
    	p.second.first = newScore - oldScore;
    	p.second.second = newScore;
    	pq.push(p);
    }
    for(ll i=0LL;i<minimum(5LL,n);i++){
    	cout<<pq.top().first<<" "<<pq.top().second.second<<endl;
    	pq.pop();
    }
    return 0;
}
