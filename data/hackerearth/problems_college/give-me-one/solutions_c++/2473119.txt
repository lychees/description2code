#include <bits/stdc++.h>
using namespace std;
#define maxi 20000007
long long dp[maxi];
void pre(  )
{
	//memset(dp,0,sizeof(dp));
long long i;
dp[1] = 0;  
for( i=2 ; i<=maxi ; i++)
{
dp[i] = 1 + dp[i-1];
if(i%2==0) dp[i] = min( dp[i] , 1+ dp[i/2] );
if(i%3==0) dp[i] = min( dp[i] , 1+ dp[i/3] );
}

}

int main()
{
   pre();
   long long t;
   scanf("%lld",&t);
   while(t--)
   {
   	long long n,cont=0;
   scanf("%lld",&n);
   cout<<dp[n]<<endl;
   }
    return 0;
}
