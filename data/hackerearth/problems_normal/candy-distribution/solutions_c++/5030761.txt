#include <bits/stdc++.h>

using namespace std;

#define ll long long int
#define pb push_back
#define f first
#define s second
#define mod 1000000007
typedef pair<int,int> ii;
typedef pair<int,pair<int,int> > iii;

map<ll,ll> mp;

ll modexp(ll a,ll b)
{
    ll res=1;
    while( b )
    {
        if( (b%2)==1 )
            res=(res*a)%mod;
        a=(a*a)%mod;
        b=b/2;
    }
    return res;
}

ll solve(ll n)
{
    if( mp.count(n)>0 )
        return mp[n];
    ll temp,i,x,res;
    res=0;
    temp=n-1;
    x=n-1;
    for(i=2;(i*i)<=x;i++)
    {
        if( (x%i)==0 )
        {
            if( (i*i)==x )
                res=(res+modexp(solve(i),x/i))%mod;
            else
                res=(res+modexp(solve(i),x/i)+modexp(solve(x/i),(i)))%mod;
        }
    }
    //if( temp>1 )
      //  res=(res+modexp(solve(temp),x/temp))%mod;
    mp[n]=res+1;
    return mp[n];
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    ll n;
    cin>>n;
    mp[1]=1;
    mp[2]=1;
    solve(n);
    cout<<mp[n]<<"\n";

    return 0;
}
