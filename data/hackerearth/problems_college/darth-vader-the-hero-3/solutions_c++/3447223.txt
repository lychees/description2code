vector<lld> V[lim];

lld A[lim2][lim2],C[lim2][lim2],P[lim];

lld one,zero,ansR,numNodes;

lld findGcd(lld a,lld b){
    if(a==0) return b;
    else return findGcd(b%a,a);
}
lld powP(lld a,lld b){
    if(b==0) return 1%mod;
    lld k;
    k=powP(a,b/2);
    k=k*k%mod;
    if(b%2==0) return k;
    else return a*k%mod;
}

void dfs(lld x){
	lld n=V[x].size(),t;
	one++;
	P[x]=0;
	rep(i,0,n-1){
		t=V[x][i];
		if(P[t]==1) dfs(t);
	}
}

int main(){
    // std::ios::sync_with_stdio(false);
    lld T,i,j,h,l,r,k,s,a,b,c,d,q,p,n,m,w,x,y,v,z,t,curr,prev,sum,ans,pos,val,countA,secondMin,indicator;
    sc(T);
    while(T--){
    	sc2(n,m);
    	M.clear();
    	rep(i,1,n*m){
    		V[i].clear();
    		P[i]=1;
    	}
    	rep(i,1,n) rep(j,1,m) C[i][j]=0;
    	rep(i,1,n) rep(j,1,m) sc(A[i][j]);
    	p=1;
    	rep(i,1,n){
    		rep(j,1,m){
    			if(A[i][j]==0) continue;
    			// PA po=mp(i,j);
    			if(C[i][j]==0) C[i][j]=p++;
    			// printf("i=%lld j=%lld t=%lld\n",i,j,t);
    			t=C[i][j];
    			rep(l,-1,1){
    				rep(r,-1,1){
    					if(l==0 && r==0) continue;
    					if(i+l>=1 && i+l<=n && j+r>=1 && j+r<=m && A[i+l][j+r]==1){
    						// po=mp(i+l,j+r);
    						if(C[i+l][j+r]==0) C[i+l][j+r]=p++;
    						// printf("i=%lld j=%lld t=%lld\n",i,j,t);
    						V[t].pb(C[i+l][j+r]);
    					}
    				}
    			}
    		}
    	}
    	one=ans=0;
    	countA=0;
    	rep(i,1,p-1){
    		one=0;
    		if(P[i]==1){
    			countA++;
    			dfs(i);
    		}
    		ans=max(ans,one);
    	}
    	prS(countA);prL(ans);
    }
    return 0;
}
