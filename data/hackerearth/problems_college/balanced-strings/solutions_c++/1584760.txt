#include <iostream>
using namespace std;

int charmap[26];
int main()
{
    int T, count, c;
    string s;
    bool isBalanced;
    cin >> T;
    while (T--) {
    	cin >> s;
      	for (int i = 0; i < 26; i++) 
    		charmap[i] = 0;
    	for (int i = 0; s[i] != 0; i++) {
    		c = s[i] - 'a';
    		charmap[c]++;
    	}
    	isBalanced = true;
    	c = charmap[0];
    	for (int i = 1; i < 26; i++)
    		if (charmap[i] != c) {
    			isBalanced = false;
    			break;
    		}

    	cout << (isBalanced ? "Yes" : "No") << "\n";
    }
    return 0;
}
