inp = raw_input()
[A, B] = inp.split(" ")
newA = ''
newB = ''
minA = ''
minB = ''
for i in A:
    if i == '5':
        i = '6'
    newA += i
for i in A:
    if i == '6':
        i = '5'
    minA += i
for j in B:
    if j == '5':
        j = '6'
    newB += j
for j in B:
    if j == '6':
        j = '5'
    minB += j
print int(minA)+int(minB), int(newA)+int(newB)
