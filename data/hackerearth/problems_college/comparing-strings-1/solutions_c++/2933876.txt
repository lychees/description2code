#include <iostream>
using namespace std;

int main() {
	long long int t,i,j,k,l,g;
	cin>>t;
	while(t--){
		g=0;
		string s,s1;
		cin>>s>>s1;
		k=s.size();
		l=s1.size();
		for(i=0;i<k;i++){
			for(j=0;j<l;j++)
			{
				if(s[i]==s1[j])
				{
					g=1;
					break;
				}
			}
			if(g==1)
			break;
		}
		
		if(g)
		cout<<"YES\n";
		else
		cout<<"NO\n";
	}
}
