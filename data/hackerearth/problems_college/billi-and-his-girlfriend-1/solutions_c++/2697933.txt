#include <iostream>
#include <cmath>

using namespace std;

long long rem(long long a, long long b,int k)
{
	long long point=pow(10,k);
	long long ans=1;
	while(b!=0)
	{
		if(b%2==1)
		{
			ans=(ans*a)%point;
		}
		a=(a*a)%point;
		b/=2;
	}
	return ans;
}

long long sum(long long a)
{
	long long sum=0;
	while(a!=0)
	{
		sum=sum+(a%10);
		a/=10;
	}
	return sum;
}

int main()
{
	long long t;
	cin >> t;
	while(t--)
	{
		long long a,b;
		int k;
		cin >> a >> b >> k;
		long long ans=rem(a, b, k);
		//cout << ans << endl;
		cout << sum(ans) << endl;
	}
}
