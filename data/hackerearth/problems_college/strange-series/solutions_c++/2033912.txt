#include<iostream>
#include<algorithm>
using namespace std;
#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<math.h>

int main()
{
    int t,n;
    long long x;
    cin>>t;
    while(t--)
    {
        x=1;
        cin>>n;
        for(int i=0;i<n;i++)
            x*=2;
        x-=1;
        cout<<x<<"\n";

    }
}
