num_cases = int(raw_input().rstrip("\n"))
d = {}
for count in range(num_cases):
	input = raw_input().rstrip("\n")
	key = input[0] + "".join(sorted(input[1:len(input) - 1])) + input[len(input) - 1]
	if d.__contains__(key):
		value = d[key]
		value.append(input)
		d[key] = value
	else:
	    d[key] = [input]
