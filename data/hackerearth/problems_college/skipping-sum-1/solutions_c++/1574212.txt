#include <bits/stdc++.h>
using namespace std;

int A[100010];
long long pre_sum[100010][11];

void pre_process(int n)
{
	int k,i,j;
	for(k=1;k<=10;k++)
	{
		for(i=1;i<=n;i++)
		{
			if(i-k>0) pre_sum[i][k]=pre_sum[i-k][k]+A[i];
			else pre_sum[i][k]=A[i];
		}
	}
}

long long skipping_sum(int l,int r, int k)
{
	return pre_sum[l+((r-l)/k)*k][k]-pre_sum[l][k]+A[l];
} 

int main()
{
    int n,q,l,r,k,i;
    scanf("%d %d\n",&n,&q);
    for(i=1;i<=n;i++) scanf("%d ",&A[i]);
    pre_process(n);
    for(i=0;i<q;i++) 
    {
    	scanf("%d %d %d\n",&l,&r,&k);
    	printf("%lld\n",skipping_sum(l,r,k));
    }
    return 0;
}
