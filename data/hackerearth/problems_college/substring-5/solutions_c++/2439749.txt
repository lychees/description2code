#include <bits/stdc++.h>
using namespace std;
bool check(string s)
{
	char a=s[0];
	for(int i=1;i<s.size();i++) if(s[i]!=a) return 0;
	return 1;
}
int main()
{
	string s;
	cin>>s;
	int ans=s.size();
	for(int i=0;i<s.size();i++)
	{
		string t("");
		t+=s[i];
		for(int j=i+1;j<s.size();j++)
		{
			t+=s[j];
			if(check(t)) ans++;
		}
	}
	cout<<ans<<endl;
	return 0;
}
