def binarysearch(a, t):
    low = 0
    high = len(a)-1
    while low <= high:
        mid = (low+high)/2
        if t == a[mid]:
            # print mid
            return True
        elif t < a[mid]:
            high = mid-1
        else:
            low = mid+1
    return False


n, q = map(int, raw_input().split())
a = sorted(list(map(int, raw_input().split())))
i = 0
while i < q:
    t = input()
    if binarysearch(a, t):
        print "YES"
    else:
        print "NO"
    i += 1
