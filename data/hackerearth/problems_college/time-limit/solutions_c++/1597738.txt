#include <iostream>
#include <cstdio>
#include <cstring>
#include <climits>
#include <algorithm>

using namespace std;

#define N 100005
#define M 400010
#define INF INT_MAX

const int maxn=500;
inline int max(int a,int b)
{
    return a>b?a:b;
}

inline int min(int a,int b)
{
    return a<b?a:b;
}

int main()
{
    int a[maxn],b[maxn];
    int n,m;
    cin>>n>>m;
    for(int i=0;i<n;i++)
        cin>>a[i];
    for(int i=0;i<m;i++)
        cin>>b[i];

    sort(a,a+n);
    sort(b,b+m);
    int res=max(2*a[0],a[n-1]);
    printf("%d\n",res<b[0]?res:-1);
    return 0;
}
