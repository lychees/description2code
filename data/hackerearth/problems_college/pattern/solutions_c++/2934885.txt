#include <iostream>
using namespace std;

int main()
{
   	int n,i,j,k;
   	cin>>n;
   	for(i=0;i<n;i++)
   		cout<<"*";
   	cout<<endl;
   	for(i=n-1;i>=2;i--)
   	{
   		for(j=1;j<=i;j++)
   		{
   			if(j==i)
   				cout<<"*";
   			else
   				cout<<" ";
   		}
   		cout<<endl;
   	}
   	for(i=0;i<n;i++)
   		cout<<"*";
    return 0;
}
