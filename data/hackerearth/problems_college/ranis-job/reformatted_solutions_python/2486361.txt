class DiskSpace:
    def minDrives(self, used, total):
        used = tuple(map(int, used.split()))
        total = tuple(map(int, total.split()))
        s = sum(used)
        total = sorted(total, reverse=True)
        i = 0
        while s > 0:
            s -= total[i]
            i += 1
        return i
