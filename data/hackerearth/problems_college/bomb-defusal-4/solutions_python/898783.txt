test=input()
for i in range(test):
    time,x,y,bombx,bomby=map(int,raw_input().split())
    cutx=(bomby-y)+(bombx+x)
    cuty=(bomby+y)+(bombx-x)
    if abs(cutx)%2==0 and abs(cuty)%2==0:
        cutx=cutx/2
        cuty=cuty/2
        if x==bombx and y==bomby:
            print "Counter-Terrorists Win !"
        elif (x==cutx and y==cuty) or (bombx==cutx and bomby==cuty):
            if time>1:
                print "Counter-Terrorists Win !"
            else:
                print "Terrorists Win !"
        elif time>2:
            print "Counter-Terrorists Win !"
        else:
            print "Terrorists Win !"
    else:
        print "Terrorists Win !"
