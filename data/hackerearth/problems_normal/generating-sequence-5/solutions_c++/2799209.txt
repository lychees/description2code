void pre()
{
   for(LL i=3;i<mx;i+=2)
   {
       if(isprime[i]==0)
       {
           for(LL j=i+i;j<mx;j+=i)
               isprime[j]=1;
       }
   }
   primes.push_back(2);
   for(LL i=3;i<mx;i+=2)
   {
       if(isprime[i]==0)
           primes.push_back(i);
   }
}
