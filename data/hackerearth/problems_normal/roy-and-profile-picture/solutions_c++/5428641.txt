#include <iostream>
using namespace std;

int main()
{
    int l;
    cin>>l;
    int n;
    cin>>n;
    int w;
    int h;
    while(n)
    {
    	cin>>w;
    	cin>>h;
    	if(w==h && w>=l)
    	{
    		cout<<"ACCEPTED"<<endl;
    	}
    	else if( w<l || h<l)
    	{
    		cout<<"UPLOAD ANOTHER"<<endl;
    	}
    	else 
    	{
    		cout<<"CROP IT"<<endl;
    	}
    	
    	n--;
    }
    return 0;
}
