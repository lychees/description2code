#include <iostream>
#include <cstring>
using namespace std;

int main()
{
	char s[1000],temp;
	int t,n,i,j;
	cin>>t;
	while(t--)
	{
		cin>>s;
		n=strlen(s);
		for(i=0;i<n-1;++i)
		{
			for(j=i+1;j<n;++j)
			{
				if(s[i]>s[j])
				{
					temp=s[i];
					s[i]=s[j];
					s[j]=temp;
				}
			}
		}
		cout<<s<<endl;
	}
	return 0;
	
}
