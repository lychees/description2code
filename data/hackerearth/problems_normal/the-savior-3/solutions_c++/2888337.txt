#include <bits/stdc++.h>
using namespace std;

int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
    	int n;
    	cin>>n;
    	int a[n];
    	for(int i=0;i<n;i++)
    	cin>>a[i];
    	int count=0;
    	for(int i=0;i<n-1;i++)
    	{
    		for(int j=i+1;j<n;j++)
    		{
    			if(((a[i]+a[j])%2 == 0) && a[i]!=a[j])
    			count++;
    		}
    	}
    	cout<<count<<endl;
    }
    return 0;
}
