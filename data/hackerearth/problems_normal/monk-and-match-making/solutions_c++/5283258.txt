#include<bits/stdc++.h>
using namespace std;
char s[100003];
int main()
{
    scanf("%s",s);
    int q;
    int l1,r1,l2,r2;
    cin>>q;
    while(q--)
    {
        cin>>l1>>r1>>l2>>r2;
        char *ptr1=(s+(l1-1));
        char *ptr2=(s+(l2-1));
        //cout<<*ptr1<<" "<<*ptr2<<"\n";
        if((r1-l1) != (r2-l2))
            cout<<"No\n";
        else
        {
            if(memcmp(ptr1,ptr2,r1-l1+1)==0)
                cout<<"Yes\n";
            else
                cout<<"No\n";
        }
    }
    return 0;
}
