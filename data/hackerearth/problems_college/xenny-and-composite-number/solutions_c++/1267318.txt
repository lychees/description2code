#include <iostream>
using namespace std;
bool IsPrime(int number)
{	
	
	int i;
	
	for (i=2; (i*i)<=number; i++)
	{
		if (number % i == 0)
		{
			return false;
		}
	}
	
	return true;	
}

int main()
{
    int t;
    cin >> t;
    while(t--){
    	int a;
    	cin >> a;
    	while(true){
    	a=a+1;
    	
    		if(!IsPrime(a))
    		{
    			cout <<a<< endl;
    			break;
    		}
    			
    	}
    }
    return 0;
}
