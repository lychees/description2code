#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int arr[n],i,c=0;
    	for(i=0;i<n;i++)
    	arr[i]=i+1;
    	for(i=0;i<n;i++)
    	{
    		for(int j=i+1;j<n;j++)
    		{
    			if((arr[i]*arr[j])%(arr[i]+arr[j])==0)c++;
    		}
    	}
    	cout<<c<<"\n";
    }
    return 0;
}
