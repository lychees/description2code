#define MAX	55

int f[MAX][MAX][MAX];

int go(int a, int b, int c) {
	if (a == 0 && b == 0 && c == 0) return 0;
	int &res = f[a][b][c];
	if (res >= 0) return res;
	set<int> v;
	rep(i, a) v.insert(go(i, b, c));
	rep(i, b) v.insert(go(a, i, c));
	rep(i, c) v.insert(go(a, b, i));
	res = -1;
	for (set<int>::iterator it = v.begin(); it != v.end(); it++) {
		if (*it != ++res) {
			return res;
		}
	}
	return res = v.size();
}
