#include <bits/stdc++.h>
using namespace std;
#define max 1000000
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long int i,n,k,a[max],r=0,sum=0;
		cin>>n>>k;
		for(i=0;i<n;i++)
		{
			cin>>a[i];
		}
		sort(a,a+n);
			for(i=n-1;i>=0;i--)
		{	if(r!=k)
			{sum=sum+a[i];
			r++;}
			else
			break;
		}
       cout<<sum<<endl;
}
return 0;
}
