#include <iostream>
#include <vector>
#include <queue>
#include <utility>
#include <set>
#include <algorithm>

using namespace std;

int main()
{
	int tc, n, m;
	int beg, end;
	bool ans;
	
	cin >> tc;
	while (tc--)
	{
		vector<pair<int, int> > range;
		
		cin >> n >> m;
		ans = true;
		
		while (m--)
		{
			cin >> beg >> end;
			if (beg <= end)
			{
				range.push_back(make_pair(beg, end));
				range.push_back(make_pair(beg + n, end + n));
			}
			else
				range.push_back(make_pair(beg, end + n));
		}
		sort(range.begin(), range.end());
		
		set<pair<int, int> > pq;
		int i, left;
		
		i = 0;
		left = 0;
		while (ans && i < range.size())
		{
			if (pq.empty())
			{
				left = range[i].first;
			}
			
			while (range[i].first == left)
			{
				pq.insert(make_pair(range[i].second, i));
				++i;
			}
			
			int idx = pq.begin()->second;
			pq.erase(pq.begin());
			
			if ((left < range[idx].first) || (left > range[idx].second))
			{
				ans = false;
				break;
			}
			++left;
		}
		
		cout << (ans ? "YES" : "NO") << endl;
	}
	return 0;
}
