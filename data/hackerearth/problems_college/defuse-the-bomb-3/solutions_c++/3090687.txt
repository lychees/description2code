#include <iostream>
#include <stdio.h>
#include<string.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	char a[12],b[12];
    	long long sum;
    	scanf("%s%s",a,b);
    	int i,j,k,x,l1=strlen(a),l2=strlen(b);
    	long y;
    	y=atoi(b);
    	for(i=0;i<=l1-l2;i++)
    	{
    		sum=0;
    		for(j=l2+i-1,k=1;j>=i;j--,k*=10)
    		{
    			x=a[j]-'0';
    			sum+=x*k;
    		}
    		printf("%lld\n",sum*y);
    	}
    }
    return 0;
}
