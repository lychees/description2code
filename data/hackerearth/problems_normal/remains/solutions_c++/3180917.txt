#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		unsigned long long x,y,n;
		scanf("%llu%llu%llu",&x,&y,&n);
		 long long int a[2],i;
		 
		unsigned long long sum; 
		a[0]=x;
		a[1]=y;
		sum=a[0]+a[1];
		long long q;
		for(i=2;i<n;i++){
			if(a[0]==0 && a[1]!=0 &&i>2){
				sum-=a[1];
				if((n-i+2)%3==0){
					sum+=((n-i+2)/3)*2*a[1];
					}
				else if((n-i+2)%3==1){
					sum+=((n-i+2)/3)*2*a[1];
					}
				else{
					sum+=(((n-i+2)/3)*2*a[1]+a[1]);
					}
				break;
			}
			q=a[0]-a[1];
			if(q>0){
				sum+=q;
				
				a[0]=a[1];
				a[1]=q;
				}
			else{
				sum+=((-1)*q);
				
				a[0]=a[1];
				a[1]=((-1)*q);
				//printf("a[0]=%d    %d\n",a[0],a[1]);
				
			}
	
				
		}
			printf("%llu\n",sum);	
		
	}
}
