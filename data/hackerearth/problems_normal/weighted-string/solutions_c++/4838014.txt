#include <iostream>
#define ll long long 
using namespace std;

int main()
{
    string s;
    int t;
    cin>>t;
    while(t--)
    {
    	ll n,i,j=0,k,a[1000005],ct=0;
		cin>>k>>s;
    	i=0;
    	while(s[i]!=NULL)
    	{
    		a[j++]=(long long)s[i]-'a'+1;
   			i++;
    	}
    	n=j;
    	int sum=0;
    	int start=0;
    	for(i=start;i<n;i++)
    	{
    		sum+=a[i];
    		if(sum==k)
    			ct++;
    		else if(sum>k)
    			{
    				i=start++;
    				sum=0;
    			}
    		
    	}
    	cout<<ct;
    	cout<<endl;
    }
    return 0;
}
