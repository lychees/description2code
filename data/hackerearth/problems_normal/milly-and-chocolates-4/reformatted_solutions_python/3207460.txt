noOfTc = raw_input()
noOfTc = int(noOfTc)
tcCheck = 0

while tcCheck < noOfTc:
    sum = 0
    tcCheck += 1
    noOfFoodItem = raw_input()
    percentage = raw_input()
    noOfFoodItem = int(noOfFoodItem)
    percentage = percentage.split()
    for item in range(0, noOfFoodItem):
        sum = sum + int(percentage[item])
    print "{0:.8f}".format(float(sum/float(noOfFoodItem)))
