N, M = map(int, raw_input().split())
fib = [None] * N
fib[0], fib[1] = 1, 1
for i in range(2, N):
	fib[i] = fib[i - 1] + fib[i - 2]
a = [int(i) for i in raw_input().split()]
a.sort()
i, j, sumSkip = 0, 0, 0
while i < len(a) or j < len(fib):
	if i >= len(a) or a[i] != fib[j] :
		sumSkip += fib[j]
	else:
		i += 1
	j += 1
print sumSkip
