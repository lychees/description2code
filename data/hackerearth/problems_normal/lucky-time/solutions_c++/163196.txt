
#include<bits/stdc++.h>
using namespace std;
bool chk(int a,int b,int c)
{
    int fnd[10],t=2;
    memset(fnd,0,sizeof(fnd));
    while(t--)
    {
        if(fnd[a%10]==0) fnd[a%10]=1;
        else return false;
        a/=10;
    }
    t=2;
    while(t--)
    {
     if(fnd[b%10]==0) fnd[b%10]=1;
        else return false;
        b/=10;
    }
    t=2;
    while(t--)
    {
        if(fnd[c%10]==0) fnd[c%10]=1;
        else return false;
        c/=10;
    }
    return true;
}
void conv(int a,int b,int c)

{
    if(a/10==0) cout<<0<<a<<':';
    else cout<<a<<':';
    if(b/10==0) cout<<0<<b<<':';
    else cout<<b<<':';
    if(c/10==0) cout<<0<<c<<endl;
    else cout<<c<<endl;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {//cout<<t;

        string st;
        int i,s=0,m=0,h=0;
        cin>>st;
        h=(st[0]-'0')*10+(st[1]-'0');
        m=(st[3]-'0')*10+(st[4]-'0');
        s=(st[6]-'0')*10+(st[7]-'0');
//cout<<h<<m<<s;
if(chk(h,m,s) ) {conv(h,m,s);continue;}
        while(1)
        {
            s++;
            if(s==60)
            {
                s=0;
                m++;
            }
            if(m==60)
            {
                m=0;
                h++;
            }
            if(h==24)
            {
                h=1;
                s=0;m=0;
            }
            if(chk(s,m,h)) break;
        }
        conv(h,m,s);
    }
    return 0;
}
