# include<iostream>
using namespace std;
int phi(int n)
{int i;
	int r=n;
	for(i=2;i*i<=n;++i)
	{
		if(n%i==0)
		r-=(r/i);
		while(n%i==0)
		n/=i;
	}
	if(n>1)
	r-=(r/n);
	return r;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,x,k,i,c,s=1;
		cin>>n>>x>>k;
		for(i=n;i>=x;i--)
		{
			c=phi(i);
		//	cout<<c<<"        ";
			if(c==k)
			{
				cout<<i<<endl;
				s=0;
				break;
			}
		}
		if(s)
		cout<<"-1\n";
	}
}
