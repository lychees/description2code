#include<stdio.h>
#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	int b[26],k=0,i;
	string a;
	cin>>a;
	for(i=0;i<26;i++)
	b[i]=0;
	for(i=0;i<a.length();i++)
	b[a[i]-'a']++;
	for(i=0;i<26;i++)
	if(b[i]%2==1)
	k++;
	if(k==1||k==0)
	printf("YES\n");
	else
	printf("NO\n");
	
}
