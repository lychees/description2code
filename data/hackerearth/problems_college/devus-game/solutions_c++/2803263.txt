#include <iostream>
using namespace std;

int main()
{
    int t,i;
    scanf("%d",&t);
    while(t--)
    {
    	int n;
    	scanf("%d",&n);
    	int a[n];
    	bool vis[n];
    	for(i=0;i<n;i++)
    	{
    		scanf("%d",&a[i]);vis[i]=false;
    	}
    	i=0;
    	while(1)
    	{
    		if(a[i]==-1) {printf("Win\n");break;}
    		else if(!vis[i])
    		{
    			int k=a[i];
    			vis[i]=true;
    			i=k;
    		}
    		else {printf("Lose\n");break;}
    	}
    }
    return 0;
}
