#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

#define MAX 1000010
#define MOD 1000000007
#define ll long long

int t;
ll a, b;
char s[MAX];

ll makeNum(int m){
	ll ret = 0;
	for (int i = 0; i < strlen(s); i++)
		ret = (ret * 10 + s[i] - '0') % m;
	return ret;
}

ll pow(ll b, ll p, ll m){
	if (p == 0)
		return 1;
	if (p == 1)
		return b % m;
	ll ret = pow(b, p/2, m);
	ret = (ret * ret) % m;
	if (p & 1)
		ret = (ret * b) % m;
	return ret;
}

int main(){
	scanf("%d", &t);
	while (t--){
		scanf("%s", s);
		a = makeNum(MOD);
		scanf("%s", s);
		b = makeNum(MOD - 1);
		printf("%lld\n", pow(a, b, MOD));
	}
	return 0;
}
