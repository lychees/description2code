#include <iostream>

using namespace std;

long dp[20][20];
long calcdp(int i,int j){
    int k=0;
    long sum = 0;
    while(k<=j){
        sum = sum + dp[i-1][j-k];
        k++;
    }
    return sum;
}
int main(){
    int i=0,j=0,n,k,t;
    long res[100000];

    while(i<20){
        dp[1][i] = 1;
        i++;
    }
    i=0;
    while(i<20){
        dp[0][i] = 0;
        i++;
    }
    i=0;
    while(i<20){
        dp[i][0] = 1;
        i++;
    }
    dp[0][0] = 0;

    for(i=2;i<20;i++){
        for(j=1;j<20;j++){
            dp[i][j] = calcdp(i,j);
        }
        cout<<endl;
    }
    cin>>t;
    i=0;
    while(i<t){
        cin>>n>>k;
        res[i] = dp[n][k];
        i++;
    }
    i=0;
    while(i<t){
        cout<<res[i]<<endl;
        i++;
    }
}
