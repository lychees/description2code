#include<iostream>

using namespace std;

int main()
{
	int t,i,n,count,array[27];
	char word[10002],x,skip[28];
	
	
	cin>>t;
	
	while(t--)
	{
		cin>>word;
		cin>>n;
		for(i=0;i<n;i++)
		{
			cin>>x;
			skip[i] = x;
		}
		skip[i] = '\0';
		
		for(i=0;i<26;i++)
			array[i] = 0;
		
		for(i=0;i<n;i++)
		{
			array[skip[i]-97] = 1;
		}
		count = 0;
		for(i=0;word[i]!='\0';i++)
		{
			if(array[word[i]-97] != 1)
				count += (word[i]-96);
		}
		cout<<count<<endl;
	}
	return 0;
}
