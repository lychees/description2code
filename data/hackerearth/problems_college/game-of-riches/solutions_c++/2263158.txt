#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int result=0;
    	int first_value, value, nG, nR;
    	int wealth[5005];
    	cin>>nR>>nG;
    	for(int i=0;i<nR;i++){
    		cin>>wealth[i];
    	}
    	first_value=wealth[0];
    	sort(wealth, wealth+nR);
    	for(int i=nR-1,p=1;i>=0;i--,p++){
    		
    		if(wealth[i]==first_value)
    			break;
    		if(p%nG==0)
    			result++;
    	}
    	cout<<result<<endl;
    
    }
    return 0;
}
