#include<iostream>
#include<string.h>
using namespace std;
int main()
{
    int t,i,c,k;
    string str;
    cin>>t;
    while(t>0)
    {
        c=0;
        cin>>str;
        for(i=0;str[i]!='\0';i++)
        {
            for(k=i;str[k]!='\0';k++)
            {
                if(str[i]==str[k+1])
                    c++;
                else
                {
                    i=k;
                    break;
                }
            }
        }
        cout<<c<<"\n";
        t--;
    }
    return 0;
}
