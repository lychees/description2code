#include <iostream>
using namespace std;

int main()
{
    long long tc, numCards;
    cin>>tc;
    while(tc--)
    	{
    		cin>>numCards;
    		if(numCards == 0)
    			{
    				cout<<"0"<<endl;
    			}
    		else if(numCards == 1 || numCards == 2)
    			{
    				cout<<"1"<<endl;
    			}
    		else if(numCards%3 == 0)
    		    {
    		    	cout<<numCards/3<<endl;
    		    }
    	
    		else 
    			{
    				cout<<numCards<<endl;
    			}
    	}
    
    return 0;
}
