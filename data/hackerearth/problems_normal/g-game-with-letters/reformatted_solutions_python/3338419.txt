for _ in range(int(raw_input())):
    n, s = raw_input().split()
    n = int(n)
    p = {}
    for i in s:
        if p.has_key(i):
            p[i] += 1
        else:
            p[i] = 1
    a = []
    for i in range(n):
        s = raw_input()
        ct = 0
        for j in s:
            ct += p.get(j, 0)
        a.append([ct, -len(s), -i])
    print -max(a)[2]+1
