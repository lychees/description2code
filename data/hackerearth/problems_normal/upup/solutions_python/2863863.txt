x=raw_input()
f=-1
l = len(x)
z=""
for i in range(0,l):
	if f==-1 and x[i]!=' ':
		z = z+(x[i]+"").capitalize()
		f=0
	elif x[i]==' ':
		z = z+" "
		f=-1
	else:
		z=z+x[i]
print z
