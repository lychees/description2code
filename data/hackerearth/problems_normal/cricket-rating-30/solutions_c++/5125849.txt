#include<bits/stdc++.h>
using namespace std;
int main(){
    int n,sum = 0,ans = 0;
    cin>>n;
    while(n--){
        int m;
        cin>>m;
        if(sum<=0){
            if(m<0)continue;
            sum = m;
        }
        else{
            sum+=m;
        }
        ans = max(sum,ans);
    }
    cout<<ans<<endl;
}
