#include "iostream"
#include "vector"
using namespace std;

void printDecimalinBin(int n){
  vector < int > result;
  while(n){
    result.push_back(n%2);
    n/=2;
  }
  for(int i=result.size()-1;i>=0;i--)
     cout << result[i];
}
 
int main(){
   int n;
   cin >> n;
   int arr[n];
   for(int i=0;i<n;i++)
      cin >> arr[i];
   int *LIS =new int[n];
   for(int i=0;i<n;i++)
   LIS[i]=1;
   for(int i=1;i<n;i++){
     for(int j=0;j<i;j++){
       if(arr[i]>arr[j] && LIS[j]+1>LIS[i])
         LIS[i]=LIS[j]+1;
       }
     }
   int max_length =1;
   for(int i=0;i<n;i++)
       max_length=max(max_length,LIS[i]);
   
   printDecimalinBin(max_length);
   return 0;
}
