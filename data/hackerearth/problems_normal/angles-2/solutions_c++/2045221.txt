#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n,k;
    cin>>n>>k;
    int a[361],b[361];
    for(int i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    for(int i=0;i<k;i++)
    {
    	cin>>b[i];
    }
    int flag;
    for(int i=0;i<k;i++)
    {
    	flag=0;
    	for(int t=0;t<n;t++)
    	{
    		for(int j=1;j<=360;j++)
    		{
    				if((a[t]*j)%360==b[i])
    				{
    					flag=1;
    					break;
    				}
    		}
    		if(flag==1)
    		{
    			break;
    		}
    	}
    	if(flag==1)
    	{
    		cout<<"YES"<<endl;
    	}
    	else
    	{
    		cout<<"NO"<<endl;
    	}
    }
    
    
}
