#include<stdio.h>
#include<iostream>
using namespace std;
#define MAXDIGITS 1100
#define MOD 1000000
#define weight 6
class verybig
{
       private:
               long long digits[MAXDIGITS];
               int last_digit;
               int first_digit;
       public: 
               
               verybig(int a=0)
               {
                        first_digit=last_digit=MAXDIGITS-1;
                       
                        do
                        {
                                 digits[first_digit--]=a%MOD;
                                 a/=MOD;
                        }while(a);
                        
               }
               int getsum();
               void multiply(int a);
                  
};


int verybig::getsum()
{
	int sum=0;
           for(int i=first_digit+1;i<=last_digit;i++)
           {
           		int digitcopy=digits[i];
           		while(digitcopy){
           			sum+=(digitcopy%10);
           			digitcopy/=10;
           		}
           	}
	return sum;
}
void verybig::multiply(int a)
{
           for(int i=last_digit;i>first_digit;i--)
           {
                   digits[i]*=a;
                   if(i!=last_digit)
                   {
                                    digits[i]+=(digits[i+1]/MOD);
                                    digits[i+1]%=MOD;                 
                   }
           }
           while(digits[first_digit+1]/MOD!=0)
           {
                                           digits[first_digit]=digits[first_digit+1]/MOD;
                                           digits[first_digit+1]%=MOD;
                                           first_digit--;
           }
}

int main()
{
	 int t;
	 cin>>t;
	 int arr[1100];
	 
	 arr[0]=1;
	 verybig ob(1);
	 for(int i=1;i<=1000;i++)
	 {
	 	ob.multiply(i);
	 	arr[i]=ob.getsum();
	 }
	 while(t--){
	 	int n;
	 	cin>>n;
		cout<<arr[n]<<endl;
	 }
}
