VII graph[5000];
int vis[5000];
int dist[5000];
int main()
{
	int n;si(n);
	for(int i=0;i<n;i++)
	{
		int u,v,w;
		si(u);si(v);si(w);
		graph[u].PB(MP(v,w));
		graph[v].PB(MP(u,w));
	}
	int u;si(u);
	for(int i=0;i<5000;i++)
		dist[i]=100000000;
	set< II > S;
	S.insert(MP(0,u));
	dist[u]=0;
	while(!S.empty())
	{
		II top = *S.begin();
		S.erase(S.begin());
		int u = top.S;
		if(vis[u])continue;
		for(int i=0;i<SZ(graph[u]);i++)
		{
			int w = graph[u][i].F;
			int wt = graph[u][i].S;
			if(dist[u] + wt < dist[w])
			{
				dist[w] = dist[u] + wt;
				S.insert(MP(dist[w],w));
			}
		}
	}
	int q;si(q);
	while(q--)
	{
		int u;si(u);
		if(dist[u]==100000000)
			puts("NO PATH");
		else
			dout(dist[u]);
	}


	return 0;
}
