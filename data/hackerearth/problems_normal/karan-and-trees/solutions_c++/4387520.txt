//Karan and Trees 

#include<iostream>
#include<cmath>
using namespace std;

bool isEven(double x){
	double quo;
	quo=x/2;
	
	if((x-(floor(quo)*2))==0){
		return true;
	}else{
		return false;
	}
}

int main(){
	int t,i;
	double a,b;
	
	cin>>t;
	for(i=0;i<t;++i){
		cin>>a>>b;
		
		while(a!=b){
			if(a>b){
				if(isEven(a)){
					a=a/2;
				}else{
					a=(a-1)/2;
				}
			}else{
				if(isEven(b)){
					b=b/2;
				}else{
					b=(b-1)/2;
				}
			}
		}
		
		cout<<a<<endl;
	}
}
