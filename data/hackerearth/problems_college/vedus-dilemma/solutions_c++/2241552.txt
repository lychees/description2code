#include<cstdio>
#include<cmath>
#include<algorithm>
using namespace std;
struct ba
{
	int d;
	int pr;
};
int compare(struct ba pri,struct ba pri1)
{
	if(pri.pr<pri1.pr)
		return 1;
	return 0;
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		int a[n+2];
		for(int i=0;i<n;i++)
			scanf("%d",&a[i]);
		int p,q;
		scanf("%d%d",&p,&q);
		struct ba b[n];
		for(int i=0;i<n;i++)
		{
			b[i].d=a[i];
			b[i].pr=(int)abs(p-a[i]);
		}
		sort(b,b+n,compare);
		/*for(int i=0;i<n;i++)
			printf("%d %d\n",b[i].d,b[i].pr);*/
		int cnt=0;
		int da[q];
		for(int i=0;i<n;i++)
		{
			int x;
			int y,rem=0;
			if(b[i].d>p)
			{
				x=b[i].d;
				y=p;
			}
			else
			{
				x=p;
				y=b[i].d;
			}
			while( y!= 0)
			{
			rem= x % y;
			x = y;
			y = rem;
			}
			if(x>1 && cnt<q)
				da[cnt++]=b[i].d;
			if(cnt>=q)
				break;	
		}
		if(cnt!=q)
			printf("-1\n");
		else
		{
			sort(da,da+q);
			for(int j=0;j<q;j++)
				printf("%d ",da[j]);
			printf("\n");
		}
	}
	return 0;
}
