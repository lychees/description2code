#include<iostream>
#include<string.h>
#include<algorithm>
#include<stdio.h>
using namespace std;
int a[26],b[26]={0};
int main()
{
    for(int i=0;i<26;i++)
        a[i]=26-i;
    int t;
    cin>>t;
    while(t--)
    {
        for(int i=0;i<26;i++)
            b[i]=0;
        char c[100];
        cin>>c;
        int f=0;
        for(int i=0;i<strlen(c);i++)
        {
            if(b[c[i]-'A']==0)
            {
                if(count(c,c+strlen(c),c[i])!=a[c[i]-'A'])
                {
                    f=1;break;
                }
                else
                    b[c[i]-'A']=1;
            }
        }
        if(f==1)
            cout<<"No\n";
        else
            cout<<"Yes\n";
    }

}
