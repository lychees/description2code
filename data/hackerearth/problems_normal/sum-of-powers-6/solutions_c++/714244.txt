#include <iostream>
#include <algorithm>
using namespace std;

#define ll long long int

bool is_perfect[1000001];
int perfect[1000001];
int tp;

int main() {
    int test, x; cin >> test;
    is_perfect[1] = true;
    for(ll i = 2; i < 1000001; ++i) {
        if(is_perfect[i]) continue;
        for(ll j = i * i; j < 1000001; j *= i)
            is_perfect[j] = true;
    }
    for(int i = 1; i <= 1000000; ++i)
        if(is_perfect[i]) perfect[tp++] = i;
    while(test--) {
        cin >> x;
        bool ok = true;
        for(int i = 0; i < tp && perfect[i] < x; ++i) {
            if(is_perfect[x - perfect[i]]) {
                ok = false;
                cout << "Yes" << endl;
                break;
            }
        }
        if(ok) cout << "No" << endl;
    }
    return 0;
}
