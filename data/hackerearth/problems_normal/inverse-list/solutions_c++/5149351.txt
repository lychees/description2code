#include<iostream>
using namespace std;
int main()
{
	int T;
	cin>>T;
	while(T--)
	{
		int a[1001];
		int n;
		cin>>n;
		int i, j=0;
		int b[1001];
		for(i=0;i<n;i++)
		{
			cin>>a[i];
			b[a[i]]=i+1;
		}
	
		int flag=0;
		for(i=0;i<n;i++)
  			if(a[i]!=b[i+1])
 			{
 				flag=1;
 				break;
 			}
		if(flag)
			cout<<"not inverse"<<endl;
		else
			cout<<"inverse"<<endl;
	}
	return 0;
}
