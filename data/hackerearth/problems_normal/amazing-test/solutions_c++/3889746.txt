#include <iostream>
using namespace std;

int dp[2][10005];
int arr[105];

int main()
{
  int t,n,x;
  cin>>t;
  while(t--)
  {
      long sum=0;
      cin>>n>>x;
      for(int i=1;i<=n;i++)
      {
      cin>>arr[i];
      sum+=arr[i];
      }
      if(sum<=x)
      {
          cout<<"YES\n";
      }
      else
      {
          for(int i=0;i<=x;i++)
          dp[0][i]=0;
          for(int i=1;i<=n;i++)
          {
              for(int j=0;j<=x;j++)
              {

              dp[i%2][j]=dp[(i-1)%2][j];
              if(j>=arr[i])
              dp[i%2][j]=max(dp[i%2][j],(dp[(i-1)%2][j-arr[i]]+arr[i]));
              }
          }
          if((sum-dp[(n%2)][x]) <= x)
          cout<<"YES"<<endl;
          else

          cout<<"NO"<<endl;
      }
  }

    return 0;
}
