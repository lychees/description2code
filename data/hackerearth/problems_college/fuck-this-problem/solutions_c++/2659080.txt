#include<iostream>
using namespace std;
int main()
{
	string line;
	getline(cin,line);
	int n;
	cin>>n;
    int len=line.length()/n;
    int i=0;
    while(i!=len)
    { 
      cout<<line.substr(i*n,n);
	  cout<<endl;
	  i++;
	 
	}
	len=line.length()-i*n;
	for(int j=line.length()-len;j<line.length();j++)
	cout<<line[j];
	return 0;
	
}
