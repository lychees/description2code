//Matching Strings
#include <iostream>
#include <cstdio>
#include <cstdlib>
#define N 501
using namespace std;

int computeLength(string &p,string &q)
{
	int i=0,j=0,count=0;
	
	while( p[i]!='\0' && q[j]!='\0')
	{
		if(p[i] == q[j])
		count++;
		i++;j++;
	}
	return count;
}

int main()
{
	int t;
	//char *p = (char*) malloc (sizeof(char)*N);
	//char *q = (char*) malloc (sizeof(char)*N);
	string p,q;
	scanf("%d",&t);
	while(t--)
	{
		//scanf("%s %s",p,q);
		cin>>p>>q;
		printf("%d\n",computeLength(p,q));
		p.clear();
		q.clear();
	}
	return 0;
}
