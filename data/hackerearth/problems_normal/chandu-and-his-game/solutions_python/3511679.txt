n, k = map(int, raw_input().split())
xx = map(int, raw_input().split())
yy = map(int, raw_input().split())
ans = 0
for i in range(k):
    x = xx[i]
    y = yy[i]
    
    mini = 1e18
    mini = min(abs(n -  x) + abs(n - y), mini); 
    mini = min(abs(1 -  x) + abs(1 - y), mini); 
    mini = min(abs(1 -  x) + abs(n - y), mini); 
    mini = min(abs(n -  x) + abs(1 - y), mini); 
    
    ans += mini
print ans
