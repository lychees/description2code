#include <iostream>
#include <stdio.h>
using namespace std;
void dfs(int a[][101],int i ,int j,bool visited[][101],int n,int &sum) {
		visited[i][j] = 1;
	if(i == n-1 && j == n-1) {
		
		sum = sum + 1;
	
	}
	if(i-1>=0 && a[i-1][j] == 0 && visited[i-1][j] == 0) {
		
		dfs(a,i-1,j,visited,n,sum);
	}
	if(i+1< n  && a[i+1][j] == 0 && visited[i+1][j] == 0) {
		
		dfs(a,i+1,j,visited,n,sum);
	}
	if(j-1>=0 && a[i][j-1] == 0 && visited[i][j-1] == 0) {
		
		dfs(a,i,j-1,visited,n,sum);
	}
	if(j+1<n  && a[i][j+1] == 0 && visited[i][j+1] == 0) {
		
		dfs(a,i,j+1,visited,n,sum);
	}
	visited[i][j] = 0;
	
}
int main()
{
	int n,a[101][101],i,j,sum = 0;
    cin>>n;
    bool visited[101][101];
    for(i = 0; i < n ; i++) {
    	for(j = 0; j < n;j++){
    	
    	cin>>a[i][j];
    	visited[i][j] = 0;
    	}
    }
    
    
    dfs(a,0,0,visited,n,sum);
    cout<<sum<<endl;
    return 0;
}
