#include <bits/stdc++.h>
using namespace std;
#define ll long long int

int main()
{
   cin.sync_with_stdio(false);
   cin.tie(0);
   ll t,n,m;
   ll a[]={89,145,42,20,4,16,37,58};
   cin>>t;
   while(t--)
   {
   	cin>>n>>m;
   	ll j=0,k=0;
   	while(m--)
   	{
   		
   		ll sum=0;
   		if(n==89)
   		{
   			j=1;
   			break;
   		}
   		if(n==1)
   		{
   			cout<<1<<endl;
   			k=1;
   			break;
   		}
   	
   		
   		while(n)
   		{
   			ll d=n%10;
   			sum=sum+d*d;
   			n/=10;
   		}
   		n=sum;
   	}
   	if(k)
   	continue;
   	m++;
   	if(j)
   	cout<<a[m%8]<<endl;
   	else
   	cout<<n<<endl;
   }
   
}
