#include <iostream>
using namespace std;

int main()
{
	int T;
	cin>>T;
	for(int i=0; i<T; i++)
	{
		int digits;
		cin>>digits;
		if(digits%3 == 0)
		{
			for(int j=0; j<digits; j++)
				cout<<"5";
			cout<<endl;
		}
		else if(digits%5 == 0)
		{
			int temp2 = digits - 5;
			bool bpos = true;
			while(temp2%3 != 0)
			{
				if(temp2 < 3 || temp2 == 4)
				{
					bpos = false;
					break;
				}
				temp2 = temp2 - 5;
			}
			if(bpos)
			{
				for(int j=0; j<temp2; j++)
				{
					cout<<"5";
				}
				for(int j=0; j<digits - temp2; j++)
					cout<<"3";
			}
			else
			{
				for(int j=0; j<digits; j++)
					cout<<"3";	
			}
			cout<<endl;
		}
		else
		{
			bool bposible = true;
			int temp = digits - 5;
			while(temp % 3 != 0)
			{
				temp = temp - 5;
				if(temp <3 || temp == 4)
				{
					cout<<"-1"<<endl;
					bposible = false;
					break;
				}
			}
			if(bposible == true)
			{
				for(int j=0; j<temp; j++)
				{
					cout<<"5";
				}
				for(int j=0; j<digits - temp; j++)
				{
					cout<<"3";
				}
				cout<<endl;
			}
		}
		
	}
	
    return 0;
}
