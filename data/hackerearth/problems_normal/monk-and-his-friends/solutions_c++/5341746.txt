#include <iostream>
#include <cstdio>
#include <set>
using namespace std;

void sameMain()
{
    int T;
    scanf("%d", &T);
    while(T--)
    {
        int N, M;
        scanf("%d%d", &N, &M);
        set<long long> classroom;
        for(int i = 0; i < N; i++)
        {
            long long x;
            scanf("%lld", &x);
            classroom.insert(x);
        }

        for(int i = 0; i < M; i++)
        {
            long long x;
            scanf("%lld", &x);
            if(classroom.find(x) != classroom.end())
                printf("YES\n");
            else
            {
                printf("NO\n");
                classroom.insert(x);
            }
        }
    }
}

int main()
{
    sameMain();
    return 0;
}
