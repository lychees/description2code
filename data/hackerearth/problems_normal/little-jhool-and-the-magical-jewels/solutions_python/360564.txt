n = int(raw_input())
while n>0:
        l = [0,0,0,0]
        a = raw_input()
        for i in a:
                if i=='r':
                        l[0] = l[0] + 1

                if i =='u':
                        l[1] = l[1] + 1
                if i =='b':
                        l[2] = l[2] + 1
                if i =='y':
                        l[3] = l[3] + 1

        n = n-1
        print min(l)
