#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
 
 int n , v , w , max = 0 , min = 0 , counter=0;
 cin>>n;
 
 string s;
 cin>>s;
 
 w = count(s.begin(),s.end(),'w');
 max = n+w;


 for(int i=0;i<n;i++){
 	
 	if(i+1<n && s.at(i)=='v' && s.at(i+1)=='w'){
 	s.at(i+1)='v';
 	s.at(i)='w';
 	}
 	
 	else if(i+1 < n && s.at(i)=='v' && s.at(i+1)=='v'){
 	i++;
 	counter++;
 	}
 }



 min=n-counter;


 cout<<min<<" "<<max;
 
    return 0;
}
