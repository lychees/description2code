#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,h,i;
    	cin>>n>>h;
    	int ar[n],num[n];
    	for(i = 0; i<n;i++)
    	{
    		cin>>ar[i];
    		if(ar[i]>=h)
    		{
    			num[i] = ar[i]-h;
    		}
    		else if(ar[i]<h)
    		{
    			num[i] = h-ar[i];
    		}
    	}
    	sort(num,num+n);
    	cout<<num[0]<<endl;
    }
    return 0;
}
