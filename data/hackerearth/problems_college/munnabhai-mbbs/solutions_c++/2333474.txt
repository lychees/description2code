#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<vector>
#include<map>
#include<queue>
#include<cstring>
#include<cmath>
#include<algorithm>
using namespace std;
vector<int> g[1001];
int in[1001];
int ans=0;
void dfs(int cur)
	{
	ans++;
	int i;
	for(i=0;i<g[cur].size();i++)
		in[g[cur][i]]--;
	}
int main()
	{
	int i,j,k;
	int T;
	int N,M;
	ios_base::sync_with_stdio(false);
	cin>>T;
	while(T--){
		ans=0;
		cin>>N>>M;
		for(i=1;i<=N;i++)
			g[i].clear(),in[i]=0;
		while(M--)
			{
			cin>>i>>j;
			g[i].push_back(j);
			in[j]++;
			}
	bool pass=true;
	while(pass){
		pass=false;
		for(i=1;i<=N;i++){
			if(in[i]==0){
				in[i]=-1;
				pass=true;
				ans++;
				for(j=0;j<g[i].size();j++)
					in[g[i][j]]--;
			}
		}
	}
	printf("%d\n",ans);
	}
	
	return 0;	
	}
