#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long t;
    cin>>t;

    int ar[100001],ar1[100001];
    while(t--)
    {
        int n;
        cin>>n;
        for(int i=0;i<n;i++)
            cin>>ar[i];
        for(int i=0;i<n;i++)
            cin>>ar1[i];
            sort(ar,ar+n);
            sort(ar1,ar1+n);
            int flag=0;
            for(int i=0;i<n;i++)
            {
                if(ar[i]<ar1[i])
                {
                    flag=1;
                    break;

                }
            }
            if(flag==1)
                cout<<"NO";
            else
                cout<<"YES";
            cout<<endl;

    }
    return 0;
}
