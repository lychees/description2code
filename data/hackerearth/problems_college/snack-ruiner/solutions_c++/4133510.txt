#include<iostream>
#include<stdlib.h>

using namespace std;
void fun(float a[],float b[],int c[],int n,float value)
{
    int index=0;
    float res=0;
    for(int i=0;i<n;i++)
    {
        if(res<=(b[i]*a[i])/value)
        {
           res=(b[i]*a[i])/value;
           index=i;
        }
    }
    cout.setf(ios::fixed,ios::floatfield);
    cout.precision(3);
    cout<<index<<" "<<res<<endl;
}
int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
        int n;
        cin>>n;
        float d[n];
        float f[n];
        int c[n];
        float value=0;
        float sum=0;
        for(int j=0;j<n;j++)
            {
                cin>>d[j];
                sum+=d[j];
            }
        for(int j=0;j<n;j++)
            {
                cin>>f[j];
                value+=f[j]*d[j];
                if(d[j]/sum>=0.5)
                    c[j]=1;
                else
                    c[j]=0;

            }
        fun(d,f,c,n,value);
    }
    return 0;
}
