#include<bits/stdc++.h>

using namespace std;
int main(int argc, char const *argv[])
{

	int T; cin>>T;
	while(T--){
	int counter =0;	
	string s1,s2; int n; cin>>s1>>s2; cin>>n;	
	for (int i = 0; i < s1.size(); ++i)
	{
		if(s1[i] != s2[i]) counter++;
	}


	if( (n - counter) >=0 && (n - counter)%2 == 0) cout<<"Yes";

	else cout<<"No";
	cout<<endl;

             }
	
	return 0;
}
