#include <iostream>
using namespace std;

void drawstar(int n){
	
	if(n == 1)
	return;
	int i;
	for(i=0;i<(n-1);i++)
		cout << " ";
	
	cout << '*';
	cout << endl;
	n--;
	
	drawstar(n);
	
	
	


}
