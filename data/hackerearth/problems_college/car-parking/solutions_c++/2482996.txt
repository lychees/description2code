#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		char str[10000];
		int arr[10000]={0};
		int cnt=0,total=0,si=0,top=-1;
		cin>>str;
		for(int i=0;str[i]!='\0';i++)
		{
			if(str[i]=='S'){
				arr[i]=-1;
				if((i-1)>-1)
				arr[i-1]=-1;
				arr[i+1]=-1;
			}
			if(str[i]=='B'){
				arr[i]=-1;
				if((i-1)>-1)
				arr[i-1]=-1;
				if((i-2)>-1)
				arr[i-2]=-1;
			}
			if(str[i]=='D'){
				arr[i]=-1;
			}
		}
		for(int i=0;i<strlen(str);i++)
		{
			//cout<<arr[i]<<" ";
			if(arr[i]!=-1)
				cnt++;
		}
		cout<<cnt<<endl;
	}
}
