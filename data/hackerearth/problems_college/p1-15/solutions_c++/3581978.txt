#include<iostream>
using namespace std;
int main()
{  long int t;
cin>>t;
while(t--)
{
    string s1,s2;
    long int i=0,flag=0,l1,l2,pos;
    cin>>s1>>s2;
    l1=s1.length();
    l2=s2.length();
    cout<<s1;
    while(i<l1)
    {
       if(s1[i]==s2[0])
       {
        if(s1.substr(i) == s2.substr(0,l1-i))
        {
           // cout<<s1.substr(i)<<endl;
            pos=l1-i;
            flag=1;
            break;
        }
        }
       i++;
    }
    if(flag==1)
    {
       for(i=pos;i<l2;i++)
            cout<<s2[i];
    }
    else
       cout<<s2;
    cout<<endl;
}
}
