#include <algorithm>
#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int a,b,c,ans=0;
    	cin>>a>>b>>c;
    	int v[3];
    	v[0]=a;
    	v[1]=b;
    	v[2]=c;
    	sort(v,v+3);
    	a=v[0];
    	b=v[1];
    	c=v[2];
    	if(b-a>c-b)
    	{
    		while(b>a+1)
    	{	c=b;
    			b--;
    			ans++;
    	}
    	}
    	else
    	{
    			while(c>b+1)
    		{ 
    			a=b;
    			b++;
    			ans++;
    			
    		}
    	}
    
    	cout<<ans<<"\n";
    
    }
    
    return 0;
}
