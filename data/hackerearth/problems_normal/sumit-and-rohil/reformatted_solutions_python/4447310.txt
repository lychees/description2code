def check(s1, s2):
    return sorted(s1) == sorted(s2)


N = input()
names = [raw_input().strip(" \n") for i in xrange(N)]

groups = []
while names:
    tmp = []
    test = names[0]
    group = [test]
    for i in names[1:]:
        if test[0] == i[0] and test[-1] == i[-1] and check(test, i):
            group.append(i)
        else:
            tmp.append(i)
    groups.append(group)
    names = tmp[:]

print len(groups)
