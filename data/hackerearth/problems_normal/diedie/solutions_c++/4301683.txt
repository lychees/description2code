ll int power(ll int a,ll int b)
{
    ll int res=1;
    while(b)
    {
        if(b%2==1)
        {
            res=((res%MOD)*(a%MOD))%MOD;
        }
        b=b>>1;
        a=((a%MOD)*(a%MOD))%MOD;
    }
    return res%MOD;
}
