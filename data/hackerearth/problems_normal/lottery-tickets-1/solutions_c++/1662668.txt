#include <stdio.h>

long long int a [101];
long long int b [101];
long long int c [101];
long long int f [101] = {0};

long long int max (long long int i, long long int j) {
	if (i > j) {
		return i;
	}
	return j;
}

int main ()
{
	long long int i, j, n, result, t, tt;
	
	scanf ("%lld", &t);
	
	while (t--) {
	
	scanf ("%lld", &n);
	result = 0;
	
	for (i = 0; i < n; i++) {
		scanf ("%lld%lld%lld", & a [i], & b [i], & c [i]);
		f [i] = 0;
	}
	
	for (i = 0; i < n; i++) {
		for (j = i + 1; j < n; j++) {
			if (b [i] > b [j]) {
				tt = b [i];
				b [i] = b [j];
				b [j] = tt;
				tt = a [i];
				a [i] = a [j];
				a [j] = tt;
				tt = c [i];
				c [i] = c [j];
				c [j] = tt;
			}
		}
	}
	f [0] = c [0];
	for (i = 0; i < n; i++) {
		f [i] = c [i];
		for (j = 0; j < i; j++) {
			if (a [i] >= b [j]) {
				f [i] = max (f [i], f [j] + c [i]);
			}
			
		}
		if (result < f [i]) {
				result = f [i];
			}
	}
	
	printf ("%0.2lf\n", double(result)/100.0);
	}
	
	return 0;
}
