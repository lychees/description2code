#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

char s[11][11];
char t[4][4][11][11];

void ConverMatrix(int n){
    for (int k = 1; k < 4; k++)
        for (int i = 0; i < 10; i++)
            for (int j = 0; j < 10; j++)
                t[n][k][10 - j - 1][i] = t[n][k - 1][i][j];
}

int isSquare(char s[11][11], char t[11][11]){
    for (int i = 0; i < 10; i++)
        for (int j = 0; j < 10; j++)
            if (s[i][j] == t[i][j])
                return 0;
    return 1;
}

int Check(int n){
    for (int i = 0; i < 4; i++)
        if (isSquare(s, t[n][i]))
            return 1;
    return 0;
}

void ReadMatrix(char s[11][11]){
    for (int i = 0; i < 10; i++){
        do{
            scanf("%s", s[i]);
        }while(strlen(s[i]) == 0);
    }
}

int main(){
    int cs;
    scanf("%d", &cs);
    while (cs--){
        memset(s, 0, sizeof(s));
        memset(t, 0, sizeof(t));
        ReadMatrix(s);
        for (int i = 0; i < 4; i++){
            ReadMatrix(t[i][0]);
            ConverMatrix(i);
        }
        for (int i = 0; i < 4; i++)
            if (Check(i)){
                printf("%d\n", i + 1);
                break;
            }
    }

    return 0;
}
