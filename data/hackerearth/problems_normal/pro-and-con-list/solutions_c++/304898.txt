#include <iostream>
using namespace std;

int main()
{
	int testCases;
	scanf("%d\n",&testCases);
		
		while(testCases--)
		{
			long long int n;
			scanf("%lld\n",&n);
			long long int favour , anger;
			long long int sum = 0,max1=0,max2=0,s;
			while(n-->0)
			{
				long long int temp;
				scanf("%lld %lld\n",&favour,&anger);
				s=favour+anger;
				if(s>max1)
				{
					temp = max1;
					max1=s;
					s=temp;
				}
				if(s>max2)
				{
					max2=s;
				}
				sum = sum+anger;
			}
			sum = max1+max2-sum;
			printf("%lld \n",sum);	
		}
    
    return 0;
}
