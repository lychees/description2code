#include <bits/stdc++.h>
using namespace std;
#define F first
#define S second
#define d double
#define MP make_pair
int cmp(pair<d,d> a,pair<d,d> b)
{
	if(a.F == b.F) return a.S<b.S;
	return a.F<b.F;
}
int main() {
	pair<d,d> p[3];d x,y;
	for(int i=0;i<3;i++)
	{
		scanf("%lf %lf",&x,&y);
		p[i]=MP(x,y);
	}
	sort(p,p+3,cmp);
	x=p[0].F+p[1].F-p[2].F;
	y=p[0].S+p[1].S-p[2].S;
	printf("%.4lf %.4lf\n",x,y);
	x=p[0].F-p[1].F+p[2].F;
	y=p[0].S-p[1].S+p[2].S;
	printf("%.4lf %.4lf\n",x,y);
	x=-p[0].F+p[1].F+p[2].F;
	y=-p[0].S+p[1].S+p[2].S;
	printf("%.4lf %.4lf",x,y);
	return 0;
}
