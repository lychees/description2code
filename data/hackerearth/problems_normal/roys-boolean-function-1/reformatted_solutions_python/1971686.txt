from fractions import gcd
import math


def pn(n):
    if n <= 1:
        return "FALSE"
    if n == 2:
        return "TRUE"
    if n % 2 == 0:
        return "FALSE"
    m = int(sqrt(n))
    for i in xrange(3, m+1, 2):
        if n % i == 0:
            return "FALSE"
    return "TRUE"


testcases = int(raw_input())
while testcases:
    n = int(input())
    count = 0
    for i in range(1, n+1):
        if gcd(n, i) == 1:
            count += 1
    print pn(count)
    testcases -= 1
