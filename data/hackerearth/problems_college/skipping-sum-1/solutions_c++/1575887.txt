#include <iostream>
#include <cstdio>
using namespace std;
long S[100005][11], t[100005];

int main()
{
    int n, q;
    cin>>n>>q;
    for(int i = 1; i <= n; i++) cin>>t[i];
    for(int k = 1; k <= 10; k++) S[1][k] = t[1];
    for(int i = 2; i <= n; i++){
        for(int k = 1; k <= 10; k++){
            int j = i-k;
            if(j < 1) S[i][k] = t[i];
            else S[i][k] = S[j][k] + t[i];
        }
    }
    // for(int i = 1; i <= n; i++){
    //     for(int k = 1; k <= 10; k++) printf("%4d ", S[i][k]);
    //     cout<<"\n";
    // }
    int l, r, k;
    while(q--){
        cin>>l>>r>>k;
        int m = (r-l)/k;
        r = l + k*m;
        cout<<S[r][k]-S[l][k]+t[l]<<endl;
    }
    return 0;
}
