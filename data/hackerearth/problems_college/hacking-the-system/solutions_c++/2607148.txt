#include <iostream>
#include <string.h>
using namespace std;
int main()
{
	char a[100000],b[100000];
	int i,count=0;
    cin>>a>>b;
    for (i=0;i<strlen(a);i++)
    {
    	if (a[i]!=b[i])
    		count++;
    }
    if (count==1 || count==0)
    cout<<"wins";
    else
    cout<<"loses";
}
