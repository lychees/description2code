#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

ll mo= 1000000007;
vector <ll> prime;

	ll t,n,A[1300000],B[1300000];
	
int main() {
	A[1]=2;
	for(ll i=2;i<1300000;i++)
	{
		if(A[i]==0)
		{
			prime.push_back(i);
			for(ll j=2*i;j<1300000;j=j+i)
			A[j]=1;
		}
		
	}
	for(ll i=2;i<=100000;i++)
	{
		A[i] = (A[i-1]*prime[i-1])%mo;
	}
	cin>>t;
	while(t--)
	{
		cin>>n;
		cout<<A[n]<<endl;
	}
	return 0;
}
