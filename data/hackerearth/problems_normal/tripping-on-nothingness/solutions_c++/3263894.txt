#include <iostream>
#include<set>
#include<algorithm>
using namespace std;
int gcd(int a,int b)
{
	if(a==0)
	return b;
	if(b==0)
	return a;
	return gcd(b,a%b);
	
}
int main()
{
	int s,a,x,q;
	cin>>s>>q;
	set<int>si;
	while(q--)
	{ cin>>a;
		x=gcd(s,a);
		if(si.find(x)==si.end())
		{
			cout<<x<<endl;
			si.insert(x);
		}
		else
		{
			cout<<"-1"<<endl;
		}
	}
    //cout << "Hello World!" << endl;
    return 0;
}
