#include<bits/stdc++.h>
using namespace std;
#define ll long long

ll arr[100005],ans[100005];
int main()
{
    ll n,i,x;
    cin>>n;
    for(i=1;i<=n;i++)
    {
        cin>>arr[i];
        ans[i]=arr[i];
    }
    for(i=1;i<=n;i++)
    {
        x=2;
        while(i*x<=n)
        {
            ans[i*x]=max(ans[i*x],ans[i]+arr[i*x]);
            x++;
        }
    }

    for(i=1;i<=n;i++)
        cout<<ans[i]<<" ";

    return 0;
}
