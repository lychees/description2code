#include <iostream>
#include<cmath>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
     int t;
   cin>>t;
   while(t--)
   {
   	int x;
   	cin>>x;
   	int a=1;
   	int diff=0,t1;
   	for(int i=a;i<x;i++)
   	{
   		int b=x-i;
   		if((pow(a,b)+pow(b,a))==x)
   		{
   			int dif=abs(a-b);
   			if(dif>diff)
   			{
   				diff=dif;
   				t1=a;
   			}
   		}
   	}
   	if(t1>(x-t1))
   	cout<<t1<<" "<<(x-t1)<<endl;
   	else
   	cout<<(x-t1)<<" "<<t1<<endl;
   }
    return 0;
}
