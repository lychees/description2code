#include <iostream>
using namespace std;

int main()
{
	int t;
	cin >>t;
	while(t--)
	{
		long long int n,num[1000],c=0,i,j;
		cin>>n;
		for(i =0;i<n;i++)
		{
			cin>>num[i];
		}
		for(i=0;i<n;i++)
		{
			for(j=i+1;j<n;j++)
			{
				if(num[i] != num[j] && (num[i]+num[j])% 2 ==0)
				c++;
			}
		}
		
		cout<<c<<endl;
	}

    return 0;
}
