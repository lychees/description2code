#include<bits/stdc++.h>
#include<iostream>

#define fre 	freopen("0.in","r",stdin),freopen("0.out","w",stdout)
#define abs(x) ((x)>0?(x):-(x))
#define MOD 1000000009
#define lld signed long long int
#define pp pop_back()
#define ps(x) push_back(x)
#define mpa make_pair
#define pii pair<int,int>
#define fi first

#define scan(x) scanf("%d",&x)
#define print(x) printf("%d\n",x)
#define scanll(x) scanf("%lld",&x)
#define printll(x) printf("%lld\n",x)
#define boost ios_base::sync_with_stdio(0)
#define MaxVal 3*5*100000
int tree[3*5*100000+5];
void update(int idx ,int val){
    while (idx <= MaxVal){
        tree[idx] += val;
        idx += (idx & -idx);
    }
}
int read(int idx){
    int sum = 0;
    while (idx > 0){
        sum += tree[idx];
        idx -= (idx & -idx);
    }
    return sum;
}
int wh[5*100000+5];
int main()
{
	//fre;
	int x,N,M;
	char c;
	cin>>N>>M;
	int L = 500000+1;
	int R = L+N-1;
	for(int i=1;i<=N;++i)
	{
		scan(x);
		wh[x] = L+i-1;
		update(wh[x],1);
	}
	lld ans = 0;
	while(M--)
	{
		scan(x);
		scanf(" %c",&c);
		int pos = read(wh[x]);
		ans += min(pos-1,N-pos);
		update(wh[x],-1);
		if(c=='l')
		{
			L--;
			wh[x]=L;
		}
		else
		{
			R++;
			wh[x]=R;
		}
		update(wh[x],1);
	}
	printll(ans);
}
