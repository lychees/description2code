    #include <bits/stdc++.h>
    using namespace std;
     
    int main()
    {
    	vector<long long > v;
        int n;
		cin>>n;
        long long  x;
        for(int i=0;i<n;i++){
        	cin>>x;
        	v.push_back(x);
        }
        	
        sort(v.begin(),v.end());
       
        long long  temp1,temp2;
        temp1=v[n-1]*v[n-2];
        temp2=v[0]*v[1];
        
        if(temp1>temp2)
        	cout<<temp1;
        else
        	cout<<temp2;
        	
        return 0;
    }
