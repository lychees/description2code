#include <iostream>
using namespace std;
const int mod = 1000000007;
int main()
{
	int N;
	cin >> N;
	int *A = new int[N];
	int lsum = 0;
	for(int i = 0; i < N; ++i)
	{
		cin >> A[i];
		lsum += A[i];
	}
	long *list[2];
	list[0] = new long[lsum+1];
	list[1] = new long[lsum+1];
	for(int i = 0; i <= lsum; ++i)
	{
		list[0][i] = list[1][i] = 0;
	}
	int start = 0;
	int end = 1;
	for(int i = 0; i < N; ++i)
	{
		start = i%2;
		end = (i+1)%2;
		for(int j = 1; j <= lsum; ++j)
		{
			list[end][j] = list[start][j];
		}
		++list[end][A[i]];
		for(int j = 1; j <= (lsum-A[i]); ++j)
		{
			list[end][j+A[i]] = (list[end][j+A[i]]+list[start][j])%mod;
		}
	};
	int pp = 0;
	int suml = lsum;
	while(suml)
	{
		pp = (pp << 1);
		++pp;
		suml = (suml >> 1);
	};
	//cout << pp << endl;
	int *v = new int[pp+1];
	for(int i = 0; i <= pp; ++i)
	{
		int p = 0;
		while(p <= pp)
		{
			int tmp = (p^i);
			if ((tmp <= lsum) && (list[end][tmp] > 0))
			{
				v[i] = tmp;
				break;
			}else
			{
				++p;
			}
		}
	}
	//cout << p << endl;
	int Q;
	cin >> Q;
	for(int i = 0; i < Q; ++i)
	{
		int M;
		cin >> M;
		M = (M&pp);
		cout << v[M] << " " << list[end][v[M]] << endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
