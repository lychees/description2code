#include<bits/stdc++.h>
using namespace std;

int primes[1000001];

  void gen_primes() 
     { 
       int i,j; 
       for(i=0;i<10000;i++) primes[i] = 1; 
       for(i=2;i<=(int)sqrt(10000);i++) 
         if (primes[i]) 
           for(j=i;j*i<10000;j++) primes[i*j] = 0; 
     } 


int main() {
	int t;
	cin>>t;
	gen_primes();
	string s;
	while(t--) {
		cin>>s;
		int f = 0;
		for(int i = 0 ; i < s.length(); i++) {
			if(!primes[s[i]]) {
				f = 1;
				cout<<s[i];
			}
		}
		if(f == 0) {
			cout<<"NULL";
		}
		cout<<"\n";
	}
	return 0;
}
