#include<bits/stdc++.h>
using namespace std;
int main()
{
   int n;
   cin>>n;
   long long int arr[n];
   for(int i=0;i<n;i++)
   {
       cin>>arr[i];
   }
   vector<pair<int,int> > vc[100005];
   vector<int> answer[200];
   for(int i=0;i<n;i++)
   {
       long long int sum=0;
       for(int j=i;j<n;j++)
       {
           sum+=arr[j];
           pair<int,int> pair1(j,i);
           vc[sum].push_back(pair1);
       }
   }
   for(int i=0;i<n;i++)
   {
       sort(vc[i].begin(),vc[i].end());
   }
   for(int i=0;i<=100000;i++)
   {
       if(vc[i].size()>=1)
       {
           int count1=1;
           int end1=vc[i][0].first;
           for(int j=1;j<vc[i].size();j++)
           {
               if(vc[i][j].second>end1)
               {
                   count1++;
                   end1=vc[i][j].first;
               }
           }
           answer[count1].push_back(i);
       }
   }
   for(int i=200;i>=0;i--)
   {
       if(answer[i].size()>=1)
       {
           cout<<i<<endl;
           for(int j=0;j<(answer[i].size());j++)
            cout<<answer[i][j]<<" ";
           cout<<endl;
           break;
       }

   }
   return 0;
}
