#include <iostream>
using namespace std;

#define N 2000000
#define M 1000000007

int p[N] = {0};

void sieve(){
	p[0] = p[1] = 1;
	for(int i = 2; i*i < N; i++){
		if(!p[i]){
			for(int j = i*i; j < N; j += i){
				p[j] = 1;
			}
		}
	}
}

int main()
{
    sieve();
    int n,min = 0, max;
    cin >> n;
    long long count = 0;
    max = 2*n-2;
    for(int i = 1; i < n; i++){
    	if(!p[min]){
    		count = (count + i)%M;
    	}
    	if(!p[max]){
    		count = (count + i)%M;
    	}
    	min++;
    	max--;
    }
    if(!p[min]){
    	count = (count + n)%M;
    }
    cout << count << endl;
    return 0;
}
