#include <iostream>
#include<vector>
using namespace std;
vector<long long >prime;
long long  i, j, y, sieve[1000009], t;
void sie()
{
	y = 1000001;
	for (i = 2; i <= y; i++)
		sieve[i] = i;
	for (i = 2; i*i <= y; i++)
		if (sieve[i] == i)
		{
		for (j = i*i; j <= y; j = j + i)
			sieve[j] = i;
		}
	for (i = 2; i <= y; i++)
		if (sieve[i] == i)
			prime.push_back(i);


}
int main()
{
	long long num, ans;
	int tc;
	cin >> tc;
	sie();
	while (tc--)
	{
		cin >> num;
		i = 0;
		ans = 0;
		while (prime[i] * prime[i] <= num)
		{
			if (num%prime[i] == 0)
			{
				num = num / prime[i];
				if (num%prime[i] == 0)
				{
					ans = 1;
					break;
				}
			}
			i++;
		}
		if (ans == 0)
			cout << "Yes\n";
		else
			cout << "No\n";
	}
	return 0;
}
