    #include <bits/stdc++.h>
     
    int main()
    {
        int t;
        scanf("%d",&t);
        while(t--)
        {
        	int n;	
        
        	scanf("%d",&n);
        	long arr[n];
        	
        	for(int i = 0 ;i<n ;i++)
        		scanf("%ld",&arr[i]);
        	long ns =0,c=0,pf=0,min=10000000000,rise;
     		int i=1,j;   
        	for(i=0;i<n-1;i++)
        	{
        		rise=0;
        		for( j = i + 1 ; j < n ; j++)
        		{
        			if ( arr[j] > arr[i] )
        			{
        				rise = 1;
        				break;
        			}
        			
        		}
        		if(rise)
        		{
        			c++;
        			pf -= arr[i] ;
        		}
        		else if( c )
        		{
        			pf += c*arr[i] ;
        			c=0;
        		}
        		
        	}
        	pf += arr[n-1]*c;
        	if(pf>0)
        		printf("%ld\n",pf);
        	else
        		puts("0");
        }
        return 0;
    }
