#include <iostream>
#include <cstring>
#include <algorithm>
#include <cstdio>
using namespace std;
int par[1000010];
int get_parent(int a) {
	if (par [a] == -1 || par [a] == a) return a;
	par[a] = get_parent(par[a]);
	return par[a];
}
void make_friend(int a, int b) {
	int x = get_parent(a), y = get_parent(b);
	if (x > y) swap(x,y);
	par[y] = x;
}
int main()
{
    int T; 
    for (cin >> T; T; T--) {
    	int N, E; cin >> N >> E;
    	memset(par, -1 ,sizeof(par));
    	for (int i = 0; i < E; i++) {
    		int a, b;
    		scanf("%d%d", &a,&b);
    		if (a > b) swap(a,b);
    		make_friend(a, b);
    	}
    	int ans =0 ;
    	for (int i = 0; i < N; i++) if (par[i] == -1 || par[i] == i) ans++;
    	cout << ans <<"\n";
    }
}
