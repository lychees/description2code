#include <bits/stdc++.h>

using namespace std;

int prim[1000004] = {0};
int sieve[1000004] = {0};

void sieve1()
{
	prim[1] = 0;
	for(int i = 2; i <= 1000000; i++) {
		if(sieve[i] == 0) {
			prim[i] = 1;
			for(int j = 2*i; j <= 1000000; j+=i) {
				sieve[j] = 1;
			}
		}
	}
	for(int i = 2; i <= 1000000; i++) {
		prim[i] = prim[i] + prim[i-1];
//		cout << prim[i] << endl;
	}
}

int main()
{
	int t,a,b;
	sieve1();
	scanf("%d",&t);
	while(t--) {
		scanf("%d %d",&a,&b);
		float count = (float)b-a+1;
		float c1 = (float)prim[b];
		if(a > 1) {
			c1 = c1 - prim[a-1];
		}
		printf("%f\n",(c1/count));
	}
	return 0;	
}
