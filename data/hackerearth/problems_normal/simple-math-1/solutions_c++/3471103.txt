#include<bits/stdc++.h>
#define MOD 10000007
using namespace std;
int main()
{
	long long n,m,i,j,k,solve[105][105]={0};
	cin>>n>>m;
	for(i=0;i<102;i++){
    solve[i][0]=1;
    solve[i][i]=1;
    }
    for(i=1;i<102;i++)
        for(j=1;j<102;j++)
            if (i!=j)
                solve[i][j]=(solve[i-1][j]%MOD+solve[i-1][j-1]%MOD)%MOD;
      if(n<m)printf("-1\n");
    else{
    n-=m;
    printf("%lld",solve[n+m-1][m-1]%MOD);}
	

}
