#include <bits/stdc++.h>
using namespace std;

int main()
{
   ios_base::sync_with_stdio(0);
   cin.tie(0);
    int t;
    string s;
   cin>>t;
   while(t--){
   	cin>>s;
   	int con=0,vow=0,tcon=0;
   	for(int i=0;i<s.length();i++){
   		if(s[i]=='a' || s[i]=='e' || s[i]=='i' || s[i]=='o' || s[i]=='u' || s[i]=='y' ){
   			con=0;
   			vow++;
   		
   		}
   		else{
   			con++;
   			tcon++;
   		
   			if(con>=3){
     				break;
   				
   			}
   		}
   		
   	}
   	if(con>=3 || vow<tcon){
   		cout<<"hard\n";
   	}
   	else{
   		cout<<"easy\n";
   	}
   }
    return 0;
}
