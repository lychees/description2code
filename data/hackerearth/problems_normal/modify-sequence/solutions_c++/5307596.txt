#include <iostream>
using namespace std;

int main()
{
    int N;
    cin>>N;
    int i,a[N];
    for(i=0;i<N;i++)
      cin>>a[i];
    bool flag=true;
    i=0;
    while(i<N){
    	a[i+1]-=a[i];
    	if(a[i+1]<0)
    	  flag=false;
      i++;	  
    }
    a[0]=0;
    if(a[i-1] !=0)
       flag=false;
    for(int j=1;j<N;j++){
    	if(a[j]!=0)
    	   flag=false;
    	else
    	   flag=true;
    } 
    if(flag == true)
       cout<<"YES";
    else
     cout<<"NO";
    return 0;
}
