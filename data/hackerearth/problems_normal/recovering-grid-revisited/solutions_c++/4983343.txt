#include<bits/stdc++.h>
using namespace std;
#define ll long long  

long long  powmod(ll a,ll b)
{
	ll res=1;
	while(b)
	{
		if(b&1) res=(res*a)%1000000007;
		a=(a*a)%1000000007;
		b>>=1;
	}
	return res;
}
int main()
{
	ll t,n,m,sum,i,j;
	ll arr[10000005];
	arr[0]=1;
	for(i=1;i<10000005;i++)
	{
	  arr[i]=(arr[i-1]*i)%1000000007;	
	}
	cin>>t;
	while(t--)
	{
		sum=0;
		cin>>n>>m;
		ll r=n*n-1;
		ll n1=r+m-1;
		ll r1=r-1;
		if (n == 1){
			cout << 0 << endl;
			continue;
		}
		ll num=arr[n1]%1000000007;
		ll denom=(arr[r1]%1000000007*arr[n1-r1]%1000000007)%1000000007;
		sum=(num*powmod(denom,1000000007-2))%1000000007;
		
		cout<<sum<<endl;
	}
}
