#include<bits/stdc++.h>
using namespace std;
int B[100010],G[100010],bt[100010],gt[100010];
int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		int N;
		cin >> N;
		map<int,int> mp;
		map<int,int> gp;
		for(int i=0;i<=N;i++)bt[i]=0;
		for(int i=0;i<=N;i++)gt[i]=0;
		for(int i=1;i<=N;i++)cin >> B[i];
		for(int i=1;i<=N;i++)cin >> G[i];
		for(int i=1;i<=N;i++)
		{
			if(G[B[i]]!=i)
			{
				bt[G[B[i]]]++;
				mp[i]=G[B[i]];
			}
			if(B[G[i]]!=i)
			{
				gt[B[G[i]]]++;
				gp[i]=B[G[i]];
			}
		}
		int ans=0;
		map<int,int>::iterator it,iit;
		for(it=mp.begin();it!=mp.end();it++)
		{
			if(mp[it->second]==it->first)ans++;
		}
		for(iit=gp.begin();iit!=gp.end();iit++)
		{
			if(gp[iit->second]==iit->first)ans++;
		}
		int mx=bt[1];
		for(int i=2;i<=N;i++)mx=max(mx,bt[i]);
		for(int i=1;i<=N;i++)mx=max(mx,gt[i]);
		cout<<mx<<" "<<ans/2<<endl;
	}
	return 0;
}
