#include<bits/stdc++.h>
using namespace std;
const int MOD = 1e9+7;
const int MAXN = 1e6+3;
long long dp[MAXN];
long long endingInE[MAXN],endingInH[MAXN];
int main() {
	int t;
	cin>>t;
	dp[1]=2;
	endingInH[1]=1;
	endingInE[1]=1;
	for(int i=2;i<MAXN;i++) {
		dp[i]=dp[i-1];
		dp[i]+=2*endingInH[i-1]+endingInE[i-1];
		if(dp[i]>=MOD)
			dp[i]%=MOD;
		endingInE[i]=endingInH[i-1];
		endingInH[i]=endingInH[i-1]+endingInE[i-1];
		if(endingInH[i]>=MOD)
			endingInH[i]%=MOD;
		if(endingInE[i]>=MOD)
			endingInE[i]%=MOD;
	
	}	
	while(t--) {
		int n;
		scanf("%d",&n);
		printf("%lld\n",dp[n]);
	}
	return 0;
}
