#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t>0){
    	int p,q,s;
    	cin >> p >> q >> s;
    	
    	int x;
    	if(s==1 && q!=s)
    		 x = 1;
    	else
    		x = 0;
    
    	for(int i=2;i*i<=q;i++){
    		if(q%i==0 &&(q/i>=s || i >= s))
    			x=1;
    	}
    	if(x&&p%2==1)
    		cout << "Bernard" << endl;
    	else
    		cout << "Havanese" << endl;
    	
    	t--;
    }
    return 0;
}
