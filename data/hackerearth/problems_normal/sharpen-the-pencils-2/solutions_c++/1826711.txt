#include<bits/stdc++.h>
using namespace std;
int main(){
	int t,n;long long int sum;
	int arr[100007],mark[100007];
	cin>>t;
	while(t--){
		memset(mark,0,sizeof(mark));
		cin>>n;
		for(int i=0;i<n;i++)
		cin>>arr[i];
		int i=0,j=n-1;
		sum=0;
		while(i<j){
			mark[j]=0;
			sum=arr[j]*2;
			while(i<j && sum>0){
				if(arr[i]<=sum){
					sum-=arr[i];
					mark[i]=1;
					i++;
				}
				else
				{
					mark[i]=1;
					arr[i]-=sum;
					sum=0;
				}
			}
			
			j--;
		}
		int c=0;
		for(int i=0;i<n;i++){
			if(mark[i]==1) c++;
		}
		cout<<c<<" "<<(n-c)<<endl;
		
	}
	return 0;
}
