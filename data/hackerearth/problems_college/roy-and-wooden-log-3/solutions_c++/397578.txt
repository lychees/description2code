#include <iostream>
#include <algorithm>
#include <cmath>

using namespace std;

typedef long long int ll;

int main()
{
   	int t;
   	ll length,w,c;
   	cin>>t;
   	while(t--)
   	{
   		cin>>length;
   		c = ceil(length/2.0)-1;
   		w = length/2 + 1;
   		cout<<w<<" "<<c<<"\n";
   	}
    return 0;
}
