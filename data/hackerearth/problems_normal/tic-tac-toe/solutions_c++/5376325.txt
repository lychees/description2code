#include <iostream>
using namespace std;
 
int maxs(int m){
	unsigned long long int n = m-1;
    unsigned long long int sum;
    unsigned long long int mod = 1000000007;
    sum = (n*(n+1))%mod;
    sum = ((2*n+1)*sum)%mod;
    sum = (sum*166666668)%mod;
    return sum;
}
 
int mins(int n){
    unsigned long long int sum = 0;
    unsigned long long int mod =  1000000007;
    unsigned long long int ematches = (n-1)/2 ;
 
    sum = (n*ematches)%mod;
    sum = (sum*ematches)%mod;
    return sum;
}
 
 
int main(){
    int tn; cin >> tn;
    for(int i=0;i<tn;i++){
        int n; cin >> n;
        cout << mins(n) << " " << maxs(n) << endl;
    }
    return 0;
}
