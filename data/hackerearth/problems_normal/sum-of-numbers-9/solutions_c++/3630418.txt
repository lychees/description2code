#include <bits/stdc++.h>
using namespace std;
#define ll long long 
#define fi first 
#define se second
#define pb push_back  
#define mp make_pair


int main()
{
   int t;
   ll flag,s,i, j, k ,a[100],ans, temp, n ;

   cin>>t;

   while( t-- )
   {
    cin>>n;
    flag=0;
    for(i=0;i<(n);i++)
      cin>>a[i];
    cin>>s;
    for(i=0;i<=(1<<n);i++)
    {
      temp=i;
      j=0;
      ans=0;
      while(temp)
      { 
        if((temp&1))
          ans+=a[j];
        j++;
        temp>>=1;
      }
      if(ans==s)
        {flag=1;break;}
    }
    if(flag)
     cout<<"YES";
    else
     cout<<"NO";
     cout<<endl;
   }
  


  return 0;
}
