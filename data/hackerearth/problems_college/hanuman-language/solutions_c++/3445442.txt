#include <stdio.h>
#include <iostream>
using namespace std ; 

int main()
{
    int t ; 
    cin >> t ; 
    getchar() ; 
    while  ( t--)
    {
       string a ; 
       string b[10000] ; 
       getline ( cin , a ) ; 
       int l = a.length() ; 
       char c[100] ; 
       int k = 0 ; 
       int j = 0 ; 
       
       int i ; 
       for ( i = 0; i< l ; i++)
       {
       	    if ( a[i] == ' ')
       	    {
       	      c[k] = '\0' ; 
       	      i = i + 1 ; 
			  b[j]	 = c  ;
			  j = j+1 ; 
			  k = 0 ; 
			}
			c[k] = a[i] ; 
			k = k+1 ; 
       	
	   }
	   c[k] = '\0' ; 
	   b[j] = c ; 
	   
	   for ( i = j; i>= 0 ;i--)
	   {
	   	cout << b[i] << " "; 
	   }
	   cout << endl; 
       
	}
    return 0;
}
