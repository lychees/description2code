#include <iostream>
#include<map>
#include<math.h>
using namespace std;

int main()
{
	map<string ,int>m;
	map<string ,int>::iterator it;
	string s1,s2;
	int n;
	cin>>n;
	int x=pow(2,n)-1;
	for(int i=0;i<x;i++)
	{
		cin>>s1>>s2;
		
		if(m.find(s1)!=m.end())
		{
			m[s1]++;
		}
		else
		{
			m[s1]=1;
		}
	}
	int max=0;
	string value;
	for(it = m.begin(); it != m.end(); ++it )
	{
	//  cout<<it->first<<" "<<it->second<<endl;
	  if( it->second >max )
	  {
	  	value=it->first;
	   max=it->second;
	  }
	}
	
	cout<<value;
   // cout << "Hello World!" << endl;
    return 0;
}
