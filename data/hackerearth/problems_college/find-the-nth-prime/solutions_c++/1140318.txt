#include <iostream>
#include <math.h>
using namespace std;

bool IsPrime (int n)
{
  bool flag = 1;
  if((n==2)||(n==3)) return true;
  for(int i=2;i<=sqrt(n);i++){
          if(n%i==0){flag = 0;break;}
  }
return flag;
}

int main()
{
                int N = 0;
                cin >> N;
                int c = 0;

                int i =2 ;
                while(c !=N){
                        if(IsPrime(i)){
                                c++;
                                //cout<<"index: "<<c<<" prime:"<< i<<endl;
                        }
                        i++;
                        //cout<< "c: "<<c<<"i: "<<i<<endl;
                }
                cout << i-1<<endl;
}
