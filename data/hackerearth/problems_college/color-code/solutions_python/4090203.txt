for i in range(input()):
	l=[int(x) for x in raw_input().split()]
	s=""
	s1=""
	for j in range(0,len(l)):
		if(len(hex(l[j])[2:])==1):
			s+="0"+hex(l[j])[2:]+" "
		else:
			s+=hex(l[j])[2:]+" "
	for h in range(0,len(s)):
		if(s[h]=="a"):
			s1+="U"
		elif(s[h]=="b"):
			s1+="V"
		elif(s[h]=="c"):
			s1+="W"
		elif(s[h]=="d"):
			s1+="X"
		elif(s[h]=="e"):
			s1+="Y"
		elif(s[h]=="f"):
			s1+="Z"
		else:
			s1+=s[h]
	print "Case",str(i+1)+":",s1
