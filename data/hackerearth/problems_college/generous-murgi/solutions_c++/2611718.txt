#include<bits/stdc++.h>
#define ll long long

using namespace std;


ll calcf(ll a,ll b)
{

    ll cnt=0;
    ll x=sqrt(a)+1;
   
    for(int i=1;i<x;i++)
    {
        if(a%i==0)
        {
            if(i>b)
                cnt++;
            if((a/i)>b)
                cnt++;
        }
    }
     if(int(sqrt(a))*int(sqrt(a))==a)
    {

      if(int(sqrt(a))>b)
        cnt--;
    }
    return cnt;
}

int main()
{

    ll t,x,y;
    cin>>t;
    while(t--)
    {
        cin>>x>>y;
        if(x==y)
        cout<<"-1\n";
        else
        cout<<calcf(x-y,y)<<"\n";
    }
}
