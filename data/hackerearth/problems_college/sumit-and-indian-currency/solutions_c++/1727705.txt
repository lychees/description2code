#include <iostream>
using namespace std;

char str[50001];
int index = 0;

void convert(string word) {
	string newStr = "";
	int length = word.length();
	index = 0;
	str[index++] = word[length-1];
	str[index++] = word[length-2];
	str[index++] = word[length-3];
	for(int i=word.length()-4;i>=0;i-=2){
		str[index++] = ',';
		str[index++] = word[i];
		if(i-1>=0)
		str[index++] = word[i-1];
	}
}

int main() {
	int tc; cin >> tc;
	while(tc--){
		string word; cin >> word;
		if(word.length()<=3){
			cout << word << endl;
		}
		else {
			convert(word);
			for(int i=index-1;i>=0;i--){
				cout << str[i];
			}
			cout << endl;
		}
	}
	return 0;
}
