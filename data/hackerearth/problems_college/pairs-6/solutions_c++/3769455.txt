#include <stdio.h>
#include<math.h>
#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n;
    	scanf("%d",&n);
    	int b,g;
    	scanf("%d %d",&b,&g);
    	int c=abs(b-g);
    	if(c>1)
    	printf("Motu\n");
    	else
    	printf("Chotu\n");
    }
    return 0;
}
