#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	vector<long> v;
    	for(int i=0;i<n;i++)
    	{
    		long val;
    		cin>>val;
    		v.push_back(val);
    	}
    	stable_sort(v.begin(),v.end(),greater<int>());
    	for(int i=0;i<v.size();i++)
    	{
    		cout<<v[i]<<" ";
    	}
    	cout<<endl;
    }
    return 0;
}
