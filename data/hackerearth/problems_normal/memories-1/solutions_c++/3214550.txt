#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef pair<int,int> ii;
typedef vector<int> vi;
typedef vector<ii> vii;

const int  MOD=1000000007;
const int  INF= int(1e9);

int main()
{
	ios_base::sync_with_stdio(false);
    int testCases;
    cin>>testCases;
    while(testCases--) {
        int n,p;
        cin>>n>>p;
        vii notAllowed(p);
        for(int i=0;i<p;i++) {
            cin>>notAllowed[i].first>>notAllowed[i].second;
            notAllowed[i].first--;
            notAllowed[i].second--;
        }
        int res=0;
        for(int mask=0;mask<(1<<n);mask++) {
            bool ok=true;
            for(ii x : notAllowed) {
                if((mask & (1<<x.first)) && (mask & (1<<x.second))) {
                    ok=false;
                    break;
                }
            }
            if(ok) {
                res++;
            }
        }
        cout<<res-1<<"\n";
    }

	return 0;

}
