#include<iostream>
#include<cstdio>
#include<algorithm>

using namespace std;
typedef long int  li;

int power_of_twos(long int num){
    int counter = 0;
    while(num>0){
        num=num>>1;
        counter++;
    }

    return counter;
}


long int get_position_of_survivor(long int n){
    if (n&(n-1)==0) return 1;
    
    long int m = 1<<(power_of_twos(n)-1);
    return (n-m)*2 + 1;
}


int main(){
    int N;
    long int test_case;
    cin>>N;
    while(N--){
        cin>>test_case;
        cout<<get_position_of_survivor(test_case)<<endl;
    }
    
}
