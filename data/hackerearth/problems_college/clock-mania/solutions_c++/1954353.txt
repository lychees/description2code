#include<iostream>
#include<math.h>
#include <stdlib.h>
using namespace std;
int min(int a, int b){
	return (a<b)?a:b;
}

int angle(int a,int b){
	int h = 0.5*(60*a + b);
	int m = 6*b;
	int result = h-m;
	if(result < 0)
		result = -1*result;
	result = min(result,360-result);
	return result;
}
int main(){
	int t;
	cin >> t;
	while(t--){
		int hour,minute;
		cin >>hour >> minute;
		int f = angle(hour,minute);
		cout << f << endl;
	}
}
