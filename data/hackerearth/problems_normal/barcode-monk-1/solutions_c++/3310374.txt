#include <bits/stdc++.h>
using namespace std;
long long mod = 1000000007;
bool superbrush[1001] = {false};
int main()
{
    long long n,k;
    cin>>n>>k;

    long long u[1001] = {0};
    long long p[1001] = {0};
    long long a[n];
    superbrush[0] = 1;
    for(long long i = 0;i< k;i++)
    {
     cin>>a[i];

     for(long long j = a[i];j<= n;j++)
     {
         superbrush[j]=superbrush[j]|superbrush[j-a[i]];
     }
    }
    u[0] = 1;
    p[0] = 0;
    for(long long i = 1;i<= n;i++)
    {
        u[i] = u[i-1] + p[i-1];
        u[i] %=mod;
        for(long long j = 1;j<= i;j++)
        {
            if(superbrush[j])
            {
                p[i]+=u[i-j];
                p[i] %=mod;
            }
        }
    }
    cout<<(p[n]+u[n])%mod<<endl;

}
