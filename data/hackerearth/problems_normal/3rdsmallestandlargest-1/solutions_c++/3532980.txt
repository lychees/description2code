#include <bits/stdc++.h>
using namespace std;
char a[100005];
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%s",&a);
    	int n = 0;
    	for(n=0;a[n];n++);
    	sort(a,a+n);
    //	printf("len %d\n",n);
    	
    	int st = 0;
    	for(st=0;st<2;st++)
    	{
    		if(!next_permutation(a,a+n))break;
    		//printf("%s ",a);
    	}
    	if(st!=2)
    	{
    		printf("Not possible!\n");
    	}
    	else {
    		printf("%s ",a);
    		sort(a,a+n);
    		reverse(a,a+n);
    		for(st=0;st<2;st++)
    		{
    			
    			if(!prev_permutation(a,a+n))break;
    		}
    		printf("%s\n",a);
    	}
    }
    return 0;
}
