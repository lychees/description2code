x1,y1=map(float,raw_input().split(" "))
x2,y2=map(float,raw_input().split(" "))
x3,y3=map(float,raw_input().split(" "))

a1 = x1 - x2 + x3
a2 = 2*x1 - a1
a3 = 2*x2 - a2
 
b1 = y1 - y2 + y3
b2 = 2*y1 - b1
b3 = 2*y2 - b2

sides=[[a1,b1],[a2,b2],[a3,b3]]
for side in sorted(sides):
	print "%.4f %.4f"% (side[0],side[1])
