#include <iostream>
using namespace std;

int main(){
    int t, n, x, y;
    
    scanf("%d", &t);
    while (t--){
    	scanf("%d", &n);
    	for (x = 0; x <= n; x += 5)
    		if ((n - x) % 3 == 0)
    			break;
    	if (x > n)
    		printf("-1\n");
    	else{
    		for (int i = 0; i < n - x; i++)
    			printf("5");
    		for (int i = 0; i < x; i++)
    			printf("3");
    		printf("\n");
    	}
    }
    
    return 0;
}
