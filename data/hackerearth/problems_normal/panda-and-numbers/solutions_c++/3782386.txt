#include <bits/stdc++.h>
using namespace std;
 int dp[1000004];
int has[1000006];
int n,m;
#define ll long long int;
int power(int x)
{
	int ans=1;
	int a=x,b=x;
	while(b)
	{
		if(b&1)
		ans*=a;
		b/=2;
		a*=a;
	}
	return ans;
}
int fun(int x)
{   
	if(x<=0)
	return 0;
	
	int &ret=has[x];
	if(ret!=-1)
	return ret;
	
	ret=0;
    int temp=x;
    
    while(temp)
    {
    	int a=temp%10;
    	temp/=10;
    	if(a!=0)
    	ret=ret | fun(x-a*a);
    	if(ret)
    	break;
    
    }
    
    return ret;
	
}
int main()
{
    int i,j,t;
   i=1;
   memset(has,-1,sizeof(has));
   while(1)
   {
   	if(power(i) >1000000)
   	break;
   	has[power(i)]=1;
   	i++;
   }
  
   for(i=1000000;i>=1;i--)
   if(has[i]==-1)
   has[i]=fun(i);

   cin>>t;
   while(t--)
   {
   cin>>n;
   if(has[n])
   cout<<"Yes\n";
   else
   cout<<"No\n";
    }
    
    
}
