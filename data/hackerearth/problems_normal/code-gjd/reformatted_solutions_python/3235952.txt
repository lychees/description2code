from pprint import pprint as pp


def GI():
    return int(raw_input())


def GIS():
    return map(int, raw_input().split())


def main():
    for t in xrange(GI()):
        GI()
        nums = GIS()
        nums.sort(reverse=True)
        print sum(reduce((lambda a, b: a.append(a[-1] + b) or a), nums, [0])) - nums[0]


main()
