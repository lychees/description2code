#include<iostream>
#include<cstring>
using namespace std;
int main(){
	int t,c1,c2,i;
	cin>>t;
	string a,b;
	int c[26],d[26];
	while(t--){
	memset(c,0,sizeof(c));
	memset(d,0,sizeof(d));
	cin.ignore();
	getline(cin,a);
	for(i=0;i<a.length();i++){
		if(a[i]!=32)
		++c[(int)a[i]-97];	
	}
	getline(cin,b);
	for(i=0;i<b.length();i++){
		if(b[i]!=32){
			if(c[(int)b[i]-97]>0)
			--c[(int)b[i]-97];
			else
			++d[(int)b[i]-97];
		}
	}
	c1=0,c2=0;
	for(i=0;i<26;i++){
		if(c[i]>0)
		c1=10;
		if(d[i]>0)
		c2=10;
	}
	if(c1==10&&c2==10)
	cout<<"You draw some.";
	else if(c1==10)
	cout<<"You win some.";
	else
	cout<<"You lose some.";
	cout<<endl;
	}
	return 0;
}
