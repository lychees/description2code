#include <bits/stdc++.h>
using namespace std;
 
int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(NULL);
    
    //Stacky my;
    int t;
    cin>>t;
    
    while(t--){
    	//int ref=my.tos;
    	stack<int> my;
    	int n,id,x;
    	char temp;
    	cin>>n>>id;
    	
    	my.push(id);
    	
    	for(int i=0;i<n;i++)
    	{
    		cin>>temp;
    		
    		if(temp=='P')
    		{
    			cin>>x;
    			my.push(x);
    			//ref=tos;
    		}
    		else
    		{
    			int x=my.top();
    			my.pop();
    			int y=my.top();
    			my.push(x);
    			my.push(y);
    		}
    	}
    	cout<<"Player "<<my.top()<<endl;
    }
    return 0;
}
