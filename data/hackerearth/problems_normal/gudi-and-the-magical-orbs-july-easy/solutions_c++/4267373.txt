#include<bits/stdc++.h>
using namespace std;
int arr[105][105];
bool dp[105][105][505];
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int k,n,m;
		scanf("%d%d%d",&n,&m,&k);
		memset(arr,0,sizeof(arr));
		memset(dp,false,sizeof(dp));
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<m;j++)
			{
				scanf("%d",&arr[i][j]);
				//dp[i][j]=arr[i][j];
			}
		}
		dp[0][0][arr[0][0]]=true;
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<m;j++)
			{
				for(int p=arr[i][j];p<=k;p++)
				{
					if(i==0&&j==0)
					{
						continue;
					}
					else if(i==0)
					{
						dp[i][j][p]=(dp[i][j][p]||dp[i][j-1][p-arr[i][j]]);
						//dp[i][j][p]=dp[i][j][p]||dp[i][j][p-arr[i][j]];
					}
					else if(j==0)
					{
						//dp[i][j][p]=dp[i][j][p]||dp[i-1][j][p-arr[i][j]];
						dp[i][j][p]=(dp[i][j][p]||dp[i-1][j][p-arr[i][j]]);
					}
					else
					{
						dp[i][j][p]=(dp[i][j][p]||dp[i-1][j][p-arr[i][j]]);
						dp[i][j][p]=(dp[i][j][p]||dp[i][j-1][p-arr[i][j]]);
						dp[i][j][p]=(dp[i][j][p]||dp[i-1][j-1][p-arr[i][j]]);
					}
	
				}
			}
		}
		int p;
		for( p=k;p>=0;p--)
		{
			if(dp[n-1][m-1][p])
			{
				cout<<p<<endl;
				break;
			}
		}
		if(p==-1)
		printf("-1\n");
	}
	return 0;
}
