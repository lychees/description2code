using namespace std;
ll mark[10000007],dp[10000007];
ll power(ll a)
{
    ll res=pow(a,a);
    return res;
}
ll solve(ll x)
{
    if(x<1)
    return 0;
    if(mark[x])
    return 1;
    if(dp[x]!=-1)
    return dp[x];
    ll temp=x,r,res=0;
    while(temp!=0)
    {
        r=temp%10;
        if(r>0)
        res=solve(x-(r*r));
        if(res==1)
        {
        	dp[x]=1;
        return 1;
        }
        temp=temp/10;
    }
    dp[x]=0;
    return 0;
}
int main()
{
    ll t,i,x,res;
    memset(mark,0,sizeof(mark));
    memset(dp,-1,sizeof(dp));
    for(i=1;i<=5;i++)
    mark[power(i)]=1;
    scanf("%lld",&t);
    while(t--)
    {
    	
        scanf("%lld",&x);
        res=solve(x);
        if(res==1)
        printf("Yes\n");
        else
        printf("No\n");
    }
	return 0;
}
