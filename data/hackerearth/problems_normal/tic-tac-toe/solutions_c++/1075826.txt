#include <iostream>
using namespace std;
inline void scan(int &a){
	char ch=a=0;
	while(ch<33)	ch=getc(stdin);
	while(ch>33)	a=a*10+ch-'0',ch=getc(stdin);
}
int main()
{
	int t,n,inv=1166666675,mod=1000000007;
	long long temp,min,max;
	scan(t);
	while(t--){
		scan(n);
		temp=n-1;
		max=temp*(temp+1)%mod;
		max=max*(temp*2+1)%mod;
		max=max*inv%mod;
		temp=n/2;
		min=(temp*temp)%mod;
		min=min*n%mod;
		printf("%lld %lld\n",min,max);
	}
    return 0;
}
