#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	int c=1;
	while(t--)
	{
		int arr[3];
		for(int i=0;i<3;i++)
            cin>>arr[i];
        char ans[4];
        cout<<"Case "<<c<<": ";
        c++;
		for(int i=0;i<3;i++)
        {
            int temp=arr[i];
            if(temp==0)
            {
            	cout<<"00 ";
            }
            else
            {
            int j=0;
            while(temp!=0)
            {
                int m=temp%16;
                if(m<10)
                 ans[j++]=m+48;
                else
                 ans[j++]=m+75;
                temp/=16;
            }
			if(j==1)
			cout<<"0";
            for(int i=j-1;i>=0;i--)
                cout<<ans[i];
                cout<<" ";
        	}
        }
        cout<<endl;
	}

    return 0;
}
