T = int(input())
for t in range(T):
    A1 = raw_input()
    A2 = raw_input()
    l1 = len(A1)
    l2 = len(A2)
    maximum = 0
    for offset in range(l1-l2+1):
        # print "offset =", offset
        counter = 0
        for i in range(l2):
            # print "i =", i
            if A1[i+offset] == A2[i]:
                # print "comparing", A1[i+offset], A2[i]
                counter += 1
        if maximum < counter:
            maximum = counter
    print l2-maximum
