import numpy

N, Q = map(int, raw_input().split())
probTime = map(int, raw_input().split())
probScore = map(int, raw_input().split())
probs = zip(probTime, probScore)
maxScores = [i[0] for i in sorted(probs, key=lambda x:x[1], reverse=True)]
maxTimesSum = numpy.cumsum(maxScores)
for i in range(Q):
	q = input()
	print maxTimesSum[q-1]
