#include <iostream>
#include <cstdio>
using namespace std;
 
int main()
{
    int t;
    
    cin>>t;
    cin.ignore();
    while(t--) {
        string a,b;
    	int hash[26]={0},i,pos=0,neg=0;
        
    	getline(cin,a);
    	getline(cin,b);
        //cout<<"a:"<<a<<" "<<"b:"<<b<<endl;
    	for(i=0;i<a.size();i++) {
    		if(a[i]==' ')
    			continue;
    			
    		hash[a[i]-'a']++;
    	}
    	
    	for(i=0;i<b.size();i++) {
    		if(b[i]==' ')
    			continue;
    			
    		hash[b[i]-'a']--;
    	}
    /*    
        for(i=0;i<26;i++) {
        	cout<<"i:"<<i<<" "<<hash[i]<<endl;
    	}	
    */
    	for(i=0;i<26;i++) {
    		if(hash[i]>0) 
    			pos=1;
    		else if(hash[i]<0) 
    				neg=1;	
    	}	
    	
        //cout<<"pos:"<<pos<<" neg:"<<neg;
    	if((pos && neg)||(!pos && !neg))
    		printf("You draw some.\n");
    	else if(pos)
    		printf("You win some.\n");
    	else if(neg)
    		printf("You lose some.\n");
    }
    return 0;
}
