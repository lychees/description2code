#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
long long dp[500005][12], A[500005];

long long power(long long a, int n){
	
	long long ans=1;
	while(n){
		if(n%2){
			ans=ans*a%mod;
		}
		n/=2;
		a=a*a%mod;
	}
	return ans;
}

int main()
{
    int n,k,m;
    cin>>n>>k>>m;
    for(int i=1; i<=n; i++)
    {
    	int x;
    	cin>>x;
    	A[x]++;
    }
    
    dp[0][0]=1;
    for(int i=1; i<=k; i++)
    	dp[i][0]=1;
    for(int i=1; i<=k; i++){
    	for(int j=1; j<=m; j++){
    		long long z=power(2,A[i])-1;
    		if(z<0)
    			z+=mod;
    		dp[i][j]=dp[i-1][j-1]*(z)%mod;
    		dp[i][j]+=dp[i-1][j];
    		dp[i][j]%=mod;
    	}
    }
    long long ans=1;
    for(int i=1; i<=m; i++){
    	ans=ans+dp[k][i];
    	ans%=mod;
    }
    //ans++;
    //ans%=mod;
    cout<<ans<<endl;
    return 0;
}
