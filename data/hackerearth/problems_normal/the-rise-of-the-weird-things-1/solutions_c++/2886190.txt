#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int n,i,j;
    int sum_even=0,sum_odd=0;
    cin>>n;
    int a[n];
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    sort(a,a+n);
    int b[n+2];
    j=0;
    for(i=0;i<n;i++)
    {
    	if(a[i]%2 == 0)
    	{
    		b[j]=a[i];
    		sum_even += b[j];
    		j++;
    	}
    }
    b[j]=sum_even;
    j=j+1;
    for(i=0;i<n;i++)
    {
    	if(a[i]%2 != 0)
    	{
    		b[j]=a[i];
    		sum_odd += b[j];
    		j++;
    	}
    }
    b[j]=sum_odd;
    for(i=0;i<(n+2);i++)
    cout<<b[i]<<" ";
    
    return 0;
}
