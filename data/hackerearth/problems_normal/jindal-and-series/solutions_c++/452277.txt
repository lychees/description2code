#include <iostream>
using namespace std;

int main()
{
    long long i,t,n,j;
    long long arr[1000000+1]={0};
   
   	arr[1]=1;arr[2]=1;arr[3]=2;arr[4]=4;
    	
    for(j=5;j<=1000000;j++)
    arr[j]=(arr[j-1]+arr[j-3]+arr[j-4])%10000007;
   
    cin>>t;
   
    for(i=0;i<t;i++)
    {
    	cin>>n;
    	cout<<arr[n]<<"\n";
    }
    
    return 0;
}
