#include <iostream>
using namespace std;
bool func(int x,int y)
{
	if(x==y||x-2==y)
	return true;
	else return false;
}
int main()
{
   int t,x,y;
   cin>>t;
   while(t--)
   {
   	cin>>x>>y;
   	if(func(x,y))
   	{
   		if(x%2==0)
   		cout<<x+y<<endl;
   		else
   		cout<<x+y-1<<endl;
   	}
   	else
   	cout<<"No Number"<<endl;
   }
}
