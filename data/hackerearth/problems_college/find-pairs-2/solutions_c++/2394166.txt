    int main()
    {
        int t;
        cin>>t;
        for(int k=0;k<t;k++)
        {
            unsigned long int n;
            cin>>n;
             long int a[1000002];
            for(int i=0;i<n;i++)cin>>a[i];
            sort(a,a+n);int c=1,sum=0;

                if(a[i]==a[i+1])c++;

            }

        }
