#include<bits/stdc++.h>

using namespace std;

int main()
	{
		int T;
		cin>>T;
		while(T--)
			{
				long long int N,K;
				scanf("%lld %lld",&N,&K);
				long long int x;
				long long v[N+1];
				for(int i=0;i<N;i++)
					{
						scanf("%lld",&v[i]);
					}
				sort(v,v+N);
				long long int sum=0;
				for(int i=N-1;i>=0 && K>0;--i)
					{
						sum=sum+v[i];
						K--;
					}
				 printf("%lld\n",sum);
			}
		return 0;
	}
