#include<bits/stdc++.h>

using namespace std;

int main()
{
	long long int t,n,sum=0;
	cin>>t;
	for(int i=0; i<t ; i++)
	{
		cin>>n;
		sum=0;
		for(int j=0, k=1; (1LL<<j)+(1LL<<k)<=n ;j++)
		{
			for(k=j+1; (1LL<<j)+(1LL<<k)<=n ;k++)
			{
				sum+=(1LL<<j)+(1LL<<k);
			}
			k=j+1;
			sum%=1000000007;
		}
		cout<<sum<<endl;

	}	
	return 0;
}
