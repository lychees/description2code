#include<cstdio>
#include<cstring>
#define MAXN 100000000

double ans[10000+11];
bool flag[MAXN+11];

void init(){
    memset(flag,true,sizeof(flag));
    flag[1]=false,flag[0]=false;
    for(int i=4;i<=MAXN;i+=2) flag[i]=false;
    for(int i=3;i<=MAXN;i+=2){
        if(flag[i]){
            for(int j=i+i;j<=MAXN;j+=i){
                flag[j]=false;
            }
        }
    }
}
void solve(){
     ans[1]=0;
     int cnt=0;
     double n=1;
     for(int i=3;i<=10000;i+=2){
        int m=i-2;
        m*=m;
        for(int j=1;j<=4;j++){
            int tmp=m+(i-1)*j;
            if(flag[tmp])
                cnt++;
        }
        n+=4;
        ans[i]=cnt/n;
     }
     return ;
}
int main(){
    init();
    int cs;
    solve();
    scanf("%d",&cs);
    while(cs--){
        int n;
        scanf("%d",&n);
        printf("%.6lf\n",ans[n]*100.0);
    }
    return 0;
}
