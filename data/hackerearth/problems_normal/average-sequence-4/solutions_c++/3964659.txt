#include<iostream>
using namespace std;
int main(){
    int n;
    cin>>n;
    long long int a[n],b[n];
    for(int i=0;i<n;i++){
        cin>>b[i];

    }
    long long temp=0;
    a[0]=b[0];
    for(int i=1;i<n;i++){
        temp=0;
        for(int j=0;j<i;j++){
            temp=temp+a[j];
        }
        a[i]=(b[i]*(i+1))-temp;
    }
    for(int i=0;i<n;i++)
        cout<<a[i]<<" ";

return 0;
}
