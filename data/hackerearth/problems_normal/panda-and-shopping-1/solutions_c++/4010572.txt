int n;
vector<ll> v;
vector<ll> h;
int main()
{
	inp (n);
	v.resize (n);
	h.resize (n);
	set<pair<ll, ll> > s;
	s.insert (MP (0, 0));
	set<pair<ll, ll> >::iterator it;
	rep (i, n) {
		inll2 (h[i], v[i]);
		if (h[i]<= 0)
			continue;
		it = s.upper_bound (MP(v[i], (ll)1e18));
		if (it == s.begin ()) {
			ll sum = h[i];
			vector<pair<ll, ll> > e;
			while (it != s.end () && it->Y <=sum) {
				e.pb (*it);
				it++;
			}
			rep (i, e.size ())
				s.erase (e[i]);
			s.insert (MP (v[i], h[i]));
		}
		else {
			it--;
			ll sum = h[i] + it->Y;
			it++;
			vector<pair<ll, ll> > e;
			while (it != s.end () && it->Y <=sum) {
				e.pb (*it);
				it++;
			}
			rep (i, e.size ())
				s.erase (e[i]);
			s.insert (MP (v[i], sum));
		}
	}
	outll (max ((ll)0, s.rbegin ()->Y));
    return 0;
}
