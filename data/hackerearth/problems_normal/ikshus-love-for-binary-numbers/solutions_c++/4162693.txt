#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

using ull = unsigned long long;

ull gcd(ull a, ull b) {
	while (b > 0) {
		ull m = a % b;
		a = b;
		b = m;
	}
	return a;
}

void cancel(ull& a, ull& b) {
	ull d = gcd(a, b);
	a /= d;
	b /= d;
}

ull count(int n, int k) {
	vector<ull> a(k + 1,0);
	a[0] = 1;
	for (int i; i < n; ++i) {
		ull sum = accumulate(a.begin(), a.end(), 0ull);
		for (int j = k; j > 0; --j)
			a[j] = a[j - 1];
		a[0] = sum;
	}
	return accumulate(a.begin(), a.end(), 0ull);
}

int main()
{
	int n, k;
	cin >> n >> k;
	ull a = count(n, k - 1);
	ull b = (1ull << n);
	cancel(a, b);
	cout << b - a << '/' << b << '\n';
    return 0;
}
