#include<bits/stdc++.h>
using namespace std;
int main()
{
int t;
cin>>t;
while(t--)
{
 int a[1000000];
 int n;
 cin>>n;
 unsigned long long int x=1;
 a[0]=n;
 for(int i=0;;i++)
 {
    if(n%2==1)
    n=3*n+1;
    else
    n=n/2;
    a[x]=n;
    x++;
    if(n==1)
    break;
 }
 unsigned long long es=0,os=0;
for(int i=0;i<x;i++)
                {
                                if(a[i]%2==0)
                                                es=es+a[i];
                                else
                                                os=os+a[i];
                }
                x=x-1;
                if(es%x>os%x)
                {
                                cout<<"Even Rules"<<endl;
                }
                else if(es%x<os%x)
                                cout<<"Odd Rules"<<endl;
                else
                                cout<<"Tie"<<endl;
}
return 0;
}
