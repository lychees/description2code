def shortest_way(a, b):
    result = 0
    while a > 1 or b > 1:
        if a > b:
            k = a/b
            a = a-k*b
            result = result + k
        else:
            k = b/a
            b = b-k*a
            result = result + k
    if result > 0:
        result -= 1
    return result
