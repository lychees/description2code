#include<bits/stdc++.h>
using namespace std;
string s;
long long dp[25][5],v[25][5];
long long rec(long long n,int ct)
{
    //cout<<n<<" "<<ct<<endl;
    if(n<ct-1)
        return dp[n][ct];
    if(v[n][ct]!=0)
        return dp[n][ct];
    if(ct==1&&n>=0)
    {
       // cout<<"fuck\n";
        long long ans=0;
        if(n==0)
        {
            v[n][ct]=1;
            dp[n][ct]=s[0]-48;
            //cout<<dp[n][ct]<<endl;
            return dp[n][ct];
        }
        if(n>=1&&(s[0]-48))
        {
            for(int i=0;i<=n;i++)
            {
                ans=(ans*10+s[i]-48);
                if(ans>1000000000000)
                    break;
            }
            if(ans<=1000000000000)
            dp[n][ct]=ans;//,cout<<ans<<endl;
            v[n][ct]=1;
            return dp[n][ct];
        }
        else
        {
            v[n][ct]=1;
            return dp[n][ct];
        }
    }
    for(int i=1;i<=n;i++)
    {
        long long ans=0;
        for(int j=n-i+1;j<=n;j++)
        {
            ans=(ans*10+s[j]-48);
            if(ans>1000000000000)
                    break;
        }
        if(s[n-i+1]=='0'&&n-i+2<=n)
            continue;
        if(ans<=1000000000000)
        dp[n][ct]=max(dp[n][ct],(dp[n-i][ct-1]=rec(n-i,ct-1))+ans),v[n-i][ct-1]=1;
    }
    return dp[n][ct];
}
