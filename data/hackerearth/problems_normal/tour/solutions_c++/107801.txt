#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int main()
{
	int n,i,j,p,q,m;
	long long x=0;
	scanf("%d\n",&n);
	char a[n][1000];
	for(i=0;i<n;i++)
	scanf("%s",a[i]);
	int c[n][n];
	for(i=0;i<n;i++)
	for(j=0;j<n;j++)
	scanf("%d",&c[i][j]);
	scanf("%d\n",&m);
	char b[m][1000];
	for(i=0;i<m;i++)
	scanf("%s",b[i]);
	for(i=0;i<m-1;i++)
	{	
		char y[1000]={0},z[1000]={0};
		//printf("%d C\n",i);	
		for(j=0;b[i][j];j++)
		y[j]=b[i][j];
		i++;
		for(j=0;b[i][j];j++)
		z[j]=b[i][j];
		i--;		
		for(j=0;j<n;j++)
		{
			if(strcmp(y,a[j])==0)
			{
				p=j;
				//printf("%d B\n",j);
				break;
			}
		}
		//printf("D %s %s\n",y,z);
		for(j=0;j<n;j++)
		{
			if(strcmp(z,a[j])==0)
			{
				q=j;
				//printf("%d A\n",j);
				break;
			}
		}
		x=x+(long long)c[p][q];
	}
	char y[1000]={0};	
	for(j=0;b[0][j];j++)
	y[j]=b[0][j];
	for(j=0;j<n;j++)
	{
		if(strcmp(y,a[j])==0)
		{
			p=j;
			//printf("%d B\n",j);
			break;
		}
	}
	x=x+(long long)c[0][p];
	cout<<x<<endl;
	return 0;
}
