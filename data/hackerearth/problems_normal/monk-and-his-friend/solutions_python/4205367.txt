
def ToBinary(decimal):
    intlist=list()
    while decimal>1:
        rem=decimal%2
        decimal=decimal/2
        intlist.append(rem)
    intlist.append(decimal)
    return sum(intlist)

T=int(input())
countlist=list()
for i in range(T):
    the_string=raw_input()
    P,M=the_string.split()
    count=ToBinary((int(P))^(int(M)))
    countlist.append(count)
