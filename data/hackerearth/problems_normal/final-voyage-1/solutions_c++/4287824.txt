#include<bits/stdc++.h>
using namespace std;
#define ll long long int
ll wt[100010],value[100100];
ll dp[2][1000100];
int main()
{
	ll t;
	scanf("%lld",&t);
	while(t--)
	{
		ll n;
		scanf("%lld",&n);
		ll weight;
		scanf("%lld",&weight);
		ll i,j;
		memset(dp,0,sizeof(dp));
		for(i=0; i<n; i++)
		{
			scanf("%lld",&wt[i]);
		}
		for(i=0; i<n; i++)
		{
			scanf("%lld",&value[i]);
		}
		for(i=0; i<=n; i++)
		{
			for(j=0; j<=weight; j++)
			{
				if( i==0 || j==0)
				{
					dp[i%2][j]=0;
					continue;
				}
				if(wt[i-1]<=j)
				{
					dp[i%2][j]=max(dp[(i-1)%2][j-wt[i-1]]+value[i-1],dp[(i-1)%2][j]);
				}
				else
				{
					dp[i%2][j]=dp[(i-1)%2][j];
				}
			}
		}
		printf("%lld\n",dp[n%2][weight]);
	}
	return 0;
}
