invalids = [1, 4, 6, 8, 9]
for _ in range(input()):
    nod, ins = map(int, raw_input().split())
    ans = 0
    while ins > 0:
        mods = ins % 10
        if mods in invalids:
            if mods == 9:
                ans += 2
            else:
                ans += 1
        ins /= 10
    print ans
