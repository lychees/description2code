#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    long long int t,d,p,r,i;
    double s;
    cin>>t;
    while(t--)
    {
    	cin>>d>>p;
    	r=d*d+4*p;
    	s=sqrt(r);
    	
    	i=int(s);
    	
    	if(i!=s)
    	{
    		cout<<0<<endl;
    		continue;
    	}
    	if(d==0)
    	{
    		cout<<2<<endl;
    		
    	}
    	
    	
    	else
    	{
    		if(r==0)
    			cout<<2<<endl;
    		else
    		{
    			if(d%2==1 && i%2==1 || d%2==0 && i%2==0)
    				cout<<4<<endl;
    			else
    				cout<<0<<endl;
    		}
    			
    	}
    	
    }
    return 0;
}
