#include<iostream>
#include<stdlib.h>
#include<vector>
using namespace std;
#define Mod 1000000007
vector<int>siev;
void sieve()
{
    bool A[2000000]={0};
    for (int i=2 ; i<2000000 ; i++)
    {
        if (A[i]==0)
        {
            int j=2;
            while(j*i<=2000000)
            {
                A[j*i]=1;
                j++;
            }
        }
    }
    for (int k=2 ; k<2000000 ; k++)
    {
        //cout<<k<<" "<<A[k]<<endl;
        if (A[k]==0)
            siev.push_back(k);
    }
}
int main()
{
    int T;
    sieve();
    //cout<<siev.size();;
    cin>>T;
    long long *A;
    A=(long long *)malloc(sizeof(long long)*100000);
    A[0]=1;
    for (int i=1 ; i<=100000 ; i++)
    {
        A[i]=(siev[i-1]*A[i-1])%Mod;
        //cout<<A[i]<<endl;

    }
    for(int i=0 ; i<T ; i++)
    {
        long long N;
        cin>>N;
        cout<<A[N]<<endl;
    }
}
