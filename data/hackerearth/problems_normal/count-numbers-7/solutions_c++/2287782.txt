long long safe(long long a,long long b)
{
	if (b==0||a==0)
		return 0;
	if (1e18/b<=a)
		return 1e18;
	return a*b;
}

long long solve(long long x)
{
	long long res=0;
	for (int mask=0;mask<(1<<n);mask++)
	{
		long long prod=1;
		int sig=1;
		for (int i=0;i<n;i++)
		{
			if (mask&(1<<i))
			{
				prod=safe(prod,p[i]);
				sig*=-1;
			}
		}
		res+=x/prod*sig;
	}
	return res;
}
