#include <iostream>
using namespace std;
int main()
{
    int t;
    cin >> t;
    while(t--){
        int table[256] = {0};
        int n,i;
    	cin >> n;
    	int flag = 1;
        string x;
    	for(i=0;i<n;i++){
	    	cin >> x;
	    	if(table[x[0]] == 1 ) {
	    		flag = 0;
	    	}
	    	else table[x[0]] = 1;
	    }
	    if(flag ) cout << "YES\n";
	    else cout << "NO\n";
    }
    return 0;
}
