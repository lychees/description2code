#include <bits/stdc++.h>
using namespace std;
int main() {
	ios_base::sync_with_stdio(false);
	long t;
	int arr[101][101];
	long total=0;
	cin>>t;
	for(int i=0;i<t;i++)
	{
		for(int j=0;j<t;j++)
		{
			cin>>arr[i][j];
		}
	}
	for(int i=0,j=0;i<t;i++,j++)
		total+=arr[i][j];
	for(int i=0,j=t-1;i<t;i++,j--)
			total-=arr[i][j];
	cout<<abs(total)<<endl;
	return 0;
}
