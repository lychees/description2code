#include <stdio.h>
#include <cstdlib>

int main ()
{
	int t;
	scanf ("%d",&t);
	while (t--)
	{
		int n;
		int max = 0;
		int p,q;
		int a[500005];
		scanf ("%d",&n);
		int i;
		for (i = 0 ; i < n ; i++)
		{
			scanf ("%d",&a[i]);
		}
		int j;
		for (i = 0 ; i < n-max ; i++)
		{
			int count = 0;
			j = i+1;
			if (abs (a[i] - a[j]) <= 1)
			{
			count = 1;
			p = a[i];
			q = a[j];
			while (j < n)
			{
				if (abs(a[j] - p) <= 1 && abs(a[j] - q)<=1)
					count++;
				else
					break;
				j++;
			}
			if (count >= max)
				max = count;
			}
		}
		printf ("%d\n",max);
	}
	return 0;
}
