#include <iostream>
using namespace std;

int main()
{
    string s;
    cin>>s;
    int k = s.length();
    int i=0;
    int count=0;
    for(i=0;i<(k/2);i++)
    {
    	if(s[i]!=s[k-i-1])
    	{
    		count ++;
    		break;
    	}
    }
    if(count==0)
    cout<<"YES";
    else
    cout<<"NO";
    return 0;
}
