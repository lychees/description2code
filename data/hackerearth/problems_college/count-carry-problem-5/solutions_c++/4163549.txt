#include <iostream>
using namespace std;

int main()
{
    unsigned long t,a,b,c;
    cin>>t;
    while(t--){
    	cin>>a>>b;c=0;
    	while(a>0&&b>0){
    		if((a%10)+(b%10)>=10)
    		c++;
    		a/=10;b/=10;
    	}
    	if(c==0)
    	cout<<"No carry operation\n";
    	else{
    	if(c==1)
    	cout<<c<<" carry operation\n";
    	else
    	cout<<c<<" carry operations\n";
    	}
    }
    return 0;
}
