#include <iostream>
using namespace std;

int main()
{
    long long c[26],n,p,total,sum,complete,remain,rsum;
    long t;
    cin >> t;
    while(t--){
    	sum = 0,rsum=0;
    	cin >> n;
    	for(int i=0; i<26; i++){
    		cin >> c[i];
    		sum+=c[i];
    	}
    	p = 12*n;
    	complete = p/26;
    	remain = p%26;
    	total=complete*sum;
    	if(complete>1){
    		total+= 13*(complete-1)*(complete);
    	}
    	for(int i=0; i<remain; i++)
    		rsum+=c[i];
    	total+=rsum;
    	total+= remain*complete;
    	cout << total << endl;
    }
}
