t = int(raw_input())


while t:
	t-=1
	N, M = map(int, raw_input().strip().split(" "))

	N_L = []
	M_L = []

	L = map(int, raw_input().strip().split(" "))
	N_L = L[:N]
	M_L = L[N:]
	
	
	N_L = set(N_L)
	
	for item in M_L:
		if item in N_L:
			print "YES"
		else:
			print "NO"
		N_L.add(item)
