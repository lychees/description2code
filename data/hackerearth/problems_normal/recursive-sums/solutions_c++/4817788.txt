#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;

unsigned long long int reduce(unsigned long long int num)
{
	vector<int> arr;
	unsigned long long sum = 0;
	while (num > 0)
	{
		arr.push_back(num % 10);
		num = num / 10;
	}
	while (arr.size() > 1)
	{
		sum = 0;
		for (vector<int>::iterator it = arr.begin(); it != arr.end(); ++it)
			sum = sum + (*it);
		arr.clear();
		if (sum / 10 < 1)
			arr.push_back(sum);
		else
		{
			while (sum > 0)
			{
				arr.push_back(sum % 10);
				sum = sum / 10;
			}
		}
	}
	return arr[0];
}
