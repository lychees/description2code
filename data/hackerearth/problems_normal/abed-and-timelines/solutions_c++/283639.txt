#include <iostream>
#include<math.h>
using namespace std;

int main()
{
  int t;
  cin>>t;
  while(t--){
  	long N,M,x1,y1,x2,y2,P;
  	cin>>N>>M;
  	cin>>x1>>y1>>x2>>y2;
  	if(x1>x2){
  		int temp;
  		temp=x1;
  		x1=x2;
  		x2=temp;
  	}
  	if(y1>y2){
  		int temp;
  		temp=y1;
  		y1=y2;
  		y2=temp;
  	}
  	cin>>P;
  	long minx=min((x2-x1), (N - (x2-x1)));
  	long miny=min((y2-y1), (M - (y2-y1)));
  	if((minx+miny)*P>1000){
  		cout<<(minx+miny)*P<<endl<<"Let go of the group."<<endl;
  	}
  	else{
  		cout<<(minx+miny)*P<<endl<<"You saved the group."<<endl;
  	}
  	
  }
  return 0;
}
