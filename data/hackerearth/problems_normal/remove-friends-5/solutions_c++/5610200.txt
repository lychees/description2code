#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
    cin>>t;
    while(t--){
    	int n,m;
    	cin>>n>>m;
    	int k;
    	
    	list<int> friends;
    	for(int i = 0;i < n;i++){
    		cin>>k;
    		friends.push_back(k);
    	}
    	
    	list<int> :: iterator it;
    	for(it = friends.begin();it!=friends.end();){
    		int temp = *(friends.begin());
    		if(m==0) break;
    		int first = *it;
    		it++;
    		int second = *it;
    		if(first<second){
    			it--;
    			int temp1 = *it;
    			friends.erase(it);
    			if(temp == temp1){
    				it = friends.begin();
    			}
    			else{
    				it--;
    			}
    			m--;
    		}
    	}
    	
    	if(k!=0){
    		for(int i=0;i<m;i++){
    			friends.pop_back();
    		}
    	}
    	
    	
    	for(it = friends.begin();it!=friends.end();it++){
    		cout<<*it<<" ";
    	}
    	cout<<endl;
    }
    
    return 0;
}
