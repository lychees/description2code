#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	long long n;
    	scanf("%lld",&n);
    	long long arr[n];
    	for(int i=0;i<n;i++)
    	scanf("%lld",&arr[i]);
    	sort(arr,arr+n);
    	reverse(arr,arr+n);
    	long long sum=0;
    	for(int i=0;i<n;i++)
    	{
    		if(i+1<n)
    		sum=sum+arr[i]-arr[i+1];
    	}
    	printf("%lld\n",sum+arr[0]-arr[n-1]);
    }
    return 0;
}
