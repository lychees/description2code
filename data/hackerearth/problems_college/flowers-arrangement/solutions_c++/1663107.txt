#include <iostream>
#include <string>
using namespace std;
int main(){
	ios::sync_with_stdio(false);
	int T=0;
	
	cin >> T;
	
	while(T--)
	{
	 string seq="";
	 string det="";
	 bool flag=true;
	 
	 cin >> seq;
	 
	 int len = seq.length();
	 det+="R";
	 
	 for(int i=0;i<len;i++)
	 {
	 	if(seq.substr(i,1) == det)
	 	{
	 	 if(seq[i+1] == 'R')
	 	  continue;
	 	 else
	 	 {
	 	 det+="Y";
	 	 if(seq.substr(i,2) == det)
	 	 {
	 	  det+="Y";
		  if(seq.substr(i,3) == det)
		  {
		   det="R";
		   i+=2;
		  }
		  else
		  {
		   det="R";
		   i+=1;
		  }
	 	 }
	 	}
	 	}
	   else
	   {
	    flag=false;
	    break;
	   }
	 }	
	
	cout << (flag ? "YES\n" : "NO\n");
	 }
	 
	 return 0;
}
