#include <iostream>
#define ll long long int
#define s(a) scanf("%lld",&a)
using namespace std;

int main()
{
    ll t,n,i,j,even,odd,x,ans;
    s(t);
    while(t--) {
    	even=odd=0;
    	s(n);
    	for(i=0;i<n;i++) {
    		s(x);
    		if(x%2==0) even++;
    		else odd++;
    	}
    	ans=odd*even;
    	cout<<ans<<endl;
    }
    return 0;
}
