#include<bits/stdc++.h>
using namespace std;
int arr[203];
bool check(int num){
    int cnt=0;
    while(num>0){
        if(num&1)
            cnt++;
        num>>=1;
    }
    if(cnt==1)return true;
    return false;
    //(cnt==1)?return true:return false;
}
int main(){
    int tc;cin>>tc;
    while(tc--){
        int n;cin>>n;
        for(int i=1;i<=n;++i){
            cin>>arr[i];
        }
        bool flag=false;
        for(int i=0;i<31;++i){
            int base=(1<<31)-1;
            for(int j=1;j<=n;++j){
                if(arr[j]&(1<<i))
                    base&=arr[j];
            }
            if(check(base)){
                flag=true;
                break;
            }
        }
        (flag)?puts("YES"):puts("NO");
    }
    return 0;
}
