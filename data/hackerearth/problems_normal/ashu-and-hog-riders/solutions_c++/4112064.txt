#include<bits/stdc++.h>
using namespace std;
typedef pair<int,int> ii;
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int n;
    cin>>n;
    int **a=new int*[n];
    int i,j;
    for(i=0;i<n;i++)
    {
        a[i]=new int[n];
    }
    for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++)
        {
            cin>>a[i][j];
        }
    }
    int max=INT_MIN,m=0;
    list<ii> q;
    for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++)
        {
            if(a[i][j]==1)
            {
                m=1;
                a[i][j]=0;
                q.push_back(ii(i,j));
                while(!q.empty())
                {
                    ii r=q.front();
                    q.pop_front();
                    int f=r.first,s=r.second;
                    if(f+1<n && a[f+1][s]==1)
                    {
                        m++;
                        a[f+1][s]=0;
                        q.push_back(ii(f+1,s));
                    }
                    if(s+1<n && a[f][s+1]==1)
                    {
                        m++;
                        a[f][s+1]=0;
                        q.push_back(ii(f,s+1));
                    }
                    if(f-1>=0 && a[f-1][s]==1)
                    {
                        m++;
                        a[f-1][s]=0;
                        q.push_back(ii(f-1,s));
                    }
                    if(s-1>=0 && a[f][s-1]==1)
                    {
                        m++;
                        a[f][s-1]=0;
                        q.push_back(ii(f,s-1));
                    }
                }
            }
            if(m>max)
            {
                max=m;
            }
        }
    }
    cout<<max;
}
