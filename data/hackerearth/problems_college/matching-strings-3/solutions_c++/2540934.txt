#include <iostream>
using namespace std;

int getMin(int x,int y){
	if(x>=y){
		return y;
	}else{
		return x;
	}
}

int main()
{
    int t,i;
    cin>>t;
    while(t--){
    	string a,b;
    	cin>>a>>b;
    	
    	int min = getMin(a.size(),b.size());
    	int c = 0;
    	for(i=0;i<min;i++){
    		if(a[i]==b[i]){
    			c++;
    		}
    	}
    	cout<<c<<endl;
    }
    return 0;
}
