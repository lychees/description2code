int main()
{
	int t ; s(t) ;
	while(t--)
	{
		int n ; int b ;
		s(n) ; s(b) ;
		int v = n/b ;
		int w1 = v*b ;
		int w2 = (v+1)*b ;
		int d1 = abs(w1-n) ;
		int d2 = abs(w2-n) ;

}
