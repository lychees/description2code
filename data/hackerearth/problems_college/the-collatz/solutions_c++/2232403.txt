#include <bits/stdc++.h>
using namespace std;

int main()
{
	int tc;
	scanf("%d",&tc);
	while(tc--)
	{
    int n;
    scanf("%d",&n);
    int steps=0;
    long long int odd=0;
    long long int even=0;
    while(n!=1)
    {
    	if(n%2==0)
    	{
    		even+=n;
    		n/=2;
    	}else
    	{
    		odd+=n;
    		n=3*n+1;
    	}
    	steps++;
    }
    odd+=1;
    if(even%steps>odd%steps)
    {
    	printf("Even Rules\n");
    }
    else if(even%steps<odd%steps)
    {
    	printf("Odd Rules\n");
    }else
    {
    	printf("Tie\n");
    }
}
    return 0;
}
