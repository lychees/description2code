#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,i,j,A[100][100],B[100][100];
    	cin>>n;
    	for(i=0;i<n;i++)
    	for(j=0;j<n;j++)
    	cin>>A[i][j];
    	if(A[0][0]==0)
    	cout<<"NOT POSSIBLE"<<endl;
    	
else{   B[0][0]=1; 	for(i=1;i<n;i++)
    	{
    		if(B[0][i-1]==1&&A[0][i]==1)
    		B[0][i]=1;
    		else
    		B[0][i]=0;
    		
    	}
    		for(i=1;i<n;i++)
    	{
    		if(B[i-1][0]==1&&A[i][0]==1)
    		B[i][0]=1;
    		else
    		B[i][0]=0;
    		
    	}
    	for(i=1;i<n;i++)
          for(j=1;j<n;j++)
          B[i][j]=A[i][j]&&(B[i-1][j]||B[i][j-1]);
          
          if(B[n-1][n-1]==1)
          cout<<"POSSIBLE"<<endl;
          else
          cout<<"NOT POSSIBLE"<<endl;
          
    	
    }
    }
    return 0;
}
