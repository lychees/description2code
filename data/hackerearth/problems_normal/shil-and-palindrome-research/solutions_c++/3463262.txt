typedef long long ll;
#include <iostream>
#include <stdio.h>
#include <cmath>
#include <algorithm>
#include <vector>
#include <string>
#include <string.h>
#include <map>
#include <queue>
#include <stack>
#include <sstream>
#include <fstream>
#include <stdlib.h>
#include <utility>

using namespace std;
ll n,bit[100007][26];
void update(ll i,ll val,ll pos)
{
	for(;i<=n;i+=(i&-i))
	bit[i][pos]+=val;
}
ll query(ll i,ll val)
{
	ll ans=0;
	for(;i>0;i-=(i&-i))
	ans+=bit[i][val];
	return ans;
}
int main()
{
	ll q,i,ty,l,r,val,c;
	char x;
	string s;
	scanf("%lld %lld",&n,&q);
	memset(bit,0,sizeof(bit));
	cin>>s;
	for(i=0;i<n;i++)
	{
		update(i+1,1,s[i]-'a');
	}
	while(q--)
	{
		scanf("%lld",&ty);
		if(ty==1)
		{
			cin>>l;
			cin>>x;
			update(l,1,x-'a');
			update(l,-1,s[l-1]-'a');
			s[l-1]=x;
		}
		else
		{
			scanf("%lld %lld",&l,&r);
			c=0;
			for(i=0;i<26;i++)
			{
				val=query(r,i)-query(l-1,i);
				if(val%2)
				c++;
			}
			if(c<=1)
			printf("yes\n");
			else
			printf("no\n");
		}
	}
	return 0;
}
