#include <bits/stdc++.h>

using namespace std;

int main() {
	int t;
	scanf("%d", &t);
	for (int tt = 1; tt <= t; tt++) {
		long long a, b;
		scanf("%lld %lld", &a, &b);
		long long res = 0;
		if (b - 1 > a) {
			res = (b - 1) & (b - 2);
		}
		res = max(res, b & (b - 1));
		printf("%lld\n", res);
	}
	return 0;
}
