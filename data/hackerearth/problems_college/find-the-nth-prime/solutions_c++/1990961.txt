#include<stdio.h>
#include<math.h>
int Prime(int x)
{
   	int i,flag=0;
   	for(i=2;i<=sqrt(x);i++)
   	{
  		if(x%i==0)
   		{
   			flag=1;
   			break;
   		}
   	}
   	return flag;
}
int main()
{
    int n,count=0,i;
    scanf("%d",&n);
    for(i=2;i<=10000;i++)
    {
   		if(Prime(i)==0)
   			count++;
   		if(count==n)
   		{
   			printf("%d\n",i);
   			break;
   		}
    }
    return 0;
}
