#include <iostream>
using namespace std;
long long int get_parent(long long int,long long int);
int main()
{
    int tc;
    long long int a,b,c;
    cin>>tc;
    while(tc--)
    {
    	cin>>a>>b;
    	
    	c=get_parent(a,b);
    	cout<<c<<endl;
    	 	
    }
}
long long int get_parent(long long int a,long long int b)
{
	while(a!=b)
	{
		if(a>b)
		{
			a=a/2;
		}
		else
		{
			b=b/2;
		}
	}
	return a;
}
