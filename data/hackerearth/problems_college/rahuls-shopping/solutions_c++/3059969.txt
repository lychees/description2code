#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    int l,b,temp=0,ans=0,area=0,mul=0,hold=0;
    float tcal=0,cal=0;
    cin>>l>>b;
    area=l*b;
    for(int i=0;i<n;i++)
    {
    	temp=a[i];
    	mul=a[i]%10;
    	mul+=a[i]/10;
    	temp*=mul;
    	hold=temp*area;
    
    	if(a[i]<=10)
    	{
    		tcal=0.25*hold;
    		cal=hold-tcal;
    	    	
    			ans+=floor(cal);
  
    	}
    		if(a[i]>=11 && a[i]<=20)
    	{
    		tcal=0.3*hold;
    		cal=hold-tcal;
    		
    			ans+=floor(cal);
    	}
    	
    		if(a[i]>=21 && a[i]<=50)
    	{
    		tcal=0.2*hold;
    		cal=hold-tcal;
    		
    			ans+=floor(cal);

    	}
    
    	
    	mul=0;
    	tcal=0;
    	cal=0;
    
    }
    cout<<ans<<"\n";
    return 0;
}
