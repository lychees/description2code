#include <iostream>
using namespace std;
int main()
{
    long long int t,r,c,o,e,tmp;
    cin>>t;
    while(t--)
    {
        cin>>r>>c;
        if(r<=c)
        {
            if(r%2==0)
            {
                tmp=r/2;
                e=tmp*(tmp+1);
                o=tmp*tmp;
            }
            if(r%2==1)
            {
                tmp=r/2+1;
                o=tmp*tmp;
                tmp=r/2;
                e=tmp*(tmp+1);
            }
        }
        else if(r>c)
        {
            tmp=r-(c-1);
            if(tmp%2==0)
            {
                e=(tmp/2)*c;
                o=(tmp/2)*c;
                r=r-tmp;
                c=c-1;
                if(r%2==0)
                {
                    tmp=r/2;
                    e=e+(tmp*(tmp+1));
                    o=o+(tmp*tmp);
                }
                else if(r%2==1)
                {
                    tmp=r/2+1;
                    o=o+(tmp*tmp);
                    tmp=r/2;
                    e=e+(tmp*(tmp+1));
                }
            }
            else if(tmp%2==1)
            {
                if(r%2==1)
                {
                    o=((tmp/2)+1)*c;
                    e=(tmp/2)*c;
                }
                else if(r%2==0)
                {
                    e=((tmp/2)+1)*c;
                    o=(tmp/2)*c;
                }
                r=r-tmp;
                c=c-1;
                if(r%2==0)
                {
                    tmp=r/2;
                    e=e+(tmp*(tmp+1));
                    o=o+(tmp*tmp);
                }
                else if(r%2==1)
                {
                    tmp=r/2+1;
                    o=o+(tmp*tmp);
                    tmp=r/2;
                    e=e+(tmp*(tmp+1));
                }
            }

        }

        cout<<e<<" "<<o<<endl;


    }

	return 0;
}
