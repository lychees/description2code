#include <iostream>
using namespace std;
#include<stdio.h>
#include<string.h>
inline long long int nI(char str[],long long int i,char ch)
{
	long long int l=strlen(str);
	if(i==l)
	return l;
	char *p=strchr(str+i+1,ch);
	if(p==NULL)
	return l;
	return (p-str);
}
int main()
{
   long long int ans,ntc,ans1,j,i,nj,c,k,len,b,a,y,x;

   while(ntc--)

   	ans=ans1=0;
   	if(a==0)

   		{

   		c++;

   		{
   		ans+=c*(c+1)/2;
   		c=0;
   		}
   		}
   		ans+=c*(c+1)/2;
   	}
   	else
   	{
   		x=y=-1;
   		i=j=nI(str,x,'R');
   		for(k=1;k<a;k++)
   		{
   			y=j;
   			j=nI(str,j,'R');
   		}
   		while(j!=len)
   		{
   			nj=nI(str,j,'R');
   			ans+=(i-x)*(nj-j);
   			j=nj;
   			x=i;
   			i=nI(str,i,'R');
   		}
   	}
   	
   	if(b==0)
   	{

   	}

   	{
   		x=y=-1;
   		j=i=nI(str,x,'K');
   		for(k=1;k<b;k++)
   		{
   			y=j;
