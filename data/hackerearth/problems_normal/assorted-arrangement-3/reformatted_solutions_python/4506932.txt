n, m = map(int, raw_input().split())
d = map(int, raw_input().split())

col = [i-1 for i in map(int, raw_input().split())]

# print col,d

c = d[0]
for i in col:
    flag = True
    while flag:
        if c % d[i] == 0:
            for j in d[i+1:]:
                if c % j == 0:
                    c = c+d[i]
                    break
            else:
                flag = False
        else:
            c = c+1
    c = c+1
print c-1
