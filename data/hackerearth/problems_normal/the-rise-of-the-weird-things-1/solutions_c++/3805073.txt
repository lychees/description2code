#include <bits/stdc++.h>
#define ll long long int
#define gt(i) scanf("%lld", &i)

using namespace std;

int main()
{
	ll i, j, k, n, m, sm1 = 0, sm2 = 0;
	gt(n);
	map <ll, ll> even;
	map <ll, ll> odd;
	map <ll, ll> ::iterator pos;
	
	for(i = 0; i < n; i++) {
		gt(j);
		if(j % 2 == 0) {
			even[j]++;
			sm1 += j;
		}
		else {
			odd[j]++;
			sm2 += j;
		}
	}
	for(pos = even.begin(); pos != even.end(); pos++) {
		while(pos->second) {
			cout << pos->first << " ";
			pos->second--;
		}
	}
	cout << sm1 << " ";
	for(pos = odd.begin(); pos != odd.end(); pos++) {
		while(pos->second) {
			cout << pos->first << " ";
			pos->second--;
		}
	}
	cout << sm2 << endl; 
	return 0;
}
