#include <iostream>
using namespace std;

unsigned long max(unsigned long a, unsigned long b) { return (a > b)? a : b;}
unsigned long LassiPrice(unsigned long price[], unsigned long n)
{
   unsigned long val[n+1];
   val[0] = 0;
   unsigned long i, j;
   
   for (i = 1; i<=n; i++)
   {
       unsigned long max_val = 0;
       for (j = 0; j < i; j++)
         max_val = max(max_val, price[j] + val[i-j-1]);
       val[i] = max_val;
   }
 
   return val[n];
}
int main()
{
	unsigned long T,i,L;
	cin >> T;
	while(T--)
	{
		cin >> L;
		unsigned long priceLassi[L];
		for(i=0;i<L;i++)cin >> priceLassi[i];
		unsigned long ans=LassiPrice(priceLassi,L);
		cout << ans << '\n';
	}
    
    return 0;
}
