#include <iostream>
#include <cmath>
#include <cstdio>
#include <algorithm>

using namespace std;

#define YES puts("Able")
#define NO puts("Not able");

int main (){
	int t; cin >> t; while (t--) {
		long n , x , y; cin >> n >> x >> y; double a[n+1] , steps = 0 , dis = 0;
		for (int i = 0; i < n; i++)
			cin >> a[i];
		a[n] = sqrt (x * x + y * y);
		sort (a , a+n+1);
		for (int i = 0; i < n; i++)
			steps += a[i];
		if (steps >= a[n])
			YES;
		else
			NO;
	}
	return 0;
}
