#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    char str[4005];
    int count[26];
    for(int i=0;i<26;i++)
    	count[i]=0;
    	
    cin>>str;
    
    for(int i=0;i<strlen(str);i++){
    	if(str[i]==' ')
    		continue;
    	count[(int)(str[i])-97]++;
    }
    int max = -1,index=0;
    
    for(int i=0;i<26;i++){
    	if(count[i]>max){
    		max = count[i];
    		index = i;
    	}
    }
    
    char hr[]="head rush";
    char cc[]="cubing club";
    int flag1=0,flag2=0;
    
    for(int i=0;i<strlen(hr);i++){
    	if((int)(hr[i])-97==index){
    		flag1=1;
    		break;
    	}
    }
    for(int i=0;i<strlen(cc);i++){
    	if((int)(cc[i])-97==index){
    		flag2=1;
    		break;
    	}
    }
    
    if(flag1==1 && flag2==1)
    	cout<<"Thanks both"<<endl;
    else if(flag1==1 && flag2==0)
    	cout<<"Thanks head rush"<<endl;
    else if(flag1==0 && flag2==1)
    	cout<<"Thanks cubing club"<<endl;
    else
    	cout<<"none";
    return 0;
}
