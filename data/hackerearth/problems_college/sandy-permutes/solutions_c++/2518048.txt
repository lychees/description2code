#include <bits/stdc++.h>
#define ll long long
using namespace std;
int a[100010];
ll int r[100010];
int h[100010],h2[100010];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int n,k;
	cin>>n>>k; //length of array & k-swaps
	int i;
	for(i=1;i<=n;i++)
	{
		cin>>a[i];
		h[a[i]]=i;//Hashing
		h2[i]=a[i];//Reverse map
	}
	int mc=n,index,element;
	for(i=1;i<=n;i++)
	{
		if(k==0)
		break;
		if(h[mc]!=i)
		{
			index=h[mc];
			element=h2[i];
			h[mc]=i;//map
			h[element]=index;
			h2[i]=mc;
			h2[index]=element;
			k--;
		}
		mc--;
	}
	for(i=1;i<=n;i++)
	{
		r[h[i]]=i;
	}
	for(i=1;i<=n;i++)
	cout<<r[i]<<" ";
	cout<<endl;
	return 0;
}
