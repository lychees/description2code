#include <iostream>
#define inf 1000000000000000000LL
using namespace std;

int main()
{
	long long dp[509][309],sum[509]={0},a;
    int n,k;
    cin>>n>>k;
    cin>>sum[0];
    for(int i=1;i<n;i++)
    {
    cin>>a;
    sum[i]=sum[i-1]+a;
    }
    for(int i=0;i<n;i++)
    {
    dp[i][1]=sum[i]*sum[i];
    for(int j=2;j<=k;j++)
    {
    dp[i][j]=inf;
    for(int l=0;l<i;l++)
    dp[i][j]=min(dp[l][j-1]+(sum[i]-sum[l])*(sum[i]-sum[l]),dp[i][j]);	
    }
    }
    long long res=(long long)k*dp[n-1][k]-(sum[n-1]*sum[n-1]);
    cout<<res<<"\n";
    return 0;
}
