#include <iostream>
using namespace std;

int main()
{
    int t,n,k,a[10000],i,c;
    cin>>t;
    while(t--)
    {
    	c=0;
    	cin>>n>>k;
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	if(k<n)
    	{
    		for(i=0;i<n;i++)
    		{
    			if(a[i]<=0)
    				c++;
    		}
    	}
    	if(c<k)
    		cout<<"YES\n";
    		else
    		cout<<"NO\n";
    }
    return 0;
}
