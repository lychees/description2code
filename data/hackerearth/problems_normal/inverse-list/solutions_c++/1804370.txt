#include <iostream>
#include<bits/stdc++.h>
#define ll long long int
using namespace std;
int main()
{
    ///cout << "Hello World!" << endl;
    int t;
    ll n,i;
    int ar[100000];
    cin >> t;
    while(t--)
    {
    	scanf("%lld",&n);
    	for(i=1;i<=n;i++)
    	{
    		scanf("%d",&ar[i]);
    		
    	}
    	int b[n+1];
    	for(i=1;i<=n;i++)
    	{
    		b[ar[i]]=i;
    	}
    	for(i=1;i<=n;i++)
    	{
    		if(ar[i]!=b[i])
    		break;
    	}
    	if(i==n+1)
    	printf("inverse\n");
    	else
    	printf("not inverse\n");
    }
    return 0;
}
