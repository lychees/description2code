#include<bits/stdc++.h>
using namespace std;

long int arr[100005];

int main()
{
    long int n, k, d, ans=0, sum=0;
    cin>>n>>k>>d;
    arr[0]=0;
    for(int i=1; i<=n; i++)
        scanf("%ld",&arr[i]);
    for(int i=1; i<k; i++)
        sum+=arr[i];
        
    for(int i=k; i<=n;i++)
	{
        sum+=arr[i];
        sum-=arr[i-k];

        //cout<<i<<" "<<sum<<endl;

        if(sum<d)
        {
            ans++;
            int t = i;
            //cout<<t<<" ";
            while(arr[t]==0)
            {
                t--;
                if(t==i-k)
                {
                    cout<<"-1"<<endl;
                    exit(0);
                }
            }
            sum-=arr[t];
            arr[t] = 1000000;
            sum+=arr[t];

            t = t+k-1;
            while(arr[t]==0)
                t--;
            arr[t] = 1000000;

            //cout<<t<<endl<<endl;
        }
    }
    cout<<ans<<endl;
    return 0;
}
