
#include <bits/stdc++.h>

using namespace std;

int main()
{long long int a[10000001];
for(long long int i=0;i<10000001;i++){
        a[i]=0;

}
a[0]=a[1]=0;
for(long long int i=2;i<10000001;i++){
    for(long long int j=2;j*i<10000001;j++){
        a[i*j]=1;
    }
}
for(long long int i=2;i<10000001;i++)
{
    if(a[i]==0)
        a[i]=a[i-1]+1;
    else
        a[i]=a[i-1];
}

long long int t,m,n,k;

cin>>t;
while(t--)
{
    cin>>m>>n;
    


    k=a[n]-a[m-1];


cout<<k<<endl;

}
    return 0;
}
