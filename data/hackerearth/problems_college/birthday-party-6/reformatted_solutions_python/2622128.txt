import math


def nCr(n, r):
    return math.factorial(n) / (math.factorial(r) * math.factorial(n-r))


T = int(raw_input())
for _ in range(T):
    M, N = raw_input().split()
    M, N = int(M), int(N)
    if N >= M + 2:
        print -1
    else:
        print(nCr(M+1, N) * math.factorial(M) * math.factorial(N)) % 1000000007
