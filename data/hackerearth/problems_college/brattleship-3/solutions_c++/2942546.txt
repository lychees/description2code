#include <iostream>
using namespace std;

int main()
{
	int i,j,k,r,c,w,t;
	cin>>t;
	for(i=1;i<=t;i++)
	{
		cin>>r>>c>>w;
		
		if(w==c)
		{
			cout<<"Case #"<<i<<": "<<w<<endl;
			
		}
		else
		{
			if(w==1)
				cout<<"Case #"<<i<<": "<<r*c<<endl;
			else
				cout<<"Case #"<<i<<": "<<(r*w)+(r*(c-(w+1)))<<endl;
			
		}
	}
    //cout << "Hello World!" << endl;
    return 0;
}
