#include <iostream>
using namespace std;
long long int compute(int n) {
if(n==0) return 0;
long long  k = (n+1)/2;
return k*k + compute(n/2);
}
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--){
    	
    long long int x;
    cin>>x;
    cout<<compute(x)<<endl;
    }
    return 0;
}
