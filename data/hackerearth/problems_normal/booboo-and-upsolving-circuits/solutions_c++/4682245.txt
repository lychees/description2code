#define per(i,b,a) for(int i=(b);i>=(a);--i)

    template<typename T>debugger& operator ,(const T& v) { cerr<<v<<" ";return *this; }

const int maxn = 1000000000;
typedef long long int ll;
int n, m;
ll T[100000+2];

int daysRequiredFor(ll t) {
	int days = 1;
	ll currSum = 0;

	rep(i,0,n) {
		if(currSum + T[i] > t) {
			days += 1;
			currSum = T[i];
		}
		else 
			currSum += T[i];
	}

	return (days <= m)? 1: 0;
}

int main() {
	cin>>n>>m;

	ll lo = 1, hi = 1LL * maxn * maxn;
	
	rep(i,0,n) {
		cin>>T[i];
		lo = max(lo, T[i]);
	}

	while(lo < hi) {
		ll mid = (lo + hi)>>1;

		if(daysRequiredFor(mid)) hi = mid;

		else lo = mid+1;
	}

	cout<<lo<<endl;
	return 0;
}
