void sort(char n[][31],int r){
	int i, j, swapped;
	char temp[31];
	do{
		swapped = 0;
		for (i=0; i<r-1; ++i){
			if (strcmp(n[i],n[i+1])>0){
				strcpy(temp, n[i]);
				strcpy(n[i],n[i+1]);
				strcpy(n[i+1],temp);
				swapped = 1;
			}
		}
		--r;
	}while (swapped==1);
}

void displayFrequency(char distinctNames[][31], int freq[], int rows){
	int i;
	for (i=0; i<rows; ++i){
		cout << distinctNames[i] << " " << freq[i] << endl;
	}	
}

void setFrequency(char names[][31], char distinctNames[][31], int freq[], int & rows){
	int i, j=0, k, match;
	for (i=0; i<rows; ++i, ++j){
		strcpy(distinctNames[j],names[i]);
		match = 1;
		freq[j] = 1;
		for (++i; i<rows; ++i){
			if (strcmp(distinctNames[j],names[i])!=0){
				match = 0;
				break;
			}
			++freq[j];
		}
		if (match == 0)
			--i;
	}
	rows = j;
}
