#include <bits/stdc++.h>
using namespace std;
#define ll long long int
unordered_map<ll,bool>visited;
unordered_map<ll,ll>mp;
ll a[10005],b[10005];
int main(){
	ll t,i,n;
	cin >> t;
	while(t--){
		visited.clear();
		mp.clear();
		cin >> n;
		for(i=0; i<n; i++){
		 cin >> a[i];
		 b[i] = a[i];
		 visited[a[i]] = false;
		}
		sort(b,b+n);
		reverse(b,b+n);
		int count =0;
		for(i=0; i<n; i++){
			if(visited[b[i]]==false)
			{
				count++;
				mp[b[i]]=count;
				visited[b[i]]=true;
			}
			else
			 count++;
		}
		for(i=0; i<n; i++){
			printf("%lld ",mp[a[i]]);
		}
		 printf("\n");
	}
	return 0;
}
