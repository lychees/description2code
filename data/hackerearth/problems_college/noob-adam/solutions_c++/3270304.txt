#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int n,k,i=0,cnt=0,j=2,m,count=0;
    	cin>>n>>k;
    	long int a[n];
    	for(;i<n;i++)cin>>a[i];
    	for(i=0;i<n-2;i++)
    	{
    		m=j;
    		while(m<n&&(a[m]-a[i])<=k)
    		{m++;}
    		cnt=m-i-1;
    		j=m;
    		if(cnt>0)
    		count+=(cnt-1)*cnt/2;
    	}
    	cout<<count<<endl;
    }
    return 0;
}
