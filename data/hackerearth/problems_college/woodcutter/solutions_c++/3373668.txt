#include <iostream>
#include <stdlib.h>

using namespace std;

int comp(const void *a, const void *b);

int main()
{
    int n, i, start, end, mid, tmp_key;
    long long int m, ans, tmp;
    cin>>n>>m;
    long long int h[n];
    
    for (i = 0; i < n; i++) {
    	cin>>h[i];
    }
    
    qsort(h, n, sizeof(long long int), comp);
    
    start = 0; end = h[0]; tmp_key = -1;
    tmp = 999999999999999999;
    while (start <= end) {
    	ans = 0;
    	mid = start + (end - start) / 2;
    	for (i = 0; i < n; i++) {
    		if (h[i] <= mid) {
    			break;
    		}
    		ans = ans + h[i] - mid;
    	}
    	if (ans == m) {
    		cout<<mid<<endl;
    		break;
    	} else if (ans < m) {
    		end = mid - 1;
    	} else {
    		if (ans <= tmp) {
    			tmp = ans;
    			tmp_key = mid;
    		}
    		start = mid + 1;
    	}
    }
    if (ans != m) cout<<tmp_key<<endl;
    
    return 0;
}

int comp(const void *a, const void *b)
{
	if (*((long long int *)b) > *((long long int *)a)){
		return 1;
	} else {
		return 0;
	}
}
