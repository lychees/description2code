#include<iostream>
#include<algorithm>
#include<vector>
#include<string>
using namespace std;

bool findd(vector<int> &sum,int low,int high,int diff)
{
	if(low>high)return false;
	else if(low<=high){
		int mid=(low+high)/2;
		if(sum[mid]==diff)return true;
		else if(sum[mid]<diff)return findd(sum,mid+1,high,diff);
		else if(sum[mid]>diff)return findd(sum,low,mid-1,diff);
	}
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int k;
		cin>>k;
		string input;
		cin>>input;
		int len=input.size();
		vector<int> sum(len);
		for(int i=0;i<len;i++)
		{
			sum[i]=input[i]-'a'+1;
		}
		for(int i=1;i<len;i++)
		{
			sum[i]=sum[i]+sum[i-1];
		}
	//	for(int i=0;i<len;i++)
	//	{
	//		cout<<sum[i]<<" ";
	//	}cout<<endl;
		int count=0;
		if(sum[0]==k){
	//				cout<<"i: "<<0<<'a'<<endl;
					count++;
				}
		for(int i=1;i<len;i++)
		{
			if(sum[i]==k){
	//				cout<<"i: "<<i<<'b'<<endl;
					count++;
				}
			else if(sum[i]>k){
				int diff=sum[i]-k;
				if(findd(sum,0,i-1,diff)==true){
	//				cout<<"i: "<<i<<'c'<<endl;
					count++;
				}
			}
		}
		cout<<count<<endl;
	}
	
	return 0;
}
