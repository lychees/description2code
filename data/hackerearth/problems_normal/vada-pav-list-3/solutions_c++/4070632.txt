#include <iostream>
#include <stdio.h>
#include <string.h>
using namespace std;

int main()
{
	long long n,i,count=0;
	int hash_table[27][27][27][27][27]={0};
	int a,b,c,d,e;
	cin>>n;
	while(n--)
	{
			int num1=0,num2=0,num3=0,num4=0,num5=0;
            char str[10];
			scanf("%s", str);
			for(i=0;i<strlen(str);i++)
			{
			if (i==0) num1=(str[0]-'a'+1);
			if (i==1) num2=(str[1]-'a'+1);
			if (i==2) num3=(str[2]-'a'+1);
			if (i==3) num4=(str[3]-'a'+1);
			if (i==4) num5=(str[4]-'a'+1);
			}
			if (hash_table[num1][num2][num3][num4][num5]==0)
			{
				hash_table[num1][num2][num3][num4][num5]=1;
				count++;	
			}
			
	}

	cout<<count<<endl;
	 for(a=0;a<27;a++)
        for(b=0;b<27;b++)
            for(c=0;c<27;c++)
                for(d=0;d<27;d++)
                    for(e=0;e<27;e++)
                        if(hash_table[a][b][c][d][e])
                        {
                            printf("%c",a+'a'-1);
                            if(b)
                            printf("%c",b+'a'-1);
                            if(c)
                            printf("%c",c+'a'-1);
                            if(d)
                            printf("%c",d+'a'-1);
                            if(e)
                            printf("%c",e+'a'-1);
                            printf("\n");
                        }
    return 0;
}
