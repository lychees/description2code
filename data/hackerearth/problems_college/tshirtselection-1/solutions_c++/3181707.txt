template<typename X> inline void input(X &n ) {
   register int ch=gc();int sign=1;n=0;
   while( ch < '0' || ch > '9' ){if(ch=='-')sign=-1; ch=gc();}
   while(  ch >= '0' && ch <= '9' ) n = (n<<3)+(n<<1) + ch-'0', ch=gc();
   n=n*sign;
}

int gcd(int a, int b) {
	if(b == 0)
		return a;
	return gcd(b, a%b);
}

ll power(ll a, ll b) {
	ll res = 1;
	while(b) {
		if(b & 1) {
			res = (res * a) % MOD;
		}
		a = (a*a) % MOD;
		b >>= 1;
	}
	return res;
}

ll expo(ll a, ll b) {
	ll res = 1;
	while(b) {
		if(b & 1) {
			res = (res * a);
		}
		a = (a*a);
		b >>= 1;
	}
	return res;
}


void solve() {
	int n;
	cin>>n;
	vector<int> a(n);
	for(int i=0;i<n;i++)
		cin>>a[i];
	int s = 0, l = 0, ml = 0, e = 0;
	int j = 0, i = 0;
	map<int, bool> m;
	for(i=0;i<n;i++) {
		if(m.find(a[i]) == m.end()) {
			m[a[i]] = true;
			if(ml < i-j+1) {
				ml = i-j+1;
				s = j;
				e = i;
			}
		}
		else {
			while(j < i && a[j] != a[i]) {
				m.erase(a[j]);
				j++;
			}
			//m.erase(a[j]);
			j++;
			if(ml < i-j+1) {
				ml = i-j+1;
				s = j;
				e = i;
			}
		}
	}
	cout<<s+1<<" "<<e+1<<"\n";
}
