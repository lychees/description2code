#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
    int k;
    cin>>k;
    string s;
    cin>>s;
    int i,j,sum,ans=0;
    for(i=0;i<s.length();i++)
    {
    	sum=0;
    	for(j=i;j<s.length();j++)
    	{
    		sum+=s[j]-'a'+1;
    		if(sum==k)
    		ans++;
    		if(sum>k)
    		break;
    	}
    }
    cout<<ans<<endl;
    }
    return 0;
}
