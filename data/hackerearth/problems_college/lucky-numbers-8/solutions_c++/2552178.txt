#include <iostream>	
#define NUM_P 1000001
using namespace std;

int main()
{
    int t;
    long long n;
    scanf("%d", &t);
    long long arr[NUM_P];
    while(t > 0) {
    	scanf("%lld", &n);
    	for(long long i=0; i<n; i++) {
    		scanf("%lld", &arr[i]);
    	}
    	
    	long long count = 0;
    	bool flag = true;
    	for(long long i=0; i<n; i++) {
    		flag = true;
    		for(long long j = i+1; j<n; j++) {
    			if(arr[i] <= arr[j] ) {
    				flag = false;
    				break;
    			}
    		}
    		if(flag) {
    			count++;
    		}
    	}
    	printf("%lld\n", count);
    	t--;
    }
    return 0;
}
