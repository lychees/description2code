#include <iostream>
#include <math.h>
#include <iomanip>
using namespace std;

int main()
{
	
	float ans,p,Time;
    int testcases;
    cin>>testcases;
    while(testcases--)
    {
    	cin>>p>>Time;
    	ans = 1-pow(1-p,Time/1440.0);
    	cout<<fixed<<setprecision(4);
    	cout<<ans<<endl;
    }
}
