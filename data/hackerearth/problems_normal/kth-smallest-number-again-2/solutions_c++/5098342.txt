#include <bits/stdc++.h>

using namespace std;

const int N = 1e5 + 10;

pair <long long, long long> pa[N];

int main() {
	int tt;
	scanf("%d", &tt);
	while (tt--) {
		int n, q;
		scanf("%d %d", &n, &q);
		for (int i = 0; i < n; i++) {
			scanf("%lld %lld", &pa[i].first, &pa[i].second);
	    }
	    sort(pa, pa + n);
		vector < pair <long long, long long> > seg;
		vector < long long > pos;
	    seg.push_back(pa[0]);
	    for (int i = 1; i < n; i++) {
	        auto &last = seg.back();
	        if (pa[i].first > last.second) {
	            seg.push_back(pa[i]);
	        } else if (pa[i].second > last.second) {
	            last.second = pa[i].second;
	        }
	    }
	    long long next = 1;
	    for (int i = 0; i < seg.size(); i++) {
	    	pos.push_back(next);
	    	next += seg[i].second - seg[i].first + 1;
	    }
	    for (int qq = 1; qq <= q; qq++) {
	    	long long k;
	    	scanf("%lld", &k);
	    	int t = upper_bound(pos.begin(), pos.end(), k) - pos.begin() - 1;
	    	if (k - pos[t] > seg[t].second - seg[t].first) {
	    		printf("-1\n");
	    	} else {
		    	printf("%lld\n", (k - pos[t]) + seg[t].first);
	    	}
	    }
	}
	return 0;
}
