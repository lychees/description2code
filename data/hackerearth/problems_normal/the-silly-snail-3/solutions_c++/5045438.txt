#include <bits/stdc++.h>
using namespace std;
vector<int> g[100005];
void dfs(int src){
	if(src==0)
		return;
	cout<<src<<" ";
	for(int i=0;i<g[src].size();++i)
		dfs(g[src][i]);
}
int main()
{
    int t,n,a,b,c;
    cin>>t;
    while(t--){
    	cin>>n;
    	for(int i=0;i<100005;++i)
    		g[i].clear();
    	for(int i=0;i<n;++i){
    		cin>>a>>b>>c;
    		g[a].push_back(b);
    		g[a].push_back(c);
    	}
    	dfs(1);
    	cout<<endl;
    }
    
    return 0;
}
