ll dfs(int u, int p, int rt) {
	if(wt[u]<wt[rt] or (wt[u]==wt[rt] and u<rt)) return 0;
	ll ans=1;
	rep(i,0,v[u].size()) {
		if(v[u][i]!=p) ans=mul(ans,dfs(v[u][i],u,rt)+1);
	}
	return ans;
}
