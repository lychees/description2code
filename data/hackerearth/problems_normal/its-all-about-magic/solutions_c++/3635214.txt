long long findFibonacciSeries(long long num)
{
    //cout<<"Number passed is "<<num<<endl;
    if(num==0)
    {
        return 0;
    }
    if(num==1)
    {
        return 1;
    }
    long long count=2,a=0,b=1,c=0,temp;
    while(count<=num)
    {
        c=a+b; //cout<<"C is"<<c;
        temp=b;
        b=c;
        a=temp;
        count++;
    }
    return c;
}
