#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	long x;
    	cin>>n>>x;
    	long prev, k;
    	cin>>prev;
    	n--;
    	long count=0;
    	while(n--)
    	{
    		cin>>k;
    		long b=0;
    		if(k<=prev)
    		{
    			long a = prev-k+1;
    			b = a/x;
    			if(a%x>0)
    				b++;
    		}
    		count+=b;
    		prev=k+(b*x);
    	}
    	cout<<count<<endl;
    }
    return 0;
}
