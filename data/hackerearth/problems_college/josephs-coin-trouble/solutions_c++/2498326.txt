#include<iostream>
using namespace std;


int count( int S[], int m, int n )
{
   
    if (n == 0)
        return 1;
    
    
    if (n < 0)
        return 0;

    
    if (m <=0 && n >= 1)
        return 0;

    
    return count( S, m - 1, n ) + count( S, m, n-S[m-1] );
}


int main()
{
    int i, j;
    int arr[3],n,m;
    cin>>n;
    if(n>9000000)
    goto end;
    for(i=0;i<3;i++)
    cin>>arr[i];
    for(i=0;i<3;i++)
    {
    	if(arr[i]>9000000)
    	goto end;
    }
    m = sizeof(arr)/sizeof(arr[0]);
    cout<<count(arr, m, n);
    end:
    return 0;
}
