#include <iostream>
#include <cmath>
#include <algorithm>

using namespace std;

int bs(int *a, int n, int f) {
	int low = 0, high = n-1;
	while(low <= high) {
		int mid = (low+high)/2;
		if(a[mid] == f) return mid;
		else if(a[mid] < f) low = mid+1;
		else high = mid-1;
	}
	return -1;
}

int main(void) {
	int n, m;
	cin >> n >> m;
	int a[n];
	for(int i = 0; i < n; i++)
		cin >> a[i];
	sort(a, a+n);
	int i;
	for(i = m; i <= a[n-1]; i += m)
		if(bs(a, n, i) == -1) break;
	cout << i-m << endl;
	return 0;
}
