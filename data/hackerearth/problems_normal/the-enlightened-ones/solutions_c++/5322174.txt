#include<bits/stdc++.h>
using namespace std;

bool isvalid(vector<int> &vec,int mid,int k)
{
	int val=vec[0]+mid;
	k--;
	for(int i=1;i<vec.size();i++)
	{
		if(val+mid>=vec[i])
			continue;
		if(k==0)
			return false;
		val=mid+vec[i];
		k--;
	}
	return true;
}

int main()
{
	ios_base::sync_with_stdio(false);
	int n,k;
	cin >> n >> k;
	vector<int> vec(n);
	for(int i=0;i<n;i++)
		cin >> vec[i];
	sort(vec.begin(),vec.end());
	int mid,low=0,high=10000000;
	bool val;
	while(low<=high)
	{
		mid=low+(high-low)/2;
		val=isvalid(vec,mid,k);
		if(val&&isvalid(vec,mid-1,k)==false)
			break;
		if(val==true)
			high=mid-1;
		else
			low=mid+1;
	}
	cout << mid << "\n";
	return 0;
}
