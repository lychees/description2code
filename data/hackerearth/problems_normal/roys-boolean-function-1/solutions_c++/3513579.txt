#include <iostream>
#include <cmath>
using namespace std;

int gcd(int a, int b) {
	while(b != 0) {
		int temp = b;
		b = a % b;
		a = temp;
	}
	return a;
}

bool isprime(int n) {
	if(n < 2) return false;
	for(int i=2; i<=sqrt(n); i++) {
		if(n%i == 0)
			return false;
	}
	return true;
}

int main()
{
    int n, t, ans;
    
    cin >> t;
    
    while(t--) {
    cin >> n;
    ans = 0;
    
    for(int i=1; i<=n; i++) {
    	if(gcd(n,i) == 1)
    		ans++;
    }
    
    // cout << n << " " << ans << endl; // debug
    
    if(isprime(ans))
    	cout << "TRUE" << endl;
   	else
   		cout << "FALSE" << endl;
    }
    
    return 0;
}
