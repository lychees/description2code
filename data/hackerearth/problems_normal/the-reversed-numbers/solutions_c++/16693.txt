#include <iostream>
using namespace std;

int main()
{
	int t,a,b,count,c,d,a1,b1;
	
	cin>>t;
	while(t--){
		d=0;
		count=0;
                a1=0;
                b1=0;
		
		cin>>a;
		cin>>b;
		
        while(a>0){
			a1=(a1*10)+(a%10);
			a=a/10;
		}
                
        while(b>0){
			b1=(b1*10)+(b%10);
			b=b/10;
		}
                
		c=a1+b1;
                
		while(c>0){
			d=(d*10)+(c%10);
			if(!(c%10) && !d){
                            count++;
                        }
			c=c/10;
		}
		
		//while(count--){
		//	cout<<"0";
		//}
		
		cout<<d<<"\n";
	}
	
    return 0;
}
