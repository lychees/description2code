#include<stdio.h>
#include<iostream>
#include<string.h>
#include<math.h>
#include<algorithm>
#include<stdlib.h>
#include<vector>
#include<map>
#include<set>

using namespace std;

int a[10000];
int mini(int n)
{
	int minimum=10000;
	int ct=0;
	for(int i=0;i<n;i++)
	{
		if(a[i]<minimum && a[i]!=-100000)
		minimum=a[i];
	}
	
	for(int i=0;i<n;i++)
	{
		if(a[i]==minimum)
		ct++;
	}
	return ct;
}

int findmin(int n)
{
	int minimum=10000;
	for(int i=0;i<n;i++)
	{
		if(a[i]<minimum && a[i]!=-100000)
		minimum=a[i];
	}
	return minimum;
}

int cont(int n)
{
	for(int i=0;i<n;i++)
	{
		if(a[i]>0)
		return 1;
	}
	return 0;
}

main()
{
	int n,ct;
	cin>>n;
	
	ct=n;
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
	}
	
	cout<<ct<<endl;
	
	while(cont(n))
	{
		int temp=findmin(n);
		int ans=mini(n);
		for(int i=0;i<n;i++)
		{
			a[i]=a[i]-temp;
			if(a[i]<=0)
			a[i]=-100000;
		}		
		ct=ct-ans;
		if(ct!=0)
		cout<<ct<<endl;
	}

}
