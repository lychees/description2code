def f():
    return raw_input().strip()


n = int(f())
k, v = {}, {}

for i in range(n):
    old, new = f().split()
    if old in v:
        ori = v[old]
        k[ori] = new
        v[new] = ori
    else:
        k[old] = new
        v[new] = old

print len(k)
for _1, _2 in k.items():
    print _1, _2
