{

	int t;

	{

		int arr[n+1];
		for(int i=0;i<n;i++)
		cin>>arr[i];
		
		int i=0;
		int j=n-1;
		int count1=0,count2=0,sum1=0,sum2=0;
		while(i<j)
		{
			//cout<<i<<" "<<j<<" "<<sum1<<" "<<sum2<<endl;
			if(sum1>sum2)
			{
				sum2+=(2*arr[j]);
				count2++;
				j--;
			}
			else if(sum2>sum1)
			{
				sum1+=arr[i];
				i++;
				count1++;
			}

				count2++;

			}
		}
		if(i==j && sum1<=sum2)
		count1++;
		if(i==j && sum1>sum2)
		count2++;
		cout<<count1<<" "<<count2<<endl;
	}
}
