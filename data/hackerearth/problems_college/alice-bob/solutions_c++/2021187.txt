#include <iostream>
#include<algorithm>
using namespace std;
int arr[10000];
int main()
{
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
	cin>>arr[i];
	sort(arr,arr+n);
	int count=0;
	int pos=0;
	for(int i=1;i<arr[n-1];i++)
	{
		if(arr[pos]==i)
		pos++;
		else
		count++;
	}
	if(count%2)
	cout<<"Alice";
	else
	cout<<"Bob";
    return 0;
}
