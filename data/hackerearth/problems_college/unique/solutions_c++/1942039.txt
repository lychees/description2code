#include <iostream>
#include <map>
using namespace std;

int main()
{
    int tno,n,i,count;
    long long int a;
    map <long long int,int> hash;
    
    cin>>tno;
    
    while(tno--)
    {
    	cin>>n;
    	count=0;
    	hash.clear();
    	for(i=0;i<n;i++)
    	{
    		cin>>a;
    		
    		if(hash[a]==0)
    		{
    			count++;
    			hash[a]=1;
    		}
    	}
    		
    	cout<<count<<endl;
    }
    return 0;
}
