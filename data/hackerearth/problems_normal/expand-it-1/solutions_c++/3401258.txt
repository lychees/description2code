#include <iostream>
#include <algorithm>
using namespace std;
long long int  arr[100000];
int main()
{
	string str;
	cin>>str;
	int l= str.length();
	string rs ="";
	string cp = str.substr(0, 1);
	
	for(int i=0;i<l;){
	
		int val = 0;
		int j=i;
		
		while(i++ <l && (str[i]-48)>=0 && (str[i]-48)<=9){
			val = val*10 + (str[i]-48);
		}
		arr[str[j]-97]= arr[str[j]-97] + val;
		
	}
	for(int v=0;v<26;v++){
		arr[v]+=arr[v-1];
	//	cout<<arr[v]<<" " ;
	}

	long long int n,val;
	cin>>n;
	int fg = 0;
	for(int i=0;i<n;i++){
		cin>>val;
		fg=0;
		for(int v =0; v<26;v++){
			if(val <= arr[v]&& v==0){
				char s = v+97;
				cout<<s<<"\n";
				fg=1;
				break;
			}else if(val <= arr[v]&& (arr[v]!=arr[v-1])){
				char s = (v)+97;
				cout<<s<<"\n";
				fg=1;
				break;
			}
		}
		if(fg==0)
		cout<<-1<<"\n";
	}
	
    return 0;
}
