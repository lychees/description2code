#include <bits/stdc++.h>
using namespace std;

int main()
{
	int tc;
	cin >> tc;
	while(tc--)
	{
		int n;
		cin >> n;
		if(n == 1) cout << 0 << "\n";
		else
		{
			int mnsteps = 999999999;
			for(int i=1;i<n;i++)
			{
				int mx = max(i, n);
				int mn = min(i, n);
				int steps = 0;
				while(1)
				{
					//cout << mx << " " << mn << endl;
					if(mx == mn)
					{
						break;
					}
					mx -= mn;
					if(mx < mn) swap(mx, mn);
					steps++;
				}
				if(mx == 1 && mn == 1)
					mnsteps = min(steps, mnsteps);
			}
			cout << mnsteps << "\n";
		}
	}
    return 0;
}
