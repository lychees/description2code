#include<iostream>
#include<stdio.h>
#include<math.h>
using namespace std;
void sort(int a[2000],int start,int end)
{
	int i,j,k,smallest,pos;
	for(i=start;i<=end;i++)
	{
		smallest=a[i];
		pos=i;
		for(j=i+1;j<=end;j++)
		{
			if(smallest>a[j])
			{
				smallest=a[j];
				pos=j;
			}
		}
		k=a[i];
		a[i]=a[pos];
		a[pos]=k;
	}
}
int main()
{
	int i,j,k,l,tc,ceo,coo,cto,sum;
	int a[2000],b[20000];
	cin>>ceo>>cto>>coo;
	cin>>tc;
	for(i=0;i<=tc-1;i++)
	{
		cin>>a[i];
	}
	sort(a,0,tc-1);
	l=0;
	for(i=0;i<=tc+2;i++)
	{
		if(i==0)
		{
			b[i]=min(cto,coo);
		}
	   else	if(i==(tc+2)/2)
	   {
	   	b[i]=ceo;
	   }
	   else if(i==tc+2)
	   {
	   		b[i]=max(cto,coo);
	   }
	   else
	   {
	   	b[i]=a[l++];
	   }
	}
    
    
    
    sum=0;
	for(i=0;i<=tc+1;i++)
	{  if(b[i]<b[i+1])
		{
		sum=sum+b[i+1]-b[i];}
		else
		{
				sum=sum-b[i+1]+b[i];
		}
	}
	cout<<"79";
}
