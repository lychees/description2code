#include <bits/stdc++.h>
using namespace std;

int main() {
	int n,i;
	cin>>n;
	vector<int> vt;
	long long int ans=0;
	for(i=0;i<n;i++){
		int x;
		cin>>x;
		vt.push_back(x);
	}
	while(vt.size()>=3){
		int maxi = -1,ind;
		for(i=1;i<vt.size()-1;i++){
			if(vt[i-1]*vt[i+1]>maxi) ind=i;
			maxi = max(maxi,vt[i-1]*vt[i+1]);
			
		}
		ans=ans+maxi;
		vector<int> vt1;
		for(i=0;i<vt.size();i++){
			if(ind==i) continue;
			vt1.push_back(vt[i]);
		}
		vt=vt1;
	}
	cout<<ans<<endl;
	return 0;
}
