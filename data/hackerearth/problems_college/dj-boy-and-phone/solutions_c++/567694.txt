#include <bits/stdc++.h>
using namespace std;
#define mod 1000000009; 
long long int dp[10005][10],a[10005];
int main()
{
    long long int i,j,k,l,m,n,t;
 
            a[0]=0;
            a[1]=10;
            
    		for(i=0;i<10;i++)
    	        dp[1][i]=1;
    	        
    	for(j=2;j<=10000;j++)
    	{
    		a[j]=0;
    	   dp[j][0]=(dp[j-1][0]+dp[j-1][8])%mod;
    	   dp[j][1]=(dp[j-1][1]+dp[j-1][2]+dp[j-1][4])%mod;
    	   dp[j][2]=(dp[j-1][2]+dp[j-1][3]+dp[j-1][1]+dp[j-1][5])%mod;
    	   dp[j][3]=(dp[j-1][3]+dp[j-1][6]+dp[j-1][2])%mod;
    	   dp[j][4]=(dp[j-1][4]+dp[j-1][7]+dp[j-1][5]+dp[j-1][1])%mod;
    	   dp[j][5]=(dp[j-1][5]+dp[j-1][8]+dp[j-1][4]+dp[j-1][6]+dp[j-1][2])%mod;
    	   dp[j][6]=(dp[j-1][6]+dp[j-1][5]+dp[j-1][9]+dp[j-1][3])%mod;
    	   dp[j][7]=(dp[j-1][7]+dp[j-1][4]+dp[j-1][8])%mod;
    	   dp[j][8]=(dp[j-1][8]+dp[j-1][7]+dp[j-1][0]+dp[j-1][9]+dp[j-1][5])%mod;
    	   dp[j][9]=(dp[j-1][9]+dp[j-1][8]+dp[j-1][6])%mod;
    	   	for(i=0;i<10;i++)
    	   a[j]=(a[j]+dp[j][i])%mod;
    	}
    	   cin >> t;
             while(t--)
    {
    	cin >> n;
    	cout << a[n] << "\n";
    }
    return 0;
}
