n = int(input())
li = []
for i in range(31):
	li.append(0)
for i in range(n):
	di = raw_input().split()
	if(di[0]=='G:') :
		for j in di:
			if j.isdigit(): li[int(j)]+=2
	if(di[0]=='M:'):
		for j in di:
			if j.isdigit(): li[int(j)]+=1
si = sorted(li,reverse=True)
if si[0]==si[1]: print("No Date")
elif (si[0]==li[19] or si[0]==li[20]): print("Date")
else : print("No Date")
