#include<iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string p,f;
		cin>>p>>f;
		
		int st[26];
			for(int i=0;i<26;i++)
				st[i]=0;                                                       //memset(st,0,sizeofrepeat)
		
		for(int i=0;i<f.size();i++)
		{
			st[f[i] - 'a']++;
			
		}
		
		 for(int i = 0; i < 26; i++){                                        // loop over characters in the given order

            int num = st[ p[i] - 'a' ];                               // print the (ith) character the number of times it is repeated

            for(int j = 0; j < num; j++) 
				
				cout << p[i];
        }
        cout<<"\n";

	}
}
