#include <iostream>
#include<vector>
#include<string>
using namespace std;

int main()
{
    string s;
    cin>>s;
    int n=s.size();
    vector< vector<int> > v(n,vector<int>(n,0));
    int c=0;
    for(int i=0;i<n;i++)
    {
    	c++;
    	for(int j=0;j<=i;j++)
    	v[i][j]=1;
    }
    for(int i=n-1;i>=0;i--)
    {
    	for(int j=n-1;j>i;j--)
    	{
    		if(s[i]==s[j] && v[i+1][j-1]==1)
    		{
    			v[i][j]=1;
    			c++;
    		}
    	}
    }
    cout<<c<<endl;
}
