#include<iostream>
#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll power[62];
void pre(){
 
 power[0]=1;
 for(int i=1;i<=60;i++){
  power[i]=2*power[i-1];
 }
}
int main()
{
    int n;
   pre();
   scanf("%d",&n);
    vector<ll> arr;
    arr.resize(n);
    for(int i=0;i<n;i++){ scanf("%lld",&arr[i]);}
    int bit[61],bit1[61];
    memset(bit,0,sizeof(bit));
    memset(bit1,0,sizeof(bit1));
    unsigned long long temp;
    for(int i=0;i<n;i++){
     for(int j=0;j<61;j++){
        temp=power[j];
        if(arr[i]&temp)bit[j]++;
     }
    }
    int OR[61];
    memset(OR,0,sizeof(OR));
    for(int i=0;i<n;i++){
      memset(bit1,0,sizeof(bit1));
      for(int j=0;j<61;j++){
        temp=power[j];
        if(arr[i]&temp)bit1[j]++;
      }
     for(int j=0;j<61;j++) bit[j]-=bit1[j];

     for(int j=0;j<61;j++){
      if((bit1[j]&&bit[j]==n-1)||(bit1[j]==0&&bit[j]==0)){
      }else
       OR[j]=true;
     }
    for(int j=0;j<61;j++) bit[j]+=bit1[j];
    }
    ll ans=0;
    for(int i=0;i<61;i++){
     if(OR[i]) ans+=power[i];
    }
    printf("%lld\n",ans);
    return 0;
}
