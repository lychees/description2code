void solve()
{
	string a, b;
	cin>>a>>b;
//	assert(a.size() <= b.size());
	if(a.size() > b.size()) {
		string t = a;
		a = b;
		b = t;
	}
	if(a.size() == b.size()) {
		int cnt = 0;
		for(int i=0;i<a.size();i++)
			if(a[i] != b[i])
				cnt++;
		cout<<cnt<<"\n";
	}
	else {
		int min = 999999999;
		for(int i=0;i<=(b.size()-a.size());i++) {
			int cnt = 0;
			for(int j=0;j<a.size();j++) {
				if(a[j] != b[i+j])
					cnt++;
			}
			if(cnt < min)
				min = cnt;
		}
		cout<<min<<"\n";
	}
}
