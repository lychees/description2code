#include <iostream>
#include <stdio.h>
using namespace std;

int ispalin(int a)
{
	int b,d=0;
	for(int i=a;i>0;i=i/10)
	{
		b=i%10;
		d=d*10+b;
		
	}
	if(a==d)
	  return 1;
	else
	  return 0;
}

int main()
{
    int a,b,t,count=0;
    scanf("%d",&t);
    for(int i=0;i<t;i++)
    {
    	int a,b;
    	scanf("%d%d",&a,&b);
    	for(int j=a;j<=b;j++)
    	{
    		if(ispalin(j))
    		   count++;
    	}
    	printf("%d\n",count);
    	count=0;
    }
    return 0;
}
