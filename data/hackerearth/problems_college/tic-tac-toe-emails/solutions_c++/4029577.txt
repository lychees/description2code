#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    scanf("%d",&n);
    while(n--)
    {
        string s,t;s.clear();
        int x;
        scanf("%d",&x);
        getline(cin,s);
        int i=0;
        set<string> a;
        set<string> ::iterator it;
        a.clear();

        while(i<s.length())
        {
            t.clear();
            while(!isspace(s[i])&&i<s.length())
            {
                t+=s[i];
                i++;
            }
            it=a.find(t);
            if(it==a.end())
            {
                cout<<t<<" ";
                a.insert(t);
            }
            i++;
        }
        printf("\n");
    }
    return 0;
}
