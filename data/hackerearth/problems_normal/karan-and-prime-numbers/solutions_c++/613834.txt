#include<iostream>
#include<cmath>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int a,b,flag;
		scanf("%d %d",&a,&b);
		int sum=0;
		if (a==1)
		a=2;
		for(int i=a;i<=b;i++)
		{  flag=0;
			for(int j=2;j<=sqrt(i);j++)
	    {
	    
			if(i%j==0)
			{
		      flag=1; 
		    }
		}
		if(!flag)
		sum+=i;
	    }
	printf("%d\n",sum);
   }
}
