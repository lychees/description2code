#include <iostream>
using namespace std;

int main()
{
    long long t;
    cin>>t;
    string str;
    cin>>str;
    t--;
    long long index;
    string temp;
    string value;
    while(t--) {
    	
    	cin>>index;
    	cin>>temp;
    	value=str.substr(0,index)+temp;
    	str = value;
    }
    cout<<value;
    return 0;
}
