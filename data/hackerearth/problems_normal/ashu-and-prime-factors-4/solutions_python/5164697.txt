a=[int(x) for x in xrange(0,10**6+1)]
length=len(a)
dic={}
for i in xrange(2,length-1):
    count=1
    if a[i]!=0:
        j=2*a[i]
        while j<10**6+1:
            if a[j]!=0:
                a[j]=0
                count+=1
            j=j+i
        dic[i]=count
 
n=input()
for i in xrange(0,n):
	a=input()
	print dic[a]
