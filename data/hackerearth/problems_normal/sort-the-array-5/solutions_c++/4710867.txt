#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

bool cmp1(pair<int,int>a,pair<int,int>b)
{
	if(a.first>b.first)
		return true;
	else if(a.first<b.first)
		return false;
	else 
	{
		if(a.second<=b.second)
			return true;
		else
			return false;
	}
}
int main()
{
	int n,k;
	cin>>n>>k;
	vector<pair<int,int>> vec;
	vec.resize(n);
	for(int i=0;i<n;i++)
	{
		cin>>vec[i].second;
		vec[i].first=vec[i].second%k;
	}
	sort(vec.begin(),vec.end(),cmp1);
	for(int i=0;i<n;i++)
		cout<<vec[i].second<<" ";
	cout<<endl;
	
}
