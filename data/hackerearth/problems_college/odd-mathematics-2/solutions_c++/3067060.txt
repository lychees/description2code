#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int n,i,j;
		cin>>n;
		long long sum=0;
		while(n!=0)
		{
			if(n%2 ==1)
			{
				j = n/2+1;
			}
			else
			{
				j = n/2;

			}
			sum = sum + j*j;
			n = n-j;
		}
		cout<<sum<<"\n";
	}
}
