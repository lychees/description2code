t = input()


def pw(a, b, k):
    if b == 0:
        return 1
    if b == 1:
        return a % 10**k
    if b % 2 == 0:
        x = pw(a, b/2, k) % 10**k
        return x*x % 10**k
    else:
        x = pw(a, b/2, k) % 10**k
        return x*x*a % 10**k


while t > 0:
    a, b, k = map(int, raw_input().split())
    x = pw(a, b, k) % 10**k
    c = 0
    for i in str(x):
        c = c+int(i)
    print c
    t = t-1
