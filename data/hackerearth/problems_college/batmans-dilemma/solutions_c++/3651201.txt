#include <bits/stdc++.h>

using namespace std;

#define pb push_back
#define mp make_pair
#define ll long long int
#define s(x) scanf("%d", &x)
#define sl(x) scanf("%lld", &x)
#define sd(x) scanf("%lf", &x)
#define mod 1000000007
#define get getchar_unlocked

ll fac[10000005];
int mrk[100005];

int main()
{
    int t, n, m, i, j;
    fac[0] = 1;
    ll ans;
    for (i = 1; i <= 10000000; ++i)
        fac[i] = (fac[i-1] * i)%mod;
    s(t);
    while (t--) {
        s(n);
        s(m);
        memset(mrk, 0, sizeof(mrk));
        for (i = 1; i <= n; ++i) {
            for (j = 1; j <= m; ++j)
                ++mrk[i|j];
        }
        ans = 1;
        for (i = 1; i <= 100000; ++i)
            ans = (ans * fac[mrk[i]])%mod;
        printf("%lld\n", ans);
    }
    return 0;
}
