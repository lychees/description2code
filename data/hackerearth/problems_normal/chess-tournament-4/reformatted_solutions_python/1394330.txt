n = input()
players = range(0, 2**n)
pow = []
for i in range((2**n)-1):
    pow.append([int(x) for x in raw_input().strip().split()])
for i in range(n):
    nextp = [0]*(len(players)/2)
    for j in range(0, len(players), 2):
        p1, p2 = players[j+1], players[j]
        if p1 < p2:
            p1, p2 = p2, p1
        x = pow[p1-1][p2]
        if x == 0:
            nextp[j/2] = p2
        else:
            nextp[j/2] = p1
    players = nextp
print players[0]+1
