#include <iostream>
using namespace std;

void compute()
{
	int n;
	cin>>n;
	int *arr = new int[n];
	int *dp = new int[n];
	int i;
	for(i=0;i<n;i++)
	{
		cin>>arr[i];
		dp[i]=0;
	}
	int j=0,count=0,temp;
	while(count<n)
	{
		dp[j] = -2;
		if(arr[j]==-1)
		{
			cout<<"Win\n";
			break ;
		}
		temp  = arr[j];
		if(dp[temp] == -2)
		{
			cout<<"Lose\n";
			break ; 
		}
		j = temp;
		count++;
	}
	//cout<<"Lose\n";
}

int main()
{
    int t;
    cin>>t;
    while(t>0)
    {
    	compute();
    	t--;
    }
    return 0;
}
