for _ in xrange(input()):
	r, s, n = map(int, raw_input().split())
	lst = sorted(map(int, raw_input().split()))
	cal = 0	
	mx = max(r,s)
	for i in xrange(n):
		cal += lst[i]
		if cal > mx:
			cal -= lst[i]
			break
	
	rb = False
	if mx == r:
		rb = True
	
	if rb:
		print "Tie" if cal <= s else "Raghu Won"
	else:
		print "Tie" if cal <= r else "Sayan Won"
