#include <iostream>
using namespace std;

int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
    	int n;
    	cin>>n;
    	unsigned long int sum=0;
    	for(int i=0;i<n;i++)
    	{
    		unsigned long int a;
    		cin>>a;
    		sum= sum+a-1;
    	}
    	sum=sum+1;
    	cout<<sum<<endl;
    
    }
    return 0;
}
