def primefactors(x):
    factorlist = []
    loop = 2
    while loop <= x:
        if x % loop == 0:
            x /= loop
            factorlist.append(loop)
        else:
            loop += 1
    return factorlist


def out(factors):
    k = 1
    while (len(factors) != 0):
        a = factors[0]
        b = factors.count(a)
        k = k*(b+1)
        factors[:] = (value for value in factors if value != a)
    print k-1
