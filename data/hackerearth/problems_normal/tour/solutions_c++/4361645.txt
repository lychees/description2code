#include<bits/stdc++.h>
#define ll long int
using namespace std;
int main()
{
	map<string,ll> mp;
	string str,temp;
	vector<ll> cost[1002];
	ll n,p,i,j,ele;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>str;
		mp[str]=i;
	}
	for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
		{
			cin>>ele;
			cost[i].push_back(ele);
		}
	}
	cin>>p;
	ll initial=0,sum=0;
	for(i=0;i<p;i++)
	{
		cin>>temp;
		ll val=mp[temp];
		sum+=cost[initial][val];
		initial=val;
	}
	cout<<sum<<endl;
	return 0;
}
