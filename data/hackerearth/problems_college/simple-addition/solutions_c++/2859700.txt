#include<bits/stdc++.h>
#define sc(n) scanf("%d",&n)
#define pr(n) printf("%d\n",n)
#define N 2147483647
using namespace std;
int cal(int a,int b)
{
    int count=0;
    while(a!=b)
    {
        int tmp=a-b;
        a=max(tmp,b);
        b=min(tmp,b);
        count++;
    }
    if(a==1)
        return count;
    else
        return N;
}
int main()
{
    int t,l,ans;
    sc(t);
    while(t--)
    {
        sc(l);
        ans=N;
        if(l==1)
        {
            pr(0);
            continue;
        }
        for(int i=1;i<l;i++)
            ans=min(ans,cal(l,i));
        pr(ans);
    }
    return 0;
}
