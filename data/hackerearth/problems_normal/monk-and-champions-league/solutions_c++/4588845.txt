#include <iostream>
#include <queue>
using namespace std;

int main()
{
	int i, m, n, a;
	long long int sum = 0;
    priority_queue<int> pq;
    cin>>m>>n;
    for(i=0; i<m; i++)
    {
    	cin>>a;
    	pq.push(a);
    }
    i = 0;
    while(i<n && !pq.empty())
    {
    	a = pq.top();
    	pq.pop();
    	sum += a;
    	a = a-1;
    	if(a>0)
    	{
    		pq.push(a);
    	}
    	i++;
    }
    cout<<sum<<endl;
    return 0;
}
