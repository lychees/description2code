#include <bits/stdc++.h>
using namespace std;
#define SIZE 199999999
int num[SIZE];
int N = 0;
void read(void) {
	scanf("%d", &N);
	return;
}
 
int used[SIZE];
 
void solve(void) {
	int i, j;
	for(j = 0, i = 0; j < N; i++)
		if(used[i] >= 2)
			j++;
 
	printf("%d\n", i - 1);
 	return;
}
 
void genNumbs(void) {
	for(int i = 1; i <= 100; i++)
		for(int j = i + 1; j <= 100; j++)
			used[i * i * i + j * j * j]++;
	
	return;
}
 
int main(void) {
	genNumbs();
	int T;
	scanf("%d", &T);
	for(int i = 0; i < T; i++) {
		read();
		solve();
	}	
	return 0;
}
