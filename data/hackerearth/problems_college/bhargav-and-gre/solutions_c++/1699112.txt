#include <bits/stdc++.h>
using namespace std;
int main()
{
    long long int tc,coun=0,coun1=0;
    string main;
    cin>>main;
    cin>>tc;
    while(tc--)
    {
        string s;
        cin>>s;
        if((main[0] == s[0]) && (main.length() == s.length()))
        {
            coun++;
        }
        else if((main[0]-'a'+1)+3 == (s[0]-'a'+1))
        {
            if(main.length() == s.length())
                coun1++;
        }
    }
    cout<<coun<<" "<<coun1<<endl;
}
