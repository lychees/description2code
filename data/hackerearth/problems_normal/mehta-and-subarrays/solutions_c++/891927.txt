#define pb push_back
#define mp make_pair
#define gl(x) scanf("%lld", &x)

using namespace std;

typedef long long LL;

LL p[1000002];
vector< pair<LL, LL> > t;

int main()
{
   // freopen("vats.txt", "r", stdin);
    LL n;
    cin>>n;
    LL s = 0;
    t.pb(mp(0, 0));
    for (LL i=1;i<=n;i++){
        cin>>p[i];
        s += p[i];
        t.pb(mp(s, i));
    }
    sort (t.begin(), t.end());
   /* for (LL i=0;i<t.size(); i++){
        cout<<t[i].first<<" "<<t[i].second<<"\n";
    }*/

    LL len = 0, c = 0;
    LL mn = t[0].second;

    for (LL i=1;i<t.size();i++){
        LL val = t[i].second - mn;
        if (val > 0){
            if (val > len){
                len = val;
                c = 1;
            } else if (val == len){
                c++;
            }
        }
        mn = min (mn, t[i].second);
    }
    if (c == 0){
        cout<<"-1\n";
    } else {
        cout<<len<<" "<<c<<"\n";
    }
    return 0;
}
