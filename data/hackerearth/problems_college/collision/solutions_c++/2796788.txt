#include <iostream>
using namespace std;

int main()
{
  int t;
  cin>>t;
  
  while(t--){
  	int n,x;
  	cin>>n;
  	
  	int a[10];
  	for(int i=0; i<10; i++){
  		a[i] = 0;
  	}
  	
  	int count = 0;
  	for(int i=0; i<n; i++){
		cin>>x;
		if(++a[x%10] > 1){
			count++;
		}
	}
	cout<<count<<endl;  		
  }
    return 0;
}
