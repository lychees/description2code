#include <iostream>
#include <cmath>
using namespace std;

int main()
{
	long long int t,t1,df1,df2;double n,tm;
	cin>>t;
	while(t--)
	{
		cin>>n;
		tm=t1=sqrt(n);
		if(tm*tm==n)cout<<"YES"<<endl;
		else
		{
		   	df1=abs(tm*tm-n);
		   	df2=abs((tm+1)*(tm+1)-n);
		   	if(df1>df2)cout<<"+"<<abs((tm+1)*(tm+1)-n)<<endl;
		   	else cout<<tm*tm-n<<endl;
		}
	}
    //cout << "Hello World!" << endl;
    return 0;
}
