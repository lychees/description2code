#include<bits/stdc++.h>
using namespace std;
int tests;
long long a,b,c;
int solve(long long num,int player)
{
	if (num<=1)
		return (player==1);
	if (player==2)
		return solve((num-1)/b+1,3-player);
	else
		return solve((num-1)/a+1,3-player);
}
int main(){
cin>>tests;
for (;tests;--tests)
{
	cin>>a>>b>>c;
	int res=solve(c,2);
	if(res)
		cout<<"Adam"<<endl;
	else
	
		cout<<"Bob"<<endl;
}
return 0;}
