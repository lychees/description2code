#include<bits/stdc++.h>
using namespace std;

#define LEN 100099
#define MAX 10004
#define SIZE 15
#define CTR 254

int main() {
	int t;


	cin >> t;
	while (t--) {
			int n;
	int crrScore;
	int maxScore;
	int maxInd;
	int maxSize;
//	char team [MAX][SIZE];
	char s[LEN];
	int freq[CTR];
	int len;
	int i;
	int j;
		// input string
		cin >> n;
		cin >> s;

		memset(freq, 0, sizeof(freq));
		len = strlen(s);

		// frequency map
		for (i = 0; i < len; i++) {
			freq[s[i]]++;
		}

		// input teams

		// calculate max score team
		maxScore = -1;
		maxSize = 1e9;
		for (i = 0; i < n; i++) {
			char team[4233];
			// find score of ith team
			cin >> team;
			len = strlen(team);
			crrScore = 0;

			for (j = 0; j < len; j++) {
				crrScore += freq[team[j]];
			}

			if (crrScore > maxScore) {
				maxScore = crrScore;
				maxInd = i;
				maxSize = len;
			} else if (crrScore == maxScore && len < maxSize) {
				maxScore = crrScore;
				maxInd = i;
				maxSize = len;
			}

		}
		printf("%d\n", (maxInd + 1));
	}
	
	return 0;
}
