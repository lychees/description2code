n=raw_input()
n=int(n)

l=raw_input().split(" ")
x=raw_input()
x=int(x)

flag=False

for i in range(0,n):
	for j in range(i+1,n):
		for k in range(j+1,n):
			if((int(l[i])+int(l[j])+int(l[k])) == x):
				flag=True
				break;
	
print str(flag)+"\n"
