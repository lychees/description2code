#include <iostream>
using namespace std;

int main()
{
    long long t,n1,n2,co,r1,r2;
    cin>>t;
    while(t--)
    {co=0;
    	cin>>n1>>n2;
    	while(n1>0||n2>0)
    	{
    		r1=n1%10;
    		r2=n2%10;
    		if(r1+r2>=10)
    		{//cout<<r1+r2<<"\n";
    			co++;
    		}
    		n1=n1/10;
    		n2=n2/10;

    	}
    	if(co==0)
    	{
    		cout<<"No carry operation\n";
    	}
    	else if(co==1)
    	{
    		cout<<"1 carry operation\n";
    	}
    	else
    	{
    		cout<<co<<" carry operations\n";
    	}
    }
    return 0;
}
