#include <iostream>
using namespace std;
int hasha[1000000];
int main()
{
	int t,n,a,r,count=0,max=0;
    cin>>t;
    while(t--)
       {
       	cin>>n;

       	for(int i=0;i<n;i++)
       	     {
       	     	cin>>a>>r;
       	     	if(r==1)
       	     	    {
                    if(hasha[a]!=1)
                    {
       	     	    	count++;
       	     	    	if(count>max)
       	     	    	max=count;
       	     	    	}
       	     	    hasha[a]=1;

       	     	    }
       	     	 else
       	     	   {
       	     	   	count=0;

       	     	   }


       	     }
       		cout<<max<<"\n";
       	max=count=0;
       for(int i=0;i<1000000;i++)
       hasha[i]=0;
       }
    return 0;
}
