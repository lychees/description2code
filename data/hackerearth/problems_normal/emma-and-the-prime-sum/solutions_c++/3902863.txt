#include <bits/stdc++.h>
#define max 1000001
using namespace std;

int main()
{
	int i,j;
	long long a[max]={0};
	for(i=2;i<=sqrt(max);i++)
	{
		for(j=i*i;j<max;j+=i)
		{
			a[j]=-1;
		}
	}
	long long k=0;
	for(i=2;i<max;i++)
	{
		if(a[i]==0)
		{
			a[i]=i+k;
			k=a[i];
		}
		else
		a[i]=a[i-1];
	}
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int x,y;
		long long ans;
		scanf("%d%d",&x,&y);
		ans=a[y]-a[x-1];
		printf("%lld\n",ans);
	}
    return 0;
}
