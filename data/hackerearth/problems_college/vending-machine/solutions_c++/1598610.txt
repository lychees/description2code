#include <bits/stdc++.h>

using namespace std;

int main()
{
	map <int, string> m;
	m[1] = "Sheldon";
	m[2] = "Leonard";
	m[3] = "Penny";
	m[4] = "Rajesh";
	m[5] = "Howard";
	long long n,val=5;
	cin >> n;
	if ( n <= 5 ) {
		cout << m[n] << endl;
		return 0;
	}
	long long cnt=2,val1=5;
	while ( 1 ) {
		val *= 2LL;
		val1 += val;
		if ( val1 >= n ) break;
        cnt *= 2LL;
	}
	n = n - (val1 - val);
	for ( long long i = 1; i <= 5; i++ ) {
		if ( cnt*i >= n ) {
			cout << m[i] << endl;
			break;
		}
	}
	return 0;
}
