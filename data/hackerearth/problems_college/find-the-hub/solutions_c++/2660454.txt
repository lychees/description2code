#include<stdio.h>
#include<iostream>
#include<string.h>
#include<math.h>
#include<algorithm>
#include<stdlib.h>
#include<vector>

using namespace std;

int a[100000];
int count(int x,int n)
{
	int i,ct=0;
	for(i=0;i<n;i++)
	{
		if(a[i]==x)
		ct++;
	}
	return ct;
}

main()
{
	 int i,j,n,max=0;
	scanf("%d",&n);
	
	for(i=0;i<n;i++)
	scanf("%d",&a[i]);
	
	int flag=0;
	
	for(i=0;i<n;i++)
	{
		int t=a[i]/2;
		if(a[i]%2!=0)
		{
		for(j=2;j<=t;j++)
		{
			if(a[i]%j==0)
			{
				flag=1;
				break;
			}
		
		}
		if(flag==0)
		{
			if(a[i]>max)
			max=a[i];
		}
		}
		flag=0;
	}
	printf("%d\n",max);
	printf("%d\n",count(max,n));
}
