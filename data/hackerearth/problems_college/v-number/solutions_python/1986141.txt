n=input()
for i in range(n):
	x=raw_input()
	l=len(x)
	a=0
	b=0
	for y in range(l/2):
		a+=(int)(x[y])
		b+=(int)(x[l-1-y])
	if a==b:
		print 'YES'
	else:
		print 'NO'
