#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
   	int TC;
   	cin>>TC;
   	for(int x=0;x<TC;x++)
   	{
   		int N;
   		cin>>N;
   		int S[N],A[N];
   		for(int i=0;i<N;i++)	cin>>S[i];
   		for(int i=0;i<N;i++)	cin>>A[i];
   		sort(S,S+N);
   		sort(A,A+N);
   		int flag=1;
   		for(int i=0;i<N;i++)
   		{
   		
   			if(A[i]>S[i])
   			{
   				flag=0;
   				break;
   			}
   		}
   		
   		if(flag==1)	cout<<"YES"<<endl;
   		else	cout<<"NO"<<endl;
   	}
    return 0;
}
