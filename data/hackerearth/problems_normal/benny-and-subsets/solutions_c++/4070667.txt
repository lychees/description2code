#include <bits/stdc++.h>
#define ll long long
#define mod 10000007
using namespace std;
ll int n,m,k,x;
ll int a[1010],b[1010],c[1010];
ll int dp[1010][1ll<<11];
ll int d[21],h[1ll<<21];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	memset(d,0,sizeof(d));
	memset(h,0,sizeof(h));
	cin>>n>>x;
	ll int i,j,ans;
	m=0;k=0;
	for(i=0;i<n;i++)
	cin>>a[i];
	for(i=0;i<n;i++)
	{
		if(a[i]>1024)
		{
			b[k++]=a[i];
		}
		else
		{
			c[m++]=a[i];
		}
	}
	for(i=0;i<=m;i++)
	{
		for(j=0;j<=((1ll<<11)-1);j++)
		{
			if(i==0)
			{
				if(j==0)
				dp[i][j]=1;
				else
				dp[i][j]=0;
			}
			else
			dp[i][j]=(dp[i-1][j^c[i-1]] +dp[i-1][j])%mod;
		}
	}
	h[0]=1;
	for(i=1;i<(1ll<<k);i++)
	{
		for(j=(k-1);j>=0;j--)
		{
			d[j]=(d[j]+1)%2;
			if(d[j])
			break;
		}
		ans=0;
		for(j=0;j<k;j++)
		{
			if(d[j])
			ans=ans^b[j];
		}
		h[ans]++;
	}
	ans=0;
	for(i=0;i<=((1ll<<11)-1);i++)
	{
		ans=(ans + ((dp[m][i] * h[x^i])%mod))%mod;
	}
	cout<<ans<<endl;
	return 0;
}
