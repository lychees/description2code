#include <stdio.h>

int main()
{
    int T,a,m,ans,k;
    scanf("%d",&T);
    while(T--)
    {
        scanf("%d%d",&a,&m);
        if(!a || m==2)
        {
            puts("YES");
            continue;
        }
        k=(m-1)/2;
        ans=1;
        while(k)
        {
            if(k%2)ans=(long long)ans*a%m;
            a=(long long)a*a%m;
            k/=2;
        }
        puts(ans==1?"YES":"NO");
    }
    return 0;
}
