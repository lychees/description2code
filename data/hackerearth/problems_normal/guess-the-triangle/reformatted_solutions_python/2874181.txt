a, m = map(float, raw_input().split())
b, n = map(float, raw_input().split())
c, o = map(float, raw_input().split())
ans = []
ans.append(((a+c-b), (m+o-n)))
ans.append(((a+b-c), (m+n-o)))
ans.append(((b+c-a), (n+o-m)))
ans.sort(key=lambda x: (x[0], x[1]))
for x in ans:
    print '%0.4f %0.4f' % (x[0], x[1])
