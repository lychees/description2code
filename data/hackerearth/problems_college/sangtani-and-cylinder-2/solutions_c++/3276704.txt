#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t, n;
    cin>>t;
    while(t--) {
    	cin>>n;
    	double r = (double)pow(((double)(n/(4*acos(0.0)))), (double)(1.0*1/3));
    	double h = (double)(n/(2*acos(0.0)*r*r));
    	double s = 4*acos(0.0)*r*(r+h);
    	printf("%0.9lf\n", s);
    }
    return 0;
}
