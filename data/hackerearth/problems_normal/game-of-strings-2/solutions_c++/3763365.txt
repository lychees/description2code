#include <bits/stdc++.h>
using namespace std;

int pr[1000004] = {0};
string s;

int kmp(int pos){
	int k=0,ma=0;
	for(int i=1;i<s.length();i++){
		while(s[i] != s[k] && k)
			k = pr[k-1];
		if(s[i] == s[k]){
			k++;
		}
		pr[i] = k;
		if(i<=pos){
			ma = max(ma,pr[i]);
		}
	}
	return ma;
}

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int p,len;
    	cin>>s;
    	cin>>p;
    	p--;
    	int k;
    	len = kmp(p);
    	k = pr[s.length() - 1];
    	while(k>len && k) k = pr[k-1];
    	if(k == 0) cout<<"Puchi is a cheat!"<<endl;
    	else{
    	for(int i=0;i<k;i++){
    		cout<<s[i];
    	}
    	cout<<endl;
    	}
    }
    return 0;
}
