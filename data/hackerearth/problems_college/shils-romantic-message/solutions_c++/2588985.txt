#include <iostream>
#include<string>
using namespace std;

int main()
{
    int t;
    cin>>t;
    string s;
    while(t--){
    	cin>>s;
    	int n = s[0]-'a';
    	for(int i=0;i<s.length();i++){
    		s[i]= ((s[i]-n));
    		if(s[i]<'a') s[i]+=26;
    	}
    	cout<<s<<endl;
    	
    }
}
