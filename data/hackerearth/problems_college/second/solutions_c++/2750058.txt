#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
	int n,m,ans = 0,*arr,i,j,key;
	cin >> n >> m;
	arr = new int[n];
	
	for(i=0;i<n;i++)
	{
		cin >> arr[i];
	}
	sort(arr,arr+n);
	j = 2;
	key = m;
	for(i=0;i<n;i++)
	{
		if(arr[i] == key)
		{
			key = m*j;
			j++;
		}
	}
	cout << key-m;
	return 0;
}
