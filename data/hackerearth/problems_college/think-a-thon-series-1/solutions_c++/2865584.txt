    #include<bits/stdc++.h>
 	  using namespace std;
    int func(int a,int b,int c){
        if(a*c>=b)
            return 0;
        
        int mx=a*sqrt((b/a));
        mx=max(mx,a*c);
        return (1+func(mx,b,c));
    }
     
    int main(){
        int t;
        cin>>t;
        for(int i=0;i<t;i++){
            int a,b,c;
            cin>>a>>b>>c;
            cout<<"Case #"<<i+1<<": "<<func(a,b,c)<<endl;
        }
    }
