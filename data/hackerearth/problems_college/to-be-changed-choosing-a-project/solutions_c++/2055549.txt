#include <bits/stdc++.h>
using namespace std;
int f(int n,int k) {
	vector<pair<int,int> > v;
	v.resize(n);
	for(int i=0;i<n;++i) {
		v[i].first=v[i].second=i;
	}
	int pos=0;
	for(int i=0;i<n-1;++i) {
		pos+=k-1;
		pos%=v.size();
		v.erase(v.begin()+pos);
	}
	return v[0].first;
}
int main() {
	ios_base::sync_with_stdio(0);
	cin.tie(0);
	int n,k;
	cin>>n>>k;
	cout<<f(n,k)+1<<"\n";
}
