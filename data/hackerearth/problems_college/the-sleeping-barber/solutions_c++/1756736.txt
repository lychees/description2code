#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		long long g,m;
		cin>>g>>m;
		long long arr[g];
		for(int i=0;i<g;i++){
			cin>>arr[i];
		}
		sort(arr,arr+g);
		long long count=0;
		long long temp=arr[0];
		int k=1;
		while(temp<=m){
			count++;
			temp+=arr[k++];
		}
		cout<<count<<endl;
	}
}
