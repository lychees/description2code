#include <bits/stdc++.h>
using namespace std;

#define f first
#define s second

#define pb push_back
#define mp make_pair

typedef long long ll;
typedef pair<int, int> ii;

int n;
vector<ii> profit, loss;

bool cmp(ii x, ii y) {
    return x.s > y.s || (x.s == y.s && x.f < y.f);
}

bool isOK(ll val) {
    for (int i = 0; i < profit.size(); i++) {
        if (val < profit[i].f) return false;
        else val += profit[i].s - profit[i].f;
    }

    for (int i = 0; i < loss.size(); i++) {
        if (val < loss[i].f) return false;
        else val += loss[i].s - loss[i].f;
    }

    return true;
}

ll bs(ll val) {
    ll res(-1);
    ll l = 1, h = val;
    while (l <= h) {
        ll m = (h - l)/2 + l;

        if (isOK(m)) res = m, h = m-1;
        else l = m+1;
    }
    return res;
}

int main() {
    //freopen("input.inp", "r", stdin);
    //freopen("output.out", "w", stdout);

    scanf("%d", &n);
    for (int i = 0; i < n; i++) {
        int t, x; scanf("%d%d", &t, &x);

        if (t <= x) profit.pb(mp(t, x));
        else loss.pb(mp(t, x));
    }

    sort(profit.begin(), profit.end());
    sort(loss.begin(), loss.end(), cmp);

    ll ans = bs(ll(1e10));
    printf("%lld", ans);
}
