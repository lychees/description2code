{

		dp[i]=dp[i-1];
		LL tmp=3*1LL*(i-1)*1LL*(i-1);
		dp[i]+=tmp;
		//trace(i,dp[i]);
		tmp=(3*i-2)*1LL*(2*i-1);
		dp[i]+=tmp;
		//trace(i,dp[i]);

		dp[i]+=tmp;
		//trace(i,dp[i]);
	}
	int t;

	{
		int n;
