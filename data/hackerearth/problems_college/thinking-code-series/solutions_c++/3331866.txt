    #include<bits/stdc++.h>
        using namespace std;
        int main()
        {
        	int t;
        	cin>>t;
        	for(int cc=1;cc<=t;cc++)
        	{
        		long long c,l,p;
        		cin>>l>>p>>c;
        		long x=l;
        		int k,i;
        		k=0;
        		i=0;
        		while(x<p)
        		{
        			x=x*c;
        			k++;
        		}
        		while((1<<i)<k) i++;
        		cout<<"Case #"<<cc<<": "<<i<<endl;
        	}
        	return 0;
        }
