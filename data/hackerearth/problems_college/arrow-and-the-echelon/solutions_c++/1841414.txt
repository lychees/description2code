
#include <iostream>
using namespace std;

int main()
{
    int n,t,k;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        string b;
        for(int j=n;j>=0;j--)
        {
            if(j%3==0 && (n-j)%5==0)
            {
                b="";
                for(int k=0;k<j;k++)
                    b+='5';
                for(int k=0;k<n-j;k++)
                    b+='3';
                break;
            }
        }
        if(b=="")
            cout<<"-1\n";
        else
            cout<<b<<endl;
    }
    return 0;
}
