#include <bits/stdc++.h>

#define mp make_pair
#define pb push_back
#define ll long long

using namespace std;

const int nm = 1010;

int m, n, a[nm][nm];
int up[nm][nm], r[nm][nm];

int main() {
#ifdef LOCAL
	freopen("input.txt", "r", stdin);
#endif
	scanf("%d%d", &m, &n);
	for (int i = 1; i <= m; ++i)
		for (int j = 1; j <= n; ++j)
			scanf("%d", &a[i][j]);
	for (int i = 1; i <= m; ++i) {
		for (int j = 1; j <= n; ++j) {
			if (a[i][j] == a[i - 1][j])
				up[i][j] = up[i - 1][j] + 1;
			else
				up[i][j] = 1;
		}
	}
	int res = 0;
	for (int i = 1; i <= m; ++i) {
		for (int j = n; j >= 1; --j) {
			if (a[i][j] == a[i][j + 1])
				r[i][j] = r[i][j + 1] + 1;
			else
				r[i][j] = 1;
			res = max(res, up[i][j] + r[i][j] - 1);
		}
	}
	printf("%d\n", res);
#ifdef LOCAL
	printf("\n%.3lf s\n", (double) clock() / CLOCKS_PER_SEC);
#endif
}
