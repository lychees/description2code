#include <iostream>
#include<string.h>
using namespace std;

int main()
{
	int t;
	char s[100];

	cin>>t;

	while(t--)
	{
		int i=0,j=0,start=0,count=0;

		cin>>s;
		j=strlen(s)-1;
		while(i<=j)
		{
			if(s[i]==s[j])
			{
				i++;
				j--;
				count++;
			}
			else
			{
				start++;
				i=start;
				j=strlen(s)-1;
				count=0;
			}

		}

		if(start==0)
			cout<<strlen(s)<<endl;
		else
		{
			/*if(strlen(s)%2==1)
			cout<<strlen(s)+start+1+count<<endl;
			else*/
				cout<<strlen(s)+start<<endl;
			
		}
		//cout<<"<"<<start<<">";
	}

    return 0;
}
