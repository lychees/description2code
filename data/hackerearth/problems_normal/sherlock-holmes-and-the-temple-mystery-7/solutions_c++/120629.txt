#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t!=0)
    {
    	int n,i,sum=0;
    	cin>>n;
    	int a[n];
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    		sum+=a[i];
    	}
    	if(sum%2==0)
    		cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;
		t-=1;
    }
    return 0;
}
