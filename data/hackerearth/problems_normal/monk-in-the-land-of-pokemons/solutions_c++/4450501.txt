#include<bits/stdc++.h>
using namespace std;

int main()
{
   int t,ans,A,B,N;
   ios::sync_with_stdio(false);cin.tie(0);cout.tie(0);
   cin>>t;
   
   while(t--) {
   	cin>>N;
   	ans = 0;map<int,int>have;
   	// fill(have.begin(),have.end(),0);
  	for(int i =0;i<N;i++) {
  		cin>>A>>B;if(A==B)continue;
  		if(!(have[B]))ans++;
  		else have[B]--;
  		have[A]++;
  	}
  	cout<<ans<<'\n';
   }
    return 0;
}
