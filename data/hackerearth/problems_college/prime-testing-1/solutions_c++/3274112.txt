#include <iostream>
using namespace std;
int checkprime(int N) {
    int count = 0;
    for( int i = 1;i * i <=N;++i ) {
         if( N % i == 0) {
         if( i * i == N )
                     count++;
                 else                                                     // i < sqrt(N) and (N / i) > sqrt(N)
                     count += 2;
      }
    }
    if(count == 2)
        return 1;
    else
        return 0;
}

int main()
{
    int t,n;
    cin>>t;
    while(t>0) {
    	int count=0,p;
    	cin>>n;
    	while(n>0) {
         	long long int a;
        	cin>>a;
         	p=checkprime(a);
        	if(p==1)
        	count++;
        	n--;
    	}
    	cout<<count<<endl;
    	t--;
    }
    return 0;
}
