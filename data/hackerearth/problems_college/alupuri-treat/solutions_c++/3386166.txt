#include <iostream>
#include <cmath>
using namespace std;

int main()
{
//    cout << "Hello World!" << endl;
  	int m;
  	cin>>m;
  	int a[m][m];
  	for(int i=0;i<m;i++)
  		for(int j=0;j<m;j++)
  			cin>>a[i][j];
  	int sum=0;
  	for(int i=0,j=m-1;i<m;i++,j--)
  		sum=sum+(a[i][i]-a[i][j]);
  	cout<<fabs(sum);
    return 0;
}
