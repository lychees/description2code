#include<iostream>
using namespace std;
int main(){
int t;
unsigned long long arr[1000003];
unsigned long long x,n,ans;
cin>>t;
	for(int i=0;i<t;i++){
	cin>>n>>x;
	if(i==0){
	arr[0]=1;
		for(int j=1;j<1000003;j++){
		   arr[j]=(j*arr[j-1])%1000003;
		}
	}
	
	if(n>=1000003)
	cout<<0<<endl;
	else{
		n=n%1000003;
		ans=((x%1000003)*(arr[n]%1000003))%1000003;
		cout<<ans<<endl;
	}
	}

return 0;
}
