#include <iostream>
#include <cstdio>
#include <string.h>

using namespace std;

int main()
{
   char msg[100];
   int i,j=0,k=0,m=0,l=0,flag=0;
   scanf("%s",msg);
   l=strlen(msg);
   for(i=0;i<l;i++)
   {
       if(msg[i]=='l')
       {
           for(j=i;j<l;j++)
           {
               if(msg[j]=='o')
               {
                   for(k=j;k<l;k++)
                   {
                       if(msg[k]=='v')
                       {
                           for(m=k;m<l;m++)
                           {
                               if(msg[m]=='e')
                                 {
                                   flag=1;
                                   break;
                                 }

                           }
                       }
                   }
               }
           }
       }

   }

   if(flag==0)
    printf("Let us breakup!");
   else
    printf("I love you, too!");
    return 0;
}
