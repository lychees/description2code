#include<iostream>
#include<stdio.h>

using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        long long int a,b,c,x,y,z,n,i,k,l,m;
        cin>>n;
        x=y=z=0;
        i=0;
        while(i<n)
        {
            cin>>a>>b>>c;
            k=a+min(y,z);
            l=b+min(x,z);
            m=c+min(x,y);
            x=k;
            y=l;
            z=m;
            i++;
        }
        cout<<min(x,min(y,z))<<endl;
    }
    return 0;
}
