#include <bits/stdc++.h>
using namespace std;
bool fun(long long int x,long long int y)
{
return x>y;
}
int main()
{
    long long int n;
    cin>>n;
    long long int a[n],b[n],i,k,count=0,j=0,temp,m,maxi;
    for(i=0;i<n;i++)
    cin>>a[i];
    for(i=0;i<n;i++)
    cin>>b[i];
    
    sort(a,a+n,fun);
    maxi=0;
    for(i=0;i<n;i++)
    {  
    for(j=0;j<n;j++)
       {
    	if(a[i]>=b[j]&&b[j]>maxi)
    	{    maxi=b[j];
    		temp=j;
    	continue;	
    	}
        }
        m=b[i];
        b[i]=b[temp];
        b[temp]=m;
        j=i+1;
        maxi=0;
        
    }
     /*for(i=0;i<n;i++)
    cout<<a[i]<<endl;
    for(i=0;i<n;i++)
    cout<<b[i]<<endl;*/
    for(i=0;i<n;i++)
    {
    	if(a[i]>b[i])
    	{
    		count=count+100;
    	}
    	
    	if(a[i]==b[i])
    	{
    		count=count+50;
    	}
    	
    }
    cout<<count<<endl;
    return 0;
}
