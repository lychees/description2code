#include <iostream>
using namespace std;

int main()
{
	long n,k,i,c=0;
	
		cin>>n>>k;
		long x[k],y[k];
		for(i=0;i<k;i++){
			cin>>x[i];
		}
		for(i=0;i<k;i++){
			cin>>y[i];
		}
		for(i=0;i<k;i++){
			if(x[i]!=1 && x[i]!=n){
			//	cout<<x[i]<<" "<<y[i];cout<<endl;
				if(x[i]-1<(n-x[i]))
				c+=x[i]-1;
				else
				c+=n-x[i];
			}
			if(y[i]!=1 && y[i]!=n){
				if(y[i]-1<(n-y[i]))
				c+=y[i]-1;
				else
				c+=n-y[i];
			}
		}
		cout<<c<<endl;
    
    return 0;
}
