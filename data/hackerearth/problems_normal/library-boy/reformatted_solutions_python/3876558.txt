N = int(raw_input())

books = {}
for x in range(N):
    book = raw_input()
    cnt = books.get(book[0], 0)
    books[book[0]] = cnt+1

numShelves = 0
for book in books:
    numBooks = books[book]
    reminder = 0 if numBooks % 10 == 0 else 1
    numShelves += books[book]/10 + reminder

print numShelves
