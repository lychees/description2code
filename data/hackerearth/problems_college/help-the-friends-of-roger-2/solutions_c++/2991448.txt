#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    char s[100];
    cin>>s;
    int len=strlen(s);
    int m;
    cin>>m;
    while(m--)
    {
    	int l,r,count=0;
    	cin>>l>>r;
    	for(int i=l-1;i<r-1;i++)
    	{
    		if(s[i]==s[i+1])
    		count++;
    	}
    	cout<<count<<endl;
    }
    return 0;
}
