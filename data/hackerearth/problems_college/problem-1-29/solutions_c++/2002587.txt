#include <bits/stdc++.h>

using namespace std;

vector <vector<int> >v;

bool getPath(int ll, int lr, int ul, int ur){
	if(ll==ul && lr==ur){
		return true;
	}	
	if(ll==ul){
		if(v[ll][lr+1]==1){
			return getPath(ll,ul,lr+1,ur);	
		}	else{
			return false;
		}
	}
	if(lr==ur){
		if(v[ll+1][lr]==1){
			return getPath(ll+1,ul,lr,ur);	
		}	else{
			return false;
		}
	}
	if(v[ll][lr]==0){
		return false;
	}
	return (getPath(ll,lr+1,ul,ur) || getPath(ll+1,lr,ul,ur) );
	}
	
int main()
{
    int t,n,x,i,j;
    bool result;
    scanf("%d",&t);
    vector<int> row;
    while(t--){
        v.clear();
    	scanf("%d",&n);
    	for(i=0;i<n;++i){
    		row.clear();
    		for(j=0;j<n;++j){
    			scanf("%d",&x);
    			row.push_back(x);	
    		}
    		v.push_back(row);
    	}
    	result=getPath(0,0,n-1,n-1);
        if(result){
    	    printf("POSSIBLE\n");
        }	else{
    	    printf("NOT POSSIBLE\n");
        }
    }
    return 0;
}
