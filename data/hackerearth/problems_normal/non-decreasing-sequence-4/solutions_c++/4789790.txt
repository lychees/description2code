#include<bits/stdc++.h>
#define N 1000000+1
using namespace std;
int s[N];
int main()
{
    int t,i;
    cin>>t;
    long long n;
    while(t--)
    {
        cin>>n;
        long long x,l;
        for(i=0;i<n;i++)
            s[i]=0;
       int  top=0;
        s[0]=-1;
        for(i=0;i<n;i++)
        {
            cin>>x;
            if(x>s[top])
            {
                s[++top]=x;
            }
            else
            {
                int low=1,high=top,mid;
                while(low<=high)
                {
                    mid=(low+high)/2;
                    if(s[mid]<x)
                    {
                        low=mid+1;
                    }
                    else
                        high=mid-1;
                }
                s[low]=x;
            }
//            for(int j=0;j<=top;j++)
//                cout<<s[j]<<" ";
//            cout<<"\n";
        }
        cin>>l;
        if(l>top)
            cout<<"-1"<<"\n";
        else
            cout<<s[l]<<"\n";
    }
    return 0;
}
