#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	string s1,s2;
    	cin>>s1>>s2;
    	int count =0;
    	int len = s1.length()>s2.length()?s2.length():s1.length();
    	for(int i=0;i<len;i++){
    		if(s1[i]==s2[i]) count++;
    	}
    	cout<<count<<endl;
    }
    return 0;
}
