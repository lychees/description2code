#include<iostream>
using namespace std;

#define lld long long int     
    
lld F[10000+5];

int main(){
	F[1] = 1;
	for(int i=2;i<=90;++i)
	{
		F[i] = F[i-1] + F[i-2];
	}	
//	rep(i,90,1003)F[i] = F[i-1];//INF

   //  cout<<"F[90]= "<<F[90]<<endl;

    for(int i=90; i<1003; i++)
    {
    	F[i]=F[i-1];
    }
    
    
	int T,N;
	lld K;
	cin>>T;
	
	while(T--)
	{
    	cin>>N>>K;
    	N++;
				
		if(F[N+1] < K){
			cout<<"-1\n";
            continue;
		}
		
		for(int i=N;i>=2;i--){
			if(F[i]<K){
				cout<<'Z';
				K -= F[i];
			}
			else{
				cout<<'O';
			}
		}
		cout<<endl; 
	}
}
