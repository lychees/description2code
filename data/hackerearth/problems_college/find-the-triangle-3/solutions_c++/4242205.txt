#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() { ios::sync_with_stdio(false); cin.tie(NULL);

    int t, n; long long a; cin>>t;
    while(t--){
		cin>>n;
		vector<long long> num;
		while(n--) {
			cin>>a;
			num.push_back(a);
		}
		sort(num.begin(), num.end());
		for(int i= num.size()-1; i>1; i--) {
			for(int j=i-1; j>0; j--) {
				
				int k=j-1;
				while(k>=0) {
					if( num[k] + num[j] > num[i]  ) {
						cout<<num[k] << " "<<num[j] <<" "<<num[i]<<endl;
						goto end;
					}
					if( num[k] + num[j] <= num[i]  ) break;
					k--;
				}
				
			}
		}
		cout<<-1<<endl;
		end: {}
    }
    return 0;
}
