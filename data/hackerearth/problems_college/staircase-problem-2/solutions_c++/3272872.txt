#include <iostream>
using namespace std;

int main()
{
	int t,i,n;
	cin>>t;
	for(i=0;i<t;i++)
	{
		cin>>n;
		int done=0,x=1,c=0;
		
		while(done==0)
		{
			if(n>=(x+1))
			{
			n=n-(x+1);
			if(n==0)
			done =1;
			else
			done=0;
				c++;
				x=x+1;
			}
			else
			{
				c++;
				done =1;
			}
			
			
			
		}
		cout<<c<<"\n";
		
		
	}
}
