#include <iostream>
using namespace std;

int main()
{
    long long int ev,od,n,test,a,i;
    cin>>test;
    while(test--)
    {
    	ev=0;
    	od=0;
    	cin>>n;
    	for(i=0;i<n;i++)
    	{
    		cin>>a;
    		if(a%2==0)
    		ev++;
    		else
    		od++;
    	}
    	cout<<ev*od<<endl;
    }
    return 0;
}
