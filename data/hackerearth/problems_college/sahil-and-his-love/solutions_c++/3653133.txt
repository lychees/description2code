#include<bits/stdc++.h>
using namespace std;
int lcs(char a[]){
    int ans=0;
    int x=strlen(a);
    int y=x;
    int dp[x+1][y+1];
    for(int i=0;i<=x;i++)
        dp[i][0]=0;
    for(int i=0;i<=y;i++)
        dp[0][i]=0;
    for(int i=1;i<=x;i++)
        for(int j=1;j<=y;j++){
        dp[i][j]=0;
        if(a[i-1]==a[x-j])
            dp[i][j]=max(dp[i][j],1+dp[i-1][j-1]);
        dp[i][j]=max(dp[i][j],max(dp[i-1][j],dp[i][j-1]));
    }
    return dp[x][y];
}
int main(){
	int t;
	scanf("%d\n",&t);
	while(t--){
		int ans=1;
		char m[1001],n[1001];
		scanf("%s",m);
		int x=strlen(m);
		for(int i=1;i<=x;i++){
			strcpy(n,m);
			for(int j=0;j<i/2;j++){
				swap(n[x-i+j],n[x-1-j]);
			}
			ans=max(ans,lcs(n));
		}
		printf("%d\n",ans);
	}
	return 0;
}
