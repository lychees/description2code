t = int(raw_input().strip())


P = [0]* 1000001

for i in xrange(2, 1000+1):
	j = 2
	while int(pow(i, j)) <= 1000000:
		pos = int(pow(i, j))
		P[pos] = 1
		j+=1

P[1] = 1

while t:
	t-=1
	
	x = int(raw_input().strip())
	
	f = 0
	for i in xrange(1, (x>>1)+1):
		if P[i] and P[x-i]:
			f = 1
			break
		
	if f:
		print "Yes"
	else:
		print "No"
