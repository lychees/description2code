#include <bits/stdc++.h>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
	int t;
	cin >> t;
	while (t--) {
		int n;
		cin >> n;
		int ans = 50;
		for (int i = 0; i < n; ++i) {
			int x;
			cin >> x;
			ans = min(ans, __builtin_popcount(x));
		}
		cout << ans << endl;
	}
	return 0;
}
