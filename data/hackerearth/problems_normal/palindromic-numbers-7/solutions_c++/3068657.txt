#include <iostream>
using namespace std;

bool is_palindrome(long num)
{
	long temp = num;
	long rev = 0;
	while(temp)
	{
		rev *= 10;
		rev += temp % 10;
		temp /= 10;
	}
	if(rev == num)
		return true;
	else
		return false;
}

int main()
{
    int test_case;
    long num1,num2,answer;
    cin >> test_case;
    while(test_case--)
    {
    	cin >> num1 >> num2;
    	answer = 0;
    	for(long  i = num1; i <= num2; ++i)
    	{
    		if(is_palindrome(i))
    			++answer;
    	}
    	cout << answer << endl;
    }
    return 0;
}
