#include <vector>
#include <iostream>
#include <sstream>
#include <algorithm>
using namespace std;


int main() {
   
    vector<long int> arr;
    long int tmp,n;
    cin >> n;
   
    while(cin >> tmp){
        arr.push_back(tmp);
       
    }
    long int dp[n+1][arr.size()],x,y;
    for(int i=0;i<arr.size();i++)
        dp[0][i]=1;
    for(int i=1;i<n+1;i++){
        for(int j=0;j<arr.size();j++){
            x=(i-arr[j]>=0)?dp[i-arr[j]][j]:0;
            y=(j>=1)?dp[i][j-1]:0;
            dp[i][j]=x+y;
        }
    }
    cout << dp[n][arr.size()-1] << endl;
}
