#include <bits/stdc++.h>
#define ll long long
using namespace std;
void myfind(string s,int flag)
{
	if(flag)
	cout<<s<<endl;
	else
	{
		if(s=="Chandu")
		cout<<"Daspal"<<endl;
		else
		cout<<"Chandu"<<endl;
	}
}
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int test;
	cin>>test;
	while(test--)
	{
		string s;
		int x,y;
		cin>>x>>y>>s;
		int flag=1;
		int t;
		while(1)
		{
			t=__gcd(x,y);
			x-=t;y-=t;
			if(x==0 || y==0)
			break;
			flag^=1;
		}
		myfind(s,flag);
	}
	return 0;
}
