#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long int a,sum,b=100,c,f=0;
    	cin>>a;
    	c=a;
    	while(b)
    	{
    		sum=0;
    		while(c)
    		{
    			sum+=c%10;
    			c/=10;
    		}
    		if(sum/10==0)
    		{
    			cout<<sum<<endl;
    			f=1;
    			break;
    		}
    		if(f==1)
    		break;
    		b=sum;
    		c=sum;
    	}
    }
    return 0;
}
