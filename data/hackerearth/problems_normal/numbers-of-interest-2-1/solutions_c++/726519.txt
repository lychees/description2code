#include <bits/stdc++.h>

using namespace std;

long long f[1000005];
long long g[1000005];

bool vis[1000005];

int main()
{
	for ( int i = 1; i <= 1000000; i++ ) {
		for ( int j = i; j <= 1000000; j += i ) {
			f[j] += i;			
		}
	}
	f[0] = f[1] = g[0] = g[1] = 0;
	
	for ( int i = 2; i <= 1000000; i++ ) {
		if ( !vis[i] ) {
			for ( int j = i; j <= 1000000; j += i ) {
				if ( !vis[j] ) g[j] = i, vis[j] = true;
			}
		}				
	}
	for ( int i = 2; i <= 1000000; i++ ) {
		f[i] = f[i-1] + f[i]-i;
		g[i] += g[i-1];
	}

	int t,n;
	cin >> t;
	while ( t-- ) {
		cin >> n;
		cout << (f[n] + g[n])%n << endl;
	}
	return 0;
}
