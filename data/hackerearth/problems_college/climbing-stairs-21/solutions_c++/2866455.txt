#include "bits/stdc++.h"
using namespace std;
typedef long long int ll;
int main()
{ios::sync_with_stdio(0);cin.tie(0);cout.tie(0);
    ll t,n,i,fib[1001];
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	fib[1]=1;fib[2]=2;
    	for(i=3;i<1001;i++) fib[i]=fib[i-1]+fib[i-2];
    	cout<<fib[n]<<endl;
    }
    return 0;
}
