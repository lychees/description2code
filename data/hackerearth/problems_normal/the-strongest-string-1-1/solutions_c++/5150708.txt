#include <iostream>
#include <deque>
#include <vector>
using namespace std;

int main()
{
	int n;
	cin >> n;

	string str;
	cin >> str;
	vector< vector<int> > indices(26);

	for( int i=0; i<str.size(); i++)
	{
		indices[ str[i] - 'a' ].push_back(i);
	}

	string ans;
	ans = "";
	ans.clear();

	int i = 25;
	int len = -1;

	while( i>= 0 )
	{
		if( indices[i].size() > 0 )
		{
			int j=0;
			while( j<indices[i].size() && indices[i][j] < len )
				j++;

			if( j!=indices[i].size() )
			{
				ans.push_back( 'a' + i );
				len = indices[i][j];
			}
		}
		i--;
	}

	cout << ans << endl;
	return 0;
}
