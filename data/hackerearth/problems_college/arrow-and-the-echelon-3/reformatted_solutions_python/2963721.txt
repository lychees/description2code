from sys import stdin
import sys

for i in xrange(int(stdin.readline().strip())):
    X = int(stdin.readline().strip())
    n = X / 3
    z = 0
    a = False
    j = n
    while j != -1:
        if (X - 3*j) % 5 == 0:
            z = 3*j
            a = True
            break
        else:
            j -= 1
    if a:
        sys.stdout.write(str(5) * z)
        sys.stdout.write(str(3) * (X-z))
    else:
        sys.stdout.write(str(-1))
    print
