void isSubsetSum(int set[], int n, int sum)
{
    bool subset[sum+1][n+1];
    for (int i = 0; i <= n; i++)
      subset[0][i] = true;
 
    for (int i = 1; i <= sum; i++)
      subset[i][0] = false;
     for (int i = 1; i <= sum; i++)
     {
       for (int j = 1; j <= n; j++)
       {
         subset[i][j] = subset[i][j-1];
         if (i >= set[j-1])
           subset[i][j] = subset[i][j] || subset[i - set[j-1]][j-1];
       }
     }
 
    // uncomment this code to print table
    int c=0;
     for (int i = 0; i <= sum; i++)
     {
       //for (int j = 0; j <= n; j++)
          if(subset[i][n]==true)
          	c++;
     } 
 	cout<<c<<"\n";
}
int main()

	int t;

		for(i=0;i<n;i++)
		{
			cin>>a[i];
			sum+=a[i];
		}
		isSubsetSum(a,n,sum);
