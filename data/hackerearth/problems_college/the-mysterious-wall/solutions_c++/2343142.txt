#include <stdio.h>
#include<string.h>
int res(int ca[],int cb[26])
{
	int i;
	for(i=0;i<26;i++)
	{
		if(ca[i]!=cb[i])
		return 0;
	}
	return 1;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	char a[100009],b[100009];
    	 int i,j,ca[26],l1=0,l2=0,cb[26],flag=0;
    	scanf("%s %s",a,b);
    	l1=strlen(a);
    	l2=strlen(b);
    	for(i=0;i<26;i++)
    	{
    		ca[i]=0;
    		cb[i]=0;
    	}

    	if(l2>l1)

    	 printf("NO\n");

    	else
    	{
        	for(i=0;i<l2;i++)
    	   {
    		cb[b[i]-'a']++;
    		ca[a[i]-'a']++;
    	    }
    	    if(res(ca,cb))
    	      printf("YES\n");
    	    else
    	    {

    	       for(i=l2;i<l1;i++)
    	       {
    		      ca[a[i-l2]-'a']--;
    		      ca[a[i]-'a']++;
    		      if(res(ca,cb))
    		      {
    		      	flag=1;
    		      	break;
    		      }


    	       }
            	if(flag==1)
    	    printf("YES\n");
    	    else
    	    printf("NO\n");
    	    }

    	}
    }
    return 0;
}
