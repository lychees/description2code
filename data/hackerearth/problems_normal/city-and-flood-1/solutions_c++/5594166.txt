#include<bits/stdc++.h>
using namespace std;
int arr[100005];
int root(int a)
{
	
	while(arr[a]!=a)
	{
		a=arr[a];
	}
	return a;
}
void unio( int a,int b)
{
	int root_a=root(a);
	int root_b=root(b);
	arr[root_a]=root_b;
}

int main()
{
	long a;
	cin>>a;
	arr[a+1];
	for(long i=0;i<=a;i++)
	{
		arr[i]=i;
	}
	long  b;
	cin>>b;
	while(b--)
	{
		long c,d;
		cin>>c>>d;
		unio(c,d);
	}
long	count=0;
/*for(long g=1;g<=a;g++)
{
	cout<<arr[g]<<endl;
}*/
	for(long j=1;j<=a;j++)
	
	{
		if(arr[j]==j)
		count++;
	}
	cout<<count<<endl;
}
