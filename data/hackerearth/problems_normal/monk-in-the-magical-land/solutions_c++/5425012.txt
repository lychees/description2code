#include <bits/stdc++.h>
using namespace std;
#define fast ios::sync_with_stdio(false); cin.tie(0); cout.tie(0)
int gcd(int a, int b){return b==0? a : gcd(b,a%b);}
int main(){
	fast;
	int t; scanf("%d",&t);
	while(t--){
		int n,m,k,x[22],c[111],z[111],ans=0,test;
		__int128 mark[22]={0},bit=1;
		scanf("%d%d%d",&n,&m,&k);
		for(int i=0;i<n;i++) scanf("%d",x+i);
		for(int j=0;j<m;j++) scanf("%d",c+j);
		for(int j=0;j<m;j++) scanf("%d",z+j);

		for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
		if(gcd(x[i],c[j])!=1) mark[i]|=(bit<<j);
		
		for(int mask=0;mask<(1<<n);mask++){
			int count=0;
			__int128 valid=0;
			for(int i=0;i<n;i++)
			if(mask&(1<<i)) count++,valid|=mark[i];
			if(count>k) continue;
			
			test=0;
			for(int i=0;i<m;i++)
			if(valid&(bit<<i)){
				test+=z[i];
			}
			if(test>ans) ans=test;
		}
		printf("%d\n",ans);
	}
	
	return 0;
}
