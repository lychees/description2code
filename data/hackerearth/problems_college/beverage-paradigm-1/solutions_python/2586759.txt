import itertools
x=input()
y=raw_input().split(" ")
y=map(int,y)
z=input()
c=[]
for i in range(1,len(y)):
	c.extend(itertools.combinations(y,i))
c=list(set(c))
con=0
for i in range(0,len(c)):
	s=0
	for j in range(0,len(c[i])):
		s+=c[i][j]
	if s==z:
		print "True"
		con=1
		break
	
if con==0:
	print "False"
