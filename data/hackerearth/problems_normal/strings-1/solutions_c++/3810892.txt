long long big_power(long long n,long long p)
{
    long long tmp;
    if(p==0)
        return 1;
    tmp=big_power(n,p/2);
    tmp=(tmp*tmp);
    if(p%2!=0)
        return (tmp*n);
    else
        return tmp;
}
