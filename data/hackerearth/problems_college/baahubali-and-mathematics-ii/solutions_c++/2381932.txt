#include <bits/stdc++.h>
using namespace std;

int a[10];

bool rec(int sum, int i) {
	
	if(i == 5) {
		
		if((sum + a[i] == 23) || (sum*a[i] == 23) || (sum-a[i] == 23))
			return true;
		else
			return false;
	}
	
	if(rec(sum+a[i], i+1)) return true;
	if(rec(sum-a[i], i+1)) return true;
	if(rec(sum*a[i], i+1)) return true;	
}

bool permute(int l, int r)
{
   int i;
   if (l == r) {
   	
   	if(rec(a[1], 2)) {
   		return true;
   	}
   	else
   		return false;
   }
   else
   {
       for (i = l; i <= r; i++)
       {
          swap(a[l], a[i]);
          if(permute(l+1, r)) return true;
          swap(a[l], a[i]);
       }
   }
   
   return false;
}

int main() {
    
    int t; cin >> t;
    
    while(t--) {
    	
    	for(int i = 1; i <= 5; i++)
    		cin >> a[i];
    		
    	if(permute(1, 5))
    		cout << "YES" << endl;
    	else
    		cout << "NO" << endl;
    }
    
    
    return 0;
}
