def change(n, k):
    if not n % k:
        return 0
    m = n % k
    return min(m, k - m)


def main():
    n, k = [int(s) for s in raw_input().split()]
    time = 10000000000
    for i in xrange(k):
        if i >= time: break
        time = min(time, i + change(n, k + i), i + change(n, k - i))
    print(time)


main()
