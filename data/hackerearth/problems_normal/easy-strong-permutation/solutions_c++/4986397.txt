#include <bits/stdc++.h>
using namespace std;
#define MAX 100005
typedef long long int ll;

int main()
{
	std::ios::sync_with_stdio(0);
	ll n;
	cin>>n;
	ll A[MAX];
	for(int i=0;i<n;i++)
	cin>>A[i];
	sort(A,A+n);
	ll B[MAX];
	ll c=0;
	ll j=n-1;
	for(int i=0;i<n/2;i++)
	{
		B[c++]=A[i];
		B[c++]=A[j];
		j--;
	}
	ll sum=0;
	for(int i=0;i<n;i++)
	{
		if(i==n-1)
		{
			sum=sum+abs(B[i]-B[0]);
		}
		else
		{
			sum=sum+abs(B[i]-B[i+1]);
		}
	}
	cout<<sum<<endl;
	return 0;
}
