#include <iostream>
using namespace std;
typedef long long int ll;
int main()
{
    ll t;
    cin>>t;
    while(t--)
    {
    	ll n;
    	ll k;
    	cin>>n>>k;
    	if(k>=n){cout<<"Nandu"<<endl;goto here;}
    	else
    	{
    		if(n%(k+1)==1)
    		cout<<"Nandu"<<endl;
    		else cout<<"Rani"<<endl;
    		goto here;
    	}
    	here: ;
    }
    return 0;
}
