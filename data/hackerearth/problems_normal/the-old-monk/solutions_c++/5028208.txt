#include<bits/stdc++.h>
#define ll long long int
using namespace std;
main()
{
    ll t,n,i,ma,x,lo,hi,mid;
    cin>>t;
    while(t--)
    {
        cin>>n;
        vector<ll> a,b;
        for(i=0;i<n;i++)
        {
            cin>>x;
            a.push_back(x);
        }
        for(i=0;i<n;i++)
        {
            cin>>x;
            b.push_back(x);
        }
        ma=0;
        for(i=0;i<n;i++)
        {
            hi=n-1;
            lo=0;
            while(lo<hi)
            {
                mid=(lo+hi+1)/2;
                if(b[mid]<a[i])
                    hi=mid-1;
                else
                    lo=mid;
            }
            if(lo-i>ma)
                ma=lo-i;
        }
        cout<<ma<<endl;
    }
    return 0;
}
