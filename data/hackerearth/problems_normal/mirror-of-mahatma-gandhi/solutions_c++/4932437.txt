#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int test;
    cin >> test;
    for(int i = 0; i < test; i++){
    	string name;
    	cin >> name;
    	//bool flag = true;
    	int n = 0;
    	for(int j = 0; j < name.length(); j++){
    		if(name[j] == '0' || name[j] == '1' || name[j] == '8')
    			n++;
    	}
    	if(n == name.length()){
    		int count = 0;
    	 	int mid = n/2;
    		if( n%2 == 0){
    			for(int j = 0, k = n - 1; j < n/2 && k >= n/2 ; j++, k--){
    			if(name[j] == name[k])
    				count++;
    			}
    		}
    		else{
    			for(int j = 0, k = n - 1; j < n/2 && k > n/2 ; j++, k--){
    				if(name[j] == name[k])
    					count++;
    			}
    		}
    		
    		if(count == mid)
    			cout << "YES" << endl;
    		else
    			cout << "NO" << endl;
    	}
    	else
    		cout << "NO" << endl;
    }
    return 0;
}
