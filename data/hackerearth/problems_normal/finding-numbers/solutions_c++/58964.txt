#include<iostream>
using namespace std;
#include<cstdio>
#include<map>
#include<algorithm>
#include<vector>
#define LL long long
int main()
{
int T;
cin >> T;
while(T--)
{
LL Z,N,M;
cin >> Z >> M >> N;
vector<LL> factors;
LL ans = 0;
LL i;
for( i =1;i*i<=Z;i++)
{
if(Z%i == 0 )
{
if( i <= M && (Z/i) <= N)
ans++;
if((Z/i) <= M && i <= N)
ans++;
}

}
if((i-1)*(i-1) == Z )
ans -- ;
cout << ans <<endl;
}
return 0;
}
