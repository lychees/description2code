#include <iostream>
using namespace std;
int prime[1000009]={0};
int count[1000009]={0};
int ans[1000009]={0};
int main()
{
	
	int i;
	for(i=2;i<=1000000;i++)
	{
		prime[i]=1;
		count[i]=0;
		
	}
	for(i=2;i<=1000;i++)
	{
		int j;
		for(j=2*i;j<=1000000;j+=i)
		prime[j]=-1;
	}	
	for(i=2;i<=1000000;i++)
	{
		if(prime[i]==1)
	count[i]=count[i-1]+1;
	else
	count[i]=count[i-1];
	}
	for(i=2;i<=1000000;i++)
	{
		if(prime[count[i]]==1)
		ans[i]=ans[i-1]+1;
		else
		ans[i]=ans[i-1];
	}
	int t;
	cin >> t;
	while(t--)
	{
		int r,l;
		cin >> l>>r;
		int x=0;
		if(prime[count[l]]==1)
		x=1;
	cout << ans[r]-ans[l]+x<<endl;
	}
    
    return 0;
}
