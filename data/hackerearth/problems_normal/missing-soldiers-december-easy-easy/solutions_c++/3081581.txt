#define ll long long
#define pll pair<ll,ll>
#define ppll pair<ll, pair<ll,ll> >
#define inf 1000000000
#define f first
#define s second

ll n,x,y,d,p,i,ans=0;
vector<pll> v;

int main()
{
    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>x>>y>>d;
        v.push_back(pll(x,x+d));
    }
    sort(v.begin(),v.end());
    //for(i=0;i<n;i++)
        //cout<<v[i].f<<" "<<v[i].s<<endl;
    p=1;
    for(i=0;i<n;i++)
    {
        p=max(p,v[i].f);
        if(v[i].s>=p)
        {
            ans+=v[i].s-p+1;
            p=v[i].s+1;
        }
    }
    cout<<ans;
}
