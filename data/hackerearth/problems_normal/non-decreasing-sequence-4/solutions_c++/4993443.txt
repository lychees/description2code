#include<bits/stdc++.h>
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t-->0)
	{
	    int n,len,length;cin>>n;
	    int arr[n+9],stack[n+9];
	    for(int i=0;i<n;i++) cin>>arr[i];
	    cin>>length;
	    len=1;
	    stack[0]=arr[0];
	    for(int i=1;i<n;i++)
	    {
	    	if(arr[i]<stack[0]) stack[0]=arr[i];
	    	else if(arr[i]>stack[len-1])
	    	{
	    		if(len<length)
	    		{
	    			stack[len]=arr[i];
	    			len++;
				}
			}
			else 
			{
				int c= lower_bound(stack,stack+len,arr[i])-stack;
				stack[c]=arr[i];
			}
			
		}
	    if(len>=length) cout<<stack[len-1]<<endl;
	    else cout<<"-1\n";
	    
	}
}
