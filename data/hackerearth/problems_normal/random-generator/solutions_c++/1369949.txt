{

    while(t--)
    {
        int a[100005]={0},i,j,n,k,p;
        scanf("%d%d%d",&n,&k,&p);
        for(i=0;i<n;i++)
        {
            scanf("%d",&a[i]);
        }
        bool flag=false;
        sort(a,a+n);
        for(i=0;i<n;i++)
        {
            int idx=upper_bound(a,a+n,a[i]+2*p)-a;
            if(idx-i+1>k)
            {
                printf("NO\n",i);
                flag=true;
                break;
            }
        }

        {
            printf("YES\n");
        }

}
