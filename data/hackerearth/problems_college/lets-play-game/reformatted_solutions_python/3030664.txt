T = input()
for test in xrange(T):
    N, K = map(int, raw_input().split())
    if N >= 1 and N <= 10**5 and K >= 1 and K <= 10**9:
        Ai = map(int, raw_input().split())
        Bi = map(int, raw_input().split())
        max_profit = 0

        for i, time in enumerate(Ai):
            no_of_stones = K/time
            profit = Bi[i]*no_of_stones
            if profit > max_profit:
                max_profit = profit
        print(max_profit)
