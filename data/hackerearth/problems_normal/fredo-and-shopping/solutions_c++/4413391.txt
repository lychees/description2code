#include <bits/stdc++.h>
using namespace std;

int n;
int arr[1000005];
vector<pair<int,int> > compressed_arr;
int BIT[1000005];

void update(int x,int val)
{
	for(;x<=n;x+=x&-x)
	{
		BIT[x]+=1;
	}
}

int query(int x)
{
	int ans=0;
	for(;x>0;x-=x&-x)
	{
		ans+=BIT[x];
	}
	return ans;
}

int main()
{
	cin>>n;
	int num;
	for(int i=1;i<=n;i++)
	{
		cin>>num;
		compressed_arr.push_back({num,i});
	}
	sort(compressed_arr.begin(),compressed_arr.end());
	int k=1,prev=compressed_arr[0].first;
	compressed_arr[0].first=k;
	for(int i=1;i<n;i++)
	{
		if(compressed_arr[i].first!=prev)
		{
			k++;
			prev=compressed_arr[i].first;
		}
		compressed_arr[i].first=k;
	}
	for(int i=0;i<n;i++)
	{
		arr[compressed_arr[i].second]=compressed_arr[i].first;
	}
	for(int i=1;i<=n;i++)
	{
		update(arr[i],1);
		cout<<query(arr[i]-1)<<" ";
	}
}
