for _ in range(int(raw_input())):
    n = int(raw_input())
    w = map(int, raw_input().split())
    h = map(int, raw_input().split())
    w1 = w.pop(0)
    w.sort()
    h.sort(reverse=1)
    h1 = h.pop(0)
    if w1 * h1 > max(h[i] * w[i] for i in range(n-1)):
        print "YES"
    else:
        print "NO"
