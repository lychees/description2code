#include<iostream>
#include<stdio.h>
using namespace std;

int main(){
long long t;
scanf("%lld",&t);
while(t--){

   long long d,i,j,c=0;
   scanf("%lld",&d);
   for(i=1;d>0;i++){
       for(j=1;j<=i;j++){
       d--;
       
      if(d==0){
      c=i;
      break;
      }
   }
}


   if(c%3==0){
   printf("SL\n");
   }else if(c%3==1){
   printf("LB\n");
   }else{
   printf("BS\n");
   }
   

}

}
