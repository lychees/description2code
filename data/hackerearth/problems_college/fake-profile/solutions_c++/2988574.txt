#include <iostream>
using namespace std;
int a[26];

{
    string name;
    cin>>name;
    int i=0,count=0;
    while(name[i]!='\0')
    {
    	a[name[i]-'a']++;
    	i++;
    }
    for(int i=0;i<26;i++)
     if(a[i]!=0)
      count++;
    if((count&1)==0)
     cout<<"CHAT WITH HER!";

    return 0;
}
