line = raw_input()

dic = {}
for i in range(len(line)):
    ch = line[i]
    if ch in dic:
        dic[ch] += 1
    else:
        dic[ch] = 1

chars_whose_count_is_odd = 0
for key, value in dic.items():
    if value % 2 == 1:
        chars_whose_count_is_odd += 1

if chars_whose_count_is_odd > 1:
    print ("NO")
else:
    print ("YES")
