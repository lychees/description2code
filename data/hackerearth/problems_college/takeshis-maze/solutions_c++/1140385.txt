#include <iostream>
#include<vector>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    scanf("%d",&t);
    while(t--){
    	int n,c,p;
    	
    	scanf("%d%d",&n,&c);
    	vector<vector<int> >matrix(n+1,vector<int>(n+1,0));
    	vector<vector<int> >dp(n+1,vector<int>(n+1,0)) ;
    	for(int i=1;i<=n;i++){
    		for(int j=1;j<=n;j++){
    			scanf("%d",&p);
    			matrix[i][j]=p;
    		}
    	}
    	for(int i=1;i<=n;i++){
    		for(int j=1;j<=n;j++){
    			dp[i][j] = max(dp[i-1][j-1],max(dp[i][j-1],dp[i-1][j]))+matrix[i][j];
    		}
    	}
    	if(dp[n][n]-c>=0){
    		printf("%d\n",dp[n][n]-c);
    	}else
    		printf("%d\n",-1);
    }
    return 0;
}
