#include <iostream>
#include<stdio.h>
#include <string.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    	char c[1001];
    	scanf("%s ",c);
    	char ch;
    	scanf("%c",&ch);
    	int i,cnt=0;
    	int l=strlen(c);
    	for(i=0;i<l;i++){
    		if(c[i]==ch)
    					cnt++;
		}
		printf("%d\n",cnt);
    }
    return 0;
}
