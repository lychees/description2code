#include<iostream>
using namespace std;
bool gcd(int a,int b)
{
	int c=0;
	int s=(a>b)?b:a;
	for(int i=2;i<=s;i++)
	{
		if(a%i==0 && b%i==0)
			c++;
	}
	if(c!=0)
		return false;
	else
		return true;
}
int main()
{
	int n;
	cin>>n;
	int ar[n+1];
	for(int i=1;i<=n;i++)
	{
		cin>>ar[i];
	}
	int count=0;
	for(int i=1;i<=n;i++)
	{
		for(int j=i+1;j<=n;j++)
		{
			if(ar[i]>ar[j])
			{
				if(gcd(ar[i],ar[j]))
					count++;
			}
		}
	
	}
	cout<<count<<endl;
	return 0;
	
}
