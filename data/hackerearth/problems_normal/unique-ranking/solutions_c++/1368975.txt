#include <bits/stdc++.h>
#define PB push_back
#define MP make_pair
#define LL long long
#define MAXN 100001
#define M 1000000007
#define F first
#define S second
#define sc(x) scanf("%d",&x)
#define scl(x) scanf("%lld",&x)
using namespace std;

int a[1123][1123];
int b[1123], c[1123];

int main(){
    int n, t, i, j, k;
    cin>>n>>k;
    for(i=1;i<=n;i++){
        for(j=1;j<=k;j++){
            sc(a[i][j]);
        }
    }
    int q, m, d, sub;
    sc(q);
    while(q--){
        sc(m);
        for(i=1;i<=m;i++)
            sc(b[i]);
        for(i=1;i<=n;i++)
            c[i] = 1;
        for(i=1;i<=m;i++){
            d = 101;
            sub = b[i];
            //cout<<sub<<endl;
            for(j=1;j<=n;j++){
              //  cout<<a[j][sub]<<" ";
                if(a[j][sub] < d && c[j] == 1){
                    d = a[j][sub];
                }
            }
            //cout<<d<<endl;
            for(j=1;j<=n;j++){
                if(a[j][sub] == d && c[j] == 1){
                    c[j] = 1;
                }
                else
                    c[j] = 0;
            }
        }
        for(i=1;i<=n;i++){
            if(c[i] == 1){
                break;
            }
        }
        printf("%d\n",i);
    }
}
