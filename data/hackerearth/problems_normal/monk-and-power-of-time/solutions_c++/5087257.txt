#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin>>n;
    int x;
    queue<int> q;
    int a[101];
    for(int i=0;i<n;i++)
    {
        cin>>x;
        q.push(x);
    }
    for(int i=0;i<n;i++)
    {
        cin>>a[i];
    }
    int k=0;
    long long int ans=0;
    int element;
    while(!q.empty())
    {
        while(q.front()!=a[k])
        {
            element=q.front();
            q.pop();
            q.push(element);
            ans++;
        }
        q.pop();
        ans++;
        k++;
    }
    cout<<ans;
    return 0;
}
