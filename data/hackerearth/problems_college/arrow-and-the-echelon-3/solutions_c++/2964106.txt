#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
    long long int t,p;
    cin>>t;

    while(t--)
    {
        cin>>p;
        long int t3,t5,flag=0;
        for(t5 =0; t5<=p; )
        {
            t3 = p - t5 ;
            if(t3%3==0)
            {
                flag =1;
                break;
            }
            else t5 =t5 +5;
        }
        if(flag==0)
            cout<<"-1"<<endl;
        else
        {
            long int ff =t3,fk = t5;
            while(ff--)
                cout<<"5";
            while(fk--)
                cout<<"3";
            cout<<endl;
        }
    }
    return 0;
}
