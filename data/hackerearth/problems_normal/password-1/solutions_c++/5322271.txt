#include <bits/stdc++.h>
#include <functional>

using namespace std;

int main()
{
	int n;
	cin>>n;
	vector<string> vec;
	int j=0;
	while(n--)
	{
		string s;
		cin>>s;
		vec.push_back(s);
	}
	int flag=0;
	for(int i=0;i<vec.size();i++)
	{
		string str(vec[i]);
		reverse(str.begin(),str.end());
		for(int j=0;j<vec.size();j++)
		{
			if(str.compare(vec[j])==0)
			{
				int length=vec[j].length();
				cout<<length<<" "<<vec[j][length/2]<<endl;
				flag=1;
				break;
			}	
		}
		if(flag==1)
			break;
	}	
}
