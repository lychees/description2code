bool palin(int start, string& s)
{
	int i = start, j = s.size()-1;
	while(i <= j) {
		if(s[i] != s[j])
			return false;
		i++, j--;
	}
	return true;
}

void solve()
{
	string s;
	cin>>s;
	int i = 0;
	for(;i<s.size();i++) {
		if(palin(i, s))
			break;
	}
	cout<<s.size()+i<<"\n";
}
