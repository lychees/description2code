#include<iostream>
#include<vector>
using namespace std;
typedef long long int ll;
#define m 1000000000000
vector<ll> v;
ll upper(ll low,ll value)
{
    ll a=low; ll b=v.size();

    while(b>a)
    {
            ll mid=(a+b)/2;
       if(v[mid]>value)
       {
           b=mid;
       }
       else
       {
           a=mid+1;
       }
       
    }
    return b;
}

ll lower(ll low,ll value)
{
    ll a=low;ll b=v.size();
    ll mid;
    
    while(b>a)
    {
        mid=(a+b)/2;
        if(v[mid]<value)
        {
            a=mid+1;
        }
        else
        {
            b=mid;
        }
    }
    return a;
}


int main()
{
     ios::sync_with_stdio(false);cin.tie(0);
    for(ll i=1; ;i++)
    {
        ll t;
        t=(i*(i+1))/2;
        v.push_back(t);
        if(t>m)
            break ;
    }
    ll h,ans;
            ll n,w;
    cin>>h;
    for(int j=0;j<h;j++)
    {
          ans=0;
        cin>>n>>w;
        for(ll i=0;i<v.size() && v[i]<=w;i++)
        {
    
            ans+=upper(i,w-v[i])-lower(i,n-v[i]);
       
        }
        cout<<ans<<endl;

    }
    return 0;
}
