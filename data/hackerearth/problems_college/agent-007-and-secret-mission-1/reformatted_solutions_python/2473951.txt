t = input()
while t > 0:
    t -= 1
    s = raw_input()
    n = input()
    val = (n*(n+1))/2
    cnt = 0
    if val % 2 == 0:
        for i in range(len(s)):
            if s[i] == '1':
                cnt += 1
    else:
        for i in range(len(s)):
            if s[i] == '0':
                cnt += 1
    print cnt
