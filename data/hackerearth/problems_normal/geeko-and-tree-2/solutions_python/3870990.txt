for x in xrange(int(raw_input())):
	res = 0
	ary, lvl = map(int, raw_input().split())
	nodes = (pow(ary, lvl + 1) - 1) / (ary - 1)
	nod = str(nodes)
	for s in nod:
		res += int(s)
	print res
