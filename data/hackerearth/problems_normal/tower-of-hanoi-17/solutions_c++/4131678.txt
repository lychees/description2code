#include <bits/stdc++.h>
#define pi pair<long long,long long>
#define pb push_back
using namespace std;
typedef long long ll;
vector<pi> vec;
ll mxh[205];
bool comp(pi a,pi b)
{
    if(a.second!=b.second)
        return a.second>b.second;
    return a.first>b.first;
}
int main()
{
    ll t,i,n,ans,x,y,sz,j;
    cin>>t;
    while(t--)
    {
        cin>>n;
        vec.clear();
        for(i=0;i<n;i++)
        {
            cin>>x>>y;
            vec.pb(make_pair(x,y));
        }
        sort(vec.begin(),vec.end(),comp);
        sz=vec.size();
        for(i=0;i<sz;i++)
           mxh[i]=vec[i].second;
        ans=mxh[sz-1];
        for(i=sz-2;i>=0;i--)
        {
            for(j=i+1;j<sz;j++)
            {
                if(vec[j].second<vec[i].second && vec[j].first<vec[i].first)
                    mxh[i]=max(mxh[i],vec[i].second+mxh[j]);
            }
            ans=max(ans,mxh[i]);
        }
        cout<<ans<<endl;
    }
    return 0;
}
