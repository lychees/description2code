n = int(raw_input())

while n > 0:

    lineInput = raw_input()
    lineInput = lineInput.split(' ')

    a = int(lineInput[0])
    b = int(lineInput[1])

    if a == b:
        print a
        n -= 1
        continue

    while True:

        if a > b:
            a = a/2
        elif a < b:
            b = b/2
        else:
            print a
            break

    n -= 1
