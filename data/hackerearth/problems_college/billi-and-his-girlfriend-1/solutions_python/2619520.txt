def fe(a,b,k):
    if(b==0):
        return 1
    else:
        ret = fe(a,b/2,k)
        ret = ret* ret
        ret %= k
        if(b%2):
            ret *= a
            ret %= k
        return ret

t = int(raw_input())
while (t!=0):
    a, b, k = raw_input().split()
    a = int(a)
    b = int(b)
    k = int(k)
    k= 10**k
    ans = fe(a,b,k)
    lA = 0
    while(ans):
        lA += ans %10
        ans /= 10
    print lA
    t -= 1
