    		cin >> k;

    		vector<int>val(1000009);

    		for(i=1;i<=1000000;i++)
    		val[i]=0;
    		
    		for(i=1;i<n;i++)

    			while(j>0 && s[j]!=s[i])
    			j=f[j-1];
    			if(s[j]==s[i])
    			j++;
    			f[i]=j;
    		}
    		val[f[n-1]]=1;
    		i=n-1;

    		{
    			if(f[i]!=0)
    			val[f[i]]=1;

    		}

    			}
