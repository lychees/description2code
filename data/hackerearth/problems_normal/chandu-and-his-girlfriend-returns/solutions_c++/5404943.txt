#include <iostream>
#include <cstdio>
using namespace std;

int c[1000000],k=0;

void merge(int *a,int alen,int *b,int blen) {
	
	int i = 0,j=0;
	while(i<alen&&j<blen) {
		if (a[i]>b[j]) {
			c[k++] = a[i++];
		}else {
			c[k++] = b[j++];
		}
	}
	
	while(i<alen) {
		c[k++] = a[i++];
	}
	while(j<blen) {
		c[k++] = b[j++];
	}
	
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--) {
    	int n,m;
    	scanf("%d %d",&n,&m);
    	
    	int a[n],b[m];
    	
    	for(int i=0;i<n;i++) {
    		scanf("%d",&a[i]);
    	}
    	
    	for (int i=0;i<m;i++) {
    		scanf("%d",&b[i]);
    	}
    	k = 0;
    	merge(a,n,b,m);
    	for (int i=0;i<k;i++) {
    		if (i==k-1) {
    			printf("%d\n",c[i]);
    		}else {
    			printf("%d ",c[i]);
    		}
    	}
    }
    return 0;
}
