#include <algorithm>
#include <cstring>
#include <iostream>
#include <math.h>
#include <stdio.h>
#include <stdlib.h>
using namespace std;
#define INF 0x7fffffff
int c, d, n, m, k, ans = INF;
int main() {
  cin >> c >> d >> n >> m >> k;
  for (int i = 0; i <= m; i++)
    ans = min(ans, i * c + max((n * (m - i) - k) * d, 0));
  cout << ans << endl;
  return 0;
}
