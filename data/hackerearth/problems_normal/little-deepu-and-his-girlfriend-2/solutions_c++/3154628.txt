#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t,n,m,s[105], i, j;
	bool dp[10005];
	scanf("%d", &t);
	while(t--)
	{
		scanf("%d%d", &m, &n);
		for(i=0;i<=m;i++)
			dp[i] = false;
		for(i=0;i<n;i++)
		{
			scanf("%d", &s[i]);
			dp[s[i]] = true;
		}
		for( i = 1; i <= m; i++)
		{
			if(!dp[i])
			{
				for( j = 0; j < n; j++)
					if(i-s[j] >= 0 && !dp[i-s[j]])
					{
						dp[i] = true;
						break;
					}
			}
		}
		if(dp[m])
			printf("Little Deepu\n");
		else
			printf("Kate\n");
	}
	return 0;
}
