#include <iostream>
using namespace std;

int gcd(int a,int b)
{
	if(b==0)
	return a;
	else
	return gcd(b,a%b);
}

int main()
{
    int n,i,j,count=0,a[10000];
    cin>>n;
    for(i=0;i<n;i++)
    cin>>a[i];
    
    for(i=0;i<n;i++)
    for(j=0;j<i;j++)
    if((a[j]>a[i])&&gcd(a[i],a[j])==1)
    count++;
    
    cout<<count;
}
