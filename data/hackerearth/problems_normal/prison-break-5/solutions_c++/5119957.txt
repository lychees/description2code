#include <iostream>
using namespace std;
int ways=0,n;
int mat[21][21],visited[21][21];

void calculateWays(int x,int y)
{
    if(x==n && y==n)
    {
        ways++;
        return ;
    }
    else{
        visited[x][y]=1;
        if((x+1)<=n && visited[x+1][y]==0 && mat[x+1][y]==0)
            calculateWays(x+1,y);
        if((y+1)<=n && visited[x][y+1]==0 && mat[x][y+1]==0)
            calculateWays(x,y+1);
        if((x-1)>0 && visited[x-1][y]==0 && mat[x-1][y]==0)
            calculateWays(x-1,y);
        if((y-1)>0 && visited[x][y-1]==0 && mat[x][y-1]==0)
            calculateWays(x,y-1);

        visited[x][y]=0;
    }
    return;
}

int main()
{
    int t,i,j;
    cin >> t;
    while(t--)
    {
        ways=0;
        cin >> n;
        for(i=1;i<=n;i++)
        {
            for(j=1;j<=n;j++)
            {
                cin >> mat[i][j];
                visited[i][j]=0;
            }
        }
        
            calculateWays(1,1);
        
        cout << ways <<endl;
    }
    return 0;
}
