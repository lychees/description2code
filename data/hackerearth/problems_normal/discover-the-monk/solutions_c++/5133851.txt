#include <iostream>
#include <algorithm>
#define MAX 9999999
using namespace std;

int main()
{
	long long A[MAX],q,n,x;
	cin>>n>>q;
	long long i = 0;
	while(i < n)
	{
		cin>>A[i++];
	}
	sort(A,A+n);
	while(q-- > 0)
	{
		cin>>x;
		if(binary_search(A,A+n,x))
			cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;
	}
	return 0;
}
