

    #include <bits/stdc++.h>

    #define IOS 		ios_base::sync_with_stdio(false); cin.tie(NULL)
    #define IN          int t; cin >> t; while(t--)
     
    long long int ans,a[203];
     
    bool chk(long long int x)
    {
    	if(!x) return false;
    	if(!(x&(x-1))) return true;
    	else return false;
    }
     
    int main()
    {
        IN
        {
        	int n;
        	cin >> n;
        	bool f = false;
        	for(int i=0;i<n;++i) cin >> a[i];
        	for(int bit = 0; bit < 32; ++bit)
        	{
        		ans = 0;
        		for(int i=0;i<n;++i)
        		{
        		if(a[i] & (1 << bit))//if bit(bit) is set in element a[i];
        		{
        			if(!ans) ans = a[i];
        			else ans &= a[i];
        		}
        		}
        		if(chk(ans))
        		{
        			f = true;
        			break;
        		}
        	}
        	if(f) cout << "YES\n";
        	else cout << "NO\n";
        }
        
        
        return 0;    
    }
