#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int main()
{
	long long int n,k,i,j,flag=0;
	cin>>n>>k;
	vector<long long> v;
	for(i=0;i<n;i++)
	{
		cin>>j;
		v.push_back(j);
	}
	sort(v.begin(),v.end());
	for(i=0;i<n-1;i++)
	{
		if(v[i+1]-v[i]>k)
		{
			cout<<v[i]<<endl;
			flag=1;
			break;
		}
	}
	if(flag==0)
	cout<<v[i]<<endl;
}
