#include <bits/stdc++.h>
using namespace std;
int prime(int n)
{
	int i;
	for(i=2;i<=sqrt(n);i++)
	{
		if(n%i==0)
		{
			return 0;
		}
	}
	return 1;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string str;
    	cin>>str;
    	long int n;
    	n=str.size();
    	long int i;
    	int flag[n];
    	for(i=0;i<n;i++)
    	{
    		flag[i]=prime(str[i]);
    	}
       	int pos=0;
    	for(i=0;i<n;i++)
    	{
    		if(flag[i]==1)
    		{
    			str.erase(i-pos,1);
    			pos++;
    		}
    	}
    	if(pos==n)
    	cout<<"NULL"<<endl;
    	else
    	cout<<str<<endl;
    	
    }
}
