#include <iostream>
#include<stack>
using namespace std;

int main()
{
    int N;
    cin>>N;
    stack<int> s;
    	int a[10000];
    	for(int i=0;i<N;i++)
    	  cin>>a[i];
    	 for(int j=0;j<N;j++){
    	 	if(a[j]>0)
    	 	  s.push(a[j]);
    	 	else
    	 	  s.pop(); 
    	 } 
    	 int sum=0;
    	 
        while(s.size() >0){
        	sum+=s.top();
        	s.pop();
        }
        cout<<sum;
    return 0;
}
