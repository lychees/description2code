#include<bits/stdc++.h>
using namespace std;
int mat[1024][1026]={0};
int dp[12][1026][1026]={0};
int n;

int f(int a,int b)
{
	if(b-a==1)
	return (mat[b-1][a-1])?b:a;
	int mid=(a+b)/2;
	int x=f(a,mid);
	int y=f(mid+1,b);
	return (mat[y-1][x-1])?y:x;
}

int main()
{
	scanf("%d",&n);
	for(int i=1;i<(1<<n);i++)
	for(int j=0;j<i;j++)
	scanf("%d",&mat[i][j]);
	printf("%d",f(1,(1<<n)));
	return 0;
	}
