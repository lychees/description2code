#include <iostream>
#include<map>
#include<malloc.h>
using namespace std;
struct Candidate
{
  long age;
  long num;
};
int main()
{
    long n,m,age=0;
    cin>>n>>m;
    map<long,long>maps;
    struct Candidate*win=(struct Candidate*)malloc(sizeof(struct Candidate));
    win->age=0;
    win->num=0;
    for(long i=0;i<n;i++)
    {
     	cin>>age;
        maps[age]+=1;
        if(maps[age]>win->num || (maps[age]==win->num && age>win->age))
        {
        	win->age=age;
        	win->num=maps[age];
        }
        cout<<win->age<<" "<<win->num<<"\n";
    }
    return 0;
}
