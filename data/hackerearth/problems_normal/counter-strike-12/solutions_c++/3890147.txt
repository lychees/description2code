#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,m,d;
		pair<int,int>pos[100000];
		pair<int,int>tar[100000];
		
		cin>>n>>m>>d;
		unsigned int x,y;
		for(int i=0;i<n;i++)
		{
			cin>>x>>y;
			pos[i]=make_pair(x,y);
		}
		
		for(int i=0;i<m;i++)
		{
			cin>>x>>y;
			tar[i]=make_pair(x,y);
		}
		int cnt=0;
		int flag=0;
		for(int i=0;i<m;i++)
		{
			for(int j=0;j<n;j++)
			{
				int dis=(abs(pos[j].first-tar[i].first)+abs(pos[j].second-tar[i].second));
				if(dis<=d)
				{
					cnt++;
					break;
				}	
			}
			if(cnt>m/2+1)
			{
				flag=1;
				break;
			}
		}
		if(flag)
			cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;
		
	}
}
