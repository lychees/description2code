#include <bits/stdc++.h>
#define ll long long int
using namespace std;
int main()
{
	ios::sync_with_stdio(false);
	ll t;
	cin>>t;
	while(t--)
	{
		char str[50];
		ll tmp2=0,ans=0;
		cin>>str;
		for(ll i=strlen(str)-1; i>=0; i--)
		{
			if(str[i]=='+')
			{
				tmp2 = str[i-1] - '0';
				i--;
				ans = ans + tmp2;
			}
			else if(str[i]=='-')
			{
				tmp2 = str[i-1] - '0';
				i--;
				ans = ans - tmp2;
			}
			else if(str[i]=='*')
			{
				tmp2 = str[i-1] - '0';
				i--;
				ans = ans * tmp2;
			}
			else
			{
				ans = str[i] - '0';
			}
		}
		cout<<ans<<endl;
	}
	return 0;
}
