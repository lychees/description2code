#include <bits/stdc++.h>
using namespace std;

int main()
{
    long int a,b,c,sarea,oarea;
    cin>>a>>b>>c;
    sarea = 2*a*b;
    long int max,min;
    if(a>=b)
    {
    	min = b;
    	max = a;
    }
    else
    {
    	min = a;
    	max = b;
    }
    
    	if(c <= min)
    	{
    		oarea = c*c;
    	}
    	else
    	if(c>min && c<=max)
    	{
    		oarea =  (min*min)+2*(c-min)*min;
    	}
    	else
    	if(c>max)
    	{
    		oarea =  2*a*b-((min+max-c)*(min+max-c));
    	}
    	
    if(c>(a+b))
    {
    	cout<<"1/1";
    }else
    {
    	int g = __gcd(oarea, sarea);
    
    	cout<<oarea/g<<"/"<<sarea/g;
    }
    return 0;
}
