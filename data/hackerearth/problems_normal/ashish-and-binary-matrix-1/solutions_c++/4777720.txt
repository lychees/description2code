#include <iostream>
#include <set>
#include <map>

using namespace std;

int main()
{
    int t;
    cin>>t;
    while (t--) {

    map<string,int> count_freq;

    int n, m;
    cin>>n>>m;
    string str;
    for (int i=0;i<n;i++)  {
        cin>>str;
        count_freq[str]++;
    }
    int o = 0;
    for (map<string,int>::iterator itr=count_freq.begin();itr!=count_freq.end();itr++) {
        if ((*itr).second>1) {
            o=1;
            break;
        }
    }
    if (o==0) cout<<"Yes\n";
    else cout<<"No\n";
    }
    return 0;
}
