#include <bits/stdc++.h>
using namespace std ;
#define LL long long 
int T,N ;
string str ;

int main(){
	scanf("%d",&T) ;
	while(T--){
		cin >> str ;
		N = str.length() ;
		LL ans1 = 0 ,ans2 = 0;
		int  i = 0;
		while(i<N){
			if(str[i] != 'x'){ i ++ ; continue ; } 
			int c =  0;
			while(i<N && str[i] == 'x'){
				c ++ ;
				i ++ ;
			}
			ans1 = ans1 + (1LL * c * (c+1)/2) ;
		}
		i = 0;
		while(i<N){
			if(str[i] != 'y'){ i ++ ; continue ; }
			int c =  0;
			while(i<N && str[i] == 'y'){
				c ++ ;
				i ++ ;
			}
			ans1 = ans1 + (1LL * c * (c+1)/2) ;
		}	
		i = 0;
		while(i<N){
			if(str[i] != 'z'){ i ++ ; continue ; }
			int c =  0;
			while(i<N && str[i] == 'z'){
				c ++ ;
				i ++ ;
			}
			ans1 = ans1 + (1LL * c * (c+1)/2) ;
		}
		i = 0;
		while(i<N){
			if(str[i] == 'z'){ i ++ ; continue ; }
			int c =  0;
			while(i<N && (str[i] == 'x' || str[i] == 'y')){
				c ++ ;
				i ++ ;
			}
			ans2 = ans2 + (1LL * c * (c+1)/2) ;
		}
		i = 0;
		while(i<N){
			if(str[i] == 'x'){ i ++ ; continue ; }
			int c =  0;
			while(i<N && (str[i] == 'z' || str[i] == 'y')){
				c ++ ;
				i ++ ;
			}
			ans2 = ans2 + (1LL * c * (c+1)/2) ;
		}
		i = 0;
		while(i<N){
			if(str[i] == 'y'){ i ++ ; continue ; }
			int c =  0;
			while(i<N && (str[i] == 'x' || str[i] == 'z')){
				c ++ ;
				i ++ ;
			}
			ans2 = ans2 + (1LL * c * (c+1)/2) ;
		}
		cout << ans2-ans1 << endl ;
	}
	return 0 ;
}
