#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    long t,z,sum=0,i;
    char a[10000];
    cin>>t;
    for(z=1;z<=t;z++)
    {
    	cin>>a;
    	if((int(a[0])-48)==1 && (a[1]<'0' || a[1]>'9'))
    	{
    		cout<<sum;
    		return 0;
    	}
    	sum=0;
    	for(i=0;i<strlen(a);i++)
    	{
    		sum+=(int(a[i])-48);
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
