#include <bits/stdc++.h>
using namespace std;

int main()
{
	//ios::sync_with_stdio( false);
    int i, t, n;
    //cin >> t;
    scanf("%d",&t);
    char ch;
    char s[10000001];
    string sub;
    scanf("%c",&ch);
    while( t--)
    {
    	//getline(cin, s);
    	gets(s);
    	n=strlen( s);
   		set<string> pr;
   		sub="";
   		i = 0;
   		while( i<n )
   		{
   			while( (i < n) && (s[i]!=' ') )
   			{
   				sub += s[i];
   				i++;
   			}
   			if( pr.find( sub)== pr.end())
   			{
   				cout << sub << " ";
   				pr.insert( sub);
   			}
   			sub= "";
   			i++;
   		}
   		printf("\n");
    }
    return 0;
}
