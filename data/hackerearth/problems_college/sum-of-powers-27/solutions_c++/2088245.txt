#define MOD 1000000007

int a[1000005];

int main(){
	
	a[1]=1;
	for(ll int i=2;i<1000001;i++) {
     	if(a[i]!=0) 
     		continue;
     	for(ll int j=i*i;j<1000001;j*=i)
     		a[j]=1;
    
    }
	int t, n;
	scanf("%d", &t);
	REP(tc, t){
		scanf("%d", &n);
		int f = 1;
		FOR(i,1,n-1){
         	if(a[n-i] && a[i]){ 
         		f=0;
         		printf("Yes\n");
         		break;
         	}
         }
         if(f)
         	printf("No\n");
	}
	return 0;
}
