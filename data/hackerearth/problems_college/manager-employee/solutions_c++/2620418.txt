#include<bits/stdc++.h>
using namespace std;
int ar[120];
int main() {
	int t,n,i,ans,temp,mini,maxi;
	scanf("%d",&t);
	while(t--) {
		scanf("%d",&n);
		for(i=0;i<n;i++) {
			scanf("%d",&ar[i]);
		}
		sort(ar,ar+n);
		ans=0;
		if(n==2) {
			printf("0\n");
			continue;
		}
		maxi=-1;
		mini=123456789;
		for(i=0;i<n/2;i++) {
			temp=ar[i]+ar[n-i-1];
			if(temp>maxi)
				maxi=temp;
			if(temp<mini)
				mini=temp;
		}
		printf("%d\n",maxi-mini);
	}
	return 0;
}
