#include<bits/stdc++.h>
using namespace std;
double area(int x1,int y1,int x2,int y2,int x3,int y3){
	double ar=(x1*(y2-y3) + x2*(y3-y1) + x3*(y1-y2))/2.0;
	if(ar<0)ar=ar*(-1);
	return ar;
}
int main(){
	int t;
	cin>>t;
	while(t--){
		int x1,x2,x3,y1,y2,y3,x4,y4;
		cin>>x1>>y1>>x2>>y2>>x3>>y3>>x4>>y4;
		double a1=area(x1,y1,x2,y2,x3,y3);
		double a2=area(x1,y1,x2,y2,x4,y4);
		double a3=area(x1,y1,x4,y4,x3,y3);
		double a4=area(x4,y4,x2,y2,x3,y3);
		if(a1==a2+a3+a4){
			cout<<"INSIDE"<<endl;
		}else{
			cout<<"OUTSIDE"<<endl;
		}
	}
	return 0;
}
