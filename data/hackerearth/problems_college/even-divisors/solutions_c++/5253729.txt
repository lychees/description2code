#include<stdio.h>
#include<math.h>
int main(){
	int i, ni; double t, d, n, c, r; scanf("%lf", &t);
	while(t-- && scanf("%lf", &n)){
		if((int)n & 1)	printf("0\n");
			else{
				c = 0, d = (int)sqrt(n), ni = (int)n;
				for(i = 2; i <= d; i++){
					r = ni % i; 
					if(!r && !(i & 1))	c++;
					if(!((ni / i) & 1) && !(ni % i))	c++;
				}
				(d * d == ni) ? printf("%.0lf\n", c) : printf("%.0lf\n", c + 1);
			}
	}
	return 0;
}
