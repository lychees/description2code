#include <iostream>

using namespace std;

int main()
{
    long long int A1,A2;
    int T,status;
    long int N,i;
    cin >> T;
    while (T--)
    {
        status=1;
        cin >> N >> A1;
        for (i=1;i<N;++i)
        {
            cin >> A2;
            if (A2<A1)
            {
                status=0;
                break;
            }
            else
                A1=A2;
        }
        for (i+=1;i<N;++i)
            cin >> A1;
        if (status)
            cout << "YES" << endl;
        else
            cout << "NO" << endl;
    }
    return 0;
}
