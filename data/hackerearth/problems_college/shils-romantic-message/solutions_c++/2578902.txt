#include <iostream>
#include<cstring>
using namespace std;
char msg[1000005];

int main()
{
   int t;
   int n,i,len;
   cin>>t;
   while(t--) {
   	n=-1;
   	cin>>msg;
   	len=strlen(msg);
   	for(i=0;i<len;i++) {
   		if(n==-1)
   		n=msg[i]-'a';
   		char x=(msg[i]-'a'-n+26)%26 + 'a';
   		cout<<x;
   	}
   	cout<<endl;
   }
   
}
