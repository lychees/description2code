#include <bits/stdc++.h>

using namespace std;

#define ll long long int

int main() {
	ll n;
	scanf("%d",&n);
	while(n--) {
		string s,s1,ans;
		ll i,j;
		cin >> s;
		ll len = s.length();
		for(i = 0;i < len;i++) {
			if(s[i] >= '0' && s[i] <= '9') {
				ll num = s[i] - '0';
				for(j = 0;j < num;j++) {
					ans = ans + s1;
				}
			
				s1 = ans;
				ans = "";
			} 
			else {
				s1 = s1 + s[i];
			}
		}
		cout << s1 << "\n";
	}

}
