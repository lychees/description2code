#include<bits/stdc++.h>
#include<iostream>
using namespace std;
int prime[1000000];
void soe(int n){
	int i,k=0;
	int a[1000000];
	for(i=0;i<=n;i++)
	a[i]=1;
	for(i = 2; i*i <=n; i++){
		if(a[i]==1){
			for(int j=i*i; j<=n; j+=i)
			a[j]=0;
		}
		if(a[i]==1)
		prime[k++]=i;
	}
	for(;i<=n;i++)
	if(a[i])
	prime[k++]=i;
}
int etf(int n)
{
	int phi=n;
	int i;
	for(i=0;i*i<=n;i++)
	{
		if(n%prime[i]==0){
			phi=phi*(prime[i]-1)/prime[i];
			while(n%prime[i]==0)
			n=n/prime[i];
		}
		
	}
	if(n!=1)
		phi=phi*(n-1)/n;
	return phi;
}
int main()
{
	int t;
	int n,d;
	cin >> t;
	while(t--){
		cin >> n;
		soe(n);
		d=etf(n);
		cout << d <<endl;
	}
}
