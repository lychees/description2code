#include <iostream>
#include <string>
#include <ctype.h>
using namespace std;

int main()
{
	int t,i,l,m,j,k,f;
	string s;
	
 	cin>>t;
	for (i=0;i<t;i++)
		{
		char a[1000000]={'\0'};
		l=0;
    	cin>>s;
    	for(j=0;j<s.length();j++)
    		{
    		if(isalpha(s[j])!=0)
    			{
    			a[l]=s[j];
    			l++;
    			}
    		else
    			{
    			m=l;
    			for(f=0;f<(s[j]-'0')-1;f++)
    				{
    				for(k=0;k<m;k++)
    					{
    					a[l]=a[k];
    					l++;
    					}
    				}
    			}
    		}
    	for (k=0;a[k]!='\0';k++)
    		{
    			cout<<a[k];
    		}
    	cout<<endl;
    	}
}
