#include<bits/stdc++.h>
#define ll unsigned long long int
using namespace std;
ll ans=0ll;
void gcd(ll x, ll y){
	while(y){
		if(y>x) swap(x,y);
		ll val=x/y;
		ans+=val;
		ll var=x%y;
		x=y;
		y=var;
	}
}
int main(){
    int t;
    cin>>t;
    while(t--){
    	ans=0;
    	ll x,y; cin>>x>>y;
    	gcd(x,y);
    	cout<<ans-1<<endl;
    }
}
