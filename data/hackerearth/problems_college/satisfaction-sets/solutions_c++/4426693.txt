#include <bits/stdc++.h>
#define ll long long int
using namespace std;

int main()
{
	ll arr[100005];
	ll brr[1001];
	
	ll n;
	cin >> n;
	for (ll i = 0; i < n; i++) {
		ll m;
		cin >> m;
		memset(brr, 0, sizeof(brr));
		
		for (ll j = 0; j < m; j++) {
			ll num;
			cin >> num;
			if (brr[num] == 0) {
				arr[num]++;
				brr[num] = 1;
			}
		}
		
	}
	ll ct = 0;
	for (ll i = 0; i < 501; i++) {
		if (arr[i] == n) {
			ct += i;
		}
	}
	
	cout << -1 * ct << endl;
    return 0;
}
