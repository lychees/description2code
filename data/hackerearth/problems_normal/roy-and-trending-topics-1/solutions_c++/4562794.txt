#include<bits/stdc++.h>
#define ll long long 
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   vector<pair<ll,pair<ll,ll> > >vec;
   ll id,old,p,l,c,s;
   int  n;
   cin >>n;
   for(int i=0;i<n;i++){
   	cin >>id >> old >> p >> l >> c >> s;
   	ll first,z;
   	first=50*p+5*l+10*c+20*s;
   	z=first-old;
   	vec.push_back(make_pair(z,make_pair(id,first)));
   }
   sort(vec.begin(),vec.begin()+n);
   vector<pair <ll, pair<ll,ll> > >::iterator it;
   int x=0;
   for(it=vec.begin()+n-1;x<5;it--){
   	cout << it->second.first << " " << it->second.second << endl;
   	x++;
   }
    return 0;
}
