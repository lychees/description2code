t=input()
while t>0:
	t-=1
	n,m1,m2=map(int,raw_input().split())
	r= (n%(m1+m2))
	if (r < m1) : 
		print 0
	elif ((r== m1)or(r >= m2)) :
		print 1
	else :
		r/= m1
		if (r&1) :
			print 1
		else :
			print 0
