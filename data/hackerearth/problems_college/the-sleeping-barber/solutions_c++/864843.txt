#include <iostream>
#include <bits/stdc++.h>

using namespace std;
int main()
{
	int tc;
	cin>>tc;
	while(tc--)
	{
		int g;
		cin>>g;
		long long int M;
		cin>>M;
		long long int gs[g];
		for(int i=0;i<g;i++)
		{
			cin>>gs[i];
		}
		sort(gs,gs+g);
		long long sum=0;
		int cnt=0;
		for(int i=0;i<g;)
		{
			sum+=gs[i];
			if(sum<M)
			{
				cnt++;
				i++;
			}else
			{
				break;
			}
		
		}
		if(sum==M)
			cnt++;
		cout<<cnt<<endl;
	}

	return 0;
}
