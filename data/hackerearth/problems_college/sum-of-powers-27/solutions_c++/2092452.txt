#include<bits/stdc++.h>
using namespace std;

typedef unsigned u_t;
typedef long l_t;
typedef unsigned long ul_t;
typedef long long ll_t;
typedef unsigned long long ull_t;
#define PRE_CPP do { ios_base::sync_with_stdio(0); cin.tie(0); } while(0)

const int MAXN = 1e6;
set<long> s;

void preset() {
    s.insert(1);
    for (long i = 2, j; i < 10000; i++) {
        j = i; 
        j *= j;
        while (j < 10000000) {
            s.insert(j);
            j *= i;
        }
    }
}

int main()
{
	PRE_CPP;
	ostringstream oss;
	int testcase;
	long x;
	bool isPower;
	cin >> testcase;
	preset();
	while (testcase--) {
        cin >> x;
        isPower = false;
        for(auto z:s) {
            if (z >= x)
                break;
            long y = z;
            if (s.find(x - y) != s.end()) {
                isPower = true;
                break;
            }
        }
        oss << (isPower? "Yes": "No") << "\n";
	}
	cout << oss.str();
 	return 0;
}
