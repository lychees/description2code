#include<stdio.h>
#include<math.h>
#include<iostream>
using namespace std;
#define f_in(st) freopen(st,"r",stdin);
#define f_out(st) freopen(st,"w",stdout);
int main()
{
    //f_in("sample_input.txt");
    //f_out("sample_output.txt");
    int test,i,j,ArraySize;
    long long int Count;
    scanf("%d",&test);
    while(test--)
    {
        scanf("%d",&ArraySize);
        int a[ArraySize];
        int cnt[101]={0};
        int count[101]={0};
        Count=0;
        for(i=0;i<ArraySize;i++)
        {
            // a[i]=1;
            scanf("%d",&a[i]);
            cnt[a[i]]++;
        }
        for(i=0;i<ArraySize;i++)
        {
            count[a[i]]++;
            //divisors of a[i];
            for(j=1;j*j<=a[i];j++)
            {
                int num=a[i];
                if(num%j==0)
                {
                    int divisor1=j;
                    int divisor2=num/j;
                    if(divisor1==divisor2)
                        Count+=(cnt[divisor1]-count[divisor1]);
                    else
                    {
                        Count+=cnt[divisor1]-count[divisor1];
                        Count+=cnt[divisor2]-count[divisor2];
                    }
                }
            }
        }
        printf("%lld\n",Count);
    }
    return 0;
}
