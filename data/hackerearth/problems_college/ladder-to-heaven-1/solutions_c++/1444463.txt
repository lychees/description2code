#include<bits/stdc++.h>

using namespace std;

int main(){
	long long int fib[90],i;
	fib[0]=1;
	fib[1]=2;
	
	for(i=2;i<89;i++){
		fib[i]=fib[i-1]+fib[i-2];
	}
	
	/*for(i=0;i<90;i++){
		cout << fib[i] << endl;
	}*/
	
	
	long long int t,ans,n,j;
	cin >> t;
	while(t--){
		cin >> n;
		ans=0;
		j=0;
		while(fib[j]<=n){
			if(fib[j]%2==0){
				ans=ans+fib[j];
			}
			j++;
		}
		cout << ans << endl;
	}
}
