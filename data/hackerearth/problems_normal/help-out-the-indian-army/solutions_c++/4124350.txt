#include<bits/stdc++.h>
#define pii pair<long long,long long>
#define fi first
#define se second
#define mp make_pair
#define vpi vector< pii >
#define pb push_back
#define ll long long int
#define mod 1000000007
#define gcd(a,b) __gcd(a,b)
#define sf(n) scanf("%lld",&(n))
#define pf(n) printf("%lld\n",(n))
#define setbits(n) __builtin_popcount((n))
#define all(v) (v).begin(),v.end()
#define pq priority_queue
using namespace std;
int main()
{
    ll n,s,e;
    sf(n);sf(s);sf(e);
    vector< pii >v;
    for(int i=0;i<n;i++)
    {
        ll a,b;
        cin>>a>>b;
        ll a1=a-b;
        ll a2=a+b;
        if(a1>=e || a2<=s)
            continue;
        v.pb(mp(max(a1,s),min(a2,e)));
    }
    if(v.size()==0)
        pf((e-s));
    else
    {
        sort(all(v));
        ll ans=max(0LL,v[0].fi-s);
        ll b=v[0].se;
        for(int i=1;i<v.size();i++)
        {
            if(v[i].se>b)
            {
                ans+=max(0LL,v[i].fi-b);
                b=v[i].se;
            }
        }
        ans+=max(0LL,e-b);
        pf(ans);

    }
}
