#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	n*=2;
    	int arr[n];
    	for(int i=0;i<n;i++) cin>>arr[i];
    	sort(arr,arr+n);
    	int minisum=0;
    	for(int i=0;i<n/2;i++)
    	{
    		int temp1=arr[i];
    		int temp2=arr[n-1-i];
    		minisum=max(minisum,temp1+temp2);
    	}
    	cout<<minisum<<"\n";
    }
    return 0;
}
