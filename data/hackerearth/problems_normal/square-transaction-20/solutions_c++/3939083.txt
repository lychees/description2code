#include <iostream>
#include <cstdio>

using namespace std;

int binary_search(int a[],int n,int key)
{
	int lb=0,ub=n-1,mid;
	while(lb<=ub){
		mid=(lb+ub)/2;
		if(a[mid]>=key && (mid==0 || a[mid-1]<key))
			return mid;
		else if(a[mid]<key)
			lb=mid+1;
		else
			ub=mid-1;
	}
	return -1;
}

int main()
{
	int n,q,i,j;
	scanf("%d",&n);
	int a[n];
	for(i=0;i<n;i++)
		scanf("%d",&a[i]);
	for(i=1;i<n;i++)
		a[i]+=a[i-1];
	scanf("%d",&q);
	while(q--){
		scanf("%d",&j);
		i = binary_search(a,n,j);
		if(i!=-1)
			i++;
		printf("%d\n", i);
	}
	return 0;
}
