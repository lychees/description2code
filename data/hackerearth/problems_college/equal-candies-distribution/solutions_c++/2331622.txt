#include<bits/stdc++.h>
using namespace std;
int main()
{
int t;
cin>>t;
while(t--)
{
int n;
cin>>n;
int a[10003],b[10003];
for(int i=0;i<n;i++)
cin>>a[i];
sort(a,a+n);
memset(b,0,sizeof(b));
for(int i=0;i<n;i++)
 b[i]=a[i]-a[0];
int s=0,c=0;
for(int i=0;i<n;i++)
{
s=b[i];
c=c+s/5;
s=s%5;
c=c+s/2;
s=s%2;
c=c+s/1;
}
cout<<c<<endl;
}
return 0;
}
