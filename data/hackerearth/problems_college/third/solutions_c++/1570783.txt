#include <iostream>
using namespace std;
 
long long int fun(long long int arr[],long long int j)
{
	long long int table[100004]={0};
	table[0]=arr[0];
	table[1]=max(arr[0],arr[1]);
	for(long long int i=2;i<j;i++)
	{
		table[i]=max(table[i-2]+arr[i],table[i-1]);
	}
	return table[j-1];
}
 
 
int main()
{
 
    int t;
    cin>>t;
    while(t--)
    {
    long long int j,arr[100000];
 
    cin>>j;
    for(long long int i=0;i<j;i++)
    {
    	cin>>arr[i];
 
    }
 
 
    cout<<fun(arr,j)<<endl;
 
 
 
    }
    return 0;
}
