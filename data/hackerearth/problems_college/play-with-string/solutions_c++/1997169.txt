#include <iostream>
using namespace std;

int main()
{
	long long T,t,i,j,k;
	cin>>T;
	string S;
	while(T--){
		cin>>S;
		k=1;
		long long n=S.size();
		for(i=1;i<S.size();i++){
			if(abs(S[i]-S[i-1])!=abs(S[n-1-i]-S[n-1-i+1])) k=0;
		}
		if(k==0) cout<<"Not Funny\n";
		else cout<<"Funny\n";
	}
    return 0;
}
