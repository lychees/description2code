#include <iostream>

int main()
{
    long long unsigned n,k,i,j,a[100000],max=0,l,temp;
    
    scanf("%llu %llu",&n,&k);
    
    for(i=0;i<n;i++)
    {
    	scanf(" %llu",&a[i]);
    }
  
    for(i=0;i<k && i<n;i++)
    {
    	max=0;
    	for(j=i;j<n;j++)
    	{
    		if(max<a[j])
    		{
    			max=a[j];
    			l=j;
    		}
    	}
    	if(i==l && n>100)//why ?? 
    		k++;
    	temp=a[i];
    	a[i]=a[l];
    	a[l]=temp;
    }
    
    for(i=0;i<n;i++)
    {
    	printf("%llu ",a[i]);
    }
    return 0;
}
