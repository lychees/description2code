#include <iostream>
#include <string>
using namespace std;

int main()
{
    int t;
    	cin >> t;
		while(t--){
    		string p;
    			cin>>p;
    		string ans ="";
    		for(int i=0; i<p.length();){
    			int k = 0;
    			while(p[i]==p[i+1+k]){
    				k++;
    			}
    			ans += p[i];
    			i = i+1+k;
    		}
    		cout << ans << endl;
    	}
    return 0;
}
