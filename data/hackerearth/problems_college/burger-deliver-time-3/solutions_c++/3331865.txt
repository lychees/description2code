    #include<iostream>
    #include<bits/stdc++.h>
    using namespace std;
    int main() {
    	int n,x,y,i;
    	cin >> n;
    	vector < pair < int,int > > v;
    	for(i=0; i<n; i++) {
    		cin >> x >> y;
    		v.push_back(make_pair(x+y,i+1));
    	}
    	cout << endl;
    	sort(v.begin(),v.end());
    	for(i=0; i<v.size(); i++) {
    		cout << v[i].second << " ";
    	}
    	cout << endl;
    	return 0;
    }
