#include <iostream>
using namespace std;
 
long long pwr2(long long y)
{
	long long x=1;
	while(x<=y)
	{
		x=2*x;
	}
	x=x/2;
	if(x==y)return 1;
	else
	{
		
		return 2*(y-x)+1;
		
	}
	
}
 
int main()
{
    int n;
    scanf("%d",&n);
    long long x;
    while(n--)
    {
    	scanf("%ld",&x);
    	printf("%ld\n",pwr2(x));
    }
    return 0;
}
