using namespace std;
int a[10];
int q(int n)
{int rem;
	while(n)
	{
		rem=n%10;
		n/=10;
		//cout<<a[rem];
		if(a[rem]==1)
		{
			return 1;
		}
	}
	return 0;
}
int main() {
	// your code goes here
	int t;
	cin>>t;
	while(t--)
	{
		long long n,rem,c=0,i,n1;
		cin>>n;
		n1=n;
		
		for(i=0;i<10;i++)
		a[i]=0;
		while(n)
		{
			rem=n%10;
			n/=10;
			if(a[rem]==0)
			a[rem]=1;
		}
		for(i=1;i*i<=n1;i++)
		{
			if(n1%i==0)
			{
				c+=q(i);

			}
		}
		--i;
		if(i*i==n1)
		c-=q(i);
		cout<<c<<endl;
	}

}
