#include <bits/stdc++.h>
using namespace std;

int n, k, cost[10000+1];
long long dp[10000+1];

long long solve(int idx) {
	if(idx >= n) return 0;

	if(dp[idx] != -1) return dp[idx];

	long long ret = max(solve(idx+1), cost[idx] + solve(idx+k+1));

	return dp[idx] = ret;
}

int main() {
	int t;
	cin>>t;
	while(t--) {
		 cin>>n>>k;

		 for(int i=0;i<n;++i) cin>>cost[i];

		 memset(dp, -1, sizeof(dp));

		 cout << solve(0) << endl;
	}
}
