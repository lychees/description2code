#include<bits/stdc++.h>
#define ll long long
#define pi 3.1415926536
#define N 6000005
#define mod 1000000007


using namespace std;
ll n,m,t,i,j,maxx,minn,temp,flag ;



int main()
{
	scanf("%lld",&t);
		ll dp[N];
		dp[1] = 1;
		dp[2] = 2;
		for(i = 3; i < N; i++)
        {
            dp[i] = dp[i-1] + (i-1)*dp[i-2];
            if(dp[i] > mod)
                dp[i] %= mod;
        }
	while(t--)
	{
		scanf("%lld",&n);
        printf("%lld\n",dp[n]);

	}




    return 0;
}
