#include <iostream>
using namespace std;

int main()
{
    int t,i,d,wt,total;
    cin>>t;
    for(i=1;i<=t;i++)
    {
    	cin>>d>>wt;
    	    if(d<=2)
    		{
    		total=50+(wt*2);
    		cout<<total<<"\n";}
    	
    else if(d>60)
    {
    	cout<<"NA\n";}
    	
    	else
    	{
    		total=50+(wt*2)+((d-2)*12);
    		cout<<total<<"\n";
    	}
    }
    return 0;
}
