#include <iostream>
using namespace std;

int main()
{
    int n,m;
    cin>>n>>m;
    int a[n][m];
    int row_sum[n];
    int col_sum[m];
    for(int i=0;i<n;i++){
    	for(int j=0;j<m;j++){
    		cin>>a[i][j];
    	}
    }
    for(int i=0;i<n;i++){
    	row_sum[i]=0;
    	for(int j=0;j<m;j++){
    		row_sum[i]=row_sum[i]+a[i][j];
    	}
    }
    for(int i=0;i<m;i++){
    	col_sum[i]=0;
    	for(int j=0;j<n;j++){
    		col_sum[i]=col_sum[i]+a[j][i];
    	}
    }
    int max=0;
    int temp=0;
    for(int i=0;i<n;i++){
    	for(int j=0;j<m;j++){
    		temp=row_sum[i]^col_sum[j];
    		if(temp>max)max=temp;
    	}
    }
    cout<<max<<endl;
    return 0;
}
