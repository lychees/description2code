#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
int comp(string X,string Y)
{
    string XY=X.append(Y);
    string YX=Y.append(X);
    return XY.compare(YX)>0?0:1;
}
vector <string> numbers(5000);
int main()
{
    int n, i;
    cin>>n;
    for(i=0; i<n; i++)
        cin>>numbers[i];
    sort(numbers.begin(), numbers.begin()+n, comp);
    for(i=0; i<n; i++)
        cout<<numbers[i];
    return 0;
}
