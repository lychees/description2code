#include <bits/stdc++.h>
#include<stdlib.h>
using namespace std;
int main()
{
    int n,i,j,a[10],p=0,q=0;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
    	scanf("%d",&a[i]);
	}
	for(i=0;i<n;i++)
	{
		p=0;
	    for(j=1;j<=a[i];j++)
	    {
	    	if(__gcd(j,a[i])==1)
	    	{
	    		p++;
	   		}
	   	}
	   	if(p==1)
	   		printf("FALSE\n");
	   	else
	   	{
	   		q=0;
		    for(j=2;j<p;j++)
		    {
		    	if(p%j==0)
		    	{
		   			q++;
		   			break;
		   		}
		   	}
		   	
			if(q==0)
			{
		   		printf("TRUE\n");	
			}
			else
			{
				printf("FALSE\n");
			}
		}
	}
    return 0;
}
