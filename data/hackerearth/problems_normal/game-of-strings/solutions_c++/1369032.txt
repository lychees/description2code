#include<bits/stdc++.h>
using namespace std;

string str1,str2;
int dp[5001][5001];

void solve() {
    int len1=str1.size();
    int len2=str2.size();
    for(int i=1;i<=len1;i++) {
        for(int j=1;j<=len2;j++) {
            if(str1[i-1]==str2[j-1]&&str1[i-1]!='a'&&str1[i-1]!='e'&&str1[i-1]!='i'&&str1[i-1]!='o'&&str1[i-1]!='u') {
                dp[i][j]=dp[i-1][j-1]+1;
            } else {
                dp[i][j]=max(dp[i][j-1],dp[i-1][j]);
            }
        }
    }
    printf("%d\012",dp[len1][len2]);
    return;
}
int main() {
    cin>>str1>>str2;
    solve();
    return 0;
}
