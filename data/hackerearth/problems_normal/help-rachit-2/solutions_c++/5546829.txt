#include<iostream>

#define MAX 10000000

using namespace std;

int   p[MAX + 10 ];
int prime[MAX+10];
int sum[10000+100];

int len;

void seive()
{
	p[1] = 1;
	
	len = 0;
	
	prime[len++] = 2;
		
	for(long long int i = 4; i<=MAX; i += 2 )
	 {
	 	p[i] = 1 ;
	 }
	 
	for(long long int i = 3; i<=MAX; i += 2)
	 {
	 	if( p[i] == 0 )
	 	 {
	 	 	prime[len++] = i;
	 	 	
	 	 	for( long long j = i + i  ; j<=MAX; j += i )
	 	 	 {
	 	 	 	 p[j] = 1;
	 	 	 }
	 	 }
	 }
}

int check_prime(long long int n)
{
	for(int i = 0; prime[i]*prime[i]<n; i++ )
	 {
	 	if( n%prime[i] == 0 )
	 	 return 0;
	 }
	 
	return 1;
}

int main()
{
	int t;

	cin >> t;
	
	seive();
		
	sum[1] = 0;
	
	for(int i=3; i<=10000 ; i +=2 )
	 {	
	    //cout << i << "\n" ;
	    
	 	sum [ i ] = sum[ i - 2];
	    
	    for(int j = 1; j<=3; j++ )
	     {
	     	long long int x = i*i - j*(i-1) ;
	     	
	     	if( x <= MAX )
			 {
			 	if( p[x] == 0 )
			 	 {
			 	 	sum[i] ++;
			 	 }
			 }
		    else
			   if( check_prime( x ) == 1 ) sum[i] ++;	 
	     }
	 }
	 	    
	while( t-- )
	 {
	 	int n;
	 	
	 	cin >> n;
	 	
	 	long double answer = ( (long double) sum[n]  / ( n+n-1)  )*100;
	 	
	 	printf("%0.6Lf\n",answer);
	 }

	return 0;
}
