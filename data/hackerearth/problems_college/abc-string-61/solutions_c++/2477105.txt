#include <iostream>
using namespace std;
#define M 1000000007
long long power(long long a,long long b)
{
	if(b==0)
	return 1;
	long long x=power(a,b/2);
	if(b%2==0)
	return (x*x)%M;
	else
	return (((x*x)%M)*a)%M;
}
	
int main()
{
    long long a,b,n;
    int t;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	a=power(3,n);
    	b=3*power(2,n)%M;
    	a=(a+3-b+M)%M;
    	cout<<a<<"\n";
    }
}
