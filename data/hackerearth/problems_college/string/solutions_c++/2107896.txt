#include <bits/stdc++.h>

using namespace std;

int k,i,min_i,h[200];
string s;

int main()
{
    cin >> s >> k;
    for(i=0;i<s.length();++i)
        ++h[s[i]];
    for(i=97;i<123;++i)
        if(h[i] == 0)
            h[i] = INT_MAX;
    while(1)
    {
        min_i = 97;
        for(i=97;i<123;++i)
            if(h[i] < h[min_i])
                min_i = i;
        if(h[min_i] <= k)
        {
            k -= h[min_i];
            h[min_i] = INT_MAX;
        }
        else
            break;
    }
    for(i=97,k=0;i<123;++i)
        if(h[i] != INT_MAX)
            ++k;
    printf("%d\n",k);
    for(i=97;i<123;++i)
        if(h[i] != INT_MAX)
            for(k=0;k<h[i];++k)
                putchar(i);
    putchar('\n');
    return 0;
}
