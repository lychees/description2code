#include <bits/stdc++.h> 
using namespace std;
#define ll long long
#define se second
#define fi first
#define pb push_back
#define mp make_pair
#define MOD 1000000007

int main()
{ 
  ll i,ans,a[100005],t,n,pre[100005],pro;
  cin>>t;

   while(t--)
   {
   	memset(pre,0,sizeof(pre));
   	cin>>n;
   	for(i=0;i<n;i++)
   		cin>>a[i];
   	sort(a,a+n);
   	pre[0]=a[0];
   	for(i=1;i<n;i++)
   		pre[i]=((pre[i-1])+(a[i]));
   	pro=1;ans=0;
   	for(i=1;i<n;i++)
   	{
   		ans=(ans)+((((pro)*(a[i])))-(pre[i-1]))%MOD;
   		pro++;
   	}
   	cout<<((ans%MOD)*(a[n-1]%MOD))%MOD<<endl;
   }

	return 0;
}
