#include <iostream>
#include<algorithm>
#include<cstdio>
using namespace std;

int main()
{
    int t,n;
    scanf("%d",&t);
    while(t--)
    {
    	long long a[100005];
    	scanf("%d",&n);
    	for(int i=0;i<n;i++)
    	scanf("%lld",&a[i]);
    	sort(&a[0],&a[n]);
    	for(int i=0;i<n;i++)
    	printf("%lld ",a[i]);
    	printf("\n");
    }
    return 0;
}
