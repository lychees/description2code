#include <iostream>
#include <stack>
#include <string.h>
using namespace std;

int main()
{
    int m;
    cin>>m;
    long long int ans = 0;
    while(m--)
    {
    	stack<char> stk;
    	string s;
    	cin>>s;
    	long long int l = s.size();
    	for(long long int i = 0; i < l; i++)
    	{
    		if(stk.empty())
    		{
    			stk.push(s[i]);
    		}
    		else
    		{
    			if(stk.top() == s[i])
    				stk.pop();
    			else
    				stk.push(s[i]);
    		}
    	}
    	if(stk.empty())
    		ans++;
    }
    cout<<ans<<endl;
    return 0;
}
