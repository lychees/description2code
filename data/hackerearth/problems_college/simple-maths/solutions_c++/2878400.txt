#include<bits/stdc++.h>
using namespace std;
int gcd(int a,int b)
{
	if(b==0)
	return a;
	else
	return gcd(b,a%b);
}
int main()
{
	unsigned long long int t;
	cin>>t;
	while(t--)
	{
		unsigned long long int n,a,b;
		cin>>n>>a>>b;
		if(a==0 || b==0 || (a==0 &&  b==0))
		{
			cout<<"-1"<<"\n";
			continue;
		}
		long long int sum =0;
		long long int p = (long long int)n/a;
		long long int q = (long long int)n/b;
		long long int d = a*b/gcd(a,b);
		long long int r = (long long int)n/d;
		sum =  p+q-r;
		cout<<sum<<"\n";
	}
}
