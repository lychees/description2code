#include <iostream>
using namespace std;

int main()
{
	char s1[1000], s2[2000];
	int n, i, x=0, y=0;
	cin>>n;
	while(n--)
	{
		cin>>s1;
		cin>>s2;
		x=0;
		y=0;
		for(i = 0;s1[i]!='\0';i++)
		{
			x+=s1[i];
		}
		for(i = 0;s2[i]!='\0';i++)
		{
			y+=s2[i];
		}
		if(x==y)
			cout<<"Yes\n";
		else
			cout<<"No\n";
	}	
}
