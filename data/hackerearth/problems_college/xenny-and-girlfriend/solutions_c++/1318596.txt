#include <iostream>
#include <vector>
using namespace std;

int main()
{
    //printf("Hello World!\n");
    int t;
    cin>>t;
    int n,m;
    int a,b;
    while(t--)
    {
    	cin>>n>>m;
    	int arr[10001]={0};
    	int i;
    	for(i=0;i<m;i++)
    	{
    		cin>>a>>b;
    		arr[a]++;
    		arr[b]++;
    	}
    	int count=0;
    	for(i=0;i<n;i++)
    	{
    		if(arr[i]%2)
    		count++;
    	}
    	if(count==0 || count==2)
    	cout<<"Yes\n";
    	else
    	cout<<"No\n";
    	
    }
    return 0;
}
