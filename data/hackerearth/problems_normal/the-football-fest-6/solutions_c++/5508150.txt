#include <iostream>
#include <stack>
#include <stdio.h>
using namespace std;

int main()
{
    stack <int> id;
    int t,n,x;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d %d",&n,&x);
    	id.push(x);
    	x=2*n;
    	while(x--)
    	{
    		char ch; int y;
    		scanf("%c",&ch);
    		if(ch=='P')
    		{
    			scanf("%d",&y);
    			id.push(y);
    		}
    		if(ch=='B')
    		{
    			int temp = id.top();
    			id.pop();
    			int temp1= id.top();
    			id.pop();
    			id.push(temp);
    			id.push(temp1);
    		}
    	}
    	printf("Player %d\n",id.top());
    }
    return 0;
}
