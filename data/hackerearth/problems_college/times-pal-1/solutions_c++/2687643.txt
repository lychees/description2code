#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	string a,b;
    	cin>>a;
    	cin>>b;
    	int la = a.length();
    	int lb = b.length();
    	int lc = la+lb;
    	char * c = new char[lc];
    	int k = 0;
    	int count = 0;
    	for(int i=0;i<lb;i++){
    		c[k++]=b[i];
    	}
    	for(int i=0;i<la;i++){
    		c[k++]=a[i];
    	}
    	int i=0;
    	for(i=0;i<lc;i++){
    			if(c[i]!=c[lc-1-i]){
    				break;
    			}
    		}
    	if(i == lc){
    		count++;
    	}
    	k=0;
    	for(int i=0;i<la;i++){
    		c[k++]=a[i];
    		for(int j=0;j<lb;j++){
    			c[k++]=b[j];
    		}
    		int temp = i;
    		while(++i<la){
    			c[k++]=a[i];
    		}
    		i = temp;
    		k = i+1;
    		
    		int x =0;
    		int flag = 1;
    		//c[k]='\0';
    		for(x=0;x<lc;x++){
    			if(c[x]!=c[lc-1-x]){
    				flag = 0;
    				break;
    			}
    		}
    		if(flag == 1){
    			count++;
    		}
    	}
    	cout<<count<<endl;
    	delete []c;
    }
    return 0;
}
