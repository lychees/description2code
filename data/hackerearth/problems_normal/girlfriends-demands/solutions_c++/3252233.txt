#include <bits/stdc++.h>
#define ll long long int
using namespace std;

int main(){
	string s;
	int n, l;
	ll a, b;
	cin >> s;
	l = s.size();
	scanf("%d", &n);
	while (n--){
		scanf("%lld %lld", &a, &b);
		a--;
		b--;
		a = a%l;
		b = b%l;
		if (s[a]==s[b])
			cout<<"Yes"<<endl;
		else
			cout<<"No"<<endl;
	}
	return 0;
}
