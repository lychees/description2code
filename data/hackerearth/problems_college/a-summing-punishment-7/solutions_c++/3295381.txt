#include <iostream>
#define ll long long int
#include <bits/stdc++.h>

using namespace std;

int main()
{
    ll n,a[100005],sum[100005],q,ans,x,l,r;
    
    memset(sum,0,sizeof(sum));
    
    cin>>n;
    for(int i = 1; i <= n; i++) {
    	cin>>x;
    	sum[i] = sum[i-1]+x;
    	//cout<<"i "<<sum[i]<<endl;
    }
    
   	cin>>q;
   	for(int i = 1; i <= q; i++) {
   		cin>>l>>r;
   		if(l > 1) {
   			ans = sum[r]-sum[l-1];
   		}
   		else {
   			ans = sum[r];
   		}
   		
   		cout<<ans<<endl;
   	}
    
    return 0;
}
