#include <iostream>
#include<stdlib.h>
#include<cstdio>
using namespace std;
int cmp(const void*a,const void*b)
{
	return *(int *)a-*(int*)b;
}
int main()
{
  	int t;
  scanf("%d",&t);
  while(t--)
  {
  	int n,k,p;
  	scanf("%d %d %d",&n,&k,&p);
  	int i;
  	int arr[k];
  	for(i=0;i<k;i++)
  	{
  		scanf("%d",&arr[i]);
  	}
  	int pre=p;
  	qsort(arr,k,4,cmp);
  	for(i=0;i<k;i++)
  	{
  		if(arr[i]<=pre)
  		pre++;
  	}
  	if(pre<=n)
  	printf("%d\n",pre);
  	else
  	printf("-1\n");
  }
    return 0;
}
