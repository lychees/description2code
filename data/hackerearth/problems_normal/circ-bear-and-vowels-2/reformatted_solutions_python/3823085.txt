def hard_easy(s):
    vo = "aeiouy"
    count = 0
    vowel_cnt = 0
    consonant_cnt = 0
    for g in s:
        if g not in vo:
            count = count + 1
            consonant_cnt = consonant_cnt + 1
            if count >= 3:
                print "hard"
                return 0
        else:
            vowel_cnt = vowel_cnt + 1
            count = 0
    if consonant_cnt > vowel_cnt:
        print 'hard'
    else:
        print 'easy'
    return 0


t = int(input())
for d in range(t):
    s = str(raw_input())
    hard_easy(s)
