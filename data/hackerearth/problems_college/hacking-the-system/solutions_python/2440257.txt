def main():
    x=raw_input()
    r=raw_input()
    c=0
    if len(x)!=len(r):
        print "loses"
    else:
        for i in range(len(x)):
            if x[i]!=r[i]:
                c=c+1
            if c>=2:
                break
    if c>=2:
        print "loses"
    else:
        print "wins"

if __name__=='__main__':
    main()
