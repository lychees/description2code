#include<iostream>
using namespace std;
void rec(long a[],long st,long ed,long  mid)
{
    long  i=st,j=mid+1,c=0,b[ed-st+1];
    while(i<=mid&&j<=ed)
    {
        if(a[i]>a[j])
            b[c++]=a[i++];
        else
            b[c++]=a[j++];
    }
    while(i<=mid)
        b[c++]=a[i++];
    while(j<=ed)
        b[c++]=a[j++];
    c--;
    while(c>=0)
    {
        a[ed--]=b[c--];
    }
}
void mer(long  a[],long st,long ed)
{
    long  c=(st+ed)/2;
    if(st!=c)
    {
        mer(a,st,c);
        mer(a,c+1,ed);
        rec(a,st,ed,c);
    }
    else if(a[st]<a[ed])
    {
        long  temp=a[st];
        a[st]=a[ed];
        a[ed]=temp;
    }
}

int main()
{
    long n,j,p,i;
    long long coun=1,price=0,prod;
    cin>>n>>p;
    long a[n],k=p;
    for(i=0;i<n;i++)
        cin>>a[i];
    mer(a,0,n-1); //Sorting the array in descending order
    for(i=0;i<n;i++)
    {
    	if(k!=0)
    	{
    		price=price+coun*a[i];
    		k--;
    	}
    	else
    	{
    		k=p;
    		i--;
    		coun++;
    	}
    }
    cout<<price<<endl;
    return 0;
}
