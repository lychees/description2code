#include <iostream>
#include <math.h>
#include <iomanip>
using namespace std;

int main()
{
	
	float ans,p,Time,q;
    int testcases;
    cin>>testcases;
    while(testcases--)
    {
    	cin>>p>>Time;
    	q=1-p;
    	Time=Time/1440.0;
    	ans=pow(q,Time);
    	ans=1-ans;
    	cout<<fixed<<setprecision(4);
    	cout<<ans<<endl;
    }
}
