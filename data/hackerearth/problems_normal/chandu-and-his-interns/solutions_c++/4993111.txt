#include <iostream>
#include <math.h>
using namespace std;

int main()
{
	long int n;
	cin>>n;
	while(n>0)
	{
		long int x,count=0;
		cin>>x;
		if (x<2)
			cout<<"NO"<<endl;
		else
	    {
		    for (long int i=1; i<=sqrt(x); i++)
		    {
		        if (x%i==0)
		        {
		            // If divisors are equal, print only one
		            if (x/i == i)
		               	count=count+1;
		 
		            else // Otherwise print both
		                count=count+2;
		           
		        }
		        if(count>=4)
		        	break;
		        
		    }
		   
		    if(count<4)
		    	cout<<"NO"<<endl;
		    else
		    	cout<<"YES"<<endl;
	    }
	    
		n--; 	
	}
    return 0;
}
