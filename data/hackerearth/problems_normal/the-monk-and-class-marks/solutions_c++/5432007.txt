 #include<bits/stdc++.h>
using namespace std;

int main()
{
int t,n,i;
vector<pair<int,string> > v;
string a;
        cin>>n;
        // when pair is sorted first sorts according to first element if they are equal then checks second 
        // we want to sort in first in decreasing but while comparing second it should be increasing only 
        // insert negative of the number in the pair and make it positive before printing.
for(i=0;i<n;i++){
   cin>>a>>t;
   v.push_back(make_pair(-1*t,a));
}
sort(v.begin(),v.end());

for(i=0;i<v.size();i++){
    cout<<v[i].second<<" "<<-1*v[i].first<<endl;
}

}
