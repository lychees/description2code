ll C(ll n,ll r) {
    if(n<r)
        return 0;
    ll ans=1;
    r = (r>n-r)?n-r:r;
    for(ll i=1;i<=r;i++,n--) {
        if(n%i==0)
            ans *= n/i;
        else
            if(ans%i==0)
                ans = (ans/i)*n;
            else
                ans = (ans*n)/i;
    }
    return ans;
}
