#include <bits/stdc++.h>
#define boost std::ios::sync_with_stdio(false)
using namespace std;
int main()
{
	long int t;
	int len,in= 0;
	boost;
	cin >> t;
	cin.ignore();
	string inp;
	std::deque<string> v;
	while(t--)
	{
		v.clear();
		getline(cin,inp);
		len = inp.length();
		in = 0;
		for (int i = 0; i < len; ++i)
		{
			if(inp[i]==' ' || inp[i]=='\t')
			{
				v.emplace_front(inp.substr(in,i-in));
				in = i+1;
			}
		}
		if(inp[len-1]!=' ' || inp[len-1]!='\t')
		{
			v.emplace_front(inp.substr(in,len-in));
		}
		for(auto x:v)
			cout << x << " ";
		cout << "\n";

	}
	return 0;
}
