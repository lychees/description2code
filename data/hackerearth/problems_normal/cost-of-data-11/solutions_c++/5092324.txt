#include <bits/stdc++.h>
using namespace std;
#include <stdio.h>
int ans;
struct node{
	 map<int,node*> child;
};
char a[100005];
node* getNode()
{
   // printf("making new node\n");
	node* cur = (node*)malloc(sizeof(node));
	ans++;
    map<int,node*> gr ;
    cur->child = gr;
	int i;
	//for(i=0;i<26;i++)
	//cur->child[i] = NULL;
    return cur;
}
void insert(node* root)
{
	int i;
	for(i=0;a[i];i++)
	{
		int in = a[i]-'a';
       // printf("inserting %c values are %d\n",a[i],root->child.size());
		if(root->child.find(in)==root->child.end())root->child[in] = getNode();
		root = root->child[in];
	}
}
int main()
{
   // printf("Hello World!\n");
   int n;
   scanf("%d",&n);
   node* root;
   if(n)root = getNode();
   while(n--)
   {
   	scanf("%s",&a);
   	insert(root);
   }
   printf("%d\n",ans);
    return 0;
}
