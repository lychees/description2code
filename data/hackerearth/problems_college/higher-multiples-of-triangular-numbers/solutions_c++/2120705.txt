#include<iostream>
using namespace std;
int main()
{
	long long int t,n,i,j,k=0,c=0;
	cin>>t;
	while(t--)
	{
		cin>>n;
		k=0;
		for(i=1;;i++)
		{   c=0;
		    k=i+k;
			for(j=1;j<=k;j++)
			{
				if(k%j==0)
				c++;
			}
			if(c>n)
			break;
		}
		cout<<k<<endl;
	}
	return 0;
}
