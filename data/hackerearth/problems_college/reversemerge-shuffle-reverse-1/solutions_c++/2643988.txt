#include <iostream>
#include <vector>
#include <deque>
#include <string>
#include <algorithm>

using namespace std;

bool dom(vector<int>& x1, vector<int>& x2, int n){
    for(int i=0;i<n;i++){
        if(x1[i] < x2[i]) return false;
    }
    return true;
}

int main(){
    string S;
    cin >> S;
    int n = S.length();
    vector<int> a(n);
    vector<int> cs(26,0);
    vector<vector<int> > left(n+1, cs);
    vector<deque<int> > cinds(26);
    
    
    

    for(int i=0;i<n;i++){
        a[i] = S[n-1-i]-'a';
        cinds[a[i]].push_back(i);
    }
     
    for(int i=n-1;i>=0;i--){
        for(int j=0;j<26;j++){
            left[i][j]=left[i+1][j];
        }
        left[i][a[i]]++;
    }

    vector<int> fin(26, 0);
    
    for(int i=0;i<26;i++) fin[i]= left[0][i]/2;
    
    
    
    int lp = 0;
    string T(n/2, '.');
    for(int t=0;t<n/2;t++){
        for(int i=0;i<26;i++){
            if(fin[i]>0 && dom(left[cinds[i].front()], fin, 26)){
                T[t] = (char)('a' + i);
                fin[i]--;
                lp = cinds[i].front();
                i = 26;
            }
        }
        
        for(int i=0;i<26;i++){
            while((!cinds[i].empty()) && cinds[i].front() <=lp){
                cinds[i].pop_front();
            }
        }
    }
    
    cout << T << endl;
}
