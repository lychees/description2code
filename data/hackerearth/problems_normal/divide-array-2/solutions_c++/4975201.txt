#include <iostream>
using namespace std;

int main()
{
    int N, Q;
    
    cin >> N;
    
    int A[N];
    
    for(int i = 0; i < N; i++) {
    	cin >> A[i];
    }
    
    cin >> Q;
    
    int multiplier = 1;
    
    for(int i = 0; i < Q; i++) {
    	int a;
    	cin >> a;
    	multiplier *= a;
    }
    
    for(int i = 0; i < N; i++) {
    	if(multiplier == 0) {
    		A[i] = 0;
    	}else {
    		A[i] /= multiplier;
    	}
    }
    
    for(int i = 0; i < N; i++) {
    	cout << A[i] << " ";
    }
    
    return 0;
}
