	ll arr[N];
	for(int i=0;i<N;i++)cin>>arr[i];
	
	for(int i=0;i<N;i++)
	{
		if(arr[i] != 0)
		Q.push_back(arr[i]);
		
		else if(arr[i] == 0)

			if(Q.empty() == false)
