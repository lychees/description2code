ll id[17][17];
void mul(ll a[17][17],ll b[17][17]){
ll c[17][17];
for(int i=0;i<17;i++){
    for(int j=0;j<17;j++){
        c[i][j]=0;
        for(int k=0;k<17;k++){
            c[i][j]+=(a[i][k]*b[k][j])%mod;
            c[i][j]%=mod;
        }
    }
}
for(int i=0;i<17;i++){
    for(int j=0;j<17;j++){
        a[i][j]=c[i][j];
    }
}
}
void bpow(ll a[17][17],ll n){
    memset(id,0,sizeof(id));
    for(int i=0;i<17;i++) id[i][i]=1;
    while(n>0){
        if(n&1) mul(id,a);
        mul(a,a);
        n/=2;
    }
}
ll bpow1(ll x,ll n){
ll ans=1;
while(n>0){
    if(n&1) ans*=x;
    x*=x;
    ans%=mod;
    x%=mod;
    n/=2;
}
return ans;
}
