#include <iostream>
using namespace std;

int main()
{
    int A, B, N, temp;
    cin >> A >> B >> N;
    if(N == 1) cout << A << endl;
    else if(N == 2) cout << B << endl;
    else {
	    while(N-- > 2) {
	    	temp = A + B;
	    	A = B;
	    	B = temp;
	    }
	    cout << temp << endl;
    }
    return 0;
}
