    #include <iostream>
    #include <cstdio>
    #include <cstring>
    using namespace std;
    int solve(char arr[],int l)
    {
    	if(l == 1)
    	return 0;
    	int dels = 0;
    	int j = 1,k =0;
    	for(int i = 0;i<l-1;i++)
    	{
    		
    		if(abs(arr[k] - arr[j]) != 1)
    		{
    			dels++;
    			j +=1;
    		}
    		else
    		{
    			k = j;
    			j += 1;
    		}
    		
    	}
    	return dels;
    }
    int main()
    {
        int t;
        cin>>t;
        while(t--)
        {
        	char arr[200000] = {0};
        	cin>>arr;
        	int l = strlen(arr);
        	printf("%d\n",solve(arr,l));
        	
        }
        return 0;
    }
