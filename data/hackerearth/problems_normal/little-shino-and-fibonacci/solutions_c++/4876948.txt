#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
long long int fibo[200000];
long long int sum_block=0;
long long int calc(long long int x)
{
	long long int no_block=(x/15000)%mod;
	x%=15000;
	long long int sum=(no_block*fibo[14999])%mod;
	sum+=fibo[x];
	sum%=mod;
	return sum;
}
int main()
{
	fibo[1]=0;
	fibo[2]=1;
	for(int i=3;i<=15000;i++)
	{
		fibo[i]=fibo[i-1]+fibo[i-2];
		fibo[i]%=10000;
	}
	fibo[0]=fibo[15000];
	for(int i=1;i<=15000;i++)
	{
		fibo[i]+=fibo[i-1];
		fibo[i]%=mod;
		sum_block+=fibo[i];
		sum_block%=mod;
	}
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long int l,r;
		scanf("%lld %lld",&l,&r);
		long long int sum=calc(r)-calc(l-1);
		sum%=mod;
		while(sum<0)
		{
			sum+=mod;
		}
		printf("%lld\n",sum);
	}
	return 0;
}
