#include <iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    char arr[30][30];
    int n,m;
    for(int i=0;i<t;i++){
    	cin>>n>>m;
    	for(int j=0;j<n;j++){
    		for(int k=0;k<m;k++){
    			cin>>arr[j][k];
    		}
    	}
    	int ans=0;
    	for(int j=n-1;j>=0;j--){
    		for(int k=m-1;k>=0;k--){
    			if(arr[j][k]=='0'){
    				ans++;{
    					for(int l=0;l<=j;l++){
    						for(int p=0;p<=k;p++){
    							if(arr[l][p]=='1')
    								arr[l][p]='0';
    							else
    								arr[l][p]='1';
    						}
    					}
    				}
    			}
    		}
    	}
    	cout<<ans<<"\n";
    }
    return 0;
}
