#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define pb push_back
#define mp make_pair
ll tim[100010], sc[100010];
vector<pair<ll, ll> > v;
int main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);
	ll t;
	// cin>>t;
	t = 1;
	while (t--)
	{
		ll n, q;
		cin >> n >> q;

		for (ll i = 0; i < n; i++)
		{
			cin >> tim[i];
			v.pb(mp(0, tim[i]));
		}
		for (ll i = 0; i < n; i++)
		{
			cin >> sc[i];
			v[i].first = sc[i];
		}
		sort(v.begin(), v.end());
		while (q--)
		{
			ll ans = 0;
			ll k;
			cin >> k;
			for (int i = n - 1; i >= n - k; i--)
			{
				ans += v[i].second;
			}
			cout << ans << "\n";
		}
	}
	return 0;
}
