string binaryconverter(LL x)
    {

		string s;
        if ( !x ) return s="0";

        s = bitset<numeric_limits<unsigned long long>::digits> ( x ).to_string();

        string::size_type n = s.find( '1' );
        return s.substr( n );
    }

LL fastmultiplier(LL a, LL x,int n)
{
    string s=binaryconverter(x);
	LL r=1;

	for(string::iterator iter=s.begin();iter!=s.end();iter++)
		{
				r=(r*r)%n;
				if(*iter=='1')
					{
						r=(a*r)%n;
					}

		}
	return r;

}


int main()
{
	int t;
    long long  x,y,result,mod=1000000007,i;
    char a[100000],b[100000];
    scanf("%d",&t);
    while(t--){
    	x=0;
    	y=0;
    	scanf("%s %s",a,b);
    	for(i=0;a[i]!='\0';i++){
    		x=( x*10 + a[i]-'0' ) % mod;
    	}
    	for(i=0;b[i]!='\0';i++){
    		y=( y*10 + b[i]-'0' ) % (mod-1);
    	}
       result= fastmultiplier(x,y,mod);
        P("%lld\n",result);

    }
}
