#include<bits/stdc++.h>
#define ll long long int

using namespace std;

double p;
double dp[1000007];

double fun(int n) {
    
    if (n < 0) return 0.0;
    if (n == 0) return 1.0;
	if ( dp[n] != -1.0) 
      return dp[n];
	 
	return dp[n] = p * fun(n-2) + (1.0 - p) * fun(n-3);
}

int main() {
	ll n,i;
	cin>>n>>p;
	p = p/100.00;
    
    for(i=0;i<1000007;i++)
    	dp[i] = -1.0;

    
    printf("%0.6f\n",fun(n));
	return 0;
}
