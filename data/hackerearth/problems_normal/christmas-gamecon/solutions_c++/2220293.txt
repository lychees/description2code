#include <iostream>
#include<vector>
using namespace std;

int main()
{
    int t,n;
    cin>>t;
    
    while(t--)
    {
    	cin>>n;
    	long long int result=0;
    	vector<long long int> v(n);
    	for(int i=0;i<n;i++)
    	{
    		cin>>v[i];
    		v[i]=v[i]%5;
    		
    		result=result^v[i];
    	}
    	if(result==0)
    	cout<<"Andrew\n";
    	
    	else
    	cout<<"Aniruddha\n";
    	
    	
    }
    return 0;
}
