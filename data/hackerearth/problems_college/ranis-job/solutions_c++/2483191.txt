#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,i;
    cin>>t;
    while(t--){
    	int n,sum=0,c=0;
    	cin>>n;
    	int a[n],b[n];
    	for(i=0;i<n;i++){
    		cin>>a[i];
    		sum+=a[i];
    	}
    	for(i=0;i<n;i++){
    		cin>>b[i];
    	}
    	sort(b,b+n);
    	for(i=n-1;i>=0;i--){
    		sum=sum-b[i];
    		if(sum>0){
    			c++;
    		}
    		else{
    			break;
    		}
    }
    cout<<c+1<<endl;
}
}
