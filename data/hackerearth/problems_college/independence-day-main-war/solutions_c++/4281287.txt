#include <bits/stdc++.h>
using namespace std;

double sum_x,sum_y;

int main()
{
    int t,n;
    double x,y;
    double ans_x,ans_y;
    double a= -std::numeric_limits<double>::infinity();
    double b= std::numeric_limits<double>::infinity();
    scanf("%d",&t);
    while(t--)
    {
        sum_x=0,sum_y=0;
        scanf("%d",&n);
        for(int i=0;i<n;i++)
        {
            scanf("%lf%lf",&x,&y);
            if(x>=a && x<=b && y>=a && y<=b)
            {
                sum_x=sum_x+x;
                sum_y=sum_y+y;
            }
        }
        ans_x=sum_x/n;
        ans_y=sum_y/n;
        printf("(%.2lf , %.2lf)\n",ans_x,ans_y);
        
    }
    return 0;
}
