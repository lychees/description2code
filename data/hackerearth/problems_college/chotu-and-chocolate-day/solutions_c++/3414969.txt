#include<bits/stdc++.h>
using namespace std;
int main () {
    int t,n,k;
    cin >> t;
    while ( t-- ) {
        cin >> n >> k;
        int a,d,sum = 0;
        a = n*n - ( n - k + 1 ) + 1;
        d = n - k + 1;
        for ( int i = 0 ; i < n ; i++ ) {
            sum = sum + a;
            a = a-d;
        }
        cout << sum <<endl ;

    }
    return 0 ;
}
