#include <iostream>
using namespace std;

int main()
{
	int t;
    //cout << "Enter no of test cases\n" << endl;
    cin>>t;
    for(int a=1;a<=t;a++)
    {
    	int l,r;
    	cin>>l>>r;
    	int max=0;
    	for(int i=l;i<=r;i++)
    	{
    		for(int j=l;j<=r;j++)
    		{
    			int x=i^j;
    			if(x>max)
    			{
    				max=x;
    			}
    		}
    	}
    	cout<<max<<endl;
    }
    return 0;
}
