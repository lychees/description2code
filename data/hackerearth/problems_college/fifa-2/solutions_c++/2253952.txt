#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int n;
   scanf("%d",&n);
   int i;
   int hash[100002]={0},a[100002],b[100002];
   for(i=0;i<n;i++)
   {
   	scanf("%d %d",&a[i],&b[i]);
   	hash[a[i]]++;
   }
   for(i=0;i<n;i++)
   {
   	printf("%d %d\n",n-1+hash[b[i]],(n-1)*2-n+1-hash[b[i]]);
   }
return 0;
}
