test_cases = int(raw_input())
for i in range(test_cases):
    students, required = raw_input().split()
    students = int(students)
    required = int(required)
    a = [0]*students
    a = raw_input().split()
    count = 0
    for i in range(students):
        a[i] = int(a[i])
        if a[i] <= 0:
            count += 1
    if count >= required:
        print "NO"
    else:
        print "YES"
