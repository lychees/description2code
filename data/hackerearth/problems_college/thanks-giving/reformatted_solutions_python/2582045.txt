str = raw_input()
dict = {}
max = 0
lst = []
state = 0

for char in str:
    dict[char] = dict.get(char, 0) + 1


for key, value in dict.items():
    if max < value:
        max = value

for key, value in dict.items():
    if value == max:
        lst.append(key)

for key in lst:
    if key in 'head rush' and key in 'cubing club':
        state = 3

    elif key in 'head rush':
        if state == 0 or state == 1:
            state = 1
        else:
            state = 3

    elif key in 'cubing club':
        if state == 0 or state == 2:
            state = 2
        else:
            state = 3

if state == 0:
    print "none"
elif state == 1:
    print "Thanks head rush"
elif state == 2:
    print "Thanks cubing club"
else:
    print "Thanks both"
