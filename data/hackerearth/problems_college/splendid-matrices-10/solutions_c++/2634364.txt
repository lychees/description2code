#include <iostream>
#include <cmath>
using namespace std;
int a[1030][1030]={0};

void generate(int num1,int num2,int p,int q,int r, int s)
{
  int numdig = (num2-num1+1);
  if(numdig==1)
  {
  	a[p][q]=num1;
  	return;
  }
  int u=(r-p+1)/2;
  generate(num1,num1+numdig/4-1,p,q,p+u-1,q+u-1);
  generate(num1+3*numdig/4,num1+numdig-1,p+u,q+u,r,s);
  generate(num1+numdig/4,num1+numdig/2-1,p,q+u,r-u,s);
  generate(num1+numdig/2,num1+3*numdig/4-1,p+u,q,r,s-u);  
}



main()
{
	int n;
	cin>>n;
	generate(1,pow(2,n)*pow(2,n),1,1,pow(2,n),pow(2,n));
	for(int i=1;i<=pow(2,n);i++)
	{
		for(int j=1;j<=pow(2,n);j++)
		{
			cout<<a[i][j]<<" ";
		}
		cout<<"\n";
	}
    return 0;
}
