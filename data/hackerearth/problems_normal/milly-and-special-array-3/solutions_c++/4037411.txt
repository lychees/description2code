#include <iostream>
#include <climits>
#include <cstdio>
#include <algorithm>
#define MAX 100007
using namespace std;
int a[100010];
int leftDP[100010];
int rightDP[100010];
int main ( )
{
    int t;
    cin >> t;
    while(t--) {
        int n;
        cin >> n;
        rightDP[n+1] = 0;
        leftDP[0] = 0;
        for ( int i = 1 ; i <= n ; i++ ) scanf ( "%d" , &a[i] );
        for ( int i = 1 ; i <= n ; i++ ) {
            if ( a[i] >= 0 ) leftDP[i] = leftDP[i-1] + 1;
            else leftDP[i] = leftDP[i-1];
        }
        for ( int i = n ; i >= 1 ; i-- )
            if ( a[i] <= 0 ) rightDP[i] = rightDP[i+1] + 1;
            else rightDP[i]= rightDP[i+1];
        int ans = INT_MAX;
        for ( int i = 1 ; i < n ; i++ )
            ans = min ( ans , leftDP[i] + rightDP[i+1] );
        printf ( "%d\n" , ans );
    }
}
