#include <iostream>
using namespace std;
#include<stdio.h>
#include<string.h>
int main()
{
	int t;
	char s[100];
	scanf("%d",&t);
	while(t--)
	{
		int flag=0;
		scanf("%s",s);
		for(int i=0;i<strlen(s);i++)
		{
			for(int j=i+1;j<strlen(s);j++)
			{
				if(s[i]==s[j])
				{
					flag=1;
					break;
				}
			}
			if(flag==1)
			{
				break;
			}
		}
		if(flag==1)
		{
			printf("Yes\n");
			
		}
		else
		{
			printf("No\n");
		}		
	}
	return 0;
}
