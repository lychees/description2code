#include <iostream>
#include <string>
#include <vector>
using namespace std;

int main()
{
    long n, q, i, count;
    string in;
    //string temp;
    vector <long> out;
    cin >> n >> in >> q;
    for (i = 0, count = 0; i < in.size(); i++){
    	//cout << in[i] << " " << temp << " ";
    	if (in.compare(i, in.size() - i, in, count, in.size() - count) < 0){
    		count = i;
    	}
    	out.push_back(count);
    	//cout << count << " " << out[i] << endl;
    }
    while (q--){
    	cin >> i;
    	cout << out[i] << endl;
    }
    return 0;
}
