#include<stdio.h>
#include<iostream>
#define mod 1000000007
using namespace std;
long long int f[400000];
long long int inv[400000];
void fn()
{
	long long int i,result,exp,base;
	f[0]=1;
	for(i=1;i<=400000;i++)
	{
		f[i]=i*f[i-1]%mod;
	}
    inv[0]=1;
	for(i=1;i<=400000;i++)
	{
	result=1;
	exp=mod-2;
	base =f[i];
	while(exp>0)
	{
		if(exp&1)
		result=(result*base)%mod;
		exp=exp>>1;
		base=(base*base)%mod;
	}
	inv[i]=result;
}
}
int main()
{
	fn();
	int t;
	scanf("%d",&t);
	int n,m,x,i;
	while(t--)
	{
		 int sum=0;
		 cin>>n>>m;
		  for(i=0;i<n;i++)
		   {
   			cin>>x;
   			sum=sum+x;
   		}
   		 m=m-sum;
   		 if(m<0)
   		 cout<<"0"<<endl;
   		 else
   		 {
   		 	
   		 long long int ans=f[m+n-1];
   		  ans=ans*inv[n-1]%mod;
   		  ans=ans*inv[m]%mod;
   		  cout<<ans%mod<<endl;
	}
	}
	return 0;
}
