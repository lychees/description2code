from collections import defaultdict
d = defaultdict(lambda:0)
def main():
    st = raw_input()
    for i in st:
        d[ord(i)] += 1
    minimum = min([ord(i) for i in st])
    maximum = max([ord(i) for i in st])
    till = 0
    obj = chr(0)
    for i in range(minimum,maximum+1):
        if d[i] > till:
            till = d[i]
            obj = chr(i)
    print obj,till
if __name__ == "__main__":
    main()
