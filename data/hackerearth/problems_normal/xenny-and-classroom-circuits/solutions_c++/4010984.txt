#include<bits/stdc++.h>

using namespace std;

int tests;
string st;
int n;
int cnt[2][300];

int main(){

	cin >> tests;
	for (; tests; --tests)
	{
		cin >> n;
		for (int i = 0; i < 2; i++)
		{
			cnt[i]['B'] = cnt[i]['G'] = 0;
		}
		cin >> st;
		for (int i = 0; i < st.size(); i++)
		{
			cnt[i % 2][st[i]]++;
		}
		cout << min(cnt[1]['G'], cnt[0]['G']) << endl;
	}

	return 0;
}
