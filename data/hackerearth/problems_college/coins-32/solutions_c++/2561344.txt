#include<stdio.h>
#include<iostream>
#include<string.h>
#include<algorithm>

using namespace std;

int a,b,c,d,r;
int ar[5]={0,1,2,5,10};
int ak[5];
long long int find(int i,int am)
{
    if(i==3 && am%5!=0)
        return 0;
    if(i==4 && am%10!=0)
        return 0;
    if(am==0)
        return 1;
    if(i>4)
        return 0;
    int v=ar[i];
    int n=ak[i];
    if(am-v<0)
        return 0;
    int j=0;
    long long int count=0;
    while(j<=n)
    {
        if(v*j>am)
            break;
        else
            count+=find(i+1,am-v*j);
        j++;
    }
    return count;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&r);
        scanf("%d %d %d %d",&a,&b,&c,&d);
        ak[1]=a;
        ak[2]=b;
        ak[3]=c;
        ak[4]=d;
        printf("%lld\n",find(1,r));
    }
    return 0;
}
