#include <bits/stdc++.h>
using namespace std;
string getstr() //To Input String
{
    char a[11];
    scanf("%s",a);
    string s=a;
    return s;
}

int main()
{
    int n;
    scanf("%d",&n);
    //Operation - 1
    map <string, vector<int> > m;
    string s;
    for ( int i = 0; i < n; i++ )
    {
        s=getstr();
        for ( int j = 0; j < (int)s.size(); j++ )
        {
            string s1("");
            for ( int k = j; k < (int)s.size(); k++ )
            {
                s1.push_back(s[k]);
                if(m[s1].size()==0)
                    m[s1].push_back(i);
                else
                {
                    int sz=m[s1].size();
                    if(m[s1][sz-1]!=i)
                        m[s1].push_back(i);
                }
            }
        }
    }

    int q;
    scanf("%d",&q);

    //Operation - 3
    while ( q-- )
    {
        int l,r;
        scanf("%d %d",&l,&r);
        s=getstr();
        l--; r--;
        int ans=0;
        if ( s.empty() )
            ans = r-l+1;
        else
        {
            int idx2 = upper_bound(m[s].begin(),m[s].end(),r) - m[s].begin();
            int idx1 = lower_bound(m[s].begin(),m[s].end(),l) - m[s].begin();
            ans = idx2-idx1;
        }
        printf("%d\n",ans);
    }
    return 0;
}
