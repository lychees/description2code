#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t>0)
    {
    	long int n,p;
    	cin>>n>>p;
    	for(int i=0;i<p;i++)
    	{
    		if(n%2==0)
    		{
    			
    			n=	(n/2 +1);
    		}
    		else
    		{
    			if(n==3)
    			{
    				break;
    			}
    			n = (ceil(n/2) + 2);
    		}
    	
    	}
    	cout<<n<<"\n";
    	
    	t--;
    }
    return 0;
}
