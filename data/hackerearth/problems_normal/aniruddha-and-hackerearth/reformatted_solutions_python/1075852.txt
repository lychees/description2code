
nums = []


def process():
    global nums
    nums.append(0)
    nums.append(2)
    nums.append(3)

    for i in xrange(3, 10**6 + 3):
        nums.append((nums[i-1] + nums[i-2]) % 1000000007)


def main():
    global nums
    process()
    for _ in xrange(int(raw_input())):
        print(nums[int(raw_input())+2]-3)


main()
