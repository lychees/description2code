#include <iostream>
using namespace std;
typedef long long ll;

int main() {
    ll t;
    cin>>t;
    while(t--) {
    	ll n;
    	ll a[100005][4];
		ll cost[100005][4];
    	cin>>n;
    	for(ll i=0;i<n;i++) {
    		for(ll j=0;j<3;j++)
    			cin>>a[i][j];
    	}
    	cost[0][0]=a[0][0];
    	cost[0][1]=a[0][1];
    	cost[0][2]=a[0][2];
    	for(ll i=1;i<n;i++) {
    		cost[i][0]=a[i][0]+min(cost[i-1][1],cost[i-1][2]);
    		cost[i][1]=a[i][1]+min(cost[i-1][0],cost[i-1][2]);
    		cost[i][2]=a[i][2]+min(cost[i-1][1],cost[i-1][0]);
    		
    	}
    	cout<<min(cost[n-1][0],min(cost[n-1][1],cost[n-1][2]))<<"\n";
    }
}
