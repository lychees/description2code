#include <iostream>
#include<math.h>
using namespace std;
void test(long int n, long int *d)
{	int i;
	if(n<2 || n>pow(10,6))
	exit(0);
	
	for(i=0; i<n; i++)
	 if(d[i]<-(pow(10,9)) || d[i]>pow(10,9))
	 exit(0);
}
int main()
{
    	long int N, mxd=0, mxb=0;
    	cin>>N;
    	long int sd[N], sb[N];
    	int i;
    	for(i=0; i<N; i++)
    	cin>>sd[i];
    	for(i=0; i<N; i++)
    	cin>>sb[i];
    	test(N,sd);
    	
    	for(i=1; i<N; i++)
    	{
    		if(abs(sd[i]-sd[i-1])>mxd)
    			mxd=abs(sd[i]-sd[i-1]);
    		if(abs(sb[i]-sb[i-1])>mxb)
    			mxb=abs(sb[i]-sb[i-1]);
    	}
    	
    	if(mxd>mxb)
    	{
    		cout<<"Dom"<<endl;
    		cout<<mxd<<endl;
    	}
    	else if(mxb>mxd)
    	{
    		cout<<"Brian"<<endl;
    		cout<<mxb<<endl;
    	}
    	else if(mxb==mxd)
    	{
    		cout<<"Tie"<<endl;
    		cout<<mxb;
    	}
    	
}
