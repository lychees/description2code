#include <iostream>
using namespace std;

int main()
{
    int n,a[10000],i,c=0,d;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    for(i=0;i<n;i++)
    {
    	c=c+a[i];
    	if(c<0)
    	{
    		c=0;
    	}
    	if(d<c)
    	{
    		d=c;
    	}
    }
    cout<<d<<endl;
    return 0;
}
