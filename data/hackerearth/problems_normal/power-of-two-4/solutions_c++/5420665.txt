#include <iostream>
using namespace std;

int main()
{
	int t, sum;
	cin >> t;
	for (int i = 0; i<t; i++)
	{
		int n;
		cin >> n;
		int arr[n];
		for (int j = 0; j<n; j++)
		{
			cin >> arr[j];
		}
		int flag = 0;
		for (int k = 0; k<32; k++)
		{
			int temp = 1 << k;
			sum = 0xffffffff;
			for (int l = 0; l<n; l++)
			{
				if (((arr[l] >> k) & 1) == 1)
				{
					sum = sum&arr[l];
				}
			}
			if (sum == temp)
			{
				flag = 1;
				break;
			}
		}
		if (flag == 1)
			cout << "YES\n";
		else
			cout << "NO\n";

	}
	return 0;
}
