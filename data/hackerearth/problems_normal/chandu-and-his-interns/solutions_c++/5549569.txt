#include<iostream>
using namespace std;

int main() {
    long long N,X,i,divCount;
    cin>>N;
    while(N--) {
        cin>>X;
        for(i=1,divCount=0;i*i<=X && divCount<4;i++)
            if (X%i==0 && i!=X/i) divCount+=2;
        if (divCount<4) cout<<"NO\n";
        else cout<<"YES\n";
    }
    return 0;
}
