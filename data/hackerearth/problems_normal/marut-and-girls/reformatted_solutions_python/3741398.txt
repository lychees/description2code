n = int(raw_input())
s = raw_input().split(" ")
d = {}
for v in s:
    d[v] = 1
# print d
m = int(raw_input())
ans = 0
while m > 0:
    t = raw_input().split(" ")
    count = 0
    for e in t:
        if d.get(e, 0) != 0:
            count += 1
    if count == n:
        ans += 1
    m -= 1
print ans
