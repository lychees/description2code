#include<bits/stdc++.h>
using namespace std;
int main()
{int t,k;
cin>>t;
string fina;

   while(t--)
   {
       long long int n;
       cin>>n>>k;
   vector<string>v;
   vector<char>v1;
    int i,j;
    for(i=0;i<n;i++)
    {
        string s;
        cin>>s;

        v.push_back(s);
    }
   for(i=0;i<k;i++)
   {
       for(j=0;j<v.size();j++)
       {
           string tmp=v.at(j);
           v1.push_back(tmp.at(i));

       }
       sort(v1.begin(),v1.end());
       if(n%2==0)
       {
           fina=fina+v1.at((n/2)-1);
       }
       else if((n%2)!=0)
       {
           fina=fina+v1.at(n/2);
       }
       v1.clear();

       }

    cout<<fina<<endl;
    fina="";
}
return 0;
}
