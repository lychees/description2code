#include <iostream>
#include <math.h>
#include <stdio.h>
using namespace std;

double PI = (2*acos(0.0));
int main()
{
    int t;
    cin>>t;
    
    while(t--)
    {
    	double v;
    	cin>>v;
    	double  r = pow(v/(2*PI),1.0/3.0);
    	double  h  =  r*2;
    	
    	double ans = 2*PI*r*(r+h);
    	printf("%.9f\n",ans);
    	
    	
    }
    return 0;
}
