#include <iostream>
using namespace std;

int main()
{
    long long int i,n;
    
    cin>>n;
    
    long long int a;
    
    long long int min,max;
    
    cin>>min;
    max=min;
    
    for(i=1;i<n;i++)
    {
    	cin>>a;
    	
    	if(a<min)
    		min=a;
    	else if(a>max)
    		max=a;
    }
    
    
    
    cout<<min*max<<endl;
    return 0;
}
