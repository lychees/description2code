#include<bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;

int main()
{
	string w,s;
	int l,n,count,i,j;
	cin>>w;
	cin>>s;
	l=w.length();
	n=s.length();
	int arr[n+5],temp[n+5];
	for(i=0;i<n+2;i++)
	{
		arr[i]=0;
		temp[i]=-1;
	}
	
	for(i=0;i<l;i++)
	{
		if(i==0)
		{
			for(j=0;j<n;j++)
			{
				if(w[i]==s[j])
				{
					temp[j]=i;
					arr[j]=1;
				}
			}
		}
		else
		{
			count=0;
			for(j=0;j<n;j++)
			{
				if(temp[j]==i-1)
				{
					count+=arr[j];
				}
				else if(w[i]==s[j])
				{
					temp[j]=i;
					arr[j]=count;
				}
			}
		}
	}
	count=0;
		for(j=0;j<n;j++)
		{
		//	cout<<temp[j]<<" "<<arr[j]<<endl;
			if(temp[j]==l-1)
			{
				count+=arr[j];
			}
		}
		cout<<count;
	return 0;
}
