#include<iostream>
#include<algorithm>
#include<stdio.h>
#include<stdlib.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t>0)
	{
		long n,i,j,k;
		long *arr;
		cin>>n;
		arr=(long*)calloc(n,sizeof(long));
		//cin>>i>>j;
		for(k=0;k<n;k++)
		cin>>arr[k];
		sort(arr,arr+n);
		cin>>i>>j;
		cout<<arr[j-1]+arr[n-i]<<"\n";
	t--;
	}
}
