#include<bits/stdc++.h>

using namespace std;


int main()
{
long long int t;
	cin >> t;
	while(t--)
	{
		long long int n,i,ans=0;
		

		cin >>n;
	long long int val;
		vector <long long int> inp(n);
	
		for(i=0;i<n;i++)
		{
			

		cin >> inp[i];
		}
		
		sort(inp.begin(), inp.end());
		if(n%2==0)
		{
			

		val= inp[(n/2)-1];
		}
		else
		{
		

	val= inp[(n/2)];

		}

		

	for(i=0;i<n;i++)
		ans+=abs(val- inp[i]);

		cout << ans 

<<endl;
	}

	return 0;
}
