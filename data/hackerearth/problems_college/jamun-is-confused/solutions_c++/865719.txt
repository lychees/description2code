#include <iostream>
using namespace std;

int gcd(int a, int b){
	if(b==0) return a;
	else return gcd(b,a%b);
}

void solve(){
	int n;
	cin>>n;
	int g, a;
	cin>>g;
	for(int i = 1; i<n; i++){
		cin>>a;
		g = gcd(a,g);
	}
	cout<<g<<endl;
}

int main(){
	int t;
	cin>>t;
	while(t--) solve();
    return 0;
}
