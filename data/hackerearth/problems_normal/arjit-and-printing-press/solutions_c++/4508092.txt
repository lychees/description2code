#include<iostream>
#include<stdio.h>
#include<string.h>
#include<algorithm>

using namespace std;	
int main()
{
	int t;
	string w,r;
	cin>>t;
	while(t--)
	{
	    cin>>w>>r;
		sort(r.begin(),r.end());
		
		for(int i=0,j=0;i<w.size();i++)
		{
			if(j>=r.size())
			break;
			if(w[i]>r[j])
			{
			w[i]=r[j];
			j++;
			}
	}
	cout<<w<<endl;
}
return 0;
}
