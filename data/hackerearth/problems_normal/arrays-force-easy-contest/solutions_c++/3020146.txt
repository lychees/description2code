#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll h[1100000];
int main()
{
	int t;
	cin>>t;
	ll a0,a1,n,mod;
	while(t--)
	{
		memset(h,0,sizeof(h));
		cin>>a0>>a1>>n>>mod;
		ll a[n],i;
		a[0]=a0,a[1]=a1;
		h[a0]++;
		h[a1]++;
		ll mx=max(a0,a1);
		for(ll i=2;i<n;i++)
		{
			a[i]=(a[i-1]+a[i-2])%mod;
			h[a[i]]++;
			mx=max(mx,a[i]);
		}
		//cout<<mx<<' '<<h[0]<<' '<<h[1]<<'\n';
		ll ans=0;
		for(ll i=0;i<=mx;i++)
		{
			ans=ans+h[i]*h[i];
		}
		cout<<ans<<'\n';
	}
	return 0;
}
