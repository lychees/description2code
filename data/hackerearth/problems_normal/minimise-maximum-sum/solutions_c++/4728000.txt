#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t>0)
    {
    	int n;
    	cin>>n;
    	vector<int> vec;
    	vec.resize(n*2);
    	for(int i=0;i<2*n;i++)
    	{
    		cin>>vec[i];
    	}
    	sort(vec.begin(),vec.end());
    	int index=2*n-1;
    	int maxsum=0;
    	for(int i=0;i<n;i++)
    	{
    		maxsum=max(maxsum,vec[i]+vec[index]);
    		index--;
    	}
    	cout<<maxsum<<endl;
    	t--;
    }
}
