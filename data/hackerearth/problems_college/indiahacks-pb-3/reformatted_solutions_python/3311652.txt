for _ in xrange(int(raw_input())):
    n, k = map(int, raw_input().split())
    arr = map(int, raw_input().split())
    pre = [None]*n
    pre[0] = max(arr[0], 0)
    for i in xrange(1, n):
        if i <= k:
            pre[i] = max(pre[i - 1], arr[i])
        else:
            pre[i] = max(pre[i - 1], arr[i] + pre[i - k - 1])
    print pre[n - 1]
