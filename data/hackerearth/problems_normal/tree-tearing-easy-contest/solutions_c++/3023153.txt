#include <iostream>
#include<bits/stdc++.h>
using namespace std;

long long int size[200001],marked[200001],k;
vector<long long int>v[200001];

void dfs1(long long int cur,long long int par)
{
    for(long long int i=0;i<v[cur].size();i++)
    {
        if(v[cur][i]!=par)
        {
            dfs1(v[cur][i],cur);
            size[cur]+=size[v[cur][i]];
        }
    }
}

void dfs(long long int cur,long long int par)
{
    for(long long int i=0;i<v[cur].size();i++)
    {
        if(v[cur][i]!=par)
        {
            if(!marked[v[cur][i]])
            {
                dfs(v[cur][i],cur);
                if(!marked[v[cur][i]])
                size[cur]+=size[v[cur][i]];
            }
        }
    }
    if(size[cur]>=k)
    marked[cur]=1;
    
}

int main() 
{
    long long int i,n,x,y,ans=0;
    cin>>k>>n;
    for(i=1;i<=n;i++)
    size[i]=1;
    for(i=2;i<=n;i++)
    {
        cin>>x;
        v[x].push_back(i);
        v[i].push_back(x);
    }
    //dfs1(1,-1);
    dfs(1,-1);
    for(i=1;i<=n;i++)
    {
        if(marked[i])
        ans++;
    }
    cout<<ans;
    return 0;
}
