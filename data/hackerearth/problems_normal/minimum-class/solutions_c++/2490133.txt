#include <iostream>
#include <limits.h>
using namespace std;

int msol(int n, int pmove)
{
	if(n==1)
	{
		return 1;
	}
	if(n<=pmove+1)
	{
		return 1;
	}
	if(n<=2*(pmove+1))
	{
		return 2;
	}
	
	return 2+ msol(n-2*(pmove+1),pmove+1);
	
}

{
    int t;
    cin>>t;

    cout<<	msol(n,0)<<endl;
