#include<iostream>
using namespace std;
int main()
{
	int a[1000002]={0},i,j,temp,c;
	   for(i=2;i<=1000;i++)
	   {
   		if(!a[i])
   		for(j=i*i;j<=1000000;j=j+i)
   		a[j]=1;
		   }
   	for(i=2;i<=1000000;i++)
   	{
	   	if(a[i])
	   	{
	   		temp=i;
	         c=0;
	   		while(temp!=0)
	   		{
		   		if(temp%10==7)
		   		{
		   			c++;
		   			break;
		   		}
		   		temp=temp/10;
		   	}
		   	if(c==0)
		   	a[i]=0;
	   	}
	   
	   }
	   int t,m,n;
	   cin>>t;
	   while(t--)
	   {
	   	int ct=0;
	   	cin>>m>>n;
   		for(i=m;i<=n;i++)
   		{
		   	if(a[i])
		   	ct++;
		   }
		   if(ct==0)
		   ct=-1;
		   cout<<ct<<endl;
   	}
   	return 0;
   	 
}
