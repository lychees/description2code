#include<stdio.h>
int main(){
int test;
scanf("%d",&test);
while(test-->0){
    unsigned long long int a,b;
    scanf("%llu%llu",&a,&b);
    if(a<b){
        printf("%llu\n",a*(a+1)/2);
        continue;
    }
    unsigned long long int sm=b*(b-1)/2;
    unsigned long long int fact=a/b;
  //  printf("=-------------- %llu\n",sm);
    unsigned long long int modval=a%b;
    unsigned long long int toadd=0;
    if(modval!=0 && modval!=1){
        toadd=((modval)*(modval+1));
        toadd=toadd/2;
    }
   else if(modval==1)
      toadd++;
  //    printf("\t%llu\t",toadd);
    printf("%llu\n",fact*sm+toadd);

}
}
