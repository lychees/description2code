#include <cstdio>

#define MAX 1000010
#define MOD 1000000007

int a[MAX], f[MAX];
int t, n, m;

int main(){
	f[1] = 2;
	a[0] = 1; a[1] = 1;
	m = 1;
	
	scanf("%d", &t);
	while (t--){
		scanf("%d", &n);
		if (n > m){
			for (int i = m + 1; i <= n; i++){
				a[i] = (a[i - 1] + a[i - 2]) % MOD;
				f[i] = ((f[i - 1] + a[i - 1]) % MOD + a[i]) % MOD;
			}
			m = n;
		}
		printf("%d\n", f[n]);
	}
	
	return 0;
}
