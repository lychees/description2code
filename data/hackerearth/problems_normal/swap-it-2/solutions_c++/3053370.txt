#include<iostream>                    
using namespace std;
int main()
{
int t;
cin>>t;
while(t--)
{
int n,k;                                  
cin>>n>>k;                      
int a[n];
for(int i=0;i<n;i++)                  
cin>>a[i];                                         
int p=0;                    int m;    
while(k>0&&p<n)                   
{  m=p;
if(n<p+k+1)
{for(int i=p;i<n;i++)
if(a[i]<a[m])
m=i;              
}
else 
for (int i=p;i<=k+p  ;i++)
if(a[i]<a[m])
m=i;
for(int i=m;i>p;i--)
{swap(a[i],a[i-1]);
k--;}
p++;
}
for(int i=0;i<n;i++)
cout<<a[i]<<" ";             
cout<<endl;
}}
