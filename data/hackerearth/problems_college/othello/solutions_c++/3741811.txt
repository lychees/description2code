#include <bits/stdc++.h>
using namespace std;
const int MAX = 1005;
int n, m, arr[MAX][MAX], row[MAX], col[MAX], tile[MAX][MAX];
int main(){
	ios :: sync_with_stdio(false);
	cin >> n >> m;
	for(int i = 1; i <= n; i++){
		for(int j = 1; j <= m; j++){
			char c;
			cin >> c;
			arr[i][j] = (c == 'B') ? (0) : (1);
            if(arr[i][j] == 0) row[i]++, col[j]++, tile[i][j]++;
		}
	}
	int res = 0;
	for(int i = 1; i <= n; i++){
		for(int j = 1; j <= m; j++){
			if( (row[i] + col[j] + tile[i][j]) % 2 == 1) res++;
		}
	}
	cout << res << '\n';
}
