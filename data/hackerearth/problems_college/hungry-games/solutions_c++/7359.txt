#define MP make_pair
#define PB push_back
#define X first
#define Y second
#define oo 2000000000
#define MOD 1000000007
#define LL long long int
#define pii pair<int,int>

using namespace std;
vector<pii> v;
LL ar[102],br[102];
int main(){
    int n,x,y,t;
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%d%d",&x,&y);
        v.PB(MP(y,x));
    }
    sort(v.begin(),v.end());
    scanf("%d",&t);
    LL cost=0,j=0;
    for(int i=0;i<t;i++) scanf("%lld",&br[i]);
    ar[0]=br[0];
    for(int i=1;i<t;i++){
        ar[i]=br[i]-br[i-1];
    }
    for(int i=0;i<t;i++){
        while(j<n && v[j].Y<=ar[i]){
            cost+=((LL)v[j].Y*(i+1)*v[j].X);
            ar[i]-=v[j].Y;
            ++j;
        }
        if(j<n){
            cost+=((LL)ar[i]*(i+1)*v[j].X);
            v[j].Y-=ar[i];
        }
    }
    while(j<n) cost+=((LL)v[j].Y*v[j].X*(t+1)) ,++j;
    printf("%lld\n",cost);
    return 0;
}
