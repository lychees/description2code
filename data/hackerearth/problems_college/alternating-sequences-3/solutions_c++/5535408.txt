#include<iostream>
using namespace std;
int main()
{
    int n;
    cin>>n;
    long long int ip[n];
    int s[n];
    int maxi[n];
    for(int i=0;i<n;i++)
    {
        cin>>ip[i];
        if(ip[i]<0)
        {
            ip[i]=-1*ip[i];
            s[i]=1;
        }
        else
            s[i]=0;
        maxi[i]=1;
    }
    int ans=1;
    for(int i=1;i<n;i++)
    {
        for(int j=i-1;j>=0;j--)
        {
            if((ip[j]<ip[i]) && (s[j]!=s[i]) && (maxi[j]>=maxi[i]))
                maxi[i]=maxi[j]+1;
        }
        if(maxi[i]>ans)
            ans=maxi[i];
    }
    cout<<ans;
}
