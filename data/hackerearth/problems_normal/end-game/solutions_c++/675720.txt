#include <iostream>

using namespace std;

int main()
{
	long long T;
	cin >> T;

	for(;T--;)
	{
		long long n, a, b, c ,d ,Move;
		cin >> n >> a >> b >>c >> d >> Move;

		if(Move == 1)
			--a;

		long long lx = n-a;
		long long ly = b-lx;
		long long ry = b+lx;

		if(n-c <= lx && d >= ly && d <= ry)
			cout << "Draw\n";
		else
			cout << "White Wins\n";
	}
	return 0;
}
