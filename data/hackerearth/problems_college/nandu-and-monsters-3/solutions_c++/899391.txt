#include <bits/stdc++.h>
using namespace std;
string s;

bool chk(int l, int r)
{
	vector<int> F, S;

	while (l) {
		F.push_back(l % 2);
		l /= 2;
	}

	reverse(F.begin(), F.end());
	int val = 1; 
	F[0] = val;
	for (int i = 1; i < F.size(); i++) {
		if (F[i] == 0) {
			F[i] = val * 2;
			val *= 2;
		}
		else if (F[i] == 1) {
			F[i] = val * 2 + 1;
			val = val * 2 + 1;
		}
	}


	while (r) {
		S.push_back(r % 2);
		r /= 2;
	}

	reverse(S.begin(), S.end());
	val = 1; 
	S[0] = val;
	for (int i = 1; i < S.size(); i++) {
		if (S[i] == 0) {
			S[i] = val * 2;
			val *= 2;
		}
		else if (S[i] == 1) {
			S[i] = val * 2 + 1;
			val = val * 2 + 1;
		}
	}

	int i = 0, j = 0;
	int n = F.size(), m = S.size();
	while (i < n && j < m) {
		if (F[i] != S[j])
			break;
		i++, j++;
	}
	i--, j--;

	for (int temp = i; temp < n; temp++) {
		if (s[F[temp] - 1] == '1')
			return true;
	}

	for (int temp = j; temp < m; temp++) {
		if (s[S[temp] - 1] == '1')
			return true;
	}

	return 0;
}

int main()
{
	int n, q;
	scanf("%d %d\n", &n, &q);
	char str[100005];
	scanf("%s", &str);
	//cin >> s;
	s = string(str);

	while (q--) {
		int l, r;
		scanf("%d %d", &l, &r);

		if (chk(l, r))
			printf("No\n");
		else
			printf("Yes\n");
	}
}
