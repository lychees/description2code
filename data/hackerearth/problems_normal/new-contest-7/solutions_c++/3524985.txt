    #include <stdio.h>
     
    double nCr(int a, int b)
    {
    	double n = (double)a;
    	double r = (double)b;
    	int i;
    	double ans;
     
    	if(b==0)
    	{
    		ans = 1.0;
    	}
    	else if(a-b < b)
    	{
    		ans = nCr(a,a-b);
    	}
    	else
    	{
    		ans = 1.0;
    		for(i=0;i<r;i++)
    			ans = ans*(n-i)/(r-i);
    	}
    	
    	return ans;
    }
     
    int main()
    {
    	int T,s,n,m,k;
     
    	int i;
    	double factN,factM,factS,factK;
    	double sol;
     
    	scanf("%d",&T);
    	
    	while(T--)
    	{	
    		scanf("%d %d %d %d",&s,&n,&m,&k);
     
    		if(k==0 || s-m<n-k+1)
    		{
    			sol=1.0;
    		}
    		else if(k>n)
    		{
    			sol=0.0;
    		}
    		else
    		{
    			sol = 0.0;
    			
    			for(i=k;i<=m && i<=n;i++)
    			{
    				sol = sol + nCr(m,i)*nCr(s-m,n-i);
    			}
    			
    			sol = sol/nCr(s,n);
    		}
     
    		printf("%.6f\n",sol);
    	}
     
    	return 0;
    }
