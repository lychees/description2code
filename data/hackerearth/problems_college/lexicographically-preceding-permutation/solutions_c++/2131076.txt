#include <iostream>
#include <algorithm>

using namespace std;

void rev(int arr[],int start , int end ){
	
	while(end>start){
		
		int temp = arr[start];
		arr[start]=arr[end];
		arr[end]=temp ;
		end--;
		start++;
	}
	
	}


int main()
{
	
	int t; 
	cin >> t ; 
	
	while(t--){
		
		
		int nu ; 
		cin >> nu ;
		
		int arr[nu];
		
		for(int i=0 ; i<nu ; i++)
			cin >> arr[i];
		
		int j;
		
		
		for( j=nu-1; j >0;j--){
			if(arr[j]>arr[j-1]){}
			else
			break;
		}
		
		if(j!=0){
			
			int x ; 
			
			for( x=j;x<nu;x++){
				if(arr[x]>arr[j-1])
				break;
						
			}
			
			
		
		int temp = arr[x-1];
		arr[x-1]=arr[j-1];
		arr[j-1]=temp ; 

		rev(arr,j,nu-1);
		
		
		}
	
	for(int i =0 ; i < nu ; i++)
	cout << arr[i] << " ";
		
		cout << endl ;
	}	
	
    
    return 0;
}
