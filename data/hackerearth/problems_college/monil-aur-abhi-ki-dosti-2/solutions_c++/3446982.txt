#include<bits/stdc++.h>
#define MAX 256
using namespace std;
 

bool compare(char arr1[], char arr2[])
{
    for (int i=0; i<MAX; i++)
        if (arr1[i] != arr2[i])
            return false;
    return true;
}
 
void search(string pat, string txt)
{
    int M = pat.size(), N = txt.size(),flag=0;
 
   
    char countP[MAX] = {0}, countTW[MAX] = {0};
    for (int i = 0; i < M; i++)
    {
        (countP[pat[i]])++;
        (countTW[txt[i]])++;
    }
 
   
    for (int i = M; i < N; i++)
    {
       
        if (compare(countP, countTW))
           flag=1;
 
       
        (countTW[txt[i]])++;
 
      
        countTW[txt[i-M]]--;
    }
 
  
    if (compare(countP, countTW))
        flag=1;
        
    if(flag==1)
    	cout<<"YES\n";
    else
    	cout<<"NO\n";
}

int main()
{
	long int t;
	cin>>t;
	string x,y;
	while(t--)
	{
		cin>>x;
		cin>>y;
		search(x,y);
	}
}
