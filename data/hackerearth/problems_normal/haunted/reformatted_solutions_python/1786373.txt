
N, M = map(int, raw_input().split())
G = [int(i) for i in raw_input().split()]
maxCount = 0
maxAge = 0
ageCount = {}
for age in G:
    if ageCount.has_key(age):
        ageCount[age] = ageCount[age] + 1
    else:
        ageCount[age] = 1
    if ageCount[age] > maxCount:
        maxCount = ageCount[age]
        maxAge = age
    elif ageCount[age] == maxCount and age > maxAge:
        maxAge = age
    print maxAge, maxCount
