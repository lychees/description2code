#include<bits/stdc++.h>
using namespace std;


int main(){
	int t;
	scanf("%d",&t);

	while(t--){
		int n,k,i,j;
		scanf("%d%d",&n,&k);
		int a[201][201];
		int r[201],c[201];
		int sr[401],sc[401];
		
		memset(r,0,sizeof(r));
		memset(c,0,sizeof(c));
		
		for(i=0;i<n;i++){
			for(j=0;j<n;j++){
				scanf("%d",&a[i][j]);

				r[i] += a[i][j];
				c[j] += a[i][j];
			}
		}
		
		int ans = 0x3f3f3f3f;
		
		for(i=1;i<=k;i++){
            int x = 0;
            for(j=0;j<n;j++){
                if(r[j] < r[x]){
                    x = j;
                }
            }
            sr[i] = r[x] + sr[i-1];
            r[x] += n;
		}
		
		for(i=1;i<=k;i++){
            int x = 0;
            for(j=0;j<n;j++){
                if(c[j] < c[x]){
                    x = j;
                }
            }
            sc[i] = c[x] + sc[i-1];
            c[x] += n;
		}
		
		for(i=0;i<=k;i++){
			ans = min(ans,sr[i] + sc[k-i] + i*(k-i));
		}
		printf("%d\n",ans);
	}
	return 0;
}
