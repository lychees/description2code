#include <iostream>
using namespace std;


int gcd(int a, int b)
{
    if (a == 0)
        return b;
    return gcd(b%a, a);
}
 
// A simple method to evaluate Euler Totient Function
int phi(unsigned int n)
{
    unsigned int result = 1;
    for (int i=2; i<n; i++)
        if (gcd(i, n) == 1)
            result++;
    return result;
}

int main()
{ unsigned int t,n;
cin>>t;
while(t--)
{cin>>n;
cout<<phi(n)<<"\n";
}}
