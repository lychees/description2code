#include <bits/stdc++.h>
using namespace std;
long int cal(long int a[],int n,int k)
{
  if(n==0 || k==0)	
	 return 0;
  if(a[n-1]>k)	
	 return cal(a,n-1,k);
  else	 
 return max(a[n-1]+cal(a,n,k-a[n-1]),cal(a,n-1,k));	 	
}

  int main()
{  int t;
   cin>>t;
   while(t--)
  {
  	long int n,k;
  	cin>>n>>k;
  	long int *a=new long int[n];
  	for(int i=0;i<n;i++)
  	  cin>>a[i];
  	cout<<cal(a,n,k)<<endl;  	
  } 
    return 0;
}
