t = input()
for _ in range(0, t):
    n = input()
    case = map(int, raw_input().split())
    d = input()
    y = 0
    x = 0
    while x < (n-y-1):
        if (case[x]+case[x+1]) % d == 0:
            case[x] = (case[x]+case[x+1])/d
            del case[x+1]
            y += 1
        else:
            x += 1
    print ' '.join(map(str, case))
