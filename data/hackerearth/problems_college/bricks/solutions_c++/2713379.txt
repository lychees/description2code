using namespace std;
typedef long long ll;
//#define DEBUG
#define mod 1000000007
#define pb push_back

ll power(ll a, ll k){
    ll ret=1;
    while(k){
        if(k & 1)
            ret = (ret*a)%mod;
        k>>=1;
        a = (a*a)%mod;
    }
    return ret;
}

ll f[1010], x[1010], ans[1010];

int main(){
    ios::sync_with_stdio(false);
    #ifdef DEBUG
    freopen("input.txt", "r", stdin);
    #endif // DEBUG

    f[0] = 1;
    f[1] = 1;
    f[2] = 2;
    f[3] = 4;
    f[4] = 8;
    for(int i=5;i<=1000;i++)
    	f[i] = (f[i-1] + f[i-2] + f[i-3] + f[i-4]) % mod;

    int t, n, m;
    cin >> t;
    while ( t-- ){
    	cin >> n >> m;
    	for(int i=1;i<=m;i++)
    		x[i] = power(f[i], n);

    	memset(ans, 0, sizeof(ans));
    	ans[1] = 1;
    	for(int i=2;i<=m;i++){
    		ans[i] = x[i];
    		for(int j=1;j<i;j++){
    			ans[i] = (ans[i] + 2*mod - ans[j] * x[i-j]) % mod;
    		}
    	}
    	cout << ans[m] << endl;
    }

    return 0;
}
