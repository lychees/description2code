#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

int main()
{
    int t;
    cin >> t;
    while ( t-- ) {
          string s;
          cin >> s;
          int sum = 0;
          for ( int i = 0; i < s.size(); i++ ) {
              if ( s[i] == '_' ) continue;
              sum += s[i]-48;
          }
          if ( sum%9 == 0 ) cout << "0 or 9" << endl;
          else {
               int val = sum/9 + 1;
               val = 9*val - sum;
               cout << val << endl;
          }
               
    }
    return 0;
}
