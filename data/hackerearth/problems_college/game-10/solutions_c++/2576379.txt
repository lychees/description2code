#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    long long int n,k,ans,position;
    cin>>n>>k;
    long long int new_k1=k+1;
    position=(2*new_k1)-1;
    if(position>n)
    {
      ans=(n-new_k1)*2;	
    }
    else
    {
    	ans=position;
    }
    cout<<ans<<endl;
    }
    return 0;
}
