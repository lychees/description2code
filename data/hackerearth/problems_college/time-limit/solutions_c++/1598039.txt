#include <bits/stdc++.h>

using namespace std;

int A[102];
int B[102];

int main()
{
	int n,m;
	cin >> n >> m;
	for ( int i = 0; i < n; i++ ) cin >> A[i];
	for ( int i = 0; i < m; i++ ) cin >> B[i];
	sort(A,A+n);
	sort(B,B+m);
	int ans;
	for ( int i = A[n-1]; i <= B[0]-1; i++ ) {
		if ( 2*A[0] <= i ) {
			ans = i;
			break;
		}
	}
	cout << ans << endl;
	return 0;
}
