#include <iostream>
using namespace std;
long int f1(long int a[],long int b[],int n,int l,int r,int flag)
{
	
	long int s=0;
	int i=l;
	while(i<=r)
	{
	 if(flag==0)
	{ s+=a[i]; flag=1;
	}
	else{
		s+=b[i];  flag=0;
	    }
	 i++;
	}
	return s;
}
int main()
{
    int n,q;
    cin>>n>>q;
   long int a[n],b[n],y[q],k=0;
    for(int i=0;i<n;i++)
    cin>>a[i];
    for(int i=0;i<n;i++)
    cin>>b[i];
    int t,l,r;
    for(int i=0;i<q;i++)
    {
    	cin>>t>>l>>r;
    	if(t==1)
    	y[k]=f1(a,b,n,l-1,r-1,0);
    	else if(t==2)
    	y[k]=f1(a,b,n,l-1,r-1,1);
    	k++;
    	
    }
    for(int i=0;i<k;i++)
    cout<<y[i]<<"\n";
    return 0;
}
