using namespace std;
vector<pair<ll,ll> > v;
int main()
{
    ll n,i,x,y;
    scanf("%lld",&n);
    for(i=0;i<n;i++)
    {
        scanf("%lld %lld",&x,&y);
        v.push_back(make_pair(x,y*-1));
    }
    sort(v.begin(),v.end());
    for(i=0;i<n;i++)
    printf("%lld %lld\n",v[i].first,v[i].second*-1);
	return 0;
}
