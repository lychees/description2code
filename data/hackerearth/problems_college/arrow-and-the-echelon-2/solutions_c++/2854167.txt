#include<iostream>
#include<stdlib.h>
#include<stdio.h>
#include<string.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		if(n%3==0&&n%5==0)
		{
			for(int i=0;i<n;i++)
				cout<<"5";
			cout<<endl;
		}
		else if(n%3==0)
		{
			for(int i=0;i<n;i++)
				cout<<"5";
			cout<<endl;
		}
		else if(n==5)
		{
			cout<<"33333"<<endl;
		}
		else
		{
			int x=0,y=0,flag=0;
			for(int i=3;i<=n;i+=3)
			{
				if((n-i)%5==0)
				{
					flag=1;
					x=i;
				y=n-i;
				}
			}
			for(int j=0;j<x;j++)
				cout<<"5";
			for(int j=0;j<y;j++)
				cout<<"3";
			if(flag==0)
				cout<<"-1";
			cout<<endl;
	
			
		}
		
	}
	
}
