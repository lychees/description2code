import math

strTime = raw_input()
arTime = strTime.split(':')
hour = float(12 if arTime[0] == 0 else arTime[0])
mint = float(arTime[1])



hrAngle = 30*hour + mint/2
minAngle = 6 * mint

angleDiff = math.fabs(hrAngle - minAngle)
if angleDiff > 180:
	angleDiff = 360 - angleDiff

print '{0:.6f}'.format(angleDiff)
