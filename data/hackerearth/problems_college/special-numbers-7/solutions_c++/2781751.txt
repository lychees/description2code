#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
int prime[10004];
int main()
{
	int i,count,j,n,result,t,num,a,k;
	vector<ll> lucky;
	vector<int> prime_num;
	prime[0]=1;
	prime[1]=1;
	k=0;
	for(i=2;i<8000;i++)
	{
		if(prime[i]==0)
		{
			prime_num.push_back(i);
			k++;
			a=2;
			while(a*i<8000)
			{
				prime[a*i]=1;
				a++;
			}
		}
	}
	k=prime_num.size();
	/*for(i=0;i<k;i++)
		cout<<prime_num[i]<<" ";
	cout<<endl;*/
	for(i=1;i<100000;i++)
	{
		count=0;
		num=i;
		j=0;
		while(count<3 && num>0 && j<k)
		{
			if(num%prime_num[j]==0)
			{
				num=num/prime_num[j];
				count++;
			}
			j++;
		}
		if(count>=3)
			lucky.push_back(i);
	}
	cin>>t;
	while(t--)
	{
		cin>>n;
		cout<<lucky[n-1]<<endl;
	}
	return 0;
}
