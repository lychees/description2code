#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   
   int t;
   cin>>t;
   while(t--)
   {
   	int n;
   	int q;
   	cin>>n>>q;
   	long long int h[n];
   	for(int i=0;i<n;i++)
   		cin>>h[i];
   	sort(h,h+n);
   	long long int cost=0;
   	for(int i=1;i<n;i++)
   		cost+=q*(h[i]-h[i-1]);
   	cout<<cost<<endl;
   }
    return 0;
}
