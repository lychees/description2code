#include <iostream>

using namespace std;

long long int C(long long int n, long long int r) {
	if(n <= 2) return 0;
	if(r == 3) {
		long long int tmp = n * (n - 1);
		tmp /= 2;
		tmp *= (n - 2);
		tmp /= 3;
		return tmp;
	}
}

static long long gcd(long long a, long long b) {
  long long r;
  while(b!=0){
    r = a % b;
    a = b;
    b = r;
  }
  return a;
}



int main()
{
    int t=0;
    cin>>t;
    while(t--)
    {
        long long int n=0,k=0;
        cin>>n>>k;

        int a[1000000];
        for(int i=0;i<n;i++)
            cin>>a[i];

        int count=0;
        for(int i=0;i<n;i++)
        {
            if(a[i]==k)
                count++;
        }



       long long int num=C(count,3);
       long long int den=C(n,3);

        //cout<<"num:"<<num<<" den:"<<den<<endl;

        if(num!=1)
        {
           long long n1=gcd(num,den);
            num=num/n1;
            den=den/n1;

       }

        cout<<num<<"/"<<den<<endl;


    }

    return 0;
}
