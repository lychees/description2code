#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
vector<vector<bool> > ss(1000002,vector<bool>(101,false));
void subset(vector<ll> &v,int n)
{
	for(int i=0;i<=n;i++)
		ss[0][i]=true;
	for(int i=1;i<=1000000;i++)
	{
		for(int j=1;j<=n;j++)
		{
			ss[i][j]=ss[i][j-1];
			if(i>=v[j-1])
				ss[i][j]=ss[i][j] || ss[i-v[j-1]][j-1];
		}
	}
}
int main()
{
	int n;
	cin>>n;
	vector<ll> v(n,0);
	for(int i=0;i<n;i++)
		cin>>v[i];
	subset(v,n);
	int q,num;
	cin>>q;
	for(int i=0;i<q;i++)
	{
		cin>>num;
		if(ss[num][n])
			cout<<"Yes"<<endl;
		else
			cout<<"No"<<endl;
	}
}
