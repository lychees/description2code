#include<bits/stdc++.h>
using namespace std;
///zone
int prior[26];
bool kuroko(string a,string b)
{
    int i;
    int p=min(a.length(),b.length());
    for(i=0;i<p;i++)
    {
        if(prior[a[i]-'a']<prior[b[i]-'a']){
            return true;}
        else if(prior[a[i]-'a']>prior[b[i]-'a']){ 
            return false;}
    }
    if(a.length()<b.length()){
        return true;}
    return false;
}
int main()
{
    int t,i,n;
    string str;
    
    cin>>t;
    while(t--)
    {
        cin>>str;
        for(i=0;i<str.length();i++)
        {
            prior[str[i]-'a']=i;
        }
        cin>>n;
        string a[n];
        for(i=0;i<n;i++)
            cin>>a[i];
  
        sort(a,a+n,kuroko);
        for(i=0;i<n;i++)
            cout<<a[i]<<endl;
    }
    return 0;
}
