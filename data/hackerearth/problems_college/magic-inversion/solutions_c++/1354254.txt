#include<bits/stdc++.h>
using namespace std;
int prime(int c)
{
	int i,flag=0;
	for(i=2;i<=sqrt(c);i++)
	{
		if((c%i)==0)
		{
			flag=1;
		break;
		}
	}
//	cout<<flag<<'\n';

	if(flag==1){return 1;}
	else{return 0;}
}
int main()
{
	int n,i,a[500000],b[500000],c=0;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
	}
	for(i=0;i<n;i++)
	{
		cin>>b[i];
		if(b[i]<a[i])
		{
			c++;
		}
	}
//	cout<<c<<'\n';
	int k=prime(c);
	if(k==0){cout<<"MAGIC INVERSION";}
	else{cout<<"SIMPLE INVERSION";}
	return 0;
}
