#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    string s,min,max;
    cin>>s;
    int i;
    for(i=0;i<n;i++)
    {
    	if(s[i]=='w')
    	{
    		max+="v";
    		max+="v";
    		continue;
    	}
    	max+=s[i];
    }
   // cout<<max.length()<<endl;
    for(i=0;i<max.length();i++)
    {
    	if(max[i]=='v'&&max[i+1]=='v')
    	{
    		min+="w";
    		i++;
    	}
    	else
    	{
    		min+=max[i];
    	}
    }
    //cout<<min.length()<<endl;
    cout<<min.length()<<" "<<max.length()<<endl;
    return 0;
}
