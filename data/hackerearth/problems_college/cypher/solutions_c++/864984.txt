#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#include<string>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string str;
		cin>>str;
		int l=str.length();
		int i,a[l];
		for(i=0;i<l;i++)
		{
			a[i]=str[i]+i;
			if(a[i]>122)
			{
				a[i]=a[i]-122;
				a[i]=a[i]%26+96;
			}
		}
		for(i=0;i<l;i++)
		str[i]=a[i];
		
		for(i=0;i<l;i++)
		cout<<str[i];
		cout<<endl;
	}
}
