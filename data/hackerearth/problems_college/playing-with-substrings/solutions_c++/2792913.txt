#include<bits/stdc++.h>

using namespace std;
bool findt(string s,string t)
{
    if(s.length()<t.length())
        return 0;

    for(int i=0;i<s.length();i++)
    {
        if(s[i]==t[0])
        {
            int j;
            for( j=0;j<t.length();j++)
                if(s[i+j]!=t[j])
                break;

            if(j==t.length())
                return true;
        }
    }

    return false;

}
int main()
{
    int l,i,j;
    string s,t;
    map<string,int>mp;


    cin>>s>>t;
    l = s.length();
    int cnt=0;
    //cout<<s<<' '<<t<<'\n';
    for(i=0;i<l;i++)
    {
        string s1="";
        for(j=i;j<l;j++)
        {
            s1 = s1 + s[j];
            //cout<<s1<<'\n';
            if(!mp[s1])
            {

                if(findt(s1,t))
                    cnt++;
                mp[s1]++;
            }
        }
    }

    cout<<cnt;

    return 0;
}
