#include <bits/stdc++.h>
using namespace std;
int a[105][105][105];
int main()
{
	int i,j,k;
    a[0][0][0]=1;
    for(i=1;i<=100;i++)
	{
		if(a[i-1][0][0]==1)
		a[i][0][0]=2;
		else
		a[i][0][0]=1;
	}    
    for(i=1;i<=100;i++)
	{
		if(a[0][i-1][0]==1)
		a[0][i][0]=2;
			else
		a[0][i][0]=1;
	}
	for(i=1;i<=100;i++)
	{
		if(a[0][0][i-1]==1)
		a[0][0][i]=2;	else
		a[0][0][i]=1;
	}
	for(i=1;i<=100;i++)
	{
		for(j=1;j<=100;j++)
		{
			if(a[i-1][j][0]==1 || a[i][j-1][0]==1 || a[i-1][j-1][0]==1)
			a[i][j][0]=2;
			else
			a[i][j][0]=1;
		}
	}
	for(i=1;i<=100;i++)
	{
		for(j=1;j<=100;j++)
		{
			if(a[i-1][0][j]==1 || a[i][0][j-1]==1 || a[i-1][0][j-1]==1)
			a[i][0][j]=2;
			else
			a[i][0][j]=1;
		}
	}
	for(i=1;i<=100;i++)
	{
		for(j=1;j<=100;j++)
		{
			if(a[0][i-1][j]==1 || a[0][i][j-1]==1 || a[0][i-1][j-1]==1)
			a[0][i][j]=2;
			else
			a[0][i][j]=1;
		}
	}
	for(i=1;i<=100;i++)
	{
		for(j=1;j<=100;j++)
		{
			for(k=1;k<=100;k++)
			{
				if(a[i-1][j][k]==1 || a[i][j-1][k]==1 || a[i][j][k-1]==1 || a[i-1][j-1][k]==1 || a[i-1][j][k-1]==1 || a[i][j-1][k-1]==1 || a[i-1][j-1][k-1]==1)
				a[i][j][k]=2;
				else
				a[i][j][k]=1;
			}
		}
	}
	int t;
	cin>>t;
	while(t--)
	{
		cin>>i>>j>>k;
		if(a[i][j][k]==1)
		cout<<"Aishwarya";
		else
		cout<<"Ashima";
		cout<<endl;
	}
    return 0;
}
