#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
	long long t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	long long a[n],i;
    	for(i=0;i<n;i++)
    		cin>>a[i];
    	sort(a,a+n);
    	for(i=n-1;i>=0;i--)
    		cout<<a[i]<<" ";
    	cout<<endl;
    }
    return 0;
}
