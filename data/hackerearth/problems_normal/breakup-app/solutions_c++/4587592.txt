#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
int main(){
	int N,i,j,count=0,day,y1,y2;
	long int date[30]={0},max;
	cin>>N;
	for(i=0;i<=N;i++){
		string s;
		getline(cin,s);
		size_t found;
		for(j=30;j>=1;j--){
			found=s.find(to_string(j),0);
			while(found!=string::npos){
				if(s[0]=='G'){
					date[j-1]+=2;
				}
				else{
					date[j-1]+=1;
				}
				found=s.find(to_string(j),found+1);
			}
		}
	}
	for(j=30;j>=10;j--){
	    if(date[j-1]!=0){
	        y2=(j-1)%10;
	        date[y2]-=date[j-1];
	        y2=j/10;
	        date[y2-1]-=date[j-1];
	    }
	}
	cout<<endl;
	max=*max_element(date,date+30);
	for(j=0;j<30;j++){
		if(date[j]==max){
			count++;
			day=j+1;
		}
	}
	if(count==1){
		if(day==19 || day==20)
			cout<<"Date";
		else
			cout<<"No Date";
	}
	else
		cout<<"No Date";
	return 0;
}
