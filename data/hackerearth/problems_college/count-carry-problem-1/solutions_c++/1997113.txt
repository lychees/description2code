#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	while(n--)
	{
		string s1,s2;
		cin>>s1>>s2;
		int l1=s1.length();
		int l2=s2.length();
		int i=l1-1;
		int j=l2-1;
		int carry=0;
		int ans=0;
		for(;i>=0&&j>=0;i--,j--)
		{
			if(s1[i]-48+s2[j]-48+carry>9)
			{
				ans++;
				carry=s1[i]-48+s2[j]-48+carry-9;
			}
		}
		while(i>=0)
		{
			if(s1[i]-48+carry>9)
			{
				ans++;
				carry=s1[i]-48+carry-9;
			}
		}
		while(j>=0)
		{
			if(s2[j]-48+carry>9)
			{
				ans++;
				carry=s2[i]-48+carry-9;
			}
		}
		if(ans>1)
		cout<<ans<<" "<<"carry operations"<<endl;
		else if(ans==0)
		cout<<"No carry operation"<<endl;
		else
		cout<<ans<<" "<<"carry operation"<<endl;
	}
	return 0;
}
