import heapq
N, Q = map(int, raw_input().split())
T = map(int, raw_input().split())
S = map(int, raw_input().split())
C = sorted(zip([-1*i for i in S], T))
# precompute ans
ans = [-1]*N
ans[0] = C[0][1]
for index in xrange(1, N):
    ans[index] = ans[index-1] + C[index][1]

for _ in xrange(Q):
    K = int(raw_input().strip())
    print ans[K-1]
