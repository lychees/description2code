#include <bits/stdc++.h>
using namespace std;

int main()
{
    int i,j,k,l,n,m,q,x,y;
    cin>>n;
    cin>>q;
    int a[n],b[n];
    for(i=0;i<n;i++)
    	cin>>a[i];
    for(i=0;i<n;i++)
    	cin>>b[i];
    while(q--)
    {
    	long long s=0;
    	cin>>k;
    	cin>>x;
    	cin>>y;
    	for(i=x-1;i<y;i+=2)
    	{
    		if(k==1)
    		{
    			s+=a[i];
    			if(i+1<y)
    				s+=b[i+1];
    			//cout<<i<<" "<<s<<"\n";
    		}
    		else
    		{
    			s+=b[i];
    			if(i+1<y)
    				s+=a[i+1];
    			//cout<<i<<" "<<s<<"\n";
    		}
    	}
    	cout<<s<<"\n";
    }
    return 0;
}
