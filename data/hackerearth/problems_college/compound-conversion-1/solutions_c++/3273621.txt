#include<bits/stdc++.h>
using namespace std;

int main() {
    int i,j,k,t,c;
    string a,b;
    cin >> t;
    for(k=1;k<=t;k++) {
    	int z=0,o=0,o1=0;
    	c=0;
    	cin  >> a >> b;
    	if(a.size() != b.size()) {
    		cout << "-1" << endl;
    	}
    	else {
    		for(i=0;i<a.length();i++) {
    			if(a[i]!=b[i]) {
    				if(a[i]=='?') {
    					if(b[i]=='1')
    						o++;
    					c++;
    				}
    				else if(a[i]=='0') {
    					z++;
    					c++; 
    				}
    				else {
    					o1++;
    					c++;
    				}
    			}

    		}
    	}
    	if(o1>z+o)
    		printf("Case %d: -1\n",k);
    	else {
    	while(z && o1) {
    			c--;
    			z--;
    			o1--;
    		}

    
    		printf("Case %d: %d\n",k,c);
    	}
    }
	return 0;
}
