#include<iostream>
#include<stdio.h>
#define lld long long int 
#define gc getchar_unlocked
using namespace std;


	inline lld read()
{
	lld x;
    register int c = gc();
    x = 0;
    int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
	return x;
}

lld number_of_possible_rectangles(lld m,lld n,lld A) 
{
lld i,a,b,number=0;

for(i=1;i*i<=A;i++)
{
 if(A%i==0)
  {
	a=i;b=A/i;
	if(a>m || b>n)
	{
		number+=0;
	}

	if((a+1)<=m && (b+1)<=n)
	{
		number+=(m-a)*(n-b);
	}

	if(((b+1)<=m && (a+1)<=n) && a!=b)
	{
		number+=(m-b)*(n-a);
	}
   }

}

return number;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    lld m,n,A;
    //scanf("%lld %lld %lld",&m,&n,&A);
    m=read();n=read();A=read();
    printf("%lld\n",number_of_possible_rectangles(m,n,A));
    }
	return 0;
}
