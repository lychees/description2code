#include<iostream>
using namespace std;
int main()
{
	int tc,ar[50005];
	cin>>tc;
	while(tc--)
	{
		int min,n,m,ind;
		cin>>n>>m;
		for(int i=0;i<=n;i++)
			ar[i]=0;
		for(int i=1;i<=m;i++)
		{
			min=1000000,ind=0;
			if(i&1==1)
			{
				for(int j=1;j<=(n+1)/2;j++)
					if(ar[j]<min)
					{
						min=ar[j];
						ind=j;
					}
			}
			else
			{
				for(int j=n;j>=(n+1)/2;j--)
					if(ar[j]<min)
					{
						min=ar[j];
						ind=j;
					}
			}
			ar[ind]++;
		}
		cout<<ind<<" "<<(ar[ind]-1)<<endl;
	}
	return 0;
}
