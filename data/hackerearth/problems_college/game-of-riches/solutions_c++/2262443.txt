#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
std::ios_base::sync_with_stdio(false);
int t;
cin>>t;
while(t--)
{
int n,m,i,k=0;
cin>>n>>m;
    int arr[n];
    for(i=0;i<n;i++)
    cin>>arr[i];
    int maxi=arr[0];
    sort(arr,arr+n);
    i=n-1;
    while(i>=0&&arr[i]>maxi)
    {
     k++;
     i--;
    }

    cout<<k/m<<endl;
}
return 0;
}
