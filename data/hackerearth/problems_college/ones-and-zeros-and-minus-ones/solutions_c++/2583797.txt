	#include <iostream>
	using namespace std;
	int main()
	{
		int T;
		cin>>T;
		
		while(T--) {
			int n, no;
			cin>>n;
			
			int x=0;
			int y=0;
			int lx=0;
			
			int m=0, sm=0;
			
			for(int i=0;i<n;i++) {
				cin>>no;
				
				if(no==1){
					x++;
				}
				else if(no==-1) {
					y++;
					
					sm=lx+x;
					
					if(sm>m){
						m=sm;
					}
					
					lx=x;
					x=0;
				}
				
				if(no!=-1 && i==n-1){
					sm=lx+x;
					if(sm>m)m=sm;
				}
			}
			
			if(y==0) {
				cout<<"0/1"<<endl;
			} else{
				cout<<m<<"/1"<<endl;
			}
		}
		 return 0;
	}
