#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    int t;
    cin>>t;
    ll x;
    while(t--)
    {
        scanf("%lld",&x);
        ll i;
        int flag=0;
        double p=sqrt(x);
        for(i=2;i*i<=p+500000;i++)
        {
            if(x%(i*i)==0)
            {
                flag=1;
                break;
            }
        }
        if(flag==1)
            printf("No\n");
        else
            printf("Yes\n");
    }

}
