#include <iostream>
using namespace std;
#define mod 1000000009
int n;
long long memo[10009][12];
long long func(int id ,int dig)
{
	if(id==n)
	{
		return 1LL;
	}
	long long &value = memo[id][dig];
	if(value!=-1)
	return value;
	else
	{
		value=0LL;
		if(dig==1)
		value = (value+ func(id+1,1)+func(id+1,4)+func(id+1,2))%mod;
		else if(dig==2)
		value = ( value+func(id+1,2)+func(id+1,1)+func(id+1,3)+func(id+1,5))%mod;
		else if(dig==3)
		value = ( value+func(id+1,2)+func(id+1,6)+func(id+1,3))%mod;
		else if(dig==4)
		value = ( value+func(id+1,1)+func(id+1,7)+func(id+1,4)+func(id+1,5))%mod;
		else if(dig==5)
		value = ( value+func(id+1,2)+func(id+1,4)+func(id+1,6)+func(id+1,5) + func(id+1,8))%mod;
		else if(dig==6)
		value = ( value+func(id+1,9)+func(id+1,6)+func(id+1,3)+func(id+1,5))%mod;
		else if(dig==7)
		value = ( value+func(id+1,4)+func(id+1,8)+func(id+1,7))%mod;
		else if(dig==8)
		value = ( value+func(id+1,5)+func(id+1,7)+func(id+1,0)+func(id+1,9)+func(id+1,8))%mod;
		else if(dig==9)
		value = ( value+func(id+1,9)+func(id+1,8)+func(id+1,6))%mod;
		else if(dig==0)
		value = ( value+func(id+1,0)+func(id+1,8))%mod;
	
		
	}
	return value;
}
int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
    	cin>>n;
    	for(int i=0;i<10007;i++)
    	  for(int j=0;j<=9;j++)
    	  memo[i][j]=-1;
    	 long long ans=0;
    	 for(int i=0;i<=9;i++)
    	 ans = (ans+func(1,i))%mod;
    	 cout<<ans<<endl;
    }
    return 0;
}
