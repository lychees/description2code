#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	long long n,k,p;
    	cin>>n>>k>>p;
    		long long arr[n];
    	
    	for(int i=0;i<n;i++)
    	cin>>arr[i];
    	sort(arr,arr+n);
    int j,l;
    int count=0;
    for(j=0,l=k-1;j<n-k,l<n;j++,l++)
    {
    	if(arr[l]<=arr[j]+2*p)
    	{
    		count=1;
    		break;
    	}
    }
    if(count)
    cout<<"NO"<<endl;
    else
    cout<<"YES"<<endl;
    	}
    	
    return 0;
}
