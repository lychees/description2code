#include <iostream>
#define N 1000000
#include<cmath>
using namespace std;

int prime[N+1];

void sieve(){
    prime[0]=1;
    prime[1]=1;
    for(long long i=2;i<=N;i++){
        if(!prime[i]){
                        prime[i]=i;
            for(long long j=i;j*i<=N;j++){
                prime[j*i]=i;
            }
        }
    }
}

long long int primeFactorize(int n){
    long long int yy=1;
    for( ; n > 1 ; ){
        int p = prime[n] , e = 0 ;
        for( ; n % p == 0 ; n /= p , e++ ) ;
        yy*= ((p-1)*(pow((double)p,e-1)));
    }
    return yy;
}



int main()
{
	sieve();
    int T,n,k,i,arr[10001];
    long long ans;
    cin>>T;
    for(i=1;i<10001;i++)
    {
    	arr[i] = primeFactorize(i);
    }
    
    while(T--)
    {
    	ans = 0;
    	cin>>n;
    	for(i=2;i<=n;i++)
    	{
    		ans = ans + arr[i];
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
