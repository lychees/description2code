n = input()
while (n > 0):
    n -= 1
    a, b = map(str, raw_input().split())
    count = 0
    k = len(b)-1
    for i in range(k, -1, -1):
        if len(str(int(a[i]) + int(b[i]))) > 1:
            count += 1

    if count == 0:
        print "No carry operation"
    elif count == 1:
        print count, "carry operation"
    else:
        print count, "carry operations"
