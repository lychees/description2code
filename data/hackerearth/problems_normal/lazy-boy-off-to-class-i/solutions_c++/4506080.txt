#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    int t;
    vector<ll> v;
    scanf("%d",&t);

    ll st,en,n,sum,k,ti;
    while(t--)
    {
        v.clear();
        sum=0;
        ti=0;
        for(int i=0;i<7;i++)
        {
            scanf("%lld",&n);
            sum+=n;
            for(ll j=0;j<n;j++)
            {
                scanf("%lld%lld",&st,&en);
                ll x=en-st;
                v.push_back(x);
            }
        }
        sort(v.begin(),v.end());
        k=ceil(.75*sum);
        //cout<<k<<" size="<<v.size()<<endl;
        for(ll i=0;i<k;i++)
            ti+=v[i];
        //cout<<ti<<endl;
        printf("%lld\n",(604800-ti));

    }
    return 0;
}
