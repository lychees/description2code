#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
   	vector <long> V;
   	long sum,x,N,Q,T;
   	cin>>T;
   	for(long i=0;i<T;i++){
   		cin>>N>>Q;
   		sum = 0;
   		for(long j=0;j<N;j++){
   			cin>>x;
   			V.push_back(x);
   		}
   		sort(V.begin(),V.end());
   		for(long j=1;j<V.size();j++)
   			sum += V[j]-V[j-1];
   		cout<<Q*sum<<endl;
   		V.clear();
   	}
    return 0;
}
