#include <iostream>
#include <map>

using namespace std;

int main() {
    
    map<int, int> hashtab;
    
    int n_tests, n_ints, sample, coll_count;
    
    cin >> n_tests;
    while(n_tests-- != 0) {
        
        coll_count = 0;
        cin >> n_ints;
        
        while(n_ints-- != 0) {
        
            cin >> sample;
            if(hashtab[sample % 10] != 1) {
                hashtab[sample % 10] = 1;
            }
            else {
                coll_count += 1;
            }
        }
        
        cout << coll_count << endl;
        hashtab.clear();
    }
    
    return 0;
}
