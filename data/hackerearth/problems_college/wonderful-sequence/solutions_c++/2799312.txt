#include <bits/stdc++.h>

using namespace std;

main()
{
	int t;
	
	long long int n,s,i,c;
	
	cin>>t;
	
	while(t--)
	{
		cin>>n;
		
		c=0;
		
		long long int a[n];
		
		for(i=0;i<n;i++)
		{
			cin>>a[i];
			
			if(!i)
			{
				s=a[i];
			}
			
			else
			{
				if(a[i]<=a[i-1])
				{
					c++;
				}
			}
		}
		
		cout<<c<<endl;
	}
}
