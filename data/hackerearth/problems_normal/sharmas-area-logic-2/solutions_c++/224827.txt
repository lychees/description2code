#include <bits/stdc++.h>

using namespace std;

int main()
{
    ios::sync_with_stdio(false);
    int t;
    cin>>t;

    while(t--)
    {
        int n;
        cin>>n;

        int maxx,maxy,minx,miny;
        maxx=INT_MIN;
        maxy=INT_MIN;
        miny=minx=INT_MAX;

        int x,y;

        for(int i=0;i<n;i++)
        {
            cin>>x>>y;
            maxx=max(maxx,x);
            minx=min(minx,x);
            maxy=max(maxy,y);
            miny=min(miny,y);
        }

        int hi=abs(maxx-minx);
        int br=abs(maxy-miny);

        if(hi==0)
        {
            cout<<br<<endl;
            continue;
        }
        if(br==0)
        {
            cout<<hi<<endl;
            continue;
        }
        cout<<hi*br<<endl;
    }
    return 0;
}
