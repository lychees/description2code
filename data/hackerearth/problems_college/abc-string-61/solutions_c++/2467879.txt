#include <iostream>
using namespace std;
const long long mod = 1000000007;
long long modPower(long long a, unsigned long long p)
{
	unsigned long long ans = 1;
	while(p)
	{
		if(p%2)
			ans = (ans*a)%mod;
		a = (a*a)%mod;
		p = p/2;
	}
	return ans;
}


int main()
{
	int t;
	cin>>t;
	long long a;
	while(t--)
	{
		cin>>a;
		if(a<3)
			cout<<"0"<<endl;
		else
		{
			long long res = ((modPower(3, a)-((3*modPower(2, a))%mod))%mod+3)%mod;
			if(res<0)
				res += mod;
			cout<<(int)res<<endl;
		}
	}
    return 0;
}
