#include <iostream>
using namespace std;

int main()
{
	int t;
    cin>>t;
    for(int i=0;i<t;i++){
    	int n;
    	cin>>n;
    	int a[n];
    	bool win = false;
    	for(int j=0;j<n;j++){
    		cin>>a[j];
    	}
    	int j = 0;
    	while(j<n){
    		if(a[j] == -1){
    			win = true;
    			break;
    		}
    		if(a[j] == -2){
    			break;
    		}
    		int temp = a[j];
    		a[j] = -2;
    		j= temp;
    	}
    	if(win){
    		cout<<"Win"<<endl;
    	}else{
    		cout<<"Lose"<<endl;
    	}
    }
    return 0;
}
