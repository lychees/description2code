#include<bits/stdc++.h>
using namespace std;
class Cmp{
public:
    bool operator()(pair<long long int,long long int> p1,pair<long long int,long long int> p2){
        return p1.second < p2.second;
    }
};
int main(){
    long long int N,ID,Z,P,L,C,S,nZ;
    pair<long long int,long long int> p;
    cin >> N;
    map<long long int,long long int>m;

    priority_queue<pair<long long int,long long int>, vector<pair<long long int,long long int> >, Cmp>q;
    while(N--){
        cin >> ID >> Z >> P >> L >> C >> S;
        nZ = (P*50) + (L*5) + (C*10) + (S*20);
        m.insert(make_pair(ID,nZ));
        Z = nZ - Z;
        q.push(make_pair(ID,Z));
    }
    /*while(!q.empty()){

        q.pop();
    }*/
    priority_queue<long long int, vector<long long int> >q2;
    int i =0;
    while(1){
        long long int value = q.top().second;
        q2.push(q.top().first);
        q.pop();
        while(value==q.top().second){
            q2.push(q.top().first);
            q.pop();
        }
        while(!q2.empty()){

            q2.pop();
