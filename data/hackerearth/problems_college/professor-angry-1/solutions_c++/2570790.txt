#include <iostream>
using namespace std;

int main()
{
    long long int test,a,b,c,i,j;
    cin>>test;
    while(test--){
    	cin>>a>>b;
    	c=0;
    	for(i=0;i<a;i++){
    		cin>>j;
    		if(j<=0)
    		c++;
    	}
    	if(c<b){
    		cout<<"YES"<<endl;
    	}
    	else
    	cout<<"NO"<<endl;
    	
    }
    return 0;
}
