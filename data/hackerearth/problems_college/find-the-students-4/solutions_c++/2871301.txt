#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    long long int n;
	cin>>n;
	long long int a[n];
	long long int i;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
	}
	sort(a,a+n);
	int c=0;
    for(i=0;i<n;i++)
	 {
	 	c=0;
	 	if(a[i]==0)
	 	{
	 		continue;
		 }
	    long long int min=a[i];
		if(a[i]>min)
		{
			min=a[i];
		}
		for(int j=i;j<n;j++)
		{
			a[j]=a[j]-min;
			c++;
		}
		cout<<c<<"\n";		
     }
 }
