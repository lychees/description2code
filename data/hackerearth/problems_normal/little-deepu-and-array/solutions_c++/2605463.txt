#include <iostream>
#include <vector>
using namespace std;

int main()
{
   	int n ; 
   	cin >> n ;
   	
   	vector<int> val(n+1,0) ; 
   	
   	for(int i = 1 ; i <= n ; i++)
   	{
   		cin >> val[i] ; 
   	}
   	
   	int m ;
   	cin >> m ;
   	//map<int,int> mp ; 
   	
   	for(int i = 0 ; i < m ; i++)
   	{
   		int temp ;
   		cin >> temp ;
   		
   		for(int j = 1 ; j <= n ; j++)
   		{
   			if (temp<val[j])
   				val[j]-- ;
   		}
   		
   	}
   	
   	
   	for(int j = 1 ; j <= n ; j++)
   	{
   		cout << val[j] << " ";
   	}
   	
   	
    return 0;
}
