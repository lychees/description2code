#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	unsigned long long int x,y,b,w,z;
    	cin>>b>>w;
    	cin>>x>>y>>z;
    	unsigned long long int m1=(b+w)*x+z*w;
    	unsigned long long int m2=(b+w)*y+z*b;
    	unsigned long long int m3= b*x+w*y;
    	if(m1<=m3 && m1<=m2){
    		cout<<m1<<endl;
    	}else if(m2<=m3 && m2<=m1){
    		cout<<m2<<endl;
    	}else if(m3<=m1 && m3<=m2){
    		cout<<m3<<endl;
    	}
    }
    return 0;
}
