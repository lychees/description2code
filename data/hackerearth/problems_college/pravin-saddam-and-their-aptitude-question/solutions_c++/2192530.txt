#include <bits/stdc++.h>
using namespace std;

int main()
{
    int arr[]={1,1,2,6,4};
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int a,b,sum=0,i;
    	scanf("%d%d",&a,&b);
    	for(i=a;i<=b;i++)
		{
			if(i<5)
				sum+=arr[i];
			else if(i>=5)
				break;
		}
		printf("%d\n",sum%10);
    }
    return 0;
}
