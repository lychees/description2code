#include <iostream>
#include <vector>
using namespace std;

auto main()->int
{
    long long n;
    cin>>n;
    vector<long long>v;
    vector<long long>count(1000000+1);
    long long unique = 0;
    for(auto i=0;i<n;i++)
    {
        long long p;
        cin>>p;
        if(count[p]==0)
        {
            unique++;
        }
        count[p]++;
    } 
    long long total = 0;
    long long taken = 0;
    for(auto i=1;i<count.size();i++)
    {
        if(count[i]!=0)
        {
            taken ++;
            total += unique - taken;
        }
    }
    cout<<total;
}
