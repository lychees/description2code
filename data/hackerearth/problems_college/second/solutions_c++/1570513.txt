#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long int n,i,max=0,b,m;
    cin>>n>>m;
    long int a[100001]={0};
    //cin>>a[0];
    for(i=0;i<n;i++){
    	cin>>b;
    	a[b]++;
    }
    
    for(i=m;i<=100000;i+=m){
    	//cout<<i<<" ";
    	if(a[i])
    		max=i;
    	else
    		break;
    }
    cout<<endl;
    cout<<max;
    return 0;
}
