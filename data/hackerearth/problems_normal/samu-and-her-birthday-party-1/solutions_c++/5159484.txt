#include <iostream>
#include <string>
#include <vector>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--) {
    	int N, K;
    	cin >> N >> K;
    	vector<string> vs;
    	string str;
    	for(int i = 0; i < N; i ++) {
    		cin >> str;
    		vs.push_back(str);
    	}
    	int ans = K;
    	for(int i = 1; i < (1 << K); i ++) {
    		int cnt = 0;
    		for(int j = 0; j < N; j ++) {
    			int h = 0;
    			for(int k = 0; k < K; k ++) {
    				if( i & (1 << k) ) {
    					if( vs[j][k] == '1' ) {
    						cnt ++;
    						h = 1;
    						break;
    					}
    				}
    			}
    			if( h == 0 ) break;
    		}
    		if( cnt == N ) {
    			cnt = 0;
    			for(int j = 0; j < K; j ++) {
    				if( i & (1 << j) ) {
    					cnt ++;
    				}
    			}
    			ans = min(ans, cnt);
    		}
    	}
    	cout << ans << endl;
    }
    return 0;
}
