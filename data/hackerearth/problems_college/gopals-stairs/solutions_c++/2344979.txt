#include<cstdio>
using namespace std;
#define mod 1000000007
long long int res[100005];
void countWaysUtil(int n,int m)
{
    res[0] = 1; res[1] = 1;
    for (int i=2; i<n; i++)
    {
       res[i] = 0;
       for (int j=1; j<=m && j<=i; j++)
         res[i] =(((res[i]%mod)+(res[i-j]%mod))%mod);
    }
}
 
// Returns number of ways to reach s'th stair
void countWays(int s,int m)
{
    countWaysUtil(s+1, m);
}

    return 0;
}
