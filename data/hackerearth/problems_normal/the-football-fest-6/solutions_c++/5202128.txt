#include <iostream>
#include <stack>

using namespace std;

int main()
{
    long totalTests=0;
    
    cin >> totalTests;
    for(long i = 0; i < totalTests; i++)
    {
    	int totalPass = 0, currentPlayer = 0, prevPlayer = 0;
    	
    	cin >> totalPass >> currentPlayer;
    	stack<long> players;
    	
    	for(long j = 0; j < totalPass; j++)
    	{
    		char passType;
    		cin >> passType;
    		
    		players.push(currentPlayer);
    		
    		switch(passType)
    		{
    			case 'P':
    				cin >> currentPlayer;
    				prevPlayer = 0;
    				//cout << "P>>" << currentPlayer << endl;
    				break;
    				
    			case 'B':
    				if(prevPlayer == 0)
    				{
    					prevPlayer = players.top();
    					players.pop();
    					currentPlayer = players.top();
    				}
    				else
    				{
    					int temp = currentPlayer;
    					currentPlayer = prevPlayer;
    					prevPlayer = temp;
    				}
    				//cout << "B>>" << prevPlayer << "-" << currentPlayer << endl;
    				break;
    		}
    	}
    	cout << "Player " << currentPlayer << endl;
    }
    return 0;
}
