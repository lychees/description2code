import itertools

count = 0


def converse(temp_line):
    global count
    count = 0
    for i in temp_line:
        if ((sum(i) % 2 == 0) and (i[0] != i[1])):
            count += 1


al_ = input()
for _ in range(al_):
    curr = input()
    line = map(int, raw_input().split())
    converse(list(itertools.combinations(line, 2)))
