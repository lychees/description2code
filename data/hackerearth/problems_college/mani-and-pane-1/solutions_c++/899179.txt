#include <bits/stdc++.h>
using namespace std;

int main()
{
	long long arr[33][2] = { { 1, 814434 }, { 2, 99888 }, { 3, 28302 }, { 4, 12102 }, { 5, 5982 }, { 6, 2958 }, { 7, 1950 }, { 8, 1176 }, { 9, 924 }, { 10, 654 }, { 11, 474 }, { 12, 312 }
	, { 13, 204 }, { 14, 204 }, { 15, 114 }, { 16, 114 }, { 17, 60 }, { 18, 60 }, {
		19, 60 }, { 20, 60 }, { 21, 24 }, { 22, 24 }, { 23, 24 }, { 24, 24 }, { 25, 24 }
		, { 26, 6 }, { 27, 6 }, { 28, 6 }, { 29, 6 }, { 30, 6 }, { 31, 6 }, { 32, 6 }, {
			33, 6 }, };

	long long div = (100 * 99 * 98);

	int t;
	scanf("%d", &t);
	while (t--) {
		long long x, z;
		scanf("%lld %lld", &x, &z);

		long long cnt = 0;
		for (int i = 0; i < 33; i++) {
			if (abs(x - z) % arr[i][0] == 0)
				cnt += arr[i][1];
		}

		printf("%0.6lf\n", (double)cnt / div);
	}
}
