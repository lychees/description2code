#include<bits/stdc++.h>


using namespace std;

int main()

{

  /*ios_base::sync_with_stdio(false);*/

  int n;
  cin>>n;
  vector<int> v;

  for(int i=0;i<n;i++)
  {
       int k;
      cin>>k;
      v.push_back(k);
  }
  int q;
  cin>>q;
  while(q--)
  {
      int l,i,s=0,flag=0;
      cin>>l;
      for(i=0;i<n;i++)
      {
        s+=v[i];
      if(s>=l)
      {
    flag=1;
      break;
      }
      }
      if(flag==1)
      cout<<i+1<<"\n";
      else cout<<"-1"<<"\n";

  }
  return 0;
}
