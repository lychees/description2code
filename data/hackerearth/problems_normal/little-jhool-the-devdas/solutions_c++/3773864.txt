#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int arr[n];
    for(int i=0;i<n;i++)
    	cin>>arr[i];
    sort(arr,arr+n);
    int count=0;
    for(int i=0;i<n;i++)
    {
    	if(arr[i]%2==1)
    	count++;
    }
    int flag=0;
    if(count==0)
    {
    	cout<<":(";
    	
    }
    else
    {
   	if(count%2==1)
    flag=0;
    else
    flag=1;
    int sum=0;
    if(flag==0)
    {
    	for(int i=0;i<n;i++)
    	sum+=arr[i];
    }
    else
    {
    	for(int i=0;i<n;i++)
    	{
    		if(arr[i]%2==1 && flag==1)
    			{
    				flag=0;
    			}
    			else
    			{
    				sum+=arr[i];
    			}
    	}
    }
    cout<<sum;
    
    }
   
    return 0;
}
