#include<cstdio>
#include<cstring>

#define MAX_N 4

char double_factorial[MAX_N+1][30] = {"1", "1", "2", "720", "620448401733239439360000"};


int main(){
    int t;
    scanf("%d", &t);
    while(t--){
        int n, m;
        scanf("%d %d", &n, &m);
        if(n > 4){
            printf("0\n");
        }
        else{
            if(m == 0){
                printf("0");
            }
            else{
                int l = strlen(double_factorial[n]);
                int ans[30];
                int j = 0;
                bool zero = true;
                int mask = -1;
                for(int i=0;i<m && i<l;i++){
                    if(double_factorial[n][l-i-1] != '0'){
                        zero = false;
                        mask = j;
                    }
                    ans[j++] = double_factorial[n][l-i-1];
                    
                }
                if(zero){
                    printf("0");
                }
                else{
                    for(int i=mask;i>=0;i--){
                        printf("%c", ans[i]);
                    }
                }   
            }
            printf("\n");
        }
    }   
}
