#include <iostream>
using namespace std;

int main()
{
    char s[100];
    cin >> s;
    string check = "lovea";
    int index = 0;
    for(int i = 0; s[i] != '\0'; i++){
    	if(s[i] == check[index]){
    		index++;
    		if(index == 4){
    			break;
    		}
    	}
    }
    if(index == 4){
    	cout << "I love you, too!" << endl;
    }else{
    	cout << "Let us breakup!" << endl;
    }
    return 0;
}
