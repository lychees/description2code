#include <bits/stdc++.h>
using namespace std;

int main() {
	
	long long int t,n,m,i,j,k,l,count;
	cin>>t;
	while(t--){
		count=0;
		string str;
		cin>>str;
		k=str.length();
		for(i=0;i<k;i++)
		{
			if(str[i]=='D' || str[i]=='B' || str[i]=='S')
			continue;
			
			if(i+1<k && (str[i+1]=='B' || str[i+1]=='S'))
			continue;
			
				
			if(i+2<k && str[i+2]=='B')
			continue;
					
			if(i-1>=0 && str[i-1]=='S')
			continue;
						
			count++;
			
		}
		cout<<count<<"\n";
	}
}
