#include <iostream>
#include<cmath>
typedef unsigned long long ull;
using namespace std;
ull a,b,c,d,k,x,t,j;
ull bs(ull n,ull k)
{ull l=0;
ull h=n;
ull m;
 while(l<h)
 { m=(l+h)/2;
    j=a*(pow(m,3)) + b*(pow(m,2)) +c*m +d;
    if(j>k)
     h=m-1;
     else if(j<k)
     l=m+1;
     else
     return m;
 }
 return l;
}
int main()
{
    cin>>t;
    while(t--)
    {cin>>a>>b>>c>>d>>k;
     x=cbrt(k);
     ull g=bs(x,k);
     if(d>=k)
     cout<<0<<endl;
     else
    { if((a*(pow(g,3)) + b*(pow(g,2)) +c*g +d)>k)
     cout<<--g<<endl;
     else
     cout<<g<<endl;
    }
    }
     
    return 0;
}
