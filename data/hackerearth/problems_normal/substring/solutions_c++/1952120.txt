#include <bits/stdc++.h>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    int tc;
    cin>>tc;
    while(tc--) {
    	string str;
    	cin>>str;
    	long long a = 0,count = 0;
    	for(int i=0;i<str.length();i++) {
    		if(str[i] != 'a' && str[i] != 'A' && str[i] != 'b' && str[i] != 'B' && str[i] != 'c' && str[i] != 'C') {
    			count += a*(a+1)/2;
    			a = 0;
    		} else {
    			a++;
    		}
    	}
    	count += a*(a+1)/2;
    	cout<<count<<endl;
    }
    return 0;
}
