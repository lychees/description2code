if __name__ == '__main__':
    n = int(raw_input())
    h_v = (map(int, raw_input().split()))
    assert 3 <= n <= 1000
    assert len(h_v) == n
    h_v.sort()
    one = []
    two = []
    counter = 1
    for each in h_v:
        if counter % 2 == 0:
            two.append(each)
        else:
            one.append(each)
        counter += 1
    two.reverse()
    final = one + two
    print max([abs(j-i) for i, j in zip(final[:-1], final[1:])])
