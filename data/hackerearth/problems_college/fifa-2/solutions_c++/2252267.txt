#include <iostream>

using namespace std;

int main() {
	int t;
	t = 1;
	
	while (t--) {
		int n;
		cin >> n;
		int x[n], y[n];
		
		int freq[1000000];
		
		for (int i = 0; i < n; i++) {
			cin >> x[i] >> y[i];
			
			freq[x[i]]++;
		}
		
		for (int i = 0; i < n; i++) {
			int a, b;
			a = b = n - 1;
			a -= freq[y[i]];
			b += freq[y[i]];
			
			cout << b << " " << a << endl;
		}
	}
}
