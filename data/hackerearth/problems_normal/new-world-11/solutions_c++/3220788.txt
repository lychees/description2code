int T,n,x[100010],k;
 
bool check(int m){
	int i=1, res = 0,j=1;
	bool ok = 0;
	while(i <= n){
		j = max(i,j);
		res++;
		while(j <= n && x[i] + m >= x[j]) j++;
		j--;
		if(j == n){
			ok = 1;
			break;
		}
		if(j == i) break;
		i = j;
	}
	return (ok && res <= k);
}
 
int main () {
	scanf("%d",&T);
	while(T--){
		scanf("%d%d",&n,&k);
		for(int i=1;i<=n;i++) scanf("%d",&x[i]);
		int l = 0,r = (int)(1e9) + 1;
		while(r - l > 1){
			int mid = (r + l) / 2;
			if(check(mid)) r = mid;
			else l = mid;
		}
		printf("%d\n",r);
	}
 
return 0;
}
