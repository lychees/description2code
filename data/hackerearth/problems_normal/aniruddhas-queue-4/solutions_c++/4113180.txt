#include <iostream>
using namespace std;

int main()
{
	long long m,i,n,x,a[100005],sum,t,l;
	cin>>t;
	while(t--)
	{
		sum=0;
		cin>>n;
		for(i=0;i<n;i++)
		{
			cin>>a[i];
			sum+=a[i];
			if(a[i]>0)
				l=i;
		}
		cin>>m;
		m=m%sum;
		if(m==0)
			cout<<l+1<<endl;
		else
		{
			i=0;
			while(m>0)
			{
				m-=a[i];
				if(m<=0)
					break;
				i++;
			}
			cout<<i+1<<endl;
		}
	}
    return 0;
}
