#!/usr/bin/env python

def days(t):
    res = 0
    cur = 0
    for i in range(n):
        if cur + q[i] > t:
            res += 1
            cur = 0
        cur += q[i]
    if cur > 0:
        res += 1
    return res


n, m = map(int, raw_input().split())
q = map(int, raw_input().split())

l = max(q)
r = int(1e18)
while r - l > 1:
    mid = (l + r) / 2
    if days(mid) > m:
        l = mid + 1
    else:
        r = mid

while days(l) > m:
    l += 1

print(l)
