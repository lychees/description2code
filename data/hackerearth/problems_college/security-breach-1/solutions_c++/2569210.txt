#include <iostream>
#include <vector>
#include <string>

using namespace std;

int main()
{
	string str;
    
    int n;
    int count=0;
    cin>>n;
    while(n--)
    {
    	vector<string> s;
    	cin>>str;
    	count=0;
    	for(int i=0;i<str.size();i++)
    	{
    		if(str[i]=='2')
    		{
    			s.push_back("cde");
    			++count;
    		}
    		if(str[i]=='7')
    		{
    			s.push_back("acf");
    			++count;
    		}
    		if(str[i]=='6')
    		{
    			s.push_back("b3");
    			++count;
    		}
    		if(str[i]=='9')
    		{
    			s.push_back("c6a");
    			++count;
    		}
    		
    	}
    	if(count==0)
    	{
    		cout<<endl;
    	}
    	else
    	{
    		for(int i=0;i<s.size();i++)
    		{
    			cout<<s[i];
    		}
    		cout<<endl;
    	}
    }
    
    return 0;
}
