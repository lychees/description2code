num_test_cases = int(raw_input())
for i in xrange(num_test_cases):
	num_inputs = int(raw_input())
	existential_map = {}
	num_baggage_needed = 0
	for _ in xrange(num_inputs):
		food_type,panda_type = map(int,raw_input().split())
		food_status = existential_map.get(food_type,0)
		existential_map[food_type]=food_status+1
		
		panda_food = existential_map.get(panda_type,0)
		if panda_food == 0:
			num_baggage_needed += 1
		else:
			existential_map[panda_type]-=1
		
	print num_baggage_needed
