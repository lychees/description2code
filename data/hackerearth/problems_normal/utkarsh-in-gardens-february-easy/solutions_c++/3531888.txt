#include <bits/stdc++.h>
using namespace std;
bitset<2010> v[2010];
int n;
int main(){
    cin>>n;
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++){
            int x;
            cin>>x;
            v[i][j]=x;
        }
    }
    long long ans=0;
    for(int i=0;i<n;i++){
        for(int j=i+1;j<n;j++){
            int curr = (v[i]&v[j]).count();
            ans += curr*(curr-1)/2;
        }
    }
    cout<<ans/2<<endl;
}
