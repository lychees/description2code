#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int t, a, n;
    vector<int> vec;
    cin >> t;
    while (t--) {
        cin >> n;
        int count = 0;
        int b = n;
        while (n--) {
            cin >> a;
            vec.push_back(a);
        }
        for (int i = 0; i < b; i++) {
            for (int j = i + 1; j < b; j++) {
                if ((vec.at(i) + vec.at(j)) % 3 == 0) {
                    count++;
                    // cout<<"\n"<<vec.at(j)<<vec.at(i)<<"\n";
                }
            }
        }
        vec.clear();
        cout << count << "\n";
    }
    return 0;
}
