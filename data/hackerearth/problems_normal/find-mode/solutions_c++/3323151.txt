#include<iostream>
#include<map>
#include<vector>
using namespace std;
int main()
{
	int t,n,i,temp,ele;
	cin>>t;
	while(t--)
	{
		cin>>n;
		temp=0;
		map<int,int>m;
		map<int,int>::iterator it;
		map<int,int>::reverse_iterator it2;
		vector<int>v;
		for(i=0;i<n;i++)
		{
			cin>>ele;
			m[ele]++;
		}
		for(it=m.begin();it!=m.end();++it)
		{
			if(it->second>temp)
			{
				temp=it->second;
			}
		}
		for(it2=m.rbegin();it2!=m.rend();++it2)
		{
			if(it2->second==temp)
			cout<<it2->first<<" ";
		}
		
		cout<<endl;
	}
}
