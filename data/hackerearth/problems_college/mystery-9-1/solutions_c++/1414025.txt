#include<bits/stdc++.h>
using namespace std ;
int main()
{
	int t , i , f ;
	cin >> t ;
	int l , flag , n ;
	string str ;
	while ( t-- )
	{
		cin >> str ;
		l = str.size() ;
	
	flag = 1 ;
	n = l - 1 ;
	for( i = 0 ; i < l/2 ; i++)
	{
		if( str[i] != str[n])
		{
			flag = 0 ;
		}
		n-- ;
	}
	if( flag == 1)
	{
		if( l % 2 == 0)
		{
			int g = l/2 ;
			
			int re = str[g-1] ;
			cout<<re<<"\n" ;
		}
		else
		{
			//l  = l / 2 ;
		  int re = str[l/2] ;
		  cout<<re<<"\n" ;
		}
	}
	if( flag == 0)
	{
		cout<<"-1\n" ;
	}
}
	return 0 ;
}
