#include <bits/stdc++.h>
using namespace std;
int cf(int n)
{
	int c=0;
	for(int i=1;i<=sqrt(n);i++)
   		{
   			if(n%i==0)
   			{
   				if(n/i==i)
   				c++;
   				else
   				c+=2;
   			}
   		}
   		return c;
   		
}
int main()
{
   	int t;
   	cin>>t;
   	int a[1001],k=1,b=1;
   	while(k<=1000)
   	{
   		a[k++]=b;
   		b=k+a[k-1];
   		//cout<<a[k-1]<<' ';
   	}
   	while(t--)
   	{
   		int n;
   		cin>>n;
   		int f=0;
   		for(int kk=1;f==0;kk++)
   		{
   			int p=cf(a[kk]);
   			if(p>n)
   			{
   				cout<<a[kk]<<endl;
   				f=1;
   			}
   		}
   	}
    return 0;
}
