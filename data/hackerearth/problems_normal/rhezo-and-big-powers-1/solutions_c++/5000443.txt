#include<bits/stdc++.h>
#define ll long long
#define Mod 1000000007
#define forl(i,a,b) for(i=a;i<b;i++)
//const long long  int Mod=pow(10,9)+7;
using namespace std;

ll power_by_iteration(ll base,ll power)
{
	ll y=1;
	if(power==0)
	{
		return 1;
	}
	else if(power==1)
	{
		return base%Mod;
	}
	else
	{
		while(power>1)
		{
			 if(power%2==0)
	           {
	  	         base=(base%Mod*base%Mod)%Mod;
	  	         power/=2;
	           } 	
	        else
	           {
	              y=(base%Mod*y%Mod)%Mod;    	
	  	         base=(base%Mod*base%Mod)%Mod;
	  	         power=(power-1)/2;
	           }
		}
	   return (y%Mod*base%Mod)%Mod;
	}

}
int main(void)
{
	ll Base,i;
	string Exponent;
	cin>>Base>>Exponent;
	
	ll power=0;
	forl(i,0,Exponent.length())
	{
		power=(power*10+(Exponent[i]-'0'))%(Mod-1);
	}
	cout<<power_by_iteration(Base,power)<<endl;
	return 0;
}
