#include<stdio.h>
#include<iostream>
#include<string>
#include<string.h>
using namespace std;

char arr[26];

void initialize(){
	int i;
	for(i=0;i<26;i++){
		arr[i]=char(97+i);
	}
}


int main(void){

int k;
string s;
cin>>k;
fflush(stdin);
cin>>s;
int l= s.length();
int i,j;
int flag=0;
int count=0;
int co;
initialize();
for(i=0;i<l-k+1;i++){
	co=0;
	for(j=i;j<l;j++){
		
		if(s[j]==arr[int(s[j])-97]){
			co++;
			arr[int(s[j])-97]='A';
		}
		if(co==k){
			count++;
		}
		if(co>k){
			break;
		}

	}
	

	initialize();
}

cout<<count;

return 0;

}
