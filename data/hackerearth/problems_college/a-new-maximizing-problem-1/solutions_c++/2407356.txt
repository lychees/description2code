#include <bits/stdc++.h>
#define ll long long
#define mod  1000000007
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int a[100010],b[100010];
	ll int test;
	cin>>test;
	while(test--)
	{
		ll int n;
		cin>>n;
		ll int i,j;
		for(i=0;i<n;i++)
		cin>>a[i];
		sort(a,a+n);
		int flag=0;
		j=0;
		ll int ans1=0,ans2=0;
		for(i=0;i<n;i++)
		{
			if(!flag)
			{
				b[i]=a[j];
			}
			else
			{
				b[i]=a[n-1-j];
				j++;
			}
			flag^=1;
			ans2=(ans2+abs(b[i]))%mod;
		    //~ cout<<b[i]<<" ";
		}
		//~ cout<<endl;
		
		for(i=0;i<n;i++)
		{
			if((i%2) ==0)
			{
				ans1=(ans1-b[i]+mod)%mod;
			}
			else
			ans1=(ans1+b[i])%mod;
		}
		ll int ans=(ans1*ans2)%mod;
		cout<<ans<<endl;
	}
	return 0;
}
