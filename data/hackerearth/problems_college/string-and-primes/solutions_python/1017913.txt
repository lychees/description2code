def prime(val):
	if val !=2:
		if(val % 2 == 0):
			return 0
		else:
			valBy = val/2
			for divider in range(3,valBy+1,2):
				if val % divider == 0:
					return 0
			return 1
	else:
		return 1

def reduce_string(string):
	new_string=''
	for char in string:
		if prime(ord(char)) == 0:
			new_string+=char
	return new_string
