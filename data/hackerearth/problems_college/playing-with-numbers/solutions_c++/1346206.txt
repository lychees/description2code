#include <cstdio>

#define ull unsigned long long

int main(){
    int t, res, win = 1;
    ull n;
    scanf("%d", &t);
    while (t--){
    	scanf("%llu", &n);
    	res = 0;
    	for (int i = 0; n >> i; i++)
    		 res += (n >> i) & 1;
    	if ((res & 1) == 0)
    		win ^= 1;
    	printf("%s\n", win ? "Sherlock" : "Watson");
    }
    return 0;
}
