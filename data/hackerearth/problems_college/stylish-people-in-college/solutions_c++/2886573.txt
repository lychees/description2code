lint solve (VI &A, int n) {
    VLI DP(n, 0);
    lint ans = 0;
    FOR (j, 3, n) {
        DP[j] = max(DP[j - 2], DP[j - 3]) + A[j];
        ans = max(ans, DP[j]);
    }
    return ans;
}
