#include <iostream>
#include <unordered_map>
using namespace std;

int main()
{
	int n, q;
	unordered_map<long, bool> map;
	cin >> n >> q;
	long temp;
	for(int i = 0; i < n; ++i) {
		cin >> temp;
		map[temp] = true;
	}
	for(int i = 0; i < q; ++i) {
		int a;
		cin >> a;
		if(map.find(a) != map.end())
			cout << "YES\n";
		else
			cout << "NO\n";
	}
	return 0;
}
