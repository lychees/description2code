#include <bits/stdc++.h>

using namespace std;


int main()
{
    int t, i, j, l;
    cin >> t;
    string ts;
    getline(cin, ts);
    while (t--) {
   		string s;
   	    getline(cin,s);
   	    l = s.size();
   	    l = sqrt(l);
   	    for(i = 0;i < l; i++) {
            for(j = i; j < s.size(); j = j + l) {
   	    		cout << s[j];
   	    	}
   	    }
   	    cout<<endl;
   	}

    return 0;
}
