#include <iostream>
#include <math.h>
using namespace std;

#define pi 3.1415

int main()
{
    int t; cin>>t;
    while(t--)
    {
    	double a,b,c;
    	cin>>a>>b>>c;
    	double r = a*b*c/ sqrt((a+b+c)*(a+b-c)*(a+c-b)*(b+c-a));
    	double ans = pi*r*r;
    	cout.precision(4);
    	cout<<fixed<<ans<<endl;
    }
}
