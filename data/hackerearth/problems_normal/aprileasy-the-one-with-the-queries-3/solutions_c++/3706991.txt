#include <iostream>
#include <vector>
using namespace std;

vector<int> loc[30];
char S[100005];
char a[100005];

int bs(int l, int r, int arr, int targ){
	int m = (l+r)/2;
	if(l >= r) return l;
	if(loc[arr][m] > targ && (m == 0 || loc[arr][m-1] <= targ)) return m;
	if(loc[arr][m] <= targ) return bs(m+1, r, arr, targ);
	else return bs(l, m, arr, targ);
}

int main()
{
	int Q;
    scanf("%s", S);
    for(int i=0; S[i]!='\0'; i++) loc[S[i]-'a'].push_back(i);
    scanf("%d", &Q);
    for(int i=0; i<Q; i++){
    	int ptr=-1;
    	int temp;
    	scanf("%s", a);
    	for(int j=0; a[j]!='\0'; j++){
    		if(loc[a[j]-'a'].size() == 0) { ptr = -2; break; }
    		temp = bs(0, loc[a[j]-'a'].size()-1, a[j]-'a', ptr);
    		if(loc[a[j]-'a'][temp] > ptr) ptr = loc[a[j]-'a'][temp];
    		else { ptr = -2; break; }
    	}
    	if(ptr != -2) printf("Yes\n");
    	else printf("No\n");
    }
    return 0;
}
