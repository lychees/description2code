#include <bits/stdc++.h>
using namespace std;

long a[1000002]={0};
long b[1000002]={0};
long ans[1000002]={0};

long smallest_prime(long n){
	if(n%2==0)
		return 2;
	for(long i=3;i<=sqrt(n);i+=2){
		if(a[i]!=-1 && n%i==0){
			return i; 
		}
	}	
	//return n;
}

long count(long n){
	int o=0;
	if(n==2)
	return 500000;
	if((n*n)<0){
		return 1;
	}
	long c=1;
	for(long i=n*n;i<=1000000;i++){
		
		if(b[i]==n){
			c++;
			if(o!=5){
			//cout<<":o  ";
			o++;
		}
		}
	}
	return c-2;
}

int main()
{
	long y=1;
    for(long i=2;i<=1000000;i++){
    	long j=i+i;
    	y=1;
    	while(j<=1000000){
    		if(a[j]==0)
    			y++;
    		a[j]=-1;
    		j+=i;
    	}
    	ans[i]=y;
    }
    /*
