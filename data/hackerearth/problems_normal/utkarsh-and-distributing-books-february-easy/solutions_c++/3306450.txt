#include<iostream>
using namespace std;
int main()
{
int t,n;
cin>>t;
while(t-->0)
    { cin>>n; int l=n,i,min,s=0;
    int a[n];
    while(n>0)
        cin>> a[l-(n--)];
    min =a[0];
    for(i=1;i<l;i++)
        {if(a[i]<min)
        min=a[i];}
        for(i=0;i<l;i++)
            {s=s+a[i];}
            cout<<min-1<<" "<<s-l<<endl;


    }
}
