#include <iostream>
using namespace std;

int main()
{
     int t,n,count,i,a[100];
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	count=0;
    	for(i=1;i<=n;i++)
    	{cin>>a[i];}
    	for(i=1;i<=n;i++)
    	{
    		if(a[i]%2==0)
    	    {count=count+1;}
    	}
    cout<<count<<endl;}
    return 0;
}
