#include <iostream>
#include<string.h>
using namespace std;

long long int reverse(long long int n)
{
	long long int reverse = 0;
	while(n>0)
   	{
      reverse = reverse * 10;
      reverse = reverse + n%10;
      n = n/10;
   	}
   	return reverse;
}

int main() {
	
	int t;
	cin>>t;
	
	while(t--)
	{
		long long int a;
		cin>>a;
		
		long long int rev = reverse(a);
		long long int temp1 = a*a;
		long long int temp2 = rev*rev;
		temp2 = reverse(temp2);
		
		if(temp1==temp2 && a!=rev)
		cout<<rev<<endl;
		else
		cout<<"No"<<endl;
	}
	return 0;
}
