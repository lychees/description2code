#include <iostream>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    	long long n;
    	scanf("%lld",&n);
    	if(n==2 ){
    		printf("1\n");
    		continue;
    	}
    	if(n%3==0){
    	printf("%lld\n",n/3);
    	}
    	else{
    	printf("%lld\n",n);	
    	}
    }
    return 0;
}
