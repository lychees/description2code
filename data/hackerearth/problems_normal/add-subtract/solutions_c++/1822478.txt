#include <iostream>
#include <climits>
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
    int t;
    cin >> t;
    
    while (t--) {
    	int n, k;
    	cin >> n >> k;
    	
    	int fin = INT_MAX;
   
    	int a[n];
    	for (int i = 0; i < n; i++) {
    		cin >> a[i];
    	}
    	
    	for (int i = 0; i < n; i++) {
    		int cc = 0;
    		vector <int> v;
    		
    		for (int j = 0;  j < n; j++) {
    			if (i != j) {
    				if (a[j] == a[i]) {
    					v.push_back(0);
    				} else if (a[j] > a[i]) {
    					
    					v.push_back((5 * (a[j] - a[i])));
    				} else {
    					v.push_back((3 *(a[i] - a[j])));
    				}
    			}
    		}
    		
    		sort(v.begin(), v.end());
    		
    		for (int j = 0; j < k - 1; j++) {
    			cc = cc + v[j];
    		}
    		
    		v.clear();
    		
    		fin = min(cc, fin);
    	}
    	cout << fin << endl;
    }
    return 0;
}
