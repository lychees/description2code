#include <iostream>
#include <cstring>
using namespace std;

int main() 
{
	int t,l,i,f=0,s=0;
	char n[19];
	cin>>t;
	while(t--)
	{
		cin>>n;
		l=strlen(n);
		for(i=0;i<l;i++)
		{
			if(n[i]=='4')
			{
				f++;
			}
			else if(n[i]=='7')
			s++;
		}
		if(f+s==4 || f+s==7)
		cout<<"YES\n";
		else
		cout<<"NO\n";
		i=0;
		s=0;
		f=0;
	}
	return 0;
}
