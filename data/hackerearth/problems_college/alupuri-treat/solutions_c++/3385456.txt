#include <iostream>
#include<math.h>
using namespace std;

int main()
{
	long int p=0,q=0;
	int m;
	int a[101][101];
	cin >> m;
	for(int i=0;i<m;i++)
	{
		for(int j=0;j<m;j++)
		{
			cin >> a[i][j];
			if(i==j)
			p+=a[i][j];
			if(i+j+1==m)
			q+=a[i][j];
		}
	}
	cout << abs(p-q) << endl;
    return 0;
}
