#include <iostream>
using namespace std;
#include<string>
int main()
{
	int x;
    cin>>x;
    for(int i=0;i<x;i++)
    {
    	int y;
    	cin>>y;
    	string str;
    	int arr[26]={0};
    	for(int j=0;j<y;j++)
    	{
            cin>>str;
            int len=str.length();
            bool check[26]={false};
            for(int k=0;k<len;k++)
            {
                int p=str[k]-'a';
                if(check[p]==false)
                {
                    check[p]=true;
                    arr[p]++;
                }
            }
    	}
    	int mx=0;
    	char c;
    	for(int w=0;w<26;w++)
        {
            if(arr[w]>mx)
            {
                mx=arr[w];
                c=(char)(w+'a');
            }
        }
        cout<<c<<endl;
    }
}
