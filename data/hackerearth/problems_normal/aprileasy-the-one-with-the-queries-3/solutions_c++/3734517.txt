const int N = 1e5+5;
char s[N];
vector<int> v[26];
int binary(int pos,int prev) {
  int low = 0;
  int high = v[pos].size()-1;
  int ans = -1;
  while(low<=high) {
    int mid = (low+high)/2;
    if(v[pos][mid]>prev) {
      ans = v[pos][mid];
      high = mid - 1;
    }
    else {
      low = mid+1;
    }
  }
  return ans;
}
bool func(const string & temp) {
  int prev = -1;
  for(int i=0;i<temp.size();i++) {
    int pos = temp[i]-'a';
    int l = binary(pos,prev);
    if( l == -1)
      return false;
    prev = l;
  }
  return true;
}
