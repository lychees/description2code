#include <bits/stdc++.h>
using namespace std;
bool isPrime(int n)
{
	if(n==1)
	return true;
	if(n==2)
	return true;
	if(n%2==0)
	return false;
	for(int i=3;i<=sqrt(n);i+=2)
	if(n%i==0 && i<n)
	return false;
	return true;
}

int main()
{
    int n,i;
    cin>>n;
    for(i=1;i<=n;i++)
    if(!isPrime(i))
    cout<<i<<endl;
    return 0;
}
