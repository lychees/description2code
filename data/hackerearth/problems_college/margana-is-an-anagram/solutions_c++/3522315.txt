#include <bits/stdc++.h>
using namespace std;
int freq1[30],freq2[30];
int main()
{
	int test;
	string a,b;
	cin >> test;
	while(test--)
	{
		int counter=0;
		cin >> a >>b;
		for(int i=0;i<26;i++)
		{
			freq2[i]=freq1[i]=0;
		}
		for(int i=0;i<a.length();i++)
		{
			freq1[a[i]-'a']++;
		}
		for(int i=0;i<b.length();i++)
		{
			freq2[b[i]-'a']++;
		}
		for(int i=0;i<26;i++)
		{
			if(freq2[i]==freq1[i])
				++counter;
			
		}
		if(counter==26)
			cout <<"Yes\n";
		else
			cout<<"No\n";
		
	}
 
    return 0;
}
