#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>
#include <cmath>

using namespace std;

int gcd(int a, int b)
{
    if ( b == 0 ) return a;
    return gcd(b,a%b);
}

int main()
{
    int t,n,r;
    cin >> t;
    while ( t-- ) {
          cin >> n >> r;
          r--;
          int t1 = (int)(sqrt(r));
          int ans = 0;
          for ( int i = 1; i <= t1; i++ ) {
              if ( r%i == 0 ) {
                   ans++;
                   if ( i != r/i ) {
                        ans++;
                   }
              }
          }
          ans = n-ans;    
          int ans1 = gcd(n,ans);
          ans /= ans1;
          n /= ans1;
          cout << ans << "/" << n << endl;
    }
    return 0;
}
