#include <cstdio>
#include <vector>
using namespace std;

typedef long long ll;
bool P[100001];
ll i,j,k,l;
vector<ll> p;

ll po(ll q,ll w)
{
	ll res=1;
	while(w--) res*=q;
	return res-1;
}

ll sod(ll x)
{
	ll c,ans=1;
	for(k=0;k<l;k++)
	{
		if(x<p[k]) break;
		if(x%p[k]==0)
		{
			c=1;
			while(x%p[k]==0)
			{
				c++;
				x/=p[k];
			}
			ans=ans*( po(p[k],c)/(p[k]-1) );
		}
	}
	if(x>1)
	{
		ans=ans*( x+1 );
	}
	return ans;
}

int main()
{
	for(i=2;i<100001;i++)
	{
		if(!P[i])
		{
			p.push_back(i);
			for(j=i<<1;j<100001;j+=i)
				P[j]=1;
		}
	}
	l=p.size();
	scanf("%lld",&i);
	while(i--)
	{
		scanf("%lld",&j);
		printf("%lld\n",(sod( j/(j&(-j)) )<<3)*(2+(j&1?-1:1)) );
	}
	return 0;
}
