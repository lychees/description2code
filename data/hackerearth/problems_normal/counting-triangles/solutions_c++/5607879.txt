#include<bits/stdc++.h>
using namespace std;
#define fore(i,a,b) for(i=a;i<=b;i++)
#define forl(i,a,b) for(i=a;i<b;i++)
#define si(d) scanf("%d",&d)
//#define ss(s) scanf("%s",s)
#define sll(d) scanf("%lld",&d)
#define pin(d) printf("%d\n",d)
#define pln(d) printf("%lld\n",d)
#define ll long long
#define ull unsigned long long
int main()
{
	int n;
	cin >> n;
	map<pair < ll , pair < ll , ll> > ,int > m;
		set<pair < ll , pair < ll , ll> > > v;
	
	for ( int i = 0 ; i < n ; i++)
	{
			ll a[4]; 
		cin >> a[0] >> a[1] >> a[2];
		sort(a,a+3);
		m[make_pair(a[0],make_pair(a[1],a[2]))] ++;
			v.insert(make_pair(a[0],make_pair(a[1],a[2]))) ;
	}
	int cnt = 0;
	set<pair < ll , pair < ll , ll> > >:: iterator it;
		for ( it = v.begin();it!=v.end();it++)
		{
			ll i = (*it).first;
			ll j = ( *it).second.first;
			ll k = (*it).second.second;
			if ( m[make_pair(i,make_pair(j,k))] == 1)
			cnt++;
		}
		cout << cnt << endl;
	return 0;
}
