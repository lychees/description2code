int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		int n,ans=0,T,cnt=0;
		cin >> n;
		vector<ii> st;
		rep(i,n)
		{
			cin >> T;
			st.pb(mp(T,1));
		}
		rep(i,n)
		{
			cin >> T;
			st.pb(mp(st[i].first + T,0));
		}
		sort(all(st));
		rep(i,st.size())
		{
			if(st[i].second==1) cnt++;
			else cnt--;
			ans = max(ans,cnt);
		}
		cout << ans<<"\n";

	}
}
