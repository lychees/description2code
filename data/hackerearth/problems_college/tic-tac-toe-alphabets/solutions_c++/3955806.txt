#include<bits/stdc++.h>
using namespace std;
int main()
{
	string s;
	int t;
	cin>>t;
	while(t--)
	{
		const char *sree;
		cin>>s;
		sree=s.c_str();
		char g[1000];
		strcpy(g,sree);
		int l=s.length();
		sort(g,g+l);
		cout<<g<<endl;
	}
}
