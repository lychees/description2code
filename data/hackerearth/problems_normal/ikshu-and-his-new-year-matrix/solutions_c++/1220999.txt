#include<stdio.h>
using namespace std;
long bins(long pr[],long w,long key)
{
    int  l=0;
    int mid;
    int u=w;
    while(l<=u)
    {
        mid=(l+u)/2;
        if(pr[mid]==key)
            return 0;
        if(pr[mid]<key)
            l=mid+1;
        else
            u=mid-1;
    }
    if(pr[l]>key)
        return pr[l]-key;
    else
        return pr[l+1]-key;

}
int main()
{
long p[10100],i,j;
p[0]=p[1]=-1;
long pr[1250];
long w=0;
for(i=2;i<=10010;i++)
    {
    	if(p[i]==0)
    	{
    		j=i;
    		pr[w++]=i;
    		while (j <=10010)
            		{
                		p[j]=i;
                		j = j + i;
            		}
    	}
    }
    long n;
    long k,tem,tot;
    long al[1010][1010];

    scanf("%ld%ld",&n,&k);
    for(i=0;i<n;i++)
    {
    	for(j=0;j<n;j++)
    	{
    		scanf("%ld",&tem);
    		int y=0,ur=0;
    		tem=bins(pr,w,tem);

    		if((tem)%k==0)
			{
    				al[i][j]=(tem)/k;
			}
            else
    			al[i][j]=-1;
    	}
    }
    long minh =100000;
    long x,y,f=0;
    for(i=1;i<n-1;i++)
    for(j=1;j<n-1;j++)
    if(al[i][j]!=-1&&al[i-1][j-1]!=-1&&al[i+1][j+1]!=-1&&al[i-1][j+1]!=-1&&al[i+1][j-1]!=-1)
    {tot=al[i][j]+al[i-1][j+1]+al[i+1][j-1]+al[i-1][j-1]+al[i+1][j+1];
    if(tot<minh)
    {
    	minh=tot;
    	x=i;
    	y=j;
    	f=1;
    }
    }
    if(f==1)
    {

    printf("yes\n");
    printf("%ld\n",minh);
    printf("%ld %ld",x+1,y+1);
    }
    else
    printf("no");

return 0;
}
