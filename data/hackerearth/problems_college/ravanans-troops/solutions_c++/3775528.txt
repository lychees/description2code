#include <iostream>
#include <cmath>
using namespace std;

#define N 600000	
int arr[N+1];

int main()
{	
	int i, x, min = 0;
	arr[0] = 0;
	for (i = 1; i*i <= N; ++i) {
		arr[i*i] = 1;
		//cout << i*i << " : "<< arr[i*i] << endl;
	}
	
	for (i = 2; i <= N; i++) {
		if (arr[i] == 1)
			continue;

		min = 1 + arr[i - 1];    // minimum of a non square
		
		for ( x = 1; i >= x*x; ++x) {
			
			if (min == 2)
				break;

			arr[i] = 1 + arr[i - x*x];
			if (arr[i] < min)
				min = arr[i];
		}
		arr[i] = min;		
		//cout << i << " : "<< arr[i] << endl;
	}
	
	cin >> i;
	while (i--) {
		cin >> x;
		cout << arr[x] << endl;		
	}
    
    return 0;
}
