#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		char a[32];
		
		cin>>a;
		int l=strlen(a);
		long long b,c=0;
		c=(a[l-1]-48);
		
		for(int i=l-2;i>=0;i-=2)
		{
			switch(a[i])
			{
				case '+':
				c=c+(a[i-1]-48);
				break;
				case '-':
				c=c-(a[i-1]-48);
				break;
				case '*':
				c=c*(a[i-1]-48);
				break;
				
			}
		}
		cout<<c<<endl;
	}
	return 0;
}
