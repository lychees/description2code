#include <iostream>
#include <string>
using namespace std;

int main()
{
    string s;
    int n=0, k, i, j, l, m, check;
    cin>>n>>k;
    m=n;
    int ascii[n][26];
    for(i=0;i<n;i++)
    	for(j=0;j<26;j++)
    		ascii[i][j]=0;
    while(n--){
    	cin>>s;
    	for(l=0;l<s.size();l++)
    		ascii[n][(int)s[l]-97]=1;
    }
    l=0;
    for(i=0;i<26;i++){
    	check=1;
    	for(j=0;j<m;j++){
    		if(ascii[j][i]==0)
    			check=0;
    	}
    	if(check==1)
    		l++;
    }
    if(l>=k)
    	cout<<":-)";
   	else
    	cout<<":-(";
    return 0;
}
