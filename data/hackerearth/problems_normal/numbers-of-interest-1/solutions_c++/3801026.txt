#include <iostream>
using namespace std;

int main()
{
    int t, i, j, k, x, y, n;
    cin>>t;
    long long answer[t];
    
    for (i=0 ; i<t ; i++)
    {
    	cin>>x;
    	cin>>y;
    	cin>>n;
    	
    	long long series[n];
    	
    	for (j=0 ; j<n && j<x  ; j++)
    		series[j] = y;
    		
    	if (x < n)
    	{
    		for (j=x ; j<n ; j++)
    		{
    			series[j] = 0;
    			for (k=j-x ; k<j ; k++)
    				series[j] += series[k];
    		}
    	}
    	
    	answer[i] = series[n-1];
    }
    for (i=0 ; i<t ; i++)
    	cout<<answer[i]<<endl;
    
    return 0;
}
