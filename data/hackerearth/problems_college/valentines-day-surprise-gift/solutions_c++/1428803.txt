LL BigMod(LL a,LL b)
{
      if(!b) return 1;
      LL x = BigMod(a,b/2);
      x = (x*x)%MOD;
      if(b&1) x = (x*a)%MOD;
      return x;
}

LL getVal(int n)
{
      int limit = 5*n + 1;
      LL ret = 0;
      FOR(x,1,limit)
      {
            if(gcd(x,n) < n) continue;
            FOR(y,1,limit)
            {
                  if(gcd(x,y) != n) continue;
                  ret = max(ret,BigMod(x,y));
            }
      }
      return ret;
}
