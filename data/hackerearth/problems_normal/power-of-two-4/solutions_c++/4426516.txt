#include <bits/stdc++.h>
using namespace std;

long long A[201];

bool power_of_two(long long x)
{
	for(long long i=0;i<=64;i++)
	{
		if(x==(1<<i))
			return true;
	}
	return false;
}

int main()
{
	int T;
	cin >> T;
	while (T--) {
		int N;
		cin >> N;
		for (int i = 0; i < N; i++)
			cin >> A[i];
		bool ans = false;
		for (int i = 0; i < 64; i++)
		{
			long long x = -1;
			for (int j = 0; j < N; j++)
			{
				if (A[j] & (1 << i))
				{
					x &= A[j];
				}
			}
			if (x == (1 << i))
			{
				ans = true;
				break;
			}
		}
		if (ans)
			cout << "YES\n";
		else
			cout << "NO\n";
	}
	return 0;
}
