noOfBooks = int(raw_input())

bookMap = {

}

for n in range(1, noOfBooks + 1):
    name = raw_input()
    book_index = name[0]
    if (bookMap.has_key(book_index) == False):
        bookMap[book_index] = 1
    else:
        bookMap[book_index] += 1


keys = bookMap.keys()
c = 0
for k in keys:
    c += bookMap[k] / 10
    if (bookMap[k] % 10 != 0):
        c += 1

print c
