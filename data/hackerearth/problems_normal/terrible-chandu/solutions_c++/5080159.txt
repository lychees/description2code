#include <iostream>
using namespace std;

void reverseString(string str);

int main()
{
	int num;
	string str;
    cin>>num;
    
    
    for(int i = 0; i < num; i++){
    	cin>>str;
    	reverseString(str);
    }
    
}

void reverseString(string str){
	int len = 0;
	
	for(int i = 0 ; str[i] != '\0'; ++i)                
       len++;
    for(int i = len-1; i >= 0; i--)   
      	cout<<str[i];
    	
   	cout<<endl;
}
