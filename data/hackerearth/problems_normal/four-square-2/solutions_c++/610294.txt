#include <bits/stdc++.h>
using namespace std;

int main()
{
	long long int i,j,t,m,n,s;
	scanf("%lld",&t);
	while(t--){
		scanf("%lld",&n);
		s=0;
			for(i=1;i<sqrt(n);i++){
				if(n%i==0){
				    if(i%4!=0){
				    	s=s+i;
				    }
				    if((n/i)%4!=0){
				    	s=s+(n/i);
				    }
				}
			}
			if(i*i==n&&i%4!=0){
				s=s+i;
			}
			s=s*8;
			printf("%lld\n",s);
		
	}
    //cout << "Hello World!" << endl;
    return 0;
}
