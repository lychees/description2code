#include<bits/stdc++.h>
using namespace std;
int main(){
int n;cin>>n;
int a[n+1],dp[n+1];
fill(dp,dp+n+1,0);
for(int i=1;i<=n;i++)
{
    cin>>a[i];
}
dp[1]=1;
for(int i=2;i<=n;i++){
    for(int j=1;j<i;j++){
        if(1ll*a[j]*a[i]<0 && abs(a[j])<abs(a[i]) && dp[i]<dp[j]+1) dp[i]=dp[j]+1;
    }
    if(dp[i]==0) dp[i]=1;
}
int max=0;
for(int i=1;i<=n;i++){
    if(dp[i]>max) max=dp[i];
}
cout<<max;
}
