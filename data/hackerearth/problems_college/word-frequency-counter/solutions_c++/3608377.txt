#include<bits/stdc++.h>

#define max 100005
using namespace std;

map<string,long int>mymap;
map<string,long int>chk;

int main()
{
    mymap.clear();

    int u,v,tag=0,temp;
    string str;
    int arr[100000];

    getline(cin,str);
    stringstream st(str);
    stringstream stt(str);
    string n,s="";

    while(st>>n)
    {
        s="";
        for(int i=0;i<n.length();i++)
             s+=toupper(n[i]);
        if(!mymap[s])
        {
            mymap[s]=++tag;
        }
        temp=mymap[s];
        arr[temp]++;
    }

    while(stt>>n)
    {
        s="";
        for(int i=0;i<n.length();i++)
             s+=toupper(n[i]);
        if(!chk[s])
        {
            chk[s]=1;
            cout<<s<<" "<<arr[mymap[s]]<<endl;
        }
    }


    return 0;
}
