#include <bits/stdc++.h>
using namespace std;
long long int a[10000001], i, t, m, n, mm;
int main()
{
    cin>>t;
    while(t--) {
            cin>>n;
            m = mm = 0;
            for(i = 0 ; i < n; i++)
            cin>>a[i];
           
            for(i = 0 ; i < n-1; i++) {
            if(a[i] >= a[i+1])
            m++;
            }
     
            cout<<m<<"\n";

    }
    return 0;
}
