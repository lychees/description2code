int a[100009];
int b[100009];

int main()
{
    int n,q;
    bool fl=true;
    LL tot=0;
    map<int, int> m;
    sd(n),sd(q);
    for(int i=0; i<n; i++){
        sd(a[i]), sd(b[i]);
        if(a[i]>b[i])swap(a[i], b[i]);
        tot += a[i];
        m[b[i]-a[i]]++;
    }
    for(auto &i: m){
        if(i.S%2)fl=false;
        tot += (LL)i.F*(i.S/2);
    }
    while(q--){
        LL x;
        sl(x);
        if(x==tot and fl)printf("YES\n");
        else printf("NO\n");
    }
    return 0;
}
