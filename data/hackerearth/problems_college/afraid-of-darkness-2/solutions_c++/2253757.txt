int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
    {
    	int n;
    	scanf("%d",&n);
    	ll arr[n];
    	for(int i=0;i<n;i++)
    	 scanf("%lld",&arr[i]);
    	if(n==0){cout<<"0\n";continue;} 
    	int cnt=0;
    	bool flag=false;
        while(cnt<n)
        {
        	sort(arr,arr+n,cmp);//sort descending order
        	for(int i=0;i<=cnt;i++)
        	{
        		arr[i]--;
        		if(arr[i]<0) 
        		{
        		  flag=true;
        		  break;
        		} 
            }
            if(flag) break;
            cnt++;
         }
         printf("%lld\n",cnt);
       }
}
