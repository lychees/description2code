#include <iostream>
#include<algorithm>

using namespace std;

int main()
{ int t;
cin>>t;
while(t--)
 {
 	
 	
 	int n;
 	cin>>n;
 	int arr[n+1];
 	for(int i=0;i<n;i++)
 	   cin>>arr[i];
 	     sort(arr,arr+n);
 	     
 	        int count=0;
 	       for(int i=0;i<=n-1;i++)
 	          {  if(i==n-1)
 	              {
 	              count++;
 	               break;
 	              }
 	          	
 	          	if(arr[i]+1!=arr[i+1])
 	          	     count++;
 	          	     else if(arr[i]+1==arr[i+1])
 	          	         {count++;
 	          	            i++;
 	          	         }
 	          	     
 	          	
 	          }
 	          
 	          cout<<count<<endl;
 }



}
