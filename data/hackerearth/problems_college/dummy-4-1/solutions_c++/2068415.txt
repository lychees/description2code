#include <iostream>
#include<vector>
#include<queue>
#include<stack>
using namespace std;
long long int ways=0,n;
long long int mat[200][200];
bool visited[200][200];
void find(long long int i,long long int j)
{  
	
	if(i<0||j<0||i>n-1||j>n-1) return ;
    if(mat[i][j]==1)  return ;
    if(visited[i][j]==true)  return ;
    visited[i][j]=true;
    if(i==n-1&&j==n-1) {ways++;}
    

find(i-1,j);
find(i+1,j);
find(i,j+1);
find(i,j-1);
visited[i][j]=false;
return ;
}
int main()
{
    long long int t,i,j;
    t=1;//cin>>t;
    while(t--)
    {   ways=0;
    	cin>>n;
        for( i=0;i<n;i++)
    	  for( j=0;j<n;j++)
    	     { cin>>mat[i][j];
    	       visited[i][j]=false;
    	     }
    	 find(0,0);
           cout<<ways;//<<endl;	     
    }
    return 0;
}
