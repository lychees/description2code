#include<stdio.h>
#include<limits.h>
#include<string.h>
#define INF 10000000
int main()
{
    long long  arr[669][669]={{}};
    int q,d,i,j,n;
    scanf("%d",&q);
    n=q;
    for(int i=0;i<=600;i++)
            for(int j=0;j<=600;j++)
        		arr[i][j]=INF;
    while(q--)
    {
        int x,y,w;
        scanf("%d%d%lld",&x,&y,&w);
        arr[x][y]=w;
        arr[y][x]=w;
    }
    int s;
    scanf("%d",&s);
    for(int k=0;k<=600;k++)
        for(int i=0;i<=600;i++)
            for(int j=0;j<=600;j++)
                if(arr[i][j]>arr[i][k]+arr[k][j])
                    arr[i][j]=arr[i][k]+arr[k][j];
    scanf("%d",&q);
    while(q--)
    {
        scanf("%d",&i);
        if(s==i)
        	puts("0");
        else if(arr[s][i]>=INF)
            printf("NO PATH\n");
        else
            printf("%lld\n",arr[s][i]);
    }
    return 0;
}
