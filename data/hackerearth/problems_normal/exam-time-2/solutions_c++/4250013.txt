#include<bits/stdc++.h>
using namespace std;
#define LL long long
#define S 1000005
#define W while
#define M 1000000007
#define str string
#define pi 3.14159265
LL mini=INT_MAX;
LL maxi=INT_MIN;
LL prime[S],seive_size=0;
void seive()
{
	for(int i=0;i<S;i++)
		prime[i]=1;
	prime[0]=0;
	prime[1]=0;
	for(LL i=0;i<S;i++)
	{
		if(prime[i]==0)
			continue;
		prime[i]=1;
		for(LL j=i*i;j<S;j=j+i)
			prime[j]=0;
	}
	int k=0;
	for (int i=0;i<S;i++)
	{
		if(prime[i]==1)
		{
			prime[k++]=i;
		}
	}
	seive_size=k;
} 
long long int power(long long int a,long long int b)
{
	if(b==0)
	return 1;
	long long int temp=power(a,b/2);
	if(b%2)
	return a*temp*temp;
	else
	return temp*temp;
}

int main()
{
	seive();
//	for(int i=0;i<20;i++)
//		cout<<prime[i]<<endl;
	int t;
	cin>>t;
	W(t--)
	{
		int n;
		cin>>n;
		LL ans=1;
		for(int i=0;i<seive_size&&n;i++)
		{
			LL count=0;
			W(n%prime[i]==0)
			{
				count++;
				n/=prime[i];
			}
			if(count)
			{
				ans=ans*power(prime[i],count-1)*(prime[i]-1);  //p^k(1-1/p)
			}
		}
		cout<<ans<<endl;
	}
	return 0;
}
