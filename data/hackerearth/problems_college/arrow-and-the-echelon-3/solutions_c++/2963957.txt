#include <iostream>
using namespace std;
#include<cstdio>
typedef long long int lli;
int main()
{
	lli n;
	scanf("%lld",&n);
	while(n--){
		lli x;
		char flag=0;
		scanf("%lld",&x);
		lli k=x/3;
		for(lli i=k;i>=0;i--){
			lli y=x-i*3;
			if(y%5==0){
				for(lli j=0;j<i;j++){
					printf("555");
				}
				for(lli j=0;j<y/5;j++){
					printf("333");
					printf("33");
				}
				flag=1;
				break;
			}
		}
		if(flag==0)
			printf("-1");
		printf("\n");
	}
   // cout << "Hello World!" << endl;
    return 0;
}
