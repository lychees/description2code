import math
a=int(raw_input())
for _ in xrange(a):
    #b=raw_input()
    l1=[]
    l1=raw_input().split()
    l1=list(map(float, l1))
    s=(l1[0]+l1[1]+l1[2])/2
    area=math.sqrt(s*(s-l1[0])*(s-l1[1])*(s-l1[2]))
    r=(l1[0]*l1[1]*l1[2]/(4*area))
    ac=(3.1415*r*r)
    print("%.4f" % round (ac,4))
