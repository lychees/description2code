#include<iostream>
#include<cstdio>
using namespace std;
long long sum,a[101],p[101],tmp;int c[101];
int main()
{
int n,i,j,t;
scanf("%d",&n);
for(i=0;i<n;i++)
scanf("%lld%d",&a[i],&c[i]);
scanf("%d",&t);
for(i=0;i<t;i++)
scanf("%lld",&p[i]);
for(i=t-1;i>0;i--)
p[i]=p[i]-p[i-1];
for(i=0;i<n;i++)
for(j=i+1;j<n;j++)
if(c[i]>c[j])
{
tmp=a[i];
a[i]=a[j];
a[j]=tmp;
tmp=c[i];
c[i]=c[j];
c[j]=tmp;
}
//for(i=0;i<n;i++)
//printf("%lld %d\n",a[i],c[i]);
j=0;
sum=0;
for(i=0;i<n&&j<t;)
{
if(a[i]>p[j])
{
sum+=p[j]*c[i]*(j+1);
a[i]=a[i]-p[j];
j++;
}
else 
{
sum+=(a[i]*c[i]*(j+1));
p[j]=p[j]-a[i];
i++;
}
}
for(i;i<n;i++)
sum+=(a[i]*c[i]*(j+1));
printf("%lld\n",sum);
return 0;
}
