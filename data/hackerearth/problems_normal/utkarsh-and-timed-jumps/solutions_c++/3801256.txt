#include <bits/stdc++.h>

using namespace std;

typedef long long ll ;

int main()
{
//    freopen("in.txt", "r", stdin) ;
    ios_base ::sync_with_stdio(false) ;
    cin.tie(false), cout.tie(false) ;
    ll m, n, tst, k, l ;
    cin >> tst ;
    while( tst-- )
    {
        cin >> n ;
        m = n ;
        n *= 2 ;
        k = 1 - ( 4 * ( - n ) ) ;
        k = sqrt(k) ;
        l =  -1 + k ;
        l /= 2 ;
        k = ( l * ( l + 1ll ) ) / 2ll ;
        if( k < m )
            l++ ;
        cout << l << endl ;
    }
    return 0 ;
}
