#include <iostream>
using namespace std;

int funSearch(int i, string S)
{
	bool flg;
	for(int j=0;j<i;j++)
	{
		flg=false;
		if(S[j]==S[i])
			break;
		flg=true;
	}
	if(flg)
		cout << S[i];
}

int main()
{
	int T;
	string S;
	for(cin>>T;T;T--)
	{
		cin >> S;
		cout << S[0];
		for(int i=1;i<S.size();i++)
		{
			funSearch(i,S);
		}
		cout << endl;
	}
}
