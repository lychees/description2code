#include <bits/stdc++.h>
using namespace std;
vector<int>a[1000];
bool visited[1000];
int dfsutil(int u){
	visited[u]=true;
	int ans=1;
	for(int i=0;i<a[u].size();i++){
		int v=a[u][i];
		if(!visited[v]){
			ans=ans+dfsutil(v);
		}
	}
	return ans;
}
void dfs(int n){
	int ans=INT_MAX;
	for(int i=0;i<n;i++){
		memset(visited,false,sizeof(visited));
		ans=min(ans,dfsutil(i));
	}
	cout<<ans<<endl;
}
int main() {
	int n,d;
	cin>>n>>d;
	while(d--){
		int u,v;
		cin>>u>>v;
		a[u-1].push_back(v-1);
	}
	dfs(n);
	return 0;
}
