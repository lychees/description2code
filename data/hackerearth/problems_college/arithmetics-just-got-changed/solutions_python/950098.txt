t = input()

def rev1(n, m):
	ns = str(n)
	ns = int(ns[::-1])
	ms = str(m)
	ms = int(ms[::-1])
	return ns + ms

def rev(n):
	while(n%10 == 0):
		n /= 10
	return int(str(n)[::-1])

for i in xrange(t):
	arr = map(int, list(raw_input().strip().split(' ')))
	a = arr[0]
	b = arr[1]
	print rev(rev1(a,b))
