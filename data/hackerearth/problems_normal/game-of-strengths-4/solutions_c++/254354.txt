#include <iostream>
#include <stdio.h>
#include <vector>
#include <algorithm>
using namespace std;
#define MOD 1000000007

int main()
{
	int tc;
	scanf("%d",&tc);
	while(tc--){
		int n;
		scanf("%d",&n);
		vector<long long int> arr;
		int i=0;
		int num;
		while(i<n){
			scanf("%d",&num);
			arr.push_back(num);
			i++;
		}
		sort(arr.begin(), arr.end());
		long long int maxval =arr[n-1];
		long long int ans;
		long long pos=0, neg=0;
		for(i=0;i<n;i++){
			pos = (pos+arr[i]*i)%MOD;
			neg = (neg+arr[i]*(n-i-1))%MOD;
		}
		if(pos>neg){
			printf("%lld\n",((pos-neg)*maxval)%MOD);
		}else{
			printf("%lld\n",((MOD+pos-neg)*maxval)%MOD);
		}
	}
}
