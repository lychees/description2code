#include <iostream>
#include <string.h>

using namespace std;

int main()
{
	long long int n;
	cin>>n;
	while(n--){
		char s[100009],s1[100009];
		cin>>s>>s1;
		long long int l=strlen(s),l1=strlen(s1),a[100009]={0};
			int fl=0;
		for(long long int i=0;i<l;i++){
			a[s[i]-'a']++;
		}
		for(long long int i=0;i<l1;i++){
			if(a[s1[i]-'a']>0){
				fl=1;break;
			}
		}
		
		if(fl) cout<<"YES"<<endl;
		else cout<<"NO"<<endl;
	}
    return 0;
}
