#include <iostream>
#include<string>
#include<bitset>
using namespace std;

int main()
{
    int t;
    long int n,count;
    cin>>t;
    while(t--){
    	cin>>n;
    	count=0;
    	string s=bitset<64>(n).to_string();
    	//cout<<s<<"\n";
    	for(int i=0;i<s.length();i++){
    		if(s.at(i)=='1')
    		   count++;
    	}
    	if(count%2==0)
    	   cout<<"even\n";
    	else
    	   cout<<"odd\n";
    }
    return 0;
}
