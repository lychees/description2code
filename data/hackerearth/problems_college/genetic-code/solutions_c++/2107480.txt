#include <iostream>
using namespace std;

int main()
{
    string s;
    cin >> s;
    
    int sum=1,ans=0;
    for(int i=0;i<s.length()-1;i++)
    {
    	if(s.at(i)==s.at(i+1))
    	{
    		sum++;
    	}
    	else
    	{
    		if(sum%2==0)
    		{
    			ans++;
    		}
    		sum=1;
    	}
    }
    
    if(sum%2==0)
    {
    	ans++;
   	}
    cout << ans ;
}
