T = int(raw_input())
for i in range(0, T):
    M, P = map(int, raw_input().split())
    m = '{0:064b}'.format(M)
    p = '{0:064b}'.format(P)
    count = 0
    for i in range(0, 64):
        if (m[i] != p[i]):
            count = count+1
    print count
