alpha = "abcdefghijklmnopqrstuvwxyz"

test = input()
for i in xrange(test):
    word = raw_input()
    d = {}
    for i in word:
        if i in d:
            d[i] += 1
        else:
            d[i] = 1

    # print d
    final = ''.join(sorted(d.keys()))
    # print final
    final1 = len(set(d.values()))

    if final == alpha and final1 == 1:
        print "Yes"
    else:
        print "No"
