#include <iostream>
using namespace std;
bool prim(int num)
{
	for(int i=2;i*i<=num;i++)
	if(num%i==0)
	return false;
	return true;
}
int main()
{
   bool prime[1001];
   int i;
   for( i=0;i<1001;i++)
   prime[i]=false;
   prime[2]=prime[3]=true;
   for(i=4;i<1001;i++)
   if(prim(i))
   prime[i]=true;
   int tc,n,num,c;
   cin>>tc;
   while(tc--)
   {
   	c=0;
   	cin>>n;
   	while(n--)
   	{
   		cin>>num;
   		if(prime[num])
   		c++;
   	}
   	cout<<c<<endl;
   }
    return 0;
}
