#include <iostream>
#include <map>
using namespace std;

int main()
{
    int t;cin>>t;
    while(t--){
    	int n;cin>>n;
    	int a[n],i,ans=0;
    	map<int,int> ma;
    	for(i=0;i<n;i++)	cin>>a[i];
    	for(i=0;i<n;i++){
    		if(ma[a[i]%10])
    			ans++;
    		ma[a[i]%10]=i+1;
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
