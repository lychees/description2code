#include <iostream>
using namespace std;
long long int fact( long long  x )
{
      long long  i, f = 1;
      for( i = x; i > 0; i-- )
      {
            f = f * i;
       }
       return (f);
}
long long NCR(long long int n,long long int r)
{
	long long int c=1 ;
	if(r>n/2) r=n-r;
	for(int i=0;i<r;i++)
	c*=(n--);
	for(int i=1;i<=r;i++)
	c/=i;
	return c;
}
int main() {
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long int a,b;
		scanf("%lld %lld",&a,&b);
		//cout<<NCR(5,1)<<" "<<fact(7)<<" ";
		long long int ans;
		ans=NCR(7,a)*NCR(5,b)*fact(a+b);
		cout<<ans<<endl;
		
	}
	return 0;
}
