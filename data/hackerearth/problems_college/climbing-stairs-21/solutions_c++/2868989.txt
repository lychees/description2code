#include <iostream>
using namespace std;

long long int answer[1000];

int main()
{
    answer[0] = 0;
    answer[1] = 1;
    answer[2] = 2;

    for(int i=3;i<=1000;i++)
    {
        answer[i] = answer[i-1]+answer[i-2];
    }

    int test;
    cin>>test;

    while(test)
    {
        test--;
        int num;
        cin>>num;

        cout<<answer[num]<<"\n";
    }

    return 0;
}
