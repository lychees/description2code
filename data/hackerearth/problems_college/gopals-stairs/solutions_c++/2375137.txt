#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
int main()
{
    int t;
    cin>>t;
    long long int a[100001];
    a[0]=1;
    a[1]=1;
    a[2]=2;
    a[3]=4;
    for(int i=4;i<100001;i++)
    {
    	a[i]=(a[i-1]+a[i-2]+a[i-3])%mod;
    }
    while(t--)
    {
    	int n;
    	cin>>n;
    	cout<<a[n]<<endl;
    	
    }
}
