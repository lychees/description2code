#include <iostream>
#include <math.h>
using namespace std;
//int alpha[26];
int mat[10000][20];
int ans[20];
int main() {
	ios_base::sync_with_stdio(0);
	int t,n,k;
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		string s;
		unsigned long long sum[k];
		for(int i=0; i<n; i++)
		{
			cin>>s;
			for(int j=0; j<k; j++)mat[i][j] = s[j]-'a';
		}
		//for(int i=0; i<26; i++)alpha[i]=0;
		//int len = s.length();
		for(int j=0; j<k; j++)
		{	
			ans[j]=0;
			//int alpha[26]={0};
			int mincost = 1500111220;
			int cost=0;
			for(int k=0; k<26; k++)
			{
				cost=0;
				for(int i=0; i<n; i++)
				{
					cost += abs(mat[i][j]-k);
				}
				if(cost<mincost) 
				{
					mincost = cost;
					ans[j]=k+1;
				}
			}
			
		}
		for(int i=0; i<k; i++)cout<<(char)(ans[i]-1+'a');
		cout<<"\n";
	}
	// your code goes here
	return 0;
}
