#include <cstdio>
 
using namespace std;

long int a[1000000], b[1000001];

int main()
{
	int n;
	
	scanf("%d", &n);
	
	int i;
	
	for (i = 0; i < n; ++i) {
		scanf("%ld", &a[i]);
		b[i] = 0;
	}
	
	b[i] = 0;
	
	int m, j, k;
	
	scanf("%d", &m);
	
	while (m--) {
		scanf("%d%d%d", &i, &j, &k);
		b[i - 1] += k;
		b[j] -= k;
	}
	
	a[0] += b[0];
	printf("%ld\n", a[0]);
	
	for (i = 1; i < n; ++i) {
		b[i] += b[i - 1];
		a[i] += b[i];
		printf("%ld\n", a[i]);
	}
	
	return 0;
}
