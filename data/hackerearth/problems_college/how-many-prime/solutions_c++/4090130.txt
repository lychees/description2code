#include <iostream>
using namespace std;
#include<set>
int main()
{
   set<int>s;
    int i,j,t,a[100001];
   for(i=0;i<100001;i++)
   {
   	a[i]=1;
   }
   a[0]=0,a[1]=0;
   for(i=2;i*i<=100000;i++)
   {
   	if(a[i]==1)
   	{
   		for(j=i*i;j<=100000;j+=i)
   		{
   			a[j]=0;
   		}
   	}
   }
    cin>>t;
    int n;
    for(i=0;i<t;i++)
    {
    	cin>>n;
    	int v[100001]={0};
    	int f=0,r1,r2;
    	for(j=2;j<=n;j++)
    	{
    	if(a[j]==1 && a[n-j]==1 && (v[j]==0) && j!=n-j)
    	{
    		f=1;
    		r1=j;
    		r2=n-j;
    		v[j]=1;
    		v[n-j]=1;
    	}
    	}
    	if(f==1)
    	cout<<r1<<" "<<r2<<"\n";
    	else
    	cout<<"-1"<<" "<<"-1"<<"\n";
    	
    }
    return 0;
}
