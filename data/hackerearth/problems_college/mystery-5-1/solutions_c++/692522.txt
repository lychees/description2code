#include <iostream>
#include<string>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while( t-- )
	{
		int alpha[26] = {0};
		string input;
		cin>>input;
		
		for(int i = 0 ; i < input.length(); i++)
			alpha[input[i]-'a']++;
			
		int maxi = 0;
		for(int i = 0 ; i < 26 ; i++)
			maxi = max(maxi,alpha[i]);
			
		cout<<maxi<<endl;
		
	}
    
    return 0;
}
