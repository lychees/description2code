#include<iostream>
#include<vector>
using namespace std;

vector<int> v[1001];
bool visit[1001]={false};

int dfs(int x)
{
	visit[x]=true;
	int c=1;
	for(int i=0;i<v[x].size();i++)
	{
		if(visit[v[x][i]]==false)
		{
			int y=dfs(v[x][i]);
			c+=y;
		}
	}
	return c;
}

int main()
{
	int n,m;
	cin>>n>>m;
	while(m--)
	{
		int x,y;
		cin>>x>>y;
		v[x].push_back(y);
	}
	int min=dfs(1);
	for(int i=2;i<=n;i++)
	{
		for(int j=1;j<=n;j++)
		    visit[j]=false;
		int z=dfs(i);
		if(min>z)
		    min=z;
	}
	cout<<min;
	return 0;
}
