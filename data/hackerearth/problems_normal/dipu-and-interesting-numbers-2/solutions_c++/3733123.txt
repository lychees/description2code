#include <bits/stdc++.h>
using namespace std;
int find (long long n) {
	int low,high,mid;
	long long v1,v2;
	low = 0;
	high = 1e9;
	if (n == 0) {
		return 0;
	}
	int ans=0;
	while (low <= high) {
		mid = (low + high) / 2;
		v1 = mid * 1LL * mid;
		v2 = (mid + 1) * 1LL * (mid + 1);
		if (v1 <= n) {
			ans=mid;
				low = mid + 1;
		} else {
			high = mid - 1;
		}
	}
	return ans;
}
int main()
{
	int t;
	scanf("%d",&t);
	while (t --) {
		long long l,r;
		int ans;
		scanf("%lld%lld",&l,&r);
		ans = find(r) - find(l - 1);
		printf("%d\n",ans);
	}
	return 0;
}
