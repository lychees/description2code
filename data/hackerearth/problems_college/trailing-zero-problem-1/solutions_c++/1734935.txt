#include <iostream>
#include<math.h>
#define PI 3.14159265359
using namespace std;

int main()
{
	int t,z=0,digit=0,N,i;
cin>>t;
while(t--)
{
	z=0;
	digit=0;
	cin>>N;
	for(i=5;N/i>=1;i=i*5)//counring no of trailing zeroes//
	{
	z=z+N/i;                     //no of zeoes//	
	}
	//formula to caculate the no of digits in n!//
	//use stirling formula//
	digit=(floor((log(2*PI*N)/2 + N*(log(N) - 1) + 1/(12*N))/log(10)) + 1);//no of didgits in n fact
	cout<<digit-z<<"\n";
}
return 0;
}
