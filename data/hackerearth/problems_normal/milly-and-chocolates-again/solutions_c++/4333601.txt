ll search(ll arr[],ll low,ll high,ll key)
{
	ll mid;
	if(key<=arr[low])
	return low;
	if(key>arr[high])	
	return -1;
	mid=(low+high)/2;
	if(arr[mid]==key)
	return mid;
	else if(arr[mid]<key)
	{
		if(mid+1<=high && key<=arr[mid+1])
		return mid+1;
		else
		return search(arr,mid+1,high,key);
	}
	
	else if(arr[mid]>key)
	{
		if(mid-1>=low && key>arr[mid-1])
		return mid;
		else
		return search(arr,low,mid-1,key);
	}
}
