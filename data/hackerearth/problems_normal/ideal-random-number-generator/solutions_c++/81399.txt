#include<iostream>
using namespace std;

int GCD(int a, int b) {
    if( b == 0 ) return a;
    return GCD(b, a % b);
}
int main() {
    long int a, b, c;
    cin >> a >> b >> c;

    int x = 2 * a * b;
    int y = c * c;
    if( a + b <= c) {
        y = x;
    } else  {
        if (c > a) {
            y -= (c - a) * (c - a);
        }
        if(c > b) {
            y -= (c - b) * (c - b);
        }
    }
    int gcd = GCD(x, y);
    cout <<  y / gcd << "/" << x / gcd; 
}
