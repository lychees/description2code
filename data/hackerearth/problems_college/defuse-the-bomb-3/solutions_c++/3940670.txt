#include <iostream>
#include <cstring>
#include <cmath>
using namespace std;

int main() {
    
    int tc;
    cin >> tc;
    
    while (tc--) {
        char a[15];
        int b;
        
        cin >> a >> b;
        int la=strlen(a), lb=log10(b)+1;
        int n = la - lb + 1;
        
        for (int i=0; i<n; i++) {
            int c=0;
            
            for (int x=i; x<lb+i; x++) {
                c = (c * 10) + (a[x] - '0');
            }
            
            cout << b * c << endl;
        }
    }
}
