#include <iostream>
using namespace std;

int GCD(int a,int b){
	if (b == 0)
		return a;
	else
		return GCD(b,a%b);
}

int main()
{
	int t;
	cin >> t;
	int x,y;
	string s;
	while (t > 0){
		cin >> x >> y >> s;
		int i = 0;
		while (x != 0 && y != 0){
			int res = GCD(x,y);
			x = x-res;
			y = y-res;
			i++;
		}
		if (i % 2 == 0){
			if (s == "Chandu")
				cout << "Daspal" << endl;
			else
				cout << "Chandu" << endl;
		}
		else{
			if (s == "Chandu")
				cout << "Chandu" << endl;
			else
				cout << "Daspal" << endl;	
		}
		t--;
	}
    return 0;
}
