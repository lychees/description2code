#include <iostream>
#include <cstdio>
#include <utility>
#include <algorithm>
#include <cmath>

using namespace std;

int main()
{
        int test_cases;
        scanf("%d",&test_cases);
        while(test_cases > 0)
        {
                int N,i=0;
                pair<int,int> arr[10000];
                scanf("%d",&N);
                while(i<N)
                {
                        scanf("%d",&arr[i].first);
                        arr[i].second=i;
                        i = i+1;
                }
                sort(arr,arr+N);
                
                int temp,max=-1;
                for(i=0;i<N;i++)
                {
                	temp=abs(arr[i].second-i);
                	if(max<temp)
                		max=temp;
                }
                //your code here
             	printf("%d\n",max);
             	
                test_cases--;
        }
        return 0;
}
