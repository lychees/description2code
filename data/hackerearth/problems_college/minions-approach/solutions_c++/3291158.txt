#include <bits/stdc++.h>
using namespace std;

int main() 
{
	int t, n, i, j, k,tmp;
	int cnt = 0;
	cin >> t;
	while (t--) {
		cin >> n;
		tmp = cnt = 0;
		if (n == 2 || n == 1) {
			cout << "Incorrect" << endl;
		}else {
			for (i = 1; i <= n; i++) {
				k = __gcd(i, n);
				if (k == 1) {
					cnt++;
				}
			}
			//cout << "KK : " << cnt << endl;
			for (i = 1; i < cnt; i++) {
				k = __gcd(i, cnt);
				if (k != 1) {
					tmp = 1;
					break;
				}
			}
			if (tmp == 1) {
				cout << "Incorrect" << endl;
			} else {
				cout << "Correct" << endl;
			}
		}
	}
	return 0;
}
