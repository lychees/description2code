#include<iostream>
#include<cstdio>
#include<cmath>
#include<vector>
#include<algorithm>
// Include headers as needed
#define ll long long int
using namespace std;

int main()
{
    // Write your code here
    int t;
    scanf("%d",&t);
    while(t>0){
        vector<ll> a;
        ll num;
        scanf("%llu",&num);
        a.push_back(1);
        if(num!=1){
        a.push_back(num);}
        for(ll i=2;i<=sqrt(num);i++){
            if(num%i==0){
                a.push_back(i);
                if(i!=sqrt(num)){
                    a.push_back(num/i);
                }
            }
        }
        sort(a.begin(),a.end());
        vector<ll>::iterator i;
        /*for(i=a.begin();i!=a.end();i++){
            cout<<*i<<" ";
        }*/
        ll sum=0;
        if(num&1){
            for(i=a.begin();i!=a.end();i++){
                sum+=*i;
            }
            printf("%llu",8*sum);
        }
        else {
            for(i=a.begin();i!=a.end();i++){
                if((*i)&1)sum+=*i;
            }
            printf("%llu",24*sum);
        }
        printf("\n");
        t--;
    }
    
    // Return 0 to indicate normal termination
    return 0;
}
