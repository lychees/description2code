#include <iostream>
using namespace std;

int main()
{
    string s;
    cin>>s;
    long long int l = s.length();
    long long int t;
    cin>>t;
    while(t--)
    {
    	long long int a,b;
    	cin>>a>>b;
    	a--; b--;
    	if(s[a%l]==s[b%l]) cout<<"Yes\n";
    	else cout<<"No\n";
    }
    return 0;
}
