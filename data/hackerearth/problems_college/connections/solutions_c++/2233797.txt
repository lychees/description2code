#include <iostream>
using namespace std;

int main()
{
    int n,t;
    cin>>n;
    cin>>t;
    int a[t][2];
    int array[t*2];
    for(int i=0;i<t;i++)
    {
    	cin>>a[i][0]>>a[i][1];
    	array[i*2] = a[i][0];
    	array[i*2+1] = a[i][1];
    
    }
    int q;
    cin>>q;
    int b[q][2];
    
    for(int i=0;i<q;i++)
    {
    	int x=0,y=0,z=0;
    	cin>>b[i][0]>>b[i][1];
    	
    	for(int j=0;j<t*2;j++)
    	{
			if(b[i][0] == array[j])
			{
				x = 1;
			}
			j++;
    	}
    	
    	if(x==0)
    	{
    		cout<<"No Connection\n";
    		continue;
    	}
    	else
    	{
    		for(int j=1;j<t*2;j++)
    		{
    			if(b[i][1] == array[j])
				{
					y = 1;
				}
				j++;
    		}
    		
    		if(y==0)
    		{
    			cout<<"No Connection\n";
    			continue;
    		}
    		else
    		{
    			for(int j=0;j<t;j++)
    			{
    				if(b[i][0]==a[j][0] && b[i][1]==a[j][1])
    				{
    					z = 1;
    				}
    		
    			}
    			if(z==1)
    			{
    				cout<<"Direct Connection\n";
    			}
    			else
    			cout<<"Indirect Connection\n";
    			continue;
    		}
    	}
    
    	
    	
    }
    
    return 0;
}
