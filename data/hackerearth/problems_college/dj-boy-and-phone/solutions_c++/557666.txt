#include <iostream>
#define MOD 1000000009
#define max(x,y) (x>y?x:y)
#define M 10005
typedef long long int LL;
using namespace std;
LL A[10][M];
void fun(){
	for(int i=0;i<10;i++)
		A[i][0]=1;
	for(int i=1;i<M;i++){
		A[0][i]=(A[0][i-1]+A[8][i-1])%MOD;
		A[1][i]=(A[1][i-1]+A[2][i-1]+A[4][i-1])%MOD;
		A[2][i]=(A[2][i-1]+A[1][i-1]+A[3][i-1]+A[5][i-1])%MOD;
		A[3][i]=(A[3][i-1]+A[2][i-1]+A[6][i-1])%MOD;
		A[4][i]=(A[4][i-1]+A[1][i-1]+A[5][i-1]+A[7][i-1])%MOD;
		A[5][i]=(A[5][i-1]+A[2][i-1]+A[4][i-1]+A[6][i-1]+A[8][i-1])%MOD;
		A[6][i]=(A[6][i-1]+A[3][i-1]+A[5][i-1]+A[9][i-1])%MOD;
		A[7][i]=(A[7][i-1]+A[4][i-1]+A[8][i-1])%MOD;
		A[8][i]=(A[8][i-1]+A[5][i-1]+A[7][i-1]+A[9][i-1]+A[0][i-1])%MOD;
		A[9][i]=(A[9][i-1]+A[6][i-1]+A[8][i-1])%MOD;		
	}	
}
int main(){
	int t;
	fun();
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		if(n==0&&cout<<"0\n")
			continue;
		long long int s=0;
		for(int i=0;i<10;i++)
			s+=(A[i][n-1]);
		cout<<s%MOD<<endl;
	}
	return 0;
}
