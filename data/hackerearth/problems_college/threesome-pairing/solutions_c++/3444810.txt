#include <iostream>
using namespace std;

int main()
{

	long int t,n,i,j,k,l,m,o,p;
	cin>>t;
	while(t--)
	{
		cin>>n;
		p=0;
		long int a[n];
		for(i=0;i<n;i++)
		cin>>a[i];
		for(i=0;i<n;i++)
		{
			for(j=i+1;j<n;j++)
			{
				l=a[i]+a[j];
				if(l%3==0)
				p++;
			}
		}
		
		cout<<p<<"\n";
	}
}
