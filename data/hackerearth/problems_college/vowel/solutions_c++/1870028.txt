#include <iostream>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    int tc;
    cin>>tc;
    while(tc--) {
    	string str;
    	cin>>str;
    	bool flag = false;
    	for(int i=0;i<str.length();i++) {
    		if(str[i] == 'a' || str[i] == 'e' || str[i] == 'i' || str[i] == 'o' || str[i] == 'u') {
    			cout<<str[i];
    			flag = true;
    		}
    		if(str[i] == 'A' || str[i] == 'E' || str[i] == 'I' || str[i] == 'O' || str[i] == 'U') {
    			cout<<str[i];
    			flag = true;
    		}
    	}
    	if(!flag) {
    		cout<<"No";
    	}
    	cout<<endl;
    }
    return 0;
}
