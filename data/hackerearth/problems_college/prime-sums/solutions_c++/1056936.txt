#include <iostream>
#include <vector>
using namespace std;
typedef unsigned long long ull;

ull primes(int n)
{
	vector<bool> v (n + 1, true);
	v[0] = false;
	v[1] = false;
	ull sum = 0;
	for (int i = 2; i < n; ++i)
	{
		if (v[i])
		{			
			sum += i;
			for (int j = i + i; j < n; j += i)
				v[j] = false;
		}
	}	
	return sum;
}

int main()
{	
	int n;
	cin >> n;
	if (n > 2)
		cout << primes(n) << "\n";
	else
		cout << 0 << "\n";
    
    return 0;
}
