#include <bits/stdc++.h>
using namespace std;

int main() {
	long long int n,m,i,j,k,l;
	cin>>n>>m;
	long long int a[m];
	for(i=0;i<m;i++)
	cin>>a[i];
	
	long long int b[n+1];
	memset(b,0,sizeof(b));
	b[0]=1;
	
	for(i=0;i<m;i++)
	{
		for(j=a[i];j<=n;j++)
		{
			b[j]+=b[j-a[i]];
		}
		
	//	cout<<b[i]<<" ";
	}
	
	cout<<b[n];
	
}
