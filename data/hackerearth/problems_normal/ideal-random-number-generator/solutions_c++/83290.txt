#include <iostream>
#define MAX(A,B) (A>B?A:B)
#define MIN(A,B) (A<B?A:B)
using namespace std;

long long gcd(long long a, long long b) 
{ 
	if (b == 0)return a;
	else return gcd(b, a % b);
}

int main()
{
	long long A,B,C,T,U,G,AT;
	cin>>A>>B>>C;

	if(A<=C && B<=C)
	{
		T = A*B*10;
		U = (MAX((A+B-C),0) * MAX((A+B-C),0))*5;
		U = T-U;
	}
	else if(A>C && B>C)
	{
		T = A*B*10;
		U = C*C*5;
	}
	else
	{
		T = A*B*10;
		AT = C*C*5;
		U = (C - MIN(A,B))*(C - MIN(A,B))*5;
		U = AT - U;
	}

	G = gcd(U,T);
	cout<<U/G<<"/"<<T/G<<"\n";
	return 0;
}
