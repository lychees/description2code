#include <iostream>
using namespace std;

int main()
{
	int a,b;
	cin>>a>>b;
	int printed=0;
	
	for(int i=a+1;i<b;i++){
	
		if(i%5==0 && i%3==0){
			
			if(printed!=0){
				cout<<",";
			}
			cout<<i;
			printed++;
		}
		else if(i%5==0 && i%3!=0){
			continue;
		}else if(i%3==0 && i%5!=0){
			continue;
		}else{
			if(printed!=0){
				cout<<",";
			}
			cout<<i;
			printed++;
		}
	}
    return 0;
}
