#include <iostream>
using namespace std;
typedef long long int ll;

ll p(ll n){

   if (n&(n-1) == 0) 
         return 1;
    else{
    	int i=1;
    	 while(i<n)
    	    i*=2;
    	      return i;
    
    }
  
}
int main()
{
   // cout << "Hello World!" << endl;
    int nn;
    cin>>nn;
    while(nn--){
    	ll x;
    	cin>>x;
    	int n=p(x);
    	if(x==1)
    	   cout<<"1"<<endl;
    	 else{
    	 	 cout<< x-(n-x) + 1<<endl; 
    	 }
    	   
    	
    }
    
    return 0;
}
