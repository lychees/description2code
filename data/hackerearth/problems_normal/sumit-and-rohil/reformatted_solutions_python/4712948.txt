def isAnagram(str1, str2):
    chars = list(str1)
    for i in range(len(str2)):
        for j, c in enumerate(chars):
            if c == str2[i]:
                del chars[j]
                break
    if chars == []:
        return True
    else:
        return False


n = int(raw_input())

names = []
for i in range(n):
    names.append(raw_input())

groups = {}

for name in names:
    keyid = 0
    while True:
        key = name[0]+name[-1]+str(keyid)
        if key not in groups:
            groups[key] = []
            groups[key].append(name)
            break
        elif isAnagram(name, groups[key][0]):
            groups[key].append(name)
            break
        else:
            keyid += 1

print len(groups)
