#include <bits/stdc++.h>
using namespace std;

int main()
{
   int i,n,k,a[1000000],f,x,y,sum;
   cin>>n;
   cin>>k;
   for(i=0;i<n;i++){
   	cin>>a[i];
   }
   sort(a,a+n);
   f=1;
   x=0,y=n-1;
   while(x<y){
   	sum=a[x]+a[y];
   	if(sum==k){
   		f=0;
   		break;
   	}
   	else if(sum>k){
   		y--;
   	}
   	else{
   		x++;
   	}
   }
   if(f){
   	cout<<"NO";
   }
   else{
   	cout<<"YES";
   }
   
   
   

    return 0;
}
