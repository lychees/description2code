#include <iostream>
#include <algorithm>
 
using namespace std;
 
int main()
{
    int test;
    cin >> test;
    while (test--) {
        string s1,s2 ;
        cin >> s1 >> s2;
        if (s1.size() > s2.size()) 
        {
        	swap(s1, s2);
        }
        int count = s1.size();
        for (int i = 0; i < s2.size() - s1.size() + 1; i++)
        {
			int flag = 0;
			for (int j = 0; j < s1.size(); j++)
			{
				if (s1[j] != s2[j + i])
				flag++;
			}
			count = min(count, flag);
		}
		cout << count << endl;
    }
    return 0;
}
