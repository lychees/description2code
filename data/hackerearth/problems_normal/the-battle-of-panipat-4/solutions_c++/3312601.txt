#include <bits/stdc++.h>
using namespace std;
vector<int>v[100005];
#define MOD 1000000007
int visit[100005];

pair<long long int,long long int>dfs(int u)
{
	visit[u]=1;
	pair<long long int,long long int>p=make_pair(1,0);
	for(int i=0;i<v[u].size();i++)
	{
		int node=v[u][i];
		if(visit[node]==0)
		{
		pair<long long int,long long int>res=dfs(node);
		p.first=(p.first*(res.first+1))%MOD;
		p.second=(p.second+(res.first+res.second)%MOD)%MOD;
		}
		
	}
	return p;
}
int main()
{
	int n;
	cin>>n;
	for(int i=0;i<=100000;i++)
	v[i].clear();
	
	memset(visit,0,sizeof visit);
	int x,y;
	for(int i=0;i<n-1;i++)
	{
		scanf("%d%d",&x,&y);
		v[y].push_back(x);
		v[x].push_back(y);
	}
	pair<long long int,long long int >ans=dfs(0);
	cout<<(ans.first+ans.second+1)%MOD;
	return 0;
	
}
