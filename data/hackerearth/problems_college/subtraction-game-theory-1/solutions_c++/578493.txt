#include<iostream>
#include<cstdio>
#include<string>
#include<map>
using namespace std;

# define ll long long
int main()
{
    ll t;
    scanf("%lld", &t);
    while(t--) {
        ll n, k;
        scanf("%lld%lld", &n, &k);
        ll p;
        p = n % (k + 1);
        p --;
        if( p != 0) printf("Rani\n");
            else printf("Nandu\n");
    }
    return 0;
}
