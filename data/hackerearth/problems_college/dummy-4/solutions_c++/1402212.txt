#include <iostream>
#include<string>
using namespace std;

int main()
{
    int n,r,i;
    cin>>n;
    string str;
    long long int ans=0;
    while(n--)
    {
    	cin>>str;
    	long long int sum=0;
    	for(i=0;i<str.length();i++)
    	{
    		sum+=str[i]-'0';
    	}
    	ans+=sum;
    	sum=ans;
    //	cout<<ans<<endl;
    	while(ans>9)
    	{
    		long long int sum1=0;
    	while(sum>0)
    	{
    		r=sum%10;
    		sum1+=r;
    		sum=sum/10;
    	}
    	sum=sum1;
    	ans=sum1;
    	}
    	
    }
    cout<<ans<<endl;
    return 0;
}
