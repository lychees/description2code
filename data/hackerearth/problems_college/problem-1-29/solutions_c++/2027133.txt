#include <iostream>
using namespace std;

#define ll long long int

bool solveMazeUtil(ll maze[1000][1000],ll n , ll x, ll y){
	if(x>n-1 || y>n-1)
		return false;
	if(x==n-1 && y==n-1)
		return true;
	if(maze[x][y]==0)
	return false;
	else
	return (solveMazeUtil(maze,n,x+1,y) || solveMazeUtil(maze,n,x,y+1));
}




void solveMaze(long long int maze[1000][1000],long long int n){
	if(solveMazeUtil(maze,n,0,0))
		cout<<"POSSIBLE"<<endl;
	else
	    cout<<"NOT POSSIBLE"<<endl;
	
	}

int main()
{
    long long int t,sz;
    long long int maze[1000][1000],i,j;
    cin>>t;
    while(t--){
    	cin>>sz;
    	for(i=0;i<sz;i++)
    		for(j=0;j<sz;j++)
    			cin>>maze[i][j];
    	solveMaze(maze,sz);
    }
    return 0;
}
