LL x[] = { 6,2,5,5,4,5,6,3,7,6 };
LL getsum(LL a)
{
	LL tmp = 0;
	if (a == 0) tmp = 6;
	while (a)
	{
		tmp += x[a % 10];
		a /= 10;
	}
	return tmp;
}

int main()
{
#ifdef _DEBUG
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
#endif
	int t;
	scanf("%d", &t);
	set<int>primes;
	primes.insert(2);
	for (int i = 3; i < 100000; i++)
	{
		bool y = true;
		for (int j = 2; j*j < i; j++)
		{
			if (i%j == 0) y = false;
		}
		if (y) primes.insert(i);
	}
	while (t--)
	{
		int n;
		scanf("%d", &n);
		if (n == 3)
		{
			printf("%d\n", 3);
			continue;
		}
		if (primes.count(n)) printf("%d\n", 2);
		else printf("%d\n", 1);
	}
	return 0;
}
