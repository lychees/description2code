#include <iostream>
#define MOD 1000000007
using namespace std;

long long ans[1000010];

int main(){
	long long dp[2][10];
	
    int t,n;
    cin>>t;
	
	ans[1]=0;
    for(int j=0;j<10;j++){
    	dp[1][j]=1;
    	ans[1]+=dp[1][j];
    }
	
	for(int i=2;i<=1000000;i++){
		ans[i]=0;
		for(int j=0;j<10;j++){
			if((j==0) || (j==1 && i==2))
				dp[i%2][j] = dp[(i-1)%2][j+1];
			else if(j==9)
				dp[i%2][j] = dp[(i-1)%2][j-1];
			else
				dp[i%2][j] = (dp[(i-1)%2][j+1] + dp[(i-1)%2][j-1])%MOD;
			
			ans[i]= (ans[i] + dp[i%2][j])%MOD;
		}
	}
    
    while(t--){
		cin>>n;
		cout<<ans[n]<<endl;
    }
    
    return 0;
}
