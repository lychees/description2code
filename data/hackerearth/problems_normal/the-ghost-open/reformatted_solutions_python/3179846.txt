import sys

TC = int(sys.stdin.readline())


for counter in range(0, TC):
    players = int(sys.stdin.readline())
    noOfMatches = 0
    if (players == 1 or players == 0):
        print 0
    else:
        while (players != 1):
            dist = players/2
            noOfMatches += dist
            ghost = players % 2
            players = dist + ghost

        print noOfMatches
