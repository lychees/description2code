#include<cstdio>
#include<set>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		set <int> s;
		int n,k;
		scanf("%d %d",&n,&k);
		for(int i=0;i<n;i++)
		{
			int t;
			scanf("%d",&t);
			s.insert(t);
		}
		set <int> :: iterator ip,ip1=s.begin();
		int max=-1,sum=0;
		for(ip=s.begin();ip!=s.end();ip++)
		{
			sum=sum+(*ip);
			if(sum>max && sum<k)
				max=sum;
			else if(sum>k)
			{
				while(sum>k && ip1!=s.end())
				{
					sum=sum-(*ip1);
					ip1++;
				}
			}
		}
		for(ip=s.begin();ip!=s.end();ip++)
		{
			sum=k/(*ip)*(*ip);
			if(sum>max)
				max=sum;
		}		
		printf("%d\n",max);
	}
	return 0;
}
