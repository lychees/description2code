#include <bits/stdc++.h>

using namespace std;

int prime(int n)
{
	for (int i = 2; i <= n / 2; i++)
	{
		if (n % i == 0)
			return 0;
	}
	return 1;
}

	for (int i = 0; i < n; i++)
	{
		int check = prime(a[i]);
		if (check == 1)
			sel[c++] = a[i];
	}
		
	int *max = max_element(sel, sel + c);

	cout << count << endl;
	return 0;
}
