#include<bits/stdc++.h>
using namespace std;
int main()
{ int t; 
  cin>>t;
  while(t--)
  { long long int ans=0,a,b,c,x,y,n,i;
    cin>>x>>y>>n;
    
    ans=x+y;
    a=x;
    b=y;
    for(i=3;i<=n;i++)
  	{ c=max(a,b)-min(a,b);
  	   if(c==0)
  	   { ans+=((  ((n-i)/3)*2+(n-i)%3  )*b);
  	   	 break;
  	   }
  	   ans+=c;
  	  
  	   a=b;
  	   b=c;
	}
  	
  	cout<<ans<<"\n";
  }
	return 0;
}
