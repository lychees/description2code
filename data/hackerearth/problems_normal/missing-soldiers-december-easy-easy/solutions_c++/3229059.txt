#include<bits/stdc++.h>
using namespace std;
const int Max = 1e5;
const int Maxv = 1e9;
int main()
 {
 	int n,x,y,d;
 	cin>>n;
 	assert(1<=n && n<=100000);
 	pair<int, int> v[Max];
 	for(int i=0;i<n;i++) {
 		cin>>x>>y>>d;
 		assert(1<=x && x<=Maxv);
 		assert(1<=y && y<=Maxv);
 		assert(1<=d && d<=Maxv);
 		v[i]=make_pair(x,x+d);
 	}
 	sort(v,v+n);
 	int res=0;
 	int cur=1;
 	for(int i=0;i<n;i++) {
 		cur = max(cur,v[i].first);
 		if(v[i].second >= cur) {
 			res = res + v[i].second - cur + 1;
 			cur = v[i].second + 1;
 		}
 	}
 	cout<<res<<endl;
 	return 0;
 }
