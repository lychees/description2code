class helper
{
	public:
		int mindist(int current,int dest)
		{
			int a,b;
			if(dest>=current)
				a = dest-current;
			else
				a = 26 - (current - dest);
			if(current>=dest)
				b = current - dest;
			else
				b = 26 - (dest - current);
			if(a<=b)
				return a;
			else
				return -b;
		}
};

int main()

	int t,i,len,current;
	string ar;
	helper h;

		cin>>ar;
		len = ar.length();
		current = 0;
		loop(i,0,len)

			printf("%d ",h.mindist(current,ar[i]-'a'));
			current = ar[i] - 'a';
		}

	}
