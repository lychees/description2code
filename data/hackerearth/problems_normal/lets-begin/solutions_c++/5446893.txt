int prime[15000];
vector<int>p;
void sieve(int n)
{
    for(int i=2;i<=n;++i)prime[i]=1;
    for(int i=2;i<=sqrt(n);++i)
    {   if(prime[i])
        for(int j=i*i;j<=n;j+=i)prime[j]=0;
    }
    for(int i=2;i<=n;++i)
        if(prime[i])p.push_back(i);
}

int dp[(int)1.2e6];

        dp[i+3]=min(dp[i+3],1+dp[i]);
        dp[i+5]=min(dp[i+5],1+dp[i]);
        dp[i+7]=min(dp[i+7],1+dp[i]);
    }
