#include<bits/stdc++.h>

struct yoo
{
    long long data;
    long long x;
};
struct yoo arr[1000006];
int main()
{
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {
        scanf("%lld",&arr[i].data);
        arr[i].x=0;
    }
    int q;
    scanf("%d",&q);
    while(q--)
    {
        int i,j,k;
        scanf("%d%d%d",&i,&j,&k);
        arr[i-1].x+=k;
        arr[j].x-=k;
    }
    long long sum=0;
    for(int i=0;i<n;i++)
    {
        sum+=arr[i].x;
        printf("%lld\n",arr[i].data+sum);
    }
    return 0;
}
