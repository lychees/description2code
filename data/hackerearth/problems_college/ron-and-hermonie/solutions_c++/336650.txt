#include<iostream>
#include<cstdio>
#define s(a) scanf("%lld",&a)
#define p(a) printf("%lld",a)
#define pl printf("\n")

using namespace std;

typedef long long ll;

ll powr(ll a,ll p, ll m)
{
    if(p==0) return 1;
    if(p%2==0)
    {
       ll x=powr(a,p/2,m);
       return (x*x)%m;
    }
    return (a*powr(a,p-1,m))%m;
}

int main()
{
    ll T,B,e,ans,ld;
    s(T);
    while(T--)
    {
    	s(B); s(e);
    	ld=B%10;
    	if(ld==0 || ld==1 || ld==5 || ld==6) ans=ld;
    	else if(ld==2)
    	{
    		if(e%4==1) ans=2;
    		if(e%4==2) ans=4;
    		if(e%4==3) ans=8;
    		if(e%4==0) ans=6;
    	}
    	else if(ld==3)
    	{
    		if(e%4==1) ans=3;
    		if(e%4==2) ans=9;
    		if(e%4==3) ans=7;
    		if(e%4==0) ans=1;
    	}
    	else if(ld==4)
    	{
    		if(e%2==1) ans=4;
    		if(e%2==0) ans=6;
    	}
    	else if(ld==7)
    	{
    		if(e%4==1) ans=7;
    		if(e%4==2) ans=9;
    		if(e%4==3) ans=3;
    		if(e%4==0) ans=1;
    	}
    	else if(ld==8)
    	{
    		if(e%4==1) ans=8;
    		if(e%4==2) ans=4;
    		if(e%4==3) ans=2;
    		if(e%4==0) ans=6;
    	}
    	else if(ld==9)
    	{
    		if(e%2==1) ans=9;
    		if(e%2==0) ans=1;
    	}
    	
    	p(ans);
    	pl;
    }
    return 0;
}
