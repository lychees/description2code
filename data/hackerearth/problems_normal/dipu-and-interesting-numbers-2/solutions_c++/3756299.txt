#include <bits/stdc++.h>
#define lli long long
using namespace std;
lli cal(lli x)
{
	lli L, R, M, ans;
	L = 1, R = 1000000000LL;
	while (L <= R) {
		M = (L + R) >> 1;
		if (M*M <= x) {
			ans = M;
			L = M + 1;
		}
		else R = M - 1;
	}
	return ans;
}
int main()
{
	int t;
	lli a, b;
	cin >> t;
	while (t--) {
		cin >> a >> b;
		lli fs = cal(a);
		if (fs*fs != a) fs++;
		lli sc = cal(b);
		cout << sc - fs + 1 << endl;
	}
	return 0;
}
