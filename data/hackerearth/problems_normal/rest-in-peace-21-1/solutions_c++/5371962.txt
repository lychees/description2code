#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,flag=0;
    	cin>>n;
    	if(n%21==0)
		{
    		flag=1;
    	}
    	else
    	{
    	while(n>=1 && flag==0)
    	{
	    	if((n-21)%100==0)
	    	{
	    		//cout<<"hi1"<<endl;
	    		flag=1;
	    		break;
	    	}
    		n/=10;
    	}
	}
    	if(flag==0)
    	{
    		cout<<"The streak lives still in our heart!"<<endl;
    	}
    	else if(flag==1)
    	{
			cout<<"The streak is broken!"<<endl;
    	}
	}
    return 0;
}
