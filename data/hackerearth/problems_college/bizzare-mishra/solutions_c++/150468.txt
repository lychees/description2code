inline int read() //faster input for integer
{
	int n = 0;
	register char c = getchar_unlocked();//register is used due to its repeated use
	while (!('0' <= c && c <= '9'))
 	{
  		c = getchar_unlocked();
 	}
 	while ('0' <= c && c <= '9')
	{
  	n = (n<<3) + (n<<1) +  c - '0';//equivalent to n = n*10 + c - 48;
  	c = getchar_unlocked();
	}
 return n;
}
