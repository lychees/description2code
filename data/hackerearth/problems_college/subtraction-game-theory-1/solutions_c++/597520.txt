#include <iostream>
#include<cstdio>
using namespace std;

inline int read()
{
char c;
int n = 0;
while ((c = getchar_unlocked()) < 48);
n += (c - '0');
while ((c = getchar_unlocked ()) >= 48)
n = n * 10 + (c - '0');
return n;
}

int main()
{
    int t,n,k;
    //scanf("%d",&t);
    t=read();
    while(t--)
    {
    	n=read();
    	k=read();
    	if(n%2==1 && k==1) cout<<"Nandu\n";
    	else if(n%2==0 && k==1)  cout<<"Rani\n";
    	else
    	{
    		//a=k+2;  
    		if((n-k-2)%(k+1)==0) cout<<"Nandu\n";
    		else cout<<"Rani\n";
    	}
    }
    return 0;
}
