#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t;
    scanf("%d", &t);

    string s;
    while (t--) {
		cin >> s;
		int i=0;
		while (i < s.size()) {
			printf("%c", s[i]);
			int j=i+1;
			while (s[i] == s[j]) {
				j++;
			}
			i=j;
		}
		cout << endl;
    }

    return 0;
}
