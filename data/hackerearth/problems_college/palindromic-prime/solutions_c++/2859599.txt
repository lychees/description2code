#include<bits/stdc++.h>
using namespace std;
int p[1000001];
vector<int> dp;
int k;
bool ispail(int n)
{
	k=1;
	int arr[10];
	int i=0;
	while(n)
	{
		arr[i]=n%10;
		n=n/10;
		if(arr[i]!=0) k=k*arr[i];
		i++;
	}
	n=i;
	for(int i=0;i<=n/2;i++)
		if(arr[i]!=arr[n-i-1]) return 0;
return 1;
}
void seive()
{
	for(int i=2;i<1000001;i++) p[i]=1;
	for(int i=2;i<=1000;i++)
	{
		if(p[i])
		{
			for(int j=i*i;j<1000001;j+=i) p[j]=0;
		}
	}
	for(int i=2;i<1000001;i++)
	{
		if(p[i])
		{
			if(ispail(i)) dp.push_back(k);
		}
	}
}
int main()
{
	seive();
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,m;
		scanf("%d%d",&n,&m);
		n=dp[n-1];
		int ans=n*m;
		int sq=sqrt(ans);
		int c=0;
		for(int i=1;i<=sq;i++)
		{
			if(ans%i==0) c+=2;
		}
		if(sq*sq==ans) c--;
		printf("%d\n",c);
	}
return 0;
}
