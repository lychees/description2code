t = input()
for _ in xrange(t):
	n, x = [int(s) for s in raw_input().split()]
	prices = [input() for _ in xrange(n)]
	start = 0
	current_sum = 0
	for p in prices:
		current_sum += p
		while current_sum > x:
			current_sum -= prices[start]
			start += 1
		if current_sum == x:
			print 'YES'
			break
	else:
		print 'NO'
