def maxscore(nType1, nType2, nType1Score, nType2Score, nOtherTypeScore):
    if (nType1 > nType2):
        nDiff = nType1 - nType2
        nMaxScore = (nType1+nType2 - nDiff)*nOtherTypeScore + nDiff*nType1Score
    else:
        nDiff = nType2 - nType1
        nMaxScore = (nType1+nType2 - nDiff)*nOtherTypeScore + nDiff*nType2Score
    return nMaxScore


nTestCases = int(input())
for i in range(0, nTestCases):
    nType1, nType2, nType1Score, nType2Score, nOtherTypeScore = raw_input().split()
    nType1 = int(nType1)
    nType2 = int(nType2)
    nType1Score = int(nType1Score)
    nType2Score = int(nType2Score)
    nOtherTypeScore = int(nOtherTypeScore)

    if (nOtherTypeScore > nType1Score) and (nOtherTypeScore > nType2Score):
        nMaxScore = maxscore(
            nType1,
            nType2,
            nType1Score,
            nType2Score,
            nOtherTypeScore)
    else:
        nTemp1Score = nType1*nType1Score + nType2*nType2Score
        nTemp2Score = maxscore(
            nType1,
            nType2,
            nType1Score,
            nType2Score,
            nOtherTypeScore)
        if (nTemp1Score > nTemp2Score):
            nMaxScore = nTemp1Score
        else:
            nMaxScore = nTemp2Score
    print(nMaxScore)
