#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int n,i,j,sum=0;
    cin>>n;
    int a[n];
    for(i=0;i<n;i++){
    	cin>>a[i];
    	sum+=a[i];
    }
    sort(a,a+n);
    int b[10000]={0};
    b[0]=1;
    for(i=0;i<n;i++){              //// for each element in the array
    	for(j=1200;j>0;j--){       //// check the sum in reverse direcction
    		if(j-a[i]>=0)
    			if(b[j-a[i]]){
    			b[j]=1;	
    			}
    	}
    }
    int s1=0;
    for(i=0;i<=1200;i++)
    {
    	if(   (    (sum%3!=0 && i>sum/3) || (sum%3==0 && i==sum/3)   ) && b[i])
    	{
    		if(a[n-1]>=i)
    		s1 = a[n-1];
    		else
    		s1=i;
    		break;
    	}
    }
    cout<<s1;
    return 0;
}
