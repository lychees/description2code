#include <iostream>
#include <vector>

using namespace std;
int main()
{
    unsigned int numBox = 0, numQuery = 0;
    cin >> numBox;
    cin >> numQuery;
    vector<int> inpVec( numBox, 0 );
    while( numQuery-- )
    {
        /*cout << numQuery << " ";
        for( unsigned int k = 0; k < inpVec.size(); k++ )
            cout << inpVec[k] << " ";
        cout << endl;*/

        unsigned int i, j, i1, j1;
        cin >> i1 >> j1;
        i = std::min( i1, j1 );
        j = std::max( i1, j1 );

        if( i == j )
            cout << "NO" << endl;
        else if( inpVec[i-1] < 0 || inpVec[j-1] < 0 )
            cout << "NO" << endl;
        else if( inpVec[i-1] != inpVec[j-1] )
            cout << "NO" << endl;
        else
        {
            cout << "YES" << endl;
            inpVec[i-1] = -1;
            inpVec[j-1] = -2;
            int firstSet = 0;
            for( unsigned int k = i; k < j-1; k++ )
            {
                if( inpVec[k] < 0 )
                {
                    if( inpVec[k] == -1 )
                        firstSet++;
                    else if( inpVec[k] == -2 )
                        firstSet--;
                    continue;
                }
                if( !firstSet )
                    inpVec[k] = numQuery;
            }
        }
    }
    return 0;
}
