n = input()
if n == 0:
    print 0

else:
    rating = raw_input()
    rating = [int(i) for i in rating.split()]
    max_ = 0
    sum_ = 0
    for i in rating:
        if (sum_+i) < 0:
            sum_ = 0
        else:
            sum_ += i
        if sum_ > max_:
            max_ = sum_
    print max_
