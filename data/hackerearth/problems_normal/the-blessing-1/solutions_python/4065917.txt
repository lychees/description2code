#!/usr/bin/env python
# -*- coding: utf-8 -*-

accuracy = .001

T = int(raw_input())
for _ in xrange(0, T):
    # P, M = (0, 3)
    P, M = [int(x) for x in raw_input().split()]

    total = 0
    nom = P + .0
    denom = 1
    first = True
    while True:
        v = nom / denom
        if first:
            first = False
        else:
            if v < accuracy:
                break
        nom += M
        denom *= 2
        total += v
    print int(round(total))
