#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int a[n+1];
    for(int i=1;i<=n;i++)
      cin>>a[i];
      
      int batches_chintu;
     cin>>batches_chintu;
      int s,visited[102]={0},diff=0,diff2=0;
      
      while(batches_chintu--)
      { 
      	cin>>s;
        int a2[s+1],p,local_sum=0,loc_sum=0;
        for(int j=1;j<=s;j++)
      	   {cin>>a2[j];
      	    
      	    
      	    if(visited[a2[j]]==0) 
      	    {   
      	        local_sum+=a[a2[j]];
      	    	visited[a2[j]]++;
      	        
      	    }
      	    loc_sum+=a[a2[j]];
      	   
      	   }
      	  // cout<<local_sum<<endl;
      	   
      	  cin>>p;
          if(local_sum>=p) diff-=local_sum-p;
            else diff+=(p-local_sum);
            
            if(p>=loc_sum) diff2+=(p-loc_sum); 
          // cout<<diff<<endl; 
      }
    if(diff2>diff) cout<<diff2;
    else
    cout<<diff;
    
    //cout << "Hello World!" << endl;
    return 0;
}
