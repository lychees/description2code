#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() 
{
     int n,m,i,j;
     cin>>n>>m;
     int array[m];
    long long int dp[n+1][m];
     for(i=0;i<m;i++)
     dp[0][i]=1;
     
     for(i=0;i<m;i++)
     cin>>array[i];
     sort(array,array+m);
     for(i=1;i<n+1;i++)
     {
     
     	for(j=0;j<m;j++)
     	{
     		long long int x=0,y=0;
    		if(i>=array[j])
     			y=dp[i-array[j]][j];
     
     		if(j>=1)
     			x=dp[i][j-1];
     
     		dp[i][j]=x+y;
     	}
     }
     cout<<dp[n][m-1]<<endl;
    return 0;
}
