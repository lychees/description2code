int t;
int n;
vector<int> v;
int main(){
	scanf("%d", &t);
	while (t--){
		v.clear();
		int countt = 0;
		for (int i = 0; i < 7; i++){
			int c = 0;
			scanf("%d", &c);
			for (int j = 0; j < c; j++){
				int a, b;
				scanf("%d%d", &a, &b);
				v.push_back(b - a);
			}
			countt += c;
		}
		sort(v.begin(), v.end());
		double ans = (double)(countt)*0.75;
		ans = ceil(ans);
		int a = ans;
		long long int sum = 60LL*60LL * 24LL * 7LL;
		for (int i = 0; i < a; i++){
			sum -= (long long int)(v[i]);
		}
		printf("%lld\n", sum);
	}
	return 0;
}
