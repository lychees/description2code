#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	int i,c=0,a[26]={0};
    	for(i=0;i<s.size();i++)
    	if(a[s[i]-'a']>0)
    	a[s[i]-'a']--;
    	else
    	a[s[i]-'a']++;
    	
    	for(i=0;i<26;i++)
    	{
    		if(a[i]>0)
    		c++;
    	}
    	if(c>0)
    	cout<<c-1<<endl;
    	else
    	cout<<0<<endl;
    }
    
    return 0;
}
