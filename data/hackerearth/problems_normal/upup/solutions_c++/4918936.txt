#include<bits/stdc++.h>
using namespace std;
int main()
{
char ch;
int flag=1;
while(scanf("%c", &ch)!=EOF)
{
 if(ch == ' ') {
        flag = 1;
    }
    else {
        if(flag && ch>='a' && ch<='z')
            ch = 65+ch-97;

        flag=0;
    }

    printf("%c", ch);
    }
    printf("\n");
    return 0;
    }
