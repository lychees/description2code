#define XXX 100069

int a[101][101];
bool dp[101][101][501];
int n,m,k;
int main()
{
	int t;
	si(t);
	while(t--)
    {
        cin>>n>>m>>k;
    	FOR(i,0,n)
            FOR(j,0,m)
                cin>>a[i][j];
        memset(dp,false,sizeof(dp));
        dp[0][0][a[0][0]] = true;
        FOR(i,0,n)
            FOR(j,0,m)
                FOR(kk,a[i][j],k+1){
                	if(i!=0 || j!=0){
	                    if(i==0)
	                        if(dp[i][j-1][kk-a[i][j]])
	                            dp[i][j][kk]=true;
	                    if(j==0)
	                        if(dp[i-1][j][kk-a[i][j]])
	                            dp[i][j][kk]=true;
	                    if(i-1>=0 && j-1>=0)
	                        if(dp[i-1][j-1][kk-a[i][j]] || dp[i][j-1][kk-a[i][j]] || dp[i-1][j][kk-a[i][j]])
	                            dp[i][j][kk]=true;
                	}
                }
        int ans=-1;        
        FOR(i,1,k+1)
            if(dp[n-1][m-1][i]==true)
                ans = i;
        if(ans==-1)
            cout<<-1<<endl;
        else
            cout<<ans<<endl;
    }
	return 0;
}
