#include <bits/stdc++.h>
using namespace std;
int a[100002];
int b[100002];
int root(int a[],int i){
	while(i!=a[i])i=a[i];
	return i;
}
bool find(int c,int d){
	int x=root(a,c);
	int y=root(a,d);
	if(x==y)return true;
	return false;
}
void un(int a[],int b[],int c,int d){
	if(!find(c,d)){
		int x=root(a,c);
		int y=root(a,d);
		if(b[x]>=b[y]){
			a[y]=x;
			b[x]+=b[y];
		}
		else{
			a[x]=y;
			b[y]+=b[x];
		}
	}
}
int main()
{
    //cout << "Hello World!" << endl;
    int n,k,c,i,d;
    cin>>n>>k;
    for(i=1;i<=n;i++){
    	a[i]=i;
    	b[i]=1;
    }
    while(k--){
    	cin>>c>>d;
    	un(a,b,c,d);
    } d=0;set<int> s;
    for(i=1;i<=n;++i){
    	s.insert(root(a,a[i]));
    }
    cout<<s.size()<<endl;s.clear();
    return 0;
}
