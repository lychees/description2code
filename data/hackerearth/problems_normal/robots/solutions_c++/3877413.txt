#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
    int t;
    cin>>t;
    int n;
    vector<int> v;
    while(t--){
    	cin>>n;
    	v.resize(n);
    	for(int i=0;i<n;i++)cin>>v[i];
    	sort(v.begin(),v.end());
    	//for(int i=0;i<n;i++)cout<<v[i]<<" ";
    	ll mod=1000000007;
    	ll m=v[0]%mod;
    	for(int i=1;i<n;i++){
    		m=m*(v[i]-i)%mod;
    	}
    	cout<<m<<endl;
    	v.clear();
    }
    return 0;
}
