#include<bits/stdc++.h>

using namespace std;

int main()

{

int n,i,j,ans,ceo_h,coo_h,cto_h;
vector<int> a,seat;
scanf("%d %d %d",&ceo_h,&coo_h,&cto_h);
scanf("%d",&n);

a.resize(n);
seat.resize(n+3,0);   //total seats are n+3 and if b[i]==0 then it means that ith position is empty

for(i=0;i<n;i++)
    scanf("%d",&a[i]);

sort(a.begin(),a.end());   // sort the given heights in ascending order
seat[0]=min(coo_h,cto_h);   // fill the left end seat with either CTO or COO(which ever has lesser height)
seat[n+2]=max(coo_h,cto_h);   // fill the right end seat with the one with greater height between the CTO and the COO
seat[(n+3)/2]=ceo_h;   // fill the middle seat with the CEO

j=0;

for(i=0;i<n;i++)
{
    while(seat[j]!=0)   // check if the i^th seat is empty
    {
        j++;
    }
    seat[j]=a[i];   // fill the remaining empty seats with the given input in ascending order        
}

ans=0;
for(i=1;i<n+3;i++)
{
    ans+=abs(seat[i]-seat[i-1]);   // calculate the total absolute difference between adjacent heights
}
printf("%d\n",ans);
return 0;
}
