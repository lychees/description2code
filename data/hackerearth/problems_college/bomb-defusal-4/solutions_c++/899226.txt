#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	scanf("%d", &t);

	while (t--) {
		int p, a, b, c, d;
		int cnt = 1;

		scanf("%d %d %d %d %d", &p, &a, &b, &c, &d);
		
		int a1 = c - a;
		int a2 = d - b;

		if ((a1 != a2) && (abs(a1 - a2) % 2 == 0)) {
			cnt++;
		}

		if (a == c && b == d) {
			printf("Counter-Terrorists Win !\n");
		}

		else if (a1 == a2 && cnt < p) {
			printf("Counter-Terrorists Win !\n");
		}

		else {
			if (cnt < p && (abs(a1 - a2) % 2 == 0))
				printf("Counter-Terrorists Win !\n");
			else
				printf("Terrorists Win !\n");
		}
	}
}
