#include <bits/stdc++.h>
using namespace std;

int main()
{
   	int n,x;
   	while(scanf("%d",&n) > 0) {
   		vector<int> a;
   		for(int i = 0; i < n; i++) {
   			cin >> x;
   			a.push_back(x);
   		}
   		bool flag = false;
   		for(int i = n-1; i >= 0; i--) {
   			if(a[i] % 2 == 0) {
   				printf("%d ", a[i]);
   				flag = true;
   			}
   		}
   		if(flag == false) {
   			cout << "None to display";
   		}
   		cout << endl;
   	}
    return 0;
}
