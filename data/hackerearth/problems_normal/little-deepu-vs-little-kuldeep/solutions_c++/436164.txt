typedef long long LL;

LL Q=1000000007;

long long binpow(long long val, long long deg, long long mod) {
        if (!deg) return 1 % mod;
        if (deg & 1) return binpow(val, deg - 1, mod) * val % mod;
        long long res = binpow(val ,deg >> 1, mod);
        return (res*res) % mod;
}

LL factmod (LL n, LL p) {
	LL res = 1;
	while (n > 1) {

		res = (res * ((n/p) % 2 ? p-1 : 1)) % p;
		for (LL i=2; i<=n%p; ++i){
			res = (res * i) % p;
           // cout<<i<<" "<<n%p<<"\n";
		}
		n /= p;
	}
	return res % p;
}
