using namespace std;
#define lint long long int
bool f(pair<lint ,lint >a, pair<lint, lint >b){
	
	if(a.second>b.second) return true;
	if(a.second<b.second) return 0;
	return(a.first>b.first);
	
	}

{

   	cin>>id>>z>>p>>l>>c>>s;
   	m[id]= z;

   	pair<lint , lint >p ={ id, change};

   }
   
 sort(a.begin(), a.end(),f);
