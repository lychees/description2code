#include <bits/stdc++.h>
#define ll long long int
char s[100000];
using namespace std;
int main()
{
    ll i, n, t;
    scanf("%lld", &t);
    while(t--) {
        scanf("%s", s);
        n = strlen(s);
        for(i = 1 ; i < n; i++) {
            if(abs(s[i]-s[i-1]) != abs(s[n-i]-s[n-i-1]))
                break;
        }
        if(i == n)
            printf("Funny\n");
        else
            printf("Not Funny\n");
    }
    return 0;
}
