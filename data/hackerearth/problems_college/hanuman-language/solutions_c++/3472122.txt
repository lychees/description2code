#include <iostream>
#include<string.h>
using namespace std;
int main()
{
    int t,i,j,k;
    cin>>t;
	int x;
	x=t;
    int a[1000];
    char c[100];
char ans[1000][100];
    while(t+1!=0)
    {
    //cout<<t<<endl;
    cin.getline(c,sizeof(c));
    
 	//cout<<c;
	k=0;
    int l=0;
    for(i=strlen(c)-1;i>=0;i--)
      {
      	 if(c[i]==' ')
      	 {
            l=k;
      	 	for(j=i+1;j<=strlen(c)-l-1;j++)
      	 	{
      	 		ans[t][k++]=c[j];
      	 	
      	 	}
            
			ans[t][k++]=' ';
			
 			
      	 }
        if(i==0)
        { 
           l=k;
          for(j=i;j<=strlen(c)-k+1;j++)
      	 	{
      	 		ans[t][l++]=c[j];
      	 		
      	 	}	
        }
      }
a[t]=strlen(c);
    
t--;

   
    }



for(j=x-1;j>=0;j--)
{
for(i=0;i<=a[j]-1;i++)
{
    cout<<ans[j][i];
}
cout<<endl;
}

}
