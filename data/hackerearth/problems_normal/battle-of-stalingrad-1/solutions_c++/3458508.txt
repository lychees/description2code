#include <iostream>
using namespace std;

float absolute(float val) {
	if(val < 0)
		val = 0 - val;
	return val;	
}

int main()
{
	int t;
	cin >> t;
	for(int i=0; i<t; i++) {
    	float ax,ay,bx,by,cx,cy,px,py;
    	cin>> ax >> ay >> bx >> by >> cx >> cy >> px >> py;
    	float a_triangle = absolute((ax*(by-cy) + bx*(cy-ay) + cx*(ay-by))/2);
    	float a1 = absolute((ax*(py-cy) + px*(cy-ay) + cx*(ay-py))/2);
    	float a2 = absolute((px*(by-cy) + bx*(cy-py) + cx*(py-by))/2);
    	float a3 = absolute((ax*(by-py) + bx*(py-ay) + px*(ay-by))/2);
    	
    	if((a1+a2+a3)==a_triangle)
    		cout<< "INSIDE\n";
    	else
    		cout<< "OUTSIDE\n";
	}
    return 0;
}
