#include <iostream>
#include<vector>
#include<string>
#include<algorithm>
#include<typeinfo>
using namespace std;

int main()
{
    int t,n,i,j;
    string s;
    cin>>t;
    while(t--){
    	vector<int> a(26);
    	cin>>n;
    	cin>>s;
    	i=0;
    	while(i<n){
    		j=s[i]-97;
    		a[j]++;i++;
    	}
    	vector<int>::iterator it = max_element(a.begin(),a.end());
    	j=*it;
    	if(j%2==0 && j!=0){
    		cout<<"Yes"<<endl;
    	}
    	else{
    		cout<<"No"<<endl;
    	}
    }
    return 0;
}
