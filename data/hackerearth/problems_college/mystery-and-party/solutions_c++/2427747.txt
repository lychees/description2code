#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t,n,k,i;
    cin>>t;
    while(t--)
    {
    	scanf("%d%d",&n,&k);
    	int a[n];
    	for(i=0;i<n;i++)
    	scanf("%d",&a[i]);
    	sort(a,a+n);
	   	for(i=0;i<n;i++)
		{
			if(i==k-1)
			{
			cout<<a[i]<<"\n";
			break;
			}
		}
		
    }
    
    
    
    
    
    return 0;
}
