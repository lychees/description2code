    #include<iostream>
    using namespace std;
    int main()
    {
    	int te;
    	cin>>te;
    	while(te--)
    	{
    		long int n,t,l;
    		cin>>n>>t;
    		while(t--)
    		{
    			if(n%2)
    			n=(n-1)/2;       		
       			else
       				n=n/2;       		
       			l=n/4;
       			n=n-l;
    		}
    		cout<<n<<endl;
    	}
    }
