#include <iostream>
using namespace std;

int main()
{
	int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	// int p[n];
    	int sum=0,temp;
    	for(int i=0;i<n;i++)
    	{
    		cin>>temp;
    		if(temp%2!=0)sum++;
    		// cin>>p[i];
    	}
    	if(sum%2!=0)
    	{
    		cout<<"No"<<endl;
    		continue;
    	}
    	cout<<"Yes"<<endl;
    }
    return 0;
}
