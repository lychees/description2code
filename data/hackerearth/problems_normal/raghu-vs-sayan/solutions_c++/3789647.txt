#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long int a,b,n;
        cin>>a>>b>>n;
        vector<long long int> v;
        int i,x;
        for(i=0;i<n;i++)
        {
            cin>>x;
            v.push_back(x);
        }
        sort(v.begin(),v.end());
        long long int c1=0,c2=0;
        for(i=0;i<n;i++)
        {
            if(a-v[i]>=0)
            {
                a-=v[i];
                c1++;
            }
            if(b-v[i]>=0)
            {
                b-=v[i];
                c2++;
            }
        }
        if(c1>c2)
        {
            cout<<"Raghu Won";
        }
        else if(c1<c2)
        {
            cout<<"Sayan Won";
        }
        else
        {
            cout<<"Tie";
        }
        cout<<"\n";
    }
}
