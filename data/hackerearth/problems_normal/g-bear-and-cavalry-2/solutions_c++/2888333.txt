#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t; cin >>t ;
    while(t--){
    	int n, ok=0;
    	cin >> n;
    	vector<int> v1(n),v2(n);
    	for(int i=0;i<n;i++) cin >> v1[i];
    	for(int i=0;i<n;i++) cin >> v2[i];
    	sort(v2.begin(),v2.end(),greater<int>());
    	long long int m1=v1[0]*v2[0];
    	sort(v1.begin()+1,v1.end());
    	for(int i=1;i<n;i++){
    		if(m1<=v1[i]*v2[i]) {
    			ok=1;break;
    		}
    	}
    	if(ok) cout << "NO"<<endl;
    	else cout << "YES" <<endl;
    }
    return 0;
}
