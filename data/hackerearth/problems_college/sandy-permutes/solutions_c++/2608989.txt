#include <iostream>
using namespace std;
#define ll long long

int main()
{	ll a,b;
    cin>>a>>b;
    ll z[a];
    for(ll i=0;i<a;++i)
    	cin>>z[i];
    for(ll i=0;i<b&&i<a;++i)
    {	ll m=0,s=0;
    	for(ll j=i+1;j<a;++j)
    		if(m<z[j])
    		{	m=z[j];
    			s=j;
    		}
    	if(m>z[i])
    	{	m=z[s];
    		z[s]=z[i];
    		z[i]=m;
    	}
    	else
    		++b;
    }
    for(ll i=0;i<a;++i)
    	cout<<z[i]<<" ";
    return 0;
}
