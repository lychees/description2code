#include <iostream>
#include <algorithm>
#include <cstdio>
using namespace std;

long long int gcd(long long int a,long long int b)
{
	if(b==0)
		return a;
	else
		return (b,a%b);
}


int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
    	int a,b,n;
    	cin>>a>>b>>n;
    	long long int ans;
    	long long lcm=__gcd(a,b);
    	lcm=(a*b)/lcm;
    	long long int l=1,r,m;
    	r=10000000000000000LL;
    	while(l<=r)
    	{
    		m=(l+r)/2;
    		long long val=m/a+m/b-m/lcm;
    		if(val==n)
    		{
    			ans=max((m/a)*a,(m/b)*b);
    			cout<<ans<<endl;
    			break;
    		}else if (val > n)
    		{
    				r=m-1;
    		}else
    			l=m+1;
    	}
    }
    return 0;
}
