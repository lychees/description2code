#include <iostream>
#include <math.h>
using namespace std;

int main()
{
    int num, part,i,j,game[1500][1500],c=1,winr=1,win[1500][1500];
    cin >> num;
    part=pow(2,num);
    i=2;
    while(i<=part)
    {
    	j=1;
    	while(j<i)
    	{
    		cin >> game[i][j];
    		if(i==j+1&&i%2==0)
    		{
    			if(game[i][j]==0)
    			 win[1][winr]=j;
    			else
    			 win[1][winr]=i;
    			winr++; 
    		}
    		j++;
    	}
    	i++;
    }
    
    while(c<=num)
    {
    	part=part/2;
    	winr=1;
    	i=2;
    	while(i<=part)
    	{
    		if(game[win[c][i]][win[c][i-1]]==0)
    		 win[c+1][winr]=win[c][i-1];
    		else
    		 win[c+1][winr]=win[c][i];
    		 winr++;
    		 i=i+2;
    	}
    	c++;
    }
    cout << win[c-1][1] << endl;
    return 0;
}
