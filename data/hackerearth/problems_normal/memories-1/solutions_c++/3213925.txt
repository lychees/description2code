#include <bits/stdc++.h>
using namespace std;
int power(int a,int x)
{
	int p=1,i;
	for(i=1;i<=x;i++)
	  p=p*a;
	return p;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n,p;
    	scanf("%d%d",&n,&p);
    	int i,x[p],y[p],j,k,ans=power(2,n)-1;
    	bool flag=0;
    	for(i=0;i<p;i++)
    	 scanf("%d%d",&x[i],&y[i]);
    	for(i=1;i<power(2,n);i++)
    	{  flag=0;
    		for(j=0;j<n;j++)
    		{
    		   if((i&(1<<j))!=0)
    		    {
    		      for(k=0;k<p;k++)
    		      {
    		      	if((j+1)==x[k])
    		      	{
    		      		if((i&(1<<(y[k]-1)))!=0)
    		      		 { ans--; flag=1; break;}
    		      	}
    		      	
    		      	if((j+1)==y[k])
    		      	{
    		      		if((i&(1<<(x[k]-1)))!=0)
    		      		  { ans--; flag=1; break; }
    		      	}
    		      }
    		    }
    		    if(flag==1)
    		     break;
    		}
    	}
    	printf("%d\n",ans);
    }
    return 0;
}
