#include <iostream>
#include <cmath>
using namespace std;

int main()
{
   long long m,sum1=0,sum2=0;
   cin >> m;
   long long mat[m][m];
   for(long long i=0;i<m;i++)
   {
   		for(long long j=0;j<m;j++)
   		{
   			cin >> mat[i][j];
   		}
   }
   for(long long i=0;i<m;i++)
   {
   		for(long long j=0;j<m;j++)
   		{
   			if(i==j)
   			{
   				sum1=sum1+mat[i][j];
   			}
   			if((i+j)==(m-1))
   			{
   				sum2=sum2+mat[i][j];
   			}
   		}
   }
   //cout << sum1 << endl;
   //cout << sum2 << endl;
   long long ans=abs(sum1-sum2);
   cout << ans << endl;
}
