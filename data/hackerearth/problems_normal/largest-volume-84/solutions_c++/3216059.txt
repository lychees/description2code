# define M_PI 3.14159265358979323846
using namespace std;

double vol(double l,double b, double h){
    return (M_PI * h * (l * l + l * b + b * b) / 3);
}

double heightc(double h, long long k){
    double b = k * 0.001;
    return sqrt(h*h - b*b);
}

int main(){
    double lr;
    cin >> lr;
    double br = lr;
    double ih;
    cin >> ih;
    double h = ih;
    long long k = 0;

    double v = vol(lr, br, h);

    br += 0.001;
    ++k;
    h = heightc(ih, k);
    while(vol(lr, br, h) > v){
        v = vol(lr, br, h);
        br += 0.001;
        ++k;
        h = heightc(ih, k);

    }
    cout << fixed << setprecision(2) << floor(v*100+0.5)/100;
}
