#include<bits/stdc++.h>
using namespace std;
int m,n;

int arr[1001][1001];
queue <pair<int,int > > q;
int check(int i,int j)
{
    if(i>=1&&i<=n&&j>=1&&j<=m&&arr[i][j]==1)
        return 1;
    return 0;
}
int main()
{
    int q1,i,j;
    scanf("%d%d%d",&n,&m,&q1);
    int cnt=0;
    for(i=1;i<=n;i++)
    {
        for(j=1;j<=m;j++)
        {
            scanf("%d",&arr[i][j]);
            if(arr[i][j]==1)
                cnt++;
        }
    }
    int a,b;
    for(i=0;i<q1;i++)
    {
        scanf("%d%d",&a,&b);
        if(arr[a][b]==0)
            printf("%d\n",cnt);
        else
        {
               cnt--;
               q.push(make_pair(a,b));
               arr[a][b]=0;
               pair<int,int> p;
               while(!q.empty())
               {
                   p=q.front();
                   a=p.first;
                   b=p.second;
                   q.pop();
               if(check(a+1,b)==1)
               {
                   q.push(make_pair(a+1,b));
                   arr[a+1][b]=0;
                   cnt--;
               }
               if(check(a,b+1)==1)
               {
                   q.push(make_pair(a,b+1));
                   arr[a][b+1]=0;
                   cnt--;
               }if(check(a-1,b)==1)
               {
                   q.push(make_pair(a-1,b));
                   arr[a-1][b]=0;
                   cnt--;
               }if(check(a,b-1)==1)
               {
                   q.push(make_pair(a,b-1));
                   arr[a][b-1]=0;
                   cnt--;
               }

               }
               printf("%d\n",cnt);
        }

    }
    return 0;
}
