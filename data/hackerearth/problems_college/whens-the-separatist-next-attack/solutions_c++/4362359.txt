#include<iostream>
#include<vector>
#include<map>
using namespace std;
typedef long long ll;
int main() {
	ll T; cin >> T;
	vector<ll> myvec;
	myvec.push_back(1);
	myvec.push_back(0);
	for(ll k=2; k<62; k++) myvec.push_back(k);
	while(T--)
	{
		string str; cin >> str;
		ll cnt = 0,ans=0;
		map<char, ll> mymap;
		for(ll i=0; i<str.length(); i++)
			if(mymap.find(str[i])==mymap.end())
				mymap[str[i]] = myvec[cnt++];
		ll len = mymap.size();
		if(len==1)
		{
			for(ll i=0; i<str.length(); i++)
				ans = ans*2+mymap[str[i]];
		}
		else
		{
		for(ll i=0; i<str.length(); i++)
				ans = ans*len+mymap[str[i]];
		}
		cout << ans << endl;
	}
	return 0;
}
