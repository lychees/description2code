def do_task(a,x,y):
	for i in xrange(0,x+1):
		for j in xrange(0,y+1):
			if a[i][j] == 1:a[i][j] = 0
			else:a[i][j] = 1
	return a
	
t = input()	
while t:
	n,m = map(int,raw_input().split(' '))
	row = n - 1
	col = m - 1
	a = []
	while n:
		a.append(map(int,list(raw_input())))
		n -= 1
	
	c = 0
	for i in xrange(row,-1,-1):
		for j in xrange(col,-1,-1):
			if a[i][j] == 0:
				a = do_task(a,i,j)
				c += 1
	
	print c
	
	t -= 1
