#include <iostream>
using namespace std;

int main()
{
    int t,n,i,x,sum;
    cin>>t;
    while(t--){
    	cin>>n;
    	sum=0;
    	for(i=0;i<n;i++){
    		cin>>x;
    		if(x%2!=0){
    			sum+=x;
    		}
    		if(sum%2==0){
    			sum=0;
    		}
    	}
    	if(sum){
    		cout<<"No"<<endl;
    	}
    	else{
    		cout<<"Yes"<<endl;
    	}
    }
    return 0;
}
