#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {   int n,a[100];
        cin>>n;
        for(int i=0;i<n;i++)
            cin>>a[i];
            sort(a+0,a+n);
        int s=((a[n/2]-a[0])+(a[n-1]-a[n/2]))*2;
        cout<<s<<endl;
    }
    return 0;
}
