#include<bits/stdc++.h>
#define ll long long
#define mod 1000000007

using namespace std;

ll arr[1025];
ll funcn(ll x)
{

    ll pro=1;
    for(int i=1;i<=x;i++)
        pro=(pro*i)%mod;
    return pro;
}



int main()
{
    for(int i=0;i<1025;i++)
        arr[i]=0;
    int n,m;
    ll sum=1;
    cin>>n>>m;
    for(int i=1;i<=n;i++)
    {
        for(int j=1;j<=m;j++)
            arr[i^j]++;
    }
    for(int i=0;i<1025;i++)
    {
        if(arr[i]!=0)
            sum = (sum* funcn(arr[i]))%mod;
    }
    cout<<sum;
}
