#include <iostream>

using namespace std;

int main()
{
        int test_cases,count;
        cin>>test_cases;
        while(test_cases > 0)
        {
        	    count =0;
                int X1,X2,Y1,Y2;
                cin>>X1>>Y1>>X2>>Y2;
                
                if(X1==Y2)
                      count++;
                if(Y1==X2)
                     count++;
                cout<<count<<"\n";
                
                test_cases --;
        }
        return 0;
}
