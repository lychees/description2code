#include <iostream>
using namespace std;

int l[1000010] = {0};

int main()
{
	int ctr = 0;
	for (int i = 0; i <= 1000009; i++) {
		if(l[i]) continue;
		l[i] = ctr + i;
		if(i + ctr <= 1000009) l[i + ctr] = i;
		ctr++;
	}
    //cout << "Hello World!" << endl;
    int t;
    cin >> t;
    while(t--) {
    	int a, b;
    	cin >> a >> b;
  		if(a > b) {
  			int temp = a;
  			a = b;
  			b = temp;
  		}
  		bool flag = false;
  		if(l[a] == b) flag = true;
    	if(flag) puts("Don't Play");
    	else puts("Play");
    }
    return 0;
}
