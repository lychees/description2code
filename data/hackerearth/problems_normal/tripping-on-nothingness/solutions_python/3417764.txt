def gcd(a, b):
    """Calculate the Greatest Common Divisor of a and b.

    Unless b==0, the result will have the same sign as b (so that when
    b is divided by it, the result comes out positive).
    """
    while b:
        a, b = b, a%b
    return a

line1 = raw_input()
out = []
S,Q = line1.split(' ')
for i in range(int(Q)):
	A_i = raw_input()
	M = gcd(int(S),int(A_i))
	if M not in out:
		print M
		out.append(M)
	else:
