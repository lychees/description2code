import math


N = input()

if (N == 1):
    print 0
elif N == 2:
    print 1
else:
    count = 1
    prev = 1
    for i in range(3, N+1):
        prime = True
        for j in range(2, int(math.sqrt(i))+1):
            if i % j == 0:
                prime = False
                break
        if prime:
            prev = prev*2
        count += prev

    print count
