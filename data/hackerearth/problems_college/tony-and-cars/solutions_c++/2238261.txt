#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;
int main(){
	int T;
	cin>>T;
	while(T!=0){
		long N;
		cin>>N;
		int c=0;
		long m=N;
		while(m!=0){
			m=m/10;
			c=c+1;}
			long d=0;
			int e=0;
			if(c==1){cout<<N<<endl;}
			else{
			while(e!=c-1){
				d=d+(e+1)*9*pow(10,e);
				e++;
				}
				long nine=0;
			for(int i=1;i<c;i++){
				nine=nine+9*pow(10,i-1);
				}
				d=d+c*(N-nine);
				cout<<d<<endl;}
		T--;}
}
