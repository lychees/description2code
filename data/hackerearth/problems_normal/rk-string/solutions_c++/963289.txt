/* here we maintain the frequency of eac h letter at every posistion 
  and then subtract 
  from N and find the value at that position 
  */

const int N=1e6+10;
ll m[N][2]={{0}},ff[N]={0};

ll solve(int N,string s,char ch)
{ 
  ll ans=0;
  int d;
  for(int i=0;s[i];i++)
  {
    if(i==0)
    {
      if(s[i]==ch)
      ff[i]=1;
      else
      ff[i]=0;
      if(ff[i]>=N)
      {
        d=ff[i]-N;
        if(s[i]!=ch)
        {
          ans+=m[d][0];
          ans++;
        }
        else if ( s[i]==ch && N!=0)
        {
          ans+=m[d][0];
          ans++;
        }
      }
      if(s[i]==ch)
      m[ff[i]][1]++;
      else
      m[ff[i]][0]++;
      
    }
    else
    {
      if(s[i]==ch)
      ff[i]=ff[i-1]+1;
      else
      ff[i]=ff[i-1];
      if(ff[i]>=N)
      {
        d=ff[i]-N;
        if(s[i]!=ch)
        {
          ans+=m[d][0];
          ans++;
        }
        else if ( s[i]==ch && N!=0)
        {
          ans+=m[d][0];
          ans++;
        }
      }
      if(s[i]==ch)
      m[ff[i]][1]++;
      else
      m[ff[i]][0]++;
    }
  }
  return ans;  
}

int main()
{
  int t,M,N;
  si(t);
  while(t--)
  { 
    memset(m,0,sizeof(m));
    si(M);
    si(N);
    string s;
    cin>>s;
    char ch='R';
    pl(solve(M,s,ch));
    memset(m,0,sizeof(m));
    ch='K';
    pl(solve(N,s,ch));nl;
  }
}
