#include <iostream>
#include<string>
#include <stack>
using namespace std;

bool cross(string str)
{
	int l =0, r= str.size()-1;
	if((r+1)%2==1) return false;
	
	stack<char> s;
	for(unsigned int i =0;i<str.size();i++)
	{
		if(!s.empty())
		{
			char top = s.top();
			if(top == str[i])
				s.pop();
			else
				s.push(str[i]);
		}
		else
			s.push(str[i]);
	}
	if(s.empty()) return true;
	return false;
}
int main()
{
    int M = 0;
    cin>>M;
    int count =0;
    for(int i =0;i<M;i++)
    {
    	string str;
    	cin>>str;
    	if(cross(str)) count++;
    	
    }
    
    cout<<count;

    return 0;
}
