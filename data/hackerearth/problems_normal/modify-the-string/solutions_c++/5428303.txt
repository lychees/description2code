#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    string name;
    cin>>name;
    
    for(int i=0;i<name.length();i++)
    {
    	if(isupper(name[i]))
    	{
    		name[i]=tolower(name[i]);
    	}
    	else
    	{
    		name[i]=toupper(name[i]);
    	}
    }
    cout<<name;
    return 0;
}
