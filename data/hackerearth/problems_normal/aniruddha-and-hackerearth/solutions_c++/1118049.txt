typedef long long int LL;
LL arr[1000005];
LL dp[1000005];
#define mod 1000000007

	pre();

    cin>>t;
    while(t--)
    {
    	cin>>n;
    	cout<<dp[n]<<endl;

void pre()
{
	//memset(arr,0,sizeof(arr));
	//memset(dp,0,sizeof(dp));
	arr[0]=0;
	arr[1]=2;
	arr[2]=3;
	dp[0]=0;
	dp[1]=2;
	dp[2]=5;
	for(int i=3;i<=1000005;i++)
	{
		arr[i]=(arr[i-1]%mod+arr[i-2]%mod)%mod;
		dp[i]=(dp[i-1]%mod+arr[i]%mod)%mod;
	}
}
