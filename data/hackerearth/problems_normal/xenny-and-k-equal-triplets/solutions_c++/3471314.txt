# include <iostream>
using namespace std;

long long GCD(long long x,long long y)
{
    while (x != 0 && y != 0)
          if (x > y)
              x = x % y;
          else
              y = y % x;
    return x + y;
}

int main()
{
    int t;  cin>>t;
    while (t--)
    {
        long long n,k,counter=0;
        cin>>n>>k;
        for (int i=0;i<n;i++)
        {
            long long x;  cin>>x;
            counter += (x == k);
        }
        long long nominator   = counter * (counter-1) * (counter-2) / 6;
        long long denominator = n * (n-1) * (n-2) / 6; 
        long long gcd = GCD(nominator,denominator);
        nominator /= gcd;
        denominator /= gcd;
        cout<<nominator<<"/"<<denominator<<endl;
    }
    
    
    return 0;
}
