#include<bits/stdc++.h>
using namespace std;
int main() {
	long long int t,wt;
	long long int d;
	scanf("%lld",&t);
	while(t--) {
		scanf("%lld%lld",&d,&wt);
		if(d>60)
			printf("NA\n");
		else {
			long long int ans=50;
			if(d>=2) {
				d-=2;
				ans+=d*12;
			}
			ans+=wt*2;
			printf("%lld\n",ans);
		}
		
	}
	return 0;
}
