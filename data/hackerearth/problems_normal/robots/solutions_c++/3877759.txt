#include <bits/stdc++.h>
using namespace std;

#define MOD 1000000007LL
#define ll long long

int main()
{
	int t,n;
	cin >> t;
	while(t--)
	{
		cin >> n;
		vector<int> A(n);
		for(int i=0;i<n;++i)
			cin >> A[i];
		sort(A.begin(),A.end());
		ll ways = 1;
		for(int i=0;i<n;++i)
			ways = (ways*(A[i]-i))%MOD;
		cout << ways << endl;		
	}
	return 0;
}
