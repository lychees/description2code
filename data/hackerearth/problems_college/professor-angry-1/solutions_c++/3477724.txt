#include <iostream>
using namespace std;

int main()
{
	long long int t,i,n,k,a[1000],cnt=0;
	cin>>t;
	while(t>0)
	{
		cnt=0;
		cin>>n>>k;
		for(i=0;i<n;i++)
		{
			cin>>a[i];
			
			if(a[i]<=0)
			cnt++;
		}
	
	if(cnt<k)
    cout << "YES" << endl;
    else
    cout << "NO" << endl;
	
		t--;
	}
    return 0;
}
