#include <bits/stdc++.h>
#define ll long long 
using namespace std;

int main()
{
    ll t,i;
    cin>>t;
    while(t--)
    {
    	ll x,y,n,tm;
    	scanf("%lld %lld %lld",&x,&y,&n);
    	
    	ll sum=x+y;
    	for(i=3;i<=n;i++)
    	{
    		tm=y;
    		if(y>x)
    		y=y-x;
    		else if(x>=y)
    		y=x-y;
    		
    		x=tm;
    		sum+=y;
    		if(x==0)
    		break;
    	}
    	ll tr=n-i;
    	ll r=tr%3;
    	if(r==0)
    	{
    		sum+=tr/3*(2*y);
    	}
    	if(r==1)
    	{
    		sum+=(tr-1)/3*(2*y);
    		sum+=y;
    		
    	}
    	if(r==2)
    	{
    		sum+=(tr-1)/3*(2*y);
    		sum+=y;
    		
    	}
    	
    printf("%lld\n",sum);
    	
    }
    return 0;
}
