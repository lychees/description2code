#include <stdio.h>
using namespace std;

int arr[1000005];

bool check(int n, int len){
	long long sum=0;
	int j, i;
	for (j=0; j<len; j++){
		sum+=arr[j];
	}
	j--;
	
	for (i=0, j; j<n; i++, j++){
		if (sum>=0)
			return true;
		sum-=arr[i];
		sum+=arr[j+1];
	}
	return false;
}


//returns the bigger length of a subarray which have sum bigger than -1//
int solve(int n){
	int left=1, right=n, mid, ans=-1;
	while (left<=right){
		mid=(left+right)/2;
		if (check(n, mid)){
			ans=mid;
			left=mid+1;
		}
		else{
			//printf ("mid=%d\n", mid);
			//printf ("left=%d\n", left);
			//printf ("right=%d\n\n", right);
			right=mid-1;
		}
	}
	//if (check(n, mid)==false)
	//	printf ("FUCCKK\n");
	//else
	//	printf ("HAHAHAHA\n");
	//printf ("\n%d\n", ans);
	return ans;
}

int main(){
	int n;
	scanf ("%d", &n);
	for (int i=0; i<n; i++)
		scanf ("%d", &arr[i]);
	
	int ans=solve(n);
	if (ans==-1){
		printf ("-1\n");
	}
	else{
		int times=0;
		int i, j;
		long long sum=0;
		for (j=0; j<ans; j++)
			sum+=arr[j];
		j--;
		for (i=0, j; j<n; i++, j++){
		//printf ("%d\n", sum);
		if (sum>=0)
			times++;
		sum-=arr[i];
		sum+=arr[j+1];
		}
		printf ("%d %d\n", ans, times);
	}
	
	
	return 0;
}
