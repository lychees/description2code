t = int(raw_input())

h = []
a = [True]*1001
for i in range(2, 1001):
    if a[i]:
        h.append(i)
        for m in xrange(2*i, 1001, i):
            a[m] = False


for i in range(0, t):
    m1 = int(raw_input())
    count = 0
    for i in range(0, m1):
        if int(raw_input()) in h:
            count = count+1
    print count
