#include<bits/stdc++.h>
int main()
{
	int t,n,m;
	scanf("%d",&t);
	int arr[1001]; int ind;
	double min,price;
	int a,b,c;
	while(t--)
	{
		scanf("%d %d",&n,&m);
		for(int i=0;i<n;i++)
		{
			min=INT_MAX; ind=-1;
			for(int j=0;j<m;j++)
			{
				scanf("%d %d %d",&a,&b,&c);
				price = ((100-a)*(100-b)*(100-c)*100)/(double)1000000.0;
				if(price < min)
				{
					min = price;
					ind = j;
				}
			}
			arr[i]=ind;
		}
		for(int i=0;i<n;i++)
			printf("%d ",arr[i]+1);
		printf("\n");
	}
}
