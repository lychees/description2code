#include<iostream>
#include<cstdio>

#ifndef ONLINE_JUDGE
#define gc getchar
#else
#define gc getchar_unlocked
#endif // ONLINE_JUDGE

using namespace std;

long long int input(){
char c=gc();
while(c<'0'||c>'9') c=gc();
long long int ans=0;
while(c>='0'&&c<='9'){
    ans=10*ans+c-'0';
    c=gc();
}
return ans;
}

int main(){
long long int t;
t=input();
while(t--){
        long long int p,x,y,u,v,time;
        p=input();
        x=input();
        y=input();
        u=input();
        v=input();
        if(y==v&&x==u) time=0;
        else if((v-y==u-x)||(v-y==-(u-x))) time=1;
        else{
            if(((x+u+v-y)%2==0&&(y+v+u-x)%2==0)||((x+u+y-v)%2==0&&(y+v+x-u)%2==0)) time=2;
            else time=1000000;
        }
        if(time<p) cout<<"Counter-Terrorists Win !\n";
        else cout<<"Terrorists Win !\n";
}
return 0;
}
