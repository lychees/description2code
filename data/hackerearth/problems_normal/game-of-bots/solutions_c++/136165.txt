#include<iostream>
#include<cstring>
using namespace std;
int checkstring(char *str1, char *str2){
        long long i=0;
        long long countB_in=0, countB_fi =0, countA_in=0, countA_fi=0;

        for(i=0;i<strlen(str1);i++){
            if(str1[i]=='B'){
               countB_in++;
               if(countA_in!=countA_fi)
                return 0;
            }
            if(str2[i]=='B'){
                countB_fi++;
                if(countB_fi>countB_in)
                return 0;
                if(countA_in!=countA_fi)
                return 0;

            }
            
            if(str2[i]=='A'){
                countA_fi++;
                if(countB_fi!=countB_in)
                return 0;
            }
            
            if(str1[i]=='A'){
                 countA_in++;
                 if(countA_in>countA_fi)
                    return 0;
                 if(countB_fi!=countB_in)
                 return 0;
            }

           
        }
        if(countB_fi!=countB_in)
        return 0;
        if(countA_fi!=countA_in)
        return 0;

    return 1;
}
 int main()
 {

     long long n;
     char str1[1000002], str2[1000002];
     cin>>n;
     while(n--){
         cin>>str1>>str2;
         int res=checkstring(str1,str2);
         if(res)
         cout<<"Yes\n";
         else cout<<"No\n";
       
     }
      return 0;
 }
