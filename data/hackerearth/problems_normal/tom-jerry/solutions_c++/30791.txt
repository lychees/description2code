void scanint(int &x)
{
register int c = gc();
x = 0;
for(;(c<48 || c>57);c = gc());
for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}

int a[101][101];
int visited[101][101];
int n;
void dfs(int i,int j,long long &ways)
{
    if(i<0||j<0||i>=n||j>=n||a[i][j]==1||visited[i][j]==1)
        return;

    if(i==n-1&&j==n-1)
    {
         ways++;
         return;
    }

    visited[i][j]=1;

    dfs(i,j+1,ways);

    dfs(i+1,j,ways);

    dfs(i-1,j,ways);

    dfs(i,j-1,ways);
    
    visited[i][j]=0;

    return;
}
int main()
{
    int j,i;
    long long ways;
    ways=0;
    scanint(n);
    for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++)
        {
            scanint(a[i][j]);
            visited[i][j]=0;
        }
    }

    dfs(0,0,ways);
    cout<<ways<<"\n";

    return 0;
}
