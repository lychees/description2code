#include <algorithm>
#include <cstdio>
#include <cstring>
#include <iostream>
#include <map>
#include <queue>
#include <stack>
#include <string>
#include <vector>

using namespace std;

int main() {
  long long int x, y, z, k;
  cin >> x >> y >> z >> k;
  if (x + y + z - 3 != 0) {
    while (x + y + z - 3 > k) {
      if (x >= y && x >= z)
        x--;
      else {
        if (y > x && y > z)
          y--;
        else
          z--;
      }
    }
    cout << x * y * z << endl;
  } else
    cout << 1 << endl;
}
