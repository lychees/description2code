import sys

n = int(raw_input())
l = [chr(i) for i in xrange(65, 91)]


def swap(a, b):
    global l
    x = l.index(a)
    y = l.index(b)
    l[x], l[y] = l[y], l[x]


for _ in xrange(n):
    a, b = raw_input().split()
    a, b = a.upper(), b.upper()
    swap(a, b)

# print l
s = sys.stdin.read()
for i in s:
    x = l[ord(i.upper())-ord('A')]
    if i.islower():
        x = x.lower()
    sys.stdout.write(x)
