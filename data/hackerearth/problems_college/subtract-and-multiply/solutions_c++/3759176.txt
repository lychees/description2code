#include <iostream>
#include<cmath>
using namespace std;

int main()
{
   long long int t,d,p;
   double ps;
   cin>>t;
   while(t--){
	   	cin>>d>>p;
	   	
	   	ps=sqrt(d*d+4*p);
	   	if(ps*ps==d*d+4*p){
	   		if(d==0)
	   		cout<<"2"<<endl;
	   		else
	   		cout<<"4"<<endl;
	   	}
	   	else
	   	cout<<"0"<<endl;
	}
    return 0;
}
