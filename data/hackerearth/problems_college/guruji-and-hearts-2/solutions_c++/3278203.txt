#include <iostream>
#include<string>
using namespace std;

int main()
{      string s;
      cin>>s;
      int i=0,j=1,count=0;
      while(s[j]!='\0')
      {
      	  if(s[i]=='<' && s[j]=='3')
      	  {
      	  	  count++;
      	  }
      	  i++;
      	  j++;
      }
      cout<<count<<endl;
      return 0;
}
