#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		
		unsigned long long int a,b,n;
		cin>>a>>b>>n;
		vector<unsigned long long int> v;
		v.resize(n);
		for(unsigned long long int i=0;i<n;i++)cin>>v[i];
		sort(v.begin(),v.end());
		unsigned long long int counta=0,countb=0;
		unsigned long long int sum=0;
		while(sum<=a&& sum+v[counta]<=a&&counta<n)
		{
			sum+=v[counta];counta++;
		}
		sum=0;
		while(sum<=b&&sum+v[countb]<=b&&countb<n)
		{
			sum+=v[countb];countb++;
		}
		if(counta>countb)cout<<"Raghu Won"<<endl;
		else if(counta==countb)cout<<"Tie"<<endl;
		else cout<<"Sayan Won"<<endl;
	}
}
