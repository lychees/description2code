#include<iostream>
using namespace std;
bool check_palindrom(std::string str1,std::string str2)
{
	int n = str1.length();
	bool res = true;
	for(int i =0 ;i< n; i++)
	{
		if(!(str1.at(i) == str2.at(n-i-1)))
		{
			res = false;
			break;
		}
	}
	return res;
}

int main()
{
    int n;
    cin>>n;
    string arr[n];
    bool found = false;
    for(int i=0;i<n;i++)
    {
    	cin>>arr[i];
    }
    
    for(int i=0;i<n;i++)
    {
    	string str= arr[i];
    	for(int j=i;j<n;j++)
    	{
    		if(str.length() == (arr[j]).length())
    		{
    			int mid = str.length()/2;
    			if(str.at(mid) == arr[j].at(mid))
    			{
    				if(check_palindrom(str,arr[j]))
    				{
    					cout<<str.length()<<" "<<str.at(mid);
    					found = true;
    					break;
    				}
    			}
    		}
    	}
    	if(found)
    		break;
    }
}
