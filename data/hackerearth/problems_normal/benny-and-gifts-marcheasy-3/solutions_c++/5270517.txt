#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s; cin>>s;
    map<pair<int,int>,int>mymap;
    int x=0,y=0,ans=0;
    mymap[make_pair(x,y)]=1;
    for(int i=0;i<s.length();i++){
    	if(s[i]=='L'){
    		y--;
    		if(mymap[make_pair(x,y)]==1) ans++;
    		mymap[make_pair(x,y)]=1;
    	}
    	else if(s[i]=='R'){
    		y++;
    		if(mymap[make_pair(x,y)]==1) ans++;
    		mymap[make_pair(x,y)]=1;
    	}
    	else if(s[i]=='U'){
    		x--;
    		if(mymap[make_pair(x,y)]==1) ans++;
    		mymap[make_pair(x,y)]=1;
    	}
    	else if(s[i]=='D'){
    		x++;
    		if(mymap[make_pair(x,y)]==1) ans++;
    		mymap[make_pair(x,y)]=1;
    	}
    }
    cout<<ans<<endl;
    return 0;
}
