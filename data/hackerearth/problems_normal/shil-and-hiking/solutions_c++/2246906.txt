#include <iostream>

using namespace std;

int main()
{
    int n,j,s=0;
    cin>>n;
    int x[n];
    int y[n];
    double ma;
    double sl[n];
    for(int i=0;i<n;i++)
        cin>>x[i]>>y[i];
    for(int i=0;i<n-1;i++)
    {
        for(j=i+1;j<n;j++)
            sl[j]=(y[j]-y[i])/((double)(x[j]-x[i]));
        ma=sl[i+1];
        s++;
        for(j=i+2;j<n;j++)
        {
            if(sl[j]>=ma)
            {
                ma=sl[j];
                s++;
            }
        }
    }
    cout<<s<<endl;

}
