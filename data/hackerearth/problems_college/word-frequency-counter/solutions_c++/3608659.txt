#include <iostream>
#include <vector>
#include <string>
using namespace std;

int main ()
{
	ios_base::sync_with_stdio(false);
	
	string temp;
	vector<string> w;

	while (cin >> temp) {
		w.push_back(temp);
	}	

	for (int i=0; i<w.size(); ++i) {
		for (int j=0; j<w[i].size(); ++j) {
			if (w[i][j] >= 97 && w[i][j] <= 122) {
				w[i][j] -= 32;
			}	
		}
	}

	for (int i=0; i<w.size(); ++i) {
		if (w[i] != "**") {
			int c = 0;
			temp = w[i];

			for (int j=i; j<w.size(); ++j) {
				if (temp == w[j]) {			
					++c;
					w[j] = "**";
				}
			}
			cout << temp << " " << c << "\n";
		}
	}

	return 0;
}
