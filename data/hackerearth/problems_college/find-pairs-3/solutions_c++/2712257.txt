#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,c1=0,i,j;
    	cin>>n;;
    	int arr[n];
    	for(i=0;i<n;i++)
    	{
    		cin>>arr[i];
    		
    	}
    	for(i=0;i<n;i++)
    	{
    		for(j=i+1;j<n;j++)
    		{
    			if(arr[i]==arr[j])
    			c1++;
 
    		}
    	}
    	cout<<c1+n<<endl;
    }
    return 0;
}
