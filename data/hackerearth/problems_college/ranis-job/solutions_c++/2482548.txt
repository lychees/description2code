#include <iostream>
using namespace std;
#include<vector>
#include<bits/stdc++.h>
#include<algorithm>
#define ll long long
#define ull unsigned long long
#define pb push_back
int main()
{
    ll test,N,i,j,s1,s2,init,max;
    cin>>test;
    while(test--)
    {
    	cin>>N;
    	vector<ll> init1;
    	vector<ll> max1;
    	s1=0;s2=0;
    	for(i=0;i<N;i++)
    	{
    		cin>>init;
    		init1.pb(init);
    		s1+=init;
    	}
    	for(i=0;i<N;i++)
    	{
    		cin>>max;
    		max1.pb(max);
    		s2+=max;
    	}
    //	cout<<s1<<" "<<s2<<endl;
    	sort(max1.begin(),max1.end());
    	ll count=0;
    	for(i=N-1;i>=0;i--)
    	{
    		s1-=(max1[i]);
    		if(s1<=0){count++;break;}
    		else{count++;}
    	}
    	cout<<count<<endl;
    }
    return 0;
}
