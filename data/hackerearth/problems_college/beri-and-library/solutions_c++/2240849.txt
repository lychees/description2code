#include<iostream>
#include<algorithm>
#include<cstdio>
using namespace std;
typedef long long int  lld;

int main()
{
    lld t;
    scanf("%lld",&t);
    while(t--){
    lld n,s=0;
    scanf("%lld",&n);
    while(n)
    {
        s+=n%2;
        n>>=1;
    }
    printf("%lld\n",s);
    }
    return 0;
}
