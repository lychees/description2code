#include <cstdio>
#include <cstring>
#include<iostream>
using namespace std;
int const size = 1e5 + 5;
int x[size], y[size], cnt[size];

int main()
{
    int tests;
    scanf("%d", &tests);
    memset(cnt, 0, sizeof(cnt));
    for(int initial = 0; initial < tests; initial++)
    {
        scanf("%d %d", &x[initial], &y[initial]);
        cnt[x[initial]] ++;
    }
    for(int initial = 0; initial < tests; initial++)
        printf("%d %d\n",  tests - 1 + cnt[y[initial]], tests - 1 - cnt[y[initial]]);
return 0;
}
