#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t; cin>>t;
	while(t--)
	{
	long long int sum=0,n,i,j=1,lsum=0,rsum=0;
	cin>>n; long long a[n];
	if(n>2)
	{
	for(i=0;i<n;i++)
	{
		cin>>a[i];
		sum=sum+a[i];
	}
	for(i=1;i<n-1;i++)
	{
		lsum=lsum+a[i-1];
		rsum=sum-a[i]-lsum;
		if(lsum==rsum)
			{
				cout<<"YES"<<endl;
				j=0;
				break;
			}
	}
	if(j==1)
		cout<<"NO"<<endl;
	
	}
	else
		cout<<"NO"<<endl;
	}
	
}
