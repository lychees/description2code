#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
int main()
{
	int i, j, n, k, pd, ed, fd;
	cin >> k;
	vector<int> nod;
	vector<int>::iterator v;
	for(i = 0; i < k; i++)
	{
		cin >> n;
		nod.push_back(n);
	}
	sort(nod.begin(), nod.end());
	
	pd = 1; fd = 0;
	for(i = nod.size() - 1; i >= 0; i--)
	{
		if(pd + nod.at(i) > fd) fd = pd + nod.at(i);
		pd++;
	}
	cout << fd + 1 << endl;
	return 0;
}
