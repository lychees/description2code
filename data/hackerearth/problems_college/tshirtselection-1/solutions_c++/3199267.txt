#include <iostream>
#include <unordered_map>
#include <map>
#include <utility>

using namespace std;

int main(void) {
	int t;
	cin >> t;
	while(t--) {
		int n;
		cin >> n;
		int a[n];
		int start, end;
		int maxlen = 1;
		start = 0;
		end = 0;
		map<int, int> m;
		
		for(int i = 0; i < n; i++)
			cin >> a[i];
		
		m.insert(make_pair(a[0], 0));
		for(int i = 1, j = 0; i < n; i++) {
			//cout << "For i = " << i << ":" << endl;
			if(m.find(a[i]) == m.end()) {
				m.insert(make_pair(a[i], i));
				if(maxlen < i-j+1) {
					maxlen = i-j+1;
					start = j;
					end = i;
					//cout << "Area 1> Updates - Start: " << start << " And End:" << end << endl;
				}
			}
			else {
				j = max(j, m[a[i]]+1);
				m[a[i]] = i;
				if(maxlen < i-j+1) {
					maxlen = i-j+1;
					start = j;
					end = i;
					//cout << "Area 2> Updates - Start: " << start << " And End:" << end << endl;
				}
			}
		}
		cout << start+1 << ' ' << end+1 << endl;
	}
	return 0;
}
