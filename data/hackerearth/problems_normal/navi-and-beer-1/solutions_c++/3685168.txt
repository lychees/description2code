#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
     long t,n,k,m;
   unsigned long long  ans=0;
    scanf("%ld",&t);
    while(t--)
    {
    	ans=0;
    	scanf("%ld%ld%ld",&n,&k,&m);
    	
    	ans=((n*n*((n+1)*(n+1)))/4)*k;	
    	if(m<ans)
    	cout<<ans-m<<endl;
    	else printf("0\n");
    }
    return 0;
}
