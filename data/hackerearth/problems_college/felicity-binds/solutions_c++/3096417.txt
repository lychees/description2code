#include <bits/stdc++.h>
using namespace std;
int cmpfunc(const void *a, const void *b)
{
	return *(int*)a-*(int *)b;
}
int main()
{
	int t;
	cin>>t;
	for(int i=0;i<t;i++)
	{
		int n;
		cin>>n;
		int men[1005],women[1005];
		for(int j=0;j<n;j++)
			cin>>men[j];
		for(int j=0;j<n;j++)
			cin>>women[j];
		qsort(men,n,sizeof(int),cmpfunc);
		qsort(women,n,sizeof(int),cmpfunc);
		int total=0;
		for(int j=0;j<n;j++)
			total+=(men[j]*women[j]);

		cout<<total<<endl;
	}
	return 0;
}
