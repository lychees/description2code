#include<iostream>
#include<algorithm>
#include<vector>

using namespace std;

class class1
{
    private:
        int n,i,cnt,prize;
        int k,arr[100000];



    public:
        bool cmp(int i,int j)
        {
            return(i<j);
        }
        void input()
        {
            cin>>n>>k;
            for(i=0;i<n;i++)
                cin>>arr[i];
            vector<int> vector1(arr,arr+n);
            sort(vector1.begin(),vector1.end());


         //   for(i=0;i<n;i++)
          //     cout<<vector1[i]<<"\n";
//
               // cout<<vector1[0]<<"\n";
            i=0;
            cnt=0;
            prize=0;

            while(i<n){
                prize=prize+vector1[i];
                if(prize>=k)
                    break;
                else{
                    i+=1;
                    cnt+=1;
                    }

            }

            cout<<cnt;
        }

};

int main()
{
    class1 v1;
    v1.input();
  //  v1.work();
    return 0;
}
