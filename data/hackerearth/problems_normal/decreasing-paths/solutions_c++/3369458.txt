#include<iostream>
#define M 1000000007
#define ll long long
using namespace std;


int dp[1005][1005]={0},n;
int  arr[1000][1000]={0};

ll int solve(int i,int j)
{
    if(dp[i][j]!=0)
        return dp[i][j];

    ll int ans=0;
    if(i!=n-1)
    if(arr[i+1][j]<arr[i][j])
    {
        if(dp[i+1][j]==0)
            dp[i+1][j]=solve(i+1,j);

        ans+=dp[i+1][j];
    }

    if(i!=0)
    if(arr[i-1][j]<arr[i][j])
    {
        if(dp[i-1][j]==0)
            dp[i-1][j]=solve(i-1,j);

        ans+=dp[i-1][j];
    }

    if(j!=n-1)
    if(arr[i][j+1]<arr[i][j])
    {
        if(dp[i][j+1]==0)
            dp[i][j+1]=solve(i,j+1);

        ans+=dp[i][j+1];
    }

    if(j!=0)
    if(arr[i][j-1]<arr[i][j])
    {
        if(dp[i][j-1]==0)
            dp[i][j-1]=solve(i,j-1);

        ans+=dp[i][j-1];
    }

    ans+=1;
    ans%=M;
    dp[i][j]=ans;

    return ans;
}


int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    
    cin>>n;

    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
            cin>>arr[i][j];

    ll int ans=0;
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
        {
            ans+=solve(i,j);
            ans%=M;
        }

    cout<<ans<<endl;
}
