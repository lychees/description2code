#include <bits/stdc++.h>
using namespace std;

typedef unsigned long long ll;
int main()
{
    ll t,low,mid,high,flag,a,b,c,d,k,check,res;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b>>c>>d>>k;
    	low=1;high=cbrt(k)+1;flag=0;res=0;
    	while(low<=high)
    	{
    		mid=(low+high)/2;
    	    check=a*mid*mid*mid+b*mid*mid+c*mid+d;
    	    if(check==k)
    	    {   flag=1;
    	    	cout<<mid<<endl;
    	    	break;
    	    }
    	    else if(check>k)
    	    high=mid-1;
    	    else
    	    {  
    	    	if(res<mid)
    	    	res=mid;
    	   low=mid+1;
    	    }
    	}
    	if(flag==0)
    	cout<<res<<endl;
    }
    return 0;
}
