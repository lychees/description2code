cse = int (raw_input())

lst = ['a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U']

while (cse):
	x = raw_input()
	lst1 = []
	for chr1 in x:
		if (chr1 in lst):
			lst1.append (chr1)
	if (len(lst1) != 0):
		print "".join (ele for ele in lst1)
	else:
		print "No"
	cse -= 1
