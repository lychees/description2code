#include<bits/stdc++.h>
using namespace std;
static char m[100001];
static int suf[100001][26];
int main(){
	int n;
	scanf("%d",&n);
	scanf("%s",m);
	for(int i=0;i<26;i++)
	   suf[0][i]=suf[1][i]=0;
	for(int i=1;i<=n;i++){
			for(int j=0;j<26;j++)
			    suf[i][j]=suf[i-1][j];
		suf[i][m[i-1]-'a']++;
	}
	int q;
	scanf("%d",&q);
	while(q--){
		int l,r;
		scanf("%d %d",&l,&r);
		int b[26];
		for(int i=0;i<26;i++)
		   b[i]=suf[r][i]-suf[l-1][i];
		int c=0,sum=0,e=0;
		for(int i=0;i<26;i++){
			if(b[i]%2!=0)
			 c++;
			if(b[i]>0)
			     e++;
			sum+=b[i];
		}
		if(c>2||sum<2||(e==1))
		printf("NO\n");
		else
		printf("YES\n");
	}
	return 0;
}
