#include <iostream>
using namespace std;

void printEven(int *n, int N)
{
	bool none = true;
	for (int i = N-1; i >= 0; i--)
	{
		if (n[i] % 2 == 0)
		{
			cout << n[i] << " ";
			none = false;
		}
	}
	if (none == true) cout << "None to display";
	cout << endl;
}

int main()
{
	int N;
	int *n;
	while (cin >> N)
	{
		n = new int[N];
		for (int i = 0; i < N; i++)
		{
			cin >> n[i];
		}
		printEven(n, N);
	}
}
