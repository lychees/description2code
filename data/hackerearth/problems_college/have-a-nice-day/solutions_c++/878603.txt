#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<string.h>
#define ll long long int
using namespace std;
int main()
{ unsigned int T;
unsigned int N;
int i;
cin>>T;
ll sum;

while(T--)
{cin>>N;
int a[N],b[N];
sum=0;
for(i=0;i<N;i++)
{ cin>>a[i];

}
sort(a,a+N);
for(i=0;i<N;i++)
{ cin>>b[i];

}
sort(b,b+N);
for(i=0;i<N;i++)
{ for(i=0;i<N;i++)
sum=sum+(a[i]*b[i]);
}
cout<<sum<<endl;
}
}
