#include<iostream>
#include<algorithm>
#include<set>
using namespace std;

set< int >s;
set< int >::iterator it;

void calc() {
    int temp;
    s.insert(1);
    for(int i=2;i*i<=1000000;i++) {
        s.insert(i*i);
        temp=i*i;
        while(i*temp<=1000000) {
            s.insert(i*temp);
            temp*=i;
        }
    }
}

int main()
{
    int t;
    cin>>t;
    calc();
    while(t--) {
        int x;
        bool found=false;
        cin>>x;
        for(it=s.begin();it!=s.end() && !found;it++) {
            if(s.find(x-(*it))!=s.end()) found=true;
        }
        if(found) cout<<"Yes\n";
        else cout<<"No\n";
    }
    return 0;
}
