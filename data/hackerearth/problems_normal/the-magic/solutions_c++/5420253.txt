#include <bits/stdc++.h>

#define REP(i,n) for(int i=0;i<(int)n;i++)
#define FOR(i,a,b) for(int i=(int)a;i<=(int)b;i++)
#define pb push_back
#define mp make_pair

using namespace std;

typedef long long int ll;

int countOne(int n){
	int res = 0;
	while (n){
		n = n&(n - 1);
		res++;
	}
	return res;
}

int main()
{
	int t;
	scanf("%d", &t);
	while (t--){
		int n;
		scanf("%d", &n);
		int ans = countOne(n);
		printf("%d\n", ans);
	}
	return 0;
}
