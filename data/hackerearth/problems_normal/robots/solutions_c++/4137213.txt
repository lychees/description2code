#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
	int t, k, i, n;
	
	cin>>t;
	
	for(k = 1; k <= t; k++)
	{
		cin>>n;
		vector<int> maxx(n);
		long long ways = 1;
		
		for(i = 0; i < n; i++)
		{
			cin>>maxx[i];
		}
		
		sort(maxx.begin(), maxx.end());
		
		for(i = 0; i < n; i++)
		{
			ways *= (maxx[i]-i);
			ways = ways % 1000000007;
		}
		
		cout<<ways<<endl;
	}
}
