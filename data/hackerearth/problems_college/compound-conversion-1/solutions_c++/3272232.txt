#include<iostream>
using namespace std;
char s1[105],s2[105];
int test,i,qo,qz,oz,zo,ans;
int main(){
	cin>>test;
	for(int c=1;c<=test;c++){
		cin>>s1>>s2;
		qo=qz=oz=zo=i=0;
		while(s1[i]){
			if(s1[i]=='?' && s2[i]=='0')
				qz++;
			else if(s1[i]=='?' && s2[i]=='1')
				qo++;
			else if(s1[i]=='1' && s2[i]=='0')
				oz++;
			else if(s1[i]=='0' && s2[i]=='1')
				zo++;
			i++;
		}
		
		cout<<"Case "<<c<<": ";
		if(zo<oz)
			if((oz-zo)>qo)
				ans=-1;
			else
				ans=zo+2*(oz-zo)+(qo-(oz-zo))+qz;
		else
			ans=(zo+qz+qo);
		cout<<ans<<"\n";
	}
}
