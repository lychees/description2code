#include<bits/stdc++.h>
using namespace std;
# define ll long long
# define mod 1000000007
inline ll bp(ll base,ll exp)
{
    ll ret=1;
    while(exp)
    {
        if(exp&1)
            ret=(ret*base)%mod;
        exp>>=1;
        base=(base*base)%mod;
    }
    return ret;
}
ll pre[20],p,pmax,pmin,q,mx,mn;
void sets(int n)
{
    int cs=bp(2,n)-1,i,tmp,sz;
    while(cs)
    {
        vector <int> v;
        tmp=cs;
        for(i=0;i<n;i++)
        {
            if(tmp&1)
            v.push_back(i);
            tmp>>=1;
        }
        q=1;
        for(i=0;i<v.size();i++)
            q=(q*pre[v[i]])%mod;
        p=q*bp((ll)v.size(),mod-2);
        p%=mod;
        if(pmax<p)
        {
            pmax=p;
            mx=(ll)v.size();
        }
        else if(pmax==p)
        mx=min(mx,(ll)v.size());
        if(pmin>p)
        {
            pmin=p;
            mn=(ll)v.size();
        }
        else if(pmin==p)
        mn=max(mn,(ll)v.size());
        cs--;
    }
}
int main()
{
    cin.sync_with_stdio(false);
    cin.tie(0);
    int t,n,k,i;
    ll x;
    cin>>t;
    while(t--)
    {
        cin>>n>>k;
        for(i=0;i<n;i++)
            pre[i]=0;
        while(k--)
            for(i=0;i<n;i++)
            {
                cin>>x;
                pre[i]+=x;
            }
        pmax=mx=INT_MIN;
        pmin=mn=LONG_LONG_MAX;
        sets(n);
        ll o1,o2;
        o1=pmax^pmin;
        o2=mx^mn;
        cout<<o1<<" "<<o2<<'\n';
    }
    return 0;
}
