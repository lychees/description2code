#include <iostream>
using namespace std;
#define ll long long
#define mod 1000000007
int main()
{
	ll i,t,n,k;
	cin>>t;
	ll a[100002];
	a[1] = 1;
	a[2] = 2;
	a[3] = 4;
	for(i = 4; i <= 100000; i++) {
		a[i] = (a[i-1]+a[i-2]+a[i-3])%mod;
	}
	while(t--) {
		cin>>n;
		cout<<a[n]%mod<<endl;
	}
    return 0;
}
