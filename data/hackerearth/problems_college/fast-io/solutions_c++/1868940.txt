#include<cstdio>
#include<cstdio>
# define gc getchar
long long input() {
  char c = gc();
  while(c<'0' || c>'9') c = gc();
  long long ret = 0;
  while(c>='0' && c<='9') {
    ret = 10 * ret + c - 48;
    c = gc();
  }
  return ret;
}

int main(){
	long long t,n,c=0,MAX=300000000;
	t=input();
	while(t--){
		n=input();
		if(n<MAX)
		c++;
	}
	printf("%lld\n",c);
}
