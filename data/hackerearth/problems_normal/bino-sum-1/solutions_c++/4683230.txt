#include <iostream>
#include<bits/stdc++.h>
#define MAX 1005
#define MOD 1000000007
using namespace std;



int rec(int n,int r,vector<vector<int> > &dp){
//	cout<<n<<"   dc"<<r<<endl;
	if(r==0 || r==n){
		dp[n][r]=1;
		return 1;
	}
	if(n<0 || r<0){
//		cout<<"y";
		dp[n][r]=0;
		return 0;
	}
	if(dp[n][r]!=-1){
//		cout<<"x"<<dp[n][r];
		return dp[n][r];
	}
//	cout<<"z";
	dp[n][r]=rec(n-1,r,dp)+rec(n-1,r-1,dp);
	if(dp[n][r]>MOD){
		dp[n][r]=dp[n][r]%MOD;
	}
//	cout<<dp[n][r]<<"here"<<r<<endl;
	return dp[n][r];
}


int main()
{
	vector<vector<int> >dp(MAX,vector<int>(MAX,-1));    
    int t;
    cin>>t;
    while(t){
    	int n,p;
    	cin>> n >> p;
    	
    	for(int i=0;i<=p;i++){
    		//cout<<rec(n,i,dp)<<"hj"<<endl;
    		rec(n,i,dp);
    	}
    	
    	int res=0;
    	for(int i=0;i<=p;i++){
    		//cout<<dp[n][i]<<" "<<i<<endl;
    		res=res+(dp[n][i]);
    		if(res>MOD){
    			res=res%MOD;
    		}
    	}
    	cout<<res<<endl;
    	t--;
    }
    
    return 0;
}
