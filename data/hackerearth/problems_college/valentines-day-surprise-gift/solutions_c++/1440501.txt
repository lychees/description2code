#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;

long long mod = 1000000007;

long long int R[1000010];
long long int P[1000010];

long long int po(int l)
{
    long long ret = 1;
    long long t = 2;
    
    while(l)
    {
        if(l%2)
        {
            ret *= t;
            if(ret>=mod)
                ret %= mod;
        }
        t *= t;
        if(t>=mod)
            t %= mod;
        l /= 2;
        
        
    }
    return ret;
}

long long int po2(long long int t,long long int l)
{
    long long ret = 1;
    
    while(l)
    {
        if(l%2)
        {
            ret *= t;
            if(ret>=mod)
                ret %= mod;
        }
        t *= t;
        if(t>=mod)
            t %= mod;
        l /= 2;
        
        
    }
    return ret;
}

void pre()
{
    P[0] = 1;
    for(int i=1;i<1000010;i++)
    {
        P[i] = 2*P[i-1];
        if(P[i]>=mod)
            P[i]-=mod;
    }
    
    R[0] = 1;
    R[1] = 1;
    R[2] = 1;
    R[3] = 3;
    R[4] = 3;
    R[5] = 15;
    R[6] = 15;
    R[7] = 105;
    R[8] = 105;
    R[9] = 105;
    R[10] = 105;
    
    for(int i=11;i<=1000000;i++)
    {
        long long tmp;
        int idx = i/4;
        if(i%4)
            idx++;
        tmp = P[idx];
        
        idx = i/5;
        if(i%5)
            idx++;
        tmp *= R[idx];
        if(tmp>=mod)
            tmp%=mod;
        idx = i/10;
        if(i%10)
            idx++;
        R[i] = (tmp*R[idx])%mod;
    }
}

long long cal(int idx)
{
    if(idx<=1000000)
        return R[idx];
    
    long long tmp;
    long long int i = idx/4;
    if(idx%4)
        i++;
    tmp = po(i);
    
    i = idx/5;
    
    if(idx%5)
        i++;
    tmp *= cal(i);
    tmp %= mod;
    
    i = idx/10;
    
    if(idx%10)
        i++;
    tmp *= cal(i);
    tmp %= mod;
    
    return tmp;
    
}


int main() {
    
    int T;
    pre();
    
    cin>>T;
    while(T--)
    {
        int N;
        cin>>N;
        long long r = cal(N);
        long long x,y;
        long long mx = 0;
        for(long long int i=1;i<=5;i++)
        {
            x = i*r;
            for(long long int j=1;j<=5;j++)
            {
               if(__gcd(i,j)==1)
               {
                   y = j*r;
               long long tmp = po2(x,y);
                if(tmp>mx)
                    mx = tmp;
               }
            }
        }
        cout<<mx<<endl;
    }
    
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */   
    return 0;
}
