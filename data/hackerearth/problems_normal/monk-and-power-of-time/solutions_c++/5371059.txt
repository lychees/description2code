#include<bits/stdc++.h>
using namespace std;
int main(){
	int n;
	cin>>n;
	queue<int>real;
	queue<int>ideal;
	for(int i=0;i<n;i++){
	    int k;cin>>k;real.push(k);	
	}
	for(int i=0;i<n;i++){
	    int k;cin>>k;ideal.push(k);	
	}
	int tt;
	while(!ideal.empty()){
		if(real.front()==ideal.front()){
			real.pop();
			ideal.pop();
		}
		else{
			int kk=real.front();
			real.pop();
			real.push(kk);
		}
		tt++;
	}
	cout<<tt<<endl;
}
