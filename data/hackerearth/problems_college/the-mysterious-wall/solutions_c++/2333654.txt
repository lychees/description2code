using namespace std;
bool check(int a[26],int b[26])
	{
	for(int i=0;i<26;i++)
		{
		if(a[i]!=b[i])
			return false;
		}
	return true;
	}
int main()
	{
	int i,j,k;
	int T;
	ios_base::sync_with_stdio(false);
	cin>>T;
	string s1,s2;
	while(T--){
		cin>>s1>>s2;
		int len=s2.length();
		int req[26]={0};	
		if(s1.length()<s2.length())
			{
			printf("NO\n");
			continue;
			}
		int cur[26]={0};
		for(i=0;s2[i];i++){
			req[s2[i]-'a']++;
			cur[s1[i]-'a']++;
		}
		if(check(req,cur))
			{
			printf("YES\n");
			continue;
			}
		for(i=len;s1[i];i++)
			{
			cur[s1[i-len]-'a']--;
			cur[s1[i]-'a']++;
			if(check(req,cur))
				{
				printf("YES\n");
				goto end;
				}
			}
			printf("NO\n");
			end:
			i=1;
	}
	return 0;	
	}
