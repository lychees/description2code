#include <iostream>
#include<cstdio>
#include<cstring>
using namespace std;

int main()
{
	int t,i,l;
	char a[105],b[105],c[105];
	scanf("%d",&t);
	while(t--)
	{
	 scanf("%s",&a);
	 scanf("%s",&b);
	 l=strlen(a);
	 for(i=0;i<l;i++)
	 {
		if(a[i]==b[i])
		c[i]='0';
		else
		c[i]='1';
	 }
	 c[i]='\0';
	 printf("%s\n",c);
	}
    return 0;
}
