#include <cstdio>
#include <cassert>
#include <algorithm>
#include <iostream>
#include <vector>
#include <set>
using namespace std;
int main() {
    int n, m;
    cin>>n>>m;
    assert(1 <= n && n <= 100000);
    assert(1 <= m && m <= 100000);
    int ans = 1E9;
    for (int i = 1; i <= 1E6; i++) {
        for (int j = 0; j <= 1E6; j += i) {
            if (j % i == 0) {
                ans = min(ans, abs(m - i) + abs(j - n) );
            }
        }
    }
    cout<<ans<<endl;
    return 0;
}
