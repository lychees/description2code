#include <iostream>
using namespace std;

int rev(int n)
{
	int sum=0;
	while(n>0)
	{
		sum=(sum*10)+(n%10);
		n/=10;
	}
	return sum;
}

int main()
{
    //printf("Hello World!\n");
    int t;
    cin>>t;
    for(;t--;)
    {
    	int a,b,sum=0,x,y;
    	cin>>a>>b;
    	x=rev(a);
    	y=rev(b);
    	sum=x+y;
    	cout<<(rev(sum))<<endl;
    }
    return 0;
}
