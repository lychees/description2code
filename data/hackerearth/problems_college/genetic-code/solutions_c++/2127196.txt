#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    string s;
    cin>>s;
    int n=0,sum=0;
    for(int i=0;i<s.length()-1;i++)
    {
    	n++;
    	if(s[i]!=s[i+1])
    	{
    		if(n>0&&n%2==0)
    		sum++;
    		n=0;
    	}
    }
    if(n>0&&(n+1)%2==0)
    	sum++;
    cout<<sum<<endl;
    return 0;
}
