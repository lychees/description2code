#include <iostream>
#include <cmath>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long z,m,n,ans=0;
    	cin>>z>>m>>n;
    	int y=sqrt(z);
    	for(long long i=1;i<=y && i<=m;i++)
    	{
    		if(z%i==0)
    		{
    			long long temp=z/i;
    			if(temp<=n)
    				ans++;
    			if(i<=n  && temp<=m && i!=temp)
    				ans++;
    			
    		}
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
