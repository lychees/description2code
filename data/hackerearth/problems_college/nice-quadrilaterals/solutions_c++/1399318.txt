#include<bits/stdc++.h>
using namespace std;

typedef unsigned long long ll;

int main() {
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        int n;
        scanf("%d",&n);
        if(n<4) {
            puts("0");
            continue;
        }
        if(n==4) {
            puts("1");
            continue;
        }
        int res1=n;
        int res2=(n-5);
        ll ans=0;
        ans+=1ll*res1*res2;
        if(res1%2==0) {
            res1/=2;
        } else {
            res2/=2;
        }
        ans+=1ll*res1*res2;
        ans+=1ll*n;
        cout<<ans<<endl;
    }
    return 0;
}
