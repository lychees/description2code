#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
    int n;
    cin>>n;
    while(n--){
    	ll a,b,c;
    	cin>>a>>b>>c;
    	ll ans=pow(a,b);
    	if(ans==0){
    		cout<<"0"<<endl;
    		continue;
    	}
    	ll t=ans/c;
    	ll a1=t*c;
    	ll a2=(t+1)*c;
    	if(ans>c){
    		if(abs(ans-a1)<=abs(ans-a2)){
    			cout<<a1<<endl;
    		}else{
    			cout<<a2<<endl;
    		}
    	}else{
    		if(abs(ans)<=abs(c-ans)){
    			cout<<"0"<<endl;
    		}else{
    			cout<<c<<endl;
    		}
    	}
    }
    return 0;
}
