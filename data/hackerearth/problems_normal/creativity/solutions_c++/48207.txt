#include<iostream>
#include<string>
#include<cstdio>
 
using namespace std;
 
int main(){
    
    string s1, s2;
    int t;
    scanf("%d", &t);
    getline(cin, s1);
    while(t--){
        getline(cin, s1);
        getline(cin, s2);
    //    cout << s1 << endl << s2 << endl;
        if(s1.length() > s2.length()){
            if(s1.find(s2) == string::npos)
                cout << "not possible" << endl;
            else
                cout << "possible" << endl;
        }
        else{
            if(s2.find(s1) == string::npos)
                cout << "not possible" << endl;
            else
                cout << "possible" << endl;
        }
    }
    
    return 0;
}
