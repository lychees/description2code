#include <iostream>
#include <string>
#include <algorithm>
#include <cstring>
using namespace std;

int main()
{
	int t;
	int cnt[30];
	string s1, s2;
    
    cin >> t;
    cin.ignore();
    while (t--)
    {
    	memset(cnt, 0, sizeof(cnt));
    	getline(cin, s1);
    	getline(cin, s2);
    	if (s1.length() < s2.length())
    		s1.swap(s2);
    	
		for (int i = 0; i < s1.length(); ++i)
		{
			if (s1[i] >= 'A' && s1[i] <= 'Z')
				cnt[s1[i] - 'A']++;
			else if (s1[i] >= 'a' && s1[i] <= 'z')
				cnt[s1[i] - 'a']++;
		}
    	for (int i = 0; i < s2.length(); ++i)
    	{
    		if (s2[i] >= 'A' && s2[i] <= 'Z')
    			cnt[s2[i] - 'A']--;
    		else if (s2[i] >= 'a' && s2[i] <= 'z')
    			cnt[s2[i] - 'a']--;
    	}
    	
    	int cnt_odd = 0, cnt_nzero = 0;
    	string front, back;
    	int center = 26;
    	front = back = "";
    	for (int i = 0; i < 26; ++i)
    	{
    		if (cnt[i] % 2)
    		{
    			center = i;
    			++cnt_odd;
    			++cnt_nzero;
    			if (cnt[i] > 1)
    			{
    				int len = cnt[i] / 2;
    				for (int j = 0; j < len; ++j)
    				{
    					front += char(i + 'a');
    					back = char(i + 'a') + back;
    				}
    				cnt[i] = 1;
    			}
    		}
    		else
    		{
    			int len = cnt[i] / 2;
    			for (int j = 0; j < len; ++j)
    			{
    				front += char(i + 'a');
    				back = char(i + 'a') + back;
    			}
    			cnt_nzero += (len > 0);
    		}
    	}
    	if (cnt_nzero == 0)
    		cout << "YES" << endl;
    	else if (cnt_odd > 1)
    		cout << "NO LUCK" << endl;
    	else
    	{
    		cout << front;
    		for (int i = 0; i < cnt[center]; ++i)
    			cout << char(center + 'a');
    		cout << back << endl;
    	}
    }
    
    return 0;
}
