#include <iostream>
#include <cmath>
using namespace std;
long long divisors(long long n)
{
	long long div=0;
	for(int i=1;i<=sqrt(n);i++)
	{
		if(n%i==0)
		{
			if(i==n/i)
			div++;
			else
			div=div+2;
		}
	}
	return div;
}

int main() {
	// your code goes here
	int t;
	cin>>t;
	while(t--)
	{
		long long n;
		cin>>n;
		bool b=true;
		long long k=0,temp=1;
		//cout<<divisors(3)<<endl;
		while(b)
		{
			k++;
			temp=k*(k+1)/2;
			if(divisors(temp)>n)
			b=false;
		}
		cout<<temp<<endl;
	}
}
