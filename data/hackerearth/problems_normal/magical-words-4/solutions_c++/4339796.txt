#include <iostream>
#include <string>
using namespace std;

bool is_pal(string a,int i,int j){
	int k,mid=(j-i+1)/2;
	for(k=0;k<mid;k++){
		if(a[i+k]!=a[j-k]) return 0;
	}
	return true;
}

int main()
{
    int T;
    cin>>T;
    while(T--){
    	string a;
    	cin>>a;
    	
    	long long int i,j,k,str=0;
    	
    	for(i=0;i<a.size();i++){
    		for(j=i,k=i;j>=0,k<a.size();j--,k++){
    			if(a[j]==a[k]) str+=(k-j+1)*(k-j+1);
    			else break;
    		}
    	}
    	for(i=0;i<a.size()-1;i++){
    		for(j=i,k=i+1;j>=0,k<a.size();j--,k++){
    			if(a[j]==a[k]) str+=(k-j+1)*(k-j+1);
    			else break;
    		}
    	}
    	cout<<str<<endl;
    }
    return 0;
}
