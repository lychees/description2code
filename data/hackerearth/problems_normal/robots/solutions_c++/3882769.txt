#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    
    int t;
    cin>>t;
    int n;
    while(t--){
    	cin>>n;
    	int arr[n];
    	int i;
    	for(i=0;i<n;i++)
    	cin>>arr[i];
    	sort(arr,arr+n);
    	long long int ans=1,p=1;
    	for(i=0;i<n;i++)
    	{	
    		int k=arr[i]-i;
    		if(k<=0)
    		{
    			ans=0;
    			break;
    		}
    		else {
    			ans=((k*p)%1000000007);
    			ans%=1000000007;
    			p=ans;
    		}
    		
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
