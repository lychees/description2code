#include <iostream>
using namespace std;

int main()
{
	long lengthofarray = 0;
	cin >>lengthofarray;
	long arr[lengthofarray];
	for(long i=0;i<lengthofarray;i++)
	{
		cin>>arr[i];
	}
	long noofdivisors=0;
	bool allzero = false;
	int count = 0;
	cin>>noofdivisors;
	for(long i=0;i<noofdivisors;i++)
	{
		long divisor =0;
		cin >> divisor;
		if(divisor>1 && !allzero)
		{
			for(long j=0;j<lengthofarray ;j++)
			{
				arr[j]=arr[j]/divisor;
				if(arr[j] == 0)
				{
					count++;
				}
			}
			if(count==lengthofarray)
			{
				allzero = true;
			}else
			{
				count=0;
			}
		}
	}
	for(long i=0;i<lengthofarray;i++)
	{
		cout<<arr[i]<<" ";
	}
    cout << endl;
    return 0;
}
