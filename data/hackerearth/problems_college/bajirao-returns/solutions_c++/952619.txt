#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
using namespace std;
int p[101][101];
int a[101][101];
int dp[101][101];
void recur(int i,int j,int n) {
    if(j<1||j>n) return;
    a[i][j]=1;
    if(a[i][2*j]==0) recur(i,2*j,n);
    if(a[i][j/2]==0) recur(i,j/2,n);
    return;
}
int main() {
    int n;
    scanf("%d",&n);
    for(int i=1;i<=n;i++) {
        for(int j=1;j<=n;j++) {
            scanf("%d",&p[i][j]);
            dp[i][j]=0;
            if(i==1) dp[i][j]=p[i][j];
        }
        recur(i,i,n);
    }
    for(int i=2;i<=n;i++) {
        for(int j=1;j<=n;j++) {
            for(int k=1;k<=n;k++) {
                if(a[k][j]) if(dp[i-1][k]+p[i][j]>dp[i][j]) dp[i][j]=dp[i-1][k]+p[i][j];
            }
        }
    }
    int ans=dp[n][1];
    for(int j=2;j<=n;j++) if(dp[n][j]>ans) ans=dp[n][j];
    printf("%d\012",ans);
    return 0;
}
