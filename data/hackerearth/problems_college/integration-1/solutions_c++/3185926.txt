#include <iostream>
using namespace std;


int main()
{
	long int N,P,U,R,S,T;
	long int i,j,k,l;
	long int res;
	
	cin>>T;
	for(i=0;i<T;i++){
		cin>>N>>P>>U>>R>>S;
		j=0;
		while(j<N){
			if(P+R <= U){
				P= P+R;
				j++;
			}
			else {
				if(P-S >= 0){
					P = P-S;
					j++;
				}
				else{
					j = N;
				}
			}
		}
		cout<<"Case #"<<i+1<<": "<<P<<"\n";

	}
    return 0;
}
