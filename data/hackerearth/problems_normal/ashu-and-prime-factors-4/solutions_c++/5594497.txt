
#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
    ll a[1000000]={0};
    
    for(long i=1;i<1000000;i++)
    {
        if(a[i]==0)
        {
            ll s=0;
            for(long j=i;j<1000000;j+=(i+1))
            {
                if(!a[j])
                {
                    a[j]=1;
                    s++;
                }
            }
            a[i]=s;
        }
    }
    long t;
    cin>>t;
    while(t--)
    {
        ll n;
        cin>>n;
        cout<<a[n-1]<<endl;
    }
    return 0;
}
