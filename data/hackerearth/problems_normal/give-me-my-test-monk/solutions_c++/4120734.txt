#include<bits/stdc++.h>
using namespace std;
#define MOD 1000000007
long long ncr[109][109],deranged[109];
long long power(long long a,long long b);


long long power(long long a,long long b)
{
    long long pw = 1;
    while(b!=0)
    {
        if(b%2==1)
            pw = ((a*pw)%MOD+MOD)%MOD;
        a = ((a*a)%MOD+MOD)%MOD;
        b/=2;
    }

    return pw;
}
int main()
{
    int N,L,R,P=0;
    long long ans,Q = 1;
    cin>>N>>L>>R;
    for(int i=1;i<=N;i++)
    {
        Q = ((Q*i)%MOD+MOD)%MOD;
        for(int j=0;j<=i;j++)
        {
            if(j==0 || i==j)
                ncr[i][j] = 1;
            else
                ncr[i][j] = ((ncr[i-1][j-1]+ncr[i-1][j])%MOD+MOD)%MOD;
        }
    }

    deranged[0] = 1;
    deranged[1] = 0;
    for(int i=2;i<=N;i++)
        deranged[i] = (((i-1)*((deranged[i-1]+deranged[i-2])%MOD))%MOD+MOD)%MOD;

    for(int i=L;i<=R;i++)
    {
        P = (P+((deranged[i]*ncr[N][i])%MOD+MOD)%MOD)%MOD;
    }

    ans = power(Q,MOD-2)%MOD;
    ans = ((ans*P)%MOD+MOD)%MOD;
    cout<<ans;
}
