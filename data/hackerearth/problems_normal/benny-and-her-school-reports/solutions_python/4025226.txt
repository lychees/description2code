def min_marks(marks, n, m, x):
    s = sum(marks)
    req = (n+1)*x - s
    if req > m:
        return None
    if req < 1:
        return 1
    return req

if __name__ == "__main__":
    t = input()
    while t > 0:
        n, m, x = map(int, raw_input().strip().split())
        marks = map(int, raw_input().strip().split())
        req = min_marks(marks, n, m, x)
        if req is None:
            print 'Impossible'
        else:
            print req
    	t -= 1
