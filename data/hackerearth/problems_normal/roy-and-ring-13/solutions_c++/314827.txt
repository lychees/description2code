ll fib(ll n)
{
  ll k,a,b;
  if(n<=2)return 1;
  k=n/2;
  a=fib(k+1);
  b=fib(k);
  if(n%2==1)return (a*a)%MOD+(b*b)%MOD;
  else return (b*(2*a-b))%MOD;
}
