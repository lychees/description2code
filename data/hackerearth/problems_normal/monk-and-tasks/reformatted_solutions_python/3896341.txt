from operator import itemgetter
for i in range(input()):
    a = input()
    s = ""
    l = [int(x) for x in raw_input().split()]
    l1 = []
    for j in range(0, len(l)):
        t = str(bin(l[j]))
        l1.append((l[j], (t.count('1'))))
    l1.sort(key=itemgetter(1))
    for m in l1:
        s += str(m[0])+' '
    print(s)
