#include<iostream>
#include<cmath>
#include<algorithm>
using namespace std;
string s[100001];

int main()
{
    long long t,n,i,count;
    
    cin>>t;
    for(i=0;i<t;i++)
    {
        cin>>s[i];
    }
    sort(s,s+t);
    for(i=0;i<t;)
    {
        cout<<s[i]<<" ";
        string prev=s[i];
        i++;
        count=1;
        while(prev==s[i])
        {
          count++;
          i++;
        }
        cout<<count<<endl;
    }
    
}
