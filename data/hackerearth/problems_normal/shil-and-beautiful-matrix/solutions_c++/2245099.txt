#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
long long fact[1000001];
int main()
{

 fact[0]=1;
 for(int i = 1 ; i <= 1000000 ; ++i )
   fact[i]= ( i * fact[i-1])%mod;
    int n , m;
    cin >> n >> m;
     int ar[n+1][m+1];
     set<int> s;
     map<int,int> mymap;
    for(int i = 1;  i<= n ;++i ){
      for(int j = 1; j <= m ; ++j )
{
          ar[i][j] = (i^j);
          mymap[i^j]++;
          s.insert((i^j));
}
    }
    long long ans = 1;
     for(set<int>::iterator it = s.begin() ; it!= s.end() ; it++)
    {
     ans  = ( ans * fact[mymap[*it]])%mod;
    }
    cout <<ans << endl;

}
