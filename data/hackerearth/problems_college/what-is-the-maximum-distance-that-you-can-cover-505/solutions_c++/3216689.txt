#include <iostream>
using namespace std;
long long int sp[100000],fl[100000];
int main()
{
    long long int i,j,k,n,m,t,tmp;
    double ans,vl;
    cin>>n;
    for(i=0;i<n;i++)scanf("%lld",&sp[i]);
    for(i=0;i<n;i++)scanf("%lld",&fl[i]);
    cin>>m;
    ans=-1;
    for(i=0;i<n;i++){
    	vl = (double)m/fl[i];
    	vl = vl*sp[i];
    	if(vl>ans)ans=vl;
    }
    tmp = ans*1000;
    printf("%0.3lf",(double)tmp/1000);
    return 0;
}
