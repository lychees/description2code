#include <bits/stdc++.h>

using namespace std;

#define pb push_back
#define mp make_pair
#define ll long long int

#define mod 1000000007
#define MX 100010
#define MX1 1001
#define get getchar//_unlocked


int isP[MX], cnt[MX];

void sieve() {
    int i, j;
    for(i = 2; i < 100005; i++) {
        if(!isP[i]) {
            isP[i] = i;
            for(j = 1; j * i < 100005; ++j) {
                isP[i*j] = i;
            }
        }
    }
}

int main()
{
    int n, q, i, j, k;
    sieve();
    cin >> n;
    for(i = 0; i < n; i++) {
        cin >> j;
        while(j > 1) {
            ++cnt[isP[j]];
            j /= isP[j];
        }
    }
    cin >> q;
    int c, tmp, minn;
    while (q--) {
        cin >> j;
        minn = 9999999999;
        while(j > 1) {
            c = 0;
            tmp = isP[j];
            while (isP[j] == tmp) {
                ++c;
                j /= isP[j];
            }
            minn = min(minn, cnt[tmp] / c);
        }
        cout << minn << endl;
    }

    return 0;
}
