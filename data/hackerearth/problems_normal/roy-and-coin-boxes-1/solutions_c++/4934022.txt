#include<bits/stdc++.h>
#define ll long long int
#define INF 0x3f3f3f3f
#define pb push_back
using namespace std;
const int mx=1e6+6;
int a[mx],cnt[mx];

int main()
{
	ios_base::sync_with_stdio(false); cin.tie(0);
	int i,m,n,l,r; cin>>n>>m;
	while(m--)
	{
		cin>>l>>r;
		a[l]++;
		a[r+1]--;
	}
	for(i=1;i<=n;i++) 
	{
		a[i]+=a[i-1];
		cnt[a[i]]++;
	}
	for(i=1;i<=mx;i++) cnt[i]+=cnt[i-1];
	int q;cin>>q;
	while(q--)
	{
		int x;cin>>x;
		cout<<n-cnt[x-1]<<endl;
	}
	return 0;
}
