num = int(input())
flag = "False"
for l in range(num):
	S = raw_input()
	R = S[::-1]
	

	
	for i in range(1,len(S)):
		one = ord(S[i])-ord(S[i-1])
		two = ord(R[i])-ord(R[i-1])
		
		#if ((abs(ord(S[i])-ord(S[i])-1)) == abs(ord(R[i])-ord(R[i])-1))):
		if (abs(one) == abs(two)):
			flag = "True"
		else:	
			flag = "False"
			break
	if flag == "True":
		print "Funny"
	else:
		print "Not Funny"
