#include <iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
// simple case of solving equations x1+x2/2=X1,y1+y2/2=Y1    , x2+x3/2=X2,y2+y3/2=Y2  ,x1+x3/2=X3,y1+y3/2=Y3  solve them
int main()
{
    double x1,y1,x2,y2,x3,y3; //mid points
    double X1,Y1,X2,Y2,X3,Y3; //actual vertices
    scanf("%lf %lf %lf %lf %lf %lf",&X1,&Y1,&X2,&Y2,&X3,&Y3);
    x1=X1+X3-X2;
    x2=X1+X2-X3;
    x3=X2+X3-X1;
    y1=Y1+Y3-Y2;
    y2=Y1+Y2-Y3;
    y3=Y2+Y3-Y1;
    pair<double,double> p[3];
    p[0]=make_pair(x1,y1);
    p[1]=make_pair(x2,y2);
    p[2]=make_pair(x3,y3);
    sort(p,p+3);
    for(int i=0;i<3;i++)
    {
        printf("%.4lf %.4lf\n",p[i].first,p[i].second);
    }
    return 0;
}
