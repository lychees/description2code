tc = input()
lr = []
while tc > 0:
    n = input()
    st = raw_input()
    ls = st.split()
    ln = []
    for i in ls:
        ln.append(int(i))
    lc = []
    count = 0
    for i in ln:
        t = i % 10
        if t in lc:
            count = count+1
        else:
            lc.append(t)
    lr.append(count)
    tc = tc-1
for i in lr:
    print i
