#include<iostream>
#include<string.h>
#include<stdio.h>
using namespace std;
int main()
{
    char s[1000000];
     int t,len;
     unsigned long long int a,b;
     scanf("%s",&s);
    len=strlen(s);
    cin>>t;
    while(t--){
    cin>>a>>b;
    if(a>len)
    		a=(a-1)%len;
    	if(b>len)
    		b=(b-1)%len;
    if(s[a]==s[b])
        cout<<"Yes\n";
        else
        cout<<"No\n";
    }
    return 0;
}
