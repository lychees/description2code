#include<bits/stdc++.h>
#define pii pair<int,int>
#define fi first
#define se second
#define mp make_pair
#define vpi vector< pii >
#define pb push_back
#define ll long long int
#define mod 1000000007
#define gcd(a,b) __gcd(a,b)
#define sf(n) scanf("%ld",&(n))
#define pf(n) printf("%ld\n",(n))
#define setbits(n) __builtin_popcount((n))
#define pq priority_queue
using namespace std;
int main()
{
	int n,m;
	sf(n);sf(m);
	int arr[2*n];
	for(int i=0;i<2*n;i++)
	{
		sf(arr[i]);
	}
	sort(arr,arr+2*n);
	int sum=0;
	int cou=0;
	int ans=0;
	for(int i=0;i<2*n;i++)
	{
		if(sum+arr[i]<=m)
		{
			sum+=arr[i];
			cou++;
		}
		else
		{
			ans=max(ans,cou);
			sum=arr[i];
			cou=1;
		}
	}
	ans=max(ans,cou);
	ans=min(ans,n);
	pf(ans);

}
