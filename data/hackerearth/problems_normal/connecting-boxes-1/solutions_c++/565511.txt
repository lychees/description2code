int main()
{
    ll n,q;
    sl(n);
    sl(q);
    vector <pll> vec;
    forall(il,0,q)
    {
        ll ti,tj;
        sl(ti);
        sl(tj);
        if(ti==tj)
        {
            cout<<"NO"<<endl;
            continue;
        }
        ll i=miN(ti,tj);
        ll j=maX(ti,tj);
        ll len=szll(vec);
        ll fl=0;
        forall(jl,0,len)
        {
            ll f=vec[jl].first;
            ll s=vec[jl].second;
            if(f==i||f==j||s==i||s==j)
            {
                fl=1;
                break;
            }
            else if(!((i<f&&j>s)||(i>f&&j<s)||i>s||j<f))
            {
                fl=1;
                break;
            }
        }
        if(fl)
            cout<<"NO"<<endl;
        else
        {
            vec.pb(mp(i,j));
            cout<<"YES"<<endl;
        }
    }
    return 0;
}
