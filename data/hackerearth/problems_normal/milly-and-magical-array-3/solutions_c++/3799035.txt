#include <iostream>
using namespace std;

long long a[100010];

int main()
{
	ios_base::sync_with_stdio(false);
	
    int t;
    cin>>t;
    
    while(t--) {
    	int n;
    	long long x;
    	cin>>n>>x;
    	for(int i = 0; i < n; ++i)
    		cin>>a[i];
    	long long ans = 0;
    	for(int i = 1; i < n; ++i) {
    		if(a[i] > a[i-1])	continue;
    		long long diff = a[i-1]-a[i] + 1ll;
    		//cout<<"to add "<<diff;
    		long long mul = (diff/x) + (diff%x == 0 ? 0 : 1);
    		//cout<<" added "<<mul*x;
    		//cout<<" before "<<a[i];
    		a[i] += (mul*x);
    		
    		//cout<<" "<<i<<" after "<<a[i]<<endl;
    		ans += mul;
    	}
    	cout<<ans<<"\n";
    }
    
    
    return 0;
}
