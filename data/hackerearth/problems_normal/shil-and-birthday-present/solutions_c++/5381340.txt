#include <iostream>
#include <cstdlib>
using namespace std;

int compare(const void *a, const void *b)
{
	return(*(int *)a-*(int *)b);
}
int main()
{
	long long int n, i, m, j, c=0, count=0, arr[1000000]={0};
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>m;
    	arr[i]=m;
    }
    qsort(arr, n, sizeof(long long int), compare);
    for(j=1;j<n;j++)
    {
    	if(arr[j]==arr[j-1])
    	{
    		c++;
    	}
    	else
    	{
    		count=count+j-c;
    	}
    }
    cout<<count;
    return 0;
}
