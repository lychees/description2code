#include <iostream>
using namespace std;

int main()
{
    int num;
    
    while(cin >> num){
    	bool check = false;
    	int arr[num];
	    for(int i=0;i<num;i++){
	    	cin >> arr[i];
	    }
	    
	    for(int a=num-1;a>=0;a--){
	    	if(arr[a]%2==0){
	    		check = true;
	    		cout << arr[a] << " ";
	    	}
	    }
	    
	    if(!check){
	    		cout << "None to display";
	    }
	    
	    cout << endl;
	    
	   
    }
}
