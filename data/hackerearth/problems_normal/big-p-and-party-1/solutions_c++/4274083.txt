#define  _CRT_SECURE_NO_WARNINGS
#include <bits/stdc++.h>

using namespace std;
int a, b, x, y;
vector<vector<int> > G;
vector<bool> vis;
vector<int> lvl;
void bfs(int s)
{
	vis[s] = 1;
	queue<int> q;
	q.push(s);
	while (!q.empty())
	{
		int u = q.front(); q.pop();
		for (auto i : G[u])
		{
			if (!vis[i] && lvl[i]==-1)
			{
				lvl[i] = lvl[u] + 1;
				vis[i] = true;
				q.push(i);
			}
		}
	}
}

int main()
{
	scanf("%d %d", &a, &b);
	G.assign(a, vector<int>());
	vis.assign(a, 0);
	lvl.assign(a, -1);
	for (int i = 0; i < b; ++i)
	{
		scanf("%d %d", &x, &y);
		G[x].push_back(y);
		G[y].push_back(x);
	}
	lvl[0] = 0;
	bfs(0);
	for (int i = 1; i < a; ++i)
		cout << lvl[i] << endl;
	//system("pause");
}
