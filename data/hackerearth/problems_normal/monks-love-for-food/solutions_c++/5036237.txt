#include <iostream>
using namespace std;

int main()
{
	//get count 
	int count = 0;
	cin >> count;

	int* arr = new int[count];
	int input, val, curPos = 0;
	for (int i = 0; i < count; i++) 
	{
		cin >> input;
		
		if (input == 1)
		{ 
			if (curPos == 0)
			{
				cout << "No Food" << endl;
			}
			else
			{
				cout << arr[--curPos] << endl; // print top item cost and decrease curpos
			}
		}
		else
		{
			cin >> arr[curPos++]; // add item and increase curpos
			
		}
	}
    return 0;
}
