#include "bits/stdc++.h"
using namespace std;

int main(int argc, char const *argv[])
{
	int n; cin >> n;
	string s;
	map < string, vector < int > > mp;
	for (int i = 0; i < n; ++i)
	{
		cin >> s;
		mp[s].push_back(i);
	}
	int q; cin >> q;
	int l, r;
	while (q--)
	{
		cin >> l >> r >> s;
		if (mp[s].size() == 0)
			cout << "0\n";
		else
		{
			l--; r--;
			vector < int > v = mp[s];
			//for( int i : v) cout << i << " ";
			//	cout << endl;
			//cout << *lower_bound(v.begin(),v.end(),l) << " " << *upper_bound(v.begin(),v.end(),r) << endl;
			int ans = upper_bound(v.begin(), v.end(), r) - lower_bound(v.begin(), v.end(), l);
			cout << ans << endl;
		}

	}
	return 0;
}
