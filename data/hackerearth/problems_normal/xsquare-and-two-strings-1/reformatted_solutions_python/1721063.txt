t = input()
while t:
    t -= 1
    p = 0
    a = raw_input()
    b = raw_input()
    a = list(a)
    b = list(b)
    x = []
    for i in range(150):
        x.append(0)
    for i in a:
        asci = ord(i)
        x[asci] += 1
    for i in b:
        asci = ord(i)
        # print asci
        if x[asci] > 0:
            p = 1
            break
    if p:
        print "Yes"
    else:
        print "No"
