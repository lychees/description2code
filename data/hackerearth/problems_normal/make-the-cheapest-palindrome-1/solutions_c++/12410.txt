# include <stdio.h>
# include <stdlib.h>
# include <algorithm>
# include <cstdio>
# include <iostream>
# include <string>
# include <cstring>
# include <sstream>
# include <math.h>
using namespace std;

int main()
{
	int cases,ccost[2],cost,mincost,left,right;
	char s[10010];
	scanf("%d\n",&cases);
	while(cases--){
		scanf("%s\n%d\n%d\n",s,ccost,ccost+1);
		left = 0;
		right = strlen(s)-1;
		mincost = min(ccost[0],ccost[1]);
		cost = 0;
		while(left < right){
			if((s[left]!='/')&&(s[right]!='/')&&(s[left]!=s[right])){
				cost = -1;
				break;
			}
			if((s[left]=='/')&&(s[right]!='/')){
				cost+=ccost[s[right]-'a'];
			}
			else if((s[right]=='/')&&(s[left]!='/')){
				cost+=ccost[s[left]-'a'];
			}
			else if((s[left] == '/')&&(s[right] == '/')){
				cost+=mincost*2;
			}
			left++;
			right--;
		}

		printf("%d\n",cost);
	}



	return 0;
}
