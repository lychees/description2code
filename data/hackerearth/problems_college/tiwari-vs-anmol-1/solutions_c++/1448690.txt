#include <bits/stdc++.h>
using namespace std;
long long prime[1000010];
#define mod 1000000007
int main()
{
	#ifndef ONLINE_JUDGE
    freopen("in.txt", "r", stdin);
	#endif
	std::ios_base::sync_with_stdio(false);
	int i,j,k,l;
	for(i=2;i<1000000;i++)
	{
		if(prime[i]>0)
			continue;
		for(j=i+i;j<=1000000;j+=i)
		{
			if(prime[j]==0)
			prime[j]=i;
		}
	}
	for(i=1;i<=1000000;i++)
	{
		prime[i]+=prime[i-1];
		prime[i]%=mod;
	}
	cin>>j;
	for(i=0;i<j;i++)
	{
		cin>>k>>l;
		cout<<(prime[l]-prime[k-1])%mod<<endl;
	}
	return 0;
}
