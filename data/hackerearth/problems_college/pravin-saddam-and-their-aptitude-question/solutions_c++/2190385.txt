#include <iostream>
using namespace std;
unsigned int fa(int n)
{
	int a=1;
	for(int i=1;i<=n;i++)
	{
        a*=i;
        a=a%10;
	}
	return (a%10);
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
   	int a,b;
   	cin>>a>>b;
   	int f=0;
   	for(;a<=b;a++)
   	{ 	f+=fa(a); 
   		f%=10; 
   	}
   	cout<<(f%10)<<endl;
   }
    //cout << "Hello World!" << endl;
    return 0;
}
