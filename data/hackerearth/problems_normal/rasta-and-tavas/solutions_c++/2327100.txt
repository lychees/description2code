using namespace std;          
const int N = int(1e6);
int d[N+10],k,n,p,m;
int lp[N+10];
int main () {
	cin >> n >> k;
	int K = int(sqrt(n+2.0));
	for(int i=2;i<=n;i+=2) lp[i] = 2;
	for(int i=3;i<=K;i+=2) if(!lp[i]){
		for(int j=i*i;j<=n;j+=(i+i)) lp[j] = i;
	}
	int ans = 0;
	for(int i=2;i<=n;i++){
		p = lp[i];
		if(!p) p = i;
		if((i / p ) % p == 0) d[i] = d[i/p];
		else d[i] = p + d[i/p];
		if(d[i] == k){
			ans++;
		}
	}
	cout << ans;
return 0;
}
