#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t,n,i,j,count,m,x,y;
    cin>>t;
    while(t--){
    	cin>>n>>m;
    	count=0;
    	char arr[n][m+1];
    	for(i=0;i<n;i++){
    			cin>>arr[i];
    		//	cout<<arr[i]<<endl;
    			
    	}
    	for(i=n-1;i>=0;i--){
    		for(j=m-1;j>=0;j--){
    			if(arr[i][j]=='1')
    				continue;
    			else{
    				count++;
    				for(x=0;x<=i;x++)
						for(y=0;y<=j;y++)
							if(arr[x][y]=='1')
								arr[x][y]='0';
							else
								arr[x][y]='1';
    			}
    		}
    	}
    	cout<<count<<endl;
    }
    return 0;
}
