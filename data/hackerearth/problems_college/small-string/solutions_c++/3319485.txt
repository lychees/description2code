struct Test
{
    string s;
    Test(){}
    string solve()
    {
        int n = SZ(s);
        string res = s;
        // Iterate which character to move...
        FOR (i,0,SZ(s))
        {
            string q = "";
            FOR (j,0,n)
                if (i != j)
                    q += s[j];
            // and where
            FOR (j,0,SZ(q)+1)
            {
                string t = "";
                FOR (k,0,j)
                    t += q[k];
                t += s[i];
                FOR (k,j,SZ(q))
                    t += q[k];
                // try to minizmie the result
                res = min(res, t);
            }
        }
        return res;
    }
    void writeTest()
    {
        cout << SZ(s) << endl;
        cout << s << endl;
    }
    void writeRes()
    {
        cout << solve() << endl;
    }
    void readTest()
    {

        cin >> s;
    }
};
