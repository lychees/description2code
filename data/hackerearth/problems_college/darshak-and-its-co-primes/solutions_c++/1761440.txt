#include <iostream>
using namespace std;

int gcd(long a,long b)
{
	while(b!=0)
	{
		int t=b;
		b=a%b;
		a=t;
	}
	return a;
}

int main()
{
    int t;
    long p,q;
    cin>>t;
    while(t--)
    {
    	cin>>p>>q;
    	if(gcd(p,q)==1)
    	     cout<<"Is a Co-Prime\n";
    	else
    	     cout<<"Not a Co-Prime\n";
    }
    return 0;
}
