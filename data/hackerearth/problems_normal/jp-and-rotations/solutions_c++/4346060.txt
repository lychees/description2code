#include <iostream>
#include <cstring>
#include <vector>
using namespace std;

int main() {
	// your code goes here
	int n,m;
	cin>>n>>m;
	vector<int>v(n),t(n);
	for(int i = 0;i < n;i++){
		cin>>v[i];
	}
	for(int i = 0;i < n;i++){
		cin>>t[i];
	}
	int pos = 0;
	for(int i = 0;i < n;i++){
		if(t[i] == v[0]){
			pos = i;
			break;
		}
	}
	int y = 0;
	int ans = -1;
	for(int i = 0;i < m;i++){
		char c;
		int x;
		cin>>c>>x;
		if(c == 'L'){
			y = ((y-x)+n)%n;
		}
		if(c == 'R'){
			y = ((y+x)+n)%n;
		}
		if(y == pos){
			ans = i+1;
			break;
		}
	}
	cout<<ans<<endl;
	return 0;
}
