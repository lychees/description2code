#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        vector<vector<int> >shop(n,vector<int>(3));
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<3;j++)
            {
                cin>>shop[i][j];
            }

        }
        vector<vector<int> >dp(n,vector<int>(3));
        for(int i=0;i<3;i++)
            dp[0][i]=shop[0][i];

        for(int i=1;i<n;i++)
        {
            dp[i][0]=min(dp[i-1][1],dp[i-1][2])+shop[i][0];
            dp[i][1]=min(dp[i-1][2],dp[i-1][0])+shop[i][1];
            dp[i][2]=min(dp[i-1][0],dp[i-1][1])+shop[i][2];

        }
        int ans=min(min(dp[n-1][0],dp[n-1][1]),dp[n-1][2]);
        cout<<ans<<"\n";
    }
}
