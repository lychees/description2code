s = raw_input()
n = len(s)
nxt = [[n for x in xrange(26)]for y in xrange(n + 1)]
for i in reversed(xrange(n)):
	nxt[i] = nxt[i + 1][:]
	nxt[i][ord(s[i]) - 97] = i
for t in xrange(input()):
	i, isGood = 0, True  
	for c in raw_input():
		if i < n and nxt[i][ord(c) - 97] < n:
			i = nxt[i][ord(c) - 97] + 1
		else:
			isGood = False
			break
	print 'Yes' if isGood == True else 'No'
