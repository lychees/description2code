			if(p.empty())

			int z=p.top();
			while(!p.empty()){
				int tmp=p.top()-z;
				p.pop();
				if(tmp!=0)
				q.push(tmp);
			}
			cnt++;
			continue;
		}
		else{
			if(q.empty())

			}
			cnt++;
			continue;
		}
	}

}
