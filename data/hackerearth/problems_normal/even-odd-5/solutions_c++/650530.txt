#include<bits/stdc++.h>
using namespace std;
#define ll long long 
int main()
{
	
	int t;
	cin>>t;
	while(t--)
	{
		
		ll r,c;
		cin>>r>>c;
		ll even=0,odd=0;
		if(r<=c)
		{
			ll n=r/2;
			even=n*(n+1);
			if(r%2==0)
			odd=(n*n);
			
			else odd=((r+1)/2)*((r+1)/2);
		}
		else
		{
			ll n=c/2;
			even=(n*(n+1));
			if(c%2==0)
			odd=(n*n);
			
			else odd=((c+1)/2)*((c+1)/2);
			
			ll j=r-c;
			if(j%2==0)
			{
				even=even+c*(j/2);
				odd=odd+c*(j/2);
				
			}
			else
			{
				if(r%2==0)
				{
				even=even+c*((j/2)+1);
				odd=odd+c*(j/2);
				}
				else
				{
				even=even+c*(j/2);
				odd=odd+c*((j/2)+1);
				}
			
			}
			
			
			
				
		}
		
		cout<<even<<" "<<odd<<endl;
	}
}
