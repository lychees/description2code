#include <iostream>

#include <iomanip>

#include <math.h>

using namespace std;


int main()
{

	int t;

	scanf("%d",&t);

	while(t--)

	{

		int m,n,k;

		scanf("%d %d %d",&m,&n,&k);

		double d=pow(n,m);

		double dp[60][10001]={};

		for(int i=1;i<=n;i++)

			dp[1][i]=1/d;

		for(int i=2;i<=m;i++)

			for(int j=i;j<=min(k,n*i);j++)

				for(int k=1;k<=n;k++)

					if(j>=k)

						dp[i][j]+=dp[i-1][j-k];

		
		double ans=dp[m][k];

		int ans2=0;

		if(ans==0)

			printf("0.000 0\n");

		else

		{

			while(ans<1)

			{

				ans*=10;

				ans2++;

			}

			printf("%0.3lf %d\n",ans,ans2);

		}

	}

   
  
    return 0;

}
