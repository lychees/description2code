#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    int n;
    while(t--)
    {
    	cin>>n;
    	int *arr=(int *)malloc(n*sizeof(int));
    	for(int i=0;i<n;i++)
    	cin>>arr[i];
    	int count=0;
    	for(int i=1;i<n;i++)
    	{
    		if(arr[i]<=arr[i-1])
    		count++;
    	}
    	cout<<count<<"\n";
    }
    return 0;
}
