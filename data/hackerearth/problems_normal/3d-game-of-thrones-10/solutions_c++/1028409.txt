#define FOR(i,a,b) for( int i=(a),_b=(b);i<=_b;i++)
#define DOW(i,b,a) for( int i=(b),_a=(a);i>=_a;i--)
#define DEBUG(x) { printf << #x << " = " << x << endl; }

using namespace std;

bool f[55][55][55];
int test, n, x, y, z, res;

bool cal(int x, int y, int z){
    FOR(i,0,x-1)
        if (f[i][y][z] == false) return true;
    FOR(i,0,y-1)
        if (f[x][i][z] == false) return true;
    FOR(i,0,z-1)
        if (f[x][y][i] == false) return true;
    return false;
}

int main()
{
    ios::sync_with_stdio(false);
    //freopen("input.txt", "r", stdin);

    FOR(i,0,50)
    FOR(j,0,50)
    FOR(k,0,50) f[i][j][k] = cal(i,j,k);

    cin >> test;
    while (test--){
        res = 0;
        cin >> n;
        FOR(i,1,n){
            cin >> x >> y >> z;
            res = res ^ x ^ y ^ z;
        }
        if (res > 0) cout << "Stark\n";
        else cout << "Lannister\n";
    }

    return 0;
}
