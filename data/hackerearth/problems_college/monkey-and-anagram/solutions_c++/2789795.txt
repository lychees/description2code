#include <bits/stdc++.h>
using namespace std;
#define M 100005
#define MOD 1000000007
char ch[M],st[M];
long long a[M],b[M],c[M],d[M];
int main() {
	long long n,m,i,j,k,l,p,q,r,s,t;
	cin>>t;
	while(t--){
		int count=0,flag=0;
		cin>>ch;
		cin>>st;
		n=strlen(ch);
		m=strlen(st);
		for(i=0;i<=26;i++){
			a[i]=0;b[i]=0;
		}
		for(i=0;i<n;i++){
			a[ch[i]-97]+=1;
		}
	
		for(i=0;i<m;i++){
			b[st[i]-97]+=1;
		}
	
		for(i=0;i<=26;i++){
			if(a[i]<b[i])
			count++;
			else if( a[i]>b[i])
			flag++;
		}
		if(flag>0)
		cout<<"-1"<<endl;
		else
		cout<<count<<endl;
	}
	
	return 0;
}
