#include<bits/stdc++.h>
#define pb push_back
#define mp make_pair
#define f first
#define maxn 101000
#define s second
#define ll long long int
#define inf 1000000014
#define infl (ll)(1e18+1)
#define mod 1000000007
#define sz(x) (int) x.size()
using namespace  std;
typedef pair<ll,ll> p;
p brr[maxn];
ll arr[maxn];
ll ans2[maxn];
ll ans1[maxn];int a,b;
int main(int argc, char const *argv[])
{
  ll n,c;
  cin>>n>>c;
  for (int i = 1; i <= n; ++i)
  {
    scanf("%d",arr+i); 
    brr[i].f=arr[i]/c;
    if(arr[i]%c!=0)
      brr[i].f+=1;
    brr[i].s=i;
  }
  sort(brr+1,brr+1+n);
  ll cur=0;
  for (int i = 1; i <= n; ++i)
  {
    ans1[i]=cur;
    ans2[brr[i].s]=i;
    cur+=min(c,arr[i]);
  }
  int q;
  cin>>q;
  while(q--)
  {
    scanf("%d%d",&a,&b);
    if(a==1)
    {
      cout<<ans1[b]<<"\n";
    }
    else
    {
      cout<<ans2[b]<<"\n";
    }
  }
  return 0;
}
