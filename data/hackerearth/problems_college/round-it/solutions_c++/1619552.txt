#include <bits/stdc++.h>
using namespace std;

int main() 
{
    int t,n,b,k,i,x,y;
    cin>>t;
    while(t--) {
        cin>>n>>b;
        k = n/b;
        for(i = 0; i <= k+1; i++) {
            x = b*i;
            if(x > n) {
                break;
            }
        }
        x = b*(i-1);
        y = b*i;
        if(abs(y-n) <= abs(x-n)) {
            cout<<y<<endl;
        }
        else {
            cout<<x<<endl;
        }
    }
	return 0;
}
