
N = int(raw_input())
A = [0]*31
for n in range(N):
	s = raw_input().split()
	if s[0][0]=="G":
		for i in s:
			try:
				x=int(i)
				A[x]+=2
			except:
				pass
	else:
		for i in s:
			try:
				x=int(i)
				A[x]+=1
			except:
				pass
if A.count(max(A)) >1:
	print "No Date"
		
else:
	if (A.index(max(A)) == 19)  or (A.index(max(A)) == 20):
		print "Date"
	else:
		print "No Date"
