#include <iostream>
#include <map>
#include <string>
#include <algorithm>
using namespace std;

int main()
{
	long long int n,max;
	map<string,int> m;
	cin>>n;
	max=1;
	for(long long int i=0;i<n;i++ ){
		string str;
		cin>>str;
		sort(str.begin(),str.end());
		if(m.find(str)==m.end())	m[str]=1;
		else{
			m[str]++;
			if(m[str]>max)	max=m[str];
		}
	}
    cout <<max<< endl;
    return 0;
}
