#include <bits/stdc++.h>
#include <string>
#include <iostream>
using namespace std;

int main()
{
    string s,rs="";
    getline(cin, s);
    
    int sum=0;
    int *A=new int[256];
    for(int i=0;i<256;i++)
    {
    	A[i]=0;
    }
    
    for(int i=0;i<s.length();i++)
    {
    	sum+=s.at(i);
    	rs+=s.at(s.length()-1-i);
    	A[s.at(i)]++;
    }
    
    sum = sum/s.length() ;
    if(sum%2==1)
    {
    	int max=-1,maxi=-1;
    	for(int i=0;i<256;i++)
    	{
    		if(A[i]>max)
    		{
    			max=A[i];
    			maxi=i;
    		}
    	}
    	cout << (char)(maxi) ;
    }
    else
    {
    	cout << rs << endl;
    }
}
