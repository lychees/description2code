#include <iostream>
using namespace std;
int main()
{
	long long int n,arr[100005]={0};
	cin >> n;
	for(int i=1;i<=n;i++)
	{
		cin >> arr[i];
	}
	for(int i=2;i<=n;i++)
	{
		long long int maxi=0;
		for(int j=1;j*j<=i;j++)
		{
			if(i%j==0)
			{
				maxi=max(maxi,arr[j]);
				if(j!=1)
				maxi=max(maxi,arr[i/j]);
			}
		}
		arr[i]+=maxi;
	}
	for(int i=1;i<=n;i++)
		cout << arr[i] << ' ';
	cout << '\n';
	return 0;
}
