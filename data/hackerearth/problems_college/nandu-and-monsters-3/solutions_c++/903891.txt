#include<bits/stdc++.h>
using namespace std;
int main(){
    int n,q;
    cin>>n>>q;
    char a[n];
    cin>>a;
    while(q--){
        int i,j,k=0;
        cin>>i>>j;
        while ((int)log2(i)>(int)log2(j)){
           if (a[i-1]=='1') k=1;
            i=i/2;
        }
        while ((int)log2(i)<(int)log2(j)){
           if (a[j-1]=='1') k=1;
            j=j/2;
        }
        while(i!=j){
           if (a[i-1]=='1') k=1;
           if (a[j-1]=='1') k=1;
            i=i/2;
            j=j/2;
        }
        if (a[i-1]=='1') k=1;
        if (k==0) cout<<"Yes"<<endl;
        else cout<<"No"<<endl;
    }
    return 0;
}
