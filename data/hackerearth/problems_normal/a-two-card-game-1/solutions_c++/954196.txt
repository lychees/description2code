#include<iostream>
#include<bits/stdc++.h>
using namespace std;
int C[1000001];
int main(){
 int n;
 cin>>n;
 assert(n<=1000000);
 for(int i=0;i<n;i++)
 {
   int x;
   cin>>x;
   assert(x<=1000000);
   C[x]++;
 }
 double num=0,counta=0,countb=0;
 countb=n;
 for(int i=0;i<n;i++){
   if(C[i]){
     counta=C[i];
     countb-=C[i];
     num+=(counta*countb);
   }
 }
 double deno=n;
 deno*=(n-1);
 printf("%0.6f %0.6f",num/deno,num/deno);
 return 0;
}
