#include <vector>
#include <iostream>
#include <algorithm>
#include <climits>
#include <queue>
using namespace std;

const int v=101;
int dist[v];
typedef pair<int,int> PII;

void dijkstra(vector<PII> edges[],int N,int s){
    for(int i=0;i<N;i++){
        dist[i]=INT_MAX;
    }
    priority_queue<PII, vector<PII>, greater<PII> > pq;
    pq.push(make_pair(dist[s]=0, s));
    while (!pq.empty()) {
        PII p = pq.top();
        pq.pop();
        int curr = p.second;
        if (dist[curr] != p.first) continue;
        for (vector<PII>::iterator it = edges[curr].begin(); it != edges[curr].end(); it++) {
            if (dist[curr] + it->first < dist[it->second]) {
                dist[it->second] = dist[curr] + it->first;
                pq.push(make_pair(dist[it->second], it->second));
            }
        }
    }
}

//https://www.hackerearth.com/code-monk-graph-theory-ii/algorithm/monk-in-the-secret-services/

int main() {
    ios_base::sync_with_stdio(false);
    int t;
    cin>>t;
    while(t--){
        int n,m;
        cin>>n>>m;
        vector<pair<int ,int>> graph[3001];
        for(int i=0;i<m;i++){
            int x,y,r;
            cin>>x>>y>>r;
            x--;y--;
            graph[x].push_back(make_pair(r,y));
            graph[y].push_back(make_pair(r,x));
        }
        int s,a,h;
        cin>>s>>a>>h;
        s--;a--;h--;
        int ans[v]={0};
        dijkstra(graph,n,s);
        for (int i = 0; i < n; ++i) {
            ans[i]=dist[i];
        }
        dijkstra(graph,n,a);
        for (int i = 0; i < n; ++i) {
            ans[i]+=2*dist[i];
        }
        dijkstra(graph,n,h);
        int result=0;
        for (int i = 0; i < n; ++i) {
            if(i==a || i==s || i==h)
                continue;
            ans[i]+=dist[i];
            result=max(result,ans[i]);
        }
        cout<<result<<endl;
    }
    return 0;
}
