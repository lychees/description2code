#include <iostream>
using namespace std;

int main()
{
   int t,i,j;
   cin>>t;
   string str;
   while(t--){
   	cin>>str;
   j=0;
   char c =str[0];
   	for(i=1;i<str.length();i++){
   		if(str[i]==c){
   			j++;
   		}else{
   			c=str[i];
   		}
   	}
   cout<<j<<endl;
   	
   }
    return 0;
}
