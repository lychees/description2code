#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;

int main()
{
	long long a,b,c;
	cin>>a>>b>>c;
    if(a<c && b<c){
    	if(a+b<=c)
    		cout<<1<<"/"<<1<<endl;
    	else{
    		long long num=c*c-(c-a)*(c-a)-(c-b)*(c-b);
	    	long long den=a*b*2;
	    	long long g=__gcd(num,den);
	    	num/=g;
	    	den/=g;
	    	cout<<num<<"/"<<den<<endl;
    	}
    }
    else if(a<c){
    	long long num=c*c-(c-a)*(c-a);
    	long long den=a*b*2;
    	long long g=__gcd(num,den);
    	num/=g;
    	den/=g;
    	cout<<num<<"/"<<den<<endl;
    }
    else if(b<c){
    	long long num=c*c-(c-b)*(c-b);
    	long long den=a*b*2;
    	long long g=__gcd(num,den);
    	num/=g;
    	den/=g;
    	cout<<num<<"/"<<den<<endl;
    }
    else{
    	long long num=c*c;
    	long long den=a*b*2;
    	long long g=__gcd(num,den);
    	num/=g;
    	den/=g;
    	cout<<num<<"/"<<den<<endl;
    }
    return 0;
}
