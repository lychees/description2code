#include<bits/stdc++.h>
using namespace std;
int pallin(char *s,int l,int h)
{
	if(l>=h)
		return 0;
	if(s[l]==s[h])
		return pallin(s,l+1,h-1);
	else
		return min(pallin(s,l+1,h),pallin(s,l,h-1))+1;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		char str[100000];
		int n;
		cin>>n;
		cin>>str;
		int x=pallin(str,0,n-1);
		cout<<x<<endl;
	}
}
