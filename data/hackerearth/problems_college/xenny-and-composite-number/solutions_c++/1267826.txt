#include<bits/stdc++.h>
using namespace std;
int prime[100001]={0};
void seive()
{
    int i,j;
    prime[0]=prime[1]=1;
    for(i=2;i<100001;i++)
    {
        if(prime[i]==0)
        for(j=2;i*j<=100001;j++)
        {
                prime[i*j]=1;
        }
    }
}

int main()
{
    seive();
    int t;cin>>t;
    while(t--)
    {
        int n,i;
        cin>>n;
        if(n==1|| n==0)
        {
                printf("4\n");
                continue;
        }
        for(i=n+1;;i++)
        {
            if(prime[i]==1)
            {
                printf("%d\n",i);
                break;
            }
        }
    }
    return 0;
}
