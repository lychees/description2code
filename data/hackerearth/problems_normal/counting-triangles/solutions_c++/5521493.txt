#include<algorithm>
#include<iostream>
#include<array>
#include<map>

using namespace std;

int main(int argc, char const *argv[]) {

  long long N, C;
  array<long long, 3> value;
  map<array<long long, 3>, long long> Hash;

  cin >> N;
  C = 0;

  while(N--) {

    cin >> value[0] >> value[1] >> value[2];
    
    sort(value.begin(), value.end());
    
    Hash[value]++;
  }

  for (auto i = Hash.begin(); i != Hash.end(); i++) {
    if (i->second == 1L) {
      C++;
    }
  }

  cout << C << endl;
  return 0;
}
