#include "bits/stdc++.h"
using namespace std;
int main()
{
	long int tc;
	cin>>tc;
	while(tc--)
	{
		long int n;
		cin>>n;
		long int arr[2*n];
		long int len=2*n;
		for(long int i=0;i<len;i++)
			cin>>arr[i];
		sort(arr,arr+len);
		long int brr[n];
		for(long int i=0;i<n;i++)
		{
			brr[i]=arr[i]+arr[len-1-i];
		}
		sort(brr,brr+n);
		cout<<brr[n-1]<<endl;
	}
}
