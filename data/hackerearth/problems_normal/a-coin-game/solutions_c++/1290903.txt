#include <iostream>
#include <vector>
#include <set>
#include <algorithm>
using namespace std;

const int MX=1000001;

int main()
{
	int t,i,j;
	
	vector<bool> is(MX,0);
	
	set<pair<int,int>> s;
	
	s.insert({0,0});
	
	for( i=t=1; ; i++ )
	{
		s.insert({t,t+i});
		
		is[t]=is[t+i]=1;
		
		while( is[t] && t<MX ) t++;
		
		if( t+1+i >= MX ) break;
	}
	
	for( cin>>t; t--;  )
	{
		cin >> i >> j ;
		
		if(i>j) swap(i,j);
		
		cout << ( s.count({i,j}) ? "Don't Play" : "Play" ) << endl ;
	}
	
	return 0;
}
