#include <iostream>

using namespace std;

int main(int argc, char const *argv[]) {

	int t;
	cin >> t;

	while (t--) {
	
		int n, a[500], index, temp, i;
		cin >> n;

		a[0] = 1;
		index = 0;
		
		for(; n >= 2; n--) {
			
			temp = 0;
			
			for(i = 0; i <= index; i++) {
				temp = (a[i]*n) + temp;
				a[i] = temp % 10;
				temp = temp / 10;
			}
			
			while(temp > 0)
			{
				a[++index] = temp % 10;
				temp = temp / 10;
			}
		}

		int count = 0;

		for(i = index; i >= 0; i--)
			if (a[i] == 4 || a[i] == 7)
				count++;

		cout << count << endl;
	}

	return 0;
}
