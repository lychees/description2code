

def main():
    N, A, B = raw_input().split(" ")
    N = int(N)
    A = int(A)
    B = int(B)
    ans = 0
    if(N % A == 0):
        n = N / A - 1
    else:
        n = N / A
    if(N % B == 0):
        m = N / B - 1
    else:
        m = N / B
    if(N % (A * B) == 0):
        o = N / (A * B) - 1
    else:
        o = N / (A * B)
    ans = (A * (n * (n + 1) / 2)) + (B * (m * (m + 1) / 2)) - ((A * B) * (o * (o + 1) / 2))
    print ans
main()
