#include <iostream>
#include<algorithm>
using namespace std;

long long int foo(string a, string b)
{
    int i;
    for(i=0; i<b.size(); i++)
        if( b[i]!=a[i])break;
    return (long long int )((long long int )(a.size())-i);
}
int main()
{
    int n,i,j;
    long long int  ans=1;
    string ar[100009];
    cin>>n;
    for(i=0; i<n; i++)
        cin >> ar[i];
    sort(ar,ar+n);
    ans+=ar[0].size();
    for(i=1; i<n; i++)
        ans+=foo(ar[i],ar[i-1]);
    cout << ans <<"\n";
    return 0;
}
