import fractions
t = raw_input()
t = int(t)
while t > 0:
    t = t-1
    a, b, c = map(int, raw_input().split())
    a = int(a)
    b = int(b)
    c = int(c)
    minn = fractions.gcd(a, fractions.gcd(b, c))
    if minn != 1:
        ans = (a*b*c)/(minn*minn*minn)
    else:
        ans = a*b*c
    ans = ans % 1000000007
    print minn, ans
