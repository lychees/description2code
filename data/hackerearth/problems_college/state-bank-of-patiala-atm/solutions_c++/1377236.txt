#include<stdio.h>
#include<string.h>
#include<stdlib.h>

int flag=0;
struct TrieNode
{ 
    int isWord;
    TrieNode* child[10]; 
};

TrieNode *newTrieNode()
{
          TrieNode* trieNode = new TrieNode;
          trieNode->isWord=0;
          int i;
          for(i=0;i<10;i++)
           trieNode->child[i]=NULL;
          return trieNode;
}

void insert(const char* word, TrieNode** root)
{
     if(*word == '\0' && *root!=NULL)
     {
              flag=1;
              return;
     }
     if(*root == NULL)
        *root = newTrieNode();
     
     if(*word != '\0')
     {
        if( (*root)->isWord == 1)
       {
           flag=1;
           return;
       }
       insert(word+1,&((*root)->child[(*word)-'0']));
      
     }
     else
     {
         (*root)->isWord=1;
     }
}

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        flag=0;
        TrieNode* root = NULL;
        int n,i;
        scanf("%d",&n);
        getchar();
        char nums[n][11];
        for(i=0;i<n;i++)
        {
              scanf("%s",nums[i]);
        }
        
        for(i=0;i<n;i++)
        {
           if(flag==1)
            break;
           insert(nums[i],&root);
        }
        if(flag==0)
         printf("YES\n");
        else
         printf("NO\n");
        
    }
    return 0;
}
