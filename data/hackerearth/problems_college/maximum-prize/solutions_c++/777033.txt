#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t,n,k,i;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		scanf("%d",&k);
		int a[n];
		long long sum=0;
		for(i=0;i<n;i++)
			scanf("%d",&a[i]);
		sort(a,a+n);
		i=n-1;
		int x=0;
		while(x<k && x<n)
		{
			sum+=a[i];
			x++;
			i--;
		}
		printf("%lld\n",sum);
	}
	return 0;
}
