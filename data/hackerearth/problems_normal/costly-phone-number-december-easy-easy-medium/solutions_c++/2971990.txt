#include<bits/stdc++.h>
#include<math.h>
#include<cmath>
using namespace std;
#define pb push_back
#define mp make_pair
#define sz(_vvv) (int)(_vvv.size())
#define all(_v) (_v).begin(),(_v).end()
#define Geti(x) scanf("%d",&x)
#define Getl(x) scanf("%ld",&x)
#define Getll(x) scanf("%64d",&x)
#define pii pair<int,int>
typedef long long ll;
const ll mod=1e9+7;
const int INF=0x7f7f7f7f;
const double eps = 1e-6;

int a[11];

int main() {
	int t; cin >> t;
	while (t --) {
		for (int i = 0; i <= 9; i ++) cin >> a[i];

		bool flag = true;
		
		while (flag) {
			flag = false;
			for (int i = 0; i <= 9; i ++) {
				for (int j = 0; j <= 9; j ++) {
					int num = (i + j) % 10;
					if(a[num] > a[i] + a[j]) {
						a[num] = a[i] + a[j];
						flag = true;
					}
				}
			}
		}

		int ans = 0;
		int n; cin >> n;
		for (int i = 0; i < n; i ++) {
			char c; cin >> c;
			ans += a[c - '0'];
		}

		cout << ans << "\n";
	}
	return 0;
}
