#include <iostream>
using namespace std;

bool Good[5000][5000];
char A[5000][5000];

int main() {
	int n, m;
	
	cin>>n>>m;
	for(int i=1; i<=n; i++) cin>>A[i] + 1;
	
	for(int i=2; i<=n; i++) {
		for(int j=2; j<=m; j++) {
			if(A[i][j] == '_' && A[i-1][j-1] == '_' &&
			   A[i-1][j] == '#' && A[i][j-1] == '#')
			      Good[i][j] = 1;
			else
				Good[i][j] = 0;
		}
	}
	
	int sol = 0;
	
	for(int t=4; t<=32; t*=2) {
		for(int i=n; i>=t; i--)
		for(int j=m; j>=t; j--) {
			if(Good[i][j] && Good[i-t/2][j] &&
			Good[i][j-t/2] && Good[i-t/2][j-t/2]) {
				Good[i][j] = 1;
				if(t == 32) sol++;
			} else {
				Good[i][j] = 0;
			}
		}
	}
	
	cout<<sol;
	
    return 0;
}
