#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
int main()
{
  long long ans=0,n,k,i,j;
  cin>>n>>k;
  long long dp[k+1][n+1];
  for(i=0;i<=k;i++)
  	for(j=0;j<=n;j++)
  		dp[i][j]=0;
  for(i=1;i<=k;i++)
  {
      dp[i][1]=i;
  }
  for(i=1;i<=k;i++)
  {
      for(j=2;j<=n;j++)
      {
          dp[i][j]=(dp[i-1][j]%mod+dp[i][j-1]%mod)%mod;
      }
  }
  /*for(i=0;i<=k;i++)
  {
  	for(j=0;j<=n;j++)
  	cout<<dp[i][j]<<"  ";
  	cout<<endl;
  }*/
      for(i=1;i<=n;i++)
      {
          ans=(ans%mod+dp[k][i]%mod)%mod;
      }
 		ans=ans%mod;
      cout<<ans<<endl;
 
 return 0;
}
