#include <bits/stdc++.h>
using namespace std;
int arr[101];
int dp[101][101],n;
int recur(int idx, int las) {
	if(idx >= n) {
		return 0;
	}
	if(dp[idx][las] != -1) {
		return dp[idx][las];
	}
	int ans = recur(idx+1,las);
	if(arr[idx] > arr[las]) {
		ans = max(ans, 1+recur(idx+1,idx));
	}
	return dp[idx][las] = ans;
}
int main()
{
    //int n;
    cin >> n;
    memset(dp,-1,sizeof dp);
    for(int i =0 ; i < n; i++) {
    	cin >>arr[i];
    }
    arr[n] = -1e9;
    int ans = recur(0,n);
    vector <int> v;
    while(ans > 0) {
    	v.push_back(ans%2);
    	ans = ans/2;
    }
    reverse(v.begin(), v.end());
    for(int i =0 ; i <v.size(); i++) {
    	cout << v[i];
    }
    return 0;
}
