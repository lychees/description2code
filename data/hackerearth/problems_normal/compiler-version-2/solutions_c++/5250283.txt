#include <iostream>
#include <string>
using namespace std;

int main()
{
	int j=0;

    string s;
    while(getline(cin,s))
    {
    
    
    for(int i=0;i<s.length();i++)
    {
    	if(s[i]=='/' && s[i+1]=='/')
    	{
    		break;
    	}
    	else if(s[i]=='-' && s[i+1]=='>' )
    	{
    		s[i]='.';
    		s.erase(i+1,1);
    	
    	}
    	
    }
    
    cout<<s<<endl;
   }
    return 0;
}
