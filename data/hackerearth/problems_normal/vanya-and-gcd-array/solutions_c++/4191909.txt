#include <iostream>
#define mod 1000000007
using namespace std;

int cache[101][101];

int gcd(int a,int b) {
	int x=a,y=b; 
	if(a<b) {
		x = b;
		y = a;
	}
	while(y>0) {
		int t = x%y;
		x = y;
		y = t;
	}
	return x;
}


int main()
{
    for(int i=1;i<=100;i++) {
    	for(int j=i;j<=100;j++) {
    		cache[i][j]=cache[j][i]=gcd(j,i);
    	}
    }
    int N;
    cin>>N;
    int arr[N];
    for(int i=0;i<N;i++) {
    	cin>>arr[i];
    }
    int seqnum[N][101];
    int sum = 0;
    for(int i=0;i<N;i++) {
    	for(int j=0;j<101;j++) {
    		seqnum[i][j] = 0;
    	}
    }
    for(int i=N-1;i>=0;i--) {
    	seqnum[i][arr[i]] = 1;
    	for(int j=i+1;j<N;j++) {
    		if(arr[i]<arr[j]) {
    			for(int k=1;k<101;k++) {
    				int x = cache[arr[i]][k];
    				seqnum[i][x] = (seqnum[i][x]+seqnum[j][k])%mod;
    			}
    		}
    	} 
    	sum = (sum+seqnum[i][1])%mod;
    }
    cout<<sum;
    return 0; 
}
