#include <iostream>
#include<stdio.h>
using namespace std;
int main()
{
    long int e,n,q,m,a,b,c,d,i;
    scanf("%ld%ld",&n,&q);
    long int z[n];
    for(i=a;i<=b;i++)
    z[i]=1;
    scanf("%ld%ld%ld",&c,&a,&b);
    for(i=a;i<=b;i++)
    z[i]=1;
    while(--q)
    {
    	e=0;
    	scanf("%ld%ld%ld",&d,&m,&n);
    	for(i=m;i<=n;i++)
    	{
    		if(z[i]==1)
    		e++;
    	}
    	printf("%ld\n",e);
    }
    return 0;
}
