#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;


int main()
{
    //cout << "Hello World!" << endl;
    
    int noOfQuestions, Queries;
    
    cin >> noOfQuestions;
    cin >> Queries;
    
	int * aTime = new int[noOfQuestions];
	int * aScore = new int [noOfQuestions];
	
	for (int i = 0; i < noOfQuestions ; i++)
		cin >> *(aTime + i);
		
	for (int i=0; i < noOfQuestions; i++)
			cin >> *(aScore + i);
	
	// sort the score array in Ascending order
	std::vector<std::pair<int,int>> items;
	
	// fill the vector
	for (int i = 0; i< noOfQuestions; i++)
		items.push_back(std::pair<int, int> (aTime[i], aScore[i]));
	
	// sort the value
	std::sort(items.begin(), items.end());
	
	
	
	for (int i= 0 ; i < Queries; i++)
	{
		int kSelectedQuestions;
		cin >> kSelectedQuestions;
	
		int iIndexCount = 0;
		for (int j = 0, i =noOfQuestions -1 ; j < kSelectedQuestions; j++, i--)
		{
			
			iIndexCount += items[i].first ;
		}
		cout << iIndexCount << endl;
	}
		
	
    return 0;
}
