#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int l,r,p=0,p1=0;
    	cin>>l>>r;
    	for(int i=l;i<=r;i++)
    	{
    		for(int j=i;j<=r;j++)
    		{
    			p=(i^j);
    			if(p>p1)
    			{
    				p1=p;
    			}
    		}
    	}
    	cout<<p1<<endl;
    }
    return 0;
}
