no_of_replace_chars =int(raw_input())
replace_strings = list()
for x in range(no_of_replace_chars):
	replace_strings.append(raw_input())

s = raw_input()

d = dict()
for key, value in enumerate(s):
	if d.has_key(value):
		d[value].append(key)
	else:
		d[value] = [key]
temp = "temp"
for r_str in replace_strings:
	r_s = r_str.split(" ")
	first = r_s[0]
	second = r_s[1]
	lower_first = r_s[0].lower()
	lower_second = r_s[1].lower()
	if d.has_key(first):
		if not d.has_key(second):
			d[second] = d[first]
			del d[first]
		else:
			temp = d[first]
			d[first] = d[second]
			d[second] = temp
	elif d.has_key(second):
		d[first] = d[second]
		del d[second]

	if d.has_key(lower_first) :
		if not d.has_key(lower_second):
			d[lower_second] = d[lower_first]
			del d[lower_first]
		else:
			temp = d[lower_first]
			d[lower_first] = d[lower_second]
			d[lower_second] = temp
	
	elif d.has_key(lower_second):
		d[lower_first] = d[lower_second]
		del d[lower_second]

del temp	
final_str =  range(len(s))
for key, values in d.items():
	for x in values:
		final_str[x]=key
print "".join(final_str )
