#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <vector>
#include <algorithm>
#include <cstring>

using namespace std;

#define SIZE 1001

vector<int> adjList[SIZE];

bool visited[SIZE] = { 0 };

int dfs(int s)
{
	int sum = 1;
	visited[s] = true;

	for (int i = 0; i < adjList[s].size(); i++)
	{
		if (visited[adjList[s][i]] == false)
		{
			sum += dfs(adjList[s][i]);
		}
	}
	return sum;
}

int main()
{
	int j, N, D, x, y, result;

#ifdef _MSC_VER
	freopen("Text.txt", "r", stdin);
#endif

	cin >> N >> D;

	for (j = 0; j < D; j++)
	{
		cin >> x >> y;
		adjList[x].push_back(y);
//		adjList[y].push_back(x);
	}

	result = N;

	for (j = 1; j <= N; j++)
	{
		result = min(result, dfs(j));
		memset(&visited[1], 0, N*sizeof(visited[0]));
	}

	cout << result << endl;

	return 0;
}
