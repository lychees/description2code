#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
#define MAXN 100001

vector<int> adj[MAXN];

int par[MAXN];
int find(int u) {
  if (par[u] != u)
    par[u] = find(par[u]);
  return par[u];
}
void merge(int u, int v) {
  par[find(u)] = find(v);
}

tuple<LL, int, int> ed[MAXN];
int main() {
  ios_base::sync_with_stdio(false);
  cin.tie(NULL); 

  int n, m; cin >> n >> m;
  for (int i = 1; i <= n; i++) par[i] = i;
  int u, v; LL w;
  for (int i = 0; i < m; i++) {
    cin >> u >> v >> w;
    ed[i] = make_tuple(w, u, v);
  }
  sort(ed, ed + m);
  LL sum = 0;
  for (int i = 0; i < m; i++) {
    tie(w, u, v) = ed[i];
    u = find(u), v = find(v);
    if (u != v) {
      merge(u, v);
      sum += w;
    }
  }
  cout << sum << endl;
  
  return 0;
}
