#include <iostream>
using namespace std;
#include<cmath>

int main()
{
   int t;
   cin>>t;
   string ts;
   getline(cin,ts);
   	while(t--)
   	{
   		string s;
   	    getline(cin,s);
   	    //cout<<s<<endl;
   	    int l=s.size();
   	    int jp=sqrt(l);
   	    for(int i=0;i<jp;i++)
   	    {
   	    	for(int j=i;j<s.size();j=j+jp)
   	    	{
   	    		cout<<s[j];
   	    	}
   	    }
   	    cout<<endl;
   	    
   		
   	}
 
    return 0;
}
