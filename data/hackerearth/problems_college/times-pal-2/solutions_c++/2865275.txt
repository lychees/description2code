#include<iostream>
#include<string>
using namespace std;

bool isPallindrome(string s){
	long long int size= s.size();
	for(int i= 0 ;i<size/2; i++){
		if(s[i]!=s[size-i-1])
			return false;
	}
	return true;
}

int main(){
	string a,b;
	long long int t, sum, size;
	cin>>t;
	
	while(t--){
		cin>>a>>b;
		size= a.size();
		sum=0;
		
		for(int i=0; i<=size; i++){
			if( isPallindrome(a.substr(0,i) + b + a.substr(i,size-i)) )
				sum++;
		}
		cout<<sum<<endl;
	}
	return 0;
}
