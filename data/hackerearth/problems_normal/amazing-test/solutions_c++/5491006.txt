using namespace std;
int n,t,a[102],x,tot=0;
int dpp[105][(int)1e4],dpn[105][(int)1e4];

//t1= first-second
//a+b=tot
//a-b=t1
int f(int i,int t1)
{
    if(t1>=0&&dpp[i][t1]!=-1)return dpp[i][t1];
    else if(t1<0&&dpn[i][-t1]!=-1)return dpn[i][-t1];

    //cout<<i<<" "<<(tot+t1)<<" "<<(tot-t1)<<" "<<2*x<<endl;
    if(i==n&&(tot+t1)>(2*x))return 0;
    if(i==n&&(tot-t1)>(2*x))return 0;

    //cout<<i<<" "<<(tot+t1)<<" "<<(tot+t1)<<" "<<2*x<<endl;
    if(i==n)return 1;
    if(t1>=0)
dpp[i][t1]=f(i+1,t1-a[i])|f(i+1,t1+a[i]);
else dpn[i][-t1]=f(i+1,t1-a[i])|f(i+1,t1+a[i]);

    return  f(i+1,t1-a[i])|f(i+1,t1+a[i]);}


 int fx(int i,int j)
 {
     if(j>=0)return dpp[i][j];
     else return dpn[i][-j];
 }

 main()
{
	cin.tie(0);cout.tie(0);
   cin>>t;
   while(t--)
   {
       for(int j=0;j<=100;++j)
       for(int i=0;i<=1e4;++i)dpp[j][i]=0,dpn[j][i]=0;
       tot=0;
       cin>>n>>x;
       for(int i=0;i<n;i++)
        {cin>>a[i];tot+=a[i];}

       sort(a,a+n);
       string ans;

       for(int j=0;j<1e4;++j)
       {
          if((tot+j)<=(2*x)&&(tot-j)<=(2*x)) dpp[n][j]=1;
          else dpp[n][j]=0;
       }

       for(int j=1;j<1e4;++j)
       {
           int k=-j;
          if((tot+k)<=(2*x)&&(tot-k)<=(2*x)) dpn[n][j]=1;
          else dpn[n][j]=0;
       }

        for(int i=n-1;i>=0;i--)
            for(int j=0;j<=1e4;++j)
        {
            //+ve j
             dpp[i][j] = fx(i+1,j+a[i])|fx(i+1,j-a[i]);
             //-ve j
             dpn[i][j]=fx(i+1,-j+a[i])|fx(i+1,-j-a[i]);
        }

      if(fx(0,0))ans="YES";
      else ans="NO"; cout<<ans<<endl;
    }
   return 0;
}
