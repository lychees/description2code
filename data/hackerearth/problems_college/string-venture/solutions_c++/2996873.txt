#include<iostream>
#include<string>
using namespace std;

int main()
{
	string a;
	int i,n;
	getline(cin,a);
	
	int sum=0;
	
	n=a.length();
	
	for(i=0;i<n;i++)
	{
		sum=sum+a[i];
	}
	sum=sum/n;
    //cout<< sum<<endl;
    
	if(sum%2==0)
	{
		for(i=n-1;i>=0;i--)
		cout<<a[i];
	}
	else
	  {
	  	int b[255],max=0,key;
	  	for(i=0;i<256;i++)
	  	{   b[i]=0;
		  }
		  
	  	for(i=0;i<n;i++)
	  	{
	  		b[a[i]]++;
		}
		
		for(i=0;i<256;i++)
		{
			if(b[i]>max)
	       	{
	
				max=b[i];
				key=i;
	       }
		}
		cout<<(char) key;
		
	  }
	return 0;
}
