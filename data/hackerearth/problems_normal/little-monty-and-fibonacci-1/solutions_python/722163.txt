fib = [0] *1005
mult = [0] * 1005
fib[1] = 1
fib[2] = 1
mult[1] = 1
mult[2] = 4

for x in xrange(3,1005):
	fib[x] = fib[x - 1] + fib[x - 2]
	mult[x] = mult[x - 1] + 1
	for i in xrange(1,x):
		if (fib[x] % fib[i] == 0):
			mult[x] += 1
