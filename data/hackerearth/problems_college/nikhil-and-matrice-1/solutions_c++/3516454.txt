#include<iostream>
using namespace std;
int main()
{
   int T,N,cnt,xl,yl,xh,yh,i,j;
   cin>>T;
   while(T--)
   {
	cin>>N;
	int arr[N][N];
	cnt=1;
	xl=yl=0;
	xh=yh=N-1;
	while(1)
	{
		i=yl;
		for(j=xl;j<=xh;j++)
			arr[i][j]=cnt++;

		if(cnt>N*N)
			break;

		j--;	yl++;
		for(i=yl;i<=yh;i++)
			arr[i][j]=cnt++;

		i--;	xh--;
		for(j=xh;j>=xl;j--)
			arr[i][j]=cnt++;

		if(cnt>N*N)
			break;

		j++;	yh--;
		for(i=yh;i>=yl;i--)
			arr[i][j]=cnt++;

		xl++;
	}
	for(i=0;i<N;i++)
	{	for(j=0;j<N;j++)
			cout<<arr[i][j];
		cout<<endl;
	}
	cout<<endl;
   }
   return 0;
}
