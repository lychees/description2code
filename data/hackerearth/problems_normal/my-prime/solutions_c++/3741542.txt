template <class X>
void input(vector<X>&a,int N){
    X temp;
    rep(i,0,N){
        cin>>temp;
        a.push_back(temp);
    }
}
void solve(){
    int N;
    cin>>N;
    vi a(N);
    rep(i,0,N){
        cin>>a[i];
    }
    rep(i,0,N){
        rep(j,i+1,N){
            if(a[i]!=-1 && a[j]%a[i]==0){
                a[j]=-1;
            }
            else if(a[j]!=-1 && a[i]%a[j]==0){
                a[i]=-1;
                break;
            }
        }
    }
    rep(i,0,N){
        if(a[i]!=-1){
            cout<<a[i]<<" ";
        }
    }
}
int main(){

	int t=1;
	while(t--){
		solve();
