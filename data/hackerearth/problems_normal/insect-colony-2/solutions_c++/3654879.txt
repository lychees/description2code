#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	long long num,sum=0;
    	for(int i=0;i<n;i++)
    	{
    		cin>>num;
    		sum=sum+num;
    	}
    	if(sum%2==0)
    	cout<<"Yes\n";
    	else
    	cout<<"No\n";
    }
    return 0;
}
