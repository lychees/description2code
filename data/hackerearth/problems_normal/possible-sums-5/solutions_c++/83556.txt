#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
	int t, n, no, count;
	cin >> t;
	while(t--)
	{
		no = count = 0;
		cin >> n;
		vector<int> c(n+1);
		for(int i = 1; i <= n; ++i)
		{
			cin >> c[i];
			no += c[i];
		}
		vector<vector<bool>> v(n+1, vector<bool> (no+1, false));
		for(int i = 0; i <= n; ++i)
			v[i][0] = true;
		for(int i = 1; i <= n; ++i)
		{
			for(int j = 1; j <= no; ++j)
			{
				v[i][j] = v[i-1][j];
				if(!v[i][j] && ((j - c[i]) >=0))
					v[i][j] = v[i-1][j-c[i]];
			}
		}
		
		for(int i = 0; i <= no; ++i)
		{
			if(v[n][i])
				++count;
		}
		cout << count << endl;
	}
	return 0;
}
