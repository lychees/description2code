def int2bin(val):
    res = ''
    while val > 0:
        res += str(val & 1)
        val = val >> 1     # or val=val/2
    return res[::-1]


# print int2bin(12)
t = input()
while t:
    t -= 1
    n = input()
    x = int2bin(n)
    # print type(x)
    l = len(x)
    c = 0
    for i in range(l):
        if x[i] == '1':
            c += 1
    if c % 2 == 0:
        print 'even'
    else:
        print 'odd'
