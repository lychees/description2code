#include <iostream>
#include<cmath>
#include <algorithm>
using namespace std;
 
int main()
{
   int t,n,a[1001],s,h,i;
   cin>>t;
   while(t--)
   {
   	
   	cin>>n>>h;
   //	sort(a,a+n);
   	s=10001;
   	for(i=0;i<n;i++)
   	cin>>a[i];
   	for(i=0;i<n;i++)
   	{
   		int l=abs(a[i]-h);
   		if(s>l)
   		s=l;
   	}
   	printf("%d\n",s);
    }return 0;
}
