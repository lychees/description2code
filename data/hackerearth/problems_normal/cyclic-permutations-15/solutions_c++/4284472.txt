#include <iostream>
using namespace std;

int main()
{
	int T;
	cin >> T;
	for(int k = 0; k < T; ++k)
	{
		string A,B;
		cin >> A;
		cin >> B;
		int M = A.size();
		string C(A);
		C.append("a");
		C.append(B);
		C.append(B,0,M-1);
		int N = C.size();
		int *Z = new int[N];
		int L =0, R = 0;
		for(int i = 1; i < N; ++i)
		{
			if (i > R)
			{
				L = R = i;
				while(R < N && (C[R] == C[R-L]))
				{
					++R;
				}
				Z[i] = R-L;
				--R;
			}else
			{
				int k = i - L;
				if (Z[k] < (R-i+1))
				{
					Z[i] = Z[k];
				}else
				{
					L = i;
					while(R < N && (C[R] == C[R-L]))
					{
						++R;
					}
					Z[i] = R-L;
					--R;
				}
			}
		}
		int count = 0;
		for(int i = 1; i < N; ++i)
		{
			count += (Z[i] >= M);
		}
		cout << count << endl;
	}
    return 0;
}
