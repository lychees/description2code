noc = input()
ans = {}
temp = []
while noc:
    noc -= 1
    s1 = raw_input()
    if ans.has_key(s1):
        ans[s1] += 1
    else:
        ans[s1] = 1
    temp.append(s1)

palins = 0

for x in temp:
    x1 = list(x)
    x1r = x1[::-1]
    x1rs = "".join(x1r)
    if ans.has_key(x1rs):
        palins += ans[x1rs]
print palins / 2
