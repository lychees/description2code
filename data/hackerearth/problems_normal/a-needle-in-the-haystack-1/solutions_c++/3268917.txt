const int cs = 0x186A1;

const int me = 1100005;

int t;
char s[me];
int c[400], d[400];
int main()
{
    //ios_base:: sync_with_stdio(false);
    //cin.tie(false);

    si(t);
    getchar();
    while(t --){
        for(int i = 0 ; i < 200 ; i ++)
            c[i] = 0;
        for(int i = 0 ; i < 200 ; i ++)
            d[i] = 0;
        gets(s); int l = strlen(s); //cout << l << endl;

        for(int i = l - 1 ; i >= 0 ; i --)
            c[ s[i] ] ++;

        gets(s);
        int ok = 0, l2 = strlen(s);

        for(int i = 0 ; !ok && i < l2 ; i ++){
            d[ s[i] ] ++;

            if(i >= l - 1){
                int ss = 1;
                for(int j = 'a' ; ss & j <= 'z' ; j ++)
                    if(c[j] != d[j])
                        ss = 0;
                ok |= ss;
            }
            if(i >= l - 1){
                d[ s[i - l + 1] ] --;
            }
        }

        if(ok) cout << "YES" << endl;
        else cout << "NO" << endl;
    }

    ///cout << "Elapsed time is: " << clock() / CLOCKS_PER_SEC << endl;

    return 0;
}
