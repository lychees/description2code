#include <iostream>
#include <cstdio>
#include <algorithm>

#define MOD 1000000007

using namespace std;

long long f[2000011];

void fact()
{
    f[0] = 1;
    int i = 1;
    while(i<2000011)
    {
        f[i] = f[i-1]*i;
        f[i] = f[i]%MOD;
        i++;
    }
}

long long fast_pow(long long a,long long n)
{
    long long result = 1;
    long long power = n;
    long long value = a;
    while(power>0)
    {
        if(power&1)
        {
            result = result*value;
            result = result%MOD;
        }
        value = value*value;
        value = value%MOD;
        power /= 2;
    }
    return result;
}

long long inv(long long x)
{
    return fast_pow(x,MOD-2);
}

int main()
{
	int t;
	long long int n;
	fact();
	scanf("%d",&t);
	while(t--)
	{
		scanf("%lld",&n);
		long long pro = (f[2*n]*inv(f[n]))%MOD;
		pro = (pro*inv(f[n]))%MOD;
		long long ans = (pro*inv(n+1))%MOD;
		printf("%lld\n",ans);
	}
    return 0;
}
