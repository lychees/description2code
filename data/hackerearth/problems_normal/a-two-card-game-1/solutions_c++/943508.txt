#include<bits/stdc++.h>
using namespace std;
#define ll	long long int
vector<ll> v;
int main()
{
	ll n,tmp;
	scanf("%d",&n);
	if(n==1)
	{
		ll tmp;
		cin>>tmp;
		printf("1.000000 0.000000\n");
		//else
		//	printf("1.000000 1.000000\n");
	}
	else
	{
		for(int i=0;i<n;++i)
		{
			scanf("%d",&tmp);
			v.push_back(tmp);
		}
		ll win=0,cnt=0;
		sort(v.begin(),v.end());
		tmp=v[0];
		cnt=1;
		for(int i=1;i<n;++i)
		{
			if(v[i]==tmp)
			{
				cnt++;
				continue;
			}
			else
			{
				tmp=v[i];
				win+=cnt*(n-i);
			cnt=1;
			}
		}
		printf("%0.6f %0.6f\n",(win*1.0)/(n*(n-1)),(win*1.0)/(n*(n-1)));
	}
	return 0;
}
