#include<bits/stdc++.h>
using namespace std;
int Div(int a,int b)
{
	if(a>b)
	return a/b;
	else
	return b/a ;
}
int main()
{
	int t,c;
	cin>>t;
    for (c=1;c <=t;c++)
	{
	    int a1,a2,b1,b2;
	    cin>>a1>>a2>>b1>>b2;
	    int i,j,a,b,result=0;
	    for(i=a1;i<=a2;i++)
	    {
	    	for(j=b1;j<=b2;j++)
	    	{
	    		a = i;
				b = j;
				int count=0;
				if(a != b)
				{
				while(Div(a,b)==1)
				{
				 if(a>b)
				 {
				 	a = a%b;
				 }
				 else
				 {
				 	b = b%a;
				 }	
				count++;	
				}
				if(count%2==0)
				{
				result++;	
				} 
			}
		}
		}
		cout<<"Case #"<<c<<": "<<result<<"\n";
	}
}
