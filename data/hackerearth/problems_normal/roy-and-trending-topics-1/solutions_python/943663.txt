n = int(raw_input())
topics = [[0 for x in range(3)] for x in range(n)]
for i in range(n):
	topics[i][0], topics[i][1], p, l, c, s = map(int, raw_input().split())
	topics[i][2] = (p*50 + l*5 + c*10 + s*20) - topics[i][1]
	topics[i][1] += topics[i][2]
topics.sort(key = lambda x : (x[2], x[0]))
topics.reverse()
for i in range(5):
	print str(topics[i][0]) + " " + str(topics[i][1])
