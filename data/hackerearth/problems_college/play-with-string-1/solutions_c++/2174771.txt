#include <iostream>
#include<stdio.h>
#include<cstring>
using namespace std;

int abs(int n)
{
	if(n<0)
	return -n;
	return n;
}
int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
    	char ch[100009];
    	scanf("\n%[^\n]",ch);
    	int len=strlen(ch);
    	int k=len-1;
    	bool f=true;
    	for(int i=1;i<len;i++)
    	{
    		if(abs(ch[i]-ch[i-1])!=abs(ch[k-i+1]-ch[k-i]))
    		{
    			f=false;
    			break;
    		}
    	}
    	if(f)
    	cout<<"Funny\n";
    	else
    	cout<<"Not Funny\n";
    }
}
