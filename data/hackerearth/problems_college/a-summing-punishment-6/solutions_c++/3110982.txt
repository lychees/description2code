#include <bits/stdc++.h>
using namespace std;
#define ll long long int
ll tree[300005]={0};
ll arr[100001];
ll build(ll fi,ll la,ll index)
{
	if(fi==la)
	{
		tree[index]=arr[fi];
		return arr[fi];
	}
	ll l,r,mid;
	mid=(fi+la)/2;
	l=build(fi,mid,index*2);
	r=build(mid+1,la,index*2+1);
	
	tree[index]=l+r;
	
	return tree[index];
}
ll query(ll index,ll fi,ll la,ll p,ll q)
{
	if(fi==p&&la==q)
	{
		return tree[index];
	}
	
	ll mid=(fi+la)/2;
	
	if(q<=mid)
	return query(index*2,fi,mid,p,q);
	else if(p>mid)
	return query(index*2+1,mid+1,la,p,q);
	else
	{
		ll l,r;
		l=query(index*2,fi,mid,p,mid);
		r=query(index*2+1,mid+1,la,mid+1,q);
		return l+r;
	}
	
}
int main()
{
    ll n,q,i,l,r;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>arr[i];
    }
    build(0,n-1,1);
    cin>>q;
    for(i=0;i<q;i++)
    {
    	cin>>l>>r;
    	l--;
    	r--;
    	
    	cout<<query(1,0,n-1,l,r)<<endl;
    }
    return 0;
}
