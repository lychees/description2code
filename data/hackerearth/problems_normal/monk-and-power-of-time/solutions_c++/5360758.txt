#include <iostream>
using namespace std;

void rotateArray(int arr[], int n, int top)
{
	int i,temp=arr[top];
	for(i=top;i<n-1;i++)
	arr[i]=arr[i+1];
		
		arr[i]=temp;
}

int main()
{
    int N;
    int count=0;
    cin>>N;
    int given_array[N];
    int ideal_array[N];
    for(int i=0;i<N;i++)
    	cin>>given_array[i];
    
    for(int j=0;j<N;j++)
    	cin>>ideal_array[j];
    	
    int top=0;
    while(1)
    {
   		while(given_array[top]!=ideal_array[top]){
   			rotateArray(given_array,N,top);
   			count+=1;
   		}
   			if(given_array[top]==ideal_array[top])
		 	{
    		top=top+1;
    		count+=1;
		 	}
    		if(top==N)
    		break;
    }
    cout<<count;
    return 0;
}
