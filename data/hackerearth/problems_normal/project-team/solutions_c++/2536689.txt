#include<bits/stdc++.h>
using namespace std;

int main()
{
    //freopen("int.txt","r",stdin);
    //freopen("out.txt","w",stdout);
    ios::sync_with_stdio(false);

    int t,n,k[1002];
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(int i=0; i<n; i++)
        {
            cin>>k[i];
        }
        sort(k,k+n);
        int max=0;
        int min=INT_MAX;
        for(int i=0; i<n/2; i++)
        {
            int s=k[i]+k[n-1-i];
            if(max<s)
                max=s;
            if(min>s)
                min=s;
        }
        cout<<max-min<<endl;
    }
    return 0;
}
