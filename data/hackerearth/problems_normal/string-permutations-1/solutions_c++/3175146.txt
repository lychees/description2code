#include <bits/stdc++.h>
using namespace std;
int main()
{
	int test;
	cin>>test;
	map<string,int> M;
	map<string,int> ::iterator it;
	while(test--)
	{
		string s;
		cin>>s;
		if(M.find(s)==M.end())
			M[s]=1;
		else
			M[s]+=1;
	}
	int maxx=-1,ans=0;
	for(it=M.begin();it!=M.end();++it)
	{
		maxx=max(maxx,it->second);
	}
	//cout<<maxx<<endl;
	for(it=M.begin();it!=M.end();++it)
	{
		if(it->second==maxx)
			ans+=maxx;
	}
	cout<<ans<<endl;
	return 0;
}
