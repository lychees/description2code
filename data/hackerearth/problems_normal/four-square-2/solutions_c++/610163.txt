#include<stdio.h>
#include<math.h>
long long int fact (long long int);
int main()
{ long long int i,t,n,z,r;
scanf("%lld",&t);
for(i=1;i<=t;i++)
{

scanf("%lld",&n);
z=fact(n);
if(n%2!=0)
{
    r=8*z;
printf("%lld\n",r);

}
else
{
    r=24*z;
 printf("%lld\n",r);
}
}

return 0;
}

long long int fact(long long int a)
{
    long long int k,s=0,l;
  for(k=1; k<=sqrt(a); k++)
  {
  if (a%k==0)
   {

    l=a/k;
    if(l%2!=0)
    s+=l;
    if (l!=k && (k%2!=0))
    s+=k;
  }

  }
  return s;
  }
