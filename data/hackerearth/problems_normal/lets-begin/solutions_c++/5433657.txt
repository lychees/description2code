#include<bits/stdc++.h>
using namespace std;
int dp[1000009];
#define MAX 1000000
int calc (int n)
{
	if(n<0)
		return MAX;
	if(dp[n] != 0 || n == 0)
		return dp[n];
	dp[n] = 1 + min(min(calc(n-2),calc(n-3)), min(calc(n-5),calc(n-7)));
	return dp[n];
}
int main()
{
	int i,t;
	for(i=0;i<=MAX;i++)
		calc(i);
	scanf("%d",&t);
	while(t--) {
		int n;
		scanf("%d",&n);
		if(dp[n]>=MAX)
			printf("-1\n");
		else
			printf("%d\n",dp[n]);
	}
	return 0;
}
