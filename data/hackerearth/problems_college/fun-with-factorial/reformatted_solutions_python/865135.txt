p = [1]
n = 2
while (n <= 1000):
    p.append(p[len(p)-1]*n)
    n += 1
t = int(input())
while (t > 0):
    n = int(input())
    if (p[n-1] < 100000):
        s = str(p[n-1])
        s.split()
        x = len(s)
        print((5-x)*'0'+s)
    else:
        if (p[n-1] % 100000 == 0):
            print(5*'0')
        else:
            print(p[n-1] % 100000)
    t -= 1
