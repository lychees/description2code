#include <iostream>

using namespace std;

int main()
{
    int n;
    int M[25];
    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>M[i];
    }

for(int m=0;m<n;m++)
{


    long int one_count[M[m]];


    one_count[0] =0;
    one_count[1] = 1;

    for(int i=2;i<M[m];i++)
    {
        one_count[i] = one_count[i-1]+one_count[i-2];
    }

    cout<<one_count[M[m]-1]<<" ";
    if(M[m]==1)
    {
        cout<<1<<endl;
    }
    else if(M[m]==2)
    {
        cout<<1<<endl;
    }
    else
    {
        cout<<(one_count[M[m]-1]+one_count[M[m]-2])<<endl;
    }


}


}
