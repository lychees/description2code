for i in xrange(int(raw_input())):
    sum = []
    # taking the input in single line and converting elements to int type
    temp = [int(j) for j in raw_input().split()]
    # sorting the input list
    chk = temp.pop(0)
    if int(chk % 2) == 0:
        pass
    else:
        quit()
    # check if the entered array satisfies the size condition
    if len(temp) == chk:
        pass
    else:
        quit()
    temp.sort()
    # for the purpose of traversing
    l = len(temp)/2
    for k in xrange(l):
        sum.append(temp[l-1-k] + temp[l+k])
    sum.sort()
    fin.append(sum[-1]-sum[0])
