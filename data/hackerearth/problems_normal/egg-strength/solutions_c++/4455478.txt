#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define pii pair<int,int>
#define pll pair<ll,ll>
#define mp make_pair
#define ff first
#define ss second
#define pb push_back
#define pf pop_front
#define IOS ios::sync_with_stdio(false)
int t , n , k , dp[505][505];
int main(){

	int i , j;
    for(i = 1 ; i <= 501 ; i++){
        for(j = 1 ; j <= 501 ; j++){
            dp[i][j] = INT_MAX;
        }
    }
    for(i = 1 ; i <= 501 ; i++){
        dp[i][1] = i;
        dp[1][i] = 1;
    }
    for(i = 2 ; i <= 501 ; i++){
        for(j = 2 ; j <= 501 ; j++){
            for(k = 1 ; k <= i ; k++){
                dp[i][j] = min(dp[i][j] , 1+max(dp[k-1][j-1],dp[i-k][j]));
            }
        }
    }
    cin >> t;
    while(t--){
        cin >> n >> k ;
        cout << dp[n][k] << endl;
    }

	return 0;
}
