#include<bits/stdc++.h>
#define mod 1000000007 // 10**9 + 7
using namespace std;
typedef long long ll;
typedef long l;
ll q[1005];
ll d1[1005];
ll d2[1005];

int main()
{
    int t;
    q[0] = 1;
    for (int i = 1; i < 1005; ++ i)
    {
        for (int j = 1; j <= 4; ++ j)
        {
            if (i - j >= 0)
            {
                q[i] += q[i - j];
                q[i] %= mod;
            }
        }
    }
    cin>>t;
    while(t--)
    {
        int a, b;
        cin>>b>>a;
        memset(d1, 0, sizeof(d1));
        memset(d2, 0, sizeof(d2));
        d1[0] = 1;
        d2[0] = 1;
        for (int i = 1; i <= a; ++ i)
        {
            ll v = 1;
            l x = q[i];
            for (int j = 1; j <= b; j <<= 1)
            {
                if (j & b) (v *= x) %= mod;
                (x *= x) %= mod;
            }
            d2[i] = v;
            for (int j = 1; j < i; ++ j)
            {
                v -= d1[j] * d2[i - j];
                ((v %= mod) += mod) %= mod;
            }
            d1[i] = v;
        }
        cout<<d1[a]<<endl;
    }
    return 0;
}
