#include <bits/stdc++.h>
using namespace std;
#define p 1000000007
#define ll long long

ll fp(ll x,ll n)
{
	if(n==1)
		return x%p;
	if(n==0)
		return 1;
	ll y=fp((x%p),n/2);
	if(n%2)
		return ((x%p)*(y%p)*(y%p))%p;
	else
		return ((y%p)*(y%p))%p;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	ll sum=0;
    	ll n;
    	cin>>n;
    	for(ll i=1;i<=n;i++)
    	{
    		sum+=i;
    		sum%=p;
    	}
    	ll power=fp(2,n-1);
    	cout<<(sum*power)%p<<endl;
    }
    return 0;
}
