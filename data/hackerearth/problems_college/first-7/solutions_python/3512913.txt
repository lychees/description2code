t = input()
A = [0,1,4]
for i in xrange(100000):
    A+=[A[-1]+4]
A1=A[:]
A2=A[:]
for i in xrange(3,len(A)):
    A1[i]+=A1[i-2]
    A2[i]+=A2[i-2]
for i in xrange(1,len(A)):
    A2[i]*=2
for i in xrange(3,len(A)):
    A2[i]+=A2[i-2]
for _ in xrange(t):
    n = input()
    if(n==1):
        print "1.000000"
    else:
        den = n*n
        ans = 0
        ans = A1[n]*(3*n - 2)
        ans += A2[n-2]
        prob = ans/float(den)
        print "%0.6f"%prob
