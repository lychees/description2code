#define MAX(a,b) (a>b?a:b)
typedef struct{int s,q;}sw;
bool cmp(sw a,sw b)
{
    return a.s>b.s?true:false;
}
int rec(int *a,int st,int en,int sum)
{
    if(st>en)
    return 0;
    for(int i=st;i<=en;i++)
    {
        if(sum-a[i]==0)
        return 1;
        else
        {
            int t=rec(a,i+1,en,sum-a[i]);
            if(t)
            return t;
            else
            return rec(a,i+1,en,sum);
        }
    }
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        int a[n],s;
        for(int i=0;i<n;i++)
        {
            cin>>a[i];
        }
        cin>>s;
        if(rec(a,0,n-1,s))
        cout<<"YES"<<endl;
        else
        cout<<"NO"<<endl;
    }
    return 0;
}
