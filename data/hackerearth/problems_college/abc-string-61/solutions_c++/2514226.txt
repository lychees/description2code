#include <iostream>
#include <assert.h>
using namespace std;
#define ll	long long int
#define MOD	1000000007
ll pow_mod(ll a, ll b)
{
	ll res=1;
	while(b)
	{
		if(b&1)
			res=(res*a)%MOD;
		a=(a*a)%MOD;
		b>>=1;
	}
	return res;
}
int main()
{
	int t;
	scanf("%d",&t);
	assert(t>=1 && t<=100000);
	while(t--)
	{
		ll n;
		scanf("%lld",&n);
		assert(n>=1 && n<=1000000);
		ll ans=pow_mod(3,n);
		ans=ans-3*pow_mod(2,n);
		while(ans<0)
			ans=ans+MOD;
		ans=(ans+3)%MOD;
		assert(ans>=0 && ans<MOD);
		printf("%lld\n",ans);
	}
	return 0;
}
