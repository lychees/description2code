#include <bits/stdc++.h>
#include<string.h>
using namespace std;
char a[100005],b[100005];
int main()
{
   // cout << "Hello World!" << endl;
   int t,i;
   scanf("%d",&t);
   while(t--)
   {
   	scanf("%s%s",&a,&b);
   	int l1,l2;
   	l1 = strlen(a);
   	l2 = strlen(b);
   	int ma = max(l1,l2);
   	reverse(a,a+l1);
   	reverse(b,b+l2);
   	int total = 0;
   	int ca = 0;
   	for(i=0;i<ma;i++)
   	{
   		int n,m;
   		n = m = 0;
   		if(i<l1)m = a[i] -'0';
   		if(i < l2)n = b[i]-'0';
   		n = n+m+ca;
   		if(n>=10){
   			total++;
   			ca = 1;
   		}
   	}
   if(total==0)printf("No carry operation\n");
   else if(total==1)printf("1 carry operation\n");
   else printf("%d carry operations\n",total);
   }
    return 0;
}
