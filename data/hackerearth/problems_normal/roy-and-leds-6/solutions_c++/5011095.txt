#include<iostream>
using namespace std;
main()
{

int t,r,g,b;
cin>>t>>r>>g>>b;
int r1=r,g1=g,b1=b;
bool rstate,gstate,bstate;
int red=0,green=0,blue=0,yellow=0,cyan=0,magneta=0,white=0,black=0;
int x=2,y=2,z=2;;
for(int i=1;i<=t;i++)
{
    if(r>0)
     {
       rstate=false;
       r--;
       x=1;
     }
    if(r<0)
    { x=2;
     rstate=true;
     r++;
    }
     if(r==0&&x==1)
    {
       r=-1*r1;
    }
    if(r==0&&x==2)
    {
       r=1*r1;
    }
    if(g>0)
    {
      gstate=false;
      g--;
       y=1;
    }
    if(g<0)
    { y=2;
     gstate=true;
     g++;
    }
     if(g==0&&y==1)
    {
       g=-1*g1;
    }
    if(g==0&&y==2)
    {
       g=1*g1;
    }

    if(b>0)
    {
      bstate=false;
      b--;
      z=1;
    }
    if(b<0)
    { z=2;
     bstate=true;
     b++;
    }
     if(b==0&&z==1)
    {
       b=-1*b1;
    }
    if(b==0&&z==2)
    {
       b=1*b1;
    }

    if(rstate&&!gstate&&!bstate)
         red++;
    else if(!rstate&&gstate&&!bstate)
       green++;
       else if(!rstate&&!gstate&&bstate)
        blue++;
        else if(rstate&&gstate&&!bstate)
        yellow++;
        else if(!rstate&&gstate&&bstate)
        cyan++;
        else if(rstate&&!gstate&&bstate)
        magneta++;
        else if(rstate&&gstate&&bstate)
        white++;
        else if(!rstate&&!gstate&&!bstate)
        black++;

        //cout<<rstate<<"  "<<gstate<<"  "<<bstate<<endl;

  }

  cout<<red<<" "<<green<<" "<<blue<<" "<<yellow<<" "<<cyan<<" "<<magneta<<" "<<white<<" "<<black;

}
