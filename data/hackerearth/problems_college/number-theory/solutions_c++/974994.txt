#include<iostream>
#include<cmath>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
long int n,k;
cin>>n;
int a[10]={0};
k=n;
while(k)
{
a[k%10]=1;
k=k/10;
}
long int count=0,i,j;
for(i=1;i<=sqrt(n);i++)
{
if(n%i==0)
{
j=i;
k=n/i;
while(j)
{
if(a[j%10]==1)
{
count+=1;
j=0;
}
j=j/10;
}
if(k!=i)
{
	j=k;
while(j)
{
if(a[j%10]==1)
{
count+=1;
j=0;
}
j=j/10;
}
}
}
}
cout<<count<<endl;
}
return 0;
}
