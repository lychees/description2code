#include<iostream>
using namespace std;
long long a[1000][1000];
main()
{

    unsigned long long i,j,k,l,temp2,temp,sum=0,t,n;
    string s[1000],c[1000];
    cin>>t;
    for(i=0;i<t;i++)
    {
        cin>>s[i];
    }
    for(i=0;i<t;i++)
    {
        for(j=0;j<t;j++)
        {
            cin>>a[i][j];
        }
    }
    cin>>n;
    temp2=0;
    for(i=0;i<n;i++)
    {
        cin>>c[i];
        for(j=0;j<t;j++)
        {
            if(c[i]==s[j])
            {
                temp=j;
                break;
            }
        }
        sum=sum+a[temp2][temp];
        temp2=j;

    }
    cout<<sum<<endl;


}
