#include <iostream>
using namespace std;
int fib(int a)
{
	if(a==1)
	return 2;
	if(a==2)
	return 3;
	else
	return (fib(a-1)+fib(a-2));
}	
int main()
{
    //cout << "Hello World!" << endl;
    int t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	int res=fib(n);
    	res=res*res;
    	cout<<res<<"\n";
    }
    return 0;
}
