#include <bits/stdc++.h>
using namespace std;

#define MAXN 100010
#define MOD 1000000007LL
#define ll long long

int main()
{
	int t;
	string s;
	cin >> t;
	while(t--)
	{
		cin >> s;
		ll res = 0, cnt = 0;
		for(int i=0;i<s.size();++i)
		{
			if((s[i]>='a' and s[i]<='c') or (s[i]>='A' and s[i]<='C'))
			{
				++cnt;
				res += cnt;
			}
			else
				cnt = 0;
		}
		cout << res << endl;
	}
	return 0;
}
