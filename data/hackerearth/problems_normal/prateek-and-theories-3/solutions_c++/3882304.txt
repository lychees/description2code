#include <bits/stdc++.h>
using namespace std;
map <int,int> arr;
void init()
{
	arr.clear();
}
void solve()
{
	init();
	int n;
	scanf("%d",&n);
	int t1,t2;
	for(int i=0;i<n;i++)
	{
		scanf("%d %d",&t1,&t2);
		arr[t1]++;
		arr[t2]--;
	}
	int sum=0;
	int ans=0;
	for(map <int,int>::iterator it=arr.begin();it!=arr.end();it++)
	{
		sum+=(*it).second;
		ans=max(ans,sum);
	}
		printf("%d\n",ans);
}
int main()
{
    int t;
    scanf("%d",&t);
    for(int i=0;i<t;i++)
    	solve();
    return 0;
}
