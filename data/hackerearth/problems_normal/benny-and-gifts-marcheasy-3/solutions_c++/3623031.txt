#include <iostream>
#include<set>
using namespace std;

int main()
{
    string s;
    cin>>s;
    int cnt = 0;
    pair<int,int> pos;
    set<pair<int,int>> icy;
    icy.emplace(pos);
    for(int i=0;i<s.size();i++){
    	switch(s[i]){
    		case 'L': --pos.first;
    		break;
    		case 'R': ++pos.first;
    		break;
    		case 'U': --pos.second;
    		break;
    		case 'D': ++pos.second;
    		break;
    	}
    	if(icy.count(pos)) ++cnt;
    	else icy.emplace(pos);
    }
    cout<<cnt;
    return 0;
}
