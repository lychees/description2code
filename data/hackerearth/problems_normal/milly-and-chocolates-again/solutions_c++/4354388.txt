#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int nrboxes;
    cin >> nrboxes;
    vector<int> boxnrs;
    vector<int> nrchocolates( nrboxes, 0 );
    int totalnrchocolates = 0;
    for ( int idx=0; idx<nrboxes; idx++ )
    {
    	cin >> nrchocolates[idx];
    	totalnrchocolates += nrchocolates[idx];
    	boxnrs.resize( totalnrchocolates, idx+1 );
    }
    
    int nrqueries;
    cin >> nrqueries;
    while ( nrqueries-- )
    {
    	int chocolateidx;
    	cin >> chocolateidx;
    	cout << boxnrs[chocolateidx-1] << "\n";
    }
    
    return 0;
}
