def f (n,r,m):
	fac=1
	rfac=1
	for i in range(1,r+1):
		fac=(fac*(n-r+i))
		rfac=rfac*i
	return (fac/rfac)%m

	t-=1
	n,m=map(int,raw_input().split())
	l=[int(i) for i in raw_input().split()]
	s=m-sum(l)
	if s==0:
		print "1"
	else:
		if s<0:
			print "0"
		else:
			print f(n+s-1,n-1,1000000007)
