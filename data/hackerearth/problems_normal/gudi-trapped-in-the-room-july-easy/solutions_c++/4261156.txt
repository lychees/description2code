#include <bits/stdc++.h>
using namespace std;
int main() {
  int T;
  scanf("%d", &T);
  for (int t = 0; t < T; t++) {
    string s;
    int n, a, h;
    cin >> s >> a >> h;
    queue<string> q;
    q.push(s);
    string ans = s;
    set<string> v;
    int len = s.length();
    h %= len;
    while (!q.empty()) {
      string f = q.front();
      q.pop();
      ans = min(ans, f);
      string g = f;
      for (int i = 1; i < len; i += 2) {
        f[i] += a;
        if (f[i] > '9') f[i] -= 10;
      }
      if (v.count(f) == 0) {
        v.insert(f);
        q.push(f);
      }
      g = g.substr(len - h) + g.substr(0, len - h);
      if (v.count(g) == 0) {
        v.insert(g);
        q.push(g);
      }
    }
    cout << ans << endl;
  }
  return 0;
}
