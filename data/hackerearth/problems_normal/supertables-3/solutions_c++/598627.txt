#include<iostream>
using namespace std;
long gcd(long a,long b){
    if(b==0) return a;
    else return gcd(b,a%b);
}
long lcm(long a,long b){
    return (a*b)/(gcd(a,b));
}
long swap(long *a,long *b){
    long temp=*a;
    *a=*b;
    *b=temp;
}
long supTab(long a,long b,long n){
    long k,g,y,z;
    if(a%b==0){long g=a*n;return g;}
    else{
        long c=lcm(a,b);
        long d=(c/a)+(c/b)-1;
        long e=(n/d);
        g=c*e;
        long h=d*e;
        if(h>n) return 999999;
        if(h==n) return g;
        if(h<n)
        {
            k=n-h;
            y=g;z=g;
            for(long i=0,j=0,x=0;i<k&&j<k&&x<k;x++){
                if((y+a)<(z+b)){
                    y=y+a;
                    i++;
                    g=y;
                }
                else{
                    z=z+b;
                    j++;
                    g=z;
                }
            }
            return g;
        }
    }
}
int main(int argc, char *argv[])
{
    long a,b,n,T;
    cin>>T;
    if(!(T>=1&&T<=300000)) return 1;
    for(long i=0;i<T;i++){
    cin>>a>>b>>n;
    if(!(a>=1&&a<=1000)&&!(b>=1&&b<=1000)&&!(n>=1&&n<=1000000000))return 2;
    if(a>b) swap(&a,&b);
    cout<<supTab(a,b,n)<<endl;
    }
}
