#include<bits/stdc++.h>
using namespace std;
int G[501];
int main()
{
    int N;
    cin>>N;
    for(int i=0 ; i<N ; i++)
    {
        int M;
        cin>>M;
        int P[501]={0};
        for (int j=0 ; j<M ; j++)
        {
            int a;
            cin>>a;
            if(P[a]==0)
                G[a]=G[a]+1;
            P[a]=1;
            //cout<<a<<" "<<G[a]<<endl;
        }
    }
    long long uni=0,diff=0;
    for (int i=1 ; i<=500 ; i++)
    {
        //cout<<i<<" "<<G[i]<<endl;
        if(G[i]!=0)
            uni+=i;
        if(G[i]!=N && G[i]!=0)
            diff+=i;
    }
    //cout<<uni<<" "<<diff<<endl;
    cout<<diff-uni<<endl;
}
