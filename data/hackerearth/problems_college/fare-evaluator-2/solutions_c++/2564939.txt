#include<cstdio>
#include<iostream>

using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,m;
		cin>>n>>m;

		if(n>60)
		{
			cout<<"NA"<<"\n";
		}
		else
		if(n==1)
		{
			cout<<50+ m*2<<"\n";
		}
		else
		{
			int sum =0;
			sum = sum + 50;
			sum = sum + (n-2)*12;
			sum = sum + m*2;
			cout<<sum<<"\n";
		}
		
	}
	
	return 0;
}
