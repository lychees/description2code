testcase_cnt = int(raw_input(''))

def yesno(result, arr):
	combinations = []
	
	for n in arr:
	    new_combinations = []
	
	    for combination in combinations:
	        new_combinations.append(list(combination) + [n])
	
	    if len(combinations) == 0:
	        combinations.append([n])
	        combinations.append([0])
	
	    combinations += new_combinations
	
	for combination in combinations:
	    if sum(combination) == result:
	        print "YES"
	        return
        print "NO"
        
for _ in range(testcase_cnt):
	arr_cnt = raw_input()
	arr = set(int(element) for element in raw_input().split())
	result = int(raw_input())
	
	yesno(result, arr)
