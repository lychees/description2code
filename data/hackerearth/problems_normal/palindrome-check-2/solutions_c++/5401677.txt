#include <iostream>
using namespace std;

int main()
{
    string s;
    int len,mid,count =1,i,j;
    getline(cin,s);
    len = s.length();
    mid = len/2;
    for(i=0,j = len-1 ; i<=mid ; i++,j--){
    	if(s[i] != s[j])
    	count =0;
    }
    if(count ==0)
    	cout<<"NO";
    else
    	cout<<"YES";
    return 0;
}
