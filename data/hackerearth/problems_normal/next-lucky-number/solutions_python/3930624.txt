def next_lucky(n):
    if len(n) == 0:
        return ''
    if int(n[0]) < 3:
        return '3'*len(n)
    elif int(n[0]) == 3:
        pos1 = int(n[0]+next_lucky(n[1:]))
        pos2 = int('5'+'3'*(len(n)-1))
        if pos1 > pos2: pos1,pos2 = pos2,pos1
        if pos1 > int(n):
            return str(pos1)
        else:
            return str(pos2)
    elif int(n[0]) < 5:
        return '5'+'3'*(len(n)-1)
    elif int(n[0]) == 5:
        pos1 = int(n[0]+next_lucky(n[1:]))
        pos2 = int('3'*(len(n)+1))
        if pos1 > pos2: pos1,pos2 = pos2,pos1
        if pos1 > int(n):
            return str(pos1)
        else:
            return str(pos2)
    elif int(n[0]) > 5:
        return '3'*(len(n)+1)
