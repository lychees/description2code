#include<bits/stdc++.h>

using namespace std;
bool chck[100000];
int main()
{
	set<int64_t>s;
	long long int t,n,val,cnt;
	
	cin>>t;
	while(t--)
	{
		cin>>n;
		memset(chck,0,sizeof(chck));
		cnt=0;
		for(int i=0;i<n;++i)
		{
			cin>>val;
			if(chck[val]==0)
			{
				chck[val]=1;
				cnt++;
			}
		}
		cout<<cnt<<endl;
		
	}
	return 0;
}
