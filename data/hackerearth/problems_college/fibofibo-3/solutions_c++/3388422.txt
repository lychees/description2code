#include <iostream>
using namespace std;

int main()
{
    int t,n,i;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	int z[n+1],o[n+1];
    	z[0]=0;
    	o[0]=0;
    	z[1]=0;
    	o[1]=1;
    	z[2]=1;
    	o[2]=1;
			for(int i=3; i<=n; ++i)
			{
				z[i]=z[i-1]+z[i-2];
				o[i]=o[i-1]+o[i-2];
				
			}
			cout<<z[n]<<" "<<o[n];
    	cout<<endl;
    }
    return 0;
}
