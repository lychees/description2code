#include <iostream>
#include <math.h>
#include <queue>

#include <stdio.h>

#include <string.h>
#include <string>
using namespace std;
struct xx {
  int x, y;
} a[222];
int main() {
  freopen("input.txt", "r", stdin);
  freopen("output.txt", "w", stdout);
  int n, m;
  while (scanf("%d%d", &n, &m) != EOF) {
    int k, i, j, z, x, y, sum, max, min;
    scanf("%d", &k);
    for (i = 0; i < k; i++) {
      scanf("%d%d", &a[i].x, &a[i].y);
    }
    max = -1;
    x = 1;
    y = 1;
    for (i = 1; i <= n; i++) {
      for (j = 1; j <= m; j++) {
        for (min = 99999999, z = 0; z < k; z++) {
          sum = abs(a[z].x - i) + abs(a[z].y - j);
          if (min > sum)
            min = sum;
        }
        if (max < min)
          max = min, x = i, y = j;
      }
    }
    printf("%d %d\n", x, y);
  }
  return 0;
}
