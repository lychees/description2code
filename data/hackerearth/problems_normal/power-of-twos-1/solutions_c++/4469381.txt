#include <cstdio>
#include <cmath>
int main()
{
    int T, N, i, j;
    unsigned long long int ans[1000001];
    
    ans[0]=0;
    for(i=1;i<=1000000;i++)
     	for(j=2*i;j<=1000000;j+=i)
        	ans[j]++;
    
    for(i=1;i<=1000000;i++)
       ans[i] += ans[i-1];
       
    scanf("%d",&T);
    
    while(T--)
    {
    	scanf("%d",&N);
    	printf("%llu\n",ans[N]);
    }
    
    
    return 0;
}
