#include <iostream>
using namespace std;

int main()
{
    int T, N, st_in, en_in, curr, prev, i, fl, st_v, end_v;
    
    cin>>T;
    while(T--)
    {
    	st_in = en_in = -1;
    	fl = 0;
    	cin>>N;
    	cin>>prev;
    	for(i = 2; i <= N; i++)
    	{
    		cin>>curr;
    		if( fl==0 && curr < prev )
    		{
    			fl = 1;
    			st_in = i-1;
    			st_v = prev;
    		}
    		else if( fl==1 && prev < curr)
    		{
    			fl = 2;
    			if( curr < st_v )
    				fl = 3;
    			en_in = i-1;
    		}
    		else if(fl==2 && curr < prev)
    		{
    			fl = 3;
    		}
    		prev = curr;
    	}
    	if( fl==1 )
    		en_in = N;
    	if( fl > 2 )
    		cout<<"-1 -1"<<endl;
    	else
    		cout<<st_in<<" "<<en_in<<endl;
    }
    return 0;
}
