#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t, n, mx=0;
    cin>>t;
    while(t--){
    	cin>>n;
        int c=0;
        for(int i=1;i<=n;i++)
            for(int j=1;j<=i;j++)
                if(i*j<=n)
                    c++;
            cout<<c<<"\n";
            mx=n;
    }
    return 0;
}
