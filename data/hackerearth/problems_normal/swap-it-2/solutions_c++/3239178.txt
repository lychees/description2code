#include<bits/stdc++.h>
using namespace std;
int main()
{
       int t;
     scanf("%d",&t);
       while(t--)
       {
       	
       	
       	
       	
       	
       	                  long int a[1005];
               int n,m;
               scanf("%d%d",&n,&m);
            
               for(int i=0;i<n;i++)
               {
                  scanf("%ld",&a[i]);
               }

              int p;
               for(int i=0;i<n;i++)
               {
                       p=i;

                      for(int j=i+1;j<n;j++)
                      {

                              if(j-i>m)
                              {
                                  break;
                              }
                              if(a[j]<=a[p])
                              {
                                    p=j;
                              }
                      }
                  //    cout<<p<<endl;
                      for(int k=p;k>i;k--)
                      {
                            swap(a[k],a[k-1]);
                           
                      }
                      m-=(p-i);
               }


               for(int i=0;i<n;i++)
               {
            cout<<a[i]<<" ";
               }
               printf("\n");
       }
       return 0;

}
