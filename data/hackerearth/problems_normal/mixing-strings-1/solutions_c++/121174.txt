#include<cstdio>
#include<iostream>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<list>
#include<utility>
#include<set>
#include<vector>

using namespace std;

#define INF 10000001
#define MAXS 100001
#define MOD 1000000007ULL

typedef unsigned long long uint64;

int N;
char str[10][20];
int length[10];
int mat[10][10]={0};
int ans=INF;


int main()
{
	ios::sync_with_stdio(false);
	cin>>N;
	for(int i=0;i<N;++i){
		cin>>str[i];
		length[i]=strlen(str[i]);
	}
	int min=10000001,ct;
	for(int i=0;i<N;++i){
		for(int j=0;j<N;++j){
			int idx1=0;
			while(idx1<length[i]){
				bool flag=true;
				while(str[i][idx1]!=str[j][0] && idx1<length[i])	idx1++;
				if(idx1==length[i])	break;
				int idx2=0,count=0;
				for(int k=idx1;k<length[i];++k,idx2++)
					if(str[i][k]==str[j][idx2])
						count++;
					else{
						flag=false;
						break;
					}
				if(flag){
					mat[i][j]=count;
					break;
				}
				idx1++;
			}
		}
	}
	for(int i=0;i<N;++i){
		int idx_table[10]={0},c=N-1,ch=i;
		int temp_ans=length[ch];
		idx_table[ch]=-1;
		while(c--){
			int max=-1,tt;
			for(int j=0;j<N;++j)
				if(idx_table[j]!=-1 && mat[ch][j]>max){
					max=mat[ch][j];
					tt=j;
				}
			temp_ans+=length[tt]-mat[ch][tt];
			ch=tt;
			idx_table[ch]=-1;
		}
		if(ans>temp_ans)	ans=temp_ans;
	}
	cout<<ans;
	//system("pause");
	return 0;
}
