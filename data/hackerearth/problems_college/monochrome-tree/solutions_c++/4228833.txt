int a[N],b[N],dp[N][2],ans[N];
vector<int> g[N];
void dfs(int z,int p) {
    REP(i,0,g[z].size()) {
        if(g[z][i]!=p) {
            dfs(g[z][i],z);
        }
    }
    REP(i,0,g[z].size()) {
        if(g[z][i]!=p) {
            dp[z][0]+=min(dp[g[z][i]][0],dp[g[z][i]][1]+1);
            dp[z][1]+=min(dp[g[z][i]][1],dp[g[z][i]][0]+1);
        }
    }
        if(a[z]==b[z])
            dp[z][0]++;
        else
            dp[z][1]++;
        ans[z]=min(dp[z][0]+1,dp[z][1]);
}
