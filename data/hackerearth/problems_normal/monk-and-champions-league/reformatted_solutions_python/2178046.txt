m, n = map(int, raw_input().split())
seat = map(int, raw_input().split())
ans = 0
hmap = []
hmap.append(0)
for i in seat:
    for j in range(i):
        try:
            hmap[j+1] += 1
        except BaseException:
            for k in range(len(hmap), j+1):
                hmap.append(0)
            hmap.append(1)
for i in range(len(hmap)-1, 0, -1):
    if (hmap[i] <= n):
        ans += i * hmap[i]
        n -= hmap[i]
    else:
        ans += i * n
        break
print ans
