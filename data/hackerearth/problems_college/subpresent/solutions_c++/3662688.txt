#include<bits/stdc++.h>'
using namespace std;
int main()
{
	string str;
	cin>>str;
	int l1  = str.length();
	int k;
	cin>>k;
	int t;
	cin>>t;
	int x = 0;
	while(t--)
	{
		string str1;
		cin>>str1;
		int l2 = str1.length();
		int i = 0;
		int j = 0;
		int c = 0;
		while( i < l1 && j < l2)
		{
			if( str1[j] == str[i] )
			{
				c++;
				i++;
				j++;
			}
			else
			{
				i++;
			}
		}
		if( c == l2)
		{
			x++;
		}
	}
	if( x >= k)
	{
		cout<<x<<" "<<"LUCKY"<<"\n";
	}
	else
	{
		cout<<x<<" "<<"UNLUCKY"<<"\n";
	}
}
