#include <iostream>
using namespace std;
#define MOD 1000000007

unsigned long int fact[2000001];

void init()
{
	fact[0] = fact[1] = 1;
	for(int i = 2; i<2000001; i++)
		fact[i] = (fact[i-1]*i)%MOD;
}

unsigned long int pow(unsigned long int a, unsigned long int b)
{
    unsigned long int x = 1, y = a;
    while(b > 0)
    {
        if(b%2 == 1)
        {
            x=(x*y)%MOD;
        }
        y = (y*y)%MOD;
       
        b /= 2;
    }
    return x;
}
unsigned long int InverseModulo(unsigned long int a, unsigned long int m)
{
    return pow(a,m-2);
}

int main()
{
	int t,n;
	unsigned long int answer;
    init();
    scanf("%d",&t);
    while (t--){
    	scanf("%d",&n);
    	answer = ((fact[2*n]*InverseModulo(fact[n],MOD)))%MOD;
    	answer=(answer*InverseModulo(fact[n+1],MOD))%MOD;
        cout<<answer<<endl;
    }
    return 0;
}
