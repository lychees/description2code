#include <iostream>
using namespace std;

int main()
{
    long long int T,L,R,K;
    cin>>T;
    while(T--!=0){
    	cin>>L;
    	cin>>R;
    	cin>>K;
    	long long int count = 1;
			for(long long int i=L; i<R;i++){
				for(long long int j=((i*K)+1);j<=R;j++){
					if(i<=(j *K) && (j<=(i * K)) ){
						
					}else{
						++count;
						i=j-1;
						break;
					}
				}
			}
			cout<<count<<'\n';
    }
}
