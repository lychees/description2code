#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t,n,x,y;
    cin>>t;
    while(t--){
    	int dp[1001][1001]={0};
    	int a[1001]={0},b[1001]={0};
    	cin>>n;
    	for(int i=0;i<n;i++){
    		cin>>x>>y;
    		a[x]=i+1;
    		b[y]=i+1;
    	}
    	for(int i=1;i<=n;i++){
    		for(int j=1;j<=n;j++){
    		     if(a[i]==b[j]){
    		     	dp[i][j]=dp[i-1][j-1]+1;
    		     }
    		     else if(dp[i-1][j]>dp[i][j-1]){
    		     	dp[i][j]=dp[i-1][j];
    		     }
    		     else{
    		     	dp[i][j]=dp[i][j-1];
    		     }
    		}
    	}
    	cout<<dp[n][n]<<"\n";
    }
    return 0;
}
