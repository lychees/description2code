#include <iostream>
using namespace std;
int dp[10010];
int main()
{
    int t;
    cin>>t;
    int n,i,j;
    dp[1]=1;
    	  dp[2]=0;
    	  dp[3]=1;
    	  dp[4]=1;
    	  
    	  for(i=5;i<10010;i++)
    	  {
    	  	if(dp[i-1]==0 || dp[i-4]==0)
    	  	  dp[i]=1;
    	  	  else
    	  	  dp[i]=0;
    	  	  
    	  	
    	  }
    	  
    
    while(t--)
    {
    	  cin>>n;
    	  
    	  if(dp[n]==1)
    	  cout<<"Yes\n";
    	  else
    	   cout<<"No\n";
    	  
    	  
    	
    }
}
