n = input()
a = []
t = []
d = []
for _ in range(n):
    x, y = map(int, raw_input().split())
    t.append(x)
    d.append(y)
    a.append(x+y)
a.sort()
c = []
# print t,d,a
for i in range(n):
    for j in range(n):
        if (t[j]+d[j] == a[i]):
            u = j+1
            if (u not in c):
                c.append(u)


print " ".join(map(str, c))
