#include<bits/stdc++.h>
using namespace std;

void swap (char *x, char *y)
{
    char temp;
    temp = *x;
    *x = *y;
    *y = temp;
}

void permute(char *a, int i, int n)
{
   int j;
   if (i == n)
     printf("%s\n", a);
   else
   {
        for (j = i; j <= n; j++)
       {
          swap((a+i), (a+j));
          permute(a, i+1, n);
          swap((a+i), (a+j)); //backtrack
       }
   }
}


int main()
{
   char a[1001];
   cin>>a;
   long int n,l,r;
   cin>>n;
   while(n--)
   {
       cin>>l>>r;
   permute(a, l, r);
   cout<<"\n";
   }
   return 0;
}
