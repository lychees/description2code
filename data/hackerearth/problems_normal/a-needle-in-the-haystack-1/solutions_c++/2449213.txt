    #include <iostream>
    #include <string>
    using namespace std;
     
    void reverseChar(char* str, int len)
    {
    for(int i=0; i<len/2; i++)
    {
    std::swap(str[i], str[len-i-1]);
    }
    }
     
    int main()
    {
    int t,f1,f2;
    string s1,s2,tmp;
    cin >> t;
    while(t--)
    {
    cin >> s1;
    cin >> s2;
    tmp = s1;
    reverseChar(&tmp[0],tmp.size());
    f1 = s2.find(s1);
    f2 = s2.find(tmp);
    if(f1!=-1 || f2!=-1)
    {
    cout << "YES" << endl;
    }
    else
    {
    cout << "NO" << endl;
    }
    }
    }
