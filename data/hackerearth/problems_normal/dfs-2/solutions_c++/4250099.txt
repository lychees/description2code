#include <bits/stdc++.h>
using namespace std;

int main()
{
	long long n,ele,query,p,q,L,R;
	cin>>n;
	vector<long long> a;
	while(n--)
	{
		cin>>ele;
		a.push_back(ele);
	}
	sort(a.begin(),a.end());
	cin>>query;
	while(query--)
	{
		cin>>p>>q;
		L=lower_bound(a.begin(),a.end(),p)-a.begin();
		R=upper_bound(a.begin(),a.end(),q)-a.begin();
		if(L==n||R==0)
		cout<<0<<endl;
		else
		cout<<R-L<<endl;
	}
    return 0;
}
