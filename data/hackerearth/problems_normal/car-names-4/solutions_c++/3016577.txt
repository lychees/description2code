#include <iostream>
#include<string>
using namespace std;

int main()
{
    int n;
    cin>>n;
    while(n--){
    	string s;
    	cin>>s;
    	int l = s.length();
    	if(l%3 != 0)
    		cout<<"Not OK\n";
    	else {
    		char c1,c2,c3;
    		c1 = s[0];
    		c2 = s[l/3];
    		c3 = s[2*l/3];
    		if(c1 == c2 || c2 ==c3 || c1 ==c3)
    			cout<<"Not OK\n";
    		else{
    			int i;
	    		for(i=0;i<l/3;i++){
	    			if(c1 != s[i] || c2 != s[l/3 + i] || c3 != s[2*l/3 + i]){
	    				cout<<"Not OK\n";
	    				break;
	    			}
	    		}
	    		if(i==l/3)
	    			cout<<"OK\n";
    		}
    	}
    }
}
