#include <iostream>
#include <math.h>

using namespace std;

int main()
{
        int T;
        cin >> T;
        while(T--) {
                double A,B,C, area, radius;
                cin >> A >> B >> C;
                double s = (A+B+C)/2;
                area = sqrt(s*(s-A)*(s-B)*(s-C));
                radius = (A*B*C)/(4*(area));
                area = 3.1415*radius*radius;
                area = area*10000 + 0.5;
                long long int iarea = area;
                area = iarea;
                printf("%1.4f\n", area/10000);
        }
}
