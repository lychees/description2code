#include <bits/stdc++.h>
using namespace std;
int main()
{
    long long n,a[1010];
    cin>>n;
    long long q=1000000007;
    for(long long i=0;i<n;i++)
		 cin>>a[i];
		 sort(a,a+n);
    long long x=1;
    for(long long i=0;i<n;i++)
    {
    	x=x*a[i];
    	if(x>q)
    	{
    		x=x%q;
    	}
    }
    cout<<x%q<<endl;
    return 0;
}
