#include<bits/stdc++.h>
using namespace std;
#define lx long long
#define INF 2000000009
int main()
{
    ios::sync_with_stdio(0);
    //freopen("input.txt","r",stdin);
    int t;
    cin>>t;
    while(t--)
    {
        int n,k;
        cin>>n>>k;
        vector <int> v;
        v.resize(n);
        for (int i=0;i<n;i++)
        {
            cin>>v[i];
        }
        sort(v.begin(),v.end());
        int mx = -INF, mn = INF;
        for (int i=0;i<n/2;i++)
        {
            mx = max(mx, v[i]+v[n-i-1]);
            mn = min(mn, v[i]+v[n-i-1]);
        }
        int ans = mx - mn;
        cout<<ans<<endl;
        if (ans<k)
            cout<<"Chick magnet Jhool!";
        else if (ans==k)
            cout<<"Lucky chap!";
        else
            cout<<"No more girlfriends!";
        cout<<endl;
    }
    return 0;
}
