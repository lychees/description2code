using namespace std;
int t;
#define MAX 52
long long int len[MAX];
vector<string> v;
int main(){
	scanf("%d", &t);
	while (t--){
		int n;
		int m;
		scanf("%d%d", &n, &m);
		v.clear();
		for (int i = 0; i < n; i++){
			string kari;
			cin >> kari;
			v.push_back(kari);
		}
		len[0] = v[0].size();
		for (int i = 1; i < n; i++){
			len[i] = len[i - 1] * 2LL + v[i].size();
		}
		while (m--){
			long long int a;
			scanf("%lld", &a);
			a++;
			int ind = n - 1;
			bool outt = false;
			while (ind > 0){
				long long int prev = len[ind - 1] * 2LL;
				if (a > prev){
					a -= prev;
					a--;
					outt = true;
					cout << v[ind][a];
					break;
				}
				prev -= len[ind - 1];
				if (a > prev){
					a -= prev;
					a = prev - a;
					a++;
				}
				ind--;
			}
			if (outt == false){
				a--;
				cout << v[0][a];
			}
		}
		puts("");
	}
	return 0;
}
