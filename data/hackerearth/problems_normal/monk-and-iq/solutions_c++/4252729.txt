#include <iostream>
#include<queue>
#include<utility>
#define ll long long
using namespace std;

int main()
{
	priority_queue<pair<ll,ll>> course;
	ll c, p, n;
	cin >> c >> p >> n;
	ll iq, iqMonk[1000005] = {0};
	vector<int> studentIq[c+1];
	for(int i = 0; i < n; i++) {
		cin >> iq;
		studentIq[i].push_back(iq);
	}	
    
    for(int i = 0; i < c; i++) {
    	ll z = studentIq[i].size();
    	if(z > 0) {
    		z *= studentIq[i][0];
    	}
    	course.push(make_pair(-z, -i));
    }
    
    for(int i = 0; i < p; i++) {
    	cin >> iqMonk[i];
    }
    ll val;
	for(int i = 0; i < p; i++) {
		val = iqMonk[i];
 		ll cNo = -1*course.top().second;
		ll z = 0;
		int size = studentIq[cNo].size();
		if(size > 0) {
			z += studentIq[cNo][size-1];
		}
		z += val;
		z *= (size+1);
		studentIq[cNo].push_back(val);
		course.pop();
		course.push(make_pair(-z, -cNo));
		cout << cNo + 1 << " ";
	}
	cout << endl;
    return 0;
}
