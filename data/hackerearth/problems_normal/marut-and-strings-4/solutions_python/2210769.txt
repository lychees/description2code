from string import ascii_uppercase
all_alphabet = [alpha for alpha in ascii_uppercase]
input_val = raw_input()
test_case = int(input_val) if input_val.isdigit() else False
if test_case <= 10 and test_case > 0:
	for _ in range(test_case):
		test_str = [item for item in raw_input()]
		lower_al, upper_al = [], []
		for item in test_str:
			if item.upper() in all_alphabet:
				if item.islower():
					lower_al.append(item)
				else:
					upper_al.append(item)
		if len(test_str) > 100 or len(test_str) < 1 or (len(lower_al) == 0 and len(upper_al) == 0):
			print "Invalid Input"
		else:
			if len(lower_al) >= len(upper_al):
				print len(upper_al)
			else:
				print len(lower_al)
else:
	print "Invalid Test"
