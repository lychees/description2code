def n_palindrome(a, b):
    count = 0
    for each in range(a, b+1):
        if str(each) == str(each)[::-1]:
            count += 1
            # print each
    return count


n = int(raw_input())
for i in range(n):
    l = map(int, list(raw_input().strip().split()))
    print n_palindrome(*l)
