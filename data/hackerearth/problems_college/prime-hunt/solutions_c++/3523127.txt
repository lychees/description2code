#include <bits/stdc++.h>
using namespace std;

vector<bool > prime(1e8 + 1, 1);
#define lim 1e8 + 1
void soe() {
    prime[1] = prime[0] = 0;
   for (int p=2; p*p<lim; p++)
    {
        // If prime[p] is not changed, then it is a prime
        if (prime[p] == true)
        {
            // Update all multiples of p
            for (int i=p*2; i<lim; i += p)
                prime[i] = false;
        }
    }
}
#define sc(n) scanf("%d", &n)
int main() {
    soe();
    int t;
    sc(t);
    while (t--) {
    	int a, b, i, cnt = 0;
    	sc(a); sc(b);

    	for (i = a; i <= b; i++) {
    		if (prime[i]) cnt++;
    	}
    	printf("%d\n", cnt);
    }
    return 0;
}
