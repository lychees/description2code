T = int(raw_input())

for i in xrange(T):
    s = raw_input()
    string = raw_input()
    temp = s
    if s in string:
        print "YES"
        continue
    cnt = 0
    f = 0
    for i in range(0, len(string)):
        if string[i] in s:
            cnt += 1
            s = s.replace(string[i], '', 1)
        else:
            cnt = 0
            s = temp
        if cnt == len(temp):
            print "YES"
            f = 1
            break
    if (f == 0):
        print "NO"
