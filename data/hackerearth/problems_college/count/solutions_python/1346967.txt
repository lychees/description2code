
n = input()


for _ in xrange(n):
    row,col = map(int,raw_input().split())
    odd_s = 0
    even_s = 0
    for each in range(1,row+1):
        if each%2==0:
            if col <= each:
                even_s += col
            else:
                even_s += each
        else:
            if col <= each:
                odd_s += col
            else:
                odd_s += each
    print even_s,odd_s
