#include <bits/stdc++.h>
#include <string>
using namespace std;
 
int main()
{
    string s;
    int N, Q;
    cin>>N>>Q;
	cin>>s;
	int hash[26][N+1];
	memset(hash, 0, sizeof(hash));

	for(int i=1; i<=N; i++)
	{
		for(int j=0; j<26; j++)
		{
			hash[j][i] = hash[j][i-1];
		}
		hash[s[i-1]-'a'][i]++;
	}

    while(Q--)
    {
    	int L, R, K;
    	scanf("%d %d %d", &L, &R, &K);
    	if(R-L+1 < K)
    	{
    		printf("Out of range\n");
    		continue;
    	}

    	
    	int smallK = 0;
    	for(int i=0; i<26; i++)
    	{
    		smallK += (hash[i][R] - hash[i][L-1]);
    		if(smallK >= K)
    		{
    			char c = 'a'+i;
    			printf("%c\n", c);
    			break;
    		}
    		
    	}
    }
    
    return 0;
}
