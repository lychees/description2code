#include <iostream>
using namespace std;

int main()
{
	int t,n,a[1000],min,sum=0;
    cin>>t;
for(int i=0;i<t;i++){
    cin>>n;
    for(int j=0;j<n;j++)
    cin>>a[j];
    min=a[0];
    for(int j=0;j<n-1;j++){
    	if(a[j+1]<min)
    	min=a[j+1];
    }
    cout<<min-1<<" ";
    
     for(int j=0;j<n;j++)
     sum+=a[j]-1;
     cout<<sum<<endl;
     sum=0;
}
    return 0;
}
