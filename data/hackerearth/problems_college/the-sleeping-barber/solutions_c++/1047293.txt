#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
	long long t = 0;
	cin>>t;
	while(t-->0)
	{
		int g ,i,count,temp;
		long long m ;
		cin >> g ;
		cin >> m;
		long int arr[g] ;
		temp = g;
		g = g-1;
		while(g>=0)
		{
			cin>>arr[g];
			g = g-1;	
		}
		
		sort(arr,arr+temp);
		i = 0;
		count = 0;
		while(m>=0)
		{
			m = m - arr[i];
			i++;
			if(m>=0) count++;
		}
		cout<<count<<endl;
		
	}
    
    return 0;
}
