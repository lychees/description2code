#include<iostream>
using namespace std;
int reverse(int x)
{
	int rev=0;
    while (x > 0)
	{
      rev = rev * 10 +( x % 10);
      x = x / 10; 
	}
	return rev;
}
int main()
{
	long long int N;
	int a,b,sum;
	cin>>N;
	while(N>0)
	{
		sum=0;
		cin>>a>>b;
	    sum=sum+reverse(a)+reverse(b);
	    sum=reverse(sum);
	    cout<<sum<<endl;
	    N--;
	}
	return 0;
}
