#include <iostream>
using namespace std;

int main()
{
	int t,i,o,flag;
	char a[20];
	cin>>t;
	int count;
	while(t--)
	{
		count=0;flag=0;
		cin>>a;
		for(i=0;a[i]!='\0';i++)
		{
			if(a[i]=='4'||a[i]=='7')
			count++;
		}
		int cc=count;
		while(count)
		{
			o=count%10;
			count=count/10;
			if(o==4||o==7);
			else
			{
				flag=10;
				break;
			}
		}
		if(flag==10||cc==0)
		cout<<"NO"<<endl;
		else
		cout<<"YES"<<endl;
	}
	return 0;
}
