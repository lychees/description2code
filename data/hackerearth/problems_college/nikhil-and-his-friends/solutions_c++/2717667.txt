#include <iostream>
using namespace std;

int main()
{
    int ts;
    cin>>ts;
    while(ts--){
    	int size,cost,i,j;
    	cin>>size>>cost;
    	int arr[size];
    	for(i=0;i<size;i++)
    	cin>>arr[i];
    	int sum,flag=0;
    	for(i=0;i<size;i++){
    		sum=0;
    		for(j=i;j<size;j++){
    			sum+=arr[j];
    				
    			if(sum==cost)
    			{flag=1;
    			break;}
    			else if(sum>cost){
    				flag=0; 	
    				break;
    			}
    		}
    		if(flag==1)
    		break;
    		}
    			if(flag==1)
    			{
    				cout<<"YES"<<endl;
    				
    				
    			}
    			else
    			{
    				cout<<"NO"<<endl;
    			}
    			
    			
    		
    	
    }
    return 0;
}
