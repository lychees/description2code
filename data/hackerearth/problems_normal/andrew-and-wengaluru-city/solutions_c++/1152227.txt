#include <iostream>
using namespace std;
typedef long long ll;
ll M = 1000000007;
ll a[100005],ma[100005];

ll min(ll a,ll b){
	return a<b?a:b;
}

ll max(ll a,ll b){
	return a>b?a:b;
}

int main(){
	ios_base::sync_with_stdio(false);
	int t,n;
	cin>>t;
	while(t--){
		cin>>n;
		for(int i=0;i<n;i++)
			cin>>a[i];
		ma[n-1] = a[n-1];

		for(int i=n-2;i>=0;i--){
			if(a[i] > ma[i+1])
				ma[i] = a[i];
			else 
				ma[i] = ma[i+1];
		}

		int i=1,j=0;
		ll ans = 0,cnt = 0,sum = 0,mx = a[0];
		for(int i=1;i<=n-1;i++){
			mx = max(mx,a[i]);
			if(a[i] < mx && ma[i] >  a[i]){
				//cout << mx << " " << ma[i] << " "<<a[i]<<" ##\n";
				ans += min(ma[i],mx)-a[i];
			}
		}		
		cout << ans%M << "\n";
	}
	return 0;
}
