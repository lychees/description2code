#include <iostream>
#include<string>
#include<algorithm>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    //t++;
    while(t--)
    {
    	string a,b;
    	cin>>a>>b;
    	int la,lb,i,flag=1;
    	la=a.length();
    	lb=b.length();
    	//cout<<la<<lb;
    	if(la!=lb)
    	{cout<<"No\n";}
    	else
    	{
    		sort(a.begin(),a.end());
    		sort(b.begin(),b.end());
    		for(i=0;i<la;i++)
    		{
    			if(a[i]!=b[i])
    			{flag=0;break;}
    			
    		}
    		if(flag==1)
    		cout<<"Yes\n";
    		else
    		cout<<"No\n";
    	}
    }
    return 0;
}
