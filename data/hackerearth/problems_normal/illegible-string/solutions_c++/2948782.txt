#include <iostream>
using namespace std;

int main()
{
	string S;
	int N;
	cin >> N;
	cin >> S;
	
	int i, max=0, min=0;
	int cv=0, cw=0;
	for (i=0;i<S.length();i++)
	{
		if (S[i]=='v')
		{
			if (cv==0)
			{
				cv++;
				min++;
				max++;
			}
			else
			{
				cv--;
				max++;
			}
		}
		else if (S[i]=='w')
		{
			cw=1;
			max+=2;
			min++;
		}
		else
		{
			min++;
			max++;
			cv=cw=0;
		}
		
	
	}
	cout << min << " " << max;
	return 0;
}
