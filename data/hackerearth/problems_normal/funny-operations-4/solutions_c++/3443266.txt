#include <bits/stdc++.h>
#define ll long long 
using namespace std;
ll int power(ll int n) {
	if(n==0) return 1;
	else return(2*power(n-1));
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
    	ll int n,p,i;
    	cin>>n;
    	ll int a[n],sum=-1,len=-1;
    	p=power(n);
    	//cout<<p<<endl;
    	for(i=0;i<n;i++) {
    		cin>>a[i];
    	}
    	for(i=1;i<p;i++) {
    		ll int k=0,x=0,an=0,cnt=0,j=i;
    		while(j>0) {
    			int tmp=j%2;
    			j=j/2;
    			if(tmp==1) {
    				//cout<<a[k]<<" ";
    				if(cnt==0) {
    					x=a[k];
    					an=a[k];
    				}
    				else{
    				x=x^a[k];
    				an=an & a[k];
    				}
    				cnt++;
    			}
    			k++;
    		}
    	//	cout<<" _" <<x << an<<endl;
    		int temp=(x|an);
    		if(temp>sum) {
    			sum=temp;
    			len=cnt;
    		}
    		else if(temp==sum && len<cnt) len=cnt;
    	}
    	cout<<sum<<" "<<len<<endl;
    }
    return 0;
}
