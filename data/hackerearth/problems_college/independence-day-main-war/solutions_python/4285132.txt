t = int(raw_input())
for i in range(t):
	n = int(raw_input())
	sumix = []
	sumiy = []
	for j in range(n):
		storm = raw_input()
		storm = storm.split()
		storm = map(float, storm)
		sumix.append(storm[0])
		sumiy.append(storm[1])
	n = float(n)
	x = (sum(sumix))/n
	y = (sum(sumiy))/n
	print "(%.2f , %.2f)"%(x,y)
