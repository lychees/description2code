#include <iostream>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		int a,b,i,k,num,c=0;
		cin >> a >> b;
		for(i=a;i<=b;i++)
		{
			num = 0;
			k = i;
			while(k>0)
			{
				num = num*10 + k%10;
				k = k / 10;
			}
			if(num == i)
			{
				c++;
			}
		}
		cout << c << endl;
	}
    
    return 0;
}
