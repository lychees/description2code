#include<bits/stdc++.h>
#define MAX 100000000
using namespace std;
bool isPrime[MAX+1];
void sieve(long long N)
{
    long long i,j;
    for(i=0;i<=N;i++)
    {
        isPrime[i]=true;
    }
    isPrime[0]=false;
    isPrime[1]=false;
    for(i=2;i*i<=N;i++)
    {
        if(isPrime[i]==true)
        {
            for(j=i*i;j<=N;j+=i)
            {
                isPrime[j]=false;
            }
        }
    }
}
int main()
{
    sieve(MAX);
    long long c=0,k=1,i;
    double a[5000];
    a[0]=0.0;
    for(i=3;i<=9999;i+=2)
    {
            k+=4;
            if(isPrime[i*i])
            {
                c++;
            }
            if(isPrime[i*i-i+1])
            {
                c++;
            }
            if(isPrime[(i-1)*(i-1)+1])
            {
                c++;
            }
            if(isPrime[(i-1)*(i-1)-i+2])
            {
                c++;
            }
            a[(i-1)/2]=(c*100.0)/k;
    }
    long long t,n;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld",&n);
        printf("%f\n",a[(n-1)/2]);
    }
    return 0;
}
