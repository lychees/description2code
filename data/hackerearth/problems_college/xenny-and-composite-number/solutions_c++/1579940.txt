#include <iostream>
#include<stdlib.h>
#include<stdio.h>
#include<string.h>
#include<math.h>
using namespace std;
int main()
{	
	 int t;
	 cin>>t;
	 while(t--)
	 {
	 	 int n;
	 	 cin>>n;
	 	 if(n==0)
	 	 {
	 	 	cout<<"4"<<endl;
	 	 	continue;
	 	 }
	 	 int flag=0;
	 	 for(int i=n+1;i<n+4;i++,flag=0)
	 	 {
	 	 	for(int j=2;j<=sqrt(i);j++)
	 	 	{
	 	 		if(i%j==0)
	 	 		{
	 	 			flag=1;
	 	 			break;
	 	 		}
	 	 	}
	 	 	if(flag==1)
	 	 	{
	 	 		cout<<i<<endl;
	 	 		break;
	 	 	}
	 	 }
	 }
	
}
