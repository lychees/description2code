#include <iostream>
using namespace std;
int main()
{
    int t,n,i;
    long long int a[100000];
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	int even=0,odd=0;
    	unsigned long long int num;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    		if(a[i]%2==0) even++;
    		else odd++;
    	}
    	num=(unsigned long long int)odd*(unsigned long long int)even;
    	cout<<num<<endl;
    	
    }
    return 0;
}
