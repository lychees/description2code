int match(string a , string b)
{
	for(int i = 0; i < sz(a); i++)
	{
		if(i + sz(b) > sz(a)) break;
		int p = 1;
		for(int j = 0; j < sz(b); j++)
		{
			if(a[i + j] != b[j])
			{
				p = 0;
				break;
			}
		}
		if(p) return 1;
	}
	return 0;
}

			s += a[j];
			arr.insert(s);
		}
	}
	int ans = 0;
	for(set<string>::iterator ii = arr.begin(); ii != arr.end(); ii++)
	{
		ans += match(*ii , b);
	}
	cout << ans << endl;
