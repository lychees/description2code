tc = input()
lr = []
while tc > 0:
    num = input()
    ln = []
    while num > 0:
        rs = raw_input()
        rsl = rs.split()
        r1 = int(rsl[0])
        r2 = int(rsl[1])
        if r1 not in ln:
            ln.append(r1)
        if r2 not in ln:
            ln.append(r2)
        num = num-1
    lr.append(len(ln))
    tc = tc-1
for i in lr:
    print i
