#include <bits/stdc++.h>
using namespace std;

#define mp make_pair
#define pb push_back

typedef long long ll;
typedef pair<int, int> pii;

const int MAXV = 480000;
const int INF = 1e9;

int n, k, m;
int prime[MAXV];
vector<int> pr;
int dp[MAXV];
set<int> bad;
vector<int> ans;

int main() {
#ifdef LOCAL
	freopen("input.txt", "r", stdin);
#endif                
	for (int i = 1; i < MAXV; ++i)
		prime[i] = i;
	for (int i = 2; i < MAXV; ++i)
		if (prime[i] == i)
			for (int j = i; j < MAXV; j += i)
				prime[j] = min(prime[j], i);
	for (int i = 0; i < MAXV; ++i) {
		dp[i] = INF;
	}
	dp[1] = 1;
	scanf("%d%d%d", &n, &k, &m);
	for (int i = 0, b; i < m; ++i) {
		scanf("%d", &b);
		bad.insert(b);
	}
	ans.pb(1);
	int cur = 2;
	while ((int)ans.size() < n) {
		assert(cur < MAXV);
		if (bad.find(cur) != bad.end() || dp[cur] <= k) {
			++cur;
			continue;
		}
		dp[cur] = 1;
		ans.pb(cur);
		for (int val = 1; val * cur < MAXV; ++val) {
			dp[val * cur] = min(dp[val * cur], dp[val] + 1);
		}
	}
	for (auto it : ans) {
		printf("%d ", it);
	}
	return 0;
}
