#include <iostream>
using namespace std;

int main()
{
    int  t;
    cin>>t;
    while(t--){
    	int ans=0;
    	string s;
    	cin>>s;
    	for(int i=0;i<s.size();++i)
    	   ans+=s[i]-'0';
    	  int f=1;
    	for(int i=2;i*i<=ans;++i)
    	  if(!(ans%i))
    	    {f=0; break;}
    	    
    	 if(f && ans!=1)
    	   printf("YES\n");
    	 else   printf("NO\n");
    }
    return 0;
}
