s = raw_input()
l = len(s)
if l < 6:
    print "Good luck!"
else:
    flag1, flag2 = 0, 0
    for i in range(l):
        if s[i:i+6].count('0') == 6:
            flag1 = 1
            break
        elif s[i:i+6].count('1') == 6:
            flag2 = 1
            break
    if flag1:
        print "Sorry, sorry!"
    elif flag2:
        print "Sorry, sorry!"
    else:
        print "Good luck!"
