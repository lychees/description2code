#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	string token;
	
	while(t--){
		cin>>token;
		
		for(int i=0;i<token.length();i++){
			switch(token[i]){
				case '2':cout<<"cde";
							break;
				
				case '6':cout<<"b3";
							break;
							
				case '7':cout<<"acf";
							break;
							
				case '9':cout<<"c6a";
							break;
				
				default: cout<<"";
							break;
			}
		}
		
		cout<<"\n";
	}
    return 0;
}
