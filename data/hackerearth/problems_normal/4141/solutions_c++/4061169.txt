#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int a0,a1,tmp,res,n,c;
    int t;
    cin>>t;
    while(t--)
    {
    	cin>>a0>>a1>>c>>n;
    	int i;
    	res=(a0*10+a1)%41;
    	if(n==1) res=a0;
    	for(i=2;i<=n-1;++i)
    	{
    		tmp = a0 + a1*c;
    		tmp %= 10;
    		res = res*10 + tmp;
    		res %= 41;
    		a0 = a1;
    		a1 = tmp;
    	}
    	if(res == 0) puts("YES");
    	else puts("NO");
    	
    }
    
    return 0;
}
