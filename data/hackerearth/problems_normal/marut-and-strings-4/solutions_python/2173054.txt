import fileinput


def numberofoperation(string):
    if(1<=len(string)<=100):    
        inputstring=list(string)
        upperletter=0
        lowerletter=0
        for letter in inputstring:
            if(65<=ord(letter)<=90):
                upperletter+=1
            elif(97<=ord(letter)<=122):
                lowerletter+=1
        if(upperletter==0 and lowerletter==0):
            print ("Invalid Input")
        else:
            print (min(lowerletter,upperletter))
    else:
        print ("Invalid Input")

inp=[]
for line in fileinput.input():
    inp.append(line.strip())
try:
    T=int(inp[0])
    if(1<=T<=10):
        for i in range(1,T+1):
            numberofoperation(inp[i])
    else:
        print ("Invalid Test")
except ValueError:
    print ("Invalid Test")
