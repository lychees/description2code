#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long n;
	cin >> n;
	
	long long value=0;
	map<long long,long long> mymap;
	for(long long i=0;i<n;i++)
	{
		long long arr[3];
		cin >> arr[0] >> arr[1] >> arr[2];
		sort(arr,arr+3);
		
		long long ans= 100*arr[2] + 10*arr[1] + arr[0];
		mymap[ans]++;
	}
	for(map<long long,long long>::iterator it=mymap.begin();it!=mymap.end();it++)
	{
		if(it->second == 1)
			value++;
	}
	cout<<value << endl;
	return 0;
}
