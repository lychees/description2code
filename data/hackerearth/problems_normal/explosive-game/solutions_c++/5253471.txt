#include<bits/stdc++.h>
using namespace std;


int main(){
	int n,k;
	cin >> n >> k;
	vector<int> v;
	long long sum=0;
	for(int i=0;i<n;i++){
		int left,right;
		cin >> left >> right;
		if(left > right) swap(left,right);
		sum+=left;
		v.push_back(right-left);
	}
	sort(v.begin(),v.end());
	bool ok=true;
	for(int i=0;i<n;i+=2){
		if(v[i]!=v[i+1]) ok=false;
		sum+=v[i];
	}
	if(!ok) sum=-1;
	
	for(int i=0;i<k;i++){
		long long x;
		cin >> x;
		if(x==sum) cout << "YES" << endl;
		else cout << "NO" << endl;
	}
	return 0;
}
