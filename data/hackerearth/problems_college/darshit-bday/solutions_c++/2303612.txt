#include <iostream>
using namespace std;

int main()
{
    long long t,X,Y,Z,B,W;
    cin>>t;
    while(t--){
    	long long result = 0;
    	cin>>B>>W;
    	cin>>X>>Y>>Z;
    	
    	if(X<=(Y + Z) && Y<=(X+Z))
    		result = B * X + W * Y;
    	else if(X>(Y+Z))
    		result = B * (Y+Z) + W * Y;
    	else if(Y > (X+Z))
    		result = B * X + W * (X+Z);
    		
    	cout<<result<<endl;
    }
}
