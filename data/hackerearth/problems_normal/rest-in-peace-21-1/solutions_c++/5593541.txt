#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    long long n;
    cin>>t;
    while(t--){
    	cin>>n;
    	if(n%21){
    		int f=1,i;
    		string s=to_string(n);
    		for(i=0;i<s.size()-1;i++){
    			if(s[i]=='2' && s[i+1]=='1'){
    				f=0;
    				break;
    			}
    		}
    		if(f){
    			cout<<"The streak lives still in our heart!"<<endl;
    		}
    		else{
    			cout<<"The streak is broken!"<<endl;
    		}
    	}
    	else{
    		cout<<"The streak is broken!"<<endl;
    	}
    }
    return 0;
}
