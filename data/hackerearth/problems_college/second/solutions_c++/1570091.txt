#include <cstdio>
#include <map>
#include <iostream>

using namespace std;

int main()
{
    map<int,int> hash;
    int N,m,t;
    scanf("%d %d",&N,&m);
    for(int i = 0; i < N; ++i)
    {
        scanf("%d",&t);
        hash[t] = 1;
    }
    int ans = 0;
    if(hash[m] == 0)
    {
        printf("0\n");
        return 0;
    }

    ans = m;
    int c = 1;
    while(true)
    {
        if( hash[ans] == 0 )
        break;
        c++;
        ans = m * c;
    }
    printf("%d\n",m*(c-1));
    return 0;
}
