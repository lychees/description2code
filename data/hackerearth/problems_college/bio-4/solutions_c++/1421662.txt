#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		string a,b;
		cin>>a;
		cin>>b;
		int i,j,c=0,minm=123456;
		int la=a.size();
		int lb=b.size();
		for(i=0;i<la;i++){
			c=0;
			for(j=0;j<lb;j++){
				if(i+j<la&&a[i+j]==b[j])
				c++;
				
			}
			minm=min(minm,lb-c);
		}
		cout<<minm<<endl;
		
	}
	return 0;
}
