#include<bits/stdc++.h>
using namespace std;

int main()
 {long long MOD=1000000007;
 	int t;
 	cin>>t;
 	int i,n,j;
 	vector<long long> a(100005);
 	a[0]=1;
 	for(i=1;i<=100000;i++)
 	  {a[i]=a[i-1];
 	   if(i>1)
 	     a[i]=(a[i]+a[i-2])%MOD;
 	    if(i>2)
 	      a[i]=(a[i]+a[i-3])%MOD;
 	  }
 	while(t--)
 	  { 
 	  	cin>>n;
 	  	cout<<a[n]<<endl;
 	  }
 }
