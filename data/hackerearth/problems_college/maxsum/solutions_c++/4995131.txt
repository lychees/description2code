#include<bits/stdc++.h>
using namespace std;
int a[100001],n,temp[100001];
int recur(int v)
{
	if(v>=n)
	  return 0;
	if(temp[v])
	  return temp[v];
	temp[v]= max(recur(v+1),a[v]+recur(v+2));
	return temp[v];
}
main()
{
	int t;
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(int i=0;i<n;i++)
	       cin>>a[i];
	    memset(temp,0,sizeof(temp));
	    cout<<recur(0)<<endl;
	}
}
