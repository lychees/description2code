#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long n,m;
	cin>>n>>m;
	long long i,j,a[n],t[n],ind,amt;
	char c;
	for(i=0;i<n;i++)
	cin>>a[i];
	for(i=0;i<n;i++)
	cin>>t[i];
	for(i=0;i<n;i++)
	if(t[i]==a[0])
	ind=i;
	long long sum=0,flag=0,temp[n];
	for(i=1;i<=m;i++)
	{
		cin>>c>>amt;
		if(c=='L')
		sum-=amt;
		
		else
		
		sum+=amt;
		sum=sum%n;
		if(sum<0)
		sum+=n;

			flag=i;

		}
	}

}
