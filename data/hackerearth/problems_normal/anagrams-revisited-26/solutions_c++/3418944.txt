#include <iostream>
#include <map>
#include <string>

using std::cin;		using std::cout;
using std::endl;	using std::map;
using std::string;

int main()
{
	int answer, num_strings;
    map<char, int> anagram;
    map<map<char, int>, int> words;
    string str;
    
    answer = 0;
    cin >> num_strings;
    
    for(int i = 0; i != num_strings; i++){
    	cin >> str;
    	for(string::iterator iter = str.begin(); iter != str.end(); iter++){
    		++anagram[*iter];
    	}
    	
    	if(++words[anagram] > answer){
    		answer = words[anagram];
    	}
    	anagram.clear();
    }
    cout << answer;
    return 0;
}
