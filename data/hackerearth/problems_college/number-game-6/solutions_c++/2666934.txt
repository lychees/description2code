ll arr[10000000];
ll binary(ll low,ll high,ll n)
{
	ll mid=(low+high)/2;
	if(arr[mid]==n)
	  return mid;
	else if( mid>0 && n<arr[mid] && n>arr[mid-1])
	  return mid-1;
	 else if(arr[mid]>n)
	  return binary(low,mid-1,n);
	 else 
	  return binary(mid+1,high,n)  ;
	
}
