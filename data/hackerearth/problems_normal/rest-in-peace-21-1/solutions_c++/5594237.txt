#include<iostream>
#include<string>
using namespace std;
int main(){
	int n;
	cin >>n;
	for(int i =0;i<n;i++){
		string number  = "";
		cin >> number;
		bool sad = false;
		for(int l = 0;l<number.length(); l++){
			if(l!=number.length()-1){
				if(number[l]=='2'&&number[l+1]=='1'){
					sad = true;
					break;
				}
			}
		}
		if(stoi(number)%21==0 || sad)
			cout << "The streak is broken!" <<endl;
		else
			cout << "The streak lives still in our heart!" <<endl;
	}
	return 0;
}
