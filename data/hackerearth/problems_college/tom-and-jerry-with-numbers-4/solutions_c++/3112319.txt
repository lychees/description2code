#include<bits/stdc++.h>
using namespace std;
int main(){
    int t;
    scanf("%d",&t);
    while(t--){
    	int n,a,b;
    	scanf("%d %d %d",&n,&a,&b);
    	bool v[n+1];
    	memset(v,false,sizeof(v));
    	for(int i=1;i<=n;i++){
    		if(i>=a&&!v[i-a])
    		   v[i]=true;
    		   if(i>=b&&!v[i-b])
    		   v[i]=true;
    		//   if(i==3)
    		 //  printf("%d %d %d\n",v[3],a,b);
    	}
    	if(v[n])
    	printf("1\n");
    	else
    	printf("0\n");
    }
	return 0;
}
