#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    long long int x,y,i,j;
    cin>>x>>y;
    long long int c[y],b[x];
    for(i=0;i<x;i++)
    {
    	for(j=0;j<y;j++)
    	{
    		cin>>c[j];
    	}
    	sort(c,c+y);
    	b[i]=c[0];
    }
    sort(b,b+x);
    cout<<b[x-1]<<endl;
    return 0;
}
