#include<stdio.h>
#include<iostream>

using namespace std;

int main()
{
    int t;
    
    long n,i,j;
    long long sum=0,max=0,neg_max;
    scanf("%d",&t);
    while(t--)
    {
        sum=0;
        scanf("%ld",&n);
        int *a=new int[n];
        for(i=0;i<n;i++)
            scanf("%d",&a[i]);
        i=0;
        max=sum;
        while(i<n)
        {
           sum+=a[i];
           if(sum<0)
           {
               sum=0;

           }
           if(sum>max)
                max=sum;
           i++;
        }
        if(max==0)
        {   neg_max=a[0];
            for(i=1;i<n;i++)
                if(a[i]>neg_max)
                    neg_max=a[i];
            max=neg_max;
        }
        printf("%lld\n",max);
        delete [] a;
    }

}
