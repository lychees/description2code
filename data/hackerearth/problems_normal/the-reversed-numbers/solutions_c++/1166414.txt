#include <iostream>
using namespace std;

unsigned long long rev( int sum )
{
	unsigned long long reverse = 0;
	for( ; sum!= 0 ; )
   	{
     	reverse = reverse * 10;
      	reverse = reverse + sum%10;
      	sum = sum/10;
   	}
   	return reverse;
}

int main()
{
	int t;
	cin>>t;
	while( t-- )
	{
		unsigned long long x, y, sum = 0, SUM = 0, reverse = 0, X, Y;
		cin>>x>>y;
		X = rev( x );
		Y = rev( y );
	 	sum = X+Y;
		SUM = rev( sum );
   		cout<<SUM<<endl;
	}
}
