#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    long long a,b,c,p,q,x;
    cin>>a>>b>>c;
    p=c*c;
    q=2*a*b;
    if(a+b<=c)
    {
    	p=q=1;
    }
    else
    {
    	if(a<c)
    	p-=(c-a)*(c-a);
    	if(b<c)
    	p-=(c-b)*(c-b);
    	x=__gcd(p,q);
    	p/=x;
    	q/=x;
    	
    }
    cout<<p<<"/"<<q;
    return 0;
}
