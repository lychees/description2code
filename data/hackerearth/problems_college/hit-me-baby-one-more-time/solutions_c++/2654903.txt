bool primes[1000010];
void seive() {
	for (int i = 2; i <= sqrt(1000004); i ++) {
		if(!primes[i]) {
			for (int j = i * i; j <= 1000004; j += i) primes[j] = true;
		}
	}
}

int main() {
    ll n; cin >> n;
	ll ans = 0;
	n --;
	ll a = n / 3;
	ans += ((a * (a + 1)) / 2) * 3;
	ll b = n / 5;
	ans += ((b * (b + 1)) / 2) * 5;
	ll c = n / 15;
	ans -= ((c * (c + 1)) / 2) * 15;
	cout << ans << endl;
	return 0;
}
