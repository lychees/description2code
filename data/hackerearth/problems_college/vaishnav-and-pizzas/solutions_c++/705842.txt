#include<cstdio>
long long int p[10005];
   long long int phi(long long int n) 
     { 
       long long int r= n; 
       for(long long int i=2;i*i<=n;i++) 
       { 
         if (n%i==0) 
		 r-=r/i; 
         while(n%i==0) 
		 n/=i; 
       } 
       if(n>1) 
	   r-=r/n; 
       return r; 
     }
     
int main()
{
	
	long long t,ans=0,i;
	for(i=1;i<=10000;i++)
	p[i]=phi(i);
	scanf("%lld",&t);
	long long n;
	while(t--)
	{
		ans=0;
		scanf("%lld",&n);
		for(i=1;i<=n;i++)
		ans+=p[i];
		if(n==0)
		ans=1;
		printf("%lld\n",ans-1);
	}
	return 0;
}
