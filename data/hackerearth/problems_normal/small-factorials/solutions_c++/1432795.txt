vector<int> v[102];



void pre()
{
  v[0].push_back(1);

  int i,c=0,j;
  for(i=1;i<=100;i++)
  {
    c=0;
    for(j=0;j<v[i-1].size();j++)
    {
      int now=v[i-1][j]*i+c;
      c=now/10;
      v[i].push_back(now%10);
    }
    while(c)
    {
      v[i].push_back(c%10);
      c/=10;
    }
  }
  for(i=1;i<=100;i++)
    reverse(v[i].begin(),v[i].end());
}

int main()
{
    //freopen("in.txt","r",stdin);

    int i,j,k,n;


    pre();




    int T;
    scanf("%d",&T);
    while(T--)
    {
      scanf("%d",&n);
      for(i=0;i<v[n].size();i++) printf("%d",v[n][i]);
      printf("\n");
    }

    return 0;
}
