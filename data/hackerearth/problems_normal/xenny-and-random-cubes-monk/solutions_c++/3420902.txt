#include <bits/stdc++.h>
#define MOD 1000000007
using namespace std;
int freq[12][26];
string temp;
int main()
{
    int n,k,i,j,sz,val;
    char c;
    long long int ans,ways;
    cin>>n>>k;
    for(i=0;i<n;i++)
    {
    	for(j=0;j<6;j++)
    	{
    		cin>>c;
    		freq[i][c-'a']++;
    	}
    }
    cin>>temp;
    ans=0;
    for(i=0;i<(1<<n);i++)
    {
    	if(__builtin_popcount(i)==k)
    	{
    		vector<int>nigs;
    		val=i;
    		j=0;
    		while(val)
    		{
    			if(val&1)
    			nigs.push_back(j);
    			j++;
    			val>>=1;
    		}
    		sort(nigs.begin(),nigs.end());
    		do
    		{
    			ways=1;
    			for(j=0;j<k;j++)
    			{
    				ways*=freq[nigs[j]][temp[j]-'a'];
    				ways%=MOD;
    			}
    			ans+=ways;
    			ans%=MOD;
    		}while(next_permutation(nigs.begin(),nigs.end()));
    	}
    }
   	cout<<ans%MOD<<"\n";
   	
    return 0;
}
