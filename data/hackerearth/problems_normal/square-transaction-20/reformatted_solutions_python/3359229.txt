n, data = input(), [int(i) for i in raw_input().rstrip().split()]
for i in range(1, n):
    data[i] += data[i-1]
q = [input() for i in range(input())]
for i in q:
    index = 0
    begin, end = 0, n-1
    while begin <= end:
        mid = (begin + end) / 2
        if data[mid] > i:
            index = mid
            end = mid-1
        elif data[mid] < i:
            begin = mid+1
        else:
            index = mid
            break
    print index+1 if data[index] >= i else -1
