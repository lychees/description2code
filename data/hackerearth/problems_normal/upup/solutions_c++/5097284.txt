#include <iostream>
#include<string>
using namespace std;

int main()
{
   string s;
   int l=0;
   getline (cin, s);
  
   while(s[l] !='\0'){l++;}
   
   int i;
   
   if(int(s[0]) >=97 && int(s[0]) <=122){
   	
   	s[0]=s[0]-32;
   }
   
   for(i=0; i<l; i++){
   	if(s[i]==' ' && int(s[i+1]) >=97 && int(s[i+1]) <=122 ){
   		
   		s[i+1]=s[i+1]-32;
   	}
   	
   }
   
   cout<<s;
   
   
}
