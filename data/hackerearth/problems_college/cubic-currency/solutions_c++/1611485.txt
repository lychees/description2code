#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
vector<int> c(31);

ll dp[12345];
const int N = 10000+123;

int build(){
    for (int i=0; i<=N; i++){
        dp[i]=1;
    }
    for (int j = 2; j <= 21; j++){
        for (int i = c[j]; i<=N; i++){
            dp[i] += dp[i - c[j]];
        }
    }
    /*for (int i=0; i<=10; i++){
        cout<<i<<" "<<dp[i]<<endl;
    }*/
}

static int test(){
    int n;
    //cin>>n;
    scanf("%d",&n);
    printf("%lld\n",dp[n]);
    //cout<<dp[n]<<endl;
    return 0;
}

int main(){
    for (int i=1; i<=28; i++){
        c[i] = (i*i*i);
    }
    build();
    int t;
    cin>>t;
    while(t--){
        test();
    }
}
