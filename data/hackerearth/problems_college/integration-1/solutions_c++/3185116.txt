#include <bits/stdc++.h>
#define lli long long int
#define s(x) scanf("%lld", &x)
#define pb push_back
#define mp make_pair
#define fr first
#define sc second
#define MAXX 10005

using namespace std;

int main()
{
    lli tcase,n,m,i,j,tt,k,ans,u,r,s,l,p;

    s(tcase);

    tt = 1;

    while (tt <= tcase) {
        printf("Case #%lld: ", tt);
        ++tt;

        s(n);
        s(p);
        s(u);
        s(r);
        s(s);

        i = 0;

        while (i < n) {
            if ((u-p) >= r) {
                k = (u-p)/r;

                k = min(k, n-i);
                p = p + k*r;
               // cout << p << " " << k << " " << u << " " << p << " " << r << endl;
                i = i+k;
            } else {
                k = r - (u-p);

                k = k/s;

                if (k == 0)
                    ++k;

                k = min(k, n-i);
				
				if (k > (p/s)) {
					k = p/s;
					
					if (k == 0)
						break;
				}
                i = i+k;
                p = p - k*s;

            //    cout << p << " " << k << endl;
            }
        }

        cout << p << endl;
    }
    return 0;
}
