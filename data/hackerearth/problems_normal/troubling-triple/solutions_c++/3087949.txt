#include <bits/stdc++.h>
using namespace std;

struct BIT{
	vector<int>	ft;
	int N;
	BIT( int n ){
		N = n+1;
		ft.assign(N, 0);
	}
	
	int LSONE( int x ){
		return ( x & (-x) );
	}
	
	void update ( int x, int v ){
		for( ; x < N; x+=LSONE(x) )
			ft[x]+=v;
	}
	
	int query ( int x ){
		int res = 0;
		for( ; x ;x-=LSONE(x) )
			res += ft[x];
		
		return res;
	}
};

int main()
{
    for( int n, k; cin>>n>>k; ) {
    	vector<int>nums(n);
    	for(int i=0; i <n ; i++)
    		cin>>nums[i];
    	BIT bit(1000000);
    	long long  val, res = 0;
    	for(int i=0; i <n; i++){
    		for(int j=i+1; j <n; j++){
    			val = nums[i]*nums[j];
    			//cout<<bit.query(val)<<"\n";
    			if( val<=k )
    				res += bit.query(k/val);
    		}
    		bit.update( nums[i], 1);
    	}
    	cout<<res<<"\n";
    }
    return 0;
}
