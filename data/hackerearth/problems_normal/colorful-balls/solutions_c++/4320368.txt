#include<bits/stdc++.h>
using namespace std;
const long long inf=100000000000000LL;
#define ll long long
ll cnt=0;
ll dp[200][11][11][11][11]={};
ll dfs(char s, ll a, ll b, ll c, ll d){
    ll &t=dp[s][a][b][c][d];
    if(a==0&&b==0&&c==0&&d==0){ return 1;}
    if(t!=-1)return t;
    t=0;
    if(s=='r'){
        if(b>0)t+=dfs('b', a, b-1, c, d);
        if(c>0)t+=dfs('y', a, b, c-1, d);
        if(d>0)t+=dfs('g', a, b, c, d-1);
    }else
    if(s=='b'){
        if(a>0)t+=dfs('r', a-1, b, c, d);
        if(c>0)t+=dfs('y', a, b, c-1, d);
        if(d>0)t+=dfs('g', a, b, c, d-1);
    }else
    if(s=='y'){
        if(b>0)t+=dfs('b', a, b-1, c, d);
        if(a>0)t+=dfs('r', a-1, b, c, d);
        if(d>0)t+=dfs('g', a, b, c, d-1);
    }else
    if(s=='g'){
        if(b>0)t+=dfs('b', a, b-1, c, d);
        if(c>0)t+=dfs('y', a, b, c-1, d);
        if(a>0)t+=dfs('r', a-1, b, c, d);
    }
    return t;
}
int main(){
    ll a, b, c, d;
    memset(dp, -1, sizeof dp);
    cin>>a>>b>>c>>d;
   cnt+= dfs('r', a-1, b ,c, d);
   cnt+= dfs('b', a, b-1, c, d);
    cnt+=dfs('y', a, b, c-1, d);
    cnt+=dfs('g', a, b, c, d-1);
    if(a==0&&b==0&&c==0&&d==0)cout<<"1\n";else
    cout<<cnt<<endl;
}
