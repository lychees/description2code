#include <iostream>
using namespace std;

int main() {
	int n, m, x;
	char c;
	cin >> n >> m;
	int *s = new int[n];
	int *t = new int[n];
	for (int i = 0; i < n; i++)
		cin >> s[i];
	for (int i = 0; i < n; i++)
		cin >> t[i];
	int start = 0;
	while (start < n) {
		if (t[0] == s[start]) {
			int count = 0;
			for (int i = 0; i < n; i++) {
				if (t[i] == s[(start+i)%n])
					count++;
			}
			if (count == n)
				break;
		}
		start++;
	}
	if (start < n) {
		int left = 0;
		int j;
		for (j = 1; j <= m; j++) {
			cin >> c >> x;
			if (c == 'R') {
				left = (left + (n-x%n))%n;
			} else {
				left = (left + x%n)%n;
			}
			if (left == start) {
				int count = 0;
				for (int i = 0; i < n; i++) {
					if (t[i] == s[(left+i)%n])
						count++;
				}
				if (count == n)
					break;
			}
		}
		if (j <= m)
			cout << j;
		else
			cout << -1;
	} else {
		cout << -1;
	}
	return 0;
}
