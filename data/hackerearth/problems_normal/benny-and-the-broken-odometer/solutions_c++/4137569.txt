#include <bits/stdc++.h>
using namespace std;
#define ll long long 
ll dp[11];

void compute(){
	dp[1]=1;
	ll n=10;
	for(int i=2; i<=10; i++){
		dp[i]=dp[i-1]*9+n;
		n*=10;
		//cout<<dp[i]<<" ";
	}
}

ll solve(ll n){
	ll d=1;
	ll copy=n;
	ll ans=0;
	ll p=1;
	while(n){
		int m=n%10;
		n/=10;
		
		if(m>3){
			ans+=p+(m-1)*dp[d-1];
		}
		else{
			ans+=m*dp[d-1];
		}
		
		d++;
		p*=10;
	}
	return copy-ans;
}

int main()
{
	compute();
    int t;
    scanf("%d", &t);
    while(t--){
    	ll n;
    	cin>>n;
    	printf("%lld\n", solve(n));
    }
    return 0;
}
