#include <iostream>
using namespace std;

int arr[100001];
int n;
void init()
{
	for(int i=1;i<=n;++i)
	{
		int cp=i,s=0,d;
		while(cp>0)
		{
			d=cp%10;
			s|=(1<<d);
			cp/=10;
		}
		arr[i]=s;
	}
	
}
int main()
{
	int q,m,s,c;
    scanf("%d",&n);
    init();
    scanf("%d",&q);
    while(q--)
    {
    	scanf("%d",&m);
    	s=0;
    	for(int i=0;i<m;++i)
    	{
    		scanf("%d",&c);
    		s|=(1<<c);
    	}
    	int ans=0;
    	for(int i=1;i<=n;++i)
    	{
    		if(s&arr[i]) ans++;
    	}
    	printf("%d\n",ans);
    }
    return 0;
}
