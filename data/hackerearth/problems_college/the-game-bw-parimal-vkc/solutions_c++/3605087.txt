#include <iostream>
#include <string.h>
using namespace std;

#define MOD 1000000007

int main() {
	char num[200005];
	gets(num);
	int len = strlen(num);
	long long factor = 1, ans = 0;

	for (int i = len-1; i >= 0; --i) {
		long long tmp = (num[i]-'0') * (i+1) * factor % MOD;
		ans += tmp;
		ans %= MOD;
		factor = (factor * 10 + 1) % MOD;
	}
	printf("%d\n", ans);

	return 0;
}
