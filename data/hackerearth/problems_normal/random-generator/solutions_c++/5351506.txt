#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int main()
{
    long int t,n,k,p,val1,val2,i,j,flag;
    scanf("%ld",&t);
    for(i=1;i<=t;i++)
    {
    	scanf("%ld %ld %ld",&n,&k,&p);
    	flag=1;
    	long int a[n];
    	for(j=0;j<n;j++)
    	scanf("%ld",&a[j]);
    	sort(a,a+n);
    	for(j=0;j<=n-k;j++)
    	{
    		if(a[j+k-1]-a[j]<=2*p)
    		{
    			flag=0;
    			break;
    		}
    	}
    	if(flag==1)
    	printf("YES\n");
    	else
    	printf("NO\n");
    }
    return 0;
}
