#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
  int min,tp,t,n,k,i,ki,j,l,u,cosi,a[105];
  cin>>t;
  int b[105];
  while(t--)
    {
      for(i=0;i<105;++i)
	b[i]=0;
      cin>>n>>k;
      for(i=0;i<n;++i)
	{
	  cin>>a[i];
	}
      sort(a,a+n);
      j=n-1;
      for(i=100;i>=1;--i)
	{
	  ki=0;
	  cosi=0;
	  l=0;u=1;
	  
	  while(ki<k)
	    {
	      while(i==a[j]&&j>=0)
		{
		  ki++;
		  j--;
		  u++;
		}
	//      if(i==a[j])
	//	l++;
	      if(ki>=k)
		break;
	      if(j-l>=0)
		{
		  if(j+u<n)
		    {
		      if((i-a[j-l])*3 < (a[j+u]-i)*5)
			{
			  cosi+=(i-a[j-l])*3;
			  l++;
			}
		      else
			{
			  cosi+=(a[j+u]-i)*5;
			  u++;
			}
		      ki++;
		    }
		  else
		    { 
		      cosi+=(i-a[j-l])*3;
		      l++;
		      ki++;
		    }
		}
	      else
		{
		  cosi+=(a[j+u]-i)*5;
		  ki++;
		  u++;
		}
	    }
	  b[i]=cosi;
	}
      min=b[1];
      for(i=1;i<101;++i)
	{
	  if(min>b[i])
	    min=b[i];
	}
      
      cout<<min<<endl;
    }

  return 0;
}
