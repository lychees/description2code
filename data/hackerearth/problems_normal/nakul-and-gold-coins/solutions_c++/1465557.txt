#include <bits/stdc++.h>
using namespace std;

int prime[1000006], all[100005], l;
long long ans[10000007];

int search(long long key, int low, int high)
{
	int mid = (low+high)/2;
	
	if(low<=high)
	{
		if(ans[mid]==key)
			return mid;
		
		else if(ans[mid]>key)
			return search(key, low, mid-1);
		
		else
			return search(key, mid+1, high);
	}
	
	if(low>=l || ans[low]>key)
		return high;
	
	return low;
}

int main()
{
    int T;
    scanf("%d", &T);
    
    for(int i=2; i<=1000; i++)
    {
    	if(prime[i]==0)
    	{
    		for(int j=i; i*j<=1000000; j++)
    			prime[i*j] = 1;
    	}
    }
    
    int k = 0;
    
    for(int i=2; i<=1000000; i++)
    	if(prime[i]==0)
    		all[k++] = i;
    
    int j;
    long long tmp;
    
    for(int i=0; i<k-1; i++)
    {
    	for(j=i+1; j<k; j++)
    	{
    		tmp = all[i]*all[j];
    		
    		if(tmp<100000000)
    			ans[l++] = tmp;
    		else
    			break;
    	}
    	
    	if(j==i+1)
    		break;
    }
    
    sort(ans, ans+l);
    
    long long L, R;
    
    while(T--)
    {
    	scanf("%lld %lld", &L, &R);
    	
    	int tmp1 = search(R, 0, l-1), tmp2 = search(L, 0, l-1);
    	
		if(ans[tmp2]<L || ans[tmp1]>R)
	    	printf("%d\n", tmp1-tmp2);
 		else
 			printf("%d\n", tmp1-tmp2+1);
    }
    
    return 0;
}
