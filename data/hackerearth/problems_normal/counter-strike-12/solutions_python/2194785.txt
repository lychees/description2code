t = int(raw_input())
while t>0:
	n, m, d = raw_input().split()
	n = int(n)
	m = int(m)
	d = int(d)
	arr = []
	for i in range(n):
		c = raw_input().split()
		arr.append([int(c[0]), int(c[1])])
	ans = 0
	for i in range(m):
		c = raw_input().split()
		c[0] = int(c[0])
		c[1] = int(c[1])
		temp = [c[0], c[1]]
		for j in range(n):
			if abs(arr[j][0]-temp[0])+abs(arr[j][1]-temp[1]) <= d:
				ans += 1
				break
	if ans >= m//2:
		print "YES"
	else:
		print "NO"
				
	t -= 1
