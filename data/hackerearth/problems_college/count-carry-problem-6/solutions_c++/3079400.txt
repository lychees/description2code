#include<iostream>
#include<vector>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while (t--)
	{
		long long int n1, n2, car = 0;
		cin >> n1 >> n2;
		long long int max , min;
		if (n1 >= n2)
		{
			max = n1;
			min = n2;
		}
		else
		{
			max = n2;
			min = n1;
		}

		vector<int> v;
		while (max != 0)
		{
			v.push_back(max % 10);
			max = max / 10;
		}
		v.push_back(0);
		int size = v.size() , r;
		for (int i = 0; i < size - 1; i++)
		{
			r = v[i] + min % 10;
			// cout << "vector element is : " << v[i] << endl ;
			if (r >= 10)
			{
				car++;
				v[i + 1]++;
			}
			r = 0;
			min = min / 10;
		}

		if (car == 0)
			cout << "No carry operation" << endl;
		else if (car == 1)
			cout << car << " carry operation" << endl;
		else
			cout << car << " carry operations" << endl;

	}
    return 0;
}
