#include<iostream>
#include<set>
#include<vector>
using namespace std;
 
int LIS(vector<int> A)
{
    int N = A.size(),i;
    set<int> s;
    set<int>::iterator k;
    for (i=0;i<N;i++)
    {
        if (s.insert(A[i]).second)
        {
           k = s.find(A[i]);
           k++;
           if (k!=s.end())
              s.erase(k);
        }
    }
    return s.size();
}

int main()
{
	int n;
	vector<int> arr;
	cin>>n;
	arr.resize(n);
	for(int i=0;i<n;i++)cin>>arr[i];
	cout<<n-LIS(arr);
}
