#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    long int i,n,j;
    cin>>n;
    for(i=1;i<=n;i++)
    {
        int c=0;
        for(j=1;j<=i;j++)
        {
            if(i%j==0)
                c=c+1;
        }
        if(c!=2&&i!=1)
            cout<<i<<"\n";
    }
}
