#include<bits/stdc++.h>
using namespace std;
int prime[1000001]={0},p[1000001]={0};
int main()
{
	int a,b,i,j,t,count;
	prime[0]=1;
	prime[1]=1;
	
	p[0]=0;
	p[1]=0;
	for(i=2;i<=1000;i++)
	{
		if(prime[i]==0)
			for(j=2;i*j<=1000000;j++)
				prime[i*j]=1;
	}
	for(i=2;i<=1000000;i++)
	{
		if(prime[i]==0)
		p[i]=p[i-1]+1;
		else
		p[i]=p[i-1];
	}
	scanf("%d",&t);
	while(t--)
	{
		count=0;
		scanf("%d %d",&a,&b);
		if(prime[a]==0)
		count=1;
		
		count= count + p[b]-p[a];
		printf("%.6f\n",(float)count/(b-a+1));
		
		
		
	}
	return 0;
}
