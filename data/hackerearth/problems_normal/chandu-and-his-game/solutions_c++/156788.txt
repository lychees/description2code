#include<iostream>
#include<stdlib.h>
#include<algorithm>
#include<math.h>
long long int min(long long int a,long long int b){
	if(a>=b)
	return b;
	else
	return a;
}
using namespace std;
int main(){
long long int n,k,m,t=0;
cin>>n>>k;
long long int x[k],y[k];
for(int i=0;i<k;i++)
cin>>x[i];
for(int i=0;i<k;i++)
cin>>y[i];

for(int i=0;i<k;i++){
m=pow(10,8);
m=min(m,x[i]-1+n-y[i]);
m=min(m,x[i]-1+y[i]-1);
m=min(m,n-x[i]+y[i]-1);
m=min(m,n-x[i]+n-y[i]);
t=t+m;
}
cout<<t<<endl;
//system("pause");
	return 0;
}
