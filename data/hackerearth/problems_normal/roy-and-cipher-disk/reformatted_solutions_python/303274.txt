import string
d = dict(zip(string.ascii_lowercase, range(0, 26)))
repeat = int(raw_input())
for i in range(repeat):
    output = []
    init = 'a'
    strin = raw_input()
    for letter in strin:
        diff = (d[letter] - d[init])
        if abs(diff) == 13:
            output.append(abs(diff))
        elif abs(diff) <= 13:
            output.append(diff)
        elif diff > 0:
            output.append(diff-26)
        else:
            output.append(diff+26)
        init = letter
    print ' '.join(str(x) for x in output)
