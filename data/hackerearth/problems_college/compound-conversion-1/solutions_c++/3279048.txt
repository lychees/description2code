#include <bits/stdc++.h>

using namespace std;

int main()
{
	string a, b;
	int t;
	scanf("%d", &t);
	for (int i = 0; i < t; i++) {
		a.clear();
		b.clear();
		cin >> a >> b;
		if (a.size() != b.size()) {
			printf("Case %d: -1\n", i + 1);
			continue;
		} else {
			int ans = 0;
			int x1, x2, x3, x4;
			x1 = x2 = x3 = x4 = 0;
			for (int j = 0; j < a.size(); j++) {
				if (a[j] == '?' && b[j] == '0') {
					x1++;
				} else if (a[j] == '?' && b[j] == '1') {
					x2++;
				} else if (a[j] == '1' && b[j] == '0') {
					x3++;
				} else if (a[j] == '0' && b[j] == '1') {
					x4++;
				}
			}
			
			if (x3 > x4) {
				if ((x3 - x4) > x2) {
					ans = -1;
				} else {
					ans = x4 + 2 * (x3 - x4) + x2 - (x3 - x4) + x1;
				}
			} else {
				ans = x1 + x2 + x4;
			}
			printf("Case %d: %d\n", i + 1, ans);
		}
		
	}
	
	return 0;
}
