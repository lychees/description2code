#include <iostream>
using namespace std;
#define MAX 101
char mat[MAX][MAX][MAX] = {'x'};
void engine(){
	int i, j, k;
	for(i = 0; i < MAX; i++ ){
		for(j = 0; j < MAX; j++ ){
			for(k = 0; k < MAX; k++ ){
				if((i > 0 && mat[i-1][j][k] == 'L')||
					(j > 0 && mat[i][j-1][k] == 'L')||
					(k > 0 && mat[i][j][k-1] == 'L')||
					(i > 0 && j > 0 && mat[i-1][j-1][k] == 'L')||
					(j > 0 && k > 0 && mat[i][j-1][k-1] == 'L')||
					(i > 0 && k > 0 && mat[i-1][j][k-1] == 'L')||
					(i > 0 && j > 0  && k > 0 && mat[i-1][j-1][k-1] == 'L')){
						mat[i][j][k] = 'W';
					}
					else 
						mat[i][j][k] = 'L';
			}
		}	
	}
}
int main()
{
   int Q, N1, N2, N3;
   cin >> Q;
   engine();
   while(Q--){
   		cin >> N1 >> N2 >> N3;
   		if (mat[N1][N2][N3] == 'W'){
   			cout<<"Ashima\n";
   		} else {
   			cout<<"Aishwarya\n";
   		}
   
   }
   
}
