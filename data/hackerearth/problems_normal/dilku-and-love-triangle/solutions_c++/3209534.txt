template<typename T>
T gcd(T q, T w) {
	while (w) {
		q %= w;
		swap(q, w);
	}
	return q;
}
template<typename T>
T lcm(T q, T w) {
	return q / gcd(q, w) * w;
}

vector<vector<int>> g;

vector<int> tin, tout;

int tim = 0;

vector<int> order;

const int L = 19;
vector<vector<int>> parent;

void dfs(int v, int p) {
	parent[0][v] = p;
	for (int j = 1; j < L; ++j) {
		parent[j][v] = parent[j - 1][parent[j - 1][v]];
	}
	tin[v] = tim++;
	order.push_back(v);
	for (int to : g[v]) {
		if (to == p) {
			continue;
		}
		dfs(to, v);
	}
	tout[v] = tim++;
}

bool upper(int a, int b) {
	return tin[a] <= tin[b] && tout[a] >= tout[b];
}

int lca(int a, int b) {
	if (upper(a, b)) {
		return a;
	}
	for (int j = L - 1; j >= 0; --j) {
		if (!upper(parent[j][a], b)) {
			a = parent[j][a];
		}
	}
	return parent[0][a];
}

const int shift = 1 << 20;
int tree[2 * shift + 5];

void update(int v, int val) {
	v += shift;
	tree[v] += val;
	v /= 2;
	while (v) {
		tree[v] = tree[2 * v] + tree[2 * v + 1];
		v /= 2;
	}
}

int rsq(int l, int r) {
	if (l >= r) {
		return 0;
	}
	if (l & 1) {
		return tree[l] + rsq(l + 1, r);
	}
	if (r & 1) {
		return rsq(l, r - 1) + tree[r - 1];
	}
	return rsq(l / 2, r / 2);
}

int get_sum(int l, int r) {
	return rsq(l + shift, r + shift);
}

void solve() {
	int n;
	int Q;
	cin >> n >> Q;
	tin.resize(n);
	tout.resize(n);
	g.resize(n);
	parent.assign(L, vector<int>(n));
	for (int i = 1; i < n; ++i) {
		int a, b;
		cin >> a >> b;
		--a; --b;
		g[a].push_back(b);
		g[b].push_back(a);
	}
	dfs(0, 0);

	while (Q--) {
		int type;
		cin >> type;
		if (type == 1) {
			int u, v;
			cin >> u >> v;
			--u; --v;
			update(tin[u], 1);
			update(tin[v], 1);
			int cur_lca = lca(u, v);
			update(tin[cur_lca], -1);
			if (cur_lca != 0) {
				update(tin[parent[0][cur_lca]], -1);
			}
		}
		else {
			int v;
			cin >> v;
			--v;
			cout << get_sum(tin[v], tout[v]) << "\n";
		}
	}

}
