#include<bits/stdc++.h>
using namespace std;
int main() {
	int t,i;
	cin>>t;
	map<string,int> mp;
	vector<string> v;
	while(t--) {
	    string a;
	    cin>>a;
	    mp[a]++;
	    v.push_back(a);
	    
	}
	sort(v.begin(),v.end());
	for(vector<string>::iterator it=v.begin(); it!=v.end(); it++) {
	    	if(mp[(*it)]) {
	    		cout<<(*it)<<' '<<mp[(*it)]<<endl;
	    		mp[(*it)]=0;
	    	}
	    }
	return 0;
}
