#include <iostream>
#include<cstdio>
#include<algorithm>
#include<cmath>
using namespace std;
#define LL long long int
int main()
{
    LL t,x,y,i,k,cnt,ans;
    scanf("%lld",&t);
    while(t--)
    {
    	scanf("%lld%lld",&x,&y);
    	ans=x^y;
    	cnt=0;
    	while(ans)
  		{
    		cnt += ans & 1;
    		ans >>= 1;
  		}		
    	printf("%lld\n",cnt);
    }
    return 0;
}
