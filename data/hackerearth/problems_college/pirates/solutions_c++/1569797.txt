#include<iostream>
#include<list>
#include<math.h>
using namespace std;
int main()
{
	int n,a[100000],i,j = 1;
	list<int> l;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
		if(i > 0)
		{
			if(abs(a[i-1] - a[i]) == 1)
			{
				j++;
			}
			else
			{
				l.push_back(j);
				j = 1;
			}
		}
	}
	l.push_back(j);
	l.sort();
	cout<<l.back();
	return 0;
}
