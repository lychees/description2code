#include <iostream>
#include <math.h>
using namespace std;


int main()
{
        int T= 0 ;
        cin >> T;
        while(T--)
        {
                std::string binA;
                std::string binB;
                std::string binC;
                cin >>binA >> binB;

                std::string::iterator itr = binA.begin();
                std::string::iterator itrb = binB.begin();
                for(;itr!=binA.end();itr++,itrb++)
                {
                        int diff = 0;
                        if( (*itr == '1' && *itrb == '0') || (*itr== '0' && *itrb=='1'))
                                diff = 1;
                        else
                                diff = 0;
                        int po = pow(2,diff);
                        if( po%2 == 0)
                                binC.append("1");
                        else
                                binC.append("0");

                }
                cout<<binC<<endl;

        }

}
