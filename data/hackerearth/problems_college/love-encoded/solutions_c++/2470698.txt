#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--) {
    	int k;
    	string str;
    	cin >> k >> str;
    	int l = str.size();
    	for(int i = 0; i < l; i++) {
    		int ind = str[i] - 65;
    		int ascii = (ind + 26 - k) % 26;
    		printf("%c", ascii + 65);
    	}
    	printf("\n");
    }
    return 0;
}
