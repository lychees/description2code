n = int(raw_input())
a = map(int, raw_input().split())
b = map(int, raw_input().split())
maxb = 0
maxd = 0
for i in range(n-1):
    dd = abs(a[i+1]-a[i])
    db = abs(b[i+1]-b[i])
    if dd > maxd:
        maxd = dd
    if db > maxb:
        maxb = db
if maxb > maxd:
    print "Brian"
    print maxb
elif maxd > maxb:
    print "Dom"
    print maxd
else:
    print "Tie"
