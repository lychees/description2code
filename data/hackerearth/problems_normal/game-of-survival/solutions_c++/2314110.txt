#include <iostream>
#include <list>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
	int test, numLen1, numLen2, num;
	bool flag = true;
	cin >> test;

	while (test--){

		cin >> numLen1;
		numLen2 = numLen1;
		vector<int> arr1(numLen1);
		vector<int> arr2(numLen1);
		
		while (numLen1--){
			cin >> num;
			arr1.push_back(num);
		}
		std::sort(arr1.begin(), arr1.end());
		
		while (numLen2--){
			cin >> num;
			arr2.push_back(num);
		}
		std::sort(arr2.begin(), arr2.end());

		vector<int>::iterator it1 = arr1.begin();
		vector<int>::iterator it2 = arr2.begin();

		for (; it1 != arr1.end(); it1++, it2++){
			if (*it1 < *it2){
				cout << "NO" << endl;
				flag = false;
				break;
			}
		}
		if (flag == true){
			cout << "YES" << endl;
		}
		
		flag = true;
		arr1.clear();
		arr2.clear();
	}

	return 0;
}
