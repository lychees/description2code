#include <iostream>
#include <algorithm>
#include<bits/stdc++.h>
#define ll long long
using namespace std;

int main()
{
    ll t,n,m=1000000;
    int a[1000001]={0},b[1000001]={0};
    	for(ll i=2;i<=m;i++){
    		ll count=0;
    		if(a[i]==1){
    			b[i]=count;
    			continue;
    		}
    		for(ll j=i;j<=m;j=j+i){
    			if(a[j]!=1){
    				count++;
    				a[j]=1;
    			}
    		}
    		b[i]=count;
    	}
    cin>>t;
    while(t--){
    	cin>>n;
    	cout<<b[n]<<endl;
    }
    return 0;
}
