std::vector< int > ndivlist;
int getDivs(ll val){
	ll sq = sqrt(val)+1;
	for(int i = 1;i<sq;i++){
		if(val%i == 0){
			ndivlist.pb(i);
			if(val/i != i)
			ndivlist.pb(val/i);
		}
	}
	return ndivlist.size();
}
