#include <bits/stdc++.h>
using namespace std;
#define sc(n) scanf("%lld",&(n))
#define mod 1000000007
long long modulo(long long a,long long b)
{
    long long x=1,y=a;
    while(b>0)
    {
        if(b&1==1){
            x=(x*y)%mod;
        }
        y = (y*y)%mod; 
        b>>=1;
    }
    return x%mod;
}
int main()
{
	long long n;
	sc(n);
	long long mul = 1;
	long long a[n],i,j,k,nof0=0;
	for(i=0;i<n;i++){
		sc(a[i]);
		if(a[i]==0){
			nof0++;
		}
		else
		mul = (mul*a[i])%mod;
	}
	long long q;
	sc(q);
	long long type;
	while(q--){
		sc(type);
		//cout<<mul<<endl;
		if(type==0){
			sc(j);
			sc(k);
			if(a[j-1]==0){
				nof0--;
			}
			else
				mul = (mul*modulo(a[j-1],mod-2))%mod;
			if(k==0){
				nof0++;
			}
			else
				mul = (mul*k)%mod;
			a[j-1] = k;
		}
		else{
			sc(j);
			long long ans;
			if(nof0==1&&a[j-1]==0){
				ans = mul%mod;
			}
			else if(nof0==0){
				ans = (mul*modulo(a[j-1],mod-2))%mod;
			}
			else{
				ans=0;
			}
			printf("%lld\n",ans);
		}
	}
	return 0;
}
