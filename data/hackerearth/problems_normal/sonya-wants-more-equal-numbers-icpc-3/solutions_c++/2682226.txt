#include <iostream>

using namespace std;
int ma[100005][102]={{0}};
int main()
{
    ios::sync_with_stdio(false);
    int n;
    cin>>n;
    int i,a[200],flag,k;
    for(i=0;i<n;i++)
        cin>>a[i];
    int j,x=0,l;
    for(i=n-1;i>=0;i--,x=0)
    {
        for(j=i;j>=0;j--)
        {
            x+=a[j];
            //cout<<x<<"\n";
            flag=0;
            for(k=0;k<n-1-i;k++)
            {
                if(ma[x][k]!=0 && ma[x][k]+k>n-1-i)
                {
                    flag=1;
                    break;
                }
            }
            if(flag==0)
            {
                //cout<<"a";
                ma[x][n-1-i]=i-j+1;
            }
            /*for(l=0;l<100005;l++)
                if(ma[l][n-1-i])
                    cout<<l<<" ";
            cout<<"\n";*/
        }
    }
    /*for(i=0;i<n;i++,cout<<"\n")
        for(j=0;j<100005;j++)
            if(ma[j][i])
                cout<<j<<" ";*/
    int cnt[100005]={0};
    for(i=1;i<100005;i++)
        for(j=0;j<n;j++)
        {
            if(ma[i][j])
                cnt[i]++;
        }
    int maxi=0;
    for(i=0;i<100005;i++)
        if(cnt[i]>maxi)
            maxi=cnt[i];
    cout<<maxi<<"\n";
    for(i=0;i<100005;i++)
        if(cnt[i]==maxi)
            cout<<i<<" ";
    return 0;
}
