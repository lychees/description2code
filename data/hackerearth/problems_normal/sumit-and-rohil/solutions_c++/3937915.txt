#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n,i,j;
	cin >> n;
	string s[n];
	for(i=0;i<n;i++){
		cin >> s[i];
	}
	int cnt=n;
	for(i=0;i<n;i++){
		for(j=0;j<i ;j++){
			 int k=0;
			 string s1=s[i];
			 int l1=(int)s1.size();
			 string s2=s[j];
			 int l2=(int)s2.size();
			 
			 if(s1[0]==s2[0] && s1[l1-1]==s2[l2-1] && l1==l2) {
			 	k++;
			 }
			 
			 sort(s1.begin(),s1.end());
	 	 	 sort(s2.begin(),s2.end());
			 if(s1==s2 && k>0){
			 	cnt--;
			 	break;
			 }
		}
	}
	cout << cnt << endl;
    return 0;
}
