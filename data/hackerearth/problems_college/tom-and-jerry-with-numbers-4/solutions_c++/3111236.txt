#include<bits/stdc++.h>
#define pb push_back
#define mp make_pair
#define f first
#define maxn 1010000
#define s second
#define ll long long int
#define inf 1000000014
#define infl (ll)(1e18+1)
#define mod 1000000007
#define sz(x) (int) x.size()
using namespace  std;
typedef pair<ll,ll> p;
int arr[maxn];int a,b,c;
 int main(int argc, char const *argv[])
{
  int n;
  int t;
  cin>>t;
  while(t--)
  {  
    scanf("%d%d%d",&a,&b,&c);
    for(int i=0;i<=a;i++)arr[i]=0;
    arr[0]=0;
    arr[b]=1;
    arr[c]=1;
    for(int i=1;i<=a;i++)
    {int cnt=0;
    if(i<=b&&i<=c)continue;
      if(i==b||i==c)continue;
      if(i>b&&arr[i-b]==1)
        cnt++;
      if(i>c&&arr[i-c]==1)
        cnt++;
      if(i>b&&i>c)
      {
        if(cnt==2)
          arr[i]=0;
         else
          arr[i]=1;
      }
      else if(i>min(b,c)&&i<max(b,c))
      {
        if(cnt==1)
          arr[i]=0;
        else
          arr[i]=1;
      }
    }
   cout<<arr[a]<<"\n";
   }
   return 0;
 }
