#include<bits//stdc++.h>
using namespace std;
int main()
{
	int t,len,i,j;
	string str;
	cin>>t;
	while(t--)
	{ 
		int c=0;
		cin>>str;
		len=str.length();
	     i=0;
	     j=len-1;
	     while(i<j)
	     {
	     	if(str[i]==str[j])
	     	  {
	     	  	i++;
	     	  	j--;
	     	  	  c++;
	     	  }
	     	  else  
	     	  { 
	     	  	i++;
	     	    c=0;
	     	    j=len-1;
	     	  	
	     	  }
	     }
	       if(c==0)
	        	     cout<<(2*len)-2*c-1<<endl;

	   else  if(c!=0 && len%2==0)
	     cout<<(2*len)-2*c-1<<endl;
	     else 
	    cout<<(2*len)-2*c<<endl;

		
	}
 
	}
