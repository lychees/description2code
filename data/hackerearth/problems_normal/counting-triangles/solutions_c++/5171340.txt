#include <bits/stdc++.h>

using namespace std;

long long n,i,a[1000009][3],b[10],sum[1000000], mod = 9e15+7,c;
map<pair<long long , pair<long long, long long>>, long long > mp;
int main(){
	cin>>n;
	for(i = 0; i < n; i++){
		cin>>a[i][0]>>a[i][1]>>a[i][2];
		b[0] = a[i][0];
		b[1] = a[i][1];
		b[2] = a[i][2];
		sort(b,b+3);
		a[i][0] = b[0];
		a[i][1] = b[1];
		a[i][2] = b[2];
	}
	for(i = 0; i < n; i++){
		mp[make_pair(a[i][0],make_pair(a[i][1],a[i][2]))]++;
	}c = 0;
	for(i = 0; i < n; i++){
		if(mp[make_pair(a[i][0],make_pair(a[i][1],a[i][2]))]==1)
			c++;
	}
	cout<<c<<endl;
	return 0;
}
