#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t,i,j,temp,ans;
	string d,s;
	cin>>t;
	while(t--)
	{
	    ans=INT_MAX;
		cin>>d>>s;
		if(d.size()>s.size())
		{
		    for(i=0;i<d.size()-s.size()+1;i++)
			{
				temp=0;
				for(j=0;j<s.size();j++)
					if(s[j]!=d[i+j])
						temp++;
				ans=min(ans,temp);
			}
		}
		else
		{
			for(i=0;i<(s.size()-d.size()+1);i++)
			{
				temp=0;
				for(j=0;j<d.size();j++)
					if(s[i+j]!=d[j])
						temp++;
				ans=min(ans,temp);
			}
		}
        cout<<ans<<endl;
	}
	return 0;
}
