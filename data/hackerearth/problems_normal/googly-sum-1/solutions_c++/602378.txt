#include<bits/stdc++.h>
using namespace std;
int main()
 {
 	int n,i,j,count=0;
 	vector<int> s(400005);
 	cin>>n;
	vector<int> a(n); 
    map<int,int> c;
 	for(i=0;i<n;i++)
 	{	
	 	scanf("%d",&a[i]);
 	  	for(j=0;j<i;j++)
 	  	  {if(s[a[i]-a[j]+200000])
 	  	    {
			   count++;
 	  	     	break;
 	  	    }
 	  	  }
 	  	  
 	  		if(c.find(a[i])!=c.end() ||  c[a[i]]<2)
 	  		for(j=0;j<=i;j++)
 	  	  	{
 	  	  		s[a[j]+a[i]+200000]=1;
 	  	  	}
 	  		else
 	  		 c[a[i]]++;
 	}
 	cout<<count<<endl;
 }
