#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	string a,b;
    	cin>>a>>b;
    	int n;
    	cin>>n;
    	
    	int l;
    	l=a.length();
    	int ue=0;
    	for(int i=0;i<l;i++){
            if(a[i]!=b[i]){
                ue++;
            }
        }
        if(n<ue) cout<<"No"<<endl;
    	if(n==ue)cout<<"Yes"<<endl;
        if(n>ue){
        	if((n-ue)%2) cout<<"No"<<endl;
        	else cout<<"Yes"<<endl;
        }
    }
    return 0;
}
