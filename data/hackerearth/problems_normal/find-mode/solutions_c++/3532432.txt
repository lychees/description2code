#include<iostream>
using namespace std;
struct p{
	int f;
	 int *ap;
};
int main(){
	unsigned int t,i,j,k=0,n,max,tax,no;
	cin>>t;
	struct p s[t];
	while(k<t){
		max=0;
		tax=0;
		no=0;
		cin>>n;
		unsigned int arr[n];
		for(i=0;i<n;i++){
		cin>>arr[i];
		if(max<arr[i])max=arr[i];}
		unsigned int a[max+1];
		for(i=0;i<=max;i++)
		a[i]=0;
		
		for(i=0;i<n;i++)a[arr[i]]++;
		
		for(i=0;i<=max;i++)if(tax<a[i])tax=a[i];
		
		for(i=0;i<=max;i++)if(a[i]==tax)no++;
		s[k].f=no;
		s[k].ap=new int[no];
		int g=no-1;
		for(i=0;i<=max&&g>=0;i++){
			if(a[i]==tax){
				s[k].ap[g]=i;
				g--;
			}
		}
		k++;
   }
   int l;
   for(i=0;i<t;i++){
   	for(j=0;j<s[i].f;j++)cout<<s[i].ap[j]<<" ";
   	cout<<"\n";
   }
   
	return 0;
	
}
