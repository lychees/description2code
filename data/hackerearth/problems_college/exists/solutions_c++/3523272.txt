#include<bits/stdc++.h>
#include<iostream>
#include<algorithm>
using namespace std;
typedef long long int lli;
typedef unsigned long long int ulli;

int main(){
	int t;
	cin >>t;
	int a[1000000],b[1000000];
	while(t>0){
	int n;
	cin >>n;
	int num;
	fill(a,a+1000000,0);
	fill(b,b+1000000,0);
	for(int i=0;i<n;i++){
	cin >>num;
	if(num>=0){
	a[num]++;
	}
	else{
	b[-num]++;
	}
	}
	int k;
	cin >>k;
	while(k>0){
	int n2;
	cin >>n2;
	if(n2>0){
	if(a[n2]>0){
	cout <<"Yes\n";
	}else{
	cout <<"No\n";
	}
	}
	else{
	if(b[-n2]>0){
        cout <<"Yes\n";
	}else{
	cout <<"No\n";
	}
	}
	k--;}
	
	t--;}
}
