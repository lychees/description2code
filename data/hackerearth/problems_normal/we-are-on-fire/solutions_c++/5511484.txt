#include<bits/stdc++.h>

using namespace std;

int arr[1007][1007],v[1007][1007];

int main()
{
	int i,n,j,c=0,x,y,x1,y1,m,q;
	
	queue<pair<int,int> > Q;
	pair<int,int> P;
	
	scanf("%d %d %d",&n,&m,&q);
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=m;j++)
		{
			scanf("%d",&arr[i][j]);
			v[i][j]=0;
			if(arr[i][j]==1)
				++c;
		}
	}
	for(i=0;i<q;i++)
	{
		scanf("%d %d",&x,&y);
		if(arr[x][y]==0 || (arr[x][y]==1 && v[x][y]==1))
			printf("%d\n",c);
		else
		{
			v[x][y]=1;
			c--;
			Q.push(make_pair(x,y));
			
			while(!Q.empty())
			{
				P=Q.front();
				Q.pop();
				
				x1=P.first;
				y1=P.second;
				
				if(x1-1>0 && arr[x1-1][y1]==1 && v[x1-1][y1]==0)
				{
					c--;
					v[x1-1][y1]=1;
					Q.push(make_pair(x1-1,y1));
				}
				if(y1-1>0 && arr[x1][y1-1]==1 && v[x1][y1-1]==0)
				{
					c--;
					v[x1][y1-1]=1;
					Q.push(make_pair(x1,y1-1));
				}
				if(x1+1<=n && arr[x1+1][y1]==1 && v[x1+1][y1]==0)
				{
					c--;
					v[x1+1][y1]=1;
					Q.push(make_pair(x1+1,y1));
				}
				if(y1+1<=m && arr[x1][y1+1]==1 && v[x1][y1+1]==0)
				{
					c--;
					v[x1][y1+1]=1;
					Q.push(make_pair(x1,y1+1));
				}
			}
			printf("%d\n",c);
		}	
	}
	return 0;
}
