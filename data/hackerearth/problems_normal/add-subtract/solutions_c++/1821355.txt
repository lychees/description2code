#include <iostream>
#include <cstdio>
#include <algorithm>
#include <vector>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--){
    	int n,k;
    	cin >> n >> k;
    	int res = 1000000000;
    	vector<int> arr;
    	for(int i = 0;i < n; ++i){
    		int x;
    		scanf("%d", &x);
    		arr.push_back(x);
    	}
    	for(int i = 0;i < n; ++i){
    		vector<int> ans;
    		for(int j = 0;j < n; ++j){
    			int g = arr[j] - arr[i];
    			ans.push_back(g > 0?5*g:-3*g);
    		}
    		sort(ans.begin(), ans.end());
    		int cur = 0;
    		for(int j = 0;j < k; ++j)cur += ans[j];
    		res = min(res, cur);
    	}
    	printf("%d\n",res);
    }
    return 0;
}
