#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	int len=(int)s.length();
    	int pos = -1;
    	for(int i=0,j=len-1;i<j;i++,j--)
    	{
    		if(s[i] != s[j])
    		{
    			if(i+1==j)
    			{
    				pos=j;
    				break;
    			}
    			
    			if(s[i] != s[j-1])
    			{
    				pos=i;
    				break;
    			}
    			else if(s[j] != s[i+1])
    			{
    				pos=j;
    				break;
    			}
    			else
    			{
    				
    				pos=i;
    				break;
    			}
    		}
    	}
    	cout << pos << "\n";
    }
}
