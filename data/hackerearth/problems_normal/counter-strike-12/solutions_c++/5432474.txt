#include<bits/stdc++.h>
using namespace std;
int main()
{
    int test;
    cin>>test;
    while(test--)
    {
        int n,m,d;
        int count=0;
        cin>>n>>m>>d;
        int arr[n][2];
        for(int i=0;i<n;i++)
        {
            cin>>arr[i][0]>>arr[i][1];
        }
        int temp =m;
        while(temp--)
        {
            int a,b;
            int dis;
            cin>>a>>b;
            for(int i=0;i<n;i++)
            {
                dis=abs(a-arr[i][0])+abs(b-arr[i][1]);
                if(dis<=d)
                {
                    count++;
                    break;
                }
            }
        }
        if(count>=m/2)
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
    }
}
