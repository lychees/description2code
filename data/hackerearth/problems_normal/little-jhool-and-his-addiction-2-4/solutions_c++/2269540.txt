#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    int n2,lim,val,min,max;
    int a[1000000];
    while(t--)
    {
    	cin>>n2>>lim;
    	for(int i=0;i<n2;i++) cin>>a[i];
    	sort(a,a+n2);
    	int l=0,r=n2-1,t;
    	min=max=a[l++]+a[r--];
    	for(;l<n2/2;l++,r--)
    	{
    		t=a[l]+a[r];
    		if(t>max) max=t;
    		if(t<min) min=t;
    	}
    	val=max-min;
    	if(val>lim) cout<<val<<"\nNo more girlfriends!\n";
    	else if(val==lim) cout<<val<<"\nLucky chap!\n";
    	else cout<<val<<"\nChick magnet Jhool!\n";   	
    	
    }
    return 0;
}
