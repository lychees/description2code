int main()
{
    int n,m,a[20],b[20];
    cin>>n>>m;
    for(int i=0;i<n;i++)
        cin>>a[i];

    for(int j=0;j<m;j++)
        cin>>b[j];

    for(int i=0;i<m;i++)
    {
        int flag=0;
        for(int j=0;j<n;j++)
        {
            int gc = gcd(360,a[j]);
            if(b[i]%gc==0)
            {
                flag=1;
                break;
            }
        }
        if(flag==1)
            cout<<"YES\n";
        else
            cout<<"NO\n";
    }
    return 0;
}
