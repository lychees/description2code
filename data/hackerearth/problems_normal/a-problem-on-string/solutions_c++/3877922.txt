#include <iostream>
#include <string.h>
using namespace std;

int main()
{
int N ,Q;
string nStr;
char sStart , sEnd;
long int result;
long int inp_str[26] = {0};
cin>>N;
cin>>nStr;
cin>>Q;

for(int i=0;i<N;i++)
inp_str[nStr[i]-'a']+=1;

while(Q)
{
cin>>sStart;
cin>>sEnd;
if(sStart !=sEnd)
result = inp_str[sStart-'a'] * inp_str[sEnd-'a'];
else
result = ((inp_str[sStart-'a']-1) * inp_str[sEnd-'a'])/2;
cout<<result<<endl;
Q--;
}

return 0;
}
