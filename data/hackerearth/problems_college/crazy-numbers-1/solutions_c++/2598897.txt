#include <iostream>
#include <cstring>

#define MOD 1000000007

using namespace std;

long long int countnumbers(int last, int N);
long long int cache[11][1000005];
long long int answers[1000005];

int main()
{
    int T;
    int N;
    int i;
    long long int ans = 0;
    cin>>T;
    
    answers[0] = 0;
    answers[1] = 10;
    memset(cache, -1, sizeof(cache));
    
    for(N = 2; N<=1000000; N++)
    {
    	ans = 0;
    	for(i=1; i<10; i++)
    	{
    		ans = (ans + countnumbers(i, N-1))%MOD; 
    	}
    	answers[N] = ans;
    }
    
    while(T--)
    {
    	cin>>N;
    	cout<<answers[N]<<endl;
    	
    }
    return 0;
}

long long int countnumbers(int last, int N)
{
	if(last<0 || last>9)
		return 0;
		
	if(N==0)
		return 1;
		
	if(cache[last][N] != -1)
		return cache[last][N];
		
	long long int ans = 0;
	
	ans = ((ans + countnumbers(last+1, N-1))%MOD+countnumbers(last-1, N-1))%MOD;
	
	return cache[last][N] = ans;
}
