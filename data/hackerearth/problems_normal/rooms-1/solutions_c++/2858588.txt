#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int i,j,k,l,m,n,x,t;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	int a[n+1];int b[n+1];x=0;
    	for(i=0;i<n;i++) scanf("%d",&a[i]);
    		for(i=0;i<n;i++) {scanf("%d",&b[i]);b[i]=b[i]+a[i];}
    		sort(a,a+n);
    		sort(b,b+n);
    		i=0;j=0;int mx=0;
    		while(i<n && j<n)
    		{
    			if(a[i]<b[j])
    			{
    				i++;x++;
    				mx=max(mx,x);
    			}
    			else{
    				j++;x--;
    			}
    		}
    		cout<<mx<<endl;
    }
    return 0;
}
