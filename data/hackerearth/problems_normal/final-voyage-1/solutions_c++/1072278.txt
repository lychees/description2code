using namespace std;
//int knapsack(v[],w[],n,maxload);

int knapsack(int v[],int w[],int n,int maxload)
{
	int i,j,k[1001][1001];
	
	for(i=0;i<=n;i++)
	{
		for(j=0;j<=maxload;j++)
		{
			if(i==0||j==0)
			{
			    k[i][j]=0;
			    //cout<<k[i][j]<<" ";
			}
			else
			{
				if(w[i-1]<=j)
				{
					k[i][j]=max(v[i-1]+k[i-1][j-w[i-1]],k[i-1][j]);
				//	cout<<k[i][j]<<" ";
				}
				else
				{
				  k[i][j]=k[i-1][j];
				  //cout<<k[i][j]<<" ";
				}
			}
		}
		//cout<<endl;
	}
	return k[n][maxload];
}


int main()


{
	int t,n,maxload,i,j;
	cin>>t;
	while(t--)
	{
		cin>>n;
		cin>>maxload;
		int v[n+1],w[n+1];
	
		for(j=0;j<n;j++)
		{
			cin>>w[j];
		}
			for(i=0;i<n;i++)
		{
			cin>>v[i];
		}
		int ans=knapsack( v, w, n, maxload);
		cout<<ans<<endl;
	}
	
	return 0;
}
