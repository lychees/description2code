#include <iostream>
using namespace std;
int min(int x,int y)
{
	if(x<=y)
	 return x;
	 return y;
}

int main()
{
	int tc,n,index,temp1,temp2;
    cin>>tc;   
    while(tc--)
    {
    	cin>>index;
    	cin>>n;
    	temp1=temp2=index;
    	while(n--)
    	{
            if(temp2>0&&temp1<2)
    	{
              temp1 = temp1+1;
              temp2=temp2-1;
              
    	}
    	else if(temp1==2)
    	{
    		temp1=temp1-1;
    		temp2 = temp1;
    	}
    	else if(temp2==0)
    	{
    		temp2=temp2+1;
    		temp1 = temp2;
    	}
       
    		
    	}
    	cout<<min(temp1,temp2)<<endl;
    }
    return 0;
}
