#print 'Hello World!'
import sys
a = []
while True:
	try:
		s = raw_input()
		for i in s:
			a.append(i)
		a.sort()
		for i in a:
			sys.stdout.write(i)
		a = []
		print("")
	except(EOFError):
		break
