from __future__ import print_function
n = input(); a = [int(i) for i in raw_input().split()];
a.sort(); e = 0; o = 0;
for i in range(0, n):
	if(a[i] % 2 == 0):	print(a[i], end = ' '); e += a[i];
	else:	o += a[i];
print(e, end = ' ');
for i in range(0, n):
	if(a[i] % 2 != 0):	print(a[i], end = ' ');
print(o, end = ' ');
