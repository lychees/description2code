#include <iostream>
#include <string>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	string s;
    	cin>>s;
    	if(s[0]-'a'<14)
    		cout<<s[0]-'a';
    	else
    		cout<<s[0]-'a'-26;
    	for(int i=1;i<s.length();i++){
    		if(s[i]-s[i-1]<=-13)
    			cout<<" "<<s[i]-s[i-1]+26;
    		else if(s[i]-s[i-1]>13)
    			cout<<" "<<s[i]-s[i-1]-26;
    		else
    			cout<<" "<<s[i]-s[i-1];
    	}
    	cout<<endl;
    }
    return 0;
}
