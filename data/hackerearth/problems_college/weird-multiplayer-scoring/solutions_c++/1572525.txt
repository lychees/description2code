#include <stdio.h>
#include<string.h>
int main()
{
    long int m,n,pt[100],p,flag=0,k,max,pwmp,i;
    scanf("%ld %ld",&m,&n);
    char name[100][100],s[100];
    k=0;
    while(n--)
    {
    	scanf("%s %ld\n",s,&p);
    	if(k==0)
    	{
    		strcpy(name[0],s);
    		pt[0]=p;
    		max=p;
    		pwmp=0;
    		k++;
    	}
    	else
    	{
    		flag=1;
    		for(i=0;i<k;i++)
    		{
    			if(!strcmp(name[i],s))
    			{
    				flag=0;
    				pt[i]=pt[i]+p;
    				if(pt[i]>max)
    				{
    					max=pt[i];
    					pwmp=i;
    					break;
    				}
    			}
    		}
    	}
    	if(flag)
    	{
    		strcpy(name[k],s);
    		pt[k]=p;
    		if(pt[k]>max)
    		{
    			max=pt[k];
    			pwmp=k;
    		}
    		k++;
    	}
    }
    printf("%s",name[pwmp]);
    return 0;
}
