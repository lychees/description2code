#include<iostream>
#include<string>
#define si(n) scanf("%d",&n)
#define pi(n) printf("%d\n",n)
#define sll(n) scanf("%lld",&n)
#define pll(n) printf("%lld\n",n)
#include<math.h>
#include<algorithm>
#define ll long long int 
using namespace std;
ll r[10001][10001];
ll f(string s,int n)
{
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
		{
			if(i==j) r[i][j] = 1;
			else r[i][j] = 0;
		//	cout<<r[i][j]<<" ";
		}
	//	cout<<endl;
	}
	ll sum = 0;
	for(int l=2;l<=n;l++)
	{
		for(int i=0;i<=n-l;i++)
		{
			int j = i+l-1;
			if(s[i]!=s[j]) r[i][j] = 0;
			else
			{	
				r[i][j] = 1;
				for(int k=1;k<l;k++)
				{
					for(int p=i+1;p<=j-k;p++)
					{
						r[i][j] += r[p][p+k-1];
						//cout<<"jscklsjd   "<<r[p][p+k-1]<<endl;;
					}
				}
			}
			
		}
	//	cout<<endl;
	}
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++) sum += r[i][j];
	}

	return sum;
}
int main()
{
	int t;si(t);
	while(t--)
	{
		string s;cin>>s;
		int n = s.length();
		ll ans = f(s,n);
		pll(ans);
	}
}
