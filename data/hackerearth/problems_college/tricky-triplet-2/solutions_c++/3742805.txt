#include <iostream>
using namespace std;

int main()
{
    long int n,i,j,k,m=0,sum=0;
    scanf("%ld",&n);
    long int a[n];
    while(m<n){
         scanf("%ld",&a[m]);
         m++;
	}
	for(i=0;i<=n-3;i++){
		for(j=i+1;j<=n-2;j++){
			for(k=j+1;k<=n-1;k++){
				if(a[j]-a[i]==a[k]-a[j])
				sum++;
			}
		}
	}
	printf("%ld",sum);
    return 0;
}
