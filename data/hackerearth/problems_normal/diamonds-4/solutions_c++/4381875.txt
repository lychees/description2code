#include <iostream>
using namespace std;

int main()
{
    int t,n,m,cnt;
    char d[1000][1000];
    cin>>t;
    while(t-- >0){
    	cin>>n>>m;
    	cnt=0;
    	for(int i=0;i<n;i++){
    		for(int j=0;j<m;j++){
    			cin>>d[i][j];
    			if(i!=0&&j!=0){
    				if(d[i][j]=='/'&&d[i-1][j-1]=='/'&&d[i][j-1]=='\\' &&d[i-1][j]==  '\\' )
    					cnt++;
    				
    			}
    		}
    		
    	}
    	cout<<cnt<<endl;
    	
    }
    return 0;
}
