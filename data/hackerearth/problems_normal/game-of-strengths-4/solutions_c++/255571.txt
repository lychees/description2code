#include<iostream>
#include<stdlib.h>
#include<algorithm>
#define MAX 100004
#define MOD 1000000007
using namespace std;
int compare (const void * a, const void * b)
{
  return ( *(int*)a - *(int*)b );
}
int main() {
	long long int arr[MAX],pre[MAX],next[MAX],max=0;
	long long int ans= 0;
	int N,T;
	scanf("%d",&T);
	while(T--) {
		max=0;
		scanf("%d",&N);
		for(int i=0;i<N;i++) {
			scanf("%lld",&arr[i]);
			if(arr[i]>max) max=arr[i];
		}
		pre[0]=0;
		next[0]=N-1;
		for(int i=1;i<N;i++) {
			pre[i] =(pre[i-1] + next[i-1] - i);
			next[i] = (N-i-1);
		}
		sort(arr,arr+N);
		ans=0;
		for(int i=0;i<N-1;i++) {
			ans += (pre[i] + next[i])*(arr[i+1]-arr[i]);
			ans = ans%MOD;
		}
		cout<<(ans*max)%MOD<<endl;
	}
}
