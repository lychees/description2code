#include <iostream>

using namespace std;

int invert_number(int number)
{
	int invert = 0;

	while(number >0)
	{
		invert*=10;
		invert+=number%10;
		number/=10;
	}

	return invert;
}

int main()
{
	int n_test_cases;
	int elem1, elem2;

	cin>>n_test_cases;

	for(int i=0;i<n_test_cases;i++)
	{		
		// Got the data
		cin>>elem1>>elem2;

		int inv_elem1 = invert_number(elem1);
		int inv_elem2 = invert_number(elem2);

		int inv_sum = inv_elem1 + inv_elem2;
		int sum = invert_number(inv_sum);

		// Give result
		cout<<sum<<endl;		
	}
	
	return 0;
}
