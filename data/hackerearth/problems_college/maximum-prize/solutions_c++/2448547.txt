#include<bits/stdc++.h>
#define si(x) scanf("%d",&x)
#define sii(x,y) scanf("%d%d",&x,&y)
#define sll(x,y) scanf("%ld%ld",&x,&y)
#define sl(x) scanf("%ld",&x)
using namespace std;
typedef long long ll;
const int MOD=1000000007;
int main(){
	int t;
	si(t);
	while(t--){
		int n,k;
		sii(n,k);
		int arr[n];
		for(int i=0;i<n;i++){
			si(arr[i]);
		}
		sort(arr,arr+n);
		ll sum=0;
		int count=0;
		for(int i=n-1;i>=0;i--){
			if(count==k){
				break;
			}
			sum+=arr[i];
			count++;
		}
		printf("%ld\n",sum);
	}
}
