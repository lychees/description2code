#include<stdio.h>
#include<string.h>
int ar[100000];
int br[100000];
int main()
{
    long long m,n,c;
    int num=10010;
    //scanf("%d",&num);
    br[1]=1;
    for(m=2; m<=num; m++)
    {
            c=0;
        if(ar[m]==0)
        {
            br[m]=1;
            ar[m]=1;
            for(n=m*m; n<=num; n+=m)
            {
                ar[n]=1;
                c++;
                if(c==0)
                break;
            }

        }
    }
    int test;
    scanf("%d",&test);
    for(int i=1;i<=test;i++)
    {
        if(!br[i])
            printf("%d ",i);
    }
return 0;
}
