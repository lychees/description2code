#!/usr/bin/env python2

def gcd(a, b):
    while b:
        a, b = b, a % b
    return a


n, k = map(int, raw_input().split())
q = 2**n
a = [0] * (n+1)
a[0] = 1
for i in range(1, n+1):
    a[i] = a[i-1]*2
    if i == k:
        a[i] -= 1
    elif i > k:
        a[i] -= a[i-k-1]
p = a[n]
p = q - p
d = gcd(p, q)
p = p // d
q = q // d
print('{}/{}'.format(p, q))
