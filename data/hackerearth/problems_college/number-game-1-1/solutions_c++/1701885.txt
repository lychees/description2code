#include <iostream>
using namespace std;

int main()
{
    int i,t,j=0;
    long int n;
    cin>>t;
    for(i=1;i<=t;i++)
    {
    	j=0;
    	cin>>n;
    	while(n!=1)
    	{
    	if(n%3==0)
    	{
    		j++;
    		n=n/3;
    		continue;
    	}
    	else if(n%2==0)
    	{
    		j++;
    		n=n/2;
    		continue;
    		
    	}
    	else
    	{
    		j++;
    		n=n-1;
    	}
    	}
    	cout<<j;
    	cout<<endl;
    		
    }
    return 0;
}
