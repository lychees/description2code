#include <bits/stdc++.h>
using namespace std;
vector<int>v[10005];
int visit[10005];
int dp[10005];
int ans=0;
int dfs(int u,int n)
{
	if(u==n)
	return 1;
	if(visit[u]==1)
	return dp[u];
	visit[u]=1;
	for(int i=0;i<v[u].size();i++)
	{
		int node=v[u][i];
		dp[u]=dp[u]+dfs(node,n);
		
		
	}
	return dp[u];
}
int main()
{
	int n;
	scanf("%d",&n);
	int x,y;
	while(1)
	{
		scanf("%d%d",&x,&y);
		if(x==0&&y==0)
		break;
		v[x].push_back(y);
	}
    int ans=dfs(1,n);
    cout<<ans;
    return 0;
}
