#include <iostream>
#include <string>
using namespace std;

int main()
{
    int N;
    cin>>N;
    while(N--){
    	int ans=0;
    	string A,B;
    	cin>>A>>B;
    	int la,lb;
    	la = A.length();
    	lb = B.length();
    
    	if(la<lb){
    		for(int i =la-1;i>=0;i--){
    			int na = A[i]-48;
    			int nb = B[i]-48;
    			if((na+nb)>10)
    			ans++;
    		}
    	} else {
    		for(int i =lb-1;i>=0;i--){
    			int na = A[i]-48;
    			int nb = B[i]-48;
    			
    			if((na+nb)>=10)
    			ans++;
    		}
    	}
    	if(ans==0){
    		cout <<"No carry operation";
    	} else if (ans ==1){
    		cout <<"1 carry operation";
    	} else {
    		cout<<ans<<" carry operations";
    	}
    	cout<<"\n";
    }
    return 0;
}
