#include<bits/stdc++.h>

using namespace std;

char s[10000000];

int main() {
    int t,i;
    gets(s);
    t = atoi(s);
    while(t--) {
        gets(s);
        string v = "";
        for(i = strlen(s)-1; i>=0 ; i--) {
            if(s[i] == ' ') {
                reverse(v.begin(),v.end());
                cout<<v<<" ";
                v = "";
            } else v.push_back(s[i]);
        }
        if(v != "") {
            reverse(v.begin(),v.end());
            cout<<v<<" ";
        }
        cout<<endl;
    }
    return 0;
}
