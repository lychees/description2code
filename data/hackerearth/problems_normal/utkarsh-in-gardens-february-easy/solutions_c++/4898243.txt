#include<bits/stdc++.h>
using namespace std;
bitset<2015>A[2015];
int main(void)
{
    long long int  i,j,n,a,count=0;
    cin>>n;
    for(i=0;i<n;i++)
        for(j=0;j<n;j++)
          {
              cin>>a;
              A[i][j]=a;
          }
    for(i=0;i<n;i++){
       for(j=i+1;j<n;j++)
       {
         a=(A[i]&A[j]).count();
         count+=a*(a-1)/2;
       }}
    cout<<count/2;
    return 0;
}
