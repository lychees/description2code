#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    sort(a,a+n);
    int p,q;
    int c=0,i;
    for(i=0;i<n;)
    {
    	p=a[i];
    	q=p+4;
    	i++;
    	while(a[i]<=q && i<n)
    	i++;
    	
    	c++;
    	
    }
    if(i<n)
    c++;
    cout<<c<<"\n";
}
