#include<bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
	string s;
	cin>>s;
	ll x=0,y=0,count=0;
	set< pair<ll,ll> >fd;
	set< pair<ll,ll> >::iterator it;
	ll len=s.length();
	ll i,j;
	fd.insert(make_pair(0,0));
	for(i=0;i<len;i++)
	{
        if(s[i]=='L')
        y--;
        else if(s[i]=='R')
        y++;
        else if(s[i]=='U')
        x--;
        else if(s[i]=='D')
        x++;
        it=fd.find(make_pair(x,y));
        if(it!=fd.end()) count++;
        else fd.insert(make_pair(x,y));
    }
	cout<<count<<endl;
	return(0);
}
