#include <iostream>
#include <vector>
using namespace std;

int main()
{	
	int N,i,j,pos=0;
    vector<int>V;
    string S;
    cin>>N>>S;
    for(i=0;i<N;i++)
        V.push_back(i+1);
    while(1)
    {
        for(j=0;j<S.size();j++)
        {
            if(pos>=V.size())
                pos=0;
            if(S[j]=='b')
                {V.erase(V.begin()+pos);pos--;}
            if(V.size()==1)
                break;
            pos++;
            //cout<<"J is "<<j<<endl;
        }
        if(V.size()==1)
            break;
    }
    cout<<V[0];
    return 0;
}
