main()
{
	
	int n;
	
	sc1(n);
	
	int a[1000001];
	
	mset(a,0);
	
	rep(n)
	{
		
		int temp1,temp2;
		
		sc1(temp1);
		sc1(temp2);
		
		a[temp1] = temp2;
		
	}
	
	int max_r[1000001];
	
	mset(max_r,0);
	
	int m=-1;
	
	for(int i = 1000000;i > 0;i--)
	{
		
		m = max(a[i],m);
		
		max_r[i] = m;
		
	}
	
	int guest ;
	long long int r=0;
	
	sc1(guest);
	
	rep(guest)
	{
		
		int temp;
		
		sc1(temp);
		
		r += (max_r[temp]/temp)*100;
	
	}
	
	pf2(r);
}
