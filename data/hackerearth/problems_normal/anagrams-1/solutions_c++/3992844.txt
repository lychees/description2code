#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	cin.ignore(100,'\n');
	while(t--){
		string str1;
		getline(cin,str1);
		string str2;
		getline(cin,str2);
		int arr1[26]={0};
		int arr2[26]={0};
		for(int i=0;i<(int)str1.size();i++){
			if((str1[i]>='A'&&str1[i]<='Z')||(str1[i]>='a'&&str1[i]<='z')){
				arr1[tolower(str1[i])-'a']++;
			}
		}
		for(int i=0;i<(int)str2.size();i++){
			if((str2[i]>='A'&&str2[i]<='Z')||(str2[i]>='a'&&str2[i]<='z')){
				arr2[tolower(str2[i])-'a']++;
			}
		}
		//int diff[26]={0};
		string res="";
		int oddCount=0;
		int odd=0;
		bool flag=true;
		bool done=false;
		for(int i=0;i<26;i++){
			int x=abs(arr1[i]-arr2[i]);
			if(x>0){
				done=true;
				if(x&1){
					oddCount++;
					odd=i;
					x-=1;
					int half=x/2;
					while(half--){
						res+=(char)(i+'a');
					}
				}
				else{
					int half=x/2;
					while(half--){
						res+=(char)(i+'a');
					}
				}
				if(oddCount>1){
					flag=false;
					break;
				}
			}
		}
		if(!done){
			cout<<"YES\n";
		}
		else if(!flag){
			cout<<"NO LUCK\n";
		}
		else{
			string rev=res;
			reverse(rev.begin(),rev.end());
			if(oddCount){
				res+=(char)(odd+'a');
			}
			cout<<(res+rev)<<endl;
		}
	}
}
