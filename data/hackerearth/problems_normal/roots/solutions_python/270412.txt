def main():
    n = int(raw_input())
    assert(0 <= n <= 100)
    tot = 0
    while n > 0:
        n += -1
        a, b, c = raw_input().split()
        a, b, c = int(a), int(b), int(c)
        assert(-100 <= a <= 100)
        assert(-100 <= b <= 100)
        assert(-100 <= c <= 100)

        if b*b >= 4 * a * c:
           tot += 1

    print tot

if __name__ == '__main__':
    main()
