for i in range(input()):
    n = input()
    s = raw_input()
    l = s.split()
    j = 0
    inverse = True
    while j < n:
        e = int(l[j])
        p = int(l[e-1])
        if j != (p-1):
            inverse = False
            break
        j += 1
    if inverse:
        print "inverse"
    else:
        print "not inverse"
