from UserString import MutableString


def solve(a, b):
    count = 0
    for i in xrange(len(a)+1):
        c = MutableString(a)
        c.insert(i, b)
        if c == c[::-1]:
            count += 1
    return count


for _ in xrange(input()):
    a = MutableString(raw_input())
    b = MutableString(raw_input())
    print solve(a, b)
