#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int N,a=0,b=0,i,j,A[100][100];
    cin >> N;
    for (i=0;i<N;i++)
    	for (j=0;j<N;j++)
    	cin >> A[i][j];
    for (i=0;i<N;i++)
    	for (j=0;j<N;j++)
    	{
    		if (i==j)
    		a+=A[i][j];
    		if(i+j==N-1)
    		b+=A[i][j];
    	}
    cout << abs(a-b);
    return 0;
}
