#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int i,j,k,n;
    int t;
    cin>>t;
    for(i=0;i<t;i++)
    {
    	cin>>n;
    	int a[n];
    	for(j=0;j<n;j++)
    	{
    	  cin>>a[j];
    	}
    	sort(a,a+n);
    	int ans=0;
    	for(j=n-1;j>=0;j--)
    	{
    		if(a[j]-1==a[j-1])
    		{
    			ans++;
    			j--;
    		}
    		else
    		ans++;
    	}
    	cout<<ans<<"\n";
    }
    return 0;
}
