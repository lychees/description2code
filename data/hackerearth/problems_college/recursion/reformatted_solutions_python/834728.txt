# Date with Avni 2
import sys
from itertools import permutations

t = int(sys.stdin.readline())

while t > 0:
    t -= 1
    n = int(sys.stdin.readline())
    dig = [x for x in range(1, n+1)]
    dig_s = ""
    for d in dig:
        dig_s += str(d)
    comb = list(permutations(dig_s, n))
    s = ""
    for per in comb:
        for char in per:
            s += char
        s += " "
    print(s)
