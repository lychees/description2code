#include <bits/stdc++.h>
using namespace std;

#define sd(x) scanf("%d", &x)
#define sld(x) scanf("%lld", &x)
#define PB push_back
#define MP make_pair
#define F first
#define S second
#define MOD 1000000007
#define INF 1000000007
#define N 112345
#define LL long long
#define D double

LL dp[N], sum[N];

inline void Solve(){
    LL x, y;
    sld(x); sld(y);
    LL ans = sum[y] - sum[x - 1];
    ans %= MOD;
    if(ans < 0) ans += MOD;
    printf("%lld\n", ans);
    return;
}

int main(){
    //freopen("input.txt", "r", stdin);
    LL t = 1, k;
    sld(t); sld(k);
    dp[0] = 1;
    sum[0] = 1;
    for(int i = 1; i <= 100100; i++){
        dp[i] = dp[i - 1];
        if(i >= k){
            dp[i] += dp[i - k];
            if(dp[i] >= MOD) dp[i] -= MOD;
        }
        sum[i] = dp[i] + sum[i - 1];
        if(sum[i] >= MOD) sum[i] -= MOD;
    }
    while(t--){
        Solve();
    }
    return 0;
}
