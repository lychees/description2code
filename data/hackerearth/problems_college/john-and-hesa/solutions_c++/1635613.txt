#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    
    while(t--){
    	int x,p,c=0;
    	cin >> x >> p;
    	
    	while(1){
    	
    		if(x%p!=0){
    			break;
    		}else{
    	
    	x=x/p;
    		c++;
    		}
    	}
    	if(x==1)x--;
    	cout << c << " " << x << endl;
    	
    }
    
    return 0;
}
