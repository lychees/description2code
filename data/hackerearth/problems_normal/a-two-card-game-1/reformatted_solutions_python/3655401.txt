n = int(raw_input())
mi = []
for i in range(n):
    a = int(raw_input())
    mi.append(a)
di = dict()
for i in mi:
    if i in di:
        di[i] += 1
    else:
        di[i] = 1
count = 0
for i in di:
    count += di[i]*(di[i]-1)
t = (len(mi)*(len(mi)-1)-count)/float((len(mi)-1)*len(mi))
print('%.6f' % (t/2)+" "+'%.6f' % (t/2))
