#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	for(int i = 0;i<t;i++)
	{
		int n,temp,*a;
		a = new int[n];
		cin>>n;
		for(int j = 0;j<n;j++)
		{
			cin>>temp;
			cin>>a[temp-1];
		}
		
		//cout<<"hello"<<endl;
		int *DP = new int[n],max = 1;
		DP[0] = 1;
		for(int k = 1; k< n;k++)
		{

			DP[k] =1;
			//cout<<"hello"<<endl;
			for(int l = k-1;l>=0;l--)
			{

				//cout<<"hello"<<endl;
				if (DP[l] + 1 > DP[k] && a[l] < a[k])
				{
					DP[k] = DP[l] + 1;
					//cout<<"hello"<<endl;
				}

				if(DP[k] > max)	
					max = DP[k];
			}
		}
		cout<<max<<endl;
	}
}
