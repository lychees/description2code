#include <bits/stdc++.h>
#define ll long long
using namespace std;
int dx[]={0,1};
int dy[]={1,0};
struct pos
{
    int x,y;
};
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        int a[n];
        for(int i=0;i<n;i++)
            cin>>a[i];
        int l=0,r=n-1;
        while(r-l>1)
        {
            if(a[r]==1)
            {
                a[r]=0;
                r--;
            }
            if(a[l]==-1)
            {
                a[l]=0;
                l++;
            }
            for(int i=l+1;i<=r;i++)
            {
                if(a[i]==-1 && a[i-1]==1)
                    swap(a[i],a[i-1]);
            }

        }
       int ans;
       if(a[l]==1&&a[r]==1)
        ans=l+1;
       else if(a[l]==-1&&a[r]==-1)
        ans=r+1;
       else if(a[l]==-1&& a[r]==1)
       ans=r+1;
       else
        ans=l+1;
       cout<<ans<<endl;
    }
    return 0;
}
