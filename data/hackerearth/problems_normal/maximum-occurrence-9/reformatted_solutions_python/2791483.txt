stri = raw_input("")
strl = list(stri)
charac = chr(127)
countc = -1
for i in range(len(strl)):
    if stri.count(strl[i]) >= countc:
        if stri.count(strl[i]) == countc:
            if ord(strl[i]) < ord(charac):
                charac = strl[i]
                countc = stri.count(strl[i])
        else:
            charac = strl[i]
            countc = stri.count(strl[i])
print "%s %d" % (charac, countc)
