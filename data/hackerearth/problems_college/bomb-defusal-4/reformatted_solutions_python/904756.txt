for i in xrange(input()):
    p, x, y, u, v = map(int, raw_input().split())
    a = abs(x-u)
    b = abs(y-v)
    diff = abs(a-b)
    if a == 0 and b == 0 and p > 0:
        print "Counter-Terrorists Win !"
    elif diff % 2:
        print "Terrorists Win !"
    elif diff == 0:
        if p > 1:
            print "Counter-Terrorists Win !"
        else:
            print "Terrorists Win !"
    else:
        if p > 2:
            print "Counter-Terrorists Win !"
        else:
            print "Terrorists Win !"
