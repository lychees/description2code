const int mod = 1000000000 + 7 ;
const int maxn = 1000000 + 10 ;
const int inf = 2*1e9 ;


void solve(){
	int n ; cin >> n ;
	vector<int> A(n+1,0) , ans(n+1,0) ;
	for(int i=1;i<=n;i++) cin >> A[i] ;
	int mx = 0 ; 
	for(int i=1;i<=n;i++){
		if(A[i] >= mx)
			ans[i] = 1 ; 
		mx = max(mx,A[i]) ;	
	}
	mx = 0 ; 
	for(int i=n;i>=1;i--){
		if(A[i] >= mx)
			ans[i] = 1 ; 
		mx = max(mx,A[i]) ;	 
	}
	for(int i=1;i<=n;i++) if(ans[i]) cout << i << " " ;
	cout << endl ;
}
int main(){


	int t ; t = 1 ;
	while(t -- ) solve() ;
	return 0 ; 
}
