#include<iostream>
using namespace std;
int check(char s[],int l)
{
    for(int i=0;i<l/2;i++)
    {
        if(s[i]!=s[l-i-1])
            return 0;
    }
    return 1;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        string s1,s2;
        cin>>s1>>s2;
        int l1,l2,i,j,c=0;
        l1=s1.length();
        l2=s2.length();
        for(i=0;i<=l1;i++)
        {
            char s3[l1+l2];
            for(j=0;j<i;j++)
                s3[j]=s1[j];
            for(j=i;j<i+l2;j++)
                s3[j]=s2[j-i];
            for(j=i+l2;j<l1+l2;j++)
                s3[j]=s1[j-l2];
            if(check(s3,l1+l2))
                c++;
                //cout<<s3<<endl;
        }
        cout<<c<<endl;
    }
}
