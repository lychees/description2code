#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int hash[26];
		char str[100000];
		long long int product=1;
		memset(hash,0,sizeof(hash));
		int cnt=0;
		cin>>str;
		for(int i=0;str[i]!='\0';i++)
		{
			hash[str[i]-'a']++;
		}
		for(int i=0;i<26;i++)
		{
			if(hash[i]%2!=0)
				cnt++;
		}
		if(cnt>1)
		{	
			for(int i=0;str[i]!='\0';i++)
			{
				product*=str[i]-'a'+1;
			}
			cout<<product<<endl;
		}
		else
			cout<<"Palindrome"<<endl;
	}

}
