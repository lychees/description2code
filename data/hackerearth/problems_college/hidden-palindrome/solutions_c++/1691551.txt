#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    
    while(t--) {
    	string s;
    	
    	cin >> s;
    	int ans = 1;
    	for(int i = 0; i < s.size(); i++) {
    		int l = 1;
    		while(i-l >= 0 && i+l < s.size() && s[i-l] == s[i+l]) {
    			l++;
    		}
    		ans = max(ans, 2*l - 1);
    		if(i == 0) continue;
    		l = i-1;
    		int r = i;
    		while(l >= 0 && r < s.size() && s[l] == s[r]) {
    			l--, r++;
    		}
    		l++, r--;
    		ans = max(ans, r-l+1);
    	}
    	cout << ans << endl;
    }
    return 0;
}
