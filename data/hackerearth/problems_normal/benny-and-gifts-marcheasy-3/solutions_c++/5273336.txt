#include "bits/stdc++.h"
using namespace std;
int main()
{
	pair<long int,long int> temp;
	map< pair<long int,long int> , bool > mymap;
	temp.first=temp.second=0;
	long int ans=0;
	string s;
	cin>>s;
	mymap[temp]=1;
	for(long int i=0;i<s.length();i++)
	{
		if(s[i] == 'U')
			temp.first++;// = temp.first + 1;
		else if(s[i] == 'D')
			temp.first--;// = temp.first - 1;
		else if(s[i] == 'L')
			temp.second--;
		else
			temp.second++;
		if(mymap[temp]==1)	
			ans++;
		mymap[temp]=1;
	}
	cout<<ans<<endl;
	return 0;
}
