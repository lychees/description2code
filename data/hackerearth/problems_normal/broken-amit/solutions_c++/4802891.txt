#define ll long long int
#define maxN 17
#define maxVal 100000000
#define minVal -100000000
#define mod 1000000007LL

#define gcd(a,b) __gcd(a,b)

using namespace std;

int n;
int l[maxN+5],d[maxN+5];
ll dp[1000005];

ll compute(int visit)
{
	if(visit==((1<<n)-1))
		return 0;
	if(dp[visit]!=-1000000000000)
		return dp[visit];
	int i,c=0;
	ll s=0,ans=-1000000000005;
	for(i=0;i<n;i++)
		if((visit&(1<<i))!=0)
			c++;
	for(i=0;i<n;i++)
		if((visit&(1<<i))==0)
			s=s+(ll)(l[i]-i*c*d[i]);
	//printf("%d\n",s);
	for(i=0;i<n;i++)
	{
		if((visit&(1<<i))==0)
			ans=max(ans,s+(ll)(l[i]-i*c*d[i])*(n-c-1)+compute(visit|(1<<i)));
	}
	return dp[visit]=ans;
}

int main()
{
    int i;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    	scanf("%d",&l[i]);
    for(i=0;i<n;i++)
    	scanf("%d",&d[i]);
    fill(dp,dp+(1<<n)+5,-1000000000000);
    printf("%lld\n",compute(0));
    return 0;
}
