#include <bits/stdc++.h>
using namespace std;
#define ll long long
ll lis(vector<ll>&v,ll n)
{
	ll i,ans=1,j,maxi=1;
	vector<ll> lis(n+1,1);
	for(i=1;i<n;i++)
	{
		for(j=0;j<i;j++)
		{
			if(v[i]>v[j])
			{
				lis[i]=max(lis[i],lis[j]+1);
			}
		}
		maxi=max(lis[i],maxi);
	}
	return maxi;
}
	
int main()
{
    vector<ll> v(100000);
    ll i,t,n,ans,cal,elem,siz,j;
    scanf("%lld",&t);
    while(t--)
    {
    	ans=0;
    	scanf("%lld",&n);
    	siz=0;
    	j=0;
    	for(i=0;i<n;i++)
    	{
    		scanf("%lld",&elem);
    		if(elem<=0)
    		{
    			ans++;
    		}
    		else
    		{
    			v[j]=elem;
    			siz++;
    			j++;
    		}
    	}
    	cal=lis(v,siz);
    	printf("%lld\n",(n-cal)+ans);
    }
    return 0;
}
