
#include<iostream>
#include<algorithm>
typedef long long LL;
using namespace std;

bool comp_rever(LL a,LL b)
{
	return (a>b);
		
	
		
}

int main()
{
	LL T;
	cin>>T;
	while(T--)
	{
		LL n;
		cin>>n;
		LL boys[10100],girls[10100];
		for(LL i=0;i<n;i++)
			cin>>boys[i];
			for(LL i=0;i<n;i++)
			cin>>girls[i];
			sort(girls,girls+n);
			sort(boys,boys+n,comp_rever);
			LL sum=0;
			for(LL i=0;i<n;i++)
				if((boys[i]%girls[i]==0)||(girls[i]%boys[i]==0))
				 sum++;
			cout<<sum<<endl;
	}
			
			
			
	
	return 0;
}
