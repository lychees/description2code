#include<bits/stdc++.h>
#define nax 1000005
using namespace std;

long long n,k,t,x,arr[nax],dp[nax],cnt[nax];

long long binarysearch(long long arr[],long long  data,long long start,long long  n)
{

   long long high=n;
   long long low =start;
   long long mid;

    while(high>=low)
    {

            mid=(high+low)/2;
            if(arr[mid]==data)
                return mid;
            else if(arr[mid]>data)
                high=mid-1;
            else
                low=mid+1;

    }

    return high;

}
int main()
{
//freopen("input1.txt", "r", stdin);

    cin>>t;
    long long  xy=1;
    while(t--)
    {
    cin>>n>>k;
    arr[0]=0;
    dp[0]=0;
    for(long  i=0;i<nax;i++)
        cnt[i]=0;
    for(int i=1;i<=n;i++)
    {
     cin>>arr[i];
     dp[i]=dp[i-1]+arr[i];
    }
    for(long i=1;i<=n;i++)
    {
        if(arr[i]>k)
            continue;
        x=binarysearch(dp,dp[i-1]+k,i,n);
        cnt[x-i+1]++;

    }

    for(long i=n-1;i>=1;i--)
        cnt[i]+=cnt[i+1];
    long long int sum=0;
    for(long i=1;i<=n;i++)
        cout<<cnt[i]<<" ";

    }
}
