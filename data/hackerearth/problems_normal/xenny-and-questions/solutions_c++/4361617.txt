{

	{
		int n,a[100005];int k;
		scanf("%d%d",&n,&k);
		for(int i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
		}
		sort(a,a+n);
		/*for(int i=0;i<n;i++)
		{
			printf("%d",a[i]);
		}*/
		int j=k;
		ll ans=0;
		for(int i=n-1,j=k;j>0 && i>=0;i--,j--)

			ans+=a[i];
		}
		printf("%lld\n",ans);
	}
	return 0;
}
