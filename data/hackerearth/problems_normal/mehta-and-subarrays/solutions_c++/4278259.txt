#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define pii pair<ll,ll>
#define pll pair<ll,ll>
#define mp make_pair
#define ff first
#define ss second
#define pb push_back
#define pf pop_front
#define IOS ios::sync_with_stdio(false)
ll n ,  a[1000006] ;
pair <ll,ll> pre[1000006];
int main(){

	ll i , j , k ;
    cin >> n ;
    for(i = 1; i <= n ; i++)cin >> a[i];
    ll tot = 0;
    pre[0].ss = 0;
    for(i = 1 ; i <= n  ; i++){
        pre[i].ff = pre[i-1].ff + a[i];
        pre[i].ss = i;
    }
    sort(pre,pre+n+1);
    ll min1 = pre[0].ss;
    ll temp = 0;
    ll cnt = 0;
    for(i = 1 ; i <= n ;i++){
        ll val = max(0ll , pre[i].ss-min1);
        if(val > temp){
            temp = val;
            cnt = 1;
        }else if(val == temp)cnt++;
        min1 = min(min1,pre[i].ss);
    }
    if(temp == 0)cout << "-1\n";
    else cout << temp << " " << cnt << endl;

	return 0;
}
