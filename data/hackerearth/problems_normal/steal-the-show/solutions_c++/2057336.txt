#include <stdio.h>
#include <bits/stdc++.h>

using namespace std;

int jos(int n, int k, int m) {
	int a;
	for (a = m + 1; a <= n; a++)
		m = (m + k) % a;
	return m;
}

typedef unsigned long long xint;

xint jos_large(xint n, xint k, xint m) {
	if (k <= 1) return n - m - 1;

	xint a = m;
	while (a < n) {
		xint q = (a - m + k - 2) / (k - 1);

		if (a + q > n)	q = n - a;
		else if (!q)	q = 1;

		m = (m + q * k) % (a += q);
	}

	return m;
}

int main(void) {

	int n;

	int a[3] ;
	int b[100];
	int c[100];
	cin >> a[0] >> a[1] >> a[2];
	cin >> n;
	for(int i = 1; i <= n; i++) {
		cin >> b[i];
		c[i] = b[i];
	}
	int ans = 0;
	sort(b+1,b+n+1);
	if(n % 2 == 0) {
		b[0] = a[1] < a[2] ? a[1] : a[2];

		for(int i = n / 2 + 1; i <= n; i++) {
			b[i + 1] = b[i];
		}

		b[n / 2 +1] = a[0];
		b[n + 2] = a[1] > a[2] ? a[1] : a[2];

		for(int i = 1; i < n + 3; i++) {
			ans += abs(b[i] - b[i-1]);
		}

		cout << ans << endl;
	} else {

		b[0] = a[1] < a[2] ? a[1] : a[2];

		for(int i = n/2 + 1; i <= n ;i++) {
			b[i + 1] = b[i];
		}

		b[n / 2 + 1] = a[0];
		b[n + 2] = a[1] > a[2] ? a[1] : a[2];

		for(int i = 1; i < n+3;i++) {
			ans += abs(b[i] - b[i-1]);
		}

		int ans1 = 0;
		sort(c + 1, c + 1 + n);
		c[0] = a[1] < a[2] ? a[1] : a[2];

		for(int i = n/2 + 2; i <= n; i++) {
			c[i+1] = c[i];
		}
		
		c[n / 2 + 2] = a[0];
		c[n + 2] = a[1] > a[2] ? a[1] : a[2];

		for(int i = 1; i < n + 3; i++) {
			ans1 += abs(c[i] - c[i - 1]);
		}
		
		cout << min(ans1, ans) << endl;
	}
	return 0;
}
