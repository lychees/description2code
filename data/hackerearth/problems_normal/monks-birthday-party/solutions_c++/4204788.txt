#include <iostream>
#include <set>
#include <string>
using namespace std;

int main()
{
    int t;
    cin >> t;
    int i,j;
    for(i=0;i<t;i++)
    {
        set <string> s;
        int n;
        cin >> n;
        for(j=0;j<n;j++)
        {
            string str;
            cin >> str;
            s.insert(str);
        }
        set <string>::iterator it;
        for(it=s.begin();it!=s.end();it++)
        {
            cout<<*it<<endl;
        }
    }
}
