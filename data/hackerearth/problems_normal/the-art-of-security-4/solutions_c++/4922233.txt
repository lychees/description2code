#include <iostream>

using namespace std;

int find_permutations(int nums[], int ans[], bool visited[], int n, int pos)
{
	int count = 0;
	long long int i;
	if(pos == n)
	{
		// for(i = 0; i < n ; i++)
			// cout << ans[i] << " ";
		// cout << endl;
		int j, sum ;

		for(i = 0; i < (1 << n); i++)
		{
			sum = 0;
			for(j = 0; j < n; j++)
			{
				if(i & 1 << j)
					sum += ans[n - j - 1];
 				else
					sum -= ans[n - j - 1];
				if(sum < 0)
					break;
			}
			if(sum >= 0)
				count++;
		}
		// cout << count << endl;
		return count;
	}

	for(i = 0; i < n; i++)
	{
		if(!visited[i])
		{
			ans[pos] = nums[i];
			visited[i] = true;
			count += find_permutations(nums, ans, visited, n, pos+1);
			visited[i] = false;
		}
	}
//	cout << count << endl;
	return count;
}

int main()
{
	int i, j, n, sum = 0;
	cin >> n;
	int nums[n];
	int ans[n];
	bool visited[n];
	for(i = 0; i < n; i++)
	{
		cin >> nums[i];
		sum += nums[i];
		visited[i] = false;
		ans[i] = -1;
	}
	int temp = find_permutations(nums, ans, visited, n, 0);
	cout << temp << " " << sum << endl;
	if(temp <= sum)
		cout << "Got no way out!" << endl;
	else
		cout << "We will win!" << endl;
	return 0;
}
