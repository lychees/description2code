L = raw_input()
N = raw_input()
L = int(L)
for i in range(int(N)):
    S = raw_input()
    Size = S.split()
    Size = map(int,Size)
    if (Size[0] < L) or (Size[1] < L):
        print "UPLOAD ANOTHER"
    elif (Size[0] >= L and Size[1] >= L) and (Size[0] == Size[1]):
        print "ACCEPTED"
    else:
        print "CROP IT"
