inp1 = raw_input()
arr = {}
arr2 = []
for i in range(int(inp1)):
    inp2 = raw_input()
    if inp2 not in arr:
        arr[inp2] = 1
    else:
        arr[inp2] = arr[inp2] + 1
for key, value in arr.iteritems():
    temp = [key, value]
    arr2.append(temp)
arr2.sort()
for i in arr2:
    print str(i[0])+' '+str(i[1])
