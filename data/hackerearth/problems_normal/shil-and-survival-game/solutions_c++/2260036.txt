#include<stdio.h>
#include<string.h>
#include<stdlib.h>
#include<cstring>
#include<iostream>
using namespace std;
int max_left(int *a,int index)
{
    int max=a[index+1];
    for(int j=0;j<=index;j++)
        if(a[j]>max)
        {
        max=a[j];
        break;
        }
    return max;
}
int max_right(int *a,int index,int n)
{
    int max=a[index-1];
    for(int j=index;j<n;j++)
       	if(a[j]>max)
       	{
        max=a[j];
        break;
       	}
    return max;
}
int main()
{
    int n;
    cin>>n;
    int i,left_max,right_max,a[n];
    for(i=0;i<n;i++)
        cin>>a[i];
        cout<<"1"<<" ";
        for(i=1;i<n-1;i++)
        {
            left_max=max_left(a,i-1);
            right_max=max_right(a,i+1,n);
            if(a[i]>=left_max || a[i]>=right_max)
                cout<<(i+1)<<" ";
        }
    cout<<n;
    return 0;
}
