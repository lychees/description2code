#include <iostream>
#include<string.h>
using namespace std;

int main()
{
   int t;
   
   cin>>t;
   while(t--)
   {
   	int i,j,hash[26]={0},flag=0;
   	char a[100000],b[100000];
   	cin>>a;
 	cin>>b;
 	for(i=0;i<strlen(a);i++)
 	{
 		hash[a[i]-97]++;
 	}
 	for(j=0;j<strlen(b);j++)
 	{
 		if(hash[b[j]-97]!=0){
 		cout<<"YES"<<endl;
 		flag=1;
 		break;
 	}
 	}
     if(flag!=1)
      cout<<"NO"<<endl;
   }
   return 0;
}
