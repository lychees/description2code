#include<bits/stdc++.h>
using namespace std;
#define ll int
bool check_vowel(char ch)
{
    if((ch=='a')||(ch=='e')||(ch=='i')||(ch=='o')||(ch=='u')||(ch=='y'))
        return true;
    return false;
}
int main()
{
    ll test,i,j,num,minn,maxx,m,a,b;
    cin>>test;
    while(test--)
    {
        string str;
        cin>>str;
        ll len=str.length();
        if(len==1)
        {
            if(check_vowel(str[0])==true)
                cout<<"easy"<<endl;
            else
                cout<<"hard"<<endl;
        }
        else if(len==2)
        {
            if((check_vowel(str[0])==true) || (check_vowel(str[1])==true))
                cout<<"easy"<<endl;
            else
                cout<<"hard"<<endl;
        }
        else
        {
            ll cons=0,vowel=0;
            bool flag=false;
            if(check_vowel(str[0])==true)
                vowel++;
            else
                cons++;
            if(check_vowel(str[1])==true)
                vowel++;
            else
                cons++;
            for(i=2;i<len;i++)
            {
                if(check_vowel(str[i])==true)
                    vowel++;
                else
                    cons++;
                if( (check_vowel(str[i])==false) && (check_vowel(str[i-1])==false) && (check_vowel(str[i-2])==false) )
                {
                    flag=true;
                    break;
                }
            }
            if(cons>vowel)
                flag=true;
            if(flag==true)
                cout<<"hard"<<endl;
            else
                cout<<"easy"<<endl;
        }
    }
}
