n = int(raw_input())
i=0
result = []
while i < n:
    pattern=raw_input()
    text=raw_input()
    reversetext =text[::-1]
    if pattern in text or pattern in reversetext:
        result.append('YES')
    else:
        result.append('NO')
    i=i+1
for te in result:
    print te
