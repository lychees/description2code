#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int a[1000001]={0};
vector<int> b(1000001,1);

int main()
{

    for(int i=2;i*i<=1000000;i++)
    {
    	if(a[i]!=0)
    	continue;
    //	b[i]=b[i]+1;
    	for(int j=2*i;j<=1000000;j=j+i)
    	{
    		if(a[j]==0)
    		{
    			a[j]=i;b[i]=b[i]+1;
    		}

    	}
    }
    int t;
    cin>>t;
    while(t--)
    {
    	int x;
    	cin>>x;
    	cout<<b[x]<<endl;
    }

    return 0;
}
