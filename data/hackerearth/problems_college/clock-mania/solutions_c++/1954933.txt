#include<iostream>
#include<vector>
#include<math.h>

using namespace std;

int main()
{
	int t;
	cin>>t;
	for(int i=0;i<t;i++)
	{
		int h,m;
		cin>>h>>m;
		h*=30;
		h=h+(m/2);
		m*=6;
		double ans;
		while(h>360)h-=360;
		while(m>360)m-=360;
		if(h>m)ans=h-m;
		else ans=m-h;
		if(ans>180)ans=360-ans;
		cout<<ans<<endl;
	}
}
