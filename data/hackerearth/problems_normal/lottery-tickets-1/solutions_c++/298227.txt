int main()
{
    int t;
    cin >> t;
    while(t--)
    {
        int n; 
        char str[100];
        int S[100],T[100],P[100],f[100]; 
        cin >> n;
        int i,j; 
        for (i=0;i<n;i++)
            scanf("%d%d%d",&S[i],&T[i],&P[i]); 
        for (i=0;i<n;i++) 
            for (j=i+1;j<n;j++) 
                if (T[j]<T[i]) 
                { 
                    swap(S[i],S[j]); 
                    swap(T[i],T[j]); 
                    swap(P[i],P[j]); 
                } 
        int result=0; 
        for (i=0;i<n;i++) 
        { 
            f[i]=P[i]; 
            for (j=0;j<i;j++) 
                if (S[i]>=T[j] && f[j]+P[i]>f[i]) 
                    f[i]=f[j]+P[i]; 
            if (f[i]>result) 
                result=f[i]; 
        } 
        printf("%.2lf\n",double(result)/100.0); 
    } 
    return 0;
}
