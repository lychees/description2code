#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int a[]={0,1,1,2,2};
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        long long int s=0;
        cin>>n;
        vector<long long int> v;
        for(int i=0;i<n;i++)
        {
            int temp;
            cin>>temp;
            v.push_back(temp);
        }
        sort(v.begin(),v.end());
        int temp=v[0];
        long long int s1=0;
        for(int i=1;i<n;i++)
        {
            v[i]+=s1;
            if(v[i]!=temp)
            {
                int dif=v[i]-temp;
                s1+=dif;
                s=s+dif/5+a[dif%5];
                temp=v[i];
            }
        }
        cout<<s<<endl;
    }
    return 0;
}
