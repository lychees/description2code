#include <iostream>
#include<assert.h>
#include<vector>
using namespace std;

int main()
{
    
    int i,j,k,l,n;
    cin>>n;
    assert(1<=n && n<=1000);
    vector<int>v(n);
    for(int i=0;i<n;i++)
    {cin>>v[i];
      assert(1<=v[i]<=1000);
    }
    for(i=0;i<n;i++)
    {
      int tmp=0;
      for(j=0;j<n;j++)
        if(v[j]>v[i])tmp++;
      cout<<tmp+1;
      if(i<n-1)cout<<" ";
    } 
    return 0;
}
