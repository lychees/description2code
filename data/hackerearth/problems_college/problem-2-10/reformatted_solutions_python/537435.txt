n, m = raw_input().split()
n = int(n)
m = int(m)
p = min(m+1, n+1)
b = False
l = 1
if (not ((1 < n < 10**9) and (1 < m < 10**9))):
    b = True
while (not b and p > 1):
    p = p-1
    if (m % p == 0 and n % p == 0):
        l = p
        b = True

num_box = l
print n/num_box, m/num_box
