#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t;
    cin >> t;
    assert(t>=1&&t<=20);
    while(t--)
    {
        int ma5=-1,ma3=-1;
        int x;
        cin >> x;
        assert(x>=1&&x<=100000);
        int i;
        for(i=0;i<=x;i+=3)
        {
            if((x-i)%5==0)
            {
                ma3=i;ma5=x-i;
            }
        }
        if(ma5==-1||ma3==-1) cout << "-1" << endl;
        else{
            for(i=0;i<ma3;i++) cout << '5';
            for(i=0;i<ma5;i++) cout << '3';
            cout << endl;
        }
    }
}
