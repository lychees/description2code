#include<bits/stdc++.h>
using namespace std;
#define lli long long int
int main()
{
	lli t;
	cin>>t;
	while(t--)
	{
		lli n,k;
		cin>>n;
		lli arr[n];
		lli i;
		for(i=0;i<n;++i)
		{
			cin>>arr[i];
		}
		cin>>k;
		lli brr[k];
		for(i=0;i<k;++i)
		brr[i]=0;
	  for(i=0;i<n;++i)
	  {
	  	if(arr[i]<k)
	  	brr[arr[i]]++;
	  }
		lli ans=0;
		for(i=1;i<=k/2;++i)
		{
           if(i*2==k && brr[i]>1)
           	ans+=(brr[i]*(brr[i]-1))/2;
           else
           	ans+=brr[i]*brr[k-i];
		}
		cout<<ans<<endl;
	}
	return 0;

}
