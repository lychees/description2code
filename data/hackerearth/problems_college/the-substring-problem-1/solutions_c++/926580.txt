#include <bits/stdc++.h>
int main()
{  
	int test,N,i,j,k,hash[26],max,great,flag[26];
	char a[104];
    scanf("%d",&test);
    while(test--)
      {
      	for(i=0;i<=26;i++)
      	hash[i]=0;
      	//memset(hash,0,sizeof(hash));
      	scanf("%d",&N);
      	for(i=0;i<N;i++)
      	{ 
      		for(k=0;k<=26;k++)
      		flag[k]=0;
      		scanf("%s",a);
      	  	//memset(flag,0,sizeof(flag));
      	  	for(j=0;a[j]!='\0';j++)
      	  	{  

      	  	 	k=a[j]-'a';
      	  	 	
      	  	 	if(flag[k]==0)
      	  	 	  {
      	  	 	  	hash[k]++;
      	  	 	  	flag[k]++;
      	  	 	  }
      	  	 }
      	}
      	max=hash[0];
      	great=0;
      	for(i=1;i<26;i++)
      	{
      	  if(max<hash[i])
      	  {
      	  	  max=hash[i];
      	  	  great=i;
      	  }
      	 }  
      	printf("%c\n",great+'a');
      }
    return 0;
}
