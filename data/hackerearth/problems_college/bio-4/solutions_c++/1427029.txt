#include<iostream>
#include<string>
#include<algorithm>
using namespace std;
int main(){
	   int t;
	   cin >> t;
	   while(t--)
	   {
	   	 int i,j;
   		string a,b;
   		cin >> a >> b;
   		int n = a.size();
   		int m = b.size();
   		int mn = 1e9;
   		for(i = 0 ; i <= n-m; ++i ){
   			int ct = 0;
		   	   for(j = 0 ; j < m ; ++j )
		   	   {
   	   			   if(a[i+j] == b[j] )
   	   			   ct++;
   	   		}
   	   		 mn = min( mn , m-ct);
   	   	   }
   		   cout << mn << endl;
   		
   	}
	 
	
}
