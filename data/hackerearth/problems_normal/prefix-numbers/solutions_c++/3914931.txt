using namespace std;
#define long long long
#define uint unsigned int

int rint() { int ret; scanf("%d", &ret); return ret; }
long rlong() { long ret; scanf("%lld", &ret); return ret; }

const int SIZE = 252;
uint tmp[SIZE][SIZE], powr[SIZE][SIZE], ret[SIZE];
void MatPow2()
{
	for (int i = 0; i < SIZE; i++)
		for (int j = 0; j < SIZE; j++)
		{
			tmp[i][j] = 0;
			for (int k = 0; k < SIZE; k++)
				tmp[i][j] = tmp[i][j] + powr[i][k] * powr[k][j];
		}
	for (int i = 0; i < SIZE; i++)
		for (int j = 0; j < SIZE; j++)
			powr[i][j] = tmp[i][j];
}

void MatPow(long p)
{
	while (p > 0)
	{
		if (p & 1)
		{
			for (int i = 0; i < SIZE; i++)
			{
				tmp[0][i] = 0;
				for (int j = 0; j < SIZE; j++)
					tmp[0][i] += powr[j][i] * ret[j];
			}
			for (int i = 0; i < SIZE; i++)
				ret[i] = tmp[0][i];
		}
		MatPow2();
		p >>= 1;
	}
}

int main()
{
#ifdef _DEBUG
	freopen("..\\input.txt", "r", stdin);
#endif

	for (int i = 0; i < SIZE; i++)
	{
		ret[i] = 0;
		for (int j = 0; j < SIZE; j++)
			powr[i][j] = 0;
		for (int d = 1; d < 10; d++)
			if (i * 10 % d == 0)
				powr[i][(i * 10 + d) % SIZE]++;
	}
	ret[0] = 1;

	long n = rlong();
	MatPow(n);
	uint ans = 0;
	for (int i = 0; i < SIZE; i++)
		ans += ret[i];
	printf("%u\n", ans);

	return 0;
}
