#include<iostream>
#include<vector>
#include<queue>
#include<algorithm>
using namespace std;
priority_queue<int >max_p;
priority_queue<int,vector<int>,greater<int> > min_p;
void sol()
{
    int cnt=100000;
    int n,p,q,i,j,arr[10004];
    cin>>n;
    for(i=0;i<n;i++)
        cin>>arr[i];
    sort(arr,arr+n);
    for(i=n-1;i>=0;i--)
    {
        int f=0;
        for(j=0;j<i;j++)
        {
            if(arr[i]<=arr[j]*3)
            {
                f=1;
                break;
            }
        }
        if(f==1) cnt=min(cnt,j+n-i-1);;
    }
    cout<<cnt<<endl;
}
int main()
{
    sol();
    return 0;
}
