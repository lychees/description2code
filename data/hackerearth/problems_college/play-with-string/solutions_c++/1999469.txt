#include <iostream>
using namespace std;

int main()
{
	             int t,c1,c2;
	             cin>>t;
	             while(t--)
	             {
                  string str;
                  cin>>str;
                  int len,i,j;
                  len=str.length();
                  int f=-1;
                  for(i=1,j=len-1;i<len,j>0;i++,j--)
                  {
                  	 c1=abs((int)str[i]-(int)str[i-1]);
                  	 c2=abs((int)str[j]-(int)str[j-1]);
                  	 if(c1==c2)
                  	 { 
                  	 	f=1;
                  	 }
                  	 else
                  	 {
                  	 	 f=0;
                  	 	 break;
                  	 }
                  }
                  if(f==1)
                  cout<<"Funny"<<endl;
                  else
                  cout<<"Not Funny"<<endl;
	            }
                  
                  
    return 0;
}
