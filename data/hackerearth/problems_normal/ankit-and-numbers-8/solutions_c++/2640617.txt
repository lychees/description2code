#include<math.h>

using namespace std;


int main(){
	
	
	int t;
	cin >> t;
	
	long long int n,num=1,result;
	fori(t,i){
		num=1;
		cin >> n;
		for(int j=0;j<n-1;j++){
			num=num << 1;
		}
		result=num*(n*(n+1))/2;
		cout << result << endl;
	}
	
	
	return 0;	
}
