#include <iostream>
#include <cstdio>
#include <cstring>
#include <cstdlib>

using namespace std;

int main()
{
	int ls;
	int mx;
	char s[1001];
	int t;
	int p[1001][1001];
	int n;
	int a[100001];
	memset(a,0,sizeof(a));
	for (int i = 2; i <= 10000; i++) {
		int j = 2;
		while (i*j <= 10000) {
			a[i*j] = 1;
			j++;
		}
	}
//	memset(p,0,sizeof(p[0][0])*1000*1000);
	scanf("%d",&t);
	for (int k = 0; k < t; k++) {
		scanf("%s",s);
	//	rd(s);
	//	gets(s);
		ls = strlen(s);
		mx = 1;

		for (int i = 1; i < ls; i++) {
			int l = i-1;
			int h = i;
			while (l >= 0 && h < ls && s[l] == s[h]) {
				if (h-l + 1 > mx)
					mx = h-l + 1;
				l--;
				h++;
			}
			l = i - 1;
			h = i + 1;
			while (l >= 0 && h < ls && s[l] == s[h]) {
				if (h - l + 1 > mx)
					mx = h - l + 1;
				l--;
				h++;
			}
		}
		if (a[mx] == 0 && mx != 1)
			printf("YES\n");
		else
			printf("NO\n");
	}
	
}
