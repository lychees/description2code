#include <iostream>

using namespace std;

bool isPalindrome(const string & S)
{
	for (int i = 0, j = S.size() - 1; i < j; ++i, --j) {
		if (S[i] != S[j]) {
			return false;
		}
	}
	
	return true;
}


int main()
{
	string S;
	cin >> S;
	
    if (isPalindrome(S)) {
    	cout << "YES";
    } else {
    	cout << "NO";
    }
    return 0;
}
