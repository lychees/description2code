#include <iostream>
#include <math.h>
#define PI 3.14
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int i=5,n,k=0;
    	cin>>n;
    	int64_t re=floor((log(2*PI*n)/2 + n*(log(n) - 1) + 1/(12*n))/log(10)) +1;
    	while(n/i)
    	{
    		k+=n/i;
    		i*=5;
    	}
    	//k=findTrailingZeros(n);
    	cout<<(re-k)<<endl;
    }
    return 0;
}
