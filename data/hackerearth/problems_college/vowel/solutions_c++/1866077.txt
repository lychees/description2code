#include<bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
	int t;
	ios::sync_with_stdio(0);
	string str;
	cin>>t;
	while(t--)
	{
		string s;
		cin>>str;
		for(int i=0;i<str.length();i++)
		{
			if(str[i]=='A'||str[i]=='a'||str[i]=='E'||str[i]=='e'||str[i]=='i'||str[i]=='I'||str[i]=='O'||str[i]=='o'||str[i]=='U'||str[i]=='u')
			s+=str[i];
		}
		if(!s.empty())
		cout<<s<<endl;
		else 
		cout<<"No"<<endl;
	}
	return 0;
}
