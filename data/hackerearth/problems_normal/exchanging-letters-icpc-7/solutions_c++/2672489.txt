#include <bits/stdc++.h>

using namespace std;

#define pb push_back
#define mp make_pair
#define REP(i, n) for (int i = 0; i < (int)(n); ++i)
typedef long long LL;
typedef pair<int, int> PII;

int n, k, t;

int c[2222][2222] = {};
const int MOD = 1e9 + 7;

int main() {
    c[0][0] = 1;
    for (int i = 1; i < 2222; ++i) {
        c[i][0] = 1;
        for (int j = 1; j < 2222; ++j) {
            c[i][j] = c[i - 1][j] + c[i - 1][j - 1];
            if (c[i][j] >= MOD) c[i][j] -= MOD;
        }
    }
    scanf("%d%d%d", &n, &k, &t);
    REP(i, k) {
        int x;
        scanf("%d", &x);
        t -= x;
    }
    t = abs(t);
    if (t % 2 != k % 2 || t > k) {
        printf("0\n");
        return 0;
    }
    int eq = (k - t) / 2;
    printf("%d\n", c[k][eq]);
    return 0;
}
