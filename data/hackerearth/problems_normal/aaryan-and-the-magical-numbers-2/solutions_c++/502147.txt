#include<iostream>
using namespace std;
long long counte[22]={0};
long long powe[22]={0};

long long current[22]={0};
#include<bits/stdc++.h>
int hashe[10]={0};
long long eval(long long n,int d)
  {//cout<<n<<" "<<d<<endl;
  	if(d==0 || n==0) return 0;
  	if(d==1)
  	  {long long ctr=0;
		for(int i=1;i<=n;i++)
  	      if(hashe[i])
  	        ctr++;
  	    return ctr;
  	  }
  	long long ten=powe[d-1];
  	//cout<<ten<<endl;
  	int first=n/ten;
  	long long rest=n%ten;
  	long long ctr=0;
	if(first)
	    ctr=counte[d-1];
  	for(int i=1;i<first;i++)
  	  {
  	  	if(hashe[i])
  	  	  ctr+=ten;
  	  	else
  	  	  ctr+=counte[d-1];
  	  }
  	if(hashe[first])
  	  ctr+=rest+1;
  	else
  	  ctr+=eval(rest,d-1);
  	return ctr;
  }
int main()
 {
