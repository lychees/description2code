#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    long double T,N,M;
    double val,val1,val2,val3;
    cin>>T;
    for(int i=0;i<T;i++)
    {
    	cin>>N>>M;
    	if(N==1)
    	{
    		cout<<0<<endl;
    	}
    	else if(N<M)
    		{
    			val=ceil(log10(N)/log10(2)) + 1;
    			cout<<val<<endl;
    		}
    	else
    		{
    			val=ceil(log10(M)/log10(2));
    			val1=pow(2,val);
    			val2=N-val1;
    			
    			if(val2<=0)
    				cout<<val+1<<endl;
    			else
    			{
    				val3=ceil(double(val2/M));
    				//cout<<val3;
    				cout<<val+val3+1<<endl;
    			}
    			
    		}
    	
    	
    }
    return 0;
}
