#include<iostream>
using namespace std;
int main()
{
int n,min=-1;
cin>>n;
int a[n],sum=0,i;
for(i=0;i<n;i++)
{
cin>>a[i];
sum+=a[i];
}
if (sum%2==0)
{
for(i=0;i<n;i++)
{
if ((min==-1)&&(a[i]%2==1))
min=a[i];
if (min>a[i])
min=a[i];
}
}
if (sum%2==1) cout<<sum<<endl;
else if(min==-1) cout<<":(\n";
else cout<<sum-min;
return 0;
}
