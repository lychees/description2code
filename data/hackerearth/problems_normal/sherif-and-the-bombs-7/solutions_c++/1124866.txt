#include <iostream>
#include <cstdio>

#define N 150

using namespace std;

struct Point{
	int x,y,r;
};

Point p[N];

int main(){
	
	int n;
	scanf("%d", &n);
		
		for(int i = 0; i < n; i++) scanf("%d %d %d", &p[i].x, &p[i].y, &p[i].r);
		
		int ct = 0;
		for(int i = 1; i <= 1000; i++)
			for(int j = 1; j <= 1000; j++){
				
				int cont = 0;
				for(int k = 0; k < n; k++){
					
					if((i - p[k].x) * (i - p[k].x) + (j - p[k].y) * (j - p[k].y) <= p[k].r * p[k].r) cont++;
					
				}
				
				if(cont >= 2) ct++;
			}
		
		printf("%d\n", ct);	
		
	
	return 0;
}
