using namespace std;
int a[205];
int visited[205];

void dfs(int u) {
    if(!visited[a[u]]) {
        visited[a[u]] = 1;
        dfs(a[u]);
    }
}
int mark[205];
int main() {
    for(int i = 0;i <= 200;i++) {
        int c = 0;
        int p = i * i;
        while(p > 0) {
            c+=p % 10;
            p/=10;
        }
        a[i] = c;
    }
    for(int i = 0;i <= 200;i++) {
        for(int j = 0;j <= 200;j++) {
            visited[j] = 0;
        }
        visited[i] = 1;
        dfs(i);
        if(visited[1] || visited[4]) mark[i] = 1;
        else mark[i] = 0;
    }
    int t; cin>>t;
    for(int i = 1;i <= t;i++) {
        long long c = 0;
        long long n = i; scanf("%lld",&n);
        if(n % 3 == 0) {
            printf("NO\n");
            continue;
        }
        if(n == 4) {
            printf("YES\n");
            continue;
        }
        n*=n;
        while(n > 0) {
            c+=n%10;
            n/=10;
        }
        if(mark[c]) printf("YES\n");
        else printf("NO\n");
    }
    
    
    
}
