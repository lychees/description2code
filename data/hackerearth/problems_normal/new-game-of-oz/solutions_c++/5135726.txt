#include <bits/stdc++.h>
using namespace std;
int a[1005],dp[1005];
int main()
{
   	int t;
   	scanf("%d",&t);
   	while(t--)
   	{
   		int n;
   		scanf("%d",&n);
   		int i;
   		for(i=1;i<=n;i++)
   		scanf("%d",&a[i]);
   		sort(a+1,a+n+1);
   		dp[1] = 1;
   		for(i=2;i<=n;i++)
   		{
   			dp[i] = dp[i-1];
   			if(a[i]==a[i-1] + 1)
   			dp[i] = max(dp[i-1],dp[i-2]+1);
   			else dp[i] = dp[i-1] + 1;
   			
   		}
   		printf("%d\n",dp[n]);
   	}
    return 0;
}
