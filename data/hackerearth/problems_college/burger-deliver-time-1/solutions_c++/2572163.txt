#include<iostream>
#include<algorithm>
using namespace std;

int main()
{
		int t, d, tc;
		cin>>tc;
		pair<int,int>a[tc];
		for(int i=0; i<tc; i++)
		{
				cin >> t;
				cin >> d;
				a[i] = make_pair(t+d, i+1);
		}

		sort(a, a+tc);
		for(int i=0; i<tc; i++)
		{
				cout<<a[i].second<<" ";
		}
		cout<<endl;
}
