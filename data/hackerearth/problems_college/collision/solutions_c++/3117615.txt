#include <iostream>
using namespace std;

int main()
{
	int t,n,b[10],c,k,i;
    scanf("%d",&t);
    while( t>0)
    {c=0;
    	cin>>n;
    	int a[n];
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	for(i=0;i<10;i++)
    	{
    		b[i]=0;
    	}
    	for(i=0;i<n;i++)
    	{
    		k=a[i]%10;
    		if(b[k]==0)
    		b[k]=1;
    		else 
    		c++;
    	}
    	cout<<c<<endl;
    	t--;
    }
    return 0;
}
