#include <bits/stdc++.h>
using namespace std;
int main()
{
    ios_base::sync_with_stdio(false);
    int n,k,t;
    vector<int>v;
    cin>>t;
    while(t--){
    cin>>n>>k;
    int num;
    for(int i=0;i<n;i++){
        cin>>num;
        v.push_back(num);
    }
    int index=0;
    for(int i=0;i<n;i++)
    {
        int minn=i;
        for(int j=i+1; j<=i+k && j<n;j++)
        {
          // if(j>i+k)
          // break;
            if(v[j]<=v[minn])
            {
                minn=j;

            }

        }
        //cout<<minn<<endl;
        for(int p=minn;p>i;p--)
        {
            swap(v[p],v[p-1]);
        }

        k-=(minn-i);
       // cout<<index<<endl;
        if(k<=0)
            break;
    }
    for(int i=0;i<n;i++)
    cout<<v[i]<<" ";
  cout<<endl;
  v.clear();
    }
	return 0;
}
