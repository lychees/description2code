#include <bits/stdc++.h>
using namespace std;
#define MAX 100000001 
long int a[MAX]={0};
int main()
{
    int t;
    scanf("%d",&t);
    long int b[101],i,j;
    for(i=1;i<101;i++)
    {
    	b[i]=i*i*i;
    }
    for(i=1;i<101;i++)
    {
    	for(j=i+1;j<101;j++)
    	{
    		a[b[i]+b[j]]++;
    	}
    }
    while(t--)
    {
    	int n,count=0;
    	i=0;
    	scanf("%d",&n);
    	while(count<n)
    	{
    		i++;
    		if(a[i]==2)
    		{
    			count++;
    		}
    	}
   	cout<<i<<endl;
    }
}
