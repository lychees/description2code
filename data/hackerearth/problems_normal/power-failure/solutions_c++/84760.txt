void sieve()
{
    ill i, j, k;
    for(i=1;i<MAX;i++)
        primes[i]=1;
    for(i=4;i<MAX;i+=2)
        primes[i]=0;
    primes[1]=0;
    for(i=3; i<LMT; i+=2)
    {
        if(!primes[i])
            continue;
        for(j=i*i, k=i<<1; j<MAX; j+=k)
            primes[j]=0;
    }
    return;
}
#define MOD 1000000007
int main()
{
    int t,m,n,i,j,k,l;
    cin(t);
    int dp[100009];
    sieve();
    for(i=1;i<=100000;i++)
        dp[i]=0;
    for(i=1;i<=100000;i++)
        dp[i]=dp[i-1]+primes[i];
    int a[100009];
    while(t--)
    {
        cin(n);cin(m);
        for(i=1;i<=n;i++)
            cin(a[i]);
        sort(a+1,a+n+1);
        ill ans=1LL;
        for(i=n;i>=1;i--)
        {
            ans=(ans*(m-(n-i)-a[i]+1-(dp[m]-dp[a[i]-1])))%MOD;
            //cout<<dp[m]<<" "<<dp[a[i]]<<"\n";
        }
        cout<<ans<<"\n";
    }
    return 0;
}
