#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,j,n,k=-1,ans=0;
	cin>>n;
	map<string,int >mp;
	while(n--)
	{
		string s;
		cin>>s;
		mp[s]++;
		if(mp[s]>k)
		k=mp[s];
	}
	map<string,int > ::iterator it;
	for(it=mp.begin();it!=mp.end();it++)
	{
		if(it->second==k)
		ans+=k;
	}
	cout<<ans<<endl;
}
