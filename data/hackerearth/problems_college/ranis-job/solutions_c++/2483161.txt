#include <iostream>
using namespace std;

int main()
{
      int t,n,i,j,l,temp;
  cin>>t;
  for(l=0;l<t;l++)
  {
      cin>>n;
      int a[n],b[n],tot=0,count=0;
      for(i=0;i<n;i++)
      {
          cin>>a[i];
          tot=tot+a[i];
      }
      for(i=0;i<n;i++)
      cin>>b[i];
      for(i=0;i<n;i++)
      for(j=0;j<n;j++)
          if(b[i]>b[j])
              {
                  temp = b[i];
                  b[i] = b[j];
                  b[j] = temp;
              }
      while(tot>0)
      {
          tot=tot-b[count];
          count++;
      }
      cout<<count<<endl;
  }
  return 0;
}
