#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(i=0;i<n;i++)
#define ll long long
#define elif else if
#define pii pair<int,int>
#define mp make_pair
#define pb push_back
#define CLEAR(array, value) memset(ptr, value, sizeof(array));
#define si(a)     scanf("%d", &a)
#define sl(a)     scanf("%lld", &a)
#define pi(a)     printf("%d", a)
#define pl(a)     printf("%lld", a)
#define pn        printf("\n")
#define SZ(x)  (int)((x).size())

int main()
{
	ios_base::sync_with_stdio(false);
	ll k,sum=0,i=0,temp,count=0,max=0;
	cin>>k;
	vector<ll> v;
	rep(i,k)
	{
		cin>>temp;
		v.pb(temp);
	}
	sort(v.begin(),v.end());
	reverse(v.begin(),v.end());
	for(int i=0;i<k;i++)
	{
	count++;
	sum=count+v[i];
	if(sum>max)
		max=sum;
	}
	cout<<max+1<<endl;
	
}
