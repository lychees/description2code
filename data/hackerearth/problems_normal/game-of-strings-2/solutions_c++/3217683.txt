
#include<bits/stdc++.h>
#include<cstring>
//#define DEBUG
using namespace std;
 
int a1[1000000];
char  str[1000001];
void computeLPS(char *s,int *lps)
{
    lps[0]=0;
    if(s[0]==s[1])
        lps[1]=1;
    else
        lps[1]=0;
    int i,curLPS=lps[1];
    for(i=2;s[i]!='\0';i++)
    {
   
        if(s[i]==s[curLPS])
        {
            curLPS++;
            lps[i]=curLPS;
        }
        else
        {
            if(curLPS>0)
            {
                curLPS=a1[curLPS-1];
                i--;
            }
            else
            {
                lps[i]=0;
            }
        }
    }
}
int main()
{
   int T;
   cin>>T;
   assert(1<=T && T<=10);
   while(T--)
   {
       
       
        int K,n,i;
        cin>>str>>K;
        n=strlen(str);
        assert(1<=n && n<=1000000 && 1<=K && K<=n);
        computeLPS(str,a1);
    
        int l=a1[n-1],len=n;
        int valid[1000001]={0};
        valid[l]=1;
        while(l!=0 && len>1)
        {
           
           
            if(len%(len-l)==0)
            {
        
                for(int j=len-l;j<=len;j+=(len-l))
                {
          
                    valid[j]=1;
                }
                len=len-l;
                l=a1[len-1];
                
            }
            else
            {
                len=l;
                l=a1[len-1];
                valid[len]=1;
           
            }
        
        
        }
        int m=0;
        for(i=1;i<K;i++)
        {
            len=a1[i];
            if(valid[len])
                m=max(m,len);
        }
   
        if(m==0)
        {
            
            cout<<"Puchi is a cheat!";
        }
        else
        for(i=0;i<m;i++)
            cout<<str[i];
        cout<<endl;
   }
   
}
