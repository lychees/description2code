#include <bits/stdc++.h>
using namespace std;

int main() {
	int n, m, q, memory, search, location, first;
	map <int, int> mind_palace;

	scanf("%d %d", &n, &m);
	if(n==0 && m==0)
		return 0;
	scanf("%d", &first);
	for (int i = 1; i < n*m; ++i) {
		scanf("%d", &memory);
		mind_palace[memory] = i;
	}

	scanf("%d", &q);
	for (int i = 0; i < q; ++i)	{
		scanf("%d", &search);
		if(location = mind_palace[search]){
			printf("%d %d\n", location/m, location%m);
		}
		else if(search == first){
			printf("0 0\n");
		}
		else{
			printf("-1 -1\n");
		}
	}
	return 0;
}
