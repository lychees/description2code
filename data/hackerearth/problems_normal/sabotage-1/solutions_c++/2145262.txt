#define FNAME ""

const int MAXN = 1e5 + 5;

int v[MAXN], u[MAXN], pr[MAXN], r[MAXN], used[MAXN];

int get(int v)
{
    if (v == pr[v])
        return v;
    return pr[v] = get(pr[v]);
}

int uni(int a, int b)
{
    a = get(a), b = get(b);
    if (a == b)
        return 0;
    if (r[a] < r[b])
        swap(a, b);
    pr[b] = a;
    if (r[a] == r[b])
        r[a]++;
    return 1;
}

int main() 
{
#ifdef LOCAL    
    freopen(FNAME".in", "r", stdin);
    freopen(FNAME".out", "w", stdout); 
#endif    
    int n, m;
    scanf("%d%d", &n, &m);
    forn (i, n)
        pr[i] = i, r[i] = 0;
    forn (i, m)
        scanf("%d%d", &v[i], &u[i]), v[i]--, u[i]--;
    fornr (i, m)
    {
        if (uni(v[i], u[i]))
            used[i] = 1;
    }
    int cnt = 0;
    forn (i, m)
        if (!used[i])
            cnt++;
    printf("%d\n", cnt);
    forn (i, m)
        if (!used[i])
            printf("%d\n", i + 1);
    return 0;
}
