#include<iostream>
#include<string>
using namespace std;
int main()
{
	int t,p,i;
	string s;
    scanf("%d",&t);
    cin.ignore();
    while(t--)
    {
    	getline(cin,s);
    	p=s.length();
    	int count=0;
    	for(i=0;i<p;i++)
    	{
    		if(s[i]>=97 && s[i]<=122)
    		{
    			count++;
    		}
    		if(s[i]=='*') {count++;break;}
    	}
    	int sum=0;
    	//cout<<count<<" ";
    	for(i=p-1;i>=0;i--)
    	{
    		if(s[i]>=97 && s[i]<=122)
    		{
    			sum++;
    		}
    		if(s[i]=='*') {sum++;}
    		if(count==sum) break;
    	}
    	cout<<s[i]<<endl;
    }
    return 0;
}
