#include<bits/stdc++.h>
using namespace std;

int main()
{
    int arr[30];
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n;
        cin>>n;
        int sum = 0;
        for(int i=0; i<n; i++)
        {
            scanf("%d", &arr[i]);
            sum += arr[i];
        }

        bool dp[sum+1];
        bool temp[sum+1];

        for(int i=0; i<=sum; i++)
        {
            temp[i] = false;
            dp[i] = false;
        }

        dp[0] = true;
        for(int i=0; i<n; i++)
        {
            for(int j=0; j<=sum; j++)
            {
                if(dp[j] == true)
                    temp[j + arr[i]] = true;
            }

            for(int j=0; j<=sum; j++)
                if(temp[j] == true)
                {
                    temp[j] = false;
                    dp[j] = true;
                }

            dp[arr[i]] = true;
        }

        int ctr = 0;
        for(int i=0; i<=sum; i++)
            if(dp[i] == true)
                ctr++;
        cout<<ctr<<endl;

    }
    return 0;
}
