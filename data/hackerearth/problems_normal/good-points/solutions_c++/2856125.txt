#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    if (n != 1 && n != 2 && n != 4) {
        cout<<"NO"<<endl;
    }
    else{
    	cout<<"YES"<<endl;
    if (n == 1) {
        cout<<"0 0"<<endl;
    }
    if (n == 2) {
        cout<<"0 0\n1 0"<<endl;
    }
    if (n == 4) {
        cout<<"0 0\n0 1\n1 1\n1 0"<<endl;
    }
    }
    return 0;
}
