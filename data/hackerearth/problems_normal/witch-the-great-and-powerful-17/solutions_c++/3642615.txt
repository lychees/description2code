#include <bits/stdc++.h>
#define ll long long 
using namespace std;
ll int a[5];
ll max(ll a,ll b){if(a>b)return a;else return b;}
int main()
{
   	ll t,p,q,r,ans;
   	cin>>t;
   	while(t--)
   	{
   		cin>>p>>q>>r;
   		a[0]=p;a[1]=q;a[2]=r;
   		ans=0;
   		
   		sort(a,a+3);
   		cout<<max(a[2]-a[1]-1,a[1]-a[0]-1)<<endl;
   	}
    return 0;
}
