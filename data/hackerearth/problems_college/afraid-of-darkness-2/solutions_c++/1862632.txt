#include <iostream>
#include<algorithm>
using namespace std;
int main()
{
    long int t;
    cin>>t;
    int n,i,k,flag,days;
    vector <int> a;
    while(t--)
    {
    	days=0;
    	flag=0;
    	k=1;
    	cin>>n;
    	a.resize(n);
    	for(int j=0;j<n;j++)
    	{
    		cin>>a[j];
    	}
    	while(true)						//descending function will take more tym so use asc			
    	{
    		sort(a.begin(),a.end());
    		i=0;
    		while(i<k)
    		{
    			if(a[n-i-1]==0)
    			{
    				flag=1;
    				break;
    			}
    			else
    			{
    				a[n-i-1]--;
    				i++;
    			}
    		}
    		if(flag==1)
    		{
    			break;
    		}
    		days++;
    		k++;
    	}
    		cout<<days<<endl;
    }
}
