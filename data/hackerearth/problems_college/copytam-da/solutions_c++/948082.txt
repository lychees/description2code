#include<iostream>
#include<string>
#include<algorithm>
#define k 1000000

using namespace std;

int main()
{
    int t;
    long n,count=0,i;
    int a[k];
    cin>>t;
    while(t--)
    {
        cin>>n;
        count=1;
        for(i=0;i<n;i++)
            cin>>a[i];
        sort(a,a+n);
        i=0;
        if(n==1)
        {
            cout<<"1\n";
            continue;
        }
        if(n==0)
        {
            cout<<"0\n";
            continue;
        }
        while(i<n-1)
        {
           if(a[i]==a[i+1])
                i++;
           else
           {
               count++;
               i++;
           }


        }
        cout<<count<<endl;
    }
}
