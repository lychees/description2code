using namespace std;

typedef unsigned long long uLong;

uLong gcd(int a, int b)
{
    uLong rem=a%b;
    if(rem==0) return b;
    else
        return gcd(b,rem);
}

int main()
{
    //Code starts here
    int t;
    cin>>t;
    while(t--)
    {
        int a,b;
        cin>>a>>b;

        uLong res=gcd(a,b);
        cout<<res<<endl;
    }
    //Code ends here
    return 0;
}
