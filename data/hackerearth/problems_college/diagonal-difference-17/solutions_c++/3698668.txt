#include <iostream>
using namespace std;

int main()
{
    int n,sum1=0,sum2=0;
    cin>>n;
    int a[n][n];
    for(int i=0;i<n;i++)
    {
    	for(int j=0;j<n;j++)
    	{
    		cin>>a[i][j];
    	}
    }
    for(int i=0;i<n;i++)
    {
    	for(int j=0;j<n;j++)
    	{
    		if(i==j)
    		sum1+=a[i][j];
    		if(i==n-1-j)
    		sum2+=a[i][j];
    	}
    }
    if(sum1-sum2>0)
    cout<<sum1-sum2;
    else
    cout<<sum2-sum1;
    return 0;
}
