#include<iostream>
using namespace std;
static int i,T,j,R[1000],EL[9]={1,2,5,10,20,50,100,500,1000};
unsigned long long int D,ctr=0;
main()
{
	cin>>T;
	for(i=0;i<T;i++)
	{
		ctr=0;
		cin>>D;
		while(D)
		{
			for(j=8;j>=0;j--)
			{
				if(D>=EL[j])
				{
					D=D-EL[j];
					++ctr;
					//cout<<"\nFirst => "<<EL[j]<<"\n";
					break;
				}
			}
		}
		R[i]=ctr;
		ctr=0;
	}
	for(i=0;i<T;i++)
	cout<<R[i]<<"\n";
}
