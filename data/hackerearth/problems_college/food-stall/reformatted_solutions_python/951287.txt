t = input()

for i in xrange(t):
    arr = map(int, list(raw_input().strip().split(' ')))
    n = arr[0]
    d = arr[1]
    arr_t = []
    for i in xrange(n):
        arr_t.append(input())
    arr_t = sorted(arr_t)
    count = 0
    i = 0
    while (i < len(arr_t) - 1):
        if arr_t[i+1] - arr_t[i] <= d:
            count += 1
            i += 2
        else:
            i += 1
    print count
