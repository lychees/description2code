import math


def solve(indx, sums, prev, curr):
    global noOfItem, cnts, wtMax, price, weight
    if indx == noOfItem or curr == cnts or sums == wtMax:
        return prev
    ans1 = solve(indx + 1, sums, prev, curr)
    ans2 = 0
    if (sums + weight[indx]) <= wtMax:
        ans2 = solve(
            indx + 1,
            sums + weight[indx],
            prev + price[indx],
            curr + 1)
    return max(ans1, ans2)
