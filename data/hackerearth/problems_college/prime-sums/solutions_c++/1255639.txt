#include<bits/stdc++.h>
using namespace std;

typedef long long ll;
const int N=4*1e6+10;

bool isPrime[N];
ll dp[N];

void sieve(int n) {
    //memset(isPrime,false,sizeof(isPrime));
    //memset(dp,0,sizeof(dp));
    for(int i=2;i<n;i++) {
        if(isPrime[i]==false) {
            for(int j=2*i;j<n;j+=i) {
                isPrime[j]=true;
            }
        }
        dp[i]+=dp[i-1];
        if(isPrime[i]==false) {
            dp[i]+=i;
        }
    }
    printf("%lld",dp[n-1]);
    return;
}
int main() {
    int n;
    scanf("%d",&n);
    sieve(n);
    return 0;
}
