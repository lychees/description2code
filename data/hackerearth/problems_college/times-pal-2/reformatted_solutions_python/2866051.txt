def check(s1, s2):
    len1 = len(s1)
    res = 0
    for i in range(len1 + 1):
        if i == 0:
            tmp = s2 + s1
        else:
            tmp = s1[0:i] + s2 + s1[i:len1]
        # print tmp, tmp[::-1]
        if tmp == tmp[::-1]:
            res += 1
    print res


t = input()
for i in range(t):
    a = raw_input()
    b = raw_input()
    check(a, b)
