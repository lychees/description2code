#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long t;
    cin>>t;
    while(t--)
    {
        string s,s1;
        cin>>s>>s1;
        int f=0;
        
        for(int i=0,j=0 ; (i<s.size()) && (j<s1.size());i++,j++)
        {
            if(s[i]==s1[j])f++;
        }
        cout<<f<<endl;
        
        
        
        
        
        
    }
    
    return 0;
}
