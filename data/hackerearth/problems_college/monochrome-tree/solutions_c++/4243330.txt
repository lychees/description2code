#include <iostream>
#include<bits/stdc++.h>
using namespace std;
#define ll   int
#define inf 1000000
#define mod 1000000007
#define pb push_back
#define mp make_pair
#define all(v) v.begin(),v.end()
#define S second
#define F first
#define boost1 ios::sync_with_stdio(false);
#define boost2 cin.tie(0);
#define mem(a,val) memset(a,val,sizeof a)

vector<ll>v[1000001];
ll have[100001],need[100001],dp[100001][2];
void dfs(ll cur,ll par)
{
	ll o=0,z=0;
	for(ll i=0;i<v[cur].size();i++)
	{
		ll x=v[cur][i];
		if(x==par)
		continue;
		dfs(x,cur);
		z+=dp[x][0];
		o+=dp[x][1];
	}
	if(have[cur]==need[cur])
	{
		dp[cur][0]=min(z,o+1+1);
		dp[cur][1]=min(o+1,z+1);
	}
	else
	{
		dp[cur][0]=min(o+1,z+1);
		dp[cur][1]=min(o,z+2);
	}
}	
int main()
{
	boost1;boost2;
	ll i,j,n,x,y;
	cin>>n;
	for(i=1;i<=n-1;i++)
	{
		cin>>x>>y;
		v[x].pb(y);
		v[y].pb(x);
	}
	for(i=1;i<=n;i++)
	cin>>have[i];
	for(i=1;i<=n;i++)
	cin>>need[i];
	dfs(1,-1);
	cout<<dp[1][0];
	return 0;
}
