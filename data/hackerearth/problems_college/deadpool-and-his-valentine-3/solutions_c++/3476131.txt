#include <iostream>

using namespace std;

int main()
{
    int M, K;
    bool B[101];
    
    cin >> K >> M;
    
    for (int i = 0; i < M; i++)
    	B[i] = false;
    	
    for (int i = 0; i < K; i++)
    {
    	int X, Y;
    	cin >> X;
    	
    	for (int j = 0; j < X; j++)
    	{
    		cin >> Y;
    		B[Y-1] = true;
    	}
    }
    
    bool ans = true;
    for (int i = 0; i < M; i++)
    {
    	if (!B[i])
    		ans = false;
    }
    
    if (ans)
    	cout << "YES" << endl;
    else
    	cout << "NO" << endl;
    
    
    return 0;
}
