#include<iostream>
using namespace std;
int main()
{
    int t,i;
    cin>>t;
    int n[t],k[t];
    for(i=0;i<t;i++)
    cin>>n[i]>>k[i];
    for(i=0;i<t;i++)
    {
        ++k[i];
        if(n[i]%k[i]==1)
        	cout<<"Nandu\n";
        else
        	cout<<"Rani\n";
        
    }
    return 0;
}
