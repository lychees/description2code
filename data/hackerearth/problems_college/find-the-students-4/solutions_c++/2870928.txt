#include <cstdio>
#include <iostream>
#include <list>
#define DEBUG if(1)
using namespace std;
int main()
{
    int N,tmp,minE = 1010101,newMin = 10101010;
    cin >> N;
    list<int> input;
    for(int i = 0; i < N; ++i)
    {
        cin >> tmp;
        input.push_back(tmp);
        if(minE > tmp) minE = tmp;
    }
    while(!input.empty())
    {
        list<int>::iterator it = input.begin();
        cout << input.size() << endl;
        newMin = 10101010;
        for(;it!=input.end();)
        {
             (*it) = (*it) - minE;
             if((*it) == 0)
                 it = input.erase(it);
             else
             {
                 if(newMin > (*it))
                     newMin = (*it);
                 ++it;
             }
             //DEBUG cout << (*it) << endl;
        }
        minE = newMin;
    }
    return 0;
}
