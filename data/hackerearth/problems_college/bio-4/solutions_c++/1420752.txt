#include <iostream>
using namespace std;

#include <string.h>

int main()
{
    int t,la,lb,c,i,j,s;
    char a[1008],b[1008];
    cin>>t;
    while(t--){
    	cin>>a>>b;
    	la=strlen(a);
    	lb=strlen(b);
    	
    	c=10000000;
    	for(i=0;i<=la-lb;i++){
    		s=0;
    		for(j=0;j<lb;j++){
    			if(a[i+j]!=b[j])
    				s++;
    		}
    		if(s<c)
    			c=s;
    	}
    	cout<<c<<endl;
    }
    return 0;
}
