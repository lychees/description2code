ll n,k,a[1000001];

int sum(ll left,ll right)
{
	if(left!=right)
	{
		if(a[left]+a[right]<k)
			sum(left+1,right);
		else if(a[left]+a[right]==k)
			return 1;
		else
			sum(left,right-1);
	}
	else
		return 0;
}
int main()
{
	ios_base::sync_with_stdio(false);
	ll i,left,right;
	cin>>n>>k;
	forf(i,0,n-1)
	cin>>a[i];
	sort(a,a+n);
	left=0;
	right=n-1;
	if(sum(left,right))
		cout<<"YES";
	else
		cout<<"NO";
	return 0;
}
