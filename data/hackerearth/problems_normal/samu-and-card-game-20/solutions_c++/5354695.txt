    #include <iostream>
int main() {
    int T;
    scanf("%d", &T);
    for(int testCase = 1; testCase <= T; testCase++)
    {
        int N, M;
        scanf("%d", &N);
        scanf("%d", &M);
        int numCard;
        scanf("%d", &numCard);
        int X = 1, Y = 1;
        int jmp = 0;
        for (int i=0; i<numCard; i++) {
            int A, B;
            scanf("%d", &A);
            scanf("%d", &B);
            while ( (X+A) <= N && (X+A) >= 1 && (Y+B) <= M && (Y+B) >= 1) {
                X+=A;
                Y+=B;
                jmp++;
            }
        }
        printf("%d\n", jmp);
    }
    return 0;
}
