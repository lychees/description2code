T = int(raw_input())

while T > 0:
    W = raw_input()
    R = raw_input()

    R = sorted(R)

    i , j = 0, 0
    W = list(W)

    while i < len(R) and j < len(W):
        if R[i] < W[j] :
            W[j] = R[i]
            i +=1
            j +=1
        elif R[i] > W[j]:
            j +=1
        else:
        	j +=1

    print "".join(W)

    T-=1
