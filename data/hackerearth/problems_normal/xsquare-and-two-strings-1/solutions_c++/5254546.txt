#include <bits/stdc++.h>
using namespace std;
 int count(string s1,string s2)
 {
 	int mask1=0;
 	int mask2=0;
 	int n1=s1.length();
 	int n2=s2.length();
 	for(int i=0;i<n1;i++)
 	{
 		mask1=mask1 | (1<<(s1[i]-'a'));
 	}
 	for(int i=0;i<n2;i++)
 	{
 		mask2=mask2 | (1<<(s2[i]-'a'));
   }
   return mask1 & mask2;
 }
int main()
{
    int t;
    cin>>t;
     while(t--)
     {
     	string s1;
     	string s2;
     	cin>>s1>>s2;
       int i=count(s1,s2);
       if(i>=1)
       cout<<"Yes"<<endl;
       else
       cout<<"No"<<endl;
      // i=count(s1,s2)?cout<<"Yes":cout<<"No";
     }
    return 0;
}
