#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long p,q,i,n,ann,bnn,l;
    string r;
    char an[15],bn[15];
    int flag=0,k;
    cin>>p>>q;
    for(i=p;i<=q;i++)
    {
    	n=i*i;
    	
    	//cout<<n<<"\n";
    	ostringstream c;
    	c<<n;
    	r=c.str();
    	//cout<<r<<"\n";
    	l=r.length();
    	//cout<<l<<"\n";
    	int j=0;
    	for( j=0;j<l/2;j++)
    	{
    		an[j]=r[j];
    	}
    	an[j]='\0';
    	if(!strlen(an))
    	{
    		an[0]='0';
    		an[1]='\0';
    	}
    	
    	//cout<<an<<"\n";
    	k=0;
    	for( j;j<l;j++)
    	bn[k++]=r[j];
    	bn[k]='\0';
    	//cout<<bn<<"\n";
    	istringstream buffer(an);
	
		buffer>>ann;
		istringstream buf(bn);
	
		buf>>bnn;
		//cout<<ann<<" "<<bnn<<"\n";
		//cout<<ann+bnn<<"\n";
		if(ann+bnn==i)
		{
			cout<<i<<" ";
			flag=1;
		}
    }
    if(!flag)
    cout<<"INVALID RANGE\n";
    return 0;
}
