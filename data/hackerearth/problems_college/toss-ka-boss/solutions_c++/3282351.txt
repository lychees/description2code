#include <iostream>
using namespace std;
#define max 1000005
#define ll long long int
int main()
{
     string s;
    ll t,i;
    scanf("%lld",&t);
    while(t--){
    	ll k = 0;
    	ll count=0;
    	cin >> s;
    	
   
    	for(i = 0; s[i]!='\0'; i++){
    		if(s[i] == 'H'){ 
    			count++;
    			k =k+count;
       }
			else
    		  count=0;
    }
    	printf("%lld\n",k);	 
    	
    }
    return 0;
}
