#include<cstdio>
#include<cmath>
int a[100004];
int main()
{
	long long int t,n,i;
	scanf("%lld",&t);
	while(t--)
	{
		long long int ans=0,g=0;
		scanf("%lld",&n);
		if(n<3)
        ans=1;
        else
       ans=ceil(log10(2*3.141592653589793*n)/2 + n*log10(n/2.7182818284590452353));
	  for(i=1;(5*i)<=n;i*=5)
       {
       g+=n/(5*i);
	   }
	   ans=ans-g;
     
     
		printf("%lld\n",ans);
	}
	return 0;
}
