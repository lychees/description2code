#define FOR(i,a,b) for (long long i=a;i<b;i++)
#define FORE(i,a,b) for (long long i=a;i<=b;i++)
#define FORD(i,a,b) for (long long i=a;i>=b; i--)
#define TR(c, it) for(typeof((c).begin()) it=(c).begin(); it!=(c).end(); it++)
#define ri(n)({\
    int neg=0;\
    n=0;\
    char ch;\
    for(ch=getchar(); ch<'0' || ch>'9'; ch=getchar()) if (ch=='-') neg=1-neg;\
    n=ch-48;\
    for(ch=getchar(); ch>='0' && ch<='9'; ch=getchar()) n=(n<<3)+(n<<1)+ch-48;\
    if (neg) n=-n;\
})

typedef long long ll;
typedef unsigned long long ull;
typedef pair<int,int> II;
typedef pair<ll,ll> LL;
const ll INF=1000000009;
const double esp=1e-13;
const double pi=3.141592653589;

typedef struct BIG{
    ll l,d[3][3];
    BIG (int v=0){
        memset(d,0,sizeof(d));
        l=v;
    }

    BIG operator *(const BIG &op)const{
        BIG ans;
        ans.l=max(op.l,l);
        FORE(i,1,ans.l)
        FORE(j,1,ans.l)
        FORE(k,1,ans.l)
        ans.d[i][j]=(ans.d[i][j]+d[i][k]*op.d[k][j]%INF )%INF;
        return ans;
    }
};

BIG tmp,c,f;
int test;
ll n;

int main(){
 //   freopen("akashs-girlfriend.inp", "r", stdin);
 //   freopen("akashs-girlfriend.out", "w", stdout);
    ri(test);
    while (test--){
        ri(n);
        if (n==1){
            cout << 1 << "\n";
            continue;
        }
        c.l=2;
        c.d[1][1]=c.d[2][2]=9;
        c.d[1][2]=c.d[2][1]=1;
        tmp=c;
        n-=2;
        while (n){
            if (n%2) tmp=tmp*c;
            c=c*c;
            n/=2;
        }
        f.l=2;
        f.d[1][1]=1;
        f.d[1][2]=9;
        f=f*tmp;
        cout << f.d[1][1] << "\n";
    }
}
