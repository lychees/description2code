#include <iostream>
using namespace std;
int hash1[1000],hash2[1000];

int main()
{
	int i;
	string s1,s2;
	cin >> s1 >> s2;
	for(int i=0;i<s1.length();i++)
		hash1[s1[i]]++;
	for(int i=0;i<s2.length();i++)
		hash2[s2[i]]++;
	for(i=65;i<=90;i++)
	{
		if(hash1[i]!=hash2[i])
		{
			printf("WRONG\n");
			return 0;
		}
	}
	printf("CORRECT\n");
    return 0;
}
