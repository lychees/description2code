#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	long long int clk,i,j,n,x;
	scanf("%lld",&n);
	char* arr[5] = {"Sheldon", "Leonard", "Penny", "Rajesh", "Howard" };
	x=0;
	clk = 1;
	i=1;
	j=5;
	while(i<n){
		i+=j;
		j*=2;
		clk*=2;
	}
	clk/=2;
	while(i>n){
		i-=clk;
		x--;
	}
	if(x<0)
		x%=5;
	if(x<0)	
		x+=5;
	printf("%s\n",arr[x]);
	return 0;
}
