def main():
    tc = int(raw_input())
    for i in range(0, tc):
        n = int(raw_input())
        strengths = sorted(map(int, raw_input().split()))
        sum = 0
        for i in range(0, n):
            sum += (2*i - n + 1) * strengths[i]
        sum = sum*strengths[n-1]
        print sum % 1000000007


main()
