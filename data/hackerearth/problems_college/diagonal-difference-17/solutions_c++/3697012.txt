#include<bits/stdc++.h>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(0);

	long long int a=0,b=0,x;
	int n;
	cin >> n;
	for(int i=1;i<=n;i++)
		for(int j=1;j<=n;j++)
			{
				cin >> x;
				if(i==j)
					a+=x;
				if(i+j==(n+1))
					b+=x;
			}
	a-=b;
	if(a<0)
		a=-a;
	cout<<a;
	return 0;
}
