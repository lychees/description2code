#include <iostream>
#include <string>
using namespace std;

int solve(string s)
{
      if(s[0] != 'R') return 0;
      int sz = s.size();
      for(int i=1;i<sz;i++)
      {
            if(s[i]!='R' && s[i]!='Y') return 0;
            if(i>2 && s[i] == 'Y' && s[i-1] == 'Y' && s[i-2] == 'Y')
            return 0;
      }
      return 1;
}

int main()
{
	long t;
	cin>>t;
	string s;
	while (t--){
		cin>>s;
		if(solve(s))
		 cout<<"YES"<<endl;
		else cout<<"NO"<<endl;
	}
	return 0;
}
