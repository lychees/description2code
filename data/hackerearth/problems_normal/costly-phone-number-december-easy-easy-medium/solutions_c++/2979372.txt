#include <bits/stdc++.h>
#define ll long long
using namespace std;
bool valid(int i)
{
	if(i>=0 && i<10)
	return 1;
	return 0;
}
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int test;
	cin>>test;
	while(test--)
	{
		ll int a[10];
		ll int i,j,t,l;
		string str;
		for(i=0;i<10;i++)
		{
			cin>>a[i]; //Cost of pressing the digits
		}
		cin>>l>>str;
		bool flag;
		while(1)
		{
			flag=0;
			for(i=0;i<10;i++)
			{
				for(j=0;j<10;j++)
				{
					t=(i+j)%10;
					if((a[i]+a[j])<a[t])
					{
						flag=1;
						a[t]=a[i]+a[j];
					}
				}
			}
			if(!flag)
			break;
		}
		ll int ans=0;
		for(i=0;i<l;i++)
		{
			ans+=a[str.at(i)-'0'];
		}
		cout<<ans<<endl;
	}
	return 0;
}
