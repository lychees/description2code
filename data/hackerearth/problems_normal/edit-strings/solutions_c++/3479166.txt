#include<iostream>
#include<string>
using namespace std;
int main()
{
	int n,k,q,i,count;
	char c;
	cin>>n>>k;
	string s;
	cin>>s;
	while(k--)
	{
		count=0;
		cin>>q;
		cin>>c;
		s[q-1]=c;
		for(i=1;s[i]!='\0';i++)
		{
			if(s[i]=='.' && s[i-1]=='.')
			{
				count++;
			}
		}
		cout<<count<<endl;
	}
	return 0;
}
