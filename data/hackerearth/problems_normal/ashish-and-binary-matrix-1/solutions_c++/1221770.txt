#include <iostream>
#include <string>
#include<set>
using namespace std;

int main()
{
   int t,n,m,i;
   cin>>t;
   set<string> s;
   while(t--){
   cin>>n>>m;
   	string str;
   	for(i=0;i<n;i++){
   	cin>>str;
   	s.insert(str);
   	}
   	if(s.size()==n){
   		cout<<"Yes"<<"\n";
   	}
   	else
   	{
   		cout<<"No"<<"\n";
   	}
   	s.clear();
   }
}
