#include <bits/stdc++.h>
using namespace std;
int n,k,curr;
char str[100005],str2[100005];
deque <int> v[30];
vector <char> v2;
bool used[100005];
int cnt[30];
int main(){
	scanf("%d %d\n",&n,&k);
	gets(str);
	int l=strlen(str);
	strcpy(str2,str);
	sort(str2,str2+l);
	for(int x=0;x<l;x++) v[str[x]-'a'].push_back(x);
	for(int x=0;x<l;x++){
		if(str[x]!=str2[x]){
			if(cnt[str2[x]-'a']){
				if(!used[x]) cnt[str[x]-'a']++,--k;
				else used[x]=0;
				str[x]=str2[x];
				cnt[str2[x]-'a']--;
				if(!k) break;
			}
			else if(k>1){
				if(!used[x]) cnt[str[x]-'a']++,k-=2;
				else used[x]=0,--k;
				used[v[str2[x]-'a'].back()]=1;
				v[str2[x]-'a'].pop_back();
				str[x]=str2[x];
				if(!k) break;
			}
			else if(used[x]){
				used[x]=0;
				used[v[str2[x]-'a'].back()]=1;
				v[str2[x]-'a'].pop_back();
				str[x]=str2[x];
				--k;
				break;
			}
			else break;
		}
	}
	cnt[26]=1;
	int c=0;
	for(int x=0;x<l;x++){
		while(cnt[c]==0) c++;
		if(used[x]){
			str[x]='a'+c;
			cnt[c]--;
		}
		else if(k&&c+'a'<str[x]){
			k--;
			cnt[c]--;
			cnt[str[x]-'a']++;
			str[x]='a'+c;
		}
	}
	puts(str);
	return 0;
}
