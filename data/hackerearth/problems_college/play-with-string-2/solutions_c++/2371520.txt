#include<iostream>
#define max 1000000
using namespace std;
int main()
{
	char a[max],b[max];
	int tc,i,l,flag,m,n;
	cin>>tc;
	while(tc--)
	{
		cin>>a;
		for(l=0;a[l]!='\0';l++);
		for(i=0;i<l;i++)
			b[l-1-i]=a[i];
		b[i]='\0';
		//cout<<a<<" "<<b;
		flag=0;
		for(i=1;i<l-1;i++)
		{
			m=a[i]-a[i-1];
			n=b[i]-b[i-1];
			if(m<0)
				m=m*-1;
			if(n<0)
				n=n*-1;
			if(m!=n)
			{
				flag=1;
				break;
			}
		}
		if(flag==1)
			cout<<"Not Funny";
		else
			cout<<"Funny ";
		cout<<endl;
	}
}
