using namespace std;

long long gcd (long a, long b){
	if(b != 0)
		return gcd(b, a%b);
	else
		return a;
}

int main()
{
	int t;
	scanf("%d",&t);
	for(int i=1; i<=t; i++){
		long long n,a,b;
		scanf("%lld %lld %lld",&n, &a, &b);
		
		if(a == 0 || b == 0)
			printf("-1\n");
		else{
			long long na = n / a;
			long long nb = n / b;
			long long lcm = (a*b) / gcd(a,b);
			long long nab = n / lcm;
			printf("%lld\n",na+nb-nab);
		}
	}
	return 0;
}
