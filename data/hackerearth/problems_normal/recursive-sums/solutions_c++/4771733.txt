#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{ int tc;
  cin>>tc;
  while(tc--)
  { int b;
    cin>>b;
    long long int sum=0;
    int count=0;
    int val=b;
    while(b--)
    { long long int val1,val2;
      cin>>val1>>val2;
      if(val2==0)
         count++;
      sum=((sum%9)+((val1%9)*(val2%9))%9)%9;
    }
    if(count==val)
        cout<<"0"<<"\n";
    else if(sum==0)
       cout<<"9"<<"\n";
    else
        cout<<sum<<"\n";
  }
  return 0;
}
