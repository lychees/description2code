#include <iostream>
using namespace std;

int main()
{
	long long a,b,fa,fb;
	int t;
	cin>>t;
	while(t--){
   cin>>a>>b;
   fa=0;fb=0;
   while(a){
   	fa=fa*10+a%10;
   	a=a/10;
   }
     while(b){
   	fb=fb*10+b%10;
   	b=b/10;
   }
   fa=fa+fb;
    while(fa){
   	a=a*10+fa%10;
   	fa=fa/10;
   }
   cout<<a<<endl;
	}
    return 0;
}
