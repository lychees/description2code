#include <iostream>
using namespace std;

int main()
{
   int t,j;
cin>>t;


for(j=0;j<t;j++)
{
	int n,r,k,i=0,c=0,p;
	cin>>n;
	p=n;
	
	int a[1000]={0};
	while(n>0)
	{
		r=n%10;
		n=n/10;
		a[i]=r;
		i++;
	}
	for(k=0;k<i-1;k++)
	{
		if(a[k]==1&&a[k+1]==2)
		{
			c=1;
		}
		
		
	}
	
	if(c==1)
{
	cout<<"The streak is broken!"<<endl;
}
	
	
else if(p%21==0)
	{
		cout<<"The streak is broken!"<<endl;
	}
else
{
	cout<<"The streak lives still in our heart!"<<endl;
}
	
}

return 0;
}
