#include <iostream>
#include <bits/stdc++.h>
#define ll long long 
using namespace std;
ll fun(int a)
{
	 if(a==1)
	  return 0;
	  
     ll t=1;
     while(1)
     {
       if(t>=a)
        break;
       t=2*(t+1)-1;
     }
     ll dif=(a-(t+1)/2);
     if(dif==0)
     {
     	return 0;
     }
     else 
     {
     	if(fun((t+1)/2-dif))
     	return 0;
     	else return 1;
     }
}
int main()
{
    //cout << "Hello World!" << endl;
    ll t;
    scanf("%lld",&t);
    while(t--)
    {
    	ll k;
    	scanf("%lld",&k);
    	//cout << endl;
    	cout << fun(k) << endl;
    }
    return 0;
}
