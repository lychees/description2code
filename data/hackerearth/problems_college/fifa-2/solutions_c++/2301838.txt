#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int hc[100005];
int main(){
	int t;
	cin>>t;
	int h[t+1];
	int a[t+1];
	for(int i=0;i<t;i++){
		cin>>h[i+1]>>a[i+1];
		hc[h[i+1]]++;
	}
	for(int i=1;i<=t;i++){
		int h_res;
		int a_res;
		h_res=t-1+hc[a[i]];
		a_res=2*(t-1)-h_res;
		cout<<h_res<<" "<<a_res<<endl;
	}
}
