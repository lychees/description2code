#include<iostream>
#include<cstdio>
using namespace std;
int main(){
	int n;
	while(scanf("%d",&n)!=EOF){
		int arr[n];
		for(int i=0;i<n;i++){
			cin>>arr[i];
		}
		int count=0;
		for(int i=n-1;i>=0;i--){
			if(!(arr[i]&1)){
				cout<<arr[i]<<" ";
				 count++;
			}
		}
		if(!count)
		cout<<"None to display\n";
		else
		cout<<endl;
	}
}
