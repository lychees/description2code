#include<iostream>
#include<algorithm>
using namespace std;
long long get(long long n)
{
    if((n&1)==1)n++;
    return n*n/4;
}
void solve()
{
    long long n,i,o=0;
    cin>>n;
    for(i=1;i<=n;i<<=1)
        o+=get(n/i);
    cout<<o<<'\n';
}
int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int t;
    cin>>t;
    while(t--)
        solve();
}
