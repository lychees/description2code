#include <bits/stdc++.h>

#define ll long long
#define pb push_back
#define mp make_pair
#define ld double

using namespace std;

int main() {
#ifdef LOCAL
	freopen("input.txt", "r", stdin);
#endif
	string s;
	cin >> s;
	int res = 0;
	for (int i = 0; i < s.length();) {
		int j = i;
		while (j + 1 < s.length() && s[j + 1] == s[j])
			j++;
		res += (j - i + 1) * (j - i + 2) / 2;
		i = j + 1;
	}
	cout << res << "\n";
#ifdef LOCAL
	printf("%.3lf s\n", (double) clock() / (double) CLOCKS_PER_SEC);
#endif
}
