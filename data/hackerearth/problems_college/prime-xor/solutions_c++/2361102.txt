#include <iostream>
using namespace std;
int arr[10000001];
void pre()
{
	long long i,j;
	for(i=0;i<10000001;i++)
		arr[i]=0;
	for(i=2;i<10000001;i++)
	{
		if(arr[i]==0)
		{
			for(j=2*i;j<10000001;j+=i)
				arr[j]=1;
		}
	}
}
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    pre();
    while(t--)
    {
    	long long a,b,i;
    	cin>>a>>b;
    	if(a>2)
    		cout<<"0"<<endl;
    	else
    	{
    		long long count_primes=0;
    		for(i=3;i<=b;i++)
    		{
    			if(arr[i]==0)
    			{
    				count_primes++;
    			}
    		}
    		cout<<count_primes<<endl;
    	}
    }
    return 0;
}
