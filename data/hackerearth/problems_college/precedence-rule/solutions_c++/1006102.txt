#include <iostream>
#include<cstdio>
#include<string>
using namespace std;

int main() 
{
	long long int t,ans;
	cin>>t;
	string s;
	while(t--)
	{
		cin>>s;
		ans=s[s.length()-1]-48;
	//	cout<<ans<<"%"<<endl;
		for(int i=s.length()-2;i>0;i=i-2)
		{
			if(s[i]=='+')
			{
				ans=ans +(s[i-1]-48);
			//	cout<<"%%"<<endl;
			}
			else if(s[i]=='-')
			{
				ans=ans-(s[i-1]-48);
			}
			else if(s[i]=='*')
			ans=ans*(s[i-1]-48);
			 //cout<<"**"<<endl;
			// cout<<ans<<endl;}
		}
		cout<<ans<<endl;
	}
	return 0;
}
