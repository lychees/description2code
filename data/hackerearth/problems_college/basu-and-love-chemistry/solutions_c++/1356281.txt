#include<bits/stdc++.h>
using namespace std;

typedef long long ll;
const int md=1e9+7;

int main() {
    string str;
    cin>>str;
    int len=str.size();
    ll ans=0;
    ll res=str[0]-'0';
    for(int i=1;i<len;i++) {
        ans=((ans+str[i-1]-'0')*10)%md;
        ans=(ans+((i)*(str[i]-'0')))%md;
        res=(res+ans)%md;
        res=(res+(str[i]-'0'))%md;
    }
    cout<<res<<endl;
    return 0;
}
