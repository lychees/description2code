numCases = int(raw_input())
for iCase in xrange(numCases):
    pars = raw_input()
    state = 0
    ans = ''
    for ch in pars:
        if ch == '(':
            state += 1
        else:
            if state == 0:
                ans = '('+ans
            else:
                state -= 1
    ans += pars
    while state > 0:
        ans += ')'
        state -= 1
    print ans
