#include <iostream>
using namespace std;

int main()
{
   int tc;
   cin>>tc;
   while(tc--)
   {
   	int n,p;
   	cin>>n>>p;
   	int flag=0;
   	for(int i=0;i<p;i++)
   	{
   		if(n==3){
   			cout<<n<<endl;
   			flag=1;
   			break;
   		}
   		if(n&1)
   		n=(n/2)+2;
   		else
   		 n=n/2+1;
   	}
   	if(!flag)
   	cout<<n<<endl;
   }
    return 0;
}
