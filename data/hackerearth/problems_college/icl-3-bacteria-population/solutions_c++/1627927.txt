#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int n;
    scanf("%d",&n);
    vector<int> born(n+1,0);
    vector<int> total(n+1,0);
    born[0]=1;
    total[0]=1;
    for(int i=0;i<=n;++i)
    {
    	if(born[i]>0)
    	{
    		if(i+2<=n) { born[i+2]+=born[i]; total[i+2]+=born[i];}
    		if(i+3<=n){ born[i+3]+=born[i];}
    	}
    	if(i>0)
    		total[i]+=total[i-1];
    }
    printf("%d\n",total[n]);
    return 0;
}
