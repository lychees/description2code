#include <bits/stdc++.h>
using namespace std;

int main()
{
	long t,i,x,n;
	vector<long> v;
	
    cin>>t;
    while(t--)
    {
        v.clear();
        cin>>n;
        for(i=0;i<n;i++)
        {
        cin>>x;
        v.push_back(x);
        }
        if(is_sorted(v.begin(),v.end()))
        {
        for(i=0;i<v.size();i++)
         cout<<v[i]<<" ";
        cout<<endl;continue;}
        else{
        prev_permutation(v.begin(),v.end()) ;
        for(i=0;i<v.size();i++)
        cout<<v[i]<<" ";
        cout<<endl;continue;
        }
    }
    return 0;
}
