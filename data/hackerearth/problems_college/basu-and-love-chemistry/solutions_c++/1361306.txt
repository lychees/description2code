#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const ll N = 1024*1024;
const ll INF = 1000000000;
const ll md =1000*1000*1000+7;
string t;
ll maxar,a[N],b[N];
int main()
{
  //freopen("input.txt","r",stdin);freopen("output.txt","w",stdout);
    cin>>t;
    b[0]=1;
    for (int i=1;i<=t.size()+2;i++)
        b[i]=(b[i-1]*10)%md;
    a[0]=b[0];
    ll ans=0;
    for (int i=1;i<=t.size()+2;i++)
        a[i]=(a[i-1]+b[i])%md;

    for (int i=t.size()-1;i>=0;i--)
    {
        maxar=t.size()-i-1;
        ans+=(t[i]-'0')*a[maxar]*(i+1);
        ans=ans%md;
    }
    printf("%lld",ans);
return 0;
}
