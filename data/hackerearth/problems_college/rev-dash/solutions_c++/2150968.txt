#include <cstdio>
#include<string.h>
using namespace std;

int main()
{
    char a[1000000];
    long n,i;
    scanf("%s",&a);
    n=strlen(a);
    for(i=0;i<n;i++)
    {
    	printf("%c",a[i]);
    	if(i<n-1)printf("-");
    }
    printf("\n");
    for(i=n-1;i>=0;i--)
    {
    	printf("%c",a[i]);
    	if(i>0)printf("-");
    }
    return 0;
}
