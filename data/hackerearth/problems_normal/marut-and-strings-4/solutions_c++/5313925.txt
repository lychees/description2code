#include <iostream>
using namespace std;

int main()
{
    int N,i;
    string s;
    
    cin>>N;
    if(N>=1 && N<=10){
	    while(N--){
	    	cin>>s;
	    	int l = 0;
	    	int u = 0;
	    	if(s.length()<=100 && s.length()>=1){
		    	for(i=0;i<s.length();i++){
		    		if(islower(s[i])){
		    			l++;
		    		}else if(isupper(s[i])){
		    			u++;
		    		}
		    	}
	    	}
	    	
	    	if(l==0 && u==0){
	    		cout<<"Invalid Input"<<endl;
	    	}else if(l==0 || u==0){
	    		cout<<0<<endl;
	    	}else{
	    		if(l<=u){
	    			cout<<l<<endl;
	    		}else{
	    			cout<<u<<endl;
	    		}
	    	}
	    }
    }else{
    	cout<<"Invalid Test"<<endl;
    }
    
    
    return 0;
}
