#include<bits/stdc++.h>
using namespace std;

int main() {
	long long int t,d,zeroes ,five_mul ,num;
	double f,j,i;
	scanf("%lld",&t);
	while(t--) {
		scanf("%lf",&f);
		zeroes = 0;
		if(f == 0 || f ==1 ) {
			d = 0;
		} else {
			j = (log10(2*3.14159265358979323846264338327950288419716939937510*f)/(2.00)+f*(log10(f/2.71828182845904523536)));
			d = j;
		}	
		five_mul = 5;
		num = f;
		while(1) {
			zeroes += num/five_mul ;
			five_mul *= 5;
			if(num < five_mul)
				break;
		}
		cout << d -zeroes + 1 << endl;
	}
	return 0;
}
