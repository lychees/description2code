#include <bits/stdc++.h>
using namespace std;

int main()
{long long int t,n,max,i,a[100000],s1[100000],s2,j,min;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	sort(a,a+n);
    	j=n-1;
    	int q=0;
    	for(i=0;i<n;i++)
    	{
    	s1[q++]=a[i]+a[j];
   j--;
    	}
    	min=99999;
    	for(i=0;i<q;i++)
    	{
    		if(s1[i]<min)
    		min=s1[i];
    	}
    	max=0;
    	for(i=0;i<q;i++)
    	{
    		if(s1[i]>max)
    		max=s1[i];
    	}
    	int diff=max-min;
    	cout<<diff<<endl;
}
    return 0;
}
