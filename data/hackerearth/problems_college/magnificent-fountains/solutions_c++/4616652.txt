#include <bits/stdc++.h>
#define ll long long int
#define w(t) while(t--)
#define fb(i,n) for(i=0;i<n;i++)
#define mod 1000000007


using namespace std;

bool notprime[10003];

ll powi(ll a,ll b)
{
    ll ans=1;
    while(b)
    {
        if(b&1)
            ans=(ans*a)%mod;
        a=(a*a)%mod;
        b=(b>>1);
    }
    return ans;
}

int main()
{
    ll i,j;
    for(i=2;i<=100;i++)
    {
        if(!notprime[i])
        {
            j=i*i;
            while(j<=10000)
            {
                notprime[j]=true;
                j+=i;
            }
        }
    }
    ll t;cin>>t;w(t){
    	ll n,i,ans=1,temp=1;
    	cin>>n;
    	ll arr[n];
    	fb(i,n)
    	{
    		cin>>arr[i];
    	}
    	for(i=2;i<=10000;i++)
        {
            if(!notprime[i])
            {
                ll maxi=0;
                for(j=0;j<n;j++)
                {
                    ll cnt=0;
                    //cout<<i<<" "<<j<<endl;
                    while(arr[j]%i==0)
                    {
                        //cout<<arr[j]<<endl;
                        arr[j]/=i;
                        cnt++;
                    }
                    maxi=max(maxi,cnt);
                }
                ans=(ans*powi(i,maxi))%mod;
            }

    }

}
