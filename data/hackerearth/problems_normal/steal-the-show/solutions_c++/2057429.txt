int main()
{

    for(i=0;i<n;i++) cin>>a[i];

        int mid = (n+3)/2;
        f[mid] = ceo;
        f[0]=min(cto,coo);
        f[n+3-1]=max(cto,coo);
        sort(a,a+n);
        j=0;
        for(i=1;i<mid;i++){
            f[i]=a[j];j++;
        }
        for(i=mid+1;i<(n+3)-1;i++){
            f[i]=a[j];j++;
        }
        int ans=0;
        for(i=0;i<n+3-1;i++) ans+=abs(f[i]-f[i+1]);
        cout<<ans;
    }
    else{
        int mid = (n+3)/2;
        f[mid] = ceo;
        f[0]=min(cto,coo);

        j=0;
        for(i=1;i<mid;i++){
            f[i]=a[j];j++;
        }
        for(i=mid+1;i<(n+3)-1;i++){
            f[i]=a[j];j++;
        }
        int temp=0,temp2=0;
        for(i=0;i<n+3-1;i++) temp+=abs(f[i]-f[i+1]);
        mid--;
        f[mid] = ceo;
        f[0]=min(cto,coo);
        f[n+3-1]=max(cto,coo);
        sort(a,a+n);
        j=0;
        for(i=1;i<mid;i++){
            f[i]=a[j];j++;
        }
        for(i=mid+1;i<(n+3)-1;i++){
            f[i]=a[j];j++;
        }
        for(i=0;i<n+3-1;i++) temp2+=abs(f[i]-f[i+1]);
        cout<<min(temp,temp2);
    }
    
    return 0;
}
