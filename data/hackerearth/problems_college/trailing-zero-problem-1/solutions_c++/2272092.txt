#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	long number;
    	cin>>number;
    	int power=1;
    	int temp=number/(long)pow(5,power);
    	long zero_count=0;
    	while(temp>=1){
    		zero_count+=temp;
    		temp=number/(long)pow(5,++power);
    	}
    	double digit_count=0;
    	for(int i=2;i<=number;i++){
    		digit_count+=log10(i);
    	}
    	digit_count++;
    //	cout<<digit_count<<" "<<zero_count<<endl;
    	cout<<(long)digit_count-zero_count<<endl;
    }
    return 0;
}
