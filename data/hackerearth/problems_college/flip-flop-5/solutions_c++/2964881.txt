#include<bits/stdc++.h>
using namespace std;
int main()
{
    char ar[100100];
    int n,k;
    cin>>n;
    while(n--)
    {
        k=0;
        scanf("%s",&ar);
        for(int i=0;i<strlen(ar)-1;i++)
        {
            if(ar[i]==ar[i+1])
                k++;
        }
        printf("%d\n",k);
    }
}
