#include <iostream>
using namespace std;

int main() {
	int tcases;
	cin>>tcases;
	while(tcases--){
		char str[100];
		cin>>str;
		int arr[53];
		for(int i=0;i<53;++i){
			if(i%2==0)
				arr[i]=0;
			else
				arr[i]=(i/2)+97;
		}
		for(int i=0;str[i]!='\0';++i){
			int val=((int)str[i]-96)*2;
			arr[val]=arr[val]+1;
		}
		//insertsort
		int temp1,temp2;
		for (int i = 4; i < 53; i=i+2)
    	{
        	for (int j = i; j >= 4; j=j-2)
        	{
           		if (arr[j] < arr[j-2])
           		{
           	   		temp2 = arr[j];
           	   		temp1 = arr[j-1];
           	   		arr[j] = arr[j-2];
           	   		arr[j-1] = arr[j-3];
           	   		arr[j-2] = temp2;
           	   		arr[j-3] = temp1;
           		}
           		else
           	   		break;
        	}
    	}
    	for(int i=1;i<53;i+=2){
    		for(int j=0;j<arr[i+1];++j){
				if(arr[i+1]!=0)
					cout<<(char)arr[i];
    		}
		}
		cout<<endl;
	}
	return 0;
}
