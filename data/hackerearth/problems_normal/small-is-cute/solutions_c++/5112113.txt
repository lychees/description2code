#include <bits/stdc++.h>

using namespace std;

const int N = 5e3 + 10;

string s[N];

int cmp(const string &a, const string &b) {
	string a2 = a;
	string b2 = b;
	return a2.append(b) < b2.append(a);
}

int main() {
	int n;
	cin >> n;
	for (int i = 0; i < n; i++) {
		cin >> s[i];
	}
	sort(s, s + n, cmp);
	for (int i = 0; i < n; i++) {
		cout << s[i];
	}
	return 0;
}
