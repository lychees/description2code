#include<iostream>
#include<cstdio>
#include<algorithm>
#include<string>
#include<vector>
using namespace std;
int main()
{
	int n;
	cin>>n;
	int a[n],b[n];
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
		b[i]=a[i];	
	}
	sort(b,b+n);
	int aa[n][n];
	
	if(a[0]==b[0])
	aa[0][0]=1;
	else
	aa[0][0]=0;
	
	for(int i=1;i<n;i++)
	{
		if(a[i]==b[0])
		{
			aa[i][0]=1;
		}
		else
		{
			aa[i][0]=aa[i-1][0];
		}
	}
	for(int i=1;i<n;i++)
	{
		if(b[i]==a[0])
		{
			aa[0][i]=1;
		}
		else
		{
			aa[0][i]=aa[0][i-1];
		}
	}
	for(int i=1;i<n;i++)
	{
		for(int j=1;j<n;j++)
		{
			if(a[i]==b[j])
			{
				aa[i][j]=aa[i-1][j-1]+1;
			}
			else
			{
				aa[i][j]=max(aa[i-1][j],aa[i][j-1]);
			}
		}
	}
	vector<int> ss;
	int aaa=aa[n-1][n-1];
	while(aaa)
	{
		ss.push_back(aaa%2);
		aaa/=2;
	}
	
	for(int i=ss.size()-1;i>-1;i--)
	{
		cout<<ss[i];
	}
	
}
