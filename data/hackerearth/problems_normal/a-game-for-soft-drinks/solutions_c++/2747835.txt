#include <bits/stdc++.h>
using namespace std;
#define max1 101
int ans[max1][max1][max1];
void solve()
{
	
	ans[0][0][0]=0;
	for(int i=0;i<max1;i++)
	{
		for(int j=0;j<max1;j++)
		{
			for(int k=0;k<max1;k++)
			{
			    if( (i > 0 && ans[i-1][j][k]==0 )|| 
			    (j>0 && ans[i][j-1][k]==0 ) || 
			    (k>0 && ans[i][j][k-1] ==0) ||
			    (i>0 && j > 0 && ans[i-1][j-1][k]==0) ||
			    (j>0 && k > 0 && ans[i][j-1][k-1]==0 ) || 
			    (i>0 && k>0 && ans[i-1][j][k-1]==0)||
			    (i>0 && j>0 && k>0 && ans[i-1][j-1][k-1]==0))
			    ans[i][j][k]=1;
			    else
			    ans[i][j][k]=0;
			}
		}
	}
	
}

int main()
{
    ///cout << "Hello World!" << endl;
    solve();
    int t;
    cin>>t;
    while(t--)
    {
    	int a,b,c;
    	cin>>a>>b>>c;
    	if(ans[a][b][c]==0)
    	cout<<"Aishwarya"<<endl;
    	else
    	cout<<"Ashima"<<endl;
    }
    return 0;
}
