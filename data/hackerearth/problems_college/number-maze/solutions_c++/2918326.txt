#include <iostream>
using namespace std;

int main()
{
    int a,b;
    cin>>a>>b;
    int flag=0;
    for(int i=a+1;i<b;i++)
    {
    	if(i%15==0)
    	{
    		cout<<i;
    		flag=1;
    	}
    	else if(i%3!=0 && i%5!=0)
    	{
    		cout<<i;
    		flag=1;
    	}
    	
    	if(flag && i!=b-1)
    	cout<<",";
    	
    	flag=0;
    }
    return 0;
}
