#include <iostream>
using namespace std;

int check(int **m,int i,int j,int n){
	int k = 0;
	if(i == n-1 && j == n-1)
		return 1;
	if(m[i][j+1] == 1)
		k = check(m,i,j+1,n);
	if(k == 1)
		return 1;
	if(m[i+1][j] == 1)
		return check(m,i+1,j,n);
	return 0;
}

int main()
{
    int t,**m,n,i,j;
    cin>>t;
    while(t--){
    	cin>>n;
    	m = new int*[n];
    	for(i=0;i<n;i++)
    		m[i] = new int[n];
    	
    	for(i=0;i<n;i++)
    		for(j=0;j<n;j++)
    			cin>>m[i][j];
    			
    	if(check(m,0,0,n))
    		cout<<"POSSIBLE";
    	else
    		cout<<"NOT POSSIBLE";
    	cout<<endl;
    	for(i=0;i<n;i++)
    		delete m[i];
    	delete m;
    }
    return 0;
}
