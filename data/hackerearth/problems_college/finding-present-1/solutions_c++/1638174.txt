#include<bits/stdc++.h>
#include<vector>
#include<math.h>
#include<string>
using namespace std;
typedef long long int ll;
#define s(x) scanf("%d",&x)
#define p(x) printf("%d\n",x)
#define f1(x,r) for(x=0;x<r;x++)
#define f2(x,r) for(x=1;x<=r;x++)
#define f3(x) for(x=0;;x++)
#define f4(x,r) for(x=r;x>0;x--)
int main()
{
    int p,q,i,x;
    vector<int> p1,p11(100010),ans;
    vector<int> q1,q11(100010);
    s(p);
    f1(i,p)
    {
        s(x);
        p1.push_back(x);
        p11[x]++;
    }
    s(q);
    f1(i,q)
    {
        s(x);
        q1.push_back(x);
        q11[x]++;
    }
    if(p>q)
    {
        for(i=0;i<p;i++)
        {
            if(p11[p1[i]]!=q11[p1[i]])
            {
                ans.push_back(p1[i]);
                p11[p1[i]]=0;
                q11[p1[i]]=0;
            }
        }
    }
    if(p<q)
    {
        for(i=0;i<q;i++)
        {
            if(p11[q1[i]]!=q11[q1[i]])
            {
                ans.push_back(q1[i]);
                p11[q1[i]]=0;
                q11[q1[i]]=0;
            }
        }
    }
    sort(ans.begin(),ans.end());
    for(i=0;i<ans.size();i++)
    {
        cout<<ans[i]<<" ";
    }
    cout<<endl;
}
