#include <iostream>
#include <algorithm>
#include <cstdlib>

using namespace std;

int main()
{
 int n;
 cin>>n;
 int a[n];
 
 for(int i=0;i<n;i++)
 {
 	cin>>a[i];
 }
 sort(a,a+n);
 ///Max first
 int ff=0,r=n-1,f=0,q[n],w[n],sq,sw;
 for(int i=0;i<n;i++)
 {
 	if(ff==0)
 	{
 		q[i]=a[r];
 		r--;
 		ff=1;
 		
 	}
 	else
 	{
 		q[i]=a[f];
 		f++;
 		ff=0;
 	}
 }
 sq=0;
 for(int i=1;i<n;i++)
 {
 	sq=sq+abs(q[i-1]-q[i]);
 }
 ///Min first
 ff=1,r=n-1,f=0;
 for(int i=0;i<n;i++)
 {
 	if(ff==0)
 	{
 		w[i]=a[r];
 		r--;
 		ff=1;
 		
 	}
 	else
 	{
 		w[i]=a[f];
 		f++;
 		ff=0;
 	}
 }
 sw=0;
 for(int i=1;i<n;i++)
 {
 	sw=sw+abs(w[i-1]-w[i]);
 }
 
 cout<<max(sw,sq);
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
    return 0;
}
