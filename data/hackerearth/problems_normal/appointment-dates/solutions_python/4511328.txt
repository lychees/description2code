from calendar import Calendar

t = int(raw_input())

for n in xrange(1, t+1):
        a = raw_input().split()


        out = Calendar().monthdayscalendar(int(a[1]), int(a[0]))
        output = ""
        for days in out:
                if days[6] != 0:
                        output += "%s " % days[6]
        print output
