#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
	int t;
	int n,k,inp;
	scanf("%d",&t);
	while(t--){
		long long max_so_far=0,curr_max=0;
		scanf("%d%d",&n,&k);
		for(int i=0;i<n;i++){
			scanf("%d",&inp);
			if(inp>k){
				curr_max=0;
			}
			else{
				curr_max+=inp;
				max_so_far=max(curr_max,max_so_far);
			}
		}
		printf("%lld\n",max_so_far);
	}
    return 0;
}
