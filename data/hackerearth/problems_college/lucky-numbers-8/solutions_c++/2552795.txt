#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int i,j,a[n];
    	int temp=0;
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	for(i=0;i<n;i++)
    	{
    		for(j=i+1;j<n;j++)
    		{
    			if (a[i]<=a[j])
    			break;
    		}
    		if(j==n)
    		temp++;
    	}
    	cout<<temp<<endl;
    }
    return 0;
}
