#include <iostream>
using namespace std;
#include<bits/stdc++.h>
int main()
{
	int t;
	cin>>t;
	while(t--){
		int n,m;
		cin>>n>>m;
		int k=m;
		while(n--){
			k=m;
			double val=10000000;
			int shop_no;
			for(int a=1;a<=k;a++){
				vector<int> v(3,0);
				double prod=1.0;
				for(int i=0;i<3;i++){
				cin>>v[i];
				prod*=(1-(double)v[i]/100.0);
				//cout<<prod<<"\t"<<val<<"\t";
				}
				if(prod<val){
					val=prod;
					//cout<<"aaya"<<a;
					shop_no=a;
				}
			}
			cout<<shop_no<<" ";
		}
		cout<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
