#include <iostream>
#include <string>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	int freq[26] = {0};
    	int n;
    	cin >> n;
    	string s;
    	cin >> s;
    	for(int i = 0; i < n; i++)
    	{
    		freq[s[i] - 'a']++;
    	}
    	int max = 0;
    	for(int i = 0; i < 26; i++)
    	{
    		if(max < freq[i])
    			max = freq[i];
    	}
    	if(max%2 == 0)
    		cout << "Yes" << endl;
    	else
    		cout << "No" << endl;	
    }
}
