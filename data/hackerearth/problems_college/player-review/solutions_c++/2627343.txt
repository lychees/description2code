#include <iostream>
#include <cstdio>
#include <algorithm>
#include <vector>

using namespace std;

int main()
{
    int n;
    scanf("%d", &n);
    int a[n];
    for (int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }
    int max_so_far = 0;
    int max_till_now = 0;
    for (int i = 0; i < n; i++) {
        max_till_now += a[i];
        if (max_till_now < 0) {
            max_till_now = 0;
        }
        if (max_till_now > max_so_far) {
            max_so_far = max_till_now;
        }
    }
    cout << max_so_far << endl;

    return 0;
}
