#include <bits/stdc++.h>
using namespace std;
bool issubset[10010][110];
int a[110];
int n,x;
int sum;
bool fun(int sum)
{
	for(int i=0;i<=n;i++)
	issubset[0][i]=1;
	for(int i=1;i<=sum;i++)
	issubset[i][0]=0;
	for(int i=1;i<=sum;i++)
	{
		for(int j=1;j<=n;j++)
		{
			issubset[i][j]=issubset[i][j-1];
			if(i>=a[j-1])
			issubset[i][j]=issubset[i][j]|(issubset[i-a[j-1]][j-1]);
			
		}
	}
	return issubset[sum][n];
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d",&n,&x);
		sum=0;
		for(int i=0;i<n;i++)
		{
			
		
		scanf("%d",&a[i]);
		sum=sum+a[i];
	}
	if(x>=sum)
	{
		printf("YES\n");
		continue;
	}
	if(x>=sum/2)
	{
		bool flag=fun(x);
		if(flag)
		{
			printf("YES\n");
			continue;
		}
		else
		{
			for(int i=x-1;i>=sum/2;i--)
			{
				if(issubset[i][n])
				{
					printf("YES\n");
					flag=1;
					break;
				}
			}
			if(flag)
			continue;
			else
			printf("NO\n");
		}
	}
	else
	printf("NO\n");
	}
	return 0;
}
