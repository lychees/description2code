#include <iostream>
using namespace std;
long long int fact[13] = {0};
void init()
{
	int i;
	fact[0] = 1;
	fact[1] = 1;
	for(i=2;i<13;i++)
	{
		fact[i] = i*fact[i-1];
	}
}
long long int compute(long long int n, long long int r)
{
	if(n == r)
		return 1;
	else if(r == 1)
		return n;
	else
		return compute(n-1,r) + compute(n-1,r-1);
}
int main()
{
	init();
    int t;
    cin>>t;
    while(t--)
    {
    	int a,b;
    	cin>>a>>b;
    	int temp = a + b;
    	a = compute(7,a);
    	b = compute(5,b);
    	cout<<(long long)a*b*fact[temp]<<endl;
    }
}
