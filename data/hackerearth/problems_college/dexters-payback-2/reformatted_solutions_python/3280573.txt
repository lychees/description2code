t = int(raw_input())
for e in range(t):
    s, x, = 0, 0
    w, h = map(int, raw_input().split())
    s, x = 2*(w//4), w % 4
    s1 = s+(x//3)
    if x == 1:
        s += 1
    elif x != 0:
        s += 2
    k, x = (h//4), h % 4
    n = 2*(s+s1)*k
    if x == 1:
        n += s
    elif x == 2:
        n += 2*s
    elif x == 3:
        n += 2*s+s1
    print n
