def print_results(P,N):
    S  = sum(P)
    print '%.8f' % (S / (N*1.0))
    

def test():
    T = int(raw_input())
    counter = 0
    while counter < T:
        N = int(raw_input())
        P = [int(x) for x in raw_input().split(' ')]
        print_results(P,N)
        counter += 1
test()
