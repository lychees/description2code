#include<bits/stdc++.h>
using namespace std;

int main()
{
    int l,max=0,min=0;
    cin>>l;
    string s;
    cin>>s;
    for(int i=0;i<s.length();i++)
    {
            int c=0;
            while(s[i]=='v' || s[i]=='w')
            {
                            
                            if(s[i]=='w')
                            {
                                         max+=1;
                                         c+=2;
                            }
                            else
                            {
                                c+=1;
                            }
                            i++;
            }
            if(c!=0)
            {
             min+=(c+1)/2;
             i-=1;
            }
            else
            min+=1;
    }
    cout<<min<<" "<<max+s.length();
}
