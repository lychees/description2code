#include <bits/stdc++.h>
using namespace std;

bool prime(int n) {
	if(n % 2 == 0) return 0;

	for(int i = 3; i * i <= n; i += 2) if(n % i == 0) return 0;

	return 1;
}

int main() {
	
	int k = 1;
	map<int, int > pfac;
	pfac[1] = 0;

	for(int i = 3; i < 1e4; i += 2) {
		long long a, b, c, d;
		a = i * i;
		b = a - (2 * k);
		c = a - (6 * k);
		d = b - (2 * k);
		k += 1;
		
		int cnt = 0;
		if(prime(a)) cnt++; 
		if(prime(b)) cnt++; 
		if(prime(c)) cnt++; 
		if(prime(d)) cnt++; 
		pfac[i] = pfac[i - 2] + cnt;
	}
	
	int q;
	cin >> q;

	while(q--) {
		int x;
		cin >> x;

		double total = (x/2) * 4 + 1;
		double prob = ((double) pfac[x] / total) * 100.0;
		printf("%.6f\n", prob);
	}

	return 0;
}
