#include <iostream>
#include <stdio.h>
using namespace std;
int main()
{
    int n;
    int i;
    

    scanf("%d",&n);

    for(i = 0; i < n; i++) {
        printf("%d ",i);
        printf("%d ",(2 * i) % n);
        printf("%d ",((2 * i) + 1) % n);
        printf("\n");

    }



}
