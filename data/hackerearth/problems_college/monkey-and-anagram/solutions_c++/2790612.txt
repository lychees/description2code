#include<bits/stdc++.h>
using namespace std;
int main()
{
    string s,t;
    int ar[30],br[30],i,test,n,ans;
    cin>>test;
    while(test--)
    {
        cin>>s>>t;
        for(i=1;i<=26;i++)
            ar[i]=br[i]=0;
        for(i=0;i<s.size();i++)
            ar[s[i]-'a']++;
        for(i=0;i<t.size();i++)
            br[t[i]-'a']++;
        bool cond=true;
        for(i=1;i<=26;i++)
        if(ar[i]>br[i])
        {
            cond=false;
            break;
        }
        int ans=0;
        if(cond)
            for(i=1;i<=26;i++)
            {
                if(br[i]>ar[i])
                    ans++;
            }
        else
            ans--;
        cout<<ans<<endl;
    }
}
