#include <stdio.h>
#include<string.h>
#include<stdlib.h>

int count,n;
char *str;

int fun(int &x,char ch)
{
	switch(ch)
    		{
    			case 'a':
    					x=x*2;
    					break;
    			case 'b':
    					x=x*3;
    					break;
    			case 'c':
    					x=x*5;
    					break;
    			case 'd':
    					x=x*7;
    					break;
    			case 'e':
    					x=x*11;
    					break;
    			case 'f':
    					x=x*13;
    					break;
    			case 'g':
    					x=x*17;
    					break;
    			case 'h':
    					x=x*19;
    					break;
    			case 'i':
    					x=x*23;
    					break;
    			case 'j':
    					x=x*29;
    					break;
    			case 'k':
    					x=x*31;
    					break;
    			case 'l':
    					x=x*37;
    					break;
    			case 'm':
    					x=x*41;
    					break;
    			case 'n':
    					x=x*43;
    					break;
    			case 'o':
    					x=x*47;
    					break;
    			case 'p':
    					x=x*53;
    					break;
    			case 'q':
    					x=x*59;
    					break;
    					
    			case 'r':
    					x=x*61;
    					break;
    					
    			case 's':
    					x=x*67;
    					break;
    			case 't':
    					x=x*71;
    					break;
    			case 'u':
    					x=x*73;
    					break;
    					
    			case 'v':
    					x=x*79;
    					break;
    					
    			case 'w':
    					x=x*83;
    					break;
    			case 'x':
    					x=x*89;
    					break;
    			case 'y':
    					x=x*91;
    					break;
    			case 'z':
    					x=x*97;
    					break;
    		}
    		return x;
}

void function(char input[1001],int i,int n,int deep)
{
	
	if(count>=4)
	return;
	int x=1;
	int k;
	for(k=0;k<3;k++)
	{
	
		if(i+k<=n && str[fun(x,input[i+k])]==0)
		{
			if(deep>count)
			count=deep;
			str[x]=1;
			function(input,i+1+k,n,deep+1);
			str[x]=0;
		}
	}
}


int main()
{
    
    str=(char*)malloc(1000000);
    char input[1001];
    scanf("%d",&n);
    while(n--)
    {
    	count=0;
    	bzero(str,1000000*sizeof(char));
    	bzero(input,1001*sizeof(char));
    	scanf("%s",input);
    	if(strlen(input)>=10)
    	{
    		printf("YES\n");
    		continue;
    	}
 
    	function(input,0,strlen(input)-1,1);
  
    	if(count>=4)
    	printf("YES\n");
    	else
    	printf("NO\n");
    }
    return 0;
}
