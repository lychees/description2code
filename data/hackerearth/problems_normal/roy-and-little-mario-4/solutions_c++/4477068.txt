#include<bits/stdc++.h>
using namespace std;
typedef long long int lli;
const lli  MOD=1000*1000*1000+7;
lli  dp[100*1000+9];
int main(){
    dp[1]=1LL;
    dp[2]=2LL;
    dp[3]=4LL;
    for(int i=4;i<=100*1000;++i){
        dp[i]=dp[i-1]+dp[i-2]+dp[i-3];
        dp[i]%=MOD;
    }
    int tc;
    cin>>tc;
    while(tc--){
        int n;
        cin>>n;
        printf("%lld\n",dp[n]);
    }


    return 0;
}
