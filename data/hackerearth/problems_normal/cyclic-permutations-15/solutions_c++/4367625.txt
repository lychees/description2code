int Z[MAX];

void Zfunction(string S){
	int L, R, N;
	N = S.length();
	L = R = 0;
	for (int i = 1; i < N; i++){
		if (i > R){
			L = R = i;
			while (R < N && S[R - L] == S[R])
				R++;
			Z[i] = R - L;
			R--;
		}
		else{
			int k = i - L;
			if (Z[k] < R - i + 1)
				Z[i] = Z[k];
			else{
				L = i;
				while (R < N && S[R - L] == S[R])
					R++;
				Z[i] = R - L;
				R--;
			}
		}
	}
}
