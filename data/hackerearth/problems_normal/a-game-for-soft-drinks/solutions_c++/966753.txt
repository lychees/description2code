    #include <iostream>
    using namespace std;
    #define MAX 101
    char mat[MAX][MAX][MAX];
    int ans = 0;
    int notans = 0;
    void pro(){
    for(int i=0;i<MAX;i++){
    for(int j=0;j<MAX;j++){
    for(int k=0;k<MAX;k++){
    if((i>0 and mat[i-1][j][k] == 'L') ||
    (j>0 and mat[i][j-1][k] == 'L' ) ||
    (k>0 and mat[i][j][k-1] == 'L') ||
    (j>0 and k>0 and mat[i][j-1][k-1] == 'L') ||
    (i>0 and k>0 and mat[i-1][j][k-1] == 'L' )||
    (i>0 and j>0 and mat[i-1][j-1][k] == 'L') ||
    (i>0 and j>0 and k>0 and mat[i-1][j-1][k-1] == 'L')){
    mat[i][j][k] = 'W';
    ans++;
    }
    else {
    mat[i][j][k] = 'L';
    notans++;
    }
    }
    }
    }
    }
    int main()
    {
    pro();
    int q;
    cin >> q;
    while(q--){
    int a,b,c;
    cin >> a >> b >> c;
    if(mat[a][b][c] == 'W'){
    cout << "Ashima" << endl;
    }
    else cout << "Aishwarya" << endl;
    }
    return 0;
    }
