

    #include<bits/stdc++.h>
    using namespace std;
    int arr[1001000]={0};
    int prime[1001000]={0};
    int ans[1001000]={0};
    int main(){
    	prime[0] = prime[1] = 1;
    	for(int i=2; i<=1000000; i++){
    		if(!prime[i]){
    			for(int j=2; j*i <= 1000000; j++)
    				prime[j*i] = 1;
    		}	
    	}
    	for(int i=1; i<=1000000; i++){
    		arr[i] = arr[i-1];
    		if(!prime[i])arr[i]++;
    	}		
    	for(int i=1; i<=1000000; i++){
    		ans[i] = ans[i-1];
    		if(!prime[arr[i]])
    			ans[i]++;
    	}
    	int t;cin>>t;
    	while(t--){
    		int l, r;
    		cin>>l>>r;
    		cout<<(ans[r]-ans[l-1])<<endl;
    	}
    	return 0;
    }
