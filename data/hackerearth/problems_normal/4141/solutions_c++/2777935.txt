#include <iostream>
using namespace std;
 
int main()
{
    int divBy41[] = {1, 10, 18, 16, 37}; //reocurrence after 5
    int t, a[50005], c, n, i, j, sum;
    cin >> t;
    
    while(t--)
    {
    	sum = 0;
    	
    	cin >> a[0] >> a[1] >> c >> n;
    	for (i = 2; i <= n - 1; i++)
    	{
    		a[i] = (a[i-1]*c + a[i-2]) % 10;    		
    		sum += a[i] * divBy41[(n - 1 - i) % 5];
    	}
    	
    	for (i = 0; i <= 1 && i <= n - 1; i++)
    	{
    		sum += a[i] * divBy41[(n - 1 - i) % 5];
    	}
    	
    	cout << ((sum % 41 == 0) ? "YES" : "NO");
    	cout << "\n";
    }
    return 0;
}
