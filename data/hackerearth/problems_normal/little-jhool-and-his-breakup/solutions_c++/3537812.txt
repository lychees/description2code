#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
    string s;
    cin>>s;
    int i,j,k,m;
    int len=0;
    for( i=0;i<s.size();i++)
    {
    	if(s[i]=='l')
    	{
    		len++;
    		break;
    	}
    }
    for(j=i;j<s.size();j++)
    {
    	if(s[j]=='o')
    	{
    		len++;
    		break;
    	}
    }
    for(k=j;k<s.size();k++)
    {
    	if(s[k]=='v')
    	{
    		len++;
    		break;
    		
    	}
    }
    for(m=k;m<s.size();m++)
    {
    	if(s[m]=='e')
    	{
    		len++;
    		break;
    		
    	}
    }
    if(len==4)
    {
    	cout<<"I love you, too!";
    }
    else
    cout<<"Let us breakup!";
    return 0;
}
