#include <iostream>
using namespace std;
 
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,k;
    	cin>>n>>k;
    	int a[n],i,count=0;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    		if(a[i]<=0)
    		   count++;   	
    	}
    	 if(count>=k)
    	 printf("NO\n");
    	 else printf("YES\n");
    }
    return 0;
}
