#include <bits/stdc++.h>
using namespace std;

int find(int arr[], long n) {
   int i, j = 0;
   if(n%2 == 0)
       arr[j++] = 2;
   while(!(n%2)) 
      n /= 2;
   for(i = 3 ; i <= n ; i += 2) {
      if(n%i == 0) {
         while(!(n%i))
            n /= i;
         arr[j++] = i;
      }
   }
   if(n > 1)
      arr[j++] = n;
   return j;    
}
 
int main() {
   int t;
   long x, m, i;
   scanf("%d", &t);
   while(t--) {
      scanf("%ld %ld", &x, &m);
      int fac[1000000], size; 
      size = find(fac, m);
      for(i = 0 ; i < size ; i++)
         if(x%fac[i])
            break;
      if(i == size)
         printf("YES\n");
      else
         printf("NO\n");    
   } 
}
