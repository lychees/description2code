#include <bits/stdc++.h>
using namespace std;
int main()
{
    map<string,string>A,B;
    int n, m, i; 
	scanf("%d%d", &n, &m);
    for(i=0;i<m;i++)
	{
        string str1,str2;
        cin>>str1>>str2;
        A[str1]=str2;
        B[str2]=str1;
    }
    for(i=0;i<n;i++)
	{
        if(i)
		printf(" ");
        string str;
        cin>>str;
        if(A[str].size()<str.size())
		cout<<A[str];
        else 
		cout<<str;
    }
    cout<<endl;
    return 0;
}
