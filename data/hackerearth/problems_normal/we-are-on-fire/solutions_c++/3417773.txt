#include <bits/stdc++.h>
using namespace std;
     
     
    int func(int mat[1001][1001],int n , int m, int x, int y,int *cnt){
    if(mat[x-1][y-1]==0)return *cnt;
    else{
    int i=x-1,j=y-1;
    mat[i][j]=0;
    (*cnt)--;
    if(i-1>=0){
    if(mat[i-1][j]==1)
    {
     func(mat,n,m,i-1+1,j+1,cnt);
    }
    }
    if(i+1<=n-1){
    if(mat[i+1][j]==1){
     func(mat,n,m,i+1+1,j+1,cnt);
    }
    }
    if(j-1>=0){
    if(mat[i][j-1]==1){
     func(mat,n,m,i+1,j-1+1,cnt);
    }
    }
    if(j+1<=m-1){
    if(mat[i][j+1]==1){
    func(mat,n,m,i+1,j+1+1,cnt);
    }
    }
    }
    return *cnt;
    }
     
    int mat[1001][1001];
    int main()
    {
    int n,m, q;
    scanf("%d",&n);
    scanf("%d",&m);
    scanf("%d",&q);
    int i,j;
  
    int cnt=0;
    for(i=0;i<n;i++){
    for(j=0;j<m;j++){
    scanf("%d",&mat[i][j]);
    if(mat[i][j]==1)cnt++;
    }
    }
   
    while(q--){
    int x,y;
  
    scanf("%d",&x);
    scanf("%d",&y);
     
    int reslt=func(mat,n,m,x,y,&cnt);
    printf("%d\n",reslt);
    }
   
    return 0;
    }
