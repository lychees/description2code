#include <iostream>
#include <string>
using namespace std;

int main()
{
    int t;cin>>t;
    while(t--){
    	int cnt[26]={0};
    	string s;cin>>s;
    	int i,n=s.length();
    	for(i=0;i<n;i++){
    		cnt[s[i]-'a']++;
    	}
    	int distinctchar=0;
    	for(i=0;i<26;i++){
    		if(cnt[i]>0)
    			distinctchar++;
    	}
    	bool ans=false;
    	if(distinctchar>=4)			ans=true;
    	else if(distinctchar==3){
    		if(n>=5)	ans=true;
    		else		ans=false;
    	}	
    	else if(distinctchar==2){
    		int temp1=0,temp2=0;
    		for(i=0;i<26;i++){
    			if(cnt[i]>0&&!temp1)	temp1=cnt[i];
    			else if(cnt[i]>0)		temp2=cnt[i];
    		}
    		
    		if(temp1>=3&&temp2>=3)								ans=true;
    		else if(temp1>=1&&temp2>=6||(temp1>=6&&temp2>=1))	ans=true;
    		else												ans=false;
    	}
    	else if(distinctchar==1){
    		if(n>=10)	ans=true;
    		else		ans=false;
    	}
    	if(ans)	cout<<"YES"<<endl;
    	else	cout<<"NO"<<endl;
   }
    return 0;
}
