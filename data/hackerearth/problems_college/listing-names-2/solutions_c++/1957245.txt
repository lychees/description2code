#include<iostream>
#include<map>
#include<string>
using namespace std;
map<string,long>mymap;
int main()
{
   // cout << "Hello World!" << endl;
   int T;
   cin>>T;
   
   string str;
   while(T--)
   {
   	cin>>str;
   	long a=mymap.find(str)->second;
  // 	cout<<"a="<<a<<endl;
   	mymap[str]=a+1;
   	
   	
   	
   }
   
   for(std::map<string,long>::iterator it=mymap.begin();it!=mymap.end();++it)
    cout<<it->first<<" "<<it->second<<endl;
   
    return 0;
}
