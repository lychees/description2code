#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
long long int n,q,k,s,i,j;
cin>>n>>q;
vector<pair<int,int>> a;
long long int b[n+1];
for(i=0;i<n;i++)
{
	cin>>b[i];
}

for(i=0;i<n;i++)
{
	cin>>s;
	a.push_back(make_pair(s,b[i]));
}

sort(a.begin(),a.end());
memset(b,0,sizeof(b));

b[0]=a[n-1].second;
for(i=1;i<n;i++)
{
	b[i]=b[i-1]+a[n-i-1].second;
}

//cout<<"\n\n";
while(q--)
{
	cin>>k;
	cout<<b[k-1]<<"\n";
}

    return 0;
}
