#include<bits/stdc++.h>
using namespace std;
int main()
{
	ios::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		long long a[100000],i,n;
		cin>>n;
		for(i=0;i<n;i++)
		    cin>>a[n-i-1];
	    long long howmuch[100000][2]={0};
	    howmuch[0][0]=a[0];
	    howmuch[0][1]=0;
	    howmuch[1][0]=a[0]+a[1];
	    howmuch[1][1]=0;
	    for(i=2;i<n;i++)
	    {
	        if(howmuch[i-1][1]>howmuch[i-2][1]+a[i-1])
	        {
	            howmuch[i][0]=a[i]+howmuch[i-1][1];
	            howmuch[i][1]=howmuch[i-1][0];
	        }
	        else
	        {
	            howmuch[i][0]=a[i]+howmuch[i-2][1]+a[i-1];
	            howmuch[i][1]=howmuch[i-2][0];
	        }
	     //   cout<<i<<" "<<howmuch[i][0]<<" "<<howmuch[i][1]<<endl;
	    }
	    if(howmuch[n-1][0]==howmuch[n-1][1])
	        cout<<"DRAW";
	    else if(howmuch[n-1][0]<howmuch[n-1][1])
	        cout<<"WIN";
	    else
	        cout<<"LOSE";
	    cout<<endl;
	}
	return 0;
}
