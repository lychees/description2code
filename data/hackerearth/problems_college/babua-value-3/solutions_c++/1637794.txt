#include <iostream>
#include <algorithm>
#include <cstdio>
#include <set>
using namespace std;

long long gcd( long long a, long long b ) {
    if( !a || !b )
        return a+b ;
    return gcd(b,a%b) ;
}

long long n, a[200000], s[200000], sum, fact, g, i ;

int main()
{
    cin >> n ;
    for(i=1;i<=n;++i)
        cin >> a[i] ;
    sort(a+1,a+1+n) ;
    for(i=1;i<=n;++i) {
        s[i] = s[i-1] + a[i] ;
        sum += (i-1)*a[i]-s[i-1] ;
    }
    sum = s[n] + 2*sum ;
    fact = n ;
    g = gcd(fact,sum) ;
    cout << sum/g << "/" << fact/g << endl ;
}
