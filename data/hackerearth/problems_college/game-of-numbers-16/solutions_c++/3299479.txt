#include <iostream>
using namespace std;

int main()
{
    long long m,n,Q;
    int T,count;
    
    cin>>T;
    
   while(T--)
    {
    	count=0;
    	cin>>Q;
    	
    	for(m=1; m<=Q; m++)
    	{
    		for(n=m+1;n<=Q;n++)
          	{
    	       if((m*n)%(m+n) == 0)
    		   count++;
    	    }
    	}
    	
    	cout<<count<<endl;
    }
    return 0;
}
