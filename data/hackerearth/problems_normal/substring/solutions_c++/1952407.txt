#include<bits/stdc++.h>
using namespace std;
int main()
{
    int T,ans,c;
    string S;
    cin>>T;
    while(T--)
    {
        ans=c=0;
        cin>>S;
        int l=S.length();
        int i=0;
        while(i<l)
        {
            while(S[i]=='A' || S[i]=='B' || S[i]=='C' || S[i]=='a' || S[i]=='b' || S[i]=='c')
            {
                c++;
                i++;
            }
            ans+=(c*(c+1))/2;
            c=0;
            i++;
        }
        cout<<ans<<"\n";
    }
}
