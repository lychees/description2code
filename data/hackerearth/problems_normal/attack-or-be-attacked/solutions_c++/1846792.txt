const int mod = 1000000007;

ll a2(ll x) {
	return x*(x-1)%mod;
}

ll f3(ll x) {
	if (x % 3 == 0) {
		return x/3*a2(x-1)%mod;
	} else if (x%3 == 1) {
		return x*(x-1)/3%mod*(x-2)%mod;
	} else {
		return x*(x-2)/3%mod*(x-1)%mod;
	}
}

int main() {
    ll n,m,p,q;
    cin >> n >> m >> p >> q;
    if (n > m) swap(n,m);
    if (p > q) swap(p,q);
    ll s = n*m%mod;
    ll res = a2(s) - 2*((m-n+1)*a2(n) + 2*f3(n)) - 4*(max(0LL, n-p)*max(0LL, m-q) + max(0LL, n-q)*max(0LL, m-p));
    cout << (res%mod + mod) % mod << endl;
    return 0;
}
