#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t,n,i,j,ans;
	scanf("%d",&t);
	while (t --) {
		scanf("%d",&n);
		int a[n];
		for (i = 0; i < n; i++) {
			scanf("%d",&a[i]);
		}
		sort(a,a+n);
		for (i = 0; i < n; i++) {
			for (j = i+1; j < n; j++) {
				if (a[i] < a[j]) {
					a[i] = -1;
					break;
				}
			}
		}
		ans = 0;
		for (i = 0; i < n; i++) {
			if (a[i] != -1) {
				ans++;
			}
		}
		printf("%d\n",ans);
	}
	
	return 0;
}
