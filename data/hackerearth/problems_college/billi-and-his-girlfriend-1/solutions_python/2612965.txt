def fast_pow(a,b):
    if b==0:
        return 1
    tmp=fast_pow(a,b/2)
    if b&1:
        return a*tmp*tmp%1000000000
    else:
        return tmp*tmp%1000000000

def sumDigits(n):
    s=0
    while n>0:
        s+=n%10
        n=int(n/10)
    return s

for i in xrange(input()):
    a,b,k=map(int,raw_input().split())
    print sumDigits(fast_pow(a,b)%(10**k))
