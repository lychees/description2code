#include <iostream>
#include <algorithm>

int main() 
{
    std::string str;
    //std::string fin;
    char fin[100000];
    char a;
    int j = 0, m = 0, n = 0;
    bool found = false;
    
    std::cin >> str;
    std::sort(str.begin(), str.end());
    
    fin[str.length()] = '\0';
    n = str.length() - 1;
    
    for(int i = 0; i < str.length(); i++)
    {
        if(str[i] != '0')
            a = str[i];
        else
            continue;
        
        j = i + 1;
        while(j < str.length())
        {
            if(str[j] == a)
            {
                found = true;
                str[j] = '0';
                break;
            }
            else
            {
                j++;
            }
        }
        if(found && m < n)
        {
            fin[m] = a;
            fin[n] = a;
            m++;
            n--;
            found = false;
        }
        else if(!found)
        {
            int l = str.length();
            fin[l / 2] = a;
        }
    }
    
    int z = str.length();
    
    for(int i = 0, j = z - 1; i < z / 2 && j > z / 2 - 1;  i++, j--)
    {
        if(fin[i] != fin[j])
        {
            std::cout << -1;
            return 0;
        }
    }
    
    std::cout << fin;
}
