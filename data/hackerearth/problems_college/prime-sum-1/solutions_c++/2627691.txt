#include <iostream>
#define ll long long
using namespace std;

int main()
{
    ll t,n,total,temp;
    cin>>t;
    while(t--)
    {
    	total=0;
    	
    	cin>>n;
    	if(n==1)
    	cout<<"-1\n";
    	else
    	{
    		temp=n%7;
    		total+=(n/7);
    		n=temp%5;
    		total+=(temp/5);
    		temp=n%3;
    		total+=(n/3);
    		n=temp%2;
    		total+=((temp/2)+n);
    		
    		cout<<total<<"\n";
    	}
    	
    }
}
