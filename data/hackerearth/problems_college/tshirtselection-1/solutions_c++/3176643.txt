using namespace std ;

int main()
{
	int i,t,n,q,a[100005];

	cin>>t;
	while(t--){
		cin>>n;
		rep(i,n)
		cin>>a[i];
		int l=0,r=0;
		map<int,int> hash;
		
		int idx_l=0,idx_r=0;
		rep(i,n){
			if(hash.find(a[r])!=hash.end()){
				if((r-l) > (idx_r-idx_l)){
					idx_r=r;
					idx_l=l;
				}
				l=max(l,hash[a[r]]+1);
			}
			hash[a[r]] = r;
			r++;
		}
		if(((r-l) > (idx_r-idx_l))){
			idx_r=r;
			idx_l=l;
		}
		cout<<idx_l+1<<" "<<idx_r<<endl;
	}
	
	
return 0;
}
