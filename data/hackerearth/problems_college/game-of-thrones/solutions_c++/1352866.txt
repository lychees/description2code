#include <bits/stdc++.h>
 
using namespace std;

 
long long int arr[100006];
 
int main() {
    int t;
    cin>>t;
    while(t--) {
        long long int i,n,num,cnt=1;
        scanf("%lld", &n);
        for(i=0;i<n;i++) {
            scanf("%lld", &arr[i]);
        }
        sort(arr,arr+n);
        for(i=1;i<n;i++) {
            if(arr[i-1]^arr[i])
                cnt++;
        }
        printf("%lld\n", n-cnt);
    }
    return 0;
}
