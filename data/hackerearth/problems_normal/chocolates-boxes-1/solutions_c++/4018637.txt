#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int sz = 100010;
map<int, int> h;
int arr[sz];
int main() {
	int t, n, k;
	ll res = 0;
	cin >> t;
	for (int i = 0; i < t; ++i) {
		cin >> n;
		res = 0;
		h.clear();
		for (int i = 0; i < n; ++i) {
			cin >> arr[i];
		}
		cin >> k;
		for (int i = 0; i < n; ++i) {
			if (h.find(k - arr[i]) != h.end()) {
				res += h[k - arr[i]];
			}
			++h[arr[i]];
		}
		cout << res << endl;
	}
	return 0;
}
