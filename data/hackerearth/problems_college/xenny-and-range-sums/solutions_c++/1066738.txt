#include <iostream>
using namespace std;

#include <stdio.h>
int a[1000000],b[1000000];
int main()
{
	int n,q,i,l,r;
   cin>>n>>q;
   b[0]=0;
   for(i=0;i<n;i++)
   {
   	cin>>a[i];
   	b[i+1]=a[i];
   }
    for(i=1;i<=n;i++)
    {
    	b[i]+=b[i-1];
    }
   while(q--)
   {
   	cin>>l>>r;
   	cout<<b[r]-b[l-1]<<" ";
   	cout<<a[r-1]-b[r-1]+b[l-1]<<endl;   }
   }
