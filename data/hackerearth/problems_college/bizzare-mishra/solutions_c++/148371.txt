#include<stdio.h>
#include<string.h>
#include<iostream>
using namespace std;
#define max 10004
int main()
{
	char str[max],str2[max],str1[max];
	int i,ans,c1,c2,n,t;
	scanf("%d",&t);
	while(t--)
	{
		c1=0;c2=0;
		scanf("%d",&n);
		scanf("%s",str);
		for(i=0;i<n;i+=2)
		{
			str1[i]='Y';
			str2[i]='N';
		}
		for(i=1;i<n;i+=2)
		{
			str2[i]='Y';
			str1[i]='N';
		}
		for(i=0;i<n;i++)
		{
			if(str[i]!=str1[i])
				c1++;
			if(str[i]!=str2[i])
				c2++;
		}
		if(c1>c2)
			ans=c2;
		else
			ans=c1;
	printf("%d\n",ans);
	}
return 0;
}
