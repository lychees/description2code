const int MOD = 1e9+7;

#define LOCALTEST 0 // change this to 1 to perform localtest on "in.txt"
int n,m;
map<pair<int,vector<int> >,int > z;
vector<pair<int,PII> > r;
vector<int> tm;
int rec(int i, vector<int> t) {
    if (i<0) return 0;
    pair<int,vector<int> > p=MP(i,t);
    if (z.count(p)) return z[p];
 
    int &ret=z[p]=rec(i-1,t);
    bool ok=true;
    int s=r[i].F,e=r[i].S.F,v=r[i].S.S;
    REPP(i,s,e) {
        if (t[i]<=0) ok=false;
    }
    if (ok) {
        REPP(i,s,e) {
            t[i]--;
        }
        smax(ret,rec(i-1,t)+v);
    }
    return ret;
}
int main(){
    if (LOCALTEST) {
        freopen("in.txt", "r", stdin);
        freopen("out.txt", "w", stdout);
    }
    CASET {
        RII(n,m);
        r.clear();
        r.resize(n);
        REP(i,n) RIII(r[i].F,r[i].S.F,r[i].S.S);
        ::tm.clear();
        ::tm.resize(m);
        REP(i,m) RI(::tm[i]);
        z.clear();
        printf("%d\n",rec(n-1,::tm));
    }
    return 0;
}
