T = int(raw_input().strip())
while T>0:
	line = raw_input().strip()
	PointList = map(int, line.split(" "))
	Px = PointList[0]
	Py = PointList[1]
	Bx = PointList[2]
	By = PointList[3]
	
	if (Px>=Bx and Py >=By) or (Py>=Bx and Px >=By):
		print "Possible"
	else:
		print "Not Possible"
	
	T = T-1
