//upsolving
#include<bits/stdc++.h>
using namespace std;
int swap(int &a,int &b)
{
	int temp=a;
	a=b;
	b=temp;
}
	
int gcd(int a,int b)
{
	if(b>a)
	swap(a,b);
	if(b==0)
	return a;
	else if(b==1)
	return 1;
	else
	return gcd(a%b,b);
}

int main()
{ 
	//precomputation
	int i,j,k,arr[105];
	memset(arr,0,sizeof(arr));
	for(i=1;i<=100;i++)
	{
		for(j=1;j<=100;j++)
		{
			for(k=1;k<=100;k++)
			if(i!=j &&j!=k &&i!=k)
			arr[gcd(gcd(i,j),k)]++;
		}
	}
	double ways=100*99*98;
	int t;
scanf("%d",&t);
	while(t--)
	{ long long x,z;
		scanf("%lld %lld ",&x,&z);
		if(z>x)
		swap(x,z);
		long long dif=(x-z);
		double prob=1.0;
		if(x==z)
		{
			printf("%lf\n",prob);
			continue;
		}
		long long ans=0;
		for(i=1;i<=100;i++)
		{
			if(dif%i==0)
			ans+=arr[i];
		}
		
		printf("%lf\n",ans/ways);
	}
	return 0;
}
