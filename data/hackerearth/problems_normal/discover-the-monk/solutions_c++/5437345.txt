#include <cstdio>
#include <algorithm>
#define l long int

using namespace std;

l arry[100005];

int compare (const void * a, const void * b)
{
  return ( *(int*)a - *(int*)b );
}

bool binary_search(l start,l end,l key)
{
	while(start<=end)
	{
		l mid = (start+end)/2;

		if(arry[mid] == key)
			return true;

		else if(arry[mid]>key)
			end = mid-1;
		else
			start=mid+1;
	}
	return false;
}

int main()
{
	l N,Q;
	scanf("%ld",&N);
	scanf("%ld",&Q);
	
	for(l i=0;i<N;i++)
		scanf("%ld",&arry[i]);

	qsort(arry,N,sizeof(l),compare);
	for(l i=0;i<Q;i++)
	{
		l tmp;
		scanf("%ld",&tmp);
		bool result = binary_search(0,N-1,tmp);

		result?printf("YES\n"):printf("NO\n");
	}
	return 0;
}
