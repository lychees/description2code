
#include <bits/stdc++.h>
using namespace std;
 
int main()
{
    #ifdef VSP4
    freopen("input.txt", "r", stdin);
    freopen("output.txt", "w", stdout);
    #endif // VSP4
 
	int n, k, i, j, calc, ans;
 
	cin >> n >> k;
 
    ans = 1e9;
 
    for (j = 2*k; j >= 1; j--)
    {
        i = n - (n % j);
        //cout << i << " " << j << " " << n % j << "\n";
        ans = min(ans, abs(n - i) + abs(k - j));
        i += j;
        //cout << i << " " << j << "\n";
        ans = min(ans, abs(n - i) + abs(k - j));
    }
 
    cout << ans;
 
    return 0;
}
