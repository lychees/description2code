N=int(raw_input())
S=map(int,raw_input().split())
F=map(int,raw_input().split())
Fuel=input()
L=[]
for i in xrange(N):
    L.append(S[i]*Fuel*1.0/F[i])
#print L

Str=str(max(L))
dot=Str.find('.')
#print Str
new_str=Str[dot+1:]

len_new_str=len(new_str)
if len_new_str> 3:
	print Str[:dot+4]
else:
	print Str[:dot+1] + new_str +( '0'* (3-len_new_str))
