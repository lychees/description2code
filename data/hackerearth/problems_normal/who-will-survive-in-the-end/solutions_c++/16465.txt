#include <iostream>
#include <cmath>

using namespace std;

int main()
{
	int tc;
	cin >> tc;

	long double n;

	while (tc--)
	{
		cin >> n;

		long long x = log(n)/log(double(2));
		long long y = pow((long double)(2), (long double) (x));
		long long result = (long long)(2) * (n-y) + (long long)(1);

		cout << result << endl;
	}

}
