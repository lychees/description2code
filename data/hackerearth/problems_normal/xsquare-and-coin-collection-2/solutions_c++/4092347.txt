#include<iostream>
#include<algorithm>

using namespace std;

int main(){

    int t;
    cin >> t;
    while(t--) {
        long long n,i,ans=0,a[100005],k,cnt=0,x=0;
        cin >> n >> k;
        for(i=1;i<=n;i++) cin >> a[i];
        for(i=1;i<=n;i++) {
            if(a[i] <= k) cnt++, x += a[i];
            else {
                ans = max(ans,x); cnt = 0; x = 0;
            }
        }
        ans = max(ans,x);
        cout << ans << "\n";
    }
    return 0;






}
