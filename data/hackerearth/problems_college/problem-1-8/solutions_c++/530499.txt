#include <bits/stdc++.h>
using namespace std;

int main(){
    string str;
	cin>>str;
	set<string> st;
	int n,i,j,len;
	len=str.size();
	cin>>n;
	string s[n];
	for(i=0;i<n;i++){
		cin>>s[i];
		if(s[i].size()>=len){
			for(j=0;j<len;j++){
				if(s[i][j]!=str[j])
				   break;
				else if(j==len-1){
					st.insert(s[i]);
				}   
			}
		}
	}	
	if(!st.empty()){
		cout<<*(st.begin());
	}
	else{
		cout<<str;
	}
	return 0;
}
