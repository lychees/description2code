#define FOR(i,n) for(i=0;i<n;i++)
#define FORS(i,j,n) for(i=j;i<n;i++)
#define FORA(i,n) for(i=1;i<=n;i++)

using namespace std;
typedef long long LL;

int main(){
	int n,m,i,j,mid;
	cin>>n>>m;
	int cycle[m+1];

	if(m&1)
	{
		mid=(m+1)/2;
		j=1;
		int tmp=2;
		for(i=mid-1;i>0;i--,tmp+=2)
			cycle[tmp]=i;
		tmp=1;
		for(i=mid;i<=m;i++,tmp+=2)
			cycle[tmp]=i;

	}
	else
	{
		mid=m/2;
		j=1;
		int tmp=1;
		for(i=mid;i>0;i--,tmp+=2)
			cycle[tmp]=i;
		tmp=2;
		for(i=mid+1;i<=m;i++,tmp+=2)
			cycle[tmp]=i;

	}
	for(i=1;i<=n;i++)
		if(i%m)
			cout<<cycle[i%m]<<"\n";
		else
			cout<<m<<endl;
	
	return 0;
}
