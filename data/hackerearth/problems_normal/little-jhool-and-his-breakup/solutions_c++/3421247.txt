#include<iostream>
#include<algorithm>
using namespace std;
main()
{
    string str, lv = "love", tmp="";
    cin>>str;
    int ctr = 0;
    for(int x=0; x<lv.size(); x++)
    {
        for(ctr; ctr<str.size(); ctr++)
        {
            if(lv[x] == str[ctr]) {tmp += str[ctr]; break;}
        }
    }
    if(tmp == "love") cout<<"I love you, too!";
    else cout<<"Let us breakup!";
}
