#include <bits/stdc++.h>
using namespace std;

typedef long long int ll;

int main()
{
   	ios_base::sync_with_stdio(false);
   	ll a,b,t,n,i;
   	cin>>t;
   	while(t--)
   	{
   		cin>>n;
   		cin>>a;
   		for(i=1;i<n;i++)
   		{
   			cin>>b;
   			a = __gcd(a,b);
   		}
   		cout<<a*n<<endl;
   	}
    return 0;
}
