#include<bits/stdc++.h>
using namespace std;

struct node{
	int c=0;
};

int main()
{
	int t,n;
	scanf("%d",&t);
	for(;t>0;t--)
	{
		int n;
		scanf("%d",&n);
		
		int e=0,o=1,c=0;
		while(n!=1)
		{
			if(n%2==0)
			{
				e+=n;
				n=n/2;
			}
			else
			{
				o+=n;
				n=3*n+1;
			}
			c++;
		}
		
		if(e%c>o%c)
		printf("Even Rules\n");
		else if(e%c<o%c)
		printf("Odd Rules\n");
		else
		printf("Tie\n");
	}
	
	return 0;
}
