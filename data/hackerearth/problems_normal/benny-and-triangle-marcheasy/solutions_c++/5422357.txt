using namespace std;
 
const double eps = 1e-8;
const double kek = 1e-6;
 
int main() {
	double a, b, p;
	cin >> a >> b >> p;
	p /= 100.0;
	double x = 0.0;
	double y = a;
	double xx = b;
	double yy = 0.0;
	double k = (y - yy) / (x - xx);
	double B = y - k * x;
	cout << fixed << setprecision(2);
	double le = 0.0 + eps;
	double ri = b - eps;
	int iter = 228;
	double my_area = a * b * 0.5 * p;
	while (iter--) {
		double mid = (le + ri) / 2.0;
		double y_inter = k * mid + B;
		double rem_x = b - mid;
		double area = y_inter * rem_x * 0.5;
		if (area + kek <= my_area) {
			ri = mid;
		} else le = mid;
	}
	cout << le << endl;
	return 0;
}
