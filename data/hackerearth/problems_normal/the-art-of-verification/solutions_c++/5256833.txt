#include <iostream>
using namespace std;

int main()
{
    string link;
    string u = "username";
    string p = "pwd";
    string pr = "profile";
    string k = "key";
    string ro = "role";
    
    cin >> link;
    int idx1 = link.find(u) + 9;
    int idx2 = link.find(p);
    
    cout <<"username: ";
    for(int x = idx1; x < idx2-1; x++)
    {
    	cout <<link[x];
    }

    idx1 = idx2 + 4;
    idx2 = link.find(pr);
    
    cout <<"\npwd: ";
    for(int x = idx1; x < idx2-1; x++)
    {
    	cout <<link[x];
    }
	
	idx1 = idx2 + 8;
    idx2 = link.find(ro);
    
    cout <<"\nprofile: ";
    for(int x = idx1; x < idx2-1; x++)
    {
    	cout <<link[x];
    }
    
    idx1 = idx2 + 5;
    idx2 = link.find(k);
    
    cout <<"\nrole: ";
    for(int x = idx1; x < idx2-1; x++)
    {
    	cout <<link[x];
    }
    
    idx1 = idx2 + 4;
    cout <<"\nkey: ";
    for(int x = idx1; x < link.length(); x++)
    {
    	cout <<link[x];
    }
    return 0;
}
