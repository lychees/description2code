#include <iostream>
using namespace std;
int readInt () {
	bool minus = false;
	int result = 0;
	char ch;
	ch = getchar();
	while (true) {
		if (ch == '-') break;
		if (ch >= '0' && ch <= '9') break;
		ch = getchar();
	}
	if (ch == '-') minus = true; else result = ch-'0';
	while (true) {
		ch = getchar();
		if (ch < '0' || ch > '9') break;
		result = result*10 + (ch - '0');
	}
	if (minus)
		return -result;
	else
		return result;
}
typedef unsigned  long long ll;
ll gcd(ll a, ll b)
{
	while (a != b)
	{

        if (a > b)
        {
        	return gcd(a - b, b);
            }

        else

        {

            return gcd(a, b - a);

        }

    }

    return a;

}
int main()
{
    //cout << "Hello World!" << endl;
    ll a,n;
    a=readInt();
    n=readInt();
    ll f[n];
    ll count=0;
    for(ll i=0;i<n;i++)
    {
    	f[i]=readInt();
    	if(gcd(a,f[i])!=1)count++;
   
   }
    cout<<count<<"\n";
    return 0;
}
