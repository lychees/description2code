#include <bits/stdc++.h>
using namespace std;
int m,n;
const int mod=1e9+7;
long long int ctr;
long long dp[1010][1010];
int main()
{
    ios_base::sync_with_stdio(0);
    int t,i,j;
    cin>>t;
    while(t--)
    {
    	cin>>n>>m;
    	ctr=0;
    	memset(dp,0,sizeof(dp));
    	dp[0][0]=1;
    	for(i=1;i<=n;i++)
    	{
    		dp[i][0]=1;
    		dp[0][i]=0;
    	}
    	for(i=1;i<=n;i++)
    	{
    		for(j=1;j<=n;j++)
    		{
    			if(i>=j*m)
    			dp[i][j]=(dp[i-1][j]%mod+dp[i][j-1]%mod)%mod;
    		}
    	}
    	for(i=0;i<=n&&(n-i)>=i*m;i++)
    	{
    		if(n-i>i)
    		ctr=(ctr%mod+dp[n-i][i]%mod)%mod;
    	}
    	cout<<ctr<<"\n";
    }
    return 0;
}
