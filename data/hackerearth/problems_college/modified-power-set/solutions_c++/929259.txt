#include<iostream>
#include<cstring>

using namespace std;

int main(){
int t,count,i;
cin>>t;
char a[1000010];
while(t--){
    int occur1[26]={0},occur2[26]={0},occur3[10]={0},count=0;
    cin>>a;
    for(i=0;i<strlen(a);i++){
            if(a[i]>='a'&&a[i]<='z'){
                if(occur1[a[i]-'a']==0){
                    count++;
                    occur1[a[i]-'a']=1;
                }
            }
            else if(a[i]>='A'&&a[i]<='Z'){
                 if(occur2[a[i]-'A']==0){
                    count++;
                    occur2[a[i]-'A']=1;
                }
            }
            else if(a[i]>='0'&&a[i]<='9'){
                if(occur3[a[i]-'0']==0){
                    count++;
                    occur3[a[i]-'0']=1;
                }
            }
    }
    /*for(i=0;i<26;i++) cout<<occur1[i]<<" ";
    cout<<"\n";
    for(i=0;i<26;i++) cout<<occur2[i]<<" ";
    cout<<"\n";
    for(i=0;i<10;i++) cout<<occur3[i]<<" ";
    cout<<"\n";*/
int ans=(count)*(count+1)/2;
cout<<ans<<"\n";
}
return 0;
}
