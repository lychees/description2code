#include<bits/stdc++.h>
using namespace std;

int main() {
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        int n,d,r;
        scanf("%d%d%d",&n,&d,&r);
        int a,b;
        a=n/d;
        b=n-d*a;
        while(b&&r) {
            r--;
            n=b*10;
            a=n/d;
            b=n-d*a;
        }
        if(r==0) {
            printf("%d\012",a);
        } else {
            puts("0");
        }
    }
    return 0;
}
