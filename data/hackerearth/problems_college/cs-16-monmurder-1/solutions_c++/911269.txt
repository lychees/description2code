using namespace std;
bool arr[10000001];
void markMultiples(int a, int n)
{
    int i = 2, num;
    while ( (num = i*a) <= n )
    {
        arr[ num-1 ] = 1;
        ++i;
    }
}

{

        for (int i=1; i<n; ++i)
        {
            if ( arr[i] == 0 )
            {
                markMultiples(i+1, n);

        }
    int T,H,win;
