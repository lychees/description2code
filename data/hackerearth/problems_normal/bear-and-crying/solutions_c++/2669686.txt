
#include<iostream>
#include<vector>
#include<cmath>
#define mod 1000000007
using namespace std;

int semi, ready,result;

void add(int& a,int b){a=(a+b)%mod;}

int main()
{
    string s;
    cin >> s;
	for(int i=0;i<s.length();i++)    
	{
		if(s[i]=='_')
			add(ready,(ready+semi)%mod);
		else
		{
			add(result,ready);
			add(semi,1);
		}
	}
    
	cout << result<< endl;
    return 0;
}
