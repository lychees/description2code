#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,x,s;
	vector<int> v;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>x;
		v.push_back(x);
	}
	sort(v.begin(),v.end());
	cin>>s;
	int c=0,ans=0;
	for(int i=n-1;i>=0;i--)
	{
		if(c<s)
		{
			c+=v[i];
			ans++;
		}
		else
		break;
	}
	cout<<ans<<"\n";
}
