#include <iostream>
#include<algorithm>
#include<string.h>
using namespace std;

int main()
{
    int t,count,flag;
    string s;
    cin>>t;
    while(t--)
    {
    	cin>>s;
    	count=0;
    	while(true)
    	{
    	  flag=0;
    	  for(int i=0;i<s.length();i++)
    	   {
    		if(((s[i]=='X')&&(s[i+1]=='O'))||((s[i]=='O')&&(s[i+1]=='X')))
    		{
    			s[i]='X';
    			s[i+1]='X';
    			flag=1;
    		}
    	   }
    	   if(flag==0)
    	   {
    	   	break;
    	   }
    	}
    	for(int i=0;i<s.length();i++)
    	{
    		if(s[i]=='O')
    		{
    			count++;
    		}
    	}
    	cout<<count<<endl;
    }
}
