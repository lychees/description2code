#include <bits/stdc++.h>
using namespace std;
#define ll long long

int main(){
	int t;
	scanf("%d",&t);
	while (t--){
		ll int a,b;
		scanf("%lld%lld",&a,&b);
		int c=0;
		int cr = 0;
		while (a||b){
			int da = a%10;
			int db = b%10;
			if (da+db+cr>9){
				cr = 1;
				c++;
			}
			a/=10;
			b/=10;
		}
		if (c==0) cout << "No carry operation" << endl;
		else if (c==1) cout << "1 carry operation" << endl;
		else cout << c << " carry operations" << endl;
	}
	return 0;
}
