#include <bits/stdc++.h>
using namespace std;

#define pb push_back
vector<int> lis;
int main() {
    int t;
    scanf("%d",&t);
    while(t--) {
    	lis.clear();
    	int n,x;
    	scanf("%d",&n);
    	for(int i=0;i<n;i++){
    		scanf("%d",&x);
    		if(x>i){
    			x-=i;
    			//cout<<x<<endl;
    			auto it = upper_bound(lis.begin(),lis.end(),x);
    			if(it==lis.end())
    				lis.pb(x);
    			else
    				*it = x;
    		}
    	}
    	//cout<<lis.size()<<endl;
    	printf("%d\n",n-lis.size());
    }
    return 0;
}
