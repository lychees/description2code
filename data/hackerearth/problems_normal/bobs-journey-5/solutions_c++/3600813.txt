#include <iostream>
#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t = 0,n = 0;
	cin>>t;
	while(t--){
		cin>>n;
		map<char,bool> ma;
		string s;
		bool flag = false;
		while(n--){
			cin>>s;
			if(ma.find(s[0]) == ma.end()){
				ma[s[0]] = true;
			}
			else{
				flag = true;
			}
		}
		if(flag) cout<<"NO"<<endl;
		else cout<<"YES"<<endl;
	}
    return 0;
}
