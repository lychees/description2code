#include <bits/stdc++.h>
using namespace std;
int cnt[1000];
int ans[1000];
long long int sum_of_square(long long int x)
{
    long long int no=x;

    long long int s;
    no=no*no;
    //cout<<no;
        s=0;

        while(no>0)
        {
            s+=no%10;
            no/=10;
        }
        //cout<<s;
    return s;
}
int preprocess(int num)
{
    int i,s,flag=0,no;
    no=num;
    for(i=0;i<=200;i++)
        cnt[i]=0;
    while(no!=1 && no !=4 && !ans[no])
    {
        //cout<<no<<endl;
        cnt[no]=1;
        no=no*no;
        s=0;

        while(no>0)
        {
            s+=no%10;
            no/=10;
        }
        no=s;
        if(cnt[no]==1)
        {
            flag=1;break;
        }

    }
    if(flag==1)
    {
        ans[num]=1;//1 means faulty
    }
    else
    {
        if(no==1 || no==4)
        ans[num]=2;//2 means good
        else
        ans[num]=ans[no];
    }
}
int main()
{
    //cout<<(long long)99999999*99999999<<endl;
    for(int i=1;i<200;i++)
        preprocess(i);
    long long int t,a;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld",&a);

        if(a>199)
            a=sum_of_square(a);

        if(ans[a]==2)
            printf("YES\n");
        else
            printf("NO\n");


    }

}
