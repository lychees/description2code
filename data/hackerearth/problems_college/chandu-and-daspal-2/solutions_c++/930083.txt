#include <iostream>
using namespace std;

int gcd(int a, int b) {
    while (a != b) {
      if (a > b)
        a = a - b;
      else
        b = b - a;
    }
    return a;
  }

int main()
{
   int t,i,x,y,j,answer=0;
   char str[10];

   				answer = gcd(x,y);
   				x = x-answer;
   				y = y-answer;
   				
   				if(x==0 || y==0){
   					if(j%2==0)
   						cout<<"Chandu"<<endl;
   					else
   						cout<<"Daspal"<<endl;
   						break;
   				}
   				j++;
   			}
   		}
   		else{
   			j=0;

   				answer = gcd(x,y);
   				x = x-answer;
