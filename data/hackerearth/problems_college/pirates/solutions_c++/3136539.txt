#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	int a[n];
	for(int i=0;i<n;i++)
	cin>>a[i];
	int c_max=1,max_so_far=1;
	for(int i=1;i<n;i++)
	{
		if(abs(a[i]-a[i-1])==1)
		c_max++;
		else
		{
			//cout<<c_max<<"\n";
			if(c_max>max_so_far)
			max_so_far=c_max;
			c_max=1;
		}
	}
	if(c_max>max_so_far)
	max_so_far=c_max;
	cout<<max_so_far<<"\n";
}
