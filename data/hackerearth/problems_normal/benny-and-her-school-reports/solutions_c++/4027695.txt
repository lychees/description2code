#include<stdio.h>
#include<string.h>
#include<stdlib.h>
int main(){

	int test;
	scanf("%d",&test);
	while(test--){

		long int n,m,x;
		scanf("%ld %ld %ld",&n,&m,&x);
		long long sum = 0;
		long long value;
		for(long int i=0;i<n;i++){
			scanf("%lld",&value);
			sum+=value;
		}

		long long int result = (x*(n+1))-sum;

		if(result>m) printf("Impossible\n");
		else if(result<0) printf("1\n");
		else{
			printf("%lld\n",result);
		}
	}
}
