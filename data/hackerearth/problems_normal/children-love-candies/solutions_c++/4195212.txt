#include <iostream>
using namespace std;

int main()
{
	int Q, T;
	long long N;

	cin >> Q;
	while (Q--)
	{
		cin >> N >> T;
		for (int i = 0; i < T; ++i)
		{
			if (N % 2 == 0)
				N = N - (N / 2);
			else
				N = N - ((N + 1) / 2);
			N = N - (N / 4);
		}
		cout << N << "\n";
	}
//	system("pause");
	return 0;
}
