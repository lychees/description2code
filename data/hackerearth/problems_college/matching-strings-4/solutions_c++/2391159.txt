#include <iostream>
#include <string>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	string s1,s2;
    	int res=0;
    	cin>>s1>>s2;
    	int c=s1.size()<s2.size()?s1.size():s2.size();
    	for(int j=0;j<c;j++)
    	{
    		if(!(s1[j]-s2[j]))	res++;	
    	}
    	cout<<res<<endl;
    }
    return 0;
}
