#include <iostream>
using namespace std;


int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int occ[26] ={0};
		string a;
		int n;
		cin>>n;
		cin>>a;
		
		for(int i =0;i<n;i++)
		{
			
			int temp = a[i]-'a';
		occ[temp]++;
	
		}
		int greatest =0;
		for(int i =0;i<26;i++)
		{
			if(greatest<occ[i])
			greatest = occ[i];
		}
		if(greatest%2)
		cout<<"No"<<endl;
		else
		cout<<"Yes"<<endl;
	}
		
	
    
    return 0;
}
