#include<bits/stdc++.h>
using namespace std;

bool done[100001]={0};
bool ans[100001]={0};
	

int sum(long long n)
{
	int ret=0;
	while(n)
	{
		ret+=(n%10);
		n/=10;
	}
	return ret;
}

void solve(long long n)
{
	//cout<<n<<"\n";
	if(done[n])
		return;
	int m=sum(n*n);
	if(done[m])
		ans[n]=ans[m];
	else if(m==1 || m==4)
		ans[n]=1;
	else if(m==n)
		ans[n]=0;
	else
	{
		solve(m);
		ans[n]=ans[m];
	}
	done[n]=1;
}

int main()
{
	int t;
	long long n,i;
	ans[4]=done[4]=1;
	done[13]=1;
	for(i=0;i<=100000;i++)
	 	solve(i);
	// for(i=0;i<=100000;i++)
	// 	if(ans[i])
	// 		printf("%lld\t%d\n",i,sum(i*i));
	scanf("%d",&t);
	while(t--)
	{
		scanf("%lld",&n);
		while(n>1000ll)
			n=sum(n*n);
		if(ans[n])
			puts("YES");
		else
			puts("NO");
	}
}
