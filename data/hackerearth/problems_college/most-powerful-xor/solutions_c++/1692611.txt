#include <iostream>
using namespace std;
#include<math.h>

long long int binary(int x)
{
	int d;int i=0;
	long long int sum=0;
	while(x>0)
	{
		d=x%2;
		sum=d*pow(10,i++)+sum;
		x=x/2;
	}
	
	
	return sum;
}

int main()
{
    int t,i,j,a,b,l;
    int max;
    cin>>t;
    for(int l=0;l<t;l++)
    {
    	cin>>a;
    	
    	cin>>b;
    	max=0;
    	for(i=a;i<=b;i++)
    	for(j=i;j<=b;j++)
    	{
    	long long int a1=binary(i);
    	long long int b1=binary(j);
    	
    	int k=0;
    	
    	int sum=0;
    	
    	while((a1!=0)&&(b1!=0))
    	{
    		if((a1%10)!=(b1%10))
    		sum=sum+pow(2,k);
    		
    		a1=a1/10;
    		b1=b1/10;
    	
    		k++;
    	}
    	
    	while(a1!=0)
    	{
    		if(a1%10==1)
    		sum=sum+pow(2,k++);
    		a1=a1/10;
    	}
    	
    	while(b1!=0)
    	{
    		if(b1%10==1)
    		sum=sum+pow(2,k++);
    		b1=b1/10;
    	}
    	
    	
    	if(max<sum)
    	max=sum;
    	}
    	 cout<<max<<endl;
    }
    
   
    return 0;
}
