#include <bits/stdc++.h>
using namespace std;

int main()
{
    int i,number_of_races,N;
    map<string, int> A;
    string name_winner,name_loser;
    scanf("%d",&N);
    number_of_races=pow(2,N)-1;
    for(i=0;i<number_of_races;++i)
    {
    	cin>>name_winner>>name_loser;
    	if (A.find(name_winner) == A.end())
    		A[name_winner]=1;
    	else
    		A[name_winner]+=1;
    	if(A[name_winner]==N)
    		break;
    }
    cout<<name_winner;
    return 0;
}
