#include<bits/stdc++.h>
using namespace std;
int coun[100005];
int main()
{
    int n,i,a,j;
    scanf("%d",&n);
    for(i=0;i<n;i++)
          {
              scanf("%d",&a);
           int s=sqrt(a)+1;
             for(j=2;j<=s;j++)
             {
                 if(a==1)
                     break;
                 if(a%j==0)
                 {
                     int r=0;
                     while(a!=1 && a%j==0)
                     {
                         a=a/j;
                         r++;
                     }
                     coun[j]+=r;
                 }
             }
             if(a!=1)
                coun[a]++;
          }

          int m;
          scanf("%d",&m);
          for(i=0;i<m;i++)
           {
             scanf("%d",&a);
             int ans=100005;

             int s=sqrt(a)+1;
             for(j=2;j<=s;j++)
             {
                 if(a==1)
                     break;

                 if(a%j==0)
                 {
                     int r=0;
                     while(a!=1 && a%j==0)
                     {
                         a=a/j;
                         r++;
                     }
                     //cout<<j<<" "<<r<<endl;
                     ans=min(ans,coun[j]/r);
                 }

             }
              if(a!=1)
                 {
                    // cout<<a<<" "<<coun[a]<<endl;
                     ans=min(ans,coun[a]);
                 }
             cout<<ans<<endl;

           }

 return 0;
}
