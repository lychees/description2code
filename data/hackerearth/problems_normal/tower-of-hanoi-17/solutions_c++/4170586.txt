#include<cstdio>
#include<algorithm>
#include<vector>
 
using namespace std;
 
 
vector<pair<int, int> > v;
long long dp[300];
 
long long MAX(long long a, long long b){
    if(a > b) return a;
    return b;
}
 
void init(){
    v.clear();
    for(int i=0;i<=201;i++) dp[i] = 0;
}
 
int main(){
    int t;
    scanf("%d", &t);
    while(t--){
        init();
        int n;
        scanf("%d", &n);
        long long ans = 0;
        int r, h;
        for(int i=0;i<n;i++){
            scanf("%d %d", &r, &h);
            v.push_back(make_pair(r, h));
        }
        sort(v.begin(), v.end());
        for(int i=0;i<n;i++){
            int tr = v[i].first;
            int th = v[i].second;
            if(i == 0){
                dp[i] = th;
                ans = th;
            }
            else{
                for(int j=0;j<i;j++){
                    if(tr > v[j].first && th > v[j].second){
                        dp[i] = MAX(dp[i], dp[j]+th);
                    }
                }
                if(dp[i] == 0){
                    dp[i] = th;
                }
                ans = MAX(ans, dp[i]);
            }
        }
        printf("%lld\n", ans);
    }
}
