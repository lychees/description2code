#include<bits/stdc++.h>
using namespace std;
int main()
{
        int t;
        cin>>t;
        string s;
        while(t--)
        {
                int l,c=0,i,ans;
                cin>>s;
                int c1=0;
                l=s.length();
                for(i=0;i<l;i++)
                {
                        if(s[i]=='0')
                        {
                                c++;
                        }
                        if(s[i]=='1')
                        {
                                c1++;
                        }
                }
                ans=l-2*min(c,c1);
                cout<<ans<<'\n';

        }
        return 0;

}
