#include<cstdio>
using namespace std;
int a[10000000]={0};
void compute()
{
	a[1]=0;
	for(int i=2;i*i<=10000000;i++)
		if(a[i]==0)
			for(int j=2*i;j<=10000000;j+=i)
				a[j]=1;
}
int main()
{
	compute();
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		long long int i=n;
		while(1)
		{
			long long int x=i,y=0;
			if(a[i]==0)
			{
				while(x!=0)
				{
					y=y*10+(x%10);
					x=x/10;
				}
			}
			//printf("huh\n");
			if(i==y)
			{
				printf("%lld\n",i);
				break;
			}
			i++;
		}
	}
	return 0;
}
