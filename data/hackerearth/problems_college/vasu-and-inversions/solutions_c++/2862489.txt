#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    long int n,k,i,j,ctr,ans;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%ld%ld",&n,&k);
    	int a[n];
    	for(i=0;i<n;i++)
    	scanf("%d",&a[i]);
    	for(i=0;i<n&&k>0;i++)
    	{
    		if(a[i]==0)
    		{
    			k--;
    			a[i]=1;
    		}
    	}
    	ctr=0;
    	for(i=0;i<n;i++)
    	{
    		if(a[i]==1)
    		ctr++;
    	}
    	if(k%2==0)
    	ans=ctr;
    	else ans=ctr-1;
    	printf("%ld\n",ans);
    }
    
    
    
    return 0;
}
