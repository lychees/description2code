#include <iostream>
#include<algorithm>
using namespace std;

int main()
{	bool flag;
	int n;int result;
	cin>>n;
	int a[n],i,j;
	for(i=0;i<n;++i)
	{
		cin>>a[i];
	}
	sort(a,a+n);
	for(i=n-1;i>=0;--i)
	{ if(a[i]%2!=0){
		flag=true;
		j=2;
		while(j<sqrt(a[i]))
		{
			if(a[i]%j==0)
			{flag=false;break;}
			++j;
		}
		if(flag)
		{
			result=a[i];
			//flag1=true;
			break;
		}
	}
	}
	int count=0;
	for(;i>=0;--i){
		if(a[i]!=result)
		break;
		else
		count++;
	}
	cout<<result<<"\n"<<count;
    return 0;
}
