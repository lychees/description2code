#include <iostream>

typedef unsigned long long ull;

using namespace std;

int main()
{
	int t;
	cin >> t;
	while (t--)
	{
		ull n;
		cin >> n;
		ull ans = n - 1, count = 0;
		if (n == 1)
		{
			ans = 0;
			count = 1;
			cout << ans << ' ' << count << endl;
			continue;
		}		
		while (ans)
		{
			++count;
			ans >>= 1;
		}
		ans = 1;
		ans <<= count;
		ans -= 1;
		count = 0;
		ull temp = n - 1, half = ans >> 1;
		while (temp > half)
		{
			++count;
			--temp;			
		}
		count <<= 1;
		cout << ans << ' ' << count << endl;
	}
	
	return 0;
}
