def total(lis):
    cum_sum = 0
    for i in lis:
        cum_sum = cum_sum + i
        yield cum_sum


def bin_srch(cumu_sum, que):
    if que <= cumu_sum[0]:
        return 1

    if que > cumu_sum[len(cumu_sum)-2]:
        return len(cumu_sum)

    first = 0
    last = len(cumu_sum)-1

    while first <= last:
        # print 'first'
        # print first
        # print 'last'
        # print last
        mid = (first+last)/2
        if cumu_sum[mid] < que and cumu_sum[mid+1] >= que:
            return mid+2
        else:
            if cumu_sum[mid] < que:
                first = mid
            else:
                last = mid
