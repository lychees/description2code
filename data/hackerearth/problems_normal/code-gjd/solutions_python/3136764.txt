from itertools import islice


def acc(iterable):
    # accumulate([1,2,3,4,5]) --> 1 3 6 10 15
    # accumulate([1,2,3,4,5], operator.mul) --> 1 2 6 24 120
    it = iter(iterable)
    try:
        total = next(it)
    except StopIteration:
        return
    yield total
    for element in it:
        total += element
        yield total


for _ in xrange(int(raw_input())):
    raw_input()
    print(sum(islice(acc(sorted((int(s) for s in raw_input().split()), reverse=True)), 1, None)))
