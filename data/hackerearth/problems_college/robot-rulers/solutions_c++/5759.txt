using namespace std;

typedef unsigned long long int ulli;

#define K 1000000
#define MAX 1000000000
int xmin, xmax;

void recalculate(int a, int b)
{
	int x, y;
	if(a > b)
	{
		y = MAX;
		x = (a + b) / 2;
		if((a + b) % 2)
			x++;
	}
	else
	{
		x = 0;
		y = (a + b) / 2;
	}

	if(x > xmin)
		xmin = x;
	if(y < xmax)
		xmax = y;
}

int main(void)
{
	int t, n, i, a, b;
	scanf("%d", &t);
	while(t--)
	{
		xmin = 0;
		xmax = MAX;
		scanf("%d", &n);
		scanf("%d", &a);
		if(n == 1)
			printf("%d %d\n", xmin, xmax);
		else
		{
			i = n - 1;
			while(i--)
			{
				scanf("%d", &b);
				recalculate(a, b);
				a = b;
			}
			if(xmin <= xmax)
				printf("%d %d\n", xmin, xmax);
			else
				printf("-1\n");
		}
	}
	return 0;
}
