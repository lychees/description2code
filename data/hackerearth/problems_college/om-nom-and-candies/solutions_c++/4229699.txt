#include <iostream>
#include <vector>
#define pb(x) push_back(x);
#define maxi 100001
using namespace std;

int main()
{
	bool mark[maxi];
	for(int i=0;i<maxi;i++)
	mark[i]=1;
	vector<int> prime;
	for(int i=2;i*i<maxi;i++)
	{
		if(mark[i]==1)
		{
			for(int j=i*i;j<maxi;j=j+i)
			{
				mark[j]=0;
			}
		}
	}
	//cout<<mark[2]<<endl;
	for(int i=2;i<maxi;i++)
		if(mark[i])
			prime.pb(i);
	int n,temp;
	cin>>n;
	int a[n+1],ans[n+1];
	for(int i=1;i<=n;i++)
	{
		cin>>a[i];
		ans[i]=0;
	}
	ans[1]=a[1];
	ans[2]=ans[1]+a[2];
	for(auto it=prime.begin();it!=prime.end();it++)
    {
    	if((*it)<=n)
    	ans[*it]=a[*it]+a[1];
    	else break;
    }
    for(int i=2;i<=n;i++)
    {
    	for(auto it=prime.begin();it!=prime.end();it++)
    	{
    		temp=*it;
    		if(i*temp<=n)
    			ans[i*temp]=max(ans[i]+a[i*temp],ans[i*temp]);
    		else break;
    	}
    }
    for(int i=1;i<=n;i++)
	{
		cout<<ans[i]<<" ";
	}
    return 0;
}
