import math


def s(n):
    t = set()
    t.add(1)
    for i in xrange(2, int(math.sqrt(n)) + 1):
        u = i * i
        while u < n:
            t.add(u)
            u *= i
    for i in t:
        if n - i in t:
            return 'Yes'
    return 'No'


T = int(raw_input())
for i in xrange(T):
    n = int(raw_input())
    print s(n)
