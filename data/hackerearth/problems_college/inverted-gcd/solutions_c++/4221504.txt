#include <iostream>
using namespace std;
#include <vector>
#define ll long long

ll cop(ll a,ll b) 
{   ll t;
    while(b!=0)
    {   t=a;
        a=b;
        b=t%b;
    }
    if(a==1)
        return 1;
    return 0;
}


int main()
{   ll n,q,i,j,s=0;
    cin>>n;
    vector<ll> z;
    for(i=0;i<n;++i)
    {   cin>>q;
        z.push_back(q);
    }
    for(i=0;i<n;++i)
        for(j=i+1;j<n;++j)
        {   if(z[i]>z[j])
            {   if(cop(z[i],z[j]))
                    s++;
            }
        }
    cout<<s<<"\n";
    return 0;
}
