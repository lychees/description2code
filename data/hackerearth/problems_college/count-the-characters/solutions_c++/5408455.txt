#include <iostream>
#include<string>
using namespace std;

int main()
{
    int t,i,hash[26]={0},j,count=0;
    cin>>t;
    string s[t];
    for(i=0;i<t;i++)
    {
    	cin>>s[i];
    }
    for(i=0;i<t;i++)
    {
    	for(j=0;j<s[i].length();j++)
    	{
    		if(hash[s[i][j]-'a']==i)
    		hash[s[i][j]-'a']++;
    	}
    }
    for(i=0;i<26;i++)
    {
    	if(hash[i]==t)
    	count++;
    }
    cout<<count;
    return 0;
}
