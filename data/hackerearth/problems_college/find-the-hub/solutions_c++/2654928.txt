#include <bits/stdc++.h>
using namespace std;
vector<int> v(100001);
int main()
{
    int i,j;
    for(i=2;i<=100000;i++)
    {
    	if(v[i]==0)
    	{
    		
    	for(j=2*i;j<=100000;j=j+i)
    	{
    		v[j]=1;
    	}
    	}
    }
    int n,maxi=-1,num;
    map<int,int> m;
    map<int,int>::iterator it;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
    	scanf("%d",&num);
    	if(v[num]==0)
    	{
    		if(num>maxi)
    			maxi=num;
    		m[num]++;
    	}
    }
    printf("%d\n%d",maxi,m[maxi]);
    return 0;
}
