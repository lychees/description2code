#include<bits/stdc++.h>
using namespace std;
#define inp(x) scanf("%d",&x)
#define il(x) scanf("%lld",&x)
#define mod 1000000007
#define pb push_back
#define mp make_pair
typedef long long int ll;
int main()
{
    ll n,c,d,m,i,x,y,a,b,z;
    priority_queue<pair<ll,ll>,vector<pair<ll,ll> >,greater<pair<ll,ll> > > pq,busy;
    vector<pair<pair<ll,ll>,ll> > v;
    il(n),il(m);
    for(i=0;i<n;i++)
    {
        il(x),il(y);
        v.pb(mp(mp(x,x+y),i));
    }
    for(i=1;i<=m;i++)
        pq.push(mp(i,0));
    vector<ll> ans(n);
    sort(v.begin(),v.end());
    pair<ll,ll> p;
    for(i=0;i<n;i++)
    {
        x=v[i].first.first,y=v[i].first.second;
        z=v[i].second;
        if(!busy.empty())
        {
            while(!busy.empty() && busy.top().first<=x)
            {
                c=busy.top().first,d=busy.top().second;
                pq.push(mp(d,c));
                busy.pop();
            }
        }
        if(pq.empty())
            ans[z]=-1;
        else
        {
            p=pq.top();
            b=p.second,a=p.first;
            pq.pop();
            ans[z]=a;
            busy.push(mp(y,a));
        }
    }
    for(i=0;i<n;i++)
        cout << ans[i] << " ";
    cout << endl;
    return 0;
}
