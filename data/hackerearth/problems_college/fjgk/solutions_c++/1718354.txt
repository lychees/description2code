#include<iostream>
#include<cstdio>
using namespace std;
double arr[10001];
int prime(int i);
int main()
{
    int i=3,r=0,t;
    while(i<=10000)
    {
      int p=i*i;
      t=2*r+2;
      if(prime(p-t))
        arr[i]++;
            if(prime(p-2*t))
        arr[i]++;
              if(prime(p-3*t))
        arr[i]++;
        arr[i]+=arr[i-2];
      i+=2;
      r++;
    }
int task;
scanf("%d",&task);
while(task--)
{
    int l;
    scanf("%d",&l);
    printf("%.6f\n",arr[l]/(4*(l-1)/2+1)*100);
}
}
int prime(int i)
{
    if(i==1)
    return 0;
    if(i<4)
        return 1;
        if(i%2==0)
            return 0;
    for(int j=3;j*j<=i;j+=2)
        if(i%j==0)
        return 0;
    return 1;
}
