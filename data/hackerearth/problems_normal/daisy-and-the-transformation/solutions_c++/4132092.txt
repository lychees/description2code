#include <iostream>
#include <string>
using namespace std;
 
int main() {
	string a,b;
	int t,n;
	cin>>t;
	while(t--)
	{
		int x=0;
		cin>>a>>b>>n;
		for(int i=0;i<a.length();i++)
		{
			if (a[i]!=b[i])
			x++;
		}
		if(x==n || (n>x && (n-x)%2==0))
		cout<<"Yes"<<endl;
		else
		cout<<"No"<<endl;
		
	}
	return 0;
}
