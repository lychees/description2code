#include <iostream>
#include <algorithm>
#include <string.h>
using namespace std;

struct pt
{
    int val;
    int idx;
};

pt a[100007];

bool cust(pt p, pt q)
{
    if(p.val==q.val) return p.idx < q.idx;
    else return p.val < q.val;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        memset(a,0,sizeof(a));
        int n;
        cin>>n;
        for(int i=0;i<n;i++)
        {
            cin>>a[i].val;
            a[i].idx=i;
        }
        sort(a,a+n,&cust);
        int ans=0;
        for(int i=0;i<n;i++)
        {
            ans = max(ans,abs(a[i].idx - i));
        }
        cout<<ans<<"\n";
    }
}
