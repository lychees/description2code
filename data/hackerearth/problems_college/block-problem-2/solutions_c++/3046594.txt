#include<vector>
#include<stdio.h>
#include<string.h>
#include<iostream>
using  namespace std;
#define ll long long int
#define mod 1000000007

ll poW(ll x, ll y)
{
    if(y==0) return 1;
    ll ans, temp = poW(x, y/2)%mod;
    ans = (temp*temp)%mod;
    if(y&1) ans=(ans*x)%mod;
    return ans;
}

ll buildWays(ll w,ll h)
{
    vector<ll> dp(w, 0), totalComb(w, 0), level((w<4?4:w), 0);
    level[0] = 1, level[1] = 2, level[2] = 4,  level[3] = 8;
    
    if(w>4) for(int i = 4; i<w; i++) level[i] = (level[i-1] + level[i-2] + level[i-3] + level[i-4])%mod;
    for(ll i=0; i<w; i++)  totalComb[i] = poW(level[i], h);
        
    dp[0] = 1; 
    for(ll i=1; i<w; i++)
    {
        dp[i] = totalComb[i]%mod;
        for(ll j=1; j<=i; j++)  dp[i] = (mod + dp[i] - (dp[i-j]*totalComb[j-1])%mod)%mod; 
    }
    return dp[w-1];
}

int main() 
{
    ll T;
    ll N, M;
    cin>>T;
    while(T--)
    {
        scanf("%lld%lld",&N,&M);
        printf("%d\n", buildWays(M,N));
    }
    return 0;
}
