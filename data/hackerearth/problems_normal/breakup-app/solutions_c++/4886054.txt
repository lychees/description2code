#include <iostream>
#include <string>
using namespace std;

int main()
{
    int n;
    cin>>n;
    //char a[1000];
    cin>>ws;
    string a;
    int dt[31];
    for(int i=0;i<31;i++)
    {
        dt[i]=0;
    }
    for(int i=0;i<n;i++)
    {
    	//cin.getline(a,1000);
    	getline(cin,a);
    	int j=0,p,m,n;

    	if(a[j]=='G')
            p=2;
        else
            p=1;
        while(j<a.length())
        {
            if(a[j]>='1'&&a[j]<='3')
            {
                if(a[j+1]>='0'&&a[j+1]<='9')
                {
                    m=(int)a[j]-48;
                    n=(int)a[j+1]-48;
                    m=m*10+n;
                    dt[m]+=p;
                    j++;
                }
                else
                {
                    m=(int)a[j]-48;
                    dt[m]+=p;
                }

            }
            else if(a[j]>='4'&&a[j]<='9')
            {
                m=(int)a[j]-48;
                dt[m]+=p;
            }
            j++;
        }


    }
    int ma=max(dt[19],dt[20]),p=1;
    
        for(int i=0;i<31;i++)
        {
            if(i==19||i==20)
                continue;
            if(dt[i]>ma)
            {
                p=0;
                break;
            }
        }
        if(dt[19]==dt[20])
        	p=0;
        if(p==0)
        {
            cout<<"No Date";
        }
        else
            cout<<"Date";
    return 0;
}
