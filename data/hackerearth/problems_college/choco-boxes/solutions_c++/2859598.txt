#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	int n, m;
	cin>>t;
	while (t--) {
		cin >> n >> m;
		int candy = m/n;
		if (m%n == 0) {
			int pos = n/2+1;
			candy = m/n-1;
			cout << pos << " " << candy << endl;
			continue;
		}
		int rem = m;
		if ( m > n) {
		 rem = m%n;
		if(rem == 0)
			rem = n;
		}
		
		if (n%2 == 0) {
			if (rem%2)
				cout << (rem+1)/2 << " " << candy << endl;
			else
				cout << (n - (rem)/2 + 1) << " " << candy << endl;
		}
		else {
			if ( (m/n)%2 ) {
				if (rem%2) {
					cout << n - (rem+1)/2+1 << " " << candy << endl;
				}
				else {
					cout << (rem)/2 << " " << candy << endl;
				}
			}
			else {
				if (rem%2)
				cout << (rem+1)/2 << " " << candy << endl;
			else
				cout << (n - (rem)/2 + 1) << " " << candy << endl;
			}
		}
	}
	return 0;
}
