from re import search
n=int(raw_input())
res=[]
pat="[b,c,d,f,g,h,j,k,l,m,n,p,q,r,s,t,v,w,x,z]{3,3}"
for i in range(n):
	s=raw_input()
	if(search(pat,s)):
		#print "pattern matched"
		res.append("hard")
		continue
	vow=0
	cons=0
	flag=1
	for j in s:
		if j in ['a','e','i','o','u','y']:
			vow+=1
		else:
			cons+=1
	if(cons>vow):
		res.append("hard")
		flag=0
			#break
	else:
		res.append("easy")
