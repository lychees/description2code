#include<iostream>
using namespace std;

typedef long long ll;
int main()
{
	int T;
	cin>>T;
	while(T--) {
		ll r,k;
		cin>>r>>k;
		ll evn=0ll,odd=0ll;
		for(int i=1;i<=r;i++) {
			if(i%2) { if (k>=i)odd+=i;else odd+=k;}
			else { if (k>=i)evn+=i;else evn+=k;}
		}
		cout<<evn<<" "<<odd<<endl;
	}
	return 0;
}
