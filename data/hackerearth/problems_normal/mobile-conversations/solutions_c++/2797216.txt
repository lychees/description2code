#include<stdio.h>
#include<algorithm>
#include<utility>
#include<vector>
using namespace std;
 
int main()
{
  int N;
  long long i, t, x, i1, i2;
  vector<pair<long,long> > s1, s2;
  long long diff_min = 0;
  scanf("%d", &N);
  for(i=0, i1=0, i2=0; i<N; i++)
    {
      scanf("%lld %lld", &t, &x);
      if(t<x)
    {
	  s1.push_back(make_pair(t, x));
	}
      else
	{
	  s2.push_back(make_pair(x, t));
	}
      diff_min += t-x;
    }
 sort(s1.begin(), s1.end());
  sort(s2.begin(), s2.end());
  diff_min += s2[0].first;
  long long max_req = diff_min, req = 0, bal = 0;
  for(i=0; i<s1.size(); i++)
    {
      req = s1[i].first - bal;
      max_req = max(max_req, req);
      bal += s1[i].second-s1[i].first;
    }
  printf("%lld\n", max(max_req,diff_min));
}
