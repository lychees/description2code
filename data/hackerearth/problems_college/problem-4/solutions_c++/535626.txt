#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    string s;
    char t;
    cin>>s;
    for(int i=0;i<s.length();i++)
    {
    	for(int j=i;j<s.length();j++)
    	{
    		if(s[j]<s[i])
    		{
    			t=s[j];
    			s[j]=s[i];
    			s[i]=t;
    		}
    	}
    }
    cout<<s;
    return 0;
}
