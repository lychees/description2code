#include <bits/stdc++.h>
using namespace std;
typedef stack<char> Stack;
int main()
{
	int T,i,l;	char Q[105],temp[105];	bool flag;
	scanf("%d",&T);
	while(T--)
	{
		Stack S;	memset(temp,0,sizeof(temp));
		scanf("%s",Q);	l=strlen(Q);	flag=false;
		for(i=0;i<l;i++)
		{
			if(Q[i]=='x')
			{
				while((!S.empty())&&S.top()=='o')
					S.pop();
				S.push(Q[i]);	flag=true;
			}
			else if(Q[i]=='o'&&flag==false)
			{
				S.push(Q[i]);
			}
			else if(Q[i]=='*')
			{
				S.push(Q[i]);	flag=false;
			}
		}
		i=0;
		while(!S.empty())
		{
			temp[i++]=S.top();	S.pop();
		}
		for(i=strlen(temp);i--;)
		{
			printf("%c",temp[i]);
		}
		printf("\n");
	}
return 0;
}
