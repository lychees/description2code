#include<stdio.h>
#include<iostream>
#include<stdlib.h>
using namespace std;
int a[1000010];
int compare(const void * a, const void * b)
{
	return (*(int*)a - *(int*)b);
}
int main()
{
	 //freopen("input.txt", "r", stdin);
	 long  long int i, count, j=0, k;
		 int n;
	 long long int i1, mat, ans = 0, sum = 0;
	cin >> n;
	for (i = 0; i < n; i++)
		cin >> a[i];
	qsort(a, n, sizeof(int), compare);
	for (i = 0; j < n; i++)
	{
		k = i + 1;
		count = k*k;
		if (count > a[j])
			ans += count - a[j++];
		else if (count < a[j])
			ans += count;
		else
			j++;

	}
	cout << ans;
	return 0;
}
