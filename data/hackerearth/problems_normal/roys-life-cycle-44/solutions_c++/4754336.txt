#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    scanf("%d",&n);
    string s[n];
    int B[n];
    for(int i=0;i<n;i++)
        cin>>s[i];
    int Rmax=0,Omax=0,u=0,k=0,trmax=0,ormax=0;
    for(int i=0;i<n;i++)
    {
        trmax=0;
        k=0;
        for(int j=0;j<s[i].size();j++)
        {
            if(s[i][j]=='C')
            {
                k++;
                u++;
            }
            else
            {
                trmax=max(trmax,k);
                k=0;
                ormax=max(u,ormax);
                u=0;
            }
        }
        trmax=max(trmax,k);
        Rmax=max(trmax,Rmax);
        //cout<<i<<" "<<trmax<<"  "<<Rmax<<endl;
    }
    ormax=max(ormax,u);
    cout<<Rmax<<" "<<ormax<<"\n";
    return 0;
}
