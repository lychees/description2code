#include<iostream>
#include<string.h>
using namespace std;
void swap (char *x, char *y)
{
    char temp;
    temp = *x;
    *x = *y;
    *y = temp;
}

void permute(char *a, int i, int n) 
{
   int j; 
   if (i == n)
     cout<<a<<"\n";
   else
   {
        for (j = i; j <= n; j++)
       {
          swap((a+i), (a+j));
          permute(a, i+1, n);
          swap((a+i), (a+j)); //backtrack
       }
   }
} 
 
/* Driver program to test above functions */
int main()
{
	int t,s,e;
	char str[10001];
	cin>>str;
	cin>>t;//no of queries
	while(t--)
	{
		cin>>s>>e;
		permute(str, s, e);
		cout<<"\n";
	}
 return 0;
}
