#include<iostream>
#include<algorithm>	
#include<cstdio>
#include<cmath>
#include<stdlib.h>
#include<string.h>
#include<set>
#include<vector>
using namespace std;

int main(){
	int t;
	cin>>t;
	while(t--){
		char s[204];
		int i;
		scanf("%s",s);
		int l=strlen(s);
		int sm=0;
		for(i=0;i<l;i++){ if(s[i]!='_') sm+=s[i]-'0'; }
		vector<int> vec;
		for(i=0;i<=9;i++){
			sm+=i;
			if(sm%9==0) vec.push_back(i);
			sm-=i;
		}
		int orr=vec.size()-1;
		for(i=0;i<vec.size();i++){
			cout<<vec[i];
			if(orr>0) { cout<<" or ";orr--;}
		}
		cout<<endl;
	}
	return 0;
}
