#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    while(t--)
    {
         long int n,k=0;
    	cin>>n;
      long int a[n];
    	
    	for(int i=0;i<n;i++)
    	   cin>>a[i];
    	   
    	   for(int i=0;i<n;i++)
    	   {    if(a[i]%2==0)
    	           k++;
    	   }
    	   
    cout<<"\n"<<k;	   
    }
    
    return 0;
}
