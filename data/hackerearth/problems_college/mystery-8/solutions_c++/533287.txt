#include<iostream>
#include<cstdio>
#include<string.h>
using namespace std;
int main()
{
    int test_cases;
    cin>>test_cases;
    getchar();
    while(test_cases>0)
    {
        char str[103];
        gets(str);
        // your code here
        int start = 0, end = strlen(str)-1;
        
        while(1)
        {
        	if(str[start]==' ' || str[start]==',' || str[start]=='.' || str[start]=='?' || str[start]=='!')
        	{
        		start++;
        		continue;
        	}
        	if(str[end]==' ' || str[end]==',' || str[end]=='.' || str[end]=='?' || str[end]=='!')
        	{
        		end--;
        		continue;
        	}
        	if(str[start] == '*')
        	{
        		cout<<str[end]<<endl;
        		break;
        	}
        	if(str[end] == '*')
        	{
        		cout<<str[start]<<endl;
        		break;
        	}
        	start++;
        	end--;
        }
        
        test_cases--;
    }
    return 0;
}
