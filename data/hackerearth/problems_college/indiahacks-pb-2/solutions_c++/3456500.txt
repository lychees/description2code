#include <iostream>
using namespace std;

string resver(string s)
{
	int k=s.length();

	string e;
	
	int p=0;
	for(int j=k-1;j>=0;j--){
		e+=s[j];
	
	p++;
	}
	
	return e;
}
int main()
{
   int ts;
   cin>>ts;
   string a[ts+1],b[ts+1];
   for(int i=0;i<ts;i++){
   	cin>>a[i];
   	
   }
    for(int i=0;i<ts;i++){
   	b[i]=resver(a[i]);
   	//cout<<b[i]<<endl;
   }
   int fl = 0;
  
   for(int u=0;u<ts;u++){
   	for(int v=0;v<ts;v++){
   		int h=((b[u]).length());
   		if((b[u]==a[v])&&u!=v){
   			
   			cout<<h<<" "<<(b[u])[(h/2)]<<endl;
   			fl = 1;
   				break;
   			
   		}
   	
   	}
   	
   	if (fl == 1)
   	break;
   }
    return 0;
}
