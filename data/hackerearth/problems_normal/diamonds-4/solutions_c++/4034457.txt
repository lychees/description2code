#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int test;
    cin>>test;
    while(test--)
         {
         	int x,y;
         	char c;
         	cin>>x>>y;
         	int a[x][y];
         	for(int i=0;i<x;i++)
         	   {
         	   	for(int j=0;j<y;j++)
         	   	   {
         	   	   	cin>>c;
         	   	   	if(c=='/')
         	   	   	a[i][j]=0;
         	   	   	else
         	   	   	a[i][j]=1;
         	   	   } 
         	   }
         	   int count=0;
         	for(int i=0;i<x-1;i++)
         	   {
         	   	for(int j=0;j<y-1;j++)
         	   	   {
         	   	   	if(a[i][j]==0 && a[i][j+1]==1 && a[i+1][j]==1 && a[i+1][j+1]==0)
         	   	   	   count++;
         	   	   } 
         	   }
         	  cout<<count<<endl; 
         }
    return 0;
}
