#include <iostream>
using namespace std;
#include <cmath>

    //cout << "Hello World!" << endl;
    int trophy(int n)
{
	if(n==0 || n==1)
	return 0;
	else 
	return (1+trophy(ceil(n/3.0)));
}

int main()
{
    //printf("Hello World!\n");
    int t;
    cin>>t;
    for(;t--;)
    {
    	long int n;
    	cin>>n;
    	int ans=trophy(n);
    	cout<<ans<<endl;
    }
    return 0;
}
