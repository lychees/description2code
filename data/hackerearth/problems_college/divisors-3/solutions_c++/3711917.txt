#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll long long int
#define pb push_back
#define mk make_pair
ll power(ll a, ll b) {
ll x = 1, y = a;
    while(b > 0) {
        if(b%2 == 1) {
            x=(x*y);
            if(x>mod) x%=mod;
        }
        y = (y*y);
        if(y>mod) y%=mod;
        b /= 2;
    }
    return x;
}
int arr[1000002];
vector<ll>x;
void sieve()
{
	ll i,j;
	for(i = 2; i <= 1000000; i++) {
		if(!arr[i]) {
			for(j = 2; j*i <= 1000000; j++) {
				arr[i*j] = 1;
			}
			x.pb(i);
		}
	}
}
int main() 
{
	sieve();
	ll n,i,j;
	int t,q;
	scanf("%Ld%d",&n,&t);
	vector<ll>di;
	for(i = 1; i <= sqrt(n); i++) {
		if(n%i == 0) {
			di.pb(i);
			j = n/i;
			if(j != i) {
				di.pb(j);
			}
		}
	}
	ll k;
	while(t--) {
		scanf("%d%Ld",&q,&k);
		ll c = 0;
		if(q == 1) {
			for(i = 0; i < di.size(); i++) {
				if(k%di[i] == 0) c++;
			}
		}
		else if(q == 2) {
			for(i = 0; i < di.size(); i++) {
				if(di[i]%k == 0) c++;
			}
		}
		else {
			for(i = 0; i < di.size(); i++) {
				if(di[i]%k != 0) c++;
			}
		}
		printf("%Ld\n",c);
	}
	return 0;
}
