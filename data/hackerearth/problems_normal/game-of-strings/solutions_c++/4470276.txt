#include <bits/stdc++.h>
using namespace std;
string str1,str2;
int dp[2][5001];
int findLcs(){
	int len1=str1.length();
	int len2=str2.length();
	for(int i=0;i<=len1;i++){
		for(int j=0;j<=len2;j++){
			//cout<<i<<" "<<j<<"\n";
			if(i==0||j==0){
				dp[1][j]=0;
			}
			else{
				if(str1[i-1]==str2[j-1]){
					//cout<<"a ";
					if(str1[i-1]=='a'||str1[i-1]=='i'||str1[i-1]=='e'||str1[i-1]=='o'||str1[i-1]=='u'||str2[j-1]=='a'||str2[j-1]=='i'||str2[j-1]=='e'||str2[j-1]=='o'||str2[j-1]=='u'){
						dp[1][j]=max(dp[1][j-1],dp[0][j]);
					}
					else{
						//cout<<"b ";
						dp[1][j]=max(dp[1][j-1],dp[0][j]);
						dp[1][j]=max(dp[1][j],dp[0][j-1]+1);
					}
				}
				else{
					//cout<<"c ";
					dp[1][j]=max(dp[1][j-1],dp[0][j]);
				}
			}
		}
		for(int j=0;j<=len2;j++){
			dp[0][j]=dp[1][j];
		}
	}
	/*for(int i=0;i<2;i++){
		for(int j=0;j<=len2;j++){
			cout<<dp[i][j]<<" ";
			
		}
		cout<<"\n";
	}*/
	return dp[1][len2];
}
int main(){
	cin>>str1;
	cin>>str2;
	//cout<<str1<<" "<<str2<<"\n";
	cout<<findLcs();
}
