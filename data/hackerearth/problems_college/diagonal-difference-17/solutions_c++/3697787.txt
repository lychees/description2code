#include <bits/stdc++.h>
using namespace std ; 
int main()
{
	int n ; 
	cin >> n ; 
	int a[100][100] ; 
	int i ; 
	int j ; 
	int p = 0 ; 
	for ( i = 0; i< n ; i++)
	{
		for ( j = 0; j< n ; j++)
		{
			cin >> a[i][j] ;
			if ( i == j )
			{
			p = p+a[i][j] ; 
			}
		}
	}
	int y = 0 ;
	int k = 0 ; 
	for ( i = n-1; i >=0 ; i--)
	{
	    k = k+ a[y][i] ; 
	    y = y+1 ; 
	}
	
	cout << abs(k-p) <<endl ;
	
	return 0 ; 
}
