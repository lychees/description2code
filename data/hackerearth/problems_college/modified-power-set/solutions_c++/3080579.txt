#include<iostream>
using namespace std;

char s[1000005];

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        cin>>s;
        bool arr[256];
        for(int i=0; i<256; i++)
            arr[i] = false;
            
        long long int ctr = 0;
        
        for(int i=0; s[i]!=NULL; i++)
            if(arr[s[i]] == false)
            {
                ctr++;
                arr[s[i]] = true;
            }
        ctr = ((ctr+1)*ctr)/2;
        cout<<ctr<<endl;
    }
    return 0;
}
