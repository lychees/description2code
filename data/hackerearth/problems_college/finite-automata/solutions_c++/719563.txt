#include <iostream>
#include <cstdio>

using namespace std;

int main()
{
    int n, i;
    scanf("%d", &n);

    int a[n][2];

    int x;
    for (i = 0; i < n; i++) {
        x = (i * 2) + 1;
        x = x % n;
        a[i][1] = x;
    }
    x = 0;
    for (i = 0; i < n; i++) {
        x = (i * 2);
        x = x % n;
        a[i][0] = x;
    }

    for (i = 0; i < n; i++) {
        printf("%d %d %d\n", i, a[i][0], a[i][1]);
    }

    return 0;
}
