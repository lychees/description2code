#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,a[100],i,j,k,c;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(i=0;i<n;i++)cin>>a[i];
    	sort(a,a+n);
    	int b[1000];k=0;
    	for(i=0;i<n/2;i++){
    		b[k++]=a[i]+a[n-i-1];
    	//	cout<<b[k++];
    	}
    	sort(b,b+k);
    	cout<<b[k-1]-b[0]<<endl;
    }
    return 0;
}
