int is_prime(int a)
{
	if(a==2)
	return 1;
	if((a%2==0)||(a==1))
	return 0;
	int c=0;
	for(int i=3;((i*i)<=a);i++)
	{
		if(a%i==0)
		c++;
	}
	if(c==0)
	return 1;
	else
	return 0;
}
