alpha = 'abcdefghijklmnopqrstuvwxyz'
t = input()
for i in range(t):
    s = raw_input()
    d = {}.fromkeys(alpha, 0)
    for j in s:
        if j in d:
            d[j] += 1
    count = 0
    for k in d:
        if d[k] >= 1:
            count += 1
    if count == 26:
        print"YES"
    else:
        print "NO"
