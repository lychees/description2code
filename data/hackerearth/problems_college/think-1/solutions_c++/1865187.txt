#include<bits/stdc++.h>
#define FOR(i,a,n) for(int i=a;i<n;i++)
#define si(x) scanf("%d",&x)
#define sll(x) scanf("%lld",&x)
#define pi(x) printf("%d\n",x)
#define pll(x) printf("%lld\n",x)
#define INF 2000000000
#define ll long long
using namespace std;
int main()
{
    int t;
    si(t);
    while(t--)
    {
        char ch[169],chr[169];
        scanf("%s",ch);
        int l=strlen(ch),n=0,rev=0;
        FOR(i,0,l)
        {
            chr[l-i-1]=ch[i];
            n=n*10+(ch[i]-48);
        }
        FOR(i,0,l)
        {
            rev=rev*10+(chr[i]-48);
        }
        if(n==rev)
        {
            puts("No");
            continue;
        }
        int sqn=n*n,sqr=rev*rev,c=0;
        while(sqn)
        {
            ch[c++]=sqn%10;
            sqn/=10;
        }
        int cr=0;
        while(sqr)
        {
            chr[cr++]=sqr%10;
            sqr/=10;
        }
        int f=0;
        if(cr==c)
        {
            FOR(i,0,c)
                if(ch[i]!=chr[c-i-1])
                    f=1;
        }
        else
            f=1;
        if(f==1)
            puts("No");
        else
            printf("%d\n",rev);
    }
    return 0;
}
