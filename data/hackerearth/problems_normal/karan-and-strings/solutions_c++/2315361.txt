#include <iostream>
#include<string>
using namespace std;

int main()
{
	int t;
	cin>>t;
	++t;
	while(--t)
	{
		 string s;
		 cin>>s;
		 int i=0;
		 char prev='m';
		 while(s[i])
		 {
		   if(prev==s[i])
		   {
		   	 ++i;
		     continue;
		   }
		   cout<<s[i];
		   prev=s[i];
		   ++i;  
 		 }
		 cout<<endl;  	 
	}
   
    return 0;
}
