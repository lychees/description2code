#include <iostream>
using namespace std;
void way(int n,int k, long long int dp[20][20])
{
	int i;
	if(n==0)
	dp[n][k]=0;
	else if(n==1)
	dp[n][k]=1;
	else if(k==0)
	dp[n][k]=1;
	else
	{
	for(i=0;i<=k;i++)
	 {
	 	dp[n][k]+=dp[n-1][i];
	 }
}
}	
int main()
{
   int t,n,k,i,j;
   long long int dp[20][20];
    cin>>t;
    	for(i=0;i<20;i++)
    	for(j=0;j<20;j++)
        	dp[i][j]=0;
        	for(j=0;j<20;j++)
    	for(i=0;i<20;i++)
    	{
       way(j,i,dp);
    	}
    while(t--)
    {
    	
    	cin>>n>>k;
    	
    	cout<<dp[n][k]<<endl;
    }
    return 0;
}
