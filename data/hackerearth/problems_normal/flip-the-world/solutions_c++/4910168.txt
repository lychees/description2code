#include<bits/stdc++.h>
using namespace std;
int mat[50][50];
int main()
{
    int n,m,t,i,j,k,l,ans;
    string str;
    cin>>t;
    while(t--)
    {
        cin>>n>>m;
        ans=0;
        for(i=0;i<n;i++)
        {
            cin>>str;
            for(j=0;j<m;j++)
            {
                mat[i][j]=str[j]-'0';
            }
        }
        for(i=n-1;i>=0;i--)
        {
            for(j=m-1;j>=0;j--)
            {
                if(mat[i][j]==0)
                {
                    ans++;
                    for(k=0;k<=i;k++)
                    {
                        for(l=0;l<=j;l++)
                        {
                            if(mat[k][l]==0)
                                mat[k][l]=1;
                            else
                                mat[k][l]=0;
                        }
                    }
                }
            }
        }

        cout<<ans<<"\n";
    }
    return 0;
}
