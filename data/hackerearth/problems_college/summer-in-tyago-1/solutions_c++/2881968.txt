inline void scan_f(int *a)
{
    char c = 0;
    while(c<33)
        //c = fgetc_unlocked(stdin);
        c = getc(stdin);
    *a = 0;
    while(c>33)
        {
            *a = (*a)*10 + c - '0';
    //c = fgetc_unlocked(stdin);
    c = getc(stdin);
    }
}

int main(){

    int n,a[200001],flag=0;
    scan_f(&n);
    scan_f(&a[0]);
    for(int i=1;i<n;i++){
        scan_f(&a[i]);
        for(int j=0;j<i;j++){
            if((a[i]-a[j]==j-i)&&flag==0){
                printf(":(");
                flag=1;
                break;
            }
        }
    }

    if(flag==0){
        for(int i=0;i<n;i++){
            for(int j=1;j<n;j++){
                if(a[j]<a[j-1]){
                    a[j]^=a[j-1];
                    a[j-1]^=a[j];
                    a[j]^=a[j-1];
                    a[j-1]++;
                    a[j]--;
                }
            }
        }
        for(int i=0;i<n;i++){
            printf("%d ",a[i]);
        }
    }
    return 0;
}
