#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    string n1, n2;
    char x[2];
    x[1] = 0;
    while(n--) {
    	cin>>n1>>n2;    	
    	int len1 = n1.length(), len2 = n2.length();    	
    	int mx = max(len1, len2);
    	//cout<<len1<<" "<<len2<<" "<<mx<<endl;
    	int num1[mx], num2[mx];
    	int i, j;
    	for(i=0;i<mx-len1;i++) {
    		num1[i] = 0;	
    	}
    	j = i;
    	for(i=0;i<len1;i++) {
    		x[0] = n1[i];
    		num1[j++] = atoi(x);    		
    	}
    	for(i=0;i<mx-len2;i++) {
    		num2[i] = 0;
    	}
    	j=i;
    	for(i=0;i<len2;i++) {
    		x[0] = n2[i];
    		num2[j++] = atoi(x);    		    		
    	}
    	int carry = 0, ans = 0;
    	for(i=mx-1;i>=0;i--) {
    		carry = (carry + num1[i] + num2[i]) / 10;
    		if(carry) ans++;
    	}
    	if(ans == 0) cout<<"No carry operation"<<endl;
    	else if(ans == 1) cout<<ans<<" carry operation"<<endl;
    	else cout<<ans<<" carry operations"<<endl;
    }
    return 0;
}
