#include <iostream>
using namespace std;

int a[100][100];
int n,c;
  
int cell(int i,int j)
{
    if(i>=n || j>=n) 
    {
    	return 0;
    }
    return a[i][j];
}
 
int main()
{
      int T; 
      cin>>T;

      for(int t=0;t<T;t++)
      {
      	int gold_coin=-1;
      	cin>>n;
      	cin>>c;

      for(int i=0;i<n;i++)
      {
      	for(int j=0;j<n;j++)
      	{
      	cin>>a[i][j];
      	}
      }
         
      for(int i=n-1;i>=0;i--)
      {
        for(int j=n-1;j>=0;j--)
		{
        a[i][j] = a[i][j]+max(cell(i+1,j),cell(i,j+1));
		}
      }

      gold_coin=a[0][0]-c;

      if(gold_coin >= 0)
      {
		cout<<gold_coin<<"\n";
      }
      else
      {
	    cout<<"-1"<<"\n";
      }

      }
	return 0;
}
