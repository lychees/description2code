typedef long long ll;
using namespace std;
const int MAXN = 510;
const ll INF = 1E17;
ll f[MAXN][MAXN];
ll a[MAXN];
ll pref[MAXN];
ll sum(int st, int fin) {
    ll res = pref[fin];
    if (st > 0) {
        res -= pref[st - 1];
    }
    return res;
}
int n, k;
int main() {
    cin>>n>>k;
    for (int i = 1; i <= n; i++) {
        cin>>a[i];
        pref[i] = pref[i - 1] + a[i];
    }
    for (int i = 1; i <= k; i++) {
        f[0][i] = INF;
    }
    for (int i = 1; i <= n; i++) {
        f[i][0] = INF;
    }
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= k; j++) {
            f[i][j] = INF;
            for (int q = i; q > 0; q--) {
                ll s = sum(q, i);
                f[i][j] = min(f[i][j], f[q - 1][j - 1] + s*s*(k - 1) - 2*sum(1, q - 1)*s);
            }
        }
    }
    cout<<f[n][k]<<endl;
    return 0;
}
