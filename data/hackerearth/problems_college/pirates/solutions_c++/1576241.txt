#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,j,k,n;
	scanf("%d",&n);
	int a[n];
	for(i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
	}
	
	int count=1;
	int count1=1;
	for(i=0;i<n-1;i++)
	{
		if(abs(a[i]-a[i+1])==1)
		count++;
		else
		count=1;
		if(count>count1)
		count1=count;
	}
	printf("%d\n",count1);
	return 0;
}
