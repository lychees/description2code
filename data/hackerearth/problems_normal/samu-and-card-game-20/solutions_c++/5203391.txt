#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long T,m,n,k,count,i,A,B,x,y;
    cin>>T;
    while(T--)
    {   count=0;
    	cin>>n>>m;
    	x=1;y=1;
    	cin>>k;
    	for(i=0;i<k;i++)
    	{
    		cin>>A>>B;
    		while((x+A<=n)&&(y+B<=m)&&(x+A>=1)&&(y+B>=1))
    		{   x+=A;
    		    y+=B;
    			count++;
    		}
     	}
     	cout<<count<<endl;
    }
    return 0;
}
