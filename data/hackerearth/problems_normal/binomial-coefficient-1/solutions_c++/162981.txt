#include <iostream>
#include <cstdio>
#define M 1000000007

using namespace std;

long long fact[1000005];

long long fast_pow(long long m, long long n)
{
          long long result = 1;
          while ( n > 0 ) {
                if ( n%2 != 0 ) {
                   result = (result*m)%M;
                }
                m = (m*m)%M;
                n = n/2;
          }
          return result;
}


int main()
{
    long long a,b,n,k;
    cin >> a >> b >> n >> k;
    fact[0] = 1;
    for ( long long i = 1; i <= n; i++ ) {
        fact[i] = (fact[i-1]*i)%M;
        fact[i] %= M;
    }
    long long val1 = fast_pow(a,k);
    long long val2 = fast_pow(b,n-k);
    val1 = (val1*val2)%M;
    long long ans1 = fact[n];
    long long ans2 = fact[k];
    long long ans3 = fact[n-k];
    ans2 = (ans2*ans3)%M;
    ans2 = fast_pow(ans2, M-2);
    ans1 = (ans1*ans2)%M;
    ans1 = (ans1*val1)%M;
    cout << ans1 << endl;
    return 0;
}
