#include <iostream>
#include<cstring>
#include<cstdio>
#include<string.h>
#include<bits/stdc++.h>
#include<cstdlib>
using namespace std;
#define ll long long int
int main() 
{
	ll t,l,i,ans;
	string s;
	cin>>t;
	while(t--)
	{
		cin>>s;
		l=(ll)s.length();
		ans=s[l-1]-'0';
		if(l==1||l==2)
			cout<<s<<endl;		
		else
		{
			/*if(s[l-2]=='*')
				ans=(ll)(s[l-3]-'0')*(s[l-1]-'0');
			if(s[l-2]=='+')
				ans=(ll) (s[l-3]-'0')+(s[l-1]-'0');
			if(s[l-2]=='-')
				ans=(ll) (s[l-3]-'0')-(s[l-1]-'0');*/
			for(i=l-2;i>=0;i-=2)
			{
				if(s[i]=='*')
					ans*=(ll)(s[i-1]-'0');
				else if(s[i]=='+')
					ans+=(ll)(s[i-1]-'0');
				else if(s[i]=='-')
					ans-=(ll)(s[i-1]-'0');
			}
		printf("%lld\n",ans);
		}		
	}
	return 0;
}
