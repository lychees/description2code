#include <cstdio>
#include <climits>
#include <vector>
#include <algorithm>
using namespace std;
int arr[1000001];
int main()
{
    int t,sum,maxi,n;
    scanf("%d",&t);
    while(t--)
    {
     scanf("%d",&n);
     for(int i=0;i<n;i++)
        scanf("%d",&arr[i]);
     int max_so_far = 0, max_ending_here = 0;
     for(int i=0;i<n;i++)
     {
     max_ending_here=max_ending_here+arr[i];
     if(max_ending_here < 0)
        max_ending_here=0;
     if(max_so_far<max_ending_here)
        max_so_far=max_ending_here;
     }
     printf("%d\n",max_so_far);
    }
    return 0;
}
