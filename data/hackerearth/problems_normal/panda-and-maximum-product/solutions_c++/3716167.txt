#include <iostream>
#include <climits>
using namespace std;

typedef long long int lli;

int main()
{
	lli n;
	cin>>n;
	lli a[n];
	lli max1, max2, min1, min2;
	max1 = max2 = LLONG_MIN;
	min1 = min2 = LLONG_MAX;
	for(lli i=0; i<n; i++){
		
		if(scanf("%lld", &a[i]) == 1){
			if(a[i]<min1){
				min2 = min1;
				min1 = a[i];
			} else if(a[i]<min2){
				min2 = a[i];
			}
		
			if(a[i]>max1){
				max2 = max1;
				max1 = a[i];
			} else if (a[i]>max2) {
				max2 = a[i];
			}
		}
	}
	lli min = min1*min2;
	lli max = max1*max2;
	if(min > max){
		cout<<min;
	} else {
		cout<<max;
	}

    return 0;
}
