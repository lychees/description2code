		if(x==u&&y==v)
		printf("Counter-Terrorists Win !\n");
		else
		{
			int dif1=v-u;
			int dif2=y-x;
			if(dif1==dif2)

				printf("Counter-Terrorists Win !\n");

				if((dif2-dif1)%2==0)

					int diff=(dif2-dif1)/2;
					if(x==(u+diff)&&y==(v-diff))
					{
						if(p>1)
						printf("Counter-Terrorists Win !\n");
						else
						printf("Terrorists Win !\n");
					}
					else if(x==(u-diff)&&y==(v+diff))

						printf("Counter-Terrorists Win !\n");
						else
						printf("Terrorists Win !\n");
					}
					else
