'''

'''
t = input()
for i in range(0, t):
    a = map(int, raw_input().split())
    s = 0
    l = a[0]
    for j in range(1, l+1):
        s = s+a[j]
    if (s % 2 == 0):
        print "Yes"
    else:
        print "No"
