l = [int(i) for i in raw_input().split()]
ans = 0
while len(l) >= 3:
    i = 0
    maximum = -1
    p = 1
    index = 0
    while i+2 < len(l):
        p = l[i]*l[i+2]
        if p > maximum:
            maximum = p
            index = i+1
        i += 1
    ans += maximum
    l.pop(index)
print ans
