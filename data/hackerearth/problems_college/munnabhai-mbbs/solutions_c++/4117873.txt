#include <bits/stdc++.h>
#include<vector>
using namespace std;
vector<int> a[1001];
int b[1001];
int i=0,j,ch=1,cnt=0,n,m;
int main()
{

    int t;
    cin>>t;
    while(t--)
    {
        
    	cin>>n>>m;
    
       for(i=1;i<=n;i++)
			a[i].clear(),b[i]=0;
    	for(ch=0;ch<m;ch++)
    	{
    		
    		cin>>i>>j;
    		a[i].push_back(j);
    		b[j]++;
    	}
    	cnt=0;ch=1;
    	for(;ch==1;)
    	{
    		ch=0;
    		for(i=1;i<=n;i++)
    		{
    		
    		if(b[i]==0)
    		{
    			b[i]=-1;
    			cnt++;
    		   for(j=0;j<a[i].size();j++)
    		     {
    		   	b[a[i][j]]--;
    		     }
    		  ch=1;
    		}
    	   }
        }
    	cout<<cnt<<endl;
    }
    return 0;
}
