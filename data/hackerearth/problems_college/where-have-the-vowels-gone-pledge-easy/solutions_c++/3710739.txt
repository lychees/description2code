#include <stdio.h>
int a[26];
int main()
{
   // printf("Hello World!\n");
   int n,i;
   char s[10001];
   scanf("%d",&n);
   scanf("%s",s);
   for(i=0;i<n;i++)
   {
   	a[s[i]-'a']++;
   }
   if(a[0]>0 && a['e'-'a']>0 && a['i'-'a']>0 && a['o'-'a']>0 && a['u'-'a']>0)
   		printf("YES\n");
   	else
   		printf("NO\n");
   
    return 0;
}
