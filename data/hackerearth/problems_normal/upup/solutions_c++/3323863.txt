#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define pii pair<int,int>
#define mp make_pair
#define pb push_back
#define si(a)     scanf("%d", &a)
#define sl(a)     scanf("%lld", &a)
#define pi(a)     printf("%d", a)
#define pl(a)     printf("%lld", a)
#define pm(str)   printf("%s\n",str)
#define pn        printf("\n")

int main(){
	string str;
	getline(cin,str);
	
	if(str[0] >= 'a' && str[0]<='z')
		str[0] = toupper(str[0]);
	for(int i = 1;i < str.length();i++){
		if(str[i-1] == ' ' && str[i]>='a' && str[i] <='z')
			str[i] = toupper(str[i]);
	}

	cout << str <<endl;
	return 0;
}
