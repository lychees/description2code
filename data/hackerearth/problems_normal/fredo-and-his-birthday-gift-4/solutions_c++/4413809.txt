#include <bits/stdc++.h>
using namespace std;
#define MOD 1000000007
#define MAXN 100

void check_using_dp(int adj[][MAXN], int n)
{
    int p=n+9;
        bool dp[n+9][1<<n];
        memset(dp,false,sizeof dp);
        for(int i=0; i<n; i++)
            dp[i][1<<i]=true;
        for(int i=0; i<(1<<n); i++)
        {
            for(int j=0; j<n; j++)
                if(i&(1<<j))
                {
                    for(int k=0; k<n; k++)
                        if(i&(1<<k) && adj[k][j] && k!=j && dp[k][i^(1<<j)])
                        {
                            dp[j][i]=true;
                            break;
                        }
                }
        }
        int mxsum[n];
        for(int i=0; i<n; i++)
        {
        	mxsum[i]=0;

            for(int j=0; j<(1<<n); j++)
            {

            	if(dp[i][j])
            	{

            		if(__builtin_popcount(j)>mxsum[i])
            		{
            			mxsum[i] = __builtin_popcount(j);
            		}
            	}
            }

        }
        for(int i=0;i<n;i++)
        {
        	cout<<mxsum[i]-1<<" ";
        }

        cout<<endl;
}

int main()
{
    //freopen("file1.txt","r",stdin);
    int T,n,m,x,y;
    cin>>T;
    while(T--)
    {
    	cin>>n>>m;
    	int adj[MAXN][MAXN];
    	memset(adj,0,sizeof adj);
    	for(int i=0;i<m;i++)
    	{
    		cin>>x>>y;
    		adj[x-1][y-1]=adj[y-1][x-1] = 1;
    	}
    	check_using_dp(adj,n);
    }
}
