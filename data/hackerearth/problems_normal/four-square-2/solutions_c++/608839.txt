#include <bits/stdc++.h>
#define sl(x) scanf("%lld",&x);
#define s(x) scanf("%d",&x);
#define ll long long
#define ull long long
#define pii pair <int, int>
#define vi vector <int>
#define pb push_back
#define mp make_pair

using namespace std;
int main()
{
	//sieve();
	int t; s(t);
	while (t--) {
		ll n; sl(n);
		ll sq = sqrt(n);
		ll sm = 0;
		if (n & 1) {
			// 8 times sum of divisor
			ll sq = sqrt(n);
			for (int i = 1; i <= sq; i++) {
				if (n % i == 0) {
					sm += i;
					if (i != n / i) sm += n / i;
				}
			}
			sm = sm * 8;
		} else {
			ll sq = sqrt(n);
			for (int i = 1; i <= sq; i++) {
				if (n % i == 0) {
					if (i & 1)sm += i;
					if (((n/i)&1) && (i != n / i)) sm += n / i;
				}
			}
			sm = sm * 24;
		}
		printf("%lld\n", sm);
	}
	return 0;
}
