         Do not go gentle into that good night,
         Old age should burn and rave at close of day;
         Rage, rage against the dying of the light.
         Though wise men at their end know dark is right,
         Because their words had forked no lightning they
         Do not go gentle into that good night.


*/
#include<bits/stdc++.h>
using namespace std;
#define pb push_back
#define MAXX 100000
vector<int>G[MAXX];
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		vector<int>v;
		v.pb(1);
		int carry=0;
		for(int i=1;i<=n;i++){
			for(int j=0;j<v.size();j++){
				int x=v[j]*i+carry;
				v[j]=x%10;
				carry=x/10;
			}
			while(carry){
				v.pb(carry%10);
				carry/=10;
			}
		}
		int sum=0;
		for(int i=0;i<v.size();i++)sum+=v[i];
		cout<<sum<<endl;
	}
	return 0;
}
