#define SET(x, a) memset(x, a, sizeof(x))

using namespace std;

int adj[105][105];
bool visited[105][105];
int cnt;
void dfs(int i, int j, int n)
{
	if ((i == n) && (j == n)) {
		cnt++;
		return;
	}
	visited[i][j] = true;
	if ((j + 1) <= n && visited[i][j + 1] == false && adj[i][j + 1] == 0) {
		dfs(i, j+ 1, n);
	} 
	if ((i + 1) <= n && visited[i + 1][j] == false && adj[i + 1][j] == 0) {
		dfs(i + 1, j, n);
	} 
	if ((j - 1) >= 1 && visited[i][j - 1] == false && adj[i][j - 1] == 0) {
		dfs(i, j - 1, n);
	}
	if ((i - 1) >= 1 && visited[i - 1][j] == false && adj[i - 1][j] == 0) {
		dfs(i - 1, j, n);
	}
	visited[i][j] = false;

	return;
}

int main()
{
	int t, n;
	//si(t);

		si(n);
		for (int i = 1; i <= n; i++) {
			for (int j = 1; j <= n; j++) {
				si(adj[i][j]);
			}
		}
		SET(visited, false);
		cnt = 0;
		dfs(1,1, n);
		cout << cnt << endl;
	//}
	
	return 0;
}
