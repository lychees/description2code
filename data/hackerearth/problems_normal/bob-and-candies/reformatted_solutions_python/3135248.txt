import itertools


def candies(N, X, A, S):
    if S <= N:
        a = list(itertools.combinations(A, S))
        for i in a:
            if sum(i) == X:
                return i
                break
        return candies(N, X, A, S+1)
    else:
        return '-1'


N, X = map(int, raw_input().split(' '))
A = map(int, raw_input().split(' '))
S = 1
res = candies(N, X, A, S)
if res == '-1':
    print '-1'
else:
    for i in res:
        print A.index(i)+1,
        A[A.index(i)] = None
