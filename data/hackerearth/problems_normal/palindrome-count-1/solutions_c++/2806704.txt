#include <iostream>
#include<string.h>
using namespace std;
int palin(char s[], int start, int end)
 {
 	for(int i=start;i<end;i++)
 	{
 		if(s[i]!=s[end])
 		{
 		 return 0;	
 		}
 		end--;
 	}
 	return 1;
 }

int main()
{
    char str[50];
    cin>>str;
    int n=strlen(str);
    int p_count=0;
    p_count=p_count+n;
    for(int i=0;i<n;i++)
    {
    	for(int j=n-1;j>i;j--)
    	{
    		if(palin(str,i,j))
    		  p_count++;
    	}
    }
    cout<<p_count;
    return 0;
}
