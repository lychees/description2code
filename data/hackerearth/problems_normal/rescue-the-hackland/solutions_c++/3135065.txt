#include <iostream>
using namespace std;
#include <bits/stdc++.h>
int main()
{
	ios::sync_with_stdio(false) ; 
	int t;
	cin >> t;
	while(t--){
		int l,r,k;
		cin >> l >> r >> k ; 
		int cont =0 ;
		int cur = l ; 
		while(cur <= r){
			cont++ ; 
			cur=cur*k +1 ;
		}
		cout << cont << endl;
	}
    return 0;
}
