void reverse(char *begin, char *end)
{
  char temp;
  while (begin < end)
  {
    temp = *begin;
    *begin++ = *end;
    *end-- = temp;
  }
}

int findwt(char* str)
{
	int len = 0;
	for(int i=0;i<strlen(str);i++)
	len+=(int)str[i];
	return len;
}
