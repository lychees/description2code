#include <iostream>
using namespace std;
int tim[100001],profit[100001];
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,k,max=-1;
    	long long pro;
    	cin>>n>>k;
    	for(int i=0;i<n;i++)
    	cin>>tim[i];
    	for(int i=0;i<n;i++)
    	cin>>profit[i];
    	
    	for(int i=0;i<n;i++)
    	{
    		int picks=k/tim[i];
    		pro=1;
    		pro=picks*profit[i];
    		//cout<<pro<<endl;
    		if(pro>max)
    		{
    			max=pro;
    		}
    	}
    	cout<<max<<endl;
    }
    return 0;
}
