for _ in xrange(int(raw_input())):
    q, n = map(int, raw_input().split())
    A = [n]
    for i in xrange(q):
        x = raw_input().split()
        if x[0] == 'P':
            A.append(x[1])
        else:
            z = A[-1]
            y = A[-2]
            A.pop()
            A.pop()
            A.append(z)
            A.append(y)
    print "Player", A.pop()
