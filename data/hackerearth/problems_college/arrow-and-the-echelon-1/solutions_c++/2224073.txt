#include <iostream>
using namespace std;

int main()
{
    int t,n,rem;
    cin>>t;
    while(t--)
    {
     rem=-1;
     cin>>n;
     for(int i=0;i<=n;i=i+5)
     {
        if((n-i)%3==0)
       {
      	rem=i;
      	break;
        }
     }
    if(rem==-1)
    {
     cout<<"-1"<<endl;
     continue;
    }
    for(int i=0;i<n-rem;i++)
    cout<<"5";
    for(int i=0;i<rem;i++)
    cout<<"3";
    cout<<endl;
   }
    return 0;
}
