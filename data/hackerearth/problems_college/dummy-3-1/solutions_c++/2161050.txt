#include <bits/stdc++.h>
using namespace std;

int main()
{
 int t;
 cin>>t;
 int n;
 while(t--){
 	cin>>n;
 	int a[n+1],b[n+1];
 	for(int i=1;i<=n;i++){
 		cin>>a[i];
 		
 	}
 	for(int i=1;i<=n;i++){
 		b[a[i]]=i;
 	}
 	int flag=0;
 	for(int i=1;i<=n;i++){
 		if(a[i]==b[i]){
 			flag=1;
 			continue;
 		}else{
 			flag=-1;
 			break;
 		}
 	}
 if(flag==1){
 	cout<<"inverse"<<"\n";
 }else{
 	cout<<"not inverse"<<"\n";
 }
 
 }
 return 0;
}
