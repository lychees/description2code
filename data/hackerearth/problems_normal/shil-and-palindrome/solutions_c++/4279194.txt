#include<iostream>
using namespace std;
int main(){
	string s;
	cin>>s;
	int a[26],i,count,x,l;
	char p;
	for(i=0;i<26;i++)
	a[i]=0;
	for(i=0;i<s.length();i++){
		++a[(int)s[i]-97];
	}
	for(i=0,count=0;i<26;i++){
		if(a[i]%2!=0){
			count++;
			if(count==2)
			break;
		}
	}
	if(count==2)
		cout<<-1;
	else{
		for(i=0,l=0;i<26;i++){
			if(a[i]%2==0){
				x=a[i]/2;
				a[i]=x;
				while(x!=0){
					cout<<(char)(i+97);
					--x;
				}
			}
			else{
				l=10;
				p=(char)(i+97);	
				--a[i];
				if(a[i]!=0){
						x=a[i]/2;
				a[i]=x;
				while(x!=0){
					cout<<(char)(i+97);
					--x;
				}
				}
			}
		}
		if(l==10)
		cout<<p;
		for(i=25;i>=0;i--){
			while(a[i]!=0){
				cout<<(char)(i+97);
				--a[i];
			}
		}
	}
	return 0;
}
