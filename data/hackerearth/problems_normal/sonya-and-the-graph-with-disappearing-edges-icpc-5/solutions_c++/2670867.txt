const int  inf=1e9+7;
const int N=1e5+5;
VPII g[N];
int diss[N];
int bfs(int s,int n)
{
	bool vis[N];
	int dist[N],i;
	queue<int> q;
	q.push(s);
	vis[s]=1;
	dist[s]=0;
	while(!q.empty())
	{
		int node=q.front();
		if(node == n)
			return dist[n];
		q.pop();
		int l=g[node].size();
		loop(i,l)
		{
			int v=g[node][i].F;
			int edgeno=g[node][i].S;
			if(!vis[v] and dist[node]+1<=diss[edgeno])
			{
				vis[v]=1;
				dist[v]=dist[node]+1;
				q.push(v);

			}
		}
	}
	return -1;
}
