#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
bool f[100001][101];
int set[101];
void pre(int n){
	for(int i=0;i<=n;i++){
		f[0][i]=true;
	}
	for(int i=1;i<100001;i++){
		f[i][0]=false;
	}
	for(int i=1;i<100001;i++){
		for(int j=1;j<=n;j++){
			f[i][j]=f[i][j-1];
			if(i>=set[j-1]){
				f[i][j]=f[i][j] || f[i-set[j-1]][j-1];
			}
		}
	}
}
int main()
{
	int t;
	cin>>t;
	int a[t];
	long long int s=0;
	for(int i=0;i<t;i++){
		cin>>set[i];
	}
	pre(t);
	int n;
	cin>>n;
	for(int i=0;i<n;i++){
		int h;
		cin>>h;
		int flag=0;
		for(int j=0;j<=t;j++){
			if(f[h][j]==true){
				printf("Yes\n");
				flag=1;
				break;
			}
		}
		if(flag==0){
			printf("No\n");
		}
	}

	
}
