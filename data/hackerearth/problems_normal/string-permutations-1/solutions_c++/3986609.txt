#include <bits/stdc++.h>
using namespace std;
//zone
int main()
{
    int t;
    cin >> t;
    int maxi = 0;
    map<string,int>m;
    map<string,int>::iterator it;
    while(t--)
    {
    	string s;
    	cin >> s;
    	m[s]++;
    	if(m[s] > maxi)
    	{
    		maxi = m[s];
    	}
    }
    int sum = 0;
    for(it = m.begin() ;it != m.end();it++)
    {
    	if(it->second == maxi)
    	{
    		sum += it->second;
    	}
    }
    cout<<sum<<endl;
    return 0;
}
