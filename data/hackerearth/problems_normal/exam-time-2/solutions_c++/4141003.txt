#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int phi[1000005];
void pre()
{
	int i,j;
	for(i=1;i<=1000000;i++)
	{
		phi[i]=i;
	}
	for(i=2;i<=1000000;i++)
	{
		if(phi[i]==i)
		{
			phi[i]=i-1;
			for(j=i*2;j<=1000000;j+=i)
			{
				phi[j]/=i;
				phi[j]*=(i-1);
			}
		}
	}
}
int main()
{
   	pre();
   	int t,n;
   	scanf("%d",&t);
   	while(t--)
   	{
   		scanf("%d",&n);
   		printf("%d\n",phi[n]);
   	}
    return 0;
}
