#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t-->0){
    	int a,b,c,d,s;
    	cin>>a>>b>>c>>d>>s;
    	if((c-s)>=a&&(c-s)<=b){
    		cout<<c-s<<endl;
    	}
    	else if(c-s>b){
    		cout<<b<<endl;
    	}
    	else{
    		cout<<"STAY IN LONDON"<<endl;
    	}
    }
    return 0;
}
