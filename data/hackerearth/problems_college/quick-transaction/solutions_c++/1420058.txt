{

    		}
    		else if(n>=500)
    		{    rem=n%500;
    		    m=n/500;
    			n=rem;
    			count=count+m;
    		}
    		else if(n>=100)
    		{   rem=n%100;
    		    m=n/100;
    			n=rem;
    			count=count+m;
    		}
    		else if(n>=50)
    		{ rem=n%50;
    		    m=n/50;
    			n=rem;
    			count=count+m;
    		}
    		else if(n>=20)
    		{ rem=n%20;
    		    m=n/20;
    			n=rem;
    			count=count+m;
    		}
    		else if(n>=10)
    		{ rem=n%10;
    		    m=n/10;
    		    	n=rem;
    			count=count+m;
    		}
    	    else if(n>=5)
    		{ rem=n%5;
    		    m=n/5;
    			n=rem;
    			count=count+m;
    		}
    	    else if(n>=2)
    		{ rem=n%2;
    		    m=n/2;
    			n=rem;
    			count=count+m;
    		}
    		else if(n>=1)
    		{ rem=n%1;
    		    m=n/1;
    			n=rem;
    			count=count+m;
    		}}
    	cout<<count<<endl;
    	}

}
