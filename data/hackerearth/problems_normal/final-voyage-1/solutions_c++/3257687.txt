#include <bits/stdc++.h>
using namespace std;

int w[1010],v[1010];
int dp[1010][1010];

int main()
{
    int t,n,mw,i,j;

    cin >> t;

    while (t--) {
        cin >> n;
        cin >> mw;
        for (i = 1; i <= n; i++)
            cin >> w[i];
        for (i = 1; i <= n; i++)
            cin >> v[i];

        for (i = 1; i <= n; i++) {
            for (j = 1; j <= mw; j++) {
                dp[i][j] = dp[i-1][j];
                if (j >= w[i]) {
                    dp[i][j] = max(dp[i][j], dp[i-1][j-w[i]] + v[i]);
                }
            }
        }
        cout <<dp[n][mw]<<endl;
    }
    return 0;
}
