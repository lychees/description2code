#include<bits/stdc++.h>
using namespace std;
char s[200];
int main()
{
    int t,i;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%s",s);
        int len=strlen(s);
        for(i=0;i<len/2;i++)
            if(s[i]!=s[len-1-i])
              break;
        if(i==len/2)
         printf("%d\n",s[len/2]);
        else
            printf("-1\n");
    }
return 0;
}
