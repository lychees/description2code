#include<bits/stdc++.h>
using namespace std;

typedef long long ll;

int cnt[16];

template<class T>
T comb(T val) {
    return (val*(val-1)*(val-2))/6;
}
int main() {
    int n;
    scanf("%d",&n);
    ll ans=0;
    for(int i=0;i<n;i++) {
        ll val;
        scanf("%lld",&val);
        int c=0;
        while(val) {
            int rem=val%10;
            if(rem==2) {
                c|=1;
            } else if(rem==3) {
                c|=2;
            } else if(rem==5) {
                c|=4;
            } else if(rem==7) {
                c|=8;
            }
            val/=10;
        }
        cnt[c]++;
    }
    for(int i=1;i<(1<<4);i++) {
        int bits=0;
        int j=i;
        while(j) {
            if(j&1) {
                bits++;
            }
            j>>=1;
        }
        ll total=0;
        for(int j=i;j<16;j++) {
            if((j&i)==i) {
                total+=cnt[j];
            }
        }
        if(bits%2) {
            ans+=comb(total);
        } else {
            ans-=comb(total);
        }
    }
    printf("%lld",ans);
    return 0;
}
