#include<bits/stdc++.h>
using namespace std;

int main()
{

    int t;
    cin>>t;

    while(t--)
    {
        char s[100];
        scanf(" %[^\n]s",s);

        int l=0;
        int weight=0;

        for(int i=0;s[i]!='\0';i++)
            {
                l++;
                weight+=(int)s[i];
            }

        weight=floor(weight/l);

        if(weight%2==0)
            {
                puts(s);
                cout<<endl;
            }
        else
        {
            for(int i=l-1;i>=0;i--)
                cout<<s[i];
                cout<<endl;
        }
    }
}
