#include <bits/stdc++.h>
typedef long long ll;

using namespace std;

ll BinaryStringToDecimal(string a)
{
    ll ans = 0, i;
    reverse(a.begin(), a.end());
    for (i = 0; i < a.length(); i++) {
        if (a[i] == '1')
            ans += (1 << i);
    }
    return ans;
}

void solve(ll n) {
    ll i, j, ans;
    string a;
    for (i = n; ; i++) {
        a = bitset<32>(i).to_string();
        ll f = 0;
        for (j = 0; j < 31; j++) {
            if ((a[j] - '0') && (a[j + 1] - '0')) {
                f = 1;
                break;
            }
        }
        if(!f) {
            ans = BinaryStringToDecimal(a);
            cout << ans << endl;
            return;
        }
    }

}

int main()
{
    ll t, n;
    cin >> t;
    while (t--) {
        cin >> n;
        solve(n);
    }

    return 0;
}
