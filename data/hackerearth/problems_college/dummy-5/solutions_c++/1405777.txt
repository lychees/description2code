using namespace std;
typedef long long ll;
int v[21],c[21],p[21],n;

bool isvalid(int gv,int gc,int gp,int i)
{
    if(i==n)
    {
        if(!gv && !gc && !gp)return true;
        return false;
    }
    if(gv<0 || gc<0 || gp<0)return false;
    bool flag;
    flag =  isvalid(gv-v[i],gc-c[i],gp-p[i],i+1);
    if(flag)
        return flag;
    else
    {
        return isvalid(gv,gc,gp,i+1);
    }
}

int main()
{
    int t,i,j,gv,gc,gp,ctr=1;
    cin>>t;
    while(t--)
    {
        cin>>gv>>gc>>gp;
        cin>>n;
        for(i=0;i<n;++i)
        {
            cin>>v[i]>>c[i]>>p[i];
        }
        if(isvalid(gv,gc,gp,0))
        {
            printf("Case #%d: yes\n",ctr++);
        }
        else
        {
            printf("Case #%d: no\n",ctr++);
        }
    }
    return 0;
}
