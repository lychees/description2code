#include <bits/stdc++.h>
#define LL long long
#define MOD 1000000007
using namespace std;

LL pascal[2100][2100];
void pasc()
{
	pascal[0][0]=1;
	for(int i=1;i<=2010;++i)
	{
		pascal[i][0]=1;
		for(int j=1;j<=i;++j)
		{
			pascal[i][j]=(pascal[i-1][j]+pascal[i-1][j-1])%MOD;
		}
	}
}
int main()
{
    pasc();
    int t,n,q,a,b;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d %d",&n,&q);
    	vector<pair<int,int> > v;
    	for(int i=0;i<q;++i)
    	{
    		scanf("%d %d",&a,&b);
    		v.push_back(make_pair(a,b));
    	}
    	v.push_back(make_pair(n-1,n-1));
    	sort(v.begin(),v.end());
    	int previ=0,prevj=0;
    	LL ans=1;
    	for(int i=0;i<v.size();++i)
    	{
    		int ni=v[i].first,nj=v[i].second;
    		LL d=pascal[ni-previ+nj-prevj][ni-previ];
    		ans*=d;
    		ans%=MOD;
    		previ=ni,prevj=nj;
    	}
    	printf("%lld\n",ans);
    }
    return 0;
}
