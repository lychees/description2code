#include<cstdio>
#include<iostream>
using namespace std;

int array[1005][3005];


int PRE();
int main()
{
	PRE();
	int T;
	cin>>T;
	while(T--)
	{
		int N;
		cin>>N;
		int t=0;
		for(int j=0;j<=3000;j++)
		t+=array[N][j];
		cout<<t<<endl;
	}
	return 0;
}
int PRE()
{
	array[0][3000]=1;
	for(int i=1;i<=1000;i++)
	{
		int c=0;
		for(int j=3000;j>=0;j--)
		{
			int rem=array[i-1][j]*i+c;
			array[i][j]=rem%10;
			c=rem/10;
		}
	}
	return 0;
}
