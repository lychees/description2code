#include <bits/stdc++.h>
using namespace std;

int main()
{
    ios::sync_with_stdio(false);
    int t;
    cin >> t;
    while(t--) {
    	string s, newstr = "";
    	cin >> s;
    	int l = s.length();
    	newstr = s[0];
    	for(int i = 1; i < l; i++) {
    		if(s[i] >= newstr[0]) newstr = s[i] + newstr;
    		else newstr += s[i];
    	}
    	cout << newstr << endl;
    }
    return 0;
}
