LL pwr( LL a,LL b) {
  if(b==0) {

  }
  LL ans=1;

    if(b&1) {
      ans=((ans%MOD)*(a%MOD))%MOD;

    a=((a%MOD)*(a%MOD))%MOD;
  }

}
