test = raw_input();
test = int(test);
for i in range(0,test):
    element = raw_input();
    check_list = []
    brchr = 0
    for j in range(0,26):
        check_list.append(0);
    for k in element:
        num = ord(k) - 97;
        check_list[num] = check_list[num] + 1;
        if check_list[num] > 1:
            print "Yes";
            brchr = 1
            break;
    if(brchr == 0):
        print "No";
