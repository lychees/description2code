num = int(raw_input().strip())
for i in xrange(num):
    n = raw_input().strip()
    if n.count('21'):
        print "The streak is broken!"
    elif int(n) % 21 == 0:
        print "The streak is broken!"
    else:
        print "The streak lives still in our heart!"
