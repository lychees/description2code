#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t , n , k;
    string s;
    
    cin >> t;
    
    while(t--){
        
        cin >> n >> k;
        s = "";
        
        int counter = 0, first = -1, second = -1;
        
        for(int i = 1; i < 10; i++){
            for(int j = 0; j < 10; j++){
                
                if(i == j) continue;
                counter++;
                if(counter == k){
                    
                    first = i;
                    second = j;
                    break;
                }
            }
        }
        
        if(first == -1) cout << first << "\n";
        else {
            
            for(int i = 0, j = 1; i < n; i++, j*= -1){
                
                if(j == 1) cout << first;
                else cout << second;
            }
            
            cout << "\n"; 
        }
    }
    
    return 0;
}
