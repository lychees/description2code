#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstring>
#include <iostream>
#include <map>
#include <queue>
#include <stack>
#include <string>
#include <vector>
using namespace std;
int a[100008];
int main() {
  int i, n, j;
  long long ans = 1ll << 60;
  scanf("%d", &n);
  for (i = 0; i < n; i++) {
    scanf("%d", &a[i]);
    if (i > 0)
      a[i] += a[i - 1];
  }
  for (i = 0; i < n; i++)
    for (j = i + 1; j - i <= 1008 && j < n; j++)
      ans = min(ans, (long long)(a[j] - a[i]) * (a[j] - a[i]) +
                         (long long)(j - i) * (j - i));
  printf("%I64d", ans);
  return 0;
}
