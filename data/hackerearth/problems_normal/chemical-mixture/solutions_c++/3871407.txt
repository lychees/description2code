#include <bits/stdc++.h>
using namespace std;


int main()
{
//    freopen("input.txt", "r", stdin);
	int nTest;
    cin >> nTest;
    while(nTest--){
    	long long m, n1, n2;
    	cin >> m >> n1 >> n2;
    	vector<long long> a1(n1);
    	vector<long long> a2(n2);
//    	cout << m << endl;
    	for (int i = 0; i < n1; ++i)
    		cin >> a1[i];
    	for (int i = 0; i < n2; ++i)
    		cin >> a2[i];

        sort(a1.begin(), a1.end());
    	reverse(a1.begin(), a1.end());

        long long best_sum = -1;
        vector<long long> best_res;

    	map<long long, bool> MAP; MAP.clear();
    	for (int mask = 0; mask < (1 << n2); ++mask){
            vector<long long> res;
            long long sum = 0;
    		long long val = 0;
    		for (int i = 0; i < n2; ++i)
    		if ((mask >> i) & 1)
    			val += a2[i];
            long long rem = m - val;
            if (rem < 0)
                continue;
            for (int i = 0; i < n1; ++i)
                if (a1[i] <= rem){
                    rem -= a1[i];
                    res.push_back(a1[i]);
                    sum += a1[i];
                }
            if (rem == 0){
                if (best_sum < sum){
                    best_sum = sum;
                    best_res = res;
                }
            }
    	}
    	if (best_sum == -1)
            cout << "NO" << endl;
        else {
            cout << "YES" << endl;
            for (int i = best_res.size() - 1; i >= 0; --i)
                cout << best_res[i] << ' ';
            cout << endl;
        }
    }
    return 0;
}
