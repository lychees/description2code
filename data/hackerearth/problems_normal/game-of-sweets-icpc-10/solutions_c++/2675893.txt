#include <bits/stdc++.h>

using namespace std;

#define pb push_back
#define mp make_pair
#define REP(i, n) for (int i = 0; i < (int)(n); ++i)
typedef long long LL;
typedef pair<int, int> PII;

int n;
bool a[100001] = {};
unsigned m[32][3130] = {};
unsigned ans[3130];

int main() {
    //freopen("input.txt", "r", stdin);
    scanf("%d", &n);
    int eq = 0;
    REP(i, n) {
        int x;
        scanf("%d", &x);
        if (a[x]) eq = 1;
        a[x] = true;
    }
    REP(i, 32) for (int j = i; j <= 100000; ++j) if (a[j]) {
        int v = j - i;
        m[i][v >> 5] |= 1 << (v & 31);
    }
    REP(i, 100001) if (a[i]) {
        int st = i >> 5;
        int sh = i & 31;
        for (int j = st; j < 3130; ++j) {
            ans[j - st] |= m[sh][j];
        }
    }
    int tot = eq;
    REP(i, 3130) tot += __builtin_popcount(ans[i]);
    --tot;
    printf("%d\n", tot);
    return 0;
}
