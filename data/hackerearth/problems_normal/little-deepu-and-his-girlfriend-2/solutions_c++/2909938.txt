#include <bits/stdc++.h>
using namespace std;

int main()
{
    int T;
    cin>>T;
    while(T--){
        int m,n;
        cin>>m>>n;
        vector<int> gameset(n);
          for(int i=0;i<n;i++){    
              scanf("%d",&gameset[i]);
          }
        int gt[m+1]; 
        for(int i=0;i<=m;i++){
            gt[i]=0;
        }
       
        
        for(int i=0;i<=m;i++){
            if(gt[i])continue;
            else{
                for(int j=0;j<n;j++){
                    if(gameset[j]+i<=m)
                       gt[i+gameset[j]]=1;
                }
                
            }
        }
        if(gt[m]){
            cout<<"Little Deepu\n";
        }
        else{
            cout<<"Kate\n";
        }
        
    }
  }
