#include <iostream>
#include<stdlib.h>
#include<stdio.h>
using namespace std;

int compare(const void * a, const void * b)
{
	return (*(int*)a - *(int*)b);
}

int main()
{
	int n, i, arr[10005];
    cin>> n;
    for(i=0;i < n;i++)
    	scanf("%d", &arr[i]);
    qsort(arr, n, sizeof(int), compare);
    for(i=0;i < n;i++)
    	printf("%d\n", arr[i]);
    return 0;
}
