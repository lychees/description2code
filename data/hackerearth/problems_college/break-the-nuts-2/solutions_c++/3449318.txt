#define pb push_back

using namespace std;

int main()
{
    int n;
    si(n);
    int arr[n+1];
    ll i,j;
    for(i=0;i<n;i++)
        si(arr[i]);
    ll sum[n+1];
    sum[0]=arr[0];
    for(i=1;i<n;i++)
        sum[i]=sum[i-1]+arr[i];
    ll ans=1;
    int prev=0;
    for(i=0;i<n;i++)
    {
        if(arr[i]==1)
        {
            prev=i;
            break;
        }
    }
    int fg=prev;
    for(i=1;i<n;i++)
    {
        if(sum[i]>sum[i-1]&&sum[i-1]>=1)
        {
            ans=(i-prev)*ans;
            //cout<<i<<" "<<prev<<endl;
            prev=i;
        }
    }
    if(fg==prev&&sum[prev]==0)
    printf("0");
    else
        cout<<ans;
    return 0;
}
