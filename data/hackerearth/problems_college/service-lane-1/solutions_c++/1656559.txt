#include <iostream>
#include<stdlib.h>
using namespace std;
int compare (const void * a, const void * b)
{
  return ( *(int*)a - *(int*)b );
}

int main() {
	int t,n;
	int a[1000001];
	int i,j;
	cin>> n>>t;
	for(int i=0;i<n;i++)
	cin>>a[i];
	
	while(t--){
		cin>>i>>j;
		int x=j-i+1;
		int y=0;
		int b[x];
		for(int k=i;k<=j;k++){
			b[y]=a[k];
			y++;
		}
		qsort (b, x, sizeof(int), compare);
//		for(int i=0;i<x;i++)

		cout<<b[0]<<endl;
	}
	return 0;
}
