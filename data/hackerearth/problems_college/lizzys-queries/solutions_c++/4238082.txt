#include <iostream>
#include <vector>
#include <stdio.h>
#include <algorithm>

using namespace std;

int check(char *c, int l, int r){
    int len = r-l+1,i;
    int mid = len/2;
    for( i=0; i<mid; ++i){
        if(c[l+i] != c[r-i]) break;
    }
    if(i == mid) return 1;
    else return 0;
}

int main(){
    int n,q,t,i,l,r;
    char x;
    cin>>n>>q;
    char a[n+1];
    scanf("%s",a);
    while(q--){
        scanf("%d",&t);
        if(t){
            scanf(" %d %d",&l,&r);
            if(check(a,l-1,r-1)){
                cout<<"Yes\n";
            }else{
                cout<<"No\n";
            }
        }else{
            scanf("%d %c",&i,&x);
            a[i-1] = x;
        }
    }
    return 0;
}
