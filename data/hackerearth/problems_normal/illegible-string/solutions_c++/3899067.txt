#include<iostream>
using namespace std;
#include<string.h>
int main()
{
	long long int i,m=0,k=0,cw=0,n,min,max,t;
	string s;
	cin>>t;
	cin>>s;
	i=0;
	n=t;
	while(i<=n)
	{
		if(s[i]=='v')
		{
			m++;
			i++;
			if(m==2)
			{
   			    k=k+1;
                m=0;
			}
			if(i==n)break;
		}
		else if(s[i]=='w')
		{
			cw++;
			i++;
			if(i==n)break;
		}
		else
		{
			m=0;
			i++;
			if(i==n)break;
		}
		if(i==n)break;
	}
	min=n-k;
	max=n+cw;
	cout<<min<<" "<<max<<endl;
	return 0;
}
