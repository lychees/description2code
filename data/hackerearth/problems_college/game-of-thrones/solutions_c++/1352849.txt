#include <bits/stdc++.h>
using namespace std;
int main() 
{
     int t=0;
     scanf("%d",&t);
     while(t--)
     {
     	long n=0,i=0,x=0;
     	set<long> s;
     	scanf("%d",&n);
     	for(i=0;i<n;i++)
     	{
     		scanf("%d",&x);
     		s.insert(x);
     	}
     	cout<<n-s.size()<<endl;
     }
    return 0;
}
