t = int(raw_input())
for i in range(t):
	string = raw_input().lower()
	r = 0
	u = 0
	b = 0
	y = 0
	count = []
	for i in string:
		if i == 'r':
			r += 1
		elif i == 'u':
			u += 1
		elif i == 'b':
			b += 1
		elif i == 'y':
			y += 1
	count.append(r)
	count.append(u)
	count.append(b)
	count.append(y)
	print min(count)
