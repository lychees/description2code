#include <bits/stdc++.h>

using namespace std;

int store [1000 + 10];
int temp [1000000 + 10];

int main()
{
    //ios::sync_with_stdio(false);cin.tie(0);

    int n , t;

    store[0] = 1;
    temp[1] = 1;

    int digit = 1;
    int j = 1;
    int remaindr = 0;
    int num;

    for(int i = 1; i <= 1000; i++){

        for(j = 1; j <= digit; j++){

           num = temp[j] * i + remaindr;
           temp[j] = num % 10;
           remaindr = num / 10;
        }

        if(remaindr > 0){

            while(remaindr != 0){

                temp[j++] = remaindr % 10;
                remaindr /= 10;
                digit++;
            }
        }

        for(int h = 1; h <= digit; h++){
            store[i] += temp[h];
        }
    }

    cin >>      t          ;

    while(t--){

        cin >> n;

        cout << store[n] << "\n";
    }

    return 0;
}
