#define loop(i,x,y) for(i=x;i<y;i++)

#define sd(x) scanf("%d",&x)

#define sllu(x) scanf("%llu",&x)

#define pd(x) printf("%d",x)
#define plld(x) printf("%lld",x)

typedef vector<int> vi;

typedef unsigned long long int llu;

int main()
{
   int n,s[100000],a[100000],i,flag,t;
   sd(t);
   while(t--)
   {
      sd(n);
      loop(i,0,n)
         sd(s[i]);
      loop(i,0,n)
         sd(a[i]);
      sort(s,s+n);
      sort(a,a+n);
      flag=0;
      loop(i,0,n)
         if(s[i]<=a[i])
         {
            flag=1;
            break;
         }
      if(flag)
         pstr("NO\n");
      else
         pstr("YES\n");
   }

   return 0;
}
