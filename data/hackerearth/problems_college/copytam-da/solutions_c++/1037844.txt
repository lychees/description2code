#include <iostream>
using namespace std;

int main()
{
    int t,x,n,i,ans;
    cin>>t;
    while(t--)
    { ans=0;
    	cin>>n;
    	int a[10000]={0};
    	for(i=0;i<n;i++)
    	{
    		scanf("%d",&x);
    		a[x]++;
    	}
    	for(i=0;i<10000;i++)
    	{
    		if(a[i]>0)
    		ans++;
    	}
    	cout<<ans<<endl;
    }
}
