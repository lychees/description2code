#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int t,n;
	cin>>t;
	while(t--)
    {
		int a,b,i;	
	cin>>a>>b;
	cin>>n;
	int p[n];
	for(i=0;i<n;i++)
	{
		cin>>p[i];
	}
	int c=0,d=0,sum=0;
	sort(p,p+n);
	for(i=0;i<n;i++)
	{
		sum=sum+p[i];
		
		if(sum<=a)
		{
			c++;
		}
		if(sum<=b)
		{
			d++;
		}
		else if(sum>a||sum>b)
		{
			break;
		}
		}
		if(c>d)
		{
			cout<<"Raghu Won"<<"\n";
		}
		else if(c<d)
		{
			cout<<"Sayan Won"<<"\n";
		}
		else if(c==d)
		{
			cout<<"Tie"<<"\n";
		}
		
    }
}
