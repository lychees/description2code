#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    string s1,s2,s3,s4,s5,s6;
    cin>>t;
    while(t--)
    {
    	cin>>s1;
    	cin>>s2;
    	s3 = s1 + s2;
    	s4 = s2 + s1;
    	s5= s3;
    	s6= s4;
    	reverse(s5.begin(), s5.end());
    	reverse(s6.begin(), s6.end());
    	if(s3.compare(s5)==0||s4.compare(s6)==0)
    	cout<<"YES"<<endl;
    	else
    	cout<<"NO"<<endl;
    }
    
}
