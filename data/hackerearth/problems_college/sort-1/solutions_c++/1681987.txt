#include<iostream>
#include<algorithm>
#include<stdio.h>

typedef long long int ll;

#define s(x) scanf("%lld",&x)

using namespace std;

void merge(int arr[],int l,int m,int r)
{
	int i,j,k;
	int n1 = m-l+1;
	int n2 = r-m;
	
	int L[n1],R[n2];
	
	for(i=0;i<n1;i++)
		L[i]=arr[l+i];
	
	for(j =0;j<n2;j++)
		R[j] = arr[m+1+j];
	
	i=0;
	j=0;
	k=l;
	
	while(i<n1 && j<n2)
	{
		if(L[i]<=R[j])
		{
			arr[k] = L[i];
			i++;
		}
		else
		{
			arr[k] = R[j];
			j++;
		}
		k++;
	}
	
	while(i<n1)
	{
		arr[k] = L[i];
		i++;
		k++;
	}
	
	while(j<n2)
	{
		arr[k] = R[j];
		j++;
		k++;
	}
}

void mergesort(int arr[],int i,int j)
{
	if(i<j)
	{		
	int mid = i+(j-i)/2;
	mergesort(arr,i,mid);
	mergesort(arr,mid+1,j);
	merge(arr,i,mid,j);
	}
}

int main()
{
	int t;
	cin>>t;
	
	while(t--)
	{
		int n;
		cin>>n;
		
		int arr[n];
		
		for(int i = 0;i<n;i++)
			cin>>arr[i];
			
		
		mergesort(arr,0,n-1);
	
		for(int i = 0;i<n;i++)
			cout<<arr[i]<<" ";
			
		cout<<"\n";
		
	}
	return 0;

}
