void fun()
{
	sum = 0;
	scanf("%d", &n);
	remain.resize(n);
	capacity.resize(n);
	cum.resize(n);
	for(int i = 0; i < n; i++) {
		scanf("%d", &remain[i]);
		sum += remain[i];
	}
	for(int i = 0; i < n; i++) {
		scanf("%d", &capacity[i]);
	}
	sort(capacity.begin(), capacity.end());
	reverse(capacity.begin(), capacity.end());
	cum[0] = capacity[0];
	for(int i = 1; i < n; i++) {
		cum[i] = cum[i - 1] + capacity[i];
	}
	for(int i = 0; i < n; i++) {
		if(cum[i] >= sum) {
			printf("%d\n", i + 1);
			return;
		}
	}
	
}

int main()
{
	int t;
	scanf("%d", &t);
	while(t--) {
		fun();
	}
	return 0;
}
