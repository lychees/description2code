#include<bits/stdc++.h>
using namespace std;

int main()
 {
 	
 	int t,n,i,j;
 	cin>>t;
 	while(t--)
 	 {
 	 	cin>>n;
 	 	vector<int> a(n),b(n);
 	 	for(i=0;i<n;i++)
 	 	 cin>>a[i];
 	 	for(i=0;i<n;i++)
 	 	 cin>>b[i];
 	 	  int ans=0;
			sort(a.begin(),a.end());
 	 	  sort(b.begin(),b.end());
 	 	  for(i=0;i<n;i++)
 	 	   {
 	 	   	ans=max(ans,abs(a[i]-b[i]));
 	 	   }
 	 	   cout<<ans<<endl;
 	 	  int best=-100,d=INT_MAX;
 	 	   for(j=-100;j<=100;j++)
 	 	    {
 	 	    	int dist=0;
 	 	    	for(i=0;i<n;i++)
 	 	    	 {dist+=abs(a[i]-j);
 	 	    	 }
 	 	    	if(dist<d)
 	 	    	  {d=dist;
 	 	    	  best=j;
 	 	    	  }
 	 	    	
 	 	    }
 	 	    cout<<best<<endl;
 	 }
 }
