#include<bits/stdc++.h>
#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long int n,x,y,i,d,ans=0,minx,maxx;
    vector < pair<long long int,long long int> > v;
    cin>>n;
   	for(i=0;i<n;++i)
    {
    	cin>>x>>y>>d;
    	v.push_back(make_pair(x,x+d));
    }
    sort(v.begin(),v.end());
    ans = v[0].second-v[0].first+1;
    minx = v[0].first;
    maxx = v[0].second;
    for(i=1;i<n;++i)
    {
    	if(v[i].first>maxx)
    	{
    		ans+=v[i].second-v[i].first+1;
    	}
    	else if(v[i].second>maxx)
    	{
    		ans+=v[i].second-maxx;
    	}
    	minx = min(minx,v[i].first);
    	maxx = max(maxx,v[i].second);
    }
    cout<<ans<<endl;
    return 0;
}
