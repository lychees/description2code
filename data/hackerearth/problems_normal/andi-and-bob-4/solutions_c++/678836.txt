#include <iostream>
#include<cstdio>
using namespace std;

int main()
{
    long long t,n,*arr,*odd,*even;
    scanf("%lld",&t);
    while(t--)
    {
    	scanf("%lld",&n);
    	arr=new long long[n+1];
    	odd=new long long[n+1];
    	even=new long long[n+1];
    	for(int i=1;i<=n;i++)
    	scanf("%lld",&arr[i]);
    	if(arr[1]%2==0)
    	{
    		even[1]=1;
    		odd[1]=0;
    	}
    	else
    	{
    		odd[1]=1;
    		even[1]=0;
    	}
    	for(int i=2;i<=n;i++)
    	{
    		if(arr[i]%2==0)
    		{
    			even[i]=even[i-1]+1;
    			odd[i]=odd[i-1];
    		}
    		else
    		{
    			even[i]=odd[i-1];
    			odd[i]=even[i-1]+1;
    		}
    	}
    	long long evcnt=0,oddcnt=0;
    	for(int i=1;i<=n;i++)
    	{
    		evcnt+=even[i];
    		oddcnt+=odd[i];
    	}
    	if(evcnt>oddcnt)
    	printf("ANDI\n");
    	else
    	printf("BOB\n");
    }
    return 0;
}
