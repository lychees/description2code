#include <iostream>
#include <string.h>
#include <stdlib.h>
using namespace std;

char A[256]="";
char B[256]="";

void calculate(int& minSum, int& maxSum)
{
	char minA[256]="";
	char minB[256]="";
	strcpy(minA, A);
	strcpy(minB, B);
	int sizeA=strlen(minA);
	int sizeB=strlen(minB);
	
	// cout<<A<<" "<<B<<endl;
	// cout<<minA<<" "<<minB<<endl;
	
	for(int i=0; i<sizeA; i++)
	{
		if(minA[i]=='6')
		{
			minA[i]='5';
		}
		if(A[i]=='5')
		{
			A[i]='6';
		}
	}
	
	for(int i=0; i<sizeB; i++)
	{
		if(minB[i]=='6')
		{
			minB[i]='5';
		}
		if(B[i]=='5')
		{
			B[i]='6';
		}
	}
	
	// cout<<A<<" "<<B<<endl;
	// cout<<minA<<" "<<minB<<endl<<endl;
	
	minSum=atol(minA)+atol(minB);
	maxSum=atol(A)+atol(B);
}

int main()
{
    // cout << "Hello World!" << endl;
    cin>>A>>B;
    // cout<<A<<" "<<B<<endl;
    int minSum, maxSum;
    calculate(minSum, maxSum);
    cout<<minSum<<" "<<maxSum;
    return 0;
}
