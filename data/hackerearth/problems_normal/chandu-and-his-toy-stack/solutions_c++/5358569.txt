#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
#define FOR(i,a,b) for(LL i = (LL )a; i < (LL )b ; i++)
#define mp make_pair
#define pb push_back
typedef vector<long long> V;
typedef pair<int,int> pii;
typedef pair<long long,long long> P;


int main()
{
    LL T,N,K,M,x,y;
    LL i,j,k,it;
    LL mn,mx,temp,init,pos,ans;
    int flag;
    LL ce,co;
    V v,s;
    
    cin>>T;
    while(T--)
    {
        cin>>N>>x>>y;
        ans=0;
        for(i=0;i<N;i++)
        {
            cin>>ce>>co;
            v.pb(ce);
            s.pb(co);
        }
        sort(v.begin(),v.end());
        sort(s.begin(),s.end());
        for(i=0;i<N;i++)
        {
            if(v[i]>s[i])
                ans+=(v[i]-s[i])*y;
            else
                ans+=(s[i]-v[i])*x;
        }
        cout<<ans<<endl;
        v.clear();
        s.clear();
    }
        
	return 0;

    
}
