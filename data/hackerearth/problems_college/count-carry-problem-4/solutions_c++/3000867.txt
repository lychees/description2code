#include <bits/stdc++.h>
using namespace std;;

int main()
{
        int n;
        long long int a, b;
        long long int i, j;
        scanf("%d", &n);
        while (n--) {
                scanf("%lld", &a);
                scanf("%lld", &b);
                long long int t1, t2, res, c;
                t1 = t2 = res = c = 0;
                while((a > 0) && (b > 0)) {
                        t1 = a % 10;
                        t2 = b % 10;
                 //       cout << t1 << " " << t2 << endl;
                        if ((t1 + t2 + c) > 9) {
                                res++;
                                c = 1;
                        } else {
                                c = 0;
                        }
                        a = a / 10;
                        b = b / 10;
                }
                if (res == 0) {
                        cout << "No carry operation" << endl;
                } else if (res == 1){
                        cout << res << " carry operation" << endl;
                } else {
                       cout << res << " carry operations" << endl; 
                }
        }
        return 0;
}
