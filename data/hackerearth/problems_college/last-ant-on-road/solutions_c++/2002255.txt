#include <bits/stdc++.h>

using namespace std;
int main(){
	int T;
	cin>>T;
 int N;
 while(T--){
 cin>>N;
 int A[N];
 int i;
 for(i=0; i<N; i++){
  cin>>A[i];
 }
 int l = 0, r = 0;
 for(i=0; i<N; i++){
  if(A[i] == -1){
   l ++;
  }
  else{
   break;
  }
 }
 
 for(i=N-1; i>=0; i--){
  if(A[i] == 1){
   r++;
  }
  else{
   break;
  }
 }
 int lant = 0;
 for(i=l; i <= N-r-1; i++){
  if(A[i] == -1){
   lant ++;
  }
 }
 if(l >= r){
  cout<<l+lant<<endl;
 }
 else{
  cout<<l+lant+1<<endl;
 }
 }
 return 0;
}
