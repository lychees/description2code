#include <iostream>
#include <vector>
using namespace std;

typedef vector<int> VV;
typedef vector<VV> Vv;
int main()
{
    
    	int N,M;
    	cin>>N>>M;
    	int Ar[105][105];
    	for(int i=0;i<N;i++)
    		for(int j=0;j<M;j++)
    			cin>>Ar[i][j];
    	Vv V(1000);
    	for(int i=0;i<N;i++)
    	{
    		for(int j=0;j<M;j++)
    			V[i+j].push_back(Ar[i][j]);
    	}
    	for(int i=0;i<M+N-1;i++)
    	{
    		for(int j=0;j<V[i].size();j++)
    			cout<<V[i][j]<<" ";
    		cout<<"\n";
    	}
    	return 0;
    
}
