#include<iostream>
#include<stdio.h>
using namespace std;

int main()
{
char arr[100000];
int a,b,x,y,n,q;
scanf("%d %d",&n,&q);

for(int i=0; i<n;i++)
 cin>>arr[i];

while(q--)
{
scanf("%d %d",&x,&y);

a=x;
b=y;
while(a!=b)
{
    if(arr[a-1]=='1')
        break;
    if(arr[b-1]=='1')
        break;
  if(a>b)
    a=a/2;
  else
    b=b/2;
}
if(a==b && arr[a-1]=='0')
    printf("Yes\n");
else
    printf("No\n");
}

return 0;
}
