#include <iostream>
#include<map>
using namespace std;

int main()
{
	int T;
	cin>>T;
	while(T--)
	{
		int n;
		string s;
		cin>>n;
		cin>>s;
		map<char,int> freq;
		int i=0;
		while(i<n)
		{
			freq[s[i]]++;
			
			i++;
		}
		
		int max=0;
		for(auto j=freq.begin();j!=freq.end();j++)
		{
			//cout<<j->first<<" "<<j->second<<endl;
			if(max<j->second)
			{
				max=j->second;
			}
		}
		
		if(max%2==0)
			cout<<"Yes";
		else
			cout<<"No";
		cout<<endl;
	}
    
    
    return 0;
}
