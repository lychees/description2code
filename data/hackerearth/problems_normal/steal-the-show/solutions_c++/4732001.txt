#include <bits/stdc++.h>
using namespace std;

int main()
{
    int ceoh,ctoh,cooh;
    cin>>ceoh>>ctoh>>cooh;
    int n;
    cin>>n;
    vector<int> v;
    int temp;
    for(int i=0;i<n;i++)
    {
    	cin>>temp;
    	v.push_back(temp);
    }
    int len=v.size();
    sort(v.begin(),v.end());
    int min,max;
    if(ctoh>cooh)
    {
    	min=cooh;
    	max=ctoh;
    }
    else
    {
    	min=ctoh;
    	max=cooh;
    }
    vector<int>::iterator it=v.begin();
    v.insert(it+n,max);
    it=v.begin();
    v.insert(it,min);
    if(len%2==0)
    {
    	it=v.begin();
    	len=v.size();
    	v.insert(it+len/2,ceoh);
    }
    else
    {
    	it=v.begin();
    	len=v.size();
    	int ans;
    	if(abs(v[(len/2)-1]-ceoh)+abs(v[(len/2)]-v[(len/2)+1])<abs(v[(len/2)+1]-ceoh)+abs(v[(len/2)]-v[(len/2)-1])) ans=(len/2)-1;
    	else ans=(len/2)+1;
    	v.insert(it+ans,ceoh);
    }
    int h=0;
    for(int i=0;i<n+2;i++)
    {
    	//cout<<v[i]<<" ";
    	h+=abs(v[i]-v[i+1]);
    }
    cout<<h;
    return 0;
}
