#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	unsigned long long int a[n],b[n],i,j,sum;
	for(i=0;i<n;i++)
	{
		cin>>b[i];
	}
	for(i=0;i<n;i++)
	{
		sum=b[i]*(i+1);
		for(j=0;j<i;j++)
		{
			sum=sum-a[j];
		}
		a[i]=sum;
	}
	for(i=0;i<n;i++)
		cout<<a[i]<<" ";
}
