#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int tc;
    cin>>tc;
    while(tc--)
    {
    	int n,i,j;
    	cin>>n;
    	int arr[n];
    	for(i=0;i<n;i++)
    	{
    		cin>>arr[i];
    		
    	}
    	long long ans=0;
    	for(i=0;i<n;i++)
    	{
    		for(j=i;j<n;j++)
    	    if(arr[i]==arr[j])
    	    ans++;
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
