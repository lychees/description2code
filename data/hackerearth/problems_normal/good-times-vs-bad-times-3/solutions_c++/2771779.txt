#include <bits/stdc++.h>
using namespace std;

int sec[86400];

inline int gcd(int a, int b)
{
	int tmp;
	while(a)
	{
		tmp = a;
		a = b % a;
		b = tmp;
	}
	return b;
}

int main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);
    for(int i = 0; i < 86400; ++i)
    {
    	int h, m, s, tmp = i;
    	s = tmp;
    	s %= 60;
    	tmp /= 60;
    	m = tmp;
    	m %= 60;
    	tmp /= 60;
    	h = tmp;
    	if(gcd(gcd(h,m), s) != 1)
    		sec[i] = 1;
    }
    for(int i = 86398; i >= 0; --i)
    sec[i] += sec[i+1];

    int t; cin>>t;
    for(int i = 0; i < t; i++)
    {
    	int h,m,s,ans;
    	cin>>h>>m>>s;
    	ans = s + (m + h*60)*60;
    	int a = sec[ans], b = (86400-ans-sec[ans]), c = ((a&&b)?gcd(a,b):1);
    	cout << a/c << ":" << b/c << "\n";   
    }
    return 0;
}
