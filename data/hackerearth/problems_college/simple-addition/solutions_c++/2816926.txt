#include <bits/stdc++.h>
using namespace std;
int r;
int MAX=100000000;
int gmd(int a,int b)
{
	if(b<=0)return MAX;
	if(a>1&&b==1)
		return a-1;
	
		return gmd(b,a - a/b*b)+a/b;
}
int main(void)
{
	int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		if(n==1){
		cout<<n-1<<endl;
		continue;}
		r = MAX;
		for(int i=1;i<=n/2;i++)
			r=min(r,gmd(n,i));
		cout<<r<<endl;
	}
}
