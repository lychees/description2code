#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
	ll t,k,i,n;
	char s[1000];
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%s",s);
		n=strlen(s);
		k=n/2;
		ll c=0;
		for(i=0;i<k;i++)
		{
           if(s[i]==s[n-i-1])
           	c++;
		}
		if(c==k)
			printf("YES\n");
		else
			printf("NO\n");
	}
	return 0;
}
