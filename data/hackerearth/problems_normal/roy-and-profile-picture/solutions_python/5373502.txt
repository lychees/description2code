l = input()
n = input()
for i in range(1,n+1):
	dim_list = map(lambda x:int(x), raw_input().split())
	if (dim_list[0]<l or dim_list[1]<l):
		print "UPLOAD ANOTHER"
	elif (dim_list[0]==dim_list[1]):
		print "ACCEPTED"
	else:
		print "CROP IT"
