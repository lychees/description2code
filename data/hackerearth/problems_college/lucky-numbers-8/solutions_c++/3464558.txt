#include<bits/stdc++.h>
using namespace std;
int main()
{
	int tc;
	cin>>tc;
	while(tc--)
	{
		int n;
		cin>>n;
		vector<int>V(n+1);
		vector<int>S(n+1);
		for(int i=1;i<=n;++i)
			cin>>V[i];
		S[n]=V[n];
		for(int i=n-1;i>=1;--i)
		{
			if(V[i]>S[i+1])
				S[i]=V[i];
			else S[i]=S[i+1];
		}
		int ans=0;
		for(int i=1;i<=n-1;++i)
		{
			if(V[i]>S[i+1])
				ans++;
		}
		ans+=1;
		cout<<ans<<endl;
	}
	return 0;
}
