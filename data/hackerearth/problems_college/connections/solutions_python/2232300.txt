t=int(raw_input());
n=int(raw_input());
c1=[];
c2=[];
for i in range(0,n):
	a,b=map(int,raw_input().split());
	c1.append(a);
	c2.append(b);
c=int(raw_input());
for i in range(0,c):
	f=0;
	e1,e2=map(int,raw_input().split());
	for j in range(0,n):
		if c1[j]==e1 and c2[j]==e2:
			print("Direct Connection");
			f=1;
			break;
		elif c1.count(e1)==0 or c2.count(e2)==0:
			print("No Connection");
			f=1;
			break;
	if f==0:
		print("Indirect Connection");
