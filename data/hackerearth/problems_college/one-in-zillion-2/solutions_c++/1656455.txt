#include <iostream>
#include <vector>
using namespace std;

vector<vector<int> > power;
int store_power()
{
	vector<int> temp;
	temp.push_back(1);
	power.push_back(temp);
	temp[0]=2;
	power.push_back(temp);
	temp[0]=4;
	power.push_back(temp);
	temp[0]=8;
	power.push_back(temp);
	for(int i=3;i<=10000;i++)
	{
		temp.clear();
		int carry=0;
		for(int j=0;j<power[i].size();j++)
		{
			int x=power[i][j]*2+carry;
			if(x>=10)
				carry=1;
			else
				carry=0;
			temp.push_back(x%10);
		}
		if(carry)	
			temp.push_back(1);
		power.push_back(temp);
	}
}
int main()
{
	int t;
	store_power();
	cin>>t;
	while(t--)
	{
		int N;
		cin>>N;
		int x=N/30+1;
		for(int i=power[x].size()-1;i>=0;i--)
			cout<<power[x][i];
		cout<<endl;
	}
	return 0;
}
