#include <stdio.h>
#include<iostream>
using namespace std;

int main()
{
        int test_cases,x1,x2,y1,y2,count;
        scanf("%d",&test_cases);
        while(test_cases > 0)
        {
                count=0;
                scanf("%d %d %d %d",&x1,&y1,&x2,&y2);
                if(x2==y1)
                count++;
                if(x1==y2)
                count++;
                cout<<count<<endl;
               
                
                test_cases --;
        }
        return 0;
}
