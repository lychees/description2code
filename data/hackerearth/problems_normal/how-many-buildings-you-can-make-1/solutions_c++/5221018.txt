#include<iostream>

using namespace std;

int main()
{

    int test;
    cin>>test;
    while(test--)
    {
        int a;
        cin>>a;
        int value, count= 0;
        for(int i =1;i<1000;i++)
        {
            int rem = a/i ;
            int value = rem - i+1;
            if(value>0)
            {
                count+=value;;
            }
            else
                break;

        }

        cout<<count<<endl;
    }
}
