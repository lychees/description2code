#include <bits/stdc++.h>
using namespace std;

int main()
{
    int d,wt,t,ans;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d%d",&d,&wt);
    	if(d>60)
    	{
    		printf("NA\n");
    	}
    	else
    	{
    		if(d<3)
    		{
    			ans=50+wt*2;
    		}
    		else
    		{
    			ans=50+(d-2)*12+wt*2;
    		}
    		printf("%d\n",ans);
    	}
    }
    return 0;
}
