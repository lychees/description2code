#include <iostream>
using namespace std;

int main()
{
    int t, n, m;
    cin >> t;
    for(int i = 0; i < t; i++){
    	cin >> n >> m;
    	int a[n], b[m];
    	for(int j = 0; j < n; j++)
    		cin >> a[j];
		for(int j = 0; j < m; j++)
			cin >> b[j];
		int temp[n + m];
		int c = 0, d = 0, l = 0;
		while(c < n && d < m){
			if(a[c] > b[d]){
				temp[l++] = a[c++];
			}
			else{
				temp[l++] = b[d++];
			}
		}
		while(d < m)
			temp[l++] = b[d++];
		while(c < n)
			temp[l++] = a[c++];
		for(int j = 0; j < n + m; j++)
			cout << temp[j] << " ";
		cout << endl;
    }
    return 0;
}
