#include <iostream>
#include <cstdio>
#include <cmath>
#define ul unsigned long long
#define mod 1000000007
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	ul n;
    	scanf("%llu",&n);
    	ul bit = log2(n);
    	ul sum=0;
    	for(ul i=0;i<bit;i++)
    	{
    		ul a=1;
    		if(i!=0)
    			a = a << i;	
    		for (ul j=i+1;j<=bit;j++)
    		{
    			ul b=1 ;
    			b = b << j;
    			if ((a+b)<=n)
    			{
    				sum += a+b;
    				sum %= mod;
    			}	
    		}
    	}
    	printf("%llu\n",sum);	
    }
    return 0;
}
