#include <iostream>
#include <string>
using namespace std;

int main()
{
	int t,n,i;
	cin>>t;
	string s;
	while(t--){
		cin>>n>>s;
		long long int count=1;
		long long sum=0;
		for(i=1;i<s.length();i++)
		{
			
			if(s[i]>=s[i-1])
			     count++;
			else{
				sum+=(count)*(count+1)/2LL;
				count=1;
			}
		}
	sum+=(count)*(count+1)/2LL;
	cout<<sum<<endl;
	}
	return 0;
}
