#include <iostream>
#include <cstring>
using namespace std;

int main() {
	int alpha[26], l, max, i;
	bool inH, inC;
	char s[4001], ans;
	cin>>s;
	l = strlen(s);
	for(i=0;i<l;i++)
		alpha[(s[i]-'a')]++;
	inH = false;
	inC = false;
	max = 0;
	for(i=0;i<26;i++)
		if(alpha[i]>alpha[max])
			max = i;
	ans = max + 'a';
	if(ans=='h'||ans=='e'||ans=='a'||ans=='d'||ans=='r'||ans=='u'||ans=='s')
		inH = true;
	if(ans=='c'||ans=='u'||ans=='b'||ans=='i'||ans=='n'||ans=='g'||ans=='l')
		inC =  true;
	if(inH && inC)
		cout<<"Thanks both";
	else if(inH && !inC)
		cout<<"Thanks head rush";
	else if(!inH && inC)
		cout<<"Thanks cubing club";
	else if(!inH && !inC)
		cout<<"none";
	return 0;
}
