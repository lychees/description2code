#include <iostream>
using namespace std;
int Arr[2000007];
 
int main()
{  int T,N,m1,m2,i;
    cin>>T;
    Arr[0]=0;
   
    while(T--)
    {
    	cin>>N>>m1>>m2;
      if(m1==1||m2==1)
       Arr[1]=1;
       else
       Arr[1]=0;
      for (int i=1;i<=N;i++)
       {
       	if(i>=m1&&!Arr[i-m1])
       	   Arr[i]=1;
       	else if (i>=m2&&!Arr[i-m2])
       	   Arr[i]=1;
       	else Arr[i]=0;
       }
       cout<<Arr[N]<<"\n";
    }
    return 0;
}
