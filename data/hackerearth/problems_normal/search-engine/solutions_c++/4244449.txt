struct trie
{
    trie *ch[27];
    int cnt;
    trie()
    {
        for(int i=0;i<27;i++) ch[i]=NULL;
        cnt=0;
    }
};

void ADD(trie *root,string s,int w)
{
//    root->cnt=max(w,root->cnt);
    for(int i=0;i<(int)s.size();i++)
    {
        int c=s[i]-'a';
        if(root->ch[c]==NULL)
        {
            root->ch[c]=new trie();
            root->ch[c]->cnt=w;
        }

        root->ch[c]->cnt=max(w,root->ch[c]->cnt);
        root=root->ch[c];
    }
}
int query(trie *root,string s)
{
    for (int i = 0; i < (int)s.size(); i++)
    {
        int p=s[i]-'a';
        if(root->ch[p]==NULL)
        {
            return -1;
        }
        if(i==s.size()-1)
        {
            return (root->ch[p]->cnt);
        }
        root=root->ch[p];
//        cout<<root->cnt<<" ";
    }
}
