#include<bits/stdc++.h>
using namespace std;
int main()
{
    int a,b,n;
    cin >> a >> b >> n;
    int i=1;
    int pres=0;
    for(;;)
    {
        if(pres+a>=n)
        {
            cout << i << endl;
            break;
        }
        pres=pres+a-b;
        i++;

    }
    return 0;
}
