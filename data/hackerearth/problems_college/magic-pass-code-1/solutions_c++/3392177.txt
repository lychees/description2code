#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s, hi;
    int maxi = 0;
    while(cin >> s){
    	int sum = 0;
    	for(int i=0; i<s.size(); i++){
    		if(s[i] != '.'){
    			sum += (s[i]-'0');
    		}
    	}
    	if(sum > maxi){
    		maxi= sum;
    		hi = s;
    	}
    }
    cout << hi << endl;
    
    return 0;
}
