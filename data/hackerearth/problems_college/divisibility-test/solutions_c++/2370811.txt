#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	char arr[51]={0};
    	cin>>arr;
    	int k;
    	cin>>k;
    	int l,r;
    	cin>>l>>r;
    	l--;r--;
    	int s = 0;
    	int res =0;
    	for(int i = l;i<=r;i++)
    	{
    		s = 0;
    		for(int j = i;j<=r;j++)
    		{
    			s = (s*10+arr[j]-'0')%k;
    			if(s == 0)
    			res++;
    		}
    	}
    	cout<<res<<endl;
    }
    return 0;
}
