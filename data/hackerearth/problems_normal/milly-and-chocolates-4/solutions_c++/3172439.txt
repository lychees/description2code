#include <iostream>
#include <vector>
#include <map>
#include <algorithm>
#include <cstdio>

int main(void)
{
    short T;
    long N;
    short p;
    long long res = 0;

    std::cin>>T;
    while(T--)
    {
        std::cin>>N;
        res = 0;
        for(long i = 0;i<N;++i)
        {
            std::cin>>p;
            res += p;
        }

        printf("%.8lf\n",((res/(N*100.0))*100.0));

    }
    return 0;
}
