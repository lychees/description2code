def number_of_set_bits(N):
    count = 0
    while N:
        count += N & 1
        N >>= 1
    return count


def diversity(P, M):
    number = P ^ M
    return number_of_set_bits(number)


T = int(raw_input())
while T != 0:
    (P, M) = [int(a) for a in raw_input().split(" ")]
    print(diversity(P, M))
    T -= 1
