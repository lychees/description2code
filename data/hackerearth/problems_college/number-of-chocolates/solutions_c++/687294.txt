using namespace std;

int t,n,q,i,g1,g2,p1,p2;
long long a[1000005],ans;

void solve()
{
    scanf("%d %d",&n,&q);
    scanf("%d",&a[0]);
    for(i=1;i<n;++i)
    {
        scanf("%d",&a[i]);
        a[i]+=a[i-1];
    }
    while(q--)
    {
        ans=0;
        scanf("%d %d",&g1,&p1);
        scanf("%d %d",&g2,&p2);
        if(p2<g1 || g2<p1)
        {
            if(g1)
                ans+=(a[g2]-a[g1-1]);
            else
                ans+=a[g2];
            if(p1)
                ans+=(a[p2]-a[p1-1]);
            else
                ans+=a[p2];
        }
        else if(p1<=g1 && g2<=p2)
        {
            if(p1)
                ans+=(a[p2]-a[p1-1]);
            else
                ans+=a[p2];
        }
        else if(g1<=p1 && p2<=g2)
        {
            if(g1)
                ans+=(a[g2]-a[g1-1]);
            else
                ans+=a[g2];
        }
        else if(p1<=g1)
        {
            if(p1)
                ans+=(a[g2]-a[p1-1]);
            else
                ans+=a[g2];
        }
        else if(g1<=p1)
        {
            if(g1)
                ans+=(a[p2]-a[g1-1]);
            else
                ans+=a[p2];
        }
        printf("%lld\n",ans);
    }
}

int main(int argc,char *argv[])
{
	solve();
    return 0;
}
