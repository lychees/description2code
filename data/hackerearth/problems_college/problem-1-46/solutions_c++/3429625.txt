#include <iostream>

using namespace std;

int main()
{
	long size,query,i,l,r,count,a=-1,b=-1;
	
	char str[100001];
	cin>>size;
	cin>>str;
	cin>>query;
	long arr[26] = {0};
	while(query--)
	{
		cin>>l>>r;
		l--;
		r--;
		if(a!=b!=-1)
		{
			if(l<a)
			{
				for(i=l;i<a;i++)
				{
					arr[((int)str[i])-97]++;
				}
			}
			else if(l>a)
			{
				for(i=a;i<l;i++)
				{
					arr[((int)str[i])-97]--;
				}
			}
			if(r<b)
			{
				for(i=r+1;i<=b;i++)
				{
					arr[((int)str[i])-97]--;
				}
			}
			else if(r>b)
			{
				for(i=b+1;i<=r;i++)
				{
					arr[((int)str[i])-97]++;
				}
			}
		}
		if(l==r)
			cout<<"NO"<<endl;
		else if((r-l+1)%2==0)
		{
			count = 0;
			for(i=0;i<26;i++)
			{
				if(arr[i]%2!=0)
				{
					count++;
				}
			}
			if(count==2||count==0)
				cout<<"YES"<<endl;
			else
				cout<<"NO"<<endl;
		}
		else
		{
			count = 0;
			for(i=0;i<26;i++)
			{
				if(arr[i]%2!=0)
				{
					count++;
				}
			}
			if(count==1)
				cout<<"YES"<<endl;
			else
				cout<<"NO"<<endl;
		}
		a=l;
		b=r;
	}
    return 0;
}
