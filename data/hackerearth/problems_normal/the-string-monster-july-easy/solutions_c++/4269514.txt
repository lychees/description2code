#include <bits/stdc++.h>
using namespace std; 

string a[33]; 
string tar;
int cnt[26],cnt1[26], cnt2[33][26];
int n;
set<string> myset;
int main() {
    ios::sync_with_stdio(0);
    int t; 
    cin >> t; 
    while (t--) {
        cin >> n; 
        memset(cnt,0,sizeof cnt);
        memset(cnt2, 0, sizeof cnt2);
        for(int i=0; i<n; i++) { 
            cin >> a[i];
            for(int j=0; j<(int)a[i].size(); j++) {
                cnt2[i][a[i][j]-'a']++;
            }
        }
        cin >> tar;
		for(int i=0; i<(int)tar.size(); i++) cnt[tar[i]-'a']++;
        myset.clear();
        int to = n/2, stop = (1 << (to+1))-1; 
        for(int mask=0; mask<=stop;mask++) {
            memset(cnt1,0, sizeof cnt1);;
            bool ok = 1; 
            for(int i=0; i<=to; i++) {
                if (mask & (1 << i)) {
                    for(int j=0; j<26; j++) {
                        cnt1[j]+=cnt2[i][j];
                    }
                }
            }
            string x; 
            for(int i=0; i<26; i++) {
                while (cnt1[i]--)x+=char('a'+i);
            }
            myset.insert(x);
        }
        
        int from = to+1;
        to = n - 1;
        to = to - from + 1;stop = (1 << (to+1))-1;
        bool ansfound = 0;
        for(int mask=0; mask<=stop;mask++) {
            for(int i=0; i<26; i++) cnt1[i]=cnt[i];
            bool ok = 1; 
            for(int i=0; i<=to; i++) {
                if (mask & (1 << i)) {
                
                    for(int j=0; j<26; j++) {
                        cnt1[j]-=cnt2[i+from][j];
                        if (cnt1[j] < 0) {
                            ok = 0; break;
                        }
                    }
                    if (!ok) break;
                }
            }
            if (!ok) continue;
            string x; 
            for(int i=0; i<26; i++) {
                while (cnt1[i]--)x+=char('a'+i);
            }
            if (myset.count(x)) {
                ansfound = 1;  break;
            }
            
        }
        
        if (ansfound) cout << "YES\n";
        else cout << "NO\n";
        
    }
    
    return 0; 
}
