#include <bits/stdc++.h>

using namespace std;

const int N = 1e6 + 10;

int pr[N], ht[N], req[N], level[N];

int main() {
	int t;
	scanf("%d", &t);
	for (int tt = 1; tt <= t; tt++) {
		int n, k;
		scanf("%d %d", &n, &k);
		for (int i = 0; i < n; i++) {
			scanf("%d", &pr[i]);
		}
		int maxh = -1, sz = 0;
		for (int i = 0; i < k; i++) {
			scanf("%d", &ht[i]);
			if (ht[i] >= maxh) {
				maxh = ht[i];
				req[sz] = ht[i];
				level[sz++] = i + 1;
			}
		}
		int maxl = -1, res;
		for (int i = 0; i < n; i++) {
			int ub = upper_bound(req, req + sz, pr[i]) - (req);
			int l = level[ub - 1];
			if (l > maxl) {
				maxl = l;
				res = i;
			} 
		}
		printf("%d\n", res);
	}
	return 0;
}
