#include <iostream>
#define ll long long int
using namespace std;

int main()
{
    ll t;
    scanf("%lld",&t);
    while(t--)
    {
    	ll n;
    	scanf("%lld",&n);
    	if(n==0)
    	{
    		cout<<"0"<<endl;
    		continue;
    	}
    	if(n<=3)
    	{
    		cout<<"1"<<endl;
    		continue;
    	}
    	if(n%3==0)
    	cout<<n/3<<endl;
    	
    	else
    		cout<<n<<endl;
    }
    return 0;
}
