#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	long int a[1000];
    	int n,k;
    	scanf("%d%d",&n,&k);
    	int i,j;
    	for(i=0;i<n;i++)
    	{
    		scanf("%ld",&a[i]);
    	}
    	int pos;
    	for(i=0;i<n;i++)
    	{
    		pos=i;
    		for(j=i+1;j<n;j++)
    		{
    			if(j-i>k)
    			break;
    			if(a[j]<=a[pos])
    				pos=j;
    		}
    		for(j=pos;j>i;j--)
    		swap(a[j],a[j-1]);
    		k-=(pos-i);
    	}
		for(i=0;i<n;i++)
		cout<<a[i]<<" ";
		cout<<endl;
	}
}
