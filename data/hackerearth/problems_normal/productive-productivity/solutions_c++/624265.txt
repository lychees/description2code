#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <vector>
 
using namespace std;
 
#define INF 1000000000
 
int main ()
{
		int t;
		cin>>t;
		while(t--){
		
        int n, m, i, j;
 
        
        scanf("%d", &n);
 		m=n;
        vector < vector <int> > a(n + 1, vector<int>(m + 1));
 
        for (i = 0; i < n; i++)
                for (j = 0; j < m; j++)
                        scanf("%d", &a[i+1][j+1]);
 
 		for (i = 0; i < n; i++)
                for (j = 0; j < m; j++)
                       a[i+1][j+1]=(-1)*a[i+1][j+1];
        vector<int> u (n+1), v (m+1), p (m+1), way (m+1);
        for (int i=1; i<=n; ++i) {
                p[0] = i;
                int j0 = 0;
                vector<int> minv (m+1, INF);
                vector<char> used (m+1, false);
                do {
                        used[j0] = true;
                        int i0 = p[j0],  delta = INF,  j1;
                        for (int j=1; j<=m; ++j)
                                if (!used[j]) {
                                        int cur = a[i0][j]-u[i0]-v[j];
                                        if (cur < minv[j])
                                                minv[j] = cur,  way[j] = j0;
                                        if (minv[j] < delta)
                                                delta = minv[j],  j1 = j;
                                }
                        for (int j=0; j<=m; ++j)
                                if (used[j])
                                        u[p[j]] += delta,  v[j] -= delta;
                                else
                                        minv[j] -= delta;
                        j0 = j1;
                } while (p[j0] != 0);
                do {
                        int j1 = way[j0];
                        p[j0] = p[j1];
                        j0 = j1;
                } while (j0);
        }
 
        vector<int> ans (n+1);
        for (j=1; j<=m; ++j)
                ans[p[j]] = j;
 
 		cout<<(-1)*int(-v[0])<<"\n";
        //printf("Result:\n");
        //for (i = 1; i <= n; i++)
          //      printf("%d ", ans[i]);
        //printf("\n");
 
        //printf("Sum: %d\n", int(-v[0]));
 
        //system("pause");
}
        //return 0;
}
