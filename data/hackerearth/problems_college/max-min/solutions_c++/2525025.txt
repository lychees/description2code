#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;


int main()
{
	int n,k,num,min=0;
		vector<int> v1;
		cin>>n>>k;
		while(n--){
				cin>>num;
					v1.push_back(num);
		}
		sort(v1.begin(),v1.end());
		min=v1[k-1];
		for(int i=1;i<v1.size()-k+1;i++){
					if(v1[i+k-1]-v1[i]<min)
						min=v1[i+k-1]-v1[i];
			
		}
		
    cout <<min<< endl;
    return 0;
}
