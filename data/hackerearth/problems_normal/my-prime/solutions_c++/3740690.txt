#include <iostream>
using namespace std;

int main()  
{  
    int a[1000000], n, i, j, flag;  
    cin >> n;
    for(i=0;i<n;i++)  
    	cin >> a[i];
    for(i=0;i<n;i++)  
    {  
     	flag = 1;  
     	for(j=0;j<n;j++)  
     	{  
      		if(i!=j && a[i]%a[j]==0)  
      		{  
       			flag = 0;  
       			break;  
      		}  
     }  
     if(flag)  
      printf("%d ",a[i]);  
    }  
    return 0;  
}
