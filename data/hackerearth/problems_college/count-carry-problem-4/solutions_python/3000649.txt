n = input()
for input in xrange(n):
    carry_count = 0
    nums = map(int, raw_input().split())
    num0 = [int(x) for x in str(nums[0])]
    num1 = [int(x) for x in str(nums[1])]

    for (x,y) in zip(num0, num1):
        if x+y > 9:
            carry_count += 1
    if carry_count == 0:
        print "No carry operation"
    elif carry_count == 1:
        print "1 carry operation"
    else:
        print carry_count, "carry operations"
