#include<bits/stdc++.h>
using namespace std;

const int N=1e6+10;
int a[N];

int main() {
    int n,k;
    scanf("%d%d",&n,&k);
    for(int i=0;i<n;i++) {
        scanf("%d",&a[i]);
    }
    sort(a,a+n);
    int i;
    for(i=0;i<n-1;i++) {
        if(a[i+1]-a[i]>k) {
            break;
        }
    }
    printf("%d",a[i]);
    return 0;
}
