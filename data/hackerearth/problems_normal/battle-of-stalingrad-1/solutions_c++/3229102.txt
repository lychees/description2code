#include<cmath>
#include <string.h>
#include <stdio.h>
#include <iostream>
#include <stdlib.h>
#include <math.h>
using namespace std; 
float area (long long int x1, long long int y1, long long int x2, long long int y2, long long int x3, long long int y3)
{
    return abs((x1*(y2-y3) + x2*(y3-y1)+ x3*(y1-y2))/2.0);
}
bool isInside(long long int x1, long long int y1, long long int x2, long long int y2, long long int x3, long long int y3, long long int x, long long int y)
{  
	float A = area (x1, y1, x2, y2, x3, y3);
	float A1 = area (x, y, x2, y2, x3, y3);
	float A2 = area (x1, y1, x, y, x3, y3);
	float A3 = area (x1, y1, x2, y2, x, y);
	return (A == A1 + A2 + A3);
}
int main()
{
	long long int tc;
	cin >> tc;
	while (tc--)
	{
		long long int a, b, c, d, e, f, g, h;
		cin >> a >> b >> c >> d >> e >> f >> g >> h;
		if (isInside(a, b, c, d, e, f, g, h))
		{ 
			printf ("INSIDE\n");
		}
		else
		{
			printf ("OUTSIDE\n");
		}
	} 
	return 0;
}
