n_tests = input()
for tc in xrange(n_tests):
    n_books = input()
    books = map(long, raw_input().strip().split(' '))
    minval = 0
    maxval = 0
    minbooks = min(books)
    if minbooks >= 2:
        minval = minbooks - 1
        maxval = sum(books) - n_books
    print minval, maxval
