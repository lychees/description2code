#include <iostream>
using namespace std;



long long int comb(long long int m,long long int n)
{
if(n==0||m==n)
return 1;
else
return comb(m-1,n-1)+comb(m-1,n);
}


long long int fact(long long int x)
{
if(x==0)
return 1;
if(x==1)
return 1;
else return x*fact(x-1);
}




int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int c,v,i,total,ans;
    	cin>>c>>v;
    	total=c+v;
    	ans=fact(total)*comb(7,c)*comb(5,v);
    	cout<<ans<<endl;
    }
    return 0;
}
