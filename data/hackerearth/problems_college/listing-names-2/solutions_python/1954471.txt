t = input();
x=list()
for i in range(0,t):
    s=raw_input()
    x.append(s)
for i in range(0,t):
    z=1
    if x[i]=='$':
        continue
    elif x[i]!='$':
        for j in range(i+1,t):
            if x[j]==x[i] :
                z=z+1
                x[j]='$'
        print x[i],""+ str(z)
