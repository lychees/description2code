def chain_length(n):
    m=0
    even=0
    odd=0
    while(n!=1):
        if n%2==0:
            even=even+n
            n=n/2
        else:
            odd=odd+n
            n=3*n+1
        m=m+1
    if even%m>(odd+1)%m:
        print "Even Rules"
    
        
    if even%m<(odd+1)%m:
        print "Odd Rules"
    if even%m==(odd+1)%m:
        print "Tie"
t=int(input())
for i in xrange(t):
	p=int(input())
	chain_length(p)
