#include <bits/stdc++.h>
using namespace std;

#define ll long long



int main()
{
    ll n,i,j;
   	scanf("%lld",&n);
   	ll arr[n+1],m[n+1];
   	for(i=0;i<n;i++) scanf("%lld",&arr[i]);
   	sort(arr,arr+n);
   	for(i=0,j=0;j<n;i++,j+=2)
   	{
   		m[j]=arr[i];
   	}
   	for(i=n-1,j=1;j<n;j+=2,i--)
   	{
   		m[j]=arr[i];
   	}
   	ll sum=0;
   	for(i=0;i<n-1;i++)
   	{
   		sum+=abs(m[i]-m[i+1]);
   	}
   	sum+=abs(m[n-1]-m[0]);
   	cout<<sum<<endl;
    return 0;
}
