#include <iostream>
#include<math.h>
using namespace std;

int main()
{
   int T;
   cin>>T;
   while(T--){
   	int d;cin>>d;
   	int n=floor((1+sqrt(1+4*d))/2);
    d-=(n*(n-1))/2;	
   	for(int i=n;;i++){
   	    d-=n;
   		if(d<=0)
   		break;
   		else n++;
   	}
   	
   	if(n%3==0)cout<<"SL"<<endl;
   	else if(n%3==1)cout<<"LB"<<endl;
   	else if(n%3==2)cout<<"BS"<<endl;
   }
}
