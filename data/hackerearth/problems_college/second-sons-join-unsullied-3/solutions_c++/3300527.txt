#include <bits/stdc++.h>
#define ll long long int
#define gt(i) scanf("%lld", &i)

using namespace std;

int main()
{

    ll test;
    gt(test);
    while(test--) {
        ll n, m, i, j, k;
        string s;
        cin >> s;
        n = s.size();
        ll u = 0, ss = 0;
        for(i = 0; i < n; i++) {
            if(s[i] == 'U')
                u++;
            else
                ss++;
        }
        if(u == ss || ss + 1 == u || u + 1 == ss)  {
            ll ct1 = 0, ct2 = 0;
            for(i = 0; i < n; i++) {
                if(i % 2 == 0 && s[i] != 'U' || i % 2 != 0 && s[i] != 'S')
                    ct1++;
                if(i % 2 == 0 && s[i] != 'S' || i % 2 != 0 && s[i] != 'U')
                    ct2++;
            }
            if(ct1 % 2 == 0)
                ct1 = ct1 / 2;
            if(ct2 % 2 == 0)
                ct2 = ct2 / 2;
            if(u == ss)
                cout << min(ct1, ct2) << endl;
            else if(u < ss) 
                cout << ct2 << endl;
            else 
                cout << ct1 << endl;
        }
        else
            cout << -1 << endl;
    }
    return 0;
}
