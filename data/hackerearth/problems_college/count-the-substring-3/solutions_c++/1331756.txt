void fun(string s)
{
	int64_t len = s.size();
	int64_t a=0,b=0,c=0;
	int64_t start = 0;
	int64_t total = 0;
	int64_t tmv = 0;
	int64_t k = 0;
	for(int i=0;i<len;i++)
	{
		if(s[i] == 'x')
			a++;
		if(s[i] == 'y')
			b++;
		if(s[i] == 'z')
			c++;

		if(a>0 && b>0 && c>0)
		{
			k=start;
			total += (i-start)*(i-start+1)/2;

			while(a>0 && b>0 && c>0)
			{
				if(s[k] == 'x')
					--a;
				if(s[k] == 'y')
					--b;
				if(s[k] == 'z')
					--c;

				k++;
			}

			i = k;
			start = k;
			int64_t cn=0;
			while(s[start]==s[k])
				k++;
			cn=k-start;
	        tmv+=cn*(cn+1)/2;


			a=b=c=0;
			i=start -1;
		}
		else if(i == len-1)
		{
			total += (i-start+1)*(i-start+1+1)/2;
		}
	}
	cout<<total-tmv<<"\n";
}

int main()
{
	string s;
	int t;
	cin>>t;
	int64_t total = 0;
	int64_t a=0,b=0,c=0;
	while(t--)
	{
		total = 0;
		cin>>s;
		for(int i=0;i<s.size();i++)
		{
				if(s[i] == 'x')
					a++;
				if(s[i] == 'y')
					b++;
				if(s[i] == 'z')
					c++;
		}
		if(a>0 && b>0 && c>0)
		{
			fun(s);
		}
		else
		{
			total = total + (s.size()*(s.size()+1)/2);
			cout<<total<<"\n";
		}
	}

    return 0;
}
