#include <bits/stdc++.h>
using namespace std;

int main()
{
  int t;
  long long n;
 long double s;
  long p,q;
  cin>>t;
  while(t--)
  {
  	cin>>n;
  	s=sqrt(n);
  	p=(int)(s);
  	q=p+1;
  	if(s==p)
  	cout<<"YES"<<endl;
  	else
  	
  	
  		(abs(p*p-n))>(abs(q*q-n))?cout<<"+"<<q*q-n<<endl:cout<<p*p-n<<endl;
  	
  	
  		
  	   
  }
  	
  	
  	return 0;
}
