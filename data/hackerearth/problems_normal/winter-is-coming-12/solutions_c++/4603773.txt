#include<iostream>
#include<vector>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        int a[n+1];
        for(int i=1;i<=n;i++)
        {
            cin>>a[i];
        }
        int ans=100010;
        int l=0,r=0,temp,sum=0;
        int left[n];
        for(int i=0;i<=n;i++)
        {
            left[i]=0;
        }
        for(int i=1;i<=n;i++)
        {
            sum=sum+a[i];
            sum%=n;
            if(left[sum]==0)
            {
                if(sum==0)
                {
                    temp=i;
                }
                else{
                    temp=100010;
                }
            }
            else{
                temp=i-left[sum];
            }
            if(temp<ans)
            {
                ans=temp;
                l=left[sum]+1;
                r=i;
            }
            left[sum]=i;
        }
        cout<<l<<" "<<r<<endl;
    }
}
