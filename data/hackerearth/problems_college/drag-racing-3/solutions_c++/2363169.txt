#include <iostream>
#include<map>
#include<string>
using namespace std;

#define ll long long
int main(){
    ll n;
    cin>>n;
    ll x=((1<<n)-1);
    string losers[x+10];
    map<string,int> mp;
    for(ll i=0;i<x;i++){
        string w,l;
        cin>>w>>l;
        mp[w]++,mp[l]++;
        losers[i]=l;
    }
    for(ll i=0;i<x;i++)
        mp[losers[i]]=-1;
    map<string,int> ::iterator it=mp.begin();
    while(it!=mp.end()){
        if((it->second)>0){
            cout<<(it->first)<<"\n";
            break;
        }
        it++;
    }
    return 0;
}
