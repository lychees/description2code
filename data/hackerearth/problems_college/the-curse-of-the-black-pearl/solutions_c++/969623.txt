#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
int main()
{
    char str[100000];
    int t,i,j,c=0,n;
    t=882;
    while(t--)
    {
        scanf("%s",str);
       
        n=strlen(str);
        for(i=0;i<(n/2);i++)
        {
        	char ch=str[i];
        	str[i]=str[n-i-1];
        	str[n-i-1]=ch;
        
        }
        i=0;
        c=0;
        while(i!=n-1)
        {
            if(str[i]=='0')
                c++,i++;
            else
            {
                j=i;
                while(str[j]!='0' && j!=n)
                {
                    str[j]='0';
                    j++;
                }
                str[j]='1';
                if(j==n)
                    n++;
                c++;
            }
        }
        printf("%d\n",c);
    }
}
