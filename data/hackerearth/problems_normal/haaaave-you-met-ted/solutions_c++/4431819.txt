#include <bits/stdc++.h>
using namespace std;

int count(int A)
{
	int ans = 0;
	while(A) {
		if (A & 1)
			ans++;
		A >>= 1;
	}
	return ans;
}

int main()
{
    int T;
    cin >> T;
    while(T--)
    {
    	int N, ans = INT_MAX, A;
    	cin >> N;
    	while(N--) {
    		cin >> A;
    		ans = min(ans, count(A));
    	}
    	cout << ans << endl;
    }
    return 0;
}
