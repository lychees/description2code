#include <iostream>
using namespace std;
int min(int a,int b)
{
	return a>b?b:a;
}	

int	minegg(int n)
{
	if(n<0)
		return 999;
	else if(n==0)
		return 0;
	else
	{
		return min(1+minegg(n-6),1+minegg(n-8));
	}	
		
}

int main()
{
//  cout << "Hello World!" << endl;
	int n,t=0,test,x=0;
	cin>>test;
	while(t++<test)
	{
		cin>>n;
		x=minegg(n);
		if(x>900)
			x=-1;
		cout<<x<<endl;
	}
	return 0;

}
