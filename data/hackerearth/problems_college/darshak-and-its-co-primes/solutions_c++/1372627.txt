#include <iostream>
using namespace std;
int gcd(int a,int b)
{
	if(a==0)
		return b;
	if(b==0)
		return a;
	return (gcd(b,a%b));
}
int main()
{
    long long int p,q;
    int t;
    cin >> t;
    while(t--)
    {
    	cin >> p >> q;
    	int k=gcd(p,q);
    	if(k==1)
    		cout << "Is a Co-Prime" << endl;
    	else
    		cout << "Not a Co-Prime" << endl;
    }
    return 0;
}
