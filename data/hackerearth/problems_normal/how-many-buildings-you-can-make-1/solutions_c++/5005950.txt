#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t , n;

    cin >> t;

    while(t--){

        cin >> n;

        int i = 1, ans = 0;

        while(n / i >= i){              // loop until root of n

            ans += n / i - (i - 1);     // subtract the repeated traingles
            i++;
        }

        cout << ans << "\n";
    }


    return 0;
}
