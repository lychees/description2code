#include<cstdio>
#include<iostream>
using namespace std;
int count(long long  int a[],long long int m,long long int n )
{
    long long int i, j, x, y;
    long long int table[n+1][m];
    for (i=0; i<m; i++)
        table[0][i] = 1;
    for (i = 1; i < n+1; i++)
    {
        for (j = 0; j < m; j++)
        {
            x = (i-a[j] >= 0)? table[i - a[j]][j]: 0;
            y = (j >= 1)? table[i][j-1]: 0;
            table[i][j] = x + y;
        }
    }
    return table[n][m-1];
}
int main()
{
          long long int n,m,i;
          scanf("%lld %lld",&n,&m);
          long long int a[m];
          for(i=0;i<m;i++)
                    scanf("%lld",&a[i]);
    long long int k = sizeof(a)/sizeof(a[0]);
    printf("%lld ", count(a, k, n));
    return 0;
}
