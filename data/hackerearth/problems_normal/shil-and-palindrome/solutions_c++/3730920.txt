#include <iostream>
#include <string>
using namespace std;

int main()
{
    string a;
    cin>>a;
    int val[26]={0};
    for(int i=0;i<(a.length());i++)
    	val[((int)a.at(i))-97]++;
    int oddat=-1;
    bool fail=false;
    for(int i=0;i<26;i++)
    {
    	if(val[i]%2==1)
    	{
    		if(oddat==-1)
    			oddat=i;
    		else
    			fail=true;
    	}
    }
    if(fail)
    	cout<<-1;
    else
    {
    	for(int i=0;i<26;i++)
    	{
    		int k=val[i]/2;
    		while(k--)
    			cout<<((char)(i+97));
    	}
    	if(oddat!=-1)
    		cout<<((char)(oddat+97));
    	for(int i=25;i>=0;i--)
    	{
    		int k=val[i]/2;
    		while(k--)
    			cout<<((char)(i+97));
    	}
    }
    return 0;
}
