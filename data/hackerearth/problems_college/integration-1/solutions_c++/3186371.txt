#include <iostream>

using namespace std;

int main()
{
	long long t,i,j;
	cin >> t;
	for(i=1;i<=t;i++)
	{
		long long n,p,u,r,s;
		cin >> n >> p >> u >> r >> s;
		for(j=0;j<n;j++)
		{
			if(p+r<=u)
			{
				p=p+r;
			}
			else if(p-s>=0)
			{
				p=p-s;
			}
		}
		cout << "Case #" << i << ": " << p << endl;
	}
}
