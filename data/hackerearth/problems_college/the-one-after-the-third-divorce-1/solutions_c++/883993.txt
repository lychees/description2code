#include <iostream>
using namespace std;
long lcm(long x,long y)
{
	long m;
    if(x>y)
         m=y;
    else
         m=x;

    while(1)
    {
         if(m%x==0&&m%y==0)
         {
             return m;
   		 }
   		 m++;
    }
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	long x,y,a,b,count=0,l;
    	scanf("%ld%ld%ld%ld",&x,&y,&a,&b);
    	l=lcm(x,y);
    	count=b/l-(a-1)/l;
    	printf("%ld\n",count);
    }
    return 0;
}
