#include <iostream>
using namespace std;

typedef long long int ll;
ll sieve[1000000];

void sieveof()
{
	ll n = 1000000;
	sieve[0] = -1;
	sieve[1] = -1;
	ll p, c, i;
	for(p = 2; p <= n; p++){
			c = 1;
			if(sieve[p] == 0){
				for(i = p*2; i <= n; i += p){
					if(sieve[i] != -1){
						sieve[i] = -1;
						c++;
					}
				}
				sieve[p] = c;
			}
	}
	
}


int main() {
	ll t, i;
	cin >> t;
	sieveof();
	while(t--){
		cin >> i;
		cout << sieve[i] << endl;
	}
	return 0;
}
