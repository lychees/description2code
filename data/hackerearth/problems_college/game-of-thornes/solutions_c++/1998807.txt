    #include <iostream>
    #include <string>
    #include <algorithm>
    using namespace std;
    int main()
    {
        string str;
        cin>>str;
        int len = str.length();
        int a[len];
        for(int i=0;i<len;i++){
        	a[i] = int(str[i]);
        }
        sort(a,a+len);
        int u_count = 0;
        for(int i=0;i<len-1;i++){
        	if(u_count > 1)break;
        	if(a[i] == a[i+1]){ i++; continue;}
        	if(a[i] != a[i+1]) u_count++;    	
        }
        if(u_count == 1 && len%2 == 1) cout<<"YES";
        else if(u_count == 0) cout<<"YES";
        else cout<<"NO";
        return 0;
    }
