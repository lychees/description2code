#include <bits/stdc++.h>
using namespace std;

int main() {
	int t,mx=0,i;
	string h;
	map <string , int> m;
	cin >> t;
	for (i = 0; i < t; i++) {
		string s;
		int a;
		cin >> s;
		cin >> a;
		m[s] = m[s] + a;
		if (m[s] > mx) {
			mx = max(m[s],mx);
			h = s;
		}
	}
	cout << h << endl; 
	return 0;
}
