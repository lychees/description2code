t = int(input())


def fac(n):
    pro = 1
    for i in range(1, n+1):
        pro = pro*i
        pro = pro % 1000000007
    return pro


def per(n, r):
    pro = 1
    for i in range(0, r):
        pro = pro*(n-i)
        pro = pro % 1000000007
    return pro


while t > 0:
    r = raw_input().split()
    g = int(r[0])
    b = int(r[1])
    if (g-b < -1):
        print("-1")
    else:
        pro = per(g+1, b)*fac(g)
        pro = pro % 1000000007
        print(pro)
    t -= 1
