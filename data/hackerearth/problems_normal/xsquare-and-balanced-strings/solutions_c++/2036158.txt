#include <iostream>
#include<string.h>
using namespace std;

int checkBalance(std::string arr){
int l = arr.length();
int lookup[26]={0};
int small=0,mismatch=0,pos=0;

for(int i=0;i<l;i++){
	++lookup[int(arr[i])-97];
}

for(int i=0;i<26;i++){
	if(lookup[i]!=0){
		pos = i;
		small = lookup[i];
		break;
	}	
}

if(small%2 !=0) return -1;

for(int i=pos+1;i<26;i++){
	    if(lookup[i]!=0){
	    if(lookup[i]%2 == 0)	
	    small = (small>lookup[i])?lookup[i]:small;	    
	    else 
	    {mismatch=1; break;}
	    }		
}

if(mismatch==1) return -1;
return 1;
}

int main()
{
    std::string arr;
    int t;    
    cin>>t;
    
    for(int i=0;i<t;i++){
    	cin>>arr;
    	cout<<checkBalance(arr)<<"\n";    	
    }      
    return 0;
}
