void PrintArray(char *a, int i, int n)
{
   int j;
   if (i == n)
     cout<<a<<endl;
   else
   {
        for (j = i; j <= n; j++)
       {
          Swap((a+i), (a+j));
          PrintArray(a, i+1, n);
          Swap((a+i), (a+j));
       }
   }
}
void Swap (char *x, char *y)
{
    char temp;
    temp = *x;
    *x = *y;
    *y = temp;
}
