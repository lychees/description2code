t = int(input())

while t:
    n1, n2 = map(int, raw_input().split())
    a = bin(n1)[2:]
    b = bin(n2)[2:]
    la = len(a)
    lb = len(b)
    if la > lb:
        b = '0'*(la-lb)+b
    else:
        a = '0'*(lb-la)+a
    # print "a",a,"b",b
    cnt = 0
    for x, y in zip(a, b):
        if x != y:
            cnt += 1
    print cnt
    t -= 1
