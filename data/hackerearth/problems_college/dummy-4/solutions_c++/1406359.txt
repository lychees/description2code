#include <bits/stdc++.h>
using namespace std;
int main()
{
    long t=0;
    long sum=0,n=0;
    scanf("%ld",&t);
    while(t--)
    {
    	char ch;
    	while(cin>>ch)
    	n+=(ch-'0');
    }
    sum = n % 9;
  while (sum == 0) 
  {
    sum += 9;
  }
    printf("%ld\n",sum);
    return 0;
}
