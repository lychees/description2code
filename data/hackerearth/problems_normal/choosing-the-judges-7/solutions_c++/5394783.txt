#include<iostream>
#include<cstdio>
using namespace std;
int main() {
	// freopen("in.txt", "r", stdin);
	long long t, n, a[10002], ans, i, c=1;
	scanf("%lld", &t);
	while(t--) {
		scanf("%lld", &n);
		ans = 0;
		for(i=0;i<n;i++) {
			scanf("%lld", &a[i]);
		}

		for(i=n-1;i>=n-2 && i>=0;i--) {
			ans = max(ans, a[i]);
		}
		if(i>-1) {
			a[i+1]=ans;
		}

		for(i=n-3;i>=0;i--) {
			a[i] = max(ans, a[i] + a[i+2]);
			ans = max(ans, a[i]);
		}

		printf("Case %lld: %lld\n", c, ans);
		c++;
	}
	return 0;
}
