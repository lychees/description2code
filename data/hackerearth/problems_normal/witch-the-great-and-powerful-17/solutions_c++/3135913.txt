#include <iostream>
//#define ll long long int;
using namespace std;

int main()
{
    
    int t;
    long long int  P,Q,R;
    cin>>t;
    while(t){
    	cin>>P;
    	cin>>Q;
    	cin>>R;
    if(Q-P >= R-Q)
    {
    	cout<<Q-P-1<<"\n";
    }
    else
    cout<<R-Q-1<<"\n";
    	
    	t--;
    	
    }
    //cout << "Hello World!" << endl;
    return 0;
}
