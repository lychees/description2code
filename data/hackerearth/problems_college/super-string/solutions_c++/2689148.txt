#include <bits/stdc++.h>
using namespace std;
int arr[256];

int main()
{
	map<char,int> m;
	int b =1;
      for (char c ='Z';c>='A';c--)
      {
      	m[c] = b++;
      }
	int t;
	scanf("%d",&t);
	
	while(t--)
	{
		string s;
		cin>>s;
		
		memset(arr,0,sizeof(arr));
		for (int i =0;i<s.length();i++)
		{
			arr[s[i]]++;
		//	cout<<s[i]<<endl;
		}
		int f=0;
		for (int i = 0;i<s.length();i++)
		{
			if(m[s[i]]!=arr[s[i]])
			{
		//	cout<<s[i]<<endl;
				f=1;
				//printf("No\n");
				break;
			}
			
		}
		if(f==1)
		cout<<"No"<<endl;
		else
		cout<<"Yes"<<endl;
		
	}
    //cout << "Hello World!" << endl;
    return 0;
}
