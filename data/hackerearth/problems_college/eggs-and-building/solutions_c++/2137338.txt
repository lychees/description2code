#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<cstring>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"
int cs,n,k;
int M[1001][11];
int solve(int i,int j)
{
  int &res=M[i][j],k;
  if(res==-1)
    if(j==1)
      res=i;
    else if(i==0)
      res=0;
    else
    {
      res=i;
      for(k=1;k<=i;k++)
        res=min(res,max(solve(k-1,j),solve(i-k,j-1))+1);
    }
  return res;
}
int main()
{
  memset(M,-1,sizeof M);
  scanf("%d",&cs);
  while(cs--)
  {
    scanf("%d%d",&n,&k);
    printf("%d\n",solve(n,k));
  }
	return 0;
}
