#include <bits/stdc++.h>

using namespace std;

int repeat [26 + 1];

int main()
{

    int t;
    string pos, input;

    cin >> t;

    while(t--){

        cin >> pos >> input;

        memset(repeat, 0, sizeof repeat);                                   // reset array in each test case

        for(int i = 0; i < input.size(); i++) repeat[ input[i] - 'a' ]++;   // number of repetition for each character

        for(int i = 0; i < 26; i++){                                        // loop over characters in the given order

            int num = repeat[ pos[i] - 'a' ];                               // print the (ith) character the number of times it is repeated

            for(int j = 0; j < num; j++) cout << (char) (pos[i]);
        }

        cout << "\n";
    }

    return 0;
}
