#include <iostream>
using namespace std;

int main()
{
    int t,N,C,M;
    cin>>t;
    while(t--)
    {
    	cin>>N>>C>>M;
    	int num = N/C;
    	int w = num;
    	if(w<M)
    	cout<<num<<"\n";
    	else
    	{
    		while(w>=M)
    		{
    			num += w/M;
    			w = w/M + w%M;
    		}
    		cout<<num<<"\n";
    	}
    }
}
