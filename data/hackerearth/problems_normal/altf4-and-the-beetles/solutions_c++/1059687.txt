#include <iostream>
#include <vector>
#include <algorithm>
#include <cstring>
using namespace std;
int n;

int main()
{
    int t;
    
    cin>>t;
    long long a,b,c;
    long long p,q;
    while(t--){
    	scanf("%d%lld%lld%lld",&n,&a,&b,&c);
    	vector<pair<long long,long long> > pq;
    	
    	for(int i=0;i<n;i++){
    		scanf("%lld%lld",&p,&q);
    		pq.push_back(make_pair(p,1));    
    		pq.push_back(make_pair(q,2));   
    	}
    	sort(pq.begin(),pq.end());
    	long long ans = n*a;
    	long long cur = ans;
    	for(int i=0;i<2*n;i++){
    		if (pq[i].second==1) cur = cur-a+b;
    		else cur = cur-b+c;
    		ans = max(ans,cur);
    	}
    	printf("%lld\n",ans);
    }
    return 0;
}
