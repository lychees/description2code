#include <iostream>
using namespace std;

int check(long long int a,long long int b);

int main()
{
    long long int n;
    cin>>n;
    
    while(n--)
    {
    	long long int p,q,t;
    	cin>>p>>q;
    	t=check(p,q);
    	if(t==1)
    	{
    		cout<<"Is a Co-Prime"<<endl;
    	}
    	else
    	{
    		cout<<"Not a Co-Prime"<<endl;
    	}
    }
    return 0;
}

int check(long long int a,long long int b)
{
	long long int temp;
	while(b!=0)
	{
		temp=b;
		b=a%b;
		a=temp;
	}
	
	return a;
	
}
