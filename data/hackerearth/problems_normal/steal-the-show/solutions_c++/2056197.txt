#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <cmath>
using namespace std;

int main()
{
    int a1,b1,c,n,i,j,dif=0;
    scanf("%d%d%d\n",&a1,&b1,&c);
    scanf("%d\n",&n);
    int a[n];
    for(i=0;i<n;i++)
    	scanf("%d",&a[i]);
    
  sort(a,a+n);
   if(b1>c)
  {
  	j=b1;
  	b1=c;
  	c=j;
  } 
     dif=abs(a[0]-b1)+abs(c-a[n-1]);
     int m=(n%2==0)?(n/2-1):(n/2);
    for(i=0;i<n-1;i++){
    	if(i==m){
    		dif+=abs(a[i]-a1)+abs(a1-a[i+1]);
    		continue;
    	}
    	dif+=abs(a[i]-a[i+1]);}
    printf("%d\n",dif);
    return 0;
}
