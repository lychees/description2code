#include <bits/stdc++.h>
using namespace std;
int arr [9];
int t(int i,int sumR,int sumL,int n,int tt[]){
	if(i==n){
		if(sumR>=sumL)return 1;
		else return 0;
	}
	if(sumR<sumL)return 0;
	int suma = 0;
	for(int j=0;j<n;++j){
		if(tt[j])continue;
		sumR+=arr[j];
		tt[j]=1;
		suma += t(i+1,sumR,sumL,n,tt);
		sumR-=arr[j];
		sumL+=arr[j];
		suma+= t(i+1,sumR,sumL,n,tt);
		sumL-=arr[j];
		tt[j]=0;
	}
	return suma;
}
int main(void)
{
    int n;
    int sum = 0;
    cin>>n;
    for(int i=0;i<n;++i)cin>>arr[i],sum+=arr[i];
    int tt[n];
    memset(tt,0,sizeof tt);
    int xx = t(0,0,0,n,tt);
    cout<<xx<<" "<<sum<<endl;
    if(xx<=sum)puts("Got no way out!");
    else puts("We will win!");
}
