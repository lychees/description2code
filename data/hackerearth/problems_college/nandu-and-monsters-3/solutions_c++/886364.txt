#include <iostream>
using namespace std;
long long int n;
int path(long int x,long int y ,char a[])
{
		if(a[x]=='1' || a[y]=='1')
		return 0;
	while(x!=y)
	{
		if(x>y)
		x=x/2;
		else if(y>x)
		y=y/2;
		if(a[x]=='1' || a[y]=='1')
		return 0;
	}
	return 1;
}
int main()
{
	char a[100003];
	char b;
	long long int q,x,y;
	cin>>n>>q;
	for(long int i=1;i<=n;i++)
	{
		cin>>a[i];
	}
	while(q--)
	{
		cin>>x>>y;
		int k=path(x,y,a);
		//int k=0;

		cout<<"Yes"<<endl;
		else
		cout<<"No"<<endl;
	}
    //cout << "Hello World!" << endl;

}
