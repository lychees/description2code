#include <iostream>
using namespace std;

int calculateScore(int R,int C,int x)
{
	if(x<= (C/2))
	{
		return R*(C-2*(x-1))+x-1;
	}
	else if(x>(C/2)&&x<C)
	{
		return R+x;
	}
	else if(x==C)
	{
		return R+x-1;
	}
	
}

int main()
{
    //cout << "Hello World!" << endl;
    int t,r,c,x;
    cin>>t;
    int count=1;
    while(t--)
    {
    	cin>>r>>c>>x;
    	cout<<"Case #"<<count++<<": "<<calculateScore(r,c,x)<<"\n";
    }
    return 0;
}
