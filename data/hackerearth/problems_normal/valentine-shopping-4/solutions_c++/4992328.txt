#include <bits/stdc++.h>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
    cin.tie(NULL);  
    int t,n,m,i,a,b,c,val,min,idx,j;
    cin>>t;
    while(t--)
    {
    	cin>>n>>m;
    	for(i=0;i<n;i++)
    	{
    		min=1000007; idx=-1;
    		for(j=0;j<m;j++)
    		{
    			cin>>a>>b>>c;
    			val=(100-a)*(100-b)*(100-c);
    			if(val<min)
    			{
    				min=val; idx=j;
    			}
    		}
    		cout<<idx+1<<" ";
    	}
    	cout<<endl;
    }
    return 0;
}
