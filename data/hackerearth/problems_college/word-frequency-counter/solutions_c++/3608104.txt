#include<stdio.h>
#include<string.h>
#include <iostream>
#include<unordered_map>
using namespace std;
unordered_map<string , int>a;
unordered_map<string , int>b;
unordered_map<string , int>::iterator it,ft;
int main()
{
    char s[1000000];
    int i,st,en,count=1;
    while(scanf("%s",&s)==1)
    {
     for(i=0;s[i]!='\0';i++)
     {
       if(s[i]>=97 && s[i]<=122)
        s[i]-=32;
     }
     it=b.find(s);
     if(it==b.end())
      b[s]=count++;
     a[s]++;
    }
    for(i=1;i<count;i++)
    {
    	for(it=b.begin();it!=b.end();it++)
    	  {
    	  	if(it->second==i)
    	  	  break;
    	  }
    	  cout<<it->first<<' '<<a[it->first]<<'\n';
    }
    return 0;
}
