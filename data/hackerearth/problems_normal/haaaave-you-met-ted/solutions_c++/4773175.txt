#include <iostream>
#include <math.h>
#include <algorithm>
using namespace std;

int main()
{
    int size;
    cin>>size;
    while(size -- ){
    	int intcount;
    	cin>>intcount;
    	int minnum;
    	for(int i = 0; i< intcount; i++){
    		unsigned long long int num;
    		cin>>num;
    		if(i == 0)
    			minnum = __builtin_popcount(num);
    		else
    			minnum = min(minnum, __builtin_popcount(num));
    	}
    	cout<<minnum<<"\n";
    }
    return 0;
}
