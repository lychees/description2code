#define MOD 1000000007
using namespace std;
long long arr[1000010];
int main()
{
	int i, n, t;
	cin >> t;
	while (t--)
	{
		cin >> n;
		for (i = 0; i<n; i++)
		{
			cin >> arr[i];
		}
		sort(arr, arr + n, greater<long long>());
		long long ans = 0, temp;
		for (i = 0; i<n - 1; i++)
		{
			ans = (ans + (n - i - 1)*arr[i]) % MOD;
		}
		temp = MOD*(long long)100000;
		for (i = n - 1; i>0; i--)
		{
			ans = (temp + ans - (i)*arr[i]) % MOD;
		}
		ans = (ans*arr[0]) % MOD;
		cout << ans << endl;
	}
	return(0);
}
