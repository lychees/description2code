#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int a[3];
    	for(int i=0; i<3; i++)
    		cin>>a[i];
    		
    	int nimsum = 0;
    	for(int i=0; i<3; i++)
    		nimsum = nimsum ^ a[i];
    		
    	if(nimsum == 0)
    		cout<<"BILL"<<endl;
    		
    	else
    		cout<<"NO BILL"<<endl;
    }
    return 0;
}
