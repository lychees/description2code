#include<bits/stdc++.h>
using namespace std;
int main()
{
    char s[100000],a[26];
    cin>>s;
    int i;
    for(i=0;i<26;i++)
        a[i]=0;
    int l=strlen(s);
    for(i=0;i<l;i++){
        a[s[i]-'a']++;
    }
    int c=0;
    for(i=0;i<26;i++)
    {
      if(a[i]%2==1)
       c++;
       if(c>1)
        break;
    }
    if(c>1)
        cout<<"NO"<<endl;
    else
        cout<<"YES"<<endl;
}
