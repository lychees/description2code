N, Q = map(int, raw_input().split())
T = map(int, raw_input().split())
S = map(int, raw_input().split())
tup = [(S[counter], t) for counter, t in enumerate(T)]
s_tup = sorted(tup, key=lambda tp: tp[0], reverse=True)
sum_val = [0 for i in xrange(N)]
sum_val[0] = s_tup[0][1]
for i in xrange(1, N):
    sum_val[i] += sum_val[i-1] + s_tup[i][1]
for i in xrange(Q):
    n = int(raw_input())
    print sum_val[n-1]
