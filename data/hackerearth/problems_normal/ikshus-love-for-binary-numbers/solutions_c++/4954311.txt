#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

int n, k;
unsigned long long arr[61][61];

unsigned long long pow(int x, int y)
{
	unsigned long long result = 1;
	for (int i = 1; i <= y; i++)
		result = result*x;
	return result;
}


int main()
{
	scanf("%d %d", &n, &k);

	int i, j;
	for (i = 1; i <= 60; i++)
		arr[i][i] = 1;

	for (i = 1; i <= 60; i++)
	for (j = 1; j <= i - 1; j++)
		arr[i][j] = 2 * arr[i - 1][j] + pow(2,i - j - 1) - arr[i - j - 1][j];

	unsigned long long numerator = arr[n][k];
	unsigned long long denominator = pow(2,n);

	while (numerator % 2 == 0)
	{
		numerator = numerator / 2;
		denominator = denominator / 2;
	}
	printf("%llu/%llu", numerator, denominator);
	return 0;
}
