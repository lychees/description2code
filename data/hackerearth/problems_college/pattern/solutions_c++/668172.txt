#include <iostream>
using namespace std;

int main()
{
    int n,j,i;
    cin>>n;
    
    for(i=0;i<n;i++){
    	
    	cout<<"*";
    }
    cout<<endl;
    
    for(i=1;i<n-1;i++){
    	for(j=n-i;j>1;j--){
    		
    		cout<<" ";
    	}
    	
    	cout<<"*";
    	cout<<endl;
    }
    
    for(i=0;i<n;i++){
    	
    	cout<<"*";
    }
    
    return 0;
}
