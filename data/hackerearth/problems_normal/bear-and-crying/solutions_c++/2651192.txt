#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;

#define MOD 1000000007

long nop(string s){
	int res=0, current=0, semicount=0;
	for(int i=0;i<s.size();i++){
		if(s[i]=='_'){
			current+=current+semicount;
			current= current%MOD;
		}else{
			res+=current;
			res=res%MOD;
			semicount++;
		}
	}
	return res;
}

int main()
{
	string s;
	cin>>s;
	cout<<nop(s);
    return 0;
}
