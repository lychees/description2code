#include<bits/stdc++.h>
using namespace std;
typedef long long int lu;
#define de 1000001
int main()
{
	lu t,l,r,k,i,j;
	scanf("%lld",&t);
	int pr[de]={0};
	pr[0]=pr[1]=1;
	for(i=2;i*i<de;i++)
	{
		k=2;
		if(pr[i]==0)
		for(j=i*k;j<de;)
		{
			//cout<<i<<" ";
			if(j%i==0)
			pr[j]=1;
			k++;
			j=i*k;
		}
	}
	
	vector<lu>ar(de);
	ar[0]=ar[1]=0;
	ar[2]=2;
	j=2;
	
	for(i=3;i<de;i++)
	{
		if(pr[i]==0)
	{ j=j+i;}
		ar[i]=j;
		
	}
	while(t--)
	{
		scanf("%lld%lld",&l,&r);
		j=ar[r]-ar[l];
		if(pr[l]==0)
		j=j+l;
		printf("%lld\n",j);
	}
	return 0;
}
