#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int a[n][n];
    for(int i=0;i<n;i++)
    	for(int j=0;j<n;j++){
    		cin>>a[i][j];
    	}
    vector<int> ans(n);
    for(int i=0;i<n;i++){
    	ans[i]=a[i][n-1];
    }
    
    for(int i=n-2;i>=0;i--){
    	ans[0]+=a[0][i];
    	for(int j=1;j<n;j++){
    		ans[j]=min(ans[j-1]+a[j][i],ans[j]+a[j][i]);
    	}
    	
    	for(int j=n-2;j>=0;j--)
    		ans[j]=min(ans[j],ans[j+1]+a[j][i]);
    }
    cout<<*min_element(ans.begin(), ans.end());
    
}
