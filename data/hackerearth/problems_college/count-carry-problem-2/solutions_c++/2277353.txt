#include<bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int a,b;
		long long int c=0;
		long long int temp=0;
		cin>>a>>b;
		while(a!=0 || b!=0)
		{
			int s=a%10;
			int z=b%10;
			
			int m=0;
			m=s+z+temp;
			
			temp=m/10;
			if(temp>0)
			c++;
			
			a/=10;
			b/=10;
		
		}
		
		if(c==0)
		cout<<"No carry operation"<<endl;
		else if(c==1)
		cout<<"1 carry operation"<<endl;
		else
		cout<<c<<" carry operations"<<endl;
		
		
		
	}
}
