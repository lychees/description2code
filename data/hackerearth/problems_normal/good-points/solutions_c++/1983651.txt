#include<bits/stdc++.h>
#define ll long long

using namespace std;

ll power(ll a,ll b,ll mod)
{
    ll ret=1;

    while(b)
    {
        if(b%2==1)
            ret=(ret*a)%mod;
        b/=2;
        a=(a*a)%mod;
    }
    return ret;
}

ll gcd(ll a,ll b)
{
    return b==0?a:gcd(b,a%b);
}
ll sumDig(ll n)
{
	ll sum=0;
	while(n)
	{
		sum+=(n%10);
		n/=10;
	}
	return sum;
}
