#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int a,b,n;
    	cin>>a>>b>>n;
    	int p[n];
    	for(int i=0;i<n;i++)
    	cin>>p[i];
    	sort(p,p+n);
    	long long k=0,s=0;
    	for(int j=0;j<n;j++)
    	{
    		s+=p[j];
    		
    		if(s>a&&s<=b)
    		{
    			k=2;
    			break;
    		}
    		else
    		if(s>b&&s<=a)
    		{
    			k=3;
    			break;
    		}
    	}
    	
    	
    	if(k==2)
    	cout<<"Sayan Won"<<endl;
    	else
    	if(k==3)
    	cout<<"Raghu Won"<<endl;
    	else
    	cout<<"Tie"<<endl;
    }
    return 0;
}
