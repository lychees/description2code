#include <bits/stdc++.h>
using namespace std;

int add(int n) {
	return n*(n+1)/2;	
}

int main()
{
	string s; cin >> s;
	int ans = 0;
	int prev = '-';
	int c = 0;
	for (int i = 0; i < s.length(); ++i) {
		if (s[i] == prev) ++c;
		else {
			ans += add(c);
			c = 1;
		}
		prev = s[i];
	}
	cout << ans + add(c);
	
	return 0;
}
