#include <iostream>
#include <vector>
#include <algorithm>
#include <string>
#include <sstream>
using namespace std;

string convert(string & strr){
	string converted;
	for(unsigned short i=0;i<strr.size();++i){
		converted+=toupper(strr[i]);
	}
	return converted;
	
}
int main()
{
    string sentence,word;
    vector<string> svec;
    vector<string> ssvec;
    getline(cin,sentence);
    istringstream iss(sentence);
    while(iss>>word){
    	svec.push_back(convert(word));
    	
    }
    for(vector<string>::iterator it=svec.begin();it!=svec.end();++it){
    	// cout<<*it<<" "<<count(svec.begin(),svec.end(),*it);
    	auto r=find(ssvec.begin(),ssvec.end(),*it);
    	if(r==ssvec.end()){
    		cout<<*it<<" "<<count(svec.begin(),svec.end(),*it)<<endl;
    		ssvec.push_back(*it);
    	}
    	else{
    		continue;
    	}
    }
    return 0;
}
