#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<vector>
using namespace std;
int T,n,m,C[1002][1002];
int main()
{
	cin>>T;
	while(T--)
	{
		cin>>n>>m;
		vector<int>A(n+1),V(n+1);
		for(int i=1;i<=n;i++)cin>>A[i];
		for(int i=1;i<=n;i++)cin>>V[i];
		
		for(int i=0;i<=n;i++)
		{
			for(int j=0;j<=m;j++)C[i][j]=0;
		}
		
		for(int i=1;i<=n;i++)
		{
			for(int j=1;j<=m;j++)
			if(j>=A[i])
			{
				C[i][j]=max(C[i-1][j],V[i]+C[i-1][j-A[i]]);
			}
			else
			{
				C[i][j]=C[i-1][j];
			}
		}
		/*for(int i=1;i<=n;i++)
		{
			for(int j=1;j<=m;j++)cout<<C[i][j]<<"  ";
			cout<<endl;
		}*/
		cout<<C[n][m]<<"\n";
	}
	return 0;
}
