for _ in range(input()):
    z = int(raw_input())
    k = map(int, raw_input().split())
    k = list(k)
    k = sorted(k)
    a = sum(k)/z
    if (z % 2 == 0):
        m = (k[z/2]+k[(z/2)-1])/2
    else:
        m = (k[z/2])
    p = 0
    q = 0
    for i in k:
        p += abs(m-i)
        q += abs(a-i)
    print min(p, q)
