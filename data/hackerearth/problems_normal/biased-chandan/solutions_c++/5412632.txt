#include<iostream>
using namespace std;
#define max 100000
int stack[max],top=-1;
void push(int value)
{
	stack[++top]=value;
}
void pop()
{
	top--;
}
void sum()
{
	int value=0;
	for(int i=0;i<=top;i++)
	value=value+stack[i];
	cout<<value<<endl;
}
int main()
{
	int n,temp;
	cin>>n;
	for(int i=0;i<n;i++)
	{cin>>temp;
	if(temp==0&&top!=-1)
	pop();
	else if(temp!=0)
	push(temp);
	}
	sum();
	return 0;
}
