#include <iostream>
using namespace std;

int main()
{
	long t, a, b, c, rc;
	
   	cin>>t;
   	
   	while (t--)
   	{
   		rc = 0;
   		cin>>a>>b>>c;
   		switch (c)
   		{
   			case 0: rc = a; break;
   			case 1: rc = b; break;
   			default:
   				for (int i = 1; i < c; i++)
   				{
   					rc = a + b;
   					a = b;
   					b = rc;
   				} 
   				break;
   		}
   		cout<<rc<<endl;
   	}
    return 0;
}
