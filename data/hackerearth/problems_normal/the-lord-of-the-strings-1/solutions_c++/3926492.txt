#include <bits/stdc++.h>

#include <set>
using namespace std;

int search(int i,string s)
{
	bool flag;
	for(int j=0;j<i;j++)
	{
		flag=false;
		if(s[j]==s[i])
			break;
		flag=true;
	}
	if(flag)
		cout << s[i];
}
int main()
{
	int t;
	string s;
	for(cin>>t;t;t--)
	{
		cin >> s;
		cout << s[0];
		for(int i=1;i<s.size();i++)
		{
			search(i,s);
		}
		cout << endl;
	}
}
