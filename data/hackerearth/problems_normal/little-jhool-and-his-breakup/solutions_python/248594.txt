from collections import deque
def jhole():
    msg = raw_input()
    mstr ,d = ['l','o','v','e'], deque()
    for i in msg:
        #import pdb;pdb.set_trace()
        if mstr[len(d)] == i:
            d.append(i)
            if len(d) == 4:
                break

    if "love" == "".join(d):
        print "I love you, too!"
    else:
        print "Let us breakup!"

if __name__ == '__main__':
    jhole()
