#include<bits/stdc++.h>
using namespace std;

int main()
{
    int flag=0;
    int max=0;
    int min=100;
    string s;
    getline(cin,s);
    map<char,int> mp;
    for(int i=0;i<s.size();++i)
    {
        mp[s[i]]++;
    }
    for(int i=0;i<s.size();++i)
    {
        int temp=mp[s[i]];
        if(temp>max)
        {
            max=temp;
            min=(int)s[i];
            flag=1;
        }
        else if(temp== max)
        {
            if((int)s[i]<min)
            {
                min=(int)s[i];
                flag=1;
            }
        }
    }
    char Ascii=(char)min;
    cout<<Ascii<<" "<<max;
    return 0;
}
