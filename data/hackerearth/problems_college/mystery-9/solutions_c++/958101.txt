#include <iostream>
#include<cstdio>
#include<cmath>
using namespace std;
int main()
{
	int t,n,i,ans;
	int a[12]={31,28,31,30,31,30,31,31,30,31,30,31};
	cin>>t;
	while(t--)
	{
		cin>>n;
		ans=0;
		for(i=0;i<12;i++)
		{
			ans+=a[i];
			if(ans>=n)
			{
				ans=i;
				break;
			}
		}
		cout<<i+1<<endl;		
	}
    return 0;
}
