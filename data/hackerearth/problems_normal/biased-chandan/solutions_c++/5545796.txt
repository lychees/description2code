#include <bits/stdc++.h>
using namespace std;
int arr[5003];
stack<int>st;
int main()
{
	int n, ans = 0;
	cin>>n;
	for(int i=1; i<=n; i++){
		cin>>arr[i];
		if(arr[i]==0){
			if(st.size())
				st.pop();
		}
		else
			st.push(arr[i]);
	}
	if(st.size())
		while(st.size()){
			ans+=st.top();
			st.pop();
		}
	
	cout<<ans<<endl;
    return 0;
}
