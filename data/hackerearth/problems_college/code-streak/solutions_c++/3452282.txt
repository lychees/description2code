#define max(a,b) a>b?a:b
#define min(a,b) a<b?a:b

void scan(int &x); //IMPORTANT: for positive only

int main(void)
{
  int t;
  
  cin >> t;
  
  while(t--)
  {
  int n,x,y,i,ans=0,tmp=0;

  cin >> n;
  
  bool solved[1000006];
  memset(solved,false,sizeof(solved));
  for(i=0;i<n;i++)
  {
    scan(x);
    scan(y);
    
    if(y)
    {
    	if(!solved[x])
    	   {
    	   	tmp++;
    	   	solved[x] = true;
    	   }
    }
    else
      tmp = 0;
      
    ans = max(tmp,ans);
  }

  cout << ans << endl;
  }
  return 0;
}
