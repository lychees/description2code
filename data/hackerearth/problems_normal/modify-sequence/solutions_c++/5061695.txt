#include<iostream>
using namespace std;
int main()
{
	int n,sum_e=0,sum_o=0;
	cin>>n;
	int a[n];
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
	}
	for(int j=0;j<n;j++)
	{
		if(j%2==0)
		{
		sum_e=sum_e+a[j];	
		}
		else
		{
			sum_o=sum_o+a[j];
		}
	}
	
	if(sum_e==sum_o)
	{
		cout<<"YES";

	}
	else
	cout<<"NO";
	return 0;
}
