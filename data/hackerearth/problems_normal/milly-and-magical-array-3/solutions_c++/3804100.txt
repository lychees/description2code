#include <iostream>
#include <vector>
using namespace std;
typedef long long int intt;
int main()
{
    intt t,N,X;
    cin>>t;
   
    for(intt i=0;i<t;i++)
    {cin>>N>>X;
   		vector <intt> arr;
   	for(intt j=0;j<N;j++)
   	{
   		intt a;
   		cin>>a;
   		arr.push_back(a);
   	}
   	intt count=0;
   	for(intt j=0;j<N-1;j++)
   	{
   		if(arr[j+1]<=arr[j])
   		{intt d;
   			d=(arr[j]-arr[j+1])/X;
   			if(arr[j+1]+X*d<=arr[j])
   			d++;
   			count+=d;
   			arr[j+1]+=X*d;
   		//	int c=count;
   		//	if (count>c)
   		//	count++;
   			
   		}
   	}
    cout<<count<<endl;	
    }
    return 0;
}
