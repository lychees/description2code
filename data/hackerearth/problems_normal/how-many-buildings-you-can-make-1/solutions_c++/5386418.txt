#include <iostream>

int f (int s) {
	if (s == 1) return 1;
	int h = s >> 1;
	int ans = 0;
	for (int width = 1 ; width <= h ; width++) {
		int cols = s / width;
		if (cols >= width) ans += cols - width + 1;
	}
	return ans;
}

int main () {
	int n = 0;
	std::cin >> n;
	// for (int v = 1 ; v <= 1000 ; v++) {
	// 	std::cout << f(v) << '\n';
	// }
	for (int i = 0 ; i < n ; i += 1) {
		int s = 0;
		std::cin >> s;
		std::cout << f(s) << '\n';
	}
}
