#include<iostream>

using namespace std;
int **a;

void splendid(int init, int x, int y,int n)
{
	if (n == 1)
	{
		a[x][y] = init;
		a[x][y + 1] = init + 1;
		a[x + 1][y] = init + 2;
		a[x + 1][y + 1] = init + 3;
		return;
	}
	else
	{
		splendid(init, x, y, n - 1);
		splendid(init + (1 << 2*(n-1)), x, y + (1 << (n - 1)), n - 1);
		splendid(init + 2 * (1 << 2*(n-1)), x + (1 << (n - 1)), y, n - 1);
		splendid(init + 3 * (1 << 2*(n-1)), x + (1 << (n - 1)), y + (1 << (n - 1)), n - 1);
	}
}

int main()
{
	int n;
	cin >> n;
	int s = 1 << n;
	a = new int*[s];
	for (int i = 0; i < s; i++)
		a[i] = new int[s];
	int init = 1;
	int x = 0, y = 0;
	splendid(init, x, y, n);

	for (int i = 0; i < s; i++)
	{
		for (int j = 0; j < s; j++)
		{
			cout << a[i][j] << " ";
		}
		cout << endl;
	}
}
