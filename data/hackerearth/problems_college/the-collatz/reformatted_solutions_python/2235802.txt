n = input()
for _ in range(n):
    oddsum = 1
    evensum = 0
    count = 0
    num = input()
    while num != 1:
        if num & 1:
            oddsum += num
            num = num*3+1
        else:
            evensum += num
            num /= 2
        count += 1
    # print oddsum,evensum,count
    if (oddsum % count) == (evensum % count):
        print 'Tie'
    elif (oddsum % count) > (evensum % count):
        print 'Odd Rules'
    else:
        print 'Even Rules'
