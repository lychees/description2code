#include <iostream>
using namespace std;

int main()
{
	int n,k = 0, t;
	cin >> t;
	while(t--){
	cin >> n;
    for(int i = 0 ; i <= n/6 ;i++){
    	for(int j = 0; j <= n/9;j++){
    		for(int z = 0 ; z <= n/20 ; z++){
    			if(6*i + 9*j + 20*z == n){
    				cout << "True" << '\n';
    				k = 1;
    				break;
    				
    			}
    			if(k == 1)break;
    		}
    		if(k == 1)break;
    	}
    }
    if(k == 0)cout << "False" << '\n';
     k = 0;
	}
    
    return 0;
}
