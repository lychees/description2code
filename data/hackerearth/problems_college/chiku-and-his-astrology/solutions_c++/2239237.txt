#include <iostream>
#include<string>

using namespace std;

int main()
{
    int t,i,j=0;
   long int l=0;
    long int n;
    cin>>t;
    while(t--)
    {
    	string str;
    	cin>>str;
    	l=str.length();
    	int count=0;
    	int left,right;
    	for(i=0;i<l-1;i++)
    	{
    		if((str[i]=='1' && str[i+1]=='0') || (str[i]=='0' && str[i+1]=='1'))
    		{
    			str[i]=str[i+1]='9';
    			count+=2;
    			left=i-1;
    			right=i+2;
    			i++;
    			int flag=1;
    			j++;
    			while(flag && left>=0 &&  right<l)
    			{
    				j--;
    				if((str[left]=='1' && str[right]=='0') || (str[left]=='0' && str[right]=='1'))
    				{
    					str[left]=str[right]='9';
    					
    					while(str[left]=='9')
    						left--;
    					right++;
    					i++;
    					flag=1;
    					count+=2;
    				}
    				else 
    				flag=0;
    			}
    		}
    		
    	}
    	cout<<(l-count)<<endl;
    }
    
    return 0;
}
