#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
#define F(i,a,b) for(ll i = (ll)(a); i <= (ll)(b); i++)
ll A[1000001],S[1000001];
int main()
{
    #ifndef ONLINE_JUDGE
        freopen("input.txt","r",stdin);
        freopen("output.txt","w",stdout);
    #endif
    ll N,M,K,a,b;
    scanf("%lld",&N);
    F(i,0,N-1) scanf("%lld",&A[i]);
    scanf("%lld",&M);
    while(M--)
    {
        scanf("%lld%lld%lld",&a,&b,&K);
        a--;
        b--;
        S[a] += K;
        S[b+1] -= K;
    }
    F(i,1,N-1) S[i] += S[i-1];
    F(i,0,N-1) A[i] += S[i];
    F(i,0,N-1) printf("%lld\n",A[i]);
    return 0;
}
