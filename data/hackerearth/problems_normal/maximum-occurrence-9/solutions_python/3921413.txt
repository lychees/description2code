from collections import Counter
strn = raw_input()
c = Counter(strn)
DictNew = {}
DictNew = c.most_common(len(c.items()))
ListKey = []
MaxVal = DictNew[0][1]
for k,v in DictNew:
    if v == MaxVal:
        ListKey.append(k)
minORD = ListKey[0]
for i in range(1, len(ListKey)):
    if ord(str(ListKey[i])) < ord(str(minORD)):
        minORD = str(ListKey[i])

print minORD, MaxVal
