n, q = map(int, raw_input().split())
l = []
time = list(map(int, raw_input().split()))
score = list(map(int, raw_input().split()))
s = 0
for i in range(n):
    tmp = []
    tmp.append(time[i])
    s += time[i]
    tmp.append(score[i])
    l.append(tmp)

l.sort()
a = []
a.append(s)
for i in range(n):
    s -= l[i][0]
    a.append(s)

for i in range(q):
    k = int(raw_input())
    print a[n - k]
