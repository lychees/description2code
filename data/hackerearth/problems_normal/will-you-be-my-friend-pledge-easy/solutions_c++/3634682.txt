#include<iostream>
using namespace std;
int gcd(int a,int b)
{
	if(b==0)
	return 0;
	if(b==1)
	return 1;
	else
	return gcd(b,a%b);
}
int main()
{
	int no,n,l,s=0,k;
	cin>>no;
	cin>>n;
	while(n--)
	{
		cin>>l;
		if(no>l)
			k=gcd(no,l);
		else
		k=gcd(l,no);
		if(k!=1)
		s++;
	}
	cout<<s<<"\n";
}
