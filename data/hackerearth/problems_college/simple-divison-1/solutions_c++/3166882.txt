#include <iostream>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    	long long int a,b,c,r,q;
    	scanf("%lld %lld %lld",&a,&b,&c);
    	printf("%lld",a/b);
    	r=a%b;
    	if(c!=0)
    		printf(".");
    	while(c--){
    		r=r*10;
    		q=r/b;
    		printf("%lld",q);
    		r-=(b*q);
    	}
    	printf("\n");
    }
    return 0;
}
