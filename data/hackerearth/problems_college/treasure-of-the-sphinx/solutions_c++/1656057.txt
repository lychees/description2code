int mat[204][20004];
int main()
{
	int t, n,  i, j, x, l, tot, s;
	inp (t);
	mat[0][0] = 0;
		rep (i, 202) {
			mat[i][0] = 0;
		}
		rep (i, 20002) {
			mat[0][i] = 0;
		}
	while (t--) {
		inp2 (n, s);
		tot = 0;
		int wt[n + 1]; 
		rep (i, n) {
			inp (wt[i + 1]);
			tot += wt[i + 1];
		}
		l = (tot + 1) / 2;
		if (l > s) {
			printf ("NO\n");
			continue;
		}
		
		for (i = 1; i <= n; i++) {
			for (j = 1; j <= l; j++) {
				if (wt[i] <= j) {
					mat[i][j] = max(mat[i - 1][j], wt[i] + mat[i - 1][j - wt[i]]);
				}
				else
					mat[i][j] = mat[i - 1][j];
			}
		}
		if (tot - mat[n][l] > s) 
			printf ("NO\n");
		else
			printf ("YES\n");
	}
    return 0;
    
}
