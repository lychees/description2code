#include <iostream>
#include<stdio.h>
#define MAX 4000010
using namespace std;

bool prime[MAX];

void sieve()
{
	for(int i=1; i<=MAX; i++)
		prime[i]=1;
	for(int i=1; i<=MAX; i++) {
		if(i==1 || prime[i]==0) {
			prime[i]=0;
			continue;
		}
		for(int j=2; i*j<=MAX; j++)
			prime[i*j]=0;
	}
}

int main()
{
    sieve();
    long n;
    scanf("%ld",&n);
    int i=2;
    long long sum=0;
    while(i<n) {
    	if(prime[i])
    		sum+=i;
    	i++;
    }
    printf("%lld",sum);
    return 0;
}
