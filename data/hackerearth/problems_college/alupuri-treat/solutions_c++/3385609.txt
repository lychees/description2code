#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
    int m;
    //int *a;
    scanf("%d",&m);
    int a[100][100];
    for(int i=0;i<m;i++)
    {
		for(int j=0;j<m;j++)
		{
			scanf("%d",&a[i][j]);
		}
    }
    int s1=0,s2=0;
    int p=m-1;
    for(int i=0;i<m;i++)
    {
    	s1=s1+a[i][i];
    	s2=s2+a[i][p];
    	p--;
    }
    s1=s1-s2;
    if(s1<0)
    s1=(-s1);
    printf("%d",s1);
    
    return 0;
}
