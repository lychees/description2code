

    #include <iostream>
    #include<algorithm>
    using namespace std;
     
    int main()
    {int t;
    cin>>t;
    while(t--){long long n;
    cin>>n;
    long long ar[n],sell[n],profit=0;
    for(int i=0;i<n;++i)
    cin>>ar[i];
    for(int i=n-1;i>=0;--i){
    if(i==n-1)
    sell[i]=ar[i];
    else
    sell[i]=max(sell[i+1],ar[i]);
    }
    for(int i=0;i<n;++i)
    profit+=(sell[i]-ar[i]);
    cout<<profit<<endl;
    }
    //cout << "Hello World!" << endl;
    return 0;
    }
