#include <iostream>
using namespace std;

int main()
{
    int t,n,i;
    cin>>t;
    while(t--){
    	cin>>n;
    	int *a = (int *)malloc(sizeof(int)*(n+2));
    	a[0] = 0;
    	for(i=1;i<=n;i++)
    	{
    		int x;
    		cin>>x;
    		a[i] = a[i-1]+x;
    	}
    	
    	int flag=0;
    	for(i=1;i<=n;i++){
    		if(a[i-1]==a[n]-a[i]){
    			flag=1;
    			break;
    		}
    	}    	
    	
    	if(flag==1){
    		cout<<"YES"<<endl;
    	}else{
    		cout<<"NO"<<endl;
    	}
    }
    
    return 0;
}
