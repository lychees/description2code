#include <iostream>
using namespace std;

int main()
{
	long long n,m,a;
	cin>>n>>m>>a;
	long long ans1=1,ans2=1;
	if((n<=a)&&(m<=a))
	{
		cout<<"1"<<endl;
	}
	else if((n>a)&&(m<a))
	{
		if(n%a==0)
			ans1*=n/a;
		else
			ans1*= (n/a)+1;
	}
	else if((n<a)&&(m>a))
	{
		if(m%a==0)
			ans2*=m/a;
		else
			ans2*= (m/a)+1;
	}
	else
	{
		if(n%a==0)
			ans1*=n/a;
		else
			ans1*= (n/a)+1;
		if(m%a==0)
			ans2*=m/a;
		else
			ans2*= (m/a)+1;
	}
	
	
	cout<<ans1*ans2<<endl;
    //cout << "Hello World!" << endl;
    return 0;
}
