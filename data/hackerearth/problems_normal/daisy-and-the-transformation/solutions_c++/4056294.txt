#include <iostream>
#include <cstring>

using namespace std;

int main()
{
    int T,N,total;
    char s1[100],s2[100];
    cin>>T;
    while (T--){
    	total=0;
    	cin>>s1>>s2>>N;
        if (strlen(s1)==strlen(s2)){
    	for (int i=0;s1[i]!='\0';i++){
    		if (s1[i]!=s2[i])
    			total++;
    	}
    	if (N==total || (N>total && (N-total)%2==0))
    		cout<<"Yes"<<"\n";
    	else
    		cout<<"No"<<"\n";
    }else{
       cout<<"No"<<"\n"; 
    }
    }
    return 0;
}
