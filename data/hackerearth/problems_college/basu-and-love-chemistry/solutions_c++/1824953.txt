#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#define N 1000000007
#define iLONG long long int
 
using namespace std;

//To get the subdigit's
iLONG getDigit(char* num)
{
  int len = strlen(num);
  iLONG result = num[len-1] - 48;

  len--;
  while(len > 0)
  {
        result = result*10 + (num[len-1]-48);
        //cout<<result<<" "<<num[len-1]<<endl;
        len--;
  }
  return result;
}
 
int main()
{
	char str[200001];// = (char*) malloc (sizeof(char)*N);
	scanf("%s",str);
	iLONG result = str[0]-'0';
  iLONG sum = result;

  for(int i=1;i<strlen(str);i++)
  {
    sum = sum*10 + (str[i]-'0')*(i+1);
    sum %= N;
    result += sum;
    result %= N; 
  }
  cout<<result<<endl;
	return 0;
}
