#include <bits/stdc++.h>
using namespace std;
int A[1001];

int findMin(int n){
	int min = 100000;
	for (int i = 0; i < n; i++)
	{
		if(A[i] < min && (A[i]!=0) ){
			min = A[i];
		}
	}
	if(min!=100000)
		return min;
	else
		return -1;
}

int count(int n, int min){
	int count=0;
	for (int i = 0; i < n; i++)
	{
		if(A[i]!=0){
			A[i]-=min;
			count++;
		}
	}
	return count;
}

int main(){
    std::ios::sync_with_stdio(false);
	int i,j,k,n,a,b;
	cin >> n;
	for (int i = 0; i < n; i++)
	{
		cin >> A[i];
	}
	int temp = findMin(n);
	while(temp!=-1){
		int val = count(n, temp);
		cout << val << endl;
		temp = findMin(n);
	}
	return 0;
}
