#include <iostream>
#include <cstdio>

using namespace std;
	
long sum[100005] = {0};
int n;

int bin_search(int l, int h, long q)
{
	while(l<h)
	{
		int mid = l+ (h-l)/2;
		if (q == sum[mid])
			return mid;
		else if (q < sum[mid])
			return bin_search(1, mid, q);
		else
			return bin_search(mid+1, h, q);
	}

	return l;
}

int main(int argc, char const *argv[])
{
	int i,Q;
	long c[100005] = {0};

	scanf("%d", &n);

	for(i=1; i<=n; i++)
	{
		scanf("%ld", &c[i]);
		sum[i] = sum[i-1] + c[i];
	}

	scanf("%d", &Q);
	while(Q--)
	{
		long q;
		int res=0;
		scanf("%d", &q);
		res = bin_search(1, n, q);
		printf("%d\n", res);
	}
	return 0;
}
