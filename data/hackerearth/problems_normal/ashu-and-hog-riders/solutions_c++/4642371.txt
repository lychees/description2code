#include<bits/stdc++.h>
bool a[1001][1001];
int n;
using namespace std;
int bfs(int i,int j)
{
	a[i][j]=0;
	int count=1;
	if(i-1>0 && a[i-1][j])
	   count += bfs(i-1,j);
	if(j-1>0 && a[i][j-1])
	   count+= bfs(i,j-1);
	if(j+1<=n && a[i][j+1])
	   count += bfs(i,j+1);
	if(i+1 <= n && a[i+1][j])
	   count += bfs(i+1,j);
	return count;
}
main()
{
	cin>>n;
	memset(a,0,sizeof(a));
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=n;j++)
		   cin>>a[i][j];
	}
	int max1=0;
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=n;j++)
		{
			if(a[i][j])
			{
			     int temp=bfs(i,j);
			     max1=max1>temp?max1:temp;
			}
		}
	}
	cout<<max1;
}
