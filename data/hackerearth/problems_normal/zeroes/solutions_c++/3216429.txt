#include <bits/stdc++.h>
#define ll long long
#define mod 1000000000
using namespace std;
ll int h[100010],h2[100010],a[100010];

ll int factorise(ll int x)
{
	ll int i,t=x;
	for(i=2;(i*i)<=t;i++)
	{
		if((x%i)==0)
		{
			while((x%i)==0)
			{
				h[i]++;
				x/=i;
			}
		}
	}
	if(x!=1)
	h[x]++;
}
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	memset(h2,0,sizeof(h2));
	memset(h,0,sizeof(h));
	ll int n,q,b;
	cin>>n;
	ll int i,j,t;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
		factorise(a[i]);
	}
	cin>>q;
	for(i=0;i<q;i++)
	{
		set<ll int> s;
		cin>>b;
		t=b;
		for(j=2;(j*j)<=t;j++)
		{
			while((b%j)==0)
			{
				s.insert(j);
				h2[j]++;
				b/=j;
			}
		}
		if(b!=1)
		{
			s.insert(b);
			h2[b]++;
		}
		ll int mc=INT_MAX;
		set<ll int>::iterator it;
		for(it=s.begin();it!=s.end();it++)
		{
			mc=min(mc,h[*it]/h2[*it]);
			h2[*it]=0;
		}
		cout<<mc<<endl;
	}
	return 0;
}
