#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll long long int
#define pb push_back
#define mk make_pair
ll power(ll a, ll b) {
ll x = 1, y = a;
    while(b > 0) {
        if(b%2 == 1) {
            x=(x*y);
            if(x>mod) x%=mod;
        }
        y = (y*y);
        if(y>mod) y%=mod;
        b /= 2;
    }
    return x;
}
int main()
{
	string s,t;
	cin>>s>>t;
	int i,j;
	int n = s.size();
	int c = 0;
	map<string,int>mp;
	for(i = 0; i < n; i++) {
		string str = "";
		for(j = i; j < n; j++) {
			str += s[j];
			if(mp[str]) continue;
			std::size_t found = str.find(t);
			if(found!=std::string::npos) c++;
			mp[str]++;
		}
	}
	cout<<c<<endl;
	return 0;
}
