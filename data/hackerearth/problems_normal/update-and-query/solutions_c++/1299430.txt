#include<iostream>
#include<cstdio>
#include<stdlib.h>
using namespace std;
int main(){
	long long int N,M,X,num,Z,count,max,min;
	char que,c;
	scanf("%lld %lld",&N,&M);
	long long int arr[N];
	for(long long int i=0;i<N;i++){
		scanf("%lld",&arr[i]);
	}
	while(M--){
		cin >> que;
		cin>>X>>Z;
		//scanf("%c",&c);
		
		//printf("%c %lld %lld",que,X,Z);
		count=0;
		max = arr[X-1];
		min = arr[X-1];
		if(que == 'Q'){
			if(Z < 0){
				printf("-1 -1\n");
			}
			else{
				for(long long int i= X-1;i<N;i++){
					if(arr[X-1]-Z <= arr[i] && arr[X-1]+Z >= arr[i]){
						count++;
						//printf("%lld ",Z);
						if(max < arr[i]){
							max = arr[i];
						}
						if(min > arr[i]){
							min = arr[i];
						}					
					}
					else{
						break;
					}  
				}
				printf("%lld ",count);
				if(abs(arr[X-1]-max) >= abs(arr[X-1]-min)){
					printf("%lld\n",(long long int)abs(arr[X-1]-max));
				}
				else{
					printf("%lld\n",(long long int)abs(arr[X-1]-min));
				}
			}
		}
		else{
			//scanf("%lld %lld",&X,&num);
			arr[X-1]= Z;		
		}
		//scanf("%c",&c);
	}
	return 0;
}
