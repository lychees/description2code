#include <iostream>
using namespace std;

int main()
{
    int t;
    long int n, x, hash[1000006], count;
    cin >> t;
    while(t--){
    	count = 0L;
    	for(long int& i : hash)
    		i = 0L;
    	cin >> n;
    	while(n--){
    		cin >> x;
    		hash[x]++;
    		cin >> x;
    		if(hash[x] > 0L){
    			hash[x]--;
    		}
    		else
    			count++;
    	}
    	cout << count << endl;
    }
    return 0;
}
