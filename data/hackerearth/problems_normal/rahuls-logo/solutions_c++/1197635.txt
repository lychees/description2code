#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;
int main() {
	int n;
	cin>>n;
	if(n!=1) {
	for(int i=1;i<n;i++) {
		cout<<" ";
		
	}
	cout<<"/\\"<<endl;
	for(int i=1;i<n-1;i++) {
		cout<<" ";
	}
	cout<<"/  \\"<<endl;
	int sp=n-3,spm=0;
	for(int i=3;i<=n;i++) {
		for(int j=sp;j>=1;j--) {
			cout<<" ";
		}
		cout<<"/ /";
		
		for(int i=0;i<spm;i++) {
			//cout<<" ";
			if(spm>2 && (i+1)%2==0 && i<(spm/2)) {
				cout<<"/";
			}
			else if(spm>2 && (i)%2==0 && i>=(spm/2)) {
				cout<<"\\";
			}
			else cout<<" ";
		}
		cout<<"\\ \\";
		cout<<endl;
		sp--;spm+=2;
		
	}
	int mid=(n-3)*2,frnt=2;
	for(int j=3;j<=n;j++) {
	for(int h=0;h<frnt;h++) {
		cout<<" ";
	}
	cout<<"\\ \\";
	for(int i=mid;i>0;i--) {
		if(mid>2 && (i+1)%2==0 && i>(mid/2)) {
				cout<<"\\";
			}
			else if((i)%2==0 && i<=(mid/2)) {
				cout<<"/";
			}
			else cout<<" ";
	}
	
	cout<<"/ /"<<endl;
	mid-=2;frnt++;
	}
	for(int i=0;i<frnt;i++) {
		cout<<" ";
	}
	cout<<"\\  /"<<endl;
	for(int i=0;i<frnt+1;i++) {
		cout<<" ";
	}
	cout<<"\\/"<<endl;
	}
	else {
		cout<<"/\\"<<endl;
		cout<<" \\/"<<endl;
	}
	
	
	
	
for(int i=1;i<n+1;i++) {
		cout<<" ";
		
	}
	cout<<"/\\"<<endl;
	for(int i=1;i<n;i++) {
		cout<<" ";
	}
	cout<<"/  \\"<<endl;
	int sp=n+1-3,spm=0;
	for(int i=3;i<=n+1;i++) {
		for(int j=sp;j>=1;j--) {
			cout<<" ";
		}
		cout<<"/ /";
		
		for(int i=0;i<spm;i++) {
			//cout<<" ";
			if(spm>2 && (i+1)%2==0 && i<(spm/2)) {
				cout<<"/";
			}
			else if(spm>2 && (i)%2==0 && i>=(spm/2)) {
				cout<<"\\";
			}
			else cout<<" ";
		}
		cout<<"\\ \\";
		cout<<endl;
		sp--;spm+=2;
		
	}
	int mid=(n+1-3)*2,frnt=2;
	for(int j=3;j<=n+1;j++) {
	for(int h=0;h<frnt;h++) {
		cout<<" ";
	}
	cout<<"\\ \\";
	for(int i=mid;i>0;i--) {
		if(mid>2 && (i+1)%2==0 && i>(mid/2)) {
				cout<<"\\";
			}
			else if((i)%2==0 && i<=(mid/2)) {
				cout<<"/";
			}
			else cout<<" ";
	}
	
	cout<<"/ /"<<endl;
	mid-=2;frnt++;
	}
	for(int i=0;i<frnt;i++) {
		cout<<" ";
	}
	cout<<"\\  /"<<endl;
	for(int i=0;i<frnt+1;i++) {
		cout<<" ";
	}
	cout<<"\\/";
	return 0;
}
