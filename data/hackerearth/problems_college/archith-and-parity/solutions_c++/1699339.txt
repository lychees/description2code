#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	long long c=0;
    	while(n>0)
    	{
    		int p=n%2;
    		if(p==1)
    		c++;
    		n=n/2;
    	}
    	if(c%2==0)
    	cout<<"even"<<endl;
    	else
    	cout<<"odd"<<endl;
    }
    return 0;
}
