#include<iostream>
using namespace std;

int check_pallindrome(string s){
	int FLAG = 1;
	for(int i = 0; i < s.length()/2; i++){
		if(s[i] != s[s.length() - i - 1])
			FLAG = 0;
	}
	return FLAG;
}

int main(){
	long long int n, mul = 1;
	cin >> n;
	while(n--){
		mul = 1;
		string s;
		cin >> s;
		int p = check_pallindrome(s);
		if(p)
			cout << "Palindrome" << "\n";
		else{
			for(int i = 0; i < s.length(); i++){
				mul *= s[i] - 96;
			}
			cout << mul << "\n";
		}
	}
}
