#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int f[n],s[n],c[n];
    
    for(int i = 0;i<n;i++)
    	{cin>>f[i]>>s[i];c[i]=f[i]+s[i];}
    sort(c,c+n);
    for(int i = 0;i<n;i++)
    	{
    		for(int j=0;j<n;j++)
    		{
    			if(c[i]==f[j]+s[j])
    			{cout<<j+1<<" ";f[j]=-1;break;}
    		}
		}
}
