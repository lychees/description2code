#include <bits/stdc++.h>

using namespace std;

#define REP(i , a) for(int i = 0 ; i < a.size() ; i++)
#define reset(a , b) memset(a , b , sizeof(a))

const int INF = 1000000007;
const int N = 100100;

int n , x , a[N] , ans = 0;

int main() {
    //freopen("input.in" , "r" , stdin);

    cin >> n;
    for (int i = 1 ; i <= n ; i++) {
        scanf("%d" , &x);
        for (int j = 2 ; j * j <= x ; j++){
            while (x % j == 0) {
                x /= j;
            	a[i]++;
            }
        }
        if (x>1) a[i]++;
        ans ^= a[i];
    }

    if (ans == 0) cout << "BOB" ; else cout << "ALICE" << endl;
}
