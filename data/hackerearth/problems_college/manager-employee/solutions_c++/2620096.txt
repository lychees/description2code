#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n,t;
    scanf("%d",&t);
    while(t--){
        scanf("%d",&n);
        int a[n];
        int ans=0,ans1=INT_MAX;
        for(int i=0;i<n;i++){
            scanf("%d",&a[i]);
        }
            sort(a,a+n);
            for(int i=0;i<n/2;i++){
                a[i]+=a[n-i-1];
                ans=max(ans,a[i]);
                ans1=min(ans1,a[i]);
            }

        printf("%d\n",ans-ans1);
    }
    return 0;
}
