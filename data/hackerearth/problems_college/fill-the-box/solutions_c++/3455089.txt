#include <iostream>
using namespace std;

int main()
{
    long n,m,i,a,b,k,j,ans=0;
    cin>>n>>m;
    long s[n+1];
    for(i=1;i<=n;i++)
    {
    	s[i]=0;
    }
    for(i=0;i<m;i++)
    {
    	cin>>a>>b>>k;
    	for(j=a;j<=b;j++)
    	{
    		s[j]+=k;
    	}
    }
    for(i=1;i<=n;i++)
    {
    	ans+=s[i];
    }
    cout<<(ans/n)<<endl;
    return 0;
}
