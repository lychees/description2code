#include<bits/stdc++.h>
using namespace std;
string s1,s2;
int main()
{int n,count1[26],count2[26],x,y;
    cin>>n;
    while(n--)
    {cin.ignore();
		getline(cin, s1);
		getline(cin, s2);
    for(int i=0;i<26;i++)
    {count1[i]=0;
    count2[i]=0;
    }
    x=0;y=0;
    for(int i=0;i<s1.size();i++)
    count1[s1[i]-'a']++;
    for(int i=0;i<s2.size();i++)
    count2[s2[i]-'a']++;
    for(int i=0;i<26;i++)
    {if(count1[i]>count2[i])
      x++;
      if(count2[i]>count1[i])
      y++;
    }
    if(x>0&&y>0)
    cout<<"You draw some."<<endl;
    if(x>0&&y==0)
    cout<<"You win some."<<endl;
    if(y>0&&x==0)
    cout<<"You lose some."<<endl;
    }

      
    return 0;
}
