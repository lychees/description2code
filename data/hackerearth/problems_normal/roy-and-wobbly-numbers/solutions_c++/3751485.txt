#include <iostream>
using namespace std;

int* pData;
int Size;
int K;

void Init()
{
	cin>>Size;
	cin>>K;
	pData=new int[Size];
	for(int i=0; i<Size; i++)
	{
		if(i%2==0)
		{
			pData[i]=1;
		}
		else
		{
			pData[i]=0;
		}
	}
}

void PrintData()
{
	for(int i=0; i<Size; i++)
	{
		// cout<<pData[i]<<" ";
		cout<<pData[i];
	}
	cout<<endl;
}

bool Increase()
{
	pData[1]++;
	if(pData[1]==pData[0])
	{
		pData[1]++;
	}
	
	if(pData[1]>=10)
	{
		pData[1]-=10;
		pData[0]++;
	}
	
	if(pData[0]>=10)
	{
		return false;
	}
	
	for(int i=2; i<Size; i++)
	{
		if(i%2==0)
		{
			pData[i]=pData[0];
		}
		else
		{
			pData[i]=pData[1];
		}
	}
	
	return true;
}

int DoWork()
{
	while(K>1)
	{
		if(!Increase())
		{
			return -1;
		}
		K--;
	}
	
	return 0;
}

void ReadInput()
{
	int Times;
	cin>>Times;
	while(Times>0)
	{
		Init();
		// PrintData();
		int result=DoWork();
		if(result<0)
		{
			cout<<result<<endl;
		}
		else
		{
			PrintData();
		}
		delete[] pData;
		Times--;
	}
}

int main()
{
    // cout << "Hello World!" << endl;
    ReadInput();
    return 0;
}
