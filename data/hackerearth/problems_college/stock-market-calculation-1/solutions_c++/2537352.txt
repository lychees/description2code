#include<iostream>
#include<algorithm>
#include<stdio.h>
#include<stdlib.h>
using namespace std; 
int main()
{
	long int i, t, n;
	long a[50000];
	long max=0, pi=0, nout=0, out=0, noutcost=0;
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(i=0; i<n; i++)
			scanf("%ld", &a[i]);
		nout=0;
		out=0;
		max=0;
		pi=0;
		while(max<n)
		{
		noutcost=0;
			pi=max;
			for(i=max+1; i<n; i++)
			{
				if(a[max]<a[i])
				{
					max=i;
				}
			}
			for(i=pi; i<max; i++)
				noutcost+=a[i];
			nout=max-pi;
			out+=((a[max]*nout)-noutcost);
			max++;
		}
		cout<<out<<endl;
	}
    return 0;
}
