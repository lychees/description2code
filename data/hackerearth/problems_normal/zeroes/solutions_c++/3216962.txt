#include<bits/stdc++.h>
#define iter(i,a,b) for(int i=a;i<b;i++)
#define pb push_back
#define si(x) scanf("%d",&x)
#define pi(x) printf("%d\n",x);
#define maxe 2147483647
using namespace std;
vector< int > vec;
int stp[100001];
void seive()
{
	iter(i,2,100001)
	{
		if(!stp[i])
		{
			vec.pb(i);
			for(int j=1;i*j<=100000;j++)
				stp[i*j]=i;
		}
	}
}
int main()
{
	seive();
	int n,x,y;
	si(n);
	int cnt[100001]={0};
	iter(i,0,n)
	{
		si(x);
		while(x>1)
		{
			y=stp[x];
			cnt[y]++;
			x=x/y;
		}
	}
	int q,mini,cn;
	si(q);
	iter(i,0,q)
	{
		si(x);
		mini=maxe;
		while(x>1)
		{
			cn=0;
			y=stp[x];
			while(stp[x]==y)
			{
				x/=stp[x];
				cn++;
			}
			mini=min(mini,cnt[y]/cn);
		}
		pi(mini);
	}
	return 0;
}
