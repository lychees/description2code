#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    long long P[7],m;
    long long t,n,i,count,sum;
    cin >> t;
    while(t--)
    {
    	cin >> n >> m;
    	for(i = 0;i < n;i++)
    	  cin >> P[i];
    	sort(P,P + n);
    	count = sum = 0;
    	for( i = n-1;i >= 0;i--)
    	{
    	   sum += P[i];
    	   count ++;
    	   if(sum >= m)
    	     break;
    	}
    	if(sum < m)
    	  cout << -1 << '\n';
    	else  
    	  cout << count << '\n';	  
    }
    return 0;
}
