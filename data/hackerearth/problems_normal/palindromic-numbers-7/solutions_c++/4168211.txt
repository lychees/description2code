#include<iostream>
using namespace std;
bool is_palindrome(int a)
{
    int o = a;
    int r = 0;
    while(a > 0)
    {
        int remainder = a % 10;
        r = 10 * r + remainder;
        a = a/10;
    }
    if(o == r) return true;
    else{return false;}
}
int main()
{
    int t;
    cin >> t;
    while(t--){
    int a,b;
    cin >> a >> b;
    int c= 0;
    for(int i = a; i <= b; i++)
    {
        if(is_palindrome(i)) c++;
    }
    cout << c << endl;
    }
    return 0;
}
