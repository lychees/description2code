#include <iostream>
#include <cstring>
using namespace std;

void populate_hash(char str[],int g_count[])
{
	int l_count[26]={0};
	int i=0;
	while(*(str+i))
	{
		int temp=*(str+i)-'a';
		if(l_count[temp]==0)
		{
			g_count[temp]+=1;
			l_count[temp]=1;
		}
		i++;
	}
}

char get_min(int g_count[])
{
	int max_index=0;
	for(int i=1;i<26;i++)
	{
		if(g_count[max_index]<g_count[i])
		{
			max_index=i;
		}
	}
	return max_index+'a';
}

int main() {
	int tc=0;
	cin>>tc;
	while(tc--)
	{
		int n=0;
		cin>>n;
		int g_count[26]={0};
		for(int i=0;i<n;i++)
		{
			char str[105];
			cin>>str;
			populate_hash(str,g_count);
		}
		cout<<get_min(g_count)<<endl;
	}
	return 0;
}
