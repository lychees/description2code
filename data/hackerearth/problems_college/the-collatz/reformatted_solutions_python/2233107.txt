t = int(raw_input())

for i in xrange(t):
    count = 0
    even = 0
    odd = 0
    n = input()
    while n > 1:
        if (n % 2 == 0):
            even += n
            n = n/2
            count = count+1
        else:
            odd += n
            n = 3*n+1
            count = count+1
    if (even % count < (odd+1) % count):
        print "Odd Rules"
    elif (even % count > (odd+1) % count):
        print "Even Rules"
    else:
        print "Tie"
