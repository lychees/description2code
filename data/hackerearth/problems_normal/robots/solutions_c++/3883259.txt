#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

bool is_small(int i, int j){
	return i < j;
}

int main()
{
    int t,i,n,j,temp,flag;
    long long ways;
    cin>>t;
    for(i=0;i<t;i++){
    	cin>>n;
    	std::vector<int> max(n);
    	for(j=0;j<n;j++){
    		cin>>max.at(j);
    	}
    	std::sort(max.begin(),max.end(),is_small);
    	ways=1;
    	flag=1;
    	for(j=0;j<n;j++){
    		temp=max.at(j)-j;
    		if(temp<= 0){
    			flag=0;
    			//cout<<temp<<endl;
    			break;
    		}
    		ways = ways * temp;
    		ways= ways % 1000000007;
    	}
    	ways= ways % 1000000007;
    	if(flag==0){
    		cout<<"0"<<endl;
    	} else {
    		cout<<ways<<endl;
    	}
    }
    return 0;
}
