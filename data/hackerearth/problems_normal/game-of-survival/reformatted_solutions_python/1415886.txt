tc = int(input())
flag = 0
while tc > 0:
    n = int(input())
    sharp = [0]*n
    strength = [0]*n
    for i in range(0, n):
        sharp[i] = int(input())
    for i in range(0, n):
        strength[i] = int(input())
    sharp.sort()
    strength.sort()
    flag = 0
    for i in range(0, n):
        if (sharp[i] < strength[i]):
            print "NO"
            flag = 1
            break
    if flag == 0:
        print "YES"
    tc = tc-1
