#include <iostream>
#include <algorithm>
#include <vector>
#include <deque>
using namespace std;

int main()
{
	int n;
	long long int value,sum=0;
    vector <long long int> numbers,answer;
    
    
    scanf("%d",&n); // Input n
    
    for(int i=0;i<n;i++)
    {	
    	scanf("%lld ",&value); //Input value of each element
    	numbers.push_back(value);
    }
    
    sort(numbers.begin(),numbers.end()); //Sort the elements
    
    if(n/2) //Check if total number of elements is odd or even
    {
	    for(int i=0;i<n/2;i++)
	    {
	    	answer.push_back(numbers[i]); //Assign firt element to i=0
	    	answer.push_back(numbers[n-1-i]); //Assign last element to i=1
 	    }
	    answer.push_back(numbers[n/2]);
    }
    else
    {
    	for(int i=0;i<=n/2;i++)
	    {
	    	answer.push_back(numbers[i]); 
	    	answer.push_back(numbers[n-1-i]);
	    }
    }
    
    for(int i=0;i<n-1;i++)
    	sum += abs(answer[i]-answer[i+1]);
    	
	sum += abs(answer[n-1]-answer[0]);
	
	printf("%lld\n",sum);
}
