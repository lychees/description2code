#include <iostream>
#include <unordered_set>

using namespace std;

int main()
{
    int t;
    long long int m,n,i,x,y;

    cin>>t;
    while(t--)
    {   unordered_set <long long int> s;
        cin>>n;
        cin>>m;
        for(i=0;i<n;i++)
        {
            cin>>x;
            s.insert(x);
        }
        while(m--)
        {
            cin>>y;
            if(s.find(y)!=s.end())
                cout<<"YES"<<endl;
            else cout<<"NO"<<endl;
            s.insert(y);
        }
    }
    return 0;
}
