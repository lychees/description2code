#include <iostream>
using namespace std;

int main()
{
    int size;
    cin>>size;
    long int p[size],ptr[size],sum;
    for(int i=0;i<size;i++){
    	sum=0;
    	cin>>p[i];
    	for(int j=0;j<i;j++){
    	sum=sum+ptr[j];	
		}
	ptr[i]=p[i]*(i+1)-sum;
    	
    }
    for(int k=0;k<size;k++)
    cout<<ptr[k]<<" ";
    return 0;
}
