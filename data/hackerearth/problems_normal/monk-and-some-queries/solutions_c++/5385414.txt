#include <iostream>
#include<bits/stdc++.h>
#include <vector>
using namespace std;
vector <int> n;
void posi(int qwe)
{
	
	int con =0;
	for(int aa=0;aa!=n.size()&&!con;aa++)
	{
		if(n[aa]==qwe)
		{
		con =1;
		n.erase(n.begin()+aa);
		
		}
		
	
	}
	if(!con)
	{
		cout<<-1<<endl;
	}
}

int main()
{
    
    int q;
    cin>>q;
    int temp,a;
    while(q--)
    {
    	cin>>temp;
    	if(temp==1)
    	{
    		cin>>a;
    		n.push_back(a);
    	}
    	else if(temp ==2)
    	{
    		cin>>a;
    		posi(a);
    	}
    	else if(temp==4)
    	{
    		if(!n.empty())
    		{
    	int min = *min_element(n.begin(),n.end());
    	cout<<min<<endl;
    		}
    		else
    		cout<<-1<<endl;
    	}
    	else if(temp ==3)
    	{
    		if(!n.empty())
    		{
    		int max = *max_element(n.begin(),n.end());
    		cout<<max<<endl;
    		}
    		else
    		cout<<-1<<endl;
    	}
    }
    return 0;
}
