#include <iostream>
#include<map>
using namespace std;


int main()
{
	int t,v,n,i,j,c;
	cin>>t;
	for(v=0;v<t;v++)
	{
		cin>>n;
		c=0;
		map<int,int>m;
		for(i=0;i<n;i++)
		{
			cin>>j;
			j=j%10;
			if(m[j]>0)
			{
				c++;
				m[j]++;
			}
			else
			m[j]++;
		}
		cout<<c<<"\n";
	}
   // cout << "Hello World!" << endl;
    return 0;
}
