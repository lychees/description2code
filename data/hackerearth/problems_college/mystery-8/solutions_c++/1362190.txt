#include<iostream>
#include<string.h>
using namespace std;


int rev(string m)
    {
    
    string x="";
    int l1=m.length();
    for(int i=l1-1;i>=0;i--)
    {
    	x=x+m.at(i);
    }
    
    //cout<<x<<" "<<m<<endl;
    
    if(x==m)
    return 0;
    else
    return 1;
}
