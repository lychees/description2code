#include <iostream>
using namespace std;
int fun(int a[][102],int i,int j,int n)
{
	int x,y;
	if(a[i][j]==0)
		return 0;
	if(i==n && j==n)
		return 1;
	//if(a[i][j+1])
		x=fun(a,i,j+1,n);
	//if(a[i+1][j])
		y=fun(a,i+1,j,n);
	return x||y;
}
int main()
{
	int t,i,n,j,a[102][102];
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(i=0;i<n;i++)
			for(j=0;j<n;j++)
				cin>>a[i][j];
		if(a[0][0]==0)
		{
			cout<<"NOT POSSIBLE\n";
			continue;
		}
		if(fun(a,0,0,n-1))
			cout<<"POSSIBLE\n";
		else cout<<"NOT POSSIBLE\n";
	}
    //cout << "Hello World!" << endl;
    return 0;
}
