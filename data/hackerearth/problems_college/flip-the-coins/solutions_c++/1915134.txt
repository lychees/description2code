#include<bits/stdc++.h>
using namespace std;
int a[1000]; // Tail '0'
int main()
{
	int i,n,q,x,y,z;
	cin>>n>>q;
	cin>>x>>y>>z;
	for(i=y;i<=z;i++)
		a[i]^=1;
	q--;
	while(q--)
	{
		int ans=0;
		cin>>x>>y>>z;
		for(i=y;i<=z;i++)
		{
			if(a[i])
				ans++;
		}
		cout<<ans<<endl;
	}
	return 0;
}
