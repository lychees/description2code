#include <bits/stdc++.h>
using namespace std;
int main() {
   char board[3005][2005];
   int i, r, c, j;
   scanf("%d %d", &r, &c);
   board[r-1][0] = '.';
if(r == 1 && c == 1) 
   printf("not exists\n");
else {
   for(i = r-1 ; i >= 0 ; i--) {
      if(i == r-1)
         for(j = 1 ; j < c ; j++) {
            if(board[i][j-1] == '.')
               board[i][j] = '#';
            else
               board[i][j] = '.';     
         }
      else    
         for(j = 0 ; j < c ; j++) {
            if(board[i+1][j] == '#')
               board[i][j] = '.';
            else
               board[i][j] = '#';     
         }
         board[i][j] = '\0';            
   }   
   for(i = 0 ; i < r ; i++) {
      puts(board[i]);
}
   } 

   return 0;
}
