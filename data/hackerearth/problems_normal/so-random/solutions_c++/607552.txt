#include <cstdio>
#include <iostream>
#include <cstring>
#include <cstdlib>
#include <cmath>

using namespace std;

int main()
{
	int t;
	float ans,p,T,q;
	scanf("%d",&t);
	while (t --) {
		scanf("%f %f",&p,&T);
		q = 1-p;
		T = T/1440.0;
		ans = pow(q,T);
		ans = 1-ans;
		//ans = p*T/1440.0;
		//ans = ans + ans/5.83;
		printf("%.4f\n",ans);
	}
	
	return 0;
}
