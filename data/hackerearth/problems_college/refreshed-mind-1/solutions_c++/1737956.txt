#include<bits/stdc++.h>
#define ll long long

#define len(s) strlen(s)

#define nl printf("\n");
using namespace std;
ll a[110];
ll cul[110];
int main()
{

    int n,i;
    ll ans=0;
    cul[0]=0;
    scanf("%d",&n);

    for(i=1;i<=n;i++)
    {
        scanf("%lld",&a[i]);
        cul[i]=1+(a[i]-1)*i;
        ans=ans+cul[i];
    }
    printf("%lld",ans);
	return 0;
}
