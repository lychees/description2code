#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,m;
    cin>>n>>m;
    int a[n],c[n];
    for(int i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    for(int i=0;i<n;i++)
    {
    	cin>>c[i];
    }
    sort(a,a+n);
    sort(c,c+n);
    int j=0,k=0,sum=0,count=0;
    while(sum<=m)
    {
    	if(a[j]<c[k])
    	{
    		sum+=a[j];
    		j++;
    	}
    	else
    	{
    		sum+=c[k];
    		k++;
    	}
    	count++;
    }
    count--;
    if(count>n)
    	cout<<n<<endl;
    else
    	cout<<count<<endl;
    return 0;
}
