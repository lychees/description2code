#include <iostream>
#define ll long long
using namespace std;

int main()
{
	ll t, m, n, i, j, c, ans;
    cin>>t;
    while(t--)
    {
    	m=0;
    	cin>>n;
    	while(n!=0)
    	{
    		c++;
    		n=n>>1;
    	}
    	ans=1;m=2;c--;
    	while(c!=0)
    	{
    		ans+=m;
    		m*=2;
    		c--;
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
