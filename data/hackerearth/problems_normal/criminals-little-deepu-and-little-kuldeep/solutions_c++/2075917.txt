#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	long long int t,n,i;
	cin>>t;
	while(t--)
	{
		cin>>n;
		long long int a[n],pre=1,flag=1;
		for(i=0;i<n;i++)
		{
			cin>>a[i];
		}
		sort(a, a+n);
		for(i=0;i<n-1;i++)
		{
				if(a[i]==a[i+1])
                {
                	flag++;
                }
                else
                {
                	 if (flag > pre)
        pre = flag;
        flag=1;
                }
		}
		  	 if (flag > pre)
        pre = flag;
   
		cout<<pre<<endl;
	}
    return 0;
}
