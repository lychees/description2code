ll inverseArray(ll n, ll m) {
    vector<ll> modInverse(n + 1,0);
    modInverse[1] = 1;
    for(ll i = 2; i <= n; i++) {
        modInverse[i] = (-(m/i) * modInverse[m % i]) % m + m;
    }
    ll i=modInverse.size()-1;
    return modInverse[i];
}

void EE(int a, int b, int& x, int& y)
{
    if(a%b == 0)
    {
        x=0;
        y=1;
        return;
    }
    EE(b,a%b,x,y);
    int temp = x;
    x = y;
    y = temp - y*(a/b);
}

int inverse(int a, int m)
{
    int x,y;
    EE(a,m,x,y);
    if(x<0) x += m;
    return x;
}

ll gcd(ll a, ll b )
{
  	ll c;
  	while ( a != 0 ) 
  	{
    	c = a;
    	a = b%a;  
    	b = c;
  	}
  	return b;
}
