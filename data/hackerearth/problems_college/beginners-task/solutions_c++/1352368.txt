#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	int sam;
    	cin>>sam;
    	int count=0;
    	for(int j=0;j<sam;j++)
    	{
    		int k;
    		cin>>k;
    		k=k&1;
    		if(k==0)
    		count++;
    	}
    	cout<<count<<endl;
    }
    return 0;
}
