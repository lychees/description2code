#include <iostream>
#include <math.h>
using namespace std;
int dist(int x1, int y1, int x2, int y2) {
	return pow((x2-x1), 2) + pow((y2-y1), 2);
}
int main()
{
	int t;
	cin>>t;
	int r, xc, yc;
	int xr, yr;
	int flag = 0;
	while(t--) {
		cin>>r>>xc>>yc;
		flag = 0;
		for(int i=0;i<4;i++) {
			cin>>xr>>yr;
			//cout<<r<<" "<<xc<<" "<<yc<<" "<<xr<<" "<<yr<<endl;
			//cout<<dist(xc, yc, xr, yr)<<endl;
			if(dist(xc, yc, xr, yr) > r*r) flag = 1;
		}
		if(flag == 0) cout<<"Yes\n";
		else cout<<"No\n";
	}
    return 0;
}
