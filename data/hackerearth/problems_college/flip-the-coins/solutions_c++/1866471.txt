#include<bits/stdc++.h>

using namespace std;

int main(){
	
	int arr[105]={0},i,type,queries,total;
	
	cin>>total>>queries;
	
	int x,y;
	
	while(queries--){
		
		cin>>type>>x>>y;
		
		if(type==0){
			
			for(i=x;i<=y;i++)
			  arr[i]=!arr[i];
					
		}
		
		if(type==1){
			
			
			int count=0;
			for(i=x;i<=y;i++)
			  if(arr[i]==1) count++;
			
			cout<<count<<"\n";
			
		}
		
		
	}
	
	
	
	return 0;
	
}
