#include<iostream>

using namespace std;
int main()
{

int n;
cin>>n;
int a[n];
for(int i=0;i<n;i++)
{
    cin>>a[i];
}

long int cntr = 0;
for(int i=0;i<n;i++)
{

  for(int j=i+2;j<n;j++)
{

   if(a[i]!=a[j])
    continue;

   for(int k=i+1;k<j;k++)
   {

   for(int l=j+1;l<n;l++)
   {
   if(a[k]==a[l])
   {
       if(a[k]!=a[i])
     cntr++;

    // cout<<i<<" "<<j<<" "<<k<<" "<<l<<endl;
   }


   }


   }




}


}

cout<<cntr%1000000007;


}
