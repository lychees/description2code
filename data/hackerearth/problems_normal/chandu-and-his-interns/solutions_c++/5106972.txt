vector<bool>is_prime;
void sieve(int N)  // O(N * log(log(N)))
{
    is_prime.resize(N+1,1);
    is_prime[1]=1;
    int i,j;
    for(i=2;i*i<=N;++i)
        if(is_prime[i])
            for(j=2*i;j<=N;j+=i)
                is_prime[j]=0;

}

   }
 }
