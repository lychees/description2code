#!/usr/bin/py
from __future__ import print_function
x = []
y = []
for i in xrange(0, 3):
    a = raw_input()
    x.append(float(a.split()[0]))
    y.append(float(a.split()[1]))
sumx = x[0] + x[1] + x[2]
sumy = y[0] + y[1] + y[2]
a = []
for i in xrange(0, 3):
    a.append((sumx - 2*x[i], sumy-2*y[i]))
a.sort()
for i in a:
    print("%.4f %.4f" % (i[0], i[1]))
