#include <iostream>
using namespace std;

int main()
{
    int t,cnt,n;
    cin>>t;
    while(t--)
    {	cnt=0;
    	cin>>n;
    	do{
    		if(n%3==0 && n!=1)
    		n=n/3;

    		else if(n%2==0 && n!=1)
    		
    			n=n/2;
    		else
    		n=n-1;
    		cnt++;
    	}while(n!=1);
    	cout<<cnt<<endl;
    }
    return 0;
}
