#include <iostream>
#include <math.h>
using namespace std;
unsigned long long int gcd(unsigned long long int a,unsigned long long int b)
{
	if(b!=0)
		return gcd(b,a%b);
	else
		return a;
}
int a[100001];
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    unsigned long long int x,y,k,i,val,m=0;
    
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%llu %llu %llu",&x,&y,&k);
    	val=gcd(x,y);
    	a[0]=1;
    	m=1;
    	
    	for(i=2;i<=val;i++)
    	{
    		if(val%i==0)
    		{
    			a[m++]=i;
    		}
    	}
    	
    	
    	if(k>m)
    	{
    		printf("No crymeth today\n");
    	}
    	else
    	{
    		printf("%d\n",a[m-k]);
    	}
    }
    return 0;
}
