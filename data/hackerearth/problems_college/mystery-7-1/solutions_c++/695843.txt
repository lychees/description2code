#include <cstdio>
#include <iostream>
#ifndef ONLINE_JUDGE
	#define gc getchar
	#define pc putchar
#else
	#define gc getchar_unlocked
	#define pc putchar_unlocked
#endif

using namespace std;

int main()
{
	unsigned int t;
	char c;
	cin>>t;
	while(t--){
		c=gc();
		while(c<'a'){
			c=gc();
		}
		while(c>='a' && c<='z'){
			if(c>='a' && c<='c'){
				pc('2');
			}
			else if(c>='d' && c<='f'){
				pc('3');
			}
			else if(c>='g' && c<='i'){
				pc('4');
			}
			else if(c>='j' && c<='l'){
				pc('5');
			}
			else if(c>='m' && c<='o'){
				pc('6');
			}
			else if(c>='p' && c<='s'){
				pc('7');
			}
			else if(c>='t' && c<='v'){
				pc('8');
			}
			else if(c>='w' && c<='z'){
				pc('9');
			}
			c=gc();
		}
		pc('\n');
	}
    return 0;
}
