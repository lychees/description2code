#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define mod 1000000007
ll prime[1000002]={0};
ll a[1000002],b[1000002];
vector<ll>x;
void sieve()
{
	int i,j;
	for(i = 2; i <= 1000000; i++) {
		if(!prime[i]) {
			x.push_back(i);
			for(j = 2; j*i <= 1000000; j++) {
				prime[i*j] = 1;
			}
		}
	}
	a[0] = a[1] = 0;
	b[0] = b[1] = 0;
	for(i = 2; i <= 1000000; i++) {
		a[i] = a[i-1] + (prime[i]+1)%2;
		if(!prime[i]) {
			b[i] = (i+b[i-1])%mod; 
		}
		else {
			b[i] = b[i-1];
		}
 	}
}
int main()
{
	ll t,n,f,k,i,s;
	sieve();
	cin>>t;
	while(t--) {
		cin>>n;
		f = 1;
		cout<<a[n]<<" "<<b[n]<<" ";
		for(i = 0; i < x.size() && x[i] <= n; i++) {
			k = x[i];
			s = 0;
			while(k <= n) {
				s += n/k;
				k *= x[i];
			}
			f = (f*(s+1))%mod;
		}
		cout<<f<<endl;
	}
    return 0;
}
