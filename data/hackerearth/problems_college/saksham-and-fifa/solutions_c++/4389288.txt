#include <iostream>
#include <bits/stdc++.h>
using namespace std;
vector <pair<int,int>> v;
int main()
{
    int ans,n,i,mn,t,p;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>t>>p;
    	v.push_back(make_pair(t,p));
    }
    mn=0;
    for(i=0;i<n;i++)
    {
    	if(mn<v[i].first)
    	{
    		mn=v[i].first;
    		ans=v[i].second;
    	}
    }
    if(ans<0)
    {
    	cout<<"Right Corner"<<endl;
    }
    else if(ans>0)
    {
    	cout<<"Left Corner"<<endl;
    }
    else
    {
    	cout<<"Either Corner"<<endl;
    }
    return 0;
}
