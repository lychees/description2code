#include <iostream>
#include<cmath>
typedef unsigned long long int ll;
using namespace std;
bool prime[4000001];
void init(ll n)
{
	for(ll i=0;i<n;i++)
		prime[i]=true;
	for(ll i=2;i<=sqrt(n);i++)
		for(ll j=i;i*j<n;j++)
			if(prime[i*j]==true)
				prime[i*j]=false;
}
int main()
{
    ll n=0;
    ll sum = 0;
    cin>>n;
    init(n);
    for(ll i=2;i<n;i++)
    {
    	if(prime[i])
    		sum+=i;	
    }
    cout<<sum<<endl;
    return 0;
}
