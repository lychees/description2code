#include <iostream>
#include <vector>
using namespace std;

long long int GCD(long long int a,long long int b)
{
   if (b==0) return a;
   return GCD(b,a%b);
}

int main()
{
	int t, n;
	cin >> t;
	while(t--)
	{
		long long int g, buf;
		cin >> n;
		cin>>buf;
		g=buf;
		for(int i=1; i<n; i++)
		{
			cin>>buf;
			g=GCD(g,buf);
		}
		if(g==1)cout<<"YES\n";
		else cout<<"NO\n";
	}
    
    return 0;
}
