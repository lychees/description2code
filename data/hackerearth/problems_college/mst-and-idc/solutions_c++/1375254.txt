#include <iostream>
#include <stdio.h>
#define ll long long
#include <algorithm>
using namespace std;
#include <math.h>
ll a[500010];    
int main()
{
   int t;
   cin >> t;
   while(t--) {
   	int n;
   	cin >> n;
   	int i;
   	for(i = 0; i < n; i++) {
   		cin >> a[i];
   	}
   	sort(a, a+n);
   	ll ans = 0;
   	for(i =1; i < n; i++) {
   		ans += (a[i] - a[i-1]);
   	}
   	ans = ans * 2;
   	cout << ans << endl;
   }
   return 0;
}
