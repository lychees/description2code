int k , n ;
vector <int> graph[10005];
int a[10005];
int subsize[100005];
int vis[100005] ,res;
int dfs(int v){
    vis[v] = 1;
    int siz = 1 ;
    for(int i = 0 ; i < graph[v].size() ; i++){
        int el = graph[v][i];
        if(!vis[el]){
            siz+=dfs(el);
        }
    }
    if(siz >= k ){
        subsize[v] = siz;
        res++;
        siz = 0;
    }
    return siz;
}

        graph[a[i]].pb(i);
    }
    int temp = dfs(1);
    cout << res << endl;    
	return 0;
}
