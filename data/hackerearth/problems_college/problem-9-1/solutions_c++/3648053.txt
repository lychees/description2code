#include <iostream>
using namespace std;

int main()
{
    int N, Q, C[1001];
    
    cin >> N >> Q;
    
    for (int i = 1; i <= N; i++)
    	C[i] = 0;
    	
    for (int i = 1; i <= Q; i++)
    {
    	int X, A, B;
    	cin >> X >> A >> B;
    	if (X == 0)
    	{
    		for (int i = A; i <= B; i++)
    		{
    			C[i] = 1 - C[i];
    		}
    	}
    	else if (X == 1)
    	{
    		int count = 0;
    		for (int i = A; i <= B; i++)
    		{
    			if (C[i] == 1)
    				count++;
    		}
    		
    		cout << count << endl;
    	}
    }
    
    return 0;
}
