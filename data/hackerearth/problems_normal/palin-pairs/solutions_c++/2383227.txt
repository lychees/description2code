#include<bits/stdc++.h>
using namespace std;

string reve(string s)
{
    string temp="";
    for(int i=s.size()-1;i>=0;i--)
        temp+=s[i];
    return temp;
}

int main()
{
    int n;
    cin>>n;
    string s;
    map<string,int> m;

    long long palin_pair=0;
    while(n--)
    {
        cin>>s;
        palin_pair+=m[s];
        m[reve(s)]++;
    }
    cout<<palin_pair<<endl;

}
