#include<iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;
int arr[100005];
int n,q;
int count(int num)
{
	if(num<arr[0])
	return 0;
	if(num>=arr[n-1])
	return n;
	int l,r,m;
	l=0;r=n-1;m=(l+r)/2;
	while(l<=r)
	{
		if(arr[m]<= num && arr[m+1]>num)
			break;
		else if(arr[m] <= num)
			l=m+1;
		else
		r=m-1;
		m=(l+r)/2;
	}
	return m+1;
}
int main()
{
	scanf("%d",&n);
	for(int i=0;i<n;i++)
	{
		scanf("%d",&arr[i]);
	}
	sort(arr,arr+n);
	scanf("%d",&q);
	while(q--)
	{
		int l,r;
		scanf("%d%d",&l,&r);
		printf("%d\n",count(r)-count(l-1));
	}
	return 0;
}
