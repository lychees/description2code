#include <iostream>
#include<stdlib.h>
#include<stdio.h>
using namespace std;
int arr[10]={6,2,5,5,4,5,6,3,7,6};
char s[1002];
int main()
{
	int t;
    cin>>t;
    while(t--)
    {
    	int sum=0,i;
    	scanf("%s",s);
    //	printf("%s",s);
    	for(i=0;s[i]!='\0';++i)
    	{
    		sum+=arr[s[i]-'0'];
    	}
    	cout<<sum<<"\n";
    }
    cin>>t;
    return 0;
}
