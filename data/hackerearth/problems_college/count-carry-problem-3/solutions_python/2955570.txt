for _ in xrange(input()):
	a,b=raw_input().split()
	l1,l2=len(a),len(b)
	if l1>l2:
		a,b=b,a
	count=0
	l=len(b)
	for i,j in enumerate(a[::-1]):
		if int(b[l-1-i])+int(j)>9:
			count+=1
	if count==0:
		print 'No carry operation'
	elif count==1:
		print "1 carry operation"
	else:
		print count,"carry operations"
