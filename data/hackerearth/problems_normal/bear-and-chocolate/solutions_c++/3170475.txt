#include <iostream>
using namespace std;

int main()
{
    int t;
   	cin>>t;
   	while(t--)
   	{
   		char choco[1001][1001],ch;
   		int row[1001]={0},col[1001]={0};
   		
   		int n,sum=0,flag=0,rowsum=0,colsum=0;
   		cin>>n;
   		for(int i=0;i<n;i++)
   		{
   			for(int j=0;j<n;j++)
   			{
   				cin>>ch;
   				choco[i][j]=ch;
   				if(ch=='#')
   				{
   					row[i]++;
   					col[j]++;
   					
   				}
   			}
   			
   		}
   		
   		for(int i=0;i<n;i++)
   		{
   			sum+=row[i];
			   		
   		}
   		
   		
   		for(int i=0;i<n;i++)
   		{
   			rowsum+=row[i];
   			colsum+=col[i];
   			if((sum-rowsum)==rowsum && rowsum<=(sum-rowsum))
   			{
   				flag=1;
   				break;
   			}
   			if((sum-colsum)==colsum && colsum<=(sum-colsum))
   			{
   				flag=1;
   				break;
   			}
   		}
   		if(flag==1)
   		{
   			cout<<"YES"<<endl;
   		}
   		else
   			cout<<"NO"<<endl;
   	}
    return 0;
}
