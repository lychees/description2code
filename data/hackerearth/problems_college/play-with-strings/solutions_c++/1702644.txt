#include<stdio.h>
#include<string.h>
int check(char a[],int len1,char b[],int len2)
    {
	int i,j=0;
	for(i=0;i<len2;++i,++j)
	   {
		if(j==len1)	
			j=0;
		if(a[j]!=b[i])
			return 0;
	   }
	return 1;
    }

int main()
    {
		int len1,len2,t,count,temp;
		char a[100001],b[100001];
		scanf("%d",&t);
		while(t-->0)
		   {
			count=0;
			scanf("%s",a);
			scanf("%s",b);
			len1=strlen(a);
			len2=strlen(b);
			if(len1<=len2)
			  {
				temp=len1;
				while(len1!=0)
				   {
					if(len2%len1==0&&temp%len1==0)
					  {
						if(check(a,len1,a,temp)==1)
						{	
						if(check(a,len1,b,len2)==1)
							++count;
						}
					  }
					--len1;
				   }
			  }
			else
			  {
				temp=len2;
				while(len2!=0)
				   {
					
					if(len1%len2==0&&temp%len2==0)
					  {
						if(check(b,len2,b,temp)==1)
						{
						if(check(b,len2,a,len1)==1)
							++count;}
					  }
					--len2;
				   }
			  }
			printf("%d\n",count);
		   }
    }
