#include<bits/stdc++.h>
main()
{
	int t,i,k,n,c;
	scanf("%d",&t);
	while(t--)
	{
		c=0;
		scanf("%d",&n);
		long a[n];
		for(i=0;i<n;i++)
		scanf("%ld",&a[i]);
		for(i=0;i<n;i++)
		{
			for(k=i+1;k<n;k++)
			{
				if(a[i]!=a[k]&&(a[i]+a[k])%2==0)
				c++;
			}
		}
		printf("%d\n",c);
	}
}
