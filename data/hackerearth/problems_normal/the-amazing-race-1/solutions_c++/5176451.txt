#include <bits/stdc++.h>
#define M 1000000007
using namespace std;
 
int dir_left[100005];
int dir_right[100005];
int A[100005];
 
inline void fi(int *a)
{
 register char c=0;
 while (c<33) c=getchar_unlocked();
 *a=0;
 int tmp = 0;
 while (c>33)
 {
     if ( c == 45 ) tmp = 1;
     else *a=*a*10+c-'0';
     c=getchar_unlocked();
 }
 if ( tmp == 1 ) *a = 0-(*a);
}
 
int main()
{
	int t,n;
	fi(&t);
	while ( t-- ) {
			fi(&n);
		for ( int i = 1; i <= n; i++ ) fi(&A[i]), dir_left[i] = i-1, dir_right[i] = i+1;
		for ( int i = n-1; i >= 1; i-- ) {
			while ( dir_right[i] <= n && A[i] > A[dir_right[i]] ) dir_right[i] = dir_right[dir_right[i]];
		}
		for ( int i = 2; i <= n; i++ ) {
			while ( dir_left[i] >= 1 && A[i] > A[dir_left[i]] ) dir_left[i] = dir_left[dir_left[i]];
		}
		long long val,mx,idx;
		for ( int i = 1; i <= n; i++ ) {
			val = (long long)(dir_right[i]-dir_left[i]-1);
			val = (val*(long long)i)%M;
			if ( i == 1 ) {
				mx = val;
				idx = i;
			}
			else {
				if ( val > mx ) {
					mx = val;
					idx = i;
				}
			}
		}
		printf("%d\n", idx);
	}
	return 0;
}
