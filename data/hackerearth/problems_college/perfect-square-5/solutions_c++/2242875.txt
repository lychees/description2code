#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		long long int n,k,z,s;
		cin>>n;
		k = sqrt(n);
		//cout<<k<<endl;
		if(k*k==n)
		cout<<"YES"<<endl;
		
		else {
			z = (k+1)*(k+1);
			s = (k)*(k);
			
			if(abs(s-n)<abs(z-n)){
				cout<<s - n<<endl;
			}
			else cout<<"+"<<z - n<<endl;
		}
	}
	return 0;
}
