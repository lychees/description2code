#include<bits/stdc++.h>
using namespace std;

#define ll long long
#define mod 1000000007
#define INF 1000000000
#define pb push_back
#define ff first
#define ss second
#define pii pair<ll,ll> 
#define mp make_pair
#define set(x,y) memset(x,y,sizeof(x))

vector <ll> v[10004];
ll dp[10004],i,n,x,y,vis[10004];

ll func(ll x)
{
	if(vis[x])
	{
		return dp[x];
	}
	vis[x]=1;
	ll cnt=0;
	for(ll i=0;i<v[x].size();i++)
	{
		cnt+=func(v[x][i]);
	}
	return dp[x]=cnt;
}

int main()
{
	ios_base::sync_with_stdio(false);cin.tie(0);
	cin>>n;
	cin>>x>>y;
	while(x!=0 && y!=0)
	{
		v[y].pb(x);
		cin>>x>>y;
	}
	set(dp,-1);
	dp[1]=0;
	set(vis,0);
	vis[1]=1;
	dp[1]=1;
	func(n);
//	for(i=1;i<=n;i++)
//		cout<<dp[i]<<" ";
	cout<<dp[n]<<endl;
	return 0;
}
