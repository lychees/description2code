#include <iostream>
#include <stdlib.h>
#include <cmath>
using namespace std;
typedef long long int ll;
int main() {
	ll n;
	cin >> n;
	ll a[n];
	ll i, p, q, j;
	ll ans[n];
	for(i = 0; i < n; i++){
		cin >> a[i];
		ans[i] = 0;
	}
	ll max = 1;
	ans[0] = 1;
	for(i = 1; i < n; i++){
		ll t = 0; 
		p = a[i];
		for(j = 0; j < i; j++){
			q = a[j];
			//cout << "yoo"<< p << "%%" << q << endl;
			if(p*q < 0 && (llabs(p) > llabs(q))){
				if(ans[j] == 0){
					t = ans[j] +2;
				}
				else{
					t = ans[j] + 1;
				}
				if( t > ans[i]){
					ans[i] = t;
				}
				//cout << ans[i] << "ay" << endl;
			}
		}
		if(ans[i] > max){
			max = ans[i];
		}
	}
	cout << max << endl;
	return 0;
}
