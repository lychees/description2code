#include <stdio.h>
#include <stdlib.h>
#include <algorithm>
#include <iostream>
#include <math.h>
#include <string.h>
#include <bitset> 
#include <vector>
#include <iomanip>
#define max 10000001
typedef unsigned long ul;
using namespace std;

int main()
{
    int t=0;
    scanf("%d",&t);
    ul n;
    while(t--)
    {
    	cin>>n;
    	ul x = sqrt(2*n);
    	ul a = x-1;
    	ul b = x+1;
    	ul L1=a,L2=b;
		ul last =L1;
    	while(1)
    	{
    		if(L1&1)
    		{
    			if((L1+1)/2*L1<=n)
    			{
    				last = L1;
    				++L1;
    			}
    			else
    			{
    				L1 = last;
					break;
    			}
    		}
    		else
    		{
    			if((L1+1)*L1/2<=n)
    			{
    				last = L1;
					++L1;
    			}
    			else
    			{
    				L1=last;
					break;
    			}
    		}
    	}
    	last = L2;
    	while(1)
    	{
    		if(L2&1)
    		{
    			if((L2+1)/2*L2>=n)
    			{
    				last = L2;
    				--L2;
    			}
    			else
    			{
					L2 = last;
    				break;
    			}
    		}
    		else
    		{
    			if((L2+1)*L2/2>=n)
    			{
    				last = L2;
    				--L2;
					//break;
    			}
    			else
    			{
					L2 = last;
					break;
    			}
    		}
    		
    	}
    	//cout<<L1<<endl;
		//cout<<L2<<endl;
		x=0;
		if(L1&1)
		{
			ul y = L1*(L1+1)/2;
			if(y==n)
				x=L1;
			else if(y<n)
			{
				x=L2;
			}
				
		}
		else
		{
			ul y = L1/2*(L1+1);
			if(y==n)
				x=L1;
			else if(y<n)
			{
				x=L2;
			}
				
		}
		//cout<<x<<endl;
    	if(x%3==0)
    		printf("SL\n");
    	if(x%3==1)
    		printf("LB\n");
    	if(x%3==2)
    		printf("BS\n");
    }
}
