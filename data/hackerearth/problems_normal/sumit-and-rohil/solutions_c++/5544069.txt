#include <iostream>
#include <map>
#include <vector>
#define c cin
#define o cout
using namespace std;

bool isAnagram(string s1, string s2){
	bool check = true;
	map <char, int> freq1;
	map <char, int> freq2;
	
	if (s1.length() != s2.length()){
		check = false;
	} else {
		for(int i = 0; i < s1.length(); i++){
			freq1[s1[i]]++;
			freq2[s2[i]]++;
		}
	}
	
	for(map<char,int>::const_iterator it = freq1.begin(), it2 = freq2.begin(); it != freq1.end() || it2 != freq2.end(); ++it, ++it2){
		if(it -> first != it2 -> first || it -> second != it2 -> second){
			check = false;
			break;
		}
	}
	
	
	return check;
}
int main(){
	int t;
	c>>t;
	vector <vector<string> > v;
	while(t--){
		string word;
		c>>word;
		vector <string> buffer;
		
		if(v.empty()){
			buffer.push_back(word);
			v.push_back(buffer);
		} else {
			bool foundMatch = false;
			for(int i = 0; i < v.size(); i++){
				string buf = v[i][0];
				if(buf[0] == word[0] && buf[buf.length()-1] == word[word.length()-1] && isAnagram(buf, word)){
					v[i].push_back(word);
					foundMatch = true;
					break;
				}
			}
			
			if(!foundMatch){
				buffer.push_back(word);
				v.push_back(buffer);
			}
		}
	}
	
	o << v.size();
	
	return 0;
}
