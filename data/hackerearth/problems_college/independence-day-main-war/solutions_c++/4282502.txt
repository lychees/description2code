#include <bits/stdc++.h>

using namespace std;

int main()
{
        int i,j,t;

        long double x,y,a,b,n;

        //freopen("input.txt","r",stdin);

        cin>>t;

        while(t--)
        {
            cin>>n;
            a=b=0;
            for(i=0;i<n;i++)
            {cin>>x>>y;
            a=(a+x);
            b=(b+y);
            }
            a=a/n;
            b=b/n;

            cout<<setprecision(2)<<fixed<<"("<<a<<" , "<<b<<")"<<endl;
        }
}
