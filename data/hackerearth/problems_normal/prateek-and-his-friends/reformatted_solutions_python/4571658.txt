import string
t = int(raw_input())
for i in xrange(0, t):
    n, x = map(int, raw_input().split())
    friend_list = []
    flag = False
    for j in xrange(0, n):
        num = int(raw_input())
        friend_list.append(num)
        if (sum(friend_list) == x and (not flag)):
            flag = True
        else:
            while ((sum(friend_list) >= x)):
                if (sum(friend_list) == x and (not flag)):
                    flag = True
                else:
                    friend_list.pop(0)
    if flag:
        print "YES"
    else:
        print "NO"
