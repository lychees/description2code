#include <iostream>
#include <string>
using namespace std;

int main()
{
    int testCases;
    cin >> testCases;
    testCases++;
    while(testCases) {
    	string str;
    	cin >> str;
    	
    	int count[26] = {0};
    	for(int i = 0; i < str.length(); i++) {
    		count[str[i]] = 1;
    	}
    	
    	for(int i = 0; i < str.length(); i++) {
    		if (count[str[i]]) {
    			cout << str[i];
    			count[str[i]] -= 1;
    		}
    	}
    	cout << endl;
    	
    	testCases--;
    }
    return 0;
}
