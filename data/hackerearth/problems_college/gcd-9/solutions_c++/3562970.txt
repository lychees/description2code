#include <bits/stdc++.h>
using namespace std;

int a[1110111];
int main()
{	
	ios_base::sync_with_stdio(0);
	cin.tie (0);	cout.tie (0);
	int n , k;	cin >> n >> k;
	for (int i = 1;i <= n;i ++)		cin >> a[i];
	
	int len = 0;
	for (int i = 1;i <= n;i ++) {
		int gcd = 0;
		for (int j = i;j <= n;j ++){
			gcd = __gcd (a[j] , gcd);
			if (gcd < k)	break;
			len = max (len , j - i + 1);
		}
	}
	cout << len;
    return 0;
}
