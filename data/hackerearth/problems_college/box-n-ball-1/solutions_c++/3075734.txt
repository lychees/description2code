#include<bits/stdc++.h>
using namespace std;
int main(){
    long long int T,L,B,H;
    double m;
    scanf("%lld",&T);
    while(T--){
        scanf("%lld%lld%lld",&L,&B,&H);
        m=L;
        if(m>B)
            m=B;
        if(m>H)
            m=H;
        m=m/(2.0);
        printf("%0.1f\n",m);
    }
    return 0;
}
