#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int cell[20][20];
int mark[20][20];
ll count_ways=0;
ll n;
void check(int i, int j)
{
    if(i==n-1 && j == n-1)
    {
        count_ways++;
        return;
    }
    mark[i][j]=1;
    if(i-1>=0 && mark[i-1][j]==0 && cell[i-1][j]==0) check(i-1,j);
    if(j+1<n && mark[i][j+1]==0 && cell[i][j+1]==0) check(i,j+1);
    if(i+1<n && mark[i+1][j]==0 && cell[i+1][j]==0) check(i+1,j);
    if(j-1 >=0 && mark[i][j-1]==0 && cell[i][j-1]==0) check(i,j-1);
    mark[i][j]=0;
    return ;
}

int main ()
{
    ll t;
    cin>>t;
    for(int i=1; i<=t; i++)
    {
        cin>>n;
        for(ll j=0; j<n ; j++)
        {
            for(ll k=0; k<n ; k++) cin>>cell[j][k];
        }
        count_ways=0;
        check(0,0);
        cout<<count_ways<<endl;;
    }
    return 0;
}
