#include<bits/stdc++.h>
using namespace std;

unsigned long long int n,s;

unsigned long long int calc(unsigned long long int x )
{
    return (x*(x+1))/2;
}
int main()
{
    
        cin>>n;
        
        s=3*calc((n-1)/3) + 5*calc((n-1)/5) - 15*calc((n-1)/15);
        cout<<s<<endl;
    
    return 0;
    
}
