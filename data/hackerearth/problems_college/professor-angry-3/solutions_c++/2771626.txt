#include <iostream>
using namespace std;

int main()
{
    int t,n,k,i;
    int *a;
    cin>>t;
    while(t--){
    	cin>>n>>k;
    	a = (int *)malloc(sizeof(int)*n);
    	for(i=0;i<n;i++)
    		cin>>a[i];
    	int c = 0;
    	for(i=0;i<n;i++){
    		if(a[i]<=0){
    			c++;
    		}
    	}
    	if(c>=k){
    		cout<<"NO"<<endl;
    	}else{
    		cout<<"YES"<<endl;
    	}
    }
    return 0;
}
