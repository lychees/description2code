#include<bits/stdc++.h>
// MY MACROS
#define pb(n) push_back(n)
#define mp(a,b) make_pair(a,b)
#define M(arr) memset(arr,0,sizeof(arr))
#define MOD 1000000007
using namespace std;
struct matches
{
	string winner, loser;
};
bool compare(matches one,matches two){
	return one.loser<two.loser;
}
int main()
{
	int n;
    cin>>n;
    string first,second;
    set<string>answer;
    answer.clear();
	long long int temp=pow(2,n)+1;
    matches arr[temp];
    for(int i=0;i<pow(2,n)-1;i++)
	{
		cin>>first;
		cin>>second;
		arr[i].winner=first;
		arr[i].loser=second;
        answer.insert(first);
        answer.insert(second);
	}
	long int size=pow(2,n)-1;
	sort(arr,arr+size,compare);
	for(int i=0;i<pow(2,n)-1;i++)
	{
		//cout<<"i="<<i<<endl;;
		answer.erase(answer.find(arr[i].loser));
	}
	set<string>:: iterator itr;
    itr=answer.begin();
    cout<< *itr << endl;

  return 0;
}
