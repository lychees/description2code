#include <iostream>
using namespace std;

int main()
{
	typedef long long int ll;
	ll t;
	cin>>t;
	while(t!=0){
		t-=1;
		string st;
		int flag=1;
		cin>>st;
		int h[256]={0};
		for(int i=0;i<st.length();i++){
			if(h[st[i]]==0)h[st[i]]++;
			else{
				h[st[i]]=0;
			}
		}
		for(int i=0;i<256;i++){
			if(h[i]==1){flag=0;
			break;
			}
		}
		if(flag==1)cout<<"1"<<endl;
		else if(flag==0)cout<<"-1"<<endl;
		
	}
    return 0;
}
