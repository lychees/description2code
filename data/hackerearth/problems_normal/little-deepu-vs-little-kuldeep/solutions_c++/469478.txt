#include<iostream>
using namespace std;
#define inf 1000000007

long long int pow(long long int a,int n)
{
     if(n==0)
        return(1);
     long long int x=pow(a,n/2);
     x=(x*x)%inf;
     if(n%2==1)
        x=(x*a)%inf;
     return(x);      
}

long long int fact[2000001];
long long int ifact[2000001];
    
int main()
{
   int i,n;
   fact[0]=1;
   for(i=1;i<=2000000;i++)
   {
         fact[i]=(fact[i-1]*i)%inf;
   }      
   ifact[0]=1;
   ifact[1]=1;
   for(i=2;i<=2000000;i++)
        ifact[i]=pow(fact[i],1000000005);
   int t;
   cin>>t;
   while(t--)
   {
       cin>>n;
       long long int num=fact[2*n];
       long long int den=(ifact[n]*ifact[n+1])%inf;
       long long int sol=(den*num)%inf;
       cout<<sol<<"\n";
   }      
   return(0);
}
