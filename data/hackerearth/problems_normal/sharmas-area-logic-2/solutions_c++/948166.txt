#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,i;
    	int x,y;
    	int xmn=10000,xmx=-200,ymn=10000,ymx=-200;
    	cin>>n;
    	for(i=0;i<n;i++)
    	{
    		cin>>x>>y;
    		if(x<xmn)
    		xmn=x;
    		if(x>xmx)
    		xmx=x;
    		if(y<ymn)
    		ymn=y;
    		if(y>ymx)
    		ymx=y;
    	}
    	int ar=(ymx-ymn)*(xmx-xmn);
    	if(ar==0)
    	ar=(ymx-ymn)+(xmx-xmn);
    	if(ar<0)
    	ar=-ar;
    	cout<<ar<<"\n";
    }
    return 0;
}
