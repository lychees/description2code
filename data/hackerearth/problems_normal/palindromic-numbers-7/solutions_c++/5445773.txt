int main()
{
	int t; cin >> t;
	while(t--)
	{
		int a, b; cin >> a >> b;
		int count = 0;
		for(int i=a; i<=b;i++)
		{
			int d = i;
			int n = 0;
			while(d > 0)
			{
				int r = d%10;
				n = n*10+r;
				d = d/10;
			}
			if(i == n) count++;
		}
		cout << count << endl;
	}
}
