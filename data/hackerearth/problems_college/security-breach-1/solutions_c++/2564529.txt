#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,l;
    string it;
    cin>>t;
    while(t--)
    {
    	cin>>it;
    	l=it.length();
    	for(int i=0;i<l;i++)
    	{
    		if(it[i]=='2')
    		cout<<"cde";
    		else if(it[i]=='6')
    		cout<<"b3";
    		else if(it[i]=='7')
    		cout<<"acf";
    		else if(it[i]=='9')
    		cout<<"c6a";
    		
    	}
    	cout<<"\n";
    	
    }
    return 0;
}
