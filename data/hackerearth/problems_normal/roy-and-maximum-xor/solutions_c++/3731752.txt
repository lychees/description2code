#include <iostream>
using namespace std;

long powerdo(int a)
{
	long result=1;
	while(a--)
		result=result<<1;
	return result;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long n;
    	cin>>n;
    	if(n==1)
    		cout<<"0 1"<<endl;
    	else
    	{
    		n--;
    		int power=0;
    		while(n>powerdo(power))
    		{
    			n-=powerdo(power);
    			power++;
    		}
    		
    		cout<<(powerdo(power+1)-1)<<" "<<(n*2)<<endl;
    	}
    }
    return 0;
}
