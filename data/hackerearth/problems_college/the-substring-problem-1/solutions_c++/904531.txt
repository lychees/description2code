#include <iostream>
#include <cstring>
using namespace std;

int arr[26];
bool check[26];
int main()
{
    int t,n;
    string word;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(int i=0;i<n;++i)
    	{
    		cin>>word;
    		memset(check,0,sizeof(check));
    		for(int j=0;j<word.length();++j)
    		{
    			if(!check[word[j]-'a'])
    			{
    				arr[word[j]-'a']++;
    				check[word[j]-'a']=true;
    			}
    		}
    	}
    	int ans=0,max=arr[0];
    	for(int i=1;i<26;++i)
    	{
    		if(arr[i]>max)
    		{
    			max=arr[i];
    			ans=i;
    		}
    	}
    	cout<<char(ans+'a')<<endl;
    	memset(arr,0,sizeof(arr));
    }
    return 0;
}
