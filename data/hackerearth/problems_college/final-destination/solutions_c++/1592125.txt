#include<iostream>
#include<queue>
 
// i'm never gonna give up
 
using namespace std;
 
 
int main()
{
    int t,start,target,mul,div,add,sub,steps,temp,ans,n;
	cin>>t;
	while(t--)
	{
		
        int bfs[1000009]={0};
		queue <int> que;        
		cin>>start>>target;
		cin>>mul>>div>>add>>sub;        
		que.push(start);
		bfs[start]=1;
		ans=0;
		
		while(!que.empty()){
			
			//cout<<que.front()<<endl;
				
			n=que.front();
			que.pop();
				
			if(n==target){
				ans=bfs[n];
				break;
			}	
			
			temp=n*mul;
			if(temp>=2 && temp<=1000000 && !bfs[temp]){
					que.push(temp);
					bfs[temp]=bfs[n]+1;
										
			}
				
			temp=n/div;
			if(temp>=2 && temp<=1000000 && !bfs[temp]){
					que.push(temp);
					bfs[temp]=bfs[n]+1;
										
			}
				
			temp=n+add;
		    if(temp>=2 && temp<=1000000 && !bfs[temp]){
					que.push(temp);
					bfs[temp]=bfs[n]+1;
										
			}
				
			temp=n-sub;
			if(temp>=2 && temp<=1000000 && !bfs[temp]){
					que.push(temp);
					bfs[temp]=bfs[n]+1;
										
			}
			
		}
		cout<<ans-1<<endl;
	}
	return 0;
}
