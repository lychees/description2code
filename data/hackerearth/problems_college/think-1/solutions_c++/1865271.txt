#include <bits/stdc++.h>

using namespace std;

int reverse(int n)
{
	int x = 0;
	while (n) {
		x = x*10 + (n%10);
		n /= 10;
	}
	return x;
}

int main()
{
	int t, n, tmp;
	cin >> t;
	while (t--) {
		cin >> n;
		tmp = n*n;
		//cout << tmp << " ";
		n = reverse(n);
		//cout << n*n <<  " " << reverse(n*n) << endl;
		if (tmp == reverse(n*n) && tmp != n*n)
			cout << n << endl;
		else
			cout << "No" << endl;
	}
	return 0;
}
