#include <cstdio>
#include <iostream>

using namespace std;
typedef unsigned long long int int32;
int main() {
	int T;
	int	tc = 0, k, i;
	int32 l, p, c, tmp;
	cin >> T;
	while(T--) 
	{
		cin >> l >> p >> c;
		k = i = 0; tmp = l;
		
		while(tmp < p) {
			tmp = tmp * c;
			k++;
		}
		while((1 << i) < k)
		{	
			i++;
		}
		printf("Case #%d: %d\n", ++tc, i);
	}
	return 0;
}
