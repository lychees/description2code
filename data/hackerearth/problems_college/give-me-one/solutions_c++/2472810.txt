#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int a[20000002];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int test;
	cin>>test;
	ll int i,x,y,z;
	a[1]=0;
	for(i=2;i<20000002;i++)
	{
		if((i%2)==0)
		x=a[i/2];
		else
		x=INT_MAX;
		if((i%3)==0)
		y=a[i/3];
		else
		y=INT_MAX;
		z=a[i-1];
		a[i]=min(min(x,y),z)+1;
	}
	while(test--)
	{
		ll int n;
		cin>>n;
		cout<<a[n]<<endl;
	}
	return 0;
}
