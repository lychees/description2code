(N, K) = map(int, str(raw_input()).split())
nums = sorted(map(int, str(raw_input()).split()))

count = 0
for i in range(N-2):
    j = i+1
    k = N-1
    while True:
        while nums[i] * nums[j] * nums[k] > K:
            k -= 1
            if k == j:
                break
        count += k-j
        j += 1
        if j >= k:
            break

print count
