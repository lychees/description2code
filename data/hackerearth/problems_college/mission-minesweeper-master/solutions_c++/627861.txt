#include<bits/stdc++.h>
using namespace std;
int x,y;
 vector<vector<char> > a(1001,vector<char>(1001));
void recur(int pi,int pj)
 {int count=0;
 	 	if(pi<0 || pi>=x || pj<0 || pj>=y || a[pi][pj]=='*')
 	         	  return;
 	         	int r,c;
 	for(r=pi-1;r<=pi+1;r++)
 	      {for(c=pj-1;c<=pj+1;c++)
 	     			 {if(r<0 || r>=x || c<0 || c>=y)
 	        	 					 continue;
 	     	 			if(a[r][c]=='*')
 	         	 			 count++;
 	        		 }
 	       }
 	     //  cout<<""<<count<<endl<<endl;
 	  a[pi][pj]=(char)(count+48);
 	  if(count==0)
 	   {for(r=pi-1;r<=pi+1;r++)
 	      {for(c=pj-1;c<=pj+1;c++)
 	         if(r==pi && c==pj) continue;
 	         else if(r<0 || r>=x || c<0 || c>=y) continue;
 	         else if(a[r][c]=='.') recur(r,c);
 	     }
 	   } 
 	
 }

int main()
 {
 	
 	int t;
 	cin>>t;
 	while(t--)
 	  {
	   int pi,pj,i,j;
	   cin>>x>>y;
 	   
 	  
 	   for(i=0;i<x;i++)
 	     {for(j=0;j<y;j++)
 	        {
 	        	cin>>a[i][j];
 	        	if(a[i][j]=='c')
 	        	   {pi=i;
 	        	    pj=j;
 	        	   }
 	        }
 	     }
 	  
 	  recur(pi,pj);
 	  for(i=0;i<x;i++)
 	     {for(j=0;j<y;j++)
 	        {
 	        	cout<<a[i][j];
 	        	
 	        }
 	        cout<<endl;
 	     }
 }
}
