int main(void)
{
 
    int n,m;
    int num;
    
    std::cin>>n>>m;
    
    std::vector<std::vector<int> > mat(n,std::vector<int>(m,0));
    std::vector<long> rowSum(n,0);
    std::vector<long> colSum(m,0);
    
    for (int i=0; i<n; ++i)
    {
        long sum = 0;
        for (int j=0; j<m; ++j)
        {
            std::cin>>num;
            sum += num;
            colSum[j] += num;
        }
        rowSum[i] = sum;
    }
    
    long maxXor = 0;
    
    for (int i=0; i<rowSum.size(); ++i)
    {
        for (int j=0; j<colSum.size(); ++j)
        {
            long Xor = rowSum[i] ^ colSum[j];
            
            if (Xor > maxXor)
            {
                maxXor = Xor;
            }
        }
    }
    
    std::cout<<maxXor<<std::endl;
    
    return 0;
}
