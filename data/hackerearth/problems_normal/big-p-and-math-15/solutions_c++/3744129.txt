#include <iostream>
#include <cassert>
#include <math.h>
using namespace std;

int main()
{
	int a,b,mina=0,minb=0,maxa=0,maxb=0,i = 1,temp,count=-1;
	cin >> a;
	assert(1<=a & a<=1000000);
	cin >> b;
	assert(1<=b & b<=1000000);
	
	while(a>0)
	{
		temp = a%10;
		count++;
		if(temp ==6 || temp ==5)
		{
			
				mina = mina + (5 * pow(10,count));
				maxa = maxa + (6 * pow(10,count));
		}
		else
		{
			mina = mina +(temp * pow(10,count));
			maxa = maxa + (temp * pow(10,count));
		}
		a/=10;
	}
	
	count = -1;
	while(b>0)
	{
		temp = b%10;
		count++;
		if(temp ==6 || temp == 5)
		{
			minb = minb + (5 * pow(10,count));
			maxb = maxb + (6* pow(10,count));
		}
		else
		{
			minb = minb + (temp * pow(10,count));
			maxb = maxb + (temp * pow(10,count));
		}
		b/=10;
	}
	
	cout << (mina+minb) << " " << (maxa + maxb);
	return 0;
}
