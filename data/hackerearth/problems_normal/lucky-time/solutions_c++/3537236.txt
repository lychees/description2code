#include <bits/stdc++.h>

using namespace std;

bool vis [10 + 1];
bool sm = false;

bool rep(int num){

    if(num < 10){

        if(vis[0]) return true;
        vis[0] = true;
        if(sm) return true;
        sm = true;
    }

    if(num == 0) return true;

    while(num != 0){

        int digit = num % 10;
        if(vis[digit]) return true;
        num /= 10;
        vis[digit] = true;
    }

    return false;
}
