#include<bits/stdc++.h>
using namespace std;

int  main(){
	int t;
	cin>>t;
	while(t--){
		int n,k;
		cin>>n>>k;
		vector<long long> arr;
		for(int i=0;i<n;i++){
			long long x;
			cin>>x;
			arr.push_back(x);
		}
		
		sort(arr.rbegin(),arr.rend());
		if(n<k)
		k=n;
		long long int sum=0;
		for(int i=0;i<k;i++){
			sum=sum+arr[i];
		}
		cout<<sum<<"\n";
	}
}
