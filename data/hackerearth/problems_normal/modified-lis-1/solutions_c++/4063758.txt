#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define md 1000000007

pair<int,int>ps[1000010],ng[1000010];

void update(pair<int,int> *tr,int t,pair<int,int> v)
{
    while(t<=1e5)
    {
        if(tr[t].first<v.first){
            tr[t].first=v.first;
            tr[t].second=v.second;
        }
        else if(tr[t].first==v.first){
            tr[t].second+=v.second;
            tr[t].second%=md;
        }
        t+=(t&(-t));
    }
}
pair<int,int> query(pair<int,int> *tr,int t)
{
    pair<int,int> ret;
    ret.first=0;
    ret.second=0;
    while(t){
        if(tr[t].first>ret.first){
            ret.first=tr[t].first;
            ret.second=tr[t].second;
        }
        else if(tr[t].first==ret.first){
            ret.second+=tr[t].second;
            ret.second%=md;
        }
        t-=(t&(-t));
    }
    return ret;
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int n,a,sn;
    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>a;
        sn=0;
        if(a<0) sn=1;
        if(sn) a=-a;
        pair<int,int> t=query(sn?ps:ng,a-1);
        if(t.second==0)
            ++t.second;
        t.first++;
        update(sn?ng:ps,a,t);
//        cout<<"OK"<<endl;
    }
    pair<int,int> Ans1=query(ng,1e5);
    pair<int,int> Ans2=query(ps,1e5);
    if(Ans1.first==Ans2.first) cout<<Ans1.first<<" "<<Ans1.second+Ans2.second<<endl;
    else if(Ans1.first>Ans2.first) cout<<Ans1.first<<" "<<Ans1.second<<endl;
    else cout<<Ans2.first<<" "<<Ans2.second<<endl;
    return 0;
}
