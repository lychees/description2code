#include <iostream>
#include <math.h>
using namespace std;

int main()
{
    int t,h,m,i,H,M,angle;
    cin>>t;
    for(i=0;i<t;i++)
    {
    	cin>>h>>m;
    	if(h==12)
    		h=0;
    	H=30*h+0.5*m;
    	M=6*m;
    	angle=abs(M-H);
    	if(angle<=360-angle)
    		cout<<angle<<"\n";
    	else
    		cout<<360-angle<<"\n";
    }
    return 0;
}
