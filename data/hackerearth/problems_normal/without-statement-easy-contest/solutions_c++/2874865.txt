#include <bits/stdc++.h>
using namespace std;
int arr[730][30];

int compute_digsquares(int n)
{
	int last, sum;
	sum=0;
	while(n)
	{
		last=n%10;
		sum+=last*last;
		n=n/10;
	}

	return sum;
}

void precompute()
{
	int a, i, j;

	// number obtained on 1 operation
	for(i=1; i<=729; i++)
		arr[i][0]=compute_digsquares(i);

	for(j=1; j<30; j++)
	{
		for(i=1; i<=729; i++)
			arr[i][j]=arr[arr[i][j-1]][j-1];
	}
	return ;
}

int pow_of_two( int n, int m)
{
	int k, val;
//	sum=0;
	k=0;
	val=n;
	while(m)
	{
		if(m&1)
		{
			val=arr[val][k];
		}
		k++;
		m=m>>1;
	}
	return val;
}

int main()
{
	int t, n, m, k, i, j, val, ans;
	int arr[730][30];
	precompute();
	cin>>t;
	while(t--)
	{
		cin>>n>>m;
		val=compute_digsquares(n);
		m--;
		ans=pow_of_two(val, m);
		cout<<ans<<endl;
	}
	return 0;
}
