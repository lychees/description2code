#include <iostream>
using namespace std;
long long max = 1000001;
long long arr[1000001];
void pre()
{
	arr[0]=0;
	arr[1]=0;
	long long i;
	long long no_2=0,no_5=0;
	for(i=2;i<1000001;i++)
    {
   		long long j = i;
   		//long long no_2=0,no_5=0;
   		while(j%2==0)
   		{
   			no_2++;
    		j=j/2;
   		}
   		while(j%5==0)
   		{
   			no_5++;
   			j=j/5;
    	}
    	arr[i]=arr[i-1]+((no_2<no_5)?no_2:no_5);
    }
}
int main()
{
    
    long long t;
    cin>>t;
    pre();
    while(t--)
    {
    	long long a,b;
    	cin>>a>>b;
    
    	cout<<arr[b]-arr[a-1]<<endl;
    	
    }
    return 0;
}
