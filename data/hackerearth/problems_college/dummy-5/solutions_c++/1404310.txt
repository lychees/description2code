#include<bits/stdc++.h>
using namespace std;

const int N=21;

int a[21][3];

bool dp[21][1010][3];

bool solve(int n,int protein,int vitamin,int cholestrol) {
    if(protein==0&&vitamin==0&&cholestrol==0) {
        return true;
    }
    if(n==0&&(protein!=0||vitamin!=0||cholestrol!=0)) {
        return false;
    }
    if(a[n-1][0]>protein||a[n-1][1]>vitamin||a[n-1][2]>cholestrol) {
        return solve(n-1,protein,vitamin,cholestrol);
    }
    return solve(n-1,protein,vitamin,cholestrol)||
        solve(n-1,protein-a[n-1][0],vitamin-a[n-1][1],cholestrol-a[n-1][2]);
}
int main() {
    int tt;
    scanf("%d",&tt);
    int cases=1;
    while(tt--) {
        int protein,vitamin,cholestrol;
        scanf("%d%d%d",&protein,&vitamin,&cholestrol);
        int n;
        scanf("%d",&n);
        for(int i=0;i<n;i++) {
            for(int j=0;j<3;j++) {
                scanf("%d",&a[i][j]);
            }
        }
        if(solve(n,protein,vitamin,cholestrol)==true) {
            printf("Case #%d: yes\012",cases);
            goto label;
        }
        printf("Case #%d: no\012",cases);
        label: cases++;
    }
    return 0;
}
