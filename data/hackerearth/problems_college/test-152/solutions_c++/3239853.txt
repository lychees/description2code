#include<bits/stdc++.h>
int main()
{
	int t,i,max;
	scanf("%d",&t);
	while(t--)
	{
	  int size;
	  scanf("%d",&size);
      char string[size+1];
      scanf("%s",string);
      int c = 0, count[26] = {0};
      while (string[c] != '\0')
      {
      if (string[c] >= 'a' && string[c] <= 'z') 
         count[string[c]-'a']++;
      c++;
      }
      max=0;
   for(i=0;i<26;i++)
   {
   	if(count[i]>max)
   	max=count[i];
   }
   if(max%2==0)
   printf("Yes");
   else
   printf("No");
   printf("\n");
}
return 0;
}
