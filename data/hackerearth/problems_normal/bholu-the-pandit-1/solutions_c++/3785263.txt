#include <iostream>
#include <algorithm>
using namespace std;
typedef long long int ll;
int main()
{
	ll mod = 1000000007;
    int t;
    cin >> t;
    while(t--) {
    	ll a, b, c, ans;
    	cin >> a >> b >> c;
    	ans = a;
    	ans = __gcd(ans, b);
    	ans = __gcd(ans, c);
    	ll ans2;
    	ans2 = ((a/ans) * (b/ans))%mod;
    	ans2 = (ans2 *(c/ans))%mod;
    	cout << ans << " " << ans2 << endl;
    }
    return 0;
}
