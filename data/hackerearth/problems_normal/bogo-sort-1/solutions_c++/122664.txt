#include <iostream>
using namespace std;

int main()
{
    int N,i,j;
    long long A[10000],temp;
    cin>>N;
    for(i=0;i<N;i++)
    {
    	cin>>A[i];
    }
    
    for(i=1;i<N;i++)
    {
    	
    	for(j=0;j<i;j++)
    	{
    		if(A[i] < A[j])
    		{
    			temp=A[i];
    			A[i]=A[j];
    			A[j]=temp;
    		}
    	}
    }
    
    
    for(i=0;i<N;i++)
    {
    	cout<<A[i]<<"\n";
    }
    return 0;
}
