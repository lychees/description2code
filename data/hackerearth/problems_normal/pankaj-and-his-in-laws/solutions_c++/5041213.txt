#include <bits/stdc++.h>
using namespace std;

int main(){
	int n;
	cin>>n;
	int seq[n], lis[n];
	for(int i=0; i<n; i++){
		cin>>seq[i];
		lis[i]=1;
	}
	for(int i=1; i<n; i++){
		for(int j=0; j<i; j++){
			if(seq[i]>seq[j] && lis[i]<(lis[j]+1)){
				lis[i]=lis[j]+1;
			}
		}
	}
	int maxi=0;
	for(int i=0; i<n; i++){
		if(maxi<lis[i]){
			maxi=lis[i];
		}
	}
	string str="";
	int temp=maxi;
	while(temp){
		int rem = temp%2;
		if(rem==1) str+="1";
		else str+="0";
		temp=temp/2;
	}
	for(int i=(str.length()-1); i>=0; i--){
		cout<<str[i];
	}
	cout<<endl;
    return 0;
}
