#include<iostream>
using namespace std;
int main()
{
	long int n,i,j,max=0,a[1000000];
	  cin>>n;
for(i=0;i<n;i++)
cin>>a[i];
for(i=0;i<n;i++)
{for(j=0;j<n;j++)
{ if(i!=j)
  {if(a[i]==a[j])
   {break;
   }}}
if(a[i]!=a[j])
  {if(a[i]>max) 
  {max=a[i];}}
}
if(max==0)
max=-1;
cout<<max<<endl;
return 0;
}
