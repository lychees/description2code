#include<bits/stdc++.h>
#define all(c) c.begin(),c.end()
#define r_all(c) (c).rbegin(),(c).rend()
#define iter(c,i) for(typeof((c).begin()) it=(c).begin();it!=(c).end();it++)
#define pb(c) push_back(c)
using namespace std;
long long pw(long long n)
{
    long long int i,anss=1;
    for(i=1;i<=n;i++)
        anss*=10;
    return anss;
}
int main()
{
    long long tmp,anss,t,n,i,j,cnt;
    cin>>t;
    while(t--)
    {
        cnt=0;
        cin>>n;
        tmp=n;
        while(tmp!=0)
        {
            tmp=tmp/10;
            cnt++;
        }
        anss=((n-pw(cnt-1))+1)*cnt;
        cnt--;
        for(i=1;i<=cnt;i++)
        {
            anss+=i*9*pw(i-1);
        }
        cout<<anss<<endl;
    }
    return 0;
}
