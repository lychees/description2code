#include<bits/stdc++.h>
using namespace std;
long long int gcd(long long int a,long long int b)
{
    if(b==0)
        return a;
    else
        return (gcd(b,a%b));
}

int main()
{
    long long int i,j,k,n,t;
    scanf("%lld",&t);
    while(t--){
    scanf("%lld",&n);
    long long int a[n];
    for(i=0;i<n;i++)
    {
        scanf("%lld",&a[i]);
    }
    long long int g;
    g=a[0];
    for(i=1;i<n;i++)
    {
        g=gcd(g,a[i]);
    }
    printf("%lld\n",g);
    }
    return 0;

}
