#include <bits/stdc++.h>
using namespace std;

int main()
{
	cin.sync_with_stdio(false);
	cout.sync_with_stdio(false);
    long long int t,a,b,ans,i;
    scanf("%lld",&t);
    while(t--)
    {
    scanf("%lld%lld",&a,&b);
    ans=a;
    for(i=a+1;i<=b;i++)
    
    	ans=ans&i;
    	printf("%lld\n",ans);
    
    }  
    return 0;
}
