#include <iostream>
using namespace std;

int main()
{
    int t;
    char ch;
    char str[1001];
    cin>>t;
    while(cin.getline(str,1001,'\n'))
    {
    	bool stat = false ,zero = false;
    	fflush(stdin);
    	for(int i=0;str[i]!='\0';++i)
    	{
    		ch = str[i];	
    		if(ch == '$')
    			{stat = true ,cout<<'$';continue;}
    		if(stat == true)
    		{
				if(ch == ' ' || (ch == '0' && zero == false))
    				continue;
    			if(ch >= '0' && ch <= '9')
    				{cout<<ch;zero=true;continue;}
    			break;
    		}
    	}
    	if(zero == false && stat != false)
    			cout<<'0';
    	cout<<endl;
	}
	return 0;
}
