#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d", &t);
    
    while (t--) {
    	int n;
    	scanf("%d", &n);
    	
    	vector< int > a(n);
    	vector< bool > done(n, false);
    	
    	for (int i = 0; i < n; i++) {
    		scanf("%d", &a[i]);
    	}
    	
    	sort(a.begin(), a.end());
    	
    	int cnt = 0;
    	for (int i = 0; i < n; i++) {
    		if (!done[i]) {
    			cnt++;
    			if (i < n - 1 && a[i + 1] == a[i] + 1) {
    				done[i + 1] = true;
    			}
    			done[i] = true;
    		}
    	}
    	
    	printf("%d\n", cnt);
    }
    return 0;
}
