#include <bits/stdc++.h>
using namespace std;
long long arr[57][100005];

int main()
{
    long long n,q,i,x,y,k,K,j,temp,count;
    vector<long long> v;
    scanf("%lld %lld",&n,&q);
    v.resize(n+1);
    for(i=1;i<=n;i++)
    {
    	scanf("%lld",&v[i]);
    }
    
    memset(arr,0,sizeof(arr));
    count=1;
    for(K=1;K<=10;K++)
    {
    	for(i=1;i<=K;i++)
    	{
    		temp=0;
    		for(j=i;j<=n;j=j+K)
    		{
    			arr[count][j]=temp+v[j];
    			temp=arr[count][j];
    		}
    		count++;
    	}
    }
    /*for(i=1;i<57;i++)
    {

    	{
    		cout<<arr[i][j]<<" ";
    	}cout<<endl;
    }*/
    while(q--)
    {
    	scanf("%lld %lld %lld",&x,&y,&k);
    	long long ind,sum;
    	if(x%k==0)
    		ind=(k*(k-1))/2,ind=ind+k;
    	else
    		ind=(k*(k-1))/2,ind=ind+x%k;
    	while(arr[ind][y]==0)y--;
    	sum=arr[ind][y]-arr[ind][x]+v[x];
    	printf("%lld\n",sum);
    }
    return 0;
}
