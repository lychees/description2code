#include <iostream>
using namespace std;
long long int m=1000000009;
long long int modpowIter(long long int a,long long  int b,long long int c) {
      long long  int ans=1;
        while(b != 0) {
                if(b%2 == 1)
                        ans=(ans*a)%c;

                a=(a*a)%c;
                b /= 2;
        }
        return ans;
}
int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    long long	int n;
    	cin >> n;
    	long long int ans=(modpowIter(10,n,m)-modpowIter(8,n,m));
    	
    	
    	ans=(ans%m*modpowIter(2,m-2,m)%m)%m;
    	while(ans<0)
    	ans+=m;
    	cout << ans << endl;
    	
    }
    return 0;
}
