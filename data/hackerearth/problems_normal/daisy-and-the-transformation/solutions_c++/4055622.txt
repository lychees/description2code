#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--) {
    	char str1[52], str2[52];
    	cin >> str1 >> str2;
    	int n;
    	cin >> n;
    	int count = 0;
    	for(int i = 0; str1[i] != '\0' && str2[i] != '\0'; ++i) {
    		if(str1[i] == 'M' && str2[i] == 'D') {
    			++count;
    		}
    		if(str1[i] == 'D' && str2[i] == 'M') {
    			++count;
    		}
    	}
    	if(count > n) {
    		cout << "No" << endl;
    	} else {
    		if((n - count) % 2 == 0) {
    			cout << "Yes" << endl;
    		} else {
    			cout << "No" << endl;
    		}
    	}
    }
    return 0;
}
