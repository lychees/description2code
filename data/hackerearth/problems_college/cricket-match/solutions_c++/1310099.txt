#include <iostream>
#include <vector>
#include <cstring>
using namespace std;
bool ok = true;
int vis[100001];
int n;
vector<int> g[100001];
bool dfs(int u){
	
	for(int v:g[u]){
		if (vis[v]!=-1 && vis[v]==vis[u]){
			
			return false;
		}
		if (vis[v]!=-1 && vis[v]!=vis[u]) continue;
		vis[v] = vis[u]^1;
		dfs(v);
	}
	return true;
}
int main()
{
    int t,m,a,b;
    scanf("%d",&t);
    while(t--){
    	scanf("%d%d",&n,&m);
    	for(int i=0;i<=n;i++) g[i].clear();
    	for(int i=0;i<m;i++){
    		scanf("%d%d",&a,&b);
    		g[a].push_back(b);
    		g[b].push_back(a);
    	}
    	memset(vis,-1,sizeof(vis));
    	ok = true;
    	for(int i=1;i<=n;i++){
    		if (vis[i]==-1){
    			vis[i] = 0;    	
    			ok &= dfs(i);
    		}
    	}
    	
    	if (ok) cout << "YES\n";
    	else cout << "NO\n";
    }
    
    
    return 0;
}
