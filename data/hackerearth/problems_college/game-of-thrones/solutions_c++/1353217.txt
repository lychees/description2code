#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */   
    long long int test,n,a[100005],i;
    cin>>test;
    while(test--)
    {
    	cin>>n;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	sort(a,a+n);
    	long long int count=0;
    	for(i=0;i<n-1;)
    	{
    		while(a[i]==a[i+1]&&i<n)
    		{
    			i++;
    			count++;
    		}
    		i++;
    	}
    	cout<<count<<endl;
    }
    return 0;
}
