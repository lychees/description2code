#include <bits/stdc++.h>

using namespace std;

#define ll long long
#define ull unsigned long long
#define pb push_back
#define f first
#define PI 3.14159265359
typedef pair<int,int> ii;
typedef pair<int,pair<int,int> > iii;

int A[105];
int dp[105][100000];
vector<int> V;

int main(int argc, const char * argv[])
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int n,i,j,sum,mx,s,k;
    cin>>n;
    for(i=1;i<=n;i++)
        cin>>A[i];
    
    int summax=accumulate(A+1,A+n+1,0);
    
    for(i=1;i<=n;i++)
    {
        sum=0;
        for(j=i;j>=1;j--)
        {
            sum+=A[j];
            for(k=j-1;k>=1;k--)
            {
                dp[i][sum]=max(dp[i][sum],dp[k][sum]);
            }
            dp[i][sum]++;
        }
        
        sum=0;
    }
    
    mx=1;
    
    for(i=1;i<=n;i++)
    {
        for(j=1;j<=summax;j++)
        {
            mx=max(mx,dp[i][j]);
        }
    }
    
    for(i=1;i<=n;i++)
    {
        for(j=1;j<=summax;j++)
        {
            if(dp[i][j]==mx)
            {
                V.pb(j);
            }
        }
    }
    
    sort(V.begin(),V.end());
    V.erase(unique(V.begin(),V.end()),V.end());
    
    cout<<mx<<"\n";
    for(i=0;i<V.size();i++)
        cout<<V[i]<<" ";
    
    
    return 0;
}
