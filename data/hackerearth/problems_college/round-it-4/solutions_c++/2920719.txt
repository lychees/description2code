#include <iostream>
#include<math.h>
using namespace std;
 
int main()
{ int tc;
cin>>tc;
while(tc--)
{
long int n,b,w,q,p=0,s=0;
cin>>n>>b;
q=n;
w=n;
if(n%b==0)
{
    cout<<n<<"\n";
}
else
{while((q%b)!=0)
{
 
    q--;
    p=q;
 
 
}
while((w%b)!=0)
{
    w++;
     s=w;
 
}
 
 
if((s-n)>(n-p))
{
    cout<<p<<"\n";
}
else if((s-n)==(n-p))
{
if(p>s)
    cout<<p<<"\n";
else
    cout<<s<<"\n";
}
else
   {
       cout<<s<<"\n";
   }
}
}
}
