#include <iostream>
using namespace std;

int main()
{
    int n,a[100][100],i,j,s1=0,s2=0;
    cin>>n;
    for(i=0;i<n;i++)
    for(j=0;j<n;j++)
    cin>>a[i][j];
    for(i=0;i<n;i++)
    for(j=0;j<n;j++)
    {
    	if(i==j)
    	s1=s1+a[i][j];
    	if(j==n-i-1)
    	s2=s2+a[i][j];
}
if(s1-s2<0)
cout<<s2-s1;
else
cout<<s1-s2;
}
