typedef long long ll;

typedef vector<double> vd;
typedef pair<int, int> pii;
typedef pair<double, double> pdd;
typedef vector<pii> vii;
typedef vector<string> vs;

int main() {
  int n,k,m;
  cin >> n >> k >> m;
  vvi bad(min(20, k), vi(1e6));
  for (int i = 0; i < m; ++i) {
    int t; cin >> t;
    if (t < bad.back().size()) bad.back()[t] = 1;
  }
  vi v(n, 1);
  for (int i = 0; i < v.size(); ++i) {
    int t = 1;
    if (i) v[i] = v[i-1] + 1;
    while (bad.back()[t]) {
      ++t;
    }
    v[i] = t;
    if (v[0] == 1 || k == bad.size()) {
      bad[0][t] = 1;
      for (int j = 0; j + 1 < bad.size(); ++j) {
        for (int x = bad[j+1].size()/t*t; x >= t; x -= t) if (bad[j][x/t]) {
          bad[j+1][x] = 1;
        }
      }
    }
  }
  for (int i = 0; i < v.size(); ++i) cout << v[i] << ' ';
  return 0;
}
