#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    int x;
    cin>>x;
    sort(a,a+n);
    int sum=0,flag=0;
    for(int i=0;i<n-2;i++)
    {
    	sum=x-a[i];
    	int j=i+1;
    	int k=n-1;
    	while(j<k)
    	{
    		if(sum==a[j]+a[k])
    		{
    			flag=1;
    			break;
    		}
    		else if(sum<a[j]+a[k])
    		{
    			k--;
    		}
    		else
    		{
    			j++;
    		}
    	}
    	if(flag==1)
    	{
    		break;
    	}
    }
    if(flag==1)
    {
    	cout<<"True"<<endl;
    }
    else
    {
    	cout<<"False"<<endl;
    }
}
