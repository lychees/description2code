#include<bits/stdc++.h>

using namespace std;
int a[10005];


bool isprime(long long int x)
{
    if(x%2==0)
        return 0;

    long long i;
    for(i=3;i*i<=x;i+=2)
        if(x%i==0)
        return 0;

    return 1;
}
void pre()
{
    long long int n,ne,nw,se,sw,dne,dnw,dse,dsw,cur,nt,total;

    ne = nw = se = sw =1;
        dne = 2;
        dnw = 4;
        dsw = 6;

        n=3;
        cur = 0;
        a[1] = 0;
        while(n<10000)
        {
           ne += dne;
           nw += dnw;
           sw += dsw;

           dne+=8;
           dnw+=8;
           dsw+=8;

           if(isprime(ne))
            cur++;

           if(isprime(nw))
            cur++;

           if(isprime(sw))
            cur++;
            //cout<<ne<<' '<<nw<<'\t'<<sw<<'\n';
            a[n]=cur;
           n += 2;

        }
}
int main()
{
    int t,n,total;
    cin>>t;
    pre();
    while(t--)
    {
        cin>>n;

        total = (n/2)*4 + 1;
        //cout<<'\n'<<cur<<'\t'<<total<<'\n';
        printf("%lf\n",(1.0 * a[n] / total)*100);
    }
    return 0;
}
