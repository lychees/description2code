#include <iostream>
using namespace std;

int main()
{
    long int t,a,b;
    cin>>t;
    while(t!=0)
    {
    	int c=0;
    	cin>>a>>b;
    	while(a!=0 && b!=0)
    	{
    		if(a%10+b%10>=10)
    		c++;
    		a=a/10;
    		b=b/10;
    	}
    	if(c==0)
    	cout<<"No carry operation\n";
    	else if(c==1)
    	cout<<"1 carry operation\n";
    	else
    	cout<<c<<" carry operations\n";
    	t--;
    }
    return 0;
}
