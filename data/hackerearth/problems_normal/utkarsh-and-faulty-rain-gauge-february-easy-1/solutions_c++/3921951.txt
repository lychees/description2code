#include"bits/stdc++.h"
using namespace std;
#define ll long long int
#define mod 1000000007
#define pb push_back
int main()
{
	ll nn;
	double p=1;
	scanf("%lli",&nn);
	while(nn--)
	{
		ll n,r;
		scanf("%lli%lli",&n,&r);
		double re=((3*n*(n+1))*1.0)/(2*r*(2*n+1));
		p*=re;
		
	}
	printf("%.4lf\n",p);
}
