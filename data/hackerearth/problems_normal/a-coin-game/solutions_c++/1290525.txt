#include<bits/stdc++.h>
using namespace std;
inline int scan(){
    char c = getchar_unlocked();
    int x = 0;
    while(c<'0'||c>'9'){
        c=getchar_unlocked();
    }
    while(c>='0'&&c<='9'){
        x=(x<<1)+(x<<3)+c-'0';
        c=getchar_unlocked();
    }
    return x;
}
set<pair<int,int> > m;
bool done[1000000]={0};
int main(){
    int prev=0;
    m.insert(make_pair(0,0));
    done[0]=1;
    for(int i=1;i<1000001;i++){
        int first;
        for(int j=prev;1;j++){
            if(done[j]==0){
                first=j;
                break;
            }
        }
        prev=first+1;
        int second;
        second = first + i;
        m.insert(make_pair(first,second));
        done[first]=1;
        done[second]=1;
        if(second>1000000){
            break;
        }
    }
    int t=scan();
    while(t--){
        int n=scan(),mm=scan();
        if(n>mm){
            swap(n,mm);
        }
        if(m.find(make_pair(n,mm))!=m.end()){
            cout<<"Don't Play";
        }
        else{
            cout<<"Play";
        }
        cout<<endl;
    }
}
