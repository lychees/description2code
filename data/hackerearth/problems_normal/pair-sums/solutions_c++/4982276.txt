#include <iostream>
#include <cstring>
using namespace std;
int arr[1000005];
int hasht[2000005];

int main()
{
    int n, k;
    bool flag = false;
    cin >> n >> k;
    memset(arr, 0, sizeof arr);
    memset(hasht, 0, sizeof hasht);
    for(int i = 1; i <= n; i++)
    {
    	cin >> arr[i];
    	if(k - arr[i] <= 0)
    		continue;
    	if(hasht[k-arr[i]] >= 1)
    		flag = true;
    	hasht[arr[i]]++;
    }
    
    if(flag == false)
    	cout << "NO" << endl;
    else
    	cout << "YES" << endl;
    
    return 0;
    
}
