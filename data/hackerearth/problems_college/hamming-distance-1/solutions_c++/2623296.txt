#include<bits/stdc++.h>
using namespace std;
int hamming(int a,int b)
{
	int r,temp,n1,n2,maxi,ans,i;
	vector<int> v1,v2;
	temp=a;
	if(temp==0)
		v1.push_back(0);
	else
	{
		while(temp>0)
		{
			r=temp%2;
			v1.push_back(r);
			temp/=2;
		}
	}
	temp=b;
	if(temp==0)
		v2.push_back(0);
	else
	{
		while(temp>0)
		{
			r=temp%2;
			v2.push_back(r);
			temp/=2;
		}
	}
	n1=v1.size();
	n2=v2.size();
	if(n1!=n2)
	{
		if(n1>n2)
		{
			for(r=1;r<=(n1-n2);r++)
				v2.push_back(0);
		}
		else
		{
			for(r=1;r<=(n2-n1);r++)
				v1.push_back(0);
		}
	}
	maxi=max(n1,n2);
	reverse(v1.begin(),v1.end());
	reverse(v2.begin(),v2.end());
	for(i=0;i<maxi;i++)
		if(v1[i]!=v2[i])
			ans++;
		return ans;
}
int main()
{
  int i,t,x,y;
  scanf("%d",&t);
  while(t--)
  {
  	scanf("%d%d",&x,&y);
  	int ans=hamming(x,y);
  	printf("%d\n",ans);
 }
    return 0;
}
