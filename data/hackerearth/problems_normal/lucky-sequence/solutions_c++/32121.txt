#include <iostream>
#include <cstdio>
#include <string>
#include <algorithm>

using namespace std;

int main()
{
long long t,n;
string p = "47";

scanf("%lld", &t);

while ( t-- ) {
	int j = 0;
scanf("%lld", &n);
char s[1125];
n--;
while ( n > 1 ) {
int x = n%2;
s[j++] = p[x];
n = n/2-1;
}
s[j++] = p[n%2];
s[j] ='\0';
//reverse(s.begin(),s.end());
for(int ik = j-1; ik >=0; ik-- ){
	printf("%c",s[ik]);
}
printf("\n");
}
return 0;
}
