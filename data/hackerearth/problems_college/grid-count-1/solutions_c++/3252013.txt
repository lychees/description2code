#include <bits/stdc++.h>
using namespace std;
int a[1005][1005] ;
long long sum[1005][1005] ;
int main()
{
    int n , k;
    cin >> n >> k ;
    if(!k){
        cout << 0 ;
        return 0 ;
    }
    for(int i = 1 ; i <= n ;i ++){
    	for(int j = 1 ; j <= n ; j++){
    		cin >> a[i][j] ;
    	}
    }
    for(int i = 1 ; i <= n ; i++){
    	for(int j = 1 ; j <= n ; j++){
    		sum[i][j] = sum[i][j-1] + sum[i-1][j] - sum[i-1][j-1] + a[i][j] ;
    	}
    }
    k-- ;
    int ans = 0 ;
    for(int i = 1 ;i <= n ; i++){
    	for(int j = 1 ; j <= n ; j++){
    		if(i+k > n || j+k > n)break ;
    		long long num = sum[i+k][j+k] - sum[i-1][j+k] - sum[i+k][j-1] + sum[i-1][j-1] ;
    		long long lo = 1 , hi = 1e6 ;
    		while(lo < hi){
    			long long mid = (lo+hi)/2 ;
    			if(mid*mid*mid<num)lo = mid+1 ;
    			else hi = mid ;
    		}
    		if(lo*lo*lo==num)ans ++ ;
    	}
    }
    cout << ans ;
    return 0;
}
