#include <iostream>
using namespace std;

int main()
{
   long long int t,i,j,k,l,m,n;
   long long int arr[26];
   long long int res;
   scanf("%lld",&t);
   while(t--)
    {
    	scanf("%lld",&n);
    	for(i=0;i<26;i++)
    	scanf("%lld",&arr[i]);
    	m=12*n;
    	k=m/26;
    	res=0;
    	for(i=0;i<26;i++)
    	res+=k*arr[i];
    	res+=13*k*(k-1);
    	l=m%26;
    	for(i=0;i<l;i++)
    	res+=arr[i];
    	if(k>=1)
    	res+=k*l;
    	printf("%lld\n",res);
    }
    return 0;
}
