T = input()


def pow(a, n, m):
    t = 1
    if (n == 0):
        return 1
    t = pow(a, n/2, m)
    if (n % 2 == 0):
        return (t*t) % m
    else:
        return (a*t*t) % m


while (T > 0):
    n = input()
    m = 10**9 + 7
    t1 = pow(3, n, m)
    t2 = (3*pow(2, n, m) - 3)
    print((t1 - t2) % m)
    T = T-1
