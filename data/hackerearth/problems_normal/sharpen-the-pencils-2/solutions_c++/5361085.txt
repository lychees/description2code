#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
#define FOR(i,a,b) for(LL i = (LL )a; i < (LL )b ; i++)
#define mp make_pair
#define pb push_back
typedef vector<long long> V;
typedef pair<int,int> pii;
typedef pair<long long,long long> P;


int main()
{
    LL T,N,K,M,x,y;
    LL i,j,k,it;
    LL mn,mx,temp,init,pos,ans,c1,c2,count;
    int flag;
    LL ce,co;
    V v,s;
    it=0;
    cin>>T;
    while(T--)
    {
        cin>>N;
        for(i=0;i<N;i++)
        {
            cin>>temp;
            v.pb(temp);
            s.pb(temp);
        }
        c1=c2=0;
        for(i=0,j=N-1;i<j;)
        {
            if(v[i]%2==0)
            {
                if(v[i]<2*v[j])
                {
                
                    c1++;
                    v[j]-=v[i]/2;
                    i++;
                    flag=1;
                }
                else if(v[i]==2*v[j])
                {
                	c1++;
                	c2++;
                	i++;
                	j--;
                	flag=2;
                }
                else
                {
                    c2++;
                    v[i]-=2*v[j];
                    j--;
                    flag=0;
                }
            }
            else
            {
                if(v[i]<2*v[j])
                {
                    c1++;
                    v[j]-=v[i]/2;
                    i++;
                    v[i]--;
                    v[j]--;
                    flag=1;
                }
                else if(v[i]==2*v[j])
                {
                	c1++;
                	c2++;
                	i++;
                	j--;
                	flag=2;
                }
                
                else
                {
                    c2++;
                    v[i]-=2*v[j];
                    j--;
                    flag=0;
                    
                }
            }
        }
        if(i==j)
        	if(flag==2)
        		c1++;
        	else if(flag==1)
        		c2++;
        	else
        		c1++;
        		
        cout<<c1<<" "<<c2<<endl;
        v.clear();
        s.clear();
                
    }
    
	return 0;

    
}
