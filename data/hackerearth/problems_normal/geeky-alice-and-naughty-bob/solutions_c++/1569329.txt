#include<iostream>
using namespace std;
int main()
{
     long long t,A,B,i,s=0;
     int a[6]={0,1,2,6,6,3};
     cin>>t;
     while(t--)
     {
        cin>>A>>B;
        if(A>=6)
          s=(B-A+1)*9;
         else if(A<6&&B>=6)
           {
             for(i=A;i<=5;i++)
              s=s+a[i];
              s=s+((B-6+1)*9);
           }
           else if(B<6&&A<6)
           for(i=A;i<=B;i++)
           s+=a[i];
           cout<<s<<"\n";
           s=0;
     }
 return 0;
}
