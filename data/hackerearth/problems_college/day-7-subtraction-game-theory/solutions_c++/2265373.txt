#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define p 1000000007
ll dp[1000010];
ll x,y;

void precomp()
{
	ll i;
	dp[0]=1;
	for(i=1;i<=1000000;i++)
	{
		dp[i]=((dp[i-1])%p*(x%p))%p;
		dp[i]=(dp[i]+((-y)%p))%p;
	}
}
int main()
{
	scanf("%lld%lld",&x,&y);
	precomp();
	ll q,k;
	scanf("%lld",&q);
	while(q--)
	{
		scanf("%lld",&k);
		printf("%lld\n",dp[k]);
	}
	return 0;
}
