#include <bits/stdc++.h>

using namespace std;

int A[102];

int main()
{
	int t,n;
	cin >> t;
	while ( t-- ) {
		cin >> n;
		for ( int i = 0; i < n; i++ ) cin >> A[i];
		sort(A,A+n);
		int idx = 0, ans = 0;
		while ( idx < n ) {
			int cnt = 1;
			if ( idx+1 < n && A[idx] == A[idx+1]-1 ) {
				while ( A[idx] == A[idx+1]-1 && idx+1 < n ) {
					cnt++;
					idx++;
				}
			}
			idx++;
			ans += (cnt/2) + (cnt%2);
		}
		
		cout << ans << endl;
	}
	return 0;
}
