#include <bits/stdc++.h>
using namespace std;

#define MOD 1000000007


long long fast_pow(int a,int n)
{
    long long result = 1;
    long long power = n;
    long long value = a;
    while(power>0)
    {
        if(power&1)
            {result = result*value;
            result = result%1000000007;}
        value = value*value;
        value = value%1000000007;
        power /= 2;
        //power >>= 1;
    }
    return result;
}

long long int fib[1000004]={0};

void Fibo()
{
	long long int a,b,c;
	a=1,b=1;
	fib[1]=1;
	fib[2]=1;
	for(int i=3;i<1000004;i++)
	{
		c=(a+b)%MOD;
		fib[i]=c;
		a=b;
		b=c;
	}
}

int main()
{
	Fibo();
    int T,N;
    cin>>T;
    while(T--)
    {
    	cin>>N;
    	long long int po;
    	//=fast_pow(2,N);
    	po=fib[N+1];
    	while(po<0)po+=MOD;
    	cout<<po<<endl;
    }
    
    return 0;
}
