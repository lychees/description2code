#include<iostream>
using namespace std;

 int sod(long x)
 {
     int sum=0;
    while(x!=0)
    {
        sum=sum+x%10;
        x=x/10;
    }
    return sum;
}


int main()
 {
int n;
long long i[1000000],sum,sum1;
sum1=sum=0;
cin>>n;

for(int j=0;j<n;j++)
{
    cin>>i[j];
}
for(int j=0;j<n;j++)
{
    sum=sum+sod(i[j]);
    sum1=sum1+i[j];

}
sum1=sod(sum1);
sum=sum%9;
sum1=sum1%9;
sum=sum-sum1;

cout<<sum;
return 0;
}
