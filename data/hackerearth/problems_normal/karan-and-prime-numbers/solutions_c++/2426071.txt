#include <iostream>
#include <math.h>
#include <stdio.h>
using namespace std;
int main()
{
unsigned int* table=new unsigned int[32000];
table[0]=2;
table[1]=3;
int current=2;
int flag=1;
int counter=4;
for(long int i=5; ;i+=2)
{
        if(i==5)
                table[current++]=5;
        if(counter==4)
        {
                counter=0;
                continue;
        }
        counter++;
        if(current==32000)
                break;
        int j=1;
        long int rooti=((long int)sqrt(i))+1;
        flag=1;
        while(table[j] <= rooti && j<current)
        {
                if( i % table[j] == 0)
                {

                        flag=0;
                        break;
                }
                j++;
        }
        if(flag==1)
        {
                table[current++]=i;
        }
}
int t;
scanf("%d",&t);
while(t--)
{
int m,n;
scanf("%d %d",&m,&n);
if(m==1)
        m=2;
flag=1;
int sum=0;
for(int i=m;i<=n;i++)
{
        flag=1;
        for(int j=0;j<32000; j++)
        {
                if( table[j] > sqrt(i) )
                        break;
                if( i % table[j]==0)
                {
                        flag=0;
                        break;
                }
        }
        if(flag==1)
                sum=sum+i;
}
cout<<sum<<endl;
}
return 0;
}
