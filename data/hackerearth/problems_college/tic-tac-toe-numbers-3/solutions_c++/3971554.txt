#include <iostream>
#include <math.h>
using namespace std;

int main()
{
    int t;
    long long n,sum,a,b,c,x,ttl,i;
    cin>>t;
    while(t)
    {
    	cin>>n;
    	sum=0;
    	a=0;b=1;c=1;
    	while(1)
    	{
    		c=a+b;
    		a=b;
    		b=c;
    		if(c<n)
    		sum+=c;
    		else
    		break;
    	}
    	ttl=n*(n-1)/2;
    	x=ttl-sum;
    	c=0;
    	for(i=2;i<=sqrt(x);i++)
    	{
    		if(x%i==0)
    		{
    			c=1;
    			break;
    		}
    	}
    	if(c==0)
    	cout<<"Tic"<<endl;
    	else if(x%2==0)
    	cout<<"Tac"<<endl;
    	else
    	cout<<"Toe"<<endl;
    	t--;
    }
    return 0;
}
