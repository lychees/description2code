#include <bits/stdc++.h>

#define ll long long
#define pb push_back
#define mp make_pair

using namespace std;

int main()
{
        ios::sync_with_stdio(false);
        ll t;
        cin>>t;
        vector<string> v;
        for (int i=0;i<t;i++) {
                string s;
                cin>>s;
                v.pb(s);
        }
        sort(v.begin(),v.end());
        vector<pair<string, int> >c;
        ll cnt=1;
        for (int i=1;i<v.size();i++) {
                if (v[i]!=v[i-1]) {

                        c.pb(mp(v[i-1],cnt));
                        cnt=1;
                } else cnt++;
        }
        c.pb(mp(v[v.size()-1],cnt));
        for (int i=0;i<c.size();i++) {
                cout<<c[i].first<<" "<<c[i].second<<"\n";
        }
        return 0;
}
