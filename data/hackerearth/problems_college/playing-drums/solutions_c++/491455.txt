#include<bits/stdc++.h>
using namespace std;
#define maxn 100005
#define F(i,p,n) for(i=p;i<n;i++)
#define I(i,p,q) for(i=p;i>=q;i--)
#define Ss(x) scanf("%s",x)
#define S(x) scanf("%d",&x)
#define P(x) printf("%d\n",x)
#define Ps(x) printf("%d ",x)
#define LL long long int
int dist[maxn];
main()
{
    int i,j,k,m,n,lo,hi,t,c,p,q,mid,element,pos;
    S(t);
    while(t--)
    {
        S(n);
        S(c);
        F(i,0,n)
        S(dist[i]);
        //sort(dist,dist+n);
        lo=0;
        hi=(dist[n-1]-dist[0])/(c-1);
        while(lo<hi)
        {
            mid=(hi+lo)/2;
            if(mid==lo)
                mid++;
           int ini=dist[0];
            k=0;
            F(i,1,c)
            {
                element=mid+ini;
                q=n-1;
                if(element>dist[n-1])
                {
                    k++;
                    break;
                }
                p=0;
                while(p!=q)
                {
                    m=(p+q)/2;
                    if(dist[m]<element)
                        p=m+1;
                    else
                        q=m;
                }
                if(dist[p]<element)
                    p++;
                ini=dist[p];
            }
            if(k)
                hi=mid-1;
            else
                lo=mid;
        }
        P(hi);
    }
}
