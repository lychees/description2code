#include <stdio.h>
#include<iostream>
using namespace std;

long long dp[100005][12];
int a[100005];

void rec(int n, int K)
{
	for(int i=1; i<=n; i++)
	{
		for(int j=1;j<=K;j++)
		{
			if(i-j>0)
			 {
			 	dp[i][j]+= dp[i-j][j];
			 }
			 dp[i][j]+=a[i];
		}
	}
}

int main()
{
      int n ,w,i,j,l,r,k;
    cin>>n>>w;
    
    for(i=1;i<=n;i++)
    {
    	cin>>a[i];
    }
    rec(n,10);
    while(w--)
    {
    	cin>>l>>r>>k;
    	int mod = (r-l)%k;
    	r = r-mod;
    	if(l-k>0)
    	{
    	cout<<dp[r][k]- dp[l-k][k]<<endl;
    	}
    	else
    	{
    	cout<<dp[r][k]<<endl;
    	}
    }
    return 0;
}
