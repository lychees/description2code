#include <iostream>
using namespace std;
struct variable{
	int data;
	struct variable *next;
};
class stack{
	struct variable *head;
	int size=0;
	public:
	void add(int i){
		if(size==0){
			struct variable *temp=new variable;
			temp->data=i;
			temp->next=0;
			head=temp;
			size++;
		}
		else{
			struct variable *temp=new variable;
			temp->data=i;
			temp->next=head;
			head=temp;
			size++;
		}
	}
	int pop(){
		struct variable *temp=new variable;
		temp=head;
		head=head->next;
		size--;
		return temp->data;
	}
	int get(){
		if(size==0) return 0;
		return head->data;
	}
	int get_size(){return size;}
};
int main()
{
    int N;
    cin >> N;
    int a,b,c;
    for(int i=0;i<N;i++){
    	cin >> a >> b;
    	stack stack2;
    	cin >> c;
    	stack2.add(c);
    	while(--a){
    		cin >> c;
    		while(stack2.get()<c && b && stack2.get_size()){
    			stack2.pop();
    			b--;
    		}
    		stack2.add(c);
    	}
    	stack stack1;
    	while(stack2.get_size()){
    		stack1.add(stack2.pop());
    	}
    	while(stack1.get_size()){
    		cout << stack1.pop() << " ";
    	}
    	cout << endl;
    }
    return 0;
}
