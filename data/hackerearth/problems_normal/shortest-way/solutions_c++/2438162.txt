#include <bits/stdc++.h>
#define lp(i,n) for(int i = 0; i < n; i++)
#define ll long long
#define mem(a, b) memset(a, b, sizeof(a))
using namespace std;
int main()
{
    int t;
    long long a, b, c,x;
    cin >> t;
    while(t--)
    {
        c=0;
        cin >> a >> b;
        while(a != 1 && b != 1)
        {
            if(a > b)
            {
                x = (a - b) / b + 1;
                c+= x;
                a -= x * b;
            }
            else
            {
                x = (b - a) / a + 1;
                c+= x;
                b -= x * a;
            }
        }
        c += max(a,b) - min(a,b);
        cout << c << endl;
    }
    return 0;
}
