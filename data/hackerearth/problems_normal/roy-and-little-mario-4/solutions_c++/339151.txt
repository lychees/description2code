#include <cstdio>
#include<iostream>
#define ll long long int
#define M 1000000007
using namespace std;
ll a[100005];

int main()
{
    ll t,n,i;
    cin>>t;
    while(t--)
    {
    	a[0]=0;a[1]=1;a[2]=2;
    	a[3]=4;
    	for(i=4;i<=100000;i++)
    	{
    		a[i]=(a[i-1]%M + a[i-2]%M + a[i-3]%M);
    	}
    	cin>>n;
    	cout<<a[n]%M<<endl;
      }
    return 0;
}
