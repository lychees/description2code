#include <iostream>
using namespace std;

typedef long long LL;

int main()
{
    int N,M,i;
    LL s1=0,s2=0;
    scanf("%d%d",&N,&M);
    int A[N],B[M];
    for(i=0;i<N;i++)
    {
    	scanf("%d",&A[i]);
    	s1+=A[i];
    }
    for(i=0;i<M;i++)
    {
    	scanf("%d",&B[i]);
    	s2+=B[i];
    }
    printf("%lld",s1*s2);
    return 0;
}
