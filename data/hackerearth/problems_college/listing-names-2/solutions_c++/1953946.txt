#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
 	int n,count;
 	cin>>n;
 	string a[100001];
 	for(int i=0;i<n;i++)
	cin>>a[i];
	sort(a,a+n);
	string temp;
	for(int i=0;i<n;)
	{
	count=1;
	temp=a[i];
	cout<<a[i]<<" ";
	i++;
	while(i<n&&a[i]==temp)
	{i++;count++;}
	cout<<count<<"\n";
}
}
