t = input()
while t:
    a, b, n = map(int, raw_input().split())
    l = sorted(map(int, raw_input().split()))
    c1 = c2 = 0
    for i in range(0, n):
        if a >= l[i]:
            a -= l[i]
            c1 += 1
        if b >= l[i]:
            b -= l[i]
            c2 += 1
    if c1 == c2:
        print "Tie"
    elif c1 > c2:
        print "Raghu Won"
    else:
        print "Sayan Won"
    t -= 1
