#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>
#include <map>
#include <queue>
#include <set>
#include <sstream>
#include <stack>
#include <string>
#include <vector>
using namespace std;
int n, x;
long long ans = 0;
int main() {
  int i, j, l1, r1, l, r;
  scanf("%d%d", &n, &x);
  l = r = x;
  while (n--) {
    scanf("%d%d", &l1, &r1);
    if (r < l1) {
      ans += l1 - r;
      l = r, r = l1;
    } else if (l > r1) {
      ans += l - r1;
      r = l, l = r1;
    } else {
      l = max(l, l1);
      r = min(r, r1);
    }
  }
  printf("%I64d\n", ans);
  return 0;
}
