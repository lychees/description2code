# include <stdio.h>
# include <iostream>
using namespace std;
static int a[10002][101],v,n,b,t,m;

int main ()
{
cin>>n;
cin>>b;
a[0][b]=1;
for(int i=1;i<n;i++)
{scanf("%d",&b);
for(int j=1;j<101;j++)
a[i][j]=a[i-1][j];
a[i][b]++;
}

//cout<<a[2][2]<<endl;
cin>>m;
for(int i=0;i<m;i++)
{
scanf("%d%d%d",&b,&t,&v);
//cout<<v;
if(b!=0)n=a[t][v]-a[b-1][v];
else n=a[t][v];
printf("%d\n",n);
}
return 0;}
