#include<bits/stdc++.h>
using namespace std;


int mini(int x,int y,int z)
{
return min(min(x,y),z);
}

int calculatemin(string str1,string str2,int m,int n)
{
int i,j,dp[m+1][n+1];

for(i=0;i<=m;i++)
{
	for(j=0;j<=n;j++)
	{
		if(i==0)
			dp[i][j]=j;
		else if(j==0)
			dp[i][j]=i;
		else if(str1[i-1]==str2[j-1])
			dp[i][j]=dp[i-1][j-1];
		else
			dp[i][j]=1+mini(dp[i][j-1],dp[i-1][j],dp[i-1][j-1]);							
	}
}
return dp[m][n];

}

int main()
{
	int t,n1,n2,m,n,no;
	cin>>t;
	string str1,str2;
	while(t--)
	{
		cin>>n1>>n2;
		cin>>m>>str1;
		cin>>n>>str2;
		no=calculatemin(str1,str2,m,n);
		if(no==0)
			n2+=10*m;
		else
			n1+=5*no;
		cout<<n1<<endl<<n2<<endl;
		if(n1>n2)
			cout<<"Swapnil"<<endl;
		else if(n1<n2)
			cout<<"Friend"<<endl;
		else
			cout<<"Tie"<<endl;
	}
	return 0;
}
