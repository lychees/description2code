#include<iostream>

using namespace std;

int main()
{
    unsigned int t;
    scanf("%u",&t);
    while(t--)
    {
              long r,c,i,cr,cc,limit;
              scanf("%ld %ld",&r,&c);
              long long sum=0;
              if(r<=c)
              {
                      limit = r;
                      }
                      else if(r>c)
                      limit = c;
                      cr = limit + ((c+1)*(limit)*(limit-1))/2;
                      cc = limit + ((r+1)*(limit)*(limit-1))/2;
                      sum = cr+cc;
                      printf("%lld\n",sum);
              }
              return 0;
}
