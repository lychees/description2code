def search(x):
	lft = 0
	rgt = len(arr) - 1
	
	while lft <= rgt:
		mid = (lft + rgt) // 2
		if arr[mid] == x:
			return True
		
		if arr[mid] > x:
			rgt = mid - 1
		else:
			lft = mid + 1
		
	return False
	
N, Q = map(int, raw_input().strip(" \n").split(" "))
arr = map(int, raw_input().strip(" \n").split(" "))
arr.sort()

for i in xrange(Q):
	x = input()
	if search(x):
		print "YES"
	else:
		print "NO"
