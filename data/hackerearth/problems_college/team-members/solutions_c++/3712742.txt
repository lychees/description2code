#include <bits/stdc++.h>
 
using namespace std;
 
#define pb push_back
#define mp make_pair
#define REP(i, n) for (int i = 0; i < (int)(n); ++i)
typedef long long LL;
typedef pair<int, int> PII;
 
LL x;
int n, mod, k;
int a[15];
const LL MX = 1e14;
int dig[10] = {};
map<LL, vector<LL> > ma;
int sh[10];
 
LL solve() {
    LL mask = 0;
    REP(i, 10) mask |= (LL)dig[i] << (i << 2);
    if (ma.count(mask)) return mask;
    vector<LL> ans(mod);
    if (mask == 0) {
        ans[0] = 1;
        ma[0] = ans;
        return 0;
    }
    REP(i, 10) if (dig[i]) {
        --dig[i];
        LL mas = solve();
        auto it = ma.find(mas);
        vector<LL> &v = it->second;
        REP(z, mod) if (v[z]) {
            int j = (10 * z + i) % mod;
            ans[j] += v[z];
        }
        ++dig[i];
    }
    ma[mask] = ans;
    return mask;
}
 
int main() {
    //freopen("input.txt", "r", stdin);
    cin >> x >> mod >> k;
    while (x) {
        ++dig[x % 10];
        x /= 10;
    }
    LL mask = solve();
    auto it = ma.find(mask);
    vector<LL> &v = it->second;
    cout << v[k] << endl;
    return 0;
}
