#include<bits/stdc++.h>
using namespace std;
#define ll long long int
int dp[4*100000];
int arr[100005];
void build(int ind,int ss,int se)
{
	if(ss == se)
	{
		dp[ind] = arr[ss];
		return ;
	}
	int mid = (ss + se) / 2;
	build(2*ind, ss, mid);
	build(2*ind + 1,mid + 1,se);
	dp[ind] = max(dp[2*ind],dp[2*ind + 1]); 
}
void update(int ind,int ss,int se,int i,int v)
{
	if(ss == se)
	{
		dp[ind] = v;
		return ;
	}
	int mid = (ss+se)/2;
	if(mid < i)
	{
		update(2*ind + 1,mid+1,se,i,v);
	}
	else
	{
		update(2*ind,ss,mid,i,v);
	}
	dp[ind] = max(dp[2*ind],dp[2*ind +1]);
}
int query(int ind,int ss,int se,int v)
{
	if(dp[ind] < v)
		return -2;
	if(ss == se)
	{
		return ss;
	}
	int mid = (ss + se)/2;
	if(dp[2*ind] >= v)
		return query(2*ind,ss,mid,v);
	else
		return query(2*ind + 1,mid+1,se,v);
}
int main()
{
	ios_base::sync_with_stdio(false);
	int n,q,i;
	cin>>n>>q;
	for(i=0;i<n;i++)
		cin>>arr[i];
	build(1,0,n-1);
	//for(i=0;i<4*n;i++)
	//	cout<<dp[i]<<" ";
	int t,a,b;
	while(q--)
	{
		cin>>a;
		if(a==1)
		{
			cin>>b;
			cout<<query(1,0,n-1,b) + 1<<"\n";
		}
		else
		{
			cin>>a>>b;
			update(1,0,n-1,a-1,b);
		}
	}
	return 0;
}
