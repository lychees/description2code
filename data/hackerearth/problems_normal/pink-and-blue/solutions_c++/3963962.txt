    #include <iostream>
    #include <vector>
     
    using namespace std;
     
    #define maxN 100000
     
    vector<int> g[maxN+5];
    char gender[maxN+5];
    int color[maxN+5];
    
    int n,m;
     
    void dfs(int u)
    {
    	int i,v;
    	for(i=0;i<(int)g[u].size();i++)
    	{
    		v=g[u][i];
    		if(color[v]==0)
    		{
    			if(color[u]==1)
    				color[v]=2;
    			else
    				color[v]=1;
    			dfs(v);
    		}
    	}
    }
     
    int compute(int s,int p)
    {
    	int i,c=0;
    	for(i=1;i<=n;i++)
    		color[i]=0;
    	color[s]=p;
    	dfs(s);
    	for(i=1;i<=n;i++)
    	{
    		if(gender[i]=='B')
    		{
    			if(color[i]==2)
    				c++;
    		}
    		else
    		{
    			if(color[i]==1)
    				c++;
    		}
    	}
    	return c;
    }
     
    int main()
    {
        int i,u,v;
        cin>>n>>m;
        
        for(i=1;i<=n;i++)
        {
        	cin>>gender[i];
        	//scanf("%c",&gender[i]);
        	//scanf("%c",&gender[i]);
        }
        while(m--)
        {
        	cin>>u>>v;
        	g[u].push_back(v);
        	g[v].push_back(u);
        }
        cout<<min(compute(1,1),compute(1,2));
        return 0;
    }
