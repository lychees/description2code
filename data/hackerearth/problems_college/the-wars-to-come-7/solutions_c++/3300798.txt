#include <iostream>
using namespace std;

int main()
{
    int n, s, tmax, c=0, i,j;
    cin>>n;
    int ar[n];
    for(i=0;i<n;i++){
    	cin>>ar[i];
    }
    cin>>s;
    while(1){
    	tmax=ar[0];
    	for(i=1;i<n;i++){
    		if(tmax<ar[i]){tmax=ar[i];j=i;}
    	}
    	ar[j]=0;
    	if(s<tmax){
    		c++;
    		break;
    	}
    	else{
    		s-=tmax;
    		c++;
    		
    		
    	}
    	
    }
    cout<<c;
    return 0;
}
