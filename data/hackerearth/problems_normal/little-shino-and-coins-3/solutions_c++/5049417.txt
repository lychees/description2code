#include <iostream>
using namespace std;

int main()
{
    int k;
    string str;
    cin>>k;
    cin>>str;
    
    
    
    int len = str.length();
    int ans = 0;
    int count = 0;
    
    for(int i=1;i<=len;i++){
    	int r = i+(k-1);
    	bool arr[27] = {false};
    	count =  0;
    	for(int j=i;j<=r && r<=len;j++){
    		int val = str[j-1];
    		val++;
    		int index = val-97;
    		//cout<<"Val = "<<index<<"\n";
    		if(arr[index] == false){
    			arr[index] = true;
    			count++;
    		//	cout<<count<<"\n";
    		}
    		
    		if(j==r){
    			if(count==k){
    				ans++;
    			}
    			if(r<len){
    				r++;
    			}
    		}
    		
    		
    		
    	}
    	
    	
    }
    cout<<ans;
    
    
    return 0;
}
