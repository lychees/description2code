#include<iostream>

using namespace std;

int main()
{
    long long t,i,j,k,c,x,s;
    long long n,a;
    cin>>t;
    for(i=0;i<t;i++)
    {
        cin>>a;
        n=a;
        j=1;
        x=1;
        c=0;
        s=0;
        while((n/10)!=0)
        {
          n=n/10;
          c=(9*x*j)+c;
          s=s+9*x;
          x=x*10;
          j=j+1;

        }
        c=((a-s)*j)+c;
        cout<<c<<"\n";
    }
}
