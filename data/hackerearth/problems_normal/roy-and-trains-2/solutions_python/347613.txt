import math

t=input()
while t:
	t0,t1,t2,v1,v2,d = raw_input().split()
	t0=int(t0)
	t1=int(t1)
	t2=int(t2)
	v1=int(v1)
	v2=int(v2)
	d=int(d)
	if(t1>=t0):
		tim1=math.ceil(t1+(float(d)/v1)*60)
		tim1=int(tim1)
	if(t2>=t0):
		tim2=math.ceil(t2+(float(d)/v2)*60)
		tim2=int(tim2)
	if(t1<t0 and t2<t0):
		print("-1")
	elif(t1>=t0 and t2<t0):
		print(tim1)
	elif(t1<t0 and t2>=t0):
		print(tim2)
	else:
		print(min(tim1,tim2))
	t=t-1
