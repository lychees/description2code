#include<iostream>
#include<cstring>
using namespace std;
#define ll long long
int prime[1<<20];
ll  prime_sum[1<<20];
#define mx (int(1e6))
void sieve()
{
	memset(prime,-1,sizeof prime);//-1 indicates it's a prime
	memset(prime_sum,0,sizeof prime_sum);
	prime[1]=0;//0 means not a prime
	for(int i=2;i<=mx;++i)
	{
		for(int j=2*i;j<=mx;j+=i)
		{
			if(prime[j]==-1)
				prime[j]=0;
			else continue;
		}
	}
	prime_sum[1]=0;
	for(int i=2;i<=mx;++i)
	{
		if(prime[i]==-1)
		{	
			//printf("%d \n",i);
			prime_sum[i]+=(i+prime_sum[i-1]);
		}
		else prime_sum[i]+=prime_sum[i-1];
	}
	return;
}
int main()
{
	int tc;
	cin>>tc;
	sieve();
	while(tc--)
	{
		int x,y;
		cin>>x>>y;
		ll ans=prime_sum[y]-prime_sum[x-1];
		cout<<ans<<endl;
	}	

	return 0;
}
