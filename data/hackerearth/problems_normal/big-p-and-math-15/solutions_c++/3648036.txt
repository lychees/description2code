#include <iostream>
#include <cstdlib>
using namespace std;

int main()
{	string firstNumber;
	string secondNumber;
	
cin>>firstNumber>>secondNumber;

//for min first number
for(int i=0;i<firstNumber.length();i++){
	if (firstNumber[i]=='6'){
		firstNumber[i]='5';
	}
	
}
long int minFirstNumber = atol(firstNumber.c_str());

//for min second number
for(int i=0;i<secondNumber.length();i++){
	if (secondNumber[i]=='6'){
		secondNumber[i]='5';
	}
	
}

long int minSecondNumber = atol(secondNumber.c_str());

//for max first number
for(int i=0;i<firstNumber.length();i++){
	if (firstNumber[i]=='5'){
		firstNumber[i]='6';
	}
	
}
long int maxFirstNumber = atol(firstNumber.c_str());
//for max second number
for(int i=0;i<secondNumber.length();i++){
	if (secondNumber[i]=='5'){
		secondNumber[i]='6';
	}
	
}
long int maxSecondNumber = atol(secondNumber.c_str());


long int minSum = minFirstNumber + minSecondNumber;
long int maxSum = maxFirstNumber + maxSecondNumber;



    cout <<minSum<<" "<<maxSum;
    
}
