#include <iostream>
#include <vector>

using namespace std;

vector<long long int> vec;

int main()
{
	ios_base::sync_with_stdio(false);
	
	int t;
	cin>>t;
	while(t--) {
    	long long int g, n;
    	cin>>g>>n;
    	vec.erase(vec.begin(), vec.end());
  	
    	for(int i = 0; i < n-1; ++i)
    		vec.push_back(g*2ll);
    	
    	vec.push_back(g*3ll);
    	
    	for(int i = 0; i < n; ++i) {
    		cout<<vec[i]<<" ";
    	}
    	cout<<"\n";
    	
	}
    return 0;
}
