#include <iostream>
#include <vector>
#define vi vector<int>
#define all(c) c.begin(), c.end()

using namespace std;

int main() {
	int mod=1000000007;
	vi res{2,7};
	int t; cin>>t;
	for(int i=0;i<t;i++){
		int n; cin >>n;
		if(res.size() < n){
			int j = res.size();
			while(j<n){
				res.push_back(((2*res[j-2])%mod+res[j-1])%mod);
				j++;
			}
		}
		cout<<res[n-1]<<endl;
	}
	return 0;
}
