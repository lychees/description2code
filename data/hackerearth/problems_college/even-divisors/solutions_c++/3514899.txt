#include <iostream>
#include<math.h>
#define ll long long
using namespace std;
//#define n 1e+12
int divisors(ll n)
{
	if(n==1)
	return 1;
         ll cnt=0;
         int root=sqrt(n)+1;
		for(int j=1;j<root;j++)
		{
		if(n%j==0&&n/j!=j)
		{
			if(j%2==0&&(n/j)%2==0)
	       	cnt+=2;
	       	else if(j%2==0||(n/j)%2==0)
	       	cnt++;
		
		}
		if(n%j==0&&n/j==j&&j%2==0)
		cnt++;
		}
		return cnt;
}
int main()
{
   // cout << "Hello World!" << endl;
   int t;
   cin>>t;
   while(t--)
   {
   	ll n;
   	cin>>n;
   	int count=divisors(n);
   	cout<<count<<endl;
   }
    return 0;
}
