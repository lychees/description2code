#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	int pk=1;
	while(t--)
	{
		unsigned long long int x,y,k,ini;
		scanf("%llu %llu",&x,&k);
		ini=x;
		int j=0;
		while(k>0)
		{
			int temp=k&1;
			while(x&(1<<j))
			{
				j++;
			}
			if(temp)
			x=x|(1<<j);
			j++;
			k=k>>1;
		}
		x=x^ini;
		printf("Case #%d: %llu\n",pk,x);pk++;
	}
    return 0;
}
