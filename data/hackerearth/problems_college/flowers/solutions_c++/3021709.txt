#include <iostream>
#include <algorithm>
#include <string.h>
#define ll long long
using namespace std;

int main()
{
	ll n, k, i, j, sum=0;
    cin>>n>>k;
    ll a[n];
    for(i=0;i<n;i++)
    	cin>>a[i];
    sort(a,a+n);
    ll count[k], cost[k];
    memset(count,0,sizeof(count));
    memset(cost,0,sizeof(cost));
    j=0;
    for(i=n-1;i>=0;i--)
    {
    	cost[j]+=(count[j]+1)*a[i];
    	count[j]++;
    	j=(j+1)%k;
    }
    for(i=0;i<k;i++)
    	sum+=cost[i];
    cout<<sum<<endl;
    return 0;
}
