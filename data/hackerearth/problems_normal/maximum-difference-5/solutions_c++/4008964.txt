#include <bits/stdc++.h>
#define llint long long int
using namespace std;
int main()
{
    llint i,j,k,max_cnt=0,n,x=0;
    int arr1[100100],arr2[100100];
    cin>>n;
    for(i=0;i<n;i++) cin>>arr1[i];
    
	sort(arr1,arr1+n);
    
	j=0;
    for(i=0;i<n;i+=2)
    	arr2[i] = arr1[j++];
    	
    k = j;
    if(n%2) i = n-2;
    else i = n-1;
    for(;i>0;i-=2)
    	arr2[i] = arr1[j++];
    	
    for(i=0;i<n-1;i++)
    max_cnt += abs(arr2[i]-arr2[i+1]);

    int z;
    for(i=0;i<n/2;i++)
    {
    	z = arr1[i];
    	arr1[i] = arr1[n-i-1];
    	arr1[n-i-1] = z;
    }
    j=0;
    for(i=0;i<n;i+=2)
    	arr2[i] = arr1[j++];
    k = j;
    if(n%2) i = n-2;
    else i = n-1;
    for(;i>0;i-=2)
    	arr2[i] = arr1[j++];
    	x=0;
    for(i=0;i<n-1;i++)
    x += abs(arr2[i]-arr2[i+1]);
    max_cnt = max(max_cnt,x);
    cout<<max_cnt;
}
