#include<bits/stdc++.h>
using namespace std;
using ll = long long int;
const ll big = 1000000007;

ll v[100000];

ll AbsDiff(ll a[], ll n)
{
	sort(a,a+n);
  	if ( n < 2 ) return 0;
       
  	ll sum = 0;
  	for(ll i=n-1;i>=0;i--)
  	{
    	sum = sum + (a[i]*(i) - a[i]*(n-i-1));
  	}
  	return sum;
}

int main()
{
	std::ios_base::sync_with_stdio (false);

	int T;
	cin >> T;
	while(T--)
	{
		ll N = 1;
		cin >> N;

		for(ll i=0; i<N; i++)
		{
			cin >> v[i];
		}
		
		ll power = AbsDiff(v,N) % big;
		ll result = (v[N-1]*power) % big;
		cout <<  result <<endl;
	}
	return 0;
}
