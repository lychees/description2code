#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cmath>
using namespace std;

int main()
{
    int T,N;
    char str[2000001];
    cin>>T;
    while(T) {
    	cin>>N;
    	cin>>str;
    	int i = 0;
    	int cnt = 0;
    	while(str[i]) {
    		if(i%2&&str[i]=='B') 
    			cnt++;
    		i++;
    	}
    	cout<<min(cnt,i/2-cnt)<<endl;
    	T--;
    }
    return 0;
}
