#include <bits/stdc++.h>
using namespace std;

int gcd(int n, int q)
{
    //cout<<"n = "<<n<<" q = "<<q<<endl;
    if(q==0)
        return n;
    return gcd(q, n%q);
}

bool prime_check(int n)
{
    if(n==1)
        return false;
    int limit = sqrt(n);
    for(int q=2; q<=limit; q++)
        if(n%q==0)
            return false;
    return true;
}

int main()
{
    int t;
    cin>>t;
    while(t!=0){
        t--;
        int n;
        cin>>n;
        int check_num = 0;
        for(int q=1; q<=n; q++)
            if(gcd(n,q)==1)
                check_num++;
            //printf("GCD of %d and %d is %d\n", n, q, gcd(n, q));
        if(prime_check(check_num))
            cout<<"TRUE"<<endl;
        else
            cout<<"FALSE"<<endl;
    }
}
