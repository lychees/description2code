#include<cstdio>


int binarySearch(int a[], int n, int k){
    int low = 0;
    int high = 1000000000;
    int ans = 0;
    while(low <= high){
        int mid = (low+high)/2;
        int position = a[0]+mid;
        int num_jump = k;
        bool can_jump = true;
        for(int i=0;i<n;i++){
            while(position >= a[i] && i < n){
                i++;
            }
            i--;
            position = a[i]+mid;
            num_jump--;
            if(num_jump < 0){
                can_jump = false;
                break;
            }
        }
        if(can_jump){
            ans = mid;
            high = mid-1;
        }
        else{
            low = mid+1;
        }
    }
    return ans;
}

int main(){
    int t;
    scanf("%d", &t);
    while(t--){
        int n, k;
        scanf("%d %d", &n, &k);
        int val[n+1];
        for(int i=0;i<n;i++){
            scanf("%d", &val[i]);
        }
        printf("%d\n", binarySearch(val, n, k));
    }
}
