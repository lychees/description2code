n = input()
marut_desire = map(int, raw_input().strip().split())
girls = input()
answer = 0
for i in range(girls):
    girl_qualities = map(int, raw_input().strip().split())
    matchs = 0
    for desire in marut_desire:
        if desire not in girl_qualities:
            break
        matchs += 1
    if matchs == n:
        answer += 1
print answer
