#include <iostream>
#include<map>
using namespace std;

int main()
{
    long n;

    map<long,long> freq;
    cin>>n;
    long len;
    long num[10000];
    long col=0;
    for(int p=0;p<n;p++)
    {
        cin>>len;
          for(int i=0;i<10;i++)
        {

           freq[i]=0;

        }


         for(long i=0;i<len;i++)
        {

            cin>>num[i];
            int x=num[i]%10;
            freq[x]=freq[x]+1;

        }
     /*     for(int i=0;i<10;i++)
        {

           cout<<i<<" "<<freq[i]<<endl;

        }
*/
        map<long,long>::iterator it = freq.begin();

        while(it!=freq.end())
        {
            if(it->second!=0)
                col+=it->second-1;;
            it++;
        }
      cout<<col<<endl;
      col=0;
    }
    return 0;
}
