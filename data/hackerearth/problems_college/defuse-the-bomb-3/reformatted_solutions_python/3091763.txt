T = raw_input()
result = []
for t in range(0, int(T)):
    nums_input = raw_input()
    A = nums_input.split(" ")[0]
    B = int(nums_input.split(" ")[1])
    digits = len(str(B))
    result.append([])
    for i in range(0, len(A)-digits+1):
        result[t].append(int(A[i:i+digits])*B)


for t in result:
    for r in t:
        print r
