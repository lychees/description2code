T = int(raw_input())
while T > 0:
    T = T-1
    N = long(raw_input())
    totPoints = N*12
    q = totPoints/26
    rem = totPoints % 26
    cost = map(long, raw_input().split())
    totCost = 0
    for i in cost:
        totCost = totCost+i
    paintCost = q*totCost + 26*((q-1)*q)/2
    for i in range(rem):
        paintCost += (cost[i]+q)

    print paintCost
