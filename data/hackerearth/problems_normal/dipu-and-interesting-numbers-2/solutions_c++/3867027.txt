#include<iostream>
#include<cmath>
#define ull unsigned long long
using namespace std;

int main()
{
	long double t;
	cin>>t;
	while(t--)
	{
		ull l,r;
		cin>>l>>r;
		ull bottom=(sqrt(l)),top=(sqrt(r));
		if((top>999999999)&&(r<1000000000000000000))
		top--;
		cout<<top-bottom+(ull)((l-(bottom*bottom))==0)<<"\n";
	}
}
