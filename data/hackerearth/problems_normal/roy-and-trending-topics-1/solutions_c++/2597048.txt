#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
//function to sort the vector according to 2nd element of 1st element in vector nested pair
bool compare(const pair<pair<LL,LL>,LL >&p1,const pair<pair<LL,LL>,LL > &p2)
{
	return p1.first.second>p2.first.second;
}
//function to sort the vector according to 1st element of 1st element in vector nested pair
bool comp(const pair<pair<LL,LL>,LL >&p1,const pair<pair<LL,LL>,LL > &p2)
{
	return p1.first.first>p2.first.first;
}
vector<pair<pair<LL,LL>,LL > >v;

int main()
{
	int n;
	scanf("%d",&n);
	for(int i=0;i<n;i++)
	{
		LL id,oldZ,p,l,c,s;
		scanf("%lld %lld %lld %lld %lld %lld",&id,&oldZ,&p,&l,&c,&s);
		
		//calculating the new z-score value
		LL newZ=50*p+5*l+c*10+s*20; 
		
		//calculating the change
		LL change=newZ-oldZ;
		
		v.push_back(make_pair(make_pair(id,change),newZ));
	}
	sort(v.begin(),v.end(),comp); //sorting according to ID
	sort(v.begin(),v.end(),compare); //sorting according to change in Z-score
	for(int i=0;i<5;i++)
	{
		cout<<v[i].first.first<<" "<<v[i].second<<endl;
	}
}
