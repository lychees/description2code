#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cstdlib>
using namespace std;
int main()
{
          string str;
          getline(cin,str);
          //cout<<str;
          int i,n,l=str.length();
          cin>>n;
          for(i=0;i<l;i++)
          {
                    string s=str.substr(i,n);
                    i=i+n-1;
                    cout<<s<<endl;
          }
          return 0;
}
