int dp[100][1025];
int bfs(){
    queue<pair<int,int> > q;
    q.push(make_pair(0,cost[0]));
    dp[0][cost[0]]=1;
    int i,ans=0,x,y;
    while(!q.empty()){
        int node=q.front().first;
        int costt=q.front().second;
        q.pop();
        ans=max(ans,costt);
        for(i=0;i<v[node].size();i++){
            x=v[node][i];
            y=costt^cost[x];
            if(dp[x][y]==-1){
                dp[x][y]=1;
                q.push(make_pair(x,y));
            }
        }
    }
    return ans;
}
