#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int n;
		cin>>n;
		long long int a[n];
		for(int i=0;i<n;i++)
		{
			cin>>a[i];
		}
		long long int l,s;
		cin>>l>>s;
		sort(a,a+n);
		int sum=a[s-1] + a[n-l];
		cout<<sum<<"\n";
	}
}
