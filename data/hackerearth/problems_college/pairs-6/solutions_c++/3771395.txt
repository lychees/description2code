#include<bits/stdc++.h>
#include<math.h>
using namespace std;
int main()
{
    int T;
    scanf("%d",&T);
    while(T--)
    {
    	int N;
    	scanf("%d",&N);
    	int CT,T;
    	scanf("%d %d",&CT,&T);
    	int K=abs(CT-T);
    	if(K>1)
    	printf("Motu\n");
    	else
    	printf("Chotu\n");
    }
    return 0;
}
