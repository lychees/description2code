#include <iostream>
using namespace std;

int main()
{
    long long int t,a,b,c,d,e;
    cin>>t;
    long long int x,y,ans;
    while(t--)
    {
    	cin>>a>>b>>c>>d>>e;
    	if(c-b>=e)
    	{
    		cout<<b<<endl;
    		continue;
    	}
    	else if(c-b>=e||c-a>=e)
    	{
    		x=c-b;
    		y=e-x;
    		ans=b-y;
    		// ans=ans-a+1;
    	}
    	else
    	{
    		cout<<"STAY IN LONDON\n";
    		continue;
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
