#include <iostream>
#include <string>
#define ll long long
#define min(a,b) a<b?a:b
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,m,c;
        cin>>n>>c>>m;
        int ans=0;
        ans+=(n/c);
        int w=ans;
        int x=(w/m);
        int exw=(w%m);
        if(ans>0)
        {
            while(x>0)
            {
                ans+=x;
                exw+=x;
                x=exw/m;
                exw=exw%m;
            }
        }
        cout<<ans<<endl;
    }
    return 0;
}
