#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n,q,k,i;
    cin>>n>>q;
    vector<long>t(n);
    long s;
    for(i=0;i<n;i++)
    	cin>>t[i];
    for(i=0;i<n;i++)
    	cin>>s;
    sort(t.begin(),t.end());
    while(q--)
    {
    	long ans=0;
    	cin>>k;
    	for(i=0;i<k;i++)
    	{
    		ans=ans+t[n-1-i];
    	}
    	cout<<ans<<endl;
    }
    
    
    return 0;
}
