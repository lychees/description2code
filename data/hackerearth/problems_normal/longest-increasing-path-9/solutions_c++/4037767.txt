#include <iostream>
using namespace std;

#define INF 1000000000;

int A[100][100], dp[100][100];

int main()
{
	int t, n, m;
	cin>>t;
	
	while(t--){
		cin>>n>>m;
		for (int i=0; i<n; i++) 
			for (int j=0; j<m; j++) 
				cin >> A[i][j];
		
		for (int i=n-1; i>=0; i--) {
			for (int j=m-1; j>=0; j--) {
				dp[i][j] = 1;
				if (i+1<n && A[i+1][j] > A[i][j])
					dp[i][j] = 1+dp[i+1][j];
				if (j+1<m && A[i][j+1] > A[i][j])
					dp[i][j] = max(dp[i][j], 1+dp[i][j+1]);
			}
		}
		/*for (int i=0; i<n; i++) {
			for (int j=0; j<m; j++) {
				cout << dp[i][j] << " ";
			}
			cout<<endl;
		}
		*/
		cout << dp[0][0] << endl;
	}
	
    return 0;
}
