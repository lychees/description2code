#include <bits/stdc++.h>
using namespace std;

#define ll long long int
#define mod 1000000007

int prime[10002]={0};

void sieve()
{
	int i;
	for(i=2;i*i<10002;i++)
	{
		if(!prime[i])
		{
			for(int j=i*2;j<10002;j=j+i)
			prime[j]=1;
		}
	}
	/*for(i=2;i<10002;i++)
	{
		if(!prime[i])
		cout << i <<" ";
	}*/
}

int main()
{
    int t;
    cin >> t;
    sieve();
    while(t--)
    {
    	ll n,i,ans=1;
    	cin >> n;
    	ll a[n];
    	for(i=0;i<n;i++)
    	cin >> a[i];
    	for(i=2;i<10002;i++)
    	{
    	ll j,p=0;
    		while(!prime[i])
    		{
    			ll f=0;
    			//cout << "prime= " << i << "\n";
    			for(j=0;j<n;j++)
    			{
    				if(a[j]%i==0)
    				{
    					a[j]=a[j]/i;
    					f++;
    				}
    			}
    			if(f>0)
    			ans=(ans*i)%mod;
    			else
    			break;
    		}
    		/*for(int k=0;k<n;k++)
    		{
    			//cout << a[k] << " ";
    			if(a[k]==1)
    			p++;
    		}
    		if(p==n)
    			break;*/
    	}
    	cout <<ans <<"\n";
    }
    return 0;
}
