#include <bits/stdc++.h>
using namespace std;

vector< vector<int> > A; 
vector<int> vis; 
void dfs(int p) {
	vis[p] = 1; 
	for(int i = 0; i < int(A[p].size()); ++i) 
		if(!vis[A[p][i]]) dfs(A[p][i]); 
}
int main() {
	int t; scanf("%d", &t); 
	while(t--) {
		int n; scanf("%d", &n); 
		int e; scanf("%d", &e); 
		vis = vector<int>(n + 1, 0); 
		A = vector< vector<int> > (n + 1); 
		for(int i = 0; i < e; i++) {
			int x, y; scanf("%d %d", &x, &y); 
			A[x].push_back(y); 
			A[y].push_back(x); 
		}
		int ans = 0; 
		for(int i = 0; i < n; i++) {
			if(!vis[i]) {
				++ans; dfs(i); 
			}
		}
		printf("%d\n", ans); 
	}
    
    return 0;
}
