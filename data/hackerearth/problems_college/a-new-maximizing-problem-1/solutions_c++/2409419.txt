#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
   // cout << "Hello World!" << endl;
   int t,n;
   scanf("%d",&t);
   while(t--)
   {
   	scanf("%d",&n);
   	vector<int> v;
   	int i,x;
   	ll fir = 0;
   	ll sec = 0;
   	for(i=0;i<n;i++)
   	{
   		scanf("%d",&x);
   		v.push_back(x);
   		if(x>=0)fir+=x;
   		else fir-=x;
   	}
   	sort(v.begin(),v.end());
   	for(i=0;i<(n+1)/2;i++)sec-=v[i];
   	for(;i<n;i++)
   	sec+=v[i];
   	ll hi = 1000000007;
   	fir%=hi;
   	sec%=hi;
   	fir = (fir+hi)%hi;
   	sec = (sec+hi)%hi;
   	printf("%lld\n",(fir*sec)%hi);
   }
    return 0;
}
