#include <iostream>
#include <string>
#include <cmath>
using namespace std;

int main()
{
	int t = 0, i = 0, j = 0;
	cin >> t;
	string str;
	while(t--){
		cin >> str;
		i = 0;
		j = str.size() - 1;
		bool flag = false;
		while(i < j){
			//cout << abs(str[i] - str[i+1]) << " " << abs(str[j]- str[j-1]) << endl;
			if(abs(str[i] - str[i+1]) != abs(str[j]- str[j-1])){
				flag = true;
				break;
			}
			i++;
			j--;
		}
		if(flag)
	    	cout << "Not Funny" << endl;
	    else
	    	cout << "Funny" << endl;
	}
    return 0;
}
