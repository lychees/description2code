#include <iostream>
using namespace std;

int main()
{
    int i,r,j;
    string n;
    r=0;
    while(getline(cin,n))
    {
    	for(i=0;i<n.length();i++)
    	 {
    	 	if(n[i]!=' ')
    	 	{
    	 		if(n.substr(i)=="puts \'congats!\'")
    	 		cout << "puts \'congrats!\'" << endl;
    	 		else
    	 		cout << n.substr(i) << endl;
    	 		break;
    	 	}
    	 }
    	if (n.substr(i,5)=="class")
    	r++;
    	else if (n.substr(i,3)=="def")
    	r++;
    	else if (n.substr(i,2)=="if")
    	cout << " ";
    	else if (n.substr(i,3)=="els")
    	cout << " ";
    	else if (n.substr(i,3)=="end")
    	r--;
    	for(j=0;j<r;j++)
    	 cout <<" ";
    	 
    }
    return 0;
}
