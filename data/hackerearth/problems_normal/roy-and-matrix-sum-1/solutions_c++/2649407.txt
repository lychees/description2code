#include <iostream>
using namespace std;

long long int s(long long int n)
{
	return (n*(n+1)/2);
}
int main()
{
    int t;
    cin>>t;
    long long int n;
    
    while(t--)
    {
    	cin>>n;
    	long long int sum=0;
    	for(long long int i=0;i<n;i++)
    	{
    	
    			sum+=s(n-1-i);
    	}
    	cout<<2*sum<<endl;
    }
    return 0;
}
