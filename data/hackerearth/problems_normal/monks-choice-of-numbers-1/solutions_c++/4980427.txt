#include <cmath>

#include <cstdio>
#include <vector>
#include <cstring>
#include <iostream>
#include <algorithm>
#include <climits>
using namespace std;
#define MOD 1000000007
#define ll long long int
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,k;
		scanf("%d%d",&n,&k);
		int a[n],x;
		for (int i = 0; i < n; ++i)
		{
			scanf("%d",&x);
			a[i] = __builtin_popcount(x);
		}
		sort(a,a+n);
		int sum = 0;
		int i = n-1;
		while(k--)
		{
			sum += a[i];
			i--;
		}
		cout<<sum<<endl;
	}
}
