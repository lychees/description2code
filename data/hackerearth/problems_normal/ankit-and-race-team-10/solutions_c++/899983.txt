#include <iostream>
#include<bits/stdc++.h>
using namespace std;

#define lim 100000
#define mod 1000000007
#define loop(i,a,b) for(int i=a;i<b;i++)
#define ll long long int
ll fac[100010];
ll invfac[100010];

ll Pow(ll a,ll b)
{
	if(b==0) return 1;
	ll res=Pow(a,b/2); res%=mod;
	res=res*res; res%=mod;
	
	if(b&1) res=res*a; res%=mod;	
	return res;
}

void pre()
{
		fac[0]=1;
		fac[1]=1;
		invfac[0]=1;
		invfac[1]=1;
		loop(i,2,lim+1)
		{
			fac[i]=fac[i-1]*i;
			fac[i]%=mod;
			invfac[i]=invfac[i-1]*Pow((ll)i,(ll)(mod-2));
			invfac[i]%=mod;
		}
		return;
}

ll C(int x,int y)
{
	ll res=fac[x];
	res*=invfac[y]; res%=mod;
	res*=invfac[x-y]; res%=mod;
	return res;
}

int n,k;
int a[100010];
void Scan()
{
	cin>>n>>k;
	loop(i,0,n) cin>>a[i];
}

void Out()
{
	ll res=0;
	sort(a,a+n);
	loop(i,0,n-k+1)
	{
		res+=a[i]*C(n-i-1,k-1); res%=mod;
		//printf("%lld ",C(0,0));
	}
	cout<<res<<endl;
	return;
}
int main()
{
	pre();
	int t; cin>>t;
	while(t--)
	{
		Scan();
		Out();
	}
    return 0;
}
