#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;

int main()
{
int T,tmp = 0, totallen=0;
cin>>T;
for (int i=0; i<T; i++){
	char str[500];
    cin>>str;
    tmp=0;
	totallen= strlen(str);
    for (int i=4;i<totallen; i++){
    	  
    	if (str[i] == 'a' || str[i] == 'e' || str[i] == 'i' || str[i] == 'o' || str[i] == 'u')
     	tmp++;
  
  }
	cout << totallen-(tmp+3) <<"/"<< totallen<<"\n";
}
   return 0;
}
