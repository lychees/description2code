def factors(n):    
    return list(set(reduce(list.__add__, 
                ([i, n//i] for i in range(1, long(n**0.5) + 1) if n % i == 0))))
for t in range(input()):
    num=sorted(map(long,raw_input().split()))
    fact=sorted(factors(num[0]),reverse=True)
    flag=False
    for d in fact:
        flag=True
        for n in num:
            if n%d!=0:
                flag=False
                break
        if flag==True:
            print len(factors(d))
            break
