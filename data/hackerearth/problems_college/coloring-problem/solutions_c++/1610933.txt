#include <bits/stdc++.h>
using namespace std;
int mat[201][201];
bool visited[201];
int color[201];
queue<int> q;
int main()
{
   int t;
   cin >> t;
   while(t--)
   {
   	memset(color,0,sizeof(color));
   memset(visited,0,sizeof(visited));
   
   int n,l,i,j,k,q1,p,flag=0,counter=0;
   	int red=0;
   	cin>>n;
   	cin>>l;
   	for(i=0;i<n;i++)
   	{
   		for(j=0;j<n;j++)
   		mat[i][j]=0;
   	}
   	for(i=0;i<l;i++)
   	{
   		cin>>p>>q1;
   		mat[p][q1]=1;
   		mat[q1][p]=1;
   	}
    q.push(0);
    visited[0]=true;
    color[0]=red;
    while(!q.empty())
    {
    	int v= q.front();   q.pop();
    	for(i=0;i<n;i++)
    	{
    		if(mat[v][i])
    		{
    			if(!visited[i])
    			   {
    			   	q.push(i);
    			   	visited[i]=true;
    			   	color[i]=!(color[v]);
    			   }
    			else 
    			{
    				if(color[i]==color[v])
    				{
    					flag=1;//non colourable
    					break;
    				}
    				
    				
    			}
    			
    			
    		}
    		
    		
    	}
    	
    	
    	
    }
   	if(flag)
   	cout << "NOT BICOLORABLE."<<endl;
   	else
   	cout << "BICOLORABLE."<<endl;
   	
   }
   
   
   
   //cout << "Hello World!" << endl;
    return 0;
}
