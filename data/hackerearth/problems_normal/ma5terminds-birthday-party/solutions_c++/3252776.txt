#include <bits/stdc++.h>
using namespace std;

#define MOD 1000000007LL
#define ll long long
#define MAXN 1000010
int pr[MAXN];

int main()
{
	pr[0] = pr[1] = 1;
	for(int i=2;i*i<MAXN;++i)
		if(!pr[i])
			for(int j=i*i;j<MAXN;j+=i)
				pr[j] = 1;
	pr[4] = 1-pr[4];
	for(int i=1;i<MAXN;++i)
		pr[i] += pr[i-1];
	cin.tie(0);
	ios::sync_with_stdio(false);
	int t,l,r;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d",&l,&r);
		printf("%d\n",pr[r]-pr[l-1]);
	}
	return 0;
}
