#include <bits/stdc++.h>
using namespace std;

#define N 10000001
#define sd(x) scanf("%d", &x)

bool isPrime[N];
int primes[N], c = 0;
int rev[50][50];

inline void Pre(){
    int i, j;
    for(i = 2; i < N; i++){
        isPrime[i] = 1;
    }
    for(i = 2; i * i < N; i++){
        if(!isPrime[i]) continue;
        for(j = i * i; j < N; j += i){
            isPrime[j] = false;
        }
    }
    for(i = 2; i < N; i++){
        if(isPrime[i]) primes[c++] = i;
    }
    int t = 1, t1 = 1, num, num1, k;
    for(i = 1; i <= 23; i++){
        t *= 2;
        num = t - 1;
        t1 = 1;
        for(j = 1; j <= 23; j++){
            t1 *= 2;
            num1 = t1 - 1;
            int o = 0;
            for(k = 0; k < c; k++){
                if(num < primes[k] or num1 < primes[k]) break;
                if(num % primes[k] == 0 and num1 % primes[k] == 0) o++;
            }
            rev[i][j] = rev[j][i] = o;
        }
    }
    return;
}

int main(){
    Pre();
    int t, x, y, i;
    sd(t);
    while(t--){
        sd(x); sd(y);
        printf("%d\n", rev[x][y]);
    }
    return 0;
}
