#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s;
    stack<int> a;
    cin>>s;
    int c=0;
    int l=s.length();
    for(int i=l-1;i>=0;i--){
    	if(s[i]>=48 && s[i]<=57 && s[i]%2==0)
    		c++;
    	a.push(c);
    }
    while(!a.empty()){
    	cout<<a.top()<<" ";
    	a.pop();
    }
    return 0;
}
