#include <bits/stdc++.h>

using namespace std;

const int N = 1e6 + 10;

int pos[N];
long long cs[N];

const int MAXN = 1e6;
bool is_prime[MAXN + 1];

void sieve() {
	fill(is_prime + 2, is_prime + MAXN + 1, true);
	for (int i = 2; i * i <= MAXN; ++i) {
		if (is_prime[i] == true) {
			for (int j = i * i; j <= MAXN; j += i) {
				is_prime[j] = false;
			}
		}
	}
}

int main() {
	sieve();
	cs[0] = 0;
	for (int i = 2; i <= MAXN + 1; i++) {
		if (is_prime[i - 1]) {
			cs[i] = cs[i - 1] + i - 1;
		} else {
			cs[i] = cs[i - 1];
		}
	}
	int tt;
	scanf("%d", &tt);
	for (int tc = 1; tc <= tt; tc++) {
		int l, r;
		scanf("%d %d", &l, &r);
		printf("%lld\n", cs[r + 1] - cs[l]);
	}
	return 0;
}
