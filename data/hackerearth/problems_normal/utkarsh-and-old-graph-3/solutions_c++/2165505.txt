#include <iostream>
using namespace std;

int main()
{
    int N,M;
    cin>>N>>M;
    int arr[N];
    for(int i=0;i<M;i++)
    {
    	arr[i]=0;
    }
    
    for(int i=0;i<M;i++)
    {
    	int x,y;
    	cin>>x>>y;
    	arr[x-1]++;
    	arr[y-1]++;
    }
    
    int count =0;
    for(int i=0;i<N;i++)
    {
    	if(arr[i]>1)
    	{
    		int z=arr[i];
    		count+=(z*(z-1))/2;
    	}
    	
    //	cout<<arr[i]<<" ";
    }
    
    if(N%2==0)
    {
    	long long x = (N-2)/2;
    	x=x*(N-1);
    	cout<<x;
    }
    else
    {
    	long long x = (N-1)/2;
    	x=x*(N-2);
    	cout<<x;    	
    }
    
    
    
    return 0;
}
