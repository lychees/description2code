#include <bits/stdc++.h>
using namespace std;
#define ll long long int
struct node
{
	ll x, y;
}a[1000001];
ll b[1000001];
int main()
{
    ll t, n, k, i, j, m;
    cin>>t;
    while(t--) {
    	cin>>n;
    	
    	for(i = 0 ; i < n; i++)
    		cin>>a[i].x>>a[i].y;
    	k  = 0;
    	m = 0;
    	memset(b, 0, sizeof(b));
    	for(i = 0 ; i < n; i++) {
    		j = i;
    		k  = 0;
    	//	memset(b, 0, sizeof(b));
    		while(a[j].y == 1 && j < n) {
    			if(b[a[j].x] == 0)
    			k++;
    			b[a[j].x] = 1;
    			j++;
    		}
    		if(a[j].y == 1 && j <= n-1 && a[j-1].x <= a[j].x && a[j-1].y == 1 && b[a[j].x] == 0)
    			k++;
   
    		i = j;
    		m = max(m, k);
    	}
    	cout<<m<<endl;
    }
    return 0;
}
