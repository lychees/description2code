#include <iostream>
using namespace std;

int main()
{
    int t,n,k,count=0,i;
    cin>>t;
    while (t--)
    {
    	cin>>n>>k;
    	count=0;
    	int a[n];
    	for (i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	for (i=0;i<n;i++)
    	{
    		if (a[i]<=0)
    		{
    			count++;
    		}
    	}
    	if (count<k)
    	{
    		cout<<"YES\n";
    	}
    	else
    	{
    		cout<<"NO\n";
    	}
    	
    }
    return 0;
}
