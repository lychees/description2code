#include <iostream>
#include<map>
#include<string>
using namespace std;

int main()
{
    map<string,int> mymap;
    int n,i;
    string s,mas;
    int x,masx=-1;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>s>>x;
    	if(mymap.find(s)!=mymap.end())
    	{
    		mymap[s]+=x;
    		if(mymap[s]>masx)
    		{	mas=s;
    			masx=mymap[s];
    		}
    	}
    	else
    	{
    		mymap.insert(make_pair(s,x));
    		if(x>masx)
    		{	mas=s;
    			masx=x;
    		}
    	}
    }
    cout<<mas;
    return 0;
}
