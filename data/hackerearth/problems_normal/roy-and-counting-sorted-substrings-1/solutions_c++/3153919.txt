#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--) {
    	int n;
    	cin>>n;
    	getchar();
    	char str[1000000];
    	cin>>str;
    	char a=str[0];
    	long long int ans=0,c=1,i;
    	for(i=1;str[i]!='\0';i++) {
    		if(str[i]>=a) {
    			c++;
    			a=str[i];
    		}
    		else{
    			ans=ans+(c*(c+1))/2;
    			c=1;
    			a=str[i];
    		}
    	}
    	ans=ans+(c*(c+1))/2;
    	cout<<ans<<endl;
    }
    return 0;
}
