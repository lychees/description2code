import math 



def next_perfect(n):
    s = int(math.sqrt(n))
    s2 = s**2
    if s2 == n:
        return "YES"
    lr = (s+1)**2
    if lr - n <= n - s2:
        return ('+'+str(lr-n))
    else:
        return ('-'+str(n-s2))


if __name__=='__main__':
    for i in range(0,int(raw_input())):
        print next_perfect(int(raw_input()))
