#include <bits/stdc++.h>
using namespace std;
long long int dp[1000000];
long long int a[1000000];
int n;
long long int foo(int i) {
	if(i>=n)
	return 0;
	if(dp[i]!=-1)
	return dp[i];
	dp[i] = max(a[i]+foo(i+2),foo(i+1));
	return dp[i];
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--) {
    	scanf("%d",&n);
    	int i;
    	for(i=0;i<n;i++) {
    	scanf("%lld",&a[i]);
    	dp[i] = -1;}
    	long long int qw = foo(1);
    	for(i=0;i<n;i++)
    	dp[i] = -1;
    	n--;
    	long long int qw1 =  max(foo(0),qw);

    	printf("%lld\n",qw1);
    }
    return 0;
}
