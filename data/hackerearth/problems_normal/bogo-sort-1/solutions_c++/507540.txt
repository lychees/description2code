#include <cstdio>
#include <algorithm>
#define sf scanf
#define pf printf
#define rn return
using namespace std;

int main(){
	int p;
	int i;
	sf("%d", &p);
	
	long long x[p];
	
	//for(i=0; i<p; i++)
	
	for(i=0; i<p; i++)
		sf("%lld", &x[i]);
		
	sort(x, x+p);
	
	for(i=0; i<p; i++)
		pf("%lld\n", x[i]);
	
	rn 0;
}
