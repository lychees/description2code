#include <iostream>
#include <string>
#include <stdlib.h>
#include <utility>
#include <string.h>

using namespace std;

#define ll long long int

ll ans[1000005];

struct node{
	node* children[26];
	int tot;
};

struct trie{
	node* root;
};

node* getNode(){
	node* temp=(node* )malloc(sizeof(node));
	for(int i=0;i<26;i++){
		temp->children[i]=NULL;
	}
	temp->tot=0;
	return temp;
}

void init(trie* t){
	t->root=getNode();
}

void insert(trie* t,string s){
	node* pCrawl=t->root;
	int len=s.length();
	for(int i=0;i<len;i++){
		pCrawl->tot++;
		if(!pCrawl->children[s[i]-'a'])
			pCrawl->children[s[i]-'a']=getNode();
		ans[i]+= pCrawl->tot - pCrawl->children[s[i]-'a']->tot - 1;
		pCrawl=pCrawl->children[s[i]-'a'];
	}
	pCrawl->tot++;
	ans[len]+= pCrawl->tot -1;
}
	


int main()
{
    //cout << "Hello World!" << endl;
    memset(ans,0,sizeof(ans));
    
    int n,m=0;
    trie t;
    string s;
    
    cin >> n;
    
    init(&t);
    
    for(int i=0;i<n;i++){
    	cin >> s;
    	int l=s.length();
    	m=max(m,l);
    	insert(&t,s);
    }
    
    // for(int i=0;i<=m;i++){
    // 	cwrapper(&t,i,i);
    // }
    for(int i=0;i<=m;i++){
    	cout << ans[i] <<" ";
    }
    cout << endl;
    
    
    return 0;
}
