#include<bits/stdc++.h>
using namespace std;
#define max 10000005
int v[max],phi[max];
void sieve()
{
	int i,j;
	for(i=2;i<max;i++)
	{
		if(v[i]==0)
		{
			phi[i]=(i-1);
			for(j=i*2;j<max;j+=i)
			{
				if(phi[j]==0)
				{
					phi[j]=j;
				}
				phi[j]/=i;
				phi[j]*=(i-1);
				v[j]=1;
			}
		}
	}
}
int main()
{
	sieve();
	int t,n;
	cin>>t;
	while(t--)
	{
		scanf("%d", &n);
		printf("%d\n", phi[n]);
	}
}
