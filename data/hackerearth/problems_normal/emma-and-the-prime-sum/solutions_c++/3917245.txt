#include <bits/stdc++.h>
#define MAX 10000005
using namespace std;

int arr[MAX];
long long int sum[MAX];
void pre()
{
        int i,j,root;
        for(i=0;i<MAX;i+=2)
                arr[i]=0;
        for(i=3;i<MAX;i+=2)
                arr[i]=1;
        root=sqrt(MAX);
        for(i=3;i<=root;i+=2)
        {
                if(arr[i])
                {
                        for(j=i*i;j<MAX;j+=(i<<1))
                                arr[j]=0;
                }
        }
        arr[2]=1;
        sum[0]=sum[1]=0;
        for(int i=0;i<MAX;i++)
        {
            if(arr[i])
                sum[i] = sum[i-1]+i;
            else
                sum[i] = sum[i-1];
        }
        
} 

int main() {
	ios_base::sync_with_stdio(0);
	
	long int t,a,b;
	
	cin>>t;
	pre();
	
	while(t--)
	{
		cin>>a>>b;
		
		cout<<sum[b]-sum[a-1]<<endl;
	}
	// your code goes here
	return 0;
}
