#include<iostream>
#define MOD 1000000007
using namespace std;
long long int pow(long long int base,long long int exp)
{
    long long int ret=1LL;
    while(exp>0)
    {
        if(exp%2LL==1)
            ret=((ret%MOD)*(base%MOD))%MOD;
        exp=exp/2;
        exp=exp%MOD;
        base=((base%MOD)*(base%MOD))%MOD;
    }
    return ret%MOD;
}
int main(void)
{
    int t;
    cin>>t;
    while(t--)
    {
        long long int n;
        cin>>n;
        if(n<=2)
        {
            cout<<"0\n";
            continue;
        }
        long long int ans;
        ans=pow(3,n)%MOD;
        //cout<<ans<<endl;
        long long int temp=pow(2,n)%MOD;
        //cout<<temp<<endl;
        ans-=(temp*3LL);
        while(ans<0)
            ans+=MOD;
        ans=(ans+3)%MOD;
        cout<<ans%MOD<<endl;
    }
    return 0;
}
