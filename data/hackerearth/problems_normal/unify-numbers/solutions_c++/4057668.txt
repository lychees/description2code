#include <iostream>
#include <stdio.h>
#include <math.h>
#include <algorithm>
using namespace std;

int main()
{

    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	scanf("%d",&n);
    	long long a[n];
    	//long long min=1000000001,max=-1;
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	//int av=3/2.0;
    	//cout<<av<<endl;
    	long long cost=0;;
    	long long val;

    	sort(a,a+n);
    	if(n%2)
    	{
    	    val=a[n/2];
    	}
    	else
    	{
    	    val=(a[n/2]+a[n/2-1])/2;
    	}
    	
    	//cout<<"val="<<val<<endl;
    	for(int i=0;i<n;i++)
    	{
    		cost+=abs(val-a[i]);
    	}
    	cout<<cost<<endl;
    }
    return 0;
}
