#include <iostream>
#include<string>
using namespace std;

int main()
{
    string S;
    cin>>S;
    int len = S.length();
    char rev[len];
    
    int i, j, count = 0;
    
    for(j = 0, i = len - 1; j < len, i >= 0; i--, j++)
		   	rev[j] = S[i];

 	rev[len] = NULL;
  // cout<<rev<<endl;
   for(i = 0; i < len; i++)
   {
   		if(S[i] == rev[i])
   			count++;
   }
   
   if(count == len)
   		cout<<"YES";
   else
   		cout<<"NO";
 return 0;
}
