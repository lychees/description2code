#include<bits/stdc++.h>
using namespace std;
unsigned long long a[3] ;
int main()
{
	int t ;
	scanf("%d",&t);
	while(t--)
	{
		unsigned long long x , y , n,i,k;
		scanf("%llu%llu%llu",&x,&y,&n);
		a[0] = x ;
		a[1] = y ;
		unsigned long long sum = x + y ;
		for(i=2;i<n;i++)
		{
			if(a[0]>a[1])
				{
					k = a[0] - a[1]  ;
					sum = sum + k ;
					a[0] = a[1] ;
					a[1] = k ;
				}
			else if(a[1]>a[0])
			{
					k = a[1] - a[0]  ;
					sum = sum + k ;
					a[0] = a[1] ;
					a[1] = k ;
			}
			else
			{
				i++;
				n = n - i ;
				if((n%3)==1)
					sum = sum + (n/3)*(a[0]*2) + a[0] ;
				else if((n%3)==2)
					sum = sum + (n/3)*(a[0]*2) + 2*a[0] ;
				else
					sum = sum + (n/3)*(a[0]*2) ;
				break ;
			}
		}
		printf("%llu\n",sum);
	}
}
