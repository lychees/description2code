#include<iostream>
#include<algorithm>
 
using namespace std;
 
struct apples
{
	int x;
	int y;
};
 
int compare(const void *a,const void *b)
{
    struct apples *a1=(struct apples *)a;
    struct apples *b1=(struct apples *)b;
    if(a1->x==b1->x)
    {
        return a1->y-b1->y;
    }
    else
        return a1->x-b1->x;
}
 
int main()
{
	struct apples cal[100009];
	int i,n,c,arr[100009],wait[100009],waitv=0,min,rank[100009];
	cin>>n>>c;
	for(i=0;i<n;i++)
	{
		cin>>arr[i];
		wait[i]=waitv;
		if(arr[i]>=c)
			waitv+=c;
		else
			waitv+=arr[i];
		min=arr[i]/c;
		if(arr[i]%c==0)
			min--;	
		cal[i].x=min;
		cal[i].y=i;
	}
	qsort(cal,n,sizeof(struct apples),compare);
	for(i=0;i<n;i++)
	{
		rank[cal[i].y]=i+1;	
	}
	int q;
	cin>>q;
	for(int i=0;i<q;i++)
	{
		int m,d;
		cin>>m>>d;
		if(m==1)
			cout<<wait[d-1]<<endl;
		else
			cout<<rank[d-1]<<endl;
	}
	return 0;
}
