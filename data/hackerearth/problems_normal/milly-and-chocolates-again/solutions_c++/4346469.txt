#include <iostream>
using namespace std;
#define SIZE 100000
int binsearch(int arr[],int s,int l,int key)
{
    int mid = (s + l)/2;
    if (mid == 0)
    {
        if (arr[mid] >= key)
        {
            return mid+1;
        }
    }
    if (key > arr[mid-1] && key <= arr[mid])
    {
        return mid+1;
    } else if (key < arr[mid])
    {
        return binsearch(arr,s,mid-1,key);
    } else
    {
        return binsearch(arr,mid+1,l,key);
    }
}
int main()
{
    int n,q,ci,index;
    int arr[SIZE] = {0};
    cin>>n;
    int sum = 0;
    for (int i = 0; i < n ; i++)
    {
        cin>> ci;
        sum += ci;
        arr[i] = sum;
    }
    cin >> q;
    for (int i = 0; i < q; i++)
    {
        cin >> index;
        cout << binsearch(arr,0,n-1,index) << endl;
    }
    return 0;
}
