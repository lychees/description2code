#include <bits/stdc++.h>
#define ll long long int
using namespace std;

void mark(vector<bool> &b,int i)
{
	int j=2;
	while(i*j<600)
	{
		int t=i*j;
		b[t]=true;
		j++;
	}
}
void seive(vector<bool> &b)
{
	int i=2;
	while(i<600)
	{
		mark(b,i);
		i++;
	}
}

int main()
{
    int n;
    scanf("%d",&n);
    set<int> s;
    unsigned ll sum=0;
    vector<bool> b(600,false);
    seive(b);
    for(int i=2;i<=n;i++)
    {
    	if(!(b[i]))
    	{
    	s.insert(i);
    	}
    	long double d=pow(2,s.size()-1);
    	sum=sum+(unsigned long long int)d;
    }
    cout<<sum<<"\n";
    return 0;
}
