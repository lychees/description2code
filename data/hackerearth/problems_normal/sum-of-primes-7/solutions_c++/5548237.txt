#include <iostream>
using namespace std;
#define num 1000003
long long int sum[1000003] = {0},size=0;
bool b[1000005] = {0};
void init(){
	b[1] =1;
	for(int i=2;i<1000002;i++){
		if(!b[i]){
			for(int j=i+i;j<1000002;j+=i){
				b[j] = 1;
			}
		}
	}
	// // cout << size<<endl;
	// for(int i=1;i<size;i++){
	// 	primes[i] += primes[i-1];
	// }
	sum[0]=0;sum[1] =0 ;
	for(long long int i=2;i<num;i++){
		if(!b[i]){
			sum[i] += (i+sum[i-1]);
		}
		else{
			sum[i] = sum[i-1];
		}
	}
	//for(int i=0;i<10;i++)cout << sum[num - 2-i] << " ";
}

int main() {
	init();
	// your code goes here
	int n,l,r;
	cin >> n;
	while(n--){
		cin >> l >> r;
		if(!b[l])cout << (sum[r]-sum[l]+l)<<endl;
		else cout << (sum[r]-sum[l])<<endl;
	}
	//cout << sum[999999];
	return 0;
}
