#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    long int x,y,n;
    int count1,count2;
    while(t--)
    {
      	count1=count2=0;
    	cin>>x>>y>>n;
    	vector <int> a;
    	a.resize(n);
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	sort(a.begin(),a.end());
    	for(int i=0;i<n;i++)
    	{
    			if(x>=a[i])
    			{
    				x=x-a[i];
    				count1++;		//win according to no. of dishes
    			}
    			if(y>=a[i])
    			{
    				y=y-a[i];
    				count2++;		//win according to no. of dishes
    			}
    	}
    	if(count1>count2)
    	{
    		cout<<"Chef Won"<<endl;
    	}
    	else if(count1<count2)
    	{
    		cout<<"Jhool Won"<<endl;
    	}
    	else
    	{
    		cout<<"Tie"<<endl;
    	}
    }
    return 0;
}
