#include <iostream>
using namespace std;

int revnum(int n)
{
    int r = 0;
    while(n!=0)
    {
        r = r*10+n%10;
        n/=10;
    }
    return r;
}

int main()
{
    int t, a, b, s;
    cin >> t;
    while(t--)
    {
        cin >> a >> b;
        a = revnum(a);
        b = revnum(b);
        s = a+b;
        s = revnum(s);
        cout << s << endl;
    }
    return 0;
}
