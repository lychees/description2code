#include <iostream>
#include <bits/stdc++.h>
using namespace std;
char s[100005];
int main()
{
	stack <string> st;
	int t; scanf ("%d", &t);
	getchar();
	while (t--) {
		gets(s);
		stringstream ss;
		ss << s;
		string str;
		while (ss >> str) {
			st.push(str);
		}
		while (!st.empty()) {
			cout << st.top() << " ";
			st.pop();
 		}
 		puts ("");
	}
    return 0;
}
