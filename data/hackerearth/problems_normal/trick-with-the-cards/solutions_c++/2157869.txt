#include <iostream>
using namespace std;
long long n[100000+1];
int main()
{
	long long t;
    cin>>t;
    
    for(int i=0;i<t;++i)
    {
    	cin>>n[i];
    }
    for(int i=0;i<t;++i)
    {
    	long long a = n[i]/3;
    	long long b = n[i]%3;
    	if(b == 0)
    	{
    		cout<<a<<endl;    		
    	}
    	else if (a==0 && n[i]==0)
    		cout<<0<<endl;
    		
    	else if (a == 0)
    		cout<<1<<endl;

    	else 
    	{
    		cout<<n[i]<<endl;
    	}
    }
    return 0;
}
