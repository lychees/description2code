#include <bits/stdc++.h>
using namespace std;
#define maxn 2*4000001

char s[maxn], ret[maxn];

int main()
{
	int n, i, k, start, comp;
	scanf("%s", s);
	n = strlen(s);
	for(i = n, k = 0; k < n; k++, i++)
		s[i] = s[k];
	s[2*n] = '\0';
	start = 0;
	comp = 1;
	for(i = 0; i + comp < 2*n; i++)
		if(s[comp + i] > s[start])
		{
			start = comp + i;
			comp = start + 1;
			i = -1;
		}
		else
			if(s[comp + i] < s[start + i])
			{
				comp += i + 1;
				i = -1;
			}
			else
				if(s[comp + i] > s[start + i])
				{
					start = comp;
					comp++;
					i = -1;
				}
	for(i = start, k = 0; k < n; k++, i++)
		ret[k] = s[i];
	ret[k] = '\0';
	printf("%s", ret);
	return 0;
}
