#include <bits/stdc++.h>
using namespace std;

int main()
{ int t,i;
cin>>t;
for(i=0;i<t;i++)
{int n,j,input;
	cin>>n;
	vector <int> v;
	for(j=0;j<n;j++)
	{ cin>>input;
		v.push_back(input);
	}
	sort(v.begin(),v.end());
      j=0;
      int k=n-1,current;
     int max=v[0]+v[n-1],min=v[0]+v[n-1];
     while(j<n/2)
     {
    current=v[j]+v[k];
     if(current>max)
     {
     	max=current;
     }
     else if(current<min)
     {
     	min=current;
     }
     j++; k--;
     }
    // cout<<min<<endl;
     cout<< max-min<<endl;








}


    //cout << "Hello World!" << endl;
    return 0;
}
