#include <cstdio>
using namespace std;

int main()
{
    int t, a0, a1, next, c, n, buf, bufn;
    scanf("%d", &t);
    while(t--)
	{
		scanf("%d %d %d %d", &a0, &a1, &c, &n);
		if(n==1)
			printf("%s\n", a0==0?"YES":"NO");
		else{
			buf = a0*10+a1;
			bufn = 2;
			while(bufn<n)
			{
				while(buf<10000000 && bufn<n)
				{
					next = (a1*c+a0)%10;
					buf *= 10;
					buf += next;
					a0 = a1;
					a1 = next;
					bufn++;
				}
				buf = buf % 41;
			}
			printf("%s\n", buf%41==0?"YES":"NO");
		}
	}
    return 0;
}
