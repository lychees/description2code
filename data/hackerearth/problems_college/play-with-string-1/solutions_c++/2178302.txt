#include <iostream>
#include <cmath>
#include<string.h>
using namespace std;
char s[100005];
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
	int	flag=0,i,j;
	cin>>s;
int n=strlen(s);
	for(i=1,j=n-2;i<n,j>=0;i++,j--)
	{
		if(abs(s[i]-s[i-1])==abs(s[j]-s[j+1]))
		{
			flag=1;
		}
		else
		{
			flag=0;
			break;
		}

	}
   		if(flag==1)
		 cout <<"Funny" << endl;
		 else
		 cout<<"Not Funny\n";
	}
    return 0;
}
