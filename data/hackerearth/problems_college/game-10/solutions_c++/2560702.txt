#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,m;
    	cin>>n>>m;
    	vector<int>v(n);
    	if(n%2==0)
    	{
    		int j=n-1;
    		for(int i=0;i<n;i=i+2)
    		{
    			v[i]=j;
    			j--;
    		}
    		j=0;
    		for(int i=1;i<n;i=i+2)
    		{
    			v[i]=j;
    			j++;
    		}
    	}
    	else
    	{
    		int j=n-1;
    		for(int i=0;i<n;i=i+2)
    		{
    			v[i]=j;
    			j--;
    		}
    		j=0;
    		for(int i=1;i<n-1;i=i+2)
    		{
    			v[i]=j;
    			j++;
    		}
    	}
    	for(int i=0;i<n;i++)
    	{
    		if(m==v[i])
    		{
    			cout<<i<<endl;
    			break;
    		}
    	}
    }
}
