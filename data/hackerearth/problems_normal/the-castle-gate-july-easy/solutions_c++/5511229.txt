#include <iostream>
using namespace std;

int main()
{
    int t,n,i,j,count=0;
    cin>>t;
    while(t--)
    {	count=0;
    	cin>>n;
    	for( i=1;i<n;i++)
    	{
    		for(j=i+1;j<=n;j++)
    		{
    			if((i^j)<=n)
    				count++;
    		}
    	}
    	cout<<count<<endl;
    	
    }
    return 0;
}
