#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++){
    	long long int n,m;
    	cin>>n>>m;
    	set<long long int> s;
    	for(int i=0;i<n;i++){
    		long long int x;
    		cin>>x;
    		s.insert(x);
    	}
    	for(int i=0;i<m;i++){
    		long long int x;
    		cin>>x;
    		if(s.find(x)!=s.end()){
    			
    			cout<<"YES\n";
    		}
    		else{
    			s.insert(x);
    			cout<<"NO\n";
    		}
    		
    	}
    }
    return 0;
}
