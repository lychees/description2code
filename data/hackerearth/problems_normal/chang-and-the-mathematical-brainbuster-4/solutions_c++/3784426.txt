#include <bits/stdc++.h>
using namespace std;
int arr[10010];
int n;
short memo[10010][10010];
int DP(int l,int r)
{
	if(l>r)
	return 1e6;
	if(3*arr[l]>=arr[r])
	return 0;
	if(memo[l][r])
	return memo[l][r];
	int x=DP(l+1,r);
	int y=DP(l,r-1);
	if(x!=1e6)x+=1;
	if(y!=1e6)y+=1;
	return memo[l][r]=min(x,y);
}
int main()
{
    cin>>n;
    for(int i=0;i<n;i++)
    cin>>arr[i];
    sort(arr,arr+n);
    cout<<DP(0,n-1)<<"\n";
    return 0;
}
