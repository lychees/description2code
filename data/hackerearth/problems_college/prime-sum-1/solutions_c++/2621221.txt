#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int mem[100007];
int solve(int x)
{
	if(x < 0)return 100000000;
	if(x == 0)return 0;
	if(mem[x] != -1)
		return mem[x];
	int s1 = 1 + solve(x-2);
	int s2 = 1 + solve(x-3);
	int s3 = 1 + solve(x-5);
	int s4 = 1 + solve(x-7);
	return mem[x] = min(s1, min(s2, min(s3, s4)));
}
int main()
{
	int t; cin >> t;
	int n;
	while(t--){
		memset(mem, -1, sizeof mem);
		cin >> n;
		cout << solve(n) << endl;
	}
    return 0;
}
