#include <iostream>
#include <stdio.h>
#include <math.h>
using namespace std;

int main() {
	long long int arr[100000],n,i;
	long long int t;
	
	cin >> t;
	while(t>0) {
		scanf("%lld",&n);
		arr[1] = 1;
	arr[2] = 2;
	arr[3] = 4;
		if(n<=3) {
			cout << arr[n] << "\n";
		}
		else {
			for(i=4;i<=n;i++) {
				arr[i] = (arr[i-1] + arr[i-2] + arr[i-3])%1000000007;
			}
			printf("%lld\n",arr[n]);
		}
		t--;
	}
	return 0;
	
}
