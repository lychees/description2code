#include <iostream>
#include<string>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    string a,b;
    int count=0;
    cin>>a>>b;
    for(int i=0;i<a.length();i++)
    {
      	if(a[i]==b[i])
      	continue;
      	else
      	count++;
    }
    if(count==0 || count==1)
    {
    	cout<<"wins\n";
    }
    else
    {
    	cout<<"loses\n";
    }
    return 0;
}
