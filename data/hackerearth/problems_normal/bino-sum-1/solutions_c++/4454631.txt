#include <bits/stdc++.h>
using namespace std;
#define MOD 1000000007
long long ncr[1001][1001];
void fun()
{
	ncr[0][0]=1;
	ncr[1][0]=1;
	ncr[1][1]=1;
	for(long long i=2;i<1001;i++)
	{
		for(long long j=0;j<=i;j++)
		{
			if(j==0)
				ncr[i][j]=1;
			else
				ncr[i][j]=(ncr[i-1][j]+ncr[i-1][j-1])%MOD;
		}
	}
}
int main()
{
	fun();
	long long t,n,p,sum=0;
	cin>>t;
	while(t--)
	{
		sum=0;
		cin>>n>>p;
		for(long long i=0;i<=p;i++)
		{
			sum=(sum+ncr[n][i])%MOD;
		}
		cout<<sum<<endl;
	}
    return 0;
}
