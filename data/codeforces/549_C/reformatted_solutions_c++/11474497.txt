#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstring>
#include <iomanip>
#include <iostream>
#include <map>
#include <set>
using namespace std;
int n, k, a, x, y, ans;

int main() {
  cin >> n >> k;
  for (int i = 1; i <= n; i++) {
    scanf("%d", &a);
    if (a & 1)
      x++;
    else
      y++;
  }
  if (n == k)
    ans = x & 1;
  else if (k & 1)
    ans = (n - k) & 1 ? x > (n - k) / 2 : x >= n - (n - k) / 2;
  else if ((n - k) & 1)
    ans = x > (n - k) / 2 && x < n - (n - k) / 2;
  else
    ans = 0;
  puts(ans ? "Stannis" : "Daenerys");
}
