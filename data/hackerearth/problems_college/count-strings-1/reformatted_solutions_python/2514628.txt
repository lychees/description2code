from collections import Counter
noc = input()
mod = 1000000007
while noc:
    noc -= 1
    count = 1
    st1 = raw_input()
    st2 = raw_input()
    cst1 = Counter(st1)
    for x in xrange(len(st2)):
        count = ((count % mod)*(cst1[st2[x]] % mod)) % mod
        cst1[st2[x]] -= 1
    print count
