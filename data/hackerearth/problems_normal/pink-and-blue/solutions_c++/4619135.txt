#include<bits/stdc++.h>
using namespace std;
int n,m;
vector<int>g[100010];
int col[100010];
int tcol[100010];
int inv=0;
int vis[100010]={};

void dfs(int u,int colo){
   vis[u]=1;
   int v;
   for(int i=0;i<g[u].size();i++){
      v=g[u][i];
      if(!vis[v]){
         if(tcol[v]==colo){
           inv++;
           tcol[v]=!colo;
         }
         dfs(v,tcol[v]);
      }
   }
}

int main(){
   scanf("%d%d",&n,&m);
   char c;
   for(int i=1;i<n+1;i++){
     scanf(" %c",&c);
     if(c=='B'){
       col[i]=0;
     }
     else{
        col[i]=1;
     }

   }
   int u,v;
    for(int i=0;i<m;i++){
   scanf("%d%d",&u,&v);
        g[u].push_back(v);
       g[v].push_back(u);
   }
   int ans=n+1;

    for(int i=0;i<22;i++){
        for(int j=1;j<n+1;j++){
           vis[j]=0;
           tcol[j]=col[j];
        }
        inv=0;
        if(col[1]==i){
            dfs(1,col[1]);
        }
        else{
            inv++;
            tcol[1]=!tcol[i];
            dfs(1,tcol[i]);
        }
       ans=min(ans,inv);
   }
  printf("%d\n",ans);

}
