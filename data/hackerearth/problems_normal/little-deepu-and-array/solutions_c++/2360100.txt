#include <iostream>
using namespace std;
int main()
{
    long int n,a[100000],m,x;
    cin>>n;
    int i;
    for(i=0;i<n;i++)
    	cin>>a[i];
    cin>>m;
    while (m--){
    	cin>>x;
    	for(i=0;i<n;i++){
    		if(a[i]>x)
    			a[i]-=1;
    	}
    }
    for(i=0;i<n;i++)
    	cout<<a[i]<<" ";
    return 0;
}
