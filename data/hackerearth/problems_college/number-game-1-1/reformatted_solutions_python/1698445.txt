import sys
sys.setrecursionlimit(10**9+7)


def solve(n):
    if n == 1:
        return 0
    answer = 1+solve(n-1)
    if n % 2 == 0:
        answer = min(answer, 1+solve(n//2))
    if n % 3 == 0:
        answer = min(answer, 1+solve(n//3))
    return answer


for __ in range(input()):
    n = input()
    answer = 0
    while n > 1:
        if n % 3 == 0:
            n /= 3
        elif n % 2 == 0:
            n /= 2
        else:
            n -= 1
        answer += 1
    print answer
