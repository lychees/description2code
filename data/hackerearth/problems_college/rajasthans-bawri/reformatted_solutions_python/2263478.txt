def f(s):
    s = s.split()
    s = map(int, s)
    px = s[0]
    py = s[1]
    bx = s[2]
    by = s[3]
    if px >= bx and py >= by:
        return "Possible"
    elif px >= by and py >= bx:
        return "Possible"
    else:
        return "Not Possible"


t = int(raw_input())
for i in range(t):
    s = raw_input()
    print f(s)
