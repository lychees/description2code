t=input()
for j in xrange(t):
	n=input()
	a=[int(x) for x in raw_input().split()]
	if n==0:
		print 'Case %s: %s'%((j+1),0)
	elif n==1:
		print 'Case %s: %s'%((j+1),a[0])
	elif n==2:
		print 'Case %s: %s'%((j+1),max(a[0],a[1]))
	else:
		u=[0]*(n)
		u[0]=a[0]
		u[1]=max(a[0],a[1])
		for i in xrange(2,n):
			u[i]=max(u[i-2]+a[i],u[i-1])
		print 'Case %s: %s'%((j+1),u[n-1])
