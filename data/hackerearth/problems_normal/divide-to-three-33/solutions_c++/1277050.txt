#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<climits>
#include<algorithm>

using namespace std;

int N,temp;
int arr[15];

int ans=INT_MAX;


void findSub(int a,int b,int c,int num){

    if(num==N){
        if(ans>max(max(a,b),c))
            ans=max(max(a,b),c);
    }
    else{
        findSub(a+arr[num],b,c,num+1);
        findSub(a,b+arr[num],c,num+1);
        findSub(a,b,c+arr[num],num+1);

    }

}

int main(){

    //int N;

    scanf("%d",&N);

    for(int i=0;i<N;i++)
        scanf("%d",&arr[i]);

        findSub(0,0,0,0);

        cout<<ans<<endl;

}
