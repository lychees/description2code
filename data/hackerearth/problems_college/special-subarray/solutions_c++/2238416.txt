#include<vector>
#include<stack>

#include<stdlib.h>
#include<math.h>
#include<string>
#include<stdio.h>
using namespace std;
typedef long long int ll;

const int N=100001;
int arr[N],l[N],r[N];
int main(){
	int n,m,i,sex,sex1;
	while(~scanf("%d%d",&n,&m)){
		for(i=1;i<=n;i++)
		scanf("%d",arr+i);
		l[1]=1,r[n]=n;
		for(i=2;i<=n;i++)
		      l[i]=(arr[i]<=arr[i-1])?l[i-1]:i;
		for(i=n-1;i>0;i--)
		      r[i]=(arr[i]<=arr[i+1])?r[i+1]:i;
		while(m--){
			scanf("%d%d",&sex,&sex1);
			puts(r[sex]>=l[sex1]?"Yes":"No");
		}
	}
	return 0;
}
