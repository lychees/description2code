#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int n;
	cin>>n;
	int i,j;
	long long int a[n+1][n+1];
	for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
		cin>>a[i][j];
	}
	long long sum1=0;
	long long sum2=0;
	for(i=n-1;i>=0;i--)
	{
	  sum2 += a[i][n-i-1]; 
	}
	for(i=0;i<n;i++)
	{
		sum1 =sum1 + a[i][i];
	}
	
	
	
	
	cout<<abs(sum1-sum2)<<"\n";
}
