#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n;
	cin>>n;
	map<string,int> map;
	for(int i=0;i<n;i++){
		string s;
		cin>>s;
		sort(s.begin(),s.end());
		map[s] +=1;
	}
	int maxx=-1;
	for(auto i=map.begin();i!=map.end();i++){
		if(i->second>maxx){
			maxx=i->second;
		}
	}
	cout<<maxx<<endl;
    return 0;
}
