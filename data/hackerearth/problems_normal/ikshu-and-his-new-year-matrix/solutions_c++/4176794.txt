#include<bits/stdc++.h>
using namespace std;
#define ll long long int
ll pr[10001],a[10000],dp[10001];
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(false);
    ll i,j,n,k,ans,px,py,p,fg;
    for(i=2;i*i<=10000;i++)
    {
        if(pr[i]==0)
        {
            for(j=i*i;j<=10000;j+=i)
            {
                pr[j]=1;
            }
        }
    }
    p=0;
    for(i=2;i<=10000;i++)
    {
        if(pr[i]==0)
        a[p++]=i;
    }
    cin>>n>>k;
    for(i=1;i<=10000;i++)
    {
        fg=0;
        for(j=0;j<p;j++)
        {
            if(a[j]>=i)
            {
                if((a[j]-i)%k==0)
                {
                    dp[i]=(a[j]-i)/k;
                    fg=1;
                    break;
                }
            }
        }
        if(fg==0)
        dp[i]=INT_MAX;
    }
    /*for(i=1;i<=10;i++)
    cout<<dp[i]<<endl;*/
    ll m[n][n],m1[n][n];
    for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++)
        {
            cin>>m[i][j];
            m1[i][j]=dp[m[i][j]];
        }
    }
    fg=0;
    ans=INT_MAX;
    px=py=INT_MAX;
    for(i=1;i<n-1;i++)
    {
        for(j=1;j<n-1;j++)
        {
            if(m1[i][j]!=INT_MAX&&m1[i-1][j-1]!=INT_MAX&&m1[i+1][j+1]!=INT_MAX&&m1[i+1][j-1]!=INT_MAX&&m1[i-1][j+1]!=INT_MAX)
            {
                if(ans>(m1[i][j]+m1[i+1][j+1]+m1[i-1][j+1]+m1[i-1][j-1]+m1[i+1][j-1]))
                {
                    ans=(m1[i][j]+m1[i+1][j+1]+m1[i-1][j+1]+m1[i-1][j-1]+m1[i+1][j-1]);
                    fg=1;
                    px=i;
                    py=j;
                }
            }
        }
    }
    if(fg==0)
    {
        cout<<"no\n";
    }
    else
    {
        cout<<"yes\n";
        cout<<ans<<"\n";
        cout<<px+1<<" "<<py+1<<"\n";
    }
    return 0;
}
