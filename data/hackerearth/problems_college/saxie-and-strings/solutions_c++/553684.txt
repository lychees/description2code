#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t,i;
    cin>>t;
    while(t--)
    {
    	string a;
    	cin>>a;
    	int flag=0;
    	int l=a.length();
    	for(i=0;i<l;i++)
    	{
    		if(a[i]!=a[l-i-1])
    		flag=1;
    	}
    	if(flag)
    	printf("NO\n");
    	else
    	printf("YES\n");
    }
    return 0;
}
