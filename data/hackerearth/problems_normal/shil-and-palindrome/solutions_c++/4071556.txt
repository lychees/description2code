#include <bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
    string str;
    ll f[26] = {0};
    cin>>str;
    ll len = str.length();
    string str1 = string(len,'0');
    int i;
    ll j;
    for(i=0;i<len;i++)
    	f[str[i]-'a']++;
    if(len%2==0)
    {
    	for(i=0;i<26;i++)
    	{
    		if(f[i]%2!=0)
    		{
    			cout<<-1;
    			return 0;
    		}
    	}
    }
    else
    {
    	ll cnt=0;
    	char x;
    	for(i=0;i<26;i++)
    	{
    		if(f[i]%2!=0)
    		{
    			cnt++;
    			x = char(i + 'a');
    		}
    	}
    	if(cnt != 1)
    	{
    		cout<<-1;
    		return 0;
    	}
    	str1[len/2] = x;
    	f[x-'a']--;
    }
    j = 0;
    for(i=0;i<26;i++)
    {
    	while(f[i]>0)
    	{
    		f[i]-=2;
    		str1[j] = char(i+'a');
    		str1[len-j-1] = char(i+'a');
    		j++;
    	}
    }
    cout<<str1;
    //return 0;
}
