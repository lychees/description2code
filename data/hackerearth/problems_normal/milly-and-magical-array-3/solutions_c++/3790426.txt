#include <iostream>

using namespace std;

#define FORI(init, size) for(int i = init; i < size; ++i)

long long N = 0, X = 0;
long long inputArr[100001] = {0}; 
long long result = 0;

void FindMagicalNumber()
{
	long long additions = 0;
	FORI(2, N + 1)
	{
		if(inputArr[i] <= inputArr[i - 1])
		{
			additions = ((inputArr[i - 1] - inputArr[i]) / X) + 1;
			result += additions;

			inputArr[i] += (additions * X);
		}
	}
}

int main()
{
	int T = 0;
	cin >> T;

	for(int tc = 1; tc <= T; ++tc)
	{
		cin >> N >> X;
		FORI(1, N + 1)
		{
			cin >> inputArr[i];
		}

		FindMagicalNumber();

		cout << result << endl;
		result = 0;
	}

	return 0;
}
