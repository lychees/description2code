#include <bits/stdc++.h>
using namespace std;

int main()
{
	ios::sync_with_stdio(false);
    int t; cin>>t;
    while (t--) {
    	int n; cin>>n;
    	int x = floor(cbrt(n));
    	priority_queue<int> maxlist;
    	map<int,int> maxcount; 
    	for (int i = 1;i<=x;i++) {
    		for(int j=1;j<=x;j++) {
    			int res = i*i*i+j*j*j;
    			if (res < n) {
					maxlist.push(res);
					maxcount[res]++;
					if (i==j) maxcount[res]++;
    			}
    		}
    	}
    	
    	while (!maxlist.empty()) {
    		int i = maxlist.top();
    		if (maxcount[i] >= 4) {
    			
    			cout<<i<<endl; goto next;
    		}
    		maxlist.pop();
    	}
    	cout<<-1<<endl;
    	
    	next: continue;
    }
    return 0;
}
