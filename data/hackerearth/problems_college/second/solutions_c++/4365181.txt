#include<stdio.h>
int main(){
	int n, m, i, h[100001] = {0}, f = 0; scanf("%d%d", &n, &m); int a[n];
	for(i = 0; i < n; i++)	scanf("%d", &a[i]), h[a[i]] = 1;
	if(!(h[m]))	printf("0\n");
		else{
			i = m;
			while(i <= 100000){
				if(!h[i]){f = 1, printf("%d\n", i - m); break;}
				i += m;
			}
			if(f == 0)	printf("%d\n", i - m);
		}
	return 0;
}
