#include <bits/stdc++.h>
using namespace std;

int h[10000000];

int main()
{
    int n,q,k;

    cin >> n >> q;
  vector<int> s;
  s.resize(n+1, 0);
int i,j;
  for( i=2; i <= n; i+=2 )  { s[i]=1; }

  for( i=3; i <= n; i+=2 )  {
      if( s[i]==0 )  {
          for( j=i; j <= n; j+=i ) { s[j]+=1; }
      }
  }
  for( i=2; i <= n; ++i )  {
      h[s[i]]++;
  }
    while(q--)
    {
    	cin >> k;
    	if(k>n)
    	cout << "0\n";
    	else
    	cout << h[k] << "\n";
    }
    return 0;

}
