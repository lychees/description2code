#include<bits/stdc++.h>
using namespace std;
#define max 1000005
int arr[max];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		for(int i=0;i<n;i++)
			cin>>arr[i];
		int ans=1;
		int maxi=arr[n-1];
		for(int i=n-2;i>=0;i--)
		{
			if(arr[i]>maxi)
				ans++,maxi=arr[i];
		}
		cout<<ans<<endl;
	}
	return 0;
}
