#include<cstdio>

using namespace std;
typedef long long int ll;
int N,i,count,j,K;
int isPrime(ll K){
	for(j=2;j<K;j++){
		if(K%j==0)
		   return 0;
	}
	return 1;
}
int main()
{



		scanf("%d",&N);
		if(N==1){
			printf("2\n");
		}
		else{
		count=1;
		i=3;
		while(i){
			if(isPrime(i))
			   count++;
			if(count==N){
				printf("%lld\n",i);
				break;
			}  
			i+=2;
		}
	   

}
}
