t=input()

while(t):
    d={}
    l=[]
    n=input()
    x=map(int,raw_input().strip().split(" "))
    for i in x:
        if i not in d:
            d[i]=1
        else:
            d[i]+=1
    m=max(d.values())
    for i in d.keys():
        if d[i]==m:
            l.append(i)
    l=sorted(l,reverse=True)
    print " ".join(map(str,l))
    t-=1
