#include<bits/stdc++.h>
using namespace std;

int main()
{
	long long int t;
	
	cin>>t;
	
	for(;t>0;t--)
	{
		long long int n,m,val,ans=0;
		scanf("%lld %lld",&n,&m);
		val = (m-1) % n;
        ans = (m-1)/n;
        val /= 2;
        val += 1;
        if(m%2 == 0)
            val = n - val + 1;
        printf("%lld %lld\n",val,ans);
	} 
	return 0;
}
