LL dp[20][20];
void PreProc(string s)
{
	int i,j,q,n;
	LL num;
	n=s.size();
	for (i=0; i<20; i++)
		for (j=0; j<20; j++)
			dp[i][j]=-1;
	for (i=0; i<n; i++)
	{
		for (j=i; j<n; j++)
		{
			if (j-q+1>13) continue;
			if (s[i]=='0' && j!=i) continue;
			num=0;
			for (q=i; q<=j; q++)
			{
				num=num*10+s[q]-'0';
			}
			dp[i][j]=num;		
		}
	}
}
LL get(int l, int r)
{
	if (dp[l][r]<=(LL)(1000)*INF) return dp[l][r];
	return -1;
}
void Solve()
{
	string s;
	cin >> s;
	PreProc(s);
	int n,i1,i2,i3,i4;
	LL a,b,c,d,answer=-1;
	n=s.size();
	for (i1=1; i1<=min(n-3,13); i1++)
	{
		for (i2=1; i2<=min(n-i1-2,13); i2++)
		{
			for (i3=1; i3<=min(n-i1-i2-1,13); i3++)
			{
				i4=n-i1-i2-i3;
				a=get(0,i1-1);
				b=get(i1,i1+i2-1);
				c=get(i1+i2,i1+i2+i3-1);
				d=get(i1+i2+i3,i1+i2+i3+i4-1);
				//cout << a << " " << b << " " << c << " " << d << "\n";
				if (a!=-1 && b!=-1 && c!=-1 && d!=-1) answer=max(answer,a+b+c+d);
			}
		}
	}
	if (answer!=-1) cout << answer << "\n";
	else cout << "unlucky\n";
}
int main()
{
	int q,t;
	cin >> t;
	for (q=0; q<t; q++)
	{
		Solve();
	}
	return 0;
}
