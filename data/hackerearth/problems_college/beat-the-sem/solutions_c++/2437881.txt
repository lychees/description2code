#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
    ll t,s,d,x,y,r1,r2;
    cin>>t;
    while(t--)
    {
        cin>>s>>d;
        x=(s+d)/2;
        r1=(s+d)%2;
        y=(s-d)/2;
        r2=(s-d)%2;
        if(r1!=0||r2!=0)
            cout<<"impossible\n";
        else if(y<0)
            cout<<"impossible\n";
        else
        {
            if(x>y)
                cout<<x<<" "<<y<<'\n';
            else
                cout<<y<<" "<<x<<'\n';
        }
    }
    return 0;
}
