#include <bits/stdc++.h>
using namespace std;
#define mp make_pair
#define pb push_back
bool comp(pair <int, int> a, pair <int ,int> b) {
	if(a.first < b.first) {
		return true;
	}
	else if(a.first == b.first) {
		if(a.second > b.second) {
			return true;
		}
	}
	return false;
}
int main()
{
    int t;
    cin >> t;
    while(t--) {
    	int n,a,b;
    	vector<pair <int, int> > v;
    	cin >> n;
    	for(int i =0 ; i < n;i++) {
    		cin >>a >>b;
    		v.pb(mp(a,1));
    		v.pb(mp(b-1,-1));
    	}
    	sort(v.begin(), v.end(),comp);
    	int ans = 0;
    	int maxa = -1;
    	for(int i = 0; i < v.size(); i++) {
    		ans += v[i].second;
    		maxa = max(maxa, ans);
    	}
    	cout << maxa << endl;
    }
    return 0;
}
