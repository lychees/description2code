#include <iostream>
using namespace std;

int expand(string s, int i, int j) {
	int ret =0;
	while(i>=0 && j<s.size()) {
		if(s[i]==s[j]) ret++;
		else break;
		i--;j++;
	}
	return ret;
}

int main()
{
	string s;
	cin >> s;
	int cnt = 0;
	for (int i=0;i<s.size();i++) {
		cnt += expand(s,i,i);
		cnt += expand(s,i,i+1);
	}
	
    cout << cnt << endl;
    return 0;
}
