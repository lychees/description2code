#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    
    while(t--)
    {
    	int k;
    	cin >> k;
    	int u=1,l=1,n=1,f=1,ut,lt,nt,ft;
    	while(k>1)
    	{
    		ut = u; lt = l; nt = n; ft = f;
    		
    		u = lt+nt;
    		l = ut+nt;
    		n = ut+lt+nt+ft;
    		f = nt;
    		
    		k--;
    	}
    	
    	cout << u+l+n+f << endl;
    	
    }
    
    
    return 0;
}
