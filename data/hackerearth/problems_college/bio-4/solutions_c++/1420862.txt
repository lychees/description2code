using namespace std;

vector<int> c = {1,2,5,10,20,50,100,500,1000};

int main(){
	int T;
	cin >> T;
	while(T--){
		string a,b;
		cin >> a >> b;
		int n = a.size();
		int m = b.size();
		int ans = 1<<30;
		for(int i=0; i<=n-m; i++){
			int tmp = 0;
			for(int j=0; j<m; j++){
				if(a[j+i] != b[j]){
					tmp++;
				}
			}
			ans = min(ans, tmp);

		}
		cout << ans << endl;
	}
	return 0;
}
