#include <iostream>
#define mod 1000000007
#define ll long long 
using namespace std;
ll power(ll pro,ll res){
	ll result=1;
	while(res>0){
		if(res&1) result=(result*pro)%mod;
		pro=(pro*pro)%mod;
		res=res>>1;
	}
	return result;
}
int gcd(int a,int b){
	if(a==0) return b;
	return gcd(b%a,a);
}
int main()
