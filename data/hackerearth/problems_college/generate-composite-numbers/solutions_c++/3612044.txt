#include <bits/stdc++.h>
using namespace std;

bool prime[10005];
void abc(int n){
	memset(prime,true,n+1);
	int p,i;
	
	for(p=2;p*p<=n;p++){
		
		if(prime[p]==true){
			for(i=p*2;i<=n;i+=p){
				prime[i]=false;
			}
		}
	}
}

int main()
{
    int n;
    cin>>n;
    
    abc(n);
   	for(int i=2;i<=n;i++){
   		if(!prime[i]) cout<<i<<" ";
   	}
   	cout<<endl;
    return 0;
}
