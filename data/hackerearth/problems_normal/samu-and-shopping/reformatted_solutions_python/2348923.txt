    n = int(raw_input())

    l = []
    for i in range(n):
        temp = raw_input().split(' ')
        for i in range(3):
            temp[i] = int(temp[i])
        l.append(temp)
    memo = l[0][:]
    for i in range(1, n):
        memo2 = [0]*3
        memo2[0] = l[i][0] + min(memo[1], memo[2])
        memo2[1] = l[i][1] + min(memo[0], memo[2])
        memo2[2] = l[i][2] + min(memo[1], memo[0])
        memo = memo2[:]
