#include<bits/stdc++.h>
using namespace std;
int main()
{
	int d,f,t;
	long long n;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%lld",&n);
		f=0;
		while(n)
		{
			d=n%10;
			n=n/10;
			if(d==4||d==7)
				f++;
		}
		if(f==4||f==7)
			printf("YES\n");
		else
			printf("NO\n");
	}
	return 0;
}
