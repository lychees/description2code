#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	n%=26;
    	string s;
    	cin>>s;
    	for(int i=0;i<s.size();++i)
    	{
    	  if(s[i]+n>'z'&&s[i]>='a' && s[i]<='z')
    	   {s[i]=s[i]-26+(n);
    	    //printf("1%c",s[i]);
    	    }
    	  else if(s[i]+n>'Z' && s[i]>='A' && s[i]<='Z')
    	   {s[i]=s[i]-26+(n);
    	   //printf("2%c",s[i]);
    	   }
    	   else s[i]+=n;
    	}
      printf("%s\n",s.c_str());
    }
    return 0;
}
