#include <iostream>
using namespace std;

int CanBePalindrome(char *str)
{
	int count[26] = { 0 };

	for (int i = 0; str[i] != NULL; i++)
	{
		count[str[i] - 'a']++;
	}

	int oddCount = 0;
	//Check Counts
	for (int i = 0; i < 26; i++)
	{
		if ((count[i] % 2) != 0 && (oddCount++) == 1)
			return false;
	}

	return true;
}

int main()
{
	char str[100001];

	cin >> str;
	if (CanBePalindrome(str))
		cout << "YES";
	else
		cout << "NO";
	return 0;
}
