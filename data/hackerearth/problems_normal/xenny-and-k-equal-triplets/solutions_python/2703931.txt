from fractions import Fraction
def f(n):
	return max(n,1)
t=int(raw_input(''))
for j in range(t):
	s=raw_input('')
	a=s.split(" ")
	n,k=int(a[0]),int(a[1])
	s=raw_input('')
	a=s.split(" ")
	c=0
	for i in range(n):
		if int(a[i])==k:
			c+=1

	tCom=(f(n)*f(n-1)*f(n-2))//6
	#print tCom,n
	mCom=(f(c)*f(c-1)*f(c-2))//6
	#print mCom,c

	if mCom==0:
		print "0/1"
	elif mCom==tCom:
		print "1/1"
	else:
		print Fraction(mCom,tCom)
