#include<bits/stdc++.h>
using namespace std;

int main()
{
	long long int i,count,t,n,x;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
		count=0;
		x=1;
		while(x<n)
		{
			x=x<<1;
			x+=1;
			if(x<=n)
			count++;
			//printf("%lld ",x);
		}
		printf("%lld\n",count);
	}
	return 0;
}
