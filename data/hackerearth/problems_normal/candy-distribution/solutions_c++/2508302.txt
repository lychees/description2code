const ll mod=1000000007;
const ld pi = 3.14159265358979323846264338327950288419716939937510;
ll po(ll a,ll b) {ll res=1%mod;a%=mod;for(;b;b>>=1){if(b&(1LL))res=res*a%mod;a=a*a%mod;}return res;}
const ld eps = 1e-10;
const int oo = 1000000009;

map<ll,int> ans;
bool isPrime(ll n){
    if(n==2 || n==3) return true;
    for(ll i=2 ; i*i<=n ; i++){
        if(n%i==0) return false;
    }
    return true;
}
int solve(ll n){
    if(n==1|| n==2) return 1;
    if(ans.find(n)!=ans.end()) return ans[n];
    if(isPrime(n-1)){
        ans[n]=1;
        return 1;
    }
    ll ret=1;
    n--;
    ll cnt=0;
    int lm=sqrt((double)n);
    for(int i=2 ; i<=lm ; i++){
        if(n%i==0){
            cnt=n/i;
            ret=ret+po(solve(i),cnt);
            ret%=mod;
            if(n/i != i){
                cnt=n/(n/i);
                ret=ret+po(solve(n/i),cnt);
                ret%=mod;
            }
        }
    }
    ans[n+1]=ret;
    return ret;
}
