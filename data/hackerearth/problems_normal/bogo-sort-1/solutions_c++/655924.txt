#include <iostream>
#include <vector>
#include <algorithm>

int main()
{
	int n, t;
	std::vector<int> v;
	std::cin >> n;
	
	while (n--)
	{
		std::cin >> t;
		v.push_back(t);
		
		std::sort(v.begin(), v.end());
	}
	
	for (std::vector<int>::iterator it = v.begin(); it != v.end(); ++it)
	{	
		std::cout << *it << std::endl;
	}
	
    return 0;
}
