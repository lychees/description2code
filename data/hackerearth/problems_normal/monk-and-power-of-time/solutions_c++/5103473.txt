#include <algorithm>
#include <iostream>
#include <queue>
#include <vector>
using namespace std;

int main() {
 	queue <int> cal;
 	queue <int> cal2;
 	int n;
 	cin>>n;
	int i;
 	int element;
 	i=0;
 	while(i<n)
 	{	
  		cin>>element;
  		cal.push(element);
  		i++;
 	}
 	i=0;
 	while(i<n)
 	{
  		cin>>element;
  		cal2.push(element);
  		i++;
 	}
 		int time = 0;
 	while(!cal.empty() && !cal2.empty())
 	{
  		if(cal.front() != cal2.front())
  		{
   			int d = cal.front();
   			cal.pop();
   			cal.push(d);
   			time++;
  		}
  		else
  		{
   			cal.pop();
   			cal2.pop();
   			time++;
  		}
 	}
 	cout<<time;
}
