#include <iostream>
using namespace std;

int max(int a,int b)
{
	if(a>=b)
		return a;
	else
		return b;
}		
		
int main()
{
   int i,j,n,mat[10][10]={0},aux[10][10]={0};
   cin>>n;
   for(i=0;i<n;i++)
   {
   		for(j=0;j<n;j++)
   			cin>>mat[i][j];
   }
   aux[0][0]=mat[0][0];
   for(i=0;i<n;i++)
   {
   		for(j=i+1;j<n;j++)
   		{
   			if(i==0)
   				aux[i][j]=mat[i][j]+aux[i][j-1];
   			else if(j==i+1)
   				aux[i][j]=mat[i][j]+aux[i-1][j];
   			else
   				aux[i][j]=mat[i][j]+max(aux[i-1][j],aux[i][j-1]);
   		}
   }
   aux[i-1][j-1]=mat[i-1][j-1]+aux[i-2][j-1];
   for(i=n-1;i>=0;i--)
   {
   		for(j=i-1;j>=0;j--)
   		{
   			if(i==n-1)
   				aux[i][j]=mat[i][j]+aux[i][j+1];
   			else if(j==i-1)
   				aux[i][j]=mat[i][j]+aux[i+1][j];
   			else
   				aux[i][j]=mat[i][j]+max(aux[i+1][j],aux[i][j+1]);
   		}
   }
   aux[i+1][j+1]=mat[i+1][j+1]+aux[i+2][j+1];
   cout<<aux[0][0];
    return 0;
}
