#include <bits/stdc++.h>

using namespace std;

int arr[100];
int DP[100];

void toBinary(int n) {
	vector<int> v;
	while(n > 0) {
		int j = n % 2;
		v.push_back(j);
		n = n / 2;
	}
	int size = v.size();
	for(int i = size - 1; i >= 0; i --) {
		cout << v[i];
	}
}

void LIS(int n) {
	for(int i = 0; i < n; i ++) {
		DP[i] = 1;
	}
	for(int i = 0; i < n; i ++) {
		for(int j = 0; j < i; j ++) {
			if(arr[j] < arr[i] && DP[j] + 1 > DP[i]) {
				DP[i] = DP[j] + 1;
			}
		}
	}
	int max = -1;
	for(int i = 0; i < n; i ++) {
		if(DP[i] > max) {
			max = DP[i];
		}
	}
	toBinary(max);
}

int main() {
    int n;
    cin >> n;
    for(int i = 0; i < n; i ++) {
    	cin >> arr[i];
    }
    LIS(n);
    return 0;
}
