
#include <bits/stdc++.h>

typedef long long ll;
using namespace std;

#define INF 1000000
set<string> S;
int k;

void f(string s, int n, int a, int h) {
  if (k > INF) {
    return;
  }
  string t = s;
  int c = 0;
  for (int i = 1; i < n; i += 2)
    t[i] = (t[i] - '0' + a) % 10 + '0';
  if (S.find(t) == S.end()) {
    S.insert(t);
    c += 1;
    k++;
    f(t, n, a, h);
  }
  t = s;
  h %= n;
  for (int i = 0; i < n; i++)
    t[(i + h) % n] = s[i];
  if (S.find(t) == S.end()) {
    S.insert(t);
    k++;
    c += 1;
    f(t, n, a, h);
  }
  if (c == 0)
    return;
}

int main() {
  ios_base::sync_with_stdio(false);
  cin.tie(NULL);
  int t, a, h;
  string s;
  cin >> t;
  while (t--) {
    cin >> s >> a >> h;
    S.clear();
    S.insert(s);
    f(s, (int)s.size(), a, h);
    cout << (*S.begin()) << '\n';
  }
  return 0;
}
