#include<bits/stdc++.h>
using namespace std;
int main()
{
    long int i,j,n,k,p,tc;
    cin>>tc;
    while(tc--)
    {
        long int a[100];
        cin>>k>>n>>p;
        if(k>=p)
            cout<<n<<endl;
        else {
                int l=1;
            for(i=1;i<=k;i++)
                 a[i]=n;
            a[i]=k*n;
                k++;
                i++;
            while(k!=p)
            {
                a[i]=a[i-1]+a[i-1]-a[l];
                //cout<<a[i]<<" ";
                i++, l++;
                k++;
            }
            cout<<a[i-1]<<endl;
        }
    }
    return 0;
}
