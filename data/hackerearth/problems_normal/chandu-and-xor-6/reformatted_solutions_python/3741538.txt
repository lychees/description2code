t = raw_input()
l = t.split()
n = int(l[0])
m = int(l[1])
matrix = list()
row_sum_list = list()
sum_coulumns_list = list()
for i in range(1, n+1):
    sm = raw_input()
    lm = sm.split()
    lm = map(int, lm)
    lm[:m]
    matrix.append(lm)

for y in matrix:
    sum_rows = sum(y)
    row_sum_list.append(sum_rows)

for x in range(m):
    sum_columns = 0
    for y in matrix:
        try:
            sum_columns = sum_columns + y[x]
        except BaseException:
            print x, y, 'error', m
    sum_coulumns_list.append(sum_columns)

big = None

for x in row_sum_list:
    for y in sum_coulumns_list:
        a = x ^ y
        if big is None:
            big = a
        elif (a > big):
            big = a

print big
