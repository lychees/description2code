using namespace std;
typedef long long ll;

#define pb push_back

int main(){
    ios::sync_with_stdio(false);
    #ifdef DEBUG
    freopen("input.txt", "r", stdin);
    #endif // DEBUG

    int T;
    cin>>T;
    int N;
    ll M;
    while(T--){
        cin>>N>>M;
        ll A[N];
        for(int i=0;i<N;i++)
            cin>>A[i];
        sort(A, A+N);
        int ans=0;
        ll sum=0;
        if(M==0){
            cout<<0<<endl;
            continue;
        }
        for(int i=N-1;i>=0;i--){
            sum += A[i];
            ans++;
            if(sum >= M)
                break;
        }
        cout<< ((sum>=M)?ans:-1) <<endl;
    }
 
    return 0;
}
