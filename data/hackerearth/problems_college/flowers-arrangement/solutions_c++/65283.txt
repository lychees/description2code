bool process(string s){
	if(s.length() == 0)return true;
	bool flag = false;
	if(s.substr(0,1).compare("R") == 0){
		flag = flag || process(s.substr(1,s.length()-1));
		if(flag) return true;
	}
	if(s.substr(0,2).compare("RY") == 0){
		flag = flag || process(s.substr(2,s.length()-2));
		if(flag) return true;
	}
	if(s.substr(0,3).compare("RYY") == 0){
		flag = flag || process(s.substr(3,s.length()-3));
		if(flag) return true;
	}
	return false;
}
int main()
{
	int t;
	scanf("%d", &t);
	while(t-- > 0){
		string s;
		cin>>s;

		if(process(s))printf("YES\n");
		else printf("NO\n");
	}
	return 0;

}
