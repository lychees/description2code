using namespace std;
int main(){
    int t;
    cin>>t;
    while(t--)
        {

            int n;
            scanf("%d", &n);
            int a[n+1];
            int i;

            for(i=0; i<n; i++)
            {
                scanf("%d", &a[i]);
            }
            int left = 0, right = 0;

            for(i=0; i<n; i++)
            {
                if(a[i] == -1)
                {
                    left ++;
                }
                else
                {
                    break;
                }
            }

            for(i=n-1; i>=0; i--)
            {
                if(a[i] == 1)
                {
                    right++;
                }
                else
                {
                    break;
                }
            }


            int leftgoingants = 0;
            for(i=left; i <= n-right-1; i++){
                if(a[i] == -1){
                    leftgoingants ++;
                }
            }
            if(left >= right){
                printf("%d\n", left + leftgoingants);
            }
            else{
                printf("%d\n", left + leftgoingants + 1);
            }

    }
	return 0;
}
