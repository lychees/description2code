

#include <iostream>
using namespace std;
    long long gcd(long long p,long long q)
    {
    	if(q==0)
    	return p;
    	gcd(q,p%q);
    }
    
    int main()
    {
    long long int A,B,C,P,Q,x;
    cin>>A>>B>>C;
    P=C*C;
    Q=2*A*B;
    if(A+B<=C)
    {
    P=Q=1;
    }
    else
    {
    if(A<C) P-=(C-A)*(C-A);
    if(B<C) P-=(C-B)*(C-B);
    x=gcd(P,Q);
    P/=x;
    Q/=x;
    }
    cout <<P<<"/"<<Q<< endl;
    return 0;
    }
