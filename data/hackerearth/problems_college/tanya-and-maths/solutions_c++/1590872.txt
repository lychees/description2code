int gcd(int a, int b)
{
    while(b) b ^= a ^= b ^= a %= b;
    return a;
}

int totient(int n)
{
    int i, ans = n;

    for (i=2;i*i<=n;i++)
    {
        if (n%i==0)
            ans-=ans/i;

        while(n%i==0)
            n/=i;
    }

    if (n!=1)
        ans-=ans/n;
    return ans;
}

int main()
{
    int t;
    int a,b,sum,i;
    cin>>t;
    while(t--)
    {
        cin>>a>>b;
        int mi = min(a,b);
        sum=0;
        for(i=1;i<=mi;++i)
        {
            if(a%i==0 && b%i==0)
                sum += totient(i);
        }
        cout<<sum<<"\n";
    }
    return 0;
}
