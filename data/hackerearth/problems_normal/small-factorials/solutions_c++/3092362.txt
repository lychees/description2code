#include<iostream>
using namespace std;
void fact(int x)
	{
		int f[1000];
		int i,len=1,carry=0,j;
                f[0]=1;
		for(i=1;i<=x;++i)
		{
			for(j=0;j<len;++j)
			{
				int t;
                                t=carry+i*f[j];
				carry=t/10;
				f[j]=t%10;
			}
			while(carry!=0)
			{
				f[j]=carry%10;
				carry/=10;
				++j;
			}
			len=j;
		}
		for(i=len-1;i>=0;--i)
		cout<<f[i];
		cout<<endl;
	}
int main()
{
	int a,i,N;
	cin>>N;
	for (i = 0; i < N; i++) {
            cin>>a;
            fact(a);
        }
	return 0;
}
