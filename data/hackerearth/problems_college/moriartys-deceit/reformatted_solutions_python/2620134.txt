from fractions import gcd

test = int(raw_input())
for no in xrange(0, test):
    meas = map(int, raw_input().split(" "))
    mem = map(int, raw_input().split(" "))
    # print meas
    # print mem
    ans_list = []
    for x in mem:
        if gcd(x, meas[1]) == 1:
            ans_list.append(str(x))
    if ans_list:
        print ' '.join(ans_list)
    else:
        print -1
