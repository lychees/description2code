def mssl(l):
    best = cur = 0
    for i in l:
        cur = max(cur + i, 0)
        best = max(best, cur)
    return best


n = int(raw_input())
a = []
while len(a) != n:
    a += map(int, raw_input().split())

print mssl(a)
