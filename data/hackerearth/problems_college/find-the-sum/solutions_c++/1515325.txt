#include<bits/stdc++.h>

using namespace std;

int main(){
    long long int t,l,r,start,end,sum1,sum2,odd1,odd2,i1,i2;
    cin>>t;
    while(t--){
        sum1=0;
        cin>>i1>>i2;
        l=1;
        r=i1-1;
        if(r<1){
            sum1=0;
            goto end1;
        }
        if(l>r) swap(l,r);
        start=ceil((sqrt(1+8*l)-1)/2);
        end=ceil((sqrt(1+8*r)-1)/2);
        //cout<<start<<" "<<end<<"\n";
        if(start==end){
            if(start%2==1){
                if((r-l)%2==0) sum1=start;
                else if((r-l)%2==1) sum1=0;
            }
            else if(start%2==0){
                if((r-l)%2==0) sum1=(-start);
                else if((r-l)%2==1) sum1=0;
            }
            //cout<<sum1<<"\n";
            goto end1;
        }
        if(start%2==0){
            if((start%4==2)&&(l%2==1)) sum1+=(-start);
            else if((start%4==0)&&(l%2==0)) sum1+=(-start);
        }
        else if(start%2==1){
            if((start%4==3)&&(l%2==0)) sum1+=(start);
            else if((start%4==1)&&(l%2==1)) sum1+=(start);
        }
        if(end%2==0){
            if((end%4==2)&&(r%2==0)) sum1+=(end);
            else if((end%4==0)&&(r%2==1)) sum1+=(end);
        }
        else if(end%2==1){
            if((end%4==3)&&(r%2==0)) sum1+=(end);
            else if((end%4==1)&&(r%2==1)) sum1+=(end);
        }
        //cout<<sum1<<"\n";
        if(start%2==0) odd1=start+1;
        else if(start%2==1) odd1=start+2;
        if(end%2==0) odd2=end-1;
        else if(end%2==1) odd2=end-2;
        odd1=(odd1-1)/2;
        odd2=(odd2+1)/2;
        if(((odd2*odd2)-(odd1*odd1))>0) sum1+=(((odd2*odd2)-(odd1*odd1)));

        end1:
        sum2=0;
        l=1;
        r=i2;
        if(l>r) swap(l,r);
        start=ceil((sqrt(1+8*l)-1)/2);
        end=ceil((sqrt(1+8*r)-1)/2);
        //cout<<start<<" "<<end<<"\n";
        if(start==end){
            if(start%2==1){
                if((r-l)%2==0) sum2=start;
                else if((r-l)%2==1) sum2=0;
            }
            else if(start%2==0){
                if((r-l)%2==0) sum2=(-start);
                else if((r-l)%2==1) sum2=0;
            }
            //cout<<sum2<<"\n";
            goto end2;
        }
        if(start%2==0){
            if((start%4==2)&&(l%2==1)) sum2+=(-start);
            else if((start%4==0)&&(l%2==0)) sum2+=(-start);
        }
        else if(start%2==1){
            if((start%4==3)&&(l%2==0)) sum2+=(start);
            else if((start%4==1)&&(l%2==1)) sum2+=(start);
        }
        if(end%2==0){
            if((end%4==2)&&(r%2==0)) sum2+=(end);
            else if((end%4==0)&&(r%2==1)) sum2+=(end);
        }
        else if(end%2==1){
            if((end%4==3)&&(r%2==0)) sum2+=(end);
            else if((end%4==1)&&(r%2==1)) sum2+=(end);
        }
        //cout<<sum2<<"\n";
        if(start%2==0) odd1=start+1;
        else if(start%2==1) odd1=start+2;
        if(end%2==0) odd2=end-1;
        else if(end%2==1) odd2=end-2;
        odd1=(odd1-1)/2;
        odd2=(odd2+1)/2;
        if(((odd2*odd2)-(odd1*odd1))>0) sum2+=(((odd2*odd2)-(odd1*odd1)));
        end2:
        //cout<<sum1<<" "<<sum2<<"\n";
        cout<<(sum2-sum1)<<"\n";
    }
return 0;
}
