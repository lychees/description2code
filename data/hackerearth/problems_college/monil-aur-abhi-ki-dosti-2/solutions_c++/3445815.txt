#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
typedef vector<int> vi;
typedef pair<int,int> pii;
typedef map<int,int> mi;

#define si(a) scanf("%d",&a)
#define sii(a,b) scanf("%d %d",&a,&b)
#define nl printf("\n");
#define pi(a) printf("%d\n",a)
#define pb push_back
#define mp make_pair
#define all(c) (c).begin(),(c).end()
#define f(i,a,b) for(i=a;i<b;i++)
#define rf(i,a,b) for(i=a;i>=b;i--)
#define clr(x,a) memset(x,a,sizeof(x))
#define MAX 100010
#define MOD 1000000007

int n,m,a[50],b[50],mk[MAX][28];

mi mm;

int main(){
    int r,k,i,c=0,x=0,y=0,j,t,l,z,x1=0,y1=0,flag;
    ll ans=-1;string p,q;

    si(t);
    while(t--){
        cin>>p>>q;
        clr(mk,0);
        clr(b,0);
        n=p.size();m=q.size();
        f(i,1,m+1){
            k=q[i-1]-'a';
            f(j,0,26){
                mk[i][j]=mk[i-1][j];
            }
            mk[i][k]++;
        }
        f(i,0,n){
            b[p[i]-'a']++;
        }
        flag=0;
        f(i,n,m+1){
            clr(a,0);
            f(j,0,26){
                a[j]=mk[i][j]-mk[i-n][j];
            }
            c=0;
            f(j,0,26){
                if(a[j]!=b[j])
                    c=1;
            }
            if(!c)
                flag=1;
        }
        if(flag)
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
    }

    return 0;
}
