#include<iostream>
int main()
{
	int n,i,j;
	std::cin>>n;
	for(i=n;i>=1;i--){
		if(i==1 || i==n)
		{
			for(j=1;j<=n;j++)
			std::cout<<"*";
			std::cout<<"\n";
		}
		else{
			for(j=1;j<=i;j++)
			{
				if(i==j)std::cout<<"*\n";
				else std::cout<<" ";
			}
		}
	}
	return 0;
}
