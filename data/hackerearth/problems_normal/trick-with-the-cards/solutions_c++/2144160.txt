#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	long long int n;
    	scanf("%lld",&n);
    	
    	if(n==0)printf("0\n");
    	else if(n<=3)printf("1\n");
    	else {
    		//n++;
    		
    		printf("%lld\n",n%3==0?n/3:n);
    	}
    	
    	
    }
}
