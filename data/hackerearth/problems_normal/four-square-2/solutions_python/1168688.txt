def sumdiv(n):
	suma=0
	#print "n",n
	for i in xrange(1,int(n**0.5)+1):
		#print "i",i
		if n%i==0:
			if i%4!=0:
				suma+=i
			if (n/i)%4!=0:
				suma+=(n/i)
	i=int(n**0.5)
	if ((i*i)==n and i%4!=0):
		suma-=i
	return int(suma)
