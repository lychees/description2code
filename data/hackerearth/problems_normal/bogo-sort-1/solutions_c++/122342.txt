#include <iostream>
#include<stdlib.h>
using namespace std;


int n,i,arr[1000006];
  	
int cmp(const void*x,const void*y)
{
	return *(int*)x - *(int*)y;
}


int main()
{
  	cin>>n;
  	
  	for(i=0;i<n;i++)
  		cin>>arr[i];
  		
  	qsort(arr,n,sizeof(int),cmp);
  	
  	for(i=0;i<n;i++)
  		cout<<arr[i]<<endl;
  	
    return 0;
}
