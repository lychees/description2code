#print 'Hello World!'
from math import sqrt
t=input()
while t>0:
	t-=1
	x1, y1 ,x2 ,y2 =map(int,raw_input().split())
	r=input()
	x=(x2-x1)
	y=(y2-y1)
	if x==0 and y==0:
		print "REPLANNING"
		continue
	d=(abs(-y1*x+x1*y)*1.0)/(sqrt(x**2+y**2))
	if d==r:
		print"JUST MISSED"
	elif d>r:
		print"SAFE"
	else:
		print"REPLANNING"
