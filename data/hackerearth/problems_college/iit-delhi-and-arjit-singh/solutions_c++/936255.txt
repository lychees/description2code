#include <bits/stdc++.h>
using namespace std;
#define Lim 1000005
#define f_in(st) freopen(st,"r",stdin);
#define f_out(st) freopen(st,"w",stdout);
long long etf[1000010];
void cal_etf()
{
    for(int i = 1 ;i <= Lim ; i++ )
    	etf[i]=i;
    for(int i=2 ; i<= Lim ; i++ )
        if ( etf[i] == i)
            for(int j = 2*i ; j <= Lim ; j += i)
            	etf[j] -=  etf[j]/ i ;
    etf[1]=1;
    for(int i=2 ; i<= Lim ; i++)
    {	if ( etf[i]==i)//if that number is prime
    		etf[i] = i-1;
    	etf[i]+=etf[i-1];
    }
}

int main() {
	cal_etf();
	//f_in("in4.txt");
	//f_out("out4.txt");
	int t;
	cin>>t;
	while(t--)
	{
		int n,x;
		cin>>n>>x;
		n=n/x;
		cout<<etf[n]<<endl;
	}
	return 0;
}
