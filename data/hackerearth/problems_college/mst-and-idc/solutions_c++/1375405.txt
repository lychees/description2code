#include<bits/stdc++.h>
using namespace std;
int a[10000000];


int main() {
	int t;
	int n;
	cin>>t;
	while(t--) {
		cin>>n;
		for(int i = 0; i < n ; i++) cin>>a[i];
		sort(a, a+ n);
		int ans = 0;
		for(int i = 1; i < n; i++) {
			ans += (a[i]-a[i-1]);
		}
		cout<<2*ans<<"\n";
	}
	return 0;
}
