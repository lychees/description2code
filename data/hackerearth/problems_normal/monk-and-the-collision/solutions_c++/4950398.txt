#include <stdio.h>
#include <map>
#include <iostream>
#define ll long long int
#define cin(a) scanf("%lld",&a)
#define cout(a) printf("%lld\n",a)
using namespace std ;
ll hash_table[102],n,sum ;
ll has(ll x) {
	return x%10 ;
}
void init () {
	ll i ;
	for (i=0;i<100;i++)
		hash_table[i]=0 ;
	sum=0 ;	 
}
int main () {
	ll i,x,t ;
	cin(t) ;
	while(t--)
	{	cin(n) ;
		init() ;
		for (i=0;i<n;i++)
		{	cin(x) ;
			x=has(x) ;
			if (hash_table[x]==1)
			  sum++ ;
			hash_table[x]=1 ;   
		}	
		cout(sum) ;
	}
	return 0 ;
}
