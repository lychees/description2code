typedef unsigned long long int uint64;
typedef long long int int64;
int l,k;
int64 val;
string s;
int64 dp[101][101];
int64 go(int idx,int k){
	if(idx==l){
		if(k)
		return -1e18;
		return 0;
	}
	if(dp[idx][k]!=-1)
	return dp[idx][k];
	int64 ret1=-1e18,ret=0;
	for(int i=idx;i<l;i++){
		ret=ret*10+s[i]-'0';
		if(ret>val){
			break;
		}
		ret1=max(ret1,ret+go(i+1,k-1));
	}
	dp[idx][k]=ret1;
	return ret1;
}
int main(){
	int i,t;
	cin>>t;
	while(t--){
	memset(dp,-1,sizeof(dp));
	cin>>k>>val;
	cin>>s;
	l=s.length();
	int64 z=go(0,k);
	if(z<0){
		printf("-1\n");
	}
	else{
		printf("%lld\n",z);
	}
}
	return 0;
}
