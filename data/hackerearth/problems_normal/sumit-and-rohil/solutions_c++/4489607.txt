bool check2(string a , string b)
{
	int index = 0;
	priority_queue <char> pq1,pq2;
	for(int i = 0 ; i < a.length() ; i++)
	pq1.push(a[i]);
	for(int i = 0 ; i < b.length() ; i++)
	pq2.push(b[i]);
	
	while(true)
	{
		char x,y;
		x= pq1.top();
		y = pq2.top();
		if( x == y)
	    {
	    	index++;
	    }
	    	pq1.pop();
	    	pq2.pop();
	    
	    if(pq1.empty())
	    break;
	}
	if(index == a.length())
	return true;
	else 
	return false;
	
}

bool check(string p,string q)
{
	if(p.length() != q.length())
	return false;
	
	if((p[0] != q[0] )|| (p[p.length()-1] != q[q.length()-1]))
	return false;
	
    if(check2(p,q))
	return true;
	else 
	return false;
	
}
int main()
{
	int n,set;
	string s[100];
	cin>>n;
	set=n;
	for(int i=0;i<n;i++)
	{
		cin>>s[i];
		for(int j=0 ; j < i ; j++)
		{
			if(check(s[i],s[j]))
			{
				set--;
				break;
			}
		}
	}
	cout<<set;
}
