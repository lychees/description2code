#include <iostream>
#include <stdio.h>
using namespace std;

#define LL long long
#define MOD 1000000007
LL a[2000000];
LL fa[2000000], fact[200000];

LL powe(LL x, LL y){
    if(y == 0) return 1;

    LL ans = powe(x, y / 2);
    ans *= ans;
    if(ans >= MOD){
        ans %= MOD;
    }
    if(y & 1) {
        ans *= x;

    }
    if(ans >= MOD) ans %= MOD;
    return ans;
}
LL inv(LL x){
    return powe(x, MOD - 2);
}


LL ncr(LL x, LL y){
    int i;
    LL ans = 1, tp;
    for(i = 1; i <= y; i++){
        tp = i;
        ans *= (x - tp + 1);
        if(ans >= MOD){
            ans %= MOD;
        }
    }
    ans *= fact[y];
    if(ans >= MOD){
        ans %= MOD;
    }
    return ans;
}

int main()
{
    int i;
    for(i = 1; i <= 100000; i++){
        fa[i] = inv(i);
    }
    fact[1] = fa[1];
    for(i = 2; i <= 100000; i++){
        fact[i] = fa[i] * fact[i - 1];
        if(fact[ i] >= MOD) fact[i] %= MOD;
    }

    LL t,  j, x, n, m, tmp, g;
    cin>>t;
    while(t--){
        cin>>n>>m;
        for(i = 1; i<= n; i++){
            scanf("%lld", &x);
            m -= x;
        }
        if(m < 0){
            cout<<0<<endl;
            continue;
        }

        LL ans = ncr(m + n - 1, n - 1);
        cout<<ans<<endl;
    }
    return 0;
}
