LL dp[300][300] = {0};

string a , b;

LL solve(int i , int j)
{
	if(i == sz(a)) return (j == sz(b));
	else if(dp[i][j] != -1) return dp[i][j];
	LL ret = solve(i + 1 , j);
	if(a[i] == b[j])
		ret += solve(i + 1, j + 1);
	return (dp[i][j] = ret);
}

int main()
{
	inp_s;
	FOR(i,0,300) FOR(j,0,300) dp[i][j] = -1;
	getline(cin , b);
	getline(cin , a);
	cout << solve(0 , 0) << endl;
	return 0;
}
