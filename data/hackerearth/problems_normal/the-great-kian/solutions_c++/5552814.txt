#include <bits/stdc++.h>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    int n;
    cin>>n;
    unsigned long long int sum1=0,sum2=0,sum3=0;
    for(int i=1;i<=n;i++)
    {
    	unsigned long long int ele;
    	cin>>ele;
    	if(i%3==1)
    	sum1+=ele;
    	if(i%3==2)
    	sum2+=ele;
    	if(i%3==0)
    	sum3+=ele;
    }
    cout<<sum1<<" "<<sum2<<" "<<sum3<<"\n";
    return 0;
}
