#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	long long int a,b,c,d,n,move,max;
    	cin >>n>> a >> b>> c>>d>>move;
    	long long int ans=abs(a-c);
    	long long int ans2=abs(b-d);
    	if(ans2>(n-c))
    		max=ans2;
    	else
    		max=n-c;
    	if(move==1)
    		max--;
    	if(max<=(n-a))
    		cout << "Draw"<<endl;
    	else
    		cout << "White Wins"<<endl;
    }
    return 0;
}
