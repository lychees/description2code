#include<bits/stdc++.h>
#include <iostream>
using namespace std;

#define lol int

int main()
{
    lol T;
    cin >> T;
    while(T--)
    {
    	lol n,countB=0,countG=0;
    	cin >> n;
    	string g;
    	cin >> g;
    	//cout << n/2 << '\n';
        for(lol i=1;i<=(2*n);i+=2)
        {
        	if(g.at(i)!='B')countB++;
        	if(g.at(i)!='G')countG++;
        }
        if(countG<countB)cout << countG << '\n';
        else cout << countB << '\n';
    }
    return 0;
}
