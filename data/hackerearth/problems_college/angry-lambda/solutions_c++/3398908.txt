#include <bits/stdc++.h>
using namespace std;

#define gt(n) scanf("%d", &n)
int main()
{
    int t;

    gt(t);

    while (t--) {
        int x;
        int n;
        int k;
        int c = 0;

        gt(n);
        gt(k);

        while (n--) {
            gt(x);
            if (x <= 0)
                c++;
        }

        if (c < k)
            printf("YES\n");
        else
            printf("NO\n");
    }

    return 0;
}
