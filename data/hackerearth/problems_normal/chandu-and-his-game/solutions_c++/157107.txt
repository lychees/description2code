#include <iostream>
using namespace std;

int main()
{
    long int n,k;
	cin>>n;
	cin>>k;
	long int *points=new long int[k*2];
	for (long int i = 0; i < k*2; i++)
	{
		cin>>points[i];
	}

	long int moveCount=0;
	for (int i = 0; i < k; i++)
	{
		long int xd= n-points[i]>points[i]-1?points[i]-1:n-points[i];
		long int yd=n-points[i+k]>points[i+k]-1?points[i+k]-1:n-points[i+k];
		moveCount+=xd+yd;
	}

	cout<<moveCount;
    return 0;
}
