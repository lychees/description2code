#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    ll i,n,t,x,y,z,b,w,ans;
    scanf("%lld",&t);
    while(t--)
    {
    	ans=0;
    	scanf("%lld%lld",&b,&w);
    	scanf("%lld%lld%lld",&x,&y,&z);
    	if(x!=y)
    	{
    		if(x>y+z)
    		{
    			ans=w*y;
    			ans+=(y+z)*b;
    		}
    		else if(y>x+z)
    		{
    			ans=b*x;
    			ans+=(x+z)*w;
    		}
    		else
    		{
    			ans=(b*x)+(w*y);
    		}
    	}
    	else
    		ans=(b*x)+(w*y);
    	printf("%lld\n",ans);
    }
    return 0;
}
