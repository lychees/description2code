#include <iostream>
#include <algorithm>
#include <cstdio>

using namespace std;

int main(int argc, char *argv[])
{

        int T, X, Y, N, W, P1, P2;

        cin >> T;

        while(T--){
                cin >> X >> Y >> N >> W >> P1 >> P2;

                double dp[N+1][W+1];
                for(int i = 0; i <= N; i++){
                        dp[i][0] = 1;
                        for(int j = 1; j <= W; j++){
                                dp[i][j] = 0;
                        }
                }

                for(int i = 1; i <= W; i++){
                        dp[0][i] = 0;
                }

                double p1 = P1 * 0.01;
                double  p2 = P2 * 0.01;

                for(int i = 1; i <= N; i++){
                        for(int j = 1; j <= W; j++){
                                dp[i][j] = max(p1 * dp[i-1][max(j-X, 0)] + (1 - p1) * dp[i-1][max(j, 0)], p2 * dp[i-1][max(j-Y, 0)] + (1 - p2) * dp[i-1][max(j, 0)]);
                        }
                }
                printf("%.6f\n", dp[N][W] * 100);
        }
        return 0;
}
