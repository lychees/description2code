#include <iostream>
#include <queue>
using namespace std;

priority_queue <long> q;
long int n,i,x,t[3],j;

int main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);
	cout.tie(0);
    cin >> n;
    for (i = 0; i < n - n + 2; i++) 
    {
    	cin >> x;
    	q.push(x);
    	cout << -1 << "\n";
    }
    for (i = 2; i < n; i++)
    {
    	cin >> x;
    	q.push(x);
    	for (j = 0; j < 3; j++)
    	{
    		t[j] = q.top();
    		q.pop();
    	}
    	cout << t[0]*t[1]*t[2] << "\n";
    	for (j = 0; j < 3; j++) q.push(t[j]);
    }
    return 0;
}
