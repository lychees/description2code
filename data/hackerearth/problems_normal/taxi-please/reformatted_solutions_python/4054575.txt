def id_cmp(user1, user2):

def get_taxi(taxi, start):
    i = 0
    for tax_avail in taxi:
        if start >= tax_avail:
            return i

        i = i+1
    return -1


class customer:
    def __init__(self, start, duration, id, taxi_no):
        self.start = start
        self.duration = duration
        self.id = id
        self.taxi_no = taxi_no

    def __cmp__(self, other):
        if self.start < other.start:
            return -1
        elif self.start == other.start:
            return 0
        else:
            return 1

    def __str__(self):
        return str(self.start) + " " + str(self.duration) + \
            " " + str(self.taxi_no) + " " + str(self.id)
