#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;
int main()
{
long long int n,k,t,a,b;
cin>>t;
while(t--)
{
	long long int i,sum=0,x[10001];
	cin>>n>>k;
	if(k==1)
	cout<<0<<endl;

	else
	{
	a=n%k;
	b=n/k;
	for(i=1;i<k;i++)
	{
	if(i<=a)
	x[i]=b+1;
	else
	x[i]=b;
	}
	if(k%2==0)
	for(i=1;i<=k/2;i++)
	{
		if(i!=k-i)
		sum+=x[i]*x[k-i];
		else
		sum+=(x[i]*(x[i]-1))/2;
	}
	else
	{
	for(i=1;i<=k/2;i++)
	{
	sum+=x[i]*x[k-i];
	}
	}
	sum+=(b*(b-1))/2;
	cout<<sum<<endl;
	}
}
return 0;
}
