#include<iostream>
#include<vector>
#include<cstdio>
using namespace std;
long long int f[4][1000003];
long long int numDistinct(string S, string T) {

        if (S.length() == 0 || T.length() == 0)
            return 0;

       // long long int f[T.length() + 1][S.length() + 1];



        for (int j = 0; j <= S.length(); j++)
            f[0][j] = 1;


        for (int i = 1; i <= T.length(); i++)
            for (int j = 1; j <= S.length(); j++)
                if (T[i - 1] == S[j - 1])
                    f[i][j] = f[i - 1][j - 1] + f[i][j - 1];
                else
                    f[i][j] = f[i][j - 1];

        return f[T.length()][S.length()];
    }
int main()
{
    string str;
    string arr="101";
    cin>>str;
    long long int ans=numDistinct(str,arr);
    cout<<ans;
    return 0;
}
