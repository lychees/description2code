def is_subseq(x, y):
    it = iter(y)
    return all(any(c == ch for c in it) for ch in x)


def is_subsequence(needle, haystack):
    current_pos = 0
    for c in needle:
        current_pos = haystack.find(c, current_pos) + 1
        if current_pos == 0:
            return False
    return True


string = raw_input()
n = int(raw_input())
for _ in range(n):
    sub = raw_input()
    if is_subsequence(sub, string):
        print 'Yes'
    else:
        print 'No'
