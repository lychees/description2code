#include<iostream>
#include<vector>
#include<string>
#include<algorithm>
using namespace std;
int main()
{
	int t,i,sum;
	cin>>t;
	string s;
	vector<string> v;
	while(t--)
	{
		cin>>s;
		v.push_back(s);
		
	}
	sort(v.begin(),v.end());
	i=0;
	while(i<v.size())
	{
		sum=1;
		while(i+1<v.size()&&v.at(i)==v.at(i+1))
		{
			++sum;
			++i;
		}
		cout<<v.at(i)<<" "<<sum<<"\n";
		++i;
	}
	return 0;
}
