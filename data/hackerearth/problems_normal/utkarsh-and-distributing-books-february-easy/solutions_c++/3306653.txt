#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t,i,result=0;
    cin>>t;
    for(i=0;i<t;i++)
    {
    	result=0;
    	int n;
    	cin>>n;
    	int arr[n];
    	for(int j=0;j<n;j++)
    		{
    			cin>>arr[j];
    			result+=arr[j]-1;
    		}
    	sort(arr,arr+n);
    	cout<< arr[0]-1<<" "<<result<<endl;
    }
    return 0;
}
