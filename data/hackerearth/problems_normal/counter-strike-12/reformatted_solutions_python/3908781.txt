def compute_game_result(targets, positions, max_distance):
    half_mark = len(targets)/2
    ctr = 0
    for target in targets:
        hit = False
        for pos in positions:
            if abs(pos[0]-target[0]) + abs(pos[1]-target[1]) <= max_distance:
                hit = True
                break
        if hit:
            ctr += 1
            if ctr > half_mark:
                return "YES"
    return "NO"


T = int(raw_input())
for i in range(T):
    (N, M, D) = map(int, raw_input().split(' '))
    positions = [map(int, raw_input().split(' ')) for j in range(N)]
    targets = [map(int, raw_input().split(' ')) for j in range(M)]
    print compute_game_result(targets, positions, D)
