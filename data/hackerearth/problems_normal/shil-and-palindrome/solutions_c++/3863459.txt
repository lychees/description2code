using namespace std;

int letters[26];

int main()
{
	memset(letters, 0, sizeof(letters));
    string s; cin >> s;
    for (char ch : s) {
    	++letters[ch - 'a'];
    }
    int odd = 0;
    for (int i = 0; i < 26; ++i) {
    	if (letters[i] & 1) ++odd;
    }
    if (odd > 1) {
    	cout << -1 << endl;
    }
    else {
    	int oddidx= -1;
    	string snew;
    	for (int i = 0; i < 26; ++i) {
    		if (letters[i] != 0 && !(letters[i] & 1)) {
    			for (int j = 0; j < letters[i]/2; ++j) {
    				snew += i + 'a';
    			}
    		}
    		else if (letters[i] != 0) {
    			oddidx = i;
    			for (int j = 0; j < letters[i]/2; ++j) {
    				snew += i + 'a';
    			}
    		}
    	}
    	
    	if (oddidx != -1) {
    		snew += oddidx + 'a';
    		cout << snew;
    		for (int i = snew.length() - 2; i >= 0; --i) {
    			cout << snew[i];
    		}
    		cout << endl;
    	}
    	else {
    		cout << snew;
    		for (int i = snew.length() - 1; i >= 0; --i) {
    			cout << snew[i];
    		}
    		cout << endl;
    	}
    }
    return 0;
}
