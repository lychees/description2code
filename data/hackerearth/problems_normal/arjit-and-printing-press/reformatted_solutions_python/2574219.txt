num_tests = int(raw_input())

for test in xrange(num_tests):
    istr = list(raw_input())
    rstr = sorted(raw_input())
    index = 0

    for i in xrange(len(istr)):
        if index < len(rstr):
            if istr[i] > rstr[index]:
                istr[i] = rstr[index]
                index += 1
        else:
            break
    print "".join(istr)
