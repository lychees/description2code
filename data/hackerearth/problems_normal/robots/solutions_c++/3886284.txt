#include<stdio.h>
#include<algorithm>
#define mxm 1000000007
int main()
{
	int t,i,j,k,l,m,n;
	long long int arr[10004];
	long long int res;
	scanf("%d",&t);
	while(t--)
	{
		int flag=1;
		res = 1;
		scanf("%d",&n);
		for(i=0;i<n;i++)
		scanf("%lld",&arr[i]);
		std::sort(arr,arr+n);
		for(i=0;i<n;i++)
		{
			if(arr[i]-i>0)
			res = (res*(arr[i]-i))%mxm;
			else
			{
				flag = 0 ;
				break;
			}
		}
		if(flag ==1)
		printf("%lld\n",res);
		else
		printf("0\n");
	}
	return 0;
}
