#include<iostream>
#include<cstdio>

using namespace std;

int main()
{
    int t,n,d,r;
    scanf("%d",&t);
    while(t--)
    {
              scanf("%d%d%d",&n,&d,&r);
              n = n%d;
              r--;
              while(r--)
              {
                 n = (n*10)%d;
              }
              printf("%d\n",(n*10)/d);
    }
    return 0;
}
