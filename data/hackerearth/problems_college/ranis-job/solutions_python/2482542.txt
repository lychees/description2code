class DiskSpace:
    def minDrives(self, used, total):
    	#print(used,total)
        used=list(map(int,used.split()))
        total=list(map(int,total.split()))
        s=sum(used)
        total=sorted(total,reverse=True)
        i=0
        while s>0:
            s-=total[i]
            i+=1
        #print(i)
        return i
