#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,b,x,y;
		cin>>n>>b;
		x=n/b;
		y=x+1;
		x=x*b;
		y=y*b;
		int z=abs(n-x);
		int z1=abs(n-y);
		if(z==z1)
		cout<<y<<endl;
		else if(z1>z)
		cout<<x<<endl;
		else
		cout<<y<<endl;
	}
	return 0;
}
