#include<stdio.h>
#include<iostream>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int h,m;
		int angle;
		scanf("%d %d",&h,&m);
		int hr,min;
		hr=60*h+m;
		hr=(hr/2);
		min=6*m;
		if(hr>min)
		{
			angle=hr-min;
		}
		else
		angle=min-hr;
		
		if((360-angle)<angle)
		angle=360-angle;
		cout << angle <<'\n';
	}
	return 0;
}
