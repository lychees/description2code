#include <bits/stdc++.h>
using namespace std;

bool equilibrium(int arr[], int n){
   int sum = 0;
   int leftsum = 0;
   int i;
   for (i = 0; i < n; ++i)
        sum += arr[i];
   for( i = 0; i < n; ++i){
      sum -= arr[i];       
      if(leftsum == sum)
        return true; 
      leftsum += arr[i];
   }
   return false;
}

int main()
{
	int t;cin>>t;
	while(t--){
		int n; cin>>n; int a[n+1];
		for(int i=0;i<n;i++) cin>>a[i];
		if(equilibrium(a,n)) cout<<"YES"<<"\n";
		else cout<<"NO"<<"\n";
	}
    //cout << "Hello World!" << endl;
    return 0;
}
