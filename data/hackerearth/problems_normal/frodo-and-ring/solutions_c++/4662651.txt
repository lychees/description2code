	int n,m,t,i,j;
	cin>>t;
	while(t--){
		cin>>n>>m;
		forp(i,0,204){
			forp(j,0,204){
				val[i][j]=-999999;
			}
		}
		forp(i,1,n){
			forp(j,1,m){

			forp(j,1,m){
				if(i!=1 or j!=1){
					if(val[i-1][j]<0 and val[i][j-1]<0){
						val[i][j]=-999999;
					}
					else{
						val[i][j]=max(val[i-1][j],val[i][j-1]);
						if(a[i][j]=='T'){
							val[i][j]+=150;
						}
						else if(a[i][j]=='D'){
							val[i][j]-=100;
						}
					}
				}
			}
		}
		if(val[n][m]>=0){
			cout<<val[n][m]<<endl;
		}
		else{
			cout<<"-1\n";
		}
	}
	return 0;
}
