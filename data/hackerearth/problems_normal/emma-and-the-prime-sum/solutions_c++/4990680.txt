#include<bits/stdc++.h>
#define lli long long int
using namespace std;
int main()
{
	bool prime[1000001];
	memset(prime,true,sizeof(prime)); prime[1]=false;
	for(lli i=2;i*i<=1000000;i++)
	{
		if(prime[i])
		{
			for(lli j=i*2;j<=1000000;j+=i)prime[j]=false;
		}
	}
	vector<lli>val(1000001);
	for(lli i=1;i<=1000000;i++)
	{
		if(prime[i])val[i]=val[i-1]+i;
		else val[i]=val[i-1];
	}
	int t; cin>>t;
	while(t--)
	{
		lli x,y; cin>>x>>y; lli sum=0;
		cout<<val[y]-val[x-1]<<endl;		
	}
	return 0;
}
