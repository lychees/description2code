#include <iostream>
#include <string>
#include <cmath>
using namespace std;

long getMax(long n) {
	if(n/12 > 0)
		return getMax(n/2) + getMax(n/3) + getMax(n/4);
	return n;
}

int main() {
	ios_base::sync_with_stdio (false);
	cin.tie(NULL);
	string str;
	while(getline(cin, str)) {
		long st = stoi(str);
		cout << getMax(st) << "\n";
	}
	return 0;
}
