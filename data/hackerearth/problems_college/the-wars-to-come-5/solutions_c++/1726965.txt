#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{

int N=0,a[1000],k=0,s=0,count=0,i,sum=0,large=0;
cin>>N;
for(i=0;i<N;i++)
cin>>a[i];
cin>>s;
while(sum<s)
{
large=a[0];
for(i=0;i<N;i++)
	{
	if(large<a[i])
	{
	k=i;
	large=a[i];
	}
	}
sum+=large;
a[k]=-1;
count++;
}
if(s==0) cout<<"\n0";
else
cout<<"\n"<<count;
}
