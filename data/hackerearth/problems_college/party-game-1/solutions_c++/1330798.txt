#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
	long t,x,y,n,i,j,k;
	cin>>t;
	while(t--){
		cin>>x>>y>>n;
		long a[n];
		for(k=0;k<n;k++)
			cin>>a[k];
		sort(a,a+n);
		
		for(i=0;i<n;i++){
			if(x<a[i])
				break;
			x-=a[i];
		}
		for(j=0;j<n;j++){
			if(y<a[j])
				break;
			y-=a[j];
		}
		if(i==j)
			cout<<"Tie\n";
		else if(i<j)
			cout<<"Jhool Won\n";
		else
			cout<<"Chef Won\n";
	}
return 0;
}
