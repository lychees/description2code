#include <iostream>
using namespace std;
 
int longestPal(string str)
{ 
	int len = str.length();
	int i, j,k;
	for(i  = 0; i < len; i++)
	{
		for (j = i, k = len-1; j < k; j++, k--)
		{
			if (str[j] != str[k]) 
			{
				break;
			}
		}
		if (j >=k)
		{
			break;
		}
	}
	cout << i + len  << endl;
}
int main()
{
	int T;
	cin >> T;
	while(T--)
	{
		string str;
		cin >> str;
		longestPal( str );
	}	
	return 0;
}
