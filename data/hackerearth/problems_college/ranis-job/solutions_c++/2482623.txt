#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll total[10000000];
int main()
{
     //freopen("in.txt","r",stdin);
  // freopen("out.txt","w",stdout);
int t;
cin>>t;
while(t--)
{
	ll n;
	ll s=0;
	cin>>n;
   for(int i=0;i<n;i++)
   {
   	ll temp;
   	cin>>temp;
   	s+=temp;
   }
	
   for(int i=0;i<n;i++)
   {
      cin>>total[i];
   }
   if(s==0)
    cout<<"0\n";
    else
    {
    
	sort(total,total+n,greater<ll>());
	int i=0;
	ll c=0;
	while(i<n)
	{
		if(s<total[i])
		 {
		   c++;
		   break;
	     }
		
		else if(s>total[i])
		{
			s-=total[i];
			c++;
	    	if(s==0)
    		break;
		} 
		else
		{
			c++;
			break;
		}
		
		
i++;		
		
		
}
		cout<<c<<"\n";
	}
	
	
	
	
	
	
	
	
	
}
  

















    return 0;
}
