#include <iostream>
using namespace std;

int main()
{
    int t;cin>>t;
    while(t--){
    	long long int x1,x2,y1,y2,r;
    	cin>>x1>>y1>>x2>>y2>>r;
    	long long int num=abs(x1*(y2-y1)-y1*(x2-x1));
    	num*=num;
    	long long int den=(x2-x1)*(x2-x1)+(y2-y1)*(y2-y1);
    	r*=r;
    	if(num<den*r)cout<<"REPLANNING"<<endl;
    	else if(num==den*r)cout<<"JUST MISSED"<<endl;
    	else cout<<"SAFE"<<endl;
    }
}
