#include <bits/stdc++.h>

using namespace std;
char p[1005], s[2005];

void add(char *a, char *b, char *c)
{
    int la = strlen(a), lb = strlen(b), k = 0, i, carry = 0, x;
    if(la < lb)
    {
        for(i = 0;i < la;++i)
        {
            x = a[i] - '0' + b[i] - '0' + carry;
            carry = x / 10;
            c[k++] = x % 10 + '0';
        }
        for(;i < lb;++i)
        {
            x = b[i] - '0' + carry;
            carry = x / 10;
            c[k++] = x % 10 + '0';
        }
        while(carry)
        {
            c[k++] = (carry % 10) + '0';
            carry /= 10;
        }
    }
    else
    {
        for(i = 0;i < lb;++i)
        {
            x = a[i] - '0' + b[i] - '0' + carry;
            carry = x / 10;
            c[k++] = x % 10 + '0';
        }
        for(;i < la;++i)
        {
            x = a[i] - '0' + carry;
            carry = x / 10;
            c[k++] = x % 10 + '0';
        }
        while(carry)
        {
            c[k++] = (carry % 10) + '0';
            carry /= 10;
        }
    }
    c[k] = '\0';
}

int digitSum(int a)
{
    int ans = 0;
    while(a)
    {
        ans += (a % 10);
        a /= 10;
    }
    return ans;
}

int main()
{
    int t;
    long long ans;
    for(scanf("%d", &t);t--;)
    {
        scanf("%s", p);
        add(p, s, s);
    }
    ans = 0;
    for(int i = 0, l = strlen(s);i < l;++i)
        ans += (s[i] - '0');
    while((ans / 10))
    {
        ans = digitSum(ans);
    }
    printf("%lld\n", ans);
}
