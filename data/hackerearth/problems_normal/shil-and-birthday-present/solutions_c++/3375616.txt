#include<iostream>
#include<set>
using namespace std;
int main()
{
	set<int>set1;
	int n;
	scanf("%d",&n);
	int a[n];
	for(int i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
		set1.insert(a[i]);
	}
	int k = set1.size();
	set<int>::iterator it;
	int i=0;
	unsigned long long int sum = 0;
	for(it=set1.begin();it!=set1.end();it++)
	{
		sum = sum + (k-i-1);
	}
	printf("%llu\n",sum/2);
}
