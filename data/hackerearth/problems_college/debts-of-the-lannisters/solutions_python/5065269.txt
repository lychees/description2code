from sys import stdin
a = stdin.readline().strip()
di = {}
for i in a:
	di[i] = di.get(i,0) + 1

t = int(stdin.readline())
for _ in xrange(t):
	b = stdin.readline().strip()
	fir = 0
	sec = 0
	for i in xrange(10):
		if a[i] == b[i]:
			fir+=1
	hi = {}
	for i in b:
		hi[i] = hi.get(i,0)+1
	for i in hi:
		x = hi.get(i,0)
		y = di.get(i,0)
		if x>=y:
			sec += y
		else:
			sec += x
	print fir,sec-fir
