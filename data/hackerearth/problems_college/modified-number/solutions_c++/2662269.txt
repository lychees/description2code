#include<iostream>
#include<cmath>
using namespace std;

    int digits(int d)
    {
        int m=0;
        while(d!=0)
        {
            m+=1;
            d/=10;
        }
        return m;


    }


    int main()
    {
    	cin.sync_with_stdio(false);
    	cout.sync_with_stdio(false);
        long long int p,q,count=0;
        cin>>p>>q;
      //  int m=digits(i);
        for(long long int i=p;i<=q;i++)
        {
                long long int sum=i*i;
                int m=digits(i);
                long long int a=sum%(int)pow(10.0,m);
                long long int b=sum/(int)pow(10.0,m);
                if(a+b==i)
                {
                    count++;
                    cout<<i<<" ";
                }
        }
            if(count==0)
                cout<<"INVALID RANGE"<<endl;




        return 0;
        }
