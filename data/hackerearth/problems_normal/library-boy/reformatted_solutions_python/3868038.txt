import sys
# f = open("books.txt")
shelf = dict()
num_of_shelves = 0
for line in sys.stdin:
    words = line.split()
    word1 = words[0]
    if word1.isalpha():
        shelf[word1[0]] = shelf.get(word1[0], 0) + 1

for v in shelf.values():
    if v < 10:
        num_of_shelves = num_of_shelves + 1
    elif v > 10:
        if v % 10 == 0:
            num_of_shelves = num_of_shelves + v/10
        else:
            num_of_shelves = num_of_shelves + 1 + v/10

print num_of_shelves
