const int MAXN = 3e5 + 256;

const double eps = (double)1e-9;
template<typename T> inline bool updmin(T &a, const T &b) {return a > b ? a = b, 1 : 0;}
template<typename T> inline bool updmax(T &a, const T &b) {return a < b ? a = b, 1 : 0;}

int n, k, pos[7][1111], dp[MAXN];
vector < int > v[MAXN];

int calc(int x) {
	if(~dp[x]) return dp[x];
	dp[x] = 1;
	for(auto &to : v[x]) updmax(dp[x], calc(to) + 1);
	return dp[x];
}

		rep(j, 1, n) {
			int x; scanf("%d", &x);
			pos[i][x] = j;
		}
	}
	rep(i, 1, n) {
		rep(j, 1, n) {
			bool ok = 1;
			rep(it, 1, k) if(pos[it][j] <= pos[it][i]) { ok = 0; break; }
			if(ok) v[i].pb(j);
		}
	}
	int res = 0;
	memset(dp, 255, sizeof dp);
	rep(i, 1, n) updmax(res, calc(i));
	printf("%d\n", res);
	return 0;
}
