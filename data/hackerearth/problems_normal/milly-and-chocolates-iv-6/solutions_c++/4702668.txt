#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
    long long int i,j,k,arr[1004][1004],max,x,y,n,m,a[1000003],sum;
    while(t--)
    {
    cin>>n>>k>>m;
    for(i=0;i<n;i++)
    cin>>a[i];
    for(j=0;j<n;j++)
    {
    	for(x=0;x<a[j];x++)
    	cin>>arr[j][x];
    }
    sum=0;
    max=10000000000000000;
    for(i=0;i<n;i++)
    {
    	for(j=0;j<a[i];j++)
    	{
    		sum+=arr[i][j]+m+k;
    	}
    	sum=sum-k;
    	if(sum<max)
    	{
    		max=sum;
    		y=i+1;
    	}
    	sum=k-m;
    }
    cout<<y<<" "<<max<<"\n";
    
    }
    return 0;
}
