#include<bits/stdc++.h>
typedef long long LL;
using namespace std;
int main(){
    int t;
    cin>>t;
    while(t--){
        string s;
        cin>>s;
        LL ans=s[s.length()-1]-'0';
        int tmp;
        for(int i=s.length()-3;i>=0;i-=2){
            tmp=s[i]-'0';
            if(s[i+1]=='+')ans+=(LL)tmp;
            else if(s[i+1]=='-')ans-=(LL)tmp;
            else if(s[i+1]=='*')ans*=(LL)tmp;
        }
        cout<<ans<<endl;
    }
    return 0;
}
