#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int n;
    cin >> n;
    for(int i=0;i<n;i++){
    	string in;
    	cin >> in;
    	int ans = 0;
    	char cur = in[0];
    	for(int j=1;j<in.length();j++){
    		if(cur == in[j])
    		ans++;
    		else
    		cur = in[j];
    	}
    	cout << ans << endl;
    }
    
    return 0;
}
