from math import factorial


def nCr(n, r):
    f = factorial
    return f(n) / f(r) / f(n-r)


def calcCom(boys, girls):
    ncrval = nCr(girls + 1, boys)
    output = ncrval * factorial(girls) * factorial(boys)
    output = output % 1000000007
    return output

        print -1
    else:
        output = calcCom(boys, girls)
