#include <iostream>
using namespace std;

int main()
{
	int t;
    cin>>t;
    while(t)
    {    int n,s=0;
    	cin>>n;
    	int a[n];
    	int c=n;
    	while(n)
    	{
    	 cin>>a[n-1];
    	 s=s+a[n-1];
    	 n--;
    	}
    	if(c>2)
    	{
    		if(s%2==0)
    		cout<<"YES"<<endl;
          else	cout<<"NO"<<endl;
    	}
       if(c==2)
    	  {
    	  	if(a[0]==a[1])
    	  	{
    	    cout<<"YES"<<endl;
    	  	}
    	  	 else	cout<<"NO"<<endl;
    	  }
       if(c<=1)
        cout<<"NO"<<endl;
       t--;
    }

    return 0;
}
