#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long n,i,j,sum;
    cin>>n;
    if(n<4)
    cout<<"0\n";
    else
    {
    	for(i=3;i<n;i+=3)
    	{
    		sum+=i;
    	}
    	for(i=5;i<n;i+=5)
    	{
    		sum+=i;
    	}
    	for(i=15;i<n;i+=15)
    	{
    		sum-=i;
    	}
    	cout<<sum<<"\n";
    }
    return 0;
}
