#include<bits/stdc++.h>

typedef long long ll;
using namespace std;

int main(){
	ios::sync_with_stdio(false);
	int t;
	ll n;
	cin>>t;
	while(t--){
		cin>>n;
		vector<ll> arr(n);
		for(int i = 0; i < n; i++)
			cin>>arr[i];
		sort(arr.begin(), arr.end());
		cout<<((arr[n-1] - arr[0])%2?"Yes\n":"No\n");
	}
	return 0;
}
