#include <iostream>
#include<map>
using namespace std;

int main()
{
    int t,i;
    string str;
    cin>>t;
    map<string,int> m;
    for(i=0;i<t;i++){
    	cin>>str;
    	m[str]++;
    }
    map<string,int>::iterator it;
    for(it=m.begin();it!=m.end();it++){
    	cout<<it->first<<" "<<it->second<<endl;
    }
    return 0;
}
