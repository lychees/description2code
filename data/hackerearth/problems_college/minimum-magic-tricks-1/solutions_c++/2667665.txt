#include<bits/stdc++.h>
#define z 1000000007
using namespace std;
long long sum=0,ans[1001];
void rec(){
    ans[1]=1,ans[2]=1;
    for(int i=3;i<=1000;i++){
            ans[i]=0;
        for(int j=1;j<=i-2;j++){
            ans[i]+=2*ans[j];
            ans[i]%=z;}
        ans[i]++;
    }
}
int main(){
   int t;
   scanf("%d",&t);
   rec();
   while(t--){
        int n;
        sum=0;
    scanf("%d",&n);
   printf("%lld\n",ans[n]);
   }
   return 0;
}
