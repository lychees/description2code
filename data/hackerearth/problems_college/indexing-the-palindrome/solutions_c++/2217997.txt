#include<bits/stdc++.h>
using namespace std;

int main()
{

    int t;
    cin>>t;

    while(t--)
    {

        string s;
        cin>>s;

        int r=s.size()-1;
        int l=0;

        int pal_index=-1;

        while(l<=r)
        {
            if(s[l]==s[r])
            {
                l++;
                r--;
            }
            else
            {
                if(s[l]==s[r-1])
                    pal_index=r;

                if(s[r]==s[l+1])
                    pal_index=l;

                break;
            }
        }
        cout<<pal_index<<endl;

    }

}
