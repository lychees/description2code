#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int n,i;
    	int count=0;
    	cin>>n;
    	long long int a[n];
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	i=0;
    while(i<n-1)
    {
    if(a[i]>=a[i+1])	
    	count++;
    	i++;
    }
    cout<<count<<endl;
    }
    return 0;
}
