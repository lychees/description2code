#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    int a[26];
    for(int i=0; i<26; i++)
        a[i]=1;
    int c=0;
    while(t--)
    {
        string s;
        cin>>s;
        int l=s.length();
        for(int i=0; i<l; i++)
        {
            if(a[s[i]-'a'])
                a[s[i]-'a']=2;
        }
        for(int i=0; i<26; i++)
        {
            if(a[i]!=2)
                a[i]=0;
                else a[i]=1;
        }
     //   for(int i=0; i<26; i++)
           // cout<<a[i];
       // cout<<endl;
    }
    for(int i=0; i<26; i++)
    {
        if(a[i])
            c++;
       // cout<<a[i]<<endl;
    }
    cout<<c<<endl;
    return 0;
}
