#include <iostream>
#include<unordered_map>
#include<ctype.h>
using namespace std;

int main()
{
	string s;
	int k;
	cin>>k;
	unordered_map<string,int> mymap;
	for(int i=0;i<k;i++)
	{
		cin>>s;
		mymap[s]=1;
	}
	
	cin>>k;
	int cnt=0;
	for(int i=0;i<k;i++)
	{
		cin>>s;
		if(mymap.find(s)==mymap.end())
		{
			cnt++;
			char ch=toupper(s.at(0));
			if(cnt!=1)
				cout<<"."<<ch;
			else
				cout<<ch;
		}
	}
	cout<<endl;
	
}
