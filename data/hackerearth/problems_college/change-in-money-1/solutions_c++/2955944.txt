#include <iostream>
using namespace std;
long long int count( long long int S[], long long int m, long long int n )
{
    long long int i, j, x, y;
    long long int table[n+1][m];
    for (i=0; i<m; i++)
        table[0][i] = 1;
    for (i = 1; i < n+1; i++)
    {
        for (j = 0; j < m; j++)
        {
            x = (i-S[j] >= 0)? table[i - S[j]][j]: 0;
            y = (j >= 1)? table[i][j-1]: 0;
            table[i][j] = x + y;
        }
    }
    return table[n][m-1];
}
int main()
{
	long long int m,n;
	cin>>n>>m;
	long long int* a = new long long int[m];
	for(int i=0;i<m;i++)
	cin>>a[i];
	cout<<count(a,m,n)<<endl;
    return 0;
}
