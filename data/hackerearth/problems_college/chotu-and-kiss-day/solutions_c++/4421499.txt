#include <iostream>
#include <bits/stdc++.h>
#define s(x) scanf("%lld",&x)
using namespace std;
long long int arr[3005];
int main()
{
    long long int n,k,i,j,flag,x,l,r;
    s(n);
    for(i=0;i<n;i++)
    {
    	s(arr[i]);
    }
    cin>>k;
    sort(arr,arr+n);
   	for(i=0;i<n;i++)
    {
		x=k-arr[i];
		flag=0;
		l=0;
		r=n-1;
		while(l<r)
    	{
    		if((arr[l]+arr[r])==x)
    		{
    			flag=1;
    			break;
    		}
    		else if((arr[l]+arr[r])<x)
    		{
    			l++;
    		}
    		else
    		{
    			r--;
    		}
    	}
    	if(flag)
    	{
    		printf("YES\n");
    		break;
    	}
    }
	if(!flag)
    {
    	printf("NO\n");
    }
    return 0;
}
