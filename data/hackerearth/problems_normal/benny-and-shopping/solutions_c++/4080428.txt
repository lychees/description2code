#include <iostream>
#include<string>
using namespace std;

int main()
{
    int t,posdol,outpos;
    string in,out;
    cin>>t;
    cin.ignore(100,'\n');
    for(int tc=0;tc<t;tc++)
    {
        out.clear();
        in.clear();
    	outpos=posdol=0;
    	getline(cin,in);
    	out="$";
    	posdol=in.find("$");
    	outpos=1;
    	posdol++;
    	//cout<<"12";
    	while(in[posdol]==' ' || ('0'<= in[posdol] && in[posdol] <= '9'))
    	{
    	    //cout<<"99";
    	    if(outpos==1)
            {
                if('1'<=in[posdol]&&in[posdol]<='9')
                {
                    out=out+in[posdol];
    			//cout<<out<<endl;
    			posdol++;
    			outpos++;
    			continue;
                }
                posdol++;
                continue;
            }
    		if('0'<=in[posdol]&&in[posdol]<='9')
    		{
    			out=out+in[posdol];
    			//cout<<out<<endl;
    			posdol++;
    			outpos++;
    			continue;
    		}
    		posdol++;
    	}
    	if(outpos==1)
            out="$0";
    	cout<<out<<endl;
    }
    return 0;
}
