#include <iostream>

using namespace std;
struct BinarySearchTree
{
    BinarySearchTree*left;
    BinarySearchTree*right;
    int data;
};

int LCA(int root,int a,int b)
{
    int left,right;
    if(root>max(a,b))
        return 0;
    if(root==0)
        return root;
    if(root==a||root==b)
        return root;
    left=LCA(2*root,a,b);
    right=LCA(2*root+1,a,b);
    if(left&&right)
        return root;
    else
        return (left?left:right);
}

int main()
{
   int t;
   long long int a,b;
   cin>>t;
   for(int i=0;i<t;i++)
   {
       cin>>a>>b;
       int s=LCA(1,a,b);
       cout<<s<<endl;

   }
}
