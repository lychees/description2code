#include<bits/stdc++.h>
using namespace std;
int main()
{
int n,k;
cin>>n>>k;
int arr[n+1][2],i,j;
for(i=0;i<n;i++)
{
	cin>>arr[i][0]>>arr[i][1];
}
int ans[n+2],mxa[n+2],mxb[n+2];
ans[0]=arr[0][0]*arr[0][1];
mxa[0]=arr[0][0];
mxb[0]=arr[0][1];
for(i=1;i<k;i++)
{
	mxa[i]=arr[i][0];
	mxb[i]=arr[i][1];
//	cout<<mxa[i]<<" "<<mxb[i]<<endl;
}
for(i=k;i<=n;i++)
{
	int p,q,z;
	if(i<n)
	{
	 p=mxa[i-1];q=mxb[i-1];z=max(p,arr[i][0])*max(q,arr[i][1]);
	for(j=i-1;j>=0&&j>=i-k;j--)
	{
		if(max(arr[i][0],mxa[j])*max(mxb[j],arr[i][1])<z)
		{
			z=max(arr[i][0],mxa[j])*max(mxb[j],arr[i][1]);
			p=mxa[j];
			q=mxb[j];
		}
	}
	mxa[i]=max(p,arr[i][0]);
	mxb[i]=max(q,arr[i][1]);
	}
	else
	{
		p=n-1;
		z=mxa[n-1]*mxb[n-1];
		for(j=i-2;j>=i-k;j--)
		{
			if(mxa[j]*mxb[j]<z)
			{
				p=j;
				z=mxa[j]*mxb[j];
			}
		}
		mxa[i]=mxa[p];
		mxb[i]=mxb[p];
	}
//	cout<<mxa[i]<<" "<<mxb[i]<<endl;
}
cout<<mxa[n]*mxb[n]<<endl;
return 0;
}
