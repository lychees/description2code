#include <iostream>
using namespace std;

int main()
{
    int n,q;
    cin>>n>>q;
    int a[n];
    for(int i=0;i<n;i++)
    	cin>>a[i];
    int f,l,r,k;
    long sum;
    char s[6]="pizza";
    for(int i=0;i<q;i++)
    {
    	sum=0;
    	cin>>f>>l>>r;
    	if(f==1)
    	{
    		for(int j=l-1;j<r;j++)
    		{
    			sum+=a[j];	
    		}
    		cout<<sum<<"\/"<<(r-l+1)<<endl;
    	}
    	else
    	{
    		k=0;
    		for(int j=l-1;j<r;j++)
    		{
    			a[j]=a[j]+(s[k%5]-'a'+1);
    			k++;
    		}
    	}
    }
    return 0;
}
