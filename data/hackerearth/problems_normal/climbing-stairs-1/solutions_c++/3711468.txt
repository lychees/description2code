#include <iostream>
#include <cmath>
using namespace std;

namespace domain{
	
	bool A_range(int A){
		if(A>=1 && A<=pow(10,9))
			return true;
	}
	
	bool B_range(int B, int A){
		if(B>=1 && B<=pow(10,9) && B<A)
			return true;
	}
	
	bool N_range(int N, int A){
		if(N>=1 && N<=pow(10,9) && N>=A)
			return true;
	}
}

int requiredNoOfDays(int A, int B, int N){

	int count = 1;	 
	int stairs_covered = 0;
	
	while(1){
		stairs_covered += A;
		if(stairs_covered >= N)
			return count;
			
		stairs_covered -= B;
		count++;
	}	
	 
}

int main()
{
    int A;
    cin>>A;
    
    if(domain::A_range(A)== false){
    	cout<<"A is not in range"<<endl;
    	exit(1);
    }
    
    int B;
    cin>>B;
    
    if(domain::B_range(B,A)== false){
    	cout<<"B is not in range"<<endl;
    	exit(2);
    }
    
    int N;
    cin>>N;
    
    if(domain::N_range(N,A)== false){
    	cout<<"N is not in range"<<endl;
    	exit(3);
    }
    
    cout<<requiredNoOfDays(A,B,N)<<endl;
    return 0;
}
