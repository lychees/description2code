#include <iostream>
#include<string.h>
#include<stdio.h>
using namespace std;

int main()
{
    int t,a[28],i,l,c;
    char b[100000];
    
    cin>>t;
    while(t--){
    	
    	scanf("%s",&b);
    	for(i=0;i<26;i++)a[i]=0;
    	
    	l=strlen(b);
    	for(i=0;i<l;i++){
    		a[b[i]-97]++;
    	}
    	c=0;
    	
    	for(i=0;i<26;i++){
    		if(a[i]%2!=0)
    		c++;
    	}
    
    	if(c==0 ||c%2!=0)cout<<"JACK"<<endl;
    	else cout<<"DAVY JONES"<<endl;
    }
    return 0;
}
