#include <bits/stdc++.h>
using namespace std;

int main() {
	int x;
	long long y;
	
	cin >> x;
	
	while(x--){
	    cin >> y;
	    
	    if(y == 2 || y == 3 || y == 5 || y == 7)
            cout << "YES\n";
            
        else if(y % 2 == 0 || y % 5 == 0 || y % 3 == 0 || y % 7 == 0 || y % 9 == 0)
            cout << "NO\n";
        
        else
            cout << "YES\n";
        
	}
	return 0;
}
