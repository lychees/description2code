#include <bits/stdc++.h>
#define ll long long int

using namespace std;

int main()
{
	int t;
	ll a,b,ans,mod,n,x,a1,b1;
	cin>>t;
	while(t--)
	{
		cin>>n>>mod;
		x=mod-1;
		if(x%2)
		{
			a=x;
			b=(x+1)/2;
		}
		else
		{
			a=x/2;
			b=(x+1);
		}
		x=n%mod;
		if(x%2)
		{
			a1=x;
			b1=(x+1)/2;
		}
		else
		{
			a1=x/2;
			b1=(x+1);
		}
		ans=(a*b*(n/mod))+(a1*b1);
		cout<<ans<<endl;
	}
	return 0;
}
