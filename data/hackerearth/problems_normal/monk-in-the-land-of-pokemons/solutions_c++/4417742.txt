#include <iostream>
#include <vector>
#include <list>
#include <set>
using namespace std;

int main()
{
    
    int t;
    int n;
    int a, b;
    
    
    cin>>t;
    while(t--){
    	multiset <int> v;
    	cin>>n;
    	int count = 0;
    	while(n--){
    		cin>>a>>b;
    		
    		if(a == b){
    			continue;
    		}
    		
    		v.insert(a);
    		if(v.find(b) == v.end()){
    			count++;
    		}
    		else{
    			v.erase(v.find(b));
    		}
    	}
    	
    	cout<<count<<"\n";
    }
    return 0;
}
