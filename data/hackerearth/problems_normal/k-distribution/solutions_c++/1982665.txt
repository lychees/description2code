#include <iostream>
#define mod 1000000007
using namespace std;

int main()
{
    int n,a[109],k;
    long long s=0;
    int dp[100009]={0},temp[100009];
    long long tot=1;
    cin>>n>>k;
    for(int i=0;i<n;i++)
    {cin>>a[i];s+=(long long)a[i];tot=(tot*2LL)%mod;}
    if(s<2LL*(long long)k)
    {cout<<0<<"\n";return 0;}
    dp[0]=1;
    for(int i=0;i<n;i++)
    {
    for(int j=0;j<k;j++)
    temp[j]=dp[j];
    for(int j=a[i];j<k;j++)
    dp[j]=(temp[j]+temp[j-a[i]])%mod;	
    }
    int sum=0;
    for(int i=0;i<k;i++)
    sum=(sum+dp[i])%mod;
    sum=(2*sum)%mod;
 	int res=(mod+((int)tot-sum)%mod)%mod;
 	cout<<res<<"\n";
    return 0;
}
