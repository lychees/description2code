int n,ar[10],s;
int ways;
vector<int> v1,v2;
int used[20];

void backtrack()
{
	int s1,s2;
	s2=s1=0;
	for (int i=0;i<v1.size();i++)
	 	s1+=ar[v1[i]];
	for (int i=0;i<v2.size();i++)
		s2+=ar[v2[i]];
	if (s1>s2)
		return ;
	if (v1.size()+v2.size()==n)
	{
		ways++;
		return;
	}
	for (int i=0;i<n;i++)
	{
		if (used[i])continue;
		used[i]=1;
		v1.push_back(i);
		backtrack();
		v1.pop_back();
		v2.push_back(i);
		backtrack();
		v2.pop_back();
		used[i]=0;
	}
}
