'''
# Read input from stdin and provide input before running code

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
demand = raw_input()
leng = len(demand)
tc = input()
while tc > 0:
    p, q = (int(i) for i in raw_input().split())
    p = p % leng
    q = q % leng
    if p == 0:
        p = leng
    if q == 0:
        q = leng
    if demand[p-1] == demand[q-1]:
        print 'Yes'
    else:
        print 'No'

    tc -= 1
