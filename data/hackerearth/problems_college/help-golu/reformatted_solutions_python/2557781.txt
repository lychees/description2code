from math import sqrt


def prime(n):
    t = 0
    for x in xrange(2, n):
        if n % x == 0:
            t = 1
            break
    if t == 0:
        return True
    else:
        return False


for _ in xrange(input()):
    n = input()
    if prime(n):
        print "1"
    elif n % 2 == 0 or prime(n-2):
        print "2"
    else:
        print "3"
