#include <bits/stdc++.h>
 
#define fi first
#define se second
#define pb push_back
#define mp make_pair
#define pii pair<int,int>
#define pll pair<ll,ll>
#define MOD 1000000007
#define ll long long int
#define MAXINT 2000000000
#define SET(x,y) memset(x,y,sizeof(x));
 
using namespace std;

ll DP[100005];
int A[100005];

int main() {
	int t,i,n,k;
	ll q;
	cin>>t;
	while(t--) {
		cin>>n>>k;
		for(i=1;i<=n;i++)	cin>>A[i];
		DP[0]=0;
		for(i=1;i<=n;i++) {
			DP[i] = DP[i-1];
			q = A[i];
			if(i>k)	q+=DP[i-k-1];
			DP[i] = max(DP[i],q);
		}
		cout<<DP[n]<<endl;
	}
	return 0;
}
