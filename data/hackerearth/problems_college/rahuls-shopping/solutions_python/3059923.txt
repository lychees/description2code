t=input()
a=map(int,raw_input().split())
b=map(int,raw_input().split())
fi=0

for i in range(0,t):
	c=str(a[i])
	temp=0
	for j in range(0,len(c)):
		temp=temp+int(c[j])
	temp=temp*a[i]*b[0]*b[1]*1.0
	temp2=temp*1.0
	if(a[i]>=1 and a[i]<=10):
		temp=temp/4
	elif(a[i]>=21 and a[i]<=50):
		temp=temp*20
		temp=temp/100
	elif(a[i]>=11 and a[i]<=20):
		temp=temp*30
		temp=temp/100
	fi=fi+int(temp2-temp)
print fi
