#include<bits/stdc++.h>

using namespace std;

int ans[100005];
int main(){
	vector<int> a;
	int t,n;
	cin >> t;
	for(int tp=1;tp<100005;tp++){
		int c=tp;
		a.resize(0);
		while(c!=0){
			a.push_back(c%2);
			c/=2;
		}
		for(int i=0;i<a.size()-1;i++){
			if(a[i]==1&&a[i+1]==1){
				ans[tp]=1;
				break;
			}
		}
	}
	while(t--){
		scanf("%d",&n);
		for(int i=n;i<100005;i++){
			if(!ans[i]){
				printf("%d\n",i);
				break;
			}
		}
	}
	return 0;
}
