#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll long long int

int prime[10000];

ll power(ll a,ll b)
{
	if(b==0)
	return 1;
	if(b==1)
	return a;
	if(b==2)
	return (a*a)%mod;
	
	if(b&1)
	return (a*power(a,b-1))%mod;
	
	return power(power(a,b/2),2);
}

int main()
{
	ll t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		memset(prime,0,sizeof(0)*10000);
		ll ans = 1;
		while(n--)
		{
			ll a;
			cin>>a;
			for(ll i = 2;i*i<=a;i++)
			{
				if(a%i==0)
				{
					int cnt = 0;
					while(a%i==0)
					{
						a = a/i;
						cnt++;
					}
					
					prime[i] = max(prime[i],cnt);
				}
			}
			
			if(a>1)
			prime[a] = max(1,prime[a]);
		}
		
		for(int i=0;i<10000;i++)
		ans = (ans*(power(i,prime[i])))%mod;
		cout<<ans<<endl;
	}
}
