#include <iostream>
using namespace std;
#define ll long long
int main()
{
    int t ;
    cin >> t;
    while(t--)
    {
    	int n; cin >> n;
    	ll a[n];
    	bool res =true;
    	for(int i=0;i<n;i++)
         {
         	cin >>a[i];
         	if(i && a[i] < a[i-1])
         	res = false;
         }
         if(res) cout<<"YES\n"; else cout<<"NO\n";
    }
    return 0;
}
