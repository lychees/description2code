#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;
long long x[1000001],p[1000000];
void f(){
    long i=0,j=0,r=0;
    for(i=2;i<=1000;i++){
        if(x[i]!=0)
            continue;
        for(j=i*i;j<=1000000;j=j+i){
            if(x[j]==0){
                x[j]=1;
                p[i]++;
            }
        }
    }
}
long long gcd(long long a,long long b){
    if(b==0)
        return a;
    else
        return gcd(b,a%b);
}
int main() {
long long t,n;
    f();
    cin>>t;
    while(t--){
        cin>>n;
        cout<<p[n]+1<<"\n";
    }
    return 0;
}
