#include <bits/stdc++.h>
using namespace std;
int a[100000];
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int flag[100000]={0},n,k=0;
		scanf("%d",&n);
		for(int i=0;i<n;i++)
			scanf("%d",&a[i]);
		while(flag[k]==0 && k<n && a[k]!=-1)
		{
			flag[k]=1;
			k=a[k];
		}
		if(k<n && a[k]==-1)
			printf("Win\n");
		else
			printf("Lose\n");
	}
    return 0;
}
