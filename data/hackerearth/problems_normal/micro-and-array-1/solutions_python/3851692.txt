nq = map(int, raw_input().split())
num = map(int, raw_input().split())

def find_index(v):
	for i in xrange(nq[0]):
		if v <= num[i]:
			return i+1;
	return -1;

def update(x, y):
	num[x-1] = y

for q in xrange(nq[1]):
	query = map(int, raw_input().split())
	if query[0]:
		print find_index(query[1])
	else:
		update(query[1], query[2])
