#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	
	while(t--){
		string a;
		cin>>a;
		
		int f=0;
		for(int i=0;i<a.size()/2;i++)if(a[i]!=a[a.size()-i-1]){
			f=1;
			break;
		}  
		if(f==1) cout<<-1<<endl;
		else{
			
				cout<<(int)a[a.size()/2]<<endl;
			
		}
	}
    return 0;
}
