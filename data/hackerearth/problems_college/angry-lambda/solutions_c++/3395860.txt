#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    
    long long i, count, t, n, k;
    cin>>t;
    while(t--){
    	count=0;
    	cin>>n>>k;
    	int a[n];
    	for(i=0;i<n;i++){
    		cin>>a[i];
    		if(a[i]<=0)
    			count++;
    		
    	}
    	
    	if(count>=k)
    		cout<<"NO"<<endl;
    	else
    		cout<<"YES"<<endl;
    	
    }
    
    return 0;
}
