#include <iostream>
using namespace std;

int main()
{
    long long N,a,b,count;
    scanf("%lld",&N);
    while(N-->0){
    	cin>>a>>b;
    	count=0;
    	while(a!=0&&b!=0){
    		int s1=a%10;
    		int s2=b%10;
    		if(s1+s2>9){
    			count++;
    		}
    		a/=10;
    		b/=10;
    	}
    	if(count==0)
    		printf("No carry operation\n");
    	else if(count==1)
    		printf("1 carry operation\n");
    	else
    		printf("%d carry operations\n",count);
    }
    return 0;
}
