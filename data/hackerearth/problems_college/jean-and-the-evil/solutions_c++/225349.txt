#include<iostream>
#include<sstream>
#include<string>
#include<algorithm>
#include<vector>
#include<stdio.h>
using namespace std;
int main()
{
 int t;
 cin>>t;
 while(t--)
    {
       int n,k;
       cin>>n;
       vector<int> p;
       int i;
       for(i=0;i<n;i++)
       {
           scanf("%d",&k);
           p.push_back(k);
       }

       long long int ans=0,stn=0;
       for(i=0;i<n;i++)
       {
           stn=stn+p[i];
           if(stn>=0)
            ans=max(stn,ans);
           else
            stn=0;
       }
    if(ans==0)
        ans=*max_element(p.begin(),p.end());
    printf("%lld\n",ans);
    }
return 0;
 }
