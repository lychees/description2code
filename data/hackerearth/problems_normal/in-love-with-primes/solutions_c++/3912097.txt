#include<iostream>

using namespace std;
int main()
{
	int h;
	cin>>h;
	for(int i=0;i<h;i++)
	{
		int c=0;
		int j;
		cin>>j;
		for(int q=1;q<=j;q++)
		{
			if(j%q==0)
			c++;
		}
		if(c==2)
		{cout<<"Arjit"<<endl;
		}
		else
		{
			cout<<"Deepa"<<endl;
		}
		
	}

	return 0;
}
