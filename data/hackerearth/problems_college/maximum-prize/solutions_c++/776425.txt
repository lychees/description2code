#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
    int t;
    cin>>t;
    long n,k,i;
    long long s;
    while(t--)
    {
    	s = 0;
    	cin>>n>>k;
    	long a[n];
    	
    	for(i=0;i<n;i++)cin>>a[i];
    	
    	sort(a,a+n);
    	
    	if(k>n)
    	{
    		for(i=n-1;i>=0;i--)s+=a[i];
    	}
    	else
    	{
    		for(i=n-1;i>=n-k;i--)s+=a[i];
    	}
    	cout<<s<<"\n";
    }
    return 0;
}
