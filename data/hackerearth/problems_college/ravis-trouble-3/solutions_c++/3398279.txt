#include <bits/stdc++.h>
using namespace std;
char s[10001],p[10001];
int main()
{
	int t1;
	cin>>t1;
	while(t1--)
	{
		cin>>s>>p;
		int j=0,i=0;
		for(j=0,i=0;j<strlen(p)&&i<strlen(s);)
		{
			if(s[i]==p[j])
			{
				i++;j++;
			}
			else
				i++;
		}
		if(j==strlen(p))
			cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;
	}
	return 0;
}
