T = input()
for _ in xrange(T):
    N = input()
    iDict = {}
    line = raw_input().strip().split()
    i = 0
    for a in line:
        i = int(a)
        if i in iDict:
            iDict[i] += 1
        else:
            iDict[i] = 1

    modeList = []
    maxCount = 0
    for curI, curC in iDict.iteritems():
        if curC > maxCount:
            maxCount = curC
            modeList = [curI]
        elif curC == maxCount:
            modeList.append(curI)
    output = sorted(modeList, reverse=True)
    print ' '.join(map(str, output))
