LL dp[1<<11][20];
vector<string> arr;
string reqd;

int k;

LL solve(int mask , int index)
{
	if(index == k) return 1;
	else if(dp[mask][index] != -1) return dp[mask][index];
	LL ret = 0;
	for(int i = 0; i < sz(arr); i++)
	{
		if(mask&(1<<i)) continue;
		int occ = 0;
		for(int j = 0; j < 6; j++)
			if(arr[i][j] == reqd[index]) occ += 1;
		if(occ)
			ret = (ret + 1LL * occ * solve(mask|1<<i , index + 1)) % MOD;
	}
	return (dp[mask][index] = ret);
}

	FOR(i,0,n)

		string a = "";
		FOR(j,0,6)
		{
			char c;

			a += c;
		}
		arr[i] = a;
	}
	cin >> reqd;
	FOR(i,0,(1<<11)) FOR(j,0,20) dp[i][j] = -1;
	cout << solve(0 , 0) << endl;
	return 0;
}
