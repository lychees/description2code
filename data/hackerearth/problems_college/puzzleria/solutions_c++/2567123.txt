#include <iostream>
using namespace std;
int com(int n,int m)
{
	if(m==0||n==m)
		return 1;
		else
		return com(n-1,m-1)+com(n-1,m);
}
int fact(int n)
{
	if(n==1||n==0)
	return 1;
	if(n==2)
	return 2;
	else
	return n*fact(n-1);
}
int main()
{
    int t,total;
    cin>>t;
    while(t--)
    {
    	long long c,v;
    	cin>>c>>v;
    	total=fact(c+v)*com(7,c)*com(5,v);
   	cout<<total<<endl;
   }
}
