#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

int dp[102],cnt[1002];

int main() 
{
	 int n,i,j,el;	
	 cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>el;
		cnt[el]++;
	}
	dp[0]=0;
	for(i=1;i<=100;i++)
	{
		dp[i]=dp[max(i-2,0)]+cnt[i]*i;
		dp[i]=max(dp[i],dp[i-1]);
	}
	cout<<dp[100]<<"\n";
    
	return 0;
}
