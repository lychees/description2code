#include<bits/stdc++.h>
using namespace std;

#define lli long long
#define pb push_back
#define mp make_pair
#define gc getchar_unlocked
#define lld "%I64d"
#define DEBUG(x) cout<<">value ("<<#x<<") : "<<x<<endl;
#define mod 1000000007

const int lmt = 1000005;
lli dp[lmt][3];

int main(){
    int t,n;
    lli a,b,c;
    scanf("%d",&t);
    while(t--){
        scanf("%d",&n);
        memset(dp,0,sizeof(dp));
        scanf("%lld %lld %lld",&dp[1][0],&dp[1][1],&dp[1][2]);
        for(int i=2;i<=n;i++){
            scanf("%lld %lld %lld",&a,&b,&c);
            dp[i][0]=min(dp[i-1][1],dp[i-1][2]) + a;
            dp[i][1]=min(dp[i-1][0],dp[i-1][2]) + b;
            dp[i][2]=min(dp[i-1][0],dp[i-1][1]) + c;
        }
        lli ans = min({dp[n][0],dp[n][1],dp[n][2]});
        printf("%lld\n",ans);
    }
    return 0;
}
