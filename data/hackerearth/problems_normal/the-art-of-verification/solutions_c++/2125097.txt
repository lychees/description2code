#include <iostream>
using namespace std;

int main()
{	long long l,i,k=0;
	string s;
    cin>>s;
    l=0;
    while(s[l]!='\0')
    	++l;
    for(i=0;i<l;++i)
    {	if(k==1)
    	{	if(s[i]=='&'&&((s[i+1]=='p'&&s[i+2]=='w')||(s[i+1]=='p'&&s[i+2]=='r')||(s[i+1]=='r'&&s[i+2]=='o')||(s[i+1]=='k'&&s[i+2]=='e')))
    			cout<<"\n";
    		else if(s[i]=='='&&((s[i-1]=='e'&&s[i-2]=='m')||(s[i-1]=='d'&&s[i-2]=='w')||(s[i-1]=='e'&&s[i-2]=='l')||(s[i-1]=='y'&&s[i-2]=='e')))
    			cout<<": ";
    		else
    			cout<<s[i];
    	}
    	if(s[i]=='?')
    		k=1;
    }
    return 0;
}
