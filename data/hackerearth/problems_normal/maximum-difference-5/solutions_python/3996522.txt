import math
n = int(raw_input())

a = map(int,raw_input().split())

a = sorted(a)

b = []
c = []

for i in xrange(n/2):
    b.append(a[i])
    b.append(a[n-1-i])
    c.append(a[n-1-i])
    c.append(a[i])

if n%2 != 0 :
    b.append(a[n/2])
    c.append(a[n/2])

ans = 0
ans1 = 0
#print b,c
for i in xrange(n-1):
    ans += abs(b[i]-b[i+1])
    ans1 += abs(c[i]-c[i+1])

print max(ans,ans1)
