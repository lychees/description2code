#include<iostream>
#include<algorithm>
using namespace std;
typedef long long ll;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		ll n,sum=0;
		cin>>n;
		ll i,a[n];
		for(i=0;i<n;i++)
		{
			cin>>a[i];
		}
		sort(a,a+n);
		for(i=1;i<=n;i=i+2)
		{
			sum=sum+a[i-1]*a[i];
		}
		cout<<sum<<endl;
	}
}
