import operator

t = int(raw_input())
while t:
	s = raw_input()
	d = {}
	i = 0
	while i <= 26:
		d[97+i] = 0
		i += 1
	for c in s:
		d[ord(c)] += 1
	d = sorted(d.items(), key=operator.itemgetter(1))
	i = 0
	ans = ""
	for k, v in d:
		if v != 0:
			while v:
				ans += chr(k)
				v -= 1
	print ans
	
	t -= 1
