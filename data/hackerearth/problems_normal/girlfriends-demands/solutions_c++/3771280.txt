#include <stdio.h>
#include<string.h>
 
int main()
{
    char s[1000000];
    int t,p;
    unsigned long long int a,b;
    scanf("%s",&s);
    p=strlen(s);
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%ld %ld",&a,&b);
    	if(a>p)
    		a=(a-1)%p;
    	if(b>p)
    		b=(b-1)%p;
    	//a--,b--;
    	if (s[a]==s[b])
    		printf("Yes\n");
    	else
    		printf("No\n");
    }
    return 0;
}
