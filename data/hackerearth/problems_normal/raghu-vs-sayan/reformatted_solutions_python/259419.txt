t = int(raw_input())
for i in range(t):
    con = map(int, raw_input().split())
    a = con[0]
    b = con[1]
    n = con[2]
    dis_cal = map(int, raw_input().split())
    dis_cal = sorted(dis_cal)
    c1 = 0
    c2 = 0
    for k in dis_cal:
        if k <= a:
            a = a - k
            c1 += 1
        if k <= b:
            b = b - k
            c2 += 1
    if c1 == c2:
        print "Tie"
    elif c1 > c2:
        print "Raghu Won"
    else:
        print "Sayan Won"
