a = raw_input()
b = a.find("username=")
c = a.find("pwd=")
d = a.find("profile=")
e = a.find("role=")
f = a.find("key=")
b1 = ""
c1 = ""
d1 = ""
e1 = ""
f1 = ""
for m in range(b+9, c-1):
    b1 += a[m]
for n in range(c+4, d-1):
    c1 += a[n]
for o in range(d+8, e-1):
    d1 += a[o]
for p in range(e+5, f-1):
    e1 += a[p]
for q in range(f+4, len(a)):
    f1 += a[q]
print "username:", b1
print "pwd:", c1
print "profile:", d1
print "role:", e1
print "key:", f1
