#include<bits/stdc++.h>
using namespace std;
main()
{
	int t;
	cin>>t;
	while(t--)	
	{
		int n;
		cin>>n;
		int count=0;
		for(int i=1;i<=n;i++)	
		{
			for(int j=1;j<=n;j++)
			{
				int sum=i+j;
				long long int product=i*j;
				if(product%sum==0 && i<j)
				{
					count++;
				}
			}
		}
		cout<<count<<endl;
	}
}
