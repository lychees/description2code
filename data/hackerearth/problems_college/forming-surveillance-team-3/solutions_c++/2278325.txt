#include<bits/stdc++.h>
#include<stdlib.h>

using namespace std;

int srch(int a[], int mx, int l)
{
    for(int i = 0; i < l; i++) {
        if(a[i] == mx) {
            return 0;
        }
    }
    return 1;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int i = 0;
        int m;
        cin>>m;
        int a[m];
        while(m--)
        {
            int mx;
            cin>>mx;
            if(srch(a,mx,i)) {
                a[i++] = mx;
            }
        }
        cout<<i<<endl;
    }
    return 0;
}
