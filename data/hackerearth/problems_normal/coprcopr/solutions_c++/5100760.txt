#include<bits/stdc++.h>
#define ll long long int
#define s(n) scanf("%lld",&n);
#define MAXN 10000000
using namespace std;
 
ll pr[MAXN+31], phi[MAXN+31], S[MAXN+31];

void precompute(){
	pr[1]=1;
	for (ll i=2;i<=MAXN;i++)
		if (pr[i]==0)
			for (ll j=i*2;j<=MAXN;j+=i)
				pr[j]=i;
	
	phi[1]=1;
	
	for (ll i=2;i<=MAXN;i++)
	{
		ll q=i;
		ll val=pr[q];
		if (val==0)
			phi[i]=q-1;
		else
		{
			while (q%val==0)
				q/=val;
			phi[i]=phi[q]*(i/q)/val*(val-1);
		}
	}
	// precompute sum of phi
	for (ll i=1;i<=MAXN;i++)
		S[i]=S[i-1]+phi[i];
}
int main(){
	precompute();
	ll t, n;
	s(t);
	while(t--)
	{
		s(n);
		printf("%lld\n",S[n]);
	}
	return 0;
}
