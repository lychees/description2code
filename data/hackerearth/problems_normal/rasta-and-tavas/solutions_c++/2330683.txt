#include <iostream>
#include <bits/stdc++.h>

using namespace std;

int s[1000000];

int main()
{
    s[0]=s[1]=0;
    for(int i=2;i<1000000;i++)
        if(s[i]==0)
            for(int j=i;j<1000000;j=j+i)
                s[j]=s[j]+i;
    int n,k,ans=0;
    cin>>n>>k;

    for(int i=0;i<=n;i++)
        if(s[i]==k)
            ans++;

    cout<<ans<<endl;
    return 0;
}
