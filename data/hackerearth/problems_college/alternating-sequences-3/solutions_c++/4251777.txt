#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int n;
    cin>>n;
    vector<int> a(n),lis(n,0);
    for(int i=0;i<n;i++){
    cin>>a[i];
    	
    }
    lis[0]=1;
    int maxim=1;
    for(int i=1;i<n;i++)
    {
    	int flag=0;
    	for(int j=0;j<i;j++)
    	{
    		if(a[i]>0 && a[j]<0 && abs(a[j])<abs(a[i]) &&  lis[i]<lis[j]+1){
    			
    		if(flag<lis[j])
    		flag=lis[j];
    		}
    		else if(a[i]<0 && a[j]>0 && abs(a[j])<abs(a[i]) && lis[i]<lis[j]+1){
    			if(flag<lis[j])
    		flag=lis[j];
    		
    		}
    	}
    	lis[i]=flag+1;
    	if(maxim<lis[i])
    	maxim=lis[i];
    }
    
    cout<<maxim<<"\n";
    
    return 0;
}
