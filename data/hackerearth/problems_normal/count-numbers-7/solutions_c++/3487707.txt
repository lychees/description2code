#include <bits/stdc++.h>;
using namespace std;
int n;
int a,b;
int arr[11];
int solve(){
	long long ans = 0;
	for(int i=1;i<(1<<n);i++){
		long long temp = 1;
		int tn = 0;
		for(int j=0;j<n;j++){
			if( (i&(1<<j))==0 )
				continue;
			if( temp*arr[j] > b || temp*arr[j]/arr[j] != temp ){
				tn = -1;
				break;
			}
			temp *= arr[j];
			tn++;
		}
		if(tn == -1)
			continue;
		long long t1 = 0, t2 = 0;
		long long low = 1 , high = b;
		while(low < high){
			long long mid = (low+high)/2;
			if( mid*temp < a ){
				low = mid+1;
			}else{
				high = mid;
			}
		}
		if( temp*low < a )
			low ++;
		t1 = low;
		low = 1 , high = b;
		while(low < high){
			long long mid = (low+high)/2;
			if( mid*temp < b ){
				low = mid+1;
			}else{
				high = mid;
			}
		}
		if( temp*(low+1) <= b )
			low++;
		if( temp*low > b )
			low--;
		t2 = low;
		if( tn%2 == 0 )
			ans -= (t2-t1+1);
		else
			ans += (t2-t1+1);
	}
	return ans;
}
int main()
{
	int T;
	cin >>n>> T;
	for (int i=0;i<n;i++)
		cin >> arr[i];
	while(T--){
		cin >> a >> b;
	
		cout << solve()<<endl;
	}
    return 0;
}
