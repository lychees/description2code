#include <string.h>
#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
#define LL long long
#define pb push_back
#define MAX 1000010
int res[MAX];
int min(int a, int b){
    if(a < b)
         return a;
    return b;
}
int main()
{
    int n,k;
    cin>>n>>k;
    int marks[n][k];
    for(int i=0 ; i<n ; i++)
            for(int j=0 ; j<k ; j++)
                    cin>>marks[i][j];

    
    int q;
    cin>>q;
    while(q--){
               int cnt;
               cin>>cnt;
               int prior[cnt];
               for(int i=0 ; i<cnt ; i++)
                       cin>>prior[i];
               vector<int> v;
               for(int i=0 ; i<n ; i++)
                       v.pb(i);
               for(int i=0 ; i<cnt ; i++)
               {
                        vector<int> temp;
                        int sub = prior[i]-1;
                        int min_score = 1000;
                        for(int i = 0 ; i<v.size() ; i++)
                                min_score = min(min_score, marks[v[i]][sub]);
                        //cout<<min_score<<endl;
                        for(int i=0 ; i<v.size() ; i++)
                        {
                                if(marks[v[i]][sub] == min_score)
                                                 temp.pb(v[i]);
                        }
                        v = temp;
                        if(v.size() == 1)
                                    break;
               }
               cout<<v[0]+1<<endl;
    }              
    
    
    //system("pause");
    return 0;  
     
    
}
