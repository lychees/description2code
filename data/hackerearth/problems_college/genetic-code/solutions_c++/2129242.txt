#include <iostream>
#include <string>
using namespace std;
int main(){
	ios::sync_with_stdio(false);
	string in = "";
	
	cin >> in;
	
	int len = in.length();
	int sequenceLength = 1;
	
	int count = 0;
	
	for(int i = 1; i < len; i++)
	{
	  if(in[i] == in[i - 1] and i == len - 1)
	   sequenceLength+=2;
	  if(in[i] == in[i - 1])
	   sequenceLength++;
	  if(in[i] != in[i - 1] or i == len - 1)
	  {
	   if(sequenceLength % 2 == 0)
	    count++;
	   sequenceLength = 1;
	  }
	}
	
	cout << count << endl;
	
	return 0;
}
