#include<iostream>
using namespace std;

int arr[1001][1001],n;
int func(int i,int j)
{
	int t=1;
	arr[i][j]=2;
	if(i+1<n and arr[i+1][j]==1)t+=func(i+1,j);
	if(i-1>=0 and arr[i-1][j]==1)t+=func(i-1,j);
	if(j+1<n and arr[i][j+1]==1)t+=func(i,j+1);
	if(j-1>=0 and arr[i][j-1]==1)t+=func(i,j-1);
	return t;
}
void solve()
{
	int ans=0;
	for(int i=0;i<n;i++)
	for(int j=0;j<n;j++)
	{
		if(arr[i][j]==1)ans=max(ans,func(i,j));
	}
	cout<<ans;
}
int main()
{
	cin>>n;
	for(int i=0;i<n;i++)
	for(int j=0;j<n;j++)cin>>arr[i][j];
	solve();
	return 0;
}
