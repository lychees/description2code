#include <iostream>
#include <cmath> 
using namespace std;
int main()
{
	int n,curr,prev,i;
	cin >> n;
	int ans = 1,max = 1;
	cin >> prev;
	for(i=1;i<n;i++)
	{
		cin >> curr;
		if(abs(curr-prev) == 1)
			ans++;
		else
		{
			ans = 1;
		}
		if(max<ans)
				max = ans;
		prev = curr;
	}
	
	cout << max;
	return 0;
}
