int
fill_arr (unsigned long long num, unsigned long long *arr)
{
  while (num != 1)
    {
      if (num % 2 == 0)
	{
	  arr[2]++;
	  num = num / 2;
	}
      else if (num % 3 == 0)
	{
	  arr[3]++;
	  num = num / 3;
	}
      else if (num % 5 == 0)
	{
	  arr[5]++;
	  num = num / 5;
	}
      else if (num % 7 == 0)
	{
	  arr[7]++;
	  num = num / 7;
	}
      else
	return -1;
    }
  return 0;
}
