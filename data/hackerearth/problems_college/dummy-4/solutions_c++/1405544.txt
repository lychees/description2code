#include<iostream>
#include<stdlib.h>
#include<string>
using namespace std;
int main()
{
	int n,i,j;
	cin >> n;
	string s[n];
	for (i=0;i<n;i++)
	{
		cin>>s[i];
	}
	
	unsigned long long int ans=0,sum,num;
	for (i=0;i<n;i++)
	{
		for (j=0;j<s[i].length();j++)
		{
			ans = ans + (s[i][j]-'0');
		}
	}
	
	while (ans <0 || ans>9 )
	{
		num = ans;
		sum = 0;
		while (num != 0)
    	{
       	 	sum = sum + num % 10;
        	num = num / 10;
    	}
    	
		ans = sum;		
	}
	cout << ans;
	return 0;
}
