#include<iostream>
#include<cmath>
using namespace std;


int sumOfDigits(long int num)	//finds the sum of digits of number
{
	int sum=0;	//initialize to 0
	long int r;
	while(num)	//continue till num doesn't become 0
	{
      r=num%10;
      num=num/10;
      sum=sum+r;
  	}
  	return sum;
}
int main()
{
	int t;
	cin>>t;
	
	long long int N;
	while(t--)	//for each test case
	{
		cin>>N;
		long int r = sqrt(N);
		bool flag = false;
		int count = 0;
		
		long int i;
		for(i=r;i>0;i--)
		{
			count++;
			if(N%i==0)
			{
				if(i*(i+sumOfDigits(i))==N)
				{
					flag = true;
					break;
				}
			}
			if(count>400)
				break;
		}
		if(flag == false)
		{
			cout<<-1<<endl;
		}
		else
		{
			cout<<i<<endl;
		}
	}
	return 0;
}
