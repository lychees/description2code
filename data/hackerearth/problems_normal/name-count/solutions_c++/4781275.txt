#include <iostream>
#define m 1000000007
#define ll long long int
using namespace std;

ll power(ll a ,ll b)
{
	ll ans=1;
	while(b!=0)
	{
		if(b%2==1){
			ans*=a;
			ans%=m;
		}
		a*=a;
		a%=m;
		b/=2;
	}
	return ans;

}

ll fact(ll num)
{
	if(num==0)
	return 1;
	else
	return ((num*fact(num-1)))%m;
}

int main()
{
    int n,k,t;
    cin>>t;
    while(t--)
    {
    	cin>>n>>k;
    	ll val1,val2,val3;
    	val1=fact(k);
    	val1%=m;
    	val2=fact(k-n);
    	val3=power(val2,m-2);
    	val3%=m;
    	ll num = (val1*val3)%m;
    	
    	cout<<num<<endl;
    	
    	
    }
    return 0;
}
