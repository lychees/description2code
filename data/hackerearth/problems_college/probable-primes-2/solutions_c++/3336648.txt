#include <bits/stdc++.h>

using namespace std;

int ans[1000005] = {0};

void fun()
{
    ans[0] = 0;
    ans[1] = 0;

    for (int i = 2;i <= 1000000;i++)
    {
        int k = 0;
        for (int j = 2;j <= sqrt(i);j++)
        {
            if (i % j == 0)
            {
                k = 1;
                break;
            }
        }

        if (k == 0)
            ans[i] = ans[i-1] + 1;
        else
            ans[i] = ans[i-1];
    }
}

int main()
{
    fun();

    int tc;
    cin >> tc;
    while(tc--)
    {
        long int a,b;
        cin >> a >> b;
        float num = float(b) - float(a) + 1.000000;
        long int sum = ans[b] - ans[a-1];

        printf("%0.6f\n",sum /(num));
    }

    return 0;
}
