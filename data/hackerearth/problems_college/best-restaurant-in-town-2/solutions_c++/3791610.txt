#include<iostream>
#include<cmath>
using namespace std;
int main()
{
    long long t,i,j,n,c,r;
    cin>>t;
    while(t--)
    {
        cin>>n>>c>>r;
        long long preptime[n],dish[c],artime[c],totaltime=0,x;
        for(i=0;i<n;i++)//inputs preparation time
        {
            cin>>preptime[i];
        }
        for(i=0;i<c;i++)//dish each customer ordered
        {
            cin>>dish[i];
        }
        for(i=0;i<c;i++)//arrival time
        {
            cin>>artime[i];
        }
        //cout<<totaltime<<" ";
            totaltime=preptime[dish[0]-1];
            cout<<totaltime;
            totaltime+=r;
            for(j=1;j<c;j++)
            {
            if(totaltime+artime[j-1]<artime[j])
                totaltime=0;
            else
            {
            x=artime[j]-artime[j-1];
                    totaltime=totaltime-x;
                    //cout<<totaltime;
                }
            totaltime+=preptime[dish[j]-1];
                cout<<" "<<totaltime;
                totaltime+=r;
            }
        cout<<"\n";
    }
}
