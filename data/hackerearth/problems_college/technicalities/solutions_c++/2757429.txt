#include <iostream>
#include <cstdio>
using namespace std;
char input[110000];

int main()
{
    int T,n;
    cin >> T;
    while(T--)
    {
        cin >> n;
        cin >> input;

        long long r = 0;
        for (int i = 0; input[i] != '\0'; ++i)
            r = (r * 10 + input[i] - '0') % 1000000007;
        cout << r << endl;
    }
    return 0;
}
