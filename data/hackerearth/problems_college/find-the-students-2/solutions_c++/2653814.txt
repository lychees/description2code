#include<iostream>
#include<bits/stdc++.h>
using namespace std;
main()
{
	long long n,m,k,ct=0,c=0,j=0;
	cin>>n;
	long long a[n];
	for(int i=0;i<n;i++)
		cin>>a[i];
	m=n;
	while(1)
	{
		sort(a,a+n);
		j=c;
		if(j==n) break;
		ct=0;c=0;k=a[j];
		for(int i=0;i<n;i++)
		{
			if(a[i]>0)
			{
				ct+=1;
			//	cout<<k<<endl;
				a[i]=a[i]-k;
			//	cout<<a[i]<<endl;
				if(a[i]<=0)
					c+=1;
			}
			else
				c+=1;		
		}
		cout<<ct<<endl;
	}
}
