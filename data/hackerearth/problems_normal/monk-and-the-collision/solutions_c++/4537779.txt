#include <bits/stdc++.h>
using namespace std;

int main()
{
   int n,t;
   cin>>t;
   while(t--)
   {
   	int a[10];
   	memset(a,0,sizeof(a));
   	int n;
   	cin>>n;
   	int temp;
   	int ans=0;
   	for(int i=0;i<n;i++)
   	{
   		cin>>temp;
   		if(a[temp%10]==1)
   		ans++;
   		a[temp%10]=1;
   	}
   	cout<<ans<<"\n";
   }
   
   
    return 0;
}
