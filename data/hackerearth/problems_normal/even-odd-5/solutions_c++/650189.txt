#include <bits/stdc++.h>
using namespace std;

int main()
{
	long long int t;
	cin>>t;
	long long int n,m;
	long long int sum,odd,even_sum,odd_sum ,temp;
	while(t--)
	{
		cin>>n>>m;
		if(n<=m)
		{
			sum= (n*(n+1))/2;
			if(n%2==0)
			odd=n/2;
			else
			odd=(n/2)+1;
			cout<<sum-(odd*odd)<<" "<<(odd*odd)<<endl; 
		}
		else
		{
			sum=(m*(m+1))/2;
			if(m%2==0)
			odd=m/2;
			else
			odd=(m/2)+1;
			even_sum=sum-(odd*odd);
			odd_sum=odd*odd;
			temp=n-m;
			if(temp%2==0)
			{
				even_sum+=(temp/2)*m;
				odd_sum+=(temp/2)*m;
			}
			else
			{
				if(m%2==0)
				{
					odd_sum+=((temp/2)+1)*m;
					even_sum+=(temp/2)*m;
				}
				else
				{
					even_sum+=((temp/2)+1)*m;
					odd_sum+=(temp/2)*m;
				}
			}
			cout<<even_sum<<" "<<odd_sum<<endl;
		}
		
	}
	return 0;
    
}
