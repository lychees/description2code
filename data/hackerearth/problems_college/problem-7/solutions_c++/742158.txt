#include<iostream>
using namespace std;
 
unsigned long int binomialCoeff(unsigned int n, unsigned int k)
{
    unsigned long int res = 1;
 
    
    if (k > n - k)
        k = n - k;
 
  
    for (int i = 0; i < k; ++i)
    {
        res *= (n - i);
        res /= (i + 1);
    }
 
    return res;
}


int main()
{
    long long n;
    cin>>n;
    long long ans=binomialCoeff(n+2,3);
    //long long ans2;
    if(n%2) 	
    	ans=ans+((n*n)-1)*((2*n)+3)/24;
    else
    	ans=ans+n*(n+2)*((2*n)-1)/24;
    cout<<ans<<"\n";
     return 0;
}
