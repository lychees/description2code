#include<bits/stdc++.h>
using namespace std;
int lcs[5005][5005];
int main()
{
    string a,b;
    cin>>a>>b;

    string a1="",b1="";
    int n=a.length();
    int m=b.length();
    for(int i=0;i<n;i++)
    {
        if(a[i]=='a'|| a[i]=='e'||a[i]=='i'||a[i]=='o'||a[i]=='u')
        {
            continue;
        }
        else a1+=a[i];
    }

    for(int i=0;i<m;i++)
    {
        if(b[i]=='a'|| b[i]=='e'||b[i]=='i'||b[i]=='o'||b[i]=='u')
        {
            continue;
        }
        else b1+=b[i];
    }

    n=a1.length();
    m=b1.length();

    memset(lcs,0,sizeof(lcs));

    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m;j++)
        {
            if(a1[i]==b1[j])
                lcs[i+1][j+1]=lcs[i][j]+1;
            else lcs[i+1][j+1]=max(lcs[i+1][j],lcs[i][j+1]);
        }
    }
    cout<<lcs[n][m]<<endl;
}
