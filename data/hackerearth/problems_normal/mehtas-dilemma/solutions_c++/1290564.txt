vector<vector<long long> > one(long sz)
{
 vector<vector<long long> > res;
 res.resize(sz);
 for (int i=0;i<sz;i++)
  res[i].resize(sz);
 for (int i=0;i<res.size();i++)
  for (int j=0;j<res.size();j++)
   res[i][j]=(i==j);
  return res;
}

vector<vector<long long> > mult(vector<vector<long long> > a,vector<vector<long long> > b)
{
 vector<vector<long long> > res;
 res.resize(a.size());
 for (int i=0;i<res.size();i++)
 {
  res[i].resize(a.size());
  for (int j=0;j<res[i].size();j++)
   res[i][j]=0;
 }
 
 for (int i=0;i<a.size();i++)
  for (int k=0;k<a.size();k++)
   for (int j=0;j<a.size();j++)
    res[i][j]=(res[i][j]+a[i][k]*b[k][j])%bs;
 return res;
}

vector<vector<long long> >pw(vector<vector<long long> > a,long long b)
{
 if (b==0)return one(a.size());
 if (b%2)return mult(a,pw(a,b-1));
 return pw(mult(a,a),b/2);
}

long gmask(long x)
{
 long res=0;
 while (x%2==0){res^=1;x/=2;}
 while (x%3==0){res^=2;x/=3;}
 while (x%5==0){res^=4;x/=5;}
 while (x%7==0){res^=8;x/=7;}
 return res;
}
