#include<iostream>
#include<cstring>
using namespace std;
 
// A utility function to find maximum of two integers
int max(int a, int b)
{
    return (a > b)? a : b;
}
 
int lcs( char *X, char *Y, int m, int n )
{
   int AKH[m+1][n+1];
   int i, j;
   for (i=0; i<=m; i++)
   {
     for (j=0; j<=n; j++)
     {
       if (i == 0 || j == 0)
         AKH[i][j] = 0;
 
       else if (X[i-1] == Y[j-1])
         AKH[i][j] = AKH[i-1][j-1] + 1;
 
       else
         AKH[i][j] = max(AKH[i-1][j], AKH[i][j-1]);
     }
   }
   return AKH[m][n];
}
 
/*Main */
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int x;
		cin>>x;
	    char X[x],Y[x];
	    cin>>X>>Y;
	    int m = strlen(X);
	    int n = strlen(Y);
	    cout <<lcs(X, Y, m, n)<<endl;
	}
    return 0;
}
