int d[502][502];

int main()
{  for(int i=0;i<=500;i++)
    {
        for(int j=0;j<=500;j++)
            {if(i!=j)d[i][j] = 100000000;
            else
            d[i][j]=0;}
    }
    int n;
    cin>>n;
    for(int i=1;i<=n;i++)
    {  
        cin>>a>>b>>c;
        d[a][b] = c;
        d[b][a] = c;
 
    }
    for(int k=0;k<=500;k++)
    {
        for(int i=0;i<=500;i++)
        {
            for(int j=0;j<=500;j++)
                d[i][j] = min(d[i][j],d[i][k]+d[k][j]);
        }
    }
    int in;
    cin>>in;
    int x;
    int q;
    cin>>q;
    while(q--)
    {
        cin>>x;
        if(d[in][x]==100000000)
            cout<<"NO PATH\n";
        else
            cout<<d[in][x]<<endl;
    }
return 0;
}
