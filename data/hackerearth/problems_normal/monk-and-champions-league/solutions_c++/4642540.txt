#include<bits/stdc++.h>
using namespace std;
typedef long long int lld;
priority_queue<int> q;

int main()
{
    int m,n,i,a,b;
    lld sum=0;
    cin>>m>>n;
    for(i=0;i<m;i++)
    {
        cin>>a;
        q.push(a);
    }
    while(n--)
    {
        b=q.top();
        q.pop();
        sum+=b;
        q.push(b-1);
    }
    cout<<sum;
}
