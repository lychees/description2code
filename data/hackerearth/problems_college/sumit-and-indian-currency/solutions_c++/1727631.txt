#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	string s,s1="";
    	int i,c=0;
    	cin>>s;
    	if(s.size()<=3) cout<<s<<endl;
    	else
    	{
    		s1+=s[s.size()-1];
    		for(i=s.size()-2;i>=0;i--)
    		{
    			c++;
    			s1+=s[i];
    			if(c%2==0)
    			{
    				//s1+=s[i];
    				s1+=",";
    				
    			}
    			
    		}
    		if(s.size()%2==0)
    		{
    		for(i=s1.size()-1;i>=0;i--)
    		cout<<s1[i];
    		cout<<endl;
    		}
    		else
    		{
    		for(i=s1.size()-2;i>=0;i--)
    		cout<<s1[i];
    		cout<<endl;
    		}
    	}
    }
    return 0;
}
