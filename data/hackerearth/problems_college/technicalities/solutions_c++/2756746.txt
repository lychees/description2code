#include <iostream>
#define ll long long int
#define M 1000000007
using namespace std;

int main()
{
	int t,n,i,num;
	string arr;
	int l;
	ll rem;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	cin>>arr;
    	l=arr.length();
    	rem=0;
    	for(i=0;i<l;i++)
    	{
    		
    		num=arr[i]-'0';
    		rem=(10*rem+num)%M;
    	}
    	cout<<rem<<"\n";
    }
    return 0;
}
