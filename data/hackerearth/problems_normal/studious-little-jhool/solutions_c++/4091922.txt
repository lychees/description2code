#include <bits/stdc++.h>

using namespace std;

int main()
{
	int tc;
	cin >> tc;
	
	while(tc--){
    int n;
    n = 120;
    int a[2];
    a[0] = 10,a[1] = 12;

    int ans[3][n+1];

    for (int i = 0;i <= n;i++)
    {
        ans[0][i] = 10000;
    }

    for (int i = 0;i <= 2;i++)
    {
        ans[i][0] = 0;
    }

    for (int i = 1;i <= 2;i++)
    {
        for (int j = 1;j <= n;j++)
        {
            if (a[i-1] > j)
                ans[i][j] = ans[i-1][j];
            else
                ans[i][j] = min(ans[i-1][j],1+ans[i][j-a[i-1]]);
        }
    }

	int num;
	cin >> num;
	
    if (ans[2][num] >= 10000)
        cout << -1 << endl;
    else
        cout << ans[2][num] << endl;
	}

    return 0;
}
