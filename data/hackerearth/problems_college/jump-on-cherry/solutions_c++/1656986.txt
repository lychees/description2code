#include <iostream>
using namespace std;

 int gcd(int a,int b)
 {
 	if(b==0)
 	return a;
 	else
 	return gcd(b,a%b);
 }
int main()
{
    int x,y,ans,t;
    cin>>t;
    while(t--)
    {
    	cin>>x>>y;
    	ans=(x*y)/gcd(x,y);
    	ans=ans/y+ans/x;
    	cout<<ans<<endl;
    	
    	
    }
    return 0;
}
