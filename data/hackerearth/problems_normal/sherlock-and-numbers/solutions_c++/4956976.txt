#include <iostream>
using namespace std;

int main()
{
    int t=0;
    cin>>t;
    unsigned int n=0,k=0,p=0;
    int c=0;
    int a=0;
    while (c<t)
    {
    	cin>>n>>k>>p;
    	if ((n-k)>=p){
    		for (int i=0;i<k;i++){
    			cin>>a;
    			if (a<=p){
    				p=p+1;
    			}
    		}
  		cout<<p<<endl;
    	}
    	else{    	    
    		for (int i=0;i<k;i++){
    			cin>>a;
    		}
    		cout<<-1<<endl;
    }
    c++;
    }
    return 0;
}
