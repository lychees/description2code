#include<iostream>
using namespace std;
int main(){
    int t,j;
    long long int ans,d,n,mid,i,l,h;
    cin>>t;
    while(t--){
        cin>>n;
        long long int a[n],b[n];
        for(i=0;i<n;i++) cin>>a[i];
        for(i=0;i<n;i++) cin>>b[i];
        i=0;
        ans=0;
        d=0;
        while(i<n){
            l=i;h=n-1;
            while(l<=h){
                mid=(l+h)/2;
                if(b[mid]>=a[i]){
                    if((mid-i)>(ans-d)){ans=mid;
                    d=i;
                    }
                    l=mid+1;
                }
                else h=mid-1;
            }
            i++;
        }
        cout<<ans-d<<endl;
    }
}
