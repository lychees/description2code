n=input()
a=map(int, raw_input().split())
if sum(a)%2!=0:
    print "NO"
else:
    i=0
    while i<n-1:
        a[i+1]=a[i+1]-a[i]
        a[i]=0
        if a[i+1]<0:
            print "NO"
            i=n
        i=i+1
    if i==n-1:
        if sum(a)==0:
            print "YES"
        else:
            print "NO"
