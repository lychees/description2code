#include <iostream>
#include <stack>
#include<cstring>

using namespace std;

int main()
{
	// get count
	int tcCount = 0;
	cin >> tcCount;
	char arr[1000000];
	int bublyCount = 0;
	string s;
	stack<char> bubblyStack;
	char c;
	
	for (int j = 0; j < tcCount; j++) {
		cin >> s;
		
		strcpy(arr, s.c_str());
		
		int length = s.length();
		
		bubblyStack.push(arr[0]);
		
		for (int i = 1; i < length; i++) {
			c = arr[i];
			
			if (bubblyStack.size() > 0 && c == bubblyStack.top()) {
				bubblyStack.pop();
				continue;
			}
			else {
				bubblyStack.push(c);
			}
		}
		
		if (bubblyStack.size() == 0) {
			bublyCount++;
		}
		else {
			while (bubblyStack.size() > 0) {
				bubblyStack.pop();
			}
		}
		
		
	}
	
	cout << bublyCount << endl; 
	
    return 0;
}
