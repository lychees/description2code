#include <iostream>
#include <algorithm>

using namespace std;

int f2(int arr[], int n, int c){
	sort(arr,arr+n,greater<int>());
	
	for(int i=0;i<c;i++){
		if(arr[i]>=1){
			arr[i]--;
		}
		else return 0;
	}
	
	return 1+f2(arr,n,c+1);
	
	
}


int func(){
	int n;
	cin>>n;
	int arr[n];
	for(int i=0;i<n;i++) cin>>arr[i];
	return f2(arr,n,1);
	
	
	
	
	
}



int main()
{
	int t;
	cin>>t;
	while(t--){
		cout<<func()<<endl;
	}


    return 0;
}
