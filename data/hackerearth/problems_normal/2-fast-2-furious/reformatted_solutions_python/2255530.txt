n = input()
a = map(int, raw_input().split())
b = map(int, raw_input().split())
ma = 0
mb = 0
for i in range(n-1):
    j = abs(a[i+1]-a[i])
    k = abs(b[i+1]-b[i])
    if (j > ma):
        ma = j
    if (k > mb):
        mb = k
if ma == mb:
    print "Tie"
    print ma
elif ma > mb:
    print "Dom"
    print ma
else:
    print "Brian"
    print mb
