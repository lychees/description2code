#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	char output[5000000]={},input[100];
    	int index=0;
        cin>>input;
        for(int i=0;input[i];i++)
        {
        	if(input[i]>48 && input[i]<58)
        	{
        		int x= input[i]-'0';
        		x--;
        		int temp=index;
        		while(x--)
        		{
        			for(int k=0;k<temp;k++)
        			{
        				output[index]=output[k];
        				index++;
        			}
        		}
        		
        	}
        	else
        	{
        		output[index]=input[i];
        		index++;
        	}
        }
        cout<<output<<endl;
    }
}
