#include <iostream>
using namespace std;

int main()
{
	long int T, N, X; 
	long int sum;
	cin>>T;
	for(int t=0;t<T;t++){
		cin>>N>>X;
		sum=0;
		long int *a = new long int[N];
		for(long int n=0;n<N;n++){
			cin>>a[n];
			sum	+= a[n];
		}
		if((sum/2)>X)	cout<<"NO"<<endl;
		else{
			 bool p[(sum/2)+1][N+1];
     		 for (long int i=0; i<=N; i++)	p[0][i] = true;
       		 for (long int i=1; i<=sum/2; i++)	p[i][0] = false;     
      		 for (long int i=1; i<=sum/2; i++){
       			for (long int j=1; j<=N; j++){
         			p[i][j]=p[i][j-1];
         			if (i>=a[j-1])	p[i][j]= (p[i][j] || p[i-a[j-1]][j-1]);
       			}        
     		}
     		long int i=sum/2;
     		while(i>=0){
     			if(p[i][N]==true)	break;
     			i--;
     		}
     		long int rem=sum-(i*2);
     		X=X-i;
     		if(X>=rem)	cout<<"YES"<<endl;
     		else	cout<<"NO"<<endl;
		}
	}
    return 0;
}
