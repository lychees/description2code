#include <iostream>
using namespace std;

unsigned long long arr[100001];
unsigned int mod=1000000007;
unsigned long long ans=0;

int main()
{
	int t,t1,n[10],max=0;
	int i=0,j=0;
	scanf("%d",&t);
	t1=t;
	while(t--)
	{
		scanf("%d",&n[i]);
		if(max<n[i])
			max=n[i];
		i++;
	}
	arr[0]=1;
	arr[1]=1;
	arr[2]=2;
	long long n_zero=1;
	long long n_one=1;
	for(i=3;i<=max;i++)
	{
		unsigned long long temp=n_one;
		n_one=(n_one+n_zero) % mod;
		n_zero=temp;
		arr[i]=(n_zero+n_one) % mod;
	}
	
	for(i=0;i<t1;i++)
	{
		ans=0;
		for(j=1;j<=n[i]-1;j++)
		{
			ans = (ans + ((arr[j-1] % mod) * (arr[n[i]-j-1] % mod)) % mod) % mod;	
		}
		printf("%lld\n",ans);
	}
    return 0;
}
