#include <iostream>
using namespace std;

int arr[25][65][65]={0};

int gcd(int a,int b)
{
	if(b==0)
		return a;
	return gcd(b,a%b);
}

int main()
{
	int t,i,j,k,sum=0;
	cin>>t;
	for(i=23;i>=0;i--)
	{
		for(j=59;j>=0;j--)
		{
			for(k=59;k>=0;k--)
			{
				int g=gcd(i,j);
				g=gcd(g,k);
				if(g==1)
					++sum;
				arr[i][j][k]=sum;
			}
		}
	}
	while(t--)
	{
    	int l,m,h,s;
    	cin>>h>>m>>s;
    	l=24*60*60-h*60*60-m*60-s;
    	int p=l-arr[h][m][s];
    	if(p==0)
    	{
    		cout<<"0"<<":"<<arr[h][m][s]<<"\n";
    	}
    	else
    	{
    		int g=gcd(p,arr[h][m][s]);
    		cout<<p/g<<":"<<arr[h][m][s]/g<<"\n";
    	}
	}
    return 0;
}
