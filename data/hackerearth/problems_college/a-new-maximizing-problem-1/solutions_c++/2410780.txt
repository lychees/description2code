#include <iostream>
#include <cmath>
#include <algorithm>
#define LL long long int
#define mod 1000000007
using namespace std;
LL ar[100001],temp[100001];
int main()
{
	LL t,n,i,A=0,B=0,j,k,ans;
	cin>>t;
	while(t--){
		cin>>n;
		for(i=1;i<=n;i++)cin>>ar[i];
		for(i=1,B=0;i<=n;i++)B=(B+abs(ar[i]))%mod;
		sort(ar+1,ar+1+n);
		i=1;j=n;k=1;
		while(i<j){
			temp[k++]=ar[i++];
			temp[k++]=ar[j--];
		}
		if(i==j)temp[k]=ar[i];
		for(i=1,A=0;i<=n;i++){
			if(i%2==1)A=(A-temp[i])%mod;
			else A=(A+temp[i])%mod;
		}
		ans=(LL)A*B%mod;
		cout<<ans<<endl;
	}
    return 0;
}
