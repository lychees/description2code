#include <iostream>
#include <cstring>
using namespace std;
int n,k;
double dp[51][10001];
double fun(int m,int sum)
{
	double ans=0;
	if(m==0)
	{
		if(sum==k)
			return 1;
		return 0;
	}
	if(dp[m][sum]>=0)
		return dp[m][sum];
	for(int i=1;i<=n;i++)
	{
		if(sum+i<=k)
			ans+=(1.0/n * fun(m-1,sum+i));	
		else
			break;
	}
	return dp[m][sum]=ans;
}
int main()
{
	int t,m,flag;
	scanf("%d",&t);
    while(t--)
    {
    	scanf("%d%d%d",&m,&n,&k);
    	for(int i=0;i<=m;i++)
    		for(int j=0;j<=k;j++)
    			dp[i][j]=-1;
    	double x=fun(m,0);
    	flag=0;
    	for(int i=1;i<=100;i++)
    	{
    		if(x>=1)
    		{
    			flag=1;
    			printf("%.3lf %d\n",x,i-1);
    			break;
    		}
    		x*=10;
    	}
    	if(!flag)
	    	printf("0.000 0\n");
    }
    return 0;
}
