#include <iostream>
#include <stdio.h>
using namespace std;

int main()
{
	int t,n,i,ans,a,b,c;
	scanf("%d",&t);
	while(t--){
		ans=0;
		c=0;
		int s[1000001]={0};
		scanf("%d",&n);
		for(i=0;i<n;i++){
			scanf("%d %d",&a,&b);
			if(b==0){
				if(ans<c) ans = c;
				c=0;
			}
			else{
				if(s[a]==0){ 
					c++;
					s[a]=1;
				}
			}
		}
		if(c>ans) ans = c;
		printf("%d\n",ans);
	}
    return 0;
}
