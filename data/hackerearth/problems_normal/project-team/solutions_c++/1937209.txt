#include <bits/stdc++.h>
using namespace std;

int main()
{
    int T; cin >> T;
    while(T--) {
    	int n; cin >> n;
    	vector<int> a(n);
    	for (int i=0; i< n; ++i) cin >> a[i];
    	sort(begin(a), end(a));
    	int mini = 1<<29, maxi=0;
    	for (int i=0;i<n/2;++i) {
    		int s=a[n-1-i] + a[i];
    		maxi = max(maxi, s);
    		mini = min(mini, s);
    	}
    	cout<<(maxi-mini)<<endl;
    }
}
