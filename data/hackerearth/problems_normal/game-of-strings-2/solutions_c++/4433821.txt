#include <iostream>
#include<vector>
#include<cstring>
using namespace std;
int main()
{
    int t;
    cin>>t;
    for(int l=0;l<t;l++)
    {
        string p;
        int m;
        int ma=0;
        cin>>p>>m;
        long n=p.length();
        int v[1000005];
        memset(v,0,sizeof(v));
        v[0]=0;
        for(int i=1;i<n;i++)
        {
            int j=v[i-1];
            while(j>0&&p[j]!=p[i])
            {
                j=v[j-1];
            }
            if(p[j]==p[i])
            {
                j++;
            }
            v[i]=j;
            if(i<m)
                ma=max(ma,v[i]);
        }
        int k=v[n-1];
        while(k>ma&&k)
            k=v[k-1];
        if(k==0)
            cout<<"Puchi is a cheat!";
        else
        {
            for(int i=0;i<k;i++)
                cout<<p[i];
        }
        cout<<endl;
    }
}
