#include <iostream>
#include <cmath>
#define ll long long int
using namespace std;
ll primes[1000005];
ll sum[1000005];

void sieve(ll n)
{
	ll i,j,k;
	for(i=0;i<n;i++)
	primes[i]=1;
	
	primes[0]=0;
	primes[1]=0;
	
	for(i=2;i<=sqrt(n);i++)
	{
		if(primes[i])
		for(j=i*i;j<=n;j+=i)
		primes[j]=0;
	}
	
	sum[0]=0;
	sum[1]=0;
	sum[2]=2;
	k=3;
	for(i=3;i<n;i++)
	{
		if(primes[i])
		sum[i]=sum[i-1]+i;
		else
		sum[i]=sum[i-1];
	}
}

int main()
{
	sieve(1000005);
	ll n;
	cin>>n;
	while(n--)
	{
		ll l,r;
		cin>>l>>r;
		cout<<sum[r]-sum[l-1]<<endl;
	}
    return 0;
}
