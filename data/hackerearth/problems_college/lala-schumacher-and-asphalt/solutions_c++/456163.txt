#include <iostream>
#include<cstdio>
using namespace std;

#define getcx getchar_unlocked
 
void inp( long long int* n )//fast input function
{
	*n=0;
        int ch=getcx();
        int sign=1;
        while( ch < '0' || ch > '9' )
        {
        	if(ch=='-')
        		sign=-1; 
        	ch=getcx();
        }
        while(  ch >= '0' && ch <= '9' )
        {
        	*n = ((*n)<<3)+((*n)<<1) + ch-'0', ch=getcx();
        }
        *n=(*n)*sign;
}


#define A(i) ((i>=0)?(i):(-i))
int main()
{
    long long int t;
    inp(&t);
    while(t--){
    	long long int i,n,X,Y;
    	inp(&n);
    	long long int z,x[n+1],y[n+1];
    	for( i=0;i<n;i++){
    		
    		inp(&z);
    		x[i]=z;
    	}
    	for( i=0;i<n;i++){
    		
    		inp(&z);
    		y[i]=z;
    	}
    	inp(&X);
    	inp(&Y);
    	long long int p,q;
    	inp(&p);
    	inp(&q);
    	long long int temp,s=0;
    	s=(A(X)+A(Y))*p;
    	for( i=0;i<n;i++){
    		long long int xv=(x[i]-X),yv=(y[i]-Y);
    		temp=(A(xv)+A(yv))*q+(A(x[i])+A(y[i]))*p;
    		if(temp<s)
    			s=temp;
    	}
    	cout<<s<<endl;
    }
    return 0;
}
