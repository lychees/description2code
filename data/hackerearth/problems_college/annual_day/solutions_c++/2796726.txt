#include <iostream>
#include <stdlib.h>
using namespace std;


int countSetBits(int n)
{
    unsigned int count = 0;
    while (n)
    {
      n &= (n-1) ;
      count++;
    }
    return count;
}

int cmpfunc(const void *a, const void *b)
{
	int *_a = (int *)a;
	int *_b = (int *)b;
	return countSetBits(*_a)-countSetBits(*_b);
}

int main()
{
    //cout << "Hello World!" << endl;
    int t,n,k,arr[200],sum;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	cin>>n>>k;
    	sum=0;
    	for(int j=0;j<n;j++)
    	cin>>arr[j];
    	qsort(arr,n,sizeof(int),cmpfunc);
    	//for(int j=0;j<n;j++)
    	//cout<<arr[j]<<" ";
    	//cout<<endl;
    	for(int j=n-1;j>=n-k;j--)
    	sum+=countSetBits(arr[j]);
    	cout<<sum<<endl;
    }
    
    return 0;
}
