using namespace std;

int n, m, k, ta, tb, tc;
char x[MAX];

int main() {
  scanf("%d%d%d%s", &n, &m, &k, x);
  m += k;
  while (m--) {
    scanf("%d", &ta);
    if (ta == 1) {
      scanf("%d%d%d", &ta, &tb, &tc);
      ta--;
      memset(x + ta, tc + '0', tb - ta);
    } else {
      scanf("%d%d%d", &ta, &tb, &tc);
      ta--;
      printf(memcmp(x + ta, x + ta + tc, tb - ta - tc) ? "NO\n" : "YES\n");
    }
  }
  // system("pause");
  return 0;
}
