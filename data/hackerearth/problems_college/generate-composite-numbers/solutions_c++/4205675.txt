// Example program
#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    int n,i,j;
    cin>>n;
    for(i=4;i<=n;i++)
    {
        for(j=2;j<(pow(i,0.5))+1;j++)
        {
            if(i%j==0)
            {
                cout<<i<<" ";
                break;
            }
        }
    }
    return 0;
}
