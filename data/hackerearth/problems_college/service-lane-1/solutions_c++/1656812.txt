#include <iostream>
using namespace std;

int arr[100000];
int main()
{
    int n,t;
    
   	cin>>n>>t;
   	
   	for(int i=0;i<n;i++)
   	cin>>arr[i];
   	
   	for(int i=1;i<=t;i++)
   	{
   		int l,m;
   		cin>>l>>m;
   		
   		int ans=3;
   		
   		for(int i=l;i<=m&&ans>=1;i++)
   		{
   			if(arr[i]<ans)
   			{
   				ans--;
   				i=l-1;
   			}
   		
   		}
   		cout<<ans<<endl;
   	}
    return 0;
}
