#include <iostream>
using namespace std;
int joe(long long int n)
{
	if(n<=1)
		return n;
	if(n%2 == 0)
		return 2*joe(n/2)-1;
	return 2*joe(n/2)+1;
}
 
int main()
{
    long long int t , n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	cout<<joe(n)<<endl;
    }
    return 0;
}
