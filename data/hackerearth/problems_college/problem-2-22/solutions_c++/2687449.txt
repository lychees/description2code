int main()
{
    int tc,r,x[5],y[5],x1,y1;

    while(tc--)
    {
    	cin>>r>>x1>>y1;
    	for(int i=0;i<4;i++)
    	{
    		cin>>x[i]>>y[i];
    	}
    	if(((x1-x[0])*(x1-x[0])+(y1-y[0])*(y1-y[0])<r*r)&&((x1-x[1])*(x1-x[1])+(y1-y[1])*(y1-y[1])<r*r)&&((x1-x[2])*(x1-x[2])+(y1-y[2])*(y1-y[2])<r*r)&&((x1-x[3])*(x1-x[3])+(y1-y[3])*(y1-y[3])<r*r))
