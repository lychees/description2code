#include <iostream>
using namespace std;

int main()
{
    int t, r, c, w;
    cin >> t;
    int cnt = 0;
    while(t--)
    {
    	cnt++;
    	cin >> r >> c >> w;
    	if(w==c)
    	{
    		cout << "Case #"<< cnt << ": " << r-1+w << endl;
    	}
    	else
    	{
    		cout << "Case #"<< cnt << ": " << (c/w)*r + w-1 << endl;
    	}
    }
    return 0;
}
