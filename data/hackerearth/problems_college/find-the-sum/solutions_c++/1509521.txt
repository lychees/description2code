#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <vector>
#include <stdlib.h>
#include <math.h>
#include <string.h>
using namespace std;
int t;
long long l,r;
long long getsum(long long x){
    if (x==0) return(0);
    long long l=0,r=100000000;
    while (l<r){
        long long mid=(l+r)/2+1;
        if ((mid*(mid+1))/2<=x) l=mid;
        else r=mid-1;
    }
    long long sum=0;
    sum=sum+((l+1)/2)*((l+1)/2);
    x-=((l*(l+1))/2);
    if (x>0) sum=sum+((x%2)==1)*(l+1);
    return(sum);
}
int main(){
    //freopen("C.inp","r",stdin);
    scanf("%d",&t);
    while (t){
        scanf("%lld %lld",&l,&r);
        printf("%lld\n",getsum(r)-getsum(l-1));
        t--;
    }
}
