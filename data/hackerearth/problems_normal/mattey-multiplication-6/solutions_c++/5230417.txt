#include <iostream>
#include<vector>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
      {
      	 int count=0;
      	vector<int> vec;
      	 long long int n,m;
      	 cin>>n>>m;
      	 int temp;
      	 int size;
      //	 cout<<"size="<<size;
      	 while(m!=0)
      	    {
      	    	 temp=m&1;
      	    	// cout<<temp<<endl;
      	    	   vec.push_back(temp);
      	    	 m=m>>1;
      	    }
      	    size=vec.size();
      	   // cout<<"size="<<size;
      	    count=0;
      	  for(int j=0;j<size;j++)
      	      {
      	      	if(vec[j]==1)
      	      	   {
      	      	   	count=j;
      	      	   	break;
      	      	   }
      	      }
      	      int k;
      	    //  cout<<" count="<<count;
      	   for(k=size-1;k>count;k--)
      	       {
      	       	 if(vec[k]==1)
      	       	    cout<<"("<<n<<"<<"<<k<<") + ";

      	       }
      	    cout<<"("<<n<<"<<"<<k<<")";
      	    cout<<endl;
      	 
      	   
      }
    return 0;
}
