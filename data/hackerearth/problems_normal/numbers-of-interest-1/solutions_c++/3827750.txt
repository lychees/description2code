#include<iostream>
using namespace std;
int main()
{
    long int t,i=0,sum=0,**num;
    int *n,*x,*y,j,k;
    cin>>t;
    num=new long int*[t];
    n=new int[t];
    x=new int[t];
    y=new int[t];
    for(i=0;i<t;i++)
    {
        cin>>x[i]>>y[i]>>n[i];
        num[i]= new long int[x[i]];
        for(j=0;j<x[i];j++)
        {
            num[i][j]=y[i];
        }
    }
    for(i=0;i<t;i++)
    {
        for(j=0;j<n[i]-x[i];j++)
        {
            sum=0;
            for(k=0;k<x[i];k++)
            {
                sum+=num[i][k];
                if(k!=x[i]-1)
                    num[i][k]=num[i][k+1];
            }
            num[i][x[i]-1]=sum;
        }
    }
    for(i=0;i<t;i++)
    {
        cout<<num[i][x[i]-1]<<endl;
    }
    for(i=0;i<t;i++)
    {
        delete num[i];
    }
    delete num;
    delete n;
    delete x;
    delete y;
    return 0;
}
