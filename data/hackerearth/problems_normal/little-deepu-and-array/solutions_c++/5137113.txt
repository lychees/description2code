#include <bits/stdc++.h>
using namespace std;

void update(vector <int> &v, int x)
{
    for(int i = 0; i < (int)v.size(); i++) {
        if(v[i] > x)
            v[i] -= 1;
    }
}
void __answer()
{
    register int i;
    int n;
    cin >> n;
    vector <int> v(n, 0);
    for(i = 0; i < n; i++) {
        cin >> v[i];
    }
    int m;
    cin >> m;
    while(m--) {
        int x;
        cin >> x;
        update(v, x);
    }
    for(int i = 0; i < (int)v.size(); i++) {
            cout << v[i] << " ";
    }
    cout << '\n';
}

int main()
{
	ios::sync_with_stdio(false); cin.tie(NULL);
    #ifdef LOCAL
      freopen("sample_input.txt","r",stdin);
    #endif
	__answer();
    return 0;
}
