#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define MAX 200000
ll n;
ll a[1000006];


int main() {
    ll t;
    cin>>t;
    while(t--) {
        vector<ll> d;
        cin>>n;
        for(int i=0;i<n;i++) {
            cin>>a[i];
        }
        ll num;
        cin>>num;
        ll ans = INT_MAX;
        bool flag = 0;
        for (int i = 0; i < n; i++) {
            ll x = a[i];
            vector<ll>::iterator it = lower_bound(d.begin(), d.end(), x);
            if (it == d.end()) d.push_back(x);
            else *it = x;
            ll sz = d.size();
            if(sz >= num) {
                ans = min(ans, d[num-1]);
                flag = 1;
            }
        }
        if(flag)
            cout<<ans<<"\n";
        else
            cout<<"-1\n";
    }
    return 0;
}
