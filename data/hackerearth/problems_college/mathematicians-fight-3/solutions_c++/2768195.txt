#include <iostream>
using namespace std;

int main()
{
    int t,a[1000],r[1000];
    cin>>t;
    for(int i=0;i<t;i++)
    {
        cin>>a[i];
        if(a[i]==1)
        r[i]=1;
        else
        {
            int m=1;
           while( m*2 <=a[i])
           {
               m*=2;
           }
           m=a[i]-m;
           r[i]=2*m+1;
        }
        
    }
    for(int i=0;i<t;i++)
    {
     cout<<r[i]<<endl;   
    }
    return 0;
}
