test_cases = int(raw_input())
vowels = set('aeiou')

# Keeping this as 4 for '.com'

#Adjusting the range of the for loop to skip www. and .com

while test_cases >= 1:
    count = 4
    website = list(map(str, raw_input()))

    for i in range(4, len(website)-4):
        if website[i] in vowels:
            continue
        else:
            count += 1
    print '{0}/{1}'.format(count, len(website))

    test_cases -= 1
