#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int freq[1<<20];
int a[1<<20];
int ans[1<<20];
int main()
{
	int tc;
	cin>>tc;
	while(tc--)
	{
		int n;
		cin>>n;
		memset(a,0,sizeof a);
		for(int i=1;i<=n;++i)
		{
			cin>>a[i];
			freq[a[i]]=1;
		}
		sort(a+1,a+n+1);
		int i=1;
		int j=n;
		bool flag=true;//associate with the largest presently and vice versa
		int size=1;
		for(;i>=1 && i<=n && j>=1 && j<=n && i<=j;)
		{
			if(flag==true && freq[a[j]]==1)
			{
				freq[a[j]]=0;
				ans[size]=a[j];
				++size;
				flag=false;
				--j;
			}
			else if(flag==false && freq[a[i]]==1)
			{
				freq[a[i]]=0;
				ans[size]=a[i];
				++size;
				flag=true;
				++i;
			}
			else if(flag==true)
			{
				--j;
			}
			else if(flag==false)
			{
				++i;
			}
		}
		--size;
		for(int i=1;i<=size;++i)
		{
			printf("%d ",ans[i]);
		}
		printf("\n");
		
		
	}
	
  //  cout << "Hello World!" << endl;
    return 0;
}
