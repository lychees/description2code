#include <iostream>

int main() {
    int N;
    scanf("%d", &N);
    int X;
    scanf("%d", &X);
    int count = 0;
    bool skip = false;
    for(int testCase = 0; testCase < N; testCase++)
    {
        int A;
        scanf("%d", &A);
        if(A>X) {
            if(skip)
                break;
            
            skip = true;
        }
        else
            count++;
    }
    printf("%d", count);
    return 0;
}
