#include<iostream>
#include<string>
using namespace std;

int main()
{
    long int T;
    cin>>T;
    while(T--)
    {
        long int L;
        char x;
        int pos;
        int i,j,k;
        char min;
        char temp;
        int flag;
        string s;
        cin>>s;
        L = s.length();
        for(i=L-1;i>0;i--)
        {
            if(s[i]>s[i-1])
                break;
        }

        if(i==0)
            cout<<"no answer"<<endl;
        else
        {
            //printf("\ni=%d\n",i);
            x=s[i-1];
            //printf("\nx=%c\n",x);
            pos=i-1;
            min=i;
            for(j=i+1;j<L;j++)
            {
                if(s[j]<s[min] && s[j]>x)
                    min=j;
            }
            //printf("\nmin=%d\n",min);
            temp=s[min];
            s[min]=s[pos];
            s[pos]=temp;
            //printf("%s\n",s);

            for(j=pos+1;j<L;j++)
            {
                for(k=pos+1;k<L;k++)
                {
                    if(s[k]>s[j])
                    {
                        temp=s[k];
                        s[k]=s[j];
                        s[j]=temp;
                    }
                }
            }
            cout<<s<<endl;
        }
    }
    return 0;
}
