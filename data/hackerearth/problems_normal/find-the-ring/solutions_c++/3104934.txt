#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	for(int i=0;i<t;i++)
	{
		int n,index;
		cin>>index>>n;
		int result=0;
		for(int j=n;j>=1;j--)
		{
			if(index==0)
			{
				result=1;
			}
			if(index==1)
			{
				result=0;
			}
			if(index==2)
			{
				result=1;
			}
		  index=result;
		}
		cout<<index<<endl;
		
	}
    return 0;
}
