#include <iostream>
using namespace std;

int main()
{
    int t,i,n,j,k,posx,posy,midx,midy;
    int x,y,ans;
    char a[50][50];
    cin>>t;
    for(k=0;k<t;k++)
    {
    	cin>>n;
    	for(i=0;i<n;i++)
    	   for(j=0;j<n;j++)
    	       cin>>a[i][j];
    	for(i=0;i<n;i++)
    	   for(j=0;j<n;j++)       
    	       if(a[i][j]=='*')
    	       {
    	       	posx=i;
    	       	posy=j;break;
    	       }
        midx=n/2;
        midy=midx;
        x=midx-posx;
        y=midy-posy;
        if(x<0)
            x=(-1)*(x);
        if(y<0)
            y=(-1)*(y);
        ans=x+y;
        cout<<ans<<endl;    
    }
    return 0;
}
