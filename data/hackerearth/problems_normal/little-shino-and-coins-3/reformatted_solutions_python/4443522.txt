K = int(raw_input())
inpstr = raw_input()
inplen = len(inpstr)

numpairs = 0
for i in xrange(inplen):
    numch = 0
    lastseen = {x: -1 for x in "abcdefghijklmnoprqstuvwxyz"}
    j = i
    while j < inplen and numch < K:
        if lastseen[inpstr[j]] == -1:
            numch += 1
        lastseen[inpstr[j]] = j
        j += 1
    if numch == K:
        m = j
        while j < inplen and lastseen[inpstr[j]] >= 0:
            j += 1
        numpairs += j-m+1
print numpairs
