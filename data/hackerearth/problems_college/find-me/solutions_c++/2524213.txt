#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n,p,i;
    	long *a;
    	cin>>n>>p;
    	a=(long *)malloc(sizeof(long)*n);
    	for(i=0;i<n;i++)
    		cin>>a[i];
    		
    	int flag=0,count=0;	
    	for(i=0;i<n;i++){
    		if(p==a[i]){
    			flag=1;
    		}
    		
    		if(p<a[i]){
    			count++;
    		}
    	}
    	
    	if(flag==0){
    		cout<<"NO 0"<<endl;
    	}else{
    		cout<<"YES "<<count<<endl;
    	}
    }
    return 0;
}
