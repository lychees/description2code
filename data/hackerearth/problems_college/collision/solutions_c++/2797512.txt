#include<iostream>
using namespace std;
int main()
{
	int t,n,num,a[14];
	scanf("%d",&t);
	while(t--)
	{
		for(int i=0;i<14;i++)
		{
			a[i]=0;
		}
		scanf("%d",&n);
		for(int i=1;i<=n;i++)
		{
			scanf("%d",&num);
			a[num%10]=a[num%10]+1;
		}
		int sum=0;
		for(int i=0;i<10;i++)
		{
			if(a[i]>1)
			sum=sum+a[i]-1;
		}
		printf("%d\n",sum);
	}
	return 0;
}
