const int MAXN = 1e3 + 3;
string config[MAXN];
void solve() {
	int k;
	string s;
	scanf("%d", &k); cin.ignore();
	getline(cin , s, '\n');
	int n = s.length();
	int cnt = 0;
	string back = s;
	int t = n / 2;
	config[0] = s;
	while (1) {
		string str = "";
		forn(i, t) {
			str += config[cnt][i];
			str += config[cnt][n - i - 1];
		}
		if (n % 2) str += config[cnt][t];
		if (str == config[0]) break;
		config[++cnt] = str;
	}
	cnt++;
	cout << config[cnt - (k % cnt)];
}
