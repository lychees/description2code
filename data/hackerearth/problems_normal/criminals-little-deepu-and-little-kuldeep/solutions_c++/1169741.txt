#include <iostream>
#include<algorithm>
using namespace std;
long a[100000];

int main()
{
  int t,i,j,n,b,c;
  cin>>t;
  while(t--)
  {
  	for(i=0;i<100000;i++)
  	  a[i]=0;
  	  
  cin>>n;
  for(i=0;i<n;i++)
  cin>>a[i];
  sort(a,a+n);
  b=1;
  c=1;
  for(i=0;i<n-1;i++)
   {j=i;
   	while(a[j]==a[j+1])
   	    {b++;
   	    j=j+1;
   	    i=j-1;}
   	 if(b>c)
   	   {c=b;
   	   }
   	
   	   b=1;
   	    
   }
     cout<<c<<endl; 
}
}
