'''

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
t = int(raw_input())
while t > 0:
    n = int(raw_input())
    a = [int(x) for x in raw_input().split()]
    j = 0
    k = 0
    while True:

        k = a[k]
        if k == -1:
            print("Win")
            break
        j += 1
        if j == n*n:
            print("Lose")
            break

    t = t-1
