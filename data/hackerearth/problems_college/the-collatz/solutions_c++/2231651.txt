#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
int t;
cin>>t;
while(t--)
{
long long int n;
  cin >> n;
  vector<long long int> even;
  long long int steps=0;
  vector<long long int> odd;
  while (n > 1) {
  	steps++;
    if (n % 2 == 0)
      {
      	even.push_back(n);
      	n = n / 2;
      }
    else 
      {odd.push_back(n);
      	n = 3 * n + 1;
      }
  }
  odd.push_back(1);
 unsigned long long int oddsum=0;
 unsigned long long int evensum=0;

for(long long int i=0;i<odd.size();i++)
  {
  oddsum=oddsum+odd[i];	
  }
  
for(long long int i=0;i<even.size();i++)
  {
  evensum=evensum+even[i];	
  }
  if(evensum%steps>oddsum%steps)
  cout<<"Even Rules"<<endl;
  else if(evensum%steps<oddsum%steps)
cout<<"Odd Rules"<<endl;
else
cout<<"Tie"<<endl;
}
  return 0;
}
