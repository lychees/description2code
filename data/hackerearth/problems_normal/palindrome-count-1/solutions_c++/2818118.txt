#include <bits/stdc++.h>
#include<algorithm>
#include<cstring>
using namespace std;
 
bool isPalindrome(string s)
 {
for(int i=0;i<s.size()/2;i++)
{
	if(s[i]!=s[s.size()-1-i])
	return false;
}
return true;
}
int main()
{  
	string s;
    cin>>s;
    int l=s.size(),count=0;
    for(int i=0;i<l;++i){
        for(int j=l-i;j>0;--j){
            if(isPalindrome(s.substr(i,j))){
                count++;
            }
        }
    }
    cout<<count<<endl;
    return 0;
}
