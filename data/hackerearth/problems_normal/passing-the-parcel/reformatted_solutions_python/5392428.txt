import itertools
N = input()
S = raw_input()
players = range(N)
pos = 0
for c in itertools.cycle(S):
    #     print pos
    #     print players
    #     print c
    #     print '---------'
    if len(players) == 1:
        break
    if c == 'b':
        players.remove(players[pos])
    else:
        pos += 1
    if pos >= len(players):
        pos = 0
print players[0] + 1
