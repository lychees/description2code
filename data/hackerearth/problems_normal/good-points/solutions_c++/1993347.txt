#include <iostream>
using namespace std;

int main()
{
    int numberOfPoints = 0;
    
    cin >> numberOfPoints;
    
    if(numberOfPoints == 1)
    	cout << "YES" << endl << "0 0" << endl;
    else if(numberOfPoints == 2)
    	cout << "YES" << endl << "0 0\n1 0" << endl;
    else if(numberOfPoints == 4)
    	cout << "YES" << endl << "0 0\n0 1\n1 1\n1 0" << endl;
    else
    	cout << "NO"<< endl;
    return 0;
}
