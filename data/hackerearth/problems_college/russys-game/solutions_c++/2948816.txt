#include <iostream>
using namespace std;

int main()
{
	int tc, a[1000], n;
	int i,k, flag=0, j;
	cin>>tc;

	for(k=0;k<tc;k++){
		flag = 0;
		cin>>n;
		for(i=0;i<n;i++)
			cin>>a[i];
		
		i = j = 0;
		while(j<n){
			if(a[i] == -1){
				flag++;
				break;
			}	
			if(a[i]>0 && a[i]<n){
				i = a[i];	
			}
			j++;	
		}
		
		if(flag)
			cout<<"Win\n";
		else
			cout<<"Lose\n";
	}
   
    return 0;
}
