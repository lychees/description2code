#include <queue>
#include <vector>
#include <stdio.h>
#include <algorithm>
#include <iostream>
using namespace std;

int main()
{
    long long int n,t,i,j,k,source,temp,x,y,r,max,s;
    vector <long long int> adj[20001];
    long long int visited[20001];
    queue <long long int> q;
    scanf("%lld",&t);
    for(k=1;k<=t;k++)
    {
        scanf("%lld",&n);
        max=0;
        std::fill(visited,visited+20001,0);
        for(i=0;i<n;i++)
        {
            scanf("%lld%lld",&x,&y);
            adj[x].push_back(y);
            adj[y].push_back(x);
            visited[x]=-1;
            visited[y]=-1;
        }
        for(i=1;i<20001;i++)
        {
            r=0;
            s=0;
            if(visited[i]==-1)
            {
                 q.push(i);
                 visited[i]=0;
                 r++;
                 while(!q.empty())
                 {
                     temp=q.front();
                     q.pop();
                     for(j=0;j<adj[temp].size();j++)
                     {
                         if(visited[adj[temp][j]]==-1)
                         {
                             visited[adj[temp][j]]=1-visited[temp];
                             q.push(adj[temp][j]);
                             if(visited[adj[temp][j]]==0)
                                r++;
                             if(visited[adj[temp][j]]==1)
                                s++;
                         }
                     }
                 }
                 if(r>=s)
                    max=max+r;
                 else
                    max=max+s;
            }
        }
        cout<<"Case "<<k<<": ";
        cout<<max;
        cout<<"\n";
        for(i=0;i<20001;i++)
            adj[i].clear();
    }
    return 0;
}
