#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main() {
    
    int n,d,k;
    cin >> n >> k >> d;
    int a[n];
    int tmp;
    a[0] = 0;
    for(int i = 1; i <= n; i++){
        cin >> tmp;
        a[i] = a[i-1] + tmp;
    }
    
    int pos = k;
    int ans = 0;
    while(pos <= n){
        //cout << "pos: " << pos << " " << a[pos] <<" pos-k: " << pos-k <<" " <<a[pos-k] << endl; 
        if((a[pos] - a[pos-k]) < d){
            
            if((a[pos] - a[pos-k]) == 0){
                cout << "-1" << endl;
                return 0;
            }
            
            if(a[pos] - a[pos-1] == 0){
            	//cout << a[pos]-a[pos-1] << endl;
            	for(int i = pos; i > (pos-k); i--){
            		if(a[i]-a[i-1] > 0){
            			ans++;
            			pos = i + k + k -1;
            			break;
            		}
            	}
            }
            else{
            	ans++;
            	pos = pos + k+k-1;
            }
            
            
            
        }
        else{
            pos= pos +1;
        }
        
    }
    
    cout << ans <<endl;
    
    
    return 0;
}
