#include<iostream>
#include<string>
#include<unordered_map>
using namespace std;
main()
{
    int T,*num,i,j,sum,count=0,max=0;
    string *S;
    unordered_map<string, int> M;
    unordered_map<string, int>::iterator it;
    cin>>T;
    num=new int[T];
    num[0] = 0;
    S = new string[T];
    for(i = 0; i < T; i++)
        cin>>S[i];
    for(i = 0; i < T; i++)
    {
        if(M.find(S[i])==M.end())
            M[S[i]] = 1;
        else
            M.find(S[i])->second++;
    }
    for(it=M.begin(); it!=M.end(); it++)
    {
       num[count] = it->second;
       count++;
    }
    for(j=0; j<=count; j++)
    {
        if(num[j] > max)
        {
            sum = num[j];
            max = num[j];
        }

        else if(num[j] == max)
            sum += num[j];
    }
    cout<<sum<<endl;



}
