import math


def f(n, m):
    x = n*n
    y = n*(n+1)
    z = (n+1)*(n+1)
    if m == x:
        return 2*n-1
    if m > x and m <= y:
        return 2*n
    elif m > y and m < z:
        return 2*n+1
    else:
        return -1


t = input()
for i in range(t):
    m = input()
    d = int(math.sqrt(m/2))
    j = d
    while f(j, m) == -1:
        j = j+1
    print f(j, m)
