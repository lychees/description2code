#define max(a,b)  ((a)>(b)?(a):(b))

#define ll long long
using namespace std;
typedef vector<int> V;
typedef long long i64;
const long long LINF = (long long)1e16 + 5;
int main()
{
	int T;
	cin >> T;
	while (T--)
	{
		int n, a[101], dp[2][100000 + 5],i,j,x;
		cin >> n >> x;
		ll sum = 0;
		for (i = 1; i <= n; i++)
		{
			cin >> a[i];
			sum += a[i];
		}
		if (sum <= x)
		{
			cout << "YES\n";
		}
		else
		{
			for (i = 0; i <= x; i++)
			{
				dp[0][i] = 0;
			}
			for (i = 1; i <= n; i++)
			{
				for (j = 0; j <= x; j++)
				{
					dp[i % 2][j] = dp[(i - 1) % 2][j];
					if (j >= a[i])
					{
						dp[i % 2][j] = max(dp[i % 2][j], a[i] + dp[(i - 1) % 2][j - a[i]]);
					}
				}
			}
			if (sum - dp[n % 2][x] <= x)
			{
				cout << "YES\n";
			}
			else
			{
				cout << "NO\n";
			}
		}
	}
}
