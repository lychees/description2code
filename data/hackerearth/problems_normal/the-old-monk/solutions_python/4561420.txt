def binsearch(B,Ai,i,N):
    mid=0
    pos=0
    m=N-1
    while i<=m:
        mid=(i+m)/2
        if B[mid]<Ai:
            m=mid-1
        elif B[mid]>=Ai:
            pos=mid
            i=mid+1
    return pos
