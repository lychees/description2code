#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define llu unsigned long long
#define pp pair<int,int>
#define mp make_pair
#define pb push_back
#define mod 1000000007
#define vi vector<int>
#define vii vector<vi >
#define vl vector<ll>
#define vll vector<vl >
#define ppl pair<ll,ll>
#define ft first
#define sd second
int main() {
	int i,t,n,m,maxi,j;
	vii a(1000,vi(1000)),b(1000,vi(1000));
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d",&n,&m);
		for(i=0;i<n;i++)
		{
			for(j=0;j<m;j++)
			{
				scanf("%d",&(a[i][j]));
			}
		}
		vector<int> mrow(n),col(m);
		for(i=0;i<n;i++)
		{
			maxi=0;
			for(j=0;j<m;j++)
			{
				scanf("%d",&(b[i][j]));
				maxi=max(maxi,b[i][j]);
			}
			mrow[i]=maxi;
		}
		int ans=0;
		for(j=0;j<m;j++)
		{
			maxi=0;
			for(i=0;i<n;i++)
			{
				maxi=max(maxi,a[i][j]);
			}
			col[j]=maxi;
		}
		for(i=0;i<n;i++)
		{
			for(j=0;j<m;j++)
			{
				if((a[i][j]==col[j])&&(mrow[i]==b[i][j]))
				ans++;
			}
		}
		printf("%d\n",ans);
	}
	return 0;
}
