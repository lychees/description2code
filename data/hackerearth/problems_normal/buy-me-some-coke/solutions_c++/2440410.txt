template <class T>  inline void smax(T &x,T y){ x = max((x), (y));}
template <class T>  inline void smin(T &x,T y){ x = min((x), (y));}
const int MOD = 1e9+7;

#define LOCALTEST 0 // change this to 1 to perform localtest on "in.txt"
int c, n1, n5, n10;
int f[151][101][51];
int go(int a, int b, int c) {
    int &ret = f[a][b][c];
    if (ret != -1) return ret;
    if (a == 0) return ret = 0;
    if (b == 0 && c == 0) return ret = a * 8;

    ret = 0x3f3f3f3f;
    if (c > 0) smin(ret, 4 + go(a-1, b+1, c-1));
    if (c > 0) smin(ret, 1 + go(a-1, b, c-1));
    if (b > 1) smin(ret, 2 + go(a-1, b-2, c));
    if (b > 0) smin(ret, 4 + go(a-1, b-1, c));
    //printf("(%d,%d,%d) = %d\n",a,b,c,ret);
    return ret;
}

    if (LOCALTEST) {
        freopen("in.txt", "r", stdin);

    }


    CASET {
        RII(c,n1);
        RII(n5,n10);
        if (n1 + n5 * 5 + n10 * 10 < c * 8) {
            printf("-1\n");
            continue;
        }
        MS1(f);

        printf("%d\n", go(c, n5, n10));
    }
    return 0;
}
