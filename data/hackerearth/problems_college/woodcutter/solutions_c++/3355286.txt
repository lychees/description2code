#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
	int i,j,n;
	long d,c=0,m,s=0;
	scanf("%d%ld",&n,&m);
	long int a[n];
	for(i=0;i<n;i++) {
	scanf("%ld",&a[i]);
	s=s+a[i];}
	if(s<m){
		printf("-1");
		return 0;
	}
	else if(s==m) {
		printf("0");
		return 0;
	}
	sort(a,a+n);
	for(i=n-1,j=1;i>0;i--,j++) {
		c=c+j*(a[i]-a[i-1]);
		if(c>=m)
		break;
	}
	if(i==0)
	{
	c=c+n*a[0];
	d=(c-m)/j;
	printf("%d",d);
	return 0;}
	d=(c-m)/j;
	printf("%d",a[i-1]+d);
	return 0;
}
