#include<iostream>
#include<cmath>
#include<cstdio>
#include<algorithm>

using namespace std;

main()
{
	long long t,i;
	cin>>t; 
	long long arr[t],count=0,j;
	i=t;
	while(t)
	{
		cin>>arr[t-1];
		t--;
	}
	
	sort(arr , arr + i);
	
	
	for(j=0;j<i-1;j++)
	{
		if(arr[j]==arr[i-1])
		{
			count++;
			
		}
		
		
		while(arr[j]<arr[i-1])
		{
			arr[j]=arr[j]*2;
			if(arr[j]==arr[i-1])
			{
				count++;
				break;
			}
		}
	}
	
	if(count==i-1)
		cout<<"YES"<<endl;
	else
		cout<<"NO"<<endl;
	
	count=0;
	
	return 0;
}
