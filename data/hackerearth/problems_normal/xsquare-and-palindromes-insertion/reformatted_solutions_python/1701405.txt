def getCount(string):
    cStr = {}
    # oddChar = {}
    for c in string:
        cCount = string.count(c)
        if cCount % 2 != 0:
            cStr[c] = cCount
    sLen = len(string)
    if sLen % 2 == 0 and cStr == {}:
        print 0
    elif sLen % 2 == 0:
        print len(cStr) - 1
    elif sLen % 2 != 0 and cStr == {}:
        print 1
    else:
        print len(cStr) - 1


tcNum = int(raw_input())
for i in range(tcNum):
    tcStr = raw_input()
    getCount(tcStr)
