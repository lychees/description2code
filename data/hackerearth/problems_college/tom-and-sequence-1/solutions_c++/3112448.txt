#include <iostream>
#include <vector>
using namespace std;
#define max 10000001
#define mod 1000000007
long series[max];
bool isprime[max];
vector<int> primes;

int main()
{
    int n,num,i,j;
    for(i=2;i<max;i++)
    isprime[i]=true;
    
    for(i=2;i<max;i++)
    {
    	if(isprime[i]==true)
    	{
    		primes.push_back(i);
    		for(j=2*i;j<max;j+=i)
    		{
    			isprime[j]=false;
    		}
    	}
    }

    series[0]=2;
    for(i=1;i<primes.size();i++)
    series[i]=((series[i-1]%mod)*(primes[i]%mod))%mod;
    cin>>n;
    while(n--)
    {
    	cin>>num;
    	cout<<series[num-1]<<endl;
    }
    
    return 0;
}
