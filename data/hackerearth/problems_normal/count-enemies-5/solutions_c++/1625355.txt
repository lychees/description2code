#include <stdio.h>
#include <stdlib.h>
#include <algorithm>
#include <iostream>
#include <math.h>
#include <string>
#include <vector>
# include <string.h>
#define MAX 1000000007
using namespace std;
int main()
{
	int t=0;
	cin>>t;
	while(t--)
	{
		char str[100001] = {'\0'};
		cin>>str;
		int start = 0;
		int temp = 0;
		int perma = 0;
		int i = 0;
		while(str[i] !='\0')
		{
			if(i==0 && str[i] == 'O')start = 1;
			if(str[i] == 'O')
			{
				if(start == 1)
					++temp;
			}
			else if(str[i] == 'X')
			{
				temp = 0;
				start = 0;
			}
			else
			{
				if(start == 1)
				{
					perma = perma + temp;
					temp = 0;
				}
				else
				{
					start = 1;
				}
			}
			++i;
		}
		perma = perma + temp;
		cout<<perma<<endl;
	
	}
	return 0;
}
