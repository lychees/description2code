#include<bits/stdc++.h>
using namespace std;
vector<string> v;
string tostring ( int Number )
	{
		stringstream ss;
		ss << Number;
		return ss.str();
	}
bool cmp(string a,string b)
	{
		string x = a + b;
		string y = b + a;
		if(x<y)
			return 1;
		return 0;
	}
int main()
	{
		int n,i,a;
		cin>>n;
		for(i=1;i<=n;++i)
			{
				cin>>a;
				v.push_back(tostring(a));
			}
		sort(v.begin(),v.end(),cmp);
		vector<string>::iterator it=v.end();
		--it;
		while(it!=v.begin())
			{
				cout<<*(it);
				--it;
			}
		cout<<*it;
	}
