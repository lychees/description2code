#include <iostream>
using namespace std;
int gcd(int a,int b)
{
	if(b==0)
	return a;
	else
	return gcd(b,a%b);
	
	
}

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,len;
		len=0;
		cin>>n;
		if(n==1)
		{
			cout<<"FALSE"<<endl;
			
		}
		else
		{
		for(int i=1;i<n;i++)
		{
			if(gcd(n,i)==1)
			len++;
		}
		bool flag=false;
		if(len==1)
		{
			flag=true;
		}
	for(int j=2;j<=len/2;j++)
	
	{
		if(len%j==0)
	    flag=true;
	
	
	}
	if(flag==true)
	cout<<"FALSE"<<endl;
	else
	cout<<"TRUE"<<endl;
	}
	}
    //cout << "Hello World!" << endl;
    return 0;

}
