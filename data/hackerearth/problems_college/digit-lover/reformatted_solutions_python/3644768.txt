def count(num):
    temp = num
    sum = 0
    while (temp != 0):
        sum += temp % 10
        temp = temp//10
    return sum


iteration = input()
cases = []
for i in range(0, iteration):
    cases.append(input())

for case in cases:
    temp = case
    while (temp > 9):
        temp = count(temp)
    print(temp)
