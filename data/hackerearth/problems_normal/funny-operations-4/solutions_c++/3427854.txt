#include<bits/stdc++.h>
using namespace std;
static int ans1,ans2,n,a[17];
static bool dp[17][1<<17];
void dp1(int cur,int sel,int xo,int an,int le){
	if(ans1<(an|xo)){
		ans1=an|xo;
		ans2=le;
	}
	else
	if(ans1==(an|xo)&&ans2<le){
		ans2=le;
	}
	if(cur==n){
		return;
	}
	if(dp[cur][sel])
	   return;
	dp[cur][sel]=true;
	dp1(cur+1,sel|(1<<cur),xo^a[cur],an&a[cur],le+1);
	dp1(cur+1,sel,xo,an,le);
}
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		memset(dp,false,sizeof(dp));
		ans1=0;ans2=0;
		scanf("%d",&n);
		for(int i=0;i<n;i++)
		   scanf("%d",&a[i]);
		dp1(0,0,0,~0,0);
		printf("%d %d\n",ans1,ans2);
	}
	return 0;
}
