#include<iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{	
	int t;
	cin>>t;
	cin.ignore(256,'\n');
	while(t--)
	{
		string s,sbstr[10001];
		int count=0,index[10001];
		getline(cin,s);
		int j=0,i=0;
		while(s[i]) 
		{	
			//int j1=s.length()-1;
			char c=s[i];
			if(isspace(c))
			{ 
			 	index[count]=i;
			 	count++;
			}
			i++;
		}
		for(int i=0;i<count;i++)
		{
			sbstr[i]=s.substr(j,index[i]-j);
			j=index[i]+1;
		}
		sbstr[count]=s.substr(j,s.length()-j);
		//for(int i=0;i<=count;i++)cout<<sbstr[i]<<endl;
		int w=count;
		for(int i=0;i<=w/2;i++)
		{
			swap(sbstr[i],sbstr[w-i]);
		}
		for(int i=0;i<=count;i++)cout<<sbstr[i]<<" ";
		cout<<endl;
	}			
}
