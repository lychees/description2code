#include<iostream>
#include<cstdio>
using namespace std;

bool visited[85][85];
int matrix[85][85];
int t;
int ways(int,int);

int main()
{
	int i,j;
	cin >> t;
	
	for(i=0;i<t;i++)
		for(j=0;j<t;j++)
		{
			cin >> matrix[i][j];
			visited[i][j] = false;
		}
	if((matrix[0][0] == 1) || (matrix[t-1][t-1] == 1))
	{
		cout << 0;
		return 0;
	}
	int d =ways(0,0);
	cout << d ;
	return 0;

}

int ways(int i,int j)
{
	if((i<0) || (j<0) || (i>=t) || (j>=t))
		return 0;
	if((i == (t-1)) && (j == (t-1)))
		return 1;
	if(visited[i][j] || (matrix[i][j] == 1))
		return 0;
	visited[i][j] = true;
	int d;
	d = ways(i-1,j) + ways(i+1,j) + ways(i,j-1) + ways(i,j+1);
	visited[i][j] = false;
	return d;	
}
