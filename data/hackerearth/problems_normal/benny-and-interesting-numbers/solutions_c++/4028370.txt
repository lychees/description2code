#include<bits/stdc++.h>
using namespace std;
int arr[1000005];
int arr1[10005];
int count1=0;
int prime[1000005];
int sieve()
{
fill(arr,arr+1000005,0);
arr[0]=1;
arr[1]=1;

for(int i=2;i<=1000;i++)
{
if(arr[i]==0)
{
for(int j=(i*i);j<=1000004;j+=i)
{
arr[j]=1;
}
}
}
for(int i=2;i<=1000005;i++)
{
if(arr[i]==0)
prime[count1++]=i;

}
}
int  main()
{
sieve();
long long t;
cin>>t;
while(t--)
{
long long n;
cin>>n;
long long flag=0;
map<int,int> m;
if(n==1)
{
cout<<"Yes\n";
continue;
}
long long s=n;
for(long long i=0;i<count1;i++)
{
long long j=prime[i];
//cout<<j<<"\n";
if(s==1)
break;
if(s<1000000 && arr[s]==0)
{
m[s]++;
if(m[s]>1)
flag=1;
break;
}
if(((j*j)>n) )
{
m[s]++;
if(m[s]>1)
flag=1;
break;
}
while(s%j==0)
{
s/=j;
m[j]++;
if(m[j]>1)
{
flag=1;
break;
}
}
if(flag==1)
break;
}
if(flag==1)
cout<<"No\n";
else
cout<<"Yes\n";
}
}
