#include<bits/stdc++.h>
using namespace std;
#define int long long
const int BOUND = 1e6+3;
const int MAXN = 2e4+3;
set<int> is;
int dp[BOUND];
#undef int 
int main() {
	#define int long long
	int n,k,m;
	cin>>n>>k>>m;
	for(int i=0;i<m;i++) {
		int x;
		scanf("%lld",&x);
		is.insert(x);
	}
	for(int i=1;i<BOUND;i++)
		dp[i]=1e9+3;
	vector<int> res;
	res.push_back(1);
	int cur=2;
	dp[1]=0;
	while((int)res.size()<n) {
		if(dp[cur]<=k or is.find(cur)!=is.end()) {
			cur++;
			continue;
		}
		int val=1;
		while(val*cur<BOUND) {
			dp[val*cur]=min(dp[val*cur],dp[val]+1);
			val++;
		}
		res.push_back(cur);
		cur++;
	}
	for(int i=0;i<(int)res.size();i++)
		printf("%lld ",res[i]);
	printf("\n");
	return 0;
}
