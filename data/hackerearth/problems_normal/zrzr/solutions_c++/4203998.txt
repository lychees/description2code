#include <iostream>
using namespace std;

using ull = unsigned long long;

ull zrzr(ull n) {
	ull res = 0;
	while (n) {
		n /= 5;
		res += n;
	}
	return res;
}

int main() {
	ios_base::sync_with_stdio(0);
	cin.tie(0);
	
	int t;
	cin >> t;
	while (t--) {
		ull n;
		cin >> n;
		cout << zrzr(n) << '\n';
	}
    return 0;
}
