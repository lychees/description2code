ll possible(ll n)
{
	ll calc,ans,sz,i;
	calc=0;
	ans=0;
	sz=a1.size();
	rep(i,sz)
	{
		n-=a1[i].F;
		if(n<0)
			return 0;
		n+=a1[i].S;
		//trace2(ans,i);
	}
	sz=a2.size();
	rep(i,sz)
	{
		if(n<a2[i].S)
			return 0;
		n=n-a2[i].S+a2[i].F;
	}
	return 1;
}
ll binary()
{
	ll low=1,high=(ll)((ll)mod * (ll)mod), mid;
	while(low<=high)
	{
		mid=(low+high)/2;
		if(possible(mid) && possible(mid-1)==0)
			return mid;
		else if(possible(mid))
			high=mid-1;
		else
			low=mid+1;
	}
	return -1;
}
