#include <iostream>
#include <cmath>
using namespace std;

char names[10001][30];
int cnt[26];
int main()
{
    int t,n,k;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d %d",&n,&k);
    	for(int i=0;i<n;++i)
    	{
    		scanf("%s",names[i]);
    	}
    	for(int i=0;i<k;++i)
    	{
    		for(int j=0;j<26;++j) cnt[j]=0;
    		for(int j=0;j<n;++j)
    		{
    			cnt[names[j][i]-'a']++;
    		}
    		int req=(n+1)>>1,let;
    		for(int j=0;j<26;++j)
    		{
    			if(cnt[j]>=req){ let=j; break;}
    			req-=cnt[j];
    		}
    		printf("%c",let+'a');
    	}
    	printf("\n");
    }
    return 0;
}
