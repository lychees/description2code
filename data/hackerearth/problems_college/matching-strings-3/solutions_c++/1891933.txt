#include<iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--){
		string s1, s2;
		cin>>s1>>s2;
		int l1 = s1.length(), l2 = s2.length(), count=0;
		if(l1<l2){
			int i=0;
			while(l1--){
				if(s1.at(i) == s2.at(i)){
					count++;
				}
				i++;
			}
		}else{
			int i=0;
			while(l2--){
				if(s1.at(i) == s2.at(i)){
					count++;
				}
				i++;
			}
		}
		cout<<count<<endl;
	}
	return 0;
}
