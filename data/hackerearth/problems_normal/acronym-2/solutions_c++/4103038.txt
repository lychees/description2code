#include <iostream>

using namespace std;

int main()
{
    int k;
    string temp;
    cin>>k;
    string str1[k];
    for(int i = 0 ; i < k ; i++){
        cin>>str1[i];
    }
    int n , ans_len = 0;
    cin>>n;
    char ans[n];
    for(int i = 0 ; i < n ; i++){
        cin>>temp;
        int flag = 0;
        for(int j = 0 ; j < k; j++){
            if(temp == str1[j]){
                flag = 1;
                break;
            }
        }
        if(flag == 0){
            ans[ans_len] = (char)temp[0] - 32;
            ans_len++;
        }
    }
    for(int i = 0 , j = 0; i < (ans_len * 2) - 1; i++){
        if(i % 2 == 0){
            cout<<ans[j];j++;
        }else cout<<".";
    }

}
