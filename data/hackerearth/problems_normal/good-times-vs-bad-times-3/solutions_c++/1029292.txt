#include <iostream>
using namespace std;

int g[24][60][60];
int b[24][60][60];
int gt[24][60][60];
int bt[24][60][60];

int gcd(int m,int n){
	if(n==0)
		return m;
	if(m==0)
		return n;
	if(m<n)
		return gcd(m,n%m);
	else
		return gcd(n,m%n);

}	
	


void precal(){
	
	int i,j,k,t;

	for(i=0;i<24;i++){
		for(j=0;j<60;j++){
			
			t = gcd(i,j);
			for(k=0;k<60;k++){
				if(gcd(t,k)==1)
				{
					g[i][j][k] = 1;
					b[i][j][k] = 0;
				}	
				else
				{
					g[i][j][k] = 0;
					b[i][j][k] = 1;
				}	
					
			}
		}
	}
	int s = 0;
	t =0;
	for(i=0;i<24;i++)
		for(j=0;j<60;j++)
			for(k=0;k<60;k++)
			{
				gt[i][j][k] = s = (s + g[i][j][k]);	
				bt[i][j][k] = t = (t + b[i][j][k]);
			}	
	
}
	

int main()
{
	int n,t,i,j,h,m,s;	
    cin>>n;
    precal();
    while(n--){
    	cin>>h>>m>>s;
    	i = gt[23][59][59] - gt[h][m][s] + g[h][m][s];
    	j = bt[23][59][59] - bt[h][m][s] + b[h][m][s];
    	t = gcd(i,j);
    	if(i==0 || j==0)
    		t = 1;
    	cout<<j/t<<":"<<i/t<<"\n";
    }
    return 0;
}
