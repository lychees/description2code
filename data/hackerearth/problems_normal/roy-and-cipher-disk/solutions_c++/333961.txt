#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int tc;
   cin>>tc;
   while(tc--){
   	
   	 string str;
   	 cin>>str;
   	 int s=0;
   	 for(int i=0;i<str.length();i++){
   	   	int check=str[i]-'a';
   	 	int count=0;
   	 	while(s!=check){
   	 		count++;
   	 		s++;
   	 		s%=26;
   	 	}
   	 	if(count>(26-count)) cout<<-1*(26-count)<<" ";
   	 	else cout<<count<<" ";
   	 } 
   	 cout<<endl;
   }
   return 0;
}
