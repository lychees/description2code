#include <iostream>
#include <string>

using namespace std;

int main()
{
	int tc;
	cin>>tc;
	while(tc){
		string a,b;
		cin>>a;
		cin>>b;
		int flag = 0;
		for(int i=0;i<a.size();i++){
			for(int j=0;j<b.size();j++)
				if(a[i] == b[j]){
					flag = 1;break;
				}
			if(flag)
				break;
		}
		if(flag)
			cout<<"YES\n";
		else
			cout<<"NO\n";
		tc--;
	}
	
	return 0;
}
