#include <bits/stdc++.h>
using namespace std;
long long npr(int n,int r,int mod)
{
	long long ans=1;
	for(int i=n-r+1;i<=n;i++)
	{
		ans*=i;
		ans%=mod;
	}
	return ans;
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		string a,b;
		cin>>a>>b;
		int alphaCount[26],mod=1000000007;
		int count[26];
		memset(count,0,sizeof(count));
		memset(alphaCount,0,sizeof(alphaCount));
		for(int i=0;i<a.length();i++)
		{
			alphaCount[a[i]-'a']++;
		}
		for(int i=0;i<b.length();i++)
		{
			count[b[i]-'a']++;
		}
		long long ans=1;
		for(int i=0;i<26;i++)
		{
			if(count[i]!=0 && alphaCount[i]==0)
			{
				ans=0;
				break;
			}
			if(count[i]!=0)
				ans*=(npr(alphaCount[i],count[i],mod));
			ans%=mod;
		}
		printf("%lld\n",ans);
	}
	return 0;
}
