#include <cstdlib>
#include <iostream>
#include <algorithm>

using namespace std;
long long T,i,j,n,a[3000][3000],c; 
int main(int argc, char *argv[])
{   cin>>T;
    while(T--)
    {   cin>>n>>c;
        
    for(i=1;i<=n;i++)
       for(j=1;j<=n;j++)
           cin>>a[i][j];
    for(j=2;j<=n;j++)
       a[1][j]=a[1][j]+a[1][j-1];
    for(i=2;i<=n;i++)
       a[i][1]=a[i][1]+a[i-1][1];
    
    for(i=2;i<=n;i++)
       for(j=2;j<=n;j++)
           a[i][j]+=max(a[i-1][j],a[i][j-1]);
    if(a[n][n]<c)
       cout<<-1<<endl;
    else
       cout<<a[n][n]-c<<endl;
          
        
              
    }
    
    return EXIT_SUCCESS;
}
