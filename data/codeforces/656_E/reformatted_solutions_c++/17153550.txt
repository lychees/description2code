using namespace std;
int x[15][15];
int main() {
  int n;
  cin >> n;
  F(int i = 0; i < n; ++i) {
    F(int j = 0; j < n; ++j) { x[i][j] = 100000000; }
    x[i][i] = 0;
  }
  F(int i = 0; i < n; ++i) {
    F(int j = 0; j < n; ++j) { cin >> x[i][j]; }
  }
  F(int k = 0; k < n; ++k) {
    F(int i = 0; i < n; ++i) {
      F(int j = 0; j < n; ++j) { x[i][j] = min(x[i][j], x[i][k] + x[k][j]); }
    }
  }
  int maxx = 0;
  F(int i = 0; i < n; ++i) {
    F(int j = 0; j < n; ++j) { maxx = max(maxx, x[i][j]); }
  }
  cout << maxx << endl;
}
