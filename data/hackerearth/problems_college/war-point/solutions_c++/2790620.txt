#define ll long long
#define S(x) scanf("%d",&x)
#define Sf(x) scanf("%f",&x)
#define Slf(x) scanf("%lf",&x)
#define Sl(x) scanf("%lld",&x)
#define P(x)  printf("%d\n", x)
#define Pf(x)  printf("%f\n", x)
#define Plf(x)  printf("%lf\n", x)
#define Pl(x)  printf("%lld\n", x)
#define mem(x,i) memset(x,i,sizeof(x))
using namespace std;
ll int a[100005];
ll int cf[100005];
int main()
{
    ll int t,n,i,sum;
    Sl(t);
    while(t--){
        Sl(n);
        sum = 0;
        for (i = 1; i <= n; i++){
            Sl(a[i]);
            sum = sum + a[i];
        }
        cf[1] = a[1];
        for(i = 2; i <= n; i++){
            cf[i] = cf[i-1]+a[i];
        }
        ll int minn = LONG_LONG_MAX;
        ll int tmp,rem;
        for (i = 1; i <= n; i++){
            tmp = cf[i];
            rem = sum-tmp;
            minn = min(minn,abs(tmp-rem));
        }
        Pl(minn);
    }
}
