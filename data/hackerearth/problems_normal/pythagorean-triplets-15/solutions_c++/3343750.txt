#include<bits/stdc++.h>
using namespace std;
int findi(int n)
{
	int c,cnt=0,k,b,a;
	for(int i=2;i<=sqrt(n-1)+1;i++)
		{
			for(int j=1;j<i;j++)
			{
				if(__gcd(i,j)==1 && (i-j)%2 && (i+j)<=n)
				{
					k=0;
					a = i*i - j*j;
					b = 2*i*j;
					c = i*i + j*j;
				//	cout<<a<<' '<<b<<' '<<c<<endl;
					while((++k)*c<=n)
						cnt++;
				}
			}
		}
		return cnt;
}
int main()
{
	int t,k,val,n,m;
	scanf("%d",&t);
	while(t--){
		scanf("%d",&n);
		val = findi(n);
		printf("%d\n",val);
	}
	return 0;
}
