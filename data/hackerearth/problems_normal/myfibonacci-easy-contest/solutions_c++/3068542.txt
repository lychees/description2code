#include <iostream>
using namespace std;

int main()
{
    int a, b,n;
    cin >> a >> b >> n;
    if(n==1){
    	cout << a << endl;
    }
    else if(n==2){
    	cout << b << endl;
    }
    else{
    	for(int i(2); i<n; ++i){
    		int tmp = b;
    		b+=a;
    		a=tmp;
    	}
    	cout << b << endl;
    }
    return 0;
}
