#include <bits/stdc++.h>
using namespace std;

int Test, N, cnt[1000010];

main()
{
    for (int m = 1; m <= 1000; m++)
        for (int n = 1; n < m; n++)
            if (__gcd(m * m - n * n, 2 * m * n) == 1)
                for (int k = 1; 1LL * k * (m * m + n * n) <= 1000000; k++)
                    cnt[k * (m * m + n * n)]++;
    for (int i = 1; i <= 1000000; i++)
        cnt[i] += cnt[i - 1];

    for (scanf("%d", &Test); Test--;)
    {
        scanf("%d", &N);
        printf("%d\n", cnt[N]);
    }
}
