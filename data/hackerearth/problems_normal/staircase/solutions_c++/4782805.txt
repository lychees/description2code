#include <bits/stdc++.h>
using namespace std;
int n;
int dp[100009][2];
int recur(int idx, int ref) {
	if(idx == n) {
		return 1;
	}
	if(idx > n) {
		return 0;
	}
	if(dp[idx][ref]!= -1) {
		return dp[idx][ref];
	}
	int ans = recur(idx+1,ref)+recur(idx+2,ref);
	if(ref) {
		ans += recur(idx+3,ref);
	}
	return dp[idx][ref] = ans;
}

int main()
{
	memset(dp,-1,sizeof dp);
	cin >> n;
	int ans = recur(0, 1);
	if(ans > 0) {
		cout << "Jack-" << ans << endl;
		ans = recur(0,0);
		cout << "Jill-" << ans << endl;
		
	}
	else {
		cout << "ERROR\n";
	}
    
    return 0;
}
