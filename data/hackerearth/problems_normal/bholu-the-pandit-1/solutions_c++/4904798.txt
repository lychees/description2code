#include <iostream>
using namespace std;
#define mod 1000000007
#define ll unsigned long long int

ll gcd(ll a, ll b)
{
    if(b == 0)  return a;
    return gcd(b, a%b);
}

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	ll x,y,z;
    	cin >> x >> y >> z;
    	x = x%mod;
    	y = y%mod;
    	z = z%mod; 
    	
    	ll g = gcd(z,gcd(x,y));
    	g = g%mod;
    	
    	x = x/g;
    	y = y/g;
    	z = z/g; 
    	
    	ll totcub = (((x*y)%mod)*z)%mod;
    	
    	cout << g << " " << totcub << '\n';
    }
    return 0;
}
