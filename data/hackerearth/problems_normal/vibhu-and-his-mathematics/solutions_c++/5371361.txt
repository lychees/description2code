//constants
const int mxn=1e5+1;
const int MOD=1e9+7;
using namespace std;
ll DP[1000001];
int main()
{
	int t=1;
	sc1(t);
	DP[1]=1;
	DP[0]=1;
	DP[2]=2;
	DP[3]=4;
	for(int i=4;i<=1000000;++i)
	{
		DP[i]=(DP[i-1]+1LL*(DP[i-2]*(i-1)));
		DP[i]%=MOD;
	}
	while(t--)
	{
		int n;
		sc1(n);
		printf("%lld\n",DP[n]);
	}
	return 0;
}
