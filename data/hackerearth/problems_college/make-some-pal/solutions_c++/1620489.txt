
#include <stdio.h>
#include<string.h>
using namespace std;
int main()
{
    char s[100],r[100];
    int i,lo,j,k,t,l;
    scanf("%d",&t);
    for(lo=0;lo<t;lo++)
    {	
    	scanf("%s",&s);
    		l=strlen(s);
    	for(i=0;i<l;i++)
  			r[l-i-1]=s[i];  	
   
     
    
    	for(i=0;i<l;i++)
    	{
    		k=0;
    		for(j=0;j<l-i;j++)
    		{
    			if(s[i+j]==r[j])
    				k=1;
    			else
    			{
    				k=0;
    				break;
    			}
    		}
    		if(k==1)
    		{
    			break;
    		}
    	}
    	printf("%d\n",i+l);
    }
    return 0;
}
