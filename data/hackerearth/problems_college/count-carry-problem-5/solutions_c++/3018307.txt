#include <bits/stdc++.h>
using namespace std;
int main()
{
	long long int t,i,f;
	cin>>t;
	while(t--){
		string str1, str2;
		cin >> str1 >> str2;
	    reverse(str1.begin(), str1.end());
		reverse(str2.begin(), str2.end());
		while (str1.size() < str2.size())
			str1+='0';
		while (str2.size() < str1.size())
			str2+='0';
		int ans=0,tr=0;
		for (i=0;i<str1.size();i++) {
                f=str1[i]-'0'+str2[i]-'0'+tr;
			if (f>9)
				{
				    ans++; tr=f-10;
				}
		}
		if (ans==0)
			cout<<"No carry operation"<<endl;
		else if (ans==1)
			cout<<ans<<" carry operation"<<endl;
		else
			cout<<ans<<" carry operations"<< endl;
	}

    return 0;
}
