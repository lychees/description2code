#include <iostream>
using namespace std;

int getSetBits(int x){
	long long int c=0;
	while(x>0){
		c+=x&1;
		x=x>>1;
	}
	return c;
}
	

int main()
{
    int t,c,i,m,n,a;
    cin>>t;
    while(t--){
    	c=0;
    	cin>>m>>n;
    	c = m^n;
    	a = getSetBits(c);
    	cout<<a<<endl;
    }
    return 0;
}
