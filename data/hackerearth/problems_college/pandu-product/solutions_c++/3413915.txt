#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll a[100009];
int main()
{
    ll n, m=0, i, j;
    scanf("%lld",&n);
    for(i=0;i<n;i++)
    	scanf("%lld",&a[i]);
    sort(a,a+n);
    if(a[0]*a[1] < a[n-2]*a[n-1])
    	printf("%lld",a[n-2]*a[n-1]);
    else
    	printf("%lld",a[0]*a[1]);
    return 0;
}
