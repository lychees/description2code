#include <iostream>
#include <cstring>
#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		char a[10],b[10];
		int c=0,i,j,d=0;
		scanf("%s",a);
		scanf("%s",b);
		i=strlen(a)-1;
		j=strlen(b)-1;
		while(i>=0||j>=0)
		{
	
			if(a[i]+b[j]+c>105)
			{d++;
				c=1;
			}
			else
			c=0;
			i--;
			j--;
		}
		if(i>j)
		if(a[i]+c>57)
		d++;
		if(j>i)
		if(b[j]+c>57)
		d++;
		if(d==0)
		cout<<"No";
		else
		cout<<d;
    	cout<<" carry operation";
    	if(d>>1)
    	cout<<"s";
    	cout<<endl;
	}
return 0;
}
