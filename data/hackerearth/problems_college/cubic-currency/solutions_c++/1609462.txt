#include <iostream>
using namespace std;

long long dp[25][10010];

void ini(){
	dp[0][0]=1;
	for(int i=1;i<=21;i++){
		for(int j=0;j<=10000;j++){
			if(j>=i*i*i) dp[i][j]=dp[i-1][j]+dp[i][j-i*i*i];
			else dp[i][j]=dp[i-1][j];
		}
	}
}

int main(){
	ini();
	int t,n;
	cin>>t;
	while(t--){
		cin>>n;
		cout<<dp[21][n]<<endl;
	}
	return 0;
}
