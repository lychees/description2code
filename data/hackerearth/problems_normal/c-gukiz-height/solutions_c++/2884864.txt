#define ll long long

#define sz(x) (int)x.size()

ll a[100005];

int main()
{
    ll n,h,i;
    scanf("%lld%lld",&n,&h);
    ll sum = 0;
    for(i=1;i<=n;i++)
    {
        scanf("%lld",&a[i]);
        sum += a[i];
        if(sum >= h - (i*(i+1))/2)
        {
            printf("%d\n",i);
            return 0;
        }
    }
    ll ans = (ll)1e16;
    for(i=1;i<=n;i++)
    {
        h -= a[i];
        ll lo = 0 , hi = INT_MAX , mid;
        ll tmp = 0;
        while(lo <= hi)
        {
            mid = (lo+hi)/2;
            ll days = n*mid + i;

            if(days >= INT_MAX)
            {
                hi = mid - 1;
                continue;
            }

            if(mid * sum >= h - (days*(days+1))/2)
            {
                //cout << "AAAA";
                tmp = mid;
                hi = mid - 1;
            }
            else
            {
                lo = mid + 1;
            }
        }
        ans = min(ans,tmp*n + i);
    }
    if(ans == (ll)1e16)
        ans = -1;
    printf("%lld\n",ans);
    return 0;
}
