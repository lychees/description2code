n=input()
alist=map(int ,raw_input().split())
track=[1 for _ in range(n)]
for i in range(n):
	for j in range(i):
		if alist[i]>0 and alist[j]<0 and abs(alist[j])<abs(alist[i]):
			track[i]=max(track[i],track[j]+1)
		if alist[i]<0 and alist[j]>0 and abs(alist[j])<abs(alist[i]):
			track[i]=max(track[i],track[j]+1)	
print max(track)
