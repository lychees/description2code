    #include <iostream>
    #include <cstdio>
    
    using namespace std;
    int array[100001];
    void cleanup(int len);
    int main() {
        int cases,n,k,input,max,total;
        scanf("%d\n",&cases);
        for(int t=0;t<cases;t++)
        {
            max = 0;
            total = 0;
            scanf("%d %d\n",&n,&k);
            for(int l=0;l<n;l++)
            {
                scanf("%d",&input);
                array[input]++;
            }
            for(int l=1;l<=k;l++)
            {
                if(array[l]>max)
                    max = array[l];
                total +=array[l];
            }
            printf("%d\n",(total-max));
            cleanup(k+1);
        }
        return 0;
    }
    
    void cleanup(int len)
    {
        for(int m=0;m<len;m++)
            array[m] = 0;
    }
