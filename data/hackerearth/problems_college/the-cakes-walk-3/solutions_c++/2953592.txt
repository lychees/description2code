#include<iostream>
#include<math.h>
using namespace std;
int main()
{
int t;
cin>>t;
while(t-- > 0)
{
	int a;
	cin>>a;
	int flag=0,i,j;
	for(i=a;i>=0;i--)
	{
		for(j=0;j<=a;j++)
		{
			int n=pow(i,j)+pow(j,i);
			if(n==a)
			{flag=1;break;}
		}
		if(flag==1)
		{break;}
	}
	cout<<i<<" "<<j<<endl;
}
}
