#include<bits/stdc++.h>
#define ll long long





using namespace std;

ll cnt[1000000];
bool mark[1000000];
ll ans[1000000];

void init(ll n)
{
    ll k=sqrt(n);
    for(ll i=1;i<=n;i++)
    {
        cnt[i]=0;
        mark[i]=true;
        ans[i]=0;
    }
    cnt[2]=1;
    ans[1]++;



    for(ll i=4;i<=n;i+=2)
    {
        mark[i]=false;
        cnt[i]++;
    }
    for(ll i=3;i<=n;i++)
    {
        if(mark[i])
        {
            cnt[i]=1;
            for(ll j=i+i;j<=n;j=j+i )
            {
                mark[j]=false;
                cnt[j]++;
            }
        }
        ans[cnt[i]]++;

    }



}
int main()
{

    ll t,q,p;
    cin>>t>>q;


    init(t);
    while(q--)
    {
        cin>>p;
        if(p>t)
        cout<<"0\n";
        else
        cout<<ans[p]<<"\n";

    }

}
