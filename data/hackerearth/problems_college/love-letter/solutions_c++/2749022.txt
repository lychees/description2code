#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll pow2[31];

void solve(){
	pow2[0]=1;
	for(int i=1;i<=31;i+=1){
		pow2[i]=2*pow2[i-1];
	}
}


int main(){
	solve();
	//cout<<pow2[30]<<endl;
	//cout<<(pow(2,30))<<endl;
	int t,n,k;
	cin>>t;
	while(t--){
		cin>>n>>k;
		int x=pow2[n]-1;
		//cout<<(x&k)<<endl;
		int y=x&k;
		if(y!=x)cout<<"NO"<<endl;
		else cout<<"YES"<<endl;
	}
	return 0;
}
