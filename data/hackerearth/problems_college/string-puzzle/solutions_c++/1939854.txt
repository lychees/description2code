#include<bits/stdc++.h>
using namespace std;
void permute(char t[],long int l,long int r)
{
    long int i;
    if(l==r)
    {
        cout<<t<<"\n";
    }
    else
    {
        for(i=l;i<=r;i++)
        {
            swap(t[i],t[l]);
            permute(t,l+1,r);
            swap(t[i],t[l]);
        }
    }
}
int main()
{
    char str[10];
    long int Q,S,E;
    cin>>str;
    cin>>Q;
    while(Q--)
    {
        cin>>S>>E;
        permute(str,S,E);
        cout<<"\n";

    }
}
