for _ in range(input()):
    a = list(raw_input())
    arr = []
    for item in a:
        if item.isalpha() or item == "*":
            arr.append(item)
    # print ''.join(arr)
    x = arr.index("*")
    if x > (len(arr)-1)/2:
        x = len(arr)-1-x
        print arr[x]
    else:
        print arr[::-1][x]
