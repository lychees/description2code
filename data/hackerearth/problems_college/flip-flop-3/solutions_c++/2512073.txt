#include<bits/stdc++.h>
#include<string>
using namespace std;

int main()
{
    int T;
    string s;
    cin>>T;
    while(T--)
    {
        cin>>s;
        int count=0;
        for(int i=0; i<s.length()-1; i++)
        {
            if(s[i]==s[i+1])
                count++;
        }
        cout<<count<<endl;
    }
    return 0;
}
