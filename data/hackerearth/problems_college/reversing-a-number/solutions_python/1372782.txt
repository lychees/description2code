'''

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
if __name__ == '__main__':
	cases = int(raw_input())
	start = 1
	while start <= cases:
		number = raw_input().strip()
		reverse_number = number[::-1]
		if int(reverse_number) % 2 == 0:
			print "EVEN"
		else:
			print "ODD"
		start = start + 1
