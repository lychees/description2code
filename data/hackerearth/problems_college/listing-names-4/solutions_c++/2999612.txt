#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	scanf("%d",&n);
	int i;
	char a[300];
	map<string,int>m;
	for(i=0;i<n;i++)
	{
		scanf("%s",a);
		m[a]++;
	}
	//std::map<string,int>iterator it;
	for(std::map<string,int>::iterator it=m.begin();it!=m.end();it++)
	{
		cout<<it->first<<" "<<it->second<<endl;
	}
	return 0;
}
