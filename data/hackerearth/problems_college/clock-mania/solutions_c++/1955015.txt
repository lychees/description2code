#include <iostream>
#include <stdio.h>

using namespace std;

int getDifference (int num1, int num2)
{
    if(num1 < num2)
    {
        int temp = num1;
        num1 = num2;
        num2 = temp;
    }

    int diff = (((360 - num1) + num2) > (num1 - num2)) ? (num1 - num2) : ((360 - num1) + num2);
    return diff;
}

int main()
{
    int T = 0;

    scanf("%d", &T);

    for(int test=0; test< T; test++)
    {
        int H, M;
        scanf("%d", &H);
        scanf("%d", &M);

        int hourAngle = ((H % 12) * 30) + (M * 0.5);
        int minuteAngle = 6 * M;

        int difference = getDifference(hourAngle, minuteAngle);
        printf("%d\n", difference);

    }

    int i; cin >> i;

    return 0;
}
