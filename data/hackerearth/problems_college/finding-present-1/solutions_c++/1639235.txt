#include <bits/stdc++.h>
#define gc getchar_unlocked

void st(long long int &x)
{
    register long long int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}

typedef long long int ll;
using namespace std;

int main()
{
   ll n,m,i,x;
   st(n);
   vector<ll> v;
   
   for(i=0;i<n;i++)
   {
   	 st(x);
   	 v.push_back(x);
   }
   map<ll,ll> a;
   st(m);
   
   for(i=0;i<m;i++)
   {
   	 st(x);
   	 if(a.find(x) == a.end())
   	    a[x] = 1;
   	 else
   	   a[x]++;
   }
   
   for(i=0;i<n;i++)
   {
   	 a[v[i]]--;
   }
   map<ll,ll>::iterator it = a.begin();
   for(it = a.begin();it!=a.end();it++)
   {
   	 if(it->second > 0)
   	   cout<<it->first<<" ";
   }
   
    return 0;
}
