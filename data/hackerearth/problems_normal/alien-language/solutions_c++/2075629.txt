#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    int t,flag;
    cin>>t;
    while(t--)
    {

    	int flag=0;
    	char s[100001];
    	cin>>s;
    	char p[100001];
    	cin>>p;

    	int hash[26]={0};
    	int i;

    	for(i=0;s[i];i++)
        {
            ++hash[s[i]-'a'];
        }

        for(i=0;p[i];i++)
        {
            if(hash[p[i]-'a'])
            {flag=1;break;}
        }

        if(flag)
    	{cout<<"YES\n";}
    	else
    	{cout<<"NO\n";}





    }
    return 0;
}
