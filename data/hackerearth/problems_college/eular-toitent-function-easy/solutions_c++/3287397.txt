#include <bits/stdc++.h>

using namespace std;

int isP[10000005];

void sieve()
{
    int i, j;
    for (i = 2; i <= 10000000; ++i) {
        if (!isP[i]) {
            for (j = 1; i*j <= 10000000; ++j)
                isP[i*j] = i;
        }
    }
}

int main()
{
    sieve();
    //freopen("in10.txt", "r", stdin);
    //freopen("out10.txt", "w", stdout);
    int t, n, last, x, ans;
    cin >> t;
    while (t--) {
        cin >> n;
        last = -1;
        ans = n;
        while (n > 1) {
            x = isP[n];
            if (x != last) {
                ans /= x;
                ans *= (x-1);
            }
            last = x;
            n /= x;
        }
        cout << ans << endl;
    }
    return 0;
}
