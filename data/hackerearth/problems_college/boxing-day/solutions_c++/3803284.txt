#include <algorithm>
#include <iostream>
using namespace std;

int main()
{
	int n, k;
	cin>>n>>k;
	int str[n];
	fill(str ,str+n,1);
	for(int i=0; i<k; i++)
	{
		int a, b;
		cin>>a>>b;
		str[a-1]=str[a-1]+str[b-1];
		str[b-1]=0;
	}
	
	int cnt=0;
	for(int i=0;i<n;i++)
		if(str[i]!=0)
			cnt++;
	
	cout<<cnt;
	
}
