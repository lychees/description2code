#include<bits/stdc++.h>
#define ll long long
#define pb push_back
#define mp make_pair
using namespace std;
vector <int> g[100005];
bool v[100005];
ll dp[100005];
int n;
ll dfs(int s)
{
	if(s==n)
		return 1;
	if(v[s])
		return dp[s];
	v[s]=1;
	ll ans=0;
	for(int i=0;i<g[s].size();i++)
		ans+=dfs(g[s][i]);
	dp[s]=ans;
	return ans;
}
int main()
{
	cin>>n;
	while(1)
	{
		int x,y;
		cin>>x>>y;
		if(x==0 and y==0)
			break;
		x--;
		y--;
		g[x].pb(y);
	}
	n--;
	memset(v,0,sizeof(v));
	cout<<dfs(0)<<endl;
	return 0;
}
