t=int(raw_input())
for i in range (t):
	sum=0
	odd=0
	c=[]
	m=''
	N=int(raw_input())
	c=raw_input()
	c=c+' '
	d=0
	j=-1
	while (d < N):
		j+=1
		if (c[j].isspace()):
			d=d+1
			a=int(m)
			sum=sum+a
			if (a%2!=0):
				odd=odd+1
			m=''
			continue
		else:
			m=m+c[j]
		
	if (sum%2!=0):
		print ('NO')
	else:
		if (odd%2==0):
			print ('YES')
		else:
			print ('NO')
