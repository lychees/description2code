#include <iostream>
#include <vector>
#define ll long long
#define MD 1000000007
using namespace std;

ll pw(ll a, ll p) {
	if(p==0) return 1;
	if(p==1) return a;
	if(p&1) return (a*pw(a,p-1))%MD;
	ll x=pw(a,p/2);
	return (x*x)%MD;
}

ll modInv(ll a) {
	ll x=pw(a,MD-2);
	//cout<<"returning "<<x<<endl;
	return pw(a,MD-2);
}

int main()
{
    int N;
    cin>>N;
    vector<int> arr(N);
    ll prod=1;
    ll zcount=0;
    for(int i=0;i<N;i++) {
    	cin>>arr[i];
    	if(arr[i]==0)
    		zcount++;
    	else
    		prod=(prod*arr[i])%MD;
    }
    //cout<<prod<<endl;
    int Q,typ,num,i;
    cin>>Q;
    while(Q--) {
    	//cout<<zcount<<endl;
    	cin>>typ;
    	if(!typ) {
    		cin>>i>>num;
    		i--;
    		if(arr[i]!=0)
    			prod=(prod*modInv(arr[i]))%MD; 
    		else
    			zcount--;
			arr[i]=num;
    		if(num==0) {
    			zcount++;
    		} else {
    			prod=(prod*arr[i])%MD;
    		}
    	} else {
    		cin>>i; i--;
    		//cout<<"returning "<<modInv(arr[i])%MD<<endl;
    		if(zcount>1)
    			cout<<0<<endl;
    		else if(zcount==1) 
    			if(arr[i]!=0)
    				cout<<0<<endl;
    			else
					cout<<prod%MD<<endl;    			
    		else
    			cout<<(prod*modInv(arr[i])%MD)<<endl;
    	}
    }
    return 0;
}
