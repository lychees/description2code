#include<iostream>
#include<stack>

using namespace std;
typedef long long int LL;
int main() {
  int n, x;
  cin >> n >> x;
  int count;
  int level;
  bool first = true;
  while (n--) {
    cin >> level;
    if (level > x) {
      if (first) {
        first = false;
        continue;
      }
      else {
        break;
      }

    }
    count++;
  }
  cout << count << endl;

  return 0;


}
