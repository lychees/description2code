#include <iostream>
using namespace std;
#define ll long long int
bool prime[10000001];
void sieve()
{
	ll i,j;
	for(i=0;i< 10000001 ; i++)
	{
		prime[i] = true;
	}
	prime[0] = false;
	prime[1] = false;
	
	for(i = 2 ; i*i < 10000001 ; i++)
	{
		if(prime[i])
		for(j = i*i ; j < 10000001 ; j+= i)
			prime[j] = false;
	}
}
int main()
{
	ll t,a,b,ans,i;
	sieve();
//    cin>>t;
    scanf("%lld",&t);
    while(t--)
    {
//    	cin>>a>>b;
    	scanf("%lld %lld",&a,&b);
    	if(a == 1 && b == 1)
    	{
    		printf("0\n");
    		continue;
    	}
    	if(a% 2 == 0)
		{
			if(a <= 2)
			{
				ans = 2;
				a = 3;
			}
			else
			{	a++;ans=0;}
		}
    	else
    	{
			if(a <= 2)
			{
				ans = 2;
				a = 3;
			}
			else
				ans=0;
    		
    	}
    	
    	for(i = a;i <= b ;i = i+2)
    	{
    		if(prime[i])
    			ans += i;
//    			cout<<i<<" ";
    	}
//    	cout<<ans<<endl;
    	printf("%lld\n",ans);
    }
    return 0;
}
