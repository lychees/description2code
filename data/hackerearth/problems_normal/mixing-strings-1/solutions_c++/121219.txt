string attach(string s,string t)
{
    int n1 = (int)s.size(),n2 = (int)t.size();
    for(int i = 0;i < n1;++i)
    {
        if((n1-i)>n2)
        {
            continue;
        }
        int k = i,j;
        bool fl = 1;
        for(j = 0;k < n1;++j)
        {
            if(t[j]!=s[k])
            {
                fl =0;
            }
            ++k;
        }
        if(fl)
        {
            s += t.substr(j);
            return s;
        }
    }
    s += t;
    return s;
}
