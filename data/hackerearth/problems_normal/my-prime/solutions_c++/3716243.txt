#include <iostream>
using namespace std;

int main()
{
    int n,i,s[100],j,f=0;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>s[i];
    }
    for(i=0;i<n;i++)
    {
    	for(j=0;j<n;j++)
    	{
    		if(i!=j&&s[i]%s[j]==0)
    		{
    			f=1;
    			break;
    		}
    	}
    	if(f==0)
    	{
    		cout<<s[i]<<" ";
    	}
    	else{f=0;}
    }
    return 0;
}
