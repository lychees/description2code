#include <bits/stdc++.h>
using namespace std;

#define ll long long int
#define mp(i,j) make_pair(i,j)
#define pb(x) push_back(x)
#define MAXN 10010
#define MOD 1000000007

ll team[MAXN][26];
ll act[26];
ll score[MAXN];
ll sz[MAXN];

vector<pair<pair<ll,ll> ,ll> > v;

int main()
{
	ll t,i,j,n,len,ans,ind;

	string s;
	string x;

	cin>>t;

	while(t--) {
		memset(team,0,sizeof(team));
		memset(act,0,sizeof(act));
		memset(score,0,sizeof(score));

		cin>>n>>s;

		for(i=0;i<n;i++) {
			cin>>x;

			len = x.length();

			sz[i] = len;

			for(j = 0;j<len;j++) {
				team[i][x[j]-'a']++;
			}
		}

		len = s.length();

		for(i=0;i<len;i++) {
			act[s[i]-'a']++;
		}

		for(i=0;i<26;i++) {
			for(j=0;j<n;j++) {
				score[j] = score[j] + (team[j][i]*act[i]);
			}
		}
		
		for(i=0;i<n;i++) {
			v.pb(mp(mp(-score[i],sz[i]),i));
		}

		sort(v.begin(),v.end());

		cout<<(v[0].second+1)<<endl;

		v.clear();
	}

	return 0;
}
