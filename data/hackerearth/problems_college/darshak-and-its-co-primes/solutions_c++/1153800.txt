#include <iostream>
using namespace std;

#define llu long long unsigned

llu gcd(llu a,llu b)
{
    if(b==0)
    return a;
    else return gcd(b,a%b);
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
       llu a,b;
       cin>>a>>b;
       a=gcd(a,b);
       if(a==1) cout<<"Is a Co-Prime"<<endl;
       else cout<<"Not a Co-Prime"<<endl;
    }
    return 0;
}
