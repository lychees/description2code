#include<iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
	ios::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
	    int n,i,dp[101][10001]={0};
        int a[101];
        cin>>n;
        i=1;
        long long sum=0;
        while(i<=n)
        {
            cin>>a[i];
            sum+=a[i];
            i++;
        }
        if(sum%2==1)
        {
            cout<<"NO"<<endl;
            continue;
        }
        for(i=0;i<=n;i++)
        {
            dp[i][0]=1;
        }
        for(i=1;i<=n;i++)
        {
            for(int j=0;j<=10000;j++)
            {
                if(dp[i-1][j]==1)
                {
                    dp[i][j]=1;
                }
                if(j-a[i]>=0 && dp[i-1][j-a[i]]==1)
                {
                    dp[i][j]=1;
                }
            }
        }
        if(dp[n][sum/2]==1)
        {
            cout<<"YES"<<endl;
        }
        else
        {
            cout<<"NO"<<endl;
        }
	}
	return 0;
}
