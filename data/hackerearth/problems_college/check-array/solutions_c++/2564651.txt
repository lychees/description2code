#include<bits/stdc++.h>
using namespace std;
int main()
{
	long int T,i,j,k,N,a[100001];
	cin>>T;
	while(T--)
	{
		cin>>N;
		long int sum=0;
		for(i=0;i<N;i++)
		{
			scanf("%ld",&a[i]);
			sum+=a[i];
		}
		long int left=0,flag=0,right;
		for(i=0;i<N;i++)
		{
			right=sum-a[i]-left;
			if(left==right)
			{
				flag=1;
				break;
			}
			left=left+a[i];
		}
		if(flag==1)
		cout<<"YES\n";
		else
		cout<<"NO\n";
	}
}
