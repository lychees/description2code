int foo(ull k, ull n)
{
    if(n == 1 || k == (n+1)/2) return 0;
    int n2 = (n+1)/2-1;
    if(k < (n+1)/2) return foo(k,n2);
    k = k-n2-1;
    return !foo(n2-k+1,n2);
}
