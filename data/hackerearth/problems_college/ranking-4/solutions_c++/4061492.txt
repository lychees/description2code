#include <bits/stdc++.h>
using namespace std;
#define ll long long int
ll a[1000001], b[1000001];
map < ll, ll > m;
int main()
{
    ll t, n, k, i, r, p, j;
    cin>>t;
    while(t--) {
    	m.clear();
    	cin>>n;
    	for(i = 0 ; i < n; i++) {
    		cin>>a[i];
    		b[i] = a[i];
    	}
    	sort(a, a+n);
    	r = 1;
    	p = 1;
    	for(i = n-1; i >= 0; i--) {
    		j = i;
    		while(a[j] == a[i]) {
    			m[a[j]] = r;
    			j--;
    			p++;
    		}
    		r = p;
    		i = j + 1;
    	}
    	for(i = 0 ; i < n; i++)
    		cout<<m[b[i]]<<" ";
    	cout<<endl;
    }
    return 0;
}
