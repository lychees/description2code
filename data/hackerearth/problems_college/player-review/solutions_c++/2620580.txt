#include <iostream>
using namespace std;
#define ll long long
ll max(ll a[], ll s)
{	ll m=0,h=0;
   for(ll i=0;i<s;i++)
   {	h=h+a[i];
     	if(h<0)
        	h=0;
     	if(m<h)
        	m=h;
    }
    return m;
} 

int main()
{	ll n;
	cin>>n;
	ll a[n];
	for(ll i=0;i<n;++i)
		cin>>a[i];
	ll x=max(a,n);
	if(x>0)
		cout<<x<<"\n";
	else
		cout<<"0\n";
   	return 0;
}
