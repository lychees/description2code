#include <iostream>
using namespace std;

int main() {
	string x;
	cin>>x;
	int n = x.length();
	int count = n,i,j,c = 1;
	for(i=0;i<n-1;i++)
	{
		if(x[i]==x[i+1])
		{
			c++;
		}
		else
		{
			count+=(c*(c-1))/2;
			c = 1;
		}
	}
	if(c!=1)
		count+=(c*(c-1))/2;
	cout<<count;
	return 0;
}
