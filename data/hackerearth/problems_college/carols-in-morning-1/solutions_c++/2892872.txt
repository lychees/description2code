#include<bits/stdc++.h>
using namespace std;

#define fastScan ios_base::sync_with_stdio(0); cin.tie(NULL);



 int arr[100][100]={0},n,sz;
 
 int res(int *ar,int ind,int count){
     if(ind==n)return count;
     int i,j,ans1,ans2=100000000;
     for(i=0;i<sz;i++){
         if(arr[ind][i]){
             if(ar[i])
               ans1=res(ar,ind+1,count);
              else{
                  ar[i]=1;
                  ans1=res(ar,ind+1,count+1);
                  ar[i]=0;
              }
              if(ans1<ans2)
                ans2=ans1;
         }
     }
     return ans2;
 }
 
 
int main()
{
    fastScan;
    int t;
    cin>>t;
    while(t--){
        int i,ans=0,j,ar[100]={0};
        for(i=0;i<100;i++){
            ar[i]=0;
            for(j=0;j<100;j++)
             arr[i][j]=0;
        }
        string str;
        cin>>n;
        for(i=0;i<n;i++){
            cin>>str;
            sz=str.size();
            for(j=0;j<str.size();j++){
                if(str[j]=='Y')
                  arr[i][j]=1;
            }
        }
        ans=res(ar,0,0);
        cout<<ans<<endl;  
    }
    return 0;
}
