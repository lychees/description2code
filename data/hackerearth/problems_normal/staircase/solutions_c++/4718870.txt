#include<bits/stdc++.h>
using namespace std;
#define ull long long int
int main()
{
    ios::sync_with_stdio(false);
    ull n,i;
    cin>>n;
    if(n<0){
        cout<<"ERROR\n";
        return 0;
    }
    ull jack[n+5],jill[n+5];
    jack[0]=1;
    jack[1]=1;
    jack[2]=2;
    for(i=3;i<=n;i++)
        jack[i]=jack[i-3]+jack[i-2]+jack[i-1];
    jill[0]=1;
    jill[1]=1;
    jill[2]=2;
    for(i=3;i<=n;i++)
        jill[i]=jill[i-1]+jill[i-2];
    cout<<"Jack-"<<jack[n]<<"\nJill-"<<jill[n]<<"\n";
    return 0;
}
