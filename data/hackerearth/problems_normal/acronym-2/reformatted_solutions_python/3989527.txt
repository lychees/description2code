T = int(raw_input())
ls = []
for i in range(0, T):
    ls.append(raw_input())
N = int(raw_input())
msg = list(raw_input().split())
lst = []
for i in range(0, N):
    if (ls.count(msg[i]) == 0):
        lst.append(chr(ord(msg[i][0])-32))
print ".".join(map(str, lst))
