#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		string key;
		cin>>key;
		int n;
		cin>>n;
		vector<pair<string,string>> v;
		for(int i=0;i<n;i++){
			string str;
			cin>>str;
			v.push_back(make_pair(str,key));
		}
		sort(v.begin(),v.end(),[](pair<string,string> p1,pair<string,string> p2){
			string key=p1.second;
			string str1=p1.first;
			string str2=p2.first;
			int i;
			for( i=0;i!=str1.size()&&i!=str2.size();i++){
				if(str1[i]!=str2[i]){
					return key.find(str1[i])<=key.find(str2[i]);
				}
			}
			if(i==(int)str1.size())
				return true;
			return false;
		});
		for(int i=0;i<v.size();i++){
		    cout<<v[i].first<<endl;
		}
	}
}
