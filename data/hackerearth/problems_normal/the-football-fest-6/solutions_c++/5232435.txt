#include <iostream>
using namespace std;

int main()
{
	int t,n,x,i,j,p,q;
	char c;
	cin>>t;
	while(t--)
	{
		cin>>n>>x;
		p=x;
		q=-1;
		while(n--)
		{
			cin>>c;
			if(c=='P')
			{
				cin>>x;
				q=p;
				p=x;
			}
			else
			{
				x=p;
				p=q;
				q=x;
			}
		}
		cout<<"Player "<<p<<endl;
	}
	
    return 0;
}
