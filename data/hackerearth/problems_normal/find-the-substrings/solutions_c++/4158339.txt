#include<bits/stdc++.h>
using namespace std;

int main()
{
	int n;
	cin>>n;
	while(n--)
	{
		string s;
		cin>>s;
		long long sum1=0,sum2=0,sum3=0,sum4=0,sum5=0,sum6=0;
		long long val=0;
		for(int i=0;i<s.length();i++)
		{
			if(s[i]=='a')
			val++;
			else
			{
				sum1+= ((val*(val+1))/2);
				val=0;
			}
		}
		sum1+= ((val*(val+1))/2);
		val=0;
		
		for(int i=0;i<s.length();i++)
		{
			if(s[i]=='b')
			val++;
			else
			{
				sum2+= ((val*(val+1))/2);
				val=0;
			}
		}
		sum2+= ((val*(val+1))/2);
		val=0;
		
		for(int i=0;i<s.length();i++)
		{
			if(s[i]=='c')
			val++;
			else
			{
				sum3+= ((val*(val+1))/2);
				val=0;
			}
		}
		sum3+= ((val*(val+1))/2);
		val=0;
		
		
		for(int i=0;i<s.length();i++)
		{
			if(s[i]!='a')
			val++;
			else
			{
				sum4+= ((val*(val+1))/2);
				val=0;
			}
		}
		sum4+= ((val*(val+1))/2);
		val=0;
		
		for(int i=0;i<s.length();i++)
		{
			if(s[i]!='b')
			val++;
			else
			{
				sum5+= ((val*(val+1))/2);
				val=0;
			}
		}
		sum5+= ((val*(val+1))/2);
		val=0;
		
		for(int i=0;i<s.length();i++)
		{
			if(s[i]!='c')
			val++;
			else
			{
				sum6+= ((val*(val+1))/2);
				val=0;
			}
		}
		sum6+= ((val*(val+1))/2);
		val=0;
		
		cout<<sum4+sum5+sum6-sum1-sum2-sum3<<endl;
	}
}
