mx, mod = 10**6+13, 10**6+3
arr = [0]*mx
arr[0] = 1
for i in range(1, mx):
    arr[i] = i * arr[i-1]
    arr[i] %= mod

for _ in range(input()):
    n, x = map(int, raw_input().split())
    if n > mx:
        print 0
    else:
        arr[n] %= mod
        cal = arr[n]*x
        cal %= mod
        print cal
