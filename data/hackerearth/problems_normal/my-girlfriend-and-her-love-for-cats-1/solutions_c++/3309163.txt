#include <iostream>
#include<algorithm>
using namespace std;
#define MAX 1000000
long long c[MAX],s[MAX];
int main()
{
	int n;
	long long sum=0;

	cin>>n;

	for(int i=0;i<n;i++)cin>>c[i];for(int i=0;i<n;i++)cin>>s[i];
	sort(s,s+n);
	sort(c,c+n);
	for(int i=0;i<n;i++){
		sum+=s[i]*c[i];
	}
	cout<<sum;
	return 0;
}
