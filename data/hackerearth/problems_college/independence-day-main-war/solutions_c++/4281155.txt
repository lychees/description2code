#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t,n,i;
    float x,y;
    cin>>t;
    while (t--)
    {
        float a1=0,a2=0;

        cin>>n;
        for (i=0;i<n;i++)
        {
            cin>>x>>y;
            if (i==0)
            {
                a1=x;a2=y;
            }
            else
            {
                a1=((a1*i)+x)/(i+1);
                a2=((a2*i)+y)/(i+1);
            }
        }
        printf("(%0.2f , %0.2f)\n",a1,a2);
    }
    return 0;
}
