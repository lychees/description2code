#include <iostream>
#include<vector>
#include<set>
#include<map>
using namespace std;
int n;
vector<int> v;
map<int,int> m;
int main()
{
	int n;
	scanf("%d",&n);
	v.resize(n);
	for(int i=0;i<n;i++){
		scanf("%d",&v[i]);
	}
	int countt=0;
	for(int i=0;i<n;i++){
		for(int j=i+1;j<n;j++){
			for(int k=j+1;k<n;k++){
				if(v[k]-v[j]==v[j]-v[i]){
					countt++;
				}
			}
		}
	}
	printf("%d\n",countt);
    //cout << "Hello World!" << endl;
    return 0;
}
