rr = raw_input
for tc in xrange(int(rr())):
    N, K = map(int, rr().split())
    A = [set(rr().split()[1:]) for i in xrange(N)]

    cache = {}

    def dp(x, h):
        if x >= len(A):
            return 99999 if len(h) < K else 0
        if (x, h) not in cache:
            ans = 1+dp(x+1, tuple(sorted(set(h) | A[x])))
            ans = min(ans, dp(x+1, h))
            cache[(x, h)] = ans
        return cache[(x, h)]
    ans = dp(0, tuple())
    print ans if ans < 99999 else -1
