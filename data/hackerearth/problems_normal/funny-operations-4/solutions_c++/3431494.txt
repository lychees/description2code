#include <bits/stdc++.h>
#define LL long long
using namespace std;

int main()
{
    LL T;
    cin>>T;
    while(T--)
    {
    	LL N;
    	cin>>N;
    	LL a[N];
    	for(LL i=0;i<N;i++)
    	cin>>a[i];
    	
    	LL ans=-1;
    		LL maxcnt=0;
    		for(LL mask=0;mask<(1<<N);mask++)
    		{
    			LL P;
    			LL xorval=0;
    			LL andval=-1;
    			LL cnt=0;
    		
    			for(LL i=0;i<N;i++)
    			{
    				if(mask & (1<<i))
    				{
    					cnt++;
    					xorval=xorval^a[i];
    					if(andval==-1)andval=a[i];
    					else andval=(andval & a[i]);
    				}
    			}
    			P=(xorval|andval) ;
    			if(P>ans)
    			{
    				ans=P;
    				maxcnt=cnt;
    			}
    			else if(P==ans)
    			{
    				ans=P;
    				maxcnt=max(maxcnt,cnt);
    			}
    			
    		}
    	
    		
    	cout<<ans<<" "<<maxcnt<<"\n";
    	
    }
    return 0;
}
