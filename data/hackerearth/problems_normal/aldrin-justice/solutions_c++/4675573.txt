#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    long long int t,a1,a2,b1,b2;
    cin>>t;
    while(t--)
    {
    	cin>>a1>>a2>>b1>>b2;
    	if(max(a1,a2)==min(b1,b2) || min(a1,a2)==max(b1,b2))
    	cout<<"Point\n";
    	else if(a1==a2 && a1>min(b1,b2) && a1<max(b1,b2) || b1==b2 && b1>min(a1,a2) && b1<max(a1,a2))
    	cout<<"Point\n";
    	else if(max(a1,a2)<min(b1,b2) || min(a1,a2)>max(b1,b2))
    	cout<<"Line\n";
    	else 
    	cout<<"Nothing\n";
    }
    return 0;
}
