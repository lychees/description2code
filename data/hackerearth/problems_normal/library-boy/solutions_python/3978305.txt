n = int(raw_input(),10)
books = []
for i in range(0,n):
    s = raw_input()
    books.append(s)

shel = {}
for i in range(97,123):
    shel[chr(i)] = 0

del i
for i in books:
    c = i[0]
    shel[c] += 1

count = 0
del i
for i in range(97,123):
    k = shel[chr(i)]
    if(k!=0):
        if(k < 10):
            count += 1
        elif(k % 10 == 0):
            count += k / 10
        else:
            count += k / 10 + 1
print count
