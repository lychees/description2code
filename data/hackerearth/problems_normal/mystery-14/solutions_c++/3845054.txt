#include <iostream>
using namespace std;

int A[100007];

int main()
{
    int t; cin >> t;
    while (t--) {
    	int n; cin >> n;
    	for (int i = 0; i < n;++i) {
    		cin >> A[i];
    	}
    	int s = 0;
    	for (int i = 0; i < n;++i) {
    		if (A[i] & 1) {
    			s += A[i];
    		}
    	}
    	cout << s << endl;
    }
    return 0;
}
