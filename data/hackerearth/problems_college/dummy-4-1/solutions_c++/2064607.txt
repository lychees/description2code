#include<iostream>
#include<bits/stdc++.h>
using namespace std;
int a[105][105],v[105][105];
unsigned long long c=0;
void mysolve(int i,int j,int n)
{
	if(i==n && j==n){ c++; return;}

	v[i][j]=1;

	if(j+1<=n && !v[i][j+1] && !a[i][j+1])
		mysolve(i,j+1,n);
	if(i+1<=n && !v[i+1][j] && !a[i+1][j] )
		mysolve(i+1,j,n);
	if(i-1>=1 && !v[i-1][j] && !a[i-1][j])
		mysolve(i-1,j,n);
	if(j-1>=1 && !v[i][j-1] && !a[i][j-1] )
		mysolve(i,j-1,n);
	v[i][j]=0;

}

int main()
{
//	freopen("grid.in","r",stdin);
	int n;
	cin>>n;

	int v[n+1][n+1];

	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=n;j++)
		{
			cin>>a[i][j]; v[i][j]=0;
		}
	}

	mysolve(1,1,n);
	cout<<c<<endl;
	return 0;
}
