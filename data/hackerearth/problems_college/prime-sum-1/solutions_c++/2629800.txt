#include <bits/stdc++.h>
using namespace std;

const int N = 1e7;
int dp[N];

int solve(int y)
{
	if(y < 0) return N;
	if(y == 0) return 0;
	
	if(dp[y] != -1) return dp[y];
	dp[y] = min(solve(y-7), min(solve(y-5), min(solve(y-3), solve(y-2)))) + 1;
	return dp[y];
}
int main()
{
	int t;
	cin >> t;
	memset(dp, -1, sizeof dp);
	dp[2] = dp[3] = dp[5] = dp[7] = 1;
	while(t--) {
		int y;
		cin >> y;
		int x = solve(y);
		if(x < N) cout <<  x << endl;
		else cout << -1 << endl;
	}
}
