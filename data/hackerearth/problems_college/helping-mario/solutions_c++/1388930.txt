#include <iostream>
#include <algorithm>
#include <cstring>
#include <unordered_map>
#include <map>
using namespace std;
typedef pair<int,int> PII;
typedef long long LL;
//map<PII,LL> cache;
int N;
int L,R;
int A[51];
//unordered_map<int, long long> mp;
map<int, long long> mp;
 
int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    	scanf("%d",&N);
    	for(int i=0;i<N;i++) scanf("%d",&A[i]);
    	scanf("%d%d",&L,&R);
    	//memset(vis,0,sizeof(vis[[0][0]])*51*100001);
    	mp.clear();
    	for(int i=0;i<N;i++){
    		for(auto it=mp.begin();it!=mp.end();it++){
    			mp[__gcd(A[i],it->first)]+=(LL)(it->second);
    		}
    		mp[A[i]]++;
    	}
    	LL res = 0;
		for(int i=L;i<=R;i++) res += mp[i];
		printf("%lld\n",res);
		
    }
    return 0;
}
