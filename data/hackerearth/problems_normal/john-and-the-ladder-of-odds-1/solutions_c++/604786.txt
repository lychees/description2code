#include <stdio.h>
#define mod 1000000009
void power(long long int fib[2][2],long long int);
void multiply(long long int fib[2][2],long long int m[2][2]);
long long int fibo(long long int n)
{
	long long int fib[2][2] = {{1,1},{1,0}};
	if(n == 0)
	return 0;
	power(fib,n - 1);
	return fib[0][0];
}
void power(long long int fib[2][2],long long int n)
{
	if( n == 0 || n == 1)
      return;
      long long int m[2][2] = {{1,1},{1,0}};
      power(fib, n/2);
      multiply(fib, fib);

  if (n%2 != 0)
     multiply(fib, m);
}
void multiply(long long int fib[2][2],long long int m[2][2])
{
	long long int x,y,z,w;
	x = fib[0][0]*m[0][0] + fib[0][1]*m[1][0];
	y = fib[0][0]*m[0][1] + fib[0][1]*m[1][1];
	z = fib[1][0]*m[0][0] + fib[1][1]*m[1][0];
	w = fib[1][0]*m[0][1] + fib[1][1]*m[1][1];
	fib[0][0] = x%mod;
	fib[0][1] = y%mod;
	fib[1][0] = z%mod;
	fib[1][1] = w%mod;
}
int main()
{
	long long int t;
	long long int i,n,m,ans1,ans2,ans;
	scanf("%lld",&t);
	while(t--){
       scanf("%lld",&n);
	   //sum = fibo(n);
	   ans1 = fibo(n);
	   ans2 = fibo(n + 1);
	   printf("%lld\n",((ans1%mod) *(ans2%mod)%mod));
	}
	return 0;
}
