#include <iostream>
#include<queue>
using namespace std;
#include<set>

int main()
{
    priority_queue<int> q;
    int m,n,answer,a,x;
    cin>>m>>n;
    int i;
    for(i=0;i<m;i++)
    {
    	cin>>a;
    	q.push(a);
    }
    for(i=0;i<n;i++)
    {
    	x=q.top();
    	answer+=x;
    	q.pop();
    	q.push(x-1);
    }
    cout<<answer<<"\n";

    return 0;
}
