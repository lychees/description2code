#include <iostream>
using namespace std;
#define ll long long
int main()
{
	ios::sync_with_stdio(false);
   ll n,x;
   cin>>n>>x;
   ll a[n+1],ans=0;
   bool flag=false;
   
   for(ll i=0;i<n;++i)
   {
   	cin>>a[i];
  
   }
   
   for(ll i=0;i<n;++i)
   {
   	if(a[i]<=x) ans++;
  
   	else if(a[i]>x &&flag==false &&n!=0) flag=true;
   	
   	else if(a[i]>x &&flag==true )break;
   }
   cout<<ans<<endl;
    return 0;
}
