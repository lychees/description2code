set<int> getDivisors(int n) {
  set<int> ans;
  for (int i = 2; i * i <= n; i++) {
    if (n % i == 0) {
      while (n % i == 0) {
        n /= i;
      }
      ans.insert(i);
    }
  }
  if (n > 1) ans.insert(n);
  return ans;
}

int a[27][27];

int main() {
  fr(x, 1, 24) {
    fr(y, 1, 24) {
      set<int> st1 = getDivisors((1<<x) - 1);
      set<int> st2 = getDivisors((1<<y) - 1);
      int ans = 0;
      foreach(it, st1) if (st2.count(*it)) ans++;
      a[x][y] = ans;
    }
  }
  int T;
  scanf("%d", &T);
  while (T--) {
    int x, y;
    scanf("%d %d", &x, &y);
    printf("%d\n", a[x][y]);
  }

  return 0;
}
