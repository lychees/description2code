#define FOR(i,a,b) for (int i=a;i<b;i++)
#define FORE(i,a,b) for (int i=a;i<=b;i++)
#define FORD(i,a,b) for (int i=a;i>=b; i--)
#define ri(n)({\
    int neg=0;\
    n=0;\
    char ch;\
    for(ch=getchar(); ch<'0' || ch>'9'; ch=getchar()) if (ch=='-') neg=1-neg;\
    n=ch-48;\
    for(ch=getchar(); ch>='0' && ch<='9'; ch=getchar()) n=(n<<3)+(n<<1)+ch-48;\
    if (neg) n=-n;\
})

typedef long long ll;

bool cmp(int i, int j){
    return i>j;
}

int n,k,a[100000+10],b[100000+10],vt[100000+10],d;

int main(){
 //   freopen("sandy-permutes.inp", "r", stdin);
 //   freopen("sandy-permutes.out", "w", stdout);
    ri(n); ri(k);
    FORE(i,1,n){
        ri(a[i]);
        b[i]=a[i];
        vt[a[i]]=i;
    }
    sort(b+1,b+n+1,cmp);
    FORE(i,1,n)
    if (k>0)
    if (a[i]!=b[i]){
        d=vt[ b[i] ];
        vt[ a[i] ]=d;
        a[d]=a[i];
        a[i]=b[i];
        k--;
    }
    FORE(i,1,n) cout << a[i] << " ";
}
