n = int(raw_input())

list1 = sorted(map(int, raw_input().split()))

m = int(raw_input())

list2 = sorted(map(int, raw_input().split()))

if n < m:
	list1, list2 = list2, list1

value = []

lt1 = set(list1)

while lt1:
	x = lt1.pop()
	if list1.count(x) != list2.count(x):
		value.append(x)

value.sort()

value = map(str, value)

print ' '.join(value)
