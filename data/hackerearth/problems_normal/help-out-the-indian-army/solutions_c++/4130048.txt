#include <iostream>
#include <vector>
#include <algorithm>
//#include <utility>
#include <cstdlib>
using namespace std;
#define ll long long int
ll s,e;
int n;
int main()
{
	cin>>n>>s>>e;
	ll x,y,maxm,minm;
	vector<pair<ll,ll>> p;
	vector<pair<ll,ll>>::iterator it,it1;
	for(int i=0;i<n;i++)
	{
		ll a, b;
		cin>>a>>b;
		minm=a-b;
		maxm=a+b;
		if(maxm<=s || minm>=e)
		continue;
		minm=max(s,minm);
		maxm=min(e,maxm);
		p.push_back(pair<ll,ll>(minm,maxm));
	}
	
	ll dif=0;
	if(p.size()==0)
	{
		cout<<e-s;
		return 0;
	}
	sort(p.begin(),p.end());
	it=p.begin();
	minm=it->first;
	maxm=it->second;
	dif=max(0LL,minm-s);
	it=p.end();
	for(it1=p.begin();it1!=it;it1++)
	{
		x=it1->first;
		y=it1->second;
		
		if(it1==p.begin())
		{
			continue;
		}
		if(y>maxm)
		{
			dif=dif+max(0LL,(x-maxm));
			maxm=y;
		}
		//cout<<dif<<endl;
	}
	dif=dif+max(0LL,(e-maxm));
	cout<<dif;
	return 0;
}
