#include<bits/stdc++.h>
using namespace std;
 
long long int arr[1000000]={0};
long long int n,q;
int chck(long long int k)
{
	long long int m=k,n=0;
	while(m>0)
	{
		n*=10;
		n+=m%10;
		m/=10;
	}
	if(n%10==k%10)
	return 1;
	else
	return 0;
}
void update(int index,int diff)
{
    index++;
    while(index<=n)
    {
        arr[index]+=diff;
        index+=index&(-index);
    }
}
int get(int index)
{
    index++;
    long long int count=0;
    while(index>0)
    {
        count+=arr[index];
        index-=index&(-index);
    }
    return count;
}
void create(long long int *a)
{
    for(int i=0;i<n;i++)
    {
    	int flg=chck(a[i]),diff=0;
    	if(flg==1)
    	diff++;
    	if(a[i]<0)
    	diff=0;
        update(i,diff);
    }
}
long long int a[1000000];

int main()
{
    cin>>n>>q;
    
    for(int i=0;i<n;i++)
    cin>>a[i];
    create(a);
    while(q--)
    {
        long long int ar,b,c;
        cin>>ar>>b>>c;
        if(ar==1)
        {
            cout<<get(c-1)-get(b-2)<<"\n";
        }
        else
        {
        	int f2=chck(a[b-1]),f3=chck(c);
        	int diff=0;
        	if(c<0)
        	f3=0;
        	if(a[b-1]<0)
        	f2=0;
        	if(f2==1&&f3==0)
        	diff--;
        	if(f2==0&&f3==1)
        	diff++;
        
        //	cout<<diff<<f2<<f3<<a[b-1]<<"hhh\n";
    update(b-1,diff);
    a[b-1]=c;
        }
    }
}
