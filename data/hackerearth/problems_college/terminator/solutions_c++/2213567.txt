#include<iostream>
using namespace std;

int main() {
	int t;
	cin>>t;
	
	while(t--) {
		char s[101], r[101];
		int i = -1, j;
		cin>>s;
		for(j=0; s[j]; j++) {
			if(i == -1) {
				r[++i] = s[j];
			} else if(s[j] == 'x') {
				if(i == -1) {
					r[++i] = s[j];
				} else if((r[i] == 'x') || (r[i] == '*')) {
					r[++i] = s[j];
				} else if(r[i] == 'o') {
					while(1) {
						if(r[i] == 'o') {
							r[i] = '\0';
							i--;
						}
						if((r[i] == 'x') || (r[i] == '*') ||
						   (i == -1)) {
						   	r[++i] = s[j];
						   	break;
						}
					}
				}
			} else if(s[j] == '*') {
				r[++i] = s[j];
			} else if(s[j] == 'o') {
				if(i == -1) {
					r[++i] = s[j];
				} else if(r[i] == 'x') {
					continue;
				} else {
					r[++i] = s[j];
				}
			}
		}
		r[++i] = '\0';
		for(int k=0; r[k]; k++) cout<<r[k];
		cout<<endl;
	}
}
