#include <iostream>
using namespace std;

int main()
{
  int n;
  int c[n+1],k[n+1];
  int a,b,d,temp,min=1,l=0;

  {
 	cin>>a>>b;
  	d=a+b;
  	c[i]=d;
  	k[i]=i;
  	}
  	for(int i=1;i<=n;i++)
  	{
  		for(int j=1;j<n;j++)
  		{
