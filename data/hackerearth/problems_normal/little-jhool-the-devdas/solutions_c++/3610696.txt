#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
long long int sum=0;
int t,i,*a;
cin>>t;
a = new int [t];
for(i=0;i<t;i++)
cin>>a[i];
sort(a,a+t);
for(i=0;i<t;i++)
sum+=a[i];
if(sum%2==0)
{
	for(i=0;i<t;i++)
	{
		if((sum-a[i])%2!=0)
		{
			sum-=a[i];
			break;
		}
	}
	if(sum%2==0)
	cout<<":(";
	else
	cout<<sum;
}
else
cout<<sum;
		
return 0;
}
