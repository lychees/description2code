#include <bits/stdc++.h>
using namespace std;

int comp(const pair<int,char> a,const pair<int,char> b)
{
	if(a.first<b.first)
	{
		return 1;
	}
	else if(a.first>b.first)
	{
		return 0;
	}
	else if(a.second>b.second)
	{
		return 1;
	}
	else
	{
		return 0;
	}
}

int main()
{
    int test=0;
    scanf("%d",&test);
    while(test>0)
    {
    	int num=0,i=0,k=0;
    	cin>>num;
    	vector<int> guest,duration;
    	for(i=0;i<num;i++)
    	{
    		cin>>k;
    		guest.push_back(k);
    	}
    	for(i=0;i<num;i++)
    	{
    		cin>>k;
    		duration.push_back(k);
    	}
    	vector<pair<int,char> > schedule;
    	for(i=0;i<num;i++)
    	{
    		schedule.push_back(make_pair(guest[i],'A'));
    		schedule.push_back(make_pair((guest[i]+duration[i]),'D'));
    	}
    	sort(schedule.begin(),schedule.end(),comp);
    	//cout<<schedule.size()<<"\n";
    	int ans=INT_MIN,cnt=0;
    	/*for(i=0;i<num;i++)
    	{
    		cout<<schedule[i].first<<" "<<schedule[i].second<<"\n";
    	}*/
    	for(i=0;i<schedule.size();i++)
    	{
    		if(schedule[i].second=='A')
    		{
    			cnt++;
    		}
    		else
    		{
    			cnt--;
    		}
    		if(cnt>ans)
    		{
    			ans=cnt;
    		}
    	}
    	cout<<ans<<"\n";
    	test--;
    }
    return 0;
}
