#include <bits/stdc++.h>
using namespace std;
int arr[105];
bool ispos[105][100005];
int main()
{
    int n,t,i,z,j;
    scanf("%d",&n);
    for(i=1;i<=n;i++)
        scanf("%d",&arr[i]);
    for(i=0;i<=n;i++)
        ispos[i][0]=1;
    for(i=1;i<100005;i++)
        ispos[0][i]=0;
    for(i=1;i<=n;i++)
    {
        for(j=1;j<100005;j++)
        {
           if(arr[i]>j)
              ispos[i][j]=ispos[i-1][j];
            else
              ispos[i][j]=ispos[i-1][j] || ispos[i-1][j-arr[i]];
        }
    }
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&z);
        if(ispos[n][z])
            printf("Yes\n");
        else
            printf("No\n");
    }
    return 0;
}
