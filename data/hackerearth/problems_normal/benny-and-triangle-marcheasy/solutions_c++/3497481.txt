	if (b == 0) return a;
	return(a % b == 0 ? b : gcd(b, a % b));

{
    bool operator()(const P &p1, const P &p2)
    {
    	if (p1.first > p2.first) return true;
    	if (p1.first == p2.first) return p1.second < p2.second;
        return false;
    }
};
