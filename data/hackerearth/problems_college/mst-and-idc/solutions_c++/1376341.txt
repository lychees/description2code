#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t,n,*x,tot;
    cin >> t;
    while(t--)
    {
    	cin >> n;
    	tot=0;
    	x=new int[n];
    	for(int i=0;i<n;i++)
    		cin >> x[i];
    	sort(x,x+n);
    	for(int i=0;i<n-1;i++)
    	{
    		tot+=x[i+1]-x[i];
    	}
    	cout << tot*2 << endl;
    }
    return 0;
}
