t = input()
while (t > 0):
    n = input()
    a = []
    i = 0
    line = raw_input().split(' ')
    while (i < n):
        a.append(int(line[i]))
        i = i+1
    a.sort(reverse=True)
    x = n-1
    temp = 0
    for i in range(0, n):
        temp += x*a[i]
        x -= 2
    print(temp*a[0]) % 1000000007
    t = t-1
