n, m = map(int, raw_input().strip().split(' '))
rows = []
cols = [0 for _ in range(m)]
for line in range(n):
	nos = map(int, raw_input().strip().split(' '))
	new_cols = [i + j for i, j in zip(cols, nos)]
	cols = new_cols
	
	rows.append(sum(nos))
max_xor = None
for i in rows:
	for j in cols:
		x = i ^ j
		if max_xor is None:
			max_xor = x
		if x > max_xor:
			max_xor = x
print(max_xor)
