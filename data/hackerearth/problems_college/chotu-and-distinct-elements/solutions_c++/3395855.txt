#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int t,n,i,j,k,x,y;
	cin>>t;
	for(i=0;i<t;i++)
	{
		cin>>n;
		long long int a[n],b[n];
		for(j=0;j<n;j++)
		cin>>a[j];
		sort(a,a+n);
		k=0;
		for(j=0;j<(n);j++)
		{
			if(a[j]!=a[j+1])
			{
				b[k]=a[j];
				k+=1;
			}
		}
		x=0;y=0;
		for(j=0;j<k;j++)
		{
			if(j%2==0)
			{
				cout<<b[k-1-x]<<" ";
				x+=1;
			}
			else
			{
				cout<<b[y]<<" ";
				y+=1;
			}
			
		}
		cout<<"\n";
	}
	return 0;
}
