#include <bits/stdc++.h>
int a[1000000],n;
using namespace std;
void mi(int j)
{
    if(a[j]==0 && j!=n)
    {
        a[j]=9;
        mi(j+1);
    }
    else
        a[j]=a[j]-1;
}
int main()
{

    int t,k,j,i;
    scanf("%d",&t);
    while(t--)
    {
        for(i=0;i<1000000;i++)
        a[i]=0;
        scanf("%d%d",&n,&k);
        a[n]=1;
        j=k;
        mi(j);
        j=n-k;
        if(a[n]==0)
            n=n-1;
        mi(j);
        a[0]+=1;
        j=n;
        while(j>-1)
        {
        printf("%d",a[j]);
        j--;
        }
        printf("\n");
    }
    return 0;
}
