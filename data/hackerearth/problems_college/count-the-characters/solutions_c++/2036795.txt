inline void readll(LL &ret)
{
    int cc = getc(stdin);
    for (;cc < '0' || cc > '9';)
        cc = getc(stdin);
    ret = 0;
    for (;cc >= '0' && cc <= '9';)
    {
        ret = ret *10 + cc - '0';
        cc = getc(stdin);
    }
}
inline LL readsi(int s[])
{
    LL i = 0;
    int c = getc(stdin);
    while(c < '0' || c > '9')
        c = getc(stdin);
    do
    {
        s[i++] = c - '0';
        c = getc(stdin);
    }while(c >= '0' && c <= '9');
    return i;
}

    forall(j, 0, t) {

        sort(a.begin(), a.end());
        LL length = a.length();
        forall(i, 0, length) {
            if(!i) {
                ++s[a[i] - 'a'];
            }
            else {
                if(a[i] != a[i - 1]) {
                    ++s[a[i] - 'a'];
                }
            }
        }
    }

        if(s[i] == t)
            ++count;
    }
