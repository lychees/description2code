# include <stdio.h>
 
void swap (char *x, char *y)
{
    char temp;
    temp = *x;
    *x = *y;
    *y = temp;
}
  

void permute(char *a, int i, int n) 
{
   int j; 
   if (i == n)
     printf("%s\n", a);
   else
   {
        for (j = i; j <= n; j++)
       {
          swap((a+i), (a+j));
          permute(a, i+1, n);
          swap((a+i), (a+j));
       }
   }
} 

int main()
{
	int n,a1,b;
   char a[200];
   gets(a);
   scanf("%d",&n);
   while(n--)
   {
   scanf("%d %d",&a1,&b);
   permute(a, a1, b);
   printf("\n");
   }
   return 0;
}
