#include <iostream>
#include<algorithm>
#include<cmath>
typedef long long ll;
using namespace std;
ll l,h,m,a[1000000],b,res,coun,n,k,ans;
int bs(ll m)
{
  
    res=1;
    coun=a[0];
     for(int i=0;i<n;i++)
      {
      	if((ceil(a[i]-coun)/2)>m)
         {res++;
          coun=a[i];
         
          
         }
      }
    return (res<=k)?1:0;
}
int main()
{ cin>>n>>k;
   for(int i=0;i<n;i++)
   {cin>>a[i];
  
   }
    sort(a,a+n);
     l=0;
     h=1e8;
     while(l<=h)
     { m=(l+h)/2;
        if(bs(m))
        {h=m-1;
        ans=m;
        }
        else
       { l=m+1;
        
       }
     }
     cout<<ans;
    
    return 0;
}
