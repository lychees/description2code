#include <iostream>
using namespace std;

int main()
{
    int N;
    cin>>N;
    int ar[N];
    for(int i=0;i<N;i++){
    	cin>>ar[i];
    }
    int adj[N][N];
    //int sum[N];
    //for(int i=0;i<N;i++){
    //	sum[i]=0;
    //}
    for(int i=0;i<N;i++){
    	for(int j=0;j<N;j++){
    		if(i!=j){
    			if(ar[i]==ar[j]){
    				adj[i][j]=ar[i];
    			//	sum[i]++;
    			}
    			else{
    				adj[i][j]=0;
    			}
    		}
    		else{
    			adj[i][j]=0;
    		}
    	}
    	
    }
    int count=0;
    for(int i=0;i<N;i++){
    	for(int j=i+1;j<N;j++){
    		if(adj[i][j]>0){
    			for(int k=i<j?(i+1):(j+1);k<(i>j?i:j);k++){
    				for(int l=0;l<(i<j?i:j);l++){
    					if(adj[k][l]>0 && adj[k][l]!=adj[i][j])(count++)%(1000000007);
    				}
    				for(int l=(i>j?(i+1):(j+1));l<N;l++){
    					if(adj[k][l]>0 && adj[k][l]!=adj[i][j]){
    						(count++)%(1000000007);
    					}
    				}
    			}
    		}
    	}
    }
    cout<<(count/2)%(1000000007)<<endl;
    return 0;
}
