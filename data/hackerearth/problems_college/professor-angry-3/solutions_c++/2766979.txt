#include<iostream>


using namespace std;
main()
{

    int a[1000],i,N,K,t,cnt=0;
    cin>>t;
    for(i=0;i<t;i++)
    {
        cin>>N;
        cin>>K;
        for(int j=0;j<N;j++)
        {
            cin>>a[j];
            if(a[j]<=0)
            {
                cnt++;
            }
        }
        if(cnt>=K)
        {
            cout<<"NO"<<endl;
        }
        else{
            cout<<"YES"<<endl;
        }
        cnt=0;
    }
}
