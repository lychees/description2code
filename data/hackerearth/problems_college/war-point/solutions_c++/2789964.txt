#include<iostream>
#include<stdio.h>
#include<bits/stdc++.h>
using namespace std;
main()
{
	long long t;
	scanf("%lld",&t);
	while(t--)
	{
		long long n;
		scanf("%lld",&n);
		long long a[n];
		for(int i=0;i<n;i++)
			scanf("%lld",&a[i]);
		long long sum=a[0],ls[n],rs[n];
		ls[0]=0;
		for(int i=1;i<n;i++)
		{
			ls[i]=sum;
			sum+=a[i];
		}
		sum=a[n-1];rs[n-1]=0;
		for(int i=n-2;i>=0;i--)
		{
			rs[i]=sum;
			sum+=a[i];
		}
		long long res=INT_MAX,ans=INT_MAX;
		for(int i=0;i<n;i++)
		{
			ans=min(abs((ls[i]+a[i])-rs[i]),abs(ls[i]-(a[i]+rs[i])));
			res=min(res,ans);
		}
		printf("%lld\n",res);
	}
}
