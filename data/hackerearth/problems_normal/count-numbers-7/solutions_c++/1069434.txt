#include<iostream>
#include<bitset>
using namespace std;
long long int gcd(long long int a,long long int b){
if(b==0)return a;
return gcd(b,a%b);
}
int main(){
	int t,k,a,b;
	cin>>k>>t;
	long long pr[k+1];
	pr[0]=0;
	for(int i=1;i<=k;i++)cin>>pr[i];
	int lim=(1<<k);
	while(t--){
		cin>>a>>b;
		int ans=0;
		for(int j=1;j<lim;j++){
			bitset<10> val(j);
			int c=val.count();
			long long v=1;
			for(int i=1;i<=k;i++){
                if(val[i-1]){
                //	cout<<j<<" "<<i<<" "<<val[i]<<" "<<pr[i]<<endl;
                    v=(v*pr[i])/gcd(v,pr[i]);
                }
                
                if(b/v==0)break;
			}
			//cout<<v<<endl;
			if(c%2){
                ans+=(b/v-(a-1)/v);
			}
			else ans-=(b/v-(a-1)/v);
		}
		cout<<ans<<endl;
	}
	return 0;
}
