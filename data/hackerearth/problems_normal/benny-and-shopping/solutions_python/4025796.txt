t=input()
for x in xrange(t):
    s=raw_input()
    for i in range(len(s)):
        if ord(s[i])==36:
            s=s[i+1:]
            break     
    for i in xrange(len(s)):
        if 65<=ord(s[i])<=122:
            s=s[:i]
            break
    s=s.replace(" ", "")
    for i in xrange(len(s)):
        if s[i]!='0':
            s=s[i:]
            break
            
    print '$'+str(int(s))
