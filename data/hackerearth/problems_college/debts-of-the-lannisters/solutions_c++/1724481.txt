#include<iostream>
using namespace std;
int main()
{
	char real[10],guess[10],temp[10];
	int n,gold,silver;
	cin>>real;
	cin>>n;
	for( int i=0 ; i<n ; i++ )
	{
		gold=0;
		silver=0;
		cin>>guess;
		for( int j=0 ; j<10 ; j++ )
			temp[j]=real[j];
		for( int j=0 ; j<10 ; j++ )
		{
			if( guess[j]==temp[j] )
			{
				gold++;
				temp[j]='0';
				guess[j]='1';
			}	
		}
		for( int j=0 ; j<10 ; j++)
		{ 
			for( int k=0 ; k<10 ; k++ )
				if( guess[j]==temp[k] )
				{
					silver++;
					temp[k]='0';
					break;
				}
		}	
		cout<<gold<<" "<<silver<<"\n";
	}
	return 0;
}
