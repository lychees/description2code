#include <bits/stdc++.h>

using namespace std;
int n;
int k;
long int dp[100005];
long int a[100005];

long int calculate_max(int index)
{
    if (index  >= n)
        return 0;
    if (dp[index] != -1)
        return dp[index];

    long int s;
        s = -INFINITY;
        s = max(s,calculate_max(index+1));
        s = max(s,a[index] + calculate_max(index+k+1));

        dp[index] = s;

    return s;
}

int main()
{
    //int n;
    int tc;
    cin >> tc;

    while(tc--){
    cin >> n >> k;
    //int a[n];

    for (int i = 0;i < n;i++)
        cin >> a[i];

    for (int i = 0;i <= n;i++)
        dp[i] = -1;
    long int ans;
    ans = calculate_max(0);

    cout << ans << endl;
    }

    return 0;
}
