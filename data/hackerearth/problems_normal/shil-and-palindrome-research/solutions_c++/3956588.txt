int i,l,q,choice,alpha[26][100021],x,y,odd;
char a[100021],c;
void update(int pos,int val,int ch)
{
    for(;pos<=l;pos+=pos & (-pos)) alpha[val-'a'][pos]+=ch;
}
int query(int ch,int pos)
{
    int sum=0;
    for(;pos>0;pos-=pos & (-pos)) sum+=alpha[ch][pos];
    ret sum;
}
int main()
{
    sc("%d%d",&l,&q);
    sc(" %s",a+sizeof(char));
    for(i=1;i<=l;i++) update(i,a[i],1);
    whi(q--)
    {
        sc("%d",&choice);
        if(choice==2)
        {
            sc("%d%d",&x,&y);
            odd=0;
            for(i=0;i<26;i++)
            {
                if((query(i,y)-query(i,x-1))%2==1) odd++;
            }
            if(odd>1) pr("no\n");
            else pr("yes\n");
        }
        else
        {
            sc("%d %c",&x,&c);
            update(x,a[x],-1);
            update(x,c,1);
            a[x]=c;
        }
    }
    ret 0;
}
