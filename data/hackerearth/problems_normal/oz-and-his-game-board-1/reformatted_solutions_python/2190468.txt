
# Read input from stdin and provide input before running code

T = int(raw_input())
while (T > 0):
    N, M, A, B, C = [int(i) for i in raw_input().split(" ")]

    least = 0
    diff = 0
    w = 0
    if (N < M):
        least = N
        diff = M-N
        w = B
    else:
        least = M
        diff = N-M
        w = A

    print max((N*A)+(M*B), (least*2*C)+(diff*w))
    T -= 1
