#include<iostream>
#include<vector>
#include<climits>
#include<algorithm>
using namespace std;
bool check(int L, vector<int> &array, int k)
{
	int ct = 0;
	for(int i = 1;i < array.size();)
	{
		if( L < array[i] - array[i-1])
			return false;
		if(ct > k) return false;
		int start = i-1;
		int end = i;
		while(end < array.size() && L >= array[end] -  array[start])
			end++;
		
		i = end;
		
		ct++;
	}
	
	if(ct <= k) return true;
	else return false;
	
}
int main()
{
	int T;
	cin >> T;
	for(int tt = 0; tt < T;tt++)
	{
		int N, K;
		cin >> N >> K;
		vector<int> array(N);
		for(int i = 0; i < N;i++)
		{
			cin >> array[i];
		}
		sort(array.begin(),array.end());
		int low = 1;
		int high = 1e9;
		//cout << "status of " << check(5,array,2) << endl;	
	//	int mid = low + (high-low)/2;
		
		int result ;
		while(low <= high){
			int mid = low + (high -low)/2;
		//	cout << "mid is " << mid <<":";
			if(check(mid,array,K))
			{
			//	cout <<"came in if " << endl;
				high = mid-1;
				result = mid;
			}
			else{
				low = mid+1;
			//	cout << "came in else " << endl;
			}	
		}
		
		cout << result << endl;
	}



    return 0;
}
