#include<iostream>
#include<string>

using namespace std;

int main(){
string s;
int t;
int flag = 0;
cin>>t;
while(t--){
	cin>>s;
	for( int i= 0;i < s.length(); i++){
		for( int j=i+1; j< s.length();j++){
			if (s[i]==s[j]) {
			flag = 1;
			break;
}
}
}

if(flag==1) cout<<"Yes"<<endl;
else cout<<"No"<<endl;
flag=0;
}

return 0;
}
