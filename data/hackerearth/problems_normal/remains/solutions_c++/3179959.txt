#include <bits/stdc++.h>

typedef long long unsigned ll;

using namespace std;

ll my_sub(ll a,ll b)
{
	if(a>b)
		return a-b;
	else
		return b-a;
}

void solve()
{
	ll x,y,n;
	cin>>x>>y>>n;
	ll ans=x+y;
	ll current,prev,temp;
	current = y;prev = x;
	ll i;
	for (i = 0; i < n-2; ++i)
	{
		temp = current;
		current = my_sub(current,prev);
		prev = temp;
		ans += current;
		if(current == 0)
			break;
	}
	if(i != n-2)
	{
		temp = n - (i + 3);
		if(temp % 3 == 2)
			cout << ((temp/3) + 1)*(prev+prev) + ans <<endl;
		else if( temp % 3 == 1)
			cout << ((temp/3))*(prev+prev) + prev + ans <<endl;
		else
			cout << (temp/3)*(prev+prev) + ans <<endl;
	}
	else
		cout<<ans<<endl;
}

int main()
{
	int t;
	cin>>t;
	while(t--)
		solve();
	return 0;
}
