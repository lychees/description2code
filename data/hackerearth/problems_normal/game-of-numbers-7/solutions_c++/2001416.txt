#include <iostream>
using namespace std;
#define ma 1000000
bool primes[1000001], flag[1000001];
int main()
{
    long long int i, j, c, start;
    primes[0]=1;primes[1]=1;
    for(i=2;i*i<=ma;i++)
    if(primes[i]==0)
    {
    	for(j=i*i;j<=ma;j+=i)
    	primes[j]=1;
    }
    long long int p[100000];
    int k=0, t, x, y;
    for(i=2;i<=ma;i++)
    {
    	if(primes[i]==0)
    	p[k++]=i*i;
    }
    scanf("%d", &t);
    while(t--)
    {
    	scanf("%d%d", &x, &y);
    	for(i=0;i<y-x+1;i++)
    	flag[i]=1;
    	c=0;
    	for(i=0;p[i]<=y;i++)
    	{
    		start=x+(p[i]-x%p[i])%p[i];
    		for(j=start;j<=y;j+=p[i])
    		flag[j-x]=0;
    	}
    	for(i=0;i<y-x+1;i++)
    	if(flag[i]==1)
    	c++;
    	printf("%lld\n", c);
    }
    return 0;
}
