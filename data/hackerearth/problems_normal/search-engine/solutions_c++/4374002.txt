#include <bits/stdc++.h>
using namespace std;

#define MAX 1000005

char s[MAX];

struct trie{
	int mx;
	trie *nxt[26];
};

trie *getNode(){
	trie* temp = (trie *)malloc(sizeof(trie));
	
	temp->mx = -1;
	for(int i=0 ; i<26 ; i++)
		temp->nxt[i] = NULL;
		
	return temp;
}

void insert(char s[], trie *root, int weight){
	trie *current = root;
	
	for(int i=0 ; s[i] ; i++){
		int index = s[i] - 'a';
		if(current->nxt[index] == NULL)
			current->nxt[index] = getNode();
		current = current->nxt[index];
		current->mx = max(current->mx, weight);
	}
}

int search(char s[], trie *root){
		trie *current = root;
	
	for(int i=0 ; s[i] ; i++){
		int index = s[i] - 'a';
		if(current->nxt[index] == NULL)
			return -1;
		current = current->nxt[index];
	}
	
	return current->mx;
}

int main()
{
	int n, q, w;
	scanf("%d %d", &n, &q);
	
	trie *root = getNode();
	
	for(int i=0 ; i<n ; i++){
		getchar();
		scanf("%s %d", s, &w);
	
		insert(s, root, w);
	}
	
	for(int i=0 ; i<q ; i++){
		getchar();
		scanf("%s", s);
		
		printf("%d\n", search(s, root));
	}
	
    return 0;
}
