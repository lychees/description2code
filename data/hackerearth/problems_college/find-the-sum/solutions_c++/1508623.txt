#include <cstdio>
#include <cmath>
#include <algorithm>
using namespace std;

#define ll long long

int main(){
	int t;
	ll l, r, sl, sr, res, x, y;
	
	scanf("%d", &t);
	while (t--){
		scanf("%lld %lld", &l, &r);
		sl = (ll)sqrt(2*l);
		sr = (ll)sqrt(2*r);
		if (sl * (sl + 1) / 2 >= l)
			sl--;
		if (sr * (sr + 1) / 2 >= r)
			sr--;
		x = (sl + 1) >> 1;
		y = (sr + 1) >> 1;
		res = y * (2 + (y - 1) * 2) / 2 - x * (2 + (x - 1) * 2) / 2;
		if (sl * (sl + 1) / 2 < l && (l - 1 - sl * (sl + 1) / 2) % 2 == 1)
			res -= sl + 1;
		if (sr * (sr + 1) / 2 < r && (r - sr * (sr + 1) / 2) % 2 == 1)
			res += sr + 1;
		printf("%lld\n", res);
			
	}
	return 0;
}
