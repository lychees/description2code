#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    int temp=0;
    while(t--)
    {
        string s;
        cin>>s;

        int i,temp2=0;;
        for(i=0;i<s.length();i++)
        {
            temp+=s[i]-'0';
        }
        while(temp/10!=0)
        {
            temp2=0;
            while(temp!=0)
            {
                temp2+=temp%10;
                temp/=10;
            }
            temp=temp2;
        }
    }
    cout<<temp;
}
