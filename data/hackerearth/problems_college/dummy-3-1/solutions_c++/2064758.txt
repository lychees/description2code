#include <iostream>
using namespace std;

int main()
{
	long t,n,a[100000],f;
	cin>>t;
	while(t--)
	{
		cin>>n; f=0;
		for(int i=0;i<n;i++)
		{
			cin>>a[i];
		}
		for(int i=0;i<n;i++)
		{
			if(i+1!=a[a[i]-1])
			{
    			f=1;
    			break;
    		}
		} 
		if(f==0) 
		cout<<"inverse"<<endl;
		else 
		cout<<"not inverse"<<endl;
	}
    return 0;
}
