#include <bits/stdc++.h>
using namespace std;
#define ll long long int
ll a[1001][1001];
int main()
{
    ll t = 0, n, k = 0, i, j;
    cin>>n;
    for(i =0 ; i < n; i++)
    	for(j = 0 ; j < n; j++)
    		cin>>a[i][j];
    for(i =0 ; i < n; i++)
    	k += a[i][i];
    for(i = 0 ; i < n; i++)
    	t += a[i][n-1-i];
    cout<<abs(t- k)<<endl;
    return 0;
}
