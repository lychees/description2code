#include <iostream>
#include <stdio.h>

using namespace std;

int main() {
	// your code goes here
    int n;
    scanf("%d",&n);
    
    int (*T)[2] = (int (*)[2])malloc(n*sizeof(*T));
    
    int b, c;
    
    int state;
    for (state=0; state<n; ++state)
    {
        b = state<<1;
        T[state][0] = (b < n)? b: b-n;
        c = (state<<1) + 1;
        T[state][1] = (c < n)? c: c-n;
    }
    
    for (state=0; state<n; ++state)
    {
    	printf("%d %d %d\n",state,T[state][0],T[state][1]);
    }
	return 0;
}
