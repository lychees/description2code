#include<bits/stdc++.h>
#include<string>
using namespace std;
#define ll long long int
#define sc1ll(x) scanf("%lld",&x)
#define sc2ll(x,y) scanf("%lld%lld",&x,&y)
#define sc3ll(x,y,z) scanf("%lld%lld%lld",&x,&y,&z)
struct node
{
    ll profit,val;
};
struct node2
{
    ll profit,waste;
};
vector<node> arr(101);
vector<vector<node2> > mat(10001,vector<node2>(101));
void knapsack(ll num,ll capacity)
{
    ll maxx=0,waste=0;
    for(ll i=0;i<=capacity;i++)
    {
        for(ll j=0;j<=num;j++)
        {
            if(i==0||j==0)
            {
                mat[i][j].profit=0;
                mat[i][j].waste=0;
                continue;
            }
            mat[i][j].profit=mat[i][j-1].profit;
            mat[i][j].waste=mat[i][j-1].waste;
            if(i>=arr[j].val)
            {
                if( (arr[j].profit+mat[i-arr[j].val][j-1].profit) > mat[i][j].profit )
                {
                    mat[i][j].profit=arr[j].profit+mat[i-arr[j].val][j-1].profit;
                    mat[i][j].waste=arr[j].val+mat[i-arr[j].val][j-1].waste;
                }
            }
            if((mat[i][j].profit-mat[i][j].waste)>maxx)
            {
                maxx=mat[i][j].profit-mat[i][j].waste;
                waste=mat[i][j].waste;
            }
        }
    }
    ll ans=capacity+maxx;
    cout<<waste<<" "<<ans<<endl;
}
int main()
{
    ll test,i,j,k,num,a,b,capacity;
    sc1ll(test);
    while(test--)
    {
        sc2ll(num,capacity);
        for(i=1;i<=num;i++)
            sc2ll(arr[i].profit,arr[i].val);
        knapsack(num,capacity);
    }
}
