#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll t,n,i,j,k,l,c,x,y,m;
int main()
{
	cin>>t;
	while(t--)
	{
		vector <bool> status(1000000,false);
		cin>>n;
		c=0;m=0;
		for(i=0;i<n;i++)
		{
			cin>>x>>y;
			if(y==0)
			{
				m=max(m,c);
				c=0;
			}
			else if(status[x]==false&&y==1)
			{
				c++;
				status[x]=true;
				m=max(m,c);
			}
			
		}
		cout<<m<<endl;
	}
}
