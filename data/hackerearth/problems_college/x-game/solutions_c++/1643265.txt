#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
   int t,n;
   unsigned long long int OR; 
   cin >> t;
    while(t--)
        {
           OR = 0;
           cin >> n;
           int tmp = n;
        while(tmp --)
            {
              unsigned long long int temp;
              cin >> temp;
            OR = OR | temp;
            }
        
        for(int i = 1; i <= n-1 ; i++)
            OR = (OR << 1) % 1000000007;
        
        cout << OR << endl;

        }
    return 0;
}
