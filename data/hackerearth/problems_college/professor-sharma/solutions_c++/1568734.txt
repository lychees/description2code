#include<iostream>
#include<math.h>
using namespace std;
main()
{
	int t,X,digitY,countcase=1,countanswer; 
	double O,b;
	cin>>t;
	long long int Y,answer,a;
	while(t--)
	{
		countanswer=1;
		cin>>X>>Y;
		digitY=log10(Y)+1;
		while(1)
		{
		O=countanswer*log10(X);
		a=O;
		b=O-a;
		answer=pow(10,b)*pow(10,digitY-1);
		if(answer==Y)
		{
			cout<<"Case "<<countcase<<": "<<countanswer<<endl;
			countcase++;
			break;
		}
		countanswer++;
		}
	}
}
