#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define mod 1000000007
ll prime[1000002]={0};
ll a[1000002],b[1000002];
vector<ll>x;
void sieve()
{
	int i,j;
	for(i = 2; i <= 1000000; i++) {
		if(!prime[i]) {
			x.push_back(i);
			for(j = 2; j*i <= 1000000; j++) {
				prime[i*j] = 1;
			}
		}
	}
}
ll power(ll a, ll b) {
ll x = 1, y = a;
    while(b > 0) {
        if(b%2 == 1) {
            x=(x*y);
            if(x>mod) x%=mod;
        }
        y = (y*y);
        if(y>mod) y%=mod;
        b /= 2;
    }
    return x;
}
int main()
{
	ll t,n,f,k,i,s;
	sieve();
	cin>>t;
	while(t--) {
		cin>>n;
		i = 0;
		f = 1;
		for(i = 0; i < x.size() && x[i] <= n; i++) {
			s = 0;
			k = x[i];
			while(k <= n) {
				k *= x[i];
				s++;
			}
			f = (f*power(x[i],s))%mod;
		}
		cout<<f<<endl;
	}
    return 0;
}
