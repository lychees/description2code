#include <bits/stdc++.h>

using namespace std;

const int N = 1e5 + 10;

int a[N], b[N], c[N];

int main() {
	int t;
	scanf("%d", &t);
	for (int tt = 1; tt <= t; tt++) {
		int n, m;
		scanf("%d %d", &n, &m);
		for (int i = 0; i < n; i++) {
			scanf("%d", &a[i]);
		}
		for (int i = 0; i < m; i++) {
			scanf("%d", &b[i]);
		}
		merge(a, a + n, b, b + m, c, greater<int>());
		for (int i = 0; i < m + n; i++) {
			printf("%d ", c[i]);
		}
		printf("\n");
	}
	return 0;
}
