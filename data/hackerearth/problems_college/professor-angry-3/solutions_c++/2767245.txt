#include <bits/stdc++.h>
using namespace std;



int main()
{
  ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int t;
	cin>>t;
	while(t--)
	{
		int n, k;
		cin>>n>>k;
		int totl = 0;
		for(int i=0; i<n; i++)
		{

			int input;
			cin>>input;
			input<=0?totl++:totl;
		}
		if(totl<k)cout<<"YES\n";
		else cout<<"NO\n";

		
	}
	return 0;
}
