#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
int n;
char a[100000][30];
int count(char *p)
{
    int f=0;
    for(int i=0;i<n;i++)
    {
        if(!strcmp(a[i],p))
            f++;
    }
    return f;
}
int main()
{
    cin>>n;
    for(int i=0;i<n;i++)
        cin>>a[i];
    for(int i=0;i<n;i++)
    {
        for(int j=i+1;j<n;j++)
        {
            if(a[i][0]>a[j][0])
            {
                char t[100];
                strcpy(t,a[j]);
                strcpy(a[j],a[i]);
                strcpy(a[i],t);
            }
        }
    }
    for(int i=0;i<n;i++)
    {
        if(i!=0)
        {
            if(strcmp(a[i-1],a[i]))
            {
                cout<<a[i]<<" "<<count(a[i])<<endl;
            }
        }
        else if(i==0)
        {
            cout<<a[i]<<" "<<count(a[i])<<endl;
        }
    }
    return 0;
}
