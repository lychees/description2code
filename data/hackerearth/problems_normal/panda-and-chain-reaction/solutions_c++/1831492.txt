#include <iostream>
using namespace std;

int main()
{
    unsigned long long t,n,x,i,res,arr[1000000]={0},n1;
    cin>>t;
    while(t--)
    {
    	cin>>n>>x;
    	x=x%1000003;
        if(n>1000003 || x>1000003)
        {
          cout<<"0"<<endl;
          continue;
        }
    	
    	n1=n%1000003;
    //	cout<<"n% "<<n1<<endl;;
    //	cout<<"x% "<<x%1000003<<endl;
    	if(arr[n1]!=0)
    	{
    		res=(x*arr[n1])%1000003;
    	}
    	else
    	{
	    	res=1;
	    	for(i=2;i<=n1;i++)
	    	{
	    		
	    		res=(res*i)%1000003;
	    		arr[i]=res;
	    		//cout<<"i "<<i<<"   arri "<<arr[i]<<endl;
	    		
	    	}
	    	res=(res*x)%1000003;
    	}	
    	cout<<res<<endl;
    }
    return 0;
}
