int etf(int n) {
	int result = n;
	for (int i=2; i*i<=n; ++i)
		if (n % i == 0) {
			while (n % i == 0)
				n /= i;
			result -= result / i;
		}
	if (n > 1)
		result -= result / n;
	return result;
}

		for(i=1;i*i<=a;i++){
			if(a%i==0){
				int val1=i;
				int val2=a/i;
				if(b%val1==0){
					ans+=etf(val1);
				}
				if((b%val2==0)&&(val2!=val1)){
					ans+=etf(val2);
				}
			}
		}
		printf("%d\n",ans);
	}
	return 0;
}
