#include<stdio.h>

typedef long long ll;

const ll MOD = (1e+9)+7;

ll n,k,L[1001],Fact[1001],ncr[1001][1001];

ll solve()
{
    for(ll i=0;i<=n;++i)
    {
        for(ll j=0;j<=i;++j)
        {
            if(i==j||j==0)
                ncr[i][j] = 1;
            else
                ncr[i][j] = (ncr[i-1][j-1] + ncr[i-1][j])%MOD;
        }
    }

    Fact[0]=Fact[1]=1;
    for(ll i=2;i<=n;++i)
    {
        Fact[i] = (i*Fact[i-1])%MOD;
    }

    L[0]=1;
    for(ll i=1;i<=n;++i)
    {
        for(ll j=0;j<i;++j)
        {
            ll ways = (Fact[j+1]*ncr[i-1][j])%MOD;
            ways = (ways*L[i-j-1])%MOD;
            L[i] += ways; //since L[n+1] = (for a=0 to n) summation{ ncr(n,a)*(a+1)!*L(n-a) } here n+1=i, n=i-1
            L[i] = L[i]%MOD;
        }
    }
    ll ans = (ncr[n][k]*L[n])%MOD;
    //printf("%lld ",L[n]);
    return ans;
}

int main()
{
    scanf("%lld%lld",&n,&k);
    printf("%lld",solve());
    return 0;
}
