#include <iostream>
#include <bits/stdc++.h>
using namespace std;
long long int arr[10000005],sum[10000005];
vector <int> pr;
void sieve()
{
	int i,j,c,p;
	for(i=2;i<=10000000;i++)
	{
		if(!arr[i])
		{
			for(j=i*2;j<=10000000;j+=i)
			{
				c=0;
				p=j;
				while(p%i==0)
				{
					p/=i;
					c++;
				}
				arr[j]+=c;
			}
		}
	}
	sum[0]=0;
	sum[1]=0;
	for(i=2;i<=10000000;i++)
	{
		if(!arr[i])
		{
			arr[i]++;
		}
		sum[i]=sum[i-1]+arr[i];
	}
	
}
int main()
{
	sieve();
	int t,n,i,j;
	cin>>t;
	//cout<<sum[2]<<" "<<sum[3]<<" "<<sum[4]<<" "<<sum[5]<<" ";
	while(t--)
	{
		scanf("%d",&n);
		printf("%lld\n",sum[n]);
	}
    return 0;
}
