#include<iostream>
#include<string>
#include<cassert>
#include<vector>

using namespace std;
int main(){
  int N=0,i=0;
  string name;
  vector<string> str;
  str.clear();
  scanf("%d",&N);
  assert ((N>=1) && (N<=100));
  
  for(i=0;i<N;i++){
    name.clear();
	cin>>name;
	str.push_back(name);
    assert(name.size()<100);
  }
  
  for(i=0;i<N;i++)
   cout<<str[i].size()<<endl;
  return 0;
}
