#include <iostream>
#include <vector>

using namespace std;

typedef long long int Lint;

int main()
{
	ios_base::sync_with_stdio( false );
	Lint t, z, x, y, i, v, ct, n;
	
	cin >> t;
	
	while( t-- ) {
		
		cin >> z >> x >> y;
		vector<Lint> A;
		
		for( i = 1;i * i <= z;i++ ) {
			if( ( z % i ) == 0 ) {
				v = z / i;
				if( i == v ) {
					if( i <= x )
						A.push_back( i );
				}
				else {
					if( i <= x )
						A.push_back( i );
					if( v <= x )
						A.push_back( v );
				}
			}
		}
		
		n = A.size();
		ct = 0;
		for( i = 0;i < n;i++ ) {
			v = z / A[ i ];
			if( v <= y )
				ct++;
		}
		
		cout << ct << endl;
	}
	return 0;
}
