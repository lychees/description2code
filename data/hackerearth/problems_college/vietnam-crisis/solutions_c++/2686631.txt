#include<iostream>
#include<cstdio>
#include<cmath>
#include<vector>
#include<bits/stdc++.h>
#define gc getchar_unlocked
using namespace std;

int read() {
  char c = gc();
  while(c<'0' || c>'9') c = gc();
  int ret = 0;
  while(c>='0' && c<='9') {
    ret = 10 * ret + c - 48;
    c = gc();
  }
  return ret;
}
int main()
{
	while(1)
	{
	long long int n ,i;
	n = read();
	if(n==0)
	{
		break;
	}
	vector<long long int>arr;
	for(i=1;i<=n;i++)
	{
		long long int num;
		num= read();
		arr.push_back(num);
	}
	sort(arr.begin(),arr.end());
	copy(arr.begin(),arr.end(),ostream_iterator<long long int>(cout," "));
	cout<<"\n";
}
}
