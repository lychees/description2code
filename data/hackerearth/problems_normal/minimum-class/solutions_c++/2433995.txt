#include<bits/stdc++.h>
#define MAX 1000000001
using namespace std;

int ans[1000001];

int sieve()
{
	ans[0]=0;ans[1]=1;ans[2]=2;
	int p=2;
	int i;
	for(i=3; ans[i-1]<MAX ; i+=2)
	{
		ans[i]=ans[i-1]+p;
		ans[i+1]=ans[i]+p;
		p++;
	}
	//cout<<i<<endl;
	return i;
}

int main()
{
	int k=sieve();
	int cases,n;
	scanf("%d",&cases);
	while(cases--)
	{
		scanf("%d",&n);
		
		int beg=0,end=k,mid,sum=0;
		mid=(beg + end)/2;
		
		while(beg<=end && n!=0)
		{
			if(ans[mid]<=n && ans[mid+1]>n) {
				if(ans[mid]==n) printf("%d\n",mid);
				else printf("%d\n",mid+1);
				break;
			}
			else {
				if(ans[mid]>n) {
					end=mid-1;
					mid=(beg + end)/2;
					continue;
				}
				else {
					beg=mid+1;
					mid=(beg + end)/2;
					continue;
				}
			}
		}
		
		
	}
	return 0;
}
