# include <iostream>
# include <stdio.h>
# include <vector>
# include <string>
using namespace std;




int main()
{
    int t;
    string s;
    scanf("%d",&t);
    while(t--)
    {

        vector <bool> A(129,false);
        int c=0;
        long long n;
        cin>>s;
        for(int i=0;i<s.length();i++)
        {
            if(!A[(int)s.at(i)])
            {
                A[(int)s.at(i)]=true;
                c++;

            }
        }
        n=(c*c+c)/2;
        printf("%lld\n",n);
    }
    return 0;
}
