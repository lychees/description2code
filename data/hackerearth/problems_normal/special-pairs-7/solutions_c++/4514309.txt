#include<bits/stdc++.h>
using namespace std;

int dp[(1<<22)+2];
int a[1000007];

long long func(int n)
{
for(int j=0;j<21;j++)
for(int mask=0;mask<(1<<21);mask++)
if(mask&(1<<j))
dp[mask]=dp[mask]+dp[mask^(1<<j)];

long long ans=0;
for(int i=1;i<=n;i++)
ans+=dp[((1<<21)-1)^a[i]];
for(int i=0;i<(1<<21);i++)
dp[i]=0;
return ans;
}

int main()
{
ios_base::sync_with_stdio(false);
//freopen("in.txt","r",stdin);
int t;
cin>>t;
while(t--)
{
int n;
cin>>n;
for(int i=1;i<=n;i++)
{
cin>>a[i];
dp[a[i]]++;
}
cout<<func(n)<<endl;
}
return 0;
}
