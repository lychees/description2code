#include<bits/stdc++.h>
using namespace std;
int const N = 1e4+10;
int n;
long long dp[N];
vector<int> adj[N];
long long solve(int v){
  if(dp[v] == -1){
    if(v == n) dp[v] = 1;
    else{
      long long ret = 0;
      for(int i = 0; i < adj[v].size(); i++){
        int w = adj[v][i];
        ret += solve(w);
      }
      dp[v] = ret; 
    }
  }
  return dp[v];
}
int main(){
  scanf("%d",&n);
  while(1){
    int x, y;
    scanf("%d %d",&x,&y);
    if(!x && !y) break;
    adj[x].push_back(y);
  }
  memset(dp,-1,sizeof(dp));
  long long ans = solve(1);
  printf("%lld\n",ans);
  return 0;
}
