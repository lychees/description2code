#include <bits/stdc++.h>
using namespace std;
int ans,val,a[13],n;
void div3(int x,int y,int z,int i)
{
	if(i==n)
	{	
		val=max(max(x,y),z);
		if(val<ans)
			ans=val;
	}
	else
	{
		div3(x+a[i],y,z,i+1);
		div3(x,y+a[i],z,i+1);
		div3(x,y,z+a[i],i+1);
	}
}
int main()
{
    int i;
    ans=1000000;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    div3(0,0,0,0);
    
    
    cout<<ans;
    
    return 0;
}
