n = int(raw_input())

while n>0:
	s = raw_input()
	a = list()
	a.append(s[0])
	for x in range(1,len(s)):
		if	s[x-1]!=s[x]:
			a.append(s[x])
	print ''.join(a)
	
	n-=1
