#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	long long n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		long long i,a[n+1],left[n+1],sum=0,temp,l,r,ans=INT_MAX;
		left[0]=0;
		for(i=1;i<=n;i++)
		{
			left[i]=0;
			cin>>a[i];
		}
		
		for(i=1;i<=n;i++)
		{
			sum=sum+a[i];
			sum=sum%n;
			if(left[sum]==0)
			{
				if(sum==0)
				{
					temp=i;
				}
				else
				temp=INT_MAX;
			}
			else
			{
				temp=i-left[sum];
			}
			if(temp<ans)
			{
				ans=temp;
				l=left[sum]+1;
				r=i;
			}
			left[sum]=i;
		}
		cout<<l<<" "<<r<<endl;
	}
	return 0;
}
