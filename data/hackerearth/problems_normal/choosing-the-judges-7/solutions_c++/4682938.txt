#include <iostream>
using namespace std;
long max(long a,long b)
{
	return a>b?a:b;
}
int main()
{
	int i,j,x,k,t,n;
	cin>>t;
	for(j=1;j<=t;++j)
	{cin>>n;
	long a[n],d[n];
	for(i=0;i<n;++i)
	{
		cin>>a[i];;
	}
	cout<<"Case "<<j<<": ";
	if(n==1)
	{
		cout<<a[0]<<endl;
		continue;
	}
	if(n==2)
	{
		cout<<max(a[0],a[1])<<endl;
		continue;
	}
	d[0]=a[0];
	d[1]=max(a[0],a[1]);
	for(i=2;i<n;++i)
	{
		d[i]=max(a[i]+d[i-2],d[i-1]);
	}
	cout<<d[i-1]<<endl;
	}
    return 0;
}
