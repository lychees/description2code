int Maximum;
int N;
vector<int> Data;
int setMaximum(int i){
	if(i>=N){
		return 0;
	}
	else{
		int val1=setMaximum(2*i);
		int val2=setMaximum(2*i+1);
		int temp=val1+val2+Data[i];
		if(temp>Maximum){
			Maximum=temp;
		}
		return (Data[i]+max(val1,val2));
	}
}

			s(Data[i]);
