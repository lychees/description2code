#include <algorithm>
#include <cmath>
#include <cstdio>
#include <iostream>

using namespace std;

int x[26];
int main() {
  int i, n, ans = 0;
  cin >> n;
  n--;
  char a, b;
  while (n--) {
    cin >> a >> b;
    a = a - 97;
    x[a]++;
    if (x[b - 65] == 0)
      ans++;
    else
      x[b - 65]--;
  }
  cout << ans;
}
