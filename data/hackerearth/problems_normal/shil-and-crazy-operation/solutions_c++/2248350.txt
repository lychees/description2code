	rep(i,n)
		cin>>a[i+1];

		if(!vis[i]){
			int cur=i;
			vector<int>v;
			while(!vis[cur]){
				vis[cur]=1;
				v.pb(cur);

			b[cycle[i][j]]=a[cycle[i][(j+shift)%cycle[i].size()]];
		}
	}
