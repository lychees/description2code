#include <iostream>
#include<cstdio>
#include<string>
using namespace std;

int main()
{
    int a [27][27][27][27][27] = {0};
   	
    int n,count=0;
    cin>>n;
    string s;
    while(n--)
    {
    	cin>>s;
    	int  i[5] = {0};
 
    	for(int j = 0;j<s.size();j++)
    		i[j] = s[j]-int('a') + 1;
    	if(a[i[0]][i[1]][i[2]][i[3]][i[4]] == 0)
    	{
    	//	cout<<"  \n string :"<<s<<"  "<<i[0]<<" "<<i[1]<<" "<<i[2]<<" "<<i[3]<<" "<<i[4];
    		a[i[0]][i[1]][i[2]][i[3]][i[4]]++;
    		count++;
    	}
    }
    cout<<count<<"\n";
    
    for(int b=0;b<27;b++)
    	for(int c =0;c<27;c++)
    		for(int d =0 ;d<27;d++)
    			for(int e =0;e<27;e++)
    				for(int f=0;f<27;f++)
    				{
    					if(b==0 && c==0 && d==0 && d==0 &&f ==0) continue;
    					if(a[b][c][d][e][f] != 0)
    					{
    						if(b!=0) printf("%c",b+'a'-1);
    						
    						if(c!=0) printf("%c",c+'a'-1);
    						
    						if(d!=0) printf("%c",d+'a'-1);
    						
    						if(e!=0) printf("%c",e+'a'-1);
    						
    						if(f!=0) printf("%c",f+'a'-1);
    				//		cout<<"  "<<b<<" "<<c<<" "<<d<<" "<<e<<" "<<f;
    						cout<<endl;
    					}
    				}
    return 0;
}
