#include <bits/stdc++.h>
using namespace std;
#define ll long long

const ll N = (1<<17)+10;

int a[N],b[N],n, m;

int check(int tt)
{
    int i = 0;
    int j = 0;
    while(i < n && j < m)
    {
        if(abs(a[i]-b[j]) <= tt)
        {
            i++;
            j++;
        }
        else
        {
            j++;
        }
    }

    if(i == n)
        return 1;
    else
        return 0;
}


int main()
{
    int t,i;
    scanf("%d", &t);
    while(t--) {
        scanf("%d %d", &n, &m);
        for(i = 0; i < n; i ++)
            scanf("%d", &a[i]);

        for(i = 0; i < m; i ++)
            scanf("%d", &b[i]);

        sort(a, a+n);
        sort(b, b+m);

        int l = 0;
        int r = 210000000;
        int mid;
        while(l <= r)
        {
            mid = (r+l)/2;
            if(check(mid) == 1)
                r = mid-1;
            else
                l = mid+1;
        }

        printf("%d\n", r+1);
    }

    return 0;
}
