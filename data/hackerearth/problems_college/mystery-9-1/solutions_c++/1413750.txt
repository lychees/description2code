#include <iostream>
#include <string.h>
#include <cstdio>
using namespace std;

int main()
{
	int t,ans=0;
	char a[105];
	scanf("%d",&t);
	while(t--)
	{
		ans=0;
		cin>>a;
		for(int i=0;i<strlen(a);i++)
		{
			if(a[i]!=a[strlen(a)-i-1])
			{
			ans=-1;
			break;
			}
		}
		if(ans==0)
		{
		printf("%d\n",a[(strlen(a)-1)/2]);
		}
		else
		cout<<-1<<endl;
		
	}
	return 0;
}
