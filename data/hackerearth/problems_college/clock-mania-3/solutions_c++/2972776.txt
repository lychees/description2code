#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
       
      int h,m;
      cin>>h>>m;
      int a,b;
      a=h*360/12;
      a+=m*30/60;
      b=m*360/60;
    int  ans=abs(a-b);
        if(ans>180)ans=360-ans;
        cout<<ans<<endl;
    }
    return 0;
}
