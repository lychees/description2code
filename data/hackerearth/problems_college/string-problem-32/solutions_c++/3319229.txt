int scan()
{
    int n = 0, ch = get();
    while (ch < '0' || ch > '9')
        ch = get();
    while (ch >= '0' && ch <= '9') {
        n = (n << 3) + (n << 1) + ch - '0';
        ch = get();
    }
    return n;
}


int SubPre(string str1)
{
    string str2 = str1;
    //reverse(str2.begin(), str2.end());
    int idx1 = 1;
    int idx2 = str2.length() - 1;
    string match = "";
    while (idx1 < str1.length() && idx2 >= 0)
    {
        //cout << idx1 << "  " << idx2 << " " << str1.substr(0, idx1) << " hu" <<  str2.substr(idx2) << endl;
        if (str1.substr(0, idx1) == str2.substr(idx2))
        {
             match = str1.substr(0, idx1);
        }
        idx1++;
        idx2--;
    }
    return (match.length());
}
