#include <iostream>
using namespace std;
typedef long long ll;
int main()
{
	ll n,a[100099];
	cin>>n;
	ll i;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
	}
	ll t,j;
	cin>>t;
	int flag;
	while(t--)
	{
		ll x;
		flag =0;
		cin>>x;
		for(i=0;i<n;i++)
		{
			if(a[i]==x)
			{
				cout<<"I Like You"<<endl;
				flag=1;
				break;
			}
		}
		if(flag==0)
		{
			for(i=0;i<n;i++)
			{
				for(j=i+1;j<n;j++)
				{
					if(a[i]+a[j]==x)
					{
						cout<<"I Like You"<<endl;
						flag=1;
						break;	
					}
				}
				if(flag==1)
				{
					break;
				}
			}
		}
		if(flag==0)
		{
			cout<<"I Hate You"<<endl;
		}
	}
    return 0;
}
