#include <iostream>
using namespace std;

int fib(int num){
	int i,a=0,b=1,c=1,d;
	if(num==0) return 1;
	if(num<=2) return 1;
	for(i=3; i<=num; i++){
		d = a+b+i*2;
		d%=10000003;
		a=b;
		b=c;
		c=d;
		
	}
	return d;
}

int main()
{	
	int t,num;
	cin>>t;
	while(t--){
		cin>>num;
		cout<<fib(num)<<endl;
	}
    
    return 0;
}
