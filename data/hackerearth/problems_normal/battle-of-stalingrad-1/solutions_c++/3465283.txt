#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n;
	cin >> n;
	while(n--)
	{


	int x1,x2,x3,x4,y1,y2,y3,y4;
	cin>>x1>>y1>>x2>>y2>>x3>>y3>>x4>>y4;

	float a1,a2,a3,a4;
	a1 = abs((x1*(y2-y3)+x2*(y3-y1)+x3*(y1-y2)));
	a2 = abs((x4*(y2-y3)+x2*(y3-y4)+x3*(y4-y2)));
	a3 = abs((x1*(y4-y3)+x4*(y3-y1)+x3*(y1-y4)));
	a4 = abs((x1*(y2-y4)+x2*(y4-y1)+x4*(y1-y2)));	


	if(a2+a3+a4 == a1)
		cout<<"INSIDE"<<endl;
	else
		cout<<"OUTSIDE"<<endl;
	}
	return 0;
}
