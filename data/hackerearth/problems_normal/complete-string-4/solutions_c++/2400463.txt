#include <iostream>
#include<map>
#include<string>
using namespace std;
int main()
{
int T;
cin>>T;
while(T--)
{
string str;
cin>>str;
map<char,int> m_str;
for(unsigned int i=0;i<str.size();i++)
{
m_str[str[i]] = 1;
}
if(m_str.size() == 26)
{
cout<<"YES"<<endl;
}
else
{
cout<<"NO"<<endl;
}
}
return 0;
}
