#include <bits/stdc++.h>
using namespace std;
long long int a[1000001]={0};
long long int b[1000001]={0};
void call(int n)
{
	int temp=n;
	while(n%2==0)
	{
		n=n/2;
		a[temp]++;
	}
	int i;
	for(i=3;i<=sqrt(n);i+=2)
	{
		while(n%i==0)
		{
			a[temp]++;
			n=n/i;
		}
	}
	if(n>2)
	{
		a[temp]++;
	}
}
void init()
{
	a[0]=b[0]=0;
	a[1]=b[1]=0;
	a[2]=b[2]=1;
	for(int i=3;i<=1000000;i++)
	{
		call(i);
	}

}

{
