#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t,a,b,ans,c,k,k1,k2;
	scanf("%d",&t);
	while (t --) {
		ans = 0;
		scanf("%d %d",&a,&b);
		k1 = 0,k2 = 0;
		int a1[33],b1[33];
		for (c = 31; c >= 0; c--)
        {
           k = a >> c;
 
          if (k & 1)
            //printf("1");
            a1[k1++] = 1;
          else
            //printf("0");
            a1[k1++] = 0;
        }
        for (c = 31; c >= 0; c--)
        {
           k = b >> c;
 
          if (k & 1)
            //printf("1");
            b1[k2++] = 1;
          else
            //printf("0");
            b1[k2++] = 0;
        }
        for (int i = 0; i < k1; i++) {
        	if (a1[i] != b1[i]) {
        		ans++;
        	}
        }
        printf("%d\n",ans);
	}
}
