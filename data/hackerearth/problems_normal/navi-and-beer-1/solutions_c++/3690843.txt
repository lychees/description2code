#include <iostream>
using namespace std;

int main()
{
	double t,n,k,m;
	long cost=0,ans;
    cin>>t;
    if(t<1)
    	t=1;
    if(t>100000)
    	t=100000;
    for(int z=1;z<=t;z++)
    {
    	cost=0;
    	cin>>n>>k>>m;
    	if(n<1)
    		n=1;
    	if(n>1000)
    		n=1000;
    	if(k<1)
    		k=1;
    	if(k>1000)
    		k=1000;
    	if(m<1)
    		m=1;
    	if(m>1000000)
    		m=1000000;
    	for(double i=1;i<=n;i++)
    	{
    		cost+=(k*i*i*i);
    	}
    	if(cost<=m)
            ans=0;
        else
            ans=cost-m;
    	cout<<ans<<endl;
    }
    return 0;
}
