#include <iostream>
#include <string>
using namespace std;

int main()
{
    int N;
    cin >> N;
    string s;
    cin >> s;
    string nS(s);
    s.append(nS);
    int cN = 0, pN = 0, wN = 0, mwN = 0, twN = 0;
    int n = 0;
    int lN = N;
    int max = 0;
    while((s[n] == 'w') && (lN > 0)){
    	n++;
    	lN--;
    }
    char cC = s[n];
    while(((s[n] == cC)||(s[n] == 'w')) && (lN > 0)){
    	n++;
    	lN--;
    }
    if(lN == 0){
    	cout << N << endl;
    	return 0;
    }
    cC = s[n];
    for(; n < 2*N; n++){
    	if(s[n]==cC){
    		cN++;
    		if(twN){
    			cN += twN;
    			twN = 0;
    		}
    	}
    	else if(s[n]=='w')twN++;
    	else{
    		int m = cN + pN + wN + twN + mwN;
    		wN = mwN;
    		mwN = twN;
    		twN = 0;
    		pN = cN;
    		cN = 1;
    		cC=s[n];
    		if(m > max) max=m;
    	}
    }
    int m = cN + wN + twN + mwN;
    if(m > max) max=m;
    cout << max << endl;
    return 0;
}
