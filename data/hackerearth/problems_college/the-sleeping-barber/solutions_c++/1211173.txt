
#include <iostream>
using namespace std;

#include<cstdlib>
int compare (const void * a, const void * b)
{

  return ( *(int*)a - *(int*)b );

}
int main()
{
    int i,j,k,g,gr[3000],m,sum,count;
    cin>>k;
    for(i=0;i<k;i++)
    {
    	cin>>g>>m;
    	for(j=0;j<g;j++)
    		{
    			cin>>gr[j];

    		}
    		qsort(gr,g,sizeof(int),compare);
    	
    		sum=0;count=0;
    		for(j=0;j<g;j++)
    		{
    		sum+=gr[j];
    		count++;
    		if(sum>m)
    		{
    			count--;
    			break;
    		}
    		}
    		cout<<count<<endl;
    }

    return 0;
}
