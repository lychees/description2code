#include <iostream>
#include<string.h>
using namespace std;

int main()
{   char s[10000];
    int ac,ab,len,t;
    cin>>t;
    while(t--)
    {
    	cin>>s>>ac>>ab;
    	len=strlen(s);
    	int flag=1,a=0,b=0;
    	for(int i=0;(i<len)&&flag;i++)
    	{
    		if(s[i]!=s[len-1-i] && s[i]!='/' && s[len-1-i]!='/')
    		{ cout<<-1<<endl;
    		   flag=0;
    		   break;
    		}
    		else if(s[i]=='/')
    		{
    			if(s[len-1-i]=='a')
    			  a++;
    			else if(s[len-1-i]=='b')
    			  b++;
    			else if(s[len-1-i]=='/')
    			{
    				if(ac<ab)
    				  a++;
    				else
    				  b++;
    				
    			}
    	
    		}
        }
        if(flag)
         cout<<a*ac+b*ab<<endl;
    }
    
    return 0;
}
