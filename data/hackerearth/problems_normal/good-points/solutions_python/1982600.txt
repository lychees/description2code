import sys
N = int(sys.stdin.readline())
if N == 1:
	print "YES"
	print "0 0"
elif N == 2:
	print "YES"
	print "0 0"
	print "1 0"
elif N == 4:
	print "YES"
	print "0 0"
	print "0 1"
	print "1 1"
	print "1 0"
else:
	print "NO"
