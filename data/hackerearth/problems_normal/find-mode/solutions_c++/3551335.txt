#include <iostream>
#include <bits/stdc++.h>

using namespace std;

void mode(vector<int> &v,int n){
	int i = 0,max_lim = INT_MIN;
	map<int,int> ma;
	for(i = 0;i < n;++i){
		ma[v[i]]++;
	}
	map<int,int>::reverse_iterator rit;
	map<int,int>::iterator it;
	for(it = ma.begin();it != ma.end();++it){
		max_lim = max(max_lim,it->second);
	}
	for(rit = ma.rbegin();rit != ma.rend();++rit){
		if(rit->second == max_lim) cout<<rit->first<<" ";
	}
	cout<<endl;
	return;
}

int main()
{
	int t = 0,n = 0;
	cin>>t;
	while(t--){
		cin>>n;
		vector<int> v(n,0);
		for(int i = 0;i < n;++i)
			cin>>v[i];
		mode(v,n);
	}
    return 0;
}
