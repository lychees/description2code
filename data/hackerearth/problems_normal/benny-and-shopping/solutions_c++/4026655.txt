#include <bits/stdc++.h>
using namespace std;

int main() {
	int n,t,sum,flag,flag2,i,j;
	char s[10001],c[10001];
	cin>>n;
	
	while(n--){
		i=0;
		cin.ignore();
		gets(s);
		sum=0,flag=0,flag2=0;
		
		for(i=0,j=0;s[i] != '\0';i++){
			if(s[i] == '$'){
				flag++;
				continue;
			}
			if((!(isdigit(s[i])))&&(flag!=0)&&(s[i] != ' ')){
				flag=0;
			}
			if((isdigit(s[i]))&&(flag!=0)){
				t = s[i] - '0';
				if((t!=0)||(flag2!=0)){
					flag2++;
					c[j]=s[i];
					sum=sum+t;
					j++;
				}
			}
		}
		cout<<"$";
		if(sum!=0){
			for(int i=0;i<j;i++){
				cout<<c[i];
			}
		}
		else{
			cout<<0;
		}
		cout<<endl;
	}
	return 0;
}
