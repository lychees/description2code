#include <iostream>
#include <math.h>
using namespace std;
int sumofdig(unsigned long long int x)
  {
 int sum=0;
 unsigned long long int y=x; 
 while(y)
   {
   sum=sum+(y%10);
   y=y/10;
   }
  return sum;
  }

int main()
  {
  unsigned long long int n,y,k,t,count=0,b=0,a;
  long long int x=-1;
  cin>>t;
  while(t--)
    {
    cin>>n;
    y=sqrt(n);
    k=y;
    count=0;
    while(k)
      {
      k=k/10;
      count++;
      }
    count=count-1;
    if(count==0)
      a=1;
    else
    a=y-(9*(count));
    b=0;
    x=-1;
    for(long long int i=a;i<=y;i++)
      {
      b=sumofdig(i);
      if(!((i*i)+(b*i)-n))
      {x=i; break;}
      }
    
      cout<<x<<endl;
    }
  return 0;
  }
