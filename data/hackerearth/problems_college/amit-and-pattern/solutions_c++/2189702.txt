#include<bits/stdc++.h>

using namespace std;

int main(){
    int n,m;
    scanf("%d %d",&n,&m);
    int a[n+1][m+1];
    for(int i=0;i<n;i++){
        for(int j=0;j<m;j++){
            scanf("%d",&a[i][j]);
        }
    }
    for(int j=0;j<m;j++){
        for(int row=0,col=j;row<=j&&col>=0&&row<n;row++,col--){
            printf("%d ",a[row][col]);
        }printf("\n");
    }
    for(int i=1;i<n;i++){
        for(int row=i,col=m-1;row<n&&col>=0;row++,col--){
            printf("%d ",a[row][col]);
        }printf("\n");
    }
    return 0;
}
