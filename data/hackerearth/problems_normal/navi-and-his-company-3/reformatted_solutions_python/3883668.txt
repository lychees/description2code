t = int(raw_input())
for i in range(0, t):
    ec = 0
    count = 0
    k = int(raw_input())
    for j in range(0, k):
        n = raw_input()
        if n.isdigit():
            ec = ec+int(n)
        else:
            if ec < 1:
                count = count+1
            else:
                ec = ec-1
    print count
