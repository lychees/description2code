#include "bits/stdc++.h"
using namespace std;
int main()
{
	long int n;
	cin>>n;
	long int arr[n];
	for(long int i=0;i<n;i++)
		cin>>arr[i];
	
	for(long int i=0;i<n-1;i++)
	{
		if(arr[i]>=0)
		{
			arr[i+1]=arr[i+1]-arr[i];
			arr[i]=0;
		}
	}
	bool check=0;
	for(long int i=0;i<n;i++)
		if(arr[i]!=0)
			check=1;
	if(check==1)
		cout<<"NO\n";
	else
		cout<<"YES\n";
	return 0;
}
