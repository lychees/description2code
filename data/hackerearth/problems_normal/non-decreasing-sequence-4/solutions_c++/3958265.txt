#include <bits/stdc++.h>

using namespace std;

int a[1000010];
int main()
{
    int t,n,i,j,l,lo,hi,minn,top,mid;
	
    //freopen("input.txt","r",stdin);
    cin>>t;
    while(t--)
    {

        cin>>n;
        int s[n];

        for(i=0;i<n;i++)
            cin>>a[i];

        cin>>l;

        minn=INT_MAX;
        s[0]=a[0];
        if(l<=1)
            minn=a[0];
        top=1;
        for(i=1;i<n;i++)
        {
            //cout<<"looping"<<i<<endl;

            if(a[i]>s[top-1])
                {
                    //cout<<"a[i]"<<a[i]<<"inserted at top"<<top<<endl;
                    s[top++]=a[i];

                    if(top>=l)
                    minn=min(minn,a[i]);
                }
            else{
                lo=0;
                hi=top-1;
                while(lo<=hi)
                {
                    mid=(hi+lo)/2;
                    if(s[mid]<a[i])
                        lo=mid+1;
                    else
                        hi=mid-1;
                }
                //cout<<"a[i]"<<a[i]<<"replaced at"<<lo<<endl;
                s[lo]=a[i];
                if(lo+1>=l)
                    minn=min(minn,a[i]);
            }
            //cout<<"top"<<top<<endl;


        }
        if(minn==INT_MAX)
            cout<<-1<<endl;
        else
            cout<<minn<<endl;

    }
}
