#include<bits/stdc++.h>
using namespace std;
#define MAX 1000015
int a[MAX];
int b[MAX];
int main()
{
	int i,j,k,n,m;
	scanf("%d",&n);

	memset(a,0,sizeof(a));
	for(i=0;i<n;i++)
	{
		int tmp;
		scanf("%d",&tmp);
		a[tmp]++;
	}
	
	scanf("%d",&m);
	
	memset(b,0,sizeof(b));
	for(i=0;i<m;i++)
	{
		int tmp1;
		scanf("%d",&tmp1);
		b[tmp1]++;
	}
	
	for(i=0;i<MAX;i++)
	{
		if(a[i]!=b[i])
		{
			printf("%d ",i);
		}
	}
	printf("\n");
	return 0;
}
