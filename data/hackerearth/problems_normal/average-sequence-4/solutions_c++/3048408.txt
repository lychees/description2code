#include <iostream>
using namespace std;

int main()
{
	int size;
	cin >> size;
	
	int *arr;
	arr = new int[size];
	
	int *arr2;
	arr2 = new int[size];
	
	int *arrTemp;
	arrTemp = new int[size];
	
	for(int i=0; i<size; i++){
		cin >> arr2[i];
	}
	
	arrTemp[0] = arr2[0];
	arr[0] = arr2[0]/1;
	
	for(int i=1; i<size; i++){
		arr[i] = arr2[i]*(i+1) - arrTemp[i-1];
		arrTemp[i] = arrTemp[i-1]+arr[i];
		
	}
	
	for(int i=0; i<size; i++){
		cout << arr[i] << " ";
	}
	
    delete arr;
    delete arr2;
    delete arrTemp;
    
    return 0;
}
