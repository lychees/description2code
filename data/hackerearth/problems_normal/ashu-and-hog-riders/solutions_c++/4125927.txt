#include<bits/stdc++.h>
#include<iostream>
using namespace std;


bool dp[1001][1001];
	int n;
	
int dfs(int i, int j)
{
   dp[i][j]=0;
	int ans=1;
	if(i>0&&dp[i-1][j])
	{
		ans+=dfs(i-1, j);
	}
	if(j>0&&dp[i][j-1])
	{
		ans+=dfs(i,j-1);
	}
	if(i<n-1&& dp[i+1][j])
	{
		ans+=dfs(i+1,j);
	}
	if(j<n-1&&dp[i][j+1]){
		ans+=dfs(i,j+1);
	}
	return ans;
}


int main(){

	cin>>n;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
		{
			cin>>dp[i][j];
		}
	}
	
          	int ans=0;
		for(int i=0;i<n;i++)
		{
		  for(int j=0;j<n;j++)
		  {
		  	if(dp[i][j])
		  	{
		  		ans=max(dfs(i,j),ans);
		  	}
		}
	}
	cout<<ans<<endl;
	
}
