#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll long long
#define sc(x) scanf("%d",&x)
#define sc2(x,y) scanf("%d%d",&x,&y)
#define sc3(x,y,z) scanf("%d%d%d",&x,&y,&z)
#define sc4(x,y,z,u) scanf("%d%d%d%d",&x,&y,&z,&u)
#define scll(x) scanf("%lld",&x)
#define scll2(x,y) scanf("%lld%lld",&x,&y)
#define scll3(x,y,z) scanf("%lld%lld%lld",&x,&y,&z)
#define pr(x) printf("%d\n",x)
#define prll(x) printf("%lld\n",x)
char s[100005],s1[100005],s2[100005];
int main()
{
    int i,j,k,l,m,n,o,t;
    sc(t);
    while(t--)
    {
        sc(n);
        cin>>s;
        l=0;m=0;j=0;k=0;
        for(i=0;i<n;i++)
        {
            if(s[i]=='0')
                l++;
            else
                m++;
        }
        for(i=0;i<l;i++)
            s1[i]='0';
        for(i=l;i<n;i++)
            s1[i]='1';
        for(i=0;i<m;i++)
            s2[i]='1';
        for(i=m;i<n;i++)
            s2[i]='0';
        for(i=0;i<n;i++)
        {
            if(s[i]!=s1[i])
                j++;
            if(s[i]!=s2[i])
                k++;
        }
        k=min(j,k);
        pr(k/2);
    }
    return 0;
}
