
#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int i,j,k,m,n;
	cin>>m>>n;
	long long int a[m][n],b[n];
	for(i=0;i<m;i++)
	{
		cin>>a[i][0];
		k=a[i][0];
		for(j=1;j<n;j++)
		{
			cin>>a[i][j];
			if(a[i][j]<k)
			k=a[i][j];
		}
		b[i]=k;
	}
	k=b[0];
	for(i=1;i<m;i++)
	{
		if(b[i]>k)
		k=b[i];
	}
	cout<<k;
	return 0;
}
