#include <iostream>
#include<math.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t>0)
	{
		char arr[25][25];
		int n;
		cin>>n;
		for(int i=0;i<n;i++)
		{
			cin>>arr[i];
		}
		int index1,index2;
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<n;j++)
			{
				if(arr[i][j]=='*')
				{
					index1=i;
					index2=j;
				}
			}
		}
		n--;
		int num=abs(n/2-index1)+abs(n/2-index2);
		cout<<num<<endl;
		t--;
	}
}
