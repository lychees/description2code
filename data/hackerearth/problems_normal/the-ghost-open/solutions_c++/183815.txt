#include <iostream>
#include <cmath>
using namespace std;

long long up[30] ;

void init( ){

      long long sum = 1 ;
      for( int i = 1 ; i < 30 ; i ++ ){
      	  
      	 up[i] = sum ;
      	  sum += pow(2,i) ;
	//	 cout<<i<<" "<<up[i]<<endl ;
      	
      }
	
}

long long solve( int num ){
     long long result ;
	if( num == 1 )
	   result = 0 ;
	else{
	   int me = log(num) / log(2) ;
	   int r = pow( 2 , me ) ;
	  // cout<<r<<" "<<num<<endl ;
	   if( num > r  )
	      result = up[me] + solve( num - r  ) + 1 ;
    	else 
	      result = up[me] ;
	}
	return result ;
	
}

int main()
{
	  init();
      int t ;
      cin>>t ;

      	// long long result = solve( num );

      	if( num > 0  ) 
          cout<<num-1<<endl ;
        else 
          cout<<0<<endl ;
      }
      return 0;
}
