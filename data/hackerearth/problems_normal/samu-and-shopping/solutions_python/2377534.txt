for t in range(input()):
    n = input()
    arr = []
    for i in range(n):
        arr.append(map(int,raw_input().split()))
    for i in range(n-2,-1,-1):
        arr[i][0] += min(arr[i+1][1],arr[i+1][2])
        arr[i][1] += min(arr[i+1][0],arr[i+1][2])
        arr[i][2] += min(arr[i+1][1],arr[i+1][0])
    print min(arr[0])
