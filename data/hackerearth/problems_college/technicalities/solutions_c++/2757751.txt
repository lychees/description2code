#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long int n;
    	cin>>n;
    	string s;
    	cin>>s;
    	unsigned long long int ans=0;
    	for(long int i=0;i<n;i++)
    	{
    		ans=(ans*10+s[i]-48)%1000000007;
    	}
    	ans=ans%1000000007;
    	cout<<ans<<endl;
    	
    }
    return 0;
}
