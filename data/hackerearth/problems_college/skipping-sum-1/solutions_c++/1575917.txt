#include<iostream>
using namespace std;

int main(){

	int number;
	cin>>number;

	int queries;
	cin>>queries;

	long *input = new long[number+1];

	for(int i=1;i<=number;i++) cin>>input[i];

	long **store = new long*[11];

	for(int i=1;i<=10;i++) store[i] = new long[number];

	int i,j;
	for(j=1;j<=10;j++){
		for(i=0;i<=number;i++){
        	 store[j][i]=0;
        }
	}

	for(j=1;j<=10;j++){
        for(i=1;i<=number;i++){
            if(i-j>=0){
                store[j][i]= input[i]+store[j][i-j];	  
            }
            else{
                store[j][i]= input[i];
            }
        }
    }

	long output;

	for(i=0;i<queries;i++){
		output = 0;
		int left,right,step;
		cin>>left;
		cin>>right;
		cin>>step;

		if((right - left)%step!=0) right = right-((right-left)%step);
		if(left-step>=1) output = store[step][right] - store[step][left-step];
		else
			output = store[step][right];

		cout<<output<<endl;
	}
	return 0;
}
