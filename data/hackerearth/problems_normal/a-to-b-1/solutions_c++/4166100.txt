using namespace std;
ll powe(ll a,ll b)
{
	ll an=1;
	while(b!=0)
	{
		if(b%2==1)
			an=(an*a)%mod;
		a=(a*a)%mod;
		b=b/2;
	}
	return an;
}
ll a[100001];
int main()
{
	int n;
	si(n);
	ll pr=1;
	int cnt=0;
	iter(i,1,n+1)
	{
		sll(a[i]);
		if(a[i]!=0)
			pr=(pr*a[i])%mod;
		else
			cnt++;
	}
	int q;
	si(q);
	while(q--)
	{
		int ch;
		si(ch);
		if(ch==0)
		{
			int id;
			ll v;
			si(id);sll(v);
			if(a[id]!=0)
				pr=(pr*powe(a[id],mod-2))%mod;
			else
				cnt--;
			if(v==0)
				cnt++;
			else
				pr=(pr*v)%mod;
			a[id]=v;
		}
		else
		{
			int id;
			si(id);
			ll an;
			if((a[id]==0 && cnt-1>0) || (a[id]!=0 && cnt>0))
				an=0;
			else if(a[id]==0 && cnt-1==0)
				an=pr;
			else
				an=(pr*powe(a[id],mod-2))%mod;
			pll(an);
		}
	}
	return 0;
}
