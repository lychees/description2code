#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
 int t,n;
 string temp ;

 cin>>t;
 
 while(t--)
 {
 	map<char,int> mp;
 	 bool lock =false;
 	cin>>n;
 

 	for(int i=0;i<n;i++)
 	{
 	
 		
 		cin>>temp;
 		
 		if(mp[temp[0]]){
 		lock=true;}
 		else{
 	     mp[temp[0]]++;}
 	}
 		
 if(lock==true){
 cout<<"NO"<<endl;}
 else{
 cout<<"YES"<<endl;}
 }
    return 0;
}
