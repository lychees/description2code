using namespace std;
unsigned LL dp[102][102][20];
int n;
unsigned LL rec(int x,int y,int s){
    if(s==0) return x==50 && y==50;
    if(~dp[x][y][s]) return dp[x][y][s];
    return dp[x][y][s]=rec(x-1,y,s-1) + rec(x-1,y+1,s-1)+rec(x+1,y,s-1)+rec(x+1,y-1,s-1)+rec(x,y+1,s-1)+rec(x,y-1,s-1);
}
int main(){
    memset(dp,-1,sizeof(dp));
    int T;
    scanf("%d",&T);
    while(T--){
        scanf("%d",&n);
        printf("%llu\n",rec(50,50,n));
    }
    return 0;
}
