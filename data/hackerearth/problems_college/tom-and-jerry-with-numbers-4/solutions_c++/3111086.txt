#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,m1,m2,i;
        cin>>n>>m1>>m2;
        bool vis[1000002]={0};
        for(i=0;i<=n;i++)
        {
        	if(i<m1&&i<m2)
        	vis[i]=0;
        	else
        	{
        		if(i-m1>=0&&vis[i-m1]==0)
        		vis[i]=1;
        		if(i-m2>=0&&vis[i-m2]==0)
        		vis[i]=1;
        	}
        
        }
        cout<<vis[n]<<endl;
    }
    return 0;
}
