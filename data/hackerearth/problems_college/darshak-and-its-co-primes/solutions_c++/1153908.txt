#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll gcd(ll a,ll b) {
    return (b==0)?a:gcd(b,a%b);
}
int main() {
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        ll a,b;
        scanf("%lld%lld",&a,&b);
        if(gcd(a,b)==1) printf("Is a Co-Prime\012");
        else printf("Not a Co-Prime\012");
    }
    return 0;
}
