#include<bits/stdc++.h>
using namespace std;
long long dp[180][1600][20];
long long arr[20];
long long gcd(long long a,long long b)
{
    if(a%b==0)
        return b;
    else
        return gcd(b,a%b);
}
long long rec(long long sum,long long s,long long dig,bool check)
{
    long long r;
    if(dig<=-1)
    {

        //cout<<"HELLPO"<<endl;
    if(__gcd(sum,s)==1)
        return 1;
    else
        return 0;
    }
        if( !check && dp[sum][s][dig]!=-1)
            return dp[sum][s][dig];
        long long ans=0;
        if(check==true)
            r=arr[dig];
        else
            r=9;
    for(long long i=0;i<=r;i++)
    {
       ans+= rec(sum+i,s+(i*i),dig-1,check && (i==r));
    }
if(!check)
    dp[sum][s][dig]=ans;
    return ans;
}

long long digits(long long i)
{
    long long k=0;
    while(i!=0)
    {
        long long d=i%10;
        arr[k++]=d;
        i=i/10;
    }
    rec(0,0,k-1,true);
}
