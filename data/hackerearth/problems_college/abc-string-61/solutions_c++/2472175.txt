#include <iostream>
using namespace std;

#define ll long long int
#define MOD 1000000007

ll power(ll a,ll b)
{
	if(b==0) return 1;
	else {
		ll p = power(a,b/2);
		
		if(b%2==0) {
			return (p*p)%MOD;
		} else return (a*((p*p)%MOD))%MOD;
	}
}

int main()
{
	ll i,n;
	ll t;
	
	cin>>t;
	
	while(t--) {
	cin>>n;
	
	if(n<3) cout<<0<<endl;
	else {
		ll ans;
		
		ans = (3 * ((((power(3,n-1) - power(2,n))%MOD + MOD)%MOD + 1)%MOD))%MOD;
		cout<<ans<<endl;
	}
	}
	return 0;
}
