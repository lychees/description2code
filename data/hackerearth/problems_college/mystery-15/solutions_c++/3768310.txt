#include<bits/stdc++.h>
using namespace std;

int main()
{
	long long int t,k,s,l=0;
	scanf("%d",&t);
	while(t--)
	{
		l=0;
		scanf("%lld",&k);
		scanf("%lld",&s);
		l+=s;
		scanf("%lld",&s);
		l+=s;
		scanf("%lld",&s);
		l+=s;
		l=pow(l,k);
		printf("%lld\n",l);
	}
	return 0;
}
