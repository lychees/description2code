#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll mod = 1000000007;
ll fast_power(ll base, ll exponent, ll modulus)
{
    ll result = 1;
    while (exponent > 0) {
        if (exponent % 2 == 1)
            result = (result * base) % modulus;
        exponent = exponent >> 1;
        base = (base * base) % modulus;
    }
    return result;
}
int main()
{
    int t;
    cin >> t;
    assert(1 <= t && t <= 5);
    while(t--) {
        int n;
        cin >> n;
        assert(2 <= n && n <= 16);
        ll a[n + 1];
        for (int i = 0; i < n; i++) {
            cin >> a[i];
            assert(1 <= a[i] && a[i] <= 1e9);
        }
        int p = (1<<n);
        ll ans = -1;
        for (int i = 1; i < p; i++) {
            vector <ll> v;
            for (int j = 0; j < n; j++) {
                if (i&1<<j) {
                    v.push_back(a[j]);
                }
            }
            if (v.size()%2 == 0) {
                ll temp1 = 1;
                ll temp2 = 1;
                sort(v.begin(), v.end());
                int z = v.size();
                for (int j = 0; j < z/2; j++) {
                    temp1 = ((temp1%mod)*(v[j]%mod))%mod;
                }
                for (int j = z/2; j < z; j++) {
                    temp2 = ((temp2%mod)*(v[j]%mod))%mod;
                }
                temp1 = (temp2*fast_power(temp1, mod - 2, mod))%mod;
                if (ans < temp1) {
                    ans = temp1;
                }
            }
        }
        cout << ans << endl;
    }
    return 0;
}
