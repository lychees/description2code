#include <bits/stdc++.h>
#include <string>
using namespace std;
int main()
{
	int flag=1;
	int odd=0;
    string n;
    cin>>n;
    int s=n.size();
    for(char i='a';i<='z';i++)
    {
    	int f=0;
    	for(int j=0;j<s;j++)
    	{
    		if(n.at(j)==i)
    		f++;
    	}
    	if(f%2!=0 && odd==0)
    	{
   			odd=1;
   			continue; 
    	}
    	if(f%2!=0 && odd==1)
    	{
    		flag=0;
    		break;
    	}
    }
    if(flag==0)
    cout<<"NO"<<endl;
    else
    cout<<"YES"<<endl;
    return 0;
}
