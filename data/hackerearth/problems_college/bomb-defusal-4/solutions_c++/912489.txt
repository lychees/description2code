#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    int p;
    int x;
    int y;
    int u;
    int v;
    int t1;

    scanf("%d",&t);

    while(t--) {
        scanf("%d%d%d%d%d",&p,&x,&y,&u,&v);

        if((abs(x-y) % 2 == 0 && abs(u-v) % 2 == 1) || abs(x-y) % 2 == 1 && abs(u-v) % 2 == 0) {
            printf("Terrorists Win !\n");
            continue;
        }

        if(x == u && y == v) {
            printf("Counter-Terrorists Win !\n");
            continue;
        }

        if((x - y) == (u - v)) {
            t1 = 1;
        }
        else {
            t1 = 2;
        }

        if(t1 < p) {
            printf("Counter-Terrorists Win !\n");

        }
        else {
            printf("Terrorists Win !\n");

        }
    }
}
