#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,a[26],b[26],i;
	cin>>t;
	while(t--)
	{
		string s1,s2;
		cin>>s1;
		cin>>s2;
		memset(a,0,sizeof(a));
		memset(b,0,sizeof(b));
		for(i=0;i<s1.length();i++)
		a[s1[i]-'a']++;
		for(i=0;i<s2.length();i++)
		b[s2[i]-'a']++;
		int flag=0;
		for(i=0;i<26;i++)
		if(a[i]!=0&&b[i]!=0)
		{
			flag=1;
			break;
		}
		if(flag)
		cout<<"Yes"<<endl;
		else
		cout<<"No"<<endl;
	}
	return(0);
}
