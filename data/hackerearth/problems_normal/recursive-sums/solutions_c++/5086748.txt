#include <iostream>
#include<cassert>
using namespace std;
#define long1 unsigned long long
int myFun(long1 m){
	if(m<10){
		return m;
	}
	int result = 0;
	while(m>0){
		result+=(m%10);
		m/=10;
	}
	return myFun(result);
}
int main()
{
    int t;
    int n,d;
    cin>>t;
    while(t--){
    	cin>>n;
    	long1 sum = 0;
    	long1 len;
    	
    	for(int i = 0;i<n;i++){
    		cin>>len>>d;
    		sum+=d*len;
    		sum = myFun(sum);
    	}
    	sum = myFun(sum);
    	cout<<sum<<endl;
    }
    return 0;
}
