#include <iostream>
using namespace std;
#define MAX 1000000
int sieve[MAX];
void precomp(){
	for(int i=0;i<=MAX;i++)sieve[i]=1;
	sieve[0]=sieve[1]=0;
	for(int i=2;i*i<=MAX;i++){
		if(sieve[i]==1){
			for(int j=i*i;j<=MAX;j+=i){
				if(sieve[j]==1)sieve[i]++;
				sieve[j]=0;
				
			}
		}
	}
}
int main()
{
	ios::sync_with_stdio(0);
	precomp();
    int t;
    cin>>t;
    while(t--){
    	int x;
    	cin>>x;
    	cout<<sieve[x]<<'\n';
    	
    }
    return 0;
}
