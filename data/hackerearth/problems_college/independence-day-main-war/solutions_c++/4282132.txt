#include <bits/stdc++.h>

using namespace std;

double xS, yS;

double Kahan_Summation(double x[], double y[], long N){

    double xSum = 0.0, ySum = 0.0;
    double xC = 0.0, yC = 0.0;

    for(long int i=0; i<N; i++){

        double tX = x[i] - xC, tY = y[i] - yC;
        double rX = xSum + tX, rY = ySum + tY;

        xC = (rX - xSum) - tX;
        yC = (rY - ySum) - tY;

        xSum = rX;
        ySum = rY;
    }

    xS = xSum;
    yS = ySum;
}


int main(){

   // FILE *in = freopen("SampleInput.txt", "r", stdin);
   // FILE *out = freopen("SampleOutput.txt", "w", stdout);
   // assert(in != NULL);

    long T;
    cin>>T;

    assert(T>=1 && T<=51);
    while(T--){

        long N;
        cin>>N;

        assert(N>=30 && N<=10001);

        double x[N], y[N];
        for(long int i=0; i<N; i++){
                cin>>x[i]>>y[i];
        }
        Kahan_Summation(x, y, N);
        printf("(%.02f , %.02f)\n", (float)(xS/(float)N), (float)(yS/(float)N));
    }

    return 0;
}
