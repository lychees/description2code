#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,m,i,j;
    long long e;
    scanf("%d",&n);
    long long arr[n];
    for(int i=0;i<n;i++)
    scanf("%lld",&arr[i]);
    scanf("%d",&m);
    vector<long long> diff(n+1,0);
    while(m-->0)
    {
    	scanf("%d %d %lld",&i,&j,&e);
    	i--;
    	j--;
    	diff[i]+=e;
    	diff[j+1]-=e;
    }
    long long update=0;
    for(int i=0;i<n;i++)
    {
    	update+=diff[i];
    	printf("%lld\n",(arr[i]+update));
    }
    return 0;
}
