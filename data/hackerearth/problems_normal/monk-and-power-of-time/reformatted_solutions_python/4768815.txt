num_processes = int(raw_input().strip())
call_order = map(int, raw_input().strip().split())
ideal_order = map(int, raw_input().strip().split())

current_index = 0
processes_done = 0
while processes_done < num_processes:
    if call_order[current_index] == ideal_order[processes_done]:
        processes_done += 1
    else:
        call_order.append(call_order[current_index])
    current_index += 1
