ll solve(int l, int cur, int L, int K) {
	if(l>=L) return 0;
	if(cur==K) return imax;
	ll &ret = dp[l][cur];
	if(ret != -1) return ret;
	ret = imax;
	if(l>=in[cur].ff.ss) ret = in[cur].ss + solve(max(l,in[cur].ff.ff), cur+1, L, K);
	ret = min(ret, solve(l, cur+1, L, K));
	return ret;
}
