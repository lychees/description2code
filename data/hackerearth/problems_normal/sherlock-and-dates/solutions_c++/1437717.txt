int f(int n, int i)
{
    return n / 100 + (n % 100 >= i);
}

int solve() 
{
    char c;
    
    int d1, m1, y1, d2, m2, y2;

    cin >> d1 >> c >> m1 >> c >> y1 >> d2 >> c >> m2 >> c >> y2;

    int ans = 0;
    for(int i = 3; i <= 13; i++) {
        int cnt = f(y2, i) - f(y1 - 1, i);
        int need_mo = i - 1;
        int need_d = need_mo - 1;
        if(y1 % 100 == i and (m1 > need_mo or (m1 == need_mo and d1 > need_d)))
            cnt--;
        if(y2 % 100 == i and (m2 < need_mo or (m2 == need_mo and d2 < need_d)))
            cnt--;
        ans += max(0, cnt);


    }

    return ans;
}

int main()
{
    int t;
    cin >> t;
    while(t-- > 0)
        cout << solve() << endl;
    return 0;
}
