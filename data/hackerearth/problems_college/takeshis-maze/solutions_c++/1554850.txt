#include <iostream>
#include<string.h>

using namespace std;

#define N 100

int mat[100][100];

void print(int m)
{
       int i=0;
	   int j=0;
	   for(i=0;i<m;i++){
		   for(j=0;j<m;j++){
			   cout<<mat[i][j]<<" ";
		   }
		   cout<<endl;
	   }
}
int move(int i,int j);
int maxnum( int i, int j);

int main()
{   
    memset(mat,0,sizeof(mat));
//	freopen("input.txt","r",stdin);

	int T;
	cin>>T;
	int n;
	int c;
	int ans=0;

	while(T--)
	{
       cin>>n;
	   cin>>c;
	   int i=0;
	   int j=0;
	   for(i=0;i<n;i++)
		   for(j=0;j<n;j++)
			   cin>>mat[i][j];

	   /* print corridors */
      // print(n);
	   /* starting position */
	   i=0;
	   j=0;
       
	   
	   for(i=0;i<n;i++)
	   {
        for(j=0;j<n;j++)
		{
			    
            int sum = 0;
            if( i > 0)
	            sum = maxnum(sum , mat[i-1][j]);
	        if( j > 0)
			     sum = maxnum(sum , mat[i][j-1]); 
			    
	            mat[i][j] = mat[i][j] + sum;
			
		}
	   }
       ans = mat[n-1][j-1] - c;
	   if( ans < 0)
	   ans = -1;

	   cout<<ans<<endl;


	   
	   i=j=n=c=ans=0;
	}
    return 0;
}


int maxnum( int i, int j)
{
	if( i > j)
	return i;
	else
	return j;
	
}
