#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		int n;
		cin >> n;
		bool t = false;
		bool o = false;
		if(n%21 == 0) cout << "The streak is broken!";
		else
		{
			while(n>0)
			{
				int x = n%10;
				if(!o && x == 1) 
				{	
					o = true;
					int m = n/10;
					int y = m%10;
					if(o && y==2) t = true;
					else o = false;
				}
				if(o && t) 
				{
					cout << "The streak is broken!";
					break;
				}
				n = n/10;
			}
			if(!(o && t)) cout << "The streak lives still in our heart!";
		}
		cout << endl;
	}
}
