#include <iostream>
using namespace std;

int main()
{
    long t;
    cin>>t;
    while(t--)
    {
    	long double a,b,c,k,l;
    	cin>>a>>b>>c;
    	long long j=(a>b)?a:b;
    	j=(j>c)?j:c;
    	if(j==a)
    	{
    		k=b;l=c;
    	}
    	else if(j==b)
    		{
    			k=a;l=c;
    		}
    		else
    		{
    			k=a;l=b;
    		}
    	if((k+l)<=j)
    		cout<<"NO\n";
    	else
    	{
    		if(a==b&&b==c)
    			cout<<"YES. It is a equilateral triangle.\n";
    		else if(a==b || b==c || a==c)
    				cout<<"YES. It is a isosceles triangle.\n";
    			else
    				cout<<"YES. It is a scalene triangle.\n";
    	}
    }
    return 0;
}
