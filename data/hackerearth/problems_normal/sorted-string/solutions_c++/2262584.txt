#include <iostream>
using namespace std;
#include<algorithm>


#include <cstring>
pair<int,int> n[26];
bool comp(pair<int,int> a,pair<int,int> b)
{
return a.second<b.second||(a.second==b.second&&a.first<b.first);	
}
int main()
{
   int t;
   cin>>t;
   char a[101];
   while(t--)
   {
   	
   	for(int i=0;i<26;i++)
   {	n[i].first='a'+i;n[i].second=0;}
   	cin>>a;
   	int i=0;
   	while(a[i])
   	{
   		n[a[i]-'a'].second++;
   		i++;
   	}
   	sort(n,n+26,comp);
   	//cout<<n[25].second;
   	for(int i=0;i<26;)
   	{
   	if(n[i].second==0)
   	i++;
   	else
   	{
   		cout<<(char)(n[i].first);
   		n[i].second--;
   	}
   	}
   	cout<<"\n";
   }	
   
    return 0;
}
