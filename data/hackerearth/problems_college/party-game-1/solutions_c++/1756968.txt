#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		long long x,y,n;
		cin>>x>>y>>n;
		long long arr[n];
		for(long i=0;i<n;i++){
			cin>>arr[i];
		}
		sort(arr,arr+n);
		long count1=0;
		long long sum1,sum2;
		sum1=sum2=0;
		long count2=0;
		long i=0;
		while(sum1<=x&&i<n){
			sum1+=arr[i];
			if(sum1<=x)
			count1++;
			i++;
		}
		i=0;
		while(sum2<=y&&i<n){
			sum2+=arr[i];
			if(sum2<=y)
			count2++;
			i++;
		}
		//cout<<count1<<" "<<count2<<endl;
		if(count1==count2){
			cout<<"Tie"<<endl;
		}
		else if(count1<count2){
			cout<<"Jhool Won"<<endl;
		}
		else{
			cout<<"Chef Won"<<endl;
		}
	}
}
