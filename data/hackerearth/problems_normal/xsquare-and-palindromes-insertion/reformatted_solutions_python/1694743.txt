t = int(input())
while (t > 0):
    s = raw_input()
    a = [0]*150
    n = len(s)
    for x in s:
        a[ord(x)] += 1

    odd = 0
    even = 0

    for i in range(0, 150):
        if (a[i] != 0):
            if (a[i] % 2 == 0):
                even += 1
            else:
                odd += 1

    if (n % 2 == 1):
        print odd-1
    else:
        if (odd >= 1):
            print odd-1
        else:
            print "0"
    t -= 1
