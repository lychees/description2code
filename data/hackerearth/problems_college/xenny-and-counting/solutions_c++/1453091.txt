#include <iostream>
using namespace std;
double fact(int x)
{
	double f=1;
	for(int i=2;i<=x;++i)
	f*=i;
	return f;
}
int main()
{
    ios::sync_with_stdio(false);
    int T;
    cin>>T;
    for(;T>0;--T)
    {
    	int N,K;
    	cin>>N>>K;
    	cout.precision(0);
    	cout<<fixed<<fact(K+N-1)/(fact(N-1)*fact(K))<<endl;
    }
}
