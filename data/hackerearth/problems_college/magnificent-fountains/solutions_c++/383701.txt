#include <bits/stdc++.h>

#define PB(a,b) a.push_back(b)
#define MP(a,b) make_pair(a,b)
#define leng(a) a.size()
#define MOD 1000000007
using namespace std;

typedef long long ll;
typedef vector<int> vi;
typedef vector<pair<int,int> > vii;
typedef vector<string> vs;

int v[100010];
int cnt;
vi vars;


ll find_lcm(){
    for(int i=0;i<leng(vars);i++){
        cnt = 0;
        while(vars[i]%2==0){
            cnt ++;
            vars[i]/=2;
        }
        v[2] = max(v[2],cnt);
        for(int j=3;j <= sqrt(vars[i]); j++){
            cnt = 0;
            while(vars[i]%j==0){
                cnt++;
                vars[i]/=j;
            }
            v[j] = max(v[j],cnt);
        }
        if(vars[i]>2)
            v[vars[i]] = max(v[vars[i]],1);
    }
    ll ans = 1;
    for(int i=2;i<100010;i++)
        if(v[i])
            ans = (ans*((ll)pow((ll)i,(ll)v[i])%MOD))%MOD;
    return ans;
}


int main(){
    int t;
    scanf("%d",&t);
    while(t--){
        int n,x;
        scanf("%d",&n);
        
        vars.clear();
        for(int i=1;i<=n;i++){
            scanf("%d",&x);
            vars.push_back(x);
        }
        printf("%lld\n",find_lcm());
    }
}
