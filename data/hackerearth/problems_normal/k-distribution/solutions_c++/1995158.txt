#include<bits/stdc++.h>
using namespace std;
#define MOD 1000000007
long long  k,A[101],dp[100001],n,total_numbers=1,ans=0;
int main()
{
    long long sum;
    cin>>n>>k;
    for(int i=0;i<n;i++)
    {
        cin>>A[i];
        sum+=A[i];
       total_numbers=total_numbers*2LL;
       total_numbers%=MOD;
       
    }
    k--;
    dp[0]=1;
    for(int i=0;i<n;i++)
    {
        for(long long  j=k-A[i];j>=0;j--)
        {
            if(dp[j])
            {
                dp[j+A[i]]+=dp[j];
                dp[j+A[i]]%= MOD;
            }
        }
    }
    for(int i=0;i<=k;i++)
    {
        long long dusra_sum=sum-i;
        if(dusra_sum<=k)  // ek baar firse count hoga
            ans=(ans+ dp[i])%MOD;
        else
            ans= (ans+(2*dp[i])) % MOD;
    }
    cout<<(total_numbers-ans+MOD)%MOD;
}
