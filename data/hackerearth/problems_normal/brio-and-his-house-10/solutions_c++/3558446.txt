#include<iostream>
#include<stdio.h>
#include<math.h>
using namespace std;
main()
{
  int t,i;
  double area,areaT,r,a,b,c,pie=3.1415,s;
  cin>>t;
  for(i=0;i<t;i++)
  {
      cin>>a>>b>>c;
      s=(a+b+c)/2;
      areaT=sqrt((s)*(s-a)*(s-b)*(s-c));
      r=(a*b*c)/(4*areaT);
      area=pie*r*r;
      printf("%.4f\n",area);
  }
}
