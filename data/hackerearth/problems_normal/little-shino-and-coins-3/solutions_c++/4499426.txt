#include <iostream>
using namespace std;

int main()
{
    int k;
    string s;
    cin>>k>>s;
    int count=0;
    int ans = 0;
    int data[26] = {0};
    for(int i=0;i<s.length();i++)
    {
    	for(int j=i;j<s.length();j++)
    	{
    		if(data[s[j]-'a']==0)
    		{
    			count++;
    			data[s[j]-'a'] = 1;
    		}
    		if(count==k)
    		{
    			ans++;
    		//	cout<<ans<<s[j]<<endl;
    		}
    		if(count>k)
    		{
    			for(int l=0;l<26;l++)
    			{
    				data[l]=0;
    			}
    			count=0;
    			break;
    		}
    	}
    	
    		for(int l=0;l<26;l++)
    			{
    				data[l]=0;
    			}
    			count=0;
    }
    cout<<ans;
    return 0;
}
