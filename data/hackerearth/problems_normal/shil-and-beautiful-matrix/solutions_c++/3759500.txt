#include <bits/stdc++.h>
using namespace std;
#define FOR(i,a,b) for(i=a;i<b;++i)
#define rep(i,n) FOR(i,0,n)
#define MOD 1000000007
typedef long long lln;
int min(int a,int b,int c){ return min(a,min(b,c));}
int max(int a,int b,int c){ return max(a,max(b,c));}

lln fact (lln a){
    if(a<=1) return 1;
    return (a*fact(a-1))%MOD;
}
vector<lln> fx(1024,0);
int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    //freopen("YYY.txt","r",stdin);
    //freopen("ZZZ.txt","w",stdout);

    int m,n;
    cin>>m>>n;
    for(int i=1;i<=m;i++){
        for(int j=1;j<=n;j++){
            fx[i^j]++;
        }
    }
    lln Ans=1;
    int i;
    rep(i,1024){
            Ans=(Ans*fact(fx[i]))%MOD;
        
    }
    cout<<Ans<<'\n';


    return 0;
}
