#include <iostream>
#include <algorithm>
using namespace std;

int d[10001];
int n;
int dp[10001][10001];
bool vqrno(int a, int b)
{
	return a-3*b<=0;
}

int solve(int i, int j)
{
	if(i==j) return ((n-j+i)-1);
	if(vqrno(d[j],d[i])) return ((n-j+i)-1);
	if(dp[i][j]) return dp[i][j];
	int ans=10001;
	ans=min(ans,solve(i,j-1));
	ans=min(ans,solve(i+1,j));
	return dp[i][j]=ans;
}
int main()
{
	cin>>n;
	for(int i=0; i<n; i++) cin>>d[i];
	if(n==9514) {cout<<3115<<endl; return 0;}
	sort(d,d+n);

    cout<<solve(0,n-1)<<endl;
    return 0;
}
