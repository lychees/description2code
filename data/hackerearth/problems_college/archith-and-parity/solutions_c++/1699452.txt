#include<cstdio>
#include<iostream>
using namespace std;
int  main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long n;
		cin>>n;
		int c=0;
		while(n>0)
		{
			long long rem = n%2;
			if(rem == 1)
			{
				c++;
			}
			n = n/2;
		}
		if(c%2==0)
		{
			cout<<"even"<<"\n";
		}
		else
		{
			cout<<"odd"<<"\n";
		}
	}
	return 0;
}
