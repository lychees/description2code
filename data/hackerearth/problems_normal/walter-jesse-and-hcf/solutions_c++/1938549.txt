#include <iostream>
using namespace std;
long long int hcf(long long int x,long long int y)
{
	if(y==0)
	return x;
	else
	return  hcf(y,x%y);
}
int main()
{   
	 long long int d,x,y,k,i,p;

	cin>>d;
	while(d--)
	{
		cin>>x>>y>>k;
		p=hcf(x,y);
		if(k==1)
		cout<<p<<endl;
		else
		{
		long long	int r=1,flag=0;
			for(i=p/2;i>=1;i--)
			{
				if(p%i==0)
				{
				r++;
				if(r==k)
				{
					flag=1;
					cout<<i<<endl;
				}
			}
			
		}
		if(flag==0)
		cout<<"No crymeth today"<<endl;
	}
	}
    return 0;
}
