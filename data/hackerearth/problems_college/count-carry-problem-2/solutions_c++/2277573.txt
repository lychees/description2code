#include <iostream>
using namespace std;

int main()
{
long int A,B;
int i,j,carry,n,num;

cin>>n;

while(n)
{
carry=num=0;
cin>>A>>B;

while(A || B)
{
i=  A%10;
j=  B%10;

if((i+j+carry)>9)
num++;

carry=(i+j)/10;

A/=10;
B/=10;
}

if(!num)
cout<<"No carry operation\n";
else if(num==1)
cout<<"1 carry operation";
else
{

cout<<num<<" carry operations";
cout<<"\n";
}
n--;
}



    return 0;
}
