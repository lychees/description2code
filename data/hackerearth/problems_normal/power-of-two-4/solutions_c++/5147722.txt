using namespace std;
typedef long long int lli;
typedef unsigned long long int ulli;
int N;
lli Total_M = 0;
bool Pow_2(int n)
{
    int m = log2(n);
    if(1<<m == n)
        return true;
    return false;
}
bool Solve(vector<lli> &Arr)
{
    if(Arr.size() == 1)
        if(Pow_2(Arr[0]))
            return true;
    for(int i = 0; i < 32; i++)
    {
        lli Total = Total_M;
        for(int j = 0; j < Arr.size(); j++)
            if(Arr[j] & (1<<i))
                Total = Total & Arr[j];
        if(Pow_2(Total))
            return true;
    }
    return false;
}
int main()
{
    //freopen("1.txt", "r", stdin);
    //freopen("2.txt", "w", stdout);
    int test;
    cin >> test;
    for(int i = 0; i < 32; i++)
        Total_M += (1<<i);
    while(test--)
    {
        cin >> N;
        vector<lli> Arr;
        lli Rec;
        for(int i = 0; i < N; i++)
        {
            cin >> Rec;
            if(Rec)
               Arr.push_back(Rec);
        }
        if(Solve(Arr))
            cout << "YES" << endl;
        else
            cout << "NO" << endl;
    }
    return 0;
}
