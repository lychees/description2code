#include<iostream>
#include<math.h>
#include<vector>
using namespace std;

int main()
{	long long unsigned t,i;
	long long unsigned c[1000000];
	c[0]=1;
	c[1]=1;
	c[2]=1+c[1];
	c[3]=1+c[2]+1;
	for(i=4;i<1000000;i++)
	{	c[i]=(c[i-1]+c[i-2]+c[i-3])%1000000007;
	}
	cin>>t;
	while(t--)
	{	long long unsigned n;
		cin>>n;
		cout<<c[n]<<"\n";
	}	
	
			
	return 0;
}
