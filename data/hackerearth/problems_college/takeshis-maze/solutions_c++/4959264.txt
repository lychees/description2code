#include <iostream>
using namespace std;
long long int dp[200][200];
long long int n;
long long int mat[200][200];
long long int dfs(long long int i,long long int j)
{
	long long int temp;
	if(dp[i][j]==-1)
	{
		if(mat[i+1][j]!=-1)
		{
			dp[i][j]=mat[i][j]+dfs(i+1,j);
		}
		if(mat[i][j+1]!=-1)
		{
			temp=mat[i][j]+dfs(i,j+1);
			if(temp>dp[i][j])
			dp[i][j]=temp;
		}
	}
	return dp[i][j];
}
int main()
{
    long long int t,c,i,j;
    long long int temp;
    cin>>t;
    while(t--)
    {
    	cin>>n>>c;
    	for(i=0;i<n;i++)
    	{
    		for(j=0;j<n;j++)
    		{
    			cin>>mat[i][j];
    			dp[i][j]=-1;
    		}
    	}
		dp[n-1][n-1]=mat[n-1][n-1];
    	for(i=0;i<n;i++)
    	mat[n][i]=mat[i][n]=-1;
    	temp=dfs(0,0);
    	if(temp<c)
    	cout<<"-1\n";
    	else
    	cout<<temp-c<<"\n";
    }
    return 0;
}
