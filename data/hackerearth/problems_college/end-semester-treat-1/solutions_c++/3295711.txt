#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    int n;
    cin >> n;
    long long int avail[n];
    for(int i=0;i<n;i++)
    {
        cin >> avail[i];            
    }
    int m;
    cin >> m;
    int alloc[m][n];
    for(int i=0;i<m;i++)
        for(int j=0;j<n;j++)
            cin >> alloc[i][j];         

    int max_claim[m][n];
    for(int i=0;i<m;i++)
        for(int j=0;j<n;j++)
            cin >> max_claim[i][j];     

    bool running[n];
    for(int i=0;i<n;i++)
        running[i] = 1;             

    int count=m;
    bool flag;
    int need[m][n];
    for(int i=0;i<m;i++)
        for(int j=0;j<n;j++)
            need[i][j] = max_claim[i][j] - alloc[i][j];      

    while(count--)
    { 
        flag = false;
        for(int i=0;i<m;i++)
        {
            for(int j=0;j<n;j++)
            {
                if(avail[j] < need[i][j])           
                {
                    flag=true;
                    break;
                }
            }

            if(!flag)
            {
                for(int j=0;j<n;j++)
                {
                    avail[j] = avail[j] + alloc[i][j];      
                    running[i] = false;
                }

            }
        }
    }

    flag = false;
    for(int i=0;i<m;i++)
    {
        if(running[i] == true)                  
            flag= true;
    }

    if(!flag)
        cout << "YES" << endl;
    else
        cout << "NO" << endl;
    }

    return 0;
}
