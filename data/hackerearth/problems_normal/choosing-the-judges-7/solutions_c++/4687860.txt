#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    scanf("%d",&t);
    int x=t;
    while(t--){
    	int n,i,a[100000],s=0,s1=0,j=0,dp[100000];
    	scanf("%d",&n);
    	for(i=0;i<n;i++)
    		scanf("%d",&a[i]);
    		dp[0]=a[0];
    		dp[1]=max(a[0],a[1]);
    	for(i=2;i<n;i++)
    	{
    		dp[i]=max(dp[i-1],dp[i-2]+a[i]);
    		
    	}
    	cout<<"Case "<<(x-t)<<": "<<dp[n-1]<<endl;
    	
    }
    
    
    return 0;
}
