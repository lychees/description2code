#include<bits/stdc++.h>

// i'm never gonna give up
 
using namespace std;
 
 
int main()
{
    int t,start,target,mul,div,add,sub,steps,temp,ans,n;
	cin>>t;
	while(t--)
	{
		
        int bfs[1000009];
        memset(bfs,-1,sizeof(bfs));
		queue <int> que;        
		cin>>start>>target;
		cin>>mul>>div>>add>>sub;        
		que.push(start);
		bfs[start]=0;
		ans=-1;
		
		while(!que.empty()){
			
			//cout<<que.front()<<endl;
				
			n=que.front();
			que.pop();
				
			if(n==target){
				ans=bfs[n];
				break;
			}	
			
			temp=n*mul;
			if(temp>=2 && temp<=1000000 && bfs[temp]==-1){
				bfs[temp]=bfs[n]+1;
				que.push(temp);										
			}
				
			temp=n/div;
			if(temp>=2 && temp<=1000000 && bfs[temp]==-1){
				bfs[temp]=bfs[n]+1;
				que.push(temp);										
			}
				
			temp=n+add;
		    if(temp>=2 && temp<=1000000 && bfs[temp]==-1){
				bfs[temp]=bfs[n]+1;
				que.push(temp);										
			}
				
			temp=n-sub;
			if(temp>=2 && temp<=1000000 && bfs[temp]==-1){
				bfs[temp]=bfs[n]+1;
				que.push(temp);										
			}
			
		}
		cout<<ans<<endl;
	}
	return 0;
}
