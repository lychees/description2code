#include<bits/stdc++.h>
using namespace std;
int main(){
	int n;
	cin>>n;
	long long int x[n+1],y[n+1],sum = 0,f = n-1,s = 0;
	for(int i=0;i<n;i++)
	cin>>x[i]>>y[i];
	sort(x,x+n);
	sort(y,y+n);
	for (int i=0;i<n;i++){
		sum +=  ((s*x[i] - f*x[i])%1000000007 + (s*y[i] - f*y[i])%1000000007)%1000000007;
		f--;s++;
	}
	cout<<sum%1000000007<<"\n";
	return 0;
}
