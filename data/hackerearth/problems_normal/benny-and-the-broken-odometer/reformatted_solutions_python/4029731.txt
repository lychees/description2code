q = [0, 1]
for i in range(1, 10):
    q.append(q[len(q)-1]*9+(10**i))
# print q
for _ in range(int(raw_input())):
    n = input()
    r = n
    c = 0
    n = list(str(n))
    k = len(n)-1
    for i in range(len(n)):
        n[i] = int(n[i])
        if n[i] > 3:
            c += (n[i]-1)*q[k]+10**k
        else:
            c += n[i]*q[k]
        k -= 1
