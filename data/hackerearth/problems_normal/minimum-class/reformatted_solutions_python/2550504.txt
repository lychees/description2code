def minDays(n):
    if n in [0, 1, 2]:
        return n
    else:
        x = int(pow(n, 0.5))
        if x*x == n:
            return x+x-1
        else:
            ans = x+x
            if n <= x*x + x:
                return ans
            else:
                return ans+1


tc = int(raw_input())
while tc != 0:
    tc -= 1
    n = int(raw_input())
    print minDays(n)
