#include <iostream>
using namespace std;

int main()
{
     int t;
    scanf("%d",&t);
    while(t--)
    {
    	int hash[255]={0};
    	string h;
    	cin>>h;
    	
    	int i;
    	for(i=0;i<h.length();i++)
    	{
    		hash[h[i]]++;
    	}
    	
    
    	int count=0;
    	int flag=0;
    	for(i=0;i<255;i++)
    	{ 

    		if(hash[i]%2!=0)
    		{
    			if(flag==0)
    			{
    				flag=1;
    			}else
    			{
    				count=count+ hash[i]%2;
    			}
    		}
    		
    	}
    	printf("%d\n",count);
    	
    }
    return 0;
}
