 #include<bits/stdc++.h>
using namespace std;
#define ll long long
#define li long int 

int main()
{
	ll t;
	cin>>t;
	while(t--)
	{
		ll n,i;
		cin>>n;
		ll a[n],b[n];
		for(i=0;i<n;i++)
		cin>>a[i];
		for(i=0;i<n;i++)
		{
			cin>>b[i];
		}
		ll ans1=0;
		for(i=0;i<n;i++)
		{
			ll temp=-1;
			ll l=0,r=n-1;
			while(l<=r)
			{
				ll mid=(l+r)/2;
				if(b[mid]>=a[i])
				{
					temp=mid;	
					l=mid+1;
				}
				else
				r=mid-1;
			}
			ans1=max(ans1,temp-i);
		}
		cout<<ans1<<endl;
	}
	
	return 0;
}
