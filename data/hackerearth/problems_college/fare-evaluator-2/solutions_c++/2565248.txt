#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int d,w;
    	cin>>d>>w;
    	int res=50;
    	if(d<=60){
    	d-=2;
    	if(d>0)
    	{
    		res+=(d*12);
    	}
    	res+=(w*2);
    	cout<<res<<endl;
    	}else
    	cout<<"NA\n";
    }
    return 0;
}
