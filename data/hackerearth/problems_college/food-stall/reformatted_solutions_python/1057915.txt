t = int(input())
while t > 0:
    count = 0
    a = map(int, raw_input().split())
    list(a)
    l = []
    for i in range(a[0]):
        n = int(input())
        l.append(n)
    l.sort()

    for i in range(len(l)):
        j = i+1
        while j < len(l):
            if abs(l[i]-l[j]) == a[1]:
                count += 1
            j += 1
    print(count)
    t -= 1
