#include <bits/stdc++.h>
#define MAX 15

using namespace std;

int n,tc;
int val[15];
int vis[15][1<<15];
int dp[15][1<<15];

int f(int idx, int mask)
{
	if ( idx == n ) return 1;
	if ( vis[idx][mask] == tc ) return dp[idx][mask];
	vis[idx][mask] = tc;
	int ans = f(idx+1, mask);
	if ( (mask & val[idx]) == 0 ) ans += f(idx + 1, mask | (1<<idx)); 
	dp[idx][mask] = ans;
	return ans;
}

int main()
{
	int t,p,x,y;
	cin >> t;
	for ( tc = 1; tc <= t; tc++ ) {
		cin >> n >> p;
		for ( int i = 0; i < n; i++ ) val[i] = 0;
		while ( p-- ) {
			cin >> x >> y;
			x--, y--;
			val[x] |= (1<<y);
			val[y] |= (1<<x);
		}
		cout << f(0,0) - 1 << endl;
	}
	return 0;
	
}
