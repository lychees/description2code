#include<iostream>
using namespace std;
void pro()
{
ios::sync_with_stdio(false);
int t,n,i,max;
char ans;
int *c= new int [26];
bool *e= new bool [26];
for(i=0 ;i<26; i++)
c[i]=0;
cin>>t;
char *s= new char [101];
for( ; t; t--)
{
cin>>n;
for( ; n; n--)
{
cin>>s;
for(i=0; i<26; i++)
    e[i]=false;
for(i=0; s[i]; i++)
    if(!(e[s[i]-'a']))
{
    e[s[i]-'a']=true;
c[s[i]-'a']++;
}
}
for(i=0,max=-1; i<26; i++)
if(max<c[i])
{
max=c[i];
ans=i+'a';
}
for(i=0; i<26; i++)
c[i]=0;
cout<<ans<<'\n';
}
delete s;
delete c;
delete e;
}
int main()
{
pro();
return 0;
}
