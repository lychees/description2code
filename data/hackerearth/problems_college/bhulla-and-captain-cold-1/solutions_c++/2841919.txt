#include <bits/stdc++.h>
#define MAX 1000000000000
using namespace std;

int a[1000006][5],n;
long long dp[1000006][5];

long long fun(int i , int j) {
	
	if(i==n)
		return 0;
	if(dp[i][j]+1)
		return dp[i][j];
	
	long long ans = MAX;
	
	for(int k=0;k<3;k++)
		if(k!=j)
			ans = min(ans,fun(i+1,k)+a[i][k]);
	
	return dp[i][j]=ans;
	
}

int main()
{
    int t;
    cin >> t;
    while(t--) {
    	memset(dp,-1,sizeof(dp));
    	scanf("%d",&n);
    	for(int i=0;i<n;i++)
    		for(int j=0;j<3;j++)
    			scanf("%d",&a[i][j]);
    	long long ans = MAX;
    	for(int i=0;i<3;i++)
    		ans = min(ans, fun(0,i));
    	printf("%lld\n",ans);
    	
    }
    return 0;
}
