#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define MOD 1000000007
int dp[1001][1001];
int main()
{
    int t,n,m;
    cin>>t;
    while(t--){
    	cin>>n>>m;
    	memset(dp,0,sizeof(dp));
    	dp[0][0]=1;
    	for(int i=1;i<=n;++i){
    		dp[0][i]=0;
    		dp[i][0]=1;
    	}
    	for(int i=1;i<=n;++i){
    		for(int j=1;j<=n;++j){
    			if(i<j*m){
    				continue;
    			}
    			dp[i][j]=(dp[i-1][j]+dp[i][j-1])%MOD;
    		}
    	}
    	int res=0;
    	for(int i=0;i<=n && (n-i)>=i*m;++i){
    		if(n-i>i){
    			res=(res+dp[n-i][i])%MOD;
    		}
    	}
    	cout<<res<<endl;
    }
    
    
    return 0;
}
