#include<iostream>
#include<algorithm>
using namespace std;
#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<math.h>

int main()
{
    char a[200][16],b[200][16];
    long long n,t,s,l[200];
    cin>>n;
    for(int i=0;i<n;i++)
    {
        scanf("%s",&a[i]);
        l[i]=strlen(a[i]);
    }
    for(int i=0;i<n;i++)
    {

        for(int j=l[i]-1;j>=0;j--)
            b[i][j]=a[i][l[i]-j-1];
        b[i][l[i]]='\0';
    }
   for(int i=0;i<n-1;i++)
   {
       for(int j=i+1;j<n;j++)
       {
         
           if(strcmp(a[i],b[j])==0)
           {
               cout<<l[i]<<" "<<a[i][l[i]/2]<<"\n";
             
           }
       }
   }
 
}
