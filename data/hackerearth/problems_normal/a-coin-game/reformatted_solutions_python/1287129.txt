used = [0 for i in range(1000001)]
pair = [-1 for i in range(1000001)]
dif = 2
pair[0] = 0
pair[1] = 2
pair[2] = 1
for i in range(3, 1000001):
    if used[i] == 0:
        if i+dif > 1000000:
            break
        pair[i] = i+dif
        pair[i+dif] = i
        used[i+dif] = 1
        dif += 1
for _ in range(input()):
    a, b = map(int, raw_input().split())
    if pair[a] == b:
        print "Don't Play"
    else:
        print "Play"
