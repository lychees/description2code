#include <iostream>
#include <utility>
#include <algorithm>
#define ul unsigned int
#define P pair <ul,ul>
using namespace std;

bool cmp(P a, P b)
{
	if(a.first > b.first)
		return 1;
	if(a.first == b.first && a.second < b.second)
		return 1;
	return 0;
}

int main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);
	cout.tie(0);
    ul n,x,k,i;
    cin >> n >> k;
    P p[n];
    for (i = 0; i < n; i++)
    {
    	cin >> x;
    	p[i] = make_pair(x % k,x);
    }
    sort(p,p+n,cmp);
    for (i = 0; i < n; i++)
      cout << p[i].second << " ";
    return 0;
}
