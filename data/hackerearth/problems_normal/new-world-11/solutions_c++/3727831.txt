#include <bits/stdc++.h>
using namespace std;

int a[1000001];
int n,k;

int check(int jump) {
    int i,ans=0,j=0,j1=0,flag=0;
    for(i=0;i<n;i++) {
        
        while(j<n&&a[j]-a[j1]<=jump) {
             j++;
        }
        ans++;
        j1=j-1;
        if(j==n){ flag=1;  break;}
    }
    
    if(ans<=k&&flag==1) return 1;
    return 0;
    
}

int main() {
    
    int t;
    scanf("%d",&t);
    while(t--) {
        scanf("%d%d",&n,&k);
        int i;
        for(i=0;i<n;i++) scanf("%d",&a[i]);
        
        int low=0,high=1000000000,mid;
        
       while(low<high) {
            
            mid=(low+high)/2;
            
            if(check(mid)) high=mid;
            else low=mid+1;
            
            
        }
        printf("%d\n",high);
    }
    	
	return 0;
}
