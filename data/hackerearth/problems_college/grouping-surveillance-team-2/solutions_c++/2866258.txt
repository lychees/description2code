#include<bits/stdc++.h>
using namespace std;

long long int dp[205];

int main()
{
    for(int i=0; i<205; i++)
        dp[i] = 0;
    dp[0] = 1;

    for(int j=2; j<=200; j++)
        for(int i=j; i<=200; i++)
            dp[i] += dp[i-j];

    int t, no;
    scanf("%d", &t);
    while(t--)
    {
        scanf("%d", &no);
        printf("%lld\n", dp[no]);
    }
    return 0;
}
