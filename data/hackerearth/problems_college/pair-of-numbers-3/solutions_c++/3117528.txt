#include <iostream>
using namespace std;

int main()
{int t;
    //cout << "Hello World!" << endl;
    cin>>t;
    while(t--){
    	int n,flag=0;
    	cin>>n;
    	for(int i=1;i<n;i++){
    		for(int j=i+1;j<=n;j++){
    			if(((i*j)%(i+j))==0){
    				flag++;
    			}
    		}
    	}
    	cout<<flag<<endl;
    }
    return 0;
}
