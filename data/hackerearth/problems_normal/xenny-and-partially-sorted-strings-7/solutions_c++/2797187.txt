#include <bits/stdc++.h>
#include <stdio.h>

using namespace std;
int comp_len;

bool comp(string a, string b){
	for(int i=0;i<comp_len;i++){
		if(a[i]<b[i])
			return true;
		if(a[i]>b[i])
			return false;
		else
			continue;
	}
	return true;
}

int main()
{
	std::ios::sync_with_stdio(false);
    int tc;
    cin>>tc;
    while(tc--){
    	int n,k,m;
    	cin>>n>>k>>m;
    	comp_len=min(n,m);
    	vector<string>ip(n);
    	for(int i=0;i<n;i++)
    		cin>>ip[i];
    	sort(ip.begin(),ip.end(),comp);
    	cout<<ip[k-1]<<endl;
    }
    return 0;
}
