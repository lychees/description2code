#include<stdio.h>
int main(){
	int t, n, cp, i; scanf("%d", &t);
	while(t-- && scanf("%d", &n)){
		if(n == 1)	printf("-1\n");
			else if(n % 3 == 0){
				for(i = 1; i <= n; i++)	printf("5");
				printf("\n");
			}
				else{
					cp = n;
					while(n--){
						if(n % 3 == 0 && (cp - n) % 5 == 0)	break;
					}
					for(i = 1; i <= n; i++)	printf("5");
					for(i = 1; i <= cp - n; i++)	printf("3");
					printf("\n");
				}
	}		
	return 0;
}
