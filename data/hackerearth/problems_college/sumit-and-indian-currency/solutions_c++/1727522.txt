#include <bits/stdc++.h>
#define ll long long int

using namespace std;

int main()
{
	int t,i,j,c,l;
	char s[100010],s1[100010];
	cin>>t;
	while(t--)
	{
		scanf("%s",s);
		l=strlen(s);
		if(l>3)
		{
		for(i=0;i<3;i++)
			s1[i]=s[l-i-1];
		s1[i]=',';
		j=i+1;
		c=0;
		for(i=3;i<l;i++)
		{
			s1[j++]=s[l-i-1];
			c++;
			if(c==2)
			{
				c=0;
				s1[j++]=',';
			}
		}
		s1[j]='\0';
		l=strlen(s1);
		if(s1[l-1]==',')
			l--;
		for(i=l-1;i>=0;i--)
			printf("%c",s1[i]);
		}
		else
			printf("%s",s);
		cout<<endl;
	}
	return 0;
}
