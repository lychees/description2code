#include <bits/stdc++.h>
using namespace std;

int diff(string str, string sub)
{
	int len1 = str.length();
	int len2 = sub.length();
	for(int i=0;i<len1-len2+1;i++)
	{
		string temp = str.substr(i,len2);
		bool ch = true;
		int j = 0;
		while(!isalpha(sub[j]))
            j++;
		int d = temp[j] - sub[j];
		if(d < 0)
            d += 26;
		for(j=j+1;j<len2;j++)
		{
			if(!isalpha(sub[j]))
                continue;
			int s = temp[j] - sub[j];
            if(s < 0)
                s += 26;
			if(s != d)
			{
				ch = false;
				break;
			}
		}
		if(ch == true)
			return d;
	}
	return -1;
}

int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		string str, sub;
		cin >> str >> sub;
		int d = diff(str,sub);
		for(int i=0;i<str.length();i++)
        {
            if(!isalpha(str[i]))
                cout << str[i] << flush;
            else if(str[i] >= 'a' && str[i] <= 'z')
            {
                char c = 'a' + (str[i] - 'a' - d + 26)%26;
                cout << c << flush;
            }
            else if(str[i] >= 'A' && str[i] <= 'Z')
            {
                char c = 'A' + (str[i] - 'A' - d + 26)%26;
                cout << c << flush;
            }
        }
        cout << endl;
	}
	return 0;
}
