#include <iostream>
using namespace std;

int main()
{
    int tt; cin >> tt;
    while(tt--) {
    	string s; cin >> s;
    	string ans = "";
    	for(int i = 0;i < s.size();i++) {
    		char c = s[i];
    		if(c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u' ||
    		   c == 'A' || c == 'E' || c == 'I' || c == 'O' || c == 'U')
    		   ans += c;
    	}
    	if(ans == "") cout << "No\n";
    	else cout << ans << endl;
    }
    return 0;
}
