#include<iostream>
#include<vector>
#include<set>
#include<algorithm>
#define ll long long
using namespace std;
int n;
ll s;
ll h[100006];
ll pre[100006];
ll g;
ll find(int x)
{
	ll curr=pre[x];
	ll d=x+1;
	ll ng=g-d*(d+1)/2;
	if(curr>=ng)return d;
	ll hi=(2200000000ll)/n + 2;
	ll lo=0;
	while(lo<hi)
	{
		ll m=(lo+hi)>>1;
		d=m*n +x+1;
		ll ng=g-((d*(d+1))/2);
		curr=m*s + pre[x];
		if(curr>=ng)hi=m;
		else lo=m+1;
	}
	return lo*n + x+1;
}
int main()
{
	cin>>n>>g;
	for(int i=0;i<n;++i)cin>>h[i];
	pre[0]=h[0];
	for(int i=1;i<=n;++i)pre[i]=pre[i-1]+h[i];
	s=pre[n];
	ll ans=1000000000000000ll;
	for(int i=0;i<n;++i){ans=min(ans,find(i));}
	cout<<ans<<endl;
	return 0;
}
