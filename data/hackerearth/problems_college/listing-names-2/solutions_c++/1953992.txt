#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	cin>>t;
	string arr[t];
	for(int i=0;i<t;i++)
	{
		cin>>arr[i];
	}
	sort(arr,arr+t);
	for(int i=0;i<t;)
	{
		string k=arr[i];int c=0;
		while(arr[i]==k && i<t)
		{
			c=c+1;
			i=i+1;
		}
		cout<<k<<" "<<c<<endl;
	}
}
