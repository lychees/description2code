/*"People live their lives bound by what they accept as correct and true... that is how they define reality. But what does it mean to be correct or true? Merely vague concepts... their reality may all be an illusion.? */
#include <bits/stdc++.h>

using namespace std;
int has[26];
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
	    	memset(has, 0, sizeof(has));
		string s;
		cin>>s;
		vector < pair <int  ,  int > >v;
		//map< char , int > m;
		for(int i = 0; i < s.size(); i++)
		{
			has[s[i] - 97]++;
		}
		for(int i = 0; i <  26; i++)
		{
		    if( has[i] != 0) 
		    v.push_back(make_pair(has[i], i));
		}
		sort(v.begin(), v.end());
		//cout<<v.size() <<endl;
	/*	for(int  i = 0; i < v.size(); i++)
		{
		    cout<<v[i].first <<" ";
		}*/
		for(int i = 0; i < v.size(); i++)
		{
		    if(v[i].first != 0) {
		    for(int j = 0; j < v[i].first; j++)
		    {
			cout<<char(v[i].second + 97);
		    }
		    }
		}
		cout<<endl;

	}	

    return 0;

}
