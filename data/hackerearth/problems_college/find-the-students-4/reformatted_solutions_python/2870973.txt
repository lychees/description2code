n = int(raw_input())


def sub(a):
    count = 0
    for x in range(n):
        if (a[x] < 2 and a[x] > 0):
            count += 1
            a[x] = 0
        elif (a[x] != 0):
            count += 1
            a[x] -= 2
    print count


str = raw_input().split()
a = []
for c in range(n):
    a.append(int(str[c]))
while (sum(a) != 0):
    sub(a)
