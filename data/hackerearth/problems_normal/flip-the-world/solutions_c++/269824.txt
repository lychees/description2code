#include <iostream>
using namespace std;
#include <bits/stdc++.h>
int main()
{
   int t,n,m;
   cin >> t;
   while(t--) {
   	cin >> n >>m;
   	vector<string> board(n);
   	for(int i = 0;i<n;i++)
   	cin >> board[i];
   	vector<vector<int> > table(n,vector<int> (m,0));
   	int res = 0;
   	for(int i = n-1;i>=0;i--) {
   		for(int j = m-1;j>=0;j--) {
   			if ( i == n-1 && j == m-1)
   			table[i][j] = 0;
   			else if (i == n-1)
   			table[i][j] = table[i][j+1];
   			else if (j == m-1) 
   			table[i][j] = table[i+1][j];
   			else
   			table[i][j] = table[i+1][j] + table[i][j+1] - table[i+1][j+1];
   			
   			if((table[i][j]%2 == 1 && board[i][j] == '1') || (table[i][j]%2 == 0 && board[i][j] == '0'))
   			table[i][j]++;
   			
   		}
   	}
   	cout << table[0][0] << endl;
   }
   return 0;
}
