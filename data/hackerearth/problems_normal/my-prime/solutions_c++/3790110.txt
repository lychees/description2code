#include <iostream>
#include <math.h>
using namespace std;

int main()
{
	int n, s[100],c,i;
	cin>>n;
	if(n<=100){
		
	for(int i=0;i<n;i++){
		cin>>s[i];
		if(s[i]<1 && s[i]>pow(10,6))
		return 0;
	}
	
	for(i=0;i<n;i++){
		c=0;
		for(int j=0;j<n;j++)
			{
				if(s[i]%s[j]==0 && s[j]!=s[i])
			      { c=1;
			      break;
			      }
			      
			}
	if(c==0)
	cout<<s[i]<<" ";
		
	}
	}
	else
	return 0;
   
}
