#include<iostream>
using namespace std;
int main()
{
    long long int test,p,x;
    cin>>test;
    while(test--)
    {
    	cin>>x>>p;
    	int c=0;
    	while(1)
    	{
    		if(x%p!=0)
    		break;
    		else
    		{
    			if(x==p)
    			{x=0;c++;break;}
    			else
    			{
    			x=x/p;c++;}
    	}
    	}
    	cout<<c<<" "<<x<<endl;
    }
    return 0;
}
