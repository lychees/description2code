#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,r,c,n;
    set <string> s;
    string m;
    cin>>t;
    while(t--)
    {
    	s.clear();
    	cin>>r>>c;
    	n=r;
    	while(r--)
    	{
    		cin>>m;
    		s.insert(m);
    	}
    	if(s.size()==n)
    		cout<<"Yes"<<endl;
    	else
    		cout<<"No"<<endl;
    }
    return 0;
}
