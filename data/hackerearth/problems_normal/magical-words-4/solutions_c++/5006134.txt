#include <bits/stdc++.h>
using namespace std;
int main()
{
	long long int t,ans,i,j,k;
	string str;
    cin>>t;
    while(t--)
    {
    	cin>>str;
    	int n=str.length();
    	ans=0;
    	for(i=0;i<str.length();i++)
    	{
    		
    		k=1;
    		for(j=0;j<=n;j++)
    		{
    			if(str[i-j]==str[i+j]&&i-j>=0&&i+j<n)
    			{ans+=k*k;
    			k+=2;
    			}
    			else
    			break;
    		}
    	}
    	
    	for(i=0;i<str.length();i++)
    	{
    		
    		k=0;
    		for(j=0;j<n;j++)
    		{
    			if(str[i-j+1]==str[i+j]&&i-j+1>=0&&i+j<n)
    			{ans+=k*k;
    			k+=2;
    			}
    			else
    			break;
    		}
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
