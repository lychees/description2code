#include<iostream>
using namespace std;
main()
{


int t;
cin>>t;
for(int z=0;z<t;z++)
{unsigned long long int ar[100004];
long long int n;
unsigned long long int x;
cin>>n>>x;
for(int j=0;j<n;j++)
cin>>ar[j];
unsigned long long diff=0;
unsigned long long int count=0;
for(int i=0;i<n-1;i++)
{
if(ar[i]<ar[i+1])
continue;
diff=ar[i]-ar[i+1];
ar[i+1]+=((diff/x)+1)*x;
count=count+(diff/x)+1;

}
cout<<count<<endl;
}
}
