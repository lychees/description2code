#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    
    if (n!=1){
    	for(int j=2; j<=n; j++){
    		int gh = 0;
    		for(int i=2; i<j-1; i++){
    			if (j%i==0){
    				gh=1;
    				break;
    			}
    			
    		}
    		if (gh==1){
    			cout << j << endl;
    		}
    		
    		
    	}
    }
    
}
