#include<bits/stdc++.h>
#define ll long long int
#define md 1000000007
using namespace std;
ll power(ll a,ll b)
{
	ll x=1,y=a;
	while(b>0)
	{
		if(b%2==1)
		{
			x=(x*y)%md;
		}
		y=(y*y)%md;
		b=b/2;
	}
	return x%md;
}
int main()
{
	ll n,q,i,j,arr[100006],q1,id,var,temp,product,val,ans,mid,zero;
	scanf("%lld",&n);
	product=1;
	zero=0;
	for(i=0;i<n;i++)
	{
		scanf("%lld",&arr[i]);
		if(arr[i]!=0)
		{
			product=(product*arr[i])%md;
		}
		else
		{
			zero++;
		}
	}
	scanf("%lld",&q);
	for(i=0;i<q;i++)
	{
		scanf("%lld",&q1);
		if(q1==0)
		{
			scanf("%lld%lld",&id,&val);
			if(!arr[id-1] && val)
			{
				arr[id-1]=val;
				product=(product*val)%md;
				zero--;
				continue;
			}
			if(arr[id-1] && !val)
			{
				product=(product*power(arr[id-1],md-2))%md;
				zero++;
				arr[id-1]=val;
				continue;
			}
			if(arr[id-1] && val)
			{
				product=(product*val)%md;
				product=(product*power(arr[id-1],md-2))%md;
				arr[id-1]=val;
			}
		}
		else
		{
			ans=1;
			scanf("%lld",&id);
			if(!arr[id-1] && zero==1)
			{
				printf("%lld\n",product);
			}
			if(!arr[id-1] && zero>1)
			{
				printf("0\n");
			}
			if(arr[id-1] && zero>0)
			{
				printf("0\n");
			}
			if(arr[id-1] && zero==0)
			{
				ans=(product*power(arr[id-1],md-2))%md;
				printf("%lld\n",ans);
			}
		}
	}
	return 0;
}
