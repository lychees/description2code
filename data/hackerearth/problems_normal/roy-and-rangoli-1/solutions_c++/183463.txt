#include <iostream>
#include <math.h> 
using namespace std;

int main()
{
    long long int n;
    cin>>n;
    long int max=2*(n-1);
    long long int sum=0;
    for(long long int i=2;i<=max;i++)
    {
    	int count=0;
    	for(int j=2;j<=sqrt(i);j++)
    	{
    		if((i%j)==0)
    		{
    			count=1;
    			break;
    		}
    	}
    	if(count!=1)
    	{
    		if(i<n)
    		 sum=sum+i+1;
    		else
    		 sum=sum+max-i+1;
    	}
    }
    cout<<sum%1000000007<<"\n";
    return 0;
}
