#include <iostream>
using namespace std;

int main()
{
	int n;
	cin>>n;
	int a[n];
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
	}
		
	int maxlocal=0,maxglobal=0;
	for(int i=0;i<n;i++)
	{
		maxlocal = max(a[i],maxlocal+a[i]);
		maxglobal = max(maxlocal,maxglobal);
	}
	cout<<maxglobal<<endl;
	
    return 0;
}
