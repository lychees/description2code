#include <iostream>
#include <cstring>
using namespace std;
#include <string.h>
int main()
{
	char str[100];
	int x,len,a[26];
	cin>>x;
	
	for( int i = 0; i < x; i++){
	for(int y = 0; y < 26;y++)
	a[y]=0;
	cin>>str;	
	len=strlen(str);
	int flag=0;
	for(int j = 0; j < len;j++){
		a[str[j]-97]++;
		
	}
	for(int y = 0; y < 26;y++){
		if(a[y]==0){
		flag=10;break;
		}
	}
	if(flag==10)

	else
	cout<<"YES"<<endl;
	}

}
