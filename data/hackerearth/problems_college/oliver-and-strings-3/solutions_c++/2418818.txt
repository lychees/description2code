#include <bits/stdc++.h>

using namespace std;

int score(char ch) {
	// Return the score of char in string
	if (ch == '$')	return 100;
	return int(tolower(ch)) - 96;
}

int main()
{
	ios_base::sync_with_stdio(0);
	long long t;
	cin >> t;
	while (t--) {
		string s;
		cin >> s;
		long long n = s.length();
		long long inside_score = 0; //stores the value all elements in string in between which is under process
		for (long long i=1;i<n-1;i++)	inside_score += score(s[i]);
		long long total = 100;	//stores the actual score which is output. Initially add s[0] value i.e. 100
		long long index = 1;	//used for indexing
		while (s[index] != '$') {	// loop until s[index] != '$'
			total += inside_score + 100;	//100 is the score of s[n-index-1]
			char note = s[index];
			inside_score -= score(note);	//subtract the score of value which changes to '$'
			if (note == s[n-index-1]) {
				total += inside_score + 100;
				inside_score -= score(s[n-index-1]);
				s[n-index-1] = '$';
			} else {
				total += score(s[n-index-1]);
				break;
			}
			index += 1;
		}
		cout << total << endl;
	}
	return 0;
}
