#include <iostream>
using namespace std;
string str="aeiouy";
bool isVowel(char c){
	for(int i=0;i<6;i++){
		if(c==str[i])
		return true;
	}
	return false;
}
int main()
{
    int t;
    string s;
    cin>>t;
    while(t--){
    	int vow=0,cons=0,consecutive=0;
    	cin>>s;
    	for(int i=0;i<s.length();i++){
    		if(isVowel(s[i])){
    		vow++;consecutive=0;}
    		else{
    		cons++;
    		consecutive++;
    		if(consecutive==3){
    			cout<<"hard"<<endl;break;
    		}
    	}
    }
    if(consecutive!=3){
    	if(cons>vow)
    	cout<<"hard"<<endl;
    	else
    	cout<<"easy"<<endl;
    }
    }
    return 0;
}
