a = int(raw_input())


def fact(m):
    j = 1
    t = 1
    for i in range(0, m):
        t = t*j
        j = j+1
    return t


for i in range(a):
    l = raw_input()
    c = l.split(' ')
    m = int(c[0])
    n = int(c[1])
    if n-m > 1:
        print -1
    else:
        ans = ((fact(m)*fact(m+1))/fact(m-n+1)) % 1000000007
        print ans
