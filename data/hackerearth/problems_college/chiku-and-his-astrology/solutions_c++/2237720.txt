#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n,i,j;
    	char a[200010];
    	scanf("%s",a);
    	n=strlen(a);
    	int flag=1;
    	while(flag==1)
    	{
    	flag=0;	
    	for(i=0;i<n-1;i++)
    	{
    		if(a[i]=='1')
    		{
    			j=i+1;
    			while(a[j]=='x')
    			{
    				j++;
    			}
    			if(a[j]=='0')
    			{
    				a[i] = 'x';
    				a[j] = 'x';
    				flag=1;
    			}
    		}
    		if(a[i]=='0')
    		{
    			j=i+1;
    			while(a[j]=='x')
    			{
    				j++;
    			}
    			if(a[j]=='1')
    			{
    				a[i] = 'x';
    				a[j] = 'x';
    				flag=1;
    			}
    		}
    	}
    	}
    	int c=0;
    	for(i=0;i<n;i++)
    	{
    		if(a[i]!='x')
    		{
    			c++;
    		}
    	}
    	printf("%d\n",c);
    }
    return 0;
}
