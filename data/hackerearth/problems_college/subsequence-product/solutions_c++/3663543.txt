#include <iostream>
using namespace std;
long int a[10005];
int main()
{
	int t,n,flag,i;
	long int h;
	long long int p;
	cin>>t;
	while(t--)
	{
		cin>>n>>h;
		p=1;
		for(i=0;i<n;i++)
		{
			cin>>a[i];
		}
		p=1;
		for(i=0;i<n;i++)
		{
			p*=a[i];
			p=p%h;
		}
		if(p==0)
		{
			cout<<"YES"<<endl;
		}
		else
		{
			cout<<"NO"<<endl;
		}
	}
    return 0;
}
