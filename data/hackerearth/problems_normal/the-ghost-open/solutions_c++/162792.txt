#include<iostream>
#include<cstring>

using namespace std;


int main()
{
	int T;
	cin>>T;
	int i, j, k;
	int N;
	int ans[10];
	int ghost;
	int matches;
	//int bits[30] = {0};
	int q =0;
	int value_zeroes;
	int two_n_one=0;
	int rem=0;
	int t=0;
	for(i=0; i<T; i++ )
	{
		cin>>N;
		t=1;
		matches =0;
		q = N/2;
		rem = N%2;
		two_n_one = 2;
		if( rem == 0 )
			value_zeroes = t;
		else
			value_zeroes = 0;
		while( q !=0 )
		{
			//cout<<q<<" "<<value_zeroes<<" "<<rem<<endl;
			rem = q%2;
			q = q/2;
			
			t = two_n_one;
			two_n_one = two_n_one*2;
			if( rem==0)
				value_zeroes = value_zeroes + t;

		}
		
		matches = two_n_one - 2 - value_zeroes;
		if(N==0)
			matches =0;
		ans[i] = matches;
	}
	for ( i=0; i<T; i++ )
		cout<<ans[i]<<endl;
	return 0;
}
