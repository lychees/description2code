#include <bits/stdc++.h>
using namespace std;

int t,l,n,p,a[10],b[10];
void func(int id, int c) {
	id%=10;
	if(b[id]>c) b[id]=c;
	else return;
	for(int i=0;i<10;++i) func(id+i,c+a[i]);
}
int main()
{
   string s;
   cin>>t;
   while(t--) {
   		for(int i=0;i<10;++i) cin>>a[i];
    	for(int i=0;i<10;++i) b[i]=INT_MAX;
    	for(int i=0;i<10;++i) {
    		func(i,a[i]);
    	}
    	cin>>l;
   		cin>>s;
   		int ans=0;
   		for(int i=0;i<l;++i) ans+=b[s[i]-'0'];
   		cout<<ans<<endl;
   }
}
