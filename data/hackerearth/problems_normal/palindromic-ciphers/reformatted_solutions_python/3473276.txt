import string

alps = {}
lows = string.lowercase
for i in range(len(lows)):
    alps[lows[i]] = i + 1


def calcProduc(st):
    prod = 1
    for i in st:
        prod *= alps[i]

    return prod


def isPalindrome(st):
    l = len(st)
    subl = l // 2
    m = l % 2
    first = st[:subl]
    end = st[(subl + m):]
    if first != end[::-1]:
        return False
    return True


t = input()

for i in range(t):
    st = raw_input().strip()
    if isPalindrome(st):
        print "Palindrome"
    else:
        print calcProduc(st)
