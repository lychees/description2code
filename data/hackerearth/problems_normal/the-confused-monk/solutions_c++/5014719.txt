#include<iostream>
#define M 1000000007
using namespace std;
int gcd(int a,int b)
{
	if(a%b==0)
		return b;
	else 
		return gcd(b,a%b);
}
int fast_pow(long long base,long long n)
{
	if(n==0)
		return 1;
	if(n==1)
		return base;
	long long halfn=fast_pow(base,n/2);
	if(n%2==0)
		return(halfn*halfn)%M;
	else
		return(((halfn*halfn)%M)*base)%M;
}
int main()
{
	int n;
	cin>>n;
	int a[n],i;
	for(i=0;i<n;i++)
		cin>>a[i];
	int g=a[0];
	for(i=0;i<n;i++)
		g=gcd(g,a[i]);
	long long mul=1;
	for(i=0;i<n;i++)
		mul=(mul*a[i])%M;
	long long ans=fast_pow(mul,g);
	cout<<ans<<endl;
	return 0;
}
