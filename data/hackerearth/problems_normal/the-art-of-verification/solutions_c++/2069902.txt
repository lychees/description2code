#include <bits/stdc++.h>
using namespace std;
void prefixfunction(string pat, int m,int p[])
{
	p[0]=0;
	int i=1,j=0;
	while(i<m)
	{
		if(pat[i]==pat[j])
		{
			p[i]=j+1;
			i++;
			j++;
		}
		else
		{
			if(j!=0)
			{
				j=p[j-1];
			}
			else
			{
				p[i]=0;
				i++;
			}
		}
	}
}
int kmp(string pat, string txt)
{
	int n=txt.size();
	int m=pat.size();
	int p[m];
	prefixfunction(pat,m,p);
	int i=0,j=0;
	while(i<n)
	{
		if(pat[j]==txt[i])
		{
			i++;
			j++;
			if(j==m)
				return (i-j);
		}
		else
		{
			if(j!=0)
			{
				j=p[j-1];
			}
			else
			{
				i++;
			}
		}
	}
}
int main()
{
    string str;
    cin>>str;
    string temp[5];
    int t[5];
    temp[0]="username";
    temp[1]="pwd";
    temp[2]="profile";

    temp[4]="key";
    t[0]=temp[0].size();
    t[1]=temp[1].size();
    t[2]=temp[2].size();
    t[3]=temp[3].size();
    t[4]=temp[4].size();
    int j;
    int pos[6];
    for(int i=0;i<=4;i++)
    {
    	pos[i]=kmp(temp[i],str);
    }
    pos[5]=str.size()+1;
    for(int i=0;i<=4;i++)
    {
    	cout<<temp[i];

    	int k=pos[i]+t[i]+1;
    	for(j=k;j<pos[i+1]-1;j++)

    }
}
