#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		unsigned long long int n;
		scanf("%llu",&n);
		long long int c =0 ;
		while( n != 0 )
		{
			if( n %2 ==1)
			{
				c++;
				n = n/2;
			}
			else
			{
				n = n/2;
			}
		}
		printf("%lld\n",c);
	}
}
