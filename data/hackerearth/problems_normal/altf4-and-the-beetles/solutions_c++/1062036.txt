#include <iostream>
#include <set>
#include <map>

using namespace std;

int main() {
	int t; scanf("%d", &t);
	while (t--) {
		int n, a, b, c; scanf("%d%d%d%d", &n, &a, &b, &c);
		map<int, int> pp;
		map<int, int> qq;
		set<int> ev;
		
		for (int i = 0; i < n; i++) {
			int p, q; scanf("%d%d", &p, &q);
			ev.insert(p);
			ev.insert(q+1);
			pp[p]++;
			qq[q+1]++;
		}
		
		long long best = n * (long long)a;
		long long curr = best;
		for (set<int>::iterator it(ev.begin()); it != ev.end(); it++) {
			int evp = pp[*it];
			int evq = qq[*it];
			curr -= evp * (long long)a;
			curr += evp * (long long)b;
			curr -= evq * (long long)b;
			curr += evq * (long long)c;
			if (curr > best) best = curr;
		}
		cout << best << endl;
	}
    return 0;
}
