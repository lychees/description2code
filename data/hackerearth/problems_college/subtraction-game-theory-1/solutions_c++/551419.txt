#include <iostream>
#include <stdio.h>
#define ll long long
using namespace std;
void solve()
{
	ll n,k;
	scanf("%lld%lld",&n,&k);
	int temp = 1;
	int flag = 1;
	temp  = (n-1)/(k+1);
	if(temp*(k+1) == (n-1))
		flag = 0;
	if(flag)
		printf("Rani\n");
	else
		printf("Nandu\n");
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		solve();
	}
}
