#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define mod 1000000007
int main()
{
    int t,i;
    ll ans,cal;
    vector<ll> v(3);
    scanf("%d",&t);
    while(t--)
    {
    	ans=1;
    	for(i=0;i<3;i++)
    	{
    		scanf("%lld",&(v[i]));
    	}
    	sort(v.begin(),v.end());
    	ans=v[0]%mod;
    	cal=(v[1]-1)%mod;
    	cal=(cal+mod)%mod;
    	ans=(ans*cal)%mod;
    	//ans=(ans+mod)%mod;
    	cal=(v[2]-2)%mod;
    	cal=(cal+mod)%mod;
    	ans=(ans*cal)%mod;
    	//ans=(ans+mod)%mod;
    	printf("%lld\n",ans);
    }
    return 0;
}
