#include <iostream>
#include <cstdio>
#include <vector>
#include <functional>

using namespace std;
#define sp ' '
#define endl '\n'
#define fi first
#define se second
#define mp make_pair

int f[105][10005];
pair<int, int> a[105];
int max_ele = 0;
int n, p;
int ferti = 2e9;
int main()
{
	int T;
	cin >> T;
	while (T--)
	{
		max_ele = 0;
		ferti = 2e9;
		memset(f, 0, sizeof(f));
		cin >> n >> p;
		for (int i = 1; i <= n; i++)
			cin >> a[i].fi >> a[i].se;
		for (int i = 1; i <= n; i++)
		{
			for (int j = 0; j <= p; j++)
			{
				f[i][j] = f[i - 1][j];
				if (a[i].se <= j)
					f[i][j] = max(f[i][j], f[i - 1][j - a[i].se] + a[i].fi);
				if (f[i][j] + p - j > max_ele)
				{
					max_ele = f[i][j] + p - j;
					ferti = j;
				}
				else if (f[i][j] + p - j == max_ele)
					ferti = min(j, ferti);
			}
		}
		if (max_ele == 0)
			cout << 0 << sp << 0 << endl;
		else
			cout << ferti << sp << max_ele << endl;
	}
}
