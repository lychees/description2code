vector< int>tmp;

int check(int num,int k)
{
 int i,j,prev=tmp[0]+num;
k--;
 for(i=1;i<tmp.size();i++)
 {
   if(prev+num>=tmp[i])
    continue;
   if(k==0)
    return 0;
   prev=tmp[i]+num;
   k--;
 }
 return 1;
}
