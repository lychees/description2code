#include <iostream>
using namespace std;

int main()
{
	int key,prev,flag=-1,n,count=0;
	long int arr[100001];
	cin>>n;
	for(int i=0;i<n;i++)
	cin>>arr[i];
   	while(1)
   	
   	{
   	prev=arr[0];
   	
   	flag=-1;
   	for(int i=1;i<n;i++)
   	{
   		key=arr[i];
   		
   		if(key<prev && key!=-1)
   		{arr[i]=-1;
   		flag=0;
   		}
   		
   		if(key!=-1)
   		prev=key;
   		
   	}
	
	if(flag==0)
	count++;
	else
	break;
   	
   	}
	
	
	cout<<count;
    return 0;
}
