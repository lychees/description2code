#include <iostream>
#include <algorithm>
#include <vector>
#include <cstdio>
using namespace std;

const int N = 1000010;
int p[N], alpha[N];

int main() {
  ios_base::sync_with_stdio(false);
  p[0] = p[1] = 1;
  for (int i = 2; i*i < N; i++) if (!p[i]) {
    for (int j = i*i; j < N; j += i) p[j] = 1;
  }
  for (int i = 1; i < N; i++) {
    int x = i, ten = 1, num = 0;
    bool ok = false;
    while (x > 0) {
      int cur = ten*(x % 10) + num;
      if (!p[cur]) ok = true;
      x /= 10;
      num = cur;
      ten *= 10;
    }
    if (ok) alpha[i] = 1;
    alpha[i] += alpha[i - 1];
  }
  int q; scanf("%d", &q);
  while (q--) {
    int l, r; scanf("%d%d", &l, &r);
    int out = alpha[r] - alpha[l - 1];
    printf("%d\n", out);
  }
  return 0;
}
