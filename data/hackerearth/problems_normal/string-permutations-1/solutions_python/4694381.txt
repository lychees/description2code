t=input()
d=dict()
for i in range(t):
    s=raw_input()
    if s not in d:
        d[s]=1
    else:
        d[s]+=1
#print sorted(d,key=d.get,reverse=True)
q=sorted(d,key=d.get,reverse=True)
li=dict()
sum=d[q[0]]
li[q[0]]=d[q[0]]
for i in range(1,len(q)):
	if d[q[i]]==sum:
		li[q[i]]=d[q[i]]
res=0
for i in li:
	res+=li.get(i)
print res
