#include <bits/stdc++.h>
using namespace std;

int main()
{
    string str;
    cin>>str;
    vector<bool>v(4,false);
    int i;
    for( i=0;i<str.length();++i)
    {
    	if(str[i]=='l')
    	{
    			v[0]=true;
    			break;
    		
    	}
    }
    for(;i<str.length();++i)
    {
    	if(str[i]=='o')
    	{
    			v[1]=true;
    			break;
    		
    	}
    }
    for(;i<str.length();++i)
    {
    	if(str[i]=='v')
    	{
    			v[2]=true;
    			break;
    		
    	}
    }
    for(;i<str.length();++i)
    {
    	if(str[i]=='e')
    	{
    			v[3]=true;
    			break;
    		
    	}
    }
    if(v[0]&&v[1]&&v[2]&&v[3])
    	cout<<"I love you, too!";
    else
    	cout<<"Let us breakup!";
    return 0;
}
