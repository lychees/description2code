string = raw_input()
wordDictonary = {}

stringLength = len(string)

startIndex = 0

while (startIndex < stringLength):
    curChar = string[startIndex]
    if (wordDictonary.has_key(curChar)):
        wordDictonary[curChar] += 1
    else:
        wordDictonary[curChar] = 1
    startIndex += 1


maxOccuranceOfAnyChar = max(wordDictonary.values())
maxList = []

for currentChar, count in wordDictonary.items():
    if (count == maxOccuranceOfAnyChar):
        maxList.append(currentChar)


maxList.sort()
print maxList[0], maxOccuranceOfAnyChar
