    #include <bits/stdc++.h>
    #define M 1000000009
    using namespace std;
    long long int fast_pow(long long int a, long long int b)
    {
    long long int res = 1;
    while ( b > 0 )
    {
    if ( b&1 ) 
    res = (res*a)%M;
    a = (a*a)%M;
    b >>= 1;
    }
    return res;
    }
    int main()
    {
    int t;
    long long int n;
    cin >> t;
    while ( t-- )
    {
    cin >> n;
    cout << fast_pow(26LL,n/2+n%2) << endl;
    }
    return 0;
    }
