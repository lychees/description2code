#include <iostream>
#include<string.h>
using namespace std;
int main()
{
    //cout << "Hello World!" << endl;
    int t,i,j;
    char str1[100000],str2[100000];
    cin>>t;
    while(t--)
    {
    	int hash[256]={0};
    	cin>>str1>>str2;
    	int l1=strlen(str1);
    	int l2=strlen(str2);
    	for(i=0;i<l1;i++)
    	{
    		hash[str1[i]]++;
    	}
    	for(j=0;j<l2;j++)
    	{
    		if(hash[str2[j]]>0)
    		break;
    	}
    	if(j==l2)
    	cout<<"NO"<<endl;
    	else
    	cout<<"YES"<<endl;
    }
    return 0;
}
