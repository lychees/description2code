#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
   int i,j,k,l,t,a,b,c;
   string s,r;
  cin>>t;
  for(i=0;i<t;i++)
   {
   	cin>>s;
   	l=s.size();
   	k=l;
   	c=0;
   	for(j=0;j<l;j++)
   	{
   		r[k-1]=s[j];
   		k--;
   	}
   	for(j=1;j<l;j++)
   	{
   		a=int(s[j]-s[j-1]);
   		a=fabs(a);
   		b=int(r[j]-r[j-1]);
   		b=fabs(b);
   		if(a==b)
   		c++;
   	}
   	if(c==l-1)
   	cout<<"Funny"<<endl;
   	else
   	cout<<"Not Funny"<<endl;
   	
   }
    return 0;
}
