#include<iostream>
#include<cstring>
#include<string.h>
using namespace std;
int main(void)
{
	long t, len, k, len_s;
	bool found = false;
	char p[1000], s[100000];
	int pcheck[1000], scheck[1000];
	cin >> t;
	while (t--)
	{
		cin >> p >> s;
		len = strlen(p);
		len_s = strlen(s);
		found = false;
		for (int i = 'a'; i <= 'z'; i++)
			pcheck[i] = scheck[i] = 0;

		for (int i = 0; i <len; i++)
		{
			scheck[p[i]]++;
		}
		for (int i = 0; i < len_s; i++)
		{
			if (i >= len)
				pcheck[s[i - len]]--;
			pcheck[s[i]]++;
			for (k = 'a'; k <= 'z'; k++)
				if (scheck[k] != pcheck[k]) break;
			if (k > 'z')
			{
				found = true;
				break;
			}
		}
		if (found == true)
			cout << "YES\n";
		else
			cout << "NO\n";
	}
	return 0;
}
