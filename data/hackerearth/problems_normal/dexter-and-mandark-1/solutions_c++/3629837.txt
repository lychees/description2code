#include<iostream>
#include<algorithm>

using namespace std;

int const N = 1e3 + 41;
int const MOD = 1e9 + 7;

int n, m, test, dp[N][N];

void clear(){
	for(int i=0;i<N;++i){
		for(int j=0;j<N;++j){
			dp[i][j] = 0;
		}
	}
}

int main(){
	scanf("%d",&test);
	for(int t=0;t<test;++t){
		scanf("%d%d",&n,&m);
		clear();
		dp[1][1] = 1;
		for(int i=1;i<n;++i){
			for(int j=1;j<=i;++j){
				dp[i+1][j+1] += dp[i][j];
				dp[i+1][j+1] %= MOD;
				int cntM = j;
				int cntD = i - j + 1;
				if(cntD * m <= cntM){
					dp[i+1][j] += dp[i][j];
					dp[i+1][j] %= MOD;
				}
			}
		}
		int ans = 0;
		for(int i=n/2+1;i<=n;++i){
			ans += dp[n][i];
			ans %= MOD;
		}
		printf("%d\n",ans);
	}







	return 0;
}
