#include<iostream>
#include<cmath>
using namespace std;
int main(){
int n,c;
cin>>n;
long int a[n];
for(int i=0;i<n;i++){
    cin>>a[i];
}
for(int j=0;j<=n-2;j++){
for(int i=0;i<n-1;i++){
    if(a[i]<a[i+1]){
        c=a[i];
        a[i]=a[i+1];
        a[i+1]=c;
    }
}}
int b[n];
for(int i=0,j=0;j<n;i++){
    b[i]=a[j];
    j++;
    if(j==n)
        break;
    b[n-1-i]=a[j];
    j++;
}
int danger=0;
for(int i=0;i<n-1;i++){
    if(abs(b[i]-b[i+1])>danger)
        danger=abs(b[i]-b[i+1]);
}

if(abs(b[n-1]-b[0])>danger)
    danger=abs(b[n-1]-b[0]);
cout<<danger;
}
