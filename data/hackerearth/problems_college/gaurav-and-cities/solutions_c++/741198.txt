int random_long(int digit=8)
{
   int len=digit;
   int x=0;
   for ( int i = 0; i < len; ++i )
   {
    int dig = rand() % 10;
    while ( x == 0 && dig == 0 )
      dig = rand() % 10;
    x = x * 10 + dig;
  }
   return x;
}

int random(int l, int r)
{
    int diff = r - l + 1;

    return l+random_long()%diff;

}
#define MOD 10000007

void mult(ll (*M)[6],ll (*A)[6]){
	ll C[6][6];
	ms(C);
	for(int i=1;i<=5;i++){
		for(int j=1;j<=5;j++){
			for(int k=1;k<=5;k++){
				C[i][j]+=(M[i][k]*A[k][j]);
				C[i][j]%=MOD;
			}
		}
	}
	for(int i=1;i<=5;i++)
		for(int j=1;j<=5;j++)
			M[i][j]=C[i][j];
} 

void init(ll (*M)[6]){

	M[1][2]=1;
	M[2][3]=1;
	M[3][4]=1;
	M[4][5]=1;
	M[5][1]=1;
	M[5][2]=2;
	M[5][3]=3;
	M[5][4]=2;
	M[5][5]=1;

}

ll power(ll n){

	ll res[6][6];
	ll M[6][6];
	ms(M);
	ms(res);
	init(M);
	for(int i=1;i<=5;i++)
		res[i][i]=1;
	
	while(n){

		if(n&1)
			mult(res,M);
		n/=2;
		mult(M,M);
	}
	ll ans=(res[1][1] + 2*res[1][2] + 3*res[1][3] + 2*res[1][4] + res[1][5])%MOD;
	return ans;
}
