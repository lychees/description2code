#include<iostream>
#include<set>
#define MAX 100001
using namespace std;
int n,q,arr[MAX],sz[MAX];
set< pair<int,int> > s;

int root(int x){
    while(arr[x]!=x){
        arr[x]=arr[arr[x]];
        x=arr[x];
    }
    return x;

}

void mergeset(int a,int b){
    int ra=root(a),rb=root(b);
    s.erase(pair<int,int>(sz[ra],ra));
    s.erase(pair<int,int>(sz[rb],rb));
    arr[rb]=arr[ra];
    sz[ra]+=sz[rb];
    s.insert(make_pair(sz[ra],ra));

}

int main(){
    int x,y;
    pair<int,int> ans1,ans2;
    set<pair<int,int> > :: iterator it1,it2;
    cin>>n>>q;
    for(int i=1;i<=n;i++){
        arr[i]=i;
        sz[i]=1;
        s.insert(make_pair(sz[i],arr[i]));
    }
    while(q--){
        cin>>x>>y;
        if(root(x)!=root(y))
            mergeset(x,y);
        it1=s.begin();
        it2=s.end();
        it2--;
        ans1=*it1;
        ans2=*it2;
        cout<<ans2.first-ans1.first<<endl;

    }

}
