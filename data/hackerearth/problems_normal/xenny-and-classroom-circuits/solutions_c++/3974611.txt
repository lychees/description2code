#include <iostream>
#include<algorithm>
#include<string>

int main()
{
	string s;
	int t,i,q,m,mm,m2,mm2,n;

	{

	for( i=0;i<2*n;i+=2)
	{
	   if(s[i]!=s[0])
	   {
	   	m++;

		for( i=0;i<2*n;i+=2)
	{
	   if(s[i]==s[0])
	   {
	   	m2++;
	   }
	}
		for( i=1;i<2*n;i+=2)
	{
	   if(s[i]!=s[1])
	   {
	   	mm++;
	   }
	}
	for( i=1;i<2*n;i+=2)
	{
	   if(s[i]==s[1])
	   {
	   	mm2++;
	   }
	}
	m=min(m,mm);

	cout <<min(m,m2) << endl;
	}
	
    
    return 0;
}
