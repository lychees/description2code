#include <stdio.h>
int main()
{
	int test, n;
	scanf("%d", &test);
	while(test--)
	{
		scanf("%d", &n);
		int dim[n], x[n];
		for(int i = 0; i < n; i++)
			scanf("%d", &dim[i]);
		for(int i = 0; i < n; i++)
			scanf("%d", &x[i]);
		int ans = 0;
		for(int i = 0; i < n; i++)
			ans = ans ^ (x[i] - 1) ^ (dim[i] - x[i]);
		if(ans == 0)
			printf("Second\n");
		else
			printf("First\n");
	}
}
