#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int arr[105];
int main()
{
    int t,n,i,no;
    cin>>t;
    while(t--)
    {
        cin>>n;
        no=0;
        for(i=0;i<n;i++)
        {
            cin>>arr[i];
            if(arr[i]%2!=0)
                no++;
        }
        if(no%2!=0)
            cout<<"No"<<endl;
        else
            cout<<"Yes"<<endl;
    }
    return 0;
}
