ll binpow(ll a, ll b)
{
	ll local_res = 1;
	while (b)
	{
		if (b & 1)
		{
			local_res *= a;
			local_res %= mod;
		}
		a *= a;
		a %= mod;
		b >>= 1;
	}
	return local_res;
}

ll solve (ll n, ll k) 
{
	return 1 + 2 * (n - binpow(2, (ll)log2(n)));
}
