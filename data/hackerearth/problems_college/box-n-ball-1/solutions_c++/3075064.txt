#include <iostream>
#include<algorithm>
using namespace std;
 
int main()
{
   long long t,i,d[3];
   double ans;
   cin>>t;
   while(t--)
   {
   	for(i=0;i<3;i++)
   	{
   		cin>>d[i];
   	}
   	sort(d,d+3);
   	ans=d[0]*.5;
   	printf("%.1lf\n",ans);
   	
   }
    return 0;
}
