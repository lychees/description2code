#include <iostream>
#include <vector>
#include<algorithm>
using namespace std;
vector<int> v;
int main()
{
    int n,x;
    cin>>n;
    int c=0;
    	for(int i=0;i<n;i++)
    	{
    		cin>>x;
    		v.push_back(x);
    	}
    	sort(v.begin(),v.end());
    	for(int i=0;i<n;i++)
    	{
    		
    		cout<<v[i]-c;
    		
    		cout<<" ";
    		c++;
    	}
    	
    return 0;
}
