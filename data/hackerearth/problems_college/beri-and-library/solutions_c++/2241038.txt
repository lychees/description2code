using namespace std;
#include <iostream>
int main()
{
    long t;
    cin>>t;
    while(t--){
        long n,c=0;
        long q;
        cin>>q;
        n = q;
        while(n){
            if(n%2)
                c++;
            n /= 2;
        }
        cout<<c<<endl;
    }
}
