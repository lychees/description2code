#include <iostream>
#include <cstdio>
#include <algorithm>
#include <vector>
#include <cstdlib>
#include <cstring>
using namespace std;

int main(void) {
    
    int N = 0,Q = 0,idx = 0;
    int hash[123][12345] = {0};
    memset(hash,0,sizeof(hash));
    scanf("%d",&N);
    for(int it = 0; it != N; ++it) {
    	scanf("%d",&idx);
    	hash[idx][it] = 1;
    }
    
    for(int it1 = 0; it1 != 121; ++it1) {
    	for(int it2 = 1; it2 != 12341; ++it2) {
    		hash[it1][it2] += hash[it1][it2 - 1];
    	}
    }
    
    scanf("%d",&Q);
    while(Q--) {
    	int X = 0, Y = 0, val = 0;
    	scanf("%d %d %d",&X,&Y,&val);
    	if(X == 0)
    		printf("%d\n",hash[val][Y]);
    	else
    		printf("%d\n",hash[val][Y] - hash[val][X - 1]);
    }
    
    
    
    return 0;
}
