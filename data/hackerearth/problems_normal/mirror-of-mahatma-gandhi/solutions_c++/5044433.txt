#include <iostream>
using namespace std;

int main()
{
    int T;
    cin >> T;
    while ( T-- > 0)
    {
    	string input;
    	string result = "YES";
    	cin >> input;
    	int i,j;
    	for( i = 0; i < input.size(); i++ )
    	{
    		if( input[i] != '0' && input[i] != '1' && input[i] != '8' )
    		{
    			result = "NO";
    			break;
    		}
    	}
    	for( i = 0, j = input.size()-1; i <= j; i++, j--)
    	{
    		if( input[i] != input[j] )
    		{
    			result = "NO";
    			break;
    		}
    	}
    	cout << result << endl;
    }
    return 0;
}
