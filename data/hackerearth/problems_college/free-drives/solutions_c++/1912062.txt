#include<iostream>
#include<stdio.h>
#include<math.h>
#include<algorithm>
using namespace std;
#define lli long long int

int main()
{
    lli t,n,ans;
    cin>>t;
    while(t--)
    {
              cin>>n;
              
              ans=n*(n-1);
              ans/=2;
              
              cout<<ans<<endl;
    }
    
    return 0;
}
