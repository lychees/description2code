#include<bits/stdc++.h>
using namespace std;
vector<int>v[10];
int prime[10000005];
int h=0;
void sieve()
{
	int maxi=-1;
	int i,j;
	prime[0]=0;
	prime[1]=0;
	for(i=2;i<=10000000;i++)
	{
		if(!prime[i])
		{
			for(j=2*i;j<=10000000;j+=i)
			{
				prime[j]++;
			}
		}
	}
	
	for(i=2;i<=10000000;i++)
	{
		if(!prime[i])
		prime[i]=1;
	}
	for(i=2;i<=10000000;i++)
	{
		int x=prime[i];
		maxi=max(maxi,x);
		v[x].push_back(i);
	}
	//printf("%d\n",maxi);
}
long long n;
int main()
{
	sieve();
	int t;
	int a,b,k;
	scanf("%d",&t);
	while(t--){
		scanf("%d %d %d",&a,&b,&k);
		if(k>=9)
		{
			printf("0\n");
			continue;
		}
		int low,high;
		low=lower_bound(v[k].begin(),v[k].end(),a)-v[k].begin();
		high=upper_bound(v[k].begin(),v[k].end(),b)-v[k].begin();
		int answ=high-low;
		printf("%d\n",answ);
	}
    return 0;
}
