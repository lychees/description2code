#include <bits/stdc++.h>
using namespace std;

int p[10000];
string s;
 
int main()
{
    int i,j,k;
 
    int t,n;
    cin>>t;
    while(t--)
    {
      cin>>n;
 
      int len;
      for(i=0;i<n;i++)
      {
        cin>>s;
        len=s.size();
 
        p[i]=0;
        for(j=0;j<len;j++)
        {
          if(s[j]=='Y')
            p[i]+=1<<j;
        }
      }
 
 	
      int mn=len;
      int pp=pow(2,len);
      for(i=0;i<pp;i++)
      {
 
        for(j=0;j<n;j++)
        {
          if( (p[j] & i)==0)
            break;
        }
 
        if(j==n)
          mn=min(mn,__builtin_popcount(i));
      }
      cout<<mn<<endl;
    }
 
 
    return 0;
}
