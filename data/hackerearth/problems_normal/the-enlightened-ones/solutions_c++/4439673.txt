#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;

vector<int> tmp;

int check(int num, int k)
{
	unsigned prev = tmp[0] + num;
	k--;
	for (int i = 1; i<tmp.size(); i++)
	{
		if (prev + num >= tmp[i]) continue;
		if (k == 0) return 0;
		prev = tmp[i] + num;
		k--;
	}
	return 1;
}

int main()
{
	int n, k, p;
	cin >> n >> k;
	for (int i = 0; i<n; i++)
	{
		cin >> p;
		tmp.push_back(p);
	}
	sort(tmp.begin(), tmp.end());
	int l = 0, h = 10000000;
	int mid;
	while (h >= l)
	{
		mid = (h + l) / 2;
		int upper = check(mid, k);
		int lower = check(mid - 1, k);
		if (upper && lower == 0) break;
		if (upper == 0)
			l = mid + 1;
		else
			h = mid - 1;
	}
	cout << mid << endl;
	return 0;
}
