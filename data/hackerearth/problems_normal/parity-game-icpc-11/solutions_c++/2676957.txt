template<class T> inline void amin(T &a, const T &b) { if (b<a) a=b; }
template<class T> inline void amax(T &a, const T &b) { if (a<b) a=b; }


int N, Q;
int S[1000];
int D[1000];
int main() {
    scanf("%d%d", &N, &Q);
    REP (i, Q) {
	int a, b;
	scanf("%d%d", &a, &b);
	S[a]++;
	S[b+1]--;
	D[a]++;
	D[b+1]++;
    }

    REP (i, N+3) S[i+1] += S[i];
    int ans = *min_element(S+1, S+N+1);
    if (1 < N) amin(ans, *min_element(D+2, D+N+1));
    printf("%d\n", ans);
    
    return 0;
}
