#include<stdio.h>
#include<string>
#include<iostream>
using namespace std;
int a[1000];
int main()
{
    int test;
    cin>>test;

    while(test--)
 {

      string s;
      cin>>s;

    int j=0,totalcount;
    for(int i=0;s[i]!='\0';)
    {
        totalcount=0;
        while((s[i]!='\0')&&(s[i]=='_'))
        {
            totalcount++;
            i++;
        }
        if(totalcount!=0)
        {
        a[j]=totalcount;
        j++;
        }
        if(s[i]=='\0')
          break;

        i++;



    }

    if(j==1&&a[j-1]==s.length())
     {

         cout<<"Bob\n";
         continue;
     }

    int xori=0;
    for(int i=0;i<j;i++)
    {

       xori=xori^a[i];
    }

    if(xori==0)
    cout<<"Bob\n";

    else
    cout<<"Alice\n";

  }

}
