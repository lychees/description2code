#include <iostream>
#include<vector>
#include<cmath>
using namespace std;

int main()
{
	long long ans=0;
    int n;
    cin>>n;
    vector<long> sweet(1000002,0);
    vector<long> qt(1000002,0);
    for(int i=0;i<n;i++)
    {
    	int x,y;
    	cin>>x;
    	++sweet[x];
    	cin>>y;
    	qt[x]=y;
    }
    
    for(int i=1000000;i>=1;i--)
    {
    	qt[i]=max(qt[i],qt[i+1]);
    }
    int t;
    cin>>t;
    while(t--)
    {
    	int q;
    	cin>>q;
      long	long z=100*floor(qt[q]/q);
    	ans+=z;
    	
    }
    cout<<ans;
    return 0;
}
