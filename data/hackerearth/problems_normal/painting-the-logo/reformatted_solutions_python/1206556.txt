def foo(N):
    import math
    n1, n2 = (lambda N: (int(math.sqrt(2*N + 1)/4. + 1/4.),
              int(math.sqrt(2*N + 1)/4. - 1/4.)))(N)
    an = max((2*n1 - 1, 2*n2))
    a, b = (an - 1, an) if an % 2 == 0 else (an, an - 1)
    a, b = (a, b) if a and b else (0, 0)
    print a, b


T = input()
for _ in range(T):
    foo(input())
