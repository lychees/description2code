#include <iostream>
#include<algorithm>
using namespace std;

int main()
{long int r,t,a,b,i,z,d[100000],s[100000],q;
  cin>>t;
  z=1;
  while(t--)
  {
  	cin>>a>>b;
  	s[z++]=a+b;
  }
  z--;
  r=1;
   for(i=1;i<=z;i++)
  {
  	d[r++]=s[i];
  }
  
   sort(d,d+r);
  
  q=1;
for(i=1;i<=z;i++)
{
    
	if(d[q]==s[i])
	{
		cout<<i;
		cout<<" ";
        s[i]=-9;
q++;
i=0;
}
if(q==r)
break;
}
    return 0;
}
