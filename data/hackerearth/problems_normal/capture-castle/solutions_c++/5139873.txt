#include<bits/stdc++.h>
using namespace std;
int cost[1000][1000];
int arr[1000][1000];
int xx[]={1,0,-1,0};
int yy[]={0,1,0,-1};

bool isSafe(int x,int y,int n,int m){
    if(x>=0 && x<n && y>=0 && y<m)
        return true;
    return false;

}
void mincost(int x,int y,int n,int m){
    for(int k=0;k<4;k++){
        int new_x=x+xx[k];
        int new_y=y+yy[k];
        if(!isSafe(new_x,new_y,n,m))
            continue;
        if(cost[new_x][new_y]>cost[x][y]+arr[new_x][new_y]){
            cost[new_x][new_y]=cost[x][y]+arr[new_x][new_y];
            mincost(new_x,new_y,n,m);
        }


    }
}

int main(){
    int t;
    int n,m;
    int xr,yr,time;
    cin>>t;
    while(t--){
        cin>>n>>m;
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                cost[i][j]=1e9;
                cin>>arr[i][j];
            }

        }
        cost[0][0]=arr[0][0];
        cin>>xr>>yr>>time;
        xr--;yr--;
        mincost(0,0,n,m);
        int ans=cost[xr][yr];
        if(ans<time){
        	cout<<"YES"<<endl;
        	cout<<time-ans<<endl;
        }
        else{
        	 cout<<"NO"<<endl;
        }
       

    }
return 0;

}
