#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int t,n,a,b,result,temp;
    cin >> t;
    while(t--)
    {
    	cin >> n;
    	temp=n;
    	vector <int> a1 (1000001,0);
    	result=0;
    	while(n--)
    	{
    		cin >> a >> b;
    		if(a!=b)
    		{
    			if(a1[b]!=0)
    			{
    				a1[b]-=1;
    				a1[a]+=1;
    			}
    			else
    			{
    				result+=1;
    				a1[a]+=1;
    			}
    		}
    		
    		
    	}
    	cout << result << endl;
    }
    return 0;
}
