#include <iostream>
using namespace std;
#include<map>
#include<vector>
#include<algorithm>
int occu[200001];
int main()
{
	string s;
	cin>>s;
	int le =  s.length();
	
   int n;
   scanf("%d",&n);
   map< char , vector<int> >  mp;
   
   for(int i=0;i<le;i++)
   {
   	   mp[s[i]].push_back(i);
   	   
   }
   
   for(int i =0;i<n;i++)
   {
   	   int num;
   	   char ct;
   	   //cin>>num>>ct;
   	  scanf("%d %c",&num,&ct);
   	  occu[mp[ct][num-1]] = -1;
   	  mp[ct].erase(mp[ct].begin()  +  num -1);
   	   
   }
   
   for(int i=0;i<le;i++)
   {
   	     if(occu[i]!=-1)
   	     {
   	     	printf("%c",s[i]);
   	     	
   	     }
   	     
   }
   printf("\n");
    return 0;
}
