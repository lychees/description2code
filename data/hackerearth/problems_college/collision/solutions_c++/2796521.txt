#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,c,x;
    int a[101]={0};
    cin>>t;
    while(t--)
    {
    	c=0;
    	cin>>n;
    	for(int i=0;i<n;i++)
    	{
    		cin>>x;
    		if(a[x%10]==0)
    		a[x%10]++;
    		else
    		c++;
    	}
    	cout<<c<<"\n";
    	for(int i=0;i<101;i++)
    	a[i]=0;
    }
    return 0;
}
