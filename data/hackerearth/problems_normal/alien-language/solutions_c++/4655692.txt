#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s,p;
    	cin>>s;
    	cin>>p;
    	
    	int l=p.size();
    	unordered_map<char , int> m;
    	for(int i=0;i<l;i++)
    	m[p[i]]=1;
    	
    	int f=0;
    	for(int i=0;i<s.size();i++)
    	if(m[s[i]]==1)
    	{
    		f=1;
    		break;
    	}
    	
    	if(f==1)
    	cout<<"YES"<<endl;
    	else
    	cout<<"NO"<<endl;
    }
    return 0;
}
