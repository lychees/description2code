#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ll int n;
	scanf("%lld",&n);
	ll int a[n],b[n];
	ll int i,j;
	for(i=0;i<n;i++)
	{
		scanf("%lld",(a+i));
	}
	ll int ans=0;
	int flag;
	while(1)
	{
		j=0;
		flag=0;
		for(i=n-2;i>=0;i--)
		{
			if(a[i]>a[i+1])
			{
				flag=1;
				continue;
			}
			else
			{
				b[j++]=a[i+1];
			}
		}
		b[j++]=a[0]; //As first element will always persist,as its not in the right of any other element
		if(flag==0)//Nothing will get deleted
		{
			break;
		}
		else if(j==n)//All elements could be deleted in just 1 iteration
		{
			ans++;
			break;
		}
		else
		{
			ans++;
			n=j;
			j=0;
			for(i=(n-1);i>=0;i--)
			{
				a[j++]=b[i];
				//~ cout<<b[i]<<" ";
			}
			//~ cout<<endl;
		}
	}
	printf("%lld\n",ans);
	return 0;
}
