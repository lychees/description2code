#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t, i, c;
    cin >> t;
    while(t--) {
    	char a[501], b[501];
    	memset(a, '\0', strlen(a));
    	memset(b, '\0', strlen(b));
    	cin >> a >> b;
    	c = i = 0;
    	while(a[i]!='\0' && b[i]!='\0') {
    		if(a[i] == b[i]) {
    			c++;
    		}
    		i++;
    	}
    	cout << c << endl;
    }
    return 0;
}
