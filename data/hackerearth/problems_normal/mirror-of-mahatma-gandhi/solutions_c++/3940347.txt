#include <iostream>
#include<cstring>
using namespace std;
 
int main()
{
    int t,m,j,l;
    char n[100];
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	m=0;
    	l=strlen(n);
    	for(j=0;j<strlen(n);j++)
    	{
    		if(n[j]==n[strlen(n)-j-1])
    		{
    		if((n[j]=='1'||n[j]=='0'||n[j]=='8') && (n[strlen(n)-j-1]=='1' || n[strlen(n)-j-1]=='0' || n[strlen(n)-j-1]=='8'))
    		{
    		{m=0;}
    		}
    		else
    		{m=1;break;}
    		}
    	  else
    	  {m=1;break;}
    	}
    if(m==0)
    {cout<<"YES"<<endl;}
    else
    {cout<<"NO"<<endl;}
    }
    return 0;
}
