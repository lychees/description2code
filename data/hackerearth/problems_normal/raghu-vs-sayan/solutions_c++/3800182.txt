#include <iostream>
#include <algorithm>

using namespace std;


int main() {
	int T, A, B, N;
	cin >> T;
	while (T > 0) {
		cin >> A >> B >> N;
		int *d = new int[N];
		for (int i = 0; i < N; i++) {
			cin >> d[i];
		}
		sort(d, d+N);
		int i = 0, cA = 0, cB = 0;
		while (i < N) {
			bool flag = false;
			if (A >= d[i]) {
				A -= d[i]; cA++;
			} else 
				flag = true;
			if (B >= d[i]) {
				B -= d[i]; cB++;
			} else
				flag = true;
			if (flag) break;
			i++;
		}
		// cout << A << " " << B << " " << i << endl;
		if (cA == cB) {
			// if (A > B)
			// 	cout << "Raghu Won\n";
			// else if (A < B)
			// 	cout << "Sayan Won\n";
			// else
				cout << "Tie\n";
		} else {
			if (cA > cB)
				cout << "Raghu Won\n";
			else if (cA < cB)
				cout << "Sayan Won\n";
			else
				cout << "Tie\n";
		}
		delete[] d;
		T--;
	}
	return 0;
}
