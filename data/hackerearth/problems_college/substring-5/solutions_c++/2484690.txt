#include<bits/stdc++.h>
using namespace std;
int main()
{
	long int i,j,k,l,n,flag,cnt;
	string str,s1;
	cin>>str;
	n=str.length();
	cnt=0;
	for(l=2;l<=n;l++)
	{
		for(i=0;i<n-l+1;i++)
		{
			s1=str.substr(i,l);
			//cout<<s1<<"\n";
			flag=0;
			for(j=0;j<l-1;j++)
			{
				if(s1[j]!=s1[j+1])
				{
					flag=1;
					break;
				}
			}
			if(flag==0)
			cnt++;
		}
	}
	cout<<cnt+str.length()<<"\n";
}
