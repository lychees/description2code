#include <iostream>
#include <string.h>
using namespace std;

int vowel(char c){
	if(c=='a'||c=='e'||c=='i'||c=='o'||c=='u'||c=='y')
	{
		return 0;
	}
	else return 1;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    	char s[50];
    	scanf("%s",s);
    	int c=0,v=0,i=0,len;
    	len=strlen(s);
    	while(i<len){
    		if(vowel(s[i])){
    				c++;
    				}
	    		else{
	    		v++;
	    		}
    		i++;
    	}
    	
    	if(c>v){
    		printf("hard\n");
    		}
    		else
    		{
    			int j=0,flag=0;
    			while(j<(len-2))
    			{
    				if(vowel(s[j]) && vowel(s[j+1]) && vowel(s[j+2])){
    				flag=1;
    				break;
    				}
    				j++;
    			}
    			
    			if(flag==1) printf("hard\n");
    			else printf("easy\n");
    		}
    		

    }
    return 0;
}
