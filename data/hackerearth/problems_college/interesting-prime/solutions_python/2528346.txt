t = int(raw_input())
while t:
	primeCt = {}
	res = 1
	n = int(raw_input())
	m = n
	i = 2
	while i*i <= n:
		while not(m % i):
			if i not in primeCt:
				primeCt[i] = 1
			else:
				primeCt[i] += 1
			m/=i
		i+=1
	if m > 1:
		primeCt[m] = 1
	for key in primeCt:
		res *= (primeCt[key]+1)
	res -= 1
	print res
	t-=1
