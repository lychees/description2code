#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    
    int i,j,k,N,T;
    
    int dp[102]={0};
    dp[0] = 1;  dp[1] = 1; dp[2] = 1; dp[3] = 2; dp[4] = 3; dp[5] = 5;
    
    for(i=6;i<102;i++)
    {
    	dp[i] = dp[i-1] + dp[i-3] + dp[i-5];
    }
    
    int table[102]={0};
    table[0] = 1;
    
    int m = 3;
    int S[3] = {1,3,5};
    
    for(int i=0; i<m; i++)
        for(int j=S[i]; j<=101; j++)
            table[j] += table[j-S[i]];
    
    cin>>T;
    
    while(T--)
    {
    	cin>>N;
    	cout<<table[N]<<endl;
    }
    return 0;
}
