#include <iostream>
using namespace std;
#include<string.h>
int gcd(int x,int y)
{
	if(y==0)
	{
		return x;
		
	}
else
gcd(y,x%y);
}
int main()
{
	char str[10];
    int test,x,y,count,l;
    cin>>test;
    while(test)
    {
    	
    	cin>>x>>y>>str;
    	if(strcmp(str,"Daspal"))
    	{
    		count=0;
    	}
    	else
    	{
    		count=1;
    	}
    	while(x>0&&y>0)
    	{
    		l=gcd(x,y);
    		x=x-l;
    		y=y-l;
    		count++;
    	}
    	if(count%2==0)
    	{
    		cout<<"Daspal"<<endl;
    	}
    		else
    	{
    		cout<<"Chandu"<<endl;
    	}
    	test--;
    }
    return 0;
}
