#include <bits/stdc++.h>
using namespace std;

int main()
{
    int tc,i,j ;
    cin>>tc ;
    tc=tc+1;
    while(tc--)
    {  char a[202] ;
    	gets(a) ;
      int le=strlen(a) ;
      int l=sqrt(le) ;
      
      for(j=0;j<l;j++)
      { for(i=j;i<le;i=i+l)
         printf("%c",a[i]) ;
      }
    	printf("\n") ;
    }
    
    return 0;
}
