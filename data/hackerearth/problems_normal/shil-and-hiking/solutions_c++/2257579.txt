#include <iostream>
using namespace std;
#define ll long long int

int main()
{
    ll n;
	cin>>n;
	double x[n],y[n];
	double _min,t;
	ll ans=0;
	
	for(int i=0;i<n;i++) cin>>x[i]>>y[i];
	for(int i=1;i<n;i++){
	    _min=999999.1;
	    for(int j=i-1;j>=0;j--){
	        t=(y[j]-y[i])/(x[j]-x[i]);
	        
	        if(t<=_min){
	            _min = t;
	            ans++;
	        }
	    }
	}
    cout<<ans;
    return 0;
}
