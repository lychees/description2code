
# Read input from stdin and provide input before running code

N = int(raw_input());
bigArr = [];
while (N > 0) :
	xyd = map(int, raw_input().strip().split(' '));
	bigArr.append(xyd);
	N -= 1;
bigArr = sorted(bigArr, key = lambda x : x[0]);
res = 0;
fncp = 1;
for i in bigArr :
	fncp = max(fncp, i[0]);
	if (fncp <= i[0] + i[2]):
		res += (i[0] + i[2] - fncp + 1);
		fncp = i[0] + i[2] + 1;
print res;
