#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define S(x) scanf("%lld", &x)

ll d[2000005];

ll solve(ll l, ll r,ll c) {
	while(r - l > 1)
	{
		long long m = (l + r) / 2;
		if(d[m] <= c) 			
			l = m;
		else
			r = m;
	}
	if(d[r] <= c) {
		return r;
	}
	if(d[l] <= c) {
		return l;
	}
	return 0;
}

int main()
{
	ll q;
	S(q);
	d[0] = 0;
	ll i;
	for(i = 1;i <= 2000000;i++) {
		d[i] = ((i + 1)*i) / 2;
	//	cout << d[i] << "\n";
	}
	ll l,r;
	while(q--) {
		ll to = 0;
		S(l);
		S(r);
		l--;
		for(i = 0; d[i] <= l;i++) {
			ll fin = solve(0, i,l - d[i]);
			to += fin;
		//	cout << d[fin] << " " << d[i] << "\n";
		}
		ll to1 = 0;
		for(i = 0; d[i] <= r;i++) {
			ll fin = solve(0, i,r - d[i] );
			to1 += fin;
		//	cout << d[fin] << " "<<d[i]   << "r\n";
		}
		cout << to1 - to << "\n";
	}
}
