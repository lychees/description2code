#include <bits/stdc++.h>
using namespace std ;

int T,N,A,B,M,X,arr[100001]; 

int main(){

	cin >> T ;
	while(T--){
		cin >> A >> B >> N ;
		M = INT_MAX ;		
		for(int i=1;i<=N;i++)
			cin >> arr[i] ;
		sort(arr+1,arr+1+N) ;
		int cntC,cntJ ;
		cntC= cntJ = 0 ;
		for(int i=1;i<=N;i++){
			if(arr[i] <= A){ 
				cntC ++ ;
				A -= arr[i] ; 
			}
			if(arr[i] <= B){
				cntJ ++ ;
				B -= arr[i] ; 
			}
		}	
		if(cntC == cntJ){
			cout << "Tie" << endl ;
		}else if(cntC > cntJ){
			cout << "Chef Won" << endl ;
		}else{
			cout << "Jhool Won" << endl ;
		}
	}
	return 0 ;
}
