#include <cstdio>
#define LIM 1500000

int memo[LIM];

long long f(int x)
{
	if (x==0) return 0;
	if (x==1) return 1;
	
	if (x<LIM && memo[x])
		return memo[x];
	
	long long g=1;
	while(g<=x) g*=2;
	g/=2;
	
	if (x<LIM) return memo[x]=x-g+1+f(x-g)+f(g-1);
	else return x-g+1+f(x-g)+f(g-1);
}

int main()
{
    int t, a, b;
    scanf("%d", &t);
    while(t--)
    {
    	scanf("%d%d", &a, &b);
    	printf("%lld\n", f(b)-f(a-1));
    }
    return 0;
}
