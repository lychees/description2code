for _ in range (int(raw_input())):
    st=raw_input()
    l=len(st)
    j=97
    for i in range (l):
        k=ord(st[i])
        if abs(k-j)>abs(26-(k-j)):
            print(k-j-26),
        elif k-j==-13 or k-j==13:
            print("13"),
        elif abs(k-j)>abs(26+(k-j)):
            print(26+(k-j)),
        else:
            print(k-j),
        j=k
    print;
