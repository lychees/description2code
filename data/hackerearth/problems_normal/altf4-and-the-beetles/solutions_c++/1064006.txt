#include <iostream>
#include<algorithm>
#include<cstdio>
#include<vector>
#include<climits>
using namespace std;
#define Int long long
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int N;Int P[20009],Q[20009],A,B,C;
		scanf("%d %lld %lld %lld",&N,&A,&B,&C);
		for(int i=0;i<N;i++)
		scanf("%lld %lld",&P[i],&Q[i]);
		P[N]=INT_MAX;
		Q[N]=INT_MAX;
		sort(P,P+N);sort(Q,Q+N);
		Int ans=N*A;
		Int curr=ans;
		int i=0,j=0;
		while(i<N || j<N)
		{
			if(P[i]<=Q[j]){curr+=B-A;i++;}
			else
			{
				curr+=C-B;j++;
			}
			ans=max(curr,ans);
		}
		printf("%lld\n",ans);

	}



    return 0;
}
