#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
vector<ll> arr;
int main(){
	int a,n;
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>a;
		arr.push_back(a);
	}
	sort(arr.begin(),arr.end());
	ll mul1=arr[0]*arr[1];
	ll mul2=arr[n-1]*arr[n-2];
	cout<<max(mul1,mul2)<<endl;
	return 0;
}
