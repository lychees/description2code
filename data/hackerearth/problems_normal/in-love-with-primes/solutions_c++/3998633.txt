#include<iostream>
#include<math.h>
using namespace std;
int prime(long int n) //Primality Test
	{
		if(n==1 || n==4)
			return -1;
		long int i;
		for(i=2;i<sqrt(n);i++)
			{
				if(fmod(n,i)==0)
					return -1;
			}
		return 1;
	}
int main()
	{
		int tc,count,flag;
		long int n,a,b;
		cin>>tc;
		for(count=1;count<=tc;count++)
			{
				flag=0;
				cin>>n;
				for(a=1;a<n;a++)
					{
						for(b=1;b<n;b++)
							{
								if((a+b)==n && prime(a)==1 && prime(b)==1)
									{
									flag=1;
									break;
									}
							}
						if(flag==1)
							break;
					}
				if(flag==1)
					cout<<"Deepa\n";
				else
					cout<<"Arjit\n";
			}
		return 0;
	}
