#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int a[100];
    
    int i;
    for(i=0;i<n;i++)
    {
     cin>>a[i];
     //cout<<a[i]<<"\n";
    }
    int b[100],j;
    	int sum;
    for(i=0;i<n;i++)
    {
    sum=0;
      if(i==0)
      {
       b[i]=a[i];
      }
      else
      {
        for( j=0; j<i; j++)
        {
         sum=sum+b[j];
         
        }
       // cout<<sum<<"\n";
        b[i]= (a[i]*(i+1))-sum;
       // cout<<b[i]<<"\n";
      }
      
    }
     for(i=0;i<n;i++)
    {
     cout<<b[i]<<" ";
    }
    
    return 0;
}
