number_of_numbers = int(raw_input())
list_of_numbers = raw_input().split()
not_primes = list()

for x in list_of_numbers:
    for y in list_of_numbers:
        if x != y and int(x) % int(y) == 0 and x in list_of_numbers:
            if x not in not_primes:
                not_primes.append(x)

for x in not_primes:
    list_of_numbers.pop(list_of_numbers.index(x))

print " ".join(list_of_numbers)
