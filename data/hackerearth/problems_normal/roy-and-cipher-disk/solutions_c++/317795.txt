#include <iostream>
#include <cstdio>
#include <string>
#include <vector>
using namespace std;

int main()
{
    int t;
    string s;
    cin>>t;
    while(t--)
    {
    	cin>>s;
    	vector<int> v;
    	if(s[0]-97 <=13)
    		v.push_back(s[0]-97);
    	else
    		v.push_back(s[0]-97-26);
    	int a;
    	for(int i=1;i<s.length();i++)
    	{
    		a = s[i]-s[i-1];
    		if(a>=0)
    		{
    			if(a<=13)
    			{
    				v.push_back(a);
    			}
    			else
    			{
    				v.push_back(a-26);
    			}
    		}
    		else
    		{
    			a*=(-1);
    			if(a<13)
    			{
    				v.push_back(-1*a);
    			}
    			else
    			{
    				v.push_back(26-a);
    			}
    		}
    	}
    	for(int i=0;i<v.size();i++)
    	{
    		cout<<v[i]<<" ";
    	}
    	cout<<endl;
    }
    return 0;
}
