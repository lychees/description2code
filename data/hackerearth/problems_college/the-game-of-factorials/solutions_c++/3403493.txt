#include <bits/stdc++.h>
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		int a[1001];
		memset(a,0,sizeof(a));
		a[0]=1;
		int limit=1;
		int i=2;
		while(i<=n)
		{
			int sum=0;
			int j=0;
			while(j<limit)
			{
				int temp=a[j]*i+sum;
				a[j]=temp%10;
				temp/=10;
				sum=temp;
				j++;
			}
			while(sum!=0)
			{
				a[limit]=sum%10;
				sum/=10;
				limit++;
			}
			i++;
		}
		limit--;
		long long total=0;
		for(int i=limit;i>=0;i--)
		{
			total+=a[i];
		}
		cout<<total<<endl;
	}
}
