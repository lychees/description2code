#include <bits/stdc++.h>
#define MAX 1000001
using namespace std;

int min(int a,int b)
{
	if(a<b) return a;
	else 
	return b;
}

int main()
{
	int p[MAX]={0};
	int i,j;
	p[0]=p[1]=-1;
	for(i=2;i<sqrt(MAX);i++)
	{
		for(j=i*i;j<MAX;j+=i)
		{
			if(p[j]==0)
			p[j]=i;
			//else
			//{
		//		p[j]=min(p[j],i);
			//}
		}
	}
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,ans;
		scanf("%d",&n);
		if(p[n]==0)
		ans=0;
		else
		ans=n-p[n];
		printf("%d\n",ans);
	}
	
	return 0;
}
