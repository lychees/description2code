#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int hash[1000001] = {0};
		int max = 0;
		int curr = 0;
		for(int i=0;i<n;i++)
		{
			int s,r;
			cin>>s>>r;
			if(!hash[s] && r)
			{
				curr++;
				hash[s] = 1;
			}
			else if(!r)
			{
				if(curr>max)
				max = curr;
				curr = 0;
			}
		}
		if(curr>max) max = curr;
		cout<<max<<endl;
	}
    return 0;
}
