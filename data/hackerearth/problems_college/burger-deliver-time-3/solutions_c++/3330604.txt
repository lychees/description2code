#include <iostream>
using namespace std;
#define ll long long
#include<vector>


int main()
{
	ll k,a,b,i,j;
	cin>>k;
	ll z[k][2];
	for(i=0 ; i<k;i++)
	{
		cin>>a>>b;
		z[i][0] = a+b;
		z[i][1] = i+1;
	}
	for(j=0;j<k-1;j++)
	{
		for(i=0 ; i<k-j-1;i++)
		{
			if(z[i][0] > z[i+1][0])
			{
				ll temp = z[i][0];
				z[i][0] = z[i+1][0];
				z[i+1][0] = temp;
				temp = z[i][1];
				z[i][1] = z[i+1][1];
				z[i+1][1] = temp;
			}
		}
	}
	for(i=0 ; i<k;i++)
	{
		cout<<z[i][1] <<" ";
	}
	cout<<"\n";
    return 0;
}
