#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    string a,b;
    cin>>t;
    while(t--)
    {cin>>a>>b;
      int r[26];
      for(int i=0;i<26;i++)
      r[i]=0;
      for(int i=0;i<b.size();i++)
      {
      	r[b[i]-'a']++;
      	
      }
    	for(int i=0;i<a.size();i++)
    	{
    		while(r[a[i]-'a']--)cout<<a[i];
    	}
    	
    	cout<<endl;
    	
    }
    return 0;
}
