#include<iostream>
#include<bits/stdc++.h>
#include<stdio.h>
using namespace std;
int dp[1001][11];
int find(int n,int k)
{
	if(n==1)
		return 1;
	if(k==1)
		return n;
	if(dp[n][k]!=-1)
		return dp[n][k];
	int mi=INT_MAX,res;
	for(int i=1;i<=n;i++)
	{
		res=max(find(i-1,k-1),find(n-i,k));
		if(res<mi)
			mi=res;
	}
	return dp[n][k]=mi+1;
}
main()
{
	int q;
	scanf("%d",&q);
	memset(dp,-1,sizeof dp);
	while(q--)
	{
		int n,k;
		scanf("%d%d",&n,&k);
		printf("%d\n",find(n,k));
	}
}
