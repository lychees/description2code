#include <iostream>
#include <stdio.h>
using namespace std;

int main()
{
    //std::cout.precision(6);
    int n,l,d,i,j,count=1;
    cin>>n>>l>>d;
    double p[n+1];
    int g[n+2];
    p[0]=1;
    g[0]=0;
    p[n+1]=1;
    g[n+1]=d;
    for(i=1;i<n+1;i++)
    {
        cin>>p[i];
    }
    for(i=1;i<n+1;i++)
    {
        cin>>g[i];
        if(g[i]-g[i-1]>l)
        count=0;
    }
    if(count==1){
    for(i=1;i<n+2;i++)
    {
        j=i-1;
        double x=0;
        while(g[i]-g[j]<=l && j>=0)
        {
            if(p[j]>x)
            x=p[j];
            j--;
        }
        p[i]=p[i]*x;
       // cout<<x<<endl;
    }
    // for(i=1;i<n+2;i++)
   // cout << p[n+1]<< endl;
     printf("%.6f\n",p[n+1]);
    }
    else
    cout<<"IMPOSSIBLE"<<endl;
    return 0;
}
