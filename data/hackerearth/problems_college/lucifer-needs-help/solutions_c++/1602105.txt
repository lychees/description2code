#include<bits/stdc++.h>
using namespace std;
int main()
{
ios_base::sync_with_stdio(false);
int n,sum;
cin>>n>>sum;
int a[n];
for(int i=0;i<n;i++)
cin>>a[i];
int flag=0;
sort(a,a+n);
for(int i=0;i<n;i++)
{
if(flag==1)
break;
for(int j=i;j<n;j++)
{
if(binary_search(a+j+1,a+n,sum-a[i]-a[j]))
{
cout<<"YES"<<endl;
flag=1;
break;
}
}
}
if(flag==0)
cout<<"NO"<<endl;
return 0;
}
