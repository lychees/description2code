#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
#define FOR(i,a,b) for(LL i = (LL )a; i < (LL )b ; i++)
#define mp make_pair
#define pb push_back
typedef vector<long long> V;
typedef pair<int,int> pii;
typedef pair<long long,long long> P;


int main()
{
    LL T,N,K,M;
    LL i,j,k,it;
    LL mn,mx,temp,init,pos;
    int flag;
    V v;
    cin>>T;
    while(T--)
    {
        cin>>N>>K;
        for(i=0;i<N;i++)
            {cin>>temp;v.pb(temp);}
        //cout<<"T  "<<T<<endl;
        init =0;
        for(i=0;i<N&&K>0;i++)
        {
            mn=v[i];
            pos=i;
            for(j=i+1;j<=i+K&&j<N;j++)
            {
                if(v[j]<mn)
                {
                    mn=v[j];
                    pos=j;
                }
            }
            temp=v[pos];
            //cout<<"Min  "<<temp<<"  Init  "<<init<<"  it "<<it<<endl;
            for(it=pos;it>i&&it>=0;it--)
            {
                v[it]=v[it-1];
            }
            v[i]=temp;
            K=K-(pos-i);

        }
        for(i=0;i<N;i++)
            cout<<v[i]<<" ";
        cout<<endl;
        v.clear();
    }
	return 0;

    
}
