#include<bits/stdc++.h>
using namespace std;
int main()
{
	string a;
	cin>>(a);
	int i,j,count=0;
	for(i=0; a[i]!='\0'; i++)
	{
		if(a[i]=='l')
		{
			count=1;
			break;
		}
	}
	for(j=i; a[j]!='\0'; j++)
	{
		if(a[j]=='o')
		{
			count=2;
			break;
		}
	}
	for(i=j; a[i]!='\0'; i++)
	{
		if(a[i]=='v')
		{
			count=3;
			break;
		}
	}
	for(j=i; a[j]!='\0'; j++)
	{
		if(a[j]=='e')
		{
			count=4;
			break;
		}
	}
	
	if(count==4)
	cout<<"I love you, too!";
	else
	cout<<"Let us breakup!";
	return 0;
}
