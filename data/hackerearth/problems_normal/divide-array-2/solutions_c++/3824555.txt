#include<iostream>
using namespace std;
int a[(1<<20)+9];
int main()
{
	int n;
	cin>>n;
	int i;
	for(i=1;i<=n;++i)
		cin>>a[i];
	int q;
	long long int  d;
	cin>>q;
	d=1;
	while(q--)
	{	
		int temp;
		cin>>temp;
		d*=temp;
	}
	for(i=1;i<=n;++i)
	{
		if(d<=(long long int)a[i] && d!=0)printf("%lld ",((long long int)a[i])/d);
		else printf("0 ");
	}
	printf("\n");
	return 0;
}
