#include <iostream>
#include <cstring>
#include <string>
using namespace std;
bool isValid(string s) {
	for (int i = 1; i < s.size(); i++) {
		if (s[i] == '1' && s[i] == s[i-1] ) return false;
	}
	return true;
}

string get_bin(int n) {
	string bin = "";
	while (n > 0) {
		if (n%2) bin = '1' + bin;
		else bin = '0' + bin;
		n /= 2;
	}
	return bin;
}
int main()
{
    int T;
    for (cin >> T; T; T--) {
    	int n; cin >> n;
    	for (int i = n; ; i++) {
    		if (isValid(get_bin(i))) {cout << i <<"\n"; break; }
    	}
    }
    return 0;
}
