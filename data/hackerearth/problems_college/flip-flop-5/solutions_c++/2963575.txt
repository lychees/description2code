#include<bits/stdc++.h>
using namespace std;
#define M 100005
char str[M];
int main()
{
  int t;
  cin>>t;
  while(t--)
  {
    cin>>str;
    int n=strlen(str);
    int res=0;
    stack<char> s;
    int i=0;
    while(i<n)
    {
      if(s.empty()||s.top()!=str[i])
      {
         s.push(str[i]);
         i++;
      }
      else if(s.top()==str[i])
      {
        res++;
        i++;
      }
    }
    cout<<res<<endl;
  }
  return 0;
}
