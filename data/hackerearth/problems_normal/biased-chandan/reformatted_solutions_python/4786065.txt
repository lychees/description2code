n = int(raw_input())
i = 0
c = []
while (i < n):
    c.append(raw_input())
    i = i+1
c = [int(x) for x in c]

b = len([i for i, val in enumerate(c) if val == 0])
j = 0
while (j < b):
    p = c.index(0)
    if (p == 0):
        c.pop(p)
        j = j+1
    else:
        c.pop(p)
        c.pop(p-1)
        j = j+1

sum1 = 0
for x in c:
    sum1 = sum1+x

print sum1
