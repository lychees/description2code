#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int main()
{
  int ts;
  cin>>ts;
  while(ts -- )
  {
    int n;
    cin>>n;
    vector<int> s;
    vector<int> a;
    for(int i=0;i<n;i++)
    {
      int a1;
      cin>>a1;
      s.push_back(a1);
    }
    for(int i=0;i<n;i++)
    {
      int a1;
      cin>>a1;
      a.push_back(a1);
    }
    sort(s.begin(),s.end());
    sort(a.begin(),a.end());
    bool flag=true;
    for(int i=0;i<n;i++)
    {
      if(s[i]<=a[i])
      {
        flag=false;
        break;
      }
    }
    if (flag)
     cout<<"YES"<<endl;
     else
       cout<<"NO"<<endl;
  }
}
