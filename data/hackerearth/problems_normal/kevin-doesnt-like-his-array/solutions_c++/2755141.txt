#include <iostream>
using namespace std;
int A[100000];
int C[100001];
int P[100001];
int main()
{
	int N; scanf("%d",&N);
	for(int i = 0; i<N; i++)
	{
		scanf("%d",&A[i]);
		C[A[i]]++;
	}
	
	for(int i = 0; i<N-1; i++)
	{ if(A[i]==A[i+1])P[A[i]]++;}
	int sum = 0; int mx = 0;
	for(int i = 0; i<=100000; i++) { sum += P[i]; if(mx<P[i])mx = P[i];}
	int r = N - (N/2);
	for(int i = 0; i<=100000; i++) if(C[i] > r) { cout<<-1<<endl; return 0;}
	if(sum - mx < mx)
		cout<<mx<<endl;
	else
		cout<<(sum+1)/2<<endl;
    return 0;
}
