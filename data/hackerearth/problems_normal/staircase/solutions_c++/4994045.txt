#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    if(n<0)cout<<"ERROR"<<endl;
    else
    {
    long dp1[n+1];long dp2[n+1];
    dp1[0]=dp1[1]=1;dp1[2]=2;dp2[0]=dp2[1]=1;
    for(int i=3;i<=n;i++)
    dp1[i]=dp1[i-1]+dp1[i-2]+dp1[i-3];
    cout<<"Jack-"<<dp1[n]<<endl;
    for(int i=2;i<=n;i++)
    dp2[i]=dp2[i-1]+dp2[i-2];
    cout<<"Jill-"<<dp2[n]<<endl;
    }

    return 0;
}
