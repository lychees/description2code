inp = raw_input()
stlen = len(inp)
dic = {}
count = 0
for i in inp:
    if i not in dic:
        dic[i] = 1
    else:
        dic[i] = dic[i] + 1
for i in dic.values():
    if i % 2 != 0:
        count = count + 1

if stlen % 2 == 0:
    if count == 0:
        print "YES"
    else:
        print "NO"
else:
    if count == 1:
        print "YES"
    else:
        print "NO"
