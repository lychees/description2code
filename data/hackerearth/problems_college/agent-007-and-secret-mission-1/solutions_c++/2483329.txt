#include <iostream>
using namespace std;

int main()
{
	long long t,n,ans,i;
	cin>>t;
	string str;
	while(t--){
		ans=0;
		i=0;
		cin>>str;
		cin>>n;
		while(str[i]!='\0'){
			if(str[i]=='1')
				ans++;
			i++;
		}
		n=(n*(n+1))/2;
		if(n%2==1)
			ans=str.length()-ans;
		
		cout<<ans<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
