#include <iostream>
using namespace std;
 
int main()
{
    long int n,min,max,i;
    cin>>n;
    long int a[n];
    cin>>a[0];
    min=a[0];
    max=a[0];
    for(i=1;i<n;++i)
    {
    	cin>>a[i];
    	if(a[i]>max)
    	{
    		max=a[i];
    	}
    	if(a[i]<min)
    	{
    		min=a[i];
    	}
    }
    cout<<max*min;
    return 0;
}
