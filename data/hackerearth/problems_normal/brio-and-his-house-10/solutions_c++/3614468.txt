#include <iostream>
#include <stdio.h>
#include <math.h>
using namespace std;

int main()
{
    int t;
    double a,b,c,e,r,s;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b>>c;
    	s=sqrt((a+b+c)*(b+c-a)*(c+a-b)*(a+b-c));
    	r=(a*b*c)/s;
    	e=3.1415*r*r;
    	printf("%.4f\n",e);
    }
    return 0;
}
