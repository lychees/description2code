#include <iostream>
#include<cstdio>
#include<ctype.h>
#include<string>
#include<string.h>
using namespace std;

int main()
{
    long long int t,k=0,i=0,n,count1=0,m;
    
    scanf("%lld",&t);
    while(t--)
    {
        count1=0;
        scanf("%lld%lld",&n,&m);
        while(n>0&&m>0)
        {
            i=n%10;
            k=m%10;
            n=n/10;
            m=m/10;
            if(i+k>9)
                count1++;

        }
        if(count1!=0)
        {
            printf("%lld",count1);
            if(count1>1)
                printf(" carry operations\n");
            else
                printf(" carry operation\n");
        }
        else
            printf("No carry operation\n");


    }
    return 0;
}
