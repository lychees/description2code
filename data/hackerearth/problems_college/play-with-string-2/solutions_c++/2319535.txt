#include<iostream>
#include<string.h>
#include<math.h>
using namespace std;
typedef long long ll;
char pos[1000002];
int main()
{
	int tc;
	cin>>tc;
	while(tc--)
	{
		ll len;
		int i,j;
		cin>>pos;
		len=strlen(pos);
		i=1; j=len-1;
		while(i<len && j>0)
		 {
		 	if(abs(pos[i]-pos[i-1])!=abs(pos[j]-pos[j-1]))
		 		break;
		 	i++; j--;
		 }
		if(i==len) cout<<"Funny\n";
		else cout<<"Not Funny\n";
	}
	return 0;
}
