long long powe(long long a,long long x)
{
	if(x==0)
	return 1;
	if(x==1)
	return a;
	long long ans=powe(a,x/2);
	ans=(ans*ans)%MOD;
	if(x&1)
	{
		ans=(ans*a)%MOD;
	}
	//printf("xx %lld\n",ans);
	return ans;
}
