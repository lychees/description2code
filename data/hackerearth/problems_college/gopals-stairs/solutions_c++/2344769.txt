#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long i,j,k,m,n,t,a,b,c,d;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	a=1;b=1;c=2;d=4;
    	if(n==0) cout<<"1"<<endl;
    	else if(n==1) cout<<"1"<<endl;
    	else if(n==2) cout<<"2"<<endl;
    	else{
    		while(n>2)
    		{
    			d=(a+b+c)%1000000007;
    			a=b;b=c;c=d;n--;
    		}
    		cout<<c<<endl;
    	}
    }
    return 0;
}
