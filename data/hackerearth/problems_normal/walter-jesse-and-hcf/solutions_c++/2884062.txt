#include<bits/stdc++.h>
using namespace std;
long long int gcd(long long int a,long long int b)
{
    if(b==0)
    return a;
    a%=b;
    return gcd(b,a);
}
int main()
{
    int d;
    cin>>d;

    while(d--)
    {
        long long int x,y,k;
        cin>>x>>y>>k;
        long long int g=gcd(x,y);
        set<long long int>s;
        for(long long int i=1;i*i<=g;i++)
        {
            if(g%i==0)
                {
                    s.insert(i);
                    s.insert(g/i);
                }
        }
        int r=s.size();
        if(k>r)
        {
            cout<<"No crymeth today\n";
            continue;
        }
        set<long long int>::reverse_iterator it;
        int count=0;
        for(it=s.rbegin();it!=s.rend();it++)
        {
           // cout<<(*it)<<endl;
            if(count==(k-1))
                break;
                count++;
        }
        cout<<(*it)<<endl;

    }
}
