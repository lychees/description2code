#include <iostream>
#include <unordered_map>
#include <string>
using namespace std;

int main(void)
{
	int t;
	cin >> t;
	unordered_map<string, int> wordMap;
	while(t--)
	{
		string s;
		cin >> s;
		wordMap[s]++;
	}
	int max = 0;
	for(unordered_map<string, int>::iterator i = wordMap.begin(); i != wordMap.end(); i++)
	{
		if(max < i->second)
		{
			max = i->second;
		}
			
	}
	int val = max;
	for(unordered_map<string, int>::iterator i = wordMap.begin(); i != wordMap.end(); i++)
	{
		if(val == i->second)
		{
			max += i->second;
		}
			
	}
	max = max - val;
	
	cout << max << endl;
}
