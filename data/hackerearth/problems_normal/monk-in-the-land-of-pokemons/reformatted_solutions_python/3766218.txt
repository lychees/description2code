
t = int(raw_input())
while t:
    n = int(raw_input())
    count = 0
    hashtable = [0]*1000001
    while n:
        food, pokemon = map(int, raw_input().split())
        hashtable[food] += 1
        if hashtable[pokemon] == 0:
            count += 1
        else:
            hashtable[pokemon] -= 1
        n -= 1
    print count
    t -= 1
