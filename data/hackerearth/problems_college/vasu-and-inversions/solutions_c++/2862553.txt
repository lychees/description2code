#include <stdio.h>
long long int a[100000];
int main()
{
   	long long int t;
   	scanf("%lld",&t);
   	while(t--)
   	{
   		long long int n,k,i,j=0,count=0,p=0;
   		scanf("%lld%lld",&n,&k);
   		for(i=0;i<n;i++)
   		scanf("%lld",&a[i]);
   		for(i=0;i<n;i++)
   		{
   			if(a[i]==0 && k>0)
   			{
   				k--;
   				a[i]=1;
   				//printf("%d ",a[i]);
   			}
	
   		}
   		for(i=0;i<n;i++)
   		{
   			if(a[i]==1)
   			p++;
   		}
   		//printf("%d %d ",count,p);

   		}

   			j=p-1;
   		}

   	}
    return 0;
}
