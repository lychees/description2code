#include<bits/stdc++.h>
using namespace std;

#define PB push_back
#define MP make_pair
#define F first
#define S second
#define LL long long
const int MOD = 1e9 + 7;
const int INF = INT_MAX;

pair<int,int> pnts[1123];

bool func(int st,int bw,int en)
{
	LL n1,d1;
	LL n2,d2;
	LL g;
	n1 = pnts[bw].S - pnts[st].S;
	d1 = pnts[bw].F - pnts[st].F;
	g = __gcd(abs(n1),abs(d1));
	n1 = n1/g;
	d1 = d1/g;

	n2 = pnts[en].S - pnts[st].S;
	d2 = pnts[en].F - pnts[st].F;
	g = __gcd(abs(n2),abs(d2));
	n2 = n2/g;
	d2 = d2/g;
	
	g = __gcd(abs(d1),abs(d2));
	n1 = (n1 * d2)/g;
	n2 = (n2  * d1)/g;

	if(n2 >= n1)
		return true;
	else 
		return false;
}

int main()
{
	int n;
	cin >> n;
	for(int i=0;i<n;i++)
		cin >> pnts[i].F >> pnts[i].S;

	int ans = 0;
	for(int i=0;i<n;i++)
	{
		int j = i + 1;
		if(j >= n)
			continue;
		ans += 1;
		int ma = j;
		j++;
		for(;j<n;j++)
		{
			if(func(i,ma,j))
			{
				ans += 1;
				ma = j;
			}
		}
	}
	cout << ans << endl;

	return 0;
}
