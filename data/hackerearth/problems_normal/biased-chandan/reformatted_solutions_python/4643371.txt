# print 'Hello World!'

n = int(raw_input())
stack = []
count = 0

for i in range(n):
    inputNo = int(raw_input())
    if inputNo == 0:
        if len(stack) != 0:
            stack.pop()
    else:
        stack.append(inputNo)

for j in stack:
    count = count + j

print count
