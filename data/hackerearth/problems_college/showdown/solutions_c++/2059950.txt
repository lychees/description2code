#include <iostream>
#include<algorithm>
#include<map>
#include<string>
#include<vector>
#include<stdio.h>
using namespace std;

int main()
{
		int N;
		vector<int> v1;
		vector<int> v2;
		
		cin>>N;
		
		int temp=N,val;
		while(temp--)
		{
			cin>>val;
			v1.push_back(val);
		}
		temp=N;
		while(temp--)
		{
			cin>>val;
			v2.push_back(val);
		}
		
		sort(v1.begin(),v1.end());
		sort(v2.begin(),v2.end());
		int countt=0;
		for(int i=v2.size()-1;i>=0;i--)
		{
		        
		        int j=v1.size()-1;
		       
				if(v1[j]==v2[i])
				{
					int k=0;
					while(v1[k]>v2[k])
					    k++;
					if(v1[k]==v1[j])
						   countt+=50;
					 v1.erase(v1.begin()+k);
				}
				else if(v1[j]<v2[i])
				{
					
						if(v1[j]==v1[0])
						   countt+=50;
						 
						 v1.erase(v1.begin()+0);
						
				}
				else
				{
					int k=j;
					while(true)
					{
						if(v1[j-1]>v2[i])
						     j--;
						 else
						      break;
					}
					countt+=100;
					v1.erase(v1.begin()+j);
				}
		
			 
		}
		
		
		cout<<countt;
		
		return 0;
}
