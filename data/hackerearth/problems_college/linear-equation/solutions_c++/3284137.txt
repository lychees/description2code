#include <iostream>
using namespace std;

int main()
{
    long long int tc, i, n;
    
    cin >> tc;
    
    while (tc--) {
    	cin >> n;
    	long long int a[n];
    	long long int sum = 0;
    	for (i = 0; i < n; i++) {
    		cin >> a[i];
    		sum += a[i];
    	}
    	
    	sum /= (n - 1);
    	
    	for (i = 0; i < n; i++) {
    		cout << sum - a[i] << " ";
    	}
    	cout << endl;
    	
    }
    
    return 0;
}
