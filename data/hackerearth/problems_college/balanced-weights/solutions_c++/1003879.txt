#include <iostream>
using namespace std;
long long int gcd ( int a, int b )
{
  int c;
  while ( a != 0 ) {
     c = a; a = b%a;  b = c;
  }
  return b;
}
int main()
{
	int t,n,i;
	scanf("%d",&t);
	while(t--) {
		scanf("%d",&n);
		int a[n];
		for(i = 0; i <n; i++) {
			scanf("%d",&a[i]);
		}
		long long int k=gcd(a[0],a[1]);
		a[1] = k;
		for(i = 2; i < n; i++) {
			k = gcd(a[i],k);
		}
		k *= n;
		cout<<k<<endl;
	}
    return 0;
}
