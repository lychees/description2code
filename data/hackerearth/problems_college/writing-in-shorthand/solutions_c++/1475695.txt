#include<iostream>
#include<cstdio>
#include<cstring> 

using namespace std;

int main() {
	int n,m,h,l1,l2,i;
	cin>>n>>m;
	char temp[11];
	char w[m][2][11];
	memset (w,'\0',sizeof(w));
	char s[n][11];
	for(i=0;i<m;i++) {
		cin>>temp;
		h=temp[0]%m;
		while(w[h][0][0]!='\0') {
			h=(h+1)%m;
		}
		strcpy(w[h][0],temp);
		cin>>w[h][1];
	}
	for(i=0;i<n;i++) {
		cin>>s[i];
		h=s[i][0]%n;
		while(strcmp(s[i],w[h][0])!=0) 
			h=(h+1)%m;
		l1=strlen(w[h][0]);
		l2=strlen(w[h][1]);
		if(l1>l2)
			strcpy(s[i],w[h][1]);
	}
	for(i=0;i<n;i++) {
		cout<<s[i]<<" ";
	}
	return 0;
}
