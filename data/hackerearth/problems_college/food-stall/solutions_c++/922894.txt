#include <bits/stdc++.h>
using namespace std;

int main() 
{
    int t,n,i,j,c;
    long long int d;
    cin>>t;
    while(t--) {
        cin>>n>>d;
        long long int a[n],b[n];
        for(i = 0; i < n; i++) {
            cin>>a[i];
        }
        sort(a,a+n);
        c = 0;
        for(i = 0; i < n-1; i++) {
        	for(j = i+1; j < n; j++) {
        		if(a[j] == a[i]+d)
        		c++;
        		else if(a[j] > a[i]+d)
        		break;
        	}
        }
        printf("%d\n",c);
    }
	return 0;
}
