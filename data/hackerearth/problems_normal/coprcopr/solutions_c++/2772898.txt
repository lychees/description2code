#include<bits/stdc++.h>
using namespace std;
#define MAX 10000007
vector <int> phi(MAX,0);
void _phi() {
    phi[1] = 1;
    for(int i=2; i<MAX; i++) {
        if(!phi[i]) {
            phi[i] = i-1;
            for(int j=(i<<1); j<MAX; j+=i) {
                if(!phi[j]) phi[j] = j;
                phi[j] = phi[j]/i*(i-1);
            }
        }
    }
}
vector <int64_t> ph(MAX,0);
void _pre(){
    for(int i=1;i<=MAX;i++){
        ph[i]=ph[i-1]+phi[i];
    }

}
int main() {
 	_phi();
 	_pre();
 	int tt;
 	cin>>tt;
 	while(tt--){
 		int n;

	 	cin>>n;

 		cout<<ph[n]<<endl;
 	}
 return 0;
}
