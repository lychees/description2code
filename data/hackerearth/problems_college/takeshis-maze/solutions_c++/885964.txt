#include<stdio.h>
#include<cstring>
#include<algorithm>

using namespace std;

int main()
{
   int T=0;
   scanf("%d",&T);
   while(T--){
   		int N = 0;
   		scanf("%d",&N);
   		long long int deduct = 0;
   		scanf("%d",&deduct);
   		long long int arr[N][N];
   		memset(arr,0,sizeof(arr));
   		for(int i=0;i<N ;i++){
   			for(int j=0;j<N;j++){
					long long int element = 0;   				
					scanf("%lld",&element);
					if(i==0 && j==0){
   						arr[i][j] = element ;
   					}
   					else if(i==0){
   						arr[i][j] = arr[i][j-1] + element;
   					}
   					else if(j==0){
   						 arr[i][j] = arr[i-1][j] + element;  					
   					}
   					else {
   						arr[i][j] = max(arr[i-1][j],arr[i][j-1])+element;
   					}
   			}
   		}
   		
   		long long int final_val = max(arr[N-2][N-1],arr[N-1][N-2]);
   		
   		if(final_val<deduct )
   			printf("-1\n");
   		else {
   			printf("%llu\n",final_val-deduct);
   		}			
   }
    return 0;
}
