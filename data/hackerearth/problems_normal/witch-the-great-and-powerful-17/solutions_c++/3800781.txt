#include <iostream>
using namespace std;

long long mod(long long a, long long b)
{
	if(a-b >0)
		return a-b;
	else
		return b-a;
}

int main()
{
    int t;
    cin>>t;
    for(int i = 0; i<t;i++)
    {
    	long long a,b,c;
    	cin>>a>>b>>c;
    	long long ans = mod(a,b) > mod(b,c) ? mod(a,b) -1 : mod(b,c) - 1;
    	cout<<ans<<endl;
    }
    return 0;
}
