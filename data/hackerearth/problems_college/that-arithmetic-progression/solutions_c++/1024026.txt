void doit(){

    if (n<3){
        if (n==1)
        printf("-1");
        else {
            if (n==2){
                if (a[0]==a[1])
                  printf("1\n%d",a[0]);
                else {
                    d=a[1]-a[0];
                    k=(a[0]+a[1])/2;
                    if ((k-a[0])==(a[1]-k)){
                        printf("3\n%d %d %d",a[0]-d,k,a[1]+d);
                    }
                    else
                        printf("2\n%d %d",a[0]-d,a[1]+d);
                }
            }
        }
        return;
    }
    d=a[n-1]-a[0];
    for (i=1;i<n;i++)
      d=min(d,a[i]-a[i-1]);
    j=0;k=0;
    for (i=1;i<n;i++)
    {
        if (a[i]-a[i-1]==d*2)
        {
            j++; k=(a[i]+a[i-1])/2;
        }
        if ((a[i]-a[i-1]!=d)&&(a[i]-a[i-1]!=(d*2)))
        {
            printf("0"); return;
        }
    }
    if(j==1)
      printf("1\n%d",k);
    else
    {
        if (j==0)
            printf("2\n%d %d",a[0]-d,a[n-1]+d);
        else if(d==0)
            printf("1\n%d",a[0]);
        else
            printf("0");
    }
}

int main(){
