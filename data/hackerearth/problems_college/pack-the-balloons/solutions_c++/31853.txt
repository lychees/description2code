#include <iostream>
#include <cstdio>
using namespace std;

int main ()
{
        long int t;
        unsigned long long int l, b, h, v;
        scanf ("%ld", &t);
        while (t--) {
                scanf ("%llu %llu %llu", &l, &b, &h);
                v = l < b ? l : b;
                v = v < h ? v : h;
                if (v % 2 == 0)
                printf ("%llu.0\n", v/2);
                else
                        printf ("%llu.5\n", v/2);

        }

        return 0;
}
