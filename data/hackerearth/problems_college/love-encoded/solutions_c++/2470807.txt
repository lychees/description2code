#include <string>
#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	int n;
    	cin >> n;
    	string s;
    	cin >> s;
    	int i,l = s.size();
    	for(i=0;i<l;i++)
    	{
    		s[i]-=n;
    		while(s[i]<65)
    		{
    			s[i]+=26;
    		}
    	}
    		for(i=0;i<l;i++)
    	{
    		cout << s[i];
    	}
    	cout << endl;
    }
    return 0;
}
