#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	int n;
	cin>>n;
	
	int a[n];
	
	for(int i=0;i<n;i++){
		cin>>a[i];
	}
		
	bool happy=true;

	for(int i=0;i<n;i++){
		for(int j=0;j<n-1;j++){
			if(a[j]<a[j+1]){
				continue;
			}else{
				if(a[j]-a[j+1]==1){
					happy=false;
					break;	
				}else{
					a[j]--;
					a[j+1]++;
					
					int temp=a[j];
					a[j]=a[j+1];
					a[j+1]=temp;
				}
			} 
		}
	}
	
	if(happy==false){
		cout<<":(";
	}else{
		for(int i=0;i<n;i++){
			cout<<a[i]<<" ";
		}
	}
	return 0;
}
