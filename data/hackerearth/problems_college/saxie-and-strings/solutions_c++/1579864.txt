#include <iostream>
#include <cstring>
#include <cmath>
#include <vector>
#include <algorithm>
#define LL long long
#define N 200
using namespace std;
int main()
{
	int t , i , j;
	cin>>t;
	while(t--)
	{
		string str , temp;
		int n = 0;
		cin>>str;
		for(i=0;str[i]!='\0';i++)
		{
			temp = temp+str[i];
			
			string rstemp;
			rstemp = temp;
			reverse(rstemp.begin() , rstemp.end());
			std::size_t found = str.find(rstemp);
			if(found==std::string::npos)
			{
				cout<<"NO\n";
				n = 1;
				break;
			}
		}
		if(n==0)
		{
			cout<<"YES\n";
		}
	
	}
	return 0;
}
