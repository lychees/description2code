using namespace std;

#define REP(i, n) for ((i) = 0; (i) < (int)(n); (i)++)
#define foreach(c, itr)                                                        \
  for (__typeof((c).begin()) itr = (c).begin(); itr != (c).end(); itr++)

int N, a[110][110];

int main(void) {
  int H, W, i, j;

  cin >> H >> W;

  if (H + W >= 3 && H + W <= 4) {
    cout << -1 << endl;
  } else {
    REP(i, H) REP(j, W) if (i % 2 != j % 2) {
      a[i][j] = N;
      N++;
    }
    REP(i, H) REP(j, W) if (i % 2 == j % 2) {
      a[i][j] = N;
      N++;
    }
    REP(i, H) REP(j, W) {
      cout << a[i][j] + 1;
      if (j == W - 1)
        cout << endl;
      else
        cout << ' ';
    }
  }

  return 0;
}
