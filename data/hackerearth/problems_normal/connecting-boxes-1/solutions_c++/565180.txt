#include<iostream>
using namespace std;
//-----------------------------------------

int main()
{
	int n;
	cin>>n;
	int arr[n+1];
	for(int i=0;i<=n;i++)
		arr[i]=0;
	
	int q;
	cin>>q;
	
	int b,a;
	while(q--)
	{
		int flag=0;
		cin>>a>>b;
		if(b<a)
		{
			int temp=a;
			a=b;
			b=temp;
		}
	
		if(arr[a]!=0 || arr[b]!=0)
		flag=1;
		
		else if(a==b)
		flag=1;
		else 
		{
			for(int i=1;i<a;i++)
			{
			//	cout<<"P";
				if(arr[i]!=0 && arr[i]<b && a<arr[i])
				{
			//		cout<<"k"<<i<<endl;
					flag=1;
					break;
				}
			}
			
			if(flag==0)
			{
					for(int i=a+1;i<b;i++)
				{
				//	cout<<"P";
					if(arr[i]!=0 && arr[i]>b)
					{
				//		cout<<"k"<<i<<endl;
						flag=1;
						break;
					}
				}
			}
		}
		if(flag==0)
		{
			cout<<"YES\n";
			arr[a]=b;
			arr[b]=a;
		}
		else
		cout<<"NO\n";
		 
	}
	
}
