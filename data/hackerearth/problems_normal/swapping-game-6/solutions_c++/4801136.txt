#include <iostream>
#include<string>
#include<vector>
using namespace std;


string change(string s)
{
	int n=s.length();
	string c=s;
	for(int i=0;i<n/2;i++)
	{
		c[2*i]=s[i];
		c[2*i+1]=s[n-i-1];
	}
	if(n%2==1)
	{
		c[n-1]=s[n/2];
	}
		return c;
	
	
}

int main()
{
    //cout << "Hello World!" << endl;
    long int k;
    cin>>k;
    string s;
    cin>>s;
    
    int p=1;
    string a=change(s);
    while(a!=s)
    {
    	p++;
    	a=change(a);
    }
    
    int left=p-k%p;
    for(int i=1;i<=left;i++)
    {
    	a=change(a);
    }
    cout<<a;
    
    
    return 0;
}
