const int N = 10000;
bool isPrime[N + 5];

void sieve() {
    for(int i = 4; i <= N; i += 2) {
        isPrime[i] = 1;
    }
    for(int i = 3; i * i <= N; i += 2) {
        if(!isPrime[i]) {
            for(int j = i << 1; j <= N; j += i) {
                isPrime[j] = 1;
            }
        }
    }
}

int main() {
    sieve();
    int n;
    cin >> n;
    for(int i = 1; i <= n; i++) {
        if(isPrime[i]) {
            cout << i << ' ';
        }
    }
    End;
}
