t = int(raw_input())
for w in range(t):
    s = raw_input()
    d = dict()
    cnt = 0
    for i in s:
        if i in d:
            d[i] += 1
        else:
            d[i] = 1
    for i in d:
        if d[i] % 2 == 0:
            cnt += 1
    if cnt == len(d):
        print 1
    else:
        print -1
