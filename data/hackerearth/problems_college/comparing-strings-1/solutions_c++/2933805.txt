#include <iostream>
#include<string.h>
using namespace std;

int main()
{
  	int t;
  	cin>>t;
  	while(t--)
  	{
  		char str[100000],str1[100000];
  		cin>>str;
  		cin>>str1;
  		int i,j,hash[26]={0},flag=0;
  		for(i=0;i<strlen(str);i++)
  		{
  			hash[str[i]-97]++;
  			
  		}
  		for(j=0;j<strlen(str1);j++)
  		{
  			if(hash[str1[j]-97]!=0){
  				cout<<"YES"<<endl;
  				flag=1;
  				break;
  			}
  			
  		}
  		if(flag!=1)
  			cout<<"NO"<<endl;
  	}
    return 0;
}
