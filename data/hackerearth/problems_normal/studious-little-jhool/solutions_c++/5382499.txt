int main()
{
    int t,n;
    cin>>t;
    while(t--){
    	cin>>n;
    	int ans = 0,minN=INT_MAX;
    	while(n>=10){
    		if(n%10==0) minN=min(minN,ans+(n/10));
    		n-=12; ans++;
    	}
    	if(n==0) minN=min(minN,ans);
    	if(minN==INT_MAX) cout<<"-1"<<endl;
    	else cout<<minN<<endl;
    }
    return 0;
}
