#include <iostream>
#include <string>

using namespace::std;


int main(int argc, const char * argv[]) {
    
    int t;
    cin>>t;
    while (t--) {
        
        int n,cherries=0;
        cin>>n;
        bool arr[1000][1000];
        for (int i=0; i<n; i++) {
            
            string str;
            cin>>str;
            for (int j=0; j<n; j++) {
                
                if (str[j]=='#') {
                    
                    arr[i][j] = true;
                    cherries++;
                }
                else
                    arr[i][j] = false;
            }
        }
        if (cherries%2!=0) {
            
            cout<<"NO"<<endl;
        }
        else{
            
            int totalRow = 0,totalColumn=0,alert=0;
            cherries = cherries/2;
            for (int i=0; i<n; i++) {
                
                for (int j=0;j<n; j++) {
                    
                    if (arr[i][j]) {
                        totalRow++;
                    }
                    if (arr[j][i]) {
                        totalColumn++;
                    }
                }
                if (totalRow==cherries||totalColumn==cherries) {
                    
                    cout<<"YES"<<endl;
                    alert++;
                    break;
                }
            }
            if (alert==0) {
                cout<<"NO"<<endl;
            }
        }
       
    }
}
