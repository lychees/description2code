#include<iostream>
#include<cstring>
#include<cstdio>
using namespace std;
void swap(char *x,char *y)
{
	char temp;
	temp=*x;
	*x=*y;
	*y=temp;
}
void permute(char *a,int i,int n)
{
	int j;
	if(i==n)
		printf("%s\n",a);
	else
	{
		for(j=i;j<=n;j++)
		{
			swap((a+i),(a+j));
			permute(a,i+1,n);
			swap((a+i),(a+j));
		}
	}
}
int main()
{
	char str[100000];
	cin>>str;
	int q;
	cin>>q;
	while(q-->0)
	{
		int l=0,r=0;
		cin>>l>>r;
		permute(str,l,r);
		cout<<endl;
	}
	return 0;
}
