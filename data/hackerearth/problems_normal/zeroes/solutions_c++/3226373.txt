#include <bits/stdc++.h>

using namespace std;

long arr[100005];
long factor[100005];

void precalc(long n){
	long t,t1;
	for(long i=0;i<n;i++){
		t=arr[i];
		t1=sqrt(t);
		for(long j=2;j<=t1;){
			if(t%j==0){
				factor[j]++;
				t/=j;
			}
			else
				j++;
		}
		factor[t]++;
	}
}

int main()
{
	ios::sync_with_stdio(false);
    long n,q,x;
    cin>>n;
    for(long i=0;i<n;i++)
    	cin>>arr[i];
    memset(factor,0,100005);
    precalc(n);
   
    cin>>q;
    while(q--){
    	cin>>x;
    	long ans=INT_MAX,x1=sqrt(x);
    	map<long,long> q;
        map<long,long>::iterator it;
    	
    	long j=2;
    	while(j<=x1){
			if(x%j==0){
				q[j]++;
				x/=j;
			}
			else
				j++;
		}
		if(x>1)
			q[x]++;
			
		
    	for(it=q.begin();it!=q.end();it++)
            ans=min(ans,factor[it->first]/it->second);
    	cout<<ans<<"\n";
    }
    return 0;
}
