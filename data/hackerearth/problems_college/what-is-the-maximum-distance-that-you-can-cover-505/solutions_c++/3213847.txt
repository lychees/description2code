#include <iostream>
#include <vector>
#include <limits.h>
#include <stdio.h>
#include <math.h>
using namespace std;
double maxAt(std::vector<double>& vector_name) {
    double max = INT_MIN;
    for (auto val : vector_name) {
         if (max < val) max = val;
    }
    return max;
}
int main()
{
	vector<double> speeds;
	vector<double> fuels;
	vector<double> ratios;
	double speed,fuel,ratio,total_fuel;
	int n;
	cin >> n;
	for(int i=0;i<n;i++){
		cin >> speed;
		speeds.push_back(speed);
	}
	for(int i=0;i<n;i++){
		cin >> fuel;
		fuels.push_back(fuel);
	}
	cin >>total_fuel;
	for(int i=0;i<n;i++){
		ratio = speeds.at(i)/fuels.at(i);
		ratios.push_back(ratio);
	}
	//cout.setf(ios::fixed,ios::floatfield);
    //cout.precision(4);
    //cout << total_fuel*maxAt(ratios)<<endl;
    double res =total_fuel*maxAt(ratios);
    //printf("%.3f\n",);
    printf("%.3f", floor(1000 * res) / 1000);
    return 0;
}
