#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int steps=0;
    	while(n!=1)
    	{
    		if(n%3==0)
    		{
    			while(n%3==0)
	    		{
	    			n=n/3;
	    			steps++;
	    		}	
    		}
    		else if(n%2==0)
    		{
    			while(n%2==0)
	    		{
	    			n=n/2;
	    			steps++;
	    		}
    		}
    		else 
    		{
    			n=n-1;
    			steps=steps+1;
    		}
    	}
    	cout<<steps<<endl;
    }
    return 0;
}
