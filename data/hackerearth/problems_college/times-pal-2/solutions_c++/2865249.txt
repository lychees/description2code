#include<bits/stdc++.h>
using namespace std;
 
int main()
{
    int t,c,i,len;
    string s1,s2;
    cin>>t;
    while(t--)
    {
    	c=0;
    	cin>>s1>>s2;
    	len=s1.length();
    	for(i=0;i<len;i++)
    	{
    		string s=s1.substr(0,i) + s2 + s1.substr(i,len);
    		string s3=s;
    		reverse(s.begin(),s.end());
    		if(s==s3)
    		c++;
    	}
    	string s4=s1+s2;
    	string s4rev=s4;
    	reverse(s4.begin(),s4.end());
    	if(s4==s4rev)
    	c++;
    	
    	cout<<c<<endl;
    }
    return 0;
}
