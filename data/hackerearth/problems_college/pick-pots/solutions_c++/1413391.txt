#include <bits/stdc++.h>
#define lli long long
#define M 1000000007

using namespace std;

int dp[1000005];

int main()
{
	int t,n;
	scanf("%d", &t);
	dp[1] = 2;
	dp[2] = 3;
	for ( int i = 3; i <= 1000000; i++ ) dp[i] = (dp[i-1] + dp[i-2])%M;
	while ( t-- ) {
		scanf("%d", &n);
		lli ans = dp[n];
		ans--;
		if ( ans < 0 ) ans += M;
		printf("%d\n", ans);
	}
	return 0;
}
