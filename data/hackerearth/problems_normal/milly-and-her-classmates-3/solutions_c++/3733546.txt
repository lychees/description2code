#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int t,*a,*b,n,i,len,found,start_point,end_point;
	cin>>t;
	while(t--)
	{
		cin>>n;
		a=new int[n];
		b=new int[n];
		for(i=0;i<n;i++)
		cin>>a[i];
		for(i=0;i<n;i++)
		b[i]=a[i];
		sort(b,b+n);
		len=0;
		for(i=0;i<n;i++)
		if(a[i]==b[i])
		len++;
		if(len==n)
		cout<<"-1 -1"<<endl;
		else
		{
		for(i=0;i<n;i++)
		if(a[i]!=b[i])
		{
		start_point=i;
		break;
	    }
	    for(i=n-1;i>=0;i--)
	    if(a[i]!=b[i])
	    {
	    	end_point=i;
	    	break;
		}
		found=0;
		for(i=end_point;i>start_point;i--)
			if(a[i]>a[i-1])
			{
			found=1;
			break;
		    }
		    if(found==1)
		    cout<<"-1 -1"<<endl;
		    else
		    cout<<start_point+1<<" "<<end_point+1<<endl;
	}		
	}
}
