#include <bits/stdc++.h>
#include <algorithm>
using namespace std;

int main()
{
	int t,i,k,min;
    vector<int> inp;
    while(cin>>t) inp.push_back(t);
    make_heap(inp.begin(),inp.end());
    t=inp.size();
    //for(int i=0;i<t;i++) cout<<inp[i]<<" ";
   // cout<<endl;
    k=t/2;
    min=inp[k];
    //cout<<min<<endl;
    for(i=k+1;i<t;i++)
    {
    	if(min>inp[i]) min=inp[i];
    }
    cout<<min<<endl;
    return 0;
}
