#include<iostream>
#include<cmath>
using namespace std;

int main()
{
    int i,T,N,sum,temp;
    long long int ans;
    cin >> T;

    while(T > 0)
    {
        sum = 0;

        cin >> N;

        for(i=0;i<3;i++)
        {
            cin >> temp;
            sum = sum + temp;
        }

        ans = pow(sum,N);

        cout << ans;
        cout << "\n";

        T--;
    }

    return 0;
}
