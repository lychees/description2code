#include <iostream>
#include<string.h>
using namespace std;

int main()
{
	int t;
	char s[100];

	cin>>t;

	while(t--)
	{
		int i=0,j=0,start=0;

		cin>>s;
		j=strlen(s)-1;
		while(i<=j)
		{
			if(s[i]==s[j])
			{
				i++;
				j--;
			}
			else
			{
				start++;
				i=start;
				j=strlen(s)-1;
			}

		}

		if(start==0)
			cout<<strlen(s)<<endl;
		else
			cout<<strlen(s)+start<<endl;	
		
	}

    return 0;
}
