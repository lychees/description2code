#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    char a[100000];
    while(t--){
    cin>>a;
    int l=strlen(a);
    int c=0;
    for(int i=0;i<l-1;i++){
        if(a[i]!=a[i+1])
            continue;
        else
        {
            c++;
        }
    }
    cout<<c<<endl;
    }
}
