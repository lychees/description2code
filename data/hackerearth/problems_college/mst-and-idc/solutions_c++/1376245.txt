#include <iostream>
#include <algorithm>
#include <cmath>
#define lli long long int
using namespace std;


int main()
{
    lli t,n,sum,ans,avg,min=1000000000000000000,max=0;
    cin>>t;
    while(t--)
    {
    	min=1000000000000000000,max=0;
    	sum=0;
    	cin>>n;
    	lli arr[n];
    	for(lli i=0;i<n;i++)
    	{
    		cin>>arr[i];
    		sum+=arr[i];
    		if(max<arr[i])
    		{
    			max=arr[i];
    		}
    		if(min>arr[i])
    		{
    			min=arr[i];
    		}
    	}
    	avg=sum/n;
    	//ans=abs(max-avg)>abs(min-avg)?abs(max-avg):abs(min-avg);
    	if(abs(max-avg)==abs(min-avg))
    	{
    		ans=abs(max-avg);
    		cout<<4*ans<<endl;
    	}
    	else
    	{
    		cout<<2*(abs(max-avg))+2*(abs(min-avg))<<endl;
    	}
    	//if(((float)sum/(float)n)==avg)
    	//{
    	//	cout<<a*ans<<endl;
    	//}
    	
    	
    }
    return 0;
}
