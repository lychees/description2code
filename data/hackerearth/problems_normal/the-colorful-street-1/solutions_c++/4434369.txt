#include<bits/stdc++.h>

#define mod 1000000007
#define ll long long
#define INF (ll)(1e18)
#define mp make_pair
#define fill(a,v) memset(a, v, sizeof a)
using namespace std;
ll N;
ll dp[25][3];
ll arr[25][3];
ll fun(ll row,ll column)
{
	ll i;
	if(row==N)
	return 0;
	if(dp[row][column]!=-1)
	return dp[row][column];
	ll temp=INF;
	for(i=0;i<3;i++)
	{
		if(i!=column)
		{
			temp=min(temp,arr[row][i]+fun(row+1,i));
		}
	}
	dp[row][column]=temp;
	return dp[row][column];
}
int main()
{
	ll T;
	cin>>T;
	while(T--)
	{
		ll i,j;
		fill(dp,-1);
		fill(arr,0);
		
		cin>>N;
		for(i=0;i<N;i++)
		{
			for(j=0;j<3;j++)
			cin>>arr[i][j];
		}
		ll ans=INF;
		for(i=0;i<3;i++)
		ans=min(ans,arr[0][i]+fun(1,i));
		cout<<ans<<endl;	
	}
	return 0;
}
