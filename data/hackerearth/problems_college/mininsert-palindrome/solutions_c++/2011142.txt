#include <bits/stdc++.h>
using namespace std;
int mininsertion(char str[],int l,int h)
{
    if(l>h)
        return INT_MAX;
    if(l==h)
        return 0;
if(l==h-1)
{
if( str[l]==str[h])
return 0;
else
return 1;
}
if(str[l]==str[h])
return mininsertion(str,l+1,h-1);
else
return min(mininsertion(str,l+1,h)+1,mininsertion(str,l,h-1)+1);
}
int main()
{
    //cout << "Hello World!" << endl;
    long long t;
    cin>>t;
    while(t--)
    {
    	int  n;
    	cin>>n;
        char str[101];
    	scanf("%s",&str);
    	cout<<mininsertion(str,0,n-1)<<endl;;
    }
    return 0;
}
