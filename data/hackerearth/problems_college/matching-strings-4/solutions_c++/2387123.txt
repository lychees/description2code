#include <iostream>
using namespace std;
#include <string>
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string a;
    	string b;
    	cin>>a;
    	cin>>b;
    	int cnt = 0;
    	int i = 0;
    	while(a[i]!='\0' && b[i]!='\0')
    	{
    		if(a[i]==b[i])++cnt;
    		++i;
    	}
    	cout<<cnt<<endl;
    	
    }
    return 0;
}
