#include <bits/stdc++.h>
#define LL long long int

#define ff first

using namespace std;
int cnt[26];
int main(){
	string s;
	int n,i,t;
	cin >> t;
	while(t--){
		memset(cnt,0,sizeof(cnt));
		cin >> s;
		n=s.length();
		for(i=0;i<n;i++)
			cnt[s[i]-'a']++;
		int val=cnt[0];
		int f=0;
		for(i=1;i<26;i++){
			if(cnt[i]!=val)f=1;
		}
		if(f==0)
			cout  << "Yes"<< endl;
		else
			cout << "No" << endl;
	}



	return 0;
}
