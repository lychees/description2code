#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	cin>>t;	
	while(t--)
	{
		int b[3];
		cin>>b[0]>>b[1]>>b[2];
		if(b[0]==0&&b[1]==0&&b[2]==0)
		{
			cout<<"0"<<endl;
			continue;
		}
		sort(b,b+3);
		int f=0,k,l;
		k=b[2];
		l=b[1];
		int c=0;
		while(k>0&&l>0)
		{
			if(f==0)
			{
				c++;k--;
				f=1;	
					
			}
			else if(f==1)
			{
				c++;l--;
				f=0;
					
			}
		}
		cout<<c+1<<endl;
	}
	
}
