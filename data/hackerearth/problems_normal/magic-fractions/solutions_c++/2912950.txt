#include <bits/stdc++.h>

using namespace std;

long long int power(int p){
	long long int ans=1;
	for(int i=1;i<=p;++i)
		ans*=2;
	return ans;
}
int main(){
 long int n,i,j,dp[505];
 dp[1]=0;
 cin>>n;
 long int isprime[505]={0};
 for(i=1;i<=500;++i)
 	isprime[i]=1;
 for(i=2;i<=23;++i){
 	if(isprime[i]){
 	for(j=i*i;j<=500;j+=i)
 		isprime[j]=0;
 	}
 }
long long int pcount =0;
 long long int ans=0;
 for(i=2;i<=n;++i){
 	pcount+=isprime[i];
 	ans+=power(pcount-1);

 }
 cout<<ans<<endl;
 return 0;
}
