#include <iostream>
using namespace std;

#define ll long long

int main()
{
    int test = 0;
 
  	cin>>test;
  	while(test--){
  		int n , pcol , prow , krow , kcol , turn;
  		cin>>n>>prow>>pcol>>krow>>kcol>>turn;
  		
  		int king_step_req = abs(kcol - pcol);
  		int king_row = krow + king_step_req ;
  		int pawn_row = prow + king_step_req ;
  		if(turn == 1 )
  			pawn_row--;
  			
  		if(pawn_row > king_row || pawn_row > n)
  			cout<<"White Wins\n";
  		else
  			cout<<"Draw\n";
  	}
    return 0;
}
