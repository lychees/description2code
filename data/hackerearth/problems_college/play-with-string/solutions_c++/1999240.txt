#include <iostream>
#include <cstring>
#include <bits/stdc++.h>
using namespace std;

char * rev(char *str,int len)
{
	char *rev=new char[len];
	rev[len-1]='\0';
	for(int i=0;i<len-1;i++)
	{
		rev[i]=str[len-i-2];
	}
	//cout<<rev;
	return rev;
}

int main()
{
	int n;
	cin>>n;
	while(n--)
	{
		string str;
		cin>>str;
		int l=str.length();
		char * temp=new char[l+1];
		strcpy(temp,str.c_str());
		char *reverse=rev(temp,l+1);
	//	cout<<reverse;
		int flag=0;
		for(int i=1;i<l;i++)
		{
			int a=abs(temp[i]-temp[i-1]);
			int b=abs(reverse[i]-reverse[i-1]);
			//cout<<a<<b<<endl;
			if(a!=b){
				cout<<"Not Funny\n";
				flag=1;
				break;
			}
		}
		if(!flag)
			cout<<"Funny\n";
	}
    //cout << "Hello World!" << endl;
    return 0;
}
