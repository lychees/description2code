#define S(x) scanf("%d",&x)

string s;
string ans;

int main() {
  int t;
  S(t);
  while(t--) {
    int n,m,p,x;
    scanf("%d%d%d%d",&n,&m,&p,&x);
    cin >> s;
    ans.clear();
    if(n == 1) {
      cout << s << "\n";
    }
    n--;

    for(int j = 31; j >= 0; j--) if((n>>j) & 1) {
      rep(i,0,m) {
        int b1 = s[i] - '0';
        int b2 = s[((i - ((1<<j) * 1LL * p) % m) + m) % m] - '0';
        int res = b1 ^ b2;
        ans += res + '0';
      }
      s = ans;
      ans.clear();
    }
    if(x) {
      rep(i,0,s.size()) s[i] = 1 - (s[i] - '0') + '0';
    }
    cout << s << "\n";
  }
  return 0;
}
