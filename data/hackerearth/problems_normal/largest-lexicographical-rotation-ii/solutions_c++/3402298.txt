#include <bits/stdc++.h>

using namespace std;


int minmove(string &inp){
	int len = inp.size();
    int ret=0,l=0,idx=1;
    while(ret+l+1<len && idx<len){
        int cp=inp[ret+l]-inp[(idx+l)%len];
        if(!cp) l++;
        else if(cp<0) idx+=l+1,l=0;
        else ret=max(ret+l+1,idx), idx=ret+1, l=0;
    }
    return ret;
}
int main() {
    
    string inp;
    cin>>inp;
    for(int i=0; i<inp.size(); i++)
    {
    	inp[i] = 122-inp[i]+98;
    }
    int start = minmove(inp);
    
    for(int i=0; i<inp.size(); i++)
    {
    	inp[i] = 122-inp[i]+98;
    }
    
    int len = inp.size();
    for(int i=0; i<inp.size(); i++)
    {
    	cout<< inp[start];
    	start = (start+1)%len;
    }
    cout<<endl;
    
    return 0;
}
