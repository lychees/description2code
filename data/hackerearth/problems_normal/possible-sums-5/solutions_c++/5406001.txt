#include<stdio.h>
#include<set>
#include<vector>
#include<iostream>
#include<string.h>
using namespace std;

set<int> ans;

void subset(int *arr, int n, int sum)
{
	vector<vector<bool>> M(n + 1, vector<bool>(sum + 1));

	for (int i = 0; i <= n; i++)
		M[i][0] = true;

	for (int j = 1; j <= sum; j++)
		M[0][j] = false;

	for (int i = 1; i <= n; i++)
	{
		for (int j = 1; j <= sum; j++)
		{
			if (arr[i - 1] > j)
				M[i][j] = M[i - 1][j];
			else
			{
				M[i][j] = M[i - 1][j] ||
					M[i - 1][j - arr[i - 1]];
			}

			if (M[i][j])
				ans.insert(j);
		}
	}
}


int main()
{
	int test;
	scanf("%d", &test);

	while (test>0)
	{
		ans.clear();
		ans.insert(0);
		int n, i, sum = 0;
		scanf("%d", &n);
		int a[n];
		for (i = 0; i<n; i++)
		{
			scanf("%d", &a[i]);
			sum += a[i];
		}
		subset(a, n, sum);
		printf("%ld\n", ans.size());
		test--;
	}
}
