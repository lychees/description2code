#include <iostream>
#include <string>
using namespace std;

int main()
{
    int t;cin>>t;
    while(t--){
    	string s;cin>>s;
    	int i,ans=0,total=0;
    	bool flag=false;
    	for(i=0;s[i];i++){
    		if(s[i]=='*'){
    			flag=false;
    			total+=ans;
    			ans=0;
    		}		
    		else if(s[i]=='X'){	
    			flag=true;
    			ans=0;
    		}
    		if(!flag&&s[i]=='O')
    			ans++;
    	}
    	if(!flag)
    		total+=ans;
    	cout<<total<<endl;
    }
    return 0;
}
