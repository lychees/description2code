#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	int n;
    	cin >> n;
    	int i,temp;
    	vector<int> vec;
    	for(i=0;i<n;i++)
    	{
    		cin >> temp;
    		vec.push_back(temp);
    	}
    	sort(vec.begin(),vec.end());
    	vector<int> fin;
    	while(!vec.empty())
    	{
    		int temp = vec.front()+vec.back();
    		vec.pop_back();
    		vec.erase(vec.begin());
    		fin.push_back(temp);
    	}
    	sort(fin.begin(),fin.end());
    	//int ans = 
    	cout << fin.back()-fin.front() << endl;
    }
    return 0;
}
