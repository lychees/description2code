#include <iostream>
#include<bits/stdc++.h>
# define ll long long int
using namespace std;
vector <int> adj[100001];
ll nodes,edges;
void bfs()
{
	bool visited[100005]={0};
	ll level[100005],u,v,i;
	for(i=0;i<=100005;i++)
	{
		level[i]=INT_MAX;
	}
	level[0]=0;
	queue <int> q;
	q.push(0);
	while(!q.empty())
	{
		u=q.front();
		q.pop();
		if(visited[u])
		{
			continue;
		}
		for(i=0;i<adj[u].size();i++)
		{
			v=adj[u][i];
			if(!visited[v])
			{
				level[v]=min(level[v],level[u]+1);
				q.push(v);
			}
		}
		visited[u]=1;
	}
	for(i=1;i<nodes;i++)
	{
		if(level[i]==INT_MAX)
		{
			cout<<"-1"<<endl;
		}
		else
		cout<<level[i]<<endl;
	}
}
int main()
{
   ll i,j,u,v;
   cin>>nodes>>edges;
   for(i=1;i<=edges;i++)
   {
   	cin>>u>>v;
   	adj[u].push_back(v);
   	adj[v].push_back(u); 
   }
   bfs();
    return 0;
}
