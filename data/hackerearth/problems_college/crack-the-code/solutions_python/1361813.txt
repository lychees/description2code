# Start it Biches

t = input()

a = 0
b = 1
fibo = [a,b]
while a<100000:
    a,b = b,a+b
    fibo.append(b)

for _ in xrange(t):
    x,y = map(int,raw_input().split())
    index = 0
    while fibo[index] < x:
        index +=1
    x_in = index
    while fibo[index] < y:
        index +=1
    if fibo[index] != y:
        index -=1
    y_in = index
    sample = fibo[x_in:y_in+1]
    #print sample
    if len(sample)%2==0:
        print "VALID"
    else:
        print "INVALID"
