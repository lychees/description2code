#include<string.h>

{

    	for(i=0;i<l;i++)

    		cout<<"b3";
    		if(a[i]=='7')
    		cout<<"acf";
    		if(a[i]=='9')
    		cout<<"c6a";
    	
    	}
    	cout<<"\n";
    }
    return 0;
}
