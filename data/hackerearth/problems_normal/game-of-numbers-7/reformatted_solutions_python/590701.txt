t = input()
while t:
    a, b = map(int, raw_input().split())
    if a > b:
        a, b = b, a
    n = [0]*(b-a+1)
    for i in range(2, int(b**0.5)+1):
        num = i*i
        mul = a/num
        if a % num != 0:
            mul += 1
        while mul*num <= b:
            n[mul*num-a] = 1
            mul += 1
    print n.count(0)
    t -= 1
