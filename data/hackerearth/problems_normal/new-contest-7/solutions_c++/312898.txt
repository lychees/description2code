#include<bits/stdc++.h>
using namespace	std;
typedef long long int i64;
double nCr[1001][1001];
int main()
{
    for(int i=0; i<1001; i++)
    {
        nCr[i][i]=nCr[i][0] = 1;
        for(int j = 1; j<=i/2; j++)
        {
            nCr[i][i-j]=nCr[i][j] = nCr[i-1][j] + nCr[i-1][j-1];
        }
        if(i%2) nCr[i][i/2 + 1] = nCr[i-1][i/2 +1] + nCr[i-1][i/2];
    }
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int s,n,m,k;
        scanf("%d%d%d%d",&s,&n,&m,&k);
        double sum =0;
        int mm = min(m,n);
        for(int i=k;i<=mm;i++)
        {
            if(n-i>s-m){continue;}
            sum += nCr[m][i]*nCr[s-m][n-i];
            //cout<<i<<" "<<m<<" "<<sum<<endl;
        }
        sum/=nCr[s][n];
        printf("%.6f\n",sum);
    }
    return 0;
}
