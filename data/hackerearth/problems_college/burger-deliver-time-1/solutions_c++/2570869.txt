#include <bits/stdc++.h>
#define ll long long
using namespace std;
typedef struct node
{
	ll int id,t;//id of order & delivery time
}node;
bool my(node a,node b)
{
	if(a.t<b.t)
	return true;
	else if(a.t==b.t)
	{
		if(a.id<b.id)
		return true;
	}
	return false;
}
node r[100010];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int n;
	cin>>n;
	ll int i,x,y;
	for(i=0;i<n;i++)
	{
		cin>>x>>y;
		r[i].id=(i+1);r[i].t=x+y;
	}
	sort(r,r+n,my);
	for(i=0;i<n;i++)
	cout<<r[i].id<<" ";
	cout<<endl;
	return 0;
}
