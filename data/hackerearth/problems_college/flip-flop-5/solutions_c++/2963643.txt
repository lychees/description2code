#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    string tmp;
    cin>>tmp;
    int length=tmp.length();
   char ch[length];
strcpy(ch, tmp.c_str());	

    int flag=1;
    int count=0;
    
    for(int k=0;k<length-1;k++)
    {
    	char b=ch[k];
    	if(b=='X')
    	{
    		if(ch[k+1]=='Y')
    		{
    			flag=1;
    		}
    		else
    		{
    			flag=0;
    			count++;
    		}
    		
    	}
    	
    		if(b=='Y')
    	{
    		if(ch[k+1]=='X')
    		{
    			flag=1;
    		}
    		else
    		{
    			flag=0;
    			count++;
    		}
    		
    	}
    }
    
    cout<<count<<endl;
    	
    }
    return 0;
}
