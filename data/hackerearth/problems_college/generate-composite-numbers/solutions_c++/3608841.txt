#include<bits/stdc++.h>
using namespace std;
int a[10000];
void multiply(int j,int n)
{
	int num,k=2;
	while((num=j*k)<=n)
	{
		a[num]=1;
		k++;
	}
}
int main()
{
	int i,n;
	cin>>n;
	for(i=2;i<=n;i++)
	{
		if(a[i]==0)
		{
			multiply(i,n);
		}
	}
	for(i=2;i<=n;i++)
	{
		if(a[i])
		cout<<i<<" ";
	}
	return 0;
}
