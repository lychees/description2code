#include<iostream>
using namespace std;

int a[1000001],track[1000001],flag[1000001],n,high;

void preorder(int i){
    if(i>high)
        return;
    if(a[i]!=0)
        cout<<a[i]<<" ";
    preorder(2*i);
    preorder(2*i+1);
}

int main(){
    int t,i,nodeIndex,x,y,z;
    cin>>t;
    while(t--){
        cin>>n;
        if(n==0){
        	cout<<1<<endl;
        	continue;
        }
        for(i=1;i<=1000000;i++){
            a[i]=0;
            track[i]=0;
        }
        a[1]=1;
        track[1]=1;
        high=0;
        for(i=0;i<n;i++){
            cin>>x>>y>>z;
            nodeIndex=track[x];
            track[y]=nodeIndex*2;
            track[z]=nodeIndex*2+1;
            if(2*nodeIndex+1>high)
                high=2*nodeIndex+1;
            a[nodeIndex*2]=y;
            a[nodeIndex*2+1]=z;
        }
        /*for(i=1;i<=high;i++)
            cout<<a[i]<<" ";
        cout<<endl;*/
        preorder(1);
        cout<<endl;
    }
    return 0;
}
