#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long a,b,mini=100000009,maxa=0,test,x;
    long long arr[105];
    long long has[100001]={0};
    scanf("%lld",&test);
     for(long long i=0;i<test;i++){
        scanf("%lld",&arr[i]);
     }
    for(long long i=1;i<=100000;i++){
            long long sum=0;long long x=0;
        for(long long j=0;j<test;j++){
            sum+=arr[j];

 //           printf("\nsum=%lld ",sum);
            if(sum>i)
            {
 //               printf("in i=%lld j=%lld sum=%lld ",i,j,sum);
                while(1){
                    sum-=arr[x];
                    x++;

                    if(sum<=i)
                        break;
                }
  //              printf("out i=%lld j=%lld sum=%lld \n",i,j,sum);
            }
            if(sum==i){
                 has[i]++;
 //                printf("in has map i=%lld j=%lld x=%lld\n",i,j,x);
                 if(maxa<has[i])
                    maxa=has[i];
                 x=j+1;
                 sum=0;
            }
        }
    }
    printf("%lld\n",maxa);
    for(long long i=1;i<100001;i++){
        if(has[i]==maxa)
            printf("%lld ",i);
    }
    return 0;
}
