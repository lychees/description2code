testCase = int(raw_input())
while (testCase > 0):
    x = int(raw_input())
    if (x == 1):
        print(-1)
    else:
        counter = 0
        while (x > 0):
            counter += 1
            if (x >= 7):
                x -= 7
            elif (x >= 5):
                x -= 5
            elif (x >= 3):
                x -= 3
            else:
                x -= 2
        print(counter)
    testCase -= 1
