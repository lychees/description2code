#include <iostream>
#include<string.h>
using namespace std;

int main() {
	long long int t,i,count,l,flag;
	string s;
	cin>>t;
	while(t--)
	{
		cin>>s;
		count=0;
		l=s.length();
		for(i=0;i<l;i++)
		{
			if(s[i]=='4' || s[i]=='7')
			{
				count++;
			}
		}
		//cout<<count<<endl;
		flag=0;
		while(1)
		{
			int k=count%10;
			//if(k==0){break;}
			count=count/10;
			if(k!=4 && k!=7)
			{
				flag=1;
				break;
			}
			if(count==0){break;}
		}
		if(flag==1){cout<<"NO"<<endl;}
		else{cout<<"YES"<<endl;}
	}
	
	return 0;
}
