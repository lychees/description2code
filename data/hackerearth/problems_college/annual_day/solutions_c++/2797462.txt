#include <bits/stdc++.h>
using namespace std;
int a[107];
void solve()
{
	a[0]=0;a[1]=1;a[2]=1;a[3]=2;
	for(int j=4;j<106;j++)
	{int c=0;
	for(int i=0;i<7;i++)
	{
		if(j&1<<i) c++;
	}
	a[j]=c;
	}
}
int main()
{
   // cout << "Hello World!" << endl;
   int t;
   solve();
   cin>>t;
   while(t--)
   {
   	
   	int n,k;
   	cin>>n>>k;
   	int b[n];
   for(int i=0;i<n;i++)
   { cin>>b[i]; b[i]=a[b[i]];
    }
   int ans=0;
   sort(b,b+n);
   
   for(int i=1;i<=k;i++)
   {
   //	cout<<b[n-i]<<" ";
   	ans+=b[n-i];
   }
   cout<<ans<<endl;
   }
    return 0;
}
