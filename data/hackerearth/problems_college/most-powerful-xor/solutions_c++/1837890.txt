#include <iostream>
#include <limits.h>
using namespace std;

int main()
{
    int t,max,res,l,r;
    //max=INT_MIN;
    cin>>t;
    while(t--)
    {
    	max=INT_MIN;
    	cin>>l>>r;
    	for(int i=l;i<=r;i++)
    	{
    		for(int j=i;j<=r;++j)
    		{
    			res=i^j;
    			if(res>max)
    			{
    				max=res;
    			}
    		}
    	}
    	cout<<max<<endl;
    }
    return 0;
}
