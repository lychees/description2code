   #include<bits/stdc++.h>
   using namespace std;
   int main ()
   {
       int t;
       scanf("%d",&t);
       while(t--)
       {
       	char s[200005];
       	int i,n,sum=0;
       	cin>>s;
       	int len=strlen(s);
       	cin>>n;
       	n=n*(n+1);
       	n/=2;
       	for(i=0;i<len;i++)
       	{
       		sum+=(s[i]-'0'+n)%2;
       	}
       	cout<<sum<<endl;
       }
       return 0;
   }
