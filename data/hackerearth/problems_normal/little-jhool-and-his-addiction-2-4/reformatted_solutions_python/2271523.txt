for i in range(input()):
    n, k = map(int, raw_input().split())
    lst = sorted(list(map(int, raw_input().split())))

    mx = -1
    mn = 10 ** 18

    for i in xrange(n // 2):
        v = lst[i] + lst[-i - 1]
        mx = max(v, mx)
        mn = min(v, mn)

    dif = mx - mn

    print dif
    if dif > k:
        print 'No more girlfriends!'
    elif dif == k:
        print 'Lucky chap!'
    else:
        print 'Chick magnet Jhool!'
