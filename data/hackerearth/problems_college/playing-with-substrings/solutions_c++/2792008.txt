set<string > s;
int main()
{
	string a, b;
	cin >> a >> b;
	for (int i = 0;i < a.size();i++)
	{
		for (int j = i;j < a.size();j++)
		{
			if (j - i + 1 < b.size())
				continue;
			string temp = "";
			for (int k = i;k <= j;k++)
				temp += a[k];
			s.insert(temp);
		}
	}
	int cnt = 0;
	forr(it, s)
	{
		string temp = *it;	
		for (int i = 0;i < temp.size();i++)
		{
			if (i + b.size() - 1 >= temp.size())
				break;
			string tt = temp.substr(i, b.size());
			if (tt == b)
			{
				cnt++;
				break;
			}
		}
	}
	pi(cnt);
	return 0;
}
