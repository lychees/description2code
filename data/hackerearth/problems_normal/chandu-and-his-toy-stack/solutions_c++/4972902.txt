#include <bits/stdc++.h>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t,n,i,x,y,ans;
   cin>>t;
   while(t--){
   	cin>>n>>x>>y;
   	ans=0;
   	int a[n],b[n];
   	for(i=0;i<n;i++)
   		cin>>a[i]>>b[i];
   	sort(a,a+n);
   	sort(b,b+n);
   	for(i=0;i<n;i++){
   		if(b[i]>a[i])
   			ans+=(b[i]-a[i])*x;
   		else
   			ans+=(a[i]-b[i])*y;
   	}
   	cout<<ans<<endl;
   }
    return 0;
}
