#include <iostream>
using namespace std;
typedef long long ll;
int main()
{
    int a[100];
    for(int i=0;i<99;i++)a[i]=1;
    for(int i=2;i<99;i++)
    {
    	if(a[i])
    	for(int j=i; (j*i)<99 ; j++)
    	{
    		a[j*i]=0;
    	}
    }
    a[1]=0;
    ll t; char abc[10];
    cin>>t;
    while(t--)
    {
    	cin>>abc;
    	int sum=0;
    	for(int i=0;abc[i]!='\0';i++)
    	 sum+= (abc[i]-'0');
    	if(a[sum])cout<<"YES\n";
    	else cout<<"NO\n";
    }
}
