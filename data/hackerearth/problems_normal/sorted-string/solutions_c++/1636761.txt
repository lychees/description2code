#include <bits/stdc++.h>
using namespace std;
#define PIC pair <int, char> 
int main()
{
    int a; cin >> a;
    for (int g=0; g<a; g++)
    {
    	string b; cin >> b;
    	vector <PIC> ans; 
    	map <char,int>y;
    	for (int z=0; z<b.length(); z++)y[b[z]]++; 
    	for (char z='a'; z<='z'; z++)
    	{
    		for (int l=0; l<y[z]; l++) ans.push_back(PIC(y[z], z)); 
    	}
    	sort(ans.begin(),ans.end());
    	for (int y=0; y<ans.size(); y++) cout << ans[y].second;
    	cout << '\n';
    }
    return 0;
}
