#include <iostream>
using namespace std;

int main()
{
    int x,y;
    cin>>x>>y;
    long int *A;
    A=new long int [1000001];
    A[0]=1;
    for(int i=1;i<1000001;i++)
    {A[i]=x*A[i-1]-y;
    A[i]%=1000000007;}
    int q;
    cin>>q;
    while(q--){
    	int n;
    	scanf("%d",&n);
    	printf("%ld\n",A[n]);
    }
}
