#include <iostream>
using namespace std;

int main()
{
    long N,count[26]={0};
    cin>>N;
    fflush(stdin);
    char ch=getchar();
    for(long i=0;i<N;i++){
    	ch=getchar();
    	count[ch-'a']++;
    }
    long Q;
    char X,Y;
    cin>>Q;
    for(long i=0;i<Q;i++){
    	cin>>X>>Y;
    	if(X==Y){
    		cout<<(count[X-'a']*(count[X-'a']-1))/2;
    	}
    	else
    	    cout<<count[X-'a']*count[Y-'a'];
    	cout<<"\n";
    }
    
    return 0;
}
