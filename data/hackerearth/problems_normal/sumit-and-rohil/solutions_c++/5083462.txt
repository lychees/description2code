#include<iostream>
#include<vector>
#include<map>
using namespace std;

int isAnagram(string s1,string s2){
	map<char,int>hm;
	for(int i=0;i<s1.length();i++)hm[s1[i]]++;
	for(int i=0;i<s2.length();i++)hm[s2[i]]--;
	map<char,int>::iterator it;
	for(it=hm.begin();it!=hm.end();it++)
	if(it->second!=0)return 0;
	return 1;
}

int main(){
    int n;
    cin>>n;
    vector<string>v(n);
    for(int i=0;i<n;i++)cin>>v[i];
    map<int,int>hm;
    int visited[n];
    for(int i=0;i<n;i++)visited[i]=0;
    for(int i=0;i<n;i++){
    	if(visited[i]==0){
    		visited[i]=1;
    		hm[i]++;
    	for(int j=i+1;j<n;j++){
    		if(v[i].at(0)==v[j].at(0) && v[i].at(v[i].length()-1)==v[j].at(v[j].length()-1)){
    			if(isAnagram(v[i],v[j])){
    				hm[i]++;
    				visited[j]=1;
    			}
    		}
    	}
    	}
    }
    cout<<hm.size();
    return 0;
}
