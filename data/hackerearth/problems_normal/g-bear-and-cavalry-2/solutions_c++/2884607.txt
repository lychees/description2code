#include <bits/stdc++.h>
using namespace std;

int n;
int w[100], h[100];

void solve(){
	cin>>n;
	for(int i=0;i<n;++i)cin>>w[i];
	int x = 0;
	for(int i=0;i<n;++i){
		cin>>h[i];
		if(h[i]>h[x])x=i;
	}
	int s = w[0]*h[x];
	w[0]=h[x]=-1;
	sort(w,w+n);
	sort(h,h+n);
	reverse(h+1,h+n);
	for(int i=1;i<n;++i)if(h[i]*w[i]>=s){cout<<"NO"<<endl;return;}
	cout<<"YES"<<endl;
}

int main()
{int t;
    cin>>t;
    while(t--) solve();
    return 0;
}
