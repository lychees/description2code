#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		int n,w;
		cin>>n>>w;
		vector<int> wt,val;
		int x;
		for(int i=0;i<n;i++){
			cin>>x;
			wt.push_back(x);
		}
		for(int i=0;i<n;i++){
			cin>>x;
			val.push_back(x);
		}
	//	for(int i=0;i<n;i++)cout<<wt[i]<<" ";cout<<endl;
	//	for(int i=0;i<n;i++)cout<<val[i]<<" ";cout<<endl;
		int dp[n+1][w+1];
		memset(dp,0,sizeof(dp));
		for(int i=0;i<=n;i++){
			for(int j=0;j<=w;j++){
				if(i==0 || j==0){
					dp[i][j]=0;
				}else if(wt[i-1]<=j){
					dp[i][j]=max(val[i-1]+dp[i-1][j-wt[i-1]],dp[i-1][j]);
				}else{
					dp[i][j]=dp[i-1][j];
				}
			}
		}
	/*	for(int i=0;i<=n;i++){
			for(int j=0;j<=w;j++){
				cout<<dp[i][j]<<" ";
			}
			cout<<endl;
		}
	*/	cout<<dp[n][w]<<endl;
	}
	return 0;
	
}
