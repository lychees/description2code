#include <iostream>
using namespace std;
int CheckIt(char s1[],char s2[])
{
	int s=0,i;
	int x,y;
	for(x=0;s1[x]!='\0';x++);
	for(y=0;s2[y]!='\0';y++);
    if(x==0||y==0)
    {
    	return 0;
    }
	if(x<y)
	{
		for(i=0;i<x;i++)
		{
			if(s1[i]==s2[i])
			{
				s++;
			}
		}
	}
	else
	{
		for(i=0;i<y;i++)
		{
			if(s1[i]==s2[i])
			{
				s++;
			}
		}
	}
	return s;
}
int main()
{
    int T,i,j;
    char s1[2000],s2[2000];
    cin>>T;
    cin.get();
    while(T--)
    {
    	cin.getline(s1,505);
    	cin.getline(s2,505);
    	cout<<CheckIt(s1,s2)<<endl;
    }
}
