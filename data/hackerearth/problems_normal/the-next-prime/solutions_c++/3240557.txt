#include <iostream>
#include <vector>
using namespace std;
int main()
{
    long long t,i,j,x;cin>>t;
    vector<bool>arr(10000000,1);arr[0]=0;
    for(i=2;i<=4000;i++)
		if(arr[i-1]==1)
			for(j=2;i*j<=10000000;j++)
				if(arr[i*j-1]==1)arr[i*j-1]=0;
	while(t--){
    	cin>>x;
    	for(i=x;arr[i]!=1;i++);
    		cout<<i+1<<endl;
    }return 0;
}
