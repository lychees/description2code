n = int(raw_input())
while(n):
	n-=1
	
	raw = raw_input()
	
	l = len(raw)
	
	
	
	if(len(list(set(raw)))!=3):
		print "Not OK"
		continue
	if(len(raw) % 3 != 0):
		print "Not OK"
		continue
	q = l/3
	flag = 0
	
	for i in range(0,3):
		if(raw[i*q] != raw[i*q+q-1]):
			flag = 1
			
	if flag == 1:
		print "Not OK"
		continue
	else:
		print "OK"
