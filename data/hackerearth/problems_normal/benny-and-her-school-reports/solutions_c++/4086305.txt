#include <bits/stdc++.h>
using namespace std;

int main()
{
    ios::sync_with_stdio(false);
    
    long int t,n,m,x,a,ctr,i;
    
    cin>>t;
    
    while(t--)
    {
    	cin>>n>>m>>x;
    	
    	ctr=0;
    	
    	for(i=0;i<n;i++)
    	{
    		cin>>a;
    		ctr+=a;
    	}
 
    	if(ctr>=((n+1)*x))
    		cout<<"1";
    	
    	else if(((n+1)*x - ctr)>m)
    		cout<<"Impossible";
    		
		else
		{
			cout<<((n+1)*x)	- ctr;
		}
		
		cout<<endl;
    	
    }
    
    return 0;
}
