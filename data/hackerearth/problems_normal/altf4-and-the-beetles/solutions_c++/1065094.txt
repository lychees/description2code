#include <iostream>
#include <bits/stdc++.h>
using namespace std;

vector<pair<long long,bool> > arr;
int main()
{
    int t,n;
    long long a,b,c,p,q;
    cin>>t;
    while(t--)
    {
    	cin>>n>>a>>b>>c;
    	for(int i=0;i<n;++i)
    	{
    		cin>>p>>q;
    		arr.push_back(make_pair(p,0));
    		arr.push_back(make_pair(q,1));
    	}
    	sort(arr.begin(),arr.end());
    	
    	long long ans=n*a;
    	int open=0,close=0;
    	
    	for(int i=0;i<2*n;++i)
    	{
    		int j=i;
    		while(j<2*n && arr[i]==arr[j])
    		{
    			//cout<<arr[j].first<<" "<<arr[j].second<<endl;
    			if(arr[j].second==0) open++;
    			else close++;
    			j++;
    		}
    		i=j-1;
    		//cout<<ans<<(b*(open-close)+c*close+(n-open)*a)<<endl;
    		ans=max(ans,b*(open-close)+c*close+(n-open)*a);
    	}
    	cout<<ans<<endl;
    	arr.clear();
    }
    return 0;
}
