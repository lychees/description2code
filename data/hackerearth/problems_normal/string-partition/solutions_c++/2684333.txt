#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
LL dp[101][101];
LL fromto[101][101];
int main()
{
	int t,k,j;
	LL z;
	char s1[110];
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%lld",&k,&z);
		
		scanf("%s",s1);
		int n=strlen(s1);
		
		

		for(int i=0;i<n;i++)
		{
			long long int num=0;
			for(j=i;j<n;j++)
			{
				num=num*10+(s1[j]-'0');
				if(num>z)
				break;
				fromto[i][j]=num;
			}
			for(;j<n;j++)
			fromto[i][j]=-1;
			
		}
		for(int i=1;i<=100;i++)
		for(int j=0;j<=100;j++)
		dp[i][j]=-1;
		for(int i=0;i<n&&fromto[0][i]!=-1;i++)
		{
			dp[1][i]=fromto[0][i];
		}
		for(int i=2;i<=k;i++)
		{
			for(int j=0;j<n;j++)
			{
				for(int k=j+1;k<n;k++)
				{
					if(fromto[j+1][k]!=-1&&dp[i-1][j]!=-1)
					{
						dp[i][k]=max(dp[i][k],dp[i-1][j]+fromto[j+1][k]);
					}
				}
			}
		}
		printf("%lld\n",dp[k][n-1]);
	}
	return 0;
	}
