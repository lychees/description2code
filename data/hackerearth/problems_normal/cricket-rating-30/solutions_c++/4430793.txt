#include <iostream>
using namespace std;
int n;
#define M 100001
int rating[M];

void input()
{
	for(int i=0;i<n;i++)
	{
		cin>>rating[i];
	}
}

int max(int i,int j)
{
	if(i>j) return i;
	else return j;
}
void solve()
{   
    int i;
	int maxsofar=rating[0];
	int curr_max=rating[0];
	for(i=1;i<n;i++)
	{
		curr_max=max(rating[i],curr_max+rating[i]);
		maxsofar=max(maxsofar,curr_max);
	}
	if(maxsofar>0)
	{
		cout<<maxsofar;
	}
	else 
	{
		cout<<"0";
	}
}



int main()
{
    cin>>n;
    input();
    solve();
    return 0;
}
