#include<bits/stdc++.h>
using namespace std;

#define mod 1000000007
#define W while
#define LL long long int
#define pb push_back
LL fun(LL n)
{
	LL i,rev,p,d;
	p=n;
	rev=0;
	W(n!=0)
	{
		d=n%10;
		rev=rev*10+d;
		n=n/10;
	}
	if(p==rev)
		return 1;
	return 0;
}

int main()
{
	LL  t,a,b,c,i,ans;
	cin>>t;
	W(t--)
	{
		c=0;
		cin>>a>>b;
		for(i=a;i<=b;i++)
		{
			ans=fun(i);
			if(ans==1)
				c++;
		}
		cout<<c<<endl;
	}
	return 0;
}
