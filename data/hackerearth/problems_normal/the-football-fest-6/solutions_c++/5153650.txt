#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   		int n,first_player;
   		cin>>n>>first_player;
   		stack <int> s;
   		s.push(first_player);
   		while(n--)
   		{
   			char type;
   			cin>>type;
   			if(type=='P')
   			{
   				int ID;
   				cin>>ID;
   				s.push(ID);
   			}
   			else
   			{
   				int x = s.top();
   				s.pop();
   				int y = s.top();
   				s.pop();
   				s.push(x);
   				s.push(y);
   			}
   		}
   		cout<<"Player "<<s.top()<<endl;
   }
}
