//recursion
#include <iostream>
#include <cmath>

using namespace std;
  	    
int f(int x1,int y1,int x2,int y2,int m[][1025],int* c,int n)
{
    if(n==1)
    {
    	m[x1][y1]=++(*c);
    }
    else
    {
    	f(0+x1,0+y1,(n/2)-1+x2,(n/2)-1+y2,m,c,n/2);
        f(0+x1,(n/2)+y1,(n/2)-1+x2,n-1+y2,m,c,n/2);
    	f((n/2)+x1,0+y1,n-1+x2,(n/2)-1+y2,m,c,n/2);
    	f((n/2)+x1,(n/2)+y1,n-1+x2,n-1+y2,m,c,n/2);
    }
}
int main()
{
    static int c = 0;
    int m[1025][1025] = {0};

    int p,n;
    cin>>p;
    n = pow(2,p);
    f(0,0,n,n,m,&c,n);
    
    for(int i=0;i<=(n-1);i++)
    {
        for(int j=0;j<=(n-1);j++)
        {
            cout<<m[i][j]<<" ";
        }
        cout<<"\n";
    }
    return 0;
}
