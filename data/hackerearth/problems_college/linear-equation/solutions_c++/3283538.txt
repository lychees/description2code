#include <bits/stdc++.h>
using namespace std;
#define ll long long int
ll a[1000001];
int main()
{
    ll t, n,  i, x;
    cin>>t;
    while(t--) {
    	cin>>n;
    	x = 0;
    	for(i = 0; i < n; i++) {
    		cin>>a[i];
    		x += a[i];
    	}
    	x = x/(n-1);
    	for(i = 0 ; i < n; i++)
    		cout<<x-a[i]<<" ";
    	cout<<endl;
    }
    return 0;
}
