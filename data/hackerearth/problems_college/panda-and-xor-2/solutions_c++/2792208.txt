#include<bits/stdc++.h>
using namespace std;
long long int fun(int n)
{
	unsigned long long int count = 0;
	while (n)
	{
		n &= (n-1) ;
		count++;
	}
	return count;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
	long long a,b;
	cin>>a>>b;
	long long x=fun(a^b);
	if(x%2==0)
		cout<<"YES"<<endl;
	else
		cout<<"NO"<<endl;
	//cout<<fun(x)<<endl;
	}
	return 0;
}
