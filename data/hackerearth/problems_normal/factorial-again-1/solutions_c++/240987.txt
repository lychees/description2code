int64 powe(int64 a,int64 b,int64 c){
	int64 ans=1;
	while(b){
		if(b&1){
			ans=ans*a;
			if(ans>=c)
			ans%=c;
		}
		b=b/2;
		a=a*a;
		if(a>=c)
		a%=c;
	}
	return ans;
}
int64 solve(int64 x,int64 m,int64 n){
	int64 ans=1,den;
	for(int64 i=x+1;i<m;i++){
		ans=ans*i;
		if(ans>=m)
		ans%=m;
	}
	ans=powe(ans,m-2,m);
	ans=(-1*ans+m)%m;
	den=powe(6,n,m);
	den=powe(den,m-2,m);
	return (ans*den)%m;
}
int main(){
	int64 t,n,p;
	cin>>t;
	while(t--){
		cin>>n>>p;
		if(3*n>=p)
		cout<<0<<endl;
		else
		cout<<solve(3*n,p,n)<<endl;
	}
	return 0;
}
