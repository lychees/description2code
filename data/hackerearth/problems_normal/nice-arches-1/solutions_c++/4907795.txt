#include <iostream>
#include <stack>
using namespace std;

int main()
{
    int n,count=0;
    cin>>n;
    //string str;
    string str;
    char s[1000000];
    int top=0;
    int i=1;
    while(n--)
    {
    	
    	i=2;
    	cin>>str;
    	s[++top]=str[0];
    	
    	while(str[i-1]!='\0')
    	{
    		
    		if(s[top]==str[i-1])
    		{
    			top--;
    		}
    			
    		else
    			s[++top]=(str[i-1]);
    		i++;
    	}
		
    	if(top==0)
    		count++;
    	top=0;	
    
    }
    cout<<count;
    return 0;
}
