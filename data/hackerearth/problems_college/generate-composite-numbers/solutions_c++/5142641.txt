#include<iostream>
#include<cstring>
#include<cmath>
using namespace std;
#define mx (10009)
bool a[mx];
int n;
void sieve()//complexity O(nloglogn)
{
    for(int i=2;i<=n;++i)
        a[i]=true;
    int lim=sqrt(n);
    for(int i=2;i<=lim;++i)
    {
        if(a[i]==true)
        {
            for(int j=i*i;j<=n;j+=i)
            {
                a[j]=false;
            }
        }
    }
 return;
}
int main()
{
    cin>>n;
    sieve();
    for(int i=2;i<=n;++i)
    {
        if(a[i]==false)cout<<i<<" ";
    }
    cout<<endl;
    return 0;//lkjhlkj
}
