#include <iostream>
using namespace std;

int compare(const void *a,const void *b)
{
	return (*(int*)a-*(int*)b);
}
int main()
{
    int i,n,k;
    cin>>n;
    cin>>k;
    int a[n];
    for(i=0;i<n;i++)
    	cin>>a[i];
    qsort(a,n,sizeof(int),compare);
    int index=n-k;
    cout<<a[index];
    return 0;
}
