ll fast_exp(l base,l exp)
 {
     if(exp==0)
        return 1;
     else
     {
         if(exp%2==0)
         {
             ll temp=fast_exp(base,exp/2);
             temp=temp*temp;
             if(temp>=MOD)
                return temp%MOD;
             else
                return temp;
         }
         else
         {
             ll temp=fast_exp(base,(exp-1)/2);
             temp=base*temp*temp;
             if(temp>=MOD)
                return temp%MOD;
             else
                return temp;

         }
     }
 }
 ll GCD(ll a, ll b)
{
    if (a == 0)
        return b;
    return GCD(b%a, a);
}
int setbit(ll N)
{
    if(N==0)
        return 0;
    return setbit(N&(N-1))+1;
}
