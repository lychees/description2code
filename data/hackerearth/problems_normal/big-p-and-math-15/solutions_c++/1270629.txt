#include <iostream>
#include<algorithm>
using namespace std;
int min(int a,int b);
int max(int a,int b);
int min(int a,int b)
{
	int temp,newa=0,newb=0,ini=1;
	while(a)
	{
		temp=a%10;
		if(temp==6)
		newa+=5*ini;
		else
		newa+=temp*ini;
		ini*=10;
		a/=10;
	}
	//cout<<newa;
	ini=1;
	while(b)
	{
		temp=b%10;
		if(temp==6)
		newb+=5*ini;
		else
		newb+=temp*ini;
		ini*=10;
		b/=10;
	}
	//cout<<newb;
	//cout<<newa<<" "<<newb<<endl;
	return newa+newb;
}
int max(int a,int b)
{
	int temp,ini=1,newa=0,newb=0;
	while(a)
	{
		temp=a%10;
		if(temp==5)
		newa+=6*ini;
		else
		newa+=temp*ini;
		ini*=10;
		a/=10;
	}
	//cout<<newa;
	ini=1;
	while(b)
	{
		temp=b%10;
		//cout<<temp<<endl;
		if(temp==5)
		newb+=6*ini;
		else
		newb+=temp*ini;
		ini*=10;
		b/=10;
		//cout<<newb<<endl;
	}
	//cout<<newb;
	//cout<<newa<<" "<<newb<<endl;
	return newa+newb;
}
int main()
{
    int a,b;
    cin>>a>>b;
    cout<<min(a,b)<<" "<<max(a,b)<<endl;
    return 0;
}
