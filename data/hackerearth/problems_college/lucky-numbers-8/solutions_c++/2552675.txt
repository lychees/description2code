#include<stdio.h>
#include<iostream>
#include<string.h>
#include<math.h>
#include<algorithm>
#include<stdlib.h>
#include<vector>
#include<map>
#include<set>

using namespace std;

main()
{
	
	int t;
	cin>>t;
	
	while(t--)
	{
		long long int n,a[1000000],ct=0,flag=0;
		cin>>n;
		
		for(long long int i=0;i<n;i++)
		{
			cin>>a[i];
		}
		
		for(long long int i=0;i<n;i++)
		{
			for(long long int j=i+1;j<n;j++)
			{
				if(a[i]<=a[j])
				{
					flag=1;
					break;
				}
			}
			if(flag==0)
			ct++;
			
			flag=0;
		}
		cout<<ct<<endl;
	}
}
