    #include <iostream>
    using namespace std;
    int dp[1000001][10];//dp[i][j] is for (i+1) digits
    int ans[1000001]={0};//ans[i] is for i digits
    #define MOD 1000000007
    int main()
    {
        //cout << "Hello World!" << endl;
        for(int i=0;i<10;i++) dp[0][i]=1;
        for(int i=2;i<=8;i++) dp[1][i]=2;
        dp[1][0]=dp[1][1]=dp[1][9]=1;
        ans[1]=10;
        ans[2]=17;
        for(int i=2;i<=1000000;i++)
        {
        	dp[i][0]=dp[i-1][1];
        	dp[i][9]=dp[i-1][8];
        	for(int j=1;j<=8;j++)
        		dp[i][j]=(dp[i-1][j-1]+dp[i-1][j+1])%MOD;
        	for(int j=0;j<=9;j++) ans[i+1]=(ans[i+1]+dp[i][j])%MOD;
        }
        int t;
        cin>>t;
        while(t--){
        	int n;
        	cin>>n;
        	cout<<ans[n]<<endl;
        }
        return 0;
    }
