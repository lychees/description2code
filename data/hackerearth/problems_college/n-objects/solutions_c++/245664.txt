#include <iostream>
using namespace std;

#define MAX 501
#define MOD 1000000007

int main()
{
	int T, N;
	long long triangle[MAX][MAX], partitions[MAX];
	
	triangle[1][1] = 1;
	partitions[1] = 1;
	
	for(int i=2;i<MAX;i++)
	{
		triangle[i][1] = triangle[i-1][i-1];
		for(int j=2;j<=i;j++)
			triangle[i][j] = (triangle[i][j-1] + triangle[i-1][j-1])%MOD;
		partitions[i] = triangle[i][i]%MOD; 
	}
	
	cin>>T;
	while(T--)
	{
		cin>>N;
		cout<<triangle[N][N]<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
