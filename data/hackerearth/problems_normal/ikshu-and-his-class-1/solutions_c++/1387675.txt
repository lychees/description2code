#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
vector <long long int> mat[100002];
long long int v[100002]={0};
long long int fac[100002];
long long int counter=1;
int generator()
{
    fac[0]=0;
    fac[1]=1;
    fac[2]=2;
    long long int i=3;
    while(i<100002)
    {
        fac[i]=(i*fac[i-1])%mod;

        i++;
    }
}
int dfs(int x)
{
    v[x]=1;
    for(int i=0;i<mat[x].size();i++)
        {
            if(v[mat[x][i]]!=1)
            {
                counter++;
                dfs(mat[x][i]);
            }
        }
}
int main()
{
      int n,k,i,x,y;
      long long int ans=1;
      cin>>n>>k;
      for(i=0;i<k;i++)
      {
          cin>>x>>y;
          mat[x].push_back(y);
          mat[y].push_back(x);
      }
    generator();
    for(i=0;i<n;i++)
    {
        if(v[i]!=1 && !mat[i].empty())
            dfs(i);
        //cout<<i<<counter<<endl;
        ans=(ans*fac[counter])%mod;
        counter=1;
    }

    cout<<ans<<endl;
}
