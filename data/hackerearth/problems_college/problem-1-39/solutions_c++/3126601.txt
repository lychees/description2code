#include<bits/stdc++.h>
using namespace std;
main()
{
	int n;
	scanf("%d",&n);
	while(n--)
	{
		int a,b,c,ans=1,final;
		scanf("%d%d%d",&a,&b,&c);
		for(int i=1;i<=abs(b);i++)
		ans*=a;
		if(b<0)
		ans=1/ans;
		if(ans==0)
		printf("0\n");
		else{
		int val1=c-(ans%c);
		int val2=ans%c;
		int k=min(val1,val2);
		if(k==val1)
		final=ans+val1;
		else
		final=ans-val2;
		printf("%d\n",final);
	}}
}
