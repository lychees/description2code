    #include <iostream>
    using namespace std;
     
    int main()
    {
    	string s;
    	int mx = 0;
    	string maxs;
        while(cin >> s) {
        	int sum = 0;
        	for(int x = 0; x<s.length(); x++) {
        		sum += s[x] - '0';
        	}
        	if(sum > mx) {
        		mx = sum;
        		maxs = s;
        	}
        }
        cout << maxs << endl;
    }
