#include <bits/stdc++.h>
#define ll long long int
using namespace std;

ll c[15];

int main()
{
	ll t, n, k;
	cin >> t;
	while (t--) {
		cin >> n >> k;
		ll x;
	//	memset(c, 0, sizeof(0));
		for(ll i=0;i<=10;i++)
			c[i]=0;
		for (ll i = 0; i < n; i++) {
			cin >> x;
			c[x]++;
		}
		ll mx = -1;
		
		for (ll i = 1; i <= k; i++) {
			if(c[i] > mx ) {
				mx = c[i];
			}
		}
		
		cout << n - mx << endl;
	}
	
	return 0;
}
