using namespace std;
#define MOD 1000000007
string s;
int main()
{
	std::ios::sync_with_stdio(false);cin.tie(0);
	long long re=0;
	long long n1=0,n2=0,t=0;
	long long nt=0;
	int i,n;
	cin>>s;
	n=s.size();
	for(i=0;i<n;i++)
	{
		if(s[i]==';')
		{
			re=(re+t)%MOD;
			t=(t+1)%MOD;
			nt++;
		}
		else
		{
			n2++;
			t=t*2%MOD;
		}
	}
	re=(re-nt*(nt-1)/2+(long long)MOD*MOD)%MOD;
	cout<<re<<endl;
//	system("pause");
	return 0;
}
