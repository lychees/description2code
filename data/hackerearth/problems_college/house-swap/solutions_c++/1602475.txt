#include<iostream>
using namespace std;

int a[1000001],pos[1000001];
int main()
{
int n,m=0;
cin>>n;
fill(pos,pos+n,0);
for(int i=0;i<n;i++)
{
cin>>a[i];
pos[i]=a[i];
if (a[i]==i)
m++;
}
int flag=0;
for(int i=0;i<n;i++)
{
if(pos[i]==i)
continue;
if(pos[pos[i]]==i)
{
flag=1;
break;
}
}

if (m==n)
cout<<n-2<<endl;
else if(flag==1)
cout<<m+2<<endl;
else if(flag==0)
cout<<m+1<<endl;
return 0;
}
