#include <iostream>
using namespace std;

long long cal(int n, int k, int m) {
	long long cost = 0;
	for(int i=1; i<=n; i++) {
		cost += (1LL*k)*((1LL*i)*i*i);
	}
	return m>cost?0:cost - m;
}

int main()
{
	ios_base::sync_with_stdio(false);
   	int test;
   	cin>>test;
   	
   	while(test--) {
   		int n,k,m;
   		cin>>n>>k>>m;
   		
   		long long ans = cal(n,k,m);
   		
   		cout<<ans<<endl;
   	}
    return 0;
}
