#include <bits/stdc++.h>
#define mod 1000000007
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long arr[3],res;
    	cin>>arr[0]>>arr[1]>>arr[2];
    	//cout<<arr[0]<<arr[1]<<arr[2];
    	sort(arr,arr+3);
    	res=arr[0];
    	res=(res*(arr[1]-1))%mod;
    	res=(res*(arr[2]-2))%mod;
    	cout<<res<<endl;
    }
    return 0;
}
