#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
const int mod = 1e9+7;
ll n;
char s[100005];
vector<ll> v;
ll solve(ll x,ll u,ll z){
 int l = u;
 int r = z;
 while(l<=r){
  ll mid = (l+r)>>1;
  if(v[mid]==x) return mid;
  else if(v[mid] > x) r = mid-1;
  else l = mid+1;
 }
 return l-1;
}
int main(){
 //freopen("qns.txt","r",stdin);
 //freopen("ans.txt","w",stdout);
 int t;
 ll sz,K;
 scanf("%d",&t);
 while(t--){
  v.clear();
  scanf("%lld %lld",&sz,&K);
  scanf("%s",s);
  for(ll i=0;i<sz;i++){
    if(s[i]=='1') v.push_back(i);
  }
  n = v.size();
  ll res = n;
  for(ll i=0;i<n-1;i++){
    ll b = v[i]+K;
    ll pos = solve(b,i+1,n-1);
    res = res+2*(pos-i);
  }
  if(res==0) printf("0/1\n");
  else{
  	ll g = __gcd(res,sz*sz);
    printf("%lld/%lld\n",res/g,sz*sz/g);
  }
 }
 return 0;
}
