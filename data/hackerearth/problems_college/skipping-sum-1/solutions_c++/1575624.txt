#include<cstdio>
#include<iostream>
#include<stack>
#include<queue>
#include<list>
#define ll long long int
using namespace std;
    ll a[11][100010];
int main()
{
    ll ar[100010];
    ll n,q;
   
    scanf("%lld%lld",&n,&q);
    for(ll i=1;i<=n;i++)
    {
      cin>>ar[i];       
    }
    for(ll i=1;i<=10;i++)
    for(ll j=n;j>=1;j--)
    a[i][j]=ar[j]+a[i][j+i];
 
    
ll l,r,k;
    while(q--)
    {
              
            scanf("%lld%lld%lld",&l,&r,&k);
            ll next_hop=((r-l)/k+1)*k+l;  
            ll res=a[k][l]-a[k][next_hop];
            cout<<res<<endl; 
    }
}
