def maxSubArraySum(arr, size):
    max_so_far = 0
    max_ending_here = 0
    for x in range(0, size):
        max_ending_here = max_ending_here + int(arr[x])
        if (max_ending_here < 0):
            max_ending_here = 0
        if (max_so_far < max_ending_here):
            max_so_far = max_ending_here
    return max_so_far


numInputs = input()
arr = raw_input().split()
print maxSubArraySum(arr, numInputs)
