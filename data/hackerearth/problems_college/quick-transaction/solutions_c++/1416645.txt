#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	long long int n;
	int a[9]={1,2,5,10,20,50,100,500,1000};
	while(t--)
	{
		cin>>n;
		long long int in=0;
		for(int i=8;i>=0;i--)
		{
			if(a[i]<=n)
			{
				in=i;
				break;
			}
		}
		//cout<<in<<endl;
		long long int coins=0;
		while(in>=0 && n!=0)
		{
			coins+=n/a[in];
			n=n%a[in];
			in--;
		}
		cout<<coins<<endl;
	}
}
