#include <bits/stdc++.h>
using namespace std;
int a[26]={0};
int main()
{
	string s[10000];
	int n,k,i,j,mx,t,mn,p;
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		for(i=0;i<n;i++)
		{
			cin>>s[i];
		}
		
		for(p=0;p<k;p++)
		{
			int mn=1000000,index=-1;
		
			for(i=0;i<26;i++)
			{
				
				int cost=0;
				for(j=0;j<n;j++)
				{
					cost=cost+abs(s[j][p]-'a'-i);
			
				}
			if(cost<mn)
			{
				mn=cost;
				index=i;
			}
	
		
			}
			cout<<(char)('a'+index);
		}
		cout<<endl;
	
	}

}
