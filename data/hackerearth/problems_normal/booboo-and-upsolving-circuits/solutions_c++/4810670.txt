#include<bits/stdc++.h>
using namespace std;
long long tym[100005];
int main()
{
    long long n,m,i,mid,lef,ryt,days,mini,temp,flag;
    cin>>n>>m;
    for(i=0;i<n;i++)
        cin>>tym[i];
    mini=LLONG_MAX;
    lef=1;
    ryt=100000000000000005;
    while(lef<ryt)
    {
    	flag=0;
        mid=(lef+ryt)/2;
        temp=0;
        days=0;
        for(i=0;i<n;i++)
        {
			if(tym[i]>mid)
			{
				flag=1;
				break;
			}
            if(temp+tym[i]>mid)
            {
                days++;
                temp=0;
            }
             temp=temp+tym[i];
        }
        if(flag==1)
        {
        	lef=mid+1;
        	continue;
        }
        if(temp<=mid)
            days++;
        if(days>m)
            lef=mid+1;
        else
        {
            mini=min(mini,mid);
            ryt=mid;
        }
    }
    cout<<mini;
    return 0;
}
