#include <bits/stdc++.h>
#define ll long long int
using namespace std;

int main()
{
    int n;
    string s;
    vector<pair<int, string> >v;

    while (cin >> s) {

        int sum = 0;
        for (int i = 0; i < s.length(); i++) {
            sum += s[i] - '0';
        }
        v.push_back(make_pair(sum, s));
    }

    sort(v.begin(), v.end());

    cout << v[v.size() - 1].second << endl;

    return 0;
}
