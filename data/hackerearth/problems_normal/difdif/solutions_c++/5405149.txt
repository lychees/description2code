#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    string s;
    cin>>s;
    set <string> ss;
    int len=s.length();
    for(int i=0;i<len;i++){
    	for(int j=0;j<len-i;j++){
    		string sub=s.substr(i,j+1);
    		//cout<<sub<<endl;
    		ss.insert(sub);
    	}
    }
    cout<<ss.size();
    return 0;
}
