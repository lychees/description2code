#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    long long int t,n,k,m,i;
    cin>>t;
    while(t--)
    {
    	cin>>n>>k>>m;
    	string str;
    	pair<string,string> p[n];
    	for(i=0;i<n;++i)
    	{
    		cin>>str;
    		string s = str.substr(0,m);
    		//cout<<s<<" ";
    		p[i].first = s;
    		p[i].second = str;
    	}
    	sort(p,p+n);
    	cout<<p[k-1].second<<"\n";
    }
    return 0;
}
