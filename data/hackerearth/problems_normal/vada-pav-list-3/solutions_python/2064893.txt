def print_list(n, name):

    adict = {}
    for x in xrange(n):
        adict[name[x]] = 0

    answer = adict.keys()
    length = len(answer)
    answer.sort()
    print(length)
    for x in xrange(length):
        print(answer[x])

if __name__ == '__main__':
    n = input()
    alist = []
    for x in xrange(n):
        alist.append(raw_input())

    print_list(n, alist)
