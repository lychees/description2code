#include <iostream>
#include <map>
#include <algorithm>
#include <cstdio>
#define ll long long
using namespace std;
#define gc getchar_unlocked

void inp(long long &x)
{
    register long long c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}
int sumdig(ll n){
	int sum = 0;
	while(n > 0){
		sum = sum + n % 10;
		n = n / 10;
	}
	return sum;
}
int main()
{
	ll t,n,b,x;
	inp(t);
	while(t--){
		inp(n);
		b = sqrt(n);
		int flag = 0;
		x = b - 83;
		if(x < 0)
			x = 0;
		for(; x < b + 1; x++){
			if(x * x + sumdig(x) * x == n){
				flag = 1;
				break;
			}
				
		}
		if(flag == 1)
			cout<<x<<endl;
		else
			cout<<"-1"<<endl;
	}
    return 0;
}
