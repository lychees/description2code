#include <bits/stdc++.h>
using namespace std;

int num_div(int num){
    int n=0;
    int s=sqrt(num);
    for(int i=1;i<=s;i++){
        if(num%i==0)
            n+=2;
    }
    if(s*s==num)
        n--;
    return n;
}
int main(){
    int num=0;
    int T;
    int i=1;
    cin>>T;
    while(T--){
        int N;
        cin>>N;
        N+=1;
        num=1;
        i=1;
        while(num_div(num)<N){
            i++;
            num=(i*(i+1))/2;
        }
        cout<<num<<endl;
    }
    return 0;
}
