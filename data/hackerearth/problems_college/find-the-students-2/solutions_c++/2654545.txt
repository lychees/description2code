#include<bits/stdc++.h>
using namespace std;
main()
{
	long long int n;
	cin>>n;
	int i,j;
	long long int a[1005];
	long long int mini=INT_MAX;
	for(i=0;i<n;i++) 
	{
	  cin>>a[i];
      mini=min(a[i],mini);
	}
	long long int ct=0;
	int f=0;
	while(!f)
	{
		long long int res=0;
		long long int minu=INT_MAX;
		for(i=0;i<n;i++)
		{
			if(a[i]!=0)
			{
				res++;
			  a[i]=a[i]-mini;
			  if(a[i]!=0)
			    minu=min(minu,a[i]);
			  if(a[i]==0)
			    ct++;
			  if(ct==n)
			    f=1;
			}
		}
		mini=minu;
		cout<<res<<endl;
	}
	
}
