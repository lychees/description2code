


    #include <bits/stdc++.h>
    using namespace std;

    #define in(a) scanf("%d",&a);
    int main()
    {
    int t,a,b; in(t);
    
    while(t--){
            in(a);in(b);
            int arr[10010]={0},m=0,ans=0;
            for(int i=1;i*i<=b;i++){
                int sq=i*i;
                for(int j=((a-1)/i+1)*i;j<=b;j+=i){
                    if(j<sq)continue;
                    if(j==sq)arr[j-a]++;
                    else arr[j-a]+=2;
                }
            }
            
            for(int i=0;i<=b-a;i++){
       //         cout<<arr[i]<<" ";
                if(m==arr[i])ans++;
                else {
                    if(m<arr[i]){
                    m=arr[i];
                    ans=1;
                }
            }
            }
     //       cout<<endl;

            printf("%d\n",ans+1);
    }

    return 0;
    }
