N, Q = map(int, raw_input().split())
arr = map(int, raw_input().split())


def update(x, y):
    global arr
    arr[x-1] = y


def disp(x):
    global arr
    for i in xrange(N):
        if arr[i] >= x:
            print i+1
            return
    print -1


for x in xrange(Q):
    querry = map(int, raw_input().split())
    if querry[0] == 0:
        update(querry[1], querry[2])
    if querry[0] == 1:
        disp(querry[1])
