#include <iostream>
#include <cstdio>

using namespace std;

int main()
{
    int T,N;
    cin>>T;
    while(T--){
    	cin>>N;
    	if ( N == 1){
    		cout<<1<<"\n";
    	}
    	else{
    		unsigned long long int a=1,b=1,c;
    		for(int i=2;i<=N;++i){
    			c = a+b;
    			a = b;
    			b = c;
    		}
    		printf("%llu\n",c);
    	}
    }
}
