#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int set_bit(int a)
{
	int c=0;
	while(a)
	{
		a=a & (a-1);
		c++;
	}
	return c;	
}
int main()
{
	int t,n,k,a[105],b;
	cin >> t;
	while(t--)
	{
		cin >> n >> k;
		for(int i=0; i < n; i++)
		{
			cin >> b;
			a[i]=set_bit(b);
		}
		int sum=0;
		for(int i=0; i < k; i++)
		{
			int m=a[0];
			int id=0;
			for(int i=1; i < n; i++)
			{
				if(a[i]!=-1)
				{
				if(a[i] > m)
				{
				m=a[i];
				id=i;
				}
				}
			}
			a[id]=-1;
			sum=sum+m;
		}
		cout << sum <<"\n";
	}
    //cout << "Hello World!" << endl;
    return 0;
}
