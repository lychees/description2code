#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    
    int n,i;
    cin>>n;
    int s = sqrt(n) +1;
    
    bool *a = new bool[n];
    
    for(i=2;i<n;i++)
    a[i]= true;
    
    for(i=2; i<s; i++)
    {
    	if(a[i])
    	{
    		int j=i;
    		while(j<n)
    		{
    			a[j] = false;
    			j=j+i;
    		}    		
    	a[i] = true;
    	}
    }
    
    long long sum=0;
    i=2;
    while(i<n)
    {
    	if(a[i]) sum += i;
    	i++;
    }
    
    cout<<sum<<endl;
    return 0;
}
