#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	string number;
    	cin >> number;
    	bool flag = false;
    	int digit_sum = 0, digit_diff = 0;
    	if(stoi(number)%21 == 0)
    	{
    		cout << "The streak is broken!" << endl; flag = true;
    	}	
    	else
    	{
    		for(int i = 0; i < number.length() - 1; i++)
    		{
    			if(number[i] == '2' && number[i+1] == '1')
    			{
    				cout << "The streak is broken!" << endl;
    				flag = true;
    				break;
    			}
    		}
    	}
    	if(flag == false)
    	{
    		cout << "The streak lives still in our heart!" << endl;
    	}
    }
    return 0;
}
