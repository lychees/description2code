#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n; cin >> n;
    string s[n];
    for(int i=0;i<n;i++)
    	cin >> s[i];
    sort(s,s+n);
    int occr=1;
    for(int i=n-2;i>=0;i--)
    {
    	if(s[i]==s[i+1])
    		occr++;
    	else
    	{
    		cout << s[i+1] << " " << occr << endl;
    		occr=1;
    	}
    }
    cout << s[0] << " " << occr << endl;
    return 0;
}
