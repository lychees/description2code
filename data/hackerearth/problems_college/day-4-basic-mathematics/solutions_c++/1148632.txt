#include <iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while( t-- )
	{
		unsigned long long n;
		cin>>n;
		int j = 0, i = 0;
		while( i < n )
		{
			j++;
			for( int k = 0; k < j; k++ )
				i += 1;
		}
		if( j%3 == 0 ) cout<<"SL"<<endl;
		else if( j%3 == 1 ) cout<<"LB"<<endl;
		else cout<<"BS"<<endl;
	}
}
