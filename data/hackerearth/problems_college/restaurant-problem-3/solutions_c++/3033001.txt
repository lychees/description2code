#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s;cin>>s;
    int count=0;
    for(int i=0;i<s.length();i++)
    	if((s[i]-'0')%8==0)
    		count++;
    for(int i=1;i<s.length();i++)
    {
    	if(s[i-1]=='0')
    		continue;
    	if( ((s[i-1]-'0')*10+(s[i]-'0'))%8==0)
    		count++;
  	}
  	int nonzeros=0;
  	for(int i=2;i<s.length();i++)
  	{
    	if(s[i-2]!='0')
    		nonzeros++;
    	if( ((s[i-2]-'0')*100+(s[i-1]-'0')*10+(s[i]-'0'))%8 == 0)
    		count+=nonzeros;
  	}
  	cout<<count;
    return 0;
}
