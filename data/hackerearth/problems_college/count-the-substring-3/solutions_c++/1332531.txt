#include <iostream>
using namespace std;


int main()
{
    int t;
    string s;
    scanf("%d",&t);
    while(t--)
    {
    	cin>>s;
    	long long ans=0;
    	int ll=s.length();
    	int l=0,r=0,x=0,y=0,z=0,i;
    	if(s[0]=='x') x++;
    	else if(s[0]=='y') y++;
    	else if(s[0]=='z') z++;
    	int last=-1;
    	for(i=1;i<ll;++i)
    	{
    		if(s[i]=='x') x++;
    		else if(s[i]=='y') y++;
    		else if(s[i]=='z') z++;
    		
    		if(x!=0 && y!=0 && z!=0)
    		{
    			int len=i-l;
    			//printf("%d %d\n",l,last);
    			if(l>last)
    				ans+=((long long)len*(len+1))/2;
    			else{
    				int len1=i-last-1;
    				ans+=((long long)len1*(len1+1))/2;
    				ans+=((long long)len1*(last-l+1));
    			}
    			last=i-1;
    			for(;l<i;++l)
    			{
    				if(s[l]=='x') x--;
    				else if(s[l]=='y') y--;
    				else if(s[l]=='z') z--;
    				if(x==0 || y==0 || z==0)
    				{
    					l++;
    					break;
    				}
    			}
    			//printf("%lld %d %d\n",ans,l,i);
    		}
    	}
    	int len=i-l;
    			if(l>last)
    				ans+=((long long)len*(len+1))/2;
    			else{
    				int len1=i-last-1;
    				ans+=((long long)len1*(len1+1))/2;
    				ans+=((long long)len1*(last-l+1));
    			}
    			last=i-1;printf("%lld\n",ans);		
    }
    return 0;
}
