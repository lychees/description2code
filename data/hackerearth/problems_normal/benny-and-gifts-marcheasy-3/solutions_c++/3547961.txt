#include <cstdio>
#include <cstring>
#include <map>

using namespace std;

const int N = 100010;
int n, x, y, a;
char s[N];
map<pair<int, int>, bool> m;

int main()
{
	scanf("%s", s + 1);
	n = strlen(s + 1);
	m[make_pair(0, 0)] = true;
	for (int i = 1; i <= n; i++) 
	{
		if (s[i] == 'U') x++; else
		if (s[i] == 'D') x--; else
		if (s[i] == 'L') y++; else
		y--;
		if (m[make_pair(x, y)]) a++; else
		m[make_pair(x, y)] = true;
	}
	printf("%d", a);
    return 0;
}
