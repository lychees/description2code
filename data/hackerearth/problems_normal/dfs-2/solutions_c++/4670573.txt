#include<iostream>
#include<cstdio>
#include<algorithm>
#include<vector>
using namespace std;
int n;


int main()
{
	scanf("%d",&n);
	vector<int>a(n);
	for(int i=0;i<n;i++)
	scanf("%d",&a[i]);
	sort(a.begin(),a.end());
	int q;
	scanf("%d",&q);
	int x,y;
	vector<int>::iterator ans1,ans2;
	int low,up;
	while(q--)
	{
		scanf("%d%d",&x,&y);
		ans1=lower_bound(a.begin(),a.end(),x);
		ans2=upper_bound(a.begin(),a.end(),y);
		low=ans1-a.begin();
		up=ans2-a.begin();
		printf("%d\n",up-low);
	}
	return 0;
}
