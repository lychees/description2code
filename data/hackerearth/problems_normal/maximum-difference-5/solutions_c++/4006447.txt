#include<iostream>
#include<cmath>
using namespace std;
int n,a[1000],s1=0,s2=0;
void sum1()
{
	int l,mini[1000],maxi[1000],b[1000];
	int minsize,maxsize;
	minsize=n/2;
	maxsize=n-minsize;
	l=n-1;
	for(int i=0;i<minsize;i++)
		mini[i]=a[i];
	for(int i=0;i<maxsize;i++)
		maxi[i]=a[l--];
	int j=0,k=0;
	for(int i=0;i<n;i++)
	{
		if(i%2==0)
			b[i]=maxi[j++];
		else
			b[i]=mini[k++];
	}
	for(int i=0;i<n-1;i++)
		s1+=abs(b[i]-b[i+1]);
}
void sum2()
{
	int l,mini[1000],maxi[1000],b[1000];
	int minsize,maxsize;
	maxsize=n/2;
	minsize=n-maxsize;
	l=n-1;
	for(int i=0;i<minsize;i++)
		mini[i]=a[i];
	for(int i=0;i<maxsize;i++)
		maxi[i]=a[l--];
	int j=0,k=0;
	for(int i=0;i<n;i++)
	{
		if(i%2==0)
			b[i]=mini[j++];
		else
			b[i]=maxi[k++];
	}
	for(int i=0;i<n-1;i++)
		s2+=abs(b[i]-b[i+1]);
}

int main()
{
	cin>>n;
	for(int i=0;i<n;i++)
		cin>>a[i];
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
		{
			if(a[i]<a[j])
			{
				int x=a[i];
				a[i]=a[j];
				a[j]=x;
			}
		}
	}
	sum1();
	sum2();
	if(s1>s2)
		cout<<s1;
	else
		cout<<s2;
	return 0;
}
