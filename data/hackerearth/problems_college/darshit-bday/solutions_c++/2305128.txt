#include <iostream>

using namespace std;
int main()
{
	long int t,b,w,x,y,z;
	cin >> t;
	while(t--)
	{
		cin >> b >> w >> x >> y >> z;
		if((x+z)<y)
		{
			cout << (w*(x+z))+(b*x)<<endl;
		}
		else if((y+z)<x)
		{
			cout << (b*(y+z))+(w*y)<<endl;
		}
		else
		{
			cout << (b*x)+(w*y)<<endl;
		}
	}
	return 0;
}
