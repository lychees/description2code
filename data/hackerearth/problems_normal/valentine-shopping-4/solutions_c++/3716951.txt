#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,m;
    	cin>>n>>m;
    	
    	while(n--)
    	{
    		int x=1000000007,inx=0;
    		
	   		for(int i=1;i<=m;i++)
	    	{
	    		int a,b,c,d;
	    		
	    		cin>>a>>b>>c;
	    		
	    		d=(100-a)*(100-b)*(100-c);
	    		
	    		if(d<x)
	    			x=d,inx=i;
	   		}
	   		
	   		cout<<inx<<' ';
    	}
    	
    	cout<<endl;
    }
    return 0;
}
