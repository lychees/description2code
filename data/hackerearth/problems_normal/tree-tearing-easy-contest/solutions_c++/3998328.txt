#include <iostream>
#include <bits/stdc++.h>
using namespace std;
vector<int> v[10002];
//int dp[10002];
int k,n;
int visited[10002];
int ans=0;
int dfs(int i)
{
	if(visited[i]==1)
	return 0;
	
	visited[i]=1;
	int a=1;
	int l=v[i].size();
	for(int j=0;j<l;j++)
	{
		if(visited[v[i][j]]==0)
		a=a+dfs(v[i][j]);
	}
	if(a>=k)
	{
		ans++;
		return 0;
	}
	return a;
}
int main()
{
    
    cin>>k>>n;
    //memset(dp,0,sizeof(dp));
    memset(visited,0,sizeof(visited));
    for(int i=2;i<=n;i++)
    {
    	int a;
    	cin>>a;
    	v[i].push_back(a);
    	v[a].push_back(i);
    }
    
    for(int i=1;i<=n;i++)
    {
    	if(visited[i]==0)
    	int a=dfs(i);
    }
    cout<<ans<<endl;
    return 0;
}
