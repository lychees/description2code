#include <bits/stdc++.h>
using namespace std;


#define cin(x)    scanf("%d",&x)

double dp[1001][1001];

double solve(int runs , int balls)
{
	if(runs <= 0) return 1.0;
	else if(balls <= 0) return 0.0;
	else if(dp[runs][balls] >= 0) return dp[runs][balls];
	double ret = 0;
	for(int i = 0; i <= 6; i++)
		if(i != 5)
			ret += solve(runs - i , balls - 1);
	ret += solve(runs , balls - 1);
	ret /= 7;
	return (dp[runs][balls] = ret);
}

int main()
{
	int t;
	cin(t);
	while(t--)
	{
		memset(dp , -1 , sizeof(dp));
		int r,b;
		cin(r);
		cin(b);
		printf("%.6lf\n", solve(r,b));
	}
	return 0;
}
