n = input()
day=[0]*31
while n:
    args = raw_input()
    arr = [int(s) for s in args.split() if s.isdigit()]
    for i in arr:
        if args[0]=='G':
           day[i]+=2
        else:
           day[i]+=1
    n-=1
date = "Date"
for i in range(1,31):
    if i == 19 or i == 20:
        continue
    if day[i] >= day[19] and  day[i] >= day[20]:
        date = "No Date"
        break
print date
