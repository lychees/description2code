#include <iostream>
using namespace std;

int multiply(int x,int res[],int size)
{
	int count=0,prod;
	for(int i=0;i<size;i++)
	{
		prod=res[i]*x + count;
		res[i]=prod%10;
		count=prod/10;
		
	}
     
     while(count)
     {
     	res[size++]=count%10;
     	count=count/10;
     }
     
     return size;
}


void factorial(int n)
{
	int res[50000],size=1;
	res[0]=1;
	
	for(int i=2;i<=n;i++)
	{
	  size=multiply(i,res,size);
	}
	int count=0;
	for(int i=size-1;i>=0;i--)
	{
		cout<<res[i];
	}
	cout<<endl;
	
}
	

int main()
