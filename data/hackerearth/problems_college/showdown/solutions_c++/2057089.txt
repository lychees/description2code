void setup()
{
	//cin.tie(0);
	ios_base::sync_with_stdio(false);
}

int main()
{
	setup();
	int n,x;
	cin>>n;
	VI a;
	VI b;
	rep(i,0,n)
	{
		cin>>x;
		a.pb(x);
	}
	rep(i,0,n){
		cin>>x;
		b.pb(x);
	}
	int ans = 0;
	sort(a.begin(),a.end());
	sort(b.begin(),b.end());
	int mark[n];
	int mk2[n];
	cl(mk2);
	cl(mark);
	int same[n];
	cl(same);
	for(int i = 0; i < n; i++)
	{
		for(int j= 0 ; j < n;j++)
		{
			if(a[i] == b[j])
			{
				same[i] = 1;
			}
		}
	}
	for(int  i = n-1;i>=0;i--)
	{
		int k = -1;
		for(int j = 0; j <n;j++)
		{
			if(b[i]< a[j] && same[j] == 0 && mark[j] == 0)
			{
				k = j;
				break;
			}
		}
		if(k!=-1)
		{
			ans = ans + 100;
			mark[k] = 1;
			mk2[i] = 1;
			continue;
		}
		for(int j = 0 ; j < n ;j++){
			if((b[i] < a[j] && same[j] != 0 && mark[j] == 0))
			{
				k = j;
				break;
			}
		}
		if(k!=-1)
		{
			ans = ans + 100;
			mark[k] = 1;
			mk2[i] = 1;
			continue;
		}
	}
	for(int i = 0; i < n;i++)
	{
		if(mark[i] == 0)
			{
				for(int  j = 0; j < n;j++)
				{
					if(a[i] == b[j] && mk2[j] == 0)
					{
						ans +=50;
						mark[i] = 1;
						mk2[j] = 1;
						break;
					}
				}
			}
	}
	cout<<ans<<endl;




}
