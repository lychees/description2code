#include <iostream>
using namespace std;
#include<vector>
int main()
{
    int n,q;
    cin>>n>>q;
    vector< vector< int> >arr(n+1);
    for(int i =0;i<q;i++)
    {
    	int xx,yy,x,y;
    		bool bt =true;
    	cin>>xx>>yy;
    		x = min(xx,yy);
    		y = max(xx,yy);
    	if(x ==y)
    		bt =false;
    	if(arr[x].size() >0 || arr[y].size()>0)
    		bt =false;
    	for(int i = x+1;i<y && bt;i++)
    	{
    		for(int j =0;j<arr[i].size() && bt;j++)
    		{
    			if(arr[i][j]<=x || arr[i][j]>=y)
    			{
    			//	for(int k =0;k<arr[i].size() && bt;k++)
    			//	cout<<arr[i][k]<<" ";
    			//	cout<<"HI"<<endl;
    				
    				bt =false;	
    			}
    		}
    	}
    	if(bt){
    	cout<<"YES"<<endl;
    	arr[x].push_back(y);arr[y].push_back(x);}
    	else
    	cout<<"NO"<<endl;
    }
    return 0;
}
