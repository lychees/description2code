'''

for _ in range(int(raw_input())):
    inP = raw_input()
    params = [int(i) for i in inP.split()]

    n = len(params) - 1

    ans = 0
    count = 1
    while n > 0:
        n -= 1
        ans ^= params[count]
        count += 1
    if ans % 2 == 0:
        print 'Yes'
    else:
        print 'No'
