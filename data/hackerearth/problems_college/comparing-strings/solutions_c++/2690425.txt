#include <iostream>
using namespace std;

int main()
{
    int tst;
    cin>>tst;
    cont:
    while(tst--)
    {
    	string a,b;
    	int flag = 0;
    	cin>>a>>b;
    	for(int i = 0 ;i<a.length();i++)
    	for(int j = 0 ;j<b.length();j++)
    	if(a[i]==b[j]) {cout<<"YES\n"; flag = 1; goto cont;}
    	if(!flag) cout<<"NO\n";
    }
    return 0;
}
