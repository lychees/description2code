#include <string.h>
#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>

#define FORi(x) for(int i=0;i<(x);i++)
#define FORj(x) for(int j=0;j<(x);j++)
#define TC int T; cin>>T; while(T--)
#define ll long long int
using namespace std;
int main()
{
    TC
    {
        int N,K;
        cin>>N>>K;
        int a[11]={0},x,maxi=0;
        FORi(N)
        {
            cin>>x;
            a[x]++;
        }
        for(int i=1;i<11;i++)
        {
        	if(maxi<a[i])
                maxi=a[i];
		}
        cout<<N-maxi<<endl;
    }
    return 0;
}
