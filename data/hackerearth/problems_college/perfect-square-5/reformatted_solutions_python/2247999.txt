import math
tc = int(input())
while tc > 0:
    s = int(input())
    t = math.sqrt(s)
    if math.floor(t) == t:
        print('YES')
    else:
        if t-math.floor(t) < math.ceil(t)-t:
            print int((math.floor(t)*math.floor(t)))-s
        else:
            print '+'+str(int(abs((math.ceil(t)*math.ceil(t))-s)))
    tc = tc-1
