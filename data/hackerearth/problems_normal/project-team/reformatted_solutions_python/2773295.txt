
tc = int(raw_input())

for t in xrange(tc):
    num = raw_input().split()
    N = int(num[0])
    kl = map(int, num[1:])
    kl = sorted(kl)
    klupper = kl[N/2:]
    kllower = kl[:N/2]
    klupper = sorted(klupper, reverse=True)
    sum = []
    for i in xrange(N/2):
        sum.append(kllower[i]+klupper[i])
    print max(sum)-min(sum)
