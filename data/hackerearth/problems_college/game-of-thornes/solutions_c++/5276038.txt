#include <iostream>
using namespace std;

int main()
{
    int flag=0;
    string s; cin>>s;
    int arr[26]={0};
    for(int i=0;i<s.length();i++){
    	arr[s[i]-'a']++;
    } 
    for(int i=0;i<26;i++){
    	if(arr[i]&1){
    		flag++;
    		//break;
    	}
    }
    if(flag>=2) cout<<"NO"<<endl;
    else cout<<"YES"<<endl;
    return 0;
}
