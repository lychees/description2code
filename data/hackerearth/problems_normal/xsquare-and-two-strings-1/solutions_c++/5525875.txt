#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	char *s1,*s2;
    	s1= new char[100000];
    	s2= new char[100000];
    	cin >>s1>>s2;
    	//cout<<s1<<endl<<s2;
    	int c1[26]={0},c2[26]={0};
    	int k=0;
    	while(s1[k])
    	{
    		c1[s1[k]-'a']++;
    		k++;
    	}
    	k=0;
    	while(s2[k])
    	{
    		c2[s2[k]-'a']++;
    		k++;
    	}
    	int flag=0;
    	for(int i=0;i<26;i++)
    	{
    		if(c1[i]!=0 && c2[i]!=0 && ((c1[i]<=c2[i]) || (c1[i]>c2[i])))
    		{
    			cout<<"Yes";
    			flag=1;
    			break;
    		}
    		else
    		{
    			continue;
    		}
    	}
    	if(flag==0)
    		cout<<"No";
    	cout<<endl;
    	delete s1;
    	delete s2;
    }
    return 0;
}
