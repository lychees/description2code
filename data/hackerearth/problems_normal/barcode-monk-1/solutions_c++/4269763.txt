#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define pii pair<int,int>
#define pll pair<ll,ll>
#define mp make_pair
#define ff first
#define ss second
#define pb push_back
#define pf pop_front
#define IOS ios::sync_with_stdio(false)
#define MOD 1000000007
int n , k;
int dp[1005][2];
int isposs[100005];
int a[100005];
int main(){

    int  i , j , el;
    cin >> n >> k;	
    for(i = 0 ; i < k ; i++){
        cin >> a[i];
        isposs[a[i]] = 1;
    }
    for(i = 1 ; i <= n ; i++){
        for(j = 0 ; j < k ; j++){
            isposs[i] = isposs[i] | isposs[i-a[j]];
        }
    }
    dp[0][0] = 1; 
    for(i = 1 ; i <= n ; i++){
        dp[i][0] = (dp[i-1][0] + dp[i-1][1])%MOD;
        for(j = 1 ; j <= i ; j++){
            if(isposs[j]){
                dp[i][1] = (dp[i][1] + dp[i-j][0])%MOD;
            }
        }
    }
    cout <<  (dp[n][0] + dp[n][1])%MOD;
	return 0;
}
