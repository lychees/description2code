#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t = 0;
	scanf("%d",&t);
	
	int arr[1000001];
	
	for(int i = 0 ; i<1000001 ; i++)
	arr[i]=-1;
	
	arr[0]=-1;
	arr[1]=-1;
	arr[2]=1;
	arr[3]=1;
	arr[4]=2;
	arr[5]=1;
	arr[6]=2;
	arr[7]=1;
	arr[8]=2;
	
	for(int i = 9 ; i<1000001 ; i++)
	{
		arr[i] = min(arr[i-2],min(arr[i-3],(min(arr[i-5],arr[i-7]))))+1;
	}
	
	while(t--)
	{
		int n = 0;
		scanf("%d",&n);
		
		printf("%d\n",arr[n]);
		
	}
	
	
	
	
}
