#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n;
    cin >> n;

    int a[n];
    for (int i = 0;i <n;i++)
        cin >> a[i];

    int w,h;
    cin >> w >> h;
    float total = 0;
    int ans = 0;
    int num;
    int sum = 0;

    for (int i = 0;i <n;i++)
    {
        num = a[i];
        int j = 0;
        sum = 0;
        while(num > 0)
        {
            sum = sum + num % 10;
            num = num / 10;
        }

        total = (a[i] * (sum));

        if (a[i] >= 1 && a[i] <= 10)
        {
            total = total - (total * 25)/100;
        } else if (a[i] >= 11 && a[i] <= 20)
        {
            total = total - (total * 30)/100;
        } else if (a[i] >= 21 && a[i] <= 50)
        {
            total = total - (total * 20)/100;
        }

        ans = ans + int(total * w * h);
        //cout << total * w * h << endl;
    }

    cout << ans << endl;

    return 0;
}
