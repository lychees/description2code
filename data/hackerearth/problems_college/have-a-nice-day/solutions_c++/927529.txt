#include <iostream>
#include<algorithm>
#include<assert.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    assert(t>=1 && t<=1500);
    while(t--){
    	int c;
    	cin>>c;
    	int m[c],w[c],total[c];
    	for(int i=0;i<c;i++)
    	cin>>m[i];
    	for(int i=0;i<c;i++)
    	cin>>w[i];
    	long int sum=0;
    	sort(m,m+c);
    	sort(w,w+c);
    	for(int i=0;i<c;i++)
    	total[i]=w[i]*m[i];
    	for(int i=0;i<c;i++)
    	sum+=total[i];
    	cout<<sum<<endl;
    }
    return 0;
}
