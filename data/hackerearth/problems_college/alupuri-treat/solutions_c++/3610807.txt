#include<iostream>
#include<cmath>
using namespace std;
int main()
{
    long long n,i,j,a[100][100],b=0,c=0;
    cin>>n;
    for(i=0;i<n;i++)
        for(j=0;j<n;j++)
            cin>>a[i][j];
    for(i=0;i<n;i++)
            b+=a[i][i];
    for(i=0;i<n;i++)
            c+=a[i][n-1-i];
    cout<<abs(b-c);
}
