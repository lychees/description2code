#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t,v,temp,x,y,i;
    scanf("%d",&t);
    while(t)
    {
        bool result=false;
        scanf("%d",&v);
        temp=v;
        x=temp/3;
        y=(temp%3)%5;
        if(v<=2)
            result=false;
        else
        {
            if(y==0)
                result=true;
            else
            {
                while(y!=0)
                {
                    temp=v;
                    y=(temp-3*x)%5;
                    if(y==0)
                    {
                        result=true;
                        break;
                    }
                    else
                        x=x-1;
                }
            }
        }

        if(result==false)
            printf("%d\n",-1);
        else
        {
            for(i=1;i<=3*x;i++)
                printf("%d",5);
            for(i=1;i<=(v-3*x);i++)
                printf("%d",3);
            printf("\n");
        }
        t=t-1;
    }
    return 0;
}
