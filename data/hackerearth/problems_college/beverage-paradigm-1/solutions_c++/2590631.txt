#include<bits/stdc++.h>
using namespace std;
main()
{
	ios_base::sync_with_stdio(0);cin.tie(0);
	long long int n,i,x,sum=0,j,k;
	cin>>n;
	long long int a[n+1];
	for(i=1;i<=n;i++)
	{
		cin>>a[i];
	}
	cin>>x;
	for(i=1;i<=n;i++)
	{
		for(j=i+1;j<=n;j++)
		{
			for(k=i+1;k<=n;k++)
			{
				sum=a[i]+a[j]+a[k];
				if(sum==x)
				goto ll;
			}
		}
	}
	 cout<<"False"<<endl;
	goto lx;
	ll :cout<<"True"<<endl;
	lx: ;
}
