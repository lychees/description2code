#include <iostream>
#include<algorithm>
using namespace std;
#include<cmath>
#define ll long long
#define ull unsigned long long
int main()
{
    //cout << "Hello World!" << endl;
    ll n,i,j,arr1[200000],arr2[200000];
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>arr1[i]>>arr2[i];
    }
    sort(arr1,arr1+n);     
    ll sum = 0;
    for(i=n-1;i>=0;i--)
    {
     sum += ((arr1[i]*(i) - arr1[i]*(n-i-1))%1000000007);
    }
    //cout<<sum<<endl;
    sort(arr2,arr2+n); 
    sum=sum%1000000007;
    ll sum1 = 0;
    for(i=n-1;i>=0;i--)
    {
      sum1 += ((arr2[i]*(i) - arr2[i]*(n-i-1))%1000000007);
    }
    sum1=sum1%1000000007;
    //cout<<sum1<<endl;
    cout<<(abs(sum)+abs(sum1))%1000000007<<endl;
    return 0;
}
