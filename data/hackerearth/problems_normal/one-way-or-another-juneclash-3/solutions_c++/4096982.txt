#include<bits/stdc++.h>
using namespace std;

typedef long long int LL;
typedef unsigned long long int ULL;
typedef pair<LL, LL> lipair;

vector< pair<lipair, LL> > v;

int main()
{
	int t, n, i, j, flag;
	LL r, c, x, col, val, tcol, tval, row, trow;
	
	scanf("%d", &t);
	
	while(t--)
	{
		v.clear();
		
		scanf("%d", &n);
		
		for(int i=0;i<n;++i)
		{
			scanf("%lld %lld %lld", &r, &c, &x);
			v.push_back(make_pair(make_pair(r, c), x));
		}
		
		sort(v.begin(), v.end());
		
		//for(i=0;i<n;++i)cout<<"("<<v[i].first.first<<", "<<v[i].first.second<<") -> "<<v[i].second<<endl;
		i = j = flag = 0;
		
		while(i<n)
		{
			//Check with previous
			if(i != 0)
			{
				row = v[i-1].first.first;
				col = v[i-1].first.second;
				val = v[i-1].second;
				
				trow = v[i].first.first;
				tcol = v[i].first.second;
				tval = v[i].second;
				
				//Even even
				if((col%2==0 && val%2==0) || (col%2==1 && val%2==1))
				{
					//Row difference is odd 
					if((trow-row)%2 == 1)
					{
						if(tcol%2== 0 && tval%2==1)flag = 0;
						else if(tcol%2== 1 && tval%2==0)flag = 0;
						else
						{
							//cout<<"Error with previous row at "<<trow<<endl;
							flag = 1;
							break;
						}	
					}
					
					//Even difference 
					else
					{
						if(tcol%2== 0 && tval%2==0)flag = 0;
						else if(tcol%2== 1 && tval%2==1)flag = 0;
						else
						{
							//cout<<"Error with previous row at "<<trow<<endl;
							flag = 1;
							break;
						}	
					}
					
				}
				
				else
				{
					//Row difference is even
					if((trow-row)%2 == 0)
					{
						if(tcol%2== 0 && tval%2==1)flag = 0;
						else if(tcol%2== 1 && tval%2==0)flag = 0;
						else
						{
							//cout<<"Error with previous row at "<<trow<<endl;
							flag = 1;
							break;
						}	
					}
					
					//Odd difference 
					else
					{
						if(tcol%2== 0 && tval%2==0)flag = 0;
						else if(tcol%2== 1 && tval%2==1)flag = 0;
						else
						{
							//cout<<"Error with previous row at "<<trow<<endl;
							flag = 1;
							break;
						}	
					}
				}
			}
			
			col = v[i].first.second;
			val = v[i].second;
			
			//Check for same row
			for(j=i+1;v[j].first.first == v[i].first.first && j<n; ++j)
			{
				tcol = v[j].first.second;
				tval = v[j].second;
				
				//Even even
				if((col%2==0 && val%2==0) || (col%2==1 && val%2==1))
				{
					if(tcol%2== 0 && tval%2==0)flag = 0;
					else if(tcol%2== 1 && tval%2==1)flag = 0;
					else
					{
						flag = 1;
						break;
					}
				}
				
				else
				{
					if(tcol%2== 0 && tval%2==1)flag = 0;
					else if(tcol%2== 1 && tval%2==0)flag = 0;
					else
					{
						flag = 1;
						break;
					}
				}
			}
			
			if(flag)
			{
				//cout<<"Error with same row"<<endl;
				break;
			}
			
			i = j;
		}
		
		if(flag)printf("No\n");
		else printf("Yes\n");
	}
	return 0;
}
