#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	int m =t,j,i;
	vector<int>count[60000];
	for( j=1;j<=m;j++)
	{
		int n;
		cin>>n;
		int i;
		int c[600]={0};
		for(i=0;i<n;i++)
		{
			int x;
			cin>>x;
			if(c[x]==0)
			{
				count[x].push_back(j);
				c[x]++;
			}
		}
	}
	long long sum1=0,sum2=0;
	for(i=0;i<=501;i++)
	{
		if(count[i].size()!=0)
		{
			sum1 = sum1 + i;
		}
		
		if(count[i].size()>=1 && count[i].size()!=m)
		{
			sum2 = sum2+i;
		}
	}
	cout<<sum2-sum1<<"\n";
}
