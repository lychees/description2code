from collections import Counter

counts = dict()
max_occur, char = 0, None
for ch in raw_input().strip():
    if ch not in counts:
        counts[ch] = 1
    else:
        counts[ch] += 1
    if counts[ch] > max_occur:
        max_occur = counts[ch]
        char = ch
    elif counts[ch] == max_occur and ord(ch) < ord(char):
        char = ch
print char, max_occur
