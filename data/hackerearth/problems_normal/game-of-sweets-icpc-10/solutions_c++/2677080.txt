#define mp make_pair
#include <iostream>
#include <bitset>
#include <algorithm>
#include <cstring>
#include <vector>
#include <cstdio>

using namespace std;

bitset<100000 + 1> bs;

int main() {
    int N, a = 0, ct = 0;
    scanf("%d", &N);
    bool zero = false;
    while(N--) {
        scanf("%d", &a);
        if(bs[a])
            zero = true;
        bs[a] = true;
    }
    if(zero)
        ct = 1;
    for(int i = 1; i <= 100000; ++i)
        if((bs & (bs >>i)).any())
            ++ct;
    printf("%d\n", ct);
    
}
