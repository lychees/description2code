#include <iostream>
using namespace std;

int main()
{
	int prime[1001];
	for(int i=0;i<1001;i++)
		prime[i]=0;
	for(int i=2;i*i<1001;i++)
		for(int j=i;j*i<1001;j++)
			prime[i*j]++;
	int ctr[1001];
	ctr[0]=0;
	ctr[1]=0;
	for(int i=2;i<1001;i++)
	{
		ctr[i]=ctr[i-1];
		ctr[i]+=prime[i];
	}
	int T;
	cin>>T;
	while(T--)
	{
		int n;
		cin>>n;
		cout<<(n+ctr[n])<<endl;
	}
    return 0;
}
