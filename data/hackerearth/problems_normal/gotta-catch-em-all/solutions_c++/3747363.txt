#include<bits/stdc++.h>
using namespace std;
#define ll long long
# define pb push_back
# define mp make_pair
int main()
{
	ll n,arr[100010];
	scanf("%lld",&n);
	ll i,tmp=0,a=0;
	for(i=0;i<n;i++)
	{
		scanf("%lld",&arr[i]);
	}
		sort(arr,arr+n);
		for(i=n-1;i>=0;i--)
		{
			if(arr[i]+1+a>tmp)
				tmp=arr[i]+1+a;
			a++;
		}
		printf("%lld\n",tmp+1);
	return 0;
}
