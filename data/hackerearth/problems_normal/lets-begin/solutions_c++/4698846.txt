#include <bits/stdc++.h>
using namespace std;
   int n,dp[1000001];
bool check(int i,int val){
	if(i-val<0)
	return false;
	if(dp[i-val]==INT_MAX)
	return false;
	return true;
}
int main()
{
    //cout << "Hello World!" << endl;
    int t,i;
    scanf("%d",&t);
    for(i=1;i<=1000000;i++){
    	dp[i]=INT_MAX;
    	if(check(i,2))
    	dp[i]=min(dp[i],1+dp[i-2]);
    	if(check(i,3))
    	dp[i]=min(dp[i],1+dp[i-3]);
    	if(check(i,5))
    	dp[i]=min(dp[i],1+dp[i-5]);
    	if(check(i,7))
    	dp[i]=min(dp[i],1+dp[i-7]);
    }
    dp[1]=-1;
    while(t--){
    	scanf("%d",&n);
 		cout<<dp[n]<<endl;
    }
    
    
    return 0;
}
