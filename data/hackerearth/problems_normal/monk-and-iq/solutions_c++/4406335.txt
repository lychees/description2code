#include <bits/stdc++.h>
#define lim 100000
#define ll long long
using namespace std;

int main()
{
    int c,p,n,i,x,j;
    ll z;
    vector<int> y1(lim,0),y2(lim,0),x2(lim,0),m(lim,0);
    priority_queue<pair<ll,int> > Q;
    cin>>c>>p>>n;
    for(i=1; i<=n;i++){
    	cin>>y1[i];
    	Q.push(make_pair(-1*y1[i],-1*i) );
    	m[i]=1;
    }
    for(i=n+1;i<=c;i++){
    	Q.push(make_pair(0,-1*i));
    }
    for(i=1;i<=p;i++){
    	cin>>x2[i];
    	pair<ll,int> t=Q.top();
    	Q.pop();
    	j=t.second;
    	j*=-1;
    	y2[j]=y1[j];
    	y1[j]=x2[i];
    	m[j]++;
    	z=(y1[j]+y2[j])*m[j];
    	cout<<j<<" ";
    	Q.push(make_pair(-1*z,-1*j) );
    }
    cout<<endl;
    return 0;
}
