x = raw_input()
x = list(x)
z = [0]*200
k = -1
for i in x:
    z[ord(i)] = z[ord(i)]+1
    if (z[ord(i)] >= k):
        k = z[ord(i)]
for i in range(0, 200):
    if (k == z[i]):
        print str(chr(i))+" "+str(z[i])
        break
