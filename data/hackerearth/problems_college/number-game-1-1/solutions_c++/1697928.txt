#include<iostream>
    using namespace std;
int main()
{
    int i,j,n,T;
    cin>>T;
    while(T--)
    {
        cin>>n;
        i=0;
        while(n>1)
        {
            if(n%3==0)
                n/=3;
            else if(n%2==0)
                n/=2;
            else
                n--;
            i++;
        }
        cout<<i<<endl;
    }
    return(0);
}
