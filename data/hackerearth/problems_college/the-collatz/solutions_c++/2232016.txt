#include <iostream>
using namespace std;
int T,N,r_e,r_o,o,e,step,temp;
int main()
{	cin>>T;

	for(int i=0;i<T;i++)
	{
	e=o=0;
    step=0;
    r_e=r_o=0;
	cin>>N;
	temp=N;
		while(temp!=1)
		{
			if(temp%2==0)
			{
				e=e+temp;
				step++;
				temp=temp/2;
	//			cout<<"\n"<<temp<<"\t"<<e;
			}
			else
			{
				o=o+temp;
				step++;
				temp=(3*temp)+1;
	//			cout<<"\n"<<temp<<"\t"<<o;
			}
			
		}
	r_e=e%step;
	r_o=(o+1)%step;
	//cout<<r_e<<"\t"<<r_o<<"\t"<<step<<"\t"<<temp<<"\n";
	if(r_e<r_o)
		cout<<"Odd Rules\n";
	 if(r_e>r_o)
		cout<<"Even Rules\n";
	if(r_e==r_o)
		cout<<"Tie\n";
	}
    
    return 0;
}
