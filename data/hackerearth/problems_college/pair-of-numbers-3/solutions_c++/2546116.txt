#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		int i,count=0,j;
		if(n<=5)
		cout<<0<<endl;
		else{
		
		for(i=3;i<=n;i++)
		{
			for(j=i+1;j<=n;j++)
			{
				if(i!=j){
				
				if((i*j)%(i+j)==0){
				count++;
				}
				}
			}
		}
		cout<<count<<endl;}
	}
}
