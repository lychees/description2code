for i in range(input()):
    num = raw_input()
    num1 = [int(x) for x in num[0:len(num)/2]]
    num2 = [int(x) for x in num[len(num)/2:]]
    sum1 = sum2 = 0
    for l in num1:
        sum1 += l
    for k in num2:
        sum2 += k
    if sum1 == sum2:
        print "YES"
    else:
        print "NO"
