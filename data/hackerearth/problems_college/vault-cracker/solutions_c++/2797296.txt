#include <iostream>
#include<string.h>
using namespace std;
int ar[1000005];
int main()
{
    string s1,s2;
    cin>>s1;
    cin>>s2;
    int l1,l2;
    l1=s1.length();
    l2=s2.length();
    
    int cnt=0;
    memset(ar,0,(l2+1));
    for(int i=0;i<l2;i++)
    	if(s1[l1-1]==s2[i])
    		ar[i]=1;
    for(int i=l1-2;i>=0;i--)
    	for(int j=l2-1;j>=0;j--)
    	{
    		if(s1[i]==s2[j])
    		{
    			for(int k=j+1;k<l2;k++)
    				if(s1[i+1]==s2[k])
    					ar[j]+=ar[k];
    		}
    	}
    for(int i=0;i<l2;i++)
    	if(s1[0]==s2[i])
    		cnt+=ar[i];
    cout<<cnt<<endl;
    return 0;
}
