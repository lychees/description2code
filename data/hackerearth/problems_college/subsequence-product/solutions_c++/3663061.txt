#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n,h,x,g;
    	scanf("%d%d",&n,&h);
    	for(int i=0;i<n;i++)
    	{
    		scanf("%d",&x);
    		if(h!=1)
    		{
    			g=__gcd(h,x);
    			h/=g;
    		}
    		//cout<<x<<" "<<h<<endl;
    	}
    	if(h==1)
    		printf("YES\n");
    	else
    		printf("NO\n");
    }
    return 0;
}
