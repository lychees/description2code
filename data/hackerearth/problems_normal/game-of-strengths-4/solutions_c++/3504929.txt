
#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
typedef long long int ll;
int main()
{
    int i,t,k;
    scanf("%d",&t);
    while(t--)
    {
    int n;
    scanf("%d",&n);
    ll a[n];
    ll sub=0,sum=0;
    for(i=0;i<n;i++)
     scanf("%lld",&a[i]);
    sort(a,a+n);
    for(i=0;i<n-1;i++)
    {
      sub+=((n-1-i)*a[i]);
    }
    for(i=1;i<n;i++)
    {
       sum+=(i*a[i]);
    }
    sum-=sub;
    ll x= ((sum%1000000007)*(a[n-1]%1000000007))%1000000007;
    printf("%lld\n",x);
    }
    return 0;
}
