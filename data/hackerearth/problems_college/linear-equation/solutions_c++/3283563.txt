#include <bits/stdc++.h>
using namespace std;
#define ll long long int
int main(){
	ll t,n,i,a[50005],sum,k,ans;
	cin>>t;
	while(t--){
		cin>>n;
		sum=0;
		for(i=0; i<n; i++)
		 {
		 	cin >> a[i];
		 	sum = sum+a[i];
		 }
		 k = sum/(n-1);
		 for(i=0; i<n; i++){
		 	ans = k - a[i];
		 	printf("%d ",ans);
		 }
		 printf("\n");
	}
	return 0;
}
