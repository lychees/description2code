#include <iostream>
#include <string>
#include <cstdlib>
using namespace std;

int main() {
    int N;
    string s;
    getline(cin, s);
    N = atoi(s.c_str());
    for (int i = 0; i < N; i++) {
        getline(cin, s);
        string ans = "";
        bool found = false;
        for (int j = 0; j < s.size(); j++) {
            
            if (s[j] == '$') {
                found = true;
                continue;
            }


            if (found) {
                if (s[j] >= '0' && s[j] <= '9') {
                    if (s[j] == '0' && ans == "") {

                    } else {
                        ans += s[j];
                    }
                } else if (s[j] != ' ') {
                    break;
                }
            }
        }
        if (ans == "") {
            ans = "0";
        }
        cout << "$" << ans << endl;
    }
}
