#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
    while(t--)
    {
    	int n,k,i,j=0;
    	cin>>n>>k;
    	int a[n];
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
		}
		int sum =0;
		int dp[n+1];
		for(i=0;i<=n;i++)
		dp[i]=0;
		
		i=0;
		int ans;
		while(i<n && sum <= k)
		{
			sum = sum + a[i];
			while(sum>k)
			{
				sum = sum - a[j];
				j++;
			}
		ans  = i-j+1;
		dp[ans]++;
		i++;
		}
		int prefix[n+1];
		sum=0;
		j=0;
		for(i=n;i>=1;i--)
		{
			sum = sum + dp[i];
			prefix[j] = sum;
			j++;
		}
		for(i=j-1;i>=0;i--)
		cout<<prefix[i]<<" ";
		
		cout<<"\n";
	}
}
