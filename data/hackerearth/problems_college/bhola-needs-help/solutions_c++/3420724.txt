vector<ll int>v;
int main()
{
		ll int a,b,i,q,l,r;

		Sl(a);Sl(b);

		ll int hcf = __gcd(a,b);

		for (i = 1; i*i < hcf; i++){
				if(hcf%i == 0){
						v.push_back(i);
						v.push_back(hcf/i);
				}
		}
		if(i*i == hcf)
				v.push_back(i);

		sort(v.begin(),v.end());
		Sl(q);

		while(q--){
				Sl(l);Sl(r);
				ll int x = upper_bound(v.begin(),v.end(),r)-v.begin();
				if(x == 0){
						printf("-1\n");
				}
				if(v[x-1] >= l && v[x-1] <= r){
						Pl(v[x-1]);
				}
				else{
						printf("-1\n");
				}
		}
}
