using namespace std;

#define MAX 100500
#define MOD 1000000007
#define INF 1e9

long long dp[MAX][5][5];
bool exist[MAX][5][5];

long long solve(int n, int present, int lastDig)
{
    if (n == 0) {
        if(present == 1) return 1;
        else return 0;
    }
    if (!exist[n][present][lastDig]) {
        long long ways = 0;
        if (lastDig == 1) {
            ways += solve(n-1, present, 0);
            ways += solve(n-1, present, 1);
            ways %= MOD;
        } else {
            ways += solve(n-1, present, 1);
            int newPresent =  ((present == 0) || (present == 1))? present+1: present;
            ways += solve(n-1, newPresent, 0);
            ways %= MOD;
        }
        exist[n][present][lastDig] = true;
        dp[n][present][lastDig] = ways;
    }
    return dp[n][present][lastDig];
}

int main()
{
    int t;
    scanf("%d", &t);
    while (t--) {
        memset(dp, 0, sizeof(dp));
        memset(exist, 0, sizeof(exist));

        int n;
        scanf("%d", &n);
        long long ways = solve(n-1, 0, 0);
        ways += solve(n-1, 0, 1);
        ways %= MOD;
        printf("%lld\n", ways);
    }
}
