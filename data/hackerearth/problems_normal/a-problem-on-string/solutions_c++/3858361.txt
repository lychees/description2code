#include <iostream>
using namespace std;

int main()
{
    long n,q,xc,yc;
    int a,ax,ay;
    string input;
    long array [26];
    char x,y;
    cin>>n;
    cin>>input;
    cin>>q;
    for (int j = 0; j < 26; j++){ 
   	  array[j]=0;
     }
     
     for (int j = 0; j < input.size(); j++){ 
   	  a=input[j];
   	  array[a-97]++;
     }
    
    for(int i=0;i<q;i++)
    {
    cin>>x;cin>>y;
    ax=x;ay=y;
    if(x!=y)
      cout << array[ax-97]*array[ay-97] << endl;
    else
       cout<<(array[ax-97])*(array[ax-97]-1)/2<<endl;
    	
    }
   return 0;
}
