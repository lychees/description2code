#include <iostream>
using namespace std;

int gcd(int a, int b)
{
    while (b != 0)
    {
        a %= b;
        a ^= b;
        b ^= a;
        a ^= b;
    }

    return a;
}

int foo(int n)

{

  int i,ret = 0;

  for( i = 1;i<=n;i++)
  {

     if (gcd(n,i)==1)

             ret += 1;
   }
  return ret;
}

int SpecialSum(int N)

{

int i,ret=0;

for (i = 1;i<=N;i++)

{

      if(N%i==0)

         ret += foo(i);
 }

 return ret;
}
