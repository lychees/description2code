#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
        int t;
        cin>>t;
        int n,c,i,j;
        while(t--)
        {
            cin>>n>>c;
            int a[n][n];
            int dp[n][n];
            for(i=0;i<n;i++)
            {
                for(j=0;j<n;j++)
                {
                    cin>>a[i][j];
                    dp[i][j]=0;
                }
            }
            dp[0][0]=0;
            for(i=0;i<n;i++)
            {
                for(j=0;j<n;j++)
                {
                    if(i>0 && j>0)
                        dp[i][j]=a[i][j]+max(dp[i-1][j],dp[i][j-1]);
                    else if(i==0 && j>0)
                        dp[i][j]=a[i][j]+dp[i][j-1];
                    else if(j==0 && i>0)
                        dp[i][j]=a[i][j]+dp[i-1][j];
                }
            }
          /*  for(i=0;i<n;i++)
            {
                for(j=0;j<n;j++)
                {
                    cout<<dp[i][i]<<" ";
                }
                cout<<endl;
            }*/
          //  cout<<dp[n][n]<<endl;
            if(dp[n-1][n-1]>=c)
                cout<<dp[n-1][n-1]-c;
            else
                cout<<"-1";
            cout<<endl;
        }

        return 0;
}
