#include <bits/stdc++.h>
using namespace std;
const long long int MOD = 1000000007;
int main()
{
	long long int a,b,c,d,e,f;
	cin>>a>>b>>c>>d>>e>>f;
	int q;
	cin>>q;
	while(q--)
	{
		long long int n;
		cin>>n;
		if(n == 0)
			cout<<0<<"\n";
		else
		{
			long long int ans = (f + (((n-1)*f)%MOD)*d)%MOD;
			cout<<ans<<"\n";
		}
	}
	return 0;
}
