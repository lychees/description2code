const int inf=1e9;
const int MOD=1e9+7;
const int N=0;

ll solve(int n,int x,int y){
    if (!n) return 0;
    if (x!=y) return x+solve(n-1,y,abs(x-y));
    return 2LL*(n/3)*x+(n%3)*x;
}
