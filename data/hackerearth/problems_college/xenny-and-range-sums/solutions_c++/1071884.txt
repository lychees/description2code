#include <iostream>
using namespace std;

int main()
{
    int n,q,l,r;
    int *a;
    cin>>n>>q;
    a=new int[n];
    for(int i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    while(q--)
    {
    	cin>>l>>r;
    	int s=a[r-1],d=a[r-1];
    	int j=r-2;
    	for(int i=r-2;i>=l-1;i--)
    	{
    		s+=a[i];
    		d-=a[i];
    	}
    	cout<<s<<" "<<d<<endl;
    }
    return 0;
}
