#include <iostream>
#include<limits>
#include <bits/stdc++.h>
#include<string.h>
using namespace std;


int subtract(int a,int b){

	if(a<b)
		return (b-a);
	else
		return (a-b);
}

int main()
{
	
    int t,j=1;
    string a,b;
    cin>>t;
    
    while(t--){
    
    	a.clear();
    	b.clear();
    	cin>>a;
    	cin>>b;
    	// if(j>7){
    	// cout<<a<<endl;
    	// cout<<b<<endl;}
    	int zero=0,one=0,qmark=0;
    	
    	for(int i=0;i<a.size();i++){
    		if(a[i]=='0' && b[i]=='1')
    			zero++;
    			
    		if(a[i]=='1' && b[i]=='0')
    			one++;
    		
    		if(a[i]=='?')
    			qmark++;
    	}
    //	cout<<zero<<" "<<one<<" "<<qmark<<endl;
    	int swapno=0; 
    	
    	if(zero==one){
    		cout<<"Case "<<(j)<<": "<<(zero+qmark)<<endl;
    	}
    	else
    	if(zero<one){
    		swapno=zero;
    		
    		one=one-zero;
    		
    		for(int i=0;i<a.size();i++){
    			if(a[i]=='?' && b[i]=='1')
    				swapno++,one--;
    			if(one==0)
    				break;
    		}
    		
    		if(one>0)
    			cout<<"Case "<<(j)<<": "<<"-1"<<endl;
    		else
    			cout<<"Case "<<(j)<<": "<<(swapno+qmark)<<endl;
    	}
    	else{
    		swapno=one;
    		cout<<"Case "<<(j)<<": "<<swapno+qmark+(zero-one)<<endl;
    		
    	}
    	j++;
    }
    return 0;
}
