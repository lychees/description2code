#include <bits/stdc++.h>
using namespace std;

bool compare(int t1,int t2)
{
	return t1>t2;
}

int main()
{
    int n,k;
    cin>>n>>k;
    int A[n+9];
    for(int i=0;i<n;i++)
    	cin>>A[i];
    sort(A,A+n,compare);
    int ans=INT_MAX;
    for(int i=0;i<n-(k-1);i++)
    	ans=min(ans,A[i]-A[i+k-1]);
    cout<<ans;
    return 0;
}
