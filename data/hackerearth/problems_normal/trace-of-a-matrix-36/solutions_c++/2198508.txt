#include<iostream>
using namespace std;
int min(int a,int b)
{
    if(a<b)
    return a;
    
    return b;
}
int main()
{
    int t;
    cin>>t;
    while(t-->0)
    {
        int r,c;
        cin>>r>>c;
        
        int n=min(r,c);
        
        long long int ans=0;
        ans=((long long int)n)*(n+1);
        ans+=((long long int)(r+c)*(n-1)*(n))/2;
        
        cout<<ans<<endl;
    }
    return 0;
}
