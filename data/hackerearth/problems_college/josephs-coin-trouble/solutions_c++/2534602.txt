#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin>>n;
    //cout<<n<<endl;
    int inp;
    string s;
    vector<int>v;
    while(cin>>s)
    {
    	inp=stoi(s);
    	v.push_back(inp);
    }
   /* for(int i=0;i<v.size();i++)
    {
    	cout<<v[i]<<" ";
    }
    cout<<endl;*/
    int T[n+1];
    memset(T,0,sizeof(T));
    T[0]=1;
    for(int j=0;j<v.size();j++)
    {
    	for(int i=v[j];i<=n;i++)
    	{
    		T[i]=T[i]+T[i-v[j]];
    	}
    }
    cout<<T[n];
    
}
