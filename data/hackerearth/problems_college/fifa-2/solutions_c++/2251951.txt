#include <iostream>
#include <stdio.h>
#include <algorithm>

using namespace std;

const int maxn = 112345;

int hm[maxn],aw[maxn],a[maxn],b[maxn];

int main()
{
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		scanf("%d%d",&a[i],&b[i]);
		hm[i]=a[i];
		aw[i]=b[i];
	}
	sort(hm,hm+n);
	sort(aw,aw+n);
	for(int i=0;i<n;i++)
	{
		int pl=upper_bound(hm,hm+n,b[i])-lower_bound(hm,hm+n,b[i]);
		printf("%d %d\n",n+pl-1,n-pl-1);
	}
	return 0;
}
