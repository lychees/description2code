#include<iostream>
#include<cstdio>
#define s(n) scanf("%d",&n)
using namespace std;

int no_of_paths=0;

void cnt(int grid[90][90], int n, int x, int y){

    if(x==n && y==n){
        no_of_paths++;
        return;
    }

    grid[x][y]=1;
    if(x-1>=1 && grid[x-1][y]==0) cnt(grid,n,x-1,y);
    if(x+1<=n && grid[x+1][y]==0) cnt(grid,n,x+1,y);
    if(y-1>=1 && grid[x][y-1]==0) cnt(grid,n,x,y-1);
    if(y+1<=n && grid[x][y+1]==0) cnt(grid,n,x,y+1);
    grid[x][y]=0;

}

int main(){

    int t, dummy,grid[90][90];
    s(t);
    for(int i=1;i<=t;i++){
        for(int j=1;j<=t;j++){
            s(dummy);
            grid[i][j]=dummy;
        }

    }

    cnt(grid,t,1,1);
    cout<<no_of_paths<<endl;


	return 0;
}
