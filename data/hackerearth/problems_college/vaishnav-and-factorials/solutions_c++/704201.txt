
#include<iostream>
using namespace std;
void calFact(int n)
{
int arr[500],m=1,i,temp=0,x,j,ct1,ct2;
arr[1]=1;
for(i=1;i<=n;i++)
{
for(j=1;j<=m;j++)
{
x=(arr[j]*i)+temp;
arr[j]=x%10;
temp=x/10;
}
while(temp!=0)
{
arr[++m]=temp%10;
temp=temp/10;
}
}
ct1=0;
ct2=0;
for(i=m;i>0;i--)
{
if(arr[i]==4 || arr[i]==7)
    ct1++;
}
cout<<ct1<<endl;
}

int main()
{
int t,n;
cin>>t;
while(t--)
{
cin>>n;
calFact(n);
}
return 0;
}
