#include <iostream>
#include <bits/stdc++.h>
#define ll long long
using namespace std;
long long int MOD=1000000007;
int main()
{
    //cout << "Hello World!" << endl;
    ios::sync_with_stdio(false);cin.tie(0);cout.tie(0);
    int t,n;
    ll a[100005];
    cin>>t;
    while(t--){
    	cin>>n;
    	for(int i=0;i<n;i++)
    	cin>>a[i];
    	sort(a,a+n);
    	ll b,sum=0,c,d,e;
    	int j=n-1;
    	for(int i=0;i<n-1;i++){
    		b=a[i+1]-a[i];
    		b=b%MOD;
    		c=(j-i)%MOD;
    		d=(i+1)%MOD;
    		e=(((b*c)%MOD)*d)%MOD;
    		sum=(sum+e)%MOD;
    	}
    	cout<<(sum*a[n-1])%MOD<<"\n";
    }
    return 0;
}
