#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	int list[n+1];
    	bool ans = true;
    	for(int i=1;i<=n;i++){
    		cin>>list[i];
    	}
    	for(int i=1;i<=n;i++){
    		int val = list[i];
    		int index = list[val];
    		if(list[index]!=val){
    			ans = false;
    			break;
    		}
    		
    	}
    if(ans){
    	cout<<"inverse"<<endl;
    }else{
    	cout<<"not inverse"<<endl;
    }
    	
    }
    return 0;
}
