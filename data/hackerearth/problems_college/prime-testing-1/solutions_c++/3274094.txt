#include <iostream>
using namespace std;

int main()
{
    int t,n,x;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	int c=0,i;
    	while(n--)
    	{
    		cin>>x;
    		for(i=2;i*i<=x;i++)
    		if(x%i == 0)
    		break;
    		if(i*i > x && x > 1)
    		c++;
    	}
    	cout<<c<<endl;
    }
    return 0;
}
