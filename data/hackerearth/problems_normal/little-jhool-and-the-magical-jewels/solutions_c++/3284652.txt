#include<bits/stdc++.h>

using namespace std;

int main () {
int t;
cin >> t;

while(t--){
    string str;
    cin >> str;
    int cnt = 0,cnt1 = 0,cnt2 = 0,cnt3 = 0;

    for (int i = 0;i < str.size();i++){
        if (str[i] == 'r')
            cnt++;

        else if(str[i] == 'u')
            cnt1++;
        else if (str[i] == 'b')
            cnt2++;
        else if (str[i] == 'y')
            cnt3++;
    }

    cout << min(min(cnt,cnt1),min(cnt2,cnt3))<<endl;
}

return 0;
}
