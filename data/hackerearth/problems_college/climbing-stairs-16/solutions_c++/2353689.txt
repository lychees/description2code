#include <iostream>
using namespace std;

long long int ways(long long int x, long long int sum)
{
	if(x==2) sum += 2;
	else if(x==1) sum +=1;
	else
	sum += ways(x-1, sum) + ways(x-2, sum);
	return sum;
}

long long int ways_DP(int x)
{
	if(x<4) return x;
	else {
	long long int sum[x];
	sum[0]=0,sum[1]=1,sum[2]=2,sum[3]=3;
	for(int i=4;i<=x;i++)
	{
		sum[i] = sum[i-2]+sum[i-1];
	}
	return sum[x];
	}
}


int main()
{
	int T,N;
	cin>>T;
	while(T--)
	{
		cin>>N;
		cout<<ways_DP(N)<<endl;
	}
    return 0;
}
