#include <iostream>
#include<algorithm>
using namespace std;
bool CheckForSum(int a[],int X,int i,int N)
{
	int st=i,end=N-1;
	while(st<end)
	{
		if(a[st]+a[end]>X)
		{
			end--;
		}
		else if(a[st]+a[end]<X)
		{
			st++;
		}
		else
		{
			return 1;
		}
	}
	return 0;
}	
int main()
{
    int N,i,a[30005],X;
    cin>>N;
    for(i=0;i<N;i++)
    {
    	cin>>a[i];
    }
    sort(a,a+N);
    cin>>X;
    for(i=0;i<N;i++)
    {
    	if(CheckForSum(a,X-a[i],i+1,N))
    	{
    		cout<<"True";
    		return 0;
    	}
    }
    cout<<"False";
    return 0;
}
