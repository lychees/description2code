#include <bits/stdc++.h>
#define MOD 1000000007
using namespace std;

int main()
{
    int t,n,num;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	long long ans=0;
    	for(int i=0;i<n;++i)
    	{
    		scanf("%d",&num);
    		ans|=num;
    	}
    	ans%=MOD;
    	for(int i=1;i<n;++i)
    	{
    		ans<<=1;
    		ans%=MOD;
    	}
    	printf("%lld\n",ans);
    }
    return 0;
}
