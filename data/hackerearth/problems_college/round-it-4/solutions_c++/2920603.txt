#include<bits/stdc++.h>
using namespace std;
int bound(float r)
{
  int near;
  near=ceil(r);
  //cout<<near<<" "<<r<<endl;
  if(near-r<=0.5)
  return near;
  else
  return near-1;
}
int main()
{
  int t;
  cin>>t;
  while(t--)
  {
    int n,b;
    cin>>n>>b;
    float res=((float)n/b);
    int mul=bound(res);
    //cout<<mul<<endl;
    cout<<mul*b<<endl;
  }
  return 0;
}
