def odd(size):
	return ((size+1)/2) ** 2

def even(size):
	size /= 2
	return size ** 2 + size

for _ in xrange(int(raw_input())):
	r, c = map(int, raw_input().strip().split())
	
	if r <= c:
		print even(r), odd(r)
	else:
		diff = r - c
		if c % 2 == 0:
			odd_length = (diff+1) / 2
			even_length = diff / 2
		else:
			odd_length = diff/2
			even_length = (diff+1)/2
		
		odds = odd_length * c
		evens = even_length * c
		
		print even(c) + evens, odd(c) + odds
