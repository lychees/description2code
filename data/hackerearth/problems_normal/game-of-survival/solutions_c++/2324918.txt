#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;
vector<int> animals,swords;
int main()
{
    int t,n,i,k;
    cin>>t;
    while(t--)
    {
    	animals.clear();
    	swords.clear();
    	cin>>n;
    	for(i=0;i<n;i++)
    	{
    		cin>>k;
    		swords.push_back(k);
    	}
    	for(i=0;i<n;i++)
    	{
    		cin>>k;
    		animals.push_back(k);
    	}
    	sort(animals.begin(),animals.end());
    	sort(swords.begin(),swords.end());
    	for(i=0;i<n;i++)
    	if(animals[i]>=swords[i])
    	break;
    	if(i==n)
    	cout<<"YES\n";
    	else
    	cout<<"NO\n";
    }
    return 0;
}
