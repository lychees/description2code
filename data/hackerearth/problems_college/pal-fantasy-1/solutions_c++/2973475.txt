#include <iostream>
#include<string>
using namespace std;

bool isPalin(string str,int start,int end)
{
	for(int i=start;i<end;i++) 
		if(str[i]!=str[end-1])
			return false;
		else end--;
	return true;
}
int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
    	string str;
    	cin>>str;
    	int len=str.length(),i;
    	for(i=0;i<len;i++)
    		if(isPalin(str,i,len))
    			break;
    	cout<<len+i<<endl;
    }
    return 0;
}
