#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;
int main() {
    int t,n,i;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	int a[n];
    	for(i=0;i<n;i++)
    	{
    		scanf("%d",&a[i]);
    	}
    	sort(a,a+n);
    	int c=0;
    	for(i=0;i<n-1;i++)
    	{
    		if(a[i]==a[i+1])
    		c++;
    	}
    	printf("%d\n",c);
    }
    return 0;
}
