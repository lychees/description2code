using namespace std;

LL a[1000006],d[1000006];

int main()
{
	int t,n;
	
	cin>>t;
	
	while(t--)
	{
		cin>>n;
	
		forall(i,0,n)
		 cin>>a[i];
		forall(i,0,n)
		{ cin>>d[i];
		   d[i]=a[i]+d[i];
		}
		
		sort(a,a+n);
		sort(d,d+n);
		
		int i=0,j=0;
		int maxcount=0,count=0;
		while(i<n && j<n)
		{
			if(a[i]<d[j])
			{ count++;
			  i++;
			}
			else if(a[i]==d[j])
			{
				i++;
				j++;
			}
			else
			{
			 count--;
			 j++;
			}
		maxcount = max(maxcount,count);
		
		}
		
		cout<<maxcount<<endl;
		
		
		
		 
	
	}
   return 0;
}
