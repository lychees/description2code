#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long count0=0;
    	long long count1=1;
    	int m;
    	cin>>m;
    	for(int i=1;i<=m;++i)
    	{
    		long long temp0=count0;
    		long long temp1=count1;
    		temp0+=count1;
    		temp1-=count1;
    		temp1+=count0;
    		count0=temp0;
    		count1=temp1;
    	}
    	cout<<count1<<" "<<count0<<endl;
    }
}
