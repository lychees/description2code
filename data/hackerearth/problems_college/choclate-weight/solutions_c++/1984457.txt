#include<iostream>
using namespace std;
int n;
int fun(int arr[],int n,int x)
{
	for(int i=0;i<n;i++)
	{
		if(arr[i]==x)
		{
			return 1;
		}
	}
	for(int i=0;i<n-1;i++)
	{
		for(int j=i+1;j<n;j++)
		{
			if(arr[j]+arr[i]==x)
			return 1;
		}
	}
	return 0;
}
int main()
{
	cin>>n;
	int arr[n];
	for(int i=0;i<n;i++)
	{
		cin>>arr[i];
	}
	int m;
	cin>>m;
	while(m--)
	{
		int x;
		cin>>x;
		if(!fun(arr,n,x))
		{
			cout<<"I Hate You\n";
		}
		else
		{
			cout<<"I Like You\n";
		}
	}
}
