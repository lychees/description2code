#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;

int main(){
	int T; 
	cin >> T;
	while( T-- ){		
		int N,maxi=-1,maxu=-1; cin >> N;
		vector<int> W(N), H(N);
		for(int i = 0; i < N; i++) cin >> W[i];
		for(int i = 0; i < N; i++){ 
			cin >> H[i];
			if( W[0] * H[i] > maxu ){
				maxu = W[0]*H[i];
				maxi = i;
			}
		}
		W.erase( W.begin() );
		H.erase( H.begin() + maxi );
		sort( W.begin(), W.end() );
		sort( H.rbegin(), H.rend() );
		//cout << maxi << " " << maxu << endl;
		bool ok = true;
		for(int i = 0; i + 1 < N && ok; i++){
			if( W[ i ] * H[ i ] >= maxu ){
				ok = false; break;
			}
		}
		cout << (ok?"YES":"NO") << endl;
	}
}
