#include <iostream>
#include <bits/stdc++.h>
#define x 1000000007
using namespace std;

long long int power(long long int a, long long int b){
	if(b==0){
		return 1%x;
	}
	long long int temp = power(a,b/2)%x;
	if(b%2==0){
		return ((temp%x)*(temp%x))%x;
	}
	else{
		return ((temp%x)*(temp%x)*(a%x))%x;
	}
}

int main()
{
    int t;
    long long int n, temp, t1, ans;
    cin>>t;
    while(t--){
    	cin>>n;
    	if(n<3){
    		ans = 0;
    	}
    	else{
    		temp = power(3,n);
    		t1 = power(2,n);
    		t1 = ((t1)%x*(3%x))%x;
    		ans = ((temp%x)-(t1%x)+(3%x))%x;
    		if(ans<0){
    			ans+=x;
    		}
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
