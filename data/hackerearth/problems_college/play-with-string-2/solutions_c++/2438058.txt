#include <bits/stdc++.h>

using namespace std;


int main() {
    int tes;
    cin>>tes;
    while(tes--)
        {
    char s[10000];
    cin>>s;
        int len = strlen(s);
        int flag=0,i;
        for(i=1;i<len;i++)
            {
            if(!(abs(s[i]-s[i-1])==abs(s[len-i]-s[len-i-1])))
            {flag=1;
             break;
            }
     
        }
        if(flag==1)
            cout<<"Not Funny\n";
        else
            cout<<"Funny\n";
    }
    return 0;
}
