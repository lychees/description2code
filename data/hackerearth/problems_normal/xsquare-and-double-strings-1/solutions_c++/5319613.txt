#include <iostream>
using namespace std;
#include <bits/stdc++.h>
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	map<char,int> m;
    	int l= s.length();
    	bool yes=false;
    	for(int i=0;i<l;i++)
    	{
    		m[s[i]] +=1;
    	}
    	for(auto it=m.begin();it!=m.end();it++)
    	{
    		if(it->second>=2)
    		{
    			yes=true;
    			break;
    		}
    	}
    	if(yes)
    	cout<<"Yes\n";
    	else
    	cout<<"No\n";
    	
    	
    }
    return 0;
}
