#include <bits/stdc++.h>
using namespace std;
int place(int a[],int n)
{
	int i=0;
	while(n>0)
	{	
		a[i]=n%10;
i++;
n/=10;
}
return i;
}
int search(int a[],int b[],int m,int n)
{int flag=-1;
	for(int i=0;i<m;i++)
	for(int j=0;j<n;j++)
	if(a[i]==b[j])
	return 1;
return flag;
}



int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int co=0;
    	int n;
    	cin>>n;
    	int a[9];
    	int i=place(a,n);
    	
    	for(int k=1;k*k<=n;k++)
    	{
    		if(n%k==0)
    		{
    			int m=n/k;
			int b[9],c[9];
    			if(m!=k)
    			{
    				int j=place(b,k);
    				if(search(a,b,i,j)==1)
    			co+=1;
    			}
			int o=place(c,m);
			
			if(search(a,c,i,o)==1)
			co+=1;
    		}
    	}
    	cout<<co<<endl;
    	
    }
    return 0;
}
