#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() 
{
    int n,i,j,cnt=1,s;
    cin>>n;
    int a[n];
    for(i=0;i<n;i++)
        cin>>a[i];
    sort(a,a+n);
    s=a[0];
    for(i=1;i<n;i++)
    {
        if(a[i]>s+4)
        {
            s=a[i];
            cnt++;
        }
    }
    cout<<cnt;
    return 0;
}
