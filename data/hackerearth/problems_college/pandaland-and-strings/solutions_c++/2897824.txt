
#include <stdio.h>

#define IOR(x) freopen(x,"r",stdin);

struct trie {
	trie *next[26];
	bool end;
	trie() {
		for(int i=0; i<26; i++) next[i] = NULL;
		end = 0;
	}
};

int main() {
	int t, n, i, p;
	char str[105];
	bool flag;
	//IOR("in.txt");
	scanf("%d", &t);
	while(t--) {
		scanf("%d", &n);
		trie *head, *tail;
		head = new trie;
		flag = 1;
		while(n--) {
			scanf("%s", str);
			if(flag) {
				tail = head;
				for(i=0; str[i]; i++) {
					if(tail->end) {
						flag = 0;
						break;
					}
					p = str[i]-'a';
					if(tail->next[p]==NULL) tail->next[p] = new trie;
					tail = tail->next[p];
				}
				tail->end = 1;
				for(i=0; i<26; i++) {
					if(tail->next[i]) {
						flag = 0;
						break;
					}
				}
			}
		}
		if(flag) printf("YES\n");
		else printf("NO\n");
	}
	return 0;
}
