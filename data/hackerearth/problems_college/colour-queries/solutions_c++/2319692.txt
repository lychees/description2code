#include <bits/stdc++.h>
using namespace std;
int a[104][104];
void dfs(int,int,int);
int count1,n,m;
int vis[104][104];
int main()
{
	int q,i,j,x,d,y,k;
	cin>>n>>m>>q;
	for(i=0;i<n;i++){
		for(j=0;j<m;j++){
		cin>>x;
		a[i][j]=x;
		}
	}
	for(k=0;k<q;k++){
		cin>>x>>y>>d;

		x--;
		y--;
    for(i=0;i<n;i++){
		for(j=0;j<m;j++)
                vis[i][j]=0;

    }
count1=0;

		dfs(x,y,d);
		cout<<count1<<endl;
	}
}
void dfs(int x,int y,int d){
	count1++;
	vis[x][y]=1;
	if(x+1<n&&vis[x+1][y]==0){
	if(abs(a[x+1][y]-a[x][y])<=d)
	dfs(x+1,y,d);
	}
	if(y+1<m&&vis[x][y+1]==0){
	if(abs(a[x][y+1]-a[x][y])<=d)
	dfs(x,y+1,d);
	}
	if(x-1>=0&&vis[x-1][y]==0){
	if(abs(a[x-1][y]-a[x][y])<=d)
	dfs(x-1,y,d);
	}
	if(y-1>=0&&vis[x][y-1]==0){
	if(abs(a[x][y-1]-a[x][y])<=d)
	dfs(x,y-1,d);
	}
}
