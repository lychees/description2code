#include <iostream>
using namespace std;

int main()
{
    int i,j,k,t;
    cin >> t;
    for(i=0;i<t;i++){
    	int n;
    	int maxindex=0;
    	cin >> n;
    	long long int x[n];
    	for(j=0;j<n;j++)    	cin >> x[j];
    	int dp[n];
    	dp[0]=x[0];
    	if(n>1)dp[1]=x[1];
    	if(n>2) dp[2] = x[2]+dp[0];
    if(n>2)	{if(dp[1]>dp[2]) maxindex = 1;
    	else maxindex = 2;}
    	for(j=3;j<n;j++){
    		long long int max=0;
    		if(j-maxindex>1){
    			dp[j] = x[j] + dp[maxindex];
    			maxindex = j;
    		}
    		else {
    		int	maxin1 = 0;
    			//dp[j] = x[j] + dp[j-2];
    			for(k=0;k<j-1;k++){
    				if(dp[k]>dp[maxin1])maxin1 = k;
    			}
    			dp[j] = x[j] + dp[maxin1];
    			if(dp[j]>dp[maxindex]) maxindex = j;
    		}
    	}
    	cout << "Case " << i+1 << ": " << dp[maxindex] << endl;
    }
}
