#include<iostream>
using namespace std;
main()
{ int t;
cin>>t;
for(int z=0;z<t;z++)
{
  int n,k,m;
  cin>>n>>k>>m;
  int p[100002];
  for(int i=0;i<n;i++)
    cin>>p[i];
    long long int ans=1000000000000;
     int index;
    for(int i=0;i<n;i++)
    {   long long int sum=0;
       for(int j=1;j<=p[i];j++)
       {
          long long int a;
          cin>>a;
          sum+=a;
       }
       sum=sum+(k*(p[i]))+(m*(p[i]-1));
       if(sum<ans)
      index=i+1, ans=sum;
     }
     cout<<index<<" "<<ans<<endl;
}
}
