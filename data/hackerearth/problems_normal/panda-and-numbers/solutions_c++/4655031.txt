#include <bits/stdc++.h>
using namespace std;
int dp[1000009];
int n1;
int x,f;


int recur(int num)
{
   if(num<0)
        return -1;
   if(dp[num]==1 || dp[num]==-1)
       return dp[num];
   if(dp[num]==2)
       return -1;
   n1 = num;
   f = -1;
   dp[num]=2;
   while(n1>0)
   {
    	x = n1%10;
    	
    	f = recur(num-x*x);
    	if(f==1)
    		break;
    	n1/=10;
   }

   dp[num] = f;
   return dp[num];
}
int main()
{
    //freopen("file1.txt","r",stdin);

	dp[0] = -1;
    dp[1] = 1;
    dp[27] = 1;
    dp[256] = 1;
    dp[4] = 1;
    dp[3125] = 1;
    dp[46656] = 1;
    dp[823543] = 1;
    int num;
    for(int i=1;i<=1000008;i++)
    {
    	if(dp[i]==0)
    		dp[i] = recur(i);
    }

    long long t,i;
    	scanf("%lld",&t);
    long long n;
    for(i=1;i<=t;i++)
    {


    	scanf("%lld",&n);
    	if(dp[n]==1)
    			printf("Yes\n");
    	else printf("No\n");
    }
}
