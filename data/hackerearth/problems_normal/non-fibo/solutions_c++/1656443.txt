#define sd(a) scanf("%d",&a)
#define pd(a) printf("%d\n",(a))
#define LL long long
#define F first
#define S second
#define MP make_pair
#define PB push_back
int main()
{
	int t;
	long long n,i,cur;
	long long fib[500],pos;
	fib[0]=1;
	fib[1]=2;
	for(i=2;fib[i-1]<=100000000000000000ll;++i)
	{
		fib[i]=fib[i-1]+fib[i-2];
	}
	sd(t);
	while(t--)
	{
		scanf("%lld",&n);
		for(i=0;fib[i]<=n;++i);
		pos=i;
		cur=n-i;
		i=n;
		while(cur<n)
		{
			i++;
			if(fib[pos]==i)
			{
				pos++;
			}
			else
				cur++;
		}
		printf("%lld\n",i);
	}
}
