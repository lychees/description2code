#include<cstdio>
#include<algorithm>
using namespace std;
inline void scan(int &a){
    char ch=a=0;
    while(ch<33)    ch=getc(stdin);
    while(ch>33)    a=a*10+ch-'0',ch=getc(stdin);
}
long long min(long long a,long long b){
    return a<b?a:b;
}
int main(){
    int t,n,k,a;
    long long sumTasks;
    long long dp[2010];
    pair<long long,long long> list[2010];
    scan(t);
    while(t--){
        scan(n),scan(k);
        for(int i=0;i<n;i++)
            scan(a),list[i].first=a;
        for(int i=0;i<n;i++)
            scan(a),list[i].second=a;
        sort(list,list+n);
        dp[0]=(list[0].second+k)*list[0].first;
        for(int i=1;i<n;i++){
            dp[i]=dp[i-1]+(list[i].second+k)*list[i].first;
            sumTasks=list[i].second;
            for(int j=i-1;j>=0;j--){
                sumTasks+=list[j].second;
                dp[i]=min(dp[i],(j?dp[j-1]:0)+(sumTasks+k)*list[i].first);
            }
        }
        printf("%lld\n",dp[n-1]);
    }
    return 0;
}
