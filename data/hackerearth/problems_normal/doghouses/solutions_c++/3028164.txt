void sll(ll &x)
{
    register char c = gc();
    x = 0;
    int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}

void wll(ll a)
{
        if(a<0)
        {
                pc('-');
                a=-a;
        }

        char snum[100];
        int i=0;
        do
        {
                snum[i++]=a%10+48;
                a=a/10;
        }
        while(a!=0);
        --i;
        while(i>=0)
        putchar_unlocked(snum[i--]);
        putchar_unlocked('\n');
}

ll power(ll a, ll b, ll mod)
{
        ll ret = 1 ;
        while(b)
        {
                if(b & 1 ) ret = ret*a % mod;
                a = a*a % mod;
                b >>= 1 ;
        }
        return ret;
}

ll gcd(ll a, ll b)
{
    while(b) b ^= a ^= b ^= a %= b;
    return a;
}


int main()
{
	ll t, i, j, k, l;
	ll n;
	sll(n);
	ll a[1010], f[1010]={0}, mn[1010], mx[1010]={0};
	vll v;
	repe(i,1,n){		
		sll(a[i]);	
		if(!f[a[i]]){
			mn[a[i]]=i;
			v.pb(a[i]);		
		}
		mx[a[i]]=i;
		f[a[i]]++;
	}
	ll res=0;
	sort(all(v));
	repl(i,0,v.sz){
		ll mnx=mn[v[i]],mxx=mx[v[i]];	
		repl(j,i+1,v.sz){
			mnx=min(mnx,mn[v[j]]);
			mxx=max(mxx,mx[v[j]]);
			ll ans=f[v[i]]+f[v[j]];
			if(mnx != mn[v[i]] && mnx != mn[v[j]])
				ans++;
			if(mxx != mx[v[i]] && mxx != mx[v[j]])
				ans++;
			res=max(res,ans);
		}
	}	
	if(v.sz == 1)
		wll(f[v[0]]);
	else 
		wll(res);
	return 0;
}
