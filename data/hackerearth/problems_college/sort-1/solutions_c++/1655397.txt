#include <iostream>
#include <algorithm>
using namespace std;
int main(){
	ios::sync_with_stdio(false);
	int T=0;
	
	cin >> T;
	
	while(T--)
	{
	 int N=0;
	 cin >> N;
	 long arr[N];
	 
	 for(long i=0;i<N;i++)
	  arr[i] = 0;
	 
	 for(long i=0;i<N;i++)
	  cin >> arr[i];
	 
	 sort(arr,arr+N);
	 
	 for(long i=0;i<N;i++)
	  cout << arr[i] << " ";
	 
	 cout << endl;
	}
	
	return 0;
}
