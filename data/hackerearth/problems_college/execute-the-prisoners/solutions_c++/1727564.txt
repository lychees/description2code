#include <cstdio>
#include <cstring>
#include <algorithm>>
using namespace std;

#define ll long long

int t, n, m;

int main(){
    scanf("%d", &t);
    while (t--){
        scanf("%d", &n);
        m = 2*n;
        while (m > n)
            m += m - 2*n - 1;
        printf("%d\n", m);
    }
    return 0;
}
