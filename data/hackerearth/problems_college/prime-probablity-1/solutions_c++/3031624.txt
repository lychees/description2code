#include<iostream>
#include<math.h>
#include<stdio.h>
using namespace std;
bool isPrime(int x)
{
	if(x==2)
		return true;
	if(x%2==0 || x==1 || x==0)
		return false;
	for(int i=3;i<=sqrt(x);i+=2)
		if(x%i==0)
			return false;
	return true;
}
int main()
{
	int i,t,a[1000001],primecount=0,l,r;
	for(i=0;i<1000001;i++)
	{
		if(isPrime(i))
			primecount++;
		a[i]=primecount;
	}
	cin>>t;
	while(t--)
	{
		cin>>l>>r;
		if(isPrime(l))
			printf("%.6f\n",(float)(a[r]-a[l]+1)/(r-l+1));
		else
			printf("%.6f\n",(float)(a[r]-a[l])/(r-l+1));
	}
}
