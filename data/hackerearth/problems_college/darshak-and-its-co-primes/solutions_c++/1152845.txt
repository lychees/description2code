using namespace std;

long long int gcd(long long int a, long long int b)
{
	while (b) b ^= a ^= b ^= a %= b;
	return a;
}

int main()
{
	int t;
	cin>>t;
	while (t--)
	{
		long long int p;
		long long int q;

		cin >> p >> q;

		(gcd(p, q) == 1) ? cout << "Is a Co-Prime" << endl : cout << "Not a Co-Prime" << endl;
	}

	return 0;
}
