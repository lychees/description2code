#include<bits/stdc++.h>
using namespace std;
int main()
{	ios_base::sync_with_stdio(0);
	int t;
	cin>>t;
	while(t--)
	{
		int d,wt; long int s=0;
		cin>>d>>wt;
		if(d>60)
		printf("NA\n");
		else{
			if(d>2)
			{
				s=(d-2)*12 + wt*2 +50;
			}
			else if(d<=2)
			{
				s=50 + wt*2;
			}
			printf("%ld\n",s);
		}
	}
}
