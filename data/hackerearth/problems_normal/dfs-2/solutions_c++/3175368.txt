#include <bits/stdc++.h>
using namespace std;
int main()
{
	int N,num;
	cin>>N;
	vector<int> V;
	vector<int>::iterator low,up;
	for(int i=0;i<N;i++){
		cin>>num;
		V.push_back(num);
	}
	sort(V.begin(), V.end());
	int Query;
	cin>>Query;
	while(Query--)
	{
		int L,R;
		cin>>L>>R;
		low=lower_bound (V.begin(), V.end(), L); 
  		up= upper_bound (V.begin(), V.end(), R);
  		cout<<((up-V.begin())-(low-V.begin()))<<endl; 
	}
	return 0;
}
