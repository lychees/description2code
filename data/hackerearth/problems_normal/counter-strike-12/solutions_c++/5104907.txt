#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int N,M,D;
    	cin>>N>>M>>D;
    	int locationX[N];
    	int locationY[N];
    	int targetX[M];
    	int targetY[M];
    	int count=0;
    	for(int i=0; i<N; i++)
    	{
    		cin>>locationX[i]>>locationY[i];
    	}
    	for(int i=0; i<M; i++)
    	{
    		cin>>targetX[i]>>targetY[i];
    	}
    	for(int i=0; i<M; i++)
    	{
    		for(int j=0; j<N; j++)
    		{
    			if(abs(targetX[i]-locationX[j])+ abs(targetY[i]-locationY[j]) <= D)
    			{
    				count++;
    				break;
    			}
    		}
    	}
    	if(count >=M/2)
    	{
    		cout<<"YES"<<endl;
    	}
    	else
    	{
    		cout<<"NO"<<endl;
    	}
    }
}
