typedef long long ll;

ll arr [1000000 + 10];


int main()
{
    //ios::sync_with_stdio(false);cin.tie(0);

    ll n , m  , k , d , t , tem1 , tem2 , tem3 , tem4 , y = 1, sum = 0 , ans = 0;
    string s , c;


    cin >>         t       ;

    while(t--){

        cin >> n >> m;
        bool found = false;
        sum = 0;

        for(int i = 1; i <= n; i++){
            cin >> arr[i];
        }
        sum = arr[1];
        for(int i = 1, j = i + 1; i <= n && !found; i++){
            sum -= arr[i - 1];


            if(sum == m){
                found = true;
                break;
            }
            for(j; j <= n && sum < m; j++){
                sum += arr[j];
                if(sum == m){
                    found = true;
                    break;
                }
            }
        }
        if(found) cout << "YES\n";
        else cout << "NO\n";

    }




    return 0;
}
