def showfile(filename):
    print filename
    l, w, c = 0, 0, 0
 
    with open(filename, 'r') as f:
        for line in f:
            words = line.split()
 
            l += 1
            w += len(words)
            c += len(line)
 
            st = os.stat(filename)
 
    return l, w, c, st[4], st[5], st[8]
