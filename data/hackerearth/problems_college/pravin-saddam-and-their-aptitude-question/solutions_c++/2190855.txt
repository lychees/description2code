#include <iostream>
using namespace std;
long int fib(long int n)
{
	if(n==0)
	return 1;
	if(n==1)
	return 1;
	else
	return n*fib(n-1);
}

int main()
{
    long int  a,b,t,i,j;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b;
    	j=0;
    	for(i=a;i<=b;i++)
    	j+=fib(i);
    	
    	cout<<j%10<<endl;
    }
    
    return 0;
}
