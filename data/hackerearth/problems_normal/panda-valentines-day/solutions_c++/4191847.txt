#include <iostream>
using namespace std;

long long dp[1000001] = {0};

long long ans(int n){
	if(n == 1){
		return dp[1];
	}
	if(dp[n]){
		return dp[n];
	}
	int p = 0,ncopy = n;
	for(int i = 2; i*i <= n; i++){
		while(n%i == 0){
			n /= i;
			p++;
		}
	}
	if(n>1){
		p++;
	}
	dp[ncopy] = ans(ncopy-1) + p;
	return dp[ncopy];
}

int main()
{
	dp[1] = 0;
	dp[2] = 1;
    int t,n;
    cin >> t;
    while(t--){
    	cin >> n;
    	cout << ans(n) << endl;
    }
    return 0;
}
