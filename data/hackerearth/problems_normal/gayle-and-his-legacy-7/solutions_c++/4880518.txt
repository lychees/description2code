#include <iostream>
#include<algorithm>
using namespace std;
#define ull unsigned long long int

int main()
{
    ull n, m;
    int t;
    cin>>t;
    while(t--){
    	cin>>n>>m;
    	vector<int> g(n),p(n);
    	vector<pair<double, int> > sack(n);
    	for(int i=0;i<n;i++) cin>>g[i]>>p[i];
    	for(int i=0;i<n;i++) sack[i] = {p[i], i};
    	sort(sack.begin(), sack.end());
    	bool cando = true;
    	for(int i=0;i<n;i++){
    		if(m <= p[sack[i].second]){
    			cando = false;
    			break;
    		}
    		m += g[sack[i].second];
    	}
    	cout<<(cando ? "YES" : "NO")<<"\n";
    }
    return 0;
}
