#include<iostream>
#include<math.h>
using namespace std;
int main()
{
    long t,n,x,y,i,flag;
    cin>>t;
    while(t--)
    {
        cin>>n;
        flag=0;
        x=n+1;
        while(1)
        {
            for(i=2;i<=sqrt(x);i++)
            {
                if(x%i==0)
                {
                    flag=1;
                    break;
                }
            }
            if(flag==1)
            {
                cout<<x<<endl;
                break;
            }
            x++;
        }
    }
    return 0;
}
