#include<bits/stdc++.h>
using namespace std;
const int MAXN = 1e6+3;
int ar[MAXN];
int main() {
	int n;
	cin>>n;
	memset(ar,0,sizeof(ar));
	for(int i=1;i<=n;i++) {
		int x,y;
		scanf("%d%d",&x,&y);
		ar[x]=y;
	}
	int last=-1;
	for(int i=MAXN-1;i>=1;i--) {
		last=max(last,ar[i]);
		ar[i]=last;
	}
	int g;
	cin>>g;
	long long ans=0;
	for(int i=1;i<=g;i++) {
		int m;
		scanf("%d",&m);
		ans+=1LL*ar[m]/m*100;
	}
	cout<<ans<<endl;
	return 0;
}
