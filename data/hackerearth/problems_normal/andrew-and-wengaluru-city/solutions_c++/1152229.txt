#include <iostream>
#include <vector>
#define MOD 1000000007
#define ll long long
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t,n;
    scanf("%d",&t);
    while(t--){
    	scanf("%d",&n);
    	vector<int>v(n);
    	for(int i=0;i<n;i++){
    		cin>>v[i];
    	}
    	int l,r;
    	ll result = 0;
    	for(l=0,r=1;r<n;r++){
    		if(v[l]<0){
    			l++;
    			continue;
    		}
    		if(v[l]<=v[r]){
    			for(int j=l+1;j<r;j++){
    				result = (result + v[l]-v[j])%MOD;
    			}
    			l=r;
    		}
    	}
    	for (int l=n-2, r=n-1; l>=0; l--) {
            if (v[r]<0) {
            	r--;
                continue;
            }
            if (v[l] >= v[r]) {
                for (int i = r - 1; i > l; i--) {
                    result = (result + v[r] - v[i])%MOD;
                    v[i] = v[r];
                }
                r = l;
            }
        }
        result = result % MOD;
        cout<<result<<endl;
    }
    return 0;
}
