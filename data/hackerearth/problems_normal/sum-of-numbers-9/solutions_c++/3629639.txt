#include <iostream>
#include <string>

using namespace std;

int main()
{
	int T, N, S, A[16];

	cin >> T;

	while (T--)
	{
		cin >> N;

		for (int i = 0; i < N; i++)
		{
			cin >> A[i];
		}
		
		cin >> S;
		
		bool found = false;
		int sum = 0;

		for (int i = 0; i < (1 << N) && !found; i++)
		{
			sum = 0;
			
			for (int j = 0; j < N; j++)
			{
				if (i & (1 << j))
				{
					sum += A[j];
				}
				if (sum == S)
				{
					found = true;
					break;
				}
			}
		}

		if (found)
			cout << "YES";
		else
			cout << "NO";

		cout << endl;
	}

	return 0;
}
