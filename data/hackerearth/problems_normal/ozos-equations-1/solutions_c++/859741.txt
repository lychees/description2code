#include<bits/stdc++.h>
#define mod 1000000007
#define MAX 100000000
using namespace std;
#define ll long long
int s(ll n)
{
	int sum=0;
	while(n!=0)
	{
		sum+=n%10;
		n/=10;
	}
	return sum;
}
int main()
{

	int t;
	scanf("%d",&t);
	while(t--)
	{
		ll l,r,n,tmp,ans;
		int flag=0;
		scanf("%lld",&n);
		l=(ll)sqrt(n)-81;
		r=(ll)sqrt(n)+1;
		for(ll i=l;i<=r;i++)
		{
			if(i>=0)
			{
				if(i*(i+s(i))==n)
				{
					ans=i;
					flag=1;
					break;
				}
			}
		}
		if(flag==1)
			printf("%lld\n",ans);
		else
			printf("-1\n");
	}
	return 0;
}
