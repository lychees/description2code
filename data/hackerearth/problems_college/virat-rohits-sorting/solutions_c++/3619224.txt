#include <iostream>
#include <algorithm> 
#include <vector>
using namespace std;

bool myfunction(pair<long int,long int> v1,pair<long int,long int> v2)
{
	if(v1.first==v2.first)
		return v1.second>v2.second;
	else
		return v1.first<v2.first;
}
int main()
{
   int N;
   long int code,runs;
   cin>>N;
   vector<pair<long int,long int>> v;
   while(N--)
   {
   	cin>>code>>runs;
   	v.push_back(pair<long int,long int>(code,runs));
   }
   sort(v.begin(),v.end(),myfunction);
   vector<pair<long int,long int>>::iterator it;
   for(it=v.begin();it!=v.end();it++)
   {
   	 cout<<it->first<<" "<<it->second<<"\n";	
   }
    return 0;
}
