#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	int n;
	scanf("%d",&n);
	int a[n],b[n];
	int dp[n+1][n+1];
	int i,j;
	for(i=0;i<n;i++)
	{
		scanf("%d",(a+i));
	}
	sort(a,a+n);
	for(i=0;i<n;i++)
	{
		scanf("%d",(b+i));
	}
	sort(b,b+n);
	for(i=0;i<=n;i++)
	{
		for(j=0;j<=n;j++)
		{
			if(i==0||j==0)
			{
				dp[i][j]=0;
			}
			else if(a[i-1]>b[j-1])
			{
				dp[i][j]=100+dp[i-1][j-1];
			}
			else if(a[i-1]==b[j-1])
			{
				dp[i][j]=max(max(50+dp[i-1][j-1],dp[i][j-1]),dp[i-1][j]);
			}
			else
			{
				dp[i][j]=max(dp[i][j-1],dp[i-1][j]);
			}
		}
	}
	printf("%d\n",dp[n][n]);
	return 0;
}
