string gen(string s[],int n)
{
    string res = s[0];
    for(int i=1;i<n;i++)
    {
        string temp = res;
        reverse(temp.begin(),temp.end());
        res = res + temp + s[i];
    }
    return res;
}


main()
{
    int n;
    cin>>n;
    while(n--)
    {
        int m,n;

        string s[m+1];
        for(int i=1;i<=m;i++)
            cin>>s[i];
        int len[m+1];
        len[0]=0;
        for(int i=1;i<=m;i++)
            len[i]=len[i-1]*2+s[i].size();

        for(int i=0;i<n;i++)
        {
            int x;
            cin>>x;
            for(int j=m;j;--j)
            {
                if(x>=len[j-1]*2)
                {
                    cout<<s[j][x-2*len[j-1]];
                    break;
                }else if(x>=len[j-1])
                {
                    x = 2*len[j-1]-x-1;
                }
            }
