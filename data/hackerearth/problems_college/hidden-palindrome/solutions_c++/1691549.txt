string s;
int dp[100][100];
void Scan()
{
	cin>>s;
}
void Out()
{
	int n=sz(s);
	int res=1;
	loope(i,0,n) dp[i][i]=1;
	loop(i,0,n-1) if(s[i]==s[i+1]) {dp[i][i+1]=1; res=2;}
	  for (int k = 3; k <= n; ++k)
    {
        
        for (int i = 0; i < n-k+1 ; ++i)
        {
            
            int j = i + k - 1;
 
            if (dp[i+1][j-1] && s[i] == s[j])
            {
                dp[i][j] = 1;
 					res=max(res,k);
            }
        }
    }
    cout<<res<<endl;
	return;
}

	{
	Scan();
	Out();
