#include <iostream>
using namespace std;
long long C(int n, int r) {
    if(r > n / 2) r = n - r; // because C(n, r) == C(n, n - r)
    long long ans = 1;
    int i;
 
    for(i = 1; i <= r; i++) {
        ans *= n - r + i;
        ans /= i;
    }
 
    return ans;
}
int factorial(int number) {
	int temp;
 
	if(number <= 1) return 1;
 
	temp = number * factorial(number - 1);
	return temp;
}
int main()
{
    int T,n1,n2;
    cin >> T;
    while(T--){
    	cin >> n1 >> n2;
    	cout << C(7,n1)*C(5,n2)*factorial(n1+n2)<<endl;
    }
    return 0;
}
