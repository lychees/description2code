#include <vector>
#include <iostream>
 
using namespace std;
 
int find_kth(long k) {
	long cnt = 0, last=0, d = 1, n = 2;
	while(cnt<k) {
		last = cnt;
		cnt += d*n;
		d+=1;
		n*=2;
	}
	
	d-=1;
	n/=2;
 
	k = k-last-1;
	long i, j;
	i = k%d;
	j = k/d;
	while(true) {
		if(i==0) {
			if(j<n/2) return 4;
			else return 5;
		}
		else {
			if(j<n/2) ;
			else j-=n/2;
			i-=1;
			d-=1;
			n/=2;
		}
	}
	return 0;
}
 
int main() {
	int N;
	cin >> N;
	for(int i=0; i!=N; ++i) {
		long tmp;
		cin>>tmp;
		cout << (find_kth(tmp)==4?"Hacker":"Earth") << endl;
	}
}
