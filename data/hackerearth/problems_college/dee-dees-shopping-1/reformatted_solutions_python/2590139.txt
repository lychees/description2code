def getCount(n):
    c = 0
    x = 0
    n1 = n
    while (n > 0):
        if n % 8 == 0:
            c += 1
            n -= 8
        else:
            x += 1
            n -= 6
    if (6*x + 8*c) == n1:
        return x+c
    else:
        return -1


if __name__ == '__main__':
    T = int(raw_input())
    for i in range(0, T):
        n = int(raw_input())
        print getCount(n)
