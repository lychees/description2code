#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,d,r;
        cin>>n>>d>>r;
        int a=n;int ans;
        for(int i=0;i<r;i++)
        {
            a*=10;
            if(i==r-1){ans=a/d;break;}
            if(a%d==0)
            {
                ans=0;
                break;
            }

            a%=d;
        }
        cout<<ans<<endl;
    }
    }
