/************************************************************************************/

long long dp[100100];

int main()
{
   int t;
   cin >> t;
   dp[0] = 0;
   dp[1] = 1;
   for (long long i = 2; i < 100100; i++) {
   	    dp[i] = dp[i - 2];
   	    dp[i] += 8 * (i - 2) * (i - 2);
   	    dp[i] += (4 * (i - 1)) * ((2 * (i - 1)) + i);
   }
   while (t--) {
   	      int n;
   	      cin >> n;
   	      double N = 1.0 * n;
   	      double ans = (double)dp[n];
   	      ans /= (N * N);
   	      printf("%.6lf\n", ans);
   }
   return 0;
}
