#include <iostream>
#include<vector>
#include<map>
using namespace std;

int main()
{
    int n, req;
    vector<int> vect;
    scanf("%d", &n);
    for(int i=0;i<n;i++)
    {
    		int temp;
    		scanf("%d", &temp);
    		vect.push_back(temp);
    }
    scanf("%d", &req);
    map<int, int> sum;
    int flag=0;
    for(int i=0;i<vect.size();i++)
    {
    	for(int j=i+1;j<vect.size();j++)
    		sum[vect[i]+vect[j]]=1;
    }
    for(int i=0;i<vect.size();i++)
	{
		if(sum.find(req-vect[i])!=sum.end())
		{
			flag=1;
			break;
		}
    
    }

	if(flag==1)
		printf("True");
	else
		printf("False");
    return 0;
}
