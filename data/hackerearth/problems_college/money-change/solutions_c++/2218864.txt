#include <iostream>
using namespace std;

long long int fun(int *arr,int n,int amt,long long int dp[1000][1000])
{
	
	if(dp[n][amt]!=0)return dp[n][amt];
	if(amt==0)
	return 1;
	if(amt<0)return 0;
	
	if(amt>=1 && n<=0)return 0;
	dp[n][amt]=fun(arr,n-1,amt,dp)+fun(arr,n,amt-arr[n-1],dp);
	return dp[n][amt];
}

int main()
{
   int amt,n;
   cin>>amt>>n;
   int *arr=new int[n];
  
   for(int i=0;i<n;i++)
   cin>>arr[i];
  long long int dp[1000][1000];
	for(int i=0;i<1000;i++)
	{
		for(int j=0;j<1000;j++)
		{
			dp[i][j]=0;
		}
	}
   cout<<fun(arr,n, amt,dp);
    
    return 0;
}
