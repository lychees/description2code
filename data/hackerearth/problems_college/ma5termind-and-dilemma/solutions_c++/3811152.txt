#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	
	while(t--)
	{
		int n;
		cin>>n;
		string str;
		cin>>str;
		
		int c0=0;
		int c1=0;
		for(int i=0;i<str.length();i++)
		{
			if(str[i]=='0')
			c0++;
			else if(str[i]=='1')
			c1++;
		}
		int sw0=0;
		int sw1=0;
		
		for(int i=0;i<c0;i++)
		{
			if(str[i]!='0')
			sw0++;
		}
		for(int i=0;i<c1;i++)
		{
			if(str[i]!='1')
			sw1++;
		}
		
		cout<<min(sw0,sw1)<<endl;
		
	}
	
}
