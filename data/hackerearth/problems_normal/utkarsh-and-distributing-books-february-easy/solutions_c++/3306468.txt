#include<iostream>
using namespace std; 
int main()
{
 	int t;
 	cin>>t;
 	while(t--)
 			  {
	 		   int pile;
	 		   cin>>pile;
	 		   int r[1000];
	 		   
	 		   int min=10000,sum=0;
			   for (int i=0;i<pile;i++)
			   	   {
  					   cin>>r[i];
				   	   if(min>r[i])
						   min=r[i];			 	   
			       }
		       for (int i=0;i<pile;i++)
		       	   		sum+=r[i];
   			
		       cout<<min-1<<" "<<sum-pile<<"\n";
			       
	           }
      
      system("pause");
      return 0;
}
