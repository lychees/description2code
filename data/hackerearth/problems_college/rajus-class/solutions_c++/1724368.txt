#include <iostream>
#include<algorithm>
#include<string.h>
using namespace std;

struct stu
 {
 	int age;
 	char name[100];
 };
 
 bool comp(const stu &x,const stu &y)
  {
  	if(x.age<y.age)return 1;
  	else if(x.age==y.age){
  	   if(strcmp(x.name,y.name)<0)
  	    return 0;
  	   else return 1;
  	} 
  	else return 0;
  }

int main()
{
    int t;
    cin>>t;
    while(t--)
     {
     	int n;
     	cin>>n;
     	stu student[n+2];
     	for(int i=1;i<=n;i++)
     	 {
     	 	cin>>student[i].age>>student[i].name;
     	 }
     	sort(student+1,student+n+1,comp);
     	int r;
     	cin>>r;
     	cout<<student[r].name<<"\n";
     }
}
