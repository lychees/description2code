#include<bits/stdc++.h>
using namespace std;
int a[1000000];
int main(){
int n,k,i,low,high,h,flag=0;
cin>>n>>k;
for(i=0;i<n;i++){
cin>>a[i];
}

sort(a,a+n);
low=0;high=n-1;
while(low<high){
if(a[low]+a[high]==k){
flag=1;
break;
}
if(a[low]+a[high]>k){
high--;

}
else{
low++;
}


}

if(flag==1){
cout<<"YES";
}
else{
cout<<"NO";
}
return 0;
}
