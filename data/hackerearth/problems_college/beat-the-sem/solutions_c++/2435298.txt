#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    
    while(n--)
    {
    	
    	  int a,b;
    	  cin>>a>>b;
    	  
    	  int x,y;
    	  
    	  x= (a+b)/2;
    	  y=(a-b)/2;
    	  
    	  if(a<b||(a+b)%2==1||(a-b)%2==1)
    	  cout<<"impossible"<<endl;
    	  else
    	  cout<<x<<" "<<y<<endl;
    	 
    }
    return 0;
}
