#include<bits/stdc++.h>
using namespace std;

int main()
{
	int n,count,count1,c,d;
	count=0,count1=0,c=0,d=0;
	cin>>n;
	while(n--)
	{
		string str;
		cin>>str;
		c=0;
 
	for(int i=0;i<str.length();i++)
	{
		if(str[i]!='C')
		{
			if(c>=count) count=c;
			c=0;
			count1=max(count1,d);
			d=0;
		}
		else
		{
			c++;
			d++;
		}
	}
	count=max(c,count);
  }
  count1=max(d,count1);	
  cout<<count<<" "<<count1;
  
}
