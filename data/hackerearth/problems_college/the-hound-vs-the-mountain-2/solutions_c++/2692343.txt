    #include <iostream>
    #include <cmath>
    #include <string>
    #include <algorithm>
    #define lint long long
    using namespace std;
    lint con(string s){
    	lint n=0;
    	for(int i=0; i<s.size(); i++){
    		n=n*2;
    		if(s[i]=='1')n++;
    	}
    	return n;
    }
    int main(){
    	int t;
    	cin>>t;
    	while(t--){
    		string s,t;
    		cin>>s>>t;
    		lint x = con(s);
    		lint y = con(t);
    		if(x>2 && y>2){
    			if(x<y)cout<<"The Mountain\n";
    			else cout<<"The Hound\n";
    		}
    		else{
    			lint a = pow(x,y);
    		lint b = pow(y,x);
    		//cout<<x<<" "<<y<<" "<<a<<" "<<b<<endl;
    		if(a>=b)cout<<"The Hound\n";
    		else cout<<"The Mountain\n";}
    	}
    }
