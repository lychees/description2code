#include <iostream>
using namespace std;
string Rev(string name){
	char ans[name.length()+1];
	for(int i=0;i<name.length();i++){
		ans[name.length()-1-i] = name[i];
	}
	ans[name.length()] = '\0';
	return ans;
	}
int main()
{
    int t;
    	cin >> t;
    string names[t];
    	for(int i=0;i<t;i++){
    		cin >> names[i];
    	}
    for(int i=0; i< t;  i++){
    	string rev = Rev(names[i]);
    		//cout << rev << endl;
 		for(int j=0;j< t;j++){
 			if(rev==names[j]){
 				cout << rev.length() << " " << rev[rev.length()/2] << endl;
 				exit(0);
 			}
 		}
    }
    
    return 0;
}
