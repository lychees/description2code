#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
    int t;
    cin>>t;
    for(;t--;)
    {
    	string s;
    	cin>>s;
    	int i,count=0;
    	for(i=0;i<s.length();i++)
    	{
    		if(s[i]=='X' && s[i+1]=='X')
    		count++;
    		else if(s[i]=='Y' && s[i+1]=='Y')
    		count++;
    	}
    	cout<<(count)<<endl;
    }
    return 0;
}
