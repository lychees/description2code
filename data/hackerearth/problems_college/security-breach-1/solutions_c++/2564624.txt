#include<bits/stdc++.h>
using namespace std;
int main() {
	string str;
	string two="cde";
	string seven="acf";
	string six="b3";
	string nine="c6a";
	int t,i;
	scanf("%d",&t);
	while(t--) {
		cin>>str;
		string ans="";
		for(i=0;i<str.size();i++) {
			if(str[i]-48==2)
				ans+=two;
			else if(str[i]-48==7)
				ans+=seven;
			else if(str[i]-48==6)
				ans+=six;
			else if(str[i]-48==9)
				ans+=nine;
		}
		cout<<ans<<"\n";
	}
	return 0;
}
