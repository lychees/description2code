#include <iostream>
using namespace std;
unsigned long int count_bits(unsigned long int n)
{
	unsigned long int count=0;
	while(n)
	{
		count++;
		n&=n-1;
	
	}
	return count;
}
int main()
{
	unsigned long int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		cout<<count_bits(n)<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
