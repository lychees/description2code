#include<bits/stdc++.h>
#include<algorithm>
int a[1000000];
int main() {
	long long int t,n,i,j,k;
	scanf("%lld",&t);
	while(t--) {
		k=0;
		scanf("%lld",&n);
		for(i=1;i<=n;i++) scanf("%lld",&a[i]);
		std::sort(a+1,a+n+1);
		j=n;
		for(i=n;i>1;i--) {
		    if((a[j]-a[i-1]>0)){
		    	k++;
		    	j--;
		    }
		}
		if(k==(n-1))puts("1");
		else if(k==0)printf("%lld\n",n);
		else printf("%lld\n",n-k);
	}
	return 0;
}
