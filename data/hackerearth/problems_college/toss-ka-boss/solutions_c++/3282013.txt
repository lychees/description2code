#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	string s;
    	long long int ans=0;
    	cin>>s;
    	int count=0;
    	for(int i=0;i<s.length();i++){
    		if (s[i]=='H') count++;
    		else count=0;
    		ans+=count;
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
