#include <iostream>
#include <cmath>
using namespace std;

long double calArea(int x1,int y1,int x2,int y2,int x3,int y3){
	long double area = (x1*(y2-y3)+x2*(y3-y1)+x3*(y1-y2))/2.0;
	return abs(area);	
}

int check(int x1,int y1,int x2,int y2,int x3,int y3,int x4,int y4){
	long double a1 = calArea(x1,y1,x2,y2,x3,y3);
	long double a2 = calArea(x1,y1,x2,y2,x4,y4);
	long double a3 = calArea(x1,y1,x4,y4,x3,y3);
	long double a4 = calArea(x4,y4,x2,y2,x3,y3);
	
	if(a1 == a2+a3+a4){
		return 1;
	}else{
		return 0;
	}
}

int main()
{
    int t,x1,y1,x2,y2,x3,y3,x4,y4;
    cin>>t;
    while(t--){
    	cin>>x1>>y1>>x2>>y2>>x3>>y3>>x4>>y4;
    	if(check(x1,y1,x2,y2,x3,y3,x4,y4)){
    		cout<<"INSIDE"<<endl;
    	}else{
    		cout<<"OUTSIDE"<<endl;
    	}
    }
    return 0;
}
