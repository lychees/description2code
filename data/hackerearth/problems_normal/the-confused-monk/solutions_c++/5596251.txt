#include <bits/stdc++.h>
using namespace std;


int main(){
	int N;
	cin >> N;
	long long f = 1;
	long long g = 0;
	long long m = 1e9 + 7;
	for(int i = 1;i <= N;++i){
		long long A;
		cin >> A;
		f = (f * A) % m;
		g = __gcd(A, g);
	}
	long long ans = 1;
	while(g > 0){
		if(g & 1){
			ans = (ans * f) % m;
		}
		f = (f * f) % m;
		g >>= 1;
	}
	cout << ans << "\n";
	return 0;
}
