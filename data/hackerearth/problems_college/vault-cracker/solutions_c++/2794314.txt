#include<bits/stdc++.h>
using namespace std;
#define ll long long int
ll ans=0;
void solve(string w, string s, int pos, int st)
{
    if(pos==w.length())
    {
        ans++;
        return ;
    }
    int i,j,k;
    for(i=st;i<s.length();i++)
    {
        if(s[i]==w[pos])
        {
            solve(w,s,pos+1,i+1);
        }
    }

}
int main(){
ios::sync_with_stdio(false);

ll i,j,k,n;
string w,s;
cin>>w;
cin>>s;
solve(w,s,0,0);
cout<<ans;



return 0;
}
