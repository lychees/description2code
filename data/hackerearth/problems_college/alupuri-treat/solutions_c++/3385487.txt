#include<bits/stdc++.h>
using namespace std;

int main() {
  int n;
  cin>>n;
  int mat[n][n];
  for(int i=0;i<n;i++) {
    for(int j=0;j<n;j++) {
      cin>>mat[i][j];
    }
  }
  long long ans1=0;
  for(int i=0,j=0;i<n&&j<n;i++,j++) {
    ans1+=mat[i][j];
  }
  long long ans2=0;
  for(int i=0,j=n-1;i<n&&j>=0;i++,j--) {
    ans2+=mat[i][j];
  }
  cout<<llabs(ans1-ans2)<<endl;
  return 0;
}
