#include<iostream>
#include<cstdio>
#include<vector>

using namespace std;

int main()
{
    int test,i;
    scanf("%d",&test);
    while(test)
    {
        int N,out=0;
        scanf("%d",&N);
        vector<long int> vect(N);
        for(i=0;i<N;i++)
        {
            scanf("%ld",&vect[i]);
            vect[i]%=5;
        }
        for(i=0;i<N;i++)
        {
            out^=vect[i];
        }
        if(out)
        {
            printf("Aniruddha\n");
        }
        else
        {
            printf("Andrew\n");
        }


        test--;
    }

    return 0;
}
