#include<bits/stdc++.h>
#define MAX 100005
#define MOD 1000000007

using namespace std;

typedef long long int lli;

lli dp[MAX][2][2],last;
lli solve(lli idx,lli pre,lli ok){
	
	if(idx==last)
	 return ok;
    
	if(dp[idx][pre][ok]!=-1) return dp[idx][pre][ok];
	
	lli count=0;
	
	if(pre==0&&!ok)
	  count=solve(idx+1,0,1)+solve(idx+1,1,0);
	  
	else if(pre==0&&ok)
	  count=solve(idx+1,1,ok);  
	  
	else if(pre==1) count=solve(idx+1,1,ok)+solve(idx+1,0,ok); 
	
	return dp[idx][pre][ok]=count%MOD;
	
}

int main(void){
	
	lli testCases;
	
	cin>>testCases;
	
	while(testCases--){
		
		cin>>last;
		memset(dp,-1,sizeof dp);
		cout<<(solve(1,0,0)+solve(1,1,0))%MOD<<endl;
			
	}
	
	return 0;
	
}
