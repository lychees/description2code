#include <iostream>
#include <string>
using namespace std;

int main()
{
    int t;
    cin >> t;
    string a;
    for(int i=0; i<t; i++)
    {
    	cin >> a;
    	bool ans = 1;
    	int l = a.length();
    	bool c[26] = {0};
    	for(int j=0; j<l; j++)
    	{
    		c[a[j]-97] = 1;
    	}
    	for(int j=0; j<26; j++)
    	{
    		if(c[j] == 0)
    		{
    			ans = 0;
    			break;
    		}
    	}
    	if(ans == 1)
    		cout << "YES" << endl;
    	else
    		cout << "NO" << endl;
    }
    return 0;
}
