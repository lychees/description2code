#include <iostream>
using namespace std;
int findAnswer(int arr[], int test){
	int count = 0;
for(int i=0; i< test-1; i++)
	for(int j=i+1; j< test; j++){
		if(arr[i]== arr[j])
			count++;
	}
	
	return test+count;

}

int main()
{
    //cout << "Hello World!" << endl;
    int n, test, result;
    cin >> n ;
    while(n--){
    	cin>> test;
    	int arr[test];
    	for(int i=0; i< test; i++)
    		cin >> arr[i];
    	result = findAnswer(arr, test);
    	cout << result <<"\n";
    }
    return 0;
}
