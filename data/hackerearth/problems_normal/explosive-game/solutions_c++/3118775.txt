#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdlib>
#include <cstdio>
#include <cstring>
#include <sstream>
using namespace std;
vector <int> vect;
long long int a;
int main()
{
    int n,q;
    cin>>n>>q;
    while(n--){
    	int a1,b1;
    	cin>>a1>>b1;
    	if(a1>b1)
    	   swap(a1,b1);
    	vect.push_back(b1-a1);
    	a+=a1;}
    	bool ff=true;
   sort(vect.begin(),vect.end());
   for(int i=0;i<vect.size();i+=2){
   	if(vect[i]!=vect[i+1])
   	   ff=false;
   	else
   	    a+=vect[i];
   	
   	   
   }
   for(int i=0;i<q;i++)
    {
    	long long int a2;
    	cin>>a2;
    	if(ff==true && a2==a)
    	   cout<<"YES"<<endl;
    	else
    	   cout<<"NO"<<endl;
    	
    }
    return 0;
}
