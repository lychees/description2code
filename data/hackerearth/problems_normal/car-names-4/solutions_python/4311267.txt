from collections import Counter
for t in range(input()):
	val = raw_input()
	counts = Counter(val)
	use = 0
	double_val = 0
	for key,value in counts.iteritems():
		if value != 0:
			double_val = value
			use = use + 1
	if use != 3:
		print "Not OK"
	else:
		print_ok = True
		rev_val = val[0:double_val]
		if val[0:double_val] != rev_val[::-1]:
			print_ok = False
			print "Not OK"
			continue
		
		rev_val = val[double_val : double_val * 2]
		if val[double_val: double_val * 2] != rev_val[::-1]:
			print_ok = False
			print "Not OK"
			continue
		
		rev_val = val[double_val * 2:]
		if val[double_val*2:] != rev_val[::-1]:
			print_ok = False
			print "Not OK"
			continue
		
		if print_ok:
			print "OK"
