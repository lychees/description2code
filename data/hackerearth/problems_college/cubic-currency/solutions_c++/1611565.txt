#include <bits/stdc++.h>
using namespace std;
long long int dp[10002];
int wt[22];
int main()
{
  int t,i,j;
  cin>>t;
  for(i=1;i<=21;i++)
  {
  	j=i*i*i;
  	wt[i]=j;
  }
  dp[0] = 1 ;
  for(int i=1;i<=21;i++){
  	for(int j=1;j<=10000;j++){
  		if(j-wt[i] >= 0 )
  			dp[j] += dp[j-wt[i]] ;
  	}
  }
  while(t--)
  {
  	int n;
  	cin>>n;
  	cout<<dp[n]<<endl;


  }



  // cout << "Hello World!" << endl;
    return 0;
}
