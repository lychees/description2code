def gcd(a,b):
	if (a%b==0):
		return b
	return gcd(b,a%b)
	
z=input()
a=map(int,raw_input().split())
a=sorted(a)
ans=0
for x in a:
	ans+=x
	
	
#for x in a:
#	for y in a:
#		ans+=abs(x-y)

start=0
aa=0
count=0
for x in a:
	aa+=2*(count*x-start)
	start+=x
	count+=1
	
ans+=aa
num=ans
den=z
g=gcd(num,den)
num,den=num/g,den/g
print str(num)+"/"+str(den)
