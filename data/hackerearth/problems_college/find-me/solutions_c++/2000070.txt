#include <iostream>
using namespace std;

int main()
{
	int t,i,flag=0,n;
	long p,a[10000],c;
	cin>>t;
	while(t--)
	{
		flag=0;
		c=0;
		cin>>n>>p;
		for(i=0;i<n;i++)
		cin>>a[i];
		for(i=0;i<n;i++)
		{
			if(a[i]==p)
			{
				cout<<"YES ";
				flag=1;
			}
			if(a[i]>p)
			c++;
		}
		if(flag==0)
		cout<<"NO 0\n";
		else
		cout<<c<<endl;
	}
}
