t = int(input())
while t > 0:
    flag = 0
    n, k = [int(x) for x in raw_input().split()]
    count = 0
    num = [int(x) for x in raw_input().split()]
    for i in num:
        if i <= 0:
            count += 1
        if count >= k:
            flag = 1
            break
    if flag == 0:
        print 'YES'
    else:
        print 'NO'
    t -= 1
