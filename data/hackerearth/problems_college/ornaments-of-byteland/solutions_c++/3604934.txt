#include <bits/stdc++.h>
using namespace std;
int main() {
   int h[30] = {0}, i, j, c = 0, t;
   char str[1000];
   scanf("%d", &t);
   getchar();
   for(j = 1 ; j <= t ; j++) {
      scanf("%s", str);
      for(i = 0 ; str[i] != '\0' ; i++)
         if(h[str[i] - 97] == j-1)
            h[str[i] - 97] = j; 
   } 
   for(i = 0 ; i < 26 ; i++)
      if(h[i] == t)
         c++;
   printf("%d\n", c);
   return 0;    
}
