#include<bits/stdc++.h>
using namespace std;
vector<int> adj[505];


int main()
{
    int n,m,u,v,s,d,x;
    int pp[505][505]={0};
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%d %d %d",&u,&v,&x);
        adj[u].push_back(v);
        adj[v].push_back(u);
        pp[u][v]=x;
        pp[v][u]=x;
    }
    int visited[509],dist[509];
    for(int i=0;i<509;i++){
            visited[i]=0;
            dist[i]=0;
    }
    //printf("Enter the nodes u want to find path.\n");
    scanf("%d",&s);
    queue<int> q;
    q.push(s);
    dist[s]=0;
    visited[s]=1;
    while(!q.empty()){
        int cur=q.front();
        //printf("%d Prev:%d\n",cur,prev[cur]);
        q.pop();
        for(vector<int>::iterator it=adj[cur].begin();it!=adj[cur].end();it++){
            if(!visited[*it]||(dist[*it]>dist[cur]+pp[*it][cur])){
                visited[*it]=1;
                dist[*it]=dist[cur]+pp[*it][cur];
                q.push(*it);
            }
        }
    }
    int test,l;
    scanf("%d",&test);
    while(test--){
        scanf("%d",&l);
        if((dist[l]==0&&l!=s)||!visited[l])printf("NO PATH\n");
        else printf("%d\n",dist[l]);
    }
    return 0;
}
