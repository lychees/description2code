#include <iostream>
using namespace std;
int fact(int n)
{
	int q=1;
	if(n==0)
	 return 1;
	else
	 for(int i=1;i<=n;i++)
	   q=q*i;
	   return q;
	 
}

int main()
{
   
    int a,b,c,i,j,k;
    cin>>a;
    for(i=0;i<a;i++)
    {
    	cin>>b>>c;
    	c=c-1;
    	k=c+b;
    	k=fact(k);
    	c=fact(c);
    
    	b=fact(b);
    	
    	j=k/(c*b);
    	cout<<j<<endl;
    }
    
    return 0;
}
