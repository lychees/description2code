#include <iostream>
using namespace std;

int main()
{
   int n ;
   scanf("%d",&n) ;
   int arr[n] ;
   for(int i=0;i<n;i++)
   scanf("%d",&arr[i]) ;
   int tc ;
  cin>>tc ;
  	int l,r,sum ;
  while(tc--)
  {
   sum=0 ;
  	cin>>l>>r ;
  	for(int i=l-1;i<r;i++)
  	sum=sum+arr[i] ;
  	
  	cout<<sum<<endl ;
  	
  }
    return 0;
}
