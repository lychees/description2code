c = 1000000007
def fact(cache):
	cache[0] = 1
	for i in range(1, 51+1):
		cache[i] = cache[i-1] * i

cache = [i for i in range(0, 55)]
fact(cache)
t = input()
for i in range(0, t):
	l = raw_input().split(' ');
	g = int(l[0])
	b = int(l[1])
	if(g < b-1):
		print -1
	else:
		print (cache[g] * (cache[g+1]/ (cache[b] * cache[g+1-b])) * cache[b] ) % c
