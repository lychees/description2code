#include<bits/stdc++.h>
using namespace std;
#define min(a,b) a<b?a:b;
int prior[26];
bool comp(string a,string b)
{
    int i;
    int p=min(a.length(),b.length());
    //cout<<p<<"   "<<a<<"   "<<b<<"   return : ";
    for(i=0;i<p;i++)
    {
        if(prior[a[i]-'a']<prior[b[i]-'a']){ //cout<<a<<"  type1-a:  "<<i<<endl;
            return true;}
        else if(prior[a[i]-'a']>prior[b[i]-'a']){ //cout<<b<<"  type1-b:  "<<i<<endl;
            return false;}
    }
    if(a.length()<b.length()){//cout<<a<<"  type2:  "<<endl;
        return true;}
    //cout<<b<<"  type3:  "<<endl;
    return false;
}
int main()
{
    int t,i,n;
    string str;
    //freopen("input.txt","r",stdin);
    cin>>t;
    while(t--)
    {
        cin>>str;
        for(i=0;i<str.length();i++)
        {
            prior[str[i]-'a']=i;
        }
        cin>>n;
        string a[n];
        for(i=0;i<n;i++)
            cin>>a[i];
        /*for(i=0;i<26;i++)
            cout<<prior[i]<<" ";
        cout<<endl;*/
        sort(a,a+n,comp);
        for(i=0;i<n;i++)
            cout<<a[i]<<endl;
    }
    return 0;
}
