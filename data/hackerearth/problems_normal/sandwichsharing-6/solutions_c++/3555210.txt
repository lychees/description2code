#include <iostream>
#include <math.h>
using namespace std;

int main()
{
    int T, N, P;
    cin >> T;
    for(int i=0; i<T; i++){
	    cin >> N;
	    cin >> P;
	    for(int j=0; j<P && N>3; j++){
	    	N=ceil((double)N/2)+1;
	    	//cout << "N=" << N << endl;
	    }
	    cout << N << endl;
    }
    return 0;
}
