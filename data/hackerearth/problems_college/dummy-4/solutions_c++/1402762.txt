vector<string> split(const string &s, char c) {
	vector<string> v;
	stringstream ss(s);
	string x;
	while (getline(ss, x, c)) v.push_back(x);
	return v;
}

const int N = 505;
string s[N];
int n;

string add(string x, string y) {
	if (x.size() < y.size()) swap(x, y);
	string ans = "";
	reverse(all(x));
	reverse(all(y));
	int carry = 0;
	rep1(i, 0, y.size()) {
		int d = (x[i] - '0') + (y[i] - '0') + carry;
		ans += '0' + (d % 10);
		carry = d / 10;
	}
	rep1(i, y.size(), x.size()) {
		int d = (x[i] - '0') + carry;
		ans += '0' + (d % 10);
		carry = d / 10;
	}
	if (carry) ans += '0' + carry;
	return ans;
}

int sum(string x) {
	int num = 0;
	rep1(i, 0, x.size()) num += x[i] - '0';
	while (num >= 10) {
		int tmp = 0;
		while (num) {
			tmp += num % 10;
			num /= 10;
		}
		num = tmp;
	}
	return num;
}

    	ans = add(s[i], ans);
