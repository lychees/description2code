#include<bits/stdc++.h>
#define ll long long int
#define MAXN 2345
using namespace std;
int n, k;
ll mat[1002][1002], dp[1002][1002];
bool is_perfect_cube(ll n) {
    ll root(round(cbrt(n)));
    return n == root * root * root;
}
int main(){
//freopen("input.txt", "r", stdin);
cin>>n>>k;
for(int i = 1; i <= n; i++){
	for(int j= 1; j <= n; j++) cin>>mat[i][j];
}
for(int i = 1; i <=n ; i++){
	for(int j = 1; j <= n; j++){
		dp[i][j] = dp[i-1][j] + dp[i][j-1] - dp[i-1][j-1] + mat[i][j];
	}
}
ll ans = 0;
for(int i = 1; i <= n; i++){
	for(int j = 1; j <= n; j++){
		int I = i + k - 1 , J = j + k - 1;
		if(I > 0 && I <= n && J > 0 && J <= n){
			ll sum = dp[I][J] - dp[I][j-1] - dp[i-1][J] + dp[i-1][j-1];
			ans += is_perfect_cube(sum);
		}
	}
}
printf("%lld\n",ans);
return 0;
}
