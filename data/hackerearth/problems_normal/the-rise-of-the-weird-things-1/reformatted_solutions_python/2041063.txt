N = int(raw_input())
powers = raw_input()
powers = powers.split(' ')
zombies = []
vampires = []
for i in powers:
    i = int(i)
    if i % 2 == 0:
        zombies.append(i)
    else:
        vampires.append(i)
zombies.sort()
vampires.sort()
zombies.append(sum(zombies))
vampires.append(sum(vampires))
print ' '.join(str(i) for i in zombies), ' '.join(str(i) for i in vampires)
