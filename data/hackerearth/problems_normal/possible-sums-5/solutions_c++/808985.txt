#include <iostream>
#include<stdio.h>
//#include<unordered_set>
#include<vector>

using namespace std;
int ans=0;
bool** dp;
int vec[101];
int main(){
    int t;
	scanf("%d",&t);
    while (t--) {
    	int n;
    	scanf("%d",&n);
		ans=0;
    	
		int sum=0;
    	for (int i=1;i<=n;i++ )   	{
    		int temp;
    		cin>>temp;
			sum+=temp;
    		vec[i]=temp;    		
    	}
    	
		dp=new bool* [sum+1];
		for (int i=0;i<sum+1;i++) 
			dp[i]=new bool[n+1];

		for (int i=0;i<=n;i++ ) 
			dp[0][i]=1;
		for (int i=1;i<=sum;i++) 
			dp[i][0]=0;
			
		int count=0;
		for (int i=1;i<=sum;i++ ) {
			bool flag=false;
			for (int j=1;j<=n;j++) {
				if (i==vec[j]) {
					dp[i][j]=1;
	
				}
				else if (i>vec[j]) {
					dp[i][j]=dp[i-vec[j]][j-1]||dp[i][j-1];
				}
				else {
					dp[i][j]=dp[i][j-1];
				}
				if (dp[i][j])
					flag=true;
			}
			if (flag)
				count++;
		}

    	printf("%d\n",count+1);
    }
    return 0;
}
