#include<bits/stdc++.h>
using namespace std;
int i,ctr,v,n,t;
int main()
{

cin>>t;
while(t--){
cin>>n;

    int ctr=0;
    for(i=2;i<=sqrt(n);i++)
    {
        v=0;
        if(n%i==0)
            {

            if(i%2==0)
                ctr++;
              v=n/i;
              if((n/i)%2==0 && (n/i)!=i){

          //cout<<v<<" "<<i<<endl;
                ctr++;
                }
            }
    }
    if(n%2!=0)
         cout<<"0"<<endl;
    else
    cout<<ctr+1<<endl;
}

return 0;
}
