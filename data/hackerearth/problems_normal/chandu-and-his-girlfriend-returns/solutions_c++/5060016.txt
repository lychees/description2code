#include <iostream>
#include <vector>
#include <algorithm>
#define ll long long
using namespace std;
 
int main() {
	
	int t,temp,first,second;
	vector<int> vec;
	cin >> t;
	while(t--) {
		cin >> first >> second;
		for(int i=0; i<first;i++) {
			cin >> temp;
			vec.push_back(temp);
		}
		for(int i=0; i<second;i++) {
			cin >> temp;
			vec.push_back(temp);
		}
		
		sort(vec.rbegin(),vec.rend());
		
		for(int i=0; i<vec.size(); i++)
			cout << vec[i] << ' ';
		cout << endl;
		
		vec.clear();
	}
    return 0;
}
