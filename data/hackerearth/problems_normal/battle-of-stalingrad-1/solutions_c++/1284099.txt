#include <iostream>
using namespace std;

double area(int x1,int y1, int x2, int y2, int x3, int y3){
	
	double result =(x1*(y2-y3)+x2*(y3-y1)+x3*(y1-y2))/2.0;
	if(result<0)
		result = result*-1;
		
	return result;
}


int main()
{
    int t,x1,y1,x2,y2,x3,y3,x4,y4;
    double a,a1,a2,a3;
    cin>>t;
    while(t--){
    cin>>x1>>y1>>x2>>y2>>x3>>y3>>x4>>y4;
    a= area(x1,y1,x2,y2,x3,y3);
    a1=area(x1,y1,x2,y2,x4,y4);
    a2=area(x1,y1,x4,y4,x3,y3);
    a3=area(x4,y4,x2,y2,x3,y3);
    //cout<<a<<" "<<a1<<" "<<a2<<" "<<a3<<endl;
   if(a==(a1+a2+a3))
    	cout<<"INSIDE"<<endl;
    else cout<<"OUTSIDE"<<endl;
    }
}
