#include <iostream>
#include <string>
#include <cstring>
#include <algorithm>
#include <sstream>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
    cin.tie(NULL);   
     
	int t;
	string s;
	cin>>t;
    int a[30]={0};
    while(t--)
    {
    	getline(cin,s);
    	int f;
    	if(s[0]=='G')
    		f=2;
    	else if(s[0]=='M')
    		f=1;
    	istringstream iss(s);
    	do
    		{
    			string sub;
        		iss >> sub;
        		int num=atoi(sub.c_str());
    			if(num>=1 && num<=30)
    			{
    				a[num-1]+=f;
    			}
    		} while (iss);
    }
   
    auto result = minmax_element (a,a+30);
    int r=result.second-a+1;
    if(r==19||r==20)
    cout<<"Date"<<endl;
    else
    cout<<"No Date"<<endl;
    return 0;
}
