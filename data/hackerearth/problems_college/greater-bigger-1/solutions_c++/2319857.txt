#include <bits/stdc++.h>
using namespace std;

int main()
{
    string inp,rev;
    int t,size,i,j;
    scanf("%d",&t);
    while(t--)
    {
    	cin>>inp;
    	size=inp.size();
    	if(inp[size-2]<inp[size-1])
        {
        	//cout<<1;
        	swap(inp[size-2],inp[size-1]);
        	cout<<inp<<"\n";
        }
        else
        {
        rev=inp;
        sort(rev.begin(),rev.end());
        if(rev==inp)      
        cout<<"no answer\n";
        else
        {
        	
        	sort(inp.begin(),inp.end());
        	swap(inp[size-2],inp[0]);
        	sort(inp.begin()+1,inp.end()-1);
        	cout<<inp<<"\n";
        }
        }
    }
    return 0;
}
