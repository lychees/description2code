#include <iostream>
using namespace std;

int main()
{
    int tno,i,ans;
    string s;
    char c;
    int skip[26];
    
    cin>>tno;
    
    
    while(tno--)
    {
    	for(i=0;i<26;i++)
    		skip[i]=0;
    	
    	cin>>s;
    	
    	cin>>i;
    	
    	while(i--)
    	{
    		cin>>c;
    		skip[c-'a']=1;
    	}
    	
    	ans=0;
    	for(i=0;i<s.length();i++)
    	{
    		if(skip[s[i]-'a']==0)
    		{
    			ans += s[i]-'a'+1;
    		}
    	}
    	
    	cout<<ans<<endl;
    	
    	
    }
    return 0;
}
