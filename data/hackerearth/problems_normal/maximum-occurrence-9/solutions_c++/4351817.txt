#include<bits/stdc++.h>
using namespace std;

int main()
{
    int a[256]={0};
    char s[1005];
    scanf("%[^\n]s",s);

    int n = strlen(s);

    for(int i=0;i<n;i++)
    {
        a[s[i]]++;
    }

    int maxx=-1;
    int index = -1;

    for(int i=0;i<256;i++)
    {
        if(a[i]>maxx)
        {
            maxx=a[i];
            index=i;
        }
    }

    printf("%c %d\n",index,maxx);
}
