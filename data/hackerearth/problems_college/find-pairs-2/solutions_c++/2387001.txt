#include<iostream>
using namespace std;
int arr[2000005];
int main()
{
std::ios_base::sync_with_stdio(false);
int t;
cin>>t;
while(t--)
{
    long long ans=0;
int n,i,x;
cin>>n;
for(i=0;i<2000001;i++)
    arr[i]=0;
for(i=0;i<n;i++)
{
    cin>>x;
arr[x+1000000]++;
}
for(i=0;i<2000001;i++)
{
    if(arr[i]>0)
    ans+=((long long )(arr[i]))*(arr[i]-1)/2+arr[i];
}
cout<<ans<<endl;
}
return 0;
}
