#include <bits/stdc++.h>
using namespace std;

int a[1000005];
int main()
{
	cin.sync_with_stdio(0);
	cin.tie(0);
    int t;
    cin >> t;
    int m1, m2, n;
    while(t--)
    {
    	memset(a, 0, sizeof a);
    	cin >> n >> m1 >> m2;
    	
    	int mi = m1 > m2 ? m2 : m1;
    	int ma = m1 > m2 ? m1: m2;
    	
    	a[mi] = 1;
    	a[ma] = 1;
    	
    	for(int i = mi + 1; i < ma; i++)
    	{
    		if(a[i - mi] == 0)
    			a[i] = 1;
    		else
    			a[i] = 0;
    	}
    	
    	for(int i = ma + 1; i <= n; i++)
    	{
    		if(a[i - mi] == a[i - ma])
    			a[i] = (a[i-mi] + 1) % 2;
    		else
    			a[i] = 1;
    	}
    	
    	cout << a[n] << '\n';
    }
    
   	/*for(int i = 0; i < n; i++)
   		cout << a[i] << ' ';
   	cout << endl;*/
    return 0;
}
