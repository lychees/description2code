#include<iostream>
using namespace std;

int main()
{
	int n,t;

	cin>>t;
	
	while(t--)
	{
		cin>>n;

		int a[n],val,prev;

		for(int i=0;i<n;i++)
		cin>>a[i];

		for(int i=0;i<n;i++)
		{
			val=0;
			while(a[i]!=0)
			if(a[i]%2!=0)
			{
				val++;
				a[i]/=2;
			}
			else
				a[i]/=2;
			
			a[i]=val;
		}
		val=a[0];
		for(int i=0;i<n;i++)
		if(a[i]<val)
		val=a[i];

		cout<<val<<"\n";
	}
return 0;
}
