#include <iostream>
using namespace std;

int find(int n){
	int r, ans=0;
	int k=0;
	int a = n;
	while(a>0){
		r = a%10;
		a = a/10;
		k=k+1;
	}
	//cout << "k=" <<k <<endl;
	int arr[k];
	int i=0;
	while(n>0){
		r = n%10;
		arr[i] = r;
		i++;
		n = n/10;
	}
	
	for(int i=0; i<k; i++){
		//cout << "arr" << i << "=" <<arr[i] <<endl;
		if(arr[i]==1 && arr[i+1]==2){
			ans=1;
			break;
		}
	}
	return ans;
} 

int main()
{
    int n,k;
    int t;
    cin >>t;
    for(int i=0; i<t; i++){
    	cin >>n;
    	if(n%21==0){
    		cout << "The streak is broken!"<<endl;
		}
    	else {
    		k = find(n);
	    	if(k==1){
    				cout <<"The streak is broken!"<<endl;
    		}else {
    				cout << "The streak lives still in our heart!" <<endl;
    		}
    	}
    }
    return 0;
}
