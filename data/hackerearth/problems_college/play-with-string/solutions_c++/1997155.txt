#include <bits/stdc++.h>
using namespace std;

int main()
{
     int t;
    cin >>  t;
    while(t--){
    	   
    	     string x;
    	     cin >> x;
    	       int flag = 1;
    	       int n = x.size();
    	       for(int i = 1 ; i < n ; ++i )
    	       {
    	       	 int a = abs(x[i]-x[i-1]);
    	       	 int b = abs(x[n-1-i]-x[n-i]);
    	       	  if( a != b)
    	       	   flag = 0;
    	       }
    	       if(flag)
    	       cout <<"Funny" << endl;
    	       else
    	       cout <<"Not Funny" << endl;
    	     
    	
    }
      return 0;
}
