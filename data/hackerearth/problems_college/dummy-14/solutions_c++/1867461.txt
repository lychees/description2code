#include <bits/stdc++.h>
using namespace std;
long long int cube(long long int n)
{
    return n*n*n;
}
vector<long long> v;
void sieve(int n)
{
    long long int a;
    long long  int b;
    long long int c;
    long long int d;

    for(a = 1; a <= n; a++) {
        for(b = a; b <= n; b++) {
            for(c = a+1; c <= n; c++) {
                for(d = c; d <= n; d++) {
                   if(a != b && b != c && c != d && a != c && a != d && b != d) {
                        if(cube(a)+cube(b) == cube(c)+cube(d)) {
                            v.push_back(cube(a)+cube(b));
                        }
                   }
                }
            }
        }
    }
}
int main()
{
        int n;
        int i;
        int t;
        sieve(170);
        sort(v.begin(),v.end());

        cin>>t;
        while(t--) {
            scanf("%d",&n);
            printf("%lld\n",v[n-1]);
        }

}
