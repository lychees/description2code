#include <iostream>
#include <vector>
#include <stdio.h>

using namespace std;

bool Check( int n ) {
		
	while( n > 0 ) {
		if( n % 10 == 7 )
			return true;
		n /= 10;
	}
	
	return false;
}

int main()
{
	int t, m, n, i, len = 1000001, j, count;
	
	vector<bool> A( len, true );
	
	A[ 0 ] = A[ 1 ] = false;
	
	for( i = 2;i*i < len;i++ )
		for( j = i*i;j < len;j += i )
			if( A[ j ] )
				A[ j ] = false;
	
	scanf("%d", &t );
	
	while( t-- ) {
			
		scanf("%d %d", &n, &m );
		count = 0;
		for( i = n;i <= m;i++ )
			if( !A[ i ] && Check( i ) )
				count++;
		if( count )
			printf("%d\n", count);
		else
			printf("-1\n");
		
	}
	return 0;
}
