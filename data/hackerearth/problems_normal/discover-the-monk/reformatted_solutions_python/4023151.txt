def binary_search(array, left, right, value):
    mid = (left + right)/2
    if (right >= left):
        if array[mid] == value:
            return 'YES'
        elif array[mid] > value:
            return binary_search(array, left, mid-1, value)
        else:
            return binary_search(array, mid+1, right, value)
    else:
        return 'NO'


n, q = map(int, raw_input().split(" "))
array = sorted(map(int, raw_input().split(" ")))
query = []
for i in range(q):
    query.append(input())
for each in query:
    print binary_search(array, 0, n-1, each)
