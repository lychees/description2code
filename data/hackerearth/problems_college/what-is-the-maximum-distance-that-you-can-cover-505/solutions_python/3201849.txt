import math
from decimal import Decimal, ROUND_DOWN
def getData():
    return map(Decimal, raw_input().split())

noc = input()
car_speed = getData()
fuel_consume_rate = getData()
fuel_availble = Decimal(raw_input())

speed_fuel_consumed = {}
for i in range(noc):
    speed_fuel_consumed[car_speed[i]] = fuel_consume_rate[i]

max_distance = 0.0
for k, v in speed_fuel_consumed.items():
   
    possible_distance = (fuel_availble / v)  * k    
    if  possible_distance > max_distance:
        max_distance = possible_distance
        
 

print Decimal(max_distance.quantize(Decimal('.001'), rounding=ROUND_DOWN))
