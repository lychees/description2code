#include<iostream>

#define mp make_pair
bool done[100][100]={0},cdone[100][100]={0},giv[100][100]={0};
int a[100][100]={0};
int solve(int i,int j,int n)
{
    if(i<n && j<n && i>=0 && j>=0 && !giv[i][j])
    {
	if(i==n-1 && j==n-1)
	    return 1;
	if(done[i][j])
	    return 0;
	done[i][j]=1;
	a[i][j]=solve(i+1,j,n)+solve(i-1,j,n)+solve(i,j+1,n)+solve(i,j-1,n);
	done[i][j]=0;
	return a[i][j];
    }
    else return 0;

}
int main()
{
    int n;
    cin>>n;
    for(int i=0;i<n;i++)
    {
	for(int j=0;j<n;j++)
	    cin>>giv[i][j];
    }
    cout<<solve(0,0,n);
    return 0;
}
