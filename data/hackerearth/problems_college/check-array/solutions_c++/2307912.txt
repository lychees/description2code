#include<cstdio>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,x;
		scanf("%d",&n);
		int a[n+2];
		a[0]=0;
		//a[n+1]=0;
		for(int i=1;i<=n;i++)
		{
			scanf("%d",&x);
			a[i]=a[i-1]+x;
		}
		int fg=0,i;
		for(i=1;i<=n;i++)
		{
			if(a[i-1]==a[n]-a[i])
			{
				fg=1;
				break;
			}
		}
		//printf("%d\n",i);
		if(fg==0)
			printf("NO\n");
		else
			printf("YES\n");
	}
	return 0;
}
