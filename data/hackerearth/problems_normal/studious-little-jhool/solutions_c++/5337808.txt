#include <bits/stdc++.h>
using namespace std;


int dp[123];

int rec(int total){
	if(total==0) return 0;
	if(total<0)return 1e9;
	if(dp[total]!=-1) return dp[total];
	
	int res=1+min(rec(total-10),rec(total-12));
	
	return dp[total]=res;
	
	
	
	
	}
int main() {
int t,n;
memset(dp,-1,sizeof dp);
cin>>t;
while(t--){
	cin>>n;
	int ans=rec(n);
	if(ans>n) cout<<"-1"<<endl;
	else cout<<ans<<endl;
	
	
}
	
	
	
	
// your code goes here
	return 0;
}
