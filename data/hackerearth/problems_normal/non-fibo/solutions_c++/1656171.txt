#include <bits/stdc++.h>

#define For(i, be, en) for(i = be; i <= en; i++)
#define Forr(i, be, en) for(int i = be; i >= en; i--)

using namespace std;

int main()
{
    int t;
    long long int ans, n, f1, f2, te;

    cin >> t;
    while (t--) {
        cin >> n;

        f1 = 3;
        f2 = 5;
        while (1) {
            if (n <= f2 - f1 - 1) {
                cout << f1 + n << endl;
                break;
            } else {
                n -= f2 - f1 - 1;
                te = f2;
                f2 = f1 + f2;
                f1 = te;
            }
        }
    }

	return 0;
}
