#include<stdio.h>
#include<string.h>
#include<algorithm>
#include<iostream>
using namespace std;
#define S(t) scanf("%d",&t)
#define S2(t,q) scanf("%d%d",&t,&q)
#define P(a) printf("%d\n",a)
#define For(i,a,b) for(i=a;i<b;i++)
typedef long long int ll;
int main()
{
    int t;
    S(t);
    while(t--)
    {
              int k,z,n,i,j,q;
              S2(k,z);
               char str[101];
               scanf("%s",str);
               n=strlen(str);
              int fromto[100][100];
              memset(fromto,-1,sizeof(fromto));
              For(i,0,n)
              {
                              ll count=0;
                              For(j,i,n)
                              {
                                count=count*10+(str[j]-'0');
                                if(count>z)
                                break;
                                fromto[i][j]=count;
                                
                              }
              }
              ll dp[101][100]={0};
             For(i,0,k+1)
             {
               For(j,0,n)
               dp[i][j]=-1;          
             }
              for(i=0;i<n&&fromto[0][i]!=-1;i++)
              dp[1][i]=fromto[0][i];
              For(i,2,k+1)
              For(j,0,n)
              For(q,j+1,n)
              if(fromto[j+1][q]!=-1&&dp[i-1][j]!=-1)
              dp[i][q]=max(dp[i][q],fromto[j+1][q]+dp[i-1][j]);
              cout<<dp[k][n-1]<<endl;
              
    }
    
    return 0;
}
