#include <iostream>
#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll mul(ll a,ll b,ll mod)
{
	ll ans=0;
	ll y=a;
	while(b>0)
	{
		if(b%2)
		ans=(ans%mod+y%mod)%mod;
		y=y*2;
		if(y>mod)
		y%=mod;
		b/=2;
	}
	return ans%mod;
	
}
ll power(ll a,ll b,ll mod)
{
	ll ans=1,y=a;
	while(b>0)
	{
		if(b%2==1)
		{
			ans=ans*y;
			if(ans>mod)
			ans%=mod;
		}
		y=y*y;
		if(y>mod)
		y%=mod;
		b/=2;
	}
	return ans%mod;
}
ll inverse(ll a,ll mod)
{
	return power(a,mod-2,mod);
}
int main()
{
    int tc,n,k,i;
    ll ans=1;
    ll mod=1000000007;
    ll fact[100005];
    fact[0]=1;
    fact[1]=1;
    for(i=2;i<=100000;i++)
    fact[i]=(i%mod)*(fact[i-1]%mod)%mod;
    cin>>tc;
    while(tc--)
    {
    	cin>>n>>k;
    	ll temp=inverse(fact[k-n],mod);
    	ans=((fact[k]%mod)*(temp%mod))%mod;
    	//cout<<fact[k]<<" "<<fact[k-n]<<" ";
    	cout<<ans<<endl;
    	
    }
    return 0;
}
