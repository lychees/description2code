#include <bits/stdc++.h>
#define PB push_back
#define MP make_pair
#define LL long long
#define MAXN 5000011
#define M 1000000007
#define sc(x) scanf("%d",&x)
#define scl(x) scanf("%I64d",&x)
using namespace std;

int a[MAXN + 100];

void prime(){
    int i, j;
    for(i=0;i<MAXN;i++)
        a[i] = 1;

    for(i=0;i<MAXN;i+=2)
        a[i] = 0;

    a[1] = 0;
    a[2] = 1;

    for(i=3;i<=sqrt(MAXN);i+=2){
        if(a[i] == 1){
            for(j=i*i;j<MAXN;j+=2*i)
                a[j] = 0;
        }
    }
    return;
}

vector<int > v1, v2;

int main(){
    int n, t, i, j;
    int x, y, ans;
    int ii, jj;
    prime();
    cin>>t;
    while(t--){
        cin>>x>>y;
        ans = 0;

        v1.clear();
        v2.clear();
        while(x != 0){
            v1.PB(x);
            x /= 2;
        }

        while(y != 0){
            v2.PB(y);
            y /= 2;
         //   cout<<"here"<<endl;
        }

/*        for(i=0;i<v1.size();i++)
            cout<<v1[i]<<" ";
        cout<<endl;

        for(i=0;i<v2.size();i++)
            cout<<v2[i]<<" ";
        cout<<endl;
     */
        int g = 0;
        for(i=0;i<v1.size();i++){
            for(j=0;j<v2.size();j++){
                if(v1[i] == v2[j]){
                    ii = i;
                    jj = j;
                    g  = 1;
                    break;
                }
            }
            if(g == 1)
                break;
        }
       // cout<<ii<<" "<<jj<<endl;
        for(i=0;i<=ii;i++){
            if(a[v1[i]] == 1)
                ans -= v1[i];
            else
                ans += v1[i];
        }

        for(i=0;i<jj;i++){
            if(a[v2[i]] == 1)
                ans -= v2[i];
            else
                ans += v2[i];
        }

        cout<<ans<<endl;
    }
}
