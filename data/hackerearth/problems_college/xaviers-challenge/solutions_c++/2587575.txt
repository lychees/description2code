#include<bits/stdc++.h>
#include<iomanip>
using namespace std;
int main(){
	long int i,t,sum,num,length,f;
	string str;
	cin>>t;
	while(t--){
		cin.ignore();
		getline(cin,str);
		length=str.length();
		sum=0;
		for(i=0;i<length;i++){
			num=str[i];
			sum=sum+num;
		}
		f=sum/length;
		if(f%2!=0){
			for(i=length-1;i>=0;i--){
				cout<<str[i];
			}
			cout<<"\n";
		}
		else{
			cout<<str<<"\n";
		}
	}
	return 0;
}
