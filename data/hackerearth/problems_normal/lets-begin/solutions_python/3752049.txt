
def getSum(l,x,r):
	for c in range(x+1):
		minsum =c 
		for i in [s for s in l if s<=c]:
			if 1+r[c-i] < minsum:
				minsum=1+r[c-i]
		r[c] = minsum
	return r[x]		

l = [2,3,5,7]
t = int(raw_input())
while t:
	t=t-1
	x = int(raw_input())
	if x ==1:
		print -1
	else:
		r = [0]*(x+1)
		print getSum(l,x,r)
