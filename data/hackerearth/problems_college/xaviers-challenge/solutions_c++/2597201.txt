#include<bits/stdc++.h>
using namespace std;
	 char s[10000] ;
int t,n;
int main()
{
    
    int t;
    cin>>t;

    while(t--)
    {
    
        scanf (" %[^\n]s",s);
        n = strlen(s);
        int sum =0;
        for(int i=0;i<n;i++)
        {
            sum += s[i];
        }
    
        if((sum/n)%2!=0)
        {
            for(int i=n-1;i>=0;i--)
            cout<<s[i];
            cout<<endl;
        }
        else cout<<s<<endl;
    }
}
