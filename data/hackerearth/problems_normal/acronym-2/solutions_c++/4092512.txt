#include <iostream>
#include <bits/stdc++.h>
using namespace std;

char c[11][105];
char tmp[105];
char rez[105];
int main()
{
    int k; scanf("%d", &k);
    for(int i = 0; i < k; i++) {
    	scanf("%s", c[i]);
    }
    int p = 0;
    int n; scanf("%d", &n);
    for(int i = 0; i < n; i++) {
    	scanf("%s", tmp);
    	int res = 1;
    	for(int j = 0; j < k; j++) {
    		if(strcmp(c[j], tmp) == 0) res = 0;
    	}
    	if(res == 1) rez[p++] = tmp[0]-'a'+'A';
    }
    for(int i = 0; i < p; i++) {
    	printf("%c", rez[i]);
    	if(i != p-1) printf(".");
    }
    return 0;
}
