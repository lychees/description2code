import math
N = input()
X = [0] * N
Y = [0] * N
F = [0] * N
for i in xrange(N):
    X[i], Y[i], F[i] = map(int, raw_input().split())
dp = [-1 << 30] * N

dp[0] = F[0]
for i in xrange(1, N):
    for j in xrange(i):
        dx = X[i] - X[j]
        dy = Y[i] - Y[j]
        d = math.sqrt(dx * dx + dy * dy)
        dp[i] = max(dp[i], dp[j] + F[i] - d)

print "%.6f" % dp[-1]
