#include <iostream>
using namespace std;

int main()
{
    int T;
    cin>>T;
    
    while(T--)
    {
    	string S;
    	cin>>S;
    	
    	int x=0,o=0;
    	for(int i=0;S[i]!='\0';i++)
    	{
    		if(S[i]=='x')x++;
    		else if(S[i]=='o')o++;
    		else
    		{
    			if(x>0)
    			{
    				for(int j=0;j<x;j++)
    				cout<<"x";
    			}
    			else
    			{
    				for(int j=0;j<o;j++)
    				cout<<"o";
    			}
    			x=0,o=0;
    			cout<<"*";
    		}
    	}
    	if(x>0)
    	{
    		for(int j=0;j<x;j++)
    		cout<<"x";
    		
    	}
    	else if(o>0)
    	{
    		for(int j=0;j<o;j++)
    		cout<<"o";
    	}
    	cout<<"\n";
    }
    return 0;
}
