#include <iostream>
using namespace std;

int main()
{
    int t,n,i;
    int a[26],largest;
    string s;
    cin>>t;
    while(t)
    {
    	cin>>n;
    	cin>>s;
    	for(i=0;i<26;i++)
    	{
    		a[i] = 0;
    	}
    	for(i=0;i<n;i++)
    	{
    		a[s[i]-97]++;
    	}
    	largest = a[0];
    	for(i=0;i<26;i++)
    	{
    		if(a[i]>largest)
    		{
    			largest = a[i];
    		}
    	}
    	if(largest % 2 == 0)
    	{
    		cout<<"Yes";
    	}
    	else
    	{
    		cout<<"No";
    	}
    	cout<<"\n";
    	t--;
    }
    return 0;
}
