#include<bits/stdc++.h>
using namespace std;
#define ll long long
void lol(ll arr[], ll n, ll sum)
{
    ll curr_sum = arr[0], start = 0, i;
    for (i = 1; i <= n; i++)
    {
        while (curr_sum > sum && start < i-1)
        {
            curr_sum = curr_sum - arr[start];
            start++;
        }
        if (curr_sum == sum)
        {
            printf ("YES\n");
            return;
        }
        if (i < n)
          curr_sum = curr_sum + arr[i];
    }
    printf("NO\n");
    return;
} 
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        ll n,sum;
        scanf("%lld%lld",&n,&sum);
        ll arr[n];
        for(ll i=0;i<n;i++)
            scanf("%lld",&arr[i]);
        lol(arr, n, sum);
    }
    return 0;
}
