#include <iostream>
#include <cmath>
using namespace std;

bool prime[100];
void init(){

	for(int i = 0;i<100;i++)
		prime[i] = true;

    prime[0] = false;
    prime[1] = false;
	for(int i = 2;i<100;i++){
		for(int j = i*2;j<100;j=j+i){
			if(j%i == 0)
				prime[j] = false;
		}
	}
}
bool isPrime(int n){
	return prime[n];
}

int main()
{
    int t;
    init();
    cin>>t;

    while(t--){
    	int n,d = 0;
    	cin>>n;

    	while(n!=0){
    		d += n%10;
    		n/=10;
    	}
    	isPrime(d)?cout<<"YES\n":cout<<"NO\n";
    }

    return 0;
}
