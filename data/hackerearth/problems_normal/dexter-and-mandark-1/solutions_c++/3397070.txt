#include<bits/stdc++.h>
using namespace std;
int const N = 1e3+10, mod = 1e9+7;
int t, n, m, dp[N][N];
int solve(int i, int j){
  if(dp[i][j] == -1){
    if(i == n){
      int D = i-j;
      if(D > j) dp[i][j] = 1;
      else dp[i][j] = 0;
    }
    else{
      int ret = 0;
      if(i+1-j-1 >= (j+1)*m) ret = (ret + solve(i+1,j+1)) % mod;
      if(i+1-j >= j*m) ret = (ret + solve(i+1,j)) % mod;
      dp[i][j] = ret;
    }
  }
  return dp[i][j];
}
int main(){
  cin >> t;
  while(t--){
    cin >> n >> m;
    memset(dp,-1,sizeof(dp));
    int ans = solve(0,0);
    cout << ans << '\n';
  }
}
