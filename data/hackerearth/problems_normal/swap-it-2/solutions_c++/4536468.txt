#include<bits/stdc++.h>
using namespace std;
int A[1009];
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,k,i,j,pos;
        cin>>n>>k;
        for(i=0;i<n;i++) cin>>A[i];
        for(i=0;i<n;i++)
        {
            pos=i;
            for(j=i+1;j<n;j++)
            {
                if(j-i>k)
                    break;
                if(A[pos]>=A[j])
                    pos=j;
            }
            for(int p=pos;p>i;p--)
                swap(A[p],A[p-1]);
            k-=(pos-i);
        }
        for(i=0;i<n;i++)
        cout<<A[i]<<" ";
        cout<<endl;
    }
}
