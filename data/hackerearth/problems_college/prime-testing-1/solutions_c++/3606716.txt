long long int a[100000];
bool check[50090];
    void sieve()
{
    for(int i=3;i<=224;i+=2)
    {
        if(!check[i])
        {
            for(int j=i*i;j<=50090;j+=i)
                check[j]=true;
        }
    }
    check[2] = false;
    
    /*for(int i=3;i<=50090;i+=2)

}
int main()
{
	long long int k,n,p,count=0;
   	cin>>n;
    sieve();
   	for(long long int i=0;i<n;i++)
   		{
            count=0;
   			cin>>k;
   			for(long long int j=0;j<k;j++)
   				cin>>a[j];

	{
		if((!check[a[p]] && a[p]!=1  && a[p]%2!=0) || a[p]==2 ){
			count++;//cout<<a[p]<<"a ";
        }
	}
        cout<<count<<"\n";
   	}	
    return 0;
}
