#include <bits/stdc++.h>
using namespace std;

vector<int> blue;
    vector<int> yellow;
bool check(int i,int j)
{
	int mi=min(blue[i],yellow[j]);
	int ma=max(blue[i],yellow[j]);
	if(2*mi*mi>ma*ma) return true;
	return false;
}

int main()
{
    //cout << "Hello World!" << endl;
    int n;
    cin>>n;
    
    int color,length;
    for(int i=0;i<n;i+=1)
    {
    	scanf("%d %d",&color,&length);
    	if(color==1)blue.push_back(length);
    	else yellow.push_back(length);
    }
    sort(blue.begin(),blue.end());
    sort(yellow.begin(),yellow.end());
    int minx=min(blue.size(),yellow.size());
    int i=0,j=0,ans=0;
    while(i<blue.size() && j<yellow.size())
    {
    	if(check(i,j))
    	{
    		ans++;
    		i++;j++;
    	}
    	else
    	{
    		if(blue[i]<yellow[j]) i++;
    		else j++;
    	}
    }
    cout<<ans<<endl;
    return 0;
}
