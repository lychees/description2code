#include <iostream>
using namespace std;
int equilibrium(int arr[], int n)
{
   long sum = 0;      // initialize sum of whole array
   long leftsum = 0; // initialize leftsum
   int i;
 
   /* Find sum of the whole array */
   for (i = 0; i < n; ++i)
        sum += arr[i];
 
   for( i = 0; i < n; ++i)
   {
      sum -= arr[i]; // sum is now right sum for index i
 
      if(leftsum == sum){
      	return 1;
    	break;
      }
        
 
      leftsum += arr[i];
   }
 
    /* If no equilibrium index found, then return 0 */
    return -1;
}
 
int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++){
    	int n;
    	cin>>n;
    	int a[n];
    	long leftSum=0;
    	for(int j=0;j<n;j++){
    		cin>>a[j];
    		
    	}
    	int x=equilibrium(a,n);
    	if(x==1){
    		cout<<"YES"<<endl;
    	}
    	else{
    		cout<<"NO"<<endl;
    	}
    }
    return 0;
}
