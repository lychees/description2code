t = int(raw_input())
while t > 0:
    s = raw_input()
    num = 0
    for i in range(0, len(s)-1):
        for j in range(i+1, len(s)):
            if s[i] == s[j]:
                num = 1
                print("Yes")
                break
        if num == 1:
            break
    if num == 0:
        print("No")
    t -= 1
