unsigned long long calculate_sum_of_digits(unsigned long long digits)
{
	unsigned long long sum = 0;

	while(digits != 0){
		sum += (digits % 10);
		digits /= 10;
	}
	return sum;
}

	unsigned long long len,sum=0;

			cin>>len>>d;
			sum+= (len * d);
			while(sum >= 10){
			sum = calculate_sum_of_digits(sum);
		}
			
		}

	}
	return 0;
}
