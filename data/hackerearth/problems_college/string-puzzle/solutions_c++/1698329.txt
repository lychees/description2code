#include <iostream>
using namespace std;

void swap(char *x,char *y)
{
	char temp;
	temp=*x;
	*x=*y;
	*y=temp;
}
void display(char *a,int s,int e)
{
	int j;
	if(s==e)
		cout<<a<<endl;
	else
	{
		for(j=s;j<=e;j++)
		{			
			swap(a+s,a+j);
			display(a,s+1,e);
			swap(a+s,a+j);
		}
	}
}
int main()
{
    char a[100];
    cin>>a;
    int i,n,s,e;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>s;
    	cin>>e;
    	display(a,s,e);
    	cout<<endl;
    }
    return 0;
}
