#include <iostream>
#include <vector>
#include <cstdio>
#include <sstream>
#include <map>
#include <string>
#include <algorithm>
#include <queue>
#include <cmath>
#include <functional>
#include <set>
#include <ctime>
#include <random>
using namespace std;

int main(){
	int t;
	cin >> t;

	auto func = [&](long long x){
		long long ret = 0;
		while(x){
			ret += (x%10) * (x%10);
			x /= 10;
		}
		return ret;
	};

	int SZ = 8000;
	vector<vector<long long>> next(SZ, vector<long long>(30, 0));
	for(int i=1; i<SZ; i++){
		next[i][0] = func(i);
	}
	for(int k=1; k<30; k++){
		for(int i=0; i<SZ; i++){
			next[i][k] = next[next[i][k-1]][k-1];
		}
	}
	while(t--){
		long long n,m;
		cin >> n >> m;
		n = func(n);
		m--;

		int cnt = 0;
		while(m){
			if(m&1) n = next[n][cnt];
			m >>= 1LL;
			cnt++;
		}
		cout << n << endl;
	}
	return 0;
}
