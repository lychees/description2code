# print 'Hello World!'
import collections
n = input()
l = []
while n > 0:
    n -= 1
    s = raw_input()
    l.append(s)
l.sort()
p = []
c = collections.Counter(l)
for key in c:
    p.append(key)
p.sort()
for key in p:
    print key, c[key]
