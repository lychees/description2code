#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long t,l,r,i,j,c=0;
	float d,p,f;
	vector<long long> v(1000001,0);
	v[0]=1;v[1]=1;
	for(i=2;i<1000001;i++)
	{
		if(!v[i])
		{
			for(j=2*i;j<1000001;j+=i)
			v[j]=1;
		}
	}
	/*for(int i=0;i<100;i++)
	cout<<v[i]<<" ";*/
	map<long long,long long> m;
	for(i=0;i<1000001;i++)
	{
		if(!v[i])
		{
			m[i]=c+1;
			c=m[i];
		}
		else
		m[i]=c;
	} 
	/*for( i=0;i<100;i++)
	cout<<m[i]<<" "<<"\n";*/
	cin>>t;
	while(t--)
	{
		
		cin>>l>>r;
		d=float(m[r]-m[l]);
		if(v[l]==0)d++;
		p=float(r-l+1);
		//cout<<d<<" "<<p<<"\n";
		f=d/p;
		cout<<fixed<<setprecision(6)<<f<<"\n";
	}
}
