n = int(input())
l = list(map(int, raw_input().split()))
ttl = int(input())
ln = len(l)
f = 0
for i in range(ln):
    for j in range(ln):
        if j != i:
            for k in range(ln):
                if k != j and k != i:
                    if l[i]+l[j]+l[k] == ttl:
                        f = 1
                        break
        if f == 1:
            break
    if f == 1:
        break
if f == 1:
    print 'True'
else:
    print 'False'
