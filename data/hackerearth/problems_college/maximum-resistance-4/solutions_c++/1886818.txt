typedef long long int ll;
const int mx = 10000;

using namespace std;

int t;
vector <int> v;
string s;



int main(){
//    freopen("data.in", "r", stdin);
    scanf("%d", &t);
    while(t--){
        cin >> s;
        stack <int> q;

        int j = 0;
        for(int i = s.size() - 1; i >= 0; i--){
            if(s[i] == 'X'){
                q.push(1);
                j++;
            }
            else{
                int n = q.top();q.pop();
                int m = q.top();q.pop();
                if(s[i] == 'A')q.push(m + n);
                else q.push(max(n, m));
            }
        }
        while(j--){
            int a;
            scanf("%d", &a);
//            cout<< a << " ";
            v.push_back(a);
        }
        sort(v.begin(), v.end(),greater <int>());
        ll ans = 0;
        for(int i = 0; i < q.top() ; i++){
            ans += 1LL*v[i];
        }
        v.clear();
        printf("%d\n", ans);
    }
    return 0;
}
