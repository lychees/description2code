#include <iostream>
using namespace std;
class queue{
	public:
	long long array[10000000];
	long long head ,tail;

	queue(){

		head=0;
		tail=0;
	}

	void enqueue(long long x){
		array[tail]=x;
		tail++;
	}

int dequeue(){
	if(head==tail)
	{
	return	0;}

	else {tail--;
        return array[tail];

	}
	}

};

int main()
{
  long long x;
  cin >> x;
  queue chef;
  for(long long i=0;i<x;i++)
  {int y;
  	cin>>y;
  	if(y==1)
  	{int w;
  	w = chef.dequeue();
  	if(w==NULL)
  	{cout<<"No Food"<<endl;}
  	else{ cout<<w<<endl;}

  	}


  	else{long long z;
  		cin>>z;
  		chef.enqueue(z);
  	}

  }
  return 0;
}
