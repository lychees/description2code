vector<pair<ll,ll> > v;
int bin(ll key)
{
    int l=0;
    int u=v.size()-1;
    while(l<=u)
    {
        int m=l+(u-l)/2;
        if(key == v[m].F)
            return m;
        if(key<v[m].F)
            u=m-1;
        else
            l=m+1;
    }
    return l;
}
