#include <bits/stdc++.h>
using namespace std;
struct node{
	int sum;
	int ini;
};
struct compare{
	bool operator()(const node& l,const node& r){
		if((l.sum+l.ini)>(r.sum+r.ini)){
			return true;
		}
		if((l.sum+l.ini)==(r.sum+r.ini)){
			if(r.ini<l.ini){
				return true;
			}
		}
		return false;
	}
};
int main()
{
    //cout << "Hello World!" << endl;
    int t,k,n,x;
    cin>>t;
    priority_queue<node,vector<node>,compare> p;
    while(t--){
    	cin>>n>>k;
    	for(int i=0;i<n;i++){
    		cin>>x;
    		p.push(node{x,x});
    		//cout<<p.top().sum<<p.top().ini<<" ";
    	}
    	while(k--){
    		node temp=p.top();
    		//cout<<temp.sum<<temp.ini<<" ";
    		p.pop();
    		temp.sum+=temp.ini;
    		p.push(temp);
    	}
    	int max=-1;
    	while(p.size()){
    		if(p.top().sum>max){
    			max=p.top().sum;
    		}
    		p.pop();
    	}
    	cout<<max<<endl;
    }
    return 0;
}
