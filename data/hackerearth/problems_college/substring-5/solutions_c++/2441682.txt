#include <iostream>
#include<stack>
using namespace std;

int main()
{
	int x=0,k,i,t=1;
	stack<char> st;
    string s;
    cin>>s;
    st.push(s[0]);
    for(i=1;i<s.length();i++)
    {
    	if(s[i]==st.top())
    	{
    		t++;
    	}
    	else
    	{
    		st.pop();
    		x+=(t*(t+1))/2;
    		st.push(s[i]);
    		t=1;
    	}
    }
    x+=(t*(t+1))/2;
    cout<<x;
}
