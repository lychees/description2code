#include<iostream>
#include<vector>
#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
    int n;
    scanf("%d",&n);
    int j=0;
    vector<int> v;
    while(j<n)
    {
        int n,rn,sn;
        scanf("%d%d%d",&n,&rn,&sn);
        for(int i=0;i<rn;i++)
        {
            int a;
            scanf("%d",&a);
            if(sn>=a)
            {
                sn++;
            }
        }
        if(sn<=n)
            cout<<sn<<endl;
        else
            cout<<"-1"<<endl;
        j++;
    }
}
