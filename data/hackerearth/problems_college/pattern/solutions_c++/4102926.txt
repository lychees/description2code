#include<iostream>
using namespace std;
int main()
{
    int n;
    cin >> n;
    for(int i=0; i<n; i++)
    {
        cout << "*";
    }
    cout << "\n";
    for(int i=n-3; i>=0; i--)
    {
        for(int j=i; j>=0; j--)
        {
            cout << " ";
        }
        cout << "*" << "\n";
    }
    for(int i=0; i<n; i++)
    {
        cout << "*";
    }
}
