#include <iostream>
#include <cmath>
using namespace std;
#define LL long long

int main()
{
    int T;
    cin >> T;
    
    while(T--)
    {
    	LL x,y,z;
    	cin >> z >> x >> y;
    	
    	int count=0;
    	
    	for(int i=1 ; i<=sqrt(z) ; i++)
    	{
    		if(z%i==0)
    		{
    			if(i<=x && y>=z/i)
    			count++;
    			
    			if(i<=y && x>=z/i)
    			count++;
    			
    			if(i==z/i)
    			count--;
    		}
    	}
    	cout << count << endl;
    }
    return 0;
}
