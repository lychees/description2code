#include <iostream>
#include <cstring>
using namespace std;

int* check_for_dislike(char word[100][100], int y, int x, char match[10][100], int p[100])
{
        int i, j;
        for(i=0; i<y; i++)
        {
                for(j=0; j<x; j++)
                {
                        if(strcmp(word[i], match[j]) == 0)
                        {
                                p[i] = 1;                               //Here 1 denotes for dislike and 0 denotes for like
                                continue;
                        }
                }
        }
}

char acronym(char word[100])
{
        char c, C;
        c = word[0];
        C = c-32;
        return C;
}

int main ()
{
        int k;                                                  //number of words harsh dislike
        cin >> k;
        char str[10][100];
        int i;
        for(i=0; i<k; i++)
        {
                cin >> str[i];
        }
        int n;                                                  //number of words in the given sentences
        cin >> n;
        char sen[100][100];
        for(i=0; i<n; i++)
        {
                cin >> sen[i];
        }
        int p[100];
        char prev = '0', curr = '0';
        check_for_dislike(sen, n, k, str, p);
        for(i=0; i<n; i++)
        {
                if(p[i] == 0)
                {
                        prev = curr;
                        curr = acronym(sen[i]);
                        if(prev != '0')
                                cout << prev << ".";
                }
        }
        cout << curr;

        return 0;
}
