typedef pair<int, int> pii;

const double eps = 1e-9;
const double pi = acos(-1.0);

const int maxn = (int)1e4 + 10;

double res[maxn];

int isPrime(int x) {
	int i = 2;
	while(i * i <= x) {
		if (x % i == 0) return 0;
		i++;
     }
     return 1;
}

int main() {
	#ifdef LOCAL
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	#endif

	int T; cin >> T;

	int cur = 0;
	int tot = 1;

	res[1] = 0.0;
	int a = 5;
	int b = 3;
	int c = 7;
	int d = 9;

	for (int i = 3; i < maxn; i += 2) {
		tot += 4;
		cur += isPrime(a) + isPrime(b) + isPrime(c) + isPrime(d);
		res[i] = cur * 100.0 / tot;
		int nb = d + i + 1;
		int na = d + 2 * (i + 1);
		int nc = d + 3 * (i + 1);
		int nd = d + 4 * (i + 1);
		a = na;
		b = nb; 
		c = nc;
		d = nd;
	}


	while(T--) {
		int x; cin >> x;
		printf("%.6f\n",res[x]);
	}
	
	return 0;
}
