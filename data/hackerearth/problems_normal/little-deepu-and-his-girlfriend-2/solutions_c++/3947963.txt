#include <iostream>
using namespace std;
void solve()
{
	int m,n;
	cin>>m>>n;
	int s[n];
	int i,j;
	for(i=0;i<n;i++)
	cin>>s[i];
	bool dp[m+1],res;
	for(i=0;i<=m;i++)
	{
	
		dp[i]=false;
	}
	for(i=0;i<=m;i++)
	{
		res=true;
		for(j=0;j<n;j++)
		{
			if(i>=s[j])
			{
				res=res&&dp[i-s[j]];
			}
		}
		if(res)
		dp[i]=false;
		else
		dp[i]=true;
	}
	if(dp[m]==true)
	cout<<"Little Deepu\n";
	else
	cout<<"Kate\n";
}
	
	

int main()
{
   int t;
   cin>>t;
   while(t--)
   solve();
    return 0;
}
