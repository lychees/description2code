#include<bits/stdc++.h>
using namespace std;
int a[1111111],t,n;
void init()
{
    a[1] = 1;
    for(int i = 2; i <= 1000;i++){
        for(int j = 2;pow(i,j) <= 1000000;j++){
            int pos = pow(i,j);
            a[pos] = 1;
        }
    }
}
bool solve()
{
    for(int i = 1;i <= n/2 ; i++)
            if(a[i] && a[n-i])
                return true;
    return false;
}
int main()
{
    memset(a,0,sizeof(a));
    init();

    cin>>t;
    while(t--){
        cin>>n;
        if (solve())
            cout<<"Yes"<<endl;
        else
            cout<<"No"<<endl;
        }
return 0;
}
