#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
int power(int x, unsigned int y)
{
	long long int temp;
    if( y == 0)
        return 1;
    temp = power(x, y/2);
    if (y%2 == 0)
        return (temp*temp)%mod;
    else
        return (x*temp*temp)%mod;
}
int main(){
	int t;
	cin>>t;
	for(int i=0;i<t;i++){
	int n;
	cin>>n;
	cout<<power(3,n)<<endl;
	}

}
