T = int(raw_input())
for i in range(0,T):
	NK = map(int,raw_input().strip().split(' '))
	N = NK[0]
	K = NK[1]
	Numbers = map(int,raw_input().strip().split(' '))
	ones_count = []
	for num in Numbers:
		ones_count.append(int(bin(num).count("1")))
	ones_count.sort()
	print sum(ones_count[N-K:N])
