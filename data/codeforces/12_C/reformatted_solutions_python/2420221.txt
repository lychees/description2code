n, m = map(int, raw_input().split())
t = sorted(map(int, raw_input().split()))
d = dict()
for i in xrange(m):
    f = raw_input()
    if f in d:
        d[f] = d[f] + 1
    else:
        d[f] = 1
s = d.values()
s.sort()
s.reverse()
mins = 0
for i in xrange(len(s)):
    mins += t[i] * s[i]
maxs = 0
t.reverse()
for i in xrange(len(s)):
    maxs += t[i] * s[i]
print mins, maxs
