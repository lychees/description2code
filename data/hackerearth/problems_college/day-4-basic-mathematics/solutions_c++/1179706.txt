#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int t; cin >> t;
    while(t--){
    	int n; cin >> n;
    	int num = (sqrt(1 + 8 * n) - 1) / 2, res;
    	if((num * (num + 1)) / 2 == n)res = num;
    	else res = num + 1;
    	//cout << res << endl;
    	if(res % 3 == 0)cout << "SL" << endl;
    	else if(res % 3 == 1)cout << "LB" << endl;
    	else cout << "BS" << endl;
    }
    return 0;
}
