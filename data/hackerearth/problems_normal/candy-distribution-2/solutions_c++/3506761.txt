#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

int main()
{
    int T;
    cin>>T;
    while(T--){
    	int N;
    	cin>>N;
    	vector<long long> arr;
    	for(int i=0;i<N;i++){
    		int a;
    		cin>>a;
    		arr.push_back(a);
    		
    	}
    	std::sort(arr.begin(),arr.end());
    	long long ans=arr[0];
    	for(int i=1;i<N;i++)
    	 ans = ans+arr[i]-1;
    	
    	cout<<ans<<"\n";
    }
    return 0;
}
