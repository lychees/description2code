#include <iostream>
using namespace std;

int main()
{
	int T;
	cin >> T;
	for(int i = 0; i < T; ++i)
	{
		string str1;
		string str2;
		cin >> str1 >> str2;
		int N1 = str1.size();
		int N2 = str2.size();
		if (N1 != N2)
		{
			cout << "No" <<endl;
		}else
		{
			int index1 = 0;
			int index2 = 0;
			bool isExist = true;
			while(index1 < N1 && index2 < N2)
			{
				if (str1[index1] == '#')
				{
					++index1;
				}else if (str2[index2] == '#')
				{
					++index2;
				}else
				{
					if (str1[index1] != str2[index2])
					{
						cout << "No" << endl;
						isExist = false;
						break;
					}else if (str1[index1] == 'A' && index2 > index1)
					{
						cout << "No" << endl;
						isExist = false;
						break;
					}else if (str1[index1] == 'B' && index2 < index1)
					{
						cout << "No" << endl;
						isExist = false;
						break;
					}else
					{
						++index1;
						++index2;
					}
				}
			}
			if (isExist)
			{
				cout << "Yes" << endl;
			}
		}
	}
    //cout << "Hello World!" << endl;
    return 0;
}
