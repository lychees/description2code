#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
   long long int k,i,j;
   cin>>k;
   long long int a[k],sum=0,b[k];
   for(i=0;i<k;i++)
   {
   	cin>>a[i];
   }
   sort(a,a+k);
   sum=0;
   for(i=0;i<k;i++)
   {
   	b[k-i-1]=a[i];
   }
   for(i=0;i<k;i++)
   {a[i]=b[i]+i+1;
   	if(a[i]>sum)
   	sum=a[i];
   }
   cout<<sum+1<<endl;
    return 0;
}
