#include <iostream>
using namespace std;
#define ll long long

int main()
{	int t;
    cin>>t;
    while(t--)
    {	ll n,x,a=0,b=0,c=0;
    	cin>>n;
    	while(n--)
    	{	cin>>x;
    		if(x%3==0)
    			++a;
    		else if(x%3==1)
    			++b;
    		else
    			++c;
    	}
    	cout<<((a*(a-1)/2)+b*c)<<"\n";
    }
    return 0;
}
