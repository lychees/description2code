#include<iostream>
#include<cstring>
using namespace std;



int main(){
	std::ios::sync_with_stdio(false);
	int size,count =0;
	cin >> size;
	
	int p[size];
	for(int i=0;i<size;i++){
		cin >> p[i];
	}
	
	while(size>2){
		int max=0,index;
		for(int k=0;k<size-2;k++){
			if(p[k]*p[k+2]>max){
				max = p[k]*p[k+2]; 
				index = k+1;
			}
		}
		count = count + max;
		for(int i=index;i<size;i++){
			p[i] = p[i+1];
	}	
		size--;
		
		
		
	}
	cout << count ;
}
