#include <iostream>
using namespace std;
int bits(long long n)
{
	int i ; 
	while(n)
	{
		if(n&1)
		i++ ; 
		n/=2;
	}
	return i ;
}
int main()
{
    int t ; 
    long long n ;
    scanf("%d",&t) ; 
    while(t--)
    {
    	scanf("%lld",&n) ; 
    	int count = bits(n) ; 
    	printf("%lld\n",count) ; 
    }
    return 0;
}
