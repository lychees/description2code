#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long int n,*a,*b;
    cin>>n;
    a= new long long int [n];
    b = new long long int [n];
    for(int i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    b[0]=a[0];
    cout<<a[0]<<" ";
    for(int i=1;i<n;i++)
    {
    	long long int c=a[i]*(i+1);
    	for(int j=0;j<i;j++)
    	{
    		c-=b[j];
    	}
    	b[i]=c;
    	cout<<c<<" ";
    }
    
    return 0;
}
