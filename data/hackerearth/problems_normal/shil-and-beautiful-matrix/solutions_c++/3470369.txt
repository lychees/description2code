#include<bits/stdc++.h>
using namespace std;
const int mod = 1000000000 + 7;
long long fact[1000011];
long long coun[1030]={0};
int main()
{
	long long n,m,i,j,k;
	cin>>n>>m;
       fact[0]=1;
	for(i=1;i<=n*m;i++)
	{
         fact[i]=(i*fact[i-1])%mod;
	}
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=m;j++)
		{
			coun[i^j]++;
		}
	}
									
     long long ans=1;
	for(i=0;i<=1029;i++)
	{
		ans=(ans*fact[coun[i]])%mod;
	}

         cout<<ans<<endl;


}
