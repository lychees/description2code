#include<bits/stdc++.h>
using namespace std;
#define ll long long 
void _divisible()
{
	ll a,b,c,d;
	scanf("%lld",&a);
	scanf("%lld",&b);
	scanf("%lld",&c);
	scanf("%lld",&d);
	for(int i=2;i*i<=c;i++)
	{
		ll p=0,q=0;
		while(a%i==0)
		{
			p++;
			a/=i;
		}
		while(c%i==0)
		{
			q++;
			c/=i;
		}
		if(p*b<q*d) 
		{
			cout<<"Not divisible"<<endl;
			return;
		}
	}
	if(c>1)
	{
		ll p=0;
		while(a%c==0)
		{
			p++;
			a/=c;
		}
		if(p*b<d)
		{
			cout<<"Not divisible"<<endl;
			return;
		}
	}
	cout<<"Divisible"<<endl;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		_divisible();
	}
}
