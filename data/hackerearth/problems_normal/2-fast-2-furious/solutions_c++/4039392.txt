#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
    int n;
    cin>>n;
    long long last;
    cin>>last;
    long long max=0;
    for(int i=1;i<n;i++)
    {
    	long long temp;
    	cin>>temp;
    	long long dif=abs(temp-last);
    	if(dif>max)
    	max=dif;
    	last=temp;
    }
    cin>>last;
    long long max1=0;
    for(int i=1;i<n;i++)
    {
    	long long temp;
    	cin>>temp;
    	long long dif=abs(temp-last);
    	if(dif>max1)
    	max1=dif;
    	last=temp;
    }
    if(max>max1)
    {
    	cout<<"Dom"<<endl<<max<<endl;
    }
    else if(max<max1)
    {
    	cout<<"Brian"<<endl<<max1<<endl;
    }
    else
    {
    	cout<<"Tie"<<endl<<max<<endl;
    }
    
    return 0;
}
