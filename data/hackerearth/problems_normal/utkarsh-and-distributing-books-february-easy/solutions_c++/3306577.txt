#include<iostream>
#include<stdio.h>
using namespace std;

int main()
{
    int T,i;
    int N,num,sum,mi;
    scanf("%d",&T);
    while(T--)
    {
        sum=0;
        scanf("%d",&N);
        scanf("%d",&num);
        mi = num;
        sum+=num;
        for(i=1;i<N;i++)
        {
            scanf("%d",&num);
            if(mi>num)
            {
                mi = num;
            }
            sum+=num;
        }
        printf("%d %d\n",(mi-1),(sum-N));
    }
    return 0;
}
