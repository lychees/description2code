#include <iostream>
using namespace std;

int main()
{
    int t; cin>>t;
    while(t--)
    {
    	int x,y,z,t; cin>>x>>y>>z>>t;
    	int a = min(max(x,y),max(z,t));
    	int b = max(min(x,y),min(z,t));
    	if(a>b) cout<<"Nothing\n";
    	else if(a==b) cout<<"Point\n";
    	else cout<<"Line\n";
    }
}
