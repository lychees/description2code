#include <bits/stdc++.h>
using namespace std;
#define MOD 1000003

long long int arr[MOD+1];
void funt()
{	arr[0] = 1;
	for(int i=1;i<= MOD;i++)
	{
		arr[i] = (arr[i-1]*(long long)i)%MOD;
	}
}
int main()
{
	int t; cin >> t;
	funt();
	while(t--)
	{
		long long int n , x; cin >> n >> x;
		x = x % MOD;
		
		if(n > MOD) n = MOD;
		cout << (x*arr[n])%MOD << endl;
		
	}
}
