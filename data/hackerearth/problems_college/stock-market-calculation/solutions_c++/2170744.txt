#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t;
   scanf("%d", &t);
    while(t--){
   	int n;
   	scanf("%lld", &n);
   	int a[n];
   	for(int i=0; i<n;i++){
   		scanf("%d", &a[i]);
   	}
   	int max[n];
   	max[n-1]=a[n-1];
   	for(int i=n-2; i>=0;i--){

    if (a[i] > max[i +1])
        max[i] = a[i];
    else
        max[i] = max[i + 1];

   	}
   	
   	int result = 0;
    for (int i = 0; i < n; i++) {
    if (a[i] < max[i]) {
        
        result += max[i] - a[i]; 
    }
    
   	
   	
   }
   printf("%lld\n", result);
    }
    return 0;
}
