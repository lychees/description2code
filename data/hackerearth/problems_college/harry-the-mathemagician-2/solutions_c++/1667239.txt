#include <iostream>
using namespace std;

int main()
{
	int n;
	cin>>n;
	
	while(n--)
	{ int m=0,n=0,c=0,c1;
		long long a,b;
		cin>>a>>b;
		c1=a-b;
		if((a-b)==1||b==1)
		{
			if((a%2)==1)
			cout<<"odd\n";
			else 
		   cout<<"even\n";
		}
		else{
		while(a/2>1)
		{
			a=a/2;
			m+=a;
		}
          	while(b>1)
		{
			b=b/2;
			n+=b;
		}	
		  	while(c1/2>1)
		{
			c1=c1/2;
			c+=c1;
		}
		if(m>(n+c))
		cout<<"even\n";
		else cout<<"odd\n";
		}
	}
    return 0;
}
