#include<cstdio>
#include<iostream>
#include<queue>
#include<vector>
#include<algorithm>
using namespace std;
char st[1000003];
int main()
{
	
	int ntc; cin>>ntc;
	while(ntc--)
	{  int N;
		cin>>N;
		scanf("%s",&st);
		int a1 = 0,a2=0;
		for(int i=0;i<2*N;i+=2)
		{
		if(st[i]=='B') a1++;	
		}
		for(int i=1;i<2*N;i+=2)
		{
		if(st[i]=='B') a2++;	
		}
		
		int ans = a1; if(ans>a2) ans= a2;
		cout<<ans<<"\n";
		
	}
	 
	 
	 
}
