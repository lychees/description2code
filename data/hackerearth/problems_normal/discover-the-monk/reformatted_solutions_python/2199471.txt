def binary_search_recursive(listi, low, high, x):
    mid = low + (high - low) / 2
    if low <= high:
        if listi[mid] == x:
            return True
        elif listi[mid] < x:
            return binary_search_recursive(listi, mid + 1, high, x)
        else:
            return binary_search_recursive(listi, low, mid - 1, x)
    return False


inp = [int(i) for i in raw_input().split()]
N = inp[0]
Q = inp[1]
A = sorted([int(i) for i in raw_input().split()])
for i in range(Q):
    x = input()
    if binary_search_recursive(A, 0, len(A) - 1, x):
        print "YES"
    else:
        print "NO"
