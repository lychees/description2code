#include <iostream>
using namespace std;

int main()
{
	ios::sync_with_stdio(false);
    int n,i;
    cin>>n;
    int x=0;
    for(i=4;i<=n;i++)
    {
    	x=0;
    	for(int j=2;j<i;j++)
    	{
    		if(i%j==0)
    		{
    			x=1;
    			break;
    		}
    	}
    	if(!(x^1))
    	{
    		cout<<i<<endl;
    	}
    	
    }
    return 0;
}
