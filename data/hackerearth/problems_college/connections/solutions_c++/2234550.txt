#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int a[n];
    int f;
    cin>>f;
    while(f--){
    	int b,c;
    	cin>>b;
    	cin>>c;
    	a[b]=c;
    }
    int q;
    cin>>q;
    for(int i=0;i<q;i++){
    	int y,z;
    	cin>>y>>z;
    	if(a[y]==z)
    	cout<<"Direct Connection"<<endl;
    	else if(a[a[y]]==z||a[a[a[y]]]==z||a[a[a[a[y]]]]==z||a[a[a[a[a[y]]]]]==z)
    	cout<<"Indirect Connection"<<endl;
    	else 
    		cout<<"No Connection"<<endl;
    }
    return 0;
}
