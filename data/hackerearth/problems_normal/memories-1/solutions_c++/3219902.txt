#include <bits/stdc++.h>

using namespace std;

#define pb push_back
#define mp make_pair
#define ll long long int

#define mod 1000000007
#define MX 100010
#define MX1 15
#define get getchar//_unlocked

int main()
{
    ll cnt, f, k, t, n, i, p, j, m, a, b;
    ll exist[MX1][MX1];
    cin >> t;
    while (t--) {
        memset(exist, 0, sizeof(exist));
        cin >> n >> p;
        for(i = 0; i < p; i++) {
            cin >> a >> b;
            a--;
            b--;
            exist[a][b] = 1;
            exist[b][a] = 1;
        }
        cnt = 0;
        for(i = 1; i < (1 << n); i++) {
            f = 0;
            for(j = 0; j < n; j++) {
                if(i & (1 << j)) {
                    for(k = 0; k < j; k++) {
                        if(i & (1 << k)) {
                            if(exist[k][j]) {
                                f = 1;
                                break;
                            }
                        }
                    }
                }
                if(f) {
                    break;
                }
            }
            if(!f) {
                cnt++;
            }
        }
        cout << cnt << endl;
    }

    return 0;
}
