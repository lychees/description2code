#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
          int n,i,j;
          scanf("%d",&n);
          int a[n],b[n],c[n];
          for(i=0;i<n;i++)
          {
                    scanf("%d %d",&a[i],&b[i]);
                    c[i]=a[i]+b[i];
          }
          sort(c,c+n);
          for(i=0;i<n;i++)
          {
                    for(j=0;j<n;j++)
                    {
                              if(a[j]+b[j]==c[i])
                              {
                                        a[j]=0;
                                        b[j]=0;
                                        printf("%d ",j+1);
                              }
                    }
          }
          return 0;
}
