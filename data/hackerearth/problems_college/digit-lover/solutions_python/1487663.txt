import sys

def Get_Sum(s) :
    answer = 0
    while s :
        answer += s%10
        s /= 10
    return str(answer)

def Recur_Check(s) :
    if len(s) == 1 :
        return s
    else :
        return Recur_Check((Get_Sum(int(s))))
