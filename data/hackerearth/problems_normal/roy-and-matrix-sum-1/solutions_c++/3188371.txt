#include <iostream>
using namespace std ;
int main()
{

    int t ;
    cin >> t ;
    while ( t--)
    {
       long long  int n ;
        cin >> n;
        long long int sum = 0;
        long long i ;
        for ( i =0 ; i<= n ; i++)
        {
            sum = sum+i*(n-i) ;
        }
        cout << 2*sum <<endl;

    }
    return 0 ;
}
