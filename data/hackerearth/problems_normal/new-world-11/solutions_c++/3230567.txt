ll bisearch(ll a[],ll lo,ll hi,ll c,ll n)
{ll mid,an,i,su;
while(lo<hi){
	mid=(lo+hi)/2;
	su=0;
	an=0;
	f(i,2,n+1)
	{
		if(su+a[i]-a[i-1]<=mid)
		su+=a[i]-a[i-1];
		else{
		
		su=a[i]-a[i-1];
		an++;
	
		}
	}
an++;
	if(an<=c)
	hi=mid;
	
	else
	lo=mid+1;
	
	}
//return mid;	
	an=0;su=0;
	f(i,2,n+1)
	{
		if(su+a[i]-a[i-1]<=mid)
		su+=a[i]-a[i-1];
		else{
		
		su=a[i]-a[i-1];
		an++;
	
		}
	}
an++;
	if(an<=c)
	return  mid;
	else return 1+mid;
 
 
}
int main()
{

	ll n,c,i,j,k,a[100009],ans;
