import sys
s=raw_input()
s=str(s)

d=[]
if s[0]!=" ":
	if ord(s[0])>=92:
		d.append(chr(ord(s[0])-32))
		
	
	else:
		d.append(s[0])
		
		
		

for j in xrange(1,len(s)):
	if s[j]!=" " and s[j-1]==" ":
		if ord(s[j])>=92:
			d.append(chr(ord(s[j])-32))
		else:
			d.append(s[j])
	else:
		d.append(s[j])
print "".join(d)
