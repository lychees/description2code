num = int(raw_input())
song = raw_input()
students = list(range(1,num+1))
ballpointer = 0
# problem
while len(students) > 1:
    for lyrics in song:
        if len(students) == 1:
            break
        if lyrics == 'b':
            del students[ballpointer]
        if lyrics == 'a':
            ballpointer += 1
        if ballpointer == len(students):
            ballpointer = 0
print(students[0])
