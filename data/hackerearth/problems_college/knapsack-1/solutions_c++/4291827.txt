#include<bits/stdc++.h>
using namespace std;
int main() 
{ 	int t;
	cin>>t;
	while(t--)
	{
		int i,n,k,a[100001];
		cin>>n>>k;
		for(i=0;i<n;i++)cin>>a[i];
		int j,dp[100001]={0};
		//dp[j]=1, if it is possible to make sum=j , else =0 
		dp[0]=1;
		for(i=0;i<n;i++)dp[a[i]]=1;
		
		for(j=1;j<=k;j++)
		   {
		   	for(i=0;i<n;i++)
		   	    {   
		   	    	if(j-a[i]>=0 && dp[j-a[i]]==1)
		   	    	  dp[j]=1;
		   	    
		   	    }
		   	   //update
				  //for(i=0;i<n;i++)dp[a[i]]=1; 
		   }
		   
		 for(j=k;j>=0;j--)
		    {
		    	if(dp[j]==1)
		    	   {
		    	   	cout<<j<<endl;
		    	   	break;
		    	   }
		    }  
	}
	return 0; 
}
