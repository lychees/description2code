def solve(S):
	uct=sct=a=b=0
	for ix,v in enumerate(S):
		if v=='U':
			uct += 1#USUSU..
			a += ix%2
			b += ~ix%2
		if v=='S':
			sct += 1 #SUSUS..
			a += ~ix%2
			b += ix%2
	if uct==sct+1: return a/2
	if sct==uct+1: return b/2
	if uct==sct: return min(a/2,b/2)
	return -1
rr = raw_input
for tc in xrange(int(rr())):
	print solve(rr())
