#include <iostream>
#include<algorithm>
using namespace std;

int main() {
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		cin>>n;
		int a[n];
		for(int i=0;i<n;i++) cin>>a[i];
		sort(a,a+n);
		int max=0,min=1000000000;
		for(int i=0;i<n/2;i++)
		{
			int x=a[i]+a[n-i-1];
			if(x> max) max=x;
			if(x<min) min=x;
		}
		cout<<max-min<<endl;
	}
	return 0;
}
