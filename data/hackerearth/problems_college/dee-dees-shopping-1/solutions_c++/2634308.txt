#include<iostream>
using namespace std;
int main(){
int t,n,m=999999;
cin>>t;
while(t--){
    cin>>n;
    for(int i=1;i<n;i++){
        int temp = n - 8*i;
        if(temp<0){
            break;
        }
        if(temp%6==0){
            temp=i+temp/6;
        if(temp<m){
            m=temp;
        }
        }
    }
    if(m==999999)
    cout<<"-1"<<endl;
    else
    cout<<m<<endl;
    m=999999;
}
return 0;
}
