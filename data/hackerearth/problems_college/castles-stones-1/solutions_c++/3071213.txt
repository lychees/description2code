#include "bits/stdc++.h"

using namespace std;

int main () {
    //freopen("in.txt", "r", stdin);
    int t; cin >> t;
    while (t--) {
        int cap; cin >> cap;
        int N; cin >> N;
        int a[N];
        
        for (int i = 0; i < N; i++) cin >> a[i];
        
        sort(a, a + N);
        reverse(a, a + N);
        
        int ans = 0;
        for (int i = 0; i < N; i++) {
        	if (a[i] <= cap) {
        		if (a[i] == cap) {
        			ans++;
        			continue;
        		} else if (a[i] + a[N-1] <= cap) {
        			ans++;
        			N--;
        		} else ans++;
        	}
        }
        
        cout << ans << endl;
    }
    return 0;
}
