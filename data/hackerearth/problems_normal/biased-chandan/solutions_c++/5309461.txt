#include <iostream>
#include <stack>

using namespace std;

int main()
{
	int n,i,a,sum=0;
	stack <int> rating;
	
	cin >> n;
	for(i=0;i<n;i++)
	{
		cin >> a;
		if ( a==0)
		{
			if(!rating.empty())
			rating.pop();
		}
		else rating.push(a);
	}
	while(!rating.empty()){
	sum+=rating.top();
	rating.pop();
}
	cout << sum;
	return 0;
}
