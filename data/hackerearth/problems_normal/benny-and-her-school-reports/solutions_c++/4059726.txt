#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int N;
    	long M,X;
    	cin>>N>>M>>X;
    	long sum=0,temp;
    	for(int i=0;i<N;i++)
    	{
    		cin>>temp;
    		sum+=temp;
    	}
    	long diff = sum - ((N+1)*X);
    	if(diff>=0)
    	{
    		cout<<"1\n";
    	}else
    	{
    		diff*=-1;
    		if(diff>M)
    		{
    			cout<<"Impossible\n";
    		}else
    			cout<<diff<<"\n";
    	}
    }
    return 0;
}
