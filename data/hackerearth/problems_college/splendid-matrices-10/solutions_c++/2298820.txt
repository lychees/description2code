#include <iostream>
#include<math.h>
using namespace std;
int k=0,n,a[1024][1024];
void assignvalue(int rlow,int clow,int rhigh, int chigh)
{   //     cout<<rlow<<"\t"<<clow<<"\t"<<rhigh<<"\t"<<chigh<<"\t"<<k<<"\t"<<endl;

    int i,j;
    int p=(rlow+rhigh)/2 , q=(clow+chigh)/2;
    if(rhigh-rlow<=2)
    {
        a[rlow-1][clow-1]=++k;
        a[rlow-1][chigh-1]=++k;
        a[rhigh-1][clow-1]=++k;
        a[rhigh-1][chigh-1]=++k;
        return;
    }
    else{
    assignvalue(rlow,clow,p,q);
    assignvalue(rlow,q+1,p,chigh);
    assignvalue(p+1,clow,rhigh,q);
    assignvalue(p+1,q+1,rhigh,chigh);
    }
}
int main()
{
    int i,j;
    cin>>n;
    n=pow(2,n);
 assignvalue(1,1,n,n);
    for(i=0;i<n;i++)
    {
    for(j=0;j<n;j++)
    {
        cout<<a[i][j]<<" ";
    }
    cout<<endl;
    }
    return 0;
}
