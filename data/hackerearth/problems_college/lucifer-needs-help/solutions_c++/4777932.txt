#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long int a[1005],i,l,r,flag=0,s,x,id1,id2,n;
    scanf("%lld%lld",&n,&s);
    for(i=0;i<n;i++)
    {
    	scanf("%lld",&a[i]);
    }
    sort(a,a+n);
    for(i=0;i<n;i++)
    {
    	x=s-a[i];
    	l=0;
    	r=n-1;
    	while(l<=r)
    	{
    	//	cout<<l<<" "<<r<<endl;
    	//	cout<<a[l]<<" "<<a[r]<<endl;
    		if((a[l]+a[r])==x)
    		{
    			id1=l;
    			id2=r;
    			if(id1!=i&&id2!=i)
    			{
    				flag=1;
    			}
    			else
    			{
					l++;
					r--;
    			}
    		}
    		else if((a[l]+a[r])>x)
    		{
    			r--;
    		}
    		else
    		{
    			l++;
    		}
    		if(flag)
    		{
    			break;
    		}
    	}
    	if(flag)
    	{
    		break;
    	}
    }
    if(flag)
    {
    	printf("YES\n");
    }
    else
    {
    	printf("NO\n");
    }
    return 0;
}
