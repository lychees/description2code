#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int a[n];
    	for(int i=0;i<n;i++)
    	cin>>a[i];
    	int chk=0;
    	for(int i=0;i<n;i++)
    	{
    		if((i+1)!=a[a[i]-1])
    		{
    			chk=1;
    			break;
    		}
    	}
    	if(chk==0)
    	cout<<"inverse"<<endl;
    	else
    	cout<<"not inverse"<<endl;
    }
    return 0;
}
