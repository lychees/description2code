#include <bits/stdc++.h>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    string s;
    getline(cin,s);

    string s1=R"(?username=)";
    string s2=R"(&pwd=)";
    string s3=R"(&profile=)";
    string s4=R"(&role=)";
    string s5=R"(&key=)";

    auto pos = s.find(s1);
    auto pos2 = s.find(s2);
    string ss="";
    for(int i=pos+10;i<pos2;i++)
    {
    	ss+=s[i];
    }
    cout<<"username: "<<ss<<"\n";


    pos = s.find(s2);
    pos2 = s.find(s3);
    ss="";
    for(int i=pos+5;i<pos2;i++)
    {
    	ss+=s[i];
    }
    cout<<"pwd: "<<ss<<"\n";



    pos = s.find(s3);
    pos2 = s.find(s4);
    ss="";
    for(int i=pos+9;i<pos2;i++)
    {
    	ss+=s[i];
    }
    cout<<"profile: "<<ss<<"\n";



   pos = s.find(s4);
    pos2 = s.find(s5);
    ss="";
    for(int i=pos+6;i<pos2;i++)
    {
    	ss+=s[i];
    }
    cout<<"role: "<<ss<<"\n";


    pos = s.find(s5);
    s3="";
    for(int i=pos+5;s[i]!='\0';i++)
    {
    	s3+=s[i];
    }
    cout<<"key: "<<s3<<"\n";

    return 0;
}
