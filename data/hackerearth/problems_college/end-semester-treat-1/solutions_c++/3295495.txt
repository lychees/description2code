#include <bits/stdc++.h>

using namespace std;

#define pb push_back
#define mp make_pair
#define ll long long int

int main()
{

    ll A[10005];
    ll B[10005];
    int t;
    cin >> t;
    int n, m, i, j;
    while(t--) {
        int mainflag = 1;
        cin >> n;
        for (i = 1; i <= n; i++)
            cin >> A[i];
        cin >> m;
        for(i = 1;i <= n; i++)
            B[i] = 0;
        int o[n+1][m+1], r[n+1][m+1];

        for(j = 1;j <= m; j++) {
            o[0][j] = 1;
            for(i = 1; i <= n; i++)
            {
                cin >> o[i][j];
                B[i] += o[i][j];
            }
        }
        for(j = 1; j <= m; j++) {
            for(i = 1; i <= n;i++) {
                cin >> r[i][j];
            }
        }
        int tmp = m;
        while(tmp)
        {
            int fl = tmp;
            for(j = 1; j <= m;j++)
            {
                if(o[0][j] == 1)
                {
                    int flag2 = 1;
                    for(i = 1; i <= n; i++)
                    {
                        if((A[i] + o[i][j]) < r[i][j])
                        {
                            flag2 = 0;
                            break;
                        }
                    }
                    if(flag2!=0)
                    {
                        for(i = 1; i <= n;i++)
                        {
                            B[i] -= o[i][j];
                            A[i] += o[i][j];
                        }
                        tmp--;
                        o[0][j] = 0;
                    }
                }
            }

            if(fl == tmp)
            {
                cout << "NO\n";
                mainflag = 0;
                break;
            }
        }
        if(mainflag!=0)
            cout << "YES\n";
    }
    return 0;
}
