#include<iostream>
#include<string.h>

using namespace std;

#define MIN(a,b) ((a)<(b) ? (a) : (b))




int pplMask[10000];
char in[1000];

int main()
{
    
    int i,j,k;

    int t,n;
    cin>>t;
    while(t--)
    {
      cin>>n;

      int len;
      for(i=0;i<n;i++)
      {
        cin>>in;
        len=strlen(in);
        pplMask[i]=0;
        for(j=0;j<len;j++)
        {
          if(in[j]=='Y')
            pplMask[i]+=1<<j;
        }
      }
     


      int mn=len;
      for(i=0;i<(1<<len);i++)
      {
        for(j=0;j<n;j++)
        {
          if( (pplMask[j] & i)==0)
            break;
        }

        if(j==n)
       {
          mn=min(mn,__builtin_popcount(i));
       }
      }
      cout<<mn<<endl;
    }


    return 0;
}
