def fib():
    a = 1
    b = 2
    ans = [a, b]
    while a < 100+10**15:
        ans.append(a+b)
        a, b = b, a+b

    return ans


fibs = fib()

t = input()
while t:
    t -= 1
    n = input()
    ans = n
    i = 0
    while i < len(fibs):
        if (fibs[i] <= ans):
            ans += 1
        else:
            break
        i += 1
    print ans
