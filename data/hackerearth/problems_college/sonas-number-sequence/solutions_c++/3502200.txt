#include <iostream>
using namespace std;

int main()
{
    int N,i, arr[100], farr[100], sum=0;
    cin>>N;
    for(i=0;i<N;i++) {
    	cin>>arr[i];
    }
    
    int k=0;
    farr[k]=arr[k];
    sum+=arr[k];
    k++;
    for(i=1;i<N;i++) {
    	farr[i]=((i+1) * arr[i]  ) - sum;
    	sum+=farr[i];
    }
    
    for(i=0;i<N;i++) {
    	cout<<farr[i]<<" ";
    }
}
