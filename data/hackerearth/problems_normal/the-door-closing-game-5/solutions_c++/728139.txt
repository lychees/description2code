#include<stdio.h>
#include<vector>
#include<string.h>
using namespace std;
int main()
{
    int T;
    scanf("%d",&T);
    while(T--)
    {
        char arr[110];
        int i,cnt,nim=0;
        vector<int> v;
        scanf("%s",&arr);
        for(i=0;arr[i]!='\0';i++)
        {
             cnt=0;
             if(arr[i]=='_')
             {
                cnt++;
                i++;
                while(arr[i]=='_')
                {
                    cnt++;
                    i++;
                }
                v.push_back(cnt);
                i--;
             }
        }
        for(i=0;i<v.size();i++)
        {
             //printf("%d\n",v[i]);
             nim=nim^v[i];
        }
        if(nim==0)printf("Bob\n");
        else 
        {
                                  if(v.size()==1&&v[0]==strlen(arr))printf("Bob\n");
                                  else printf("Alice\n");
        }
        
    }
    return 0;
}
