#include <iostream>
using namespace std;
long int change(long int *a,long int n,long int m)
{
	long int dp[n+1][m],i,j,x,y;
	for(i=0;i<m;i++)
	{
		dp[0][i]=1;
	}
	for(i=1;i<=n;i++)
	{
		for(j=0;j<m;j++)
		{
			x=(i-a[j]>=0)?dp[i-a[j]][j]:0;
			y=(j>=1)?dp[i][j-1]:0;
			dp[i][j]=x+y;
		}
	}
	return dp[n][m-1];
}
int main()
{
  long int n,i,m;
  cin>>n>>m;
  long int a[n];
  for(i=0;i<m;i++)
  {
  	cin>>a[i];
  }
  cout<<change(a,n,m)<<endl;
    return 0;
}
