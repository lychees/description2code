#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;
#define period 10

int main() {
    
    int n,q,l,r,k;
    int fact,rht,i,p,diff,t;
    cin>>n>>q;
    int * input= new int [n+1];
    long int **ksum;
    ksum=new long int* [n+1];
    ksum[0]=new long int[11];
    for(i=1;i<=n;i++)
        {
            cin>>input[i];
            ksum[i]=new long int [11];
        }
        
    for(p=n;p>=1;p--)
        {
           for(k=1;k<=period;k++)
           ksum[p][k] =(p+k)>n ? input[p]:input[p]+ksum[p+k][k];
        }
    
    for(t=1;t<=q;t++)
        {
           cin>>l>>r>>diff;
           fact=(r-l)/diff;
           rht=l+fact*diff;
           //if(rht>l)
           cout<<ksum[l][diff]-ksum[rht][diff]+input[rht]<<endl;
        }
    
    //gdfgd
    
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */   
    return 0;
}
