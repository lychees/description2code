#include <iostream>
using namespace std;

int main()
{
    int h,m;
    float angle,hr,mn;
    scanf("%d:%d",&h,&m);
    if(h<0||m<0||h>12||m>60)
    return 0;
    else
    {
        if(h==12)
        h=0;
        if(m==60)
        m=0;
        hr=0.5*(h*60+m);
         mn=6*m;     
        angle=(hr>mn)?hr-mn : mn-hr;
        if(360-angle<angle)
        angle=360-angle;
        printf("%0.6f",angle);        
    }   
    return 0;
}
