import math

noOfCases = int(raw_input())
caseM = []
caseN = []
for i in range(0, noOfCases):
    caseValues = raw_input().split(" ")
    caseM.append(int(caseValues[0]))
    caseN.append(int(caseValues[1]))

for i in range(0, noOfCases):
    m = caseM[i]
    n = caseN[i]
    if m < n-1:
        print -1
    else:
        result = math.factorial(m) * math.factorial(m +
                                                    1) / math.factorial(m+1-n)
        print result % 1000000007
