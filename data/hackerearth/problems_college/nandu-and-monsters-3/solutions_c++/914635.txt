#include <iostream>
using namespace std;
int main()
{

    int n;int s;int i;int j;
    cin>> n;
    cin>> s;
    char  a[100005];
    scanf("%s", a);
    for(int p=1;p<=s;p++)
    {
        cin>> i;
        cin>> j;
        if(a[j-1]=='1'||a[i-1]=='1')
            cout<<"No"<<endl;
        else if(i==j)
        {
            if(a[i-1]=='1')
                cout<< "No" <<endl;
            else
                cout<< "Yes" <<endl;
        }
        else
        {
            while((i-1)!=0||(j-1)!=0)
            {
                if(j<=i)
                {
                   i=i/2;
                }
                else
                    j=j/2;
                if(i==j)
                {
                    cout<< "Yes" <<endl;
                    break;
                }
                else if(a[i-1]=='1'||a[j-1]=='1')
                {
                    cout<< "No" <<endl;
                    break;
                }

            }
        }
    }
    return 0;
}
