#include<math.h>
#include<iostream>
using namespace std;
int A[1000010];
int main()
{
    int n,k;
    cin>>n>>k;
    for(int i=0;i<n;i++)
    {
        cin>>A[i];
    }
    int count=0;
    for(int i=0;i+k<n;i++)
    {
        if(((A[i]-A[i+k])%k==0 )||((A[i]+A[i+k])%k==0 ))
            count++;
    }
    cout<<count<<endl;
    return 0;
}
