#include <bits/stdc++.h>
using namespace std;

int mark[20][20],pcell[20][20],ans,n;

void rec(int i,int j){
	if(i==n && j==n){
		ans++;
		return;
	}
	mark[i][j]=1;
	if(i+1<=n && pcell[i+1][j]==0 && mark[i+1][j]==0) rec(i+1,j);
	if(j+1<=n && pcell[i][j+1]==0 && mark[i][j+1]==0) rec(i,j+1);
	if(i-1>=1 && pcell[i-1][j]==0 && mark[i-1][j]==0) rec(i-1,j);
	if(j-1>=1 && pcell[i][j-1]==0 && mark[i][j-1]==0) rec(i,j-1);
	mark[i][j]=0;
	return;
}

int main()
{
    int t;
    cin>>t;
    while(t--){
    	cin>>n;
    	for(int i=1;i<=n;i++)
    		for(int j=1;j<=n;j++){
    			cin>>pcell[i][j];
    			mark[i][j]=0;
    		}
    	ans=0;
    	rec(1,1);
    	cout<<ans<<endl;
    }
    return 0;
}
