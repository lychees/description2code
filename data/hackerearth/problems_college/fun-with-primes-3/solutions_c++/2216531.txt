#include <bits/stdc++.h>
using namespace std;

int is[1000006];

void gen()
{
	memset(is, 1, sizeof(is));
	is[0] = is[1] = 0;
	for(int i = 2; i * i < 1000006; i++) {
		if(is[i]) {
			for(int j = i * i; j < 1000006; j += i) {
				is[j] = 0;
			}
		}
	}
}

int main()
{
    gen();
    int t;
    cin >> t;
    while(t--) {
    	int n;
    	cin >> n;
    	int sum = 0;
    	while(n) {
    		sum += (n % 10);
    		n /= 10;
    	}
    	if(is[sum]) {
    		puts("YES");
    	}
    	else {
    		puts("NO");
    	}
    }
    return 0;
}
