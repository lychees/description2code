#include <bits/stdc++.h>

using namespace std;
int a[1000];
int b[1000];
int main() {
    int t;
    cin>>t;
    while(t--) {
        int n;
        cin>>n;
        for(int i = 0; i < n; i++) {
            cin>>a[i];
        }
        int mx1 = INT_MIN;
         for(int i = 0; i < n; i++) {
            cin>>b[i];
            if( b[i] > mx1) {
                mx1 = b[i];
            }
        }
        sort(a+1, a+n);
        sort(b , b+n);
        reverse(b, b+n);
        int flag = 0;
        for(int i = 1; i < n; i++) {
            int temp = a[i] * b[i];
            //cout<<temp<<endl;
            if( temp >= (a[0] * mx1)) {
                flag = 1;
            }
        }
        if( flag == 1) {
            cout<<"NO"<<endl;
        } else {
            cout<<"YES"<<endl;
        }
    }


}
