#include <iostream>
using namespace std;

int check(char W, char M[], int N)
{
	for(int i = 0; i < N; i++)
	{
		if(W == M[i])
		{
			return 0;
		}
	}
	
	return ((int)W-96);
}

int main()
{
    int t;
    int weight = 0;
    int flag = 0;

    cin>>t;
    
    string W;
    int N;
    
    while(t--)
    {
    	cin>>W;
    	cin>>N;
    	
    	char *M = (char*)malloc(N*sizeof(char));
    	
    	for(int i=0; i< N; i++)
    	{
    		cin>>M[i];
    	}
    	
    	int k = 0;
    	while(W[k] != '\0')
    	{
    		weight += check(W[k],M,N);
    		k++;

    	}
    	
    	cout<<weight<<endl;
    	weight = 0;
    }
}
