#include <iostream>
using namespace std;
int main()
{
	long long k,pow[34],sum[34];
	int i,t;
	pow[0]=1;
	for(i=1;i<=33;i++) pow[i]=pow[i-1]*3;
	sum[0]=pow[0];
	for(i=1;i<=33;i++) sum[i]=pow[i]+sum[i-1];
	cin>>t;
	while(t--)
	{
		cin>>k;
		i=33;
		while(pow[i]>k) i--;
		CONT:
		if(i==0)
		{
			if(k==1) cout<<'2'<<endl;
			else if(k==2) cout<<'3'<<endl;
			else cout<<'4'<<endl;
			goto END;
		}
		else if((3*pow[i]+sum[i-1])<=k) {cout<<'4';k-=3*pow[i];}
		else if((2*pow[i]+sum[i-1])<=k) {cout<<'3';k-=2*pow[i];}
		else if((pow[i]+sum[i-1])<=k) {cout<<'2';k-=pow[i];}
		i--;
		goto CONT;
		END:
		;

	}
}
