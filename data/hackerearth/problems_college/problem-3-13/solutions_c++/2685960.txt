#include <iostream>
#define max 1000
using namespace std;
int main()
{
    int i,j,k,w,r,c,cnt=0;
    int a[1000][1000];
    int temp,m,n;
    //cout<<"enter the no of vertices";
    cin>>r;
    int countmax=0;
    //cout<<"enter the no of edges";
    cin>>c;
    for(i=0;i<r;i++)
    {
    	for(j=0;j<c;j++)
       {cin>>temp;
       	 a[i][j]=temp;
    }
    
    }
    for(i=0;i<r;i++)
    {
    	for(j=0;j<c;j++)
       {temp=a[i][j];
       m=1;
       n=1;
       while(temp==a[i+m][j])
       {
       	m++;
       	cnt++;
       }
       while(temp==a[i-1+m][j+n])
       {
       	n++;
       	cnt++;
       }
    
    if(cnt+1>countmax)
    {
    	countmax=cnt+1;
    	
    }
    cnt=0;
    }
    }
    cout<<countmax;
}
