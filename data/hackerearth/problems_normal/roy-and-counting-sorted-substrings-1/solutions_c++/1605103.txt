#include <iostream>
using namespace std;

int main()
{
    // cout << "Hello World!" << endl;
    int t ; 
    
    string s ; 
    cin >> t ; 
    int length ; 
    int i; 
    while(t-- )
    {
    	cin >> length ;
    	cin >> s ;
    	long long int ans = 1 , j  = 1 ;
    	for( i = 1 ; i < length ; i ++)
    	{
    		if(s.at(i) >= s.at(i-1))
    		{
    			j++ ; 
    			ans+= j ; 
    		}
    		else
    		{
    			ans += 1 ; 
    			j = 1 ;
    		}
    		
    	}
    	cout << ans << endl ; 
    	
    }
    return 0;
}
