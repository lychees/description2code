#include <iostream>

using namespace std;

int main()
{
    char c;
    int t,n,l,lift,stairs,flo;
    cin>>t;
    while(t--)
    {
        cin>>n>>l>>flo>>c;
        stairs=l*10;
        if(c=='U')
        {
            if(l==n)
            {
                if(flo==n)
                {
                    lift=10+n*5+10;
                }
                else
                {
                    lift=5*(n-flo)+10+n*5+10;
                }
            }
            else
            {
                lift=5*(n-flo)+10+10+n*5+10;
            }
        }
        else
        {
            if(l<=flo)
            {
            lift=flo*5+10+10;
            }
            else
            {
            lift=flo*5+10+2*n*5+20+10;

            }
        }
        //cout<<lift<<" "<<stairs<<endl;
        if(lift<=stairs)
        {
            cout<<"Lift "<<lift<<endl;
        }
        else
        {
            cout<<"Stairs "<<stairs<<endl;
        }
    }

    return 0;
}
