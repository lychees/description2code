
from operator import indexOf
t_cases = int(raw_input())

for i in range(t_cases):
    [n, k, m] = list(map(int, raw_input().split()))
    p = list(map(int, raw_input().split()))
    times = [[] for i in range(n)]
    eating_times = [0] * n
    for i in range(n):
        times[i] = list(map(int, raw_input().split()))
        eating_times[i] = sum(times[i]) + (p[i]) * k + (p[i]-1) * (m)

    print indexOf(eating_times, min(eating_times)) + 1, min(eating_times)
