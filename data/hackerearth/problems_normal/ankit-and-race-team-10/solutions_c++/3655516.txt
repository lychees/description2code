#include <bits/stdc++.h>
using namespace std;
long long int a[100001]={0};
long int c=1000000007;
long int inverse(long int a,long int b) {
        long int ans=1;
        long int c=b-2;
        while(c != 0) {
                if(c%2 == 1)
                        ans=(ans*a)%b;

                a=(a*a)%b;
                c/= 2;
        }
        return ans;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {  
    	int n,k;
    	cin>>n>>k;
    	for(int i=0;i<n;i++)
    	cin>>a[i];
    	
    	sort(a,a+n);
    	
    	long long int ans=a[n-k];
    	long long int temp=1;
    	
    	for(int i=n-k-1;i>=0;i--)
    	{   
    		temp= (((temp*(n-i-1))%c)*(inverse(n-i-k,c)))%c;
    		
    		ans=(ans+(temp*a[i])%c)%c;
    		    	}
    	
    	cout<< ans <<endl;
    	
    	
    	
    	
    }
    return 0;
}
