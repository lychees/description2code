#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long t,n;
	string s,temp,mins;
	string tempch;
	cin>>t;
	while(t--)
	{
		cin>>n>>s;
		mins=s;
		for(int i=0;i<s.length();i++)
		{
			for(int j=0;j<s.length();j++)
			{
				tempch=s[i];
				temp=s;
				temp.erase(temp.begin()+i);
				temp.insert(j,tempch);
				if(temp<mins)
				{
					mins=temp;
				}
			}
		}
		cout<<mins<<"\n";
	}
return 0;
}
