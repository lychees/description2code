#include<iostream>
#include<string.h>
#include<vector>
using namespace std;
main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string s,s1;
		cin>>s>>s1;
		vector<string> v;
		int i,n=s.length();
		string temp="";
		for(i=0;i<n;i++)
		{
			temp+=s.substr(0,i+1);
			temp+=s;
			temp+=s.substr(i+1,n-i-1);
			//cout<<"Pushing "<<temp<<endl;
			v.push_back(temp);
			temp="";
		}
		for(i=0;i<v.size();i++)
		{
			if(v[i].compare(s1)==0)
			{
				cout<<"Possible"<<endl;
				break;
			}
			
		}
		if(i==v.size())
		cout<<"Impossible"<<endl;
	}
}
