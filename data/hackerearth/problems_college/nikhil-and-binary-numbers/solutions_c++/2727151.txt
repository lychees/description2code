#include <iostream>
using namespace std;

int main()
{
    int t,n,r,c0,c1,i;
    int *a;
    
    cin>>t;
    while(t--){
    	cin>>n;
    	a = (int *)malloc(sizeof(int)*n);
    	c0=c1=0;
    	for(i=0;i<n;i++){
    		cin>>a[i];
    		if(a[i]==1){
    			c1++;
    		}else{
    			c0++;
    		}
    	}
    	
    	r=a[0];
    	for(i=1;i<n;i++){
    		r^=a[i];
    	}
    	if(r==1){
    		cout<<c1<<endl;
    	}else{
    		cout<<c0<<endl;
    	}
    	
    }
    
    return 0;
}
