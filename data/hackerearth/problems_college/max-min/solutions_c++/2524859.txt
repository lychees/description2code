#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int n,k,i,l,m;
    cin>>n>>k;
    int a[n];
    for(i=0;i<n;i++)
    cin>>a[i];
    sort(a,a+n);
    l=a[k-1]-a[0];
    for(i=0;i<n-k;i++)
    {
    	m=a[k+i-1]-a[i];
    	if(m<l)
    	l=m;
    }
    cout<<l;
    return 0;
}
