#include<iostream>
using namespace std;
#include<string.h>
#include<stdio.h>

int main()
{
    int t,l,i,j;
    scanf("%d",&t);
    getchar();
    while(t--)
    {
        char str[109];
        char A[109];
        cin.getline(str,101);
        l=strlen(str);
        j=0;
        i=0;
        while(i<l)
        {
        	if(str[i]=='!'||str[i]==','||str[i]=='.'||str[i]=='?'||str[i]==' ')
        	 i++;
        	else
        	{
        	 A[j]=str[i];
        	 j++;
        	 i++;
        	} 
        }
        i=0;
        int x;
        while(i<l)
        {
        	
        	if(A[i]=='*')
        	{
        	 x=i;
        	 break;
        	}
        	i++;
        }
        x++;
        x=j-x;
        printf("%c\n",A[x]);
    }
    return 0;
}
