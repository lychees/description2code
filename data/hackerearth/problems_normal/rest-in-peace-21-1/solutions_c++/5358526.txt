#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t,n,count;
    scanf("%d",&t);
    while(t--){
        scanf("%d",&n);
        count=0;
        if(n%21==0){
            printf("The streak is broken!\n");
        }
        else{
            while(n!=0){
                if(n%100==21){
                    printf("The streak is broken!\n");
                    break;
                }
                n/=10;
            }
            if(n==0){
                printf("The streak lives still in our heart!\n");
            }
        }
    }
    return 0;
}
