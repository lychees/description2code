#include <iostream>
using namespace std;

int main()
{
	int T;
	cin >> T;
	for(int i = 0; i < T; ++i)
	{
		string str;
		cin >> str;
		int N = str.size();
		int list[26];
		for(int j = 0; j < 26; ++j)
		{
			list[j] = 0;
		}
		for(int j = 0; j < N; ++j)
		{
			++list[str[j]-'a'];
		}
		int count = 0;
		for(int j = 0; j < 26; ++j)
		{
			count += (list[j]%2);
		}
		cout << max(count-1,0)<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
