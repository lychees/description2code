#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

const int C = 1000000;

int main()
{
    int n, k, m;
    cin >> n >> k >> m;
    vector<int> repr(C + 1, k + 1);
    repr[1] = 0;
    //vector<vector<int>> vecs(k + 1);
    //vecs[0].push_back(1);
    for (int i = 0; i < m; ++i) {
    	int cur;
    	cin >> cur;
    	if (cur >= C) {
    		continue;
    	}
    	repr[cur] = k;
    	//vecs[k].push_back(cur);
    }
    
    vector<int> result;
    result.push_back(1);
    for (int num = 2; num < C && result.size() < n; ++num) {
    	if (repr[num] <= k) {
    		continue;
    	}
    	result.push_back(num);
    	for (int i = 1; i * 1LL * num < C; ++i) {
    		if (repr[i * num] > repr[i] + 1) {
    			repr[i * num] = repr[i] + 1;
    		}
    	}
    }
    
    for (auto item : result) {
    	cout << item << ' ';
    }
    
    return 0;
}
