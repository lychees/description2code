
import sys

testcases = int(sys.stdin.readline())

def careful_zip(li, li_subset, m):
	result = []
	for i in li_subset:
		for counter in range(len(li)):
			if li[counter][:m] == i:
				result.append(li[counter])
				del li[counter]
				break
	return result

for testcase in range(testcases):
	li = []
	n, k, m = [int(i) for i in sys.stdin.readline().split(" ")]
	for x in range(n):
		li.append(sys.stdin.readline().replace("\n",""))
	li_subset = [i[:m] for i in li]
	li_subset.sort()
	result = careful_zip(li, li_subset, m)
	print(result[k-1])
