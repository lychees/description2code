#include <iostream>
#include <vector>
using namespace std;

int main()
{
	
    long int T=0;
    long int N=0;
    long int P=0;
    vector<long int> nVector;
    vector<long int> pVector;
    cin >> T;
    long i=0;
    for(i=0;i<T;i++){
    	cin >> N;
    	cin >> P;
    	nVector.push_back(N);
    	pVector.push_back(P);
    }
   
	for(i=0;i<T;i++){
		N=nVector[i];
		P=pVector[i];
		int j=0;
		if(P!=0){
			for(j=1;j<=P;j++){
				if(N%2==0){
					N=N/2+1;
				}
				else{
					N=N/2+2;
				}
				if(N==3){
					break;
				}
			}
		}
		cout << N << endl;
	}    
    return 0;
}
