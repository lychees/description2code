#include <bits/stdc++.h>
using namespace std;

int main()
{long long int b[100000],l=0,n,i,j,a[100000],ans,c=0,flag=0;
 cin>>n;
 for(i=0;i<n;i++)
 cin>>b[i];
 sort(b,b+n);
 for(i=n-1;i>=0;i--)
 a[l++]=b[i];
 for(i=0;i<l;i++)
 {
 	for(j=2;j<=a[i]/2;j++)
 	{
 		if(a[i]%j==0)
 		{flag=0;
 			break;
 		}
 		else flag=1;
 	}
 	if(flag==1)
 	{
 		ans=a[i];
 		break;
 	}
 }
 c=0;
 for(i=0;i<n;i++)
 {
 	if(a[i]==ans)
 	c++;
 }
 cout<<ans<<endl;
 
 cout<<c<<endl;
 return 0;
}
