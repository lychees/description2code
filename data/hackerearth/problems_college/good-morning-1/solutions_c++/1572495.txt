#include <iostream>
using namespace std;
int distace(int fuel,int distaceRemaining);

int main()
{
   // cout << "Hello World!" << endl;
   int t;
   int a,b;
   cin>>t;
   while(t){
   	cin>>a>>b;
   	if(distace(a,b))
   		cout<<"Yes\n";
   	else
   		cout<<"No\n";
   	t--;
   }
   
    return 0;
}

int distace(int fuel,int distaceRemaining){
	if(distaceRemaining <=0)
		return 1;
	if(fuel <=0)
		return 0;
	if(fuel>=3)
	distace(fuel-3+2,distaceRemaining-6);
	else
	distace(fuel-1,distaceRemaining-2);
}
