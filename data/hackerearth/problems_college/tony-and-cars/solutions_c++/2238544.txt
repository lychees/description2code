#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long int sum,test,num,n,i,j,remain;
    scanf("%lld",&test);
    while(test--)
    {
        sum=0;
        scanf("%lld",&num);
        if(num<10)
          printf("%lld\n",num);
        else
        {
            i=1;
            while(num>=pow(10,i))
            {
                sum+=i*(pow(10,i)-pow(10,i-1));
                i++;
            }
            remain=num-pow(10,i-1)+1;
            sum+=i*remain;
            printf("%lld\n",sum);
        }
    }
}
