#include<bits/stdc++.h>

using namespace std;

int main()
{
	long long int n;
	cin>>n;
	long long int s[n],c[n];
	
	for(long long int i=0;i<n;i++)
		cin>>s[i];
	for(long long int i=0;i<n;i++)
		cin>>c[i];
		
	sort(s,s+n);
	sort(c,c+n);
	
	long long int ans=0;
	for(long long int i=0;i<n;i++)
		{
			ans+= c[i]*s[i];
		
		}
		cout<<ans<<endl;
		return 0;
}
