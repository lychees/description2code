#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
   	int t,n;
   	int a[100001];
   	cin>>t;
   	while(t--)
   	{	cin>>n;
   		for(int i=0;i<n;i++)
   			cin>>a[i];
   		unordered_set<int> st;
   		int beg=0,end=0,mlen=-1,mbeg=0,mend=0;
   		while(end<n)
		{	if(st.find(a[end])!=st.end())
			{	
				if(end-beg>mlen)
				{	mlen=end-beg;
					mend=end-1;
					mbeg=beg;
				}
				while(a[beg]!=a[end])
				{	st.erase(a[beg]);
					beg++;
				}
				beg++;
			}
			else
			{	st.insert(a[end]);
			}
			end++;
		}
		if(end-beg>mlen)
		{	mbeg=beg;
			mend=end-1;
		}
   		cout<<mbeg+1<<" "<<mend+1<<endl;
   	}
   		
    return 0;
}
