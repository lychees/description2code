#include <iostream>
using namespace std;

long int pow(int i,int j)
	{
		int s=1;
		for(int k = 0;k<j;k++)
		{

			s= s*i;
		}
		return s;

	}
int main()
{
    int array[1000010]={0};
    array[1]=1;
    int i,j;
    for(i=2;i<1001;i++)
    {
    	for(j=2;pow(i,j) <1000010;j++)
    	{
    		int pos = pow(i,j);
    		array[pos] = 1;
    	}
    }
    //for(i=0;i<1000010;i++)
    //cout <<
    int t,n;
    cin>>t;
    while(t--){
        cin>>n;
        bool chk = false;
        for(int i = 1;i <= n/2 ; i++){
            if(array[i] && array[n-i]) {
                chk = true;
                break;
            }
        }
        if (chk) cout<<"Yes"<<endl;
        else cout<<"No"<<endl;
        }
}
