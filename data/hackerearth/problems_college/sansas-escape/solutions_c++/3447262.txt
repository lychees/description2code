#include<bits/stdc++.h>
using namespace std;
static int bits[100001][32];
int main(){
	int n;
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		int x;
		scanf("%d",&x);
		if(i>0)
		for(int j=0;j<30;j++)
		   bits[i][j]=bits[i-1][j];
		 else
		 for(int j=0;j<30;j++)
		    bits[i][j]=0;
		for(int j=0;j<30;j++){
			if((x>>j)&1)
			   bits[i][j]++;
		}
	}
	int q;
	scanf("%d",&q);
	while(q--){
		int a,b;
		scanf("%d %d",&a,&b);
		int sel[30];
		for(int i=0;i<30;i++)
		   sel[i]=bits[b][i];
		if(a>0)
		for(int i=0;i<30;i++)
		   sel[i]-=bits[a-1][i];
		long long ans=0;
		for(int i=0;i<30;i++){
			if(sel[i]%2){
				ans+=pow(2,i);
			}
		}
		printf("%lld\n",ans);
	}
	return 0;
}
