#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int main() {
	int n,m,i,sum=0,inp;
	cin>>m>>n;
	priority_queue<ll> pq;
	for(i=0;i<m;i++){
	    cin>>inp;
	    pq.push(inp);
	    sum=sum+inp;
	}
	ll ans=0,val;
	while(n-- && sum--){
	    val=pq.top();
	    ans=ans+val;
	    pq.pop();
	    pq.push(val-1);
	    
	}
	cout<<ans<<endl;
	return 0;
}
