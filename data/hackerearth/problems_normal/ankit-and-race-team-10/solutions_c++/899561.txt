#include <iostream>
#include<algorithm>
using namespace std;
#define m 1000000007
int fact[100001];

long long dencalc(long long a, int b){
	if(b==0){return 1;}
	if(b==1){return a;}
	long long ans;
	ans=dencalc(a,b/2);
	ans=(long long)ans*ans;
	ans=ans%m;
	if(b%2!=0){return (ans*a)%m;}
	return ans;
	
}
void solve(int *a,int n,int k){
	int i;
	long long ans=0,den,t;
	for(i=0;i<=n-k;i++){
		den=(long long) fact[n-i-k]*fact[k-1];
		den=den%m;
		den=dencalc(den,m-2);
		den=den%m;
		t=(long long)a[i]*fact[n-i-1];t=t%m;
		t=t*den;
		t=t%m;
		ans+=t;
		ans=ans%m;
		//cout<<ans<<" ";
	}
	cout<<ans<<endl;
}


int main()
{
	int i;
	long long t;
    fact[0]=1;
    fact[1]=1;
    for(i=2;i<=100000;i++){
    	t=(long long)i*fact[i-1];
    	fact[i]=t%m;
    }
    int nt,n,k;
    cin>>nt;
    while(nt--){
    	cin>>n>>k;
    	int *a=new int[n];
    	for(i=0;i<n;i++){cin>>a[i];}
    	sort(a,a+n);
    	solve(a,n,k);
    }
}
