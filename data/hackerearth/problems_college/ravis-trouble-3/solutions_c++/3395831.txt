
#include<bits/stdc++.h>
#include<cstring>
using namespace std;

bool isSubSequence(char str1[], char str2[], int m, int n)
{
   int j = 0; 
   for (int i=0; i<n&&j<m; i++)
       if (str1[j] == str2[i])
         j++;

   return (j==m);
}

int main()
{
    char str1[100001], str2[100001];
    int t;
    cin>>t;
    while(t--) {
    cin>>str1;
    cin>>str2;
    int m = strlen(str1);
    int n = strlen(str2);
    isSubSequence(str2, str1, n, m)? cout << "YES\n":cout << "NO\n";
    }
    return 0;
}
