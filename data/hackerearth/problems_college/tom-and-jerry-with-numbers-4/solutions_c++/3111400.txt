#include <iostream>
#include <algorithm>
#include <stdio.h>
#include <string.h>
using namespace std;
char ar[1000000];
bool f(int n, int a, int b);
bool f(int n, int a, int b){

   if(n<a)
            return false;

   else if(ar[n]!='\0')
   {
   if(ar[n]=='w')
   return true;
   else return false;
   }
       else{     if( (n>=a) && !f(n-a, a, b) || (n>=b) && !f(n-b, a, b) )
       {
       	ar[n]='w';
       }
       else ar[n]='l';
       if(ar[n]=='w')
   return true;
   else return false;
       
}
}
int main() {
	int t,n,a,b;
	
	cin>>t;
	while(t--)
	{
	cin>>n>>a>>b;memset(ar,'\0',n+1);
	if(f(n,a,b))
	cout<<"1"<<endl;
	else cout<<"0"<<endl;
	}
	return 0;
}
