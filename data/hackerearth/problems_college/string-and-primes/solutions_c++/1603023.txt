#include <iostream>
#include<string>
#include<cmath>
using namespace std;
int f(int n)
{
	if(n<2)
	return false;
	else
	{
	for(int i=2;i<=sqrt(n);i++)
	{
		if(n%i==0)
		return false;
	}
	return true;
    }
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string str,test="";
    	cin>>str;
    
    	for(int i=0;i<str.length();i++)
    	{
        	if(!f(str[i]))
             test+=str[i];

       }
       if(test=="")
       cout<<"NULL"<<endl;
       else
     cout<<test<<endl;
 }
    return 0;
}
