#include<iostream>
using namespace std;
int main()
{
    int i,l,j,n,s,t,z,flag=0,tmp;
    int b[15];
    int final[32768]={0};
    // cout<<"enter no of test cases"<<"\n";
    cin>>t;
    for(i=0;i<t;i++)
    {
        flag=0;
        z=0;
        for(j=0;j<32768;j++)
        final[j]=0;
        //cout<<"enter no of nums"<<"\n";
        cin>>n;
        for(j=0;j<n;j++)
        {
            cin>>b[j];
        }
        cin>>s;
        if(s==0)
            cout<<"YES"<<"\n";
        else
        {

            for(j=0;j<n;j++)
               {

                   for(l=z;l>=0;l--)
                   {
                       tmp=b[j]+final[l];
                      // cout<<b[j]<<" "<<final[l]<<" ";
                      // cout<<tmp<<"\n";
                       if(tmp==s)
                        break;
                        final[z++]=tmp;

                   }
                   if(tmp==s){
                       // cout<<"in here";
                       cout<<"YES"<<"\n";
                       flag=1;
                         break;

                   }


               }



        }
        if(flag==0)
            cout<<"NO"<<"\n";
    }
}
