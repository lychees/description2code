#include <iostream>
#include<stdio.h>
using namespace std;
typedef long long int64;
int64 mo = 1000000007;
int64 po(int64 a,int64 b){
	if(b==0)return 1;
	if(a==0)return 1;
	int64 x = po(a,b/2);
	x = (x*x)%mo;
	if(b%2)x=(x*a)%mo;
	return x;
}
int64 fa[1000005];
char a[1000005], b[1000005],c[1000005];
int is(int a)
{
	char b[] = "aeiou";
	int j;
	for(j=0;j<5;j++)if(b[j]==a)return 1;
	return 0;
}
int64 get(char *a)
{
	int co[26];
	int i;
	for(i=0;i<26;i++)co[i] = 0;
	int l;
	for(l=0;a[l];l++)co[a[l]-'a']++;
	int64 fir = fa[l];
	int64 sec = 1;
	for(i=0;i<26;i++)sec = (sec * fa[co[i]])%mo;
	sec = po(sec,mo-2);
	fir = (fir*sec)%mo;
	return fir;
}
int64 ncr(int64 n,int64 r){
	if(r>n)return 0;
	int64 fir = fa[n];
	int64 sec = fa[r] * fa[n-r];
	sec%=mo;
	sec = po(sec,mo-2);
	fir = fir*sec;
	return fir%mo;
}
