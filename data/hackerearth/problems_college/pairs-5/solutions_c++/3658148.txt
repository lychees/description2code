#include <bits/stdc++.h>
using namespace std;
#define vi vector<int>
int dp[15][1301][1301];
int sum=0,n;
int v[15];
int solve(int index,int gr1,int gr2)
{
	if(index==n)
	{
		int val=sum-gr1-gr2;
		if((val>=gr1)&&(val>=gr2))
		{
			return val;
		}
		else
			return (INT_MAX);
	}
	if(dp[index][gr1][gr2]!=-1)
		return dp[index][gr1][gr2];
	dp[index][gr1][gr2]=min(solve(index+1,gr1+v[index],gr2),min(solve(index+1,gr1,gr2),solve(index+1,gr1,gr2+v[index])));
	return dp[index][gr1][gr2];
}
int main()
{
 	int i,num;
 	scanf("%d",&n);
 	memset(dp,-1,sizeof(dp));
 	for(i=0;i<n;i++)
 	{
 		scanf("%d",&(num));
 		v[i]=num;
 		sum+=num;
 	}
 	int ans=solve(0,0,0);
 	printf("%d",ans);
    return 0;
}
