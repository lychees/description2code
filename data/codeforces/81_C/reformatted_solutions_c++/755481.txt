using namespace std;
#define li long long int
#define rep(i, to) for (li i = 0; i < ((li)(to)); ++i)
#define pb push_back
#define sz(v) ((li)(v).size())
#define bit(n) (1ll << (li)(n))

#define MAX 100005
int main() {
  li n, a, b, ans[MAX];
  pair<li, li> c[MAX];
  cin >> n >> a >> b;
  rep(i, n) cin >> c[i].first;
  rep(i, n) c[i].second = (b < a) ? i : (-i);
  if (a != b)
    sort(c, c + n);
  if (a < b)
    reverse(c, c + n);
  rep(i, n) ans[i] = 2;
  rep(i, a) ans[abs(c[i].second)] = 1;
  rep(i, n) cout << ans[i] << " ";
  cout << endl;
}
