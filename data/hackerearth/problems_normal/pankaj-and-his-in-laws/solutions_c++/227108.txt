#include <iostream>
#include <stack>
using namespace std;

int main()
{
   int n;
   cin>>n;
   int a[n+1],dp[n+1];
   for(int i=1;i<=n;i++)
    cin>>a[i];
    stack<int> st;
    a[0]=-100000;
   dp[0]=0; 
    int maxi=0;
    for(int i=1;i<=n;i++)
     {
     	dp[i]=0;
     	for(int j=0;j<i;j++)
     	   if(a[j]<a[i])
     	     dp[i]=max(dp[i],dp[j]+1);
     	maxi=max(dp[i],maxi);     
     	     
     }
    
    
    
    while(maxi)
      {
      	
        st.push(maxi%2);
      	maxi=maxi/2;
      }
   
   
   
   
   
   while(!st.empty())
   {
   	cout<<st.top();
   	st.pop();
   }
    return 0;
}
