#include <iostream>
#include <math.h>
#define M_PI 3.14159265358979323846

using namespace std;

int main()
{
        long int t;
        long long int temp;
        long long int multiple[10]={25,125,625,3125,15625,78125,390625,1953125,9765625,48828125};
        cin>>t;
        long long int n=0;
        while(t--)
        {
                cin>>n;
                long long int res=n/5;

                for(int i=0;i<10;i++)
                {
                        temp=n/multiple[i];
                        if(temp==0)
                                break;
                        else
                                res=res+temp;	//res is the no of trailing zeros
                }
                long int nod=(( (n* (log(n) ) )-n)+(0.5*log(2*M_PI*n)))/log(10);
                //cout<<endl<<"number of digits are: "<<nod;
                //cout<<endl<<"number of trailing zeros are: "<<res;
                cout<<nod-res+1<<endl;
        }
        return 0;
}
