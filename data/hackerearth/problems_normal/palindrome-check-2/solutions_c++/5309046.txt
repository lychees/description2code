#include <iostream>
#include <string.h>
using namespace std;

int main()
{
	int i,k,flag=0;
	char s[10000];
	cin.getline(s,10000);
	k=strlen(s);
	if(strlen(s)<=100)
	{
		for(i=0;i<((k/2)+1);i++)
		{
			if(s[i]==s[k-1-i])
				flag++;
		}
		if(flag==((k/2)+1))
		cout<<"YES"<<endl;
		else
		cout<<"NO"<<endl;
	}
	else
	cout<<"not in range"<<endl;
 return 0;
}
