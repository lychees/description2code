#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
typedef vector<int> vi;
typedef pair<int, int> pii;
const int MOD = (int) 1e9 + 7;

int n;
string line;

ll memo[100100][5];
ll dp(int id, int stage){
	if (memo[id][stage] != -1) return memo[id][stage];
	if (id == n){
		if (stage == 4) return 1;
		return 0;
	}
	char c = line[id];
	ll res = 0;
	if (stage == 0){
		if (c == ';') res += dp(id + 1, 1);
		res += dp(id + 1, stage);
	} else if (stage == 1){
		if (c == '_') {
			res += dp(id + 1, 2);
		}
		res += dp(id + 1, stage);
	} else if (stage == 2){
		if (c == '_') {
			res += dp(id + 1, 2);
		} else {
			res += dp(id + 1, 4);
		}
		res += dp(id + 1, stage);
	} else if (stage == 3){
		if (c == ';') res += dp(id + 1, 4);
		res += dp(id + 1, stage);		
	} else {
		res += dp(id + 1, stage);
	}
	return memo[id][stage] = res % MOD;
}

int main(){
	cin >> line;
	n = (int) line.length();

	memset(memo, -1, sizeof memo);
	ll res = dp(0, 0);
	cout << res << endl;

	return 0;
}
