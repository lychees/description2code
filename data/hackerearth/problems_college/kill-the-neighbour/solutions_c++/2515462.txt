#include <iostream>
using namespace std;

int main()
{
    int arr[500000] = {0};
    int n;
    cin>>n;
    for(int i =0;i<n;i++)
    cin>>arr[i];
    int xkills = 0,steps = 0;
    int i,j;
    while(1)
    {
    	
    	xkills = 0;
    	i = n-1;
    	while(i>=0)
    	{
    		if(arr[i] != -1)
    		{
    			j = i-1;
    			while(arr[j]== -1 && j>=0)
    			j--;
    			if(j>=0 && arr[j] > arr[i])
    			{
    				xkills++;
    				arr[i] = -1;
    				i = j;
    				
    			}
    			else
    			i--;
    		}
    		else
    		i--;
    	}
    	if(xkills == 0)
    	break;
    	steps++;
    }
    cout<<steps<<endl;
    return 0;
}
