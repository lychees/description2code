#include <iostream>
using namespace std;

int main()
{
    int t;
    long int temp;
    long int a[3]; 
    cin>>t;
    long int length;
    for(int i=0;i<t;i++)
    {
    	cin>>a[0]>>a[1]>>a[2];
    	for(int j=0;j<3;j++)
    		for(int k=j;k<3;k++)
    			if(a[j]<a[k])
    			{	temp=a[k];
    				a[k]=a[j];
    				a[j]=temp;
    			}
    	if(a[0]>a[1])
    	length=a[1]+a[1]+1;
    	else
    	length=a[1]+a[1];
    	cout<<length<<endl;
    	
    }
    return 0;
}
