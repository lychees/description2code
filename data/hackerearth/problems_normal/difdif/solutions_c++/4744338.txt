#include <iostream>
#include <set>
using namespace std;

int main()
{
   set <string> sub;
   set <string> :: iterator it;
   string str;
   getline(cin,str);
   for(int i = 0; i<str.length(); i++){
   	for(int j = i; j<str.length(); j++){
   		string subString = "";
   		subString = str.substr(i, j-i+1);
   		sub.insert(subString);
   	}
   }
   cout<<sub.size()<<"\n";

    return 0;
}
