#include<string.h>
#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    int a[t];
    for(int i=0;i<t;i++) cin>>a[i];
    
    int m[t];
    for(int i=0;i<t;i++) m[i]=1;
    for(int i=1;i<t;i++){
    	for(int j=0;j<i;j++){
    		if(a[i]>a[j]&&m[i]<m[j]+1)
    		m[i]= m[j]+1;
    	}
    	
    }
    
   
    
    string s="";
    int n= 1;
    for(int i=0;i<t;i++) n= max(n,m[i]);
    while(n){
    	
    	int k = n%2; 
    	s+= '0'+k;
    	n/=2;
    	
    }
    reverse(s.begin(),s.end());
    if(s=="") s="0";
    cout<<s;
    
}
