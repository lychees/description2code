#include <iostream>
#include <algorithm>
#include <string>
#include <stdlib.h>
#include <vector>
#include <limits>
 
std::vector<int> arr(31,0);
using namespace std;
int main()
{
	
	int t;
	cin>>t;
	
	while(t--)
	{
		std::cin.ignore(std::numeric_limits<std::streamsize>::max(), '\n');
		string s;
		int mult=1;
		cin.ignore();
		getline(cin, s);
		if(s[0]=='G')
			mult = 2;
 
		for(int i=0; i<s.size(); ++i)
		{
 
			int digit;
			if(isdigit(s[i]))
			{
				digit = s.at(i)-'0';
				if(isdigit(s[i+1]))
				{
					digit = 10 *  digit + (s[i+1] -'0');
					i++;
				}					
				//cout<<digit<<endl;
				arr[digit] += (mult);
			}
		}		
	}
	vector<int> cp;
	cp.push_back(*std::max_element(arr.begin(), arr.end()));
	
	if((arr[19]==cp[0] || arr[20]==cp[0])&&arr[19]!=arr[20])
		{
			cout<<"Date";
		}
		else
			cout<<"No Date";
		cout<<endl;
}
