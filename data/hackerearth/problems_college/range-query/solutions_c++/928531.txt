#include <iostream>
#include <string.h>
#include <assert.h>
#define ll long long
using namespace std;

int dp[10002][101];

int a[10002];

int main()
{
    int n;
    cin>>n;
    assert(n<=10000);
    for(int i=1;i<=n;i++) 
    {
        cin>>a[i];
        assert(a[i]<=100);
    }
    for(int i=0;i<=100;i++) dp[0][i]=0,dp[i][0]=0;
    for(int i=1;i<=n;i++)
    {
        for(int j=1;j<=100;j++)
        {
            dp[i][j] = dp[i-1][j];
            if(a[i]==j) dp[i][j]+=1;
        }
    }
    /*
    for(int i=0;i<10;i++)
    {
        for(int j=0;j<10;j++)
        {
            cout<<dp[i][j]<<"\t";
        }
        cout<<"\n";
    }
    */
    int q;
    cin>>q;
    while(q--)
    {
        int l,r,val;
        cin>>l>>r>>val;
        l++,r++;
        cout<<dp[r][val] - dp[l-1][val]<<"\n";
    }
}
