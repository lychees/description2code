#include <iostream>
using namespace std;
#include <set>
#include <vector>
#include<bits/stdc++.h>


int main()
{
    int t,n;
    cin>>t;
    while(t--){
    	cin>>n;
    	set<string> s;
    	
    	for(int i=0;i<n;i++){
    		string o;
    		cin>>o;
    		if(s.find(o)==s.end())
    		s.insert(o);
    		
    		
    	}
    	set<string>::iterator it;
    	for(it=s.begin();it!=s.end();it++)
    	cout<<*it<<endl;
    	
    	
    	
    	
    }
    return 0;
}
