#include <iostream>
using namespace std;

long long dp[1000001] = {0};
long long m;
long long fact(long long n){
	if(dp[n]){
		return dp[n];
	}
	dp[n] = (n*fact(n-1))%m;
	return dp[n];
}

int main()
{
	dp[0] = dp[1] = 1;
    long long n;
    cin >> n >> m;
    cout << fact(n) << endl;
    return 0;
}
