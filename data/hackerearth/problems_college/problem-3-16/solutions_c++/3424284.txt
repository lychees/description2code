#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
	int t,n,k,*ind,i,ans,tmp;
	cin >> t;
	while(t--)
	{
		cin >> n >> k;
		ind = new int[k+1];
		for(i=0;i<=k;i++)
			ind[i] = 0;
		for(i=0;i<n;i++)
		{
			cin >> tmp;
			ind[tmp]++;
		}
		sort(ind,ind+k+1);
		ans = 0;
		for(i=1;i<k;i++)
		{
				ans+=ind[i];
		}
		cout << ans << "\n";
	}
	return 0;
}
