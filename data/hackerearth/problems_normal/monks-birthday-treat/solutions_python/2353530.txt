n,d=map(int,raw_input().split())
ed=[[] for _ in range(n)]
for s in range(d):
	a,b=map(int,raw_input().split())
	ed[a-1].append(b-1)
mi=n+1

for i in range(n):
	coming=[0]*n
	check2=[i]
	coming[i]=1
	inv =1
	while check2:
		c=check2.pop()
		for e in ed[c]:
			if not coming[e]:
				coming[e]=1
				check2.append(e)
				inv+=1
	mi=min(mi,inv)			
print mi
