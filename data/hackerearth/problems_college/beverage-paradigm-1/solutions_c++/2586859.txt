#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    int flag,i,n,req,val,j,k;
    scanf("%d",&n);
    vector<int> v(n);
    for(i=0;i<n;i++)
    	scanf("%d",&(v[i]));
    scanf("%d",&req);
    flag=0;
    for(i=0;i<n-2;i++)
    {
    	for(j=i+1;j<n-1;j++)
    	{
    		for(k=j+1;k<n;k++)
    		{
    			val=v[i]+v[j]+v[k];
    			if(val==req)
    			{
    				flag=1;
    				break;
    			}
    		}
    		if(flag)
    			break;
    	}
    	if(flag)
    		break;
    }
    if(flag)
    	printf("True");
    else
    	printf("False");
    return 0;
}
