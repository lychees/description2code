#include <bits/stdc++.h>
using namespace std;
int f(int availableCash, int price, int wrapperDiscount){
    int eaten = 0;
    int wraps = 0;
     
    wraps = eaten = availableCash/price;
     
    while (wraps >= wrapperDiscount){
        int newlyEaten = wraps/wrapperDiscount;
        eaten += newlyEaten;
        wraps %= wrapperDiscount;
        wraps += newlyEaten;
    }
     
     
    return eaten;
}
 
int main() {
    int t,n,c,m;
    cin>>t;
    while(t--){
        cin>>n>>c>>m;
        int ans=0;
        ans=f(n,c,m);
        cout<<ans<<endl;
    }
    return 0;
}
