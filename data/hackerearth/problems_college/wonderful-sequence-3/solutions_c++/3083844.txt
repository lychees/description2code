#include <vector>
#include <iostream>
#include <algorithm>
#include <set>
//#include <cstring>
using namespace std;

typedef unsigned long long ull;

/* O(n^2) DP solution */
int main() {

    int t;
    cin>>t;

    while(t!=0){

    int n;
    cin>>n;

    ull a[n];

    for(int i=0;i<n;i++)
    {
        cin>>a[i];
    }

  set<ull> st;
set<ull>::iterator it;
st.clear();
for(int i=0; i<n; i++) {
    it = st.lower_bound(a[i]);
    if (it != st.end()) st.erase(it);
    st.insert(a[i]);
}
cout<< n-st.size()<<endl;

  t--;
    }
  return 0;
}
