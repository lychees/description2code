#include <iostream>
#include <vector>
using namespace std;

int main() {
	int n;
	vector<int> p;
	cin >> n;
	p.resize(n);
	for(int i = 0;i < n;i++) cin >> p[i];
	int score = 0;
	while(p.size() >= 3)
	{
		int max2 = 0;
		int i;
		int max_i;
		for(i = 1;i < p.size()-1;i++)
		{
			if(p[i-1]*p[i+1] > max2)
			{
				max_i = i;
				max2 = p[i-1]*p[i+1];
			}
		}
		p.erase(p.begin()+max_i);
		score += max2;
	}
	cout << score << endl;
	return 0;
}
