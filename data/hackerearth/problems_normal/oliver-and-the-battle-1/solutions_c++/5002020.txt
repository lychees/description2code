#include<bits/stdc++.h>
using namespace std;
#define LL long long
#define S 1000005
#define W while
#define M 1000000007
#define str string
#define pi 3.14159265
LL mini=INT_MAX;
LL maxi=INT_MIN;
int  a[1001][1001];
LL v[1001][1001];
int n,m;
LL bfs(int i,int j)
{
	queue<pair<int,int> >  q;
	q.push(make_pair(i,j));
	v[i][j]=1;
	int cnt1=0;
	W(!q.empty())
	{
		int x=q.front().first;
		int y=q.front().second;
		q.pop();
		cnt1++;
		if(x<n&&a[x][y+1]&&y+1<m&&v[x][y+1]==0)
		{
			q.push(make_pair(x,y+1));
			v[x][y+1]=1;
		}
		if(a[x][y-1]&&y-1>=0&&v[x][y-1]==0)
		{
			q.push(make_pair(x,y-1));
			v[x][y-1]=1;
		}
		if(x+1<n&&a[x+1][y+1]&&y+1<m&&v[x+1][y+1]==0)
		{
			q.push(make_pair(x+1,y+1));
			v[x+1][y+1]=1;
		}
		if(x-1>=0&&a[x-1][y-1]&&y-1>=0&&v[x-1][y-1]==0)
		{
			q.push(make_pair(x-1,y-1));
			v[x-1][y-1]=1;
		}
		if(x+1<n&&a[x+1][y]&&v[x+1][y]==0)
		{
			q.push(make_pair(x+1,y));
			v[x+1][y]=1;
		}
		if(x-1>-1&&a[x-1][y]&&v[x-1][y]==0)
		{
			q.push(make_pair(x-1,y));
			v[x-1][y]=1;
		}
		if(x-1>=0&&a[x-1][y+1]&&y+1<m&&v[x-1][y+1]==0)
		{
			q.push(make_pair(x-1,y+1));
			v[x-1][y+1]=1;
		}
		if(x+1<n&&a[x+1][y-1]&&y-1>=0&&v[x+1][y-1]==0)
		{
			q.push(make_pair(x+1,y-1));
			v[x+1][y-1]=1;
		}
	}
	return cnt1;
}
int main()
{
  int t;
  cin>>t;
  int y=0;
  W(y<t)
  {
	cin>>n>>m;
	maxi=0;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
		{
			cin>>a[i][j];
			v[i][j]=0;
		}
	}
	int cnt=0;
	for(int i=0;i<n;i++)
	{	
		for(int j=0;j<m;j++)
		{
			if(!v[i][j]&&a[i][j])
			{
				maxi=max(maxi,bfs(i,j));
				cnt++;
			}
		}
	}
	y++;
	cout<<cnt<<" "<<maxi<<endl;
  }
  return 0;
}
