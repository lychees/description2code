#include<iostream>
#include<cstdio>

using namespace std;
int count=0;

inline void fast(int &x)
{

    register int c = getchar_unlocked();
    x=0;

    for(;((c<48 || c>57));c=getchar_unlocked());

    for(;c>47 && c<58; c = getchar_unlocked())
    {
        x = (x<<1) + (x<<3) + c -48;
	
    }
	if(x <= 300000000) count++;	

}

main()
{

    
    int i,n;
    scanf("%d",&n);
for(int j=0;j<n;j++)
{
    fast(i);
    
}
printf("%d\n",count);
}
