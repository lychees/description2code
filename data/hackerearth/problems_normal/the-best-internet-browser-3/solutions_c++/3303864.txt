#include <iostream>
#include <cstring>
#include <string>
#include <cstdlib>
using namespace std;

int main()
{
	int iNumStr=0;
	cin>>iNumStr;
	
	//Test Limits
	if( iNumStr >=1 && iNumStr<=100 ){}
	else{ return 1;}
	
	for(int iIndex=0;iIndex<iNumStr;iIndex++)
	{
		string cStr;
		cin>>cStr;
		int iLength = cStr.length();
		
		if( iLength >=1 && iLength<=200 ){}
		else{	return 1;}
		
		int iCount=0;
		for( int jIndex=4;jIndex<iLength-8+4;jIndex++)
		{
			if( cStr[jIndex] == 'a' ||
				cStr[jIndex] == 'e' ||
				cStr[jIndex] == 'i' ||
				cStr[jIndex] == 'o' ||
				cStr[jIndex] == 'u' )
				{
					iCount++;
				}
		}
		
		cout<<(iLength-4-iCount)<<"/"<<iLength<<endl;
	}
    
    return 0;
}
