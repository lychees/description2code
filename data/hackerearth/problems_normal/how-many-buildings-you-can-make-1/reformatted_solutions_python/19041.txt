a = [-1]*1001
a[0] = 0
a[1] = 1
a[2] = 2
a[3] = 3


def G(n):
    if (a[n] != -1):
        return a[n]
    c = 0
    for i in range(1, int(n**.5)+1):
        if (n % i == 0):
            c = c+1
    v = c+G(n-1)
    a[n] = v
    return v


num = int(raw_input())
for i in range(num):
    val = int(raw_input())
    print G(val)
