#include <iostream>
using namespace std;

int main()
{
    int n,sum=0;
    cin>>n;
    int a[n];
    for(int i=0;i<=n-1;i++)
    {
    	cin>>a[i];
    }
    for(int i=0;i<=n-1;i++)
    {
    	cin>>a[i];
    }
    if(n>1)
    {
    	for(int i=1;i<=n-1;i++)
    	{
    		if(a[i]==0)
    		{
    			int j=i-1;
    			while(a[j]==0 && j>=0)
    			{
    				j--;
    			}
    			if(j>=0)
    			{
    				a[j]=0;
    			}
    		}
    	}
    }
    for(int i=0;i<=n-1;i++)
    {
    	sum+=a[i];
    }
    cout<<sum;
}
