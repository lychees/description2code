#include<bits/stdc++.h>
using namespace std;
long long int t,n,i,j,k,l,s,q,r,a[200005],bit[200005];
long long int check(long long int x)
{
    j=x%10;
    while(x>=10)
    {
    	x=x/10;
    }
    if(x==j)
    return 1;
    else
    return 0;
}
void update(long long int x)
{
       	for(;x<=n;x+=x&-x)
        {
           	bit[x]+=1;
        }
}
void update1(long long int x)
{
       	for(;x<=n;x+=x&-x)
        {
           	bit[x]-=1;
        }
}
long long int query(long long int x)
{
     long long int sum = 0;
     for(;x>0;x-=x&-x)
        sum+=bit[x];
     return sum;
}
int main()
{
    scanf("%lld%lld",&n,&q);
    for(i=1;i<=n;i++)
    {
    	scanf("%lld",&a[i]);
    	if(check(a[i]))
    	update(i);
    }
   while(q--)
    {
    	scanf("%lld",&t);
    	if(t==2)
    	{
    		scanf("%lld%lld",&l,&r);
    		if(check(a[l]))
    	    update1(l);
    	    a[l]=r;
    	    if(check(r))
    		update(l);
    	}
    	else
    	{
    		scanf("%lld%lld",&l,&r);
    		if(l!=1)
    		j=query(l-1);
    		else
    		j=0;
    		k=query(r);
    		printf("%lld\n",(k-j));
    	}
    }
    return 0;
}
