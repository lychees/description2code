#include <iostream>
#include<string.h>
using namespace std;
typedef long long ll;
int main()
{
    ll t;
    cin>>t;
    while(t--)
    {
    	char a[100099];
    	ll x,i,sum1=0,sum2=0;
    	cin>>a;
    	x=strlen(a);
    	for(i=0;i<x/2;i++)
    	{
    		sum1=sum1+a[i];
    		sum2=sum2+a[x-i-1];
    	}
    	if(sum1==sum2)
    	{
    		cout<<"YES"<<endl;
    	}
    	else
    	{
    		cout<<"NO"<<endl;
    	}
    }
    return 0;
}
