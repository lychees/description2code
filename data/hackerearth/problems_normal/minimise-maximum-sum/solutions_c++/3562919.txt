#include <iostream>
#include<algorithm>
using namespace std;

int main()
{     long int n,a[200000],b[100000];
      int t;
      cin>>t;
      for(int i=0;i<t;i++)
      {
      	cin>>n;
      	for(int j=0;j<2*n;j++)
      	cin>>a[j];
      	sort(a,a+2*n);
      	for(int j=0;j<n;j++)
      	b[j]=a[j]+a[2*n-1-j];
      	sort(b,b+n);
      	cout<<b[n-1]<<endl;
      }

    
    return 0;
}
