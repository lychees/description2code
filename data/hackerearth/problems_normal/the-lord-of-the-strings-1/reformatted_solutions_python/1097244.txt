from sets import *
import sys
# print 'Hello World!'
n = int(raw_input())
s = set()
i = 0
while (i < n):
    tc = raw_input()
    for x in tc:
        if (not (x in s)):
            # print "no"
            s.add(x)
            sys.stdout.write(x)
    # print s
    print("")
    s.clear()
    i = i + 1
