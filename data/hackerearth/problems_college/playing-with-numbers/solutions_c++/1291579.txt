#include <iostream>
using namespace std;

int main()
{
    int t;
    unsigned long long n;
   	scanf("%d",&t);
   	int player = 0;
   	while(t--)
   	{
   		scanf("%llu",&n);
   		int cnt = __builtin_popcountll(n);
   		if(cnt&1)
   		{
   			if(player==0)
   				printf("Sherlock\n");
   			else if(player==1)
   				printf("Watson\n");
   		}
   		else
   		{
   			if(player==0)
   				printf("Watson\n");
   			else if(player==1)
   				printf("Sherlock\n");
   			player = 1-player;
   		}
   	}
    return 0;
}
