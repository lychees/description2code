#include <iostream>
using namespace std;

int main()
{
    long long int i,j,n,sum,pro,cnt=0,t;
    cin>>t;
    while(t--)
    {
    cin>>n;cnt=0;
    
    for(i=1;i<=n;i++)
    {
    	for(j=1;j<i;j++)
    	{
    		sum=i+j;
    		pro=i*j;
    		if(pro%sum==0)
    		cnt++;
    	}
    }
    cout<<cnt<<"\n";
    }
    return 0;
}
