def checkVal(iterator):
    return len(set(iterator)) <= 1


def car_names(inp_str):
    count_dict = {}
    for i in range(len(inp_str)):
        if inp_str[i] in count_dict:
            count_dict[inp_str[i]].append(i)
        else:
            count_dict[inp_str[i]] = []
            count_dict[inp_str[i]].append(i)
    # print count_dict
    # print
    if len(count_dict.keys()) != 3:
        print "Not OK"
        return
    else:
        it = []
        bool = True
        for x in count_dict.values():
            for y in range(1, len(x)):
                if x[y] - x[y-1] > 1:
                    bool = False
            it.append(len(x))
        if not bool:
            print "Not OK"
            return
        result = checkVal(it)
        if result:
            print "OK"
            return
        else:
            print "Not OK"
            return


N = int(raw_input())
for i in range(N):
    input = raw_input()
    car_names(input)
