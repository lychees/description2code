#include <iostream>
using namespace std;
int main(int argc, char *argv[])
{
  int n, m, q, x, y, ans;
  cin>>n>>m>>q;
  int a[n][m], inp[n];

  for (int i = 0; i < n; i++) {
    for (int j = 0; j < m; j++) {
      cin>>a[i][j];
    }
  }

  while (q--) {
    for (int i = 0; i < n; i++) {
      cin >> inp[i];
    }

    ans = m-1;
    for (int i = 0; i < n; i++) {
      while (ans > 0 && inp[i] < a[i][ans]) {
	ans--;
      }
    }
    std::cout << ans+1 << std::endl;
  }


  return 0;
}
