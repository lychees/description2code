#include<iostream>
#include<algorithm>
#include<vector>
#include<string.h>
using namespace std;

bool st[100005];

int main()
{
    long long i,j,n,m,s,q,temp,g;
    cin>>s>>q;
    memset(st,0,sizeof(st));
    while(q--) {
        cin>>temp;
        g=__gcd(temp,s);
        if(!st[g]) st[g]=1,cout<<g<<"\n";
        else cout<<"-1\n";
    }
    return 0;
}
