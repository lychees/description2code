set<string> s;
int add,shift;

void mix(string str)
{
  
	 s.insert(str);
	 string newstr=str;
	 
	 int i,n=str.size();
		 for(i=0;i<n;i++)
			   newstr[(i+shift)%n]=str[i];
	
	 if(s.find(newstr)==s.end())
			   mix(newstr);
			   
	 newstr=str;
	 for(i=0;i<n;i++)
			  if(i%2)
				  newstr[i]= '0'+ (newstr[i]-'0'+add)%10;
	 
	 if(s.find(newstr)==s.end())
			   mix(newstr);
	 return;
}
