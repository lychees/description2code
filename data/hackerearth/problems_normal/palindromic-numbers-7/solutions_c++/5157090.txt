#include<iostream>
using namespace std;
int main()
{
    int n;
    cin>>n;
    int i=0;
    while(i<n)
    {
        int m1;
        int m2;
        cin>>m1>>m2;
        int count=0;
        for(int jj=m1;jj<=m2;jj++)
        {
            string j=to_string(jj);
            if(j == string(j.rbegin(),j.rend()))
            {
                count++;
                
            }
        }
        cout<<count<<endl;
        i++;
    }
}
