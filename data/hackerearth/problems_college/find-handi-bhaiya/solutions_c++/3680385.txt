#include <iostream>

{

    long t;
    cin>>t;
    long long a,b,c;

    while(t--)
    {
    	cin>>a>>b>>c;
    	if(a+b>c && a+c>b && b+c>a)
    	{
    		if(a==b && b==c)
    			cout<<"YES. It is a equilateral triangle."<<endl;

    	}

    	}

}
