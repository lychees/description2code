#include <iostream>
#include <string>
using namespace std;

string decrypt(string in){
  string temp="";
  int len = in.length();
  for(int i=0;i<len;i++)
  {
  switch(in[i]){
   case 'a':
   case 'b':
   case 'c': temp+="2";break;
   case 'd':
   case 'e':
   case 'f': temp+="3";break;
   case 'g':
   case 'h':
   case 'i': temp+="4";break;
   case 'j':
   case 'k':
   case 'l': temp+="5";break;
   case 'm':
   case 'n':
   case 'o': temp+="6";break;
   case 'p':
   case 'q':
   case 'r':
   case 's': temp+="7";break;
   case 't':
   case 'u':
   case 'v': temp+="8";break;
   case 'w':
   case 'x':
   case 'y':
   case 'z': temp+="9";break;
  }
  }

  return temp;
}
