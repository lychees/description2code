#include <iostream>
#include <cstring>
#include <cstdio>
#include <string>
using namespace std;
#define MAX 110
int t;
string s1, s2;
bool F[26];

bool solve() {
	memset(F,true,sizeof(F));
	for (int i = 0; s1[i]; i++) {
		F[s1[i] - 'a'] = false;
	}
	for (int i = 0; s2[i]; i++) {
		if (F[s2[i] - 'a'] == false) {
			return true;
		}
	}
	return false;
}

int main()
{
	scanf("%d", &t);
	while (t--) {
		cin >> s1 >> s2;
		if (solve() == true) {
			printf("Yes\n");
		}
		else {
			printf("No\n");
		}
	}
    return 0;
}
