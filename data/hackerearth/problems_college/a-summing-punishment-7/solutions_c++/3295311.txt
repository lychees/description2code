#include <iostream>
using namespace std;

int main()
{
    //printf("Hello World!\n");
    int s;
    cin>>s;
    int a[s],ar[s];
    for(int i=0;i<s;i++)
    cin>>a[i];
    ar[0]=a[0];
    for(int j=1;j<s;j++)
    ar[j]=a[j]+ar[j-1];
    int q,n,b;
    cin>>q;
    while(q--)
    {
    	cin>>n>>b;
    	if(n-1>0)
    	{cout<<ar[b-1]-ar[n-2]<<endl;}
    	else
    	cout<<ar[b-1]<<endl;
    }
    
    return 0;
}
