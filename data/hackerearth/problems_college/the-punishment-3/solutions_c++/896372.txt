#include<iostream>
#include<string>
#include<vector>

using namespace std;

int max(int a,int b)
{
    return(a>b)?
    a:b;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,k;
        cin>>n>>k;
        int m=n-k;
        int l;
        (m==k)?l=1:l=0;

        if(l==1)
        {int f=0;
        for(int i=1;i<=n;i++)
        {

            if(i==n){cout<<1;continue;}
            if(n+1-i==k+1){f=1;cout<<8;continue;}
            if(f==1&&i!=n){cout<<0;continue;}
            cout<<9;
        }
        }
        if(l==0)
        {
            int f=0,g=0;
            int y=max(m,k);
            int z=m+k-y;
            for(int i=1;i<=n;i++)
            {
                if(i==n){cout<<1;continue;}
                if(n+1-i==y+1){f=1;cout<<8;continue;}
                if(n+1-i==z+1){g=1;cout<<9;continue;}
                if(f==1&&g==1){cout<<0;continue;}
                cout<<9;
            }
        }
        cout<<endl;

    }

    return 0;
}
