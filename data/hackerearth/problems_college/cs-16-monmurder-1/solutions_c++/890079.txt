#include <iostream>
#include <cmath>
#include <vector>
using namespace std;

int main()
{
    int t;
    cin>>t;
    long n,c,ans,i,sq,s,count,n1;
    long piles[1000];
    while(t--)
    {
    	cin>>n;
    	n1 = n;
    	count = 0;
    	sq = sqrt(n);
    	for(i=2;i<=sq;i++)
    	{
    		
    		if(n%i==0)
    		{
    			c = 0;
    			while(n%i==0)
    			{
    				c++;
    				n/=i;
    			}
    			piles[count++] = c;
    		}
    	}
    	
    	if(n!=1)
    	{
    		piles[count++] = 1;
    	}

    	ans = piles[0];

    	for(i=1;i<count;i++)ans = (ans^piles[i]);
    	

    	
    	if(ans || n1==1)
    	{
    		cout<<"Printf{}\n";
    	}
    	else
    	{
    		cout<<"Pappu Bhujia\n";
    	}
    	
    }
    return 0;
}
