#include<iostream>
#include<math.h>
using namespace std;
int main()
{
	long long  n,k,t,i,test,count1,num,sum;
	scanf("%lld",&t);
	for(test=1;test<=t;test++)
	{
	count1=0;
	
	scanf("%lld",&n);
	
	scanf("%lld",&k);
	for(i=1;i<10000000001;i++)
	{
		num=n^i;
		sum=n+i;
		if(num==sum)
		{
		//	printf("%lld %lld  %lld isn xor sum\n",n^i,n+i,i);
			count1++;
		}
		if(count1==k)
		{
			printf("Case #%lld: %lld\n",test,i);
			break;
		}
	}
	}
	return 0;
	
}
