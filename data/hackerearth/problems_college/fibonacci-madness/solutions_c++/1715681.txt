#include<iostream>
using namespace std;
int main()
{
	int limit,n;
	cin>>n;
	for(int m=0;m<n;m++)
	{
		long k=0,l=1;
		cin>>limit;
		for(int i=0; i< limit;i++)
		{
			k=k+l;
			l=k-l;
		}
		for( int i=0;i<limit;i++)
		{
			l=k-l;
			k=k-l;
			cout<<k<<" ";
		}
		if(limit!=0)
			cout<<"\n";
	}
	return 0;
}
