#include <bits/stdc++.h>
using namespace std;
map<char,int> mp;
int main ()
{
    mp.clear();
    int n;
    scanf("%d",&n);
    while (n--){
        string s;
        cin>>s;
        mp[s[0]]++;
    }
    map<char,int>::iterator it;
    int ans=0;
    for (it=mp.begin();it!=mp.end();it++){
        int temp= it->second;
        if (temp<10) ans++;
        else {
            ans+=temp/10;
            if (temp%10!=0) ans++;
        }
        //cout<<temp<<" ";
    }
    cout<<ans<<endl;
}
