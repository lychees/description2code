#include <bits/stdc++.h>

using namespace std;

const int N = 1e5 + 10;

pair <int, int> pa[N];
vector < pair <int, int> > seg;

int main() {
    int x, y, d, n;
    scanf("%d", &n);
    for (int i = 0; i < n; i++) {
        scanf("%d %d %d", &x, &y, &d);
        pa[i] = make_pair(x, x + d);
    }
    sort(pa, pa + n);
    seg.push_back(pa[0]);
    for (int i = 1; i < n; i++) {
        auto &last = seg.back();
        if (pa[i].first > last.second) {
            seg.push_back(pa[i]);
        } else if (pa[i].second > last.second) {
            last.second = pa[i].second;
        }
    }
    int res = 0;
    for (auto it = seg.begin(); it != seg.end(); ++it) {
        res += it->second - it->first + 1;
    }
    printf("%d\n", res);
    return 0;
}
