inline int readInt(){
	long int n = 0, c = gc(), f = 1;
	while(c != '-' && (c < '0' || c > '9')) c = gc();
	if(c == '-') f = -1, c = gc();
	while(c >= '0' && c <= '9')
	n = (n<<3) + (n<<1) + c - '0', c = gc();
	return n * f;
}
long long f[32][32];
string getAllKeys(int n, int m) 
{ 
    memset(f, 0, sizeof(f)); 
    f[1][1] = 1; 
    int i, j; 
    for (i=2; i<=n; i++) 
      for (j=1; j<=i; j++) 
        f[i][j] = f[i-1][j-1] + (i-1) * f[i-1][j]; 
    long long a = 0; 
    for (i=1; i<=m; i++) 
      a += f[n][i]; 
    long long b = 1; 
    for (i=1; i<=n; i++) 
      b *= i; 
    long long d = gcd(a, b); 
    a /= d; 
    b /= d; 
    char s[50]; 
    sprintf(s, "%lld/%lld", a, b); 
    return s; 
}
