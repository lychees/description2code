#include<iostream>
#include<string>
#define si(n) scanf("%d",&n)
#define pi(n) printf("%d\n",n)
#define sll(n) scanf("%lld",&n)
#define pll(n) printf("%lld\n",n)
#include<math.h>
#include<algorithm>
#define ll long long int 
using namespace std;
int r[1000001];
int f()
{
	for(ll i=0;i<=1000000;i++) r[i] = -1;
	r[1] = 1;
	for(ll i=2;i<=1000000;i++)
	{
		if(r[i]==-1)
		{
			r[i] = 0;
			ll x = i*i;
			while(x<=1000000)
			{
				r[x] = 1;
				x*=i;
			}
		}
	}
	//for(int i=0;i<=10;i++) cout<<r[i]<<" ";
}
int main()
{
	f();
	int t;si(t);
	while(t--)
	{
		int n;si(n);
		bool c = false;
		for(int i=1;i<n;i++)
		{
			if(r[i]&&r[n-i])
			{
				c = true;
				break;
			}
		}
		if(c) printf("Yes\n");
		else printf("No\n");
	}
}
