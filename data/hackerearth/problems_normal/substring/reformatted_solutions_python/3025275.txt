t = input()
while t > 0:
    t -= 1
    string = raw_input()
    c = 0
    s = 0
    for char in string:
        if char not in 'ABCabc':
            c += s*(s+1)/2
            s = 0
        else:
            s += 1
    c += s*(s+1)/2
    print c
