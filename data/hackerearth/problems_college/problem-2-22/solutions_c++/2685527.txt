#include <bits/stdc++.h>

#define mp make_pair
#define pb push_back
#define ll long long

using namespace std;

double xc, yc, r;
double x[4], y[4];

double sqr(double x) {
	return x * x;
}

bool inside(int i) {
	return sqr(x[i] - xc) + sqr(y[i] - yc) < r * r + 1e-9;
}

int main() {
#ifdef LOCAL
	freopen("input.txt", "r", stdin);
#endif
	int t;
	scanf("%d", &t);
	while (t--) {
		scanf("%lf%lf%lf", &r, &xc, &yc);
		for (int i = 0; i < 4; ++i)
			scanf("%lf%lf", &x[i], &y[i]);
		bool kt = 1;
		for (int i = 0; i < 4; ++i) {
			if (!inside(i)) {
				kt = 0;
				break;
			}
		}
		if (kt)
			printf("Yes\n");
		else
			printf("No\n");
	}
#ifdef LOCAL
	printf("\n%.3lf s\n", (double) clock() / CLOCKS_PER_SEC);
#endif
}
