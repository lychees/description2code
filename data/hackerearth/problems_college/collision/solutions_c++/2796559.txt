#include <bits/stdc++.h>
using namespace std;
int main()
{
	set<int >s;
	int t,n,x;
	scanf("%d",&t);
	while(t--){
		scanf("%d",&n);
		int coun=0;
		for(int i=0;i<n;++i){
			scanf("%d",&x);
			int temp=(int)s.size();
			s.insert(x%10);
			if(temp==(int)s.size())
			coun++;
			
		}
		cout<<coun<<endl;
		s.clear();
	}
    return 0;
}
