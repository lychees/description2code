#include <bits/stdc++.h>
using namespace std;

int scan()
{
	int n=0,c;
	c=getchar_unlocked();
	while(c<'0' || c>'9'){c=getchar_unlocked();}
	while(c>='0' && c<='9')
	{
		n=(n<<3)+(n<<1)+(c-'0');
		c=getchar_unlocked();
	}
	return n;
}

void solve()
{
	int n;
	cin>>n;
	int count = 0;
	for (int i = 1; i < n; ++i)
	{
		for (int j = i+1 ; j <= n; ++j)
		{
			if((i*j)%(i+j) == 0)
				count++;
		}
	}
	cout<<count<<endl;

}

int main()
{
      int t;
      t=scan();
      while(t--)
            solve();
      return 0;
}
