from string import letters


def readint():
    return int(raw_input())


def readfloat():
    return float(raw_input())


def readarray(N, foo):
    res = []
    for _ in xrange(N):
        res.append(foo())
    return res


def readlinearray(foo):
    return map(foo, raw_input().split())

def get_npath(path):
    dirs = [d for d in path.split('/') if d != '']
    return '/' + '/'.join(dirs)
