#include <iostream>
using namespace std;

int main()
{
 //   cout << "Hello World!" << endl;
 int t;
 cin>>t;
 while(t--)
 {
 	long long int n;
 	cin>>n;
 	int count=0;
 	while(n>0)
 	{
 		if(n&1)
 		count++;
 		n=n>>1;
 	}
 	if(count%2==0)
 	cout<<"even"<<endl;
 	else
 	cout<<"odd"<<endl;
 }
 
    return 0;
}
