#include<bits/stdc++.h>
using namespace std;
#define ll long long
map<ll,ll>memory;
map<ll,ll>::iterator itr;
ll sweet[1000009],qty[1000009];
int main()
{
    ll a,b,n,g,ans=0;
    scanf("%lld",&n);
    for(int i=1;i<=n;i++)
    {
        scanf("%lld%lld",&a,&b);
        sweet[i]=a;
        memory[a]+=b;
    }
    sort(sweet+1,sweet+n+1);
    qty[n+1]=0;
    for(int i=n;i>=1;i--)
        qty[i]=max(memory[sweet[i]],qty[i+1]);
//for(int i=1;i<=n;i++) cout<<qty[i]<<" ";cout<<endl;
//for(itr=memory.begin();itr!=memory.end();itr++)cout<<itr->first<<" "<<itr->second<<endl;
    scanf("%lld",&g);
    while(g--)
    {
        scanf("%lld",&a);
        int x=lower_bound(sweet+1,sweet+n+1,a)-sweet;
        //cout<<"index is "<<x<<endl;
        if(x>n) continue;
        ans+=100*(qty[x]/a);
    }
    printf("%lld",ans);

    //sort(guest+1,guest+g+1);



}
