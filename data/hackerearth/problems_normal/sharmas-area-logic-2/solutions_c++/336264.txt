#include <iostream>
using namespace std;

int x[1000];
int y[1000];
int main()
{
   	int t, test;
	cin>>test;
	for(t=0;t<test;t++)
	{
		int n, cur_x, cur_y;
		cin>>n;
		int min_x=-90,max_x=90;
		int min_y=-180,max_y=180;
		cin>>x[0]>>y[0];
		min_x=x[0];
		max_x=x[0];
		min_y=y[0];
		max_y=y[0];
		for(int i=1;i<n;i++)
		{
			cin>>x[i]>>y[i];
			if(min_x > x[i])
				min_x=x[i];
			if(min_y > y[i])
				min_y=y[i];
			if(max_x < x[i])
				max_x=x[i];
			if(max_y < y[i])
				max_y=y[i];
		}

		int area = (max_x - min_x)  *  (max_y - min_y);
		if(area == 0)
		{
			if(min_x == max_x)
				area = max_y - min_y;
			else
				area = max_x - min_x;
		}
		cout<<area <<" \n";


	}
    return 0;
}
