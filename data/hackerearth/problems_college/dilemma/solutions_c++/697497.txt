#include<bits/stdc++.h>
using namespace std;

int arr[5000],arr1[5000],arr2[5000],arr3[25000],arr4[125000];
int main()
{
int t,n,a,b,c;
cin>>t;
while(t--)
{
	cin>>n>>a>>b>>c;
	int i,j;
	for(i=0;i<=a;i++)
arr[i]=1;
	for(i=0;i<=b;i++)
	arr1[i]=1;
	for(i=0;i<=c;i++)
	arr2[i]=1;
	if(n==0)
	cout<<"1"<<endl;
	else
	{
	for(i=0;i<=a;i++)
	{
		for(j=0;j<=b;j++)
		{
			arr3[i+j]=arr3[i+j]+arr[i]*arr1[j];
		}
	}
	//for(i=0;i<=(b+a);i++)
	//cout<<arr3[i];
	for(i=0;i<=c;i++)
	{
		for(j=0;j<=(a+b);j++)
		{
			arr4[i+j]=arr4[i+j]+arr2[i]*arr3[j];
		}
	}
	long long  sum=0;
	for(i=0;i<=n;i++)
	{
	sum=sum+arr4[i];
	}
	cout<<sum<<endl;
	for(i=0;i<=(a+b);i++)
	arr3[i]=0;
	for(i=0;i<=(a+b+c);i++)
	arr4[i]=0;
}	
}
return 0;	
}
