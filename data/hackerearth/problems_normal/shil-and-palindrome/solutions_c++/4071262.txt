#include <bits/stdc++.h>
using namespace std;
#define ll long long int
int main(){
    string s;
    cin>>s;
    ll n=s.length();
    ll f[26]={0};
    string ans=string(n,'0');
    for(int i=0;i<s.length();i++) f[s[i]-'a']++;
    if(n%2==0){
        for(int i=0;i<26;i++){
            if(f[i]%2!=0){
                cout<<-1;
                return 0;
            }
        }
    }
    else{
        ll cnt=0;
        char c;
        for(int i=0;i<26;i++){
            if(f[i]%2!=0) cnt++,c=char(i+'a');
        }
        if(cnt!=1){
            cout<<-1;
            return 0;
        }
        ans[n/2]=c;
        f[c-'a']--;
    }
    ll j=0;
    for(int i=0;i<26;i++){
        while(f[i]>0){
            f[i]-=2;
            ans[j]=char(i+'a');
            ans[n-j-1]=char(i+'a');
            j++;
        }
    }
    cout<<ans;
}
