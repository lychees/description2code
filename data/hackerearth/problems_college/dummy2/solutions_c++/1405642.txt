#include <iostream>
#include<string.h>
#include<bits/stdc++.h>
using namespace std;

char str1[1000000];
int j=0;
char ans[1000000];

int main()
{
    int t;
    char str[25];
    cin>>t;
    while(t--)
    {
    	cin>>str;
    	memset(str1,'\0',sizeof(str1));
    	j=0;
    	for(int i=0;i<strlen(str);i++)
    	{   
    		if(str[i]>='1' && str[i]<='9')
    		{
    			if(i==0)
    				continue;;
    			int k;
				k=str[i]-'0';
				strcpy(ans,str1);
				for(int i=0;i<k-1;i++)
				{
					strcat(str1,ans);
				}
				j=strlen(str1);
				continue;
    		}
    		str1[j]=str[i];
    		j++;
    		str1[j]='\0';
    	
    	}
    	cout<<str1;
    	cout<<endl;
    	
    }
    
    return 0;
}
