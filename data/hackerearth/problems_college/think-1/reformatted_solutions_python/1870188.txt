tc = int(raw_input())
while tc != 0:
    n = str(raw_input())
    if n == str(n[::-1]):
        print "No"
    else:
        m = str(int(n)**2)
        # print m
        k = str(int(n[::-1]))
        # print k
        k = str(int(k)**2)
        # print k
        k = str(int(k[::-1]))
        # print k
        if m == k:
            print str(n[::-1])
        else:
            print "No"
    tc -= 1
