#include <iostream>
#include <cstring>

using namespace std;

long int getSetBits(string s)
{
	long int count = 0;
	for(int i = 0; i < s.length(); i++)
		if(s[i] == '1')
			count++;
	return count;
}

int main()
{
    int t;
    long int n;
    string s;
    cin>>t;
    while(t--)
    {
    	cin>>s;
    	cin>>n;
    	n = (n*(n+1))/2;
    	if(n%2 == 0)
    		cout<<getSetBits(s)<<endl;
    	else
    		cout<<(s.length()-getSetBits(s))<<endl;
    }
    return 0;
}
