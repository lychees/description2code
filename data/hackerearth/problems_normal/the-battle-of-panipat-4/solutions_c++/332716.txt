#include<iostream>
#include<vector>
#include<cstring>
#define MOD 1000000007
using namespace std;
long long res[100000];
int check[100000];
vector<int>cities[100000];
    
void dfs(int current){
    res[current]=1;
    check[current]=1;
    for(int i=0;i<cities[current].size();i++){
        if(check[cities[current][i]]==0){
            //cout<<"dfs: "<<cities[current][i]<<endl;
            dfs(cities[current][i]);
            res[current]=(res[current]*((res[cities[current][i]]+1)%MOD))%MOD;
            res[current]%=MOD;
            //cout<<current<<"\t"<<res[current]<<endl;
        }    
    }    
}    

int main(){
    memset(check,0,sizeof(check));
    int n,u,v;
    cin>>n;
    for(int i=0;i<n-1;i++){
        cin>>u>>v;
        cities[u].push_back(v);
        cities[v].push_back(u);
    }    
    dfs(0);
    long long ans=0;
    for(int i=0;i<=n-1;i++){
        ans=(ans+res[i])%MOD;
    }    
    cout<<(ans+1)%MOD<<endl;
    //system("pause");
    return 0;
}
