#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,j,k,n;
	cin>>n;
	int a[n];
	for(i=0;i<n;i++)
	{
		cin>>a[i];
	}
	int max_ending_here=0,max_so_far=0;
	for(i = 0; i < n; i++)
   {
     max_ending_here = max_ending_here + a[i];
     if (max_ending_here < 0)
        max_ending_here = 0;
     if (max_so_far < max_ending_here)
        max_so_far = max_ending_here;
    }
   
	cout<<max_so_far<<endl;
	return 0;
}
