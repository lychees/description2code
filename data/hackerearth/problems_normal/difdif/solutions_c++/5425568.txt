#include <iostream>
#include <set>
using namespace std;

int main()
{
	string inp;
	cin >> inp;
	set<string> substr;
	string temp;
	for(int i=0;i<inp.size();i++){
		for(int j=i;j<inp.size();j++){
			temp = inp.substr(i,j-i+1);
			substr.insert(temp);
		}
	}
    cout << substr.size() << endl;
    return 0;
}
