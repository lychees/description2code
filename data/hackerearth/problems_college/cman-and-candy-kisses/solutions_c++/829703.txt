#include<bits/stdc++.h>
#define CONST 1000009
#define MOD 1000000007
using namespace std;
long long int fact[CONST];
long long int invfact[CONST];
long long int mod_expo(long long int a,long long int b,long long int p)
{
   long long int ans=1;
   a%=p;
   while(b)
   {
      if(b&1)
      ans=(ans*a)%p;
      b>>=1;
      a=(a*a)%p;
   }
   return ans;
}
void pre_calculate()
{
   fact[1]=1;
   invfact[1]=1;
   for(int i=2;i<CONST;i++)
   {
      fact[i]=(fact[i-1]*i)%MOD;
      invfact[i]=mod_expo(fact[i],MOD-2,MOD);
   }
}
long long int nCr(int n,int r)
{
   if(r==0||r==n)
      return 1;
   if(n<r)
      return 0;
   long long int ans=(fact[n]*((invfact[r]*invfact[n-r])%MOD))%MOD;
   return ans;
}
int main()
{
   pre_calculate();
   int test;
   scanf("%d",&test);
   while(test--)
   {
      int n,R,p=0,x;
      scanf("%d%d",&n,&R);
      for(int i=0;i<n;i++)
      {
         scanf("%d",&x);
         p+=x;
      }
      int num=R-p;
      num+=n-1;
      int den=n-1;
      printf("%lld\n",nCr(num,den));
   }
   return 0;
}
