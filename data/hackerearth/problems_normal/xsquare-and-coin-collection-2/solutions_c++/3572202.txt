#include<iostream>
#include<cstdio>
using namespace std;

long long int arr[1000000];
int main()
{
	long long int t,n,k,i,max,count;
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		for(i=0;i<n;i++)
		cin>>arr[i];
		
		i=0;
		count=0;
		max=0;
		while(1)
		{
			if(i>=n)
			break;
			if(arr[i]<=k)
			{
			count+=arr[i];
			if(count>max)
			max=count;
		    }
		    else
		    {
		    	count=0;
		    }
			i++;
		}
		cout<<max<<"\n";
	}
	return 0;
}
