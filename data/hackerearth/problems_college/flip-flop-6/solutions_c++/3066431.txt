#include <bits/stdc++.h>

using namespace std;

int main()
{
   int t;
   cin >> t;
   string s;
   while(t--){
    cin >> s;
    int no = 0;
    for (int i = 0 ; i < s.size() ; i++){
        char x = s[i];
        int j = i +1;
        for ( j = i+1 ; j < s.size() ; j++){
            if (x == s[j]){no++;}
            else break;
        }
        i = j - 1;
    }
    cout << no << endl;
   }
}
