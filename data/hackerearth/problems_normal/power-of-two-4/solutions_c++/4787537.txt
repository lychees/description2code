#include<iostream>

using namespace std;

#define ll long long int

bool ispower(ll x)
{
	return(x && !(x&(x-1)));
}

int main()
{
	ll t,n,g,p,i,j,ans;
	
	for(i=0;i<32;i++)
		p+=(1<<i);
	
	cin>>t;
	while(t--)
	{
		g=0;
		cin>>n;
		
		ll A[n];
		for(i=0;i<n;i++)
		{
			cin>>A[i];
		}
		
		if(n==1)
		{
			g=ispower(A[0]);
			if(g==1)
				cout<<"YES"<<endl;
			else
				cout<<"NO"<<endl;
		}
		else
		{
			for(i=0;i<32;i++)
			{
				ans=p;
				for(j=0;j<n;j++)
				{
					if(A[j]&(1<<i))
						ans&=A[j];
				}
				if(ispower(ans))
				{
					g=1;
					break;
				}
			}
			
			if(g==1)
				cout<<"YES"<<endl;
			else
				cout<<"NO"<<endl;
		}
	}
	return 0;
	
}
