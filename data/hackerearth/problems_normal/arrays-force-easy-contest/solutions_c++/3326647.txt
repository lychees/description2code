using namespace std;
#define ll long long

#define ppll pair<ll, pair<ll,ll> >
#define inf 1000000000

ll t,a,b,n,i,mod,temp,ans,lim;
vector<ll> fib;

int main()
{
    cin>>t;
    while(t--)
    {
        cin>>a>>b>>n>>mod;
        fib.resize(1000005,0);
        fib[a]++;
        fib[b]++;
        lim=max(a,b);
        for(i=2;i<n;i++)
        {
            temp=(a+b)%mod;
            lim=max(lim,temp);
            fib[temp]++;
            a=b;
            b=temp;
        }
        ans=0;
        for(i=0;i<=lim;i++)
        {
            if(fib[i])
                ans+=fib[i]*fib[i];
        }
        cout<<ans<<endl;
        fib.clear();
    }
}
