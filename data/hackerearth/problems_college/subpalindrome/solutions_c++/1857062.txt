#include<bits/stdc++.h>

using namespace std;

int N;
string S;
vector<vector<long long> > best;
 
long long solve(int zac, int kon) {
long long &res = best[zac][kon];
if (res >= 0) return res;
res = 0;
if (zac == kon) return res = 1;
if (zac+1 == kon) return res = 2;
res = solve(zac,kon-1) + solve(zac+1,kon);
if (S[zac] != S[kon-1]) res -= solve(zac+1,kon-1);
return res;
}
 
int main() {
int T; cin >> T;
while (T--) {
cin >> S;
best.clear();
N = S.size();
best.resize(N+1, vector<long long>(N+1,-1));
cout << solve(0,N)-1 << endl;
}
}
