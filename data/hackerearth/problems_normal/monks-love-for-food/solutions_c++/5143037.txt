#include <bits/stdc++.h>
using namespace std;

int main()
{
    int q;
    stack<int> s;
    cin>>q;
    while(q--)
    {
    	int type;
    	cin>>type;
    	if(type==1)
    	{
    		if(s.size()==0)
    			cout<<"No Food\n";
    		else
    		{
    			cout<<s.top()<<"\n";
    			s.pop();
    		}
    	}
    	else
    	{
    		int x;
    		cin>>x;
    		s.push(x);
    	}
    }
    return 0;
}
