int get_num(){
    int num = 0;
    char c = gc();
    int flag = 0;
    while(!((c>='0' & c<='9') || c == '-'))
        c=gc();
    if(c == '-')
    {
        flag = 1;
        c=gc();
    }
    while(c>='0' && c<='9')
    {
        num = (num<<1)+(num<<3)+c-'0';
        c=gc();
    }
    if(flag==0)
        return num;
    else
        return -1*num;
}
