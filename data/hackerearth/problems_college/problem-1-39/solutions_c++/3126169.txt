#include <bits/stdc++.h>
using namespace std;
int p(int a, int b){
    int tmp = 1;
    while(b--){
       tmp = tmp*a;
       assert(tmp <= 1e9);
    }
    return tmp;
}

int main(){
    int t,a,b,c;
    int y,g,ans,dist;
    cin >>t;
    while(t--){
        scanf("%d %d %d", &a, &b, &c);
        assert(1<=c&& c<=1e9);
        if(b<0)y=(a==1);
        else 
		y = p(a,b);
        g = y%c;
        ans = y - g;
        dist = g;
        g = c - g;
        if(g < dist) ans = y + g;
        printf("%d\n", ans);
    }
}
