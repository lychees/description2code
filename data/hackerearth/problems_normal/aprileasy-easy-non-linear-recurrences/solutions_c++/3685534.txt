#include <bits/stdc++.h>
using namespace std;

const int MOD = 1e9 + 7;

int a, b, c, d, e, f;
int q, n;

int main() {
  scanf("%d%d%d%d%d%d", &a, &b, &c, &d, &e, &f);
  scanf("%d", &q);
  for (int tc = 1; tc <= q; ++tc) {
    scanf("%d", &n);
    int res = f;
    res = (f + ((n - 1) * 1ll * d) % MOD * 1ll * f) % MOD;
    if (n == 0) res = 0;
    printf("%d\n", res);
  }
  return 0;
}
