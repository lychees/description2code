#include <iostream>
#include <bitset>
using namespace std;

int main()
{
    int t;
    long long a,b,temp,c;
    cin>>t;
    while(t--){
    	cin>>a>>b;
    	c=a^b;
    	std::bitset<32> ones (c);
    	temp=ones.count();
    	if(temp%2==0)
    	cout<<"YES"<<endl;
    	else
    	cout<<"NO"<<endl;
    }
    return 0;
}
