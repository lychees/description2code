ll mod_pow(ll a,ll b,ll mod) {
	// a ^ b % mod;
	if(b == 0) return 1;
	ll p;
	if(b % 2 == 0) {
		p = mod_pow(a,b / 2,mod);
		p = (p * p + mod) % mod;
	}
	else p = (a * mod_pow(a,b - 1,mod) + mod) % mod;

	return p;
}

int main() {
	ios_base::sync_with_stdio(false);

	int t; cin >> t;
	while (t --) {
		ll n,mod; cin >> n >> mod;

		if(3 * n >= mod) {
			cout << 0 << '\n';
			continue;
		}
		
		ll ft = mod_pow(mod_pow(6,n,mod),mod - 2,mod);

		ll st = mod - 1;
		for (ll i = 3 * n + 1; i < mod; i ++)
			st = (st * i + mod) % mod;
		
		st = mod_pow(st,mod - 2,mod);

		cout << (ft * st + mod) % mod << '\n';
	}
	return 0;
}
