#include<bits/stdc++.h>
using namespace std;
int main()// d should be less than i
{
    long long  t;
    cin>>t;
    long dp[33][33];
    memset(dp,0,sizeof(dp));
    for(long long  i=0;i<=31;i++)
        dp[0][i]=1;

    for(long long  d=1;d<=31;d++)
    {
        for(long long  i=1;i<=31;i++)
        {
            if(d>i)
            {
                dp[d][i]=0;
                continue;
            }
            dp[d][i]+=dp[d-1][i];
            if(d<i)
                dp[d][i]+=dp[d][i-1];
        }
    }
    while(t--)
    {
        long long  n;
        cin>>n;
        cout<<dp[n][n]<<endl;
    }
}
