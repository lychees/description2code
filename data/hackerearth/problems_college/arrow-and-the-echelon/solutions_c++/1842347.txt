#include<iostream>
using namespace std;
int main()
{
	int t,n,i;
	cin>>t;
	while(t--)
	{
		cin>>n;
		if(n%3==0)
		{
			for(i=0;i<n/3;i++)
				cout<<"555";
		}
		else
		{
			int x3=0,x5=0;
			while(n>0)
			{
				n-=5;
				x5++;
				if(n%3==0)
				{
					x3=n/3;
					break;
				}
		    }
				if(n%3==0)
				{
					for(i=0;i<3*x3;i++)
						cout<<"5";
					for(i=0;i<5*x5;i++)
						cout<<"3";
				}
				else
				{
					cout<<"-1";
				}
		}
		cout<<endl;
	}
}
