{
   int t;
   cin>>t;
   while(t--)
   {
   	long long int n,max1,min1;
   	cin>>n;
   	long long int a[n],b[n];
   	for(int i=0;i<n;i++)
   	{
   	cin>>a[i];
   	b[i]=a[i];
   	}
   	sort(a,a+n);
   	sort(b,b+n);
   	reverse(b,b+n);
   	for(int i=0;i<n;i++)
   	a[i]=a[i]+b[i];
   	max1=*max_element(a,a+n);
   	min1=*min_element(a,a+n);
   	cout<<max1-min1<<endl;
   }
    return 0;
}
