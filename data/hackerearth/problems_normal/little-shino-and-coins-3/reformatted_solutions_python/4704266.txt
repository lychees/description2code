K = int(raw_input())
S = raw_input()
out = 0
n = len(S)
for i in xrange(n):
    str_ = S[i]
    cnt = 0
    for j in xrange(i, n):
        if (S[j] not in str_):
            cnt += 1
            if (cnt > K):
                break
            else:
                str_ += S[j]
        if len(str_) == K:
            out += 1
print out
