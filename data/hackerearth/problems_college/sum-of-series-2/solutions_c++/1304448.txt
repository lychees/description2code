#include <iostream>
#include <cmath>
#define mod 1000000007
#define P 1001000
using namespace std;

int fact[2000005],prime[2000005],res[2000005];

int power(long long p,long long q){
	long long ans=1;
	while(q){
		if(q&1) ans=(ans*p)%mod;
		p=(p*p)%mod;
		q/=2;
	}
	ans=ans%mod;
	return ans;
}
int func(int k){
    int num,den,ans;
    num=(1L*k*(k-1))%mod;
    num=(1L*num*fact[2*k-2])%mod;
    den=(1L*fact[k-2]*fact[k])%mod;
    //cout<<num<<" "<<den<<" "<<endl;
    ans= (1L*num*power(den,mod-2))%mod;
    //cout<<ans<<endl;
    num=(1L*k*fact[2*k-1])%mod;
    den=(1L*fact[k-1]*fact[k])%mod;
    //cout<<num<<" "<<den<<" "<<endl;
    ans=(ans+((1L*num*power(den,mod-2))%mod))%mod;
    return ans;
}

void pre(){
    int i,j;
    fact[0]=1;
    for( i=1;i<2000001;i++)
        fact[i]=(long (i%mod)*(fact[i-1]%mod))%mod;


    for(i=3;i<=P;i+=2) prime[i]=1;
	prime[2]=1;
	for(i=3;i<=sqrt(P);i+=2){
		if(prime[i]) for(j=i*i;j<=P;j+=i+i) prime[j]=0;
	}

	res[0]=res[1]=0;
	for(i=2;i<=1000001;i++){
		if(prime[i])
            res[i]=(long(res[i-1]+func(i)))%mod;
        else
            res[i]=res[i-1];
    }
}
int main(){
	pre();
	int t,n;
	cin>>t;
	while(t--){
	 cin>>n;
	cout<<res[n]<<endl;
	}
}
