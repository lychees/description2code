from sys import stdin
d = {}
x = map(int, stdin.readline().rstrip("\n").split())
for i in range(97, 123):
    d[chr(i)] = x[i-97]
q = int(stdin.readline())
m = 0
k = 0
for i in range(0, q):
    s1 = stdin.readline().rstrip("\n")
    s2 = stdin.readline().rstrip("\n")
    for x in s1:
        m += d[x]
    for x in s2:
        k += d[x]
# print k,m
if k > m:
    print "Devil"
elif k == m:
    print "Draw"
else:
    print "Marut"
