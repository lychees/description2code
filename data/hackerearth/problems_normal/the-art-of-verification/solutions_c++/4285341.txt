#include<iostream>
#include<string>
using namespace std;
int main()
 {
    string url;
    cin>>url;
    int l=url.size();
    string user="", password="", profile="", role="",key="";
    int flag=0;
    for(int i=0;i<l;i++){
        if(url[i]=='u'&&url[i+1]=='s'&&url[i+2]=='e'&&url[i+3]=='r'&&url[i+4]=='n'&&url[i+5]=='a'&&url[i+6]=='m'&&url[i+7]=='e'&&url[i+8]=='='){
            i+=9;
            while(!(url[i]=='&'&&url[i+1]=='p'&&url[i+2]=='w'&&url[i+3]=='d'&&url[i+4]=='=')){
                user+=url[i];
                i++;
            }
            i+=5;
            while(!(url[i]=='&'&&url[i+1]=='p'&&url[i+2]=='r'&&url[i+3]=='o'&&url[i+4]=='f'&&url[i+5]=='i'&&url[i+6]=='l'&&url[i+7]=='e'&&url[i+8]=='=')){
                password+=url[i];
                i++;
            }
            i+=9;
            while(!(url[i]=='&'&&url[i+1]=='r'&&url[i+2]=='o'&&url[i+3]=='l'&&url[i+4]=='e'&&url[i+5]=='=')){
                profile+=url[i];
                i++;
            }
            i+=6;
            while(!(url[i]=='&'&&url[i+1]=='k'&&url[i+2]=='e'&&url[i+3]=='y'&&url[i+4]=='=')){
                role+=url[i];
                i++;
            }
            i+=5;
            flag=1;
        }
        if(flag)
            key+=url[i];
    }
    cout<<"username: "<<user<<endl;
    cout<<"pwd: "<<password<<endl;
    cout<<"profile: "<<profile<<endl;
    cout<<"role: "<<role<<endl;
    cout<<"key: "<<key<<endl;
	return 0;
}
