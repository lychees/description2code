#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int N,Q,t,a,b;
    cin>>N>>Q;
    vector<bool> vb(N,false);
    for(int q=0;q<Q;q++){
    	cin>>t>>a>>b; b++;
    	if(t){
    		int res = 0; 
    		for(int i=a;i<b;i++) if(vb[i])res++;
    		cout<<res<<endl;
    	}
    	else{
    		for(int i=a;i<b;i++) vb[i] = !vb[i];
    	}
    }
    return 0;
}
