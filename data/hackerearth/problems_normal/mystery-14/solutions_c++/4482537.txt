#include <iostream>
using namespace std;

int main()
{
    int t,ar[100000],i,n;
    for(scanf("%d",&t);t--;)
        {   cin>>n;
        	for(i=0;i<n;i++)
               cin>>ar[i];
            int sum=0;
           for(i=0;i<n;i++)
               { if((ar[i]%2)!=0)
                    sum+=ar[i];
               }
              cout<<sum<<'\n';
        }
    return 0;
}
