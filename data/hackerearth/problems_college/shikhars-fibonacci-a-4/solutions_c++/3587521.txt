#include <iostream>
using namespace std;


int main()
{
    long a,b,n;
    cin>>a>>b>>n;
    long c;
    if(n==1)
    cout<<a<<endl;
    else if(n==2)
    cout<<b<<endl;
    else
    {
    	n=n-2;
    	while(n>0)
    	{
    		c=a+b;
    		a=b;
    		b=c;
    		n--;
    	}
      cout<<c<<endl;
    }
    return 0;
}
