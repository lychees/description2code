list1 = []
for i in range(0, 10):
    a = input()
    list1.append(a)


i = 0
list2 = []
length = len(list1)
while i < length:
    list2.append(list1[i]-1)

    for j in range(length-1, i, -1):
        if list1[i] == list1[j]:
            del list1[j]
            length -= 1
    i += 1

for numbers in sorted(list2):
    print numbers
