#include <bits/stdc++.h>
#define LL long long int
using namespace std;
LL dp[1000002];
int phi(int n) {
	int result = n;
	for(int i = 2; i * i <= n; ++i)
		if(n % i == 0) {
			while(n % i == 0)
				n /= i;
			result -= result / i;
		}
	if(n > 1)
		result -= result / n;
	return result;
}
void com(){
	dp[1]=0;dp[0]=0;
	for(LL i=2;i<1000001;i++)
			dp[i]=dp[i-1]+phi(i);
	return ;
}
int main(){
	LL t,n;
	com();
	ios_base::sync_with_stdio(false); cin.tie(NULL);
	cin >> t;
	while(t--){
		cin  >> n;
		cout << (n*(n-1))/2 -  (dp[n]) << "\n";
	}
	return 0;
}
