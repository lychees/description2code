#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	long long int a[10000],n;
	long long int dp[10000];
	cin >> t;
	for(int k=1; k <=t; k++)
	{
		cin >> n;
		for(int i=0; i < n; i++)
		{
			 cin >> a[i] ;
		}
		
		long long int m=max(a[0],a[1]);
		dp[0]=a[0];
		dp[1]=m;
		for(int i=2; i < n; i++)
		{
			dp[i]=max(dp[i-1],a[i]+dp[i-2]);
		}
		cout << "Case " << k  <<": " << dp[n-1] <<"\n";
	
	}
    //cout << "Hello World!" << endl;
    return 0;
}
