#include <iostream>
using namespace std;

int main()
{
	int i,t;
     cin >> t;
    int ans=1,c1=0,c2=2;
    int arr[100001];
    arr[0]=0;
    for(i=1;i<=100000;i++)
    {
      if(c1<c2)
      {
      arr[i]=ans;
      c1++;
      }
      else
      {
      	c1=1;
      	c2=c2+1;
      	arr[i]=++ans;
      }
    }
    for(i=0;i<t;i++)
    {
    	int n;
    	cin >> n;
        cout <<arr[n]<<endl;
    }
    return 0;
}
