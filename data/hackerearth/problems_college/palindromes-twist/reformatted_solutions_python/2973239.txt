#!usr/bin/python
t = input()
while (t > 0):
    a = raw_input()
    ac = input()
    bc = input()
    l = len(a)
    s = 0
    f = 0
    for i in range(0, (l / 2)):
        if (a[i] == '/' and a[l - i - 1] == '/'):
            s += (min(ac, bc) * 2)
        elif ((a[i] == 'a' and a[l - i - 1] == '/') or (a[i] == '/' and a[l - i - 1] == 'a')):
            s += ac
        elif ((a[i] == 'b' and a[l - i - 1] == '/') or (a[i] == '/' and a[l - i - 1] == 'b')):
            s += bc
        elif (a[i] != a[l - i - 1]):
            f = 1
    if (s == 0 or f == 1):
        print -1
    else:
        print s
    t -= 1
