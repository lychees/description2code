'''
# Read input from stdin and provide input before running code

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
if __name__ == '__main__':
    input_numbers = [int(x) for x in raw_input().strip().split(' ')]
    N = input_numbers[0]
    a = input_numbers[1]
    b = input_numbers[2]
    start = a
    count = 0
    while start < N:
        if start % a == 0 or start % b == 0:
            count = count + start
        start = start + 1
    print count
