#include <iostream>
using namespace std;

int main()
{
    int n,temp,count=0,arr[3]={0};
    bool left = false;
    cin>>n;
    for(int i=0;i<n;i++)
    {
   		cin>>temp;
   		if(temp == 4)count++;
   		else arr[temp-1]++;
    }
    
    count+=arr[2];
    if(arr[2] < arr[0])
    {
    	arr[0]-=arr[2];
    	arr[1]+=arr[0]/2;
    	if(arr[0] % 2)left=true;
    }
    count+=arr[1]/2;
    if(arr[1]%2)left=true;
    
    if(left)count++;
    cout<<count<<endl;
    return 0;
}
