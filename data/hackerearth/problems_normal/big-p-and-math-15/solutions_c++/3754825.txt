#include <iostream>
using namespace std;

int main()
{
 int a,b;
 cin>>a>>b;
 int sum=a+b;
 int mul=1;
 int temp5=0;
 int temp6=0;
 while(a>0)
 {
 	int x=a%10;
 	if(x==5)
 	temp5+=mul;
 	if(x==6)
 	temp6+=mul;
 	a/=10;
 	mul*=10;
 }
 	mul=1;
 	while(b>0)
 	{
 		int x=b%10;
 		if(x==5)
 		temp5+=mul;
 		if(x==6)
 		temp6+=mul;
 		b/=10;
 		mul*=10;
 	}
 	cout<<sum-temp6<<" "<<sum+temp5<<endl;
    return 0;
}
