#include <iostream>
using namespace std;

int main()
{
    int n,a,b;
    cin>>n;
    int x;
    int mat[n+1][n+1];
    int bound = (1<<n);
    for(int i=0;i<bound;i++){
    	for(int j=0;j<bound;j++){
    		x = 0;
    		a = i;
    		b = j;
    		for(int k=0;k<2*n;k++){
    			x = (1<<k)*(b&1) + x;
    			k++;
    			x = (1<<k)*(a&1) + x;
    			a = a>>1;
    			b = b>>1;
    		}	
    		cout<<(x+1)<<" ";
    	}
    	cout<<endl;
    }
    return 0;
}
