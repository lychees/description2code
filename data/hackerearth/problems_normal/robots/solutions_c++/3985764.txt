#include <iostream>
#include <vector>
#include <algorithm>
#define ll long long
using namespace std;
const ll MOD = 1e9+7;
int main(void){
	int T;
	cin >> T;
	while(T--){
		int N;
		cin >> N;
		ll result=1;
		int temp;
		vector <ll> input;
		for(int i=0;i<N;i++){
			cin >> temp;
			input.push_back(temp);
			//result = (result * (temp-i))%MOD;
		}
		sort(input.begin(),input.end());
		for(int i=0;i<N;i++){
			//cin >> temp;
			//input.push_back(temp);
			result = (result * (input[i]-i));
			result = result % MOD;
		}
		cout << result % MOD << endl;

	}
}
