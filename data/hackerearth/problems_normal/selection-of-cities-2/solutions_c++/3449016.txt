using namespace std;
typedef long long ll;


ll get_mod(ll a,ll b,ll modd)
{
	ll res = 1;
	while (b > 0)
	{
		if (b % 2 == 1)
			res = (res * a) % modd;
		b = b / 2;
		a = (a*a) % modd;
	}
	return res;
}

int main()
{
    ll t,n; cin>>t;
    while (t--)
    {
        cin>>n;
        cout<<get_mod(2,n,1000000007)-1<<endl;
    }
    return 0;
}
