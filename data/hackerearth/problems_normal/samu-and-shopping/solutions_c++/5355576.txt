#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		long long a[n][3];
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<3;j++)
			{
				cin>>a[i][j];
			}
		}
		for(int i=1;i<n;i++)
		{
			a[i][0] += min(a[i-1][1],a[i-1][2]);
			a[i][1] += min(a[i-1][0],a[i-1][2]);
			a[i][2] += min(a[i-1][0],a[i-1][1]);
		}
		long long m = min(a[n-1][0],a[n-1][1]);
		m = min(m,a[n-1][2]);
		cout<<m<<endl;
	}
    return 0;
}
