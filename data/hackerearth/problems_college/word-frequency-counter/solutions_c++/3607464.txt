#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s;
    map<string,bool>done;
    map<string,int>M;
    vector<string>v;
    while(cin>>s){
    	int n = s.size();
    	for(int i=0;i<n;i++){
    		if(s[i]>='a'&&s[i]<='z'){
    			s[i] = 'A' + (s[i]-'a');
    		}
    	}
    	M[s]++;
    	v.push_back(s);
    }
    for(auto it : v){
    	if(done[it]==false){
    		cout << it << " " << M[it] << endl;
    		done[it]=true;
    	}
    }
    return 0;
}
