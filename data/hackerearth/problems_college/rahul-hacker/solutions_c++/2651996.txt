#include <iostream>
using namespace std;

int main()
{
    int t,rev=0,v=0,re=0,i;
    long int a,b,c;
    cin>>t;
    for(i=0;i<t;i++)
    {
    	rev=v=re=0;
    	cin>>a>>b;
    	while(a)
    	{rev=rev*10;
    	rev=rev+a%10;
    	a=a/10;
    	
    	}
    	
    	while(b)
    	{v=v*10;
    	v=v+b%10;
    	b=b/10;
    	
    	}
    	c=rev+v;
    	while(c)
    {
    	re=re*10;
    	re=re+c%10;
    	c=c/10;
    }
    cout<<re<<"\n";	
    }

return 0;
}
