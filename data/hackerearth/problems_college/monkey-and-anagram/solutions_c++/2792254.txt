#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int n,i,j,t;
	string s1,s2;
	cin>>t;
	while(t--)
	{
		int c1[200]={0},c2[200]={0},c=0,f=0;
		cin>>s1>>s2;
		for(i=0;i<s1.size();i++)
		{
			c1[s1[i]]++;
		}
		for(i=0;i<s2.size();i++)
		{
			c2[s2[i]]++;
		}
		for(i=0;i<200;i++)
		{
			if(c1[i]>c2[i])
			{
				cout<<"-1"<<endl;
				f=-1;
				break;
			}
			else if(c1[i]<c2[i])
			{
				c++;
			}
		}
		if(f!=-1)
		cout<<c<<endl;
	}
	return 0;
}
