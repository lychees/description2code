#include <iostream>
using namespace std;

int main()
{
	int t,n,i,j,cnt;
	
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	int a[n];
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	int cnt=0;
    	for(i=0;i<n;i++)
    	{
    		for(j=i;j<n;j++)
    		{
    			if(a[i]==a[j]) cnt++;
    		}
    	}
    	cout<<cnt<<endl;
    }
}
