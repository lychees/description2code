#include <bits/stdc++.h>
using namespace std;

double f1(double d , int n){
	d = int(d*pow(10,n));
	return d /=pow(10,n);
}
int main()
{
	int n;
	cin >> n;
	double a[n],b[n];
	for(int i=0;i<n;i++){
		cin >> a[i];
	}
	for(int i=0;i<n;i++){
		cin >> b[i];
	}
	double f;
	cin >> f;
	double res=0;
	for(int i=0;i<n;i++){
			res = max(res,(f*a[i])/b[i]);
	}
	cout << fixed << setprecision(3) << f1(res,3) << endl;
    return 0;
}
