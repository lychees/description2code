#include<bits/stdc++.h>
#define pb push_back
#define fi first
#define se second
#define endl '\n'
typedef long long LL;
using namespace std;

int n, k;
int sz[111111];
int par[111111];
bool mark[111111];

int get(int v){
    if(par[v] == -1) return v;
    else{
        par[v] = get(par[v]);
        return par[v];
    }
}

void unite(int v, int u){
    v = get(v);
    u = get(u);
    if(u == v) return;
    sz[v] += sz[u];
    par[u] = v;
    mark[u] = true;
    return;
}

int main(){
	ios_base::sync_with_stdio(false);

    cin >> n;
    cin >> k;

    for(int i = 1; i <= n; i++){
        sz[i] = 1;
        par[i] = -1;
    }

    while(k--){
        int x, y;
        cin >> x >> y;
        unite(x, y);
    }

    int cnt = 0;
    for(int i = 1; i <= n; i++)
        if(!mark[i]) cnt++;

    cout << cnt;

	return 0;
}
