def winner(N):
    ls = [1]
    for num in range(2, N + 1):
        if num > 1:
            for i in range(2, num):
                if (num % i) == 0:
                    break
        else:
            ls.append(num)
    if len(ls) % 2 == 0:
        return '2'
    else:
        return '1'


for i in xrange(int(raw_input())):
    print winner(int(raw_input()))
