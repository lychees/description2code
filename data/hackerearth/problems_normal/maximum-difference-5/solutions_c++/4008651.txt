#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n , a[1000] , b[1000] , c[1000];
    scanf("%d",&n);

    for(int i = 0 ; i < n ; i++)
    	scanf("%d",&a[i]);

    sort(a, a+n);

    int st = 0 , end = n-1 , ind = 0 ;
	long diff1 = 0 , diff2 = 0 ;

    while(st < end) {
        b[ind] = a[st] ;
        st++; ind ++ ;
        b[ind] = a[end];
        end--; ind ++ ;
    }
    if(st == end) b[ind] = a[st] , ind++ ;

    for(int i = 0 ; i < ind-1 ; i++) {
    	 diff1 += fabs(b[i] - b[i+1]);
    }

    st = 0 , end = n-1 , ind = 0 ;
	while(st < end) {
        c[ind] = a[end] ;
        end--; ind ++ ;
        c[ind] = a[st];
        st++ ; ind ++ ;
    }
    if(st == end) c[ind] = a[st] , ind++ ;
    
    for(int i = 0 ; i < ind-1 ; i++) {
    	 diff2 += fabs(c[i] - c[i+1]);
    }

    if(diff1 < diff2) printf("%ld\n",diff2);
    else printf("%ld\n",diff1);

    return 0;
}
