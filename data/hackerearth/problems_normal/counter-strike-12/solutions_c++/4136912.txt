#include <iostream>
#include<vector>
#include<algorithm>
#include<cmath>
using namespace std;

int main()
{
    int t;
    cin>>t;
   // cout<<t<<endl;
    while(t--)
    {
    	int n,m,d;
    	cin>>n>>m>>d;
    
    	int nx[n];
    
    
    	int ny[n];
    	
    	
    vector<int> mx;
    
    	vector<int> my;
    	
    
    	
    	for(int i=0;i<n;i++)
    	cin>>nx[i]>>ny[i];
    	
    	for(int j=0;j<m;j++)
   		{	int r;
   			cin>>r;
   			mx.push_back(r);
   			cin>>r;
   			my.push_back(r);
   		}
   		
   			m=floor(m/2);
    vector<int> z;
    
    //int siz=mx.size();
    for(int p=0;p<n;p++)
   {
    	for(int q=0;q<mx.size();q++)
    	{
    		if((abs(nx[p]-mx[q])+abs(ny[p]-my[q]))<=d)
    		{
    			--m;
    			//siz=mx.size();
    			z.push_back(q);
    		}
    	}
    for(int k=0;k<z.size();k++)
    {
    	mx.erase(mx.begin()+z[k]-k);my.erase(my.begin()+z[k]-k);
    }
    z.clear();
    }
    if(m<=0)
    cout<<"YES"<<endl;
    else
    cout<<"NO"<<endl;
    }
    return 0;
}
