#include <bits/stdc++.h>

using namespace std;

int main()
{
	int a[100005];
	a[0] = 0, a[1] = 0;
	a[2] = 0;
	a[3] = 0;
	a[4] = 0;

	for (int i = 5; i <= 100000; i++) {
		int s[33];
		int k = 0;
		int p = i;
		while (p != 0) {
			if (p & 1) {
				s[k++] = 1;
			} else {
				s[k++] = 0;
			}
			p = p / 2;
		}
		/*for (int j = 0; j < k; j++) {
			printf("%d",s[j]);
		}*/
		int ans = 0;
		for (int j = 0; j <= k - 3; j++) {
			if (s[j] == 1 && s[j+1] == 0 && s[j+2] == 1) {
				ans++;
			}
		}
		a[i] = ans;
	}
	//printf("%d\n",a[5]);
	int t,R,K;
	scanf("%d",&t);
	while (t --) {
		scanf("%d %d",&R,&K);
		int ans = 0;
		for (int i = 1; i <= R; i++) {
			if (a[i] >= K) {
				ans++;
			}
		}
		printf("%d\n",ans);
	}
	
	return 0;
}
