#include<cstdio>
#include<iostream>
#include<cstring>
int main()
{
    int i,j,k,l,m,n,s,t;
    char a[10001],b[10001],c[10001];
    for(i=0;i<10001;i++)
        if(i%2==0)
            a[i]='N';
        else
            a[i]='Y';
    for(i=0;i<10001;i++)
        if(i%2==0)
            b[i]='Y';
        else
            b[i]='N';
    scanf("%d",&t);
    while(t--)
    {
        s=0;
        k=0;
        scanf("%d",&l);
        scanf("%s",c);
        for(i=0;i<l;i++)
            if(a[i]!=c[i])
                s++;
        for(i=0;i<l;i++)
            if(b[i]!=c[i])
                k++;
        if(k>s)
            printf("%d\n",s);
        else
            printf("%d\n",k);
    }
    return 0;
}
