#include <iostream>
using namespace std;
typedef long long int lli;
int main()
{
	int t;
	cin>>t;
	while(t--){
		lli a, b, m;
		cin>>a>>b>>m;
		lli ans = 0;
		
		if(a%m != 0){
			a = a/m;
			a++;
			a = a*m;
		}
		
		if(b%m != 0){
			b = b/m;
			b = b*m;
		}
		
		ans = (b-a)/m;
		ans++;
		
		cout<<ans<<endl;
	}
    return 0;
}
