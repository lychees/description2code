inline ll scan(){
    char c = getchar_unlocked();
    register ll x = 0;
    bool neg=false;
    if(c=='-'){
        neg=true;
    }
    while(c<'0'||c>'9'){
        c=getchar_unlocked();
    }
    while(c>='0'&&c<='9'){
        x=(x<<1)+(x<<3)+c-'0';
        c=getchar_unlocked();
    }
    if(neg==false) return x;
    else if(neg==true) return -1*x;
}

ll n,m;
ll ar[1123456],arsz[123];ll sz[102][1002],dp[2004];

int main(){
	n=scan();m=scan();
	register ll i,j,k,l,len,p,q,x,y;
	for(i=1;i<=n;i++){
		arsz[i]=scan();ar[0]=0;
		for(j=1;j<=arsz[i];j++){
			q=scan();
			ar[j]=ar[j-1]+q;
		}
		for(l=1;l<=min(arsz[i],m);l++){
			k=0;j=l;
			while(j<=arsz[i]){
				sz[i][l]=max(sz[i][l],ar[j]-ar[k]);j++;k++;
			}
			//cout<<sz[i][l]<<" ";
		}
		//cout<<endl;
	}
	reset(dp,0);
	for(i=1;i<=n;i++){
		for(j=m;j>=0;j--){
			for(l=0;l<=m;l++){
				if(j>=l){
					dp[j]=max(dp[j],dp[j-l]+sz[i][l]);
				}
			}
		}
	}
	cout<<dp[m]<<endl;
	return 0;
}
