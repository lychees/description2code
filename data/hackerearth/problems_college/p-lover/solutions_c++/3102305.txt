#include<stdio.h>

int isp(int n){
	int i;
	for(i = 2; i * i <= n; i++)	if(n % i == 0)	return 0;
	return 1;
}

int main(){
	int t, n, cp, s; scanf("%d", &t);
	while(t-- && scanf("%d", &n)){
		while(n){
			cp = n;
			for(s = 0; cp != 0; s = (s * 10) + (cp % 10), cp /= 10);
			if(s == n && isp(n)){printf("%d\n", s); break;}	
			n++;
		}
	}		
	return 0;
}
