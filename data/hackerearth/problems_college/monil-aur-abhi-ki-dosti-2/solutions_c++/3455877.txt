#include<bits/stdc++.h>

#define MAX 1000005
#define ll long long int
#define PI 3.14159265359
#define MOD 1000000007
#define SET(x,y) memset(x,y,sizeof(x));

using namespace std;

char p[1005],s[100005];

int a[200],b[200];

int main(){

	ll i,n,len,flag=0,t,j;
	

	scanf("%lld",&t);

	while(t--){

		flag=0;

	scanf("%s",p);
	scanf("%s",s);

	
	SET(a,0);
	SET(b,0);
    
    n=strlen(p);
    len=strlen(s);

    if(n>len) printf("NO\n");
    else{

    for(i=0;i<n;i++)
    	b[(int)(p[i])]++;
    
	for(i=0;i<n;i++)
		a[(int)(s[i])]++;

	for(i=97;i<=122;i++)
		if(a[i]!=b[i])
			flag=1;
    

    if(flag!=0){

    	a[(int)(s[0])]--;

    for(i=n;i<len;i++){
    	flag=0;

    	a[(int)(s[i])]++;

    	for(j=97;j<=122;j++)
    		if(a[j]!=b[j])
    			flag=1;

        if(flag==0) break;

        a[(int)(s[i-n+1])]--;
    }
  }

  if(flag==1) printf("NO\n");
  else
  	printf("YES\n");
   }
 }
	return 0;
}
