#include <iostream>
using namespace std;

long long int gcd(long long int a,long long int b)
{
	if(a%b==0)
	return b;
	else
	return gcd(b,a%b);
}

int factors(long long int x)
{
	int ans = 0;
	for(int i=1;i*i<=x;i++)
	{
		if(x%i==0)
		{
			if(i*i == x)
			ans++;
			else
			ans+=2;
		}
	}
	return ans;
}

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int n,r;
		cin>>n>>r;
		r=r-1;
		int f = factors(r);
		r = n-f;
		long long int g = gcd(r,n);
		cout<<r/g<<"/"<<n/g<<endl;
	}
    return 0;
}
