#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	long int n;
    	scanf("%ld",&n);
    	int count=0;
    	while(n!=1)
    	{
    		if(n%3==0)
	    	{
	    		n=n/3;
	    		count++;
	    	}
	    	else if(n%2==0)
	    	{
	    		n=n/2;
	    		count++;
	    	}
	    	else
	    	{
	    		n--;
	    		count++;
	    	}
    	}
    	cout<<count<<endl;
    }
}
