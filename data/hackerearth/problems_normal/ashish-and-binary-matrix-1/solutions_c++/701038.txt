#include <stdio.h>
#include <iostream>
#include <set>
#include <string>
using namespace std;

int getLog2(int n){
	int log=0;
	for(;n>1;++log,n=n/2);
	return log;
};

int main()
{
    int nTestCases;
    scanf("%d",&nTestCases);
    for(;nTestCases >0;--nTestCases){
    	int n=0,m=0;
    	scanf("%d %d",&n, &m);
    	std::set<std::string> strMap;
    	bool repeat=false;
    	for(int i=0;i<n;++i){
    		std:string curStr;
    		std::cin >> curStr;
    		if(strMap.find(curStr)==strMap.end()){
    			strMap.insert(curStr);
    		}else{
    			repeat=true;
    		}
       	}
    	
    	if(repeat){
    		printf("No\n");
    		continue;
    	}
    	int log2= getLog2(n);
    	//printf("%d ",log2);
    	if((m-1)<log2){
    		printf("No\n");
    	}else{
    		printf("Yes\n");
    	}
    }
    return 0;
}
