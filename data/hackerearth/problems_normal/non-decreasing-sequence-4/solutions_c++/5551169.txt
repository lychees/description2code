#include<iostream>
using namespace std;
int main()
{
    long long int t,n,tl,i,a[1000002];
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(i=0;i<n;i++)
            cin>>a[i];

        long long int mini[n+1];
        cin>>tl;
        mini[0]=a[0];
        long long int len=1;
        for(i=1;i<n;i++)
        {
            if(a[i]<mini[0])
            {
                mini[0]=a[i];
            }
            else if(mini[len-1]<a[i])
            {
                if(len<tl)
                {
                    mini[len]=a[i];
                    len++;
                }
            }
            else
            {
                long long int l=-1,h=len-1,m;
                while(l<h)
                {
                    m=l+(h-l)/2;
                    if(mini[m]<a[i])
                    {
                        l=m+1;
                    }
                    else
                    {
                        h=m;
                    }
                }
                mini[h]=a[i];
            }
        }
        if(len<tl)
            cout<<"-1"<<endl;
        else
            cout<<mini[len-1]<<endl;

    }
}
