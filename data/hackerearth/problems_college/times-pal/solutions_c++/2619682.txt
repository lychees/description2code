#include<bits/stdc++.h>
using namespace std;
bool palin(string s)
{
	int len=s.length();
	for(int i=0,j=len-1; i < j ; i++,j--)
	{
		if(s[i]!=s[j])
		return false;
		
	}
	return true;
}
int main()
{
	int t;
	string s1,s2;
	cin >>t;
	while(t--)
	{
		cin >> s1;
		cin >> s2;
		int count=0,len=s1.length();
		if(palin(s2+s1))
		count++;
		for(int i=1; i < len; i++)
		{
			string s3=s1.substr(0,i);
			string s4=s1.substr(i,len-i);
			if(palin(s3+s2+s4))
			count++;
		}
		if(palin(s1+s2))
		count++;
		cout <<count <<"\n";
	}
}
