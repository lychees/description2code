#define MOD 1000000007
int a[1111111];

int main(void){
	std::ios::sync_with_stdio(false);cin.tie(0);
	int T;
	int i,j,n;
	for(i=2;i<=1000000;i++)
		a[i]=1;
	for(i=2;i<=1000000;i++)
	{
		for(j=i+i;j<=1000000;j+=i)
			a[j]++;
	}
	for(i=2;i<=1000000;i++)
		re[i]=re[i-1]+a[i];
	cin>>T;
	while(T--)
	{
		cin>>n;
		cout<<re[n]<<'\n';
	}
//	system("pause");
	return 0;
}
