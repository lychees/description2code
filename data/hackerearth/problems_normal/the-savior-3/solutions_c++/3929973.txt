#include<iostream>

using namespace std;

int main()
{
	int t,i,j;
	cin>>t;
	while(t--)
	{
		int n,k=0;
		cin>>n;
		int a[n];
		for(i=0;i<n;i++)
		{
			cin>>a[i];
		}
		for(i=0;i<n;i++)
		{
			for(j=i+1;j<n;j++)
			{
				int c=0;
				if(a[i]!=a[j])
				{
				c=a[i]+a[j];
				if(c%2==0)
				k++;
			    }
			}
		}
		cout<<k<<endl;
	}
	return 0;
}
