#include <iostream>
#include <cmath>
#include <cstdio>
#include <vector>
#include <algorithm>
using namespace std;

char find(long long k)
{
	char x;
	if(k%4!=0)
	{
		if((k/4)%2==0)
		{
			if(k%4==3)
			x='C';
			else
			x='G';
		}
		else
		{
			if(k%4==1)
			x='G';
			else
			x='C';
		}
	}
	else
	{
		k=k/4;
		x=find(k);
	}
	return x;
}

int main()
{
	long long t,n,k;
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		char x;
		if(k>(pow(2,n)-1))
		cout<<"-1\n";
		else
		{
		x=find(k);
		cout<<x<<"\n";
		}
	}
return 0;
}
