#include<iostream>
#include<vector>

using namespace std;

int main()
{
    int t;
    cin>>t;

    while(t!=0)
    {
    int n;
    cin>>n;

    string s;
    cin>>s;

    vector<int> freq(26,0);

    for(int i=0;i<s.size();i++)
        freq[s[i]-'a']++;

    int Max=0;
    int Size=999999999;
    int ans=0;

    for(int k=1;k<=n;k++)
    {
        string ctr;
        cin>>ctr;

        int temp=0;

        for(int i=0;i<ctr.size();i++)
            temp+=freq[ctr[i]-'a'];

        if(temp>Max)
        {
            Max=temp;
            ans=k;
            Size=ctr.size();
        }

        if(temp==Max)
        {
            if(ctr.size()<Size)
            {
                Size=ctr.size();
                ans=k;
            }
        }
    }

    cout << ans << '\n';

    t--;
}
}
