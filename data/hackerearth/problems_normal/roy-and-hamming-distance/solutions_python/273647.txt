def roy_and_hamming_distance():
    n = input()
    s = raw_input()
    a = [0,0,1]
    #since only one permutation is possible, and hamm distance for that is 0
    #since no other hamm distance is possible, 0 is max hamm distance and number of permutations with 0 hamm distance is 1
    if n==1:
        print 1
    elif n==2:
        print a[2]
    else:
        for i in xrange(3,n+1):
            a.append(((i-1)*(a[i-1]+a[i-2]))%1000000007)
        print a[n]
roy_and_hamming_distance()
