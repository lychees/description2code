#include<iostream>
using namespace std;
#include<algorithm>
#define ll long long 
int main()
{ ll t;
cin>>t;
while(t--)
{  ll a,b,n;
	cin>>a>>b>>n;
	ll arr[n];
	for(int i=0;i<n;i++)
	cin>>arr[i];
	sort(arr,arr+n);
	ll count1=0,count2=0;
	for(int i=0;i<n && (a>=0 || b>=0);i++)
	{
		if(a-arr[i]>=0)
		{count1++;
		a-=arr[i];
		}
		if(b-arr[i]>=0)
		{
			count2++;
			b-=arr[i];
		}
		
		
	}
	if(count1>count2)
	cout<<"Raghu Won"<<endl;
	else if(count1<count2)
	cout<<"Sayan Won"<<endl;
	else
	cout<<"Tie"<<endl;
	
}
	return 0;
	}
