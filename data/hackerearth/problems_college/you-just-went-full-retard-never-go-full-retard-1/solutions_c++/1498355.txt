#include <iostream>
#include<cmath>
using namespace std;


int prime(long long num)
{
	if(num==1||num==2||num==3)
	return 1;
	else if(num%2==0)
	return 0;
	else
	{
		for(long long j=3;j<=sqrt(num);j=j+2)
		if(num%j==0)
		return 0;
	}
	return 1;
}
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	long long num;
    	cin>>num;
    	int f=prime(num);
    	if(f==1)
    	cout<<"YES"<<endl;
    	else
    	cout<<"NO"<<endl;
    	
    }
    return 0;
}
