#include <iostream>
#define maxi 1000001
using namespace std;
long long a[maxi]={0};
void pre()
{  
	a[0]=a[1]=1;
	long long i,j;
	for(i=2;i<maxi;i++)
	{
		for(j=i*2;j<maxi;j+=i)
		{
			a[j]=1;
		}
	}
}
int main()
{
	pre();
    //cout << "Hello World!" << endl;
    long long t;
    cin>>t;
    while(t--)
    {
    	long long n;
    	cin>>n;
    	if(n==3)
    	cout<<"3"<<endl;
    	else if(a[n]==0)
    	cout<<"2"<<endl;
    	else
    	cout<<"1"<<endl;
    }
    return 0;
}
