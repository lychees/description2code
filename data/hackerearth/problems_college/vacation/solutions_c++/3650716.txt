#include <bits/stdc++.h>
using namespace std;
#define getcx getchar_unlocked
inline void inp( int &n )//fast input function
{
   n=0;
   int ch=getcx();int sign=1;
   while( ch < '0' || ch > '9' ){if(ch=='-')sign=-1; ch=getcx();}

   while(  ch >= '0' && ch <= '9' )
           n = (n<<3)+(n<<1) + ch-'0', ch=getcx();
   n=n*sign;
}

int t,n;

int main()
{
	ios_base::sync_with_stdio(false);
	
	cin>>t;
while(t--)
{
	cin>>n;
	if(n==1)
	printf("Bob\n");
	else printf("Alice\n");
}
    return 0;
}
