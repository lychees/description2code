#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
	ll a[10000], n, t, b=0, c, i;
	cin>>t;
	while(t--)
	{
		cin>>n>>c;
		for(i=0; i<n; i++)
		{
			cin>>a[i];
		}
		sort(a, a+n);
		for(i=0; i<n-1; i++)
		{
			b = b+((abs(a[i]-a[i+1]))*c);
			
		}
		cout<<b<<"\n";
		b=0;
	}
}
