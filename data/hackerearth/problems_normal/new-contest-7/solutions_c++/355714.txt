#include<iostream>
#include<stdio.h>

using namespace std;

double * binomialCoeff(int n, int k)
{
   double *C = (double*)calloc(k+1, sizeof(double));
    int i, j;

    C[0] = 0.000000000000001;

    for(i = 1; i <= n; i++)
    {
        for(j = min(i, k); j > 0; j--)
            C[j] = C[j] + C[j-1];
    }
    return C;


}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    int s,n,m,k;
    cin>>s>>n>>m>>k;
    if((s-n<=m-k)||(s==n))
        cout<<"1.000000"<<endl;
    else
    {
        double *M = binomialCoeff(m,m);
        double *S = binomialCoeff(s,s);
        double *SM;
        if(m==s-m)
            SM=M;
        else
           SM = binomialCoeff(s-m,s-m);
            double sum=0;
      for(int i=k;i<=m&&i<=n;i++)
      {
          double temp = M[i]/S[n];
          //cout<<"M[k] "<<M[k]<<endl;
          //cout<<"S[n] "<<S[n]<<endl;
         // cout<<"S-M[k] "<<SM[n-k]<<endl;
          //cout<<"temp is "<<temp<<endl;
          temp*=SM[n-i];
          temp*=1000000000000000;
        //  cout<<"After multiplication temp is "<<temp<<endl;
          sum+=temp;
      }

}
