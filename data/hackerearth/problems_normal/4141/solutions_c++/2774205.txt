#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int t,n,temp,x,i,c;
	scanf("%lld",&t);
	while(t--){
		long long int a[100010];
		temp = 0;
		scanf("%lld%lld%lld%lld",&a[0],&a[1],&c,&n);
		if(n==1){
			if(a[0] == 0)
				printf("YES\n");
			else
			printf("NO\n");
			continue;
		}
		temp += a[0]*10 + a[1];
		temp = temp%41;
		for(i=2;i<n;i++){
			a[i] = (a[i-1]*c + a[i-2])%10;
			temp = (temp*10) + a[i];
			temp = temp%41;
		}
		if(temp == 0)
			printf("YES\n");
		else
			printf("NO\n");
	}
	return 0;
}
