#include <iostream>
#include <string>
using namespace std;

int issafe(string str, int *temp)
{
	for(int i=0;i<str.size();i++)
	if(str[i] == 'Y' && temp[i] == 1)	return 1;
	return 0;
}

int docalc(string str[], int index, int n, int *temp)
{
	int count = 9999999999;
	int cnt = 0;
	if(index==n) return 0;
	if(issafe(str[index],temp))	{/*cout<<"safe\n";*/return docalc(str,index+1,n,temp);}
	else
	{
		for(int i=0;i<str[index].size();i++)
		{
			if(str[index][i] == 'Y')
			{
				//cout<<"setting for string "<<index<<" at "<<i<<endl;
				temp[i] = 1;
				cnt = 1 + docalc(str,index+1,n,temp);
				if(count > cnt)	count = cnt;
				temp[i] = 0;
			}			
		}
		return count;
	}
}

int main()
{
    //cout << "Hello World!" << endl;
    int t,n;
    string str[1000];
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	cin>>n;
    	for(int j=0;j<n;j++)
    	cin>>str[j];
    	int len = str[0].size();
    	int *temp = new int[len]();
    	//cout<<len<<endl;
    	cout<<docalc(str,0,n,temp)<<endl;
    	delete[] temp;
    }
    return 0;
}
