#include<bits/stdc++.h>
using namespace std;
void sol()
{
   long long int i,n,x;
    cin>>n;
   long long int a[n+1];
   long long int sum[n+1];
   long long int low=1,up=100000,d;
   long long int has[n+1];
    for(i=0;i<=n;++i) has[i]=0;
     sum[0]=0;
     for(i=1;i<=n;++i)
     {
         cin>>a[i];
         sum[i]=sum[i-1]+a[i];
     }
      for(i=1;i<=n;++i)
      {
          if(a[i]%n==0)
          {
              cout<<i<<' '<<i<<endl;
              return ;
          }
        d=sum[i]%n;
        if(d==0)
        {
            if(up-low>i)
            {
                low=1;up=i;
            }
        }
         else if(has[d]!=0)
        {
            if(up-low+1>i-has[d])
            {
                low=has[d]+1;
                up=i;
            }
            has[d]=i;
        }
        else has[d]=i;
      }
      cout<<low<<' '<<up<<endl;
}
int main()
{
    int t;
    cin>>t;
    while(t--) sol();
    return 0;
}
