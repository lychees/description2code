# include <bits/stdc++.h>
using namespace std;
long long int a[10000];
long long int dp[10000];
int main(){
	int t;
	cin>>t;
	while(t--){
		int n,k;
		cin>>n>>k;
		memset(a,0,sizeof(a));
		memset(dp,0,sizeof(dp));
		for(int i=0;i<n;i++){
			cin>>a[i];
		}
		for(int i=0;i<n;i++){
			if(i==0){
				dp[i]=a[i]>0?a[i]:0;
			}
			else if(i<k+1){
				dp[i]=max(dp[i-1],a[i]);
			}
			else{
				dp[i]=max(dp[i-1],a[i]+dp[i-k-1]);
			}
		}
		cout<<dp[n-1]<<endl;
	}
}
