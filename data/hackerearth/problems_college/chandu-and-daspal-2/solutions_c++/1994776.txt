#include <iostream>
using namespace std;
int GCD(int A,int B);
int GCD(int A, int B) {
    if(B==0)
        return A;
    else
        return GCD(B, A % B);
}

int main()
{
	long long  int i,j,k,l,m,x,y,t;
	
	string s1,s;
	cin>>t;
	while(t--)
	{
		
		cin>>i>>j;
		m=0;
		cin>>s1;
		s="Chandu";
		if(s==s1)
		m=0;
		else
		m=1;
		while((i!=0)&&(j!=0))
		{
			y=GCD(i,j);
		//	cout<<y;
			i=i-y;
			j=j-y;
//cout<<i<<j<<endl;
			m++;
		}
		if(m%2==0)
		cout<<"Daspal"<<endl;
		else
		cout<<"Chandu"<<endl;
	}
    return 0;
}
