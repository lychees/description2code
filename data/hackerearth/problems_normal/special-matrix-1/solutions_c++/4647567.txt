#include<iostream>
#include<cmath>
using namespace std;
int main()
{
    //char ch[][];
    int test;
    int row,col;
    int num;
    cin>>test;
    while(test--)
    {
        cin>>num;
        char a[num][num];
        for(int i=0;i<num;i++)
        {
            for(int j=0;j<num;j++)
            {
                cin>>a[i][j];
                if(a[i][j]=='*')
                 {
                     row=i;
                     col=j;
                 }
            }
        }
        num=(num-1)/2;
        cout<<abs(num-row)+abs(num-col)<<endl;
    }
}
