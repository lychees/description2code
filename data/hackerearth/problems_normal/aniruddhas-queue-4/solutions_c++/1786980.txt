#include<iostream>
using namespace std;

int main()
 {
     int t;
     cin>>t;
     while(t--)
     {
         int y;
         cin>>y;
         long int a[y], sum=0;
         for(int i=0;i<y;i++)
         {
             cin>>a[i];
             sum+=a[i];// cout<<"i"<<sum;
         }
        long int last;
         cin>>last;
          if(last==0)
         cout<<0<<endl;
        else{last%=sum;
        if(last==0)
         {
           for(int i=y-1;i>=0;i--)
             {
             	if(a[i]!=0)
             	 {cout<<i+1<<endl;
                   break;}
             }
         }
         else{
          for(int i=0;i<y;i++)
          {
           if(last<=a[i])
           { cout<<i+1<<endl;
               break;
           }
           else last-=a[i];
          }
         }
     }
     }

     return 0;
 }
