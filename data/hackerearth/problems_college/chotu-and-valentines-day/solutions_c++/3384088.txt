#include <iostream>
using namespace std;

long long int vis[1000000];

int main()
{
    int n,q,l,r;
    cin>>n>>q;
    int a[n];
    for(int i=0;i<n;i++)
    {
    	cin>>a[i];
    	vis[i]=vis[i-1]+a[i];
    }
    while(q--)
    {
    	cin>>l>>r;
    	long long int sum=0;
    	for(int i=0;i<l-1;i++)
    	{
    		sum+=a[i];
    	}
    	cout<<vis[r-1]-sum<<endl;
    }
    return 0;
}
