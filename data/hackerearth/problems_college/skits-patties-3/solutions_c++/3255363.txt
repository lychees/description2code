#include<bits/stdc++.h>
using namespace std;

int dp[10000000]={0};

int main()
{
	int t;
	cin>>t;
	long int a[t];
	long int mx=0;
	
	for(int i=0;i<t;i++)
	{
		cin>>a[i];
		mx=max(mx,a[i]);
	}
	dp[6]=dp[9]=dp[20]=1;
	
	for(int i=15;i<=mx;i++)
	{
		if(dp[i-6]==1 || dp[i-9]==1 ||dp[i-20]==1)
		{
			dp[i]=1;
			//cout<<i<<" ";
		}
		
	}
	for(int i=0;i<t;i++)
	{
		if(dp[a[i]]==1)
		{
			cout<<"True\n";
		}
		else
		{
			cout<<"False\n";
		}
	}
	
	
	
}
