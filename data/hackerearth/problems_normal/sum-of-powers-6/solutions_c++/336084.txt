#include<iostream>
#include<set>
#include<cmath>
using namespace std;
typedef long long ll;

set<ll> myset;
set<ll> :: iterator it;

int main()
{
    ll i,j,pp,N,X,val;
    //myset.insert(0); 
    myset.insert(1);
    for(i=2;i<=1000;i++)
    {
    	for(j=2;;j++)
    	{
    		pp=pow(i,j);
    		if(pp<=1000000) myset.insert(pp);
    		else break;
    	}
    }
    
    cin>>N;
    while(N--)
    {
    	cin>>X;
    	for(it=myset.begin();it!=myset.end();it++)
    	{
    		val=X-(*it);
    		if(myset.find(val)!=myset.end())
    			break;
    	}
    	if(it==myset.end()) cout<<"No"<<endl;
    	else cout<<"Yes"<<endl;
    }
    return 0;
}
