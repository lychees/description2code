#include<iostream>
#define mx 10000000
using namespace std;
int pr[mx+30];
int phi[mx+30];
long long s[mx+30];
int main()
{
	 pr[1]=1;
	 for(int i=2;i<=mx;i++)
	 	if(pr[i]==0)
	 		for(int j=i+i;j<=mx;j+=i)
	 			pr[j]=i;
	 			
	phi[1]=1;
	
	for(int i=2;i<=mx;i++)
	{
		int q=i;
		int val=pr[i];
		if(val==0)
		phi[i]=q-1;
		else
		{
			while(q%val==0)
			q/=val;
			
			phi[i]=phi[q]*(i/q)/val*(val-1);
		}
	}
	for(int i=1;i<=mx;i++)
	s[i]=s[i-1]+phi[i];
	int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		cout<<s[n]<<"\n";
	}
	return 0;
}
