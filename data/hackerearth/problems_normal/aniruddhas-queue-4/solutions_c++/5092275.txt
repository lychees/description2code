#include <iostream>
using namespace std;

int main()
{
    int T,N,i,ith_day;
    
    scanf("%d",&T);
    while(T--){
    	scanf("%d",&N);
    	long int *D = (long int *)malloc(sizeof(long int)*N);
    	long long int total_d,sum = 0,rem;
    	ith_day = 0;
    	for(i=0;i<N;i++){
    		scanf("%ld",&D[i]);
    	}
    	
    	scanf("%lld",&total_d);
    	for(i=0;i<N;i++){
    		sum += D[i];
    	}
    	
    	rem = total_d % sum;
    	//cout<<rem<<endl;
    	if(rem==0){
    		i = N-1;
    		while(1){
    			if(D[i]!=0){
    				ith_day = i+1;
    				break;
    			}
    			i--;
    		}
    	}else{
    		i = 0;
    		while(rem>0){
    			rem -= D[i];
    			
    			if(rem<=0){
    				ith_day = i+1;
    				break;
    			}
    			
    			i = (i+1)%N;
    		}
    	}
    	printf("%d\n",ith_day);
    }
    return 0;
}
