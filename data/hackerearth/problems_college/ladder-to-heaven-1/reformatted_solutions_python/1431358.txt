import sys
f = sys.stdin
T = int(f.readline())
while T:
    N = int(f.readline())
    a, b, c = 1, 2, 0
    ans = 0
    if N == 1:
        ans = 0
    elif N == 2:
        ans = 2
    else:
        ans = 2
        while True:
            c = a + b
            if c > N:
                break
            if c % 2 == 0:
                ans += c
            a = b
            b = c
    print ans
    T -= 1
