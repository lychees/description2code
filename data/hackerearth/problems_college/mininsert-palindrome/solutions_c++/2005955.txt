#include <iostream>
using namespace std;
int dp[25][25];
int max(int x,int y)
{
	return x>y?x:y;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n;
    	cin>>n;
    	string s1,s2="";
    	cin>>s1;
     n=s1.length();
    	for(int i=n-1;i>=0;i--)
    	s2+=s1[i];
    	for(int i=0;i<=n;i++)
    	for(int j=0;j<=n;j++)
    	{
    		if(i==0||j==0)
    		dp[i][j]=0;
    		else if(s1[i-1]==s2[j-1])
    		dp[i][j]=dp[i-1][j-1]+1;
    		else
    		 dp[i][j]=max(dp[i-1][j],dp[i][j-1]);
    	}
    printf("%d\n",n-dp[n][n]);	
    }
    return 0;
}
