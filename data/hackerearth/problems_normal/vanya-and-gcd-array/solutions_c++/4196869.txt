#include<iostream>
using namespace std;
#define M 1000000007
 
long long int ar[1000];
long long int matIndexIGCDJ[510][110];
 
long long int gcd(long long int a, long long int b) {
	if(b == 0) return a;
	return gcd(b, a%b);
}
 
int main() {
	long long int n,i,j,k,ans,g;
	for(i = 0; i < 510; i++)
		for(j = 0; j < 110; j++)
			matIndexIGCDJ[i][j] = 0;
 
	cin>>n;
	for(i = 0; i < n; i++) {
		cin>>ar[i];
		matIndexIGCDJ[i][ar[i]] = 1;
	}
	ans = 0;
	for(i = 0; i < n; i++) {
		for(j = 0; j < i; j++) {
			if(ar[i] > ar[j]) {
				for(k = 0; k < 110; k++) {
					g = gcd(k, ar[i]);
					matIndexIGCDJ[i][g] += matIndexIGCDJ[j][k];
					matIndexIGCDJ[i][g] = matIndexIGCDJ[i][g] % M;
				}
			}
		}
	}
 
	for(i = 0; i < n; i++) {
		ans += matIndexIGCDJ[i][1];
		ans = ans % M;
	}
 
	cout<<ans<<endl;
 
	return 0;
}
