#include <iostream>
using namespace std;
long long int max(long long int a,long long int b){
	if(a>=b)return(a);
	else return b;
}
long long int maxarray(int a[],int n){
	long long int max_sofar=a[0],current=a[0];
	int i;
	for(i=1;i<n;i++){
		current=max(a[i],current+a[i]);
		max_sofar=max(max_sofar,current);
	}
	return(max_sofar);
}
int main() {
	int n,i;
	long long int x;
	cin>>n;
	int a[n];
	for(i=0;i<n;i++){
		cin>>a[i];
	}
	x=maxarray(a,n);
	if(x>=0){
		cout<<x<<"\n";
	}
	else cout<<'0'<<"\n";
	return 0;
}
