#include <iostream>
using namespace std;
void swap(int arr[], int &a, int &b)
{
	int temp;
	temp = arr[a];
	arr[a] = arr[b];
	arr[b] = temp;
}
int main()
{
    int n,k;
    cin>>n>>k;
    int arr[1000001];
    for(int i =0;i<n;i++)
    cin>>arr[i];
    for(int i = 0;i<k;i++)
    {
    	int max = i;
    	for(int j = i;j<n;j++)
    	{
    		if(arr[j]>arr[max])
    		{
    			
    		//cout<<"here\n";
    		max = j;
    		}
    	}
    	if(i != max)
    	swap(arr,i,max);
    	else
    	k++;
    }
    for(int i = 0;i<n;i++)
    cout<<arr[i]<<" ";
    return 0;
}
