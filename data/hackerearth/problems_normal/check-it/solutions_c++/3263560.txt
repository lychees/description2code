#include<bits/stdc++.h>
#define mp make_pair
using namespace std;
int n,ver,edg;
vector <int> G[100005];
bool visited[100005];
pair<int,int> edges[100005];
void dfs(int v)
{
	visited[v]=true;
	for(int i=0;i<G[v].size();i++)
	{
		if(!visited[G[v][i]])
			dfs(G[v][i]);
	}
}
int main()
{
	int i,j,fl,test,x,y,m;
	scanf("%d",&test);
	while(test--)
	{
		memset(visited,false,sizeof(visited));
		scanf("%d %d",&n,&m);
		for(i=0;i<=n;i++)
			G[i].clear();
		for(i=1;i<=m;i++)
		{
			scanf("%d%d",&x,&y);
			edges[i]=mp(x,y);
		}
		for(i=0;i<n-1;i++)
		{
			scanf("%d",&edg);
			G[edges[edg].first].push_back(edges[edg].second);
			G[edges[edg].second].push_back(edges[edg].first);
		}
		dfs(1);
		for(i=2;i<=n;i++)
		{
			if(!visited[i])
			{
				printf("NO\n");
				break;
			}
		}
		if(i==n+1)
			printf("YES\n");
	
	}
	return 0;
}
