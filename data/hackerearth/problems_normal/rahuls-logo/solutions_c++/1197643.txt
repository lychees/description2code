#include <iostream>

using namespace std;

void printPat1(int i)
{
    for(int j=0;j<=i;j++){
        if(j%2)
        {
            cout<<string(1,' ');
        }else{
            cout<<"/";
        }
    }
    for(int j=i;j>=0;j--){
        if(j%2)
        {
            cout<<string(1,' ');
        }else{
            cout<<"\\";
        }
    }
}
void printPat2(int i)
{
    for(int j=0;j<=i;j++){
        if(j%2)
        {
            cout<<string(1,' ');
        }else{
            cout<<"\\";
        }
    }
    for(int j=i;j>=0;j--){
        if(j%2)
        {
            cout<<string(1,' ');
        }else{
            cout<<"/";
        }
    }
}

void func(int N)
{
    if(N<2) return;
    for(int i=0;i<N;i++)
    {
        cout<<string(N-i-1,' ');
        printPat1(i);
        cout<<endl;

    }
    for(int i=N-1;i>=0;i--)
    {
        cout<<string(1,' ');
        cout<<string(N-i,' ');
        printPat2(i);
        cout<<endl;

    }
}
