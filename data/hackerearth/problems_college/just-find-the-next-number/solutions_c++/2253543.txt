#include<bits/stdc++.h>
#define MAX 100001
int arr[MAX];
using namespace std;
int main()
{
	int p,curr,i,j,k,m,n,t,c=0;
	for(i=1;i<MAX;i++)
	{
		m=i;
		curr=0;
		p=0;
		while(m!=0)
		{
			if(m%2==0)
				curr=0;
			else
				curr=1;

			if(curr==1 && p==1)
				break;
			m=m/2;
			p=curr;
		}
		if(m==0)
			arr[c++]=i;

	}

	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		{
			for(i=0;i<c;i++)
			{
				if(arr[i]>=n)
					break;
			}
			printf("%d\n",arr[i]);
		}
	}
}
