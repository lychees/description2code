numCases = int(raw_input())
for iCase in xrange(numCases):
    numMatches = int(raw_input())
    matches = []
    for i in xrange(numMatches):
        st, et, wp = map(int, raw_input().split())
        matches.append([et, st, wp])
    matches.sort()
    states = {0: 0}
    for et, st, wp in matches:
        newpairs = []
        for et0 in states:
            if et0 <= st:
                newpairs.append((et, states[et0]+wp))
        for et, wp1 in newpairs:
            if et not in states:
                states[et] = wp1
            elif states[et] < wp1:
                states[et] = wp1
    solution = max(states.values())
    print "%.2f " % ((solution+0.0)/100)
