#include<bits/stdc++.h>
using namespace std;
long int check(long int n)
{
	if(n==2) return 0;
	
	if(n%2==0) return 1;
	long int i,flag=0;
	for(i=3;i<=sqrt(n);i+=2)
	{
		if(n%i==0)
		{
			flag=1;
			break;
		}
	}
	if(flag==1)
		return 1;
	else
		return 0;
}
int main()
{
	long int N,i,j,k;
	cin>>N;
	for(i=4;i<=N;i++)
	{
		j=check(i);
		if(j==1)
			cout<<i<<"\n";
	}
}
