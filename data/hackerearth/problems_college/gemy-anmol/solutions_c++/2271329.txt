#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long int t,i,j,k,a[50000],n,p,m;
    cin>>t;
    while(t--)
    {
    	cin>>m>>p;
    	for(i=0;i<m;i++)
    	cin>>a[i];
    	sort(a,a+m);
    	j=0;k=0;
    	for(i=m-1;i>=0;i--)
    	{
    		if(k<p)
    		{k=k+a[i];j++;}
    	}
    	if(k>=p)
    	cout<<j<<endl;
    	else
    	cout<<"-1"<<endl;
    }
    return 0;
}
