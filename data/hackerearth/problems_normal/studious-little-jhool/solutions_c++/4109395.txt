#include <iostream>
#include <math.h>
using namespace std;

int main()
{
    int T,N, x, y, pages;
    cin >> T;
    while(T--) {
    	
    	pages = -1;
    	cin >> N;
    	x = floor(N/10);
    	y = floor(N/12);
    	for (int i = 0; i <= y; i++) {
    		for (int j = 0; j <= x; j++) {
    			if ((12*i + 10*j) == N) {
    				pages = i + j;
    				break;
    			}
    		}
    	}
    	cout << pages << endl;
    }
    return 0;
}
