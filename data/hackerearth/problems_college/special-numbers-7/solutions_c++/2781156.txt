#include<iostream>
using namespace std;
int prime[1000001],lucky[1000];
void seive(int N)
{
	int i,j;
	for(i=0;i<=N;i++)
		prime[i]=1;
	prime[0]=0;
	prime[1]=0;
	for(i=2;i*i<=N;i++){
	if(prime[i]==1)
	{
	for(j=i*i;j<=N;j+=i)
	prime[j]=0;
	}
	}
}
int main()
{
	int t,n,i,k,j;
	int Prime[1000000],temp;
	seive(1000000);
	k=0;
	for(i=0;i<1000000;i++)
	{
		if(prime[i]==1)
		{
			Prime[k]=i;
		//	cout<<Prime[k]<<' ';
			k++;
		}
	}
	k=0;
	for(i=2;;i++)
	{
		if(k==1000)
			break;
		j=0;
		temp=0;
		while(Prime[j]<i)
		{
			if(i%Prime[j]==0)
				temp++;
			if(temp==3)
				{
					lucky[k]=i;
					k++;
					//cout<<i<<' ';
					break;
				}
				j++;
		}
	}
	cin>>t;
	while(t--)
	{
		cin>>n;
		cout<<lucky[n-1]<<endl;
	}
	return 0;
}
