#include <iostream>
using namespace std;

int main()
{   int t;
	long a,b,num1,num2,set,t1,t2,c;
	cin>>t;
	while(t--)
	{	c=0;
		cin>>a>>b;
		num1=a;
		num2=b;
		int set=0;
		while(num1 || num2)
		{
			t1=num1%10;
			t2=num2%10;
			num1/=10;
			num2/=10;
			if(t1+t2+set>9)
			{
			  c++;
			  set=1;
			}
			else 
				set=0;
		}
	
	if(c==1)
		cout<<c<<" carry operation\n";
	else if(c>1)
		cout<<c<<" carry operations\n";
	else
		cout<<"No carry operation\n";
	}
   return 0;
}
