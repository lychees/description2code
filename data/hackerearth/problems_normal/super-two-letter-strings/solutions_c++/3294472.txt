#include<bits/stdc++.h>
using namespace std;
#define MOD 1000000007
int main(void)
{
	int T,N,P;
	cin>>T;
	while(T--)
	{
		cin>>N>>P;
		int dp[N+1];
		memset(dp,0,sizeof(dp));
		dp[0] = 1;
		for(int i=1;i<=N;i++)
			for(int j=i-1;j>=0 && i-j <= P;j--)
				dp[i] = (dp[i]%MOD + dp[j]%MOD)%MOD;
		cout<<dp[N]<<endl;
	}
	return 0;
}
