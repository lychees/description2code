#include <iostream>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    int t,n;
    int a[20];
    cin>>t;
    long i,possi,c,x,su,ans;
    bool s[2001];
    while(t--)
    {
    	cin>>n;
    	for(i=0;i<n;i++)cin>>a[i];
    	for(i=1;i<=2000;i++)s[i]=false;
    	possi = 1;
    	for(i=0;i<n;i++)possi*=2;
    	ans = 0;
    	for(i=0;i<possi;i++)
    	{
    		x = i;
    		c = 0;
    		su = 0;
    		while(x!=0)
    		{
    			if(x%2!=0)su+=a[n-c-1];
    			c++;
    			x/=2;
    		}
    		s[su]=true;
    	}
    	
    	for(i=1;i<=2000;i++)if(s[i])ans++;
    	cout<<ans+1<<"\n";
    }
    return 0;
}
