#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		float d[n],f[n];
		for(int i=0;i<n;i++)
		{
			cin>>d[i];
		}
		for(int i=0;i<n;i++)
		{
			cin>>f[i];
		}
		float g[n];
		for(int i=0;i<n;i++)
		{
			g[i]=d[i]*f[i];
		}
		float max=g[0];
		int c=0;
		for(int i=1;i<n;i++)
		{
			if(g[i]>max)
			{
				max=g[i];
				c=i;
			}
		}
		float k=0.0;
		for(int i=0;i<n;i++)
		{
			k+=g[i];
		}
		float pr;
		pr=max/k;
		cout<<c<<" ";
		printf("%5.3f",pr);
		cout<<endl;
	}
	return 0;
}
