#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int test;
	cin>>test;
	while(test--)
	{
		ll int n;
		cin>>n;
		ll int i,mc=0;
		for(i=1;i*i<=n;i++)
		{
			if(n%i==0)
			{
				if(i==n/i)
				{
					//~ cout<<i<<" ";
					mc++;
				}
				else
				{
					//~ cout<<i<<" "<<n/i<<" ";
					mc+=2;
				}
			}
		}
		//~ cout<<endl;
		cout<<mc-1<<endl;
	}
	return 0;
}
