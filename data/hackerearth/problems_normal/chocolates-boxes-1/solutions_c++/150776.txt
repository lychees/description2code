#include<iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int *n,k,q,num,l,i,r,cnt,p;
        cin>>num;
        n = new int[num];
        for(i=0;i<num;i++)
        {
            cin>>n[i];
        }
        cin>>k;
        std::sort(n, n+num);
        l=0;r=num-1;
        cnt =0;
        while(l<r)
        {
                if(n[l]+n[r] == k)
                {
                    p=r;
                    while(n[l]+n[p]==k&&l<p)
                    {
                        cnt++;
                        p--;
                    }
                    l++;
                }
                else if(n[l]+n[r]<k){
                    l++;}
                else{
                    r--;}
            }
            cout<<cnt<<endl;
    }
        return 0;
    }
