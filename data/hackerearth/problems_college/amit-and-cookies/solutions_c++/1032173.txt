#include<iostream>
using namespace std;
int main(){
	long long int n,i,j,count;
		unsigned long int k,sum;
		cin>>n>>k;
		i=0;
		unsigned long int a[n];
		while(i<n){
			cin>>a[i];
			i++;
		}
		i=0;
		while(i<n){
			j=i+1;
			while(j<n){
				if(a[i]>a[j]){
					a[j]=a[i]+a[j];
					a[i]=a[j]-a[i];
					a[j]=a[j]-a[i];
				}
				j++;
			}
			i++;
		}
	i=0;
	count=0;
	sum=0;
	while(i<n){
		sum+=a[i];
		if(sum<=k){
			count++;
		}
		else{
			break;
		}
		i++;
	}
	cout<<count<<endl;
	return 0;
}
