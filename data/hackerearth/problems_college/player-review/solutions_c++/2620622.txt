#include <iostream>
using namespace std;

int main()
{
    int n,sum=0,i,maxsum=0;
    cin>>n;
    int a[n];
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    	
    }
    for(i=0;i<n;i++)
    {
    	sum=sum+a[i];
    	if(sum<0)
    	sum=0;
    	if(sum>maxsum)
    	{
    		maxsum=sum;
    	}
    }
    if(sum==0)
    cout<<sum<<endl;
    else
    cout<<maxsum;
    return 0;
}
