#include<bits/stdc++.h>
using namespace std;
int main()
{
	ios::sync_with_stdio(false);
	long long int t;
	cin>>t;
	while(t--)
	{
		long long int a,c,n;
		cin>>n;
		for(long long int i=0;i<n;i++)
		{
			cin>>a;
			if(i==0)
			 c=a;
			else
			 c=__gcd(a,c);
		}
		c=c*n;
		cout<<c<<endl;
	}
	return 0;
}
