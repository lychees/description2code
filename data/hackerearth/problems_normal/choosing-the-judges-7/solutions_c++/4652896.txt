#include <iostream>
using namespace std;
typedef long long int ll;

int main()
{
	ll t,j;
	cin >> t;
	j = 1;
	while(t--) {
		ll n;
		cin >> n;
		ll a[n],i;
		ll sum[n];
		for( i = 0; i < n; i++) {
			cin >> a[i];
		}
		for( i = 0; i < n; i++) {
			if ( i == 0 ) {
				sum[i] = a[i];
			} else if ( i == 1) {
				sum[i] = max(sum[i-1],a[i]);
			}else {
				sum[i] = max(sum[i-1],sum[i-2] + a[i]);
			}
		}
		cout << "Case" << " " << j << ":" <<  " " << sum[n-1] << endl;
		j++;
		
	}
    //cout << "Hello World!" << endl;
    return 0;
}
