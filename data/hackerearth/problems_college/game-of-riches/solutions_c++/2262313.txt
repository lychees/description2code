#include <cstdlib>
#include <iostream>

using namespace std;
int T,I,g,r,i,num,numi,t;
int compare (const void * a, const void * b)
{
  return ( *(int*)b - *(int*)a );
}

int main(int argc, char *argv[])
{   cin>>T;
    for(I=1;I<=T;I++)
    {   cin>>g>>r;
        int x[g];
        for(i=0;i<g;i++)
           cin>>x[i];
        
        num=x[0];
        
        qsort (x, g, sizeof(int), compare);
        
        for(i=0;i<g;i++)
           if(x[i]==num)
           {  numi=i;
              break;
           }
      
        
        
        {   t=0;
            for(i=numi%r;i<g;i+=r)
            {  
                if(x[i]>x[numi])
                   t++;
            }
            
            cout<<t<<endl;
            
        }
                        
    }
   
    return EXIT_SUCCESS;
}
