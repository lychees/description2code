#include <iostream>
using namespace std;

int main()
{
    int n,q;
    cin>>n>>q;
    int a[n];
    int i,j;
    int l,m,p,r=0;
    for(i=0;i<n;i++)
    cin>>a[i];
   
    for(j=0;j<q;j++){
    	cin>>l>>m;
    	r=0;
    	
    	if(l==0)
    {	cin>>p;
       a[m-1]=p;
    }
    	else
    	for(i=0;i<n;i++){
    		r=r+1;
    		if(a[i]>=m)
    	{
    		i=i+1;
        	cout<<i<<endl;
        	break;
    	}
    	if(r==n-1)
    	cout<<-1<<endl;
    	
    	}
    }
    
    return 0;
}
