#include<bits/stdc++.h>
using namespace std;
int a[1000000],b[1000000],cd[1000000];
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,i,j,k=0,c,l=0;
		cin>>n;
		for(i=0;i<n;i++)
		{
			cin>>a[i];
			b[n-i]=a[i];
			cd[i]=a[i];
		}
	
			sort(cd,cd+n);
		for(i=0;i<n;++i)
	{
			if(cd[i]!=b[i])
		{
			l=1;
			break;
		}
	
	}
	if(l==0)
	cout<<n<<endl;
	else
		
	{
			for(i=n-1;i>=0;i--)
		{
			c=0;
			for(j=i+1;j<n;j++)
			{
				if(a[i]<=a[j])
				{
					c=1;
					break;
				}
			}
			if(c!=1)
			{
				k++;
			}
		}
		cout<<k<<endl;
	}
		
		
	}
	return 0;
}
