#include <bits/stdc++.h>

using namespace std;

int m, n;
int h, g;
int u, v;
int t;
multiset<pair<int, int> > great;
multiset<int> good;

int main(){
    cin>>t;

    for(int inst=0; inst<t; inst++){
        great.clear();
        good.clear();
        cin>>m>>n;
        for(int i=0;i<m;i++){
            cin>>u>>v;
            great.insert(make_pair(u, v));
            good.insert(u);
        }
        int qtd_good=0, qtd_great=0;
        for(int i=0;i<n; i++){
            cin>>u>>v;
            if(good.find(u)!=good.end())qtd_good++;
            if(great.find(make_pair(u,v))!=great.end())qtd_great++;
        }
        cin>>h>>g;
         cout<<":(\n";
    }

    return 0;
}
