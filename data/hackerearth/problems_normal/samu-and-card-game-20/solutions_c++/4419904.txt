#include <iostream>
typedef long long int ll;

using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	ll n,m,i,k,j,x=1,y=1,cn=0;//int flag=0;
    	cin>>n>>m;
    	cin>>k;
    	while(k--)
    	{
    		cin>>i>>j;
    		while(x+i<=n&&y+j<=m)
    		{
    			if(x+i<1||y+j<1) break;
    			x=x+i;y=y+j;cn++;
    		}
       	}
    	cout<<cn<<endl;
    }
    return 0;
}
