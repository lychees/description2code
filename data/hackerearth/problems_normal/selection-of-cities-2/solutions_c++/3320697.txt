#include<iostream>
using namespace std;
long long int u=1000000007;
long long int ans(long long int n){
	if(n==1)return 2;
	long long int h=ans(n/2);
	if(n%2==0){
		h=h*h;h=h%u;return h;
	}
	h=h*h;h=h%u;h=h*2;h=h%u;return h;
}
int main(){
	long t;cin>>t;long long int n;
	for(long i=0;i<t;i++){
		cin>>n;
		long long int h=ans(n);
		cout<<h-1<<endl;
	}
}
