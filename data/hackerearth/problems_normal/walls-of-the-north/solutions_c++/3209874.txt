#include <bits/stdc++.h>

using namespace std;

#define s(x) scanf("%d", &x)
#define ll long long int
#define pb push_back
#define mp make_pair

int B[100005];
int S[100005];
int a[100005];
int dp[100005][2];

int main()
{
    int t, n, i, j;
    s(t);
    while (t--) {
        s(n);
        a[0] = INT_MAX;
        for (i = 1; i <= n; ++i)
            s(a[i]);
        B[1] = 0;
        for (i = 2; i <= n; ++i) {
            j = i-1;
            while (a[i] > a[j])
                j = B[j];
            B[i] = j;
        }
        a[0] = INT_MIN;
        S[1] = 0;
        for (i = 2; i <= n; ++i) {
            j = i-1;
            while (a[i] < a[j])
                j = S[j];
            S[i] = j;
        }
        dp[0][0] = 0;
        dp[0][1] = 0;
        for (i = 1; i <= n; ++i) {
            dp[i][0] = max(dp[S[i]][1]+1, dp[S[i]][0]);
            dp[i][1] = max(dp[B[i]][0]+1, dp[B[i]][1]);
        }
        printf("%d\n", max(dp[n][0], dp[n][1]));
    }
    return 0;
}
