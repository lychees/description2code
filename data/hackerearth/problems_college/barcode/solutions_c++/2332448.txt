char a[1010][1010];
int sum[1010][2];
int dp[1010][2];
int main()
{
	int n,m,x,y;
	scanf("%d%d%d%d",&n,&m,&x,&y);
	for(int i=0;i<n;i++)
		scanf("%s",a[i]);
	sum[0][0]=sum[0][1]=0;
	for(int j=0;j<m;j++){
		int cnt1=0,cnt2=0;
		for(int i=0;i<n;i++){
			if(a[i][j]=='.')
				cnt1++;
			else cnt2++;
		}
		sum[j+1][0]=sum[j][0]+cnt1;
		sum[j+1][1]=sum[j][1]+cnt2;
	}
	for(int i=1;i<=m;i++)
		dp[i][0]=dp[i][1]=inf;
	dp[0][0]=dp[0][1]=0;
	for(int i=1;i<=m;i++){
		for(int j=max(i-y,0);j<=i-x;j++){
			dp[i][0]=min(dp[j][1]+sum[i][1]-sum[j][1],dp[i][0]);
			dp[i][1]=min(dp[j][0]+sum[i][0]-sum[j][0],dp[i][1]);
		}
	}
	printf("%d\n",min(dp[m][0],dp[m][1]));
}
