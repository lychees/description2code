#include <bits/stdc++.h>
#include <iostream>
#include <stdint.h>

using namespace std;

int64_t fact(int64_t n)
{
	if(n==0 || n==1)
	return 1;
	else 
	return n*fact(n-1);
}
int64_t comb(int64_t N, int64_t K)
{
   if (K > N) return 0;
   int64_t r = 1,d;
   for (d = 1; d <= K; d++)
   {
      r *= N--;
      r /= d;
   }
   return r;
}
int64_t gcd(int64_t m,int64_t n)
{
	while ( n != 0) {
        int64_t r = m % n;
        m = n;
        n = r;
    }
    return m;
}
int main() {
	
	std::ios::sync_with_stdio(false);
	int64_t t,n,k,a[1000000],num1,num2,g,u,v;
	
	cin >> t;
	while(t--)
	{
		cin >> n >> k;
		uint64_t i,c = 0;
		for(i = 0;i < n;i++)
		{
			cin >> a[i];
			if(a[i] == k)
			{
				c++;	
			}	
		}
		if(c < 3 || n < 3)
		{
			cout<<"0/1"<<endl;			
		}
		else if(n==c)
		{
			cout<<"1/1"<<endl;
		}
		else{
		num1 = comb(c,3);
		num2 = comb(n,3);
		
		g = gcd(num2,num1);
		u = num1/g;
		v = num2/g;
		cout<< u<<"/"<<v<<"\n";
		}
	}
	return 0;
}
