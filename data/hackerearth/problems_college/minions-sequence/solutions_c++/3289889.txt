#include<iostream>
using namespace std;

int main()
{

	int n,t,length,i;
	cin>>t;
	while(t--)
	{
		cin>>n;
		int a[n];
		for(i=0;i<n;i++)
		{
			cin>>a[i];
		}
		cin>>length;
		int min[n],len=1;
		min[0]=a[0];
		for(i=1;i<n;i++)
		{
			if(a[i]<min[0])
			{
				min[0]=a[i];
			}
			else if(a[i]>min[len-1])
			{
				if(len<length)
				{
					min[len]=a[i];
					len++;
				}
			}
			else
			{
				int low=-1,high=len-1;
				//cout<<a[i]<<" ";
				while(low<high)
				{
					int mid=low+(high-low)/2;
					//cout<<mid<<" ";
					if(min[mid]<a[i])
					{
						low=mid+1;
					}
					else
					{
						high=mid;
					}
					//cout<<low<<endl;
				}

				min[high]=a[i];
			}
		}
		if(len<length)
		{
			cout<<"-1"<<endl;
		}
		else
		{
			cout<<min[length-1]<<endl;
		}
	}
	return 0;
}
