#include<bits/stdc++.h>
using namespace std;
int bit[26+4][100*1000+9];
int N,Q;
char s[100*1000+7];
inline void upd(int at,int by,int pos){
    for(;at<=N;at+=at&-at)
        bit[pos][at]+=by;
    return;
}
inline int qry(int at,int pos){
    int sum=0;
    for(;at>=1;at-=at&-at)
        sum+=bit[pos][at];
    return sum;
}
int main(){
    scanf("%d %d",&N,&Q);
    scanf("%s",s);
    for(int i=0;i<N;++i){
        upd(i+1,1,s[i]-'a');
    }
    while(Q--){
        int typ;scanf("%d",&typ);
        if(typ==1){
            int at;char d;
            scanf("%d %c",&at,&d);
            upd(at,-1,s[at-1]-'a');
            //puts("asdf1");
            upd(at,1,d-'a');
            //puts("adf 2");
            s[at-1]=d;
            //puts("adsf 3");
        }else if(typ==2){
            int l,r;scanf("%d %d",&l,&r);
                int odd_cnt=0;
            int even_cnt=0;
            for(int i=0;i<26;++i){
                if((qry(r,i)-qry(l-1,i))%2)
                    odd_cnt++;
                else even_cnt++;
            }
            if(odd_cnt==0 || odd_cnt==1)
                puts("yes");
            else puts("no");
        }

    }
    return 0;
}
