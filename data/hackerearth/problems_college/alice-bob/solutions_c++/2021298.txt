#include <iostream>
#include <stdio.h>
#include <cmath>
using namespace std;

int main()
{
    int n,i,j,k;
    scanf("%d\n",&n);
    int a[100000],d[100000],top=-1,win=2;
    for(i=0;i<n;i++)
    	scanf("%d\n",&a[i]);
   while(1){
   	
   for(i=0;i<n-1;i++)
    {
    	for(j=i+1;j<n;j++){
    		d[++top]=fabs(a[i]-a[j]);
    	for(k=n-1;k>=0;k--){
    		if(d[top]==a[k])
    		{
    			top--;
    			break;
    		}
    	}
     }}
    
    if(top==-1)
    	break;
    else
    {
    	if(win==1)
    		win=2;
    	else
    		win=1;
    }
    a[n++]=d[top--];
   }
   if(win==1)
   	printf("Alice\n");
   else
   	printf("Bob\n");
  
    
    
    return 0;
}
