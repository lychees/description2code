#include <iostream>
using namespace std;
bool store[10001][10001];
void inti()
{
	for(int i=0;i<10001;i++)
	{
		for(int j=0;j<10001;j++)
		{
			store[i][j]=false;
		}
	}
}
bool fun(int x,int y)
{
	
	store[1][1]=false;
	store[1][2]=store[2][1]=true;
	bool l,r;
	for(int i=2;i<=x;i++)
	{
		for(int j=2;j<=y;j++)
		{
			l=r=false;
		
		for(int k=1;k<=j/2;k++)
		{
			if(!store[j-k][k]){l=true;break;}
		}
		for(int k=1;k<=i/2;k++)
		{
			if(!store[k][x-k]){r=true;break;}
		}
		 store[i][j]=l||r;
		 store[j][i]=l||r;
		 
		}
	}
		
 return store[x][y];
}
