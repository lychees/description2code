#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;

long long int coin[5];
long long int dp[255][5];
long long int m;
long long int cool(long long int n,long long int j)
    {
     if(dp[n][j]!=-1)
        return dp[n][j];
    
     if(n==0)
         return 1;
    
       long long int res=0;
    
        for(long long int i=j;i<m;i++)
            if(n-coin[i]>=0)
           res+=cool(n-coin[i],i);
                       
    dp[n][j]=res;
    return res;
    
}

 int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */  
    
    m=3;
    coin[0]=1;coin[1]=3;coin[2]=5;
    int t;cin>>t;while(t--)
    {
    	for(long long int i=0;i<255;i++)
        for(long long int j=0;j<5;j++)
        dp[i][j]=-1;
    
      long long int cost;
      cin>>cost;
   
      cout<<cool(cost,0)<<endl;
    }
    
    return 0;
}
