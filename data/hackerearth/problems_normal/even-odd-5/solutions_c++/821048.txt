#include <stdio.h>
int main()
{
	int t;
	
	scanf("%d",&t);
	while(t--)
	{
		unsigned long long r,n,c,o=0,e=0;
		
		scanf("%llu %llu",&r,&c);
		n=r-c;
		if(r<=c)
		{
			if(r%2)
			{
				o=(((r+1)/2)*(1+r))/2;
				e=((r/2)*(1+r))/2;
			}
			else
			{
				o=((r/2)*(r))/2;
				e=((r/2)*(2+r))/2;
			}
		}
		else
		{
			if(c%2)
			{
				o=(((c+1)/2)*(1+c))/2;
				e=((c/2)*(1+c))/2;
				
				if(n%2)
				{
					e+=(((n+1)/2)*c);
					o+=((n/2)*c);
				}
				else
				{
					e+=((n/2)*c);
					o+=((n/2)*c);
				}
			}
			else
			{
				o=((c/2)*(c))/2;
				e=((c/2)*(2+c))/2;
				
				if(n%2)
				{
					e+=((n/2)*c);
					o+=(((r-c+1)/2)*c);
				}
				else
				{
					e+=((n/2)*c);
					o+=((n/2)*c);
				}
			}
			
		}
		printf("%llu %llu\n",e,o);
		
	}
	return(0);
}
