#include <bits/stdc++.h>

#define ll long long
#define llu unsigned long long
#define ld long

#define MOD 1000000007

#define CLR(a,x) memset(a,x,sizeof(a))
#define PB push_back

using namespace std;

bool primes[100001];

void initialize() {
	memset(primes, 1, sizeof primes);

	primes[0] = 0;
	primes[1] = 0;
	for(int i=2; i<100001; i++) {
		if(primes[i]) {
			for(int j=i+i; j<100001; j+=i) {
				primes[j] = 0;
			}
		}
	}
}

int main(int argc, char const *argv[]) {
	ios::sync_with_stdio(false);

	int T, m, n, arr[100001];
	int batteries_count[100002];
	initialize();
	cin>>T;

	while(T--) {
		cin>>n>>m;

		for(int i=0; i<n; i++) {
			cin>>arr[i];
		}

		sort(arr, arr+n);

		ll prod = 1;
		memset(batteries_count, 0, sizeof batteries_count);

		batteries_count[m+1] = 0;
		for(int i=m; i>=1; i--) {
			if(!primes[i]) {
				batteries_count[i] = batteries_count[i+1] + 1;
			} else {
				batteries_count[i] = batteries_count[i+1];
			}
		}

		// for(int i=1; i<=m; i++) {
		// 	cout<<batteries_count[i]<<endl;
		// }
		int j=0;
		for(int i=n-1; i>=0; i--, j++) {
			prod *= (batteries_count[arr[i]] - j);
			prod %= MOD;
		}

		if(prod >= 0)
			cout<<prod<<"\n";
		else
			cout<<"0\n";
	}
	return 0;
}
