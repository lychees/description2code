long long palin(long long i,long long j)
{
	while(i<j)
	{
		if(str[i]!=str[j])
		return 0;
		i++;
		j--;
	}
	return 1;
}
long long findindex(long long n)
{
	long long i,j;
	for(i=0,j=n-1;i<n&&j>=0;i++,j--)
     	{
     		if(str[i]!=str[j])
     		{
     			if(palin(i+1,j))
     			return i;
     			if(palin(i,j-1))
     			return j;
     		}
     	}
     	return -1;
}
