#include <iostream>
using namespace std;

int main()
{
	long long int n,q,l,r;
	cin>>n>>q;
	
	long long int ar[n];
	for(long long int i=1;i<=n;i++)
	{
		cin>>ar[i];
	}
	while(q--)
	{
		int s=0;
		int d;
		
		cin>>l>>r;
		for(int i=l;i<=r;i++)
		{	
			if(i==r)
				d=ar[i]-s;
			s+=ar[i];		
		}
	
		cout<<s<<" "<<d<<endl;
	}
	
    //cout << "Hello World!" << endl;
    return 0;
}
