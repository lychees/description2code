#include <bits/stdc++.h>
using namespace std;
int dp[1000005];
bool prime(int n){
for(int i=2;i<=sqrt(n);i++)
    if(n%i == 0)
     return false;
return true;
}
bool have7(int n){
int d;
while(n){
    d = n%10;
    n = n/10;
    if(d == 7)
        return true;
}
return false;
}
void seive(){
 for(int i=2;i<=1000002;i++){
    if(!prime(i) && have7(i))
        dp[i] = dp[i-1]+1;
    else
        dp[i] = dp[i-1];
 }
}
int main(){
int t,n,m,ans;
seive();
cin>>t;
//for(int i=0;i<40;i++)
  //  printf("%d %d\n",i,dp[i]);
while(t--){
    cin>>n>>m;
    ans = dp[m]-dp[n-1];
    if(ans == 0)
     printf("-1\n");
    else
     printf("%d\n",ans);
}
return 0;
}
