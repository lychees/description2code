#define mod 1000000007

#define N 212345

void solve()
{
    int n,i,j;
    int a[20];
    sd(n);
    for(i=0;i<n;++i)
        sd(a[i]);
    int mx = 0,mxi;
    for(i=1;i<(1<<n);++i)
    {
        int xr=0,an=-1;
        for(j=0;j<n;++j)
        {
            if(i&(1<<j))
            {
                xr ^= a[j];
                an &= a[j];
            }
        }
        int rs = (xr|an);
        if(rs>mx)
        {
            mx = rs;
            mxi = i;
        }
        else if(rs==mx)
        {
            if(__builtin_popcount(i) > __builtin_popcount(mxi))
                mxi = i;
        }
    }
    printf("%d %d\n",mx,__builtin_popcount(mxi));
}

int main()
{
    int t = 1;
    scanf("%d",&t);
    while(t--)
    {
        solve();
    }
}
