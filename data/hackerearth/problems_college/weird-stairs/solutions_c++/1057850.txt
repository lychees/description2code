#include <iostream>
#include <stdint.h>
using namespace std;

int main()
{
    int64_t w,n,t,map;
    cin>>t;
    long long int sum;
    int64_t i;
    while(t--){
    	sum = 0;
    	cin>>w;
    	cin>>n;
    	for(long long int i=1;i<=n;i++)
    		{
    			cin>>map;
    			if(map == 1)
    				sum = 2 * i; // that much height has to be covered
    			else
    				sum = sum + 1;
    		}
    	cout<<sum + w * n<<"\n";
    }
    return 0;
}
