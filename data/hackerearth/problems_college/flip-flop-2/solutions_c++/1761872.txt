#include <iostream>
using namespace std;

int main()
{
    int T;
    scanf("%d", &T);
    while (T--) {
    	char c, s[100001];
    	scanf("%s", s);
    	while ((c = getchar()) != '\n' && c != EOF);
    	
    	long d = 0, j = 0;
    	for (long i = 1; s[i] != '\0'; i++) {
    		if (s[j] == s[i]) d++;
    		else {
    			j = i;
    		}
    	}
    	printf("%ld\n", d);
    }
    return 0;
}
