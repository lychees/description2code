#include <bits/stdc++.h>
#define rep(i,x,y) for (int i = (x); i<=(y); i++)
#define repe(i,x,y) for (int i = (x); i < (y);i++)
#define drep(i,x,y) for (int i = (x); i >= (y); i--)
#define mp make_pair
#define pb push_back
#define MAX(a,b) (((a)>(b))?(a):(b))
#define MIN(a,b) (((a)<(b))?(a):(b))
#define MAXN 100100
#define sf(n) scanf("%Lf",&n)
#define prf(n) printf("%Lf",n)
#define	s(n) scanf("%d",&n)
#define sl(n) scanf("%lld",&n)
#define pr(n) printf("%d",n)
#define prl(n) printf("%lld",n)
#define endc printf("\n")
#define psp printf(" ")

using namespace std;
typedef long long ll;
typedef long double ld;
typedef pair<ll,int> pii;

pii B[MAXN];

void solve() {
    int n; s(n); ll sum = 0; B[0] = mp(0,0);
    rep(i,1,n) { int x; sl(x); sum = (sum + x)%n; B[i] = mp(sum,i); }
    sort(B,B+n+1);
    int best = 1e7, x = -1, y = -1;
    rep(i,0,n) {
	if (i+1 <= n && B[i].first == B[i+1].first) {
	    int cur = abs(B[i].second - B[i+1].second);
	    int cx = max(B[i].second,B[i+1].second);
	    int cy = min(B[i].second,B[i+1].second);
	    cy++;
	    if (cur < best || (cur == best && cy < y)) { 
		best = cur; x = cx;
		y = cy;
	    }
	   
	}
    }
    assert(x > 0);
    pr(y); psp; pr(x); endc;
}

int main() {
    int t; s(t); while(t--) solve();
}
