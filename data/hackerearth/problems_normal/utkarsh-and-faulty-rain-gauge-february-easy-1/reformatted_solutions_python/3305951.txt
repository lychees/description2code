T = int(input())
expected_value_list = []
for x in xrange(T):
    N, R = [int(x) for x in raw_input().strip().split()]
    a = R*R*6.0/(N * (N+1)*(2*N+1))
    expected_value_list.append((a/R**3) * (N * (N + 1)/2) ** 2)

result = 1
for val in expected_value_list:
    result *= val
print('%.4f' % (result,))
