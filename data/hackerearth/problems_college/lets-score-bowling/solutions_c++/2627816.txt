#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,a,b,c,ans;
    scanf("%d",&t);
    while(t--)
    {
    	ans=0;
    	scanf("%d%d%d",&a,&b,&c);
    	if((a!=9)&&(b!=9)&&(c!=9))
    		printf("0\n");
    	else
    	{
    		if(a!=9)
    		{
    			if(a>0)
    			{
    				if(a<7)
    					ans+=1;
    				else
    					ans+=2;
    			}
    			if(b!=9)
    			{
    					if(b>0)
    					{
    						if(b<7)
    							ans+=1;
    						else
    							ans+=2;
    					}
    					ans+=4;
    			}
    			else
    				ans+=7;
    			
    			printf("%d\n",ans);	
    		}
    		else
    			printf("10\n");
    	}
    }
    return 0;
}
