#include <bits/stdc++.h>
using namespace std;
int main()
{
	int i,j,k;
	int  t,count,p,l,n,a[1005],b[1005];
	scanf("%d",&t);
	while (t--) {
		scanf("%d%d",&n,&k);
		for (i = 0; i < n; i++) {
			scanf("%d",&a[i]);
			b[i] = __builtin_popcount(a[i]);
			//cout<<b[i]<<" ";
		}
		//cout<<endl;
		sort(b,b+n,greater<int>());
		int ans = 0;
		for (i = 0; i < k; i++) {
			ans += b[i];
		}
		printf("%d\n",ans);
	}
	return 0;
}
