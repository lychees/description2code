string tostring(int n)
{
    string s;
    while (n) {
        s.push_back(n % 10 + '0');
        n /= 10;
    }
    reverse(s.begin(), s.end());
    return s;
}

int main()
{
    ios::sync_with_stdio(false);
    int x, y;
    cin >> x >> y;
    if (y == 1) {
        for (int i = 0; i < x; i++)
            cout << 1 << endl;
        return 0;
    } else if (y == 2){
        for (int i = 0; i < x; i++)
            if (i % 2 == 0) cout << 1 << endl;
            else cout << 2 << endl;
        return 0;
    }
    int mid = (y-1)/2;
    int flag = 0;
    int l = mid, r = mid;
    for (int i = 0; i < x; i++) {
        if (flag == 0) {
            cout << mid + 1 << endl;
            if (y % 2 == 0) {flag = 2;}
            else flag = 1;
        } else if (flag == 1) {
            l--;
            cout << l + 1 << endl;
            if (l == 0) l = mid;
            flag = 2;
        } else if (flag == 2) {
            r++;
            cout << r + 1 << endl;
            if (r == y-1) {
                flag = 0;
                r = mid;
            } else {
                flag = 1;
            }
        }
    }
}
