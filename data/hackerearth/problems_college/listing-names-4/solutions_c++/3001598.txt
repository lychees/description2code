#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main(){
	int n;
	cin>>n;
	set<string> s;
	map<string,int> mp;
	pair<map<string,int>::iterator,bool> ck;
	while(n--){
		string str;
		cin>>str;
		
		ck= mp.insert(make_pair(str,1));
		if(!(ck.second))
			mp[str]=mp[str]+1;
		s.insert(str);
	}
	vector<string> v(s.begin(),s.end());
	sort(v.begin(),v.end());
	for(int i=0;i<v.size();i++)
		cout<<v[i]<<" "<<mp[v[i]]<<endl;
}
