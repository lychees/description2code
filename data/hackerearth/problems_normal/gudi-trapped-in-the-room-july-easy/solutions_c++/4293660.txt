string s;
int a,h;
set<string> st;

void solve(string ss){
	if(st.count(ss)){
		return ;
	}

	st.insert(ss);

	// case-1 rotate
	int n=ss.length();
	string temp (n,'x');

	for(int i=0;i<n;i++){
		temp[(i+h)%n]=ss[i];
	}
//	db(temp) sj
	solve(temp);

	// case-2 add-even
	string temp1 = ss;
	for(int i=0;i<n;i++){
		if(i%2==1){
			int val = temp1[i]-'0';
			val = (val+a)%10;
			char c = val+'0';
			temp1[i]=c;
		}
	}
//	db(temp1) sj
	solve(temp1);
};
