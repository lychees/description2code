#include <cstdio>

using namespace std;

#define MOD 1000000009LL

long long pow(long long n, int val){
	long long temp = 1LL;
	long long v = (long long)val;
	while(n>0){
		if(n%2==1)
			temp = (temp*v)%MOD;
		v = (v*v)%MOD;
		n = n/2;
	}
	return temp;
}

int main(){
	int t;
	long long n;
	scanf("%d", &t);
	while(t--){
		scanf("%lld", &n);
		long long t = pow(n, 10);
		long long e = pow(n, 8);
		long long inv = pow(MOD-2, 2);
		long long ans = ((t-e)*inv)%MOD;
		ans = (ans + MOD)%MOD;
		ans = (ans + MOD)%MOD;
		printf("%lld\n", ans);
	}
	return 0;
}
