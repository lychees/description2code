#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */   
   unsigned long long int n,k,max=0,s=0,temp;
    cin>>n>>k;unsigned long long int a[n];
    for(unsigned long long int i=0;i<n;i++)
    { 
        cin>>a[i];
    }
   
    for(unsigned long long int i=0;i<n;i++)
        {
        if(s==k)
            break;
        max=i;
        for(unsigned long long int j=i+1;j<n;j++)
            {
            if(a[j]>a[max])
                max=j;
        }
        if(i!=max){
        temp=a[i];
        a[i]=a[max];
        a[max]=temp;
            s++; }
        
    }
    while(s<k)
        {          
        temp=a[n-1];
        a[n-1]=a[n-2];
        a[n-2]=temp;
        s++;
    }
    for(unsigned long long int i=0;i<n;i++)
        cout<<a[i]<<" ";
    
    return 0;
}
