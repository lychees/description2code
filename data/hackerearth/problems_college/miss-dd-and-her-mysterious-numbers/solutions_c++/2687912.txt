#include<bits/stdc++.h>
int prime[1000001]={0};
int arr[1000001]={0};
int main()
{
    int n,q,i,j;
    int x=1000001;
    for(i=2;i<=x;i+=2)
        prime[i]=1;
    for(i=3;i<=x;i+=2)
    {
        if(prime[i]==0)
        {
            for(j=i;j<x;j+=i)
                prime[j]++;
        }
    }
    scanf("%d",&n);
    scanf("%d",&q);
    for(i=2;i<=n;i++)
    {
        arr[prime[i]]++;
    }
    while(q--)
    {
        int k;
        scanf("%d",&k);
        if(k<=1000000)
            printf("%d\n",arr[k]);
        else
            printf("0\n");
    }
    return 0;
}
