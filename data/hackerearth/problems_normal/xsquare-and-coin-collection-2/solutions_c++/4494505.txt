#include <iostream>
using namespace std;
//#define max 1000000001;
int main()
{
	long long int t,n,k;
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		long long int a[10000001],dp[10000001],sum=0;
		for(long long int i=0;i<n;i++)
		cin>>a[i];
		if(a[0]<=k)
		dp[0]=a[0];
		
		else
		dp[0]=0;
		sum=dp[0];
		for(long long int i=1;i<n;i++)
		{
			if(a[i]<=k)
			dp[i]=dp[i-1]+a[i];
			else
			dp[i]=0;
			if(dp[i]>sum)
			sum=dp[i];
		}
		cout<<sum<<"\n";
		
	}
    //cout << "Hello World!" << endl;
    return 0;
}
