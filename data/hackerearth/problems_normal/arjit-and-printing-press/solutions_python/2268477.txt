def small(w,r):
    wlist = list(w)
    rlist = list(r)
    rlist.sort()
    len_w = len(w)
    len_r = len(r)
    wi = 0
    ri = 0
    while wi < len_w and ri < len_r:
        if wlist[wi] >  rlist[ri]:
            wlist[wi] = rlist[ri]
            wi+=1
            ri+=1
        else:
            wi+=1

    print(''.join(wlist))

if __name__ == '__main__':
    t = int(raw_input())
    for x in xrange(t):

         small(raw_input(), raw_input())
