/* to find such numbers whose Sum of distinct primes(prime fctorization) is divisible by k for every number 1<=a<=n*/
//or to find sum of distinct primes of a number where 1<=n,k<=10^6
#include<iostream>
using namespace std;
int pr[1<<20];
int main()
{
	int n,k,c=0;
	cin>>n>>k;
	/*concept is similar to prime sieve just that we
	are adding every multiple of prime as we cross over in sieve
	so that common multiples in various numbers get prime sum*/
	
	for(int i=2;i<=n;i++)
		if(pr[i]==0)
			for(int j=i;j<=n;j+=i)
				pr[j]+=i;
	
	for(int i=1;i<=n;i++)
		if(pr[i]==k)
			c++;
			
		cout<<c<<"\n";
	
			
}
