#include<iostream>
#include <string.h>
#include <string>
using namespace std;

int main()
{
    int n;
    cin >> n;
    string city[1001];
    for(int i=0;i<n;i++)
    {
        cin >> city[i];
    }
    long long cost[n][n];
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<n;j++)
        {
            cin >> cost[i][j];
        }
    }
    int p;
    cin >>p;
    string f;
    int asdf[p];
    for(int j=0;j<p;j++){
        
        cin >> f;
        for(int i=0;i<n;i++)
        {
            if(f.compare(city[i]) == 0)
            {
             //   cout << city[i]<<endl;
                asdf[j] = i;
                break;
            }
        }
    }
    long long sum =cost[0][asdf[0]];
    for(int i=1;i<p;i++)
    {
        sum = sum + cost[asdf[i-1]][asdf[i]];
      //  printf("%d\n",asdf[i]);
    }
    printf("%lld\n",sum);
    return 0;
}
