#include <iostream>
#include <iomanip>
using namespace std;

int main()
{
    unsigned long long int noofcases,N,X,i,j,temp,fact;
    double count=0;
    cin>>noofcases;
    cout<<fixed<<setprecision(0);
    for(i=0;i<noofcases;++i)
    {
    	count=0;
    	cin>>N>>X;
    	double A[N];
    	cin>>A[0];
    	for(j=1;j<N;++j)
    	{
    		cin>>A[j];
    		if(A[j-1]>=A[j])
    		{
    			temp = A[j-1]-A[j];
    			fact = 1;
    			if(temp%X==0&&temp!=0)
    			{
    				fact=(temp/X);
      			}else
    			{
    				fact=(temp/X)+1;
     			}
     			count=count+fact;
    			A[j]+=(X*fact);
    		}
    	}
    	cout<<count<<"\n";
    	
    }
    return 0;
}
