#include <iostream>
using namespace std;

int main()
{
	long long tri[100010];
	tri[0]=0;
	tri[1]=0;
	tri[2]=1;
	for(int i=3;i<100010;i++	)
	{
		tri[i] = (tri[i-1] + tri[i-2] + tri[i-3])%1000000007;
	}

	int times;
	cin >> times;

	while(times>0)
	{
		int n;
		cin >> n;
		cout << tri[n+2]<< endl;

		times--;
	}
	return 0;

}
