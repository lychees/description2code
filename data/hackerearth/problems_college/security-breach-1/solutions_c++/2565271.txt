#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t,i;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	for(i=0;s[i];i++)
    	{
    		if(s[i]=='2')
    		cout<<"cde";
    		else if(s[i]=='7')
    		cout<<"acf";
    		else if(s[i]=='9')
    		cout<<"c6a";
    		else if(s[i]=='6')
    		cout<<"b3";
    		
    	}
    	cout<<endl;
    }
    return 0;
}
