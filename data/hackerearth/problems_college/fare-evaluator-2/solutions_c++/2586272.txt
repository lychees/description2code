#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,d;
    	cin>>n>>d;
    	if(n > 60)
    		{
    			cout<<"NA"<<endl;
    			continue;
    		}
    	long int tot;
    	if(n>2)
    		{
    			tot = 50 + (n-2) *12 + 2* d;
    		}
    	else
    		{
    			tot = 50 + 2 *d;
    		}
    	cout<<tot<<endl;
    }
}
