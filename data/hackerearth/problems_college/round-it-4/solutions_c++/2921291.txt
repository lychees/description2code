#include <iostream>
#define ll long long
using namespace std;

int main()
{
	ll t, n, n1, n2, b;
    cin>>t;
    while(t--)
    {
    	cin>>n>>b;
    	n1=n2=n;
    	while(n1%b!=0 && n2%b!=0)
    	{
    		n1--;n2++;
    	}
    	if(n1%b==0 && n2%b==0)
    		cout<<n2<<endl;
    	else if(n1%b==0)
    		cout<<n1<<endl;
    	else
    		cout<<n2<<endl;
    }
    return 0;
}
