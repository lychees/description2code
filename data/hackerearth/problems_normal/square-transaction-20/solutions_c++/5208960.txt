#include<bits/stdc++.h>

using namespace std;

 
int main() {
	long t, i, q, query, arr[100010], j;
	
	cin>>t;
	for(i=0;i<t;i++) {
		cin>>arr[i];
		if(i!=0)
			arr[i] += arr[i-1];
	}
	
	cin>>q; 
	
	for(i=0;i<q;i++) {
		
		cin>>query;
		if(arr[t-1] < query)
			cout<<-1;
		else {
			for(j=0;j<t;j++) {
				if(arr[j] >= query) {
					cout<<j+1;
					break;
				}
			}
		}
		cout<<"\n";
	}
 
	return 0;
}
