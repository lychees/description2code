int calccnt(int foo) {
	int cnt;
	cnt=0;
	while(foo!=0) {
		if(foo>5) {
			cnt++;
			foo-=5;
		}
		else {
			cnt++;
			if(foo==1)
				foo=0;
			else if(foo==2)
				foo=0;
			else if(foo==5)
				foo=0;
			else
				foo-=2;
		}
	}
	return cnt;
}

int main()
{
	ios_base::sync_with_stdio(false);
	
	int t,N,i,mincnt,pre,inc;
	cin>>t;

	while(t--) {
		cin>>N;
		vector<int> candy(N);
		for(i=0;i<N;i++)
			cin>>candy[i];
		
		sort(candy.begin(),candy.end());
		mincnt=0;

		pre=candy[0];
		inc=0;
		for(i=1;i<N;i++) {
			candy[i]+=inc;
			if(pre!=candy[i])
				mincnt+=calccnt(candy[i]-pre);
			inc+=candy[i]-pre;
			pre=candy[i];
		}

		cout<<mincnt<<"\n";
	}
	
	return 0;
}
