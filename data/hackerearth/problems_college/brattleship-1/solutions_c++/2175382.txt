#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    
    for(int i=1;i<=n;i++)
    {
    	int r,c,w;
    	cin>>r>>c>>w;
    	int count=0;
    	
    	int temp=c/w;
    	count=temp+w-1;
    	
    	count=count*r;
    	
    	cout<<"Case #"<<i<<": "<<count<<"\n";
    }
    return 0;
}
