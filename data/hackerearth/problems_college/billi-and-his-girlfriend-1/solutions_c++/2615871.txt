#include<bits/stdc++.h>
#define ull unsigned long long
using namespace std;
int func(ull base,ull exponent,ull modulo){
    ull result =1;
    base=base%modulo;
    while(exponent>0){
        if(exponent%2==1){
            result=(result*base)%modulo;
        }
        exponent=exponent>>1;
        base=(base*base)%modulo;
    }
    int p=0;
  //  cout<<result<<endl;
    while(result>0){
        p+=result%10;
        result/=10;
    }
    return p;
}
int main(){
    int t;
    cin>>t;
    while(t--){
        ull a,b,k;
        cin>>a>>b>>k;
        ull p=pow(10,k)+0.5;
        //cout<<p<<endl;
        cout<<func(a,b,p)<<endl;
    }
}
