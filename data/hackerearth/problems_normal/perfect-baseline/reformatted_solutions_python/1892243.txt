for _ in xrange(input()):
    n, k = map(int, raw_input().split())
    lst = []
    for _ in xrange(n):
        lst.append(raw_input())

    a = []
    ans = []
    for j in xrange(k):
        a = sorted([lst[i][j] for i in xrange(n)])
        ans.append(min(a[n/2], a[(n-1)/2]))

    print "".join(ans)
