#include <algorithm>
#include <cstdio>
#include <cstdlib>
#include <cstring>
using namespace std;
long long ans, a[120000];
int i, j, n;
int main() {
  scanf("%d", &n);
  for (i = 1; i <= n; ++i)
    scanf("%d", &a[i]);
  for (i = 1; i < n; ++i) {
    for (j = 0;; ++j)
      if (i + (1 << j) > n)
        break;
    j--;
    a[i + (1 << j)] += a[i];
    ans += a[i];
    a[i] = 0;
    printf("%I64d\n", ans);
  }
}
