#include<bits/stdc++.h>
using namespace std;
int main(){
	int  t;
	scanf("%d",&t);
	while(t--){
		int n;
		scanf("%d",&n);
		int a[n];
		for(int i=0;i<n;i++)
		  scanf("%d",&a[i]);
		sort(a,a+n);
		int added=0,ans=0;
		for(int i=1;i<n;i++){
			a[i]+=added;
			int x=a[i]-a[i-1];
			added+=x;
			ans+=x/5;
			x=x%5;
			ans+=x/2;
			x%=2;
			ans+=x;
		}
		printf("%d\n",ans);
	}
	return 0;
}
