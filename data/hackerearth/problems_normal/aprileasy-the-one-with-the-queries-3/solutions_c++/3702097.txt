#include <bits/stdc++.h>
using namespace std;

int main()
{
	char s[100005];
	scanf("%s",s);
    map<char,vector<int> > m;
    int a[27];
    for(int i=0;i<26;i++)
    	a[i]=0;
    for(int i=0;s[i]!='\0';i++)
    {
    	m[s[i]].push_back(i);
    	a[s[i]-97]++;
    }
    int n;
    scanf("%d",&n);
	while(n--)
	{
		char str[100005];
		scanf("%s",str);
		int prev=-1;
		bool ans=true;
		int temp[27]={0};
		for(int i=0;str[i]!='\0';i++)
		{
			temp[str[i]-97]++;
		}
		for(int i=0;i<26;i++)
		{
			if(temp[i]>a[i])
			{
				ans=false;
				break;
			}
		}
		if(ans)
		{
			for(int i=0;str[i]!='\0';i++)
			{
				if(m[str[i]].size()!=0)
				{
					bool flag=false;
					vector<int> :: iterator itr=m[str[i]].begin();
					vector<int> :: iterator itr1=m[str[i]].end();
					//cout<<prev<<endl;
					int pos=upper_bound(itr,itr1,prev)-itr;
					//cout<<pos<<" "<<temp[pos]<<endl;
					//cout<<pos<<endl;
					vector<int> temp=m[str[i]];
					if(pos<m[str[i]].size())
					{
						//cout<<temp[pos]<<endl;
						prev=temp[pos];
						flag=true;
					}
					else
					{
						ans=false;
						break;
					}
				}
				else
				{
					ans=false;
					break;
				}
			}
		}
		if(ans)
		printf("Yes\n");
		else
		printf("No\n");
	}
    
    return 0;
}
