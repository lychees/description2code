#include <bits/stdc++.h>
 
#define fi first
#define se second
#define pb push_back
#define mp make_pair
#define pii pair<int,int>
#define pll pair<ll,ll>
#define MOD 1000000007
#define ll long long int
#define MAXINT 2000000000
#define SET(x,y) memset(x,y,sizeof(x));
 
using namespace std;

map<int,int> M;

int main() {
	int n,i,x,y,ans;
	cin>>n;
	for(i=1;i<=n;i++) {
		cin>>x;
		if(M.find(x)==M.end())	M[x]=1;
		else					M[x]++;
	}
	ans=-1;
	for(map<int,int> :: iterator  it =M.begin();it!=M.end();it++) {
		x=it->fi;
		y=it->se;
		if(y == 1)	ans=x;
	}
	cout<<ans<<endl;
	return 0;
}
