#include <iostream>
using namespace std;
typedef long long int lli;

lli gcd(lli a, lli b){
	if(a == 0)
		return b;
	else if(b == 0)
		return a;
	else
		return gcd(b, a%b);
}

int main()
{
	int t;
	cin>>t;
	while(t--){
		lli a,b,c;
		cin>>a>>b>>c;
		lli g = gcd(a, gcd(b,c));
		if(g == 1){
			
			lli a1, a2;
			
			if(a>b && a>c){
				a1 = a*a;
				a2 = b*b + c*c;
			} else if(b>a && b>c){
				a1 = b*b;
				a2 = a*a + c*c;
			} else {
				a1 = c*c;
				a2 = a*a + b*b;
			}
			
			if(a1 == a2){
				cout<<"YES\n";
			} else {
				cout<<"NO\n";
			}
		} else {
			cout<<"NO\n";
		}
	}
    return 0;
}
