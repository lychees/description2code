#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#define MAX 51
using namespace std;

long int coin_ch(int array[MAX],int amount,int no_coins){
    long int table[amount+1][no_coins];
    int i,j;
    long int inc,non_inc;
    for(i=0;i<no_coins;i++)
        table[0][i]=1;
    for(i=1;i<=amount;i++)
    {
        for(j=0;j<no_coins;j++)
        {
            inc=(i-array[j]>=0)?table[i-array[j]][j]:0;
            non_inc=(j>=1)?table[i][j-1]:0; 
            table[i][j]=inc+non_inc;
        }
    }
    return table[amount][no_coins-1];
}

int main() {
    int amount,no_coins;
    cin>>amount>>no_coins;
    int array[MAX];
    for(int i=0;i<no_coins;i++)
        cin>>array[i];
    long int count=coin_ch(array,amount,no_coins);
    printf("%ld",count);
    return 0;
}
