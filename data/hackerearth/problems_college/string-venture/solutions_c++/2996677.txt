#include <iostream>
#include <string>
#include <cstdlib> 
#include <bits/stdc++.h>

using namespace std;

int main()
{
	string s;
	getline(std::cin,s);
	int sum=0;
	int *arr = new int[26];
	for(int i=0;i<26;i++) arr[i]=0;
	for(int i=0;i<s.length();i++)
	{
		sum=sum+s[i];
		if(s[i]-'a'<26&&s[i]-'a'>-1)
		arr[s[i]-'a'] += 1;
		else 
		arr[s[i]-'A'] +=1;
}

	sum=sum/(s.length());
//cout<<s.length();
	if(sum%2==1) {
		int max = 0;
		int m = 0;
		for(int i=0;i<26;i++) {
			if(arr[i]>max) {max = arr[i]; m=i;
			}
		}
		cout<<(char('a'+m));
	}
	else {
		for(int i=0;i<s.length();i++) cout<<s[s.length()-i-1];
	}
}
