#include <iostream>
using namespace std;

int main()
{
    long long n,q;
    cin>>n>>q;
    long long a[n],i;
    for(i=0;i<n;i++)
    cin>>a[i];
    long long x,b,c,k;
    long long h[]={16,9,26,26,1};
    while(q--)
    {
    	cin>>x>>b>>c;
    	if(x==0)
    	{
    		for(i=b-1;i<c;i++)
    		a[i]=a[i]+h[(i-b+1)%5];
    	}
    	else
    	{
    		k=0;
    		for(i=b-1;i<c;i++)
    		k=k+a[i];
    		cout<<k<<"/"<<c-b+1<<"\n";
    	}
    }
    return 0;
}
