#include <iostream>
using namespace std;

int main()
{
    int T,N,K;
    cin>>T;
    int a,b;
    bool flag;
    while(T--)
    {
    	flag=false;
    	cin>>N>>K;
    	if(K<=81)
    	{
    		a=K/9;
    		a++;
    		b=K%9;
    		if(b==0)
    		{
    			b=9;
    			a--;
    				
    		}
    		else
    		{	
    			if(b<=a)
    				b--;
				//else
				//	b--;
    		}
    		if(a==b)
    		{	
    			if(a==9)
    				b--;
    			//else
    			//	b++;
    		}
    		while(N--)
    		{	
    			flag=!flag;
    			if(flag)
    				cout<<a;
    			else
    				cout<<b;
    		}
    	}
    	else
    		cout<<-1;
    	cout<<endl;
    }
    return 0;
}
