#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t ; cin>>t ; 
    while(t--){
    	int n ; cin >>n ; int c2[n+1],c1[n+1]; string ans ="inverse";
    	for(int i=1;i<=n;i++){
    		int x ; cin >>x ; c1[i]=x ; 
    		c2[x]=i ; 
    	}
    	for(int i=1;i<=n;i++){
    		if(c1[i]!=c2[i]){
    			 ans="not inverse"; break ;
    		}
    	}
    	cout<<ans<<"\n";
    }
    return 0;
}
