#include <iostream>
#include <math.h>
using namespace std;

int main()
{
    long long int n;
    cin>>n;
    bool p[n+1];
    p[0]=p[1]=false;
    for(int i=2;i<n+1;i++)
    {
        p[i]=true;
    }
    for(int i=2;i<sqrt(n+1);i++)
    {
        if(p[i]==true)
        {
            for(int j=i;i*j<n+1;j++)
            {
                p[i*j]=false;
            }
        }
    }
    long long int sum=0,z;
    if(n>2)
    {
        sum=2;
    }
    for(z=3;z<n;z++)
    {
        if(p[z]==true)
        {
            sum+=z;
        }
    }
    cout<<sum<<endl;
    return 0;
}
