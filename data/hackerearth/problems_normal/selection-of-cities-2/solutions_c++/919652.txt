#include <iostream>
using namespace std;
#define mod 1000000007
long long int modpow(long long int a, long long int n) {
	long long int x = 1, y = a;
	while(n>0) {
		if(n&1) {
			x = x*y;
			x %= mod;
		}
		y = y*y;
		y %= mod;
		n /= 2;
	}
	return x;
}

int main()
{
	int T; cin>>T;
	while(T--) {
		long long int n; cin>>n;
		long long int ans = modpow(2, n) % mod;
		ans--;
		if(ans <0) ans += mod;
		cout << ans << endl;
	}
    
    return 0;
}
