bool isPal[2000][2000];

void solve(){
	
	string s; cin >> s;
	int n = s.size();
	
	for(int i = 0; i < n; ++i){
		isPal[i][i] = 1;
	}	
	for(int i = 0; i + 1 < n; ++i){
		if(s[i] == s[i + 1])
			isPal[i][i + 1] = 1;
	}

	for(int len = 3; len <= n; ++len){
		for(int  l = 0; l < n; ++l){
			int r = l + len - 1;
			if(r >= n) continue;
			if(s[l] == s[r])
				isPal[l][r] = isPal[l + 1][r - 1];
		}
	}

	ll ans = 0;
	for(int i = 0; i < n; ++i){
		for(int j = i; j < n; ++j){
			if(isPal[i][j])
				ans += 1ll * (j - i + 1) * (j - i + 1);	
		}
	}
	cout << ans << "\n";
}

int main(){
	std::ios::sync_with_stdio(false);
	//freopen("input.in","r",stdin);
	//freopen("output.out","w",stdout);

	int t; cin >> t;
	while(t--){

		memset(isPal,false,sizeof(isPal));
	
		solve();
	}
	
	return 0;
}
