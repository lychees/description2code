#include <iostream>
#include <cmath>
using namespace std;

unsigned sym_diff(unsigned a, unsigned b) {
	if (a > b)
		return a - b;
	else
		return b - a;
}

unsigned gcd(unsigned a, unsigned b) {
	while (b) {
		unsigned m = a % b;
		a = b;
		b = m;
	}
	return a;
}

int main()
{
	int m;
	cin >> m;
	unsigned x0;
	cin >> x0;
	unsigned d = 0;
	for (int i = 0; i < m - 1; ++i) {
		unsigned xi;
		cin >> xi;
		d = gcd(d, sym_diff(x0, xi));
	}
	unsigned a = (unsigned)sqrt((double)d);
	for (int i = 2; i < a; ++i) 
		if (d % i == 0)
			cout << i << ' ';
	if (d % a == 0 && a * a != d)
		cout << a << ' ';
	for (int i = a; i >= 2; --i) 
		if (d % i == 0)
			cout << d / i << ' ';
	cout << d << '\n';
    return 0;
}
