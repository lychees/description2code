ll dp[4005][4005];

void pre_process(){
	rep(i,4005){
		dp[0][i] = 1;
		dp[i][0] = 1;
	}
	reps(i,1,4005){
		reps(j,1,4005){
			dp[i][j] = (dp[i-1][j] + dp[i][j-1] + dp[i-1][j-1])%MOD;
		}
	}
}
