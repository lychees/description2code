#include <bits/stdc++.h>

using namespace std;

int main()
{
    char sym;
    int n;
	//getchar();
    while(cin >> sym >> n)
    {
    	//cin >> sym >> n;
        if (n <= 1)
            cout << "INVALID" << endl;
        else if (((sym - 'a') + n) > 26 && (sym >= 'a' && sym <= 'z'))
        	cout << "INVALID" << endl;
        else if (((sym - 'A') + n) > 26 && (sym >= 'A' && sym <= 'Z'))
        	cout << "INVALID" << endl;
        else {
        //cin >> sym >> n;
        int k = 1;
        for (int i = 0;i < n;i++)
        {
            for (int j = 0;j < k;j++)
                cout << sym << " ";
            sym++;
            k++;
            cout << endl;
        }
        }
        
        cout << endl;
    }

    return 0;
}
