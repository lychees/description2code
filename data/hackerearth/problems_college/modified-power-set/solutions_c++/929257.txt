#include <bits/stdc++.h>
using namespace std;
#define MAX 1000005
char s[MAX];

int main()
{
	int t;
	scanf("%d",&t);
	while (t--) {
		int n = 0, ans;
		int a[150] = {};
		scanf("%s", s);
		for (int i = 0; s[i] != '\0'; i++) {
			a[s[i]] = 1;
		}
		for (int i = 0; i < 150; i++) {
			n = n + a[i];
		}
		ans = n * (n + 1) / 2;
		printf("%d\n",ans);
	}
    return 0;
}
