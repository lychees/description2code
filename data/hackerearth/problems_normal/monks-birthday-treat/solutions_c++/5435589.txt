#include<iostream>
#include<vector>
#include<bits/stdc++.h>
using namespace std;
int dfs(int,vector<int> *);
vector<bool> v1(100001);
int c=0;

int main()
{ int n,a,m,x,y,i;
int minm;
	cin>>n>>m;
	minm=INT_MAX;
		vector<int> v[n+1];
	for(i=0;i<m;i++)
	{	cin>>x>>y;
		v[x].push_back(y);
	}
	for(i=1;i<=n;i++)
	{
		if(v[i].size()==0)
		 {
		 	cout<<"1";
		    exit(0);
		 	
		 }
	}
	for(i=1;i<=n;i++)
	{c=0;
	for(int j=1;j<=n;j++)
		{
			v1[j]=0;
		}
		dfs(i,v);
		if(minm>c)
		minm=c;
	
	}

	cout<<minm;
		
		
return 0;
}
int dfs(int a,vector<int> *v)
{int i;
	
	v1[a]=1;
	c++;
     //cout<<endl<<c<<endl;
     //cout<<endl<<v[a].size()<<endl;
	for(i=0;i<v[a].size();i++)
	{//cout<<endl<<"a"<<endl;
		
		if(v1[v[a][i]]==0)
		{//cout<<endl<<"b"<<endl;
		dfs(v[a][i],v);
	}
	}
//cout<<endl<<"e"<<endl;
	return 0;
}
