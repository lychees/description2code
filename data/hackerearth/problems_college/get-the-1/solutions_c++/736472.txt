#include <iostream>
using namespace std;

int tens(int n)
{
	if(n<10) return n;
	int s=0;
	while (n>0)
	{s+=(n%10)*(n%10);
	n=n/10;}
	return s;
}

int main()
{
	int t,n;
	cin>>t;
	for(int i=1;i<=t;i++)
	{
		cin>>n;
		while(n>=10)
		{n=tens(n);}
		if(n==1) cout<<"Case #"<<i<<":YES\n";
		else cout<<"Case #"<<i<<":NO\n";
}
    return 0;
}
