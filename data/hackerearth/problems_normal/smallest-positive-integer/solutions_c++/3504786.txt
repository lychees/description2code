LL a[10];

bool compute(LL n)
{
	memset(a, 0, sizeof(a));
	while(n%9==0){
		a[9]++;
		n=n/9;
	}
	while(n%8==0){
		a[8]++;
		n=n/8;
	}
	while(n%7==0){
		a[7]++;
		n=n/7;
	}
	while(n%6==0){
		a[6]++;
		n=n/6;
	}
	while(n%5==0){
		a[5]++;
		n=n/5;
	}
	while(n%4==0){
		a[4]++;
		n=n/4;
	}
	while(n%3==0){
		a[3]++;
		n=n/3;
	}
	while(n%2==0){
		a[2]++;
		n=n/2;
	}
	if(n==1)
		return true;
	else
		return false;
}
