#include<iostream>
#include<string.h>
using namespace std;
int main()
{
        int n;
        int a[2000001];
        cin>>n;
        while(n--)
        {
		int t;
		cin>>t;
		int x;
		long long int count=0;	
		memset(a,0,4*2000001);
		while(t--)
		{
			cin>>x;
			if(x<0)
			x=1000001-x;
			a[x]++;
			count+=a[x];
		}
		cout<<count<<"\n";
	}
	return 0;
}
