#include <bits/stdc++.h>
#include <string>
#include <cmath>
#include <algorithm>
using namespace std;
typedef long long int ll;
int main()
{
	ll i,j,k,test,n;
	cin >> n;
		string a,b;
	for(j=0;j<n;j++)
	{
		cin >> a;
		cin >> b;
		ll ans=0;
		ll temp=0;
		for(i=a.length()-1;i>=0;i--)
		{
			ll s=a[i]-'0';
			ll d=b[i]-'0';
			if(s+d+temp> 9)
			{
				ans++;
				temp=s+d+temp-9;
			}
			else
				temp=0;
		}
		if(ans==0)
			cout<<"No carry operation"<<endl;
		else if(ans==1)
			printf("1 carry operation\n");
		else
			printf("%lld carry operations\n",ans );
	}
}
