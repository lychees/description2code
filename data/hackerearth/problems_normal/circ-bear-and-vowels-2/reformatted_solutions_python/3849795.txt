import re
pattern = re.compile('[bcdfghjklmnpqrstvwxz]{3,}')
n = int(raw_input())
for num in range(0, n):
    str = raw_input()
    count = 0
    vowels = ['a', 'e', 'i', 'o', 'u', 'y']
    for char in str:
        if char in vowels:
            count += 1
    if bool(pattern.search(str)):
        print 'hard'
    elif 2*count < len(str):
        print 'hard'
    else:
        print 'easy'
