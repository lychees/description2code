ary = [map(float, raw_input().split()) for i in xrange(3)]
ans = sorted([(ary[0][0]-ary[1][0]+ary[2][0],
               ary[0][1]-ary[1][1]+ary[2][1]),
              (ary[0][0]+ary[1][0]-ary[2][0],
               ary[0][1]+ary[1][1]-ary[2][1]),
              (-ary[0][0]+ary[1][0]+ary[2][0],
               -ary[0][1]+ary[1][1]+ary[2][1])])
for i in ans:
    print "%.4f %.4f" % (i[0], i[1])
