#include<iostream>
using namespace std;

int check(char str[],int l)

{
	int c=0;
	for(int x=0;x<l-1;x++)
	{
		if(str[x]=='.' && str[x+1]=='.')
		c++;
	}
	cout<<c<<endl;;
	
}

int main()
{
	int n,l;
	cin>>l>>n;
	char str[l];
	for(int x=0;x<l;x++)
	cin>>str[x];
	
	for(int x=0;x<n;x++)
	{
		int y;
		char c;
		cin>>y>>c;
		str[y-1]=c;
		check(str,l);
	}
}
