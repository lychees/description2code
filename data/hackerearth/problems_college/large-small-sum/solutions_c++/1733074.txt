#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
    int n,k,h,l;
    cin>>n;
    while(n--)
    {
        cin>>k;
        int a[k];
        for(int i=0;i<k;i++)
        {
            cin>>a[i];
        }
        cin>>h>>l;
        h=k-h;
        l--;
        sort(a,a+k);
        cout<<a[l]+a[h]<<endl;
    }

    return 0;
}
