#include <bits/stdc++.h>
using namespace std;

typedef long long LL;

#define PII       pair<int,int>
#define all(c)    c.begin(),c.end()
#define sz(c)     (int)c.size()
#define clr(c)    c.clear()
#define pb        push_back
#define mp        make_pair
#define cin(x)    scanf("%d",&x)
#define MOD		1000000007
#define EPS		1E-10

const int maxn = 100001;

int dp[maxn][2];
vector<int> edge[maxn];
int reqd[maxn],have[maxn];

int dfs(int u,int p,int inv)
{
	int changeReqd = have[u] ^ inv ^ reqd[u];
	if(dp[u][inv] >= 0) return dp[u][inv];
	int ret = 1000000;
	if(changeReqd)
	{
		int a = 0,b = 0;
		for(int v: edge[u])
			if(v != p)
			{
				a += dfs(v,u,inv);
				b += dfs(v,u,!inv);
			}
		a += 1;
		b += 1;
		ret = min(a,b);
	}
	else
	{
		int a = 0,b = 0;
		for(int v: edge[u])
			if(v != p)
			{
				a += dfs(v,u,inv);
				b += dfs(v,u,!inv);
			}
		b += 2;
		ret = min(a,b);
	}
	return (dp[u][inv] = ret);
}

int main()
{
	memset(dp,-1,sizeof(dp));
	int n;
	cin(n);
	for(int i = 1; i < n; i++)
	{
		int u,v;
		cin(u);
		cin(v);
		edge[u].pb(v);
		edge[v].pb(u);
	}
	for(int i = 1; i <= n; i++)
		cin(have[i]);
	for(int i = 1; i <= n; i++)
		cin(reqd[i]);
	printf("%d", dfs(1,0,0));
	return 0;
}
