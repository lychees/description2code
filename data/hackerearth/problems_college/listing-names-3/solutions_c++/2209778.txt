#include <iostream>
#include<algorithm>
#include<string.h>
using namespace std;

int main()
{
    int n;
    	cin >> n;
    	string ch[n];
    	int i,j=0;
    	for(i=0;i<n;i++)
    	{
    		cin >> ch[i];
    	}
    	sort(ch,ch+n);
    	for(i=0,j=0;i<n;i++)
    	{
    		int t=0;
    		while(ch[j]==ch[i] && j<n)
    		{
    			j++;
    			t++;
    		}
    		cout << ch[i] << ' ' << t <<'\n';
    		i=j-1;
    		t=0;
    		
    	}
    return 0;
}
