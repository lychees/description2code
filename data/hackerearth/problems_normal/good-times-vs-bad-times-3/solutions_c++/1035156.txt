#include <iostream>
#include<algorithm>
using namespace std;
#include<stdio.h>
long long int ar[25][61][61],ar1[25][61][61];
int main() {
	long long int t;
	long long int x,y,z,c,b,i,j,k;
	long long int d,e;
	scanf("%lld",&t);
	b=0;
	c=0;
	for(i=23;i>=0;i--)
		{
			
			for(j=59;j>=0;j--)
				{
					///if(i==23 && j==60)
					//	j=59;
					for(k=59;k>=0;k--)
					{
					//	if(k==60 && j==59)
					//		k=59;
						 c++;
					e=__gcd(i,j);
					d=__gcd(e,k);
					if(__gcd(d,e)!=1)
						{	b++;
							ar[i][j][k]=b;
							
						//printf("%lld %lld %lld ",i,j,k);
						}
					else
						ar[i][j][k]=b;
						
						ar1[i][j][k]=c;
					}
				}
		}
		
	while(t--)
	{
		c=0;
		b=0;
		scanf("%lld %lld %lld",&x,&y,&z);
		
		
		b=ar1[x][y][z]-ar[x][y][z];
		c=ar[x][y][z];
		long long int e;
		e=__gcd(b,c);
		if(b!=0 &&c!=0 )
		{	while(e!=1)
			{
				b/=e;
				c/=e;
				e=__gcd(b,c);
			
			}
		}
		printf("%lld:%lld\n",c,b);
	
		
	}
	return 0;
}
