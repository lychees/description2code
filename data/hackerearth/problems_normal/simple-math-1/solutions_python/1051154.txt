import sys

def fac(n):
	if n<=1:
		return 1
	else:
		return n*fac(n-1)

n,r = map(int,sys.stdin.readline().split())
if(n>=r):
	print (fac(n-1)/(fac(r-1)*fac(n-r)))%10000007
else:
	print -1
