#include <iostream>
using namespace std;

int main()
{
    int t,bx,by,px,py;
    cin>>t;
    while(t--)
    {
    	
    	cin>>px>>py>>bx>>by;
    	if((py>by&&px>=bx)||(py>=bx&&px>by)||(py>=by&&px>bx)||(py>bx&&px>=by))
    	cout<<"Possible\n";
    	else
    	cout<<"Not Possible\n";
    }
    return 0;
}
