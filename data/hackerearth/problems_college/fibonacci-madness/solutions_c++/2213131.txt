#include <iostream>
using namespace std;
int main()
{
    int i;
   unsigned long long int *dp=new unsigned long long int[103];
    dp[1]=0;
    dp[2]=1;
    
    for(i=3;i<=103;i++)
     {
     	dp[i]=dp[i-1]+dp[i-2];
     }
     
     int t;
          scanf("%d",&t);
     while(t--)
      {
           int t1;
           scanf("%d",&t1);
      	for(i=t1;i>0;i--)
      	 {
      	 	printf("%lld ",dp[i]);
      	 }
      	 
      printf("\n");
      }
    return 0;
}
