#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
#define ll long long
ll cap[10000000];
int cmp(ll a,ll b)
{
	if(a>b)
	return 1;
	return 0;
}
int main()
{
	ll num,t,n,i,count,total;
	scanf("%lld",&t);
	while(t--)
	{
		count=0;
		scanf("%lld",&n);
		total=0;
		for(i=0;i<n;i++)
		{
			scanf("%lld",&num);
			total+=num;
		}
		for(i=0;i<n;i++)
		{
			scanf("%lld",&cap[i]);
		}
		if(total==0)
		cout<<"0\n";
		else
		{
			sort(cap,cap+n,cmp);
		i=0;
		while(i<n)
		{
			if(total<cap[i])
			{
				count++;
				break;
			}
			else if(total>cap[i])
			{
				total-=cap[i];
				count++;
				if(total==0)
				break;
			}
			else
			{
				count++;
				break;
			}
			i++;
		}
		printf("%lld\n",count);
		}
		
	}
return 0;
}
