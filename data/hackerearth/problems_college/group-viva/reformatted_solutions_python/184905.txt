
t = int(raw_input())


def factors(n):

    count = []
    for i in xrange(1, (int(n**.5))+1):
        if n % i == 0:
            count.append(i)
            count.append(n/i)
    return len(set(count))


def gcd(a, b):
    if a % b == 0:
        return b
    return gcd(b, a % b)


def findTotal(n, r):
    return n-factors(r-1)


for asfsd in xrange(t):
    n, r = (int(e) for e in raw_input().split(' '))
    p = findTotal(n, r)

    gc = gcd(p, n)
    print "%d/%d" % (p/gc, n/gc)
