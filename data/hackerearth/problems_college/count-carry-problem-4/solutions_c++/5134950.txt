#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t;
  cin>>t;
   for(;t--;)
   {
   	long long int a,b,count=0,sum=0;
   	cin>>a>>b;
   	while(a!=0 && b!=0)
   	{
   		sum=(a%10)+(b%10);
   		a/=10;
   		b/=10;
   		if(sum>=10)
   		count++;
   	}
   	if(count==0)
   	cout<<("No carry operation")<<endl;
   	else if(count==1)
   	cout<<("1 carry operation")<<endl;
   	else
   	cout<<count<<" carry operations"<<endl;
   }
    return 0;
}
