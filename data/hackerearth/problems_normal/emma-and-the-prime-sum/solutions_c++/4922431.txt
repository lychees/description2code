#include <iostream>
#include <vector>

using namespace std;

constexpr unsigned ymax = 1000000;

vector<unsigned long long> prepare() {
	vector<unsigned long long> sieve(ymax + 1);
	sieve[1] = 1;
	for (unsigned i = 2; i <= 1000; ++i)
		if (sieve[i] == 0) 
			for (unsigned j = i * i; j <= ymax; j += i)
				sieve[j] = 1;
	unsigned long long sum = 0;
	for (unsigned i = 1; i <= ymax; ++i) {
		if (sieve[i] == 0)
			sum += i;
		sieve[i] = sum;
	}
	return sieve;
}

int main() {
	ios::sync_with_stdio(0);
	cin.tie(0);
	auto sums = prepare();
	int t;
	cin >> t;
	while (t--) {
		int x, y;
		cin >> x >> y;
		cout << sums[y] - sums[x - 1] << '\n';
	}
    return 0;
}
