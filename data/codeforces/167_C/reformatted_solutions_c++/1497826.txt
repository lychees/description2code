#include <algorithm>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>

using namespace std;

bool check(long long a, long long b) {
  if (!a || !b)
    return false;
  if (a > b)
    swap(a, b);
  if (!check(a, b % a))
    return true;
  return !(((b / a) % (a + 1)) & 1);
}

int main() {
  int T;
  cin >> T;
  for (; T; --T) {
    long long a, b;
    cin >> a >> b;
    printf("%s\n", (check(a, b)) ? "First" : "Second");
  }
}
