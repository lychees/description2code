#include <iostream>
#include<algorithm>
using namespace std;
int arr[3];
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	for(int i=0;i<3;i++)cin>>arr[i];
    	sort(arr,arr+3);
    	if(arr[2]==arr[1])
    	{
    		arr[1] = arr[1] + arr[2];
    	}
    	else 
    	{
    		arr[1] = arr[1]*2;
    		arr[1] = arr[1] + 1;
    	}
    	cout<<arr[1]<<endl;
    }
    return 0;
}
