testCases = int(raw_input())


def closest(a, b, c):
    a, b, c = int(a), int(b), int(c)
    power = int(pow(a, b))
    num1 = power + c - (power % c)
    num2 = power - (power % c)

    if (c - (power % c) > (power % c)):
        return num2
    elif (c - (power % c) < (power % c)):
        return num1
    else:
        return num2


for j in range(1, testCases + 1):
    a, b, c = raw_input().split(" ")
    print closest(a, b, c)
