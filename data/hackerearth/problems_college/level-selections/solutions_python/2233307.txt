x=0
swap_i=0
while k:

    k-=1
    for i in range(x,n):
        if(l[i] >= max_e ):

            swap_i=i
    l[x],l[swap_i]=l[swap_i],l[x]
    x+=1
    
    if(swap_i == x):
    	k+=1
