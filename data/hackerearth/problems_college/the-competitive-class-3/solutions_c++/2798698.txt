#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    int n;
    int frq[1001];
    cin>>n;
    int m[n];
    memset(frq,0,sizeof(frq));
    for(int i=0;i<n;i++){
    	cin>>m[i];
    	frq[m[i]]++;
    }
    for(int i = 1;i<=1000;i++)
    	frq[i]+=frq[i-1];
    	
    for(int i=0;i<n;i++)
    	cout<<(frq[1000]-frq[m[i]])+1<<" ";
    
    
    return 0;
}
