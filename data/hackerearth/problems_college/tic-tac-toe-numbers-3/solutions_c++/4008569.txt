#include <bits/stdc++.h>
using namespace std;
int a[1000006];
int fib(long long int k)
{
	
	int p;
	if(k==0)
	{
		p = 0;
	}
	else if(k==1)
		p = 1;
	else
		p = fib(k-1)+fib(k-2);
	return p;


}
bool pr(long long int s)
{
	int cnt = 0;
	for (int i = 1; i < sqrt(s); ++i)
	{
		if (s%i==0)
		{
			cnt++;
		}
	}
	if(cnt>1)
		return false;
	return true;
}
int main()
{
	int t;
	cin>>t;

	while(t--)
	{
		
		long long int n;
		cin>>n;
		long long int i= 0;
		long long int sum = 0;
		vector < long long int > a;
		while(fib(i)<=n)
		{
			a.push_back(fib(i));
			i++;
		}
		long long int b[1000007];
		for (long long int k = 0; k < n; ++k)
		{
			b[k] = k;
		}
		for (int j = 0; j < a.size(); ++j)
		{
				b[a[j]] = 0;			
		}
		for (int k = 0; k < n; ++k)
		{
			sum = sum+b[k];
		}
		if(pr(sum))
			cout<<"Tic";
		else if(sum%2 ==0)
			cout<<"Tac";
		else
			cout<<"Toe";
		cout<<endl;


	}
}
