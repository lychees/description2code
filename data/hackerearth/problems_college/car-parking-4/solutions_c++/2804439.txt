#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	char s[100000];

	while(n--)
	{
		int count=0,i;
		cin>>s;
		int x=strlen(s);
		int a[x];
		for(i=0;i<x;i++)
			a[i]=0;
		for(i=0;i<x;i++)
		{
			if(s[i]=='D')
			{
				//cout<<i<<endl;
				a[i]=1;
			}
			else if(s[i]=='B')
				{
					//cout<<i<<endl;
					a[i]=1;
					if(i>0)
					{
						a[i-1]=1;
						//cout<<i-1<<endl;
					}
					if(i>1)
					{
						a[i-2]=1;
						//cout<<i-2<<endl;
					}
				}
			else if(s[i]=='S')
			{
				//cout<<i<<endl;
				a[i]=1;
				if(i>0&&i<x)
					{
						a[i+1]=1;
						a[i-1]=1;
						//cout<<i+1<<endl;
						//cout<<i-1<<endl;
					}
				if(i==0)
					a[1]=1;
				if(i==x-1)
					a[x-2]=1;

			}

		}
		for(i=0;i<x;i++)
				if(a[i]==0)
					count++;
			cout<<count<<endl;
	}
return 0;
}
