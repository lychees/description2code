							#include <iostream>
							#include<cstring>
							using namespace std;
							
							int main()
							{
								int tc;
								cin>>tc;
								while(tc!=0){
								char names[200];
									cin>>names;
								
								int OrLen=strlen(names);
								//cout<<names;
								int len=4;
								for(int i=4;i<OrLen-4;i++){
								
									if(names[i]=='a' || names[i]=='e' || names[i]=='i' || names[i]=='o' || names[i]=='u'){
										continue;
									}
									else
									{
										len++;
									}
						}
						cout<<len<<"/"<<OrLen<<endl;
						tc--;
						}
					   // cout << "Hello World!" << endl;
					    return 0;
}
