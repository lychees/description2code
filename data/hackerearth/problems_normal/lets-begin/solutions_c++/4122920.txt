#include<iostream>
#include<vector>
#include<limits.h>
using namespace std;

typedef long long int lli;
void printVector(vector<lli> vec1)
{
	for(int i=0;i<vec1.size();i++)
		cout<<i<<" "<<vec1[i]<<endl;
	
	cout<<endl;
}




int main()
{
	lli NTest;
	cin>>NTest;
	
	lli num;
	//cin>>num;
	num=10000060;
	
	vector<lli> list(num,0);
	vector<lli> prime;
	
	int k=2;
	while(1)
	{
		//cout<<"1 ";
	
		if(k==num) break;
		for(int i=k;i<num;i++)
		{
			if(list[i] == 0)
			{
				k=i;
				break;
			}
		}//k selection completes
		
		//	cout<<k<<endl;
			prime.push_back(k);
			
		for(int i=k;i<num;i=i+k)
			list[i]=1;
		
		k++;
	};
	//prime no up to N has been calculated
	
		lli N;
	//	cin>>N;
	N=1000000;
		
		N=N+1;
		vector<lli> table(N,INT_MAX);
		
		table[0]=0;
		table[1]=-1;table[2]=1;table[3]=1;table[4]=2;table[5]=1;
		table[6]=2;table[7]=1;table[8]=2;table[9]=2;
		
		vector<int> alist={2,3,5,7};
		for(int i=10;i<N;i++)
		{
			for(int j=0;j<alist.size();j++)
			{

				int pnum=alist[j];
				if(pnum >= i ) break;
				
				table[i]=min(table[i],table[i-pnum]) +1;
				//table[i]=min(table)
			}
		}
		
	//	cout<<"Printing the table\n";
	//	printVector(table);
		
		for(int i=0;i<NTest;i++)
		{
			lli N;
			cin>>N;
		
		lli ans=table[N];
		cout<<ans<<endl;
		
		}
		
	
	
	
}
