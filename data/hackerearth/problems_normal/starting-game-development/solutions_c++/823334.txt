#include<bits/stdc++.h>
using namespace std;
#define gc getchar
inline long long int read()
{
	register char c = gc();
	while(c<'0' || c>'9') c = gc();
	long long int ret = 0;
	while(c>='0' && c<='9')
	{
		ret = 10 * ret + c - 48;
		c = gc();
	}
	return ret;
}
int a[101][5001],b[101];
int check(int n,int m)
{
    int sum = m;
	for(int i=0;i<n;i++)
		while(b[i]<a[i][sum-1])
			sum--;
	return sum;	
}

}
