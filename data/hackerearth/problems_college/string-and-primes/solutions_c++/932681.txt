#include<bits/stdc++.h>
using namespace std;
#define MAX 150
long long int arr[MAX],temp[MAX];
long long int s=0;
void prime(long long int arr[],long long int i,long long int n)
{
  long long int a=2;
  while(a*i<=n)
  {
     arr[a*i]=1;
     a++;
   }
}
void sieve(){
 long long int i;
 memset(arr,0,sizeof(arr));
 memset(temp,0,sizeof(temp));
 arr[0]=1;
 arr[1]=1;
 for(i=2;i<=(MAX);i++)
 {
    if(arr[i]==0){
    temp[s++]=i; 
    prime(arr,i,MAX);
    }  
   }
}
int main()
{
  long long int t,i;
  scanf("%lld",&t);
  sieve();
  while(t--){
    string s,ans="";
    cin>>s;
    for(i=0;i<s.length();i++)
        if(arr[(int)(s[i])]==1)
          ans+=s[i];
    if(ans.length()==0 || s.compare(ans)==0) printf("NULL\n");
    else cout<<ans<<endl;
  }
  return 0;
}
