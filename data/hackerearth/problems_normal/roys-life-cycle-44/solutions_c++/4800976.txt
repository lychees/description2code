#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,m;
    cin>>n;
    m=(n*1440)+1;
    char a[m],b[1445];
    
   
    int j=0;
    int k=0;
    int max_day=0,max_day_final=0;
    int max_tot=0,max_tot_final=0;
    while(n--)
    {
    	max_day=0;
    	
    	for(int i=0;i<1440;i++)
	    {
	    	
	    	cin>>b[i];
	    	if(b[i]=='C')
			{
				max_day+=1;
			}
			else
			{	
				if(max_day>max_day_final)
				max_day_final=max_day;
				max_day=0;
			}
	    		
	    		
	    	k=1440*j+i;
	    	a[k]=b[i];
	    	if(a[k]=='C')
	    	{
	    		max_tot+=1;
	    		if(max_tot>0 && max_tot>max_tot_final)
	    			max_tot_final=max_tot;
	    	}else
	    	{
	    		if(max_tot>0 && max_tot>max_tot_final)
	    			max_tot_final=max_tot;
	    		max_tot=0;
	    	}
	    	
	    }
	    if(max_day>max_day_final)
		max_day_final=max_day;
	    j+=1;
	    
	    
	    
	    
    }
    cout<<max_day_final<<" "<<max_tot_final;
    
    
    return 0;
}
