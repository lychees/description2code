#include <iostream>
#include<cmath>
using namespace std;

int getAngle(int h,int m){
	if(h == 12)
		h = 0;
	if(m == 60)
		m = 0;
	int hang,mang,angle;
	hang = (h*60 + m)/2;
	mang = 6*m;
	angle = abs(hang - mang);
	return min((360-angle),angle);
	
}
int main()
{
    int t;
    cin>>t;
    int h,m;
    while(t--){
    	cin>>h>>m;
    	cout<<getAngle(h,m)<<endl;
    }
    return 0;
}
