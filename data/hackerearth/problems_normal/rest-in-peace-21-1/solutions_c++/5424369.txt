#include <iostream>
using namespace std;

int main()
{
    int t,n,r,count,i,p;
    cin>>t;
    while(t)
    {
    	count=0;
    	cin>>n;
        p=n;
        while(n>0)
    		{
    		r=n%10;
    		if(r==1)
    		{
    			i=n/10;
    			if((i%10)==2)
    			{
    				count=1;
    			}
    		}

    			n=n/10;
    		}
        if(p%21==0)
    	{
    	cout<<"The streak is broken!"<<endl;
    	}
    	else if(count==1)
        {
            cout<<"The streak is broken!"<<endl;
        }
        else
        {
        cout<<"The streak lives still in our heart!"<<endl;
        }

    t--;
    }
    	return 0;
    	}
