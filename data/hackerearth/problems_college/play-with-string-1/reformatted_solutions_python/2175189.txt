
t = input()
while t > 0:
    m = raw_input()
    s = m[::-1]
    # print s
    l = len(m)/2
    f = 0
    i = 0
    # print ord(m[1])-ord(m[0])
    while i <= l:
        if abs(ord(m[i+1])-ord(m[i])) == abs(ord(s[i+1])-ord(s[i])):
            f = 1
        else:
            f = 0
            break
        i = i+1
    if f == 1:
        print "Funny"
    else:
        print "Not Funny"
    t = t-1
