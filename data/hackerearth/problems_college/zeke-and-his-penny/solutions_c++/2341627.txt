#include <iostream>
#include<algorithm>
#define MAX 1000000007
using namespace std;
int main()
{
    long long int arr[3],n,i,x;
    cin>>n;
    while(n--)
    {
    	for(i=0;i<3;i++)
    	{
    		cin>>arr[i];
    	}
    	sort(arr,arr+3);
    	arr[0]=arr[0]%MAX;
    	arr[1]=(arr[1]-1)%MAX;
    	arr[2]=(arr[2]-2)%MAX;
    	x=(arr[0]*arr[1])%MAX;
    	cout<<(x*arr[2])%MAX<<endl;
    }
    return 0;
}
