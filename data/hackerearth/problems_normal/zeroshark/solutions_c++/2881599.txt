#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
long long sum=0;
long long dp[100005][2][3];
long long func(long long pos,long long prev,long long n,long long count)
{
        if(pos==n)
        {
            if(count==1)
                return 1;
            else
                return 0;
        }

        if(count>2)
        return 0;
        if(dp[pos][prev][count]!=-1)
        return dp[pos][prev][count]%mod;
        if(prev==1)
        {
            return dp[pos][prev][count]= (func(pos+1,1,n,count)%mod+func(pos+1,0,n,count)%mod)%mod;
        }
        if(prev==0)
        {
            return dp[pos][prev][count]=(func(pos+1,0,n,count+1)%mod+func(pos+1,1,n,count)%mod)%mod;
        }
}
int main()
{
    long long t;
    cin>>t;

    while(t--)
    {
    	memset(dp,-1,sizeof(dp));
        long long n;
        cin>>n;
        cout<<(func(1,1,n,0)%mod+func(1,0,n,0)%mod)%mod<<endl;
    }
}
