#include<iostream>
using namespace std;
#include<algorithm>
long long solve()
{
	
	long long n;
	long long start[100001];
	long long end[100001];
	
	cin>>n;
	for(long long i=0;i<n;i++)
	cin>>start[i];

	for(long long i=0;i<n;i++)
	{
		cin>>end[i];
		end[i]+=start[i];
	}
	sort(start,start+n);
	sort(end,end+n);
	long long s=1;
	long long e=0;
	long long ans=1;
	long long count=1;
	for(long long i=1;i<n;i++)
	{
		if(start[i]<end[e])
		count++;
		//else if(start[i]==end[i])
		//e++;
		else
		{
			i--;
			count--;
			e++;
		}
		if(count>ans)
		ans=count;
		
	}
	
	cout<<ans;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		solve();
		cout<<endl;
	}
}
