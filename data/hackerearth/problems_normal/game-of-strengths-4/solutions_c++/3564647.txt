#include <iostream>
#include<algorithm>
#define mod 1000000007
using namespace std;
void sol()
{
	long long int arr[100001],i,n,max=-1;
	long long int total=0,sum[100001],temp;
	sum[0]=0;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>arr[i];
		if(arr[i]>max) max=arr[i];
	}
	sort(arr,arr+n);
	for(i=0;i<n;i++)
        sum[i+1]=(sum[i]+arr[i]);
 for(i=0;i<n;i++)
 {
  temp=((sum[n]-sum[i+1]-(n-i-1)*arr[i]))%mod;
  //cout<<temp<<'\t';
  total=(total+temp)%mod;
 }
// cout<<total<<endl;
max%=mod;
 cout<<(max*total)%mod<<endl;
}
int main()
{
	int t;
	cin>>t;
	while(t--) sol();
	return 0;
}
