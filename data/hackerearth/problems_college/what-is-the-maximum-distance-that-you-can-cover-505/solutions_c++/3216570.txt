#include<iostream>
#include<limits.h>
#include<cstdio>
#include<stdlib.h>
using namespace std;

int main()
{
    int n,i,index;
    scanf("%d",&n);
    double sp[n], fu[n],rat,maxi,inp;
    int k;
    for(i=0;i<n;i++)
    {
        scanf("%lf",&sp[i]);
    }
    for(i=0;i<n;i++)
    {
        scanf("%lf",&fu[i]);
    }
    scanf("%lf",&inp);
    for(i=0;i<n;i++)
    {
        rat=sp[i]/fu[i];
        
        if(rat>maxi)
        {
            maxi=rat;
            index=i;
        }
    }

    inp/=fu[index]; inp*=sp[index];
    k=inp*(1000); inp=k*(.001);
    printf("%.3lf\n",inp);
    return 0;
}
