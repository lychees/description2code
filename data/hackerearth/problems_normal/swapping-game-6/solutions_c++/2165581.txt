#define trace2(x, y)             cerr << #x << ": " << x << " | " << #y << ": " << y << endl;

#define MAX 100010
#define MOD 1000000007
string s[100010];
string text;
LL x = 1;
void precompute(string str){
	int l = str.length();
	string sub = "";
	s[0] = str;
	while(1){
		for(int j=0;j<l/2;j++){
			sub += str[j];
			sub += str[l-j-1];
		}
		if((l&1))
			sub += str[l/2];

		if(sub == text)	break;

		s[x++] = sub;
		str = sub;
		sub = "";
	}
}
int main()
{_
	LL k;
	cin>>k>>text;
	precompute(text);
	int idx = (k*x-k)%x;
	cout<<s[idx]<<endl;
	return 0; 
}
