#include<iostream>
using namespace std;
main()
{
	int T,N,EL[100005],L,R,i,j,tmp;
	cin>>T;
	while(T--)
	{
		cin>>N;
		for(i=0;i<N;i++)
			cin>>EL[i];
		for(i=0;i<N-1;i++)
		{
			for(j=0;j<N-i-1;j++)
			{
				if(EL[j]<EL[j+1])
				{
					tmp=EL[j];
					EL[j]=EL[j+1];
					EL[j+1]=tmp;
				}
			}
		}
			cin>>L>>R;
		cout<<EL[L-1]+EL[N-R]<<"\n";
	}
}
