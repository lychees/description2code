#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--){
    	long long int n,a,b,c,ans=0;
    	cin>>n>>a>>b>>c;
    	for(int i=0;i<=a;i++){
    		for(int j=0;j<=b;j++){
    			long long int k=n-i-j;
    			if(k>=0)ans+=min(k+1,c+1);
    		}
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
