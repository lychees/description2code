
iters=int(raw_input())

lis=['U','V','W','X','Y','Z']
ans=""

for i in range(iters):
	x=raw_input().split(" ")
	x=[int(x[k]) for k in range(len(x))]
	dumdum=""
	for j in x:
		ans=""
		if(j==0):
			ans='0'
		while(j>0):
			quo=j%16
			if(quo>9):
				helper=quo%9-1
				ans=str(lis[helper])+ans
			else:
				ans=str(quo)+ans
			j=j//16
		if(len(ans)<2):
			ans='0'+ans
		dumdum+=str(" ")+str(ans)
	print 'Case',str(i+1)+':'+str(dumdum)
