#include<stdlib.h>
#include<math.h>
#include <iostream>
using namespace std;

int main() {
	// your code goes here
	int n, t, k, i, j, c, a[21]={}, x, y;
	cin>>n>>t;
	long double z=0;
	for(i=0; i<n; i++){
		cin>>a[i];
	}
	for(i=1; i<(1<<n)-1; i++){
		c=0;
		for(j=0; j<n; j++){
			for(k=j+1; k<n; k++){
				x=1<<j;
				x=x&i;
				x=(x==0)?0:1;
				y=1<<k;
				y=y&i;
				y=(y==0)?0:1;
				if((x==0&&y==1)&&fabs((a[j]-a[k])/(2.0))<=t)c++;
			//	cout<<i<<" "<<j<<" "<<k<<" "<<x<<" "<<y<<" "<<c<<endl;
			}
		}
		//cout<<c;
		z+=(c*(1.0))/(1<<n);
	}
	cout<<z;
	return 0;
}
