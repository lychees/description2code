#include <bits/stdc++.h>
 
#define ll long long int
using namespace std;
 
int main() 
{
     
      ll n,c,p,i,iq1[100001],iq2[100001];
      scanf("%lld %lld %lld",&c,&p,&n);
      for(i=1;i<=100001;i++)
      {
      	iq1[i]=0;
      	iq2[i]=0;
      }
      map<long long int,pair<long long int,long long int> > m;
     
      priority_queue<pair<long long int,long long int>,vector<pair<long long int,long long int> >,greater<pair<long long int,long long int> > > pq;
       ll count[100005]={0};
      for(i=1;i<=c;i++)
      {
          m[i]=make_pair(0,0);
          count[i]=0;
      }
      for(i=1;i<=n;i++)
      {
          scanf("%lld",&iq1[i]);
          m[i]=make_pair(0,iq1[i]);
          count[i]=1;
         // pq.push(make_pair(iq1[i],i))
 
      }
      for(i=1;i<=c;i++)
      {
          ll z;
          z=m[i].first+m[i].second;
          z=z*count[i];
          pq.push(make_pair(z,i));
 
      }
      for(i=1;i<=p;i++)
      {
      	
          scanf("%lld",&iq2[i]);
          
          ll index=pq.top().second;
          printf("%lld ",index);
          m[index].first=m[index].second;
          m[index].second=iq2[i];
          pq.pop();
          ll temp=m[index].first+m[index].second;
          count[index]++;
          temp=temp*count[index];
          
          pq.push(make_pair(temp,index));
      }
 
     return 0;
}
