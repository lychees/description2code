#include<bits/stdc++.h>
using namespace std;

int arr[101][101];

int main()
{
    int n;
    cin>>n;

    for(int i=1; i<=n; i++)
        for(int j=1; j<=n; j++)
            scanf("%d", &arr[i][j]);

    for(int i=n-1; i>=1; i--)
    {
        for(int j=1; j<=n; j++)
        {
            int m = 0;
            int ma = j;

            while(true)
            {
                if(ma == 0)
                    break;
                int t = ma;
                while(t <= n)
                {
                    if(arr[i+1][t] > m)
                        m = arr[i+1][t];
                    t *= 2;
                }
                ma /= 2;
            }

            arr[i][j] += m;
        }
    }

    int ans = 0;
    for(int i=1; i<=n; i++)
        ans = max(ans, arr[1][i]);

    cout<<ans<<endl;
    return 0;
}
