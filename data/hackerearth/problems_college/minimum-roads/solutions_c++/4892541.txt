int visited[100001];
vli adj[100001];

void dfs(li s){
        visited[s]=1;
        li i;
        rep(i,0,adj[s].sz){
                if(!visited[adj[s][i]])
                        dfs(adj[s][i]);
        }
}

int main(){
        li i,n,m,cnt=0L;
        sld(n),sld(m);
        rep(i,0,m){
                li x,y;
                sld(x),sld(y);
                x--,y--;
                adj[x].pb(y);
                adj[y].pb(x);
        }
        rep(i,0,n){
                if(!visited[i]){
                        dfs(i);
                        cnt++;
                }
        }
        pld(cnt-1);
        return 0;
}
