#include <iostream>
using namespace std;


bool a[10000005];
int main()
{
    for(int i=1;i<=10000000;i++)
    	a[i]=false;
    int vals[]={6,9,20};
    a[6]=a[9]=a[15]=a[12]=a[18]=a[20]=true;
    for(int i=21;i<=10000000;i++)
    {
    	for(int j=0;j<3;j++)
    	{
    		if(a[i-vals[j]])
    			a[i]=true;
    	}
    }
    int T,n;
    cin >> T;
    while(T--)
    {
    	cin >> n;
    	if(a[n])
    		cout << "True" << endl;
    	else
    		cout << "False" << endl;
    }
    return 0;
}
