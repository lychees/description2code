#include <iostream>
#include<cstring>
#include<climits>
using namespace std;
int table[11][1004];
int eggdrop(int n,int k)
{
	memset(table,0,sizeof(table));
	for(int i=1;i<=n;i++)
	table[1][i]=i;
	for(int i=1;i<=k;i++)
	table[i][1]=1;
	for(int i=2;i<=k;i++)
	{
		for(int j=2;j<=n;j++)
		{
			table[i][j]=INT_MAX;
			for(int x=1;x<j;x++)
			{
				 int res = 1 + max(table[i-1][x-1], table[i][j-x]);
                if (res < table[i][j])
                    table[i][j] = res;
			}
			
		}
		
	}
	return table[k][n];
	
}
int main()
{
   int q,n,k;
   cin>>q;
   while(q--)
   {
   	cin>>n>>k;
   	cout<<eggdrop(n,k)<<"\n";
   }
    return 0;
}
