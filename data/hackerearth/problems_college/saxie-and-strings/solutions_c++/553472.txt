#include<cstdio>
#include<iostream>
#include<cstring>
using namespace std;
main()
{
int t;
scanf("%d",&t);
while(t--)
{
char s[200];
scanf("%s",s);
int l=strlen(s),done=0;
for(int i=0,j=l-1;i<j;i++,j--)
{
if(s[i]!=s[j])
{
done=1;
break;
}
}
if(done==0)
printf("YES\n");
else
printf("NO\n");
}
return 0;
}
