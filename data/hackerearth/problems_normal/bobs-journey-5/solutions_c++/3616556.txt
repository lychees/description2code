#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
	ll i,t,n,flag,index;
	char temp[50],temp1;
	
	cin>>t;
	while(t--)
	{
		int v[26]={0};
		flag=1;
		cin>>n;
		for(i=0;i<n;i++)
		{
			scanf("%s",temp);
			temp1=temp[0];
			v[temp1-'a']++;
		}
		for(i=0;i<26;i++)
		{
			if(v[i]>=2)
				flag=0;
		}
		if(flag==1)
			cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;
	}
	
	return 0;
}
