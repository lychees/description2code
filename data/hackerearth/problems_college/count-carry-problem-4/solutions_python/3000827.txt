for _ in range(int(raw_input())):
    H,M=raw_input().split()
    x=map(int,H)[::-1]
    y=map(int,M)[::-1]
    carry=0
    ans=0
    for i in range(len(x)):
        if x[i]+y[i] + carry>9:
            carry+=1
            ans+=1

        else:
            carry=0

    if ans==0:
        print "No carry operation"
    
    elif ans==1:
    	print "1 carry operation"

    else:
        print ans,"carry operations"
