#include<iostream>
#include<utility>
#include<vector>
#include<algorithm>
using namespace std;
int main()
{
    int n,k,q,x,sub;
    ios::sync_with_stdio(false);
    cin>>n>>k;
    int arr[n][k];
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<k;j++)
            cin>>arr[i][j];
    }
    cin>>q;
    while(q--)
    {
        cin>>x;
        int num = n;
        int ar[n];
        for(int i=0;i<n;i++)
        ar[i]=i;
        for(int i=0;i<x;i++)
        {

            cin>>sub;
            sub-=1;
            //cout<<sub<<endl;
            vector<pair<int,int> > vec;
            for(int j=0;j<num;j++)
            {
                vec.push_back(make_pair(arr[ar[j]][sub],ar[j]));
            }
            sort(vec.begin(),vec.end());
            //for(int a=0;a<vec.size();a++)
              //  cout<<vec[a].first<<' '<<vec[a].second<<endl;
            //cout<<endl;
            int counter = 0;int s = vec[0].first,l=1;ar[0]=vec[0].second;
            while(l<vec.size()&&vec[l].first==s)
            {
                ar[l]=vec[l].second;
                l++;
            }
            num= l;
        }
        sort(ar,ar+num);
        cout<<ar[0]+1<<endl;
    }


}
