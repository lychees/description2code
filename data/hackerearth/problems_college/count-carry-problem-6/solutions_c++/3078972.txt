#include<iostream>
using namespace std;
int rem(long long);
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long a,b;
		int k=0;
		cin>>a>>b;
		while(a!=0||b!=0)
		{
			if(rem(a)+rem(b)>=10)
			{
				k++;
				
			}
			a/=10;
			b/=10;
		}
		if(k==0)
		cout<<"No carry operation"<<endl;
		else if(k==1)
		cout<<"1 carry operation"<<endl;
		else
		cout<<k<<" "<<"carry operations"<<endl;
		
		
	}
}
int rem(long long n)
{
	int r;
	r=n%10;
	return r;
}
