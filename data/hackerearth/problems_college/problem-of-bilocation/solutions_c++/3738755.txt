#include <iostream>
using namespace std;

int main()
{
    long long int t;
    cin>>t;
    while(t--){
    	long long int A[5];
    	for(int i=0; i<5; i++){
    		cin>>A[i];
    	}
    	long long int r = A[2]-A[4];
    	if(r>=A[0] && r<=A[1])
    		cout<<r<<endl;
    	else if(r<A[0])
    		cout<<"STAY IN LONDON"<<endl;
    	else
    		cout<<A[1]<<endl;
    }
    return 0;
}
