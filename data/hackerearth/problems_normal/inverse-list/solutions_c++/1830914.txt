#include <iostream>
using namespace std;
bool isinverse(int* a,int n)
{
	int i=0;
	while(i<n)
	{
		if(a[a[i]-1]!=i+1){return false;}
		i++;
	}
	return true;
}
int main()
{
    int i=0,j,t,n;int* a=new int;
    cin>>t;
    while(i<t)
    {
    	i++;
    	cin>>n;j=0;
    	while(j<n)
    	{
    		cin>>a[j];
    		j++;
    	}
    	if(isinverse(a,n)){cout<<"inverse"<<endl;}
    	else {cout<<"not inverse"<<endl;}
    }
    return 0;
}
