#include <bits/stdc++.h>
using namespace std;

#define long long long 
int n, m;
vector<long> a;

bool ok(long t) {
	int p = 0;
	long s = 0;
	for (int i = 0; i < n; ++i) {
		s += a[i];
		if (s > t) {
			++p;
			s = a[i];
		}
	}
	++p;
	return p <= m;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cin >> n >> m;
    a.resize(n);
    for (int i = 0; i < n; ++i) cin >> a[i];
    
    long lo = *max_element(a.begin(), a.end());
    long hi = LLONG_MAX - 10;
    while (lo < hi) {
    	long mid = lo + (hi - lo >> 1);
    	if (ok(mid)) hi = mid;
    	else lo = mid + 1;
    	// cout << lo << ' ' << hi << endl;
    }
    cout << lo;
    return 0;
}
