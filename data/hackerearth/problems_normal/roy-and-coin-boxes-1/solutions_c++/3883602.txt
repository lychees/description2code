#include <stdio.h>
#include<iostream>
using namespace std;
int main()
{
    int n,lazy[1000004]={0},c[1000004]={0},m,l,r,x,q,m1;
    scanf("%d",&n);
    scanf("%d",&m);
    m1=m;
    while(m1--)
    {
    	scanf("%d %d",&l,&r);
    	lazy[l]+=1;lazy[r+1]-=1;
    }
    x=0;
    for(int i=1;i<=n;i++)
    {
    	x+=lazy[i];
    	c[x]++;
    }
    for(int i=1;i<=m;i++)
    {
    	//cout<<c[i]<<endl;
    	c[i]+=c[i-1];
    }
    scanf("%d",&q);
    
    while(q--)
    {
    	scanf("%d",&x);
    	if(x<=m)
    	printf("%d\n",c[m]-c[x-1]);
    	else
    	printf("0\n");
    }
    
    return 0;
}
