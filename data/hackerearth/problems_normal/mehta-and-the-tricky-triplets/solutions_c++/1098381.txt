#include<iostream>
using namespace std;
int main()
{
    int n;
    cin>>n;
    long long int a[n],ans,x,rem,ptrn,tmp,i,t;
    long long int c[20]={0};
    for(i=0;i<n;i++)
        cin>>a[i];
    for(i=0;i<n;i++)
    {
        tmp=a[i];
        ptrn=0;
        while(tmp>0)
        {
            rem=tmp%10;
            if(rem==2)
            ptrn|=1;
                if(rem==3)
                  ptrn|=2;
                if(rem==5)
                 ptrn|=4;
                if(rem==7)
                 ptrn|=8;
            tmp/=10;
        }
        c[ptrn]++;
    }
    ans=0;
    int bc,j;
    for(i=1;i<16;i++)
    {
        bc=0;
        for(j=0;j<4;j++)
            if(i&(1<<j))
            bc++;
            t=0;
            for(j=i;j<16;j++)
                if((j&i)==i)
                t+=c[j];
        x=t;
        x=x*(x-1)*(x-2);
        x/=6;
        if(bc%2)
        ans+=x;
            else
            ans-=x;
            }
cout<<ans<<endl;
    return 0;
}
