#include <iostream>
#include<vector>
#include<cstdio>
using namespace std;

vector<int> primes(1000001);

void call(vector<int>& dp, int n,int y){
	for(int i=2;(i*n)<=y;i++){
		if(primes[i*n]==0)
			primes[i*n] = n;
		dp[i*n] = 1;
	}
}

int main()
{
    vector<int> dp(1000001);
    for(int i=2;i<=1000000;i++){
    		if(dp[i]==0){
    			primes[i] = i;
    			call(dp,i,1000000);
    		}
    }
    int t,n;
    scanf("%d",&t);
    while(t--){
    	scanf("%d",&n);
    	printf("%d\n",(n-primes[n]));
    }
    return 0;
}
