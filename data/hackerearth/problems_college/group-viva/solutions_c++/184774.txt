#include<iostream>
using namespace std;

int gcd(int a, int b){
	if(a < b){
		return gcd(b, a);
	}
	if(b == 0){
		return a;
	}
	return gcd(b, a % b);
}

int main(){
    int t, n, r, g, ans, i;
    cin>>t;
    while(t--){
    	cin>>n>>r;
    	r--;
    	ans = 0;
		for(i = 1; i * i <= r; i++){
    		if(r % i == 0){
    			ans += 2;
    			if(r / i == i){
	    			ans--;
	    		}
    		}
    	}
    	g = gcd(n - ans, n);
    	cout<<(n - ans) / g<<"/"<<n/g<<endl;
    }
	return 0;
}
