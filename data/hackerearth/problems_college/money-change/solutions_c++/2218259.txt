#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int c[60],dp[260][60]; //sum & coins
int main()
{
	ll int n,m;//Make sum of n out of m available coins
	ll int i,j;
	scanf("%lld%lld",&n,&m);
	for(i=0;i<m;i++)
	scanf("%lld",(c+i));//Coin values
	for(i=1;i<=n;i++)
	dp[i][0]=0; //Can't make money without any coins
	for(i=0;i<=m;i++)
	{
		dp[0][i]=1;//I have to mak 0-money out of n-coins
	}
	ll int x,y;
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=m;j++)
		{
			if(c[j-1]>i)
			x=0;
			else
			x=dp[i-c[j-1]][j];
			y=dp[i][j-1];
			dp[i][j]=x+y;
		}
	}
	printf("%lld\n",dp[n][m]);
	return 0;
}
