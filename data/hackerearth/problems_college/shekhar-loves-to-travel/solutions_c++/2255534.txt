int find_jump(int *a,int n)
{
    int max_el=0;
    int max1=a[0];
    int count=0;
    int i=0;
    while(i<n)
    {
        if(max1==0)
        {
            return -1;

        }
        max_el=i+1;int j;
        for( j=i+1;j<n&&j<=i+max1;j++)
        {
            if((a[max_el]+max_el)<(a[j]+(j)))
                max_el=j;
               // cout<<"j"<<j<<" max_el"<<max_el;

        }
       // cout<<endl<<"hi";
        if(j==n)
          {

            i=n;
           // cout<<a[n-1];
          }
        else
        {i=max_el;
        //cout<<a[i]<<" ";
        }
        count++;
        max1=a[max_el];

    }
    return count;
}
