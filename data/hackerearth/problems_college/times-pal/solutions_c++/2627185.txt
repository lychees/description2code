#include<cstdio>
#include<iostream>
#include<cmath>
#include<cstring>
#include<algorithm>
using namespace std;
int main()
{
	int t ;
	cin>>t;
	while(t--)
	{
		string str1,str2;
		cin>>str1>>str2;
		int c=0;
		for(int i=0;i<=str1.length();i++)
		{
		 string s;
			if(i== str1.length())
			{
				s = str1.substr(0,i)+ str2;
			}
			else
			{
		 s = str1.substr(0,i)+str2+str1.substr(i);
	}
		string s1 = s;
		reverse(s1.begin(),s1.end());
		if(s1 == s)
		{
			c++;
		}
		}
		cout<<c<<"\n";
	}
	return 0;
}
