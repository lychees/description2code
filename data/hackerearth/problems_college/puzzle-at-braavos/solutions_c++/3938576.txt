using namespace std;

long long int dp[1200000][20];

int main()
{	
	int i, j, t;
	
	dp[3][0] = 4;
	dp[0][0] = 1;
	
	for(i = 1; i <= 18; i++)
	{
		for(j = 0; j <= 100000; j++)
		{
			dp[j][i] = dp[j][i] + dp[j][i-1];
			
			if(3*(1<<i) <= j)
			{
				dp[j][i] += (dp[j - (3*(1<<i))][i-1]*4);
			}
		}
	}
	
	cin>>t;
	
	while(t--)
	{
		long long int n;
		cin>>n;
		
		long long int temp;
		
		temp = (n+3)*(n+2)*(n+1);
		temp = temp/6;
		
		double ans;
		
		ans = dp[n][18];
		ans  = ans/temp;
		ans = 1 - ans;
		
		cout<<fixed<<setprecision(6)<<ans<<"\n";
	}		
}
