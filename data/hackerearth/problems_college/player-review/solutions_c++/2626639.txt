#include <bits/stdc++.h>
typedef long long int ll;
using namespace std;

int main()
{
	ll n, i, j, k;
	cin >> n;
	ll A[n + 1], B[n + 1];
	for(i = 1; i <= n; i++) {
		cin >> A[i];
	}
	ll mm = 0;
	B[0] = INT_MIN;
	for(i = 1; i <= n; i++) {
		B[i] = max(A[i], B[i - 1] + A[i]);
		mm = max(mm, B[i]);
	}
	
	cout << mm << endl;
	
    return 0;
}
