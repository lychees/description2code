#include<bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
int t;
cin>>t;
while(t--){
    string s1,s2;
    cin>>s1>>s2;
    int len1 = s1.length();
    int len2 = s2.length();
    if(len1 == len2){
        for(int i = 0 ; i < len1 ; i++){
            if(s1[i] == s2[i]){

                cout<<"0";
            }else{
             cout<<"1";
            }
        }
    }cout<<endl;
 }
 return 0;
}
