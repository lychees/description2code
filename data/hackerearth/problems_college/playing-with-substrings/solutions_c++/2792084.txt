#include<bits/stdc++.h>
using namespace std;
int main()
{
	string s,t;
	cin>>s;
	cin>>t;
	//string s(str);
	vector<string> v;
		int i,j;int k=0;
		for(i=0;i<s.size();i++)
		{
			k=1;
			for(j=i;j<s.size();j++)
			{
				v.push_back(s.substr(i,k));
				k++;
			}
		}
		sort(v.begin(),v.end());
		long long co=0;
		for(i=0;i<v.size();i++)
		{
			
			if(i==0)
			{
				if(v[i].find(t) != string::npos)
				co++;
			}
			else 
			{
				if(v[i]!=v[i-1])
				{
						if(v[i].find(t) != string::npos)
						co++;
				}
			}
		}
		printf("%lld\n",co);
		return 0;
}
