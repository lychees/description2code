#include <bits/stdc++.h>
using namespace std;

int g[100100];

int main()
{
  ios::sync_with_stdio(0);
  
  for (int i = 2; i <= 100000; i++)
  {
    set <int> s;
    s.insert(0);
    for (int j = 2; j * j <= i; j++)
      if (i % j == 0)
      {
        s.insert(g[i / j]);
        s.insert(g[j]);
      }
    while (s.count(g[i])) g[i]++;
  }
  
  int n, sumXor = 0, x;
  cin >> n;
  while (n--) 
  {
    cin >> x;
    sumXor ^= g[x];
  }
  
  cout << (sumXor ? "ALICE" : "BOB") << endl;
}
