def letsee(name):
    l=len(name)
    s=name[0]
    e=name[l-1]
    t=sorted(name)
    return {"length":l,"start":s,"end":e,"alpha":t}

group=[]

N1=int(raw_input())
for j in range(N1):
    abc=raw_input()
    d=letsee(abc)
    if d not in group:
	    group.append(d)
print len(group)
