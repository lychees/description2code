#include <bits/stdc++.h>
using namespace std;

int main() {
int t,i,d;
	string str1,str,str2;
	cin>>t;
	while(t--)
	{
	str1.clear();str.clear();str2.clear();
	cin>>str1;
	str.assign(str1);
	reverse(str.begin(),str.end());
	if(str1.compare(str)==0)
	{cout<<"-1"<<endl;continue;}
	else{ 
	for(i=0;i<str1.length()/2;i++)
	{
	if(str1[i]==str1[str.length()-i-1])
	continue;
	else{
	d=str1.length();
	str1.erase(i,1);
	str2=str1;
	reverse(str2.begin(),str2.end());
	if(str1.compare(str2)==0)
	{cout<<i<<endl;break;}
	else {cout<<d-i-1<<endl;break; }
	
	}
	}
	}
	}
	return 0;
}
