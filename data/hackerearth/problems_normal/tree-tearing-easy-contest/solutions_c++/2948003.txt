   #include <iostream>
   #include <vector>
   using namespace std;

   std::vector<int> g[10001];
   int k, cnt=0;
   int vis[10001];
   
   void readInp()
   {
      int n, val;
      cin>>k>>n;
      for(int i=2; i<=n; i++){
      cin>>val;
         g[i].push_back(val);
         g[val].push_back(i);
      }
   }

   int dfs(int source)
   {
      if(vis[source])
      return 0;
      else
      vis[source] = 1;

      int siz = g[source].size();
      int s=1,t=0;
      for(int i=0; i<siz; i++){
         t = dfs(g[source][i]);
         if(t >= k){
         t = 0;
         cnt++;
         }
         s += t;
      }
      return s;
   }

   void calc()
   {
      int z = dfs(1);
      if(z >= k)
      {
         cnt++;
         z = 0;
      }
      cout<<cnt<<endl;
   }

   int main()
   {
      readInp();
      calc();
      return 0;
   }
