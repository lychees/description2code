#include<bits/stdc++.h>
using namespace std;
long long phi(int n) {
    long long result = n;
    for(int i = 2; i * i <= n; ++i)
        if(n % i == 0) {
            while(n % i == 0)
                n /= i;
            result -= result / i;
        }
    if(n > 1)
        result -= result / n;
    return result;
}

int fast_int()
{
	int n = 0,f=0; 
	char c = getchar_unlocked();

    while (!(('0' <= c && c <= '9') || c=='-')) 
    {
    	c = getchar_unlocked();
   	}
   	if(c=='-')
   	{ f=1;
   	c=getchar_unlocked(); 
   	}
   	while('0' <= c && c <= '9')
   	{
   		n = (n<<3)+(n<<1) + c - '0';  // same as n=n*10 + (c-'0');
   		c = getchar_unlocked();
   		}
   		if(f==1)
   		n*=-1;
   		
   		return n;
   }

int main()
{
   int t;
   t=fast_int();
   while(t--)
   {
   	 int n,m;
   	 n=fast_int();
   	 m=fast_int();
   	 long long ans=0;
   	 for(int i=1;i*m<=n;i++)
	{
		ans+=(phi(i));
	}	
       printf("%lld\n",ans);
   }
	return 0;
	
}
