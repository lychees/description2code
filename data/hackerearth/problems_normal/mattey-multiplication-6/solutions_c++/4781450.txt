// Example program
#include <iostream>
#include <string>
#include <bitset>
#include <set>

using namespace std;

int main()
{     
    int size;
    cin>>size;
    while(size--)
    {
        unsigned long long n, k;
        cin>> n>>k;
        bitset<64> bitstr = k;
        string str = bitstr.to_string();
        str.erase(0, min(str.find_first_not_of('0'), str.size()-1));
        bool once = false;
        string ans = "";
        for(int i = 0; i < str.length(); i++){
            if(str[i] == '1'){                
                ans = ans + "(" + to_string(n) + "<<"+ to_string(str.length() - i - 1) + ") + ";
            }
        }
        size_t endpos = ans.find_last_of(")");
        if( string::npos != endpos )
        {
            ans = ans.substr( 0, endpos+1 );
        }
        cout<<ans<<"\n";
    }
    return 0;
}
