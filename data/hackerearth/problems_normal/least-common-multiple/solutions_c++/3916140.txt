bool is_prime(long long n){
	for (long long i = 2; i * i <= n; i++){
		if (n % i == 0){
			return false;
		}
	}
	return true;
}

bool is_good(long long n){
	for (long long i = 2; i * i <= n; i++){
		if (n % i == 0){
			while (n % i == 0)
				n /= i;
			return (n == 1);
		}
	}
	return true;
}

int main(){

	long long n;
	cin >> n;
	n = n / 2 + 1;
	while (!is_good(n))
		n++;
	cout << n << "\n";

	return 0;
}
