#include<bits/stdc++.h>
using namespace std;

const int maxn = 1e5+5;
int TC,mp[maxn*2];
char b[2*maxn],a[maxn];
int main() {
    //freopen("input.txt","r",stdin);
    scanf("%d",&TC);
    while(TC--){
        int res = 0;
        scanf("%s%s",a,b);
        int n = strlen(a);
        for(int i = n; i < 2*n; ++i) b[i] = b[i-n];
        mp[0]=-1;
        for(int i = 0, j = -1; i < n;) {
            while(j > -1 && a[i] != a[j]) j = mp[j];
            mp[++i] = ++j;
        }
        for(int i = 0, j = 0; i < 2*n;) {
            while(j > -1 && b[i] != a[j]) j = mp[j];
            ++i; ++j;
            if(i < 2*n &&j >= n){
                ++res;
                j = mp[j];
            }
        }
        printf("%d\n",res);
    }
}
