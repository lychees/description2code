#include <iostream>
#include<map>
#include<string.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string str;
    	cin>>str;
    	string str1 = "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
    	int m = 26;
    	int count[26];
    	memset(count,0,sizeof(count));
    	for(int i=0; i<26; i++)
    	{
    	    count[str1[i]] = m;
    	    m--;
    	}
    	map<char, int> chars;
    	int n = 0;
    	for(int i=0; i<str.length(); i++)
    	{
    		if(chars[str[i]]==0)
    		{
    			n += count[str[i]];
    			chars[str[i]] = 1;
    		}
    	}
    	if(n == str.length())
    	cout<<"Yes\n";
    	else
    	cout<<"No\n";
    	
    }
    return 0;
}
