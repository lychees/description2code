#include <iostream>
using namespace std;
int arr[1000001];
void comp() {
	int i=0;
	arr[0]=0;
	for(i=1;i<100000;i++){
		arr[i]=i+arr[i-1];
	}
}
int main()
{
    //cout << "Hello World!" << endl;
    comp();
    int test;
    cin>>test;
    while(test--){
    	int n;
    	cin>>n;
    	cout<<arr[n-1]<<endl;
    }
    return 0;
}
