#include<bits/stdc++.h>
using namespace std;

char s[101][16];
int main()
{
	ios_base::sync_with_stdio(false);
	int n;
	cin>>n;
	int i=0;
	while(i<n)
	{
		cin>>s[i];
		i++;
	}
	for(int i=0;i<n-1;i++)
	{

		for(int j=i+1;j<n;j++)
		{
			bool flag=true;
			char p[16];
			strcpy(p,s[j]);

			if(strlen(p)==strlen(s[i]))
			{
				for(int k=strlen(s[i])-1;k>=0;k--)
				{
					if(s[i][strlen(p)-k-1]!=p[k])
					{
						flag=false;
						break;
					}
				}
				if(flag)
				{
					cout<<strlen(p)<<" "<<s[i][(strlen(p)/2)]<<endl;
					return 0;
				}
			}
		}
	}
	return 0;
}
