#include <iostream>
using namespace std;

int main()
{
    int test_case;
    cin>>test_case;
    for(int t=0;t<test_case;t++)
    {
        unsigned long int num1,num2;
    	cin>>num1>>num2;
    	int count_carry=0,carry=0;
    	while(num1>0 || num2>0)
    	{
    		int rem1=num1%10;
    		int rem2=num2%10;
    		num1=num1/10;
    		num2=num2/10;
    		int sum=rem1+rem2+carry;
    		if(sum>9)
    		{
    			count_carry++;
    			carry=sum/10;
    		}
    		else
    		    carry=0;
    	}
    	if(count_carry==0)
    	cout<<"No carry operation\n";
    	else if(count_carry==1)
    	cout<<count_carry<<" carry operation\n";
    	else
    	cout<<count_carry<<" carry operations\n";
    }
}
