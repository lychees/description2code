'''

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''

import itertools

T = int(raw_input())

for i in range(T):
    N = int(raw_input())
    A = [int(x) for x in raw_input().split()]
    S = int(raw_input())

    f = False

    for j in range(1, N+1):
        combs = itertools.combinations(A, j)

        for comb in combs:
            sl = 0
            for item in comb:
                sl += item
            if sl == S:
                f = True
                break
        if f:
            break

    if f:
        print "YES"
    else:
        print "NO"
