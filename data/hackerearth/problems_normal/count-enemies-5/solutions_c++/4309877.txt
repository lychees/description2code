#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define MOD 1000000007
#define MAX 100005
ll modexp(ll a, ll b)
{
	ll res = 1;
	while(b)
	{
		res=(res*a)%MOD;
		a=(a*a)%MOD;
		b/=2;
	}
	return res;
}
string str;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int i;
		cin>>str;
		int ans = 0;
		int cans = 0;
		int flag = 0;
		//flg = 0, means no x encountered so far
		for(i=0;i<str.size();++i)
		{
			if(flag == 0)
			{
				if(str[i] == 'O') cans++;
				else if(str[i] == 'X') 
				{
					cans = 0;
					flag = 1;
				}
				if(str[i] == '*')
				{
					ans += cans;
					cans = 0;
				}
			}
			else
			{
				if(str[i] == '*') 
				{
					flag = 0;
					cans = 0;
				}
			}
		}
		ans+=cans;
		cout<<ans<<endl;
	}
    return 0;
}
