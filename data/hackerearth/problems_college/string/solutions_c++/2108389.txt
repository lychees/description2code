#include <iostream>
using namespace std;

int main()
{
    string s;
    cin >> s;
    
    int n;
    cin >> n;
    
    int *A=new int[26];
    for(int i=0;i<26;i++)
    {
    	A[i]=0;
    }
    
    for(int i=0;i<s.length();i++)
    {
    	A[s.at(i)-97]++;
    }
    
    int sum=0;
    for(int i=0;i<26;i++)
    {
    	if(A[i]>0)
    	sum++;
    }
    
    int c=0,x=1,len=s.length();
    while(c<n && sum!=1)
    {
    	for(int i=0;i<s.length();i++)
    	{
    		if(c==n || len==1)
    		{
    			c=n;
    			break;
    		}
    		if(A[s.at(i)-97]==x)
    		{
    			s.at(i)='\0';
    			len--;
    			c++;
    		}
    	}
    	for(int i=0;i<26;i++)
    	{
    		if(A[i]==x)
    		{
    			A[i]=0;
    		}
    	}
    	
    	x++;
    }
    
    sum=0;
    for(int i=0;i<26;i++)
    {
    	if(A[i]>0)
    	sum++;
    }
    
    cout << sum << endl;
    cout << s;
}
