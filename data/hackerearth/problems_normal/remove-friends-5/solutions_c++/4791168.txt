#include <bits/stdc++.h>
using namespace std;
int main()
    {
    	int t;
    	cin>>t;
    	while(t--)
    	{
    		int n,k,temp;
    		cin>>n;
    		cin>>k;
    		list<int> v;
    		list<int>::iterator it;
    		for(int i=0;i<n;i++)
    		{
    			cin>>temp;
    			v.push_back(temp);
    		}
    		for(it=v.begin();it!=v.end();)
    		{
    			int temp1=(*v.begin());
    			if(k==0)
    			break;
    			int val=*it;
    			it++;
    			int val1=*it;
    			if(val<val1)
    			{
    					--it;
    				v.erase(it);
    				int temp2=*it;
    				if(temp1!=temp2)
    				--it;
    				else
    				it=v.begin();
    				k--;
    			}
    			else
    			continue;
    			
    			
    		}
    		for(it=v.begin();it!=v.end();++it)
    		cout<<*it<<" ";
    		cout<<endl;
    	}	
      return 0;
    }
