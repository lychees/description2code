#include <bits/stdc++.h>

using namespace std;

#define for(i, a, b) for(int i = (a); i < int(b); ++i)

typedef long long ll;

int x[3000];
ll dp[3000][3000];
ll n, S, M, P;

ll getCost(ll from, ll to)
{
	if(to > from)
		return P*(to-from);
	return M*(from-to);
}

int main()
{
	cin>>n>>S>>M>>P;		
	for(i,0,n)
		cin>>x[i];
	for(i,0,n+1)
	for(j,0,n+1)
		dp[i][j]=-1000000000000000000LL;
	for(j,0,n)
		dp[0][j]=-getCost(x[0], x[j]);
	for(i,1,n)
	{
		ll bestPrev=0;
		for(j,0,n)
		{
			bestPrev=max(bestPrev, dp[i-1][j]);
		}
		for(j,0,n)
		{
			dp[i][j]=-getCost(x[i], x[j])+max(bestPrev, dp[i-1][j]+S);
		}
	}

	ll ans=0;
	for(j,0,n)
		ans=max(ans, dp[n-1][j]);
	cout<<ans<<endl;
}
