#include <iostream>
using namespace std;

int main()
{
    long T,cc=0,n1,n2,r1,r2,dn,tmpa;
    cin>>T;
    for(int i=0;i<T;i++)
    {
    	cin>>n1;
    	cin>>n2;
    	while(n1!=0 || n2!=0)
    	{
    		r1=n1%10;
    		n1=n1/10;
    		r2=n2%10;
    		n2=n2/10;
    		tmpa=r1+r2;
    		if(tmpa>=10)
    		{
    			cc++;
    		}
    		tmpa=0;
    	}
    	if(cc==0)
    	{
    		cout<<"No carry operation"<<endl;
    	}
    	else if(cc==1)
    	{
    		cout<<cc<<" carry operation"<<endl;
    	}
    	else
    	{
    		cout<<cc<<" carry operations"<<endl;
    	}
    	cc=0;
    }
    return 0;
}
