bool isvowel(char t) {

	

	t = tolower(t);
	if(t == 'a' ||
		t == 'e' ||
		t == 'i' ||
		t == 'o' ||
		t == 'u') {
		return true;
	}
	else
		return false;
}

bool iscons(char t)
{
	if(isvowel(t))
		return false;
	else
		return true;
}
