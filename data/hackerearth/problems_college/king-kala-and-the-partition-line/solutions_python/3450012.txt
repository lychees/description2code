t=input()
while t>0:
	t-=1
	n=input()
	a,b,c=map(int,raw_input().split())
	p=0
	z=0
	while n>0:
		n-=1
		x,y=map(int,raw_input().split())
		if a*x+b*y+c>0:
			p+=1
		elif a*x+b*y+c<0:
			z+=1
	print max(p,z)
