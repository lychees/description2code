#include <iostream>
#include <stdio.h>
#include <string.h>

using namespace std;

main(){
	int n,count=0;
	cin>>n;
	int counter[300];
	while(n){
		char a[200];
		cin>>a;
		int length=0;
		for(int i=3;i<strlen(a);i++){
			if(a[i]=='a'||a[i]=='e'||a[i]=='i'||a[i]=='o'||a[i]=='u'){

			}else{
				length++;
			}
		}
		counter[count]=length;
		count++;
		counter[count]=strlen(a);
		count++;
		n--;
	}
	for (int i = 0; i < count; ++i){
		cout<<counter[i]<<"/"<<counter[i+1]<<endl;
		i++;
	}
}
