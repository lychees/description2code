#include <iostream>
using namespace std;
#define M 1000000007
int main()
{
    int t;
    cin>>t;
    int a,b,c,d;
    
    	c = 1004;
    	d = 1004;
    	int mat[c+1][d+1];
    	mat[c][d]=1;
    	for(int i=c-1;i>=0;i--){
    		mat[i][d]=mat[i+1][d];
    	}
    	for(int i=d-1;i>=0;i--){
    		mat[c][i]=mat[c][i+1];
    	}
    	for(int i=c-1;i>=0;i--){
    		for(int j=d-1;j>=0;j--){
    			mat[i][j]=((mat[i+1][j]+mat[i+1][j+1])%M+mat[i][j+1])%M;
    		}
    	}
    	while(t--){
    	cin>>a>>b>>c>>d;
    	printf("%d\n",mat[1004-c+a][1004-d+b]);
    }
    return 0;
}
