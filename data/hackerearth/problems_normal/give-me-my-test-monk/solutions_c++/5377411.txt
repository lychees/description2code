long long pw(long long x,long long y)
{
	long long res = 1;
	while(y){
		if(y&1)
			res = (res*x)%mod;
		y >>=1;
		x =(x*x)%mod;
	}
	return res;
}

long long der[105];
long long cnr[105][105];
void shark_s()
{
	der[0]=1;
	der[1]=0;
	der[2]=1;
	for(int i=3;i<=100;i++)
	{
		der[i] = ((i-1)*(der[i-1]+der[i-2])%mod)%mod;
	}
	for(int i=0;i<=100;i++)
		for(int j=0;j<=i;j++)
		{
			if(i==j || j==0)
				cnr[i][j]=1;
			else 
				cnr[i][j] = (cnr[i-1][j]+cnr[i-1][j-1])%mod;
		}

	int n,l,r;
	cin>>n>>l>>r;
	long long ways = 0,p=0,q=0;
	for(int i=0;i<=n;i++)
	{

		ways = (der[i]*cnr[n][i])%mod;
		if(i<=r && i>=l){
			p+=ways;
			p%=mod;
		}
		q+=ways;
		q%=mod;
	}
	cout<<(p*(pw(q,mod-2))%mod)<<endl;
	return ;
}
