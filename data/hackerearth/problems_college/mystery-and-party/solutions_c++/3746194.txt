#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
   // cout << "Hello World!" << endl;
   int t;
   cin>>t;
   while(t--)
   {
   	int n,k;
   	cin>>n>>k;
   	ll a[n+1];
   	for(int i=0;i<n;i++)
   	cin>>a[i];
   	sort(a,a+n);
   	cout<<a[k-1]<<endl;
   }
    return 0;
}
