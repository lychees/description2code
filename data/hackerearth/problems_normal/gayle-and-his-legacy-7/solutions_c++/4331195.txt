#include <bits/stdc++.h>
using namespace std;
struct g{
	long long g,p;
};

long long cmp(struct g a,struct g b){
	if(a.p != b.p)
	return a.p<b.p;
	return a.g>b.g;
}
int main(){
	long long n,t,i,power;
	cin>>t;
	while(t--){
		cin>>n>>power;
		struct g arr[n];
		long long flag=0;
		for(i=0;i<n;i++)
			cin>>arr[i].g>>arr[i].p;
		sort(arr,arr+n,cmp);
		//cout<<endl;

		for(i=0;i<n;i++){
			if(power >= arr[i].p)
			power+=arr[i].g;
			else{
				flag=1;
				break;
			}
			//cout<<arr[i].g<<" "<<arr[i].p<<endl;
		}
		if(flag)
			cout<<"NO\n";
		else
			cout<<"YES\n";

	}
	return 0;
}
