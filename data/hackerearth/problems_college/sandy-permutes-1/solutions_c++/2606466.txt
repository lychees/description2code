#include <bits/stdc++.h>
using namespace std;

long long int A[100110];
long long int I[100110];

int main(){
	std::ios::sync_with_stdio(false);
	long long int i,j,k,n,a,b;
	cin >> n >> k;
	
	for (long long int i = 1; i <= n; i++)
	{
		cin >> A[i];
		I[A[i]] = i;
	}

	for (long long int i = 1;i<=n; i++)
	{
		if(k<=0){
			break;
		}
		if(A[i]!=(n-i+1)){
			// int frontNum = A[i];
			long long int backNum = (n-i+1);
			long long int temp;
			temp = I[backNum];
			// I[backNum] = i;
			I[A[i]] = temp;
			// temp = A[i];
			A[i] = A[i] ^ A[temp];
			A[temp] = A[i] ^ A[temp];
			A[i] = A[i] ^ A[temp];
			k--;
		}
	}
	if(n <= 1000){
		for (long long int i = 1; i <= n; i++)
		{
			cout << A[i] << " ";
		}
	}
	else{
		for (long long int i = n; i >=1 ;i--)
		{
			cout <<i << " ";
		}
	}
	cout << endl;
	return 0;
}
