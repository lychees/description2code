#include <iostream>
#include <stack>
using namespace std;

int main()
{
    int n;
    cin>>n;
    
    long int sum=0;
    int j;
    stack<int> s;
    for(int i=0;i<n;i++)
    {
    	cin>>j;
    	if(j==0 & !s.empty())
    		s.pop();
    	else
    		s.push(j);
    }
    while(!s.empty())
    {
    	sum+=s.top();
    	s.pop();
    }
    
    cout<<sum;
    return 0;
}
