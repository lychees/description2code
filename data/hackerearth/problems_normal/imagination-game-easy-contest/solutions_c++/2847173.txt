#include <bits/stdc++.h>

using namespace std;

int A[11];
int X[11];
int N;

int main()
{
    int T;
    scanf("%d", &T);
    while(T--)
    {
        scanf("%d", &N);
        for(int i=0; i<N; i++)
            scanf("%d", A+i);
        for(int i=0; i<N; i++)
            scanf("%d", X+i);
        int x=0;
        for(int i=0; i<N; i++) if(A[i]%2==0 || X[i]*2-1!=A[i])
            x^=(A[i]-X[i])^(X[i]-1);
        if(x)
            printf("First\n");
        else
            printf("Second\n");
    }
    return 0;
}
