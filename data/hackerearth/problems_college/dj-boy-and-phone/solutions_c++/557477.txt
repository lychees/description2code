#include<stdio.h>
#include<string.h>
#include<iostream>
#define MOD 1000000009
using namespace std;
long long  int t,dp[10002][10];
int main()
{
    scanf("%lld",&t);
    long long int i,j;
    for(i=0;i<10002;i++)
    {
        for(j=0;j<10;j++)
        {
            if(i==0)
            dp[i][j]=0;
            else if(i==1)
            dp[i][j]=1;
            else
            {
                if(j==0)
                    dp[i][j]=(dp[i-1][8]+dp[i-1][0])%MOD;
                else if(j==1)
                    dp[i][j]=((dp[i-1][1]+dp[i-1][2])%MOD+dp[i-1][4])%MOD;
                else if(j==2)
                    dp[i][j]=((dp[i-1][1]+dp[i-1][2])%MOD+(dp[i-1][3]+dp[i-1][5])%MOD)%MOD;
                else if(j==3)
                    dp[i][j]=((dp[i-1][2]+dp[i-1][3])%MOD+dp[i-1][6])%MOD;
                else if(j==4)
                    dp[i][j]=((dp[i-1][1]+dp[i-1][4])%MOD+(dp[i-1][5]+dp[i-1][7])%MOD)%MOD;
                else if(j==5)
                    dp[i][j]=((dp[i-1][2]+dp[i-1][4])%MOD+((dp[i-1][5]+dp[i-1][6])%MOD+dp[i-1][8])%MOD)%MOD;
                else if(j==6)
                    dp[i][j]=((dp[i-1][3]+dp[i-1][5])%MOD+(dp[i-1][6]+dp[i-1][9])%MOD)%MOD;
                else if(j==7)
                    dp[i][j]=((dp[i-1][4]+dp[i-1][7])%MOD+dp[i-1][8])%MOD;
                else if(j==8)
                    dp[i][j]=((dp[i-1][5]+dp[i-1][7])%MOD+((dp[i-1][8]+dp[i-1][9])%MOD+dp[i-1][0])%MOD)%MOD;
                else
                    dp[i][j]=((dp[i-1][8]+dp[i-1][6])%MOD+dp[i-1][9])%MOD;
            }
        }
    }
    while(t--)
    {
        long long  int n,sum=0;
        scanf("%lld",&n);
        for(i=0;i<10;i++)
        {
            sum=(sum+dp[n][i])%MOD;
        }
        printf("%lld\n",sum);
    }
    return 0;
}
