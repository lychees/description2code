#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	long long unsigned int ans;
    	cin>>n;
    	if(n==1)
    	cout<<1<<endl;
    	else
    	{
    		ans=pow(2,n-1)*((n*(n+1))/2);
    		cout<<ans<<endl;
    	}
    	
    }
    return 0;
}
