    #include <bits/stdc++.h>
    using namespace std;char arr[1000001];
    #define PII pair <int,int>
    PII m [1000001];
    int main()
    {
    int a;scanf("%d", &a);
    for (int g=0; g<a; g++)
    {
    string a; scanf("%s", arr);
    int k=strlen(arr); int la=-1; int lb=-1; int lc=-1;
    for (int g=k-1; g>=0; g--)
    { m[g]=PII(-1,-1);
    if (arr[g]=='x')
    {
    m[g]=PII(lb,lc); la=g;
    }
    else if (arr[g]=='y')
    {
    m[g]=PII(la,lc); lb=g;
    }
    else
    {
    m[g]=PII(la,lb); lc=g;
    }
    }long long ans=0;
    for (int g=0; g<k; g++)
    {
    if (m[g].first==-1 || m[g].second==-1)
    {
    ans+=k-g;
    }
    else
    {
    long long r=max(m[g].first,m[g].second);
    ans+=r-g;
    }
    } printf("%lld\n", ans);
    }return 0;
    }
