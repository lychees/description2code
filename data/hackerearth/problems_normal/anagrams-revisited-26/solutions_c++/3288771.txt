#include <iostream>
#include<string>
#include<map>
#include<algorithm>
#include<cstdio>
using namespace std;

int main()
{
	int n;
	cin>>n;
	map<string ,int > m;
	map<string,int>::iterator it;
	for(int i=0;i<n;i++)
	{
		string s;
		cin>>s;
		sort(s.begin(),s.end());
		m[s]+=1;
	}
	int max=-1;
	it=m.begin();
	while(it!=m.end())
	{
		if(it->second>max)
		max=it->second;
		it++;
	}
	cout<<max<<endl;
    //cout << "Hello World!" << endl;
    return 0;
}
