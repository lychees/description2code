#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n,m;
    scanf("%d %d",&n,&m);
    long long a[n],b[m];
    long long ans1=0,ans2=0;
    for(int i=0;i<n;i++)
    {
    	scanf("%lld",&a[i]);
    	ans1+=a[i];
    }
    for(int i=0;i<m;i++)
    {
    	scanf("%lld",&b[i]);
    	ans2+=b[i];
    }
    printf("%lld\n",ans1*ans2);
    return 0;
}
