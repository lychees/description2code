import sys

cache = range(12)
for i in xrange(12, 1000001):
    cache.append(cache[i/2] + cache[i/3] + cache[i/4])


def solve(n):
    if n <= 1000000:
        return cache[n]
    r1 = n/2
    r2 = n/3
    r3 = n/4
    if r1+r2+r3 <= n:
        return n
    else:
        return solve(r1) + solve(r2) + solve(r3)


ll = map(int, sys.stdin.readlines())
for n in ll:
    print solve(n)
