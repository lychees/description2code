#define mod 1000000007
long long int a[100009];
long long int solve(int n)
{
	if(n<0)
		return 0;
	if(a[n]!=-1)
		return a[n];
	if(n==0)
		return a[n]=1;
	long long int ans=0;
	ans = (ans +solve(n-1))%mod;
	ans = (ans +solve(n-2))%mod;
	ans = (ans +solve(n-3))%mod;
	return a[n] =ans;
}
