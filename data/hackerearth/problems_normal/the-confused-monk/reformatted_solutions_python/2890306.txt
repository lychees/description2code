def gcd(a, b):
    a = abs(a)
    b = abs(b)
    while a:
        a, b = b % a, a
    return b


def gcd_list(inp):
    return reduce(gcd, inp)


def mul(inp):
    val = 1
    for i in inp:
        val *= i
    return val


if __name__ == "__main__":
    tc = input()
    inp = map(int, raw_input().split())
    gcd_val = gcd_list(sorted(inp))
    mul_val = mul(inp)
    monk_quo = (10 ** 9) + 7
    print(mul_val ** gcd_val) % monk_quo
