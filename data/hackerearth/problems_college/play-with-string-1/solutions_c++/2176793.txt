#include<iostream>
#include<string.h>
#include<stdlib.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		char str[100000];
		int flag=0;
		cin>>str;
		int l=strlen(str);
		char rev[10000];
		int j=0;
		for(int i=l-1;i>=0;i--)
		{
			rev[j++]=str[i];
		}
		for(int i=1;i<l;i++)
		{
			if(abs(str[i]-str[i-1])!=abs(rev[i]-rev[i-1]))
			{
				flag=1;
				break;
			}
		}
		if(flag)
			cout<<"Not Funny"<<endl;
		else
			cout<<"Funny"<<endl;
	}
}
