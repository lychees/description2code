#include <iostream>
#include<vector>
#include<queue>

using namespace std;

typedef pair<int,int> pii;
typedef vector<pii> vii;
typedef vector<vii> vvii;

int n;
int dist[505];
vvii A;

void dj(int s)
{
	priority_queue <pii, vector<pii>,greater<pii> > pq;
	pq.push(pii(0,s));
	for(int i=0;i<505;i++)
		dist[i]=100000009;
	dist[s]=0;
	while(!pq.empty())
	{
		pii k=pq.top();
		pq.pop();
		int d=k.first;
	//	cout<<d<<"..";
		int u=k.second;
		if(d==dist[u])
		{
	//		cout<<"lk";
			for(int i=0;i<A[u].size();i++)
			{
				if(dist[A[u][i].first]>dist[u]+A[u][i].second)
				{
					dist[A[u][i].first]=dist[u]+A[u][i].second;
					pq.push(pii(dist[u]+A[u][i].second,A[u][i].first));
				}
			}
		}
	}
}
int main()
{
    cin>>n;
    int u,v,w;
    A.clear();
    A.resize(505);
    while(n--)
    {
    	cin>>u>>v>>w;
    	A[u].push_back(pii(v,w));
    	A[v].push_back(pii(u,w));
    }
    
    int s;
    cin>>s;
    dj(s);
    int q;
    cin>>q;
    while(q--)
    {
    	cin>>u;
    	if(dist[u]!=100000009)
    		cout<<dist[u]<<endl;
    	else
    		cout<<"NO PATH\n";
    }
    return 0;
}
