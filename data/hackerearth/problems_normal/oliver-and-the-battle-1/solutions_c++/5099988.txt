#include <bits/stdc++.h>
#define maxN 1005
using namespace std;
int M[maxN][maxN];
int used[maxN][maxN];
int xx[8]={-1,-1,-1,0,1,1,1,0};
int yy[8]={-1,0,1,1,1,0,-1,-1};
int ans,bomb,t,n,a,b,m,fbomb;
void bfs(int i,int j)
{
	list< pair<int,int> >q;
	q.push_back(make_pair(i,j));
	ans++;
	bomb = 1;
	used[i][j] = 1;
	while(!q.empty()){
		int x = q.front().first;
		int y = q.front().second;
		q.pop_front();
		for(int i=0;i<8;i++)
		{
			int x1 = x+xx[i];
			int y1 = y+yy[i];
			if(x1>=0 && x1<n && y1>=0 && y1<m ){
		    if(used[x1][y1]==0 && M[x1][y1]==1){
				used[x1][y1] = 1;
				bomb++;
				q.push_back(make_pair(x1,y1));
			 }
			}
		}
	}
	if(fbomb < bomb) fbomb = bomb;
}
int main()
{
    cin>>t;
    while(t--){
               cin>>n>>m;
    for(int i=0;i<n;i++)
       for(int j=0;j<m;j++)
           cin>>M[i][j];
   ans = 0,fbomb=0;
   memset(used,0,sizeof(used));
   for(int i=0;i<n;i++)
       for(int j=0;j<m;j++)
          if(M[i][j]==1 && used[i][j]==0){
          	//cout<<i<<" "<<j<<endl;
          	bfs(i,j);
          }
    cout<<ans<<" "<<fbomb<<endl;
    }
    return 0;
}
