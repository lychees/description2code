#include <bits/stdc++.h>
using namespace std;

#define gc getchar_unlocked

void scanint(int &x)
{
    register int c = gc();
    x = 0;
    int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}

int main()
{
    int t,n,i,k,x,temp;
    scanint(t);
    while(t--)
    {
    	vector<int> v(200001);
    	scanint(n);
    	for(i=0;i<n;i++)
    	{
    		scanint(temp);
    	    v[temp+100000]=temp+100000;
    	}	
    	scanint(k);
    	while(k-- >0)
    	{
    	  scanint(x);
    	  if(v[x+100000]!=0)
    	  	printf("Yes\n");
    	  else
    	  	printf("No\n");
    	}
    }
    return 0;
}
