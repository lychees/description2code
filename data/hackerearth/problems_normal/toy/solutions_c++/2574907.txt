#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    vector<int> a;
    int dp[100005]={0};
    int x;
    for(int i=0;i<n;i++){
    	cin>>x;
    	a.push_back(x);
    }
    
    sort(a.begin(),a.end());
    int cnt=0;
    
    for(int i=0;i<n;i++){
    	if(dp[a[i]]==0){
	    	for(int j=0;j<=4;j++)
	    		dp[a[i]+j]=1;
	    	cnt++;
   	 	}    
   	 	
   	 //	cout<<a[i]<<" "<<cnt<<endl;
   	}
   	
    cout<<cnt;
    return 0;
}
