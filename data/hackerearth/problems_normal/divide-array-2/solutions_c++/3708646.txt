#include <iostream>
using namespace std;

int main()
{
	int n;
	cin>>n;
	int a[n], max_;
    for(int i=0; i<n; i++){
    	cin>>a[i];
    	if(i==0){
    		max_ = a[i];
    	}else{
    		max_ = max(max_, a[i]);
    	}
    }
    int q;
    cin>>q;
    long d = 1, x;
    for(int i=0; i<q; i++){
    	cin>>x;
    	if(max_>d)
    	d = d*x;
    	else if(max_>1)
    	d=max_+1;
    }
    //cout<<"d: "<<d<<endl;
    for(int i=0; i<n; i++){
    	cout<<(a[i]/d)<<" ";
    }
    cout<<endl;
    return 0;
}
