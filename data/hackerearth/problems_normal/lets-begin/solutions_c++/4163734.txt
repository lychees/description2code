#include <bits/stdc++.h>
using namespace std;
#define maxl 1000005
#define ll long long int
ll dp[maxl];

void precom(){
	ll i,j;
	memset(dp,10001,sizeof(dp));
	dp[2]=dp[3]=dp[5]=dp[7]=1;
	dp[4]=2;dp[6]=2;
	for(i=8;i<maxl;i++){
		dp[i]=min(dp[i-2],min(dp[i-3],min(dp[i-5],dp[i-7])))+1;
	}
}

int main()
{
	ll t,n;
	cin >> t;
	precom();
	while(t--){
		cin >> n;
		if(abs(dp[n]<10001))
		cout << dp[n] << endl;
		else cout << "-1" << endl;
	}
    return 0;
}
