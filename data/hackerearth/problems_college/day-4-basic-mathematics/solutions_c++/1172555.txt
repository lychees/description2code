#include <iostream>
using namespace std;
int findday(int);
int main()
{
    int t,n,i,result;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	result=findday(n);
    	if(result%3==0)
    		cout<<"SL"<<endl;
    	else if(result%3==1)
    		cout<<"LB"<<endl;
    	else
    		cout<<"BS"<<endl;
    }
    return 0;
}
int findday(int x)
{
	int i=1;
	int n=0;
	while(1)
	{
		if(i+n>=x)
			return i;
		n=n+i;
		i++;
	}
}
