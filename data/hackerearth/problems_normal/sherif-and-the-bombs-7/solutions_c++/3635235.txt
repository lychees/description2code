#include<iostream>
#define lli long long int
using namespace std;
int main()
{
    int n;
    cin>>n;
    lli a[1001][1001];
    for(lli i=0;i<1001;i++)
    {
        for(lli j=0;j<1001;j++)
            a[i][j]=0;
    }
    lli result=0;
    for(lli i=1;i<=n;i++)
    {
        lli x,y,r;
        cin>>x>>y>>r;
        for(lli j=1;j<1001;j++)
        {
            for(lli k=1;k<1001;k++)
            {
                lli dis=0;
                dis=(x-j)*(x-j)+(y-k)*(y-k);
                if(dis<=(r*r))
                    a[j][k]++;
            }
        }
    }
    for(lli i=0;i<1001;i++)
    {
        for(lli j=0;j<1001;j++)
        {
            if(a[i][j]>1)
                result++;
        }
    }
    cout<<result<<endl;
    return 0;
}
