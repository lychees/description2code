#include <iostream>

using namespace std;
 
int main()
{
    int t,i,n,p;
    cin>>t;
    while(t--)
    {
       cin>>n;
       int arr[n],flag=0;
       cin>>p;
       int a=0;
       for(i=1;i<=n;i++)
       {
       	cin>>arr[i];
       	if(arr[i]==p)
        flag=1;
       	if(arr[i]>p)
       	a++;
       }
       if(flag==1)
       cout<<"YES "<<a<<endl;
       else
       cout<<"NO 0"<<endl;
    }
    return 0;
}
