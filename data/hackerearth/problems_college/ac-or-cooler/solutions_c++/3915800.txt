#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    ll n,b,a[100005],c[100005];
    cin>>n>>b;
    for(int i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    for(int i=0;i<n;i++)
    {
    	cin>>c[i];
    }
    sort(a,a+n);
    sort(c,c+n);
    int cnt=0,i=0,j=0,ans=0;
    while(1)
    {
        ans = min(a[i],c[j]);
    	if(ans==a[i])i++;
    	else j++;
    	if(b-ans>0)
    	{
    		b-=ans;
    		cnt++;
    	}
    	else
    	break;
    }
    cout<<cnt<<endl;
    return 0;
}
