using namespace std;

inline int getint()
{
int num=0;
char c=getchar_unlocked();
while(!(c>='0' && c<='9'))
c=getchar_unlocked();
while(c>='0' && c<='9')
{
num=(num<<3)+(num<<1)+c-'0';
c=getchar_unlocked();
}
return num;
}

int main()
{
    unsigned long long int t,n,k;
    t = getint();
    while(t--)
    {
        n = getint();
        k = getint();
        if(n%(k+1)==1)
            printf("Nandu\n");
        else
            printf("Rani\n");
    }
}
