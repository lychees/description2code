#include <bits/stdc++.h>
using namespace std;
long long int fib[200];
void init()
{
	int i;
	for(i=2;i<200;i++)
	{
		fib[i]=fib[i-1]+fib[i-2];
	}
}
int main()
{
	fib[0]=1;
	fib[1]=2;
	init();
    int t;
    cin>>t;
    while(t--)
    {
    	long long int temp;
    	cin>>temp;
    	int i=-1;
    	long long int sum=0;
    	while(fib[i]<=temp)
    	{
    		if(fib[i]%2==0)
    			sum+=fib[i];
    		i++;
    	}
    	cout<<sum<<endl;
    }
}
