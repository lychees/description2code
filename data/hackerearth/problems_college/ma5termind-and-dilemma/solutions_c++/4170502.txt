#include<bits/stdc++.h>
using namespace std;

string a;

int main()
{
    int t;
    int n;

    cin>>t;

    while(t--){
        cin>>n;
        cin>>a;
        int ans = 0;
        int ans2 = 0;
		int cnt = 0;
      	for(int i=0;i<n;i++){
      		if(a[i]=='1')
      			cnt++;
      	}
      	
      	for(int i=0;i<cnt;i++){
      		if(a[i]=='0')
      			ans++;
      	}
      	
      	for(int i=0;i<(n-cnt);i++){
      		if(a[i]=='1')
      			ans2++;
      	}
      		
      	cout<<min(ans,ans2)<<endl;
    }
    return 0;
}
