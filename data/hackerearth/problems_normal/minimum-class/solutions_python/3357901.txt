def read_input():
    t = int(raw_input())
    cases = []
    for i in range(0,t):
        cases.append(int(raw_input()))
    return cases

def day_getter(classes):
    if classes == 1 or classes ==2:
        return classes
    elif classes < 0:
        return 0
    else:
        prev  = 1
        days = 0
        sum_all = 0
        while sum_all < classes:
            days += 1
            sum_all += prev
            if days % 2 == 0:
                prev+=1
        return days

for i in read_input():
    print day_getter(i)
