#include<bits/stdc++.h>
using namespace std;

set <int> node[1001];
bool vis[1001];
int n;

int bfs(int start)
{
    for(int i=0;i<=n;i++)
        vis[i]=false;

    int sum=0;
    stack <int> st;
    st.push(start);
    vis[start]=true;
    sum++;
    while(!st.empty())
    {
        int g=st.top();
        st.pop();
        for(set <int>::iterator it = node[g].begin();it!=node[g].end();it++)
        {
            if(vis[*it]==false)
            {
                sum++;
                st.push(*it);
                vis[*it]=true;
            }
        }
    }
    return sum;
}

int main()
{
    int d;
    cin>>n>>d;
    int x,y;
    for(int i=1;i<=d;i++)
    {
        cin>>x>>y;
        node[x].insert(y);
    }
    int mx = INT_MAX;
    for(int i=1;i<=n;i++)
    {
        int m = bfs(i);
        if(m<mx)
            mx=m;
    }
    cout<<mx;

    return 0;
}
