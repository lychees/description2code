#include <iostream>
#include<cstring>
using namespace std;
bool func(char *a,char *b)
{
	int m=strlen(a);
	int n=strlen(b);
	if(a[0]!=b[0]||a[m-1]!=b[m-1])
	return false;
	int alpha[26]={0};
	for(int i=0;i<m;i++)
	{
		alpha[a[i]-'a']++;
	}
	for(int i=0;i<n;i++)
	{
		alpha[b[i]-'a']--;
	}
	for(int i=0;i<26;i++)
	 if(alpha[i]!=0)
	 return false;
	 return true;
}
int main()
{
    char a[100][100];
    int n;
    cin>>n;
    char temp[100][100];
    for(int i=0;i<n;i++)
    cin>>a[i];
    int count=0;
    for(int i=0;i<n;i++)
    {
    	bool s=1;
    	for(int j=0;j<count;j++)
    	{
    		if(func(temp[j],a[i])==true)
    		{
    			s=0;
    			break;
    		}
    	}
    	if(s==1)
    	{
    		strcpy(temp[count++],a[i]);
    	}
    }
    printf("%i",count);
    return 0;
}
