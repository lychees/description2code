#include <iostream>
using namespace std;

int main()
{
    int t,k,a,b,bits;
    cin >> k >> t;
    long long p[k];
    for(int i = 0; i < k; i++){
    	cin >> p[i];
    }
    while(t--){
    	cin >> a >> b;
    	a--;
    	long long count = 0,product = 1;
    	for(int mask = 1; mask < 1<<k; mask++){
    		bits = 0;
    		product = 1;
    		for(int i = 0; i < k; i++){
    			if(1<<i & mask){
    				bits++;
    				product *= p[i];
    			}
    		}
    		if(bits%2 == 0){
    			count -= b/product - a/product;
    			continue;
    		}
    		count += b/product - a/product;
    	}
    	cout << count << endl;
    }
    return 0;
}
