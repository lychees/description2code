#include <iostream>
#include <math.h>

using namespace std;

int main()
{
	ios_base::sync_with_stdio(0);
	long long int testCases;
	cin>>testCases;
	for(long long int k=0;k<testCases;k++)
	{
		string a, b, c = "";
		cin>>a;
		cin>>b;
		for(long long int i=0;i<a.length();i++)
		{
			int diff = a[i] - b[i];
			if((int)pow(2, diff) % 2 == 0)
				c += '1';
			else
				c += '0';			
		}	
		cout<<c<<"\n";
	}
}
