#include<bits/stdc++.h>
using namespace std;
vector<int> adj[100001];
static bool v[100001];
void dfs(int curr){
    if(v[curr])
      return;
     v[curr]=true;
     for(int i=0;i<adj[curr].size();i++)
        dfs(adj[curr][i]);
}
int main(){
  int t;
  scanf("%d",&t);
  while(t--){
  	int n,e;
  	memset(v,false,sizeof(v));
  	scanf("%d %d",&n,&e);
  	while(e--){
  		int a,b;
  		scanf("%d %d",&a,&b);
  		adj[a].push_back(b);
  		adj[b].push_back(a);
  	}
  	int sum=0;
  	for(int i=0;i<n;i++){
  		if(v[i])
  		continue;
  		dfs(i);
  		sum++;
  	}
  	printf("%d\n",sum);
  	for(int i=0;i<n;i++)
  	adj[i].clear();
  }
  return 0;	
}
