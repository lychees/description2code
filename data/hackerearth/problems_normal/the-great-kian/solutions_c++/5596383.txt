#include <iostream>
using namespace std;

int main()
{
long long int n,i,array[100004],sum1=0,sum2=0,sum3=0;
cin>>n;
for(i=0;i<n;i++)
{
	cin>>array[i];
}

array[n]=0;
array[n+1]=0;
array[n+2]=0;

for(i=0;i<n;i=i+3)
{
	sum1 += array[i];
	sum2 += array[i+1];
	sum3 += array[i+2];
}
cout<<sum1<<" "<<sum2<<" "<<sum3;	
	
	
  return 0;
}
