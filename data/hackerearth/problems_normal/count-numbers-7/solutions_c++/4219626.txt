#include <iostream>
using namespace std;
typedef long long int ll;
int main()
{
    ll k, q;
    cin >> k >> q;
    ll i;
    ll a[k];
    for(i = 0; i < k; i++) {
    	cin >> a[i];
    }
    ll max = 1000000000;
    ll size = 1;
    size = size<<k;
    ll arr[size];
    ll c = 0;
    ll j;
    for(i = 1; i < size; i++) {
    	ll p = 1;
    	ll t = 0;
    	for(j = 0; j < k; j++) {
    		if(i&(1<<j)) {
    			t++;
    			if(p <= max) {
    				p = p * a[j];
    			}
    		}
    	}
    	if(t%2 == 0) {
    		t = -1;
    	}
    	else {
    		t = 1;
    	}
    	if(p <= max) {
    		arr[c++] = t * p;
    	}
    }
    while(q--) {
    	ll x, y;
    	cin >> x >> y;
    	x--;
    	ll ans = 0;
    	for(i = 0; i < c; i++) {
    		ans = ans + y/arr[i];
    		ans = ans - x/arr[i];
    	}
    	cout << ans << endl;
    }
    return 0;
}
