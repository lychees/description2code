#include <stdio.h>
int ar[15];
int ans=100000;
int n;

int min(int a,int b)
{
	return (a<b?a:b);
}

int max(int a,int b)
{
	return (a>b?a:b);
}

void sol(int a,int b,int c,int ind)
{
	if(ind>=n)
	{
		ans=min(ans,max(a,max(b,c)));
		return ;
	}
	sol(a+ar[ind],b,c,ind+1);
	sol(a,b+ar[ind],c,ind+1);
	sol(a,b,c+ar[ind],ind+1);
}

int main()
{
    int i;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    scanf("%d",&ar[i]);
    sol(0,0,0,0);
    printf("%d",ans);
    return 0;
}
