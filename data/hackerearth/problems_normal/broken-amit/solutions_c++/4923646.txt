using namespace std;
ll dp[1000000],l[20],d[20],n;
ll compute(ll in,ll mask)
{
    ll i,v=0,ans=-9999999999;
    if(in==n)
        return 0;
    if(dp[mask]!=-1)
        return dp[mask];
    for(i=0;i<n;i++)
    {
        if(!(mask&(1<<i)))
        {
            v= n*l[i] + i*d[i]*in*(in+1)/2-n*i*d[i]*in;
            ans = max(ans,v+compute(in+1,mask|(1<<i)));
        }
    }
    dp[mask] = ans;
    return dp[mask];

}
int main()
{
    ll i;
    ios::sync_with_stdio(false);
    memset(dp,-1,sizeof(dp));
    cin >> n;
    for(i=0;i<n;i++)
        cin >> l[i];
    for(i=0;i<n;i++)
        cin >> d[i];
    cout << compute(0,0) << endl;
}
