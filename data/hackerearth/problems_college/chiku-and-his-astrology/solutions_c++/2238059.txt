#include <iostream>
#include<stack>
using namespace std;
int check(string s)
{
int l=s.length();
stack<char>st;
for(int i=0;i<l;i++)
{
	if(st.empty())
	     st.push(s[i]);
	 else if((st.top())=='1'&&(s[i]=='0'))
           st.pop();
     else if((st.top())=='0'&&(s[i]=='1'))
          st.pop();
     else
          st.push(s[i]);
       //cout<<st.size()<<endl;
}
/*for(int i=0;i<st.size();i++)
{
	cout<<st.top();st.pop();
}*/
return st.size();//st.size();
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string s;
		cin>>s;
		int p=check(s);
		cout<<p<<endl;
	}
 //   cout << "Hello World!" << endl;
    return 0;
}
