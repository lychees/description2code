#include <iostream>
using namespace std;

int main()
{
    string s; cin >> s;
    int a[26]={0},i,flag=0,ans=0;
    for( i=0;i<s.length();i++)
    	a[s[i]-'a']++;
    
    for(i=0;i<26;i++)
    {
    	if(a[i]%2 && flag)
    	{
    		ans=1;	break;
    	}

    	if(a[i]%2 && flag==0)
    	{
    		flag=1;
    	}
    }
    
    cout << (ans?"NO":"YES");
    return 0;
}
