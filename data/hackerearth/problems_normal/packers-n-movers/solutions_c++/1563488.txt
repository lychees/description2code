#include <bits/stdc++.h>
using namespace std;

int main()
{
	int m,p;
	int arr[1010];
	int dp[20][1010];
	
	scanf("%d%d",&m,&p);
	for(int i=0;i<p;i++) {
		scanf("%d",&arr[i]);
	}
	
	dp[0][0] = arr[0];
	for (int i=1;i<p;i++) {
		dp[0][i] += dp[0][i-1] + arr[i];
	}
	
	for (int i=1;i<m;i++) {
		for (int j=0;j<p;j++) {
			dp[i][j] = INT_MAX;
			for (int k=0;k<=j;k++) {
				dp[i][j] = min(dp[i][j],max(dp[i-1][j-k],dp[0][j]-dp[0][j-k]));
			}
		}
	}
	
	printf("%d",dp[m-1][p-1]);
	
    return 0;
}
