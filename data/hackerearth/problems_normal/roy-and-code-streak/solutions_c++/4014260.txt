#include <iostream>
#include <bits/stdc++.h>
#include <unordered_map>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	
    	vector< pair<int, int> > v;
    	for(int i=0;i<n;i++)
    	{
    		int a,b;
    		cin>>a>>b;
    		v.push_back(make_pair(a,b));
    	}
    	unordered_map< int, int > m;
    	int ma=0,c=0;
    	for(int i=0;i<n;i++)
    	{
    		if(v[i].second==0)
    		c=0;
    		else
    		{
    			if(m[v[i].first]==0)
    			{
    				c++;
    				m[v[i].first]=1;
    			}
    		}
    		ma=max(ma,c);
    	}
    	cout<<ma<<endl;
    }
    return 0;
}
