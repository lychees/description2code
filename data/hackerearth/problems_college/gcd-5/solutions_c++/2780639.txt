#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    long long n,a[100000],i;
    int test,f;
    cin>>test;
    while(test--)
    {
    	f=1;
    	cin>>n;
    	for(i=0;i<n;i++)
    		cin>>a[i];
    	for(i=0;i<n-1;i++)
    	{
    		if(a[i]>a[i+1])
    		{
    			f=0;
    			break;
    		}
    	}
    	if(f==1)
    		cout<<"YES"<<endl;
    	else
    		cout<<"NO"<<endl;
    }
    return 0;
}
