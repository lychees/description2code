T = int(raw_input())
for iCase in xrange(T):
    N = int(raw_input())
    arrW = map(int, raw_input().split())
    arrH = sorted(map(int, raw_input().split()))
    best = arrW[0]*arrH[N-1]
    arrW = arrW[1:]
    arrW.sort()
    ok = True
    for i in xrange(N-1):
        if arrW[i]*arrH[N-1-i-1] >= best:
            ok = False
    print "YES" if ok else "NO"
