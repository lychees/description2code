#include <iostream>
#include<list>
using namespace std;

int main()
{
	int flag=0;
	list<int> l; 
   	long long int i,n,x,k,count=0;
   	cin>>n>>x;
   	for(i=0 ; i<n ; i++)
   	{
   		cin>>k;
   		l.push_back(k);
   	}
   	while(flag<=1)
   	{
   		if(l.front()<=x)
   		{
   			l.pop_front();
   			count++;
   			if(l.empty())
   			break;
   		}
   		else
   		{
   			l.pop_front();
   			flag++;
   			if(l.empty())
   			break;
   		}
   	}
   	cout<<count<<"\n";
    return 0;
}
