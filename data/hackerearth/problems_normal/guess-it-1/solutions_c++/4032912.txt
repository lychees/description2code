#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int sum=0;
		string s;
		cin>>s;
		for(int i=0;s[i]!='\0';i++)
		{
			if(s[i]!='_')
			sum+=(int)(s[i]-'0');
		}
		if(sum%9==0)
			cout<<"0 or 9"<<endl;
		else
		{
			for(int i=0;i<9;i++)
			{
				sum+=i;
				if(sum%9==0)
				{
					cout<<i<<endl;
					break;
				}
				sum-=i;	
			}
		}
	}
}
