#include <bits/stdc++.h>
using namespace std;

int main()
{ int t,n,m,count,temp,i,v,ans;
cin>>t;
while(t--)
 {  cin>>n>>m;
    count=1;v=0;
    int a[n+1];
    cin>>a[0];
    ans=a[0];
    for(i=1;i<n;i++)
      {cin>>a[i];
      if(a[i] > ans)
        count++;
      }
      temp=n-count;
     sort(a,a+n);
    for(i=temp;i<n;i+=m)
      {
      if(ans < a[i])
         v++;
      }
      cout<<v<<endl;
    
 }
    return 0;
}
