#include <iostream>
#define MOD 1000000007
typedef long long int LL;
using namespace std;

LL powmod(LL b,LL p)
{
	if(p==0)
		return 1;
	LL res=powmod(b,p/2);
	res=(res*res)%MOD;
	if(p%2==0)
	{
		return res;
	}
	else
	{
		return (res*b)%MOD;
	}
	
}


int main()
{
    LL t,n,ans;
    scanf("%lld",&t);
    while(t--)
    {
    	scanf("%lld",&n);
    	ans=powmod(2,n);
    	ans=(ans-1)%MOD;
    	printf("%lld\n",ans);
    }
    return 0;
}
