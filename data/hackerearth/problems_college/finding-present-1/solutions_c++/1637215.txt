#include <bits/stdc++.h>
#define ll long long int
using namespace std;
ll c[10001]={0};
ll d[10001]={0};
ll a[10000000];
ll b[10000000];
int main(){
     ll p,q,i,j;
     cin>>p;
     for(i=0;i<p;i++){
        cin>>a[i];
        c[a[i]]++;
     }
     cin>>q;
     for(i=0;i<q;i++){
        cin>>b[i];
        d[b[i]]++;
     }
     for(i=1;i<10001;i++){
        if (c[i]>d[i]) cout<<i<<" ";
        else if (c[i]<d[i]) cout<<i<<" ";
     }
     cout<<endl;
    return 0;
}
