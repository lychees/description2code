#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	std::cin>>t;
	while(t--){
		long long a,b,c;
		std::cin>>a>>b>>c;
		if(c==0){
			std::cout<<a<<endl;
		}
		else if(c==1){
			std::cout<<b<<endl;
		}
		else{
			long long sum=0;
			for(int i=0;i<c-1;i++){
				sum=a+b;
				a=b;
				b=sum;
			}
			std::cout<<sum<<endl;
		}
	}
}
