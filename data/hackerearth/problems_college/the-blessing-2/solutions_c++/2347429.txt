#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long int t,p,m,a[100000],i=0,s=0;
    cin>>t;
    while(t--)
    {	
    	
    	a[0]=1;
    	cin>>p>>m;
    	s=p;
    	for(i=1;i<3;i++)
    	{
    		a[i]=a[i-1]*2;
    		s=s+m;
    	}
    	cout<<s+p<<"\n";
    }
    return 0;
}
