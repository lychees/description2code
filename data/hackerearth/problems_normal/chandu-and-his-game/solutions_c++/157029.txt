#include <iostream>
using namespace std;


int main()
{
    int n,k;
    
    cin >>k>>n;
    
    int a[n],b[n];
    
    for(int i=0;i<n;i++ ){
    	cin>>a[i];
    	
    }
    
    
    for(int i=0;i<n;i++ ){
    	cin>>b[i];
    	
    }
    
    long long total=0;
    
    for(int i=0;i<n;i++ ){
    	int x = min(a[i]-1,k-a[i]);
    	int y = min(b[i]-1,k-b[i]);
    	
    	total+=x+y;
    	
    }
    cout<<total<<endl;
    
    return 0;
}
