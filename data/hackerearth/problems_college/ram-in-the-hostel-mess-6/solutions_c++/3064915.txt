#include <bits/stdc++.h>
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--){
		long long int n,arr[10000]={0};
		cin>>n;
		if(n==1)
		{
			cout<<"4"<<endl;
			continue;
		}
		arr[0]=0;
		arr[1]=2;
		arr[2]=3;
		for(int i=3;i<10000;i++)
			arr[i]=arr[i-1]+arr[i-2];
		cout<<arr[n]*arr[n]<<endl;

	}
	return 0;
}
