#include <bits/stdc++.h>
using namespace std;
pair <int,int> mp[20005];
int main()
{
    int n,x,y; cin >> n;
    for(int i=1;i<=n;i++)
    {
    	cin >> x >> y;
    	mp[i] = make_pair(x+y,i);
    }
    sort(mp+1,mp+n+1);
    for(int i=1;i<=n;i++)
    	cout << mp[i].second << " ";
    return 0;
}
