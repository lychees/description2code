#include <bits/stdc++.h>
using namespace std;


bool f(string a,string b)
{
		int vis[26];
		memset(vis,0,sizeof  vis);
		for(int i=0;i<a.length();i++)
    	{
    		vis[a[i]-'a']++;
    	}
    	
    	for(int i=0;i<b.length();i++)
    	{
    		if(vis[b[i]-'a']!=0)
    		return true;
    	}
    	return false;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string a;
    	string b;
    	cin>>a;
    	cin>>b;
    	if(f(a,b))
    	cout<<"Yes"<<endl;
    	else
    	cout<<"No"<<endl;
    }
}
