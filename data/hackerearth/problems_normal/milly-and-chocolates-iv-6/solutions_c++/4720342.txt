#include<bits/stdc++.h>
using namespace std;
void solve()
{
    long long n,k,m,minn=100000000000ll,ans=-1;
    long long sizes[1024];
    cin>>n>>k>>m;
    for(int i=0;i<n;i++)
        cin>>sizes[i];
    for(int i=0;i<n;i++)
    {
        long long teksum=0;
        for(int j=0;j<sizes[i];j++)
        {
            long long now;
            cin>>now;
            teksum+=now;
        }
        teksum+=m*(sizes[i]-1)+k*sizes[i];
        if(teksum<minn)
        {
            minn=teksum;
            ans=i+1;
        }
    }
    cout<<ans<<" "<<minn<<endl;
}
int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
        solve();
    return 0;
}
