#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	int a[n];
    	for(int i=0;i<n;i++){
    		cin>>a[i];
    	}
    	int v[n];
    	for(int i=0;i<n;i++)
    		v[i]=0;
    	int flag=1;
    	int i=a[0];
    	v[0]=1;
    	while(flag!=0){
    		if(i==-1){
    			cout<<"Win\n";
    			flag=0;
    			break;
    		}
    		else if(v[i]==0){
    			v[i]=1;
    		}
    		else if(v[i]==1){
    			cout<<"Lose\n";
    			flag=0;
    			break;
    		}
    		i=a[i];
    	}
    }
    return 0;
}
