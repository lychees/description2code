#include <bits/stdc++.h>

#define REP(i,n) for(int i=0;i<(int)n;i++)
#define FOR(i,a,b) for(int i=(int)a;i<=(int)b;i++)
#define pb push_back
#define mp make_pair

using namespace std;

typedef long long int ll;

int main()
{
	int t;
	scanf("%d",&t);
	while(t--){
		ll a,b,c,d,k;
		scanf("%lld%lld%lld%lld%lld",&a,&b,&c,&d,&k);
		
		ll root = (ll)ceil(cbrt((double)k*1.0/a));
		ll ans = 0;
		
		//printf("root : %lld, root: %lf, k/a : %lf\n",root,cbrt((double)k*1.0/a),(double)k*1.0/a);
		REP(t,root+1){
			ll ft = a*t*t*t + b*t*t + c*t + d;
			if(ft<=k) ans = t;
			else break;
		}
		printf("%lld\n",ans);
	}
    return 0;
}
