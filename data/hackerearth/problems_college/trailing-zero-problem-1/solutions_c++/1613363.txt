#include<iostream>
#include<cmath>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        long n,i,a,sum1=0;
        double sum=0,k;
        scanf("%ld",&n);
        for(i=n;i>=1;i--)
        {
            k=log10(i);
          //  cout<<k<<' ';
            sum+=k;
        }
        a=ceil(sum);
        //printf("\n%d ",a);
      while(n)
        {
            sum1+=n/5;
            n/=5;
        }
        printf("%ld\n",a-sum1);
    }
    return 0;
}
