#include<bits/stdc++.h>
using namespace std;
long long power(long long a, long long b)
{
    long long x=1,y=a;
    while(b>0)
    {
     if(b&1) x=x*y;
     y= y*y;
     b>>=1;
    }
    return x;
}
int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
        long long int n;
        cin>>n;
        long long sum =0;
        sum = sum + 1 + 2*n;
        long long x = power(2,1+log2(n));
        sum = sum - x;
        cout<<sum<<endl;
    }
}
