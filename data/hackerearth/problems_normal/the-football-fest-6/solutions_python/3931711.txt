cases = int(raw_input())

for _ in xrange(cases):
	in_data = map(int, raw_input().split())
	passes = in_data[0]
	p1, p2 = 0, in_data[1]
	
	for _ in xrange(passes):
		entry = raw_input().split()
		if entry[0] == 'P':
			p1 = p2
			p2 = int(entry[1])
			
		elif entry[0] == 'B':
			p1, p2 = p2, p1

	print 'Player', p2
