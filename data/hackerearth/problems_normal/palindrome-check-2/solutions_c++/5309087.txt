#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    string s,s1;
    getline(cin,s);
    int i=0,l=s.size();
    for(int j=l;j>=0;)
        s1[--j]=s[i++];
    for(i=0;i<l;i++)
        {
        	if(s[i]!=s1[i])
                {
        	   cout<<"NO";
        	    break;
                }
        }
        if(i==l)
         cout<<"YES";
    return 0;
}
