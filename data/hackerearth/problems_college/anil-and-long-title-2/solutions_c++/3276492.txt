#include <bits/stdc++.h>
#define ll long long
#define pb push_back
#define mk make_pair
#define pi pair<int, int>
#define fi first
#define se second
using namespace std;

int main() {
	string s;
	cin >> s;
	int n = s.length(), ch = 0;
	for(int i = 0; i < n; i++) {
		if(s[i] != '_' && (i == 0 || ch == 1)) {
			s[i] = s[i] - 'a' + 65;
			if(ch == 1) ch = 0;
		}
		if(s[i] == '_') ch = 1;
	}
	cout << s;
	return 0;
}
