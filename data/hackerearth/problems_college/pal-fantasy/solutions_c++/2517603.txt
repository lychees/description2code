bool is_palindrome(string a) {
//	cout << " fun " << a << endl;
	int i = 0;
	int j = a.length() - 1;

	while (i < j) {
		if (a[i] == a[j]) {
			i++;
			j--;
		} else {
			return false;
		}
	}
	return true;
}
