#include<bits/stdc++.h>
using namespace std;
#define maxn 51
#define F(i,p,n) for(i=p;i<n;i++)
#define Ss(x) scanf("%s",x)
#define S(x) scanf("%d",&x)
#define P(x) printf("%d\n",x)
#define Ps(x) printf("%d ",x)
#define LL long long int
int a[maxn],b[maxn];
int func(int k/*,int c*/)
{
    int i,j,m,n,s=0,ans,d[maxn][2550],diff;
    memset(d,0,sizeof(d));
    F(i,1,k)
    s+=b[i];
    m=(s+1)>>1;
    /*printf("in case %d:\n\n",c);
    printf("arr D ini\n");
    F(i,0,k)
    {
        F(j,0,m+1)
        printf("%d ",d[i][j]);
        printf("\n");
    }
    printf("\n");

    F(i,1,k)
    printf("b[%d]=%d ",i,b[i]);
    printf("\n");*/

    F(i,1,k)
    {
        F(j,0,m+1)
        {
            if(j<b[i])
                d[i][j]=d[i-1][j];
            else
            {
                if(d[i-1][j-b[i]]+b[i]>d[i-1][j])
                    d[i][j]=d[i-1][j-b[i]]+b[i];
                else
                    d[i][j]=d[i-1][j];
            }
        }
    }
    /*printf("arr D final\n");
    F(i,0,k)
    {
        F(j,0,m+1)
        printf("%d ",d[i][j]);
        printf("\n");
    }*/
    diff=s-2*d[k-1][m];
    if(diff<0)
    diff*=-1;
    //printf("s=%d m=%d value=%d diff=%d\n",s,m,d[k-1][m],diff);
    return diff;
}
main()
{
    int i,j,k,m,n,t,minn,num;
    S(t);
    while(t--)
    {
        minn=10000;
        S(n);
        F(i,0,n)
        S(a[i]);
        F(i,0,n)
        {
            k=1;
            F(j,0,n)
            if(j!=i)
                b[k++]=a[j];
            m=func(k/*,i+1*/);
            if(m<minn)
            {
                minn=m;
                num=a[i];
                //P(minn);
                //P(num);
            }
            else if(m==minn)
            {
                if(num>a[i])
                    num=a[i];
            }
        }
        Ps(minn);
        P(num);
    }
}
