from math import sqrt
def isprime(n):
    if(n==2 or n==3 or n==1):
        return True
    if(n%2==0):
        return False
    else:
        for y in xrange(3,int(sqrt(n))+1,2):
            if(n%y==0):
                return False
        return True

n=input()
for x in xrange(0,n):
    p=input()
    q=sqrt(p)
    flag=0
    if(q.is_integer()):
        flag=1
    if(isprime(p) or (flag==1 and isprime(q))):
        print("NO")
    else:
        print("YES")
