#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);

    long long n,mod,ans;
    while(t--)
    {
        scanf("%lld%lld",&n,&mod);
        if(mod%2==0)
        {
            if(n%mod<(mod/2))
            {
                ans = (n/mod)*mod;
            }
            else
                ans = (n/mod+1)*mod;
        }
        else
        {
            if(n%mod<=(mod/2))
                ans = (n/mod)*mod;
            else
                ans = (n/mod+1)*mod;
        }
        printf("%lld\n",ans);
    }
}
