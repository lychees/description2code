def isPrime(num):
    for i in xrange(2, int(num ** 0.5) + 1):
        if num % i == 0:
            return False
    return True


N = int(raw_input())
for i in range(N):
    X = int(raw_input())
    if isPrime(X):
        print 'NO'
    elif int(X ** 0.5) == (X ** 0.5) and isPrime(X ** 0.5):
        print 'NO'
    else:
        print 'YES'
