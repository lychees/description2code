#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
using namespace std;
bool pal(char s[],int a,int b)
{
    int i,j;
    for(i=a,j=b;i<j;i++,j--)
    {
        if(s[i]!=s[j])
            return false;
    }
    return true;
}
int main()
{
    int t,len,i,j,maxi,check;
    char str[21];
    scanf("%d",&t);
    while(t--)
    {
        maxi=0;
        scanf("%s",str);
        len=strlen(str);
        for(i=0;i<len;i++)
        {
            for(j=i+1;j<len;j++)
            {
                if(pal(str,i,j))
                {
                   check=abs(j-i)+1;
                   if(check>maxi)
                   {
                        maxi=check;
                    }
                }
            }
        }
        printf("%d\n",maxi);
    }
    return 0;
}
