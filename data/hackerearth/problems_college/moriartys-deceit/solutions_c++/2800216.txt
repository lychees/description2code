#include<iostream>
#include<algorithm>
#include<assert.h>
#include<vector>
using namespace std;

int main()
{
 int t,n,x;
 cin>>t;
 assert(1<=t && t<=10);
 while(t--)
 {
  cin>>n>>x;
  assert(1<=n && n<=10000);
  assert(2<=x && x<=10000000);
  vector<int> arr(n);
  vector<int> ans;
  for(int i=0;i<n;i++){
    cin>>arr[i];
    assert(2<=arr[i] && arr[i]<=10000000);
    int p=std::__gcd(arr[i],x);
  	if(p==1) ans.push_back(arr[i]);
  }  
  if(ans.size()==0) cout<<"-1\n";
  else{
  for(int i=0;i<ans.size();i++)
   cout<<ans[i]<<" ";
  }
  if(t>0) cout<<"\n";
 }
 return 0;
}
