#include <iostream>
#include <algorithm>
#include <locale>
using namespace std;

int main()
{
	locale loc;
    int t;
    cin>>t;
    if(t<1||t>10)cout<<"Invalid Test"<<endl;
    else
    {
    	while(t--)
    {
    	string s;
    	cin>>s;
    	bool flag=true;
    	int i=0,l=0,u=0;
    	if(s.size()>100)
    	{
    		cout<<"Invalid Input"<<endl;
    		continue;
    	}
    	while(i<s.size())
    	{
    		if(isalpha(s[i]))
    		{
    			if(islower(s[i]))l++;
    			else u++;
    			flag=false;
    		}
    		i++;
    	}
    	if(flag) cout<<"Invalid Input"<<endl;
    	else
    	{
    		cout<<min(u,l)<<endl;
    	}
    }
    }
    return 0;
}
