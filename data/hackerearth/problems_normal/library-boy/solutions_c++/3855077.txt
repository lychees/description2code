#include <iostream>
using namespace std;

int main()
{ int n;
  cin>>n;
  char book[n][20];
  int count[26]={0};
  int i;
  for(i=0;i<n;i++)
  { cin>>book[i];
    count[book[i][0]-97]++;
  }
  int shelves=0;
  for(i=0;i<26;i++)
  { shelves+=(count[i]/10);
    if(count[i]%10!=0)
     shelves++;
  }
  
  cout<<shelves;
  
  return 0;
}
