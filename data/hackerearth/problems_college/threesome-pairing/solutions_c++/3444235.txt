#include<iostream>
using namespace std;
int main(){
	int t;
	cin>>t;
	int n;
	int count;
	while(t--){
		cin>>n;
		int trop[n];
		count= 0;
		for(int i=0;i<n;i++){
			cin>>trop[i];
		}
		for(int i=0;i<n;i++){
			for(int j=i+1;j<n;j++){
				if(((trop[i]+trop[j])%3)==0){
					count++;
					
				}
			}
		}
		cout<<count<<endl;
	}
}
