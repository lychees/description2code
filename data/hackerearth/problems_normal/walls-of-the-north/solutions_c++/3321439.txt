#include <stdio.h>
#include <algorithm>
#define INF 0x3f3f3f3f
using namespace std;
int t, n;
int A[100010];
int f(bool inc){
	int ans = 1;
	int x = A[0];
	for(int i=1; i<n; i++){
		if((inc && A[i] >= x) || (!inc && A[i] <= x)){ ans++; x = A[i]; inc = !inc; }
		else if(inc) x = min(x, A[i]);
		else x = max(x, A[i]);
	}
	return ans;
}
int main(){
	scanf("%d", &t);
	while(t--){
		scanf("%d", &n);
		for(int i=0; i<n; i++) scanf("%d", &A[i]);
		printf("%d\n", max(f(true), f(false)));
	}
	return 0;
}
