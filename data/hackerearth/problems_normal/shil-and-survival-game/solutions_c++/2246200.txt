#include <iostream>
#include<vector>
using namespace std;

int main()
{
    int n,i,temp,max;
    vector<int> a,b;
    	scanf("%d",&n);
    for(i=0;i<n;i++)
    {
    	scanf("%d",&temp);	
    
    	a.push_back(temp);
    }
    b.push_back(1);
   	
    max=a[0];
    for(i=1;i<n-1;i++)
    {
    	if(a[i]>max)
    	{
    		b.push_back(1);
    		max=a[i];
    	}
    	else
    		b.push_back(0);
    }
    b.push_back(1);
    	
    	 max=a[n-1];
    for(i=n-2;i>0;i--)
    {
    	if(a[i]>max)
    	{
    		b[i]=1;
    		max=a[i];
    	}
    }
    for(i=0;i<n;i++)
    {
    	if(b[i]==1)
    		printf("%d ",i+1);
    }
    
    
  
    return 0;
}
