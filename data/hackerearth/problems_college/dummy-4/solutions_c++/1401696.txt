#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
typedef long long ll;
#define sz 1000004


int main(){
	int n,sum=0,i,j,k,p;
	char a[1002];
	cin>>n;
	
	for(i=0;i<n;i++){
		cin>>a;
		for(j=0;a[j];j++)
			sum+=a[j]-'0';
	}
	while(sum>9){
		k=sum;
		p=0;
		while(k>0){
			p+=k%10;
			k=k/10;
		}
		sum=p;
	}
	cout<<sum<<endl;
	return 0;
}
