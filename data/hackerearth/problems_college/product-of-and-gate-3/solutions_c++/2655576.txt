#include <iostream>
#include <stdint.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long int A,B;
    	cin>>A>>B;
    	if(A==0)
    	{
    		cout<<0<<endl;
    		continue;
    	}
       	long int res = A;
    	for(long int i = A+1;i<=B;i++)
    		res = res&i;
    	cout<<res<<endl;
    }
}
