#include<iostream>
#include<cstdio>
#include<cmath>
#include<cstring>
#include<algorithm>
#include<list>
#include<vector>
#include<cstring>
#include<queue>
#include<set>
#define mod 1000000007
#define ll long long
using namespace std;


string arr;

int main()
{
	int t,i,n,j,count=0,count1=0;
	cin>>t;
	while(t--)
	{
		cin>>n;
		cin>>arr;
		
		i=0;
		j=n-1;
		count=0;
		while(1)
		{
			if(i>=j)
			break;
			
			if(arr[i]=='0')
			i++;
			else
			if(arr[j]=='1')
			j--;
			else
			if(arr[i]=='1' && arr[j]=='0')
			{
				count++;
				i++;
				j--;
			}
			
		}
		i=0;
		j=n-1;
		count1=0;
		while(1)
		{
			if(i>=j)
			break;
			
			if(arr[i]=='1')
			i++;
			else
			if(arr[j]=='0')
			j--;
			else
			if(arr[i]=='0' && arr[j]=='1')
			{
				count1++;
				i++;
				j--;
			}
			
		}
		cout<<min(count,count1)<<"\n";
	}
	return 0;
}
