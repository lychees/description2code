#include <bits/stdc++.h>
#define ll long long
using namespace std;
char x[12],s[12];
ll int l;
ll int myupper()
{
	ll int i,j;
	bool flag=0;
	for(i=0;i<l;i++)
	{
		if(((s[i]-'0')&1))//odd
		{
			if(s[i]=='9')
			{
				for(j=i-1;j>=0;j--)
				{
					if((((s[j]-'0')+2)%10)!=0)
					{
						x[j]='0'+(((s[j]-'0')+2)%10);
						i=j+1;
						flag=1;
						break;
					}
					//~ cout<<" "<<(((s[j]-'0')+2)%10)<<" "<<j<<endl;
				}
				if(flag)
				break;
			}
			else
			{
				x[i]=('0'+(((s[i]-'0')+1)%10));
				i++;
				break;
			}
		}
		else
		x[i]=s[i];
	}
	for(;i<l;i++)
	x[i]='0';
	x[i]='\0';
	//~ cout<<x<<endl;
	ll int t2;
	sscanf(x,"%lld",&t2);
	return t2;
}
ll int mylower()
{
	ll int i;
	for(i=0;i<l;i++)
	{
		if(((s[i]-'0')&1))//odd
		{
			x[i]=('0'+(((s[i]-'0')-1)%10));
			i++;
			break;
		}
		else
		x[i]=s[i];
	}
	for(;i<l;i++)
	x[i]='8';
	x[i]='\0';
	//~ cout<<x<<endl;
	ll int t2;
	sscanf(x,"%lld",&t2);
	return t2;
}
bool mycheck()
{
	bool flag=1;
	ll int i;
	for(i=0;i<l;i++)
	{
		if(((s[i]-'0')&1))//odd
		{
			flag=0;
			break;
		}
	}
	return flag;
}
int main()
{
	ll int test;
	scanf("%lld",&test);
	while(test--)
	{
		ll int i,x,y,z,g,num,den;
		scanf("%s",s);
		sscanf(s,"%lld",&z);
		l=strlen(s);
		for(i=l;i>0;i--)
		s[i]=s[i-1];
		s[l+1]='\0';
		s[0]='0';
		l++;
		if(mycheck())
		{
			printf("Unlimited Power\n");
		}
		else
		{
			x=myupper();
			y=mylower();
			//~ cout<<x<<" "<<y<<endl;
			num=(x-z);
			den=(z-y);
			g=__gcd(num,den);
			num/=g;den/=g;
			printf("%lld/%lld\n",num,den);
			
		}
	}
	return 0;
}
