#include <iostream>
#include<cmath>
using namespace std;

int main() {
int t;
cin>>t;
while(t--)
{
long long n,a,b,z;
cin>>n;
double x=sqrt(n);
long y=(long)x;
if(x-y==0)
cout<<"YES"<<endl;
else
{
    a=y*y;
    b=(y+1)*(y+1);
    if(abs(a-n)<abs(b-n))
    z=a-n;
    else
    z=b-n;
if(z<0)
cout<<z<<endl;
else
cout<<"+"<<z<<endl;
}
}
	return 0;
}
