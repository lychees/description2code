#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,n;
	scanf("%d",&n);
	int ar[n+10];
	ar[0]=0;
	for(i=1;i<=n;i++)
	{
		scanf("%d",&ar[i]);
		ar[i]+=ar[i-1];
	}
	int q,l,r;
	scanf("%d",&q);
	while(q--)
	{
		scanf("%d%d",&l,&r);
		printf("%d\n",ar[r]-ar[l-1]);

	}
	
	return 0;
}
