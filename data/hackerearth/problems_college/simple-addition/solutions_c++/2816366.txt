#include <iostream>
#include <algorithm>
using namespace std;

int sol( int n, int i )
{
	int x = i ;
	int y = n-i ;
	if( __gcd(x,y) > 1 )
		return n ;
	int count = 0 ;
	while( x>0 && y>0 )
	{
		if(x<y)  y-=x ;
		else  x-=y ;
		count++;
	}
	if( abs(x-y) == 1 )
		return count;
	return n;
}

int main()
{
	int t,n,ans,i;
	for( cin>>t; t--; )
	{
		cin >> n ;
		if(n==1)  
		{
			cout << "0\n" ; 
			continue ;  
		}
		ans = n-1 ;
		if(n%2)  ans = n/2+1 ;
		for( i=3; i<=n/2; i++ )
		{
			int x = sol(n,i);
			if( ans > x )
				ans = x ;
		}
		cout << ans << endl ;
	}
	
	return 0;
}
