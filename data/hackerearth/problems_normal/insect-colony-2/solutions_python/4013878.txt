n = int(raw_input())
for i in range(n):
	n_i = [int(x) for x in raw_input().split()]
	x = 1
	for i in range(1, len(n_i)):
		if (n_i[i] & 1) == 1:
			x = x^1
	if x == 1:
		print "Yes"
	else:
		print "No"
