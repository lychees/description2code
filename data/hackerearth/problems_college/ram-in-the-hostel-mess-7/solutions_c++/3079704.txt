#include <iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	int a[t];
	int max=-1;
	for(int i=0;i<t;i++)
	{
		cin>>a[i];
		if(a[i]>max)
		max=a[i];
	}
	int b[max];
	b[0]=2;
	b[1]=3;
	for(int i=2;i<max;i++)
	{
		b[i]=b[i-1]+b[i-2];
	}
	for(int i=0;i<t;i++)
	{
		cout<<b[a[i]-1]*b[a[i]-1]<<endl;
	}
	return 0;
}
