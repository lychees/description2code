#include <iostream>
using namespace std;

int main()
{
    int t;
    long long int n,points,tmp,res;
    cin>>t;
    
    while(t--){
    	int arr[26];
    	res=0;
    	cin>>n;
    	points=(2*n)*6;	
    	for(int i=0;i<26;i++){
    		cin>>arr[i];
    	}
    	
    	if(points<27){
    		for(int i=0;i<points;i++){
    			res+=arr[i];
    		}
    		cout<<res<<endl;
    	}
    	else{
    		for(int i=0;i<26;i++)
    			res+=arr[i];
    			
    	    long long int p=points/26;
    	   res=res*p;
    	    res+=((p*(p-1))/2)*26;
    	    
    	    int m=points%26;
    	    
    	    for(int i=0;i<m;i++)
    	      res+=arr[i]+p;
    	      
    	    cout<<res<<endl;    		
    		
    	}
    }
    
    return 0;
}
