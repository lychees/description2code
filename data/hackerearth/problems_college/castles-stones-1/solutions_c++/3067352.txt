#include <bits/stdc++.h>
using namespace std;

int main() {
    int t,m,n,temp,count,t1,t2;
    cin>>t;
    while(t--)
    {
        count=0;
        cin>>m;
        cin>>n;
        vector<int>v;
        for(int i=0;i<n;i++)
        {
            cin>>temp;
            v.push_back(temp);
        }
        sort(v.begin(),v.end());
        int i=0;
        int j=v.size()-1;
        while (v[j]>m)
        	j--;
        while (i<=j)
        {
	        if((v[i]+v[j])<=m)
	        {
	            i++;
	            j--;
	        }
	        else
	        	j--;
	        
	        count++;
	        
        }
        cout<<count<<"\n";
    }
	return 0;
}
