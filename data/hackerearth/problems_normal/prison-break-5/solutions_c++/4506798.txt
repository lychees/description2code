typedef long long ll;
ll MOD=1000000007;
ll pw(ll a,ll b){
  ll r=1;
  while(b>0){
     if(b&1){
       r=(r*a)%MOD;
        }
     b=b>>1;
     a=(a*a)%MOD;
  }
return (r)%MOD;
}
int n;
int g[22][22];
int vis[22][22];
ll ans=0;

void dfs(int r,int c){
	//cout<<"with "<<r<<" "<<c;pln;
   if(r<1||c<1||r>n||c>n)return;
   if(vis[r][c])return;
   if(!(r==1&&c==1))if(g[r][c])return;
   vis[r][c]=1;
   if(r==n&&c==n){
      ans=(ans+1)%MOD;
     vis[r][c]=0;
    return;
   }
  dfs(r+1,c);
  dfs(r-1,c);
  dfs(r,c+1);
  dfs(r,c-1);
  vis[r][c]=0;
  return ;
}


int main(){
  int t;
  si(t);
  while(t--){
    si(n);
    fr(i,1,n+1){
       fr(j,1,n+1){
          si(g[i][j]);
          vis[i][j]=0;
       }
    }
    ans=0;
    dfs(1,1);
    pll(ans);pln;
  }

return 0;
}
