#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--) {
        long long i,j,n,m,temp,x,y,o=-1,e=-1;
        bool ans=true;
        cin>>n;
        for(i=1;i<=n;i++) {
            cin>>x>>y>>temp;
            x+=y;
            if(x%2==1) {
                if(o==-1 && e==-1) o=temp%2;
                else if(o==-1 && e!=-1 && e==temp%2) ans=false;
                else if(o!=-1 && temp%2!=o) ans=false;
            } else if(x%2==0) {
                if(e==-1 && o==-1) e=temp%2;
                else if(e==-1 && o!=-1 && o==temp%2) ans=false;
                else if(e!=-1 && e!=temp%2) ans=false;
            }
        }
        if(ans) cout<<"Yes";
        else cout<<"No";
        cout<<"\n";
    }
    return 0;
}
