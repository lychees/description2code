#include <iostream>
#include<string>
#include<algorithm>
using namespace std;

int main()
{
    int t, ff,ss;
    string str;
    string p = ("www.");
    cin>>t;
    while(t--)
    {
        cin>>str;
        string k = str.substr(0,4);
        ss = str.size();
        if(k == p)
            str.erase(0,4);
        str.erase(std::remove(str.begin(), str.end(), 'a'), str.end());
        str.erase(std::remove(str.begin(), str.end(), 'e'), str.end());
        str.erase(std::remove(str.begin(), str.end(), 'i'), str.end());
        str.erase(std::remove(str.begin(), str.end(), 'o'), str.end());
        str.erase(std::remove(str.begin(), str.end(), 'u'), str.end());
        ff = str.size() +1;
        cout<<ff<<"/"<<ss<<endl;
    }
    return 0;
}
