int p[1000001];
int a[1000001];
int sieve(){
	int i,j,k=0;
	a[0]=1;a[1]=1;
	for(j=4;j<1000001;j+=2){
		a[j]=1;
	}
	for(i=3;i<sqrt(1000001);i+=2){
		if(a[i]==0){
			for(j=i*i;j<1000001;j+=2*i){
				a[j]=1;
			}
		}
	}
	p[0]=2;
	j=1;
	for(i=3,j=1;i<(1000001);i+=2){
		if(a[i]==0){
			p[j++]=i;
		}
    }
    return j;
}

{
