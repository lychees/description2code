 #define Max(a,b) ((a)>(b)?(a):(b))

typedef long long y; 
int ts(int w, int h) { 
    return (w * h + 1) / 2; 
    } 
   
  int totalstones(int width, int height) { 
    
    return ts(width/2, height/2) + 
      ts((width+1)/2, (height)/2) + 
      ts((width+1)/2, (height+1)/2) + 
      ts((width+0)/2, (height+1)/2); 
    } 
    
    
int main()
{
    int T;
    cin>>T;
     while(T--)
    {
        int w,h;
        cin>>w>>h;
        y res = totalstones(w,h);
        cout<<res<<endl;
    }
   
    
    
    return 0;
}
