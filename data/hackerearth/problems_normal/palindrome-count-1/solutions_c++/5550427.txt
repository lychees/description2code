#include <bits/stdc++.h>
using namespace std;

int main()
{
		string s; cin >> s;
		int l = s.length();
		bool dp[l][l];
		
		for(int i=0;i<l;i++) 
		{
			dp[i][i] = true;
		}
		int L, i, j;
		long long int sum = l;
		for(L = 2; L <= l; L++)
		{
			for(i=0;i<l-L+1;i++)
			{
				j = L+i-1;
				if(L==2)
				{
					if(s[i] == s[j]) dp[i][j] = true;
					else dp[i][j] = false;
				}
				else
				{
					if(s[i] == s[j]) dp[i][j] = dp[i+1][j-1];
					else dp[i][j] = false;
				}
				if(dp[i][j]) sum ++;
			}
		}
		cout << sum << endl;
}
