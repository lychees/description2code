#include <iostream>
#include <iomanip>
using namespace std;

int main()
{
	int T;
	cin>>T;
	while(T--){
		int N;
		cin>>N;
		double sum=0.0;
		for(int i=0; i<N; i++)
		{
			double P = 0.0;
			cin>>P;
			sum += P;
		}
		double avg = sum/N;
		cout<<fixed<<setprecision(8)<<avg<<endl;
	}
    return 0;
}
