#include<iostream>
#include<cmath>
#include<cstring>
#include<string>
#include<algorithm>
#include<vector>
#include<map>
#include<stack>
#include<queue>
#define MIN(a,b) ((a)<(b)?(a):(b))
#define MAX(a,b) ((a)>(b)?(a):(b))

typedef long long ll;
typedef long l;

using namespace std;
const double pi = acos(-1.0);

int main()
{

    int t;
    scanf("%d",&t);
    while(t--)
    {
              ll k;
              scanf("%lld",&k);
              ll sum=0;
              int i=1;
              ll l_num=0,digit=0,size=0;
              while(1)
              {
                      sum+=(ll)pow(2.0,i)*i;
                      if(sum==k)
                      {
                              l_num= (ll)pow(2.0,i)-1;
                              size=i;
                              digit=i;
                              break;
                      }
                      if(sum>k)
                      {
                               sum-=(ll)pow(2.0,i)*i;
                               ll residue = k-sum;
                               if(residue%i==0)
                               {
                                       l_num= residue/i-1;
                                       digit=i;
                               }
                               else
                               {
                                       l_num= residue/i;
                                       digit=residue%i;
                               }  
                               size=i;
                               break;
                      } 
                      i+=1;
              }
              
              int p=0;
              vector<int>v;
              while(l_num>0)
              {
                            v.push_back(l_num%2);
                            p++;
                            l_num/=2;
              }
              int y=size-p;
              string s="";
              for(int i=1;i<=y;i++)
                 s+="0";
              for(int j=v.size()-1;j>=0;j--)
                    s.push_back(v[j]+'0');
              
              if(s[digit-1]=='0')
                 printf("Hacker\n");
              else
                 printf("Earth\n");
    }
    
    
    return 0;
}
