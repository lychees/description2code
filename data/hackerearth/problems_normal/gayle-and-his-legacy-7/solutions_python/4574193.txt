for _ in range(input()):
    arr=[]
    n,m=map(int,raw_input().split())
    for i in range(n):
        #[gi,pi]
        pi=map(int,raw_input().split())
        arr.append(pi[::-1])
    arr=sorted(arr)
    power=m
    i=0
    #print arr
    while i<n:
        if power>=arr[i][0]:
            power+=arr[i][1]
        else:
            print"NO"
            break
        #print power
        i+=1
    else:
        print "YES"
