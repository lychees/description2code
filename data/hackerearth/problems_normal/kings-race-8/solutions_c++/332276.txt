#include<cstdio>
#include<algorithm>
using namespace std;
const int MAX=1000005;
int str[MAX],h[MAX],mymax[MAX];


int main()
{
    int T;
    scanf(" %d",&T);
    while(T--){
        int N,K;
        scanf(" %d %d",&N,&K);
        int i;
        for(i=0;i<N;i++) scanf(" %d",&str[i]);
        for(i=0;i<K;i++) scanf(" %d",&h[i]);
        int ans=-1;
        int id=0;
        for(i=0;i<K;i++){
            mymax[i]=max(h[i],i?mymax[i-1]:0);
        }
        for(i=0;i<N;i++){
            int dist=distance(mymax,upper_bound(mymax,mymax+K,str[i]));
            if(dist > ans){
                ans=dist;
                id=i;
            }
        }
        printf("%d\n",id);
    }
    return 0;
}
