#include <bits/stdc++.h>
using namespace std;

int a[1000000],countp[1000001],countn[1000001];
int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	memset(countp,0,sizeof(countp));
    	memset(countn,0,sizeof(countn));
    	for(int i=0;i<n;i++){
    		cin>>a[i];
    		if(a[i]>0) countp[a[i]]++;
    		else countn[-a[i]]++;
    	}
    	int count=n;
    	for(int i=1;i<=1000000;i++){
    		if(countp[i]>1) count+=((countp[i]*(countp[i]-1))/2);
    		if(countn[i]>1) count+=((countn[i]*(countn[i]-1))/2);
    	}
    	cout<<count<<endl;
    }
    return 0;
}
