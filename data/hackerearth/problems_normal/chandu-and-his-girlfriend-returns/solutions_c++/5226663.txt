#include <iostream>
#include<bits/stdc++.h>
using namespace std;
long long int a[100001];
int main()
{	int t;
long long int i,n1,n2;
cin>>t;
		while(t--){
			cin>>n1;
			cin>>n2;
			for(i=0;i<n1+n2;i++){
				cin>>a[i];
			}
			sort(a,a+n1+n2);
			for(i=n1+n2-1;i>=0;i--){
				cout<<a[i]<<" ";
			}
			cout<<endl;
		}
    
}
