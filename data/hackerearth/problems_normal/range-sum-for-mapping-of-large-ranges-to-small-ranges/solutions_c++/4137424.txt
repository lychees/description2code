#include <iostream>
#include <bits/stdc++.h>
#define ll long long int
using namespace std;
vector<int>adj[100001];
bool vis[100001];
ll dfs(int s)
{    ll t=1;
	vis[s]=true;
	for(int i=0;i<adj[s].size();i++)
	{
		if(vis[adj[s][i]]==false){ t+=dfs(adj[s][i]);}

	}
	return t;

}
void initialize() {
    for(int i = 0;i < 100001;++i)
     vis[i] = false;
}
int main()
{    
	int T;
    cin>>T;
    int x,y,n,k;
    while(T--)
    {    
    	cin>>n>>k;
    	for(int i=0;i<=n;i++)
	{
		adj[i].clear();
		vis[i]=false;
	}
      for(int i=0;i<k;i++)
      {
      	cin>>x>>y;
      	adj[x].push_back(y);
      	adj[y].push_back(x);
      }
      initialize();
	
       ll ans=1;
      for(int i=1;i<=n;i++)
      {
        ll t=dfs(i);
      	if(t>=2)
      	{
      		t=t*(t-1)/2;
      		ans=(ans*t)%1000000007; 
      	}
     
      }
      cout<<ans<<endl;

    }

    return 0;
}
