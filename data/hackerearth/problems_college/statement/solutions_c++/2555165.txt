#include <bits/stdc++.h>
#define ll long long
using namespace std;
int a[1000010],dp[1000010],h[1000010];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int test;
	cin>>test;
	while(test--)
	{
		memset(h,0,sizeof(h));
		memset(dp,0,sizeof(dp));
		int n;
		cin>>n;
		int i,j,ans=1;
		for(i=0;i<n;i++)
		{
			cin>>a[i];
			h[a[i]]++;
			dp[a[i]]=h[a[i]];
			ans=max(ans,dp[a[i]]);
		}
		sort(a,a+n);
		for(i=1;i<1000010;i++)
		{
			if(h[i])
			{
				for(j=1;j*j<=i;j++)
				{
					if(j!=i && (i%j)==0 && h[j])
					{
						dp[i]=max(dp[i],h[i]-1+dp[j]+1);
						ans=max(ans,dp[i]);
					}
					if(i/j!=i && (i%j)==0 && h[i/j])
					{
						dp[i]=max(dp[i],h[i]-1+dp[i/j]+1);
						ans=max(ans,dp[i]);
					}
				}
			}
		}
		cout<<ans<<endl;
	}
	
	return 0;
}
