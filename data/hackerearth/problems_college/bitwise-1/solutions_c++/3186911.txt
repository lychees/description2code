#include <bits/stdc++.h>
 
#define fi first
#define se second
#define pb push_back
#define mp make_pair
#define pii pair<int,int>
#define pll pair<ll,ll>
#define MOD 1000000007
#define ll long long int
#define MAXINT 2000000000
#define SET(x,y) memset(x,y,sizeof(x));
 
using namespace std;

vector<int> zeroes;

void solve(void) {
	ll x,k,q,z,i,ans=0,j,flag;
	cin>>x>>k;
	zeroes.clear();
	z = 0;
	for(i=0;i<=60;i++) {
		q=1ll<<i;
		if(q>x)	break;
		if(x&q)	;
		else	z++,zeroes.pb(i);
	}
	j=i;
	//k--;
	ans = 0;
	flag=0;
	for(i=0;i<=40;i++) {
		q=1ll<<i;
		if(q>k)	break;
		if(i>=z)	{flag=1;break;}
		if(k&q) {
			ans |= (1ll<<zeroes[i]);
		}
	}
	if(flag) {
		k>>=i;
		//cout<<k<<" "<<j<<" ";
		ans+=(k<<j);
	}
	cout<<ans<<endl;
}

int main () {
	int t,i;
	cin>>t;
	for(i=1;i<=t;i++) {
		cout<<"Case #"<<i<<": ";
		solve();
	}
	return 0;
}
