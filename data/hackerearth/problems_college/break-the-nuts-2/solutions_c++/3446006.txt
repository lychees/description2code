{

     scan(tem);

        b[k++]=i;
  }
  if(k==0)
  {

      return 0;
  }
  for(i=1,ans=1;i<k;i++)
  {
      ans*= (b[i]-b[i-1]);
  }
  cout << ans;
  return 0;
}
