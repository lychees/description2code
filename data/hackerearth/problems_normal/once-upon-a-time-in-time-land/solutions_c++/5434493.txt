#include<bits/stdc++.h>
using namespace std;

long long int max(long long int a, long long int b)
{
	if(a>b)
	return a;
	return b;
}
int main()
{
int T;
cin >> T;
while(T--)
{
	int N,K;
    vector<long long int> A;
    vector<long long int> DP;
	cin >> N >> K;
	for(int i = 0; i<N;i++)
	{
		int x;
		cin >> x;
		A.push_back(x);
	}
   int i = 1;
	DP.push_back(max(0,A[0]));
	while(i<N)
	   {
	   	  if(i-K-1 >=0)
		   DP.push_back(max(DP[i-1],DP[i-K-1]+A[i]));
		  else
		   DP.push_back(max(DP[i-1],A[i]));
		   
		i++; 
		}
   printf("%lld\n",DP[N-1]);
	
}
}
