#define MOD 1000000007
long long gcd(long long x,long long y)
{
	if(y==0)
		return x;
	return gcd(y,x%y);
}
int main(void){
	std::ios::sync_with_stdio(false);cin.tie(0);
	long long a,b,c,t;
	long long re1,re2;
	cin>>a>>b>>c;
	re2=a*b*2;
	if(a>b)
	{
		t=a;
		a=b;
		b=t;
	}
	if(c>=a+b)
		re1=a*b*2;
	else if(c<=a)
	{
		re1=c*c;
	}
	else if(c<=b)
	{
		re1=a*a+a*(c-a)*2;
	}
	else
	{
		re1=re2-(a+b-c)*(a+b-c);
	}
	//cout<<re1<<endl;
	long long g=gcd(re1,re2);
	cout<<re1/g<<'/'<<re2/g<<endl;
	//system("pause");
	return 0;
}
