int main()
{
#define int long long
	int t; cin >> t;
	while (t--)
	{
		string a; cin >> a;
		int n = a.size(), ret = 0;
		for (int i = 0; i < n; i++)
			if (a[i] == '-')
			{
				if (i + 1 < n and a[i + 1] == 'B') continue;
				if (i + 2 < n and a[i + 2] == 'B') continue;
				if (i + 1 < n and a[i + 1] == 'S') continue;
				if (i - 1 >= 0 and a[i - 1] == 'S') continue;
				ret++;
			}
		cout << ret << "\n";
	}
	return 0;
}
