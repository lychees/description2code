#include "bits/stdc++.h"
using namespace std;
#define ll long long
#define s(n) scanf("%d", &n);
#define sll(n) scanf("%lld", &n);
#define p(n) printf("%d\n", n)
#define pll(n) printf("%lld\n", n)
#define MOD 1000000007

int dp[1000005];
int main()
{
	int n,t,i,j,m1,m2; 
	s(t);
	while(t--)
	{
		memset(dp,0,sizeof dp);
		s(n)s(m1)s(m2);
		for(i=n; i>=1; i--)
		{
			if(i+m1 <= n+1)
			{
				if(dp[i+m1] == 0)
					dp[i] = 1;
			}
			if(i+m2 <= n+1)
			{
				if(dp[i] == 0)
					if(dp[i+m2] == 0)
						dp[i] = 1;
			}
		}
		printf("%d\n", dp[1]);
	}
	return 0;
}
