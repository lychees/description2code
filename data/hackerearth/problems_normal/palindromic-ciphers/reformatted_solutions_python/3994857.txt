test = int(raw_input())


def palindrome(s):
    if len(s) == 0 or len(s) == 1:
        return True
    return palindrome(s[1:-1]) and s[0] == s[-1]


scr = {}
summ = 1
for i in 'abcdefghijklmnopqrstuvwxyz':
    scr[i] = summ
    summ += 1

for t in range(test):
    strg = raw_input()
    if palindrome(strg):
        print "Palindrome"
    else:
        score = 1
        for let in strg:
            score = score*scr[let]
        print score
