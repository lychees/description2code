#include<iostream>
#include<math.h>
using namespace std;
inline int isrand(long int n)
{
	if((n%2==0&&n!=2)||n==1)
		return 0;
	long long int sq=sqrt(n);
	 long long int i=3;
	while(i<=sq)
	{
		if(!(n%i))
			return 0;
		i=i+2;	
	}
	return 1;
}
int main()
{
	long long int a,b,i;
	long long int sum=0;
	int t;
	cin>>t;
	while(t--)
	{
		sum=0;
		cin>>a>>b;
		i=a;
		while(i<=b)
		{
			if(isrand(i))
			 sum+=i;
			i++; 
		}
		cout<<sum<<"\n";
	}
	return 0;
}
