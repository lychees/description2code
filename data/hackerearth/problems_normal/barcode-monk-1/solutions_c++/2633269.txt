typedef unsigned long long int ull;
typedef long long int ll;
typedef vector<vector<ll> > mat;

ll m=1e9+7,dp[1005][2]={0},w[1005]={0};

int main()
{ios_base::sync_with_stdio(false);
int n,k,a;
cin>>n>>k;
w[0]=1;
rep(i,0,k-1)
{
  cin>>a,w[a]=1;
  
  for(int j=a;j<=n;j++)
  w[j]|=w[j-a];
}


dp[0][0]=1;dp[0][1]=1;
dp[1][0]=1;
w[0]=1;
if(w[1]==1)
dp[1][1]=1;
rep(i,2,n)
{
	dp[i][0]=(dp[i-1][0]+dp[i-1][1])%m;
	rep(j,1,1000)
	{
		if(w[j]==1)
		{
			int l=i-j;
			if(l>=0)
			dp[i][1]=(dp[i][1]+dp[l][0])%m;
		}
	}
//	o2(i,dp[i][1]+dp[i][0]);on;
}
ol((dp[n][0]+dp[n][1])%m);

}
