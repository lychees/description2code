#include<stdio.h>
using namespace std;
typedef long long int ll;

const int kallu=100001;
int arr[kallu],l[kallu],r[kallu];
int main(){
	int king,love,i,sex,sex1;
	while(~scanf("%d%d",&king,&love)){
		for(i=1;i<=king;i++)
		scanf("%d",arr+i);
		l[1]=1,r[king]=king;
		for(i=2;i<=king;i++)
		      l[i]=(arr[i]<=arr[i-1])?l[i-1]:i;
		for(i=king-1;i>0;i--)
		      r[i]=(arr[i]<=arr[i+1])?r[i+1]:i;
		while(love--)
		{
			scanf("%d%d",&sex,&sex1);
			puts(r[sex]>=l[sex1]?"Yes":"No");
		}
	}
	return 0;
}
