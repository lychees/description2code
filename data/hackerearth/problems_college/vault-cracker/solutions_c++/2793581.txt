#include <iostream>
using namespace std;

int main()
{
	string a,b;
	int i,j,ans[10001];
	for (i=0;i<10001;i++) ans[i]=0;
    cin>>a>>b;
    for (i=0;i<b.length();i++)
    {
    	for (j=0;j<a.length();j++)
    	{
    		if (a[j]==b[i])
    		{
    			if (j==0) ans[j]+=1;
    			else ans[j]+=ans[j-1];
    		}
    	}
    }
    cout<<ans[a.length()-1]<<endl;
    return 0;
}
