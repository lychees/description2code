#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin >> t;
    while(t--){
    	int n;
    	cin >> n;
    	vector<int> v(n);
    	for(int i=0;i<n;i++) cin >> v[i];
    	sort(v.begin(),v.end());
    	long long int ans = 1;
    	for(int i=0;i<n;i++){
    		if(v[i]-i>0) ans*=v[i]-i;
    		else ans=0;
    		ans = ans%1000000007;
    	}
    	cout << ans%1000000007 << endl;
    }
    return 0;
}
