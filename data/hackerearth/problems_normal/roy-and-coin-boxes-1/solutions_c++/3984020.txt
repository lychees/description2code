#include <bits/stdc++.h>
using namespace std;
struct ${$(){ios_base::sync_with_stdio(false);cin.tie(NULL);}}$;

const int Maxn = 1000002;

int a[Maxn];
int n, m, l, r, q, ans;

int binary_search(int x, int st, int end) {
   while(st + 1 < end) {
      int mid = (st + end) / 2;
      if(a[mid] >= x)
         end = mid;
      else
         st = mid;
   }
   if(a[st] >= x)
      return st;
   else if(a[end] >= x)
      return end;
   else
      return n + 1;
}


int main() {
   cin >> n >> m;  
   for(int i = 0; i < m; i++) {
      cin >> l >> r;
      a[l]++;
      a[r + 1]--;
   }
   for(int i = 1; i <= n; i++) {
      a[i] = a[i - 1] + a[i];
   }
   sort(a + 1, a + n + 1);
   int x;
   cin >> q;
   while(q--) {
      cin >> x;
      ans = binary_search(x, 1, n);
      cout << (n - ans + 1) << endl;
   }
   return 0;
}
