#include <bits/stdc++.h>
using namespace std;
int a[102][102];
int dp[102][102];
int f(int m,int n)
{
	int i,j,k;
	
	if(n<0 || m<0) return 0;
	if(dp[m][n]!=-1) return dp[m][n];
	if(n==0 && m==0) {dp[m][n]=a[m][n];return a[m][n];} 
	k=a[m][n]+max(f(m-1,n),f(m,n-1));
	dp[m][n]=k;
	return k;
}
int main()
{
    //cout << "Hello World!" << endl;
    int i,j,k,m,x,y,t,c,n;
    cin>>t;
    while(t--)
    {
    	cin>>n>>c;
    	
    	for(i=0;i<n;i++)
    	{
    		for(j=0;j<n;j++)
    		{
    		scanf("%d",&a[i][j]);dp[i][j]=-1;
    		}
    	}
    	x=f(n-1,n-1);
    	if(x>=c){
    		cout<<x-c<<endl;
    	}
    	else cout<<"-1"<<endl;
    }
    return 0;
}
