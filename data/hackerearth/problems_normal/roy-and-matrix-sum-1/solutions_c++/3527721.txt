#include<cstdio>

int main(){
    int t;
    scanf("%d", &t);
    while(t--){
        int n;
        scanf("%d", &n);
        long long ans = 0;
        long long temp = 2;
        for(int i=n-1;i>=1;i--){
            ans += i*temp;
            temp+=2;
        }
        printf("%lld\n", ans);
    }
}
