t = int(raw_input())


def unique(s):
    return len(set(s)) == len(s)


while (t > 0):
    lst = []
    cnt = 0
    s = raw_input()
    s = ''.join(sorted(s))
    s = ''.join(list(set(s)))
    l = len(s)
    for i in range(l):
        lst.extend([s[i:l]])
    for i in lst:
        if unique(i):
            cnt = cnt+1
    print cnt
    t = t-1
