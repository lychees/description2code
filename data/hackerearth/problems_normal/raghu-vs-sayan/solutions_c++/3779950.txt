#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main(){
	int t;
	cin>>t;
	while(t--){
		ll a,b;int n;
		cin>>a>>b>>n;
		vector<ll> v;
		ll x;
		for(int i=0;i<n;i++){
			cin>>x;
			v.push_back(x);
		}
		sort(v.begin(),v.end());
		ll total=0;
		int a1=0;
		int a2=0;
		for(int i=0;i<n;i++){
			total+=v[i];
			if(total<=a){
				a1++;
			}
			if(total<=b){
				a2++;
			}
			if(total>a || total>b)break;
		}
		if(a1>a2){
			cout<<"Raghu Won"<<endl;
		}else if(a2>a1){
			cout<<"Sayan Won"<<endl;
		}else{
			cout<<"Tie"<<endl;
		}
	}
	return 0;
}
