#include <bits/stdc++.h>
using namespace std;


char s[50005];
int c[50005][30], curr[30];

int main()
{
	int n, q, l, r, k, i, j, d;
	char ans;
	memset(c, 0, sizeof(c));
	cin>>n;
	cin>>q;
	cin>>s;
	for(i = 0; s[i]!='\0'; i++){
		d = s[i] - 'a';
		for(j = 0; j<26; j++){
			c[i+1][j] = c[i][j];
		}
		c[i+1][d]++;
	}
	for(i = 0; i<q; i++){
		cin>>l>>r>>k;
		//cout<<l<<r<<k<<endl;
		if(k - 1>r - l){
			cout<<"Out of range"<<endl;
			continue;
		}
		for(j = 0; j<26; j++){
			curr[j] = c[r][j] - c[l-1][j]; 
		}
		j = 0;
		while(k>curr[j]){
			k -= curr[j];
			j++;
		}
		ans = 'a' + j;
		cout<<ans<<endl;
		
	}
    return 0;
}
