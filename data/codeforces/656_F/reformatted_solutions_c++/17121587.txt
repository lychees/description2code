#include <algorithm>
#include <bitset>
#include <cstdio>
#include <iostream>
#include <map>
#include <math.h>
#include <queue>
#include <stdio.h>
#include <string.h>
using namespace std;
int main() {
  char n;
  int ans = 0;
  for (int i = 0; i < 7; i++) {
    scanf("%c", &n);
    if (n == 'A') {
      ans += 1;
    } else if (n == '0') {
      ans += 9;
    } else {
      ans += n - '0';
    }
  }
  printf("%d\n", ans);
  return 0;
}
