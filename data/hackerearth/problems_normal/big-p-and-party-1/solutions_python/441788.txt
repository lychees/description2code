import sys
vals = raw_input().split(" ")
A, B = map(lambda x: int(x),vals)
graph, distTo, marked =[],[],[]
for i in range(A):
	graph.append([])
	marked.append(False)
	distTo.append(-1)
	
distTo[0] = 0

while B > 0:
	x, y = map(lambda x: int(x),raw_input().split(" "))
	graph[x].append(y)
	graph[y].append(x)
	B -=1
 
 
#BFS
from collections import deque

queue = deque([])
queue.append(0)
while queue:
	s = queue.popleft();
	marked[s]= True
	for w in graph[s]:
		if marked[w]: continue
		queue.append(w)
		marked[w] = True
		distTo[w] = distTo[s]+1

for i in range(1,len(distTo)):
	print distTo[i]
