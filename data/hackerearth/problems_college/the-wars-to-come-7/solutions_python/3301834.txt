count = int(raw_input())

strenghts = raw_input().strip().split(" ")
istrenghts = []

requiredStrength = int(raw_input())


for s in strenghts:
	istrenghts.append(int(s))

istrenghts = sorted(istrenghts, reverse=True)

start = 0
count = 0
su = 0

while(start < len(istrenghts) and su < requiredStrength):
	su += istrenghts[start]
	count += 1
	start += 1
