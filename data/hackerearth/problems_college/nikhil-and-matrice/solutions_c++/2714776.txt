#include<stdio.h>


int a[11][11],i, j, k, num, choice, c, p, q, r, s;

int right(){
    for(j = 0; j < num; j++){
        if(a[p][j] == 0){
            a[p][j] = c;
            c++;
        }
    }
    p++;
    return c;
}


int down(){
    for(i = 0; i < num; i++){
        if(a[i][q] == 0){
            a[i][q] = c;
            c++;
        }
    }
    q--;
    return c;
}


int left(){
    for(j = num - 1; j >= 0; j--){
        if(a[r][j] == 0){
            a[r][j] = c;
            c++;
        }
    }
    r--;
    return c;
}


int up(){
    for(i = num - 1; i >= 0; i--){
        if(a[i][s] == 0){
            a[i][s] = c;
            c++;
        }
    }
    s++;
    return c;
}


int main(){
    int t;
    scanf("%d ",&t);
    while(t--)
    {

     i, j, k, num, choice, c = 1, p = 0, q, r, s = 0;

    scanf(" %d",&num);
    for(i=0;i<11;i++)
    {
    	for(j=0;j<11;j++)
    	{
    		a[i][j]=0;
		}
	}
    choice = num * num;
    q = num - 1;
    r = num - 1;
    while(c <= choice){
        c = right();
        if(c > choice)
            break;
        c = down();
        if(c > choice)
            break;
        c = left();
        if(c > choice)
            break;
        c = up();
        if(c > choice)
            break;
    }
    
    for(i=0;i<num;i++){
        for(j=0;j<num;j++){
            printf("%d",a[i][j]);
        }
        printf("\n");
    }
    printf("\n");
}
return 0;
}
