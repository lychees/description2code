inline int inp()
{
   int n=0;
   int ch=getcx();int sign=1;
   while( ch < '0' || ch > '9' ){if(ch=='-')sign=-1; ch=getcx();}

   while(  ch >= '0' && ch <= '9' )
           n = (n<<3)+(n<<1) + ch-'0', ch=getcx();
   return n*sign;
}
inline long long in()
{
   long long n=0;
   long long ch=getcx();long long sign=1;
   while( ch < '0' || ch > '9' ){if(ch=='-')sign=-1; ch=getcx();}

   while(  ch >= '0' && ch <= '9' )
           n = (n<<3)+(n<<1) + ch-'0', ch=getcx();

void mul(l b[2][2],l a[2][2])
{
    l c[2][2];
    for(int i=0;i<2;++i)
    {
        for(int j=0;j<2;++j)
            c[i][j]=b[i][j];
    }
    b[0][0]=((((c[0][0]%mod)*(a[0][0]%mod))%mod)+(((c[0][1]%mod)*(a[1][0]%mod))%mod))%mod;
    b[0][1]=((((c[0][0]%mod)*(a[0][1]%mod))%mod)+(((c[0][1]%mod)*(a[1][1]%mod))%mod))%mod;
    b[1][0]=((((c[1][0]%mod)*(a[0][0]%mod))%mod)+(((c[1][1]%mod)*(a[1][0]%mod))%mod))%mod;
    b[1][1]=((((c[1][0]%mod)*(a[0][1]%mod))%mod)+(((c[1][1]%mod)*(a[1][1]%mod))%mod))%mod;

}
l term(l n)
{
    l a[2][2],b[2][2],d[2][2];
        a[0][0]=1;
        a[0][1]=1;
        a[1][0]=1;
        a[1][1]=0;
        b[0][0]=1;
        b[0][1]=0;
        b[1][0]=0;
        b[1][1]=1;
        while(n)
        {
            if(n%2)
                mul(b,a);
            n/=2;
            for(int i=0;i<2;++i)
                for(int j=0;j<2;++j)
                    d[i][j]=a[i][j];
            mul(a,d);
        }
        return b[0][1];
}
