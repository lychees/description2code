const ll Max=10e17;

ll visited[100010];

vector<ll> G[100010];

vector<ll> wealth;
ll ans=Max;
ll W[100010];

void dfs(ll idx)
{
	ll i,n=G[idx].size();
	visited[idx]=1;
	ans=min(ans,W[idx]);
	for(i=0;i<n;i++) if(visited[G[idx][i]]==0) dfs(G[idx][i]);

}

int main()
{   
    ios::sync_with_stdio(false);
    ll T;
    cin>>T;
    while(T--)
    {
    	MEM(visited,0);
    	MEM(W,0);
    	wealth.clear();
    	for(int i=0;i<100010;i++) G[i].clear();
    	ll n,m,i,j,k;
    	cin>>n>>m;
    	for(i=1;i<=n;i++) cin>>W[i];
    	for(i=0;i<m;i++)
    	{
    		cin>>j>>k;
    		G[j].pb(k);
    		G[k].pb(j);
    	}
    	for(i=1;i<=n;i++)
    	{
    		if(visited[i]==0)
    		{
    			ans=Max;
    			dfs(i);
    			wealth.pb(ans);
    		}
    	}
    	sort(wealth.begin(),wealth.end());
    	m=wealth.size();
    	ll sum=0;
    	for(i=1;i<m;i++) sum+=wealth[i];
    	cout<<(wealth[0]*sum)<<endl;
    }

    return 0;
}
