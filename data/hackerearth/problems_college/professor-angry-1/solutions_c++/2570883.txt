#include <iostream>
using namespace std;
#include<vector>
#define ll long long
#define ull unsigned long long
#define pb push_back
int main()
{
    ll test,i,j,N,K;
    
    cin>>test;
    
    while(test--)
    {
    	ll num;
    	cin>>N>>K;
    	vector<ll> v;
    	ll count=0;
    	for(i=0;i<N;i++)
    	{
    		cin>>num;
    		v.pb(num);
    		if(v[i]<=0){count++;}
    	}
    	if(count<K){cout<<"YES\n";}
    	else {cout<<"NO\n";}
    }
    return 0;
}
