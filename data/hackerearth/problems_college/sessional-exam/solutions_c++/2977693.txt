#include<bits/stdc++.h>
using namespace std;
int main(){


    int n , m;
    cin >> n >> m;
    string x[n+1];
  for(int i = 1 ; i <= n ; ++i )
   cin >> x[i];
   int hsh[n+1];
   memset( hsh , 0 , sizeof hsh );

    for(int i = 0 ; i < m ; ++i ){
      int mx = 0;
     for(int j = 1 ; j <= n ; ++j ){
         mx = max( mx , x[j][i]-'0');
     }
     for(int j = 1 ; j <= n ; ++j ){
       if(x[j][i]-'0'==mx)
        hsh[j]=1;
     }

    }
    int ans = 0;
    for(int i = 1 ; i <= n ; ++i )
     ans = ans + hsh[i];
     cout << ans << endl;


}
