#include <iostream>
#include <string.h>
#include <math.h>
using namespace std;

int main() {
int t;
cin>>t;
while(t--)
{
	int n1,n2,count=0,c=0;
	cin>>n1>>n2;
	for(int i=n1;i<=n2;i++)
	{
		int s=sqrt(i);
		
	
		c=0;
		if(i==1)
		c=1;
			for(int j=2;j<=s;j++)
			{
				if(i%j==0)
				{
					c=1;break;
				}
			}
			if(c==0)
			count++;
		
	}
	cout<<count<<endl;
}
	return 0;
}
