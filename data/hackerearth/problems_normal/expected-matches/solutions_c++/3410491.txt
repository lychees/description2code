const int N = 1e3+5;
double dp[N][N];
bool found[N][N];
double p[N][N];
int n,m;
double rec(int i,int j) {
  if(i==n+1 || j==m+1)
    return 0.0;
  if(found[i][j])
    return dp[i][j];
  found[i][j] = true;
  double ans = 0.00;
  ans+=1+p[i][j]*rec(i,j+1) + (1-p[i][j])*(rec(i+1,j));
  return dp[i][j] = ans;
}
int main() {
  int t;
  sci(t);
  while(t--) {
    M(found,0);
    sci(n);sci(m);
    F(i,1,n+1) {
      F(j,1,m+1) {
        scanf("%lf",&p[i][j]);
      }
    }
    printf("%.6lf\n",rec(1,1));
  }
}
