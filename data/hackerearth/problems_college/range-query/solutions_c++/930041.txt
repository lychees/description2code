#include <iostream>
#include <stdio.h>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
	
	int n, i, q, x, y, val, v, count;

	scanf("%d", &n );

	vector<int> A( n );
	vector< vector<int> > B( 101, vector<int> () );
	vector<bool> C( 101, false );
	
	for( i = 0;i < n;i++ ) {
		scanf("%d", &v );
		A[ i ] = v;
		B[ v ].push_back( i );
		C[ v ] = true;
	}	

	scanf("%d", &q );

	while( q-- ) {

		scanf("%d %d %d", &x, &y, &val );
		
		if( C[ val ] ) {

			vector<int> :: iterator it;
			count = 0;

			for( it = B[ val ].begin();it != B[ val ].end();it++ ) 
				if( ( ( *it ) >= x ) && ( ( *it ) <= y ) )
					count++;

			printf("%d\n", count );
		}
		else 
			printf("0\n");
	}

	return 0;
}
