#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    string a;
    scanf("%d",&t);
    while(t--) {
    	cin >> a;
    	int c1,c2;
    	c1 = c2 = 0;
    	for(int i = 0; i < a.size()-1; i++) {
    		if(a[i] == a[i+1]) {
    			c1++;
    		}
    	}
    	for(int i = a.size()-1; i>0;i--) {
    		if(a[i] == a[i-1]) {
    			c2++;
    		}
    	}
    	printf("%d\n",min(c1,c2));
    }
    return 0;
}
