#include<bits/stdc++.h>
using namespace std;
long long int a[100005],b[100005];
int main()
{
    long long int n,i,sum;
    cin>>n;
    for(i=0;i<n;i++)
    {
      cin>>a[i];
        b[i]=1;
    }
    for(i=1;i<n;i++)
    {
        if(a[i]>a[i-1])
            b[i]=b[i-1]+1;

    }
    for(i=n-2;i>=0;i--)
    {
        if(a[i]>a[i+1]&&b[i]<=b[i+1])
            b[i]=b[i+1]+1;
    }
    sum=0;
    for(i=0;i<n;i++)
    {
        sum+=b[i];
    }
    cout<<sum<<endl;
    return 0;
}
