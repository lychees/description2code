#include <bits/stdc++.h>
using namespace std;
long long n,i,a[1000000],q,top,ans,ans2,l,r;
  map<long long,long long> mp;
       	
long long first(long long arr[], long long low, long long high, long long x, long long n)
{
  if(high >= low)
  {
    long long mid = (low + high)/2;  /*low + (high - low)/2;*/
    if( ( mid == 0 || x > arr[mid-1]) && arr[mid] == x)
      return mid;
    else if(x > arr[mid])
      return first(arr, (mid + 1), high, x, n);
    else
      return first(arr, low, (mid -1), x, n);
  }
  return -1;
}
long long last(long long arr[], long long low, long long high, long long x, long long n)
{
  if(high >= low)
  {
    long long mid = (low + high)/2;  /*low + (high - low)/2;*/
    if( ( mid == n-1 || x < arr[mid+1]) && arr[mid] == x )
      return mid;
    else if(x < arr[mid])
      return last(arr, low, (mid -1), x, n);
    else
      return last(arr, (mid + 1), high, x, n);      
  }
  return -1;
}
int main(){
 
	cin>>n;
	for(i=0;i<n;i++){
		cin>>a[i];
		mp[a[i]]=1;
	}
	sort(a,a+n);
	cin>>q;
	for(i=0;i<q;i++){
		cin>>l>>r;
		if(r>a[n-1])r=a[n-1];
		if(l<a[0])l=a[0];
		while(mp[l]!=1&&l<r){
			l++;
		}
		while(mp[r]!=1&&r>l){
			r--;
		}
		
		ans=first(a,0,n-1,l,n);
		ans2=last(a,0,n-1,r,n);
		
		top=ans2-ans+1;
		if(ans==-1||ans2==-1)
		top=0;
		cout<<top<<endl;
	}
	
}
