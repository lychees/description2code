#include<iostream>
#include<cstdio>
using namespace std;

int main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);

    int t;
    cin>>t;

    while(t--)
    {
        double arr[55][10001];
        int n,m,k;
        cin>>m>>n>>k;

        for(int i=1;i<=50;i++)
            arr[i][1]=0.0000;

        if(n<=k)
        {
            for(int i=1;i<=n;i++)
                arr[1][i]=1.0/n;

            for(int i=n+1;i<=k;i++)
                arr[1][i]=0.0;
        }
        else
        {
            for(int i=1;i<=k;i++)
                arr[1][i]=1.0/n;
        }

        for(int i=2;i<=m;i++)
        {
            for(int j=1;j<=k;j++)
            {
                double sum=0.0;
                for(int x=n;x>0;x--)
                {
                    if(x>j)
                        continue;

                    sum+=(1.0/n)*arr[i-1][j-x];
                }

                arr[i][j]=sum*1.0;
            }
        }

        double temp=arr[m][k]*1.0;
        int c=0;
        while(temp<1.0 && temp>0.0)
        {
            temp*=10.0;
            c++;
        }

        printf("%.3f %d\n",temp,c);
    }
}
