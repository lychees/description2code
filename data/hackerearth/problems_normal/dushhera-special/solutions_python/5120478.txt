test = int(raw_input())

while(test):
	N,K = raw_input().split()
	N = int(N)
	K = int(K)
	j = 0
	ans = []
	j = K + 1
	for i in range(0,N):
		ans.append(j+(j*i))
	lst = map(str, ans)  
	print " ".join(lst)
	
	test -= 1
