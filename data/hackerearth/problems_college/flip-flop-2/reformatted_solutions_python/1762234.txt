t = int(raw_input())
for _ in range(t):
    s = raw_input()
    if s[0] == 'X':
        ex = 'Y'
    else:
        ex = 'X'
    count = 0
    for ch in s[1:]:
        if ch != ex:
            count += 1
        else:
            if ch == 'X':
                ex = 'Y'
            else:
                ex = 'X'
    print count
