#pragma GCC optimize("-O2")

#include<iostream>
using namespace std;

int M,P;
int p[250],m[15]={0};
int sum=0,max_g=99999;
float avg;

//i'th mover
//j is the package to be carried out
void mover(int i,int j){
	m[i]=0;
	if(i!=M-1){			//If not the last guy
		while((m[i]+p[j]<avg)&&(j<P))
			m[i]+=p[j++];
		mover(i+1,j);	//Less than avg
		m[i]+=p[j++];
		mover(i+1,j);	//Greater than avg
		if(j<P){
		    m[i]+=p[j++];
		    mover(i+1,j);	//Greater than avg (+1 package)
		}
	}
	else{				//Last guy
		while(j<P)
			m[i]+=p[j++];
		register int k,max_c=0;
		for(k=0;k<M;++k)
			if(max_c<m[k])
				max_c=m[k];
		if(max_g>max_c)
			max_g=max_c;
	}
}

int main(){
	register int i;
	cin>>M>>P;
	for(i=0;i<P;++i){
		cin>>p[i];
		sum+=p[i];
	}
	avg=((float)sum)/M;
	mover(0,0);
	cout<<max_g<<'\n';
	return 0;
}
