#include  <stdio.h>
int n,cnt=0;
bool vst[101]={},g[101][101]={};
inline void dfs(int a){
	++cnt;
	vst[a]=1;
	for(int i=1;i<=n;++i)
		if(g[a][i]&&!vst[i])
			dfs(i);
}
int main() {
	int m,x,y;
	scanf("%d%d",&n,&m);
	if(n==m){
		while(m-->0){
			scanf("%d%d",&x,&y);
			g[x][y]=1;
			g[y][x]=1;
		}
		dfs(1);
		if(cnt==n){
			printf("FHTAGN!");
			return 0;
		}
	}
	printf("NO");
	return 0;
}
