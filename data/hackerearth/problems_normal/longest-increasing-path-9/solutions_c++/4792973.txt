#include <iostream>
#include<cstring>
using namespace std;
void solve()
{
	int n,m;
	cin>>n>>m;
	int arr[n+1][m+1];
	int i,j;
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=m;j++)
		{
			cin>>arr[i][j];
		}
	}
	int dp[n+1][m+1],r1=0,r2=0,res=-100;
	memset(dp,0,sizeof(dp));
	dp[1][1]=1;
	
	for(i=2;i<=m;i++)
	{
	if(arr[1][i-1]<arr[1][i]&&dp[1][i-1]!=0)	
	dp[1][i]=1+dp[1][i-1];
	
	}
	for(i=2;i<=n;i++)
	{
	if(arr[i-1][1]<arr[i][1]&&dp[i-1][1])	
	dp[i][1]=1+dp[i-1][1];
		
	}
	for(i=2;i<=n;i++)
	{
		for(j=2;j<=m;j++)
		{
			r1=0;r2=0;
			if(arr[i-1][j]<arr[i][j]&&dp[i-1][j]!=0)
			r1=dp[i-1][j]+1;
			if(arr[i][j-1]<arr[i][j]&&dp[i][j-1]!=0)
			r2=dp[i][j-1]+1;
			dp[i][j]=max(r1,r2);
		
		}
	}
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=m;j++)
		{
			//cout<<dp[i][j]<<" ";
			res=max(res,dp[i][j]);
		}
	//	cout<<"\n";
	}
	cout<<res<<"\n";
	
}
int main()
{
   int t;
   cin>>t;
   while(t--)
   solve();
    return 0;
}
