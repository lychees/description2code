#include <iostream>
using namespace std;

int main()
{
    int t;cin>>t;
    while(t--){
    	long long a,b;cin>>a>>b;
    	long long temp=0,ans=0;
    	for(int i=0;i<32;++i){
    		if(b<a+(1LL<<i)-temp && a&(1LL<<i)){
    			ans|=(1LL<<i);
    		}
    		temp|=(a&(1LL<<i))?(1LL<<i):0;
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
