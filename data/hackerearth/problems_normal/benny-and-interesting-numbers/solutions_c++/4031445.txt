#include <iostream>
#include <cmath>
using namespace std;


int main() {
	// your code goes here
	int t;
	cin>>t;
	
	for(int i=0;i<t;i++) {
	    
	    int f = 0;
	    
	    
	    long long x;
	    cin>>x;
	    
	    for(int j=2;j<=sqrt(x);j++) {
	        if(x%j == 0) {
	            int c = 0;
	            while(x%j == 0) {
	                c++;
	                x = x/j;
	            }
	            if(c>1)
	                f = 1;
	        }
	    }
	    if(f) {
	        cout<<"No"<<endl;
	    }
	    else {
	        cout<<"Yes"<<endl;
	    }
	    
	    
	    
	}
	
	
	return 0;
}
