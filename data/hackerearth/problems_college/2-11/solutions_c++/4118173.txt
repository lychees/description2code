#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int d[30][30],adj[30][30];
void fw()
{
	int i,j,k;
	for(i=0;i<26;i++)
	{
		for(j=0;j<26;j++)
		{
			if(i==j)
			{
				d[i][j]=30;
			}
			else
			{
				d[i][j]=adj[i][j];
			}
		}
	}
	for(k=0;k<26;k++)
	{
		for(i=0;i<26;i++)
		{
			for(j=0;j<26;j++)
			{
				if(d[i][k]+d[k][j]<d[i][j])
				{
					d[i][j]=d[i][k]+d[k][j];
				}
			}
		}
	}
}
int main()
{
    int i,l,n,j,k;
    string s1,s2;
    long int ans;
    cin>>n;
    for(i=1;i<=26;i++)
    {
    	for(j=1;j<=26;j++)
    	{
    		cin>>adj[i-1][j-1];
    	}
    }
    fw();
    while(n--)
    {
    	cin>>s1>>s2;
    	l=s1.size();
    	ans=0;
    	for(i=0;i<l;i++)
    	{
    		if(s1[i]!=s2[i])
    		{
    			ans+=d[s1[i]-'a'][s2[i]-'a'];
    		}
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
