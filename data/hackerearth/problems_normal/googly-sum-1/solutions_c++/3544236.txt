#define MOD 1000000007
int a[5555];
int b[333333],c[333333];
int main(void){
	std::ios::sync_with_stdio(false);cin.tie(0);
	int i,j,n;
	int re=0;
	int t;
	cin>>n;
	for(i=0;i<n;i++)
		cin>>a[i];
	t=a[0]+a[0];
	if(t>=0)
		b[t]=1;
	else
		c[-t]=1;
	for(i=1;i<n;i++)
	{
		for(j=0;j<i;j++)
		{
			t=a[i]-a[j];
			if(t>=0&&b[t]==1)
			{
				re++;
				break;
			}
			else if(t<0&&c[-t]==1)
			{
				re++;
				break;
			}
		}
		for(j=0;j<=i;j++)
		{
			t=a[i]+a[j];
			if(t>=0)
				b[t]=1;
			else
				c[-t]=1;
		}
	}
	cout<<re<<endl;
	//system("pause");
	return 0;
}
