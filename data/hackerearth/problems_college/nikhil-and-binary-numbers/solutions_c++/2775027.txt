#include <iostream>
using namespace std;

int main()
{
int t;
scanf("%d",&t);
while (t--)
{ int n;
	scanf("%d",&n);
	int a[100001],i,k1=0,k2=0;
	for(i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
		if(a[i]==0)
		k1++;
		else k2++;
		
	}
	if(k2%2==0)
	printf("%d\n",k1);
	else printf("%d\n",k2);
}
    return 0;
}
