#include <bits/stdc++.h>
using namespace std;
int L[1001][1001];
int main()
{
    int test;
    scanf("%d",&test);
    while(test--){
    	string s1,s2;
    	cin>>s1>>s2;
    	int m = s1.length();
    	int n = s2.length();
    	int i,j;
    	int result = 0;
    	int pos = -1,idx = -1;;
    	 for ( i=0; i<=m; i++){
     		for (j=0; j<=n; j++){
       			if (i == 0 || j == 0)
         			L[i][j] = 0;
  
       			else if (s1[i-1] == s2[j-1]){
         			L[i][j] = L[i-1][j-1] + 1;
         			if(result<L[i][j]){
         				result = L[i][j];
         				pos = j;
         				idx = i;
         			}
         		}
  
		       	else
         		L[i][j] = 0;
     		}
   		}
   		//cout<<result<<endl;
   		if(result==0){
   			printf("%d\n",n);
   			continue;
   		}
   		i = idx-result;
   		j = pos-result;
   		int ans = 0;
   		while(j>=0&&i>=0){
   			if(s1[i]!=s2[j]) ans++;
   			i--;j--;
   		}
   		i = idx;
   		j = pos;
   		//cout<<i<<" "<<j<<endl;
   		while(j<n&&i<m){
   			if(s1[i]!=s2[j]){
   				ans++;
   			}
   			i++;j++;
   		}
   		printf("%d\n",ans);
    }
    return 0;
}
