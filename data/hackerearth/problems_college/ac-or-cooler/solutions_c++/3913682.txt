#include <bits/stdc++.h>
using namespace std;

int main()
{
    long int n,m,i,l,j;
    vector <long int> a,b;
    
    cin>>n;
    cin>>m;
    
    for(i=0;i<n;i++){
    	cin>>l;
    	a.push_back(l);
    }
    for(i=0;i<n;i++){
    	cin>>l;
    	b.push_back(l);
    }

    sort(a.begin(),a.end());
    sort(b.begin(),b.end());
    
    long long int sum=0;
    l=0;i=0;j=0;
    
    while(sum<=m  && l<=n && i<n && j<n){
    	if(a[i]<b[j]){
    		sum+=a[i];
    		i++;
    	}
    	else{
    		sum+=b[j];
    		j++;
    	}
    	
    //	cout<<sum<<" ";
    	l++;
    }
    
    
    while(sum<=m && j<n && l<=n){
    	sum+=b[j];
    	j++;
    }
    
    while(sum<=m && i<n && l<=n){
    	sum+=a[i];
    	i++;
    }
    
    cout<<--l<<endl;
    
    return 0;
}
