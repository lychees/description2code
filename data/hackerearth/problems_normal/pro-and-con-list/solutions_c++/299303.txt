#include<stdio.h>
#include<algorithm>
using namespace std;
struct node
{
	long long int a1,a2,diff;
}m[100000];
int comp(node m1,node m2)
{
	return m1.diff<m2.diff;
}
int main()
{
	int t,n,i;
	long long int m3,s=0;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		for(i=0;i<n;i++)
		{
			scanf("%d",&m[i].a1);
		scanf("%d",&m[i].a2);
		m[i].diff=m[i].a1+m[i].a2;
	
	s+=m[i].a2;
	}
	
	sort(m,m+n,comp);
/*for(i=0;i<n;i++)
	{
		printf("%lld ",m[i].a1);
	}*/
m3=m[n-1].a1+m[n-2].a1+m[n-1].a2+m[n-2].a2-s;
	printf("%lld\n",m3);
	s=0;
	}
	return 0;
}
