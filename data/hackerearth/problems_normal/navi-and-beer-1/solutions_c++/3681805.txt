#include <iostream>
using namespace std;

long int reqMoney(long int n,long int k, long int m)
{
	long int sum=0;
	for(int i=1;i<=n;i++)
	{
		sum=sum+ k*i*i*i;
	}
	long int money = sum-m;
	if(money>0)
	return money;
	else
	return 0;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long int n,k,m;
    	cin>>n>>k>>m;
    	cout<<reqMoney(n,k,m)<<endl;
    }
    return 0;
}
