#include <stdio.h>
#include <string>
#include <iostream>
#include <algorithm>

using namespace std;

string s[100 * 1000];

void PalinPair()
{
	int N;
	cin >> N;

	long long int count = 0;

	for (int i = 0; i < N; i++)
	{
		cin >> s[i];
	}

	sort(s, s + N);


	for (int i = 0; i < N; i++)
	{
		string temp = s[i];
		reverse(temp.begin(), temp.end());

		string* l = lower_bound(s + i + 1, s + N,temp);
		string* u = upper_bound(s + i + 1, s + N, temp);

		count += u - l;
	
	}

	cout << count;
}


int main()
{
	PalinPair();
}
