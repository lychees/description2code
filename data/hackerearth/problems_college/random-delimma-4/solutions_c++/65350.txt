int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        ll k,n;
        cin>>k>>n;
        double ans=1.0;
        if(k>n)
            printf("%.9lf\n",ans);
        else{
            for(int i=0;i<k;i++)
                ans=ans*(((n-i)*1.0)/n);
            printf("%.9lf\n",1.0-ans);
        }
    }

    return 0;
