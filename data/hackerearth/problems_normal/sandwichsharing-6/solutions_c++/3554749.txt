#include <iostream>
using namespace std;
#define ulli unsigned long long int

int main()
{
    int T, i;
    ulli N, P, result, j, temp;
    
    cin>>T;
    
    for( i = 1; i <= T; i++ )
    {
    	result = 0;
    	
    	cin>>N;
    	cin>>P;
    	temp = N;
    	
    	for( j = 1; j <= P; j++ )
    	{
    		if( temp == 3 )
    		{
    			result = 3;
    			break;
    		}
    		if( temp%2 != 0 )
    		{
    			temp = (temp-1)/2 + 2;
    		}
    		else
    		{
    			temp = (temp-2)/2 + 2;
    		}
    	}
    	
    	if( j == P+1 )
    		result = temp;
    		
    	cout<<result<<endl;
    }
    return 0;
}
