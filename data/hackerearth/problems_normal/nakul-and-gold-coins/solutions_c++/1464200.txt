bool prime[1000007];

	memset(prime, false, sizeof(prime));

		if(prime[i] == false) 
			for(j = 2*i;j <= 1e6;j += i) 
				prime[j] = true;
				
	for(i = 2;i <= 1e6;i++) 
		if(prime[i] == false)
