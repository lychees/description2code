#include <iostream>
using namespace std;
#define mod 1000000007
int main() {
	// your code goes here
	int t,tm;
	cin>>t;
	long long int a[10][2];
	long long int n;
	while( t-- )
	{
		a[0][0]=0;
		for(int i=1 ; i<10 ; i++)
		{
			a[i][0]=1;
		}
		int sum=0;
		int in1=0;
		int in2=1;
		//cout<<"he";
		cin>>n;
		if(n==0)
		{
			cout<<0<<endl;
			continue;
		}
		//cout<<"ge";
		n=n-1;
		if(n==0)
		sum=1;
		while(n--)
		{
			for(int i=0;i<10;i++)
				a[i][in2]=0;
			for(int i=0;i<10;i++)
			{
				if(i-1>=0)
					a[i-1][in2]=(a[i-1][in2]+a[i][in1])%mod;
				if(i+1<10)
					a[i+1][in2]=(a[i+1][in2]+a[i][in1])%mod;
			}
			tm=in1;
			in1=in2;
			in2=tm;
		}
		for(int i=0;i<10;i++)
		sum=(sum+a[i][in1])%mod;
		cout<<sum<<endl;
	}
	return 0;
}
