#include<iostream>
unsigned int calculate(unsigned int sides,unsigned int  person){
  if(0 == person ){
    return sides;
  }
  if(3 == sides ){
    return 3;
  }
  return (sides%2)==0?calculate((sides/2+1),person-1)
    :calculate((sides/2+2),person-1);
}
int main()
{
  unsigned int Testcases;
  std::cin>>Testcases;
  while(Testcases--){
    unsigned int sides;
    std::cin>>sides;
    unsigned int person ;
    std::cin>>person;
    std::cout<<calculate(sides,person) <<std::endl;
  }

  return 0;
}
