#include<stdio.h>
#include<iostream>
using namespace std;
int main()
{
    int n,a[101];
    scanf("%d",&n);
    int j=0,i,count=1;
    long long int ans=1;
    for(i=0;i<n;i++)
      scanf("%d",&a[i]);
    while(a[j]==0 && (j<n))
     j++;
     if(j==n)
      printf("0\n");
      else
      {
          for(i=j;i<n;i++)
          {
              if(a[i]==1)
               {
                   ans*=count;
                   count=1;
               }
              else
               count++;
          }
         cout<<ans;
         // printf("%lld\n",ans);
      }
    return 0;
}
