#include <bits/stdc++.h>

using namespace std;

#define ll long long
#define ull unsigned long long
#define pb push_back
#define f first
#define s second
#define PI 3.14159265359
typedef pair<int,int> ii;
typedef pair<int,pair<int,int> > iii;

double dp[1001][1001];
static double P[1001][1001];

int main(int argc, const char * argv[])
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int t,n,m,i,j;
    double res;
    
    cin>>t;
    while(t--)
    {
        cin>>n>>m;
        
        for(i=1;i<=n;i++)
            for(j=1;j<=m;j++)
                cin>>P[i][j];
        
        for(i=0;i<=n;i++)
            dp[i][0]=0;
        for(j=1;j<=m;j++)
            dp[0][j]=0;
        res=1.0;
        
        dp[1][1]=1;
        for(i=1;i<=n;i++)
            for(j=1;j<=m;j++)
            {
                if(i==1 && j==1)
                    continue;
                dp[i][j]=dp[i][j-1]*P[i][j-1]+dp[i-1][j]*(1-P[i-1][j]);
                res+=dp[i][j];
            }
        
        
//        for(i=1;i<=n;i++)
//            res+=dp[i][m]*P[i][m];
//      
//        for(j=1;j<=m;j++)
//            res+=dp[n][j]*(1-P[n][j]);
        
        cout<<fixed;
        cout<<setprecision(6)<<res<<"\n";
        
    }
    
    return 0;
}
