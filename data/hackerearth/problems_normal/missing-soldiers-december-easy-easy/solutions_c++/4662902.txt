#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

int main()
{
	int N, x, y, d;
	vector <pair <int, int>> a;
	
	cin >> N;
	
	for (int i = 0; i < N; i++)
	{
		cin >> x >> y >> d;
		a.push_back(make_pair(x, x+d));
	}
	
	sort(a.begin(), a.end());
	
	int ans = 0;
	
	// for (int i = 0; i < a.size(); i++)
		// cout << a[i].first << " " << a[i].second << endl;
	
	int left = a[0].first;
	int right = a[0].second;
	
	for (int i = 1; i < a.size(); i++)
	{
		if (a[i].first <= right)
		{
			right = max(a[i].second, right);
		}
		else
		{
			ans += right - left + 1;
			left = a[i].first;
			right = a[i].second;
		}
	}
	
	ans += right - left + 1;
	
	cout << ans << endl;
	
    return 0;
}
