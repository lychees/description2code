#include <iostream>
#include <vector>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <cmath>
#include <climits>
#include <cstring>
using namespace std;

bool sortB(const pair<long, long>& l, const pair<long, long>& r) {
	return l.first < r.first;
}
int main(int argc, char const *argv[])
{
	int t, A[160010], min;
	long n, l, x, y;
	cin >> t;
	while (t != 0) {
		vector<pair<long, long> > v;
		cin >> n >> l;
		for (int i = 0; i < l; i++) {
			cin >> x >> y;
			v.push_back(make_pair(x, y));
		}
		sort(v.begin(), v.end(), sortB);
		A[n] = 0;
		A[n - 1] = INT_MAX;
		for (int i = n - 2; i >= 1; i--) {
			if (l != 0) {
				if (i == v[l - 1].first && v[l - 1].first != 2) {
					A[i] = A[v[l - 1].second];
					l--;
					continue;
				}
			}
			min = INT_MAX;
			for (int j = i + 2; j <= i + 12 && j <= n; j++) {
				if (A[j] < min) min = A[j];
			}
			A[i] = min + 1;
		}
		//for (int i = 1; i <= n; i++) cout << A[i] << endl;
		cout << A[1] << endl;
		memset(A, 0, sizeof(A));
		t--;
	}
	return 0;
}
