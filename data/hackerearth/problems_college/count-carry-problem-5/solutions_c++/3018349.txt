#include <bits/stdc++.h>
using namespace std;
 
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		unsigned long long int a,b;
		cin>>a>>b;
		unsigned long long int tempa=a;
		unsigned long long int tempb=b;
		int dig1,dig2;
		int carry=0;
		int ta=0;
		while(tempa || tempb)
		{
			dig1=tempa%10;
			dig2=tempb%10;
			if(dig1+dig2+ta>9)
			{
				carry++;
				ta=1;
			}
			else
			ta=0;
			tempa=tempa/10;
			tempb=tempb/10;
		}
	if(carry==0)	cout<<"No carry operation\n";
	else if(carry==1) cout<<"1 carry operation\n";
	else cout<<carry<<" carry operations\n";
	}
    return 0;
}
