#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	long long b,w,sum=0,x,y,z;
    	cin>>b>>w;
    	cin>>x>>y>>z;
    	if(y+z<x && x+z<y){
    		sum=b*(y+z)+(w*(x+z));
    	}
    	else if(y+z<x){
    		sum=b*(y+z)+w*y;
    	}
    	else if(x+z<y){
    		sum=b*x+w*(x+z);
    	}
    	else sum=(b*x)+(w*y);
    	cout<<sum<<"\n";
    }
    return 0;
}
