#include <iostream>
using namespace std;

int main()
{
    int t,i;
    long int n,sum=0;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	int A[n];
    	for(i=0;i<n;i++)
    	{
    		cin>>A[i];
    		if(A[i]%2!=0)
    			sum+=A[i];
    	}
    	cout<<sum<<endl;
    	sum=0;
    }
    
    return 0;
}
