#include <cstring>
#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

int main() {
	int T;
	cin >> T;
	while (T--) {
		int N;
		cin >> N;
		vector<int> V1(N), V2(N);
		for (int i = 0; i < N; i++) cin >> V1[i];
		for (int i = 0; i < N; i++) cin >> V2[i];
		int ans = 0;
		sort(V1.begin(), V1.end());
		sort(V2.begin(), V2.end());
		for (int i = 0; i < N; i++) {
			ans += V1[i] * V2[i];
		}
		cout << ans << '\n';
	}

	return 0;
}
