#include<bits/stdc++.h>
using namespace std;

long long int find_ans( long long int a[], int idx , int n  , int k )
{
	
	
	long long int ans[n+1];
	ans[0] = max((long long )0,a[idx]) ;
	for( int i = 1  ; i <=k ; i++ )
	{
		ans[i] = max(ans[i-1],a[i]) ;
	}
	
	for( int i = k+1 ; i < n ; i++ )
	{
		ans[i] = max(ans[i-1],ans[i-k-1]+a[i]);
	}
	return ans[n-1];
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n ,k;
		cin>>n>>k;
		long long int a[n] ;
		for( int i = 0 ; i  < n ; i++ )
		{
			cin>>a[i] ;
		}
		
		cout<<find_ans(a,0,n,k)<<"\n";
	}
}
