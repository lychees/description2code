#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long t,n,x,c=0;
    cin>>t;
    while(t--){
    	cin>>n>>x;
    	c=0;
    	long long a[n+1];
    	cin>>a[1];
    	for(long long i=2;i<=n;i++){
    		cin>>a[i];
    		if(a[i]==a[i-1]){
    			c++;
    			a[i]+=x;
    		}
    		else if(a[i]<a[i-1]){
    			/*while(a[i]<a[i-1]){
    				c++;
    				a[i]+=x;
    			}*/
    			long long step=ceil((a[i-1]-a[i])/x);
    			c+=step;
    			a[i]+=(x*step);
    			if(a[i]<=a[i-1]){
    			c++;
    			a[i]+=x;
    		}
    		}
    	}
    	cout<<c<<endl;
    }
    return 0;
}
