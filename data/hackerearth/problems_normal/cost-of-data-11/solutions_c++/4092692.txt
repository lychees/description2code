{

void insert(string s)
{
    int node=ROOT,i=0;
    while(i<s.size())
    {
        if(trie[node][s[i]-'a']==-1)
            node=trie[node][s[i++]-'a']=nxt++;
        else
            node=trie[node][s[i++]-'a'];
    }
}
