#include <iostream>
using namespace std;

int main()
{
   unsigned long long int A[100000],e=0,o=0,n,i,t;
   
   cin>>t;
   
   while(t)
   {  
   	   e=0;  o=0;
   	   
   	 cin>>n;
   
   	 for(i=0;i<n;i++)
   	   cin>>A[i];
   	   
   	 for(i=0;i<n;i++) 
   	   {
   	   	if(A[i]%2==0)
   	   	  e++;
   	   	else 
   	   	  o++;
   	   }
  	  
  // 	  cout<<"odd= "<<o<<endl; 
   //	  cout<<"even= "<<e<<endl;
   	  cout<<o*e<<endl;
   	t--;
   }
   
    return 0;
}
