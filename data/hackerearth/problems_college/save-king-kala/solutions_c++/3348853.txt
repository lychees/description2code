typedef unsigned long long int ull;

vll a[10005],an[10005];ll pr[100005],mod=10000000000037ll,power;

ll mul(ll a,ll b)
{
	ll r=0;
	while(b)
	{
		if(b%2)
		r=(r+a)%mod;
		a=(a+a)%mod;
		b/=2;
	}
	return r;
}

ll p(ll a,ll b)
{
	ll r=1;
	while(b)
	{
		if(b%2)
		r=mul(r,a);
		a=mul(a,a);
		b/=2;
	}
	return r;
}

void seive()
{
	pr[0]=1;pr[1]=1;
	rep(i,2,100004)
	{
		if(pr[i])continue;
		for(int j=2*i;j<=100004;j+=i)
		pr[j]=1;
	}
}

void solve(ll n)
{
	ll y=n;n=n*n;
	ll i,ans=1,c1=0;
	for(i=2;i*i<n;i++)
	{
		if(n%i==0)
		{
			ll cnt=0;
			while(n%i==0)
			{
				cnt++;
				n/=i;
			}
			ans=ans*(cnt+1);
						
		}
		ll x=sqrt(n);
		if(pr[x]==0&&x*x==n)
		{
				ans*=3;
				n=1;
				break;
		}
	}
	ll x=sqrt(n);
		if(pr[x]==0&&x*x==n)
		{
				ans*=3;
				
		}
	if(ans<=10004)
	{
	 a[ans].pb(y*y);
	 an[ans].pb(p(y*y,power));
    }
}
