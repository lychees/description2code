#include <bits/stdc++.h>
using namespace std;

string s;
int str[256];
int k;

void pre() {
    str['a']=1;
    str['b']=2;
    str['c']=3;
    str['d']=4;
    str['e']=5;
    str['f']=6;
    str['g']=7;
    str['h']=8;
    str['i']=9;
    str['j']=10;
    str['k']=11;
    str['l']=12;
    str['m']=13;
    str['n']=14;
    str['o']=15;
    str['p']=16;
    str['q']=17;
    str['r']=18;
    str['s']=19;
    str['t']=20;
    str['u']=21;
    str['v']=22;
    str['w']=23;
    str['x']=24;
    str['y']=25;
    str['z']=26;
}

int main()
{
    int t, i, j;
    pre();
    cin >> t;
    while (t--) {
        cin >> k >> s;
        int cnt = 0, sum = 0;
        for (i = 0, j = 0; i < s.size(); i++) {
            sum += str[s[i]];
            while (sum > k && j <= i) {
                sum -= str[s[j++]];
            }
            if (sum == k) {
                cnt++;
            }
        }
        cout << cnt << endl;
    }
    return 0;
}
