#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long n;
    	cin>>n;
    	long long g[n],b[n],i;
    	for(i=0;i<n;i++)
    		cin>>g[i];
    	for(i=0;i<n;i++)
    		cin>>b[i];
    	sort(g,g+n);
    	sort(b,b+n);
    	long long ans = 0;
    	for(i=0;i<n;i++)
    	{
    		if(g[i]%b[n-i-1]==0 || b[n-i-1]%g[i]==0)
    			ans++;
    	}
    	cout<<ans<<endl;
    }
    
    return 0;
}
