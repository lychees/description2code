#include<bits/stdc++.h>
#define LL long long int
#define MAX 100009
using namespace std;
LL dp[MAX],val[MAX],pre[MAX];

int main() 
{
	ios::sync_with_stdio(false);
	for(int i=0; i<MAX; i++)
	{
		dp[i]=0;
	}
    int t; 
    cin >> t;
    while(t--) 
    {
        pre[0]=0;
        LL n,i,j; 
        cin >> n;
        for(i=n; i>0; i--)
        {
            cin >> val[i];
        } 
        for(i=1; i<=n; i++) 
            pre[i]=pre[i-1]+val[i];
        dp[0]=0;
        dp[1]=val[1];
        dp[2]=dp[1]+val[2];
        dp[3]=dp[2]+val[3]; 
        for(i=3; i<=n; i++)
        {
            dp[i]=max(-1ll,pre[i-1]-dp[i-1]+val[i]);
            dp[i]=max(dp[i],pre[i-2]-dp[i-2]+val[i]+val[i-1]);
        }
        LL davy, bets;
        davy=dp[n];
        bets=pre[n]-davy;
        if(davy>bets)
        cout<<"LOSE"<<endl;
        else if(davy<bets)
        cout<<"WIN"<<endl;
        else
        cout<<"DRAW"<<endl;
    }
    return 0; 
}
