#include <stdio.h>
long int a[10000005], pr[10000005], phi[10000005];
void pri(){
	int i, p;
	for (p = 2; p <= 10000005; p++){
		if (pr[p] == 0){
			for (i = p * 2; i<10000005; i += p){
				pr[i] = p;
			}
		}
	}
}
void init(){
	long int i;
	phi[1] = 1;
	for (i = 2; i<10000005; i++){
		long int q = i, val = pr[i];
		if (val == 0)
			phi[i] = i - 1;
		else
		{
			while (q%val == 0)
				q /= val;

			phi[i] = phi[q] * (i / q) / val*(val - 1);
		}
	}
	a[1] = 1;
	for (i = 2; i<10000005; i++)
		a[i] = a[i - 1] + phi[i];
}
int main()
{
	//printf("Hello World!\n");
	pri();
	init();

	long int t;
	scanf("%ld", &t);
	while (t--){
		long int n;
		scanf("%ld", &n);
		printf("%ld\n", a[n]);
	}
	return 0;
}
