#include <iostream>
#include <cstring>
#include <cmath>
using namespace std;

int main()
{
    int n,size,length,i,j,c=0,t;
    
    char mat[1000][1000];
    cin>>n;
    t=n+1;
    while(t--)
    {
    	char str[1000];
    	gets(str);
    	//cout<<str<<"\n";
    	
    
    length=strlen(str);
    //cout<<length<<"\n";
    size=sqrt(length);
    
   // cout<<size<<"\n";
   c=0;
    for(i=0;i<size;i++)
    {	
    	for(j=0;j<size;j++)
    	{
    		
    		mat[i][j]=str[c];
    		c++;
    	}
    	
    
    	
    }
    
    for(j=0;j<size;j++)
    {
    	for(i=0;i<size;i++)
    	{
    		cout<<mat[i][j];
    	}
    }
    cout<<"\n";
    }
    
    return 0;
}
