#include<stdio.h>
#include<vector>
//#include<algorithm>
using namespace std;
vector<long int> ar;

int main(){
	long int n,i,Q, j, div=1;

scanf("%ld", &n);
for(i=0; i<n; i++)
{	long int k;
	scanf("%ld", &k);
	ar.push_back(k);
}

scanf("%ld", &Q);
for(i=0; i<Q; i++)
{	long int k;
	scanf("%ld", &k);
	
	div=div*k;
	

}

for(i=0; i<n; i++)
{if(div!=0)
printf("%ld ", ar[i]/div);
else{
	ar[i]=0;
	printf("%ld ", ar[i]);
}
}
return 0;
}
