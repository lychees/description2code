#include <iostream>
using namespace std;

unsigned long long int lk[1002];

unsigned long long int fib(int n){
	
	if(lk[n]==0){
		if(n<=1)
			lk[n]=n;
		else
			lk[n]=fib(n-1)+fib(n-2);
	}
	return lk[n];
}	
int main()
{
 	int t,n;
 	cin >> t;
 	while(t--){
 		cin >> n;
 		cout << fib(n+1) << endl;
 	}
}
