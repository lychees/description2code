#include <iostream>
#include<cstring>
using namespace std;
long long a[1000000];
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    memset(a,-1,sizeof(a));
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	if(a[n]==-1){
    	for(long long i=1;i<1000000;i++)
    	a[i]=i;
    	
    	for(long long i=2;i<1000000;i++){
    		if(a[i]==i){
    		for(int j=2*i;j<1000000;j=j+i)
    		a[j]-=a[j]/i;}
    		
    		
    	}
    	for(long long i=2;i<1000000;i++)
    	if(a[i]==i)
    	a[i]=i-1;
    	}
    	cout<<a[n]<<endl;
    	
    }
    return 0;
}
