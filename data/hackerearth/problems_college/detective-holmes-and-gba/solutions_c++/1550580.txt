#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
	  unsigned long long int stu,can,pos;

		unsigned long long int res=(pos+(can%stu)-1)%stu;//the last mod if the index again exceeds last value
		cout<<res<<endl;
		
	}

    return 0;
}
