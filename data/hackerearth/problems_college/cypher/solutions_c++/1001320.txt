#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	char s[101];
    	cin>>s;
    	int l=strlen(s);
    	for(int i=0;i<l;i++)
    	{
    		int k=0;
    		k=i%l;
    		int p=s[i]+k;
    		if(p>122)
    		{
    			int q=p-26;
    			char c=q;
    			s[i]=c;
    		}
    		else
    		s[i]=s[i]+k;
    		
    	}
    	cout<<s<<endl;
    }
    return 0;
}
