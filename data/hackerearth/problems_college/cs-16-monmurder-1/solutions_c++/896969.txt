ll a[2000000] = {0};
ll b[2000000] = {0};

int main() {
	ll n;
	ll i,j;
	vector<ll> v;
	for(i = 2;i <= 1000;i++) {
		if(a[i] == 0) {
			for(j = 2*i;j <= 1000000;j+= i) {
				a[j] = 1;
			}
		}
	}
	for(i = 2;i <= 1000000;i++) {
		if(a[i] == 0) {
			v.pb(i);
		}

		ll val = v[i];
		for(;val <= 1000000;val *= v[i]) {
			b[val] = 1;
		}
	}
//	cout << v.size() << "\n";
	S(n);
	while(n--) {
		ll nu;
		S(nu);
		ll val= 0;

			ll co = 0;
			while(nu % v[i] == 0) {

				nu = nu / v[i];
			}
			val = val ^ co;
 		}
