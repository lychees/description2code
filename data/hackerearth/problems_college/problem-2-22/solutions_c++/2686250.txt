#include<bits/stdc++.h>
using namespace std;
long long int t,r,xc,x[4][2],yc,d,ans;

int main(){
    cin>>t;
    while(t--){
    cin>>r>>xc>>yc;
    ans=0;
    for(int i=0;i<4;i++){
        cin>>x[i][0]>>x[i][1];

        d=sqrt((pow((xc-x[i][0]),2))+(pow((yc-x[i][1]),2)));
        //cout<<"d="<<d<<endl;
        if(d<r)
            ans++;


    }
    if(ans==4)
        cout<<"Yes"<<endl;
    else
        cout<<"No"<<endl;
    
    }


return 0;
}
