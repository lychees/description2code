#include <stdio.h>

using namespace std;

int main()
{
	long long int t;
    scanf("%lld",&t);
        
    while(t--)
	{
		long long int odd=0,even=0;
		long long int R,C;
		scanf("%lld %lld",&R,&C);
		long long e=0,o=0,p=0;
		
		if(R%2)	//R is Odd
		{
			p=R;
			o=(R+1)/2;  e=R-o;
		}
		else 		//R is Even
		{
			p=R;
			e=o=R/2;
		}
		
		if(R<=C)
		{
			odd=o*o;
			even = R*(R+1)/2 - odd;
		}
		else
		{
			if(C%2)	//C is Odd
			{
				//printf("\nNum1,C=%d",C);
				long long int No_of_Odd= (C+1)/2;
				long long int No_of_Even= C- No_of_Odd;
				
				odd  = (No_of_Odd)  * (1+C)/2;
				even = (No_of_Even) * (1+C)/2;
				//printf("\nNum2 ,odd= %d , even=%d",odd,even);
			}
			else    //C is Even
			{
				//printf("\nNum3,C=%d",C);
				long long int No_of_Even = (C/2);	//No_of_odd = No_of_Even
				//long No_of_Odd  = C- No_of_Even;
				
				odd  = (No_of_Even  * C)/2;
				even = (No_of_Even * (2+C))/2;
				//printf("\nNum4 ,odd= %d , even=%d",odd,even);
			}
			
			int n=C+1,k=R-n+1;
			if(n%2) // n is Odd
			{
				//printf("\nNum5,n=%d",n);
				long long int No_of_Odd  = ((k+1)/2);
				long long No_of_Even = k- No_of_Odd;
				//printf("\nNum5.1, No_of_Odd=%d, No_of_Even=%d",No_of_Odd,No_of_Even);
				
				odd  = odd  + (No_of_Odd * C);
				even = even + (No_of_Even * C);
				//printf("\nNum6 ,odd= %d , even=%d\n",odd,even);
			}
			else 	// n is Even
			{
				//printf("\nNum7,n=%d",n);
				long long int No_of_Even = ((k+1)/2);
				long long int No_of_Odd  = k- No_of_Even;
				
				odd  = odd  + (No_of_Odd * C);
				even = even + (No_of_Even* C);
				//printf("\nNum8 ,odd= %d , even=%d\n",odd,even);
			}
			
		}
		
		printf("%lld %lld\n",even,odd);
		
	}
    
    return 0;
}
