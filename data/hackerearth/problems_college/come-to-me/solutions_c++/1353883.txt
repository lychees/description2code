#include<iostream>
#include<string>
#include<cstring>
#include<algorithm>
using namespace std;
int main(){
	  int t,i;
	  cin >> t;
	while(t--){
		   string s;
		   cin >> s;
		   int ct ,mx = 0;
		   for( i = 0 ; i < s.size() ; ++i ){
   			  int hsh[26]={0};
   			    int r = i;
   			    ct = 0;
   			    while( hsh[s[r]-'A'] == 0 )
   			    {
    			   ct++;
				   hsh[s[r]-'A'] = 1;
				   r++;
		    	}	   
   		             if(ct > mx)
   		             mx = ct;
   		}
	               cout << mx << endl;	
	}
         return 0;	
}
