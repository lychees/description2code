#include<iostream>
#include<cstdio>
using namespace std;

char game[105][105];
bool visited[105][105];
int x,y;
void solve(int r,int c)
{
	if(r<0||r>=x||c<0||c>=y||game[r][c]=='*'||visited[r][c])
		return;
	visited[r][c]=true;
	int cnt=8;

	if((r-1)>=0&&(c-1)>=0&&game[r-1][c-1]=='*')
		cnt--;
	if((r-1)>=0&&game[r-1][c]=='*')
		cnt--;
	if((r-1)>=0&&(c+1)<y&&game[r-1][c+1]=='*')
		cnt--;

	if((c-1)>=0&&game[r][c-1]=='*')
		cnt--;
	if((c+1)<y&&game[r][c+1]=='*')
		cnt--;

	if((r+1)<x&&(c-1)>=0&&game[r+1][c-1]=='*')
		cnt--;
	if((r+1)<x&&game[r+1][c]=='*')
		cnt--;
	if((r+1)<x&&(c+1)<y&&game[r+1][c+1]=='*')
		cnt--;

	game[r][c]=(8-cnt)+'0';
	if(cnt==8)
	{
		solve(r-1,c-1);
		solve(r-1,c);
		solve(r-1,c+1);
		solve(r,c-1);
		solve(r,c+1);
		solve(r+1,c-1);
		solve(r+1,c);
		solve(r+1,c+1);
	}
}
int main()
{
    int t,r,c;
    //freopen("input.txt","r",stdin);
    scanf(" %d",&t);
    while(t--)
    {
    	scanf(" %d %d",&x,&y);
    	for(int i=0;i<x;i++)
            scanf("%s",game[i]);

    	for(int i=0;i<x;i++)
    	{
    		for(int j=0;j<y;j++)
    		{
    			if(game[i][j]=='c')
    				{
    					r=i;c=j;
    				}
    		}
    	}
    	for(int i=0;i<105;i++)
    		for(int j=0;j<105;j++)
    			visited[i][j]=false;
    	solve(r,c);
    	for(int i=0;i<x;i++)
    	{
    		for(int j=0;j<y;j++)
    		{
    			printf("%c",game[i][j]);
    		}
    		printf("\n");
    	}

    }
    return 0;
}
