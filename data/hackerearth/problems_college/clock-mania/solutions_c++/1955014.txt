#include <bits/stdc++.h>
using namespace std;

int main()
{
   	int  t;
   	float h,m;
   	cin>>t;
   	while(t--) {
   		cin>>h>>m;
   		int mhand = 360*(m/60);
   		int hhand = ((360 * ((float)((int)h%12))/12) + (360*(m/60)*((float)1/12)));
   		
   		//cout<<hhand<<" ";
   		//cout<<mhand<<" ";
   		int angle = abs(((int)(mhand-hhand)));
   		if(angle<=180)
   			cout<<angle<<endl;
   		else
   			cout<<360-angle<<endl;
   	}
    return 0;
}
