#include <iostream>
#include <algorithm>
#include <string>
using namespace std;
string str;
int main()
{
    //cout << "Hello World!" << endl;
    getline(cin,str);
    int n;
    cin>>n;
    int len = str.size();
    for(int i=0;i<len;i+=n)
    {
    	int temp = n;
    	for(int j=i;temp>0;j++,temp--)cout<<str[j];
    	cout<<endl;
    }
    return 0;
}
