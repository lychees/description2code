#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int c=0;
    	long long int arr[n];
    	int i;
    	for(i=0;i<n;i++)
    	cin>>arr[i];
    	for(i=0;i<n-1;i++)
    	if(arr[i]>=arr[i+1])
    	c++;
    	cout<<c<<endl;
    }
    return 0;
}
