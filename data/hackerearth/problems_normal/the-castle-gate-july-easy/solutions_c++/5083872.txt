#include <iostream>
using namespace std;

int main()
{
    int i,j,n,a[100],count,k;
    
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    	count=0;
    	for(j=1;j<a[i];j++)
    	{
    		for(k=j+1;k<=a[i];k++)
    		{
    			if((j^k)<=a[i])
    			{	
    				count++;
    			}
    		}
    	}
    	
    	cout<<count<<endl;
    }
    return 0;
}
