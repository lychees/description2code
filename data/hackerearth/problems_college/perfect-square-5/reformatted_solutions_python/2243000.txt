t = int(raw_input())
n = []
for i in range(t):
    n.append(int(raw_input()))


def p_j(a):
    j = [int(a**0.5)+1, int(a**0.5)]
    b = a-j[1]**2
    A = min(j[0]**2-a, b)
    if (A == b):
        return '-'+str(A)
    return '+'+str(A)


for i in n:
    if (i**0.5 == int(i**0.5)):
        print('YES')
    else:
        print(p_j(i))
