n=raw_input()
n=int(n)
a=map(int,raw_input().split())
m_a=a[0]
f=1
for j in a:
	f*=j
	if j<=m_a:
		m_a=j

g=0
for k in xrange(1,m_a+1):
	count=0
	for l in a:
		if l%k!=0:
			break
		else:
			count+=1
	if count==len(a):
		g=k
if g==0:
	g=1
print pow(f,g)%(1000000007)
