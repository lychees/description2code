#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int N, i, j;
    string S;
    vector<int> v;
    
    cin >> N;
    cin >> S;
    
    for(i=0; i<N; i++)
    	v.push_back(i+1);
    
    i=0, j=0;	
    int l=N;
    int len = S.size();
    
    while( l != 1 )
    {
    	if( j >= len)
    		j=0;
    	if(i >= l)
    		i=0;
    	if(S[j] == 'a')
    		i++;
    	else if(S[j] == 'b')
    	{
    		v.erase( v.begin()+i );
    		l--;
    	}
    	j++;
    }
    cout << v[0] << endl;
    return 0;
}
