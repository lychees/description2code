#include <bits/stdc++.h>
using namespace std;

int main()
{
    int u;
    cin >> u;
    while(u--) {
    	string s, t;
    	cin >> s;
    	reverse(s.begin(), s.end());
    	t = s;
    	reverse(s.begin(), s.end());
    	int flag = 0;
    	for(int i = 1; i < s.size(); i++) {
    		if(abs(s[i] - s[i - 1]) != abs(t[i] - t[i - 1])) {
    			flag = 1;
    			break;
    		}
    	}
    	if(flag) {
    		puts("Not Funny");
    	}
    	else {
    		puts("Funny");
    	}
    }
    return 0;
}
