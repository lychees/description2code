#include <bits/stdc++.h>
using namespace std;

int boxes(int arr[], int n)
{
int dp[1001]={0};

for(int i=1; i<=n; i++)
{
	dp[i]=arr[i];
	for(int j=1; j<i; j++)
	{
		dp[i] = max(dp[i], dp[j] + dp[i-j]);
		

	}

}
	return dp[n];
}

int main()
{

	int t;
	cin>>t;
	while(t--)
	{

		int n, k;
		cin>>n>>k;
		int arr[n+1];
		for(int i=1; i<=n; i++){cin>>arr[i]; arr[i]-=k;}
		cout<<boxes(arr, n)<<endl;
			
	}
	return 0;
}
