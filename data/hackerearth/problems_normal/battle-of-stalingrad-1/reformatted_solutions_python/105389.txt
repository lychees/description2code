a = []


def main():
    t = input()
    global a
    for i in range(0, t):
        a = map(lambda x: float(x), raw_input().split(' '))
        if area(*(a[:6])) == (area(*(a[:4]+a[6:])) +
                              area(*(a[:2]+a[4:]))+area(*a[2:])):
            print 'INSIDE'
        else:
            print 'OUTSIDE'


def area(x1, y1, x2, y2, x3, y3):
    return abs((x1*(y2 - y3) + x2*(y3 - y1) + x3*(y1-y2))/2.0)


main()
