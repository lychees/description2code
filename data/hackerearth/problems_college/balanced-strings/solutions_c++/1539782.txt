#include <bits/stdc++.h>
using namespace std;
#define ll long long

int main() {
  ios_base::sync_with_stdio(0);
  int t;
  cin>>t;
  while(t-->0) {
    string s;
    cin>>s;
    int a[26];
    for(int i=0; i<26; i++) a[i] = 0;
    for(int i=0; i<s.size(); i++) {
      a[s[i]-'a']++;
    }
    int res = a[0];
    bool f = true;
    for(int i=0; i<26; i++) {
      if(a[i]!=res) f = 0;
    }
    if(f) cout<<"Yes"<<endl;
    else cout<<"No"<<endl;
  }
  
  return 0;
}
