#include <iostream>
#include <cstdio>
#include <cmath>

using namespace std;

bool gcd(unsigned long long p,unsigned long long q)
{
    unsigned long long r,k,m;

    if(p>q){k=p;m=q;}
    else{k=q;m=p;}

    while(m!=0)
    {
        r=k%m;
        k=m;
        m=r;

    }
    if(k==1)
    {
        return true;
    }
    return false;
}

int main()
{
    int t;
    unsigned long long p,q;
    cin>>t;
    while(t--)
    {
        fscanf(stdin,"%lld %lld",&p,&q);
        if(gcd(p,q)){printf("Is a Co-Prime\n");}
        else{printf("Not a Co-Prime\n");}

    }

    return 0;
}
