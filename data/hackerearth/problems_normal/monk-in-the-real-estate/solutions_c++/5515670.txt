#include <bits/stdc++.h>
using namespace std;
bool visited[10001];
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int e;
    	cin>>e;
    	int count=0;
    	for(int i=0;i<10001;i++)
    	visited[i]=false;
    	for(int i=0;i<e;i++)
    	{
    		int x,y;
    		cin>>x>>y;
    		if(visited[x]==false)
    		{
    			count++;
    			visited[x]=true;
    		}
    		if(visited[y]==false)
    		{
    			count++;
    			visited[y]=true;
    		}
    	}
    	cout<<count<<endl;
    }
    return 0;
}
