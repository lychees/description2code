#include<vector>

using namespace std;
int pre[1000005],mxp=0;
string ss;
void pre_compute(int pos)
{
    int i=0,k=0;
    pre[0]=0;
    for(i=1;i<ss.size();i++)
    {
        while(k>0&&ss[i]!=ss[k])
            k=pre[k-1];
        if(ss[i]==ss[k]) k++;
        pre[i]=k;
        if(i<=pos) mxp=max(mxp,pre[i]);
    }
}
void sol()
{
    int n,k,i;
    cin>>ss>>k;
    memset(pre,0,sizeof(pre));

    if(k==0) cout<<"Puchi is a cheat!"<<endl;

}
