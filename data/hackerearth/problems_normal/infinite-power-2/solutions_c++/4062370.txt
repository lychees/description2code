#include <bits/stdc++.h>
using namespace std;
 
int UnlimitedPower(int);
int NextUnlimitedPower(int);
int PrevUnlimitedPower(int);
int calgcd(int,int);
 
int main()
{
    int t,n;
    scanf("%d",&t);
    while(t--){
    	scanf("%d",&n);
    	if(UnlimitedPower(n)==1){
    		printf("Unlimited Power\n");
    	}
    	else{
    		int next=NextUnlimitedPower(n);
    		while(UnlimitedPower(next)!=1){
    			next=NextUnlimitedPower(next);
    		}
    		int prev=PrevUnlimitedPower(n);
            int gcd=calgcd(next-n,n-prev);
           // cout<<next<<" "<<prev<<" ";
    		printf("%d/%d\n",(next-n)/gcd,(n-prev)/gcd);
    	}
    }
    return 0;
}
 
int UnlimitedPower(int num){
	while(num>0){
		if((num%2)%2==0){
			num=num/10;
		}
		else{
			return 0;
		}
    }
	if(num==0){
		return 1;
	}
    return 0;
}
 
int NextUnlimitedPower(int num){
    int n1=0;
    int digits1[10]={};
    int i=0,j;
    while(num>0){
        digits1[i]=num%10;
        num=num/10;
        i++;
    }
 
    for(j=i-1;j>=0;j--)
        {
        if(digits1[j]%2!=0){
            digits1[j]++;
            if(digits1[j]==10){
               digits1[j]=20;
            }
            break;
        }
    }
    for(j=j-1;j>=0;j--){
        digits1[j]=0;
    }
   
    for(j=i-1;j>=0;j--){
        n1=n1*10+digits1[j];
        
    }
    return n1;
}
 
int PrevUnlimitedPower(int num){
    int n2=0;
    int digits2[10]={};
    int i=0,j;
    while(num>0){
        digits2[i]=num%10;
        num=num/10;
        i++;
    }
 
    for(j=i-1;j>=0;j--)
        {
        if(digits2[j]%2!=0){
            digits2[j]--;
            break;
        }
    }
    for(j=j-1;j>=0;j--){
        digits2[j]=8;
    }
   
    for(j=i-1;j>=0;j--){
        n2=n2*10+digits2[j];        
    }
 
    return n2;
 
}
 
int calgcd(int a,int b){
    int temp;
    while(true){
        if(a%b==0){
            return b;
        }
        else{
            temp=a%b;
            a=b;
            b=temp;
        }
    }
}
