#include <bits/stdc++.h>
#define ll long long 
using namespace std;

ll check(string str)
{
	ll n=str.size();
	ll i=0,j=n-1;
	while(i<j)
	{
		if(str[i]!=str[j])
		{
			break;
		}
		i++;j--;
	}
	if(i==j || i>j)
	return 1;
	return 0;
}
int main()
{
    ll t;
    cin>>t;
    while(t--)
    {
    	string a,b;
    	cin>>a>>b;
    	ll n=a.size(),i,j,cnt=0;

    	for(i=0;i<=n;i++)
    	{
    		string str=a.substr(0,i);

    		if(check(msg)==1)
    		cnt++;
    		
    	}

    }

}
