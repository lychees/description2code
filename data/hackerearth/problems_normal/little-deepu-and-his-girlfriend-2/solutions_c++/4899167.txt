#include <iostream>
using namespace std;
 
int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	int m,n;
    	cin >> m >> n;
    	bool dp[m];
    	int a[n];
    	for(int i=1;i<=m;i++)
    	{
    		dp[i]=0;
    	}
    	for(int i=0;i<n;i++)
    	{
    		cin >> a[i];
    		dp[a[i]]=1;
    	}
    	for(int i=1;i<=m;i++)
    	{
    		if(dp[i]==0)
    		{
    			for(int j=0;j<n;j++)
    			{
    				if(i-a[j]>=0 && dp[i-a[j]]==0)
    				{
    					dp[i]=true;
    					break;
    				}
    			}
    		}
    	}
    	if(dp[m])cout << "Little Deepu" << endl;
    	else cout << "Kate" << endl;
    	
    	
    	
    }
    return 0;
}
