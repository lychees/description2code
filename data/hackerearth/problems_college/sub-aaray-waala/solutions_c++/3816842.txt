#include <bits/stdc++.h>
using namespace std;
#define M 1000000007
typedef long long int ll;
map<ll,ll>m;
ll power(ll a,ll b)
{
	if(b==0)
	return 1;
	ll res=1;
	while (b)
	{
		if (b%2)
		{
			res*=a;
			res%=M;
		}
		a*=a;
		a%=M;
		b/=2;
	}
	return res;
}
int main()
{
    ll t,n,temp,i,count;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	count=0;
    	for(i=0;i<n;i++)
    	{
    		cin>>temp;
    		if (m.find(temp)!=m.end())
             continue;
    		else
    		m[temp]=1;
    	}
    	map<ll,ll>::iterator it;
    	count=m.size();
    	cout<<power(n,count)<<endl;
    	m.clear();
    }
    return 0;
}
