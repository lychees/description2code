    #include <bits/stdc++.h>
    #define ll long long
    using namespace std;
    int main()
    {
    char str[10010],str2[10010];
    scanf("%s",str2);
    int h[26],used[26],need[26];
    int i,j,k,l,index,x;
    memset(h,0,sizeof(h));
    l=strlen(str2);
    vector<int> v[26];
    for(i=0;str2[i]!='\0';i++)
    {
    str[i]=str2[l-1-i];
    h[str[i]-'a']++;
    v[str[i]-'a'].push_back(i);
    }
    for(i=0;i<26;i++)
    {
    used[i]=0;
    need[i]=h[i]/2;
    }
    string res="";
    int flag;
    int last=-1;
    vector<int>::iterator it;
    for(i=0;i<l/2;i++)
    {
    for(j=0;j<26;j++)
    {
    if(used[j]==need[j])
    continue;
    else
    {
    flag=1;
    it=upper_bound(v[j].begin(),v[j].end(),last);
    index=*it;
    
    for(k=0;k<26;k++)
    {
    if(k==j||used[k]==need[k])
    continue;
    else
    {
    it=upper_bound(v[k].begin(),v[k].end(),index);
    if(it==v[k].end())
    {
    x=0;
    }
    else
    {
    x=(v[k].end()-it);
    }
    if(x+used[k]>=need[k])
    {
    continue;
    }
    else
    {
   
    flag=0;
    break;
    }
    }
    }
    if(flag)
    {
   
    last=index;
    res+=(char)('a'+j);
    used[j]++;
    break;
    }
    }
    }
    }
    
    cout<<res<<endl;
    return 0;
    }
