#include <iostream>
#include <string>
#include <stdlib.h>

using namespace std;

int main()
{
	int t; // read number of lines
	string sNum; // read num
	string x ("21");
	int iNum;
	cin >> t;
	for(int i = 0; i < t; i++)
	{
		cin >> sNum;
        iNum = atoi(sNum.c_str());
		if((iNum % 21) == 0)
			cout << "The streak is broken!" << endl;
		else
		{
			//cout << sNum << endl;
			if(sNum.find(x) != (-1))
				cout << "The streak is broken!" << endl;
			else
				cout << "The streak lives still in our heart!" << endl;
		}
	}
    return 0;
}
