def main():
    t = int(raw_input().strip())
    while t != 0:
        line = raw_input().split()
        arr = map(int, line)
        n = arr[0]
        temp = arr[1]
        for i in xrange(1, n):
            temp ^= arr[i+1]
        if temp % 2 == 0:
            print "Yes"
        else:
            print "No"
        t -= 1


if __name__ == '__main__':
    main()
