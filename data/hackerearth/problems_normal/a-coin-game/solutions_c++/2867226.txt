#include<bits/stdc++.h>
using namespace std;
const int maxn = (int)1e6 + 7;
int lose[maxn];
int main(){
	int d = 0;
	for(int i=0; i<maxn; ++i){
		if(lose[i])
			continue;
		lose[i] = i + d;
		if(i+d < maxn)
			lose[i+d] = i;
		d++;
	}
	int tt;
	scanf("%d", &tt);
	while(tt--){
		int n1, n2;
		scanf("%d %d", &n1, &n2);
		if(n1 > n2)
			swap(n1, n2);
		if(n2 == lose[n1])
			puts("Don't Play");
		else
			puts("Play");

	}
}
