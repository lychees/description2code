#include "bits/stdc++.h"
using namespace std;
const int N = 1e5 + 5;
int n;
int arr[N];
int pre[N];
int suf[N];
int curl , curr;
int maxl;
int cur;
int val;
int main(){
    cin >> n;
    for(int i = 1 ; i <= n ; ++i){
        cin >> arr[i];
    }
    pre[0] = 0;
    for(int i = 1 ; i <= n ; ++i){
        pre[i] = max(pre[i - 1] , arr[i]);
    }
    suf[n + 1] = 0;
    for(int i = n ; i >= 1 ; --i){
        suf[i] = max(suf[i + 1] , arr[i]);
    }
    cur = 0;
    maxl = 0;
    for(int i = 1 ; i <= n ; ++i){
        if(arr[i] == pre[n]){
            ++cur;
        }
        else{
            cur = 0;
        }
        if(cur > maxl){
            maxl = cur;
            curl = curr = i;
        }
    }
    cout << "1";
    val = arr[curl];
    for(int i = 1 ; i < n ; ++i){
        if(pre[curl - 1] > suf[curr + 1]){
            val = min(val , arr[--curl]);
        }
        else if(pre[curl - 1] < suf[curr + 1]){
            val = min(val , arr[++curr]);
        }
        else if(arr[curl - 1] >= arr[curr + 1]){
            val = min(val , arr[--curl]);
        }
        else{
            val = min(val , arr[++curr]);
        }
        cout << (val >= max(pre[curl - 1] , suf[curr + 1]));
    }
}
