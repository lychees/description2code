#include <bits/stdc++.h>

using namespace std;

int main(int argc, char const *argv[]){
    
    int cases;
    scanf("%d", &cases);

    while(cases--) {
        string s;
        int m;
        cin >> s >> m;
        long long ans = 0;
        for(int i=0; i<s.size(); i++) {
            ans = (ans * 10 + (s[i] - '0')) % m;
        }
        printf("%lld\n", ans);
    }
    return 0;
}
