#include <iostream>
#include<algorithm>
using namespace std;
int minimum_number[101]={0};
int main() {
	// your code goes here
	int t,n;
	 for(int i=0;i<=101;i++)
	 minimum_number[i]=999;
	 minimum_number[6]=minimum_number[8]=1;
	for(int i=12;i<101;i++)
	{
	  minimum_number[i]=min(minimum_number[i-6]+1,minimum_number[i-8]+1);
	 
	
	}
	cin>>t;
	while(t--)
	{
	 cin>>n;
	 if(minimum_number[n]<=13)
	 cout<<minimum_number[n]<<endl;
	 else
	 cout<<"-1\n";
	
	}
	
	return 0;
}
