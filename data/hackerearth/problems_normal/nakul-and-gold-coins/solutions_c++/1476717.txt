#include<bits/stdc++.h>
using namespace std;
#define MAXN 1000009
#define pb push_back
char str[MAXN];
vector<int> ar;
vector<int>::iterator up,low;
int cnt;
void gen()
{
   for(int i=2; i*i<=MAXN; i++)
     {
        int k=i*i;
        for(int j=k;j<=MAXN;j+=i)
           str[j]=1;
     }
    for(int i=2; i<=MAXN-9; i++)
      if(str[i]==0)
        ar.pb(i);
}
int main()
{
    gen();
    //for(int i=0;i<=ar.size();i++)
      //  printf("%d %d\n",i,ar[i]);
    int t,l,r,ll,rr;
    scanf("%d",&t);
    cnt=ar.size();
    while(t--)
    {
      scanf("%d %d",&l,&r);
      int ans=0;
      for(int i=0;(i<cnt)&&(ar[i]<=r);i++)
      {
        ll=l/ar[i]+(l%ar[i]>0);
        ll=max(ll,ar[i]+1);
        rr=r/ar[i];
        if(rr<ll)continue;
         up=upper_bound(ar.begin(),ar.end(),rr);
         low=lower_bound(ar.begin(),ar.end(),ll);
         ans+= (int)(up-low);
         //printf("%d %d %d %d %d\n",ar[i],low-ar.begin(),up-ar.begin(),ll,rr);
      }
    printf("%d\n",ans);
   }
    return 0;
}
