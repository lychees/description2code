#include <iostream>
#include<string>
using namespace std;

int main()
{

	//FR;
	int TCs;	
	cin>>TCs;

	string t;
    getline(cin,t);
	while(TCs)
    {
		t.clear();    	
        getline(cin,t);
		int nCount=0;
		int Limit;
		int Len = t.size();
		Limit=Len/2 -1;
		int i,j;
		for(i=0,j=Len-1; i<=Limit ; i++,j-- )
		{		
				if(t.at(i) != t.at(j))
					nCount += abs(t.at(i) - t.at(j))	;		
		}
		cout<<nCount<<endl;
		TCs--;
	}
}
