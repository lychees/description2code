#include <iostream>
#include<stdio.h>
using namespace std;
#include<vector>
#include<stdlib.h>
int main() {
	// your code goes here
	int t,n;
	long long x,y;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		int i;
		vector <int> v;
		vector <int> q;
		for(i=0;i<n;i++)
		{
			scanf("%lld",&x);
			v.push_back(x);
			//q.push_back(y);
		}
		for(i=0;i<n;i++)
		{
			scanf("%lld",&y);
			//v.push_back(x);
			q.push_back(y);
		}
		long long fx,fy,c1,c2,tem,ans=-1;
		scanf("%lld%lld",&fx,&fy);
		scanf("%lld%lld",&c1,&c2);
		ans=(abs(fx)+abs(fy))*c1;
		for(i=0;i<n;i++)
		{
			tem=(abs(v[i])+abs(q[i]))*c1+(abs(fx-v[i])+abs(fy-q[i]))*c2;
			if(tem<ans || ans==-1) ans=tem;
		}
		printf("%lld\n",ans);
	}
	return 0;
}
