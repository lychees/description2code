#include<iostream>
using namespace std;
unsigned long int rev(unsigned long int x)
{
    unsigned long int r,s=0;
    while(x!=0)
    {
        r=x%10;
        s=s*10+r;
        x/=10;
    }
    return s;
}
int main()
{
    int t;
    unsigned long int a,b,s;
    cin>>t;
    while(t--)
    {
        cin>>a>>b;
        a=rev(a);
        b=rev(b);
        s=a+b;
        s=rev(s);
        cout<<s<<endl;
    }
    return 0;
}
