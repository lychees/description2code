#include <bits/stdc++.h>
using namespace std;
#include<algorithm>
int main()
{
    int n;
    cin>>n;
    int sum=0;
    int a[n];
    for(int i=0;i<n;i++)
    {
    	cin>>a[i];
    	sum=sum+a[i];
    }
    int y,i;
    int count=0;
    int p=(1<<n);
    for(y=0;y<p;y++)
    {
    	sort(a,a+n);
    	do
    	{
    		int flag=1;
    		int c=0;
    		for(i=0;i<n;i++)
    		{
    			if(y&(1<<i))
    				c=c+a[i];
    			else
    				c=c-a[i];
    			if(c<0)
    			{
    				flag=0;
    				break;
    			}
    		}
    		if(flag)
    			count++;
    	}while(next_permutation(a,a+n));
    }
   	cout<<count<<" "<<sum<<endl;
    if(count<=sum)
    {
    	cout<<"Got no way out!"<<endl;
    }
    else
    {
    	cout<<"We will win!"<<endl;
    }
    return 0;
}
