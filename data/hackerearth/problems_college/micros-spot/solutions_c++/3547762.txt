#include <bits/stdc++.h>

#define ll long long

using namespace std;

ll a[1010][1010],b[1010][1010];
int main() {
	ll t;
  	scanf("%lld",&t);
	while(t--)
	{
		ll c[1010][1010]={0};
		ll m,n;
		ll maxc[1010]={0};
		ll maxr[1010]={0};
		scanf("%lld %lld",&n,&m);
		
		for(int i=0;i<n;i++)
			for(int j=0;j<m;j++)
				scanf("%lld",&a[i][j]);
		
		for(int i=0;i<n;i++)
			for(int j=0;j<m;j++)
				scanf("%lld",&b[i][j]);
			  
		for(int i=0;i<m;i++)
		{
			for(int j=0;j<n;j++)
			{
				if(maxc[i]<a[j][i])
				 maxc[i]=a[j][i];
			}
		//	cout<<i<<"=i "<<maxc[i]<<endl;
		}
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<m;j++)
			{
			
				if(maxr[i]<b[i][j])
				 maxr[i]=b[i][j];
			}
		//	cout<<i<<"=i "<<maxr[i]<<endl;
		}
	
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<m;j++)
			{
				if(maxr[i]==b[i][j])
				 {c[i][j]++;}
			}
		}
		
		for(int i=0;i<m;i++)
		{
			for(int j=0;j<n;j++)
			{
				
				if(maxc[i]==a[j][i])
				{ c[j][i]++;}
			}
		}
		ll cnt=0;
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<m;j++)
			{
				
				if(2==c[i][j])
				 cnt++;
			}
		}
		cout<<cnt<<endl;
	}
	
	return 0;
}
