#define md 1000000007ll

ll l,r,n,fc[111],d[111],c[111][111];

int main()
{
	I_O;
	
	cin>>n>>l>>r;
	
	fc[0]=1;
	for(int i=1;i<=101;i++){
		fc[i]=(fc[i-1]*i)%md;
	}
	
	for(int i=0;i<=n;i++){
		for(int j=0;j<=i;j++){
			if(i==j || j==0){
				c[i][j]=1;
			}
			else{
				c[i][j]=(c[i-1][j]+c[i-1][j-1])%md;
			}
		}
	}
	
	d[0]=1;
	d[1]=0;
	for(int i=2;i<=n;i++){
		d[i]=(i-1)*(d[i-1]+d[i-2])%md;
		d[i]%=md;
	}
	ll anukul=0;
	ll total=fc[n];
	for(int i=l;i<=r;i++){
		anukul+=(d[i]*c[n][i])%md;
		anukul%=md;
	}
	
	cout<<(anukul*MODINV(total,md))%md<<endl;
	return 0;
}
