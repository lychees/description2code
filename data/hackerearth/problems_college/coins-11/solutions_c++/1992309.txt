#include <iostream>
#include<math.h>
using namespace std;
int main()
{
    int t,n,k,i;
    cin>>t;
    while(t--)
    {
    	k=0;
    	cin>>n;
    	if(n==3)
    	cout<<"3\n";
    	else if(n==1)
    	cout<<"1\n";
    	else
    	{
    	for(i=2;i<=pow(n,0.5);i++)
    	{
    		if(n%i==0)
    		{
    			k=1;break;
    		}
    	}
    	if(k==1)
    	cout<<"1\n";
    	else cout<<"2\n";
    }
    }
    return 0;
}
