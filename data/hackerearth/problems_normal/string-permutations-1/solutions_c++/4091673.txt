#include <iostream>
#include <map>
using namespace std;
typedef long long ll;
int main()
{
	ios_base::sync_with_stdio(false);
	ll t;
	cin >> t;
	ll max = 0;
	ll count = 0;
	map<string, ll> hash;
	while(t--)
	{
		string name;
		cin >> name;
		hash[name]++;
		
		if(hash[name] == max)
		{
			count++;
		}
		else if(hash[name] > max)
		{
			max = hash[name];
			count = 1;
		}
	}
	cout << count * max << endl;
	return 0;
}
