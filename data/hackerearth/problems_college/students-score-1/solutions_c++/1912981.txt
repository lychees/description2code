#include <iostream>
using namespace std;
long ar[100100],s[100100];
int main()
{
    long i,n,j,ans;
    cin>>n;
    s[0]=0;
    ans=-1;
    for(i=1;i<=n;i++)
    {
    	cin>>ar[i];
    	s[i]=s[i-1]+ar[i];
    }
    for(i=1;i<=n;i++)
    {
    	for(j=i+1;j<=n;j++)
    	{
    		ans=max((s[j]-s[i-1]),ans);
    	}
    }
    if(ans>0)
    cout<<ans<<endl;
    else
    cout<<0<<endl;
    return 0;
}
