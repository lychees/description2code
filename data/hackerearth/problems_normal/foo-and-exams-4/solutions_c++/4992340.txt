 #include<iostream>
    #include<cmath>
    using namespace std;
    int main()
    {
    	unsigned long k,a,b,c,d,t,n;
    	cin>>t;
    	while(t--)
    	{
    		cin>>a>>b>>c>>d>>k;
    		n=(int)floor(cbrt((float)((k)/a)))+1; 
    		while(n--)						
    		if((d+(n)*(c+(n)*(b+a*(n))))<=k||n==0) 
    		{
     		cout<<n<<"\n";					
    		break;					
     		}
    	}
        return 0;
    }
