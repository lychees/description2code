#include <iostream>
using namespace std;

int main()
{
    unsigned long long int i=0, j=0, t=0,n=0, sum=0;
    cin>>t;
    while(t-- > 0)
    {
    	cin>>n;
    	sum=0;
    	for(i=2; i<n;i=i*2)
    	{
    		for(j=1; j<i; j=j*2)
    		{
    			if((i+j)<=n)
    			{
    				sum = (sum+i+j)%1000000007;
    			}
    		}
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
