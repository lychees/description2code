#include<bits/stdc++.h>
using namespace std;
map<string,bool> m;
int a,h;
int n;
void cal(string s)
{
	if(m[s])
	return;
	else
	{
		m[s]=1;
		string s1=s;
		int t;
		for(int i=0;i<n;i++)
		{
			if(i%2==1)
			{
				t=s[i]-'0';
				t=(t+a)%10;
				s1[i]=t+'0';
			}
		}
		cal(s1);
		s1=s;
		for(int i=0;i<n;i++)
		s1[i]=s[(i+h)%n];
		cal(s1);
		return;
	}
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string s;
		cin>>s;
		n=s.length();
		cin>>a>>h;
		cal(s);
		cout<<m.begin()->first<<endl;
		m.clear();
	}
	return 0;
}
