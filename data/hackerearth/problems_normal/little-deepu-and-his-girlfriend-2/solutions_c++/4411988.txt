#define ll long long int
#define maxN 100
#define maxVal 100000000
#define minVal -100000000
#define mod 1000000007LL

#define gcd(a,b) __gcd(a,b)

using namespace std;

int n;
int a[maxN+5];
int dp[maxN*maxN+5];

int compute(int i)
{
	if(dp[i]!=-1)
		return dp[i];
	int j,ans;
	vector<int> g;
	for(j=0;j<n;j++)
	{
		if((i-a[j])>=0)
			g.push_back(compute(i-a[j]));
	}
	ans=0;
	sort(g.begin(),g.end());
	for(j=0;j<(int)g.size();j++)
		if(g[j]==ans)
			ans++;
		else
			break;
	return dp[i]=ans;
}

int main()
{
	ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int t,i,j,m;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d%d",&m,&n);
    	for(i=0;i<n;i++)
    		scanf("%d",&a[i]);
    	memset(dp,0,sizeof(dp));
    	for(i=1;i<=m;i++)
    	{
    		for(j=0;j<n;j++)
    		{
    			if((i-a[j])>=0&&dp[i-a[j]]==0)
    			{
    				dp[i]=1;
    				break;
    			}
    		}
    	}
    	if(dp[m])

}
