#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int T;
    cin >> T;
    
    while(T--)
    {
    	int key;
    	char s[1000];
    	
    	cin >> key;
    	scanf("%s",&s);
    	
    	for(int i=0 ; i<strlen(s) ; i++)
    	{
    		if (islower(s[i]))
    		s[i]=(s[i]-'a' + key) % 26 + 'a';
    		if (isupper(s[i]))
    		s[i]=(s[i]-'A' + key) % 26 + 'A';
    	}
    	printf("%s\n",s);
    }
    return 0;
}
