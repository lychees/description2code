#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define set_bits(x) __builtin_popcount(x)
#define sc1ll(x) scanf("%lld",&x)
#define sc2ll(x,y) scanf("%lld%lld",&x,&y)
#define sc3ll(x,y,z) scanf("%lld%lld%lld",&x,&y,&z)
struct node
{
    ll price,wt;
};
vector<node> arr(21);
ll find_max_weight(ll num,ll weight,ll items)
{
    ll max_wt,ans=-1,i,j,maxx=-1,cnt,price;
    ll total=(1<<num);
    for(ll i=0;i<total;i++)
    {
        cnt=set_bits(i);
        if(cnt<=items)
        {
            max_wt=0;
            price=0;
            for(ll j=0;j<num;j++)
            {
                if(i&(1<<j))
                {
                    max_wt=max_wt+arr[j].wt;
                    price=price+arr[j].price;
                }
            }
            if(max_wt<=weight && max_wt!=0)
                ans=max(ans,price);
        }
    }
    return ans;
}
int main()
{
    ll ans,weight,cnt,test,i,j,num;
    sc1ll(test);
    for(j=1;j<=test;j++)
    {
        sc1ll(num);
        for(i=0;i<num;i++)
            sc2ll(arr[i].price,arr[i].wt);
        sc2ll(weight,cnt);
        ans=find_max_weight(num,weight,cnt);
        printf("For Day #%lld:\n",j);
        printf("%lld\n",ans);
    }
    return 0;
}
