#include<bits/stdc++.h>
#define ll long long
using namespace std;

int main(void)
{
	ll t,n,val,mx;
	scanf("%lld",&t);
	while(t--)
	{
		mx=-1;
		map<ll,ll> m;
		map<ll,ll> :: iterator it;
		scanf("%lld",&n);
		while(n--)
		{
			scanf("%lld",&val);
			m[val]++;
		}
		for(it=m.begin();it!=m.end();it++)
		{
			if(it->second > mx)
				mx=it->second;
		}
		printf("%lld\n",mx);
	}
}
