#include "cstdio"
int a,b,l,c,d,e,f,w1,w2;
bool ok;
int main()
{
	scanf ("%d%d", &a, &l);
	
	for (b=a+1;;b++)
	{
		c=a,d=b;
		ok=true;
		for (int i=0; i<l; i++)
		{
			e=c+i;
			f=d+i;
			w1=0,w2=0;
			while (e>0)
			{
				if (e%10==4 || e%10==7) w1++;
				e/=10;
			}
			while (f>0)
			{
				if (f%10==4 || f%10==7) w2++;
				f/=10;
			}
			if (w1!=w2)
			{
				ok=false;
				break;
			}
		}
		if (ok)
		{
			printf ("%d\n", b);
			return 0;
		}
	}

	return 0;
}
