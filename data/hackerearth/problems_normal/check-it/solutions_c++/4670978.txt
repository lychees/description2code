#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		int n,m;
		cin >> n >> m;
		int x,y;
		pair<int,int> p[m+1];
		for(int i=1;i<=m;i++)
		{
			cin >> x >> y;
			p[i]=make_pair(x,y);
		}
		vector<int> vec(n-1);
		for(int i=0;i<n-1;i++)
			cin >> vec[i];
		set<int> s;
		for(int i=0;i<vec.size();i++)
		{
			s.insert(p[vec[i]].first);
			s.insert(p[vec[i]].second);
		}
		if(s.size()==n)
			cout << "YES" << endl;
		else
			cout << "NO" << endl;
	}
	return 0;
}
