#include<iostream>
#include<stdio.h>
#include<string.h>
#include<math.h>

using namespace std;

int main()
{

    int t,n,m,i,p,sum,ans,kk;
    double q;
    scanf("%d",&t);
    for(int kk1=1;kk1<=t;kk1++)
    {
        scanf("%d %d",&n,&m);
        sum=0;
        p=1;
        q=0.0;
        if(n==1)
        {
            cout<<"0"<<endl;
            continue;
        }

        if(n>=m)
        {
            for(i=1;;i++)
            {
                p=p*2;
                if(p>=m)
                    break;
            }


            if(p>m)
            {
                p=p/2;
                sum=i+1;
                ans=p+p;



                if(ans<n)
                {
                    kk=n-ans;
                    q=(double)(kk)/(double)m;
                    sum=sum+ceil(q);

                }




            }
            else if(p==m)
            {
               sum=i+1;
               ans=p;

               if(ans<n)
               {
                   kk=n-ans;
                   q=(double)(kk)/(double)m;
                   sum=sum+ceil(q);
               }

            }


        }

        else
        {
            for(i=1;;i++)
            {
                p=p*2;
                if(p>=n)
                    break;
            }

            sum=i+1;
        }

        cout<<sum<<endl;


    }



}
