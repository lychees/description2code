string = raw_input()
length = len(string)
a = input()
m = []
count = 0
while (count < a):
    m.append(raw_input())
    count += 1
count = 0
while (count < a):
    put = m[count]
    cac = put.split(" ")
    first = int(cac[0])
    last = int(cac[1])
    first = first % length
    last = last % length
    if (string[first - 1] == string[last - 1]):
        print 'Yes'
    else:
        print 'No'
    count += 1
