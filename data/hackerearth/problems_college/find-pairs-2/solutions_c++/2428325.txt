#include <iostream>
#include <cstdio>

using namespace std;

int main()
{
 	int t; long int n,i,j,ans;
     
 	scanf("%d",&t);
 	while(t--)
 	{
 		scanf("%ld",&n);
 		long int a[n];
 		for(i=0;i<n;++i)
 		{
 			scanf("%ld",&a[i]);
 			
 		}
 		ans=0;
 		for(i=0;i<n;++i)
 		{
 			for(j=i;j<n;++j)
 			{
 				if(a[i]==a[j])
 				{
 					ans++;
 				}
 			}
 			
 		}
 		
 		printf("%ld\n",ans);
 		
 		
 	}
 	
 	
    return 0;
}
