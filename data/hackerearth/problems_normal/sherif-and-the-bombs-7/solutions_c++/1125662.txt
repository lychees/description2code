#include <iostream>
#include <math.h>
#include <stdio.h>
using namespace std;
int isInsideCircle(int c1,int c2,int r, int p ,int q)
{
	p = p - c1;
	q = q - c2;
	if(p*p + q*q <= r*r )return 1;
	else
		return 0;
}
int main()
{
	int n = 0;
	scanf("%d",&n);
	int cord[n][3];
	int k = 0;
	while(k < n)
	{
		int x,y,r;
		scanf("%d",&x);
		scanf("%d",&y);
		scanf("%d",&r);
		cord[k][0] = x;
		cord[k][1] = y;
		cord[k][2] = r;
		++k;
	}
	
	int m,l;
	int count = 0;
	for(m = 1;m<1001;m++)
	{
		for(l=1;l<1001;l++)
		{
			k = 0;
			int isval = 0;
			int flag = 1;
			for(k;k<n && flag;k++)
			{
				int a = cord[k][0];
				int b = cord[k][1];
				int c = cord[k][2];
				if(isInsideCircle(a,b,c,m,l)){
					++isval;
					if(isval == 2){
						flag=0;
						++count;
					}
					
				}
			}
		}
	}
	printf("%d\n",count);
	return 0;
}
