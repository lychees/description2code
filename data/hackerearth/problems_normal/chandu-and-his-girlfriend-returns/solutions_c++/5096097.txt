#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--){
    int n,m;
    cin>>n>>m;
    int a[n],b[m],c[n+m],i,j,p,k;
    for(i=0;i<n;i++)
        cin>>a[i];
    for(i=0;i<m;i++)
        cin>>b[i];
        i=0,j=0,k=0;
     while(i<n && j<m)
     {
         if(a[i]>=b[j])
         {
            c[k++]=a[i];
            i++;
         }
         else
          {
            c[k++]=b[j];
            j++;
          }
     }
     if(i>=n)
     {
         for(p=j;p<m;p++)
            c[k++]=b[p];
     }
     else if(j>=m)
     {
        for(p=i;p<n;p++)
            c[k++]=a[p];
     }
     for(i=0;i<(n+m);i++)
        cout<<c[i]<<" ";
        cout<<"\n";
    }
    return 0;
}
