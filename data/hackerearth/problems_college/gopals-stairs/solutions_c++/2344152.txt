#include <iostream>
using namespace std;
#define MOD 1000000007
long long int a[100000]={0};
int main()
{
    int t,n;
    a[0]=1,a[1]=2,a[2]=4;
    for(int i=3;i<100000;i++)
    {
    	a[i]=(a[i-1]%MOD+a[i-2]%MOD+a[i-3]%MOD)%MOD;
    }
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	cout<<a[n-1]%(MOD)<<endl;
    }
}
