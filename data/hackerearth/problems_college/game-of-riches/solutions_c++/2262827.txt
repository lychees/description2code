#include<iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long t,nr,ng,im,a[101],b[101],i;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&nr);
		scanf("%lld",&ng);
		for(i=0;i<nr;i++)
		{
			scanf("%lld",&a[i]);
			
		}
		im=a[0];
		sort(a,a+nr);
		long long j=1,impo=0,count=0;
		for(i=0;i<nr;i++)
		{
			
			b[i]=j;
			j++;
			if(j>ng)
			{
				j=1;
			}
		}
		
		for(i=0;i<nr;i++)
		{
			if(a[i]==im)
			{
				impo=b[i];
			}
			
			
		}
		for(i=0;i<nr;i++)
		{
			if(b[i]==impo&&a[i]>im)
			{
				count++;
			}
		}
		printf("%lld\n",count);
		
	}
	
	
	
	
	return 0;
}
