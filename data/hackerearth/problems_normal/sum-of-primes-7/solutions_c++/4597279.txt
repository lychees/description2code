#include<iostream>
#include<stdlib.h>
#define SIZE 1000000
using namespace std;
int *prime;
long long *sum;
void generateSieve()
{
    prime[0]=prime[1]=1;
    int i=2,j;
    while(i<=SIZE)
    {
        j=2*i;
        while(j<=SIZE){
            prime[j]=1;
            j+=i;
        }
        i++;
        while(prime[i])
            i++;
    }
    i=2;
    sum[0]=sum[1]=0;
    while(i<=SIZE){
        sum[i]=sum[i-1];
        if(!prime[i])
            sum[i]+=(long long)i;
        i++;
    }
}
int main()
{
    prime=(int *)calloc(SIZE+1,sizeof(int));
    sum=(long long  *)calloc(SIZE+1,sizeof(long long));
    generateSieve();
    int i,l,r,n;
    cin>>n;
    i=0;
    while(i<n)
    {
        cin>>l>>r;
        if(!prime[l])
            cout<<sum[r]-sum[l]+(long long)l<<endl;
        else cout<<sum[r]-sum[l]<<endl;
        i++;
    }
    return 0;
}
