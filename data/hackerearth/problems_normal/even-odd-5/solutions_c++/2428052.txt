#include <iostream>
using namespace std;

int main()
{
    long long int t,r,c;
    long long int sum1,sum2;
    cin>>t;
    while(t--)
    {
    	cin>>r>>c;
    	sum1=0;
    	sum2=0;
    	if(r>c)
    	{
    		if((r-c)%2==0)
    		{
    			sum1=((r-c)/2)*c;
    			sum2=((r-c)/2)*c;
    		}
    		else if(c%2==0)
    		{
    				sum1=((r-c)/2)*c;
    			sum2=((r-c)/2)*c + c;
    		}
    		else if(c%2!=0)
    		{
    				sum2=((r-c)/2)*c;
    			sum1=((r-c)/2)*c + c;
    		}
    		r=c;
    	}
    	if(r%2==0)
    	{
    		sum1=sum1+(r/2)*((2+r)/2);
    		sum2=sum2+(r/2)*(r/2);
    	}
    	else
    	{
    		sum2=sum2+((r+1)/2)*((1+r)/2);
    		sum1=sum1+(r/2)*((1+r)/2);
    		
    	}
    	cout<<sum1<<" "<<sum2<<"\n";
    }
    return 0;
}
