#include <iostream>
using namespace std;

int main()
{
    string s;
    cin>>s;
    long long one=0,oneZero=0,ans=0;
    int n=s.size();
    for(int i=0;i<n;i++){
    	if(s[i]=='0'){
    		oneZero+=one;
    	}
    	else{
    		ans += oneZero;
    		one++;
    	}
    	
    //	cout<<one<<" "<<oneZero<<endl;
    }
    cout<<ans;
    return 0;
}
