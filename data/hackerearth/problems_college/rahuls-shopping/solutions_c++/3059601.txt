int arr[1000006];

int main()
{
	int i, j;

	int n;

	cin>>n;

	fl(i,0,n)
		cin>>arr[i];

	int ans = 0;
	
	int l, b;
	
	cin>>l>>b;

	fl(i,0,n)
	{
		int temp = arr[i];

		int s = 0;

		while(temp)
		{
			s += temp%10;
			temp/=10;
		}
		
		//cout<<s;nline;

		int cost = s*arr[i]*l*b;

		if(arr[i]>=1 && arr[i]<=10)
		{
			cost = int((double)cost - (cost/4.0));
		}
		else if(arr[i]>=11 && arr[i]<=20)
		{
			cost = int((double)cost - (cost*3.0/10.0));
		}
		else if(arr[i]>=21 && arr[i]<=50)
		{
			cost = int((double)cost - (cost/5.0));
		}
			
		//cout<<cost; nline;

		ans += cost;
	}


	cout<<ans;

	return 0;
}
