#include <iostream>
#include <map>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	int count = n;
    	map<int,int> m;
    	for(int i=0;i<n;i++){
    		int c;
    		cin>>c;
    		if(m[c]){
    			count+=m[c];
    		}
    		m[c]++;
    	}
    	cout<<count<<endl;
    }
    return 0;
}
