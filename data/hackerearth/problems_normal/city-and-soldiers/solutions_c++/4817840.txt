using namespace std;


int root(int a,int id[])
{
    while(a!=id[a])
    {
        a=id[a];
    }
    return a;
}
int un(int a,int b,int id[])
{
    int x=root(a,id);
    int y=root(b,id);
    id[x]=id[y];
}


int main()
{
    int n,q;
    cin>>n>>q;
    int id[n+1];
    for(int i=0;i<=n;i++)
    {
        id[i]=i;
    }
    for(int i=0;i<q;i++)
    {
        int x;
        cin>>x;
        if(x==1)
        {
            int y,z;
            cin>>y>>z;
            un(y,z,id);
        }
        else if(x==2)
        {
            int y;
            cin>>y;
            int z=root(y,id);
            id[z]=y;
            id[y]=y;

        }
        else{
            int y;
            cin>>y;
            cout<<root(y,id)<<endl;
        }
    }

return 0;
}
