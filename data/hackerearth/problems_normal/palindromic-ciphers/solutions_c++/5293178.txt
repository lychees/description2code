#include<bits/stdc++.h>
using namespace std;
int main ()
{
	long long int T,i,k,flag,pro=1;
	cin>>T;
	string str;	
	while(T--)
	{flag=0;
	pro=1;
	cin>>str;
	for(i=0,k=str.length()-1;i<str.length()/2;++i,k--)
	{
	if(str[i]!=str[k])
	{flag=1;break;}
	}
	if(flag==0)
	cout<<"Palindrome\n";
	if(flag==1)
	{
		for(i=0;i<str.length();++i)
		{pro=pro*(int)((str[i]-'a')+1);}
		cout<<pro<<endl;
	}
	}
	return 0;
	}
