#include <bits/stdc++.h>
using namespace std;

	
int main()
{	int i,j,N,M,Q,flag,down,up,mid;
	long long Y;
    cin>>N>>M;
    long long MP[N][M];
    
    for(i=0;i<N;i++)
    	for(j=0;j<M;j++)
    		 cin>>MP[i][j];
    	 	
    cin>>Q;
    for(i=0;i<Q;i++)
    	{
    		cin>>Y;
    		flag=0;
    		if(!((Y<MP[0][0])||(Y>MP[N-1][M-1])))
    			
    		
    		for(j=0;j<N;j++)
    		{		
    				down=0;
    				up=M-1;
    				mid=(up+down)/2;
    				while(down<=up)
    				{
    			 		if(MP[j][mid]==Y)
    						{
    							flag=1;
    				 			break;
    						}
    					if(MP[j][mid]>Y)
    						{
    							up=mid-1;
    							mid=(up+down)/2;
    						}
    					else
    						{
    							down=mid+1;
    							mid=(up+down)/2;
    						}
    				}
    				if(flag==1)
    					break;
    		}
	if(flag==1)
		cout<<j<<" "<<mid<<endl;
			
	else
		cout<<-1<<" "<<-1<<endl;
    }
    	
return 0;
}
