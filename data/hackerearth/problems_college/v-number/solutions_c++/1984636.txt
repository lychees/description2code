#include <iostream>
using namespace std;

int main()
{   ios::sync_with_stdio(0);
    int t;
    cin>>t;
    while(t--){
    	string s;
    	cin>>s;
    	long long int a=0,b=0;
    	for(int i=0,j=s.size()-1;i<j;i++,j--)
    	{
    		a+=s[i]-'0'; b+=s[j]-'0';
    	}
    	if(a==b) printf("YES\n"); else printf("NO\n");
    }
    return 0;
}
