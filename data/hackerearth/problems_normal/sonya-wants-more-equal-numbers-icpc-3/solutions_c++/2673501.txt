#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define MAXN 105
ll A[MAXN];

ll getSum(int l,int r)
{
	if(l==0)
		return A[r];
	else
		return A[r]-A[l-1];
}

ll getVal(int i)
{
	if(i>=0)
		return A[i];
	else
		return 0;
}

int main()
{
	int n;
	scanf("%d",&n);
	for(int i=0;i<n;++i)
		scanf("%lld",A+i);
	A[n] = 1e9;
	for(int i=1;i<n;++i)
		A[i] += A[i-1];
	int res = 0;
	set<ll> st;
	for(int i=0;i<n;++i)
		for(int j=i;j<n;++j)
		{
			ll tar = getSum(i,j);
			int cnt = 0;
			for(int k=0;k<n;)
			{
				ll val = *lower_bound(A,A+n,tar+getVal(k-1));
				if(val==tar+getVal(k-1))
				{
					++cnt;
					k = lower_bound(A,A+n,tar+getVal(k-1))-A;
				}
				++k;
			}
			if(cnt>res)
			{
				res = cnt;
				st.clear();
				st.insert(tar);
			}
			else if(cnt==res)
			{
				st.insert(tar);
			}
		
		}
	cout << res << endl;
	for(set<ll>::iterator it=st.begin();it!=st.end();++it)
		cout << *it << " ";
	return 0;
}
