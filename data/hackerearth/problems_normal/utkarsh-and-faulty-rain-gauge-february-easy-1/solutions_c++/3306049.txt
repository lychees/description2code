#include <bits/stdc++.h>
using namespace std;

int main()
{
    int T;
    scanf("%d", &T);
    
    double ans = 1;
    
    for(int i=1; i<=T; i++)
    {
    	int N, R;
    	scanf("%d %d", &N, &R);
    	
    	double a = (double)6*R*R/N/(N+1)/(2*N+1);
    	double sum = 0;
    	
    	for(int i=1; i<=N; i++)
    		sum+= (double)i/R*a*i*i/R/R;
    	
    	ans*= sum;
    }
    
    printf("%.4lf\n", ans);
    
    return 0;
}
