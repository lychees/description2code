#include<iostream>
#include<bits/stdc++.h>

using namespace std;

int a[1000000]={0};
int main ()
{
	ios::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		int i,j,k=0,l=0,q,n,z,x,c;
		 
		
		cin>>n;
		for(i=0;i<n;i++)
		cin>>a[i];
	for(i=n-1;i>0;i--)
		{
			if(a[i-1]>a[i])
			{
			
				
			    for(j=n-1;j>=i;j--)
				{
					if(a[j]<a[i-1]&&a[j]>=a[i])
					{
						c=a[i-1];
						a[i-1]=a[j];
						a[j]=c;
						l=1;
					
						for(k=i;k<n;k++)
					{
						for(q=k+1;q<n;q++)
						{
							if(a[k]<a[q])
							{
							  z=a[k];
							  a[k]=a[q];
							  a[q]=z;	
							  
							}
						}
						
					}
					
					
						
					}
					if(l==1)
					break;
						
				}	
			
			}
			if(l==1)
     		  break;
		}
		for(i=0;i<n;i++)
		cout<<a[i]<<" ";
		cout<<endl;
	}
	return 0;
}
