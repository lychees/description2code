#include <bits/stdc++.h>
#define _ ios_base::sync_with_stdio(0);cin.tie(0);

using namespace std;
long long x[1000001]={0};
long long y[1000001]={0};
long long f[1000001]={0},g[1000001]={0};
int main() {

 
    for(int i=2;i<=1000000;i++){
        for(int j=i;j<=1000000;j=j+i){
            if(x[j]==0) x[j]=i;
            y[j]=y[j]+i;
        }
    }
   
    for(int i=2;i<=1000000;i++){
        f[i]=f[i-1]+x[i];
        g[i]=g[i-1]+y[i]-i+1;
    }
    int t,n;
    scanf("%d",&t);
    
    while(t--){
        scanf("%d",&n);
        printf("%d\n",(f[n]+g[n])%n);
    }
    
    return 0;
}
