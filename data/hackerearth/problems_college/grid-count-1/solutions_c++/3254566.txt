#include<iostream>
#include<cstdio>
#include<cmath>
#include<map>
using namespace std;
#define ll long long int
#define MAX 1005
map<ll,bool> hm;
int countSum(ll *arr[MAX], int n, int k){
   int counts=0;
   //

   for(int i=1;i<=465;i++){
         hm[i*i*i]=1;
   }

   //
  ll **sum1=new ll*[n];
  for(int i=0;i<n;i++){
    sum1[i]=new ll[n];
  }
   for(int i=0;i<n;i++){
    sum1[i][0]=arr[i][0];
   for(int j=1;j<n;j++){
     sum1[i][j]=sum1[i][j-1]+arr[i][j];
   }
  }

ll **sum2=new ll*[n];
  for(int i=0;i<n;i++){
    sum2[i]=new ll[n];
  }
  for(int i=0;i<n;i++){
    sum2[0][i]=sum1[0][i];
   for(int j=1;j<n;j++){
     sum2[j][i]=sum2[j-1][i]+sum1[j][i];
   }
  }

  for(int i=0;i<n;i++){
   for(int j=0;j<n;j++){
     if( (i+k <= n) && (j+k <= n)){
      ll val;
        if(i!=0 && j!=0){
             val=sum2[i+k-1][j+k-1] - sum2[i-1][j+k-1] - sum2[i+k-1][j-1] + sum2[i-1][j-1];
        }
         if(i!=0 && j==0){
             val=sum2[i+k-1][j+k-1] - sum2[i-1][j+k-1];
        }
         if(i==0 && j!=0){
             val=sum2[i+k-1][j+k-1] - sum2[i+k-1][j-1];
        }
         if(i==0 && j==0){
             val=sum2[i+k-1][j+k-1];
        }
        if(hm[val])
          counts++;
     }
   }
  }

  return counts;
}

int main(){
  int n,k;
  scanf("%d%d",&n,&k);
  ll **arr=new ll*[n];
  for(int i=0;i<n;i++){
    arr[i]=new ll[n];
  }
  for(int i=0;i<n;i++){
   for(int j=0;j<n;j++){
      scanf("%lld",&arr[i][j]);
   }
  }
  printf("%d\n",countSum(arr,n,k));
  return 0;
}
