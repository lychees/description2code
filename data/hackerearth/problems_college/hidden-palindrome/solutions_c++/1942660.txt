#include <bits/stdc++.h>

using namespace std;

int longPalin ( char str[] )
{
	int n = strlen(str);
	bool table[n][n];
	memset(table,0,sizeof(table));
	
	for ( int i=0; i<n; i++)
	{
		table[i][i]=true;
	}
	
	int max=1;
	int ini=0;
	
	for ( int i = 0 ; i<n-1; i++ )
	{
		if ( str[i]==str[i+1] )
		{
			table[i][i+1]=true;
			ini=i;
			max=2;
		}
	}
	
	for ( int k=3; k<=n; k++ )
	{
		for ( int i=0; i<n-k+1; i++ )
		{
			int j = i+k-1;
			
			if ( table[i+1][j-1] && str[i]==str[j] )
			{
				table[i][j]=true;
				if(k>max)
				{
					ini=i;
					max=k;
				}
			}
		}
	}
	return max;
}

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		char str[21];
		cin>>str;
		int z = longPalin(str);
		cout<<z<<endl;
	}
    
    return 0;
}
