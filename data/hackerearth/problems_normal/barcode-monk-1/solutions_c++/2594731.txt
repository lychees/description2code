#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
int main()
{
    int n;
    int k;
    scanf("%d%d",&n,&k);
    int i,j;
    int brushes[k];
    for(i=0;i<k;i++)
        scanf("%d",&brushes[i]);
    bool superbrush[n+1];
    for(i=0;i<=n;i++)
        superbrush[i]=0;
    int s;
    superbrush[0]=1;
    for(i=0;i<k;i++)
    {
        s=brushes[i];
        for(j=s;j<=n;j++)
        {
            superbrush[j]=superbrush[j]|superbrush[j-s];
        }
    }
    //for(i=0;i<=n;i++)
       // cout<<superbrush[i]<<" ";
    long long u[n+1],p[n+1];
    for(i=0;i<=n;i++)
            u[i]=p[i]=0;
    u[0]=1;
    for(i=1;i<=n;i++)
    {
        u[i]=(p[i-1]+u[i-1])%mod;
        for(j=1;j<=i;j++)
        {
            if(superbrush[j]==1)
                p[i]=(p[i]+u[i-j])%mod;
        }
        //cout<<u[i]<<" "<<p[i]<<endl;
    }

    long long ans=0;
    ans=(p[n]+u[n])%mod;
    cout<<ans;
    return 0;
}
