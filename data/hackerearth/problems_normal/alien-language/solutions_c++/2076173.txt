#include <iostream>
#include <string>
using namespace std;

int main() {
    int t;
    string text, pattern;
    bool textC[26], patternC[26], found;
    long textL, patternL, i;
    cin>>t;
    while(t--) {
    	found = false;
    	for(i=0; i<26; i++) {
    		textC[i] = patternC[i] = false;
    	}
    	cin>>text>>pattern;
    	textL = text.length();
    	patternL = pattern.length();
    	for(i=0; i< textL; i++) {
    		textC[text[i] - 'a'] = true;
    	}
    	for(i=0; i< patternL; i++) {
    		patternC[pattern[i] - 'a'] = true;
    	}
    	for(i=0; i<26; i++) {
    		if(textC[i] && patternC[i]) {
    			found = true;
    			break;
    		}
    	}
    	cout<<(found?"YES":"NO")<<endl;
    }
    return 0;
}
