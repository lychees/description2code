using namespace std;

int dp[20000005];

void init(){
	int x;
	for(int i=2 ; i<=20000001 ; i++){
		x = dp[i-1];
		if(i%2==0) x= min(x,dp[i/2]);
		if (i%3==0) x= min(x,dp[i/3]);
		dp[i]=x+1;
	}
}
int main()
{
	int t,n;
	dp[1]=0;
	init();
	cin>>t;
	while(t--){
		cin>>n;
		cout<<dp[n]<<"\n";
	}
	return 0;
}
