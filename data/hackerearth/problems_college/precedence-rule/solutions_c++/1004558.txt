#include <iostream>
#include<stack>
using namespace std;
int main()
{
    int t,i,j;
    string str;
    char sg;
    long int a,b;
    stack<long int> st1;
    stack<char>st2;
    
    cin>>t;
    
    while(t--)
    {
    	 cin>>str;
    	j=0;
    	for(i=0;i<(str.size());i++)
    	{
    		 if(str[i]>='0' && str[i]<='9')
    		  {
    		   	  st1.push(str[i]-'0');  
    		  } 
    		  else
    		   st2.push(str[i]);
    	}
    	while(!st2.empty())
    	{
    		 sg=st2.top();
    		 st2.pop();
    		 
    		 a=st1.top();
    		 st1.pop();
    		 b=st1.top();
    		 st1.pop();
    		 switch(sg)
    		 {
    		 	           case '+' : st1.push(a+b);
    		 	                      break;
    		 	           case '-' : st1.push(a-b);
    		 	           break;
    		 	           case '*' : st1.push(a*b);
    		 	             break;
    		 }
    	}
    	cout<<st1.top()<<endl;
    	st1.pop();
    	
    }
    return 0;
}
