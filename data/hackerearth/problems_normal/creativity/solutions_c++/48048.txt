#include<iostream>
#include<string>
#include<sstream>
//#include<ctype>
using namespace std;
int main()
{
    int t;
    string s1 = "", s2 = "", s3 = "";
    cin>>t;
    for(int i = 0; i < t; i++)
    {
        int j = 0, k =0;
        if(i == 0)
            getline(cin, s3);
        getline(cin, s1);
        getline(cin, s2);
        j = s1.length();
        k = s2.length();
        int l = 0, m = 0;
        while(m != k)
        {
            if(s1[l] == s2[m])
            {
                l++;
                m++;
            }
            else
            {
                m++;
                l = 0;
            }
            if(l == j)
                break;

        }
        if(l == j)
            cout<<"possible"<<endl;
        else
            cout<<"not possible"<<endl;
    }
}
