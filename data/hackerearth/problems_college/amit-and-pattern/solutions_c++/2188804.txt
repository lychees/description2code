#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n,m;
	cin >> n >> m;
	int A[n][m];
	for(int i=0;i<n;++i)
		for(int j=0;j<m;++j)
			cin >> A[i][j];
	for(int i=0;i<m;++i)
	{
		int x = 0, y = i;
		while(x<n and y>=0)
		{
			//cout << "Debug : " << x << " " << y << endl;
			cout << A[x++][y--] << " ";
		}
		cout << endl;
	}
	for(int i=1;i<n;++i)
	{
		int x = i, y = m-1;
		while(x<n and y>=0)
		{
			//cout << "Debug : " << x << " " << y << endl;
			cout << A[x++][y--] << " ";
		}
		cout << endl;
	}
	return 0;
}
