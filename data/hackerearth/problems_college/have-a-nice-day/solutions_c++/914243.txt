#include <iostream>
using namespace std;
#include<algorithm>
int main()
{
    //cout << "Hello World!" << endl;
    int a[1001];
    int b[1001];
    int c[1500];
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	for(int i=0;i<n;i++)
    	cin>>a[i];
    	for(int i=0;i<n;i++)
    	cin>>b[i];
    	for(int i=0;i<1500;i++)
    	c[i]=0;
    	sort(a,a+n);
    	sort(b,b+n);
    	int sum,max=-1;
    	for(int i=0;i<n;i++)
    	{
    		sum=a[i]*b[i];
    		int j=-1,s=0,p;
    		while(sum)
    		{
    			j=j+1;
    			p=c[j]+(s+sum%10);
    			c[j]=p%10;
    			s=p/10;
    		sum=sum/10;
    		}
    		while(s)
    		{
    			j=j+1;
    			p=c[j]+s;
    			c[j]=p%10;
    			s=p/10;
    		}
    		if(max<j)
    		max=j;
    	}
    	//cout<<max;
    	for(int i=max;i>=0;i--)
    	{
    		cout<<c[i];
    	}
    	cout<<endl;
    	//cout<<sum<<endl;
    }
    return 0;
}
