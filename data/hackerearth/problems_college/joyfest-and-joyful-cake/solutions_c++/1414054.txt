#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    int test;
    scanf("%d",&test);
    while(test--){
    	ll n;
    	scanf("%lld",&n);
    	n = 2*n-2;
    	ll x = sqrt(n);
    	//cout<<x<<endl;
    	if(x*(x+1)<n)x++;
    	ll ans = x;
    	printf("%lld\n",ans);
    }
    return 0;
}
