#include<bits/stdc++.h>
//zone
using namespace std;
int main()
{

    int t;
    cin >> t;
    int n;
    int s;
    int i;
    int po;
    int j,sum;
    int flag = 0;
    int a[20];
    while(t--)
    {
        flag = 0;
        po = 0;
        sum = 0;
        cin >> n;
        for(i = 0 ; i < n ; i++)
        {
            cin >> a[i];
        }
        cin >> s;
        po = 1 << n;
        for(i = 0 ; i < po ; i++)
        {
            sum = 0;
            for(j = 0 ; j < n; j++)
            {
                //int x;
                if((i>>j)&1)
                {
                   // x = i >> j;
                    //cout<<"i-->" << i <<" " << "j-->"<<j << " " <<"i >>j -->" << x <<endl;
                    sum = sum + a[j];
                }

            }
            if(sum == s)
            {
                flag = 1;
                break;
            }
        }
        if(flag == 0)
        {
            cout<<"NO\n";
        }
        else
            cout<<"YES\n";

    }
    return 0;
}
