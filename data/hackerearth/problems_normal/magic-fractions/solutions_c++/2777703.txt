#include <iostream>
#include <cstdio>
#include <vector>
#include <string>

using namespace std;

#define ll long long

int main()
{
	ios_base::sync_with_stdio(false);
	
	int i,j,n;

	cin>>n;
	
	vector<bool> noprime(n+1,0);
	vector<ll> nfractions(n+1,0);
	
	ll cnt=0;
	noprime[1]=1;
	nfractions[1]=0;
	for(i=2;i<=n;i++)
	{
		if(noprime[i]==0)
		{
			j=i*i;
			while(j<=n)
			{
				noprime[j]=1;
				j+=i;
			}
			if(cnt==0)
			{
				cnt=1;
				nfractions[i]=cnt;
			}
			else
			{
				cnt*=2;
				nfractions[i]=nfractions[i-1]+cnt;
			}
		}
		else
			nfractions[i]=nfractions[i-1]+cnt;
	}

	cout<<nfractions[n]<<endl;
	return 0;
}
