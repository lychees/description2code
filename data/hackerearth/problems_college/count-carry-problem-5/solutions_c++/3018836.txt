#include <iostream>
using namespace std;

int main()
{
    
    unsigned long a, d, t, m1, m2, p, n1, n2;
    cin>>t;
    for(a=0; a<t; a++)
    {
    	p=0;d=0;
    	cin>>m1>>m2;
    	while((m1>0)&&(m2>0))
    	{   
    		n1=m1%10;
    		m1=m1/10;
    		n2=m2%10;
    		m2=m2/10;
    		if(d!=0)
    		n1++;
    		if((n1+n2)>9)
    		{
    			p++;
    			d=1;
    		}
    		else
    		d=0;
    	}
    	if(p==0)
    	cout<<"No carry operation\n";
    	else if(p==1)
    	cout<<p<<" carry operation\n";
    	else
    	 cout<<p<<" carry operations\n";
    	
    	 
    }
    return 0;
}
