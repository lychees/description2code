#include <iostream>
using namespace std;
int main()
{
	
    int N,M;
    cin >> N >> M;
	int a[100001],in=0;
	int m = M/2,i;
	
	if(M%2){ a[in]=(M/2 + 1); in++; }

	for(i=m;i>=1;i--)
	{
		a[in]=i;
		in++;
		a[in]=(M+1-i);
		in++;
	}
	
	int k = N/M;
	for(i=0;i<k;i++)
	 for(int j=0;j<M;j++)
	  cout << a[j] << endl;
	  
	k = N%M;
	
	for(i=0;i<k;i++)
	cout << a[i] << endl;
}
