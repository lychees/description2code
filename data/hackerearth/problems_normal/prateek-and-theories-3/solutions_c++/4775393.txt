#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

int main() {
    int t;
    cin>>t;
    while (t) {
        t--;
        int n;
        cin>>n;
        vector <long> starts (n);
        vector <long> ends(n);

        for (int i = 0; i < n; ++i) {
            cin>>starts[i]>>ends[i];
        }

        sort(starts.begin(), starts.end());
        sort(ends.begin(), ends.end());

        int s = 0, e = 0;
        int max_popularity = 0, current_popularity = 0;
        while ( s < n && e < n ) {
            if (starts[s] < ends[e]) {
                s++;
                current_popularity++;
            } else if (starts[s] > ends[e]) {
                e++;
                current_popularity--;
            } else {
                s++;
                e++;
            }
            max_popularity = max(max_popularity, current_popularity);
        }
        cout<<max_popularity<<"\n";
    }


}
