#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,flag=0,fflag=0,i,j;
        long long int tot=0;
        cin>>n;
        char a[n][n];
        for(i=0;i<n;i++)
        {
            for(j=0;j<n;j++)
            {
               
                cin>>a[i][j];
                
                if(a[i][j]=='#')
                {
                    tot++;
                }

            }
        }
        if(tot%2==0)
        {


long long int sum=0,sum2=0,c=0,c2=0;

        for(i=0;i<n;i++)
        {
            for(j=0;j<n;j++)
            {
                if(a[i][j]=='#')
                {
                    c++;
                    
                    
                }
                
               
            }
            sum=sum+c;
            c=0;
             
            if(sum==(tot/2))
            {
                flag=1;
                    break;

            }
        }
        if(flag==0)
        {
        for(j=0;j<n;j++)
        {
            for(i=0;i<n;i++)
            {
                if(a[i][j]=='#')
                {
                    c2++;
                     
                }
               
                
            }
            sum2=sum2+c2;
            c2=0;
            
            if(sum2==(tot/2))
            {
                fflag=1;
                
                    break;

            }
        }
        }

        if((flag==1)||(fflag==1))
        {
            cout<<"YES"<<endl;

        }
        else{

            cout<<"NO"<<endl;
        }
        flag=0;
        fflag=0;
        }
        else
        {
        	cout<<"NO"<<endl;
        }
        
        }
        return 0;
    }
