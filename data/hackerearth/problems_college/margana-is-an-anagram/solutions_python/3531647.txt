t = int(input())
for i in range(0,t):
	st=raw_input()
	s=st.split()
	s[0]=''.join(sorted(s[0]))
	s[1]=''.join(sorted(s[1]))
	if s[0]==s[1]:
		print 'Yes'
	else:
		print 'No'
