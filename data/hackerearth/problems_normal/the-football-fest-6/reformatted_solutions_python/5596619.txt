def tc_match():
    np, fid = map(int, raw_input().split())
    cholder = fid
    preholder = fid
    if np >= 1 and np <= 100000:
        for i in range(np):
            query = raw_input()
            if query.startswith('B'):
                cholder, preholder = back(cholder, preholder)
            else:
                temp = query.split()
                id = int(temp.pop())
                if id >= 1 and id <= 1000000:
                    cholder, preholder = push(id, cholder, preholder)
    return cholder


def push(id, cholder, preholder):
    preholder = cholder
    cholder = id
    return cholder, preholder


def back(cholder, preholder):
    temp = cholder
    cholder = preholder
    preholder = temp
    return cholder, preholder


def main():
    tc = input()
    if tc >= 1 and tc <= 100:
        for i in range(tc):
            holder = tc_match()
            print "Player %d" % (holder)


if __name__ == '__main__':
    main()
