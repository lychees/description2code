#include <iostream>
using namespace std;

int main()
{
   int l,q,i,j,count;
   cin>>l>>q;
   int p;
   char rep;
   char s[l];
   for(i=0;i<l;i++)
   {
   		cin>>s[i];
   }
   for(i=0;i<q;i++)
   {    count=0;
   		cin>>p;
   		p--;
   		cin>>rep;
   		s[p]=rep;
   		for(j=0;j<l-1;j++)
   		{
   			if(s[j]=='.'&&s[j+1]=='.')
   			{
   				count++;
   			}
   		}
   		cout<<count<<endl;
   }
    return 0;
}
