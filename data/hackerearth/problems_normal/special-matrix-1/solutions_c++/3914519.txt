#include <iostream>
using namespace std;

int main()
{
    int t,n,i,j,c,r;
    cin>>t;
    for(int k=0;k<t;k++)
    {
    	cin>>n;
    	char ar[n][n];
    	for(i=0;i<n;i++)
    	{
    		for(j=0;j<n;j++)
    		{
    			cin>>ar[i][j];
    		}
    	}
    	for(i=0;i<n;i++)
    	{
    		for(j=0;j<n;j++)
    		{
    			if(ar[i][j]=='*')
    			{
    				r=i;
    				c=j;
    			}
    		}
    	}
    	//cout<<r<<" "<<c<<endl;
    	cout<<abs((n/2)-r)+abs((n/2)-c)<<endl;
    }
    return 0;
}
