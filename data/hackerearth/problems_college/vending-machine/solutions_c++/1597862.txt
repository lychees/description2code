#include<cstdio>
#include<cmath>
const char name[5][8] = {"Sheldon", "Leonard", "Penny", "Rajesh", "Howard"};

int main(void)
{
	int n;
	scanf("%d", &n);
	int doublenum = 1 << ((int)(log10((double)(n + 4) / 5) / log10(2)));
	printf("%s", name[(n - 5 * doublenum + 4) / doublenum]);
	return 0;
}
