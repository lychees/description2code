#include <iostream>
using namespace std;

int main()
{
    long int t,m,count=0;
    long int arr[100000];
    cin>>t;
    while(t!=0)
    {
    	count=0;
    	cin>>m;
    	for(int i=0;i<m;i++)
    	  cin>>arr[i];
       for (int i=0; i<m; i++)
    {
        int j;
        for (j=0; j<i; j++)
           if (arr[i] == arr[j])
               break;
 
        if (i == j)
          count++;
    }
    cout<<count<<endl;
    --t;
    }
    return 0;
}
