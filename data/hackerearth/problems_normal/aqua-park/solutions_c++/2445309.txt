#include <iostream>
using namespace std;
typedef long long ll;

const int MAXN = 105;
int N;
ll posx, posy = 1<<30;
ll  X1[MAXN], Y1[MAXN], X2[MAXN], Y2[MAXN];

bool better(int a, int b) //is seg b higher than seg a but lower than current point
{
	ll yb = (Y2[b] - Y1[b])*posx + Y1[b]*X2[b] - Y2[b]*X1[b];	
	if (yb >= posy*(X2[b] - X1[b]))
		return false;
		
	if (a == -1)
		return true;
		
	ll ya = (Y2[a] - Y1[a])*posx + Y1[a]*X2[a] - Y2[a]*X1[a];
	return ya*(X2[b] - X1[b]) < yb*(X2[a] - X1[a]);
}

int main()
{
    ios::sync_with_stdio(0);
    
    cin >> posx >> N;
    for (int i = 0; i < N; i++)
    	cin >> X1[i] >> Y1[i] >> X2[i] >> Y2[i];
    	
    while (true)
    {
    	int idx = -1;
    	for (int i = 0; i < N; i++)
    		if (X1[i] <= posx && posx <= X2[i] && better(idx, i))
    			idx = i;
    	
    	if (idx == -1)
    		break;
    	
    	if (Y1[idx] > Y2[idx])
    		posx = X2[idx], posy = Y2[idx];
    	else
    		posx = X1[idx], posy = Y1[idx];
    }
    
    cout << posx << "\n";
    
    return 0;
}
