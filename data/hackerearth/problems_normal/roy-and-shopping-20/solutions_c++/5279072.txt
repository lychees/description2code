#include <bits/stdc++.h>
using namespace std;

int main()
{
    int prime[1000010]={0};
    for(int i=2;i<1000010;i++){
    	if(prime[i]==0){
    		for(int j=1;i*j<1000010;j++){
    			if(prime[i*j]==0) prime[i*j]=i;
    		}
    	}
    }
    int t; scanf("%d",&t);
    while(t--){
    	int n; scanf("%d",&n);
    	n-=prime[n];
    	printf("%d\n",n);
    }
    return 0;
}
