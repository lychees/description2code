#include <iostream>
#include<algorithm>
using namespace std;
int main()
{
	int sum,count,a[100005],i,n,k;
    scanf("%d %d",&n,&k);
    for(i=0;i<n;i++)
    {
    	scanf("%d",&a[i]);
    }
    sort(a,a+n);
    sum=0;
    count=0;
    for(i=0;i<n;i++)
    {
    	if(sum+a[i]<=k)
    	{
    		sum+=a[i];
    		count++;
    	}
    	else
    	{
    		break;
    	}
    }
    printf("%d\n",count);
    return 0;
}
