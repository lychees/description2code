#include <iostream>
#include<bits/stdc++.h>

using namespace std;
map<string , vector<int> > mp;
map<string , vector<int> >::iterator it;
int main()
{
	int n,i,j,k,q,l,r,ans;
	string s,s1;
    cin>>n;
    for(i = 0 ; i < n ; i++)
    {
    	cin>>s;
    	for(j = 0 ; j < (int)s.size() ; j++)
    	{
    		s1 = "";
    		for(k = j ; k < (int)s.size() ;k++)
    		{
    			s1.push_back(s[k]);
//    			s1 = s1 + s[k];
    			mp[s1].push_back(i);
    		}
    	}
    }
    
    for(it = mp.begin() ; it != mp.end(); it++)
    {
    vector<int>::iterator si =  unique(it->second.begin() , it->second.end());
    	(it->second).resize(distance( (it->second).begin() , si));
    }
    
    cin>>q;
    while(q--)
    {
    	cin>>l>>r;
    	cin>>s;
    	l--;
    	r--;
    	
    	if(s.empty())
    		ans = r-l+1;
    	else
    	{
    		int lit = lower_bound(mp[s].begin() , mp[s].end() , l ) - mp[s].begin();
    		int rit = upper_bound(mp[s].begin() , mp[s].end() , r) - mp[s].begin();
    		ans = rit - lit;
    	}
    	cout<<ans<<endl;
    }
    
    return 0;
}
