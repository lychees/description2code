#include <iostream>
using namespace std;

int main()
{
	int n;
	cin>>n;
	long ar[n][n];
	for(int i=0;i<n;i++)
	for(int j=0;j<n;j++)
	{
		cin>>ar[i][j];
		
	}
	long sum1=0,sum2=0;
	
	for(int i=0;i<n;i++)
	sum1=sum1+ar[i][i];
    //cout << "Hello World!" << endl;
    
    for(int i=0;i<n;i++)
     sum2=sum2+ar[i][n-1-i];
     
     sum1=sum1-sum2;
     if(sum1>=0)
     cout<<sum1;
     else
     cout<<sum1*(-1);
   
     
    return 0;
}
