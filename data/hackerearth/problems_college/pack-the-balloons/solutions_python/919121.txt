n = int(input())

assert(n>=1 and n<= 10**5)

for i in range(n):
	arr = map(int, list(raw_input().strip().split(' ')))
	assert(arr[0] >= 1 and arr[0] <= 10**18 and arr[1] >= 1 and arr[1] <= 10**18 and arr[2] >= 1 and arr[2] <= 10**18)
	t = min(arr)
	if t%2 == 0:
		print str(t/2) + ".0"
	else:
		print str(t/2) + ".5"
