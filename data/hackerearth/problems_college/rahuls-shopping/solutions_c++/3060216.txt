#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    int t,l,b,num,mega=0;
    cin>>t;
    int ar[t];
    for(int i=0;i<t;i++)
    	cin>>ar[i];
    cin>>l>>b;
    for(int i=0;i<t;i++)
    {
    	int sum=0;
    	float disc=0.00,res=0,sm=0;
    	num=ar[i];
    	while(num>0)
    	{
    		sum=sum+(num%10);
    		num/=10;
    	}
    	res=ar[i]*sum;
    	if(ar[i]<11)
    		disc=0.75;
    	else if(ar[i]<21)
    		disc=0.70;
    	else 
    		disc=0.80;
    	sm=res*disc*l*b;
		mega+=floor(sm);
    }
    cout<<mega;
    return 0;
}
