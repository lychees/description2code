#include <iostream>
#include <stdlib.h>
#include <algorithm>
#include <limits.h>
#include <vector>
using namespace std;
#define gc getchar_unlocked
#define rd readInt()
inline long long readInt(){
    long long n = 0, c = gc(), f = 1;
    while(c != '-' && (c < '0' || c > '9')) c = gc();
    if(c == '-') f = -1, c = gc();
    while(c >= '0' && c <= '9')
        n = (n<<3) + (n<<1) + c - '0', c = gc();
    return n * f;
}
vector<long long> nom;
int main()
{  
    long long i,n,v;
    n = rd;
    for(i=0;i<n;++i)
    { v = rd; nom.push_back(v);}
    
    sort(nom.begin(),nom.end());
    
    v = nom[n-1];
    long long a,sor = 0;
    for(i=n-2;i>=0;i--)
    {
        a = nom[i] ^ v;
        sor  = (sor | a);
    }
    printf("%lld\n",sor);
    
    nom.clear();
    
    return 0;
}
