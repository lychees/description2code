#include <iostream>
using namespace std;
long fun (long i,long no,long j)
{
	static long count=0;
	long temp;
	if (j<=no && i<no)
	{
	if ((i*j)%(i+j)==0)
		count++;
	fun (i,no,j+1);
	if (j==i+1)
		fun (i+1,no,i+2);
    }
	return count;
}
int main ()
{
	long T,no,sum=0,temp;
	cin>>T;
	while (T--)
	{
		cin>>no;
		temp=fun (1,no,2);
		cout<<temp-sum<<"\n";
		sum=temp;
	}
}
