typedef long long ll;
using namespace std;
const int MAXM = 1E6 + 100;
long long c[MAXM];
int n;
int a[MAXM], MOD;
void solve() {
    cin>>a[0]>>a[1]>>n>>MOD;
    for (int i = 0; i < MOD; i++) {
        c[i] = 0;
    }
    for (int i = 2; i < n; i++) {
        a[i] = a[i - 1] + a[i - 2];
        a[i] %= MOD;
    }
    for (int i = 0; i < n; i++) {
        //cerr<<a[i]<<" ";
        c[ a[i] ]++;
    }
    //cerr<<endl;
    long long ans = 0;
    for (int i = 0; i < MOD; i++) {
        ans += c[i] * c[i];
    }
    cout<<ans<<endl;
}
int main() {
    int t;
    cin>>t;
    while (t--) {
        solve();
    }
    return 0;
}
