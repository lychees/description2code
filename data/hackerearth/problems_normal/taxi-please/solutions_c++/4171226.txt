#include<bits/stdc++.h>
#define pb push_back
#define ull unsigned long long
#define lli long long int
#define max(a,b) ((a>b)?a:b)
#define min(a,b) ((a<b)?a:b)
#define pulli(n) printf("%llu\n",n);
#define sulli(n) scanf("%llu",&n);
#define slli(n) scanf("%lld",&n)
#define si(n) scanf("%d",&n)
#define sd(n) scanf("%lf",&n)
#define pd(n) printf("%lf\n",n)
#define pi(n) printf("%d\n",n)
#define plli(n) printf("%lld\n",n)
#define iofast (ios_base::sync_with_stdio(false))
using namespace std;
int main()
{
    iofast;
    int n,m,i,st,dur;
    cin>>n>>m;
    set<pair<pair<int,int>,int>> c;
    set<pair<int,int>> s;
    set<pair<int,int>>::iterator it,tempit;
    set<pair<pair<int,int>,int>>::iterator cit;
    int a[n+1000];
    for(int i=1;i<=n;i++)
    {
    	cin>>st>>dur;
    	c.insert(make_pair(make_pair(st,st+dur),i));
    }
    for(i=1;i<=m;i++)
    	s.insert(make_pair(1,i));
    cit=c.begin();
    it=s.begin();
    for(i=0;i<n;i++)
    {
    	//it=s.begin();
    	if((*it).first<=(*cit).first.first)
    	{
    		int minidx=(*it).second;
    		tempit=it;
    		while(((*it).first<=(*cit).first.first)&&(it!=s.end()))
    		{
    			if((*it).second<minidx)
    			{
    				tempit=it;
    				minidx=(*it).second;
    			}
    			it++;
    		}
    		it=tempit;
    		//cout<<(*it).second<<" ";
    		a[(*cit).second]=(*it).second;
    		s.insert(make_pair((*cit).first.second,(*it).second));
    		s.erase(it);
    		it=s.begin();
    	}
    	else
    		a[(*cit).second]=-1;
    	c.erase(cit);
    	cit=c.begin();
    }
    for(i=1;i<=n;i++)
    	cout<<a[i]<<" ";
    return 0;
}
