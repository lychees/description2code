#include<cstdio>
#include<iostream>
#include<algorithm>
#include<stdio.h>
using namespace std;
#define FOR(i,a,b) for(i= a ; i < b ; ++i)
#define rep(i,n) FOR(i,0,n)
#define INF INT_MAX
#define pln(n) printf("%lld\n",n)
#define sl(n) scanf("%lld",&n)
#define mod (int)(1e9 + 7)
#define ll long long int
ll modpow(ll a,ll n,ll temp){ll res=1,y=a;while(n>0){if(n&1)res=(res*y)%temp;y=(y*y)%temp;n/=2;}return res%temp;} 
ll arr[1000006];
ll checkit(ll n)
{
	if(n>=mod)
		n%=mod;
	return n;
}
int main()
{
	ll t,n,i;
	ll calc,c1,c2,val;
	sl(t);
	while(t--)
	{
		calc=1;
		c1=0;
		sl(n);
		rep(i,n)
			sl(arr[i]);
		sort(arr,arr+n);
		val=n/2+n%2;
		val=n-val;
		rep(i,val)
			c1+=arr[i];
		for(i=val;i<n;i++)
			c1-=arr[i];
		c1*=-1;
		calc*=c1;
		calc=checkit(calc);
		c2=0;
		rep(i,n)
		{
			if(arr[i]>0)
				c2+=arr[i];
			else
				c2-=arr[i];
		}
		c2=checkit(c2);
		calc*=c2;
		calc=checkit(calc);
		pln(calc);
	}
	return 0;
}
