#include <iostream>
using namespace std;

int main()
{
    int a, b, n;
    cin >> a >> b >> n;
    if (n == 1) {
    	cout << a << endl;
    	
    } else if (n == 2) {
    	cout << b << endl;
    } else if (n > 2) {
    	int val;
    	int pre1 = a;
    	int pre2 = b;
    	int tmp;
    	for (int i = 3; i <= n; i++) {
    		tmp = pre2;
    		val = pre1 + pre2;
    		pre1 = tmp;
    		pre2 = val;
    	}
    	cout << val << endl;
    }
    return 0;
}
