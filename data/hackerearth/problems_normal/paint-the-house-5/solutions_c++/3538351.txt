#include <bits/stdc++.h>
#define ll long long
using namespace std;
struct worker {
	int t,x,y;
};
bool fun(struct worker a,struct worker b) {
	if(a.t==b.t) {
		if(a.y==b.y) {
			return (a.x<b.x);
		}
		return (a.y>b.y);
	}
	return (a.t<b.t);
}
int main()
{
   	ll int n,d,i;
    cin>>n>>d;
    struct worker a[n];
    for(i=0;i<n;i++) {
    	cin>>a[i].t>>a[i].x>>a[i].y;
    	//a[i].val=((float) a[i].y)/a[i].x;
    }
    sort(a,a+n,fun);
    ll int cost=a[0].x;
    int temp=a[0].y;
    i=1;
  	while(d>0 && i<n) {
  	//	cout<<a[i].t<<" "<<a[i].x<<" "<<a[i].y<<" "<<d<<endl;
  		d=d-(a[i].t-a[i-1].t)*temp;
  		if(d>0 && a[i].y>temp ){
  			temp=a[i].y;
  			cost+=a[i].x;
  		}
  		i++;
  	}
  	cout<<cost;
    return 0;
}
