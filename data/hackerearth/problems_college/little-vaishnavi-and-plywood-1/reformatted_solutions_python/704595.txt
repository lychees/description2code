T = input()
for _ in xrange(T):
    [M, K] = map(int, raw_input().split())
    ans = 0
    if (K == 0):
        print M-1
    else:
        ans = (K-1)*4 + 5
        if (M != 5 and M != 1):
            if (K & 1 == 1):
                ans += 4-M
            elif (K & 1 == 0):
                ans += M-2
        else:
            if (M == 1):
                ans = 8*K
            elif (M == 5):
                ans = 4+8*K
        print ans
