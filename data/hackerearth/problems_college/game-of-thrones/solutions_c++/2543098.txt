#include <cmath>
#include <cstdio>
#include <vector>
#include <set>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
    int t,n,i;
    long long int *x;
    
    cin>>t;
    while(t--){
    	cin>>n;
    	x = (long long int *)malloc(sizeof(long long int)*n);
    	for(i=0;i<n;i++){
    		cin>>x[i];
    	}
    	
    	set<long long int> s;
    	for(i=0;i<n;i++){
    		s.insert(x[i]);
    	}
    	
    	cout<<(n-s.size())<<endl;
    }
    
    return 0;
}
