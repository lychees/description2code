int check(string a,string b)
{
    for(int i=0;i<a.size();i++)
    {
        int j;
        for( j=0;j<b.size() && i+j<a.size() ;j++)
        {
            if(a[i+j]==b[j]) continue;
            break;
        }
        if(j==b.size()) return 1;
    }
    return 0;
}
int main()
{
    int ans=0;
    map<string,int> m;
    string a,b; cin>>a>>b;
    for(int i=0;i<a.size();i++)
    {
        string s="";
        for(int j=i;j<a.size();j++)
        {
            s+=a[j];
            if(m[s]) continue;
            m[s]=1;
            if(check(s,b)) ans++;
        }
    }
    cout<<ans<<endl;
	return 0;
}
