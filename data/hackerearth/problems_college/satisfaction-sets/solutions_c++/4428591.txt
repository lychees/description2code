ll mod= 1000000007;
int a[103][503]={},c[503]={},b[100002];
int main()
{
    int i,j;
    ll n,m,s=0,in=0,cnt=0;
    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>m;
        for(j=0;j<m;j++)
        {
        cin>>b[j];
        c[b[j]]++;
        a[i][b[j]]++;
        }
    }
    for(i=0;i<502;i++)
        if(c[i]!=0)
        s+=i;
    for(j=0;j<=500;j++)
    {
        cnt=0;
        for(i=0;i<n;i++)
        {
            if(a[i][j]!=0)
                cnt++;
            //if(a[i][j]>mx)
              //  mx=a[i][j];
        }
        if(cnt==n)
            in+=j;
    }
    in=s-in;
    cout<<((in-s)%mod)<<endl;
    //cout<<s;
    return 0;
}
