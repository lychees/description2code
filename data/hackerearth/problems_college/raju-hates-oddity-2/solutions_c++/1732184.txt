#include<bits/stdc++.h>
using namespace std;
long int dp[1000001];
long int dynamic(long int n)
 {
       if(dp[n]!=-1) return dp[n];
       else
              {
                     dp[n]=dynamic(n-1)+dynamic(n-2);
              }
              return dp[n];
 }
int main()
{
long int t,n,i; cin>>t; while(t--)
{
for(i=3;i<100001;i++) dp[i]=-1;
dp[0]=0;
dp[1]=1;
dp[2]=2;
cin>>n;
cout<<dynamic(n)<<"\n";
}
return 0;
}
