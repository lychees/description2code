int main()
{

    ll mul;

    while(mul < 5) mul = (rand() << 10) ^ rand();

    cin >> n >> k >> l >> q;

        cin >> s;

        ll h = 0;

        for(int i = 0 ; i < k ; i ++){
            cin >> x;
            h = (h * mul + x) % mod;
        }
        m[h] = s;
    }

    while(q --){
        ll h = 0;
        for(int i = 0 ; i < k ; i ++){
            cin >> x;

    }
