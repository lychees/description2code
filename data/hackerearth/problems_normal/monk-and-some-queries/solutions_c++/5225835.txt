#include <bits/stdc++.h>
using namespace std;
long long n,q,x;
multiset<long long> s;
multiset<long long>::iterator it;
int main()
{
	ios::sync_with_stdio(false);
	cin.tie(NULL);
	cin>>n;
	while(n--)
	{
		cin>>q;
		if(q==1)
		{
			cin>>x;
			s.insert(x);
		}
		if(q==2)
		{
			cin>>x;
			if(s.find(x)!=s.end())
			{
				s.erase(s.find(x));
			}
			else
			{
				cout<<"-1\n";
			}
		}
		if(q==3)
		{
			if(s.empty())
			{
				cout<<"-1\n";
				continue;
			}
			it=s.end();
			it--;
			cout<<*it<<endl;
		}
		if(q==4)
		{
			if(s.empty())
			{
				cout<<"-1\n";
				continue;
			}
			it=s.begin();
			cout<<*it<<endl;
		}
	}
	return 0;
}
