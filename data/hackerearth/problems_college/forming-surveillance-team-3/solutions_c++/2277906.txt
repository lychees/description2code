typedef long long ll;

void fastIN(long long &a)
{
    char c;
    for(c=getchar(); !(c>='0' && c<='9') ; c=getchar());
    a = c-'0';
    for(c=getchar(); (c>='0' && c<='9') ; c=getchar())
    {
        a = a*10 + c-'0';
    }
}
