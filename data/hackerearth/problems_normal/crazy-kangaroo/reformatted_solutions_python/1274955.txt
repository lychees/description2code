test_cases = int(raw_input())
for i in range(0, test_cases):
    start, end, hop = raw_input().split()
    start = int(start)
    end = int(end)
    hop = int(hop)
    count = 0
    if (hop <= end):
        if start <= hop:
            start = hop
        else:
            if start % hop != 0:
                start = ((start / hop) + 1) * hop
        count = (end / hop) - (start / hop) + 1
    print count
