#include<bits/stdc++.h>
using namespace std;
void rec(int *s,int st,int ed,int mid)
{
    int a[ed-st+1];
    int c=0;
int     i=st;
   int  j=mid+1;
    while(i<=mid&&j<=ed)
    {
        if(s[i]<s[j])
            a[c++]=s[i++];
        else
            a[c++]=s[j++];
    }
    while(i<=mid)
        a[c++]=s[i++];
    while(j<=ed)
        a[c++]=s[j++];
        c--;
    for(i=ed;i>=st;i--)
    {
        s[i]=a[c--];
    }
}
void sor(int*s,int st,int ed)
{
   int mid=(st+ed)/2;
   if(st!=mid)
   {
       sor(s,st,mid);
       sor(s,mid+1,ed);
       rec(s,st,ed,mid);
   }
   else
   {
       if(s[st]>s[ed])
       {
           int temp=s[st];
           s[st]=s[ed];
           s[ed]=temp;
       }
   }
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int c=0,m=0;
        int d,n;
        cin>>n;
        int s[n];
        int i;
        for(i=0;i<n;i++)
        {
            cin>>s[i];
        }
        sor(s,0,n-1);
        int j;
        for(j=1;j<n;j++)
        {
            s[j]=s[j]+m;
            if((s[j]-s[j-1])>=5)
            {
                d=(s[j]-s[j-1])/5;
                c=c+d;
                m=m+d*5;
                                s[j-1]=s[j-1]+d*5;
            }
            if(s[j]-s[j-1]>=2)
            {
                 d=(s[j]-s[j-1])/2;
                c=c+d;
                m=m+d*2;
                s[j-1]=s[j-1]+d*2;
                    }
            if(s[j]-s[j-1]>=1)
            {
                c++;
                m++;
            }
        }
        cout<<c<<endl;
    }
    return 0;
}
