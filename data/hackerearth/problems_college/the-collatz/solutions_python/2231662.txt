T=input()
for _ in range(T):
	esum=0
	osum=0
	n=0
	s=input()
	if s<2:
		print "Tie"
		break
	if s%2:
		osum+=s
	else:
		esum+=s
	while s!=1:
		if s%2:
			s=3*s+1
		else:
			s/=2
		if s%2:
			osum+=s
		else:
			esum+=s
		n+=1
	oremain=esum%n
	eremain=osum%n
	if oremain<eremain:
		print "Odd Rules"
	elif eremain<oremain:
		print "Even Rules"
	else: print "Tie"
