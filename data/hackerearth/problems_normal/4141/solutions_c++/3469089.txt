#include <bits/stdc++.h>
using namespace std;

int *a, c, n;

bool F4141(){
    int ans=0;
    for(int i=0;i<n;i++){
        if(i>1)
            a[i] = (a[i-1]*c + a[i-2])%10;
        ans = (ans+a[i])%41;
        ans = (ans*10)%41;
    }
    if(ans == 0)
        return true;
    return false;
}
int main(){
    //freopen("in.txt", "r", stdin);
    bool ans;
    int t, a0, a1;
    cin>>t;
    while(t--){
        cin>>a0>>a1>>c>>n;
        a = (int*)calloc(sizeof(int), n+1);
        a[0] = a0;
        a[1] = a1;
        if(n==1){
            if(a0==0)
                cout<<"YES\n";
            else
                cout<<"NO\n";
        }
        else{
            if(F4141())
                cout<<"YES\n";
            else
                cout<<"NO\n";
        }
        free(a);
    }
    return 0;
}
