#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
ll mod=1000000007;
void multiply(ll a[2][2],ll b[2][2])
{
    ll w,x,y,z;
    w=((a[0][0]*b[0][0])%mod+(a[0][1]*b[1][0])%mod)%mod;
    x=((a[0][0]*b[0][1])%mod+(a[0][1]*b[1][1])%mod)%mod;
    y=((a[1][0]*b[0][0])%mod+(a[1][1]*b[1][0])%mod)%mod;
    z=((a[1][0]*b[0][1])%mod+(a[1][1]*b[1][1])%mod)%mod;
    a[0][0]=w;
    a[0][1]=x;
    a[1][0]=y;
    a[1][1]=z;
    return;
}
void power(ll mat[2][2],ll n)
{
    ll f[2][2]={1,1,1,0};
    if(n==0 || n==1)
        return;
    power(mat,n/2);
    multiply(mat,mat);
    if(n%2!=0)
        multiply(mat,f);
    return;
}
int main()
{
    ll n,q,x;
    cin>>q;
    while(q--)
    {
        ll mat[2][2]={1,1,1,0};
        cin>>n;
        n=(ceil)(n/2.0);
        power(mat,n-1);
        x=mat[0][0];
        cout<<(x*x)%mod<<endl;
    }
    return 0;
}
