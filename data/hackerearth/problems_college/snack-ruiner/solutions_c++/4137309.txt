int main()
{
	ios_base::sync_with_stdio(0);
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        float f[n];
        float d[n];
        float ans[n];
        float sum=0;
        for(int i=0;i<n;i++)
        {
            cin>>d[i];
        }
        for(int i=0;i<n;i++)
        {
            cin>>f[i];
        }
        for(int i=0;i<n;i++)
        {
            sum=sum+(d[i]*f[i]);
        }
        for(int i=0;i<n;i++)
        {
            ans[i]=(d[i]*f[i])/(sum);
        }
        float maxi=ans[0];
        int index=0;
        for(int j=0; j<n;j++)
        {
            if(maxi<ans[j])
            {
                maxi=ans[j];
                index=j;
            }
        }
        cout<<index<<" "<<setprecision(3)<<fixed<<maxi<<"\n";
    }
    return 0;
}
