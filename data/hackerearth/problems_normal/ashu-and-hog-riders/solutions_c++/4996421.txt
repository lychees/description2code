#include<bits/stdc++.h>
using namespace std;
long long mi=0;
long long graph[1001][1001];
void dfs(long long x,long long y,long long n)
{
	if(x<0||y<0||x==n||y==n)
	return;
	if(graph[x][y]==0)
	 return;
	 if(graph[x][y]==1)
	 {
	  graph[x][y]=0;
	  mi++;
	  //cout<<mi<<endl;
     }
	  
	  dfs(x+1,y,n);
	  dfs(x,y+1,n);
	  dfs(x-1,y,n);
	  dfs(x,y-1,n);
	  
}
int main()
{
	long long  ma=INT_MIN;
	long long n,i,j;
	cin>>n;
	for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
		{
			cin>>graph[i][j];
		}
	}
	for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
		{
			if(graph[i][j]==1)
			{
			
		      mi=0;
		      dfs(i,j,n);
			  ma=max(ma,mi);	
		   }
		}
	}
	cout<<ma<<endl;
	
}
