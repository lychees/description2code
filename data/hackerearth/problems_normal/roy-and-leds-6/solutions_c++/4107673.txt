#include <iostream>
using namespace std;

int main()
{
    long long t,r,g,b,red,green,blue,yellow,cyan,magenta,white,black,i,x;
    red=green=blue=yellow=cyan=magenta=white=black=0;
    int rs,gs,bs;
    cin>>t>>r>>g>>b;
    for(i=0;i<t;i++)
    {
    	if(i%r==0)
    	{
    		if((i/r)%2==0)
    		{
    			rs=0;
    		}
    		else
    		{
    			rs=1;
    		}
    	}
    	if(i%g==0)
    	{
    		if((i/g)%2==0)
    		{
    			gs=0;
    		}
    		else
    		{
    			gs=1;
    		}
    	}
    	if(i%b==0)
    	{
    		if((i/b)%2==0)
    		{
    			bs=0;
    		}
    		else
    		{
    			bs=1;
    		}
    	}
    	x=rs*100+gs*10+bs;
    	switch(x)
    	{
    		case 100: red++; break;
    		case 10: green++; break;
    		case 1: blue++; break;
    		case 110: yellow++; break;
    		case 11: cyan++; break;
    		case 101: magenta++; break;
    		case 111: white++; break;
    		case 0: black++; break;
    	}
    	
    }
    cout<<red<<" "<<green<<" "<<blue<<" "<<yellow<<" "<<cyan<<" "<<magenta<<" "<<white<<" "<<black;
    return 0;
}
