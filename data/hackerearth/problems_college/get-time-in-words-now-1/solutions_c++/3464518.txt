#include<bits/stdc++.h>
#include<cstdio>
using namespace std;
int main()
{
	string str[]={"zero","one","two","three","four","five","six","seven","eight",
		"nine","ten","eleven","twelve","thirteen","fourteen","fifteen","sixteen","seventeen",
		"eighteen","nineteen","twenty","twenty one","twenty two","twenty three",
		"twenty four","twenty five","twenty six","twenty seven","twenty eight",
		"twenty nine"};
	int hours,minutes;
	cin>>hours>>minutes;
	if(minutes==0)
		cout<<str[hours]<<" o' clock "<<endl;
	else if(minutes==15)
		cout<<"quarter past "<<str[hours]<<endl;
	else if(minutes==30)
		cout<<"half past "<<str[hours+1]<<endl;
	else if (minutes==45)
		cout<<"quarter to "<<str[hours+1]<<endl;
	else if(minutes<30)
	{
		if(minutes>=10)
			cout<<str[minutes]<<" minutes past "<<str[hours]<<endl;
		else 
			cout<<str[minutes]<<" minute past "<<str[hours]<<endl;
	}
	else 
	{
		if((60-minutes)>=10)
			cout<<str[60-minutes]<<" minutes to "<<str[hours+1]<<endl;
		else 
				cout<<str[60-minutes]<<" minute to "<<str[hours+1]<<endl;
	
	}

	return 0;
}
