#include <bits/stdc++.h>
using namespace std;

int foo(int n)
{
	int ret = 0;

  for(int i=1;i<=n;i++)
  {
	if(__gcd(n,i)==1)
    ret += 1;
  }   
  return ret;	
}


int SpecialSum(unsigned long n)
{
	unsigned long ret=0;

  for(int i=1;i<=n;i++)
  {

      if(n%i==0)

  }

 return ret;
}
