#include<cstdio>
long long int s, n, i, x, w[251], dp[251];

int main()
{
  scanf("%lld%lld", &s, &n);
  
  dp[0] = 1;
  
  while(n--)
  {
    scanf("%lld", &x);
    if (x <= s && w[x] == 0)
    {
      w[x] = 1;
      for(i = 0; i + x <= s; i++)
      {
	dp[i + x] += dp[i];
      }
    }
  }
  
  printf("%lld\n", dp[s]);
  
return 0;
}
