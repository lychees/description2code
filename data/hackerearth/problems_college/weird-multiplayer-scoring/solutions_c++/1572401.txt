#include <iostream>
#include <algorithm>
#include <utility>
#include<string>
using namespace std;
vector<pair<string,long long int> >p;
long long int mmax[101]={0};

int main()
{
   long long int N,M,i,j;
   long long int b;
   string a;
   long long int mmmx=-1,ind=-1;
   cin>>N>>M;
   for(j;j<M;j++)
   {cin>>a>>b;
   int flag=0;
   for(i=0;i<p.size();i++)
   {
   	if(a.compare(p[i].first)==0)
   	{
   		
   		p[i].second+=b;
   		if(p[i].second>mmax[i])
   		{
   			mmax[i]=p[i].second;
   			if(mmmx<mmax[i])
   			{
   				mmmx=mmax[i];
   				ind=i;
   			}
   		}
   		flag=1;
   		break;
   	}
   }
   pair <string,int>abc;
   if(flag==0)
   {
   	abc=make_pair(a,b);
   	p.push_back(abc);
   		if(b>mmax[p.size()-1])
   		{
   			mmax[p.size()-1]=b;
   		}
   		if(mmmx<mmax[p.size()-1])
   			{
   				mmmx=mmax[p.size()-1];
   				ind=p.size()-1;
   			}
   }
   }
   
   
   cout<<p[ind].first<<endl;
  	
   
    return 0;
}
