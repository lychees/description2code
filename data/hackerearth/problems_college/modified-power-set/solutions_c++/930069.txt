#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	scanf(" %d",&t);
	while(t--)
	{
		char s[1000001] = {0};
		scanf(" %s",s);
		long long int len = strlen(s);
		int a[150] = {0};
		int i = 0;
		while(i < len)
		{
			a[s[i]] = 1;
			i++;
		}
		i = 0;
		long long int count = 0;
		while(i < 150)
		{
			if(a[i] == 1)
				count++;
			i++;
		}
		printf("%lld\n",((count)*(count+1))/2);
	}
	return 0;
}
