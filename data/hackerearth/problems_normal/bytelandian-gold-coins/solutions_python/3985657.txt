import sys
arrmax=[0]*1000000
def getmax(coin):
    coin=int(coin)
    if coin < 1000000 and arrmax[coin]!=0:
    	return arrmax[coin]
    if coin >= 1000000000 or coin < 10:
        return coin
    else:
        maximum=max(coin,(getmax(coin//2)+getmax(coin//3)+getmax(coin//4)))
        if coin < 1000000:
            arrmax[coin]=maximum
        return maximum
    
for inpt in sys.stdin:
    coin=int(inpt)
    print(getmax(coin))
