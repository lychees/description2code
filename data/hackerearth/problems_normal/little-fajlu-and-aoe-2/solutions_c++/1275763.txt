    #include <iostream>
    #include <algorithm>
    #include <cassert>
    using namespace std;
    int main()
    {
    int test;
    cin>>test;
    assert(test>=1 && test<=100);
    while(test--)
    {
    int Food,Gold,S,P,C,LaalBaadshah,Samurai,Paladin,Champion,Answer=-1;
    cin>>Food>>Gold>>S>>P>>C>>LaalBaadshah;
    assert(Food>=1 && Food<=40000);
    assert(Gold>=1 && Gold<=40000);
    assert(S>=1 && S<=1000);
    assert(P>=1 && P<=1000);
    assert(C>=1 && C<=1000);
    for(Samurai=0;Samurai*100<=Food;Samurai++)
    {
    for(Paladin=0;Paladin*125+100*Samurai<=Food && Paladin*50<=Gold;Paladin++)
    {
    int Food_Left=(Food-100*Samurai-125*Paladin);
    int Gold_Left=(Gold-50*Paladin);
    Champion=min(Food_Left/50,Gold_Left/100);
    Answer=max(Answer,S*Samurai+P*Paladin+C*Champion);
    }
    }
    if(Answer>LaalBaadshah)
    cout<<Answer-LaalBaadshah<<endl;
    else
    cout<<-1<<endl;
    }
    return 0;
    }
