#include <cstdio>
#include <iostream>

using namespace std;

int main()
{
    int t, n, k;
    int count[100001], i, j;
    int temp;
    long int total;
    int a;
    cin>>t;
    while(t--)
    {
    	for(i=0; i<100001; i++)
    		count[i]=0;
    	total=0;
    	cin>>n;
    	//a=new int[n];
    	for(i=0; i<n; i++)
    	{
    		cin>>a;
    		count[a]++;
    	}
    	cin>>k;
    	for(i=1; i<=k/2; i++)
    	{
    		if(i==k-i)
    			temp=(count[i]*(count[i]-1))/2;
    		else
    			temp=count[i]*count[k-i];
    		total=total+temp;
    	}
    	cout<<total<<endl;
    }
    return 0;
}
