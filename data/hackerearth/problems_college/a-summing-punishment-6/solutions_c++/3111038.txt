#include<bits/stdc++.h>
using namespace std;

main(){
 ios_base::sync_with_stdio(0);
 long long n,q,l,r,a[100002]={0};
 cin>>n;
 for(int i=0;i<n;i++){
  cin>>a[i];
 }
 cin>>q;
 for(int i=0;i<q;i++){
  cin>>l>>r;
  long long sum=0;
  for(int j=l-1;j<r;j++){
   sum=sum+a[j];
  }
  cout<<sum<<endl;
 }
}
