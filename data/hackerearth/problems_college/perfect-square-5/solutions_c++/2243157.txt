#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    long long int t,n,ans,temp,temp2;
    long double t1;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	t1=sqrt(n);
    	temp=t1;
    	if(t1-temp==0)
    		cout<<"YES"<<endl;
    	else

    		temp2=(temp+1)*(temp+1);

    		if(temp2-n > n-temp)

    	}
    	
    }

}
