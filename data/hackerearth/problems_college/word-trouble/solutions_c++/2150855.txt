#include "iostream"
using namespace std;
string a,b;
int t,w,x,pom;
int main()
{
	ios_base::sync_with_stdio(0);
	cin>>t;
	while (t--)
	{
		cin>>a>>b;
		w=0;
		(a.size()<b.size()) ? x=a.size() : x=b.size();
		for (int i=0; i<x; i++)
		{
			if (a[i]!=b[i]) w++;
		}
		pom=a.size()-b.size();
		if (pom<0) pom*=(-1);
		w+=pom;
		if (w==0) cout<<"Exact\n";
		else cout<<"Non Exact "<<w<<"\n";
	}

	return 0;
}
