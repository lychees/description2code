#include <bits/stdc++.h>
using namespace std;

int main() {
	long long int t,i,j,k,l,sum;
	
	cin>>t;
	cin.ignore();
	while(t--){
	
		sum=0;
		
		vector<string> v;
		
		string str,s;
		
		getline(cin,str);
		
		k=str.size();
		
		s="";
		
		for(i=0;i<k;i++)
		{
			if(str[i]==' ' || i==k-1)
			{
				if(i==k-1)
				s+=str[i];
				
				//cout<<s<<" ";
							
				long long int r=s.size();
				//cout<<r<<" ";
				
				string str1="";
				
				for(j=0;j<r;j++)
				{
					if(s[j]>='0' && s[j]<='9')
					{sum+=(s[j]-'0');
					//cout<<s[j]<<" "<<sum<<" ";}
					}		
					//cout<<s[j];	
					
					else{
					str1+=s[j];
					//cout<<str1;}
					}
				}
				
				if(str1.size()>0)
				v.push_back(str1);
			
				s="";
			}
			
			else
			s+=str[i];
			
		}
	
	sort(v.begin(),v.end());
	
	for(i=0;i<v.size();i++)
	{
		string q=v[i];
		cout<<v[i]<<" ";
		long long int z=q.size();
		
		sum-=z;
		
		long long e=abs(sum);
			
		cout<<e<<" ";
	}
	cout<<"\n";
		
	//	cout<<v.size()<<" "<<sum<<"\n";
		
	}
}
