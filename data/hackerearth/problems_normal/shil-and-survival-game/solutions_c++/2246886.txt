const int INF = int(2e9);

class segment_tree{
	private:
		VI ST;
		VI A;	//base array
		int n;
		void build(int node,int l,int r)
		{
			if(l==r-1)
			{
				ST[node]=l;
				return;
			}
			int leftChild = node<<1;
			int rightChild = leftChild | 1;
			int mid = (l+r)>>1;
			build(leftChild,l,mid);
			build(rightChild,mid,r);
			//Operation of segment tree
			ST[node] = (A[ST[leftChild]]>A[ST[rightChild]]?ST[leftChild]:ST[rightChild]);
		}

		int range_query(int L,int R,int node,int l,int r)
		{
			if(l>=R || r<=L)
			//Depending on your operation return a value which indicates that there's nothing here
				return -1;
		
			int leftChild = node<<1;
			int rightChild = leftChild|1;
			if(L<=l && r<=R)
				return ST[node];
			int mid = (l+r)>>1;
			int la = range_query(L,R,leftChild,l,mid);
			int ra = range_query(L,R,rightChild,mid,r);
			//Operation of segment tree
			if(la==-1)return ra;
			if(ra==-1)return la;
			return (A[la]>A[ra]?la:ra);

		}


	public:
		segment_tree(){ST.clear();A.clear();n=0;}
		segment_tree(int *begin,int *end)	//construct using an array
		{
			A = VI(begin,end);
			n = SZ(A);
			ST = VI(4*n,0);
			build(1,0,n);
		}
		//returns [L,R)
		int query(int L,int R)
		{
			return range_query(L,R,1,0,n);
		}
};
const int N =int(1e5)+10;
int ans[N];
int len;
int A[N];
segment_tree S;
void solve(int l,int r,int flag)
{
	if(l>r)return;
	ans[len++]=S.query(l,r+1);
	if(l==r)return;
	if(!flag)
		solve(l,ans[len-1]-1,flag);
	else
		solve(ans[len-1]+1,r,flag);
}
int main()
{
	int n;si(n);
	for(int i=0;i<n;i++)
		si(A[i]);
	S = segment_tree(A,A+n);
	int mx = S.query(0,n);
	solve(0,mx-1,0);
	solve(mx+1,n-1,1);
	ans[len++]=mx;
	sort(ans,ans+len);
	for(int i=0;i<len;i++)
		printf("%d ",1+ans[i]);
	printf("\n");

	return 0;
}
