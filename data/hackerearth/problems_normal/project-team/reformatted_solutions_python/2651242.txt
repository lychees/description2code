def solve(students, n):
    maxknow = -1
    minknow = 999999999
    for i in xrange(n/2):
        pairknow = students[i] + students[n-i-1]
        maxknow = max(maxknow, pairknow)
        minknow = min(minknow, pairknow)
    return maxknow - minknow

    # print students
    n = students[0]
    students = sorted(students[1:])
    # print students
    print solve(students, n)
