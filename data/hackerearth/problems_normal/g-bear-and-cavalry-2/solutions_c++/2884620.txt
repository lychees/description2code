#include <bits/stdc++.h>

#define LLI long long int
#define LD long double
#define PB push_back
#define MP make_pair
#define FORi(i, a, b) for(int i = a; i < b ; ++i)
#define FORd(i, a, b) for(int i = a; i > b ; --i)

using namespace std;

const int mod = 1e9 + 7;
    
int main(){
    int t,n;
    bool is_stronger;
    cin >> t;
    vector<int> w(100),h(100);
    while (t--){
    	cin >> n;
    	FORi(i,0,n)
    		cin >> w[i];
    	FORi(i,0,n)
    		cin >> h[i];
    	is_stronger=0;
    	sort(w.begin()+1,w.begin()+n);
    	sort(h.begin(),h.begin()+n,greater<int>());
    	FORi(i,1,n){
    		if (w[i]*h[i]>=w[0]*h[0])
    			is_stronger=1;
    	}
    	if (is_stronger)
    		cout << "NO" << endl;
    	else
    		cout << "YES" << endl;
    }
    return 0;
}
