#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long a,b,p;
	scanf("%lld %lld %lld",&a,&b,&p);
	double ar=0.5*a*b;
	double ans=1000000000;
	double ar1=0.5*(double)((double)p/(double)100)*a*b;
	double low=0;
	double high=b;
	double mid;
	int co=0;
	while(low<high)
	{
		mid=(low+high)/(double)2;
		co++;
		double as=(0.5*(double)(b-mid)*(double)(a*b-(a*mid)))/(double)b;
		//printf("%lf %lf %lf %lf\n",low,high,mid,as);
		if(as>=ar1)
		{
			low=mid;
		}
		else
		{
			ans=min(ans,mid);
			high=mid;
		}
		if(co==100000)
		break;
	}
	//printf("%lf\n",ar1);
	/*for(double i=0;i<b;i+=0.01)
	{
		double as=(0.5*(double)(b-i)*(double)(a*b-(a*i)))/(double)b;
		if(as<=ar1)
		{
			ans=i;
			break;
		}
	}*/
	printf("%.2lf\n",ans);
}
