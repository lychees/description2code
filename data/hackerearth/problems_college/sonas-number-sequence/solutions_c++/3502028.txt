#include <iostream>
using namespace std;

int main()
{
    int i, j, k, n, m;
    cin>>n;
    int b[n];
    for(i=0;i<n;i++) {
    	cin>>b[i];
    }
    
    int sum = b[0];
    int a[n];
    a[0] = b[0];
    
    for(i=1;i<n;i++) {
    	int x = b[i]*(i+1);
    	a[i] = x - sum;
    	sum = sum + a[i];
    }
    
    for(i=0;i<n;i++)cout<<a[i]<<" ";
    return 0;
}
