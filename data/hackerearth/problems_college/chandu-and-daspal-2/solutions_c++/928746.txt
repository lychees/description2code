#include<bits/stdc++.h>
using namespace std;
#define gc getchar
inline long long int read()
{
	register char c = gc();
	while(c<'0' || c>'9') c = gc();
	long long int ret = 0;
	while(c>='0' && c<='9')
	{
		ret = 10 * ret + c - 48;
		c = gc();
	}
	return ret;
}
int gcd(int a, int b)
{
        int r;
        while(b != 0)
        {
                r = a % b;
                a = b;
                b = r;
        }
        return a;
}
int main()
{
    int t;
    t=read();
    while(t--)
    {
    	int a,b,c,i,j,k;
    	char x[10];
    	scanf("%d%d %s",&a,&b,x);
    	k=0;
    	while(a&&b)
    	{
    		c=gcd(a,b);
    		a=a-c;
    		b=b-c;
    		if(a==0||b==0)
    		{
    			if(k==0)
    			{
    				printf("%s\n",x);
    			}
    			else
    			{
    				if(x[0]=='C')
    				{
    					printf("Daspal\n");
    				}
    				else
    				{
    					printf("Chandu\n");
    				}
    			}
    			break;
    		}
    		k++;
    		k=k%2;
    	}
    	
    }
	return 0;
}
