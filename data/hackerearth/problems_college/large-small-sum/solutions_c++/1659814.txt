#include<iostream>
using namespace std;
int main()
{
    int t,n;
    long int a[1000],temp,l,h,i,j;
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(i=0;i<n;i++)
            cin>>a[i];
            cin>>l;
            cin>>h;
        for(i=0;i<n;i++)
        {
            for( j=1;j<n;j++)
        {
            if(a[j]>a[j-1])
            {
                temp=a[j];
                a[j]=a[j-1];
                a[j-1]=temp;
            }
        }
        }
        cout<<a[l-1]+a[n-h]<<"\n";
    }
}
