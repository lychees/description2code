#include "bits/stdc++.h"
using namespace std;
int main()
{
	long int tc;
	cin>>tc;
	while(tc--)
	{
		long int n;
		cin >> n;
		
		long int arr[2*n];
		long int n1=n*2;
		for(long int i=0;i<n1;i++)
			cin>>arr[i];
		
		sort(arr,arr+n1);
		
		long int brr[n];
		for(long int i=0,j=n1-1;i<j;i++,j--)
			brr[i] = arr[i] + arr[j];
		
		sort(brr,brr+n);
		cout<<brr[n-1]<<endl;
	}
	return 0;
}
