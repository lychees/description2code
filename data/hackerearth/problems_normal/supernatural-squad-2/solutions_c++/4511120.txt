int m, k;
int dp[201][201];

int solve(int n, int p){
	if (n < 0) return 0;
	if (n == 0) return 1;
	if (dp[n][p] != -1) return dp[n][p];
	dp[n][p] = 0;
	rep(i, k, p+1)
		dp[n][p] += solve(n-i, i);
	return dp[n][p];
}

    	rep(i, 0, m+1)
    		rep(j, 0, m+1)
				dp[i][j] = -1;
    	cout << solve(m, m) << endl;
    }
}
