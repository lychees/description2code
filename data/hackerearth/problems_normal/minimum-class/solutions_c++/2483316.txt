 #include <iostream>
 #include <algorithm>
 using namespace std;

 int max_sum(int n)
 {
     if(!(n%2))  return n*(n/2+1)/2;
     else return (n/2 + 1)*(n/2 + 1);
 }
 int a[100001];

 int main()
 {
     for(int i=0;i<100001;i++)
     {
         a[i] = max_sum(i);
         if(a[i] > (int)10e9) break;
         //cout << a[i] << endl;
     }

     int t;
     cin >> t;
     while(t--)
     {
         int n;
         cin >> n;
         int i=0;

         while(a[i] < n)
         {
             i++;
         }
         cout << i << endl;
     }
     return 0;
 }
