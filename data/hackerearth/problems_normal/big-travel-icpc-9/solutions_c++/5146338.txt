#include <bits/stdc++.h>

using namespace std;

int main()
{
    vector<long long> X;
    vector<long long> Y;
    long long n, x, y;
    cin >> n;
    for (int i = 0; i < n; ++i)
    {
        cin >> x >> y;
        X.push_back(x);
        Y.push_back(y);
    }

    sort(X.begin(),X.end());
    sort(Y.begin(),Y.end());

    long long sX = 0, sY = 0, sum = 0;

    for (int i = 1; i < X.size(); ++i)
    {
        sum += X[i-1];
        sX += (X[i]*i - sum)%1000000007;
    }

    sX %=  1000000007;
    sum = 0;
    for (int i = 1; i < Y.size(); ++i)
    {
        sum += Y[i-1];
        sY += (Y[i]*i - sum)%1000000007;
    }

    sY %=  1000000007;
    cout << (sX+sY)%1000000007 << endl;
    return 0;
}
