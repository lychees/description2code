#include <iostream>
using namespace std;
#include<bits/stdc++.h>
int main()
{
	int t,maxx ;
	cin>>t;
	while(t--)
	{
		maxx = 0;
		int n,maxi = 0;
		cin>>n;
		vector<int>arr(n);
		for(int i  =0;i<n;i++)cin>>arr[i];
		vector<int>dp(n);
		for(int i = 0;i<n;i++)
		{
			if(arr[i]>i)
			{
				maxi = 0;
				for(int j = 0;j<i;j++)
				{
					if(arr[i]>arr[j] && dp[j]>maxi)
					{
						maxi = dp[j];
					}
				}
				dp[i] = maxi+1;
			}
			else dp[i] = 1;
			if(dp[i]>maxx)maxx = dp[i];
		}
		cout<<n-maxx<<endl;
	}
    return 0;
}
