#include <cstdio>
#include <algorithm>
#define gc getchar_unlocked
using namespace std;

int main(){
	int T;
	scanf("%d",&T);
	while(T--){
		int bx,by,px,py;
		scanf("%d %d %d %d",&px,&py,&bx,&by);
		if(bx < by) swap(bx,by);
		if(px < py) swap(px,py);
		if((px >= bx && py > by) || (px > bx && py >= by)) printf("Possible\n");
		else printf("Not Possible\n");
	}
	return 0;
}
