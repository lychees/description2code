N=input()
if N==0:
	print 1,0
	print "We will win!"
	exit(0)
X=map(int,raw_input().split())
used=[0]*N;
#assert N>0
def countWays(i,l,r):
	if i==0:
		return l<=r
	if l>r: return 0	
	s=0	
	for x in range(N):
		if not used[x]:
			used[x]=1
			s+=countWays(i-1,l+X[x],r)
			s+=countWays(i-1,l,r+X[x])
			used[x]=0
	return s		
		
mess=["Got no way out!","We will win!"]		
res=countWays(N,0,0)
if N==0: res=0
print res, sum (X)
print mess[res>sum(X)]
exit(0)
