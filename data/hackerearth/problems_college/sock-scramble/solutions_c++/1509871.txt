#include <stdio.h>
#include<iostream>
using namespace std;
int main()
{
	int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n;
    	int temp;
    	scanf("%d",&n);
    	int a[n];
    	for(int i=0;i<n;i++)
    	{
    		scanf("%d",&a[i]);	
    	}
    	temp=a[0];
    	for(int i=1;i<n;i++){temp^=a[i];}
    	printf("%d\n",temp);
    }
}
