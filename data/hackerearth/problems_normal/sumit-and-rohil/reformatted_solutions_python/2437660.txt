l = []
for _ in range(int(raw_input())):
    s = raw_input()
    flag = 0
    for i in l:
        if (sorted(s) == sorted(i)) and (s[:1] == i[:1]) and (
                s[len(s)-1:] == i[len(i)-1:]):
            flag = 1
            break
    if flag == 0:
        l.append(s)
print len(l)
