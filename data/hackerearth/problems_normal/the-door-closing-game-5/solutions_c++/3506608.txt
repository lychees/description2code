#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(;t;t--)
    {
    	char s[105];
    	cin>>s;
    	int xr=0,cnt=0,flag=0;
    	for(int i=0;i<strlen(s);i++)
    	{
    		if(s[i]=='_') cnt++;
    		else if(s[i]=='|')
    		{
    			flag=1;
    			xr^=cnt;
    			cnt=0;
    		}
    	}
    	xr^=cnt;
       //cout<<xr<<endl;
      if(xr==0||flag==0) cout<<"Bob\n";
      else cout<<"Alice\n";
    }
}
