#include<iostream>
using namespace std;
int main()
{
	int t,n,k,i,j,x,m;
	cin>>t;
	while(t--)
	{
		
		cin>>n>>k;
		int arr[n];
		for(i=0;i<n;i++)
		cin>>arr[i];
		for(i=0;i<n;i++)
		{
			m=arr[i];
			x=i;
			if(k==0)
			break;
			for(j=min(i+k,n-1);j>i;j--)
			{
				if(m>arr[j]&&(k-(j-i))>=0)
				x=j,m=arr[j];
			}
			k-=(x-i);
			for(j=x;j>i;j--)
			{
				int tmp=arr[j];
				arr[j]=arr[j-1];
				arr[j-1]=tmp;
			}
			if(k==0)
			break;
		}
	for(i=0;i<n;i++)
	cout<<arr[i]<<" ";
	cout<<endl;
	}
	return 0;
}
