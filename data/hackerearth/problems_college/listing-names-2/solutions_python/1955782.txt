import sys

mydict = { }
test_case = int(raw_input())
while test_case > 0:
	value = raw_input()
	if mydict.has_key(value):
		mydict[value] = mydict[value] + 1
	else:
		mydict[value] = 1
	test_case = test_case - 1
	
mylist = mydict.keys()
mylist.sort()
for val in mylist:
	print "%s %d" %(val , mydict[val])
