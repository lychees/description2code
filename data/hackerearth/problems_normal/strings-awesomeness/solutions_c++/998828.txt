#include<cstdio>
#include<iostream>
#include<vector>
#include<algorithm>
#include<string>
#include<cstring>
using namespace std;

typedef long long LL;
typedef vector<int> VI;

#define REP(i,n) for(int i=0; i<int(n); i++)
#define EACH(i,c) for(__typeof((c).begin()) i=(c).begin();i!=(c).end();i++)


string S;
int Q;
int L, R;

int cnt[256], sums[500011];

int main() {
    cin >> S >> Q;

    REP (i, S.size()) {
	char c = tolower(S[i]);
	cnt[c]++;

	if (cnt['h'] == 2 &&
	    cnt['a'] == 2 &&
	    cnt['c'] == 1 &&
	    cnt['k'] == 1 &&
	    cnt['e'] == 2 &&
	    cnt['r'] == 2 &&
	    cnt['t'] == 1)
	    sums[i+1]++;

	if (i>=10) {
	    c = tolower(S[i-10]);
	    cnt[c]--;
	}
    }

    REP (i, S.size()) sums[i+1] += sums[i];

    REP ($, Q) {
	scanf("%d%d", &L, &R);
	L--;
	L += 10;
	L = min(L, R);
	printf("%d\n", sums[R] - sums[L]);
    }

    return 0;
}
