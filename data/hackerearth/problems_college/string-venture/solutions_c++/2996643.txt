#include<cstdio>
#include<cstring>
#include<cmath>
using namespace std;
int main()
{
	char a[100001];
	scanf("%[^\n]s",a);
	int w=0;
	int l=strlen(a);
	for(int i=0;i<l;i++)
		w=w+a[i];
	double x=w*1.0;
	double y=l*1.0;
	int c=(int)floor(x/y);
	int cnt[256]={0};
	if(c%2==1)
	{
		for(int i=0;i<l;i++)
		{
			int xi=a[i];
			cnt[xi]++;
		}
		int max=-1,ini=-1;
		for(int i=0;i<256;i++)
		{
			if(max<cnt[i])
			{
				max=cnt[i];
				ini=i;
			}
		}
		char ch=(char)(ini);
		printf("%c\n",ch);
	}
	else
	{
		for(int i=l-1;i>=0;i--)
		printf("%c",a[i]);
		printf("\n");
	}
	return 0;
}
