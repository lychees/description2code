#include <iostream>
#include<math.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int m;
		cin>>m;
		long long int i=0;
	  while(i*i<m)
	  {
	  	i++;
	  }
	  if(i*i==m)
	  {
	  	cout<<"YES"<<endl;
	  }
	  else
	  {
	  	long long int k=i*i-m;
	    long long int d=m-(i-1)*(i-1);
	    if(k<d)
	    cout<<"+"<<1*k<<endl;
	    else
	    cout<<"-"<<d<<endl;
	    
	  }
		
	}
   
    return 0;
}
