#include <iostream>
#include <cassert>
#include <math.h>
using namespace std;

int main()
{
    int t,n,i;
	long long int result;
	
	cin >> t;
	assert(1 <= t & t <= 42);
	for (i = 0; i < t; i++)
	{
		cin >> n;
		assert(1 <= n & n <= 42);

		result = (n*(n + 1) / 2)*pow(2, n - 1);
		cout << result << endl;
	}
	
    return 0;
}
