#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
typedef long long ll;
using namespace std;
const int maxn = 5100;

ll num[maxn], sum[maxn], dp[maxn][maxn];
ll n, m, k;

int main() {
	cin >> n >> m >> k;
	for (int i = 1; i <= n; i++) {
		cin >> num[i];
		sum[i] = sum[i-1] + num[i];
	}

	for (int i = m; i <= n; i++)
		for (int j = k; j >= 1; j--)
			dp[i][j] = max(dp[i-1][j], dp[i-m][j-1]+sum[i]-sum[i-m]);

	cout << dp[n][k] << endl;
	return 0;
}
