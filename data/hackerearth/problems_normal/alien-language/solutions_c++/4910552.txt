#include <bits/stdc++.h>
using namespace std;
int main()
{
   int test;
	cin>>test;
	while(test--)
	{
		string Text,Pattern;
		cin>>Text>>Pattern;
		bool HashText[26],HashPattern[26],Answer=false;
		memset(HashText,0,26);
		memset(HashPattern,0,26);
		for(int i=0;i<Text.size();i++)
			HashText[Text[i]-'a']=1;
		for(int i=0;i<Pattern.size();i++)
			HashPattern[Pattern[i]-'a']=1;
		for(int i=0;i<26;i++)
			if(HashText[i] && HashPattern[i])
				Answer=true;
		if(Answer)
			cout<<"YES\n";
		else
			cout<<"NO\n";
	}
	return 0;
}
