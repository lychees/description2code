#include <iostream>
#include <cstdio>
#include <queue>
using namespace std;

int main()
{
    int i,n,C[300],I[300],s,temp,coun=0,k;
    cin>>n;
    for(int i=0;i<n;i++)
        cin>>C[i];
    for( i=0;i<n;i++)
        cin>>I[i];
    queue <int> q;
    for( i=0;i<n;i++)
        q.push(C[i]);
    i=0;
    while(i<n)
    {
        if(q.front()==I[i])
        {
            i++;
            q.pop();
            s++;
            coun++;
        }
        else {
            temp=q.front();
            q.push(temp);
            q.pop();
            coun++;
        }
    }
    cout<<coun<<endl;
    return 0;
}
