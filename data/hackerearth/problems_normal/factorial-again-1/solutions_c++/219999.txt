#define ll long long
#define vt vector

ll power(ll a,ll b, ll c)
{
	ll x=1,y=a;
	while(b>0)
	{
		if(b%2)
			x=(x*y)%c;
		y=(y*y)%c;
		b/=2;
	}
	return x;
}

ll invmod(ll a,ll b)
{
	return power(a,b-2,b);
}

ll factmod(ll a,ll b)
{
	ll ans=1;
	if(a>=b)
		return 0;
	for(ll i=a+1;i<=b-1;i++)
		ans = (ans*i)%b;
	ans = ((b-1)*invmod(ans,b))%b;
	return ans;
}

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		ll n,p;
		scanf("%lld %lld",&n,&p);
		ll ans = power(6,n,p);
		ans = power(ans,p-2,p);
		ll temp = factmod(3*n,p);
		cout << (ans*temp)%p << endl;
	}
	return 0;
}
