#include <iostream>
#include <vector>
#include<algorithm>

using namespace std;

int main() {
	int m;
	int n,i,x,t;

	vector<int> v,g;
	
	

	cin>>n;

	for(i=0;i<n;i++)
	{ cin>>x; v.push_back(x); }

	sort(v.begin(),v.end());

	g.push_back(v[0]);

	i=1;

	while(i<v.size())
	{ g.push_back(v[i]);
	 i++;

	if(i<v.size())
	g.insert(g.begin(),v[i]);

	i++;
	}

	t= abs(g[0]-g[g.size()-1]);

	for(int i=1;i<g.size();i++)
	{if(t<abs(g[i-1]-g[i]))
	t=abs(g[i-1]-g[i]);}

	cout<<t<<"\n";

	v.clear(); g.clear();

	

	return 0;

}
