#include<iostream>
#include<cstdio>
#include<vector>
#include<algorithm>
#include<map>
using namespace std;
int main()
{
	char string[10001];
	char removedPart[27];
	map<char , int> hash;
	int T , i , totalWeight , N;
	char ch = 'a';
	cin >> T;
	while(T > 0){
		cin.ignore();
		gets(string);
		cin >> N;
		for(i = 0 ; i < N ; i++){
			cin >> removedPart[i];
		}
		ch = 'a';
		for(i = 1 ; i < 27 ; i++ , ch++){
			hash[ch] = i;
		}
		totalWeight = 0;
		i = 0;
		while(i < N){
			hash[removedPart[i]] = 0;
			i++;
		}
		i = 0;
		while(string[i] != '\0'){
			totalWeight += hash[string[i]];
			i++;
		}
		cout<< totalWeight << "\n";
		hash.clear();
		T--;
	}
	return 0;
}
