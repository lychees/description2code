#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t,a[3];
    cin>>t;
    while(t--){
    	for(int i=0;i<3;i++){
    		cin>>a[i];
    	}
    	sort(a,a+3);
    	int left = (a[1]-a[0]-1);
    	int right = (a[2]-a[1]-1);
    	cout<<((left>=right)?left:right)<<endl;
    }
    return 0;
}
