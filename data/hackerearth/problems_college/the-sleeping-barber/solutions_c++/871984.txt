#include<iostream>
#include<string>
#include<stdlib.h>
using namespace std;
#define MyType long long int
int compare(const void * a, const void * b)
{
  if ( *(MyType*)a <  *(MyType*)b ) return -1;
  if ( *(MyType*)a == *(MyType*)b ) return 0;
  if ( *(MyType*)a >  *(MyType*)b ) return 1;
}

long long int a[3000];

main()
{
    long long int d,g,m,count;
    cin>>d;
    while(d--)
    {
        count=0;
        cin>>g>>m;
        for(int i=0;i<g;i++)
            cin>>a[i];
        qsort(a,g,sizeof(long long int),compare);
        for(int i=0;i<g;i++)
        {
            if(m<=0) break;
            if(a[i]<=m)
            {
                m=m-a[i];
                count++;
            }
        }
        cout<<count<<endl;
    }
    return 0;
}
