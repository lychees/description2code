#include <iostream>
#include <string>
using namespace std;

char check(int i){
    char ch;
    if(i>0 && i<=9){
        ch = i+'0';
        return ch;
    }
    switch(i){
        case 10 :   ch = 'U';
                    break;
        case 11 :   ch = 'V';
                    break;
        case 12 :   ch = 'W';
                    break;
        case 13 :   ch = 'X';
                    break;
        case 14 :   ch = 'Y';
                    break;
        case 15 :   ch = 'Z';
                    break;

    }
    return ch;
}

void convert(int n){
    string s;
    if(n == 0){
        cout << "00";
        return;
    }
    while(n > 0){
        int rem = n % 16;
        char ch = check(rem);
        s.push_back(ch);
        n = n / 16;
    }
    if(s.size() == 1){
        cout << "0" << s;
    } else {
        for(int i=s.size()-1 ; i>=0 ; i--)
        cout << s[i];
    }
}

int main(){
    int r,g,b,t,i=1;
    cin >> t;
    while(i<=t){
        cin >> r >> g >> b;
        cout << "Case " << i << ": ";
        convert(r);
        cout << ' ' ;
        convert(g);
        cout << ' ' ;
        convert(b) ;
        cout << '\n';
        i++;
    }
    return 0;
}
