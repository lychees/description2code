#include<bits/stdc++.h>
#include<cstring>
using namespace std;
char a[1000001],b[1000001],c[1000001];
main()
{
	long long t;
	scanf("%lld",&t);
	while(t--){
	int l,i,j,flag=0;
scanf("%s",a);
l=strlen(a);
for(i=l-1,j=0;i>=0;i--,j++)
{
	b[j]=a[i];
}
b[j]='\0';
strcpy(c,b);

for(i=0;i<l-1;i++)
{
	if(abs((int)a[i+1]-(int)a[i])==abs((int)c[i+1]-(int)c[i]))
	flag=1;
	else
	{
		flag=0;
		printf("Not Funny\n");
		break;
	}
}
if(flag==1)
printf("Funny\n");}
}
