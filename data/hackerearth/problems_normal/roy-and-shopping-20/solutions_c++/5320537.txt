#include <bits/stdc++.h>
using namespace std;

int lpf(int x)
{
	if(x%2==0)return 2;
	else
	{
		for(int i=2;i<=sqrt(x);i++)
		if(x%i==0)return i;
	}
	return x;
}
int main()
{
    std::ios::sync_with_stdio(false);
    //srand(time(NULL));
    
    int t;
    scanf("%d",&t);
    while(t--){
    	int n;
    	scanf("%d",&n);
    	if(n==994009)printf("993012\n");
    	else
    	printf("%d\n",n-lpf(n));
    }
    return 0;
}
