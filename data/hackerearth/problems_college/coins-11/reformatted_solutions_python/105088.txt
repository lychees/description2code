m = 100005
a = [True]*m
a[0] = a[1] = False
for (i, p) in enumerate(a):
    if (p):
        for j in range(2*i, m, i):
            a[j] = False
t = input()
t = int(t)
while t > 0:
    n = input()
    n = int(n)
    if n == 3:
        print("3")
    elif a[n]:
        print("2")
    else:
        print("1")
    t = t-1
