#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,n;
	double x, y;
	cin>>t;
	while(t--)
	{
		cin>>n;
		int m = n;
		double sumx = 0.0;
		double sumy = 0.0;
		while(n--)
		{
			cin>>x>>y;
			sumx += x;
			sumy += y;
		}
		sumx /= m;
		sumy /= m;
		printf("(%.2lf , %.2lf)\n",sumx,sumy);
	}
	return 0;
}
