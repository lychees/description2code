#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

double a,b;
vector<double> v1,v2;

vector<double> solveIt(vector<double> v){
	vector<double> res;
	res.resize(3);
	double sum = v[0]+v[1]+v[2];
	res[0] = sum - 2*v[1];
	res[1] = sum - 2*v[2];
	res[2] = sum - 2*v[0];
	return res;
}



int main()
{
	int i;
	for(i=1;i<=3;i++){
		cin>>a>>b;
		v1.push_back(a);
		v2.push_back(b);
		
	}    
    
    vector<double> V1 = solveIt(v1);
    vector<double> V2 = solveIt(v2);
    
    vector<pair<double,double>> ans;
    
    for(i=0;i<3;i++){
    	ans.push_back(make_pair(V1[i] , V2[i]));
    }
    
    sort(ans.begin(),ans.end());
        
    for(i=0;i<3;i++){
    	printf("%.4f %.4f\n",ans[i].first,ans[i].second);
    }
    
    return 0;
}
