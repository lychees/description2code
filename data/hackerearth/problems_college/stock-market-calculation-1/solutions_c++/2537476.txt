#include <bits/stdc++.h>
using namespace std;
#define gc() getchar_unlocked()
inline long long  int read_int()
{
    char c = gc();
    while(c<'0' || c>'9')
    c = gc();
    long long int ret = 0;
    while(c>='0' && c<='9')
    {
    ret = (ret<<3) + (ret<<1) + c - 48;
    c = gc();
    }
    return ret;
} 
int main()
{
	long long int t,n,profit,i;
	t = read_int();
	while(t--)
	{
		n = read_int();
		long long int a[n],sell[n];
		profit = 0;
		for(i=0;i<n;i++)
		{
			a[i] = read_int();
		}
		for(i=n-1;i>=0;--i)
		{
			if(i==n-1)
			{
     				sell[i]=a[i];
     			}
    			else
    			{
     				sell[i]=max(sell[i+1],a[i]);
     			}
         	}
		for(int i=0;i<n;i++)
		{
    			profit = profit+(sell[i]-a[i]);
    		}
		cout << profit << endl;
	}
    	return 0;
}
