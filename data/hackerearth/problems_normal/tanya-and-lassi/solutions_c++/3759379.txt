#include<iostream>
#include<string.h>
using namespace std;
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    long int t,j,i,l;
    cin>>t;
    while(t--)
    {
        cin>>l;
        long int arr[l+1];
        for(i=0;i<l;i++)
            cin>>arr[i];
        long int dp[l+1];
        memset(dp,0,sizeof(dp));
        for(i=1;i<=l;i++)
            for(j=1;j<=i;j++)
                dp[i]=max(arr[j-1]+dp[i-j],dp[i]);
            cout<<dp[l]<<"\n";
    }
    return 0;
}
