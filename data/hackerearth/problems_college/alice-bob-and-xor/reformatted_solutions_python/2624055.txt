def f(a):
    res = [a, 1, a+1, 0]
    return res[a % 4]


def getXor(a, b):
    return f(b) ^ f(a-1)


tt = raw_input()
for x in range(0, int(tt)):
    line = raw_input()

    w = line.split(' ')
    l = int(w[0])
    r = int(w[1])

    ans = getXor(l, r)

    print ans
