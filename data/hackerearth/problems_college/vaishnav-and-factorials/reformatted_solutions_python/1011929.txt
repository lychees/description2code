def fact(n):
    f = 1
    while n > 0:
        f = f*n
        n -= 1
    return f


def count(ans):
    n = len(ans)
    i = 0
    count = 0
    while i < n:
        if ans[i] == '4' or ans[i] == '7':
            count += 1
        i += 1
    return count
