#include<iostream>
#include<vector>
using namespace std;
int main()
{
	unsigned long long n;
	long long int c[3],input;
	cin>>n;
	vector<long long int> vec;
	for(int i=1;i<=n;i++)
	{
	 cin>>input;
	 vec.push_back(input);
    }
	for(int j=0;j<3;j++)
	{    c[j]=0;
		for(int i=j;i<vec.size();i+=3)
		  c[j]=c[j]+vec[i] ;
	        
	       
    }
	    for(int i=0;i<3;i++)
	      cout<<c[i]<<" ";
	
	 
	
	 return 0;
}
