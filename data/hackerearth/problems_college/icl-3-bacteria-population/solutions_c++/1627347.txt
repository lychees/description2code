#include<stdio.h>
#include<queue>
using namespace std;

int main()
{
   int n;
   queue <long long int> que;
   long long int ans=0,back,c;
   scanf("%d",&n);
   if(n==0 || n==1)
   ans=1;
   else if(n==2 || n==3)
   ans=2;
   else if(n==4)
   ans=3;
   else
   {
   	que.push(1);
   	que.push(1);
   	que.push(2);
   	que.push(2);
   	que.push(3);
   	c=4;
   	back=3;
   	while(c!=n)
   	{
   		ans=back+(que.front());
   		que.pop();
   		back=ans;
   		que.push(ans);
   		c++;
   	}
   }
   	printf("%lld\n",ans);
    return 0;
}
