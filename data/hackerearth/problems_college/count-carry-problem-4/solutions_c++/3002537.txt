#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main(){
	int t;
	cin>>t;
	while(t--){
		int carry=0;
		string str1;
		string str2;
		cin>>str1>>str2;
		int count_carry=0;
		int lim=min(str1.length(),str2.length());
		for(int i=0;i<lim;i++){
			if((carry+str1[i]+str2[i]-2*48)>=10){
				carry=(carry+str1[i]+str2[i]-2*48)/10;
				count_carry++;
			}
			else{
				carry=0;
			}
		}
		if(str1.length()>lim){
			while(lim!=str1.length()){
				if(str1[lim]+carry>=10&&lim<str1.length()-1){
					carry=(str1[lim]+carry)/10;
					count_carry++;
				}
				else
					break;
				lim++;
			}
		}
		else{
			while(lim!=str2.length()){
				if(str2[lim]+carry>=10){
					carry=(str2[lim]+carry)/10;
					count_carry++;
				}
				else
					break;
				lim++;
			}
		}
		if(!count_carry)
			cout<<"No carry operation"<<endl;
		else if(count_carry==1)
		cout<<"1 carry operation"<<endl;
		else{
			cout<<count_carry<<" carry operations"<<endl;
		}
	}
}
