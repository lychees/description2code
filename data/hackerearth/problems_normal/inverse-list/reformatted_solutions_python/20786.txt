t = int(raw_input())
for i in xrange(0, t):
    flag = 0
    list = []
    listlen = int(raw_input())
    strin = raw_input()
    y = strin.split()
    for j in xrange(0, listlen):
        list.append(int(y[j]))

    for j in xrange(0, listlen):
        if list[list[j]-1] != j+1:
            flag = 1
            break

    if flag == 0:
        print "inverse"
    else:
        print "not inverse"
