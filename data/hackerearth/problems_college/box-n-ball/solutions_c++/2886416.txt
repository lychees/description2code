#include <iostream>
#include <iomanip>
using namespace std;

int main()
{
	int t;
	long long int l,b,h;
	
	cin>>t;
	
	while(t--){
		cin>>l>>b>>h;
		
		long long int temp= (l<b)?l:b;
		long long int smallest= (temp<h)?temp:h;
		
		double radius= (double)smallest/2.0;
		
		cout<<fixed<<setprecision(1)<<radius<<"\n";
	}
    return 0;
}
