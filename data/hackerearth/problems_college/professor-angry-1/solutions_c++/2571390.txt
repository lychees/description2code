#include<iostream>
using namespace std;
int m=0;
int main()
{
    int t,n,k,count1;
    cin>>t;
    char result[t];
    while(t>0)
    { count1=0;
        cin>>n>>k;
        int input[n];
        for(int i=0;i<n;i++)
            cin>>input[i];
        for(int i=0;i<n;i++)
        {
            if(input[i]<=0)
            {count1++;}
        }
        if(count1>=k)
            result[m]='N';
        else
            result[m]='Y';
        m++;
        t--;
    }
    for(int i=0;i<m;i++)
    {
        if(result[i]=='Y')
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
    }
}
