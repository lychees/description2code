#include <iostream>

using namespace std;

int main()
{
    int t;
    long n,m,A,B;
    long long ans,C;
    cin>>t;
    for(int i=0;i<t;i++)
    {
        cin>>n>>m>>A>>B>>C;
        ans = 2*min(n,m)*C;
        if(n>m)
            ans = ans + (n-m)*A;
        else
            ans = ans + (m-n)*B;
        C = n*A+m*B;
        ans = max(C,ans);
        cout<<ans<<endl;
    }
    return 0;
}
