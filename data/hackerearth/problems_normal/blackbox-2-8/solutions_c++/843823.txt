

#include<stdio.h>
#include<iostream>
using namespace std;

int fact[1005][3505];

int main()
{
	fact[0][3000]=1;
	for(int i=1;i<=1000;i++)
	{
		int c=0;
		for(int j=3000;j>=0;j--)
		{
			int rem=fact[i-1][j]*i+c;
			fact[i][j]=rem%10;
			c=rem/10;
		}
	}
	int T;
	cin>>T;
	while(T--)
	{
		int N;
		cin>>N;
		int s=0;
		for(int j=0;j<=3000;j++)
		s+=fact[N][j];
		cout<<s<<endl;
	}
}
