#include <bits/stdc++.h>
#define foru(i, a, b) for(int i = a; i <= b; ++i)
using namespace std;
typedef unsigned long long qword;
typedef vector<vector<qword> > matrix;
const qword BASE = 1LL << 32;
matrix a;
qword n, ans;
qword c[252][252];

void Mul(matrix &a, matrix &b) {
	foru(i, 0, 251) foru(j, 0, 251) {
		c[i][j] = 0;
		foru(k, 0, 251) c[i][j] = (c[i][j] + a[i][k] * b[k][j]) % BASE;
	}
	foru(i, 0, 251) foru(j, 0, 251) a[i][j] = c[i][j];
}

matrix Pow(qword n) {
	matrix ans; ans = a; --n;
	while (n) {
		if (n & 1) Mul(ans, a);
		Mul(a, a); n >>= 1;
	}
	return ans;
}

int main() {
	//freopen("input.txt", "r", stdin);
	//freopen("output.txt", "w", stdout);
	cin >> n;
	if (n == 1) {
		cout << 9; return 0;
	}

	a.resize(252); foru(x, 0, 251) a[x].resize(252);
	foru(x, 0, 251) foru(c, 1, 9)
		if ((10 * x + c) % c == 0) {
			int y = (10 * x + c) % 252;
			a[x][y] = 1;
		}

	a = Pow(n - 1); 
	foru(x, 0, 251) foru(c, 1, 9) 
		if ((10 * x + c) % c == 0) {
			ans = (ans + a[0][x]) % BASE;
		}
	cout << ans;
	return 0;
}
