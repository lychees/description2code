#include<bits/stdc++.h>
#include<iostream>
using namespace std;
long long modular(long long base, long long exponent, int modulus)
{
    long long result = 1;
    while (exponent > 0)
    {
        if (exponent % 2 == 1)
            result = (result * base) % modulus;
        exponent = exponent >> 1;
        base = (base * base) % modulus;
    }
    return result;
}
int main()
{
    long long t;
    cin>>t;
    while(t--)
    {
        long long a,b,k,sum=0,i,z;
        cin>>a>>b>>k;
        long long x=modular(a,b,pow(10,k));
        //cout<<x<<endl;
        for(i=0;i<k;i++)
        {
            z=x%10;
            sum=sum+z;
            x=x/10;
        }
        cout<<sum<<endl;
    }
}
