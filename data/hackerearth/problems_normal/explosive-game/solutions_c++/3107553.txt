#include <bits/stdc++.h>

using namespace std;
int main(){
    long long int n,q;
    cin>>n>>q;
    long long int ans = 0;
    map<long long int,long long int> mapa;
    mapa.clear();
    for(long long int i=0;i<n;i++){
        long long int ai,bi;
        cin>>ai>>bi;
        long long int ci = abs(ai-bi);
        if(ai==bi){
            ans+=ai;
        }
        else if(mapa[ci]%2==0){
            ans+= max(ai,bi);
            mapa[ci]++;
        }else{
            ans+= min(ai,bi);
            mapa[ci]++;
        }
    }
    for(map<long long int,long long int>::iterator it=mapa.begin(); it!=mapa.end(); ++it){
        if(it->second%2!=0){
            ans=-1;
        }
    }
    for(long long int i=0;i<q;i++){
        long long int query;cin>>query;
        if(query==ans)cout<<"YES"<<endl;
        else{
            cout<<"NO"<<endl;
        }
    }

}
