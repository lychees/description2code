#include<iostream>
using namespace std;
int main()
{
    long long n,minodd=1000000000,mineven=1000000000,i,sum=0;
    cin>>n;
    int a[n];
    for(i=0;i<n;i++)
    {
        cin>>a[i];
        if(a[i]%2==0&&mineven>a[i])
        {
            mineven=a[i];
        }
        if(a[i]%2!=0&&minodd>a[i])
        {
            minodd=a[i];
        }
        sum+=a[i];
    }
    if(sum%2==0)
    {
        if((sum-minodd)%2!=0)
        {
            sum-=minodd;
        }
        else if((sum-mineven)%2!=0)
        {
            sum-=mineven;
        }
        else
        {
            cout<<":(\n";
        }
    }
    if(sum%2!=0)
        cout<<sum<<"\n";

}
