#include<bits/stdc++.h>
using namespace std;
main()
{
	int count[100000]={0};
	int n;
	cin>>n;
	int arr[n];
	for(int i=0;i<n;i++)
	{
		cin>>arr[i];
		count[arr[i]]++;
	}
	sort(arr,arr+n);
	int t;
	cin>>t;
	while(t--)
	{
		int q;
		cin>>q;
		int flag=0;
		for(int i=0;i<n;i++)
		{
			if(q==arr[i] && count[arr[i]]==1)
			{
				flag=1;
				break;
			}
			else if(q==arr[i])
			{
				flag=7;
				break;
			}
		}
		if(flag==0 && flag!=7)
		{
			for(int i=0;i<n;i++)
			{
				for(int j=i+1;j<n;j++)
				{
					if(arr[i]+arr[j]==q)
					{
						flag=1;
						break;
					}
				}
			}
		}
		if(flag==1)
		cout<<"I Like You"<<endl;
		else
		cout<<"I Hate You"<<endl;
	}
}
