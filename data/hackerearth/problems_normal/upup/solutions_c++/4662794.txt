#include <bits/stdc++.h>
using namespace std;

int main() 
{
	string s;
	getline(cin,s);
	int k = s.length();
	for(int i=1;i<k;i++)
	{
		if(s[0]>=96 && s[0]<=122)
		{
			s[0]=s[0]-32;
		}
		if(s[i]==' ')
		{
			if(s[i+1]>=97 && s[i+1]<=122)
			{
				s[i+1]=s[i+1]-32;
			}
			
		}
	}
	cout<<s<<endl;
	return 0;
}
