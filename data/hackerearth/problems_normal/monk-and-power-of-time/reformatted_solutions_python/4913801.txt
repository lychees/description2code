from collections import deque

n = input()
s = raw_input().split(' ')
s = deque(s)
req = raw_input().split(' ')
i = 0
count = 0
while True:
    try:
        p = s.popleft()
    except BaseException:
        break
    if req[i] == p:
        i += 1
        count += 1
    else:
        s.append(p)
        count += 1
print count
