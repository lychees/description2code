#include<iostream>
#include<bits/stdc++.h>
using namespace std;
long long p2(long long n) {
    return pow(2, floor(log(n) / log(2)));
}
int main()
{
    int n,i;
    long long a,k,l,ans;
    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>a;
        k=p2(a);
        l=a-k;
        ans=2*l+1;
        cout<<ans<<endl;
    }
}
