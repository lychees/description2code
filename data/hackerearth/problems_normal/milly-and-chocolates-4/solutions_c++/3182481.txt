#include <iostream>
// #include<cstdio>
#include <bits/stdc++.h>
using namespace std;

#define lli long double

int main()
{
	int t;
	cin>>t;
	lli n,sum,temp;
	while(t--){
		sum=0;
		cin>>n;
		for(lli i=0;i<n;i++){
			scanf("%Lf",&temp);
			sum+=temp;
		}
		sum/=n;
		printf("%.8Lf\n",sum);
	}
    return 0;
}
