#include<stdio.h>
int main()
{
    int t;

    int n,i,j,temp,m,x;

    scanf("%d",&t);
    while(t--)
    {
        int a[20000];
       scanf("%d",&n);
       a[0]=1;
       m=1;

       temp = 0;
       for(i=1;i<=n;i++)
       {
            for(j=0;j<m;j++)
            {
               x = a[j]*i+temp;
               a[j]=x%10;
               temp = x/10;
            }
             while(temp>0)
             {
               a[m]=temp%10;
               temp = temp/10;
               m++;
             }
      }
      long int sum = 0;
              for(i=m-1;i>=0;i--)
                sum += a[i];
                //printf("%d",a[i]);

              printf("%ld\n",sum);
    }
    return 0;
}
