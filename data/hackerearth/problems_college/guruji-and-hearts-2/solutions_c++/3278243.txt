#include <bits/stdc++.h>
using namespace std;
char str[102];
int main()
{
    
    
    
    cin>>str;
    int n ,count1;
    count1 = 0;
    n = strlen(str);
    for(int i = 0; i < n-1; i++){
    	if(str[i] == '<' && str[i+1] == '3'){
    		count1++;
    	}
    }
    
    cout<<count1<<endl;
    return 0;
}
