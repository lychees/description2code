#include <iostream>
using namespace std;

int g[100001];
int f[100001];

int main() {
	for (int i = 2; i <= 100000; i++) {
		f[g[1]] = i;
		for (int j = 2; j*j <= i; j++) if (i % j == 0) {
			f[g[j]] = i;
			f[g[i/j]] = i;
		}
		for (int j = 0; true; j++) if (f[j] != i) { g[i] = j; break; }
	}
	int res = 0;
	int n; cin >> n;
	while (n--) {
		int x; cin >> x;
		res ^= g[x];
	}
	cout << (res == 0 ? "BOB" : "ALICE") << endl;
    return 0;
}
