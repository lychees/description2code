#include<bits/stdc++.h>

using namespace std;

int gcd(int a ,int b){
if( b == 0 ) 
    return a;
 else
   return gcd(b , a % b);
}

int main(){
int n,i,j;
int a[100005]; 
long int c = 0;
cin>>n;

for( i = 0 ; i < n ; i++)
   cin>>a[i];

for( i = 0 ; i < n ; i++ ){
   for( j = i + 1 ; j < n ; j++){
          if((a[i] > a[j] ) && gcd(a[i],a[j]) == 1)
                c++;  

             }
 
}
cout<<c;

return 0;
}
