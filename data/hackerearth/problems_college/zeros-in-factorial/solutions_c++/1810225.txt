#include <iostream>
using namespace std;
#define max 1000000
long int arr[max];

int main()
{
    long int i,j;
    arr[0]=arr[1]=arr[2]=arr[3]=arr[4]=0;
    for(i=5;i<=max;i++)
    {
    	long int count=0;
    	for(j=5;i/j>=1;j*=5)
    	{
    		count+=(i/j);
    	}
    	arr[i]=count+arr[i-1];
    }
    long int t;
    cin>>t;
    while(t--)
    {
long int a,b;
    	cin>>a>>b;
        cout<<arr[b]-arr[a-1]<<endl;
    }
    return 0;
}
