#include <iostream>
using namespace std;
#include <stdio.h>

int main()
{
    int a,b,c,x,y;
    cin>>a>>b>>c;
    for(int i = -5;i<=6;i++){
    	 for(int j = -5;j<=6;j++){
    	 	if(a*i + b*j + c == 0){
    	 		x = i;
    	 		y = j;
    	 		break;
    	 	}	
    	 }
    	 
    }
    if(x<-5 || y<-5) cout<<-1;
    else cout<<x<<" "<<y;
    return 0;
}
