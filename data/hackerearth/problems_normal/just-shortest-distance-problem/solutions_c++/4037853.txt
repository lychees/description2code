#include<bits/stdc++.h>
using namespace std;
#define ull long long int
vector<vector<ull> >graph;
ull dist[1005];
void update(ull b)
{
    for(int j=0;j<graph[b].size();j++)
    {
        if(dist[graph[b][j]]>dist[b]+1)
        {
            dist[graph[b][j]]=dist[b]+1;
            update(graph[b][j]);
        }
    }
    return;
}
int main()
{
    ios::sync_with_stdio(false);
    ull n,m,i,x,y,k;
    cin>>n>>m;
    graph.resize(n+1);
    for(i=2;i<=n;i++)
        dist[i]=1e9;
    dist[1]=0;
    while(m--)
    {
        cin>>k;
        if(k==2)
        {
            cin>>x>>y;
            graph[x].push_back(y);
            if(dist[y]>dist[x]+1)
            {
                dist[y]=dist[x]+1;
                update(y);
            }
        }
        else if(k==1)
        {
            cin>>x;
            if(dist[x]>1e8)
                cout<<"-1\n";
            else
                cout<<dist[x]<<"\n";
        }
    }
    return 0;
}
