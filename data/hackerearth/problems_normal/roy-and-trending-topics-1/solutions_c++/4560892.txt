#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long n,id,od,po,l,c,s;
    cin >> n;
    map<long long,long long> m;
    priority_queue<pair<long long,long long>> p;
    while(n--)
    {
    	cin >> id >> od >> po >> l >> c >> s;
    	long long x=(po*50+l*5+c*10+s*20);
    	long long v=abs(x-od);
    	m[id]=x;
    	
    	p.push(make_pair(v,id));
    	
    }
    for(int i=0;i<5;i++)
    {
    	cout << p.top().second <<" "<< m[p.top().second] << endl;
    	p.pop();
    }
    return 0;
}
