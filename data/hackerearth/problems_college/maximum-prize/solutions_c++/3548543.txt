#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll a[1<<20];
int main()
{
	int tc;
	cin>>tc;
	while(tc--)
	{
		ll n,k;
		cin>>n>>k;
		for(int i=1;i<=n;++i)
			cin>>a[i];
		sort(a+1,a+n+1);
		reverse(a+1,a+n+1);
		ll ans=0;
		for(int i=1;i<=k && i<=n;++i)
		{
			if(a[i]>=0)
			ans+=a[i];
		}
		cout<<ans<<endl;
	}


	return 0;
}
