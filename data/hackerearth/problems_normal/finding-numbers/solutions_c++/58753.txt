#include <iostream>
#include<algorithm>
#include<stdlib.h>
#include <stdlib.h>
#include <time.h>
#include<stdio.h>
#include<math.h>
#define LL unsigned long long
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	LL m,n,z,count,i,j;
	while(t--)
    {
        count=0;
        scanf("%lld%lld%lld",&z,&m,&n);
        for(i=1;i*i<=z;i++)
        {
            if(z%i==0)
            {
               // printf("%lld %lld\n",i,z/i);
                if(i==z/i)
                    count++;
                else if((i<=m && i<=n) && (z/i<=n && z/i<=m))
                    count+=2;
                else if((i<=m || i<=n) && (z/i<=n || z/i <=m))
                    count++;
            }
        }
        printf("%lld\n",count);
    }
	return 0;
}
