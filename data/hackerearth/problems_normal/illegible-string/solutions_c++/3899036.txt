#include<iostream>
using namespace std;
#include<string.h>
int main()
{
	long long int i,m=0,d=0,k=0,ncv=0,cw=0,r=0,n,min,max,t,b,f=0,cv=0;
	string s;
	cin>>t;
	cin>>s;
	i=0;
	n=t;
	
	while(i<=n)
	{
		if(s[i]=='v')
		{
			cv++;
//			cout<<"cv="<<cv<<endl;
			m++;
//			cout<<"m="<<m<<endl;
			i++;
//			cout<<"i="<<i<<endl;
			if(m==2)
			{
				d=d+m;
 //  			cout<<"d="<<d<<endl;
   			    k=k+1;
//				cout<<"k="<<k<<endl;
				m=0;
			}
			if(i==n)break;
		}
		else if(s[i]=='w')
		{
			cw++;
//		    cout<<"cw="<<cw<<endl;
			i++;
//			cout<<"i="<<i<<endl;
			if(i==n)break;
		}
		else
		{
			m=0;
			f=1;
			r++;
//			cout<<"r="<<r<<endl;
			i++;
//			cout<<"i="<<i<<endl;
			if(i==n)break;
		}
		if(i==n)break;
	}
//	cout<<"i="<<i<<endl;
//	cout<<"d="<<d<<endl;
//	cout<<"k="<<k<<endl;
//	cout<<"r="<<r<<endl;
//	cout<<"cv-d="<<cv-d<<endl;
//	cout<<"f="<<f<<endl;
//	cout<<"cw="<<cw<<endl;
	if(k==0||d==0)
	{
		min=n-k;
		max=n+cw;
		cout<<min<<" "<<max<<endl;
	}
	else if(f==1&&cw==0&&cv==0)
	{
//		cout<<"in r part"<<endl;
		min=n-k;
		max=n-cw;
		cout<<min<<" "<<max<<endl;
	}
	else 
	{
//		cout<<"in non cv part"<<endl;
		ncv=n-(d+r+cw);
//		cout<<"ncv="<<ncv<<endl;
		min=n-k;
		max=n+cw;
	    cout<<min<<" "<<max<<endl;	
    }
	return 0;
	
//	i=0,m=0,d=0,k=0,ncv=0,cw=0,r=0,n=0;
 //   }
}
