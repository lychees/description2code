#include <bits/stdc++.h>

using namespace std;

int tc;
int A[2002];
int n,k;
int dp[2002][2002];
int vis[2002][2002];

int f(int idx, int sum)
{
	if ( idx == n ) return sum;
	if ( vis[idx][sum] == tc ) return dp[idx][sum];
	vis[idx][sum] = tc;
	int ans = f(idx+1,sum);
	
	if ( sum+A[idx] <= k ) {
		ans = max(ans, f(idx+1,sum+A[idx]));
		ans = max(ans, f(idx,sum+A[idx]));
	}
	dp[idx][sum] = ans;
	return ans;
}

int main()
{
	int t;
	scanf("%d", &t);
	for ( tc = 1; tc <= t; tc++ ) {
		scanf("%d%d", &n, &k);
		for ( int i = 0; i < n; i++ ) scanf("%d", &A[i]);
		int ans = f(0,0);
		printf("%d\n", ans);
	}
	return 0;
}
