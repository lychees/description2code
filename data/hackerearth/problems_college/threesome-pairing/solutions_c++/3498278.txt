#include <bits/stdc++.h>
//zone
using namespace std;
int main()
{int t;
cin >> t;
int n;
int counti = 0;
while(t--)
{
	cin >> n;
	counti = 0;
	int a[n + 1];
	int i,j;
	for(i = 0 ; i < n ; i++)
	{cin >> a[i];
	}
	for(i = 0 ; i < n ; i++)
	{
		for(j = i + 1 ; j <n ;j++)
		{
			if((a[i] + a[j]) % 3 == 0 )
			{counti++;
			}
		}
	}
	cout<<counti<<endl;
}
return 0;
}
