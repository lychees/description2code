#include<iostream>
#include<algorithm>
#include<fstream>
using namespace std;
int main()
{
	//ifstream cin("testcase.txt");
	long long int n;
	cin>>n;
	long long int a[n],b[n];
	long long int i;
	for(i=0;i<n;i++)
		cin>>a[i]>>b[i];
	sort(a,a+n);
	sort(b,b+n);
	long long int q=-(n-1);
	long long int sum=0;
	long long int j=0;
	while(1)
	{
		sum=(sum%1000000007+(a[j]*q)%1000000007)%1000000007;
		q=(q%1000000007)+(2%1000000007);
		j++;
		if(q>n)
			break;
	}
	q=-(n-1);
	j=0;
	while(1)
	{
		sum=(sum%1000000007+(b[j]*q)%1000000007)%1000000007;
		q=(q%1000000007)+(2%1000000007);
		j++;
		if(q>n)
			break;
	}
	cout<<sum%1000000007<<endl;
}
