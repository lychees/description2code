#include<bits/stdc++.h>
using namespace std;
long long int a[200005];
long int n,k;
void init()
{
  scanf("%ld %ld",&n,&k);
  long int i;
  for(i=0;i<n;i++)
  scanf("%lld",&a[i]);
  a[n] = 10000000000;
  long long int c = 0;
  long int x = 0;
  for(int i=0;i<=n;i++)
  {
     if(a[i]<=k)
     {
       x++;
       continue;
     }
     c+=(x*(x+1))/2;
     x = 0;
  }

  c = (n*(n+1))/2 - c;
  cout<<c<<endl;
}

int main()
{
  int t;
  cin>>t;
  while(t--)
  init();
  return 0;
}
