#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
using namespace std;


int min(int a , int b){

	return a<b?a:b;
}

int arr[101][101];

int WayToEnd(int n, int i, int j, int ans){
	
	//cout<<"IN "<<i<<" "<<j<<" "<<ans<<"\n";
	if (arr[i][j] == 0 || i<0 || j<0){
		return 100000;
	}
	
	if (i==0 && j==0){
		return 1;
		
	}
	return 1 + min( WayToEnd(n,i-1,j,ans), WayToEnd(n,i,j-1,ans) );	
	cout<<"OUT "<<i<<" "<<j<<" "<<ans<<"\n";
	return ans;
	

}


int main()
{
    int t,n,k;
    cin>>t;
    for (int i =0;i<t;i++){
    	cin>>n>>k;
    	//int arr[n][n];	
    	string temp;
    	for (int a= 0;a<n;a++){
    		cin>>temp;
    		//cout<<temp<<"\n";
    		for (int b= 0;b<n;b++){
    			//cout<<temp[b]<<" ";
    			arr[a][b] = temp[b] - '0';
    		}	
    	}
    //cout<<"before"<<"\n";	
   	int ans = WayToEnd(n,k-1,k-1,0); 
   	//cout<<"after"<<ans<<"\n";
   	if( ans> 99999 ) 	
   		cout<<-1<<"\n";
   	else
   		cout<<ans-1<<"\n";
    	
    }
    return 0;
}
