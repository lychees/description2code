#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
long long int n,i;
cin>>n;
	long long int s[n],c[n];
	long long int sum=0;
	for(i=0;i<n;i++)
	cin>>s[i];
	for(i=0;i<n;i++)
	cin>>c[i];
	std::sort(s,s+n);
	std::sort(c,c+n);
	for(i=0;i<n;i++)
	sum+=s[i]*c[i];
	cout<<sum<<endl;
    return 0;
}
