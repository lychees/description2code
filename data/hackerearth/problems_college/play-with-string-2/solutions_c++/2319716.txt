#include <iostream>
#include <string>
#include <algorithm>
#include <cmath>

using namespace std;

int main()
{
    int t,flag;
    cin >> t;
    string s;
    for(int i=0;i<t;++i)
    {
        flag=0;
        cin >> s;
        string c(s);
        reverse(c.begin(),c.end());
        for(int j=1;j<s.size();++j)
        {
            if(abs(s[j]-s[j-1])!=abs(c[j]-c[j-1]))
            {
                flag=1;
                break;
            }
        }
        if(flag==0)
        {
            cout << "Funny" << endl;
        }
        else if(flag==1)
        {
            cout << "Not Funny"<< endl;
        }
    }
    return 0;
}
