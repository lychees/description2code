s,q = map( int, raw_input().split() )

M = []

i = 1
while i*i <= s :
	if s%i == 0 :
		M.append(i)
		M.append(s/i)
	i+=1

M = sorted(M)[::-1]

said = [0]*(s+1)

for QQ in range(q):
	a = input()
	ans = -1
	for fac in M :
		if a%fac == 0 :
			ans = fac
			break
		
	print -1 if said[ans] else ans
	
	said[ans] = 1
