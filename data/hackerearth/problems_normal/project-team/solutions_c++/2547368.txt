#include <iostream>
#include<algorithm>
#include<cmath>
#define ll long long 
using namespace std;
int a[1005];

int ans(int n)
{	//int dp[105][105]={0}
	int i,j,minT=10000000,maxT=-1;
	for(i=0;i<n/2;i++)
	{
		minT=min(minT,a[n-i-1]+a[i]);
		maxT=max(maxT,a[n-i-1]+a[i]);
	}
	cout<<maxT-minT<<endl;
	return maxT-minT;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int  n;
    	cin>>n;
    	int i;
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	
    	sort(a,a+n);
    	
    	ans(n);
    	
    }
    return 0;
}
