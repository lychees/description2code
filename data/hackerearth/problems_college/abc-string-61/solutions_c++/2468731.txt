#include <bits/stdc++.h>
#define ll long long int
#define mod 1000000007
using namespace std;
ll pwermod(ll a,ll b)
{
   ll x=1, y=a;
   while(b)
   {
     if(b & 1)
        x = (x*y)%mod;
     y = (y*y)%mod;
     b>>=1;
    }
    return x%mod;
}
int main() 
{
	ll t,n,ans,r1,r2;
	cin>>t;
	while(t--)
	{
		cin>>n;
		if(n>=3)
		{
			r1=pwermod(3,n);
			r2=pwermod(2,n);
			ans=((r1 % mod) + (mod+3-(3*r2)%mod)%mod)%mod;
		}
		else
		{
			ans=0;
		}
		cout<<ans%mod<<endl;
	}
	return 0;
}
