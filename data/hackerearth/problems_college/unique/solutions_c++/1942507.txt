#include<bits/stdc++.h>
using namespace std;
#define MAX1 100000

int main(){
	int test;
	cin>>test;
	while(test--){
		int n;
		cin>>n;
		long a[MAX1];
		for(int i=0;i<n;i++){
			cin>>a[i];
			
		}
		int count=1;
		for(int i=1;i<n;i++){
			int j;
			for(j=i-1;j>=0;j--){
				if(a[j]==a[i]){
					break;
				}
			}
			if(j<0){
				count++;
			}
		}
		cout<<count<<endl;
	
	}
}
