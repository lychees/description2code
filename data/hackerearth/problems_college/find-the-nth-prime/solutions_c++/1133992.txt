#include<iostream>
typedef unsigned long int ll;
#define MAX 1000000
using namespace std;
bool arr[MAX];

void init(void)
{
	for(ll i=2;i<MAX;i++)
		arr[i] = true;
		
	for(ll i=2;i<MAX;i++)
	{
		for(ll j=i;i*j<MAX;j++)
			arr[i*j]=false;
	}
}

int main(void)
{
	init();
	ll test = 0 ;
	ll count = 0;
	cin>>test;
	for(ll i=2;i<MAX;i++)
	{
		if(arr[i] == true)
		{
			count++;
			if(count == test)
			{
				cout<<i<<endl;
				break;
			}
		}
	}
	return 0;
}
