#include <bits/stdc++.h>
using namespace std;

typedef long long LL;

#define PII       pair<int,int>
#define all(c)    c.begin(),c.end()
#define sz(c)     (int)c.size()
#define clr(c)    c.clear()
#define pb        push_back
#define mp        make_pair
#define cin(x)    scanf("%d",&x)
#define MOD		1000000007
#define EPS		1E-10

#define MAXN	100000

bool marker[MAXN + 1] = {0};
int lpf[MAXN + 1];

vector<int> primes;

void sieve(int limit = MAXN)
{
	for(int i = 2; i <= limit; i++) marker[i] = true , lpf[i] = i;
	for(int i = 2; i <= limit; i++)
	{
		if(!marker[i]) continue;
		primes.pb(i);
		int j = 2*i;
		while(j<=limit)
		{
			if(marker[j])
			{
				marker[j] = 0;
				lpf[j] = min(lpf[j] , i);
			}
			j += i;
		}
	}
}

LL G(LL a , int b)
{
	LL ret = 1;
	while(a)
	{
		ret += (a/b);
		a /= b;
	}
	return ret;
}

inline LL F(LL a)
{
	LL ret = 1;
	for(int i = 0; i < sz(primes); i++)
	{
		ret *= G(a , primes[i]);
		if(ret >= MOD)
			ret %= MOD;
	}
	return ret;
}

int main()
{
	sieve();
	int t;
	cin(t);
	while(t--)
	{
		int a,b;
		cin(a);
		cin(b);
		if(a <= b)
			printf("0\n");
		else
		{
			LL ans = F(a) - F(b);
			if(ans < 0)
				ans += MOD;
			printf("%lld\n", ans);
		}
	}
	return 0;
}
