#include<iostream>
#include<string>
using namespace std;
int main() {
    string s;
    cin>>s;
    int i,even=0;
    for(i=0;s[i]!=0;i++){
        if(s[i]=='2'|| s[i]=='4'|| s[i]=='6' || s[i]=='8' )
            even++;
    }
    for(i=0;s[i]!=0;i++){
        if(s[i]=='2'|| s[i]=='4'|| s[i]=='6' || s[i]=='8' )
            cout<<even--<<" ";
        else
            cout<<even<<" ";
    }
    return 0;
}
