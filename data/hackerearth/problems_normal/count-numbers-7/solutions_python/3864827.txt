#!/usr/bin/env python
# -*- coding: utf-8 -*-

# https://www.hackerearth.com/problem/algorithm/count-numbers-7/

K, T = [int(x) for x in raw_input().split()]
primes = [int(x) for x in raw_input().split()]


def count_of_divisible(x):
    result = 0
    for i in xrange(0, 1 << K):
        j = i
        product = 1
        k = 0
        count_of_ones = 0
        while j > 0:
            if j & 1:
                product *= primes[k]
                count_of_ones += 1
            k += 1
            j >>= 1
        if count_of_ones & 1:
            result -= x // product
        else:
            result += x // product
    return x - result

for i in xrange(0, T):
    A, B = [int(x) for x in raw_input().split()]
    print(count_of_divisible(B) - count_of_divisible(A - 1))
