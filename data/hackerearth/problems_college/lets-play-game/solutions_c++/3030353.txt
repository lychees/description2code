#include <iostream>
using namespace std;

int main()
{
    long long int t,n,k;
    long long int a[100005],b[100005],p;
    long long int mx;
    cin>>t;
    while(t--){
    	cin>>n>>k;
    	mx = 0;
    	for(long long int i=0;i<n;i++){
    		cin>>a[i];
    	}
    	for(long long int i=0;i<n;i++)
    	cin>>b[i];
    	for(long long int i=0;i<n;i++){
    		p = (k/a[i])*b[i];
    		if(p>mx)
    		mx = p;
    	}
    	cout<<mx<<endl;
    }
    return 0;
}
