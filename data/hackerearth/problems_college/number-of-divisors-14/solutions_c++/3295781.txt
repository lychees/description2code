#include <iostream>
using namespace std;
long int arr[1000001]={0},pr[80000][2]={0},k=0;
void seive()
{
	long long i,j;
	arr[0]=arr[1]=1;
	for(i=2;i<1000001;i++)
	{
		if(arr[i]!=1)
		{
			pr[k++][0]=i;
			for(j=2*i;j<1000001;j+=i)
				arr[j]=1;
		}
	}
}
void prdiv(long long  i)
{
	long long  j,t,x;
	t=i;
	for(j=0;j<k;j++)
	pr[j][1]=0;
			for(j=0;j<k;j++)
			{
				while(t!=1&&t%pr[j][0]==0)
				{
					t/=pr[j][0];
					pr[j][1]++;
				}
				if(i%pr[j][0]==0)
				{
					if(t!=1)
					{
						for(x=0;x<k;x++)
						if(pr[x][0]==t)
						{
							t=1;
							pr[x][1]++;
							break;
						}
					}
				}
			}
}
int main()
{
    seive();
    long long  t,i,n,p;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	prdiv(n);
    	p=1;
    	for(i=0;i<=k;i++)
    		if(pr[i][1]!=0)
    		p*=(pr[i][1]+1);
    	cout<<p<<"\n";
    	
    	
    }
    return 0;
}
