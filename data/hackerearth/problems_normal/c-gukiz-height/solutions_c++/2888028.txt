#define REP(i,l,r) for(auto i=l;i<r;i++)
#define FORD(i,l,r) for(auto i=l;i>=r;i--)

typedef long long ll;
typedef pair<int,int> ii;

const ll inf=2e9+10;
const int MOD=1e9+7;
const int N=1e5+10;

int n,m,a[N];
ll s[N];
void prepare(){
    scanf("%d%d",&n,&m);
    FOR(i,1,n) scanf("%d",a+i);
    FOR(i,1,n) s[i]=s[i-1]+a[i];
}
ll cal(int idx,ll pha){
    ll day=pha*n+idx;
    return s[n]*pha+s[idx]+day*(day+1)/2;
}
ll findst(int idx){
    if (cal(idx,0)>=m) return idx;
    ll L=0,R=inf/n+20;
    while (L<=R){
        ll mid=(L+R)/2;
        if (cal(idx,mid)<m) L=mid+1;
        else R=mid-1;
    }
//    printf("%d %lld\n",idx,L);
    return L*n+idx;
}
ll solve(){
    ll ans=inf;
    FOR(i,1,n) ans=min(ans,findst(i));
    return ans;
}
int main(){
    prepare();
    cout<<solve();
}
