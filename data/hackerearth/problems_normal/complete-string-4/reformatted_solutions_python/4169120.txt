import sys

L = map(str, sys.stdin.read().split())

L = L[1:]

alpha = 'qwertyuioplkjhgfdsazxcvbnm'
for x in L:
    if len(x) < 26:
        print 'NO'
    else:
        flag = True
        for y in alpha:
            if y not in x:
                flag = False
                break
        if flag:
            print 'YES'
        else:
            print 'NO'
