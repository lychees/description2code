chocolates = map(int, raw_input().split( ))
chocolate_index = [-1]*(len(chocolates)+1)
chocolate_index[0] = 0
for index in range(1, 1+len(chocolates)):
	chocolate_index[index] = chocolate_index[index-1]+chocolates[index-1]
	
query_count = int(raw_input())
for x in range(query_count):
	index = int(raw_input())
	left_chocolate_index = 0
	right_chocolate_index = len(chocolates)
	mid_index = (left_chocolate_index + right_chocolate_index)/2
	prev_mid_index = mid_index
	mid_box_start_index = 1+chocolate_index[mid_index-1]
	mid_box_end_index = chocolate_index[mid_index]
	while not mid_box_start_index <= index <=  mid_box_end_index:
		if index < mid_box_start_index:
			right_chocolate_index = mid_index
		if index > mid_box_end_index:
			left_chocolate_index = mid_index
		mid_index = (left_chocolate_index + right_chocolate_index)/2
		if mid_index == prev_mid_index:
			mid_index = mid_index+1
		mid_box_start_index = 1+chocolate_index[mid_index-1]
		mid_box_end_index = chocolate_index[mid_index]
		prev_mid_index = mid_index
