#include <iostream>
using namespace std;
 
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	long long int n,sum=0;
    	cin>>n;
    	while(n)
    	{
    		long long int x=(n+1)/2;
    		x=x*x;
    		sum=sum+x;
    		n=n/2;
    		
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
