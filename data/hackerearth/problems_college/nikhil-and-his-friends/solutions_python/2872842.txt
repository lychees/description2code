t=input()
for _ in range(t):
	try:
		n,x=map(int,raw_input().split())
	except:
		exit()
	a=[]
	for i in xrange(n): 
		
		try:
			y=raw_input()
			y=int(y)
		except:
			break
		a.append(y)
	n=len(a)
	last=-1
	first=-1
	su=0
	while last<n and first<n:
		if su<x:
			last+=1
			if last<n:
				su+=a[last]
			else: break	
		elif su>x:
			first+=1
			if first<n:
				su-=a[first]
			else: break	
		else: 
			break
	if su==x: print "YES"
	else: print "NO"
