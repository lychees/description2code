#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	cin.ignore();
	while(t--)
	{
		string str;
		int l,i,temp;double sum=0;
		getline(cin,str);
		l=str.length();
		for(i=0;i<l;i++)
		{
			sum=sum+int(str[i]);
		}
		sum=sum/double(l);
		temp=int(sum);
		if(temp%2!=0)
		{
		
			reverse(str.begin(),str.end());
			cout<<str<<endl;
			continue;
		}
		else
		cout<<str<<endl;
	}
}
