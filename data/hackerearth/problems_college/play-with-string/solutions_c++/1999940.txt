#include <iostream>
using namespace std;
string rev(string s) {
	string k = s;
	int l = k.length();
	
	int m = l/2;
	for(int i = 0; i < m; i++) {
		swap(k[i], k[l-1]);
		l--;
	}
	return k;
}

int fun(string s, string r,int l) {
	int count = 0;
	for(int i = 0; i < l-1; i++) {
		if(abs(s[i+1]-s[i]) == abs(r[i+1]-r[i])) {
			count++;
		}
	}
	return count;
}

int main()
{
    int t,l;
    string s,r;
    cin>>t;
    while(t--) {
    	cin>>s;
    	l = s.length();
    	r = rev(s);
    	
    	if(fun(s,r,l) == l-1) {
    		cout<<"Funny"<<endl;
    	}
    	else {
    		cout<<"Not Funny"<<endl;
    	}
    }
    return 0;
}
