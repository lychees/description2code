#include<bits/stdc++.h>
#define m 20000000
#define mi(a,b) ((a>b)?b:a)
using namespace std;
int dp[m+2];
int main()
{
int t;
dp[1]=0;
for(int i=2;i<=m;i++)
    {
        dp[i]=dp[i-1]+1;
        if(i%2==0)
            dp[i]=mi(dp[i],1+dp[i/2]);
        if(i%3==0)
            dp[i]=mi(dp[i],1+dp[i/3]);
    }
    cin>>t;
while(t--)
{
    int n;
    cin>>n;
    printf("%d\n",dp[n]);
}
return 0;
}
