#include <bits/stdc++.h>
using namespace std;

int main() {
	int n,m,p,count=0,i,j;
	cin>>n>>m;
	string str[n];
	bool max[105]={false};
	for(i=0;i<n;i++) cin>>str[i];
		for(i=0;i<m;i++)
		{
			char ch=str[0][i];
			for(j=1;j<n;j++) if(ch>str[j][i]) {ch=str[j][i];}
			for(j=0;j<n;j++) if(str[j][i]==ch) max[j]=true;
		}
		for(i=0;i<n;i++) if(max[i]) count++;
		cout<<count<<endl;

	return 0;
}
