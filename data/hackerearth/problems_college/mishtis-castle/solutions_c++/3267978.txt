#include <bits/stdc++.h>
 
#define fi first
#define se second
#define pb push_back
#define mp make_pair
#define pii pair<int,int>
#define pll pair<ll,ll>
#define MOD 1000000007
#define ll long long int
#define MAXINT 2000000000
#define SET(x,y) memset(x,y,sizeof(x));
 
using namespace std;

int A[300][300],DP[300][300];

void solve(void) {
	int i,j,n,k;
	cin>>n>>k;
	for(i=1;i<=n;i++)
		for(j=1;j<=n;j++)
			cin>>A[i][j];
	SET(DP,0);
	for(i=1;i<=n;i++)
		for(j=1;j<=n;j++)
			DP[i][j]=A[i][j]+max(DP[i-1][j],DP[i][j-1]);
	if(DP[n][n]>=k)	cout<<DP[n][n]-k<<endl;
	else			cout<<-1<<endl;
}

int main () {
	int t,i;
	cin>>t;
	for(i=1;i<=t;i++) {
		solve();
	}
	return 0;
}
