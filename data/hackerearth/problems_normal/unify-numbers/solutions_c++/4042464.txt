#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
	long long int t,n,cost=0,i,a[1000000]={0};
	int median=0;
	cin>>t;
	while(t--){
		cin>>n;
		median=0;
		cost=0;
		for(i=0;i<n;i++){
			cin>>a[i];
		}
		sort(a,a+n);
		if(n%2==0){
			median=(a[(n-1)/2]+a[n/2])/2;
		}
		else{
			median=a[n/2];
		}
		for(i=0;i<n;i++){
			cost+=abs(a[i]-median);
		}
		cout<<cost<<endl;
	}
    return 0;
}
