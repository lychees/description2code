#include <iostream>
#include <string>

using namespace std;

int tx[30], pt[30];

void init()
{
	for (int i = 0; i < 30; i++)
	{
		tx[i] = 0;
		pt[i] = 0;
	}
}

bool check()
{
	for (int i = 0; i < 30; i++)
	{
		if (tx[i] != pt[i]) return false;
	}

	return true;
}

int main()
{
	int testcase;
	cin >> testcase;
	while (testcase--)
	{
		string text, pattern;
		cin >> pattern >> text;
		int txlen = text.length(), ptlen = pattern.length();
		if (ptlen > txlen)
		{
			cout << "NO" << endl;
			continue;
		}

		init();
		for (int i = 0; pattern[i] != '\0'; i++)
		{
			pt[pattern[i] - 'a']++;
		}
		for (int i = 0; i < ptlen; i++)
		{
			tx[text[i] - 'a']++;
		}
		bool result = check();
		if (!result)
		{
			for (int i = 0; i + ptlen < txlen; i++)
			{
				tx[text[i] - 'a']--;
				tx[text[i + ptlen] - 'a']++;
				if (check())
				{
					result = true;
					break;
				}
			}
		}

		if (result) cout << "YES" << endl;
		else cout << "NO" << endl;
	}

	return 0;
}
