#include <stdio.h>
#include<math.h>
int main()
{
	
	int t;
   scanf("%d",&t);
    while(t--)
    {
    	
       long int i,n,c=0;
      scanf("%ld",&n);
   
    	for(i=2;i<=sqrt(n);i++)
    	{
    	  
    	  if(n%i==0)
    	  {
    	  if(i%2==0)
    	  {
    	  	c++;
 
    	  }
    	  if((n/i)!=i && (n/i)%2==0)
    	  	 c++;
    	}
    	}
    	if(n%2==0)
    	 c++;
    printf("%ld\n",c);
    	
    }
    return 0;
}
