#include <iostream>
#include <cstdio>
using namespace std;

long long ans = 0;
int n;
long long arr[100010];

void recur(int id,int x)
{
	//cout << x << " ";
	if(id >= 4) {
            //cout << "hah";
		if(x == 0) return;
		int l = 0;
		for(int i=0;i<4;i++) {
			if(x&(1<<i)) {
				l++;
			}
		}
		long long xy = 0;
		for(int i=0;i<n;i++) {
			int x2,x3,x5,x7;
			x2=x3=x5=x7=0;
			long long t = arr[i];
			while(t) {
				if(x&1) {
					if(t%10 == 2) x2=1;
				}
				if(x&2) if(t%10==3) x3=1;
				if(x&4) if(t%10==5) x5=1;
				if(x&8) if(t%10==7) x7=1;
				t /= 10;
			}
			bool f = 1;
			if(x&1) if(!x2) f = 0;
			if(x&2) if(!x3) f = 0;
			if(x&4) if(!x5) f = 0;
			if(x&8) if(!x7) f = 0;
			if(f) xy++;
			//cout << "cool";
		}
		if(xy>2) {
                if(l%2) ans += (xy*(xy-1)*(xy-2))/6;
                else ans -= (xy*(xy-1)*(xy-2))/6;
		}
		return;
	}
	recur(id+1,x);
	recur(id+1,(x|(1<<id)));
}

int main()
{
	scanf("%d",&n);
	//cin >> n;
	for(int i=0;i<n;i++) {
		scanf("%lld",&arr[i]);
		//cin >> arr[i];
	}

	recur(0,0);
	cout << ans;
    //cout << "Hello World!" << endl;
    return 0;
}
