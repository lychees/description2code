#include <bits/stdc++.h>
using namespace std;

int main(void) {
    int T = 0;
    scanf("%d", &T);
    while(T--) {
    	vector<int> count(27, 0);
    	string s;
    	cin >> s;
    	for(auto it = s.begin(); it != s.end(); ++it) {
    		count[*it - 'A'] += 1;
    	}
    	s = "";
    	int idx = 0;
    	for(auto it = count.begin(); it != count.end(); ++it, ++idx) {
    		int c = *it;
    		string temp = "";
    		while(c--) {
    			temp += char('A' + idx);
    		}
    		s += temp;
    	}
    	cout << s << endl;
    }
    return 0;
}
