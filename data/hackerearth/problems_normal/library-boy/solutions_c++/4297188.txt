#include<iostream>
#include<map>
#include<string>
#include<cassert>
#include<math.h>

using namespace std;

int main(){

 int N=0,i=0,count=0;
 string book;
 typedef map<char,int> shelf;
 shelf shelves;
 shelves.clear();
 cin>>N;
 assert( (N>=1) && (N<=1000));
 for(i=0;i<N;i++){
  cin>>book;
  shelves[book[0]]++;
  book.erase();
 }
 
 for( shelf::const_iterator it = shelves.begin();it!=shelves.end();it++){
   count = count + (int)ceil(it->second/10.0); 
 }
 
 cout<<count<<endl;
 
 return 0;
}
