#include<bits/stdc++.h>
#define ll long long
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        ll x,p,n,amt,cnt=0;
        scanf("%lld%lld",&x,&p);
        amt=x;
        if(x<p)
        {
            cnt=0;
            amt=x;
        }
        else
        {
            while(x%p==0)
            {
                cnt++;
                x=x/p;
                //printf("%lld %lld\n",x,cnt);
            }
            if(x==1)
                x=0;
        }
        printf("%lld %lld\n",cnt,x);
        //cout<<cnt;
    }
    
    return 0;
}
