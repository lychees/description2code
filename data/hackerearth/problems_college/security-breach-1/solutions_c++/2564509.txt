#include <iostream>
#include <stdio.h>
using namespace std;

int main() {
	int t,i;
	char s[25];
	scanf("%d",&t);
	while(t--)
	{
		scanf("%s",s);
		for(i=0;s[i]!='\0';i++)
		{
			if(s[i]=='2')
			{
				printf("cde");
			}
			else if(s[i]=='7')
			{
				printf("acf");
			}
			else if(s[i]=='6')
			printf("b3");
			else if(s[i]=='9')
			printf("c6a");
			
		}
		printf("\n");
	}
	
	return 0;
}
