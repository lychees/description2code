'''

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
for _ in range(int(raw_input())):
	l=[0 for i in range(26)]
	flag=0
	for i in raw_input():
		l[ord(i)-97]+=1
	temp=l[0]
	for i in l:
		if i!=temp:
			flag=1
			break;

	if flag==0:
		print 'Yes'
	else:
		print 'No'
