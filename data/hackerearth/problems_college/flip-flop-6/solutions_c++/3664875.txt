#include <iostream>
#include<string.h>
using namespace std;


int main()
{
    int n,c,i,l;
    char s[100002];
    cin>>n;
    while(n--)
    {
    	cin>>s;
    	l=strlen(s); c=0;
    	for(i=0;i<l;i++)
    	{
    		if(s[i]==s[i+1])
    		c++;
    	}
    
    	cout<<c<<'\n';
    }
  
    return 0;
}
