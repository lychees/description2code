#include <iostream>
using namespace std;

long area (long k)
{
if(k==1||k==2||k==3 )
return 0;

else
 {
if(k%2==0)
   {
	if((k/2)%2==0)
	return (k*k)/16;
	
	else return ((k/2 +1)*(k/2-1))/4;
   }
   
 else return area(k-1)	;
 }	
}


int main()
{  long t,N;
   //cout << "Hello World!" << endl;
   cin >>t;
   for(long i=1;i<=t;i++)
   {
   	cin>>N;
   	cout<<area(N)<<endl;
   }
    return 0;
    
    
}
