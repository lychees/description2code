#include <iostream>
using namespace std;

int main()
{
    long long int n,no,i,a[2]={0},l;
    cin>>no;
    while(no--)
    {
    	cin>>n;
    	for(i=0;i<n;i++)
    	{
    		cin>>l;
    		if(l&1)
    		++a[0];
    		else
    		++a[1];
    	}
    	if(a[0]%2==0)
    	cout<<"Yes\n";
    	else
    	cout<<"No\n";
    	a[0]=0,a[1]=0;
    }
    return 0;
}
