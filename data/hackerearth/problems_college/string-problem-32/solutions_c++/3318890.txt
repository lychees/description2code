#include<bits/stdc++.h>
#define SIZE 100200
using namespace std;
int lps[SIZE];
void LPS(char str[],int len)
{
    int pre,i,j;
    pre = 0;
    lps[0] = pre;
    i = 1;
    while(i < len)
    {
        if(str[i] == str[pre])
        {
            pre++;
            lps[i] = pre;
            i++;
        }
        else
        {
            if(pre)
            {
                pre = lps[pre - 1];
            }
            else
            {
                lps[i] = 0;
                i++;
            }
        }
    }
}
int main()
{
    int t,len,j=1;
    char str[SIZE];
	cin>>t;
	while(t--)
    {
        cin>>str;
        len = strlen(str);
        LPS(str,len);
        cout<<"Case "<<j++<<": "<<(lps[len-1]?lps[len-1]:-1)<<"\n";
    }
	return 0;
}
