#include <bits/stdc++.h>
using namespace std;

int main(void) {
    int T = 0;
    scanf("%d", &T);
    cin.ignore();
    while(T--) {
    	string raw;
    	getline(cin, raw);
    	stringstream ss (raw);
    	vector<string> token;
    	set<string> s;
    	string work;
    	while(ss >> work) {
    		if (s.find(work) == s.end()) {
    			s.insert(work);
    			token.push_back(work);
    		}
    	}
    	for(auto it = token.begin(); it != token.end(); ++it) {
    		cout << *it << " ";
    	}
    	cout << endl;
    }
    return 0;
}
