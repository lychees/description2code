def pages(num):
    ans = 100000
    for i in xrange(113):
        for j in xrange(113):
            if (i*10+j*12 == num):
                ans = min(ans, i+j)
    if (ans == 100000):
        print "-1"
    else:
        print ans


test = int(raw_input())
while (test):
    test -= 1
    num = int(raw_input())
    pages(num)
