using namespace std;

int dp[10][1000010];

int main(){	
	int t,n;
	cin>>t;
	while(t--){  
		cin>>n;
		
		for(int i =0;i<10;i++)
		  dp[i][1]=1;
		
		int j;
		
		for(int i = 2; i<=n;i++){
			if(i==n) j = 1;
			else j = 0;
			for(;j<=9;j++){
				dp[j][i]=0;
				if(j-1>=0){
				dp[j][i] += dp[j-1][i-1];
				dp[j][i] = dp[j][i] % mod;
				}
				if(j+1<=9){
				dp[j][i]+= dp[j+1][i-1];
				dp[j][i] = dp[j][i] % mod;
				
				}
			}	
		}
		int i, ans = 0;
		if(n==1)
		i = 0;
		else
		i = 1;
		for(;i<=9;i++){
			ans += dp[i][n];
			ans = ans % mod;
		}
		cout<<ans<<endl;
	}
	
	return 0;
	
}
