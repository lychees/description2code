#include<iostream>
using namespace std;
int leap(int y)
{
	if(y%100==0)
	{
		if(y%400==0)
		return 1;
		else
		return 0;
	}
	else if(y%4==0)
	return 1;
	return 0;
}
int main()
{
	int firstday[12]={5,1,1,4,6,2,4,0,3,5,1,3};
	int date[12]={31,28,31,30,31,30,31,31,30,31,30,31};
	int m,y,t,count,i,change,day;
	scanf("%d",&t);
	while(t--)
	{
		date[1]=28;
		count=0;
		scanf("%d %d",&m,&y);
		for(i=1999;i<=y;i++)
		{
			if(leap(i))
			count++;
		}
		change=y-1999+count;
		if(leap(y))
		{
			if(m<=2)
			change--;
			date[1]=29;
		}
		//cout<<firstday[m-1]<<"     "<<change<<"  "<<"count = "<<count<<"\n";
		day=(firstday[m-1]+change)%7;
		//cout<<"day = "<<day<<"\n";
		if(day>0)
		day=(8-day);
		else
		day=1;
		while(day<=date[m-1])
		{
			printf("%d ",day);
			day+=7;
		}
		printf("\n");
	}
}
