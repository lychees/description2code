# Read input from stdin and provide input before running code
matches = raw_input()
win = {""}
lose = {""}
for match in range(0, pow(2, int(matches))-1):
    contestant = raw_input().split(' ', 1)
    lose.add(contestant[1])
    if contestant[0] not in lose:
        win.add(contestant[0])
for winner in win:
    if winner not in lose:
        print(winner)
