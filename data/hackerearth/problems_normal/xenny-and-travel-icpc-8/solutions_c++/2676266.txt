// Numeric Constants
#define N 1000000007
#define maxs 2000005
#define mins 1005
#define eps 0.000000000001
#define imax 2000000200
#define llmax 1000000002000000000ll
#define pi 3.141592653589793

// Others
#define ll long long
#define pb push_back
#define gc getchar_unlocked
#define iosbase ios_base::sync_with_stdio(false)
#define pii pair<int,int>
#define pll pair<ll,ll>
#define ppi pair<pair<int,int>,int>
#define ppl pair<pll,ll>
#define vi vector<int>
#define sc scanf
#define pr printf
#define lld I64d
#define F first
#define S second
#define siter set<int>::iterator
#define p_pq priority_queue
#define ub upper_bound
#define lb lower_bound

class comp{
public:
	bool operator()(pii & p1,pii & p2){
		return p2.S<p1.S;
	}
};

p_pq<pii,vector<pii>,comp>pq;
int dist[4][1255][1255],mark[4][1255],dp[4][1255];

int main()
{
	int n,u,v,i,j,k;
	sc("%d",&n);
	for(i=0;i<n;i++){
		for(j=0;j<n;j++){
			sc("%d",&dist[0][i][j]);
		}
	}
	for(i=0;i<n;i++){
		for(j=0;j<n;j++){
			sc("%d",&dist[1][i][j]);
		}
	}
	for(i=0;i<n;i++){
		for(j=0;j<n;j++){
			dist[2][i][j]=dist[0][j][i];
			dist[3][i][j]=dist[1][j][i];
		}
	}
	sc("%d %d",&u,&v);
	u--;
	v--;
	for(i=0;i<4;i++){
		for(j=0;j<=n;j++){
			dp[i][j]=imax;
			mark[i][j]=0;
		}
	}
	for(i=0;i<4;i++){
		if(i==0 || i==1){
			dp[i][u]=0;
			pq.push({u,0});
		}
		else{
			pq.push({v,0});
			dp[i][v]=0;
		}
		while(!pq.empty()){
			pii p1=pq.top();
			pq.pop();
			if(mark[i][p1.F]==1){
				continue;
			}
		//	cout<<i<<" "<<p1.F<<" "<<p1.S<<endl;
			mark[i][p1.F]=1;
			for(j=0;j<n;j++){
				int nd=dist[i][p1.F][j]+p1.S;
				if(dp[i][j]>nd){
					dp[i][j]=nd;
					pq.push({j,nd});
				}
			}
		}
	}
	// for(i=0;i<4;i++){
	// 	for(j=0;j<n;j++){
	// 		cout<<dp[i][j]<<" ";
	// 	}
	// 	cout<<endl;
	// }
	int ans=imax;
	for(i=0;i<n;i++){
		if(i==u || i==v){
			continue;
		}
		ans=min(ans,dp[0][i]+dp[3][i]);
		ans=min(ans,dp[1][i]+dp[2][i]);
	//	cout<<ans<<" ";
	}
	pr("%d\n",ans);
	return 0;
}
