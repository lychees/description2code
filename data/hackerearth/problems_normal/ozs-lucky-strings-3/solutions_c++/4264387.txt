#include <iostream>
#include <string>
#include <vector>
using namespace std;

int main()
{
    int p1,p2,t;
    string str;
    cin>>t;
    while(t--)
    {
    	cin>>str;
    	vector<int> countR(str.size(),0),countK(str.size(),0);
    	if(str[0]=='R')
    		countR[0]=1;
    	if(str[str.size()-1]=='K')
    		countK[str.size()-1]=1;
    	for(p1=1;p1<str.size();p1++)
    		if(str[p1]=='R')
    			countR[p1]=1+countR[p1-1];
    		else
    			countR[p1]=countR[p1-1];
    	for(p1=str.size()-2;p1>=0;p1--)
    		if(str[p1]=='K')
    			countK[p1]=countK[p1+1]+1;
    		else
    			countK[p1]=countK[p1+1];
    	for(p1=0;p1<str.size();p1++)
    		if(countR[p1]>=countK[p1])
    			break;
    	if(p1<str.size())
    		cout<<2*countK[p1]<<endl;
    	else
    		cout<<0<<endl;
    }
    return 0;
}
