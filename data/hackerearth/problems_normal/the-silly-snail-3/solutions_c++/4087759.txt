using namespace std;
map<int,pii>m;
void preorder(int node)
{
    if(node==0)
        return;
    cout<<node<<" ";
    preorder(m[node].fi);
    preorder(m[node].se);
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        m.clear();
        for(int i=0;i<n;i++)
        {
            int a,b,c;
            cin>>a>>b>>c;
            m[a].fi=b;
            m[a].se=c;
        }
        preorder(1);
        cout<<"\n";
    }
}
