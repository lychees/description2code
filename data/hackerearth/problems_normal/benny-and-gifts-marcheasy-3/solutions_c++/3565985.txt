#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s;
    cin>>s;
    set<pair<int, int> > k;
    k.insert(pair<int, int>(0,0));
    int x = 0;
    int y = 0;
    int z = 0;
    for(int i=0;i<s.size();i++) {
    	if(s[i]=='L')
    		y--;
    	else if(s[i]=='R')
    		y++;
    	else if(s[i]=='U')
    		x--;
    	else
    		x++;
    	if(k.find(pair<int, int>(x,y))!=k.end())
    		z++;
    	else
    		k.insert(pair<int, int>(x,y));
    }
    cout<<z;
    
    return 0;
}
