tc = input()
while tc:
    tc -= 1
    N, Q = map(int, raw_input().split())
    List = map(int, raw_input().split())
    City_List = sorted(List, reverse=True)
    if N == 1:
        print 0
    else:
        i = 0
        for j in xrange(N-1):
            i += City_List[j]-City_List[j+1]

        print Q*i
