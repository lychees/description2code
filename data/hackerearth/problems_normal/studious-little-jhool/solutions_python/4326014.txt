def process(N):
	if N == 10:
		return 1
	if N == 12:
		return 1
	if N < 10:
		return -1
	if N % 12 == 0:
		return N / 12
	k = []
	tmp = N
	c = 0
	while N >= 0:
		N -= 12
		c += 1
		if N % 10 == 0 and N > 0:
			k.append((12*c, c))
	if not k:
		if tmp % 10 == 0:
			return tmp/10
		return -1
	remain = k[-1][1] +  (tmp-k[-1][0])/10
	return remain

T = input()
for i in xrange(T):
	N = input()
	print process(N)
