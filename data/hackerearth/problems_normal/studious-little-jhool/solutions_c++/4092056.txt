#include<iostream>
using namespace std;

int main()
{
	int NTest;
	cin>>NTest;

	for(int i=0;i<NTest;i++)
	{
		int NQue;
		cin>>NQue;
		
		NQue+=1;	//'0' index will be garbage
		
		int table[NQue];
		
		for(int i=0;i<NQue;i++)
		{
			if(i <=12)
			{
				if(i<10) table[i]=-1;
				if(i==10) table[i]=1;
				if(i==11) table[i]=-1;
				if(i==12) table[i]=1;
				continue;
			}
			
			int a=table[i-10];
			int b=table[i-12];
			
			if((i) % 12 ==0)
			{	table[i]=(i)/12;
				continue;
			}
			if(a==-1 && b==-1) table[i]=-1;
			else table[i]=max(a+1,b+1);
		}	//table calculatin ends here
		
//		for(int i=0;i<NQue;i++)
//			cout<<table[i]<<"["<< i <<"]"<<" ";
//		
//		cout<<endl;
//		cout<<"The answer is\t";
		cout<<table[NQue-1]<<endl;
	}
}
