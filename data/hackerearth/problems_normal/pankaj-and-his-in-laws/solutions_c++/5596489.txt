#include <iostream>
#include<string.h>
using namespace std;
int arr[100],brr[100];
void longestincreasing(int n){
	for(int i=0;i<n;i++)brr[i]=1;
	for(int i=1;i<n;i++){
		for(int j=0;j<i;j++){
			if(arr[j]<arr[i])brr[i]=max(brr[i],brr[j]+1);
		}}
		 std::string r;
		 int maxy=0;
		 for(int i=0;i<n;i++)maxy=max(maxy,brr[i]);
		 n=maxy;
    while(n!=0) {r=(n%2==0 ?"0":"1")+r; n/=2;}
	cout<<r<<endl;
	
}
int main()
{
	int t;
	cin>>t;
	for(int i=0;i<t;i++)cin>>arr[i];
	longestincreasing(t);
}
