{long int n,q,i;

 for(i=1;i<=n;i++)
 {cin>>s[i];
  sum=0;
  for(j=0;j<k;j++)

     sum=(sum*(l+1))+a[j];
   }

 }
 for(i=1;i<=q;i++)

 }

}
