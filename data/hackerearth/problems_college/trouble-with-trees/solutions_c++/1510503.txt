#include<iostream>
using namespace std;
unsigned long int mod=1000000007;
void catalan1(unsigned int n,unsigned long int catalan[])
{
    
    catalan[0] = catalan[1] = 1;
    for (int i=2; i<=n; i++)
    {
        catalan[i] = 0;
        for (int j=0; j<i; j++)
            catalan[i] =(catalan[i]+(( catalan[j] * catalan[i-j-1])%mod))%mod;
    }
}

int main()
{
		unsigned long int catalan[1001];
    catalan1(1000,catalan);
    int t;
    scanf("%d",&t);
    while(t--)
	{
		int n;
		scanf("%d",&n);
		cout<<catalan[n]%mod<<"\n";
	}
    return 0;
}
