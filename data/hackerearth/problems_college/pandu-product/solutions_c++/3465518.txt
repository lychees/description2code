#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
   int n;
   cin>>n;
   long long int ar[n],br[n];
   int i = 0,j = 0,k = 0;
   while(i<n)
   {
   		long long int v;
   		cin>>v;
   		if(v>=0)
   		{
   			ar[j] = v;
   			j++;
   		}
   		else
   		{
   			br[k] = v;
   			k++;
   		}
   	i++;
   }
   sort(ar,ar+j);
   sort(br,br+k);
   long long int num1,num2; 
   num1 = ar[j-1] * ar[j-2];
   num2 = br[0] * br[1];
  // cout<<"num "<<num1<<" "<<num2<<endl;
   if(num1>num2)
   {
   	cout<<num1<<endl;
   }
   else
   {
   	cout<<num2<<endl;
   }
    return 0;
}
