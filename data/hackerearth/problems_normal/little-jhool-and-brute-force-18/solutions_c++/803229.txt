const int N = 1000000, M = 100;
int tab[2 * N], good[N + 5], res[N + 5];
void prepro()
{
	FOR(i, 1, M)
		FOR(j, i, M)
		{
			tab[i * i * i + j * j * j]++;
		}
	FOR(i, 1, N)
		if(tab[i] > 1)
			good[i] = 1;
	
	int last = -1;
	FOR(i, 1, N)
	{
		res[i] = last;
		if(good[i])
			last = i;
	}
}
