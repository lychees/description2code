#include<stdio.h>
#include<math.h>
#define MAX 1000006
#define ll long long
ll arr[MAX];

ll no_of_zero(ll n)
{
	ll a=1,sum=0;
	while(1)
	{
		a=a*5;
		ll z=n/a;
		if(z==0)
		{
			break;
		}
		else
		{
			sum=sum+z;
		}
	}
	return sum;
}

void generate()
{
    arr[0] = 0;
    for(ll i = 0; i < MAX; i++) {
        arr[i] += arr[i - 1] + no_of_zero(i);
    }
}

int main()
{
    generate();
	ll t;
	scanf("%lld", &t);
	while(t--) {
        ll a, b;
        scanf("%lld%lld", &a, &b);
        printf("%lld\n", arr[b] - arr[a - 1]);
	}
	return 0;
}
