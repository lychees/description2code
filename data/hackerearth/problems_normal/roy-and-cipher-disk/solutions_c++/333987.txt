//Roy and cipher disk
#include<iostream>
#include<string.h>
using namespace std;

int main()
{
	int i, test, diff;
	char strng[102], prev;
	cin>>test;
	while(test--)
	{
		cin>> strng;
		prev = 'a';
		for(i=0;i < strlen(strng);i++)
		{
			if(strng[i] > prev)
			{
				diff=strng[i]-prev;
				if(diff > 13)
					cout<< diff - 26 << " ";
				else
					cout<< diff << " ";
			}
			else
			{
				diff=prev-strng[i];
				if(diff >= 13)
					cout<< 26-diff << " ";
				else
					cout<< -1*diff << " ";
			}
			prev = strng[i];
		}
		cout<< endl;
	}
	return 0;
}
