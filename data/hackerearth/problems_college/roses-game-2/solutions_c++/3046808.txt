#include <iostream>
#include <stdbool.h>
#define MAX 100000

using namespace std;
long long int a[MAX] = {0};
bool marked[MAX] = {0};

int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		long int n, i = 0, j, prev;
		scanf("%ld",&n);
		
		for(j = 0; j < n ; j++){
			scanf("%ld",&a[j]);
			marked[j] = 0 ;
		}
			
			
		for( ; j < n+5 ; j++)
			a[j] = -2 ;
		
		
		while(marked[i] == 0)
		{
			marked[i] = 1 ;
			if(a[i] != -1)
				i = a[i];
			else
				break ;
		}
		
		if(a[i] == -1)
			printf("Win\n");
		else
			printf("Lose\n");
	}
}
