#include<iostream>
#include<algorithm>
#include<cstdio>
using namespace std;
typedef long long ll;
int main()
{
    ll t,l,b,h,r;
    cin>>t;
    while(t--)
    {
        cin>>l>>b>>h;
        r=min(l,b);
        r=min(r,h);
        if(r%2==0)
            printf("%lld.0\n",r/2);
        else
            printf("%lld.5\n",r/2);
    }
    return 0;
}
