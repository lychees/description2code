#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--) {
    	int w = 0;
    	string s;
    	int a[256];
    	for(int i = 0; i < 256; i++)a[i] = 0;
    	cin>>s;
    	for(int i = 0; i < s.size(); i++)a[s[i]]++;
    	for(int i = 0; i < 256; i++)if(a[i] % 2 != 0)w++;
    	cout<<max(0,w - 1)<<"\n";
    	
    	
    }
}
