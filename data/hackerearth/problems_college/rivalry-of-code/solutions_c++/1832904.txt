#include <iostream>
#include <limits.h>
using namespace std;

int main()
{
    int t,k,j,i,count,mincount;
    char str[5001],str1[5001];
    cin>>t;
    while(t--)
    {
    	cin>>str1;
    	cin>>str;
    	mincount=INT_MAX;
    	for(int i=0;str1[i]!=NULL;i++)
    	{
    		k=i;
    		count=0;
    
    		for(int j=0;str[j]!=NULL;++j)
    		{
    	
    			if(str[j]!=str1[k])
    			{
    			
    				count++;
    			
    			}
    			k++;
    		}
    	
    		if(count<mincount)
    		 {
    		 	mincount=count;
    		 }
    		
    	}
    	cout<<mincount<<endl;
    }
    return 0;
}
