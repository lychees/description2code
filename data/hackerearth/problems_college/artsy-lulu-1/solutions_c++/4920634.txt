#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
	
	int n;
	char ch;
	while(cin >> ch >> n) {

		if(n<2 || isalpha(ch+(n-1))==false) {
			cout << "INVALID" << endl << endl;
			continue;
		}
		else {
			for(int i=0; i<n; i++) {
				for(int j=0; j<i+1; j++) {
					cout << char(ch+i) << ' ';
				}
				cout << endl;
			}
			cout << endl;
		}
	}
	
	return 0;
}
