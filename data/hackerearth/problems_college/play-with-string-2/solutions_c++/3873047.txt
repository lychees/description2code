#include <iostream>
#include<string.h>
using namespace std;
int main()
{
	int t,temp,i;
	cin>>t;
	string s;
	while(t--)
	{
		temp=1;
		cin>>s;
		int l1=s.length()-1;
		for(i=0;i<s.length()/2;i++)
		{
			if(abs(s[i+1]-s[i])!=abs(s[l1-i]-s[l1-i-1]))
			{
				cout<<"Not Funny\n";
				temp=0;
				break;
			}
		}
		if(temp)
		cout<<"Funny\n";
	}
    return 0;
}
