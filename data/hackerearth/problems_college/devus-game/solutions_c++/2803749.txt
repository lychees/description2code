#include <iostream>
using namespace std;
int main()
{
    int t,n,flag =0,j=0,pos,i =0;
   
    int *a = NULL;
    int *count = NULL;
    
    i =0;
    cin>>t;
    while(t--)
    {
    	
    	cin>>n;
    	a= new int[n];
        count = new int[n];
        for(i=0;i<n;i++)
           {cin>>a[i];
            count[i]=0;
           }
     i=0;
    while( i < n && flag !=1)
		 {
		 pos = a[j];
		 if(pos!=-1)
		 count[pos]++;
		 j = pos;
		 if(j==-1)
		   flag =1;
		   i++;
		 }
		 for(i=0;i<n;i++)
		     if(count[i]>1)
			    flag =-1;
		 if(flag==1)
		   cout<<"Win\n";
		 else 
		  cout<<"Lose\n";
	flag =0;
	j=0;
    }
		    return 0;
}
