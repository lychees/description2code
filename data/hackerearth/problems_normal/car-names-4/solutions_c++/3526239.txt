#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n;
    cin >> n;

    while(n--)
    {
        string a;
        cin >> a;

        int ans[26] = {0};

        for (int i = 0;i < a.size();i++)
        {
            ans[a[i]-'a']++;
        }

        int count = 0;
        sort(ans,ans+26);

        for (int i = 25;i >= 0;i--)
        {
            if (ans[i] != 0)
                count++;
        }
		
		int fl = 0;
        if (count == 3)
        {
            //cout << ans[25] << "   " << ans[24] << "   " << ans[23] << endl;
            if ((ans[25] == ans[24]) && (ans[23] == ans[24]))
            {
            	int l = ans[25];
            	for (int i = 0;i < a.size();i= i + l)
            	{
            		for (int j = i;j < i+l;j++)
            		{
            			if (a[i] != a[j])
            			{
            				fl = 1;
            				break;
            			}
            		}
            	}
            }
            
            if (fl == 0)
                cout << "OK" << endl;
            else
                cout << "Not OK" << endl;
        } else
            cout << "Not OK" << endl;
    }

    return 0;
}
