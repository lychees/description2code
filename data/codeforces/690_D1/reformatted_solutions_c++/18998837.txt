using namespace std;
typedef long long LL;
char ch[1000][1000];
int main() {
  int n, m;
  cin >> n >> m;
  for (int i = 1; i <= n; i++)
    scanf("%s", ch[i] + 1);
  int cnt = 0;
  for (int i = 1; i <= m; i++) {
    // cout<<ch[n][i]<<endl;
    if (ch[n][i] == 'B' && (i == 1 || ch[n][i - 1] == '.'))
      cnt++;
  }
  cout << cnt << endl;
}
