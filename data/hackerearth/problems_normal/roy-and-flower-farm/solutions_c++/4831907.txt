#include <bits/stdc++.h>
using namespace std;
#define forall(i,n) for(int i=0;i<n;i++)
#define sc(t) scanf("%d",&t)
#define pr(x) printf("%d\n",x)
#define MOD 1000000009
#define pb(x) push_back(x)
#define mp(a,b) make_pair(a,b)
#define INF 1<<28
#define ll long long
#define scll(x) scanf("%I64d",&x)
#define prll(x) printf("%I64d\n",x)
#define pii pair<int,int>
int X[105],Y[105];
int dp[105][10006];
int main()
{
    //ios_base::sync_with_stdio(false);
	int t;
	sc(t);
	while(t--)
	{
		memset(dp,0,sizeof(dp));
		int n,p;
		sc(n);
		sc(p);
		for(int i=1;i<=n;i++)
		{
			sc(X[i]);
			sc(Y[i]);
		}
		
		for(int i=1;i<=n;i++)
		{
			for(int j=0;j<=p;j++)
				{
					if(j-Y[i]>=0)
					{
						dp[i][j]=max(dp[i-1][j],X[i]+dp[i-1][j-Y[i]]);
					}
					else
						dp[i][j]=dp[i-1][j];
				}
		}
		int minspend=0;
		for(int i=0;i<=p;i++)
		{
			if(dp[n][minspend]-minspend<dp[n][i]-i)
				minspend=i;
		}
		printf("%d %d\n",minspend,p+dp[n][minspend]-minspend);
	}
    return 0;
}
