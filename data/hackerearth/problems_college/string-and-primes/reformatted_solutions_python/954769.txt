p = [2, 3, 5]
j = 7
while p[-1] < ord('z'):
    i = 0
    q = int(j**0.5)+1
    while p[i] < q:
        if j % p[i] == 0:
            break
        i += 1
    else:
        p.append(j)
    j += 2
for i in xrange(input()):
    q = filter(lambda x: ord(x) not in p, raw_input())
    if q:
        print ''.join(q)
        continue
    print 'NULL'
