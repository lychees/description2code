#include <bits/stdc++.h>


using namespace std;

int main()
{

    int cs;
    cin >> cs;
    for (int i = 0; i < cs; ++i)
    {
        int n, k;
        long long ans = 1;
        cin >> n >> k;
        for (int j = 0; j < n; ++j)
        {
            ans = (ans*k)%1000000007;
            --k;
        }
        cout << ans%(1000000007) << endl;
    }
    return 0;
}
