int main(int argc, char const *argv[])
{
	int i,n,t,k,j,ans,sum;
	cin>>t;
	TEST(t) {
		cin>>n>>k;
		ans = 1000000;
		sum = 0;
		int A[n];
		int cost[n];
		for(i=0;i<n;i++) {
			cin>>A[i];
		}
		for(i=1;i<=100;i++) {
			sum = 0;
			for(j=0;j<n;j++) {
				if(A[j] < i) {
					cost[j] = (i - A[j])*3;
				} else {
					cost[j] = (A[j] - i)*5;
				}
			}
			sort(cost,cost+n);
			for(j=0;j<k;j++) {
				sum += cost[j];
			}
			ans = min(ans, sum);
		}
		cout<<ans<<endl;
	}
	return 0;
}
