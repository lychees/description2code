#define ll long long

#define inf 1000000000
#define mod 1000000007

int dp[201][201];
int x[201], y[201], c[201];

int main()
{
    int t;
    scanf("%d", &t);
    while(t-- > 0)
    {
        int n, m;
        scanf("%d %d", &n, &m);
        for(int i = 0; i < n; i++)
        {
            scanf("%d %d", &x[i], &y[i]);
            c[i] = 1;
        }    
        for(int i = n; i < n+m; i++)
        {
            scanf("%d %d", &x[i], &y[i]);
            c[i] = 0;
        }
        int ans = 0;
        for(int i = 0; i < n+m; i++)
            for(int j = 0; j < n+m; j++)
                dp[i][j] = (x[i] - x[j]) * (x[i] - x[j]) + (y[i] - y[j]) * (y[i] - y[j]);
        for(int i = 0; i < n+m; i++)
            if (c[i] == 0)
            for(int j = 0; j < n+m; j++)
                for(int k = 0; k < n+m; k++)
                    dp[j][k] = min(dp[j][k], max(dp[i][j], dp[i][k]));
        for(int i = 0; i < n+m; i++)
            for(int j = 0; j < n+m; j++)
                if (c[i] == 0 && c[j] == 1)
                    ans = max(ans, dp[i][j]);
        printf("%.6f\n", sqrt(ans));
    }
	return 0;
}
