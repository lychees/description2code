#include <stdio.h>
#include<stdlib.h>
int cmpfunc(const void *a,const void *b)
{
	return -(*(int *)a-*(int *)b);
}	
int main()
{
   int T, N, i, j, count;
   int a[100000], s[100000];
   
   scanf("%d",&T);
   while(T--)
   {
   	count=0;
   	scanf("%d",&N);
   	
   	for(i=0;i<N;i++)
   	  scanf("%d",&s[i]);
   	  qsort(s,N,sizeof(int),cmpfunc);
   	  
   	  	for(i=0;i<N;i++)
   	  scanf("%d",&a[i]);
   	  qsort(a,N,sizeof(int),cmpfunc);
   	  
   	  
   	  for(i=0,j=0;i<N && j<N;)
   	    {
   	    	if(s[i] > a[j]) {count++; i++; j++; }
   	    	else
   	    	j++;
   	    }
   	    
   	  if(count == N) printf("YES\n"); else  printf("NO\n");
   	  
   }
    return 0;
}
