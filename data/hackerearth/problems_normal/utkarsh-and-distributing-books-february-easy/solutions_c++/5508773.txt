#include <iostream>
#include<algorithm>
#include<vector>
using namespace std;
int main()
{
    int T;
    cin>>T;
    while(T--){
    	int N;
    	cin>>N;
    	vector<int> V;
    	for(int i=0;i<N;i++){
    		int t;
    		cin>>t;
    		V.push_back(t);
    	}
    		
    	int sum=0,min=V[0];
    	for(int i:V){
    		sum+=i;
    		if(min>=i)
    			min=i;
    	}
    		
    	cout<<min-1<<" "<<sum-N<<endl;
    }
    return 0;
}
