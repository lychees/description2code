#include<iostream>
#include<cstdio>
#define s(a) scanf("%lld",&a)

using namespace std;

typedef long long ll;

ll ar[1000010];

int main()
{
    ll T,N,i,msf,meh;
    s(T);
    while(T--)
    {
    	s(N);
    	for(i=0;i<N;i++) s(ar[i]);
    	
    	msf=0; meh=0;
    	for(i=0;i<N;i++)
    	{
    		meh=meh+ar[i];
    		if(meh<0) meh=0;
    		if(msf<meh) msf=meh;
    	}
    	printf("%lld\n",msf);
    }
    return 0;
}
