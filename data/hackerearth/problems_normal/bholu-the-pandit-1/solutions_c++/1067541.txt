#include <bits/stdc++.h>

#define ll long long int
#define s(a) scanf("%lld",&a)

using namespace std;

ll mod=1000000007;

int main()
{
    ll t,a,b,c,ans;
    s(t);
    while(t--) {
        s(a);s(b);s(c);
        ans=__gcd(a,b);
        ans=__gcd(ans,c);
        cout<<ans<<" ";
        a=a/ans;
        b=b/ans;
        c=c/ans;
        ans=1;
        ans=(a*b)%mod;
        ans=(ans*c)%mod;
        cout<<ans<<endl;
    }
    return 0;
}
