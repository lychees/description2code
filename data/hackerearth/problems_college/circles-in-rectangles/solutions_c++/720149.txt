#include <iostream>

using namespace std;

int main()
{
   
    long long int l,b,d,ans;
    int t;
    cin>>t;
    while(t--)
    {
    	cin>>l>>b>>d;
    	//ans=((l/d)*(b/d))%1000000007;
    	ans=(l/d)%1000000007;
    	ans=(ans*((b/d)%1000000007))%1000000007;
    	cout<<ans<<endl;
    }
    return 0;
}
