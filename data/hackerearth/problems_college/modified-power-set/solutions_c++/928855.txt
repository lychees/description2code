#include <bits/stdc++.h>
using namespace std;

char s[1000005];

int main()
{
	int t;
	scanf("%d",&t);
	while (t --) {
		int i,n,ans;
		int a[150] = {};
		scanf("%s",s);
		for (i = 0; s[i] != '\0'; i++) {
			a[s[i]] = 1;
		}
		n = 0;
		for (i = 0; i < 150; i++) {
			n = n + a[i];
		}
		ans = n * (n + 1) / 2;
		printf("%d\n",ans);
	}
    return 0;
}
