    #include <iostream>
    using namespace std;
     
    int main()
    {
        string str;
        int t,s,e,count;
        cin>>t;
        
        while(t--)
        {
        	cin>>str;
        	count=0;
        	e=str.size()-1;
        	s=0;
        	while(s<=e)
        	{
        		if(str[s]!=str[e])
        		{
        			 e=str.size()-1;
        			 count=0;
        			 s++;
        		}
        		else
        		 {
        		 	  (s!=e) ? count+=2 : count+=1; 
        		 	  s++;e--;
        		 }
        	}
        	
        	cout<<count+2*(str.size()-count)<<endl;
        	
        }
        return 0;
    }
