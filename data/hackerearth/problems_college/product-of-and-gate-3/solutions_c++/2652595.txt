#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
       long long  int a,b;
        cin>>a>>b;
        long long int x=a^b;
        long long int temp=x>>1;
        while(temp)
        {
            x=x|temp;
            temp=temp>>1;
        }
        cout<<(a&b&~x)<<endl;
    }
}
