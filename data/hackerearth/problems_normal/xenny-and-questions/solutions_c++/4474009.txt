#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n,k;
    	scanf("%d %d",&n,&k);
    	int i;
    	vector<int> a(n);
    	for(i=0;i<n;i++)
    		scanf("%d",&a[i]);
    	sort(a.begin(),a.end(),greater<int>());
    	long long ans=0;
    	for(i=0;i<min(n,k);i++)
    	{
    		ans+=a[i];
    	}
    	printf("%lld\n",ans);
    }
    return 0;
}
