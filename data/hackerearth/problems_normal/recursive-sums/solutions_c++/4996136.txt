#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int m;
		int flag=0;
		cin>>m;
		int b[m];
		long long int a[m];
		for(int i=0;i<m;i++)
		{
			cin>>a[i]>>b[i];
		}
		long long int sum=0;
		for(int i=0;i<m;i++)
		{
			sum=(sum+(a[i]*b[i]))%9;
			if (flag==0)
			{
				flag=b[i];
			}
		}
		if(sum==0 && flag!=0)
		{
			cout<<"9"<<endl;
		}
		else if(sum==0 && flag==0)
		{
			cout << "0"<< endl;
		}
		else
		{
			cout<<sum<<endl;
		}

	}
	return 0;
}
