#include <iostream>
using namespace std;
const long long MOD= 1000000009;

void mult(unsigned long long a[2][2],unsigned long long b[2][2])
 {unsigned long long x,y,z,p;
  x=(a[0][0]*b[0][0]+a[0][1]*b[1][0])%MOD;
  y=(a[0][0]*b[0][1]+a[0][1]*b[1][1])%MOD;
  z=(a[1][0]*b[0][0]+a[1][1]*b[1][0])%MOD;
  p=(a[1][0]*b[0][1]+a[1][1]*b[1][1])%MOD;
  a[0][0]=x;
  a[0][1]=y;
  a[1][0]=z;
  a[1][1]=p;
  
 }

 void power(unsigned long long f[2][2],unsigned long long n)
 {if(n==0 || n==1) return;
 
  unsigned long long m[2][2]={{1,1},{1,0}};
  power(f,n/2);
  mult(f,f);
  if(n%2==1)
     mult(f,m);
  
 }

unsigned long long fib(unsigned long long n)
 {unsigned long long f[2][2]={{1,1},{1,0}};
  if (n==0) 
    return 1;
  power(f,n);
  return f[0][0]%MOD;
 }
 
int main()
{  unsigned long long n;
   unsigned  long long t;
    cin>>t;
    while(t--)
      {cin>>n;
       cout<<(fib(n)*fib(n-1))%MOD<<endl;
      }
}
