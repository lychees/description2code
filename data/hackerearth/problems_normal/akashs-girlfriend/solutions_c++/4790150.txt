#include <bits/stdc++.h>
#define ll long long int
using namespace std;

const int mod=1e9+9;

ll power (ll x, ll y)
{
	ll res=1;
	while (y>0)
	{
		if (y&1)
		{
			res=(res*x)%mod;
			y--;
		}
		x=(x*x)%mod;
		y/=2;
	}
	return res;
}

int main()
{
    ll t,n,ans;
    scanf ("%lld",&t);
    while (t--)
    {
    	scanf ("%lld",&n);
    	ans=(power(10,n)*power(2,mod-2))%mod;
    	ans-=(power(8,n)*power(2,mod-2))%mod;
		if (ans<0)
			ans+=mod;
    	printf ("%lld\n",ans);
    }
    return 0;
}
