#include <bits/stdc++.h>

using namespace std;

typedef long long ll;

int f[26];
int sec[26];
int main () {
	int n;
	cin >> n;
	string s;
	cin >> s;
	for (int i = 0; i < s.length(); i++) {
		if (i < s.length() / 2)
			f[s[i] - 'a']++;
		else
			sec[s[i] - 'a']++;
	}
	int budget = 0;
	int case1 = 0;
	for (int i = 0; i < 25; i++) {
		if (f[i] + budget <= sec[i + 1]) {
			case1 += f[i] + budget;
			budget = 0;
		} else {
			case1 += sec[i + 1];
			budget = budget + f[i] - sec[i + 1];
		}
	}
	budget = 0;
	int case2 = 0;
	for (int i = 0; i < 25; i++) {
		if (sec[i] + budget <= f[i + 1]) {
			case2 += sec[i] + budget;
			budget = 0;
		} else {
			case2 += f[i + 1];
			budget = budget + sec[i] - f[i + 1];
		}
	}
	int case3 = 0;
	for (int i = 0; i < 26; i++)
		case3 += min (f[i], sec[i]);
	int l = s.length() / 2;
	cout << min (l - case3, min (max (l - case1, f[25] + sec[0]), max (l - case2, f[0] + sec[25])));
}
