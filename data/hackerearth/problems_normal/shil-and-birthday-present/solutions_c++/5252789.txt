#include<bits/stdc++.h>
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	
	long long n;
	
		scanf("%lld",&n);
	
	set<long long> s;
	while(n--)
	{
		long long a;
		scanf("%lld",&a);
		s.insert(a);
	}
	long long size=s.size();
	
	cout<<(size*(size-1))/2 << endl;
	return 0;
}
