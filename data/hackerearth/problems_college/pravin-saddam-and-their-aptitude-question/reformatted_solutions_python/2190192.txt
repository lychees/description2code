li = [1, 2, 6, 4]
for _ in range(int(raw_input())):
    sum = 0
    m, n = map(int, raw_input().split())

    if m > 4:
        print("0")
    else:
        if n > 4:
            n = 4
        m -= 1
        for i in range(m, n):
            sum += li[i]
            sum = sum % 10
        print(sum % 10)
