#include<bits/stdc++.h>
#define ll long long
#define ld long double
#define f first
#define s second
#define pb push_back
#define M 1000000001
#define MX 1000000
#define PI 3.141593
using namespace std;
typedef pair<ll,ll> ii;
ll arr[1000005],diff=0;
void pre()
{
    for(ll i=1;i<=MX;i++)
        arr[i]=-1;
    for(ll i=1;i<=MX;i++)
    {
        if(arr[i]==-1)
        {
            diff++;
            arr[i]=i+diff;
            if(i+diff<=MX)
              arr[i+diff]=i;
        }
    }
}
int main()
{
    pre();
    ll t;
    cin>>t;
    while(t--)
    {
        ll x,y;
        cin>>x>>y;
        if(!x||!y)
            {
                if(!x&&!y)
                    cout<<"Don\'t Play\n";
                else cout<<"Play\n";
            }
        else if(arr[x]==y)
            cout<<"Don\'t Play\n";
        else cout<<"Play\n";
    }
}
