#include<iostream>
#include <stdio.h>
#include <algorithm>
#include <list>
using namespace std;
#define push(a) push_back(a)
#define TEST while(t--)
#define SCAN(a) scanf("%d",&a)

int main()
{
   int t,n,i,j,tmp,tot;
   
	SCAN(t);
	TEST
	{
		list<int> l1,l2;
		SCAN(n);tot=0;
		for(i=0;i<n;i++)
		{
			SCAN(tmp);l1.push(tmp);
		}
		for(i=0;i<n;i++)
		{
			SCAN(tmp);l2.push(tmp);
		}
		l1.sort();
		l2.sort();
		
		list<int>::iterator p1=l1.begin();
		list<int>::iterator p2=l2.begin();
		for(;p1!=l1.end()&&p2!=l2.end();p1++,p2++)
			tot+=*(p1)**(p2);
		printf("%d\n",tot);
	}
    return 0;
}
