
#include <iostream>
#include <string>

int main()
{
	int t;
	std::cin>>t;
	while(t--)
	{
		std::string s;
		std::cin>>s;

		int ans=-1;
		for(int i=0 ; i<s.size()/2 ; ++i)
		{
			if(s[i]!=s[s.size()-1-i])
			{
				if(s[i]==s[s.size()-1-(i+1)])
					ans=s.size()-1-i;
				else
					ans=i;
				break;
			}
		}
		std::cout<<ans<<std::endl;
	}
}
