using namespace std;
ll parent[100005];
void ini()
{
    for(ll i=0;i<=100001;i++)
        parent[i]=i;
}
ll root(ll a)
{
    while(a!=parent[a])
    {
        parent[a]=parent[parent[a]];
        a=parent[a];
    }
    return a;
}
void uni(ll a,ll b)
{
    ll p=root(a);
    ll q=root(b);
    parent[p]=q;
}
int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
   ll t;
   cin>>t;
   while(t--)
   {
       ini();
       ll n,m;
       cin>>n>>m;
       ll arr[n+1],brr[n+1];

       for(ll i=1;i<=n;i++)
        cin>>arr[i];
       for(ll i=1;i<=n;i++)
        cin>>brr[i];
       for(ll i=1;i<=m;i++)
       {
           ll u,v;
           cin>>u>>v;
           if(root(arr[u])!=root(arr[v]))
           uni(arr[u],arr[v]);
       }
       ll flag=0;
       for(ll i=1;i<=n;i++)
       {

           if(root(arr[i])!=root(brr[i]) && arr[i]!=brr[i])
           {
               flag=1;
               break;
           }
       }
        if(flag==0)
            cout<<"YES\n";
        else
            cout<<"NO\n";
   }
}
