#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <algorithm>

using namespace std;

int min(int a, int b){
    return (a>b)?b:a;
}

int max_len(char *str)
{
    int n = strlen(str);
    int cur_len = 1; 
    int max_len = 1; 
    int prev_index;  
    int i;
    int *visited = (int *)malloc(sizeof(int)*10000);
    for (i = 0; i < 10000;  i++)
        visited[i] = -1;
    visited[str[0]] = 0;
    for (i = 1; i < n; i++){
        prev_index =  visited[str[i]];

        if (prev_index == -1 || i - cur_len > prev_index)
            cur_len++;
        else{
            if (cur_len > max_len)
                max_len = cur_len;
 
            cur_len = i - prev_index;
        }
 
        visited[str[i]] = i;
    }
    if (cur_len > max_len)
        max_len = cur_len;
 
    free(visited); 
 
    return max_len;
}

int main(){
	
	int t;
	scanf ("%d",&t);
	while (t--){
	 char str[10000];
	 scanf ("%s",str);
	 int n = max_len(str); 
	 printf("%d\n",n);
	}
	return 0;
}
