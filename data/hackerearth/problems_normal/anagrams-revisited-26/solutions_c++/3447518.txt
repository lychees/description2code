#include <iostream>
#include <algorithm>
#include <map>
#include <stdio.h>
#include <string.h>

using namespace std;

int main()

{
	long int n;
	scanf("%ld",&n);
	map <string,long int> m;
	while(n--)
	{
		char a[50];
		scanf("%s",a);
		int l;
		l=strlen(a);
		sort(a,a+l);
		
		m[a]++;
	
	}
	map<string,long int>::iterator it;
	long int max=-100000;
	for(it=m.begin();it != m.end();it++)
	{
		
		if(max < it->second)
			max=it->second;
	}
	cout <<max<<endl;
   
    return 0;
}
