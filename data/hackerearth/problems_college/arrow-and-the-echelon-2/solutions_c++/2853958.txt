#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
	    vector<int> v;
		int x;
		cin>>x;
		if(x%3==0)
        {
            for(int i=0;i<x;i++)
            cout<<"5";
            cout<<endl;
        }
        else if(x%5==0)
        {
            for(int i=0;i<x;i++)
            cout<<"3";
            cout<<endl;
        }
		else if((x%8)%3==0)
		{
			for(int i=0;i<(x-5);i++)
            v.push_back(5);
			for(int i=0;i<5;i++)
			v.push_back(3);
			sort(v.rbegin(),v.rend());
            for(int i=0;i<v.size();i++)
            cout<<v[i];
            cout<<endl;
		}
		else if((x%8)%5==0)
		{
			for(int i=0;i<(x-3);i++)
            v.push_back(3);
			for(int i=0;i<3;i++)
			v.push_back(5);
			sort(v.rbegin(),v.rend());
            for(int i=0;i<v.size();i++)
            cout<<v[i];
            cout<<endl;
		}
		else
		{
			cout<<"-1";
			cout<<endl;
		}
	}
return 0;
}
