#include<cstdio>
#include<iostream>
#include<string.h>
using namespace std;
main()
{
	int T;
	cin>>T;
	while(T>0)
	{
	string S = "";
	cin>>S;
	string temp = "";
	temp = temp + S[0];
	int len = S.length();
	for(int i=1;i<len;i++)
	{
		if(S[i]!=S[i-1])
		{
			temp = temp + S[i];	
		}
	}
	cout<<temp<<endl;
	T=T-1;
   }
}
