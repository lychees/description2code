def even(n):
    return n % 2 == 0


T = input()


while T > 0:

    N = input()
    A, B, C = map(int, raw_input().split())
    S = ''
    flag = True
    if even(C):
        NUM_PARITY = 'e'
    else:
        NUM_PARITY = 'o'

    if even(A):
        S += 'e'
    else:
        S += 'o'

    if even(B):
        S += 'e'
    else:
        S += 'o'

    if S == 'oe' or S == 'eo':
        S = 'T'

    if S == 'oo' or S == 'ee':
        S = 'W'

    N -= 1
    while N > 0:
        A, B, C = map(int, raw_input().split())
        s = ''

        if even(C):
            np = 'e'
        else:
            np = 'o'

        if even(A):
            s += 'e'
        else:
            s += 'o'

        if even(B):
            s += 'e'
        else:
            s += 'o'

        if s == 'oe' or s == 'eo':
            s = 'T'

        if s == 'oo' or s == 'ee':
            s = 'W'

        if (NUM_PARITY == np):
            if (S != s):
                flag = False
        else:
            if (S == s):
                flag = False

        N -= 1

    if (flag):
        print "Yes"
    else:
        print "No"

    T -= 1
