#include <bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;
ll int a[1000010],dp[1000010];
ll int myfind(ll int s,ll int e)
{
	return 2*(dp[e]-dp[s-1]);
}
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int n,k;
	cin>>n>>k;
	ll int i;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
		if(i==0)
		dp[i]=a[i];
		else
		dp[i]=a[i]+dp[i-1];
	}
	ll int s1=0,s2=0;
	for(i=0;i<k;i++)
	{
		s1=(s1+((((i+1)*(i+1))%mod)*a[i])%mod)%mod;
		s2=(s2+(((2*i)+1)*a[i])%mod)%mod;
		
	}
	cout<<s1<<" ";
	for(i=k;i<n;i++)
	{
		s1=(s1-s2+mod)%mod;
		s1=(s1+(((k*k)%mod)*a[i])%mod)%mod;
		s2=(s2+(((2*(k-1))+1)%mod)*a[i])%mod;
		s2=(s2-a[i-k]+mod)%mod;
		s2=(s2-myfind(i-k+1,i-1)+mod)%mod;
		cout<<s1<<" ";
	}
	cout<<endl;
	return 0;
}
