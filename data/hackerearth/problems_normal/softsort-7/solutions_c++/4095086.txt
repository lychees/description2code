#include <stdio.h>
#include<iostream>
#include<cstring>
#define gc getchar_unlocked
#define M 1000000007
using namespace std;
typedef long long ll;

void scanint(int &x)
{
    register int c = gc();
    x = 0;
    int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}
void scanlong(long long &x)
{
    register int c = gc();
    x = 0;
    int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}
long long a[1000001];
void func(){
	a[0]=1;
	for(int i=1;i<=1000000;i++){
		a[i]=(a[i-1]*(i))%M;
	}
}
int main(void) {
    int t;
    scanint(t);
    func();
    while(t--)
    {
    	int n;
    	scanint(n);
    	cout<<((3*a[n])%M+3)%M<<endl;
    }
	return 0;
}
