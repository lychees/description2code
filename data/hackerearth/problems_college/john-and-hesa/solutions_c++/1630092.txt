#include <iostream>
using namespace std;

int main()
{
    long int t,x,p,count;
    
    cin>>t;
    
    while(t--)
    {
    	cin>>x>>p;
    	
    	count =0;
    	
    	while(x%p==0 && x!=1)
    	{
    		count++;
    		x=x/p;
    	}
    	
    	if(x==1)
    	{
    		x=0;
    	}
    	
    	cout<<count<<" "<<x<<endl;
    	
    	
    }
    return 0;
}
