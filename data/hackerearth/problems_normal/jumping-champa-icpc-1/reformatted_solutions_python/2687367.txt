t = int(raw_input())
list_a = []
minh = -1
maxh = 3e6+5


def cal(list_a, mn, mx):
    mn = int(mn)
    mx = int(mx)
    n = list_a[0]
    q = list_a[1]
    f = map(int, raw_input().split())
    f = list(f)
    for xf in f[:]:
        mn = min(mn, xf)
        mx = max(mx, xf)
    ans = q * abs(mx-mn)
    print(ans)


for x in range(t):
    list_a = map(int, raw_input().split())
    list_a = list(list_a)
    cal(list_a, maxh, minh)
