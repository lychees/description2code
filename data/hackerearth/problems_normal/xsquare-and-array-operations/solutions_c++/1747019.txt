using namespace std;

typedef pair<int, int> pii;

const double eps = 1e-9;
const double pi = acos(-1.0);

const int maxn = (int)1e5 + 10;
const int inf = (int)2e9;

int a[maxn];
int nx[maxn],pr[maxn];

int main() {
	#ifdef LOCAL
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	#endif

	int T; cin >> T;

	while(T--) {
		int n; scanf("%d",&n);
		multiset<pii> st;
		for (int i = 0; i < n; i++) {
			scanf("%d",&a[i]);
			st.insert(mp(a[i],i));
			nx[i] = i + 1;
			pr[i] = i - 1;
		}
		nx[n - 1] = -1;
		ll res = 0;
		for (int i = 0; i < n - 1; i++) {
			pii cur = *st.begin();
			st.erase(st.begin());
			int val = cur.first;
			int sec = inf;
			int pos = cur.second;
			if (pr[pos] != -1) {
				sec = min(sec,a[pr[pos]]);
				nx[pr[pos]] = nx[pos];
			}
			if (nx[pos] != -1) {
				sec = min(sec,a[nx[pos]]);
				pr[nx[pos]] = pr[pos];
			}
			res += sec;
		}
		cout << res << endl;
	}
	
	return 0;
}
