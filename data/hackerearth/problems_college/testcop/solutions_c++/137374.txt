#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	 unsigned long long int n,t,g,arr[101];
	 cin>>t;
	 while(t--)
	 {
	 	cin>>n;
	     	for(int i=0;i<n;i++)
	 	{
	 		//g=__gcd
	 		cin>>arr[i];
	 	}
	     
	     if(n==1)
	     cout<<"YES"<<"\n";
	     else
	     {
	     	g=arr[0];
	     
	 	for(int i=1;i<n;i++)
	 	{
	 		g=__gcd(g,arr[i]);
	 	}
	     }
	     if(g==1)
	     cout<<"YES"<<"\n";
	     else
	     cout<<"NO"<<"\n";
	 }
	 
	 
    //cout << "Hello World!" << endl;
    return 0;
}
