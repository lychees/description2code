t = input()
for _ in xrange(t):
    n = input()
    costs = []
    for i in xrange(n):
        x, y, z = map(int, raw_input().split())
        costs += [[x, y, z]]
    for i in xrange(1, n):
        c1 = costs[i-1]
        c2 = costs[i]
        c2[0] += min(c1[1], c1[2])
        c2[1] += min(c1[0], c1[2])
        c2[2] += min(c1[1], c1[0])
        costs[i] = c2[:]
    print min(costs[n-1])
