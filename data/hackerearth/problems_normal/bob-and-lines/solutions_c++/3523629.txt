#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    while(t!=0)
    {
    	int n,q;
    	cin>>n>>q;
    	
    	vector<int> x(100001,0);
    	vector<int> y(100001,0);
    	
    	for(int i=0;i<n;i++)
    	{
    		int x1,y1,x2,y2;
    		cin>>x1>>y1>>x2>>y2;
    		
    		int minx=min(x1,x2);
    		int maxx=max(x1,x2);
    		x[minx]+=1;
    		x[maxx+1]-=1;
    		
    		int miny=min(y1,y2);
    		int maxy=max(y1,y2);
    		y[miny]+=1;
    		y[maxy+1]-=1;
    	}
    	
    	
    	
    	for(int i=1;i<=100000;i++)
    	{
    		x[i]+=x[i-1];
    		y[i]+=y[i-1];
    	}
    	
    	
    	
    	while(q!=0)
    	{
    		int k,l;
    		cin>>k>>l;
    		
    		if(k==0)
    		{
    			cout << y[l] << '\n';
    		}
    		else
    		{
    			cout << x[k] << '\n';
    		}
    		
    		q--;
    	}
    	
    	t--;
    }
    return 0;
}
