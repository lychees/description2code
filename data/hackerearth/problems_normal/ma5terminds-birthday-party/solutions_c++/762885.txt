#include <iostream>
#include <cstdio>
#include <cmath>
#include <cstring>

#define MAX_SIZE 1000001
using namespace std;

int composite[MAX_SIZE];
void runEratosthenesSieve(int upperBound)
{
    int upperBoundSquareRoot = (int)sqrt((double)upperBound);
    memset(composite, 0, sizeof(int) * (MAX_SIZE - 1));
    for (int m = 2; m <= upperBoundSquareRoot; m++)
    {
        if (!composite[m])
        {
            for (int k = m * m; k <= upperBound; k += m)
                composite[k] = 1;
        }
    }
    composite[1] = 1;
    composite[4] = 0;
    for (int  i = 1; i < MAX_SIZE; i++)
        composite[i] += composite[i-1];
}


int main()
{
    int T, L, R;
    runEratosthenesSieve(MAX_SIZE-1);
    scanf("%d", &T);
    while(T-- > 0)
    {
        scanf("%d %d", &L, &R);
        printf("%d\n", composite[R] - composite[L-1]);
    }
    
    return 0;
}
