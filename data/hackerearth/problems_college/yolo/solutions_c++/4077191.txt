#include <bits/stdc++.h>
using namespace std;
int a[55],b[5],c[5],d[5];
int main()
{
    
   while(scanf("%d",&b[0])==1)
   {
   	if(b[0]==0)break;
   	scanf("%d%d%d%d",&b[1],&b[2],&c[0],&c[1]);
   	sort(b,b+3);
   	int i;
   	for(i=0;i<=52;i++)
   	a[i] = 0;
   	a[b[0]] = a[b[1]] = a[b[2]] = a[c[0]] = a[c[1]] = 1;
   	int ff = 0;
   	for(i=1;i<=52;i++)
   	{
   	if(a[i]==0)
   	{
   		d[0] = c[0];
   		d[1] = c[1];
   		d[2] = i;
   		sort(d,d+3);
   		int u[3];
   		u[0]  = u[1] = u[2] = 0;
   		int w = 0;
   		int j;
   		int k;
   		for(j=0;j<3;j++)
   		{
   			for(k=0;k<3;k++)
   			{
   				if(u[k]== 0 && b[k] > d[j]){
   					u[k] = 1;
   					break;
   				}
   			}
   		}
   		if(u[0] + u[1] + u[2] <2){
   			ff = 1;
   			break;
   		}
   	}
   	}
   	if(ff)printf("%d\n",i);
   	else printf("-1\n");
   }
    return 0;
}
