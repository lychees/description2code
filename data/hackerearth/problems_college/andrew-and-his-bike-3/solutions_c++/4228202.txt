#include<iostream>
#include<vector>
using namespace std;
vector<int> g[100005];
int n;
int ans[100005],vis[100005];
int dfs(int x)
{
	vis[x]=1;
	int wt=0;
	for(int i=0;i<g[x].size();i++)
	{
		if(!vis[g[x][i]])
		wt+=dfs(g[x][i]);
	}
	ans[x]=wt;
	return ans[x]+1;
	
}
int main()
{
	cin>>n;
	for(int i=1;i<n;i++)
	{
		int u,v;
		cin>>u>>v;
		g[u].push_back(v);
		g[v].push_back(u);
	}
	ans[1]=dfs(1);
	for(int i=1;i<=n;i++)
	cout<<ans[i]+((i==1)?-1:1)<<" ";
	return 0;
}
