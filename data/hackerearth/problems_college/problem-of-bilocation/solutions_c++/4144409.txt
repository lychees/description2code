#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int a,b,c,d,s;
		scanf("%d%d%d%d%d",&a,&b,&c,&d,&s);
		if(b<=c-s)
		cout<<b<<endl;
		else
		{
			if(c-s>=a)
			cout<<c-s<<endl;
			else
			cout<<"STAY IN LONDON"<<endl;
		}
	}
	return 0;
}
