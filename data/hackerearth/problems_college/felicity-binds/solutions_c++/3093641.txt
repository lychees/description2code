#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t,n,ar1[1009],ar2[1009],i,j;
    scanf("%d",&t);
    while(t--)
    {
    	int ans=0;
    	scanf("%d",&n);
    	for(i=0;i<n;i++)
    	scanf("%d",&ar1[i]);
    	for(i=0;i<n;i++)
    	scanf("%d",&ar2[i]);
    	sort(ar1,ar1+n);
    	sort(ar2,ar2+n);
    	for(i=0;i<n;i++)
    	ans+=ar1[i]*ar2[i];
    	printf("%d\n",ans);
    }
    return 0;
}
