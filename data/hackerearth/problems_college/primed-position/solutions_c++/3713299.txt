#include<bits/stdc++.h>
using namespace std;
#define N 10000010
#define mod 1000000007
typedef long long ll;
int a[N]={0};
int p=0;
int primes[N];
int pos[N];
void sieve()
{
    a[0]=a[1]=1;
    for(int i=2;i*i<N;i++)
    {
        if(a[i]==0)
        {
            for(int j=2;i*j<N;j++)
            {
                a[i*j]=1;
            }
        }
    }
    for(int i=2;i<N;i++)
        if(a[i]==0)
            primes[p++]=i;
    int j=0;
    for(int i=2;i<N;i++)
    {
        if(a[i]==0)
        {
            pos[j++]=primes[i-1];
        }
    }
}
int main()
{
    int t;
    sieve();
    scanf("%d",&t);
    while(t--)
    {
        int n,m;
        scanf("%d %d",&n,&m);
        ll a = pos[n-1];
        ll b = pos[m-1];
        ll ans = ((a%mod)*(b%mod))%mod;
        printf("%lld\n", ans);
    }
    return 0;
}
