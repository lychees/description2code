#include<iostream>
#include<algorithm>
using namespace std;
typedef long long int ll;
int main()
{
    ll t,n,i;
    cin>>t;
    while(t--)
    {
        cin>>n;
        ll x[n];
        for(i=0;i<n;i++)
        {
            cin>>x[i];
        }
        sort(x,x+n);
        for(i=0;i<n;i++)
        {
            cout<<x[i]<<' ';
        }
        cout<<'\n';
    }
    return 0;
}
