T = int(raw_input())
for t in range(T):
    N = int(raw_input())
    arr = [int(x) for x in raw_input().split()]
    count_odd = 0
    count_even = 0
    for i in range(N):
        if arr[i] % 2 == 0:
            count_even += 1
        else:
            count_odd += 1

    print count_odd*count_even
