# include <stdio.h>
# include <iostream>
# include <cstring>
# include <algorithm>
using namespace std;
static long long n,ans,c;
char a[1000005];
int main ()
{
int t;
cin>>t;
while(t--)
{
scanf("%s",a);
n=strlen(a);
sort(a,a+n);
c=1;
for(int i=1;i<n;i++)
if(a[i]!=a[i-1])c++;
ans=c*(c+1);
ans/=2;
printf("%lld\n",ans);
}
return 0;}
