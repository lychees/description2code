#include <iostream>
using namespace std;

int main()
{
    int m;
    cin >> m;
    int ar[m][m];
    int i,j;
    for(i = 0;i < m;i++)
    {
    	for(j = 0;j < m;j++)
    	{
    		cin >> ar[i][j];
    	}
    }
    
    long long int a=0;
    long long int b=0;
    
    for(i = 0;i < m;i++)
    {
    	a += ar[i][i];
    	b += ar[i][m-i-1];
    }
    //cout << a << " " << b << endl;
    long long int ans= abs(a-b);
    cout << ans << endl;
    return 0;
}
