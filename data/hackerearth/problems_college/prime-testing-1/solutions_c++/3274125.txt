#include <bits/stdc++.h>

using namespace std;

int main()
{
    long long int i, n, a[1005], b[1005], j, t, ans, k;

	for (i = 0; i < 1005; i++) {
		a[i] = 1;
	}

	a[0] = 0;
	a[1] = 0;
	for (i = 2; i < 1005; i++) {
		if (a[i]) {
			for(j = 2 * i; j < 1005; j+=i) {
				a[j] = 0;
			}
		}
	}
	
	cin>>t;
	
	while(t--) {
        cin>>n;
        ans = 0;
        for (i = 0; i < n; i++) {
            cin>>k;
            
            if (a[k]) {
                ans++;
            }
        }
        cout<<ans<<endl;
	}
	
	
    return 0;
}
