int numbits(int n) {
  return n ? 1 + numbits(n & (n - 1)) : 0;
}

int main() {
  //freopen("in.txt", "r", stdin);
  //freopen("out.txt", "w", stdout);
  int n; cin >> n;
  vector<int> vi(n);
  ll res = 0;
  for (int i = 0; i < n; ++i) {
    scanf("%d", &vi[i]);
    res += ll(i + 1) * vi[i];
  }
  ll total = accumulate(all(vi), 0LL);
  ll cur = res;
  for (int i = 0; i < n; ++i) {
    cur -= total;
    cur += ll(n) * vi[i];
    res = max(res, cur);
  }
  cout << res << endl;
  return 0;
}
