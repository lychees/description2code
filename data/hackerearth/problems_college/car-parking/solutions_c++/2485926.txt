#include <iostream>
#include<string>
using namespace std;

bool checkSpot(string str,int index);

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int count=0,i;
    	string str;
    	cin>>str;
    	for(i=0;i<str.size();i++)
    	{
    		if(checkSpot(str,i))
    		{
    			count++;
    		}
    	}
    	cout<<count<<endl;
    }
    return 0;
}

bool checkSpot(string str,int index)
{
	if(str[index]=='B' || str[index]=='S' || str[index]=='D')return false;
	else if(index < str.size()-1 && (str[index+1]=='B' || str[index+1]=='S'))return false;
	else if(index < str.size()-2 && str[index+2]=='B')return false;
	else if(index >0 && str[index-1]=='S')return false;
	return true;
}
