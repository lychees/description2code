#include <bits/stdc++.h>
using namespace std;
int main()
{
 int i,t,n,k[10],a;
 cin>>t;
 while(t--){
     cin>>n;
     for(i=0;i<10;i++)
     k[i]=0;
     
     for(i=0;i<n;i++){
     cin>>a;
     k[a%10]++;}
     int s=0;
     for(i=0;i<10;i++)
     if(k[i]>0)
     s=s+(k[i]-1);
     
     cout<<s<<endl;
     
     
 }

}
