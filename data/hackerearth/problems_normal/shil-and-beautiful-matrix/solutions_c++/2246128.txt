#include <bits/stdc++.h>

using namespace std;

typedef long long ll;

const long long mod = 1e9+7;

long long q [1500000];
long long fact [1500000];

int main () {
	int n, m;
	cin >> n >> m;
	fact[1] = 1;
	fact[0] = 1;
	for (long long i = 2; i < 1500000; i++) 
		fact [i] = (i * fact [i - 1]) % mod;
	for (int i = 1; i <= n; i++) {
		for (int j = 1; j <= m; j++)
			q[i^j]++;
	}
	long long ans = 1;
	for (int i = 0; i < 1500000; i++)
		ans = (ans * fact[q[i]]) % mod;
	cout << ans;
}
