#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long n;
    cin>>n;
    int ar[100001];
    for(int i=0;i<n;i++)
    {
        cin>>ar[i];
    }
    long long sum,msum;
    msum=sum=0;

    for(int i=0;i<n;i++)
    {
        sum+=ar[i];
        if(sum<0)
        sum=0;
        if(sum>msum)
            msum=sum;

    }
    cout<<msum;
    return 0;
}
