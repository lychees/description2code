#include <iostream>
#include<sstream>
#include<string>
#include<cstring>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	string n;
   	cin>>n;
   	int len=n.length();
   	for(long long int i=0;i<len;i++)
   	{
   	switch(n[i])
   	{
   		case '2':cout<<"cde";
   		break;
   		case '6':cout<<"b3";
   		break;
   		case '7':cout<<"acf";
   		break;
   		case '9':cout<<"c6a";
   		break;
   	}
   	
   }
   cout<<"\n";
   
}
}
