n = input()
s = raw_input().strip()
arr = [0 for i in xrange(n)]
for j in xrange(2, n+1):
    for i in xrange(n-j+1):
        sub = s[i:i+j]
        if sub[-1] == '0' and sub[-2] == '0':
            arr[j-1] += 1
for i in xrange(input()):
    k = input()
    if k > 1:
        print arr[k-1]
    else:
        print 0
