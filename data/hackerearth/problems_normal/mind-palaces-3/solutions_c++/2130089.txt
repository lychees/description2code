#include <stdio.h>

int main()
{
    long int n,m,i,j,q,z,k;
    long long int a[1001][1001];
    scanf("%ld%ld",&n,&m);
    for(i=0;i<n;i++)
    {
    	for(j=0;j<m;j++)
    	{
    		scanf("%lld",&a[i][j]);
    	}
    }
    scanf("%ld",&q);
    while(q--)
    {
    	k=0;
    	scanf("%ld",&z);
    	for(i=0;i<n;i++)
    	{
    		for(j=0;j<m;j++)
    		{
    			if(a[i][j]==z)
    			{
    				printf("%ld %ld\n",i,j);
    				k++;
    				break;
    			}
    		}
    		if(k==1)
    			break;
    	}
    	if(k!=1)
    		printf("-1 -1\n");
    }
    return 0;
}
