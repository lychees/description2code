#include <iostream>
#include <algorithm>
using namespace std;

int a[1000007];

int main()
{
    int n,k;
    cin>>n>>k;
    for(int i=0;i<n;i++) cin>>a[i];
    int lf=0, rg=k;
    int ans=0;
    while(rg < n)
    {
        int x = a[lf];
        int y = a[rg];
        int z = abs(x-y);
        int w = abs(x+y);
        if(z%k==0 || w%k==0) ans++;
        rg++;
        lf++;
    }
    cout<<ans<<"\n";
}
