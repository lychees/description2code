#include<bits/stdc++.h>
using namespace std;

int check(queue<int> calling, queue<int> ideal)
{
	int count = 0;
	while(!(calling.empty() && ideal.empty()))
	{
		if(calling.front() == ideal.front())
		{
			calling.pop();
			ideal.pop();
			count++;
		}
		else 
		{
			int element =  calling.front();
			calling.push(element);
			calling.pop();
			count++;
		}
	}
	return count;
}

int main()
{
	int n;
	cin >> n;
	queue<int> calling;
	queue<int> ideal;
	for(int i=0;i<n;i++)
	{
	int k;
	cin >> k;
	calling.push(k);
    }
    while(n--)
    {
    	int k;
    	cin >> k;
    	ideal.push(k);
    }
    int count = check(calling,ideal);
    cout << count << endl;
    return 0;
}
