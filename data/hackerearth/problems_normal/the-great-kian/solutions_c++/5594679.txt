#include <iostream>
using namespace std;

int main()
{
    int n,i=0;
    cin>>n;
    long int arr[n];
    while(i!=n)
    {
    cin>>arr[i];
    i++;
    }
   
   for(long int j=0;j<3;j++)
   {
   long int sum=0;
    for(i=j;i<n;i=i+3)
    {
		sum=sum+arr[i]; 	
    }
    cout<<sum<<" ";
   }
    return 0;
}
