#include <iostream>
#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll area(ll x1, ll y1, ll x2, ll y2, ll x3, ll y3)
{
   return abs((x1*(y2-y3) + x2*(y3-y1)+ x3*(y1-y2)));
}
int main()
{
  ll t,x1,x2,x3,x4,y1,y2,y3,y4;
  cin>>t;
  while(t--){
  cin>>x1>>y1>>x2>>y2>>x3>>y3>>x4>>y4;
    ll A=area(x1,y1,x2,y2,x3,y3);
    ll a1=area(x1,y1,x2,y2,x4,y4);
    ll a2=area(x1,y1,x4,y4,x3,y3);
    ll a3=area(x4,y4,x2,y2,x3,y3);
  
    if(a1+a2+a3==A)
     cout<<"INSIDE"<<endl;
     else 
      cout<<"OUTSIDE"<<endl;
        
        
  }  
    
    return 0;
}
