import math
def is_prime(n):
	if n==2:
		return True
	if n%2==0 or n<=1:
		return False
	sqr=int(math.sqrt(n))+1
	for d in range(3,sqr,2):
		if n%d==0:
			return False
	return True
x=input()
m=[]
for i in range(2,x+1):
	if is_prime(i)==False:
		m.append(i)
print ' '.join(str(x) for x in m)
