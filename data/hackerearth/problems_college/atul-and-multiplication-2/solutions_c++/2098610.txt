#include<bits/stdc++.h>
#define ll long long int
#define sc(x) scanf("%lld",&x)
using namespace std;
int main()
{
    ll n;
    sc(n);
    while(n--)
    {
      ll p,q,t;
      sc(p);
      sc(q);
      sc(t);
      if(p*q==t)
      {
          printf("YES\n");
      }
      else{
        printf("NO\n");
      }

    }
return 0;
}
