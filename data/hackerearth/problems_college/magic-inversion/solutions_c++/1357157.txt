#include <iostream>
#include <vector>
#include <string>
using namespace std;
typedef vector<int> vi;
typedef long long LL;
#define forn(i, n) for (int i = 0; i < n; ++i)

const int maxv = 1e6 + 100;
LL t[maxv];
void add(int i, LL v) { for (; i < maxv; i |= i + 1) t[i] += v; }
LL sum(int r) { LL ans = 0; for (; r >= 0; r &= r + 1, --r) ans += t[r]; return ans; }
LL sum(int l, int r) { return sum(r) - sum(l - 1); }

bool solve()
{
    int n; scanf("%d", &n);
    vi a(n); forn(i, n) scanf("%d", &a[i]);
    vi b(n); forn(i, n) scanf("%d", &b[i]);

    LL inv = 0;
    forn(i, n)
    {
        inv += sum(b[i] + 1, maxv - 1);
        add(a[i], 1);
    }

    for (LL d = 2; d * d <= inv; ++d)
        if (inv % d == 0)
            return false;
    return true;
}

int main()
{
    puts(solve() ? "MAGIC INVERSION" : "SIMPLE INVERSION");
    return 0;
}
