#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    while(n--)
    {
    	long int a,b,sum=0;
    	int carry=0;
    	cin>>a>>b;
    	while(a!=0&&b!=0)
    	{
    		int temp1=a%10;
    		int temp2=b%10;
    		if((temp1+temp2+carry)>9)
    		{
    			sum++;
    			carry=(temp1+temp2)-9;
    		}
    		a/=10;
    		b/=10;
    	}
    	if(sum==0)
    		cout<<"No";
    	else
    		cout<<sum;
    	cout<<" carry ";
    	cout<<((sum>1)?"operations":"operation")<<endl;
    }
    return 0;
}
