#include <cstdlib>
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
    int n,temp,i,j;
    scanf("%d",&n);
    vector<int > finance;
    vector<int > market;
   // cout << i << ' ' << j << ' ' << finance.size() << '\n';

   // cout << i << ' ' << j << ' ' << finance.size() << '\n';
    vector<int >::iterator it, it2;
    for(i = 0;i<n;++i){
        scanf("%d",&temp);
        finance.push_back(temp);
    }   
    //cout << i << ' ' << j << ' ' << finance.size() << '\n'; 
    for(i = 0;i<n;++i){
        scanf("%d",&temp);
        market.push_back(temp);
    }   
   // cout << i << ' ' << j << ' ' << finance.size() << '\n'; 
    sort(finance.begin(),finance.end());
    sort(market.begin(),market.end());
   // cout << i << ' ' << j << ' ' << finance.size() << '\n';
    i = 0;j=0;
   // cout << i << ' ' << j << '\n';
    long long int ans = 0;
    while(i < finance.size() && j < market.size()){
         // cout << i << ' ' << j << ' ' << finance.size() << '\n';  
          if(finance[i] > market[j]){
             ans+=100;
             it = finance.begin();
             advance(it,i);
             it2 = market.begin();
             advance(it2,j);
             finance.erase(it);
             market.erase(it2);
           //  cout << "Entered " << ans << ' ' << i << ' ' << j << '\n';
            // ++i;
            // ++j; 
          }
          else if(finance[i] == market[j]){
               ++i;
               ++j;
          }
          else ++i;
    }
    for(i = 0;i<finance.size();++i){
        for(j=0;j<market.size();++j){
           // cout << finance[i] << ' ' << market[j] << '\n';                         
            if(finance[i] == market[j]){
             ans+=50;
             it = finance.begin();
             advance(it,i);
             it2 = market.begin();
             advance(it2,j);
             finance.erase(it);
             market.erase(it2);
            // ++i;
            // ++j;  
            --j;
           // cout << "Entered 2 " << ans << ' ' << i << ' ' << j << '\n';
            }
            else if(market[j] > finance[i]) break; 
        }
    }         
    printf("%lld",ans)    ;                                           
            
    
    
    return 0;
}
