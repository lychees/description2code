#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cmath>
#include <list>
#include <stack>
#include <vector>
#include <queue>
#include <cstring>
#include <algorithm>

using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	long n;
    	scanf("%ld",&n);
    	int a[n],s[n],ma,ms;
    	for(long i=0;i<n;i++)
    	{
    		scanf("%d",&s[i]);
    		if(i==0||s[i]<ms)
    			ms=s[i];
    	}
    	for(long i=0;i<n;i++)
    	{
    		scanf("%d",&a[i]);
    		if(i==0||a[i]<ma)
    			ma=a[i];
    	}
    	if(ms<ma)
    		printf("NO\n");
    	else
    	{
    		bool b=0;
    		sort(a,a+n);
    		sort(s,s+n);
    		for(long i=0;i<n;i++)
    		{
    			if(s[i]<a[i])
    			{
    				b=1;
    				break;
    			}	
    		}
    		if(b)
    			printf("NO\n");
    		else
    			printf("YES\n");
    	}
    }
    return 0;
}
