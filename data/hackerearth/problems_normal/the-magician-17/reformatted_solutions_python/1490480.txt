import math

n = int(raw_input())

for ii in range(n):
    N, K = map(int, raw_input().split())

    if N % 2 == 0 and K <= N/2:
        print K*2-1
    elif N % 2 == 1 and K <= N/2+1:
        print K*2-1
    elif N-K == 0:
        pp = int(math.log(N)/math.log(2) - 1e-5)
        print int(2*(N-math.pow(2, pp)))
    else:
        ff = (3+2*(N-K-1))
        pp = int(math.log((N - 1e-5)/ff)/math.log(2))+1
        print int(2*N-ff*math.pow(2, pp))
