#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	
	signed int a[1000000];
	int n,count;
	while(t--){
		cin>>n;
		
		for(int i=0;i<n;i++){
			cin>>a[i];
		}
		
		count=0;
		for(int i=0;i<n;i++){
			for(int j=i;j<n;j++){
				if(a[i]==a[j]){
					count++;
				}	
			}
		}
		
		cout<<count<<"\n";
	}
    return 0;
}
