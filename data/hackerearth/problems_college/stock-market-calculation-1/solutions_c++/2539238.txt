#include <stdio.h>
 
int main()
{
	long int i=0, t=0, n=0, a[50000], max=0, pi=0, nout=0, out=0, noutcost=0;
	scanf("%ld", &t);
	while(t--)
	{
		scanf("%ld", &n);
		for(i=0; i<n; i++)
			scanf("%ld", &a[i]);
		nout=0;
		out=0;
		max=0;
		pi=0;
		while(max<n)
		{
		noutcost=0;
			pi=max;
			for(i=max+1; i<n; i++)
			{
				if(a[max]<a[i])
				{
					max=i;
				}
			}
			for(i=pi; i<max; i++)
				noutcost+=a[i];
			nout=max-pi;
			out+=((a[max]*nout)-noutcost);
			max++;
			
		}
		printf("%ld\n", out);
	}
    return 0;
}
