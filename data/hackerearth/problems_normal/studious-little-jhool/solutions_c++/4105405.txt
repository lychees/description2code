#include <bits/stdc++.h>
using namespace std;
#define ll long long
ll max(ll a,ll b){
    if(a>b){return a;}
    else{return b;}
}


int main(){
    int t;
    cin>>t;
    for(int j=0;j<t;j++){
        int n;
        cin>>n;
        int arr[n+1+13];
        for(int i=0;i<=n;i++){
            arr[i]=0;
        }
        arr[10]=1;
        arr[12]=1;
        for(int i=10;i<=n;i++){
            if(arr[i]!=0){if(arr[i+10]<arr[i]){arr[i+10]=arr[i]+1;}if(arr[i+12]<arr[i]){arr[i+12]=arr[i]+1;}}
        }
        if(arr[n]==0){cout<<"-1"<<endl;}
        else{cout<<arr[n]<<endl;}

    }
}
