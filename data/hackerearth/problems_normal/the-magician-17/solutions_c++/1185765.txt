ll mod = 1000000007;

inline ll mpow(ll b, ll ex){
	if (b==1)return 1;
    ll r = 1;
    while (ex ){
        if (ex&1)r=(r * b)%mod;
        ex = ex >> 1;
        b = (b * b)%mod;}
    return r;
}

int f(int n,int k){
	assert(k<=n);
	if (k<=(n+1)/2)
	{
		return 2*k-1;
	}else{
		if (n&1)
		{
			int ans=2*f(n/2,k-(n+1)/2)+2;
			return ans>n?2:ans;
		}else{
			return 2*f(n/2,k-n/2);
		}
		
	}
}
