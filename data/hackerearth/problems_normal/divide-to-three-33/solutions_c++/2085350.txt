#include<iostream>
using namespace std;

int arr[100];
int n,ans=10000000;

void splitintosums(int i,int j, int k, int index)
{
	int val;
	if(index==n)
	{
		val=max(i,max(j,k));
		if(val<ans)
			ans=val;
	}
	else
	{
		splitintosums(i+arr[index],j,k,index+1);
		splitintosums(i,j+arr[index],k,index+1);
		splitintosums(i,j,k+arr[index],index+1);
	}
}

int main()
{
	int i;
	cin>>n;
	for(i=0;i<n;i++)
		cin>>arr[i];
		
	splitintosums(0,0,0,0);
	cout<<ans<<endl;
}
