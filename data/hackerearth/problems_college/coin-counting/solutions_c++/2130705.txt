#include <iostream>
using namespace std;

int main()
{
	
	int t  ; 
	
	cin >> t ;
	
	while(t--){
		
		
		int number ; 
		
		cin >> number  ;
		
		int arr[number+1];
		
		
		for(int i=0 ;  i <=number;i++)
		arr[i]=0;
		arr[0]=1;
		
		int coin[3];
		coin[0]=1;
		coin[1]=3;
		coin[2]=5;
		
		for(int i =0 ;  i<3 ; i++ )
			for(int j=coin[i] ; j <=number ;j++)
				arr[j]+=arr[j-coin[i]];
		
		cout << arr[number] << endl;
		
	}
	
	
    
    return 0;
}
