t = int(raw_input())
while t > 0:
    t -= 1
    l = sorted(map(int, raw_input().split()[1:]))
    mx = l[0] + l[len(l) - 1]
    mn = l[0] + l[len(l) - 1]
    for i in range(len(l) // 2):
        team = l[i] + l[len(l) - i - 1]
        if team > mx:
            mx = team
        if team < mn:
            mn = team
    print(mx - mn)
