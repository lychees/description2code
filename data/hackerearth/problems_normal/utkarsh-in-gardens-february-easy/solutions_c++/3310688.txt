#include <bits/stdc++.h>
using namespace std;
const int N = 2007;
vector<int> v[N];
int c[N][N];
int n,m;
int main()
{
	cin>>n;
	for(int i=0;i<n;i++){
		for(int j=0;j<n;j++){
			scanf("%d",&m);
			if(m == 1) v[i].push_back(j);
		}
	}
	long long ans = 0;
	for(int i=0;i<n;i++){
		for(auto x:v[i]){
			for(auto y:v[i]){
				c[x][y]++;
			}
		}
	}
	for(int i=0;i<n;i++){
		for(int j=i+1;j<n;j++){
			ans += c[i][j]*1ll*(c[i][j]-1)/2;
		}
	}
	cout<<ans/2<"\n";
    return 0;
}
