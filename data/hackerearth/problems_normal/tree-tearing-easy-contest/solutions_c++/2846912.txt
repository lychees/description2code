#include<bits/stdc++.h>
using namespace std;


typedef pair<int,int>pint;
typedef vector<int>vint;
#define pb push_back
#define mp make_pair
#define rep(i,n) for(int i=0;i<(n);i++)
template<class T,class U>void chmin(T &t,U f){if(t>f)t=f;}
template<class T,class U>void chmax(T &t,U f){if(t<f)t=f;}

int K,N;
vint G[10000];
int ans;
int dfs(int v,int p){
    int sum=0;
    rep(i,G[v].size()){
        int to=G[v][i];
        if(to==p)continue;
        sum+=dfs(to,v);
    }
    if(sum+1>=K){
        ans++;
        return 0;
    }
    return sum+1;
}

int main(){
    cin>>K>>N;

    rep(i,N-1){
        int p;cin>>p;
        p--;
        G[p].pb(i+1);
        G[i+1].pb(p);
    }
    dfs(0,-1);

    cout<<ans<<endl;

    return 0;
}
