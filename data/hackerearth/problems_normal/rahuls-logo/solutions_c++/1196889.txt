#include<bits/stdc++.h>
#define ll long long
#define N 1000000007
using namespace std;
void printeve(int n)
{
	int p=1,i,j,cnt;
	for(i=0;i<n;i++)
	{
		for(j=0;j<n-i-1;j++)
		{
			cout<<" ";
		}
		cnt=0;
		for(;cnt<p;j++)
		{
			if(cnt!=p-1)
			{
			cout<<"/ ";
			j++;
			}
			else
			cout<<"/";
			cnt++;
		}
		cnt=0;
		while(j<n)
		{
			cout<<" ";
			j++;
			cnt++;
		}
		for(j=0;j<cnt;j++)
		{
			cout<<" ";
		}
		for(j=0;j<p;j++)
		{
			if(j!=p-1)
			cout<<"\\ ";
			else
			{
				cout<<"\\";
			}
		}
		cout<<endl;
		if(i%2==1)
		{
			p++;
		}
	}
}
void printodd(int n)
{
	int i,j;
	int p=n/2+n%2,cnt;
	for(i=0;i<n;i++)
	{
		cout<<"  ";
		for(j=0;j<i;j++)
		{
			cout<<" ";
		}
		cnt=0;
		for(;cnt<p;j++)
		{
			if(cnt!=p-1)
			{
			cout<<"\\ ";
			j++;
			}
			else cout<<"\\";
			cnt++;
		}
		cnt=0;
		while(j<n)
		{
			cout<<" ";
			j++;
			cnt++;
		}
		for(j=0;j<cnt;j++)
		{
			cout<<" ";
		}
		for(j=0;j<p;j++)
		{
			if(j!=p-1)
			cout<<"/ ";
			else
			cout<<"/";
		}
		cout<<endl;
		if(n%2==1 && i%2==0)
		{
			p--;
		}
		else if(n%2==0 && i%2==1)
		{
			p--;
		}
		
	}
}
int main()
{
	int n;
	cin>>n;
	printeve(n);
	printodd(n);
	printeve(n+1);
	printodd(n+1);
	return 0;
}
