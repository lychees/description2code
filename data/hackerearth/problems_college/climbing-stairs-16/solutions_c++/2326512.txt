#include<iostream>
using namespace std;
long long int ways(int n);
 
int main()
{
int t;
cin >> t;
for(int i=0;i<t;i++)
{
    int n;
    cin >> n;
    cout << ways(n) << endl;
}
}
 long long int ways(int n)
{
    if(n==1)
    return 1;
    if(n==2)
    return 2;
    else
    {
    	
    	long long int a=2, b=1,c;
    	for(int i=3;i<=n;i++)
    	{
    		c = a+b;
    		
    		b=a;
    		a=c;
    	}
    	return c;
    }
        
 
 
}
