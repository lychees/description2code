#include <bits/stdc++.h>

using namespace std;
vector <int> v[1009];
int a[1009];
int cnt[1009];
int main()
{
	int i,j,k,l,m,n,A,B,c,x,y;
	scanf("%d%d",&A,&B);
	for(j=0;j<B;j++)
	{
		scanf("%d%d",&x,&y);
		v[x].push_back(y);
		v[y].push_back(x);
	}
	queue <int> q;
	q.push(0);
	for(i=0;i<A;i++) cnt[i]=INT_MAX;
	cnt[0]=0;c=0;
	vector <int> ::iterator itr;
	while(!q.empty())
	{
		int u=q.front();
		q.pop();

		for(itr=v[u].begin();itr!=v[u].end();itr++)
		{
			if(cnt[*itr]>cnt[u])
			{
				cnt[*itr]=cnt[u]+1;
				q.push(*itr);
				 //printf("%d\n",*itr);
			}
		}
	}
	for(i=1;i<A;i++)
	{
		if(cnt[i]!=INT_MAX) printf("%d\n",cnt[i]);
		else printf("-1\n");
	}
	return 0;
}
