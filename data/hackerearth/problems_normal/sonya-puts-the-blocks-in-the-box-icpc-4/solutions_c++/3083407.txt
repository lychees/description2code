lli cum[N],dp2[105][1005],dp1[105][1005];
int len[105];
vi v[105];
int main()
{
	int n,m,x;
	sc(n,m);
	lli ans=0;
	int cum_len=0,tot_len=0;
	Fore(i,1,n){
		sc(len[i]);
		tot_len+=len[i];
		v[i].pb(len[i]);
		cum[0]=0;
		Fore(j,1,len[i]){
			sc(x); 
			v[i].pb(x);
			cum[j]=cum[j-1]+x;
		}
		Fore(k,1,min(len[i],1000)){
			for(int j=k;j<=len[i];j++){
				dp1[i][k]=max(dp1[i][k],cum[j]-cum[j-k]);
				//prl(i,k,dp1[i][k]);
			}
		}
		//prl();
		Fore(j,1,1000){
			if(j>tot_len) break;
			dp2[i][j]=dp2[i-1][j];
			Fore(k,0,j){
				if(k>cum_len) break;
				dp2[i][j]=max(dp2[i][j],dp2[i-1][k]+dp1[i][j-k]);
				//if(j<=10) prl(i,j,k,dp2[i-1][k],dp1[i][j-k],dp2[i][j]);	
			}			
			if(j<=m) ans=max(ans,dp2[i][j]);
		}
		cum_len=max(cum_len,len[i]);

	}
	prl(ans);
}
