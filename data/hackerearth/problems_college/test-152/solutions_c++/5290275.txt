#include<bits/stdc++.h>
using namespace std;


int main(){
    int t,n,a;
    string s;
    cin >> t;
    while(t--){
    	int M=-1;
    	cin >> n >>s;
    	sort(s.begin(),s.end());
    	for(int i=0;i<s.size();i++){
    		a=upper_bound(s.begin(),s.end(),s[i]) - lower_bound(s.begin(),s.end(),s[i]);
    		M=max(M,a);
    	}
    	
    	if(M%2) cout << "No" << endl;
    	else cout << "Yes" << endl;
    	
    }
    return 0;
}
