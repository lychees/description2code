#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n,t;
    scanf("%d",&t);
    while(t--){
        scanf("%d",&n);
        long long count=0;
        count+=2*(n/14);
        n%=14;
        //printf("%lld\n",count);
        count+=(n/12)*2;
        //printf("%lld\n",count);
        n%=12;
        count+=2*(n/10);
        n%=10;
        count+=2*(n/9);
        n%=9;
        //printf("%lld\n",count);1
        count+=2*(n/8);
        //printf("%lld\n",count);
        n%=8;
        count+=n/7;
        //printf("%lld\n",count);
        n%=7;
        count+=n/5;
        //printf("%lld\n",count);
        n%=5;
        count+=n/3;
        //printf("%lld\n",count);
        n%=3;
        count+=n/2;
        n%=2;
        if(n==1)count++;
        printf("%lld\n",count);
    }
    return 0;
}
