#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long int i,p,x,y,res,a[26];
    int t,n;
    cin>>n;
    for(t=0;t<n;t++)
    {
        cin>>p;
        for(i=0;i<26;i++)
            cin>>a[i];
        res=0;
        p=p*12;
        x=p/26;
        y=p%26;
        if(x==0){
            for(i=0;i<p;i++)
                res+=a[i];
        }
        else{
            for(i=0;i<y;i++)
            {
                a[i]=(x+1)*a[i]+ x*(x+1)/2;
                res+=a[i];
            }
            for(i=y;i<26;i++)
            {
                a[i]=x*a[i]+ x*(x-1)/2;
                res+=a[i];
            }
        }
        cout<<res<<endl;
    }
return 0;
}
