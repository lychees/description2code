{

    		if(s[i]=='R' && s[i+1]=='Y' && s[i+2]=='Y'){
    			i += 2;
    		}else if(s[i]=='R' && s[i+1]=='Y'){
    			i += 1;
    		}else if(s[i]=='R'){}
    		else{
    			cout<<"NO"<<endl;
    			break;
    		}
    	}

    		cout<<"YES"<<endl;
    	}
    }
    return 0;
}
