#include <bits/stdc++.h>
using namespace std;

int main() {
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		int m=-1;
		int c=0;
		int t1,t2;
		vector<int> st;
		vector<int> en;
		for(int i=0;i<n;i++){
			cin>>t1>>t2;
			st.push_back(t1);
			en.push_back(t2);
		}
		sort(st.begin(),st.end());
		sort(en.begin(),en.end());
		int i=0,j=0;
		while(j!=en.size()){
			if(i<st.size()&&st[i]<en[j]){
				c++;
				i++;
				if(c>m){
					m=c;
				}
			}
			else{
				c--;
				j++;
			}
		}
		cout<<m<<endl;
	}
	return 0;
}
