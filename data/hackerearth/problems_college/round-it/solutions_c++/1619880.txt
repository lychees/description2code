#include <iostream>
#include <cmath>
#define ll long long

using namespace std;

int main()
{
    ll n = 0, a = 0, b = 0, ans = 0;
    cin >> n;
    while (n--){
    	cin >> a >> b;
    	if (b > a)
    		cout << b << endl;
    		//;
    	else{
    		//cout << (a / b)*b << endl;
    		//cout << (a+b) / b)*b << endl;
    		if ( abs((a/b)*b - a) == abs(((a+b) / b)*b - a) )
    			cout << ((a+b) / b)*b << endl;
    		else if ( abs((a/b)*b - a) < abs(((a+b) / b)*b - a) )
    			cout << (a/b)*b << endl;
    		else if ( abs((a/b)*b - a) > abs(((a+b) / b)*b - a) )
    			cout << ((a+b) / b)*b << endl;
    	}
    }
    return 0;
}
