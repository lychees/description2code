#include <bits/stdc++.h>
using namespace std;

int main()
{
	string s;
	getline(cin,s);
	int len=s.length(),loop;
	long long int a,b;
	scanf("%d",&loop);
	while(loop--)
	{
		scanf("%lld %lld",&a,&b);
		if( s.at((a-1)%((long long int)len)) == s.at((b-1)%((long long int)len)) )
			printf("Yes\n");
		else printf("No\n");
	}

	return 0;
}
