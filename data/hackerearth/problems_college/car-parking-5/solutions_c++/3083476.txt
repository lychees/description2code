#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int t,c,i;
    char a[200];
    cin>>t;
    while(t--)
    {
      cin>>a;
      c=0;
      for(i=0;i<strlen(a);i++)
      {
          if(a[i]!='D'&&a[i]!='B'&&a[i]!='S')
          {
              if(a[i+1]!='B'&&a[i+2]!='B'&&a[i+1]!='S'&&a[i-1]!='S')
              c++;
          }
      }
      cout<<"\n"<<c;
    }
    return 0;
}
