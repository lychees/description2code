#include<stdio.h>
#include<iostream>
#include<queue>
#include<map>
using namespace std;
int main()
{
    int i,j,k,l,ans,rem,a,b,c,t;
    cin>>t;
    int m[100010];
    while(t--)
    {
    cin>>a>>b>>c;
    ans=a/b;
    rem=a%b;
    j=0;
    for(i=0;i<c;i++)
    {
        rem=rem*10;
        m[j]=rem/b;
        rem=rem%b;
        j++;
    }
    cout<<ans;
    if(c>0)
    {
        cout<<".";
        for(i=0;i<j;i++)
            cout<<m[i];
    }
    cout<<"\n";
    }
    //cout<<"\n";
}
