#include <bits/stdc++.h>

using namespace std;

void printA(int a[][100],int n)
{
	for (int i = 0; i < n; ++i)
	{
		for(int j = 0;j < n;j++)
			printf("%d ", a[i][j]);
		printf("\n");
	}
}

void printB(int a[][100],int n)
{
	for(int j = 0;j < n;j++)
	{
		for(int i = n-1;i >= 0;i--)
			printf("%d ",a[i][j]);
		printf("\n");
	}
}

void printC(int a[][100],int n)
{
	for(int i = n-1;i >= 0;i--)
	{
		for(int j = n-1;j >= 0;j--)
			printf("%d ",a[i][j]);
		printf("\n");
	}
}

void printD(int a[][100],int n)
{
	for(int j = n-1;j >= 0;j--)
	{
		for(int i = 0;i < n;i++)
			printf("%d ",a[i][j]);
		printf("\n");
	}
}

int main()
{
	int n,m;
	scanf("%d %d",&n,&m);
	int a[100][100];
	int i = 0;
	int j = 0;
	while(i < n)
	{
		j = 0;
		while(j < n)
		{
			scanf("%d",&a[i][j]);
			j++;
		}
		i++;
	}
	i = 0;j = 0;
	int q = 0;
	while(m--)
	{
		scanf("%d",&q);
		q%=360;
		if(q == 0)
			printA(a,n);
		else if(q == 90)
			printB(a,n);
		else if(q == 180)
			printC(a,n);
		else if(q == 270)
			printD(a,n);
		printf("\n");
	}
	return 0;
}
