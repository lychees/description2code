#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long a,b=0,c,d,ar[1000],br[1000],i,j;
    scanf("%lld",&a);
    for(i=0;i<a;i++){
    	scanf("%lld",&ar[i]);
    }
    for(i=0;i<a;i++){
    	scanf("%lld",&br[i]);
    
    }
    sort(ar,ar+a);
    sort(br,br+a);
    for(i=0;i<a;i++){
    	if(ar[i]>br[i]){
    		b+=100;
    	}
    	if(ar[i]==br[i]){
    		b+=50;
    	}
    }
    long long max=b;
    i=a-1;
    b=0;
    for(j=a-1;j>=0;j--){
    	if(ar[i]>br[j]){
    		b+=100;
    		i--;
    	}
    }
    if(b>max){
    	max=b;
    }
    i=a-1;b=0;
    for(j=a-1;j>=0;j--){
    	if(ar[i]>br[j]){
    		b+=100;
    		i--;
    	}else if(ar[i]==br[j]){
    		b+=50;
    		i--;
    	}
    }
    if(b>max){
    	max=b;
    }
    cout<<max<<endl;
    return 0;
}
