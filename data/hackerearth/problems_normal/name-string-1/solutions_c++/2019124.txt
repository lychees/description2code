#include <iostream>
using namespace std;

int main()
{
	int t,i,pos;
	long a[3],n1,n2,max;
	cin>>t;
	while(t--)
	{
		n1=n2=max=0;
		cin>>a[0]>>a[1]>>a[2];
	
	for(i=0;i<3;i++)
	{
		if(a[i]>max)
		{
		max=a[i];
		pos=i;
	}
	}
	n1=max;
	a[pos]=0;
	max=0;
	for(i=0;i<3;i++)
	{
		if(a[i]>max)
		max=a[i];
	}
	n2=max;
//	cout<<n1<<" "<<n2<<endl;
if(n1==n2)
	cout<<n1*2<<endl;
	else if(n2==0)
	cout<<1<<endl;
	else if(n1>=n2+1)
	cout<<2*n2+1<<endl;
	
	else
	cout<<2*n2+1;
	
	}
}
