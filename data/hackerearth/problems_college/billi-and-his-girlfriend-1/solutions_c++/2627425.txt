#include<bits/stdc++.h>
#include <iostream>
using namespace std;
#define ll long long
ll power(ll exp,ll base,ll mod)
{
	ll res=1;
	while(exp>0)
	{
		if(exp&1)
		{
			res=(res*base)%mod;
		}
		base=(base*base)%mod;
		exp=exp/2;
	}
	return res;
}
int main()
{
   ll a,b,n,k,j,i,l,ct,t;
   scanf("%lld",&t);
   while(t--)
   {
   	ll sum=0;
   	scanf("%lld %lld %lld",&a,&b,&k);
   	ll res=power(b,a,pow(10,k));
   	while(res)
   	{
   		sum=sum+res%10;
   		res=res/10;
   	}
   	cout<< sum<<endl;
   }
    return 0;
}
