#include<bits/stdc++.h>
using namespace std;
#define MAX 550
#define lli long long int
string ss[MAX];
lli len[MAX];
char f(lli i,lli n){
	if(n==0)return ss[n][i];
	if(i>=2*len[n-1])return ss[n][i-2*len[n-1]];
	if(i<len[n-1])return f(i,n-1);
	return f(2*len[n-1]-i-1,n-1);
}
int main(){
	//std::ios::sync_with_stdio(false);
	ios::sync_with_stdio(0);
	int t;
	cin>>t;
	while(t--){
		lli n,m,x;
		cin>>n>>m;
		for(lli i=0;i<n;i++)cin>>ss[i];
		len[0]=ss[0].size();
		for(lli i=1;i<n;i++){
			len[i]=2*len[i-1]+ss[i].size();
		}
		for(lli i=0;i<m;i++){
			cin>>x;
			cout<<f(x,n-1);
		}
		cout<<"\n";
	}
	
	
	return 0;
}
