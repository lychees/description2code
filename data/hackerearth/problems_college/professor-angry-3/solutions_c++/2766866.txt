#include <bits/stdc++.h>
using namespace std;

int main()
{   int t;
    int n,k,i,j=0;
    
    
    cin>>t;
    
    while(t--)
    {
    	j=0;
    	cin>>n>>k;
    	int a;
    	for(i=0;i<n;i++)
    	{
    		cin>>a;
    		
    		if(a<=0)
    		j++;
    	}
    	
    	
    	if(j>=k)
    	cout<<"NO"<<endl;
    	else
    	cout<<"YES\n";
    }
    
}
