//dynamicprogramming
//once upon a time in landland
#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int t,n,i,k;
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		long long int arr[n],dp[n];
		for(i=0;i<n;i++)
		dp[i]=0;
		for(i=0;i<n;i++)
		cin>>arr[i];
		dp[0]=arr[0];
		for(i=1;i<n;i++)
		{
			if(i-k-1>=0)
			{
				dp[i]=max(dp[i-1],max(dp[i-k-1]+arr[i],arr[i]));
			}
			else
			{
				dp[i]=max(dp[i-1],arr[i]);
			}
		}
		if(dp[n-1]<0)
		dp[n-1]=0;
		cout<<dp[n-1]<<endl;
	}
}
