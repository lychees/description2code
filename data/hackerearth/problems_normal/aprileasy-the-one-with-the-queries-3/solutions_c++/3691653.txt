#include <iostream>
#include <string>
#include <vector>
#include <list>
using namespace std;

int main() {
	string st,part;
	cin>>st;

	int q,len,i,index,currindex;
	bool fg;
	vector< list<int> > foo(26,list<int>());
	len=st.length();
	for(i=0;i<len;i++)
		foo[st[i]-'a'].push_back(i);
	
	vector< list<int>::iterator > ls(26);
	cin>>q;
	while(q--) {
		cin>>part;
		for(i=0;i<26;i++)
			ls[i]=foo[i].begin();
		len=part.length();
		fg=0;
		currindex=-1;
		
		for(i=0;i<len;i++) {
			index=part[i]-'a';
			
			if(ls[index]==foo[index].end()) {
				fg=1;
				break;
			}
			
			while(ls[index]!=foo[index].end() && *ls[index]<=currindex)
				ls[index]++;
			if(ls[index]==foo[index].end()) {
				fg=1;
				break;
			}
		
			currindex=*ls[index];
			ls[index]++;
		}

		if(fg)
			cout<<"No\n";
		else
			cout<<"Yes\n";
	}

	return 0;
}
