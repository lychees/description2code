n = input()
d = []
for i in xrange(n):
    a = raw_input()
    a = a.split()
    a[0] = long(a[0])
    if a[0] > 1:

        b = int(a[1])
        for j in xrange(1, a[0]):
            c = int(a[j+1])
            b = b ^ c

        if b % 2 == 0:
            d.append("Yes")
        else:
            d.append("No")
    else:
        d.append("No")

for i in d:
    print i
