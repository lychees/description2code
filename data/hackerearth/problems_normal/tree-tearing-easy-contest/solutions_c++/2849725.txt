#include <bits/stdc++.h>
using namespace std;

vector<vector<int>> ch;
int k, n, v = 0;

int search(int cur)
{
	int cht = 1;
	for (int chi : ch[cur])
		cht += search(chi);
	if (cht >= k)
		v++, cht = 0;
	return cht;
}

int main()
{
	scanf("%d %d", &k, &n);
	ch.resize(n + 1);
	for (int i = 2; i <= n; i++) {
		int cur;
		scanf("%d", &cur);
		ch[cur].push_back(i);
	}
	search(1);
	printf("%d\n", v);
}
