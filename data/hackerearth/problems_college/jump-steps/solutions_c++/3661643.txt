#include <bits/stdc++.h>
using namespace std;

long long int po(long long base,long long e)
{
	long long ans = 1;
	while(e>0)
	{
		if (e&1)
		  ans = ans * base;
		  
          e = e>>1;
          base = base *base;
	}
	return ans;
}

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long int n;
		scanf("%lld",&n);
		long long int cnt=0;
		long long int temp, diff;
		diff = n;
		while(diff>1){
			temp = (long long int)(log(diff)/log(2));
			diff -= po(2, temp);
			cnt++;
		}
		if(diff)
		 cnt++;
		printf("%lld\n",cnt);
	}
    return 0;
}
