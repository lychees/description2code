#include<bits/stdc++.h>
using namespace std;
bool comp(pair<long long int,pair<long long int,long long int> > a,pair<long long int,pair<long long int,long long int> >  b)
{
	if(a.second.second!=b.second.second)
	return a.second.second>b.second.second;
	else
	return a.second.first>b.second.first;
}
int main()
{
	ios::sync_with_stdio(false);
	
	vector<pair< long long int,pair< long long int,long long int > > > vec(1000099);
	long long int n,l,i,m,z,p,c,s,id,nee;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>id>>z>>p>>l>>c>>s;
		nee=p*50+l*5+c*10+s*20;
		vec[i].first=nee;
		nee=nee-z;
		vec[i].second.second=nee;
		vec[i].second.first=id;
	}
	stable_sort(vec.begin(),vec.end(),comp);
	for(i=0;i<n&&i<=4;i++)
	{
	cout<<vec[i].second.first<<" "<<vec[i].first<<endl;
	}	
	
	return 0;
}
