#include<iostream>
#include<math.h>
using namespace std;





int main()
{
   int T,a,b,a1,b1,a2,b2,c=0,d1,d2,d;
   scanf("%d",&T);
   while(T--)
   {d1=d2=d=0;
       scanf("%d %d",&a,&b);
       scanf("%d %d",&a1,&b1);
       scanf("%d %d",&a2,&b2);
       scanf("%d",&c);
       if(b1!=b2)
        {d1=abs(b2-b1);
        d2=b-d1;
        if(d1<d2)
            d=d1;
        else d=d2;
        }
       if(a1!=a2)
        {d1=abs(a1-a2);
        d2=a-d1;
        if(d1<d2)
            d=d+d1;
        else
            d=d+d2;
        }
        c=d*c;
       printf("%d\n",c);
       if(c<=1000)
        printf("You saved the group.\n");
       else
        printf("Let go of the group.\n");
       c=0;
   }
   return 0;
   }
