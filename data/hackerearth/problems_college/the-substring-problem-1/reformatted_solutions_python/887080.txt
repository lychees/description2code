cases = int(raw_input())
for i in xrange(cases):
    words_lst = [0]*26
    n = int(raw_input())
    high = 0
    for j in xrange(n):
        word = set(raw_input())
        for k in word:
            ind = ord(k)-97
            words_lst[ind] += 1
            if words_lst[ind] > high:
                high = words_lst[ind]
    print chr(words_lst.index(high)+97)
