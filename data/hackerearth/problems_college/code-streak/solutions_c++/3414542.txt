#include <iostream>
#include <limits>
#include <string.h>
using namespace std;
int ar[1000005];
#define ll long long int
int main()
{
    ll t,n,s,r;
    cin>>t;
    while(t--){
    	cin>>n;
    	ll ans=0,bef,pans=0;
    	memset(ar,0,100005);
    	for(int i=0;i<n;i++){
    		cin>>s>>r;
    		if(ar[s]==0 && r==1){
    			pans++;
    			bef=1;
    			ar[s]=1;
    		}
    		else if(r==0){
    			if(pans>ans)
    			ans=pans;
    		//	cout<<pans<<" ";
    			pans=0;
    		}
    	}
    	if(pans>ans)
    	ans=pans;
    	cout<<ans<<endl;
    }
    return 0;
}
