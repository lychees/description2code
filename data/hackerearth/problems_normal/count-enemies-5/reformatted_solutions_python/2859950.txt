test = int(raw_input())
while (test):
    test -= 1
    s = raw_input()
    xc = oc = poc = 0
    length = len(s)
    i = 0
    while (i < length):
        if (s[i] != '*'):
            if (s[i] == 'X'):
                xc += 1
            if (s[i] == 'O'):
                oc += 1
        if (s[i] == '*'):
            if (xc <= 0):
                poc += oc
            xc = oc = 0
        i += 1
    if (xc == 0):
        poc += oc
    print poc
