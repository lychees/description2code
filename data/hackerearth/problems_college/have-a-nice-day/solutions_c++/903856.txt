#include <iostream>
#include<algorithm>
using namespace std;

int main()
{   
	int t,n,i,m[1001],f[1001];long r;
	cin>>t;
	while(t--)
	 {  r=0;
	 	cin>>n; for(i=0;i<n;i++)cin>>m[i]; for(i=0;i<n;i++)cin>>f[i];
	 	sort(m,m+n); sort(f,f+n); for(i=0;i<n;i++)r+=(m[i]*f[i]); 
		cout <<r<< endl;
	}
    
    return 0;
}
