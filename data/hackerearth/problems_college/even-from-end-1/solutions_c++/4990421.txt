#include <iostream>
#include <stack>
#define c cin
#define o cout
using namespace std;

int main(){
	int t;
	
	while(c>>t){
		stack <int> even;
		int num;
		while(t--){
			c>>num;
			if(num%2==0)
				even.push(num);
		}
		
		if(even.empty()){
			cout << "None to display" << endl;
			continue;
		}
		while(!even.empty()){
			cout << even.top() << " ";
			even.pop();
		}
		cout << endl;
	}
	return 0;
}
