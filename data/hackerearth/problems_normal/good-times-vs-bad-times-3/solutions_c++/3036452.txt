#include <iostream>
#include <vector>
#include <cmath>
using namespace std;

int __gcd(int a, int b){
if(!b )return a;
return __gcd(b,a%b);
}
int main()
{
    int tot=24*60*60;
    vector<int> good(tot);
    for(int i=0;i<tot;i++){
    int s=i%60;
    int mn=i/60;
    int m=mn%60;
    int h=mn/60;
    int g=__gcd(h,m);
    g=__gcd(g,s);
    if(g==1)good[i]=1;
    }
    int su=0;
    for(int i=0;i<tot; i++)good[i]=su=su+good[i];
    
    int T; cin>>T;
    for(int t=0; t<T; t++){
    int h,m,s;
    cin>>h>>m>>s;
    int j=h*60*60+m*60+s;
    int gs=good.back();
    if(j>0)gs-=good[j-1];
    int bs=tot-gs-j;
    if(bs &&gs){
    	int g=__gcd(bs,gs);
    	bs/=g; gs/=g;
    	
    }
    cout<<bs<<':'<<gs<<'\n';
    }
    return 0;
}
