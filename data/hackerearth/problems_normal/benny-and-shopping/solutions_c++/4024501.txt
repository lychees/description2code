#include <bits/stdc++.h>

using namespace std;
#define fi      first
#define sec     second
#define lb      lower_bound
#define ub      upper_bound
#define pb      push_back
#define pf      push_front
#define mp      make_pair
#define gcd     __gcd
#define all(x)  x.begin(), x.end()
#define S(x)    scanf("%d", &x)
#define Sl(x)   scanf("%lld", &x)
#define P(x)    printf("%d", x)
#define Pl(x)   printf("%lld", x)
#define PS(x)   printf("%s", x)
#define MOD     1000000007LL

typedef long long       ll;
typedef vector<int>     vi;
typedef vector<vi>      vvi;
typedef pair<int, int>  ii;
typedef vector<ii>      vii;

int main()
{
	int N;
	scanf("%d ", &N);
	string x, ans;
	while(N--) {
		ans = "";
		getline(cin, x);
		int i = 0;
		while(i < x.size()) {
			if(x[i] == '$') {
				break;
			}
			i++;
		}
		ans = ans + '$';
		i++;
		while(i < x.size()) {
			if(x[i] == ' ') {
				i++;
				continue;
			}
			else if(!(x[i] >= '0' && x[i] <= '9')) {
				break;
			} else {
				ans += (char)(x[i]);
			}
			i++;
		}
		i = 1;
		while(i < ans.size() && ans[i] == '0') {
			i++;
		}
		ans = ans.substr(i);
		if(ans == "") ans = "0";
		ans = '$' + ans;
		cout << ans << endl;
	}
	return 0;
}
