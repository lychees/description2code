#include<bits/stdc++.h>
using namespace std;

#define mod 1000000007
#define W while
#define LL long long int
#define pb push_back
string fun(string str,LL n)
{
	LL start,end;
	start=0;
	end=n-1;
	
	while(start<=end)
	{
		char ch;
		ch=str[start];
		str[start]=str[end];
		str[end]=ch;
		start++;
		end--;
	}
	return str;
}

int main()
{
	LL  t,n,i;
	cin>>t;
	W(t--)
	{
		string str,str1;
		cin>>str;
		str1=str;
		n=str.length();
		str=fun(str,n);
		
		if(str1==str)
			cout<<"Palindrome"<<endl;
		else
		{
			LL p;
			p=1;
			for(i=0;i<n;i++)
				p=p*(str[i]-'a'+1);
			
			cout<<p<<endl;
			
		}
	}
	return 0;
}
