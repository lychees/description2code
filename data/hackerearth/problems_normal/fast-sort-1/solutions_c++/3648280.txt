#include <bits/stdc++.h>

using namespace std;

int repeat [26 + 1];

int main()
{

    int t;
    string p, f;

    cin >> t;

    while(t--){

        cin >> p >> f;

        memset(repeat, 0, sizeof repeat);                                   // reset array in each test case

        for(int i = 0; i < f.size(); i++)
        	repeat[ f[i] - 'a' ]++;   // number of repetition for each character

        for(int i = 0; i < 26; i++){                                        // loop over characters in the given order

            int num = repeat[ p[i] - 'a' ];                               // print the (ith) character the number of times it is repeated

            for(int j = 0; j < num; j++) cout << (char) (p[i]);
        }

        cout << "\n";
    }

    return 0;
}
