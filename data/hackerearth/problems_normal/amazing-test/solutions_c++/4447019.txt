#include <iostream>
#include <cstring>
using namespace std;

int main()
{
	int T;
	cin >>T;
	while(T){
		int N, K;
		cin >> N >> K;
		int A[N+1];
		int total = 0;
		for(int i = 1; i <= N; i++){
			cin >> A[i];
			total = total+A[i];
		}
		if(total <= K){
			cout << "YES" << endl;
			goto next;
		}else{
		int table[N+1][total+1];
		memset(table, 0, sizeof(table[0][0]) * (N+1) * (total+1));

		for(int i = 1; i <= N; i++){
			table[i][A[i]] = 1;
		}
		for(int i = 2; i <= N; i++){
			for(int j = 1; j <= total; j++){
				if(j >= A[i]){
					table[i][j] = table[i-1][j] || table[i-1][j-A[i]];
				}else{
					table[i][j] = table[i-1][j];
				}
				if(table[i][j]){
					if( j <= K && (total - j) <= K){
						cout << "YES" << endl;
						goto next;
					}
				}
			}
		}
		cout << "NO" << endl;
		}
	next:	T--;
	}
    return 0;
}
