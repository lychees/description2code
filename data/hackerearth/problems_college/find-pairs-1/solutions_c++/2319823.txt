#include <iostream>
#include <algorithm>
using namespace std;
 
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long n;
    	cin>>n;
    	long long a[n],i;
    	for(i=0;i<n;i++)
    		cin>>a[i];
    	sort(a,a+n);
    	int m=a[0];
    	long long ans=0;
    	for(i=1;i<n;i++)
    	{
    		long long j=i;
    		while(a[i]==m)
    			i++;
    		ans = ans + (i-j)*(i-j+1)/2;
    		m = a[i];
    	}
    	cout<<ans+n<<endl;
    }
    return 0;
}
