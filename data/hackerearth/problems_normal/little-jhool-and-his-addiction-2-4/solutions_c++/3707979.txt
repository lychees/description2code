#include<bits/stdc++.h>
using namespace std;
int a[1000001];
int main()
{
    int n,t,k;
    scanf("%d",&t);
    while(t--){
        scanf("%d%d",&n,&k);
        for(int i=0;i<n;i++){
            scanf("%d",&a[i]);
        }
        sort(a,a+n);
        int mi = INT_MAX,mx = INT_MIN,sum;
        for(int i=0,j=n-1;i<n/2||j>=n/2;i++,j--){
            if(i!=j){
                sum = a[i]+a[j];
            }
            else{
                sum = a[i];
            }
            if(sum>mx){
                mx = sum;
            }
            if(sum<mi){
                mi = sum;
            }
        }
        int dif = mx-mi;
        cout<<dif<<endl;
        if(dif>k){

            cout<<"No more girlfriends!\n";
        }
        else if(dif==k){
            cout<<"Lucky chap!\n";
        }
        else if(dif<k){
            cout<<"Chick magnet Jhool!\n";
        }
    }
}
