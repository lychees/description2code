#include <iostream>
#include<stdbool.h>
#define max 100111
using namespace std;

bool isprime[max];
int prime[max];
int primefun()
{
int i,j,ind=0;
for(i=0;i<max;i++)
  isprime[i]=true;
 
isprime[1]=false;

for(i=2;i<max;i++)
{
	if(isprime[i])
	{
      prime[ind++]=i;
	  for(j=i;j<=max/i;j++)
	     isprime[i*j]=false;
	}
 }
}
int count(int a)
{
	int cnt=0,cur=0;
	while(a!=1)
	{
	  if(isprime[a])
	 {
		cnt++;
		break;
	 }
	 while(a%prime[cur]==0)
	 {
		a/=prime[cur];
		cnt++;
	 }
	  cur++;
	}
	return cnt;
}
int main()
{
    int i,n,a[max];
    cin>>n;
    for(i=0;i<n;i++)
      cin>>a[i];
      
    primefun();
    
    int tsum=count(a[0]);
    
    for(i=1;i<n;i++)
     tsum=tsum ^ count(a[i]);
     
    if(tsum==0)
        cout<<"BOB" ;
        else
          cout<<"ALICE"; 
    return 0;
}
