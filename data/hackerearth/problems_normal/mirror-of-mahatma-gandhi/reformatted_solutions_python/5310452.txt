'''

print 'Hi, %s.' % name
'''
n = int(raw_input())
mirrors = {'0', '1', '8'}

for i in range(n):
    inp = raw_input()
    is_mirror = True
    for char in range((len(inp)/2)+1):
        if inp[char] in mirrors and inp[char] == inp[len(inp)-char - 1]:
            continue
        else:
            is_mirror = False
            break
    if is_mirror:
        print 'YES'
    else:
        print 'NO'
