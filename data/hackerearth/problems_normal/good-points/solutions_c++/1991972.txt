#include<stdio.h>
using namespace std;
void me()
{
	printf("YES\n");
}
int main()
{
	int N;
	scanf("%d",&N);
	
	
	if(N==1)
	{
		me();
		printf("0 0\n");
		return 0;
	}
	
	
	if(N==2)
	{
		me();
		printf("0 0\n1 0\n");
		return 0;
	}
	
	if(N==4)
	{
		me();
		printf("0 0\n0 1\n1 1\n1 0\n");
		return 0;
	}
	printf("NO\n");
	return 0;
	
}
