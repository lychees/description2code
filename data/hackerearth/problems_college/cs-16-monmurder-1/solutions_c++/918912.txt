#include <iostream>
using namespace std;
int notprime[1000001]; 
int main()
{
	for (int g=2;g*g<=1000000; g++)
	{
		if (notprime[g]) continue;
		for (int y=2*g; y<=1000000; y+=g)notprime[y]=1; 
	}
	int a; cin >> a;
	for (int g=0;g<a; g++)
	{
		int b; cin >> b; int tot=0; 
		for (int y=2;y<=b; y++)
		{
			if (notprime[y]) continue;
			int sume=0;
			int x=b; while (x%y==0) {x/=y; sume++;} tot^=sume;
		}
		if (!tot) cout << "Pappu Bhujia" << '\n';

	}

}
