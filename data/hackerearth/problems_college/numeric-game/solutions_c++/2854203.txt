#include <bits/stdc++.h>
using namespace std;
int main()
{
    int a1,a2,b1,b2,C;
    scanf("%d",&C);
  double ratio = (1 + sqrt(5)) / 2;
    	for (int c = 1; c <= C; c++) {
       scanf("%d%d%d%d",&a1,&a2,&b1,&b2);
        long  long count = 0;
        for (int i = a1; i <= a2; i++) {
            double up = i * ratio;
            double down = i / ratio;
            if (b1 >= up || b2 <= down)
                count += b2 - b1 + 1;
            else {
                if (b1 <= down)
                    count += max(0, (int) floor(down) - b1 + 1);
                if (b2 >= up)
                    count += max(0, b2 - (int) ceil(up) + 1);
            }
        }
        printf("Case #%d: %lld\n", c, count);
 
    }
    return 0;
}
