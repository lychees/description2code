using namespace std;
#include<iostream>
#include<math.h>
int main()
{
    int t;
    cin>>t;
    while(t>0)
    {
        unsigned long long  n,m;
        cin>>n;
        cin>>m;
        unsigned long long cnt=0,i=0;
        if(n==1)
            cout<<"0"<<endl;
        else{

        while(cnt<n&&cnt<m)
        {
            cnt=pow(2,i);
            i++;
        }
        if(cnt>=n)
            cout<<i<<endl;
        else
        {

        while(cnt<n)
        {
            cnt=cnt+m;
            i++;
        }
        cout<<i<<endl;
        }
        }
        t--;
    }
}
