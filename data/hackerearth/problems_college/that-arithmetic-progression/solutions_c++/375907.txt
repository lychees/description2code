#include <iostream>
#include <algorithm>
using namespace std;

void findArith(int *a, int n) 
{
	int d,k,i,count=0;
	sort(a,a+n);
	if(n==2){
		d = a[1] - a[0];
		if((d%2)==0) {
			k = a[0] + (a[1]-a[0])/2;
			cout<<"3\n"<<a[0]-d<<" "<<k<<" "<<a[1]+d<<endl;
		}
		else {
			cout<<"2\n"<<a[0]-d<<" "<<a[1]+d<<endl;
		}
	}
	else {
		d = a[n-1] - a[0];
		for(i=1;i<n;i++)
			d = min(d,a[i]-a[i-1]);
		for(i=1;i<n;i++)
		{
			if(a[i] - a[i-1] == d)
				continue;
			if(a[i] - a[i-1] == 2*d){
				k = (a[i] - d);
				count++;
			}
			else {
				cout<<"0\n"; 
				return;
			}
		}
		if(d == 0)
			cout<<"1\n"<<a[0]<<endl;
		else if (count > 1)
			cout<<"0\n";
		else if (count == 1)
			cout<<"1\n"<<k<<endl;
		else if (count == 0)
			cout<<"2\n"<<a[0]-d<<" "<<a[n-1]+d<<endl;
	}
}

int main()
{
    int n,i;
    cin>>n;
    if(n==1)
    	cout<<"-1\n";
    else {
    	int *a = new int[n];
    	for(i=0;i<n;i++) cin>>a[i];
    	findArith(a,n);
        delete []a;
    }
   	
    return 0;
}
