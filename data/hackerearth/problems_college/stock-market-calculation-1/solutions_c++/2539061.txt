#include <iostream>
using namespace std;

int main()
{
	int t; cin>>t;
	while(t--)
	{
		int n; cin>>n;
		int a[n];
		for(int i=0;i<n;i++) cin>>a[i];
		int m=a[n-1];
		long ans=0;
		for(int i=n-2;i>=0;i--)
		{
			if(a[i]>m)
			{
				m=a[i];
			}
			else ans+=(m-a[i]);
		}
		cout<<ans<<endl;
		
	}
	return 0;
}
