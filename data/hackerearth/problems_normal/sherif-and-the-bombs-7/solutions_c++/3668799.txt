using namespace std;
int mat[1001][1001];
int main()
{	 int n;
    cin>>n;
    memset(mat,0,sizeof(mat));
    int x,y,r;
    int count = 0;
    for(int k = 0 ; k<n ; k++)
    {
        cin>>x>>y>>r;
        for(int i = max(x-r,1); i <= min(x+r,1000) ; i++)
            for(int j = max(y-r,1); j <= min(y+r,1000) ; j++)
            {
                if( r*r >= (i-x)*(i-x) + ( j - y )*(j-y) )
                {   if(mat[i][j])
                        mat[i][j] = 2;
                    else
                        mat[i][j] = 1;
                }
            }
    }
    for(int i = 1; i<=1000; i++)
        for(int j = 1; j<=1000; j++)
            if(mat[i][j] == 2)
            {
                count++;
            }
    cout<<count<<endl;
	return 0;
}
