#include<bits/stdc++.h>
long long int count,dp[100001];
int main(){
    int t,n;
    scanf("%d",&t);
    while(t--){
        long long int count=0;
        scanf("%d",&n);
        dp[0]=dp[1]=1;
        dp[2]=2;
        for(int i=3;i<=n;i++){
            dp[i]=(dp[i-1]+dp[i-2]+dp[i-3])%1000000007;
        }
        printf("%lld\n",dp[n]);
    }
    return 0;
}
