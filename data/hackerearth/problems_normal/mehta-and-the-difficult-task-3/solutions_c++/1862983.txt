typedef unsigned long long int ull;
typedef long long int ll;
typedef vector<vector<ll> > mat;

ll m=1e9+7;
mat in(16,vll(16)),res(16,vll(16));
mat I(16,vll(16));

mat mul(mat a,mat b)
{
	int n=a.size();
	mat c(n,vll(n));
	rep(i,0,n-1)rep(j,0,n-1)rep(k,0,n-1)
	{
		c[i][j]=(c[i][j]+a[i][k]*b[k][j])%m;
	}
	return c;
}

mat p(mat a,ll b)
{
	mat r=I;
	while(b)
	{
		if(b%2)
		r=mul(r,a);
		a=mul(a,a);
		b/=2;
	}
	return r;
}
