#include <iostream>
#include <string.h>
#define ll long long
using namespace std;
ll n, k;
ll dp[2000][2000];
ll solve(ll s, ll a[], ll i)
{
	if(i==n) return s;
	if(s==k) return k;
	if(dp[i][s]!=-1) return dp[i][s];
	else
	{
		ll res=0, n1=0;
		if((s+a[i])<=k)
			n1=solve(s+a[i],a,i);
		res=max(n1,solve(s,a,i+1));
		dp[i][s]=res;
		return res;
	}
}
int main()
{
	ll t, i, ans;
    cin>>t;
    while(t--)
    {
    	memset(dp,-1,sizeof(dp));
    	cin>>n>>k;
    	ll a[n];
    	for(i=0;i<n;i++)
    		cin>>a[i];
    	ans=solve(0,a,0);
    	cout<<ans<<endl;
    }
    return 0;
}
