t = input()
for i in range(t):
    n = input()
    a = raw_input()
    b = map(int, a.split(" "))
    c = b[0]
    t = 0
    for j in range(1, n):
        d = b[j]
        if (c == d):
            print "bad luck"
            break
        else:
            t = t + 1
            c = b[j]
    if (t == n-1):
        print "can do"
