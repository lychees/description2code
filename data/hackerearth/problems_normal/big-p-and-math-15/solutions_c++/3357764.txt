#include <iostream>
#include<cmath>
using namespace std;

long long findmin(long long num)
{   
	long long temp=num;
	int i=0;
	while(temp)
	{
		if(temp%10 == 6)
		{
			num-=pow(10,i);
		}
		temp/=10;
		i++;
	}
	return num;
}

long long findmax(long long num)
{
	long long temp=num;
	int i=0;
	
	while(temp)
	{
		if(temp%10 == 5)
		{
			num+=pow(10,i);
		}
		temp/=10;
		i++;
	}
	return num;
}



int main()
{   long long A,B,minA,minB,maxA,maxB,minval,maxval;
    cin>>A>>B;
    minA=findmin(A);
    minB=findmin(B);
    maxA=findmax(A);
    maxB=findmax(B);
    minval=minA+minB;
    maxval=maxA+maxB;
    cout<<minval<<" "<<maxval;
    return 0;
}
