#include<bits/stdc++.h>
using namespace std;
#define ll long long int
struct box
{
    ll filled,capa;
};
struct myfunc
{
    bool operator()(struct box a,struct box b)
    {
       return (a.capa>=b.capa);
    }
};
int main()
{
    ll sum,test,i,j,total,num,m,n,remain;
    string str;
    cin>>test;
    struct box arr[100001];
    while(test--)
    {
        total=0;
        cin>>num;
        for(i=0;i<num;i++)
            cin>>arr[i].filled;
        for(i=0;i<num;i++)
        {
            cin>>arr[i].capa;
            total+=(arr[i].filled);
        }
        sum=0;
        sort(arr,arr+num,myfunc());

        //total=arr[0].capa-arr[0].filled;
        for(i=0;i<num;i++)
        {
            total-=arr[i].filled;
            sum+=1;
            remain=arr[i].capa-arr[i].filled;
            if(remain>=total)
                break;
            else
                total=total-remain;
        }
        cout<<(sum)<<endl;
    }
}
