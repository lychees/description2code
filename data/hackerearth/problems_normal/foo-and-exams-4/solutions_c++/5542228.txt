#include <iostream>
using namespace std;

int main()
{
    int t,i=0;
    unsigned long long int ft,T;
    scanf("%d", &t);
    int *a=new int[t],
    	*b=new int[t],
    	*c=new int[t],
    	*d=new int[t];
    unsigned long long *k=new unsigned long long[t];
    while(t--)
    {
    	scanf("%d%d%d%d%llu", (a+i),(b+i), (c+i), (d+i), (k+i));
    	T=1;
    	while(1)
    	{
    		ft=((*(a+i))*T*T*T) + ((*(b+i))*T*T) + ((*(c+i))*T) + (*(d+i));
    		//printf("%llu ",ft);
  			if(ft < *(k+i))
  				T++;
  			else
  			{
  					printf("%llu\n",T-1);
  					break;
  			}
    	}
    
    	i++;
    }
    return 0;
}
