#include <bits/stdc++.h>
using namespace std;
vector<int>g[600],cost[600];
int dis[600];
struct node{
	int u,w;
	node(int a, int b)
	{
		u=a;w=b;
	}
	bool operator < (const node& p) const  {return w>p.w;}
};
int dj(int s,int n)
{
	//memset(dis,0,sizeof(dis));
	for(int i=0;i<600;i++)dis[i]=10000000;
	priority_queue< node >q;
	q.push(node(s,0));
	dis[s]=0;
	while(!q.empty())
	{
		node top = q.top();
		q.pop();
		int u = top.u;
		//cout<<"u"<<u<<endl;
		//int v = node.second;
		if(u==n)return dis[n];
		for(int i=0;i<g[u].size();i++)
		{
			int v = g[u][i];
			//cout<<v<<' '<<dis[v]<<endl;
			if(dis[u]+cost[u][i]<dis[v])
			{
				dis[v] = dis[u] + cost[u][i];
				q.push(node(v,dis[v]));
			}
		}
	}
	return -1;
}
int main() 
{
	//for(int i=0;i<600;i++)dis[i]=10000000;
	int n,u,v,w; cin>>n;
	for(int i=1;i<=n;i++)
	{
		cin>>u>>v>>w;
		g[u].push_back(v);
		g[v].push_back(u);
		cost[u].push_back(w);
		cost[v].push_back(w);
	}
	int q,s,d; cin>>s>>q;
	while(q --)
	{
		cin>>d;
		int ttt = dj(s,d);
		//cout<<dis[d]<<endl;
		if(ttt==-1)cout<<"NO PATH"<<endl;
		else cout<<ttt<<endl;
	}
}
