def find_winner(players, size):
	global mat
	if size == 1:
		return players[0]+1

	next_players = []	
	for i in range(0, size, 2):
		winner = players[i] if mat[players[i+1]-1][players[i]] == 0 else players[i+1]
		next_players.append(winner)

	return find_winner(next_players, size/2)	


n = int(raw_input())
no_of_players = 2**n

mat = [map(int, raw_input().split()) for i in range(no_of_players-1)]
print find_winner(range(no_of_players), no_of_players)
