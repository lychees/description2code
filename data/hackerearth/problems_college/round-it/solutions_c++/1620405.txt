#include <iostream>
using namespace std;

int main()
{
    int t,b,x;
    float y,i;
    long n;
    scanf("%d",&t);
    while(t--)
    {	scanf("%ld %d",&n,&b);
    	i=n/b;
    	x=n-i*b;
    	y=(i+1)*b-n;
    	if(x<y)
    		n=i*b;
    	else
    		n=(i+1)*b;
    		
    		
    	printf("%ld\n",n);    		
    		
    		
    }
    return 0;
}
