#include <iostream>
using namespace std;

int main()
{
   long long int n,count,p,m,j,k,max;
   cin>>n;
   for(int i=0;i<n;i++)
   { count=0;
   	 cin>>p>>m;
   	  int pr[100],mr[100];
   	 j=0;k=0;
   	 while(p!=0)
   	 { 
   	 	pr[j]=p%2;
   	 	p=p/2;
   	 	j=j+1;
   	 }
   	 while(m!=0)
   	 {
   	 	mr[k]=m%2;
   	 	m=m/2;
   	 	k=k+1;
   	 }
   	 if(j>k)
   	 {
   	 	for( ;k<j;k++)
   	 	mr[k]=0;
   	 	max=j;
   	 }
   	 else
   	 {
   	 	for( ;j<k;j++)
   	 	pr[j]=0;
   	 	max=k;
   	 }
   	 for(int l=0;l<max;l++)
   	 {
   	 	if(pr[l]!=mr[l])
   	 	count++;
   	 }
   	 cout<<count<<"\n";
   }
    return 0;
}
