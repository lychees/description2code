t = int(raw_input())
for i in range(t):
	n = int(raw_input())
	arr = map(int,raw_input().split(' '))
	arr.sort()
	ans=1
	is_succ = False
	for j in range(1,n):	
		if is_succ:
			is_succ = False
			ans+=1
			continue
		if(arr[j]-arr[j-1])==1:
			is_succ = True
			continue
		ans+=1
	print ans
