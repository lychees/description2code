#include <stdio.h>

int main()
{
    int a[100000],n;
    scanf("%d",&n);
   int i=0;
   for(i=0;i<n;i++)
   scanf("%d",&a[i]);
   int m,j,k=0;
   for(i=0;i<n;i++)
   { if(a[i]!=0) 
         m=a[i];
      else  
        continue;
        
      for(j=0;j<n;j++)
      if(m==a[j]&&j!=i)
       {  k++;
          a[j]=0;
       }
   }
   printf("%d\n",k);
   return 0;
       
}
