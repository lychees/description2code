#include <cstdio>
#include <algorithm>
 
using namespace std;
 
const int N = 2010, inf = 1000000007;
int n, c, x, y, z, t;
int p[N], dp[N][N];
char a[N], b[N];
 
int ce(char a, char b)
{
    if (a == b) return 0;
    if (a == '0') return x;
    if (a == '1') return y;
    return z;
}
 
void solve()
{
    scanf("%d %d %d %d %d", &n, &x, &y, &z, &t);
    scanf("%s %s", a + 1, b + 1);
    c = 0;
    for (int i = 1; i <= n; i++) if (b[i] == '0') p[++c] = i;
    for (int i = 0; i <= n; i++)
        for (int j = 0; j <= n; j++) dp[i][j] = inf;
    dp[0][0] = 0;
    for (int i = 1; i <= n; i++)
    {
        dp[i][0] = dp[i - 1][0] + ce(a[i], '1');
        for (int j = 1; j <= i; j++) dp[i][j] = min(ce(a[i], '1') + dp[i - 1][j], ce(a[i], '0') + dp[i - 1][j - 1] + t *abs(i - p[j]));
    }
    printf("%d\n", dp[n][c]);
}
 
int main()
{
    int t;
    scanf("%d", &t);
    while (t--) solve();
    return 0;
}
