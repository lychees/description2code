#include<bits/stdc++.h>
//zone
using namespace std;
int main()
{
    int n;
    cin >> n;
    int a[n + 1];
    int i;
    int ans;
    for(i = 1 ; i <= n ; i++)
    {
        cin >> a[i];
    }
    ans = 0;
    cout<<a[1]<< " ";
    int fo = a[1];
    for(i = 1 ; i < n ;i++)
    {
        ans = (i + 1) * a[i + 1] - fo;
        fo = fo + ans;
        //cout<<fo<<endl;
        cout<<ans<<" ";
    }
    return 0;
}
