#include<bits/stdc++.h>
using namespace std;

#define fr(i,n) for(int i=0;i<n;i++)
#define fr1(i,n) for(int i=1;i<=n;i++)
#define ll long long
#define pii pair<int,int>
#define ff first
#define ss second
#define mp make_pair
#define pb push_back
#define MOD 1000000007
ll cnt[30];

ll pow(ll a,ll b){
	ll ans=1;
	while(b){
		if(b%2){
			ans=ans*a;
		}
		b/=2;
		a*=a;
	}
	return ans;
}

int main(){
	int t,n;
	cin>>t;
	ll ans1,ti;
	while(t--){
		cin>>n;
		memset(cnt,0,sizeof(cnt));
		ll max1=-30;
		ll cnt1=0,min1=30;
		if(n==1){
			cin>>ti;
			cout<<ti<<" "<<ti<<endl;
			continue;
		}
		fr(i,n){
			cin>>ti;
			cnt[ti+10]++;
			min1=min(ti,min1);
			if(ti<0)cnt1++;
			max1=max(ti,max1);
		}
		ll ans=1;
		ll cnt0=0;;
		if(!cnt1){
			for(ll i=20;i>10;i--){
				if(cnt[i]){
					cnt0++;
					ans*=pow(1ll*(i-10),cnt[i]);
				}
			}
			(cnt0)?cout<<ans<<" ":cout<<"0"<<" ";
			cout<<min1<<endl;
			continue;
		}
		ans=1;
		if(cnt1)
		for(int k=11;k<=20;k++){
			if(cnt[k])ans*=pow(1ll*(k-10),cnt[k]);
		}
		if(cnt1%2){
			fr(j,10){
				if(cnt[j])
				ans*=pow(j-10,cnt[j]);
			}
			ans1=ans;
			
			int l;
			for(l=9;l>0;l--){
				if(cnt[l]){
					ans/=(l-10);
					break;
				}
			}
			if(max1==0&&cnt1==1){
				cout<<"0"<<" "<<ans1<<endl;
				continue;
			}
			else cout<<ans<<" "<<ans1<<endl;
		}
		else{
			fr(j,10){
				if(cnt[j])
					ans*=pow(1ll*(j-10),cnt[j]);
			}
			int l;
			for(l=9;l>0;l--){
				if(cnt[l]){
					ans/=(l-10);
					break;
				}
			}
			if(max1==0&&cnt1==1){
				cout<<"0"<<" "<<ans<<endl;	
			}
			else cout<<ans*(l-10)<<" "<<ans<<endl;
		}
		
	}	
}
