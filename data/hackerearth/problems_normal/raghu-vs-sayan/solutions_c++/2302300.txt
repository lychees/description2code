#include <bits/stdc++.h>
using namespace std;

int main()
{
   int tc;
   long int arr[10000];
   cin>>tc;
   while(tc--)
   {
   	long int A,B,N;
   	cin>>A>>B>>N;
   	for(long int i=0;i<N;i++)
   	{
   		cin>>arr[i];
   	}
   	sort(arr,arr+N);
   	long int counta=0,countb=0;
   	for(long int i=0;i<N;i++)
   	{
   		if(arr[i]<=A)
   		{
   			counta++;
   			A=A-arr[i];
   		}
   		else
   		{
   			break;
   		}
   	}
   	for(long int i=0;i<N;i++)
   	{
   		if(arr[i]<=B)
   		{
   			countb++;
   			B=B-arr[i];
   		}
   		else
   		{
   			break;
   		}
   	}
   	if(counta>countb)
   	{
   		cout<<"Raghu Won"<<endl;
   	}
   	else if(counta<countb)
   	{
   		cout<<"Sayan Won"<<endl;
   	}
   	else if(counta==countb)
   	{
   		cout<<"Tie"<<endl;
   	}
   }
}
