T = int(raw_input())
S = []

for i in range(T):
	S.append(raw_input())

for i in range(T):
	F = []
	for j in range(len(S[i])):
		if S[i][j] not in F:
			F.append(S[i][j])
	if len(F) == 26:
		print "YES"
	else:
		print "NO"
