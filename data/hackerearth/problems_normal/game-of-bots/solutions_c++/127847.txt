#include <bits/stdc++.h>
using namespace std;
#define pll make_pair
int main()
{
	int T;
	cin>>T;
	while(T--)
	{
		string A,B;
		std::vector<pair<int,int> > V,X;
		cin>>A>>B;
		for(int i = 0;i<A.length();i++)
		{
			if(A[i]!='#')
				V.push_back(pll(i,A[i]-'A'));
			if(B[i]!='#')
				X.push_back(pll(i,B[i]-'A'));
		}
		if(V.size()!=X.size())
		{
			cout<<"No"<<endl;
			continue;
		}
		int F = 0;
		for(int i = 0;i<V.size();i++)
		{
			if(V[i].second!=X[i].second){
				F = 1;
				continue;
			}
			if(V[i].first==X[i].first)
				continue;
			if(V[i].second==0&&V[i].first<X[i].first)
				F = 1;
			if(V[i].second==1&&V[i].first>X[i].first)
				F = 1;
		}
		if(F==1)
			cout<<"No"<<endl;
		else
			cout<<"Yes"<<endl;

	}
	return 0;
}
