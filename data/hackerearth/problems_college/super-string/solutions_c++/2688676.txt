#include<cstdio>
#include<cstring>
#include<iostream>


using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string str;
		cin>>str;
		int i,count[256]={0};
		int len = str.length();
		for(i=0;i<len;i++)
		{
			count['Z'-str[i]+1]++;
		
			
		}
		int flag=0;
		for(i=1;i<=26;i++)
		{
			if(count[i]!=0)
			{
			if(count[i]!=i)
			{
				
				flag =1;
				break;
			}
		}
		}
		if(flag==1)
		{
			cout<<"No"<<"\n";
		}
		else
		{
			cout<<"Yes"<<"\n";
		}

}}
