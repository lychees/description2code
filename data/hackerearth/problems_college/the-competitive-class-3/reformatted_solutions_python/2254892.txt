inputStrList = raw_input().split()
inputIntList = []
for strNum in inputStrList:
    inputIntList.append(int(strNum))

inputIntList.sort(reverse=True)
# print inputIntList


rankDict = {}
firstRank = 1
for num in inputIntList:
    rankDict[firstRank] = num
    firstRank = firstRank+1


# print rankDict

resultString = ""
for inputMarks in inputStrList:
    for rank, marks in rankDict.items():
        if int(inputMarks) == marks:
            resultString = resultString+str(rank)+" "
            break
