#include <iostream>
#include <bits/stdc++.h>
using namespace std;
multiset <int> se;
multiset <int> :: iterator it;
int main()
{
    int n,m,x,i,ans;
    cin>>n>>m;
    for(i=0;i<n;i++)
    {
    	cin>>x;
    	se.insert(x);
    }
    for(i=0;i<n;i++)
    {
    	cin>>x;
    	se.insert(x);
    }
    ans=0;
    for(it=se.begin();it!=se.end();it++)
    {
    	if(m>=*it)
    	{
    		m-=*it;
    		ans++;	
    	}
    	if(ans>=n||m<=0)
    	{
    		break;
    	}
    }
    cout<<ans<<endl;
    return 0;
}
