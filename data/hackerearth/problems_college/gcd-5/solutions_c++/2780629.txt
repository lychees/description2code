#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   long long int t,n,arr[10000],i,flag;
   cin>>t;
  
   while(t--)
   {
   	cin>>n;
   	flag=1;
   	for(i=0;i<n;i++)
   	cin>>arr[i];
   	
   	for(i=0;i<n-1;i++)
   	{
   	 if(arr[i]>arr[i+1])
   	 {
   	  	flag=0;
   	  	break;
   	 }
   	}
   	if(flag==1)
   	cout<<"YES"<<endl;
   	else
   	cout<<"NO"<<endl;
   }
    return 0;
}
