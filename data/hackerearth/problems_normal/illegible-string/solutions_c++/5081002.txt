#include "bits/stdc++.h"
using namespace std;
int main()
{
	long int n;
	cin >> n;
	
	string s;
	cin >> s;
	
	long int count=0;
	for(long int i=0;i<s.length();i++)
		if(s[i] == 'w')	
			count++;
	long int maxi=count+s.length();
	
	char arr[1000001];
	long int arrC=1;
	arr[0]='A';
	for(long int i=0;i<s.length();i++)
	{
		if(s[i]=='v')
		{
			if(arr[arrC-1]=='v')
			{
				arr[arrC-1]='w';
			}
			else
			{
				arr[arrC++]=s[i];
			}
		}
		else if(s[i]=='w')
		{
			if(arr[arrC-1]=='v')
			{
				arr[arrC++]='w';
				swap(arr[arrC-2],arr[arrC-1]);
			}
			else
			{
				arr[arrC++]=s[i];
			}
		}
		else
		{
			arr[arrC++]=s[i];
		}
	}
	cout << arrC-1 << " " << maxi << endl;
	return 0;
}
