#include <iostream>
#include <string>
#include <vector>
#include <cctype>

using namespace std;

int main()
{
	string w;
	vector <pair <string, int> > words;

	while (cin >> w)
	{
		bool found = false;
		for (int i = 0; i < w.length(); i++)
		{
			w[i] = toupper(w[i]);
		}

		for (int i = 0; i < words.size(); i++)
		{
			if (words[i].first == w)
			{
				words[i].second++;
				found = true;
				break;
			}
		}

		if (!found)
		{
			words.push_back(make_pair(w, 1));
		}
	}

	for (int i = 0; i < words.size(); i++)
	{
		cout << words[i].first << " " << words[i].second << endl;
	}

	return 0;
}
