#include <iostream>
using namespace std;

int main()
{
    int n,m,i,j,k,max,t;
    cin>>n>>m;
    int *c=new int[m];
    int *p=new int[n];
    int *a=new int[n];
    for(i=0;i<m;i++)
    	cin>>c[i];
    for(i=0;i<n;i++)
    	cin>>p[i];
    max=0,t;
    for(i=0;i<n;i++)
    {
    	t=c[p[i]-1];
   		j=((max/t)*t)+t;
   		k=p[i];
    	for(;k<m;)
    	{
    		if(j%c[k]==0)
    		{
    			j=j+t;
    			k=p[i];
    		}
    		else
    			k++;
    	}
    	a[i]=max=j;
    }
    cout<<a[n-1];
    	
    
    
    return 0;
}
