#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    int t;
    long long a[1000008],d,count,n,i;
    cin>>t;
    while(t--)
    {
        count=0;
        cin>>n>>d;
        for(i=0;i<n;i++)cin>>a[i];
        sort(a,a+n);
        for(i=0;i<n;i++)
        {
            if(a[i+1]-a[i]<=d)
            {
                count++;
                i=i+2;
            }
            else
            {
             i++;   
            }
        }
        cout<<count<<"\n";
    }
    return(0);
}
