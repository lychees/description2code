#include <iostream>
using namespace std;

int main()
{   int n;
   cin>>n;
  
   int arr[n];
   bool flag=true;
   int i;
   for(i=0;i<n;i++)
      cin>>arr[i];
    
    for(i=0;i<n-1;i++)
    {
    	arr[i+1]=arr[i+1]-arr[i];
    	arr[i]=0;
    	
    }
    
    for(i=0;i<n;i++)
    {
    	if(arr[i]!=0)
    	{
    		flag=false;
    		break;
    	}
    }
   
   if(flag)
    cout << "YES" << endl;
    else
    cout << "NO" << endl;
  
    return 0;
}
