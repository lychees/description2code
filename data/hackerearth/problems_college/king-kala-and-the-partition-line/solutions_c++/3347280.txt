#include <iostream>

using namespace std;

int main()
{
	int t, n, a, b, c, i;
	int part1, part2;
	cin >> t;
	while(t--){
		i = 0;
		part1 = 0;
		part2 = 0;
		cin >> n;
		cin >> a >> b >> c;
		int x[n], y[n];
		while(n--){
			cin >> x[i] >> y[i];
			if(a*x[i] + b*y[i] + c  < 0)
				part1++;
			else if(a*x[i] + b*y[i] + c  > 0)
				part2++;
			i++;
		}
		if(part1 > part2)
			cout << part1 << endl;
		else
			cout << part2 << endl;
		
		
	}
    return 0;
}
