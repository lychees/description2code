#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,a;
    	cin>>n;
    	double arr[n];
    	for(int i=0;i<n;i++)
    	cin>>arr[i];
    	
    	
    	int l=0,h=n-1;
    	int c1=0,c2=0;
    	while(l<=h)
    	{
    		double lp=arr[l],rp=arr[h];
    		//cout<<l<<" "<<h<<" "<<lp<<" "<<rp<<" "<<c1<<" "<<c2<<endl;
    		if(l==h)
    		{
    			c1++;
    			break;
    		}
    		else if(lp==2.0*rp)
    		{
    			c1++;
    			c2++;
    			l++;
    			h--;
    			if(l==h)
    			{
    				arr[l]=0;
    				c2--;
    				break;
    			}
    		}
    		else if(lp>2.0*rp)
    		{
    			arr[l]=arr[l]-2*rp;
    			c2++;
    			h--;
    			if(l==h)
    			{
    				c1++;
    				break;
    			}
    		}
    		else
    		{
    			arr[h]=arr[h]-lp/2.0;
    			c1++;
    			l++;
    			if(h==l)
    			{
    				c2++;
    				break;
    			}
    		}
    	}
    	cout<<c1<<" "<<c2<<endl;
    }
    return 0;
}
