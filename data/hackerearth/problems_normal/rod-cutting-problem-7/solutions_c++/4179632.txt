template<typename Num>
inline bool requiresStepTwo(Num x) {return x&(x-1);}

#include<iostream>
#include<vector>
#include<cmath>

typedef unsigned long long ull;

int main()
{
	short int t;
	std::cin>>t;
	while(t--)
	{
		unsigned int rod;
		std::cin>>rod;
				
		std::vector<ull> rods(1,rod);
		
		while(rod!=1)
		{
			(rod&1)?rods.push_back(rod/2):rods.push_back(rod/2+1);
			rod/=2;
		}
		ull ans=0;
		
		for(auto& r:rods)
			ans+=requiresStepTwo(r);
		std::cout<<ans<<'\n';
	}
}
