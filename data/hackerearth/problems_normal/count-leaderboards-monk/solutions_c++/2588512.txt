const int maxN = 1000 + 5;
const int mod = 1000*1000*1000 + 7;

int c[maxN][maxN], dp[maxN];
int fac[maxN];

main()
{
    ios::sync_with_stdio(0); cin.tie(0); 
    rep(i,0,maxN) c[i][0] = 1;
    rep(i,1,maxN) rep(j,1,maxN) c[i][j] = (c[i-1][j] + c[i-1][j-1]) % mod;
    int n, k; cin >> n >> k;

    fac[0] = 1;
    rep(i,0,n) fac[i+1] = (1LL) * (i+1) * fac[i] % mod;

    dp[0] = 1;
    rep(i,1,n+1)
    {
	  rep(j,0,i)
	  {
		int tmp = (1LL) * fac[i-j] * dp[j] % mod;
		tmp = (1LL) * tmp * c[i-1][j] % mod;
		dp[i] = (dp[i] + tmp) % mod;
	  }
    }

    int ans = (1LL) * dp[n] * c[n][k] % mod;
    cout << ans << endl;
    return 0;
}
