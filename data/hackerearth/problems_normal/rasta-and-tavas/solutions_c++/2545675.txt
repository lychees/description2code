#include <cstdio>
#include <algorithm>

using namespace std;

int n,k,i,j,first[1000005],dp[1000005];

int main(){
	scanf("%d%d", &n, &k);
	for(i = 2; i <= n; i++)
	first[i] = i;
	int p = 2;
	while(p * p <= n){
		if(first[p] == p){
			int x = p * p;
			while(x <= n){
				first[x] = p;
				x += p;
			}
		}
		p++;
	}
	int cnt = 0;
	for(i = 2; i <= n; i++){
		int x = i / first[i];
		if(x % first[i] == 0)
		dp[i] = dp[x];
		else
		dp[i] = dp[x] + first[i];
		if(dp[i] == k)
		cnt++;
	}
	printf("%d\n", cnt);
}
