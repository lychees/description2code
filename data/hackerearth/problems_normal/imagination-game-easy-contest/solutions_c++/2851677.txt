#include <iostream>
#include <cstdio>
using namespace std;
int a[11];
int main()
{
    int t;
    cin>>t;
    
    while(t--) {
    	//ios_base::sync_with_stdio(false);
    	int n;
    	scanf("%d", &n);
    	int xorr = 0;
    	for(int i = 0; i < n; ++i) {
    		scanf("%d", &a[i]);
    	}
    	for(int i = 0; i < n; ++i) {
    		int x;
    		scanf("%d", &x);
    		xorr ^= a[i]-x;
    		xorr ^= x-1;
    	}
    	if(xorr != 0) {
    		printf("First\n");
    	}
    	else {
    		printf("Second\n");
    	}
    }
    
    return 0;
}
