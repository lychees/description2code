#include <iostream>
using namespace std;

int main()
{
   int t,n,x,y;
   cin>>t;
   while(t--)
   {
   	int x,y,n;
   	cin>>x>>y>>n;
   	long long int a[n];
   	int i=0;
   	long long int sum=0;
   	while(i<x&&i<n)
   	{
   		
   		a[i]=y;
   		
   		sum+=a[i];
   		//cout<<a[i]<<" ";
   		i++;
   	}
   	//cout<<sum<<" sum "<<endl;
   int j=0;
   	while(i<n)
   	{
   		
   		a[i]=sum;
   		sum+=a[i]-a[j];
   		i++;
   		j++;
   	}
   	cout<<a[n-1]<<endl;
   }
    return 0;
}
