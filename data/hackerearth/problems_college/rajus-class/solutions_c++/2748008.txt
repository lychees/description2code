#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

int main()
{
    int t,n,r,a,i;
    string name;
    
    cin>>t;
    while(t--){
    	cin>>n;
    	vector <pair<int,string>> vp;
    	for(i=0;i<n;i++){
    		cin>>a>>name;
    		vp.push_back(make_pair(a,name));
    	}
    	
    	sort(vp.begin(),vp.end());
    	cin>>r;
    	cout<<vp[r-1].second<<endl;
    }
    return 0;
}
