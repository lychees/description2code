#include <iostream>
#include<stdio.h>
#include<algorithm>
#include<vector>
using namespace std;
long long int a[1000005];
int main()
{
	int t,n,i;
	long long int ans;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		for(i=0;i<n;i++)
			scanf("%lld",&a[i]);
		sort(a,a+n);
		ans=0;
		for(i=0;i<n;i+=2)
			ans+=(a[i]*a[i+1]);
		printf("%lld\n",ans);
	}
    return 0;
}
