#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin >>t;
    while(t--)
    {
    string a,b;
    int count=0;
    cin >>a>>b;
    for(int i=0;i<=a.length();i++)
    {
      string x,y;
      if(i==0)
      {
      	x=b+a.substr(i);
      }
      else if(i==a.length())
      {
      	x=a.substr(0,i)+b;
      }
      else
      {
      	x=a.substr(0,i)+b+a.substr(i);
      }
      y=x; 
       reverse(x.begin(),x.end());
      
      if(x==y)
      count++;
    }
    cout <<count <<endl;
    }
    return 0;
}
