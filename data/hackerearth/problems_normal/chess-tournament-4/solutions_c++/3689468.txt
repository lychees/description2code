#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n,k;
    cin>>n;k=pow(2,n);
    int b[k][k],a[k];
    for(int i=0;i<k;i++)
    {
    	a[i]=i+1;
    	for(int j=0;j<k;j++)
    	{
    		if(i>j)cin>>b[i][j];
    	}
    }
    for(int i=0;i<n;i++)
    {
    	for(int j=0;j<pow(2,n-1-i);j++)
    	{
    		if(b[a[2*j+1]-1][a[2*j]-1]==0)a[j]=a[2*j];
    		else a[j]=a[2*j+1];
    	}
    }
    cout<<a[0];
    return 0;
}
