#include<bits/stdc++.h>
using namespace std;
int main()
{

vector<int>a;
int k;
cin>>k;
for(int i=0;i<k;i++)
{
int temp;
cin>>temp;
a.push_back(temp);
}
sort(a.begin(),a.end());
reverse(a.begin(),a.end());
int count=0;int sum=0;int max=-1;
for(int i=0;i<k;i++)
{
++count;
sum=count+a[i];
if(sum>max)max=sum;

}
cout<<max+1<<endl;



return 0;
}
