#include <bits/stdc++.h>
using namespace std;
int main()
{
    int tst;
    cin>>tst;
    while(tst--)
    {
        string str="";
        while(true)
        {
            char ttc;
            cin>>ttc;
            if(ttc=='.')break;
            if(ttc==' ')continue;
            str+=ttc;
        }
        stack<char>rs;
        for(int i=0;i<str.size();i++)
        {
            if(str[i]=='{'||rs.empty())
                {
                    rs.push(str[i]);
                }
            else
            {
                if(str[i]=='}'&&rs.top()=='{')
                {
                    rs.pop();
                }
                else if(str[i]=='}'&&rs.top()=='}')
                    rs.push('}');
            }
        }
        if(rs.size()==0)
        {
            cout<<"0\n";
        }
        else
        {
            int64_t tt=0,xx=0,yy=0,ss=rs.size();
            if(rs.size()%2!=0)
            {
                cout<<"-1\n";
                continue;
            }
            while(!rs.empty())
            {
                if(rs.top()=='}')
                {
                  xx++;
                }
                if(rs.top()=='{')
                    {
                        yy++;
                    }
                rs.pop();
                tt++;
            }
            if(xx==ss/2&&yy==ss/2)
            {
                cout<<tt<<"\n";
            }
            else
                cout<<tt/2<<"\n";
        }
    }
     return 0;
}
