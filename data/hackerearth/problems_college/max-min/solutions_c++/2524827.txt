#include<stdio.h>
#include<iostream>
#include<algorithm>
#define size 100002

using namespace std;

long long a[size];
int main()
{
     long long n,i,k,j,min=10000000000000;

     scanf("%lld%lld",&n,&k);

     for(i=0;i<n;i++)
        scanf("%lld",&a[i]);

     sort(a,a+n);

     j=k-1;
     for(i=k-1;i<n;i++)
     {
         if((a[i]-a[i-j])<min)
            min=a[i]-a[i-j];
     }

     printf("%lld\n",min);
}
