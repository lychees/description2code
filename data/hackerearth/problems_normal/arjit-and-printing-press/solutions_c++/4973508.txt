#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t,x,y,i,j;
    cin>>t;
    char w[100000];
    char r[100000];
    while(t--){
    	cin>>w;
    	cin>>r;
    	x=strlen(w);
    	y=strlen(r);
    	sort(r,r+y);
    	i=j=0;
    	while(w[i]!='\0'){
    		if(r[j]<w[i]&&j<y){
    			w[i]=r[j];
    			i++;
    			j++;
    		}
    		else
    			i++;
    	}
    	cout<<w<<endl;
    }
    return 0;
}
