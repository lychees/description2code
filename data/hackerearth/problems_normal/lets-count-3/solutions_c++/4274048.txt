ll gcd(ll a,ll b)
{
	if(a == 0) return b;
	if(b == 0) return a;
	
	if(a>b)
	return gcd(b,a%b);
	return gcd(a,b%a);
	
}

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    
    int t;
    cin>>t;
    
    while(t--)
    {
    	ll a,b,c;
    	cin>>a>>b>>c;
    	
    	ll m=min(a,min(b,c));
    	
    	ll x,ans=0;
    	
    	rep(x,1,sqrt(m),1)
    	{
    		if(m%x == 0)
    		{
    			if(m/x == x)
    			{
    				if(a%x==0 && b%x==0 && c%x==0) ans++;
    			}
    			else
    			{
    				if(a%x==0 && b%x==0 && c%x==0) ans++;
    				if(a%(m/x)==0 && b%(m/x)==0 && c%(m/x)==0) ans++;
    			}
    			
    		}
    	}
    	
    	cout<<ans<<"\n";
    }
    
    return 0;
}
