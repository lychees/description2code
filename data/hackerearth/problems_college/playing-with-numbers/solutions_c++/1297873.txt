#include <iostream>
using namespace std;

int main()
{
    long long int n,t,n1,twp,fplayer=1,count;
    scanf("%lld",&t);
    while(t--)
     {
     	scanf("%lld",&n);
     	count=0;
     	
     	while(n>0)
     	 {
     	 	twp=1;
     	 	n1=n;
     	 while(n1>0)
     	 	 {
     	 	 	
     	 	 	twp=twp*2;
     	 	 	n1=n1/2;
     	 	 }
     	 	 twp=twp/2;
     	 	 n=n-twp;
     	 count++;	 
     	 }
     //	cout<<count<<endl;
     	if(count%2==0)
     	 {
     	 	if(fplayer==1)
     	 	 {
     	 	 	fplayer=0;
     	 	 	printf("Watson\n");
     	 	 }
     	 	 else
     	 	  {
     	 	  	fplayer=1;
     	 	  	printf("Sherlock\n");
     	 	  }
     	 	 
     	 }
     	else
     	 {
     	 	if(fplayer==1)
     	 	 printf("Sherlock\n");
     	 	 
     	 	 else
     	       printf("Watson\n");
     	 	  	
     	 }
     	 
     }
    return 0;
}
