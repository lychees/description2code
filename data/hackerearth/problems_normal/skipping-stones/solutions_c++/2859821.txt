#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n;
	int l,d1;
	cin>>n>>l>>d1;
	vector<double> p(n+2,0),d(n+2,0);
	p[0]=1.0;
	d[0]=0;
	p[n+1]=1.0;
	d[n+1]=d1;
	for(int i=1;i<=n;i++){
		cin>>p[i];
	}
	for(int i=1;i<=n;i++){
		cin>>d[i];
	}
	for(int i=1;i<=n+1;i++){
		int j=i-1;
		double val=-1.0;
		while(j>=0 && d[i]-d[j]<=l){
			val=max(val,p[j]);
			j--;
			
		}
		p[i]=p[i]*val;
		if(p[i]<0.0){
			cout<<"IMPOSSIBLE\n";
			return 0;
		}
	}
	cout<<fixed<<setprecision(6)<<p[n+1]<<endl;
    //cout << "Hello World!" << endl;
    return 0;
}
