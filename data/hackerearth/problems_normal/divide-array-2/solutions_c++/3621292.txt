#include<iostream>
using namespace std;
int main()
{
    unsigned long long int  N,i,Q,x,D=1,max,flag=0;
    cin>>N;
    unsigned long long int a[N];
    for(i=0;i<N;i++)
    {
        cin>>a[i];
        if( i==0)
            max = a[0];
        if(max < a[i])
            max = a[i];
    }
    cin>>Q;
    for(i=0;i<Q;i++)
    {
        cin>>x;
        D = D*x;
        if(max < D)
            flag=1;
    }
    for(i=0;i<N;i++)
    {
        if(flag==1)
            cout<<"0 ";
        else
            cout<<(a[i]/D)<<" ";
    }
}
