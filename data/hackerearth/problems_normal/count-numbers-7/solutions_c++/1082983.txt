
#include <iostream>
#include<math.h>
using namespace std;
int main()
{
	int q,t;
	long long int a,b,ans,count,arr[10];
	unsigned long long int i,j,prod;
	cin>>q>>t;
	for(j=0;j<q;j++)
	cin>>arr[j];
	while(t--)
	{
		ans=0;
		cin>>a>>b;
		a--;
		for(i=1;i<pow(2,q);i++)
		{
			count=0;
			prod=1;
			for(j=0;j<q;j++)
			{
				if((i>>j)&1)
				{
					count++;
					prod*=arr[j];
				}
			}
			if(prod>b)
			continue;
			if(count%2==1)
			ans+=(b/prod-a/prod);
			else
			ans-=(b/prod-a/prod);
		}
		cout<<ans<<"\n";
	}
    //cout << "Hello World!" << endl;
    return 0;
}
