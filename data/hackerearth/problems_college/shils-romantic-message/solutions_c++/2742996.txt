#include <iostream>
#include <string>

using namespace std;

int main()
{
    int t;
    string s;
    
    cin >> t;
    
    while (t--) {
    	cin >> s;
    	int x = s[0]- 'a';
    	for (int i = 0; i < s.length(); i++) {
    		cout << (char) (((s[i] - 97) + 26 - x) %26 + 97);  
    	}
    	cout << endl;
    }
    return 0;
}
