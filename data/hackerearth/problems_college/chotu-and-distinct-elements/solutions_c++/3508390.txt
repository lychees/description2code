#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t,n,i,j,y,z;
    cin>>t;
    while(t--)
    {   cin>>n;
        int a[n];
    	for(int i=0;i<n;i++)
    	cin>>a[i];
    	sort(a,a+n,greater<int>());
    	i=0;
    	j=n-1;
    	y=a[0]+1;
    	z=a[n-1]-1;
    	//while(a[i]==[i+1]&&i!=n)
    	while(i<j)
    	{
    		h:
    		if(i<j)
    		{
    		if(a[i]<y && a[i]>z)
    		{
    			cout<<a[i]<<" ";
    		}
    		else
    		{
    			i++;
    			goto h;	
    		}
    		}
    		y=a[i];
    		g:
    		if(i<j)
    		{
    		if(a[j]<y && a[j]>z)
    		{
    			cout<<a[j]<<" ";
    		}
    		else
    		{
    			j--;
    			goto g;	
    		}
    		}
    		z=a[j];
    	}
    	cout<<endl;
        
    	
    	
    	
    	
    }
    
    return 0;
}
