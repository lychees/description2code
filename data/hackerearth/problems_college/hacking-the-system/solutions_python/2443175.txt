N1 = int(raw_input())
N2 = int(raw_input())

mod = False
res = "loses"
allequal = True
while(N1>0):
	a = N1 % 10
	b = N2 % 10
	if (a != b):
		allequal = False
		if not(mod):
			mod = True
			res = "wins"
		else:
			res = "loses"
			break
	N1 = N1/10
	N2 = N2/10

if(allequal):
	print("wins")
else:
	print(res)
