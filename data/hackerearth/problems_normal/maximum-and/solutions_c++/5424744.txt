#include <bits/stdc++.h>
using namespace std;
#define ll long long

int main()
{
    ll t,a,b;
    cin>>t;
    while(t--){
    	cin>>a>>b;
    	
    	if(b-a==1){
    		ll x=b & a;
    		cout<<x<<endl;
    	}
    	else{
    		ll temp=b & (b-1);
    		if(temp==0){
    			cout<<b-2<<endl;
    		}
    		else{
    			cout<<max( b&(b-1) , (b-1)&(b-2))<<endl;
    		}
    	}
    }
}
