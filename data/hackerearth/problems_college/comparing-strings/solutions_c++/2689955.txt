#include <iostream>
using namespace std;

int main()
{
    long int t,i,j;
    char a[100000],b[100000];
    cin>>t;
    while(t--)
    {
    	int flag=0;
    	cin>>a>>b;
    	 for(i=0;a[i]!='\0';i++)
    	 {
    	 	for(j=0;b[j]!='\0';j++)
    	 	{
    	 		if(a[i]==b[j])
    	 		{
    	 		 cout<<"YES"<<endl;
    	 		 flag=1;
    	 		 break;
    	 		}
    	 	}
    	 	if(flag==1)
    	 	break;
    	 }
    	 if(flag==0)
    	 cout<<"NO"<<endl;
    }
    return 0;
}
