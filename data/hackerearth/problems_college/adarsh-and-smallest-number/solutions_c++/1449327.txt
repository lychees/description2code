void sieve()
{
    int lim= 500005;
    memset(prime,true,sizeof(prime));
    for(int i=2;i*i<=lim;i++)
    {
        if(prime[i])
        {
            for(int j=i*i;j<=lim;j+=i)
            {
                prime[j]=false;
            }
        }
    }
    for(int i=2;i<=lim;i++)
    {
       if(prime[i])
       {
           LL j=i;
           while(j<lim)
           {
               ma[j]=i;j=j*i;
           }
       }
    }
    ans[1]=1;
    for(int i=2;i<=lim;i++)
    {
        if(ma[i]!=0)
        {
            ans[i]=(ans[i-1]*ma[i])%mod;
        }
        else ans[i]=ans[i-1];
    }
}
