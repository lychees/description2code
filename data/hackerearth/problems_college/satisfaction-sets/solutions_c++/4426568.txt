#include<iostream>
#include<cstdio>
#include<cassert>
#include<set>
#define ll long long int
#define mod 1000000007
using namespace std;
int main()
{
    int t;
    cin>>t;
    ll ar[1005]={0};
    //ll dp[100005]={0};
    set <ll> st;
    for(int i=0;i<t;i++)
    {
        int m;
        cin>>m;
        ll dp[100005]={0};
        for(int j=0;j<m;j++)
        {
            ll a;
            scanf("%lld",&a);
            st.insert(a);
            if(dp[a]!=1)
            {
                   ar[a]++;
                   dp[a]++;
            }
        }
    }
    ll un=0;
    ll in=0;
    set<ll>::iterator it;
    it=st.begin();
    while(it!=st.end())
    {
        un=(un+*it)%mod;
        it++;
    }
    it=st.begin();
    ll as=0;
    for(ll j=0;j<=500;j++)
    {
        if(ar[j]>=t)
        {
            //st.erase(j);
            as=as+j;
        }
    }
    while(it!=st.end())
    {
        in=(in+*it)%mod;
        it++;
    }
    cout<<(in-un-as)%mod<<endl;
    //cout<<f<<endl;
    return 0;
}
