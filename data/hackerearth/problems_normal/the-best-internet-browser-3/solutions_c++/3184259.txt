#include <iostream>
#include <algorithm>
#include <string>
using namespace std;
int main()
{
	int t,sum,len;
	string s;
	cin>>t;
	for(int i=0;i<t;i++)
	{
		sum=0;
		cin>>s;
		len=s.length();
		s=s.substr(4,len-8);
		sum=count(s.begin(),s.end(),'a')+count(s.begin(),s.end(),'e')+count(s.begin(),s.end(),'i')+count(s.begin(),s.end(),'o')+count(s.begin(),s.end(),'u');
		cout<<len-4-sum<<"/"<<len<<endl;
	}
	return 0;
}
