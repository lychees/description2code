#include<bits/stdc++.h>

using namespace std;

long int fib(int a, int b, int n)
{
	int i,ar[n+1];
	ar[0]=a,ar[1]=b;
	for(i=2; i<n+1; i++)
		ar[i]=ar[i-1]+ar[i-2];
		
	return ar[n];
}


int main()
{
	int t,n,i,x,y;
	cin>>t;
	while(t--)
	{
		cin>>n;
		x=fib(1,0,n);
		y=fib(0,1,n);
		
		cout<<x<<" "<<y<<endl; 
	}
	return 0;
}
