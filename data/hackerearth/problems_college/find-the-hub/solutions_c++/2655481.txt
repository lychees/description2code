#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cmath>
using namespace std;
int isprime(int n)
{
	for(int i=2;i<=sqrt(n);i++)
	{
		if(n%i==0)
		{
			return 0;
		}
	}
	return 1;
}

int main()
{
	int n;
	cin>>n;
	int i,num,max=0,count[100000]={0},c;
	for(i=0;i<n;i++)
	{
		cin>>num;
		if(isprime(num))
		{
			count[num]++;
			if(num>=max )
			{
				max = num;
				c = count[num];
			}
		}
	}
	cout<<max<<"\n"<<c<<"\n";

	
}
