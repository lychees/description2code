#include<bits/stdc++.h>
#define max(a,b) (a)>(b)?(a):(b)
using namespace std;
const long double pi=3.141592653589793238;
const long double pi2=1.570296326794896619;
long double r,s,i,maxx=-1.0;
inline long double fun(long double th)
{
    return (pi*s*cos(th)*(r*r+(r+s*sin(th))*(r+s*sin(th))+r*(r+s*sin(th)))/3.0);
}
int main()
{
    long long x;
    cin>>r>>s;
    for(i=0.000000;i<=pi2;i=i+0.0000005)
    {
        maxx=max(maxx,fun(i));
    }
    cout<<fixed;
    cout<<setprecision(2);
    //x=maxx*100;
    //maxx=x/100.0;
    cout<<maxx;
}
