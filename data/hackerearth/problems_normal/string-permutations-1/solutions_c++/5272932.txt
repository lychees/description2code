#include <bits/stdc++.h>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
    int n;
    cin>>n;
    map<string,int> m;
    while(n--)
    {
    	string s;
    	cin>>s;
    	if(m.find(s)==m.end())
    	{
    		m.insert(make_pair(s,1));
    	}
    	else
    	{
    		m[s]+=1;
    	}
    }
    
    map<int,int,greater<int>> m2;
    for(auto i : m)
    {
    	if(m2.find(i.second)==m2.end())
    	m2.insert(make_pair(i.second,1));
    	else
    	m2[i.second]+=1;
    }
    
    auto itr = m2.begin();
    
    cout<<(itr->first)*(itr->second);
    return 0;
}
