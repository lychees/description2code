   #include<bits/stdc++.h>
   using namespace std;
   int n,maze[51][51],visited[51][51];

   int safe(int x , int y)
     {
          if(x>=0 && x<n && y>=0 && y<n && maze[x][y]==1 && visited[x][y]==0)
          return 1;
          return 0;
     }

   int path( int x  , int y)
    {
       if(x==n-1 && y==n-1) return 1;
       else
            {
              if(safe(x,y))
                  {
                     visited[x][y] = 1 ;
                     if(path(x+1,y)) return 1;
                     if(path(x,y+1)) return 1;
                  }
              return 0;
            }

    }





   int main()
   {
       int t; cin>>t; while(t--)
       {
              int i , j ;
              cin>>n;
              for(i=0;i<n;i++) for(j=0;j<n;j++) maze[i][j]=0,visited[i][j];
              for(i=0;i<n;i++) for(j=0;j<n;j++) cin>>maze[i][j];
              if(path(0,0))
              cout<<"POSSIBLE\n";
              else cout<<"NOT POSSIBLE\n";
       }
    return 0;
   }
