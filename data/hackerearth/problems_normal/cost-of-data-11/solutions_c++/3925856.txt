#include <bits/stdc++.h>
using namespace std;
struct Trie
{
	Trie* child[26];
	bool marker;
};
Trie* getTrie()
{
	Trie* trie=new Trie;
	trie->marker=false;
	for(int i=0;i<26;i++)
		trie->child[i]=NULL;
	return trie;
}
void insert(Trie* trie,string s,int& cnt)
{
	Trie* curr=trie;
	for(string::iterator it=s.begin();it!=s.end();it++)
	{
		if(curr->child[(*it)-'a']==NULL)
		{
			curr->child[(*it)-'a']=getTrie();
			cnt++;
		}
		curr=curr->child[(*it)-'a'];
	}
	curr->marker=true;
}
int main()
{
	int n;
	cin>>n;string s;vector<string> v;
	for(int i=0;i<n;i++)
	{
		cin>>s;
		v.push_back(s);
	}
	Trie* trie=getTrie();
	int cnt=1;
	for(int i=0;i<n;i++)
		insert(trie,v[i],cnt);
	cout<<cnt;
	return 0;
}
