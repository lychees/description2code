typedef long long ll;

bool vis [10000 + 10];
int pri [10000 + 10];
int siz = 0;

int N = 10000 + 10;

void sieve(){

    for(long long i = 2; i * i < N; i++){
        if(!vis[i]){
            for( long long j = i * i; j < N; j+= i){
                vis[j] = true;
            }
        }
    }
    for(int i = 2; i < N; i++){
        if(!vis[i]) pri[siz++] = i;
    }

}

int main() {

    ios::sync_with_stdio(false);cin.tie(0);

    sieve();
    vis[0] = vis[1] = true;
	int t; scan(t);

	while(t--)
	{
		int n; scan(n); int arr[n];
		for (int i=0; i<n; i++)
		{
			scan(arr[i]);
		}
		set<int> st; set<int> :: iterator itr;
		st.insert(0);
		for (int i=0; i<n; i++)
		{
			for (itr=st.begin(); itr!=st.end(); ++itr)
			{

				int res=arr[i]^(*itr);

				st.insert(res);
			}
			st.insert(arr[i]);
		}
		int cnt=0;
		for (itr=st.begin(); itr!=st.end(); ++itr)
		{
			if (!vis[*itr])
			{
				cnt+=1;
			}
		}
		cout<<cnt<< "\n";


	}
	return 0;
}
