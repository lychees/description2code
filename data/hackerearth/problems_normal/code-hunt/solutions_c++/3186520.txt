#include <bits/stdc++.h>
using namespace std;

#define MOD 1000000007LL
#define ll long long
#define sd(x) scanf("%d",&x)
#define sll(x) scanf("%lld",&x)
#define MAXN 105
#define MAXV 100010
vector<int> d[MAXN];
ll cnt[MAXN];
int A[MAXV];

int main()
{ 
	for(int i=1;i<MAXN;++i)
		for(int j=i;j<MAXN;j+=i)
			d[j].push_back(i);
	int t;
	sd(t);
	while(t--)
	{
		memset(cnt,0,sizeof cnt);
		int n;
		sd(n);
		for(int i=0;i<n;++i)
		{
			sd(A[i]);
			++cnt[A[i]];
		}
		ll res = 0;
		for(int i=0;i<n;++i)
		{
			--cnt[A[i]];
			for(int j=0;j<d[A[i]].size();++j)
				res += cnt[d[A[i]][j]];
		}
		cout << res << endl;
	}
	return 0;
}
