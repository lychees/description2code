#include <iostream>

using namespace std;

int findNumWhite(int years, int age)
{
	if(years == 0)
		return 1;

	years--;

	if(age == 0)
		return findNumWhite(years, 1);

	if(age == 1)
	return (findNumWhite(years, 0)+findNumWhite(years, 2));

	if(age == 2)
		return findNumWhite(years, 0);

	return 0;
}

int main() {
	int i;
	std::cin>>i;
	std::cout<<findNumWhite(i, 0);
	return 0;
}
