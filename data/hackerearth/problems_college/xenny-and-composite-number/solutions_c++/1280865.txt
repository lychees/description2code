#include <iostream>
#include<cmath>
 
using namespace std;
 
 
bool isPrime(long long int number){
 
    if(number < 2) 
    return false;
    if(number == 2) 
    return true;
    if(number % 2 == 0)
    return false;
    for(int i=3; (i*i)<=number; i+=2)
    {
        if(number % i == 0 )
           return false;
    }
    return true;
 
}
 
 
int main()
{
    long long int t,n,flag=1,i;
    cin>>t;
    while(t--)
    { flag=1;
    	cin>>n;
    	if(n==1||n==0)
    	cout<<"4"<<endl;
    	else if(isPrime(n+1))
    	cout<<n+2<<endl;
    	else
    	cout<<n+1<<endl;
    }
    return 0;
}
