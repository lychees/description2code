#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    vector<int> arr;
    int N;
    int num;
    cin>>N;
    for(int i=0;i<N;i++){
    	cin>>num;
    	arr.push_back(num);
    }
    sort(arr.begin(), arr.end());
    int ref;
    int units=0;
    for(int i=0;i<N;i++){
    	ref=arr[i];
    	units++;
    	i++;
    	while(i<N && arr[i]<=(ref+4)){
    		i++;
    	}
    	i--;
    }
    cout<<units;
    return 0;
}
