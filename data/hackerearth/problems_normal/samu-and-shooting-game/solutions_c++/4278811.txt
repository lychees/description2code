#include <bits/stdc++.h>
using namespace std;
int x,y,n,w;
long double p1,p2;
long double dp[1001][1001];
long double win(int coins,int points)
{
	//cout<<coins<<" "<<points<<endl;
	if(coins==0)
		return (points>=w);
	if(points>w)points=w;
	//cout<<dp[coins][points]<<endl;
	if(dp[coins][points]!=-1)return dp[coins][points];
	//cout<<"FUCK"<<endl;
	long double ans1=p1*win(coins-1,points+x)+(1-p1)*win(coins-1,points);
	long double ans2=p2*win(coins-1,points+y)+(1-p2)*win(coins-1,points);
	return dp[coins][points]=max(ans1,ans2);
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	cin>>x>>y>>n>>w>>p1>>p2;
    	p1/=100.0;
    	p2/=100.0;
    	for(int i=0;i<=1000;i++)
    		for(int j=0;j<=1000;j++)
    			dp[i][j]=-1;
    	//memset(dp,-1,sizeof(dp));
    	cout<<fixed<<setprecision(6)<<(100.00)*win(n,0)<<"\n";
    }
    return 0;
}
