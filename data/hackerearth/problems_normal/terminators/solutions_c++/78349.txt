#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
	int t,n,i,x,y,j;
	cin >> t;
	while ( t-- ) {
		cin >> n;
		pair <int,int> a[1002],p;
		for ( i = 0; i < n; i++ ) {
			cin >> a[i].first >> a[i].second;
		}
		sort(a,a+n);
		int dp[1002],mx=1;
		dp[0] = 1;
		for ( i = 1; i < n; i++ ) {
			dp[i] = 1;
			for ( j = 0; j < i; j++ ) {
				if ( a[i].second > a[j].second ) {
					dp[i] = max(dp[i],dp[j]+1);
				}
			}
			mx = max(mx,dp[i]);
		}
		cout << mx << endl;
	}
    return 0;
}
