#include <iostream>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    	string s;
    	cin>>s;
    	if(s[0]!='R') puts("NO");
    	else {
    		int ex=1;
    		int n=s.length();
    		int y=0;
    		for(int i=0;i<n && ex;i++){
    			if(s[i]!='R' && s[i]!='Y') ex=0;
    			else if(s[i]=='Y') {
    				if(y>1) ex=0;
    				else y++;
    			}
    			else y=0;
    			
    		}
    		if(ex==1) puts("YES");
    		else puts("NO");
    	}
    }
    return 0;
}
