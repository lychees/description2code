def issubstr(substr, mystr, start_index=0):
    for letter in substr:
        start_index = mystr.find(letter, start_index) + 1
        if start_index == 0: return False
    return True

st = raw_input()
n = int(raw_input())

for i in xrange(n):
	sub = raw_input()
	if issubstr(sub, st):
		print "Yes"
	else:
		print "No"
