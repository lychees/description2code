#include <iostream>
#include <vector>

using namespace std;

int main() {
	int T;
	cin >> T;
	while(T--) {
		vector<int> cost(3,0);
		int N,pink,orange,yellow;
		cin >> N;
		while(N--) {
			vector<int> tmp(3);
			cin >> pink >> orange >> yellow;
			tmp[0] = min(cost[1],cost[2]) + pink;
			tmp[1] = min(cost[0],cost[2]) + orange;
			tmp[2] = min(cost[0],cost[1]) + yellow;
			cost = tmp;
		}
		cout << min(min(cost[0],cost[1]),cost[2]) << endl;
	}
}
