#include<iostream>

using namespace std;

int main()
{
    long int t;
    cin>>t;

    while(t--)
    {
        int m,n;
        cin>>m;

       if(m<=31)
         n=1;
       else if(m<=59)
        n=2;
       else if(m<=90)
       n=3;
       else if(m<=120)
        n=4;
        
        else if(m<=151)
        n=5;
        
        else if(m<=181)
        n=6;
        
        else if(m<=212)
        n=7;
        
        else if(m<=243)
        n=8;
        
        else if(m<=273)
        n=9;
        
        else if(m<=304)
        n=10;
        
        else if(m<=334)
        n=11;
        
        else 
        n=12;
        
        cout<<n<<endl;
    }
    
    

}
