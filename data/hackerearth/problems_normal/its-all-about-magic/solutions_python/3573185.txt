testcases = int(raw_input())

for i in range(0, testcases):
    n = int(raw_input())
    numberof0 = 1
    numberof1 = 0
    
    if n == 1:
        print 0,1
    else:
        for j in range(1,n):
            previousnumberof1 = numberof1
            previousnumberof0 = numberof0
            numberof0 = numberof0 + previousnumberof1
            numberof1 = previousnumberof0
        print numberof1,numberof0
