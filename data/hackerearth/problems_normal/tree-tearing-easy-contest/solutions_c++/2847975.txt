using namespace std;
vector<int>v[10005];
bool h[10005];
int ans;
int n,k;
int rec(int node);
int main()
{
    int i, j,d;
    ans = 0;
    S(k);S(n);
    for (i = 2; i <= n; i++){
        S(d);
        v[i].push_back(d);
        v[d].push_back(i);
    }
    //h[1] = 1;
    rec(1);
    cout << ans << endl;
}
int rec(int node)
{
    int i,j,ret = 1;
    h[node] = 1;

    for(i = 0; i < v[node].size(); ++i) {
        if(!h[v[node][i]])
            ret += rec(v[node][i]);
    }

    if(ret >= k) {
        ans++;
        ret = 0;
    }
    return ret;
}
