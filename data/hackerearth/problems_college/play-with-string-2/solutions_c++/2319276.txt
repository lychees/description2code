#include<iostream>
#include<algorithm>
#include<cmath>
#include<vector>
#include<string.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    string s,s1="";
    cin>>s;
    long long int len,i,x=0;
    len=s.length();
    for(i=len-1;i>=0;i--)
    {
        s1+=s[i];
    }
    for(i=1;i<len;i++)
    {
        if(abs(s[i]-s[i-1])==abs(s1[i]-s1[i-1]))x++;
    }
    if(x==len-1)cout<<"Funny";
    else cout<<"Not Funny";
    cout<<endl;
    }
    return 0;
}
