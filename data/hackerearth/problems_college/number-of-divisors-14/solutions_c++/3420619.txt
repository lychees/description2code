#include<iostream>
#include<cstdio>
#include<cmath>
using namespace std;

int main()
{
    long long t,n,sq,cnt,ans;
    scanf("%lld", &t);
    while(t--){
        ans = 1;
        scanf("%lld", &n);
        for(long long i=2; i<=1000002; ++i){
            if(n%i==0){
                cnt = 0;
                while(n%i==0){
                    n = n/i;
                    cnt++;
                }
                ans*=cnt+1; //since no. of divisor = (p1+1)(p2+1)(p3+1)...
            }

        }
        printf("%lld\n", ans);
    }
    return 0;
}
