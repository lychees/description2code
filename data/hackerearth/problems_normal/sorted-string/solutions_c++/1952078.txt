#include<stdio.h>
int main()
{
	int t,hash[256]={0},i,j,temp;
	char a[100];
	scanf("%d",&t);
	while(t--)
	{
		scanf("%s",a);
		for(i=0;a[i]!='\0';i++)
		{
			hash[a[i]]++;
		}
		for(i=1;i<=100;i++)
		{
			for(j=0;j<256;j++)
			{
				if(hash[j]==i){
					temp=i;
					while(temp--)
					printf("%c",j);
				}
					
			}
		}
		printf("\n");
		for(i=0;i<256;i++)
			hash[i]=0;
	}
	
	return 0;
}
