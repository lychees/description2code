#include <bits/stdc++.h>
using namespace std;
struct st{
	int le,ri;
	long long ra;	
};
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,m,l,r;
    	long long c;
    	cin>>n>>m;
    	st func[m+1];
    	for(int i=1;i<=m;i++)
    	{
    		cin>>l>>r>>c;
    		func[i].le=l;
    		func[i].ri=r;
    		func[i].ra=c;
    	}
    	int q;
    	cin>>q;
    	int countf[100005]={0};
    	while(q--)
    	{
    		int x,y;
    		cin>>x>>y;
    		countf[x]++;
    		countf[y+1]--;
    	}
    	int times[m+1];
    	times[0]=0;
    	for(int i=1;i<=m;i++)
    	{
    		times[i]=times[i-1]+countf[i];
    	}
    	long long rat[100005]={0};
  		
    	for(int i=1;i<=m;i++)
    	{
    		rat[func[i].le]+= times[i]*func[i].ra;
    		rat[func[i].ri+1]-= times[i]*func[i].ra;
    	}
    	long long ans[n+1];
    	ans[0]=0;
    	for(int i=1;i<=n;i++)
    		ans[i]=ans[i-1]+rat[i];
    	for(int i=1;i<=n;i++)
    		cout<<ans[i]<<" ";
    	cout<<endl;
    	
    	/*int q;
    	cin>>q;
    	long long rat[1002]={0};

    	{
    		int x,y;
    		cin>>x>>y;
    		for(int i=x-1;i<y;i++)
    		{

    				rat[j]+=func[i].ra;
    		}
    	}
    	for(int i=0;i<n;i++)
    		cout<<rat[i]<<" ";
    	cout<<endl;*/
    }
    return 0;
}
