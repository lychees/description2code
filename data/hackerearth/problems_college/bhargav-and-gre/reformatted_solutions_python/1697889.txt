arr = ['a', 'b', 'c', 'd', 'e', 'f', 'g', 'h', 'i', 'j', 'k', 'l', 'm',
       'n', 'o', 'p', 'q', 'r', 's', 't', 'u', 'v', 'w', 'x', 'y', 'z']
w = raw_input()
ant, syn = 0, 0
n = input()
for i in range(n):
    s = raw_input()
    if (len(s) == len(w)):
        if (s[0] == w[0] and s != w):
            syn += 1
        elif (arr[((arr.index(w[0])+3) % 26)] == s[0]):
            ant += 1
print str(syn) + " " + str(ant)
