#include <iostream>
using namespace std;

void quick_sort(int *niz, int ind_od, int ind_do) {
    if (ind_od < ind_do) {
        int a = ind_od;
        int b = ind_do;
        int element = niz[a];
        while (1) {
            while (a < ind_do && niz[++a] > element);
            while (b > ind_od && niz[--b] < element);
            if (a < b) {
                int temp = niz[a];
                niz[a] = niz[b];
                niz[b] = temp;
            }
            else break;
        }
        int temp = niz[ind_od];
        niz[ind_od] = niz[b];
        niz[b] = temp;
        quick_sort(niz, ind_od, b);
        quick_sort(niz, b + 1, ind_do);
    }
}

int main() {
	int T;
	cin >> T; cin.ignore();
	while (T--) {
		int N;
		cin >> N; cin.ignore();
		int A[N];
		for (int i = 0; i < N; ++i) {
			cin >> A[i]; cin.ignore();
		}
		quick_sort(A, 0, N);
		for (int i = 0; i < N; ++i) {
			cout << A[i] << " ";
		}
		cout << endl;
	}
}
