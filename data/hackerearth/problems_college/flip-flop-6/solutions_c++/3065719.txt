#include <iostream>
#include<string>
using namespace std;

int main()
{
	int tc;
	string str;
	cin>>tc;
	while(tc--)
	{
		int cnt=0,j=0,cur=0;
		cin>>str;
		for(int i=1;i<str.length();)
		{
			if(str[j]==str[i])
			{
				cur++;
				cnt++;
				i++;
			}
			else
			{
				j+=(cur+1);
				i++;
				cur=0;
			}
		}
		cout << cnt << endl;
	}
    return 0;
}
