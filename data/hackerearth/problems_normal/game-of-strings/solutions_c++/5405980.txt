#include<iostream>
#include<cstring>
#define size 5005
using namespace std;

int cost[2][size];

char A[size], B[size];

int max(int a, int b)
{
	if (a>b)
		return a;
	return b;
}

bool notvowel(char cc)
{
	if (cc == 'a' || cc == 'e' || cc == 'i' || cc == 'o' || cc == 'u')
		return false;
	return true;
}

int main()
{
	int i, j, k, a, b, curr, prev;
	scanf("%s %s", A, B);
	a = strlen(A);
	b = strlen(B);

	for (i = 0; i<b; i++)
		cost[0][i] = 0;

	for (i = 1; i <= a; i++)
	{
		curr = i % 2;
		prev = !curr;
		cost[curr][0] = 0;
		if (notvowel(A[i - 1]))
		{
			for (j = 1; j <= b; j++)
				if (A[i - 1] == B[j - 1])
					cost[curr][j] = max(cost[prev][j - 1] + 1, max(cost[curr][j - 1], cost[prev][j]));
				else
					cost[curr][j] = max(cost[curr][j - 1], cost[prev][j]);
		}
		else
		{
			for (j = 1; j <= b; j++)
				cost[curr][j] = cost[prev][j];
		}
	}
	printf("%d\n", cost[!(i % 2)][b]);
	return 0;
}
