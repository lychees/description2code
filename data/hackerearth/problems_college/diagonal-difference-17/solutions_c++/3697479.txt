#include<iostream>
#include<cmath>
using namespace std;
main()

{
    int a[100][100],i,j,k,t,sum=0,sum1=0;
    cin>>t;
    for(i=0;i<t;i++)
    {
        for(j=0;j<t;j++)
        {
            cin>>a[i][j];
            if(i==j)
            {
                sum=sum+a[i][j];
            }
        }

    }
   // cout<<sum<<endl;
    i=0;
    j=(t-1);
    while(i!=t)
    {
        sum1=sum1+a[i][j];
        i++;
        j--;
    }
    cout<<abs(sum1-sum)<<endl;

}
