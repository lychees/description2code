#include <iostream>
#include <iomanip>
#include <vector>
#include <string.h>

using namespace std;
 
int main()
{
   int T;
   char S[100001];
   
   cin >> T;
   
   while(T--)
   {
   	int army = 0, enemy = 0, survivors = 0;
    cin >> S;
    
   	for(int i=0; i<strlen(S) ;i++)
   	{
   		
   		if(S[i]=='X')army++;
   		if(S[i]=='O')enemy++;
   		if(S[i]=='*')
   		{
   			if(army==0)
   			survivors += enemy;
   			army=0;
   			enemy=0;
   		}
   	}
   	if(army==0)
   	survivors+=enemy;
   	cout<<survivors<<endl;
   }
return 0; }
