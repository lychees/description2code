#include<bits/stdc++.h>
using namespace std;
set<int> st;
vector<int> vt;
int main()
{
	int t; cin>>t;
	while(t--)
	{
		int n; int i; int sz;
		vt.clear(); st.clear();
		cin>>n;
		if(n == 1)
		{ cout<<0<<endl; continue; }
		int a,b,c; a=b=c=-1;
		int ans = 0;
		vt.push_back(n); st.insert(n);
		int flag = 0;
		i=0;
		a = b = c = -1;
		while(1)
		{
			sz = vt.size();
			for(;i<sz;++i)
			{
				
				if(vt[i]%2==0)
				{
					a = vt[i]/2;
					if(st.find(a) == st.end())
					{
						st.insert(a);
						vt.push_back(a);
					}
				}
				if(vt[i]%3==0)
				{
					b = vt[i]/3;
					if(st.find(b) == st.end())
					{
						st.insert(b); vt.push_back(b);
					}
				}
				c = vt[i]-1;
				if(c > 0 && st.find(c) == st.end())
				{
					st.insert(c); vt.push_back(c);
				}
				if(a == 1 || b == 1 || c == 1)
				{
					flag=1;
					break;
				}
				
			}
			ans++;
			if(flag == 1) break;
			
		}
		cout<<ans<<endl;
	}
}
