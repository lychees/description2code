#include <iostream>
using namespace std;

int a[1<<10 + 1][1<<10 + 1];

int ans(int first, int last){
	if(first == last){
		return first;
	}
	int mid = (first + last)/2;
	int win1 = ans(first, mid);
	int win2 = ans(mid+1, last);
	if(a[win2][win1] == 1){
		return win2;
	}else if(a[win2][win1] == 0){
		return win1;
	}
}

int main()
{
    int n,p;
    cin >> n;
    p = 1<<n;
    for(int i = 2; i <= p; i++){
    	for(int j = 1; j < i; j++){
    		cin >> a[i][j];
    	}
    }
    cout << ans(1, p) << endl;
    return 0;
}
