#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
using namespace std;
typedef long long ll;
int gcd(int a,int b) {
    return b==0?a:gcd(b,a%b);
}
const int nmax=1e6+5;
int a[nmax];
int main() {
    int t;
    scanf("%d",&t);
    while(t--) {
        int n;
        scanf("%d",&n);
        for(int i=0;i<n;i++) scanf("%d",&a[i]);
        int val=a[0];
        for(int i=1;i<n;i++) val=gcd(val,a[i]);
        printf("%lld\012",ll(val)*n);
    }
    return 0;
}
