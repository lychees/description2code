#include <iostream>
using namespace std;

int main()
{
   int t;
   cin >> t;
   while(t--)
   {
    int n,k;
   	cin >> n >> k;
   	int arr[n];
   	int i;
   	for(i=0;i<n;i++)cin >> arr[i];
   	i=0;long long int max=0,sum=0;
   	while(i<n)
   	{sum=0;
   	while(arr[i]>k&&i<n)
   	i++;
   		while(arr[i]<=k&&i<n)
   		{
   			sum+=arr[i];
   			i++;
   		}
   		
   		if(sum>max)
   		max=sum;
   		
   		
   	}
   	cout << max << endl;
   }
    return 0;
}
