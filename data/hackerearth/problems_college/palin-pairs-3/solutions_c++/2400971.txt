#include <iostream>
#include <string>
#include <algorithm>
#include <map>
using namespace std;

int main()
{
    
    int n;
    cin>>n;
    string st;
    int i = 0;
    long counter = 0;
    map<string,int> mp;
    for(i;i<n;++i)
    {
    	cin>>st;
		++mp[st];
		reverse(st.begin(),st.end());
		counter +=mp[st];
    	
    }
    cout<<counter<<endl;
    
    return 0;
}
