#include <iostream>
#include<string.h>
using namespace std;

int main()
{
	int t;
	char str1[1000000];
	char str2[1000000];
	cin>>t;
	while(t--)
	{
		int flag=0;
		cin>>str1;
		cin>>str2;
		int a=strlen(str1);
		int b=strlen(str2);
		if(strcmp(str1,str2))
		{
			
			if(a>b)
			{
				for(int i=0;i<b;i++)
				{
					if(str1[i]!=str2[i])
					{
						flag++;
					}
				}
				flag=flag+(a-b);
			}
			else
			{
				for(int i=0;i<a;i++)
				{
					if(str1[i]!=str2[i])
					{
						flag++;
					}
				}
				flag=flag+(b-a);
			}
			cout<<"Non Exact "<<flag<<endl;;
		}
		else
		cout<<"Exact"<<endl;
	}
	
}
