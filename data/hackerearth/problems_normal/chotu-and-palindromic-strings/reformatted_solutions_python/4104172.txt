def palindrome(a):
    length = len(a)
    iterator = 0
    # original_length=length-1
    while iterator < length:
        if a[iterator] == a[length-iterator-1]:
            pass
        else:
            return False
        iterator = iterator+1
        if iterator == length:
            return True


for _ in range(input()):
    a = raw_input()
    b = raw_input()
    c = a+b
    d = b+a
    if palindrome(c):
        print "YES"
    elif palindrome(d):
        print "YES"
    else:
        print "NO"
