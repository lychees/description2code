#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,a[26]={};
    string s,ss;
    cin>>t;
    while(t--)
    {
        int flag=0,a[26]={};
        cin>>s>>ss;
        for(int i=0;i<s.size();i++)
            a[s[i]-'a']=1;
        for(int i=0;i<ss.size();i++)
        {
            if(a[ss[i]-'a']==1)
            {
                flag=1;
                break;
            }
        }
        if(flag==1)
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
    }
}
