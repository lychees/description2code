#include <bits/stdc++.h>
using namespace std;
bool check(string s){
	int i, j;
	int len = s.size();
	j = len - 1;
	for(i=0; i<len; i++)
	{
			if(s[i]!=s[j])
				return 0;
			j--;
	}
	return 1;
}
int main()
{
    int t;
    cin>>t;
    while(t--){
    	string s1, s2;
    	cin>>s1>>s2;
    	if(check(s1+s2)==1 || check(s2+s1)==1)
    		cout<<"YES"<<endl;
    	else
    		cout<<"NO"<<endl;
    }
    return 0;
}
