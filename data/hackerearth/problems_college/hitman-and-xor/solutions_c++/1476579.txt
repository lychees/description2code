#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t,n,i,j,c1,c2;
	scanf("%d",&t);
	while (t --) {
		c1 = c2 = 0;
		scanf("%d",&n);
		for (i = 0; i < n; i++) {
			scanf("%d",&j);
			if (j % 2) {
				c1++;
			} else {
				c2++;
			}
		}
		printf("%lld\n",(long long)c1*c2);
	}
	
	return 0;
}
