#include<stdio.h>
#include<string.h>
#include<iostream>
using namespace std;
int dp[59][59];
string str;
int solve(int i,int j)
{
	if(j<i)
	return 0;
	if(dp[i][j]!=-1)
	return dp[i][j];
	//printf("...%d %d %c %c\n",i,j,str[i],str[j]);
	if(i==j)
	return dp[i][j]=1;
	if(j-i==1 && str[i]==str[j])
	return dp[i][j]=1;
	if(str[i]==str[j])
	return dp[i][j]=solve(i+1,j-1);
	else if(str[i]!=str[j])
	return dp[i][j]=0;
	return dp[i][j];
}
int main()
{
	cin>>str;
	memset(dp,-1,sizeof(dp));
	int n=str.size();
	int ans=0;
	//solve(0,2);
	for(int i=0;i<n;i++)
	{
		for(int  j=i;j<n;j++)
		{
			solve(i,j);
			if(dp[i][j]==1)
			ans++;
			//printf("---%d %d %d\n",i,j,dp[i][j]);
		}
	}
	printf("%d\n",ans);
	return 0;
}
