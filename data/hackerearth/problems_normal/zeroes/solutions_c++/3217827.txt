#include<bits/stdc++.h>
#define ll int
#define mod 1000000007
using namespace std;
int temp[100005];
ll arr[100005];
void sieve(ll n)
{
	for(ll i=2;i<=n;i++)
	arr[i]=0;
	
	for(ll i=2;i*i<=n;i++)
	{
		if(arr[i]==0)
		{
			for(ll j=i*2;j<=n;j=j+i)
			{
				if(arr[j]==0)
				arr[j]=i;
			}
		}
	}
	
	for(ll i=2;i<=n;i++)
	{
		if(arr[i]==0)
		arr[i]=i;
	}
}

int main()
{
	sieve(100005);
	int n;
	cin>>n;
	int z[n+5];
	for(int i=0;i<n;i++)
	cin>>z[i];
	
	for(int i=0;i<n;i++)
	{
		int q=z[i];
		while(q!=1)
		{
			temp[arr[q]]++;
			q=q/arr[q];
		}
	}

	int qu;
	cin>>qu;
	while(qu--)
	{
		int x;
		cin>>x;
		int ans=INT_MAX;
		while(x!=1)
		{
			int p=arr[x];
			int count=0;
			while(x%p==0)
			{
				x=x/p;
				count++;
			}
		//	cout<<count<<endl;
			ans=min(ans,temp[p]/count);
		}
		cout<<ans<<endl;
	}

	return 0;
}
