#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,i;
    cin >> t;
    
    while(t--)
    {
    	long long n;
    	long long k;
    	cin >> n;
    	k = n;
    	long long c = 0;
    	long long sum = 0;
    	
    	while(k!=0)
    	{
    		c++;
    		k /= 10;
    		
    	}
    	
    	for(i=1;i<c;i++)
    	{
    		sum += (pow(10,i) - 1);
    	}
    	
    	cout <<  (c*n) - sum<< endl;
    }
    return 0;
}
