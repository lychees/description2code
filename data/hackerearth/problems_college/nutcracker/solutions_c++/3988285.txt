#include <iostream>
using namespace std;

int main()
{
    int arr[10004];
    int t,n;
    
    cin>>t;
    arr[1] = 1;
    arr[2] = 0;
    arr[3] = 1;
    arr[4] = 1;
    
    for(int i = 5;i <= 10000;i++){
    	arr[i] = (1-arr[i-4])||(1-arr[i-1]);	
    }
    while(t--){
    	cin>>n;
    	if(arr[n] == 1){
    		cout<<"Yes"<<endl;
    	}
    	else{
    		cout<<"No"<<endl;
    	}
    }
}
