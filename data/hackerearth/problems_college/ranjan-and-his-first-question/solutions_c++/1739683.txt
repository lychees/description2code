#include<iostream>
using namespace std;
int main()
{
    long long t,n,m,k,sum;
    cin>>t;
    while(t--)
    {
        cin>>n>>m;
        sum=((m-1)*(m))/2;
        k=n/m;
        sum=sum*k;
        k=n%m;
        sum=sum+(k*(k+1)/2);
        cout<<sum<<"\n";
    }
    return(0);
}
