#include <iostream>
#include <string>
#include <map>
using namespace std;

int main()
{
    string s;
    cin>>s;
    map<string,string> m;
    int n=s.size();
    int i=s.find('?'),j,k;
    {
    //key
    j=s.find("key",i+1);
    k=s.find('=',j+1);
    m["key"]=s.substr(k+1,n-k-1);
    n=j-1;
    //role
    j=s.find("role",i+1);
    k=s.find('=',j+1);
    m["role"]=s.substr(k+1,n-k-1);
    n=j-1;
    //profile
    j=s.find("profile",i+1);
    k=s.find('=',j+1);
    m["profile"]=s.substr(k+1,n-k-1);
    n=j-1;
    //pwd
    j=s.find("pwd",i+1);
    k=s.find('=',j+1);
    m["pwd"]=s.substr(k+1,n-k-1);
    n=j-1;
    //username
    j=s.find("username",i+1);
    k=s.find('=',j+1);
    m["username"]=s.substr(k+1,n-k-1);
    }
    {
    	cout<<"username: "<<m["username"]<<endl;
    	cout<<"pwd: "<<m["pwd"]<<endl;
    	cout<<"profile: "<<m["profile"]<<endl;
    	cout<<"role: "<<m["role"]<<endl;
    	cout<<"key: "<<m["key"]<<endl;
    }
}
