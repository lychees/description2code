#include <iostream>
#include <stdio.h>
#include <string.h>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(1)
   {
   		char ch[1000000];
   		int i;
   		for(i = 0;;i++){
   		scanf("%c",&ch[i]);
   		if(i>3){
   		if(ch[i] == '\n')
        {
            break;
        }
   		}
   		}
   		for(i = 0; i<strlen(ch);i++)
   		{
   		    if(ch[i] == '\0')
            {
                break;
            }
   			if(ch[i] != ch[i+1])
   			{
   			    if(ch[i] == '\n' || ch[i] == '0')
   			    {
   			        continue;
   			    }
   				cout<<ch[i];
   			}
   		}
   		for(i = 0; i<strlen(ch);i++)
        {
            ch[i] = '0';
        }
   		cout<<endl;
   		t--;
       if(t == 0)
       {
           break;
       }

   }
    return 0;
}
