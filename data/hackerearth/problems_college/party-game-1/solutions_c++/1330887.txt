#include<bits/stdc++.h>
using namespace std;
int main()
{
    int i,j,k,n,a,b,t;
    scanf("%d",&t);
    while(t--){
    scanf("%d%d%d",&a,&b,&n);
    int arr[n];
    for(i=0;i<n;i++)
    {
        scanf("%d",&arr[i]);
    }

    sort(arr,arr+n);

    int count=0;
    int sum=0;

    for(i=0;i<n;i++)
    {
        //sum=sum+arr[i];
            sum=sum+arr[i];
            if(a>=sum){
            count++;
            }
    }

    int c=0;
    int s=0;

    for(j=0;j<n;j++)
    {
            s=s+arr[j];
            if(b>=s){
            c++;
            }

    }

    if(count==c)
    printf("Tie\n");
    else if(count>c)
        printf("Chef Won\n");
    else
        printf("Jhool Won\n");
    }
    return 0;
}
