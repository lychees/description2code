#include<stdio.h>
#include<iostream>
#include<vector>
#include<cmath>
#include <fstream>
#include <string>
#include<stack>
using namespace std;
stack<int> s;
long long power(int a,int b)
{
    int i;
    long long ans=1;
    for(i=1;i<=b;i++)
    ans*=a;
    return ans;
}
main()
{
    long long int i,j,k,n,m,t,ans,u,flag;
    int d;
    cin>>t;
    while(t--)
    {
        flag=0;
        ans=0;
        d=0;
        scanf("%lld",&n);
        m=n;
        while(n!=0)
        {
            j=n%10;
            s.push(j);
            n/=10;
            d++;
        }
        //ans+=power(6,d-1);
        for(i=1;i<=d;i++)
        {
            u=s.top();
            s.pop();
            k=power(6,d-i);

            for(j=0;j<u;j++)
            {
                if(j==2||j==5||j==7||j==3)
                continue;
                ans+=k;

            }
            if(u==2||u==5||u==3||u==7)
            {
                flag=1;
                break;

            }

        }
        //cout<<m-ans+flag<<endl;
        printf("%lld\n",m-ans+flag);

    }
    return 0;
}
