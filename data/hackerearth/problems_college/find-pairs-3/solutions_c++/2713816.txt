#include <bits/stdc++.h>
using namespace std;
int main() {
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int a[n+1];
		long long res=0,count=1,i;
		for(i=0;i<n;i++)
			cin>>a[i];
		sort(a,a+n);
		for(i=1;i<n;i++)
		{
			if(a[i-1]==a[i])
				count++;
			else
			{
				res=res+((count*(count+1))/2);
				count=1;
			}
		}
		res=res+((count*(count+1))/2);
		cout<<res<<endl;
	}
	return 0;
}
