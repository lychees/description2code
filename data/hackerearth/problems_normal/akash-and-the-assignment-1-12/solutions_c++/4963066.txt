#include <iostream>
using namespace std;

int main()
{
    int n,t;
    cin>>n>>t;
    string a;
    cin>>a;
    int temp;
    int frq[n+1][26];
    for(int i=0;i<26;i++)
    {
    	frq[0][i]=0;
    }
    for(int i=1;i<=n;i++)
    {
    	temp=(int)a[i-1]-97;
    	for(int j=0;j<26;j++)
    	{
    		if(temp!=j)
    			frq[i][j]=frq[i-1][j];
    		
    		else
    			frq[i][j]=frq[i-1][j]+1;
    	}
    }
    
    while(t--)
    {
    	int l,r,k;
    	cin>>l>>r>>k;
    	int sum[26]={0};
    	for(int i=0;i<26;i++)
    	{
    		sum[i]=(frq[r][i]-frq[l-1][i]);
    		
    	}
    	int total=0,i;
    	for( i=0;i<26;i++)
    	{
    		total+=sum[i];
    		if(total>=k)
    			break;
    	}
    	
    	if(total<k)
    		cout<<"Out of range"<<endl;
    	else
    		cout<<(char)(i+97)<<endl;
    }
    return 0;
}
