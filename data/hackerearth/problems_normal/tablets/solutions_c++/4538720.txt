#include <iostream>
using namespace std;
int N,ans,arr[100005],dp[100005];
 
int main()
{
    cin>>N;
	int i,j;
	for(i=1;i<=N;i++)cin>>arr[i];
	dp[1]=1;ans=0;
	for(i=2;i<=N;i++)
	{
		if(arr[i]>arr[i-1])dp[i]=dp[i-1]+1;
		else
		dp[i]=1;
	}
	
	for(i=N-1;i>=1;i--)
	{
		if(arr[i]>arr[i+1] && dp[i]<=dp[i+1])dp[i]=dp[i+1]+1;
		
	}
	
	
	for(i=1;i<=N;i++)ans+=dp[i];
	cout<<ans<<endl;
    return 0;
}
