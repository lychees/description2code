#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

typedef pair<int,int> pii;
typedef vector<pii> vp;
typedef vector<int> vi;

int home[100005];
int away[100005];
int main()
{
    int n;
    cin >> n;
    vp a(n);
    for(int i=0;i<n;i++)
    {
    	cin >> a[i].first >> a[i].second;
    	home[a[i].first]++;
    	away[a[i].second]++;
    }
	
	for(int i=0;i<n;i++)
	{
		int homecolor = a[i].first;
		int awaycolor = a[i].second;
		
		int mh = n-1;
		int ma = n-1;
		
		ma -= home[awaycolor];
		mh += home[awaycolor];
	
		cout << mh << " " << ma << endl;
	}
    
    
    return 0;
}
