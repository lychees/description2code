#include <bits/stdc++.h>
using namespace std;

int R,C;
int a[100][100];

void spiralPrint(int m, int n)
{
    int i, k = 0, l = 0;

    /*  k - starting row index
        m - ending row index
        l - starting column index
        n - ending column index
        i - iterator
    */

    int val=1;
    while (k < m && l < n)
    {
        /* Print the first row from the remaining rows */
        for (i = l; i < n; ++i)
        {
            a[k][i]=val++;
        }
        k++;

        /* Print the last column from the remaining columns */
        for (i = k; i < m; ++i)
        {
          a[i][n-1]=val++;
        }
        n--;

        /* Print the last row from the remaining rows */
        if ( k < m)
        {
            for (i = n-1; i >= l; --i)
            {
                a[m-1][i]=val++;
            }
            m--;
        }

        /* Print the first column from the remaining columns */
        if (l < n)
        {
            for (i = m-1; i >= k; --i)
            {
                a[i][l]=val++;
            }
            l++;
        }
    }
}

/* Driver program to test above functions */
int main()
{
  int tt;
  cin>>tt;
  while(tt--) {
    cin>>R;
    C=R;
    spiralPrint(R, C);
    for(int i=0;i<R;i++) {
      for(int j=0;j<C;j++) {
        cout<<a[i][j];
      }
      cout<<endl;
    }
    cout<<endl;
  }
    return 0;
}
