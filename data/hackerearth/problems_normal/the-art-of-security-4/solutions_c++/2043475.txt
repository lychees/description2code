#include <iostream>
using namespace std;

long long int possibleways(int *arr, int n, int one, int two, int mask);
bool taken[10];

int main()
{
	int n;
	int i;
	int arr[10];
	long long int sum=0;
	
	cin>>n;
	for(i=0; i<n; i++)
	{
		cin>>arr[i];
		sum+=arr[i];
	}
	for(i=0; i<10; i++)
		taken[i]=false;
		
	long long int ans=possibleways(arr, n, 0, 0, 0);
		
	cout<<ans<<" "<<sum<<endl;
	if(ans<=sum)
		cout<<"Got no way out!"<<endl;
	else
		cout<<"We will win!"<<endl;
		
	
    return 0;
}

long long int possibleways(int *arr, int n, int one, int two, int mask)
{
	if(one<two)
		return 0;
		
	if(mask==(1<<n)-1)
		return 1;
	
	long long int ans=0;
	for(int j=0; j<n; j++)
	{
		if(!(mask & (1<<j)))
		{
			mask=mask | (1<<j);
			ans=ans+possibleways(arr, n, one+arr[j], two, mask)+possibleways(arr, n, one, two+arr[j], mask);
			mask=mask ^ (1<<j);
		}
	}
	
	return ans;
}
