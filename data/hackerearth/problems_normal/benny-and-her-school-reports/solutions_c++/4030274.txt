#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    	{
    		long long int n,m,x;
    		cin>>n>>m>>x;
    		long long int sum=0,y;
    		for(int i=1;i<=n;i++)
    		 {
    		 	cin>>y;
    		 	sum+=y;
    		 }
    		if(m>=(n+1)*x - sum)
    		 {
    		 	if(sum<((n+1)*x))
    		 	cout<<(n+1)*x - sum;
    		 	else
    		 	cout<<1;
    		 }
    		else
    		cout<<"Impossible";
    		cout<<"\n";
    	}
    return 0;
}
