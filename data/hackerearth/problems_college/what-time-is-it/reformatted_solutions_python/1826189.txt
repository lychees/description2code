time = raw_input()
hr = int(time.split(':')[0])
min = int(time.split(':')[1])

if (hr == 12):
    hr = 0

if (min == 60):
    min = 0


ang_hr = ((60.00000*hr)+min)/2
ang_min = 6.00000*min
result = abs(ang_hr - ang_min)
if result > 180:
    result = 360 - result
# result = abs((30*hr)-11*(min/2))
print '{0:.6f}'.format(result)
