#include <iostream>
using namespace std;

int main()
{
   int n,i,r,s,s1=0,temp,l,b;
   cin>>n;
   int a[n];
   for(i=0;i<n;i++)
   {
   	cin>>a[i];
   }
   cin>>b>>l;
   for(i=0;i<n;i++)
   {
   	temp=a[i];
   	s=0;
   	while(temp!=0)
   	{
   		r=temp%10;
   		s=s+r;
   		temp=temp/10;
   	}
   	if(a[i]>0 && a[i]<=10)
   	s1=s1+(a[i]*s*l*b*.75);
   	else if(a[i]>10 && a[i]<=20)
   	s1=s1+(a[i]*s*l*b*.70);
   	else if(a[i]>20 && a[i]<=50)
   	s1=s1+(a[i]*s*l*b*.80);
   }
   cout<<s1;
    return 0;
}
