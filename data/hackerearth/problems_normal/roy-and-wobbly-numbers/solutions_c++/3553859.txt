#include<string>
#include<sstream>
#include<iostream>
#include<stdio.h>
using namespace std;
string repeat(const string &word, int times){
	string result;
	result.reserve(times*word.length());
	for(int i=0; i<times; i++)
		result+=word;
	return result;
}
string my_to_string(int i, int j)
{
	stringstream s;
	s<<i;
	s<<j;
	return s.str();
}
string my_to_string(int i)
{
	stringstream s;
	s<<i;
	return s.str();
}
int main(){
	int t, n,k,a,b;
	string res, fi;
	scanf("%d\n", &t);
	for(int i=0; i<t; i++)
	{
		scanf("%d", &n);
		scanf("%d", &k);
		if(k>81){
			cout<<my_to_string(-1)<<'\n';
			continue;
		}
		a=k/9;
		b=k%9;
		if(b!=0){
			a=a+1;
		}
		if(b<=a){
			b=b-1;
		}	
		if(b==-1){
			if(a!=9)
				b=9;
			else
				b=8;	
		}
		res = my_to_string(a,b);		
		if(n%2!=0){
			n-=1;
			fi=repeat(res,(n/2));
			fi+=my_to_string(a);
		}
		else{
			fi=repeat(res,(n/2));
		}		
		cout<<fi<<'\n';
	}
	return 0;
}
