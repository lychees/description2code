#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int n;
    	cin>>n;
    	unsigned long long int a[n] , flag = 1;
    	for(long long int i = 0 ; i < n ; i++)
    		cin>>a[i];
    	for(long long int i = 0 ; i < n ; i++)
    	{
    		if(a[i] > a[i+1])
    			flag = 0;
    	}
    	if(flag == 0)
    		cout<<"NO\n";
    	else
    		cout<<"YES\n";
    }
    return 0;
}
