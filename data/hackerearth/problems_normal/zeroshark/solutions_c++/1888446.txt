#include <bits/stdc++.h>
using namespace std;
const int MAX = 100001;
const int mod = (int) 1e9+7;
int n;
int dp[MAX][2][2];

int fun(int step,bool last,bool ok){
	if(step==n) return ok;
	int& r = dp[step][last][ok];
	if(r!=-1) return r;
	r = 0;
    if(last!=0){
        r += fun(step+1,0,ok) + fun(step+1,1,ok);
        r %= mod;
    }
    if(last==0){
        r += fun(step+1,1,ok);
        r %= mod;
        if(!ok) r += fun(step+1,0,1);
        r %= mod;
    }
    return r%mod;
}

int main()
{
    int T;
    scanf("%d",&T);
    while(T--){
    	scanf("%d",&n);
    	memset(dp,-1,sizeof dp);
    	cout << fun(0,-1,0) << endl;
    }
    return 0;
}
