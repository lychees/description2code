#include <iostream>
using namespace std;

#define ch2d(ch)  (ch-'a')
#define d2ch(d)  ('a'+d)
#define d2d(d) (d-'0')

int main()
{
    string str; 
    cin >>str;
    unsigned long long arr[26],i,t,x;
    
    for(i=0;i<26;i++) arr[i]=0;
    
    unsigned long long len=str.size(),j=0,number = 0;
    for( i=0 ; i < len ; i++){
    	char ch = str.at(i);
    	
    	//collect numbers
    	number=0;
    	for(j=i+1; j< len; j++){
    		char num_ch = str.at(j);
    		if( num_ch >= '0' && num_ch <= '9' ){
    			number = number*10+d2d(num_ch);
    		}else{
    			i = j-1;
    			break;
    		}
    	}
    	arr[ch2d(ch)]+= number;
    }
   
   
   for( i = 1 ; i < 26; i++) {	
   	arr[i] += arr[i-1];
   }
   
   //Collecting the random inputs
   cin >> t;
   while(t--){
   	cin >> x;
   	if( x > arr[25] ){
   		cout << "-1\n";
   		continue;
   	}
   	for(i=0;i<26;i++){
   		if(arr[i]>=x){
   			cout << (char)d2ch(i) << "\n";
   			break;
   		}
   	}
   }
    	
    return 0;
}
