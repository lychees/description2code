#include<cstdio>
using namespace std;
int main()
{
	int t,n;
	long int num,nn;
	scanf("%d",&t);
	while(t--)
	{
		int flag=0,count=0;
		scanf("%d%ld",&n,&num);
		while(n--)
		{
			scanf("%ld",&nn);
			if(nn==num)
			{
				flag=1;
			}
			if(nn>num)
			count++;
		}
		if(flag==1)
		{
			printf("YES %d\n",count);
		}
		else
		{
			printf("NO 0\n");
		}
	}
	return 0;
}
