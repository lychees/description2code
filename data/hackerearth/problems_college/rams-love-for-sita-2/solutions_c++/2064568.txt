#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,c,m;
    	cin>>n>>c>>m;
    	int countc=n/c;
    	int countr=countc;
    	while(countr>=m)
    	{
    		countc++;
    		countr=countr-m+1;
    	}
    	cout<<countc<<endl;
    	
    }
}
