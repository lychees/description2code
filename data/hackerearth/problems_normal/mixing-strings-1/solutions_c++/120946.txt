void Combine(string a,string &ans)
{
	int len=1,maxl=0;
	for(int i=a.length()-1;i>=0;--i)
	{
		string p=a.substr(i);
		string q=ans.substr(0,len);
		++len;
		if(p==q)
		{
			if(p.length()>maxl)maxl=p.length();
		}
	}
			ans=a+ans.substr(maxl);
}

int main()
{
	int N;
	cin>>N;
	string str;
	vector<int> vi;
	vector<string> ki;
	for(int i=0;i<N;++i)
	{
		vi.push_back(i);
		cin>>str;
		ki.push_back(str);
	}
	string ans;
	int mini=INT_MAX;
	do
	{
		ans=ki[vi[0]];
		for(int i=1;i<ki.size();++i)
		Combine(ki[vi[i]],ans);
		
		if(ans.size()<mini)
		mini=ans.size();
	}while(next_permutation(vi.begin(),vi.end()));
	cout<<mini;
	return 0;
}
