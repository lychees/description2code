#include<iostream>
#include<algorithm>
#define mod 1000000007;
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long int n,x=0,i,j,y=0,ans=0,z=0,a[105],b[105],c[105],x1=1,x2=1,x3,x4;
        cin>>n;
        x=n/2;y=n%2;z=0;x3=x;
        for(z=0;z<=x3;z++)
        {
            for(i=0;i<=x+y;i++){if(i==x+y){a[i]=1;continue;}a[i]=(x+y)-i;}
            for(i=0;i<=x;i++){if(i==x){b[i]=1;continue;}b[i]=x-i;}
            for(i=0;i<=y;i++){if(i==y){c[i]=1;continue;}c[i]=y-i;}
            for(i=0;i<=x;i++)
            {
                for(j=0;j<=x+y;j++){x4=__gcd(a[j],b[i]);a[j]=a[j]/x4;b[i]=b[i]/x4;}
            }
            for(i=0;i<=y;i++)
            {
                for(j=0;j<=x+y;j++){x4=__gcd(a[j],c[i]);a[j]=a[j]/x4;c[i]=c[i]/x4;}
            }
            for(i=0;i<=x+y;i++)x1*=a[i];
            for(i=0;i<=x;i++)x2*=b[i];
            for(i=0;i<=y;i++)x2*=c[i];
            ans+=(x1/x2);
            x--;y+=2;x1=1;x2=1;
        }
        cout<<ans;
        cout<<endl;

    }
    return 0;
}
