#include<iostream>
#include<algorithm>

using namespace std;

int main()
{
	int t;
	cin>>t;
	
	while(t--)
	{
		int n;
		cin>>n;
		
		int sharp[n];
		int str[n];
		
		int i;
		for(i = 0;i<n;i++)
			cin>>sharp[i];
		
		for(i = 0;i<n;i++)
			cin>>str[i];
			
			
		sort(sharp,sharp+n);
		sort(str,str+n);
		
		bool flag = true;
		
		for(i =0;i<n;i++)
			if(sharp[i]<str[i])
				flag = false;
				
		if(flag)
			cout<<"YES\n";
		else
			cout<<"NO\n";
	}
	
	return 0;
}
