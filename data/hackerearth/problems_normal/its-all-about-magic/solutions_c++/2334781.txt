#include <iostream>
using namespace std;
long long int cs[2][91];
int main()
{
    int t,m,i,j;
    cin>>t;
    cs[0][0]=0;
    cs[1][0]=1;
    for(i=1;i<=90;i++)
    {
    	cs[0][i]=cs[0][i-1]+cs[1][i-1];
    	cs[1][i]=cs[0][i-1];
    }
    while(t--)
    {
    	cin>>m;
    	cout<<cs[1][m]<<" "<<cs[0][m]<<"\n";
    }
    return 0;
}
