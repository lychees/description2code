#include <iostream>
#include <algorithm>
using namespace std;
long long n;
long long sum;
long long a[100000];


long long makePositive(long long x, long long y){
if(x>y)
   return x-y;
 return y-x;
}


int main()
{
 long long i,j;
 cin>>n;
 sum = 0;
 for(i=0;i<n;i++)
	cin>>a[i];
 sort(a,a+n);
 int flag =0;

for(i=0,j=n-1;i<j;){

  sum = sum + makePositive(a[i],a[j]);
  
  
  if(flag==0){
  i++; 
  flag=1;
  }
  else{
   j--; 
  flag=0;
  }

}
sum = sum + makePositive(a[j],a[0]);
cout<<sum<<endl;
return 0;


}
