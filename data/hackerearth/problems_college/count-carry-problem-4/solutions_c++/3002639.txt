#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		unsigned long long int n1,n2;
		cin>>n1>>n2;
	unsigned long long 	int sum =0;
		while(n1!=0 && n2!=0)
		{
			int p1 = n1%10;
			n1 = n1/10;
			int p2 = n2%10;
			n2 = n2/10;
			if(p1+p2>=10)
			{
				sum = sum + 1;
				n1 = n1+1;
			}
		}
		
		if(sum==1)
		{
			cout<<sum<<" "<<"carry operation"<<"\n";
		}
		else if(sum>1)
		{
			cout<<sum<<" "<<"carry operations"<<"\n";	
		}
		else if(sum==0)
		{
			cout<<"No"<<" "<<"carry operation"<<"\n";		
		}
	}
}
