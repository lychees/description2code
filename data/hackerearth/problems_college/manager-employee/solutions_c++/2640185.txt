#include<bits/stdc++.h>
using namespace std;
bool fun(long int i,long int j){return i>j;}
int main()
{
	long int t,n,i,j,k,a[1001],b[1001],c[1001];
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(i=0;i<n;i++)
		{
			cin>>a[i];
			b[i]=a[i];
		}
		sort(a,a+n);
		sort(b,b+n,fun);
		for(i=0;i<n;i++)
		{
			c[i]=a[i]+b[i];
		}
		sort(c,c+n);
		long int min=c[0];
        sort(c,c+n,fun);
		long int max=c[0];
		//cout<<max<<" "<<min<<"\n";
		cout<<max-min<<"\n";
	}
}
