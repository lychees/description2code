#include <bits/stdc++.h>
using namespace std;
vector<int> a[105];
int v[105]={0};
int main()
{
    int n,m;
    cin>>n>>m;
    for(int i=0;i<m;i++){
        int x,y;
        cin>>x>>y;
        a[x].push_back(y);
        a[y].push_back(x);
        
    }
    
    queue<int> my;
    //int count=2;
    my.push(1);
    v[1]=2;
    int tmp=1,ans=0,flag=0;
    while(!my.empty()){
        int tmp=my.front();
        my.pop();
        flag=0;
        for(int i=0;i<a[tmp].size();i++){
            if(v[a[tmp][i]]==v[tmp]-1 && flag==0){flag=1;continue;}
            if(v[a[tmp][i]]!=0){ans++;continue;}
            else v[a[tmp][i]]=v[tmp]+1;
            my.push(a[tmp][i]);
            
        }
    }flag=0;
    
    for(int i=1;i<=n;i++){
      //  cout<<v[i]<<' ';
        if(v[i]==0){flag=1;break;}
    
    }
    if(ans!=2){cout<<"NO";return 0;}
    if(flag==1)cout<<"NO";
    else cout<<"FHTAGN!";
    
    return 0;
}
