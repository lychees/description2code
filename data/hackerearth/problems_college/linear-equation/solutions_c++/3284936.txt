#include<bits/stdc++.h>
using namespace std;


int main()
{
	//freopen("input.txt","r",stdin);
int n,t;
cin>>t;
while(t--)
{cin>>n;
long long sum=0,lamda;
int a[n];
for(int i=0;i<n;i++)
{
	cin>>a[i];
	sum+=a[i];
}
lamda=sum/(n-1);

{
	cout<<lamda-a[i]<<" ";
}

}
}
