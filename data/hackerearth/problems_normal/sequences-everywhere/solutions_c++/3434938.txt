using namespace std;

const int N = 100005;

ll a,b;
ll x;

template<typename T>
T gcd(T c, T d){
	if(d==0) return c;
	else return gcd(d,c%d);
}

template<typename T>
T lcm(T c, T d){return (c/gcd(c,d))*d;}

ll funn(ll k){
	if(k == 2) return 1;
	REP(i,2,20){
		if(k % i != 0) return funn(i)+1;
	}
}

ll fun(ll p,ll q){
	if(q==2){
		ll ans = b/2-(a-1)/2;
		ans = (b-a+1) - ans;
		return ans;
	}
	if(p > b){
		return (b-a+1)-x;
	}
	ll ans=0;
	ll k = lcm(p,q);
	ans += b/p - (a-1)/p - b/k + (a-1)/k;
	return ans;
}

int main(){
	
	cin>>a>>b;
	ll ans=0;
	ll temp = 1;
	ll n = (b-a+1);
	if(a==2){
		ans += 1;
		n--;
		a++;
	}
	ll i =2;
	while(temp <= inf){
		if(x == n) break;
		ll kk = fun(temp ,i);
		ll k = funn(i);
		ans += kk*(k+1);
		x += kk;
		temp = lcm(temp,i);
		i++;
	}

	printf("%lld\n",ans);
	return 0;
}
