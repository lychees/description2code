#include <cstdio>
#include <vector>
#include <queue>
#include <memory.h>
#define MAX 1000000
using namespace std;
int visited[MAX+1];
int main()
{
    int t;
    scanf("%d",&t);
    
    while(t--)
    {
	int a,b;
	scanf("%d %d",&a,&b);
	int mul,div,sub,add;
	scanf("%d %d %d %d",&mul,&div,&add,&sub);
	queue<int> q;
	q.push(a);
	q.push(-1);
	int curlevel=0;
	memset(visited,0,sizeof(visited));
	int found=0;
	while(!q.empty())
	{
	    if(q.front()==-1)
	    {
		q.pop();
		if(!q.empty())
		{
		    curlevel++;
		    q.push(-1);
		}
		else
		    break;
	    }
	    //visited[q.front()]++;
	    if(q.front()==b)
	    {
		found=1;
		break;
	    }
	    int i=q.front();
	    if(i*mul<=MAX && i*mul>=2 )
	    {
		if(!visited[i*mul])
		{
		    visited[i*mul]++;
		    q.push(i*mul);
		}
	    }
	    if(i+add<=MAX && i+add>=2 )
	    {
		if(!visited[i+add])
		{
		    visited[i+add]++;
		    q.push(i+add);
		}
	    }
	    if(i-sub<=MAX && i-sub>=2)
	    {
		if(!visited[i-sub])
		{
		    visited[i-sub]++;
		    q.push(i-sub);
		}
	    }
	    if(i/div<=MAX && i/div>=2)
	    {
		if(!visited[i/div])
		{
		    visited[i/div]++;
		    q.push(i/div);
		}
	    }
	    q.pop();
	}
	if(found)
	printf("%d\n",curlevel);
	else
	    printf("-1\n");
	
    }
    return 0;
}
