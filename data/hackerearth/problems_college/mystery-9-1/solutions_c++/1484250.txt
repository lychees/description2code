#include <iostream>
#include <string>
using namespace std;

void mystery(string in){
	int len = in.length();
	int last = len-1;
	
	for(int i=0;i<len;i++)
	{
	 if(in[i] != in[last])
	 {
	  cout << "-1\n";
	  break;
	 }
	 else if(i == last || i > last)
	 {
	  cout << (int)in[i] << endl;
	  break;
	 }
	 last--;
	}
}

int main(){
	int T=0;
	string in;
	
	cin >> T;
	
	while(T--)
	{
	 in="";
	 cin >> in;
	 mystery(in);
	}
	
	return 0;
}
