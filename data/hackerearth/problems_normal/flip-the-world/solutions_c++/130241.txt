#include<iostream>

using namespace std;

int test,n,m,i,j,flip[50][50],count;
char mat[50][50];


int noOfChildren(int x,int y)
{
	int count=0,i,j;

	for(j=y+1;j<m;j++)
		if(flip[x][j])
			count++;

	for(i=x+1;i<n;i++)
		for(j=y;j<m;j++)
			if(flip[i][j])
				count++;

	return count;
}

int main()
{
	
	cin>>test;
	while(test--)
	{
		cin>>n>>m;

		for(i=0;i<n;i++)
			cin>>mat[i];

		count=0;
		for(i=n-1;i>=0;i--)
		{
			for(j=m-1;j>=0;j--)
			{

				if((noOfChildren(i,j)+(mat[i][j]-48))%2)
					flip[i][j]=0;
				else
					flip[i][j]=1;
			
				if(flip[i][j])
					count++;
			}

			
		}

		cout<<count<<endl;


	}
	return 0;

}
