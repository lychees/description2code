def sumlval(lst,n):
	res=0
	for i in range(n):
		res+=lst[i][0]
	return res
	
def update(lst,k,n):
	for i in range(k,n):
		lst[i][0]-=(lst[i][1]*lst[i][2])
		
		
n = int(raw_input())
lval = list(map(int,raw_input().split()))
dval = list(map(int,raw_input().split()))
lst=[]
for i in range(n):
	lst.append([lval[i],dval[i],i])
kfunc = lambda x: -(x[1]*x[2])
lst.sort(key=kfunc)
ans=0
for i in range(n):
	ans+=sumlval(lst,n)
	update(lst,i+1,n)
print(ans)
