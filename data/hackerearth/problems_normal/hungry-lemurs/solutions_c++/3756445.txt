#include <bits/stdc++.h>
using namespace std;
 
int main() {
	// your code goes here
	long long int n,k,i;
	long long int  Ans = 1e9;
 
		cin>>n>>k;
 
		for(i=1; i<=2*k; i++)
		{
			long long int tmp = n%i;
			long long int tmp1 = abs(k-i);
			long long int tmp2 = i - tmp;
			tmp = min(tmp, tmp2);
			Ans = min(Ans,tmp+tmp1);
		}
		cout<<Ans<<endl;
	return 0;
}
