#include <bits/stdc++.h>

using namespace std;

string s,res;
int n,sn;
char mat[200][200];

int main()
{
    int t;
    cin >> t;
    getline(cin,s);
    while(t--)
    {
    	getline(cin,s);
    	n=s.size();
    	int i,j;
    	sn=(int)(sqrt(n)+.5);
    	int curc=-1;
    	for(i=0;i<s.size();i++)
    	{
    		if(i%sn==0) curc++;
    		mat[curc][(i%sn)]=s[i];
    	}
    	res="";
    	for(i=0;i<sn;i++)
    	{
    		for(j=0;j<sn;j++)
    		{
    			res+=mat[j][i];
    		}
    	}
    	cout << res << endl;
    	
    }
    return 0;
}
