#include<cstdio>
#include<vector>
using namespace std;
int main()
{
	long long n;
	scanf("%lld", &n);
	vector<char> names;
	names.push_back('s');
	names.push_back('l');
	names.push_back('p');
	names.push_back('r');
	names.push_back('h');
	int i = 0;
	while (names.size() <= n)
	{
		if (names[i] == 's')
		{
			names.push_back('s');
			names.push_back('s');
		}
		else if (names[i] == 'l')
		{
			names.push_back('l');
			names.push_back('l');
		}
		else if (names[i] == 'p')
		{
			names.push_back('p');
			names.push_back('p');
		}
		else if (names[i] == 'r')
		{
			names.push_back('r');
			names.push_back('r');
		}
		else if (names[i] == 'h')
		{
			names.push_back('h');
			names.push_back('h');
		}
		i++;
	}
	if (names[n - 1] == 's')
		printf("Sheldon");
	else if (names[n - 1] == 'l')
		printf("Leonard");
	else if (names[n - 1] == 'p')
		printf("Penny");
	else if (names[n - 1] == 'r')
		printf("Rajesh");
	else if (names[n - 1] == 'h')
		printf("Howard");
    return 0;
}
