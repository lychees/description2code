#include <bits/stdc++.h>

#define Nmax int(1e5 + 10)

using namespace std;

typedef long long ll;
typedef long double ld;
typedef unsigned int ui;
typedef unsigned long long ull;
typedef pair<int, int> ii;
typedef vector<ii> vii;
typedef vector<int> vi;
typedef vector<vi> vvi;
typedef vector<double> vd;
typedef vector<string> vs;

int f[11][2], n;
string s;
ll mu_10[11];

int cal(int pos, bool lower){
    int &res = f[pos][lower];
    if(pos==n) res = 0;
    if(res!=-1) return res;

    res = 0;

    int Min = 0, Max = 9;
    if(!lower) Max = s[pos] - '0';
    for(int i=Min; i<=Max; i++){
        if(i==3) {
            res += mu_10[n-pos-1];
        }
        else{
            res+= cal(pos+1, lower | (i<s[pos]-'0'));
        }
    }

    return res;
}

void sol(){
    int x, tmp; cin >> tmp;

    x = tmp;
    s.clear();
    while(tmp>0){
        s = char(tmp%10 + '0') + s;
        tmp/=10;
    }

    memset(f, -1, sizeof(f));

    s = '0' + s;
    n = s.size();

    cout << x - cal(0, 0) << "\n";
}

int main(){

	std::ios::sync_with_stdio(false);
	cin.tie(NULL);

	mu_10[0] = 1;
	for(int i=1; i<11; i++) mu_10[i] = mu_10[i-1] * 10;

	int test; cin >> test;
	while(test--) sol();

    return 0;
}
