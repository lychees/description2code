#include <bits/stdc++.h>
#define ll long long
#define pb push_back
#define mk make_pair
#define pi pair<int, int>
#define fi first
#define se second
using namespace std;

int main() {
	int t;
	cin >> t;
	while(t--) {
	double v;
	cin >> v;
	double p = 2*acos(0.0);
	//cout << p << endl;
	double l = 0, r = 100, v1 = v / (2 * p);
	for(int i = 0; i < 200; i++) {
		double mid = (l + r) / 2;
		if(mid * mid * mid > v1) {
			r = mid;
		} else {
			l = mid;
		}
	}
	double rad = l;
	//cout << rad << endl;
	double h = v / (p * rad * rad);
	//cout << h << endl;
	double ans = 2 * p * rad * (rad + h);
	printf("%.9lf\n", ans);
	}
	return 0;
}
