#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int st,endv;
int dp[10005];
int vis[10005]={0};
vector<int>v[10005];
int solve(int node){
   if(node==endv){
   	return 1;
   }
   if(dp[node]!=-1){
   	return dp[node];
   }
   vis[node]=1;
   int ans=0;
   for(int i=0;i<v[node].size();++i){
   	ans+=solve(v[node][i]);
   }
   dp[node]=ans;
   return ans;
}

int main()
{
   st=1;
   scanf("%d",&endv);
   int u,v1;
   while(1){
   	scanf("%d%d",&u,&v1);
   	v[u].push_back(v1);
   	if(u==0&&v1==0){
   		break;
   	}
   }
   memset(dp,-1,sizeof(dp));
   printf("%d\n",solve(1));
    return 0;
}
