#include <iostream>
#include <cstdio>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n;
    	scanf("%d",&n);
    	int i,a[n];
    	for(i=0;i<n;i++)
    	{
    		scanf("%d",&a[i]);
    	}
    	int c = 0,j;
    	for(i=0;i<n;i++)
    	{
    		for(j=i;j<n;j++)
    		{
    			if(a[i]==a[j])
    			{
    				c++;
    			}
    		}
    	}
    	printf("%d\n",c);
    }
    return 0;
}
