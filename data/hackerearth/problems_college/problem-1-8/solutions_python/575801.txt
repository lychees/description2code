import sys

def get_line():
	return sys.stdin.readline().rstrip()

partial = get_line()

words = int(get_line())

shortestMatch = None

while words > 0:
	words -= 1
	
	word = get_line()
	
	if shortestMatch is not None and len(shortestMatch) < len(word):
		continue
	
	if len(word) < len(partial):
		continue
	
	if word.startswith(partial):
		shortestMatch = word
		
print shortestMatch or partial
