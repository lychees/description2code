#include <iostream>
using namespace std;

int main()
{
    int k,t;
    cin>>k>>t;
    long long a[k];
    for(int i=0;i<k;i++)
    cin>>a[i];
    long long m[1500];
    int l=0;
    m[0]=1;
    l++;
    int n=1;
    int q[1500]={0};
    int w[1500]={1};
    for(int i=0;i<k;i++)
    {
    	for(int j=0;j<n;j++)
    	{
    		if(m[j]>100000000000000LL)
    		continue;
    		m[l]=m[j]*a[i];
    		w[l]=(w[j]+1)%2;
    		if(m[l]==a[i])
    		q[l]=1;
    		l++;
    	}
    	n=l;
    }
    q[n-1]=1;
   // cout<<n<<t;
    while(t--)
    {
    	long long b,c;
    	cin>>b>>c;
    	long long s=0,d;
    	for(int i=1;i<n;i++)
    	{
			if(b%m[i]==0)
    		{
    			d=(c-b)/m[i];
    			if(w[i]==0)
    			s=s+(d+1);
    			else
    			s=s-(d+1);
    		}
    		else
    		{
    			long long f=b/m[i];
    			f++;
    			f=f*m[i];
    			if(f>c)
    			continue;
    			d=(c-f)/m[i];
    			if(w[i]==0)
    			s=s+(d+1);
    			else
    			s=s-(d+1);
    		}
    	}
    	cout<<s<<endl;
    }
}
