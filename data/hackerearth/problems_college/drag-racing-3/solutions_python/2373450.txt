#!/usr/bin/python
class Counter(dict):
    def __missing__(self, key):
        return 0

class MatchWinner:

    def __init__(self, count):
        self.rounds = count;
        self.participant = Counter();

    def getwinner(self):
        match_count = 2 ** self.rounds - 1
        for i in range(match_count):
            winner, loser = raw_input().split(" ");
            self.participant[winner] = self.participant[winner] + 1

        max_count = 0
        probable_winner = None;
        for person in self.participant:
            if max_count < self.participant[person]:
                max_count = self.participant[person]
                probable_winner = person

        return probable_winner


def main():
    count = input('');
    w = MatchWinner(count);
    print w.getwinner()

if __name__ == "__main__":
    main()
