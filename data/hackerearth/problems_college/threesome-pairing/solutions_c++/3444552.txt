#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
   int t;
   scanf("%d",&t);
   while(t--)
   {
   	int i,j,p,count=0,flag=0;
   	scanf("%d",&p);
   	int a[p];
   	for(i=0;i<p;i++)
   	{
   	scanf("%d",&a[i]);
   	}
   	for(i=0;i<p-1;i++)
   	{
   	for(j=i+1;j<p;j++)
   	{
   	if((a[i]+a[j])%3==0)
   	 count++;
   	}
   	}
   	cout<<count<<endl;
   }
   return 0;
}
