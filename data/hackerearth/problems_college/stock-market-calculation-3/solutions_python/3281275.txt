for i in range(int(input())):
    n=int(input())
    a=list(map(int,raw_input().split()))
    i=0
    ans=0
    while(i!=len(a)-1):
        m=max(a[i:])
        r=0
        c=0
        while(a[i]!=m):
            r+=a[i]
            c+=1
            i+=1
        i+= 1
        ans+=c*m-r
        if i >= len(a):
            break
    print(ans)
