#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
long long int big_power(long long n,long long p);
int main()
{
    long int t;
    scanf("%ld",&t);
    while(t--)
    {
        long long n,ans,p,q;
        scanf("%lld",&n);
        p=2;
        q=big_power(2,n)%mod;
        q=big_power(q,mod-2)%mod;
        ans=(p*q)%mod;
        printf("%lld\n",ans);
    }
    return 0;
}
long long int big_power(long long n,long long p)
{
    long long int tmp;
    n=n%mod;
    if(p==0)
        return 1;
    tmp=big_power(n,p/2)%mod;
    tmp=(tmp*tmp)%mod;
    if(p%2!=0)
        return (tmp*n)%mod;
    else
        return tmp;
}
