#define ll long long

#define mod 1000000007
#define pb push_back
ll modpow ( ll a , ll b )
{
ll res =1;
while ( b )
{
if ( b &1) res =( res * a )  ;
a =( a * a ) ;
b >>=1;
}
return res ;
}
ll n,m,i,j,k,ans=0,sum=0,q,x,y,t,r,h,w,b,c,d,z;
vector<vector<ll> > v;
vector<ll> visit;
ll dfs(ll x){
  ll cnt=1;
  visit[x]=1;
  for(ll i=0;i<v[x].size();i++){
    if(!visit[v[x][i]]){
      cnt+=dfs(v[x][i]);
    }
  }
  return cnt;
}
int main(){
  
 
    scanf("%lld",&t);
    while(t--){
      scanf("%lld %lld",&n,&m);
      v.clear();v.resize(n);
      visit.clear();visit.resize(n,0);
      for(i=0;i<m;i++){
        scanf("%lld %lld",&x,&y);
        x--;y--;
        v[x].pb(y);
        v[y].pb(x);
      }
      ans=1;
      for(i=0;i<n;i++){
        if(!visit[i]){
          x=dfs(i);
          if(x>1){
            if(x%2==0){
              x=((x/2)*(x-1))%mod;
            }
            else{
              x=(x*((x-1)/2))%mod;
            }
            ans=(ans*x)%mod;
          }
        }
      }
      cout<<ans<<endl;
    }
    
  return 0;
    
    
}
