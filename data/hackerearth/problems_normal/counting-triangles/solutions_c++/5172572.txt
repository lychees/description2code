#include <bits/stdc++.h>
#include <array>
using namespace std;
int main()
{
    array<long long ,3> a;
    map<array<long long int,3>,long long int> m;
    map<array<long long int,3>,long long int>:: iterator ii;
    long long int n,x,y,z,ans=0;
    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>x>>y>>z;
        a[0]=(x<y)?((x<z)?x:z):(y<z?y:z);
        a[2]=(x>y)?((x>z)?x:z):(y>z?y:z);
        a[1]=x+y+z-a[0]-a[2];
        m[a]++;
    }
    for(ii=m.begin();ii!=m.end();ii++)
    {
        if(ii->second==1) ans++;
    }
    cout<<ans;
    return 0;
}
