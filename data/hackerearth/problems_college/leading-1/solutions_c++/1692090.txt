#include <iostream>
using namespace std;
int fact(int);
int main()
{
int m,n,k,t,init=1;
cin>>t;
    while(init<=t){
 
	cin>>n;
	
    
    
	cin>>m;
	
    
	k=(fact(m+n-1))/((fact(m-1))*fact(n));
	cout<<k<<endl;
	init++;
    }
}
int fact(int l)
{
	
	if(l==0)
	{
	return 1;
	}
	if(l>0)
	l=l*fact(l-1);
	return l;

}
