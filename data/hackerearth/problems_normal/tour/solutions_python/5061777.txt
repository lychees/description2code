def search():
    t = raw_input()
    i = 0
    d = {}
    while i<int(t):
        d[raw_input()] = i
        i+=1
    s = [0 for  i in range(int(t))]
    for  i in range(int(t)):
        s[i] = raw_input().split()
    n = raw_input()
    j = 0
    sum = 0
    l = 0
    while j<int(n):
        r = d[raw_input()]
        sum +=  int(s[l][r])
        l = r
        j+=1
    print sum
        
search()
