#include<iostream>
#define ll long long
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		ll int n,i,idx,sum=0;
		cin>>n;
		ll a[n],x[n];
		for(i=1;i<=n;i++)
		{
			cin>>a[i];
		}
		idx=n;
		x[n]=n;
		for(i=n-1;i>0;i--)
		{
			if(a[idx]<a[i])
			idx=i;
			x[i]=idx;
		}
		for(i=1;i<=n;i++)
		{
			if((a[x[i]]-a[i])>=0)
			sum+=(a[x[i]]-a[i]);
		}
		cout<<sum<<endl;
	}
}
