#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n; cin>>n;
    int arr1[n],arr2[n];
    queue<int>q;
    for(int i=0;i<n;i++){
    	cin>>arr1[i];
    	q.push(arr1[i]);
    } 
    for(int i=0;i<n;i++) cin>>arr2[i];
    int i=0,j=0,t=0;
    while(!q.empty()){
    	int temp=q.front();
    	if(arr2[i]!=temp){
    		q.pop();
    		q.push(temp);
    		t++;
    	}
    	else{
    		i++,t++;
    		q.pop();
    	}
    }
    cout<<t<<endl;
    return 0;
}
