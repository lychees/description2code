#include <iostream>
#include <algorithm>
using namespace std;

string dp[10010][2];
string add(const string& a,const string& b)
{
	string s1=a;
	reverse(s1.begin(),s1.end());
	string s2=b;
	reverse(s2.begin(),s2.end());
	string s="";
	int l1=a.length(),l2=b.length();
	int carry=0;
	for(int i=0;i<min(l1,l2);++i)
	{
		int d=s1[i]-'0' + s2[i]-'0'+carry;
		carry=d/10;
		d%=10;
		char ch=d+'0';
		s+=ch;
	}
	if(l1<l2)
	{
		for(int i=l1;i<l2;++i)
		{
			int d=s2[i]-'0'+carry;
			carry=d/10;
			d%=10;
			char ch=d+'0';
			s+=ch;
		}
	} else if(l2<l1)
	{
		for(int i=l2;i<l1;++i)
		{
			int d=s1[i]-'0'+carry;
			carry=d/10;
			d%=10;
			char ch=d+'0';
			s+=ch;
		}
	}
	if(carry>0)
	{
		char ch=carry+'0';
		s+=ch;
	}
	reverse(s.begin(),s.end());
	return s;
}
void init()
{
	dp[1][0]="1";
	dp[1][1]="1";
	for(int i=2;i<=10000;++i)
	{
		dp[i][1]=add(dp[i-1][0],dp[i-1][1]);
		dp[i][0]=dp[i-1][1];
	}
	/*for(int i=2;i<=10;++i)
	{
		cout<<dp[i][0]<<" "<<dp[i][1]<<endl;
	}*/
}
int main()
{
    init();
    int t,n;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	
    	string ans=add(dp[n][0],dp[n][1]);
    	cout<<ans<<endl;
    }
    return 0;
}
