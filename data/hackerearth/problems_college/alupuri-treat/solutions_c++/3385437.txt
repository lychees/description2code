#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    int m,s=0,sum=0;
    cin>>m;
    int a[m][m];
    for(int i = 0; i < m; i++){
    	for(int j = 0; j < m ;j++){
    		cin>>a[i][j];
    	}
    }
    for(int i = 0; i < m; i++){
    	s+=a[i][i];
    }
    for(int i = 0,j=m-1; i < m,j>=0; i++,j--){
    	   sum += a[i][j];
     }
     cout<<abs(s-sum)<<endl;
    return 0;
}
