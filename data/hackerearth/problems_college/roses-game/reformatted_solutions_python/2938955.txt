t = int(raw_input())
while t > 0:
    t -= 1
    n = int(raw_input())
    list = map(int, raw_input().split())
    i = 0
    j = 1
    while i < n:
        if list[i] == -1:
            j = 1
            break
        elif list[i] == -2:
            j = 0
            break
        else:
            j = i
            i = list[i]
            list[j] = -2
    if j == 1:
        print("Win")
    else:
        print("Lose")
