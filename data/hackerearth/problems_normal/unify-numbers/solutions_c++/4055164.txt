#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
using namespace std;

int main()
{
    int testCases;
    cin >> testCases;
    
    for(int l = 0; l < testCases; l++){
    	int n;
    	cin >> n;
    	vector<int> vect(n);
    	int k;
    	for(int i = 0; i < n; i++){
    		cin >>k;
    		vect[i] = k;
	    }
    
	
 		sort (vect.begin(), vect.end());
 		
 		if(n%2 != 0){
 			int mean1 = vect[n/2]; 
 			int mean2 = vect[n/2 + 1];
    	
    		long long cost1 = 0;
    		for(int i = 0; i < n; i++){
    			cost1 += abs(mean1 - vect[i]);
    		}
    		
    		long long cost2 = 0;
    		for(int i = 0; i < n; i++){
    			cost2 += abs(mean2 - vect[i]);
    		}
    		
    		if(cost1 < cost2)
    			cout <<cost1 <<endl;
    		else
    			cout <<cost2 <<endl;
 		}
 		
 		else{
 			int mean1 = vect[n/2]; 
    	
    		long long cost1 = 0;
    		for(int i = 0; i < n; i++){
    			cost1 += abs(mean1 - vect[i]);
    		}
    		cout <<cost1 <<endl;
 		}
    }
    return 0;
}
