#include <iostream>
using namespace std;
int fn(int number){
	int count=0;
	for(int j=2;j<number+1;j++){
		for(int k=1;k<j;k++){


                int r = k^j;
				if(r <= number){

					count++;
							}

		}
	}
	return count;
}
int main()
{
	int n=0;
	cin>>n;
	for(int i=0;i<n;i++){
		int co;
		cin>>co;
		cout<<fn(co)<<endl;
	}

}
