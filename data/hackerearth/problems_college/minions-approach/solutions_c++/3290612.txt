#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    vector<bool> prime(100001,true);

    prime[0]=false;
    prime[1]=false;

    for (int p=2; p*p<=100000; p++)
    {

        if (prime[p] == true)
        {
            for (int i=p*2; i<=100000; i += p)
                prime[i] = false;
        }
    }
    int t;
    cin>>t;
    
    while(t!=0)
    {
    	int n;
    	cin>>n;
    	
    	int ans=0;
    	
    	for(int i=1;i<=n;i++)
    	{
    		if(__gcd(i,n)==1)
    		ans++;
    	}
    	
    	if(prime[ans])
    	cout << "Correct\n";
    	else
    	cout << "Incorrect\n";
    	
    	t--;
    }
}
