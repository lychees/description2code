#include<iostream>
using namespace std;
int flip(int n)
{
    int r,temp=0;
    while(n!=0)
    {
        r=n%10;
        n=n/10;
        temp=(temp*10) + r;
    }
    return temp;
}
int main()
{
    int n;
    cin>>n;
    int arr[n];
    for(int i=0;i<n;i++)
    {
        int n1,n2;
        cin>>n1>>n2;
        int c=flip(n1)+flip(n2);
        arr[i]=flip(c);
    }
    for(int i=0;i<n;i++)
        cout<<arr[i]<<endl;
    return 0;
}
