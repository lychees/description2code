#include <iostream>
using namespace std;

int main()
{
    long long int tc,N,n;
    cin>>tc;
    while(tc--)
    {
    	n=1;
    	cin>>N;
        while(n*(n+1)<N)
        {
        	n++;
        }
        if(n*n>=N)
        {
        	cout<<2*n-1<<endl;
        }
        else
        {
        	cout<<2*n<<endl;
        }
    }
}
