#include<bits/stdc++.h>
using namespace std;

int main()
 {
 	int n,x,y;
 	cin>>n;
 	vector<pair<int,int> > v;
 	
  	cin>>x>>y;
 	 v.push_back(make_pair(x,y));
 	 
 	vector<int> a(n+1);
 
 	a[1]=1;
 	while(!(x==0 && y==0))
 	  {
 	  
 	  	cin>>x>>y;
 	  	v.push_back(make_pair(x,y));
 	 
 	  
 	  }
 	sort(v.begin(),v.end());
 	for(int i=0;i<v.size();i++)
 	  a[v[i].second]+=a[v[i].first];
 	cout<<a[n];
 }
