#include<iostream>
#include<algorithm>
using namespace std;
struct lottery
{
       int start;
       int end;
       int weight;
};
bool comparator(struct lottery a,struct lottery b){
     if(a.end==b.end)
        return (a.start<b.start);
     return (a.end<b.end);
}
int max(int a,int b)
{
    return (a>b)?a:b;
}
main()
{
      struct lottery lucky[1000];
      float ans[1000],final,maxsumsofar;
      int t,n,i,x;
      cin>>t;
      while(t--){
         cin>>n;
         for(i=0;i<n;i++)
             cin>>lucky[i].start>>lucky[i].end>>lucky[i].weight;
         sort(lucky,lucky+n,comparator);
         ans[0]=(float)lucky[0].weight;
         for(i=1;i<n;i++){
             x=i-1;
             maxsumsofar=lucky[x].weight;
             while(x>=0){
                 if(lucky[x].end<=lucky[i].start)
                     break;
                 maxsumsofar=max(maxsumsofar,ans[x]);
                 x--;
             }
             if(x!=-1)
                 ans[i]=ans[x]+lucky[i].weight;
             else
                 ans[i]=lucky[i].weight;
             ans[i]=max(ans[i],maxsumsofar);
         }
         printf("%.2f\n",ans[n-1]/(float)100);
      }
      return 0;
}
