#include<bits/stdc++.h>
using namespace std;
#define MOD 1000000007
long long L[1009];
long long fact[1009];
long ncr[1009][1009];


int main()
{
    int N,K;
    cin>>N>>K;

    fact[0] = fact[1] = 1;
    L[0] = 1;
    L[1] = 1;

    for(int i = 2;i<=N;i++)
        fact[i] = ((i*fact[i-1])%MOD+MOD)%MOD;

    for(int i=1;i<=N;i++)
    {
        for(int j=0;j<=i;j++)
        {
            if(j==0 || j==i)
                ncr[i][j] = 1;
            else ncr[i][j] = (ncr[i-1][j]+ncr[i-1][j-1])%MOD;
        }
    }

    for(int i=1;i<=N;i++)
    {
        for(int a = 0;a<i;a++)
        {
            long long b = ((ncr[i-1][a]*fact[a+1])%MOD+MOD)%MOD;
            b = ((b*L[i-a-1])%MOD+MOD)%MOD;
            L[i] = (L[i]+b)%MOD;
            //L[i+1] = (L[i+1]+(((ncr[i][a]*fact[a+1])%MOD)*L[i-a])%MOD)%MOD;
        }
    }

    long long ans = ((L[N]*ncr[N][K])%MOD+MOD)%MOD;
    cout<<ans;
}
