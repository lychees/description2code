#include<bits/stdc++.h> 
using namespace std; 
bool isv(int i, int j,int R,int C)
{
    if (i < 0 || i >= R || j >= C || j < 0) return false;
    return true;
} 
void dia(int a[][101],int b,int c)
{
    for (int k=0; k<c; k++)
    {
        cout << a[0][k] << " ";
        int i = 1;    int j = k-1; 
        while (isv(i,j,b,c))
        {
            cout << a[i][j] << " ";
            i++;
            j--;
        }
        cout << endl;
    } 
   for (int k = 1; k < b; k++)
    {
        cout << a[k][c-1] << " ";
        int i = k+1;  
        int j = c-2;   
        while (isv(i,j,b,c))
        {
            cout << a[i][j] << " ";
            i++;
            j--;  
        }
        cout << endl;
    }
} 
int main()
{ int n,m,b,c;
    cin>>n>>m;
    int a[101][101];
    for(int i=0;i<n;i++)
        {
        for(int j=0;j<m;j++)
        {
        cin>>a[i][j];
        }
        }
    b=n;c=m;
    dia(a,b,c);
    return 0;
}
