#include <iostream>
#include<string>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string a,b;
    	cin>>a>>b;
    	int x[26]={},y[26]={};
    	int i;
    	i=0;
    	while (a[i]!=NULL)
    	{
    		x[a[i]-'a']++;
    		i++;
    	}
    	i=0;
    	while (b[i]!=NULL)
    	{
    		y[b[i]-'a']++;
    		i++;
    	}
    	for(i=0;i<26;i++)
    	{
    		if (x[i]!=0 && y[i]!=0)
    		break;
    	}
    	if(i==26)
    	cout<<"NO";
    	else
    	cout<<"YES";
    	cout<<endl;
    }
    return 0;
}
