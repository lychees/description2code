#include<bits/stdc++.h>
using namespace std;
map<long long int,long long int> mp;
long long int count(long long int n){
    if(n==0)
        return 0;
    long long int a;
    a=mp[n];
    if(a==0){
        a= max(n,count(n/2)+count(n/3)+count(n/4));
        mp[n]=a;
    }
    return a;
}
int main(){
    long long int n;
    scanf("%lld",&n);
    mp.clear();
    cout<<count(n)<<endl;
    return 0;
}
