#include<bits/stdc++.h>

using namespace std;




int main()
{
	int t,n,i,j,d,c;
	cin >> t;
	while(t--)
	{
		cin >> n;
		int arr[n];
		for (i = 0; i <n; i++)
		{
			cin >> arr[i];
		}
		
		sort(arr,arr+n);
		c = 0;
		d = 0;
		for(i = 0; i < n-1 ; i++)
		{
			d = arr[ i+ 1] - arr[i] + d;
			if(d > 0)
			{
				c = c + (d/5) + ((d%5)/2) + ((d%5)%2);
			}
		}
		cout << c << endl;
	}
	return 0;
}
