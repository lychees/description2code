#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int x1, y1, r1, x2, y2, r2;
        cin>>x1>>y1>>r1>>x2>>y2>>r2;
        float d = sqrt(((x1-x2)*(x1-x2))+((y2-y1)*(y2-y1)));
        int sm,mm;
        if(r1>r2)
        {
            sm=r2;
            mm=r1;
        }
        else
        {
            sm=r1;
            mm=r2;
        }
        d += sm;
        if(float(mm)>=d)
        {
            if(r1>r2)
                cout<<"C1CC2"<<endl;
            if(r2>r1)
                cout<<"C2CC1"<<endl;
            if(r2==r1)
                cout<<"C2~C1"<<endl;
        }
        else
            cout<<"NOT SATISFIED"<<endl;
    }
    return 0;
}
