#define fi first

#define MOD 1000000007
#define ll long long int
#define MAXINT 2000000000
#define SET(x,y) memset(x,y,sizeof(x));
 
using namespace std;

ll LIM;

ll power(ll a,ll b) {
	ll ans=1,i;
	for(i=1;i<=b;i++) {
		double x = ((double)ans)*a;
		if(x>LIM)	return -1;
		else		ans=x;
	}
	return ans;
}

void solve(void) {
	ll p,q1,q,k,r,s,a,x,ans=0;
	cin>>p>>q>>k;
	LIM = k+10;
	for(x=0; ;x++) {
		q1=power(x,p);
		if(q1==-1)	break;
		r=k-x-q1;
		if(r<0)		break;
		s = round(pow(r,1.00/q));
		a = pow(s,q);
		//cout<<x<<" "<<q1<<" "<<r<<" "<<s<<" "<<a<<endl;
		if(a == r)	ans++;
	}
	cout<<ans<<endl;
}

int main () {
	int t,i;
	cin>>t;
	for(i=1;i<=t;i++) {
		cout<<"Case #"<<i<<": ";
		solve();
	}
	return 0;
}
