#include <iostream>
#include<stdlib.h>
#include<stdio.h>
#include<math.h>
#include<string.h>
#define MAX 604800
using namespace std;
int compare (const void * a, const void * b)
{
  return ( *(int*)a - *(int*)b );
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int arr[100000],ai=0;
		int p=7;
		int cnt=0;
		while(p!=0)
		{
			
			int x,a,b;
			cin>>x;
			cnt+=x;
			for(int i=0;i<x;i++)
			{
				cin>>a>>b;
				int x=b-a;
				arr[ai++]=x;
			}
		p--;
		}
		qsort (arr, ai, sizeof(int), compare);
		float ans=(cnt*75.0)/100;
		ans=ceil(ans);
		int sum=0;
		for(int i=0;i<ans;i++)
		sum+=arr[i];
		
		cout<<(MAX-sum)<<endl;
	}
}
