#!/usr/bin/env python
# -*- coding: utf-8 -*-

# https://www.hackerearth.com/problem/algorithm/xsquare-and-balanced-strings/

from collections import defaultdict

T = int(raw_input())
for t in xrange(0, T):
    s = raw_input()
    m = defaultdict(int)
    for ch in s:
        m[ch] += 1
    breakable = True
    for ch, i in m.iteritems():
        if i % 2 == 1:
            breakable = False
            break
    if breakable:
        print 1
    else:
        print -1
