#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t,n,i,j,flag;
	char a,b,c;
	int row[1001],col[1001],for_row[1001],for_col[1001],back_row[1001],back_col[1001];
	cin>>t;
	while(t--)
	{
		for(i=0;i<1001;i++)
		{
		//	cout<<"yeah\n";
			for_row[i]=0;
			for_col[i]=0;
			back_col[i]=0;
			back_row[i]=0;
			row[i]=0;
			col[i]=0;
		}
		flag=0;
		cin>>n;
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
			{
				cin>>a;
				if(a=='#')
				{
					row[i]++;
					col[j]++;
				}
			}
		//	cin>>a;
		}
		for_row[0]=row[0];
		for_col[0]=col[0];
		back_row[n-1]=row[n-1];
		back_col[n-1]=col[n-1];
		for(i=1;i<n;i++)
		{
			for_col[i]=for_col[i-1]+col[i];
			for_row[i]=for_row[i-1]+row[i];
		}
		for(i=n-1;i>=0;i--)
		{
			back_col[i]=back_col[i+1]+col[i];
			back_row[i]=back_row[i+1]+row[i];
		}
	/*	for(i=0;i<n;i++)
		{
			cout<<for_col[i]<<" ";
		} */
	//	cout<<"\n";
		for(i=0;i<n-1;i++)
		{
			if(for_row[i]==back_row[i+1])
			{
				cout<<"YES\n";
				flag=1;
				break;
			}
			if(for_col[i]==back_col[i+1])
			{
				cout<<"YES\n";
				flag=1;
				break;
			}
		}
		if(flag!=1)
			cout<<"NO\n";
	}
	return 0;
}
