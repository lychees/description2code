

    #include<bits/stdc++.h>
    #define MAG 10000000
    using namespace std;
     
    int pr[MAG+31];
    int phi[MAG+31];
    long long S[MAG+31];
    int tests,n;
     
    int main(){
    	ios_base::sync_with_stdio(0);
    	//cin.tie(0);
    	
    	pr[1]=1;
    	for (int i=2;i<=MAG;i++)
    		if (pr[i]==0)
    			for (int j=i*2;j<=MAG;j+=i)
    				pr[j]=i;
    	
    	phi[1]=1;
    	
    	for (int i=2;i<=MAG;i++)
    	{
    		int q=i;
    		int val=pr[q];
    		if (val==0)
    			phi[i]=q-1;
    		else
    		{
    			while (q%val==0)
    				q/=val;
    			phi[i]=phi[q]*(i/q)/val*(val-1);
    		}
    	}
    	
    	for (int i=1;i<=MAG;i++)
    		S[i]=S[i-1]+phi[i];
    	
    	cin>>tests;
    	for (;tests;--tests)
    	{
    		cin>>n;
    		
    		cout<<S[n]<<endl;
    	}
    	
    	return 0;
    }
