#include <iostream>
#include<algorithm>
using namespace std;
long long M = 1000000007;
int main()
{
  ios_base::sync_with_stdio(false);
  long long trip[3];
  int T;
  cin>>T;
  while(T--)
   {
   	 cin>>trip[0]>>trip[1]>>trip[2];
   	 sort(trip,trip+3);
     trip[0]=(trip[0])%M;
     trip[1]=(trip[1]-1)%M;
     trip[2]=(trip[2]-2)%M;  
   	 cout<<((((trip[0])*(trip[1]))%M)*(trip[2]))%M<<"\n";
   	}
return 0;
}
