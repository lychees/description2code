#include <bits/stdc++.h>
using namespace std;

int main(void) {
    ios::sync_with_stdio(true);
    int T = 0;
    scanf("%d",&T);
    while(T--) {

    	string s1, temp;
    	cin >> s1;
    	temp = s1;
    	int i = s1.size() - 1, amount = 987654321, k = 0, idx = 0, flag = 0;
    	if(s1.size() != 1)
    		for(; i > 0; --i)
    			if(s1[i - 1] < s1[i]) {
    				flag += 1;
    				break;
    			}
    	if(flag) {
    		k = i - 1;
    		amount = s1[i] - s1[k];
    		idx = i;
    		for(; i < s1.size(); ++i)
    			if(s1[i] - s1[k] < amount && s1[i] - s1[k] > 0) {
    				amount = s1[i] - s1[k];
    				idx = i;
    		}
    		swap(s1[k],s1[idx]);
    		sort(s1.begin() + k + 1, s1.end());
    	}
    	if(temp == s1)
    		puts("no answer");
    	else
    		cout << s1 << endl;
    }
    return 0;
}
