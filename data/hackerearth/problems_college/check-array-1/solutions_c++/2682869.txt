#include<stdio.h>
#include<iostream>
#include<string.h>
#include<math.h>
#include<algorithm>
#include<stdlib.h>
#include<vector>

using namespace std;

main()
{
	int t;
	cin>>t;
	
	while(t--)
	{
		int n,a[100000];
		cin>>n;
		int sum=0;
		a[0]=0;
		for(int i=1;i<=n;i++)
		{
			cin>>a[i];
			sum=sum+a[i];
			a[i]=a[i]+a[i-1];
		}
		
		int flag=0;
		
		for(int i=1;i<n;i++)
		{
			if(a[i]+a[i+1]==a[n])
			{
				flag=1;
				break;
			}
		}
		if(flag==1)
		cout<<"YES\n";
		else
		cout<<"NO\n";
	}
}
