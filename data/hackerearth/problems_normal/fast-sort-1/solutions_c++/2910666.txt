#include<bits/stdc++.h>
#define ll long long int
#define inp(x) cin >> x
using namespace std;
map< char , ll > m;
int main(){
	ll t,x;
	string s,r;
	inp(t);
	while(t--){
		inp(s);inp(r);
		for(ll i=0;i<r.length();++i){
			m[r[i]]++;
		}
		for(ll i=0;i<s.length();++i){
			if(m[s[i]]!=0){
				x = m[s[i]];
				while(x--){
					cout<<s[i];
				}
			}
		}cout<<endl;
		m.clear();
	}
	return 0;
}
