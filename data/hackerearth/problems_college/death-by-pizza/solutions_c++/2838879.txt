#include<iostream>
#define N 20000
using namespace std;

int main(void)
{
    long long int q,f=0,in1=0,in2=0,a[N],sum=0,i=0,n;
    cin >> n >> q;
    int pizza[5]={16,9,26,26,1},pizzCt=0;
    for(i=0;i<n;i++)
        cin >> a[i];
    while(q--)
    {
        cin >> f >> in1 >> in2;
        if(!f)
        {
            pizzCt=0;
            for(i=in1-1;i<=in2-1;i++)
            {
                if(pizzCt==5) pizzCt=0;
                a[i]+=pizza[pizzCt];
                pizzCt++;
            }
        }
        else
        {
            sum=0;
            for(i=in1-1;i<=in2-1;i++)
                sum+=a[i];
            cout << sum << "/" << in2-in1+1;
            cout << endl;
        }
    }

    return 0;
}
