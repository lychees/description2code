#include <iostream>
#include <cmath>
using namespace std;

int main()
{
	int t;
	unsigned long count, x, y;
	cin>>t;
	while(t--) {
		cin>>x>>y;
		unsigned long i, sqr, j, k, n, sq, rem;
		n = y-x+1;
		bool special[y-x+1];
		for(i=0; i<n; i++) {
			special[i] = true;
		}
		sqr = sqrt(y);
		for(i=2; i<=sqr; i++) {
			sq = i*i;
			rem = x % sq;
			if(rem == 0) {
				k = x;
			} else {
				k = x+(sq-rem);
			}
			for(j=k; j<=y; j=j+sq) {
				special[j-x] = false;
			}
		}
		
		count = 0;
		i=0;
		while(i<n) {
			if(special[i++]) {
				count++;
			}
		}
		cout<<count<<endl;
	}
    return 0;
}
