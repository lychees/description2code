#include<bits/stdc++.h>
using namespace std;
#define MAX 1000000
#define MOD 1000000007
long long ans[MAX+4]={0};
long long nos[MAX+4][2]={0};


int main()
{
	int t,n,i;
	nos[1][0]=nos[1][1]=1;
	ans[1]=2;
	for(i=2;i<=MAX;i++)
	 {
	 	nos[i][0]=(nos[i-1][0]+nos[i-1][1])%MOD;
	 	nos[i][1]=nos[i-1][0];
	 	ans[i]=(ans[i-1]+nos[i][0]+nos[i][1])%MOD;
	 }
	
	cin>>t;
	
	while(t--)
	 {
	 	cin>>n;
	 	assert(n>=1 && n<=MAX);
	 	cout<<ans[n]<<endl;
	 }
}
