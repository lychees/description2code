t = input()
while t:
    a, b = map(str, raw_input().split())
    a = list(a)
    b = list(b)
    x = len(a)
    y = len(b)
    count = 0
    z = max(x, y)
    for i in xrange(0, z):
        if int(a[i])+int(b[i]) > 9:
            count += 1
    if count == 0:
        print "No carry operation"
    elif count == 1:
        print "%d %s" % (count, "carry operation")
    else:
        print "%d %s" % (count, "carry operations")
    t -= 1
