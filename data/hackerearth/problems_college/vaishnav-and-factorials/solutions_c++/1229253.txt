#include<bits/stdc++.h>
using namespace std;

int a[500];

int main() {
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        int n;
        scanf("%d",&n);
        int idx=0;
        int carry=0;
        for(int i=1;i<=n;i++) {
            if(idx==0) {
                a[idx++]=1;
                continue;
            } else {
                carry=0;
                for(int j=0;j<idx;j++) {
                    int val=i*a[j];
                    val+=carry;
                    a[j]=val%10;
                    carry=val/10;
                }
            }
            while(carry) {
                a[idx++]=carry%10;
                carry/=10;
            }
        }
        while(carry) {
            a[idx++]=carry%10;
            carry/=10;
        }
        int cnt=0;
        for(int i=0;i<idx;i++) {
            if(a[i]==4||a[i]==7) {
                cnt++;
            }
        }
        printf("%d\012",cnt);
    }
    return 0;
}
