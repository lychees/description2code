#include<iostream>
#include<sstream>
#include<vector>
#include<string>
#include<algorithm>
#include<math.h>
using namespace std;
#define IOS ios_base::sync_with_stdio(false);cin.tie(0)
typedef long long LL;
int main() {
  IOS;
  int tc;
  LL a, b, c, d, k;
  cin >> tc;
  while (tc--) {
    cin >> a >> b >> c >> d >> k;
    LL t = pow(k / a, 1.0 / 3.0);
    while (a*t*t*t + b*t*t + c*t + d > k&&t>0) {
      t--;
    }
    cout << t << endl;
  }

  return 0;
}
