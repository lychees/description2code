#include <iostream>
#include <algorithm>
using namespace std;

long long C[100][30];

int main()
{
    long long i, j;
    long long n = 99;
    long long k = 26;
    for (i=0;i<=n;i++)
    {
        for (j=0; j<=min(i, k);j++)
        {
            if (j==0||j==i)
                C[i][j]=1;
            else
                C[i][j]=C[i-1][j-1] + C[i-1][j];
        }
    }
    int t;
    cin>>t;
    while(t--)
    {
        int n,k;
        cin>>n>>k;
        cout<<C[n+k-1][n-1]<<"\n";
    }
}
