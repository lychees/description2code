#include <iostream>
#include<cmath>
using namespace std;

int64_t gcd(int64_t a,int64_t b)
{
  if(b==0)
    return a;
  if(a%b==0)
   return b;
  else
    return gcd(b,a%b);
}
int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	int64_t x1,x2,y1,y2;
   	cin>>x1>>y1>>x2>>y2;
   	if(x1==x2&&y1==y2)
   	 cout<<0<<"\n";
   	else{
   		int64_t a=abs(x1-x2);
   	int64_t b=abs(y1-y2);
   	int64_t ans=gcd(a,b)-1;
   	cout<<ans<<"\n";
   	}
   	
   	
   }
    return 0;
}
