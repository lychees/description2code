#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <string>
#include <bits/stdc++.h>
using namespace std;

int main() {
	int t;
	scanf("%d",&t);
	getchar();
	while (t--) {
		string s1,s2;
		cin >> s1 >> s2;
		int d1,m1,y1,d2,m2,y2;
		string temp = s1.substr(0,2);
		d1 = atoi(temp.c_str());
		m1 = atoi((s1.substr(3,2)).c_str());
		y1 = atoi((s1.substr(6,4)).c_str());
		d2 = atoi((s2.substr(0,2)).c_str());
		m2 = atoi((s2.substr(3,2)).c_str());
		y2 = atoi((s2.substr(6,4)).c_str());
		int ctr = 0;
		int temp1 = (y1%100)-1;
		int temp2 = temp1-1;
	//	cout << temp1 << temp2;
		if (temp1 > m1  && temp1 > 0 && temp2 > 0 && y1 > 0 && y1%100 <= 13) {
			if (y1 == y2) {
				if (temp1 < m2)
				ctr++;
				else if (temp1 == m2 && temp2 <= d2)
					ctr++;
			}
			else
				ctr++;
		}	
		else if (temp1 == m1) {
			if (temp2 >= d1  && temp1 > 0 && temp2 > 0 && y1 > 0 && y1%100 <= 13) {
				if (y1 == y2) {
				if (temp1 < m2)
				ctr++;
				else if (temp1 == m2 && temp2 <= d2)
					ctr++;
				}
				else
					ctr++;
			}	
		}	
		y1++;	
		while (y1 <= y2) {
			if (y1 > y2)
				break;
			if (y1%100 <= 13) {
				m1 = (y1%100)-1;
				d1 = m1-1;
				if (y1 < y2 && y1 > 0 && m1 > 0 && d1 > 0) {
					ctr++;
					//cout << y1 << m1 << d1 << endl;
				}	
				else if (y1 == y2) {
					if (m1 < m2 &&  y1 > 0 && m1 > 0 && d1 > 0) {
						ctr++;
					//	cout << y1 << m1 << d1 << endl;
					}
					else if (m1 == m2)
						if (d1 <= d2 &&  y1 > 0 && m1 > 0 && d1 > 0) {
							ctr++;
							//cout << y1 << m1 << d1 << endl;
						}
				}
			y1++;
			}
			else if ((y1 %100) > 13)
				y1 = y1+100-(y1%100);
		}
		printf("%d\n",ctr);
	}
	return 0;
}
