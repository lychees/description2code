#include<bits/stdc++.h>

using namespace std;

int main()
	{
		int T;
		cin>>T;
		while(T--)
			{	
				int N,L,a,b,flag;
				cin>>N>>L;
				vector<pair<int,int>>s;
				for(int i=0;i<N;i++)
					{
						cin>>a>>b;
						s.push_back(make_pair(a,b));
					}
				sort(s.begin(),s.end());
				flag=0;
				for(int i=0;i<s.size();i++)
					{
						int maxi=s[i].first+L;
						int curr_max=s[i].second;
						for(int j=0;j<s.size();j++)
							{
								if(s[j].first<=curr_max && s[j].second<=maxi && s[j].first>s[i].first)
									curr_max=max(curr_max,s[j].second);
							}
						if(curr_max==maxi)
							{
								flag=1;
								break;
							}
					}
				if(flag==1)
					cout<<"Yes"<<endl;
				else
					cout<<"No"<<endl;
			}
		return 0;
	}
