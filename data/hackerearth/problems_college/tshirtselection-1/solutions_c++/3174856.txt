#define MAX 100010
using namespace std;

//ifstream fin("input.cpp");
int N;
int A[MAX];
int r[MAX];

void fillr()
{
	map<int,int> mp;
	REP(i,N)
	{
		if(mp.find(A[i])==mp.end())
		{
			mp[A[i]]=i;
			r[i]=i;
		}
		else
		{
			r[i]=mp[A[i]];
			mp[A[i]]=i;
		}
	}
	r[N]=N-1;
}




int posl,posr;
int ans()
{
	int i=0;
	int res=1;posl=posr=0;
	REPS(j,1,N+1)
	{
		if(r[j]<j && i<=r[j])
		{
			if(j-i>res)
			{
				res=j-i;
				posl=i;posr=j-1;
			}
			i=r[j]+1;
		}
	}
	return res;
}

int main()
{
	int t;cin>>t;while(t--)
	{
		cin>>N;REP(i,N)cin>>A[i];fillr();ans();
		cout<<posl+1<<" "<<posr+1<<endl;
	}
}
