#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define mod 1000000007
ll a[1000006];
void pre()
{
	a[0] = 0;
	a[1] = 1;
	a[2] = 1;
	for(int i = 3;i<=1000004;i++){
		a[i] = (a[i-1]+a[i-2])%mod;
	}
	for(int i = 4;i<=1000004;i++){
		a[i] = (a[i]+a[i-1])%mod;
	}
}
int main()
{
	pre();
	int test;
	scanf("%d",&test);
	while(test--){
		int n;
		scanf("%d",&n);
		printf("%lld\n",a[n+2]);
	}
    return 0;
}
