#include <iostream>
#include <stdio.h>

using namespace std;

int compare (const void * a, const void * b){
  return ( *(int*)a - *(int*)b );
}

int main() {
	int n, q, t[100005], s[100005], k;

	long long int ans;

	scanf("%d %d", &n, &q);

	for(int i = 0; i < n; ++i) {
		scanf("%d", &t[i]);
	}

	for(int i = 0; i < n; ++i) {
		scanf("%d", &s[i]);
	}

	qsort(t, n, sizeof(int), compare);
	qsort(s, n, sizeof(int), compare);

	while(q--) {
		scanf("%d", &k);
		ans = 0;
		for(int j = 0; j < k; ++j) {
			ans += t[n - 1 - j];
		}

		printf("%lld\n", ans);
	}

	return 0;
}
