#include <iostream>
using namespace std;
 
int main()
{
    long int t,n,p,a,i;
   cin>>t;
   while(t--)
   {int z=0,x=0;
   	cin>>n>>p;
   	for (i=0;i<n;i++)
   	{
   		cin>>a;
   		if(a==p) z=1;
   		if(a>p) x+=1;
 
   	}
   	if(z==1)cout<<"YES "<<x<<endl;
   	else cout<<"NO 0"<<endl;
   }
}
