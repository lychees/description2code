#include <iostream>
#include<stdbool.h>
#include<string.h>
using namespace std;

int main()
{
    int t,m,n,i,j,count[10002],odd;
    bool f;
    cin>>t;
    while(t--)
    {
    	cin>>n>>m;
    	f=true;
    	memset(count,0,sizeof(count));  
    	while(m--)
    	{
    		cin>>i>>j;
    		count[i]++;
    		count[j]++;
    	}
    	odd=0;
    	for(i=1;i<=n;i++)
    	{
    		if(count[i]==0)
    		{
    			f=false;
    			break;
    		}
    		if(count[i]%2)
    		  odd++;
    	}
    	if(f &&(odd==0 || odd==2))
    	 cout<<"Yes\n";
    	else
    	 cout<<"No\n";
    	
    }
    return 0;
}
