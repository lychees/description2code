#include <iostream>
using namespace std;

int main()
{
	int t,d,wt,fare;
    cin>>t;
    for(t;t>0;t--)
    {
    	int fare=50;
    	cin>>d>>wt;
    	if(d>60)
    	{
    		cout<<"NA"<<endl;
    	}
    	else if(d-2>=0)
    	{
    		fare=fare+(wt*2)+((d-2)*12);
    		cout<<fare<<endl;
    	}
    	else
    	{
    		fare=fare+(wt*2);
    		cout<<fare<<endl;
    	}
    	
    }
    return 0;
}
