#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main() {
	ll t;
	cin>>t;
	while(t--){
		ll n;
		cin>>n;
		ll a[200010],max=0,min=30000000;
		for(int i=0;i<n;i++)
		{
			scanf("%lld",&a[i]);
			if(a[i]<=min)
				min=a[i];
			if(a[i]>=max)
				max=a[i];
		}
		if((max-min)%2==0)
			printf("No\n");
		else printf("Yes\n");	
	}
	
	return 0;
}
