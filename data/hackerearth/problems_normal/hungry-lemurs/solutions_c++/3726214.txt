#include <bits/stdc++.h>
using namespace std;
int n,k;
int main(){
    cin>>n>>k;
    int ans = 1000000009;
    for(int i=k;i>0;i--){
        ans = min(ans , (k-i)+min(n%i , i - (n%i)));
    }
    for(int i=k;i<=n;i++){
        ans = min(ans , (i-k)+min(n%i , i - (n%i)));
    }
    cout<<ans<<endl;
}
