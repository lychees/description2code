#include<bits/stdc++.h>
using namespace std;
#define MAX 1000006
#define ll long long int
int primes[MAX];
int lp[MAX];
int pr;
ll sum[MAX];
void siever(void)
{
	int i,j;
	pr = 0;
	sum[0]=2;
	for(i=2;i<MAX;++i)
	{
		if(lp[i] == 0)
		{
			primes[pr] = i;
			if(pr > 0) sum[pr] = sum[pr-1]+primes[pr];
			pr++;
			for(j=i;j<MAX;j+=i)
				lp[j]=1;
		}
	}
	
}
int main()
{
	ios_base::sync_with_stdio(false);
	int t;
	siever();
	cin>>t;
	while(t--)
	{
		int l,r;
		cin>>l>>r;
		int a,b;
		a = lower_bound(primes,primes+pr,l) - primes;
		b = lower_bound(primes,primes+pr,r) - primes;
		if(primes[b] != r) b--;
		if(a > 0) cout<<sum[b]-sum[a-1]<<endl;
		else cout<<sum[b]<<endl;
	}
}
