#define si(x) scanf("%d",&x)
#define sll(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
#define pi(x) printf("%d\n",x)
#define pll(x) printf("%lld\n",x)

#define M 1000000007
#define ll long long
#define mp make_pair
#define pb push_back
#define fr first
#define se second
using namespace std;
vector<ll> v[100069];
ll arr[20069],numi;
int vis[2069][2069]={{}};
ll dfs(int x,int prev,int num)
{
    ll yo=1,sum=0,mul=1,f=0;
    FOR(i,0,v[x].size())
    {
        if(v[x][i]!=prev&&arr[v[x][i]]>num)
        {
            ll ret=dfs(v[x][i],x,num);
            mul=(mul*(ret+1))%M;
        }
        else if(v[x][i]!=prev&&arr[v[x][i]]==num&&!vis[v[x][i]][numi])
        {
        	ll ret=dfs(v[x][i],x,num);
            mul=(mul*(ret+1))%M;
            vis[numi][v[x][i]]=1;
            vis[v[x][i]][numi]=1;
        }
    }
    yo=(yo*mul)%M;
    return (yo);
}
map<ll,ll> m,ans;
int main()
{
    ll n,x;
    sll(n);
    FOR(i,1,n+1)
        sll(arr[i]);
    FOR(i,0,n-1)
    {
        ll y;
        sll(x),sll(y);
        v[x].pb(y);
        v[y].pb(x);
    }
    FOR(i,1,n+1)
    {
    	numi=i;
    	ans[arr[i]]=(ans[arr[i]]+(dfs(i,-1,arr[i])))%M;
    }
    int q;
    si(q);
    while(q--)
    {
        int num;
        si(num);
        pll(ans[num]);
    }
    return 0;
}
