#include <bits/stdc++.h>
#include <cmath>

using namespace std;


bool cb (int s)
{
  for (int l=0; l<=10; l++)
	{
	  int qb = l*l*l;
	  if (s == qb || s == -qb)
		{
		  return true;
		}
	}
  return false;
}

bool is_perfect_cube(int n) {
    int root(round(cbrt(n)));
    return n == root * root * root;
}

int main()
{
  int n,k;
  int i,j;
  // int p,r;
  int cnt=0;

  cin >> n >> k;
  int m[n][n];

  for (i=0; i<n; i++)
	for (j=0; j<n; j++)
	  cin >> m[i][j];

  for (i=0; i<=n-k; i++)
	{
	  int s=0;
	  for (int p=0; p<k; p++)
		{
		  for (int r=0; r<k; r++)
			s += m[i+p][r];
		}

	  if (is_perfect_cube(s))
		cnt++;

	  //cout << s << endl;
	  
	  for (j=k; j<n; j++)
		{
		  for (int r=0; r<k; r++)
			{
			  s += m[i+r][j];
			  s -= m[i+r][j-k];
			}

		  //cout << s << endl;
	  
		  if (is_perfect_cube(s))
			cnt++;
		}
	}

  cout << cnt << endl;
  
  return 0;
}
