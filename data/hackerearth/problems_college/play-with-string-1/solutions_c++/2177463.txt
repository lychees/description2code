#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	char s[10000];
    	cin>>s;
    	int i,l,flag=0;
    	l=strlen(s);
    	for(i=1;i<l;i++,l--){
    		if(abs(s[i]-s[i-1])!=abs(s[l-1]-s[l-2])){
    			flag=1;
    			break;
    		}
    	}
    	if(flag==0)
    		cout<<"Funny"<<endl;
    	else
    		cout<<"Not Funny"<<endl;
    }
    return 0;
}
