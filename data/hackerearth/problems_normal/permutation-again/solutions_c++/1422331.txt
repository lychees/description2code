#include<iostream>
#include<cstdio>
#define ll long long
#include<cstring>
#include<algorithm>
#include<cmath>
using namespace std;
int main()
{
	ll t,n,ans;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
		if(n==1||n==2)
		{
			cout<<"1\n";
			continue;
		}
		else if(n==3)
		{
			cout<<"3\n";
			continue;
		}
		ans=n*n/2-1;
		cout<<ans<<endl;
		
	}
	return 0;
}
