#include <iostream>
#include<stdio.h>

using namespace std;

int main()
{
    int T;
    string N,M;
    scanf("%d",&T);
    while(T--)
    {
    	cin>> N >> M;
    	
    	if((N==M) || (N =="2"&& M =="4") || (N =="4"&& M =="2"))
    		printf("YES\n");
    	else
    		printf("NO\n");
    }
    return 0;
    
}
