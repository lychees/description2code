#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll a[1000050];
int main()
{
	a[1]=1;
	for(ll i =2;i<1000050;i++)
	{
		if(!a[i])
		{
			for(ll j = i*i;j<1000050;j=j*i)a[j]=1;
		}
	}
	
  ll test; cin>>test;
  while(test--)
  {
  	ll n; cin>>n;
  	int c=0;
  	for(ll i = 1;i<n;i++)
  	{
  		if(a[i]&&a[n-i])
  		{
  			c=1;break;
		  }
	  }
	  if(c)cout<<"Yes"<<endl;
	  else cout<<"No"<<endl;
  }
}
