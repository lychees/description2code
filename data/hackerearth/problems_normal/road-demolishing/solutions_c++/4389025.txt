#include<iostream>
#include<cstdio>
#include<cassert>
#include<math.h>
using namespace std;
int main()
{
	int t=0;
	cin>>t;
	assert(1<=t && t<=100000);
	while(t--)
	{
		long long int n,q;
		cin>>n>>q;
		assert(3<=n && n<=1000000);
		assert(3<=q && q<=n);
		q--;
	long long int  ans=n*n-(n%q)*((n+q-1)/q)*((n+q-1)/q);
		ans=ans-(q-n%q)*(n/q)*(n/q);
		ans/=2;
		ans=(n*(n-1))/2-ans;
		assert(ans>0);
		cout<<ans<<endl;
	}
	return 0;
}
