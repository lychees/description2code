#include<bits/stdc++.h>
using namespace std;
int is_valid(long long val,long long *arr,long long n,long long m)
{
    long long sum = 0;
    int count = 0;
    for (int i=0;i<n;i++) {
        sum += arr[i];
        if (sum >= val) {
           sum = 0;
           count++;
        }
    }
//if(sum>=val)
//count++;
    return count >= m;
}
int main()
{
   // freopen("in4.txt","r",stdin);
    //freopen("outt4.txt","w",stdout);
    long long t,n,m,arr[100010];
    scanf("%lld",&t);
//
 
    while (t--) {
          scanf("%lld %lld",&n,&m);
//cout<<endl<<endl;
          for (int i=0;i<n;i++) {
             scanf("%lld",&arr[i]);
          }
          long long l = 1,r = LLONG_MAX,ans=-1;
          long long mid = (r-l)/2+l;
          while (l<=r) {
               if(is_valid(mid,arr,n,m)) {
                    l=mid+1;
                    ans=max(ans,mid);
                } else {
                    r=mid-1;
                }
                mid = (r-l)/2+l;
          }
          cout<<ans<<endl;
    }
    return 0;
}
