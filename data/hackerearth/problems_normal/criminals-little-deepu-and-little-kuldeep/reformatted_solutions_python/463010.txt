T = int(raw_input())
for x in range(0, T):
    n = int(raw_input())
    a = [0]*n
    for i in range(0, n):
        a[i] = int(raw_input())
    a.sort()
    p = a[0]
    max = 1
    count = 1
    for i in range(1, n):
        if p == a[i]:
            count = count+1
            if max < count:
                max = count
        else:
            count = 1
        p = a[i]
    print max
