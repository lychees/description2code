#include <iostream>
using namespace std;

int main()
{
    int t,d,w;
    cin>>t;
    while(t--)
    {
    	cin>>d>>w;
    	int rp;
    	if(d>60)
    	{
    	 	cout<<"NA"<<"\n";
    	   continue;
    	}
    	if(d==0)
    	  rp=0;
    	else
        {
        	if(d<=2)
        	  rp=50;
        	else
        	 rp=50+(d-2)*12;
        }
        rp+=w*2;
        cout<<rp<<"\n";
    }
    return 0;
}
