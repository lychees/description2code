#include <bits/stdc++.h>

using namespace std;

int main()
{
	int q,t;
	
	long long n;
	
	cin >> q;
	
	assert(q>=1 && q<=10);
	
	while(q--)
	{
		cin >> n >> t;
		
		assert(n>=1 && n<=10000000000000000);
		assert(t>=1 && t<=50);
		
		while(t--)
		{
			if(n%2)
			{
				n -= ((n+1)/2);
				
			}
			else
			{
				n -= (n/2);
				
			}
			
			n -= (n/4);
			
		}
		
		cout << n << endl;
		
	}
    
    return 0;
}
