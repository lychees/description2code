#include <iostream>
#include <string>
using namespace std;

int main(){
    int tc;
    cin >> tc;
    while(tc--){
    	int cnt = 0;
    	int flag = 0;
    	string s;
    	cin >> s;
    	int length = s.length();
    	int sum = 0;
    	for(int i=0;i<length;i++){
    		if(s[i]=='O' and flag==0){
    			cnt+=1;
    		}
    		else if(s[i]=='X'){
    			flag = 1;
    			cnt = 0;
    		}
    		else if(s[i]=='*'){
    			sum+=cnt;
    			cnt = 0;
    			flag = 0;
    		}
    		if(i==length-1){
    			sum+=cnt;
    		}
    	}
    	cout << sum << endl;	
    }
}
