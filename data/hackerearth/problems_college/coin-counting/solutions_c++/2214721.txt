#include <iostream>
using namespace std;

void c3(int temp, int &count)
{
	while (temp >= 3 && count++)
		temp -= 3;
}

void c5(int temp, int &count)
{
	while (temp >= 5)
	{
		temp -= 5;
		count++;
		c3(temp, count);
	}
}

int main()
{
	int a, t;
	cin >> t; 
	while (t--)
	{
		cin >> a;
	    int count = 1;
	    if (a)
		 count = 1;
	    c3(a, count);
	    c5(a, count);
	    cout << count <<endl;
	}
	return 0;
}
