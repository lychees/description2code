#include<bits/stdc++.h>
using namespace std;
typedef long long int uli;
const int mx=1e6+10;
bool criba[mx];
int f[mx];
int main(){
   memset(criba,true,sizeof criba);
   criba[0]=criba[1]=false;
   for(int i=0;i<mx;i++){
      if(criba[i]){
         for(int j=i+i;j<mx;j+=i){
            criba[j]=false;
         }
      }
   }
   criba[1]=true;
   int n;
   scanf("%d",&n);
   memset(f,0,sizeof f);
   int ans=-1;
   for(int i=0;i<n;i++){
      int v;
      scanf("%d",&v);
      f[v]++;
      if(criba[v]){
         ans=max(ans,v);
      }
   }
   printf("%d\n%d\n",ans,f[ans]);
   return 0;
}
