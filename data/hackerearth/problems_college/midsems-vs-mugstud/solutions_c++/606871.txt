#include<iostream>

//using namespace std;

int main()
{
        int t;
        scanf("%d",&t);
        while(t--)
        {
                int n;
                scanf("%d",&n);
                int b[n][n][n],a[n][n][n],i,j,k,c1,c2,c3,sum=0;
                for(i=0;i<n;i++)
                        for(j=0;j<n;j++)
                                for(k=0;k<n;k++)
                                        scanf("%d",&b[i][j][k]);
                for(i=0;i<n;i++)
                {
                        for(j=0;j<n;j++)
                        {
                                //nu=0;
                                for(k=0;k<n;k++)
                                {
                                        a[i][j][k]=b[i][j][k];
                    if(k>=1) a[i][j][k]-=b[i][j][k-1];
                    if(j>=1) a[i][j][k]-=b[i][j-1][k];
                    if(k>=1 && j>=1) a[i][j][k]+=b[i][j-1][k-1];
                    if(i>=1) {
                        int p=i;
                        while(p>0) {
                            a[i][j][k]-=a[p-1][j][k];
                            p--;
                        }
                    }
                                }
                        }
                }
                for(i=0;i<n;i++)
                {
                        for(j=0;j<n;j++)
                        {
                                for(k=0;k<n;k++)
                                        printf("%d ",a[i][j][k]);
                                printf("\n");
                        }
                }
        }
        return 0;
}
