# from collections import defaultdict
s = list(raw_input())
f = ord('a')
h = [[] for _ in xrange(26)]
for i in xrange(len(s)):
    h[ord(s[i])-f].append(i)
for i in xrange(input()):
    k, x = raw_input().split()
    k = int(k)
    l = ord(x)-f
    s[h[l][k-1]] = '0'
    del h[l][k-1]
print ''.join(filter(lambda x: x != '0', s))
