#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main()
{
    int t, px, py, qx, qy, rx, ry;
    cin >> t;

    while (t--) {
        cin >> px >> py >> qx >> qy;
        rx = 2 * qx - px;
        ry = 2 * qy - py;

        cout << rx << " " << ry << endl;
    }
}
