#include<iostream>
#include<cstring>

using namespace std;

int main()
{
	string s;
	int i,j,cnt;
	
	cin >> s;
	 
	for(j = 0;j <  s.length();j++)
	{	
		cnt=0;
		for(i = j;i <  s.length();i++)
		{	
			if(s[i] == '2' || s[i] == '4' || s[i] == '6' || s[i] == '8')
			{
 		       cnt++;
			}
		}
		cout << cnt << " ";
	}
	cout << endl;	
	return 0;
}
