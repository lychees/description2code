#include <iostream>
using namespace std;

int main()
{
    int t,flag=0;
    cin>>t;
    string s;
    while(t--)
    {
    	cin>>s;
    	flag=0;
    	for(int i=0;i<s.length();i++)
    	{
    		
if(s[i]=='a'||s[i]=='e'||s[i]=='i'||s[i]=='o'||s[i]=='u'||s[i]=='A'||s[i]=='E'||s[i]=='I'||s[i]=='O'||s[i]=='U')
    		{cout<<s[i];flag=1;}
    		
    	}
    	if(flag==0)
    	{cout<<"No";}
    	
    	cout<<endl;
    }
    return 0;
}
