#include <iostream>
#include<math.h>
using namespace std;

int main()
{
   long int t,d,n;
   int flag;
   long long int x,y,i,cnt;
   cin>>t;
   while(t>0)
   {
   		n=1;
    	cin>>x;
    	cin>>y;
    	d=x-y;
    	cnt=0;
    	while(n<=(sqrt(d)))
    	{
    		flag=0;
    		if(d%n==0)
    		{
    			flag = 1;
    			if(n>y)
    			{
    				cnt++;
    			}
    		}
    		if(flag==1)
    		{
    			if(n!=d/n && d/n>y)
    			{
    				cnt++;
    			}
    		}
    		n++;
    	}
    	
    	if(x==y)
		{
			cnt=-1;
		}
    	cout<<cnt<<endl;
    	t--;
   }
   return 0;
}
