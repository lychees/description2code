#include <iostream>
using namespace std;

int main()
{
    long long int t,n,m,x1,x2,y1,y2,p,z,a;
    cin>>t;
    while(t--)
    {
    	cin>>n>>m;
    	cin>>x1>>y1;
    	cin>>x2>>y2;
    	cin>>p;
    	if(x2>=x1)
    	{if((x2-x1)>=(n-x2+x1))
    	z=p*(n-x2+x1);
    	else 
    	z=p*(x2-x1);
    	}
    	else 
    	{
    		if((x1-x2)>=(n-x1+x2))
    		z=p*(n-x1+x2);
    		else z=p*(x1-x2);	
    	}
    	if(y2>=y1)
    	{
    		if((y2-y1)>(m-y2+y1))
    		a=p*(m-y2+y1);
    		else a=p*(y2-y1);
    	}
    	else 
    	{
    		if((y1-y2)>(m-y1+y2))
    		a=p*(m-y1+y2);
    		else a=p*(y1-y2);
    	}
    	if(a+z>1000)
    	cout<<a+z<<"\n"<<"Let go of the group.\n";
    	else cout<<a+z<<"\n"<< "You saved the group.\n";
    }
    return 0;
}
