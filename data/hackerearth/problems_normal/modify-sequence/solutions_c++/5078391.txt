#include <iostream>
using namespace std;

int main()
{
    int n,i;
    cin>>n;
    long long a[n];
    for(i=0;i<n;i++){
    	cin>>a[i];
    	
    }
    long long sum=a[0];
    for(i=1;i<n;i++){
    	sum=a[i]-sum;
    	if(sum<0){
    		cout<<"NO";
    		break;
    	}
    }
    if(sum==0){
    	cout<<"YES";
    }
    else if(sum>0){
    	cout<<"NO";
    }
    return 0;
}
