#include <iostream>
#include<cmath>
using namespace std;
typedef long long ll;
int main()
{
    int t; cin >> t;
    while(t--)
    {
    	ll x,y,z; cin >> z >> x >> y;
    	ll ans=0,s=sqrt(z);
    	for(ll i=1;i<=s ;i++)
    	{
    		if(z%i==0 && z/i<=y && i<=x)
    			ans++;
    		if(z%i==0 && i<=y && z/i<=x && i!=z/i)
    			ans++;
    	}
    	cout << ans << endl;
    }
    return 0;
}
