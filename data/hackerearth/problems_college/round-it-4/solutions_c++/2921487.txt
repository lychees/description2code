#include<bits/stdc++.h>
using namespace std;
int main()
{ int t,n,b;
  cin>>t;
  while(t--)
  { cin>>n>>b;
    
    if(b%2!=0)
    {if(n%b<=b/2 )
    cout<<n-n%b<<"\n";
    else 
    cout<<n-n%b+b<<"\n";
    }
    
    if(b%2==0)
    {if(n%b<b/2 )
    cout<<n-n%b<<"\n";
    else 
    cout<<n-n%b+b<<"\n";
    }
  }
	return 0;
}
