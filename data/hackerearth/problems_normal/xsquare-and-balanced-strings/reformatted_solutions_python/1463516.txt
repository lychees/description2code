N = int(raw_input())

count_occ = {}
while N > 0:
    N -= 1
    input_string = raw_input()
    count_occ = {}
    for c in input_string:
        if c in count_occ:
            count_occ[c] += 1
        else:
            count_occ[c] = 1
    is_balanced = True
    for count in count_occ.itervalues():
        if count % 2 == 1:
            print -1
            is_balanced = False
            break

    if is_balanced:
        print 1
