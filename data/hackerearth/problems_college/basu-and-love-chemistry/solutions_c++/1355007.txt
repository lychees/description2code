using namespace std;

#define ll long long
#define vt vector
#define mod 1000000007

int main()
{
	string n;
	cin>>n;
	ll val = n[0]-'0';
	ll temp = n[0] - '0';
	for(int i=1;i<n.length();i++)
	{
		temp = temp*10 + (i+1)*(n[i]-'0');
		temp %= mod;
		val = (val+temp)%mod;
	}
	cout << val << endl;
	return 0;
}
