#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int hash[257]={0}, count=0;
    	char s[1000009];
    	cin>>s;
    	for(int i=0; s[i]; i++)
    	{
    		int j=(int)s[i];
    		if(hash[j]==0)
    		{
    			hash[j]=1;
    			count++;
    		}
    	}
    	long long result=(count*(count+1))/2;
    	cout<<result%1000000007<<endl;
    }
    return 0;
}
