#include <iostream>

// https://www.hackerearth.com/problem/algorithm/shortest-way/description/

using namespace std;

long long shortest_way(long long a, long long b) {
	if (b == 1) 
		return a - 1;
	else 
		return a / b + shortest_way(b, a % b);
}

void run_test() {
	long long a, b; 
    cin >> a >> b;
	cout << shortest_way(a, b) << '\n';
}

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    
    int T;
    cin >> T;
    for (int i = 0; i < T; ++i)
        run_test();
    return 0;
}
