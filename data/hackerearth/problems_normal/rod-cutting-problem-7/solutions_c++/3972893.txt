#include <iostream>
using namespace std;

int main()
{
    long long int dp[34];
    dp[0]=1;
    for(int i=1;i<34;i++)
    {
    	dp[i] = 2*dp[i-1];
    }
    
    int t;
    cin>>t;
    while(t--)
    {
    	long long int n;
    	cin>>n;
    	for(int i=33;i>=1;i--)
    	{
    		if(n>=dp[i]-1)
    		{
    			cout<<i-1<<"\n";
    			break;
    		}
    	}
    }
}
