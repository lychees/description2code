for _ in range(int(raw_input())):
	a, b = raw_input().strip().split()
	x = map(int, a)[::-1]
	y = map(int, b)[::-1]
	ans = 0
	carry = 0
	for i in range(len(a)):
		if x[i] + y[i] + carry > 9:
			ans += 1
			carry += 1
		else:
			carry = 0
		
	print  ans if ans else "No",
	print "carry operation" if ans < 2 else "carry operations"
