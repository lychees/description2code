
def check(n, a):
    i = 0
    left = 0
    right = sum(a[1:])
    over = False
    while not over:
        if left == right:
            return "YES"

        left += a[i]
        i+=1
        if i == n-1:
            right = 0
            over= True

        right -= a[i]

    return "NO"

if __name__=='__main__':
    t = int(raw_input())
    for x in xrange(t):
        print(check(int(raw_input()), map(int, raw_input().strip().split())))
