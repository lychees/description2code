#include<iostream>
#include<cstdio>
 
using namespace std;
 
int arry[1000000]={0},Length=0;
int main()
{int t,n,j,i,myint;

Length=0;
cin>>n;
for( j=0;j<n;j++)
{scanf("%d",&myint);
i=myint-j;
if(i>0)
{int left=0,right=Length-1,y=-1;
while(left<=right)
{int mid=(left+right)>>1;
if(arry[mid]<=i)
left=mid+1;
else {
y=mid;
right=mid-1;
}}arry[(y >= 0)?(y):(Length++)] =i;
}}cout<<n-Length<<endl;}
