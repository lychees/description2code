#include <bits/stdc++.h>

using namespace std;

int checkPrime(int n)
{
	if(n==2 || n==3)
		return 1;
	if(n%2==0)
		return 0;
	if(n%3==0)
		return 0;

	int i = 5,w = 2;
    while (i * i <= n)
    {
        if (n % i == 0)
            return 0;
        i += w;
        if(i>=n/2+1)
        return 0;
       // w = 6 - w;
    }
    return 1;
	
}
int main()
{
	int t;
	cin>>t;
		while(t--)
			{
				int m,n;
				cin>>m>>n;
				int count=0;
				for(int i=m;i<=n;i++)
				{
					int r=checkPrime(i);
					if(r==1)
						count++;
				}
                if(m==1)
				cout<<count-1<<endl;
                else
                    cout<<count<<endl;
			}
	return 0;
}
