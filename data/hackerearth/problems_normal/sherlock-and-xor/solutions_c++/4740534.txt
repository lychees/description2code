#include <iostream>
using namespace std;

int main()
{
    int size;
    cin>>size;
    while(size -- ){
    	unsigned long long int tc;
    	cin>>tc;
    	unsigned long long int odd = 0, even = 0;
    	for(int i = 0; i< tc; i++){
    	    unsigned long long int num;
    	    cin>>num;
    	    if(num % 2 == 0)
    	        even = even + 1;
    	    else
    	        odd = odd + 1;
    	}
    	cout<<odd * even<<"\n";
    }
    return 0;
}
