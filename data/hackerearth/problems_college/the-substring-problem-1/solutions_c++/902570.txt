#include<iostream>
using namespace std;
//substring

int main()
{
 int T,n,maxm;
 cin>>T;
 string str;
 bool mark[26];
 int count[26];
 char ch;
 for(int i = 0; i<26; i++)
 count[i]=0;
 
 while(T--)
 {
 cin>>n;
  for(int i = 0; i<26; i++)
 count[i]=0;
 while(n--)
 {
 cin>>str;
 for(int i = 0; i<26; i++)
 mark[i]=false;
 
 for(int i = 0; i<str.length(); i++)
 mark[(int)(str[i]-'a')]=true;
 
 for(int i = 0; i<26; i++)
 {
 if(mark[i]==true)
 count[i]++;
 }   
 }//n strings
 maxm=0;
 for(int i = 0; i<26; i++)
 {
 if(count[i]>maxm)
 {
 maxm=count[i];
 ch = i+'a';
 }
 }
 cout<<ch<<"\n";
}//Test cases
 
  
    return 0;
}
