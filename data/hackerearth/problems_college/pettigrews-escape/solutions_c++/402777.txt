#include<iostream>
using namespace std;

long int path(int m[87][87],int x,int y)
{
	if(m[x][y]==0)
	return 0;
	else if(x==1 && y==1)
	{   
	  return 1;
	}
	else
	{  
		
		
		m[x][y]=0;
	     int p=path(m,x-1,y)+path(m,x,y-1)+path(m,x+1,y)+path(m,x,y+1);
	     
	     m[x][y]=1;
	    return p;
	}
}


int main()
{
	int m[87][87]={0};
	int n,a;
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=n;j++)
		{cin>>a;
		  m[i][j]=1-a;
		}
	}
	
	
	
	cout<<path(m,n,n)<<endl;
	return 0;
}
