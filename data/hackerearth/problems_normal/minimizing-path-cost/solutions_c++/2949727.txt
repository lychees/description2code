#include <iostream>
#include <string>
#include <map>
#include <climits>
using namespace std;

int main()
{
    int n, m, d;
    cin >> n >> m;
    string a , b;
    int dist[n+1][n+1];
    map<string,int> has;
    int cnt = 0;
    for(int i = 0; i < n+1 ; i++) {
    	for(int j = 0 ; j < n+1 ; j++) {
    		dist[i][j] = INT_MAX;
    	}
    }
    for(int i = 0 ; i < n ; i++) {
    	cin >> a;
    	has[a] = ++cnt;
    	dist[has[a]][has[a]] = 0;
    }
    for(int i = 0 ; i < m ; i++) {
    	cin >> a >> b >> d;
    	dist[has[a]][has[b]] = d;
    	dist[has[b]][has[a]] = d;
    }
    for(int k = 1; k <= n ; k++) {
    	for(int i = 1 ; i <= n; i++) {
    		for(int j = 1 ; j <= n; j++) {
    			if(dist[i][k]==INT_MAX || dist[k][j]==INT_MAX) continue;
    			//cout << dist[i][j] << " " << dist[i][k] << " " << dist[k][j] << endl;
    			if(dist[i][j] > dist[i][k] + dist[k][j]) {
    				dist[i][j] = dist[i][k] + dist[k][j];
    			}
    		}
    	}
    }
    int q;
    cin >> q;
    while(q--) {
    	cin >> a >> b;
    	cout << dist[has[a]][has[b]] << endl;
    }
    return 0;
}
