#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
 
int main() {
	int t;
	cin>>t;
	while(t--)
	{
		char a[100005]="NULL";
		cin>>a;
		int i=0,j,l,sum=0;
		l=strlen(a);
		while(i<=l)
		{
		if(a[i]=='0' ||a[i]=='6' ||a[i]=='9')
			sum=sum+6;
			else if(a[i]=='1')
			sum=sum+2;
			else if(a[i]=='2' ||a[i]=='3' ||a[i]=='5')
			sum=sum+5;
			else if(a[i]=='4')
			sum=sum+4;
			else if(a[i]=='7')
			sum=sum+3;
			else if(a[i]=='8')
			sum=sum+7;
			i++;
 
		}
		cout<<sum<<endl;
	}
	return 0;
}
