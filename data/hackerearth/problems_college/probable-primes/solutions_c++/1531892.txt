#include <iostream>
#include <math.h>
#include <stdio.h>
#include <iomanip>
using namespace std;

int main()
{
	int a[1000001]={0,1,0,0};
	for(int i=2;i<1000001;i++)
		for(int j=2;i*j<1000001;j++)
			a[i*j]=1;
	for(int i=1;i<1000001;i++)
	{
		if(a[i] == 1)
			a[i] = a[i-1];
		else
			a[i] = a[i-1]+1;
	}
    int t;
    cin>>t;
    while(t--)
    {
    	int l,r;
    	cin>>l>>r;
    	printf("%.6f\n",(a[r]-a[l-1])/(float)(r-l+1));
    }
    return 0;
}
