#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int tc,n;
	cin>>tc;
	while(tc--)
	{
		int a,b,i;
		cin>>a>>b>>n;
		int p[n];
		for(i=0;i<n;i++)
			cin>>p[i];
		int c=0,d=0,sum=0;
		sort(p,p+n);
		for(i=0;i<n;i++)
		{
			sum+=p[i];
			if(sum<=a)
				c+=1;
			if(sum<=b)
				d+=1;
			if(sum>a||sum>b)
				break;
		}
		if(c>d)
			cout<<"Raghu Won";
		else if(c<d)
			cout<<"Sayan Won";
		else
			cout<<"Tie";
		cout<<endl;
	}
	return 0;
}
