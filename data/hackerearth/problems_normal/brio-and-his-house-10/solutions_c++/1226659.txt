#include <iostream>
#include <cmath>
typedef double u;
using namespace std;

int main()
{
     u a,b,c,pi=3.1415,peri_meter=0,circum_radius,temp;
     int t,i;
     u denominator;
     cin>>t;
     cout.precision(4);
     while(t--)
     {

         cin>>a>>b>>c;
         peri_meter = a + b + c;
         temp= a*b*c;

         denominator = peri_meter*(peri_meter-(2*a))*(peri_meter-(2*b))*(peri_meter-(2*c));
         denominator =sqrt(denominator);
         circum_radius = temp / denominator ;

         temp= pi*circum_radius*circum_radius;
         cout <<fixed<<temp<< endl;
         //cout<<fixed<<setprecision(3)<<temp<<endl;
     }
}
