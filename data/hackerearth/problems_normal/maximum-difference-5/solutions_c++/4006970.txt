#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;

int main()
{
	int n;
    cin>>n;
    int arr[n];
    for(int i=0; i<n; i++)
    	cin>>arr[i];
    sort(arr, arr+n);
    
    int result_arr1[n];
    int result_arr2[n];
    
    int l=0, r=n-1;
    int k=0;
    while(l < r)
    {
    	result_arr1[k++] = arr[r];
    	result_arr1[k++] = arr[l];
    	l++;
    	r--;
    }
    result_arr1[k] = arr[l];
    
    
    l=0, r=n-1;
    k=0;
    while(l < r)
    {
    	result_arr2[k++] = arr[l];
    	result_arr2[k++] = arr[r];
    	l++;
    	r--;
    }
    
    result_arr2[k] = arr[l];
    
    long long int sum1 = 0, sum2 = 0;
    
    for(int i=1; i<n; i++)
    {
    	sum1 = sum1 + abs(result_arr1[i] - result_arr1[i-1]);
    	sum2 = sum2 + abs(result_arr2[i] - result_arr2[i-1]);
    }
    if(sum1>= sum2)
    	cout<<sum1<<endl;
    else
    	cout<<sum2<<endl;
    return 0;
}
