#include <iostream>
#include <stdio.h>
#include <math.h>
using namespace std;

int main()
{
    int T;
    cin>>T;
    for(int i=0; i<T; i++)
    {
    	float P;
    	int t;
    	cin>>P>>t;
    	float x = (1440/t);
    	if(P==0)
    	{
    		printf("0.0000\n");
    	}
    	else
    	{
    	float y = 1 - pow((double)(1-P), (double) (1/x));
    	printf("%.4f\n", y);
    	}
    }
}
