#include<bits/stdc++.h>
using namespace std;
int main()
{
	string str;
	cin>>str;
	int len = str.length();
	int i=0,j=1,k=0;
	string s = str + str;
	while(i+k < s.size() && j+k <s.size())
	{
		if(s[i+k] == s[j+k])
		{
			k++;
		}
		else
		if(s[i+k]<s[j+k])
		{
			i = i + k +1;
			if(i<=j)
			{
				i = j + 1;
			}
			k=0;
		}
		else
		{
			j = j + k + 1;
			if(j<=i)
			{
				j = i + 1;
			}
			k=0;
		}
	}
	int  p = min(i,j);
	for(i = p ;i<str.length();i++)
	{
		cout<<str[i];
	}
	for(i=0;i<p;i++)
	cout<<str[i];
}
