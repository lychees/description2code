#include <iostream>
using namespace std;


int gcd( int u,  int v)
{
    
    if (u == v)
        return u;
 
    if (u == 0)
        return v;
 
    if (v == 0)
        return u;
 
    
    if (~u & 1) 
    {
        if (v & 1) 
            return gcd(u >> 1, v);
        else 
            return gcd(u >> 1, v >> 1) << 1;
    }
 
    if (~v & 1)
        return gcd(u, v >> 1);
 
    
    if (u > v)
        return gcd((u - v) >> 1, v);
 
    return gcd((v - u) >> 1, u);
}
  
int max(int x,int y){return x>y?x:y;}
int min(int x,int y){return x<y?x:y;}


int main()

{
	int t;
	int a,b,c;
	
	cin>>t;
	while(t--)
	{
		scanf("%d", &a);
		scanf("%d", &b);
		scanf("%d", &c);
		int m=max(a,b);
		int maxv=max(m,c);
		int mn=min(a,b);
		int min1=min(c,mn);
		int min2;
		if(a!=min1 && a!=maxv) min2=a;
		else if(b!=min1 && b!=maxv) min2=b;
		else min2=c;
		
		if(gcd(a,b)==1 && gcd(a,c)==1 && gcd(c,b)==1)
		{
			if((min1)*(min1)+(min2)*(min2)==(maxv)*(maxv)) cout<<"YES"<<endl;
			else cout<<"NO"<<endl;
			
		}
		else
		cout<<"NO"<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
