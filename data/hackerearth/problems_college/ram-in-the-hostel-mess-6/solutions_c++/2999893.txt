/*LCA between two nodes u and v , u and v >=1 */
#include<bits/stdc++.h>
using namespace std;

#define lli long long
#define pb push_back
#define mp make_pair
#define gc getchar_unlocked
#define lld "%I64d"
#define DEBUG(x) cout<<">value ("<<#x<<") : "<<x<<endl;
#define mod 1000000007

const int lmt = 22;
lli dp[lmt];

int main(){
    dp[0] = 1;
    dp[1] = 2;
    for(int i=2;i<lmt;i++)
        dp[i] = dp[i-1] + dp[i-2];
    int t,n;
    scanf("%d",&t);
    while(t--){
        scanf("%d",&n);
        printf("%lld\n",dp[n]*dp[n]);
    }
    return 0;
}
