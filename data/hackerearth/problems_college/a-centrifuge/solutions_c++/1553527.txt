#include <iostream>
using namespace std;
#include <cmath> 
int t;
int m;
int n;
 
int main()
{
	 cin>>t;
	 while(t--){
	 	cin>>m>>n;
	 	int ans=1;
	 	if(n==0){
	 		cout<<"0"<<endl;
	 		continue;
	 	}else if(m<n)
			ans = ceil((float)2*n/m);
		else
			ans=2;
		cout<<ans*5<<endl;
	 }
    return 0;
}
