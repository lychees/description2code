#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cmath>

using namespace std;

int main()
{
        int t=0;
        scanf("%d",&t);
        while(t--)
        {
                int N;
                scanf("%d",&N);
                int x,y;

                scanf("%d %d",&x,&y);
                double sum=0,max=-1;

                while(N--)
                {
                        int temp;
                        scanf("%d",&temp);

                        if(temp>max)
                                max=temp;
                        sum = sum + (double)temp;
                }

                double shortest_path=sqrt(x*x +y*y);
                sum= sum+shortest_path;

                if(shortest_path>max)
                        max= shortest_path;

                if(max<=(sum/2))
                        printf("Able\n");
                else
                        printf("Not able\n");

        }
return 0;
}
