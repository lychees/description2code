#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <string>
#include <vector>

using namespace std;

long long mas[2222];

int main() {
    ios::sync_with_stdio(0);
    int n;
    long long k;
    cin >> n >> k;
    for(int i = 0; i < n; ++i)
        cin >> mas[i];
    sort(mas, mas+n);

    long long ans = 0;
    for(int i = 0; i < n; ++i)
        for(int j = 0; j < n; ++j)
        if (i != j) {
            if (mas[i] * mas[j] * mas[0] > k) continue;
        
            int p = 0;
            for(int s = 12; s >= 0; --s)
            if (p + (1 << s) < n)
                if (mas[i] * mas[j] * mas[p + (1 << s)] <= k)
                    p += (1 << s);
            
            ans += p + 1 - (i <= p) - (j <= p);
        }
    cout << ans / 6 << "\n";
    return 0;
}
