#include <iostream>
using namespace std;

int main()
{
    int n,i;
    cin>>n;
    int ca[n],id[n];
    for(i=0;i<n;i++)
    	cin>>ca[i];
    for(i=0;i<n;i++)
    	cin>>id[i];
    	int j=0,temp,count=0;
    	for(j=0;j<n;j++)
    	{
    while(ca[j]!=id[j])
    {
    	temp=ca[j];
    	for(int k=j;k<n-1;k++)
    		ca[k]=ca[k+1];
    	ca[n-1]=temp;
    
    	count++;
    	
    }
    	}
    cout<<count+n<<endl;
    return 0;
}
