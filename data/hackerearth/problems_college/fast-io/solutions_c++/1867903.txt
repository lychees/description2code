#include <iostream>
#include <cstdio>
#define gc getchar_unlocked
using namespace std;

void scant(long &x) {
	register long c = gc();
	x=0;
	for(;(c<48||c>57);c=gc());
	for(;c>47&&c<58;c=gc()) {
		x= (x<<1) + (x<<3) +c -48;
	}
}
int main()
{
    long N,K;
    int count=0;
    scant(N);
    while(N--) {
    	scant(K);
    	if(K<300000000) count++;
    }
    cout<<count;
    return 0;
}
