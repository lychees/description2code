#include <iostream>
using namespace std;

int main()
{
    int t;
	cin >> t;
	while (t--)
	{
		int n;
		cin >> n;
		int i;
		int a[1000];
		int sum = 0;
		int t = 10001;
		for (i = 0; i < n; i++)
		{
			cin >> a[i];
			sum += a[i];
			if (a[i] < t)
			{
				t = a[i];
			}
		}
		cout << t - 1 << " " << sum - n <<endl;
	}
    return 0;
}
