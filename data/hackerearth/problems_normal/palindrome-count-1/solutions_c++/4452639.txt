#include <bits/stdc++.h>
using namespace std;

int main()
{
    int i,j,l;
    char str[60];
    scanf("%s",str);
    int n=strlen(str);
    bool dp[55][55];
    for(i=0;i<n;i++)
    dp[i][i]=1;
    int ans=n;
    for(l=2;l<=n;l++)
    {
    	for(i=0;i<n-l+1;i++)
    	{
    		j=i+l-1;
    		if(str[i]==str[j])
    		dp[i][j]=1;
    		else dp[i][j]=0;
    		if(l!=2)
    		{
    			dp[i][j]=dp[i][j]&&dp[i+1][j-1];
    		}
    		if(dp[i][j])
    		ans++;
    	}
    }
    printf("%d\n",ans);
    return 0;
}
