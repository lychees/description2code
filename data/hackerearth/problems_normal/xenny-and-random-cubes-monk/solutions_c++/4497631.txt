#include <iostream>
#include <string>
using namespace std;

using ull = unsigned long long;
constexpr ull mod = 1000000007;

constexpr unsigned lettn = 'z' - 'a' + 1;
unsigned lett[10 * lettn] = {};

char readc() {
	while (true) {
		char c;
		cin >> c;
		if (c >= 'a' && c <= 'z')
			return c;
	}
}

int n, k;
string s;

ull results[2048] = {};

ull getcount(int i, unsigned used) {
	if (i == k)
		return 1;
		
	if (results[used])
		return results[used] - 1;
		
	ull res = 0;
	for (int j = 0; j < n; ++j) 
		if (!(used & (1 << j)) && lett[j * lettn + (s[i] - 'a')])
			res += lett[j * lettn + (s[i] - 'a')] * getcount(i + 1, used | (1 << j));
	
	res %= mod;
	
	results[used] = res + 1;
	return res;
}

int main() {
	ios_base::sync_with_stdio(0);
    cin.tie(0);
    
    cin >> n >> k;
    
    for (int i = 0; i < n; ++i) 
    	for (int j = 0; j < 6; ++j) {
    		char c = readc();
    		++lett[i * lettn + (c - 'a')];
    	}
    
    
    cin >> s;
    cout << getcount(0, 0) << '\n';
    return 0;
}
