#include<bits/stdc++.h>
using namespace std;

#define max(a,b) (a)>(b)?(a):(b)
#define min(a,b) (a)>(b)?(b):(a)

#define rep(i,a,b) for(i=a;i<b;i++)
#define F first
#define S second
#define pb push_back
#define sqr(a) (a)*(a)
#define Pi 3.141592653589793
#define MOD 1000000007

// For i/o
#define sd1(a) scanf("%d",&a)
#define sd2(a,b) scanf("%d %d",&a,&b)
#define sd3(a,b,c) scanf("%d%d%d",&a,&b,&c)
#define sf scanf
#define pf printf
// For debugging
#define deb(a) printf("deb%d\n",a)

int i,j;

int main()
{
	int t;
	sd1(t);
	while(t--)
	{
		map <int,int> mp;
		int n;
		sd1(n);
		rep(i,0,n)
		{
			int a,b;
			sd2(a,b);
			mp[a] = mp[a] + 1;
			mp[b] = mp[b] - 1;
		}
		int ans = 0;
		int cnt = 0;
		for(auto it = mp.begin();it!=mp.end();it++)
		{
			cnt += (*it).second;
			if(cnt > ans)
				ans = cnt;
		}
		pf("%d\n",ans);
	}
	return 0;
}
