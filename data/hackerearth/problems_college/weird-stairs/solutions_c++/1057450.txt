#include <iostream>
using namespace std;
int main()
{
    int t;
    cin >> t;
    while(t--)
    {
        long long int w,n;
        cin >> w >> n;
        int arr[n+1];
        arr[0]=0;
        int last=0;
        long long int total=0;
        for(int i=1;i<=n;i++)
        {
            cin >> arr[i];
            if(arr[i]==1)
            {
                total= total + (2*i - arr[i-1]);
                arr[i]=2*i;
            }
            else
            {
                total = total + 1;
                arr[i]=arr[i-1]+1;
            }
        }
        total = total + (long long int)w*(long long int)n;
        cout << total << "\n";
    }
    return 0;
}
