#include<bits/stdc++.h>
using namespace std;




void printSplendid(int n,int i,int j,int& count,int **a){
	
	if(n==1){
		a[0+i][0+j] = count;
		count++;
		a[0+i][1+j] = count;count++;
		a[1+i][0+j] = count;count++;
		a[1+i][1+j] = count;count++;
		
		return;
	}
	else{
		int plusAmount = pow(2,n-1);
		printSplendid(n-1,i,j,count,a);
		printSplendid(n-1,i,j+plusAmount,count,a);
		printSplendid(n-1,i+plusAmount,j,count,a);
		printSplendid(n-1,i+plusAmount,j+plusAmount,count,a);
	}
}

int main(){
	int count=1;
	int n;
	cin>>n;
	int size = pow(2,n);
	int **a = new int*[size];
	
	for(int i=0;i<pow(2,n);i++)
	a[i] = new int[size];
	
	printSplendid(n,0,0,count,a);
	for(int i=0;i<size;i++){
		for(int j=0;j<size;j++)
		cout<<a[i][j]<<" ";
		cout<<endl;
	}
}
