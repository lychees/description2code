#include <iostream>
#include <cmath>
#define ull unsigned long long
using namespace std;
const ull c=(ull)6;
int b[20];
int count[10]={0,1,2,2,2,3,3,4,4,5};
ull compute(int i)
{
ull temp=(ull)count[b[i]];
if(i==0)
return temp;
ull temp2=(ull)pow(c,i);
temp=temp*temp2;
if(b[i]==2 || b[i]==3 || b[i]==5 || b[i]==7)
return temp;
return temp+compute(i-1);
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    ull n;
    scanf("%llu",&n);
    if(n==0)
    {cout<<0<<"\n";continue;}
    ull temp=n;
    int k=0;
    bool flag=false;
    while(temp!=0)
    {
    b[k]=temp%10;
    temp=temp/10;
    if(b[k]==2 || b[k]==3 || b[k]==5 || b[k]==7)
    flag=true;
    k++;
    }
    if(!flag)
    cout<<n-compute(k-1)<<"\n";
    else
    cout<<n-compute(k-1)+1<<"\n";
    }
    return 0;
}
