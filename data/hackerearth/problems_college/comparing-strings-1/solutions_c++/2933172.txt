#include <iostream>
#include<string>
#include<algorithm>
using namespace std;

int main()
{int tc;
cin>>tc;
while(tc--){	
	string s1,s2;
	cin>>s1>>s2;
	
	sort(s1.begin(),s1.end());
	sort(s2.begin(),s2.end());
	


	
	int n1,n2;
	n1=s1.size();
	n2=s2.size();
		int flag=0;
	for(int i=0;i<n1;i++)
	{
	if(s2.find(s1[i])!=string::npos)
	{flag=1;
	break;
	}}
	

	
	
	if(flag)
	cout<<"YES"<<endl;
	else
	cout<<"NO"<<endl;}
    //cout << "Hello World!" << endl;
    return 0;
}
