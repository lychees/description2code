LL dp[1<<21 + 1];
LL d[100];
LL val[100];

LL func(LL ind, LL mask) {
           
           if(ind == n)
                  return 0;
           
           if(dp[mask] != (LL)-99999999*(LL)9999999)
                       return dp[mask];
           
           LL v = 0;
           
           LL i,j,k;
           
          
           k = (LL)-9999999*(LL)9999999;
           for(i=0;i<n;i++) {
                            if(!(mask&(1<<i))) {
                                              
                                           v= n*val[i] + i*d[i]*ind*(ind+1)/2-n*i*d[i]*ind;
                                               k = max(k,v+func(ind+1,mask|(1<<i)));
                                              
                            }
           }
           return dp[mask] = k;
}
