#include<iostream>
#include<list>
#include<queue>
using namespace std;
int main()
{
int v,e;
cin>>v>>e;
list<int>*adj;
adj=new list<int>[v];
while(e--)
{
int a,b;
cin>>a>>b;
adj[a].push_back(b);
adj[b].push_back(a);	
}
bool visited[v];
int value[v];
for(int i=0;i<v;i++)
{ value[i]=0;
visited[i]=false;
}
value[0]=0;
queue<int>q;
list<int>::iterator it;
for(it=adj[0].begin();it!=adj[0].end();it++)
{
q.push(*it);
value[*it]=1;
visited[*it]=true;
}	
while(!q.empty())
{
int node=q.front();
q.pop();
list<int>::iterator i;
if(node)
for(i=adj[node].begin();i!=adj[node].end();i++)
{
if(visited[*i]==false)
{
q.push(*i);
value[*i]=value[node]+1;
visited[*i]=true;
}
}
}
for(int i=1;i<=v-1;i++)
{
if(value[i]>0)
cout<<value[i]<<endl;
else
cout<<"-1"<<endl;	
}	
}
