#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    int index;
    int N;
    while(t--)
    {
    	cin>>index;
    	cin>>N;
        while (N--> 0) {
            if (index == 0) index = 1;
            else if (index == 1) index = 0;
            else if (index == 2) index = 1;
        }
        cout<<index<<endl;
    }
    return 0;
}
