#include <bits/stdc++.h>

using namespace std;

const int MAXN = 1e6;
int prime[MAXN + 1];

void sieve() {
	prime[0] = 1;
	prime[1] = 1;
	for (size_t i = 2; i <= MAXN; i++) {
		if (!prime[i]) {
			prime[i] = i;
			for (size_t j = i; j * i <= MAXN; j++) {
				prime[j * i] = i;
			}
		}
	}
}

int phi(int n) {
	int res = 1;
	while (n > 1) {
		int p = prime[n], e = 0;
		while (n % p == 0) {
			n /= p;
			e++;
		}
		res *= (p - 1) * pow((double) p, e - 1);
	}
	return res;
}

int main() {
	sieve();
	int t;
	scanf("%d", &t);
	for (int tt = 1; tt <= t; tt++) {
		int n;
		scanf("%d", &n);
		printf("%d\n", phi(n));
	}
	return 0;
}
