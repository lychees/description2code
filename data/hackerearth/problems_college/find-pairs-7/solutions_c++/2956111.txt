#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

int main()
{
	int t;
	long int n, a;
	
	cin >> t;
	
	while(t--)
	{
		cin >> n;
		long int answer = n;
		vector<long int> A(n);
		
		for(int i=0;i<n;i++)
		{
			cin >> a;
			A[i] = a;
		}
		
		for(int i=1;i<n;i++)
		{
			for(int j=0;j<i;j++)
			{
				if(A[i] == A[j])
					answer++;
			}
		}
		
		cout << answer << endl;
		
	}
	
	return 0;
}
