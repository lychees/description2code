topper = ''
topscore = 0
t = int(raw_input())
scores = {}

for i in range(t):
    inp = raw_input().split(' ')
    name = inp[0]
    score = int(inp[1])
    if name in scores:
        scores[name] += score
    else:
        scores[name] = score

    temp = max(scores, key=scores.get)
    tc = scores[temp]
    toppers = [key for key in scores if scores[key] == tc]
    if len(toppers) == 1:
        topper = temp
    if len(toppers) == 1:
        topper = toppers[0]
print topper
