#include <iostream>
#include <stdio.h>
#include <string.h>

using namespace std;

inline int fastread()
{
    int i=0;
    char c=getchar_unlocked();
    while(c<'0')
    c = getchar_unlocked();
    while(c>='0' && c<='9')
    {
        i = (i<<3)+(i<<1)+c-'0';
        c = getchar_unlocked();
    }
    return i;
}

int main()
{
    int t=fastread();
    char str1[102],str2[102];
    while(t--)
    {
        cin>>str1>>str2;
        int len =strlen(str1);
        for(int i=0;i<len;i++)
        {
			if(str1[i]!=str2[i]) printf("1");
			else printf("0");
		}
		printf("\n");
    }
    return 0;
}
