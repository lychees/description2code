#include <bits/stdc++.h>
#define _ ios_base::sync_with_stdio(false);cin.tie(0);
using namespace std;
#define pb push_back
#define pob pop_back
#define pf push_front
#define pof pop_front
#define mp make_pair
#define all(a) a.begin(),a.end()
#define bitcnt(x) __builtin_popcountll(x)
#define MOD 1000000007
#define MAXN 100005
typedef unsigned long long int uint64;
typedef long long int int64;

vector<int>factor[100005];
bool prime[100006];
int cnt[100005];
int dp[100005];
int main(){
	int i,j,n,val;
	for(i=2;i<=100000;i++){
		if(prime[i]==false){
			factor[i].pb(i);
			for(j=2*i;j<=100000;j+=i){
				prime[j]=true;
				factor[j].pb(i);
			}
		}
	}
	int ans=0;
	cin>>n;
	for(i=1;i<=n;i++){
		scanf("%d",&val);
		dp[i]=1;
		for(j=0;j<factor[val].size();j++){
			int req=cnt[factor[val][j]];
			dp[i]=max(dp[i],dp[req]+1);
		}
		for(j=0;j<factor[val].size();j++){
			int req=cnt[factor[val][j]];
			if(dp[i]>dp[req])
			cnt[factor[val][j]]=i;
		}
		ans=max(ans,dp[i]);
	}
	printf("%d\n",ans);
	return 0;
}
