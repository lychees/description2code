#include <iostream>

using namespace std;

int gcd(int a, int b)
{
    int c;
    while (a != 0) {
        c = a;
        a = b % a;
        b = c;
    }

    return b;
}

int main()
{
    int t;

    cin >> t;

    while (t--) {
        int x, y, t1, count = 0;

        cin >> x >> y;

        string s;

        cin >> s;

        while (x != 0 && y != 0) {
            t1 = gcd(x, y);
            //cout << x << " " << y << " " << t1 << endl;
            x -= t1;
            y -= t1;
            count++;
        }

        if ((count % 2) && s[0] == 'C')
            cout << "Chandu\n";

        else if ((count % 2) && s[0] == 'D')
            cout << "Daspal\n";

        else if ((count % 2 == 0) && s[0] == 'C')
            cout << "Daspal\n";

        else
            cout << "Chandu\n";
    }

    return 0;
}
