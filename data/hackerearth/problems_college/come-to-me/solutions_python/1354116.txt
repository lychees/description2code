def maxselect():
    for test in range(input()):
        force = [i for i in raw_input()]
        flen = len(force)
        i = 0
        j = 0
        ldir = {i: 0 for i in "ABCDEFGHIJKLMNOPQRSTUVWXYZ"}
        countlst = []
        while i < flen and j <= flen:
            if j < flen and ldir.get(force[j]) == 0:
                ldir[force[j]] = 1
                j += 1
            else:
                countlst.append(abs(j - i))
                ldir = {i: 0 for i in "ABCDEFGHIJKLMNOPQRSTUVWXYZ"}
                i += 1
                j = i
        print max(countlst)
maxselect()
