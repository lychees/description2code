int a[1000005],b[1000005];
int p[1000005];
int main()
{
	int t,i,j,flag=0;
	int n,cnt,m,s1,e;
	p[0]=0;
	p[1]=0;
	p[2]=1;
	for(i=2;i<=1000000;++i)
	p[i]=1;
		for(i=4;i<=1000000;i=i+2)
		p[i]=0;
		for(i=3;i<=1000;++i)
		{
			if(p[i]==1)
			{
				for(j=i*i;j<=1000000;j=j+i)
				{
					if(p[j]==1)
					p[j]=0;
				}
				
			}
			
		}
			
		scanf("%d",&n);
		
		for(i=0;i<n;++i)
		{
			scanf("%d",&a[i]);
		}
		for(i=0;i<n;++i)
		{
			scanf("%d",&b[i]);
		}
		sort(a,a+n)	;
		sort(b,b+n)	;
		cnt=0;
		for(i=0,j=0;i<n&&j<n;)
		{
			if(a[i]>b[j])
			{
				if(i<j)
				cnt+= j-1-i+1;
				++j;
			}
			else
			{
				++i;
			}
		}
		if(cnt<=1000000)
		{
			if(p[cnt]==1)
			flag=1;
			
		}
		else
		{
			//flag=1;
			for(i=2;i*i<=cnt;++i)
			{
					if(p[i]==1&&cnt%i==0)
					{
					flag=0;
					break;
					}
				
			}
			
		}
		if(flag==1)
		printf("MAGIC INVERSION\n");
		else
		printf("SIMPLE INVERSION\n");
	
		
	
	return 0;
}
