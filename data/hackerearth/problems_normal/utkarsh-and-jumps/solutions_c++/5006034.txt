#include<bits/stdc++.h>
#define ll long long
#define sl(x) scanf("%lld",&x)
#define sd(x) scanf("%f",&x)
using namespace std;
double dp[1000001];
void calc(double t2,double t3)
{
    ll i=0;
    dp[0]=1;
    dp[1]=0;
    dp[2]=t2;
    dp[3]=t3;
    for(i=4;i<1000001;i++)
    {
        dp[i]=(dp[i-2]*dp[2])+(dp[i-3]*dp[3]);
    }
}
main()
{
	ll n,p;
	sl(n);
	sl(p);
	double t2=(double)p/100;
	double t3=1-t2;
    calc(t2,t3);
    cout<< fixed;
    cout<< setprecision(6) << dp[n];
}
