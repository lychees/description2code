def hcf(a, b):
    if b == 0:
        return a
    else:
        return hcf(b, a % b)


def gcd(a, b, c):
    return reduce(hcf, (a, b, c))


t = input()

for y in xrange(t):
    a, b, c = sorted(map(int, raw_input().split()))

    if (((a*a) + (b*b)) == (c*c)) and gcd(a, b, c) == 1:
        print "YES"
    else:
        print "NO"
