#include "bits/stdc++.h"
using namespace std;
int pri[1000000]={0};
int p[1000000]={0},k=0;
 
void sieve()
{
	pri[0]=pri[1]=1;
	int x,y;
	for(x=2;x<=1000000;x++)
	{
		if(!pri[x])
		{
			for(y=2*x;y<=1000000;y+=x) pri[y]=1;
			p[k++]=x;
		}
	} 
}
 
int ans[1000000]={0};
 
void sum()
{
	ans[0]=ans[1]=0;	
	int x,y;
	for(x=2;x<=1000000;x++)
	{
		int n=x,c=0;
		for(y=0;p[y]*p[y]<=n;y++)
		{
			if(n%p[y]==0)
			{
				while(n%p[y]==0) { c++; n/=p[y]; }
			}
		}
		if(n>1) c++;
		ans[x]=ans[x-1]+c;
	}
}

int main()
{   
    sieve();
    sum();
    
    int t;
    cin>>t;
    
    while(t--)
    
    {
    	int n;
    	cin>>n;
    	
    	cout<<ans[n]<<"\n";
    }
    
    return 0;
    
}
