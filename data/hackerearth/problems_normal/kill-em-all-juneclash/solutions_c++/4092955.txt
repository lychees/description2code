#include <iostream>
#include <bits/stdc++.h>
#define MOD 1000000007
using namespace std;
long long int power(long long int a,long long int b){
	if(b==0)
		return 1;
	long long int f=power(a,b/2);
	f=(f*f)%MOD;
	if(b%2==1)
		return (a*f)%MOD;
		return f;

}
long long int fact[1000005];
long long int po[1000005];
long long int comb(long long int n,long long int r){
	if(r==0 || n==r)
		return 1ll;
    if(n<r)
        return 0;
	long long int f=(po[fact[r]]*po[fact[n-r]])%MOD;
	f=(f*fact[n])%MOD;
//	cout << f << endl;
//	cout << fact[n] << endl;
	return f;

}
long long int two[1000005];
int main()
{	fact[0]=1;
    two[0]=1;

  		}
  	int t=0;
