 #include<iostream>
#include<algorithm>
#include <bits/stdc++.h>
#include<vector>
#include <deque>          // std::deque
#include <list>           // std::list
#include <queue>
using namespace std;
//bool visited[10001];
bool a[1000][1000];
int n;
 int dfs(int i,int j){
    a[i][j]=0;
    int ans=1;
    if(i>0&&a[i-1][j])
        ans+=dfs(i-1,j);

    if(j<n-1&&a[i][j+1])
        ans+=dfs(i,j+1);
    if(i<n-1&&a[i+1][j])
        ans+=dfs(i+1,j);
    if(j>0&&a[i][j-1])
        ans+=dfs(i,j-1);

    return ans;
}
int main()
{

    cin>>n;
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
         cin>>a[i][j];
    int ans=0;
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++){
            if(a[i][j]){
                ans=max(dfs(i,j),ans);
            }
        }
    }
    cout<<ans<<endl;
}
