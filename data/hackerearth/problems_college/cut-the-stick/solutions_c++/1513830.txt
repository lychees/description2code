#include<bits/stdc++.h>
using namespace std;
int main(){
	int t,S,n,L,x,j,z;
	long long ans;
	cin>>t;
	while(t--){
		vector<int> weldLocations;
		ans=0;
		
		cin>>S>>n>>L;
		
		for(int i=0;i<n;i++){
			cin>>x;
			weldLocations.push_back(x);
		}
		sort(weldLocations.begin(),weldLocations.end());
		vector<int> start,nd;
		
		for(int i=0;i<n;i++){
			x=weldLocations[i];
			start.push_back(x);
			nd.push_back(x);
		}
		
		j=start.size();
		for(int i=0;i<start.size();i++){
			if(start[i]>L){
				j=i;
				ans+=((start.size()-i)*(start.size()-i-1))/2;
				break;
			}
		}
		start.resize(j);
		//cout<<ans<<"A\n";
		vector<int>::iterator e;
		for(int i=0;i<nd.size();i++){
			if(S-nd[i]>L){
				e=lower_bound(start.begin(),start.end(),nd[i]);
				z=e-start.begin();
				ans+=z;
			}
			else{
				e=lower_bound(start.begin(),start.end(),nd[i]-L);
				z=e-start.begin();
				ans+=z;
			}
			//cout<<ans<<endl;
		}
		cout<<ans<<endl;
	}
	return 0;
}
