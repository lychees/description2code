#include <iostream>
#include <string.h>
#include<stdio.h>
using namespace std;
int longstpal(char *str)
{
	int maxLen=1,l,h,i,start=0;
	
	int len=strlen(str);
	for(i=0;i<len;++i)
	{
		l=i-1;
		h=i;
		while(l>=0 && h<l && str[l]==str[h])
		{
			if(h-l+1>maxLen)
			{
			start=l;
			maxLen=h-l+1;
			
		}
		--l;
		++h;
	}
	l=i-1;
	h=i+1;
	while(l>=0 && h<len && str[l]==str[h])
	{
		if(h-l+1>maxLen)
		{
			start=l;
			maxLen=h-l+1;
		}
		--l;
		++h;
	}
	}
	return maxLen;
}

int main()
{
   long long int T;
   cin>>T;
   
   while(T--)
   {
   	char str[10000];
   	cin>>str;
   	cout<<longstpal(str)<<endl;
   }
   
   return 0;
}
