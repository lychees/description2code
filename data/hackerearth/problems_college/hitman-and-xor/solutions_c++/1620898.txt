#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,n,x,e,o,i;
    cin>>t;
    while(t--)
    {
        e=o=0;
        cin>>n;
        for(i=0;i<n;i++)
            {
                cin>>x;
                if(x%2)o++;
                else e++;
            }
cout<<(long long)e*(long long)o<<endl;
    }
    return 0;
}
