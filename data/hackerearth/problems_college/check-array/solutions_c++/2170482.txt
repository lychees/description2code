#include <bits/stdc++.h>
using namespace std;
vector <int> arr;
int equilibrium(int n)
{
   int sum = 0;      // initialize sum of whole array
   int leftsum = 0; // initialize leftsum
   int i;
 
   /* Find sum of the whole array */
   for (i = 0; i < n; ++i)
        sum += arr[i];
 
   for( i = 0; i < n; ++i)
   {
      sum -= arr[i]; // sum is now right sum for index i
 
      if(leftsum == sum)
        return i;
 
      leftsum += arr[i];
   }
 
    /* If no equilibrium index found, then return 0 */
    return -1;
}
 
int main()
{
  int t, n;
  scanf("%d", &t);
  while (t--) {
  	scanf("%d", &n);
  	arr.clear();
  	arr.resize(n);
  	for (int i = 0; i < n; i++) {
  		scanf("%d", &arr[i]);
  	}
  	
  	int ans = equilibrium(n);
  	if (ans != -1) {
  		printf("YES\n");
  	} else {
  		printf("NO\n");
  	}
  }
  return 0;
}
