#include <iostream>
#include <algorithm>
#include <limits.h>
using namespace std;

int main()
{
    int minv,maxv,t,n,*arr;
    cin >>t ;
    for(int i=0;i<t;i++)
    {
    	cin >> n;
    	arr = new int[n];
    	for(int j=0;j<n;j++)
    	{
    		cin >> arr[j] ;
    	}
    	sort(arr,arr+n);
    	minv = INT_MAX ;
    	maxv = INT_MIN ;
    	for(int j=0;j<n/2;j++)
    	{
    		if(arr[j]+arr[n-1-j] < minv )
    		{
    			minv = arr[j] + arr[n-1-j];
    		}
    		if(arr[j]+arr[n-1-j] >  maxv )
    		{
    			maxv = arr[j] + arr[n-1-j];
    		}
    	}
    	cout<< maxv - minv <<endl;
    }
    return 0;
}
