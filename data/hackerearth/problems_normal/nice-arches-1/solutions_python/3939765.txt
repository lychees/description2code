T = int(raw_input())
b=0
for i in range(T):
	S = raw_input()
	lenS = len(S)
	if lenS%2!=0:continue
	j=0
	stack = []
	
	while(j<lenS):
		
		
		if len(stack)>=1 and stack[-1]==S[j]:
			stack.pop()
		else:
			stack.append(S[j])
		j+=1
	if len(stack)==0:b+=1

print b
