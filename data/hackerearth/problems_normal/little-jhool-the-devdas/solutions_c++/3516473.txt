#include <iostream>
using namespace std;

int main()
{
    int r,p,odd[100],e=0,o=0;
    cin>>r;
    for(int i=0;i<r;i++) {
    	cin>>p;
    	if(p%2==0) {
    		e=e+p;
    	} else {
    		odd[o++]=p;
    	}
    }
    if(o==0) {
    	cout<<":(";
    	return 0;
    } else if(o%2==0) {
    	for(int i=0;i<o-1;i++) {
    		for(int j=i+1;j<o;j++) {
    			if(odd[i]<odd[j]) {
    				int x = odd[i];
    				odd[i] = odd[j];
    				odd[j] = x;
    			}
    		}
    		e=e+odd[i];
    	}
    } else {
    	for(int i=0;i<o;i++) {
    		e=e+odd[i];
    	}
    }
    cout<<e;
    return 0;
}
