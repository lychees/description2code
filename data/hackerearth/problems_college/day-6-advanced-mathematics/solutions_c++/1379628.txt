typedef long long ll;

template <class T> T& get(T &n)
{
    static int p;
    static bool f;
    f=false;
    n = 0;
    do
    {
        p=getcx();
    }
    while( !isdigit(p) && p!='-');
    if(p=='-'){
        f=true;
        p=getcx();
    }
    while ( isdigit(p) ){
        n = (n << 3) + (n << 1) + (p - '0');
        p = getcx();
    }
    n = (f ? -n : n);
    return n;
}

int getstring(char* s)
{
    static int p;
    p=0;
    static char c;
    do
    {
        c=getcx();
    }while(!isalpha(c));
    while(c!='\n'&&c!=EOF)
    {
        s[p++]=c;
        c=getcx();
    }

    return p;
}
