#include <iostream>
#include <stack>
using namespace std;

int main()
{
	int Q, tmp;
	cin >> Q;
	stack<int> food;
	for(int i=0; i<Q; i++){
		cin >> tmp;
		if(tmp == 1){
			if(food.empty()){
				cout << "No Food" << '\n';
			}
			else{
				cout << food.top() << '\n';
				food.pop();
			}
		}
		else{
			cin >> tmp;
			food.push(tmp);
		}
	}
    return 0;
}
