t = input()
while t:
	t-=1
	
	n = input()
	c = 0
	if n >= 7:
		x = n / 7 	
		n = n - (x*7)
		c = c + x
	if n >= 5:
		x = n / 5 	
		n = n - (x*5)
		c = c + x
	if n >= 3:
		x = n / 3 	
		n = n - (x*3)
		c = c + x
	if n >= 2:
		x = n / 2 	
		n = n - (x*2)
		c = c + x
	if n == 1 and c>0:
		c = c + 1
	
	print c if c>0 else '-1'
