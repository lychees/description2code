#include <bits/stdc++.h>
using namespace std;

int main() {
    int t;
    cin>>t;
    while(t--) {
    	int A,B,n;
    	cin>>A>>B>>n;
    	vector<int> v(n);
    	for(int i=0;i<n;i++)
    		cin>>v[i];
    	sort(v.begin(),v.end());
    	int c1 = 0, c2 = 0;
    	for(int i=0;i<n;i++) {
    		if(A-v[i] < 0)
    			break;
    		A -= v[i];
    		c1 += 1;
    			
    	}
    	for(int i=0;i<n;i++) {
    		if(B-v[i] < 0)
    			break;
    		B -= v[i];
    		c2 += 1;
    	}
    	if(c1 == c2)
    		cout<<"Tie"<<endl;
    	else
    		cout<<(c1>c2 ? "Raghu Won" : "Sayan Won")<<endl;
    }
    return 0;
}
