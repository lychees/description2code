#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
	unsigned long int n,s[1000000],c[1000000];
	unsigned long int ans=0;
	cin>>n;
	
	for(int i=0;i<n;i++)
	     cin>>s[i];
	     
	for(int i=0;i<n;i++)
	  cin>>c[i];
	
	sort(s,s+n);
	sort(c,c+n);
	
	for(int i=0;i<n;i++)
	ans +=c[i]*s[i];
	
    cout<<ans;
    
    return 0;
}
