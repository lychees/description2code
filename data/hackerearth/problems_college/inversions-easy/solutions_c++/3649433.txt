#include <bits/stdc++.h>
using namespace std ;
#define abc long long int
#define p cin
#define c cout
int main()
{
    abc n ;
    cin >> n ;
    abc  i ;
    abc a[100000] ;
    for ( i = 0; i< n ; i++)
    {
       p >> a[i] ;
     }
     abc j ;
     abc sum = 0 ;
     for ( i = 0; i< n ; i++)
     {
       for ( j = 0; j< n ; j++)
       {
           if ( i < j && a[i]>a[j])
           {
            sum = sum + 1 ;
           }
       }
     }

     cout << sum <<endl;
     return 0 ;
}
