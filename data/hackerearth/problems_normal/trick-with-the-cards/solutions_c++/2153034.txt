#include <iostream>
using namespace std;

typedef long long int LLD;

LLD findRes(LLD n){
	if(n == 1 || n == 2)
		return 1;
	if(n % 3 == 0)
		return (n / 3);
	if(n % 3 == 1)
		return ((n / 3) + 1 + ((2 * n - 2) / 3));
	return ((n / 3) + 1 + ((2 * n - 1) / 3));
}

int main()
{
	LLD n;
	int t;
	cin >> t;
	while(t--){
		cin >> n;
		cout << findRes(n) << endl;
	}
    return 0;
}
