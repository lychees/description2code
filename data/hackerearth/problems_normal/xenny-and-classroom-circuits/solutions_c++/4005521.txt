#include <iostream>
#include <stdio.h>
#include <vector>
#include <string>
#include <queue>
#include <algorithm>
#include <map>

#include <cassert>

#define si(n) scanf("%d",&n)
#define sd(n) scanf("%lf",&n)
#define sl(n) scanf("%lld",&n)
#define ss(str) scanf("%s",str)
#define FOR(i,a,b) for(int i = a; i<b; i++)
#define pb push_back
#define mp make_pair

typedef long long int ll;
typedef unsigned long long int ull;

using namespace std;

int main()
{
    int t;
    si(t);
    while(t--)
    {
        int n;
        si(n);
        string sequence;
        cin>>sequence;
        int bcount=0;
        int gcount=0;

        //start with B;
        FOR(i,0,sequence.size())
        {
            if(sequence[i] != 'B')
                bcount++;
            i++;
        }

        FOR(i,0,sequence.size())
        {
            if(sequence[i] !='G')
                gcount++;
            i++;
        }

        cout<<min(bcount,gcount)<<endl;
    }
    
    return 0;
}
