long long powermod(long long x,long long y){
	long long res = 1;
	while(y){
		if(y&1){
			res *=x;
			res %=mod;
		}
		x *=x;
		x %=mod;
		y/=2;
	}
	return res;
}
long long com(int n,int r){
	long long res = fact[n];
	res *= ifact[r];
	res %=mod;
	res *=ifact[n-r];
	res %=mod;
	return res;

}
