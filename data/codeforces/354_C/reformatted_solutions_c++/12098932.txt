#include <iostream>
#include <map>
#include <math.h>

#include <stdio.h>
#include <vector>
using namespace std;
const int A = 1e9, N = 328228;

int o = A, a[N], ok = 1, i, j, n, m;

int main() {
  cin >> n >> m;
  for (i = 0; i < n; i++)
    scanf("%d", &a[i]), o = min(a[i], o);
  while (ok) {
    ok = 0;
    for (i = 0; i < n; i++)
      while (a[i] % o > m)
        o--, ok = 1;
  }
  cout << o;
}
