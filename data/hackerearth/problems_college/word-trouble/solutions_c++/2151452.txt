#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    string s, t;
    int T, i, N, M, aux;
    cin>>T;
    while(T--)
    {
    	cin>>s>>t;
    	if(s==t)
    	{
    		cout<<"Exact"<<endl;
    		continue;
    	}
    	cout<<"Non Exact ";
    	N=s.length();
    	M=t.length();
    	aux=min(N, M);
    	int cnt=0;
    	for(i=0; i<aux; ++i)
    		if(s[i]!=t[i])
    			cnt++;
    	cnt+=abs(N-M);
    	cout<<cnt<<endl;
    	
    }
    return 0;
}
