void update(int idx, char v,int val){

	int vv = v - 'a';
	while( idx < N){


		BIT[vv][idx] += 1;
		idx += idx & -idx ;
	}

	
}

void build(){

	int i;
	FOR(i,1,n+1){

		update(i,s[i-1],1);
	}


}




int read( int idx, int v){


	int res = 0;

	while( idx > 0){


		res += BIT[v][idx];
		idx -= idx & -idx;
	}


	return res;	

}
