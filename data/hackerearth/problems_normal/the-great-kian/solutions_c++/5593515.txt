#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int nSz, count = 0;
    long int sum = 0;
    cin >> nSz;
    
    vector <long int> arr;
    for (int idx = 0; idx < nSz; idx++) {
    	int temp;
    	cin >> temp;
    	arr.push_back (temp);
    }
    
    while (count < 3) {
    	sum = 0;
    	for (int idx = count; idx < nSz; idx = idx + 3) {
    		sum += arr[idx];
    	}
    	cout << sum << " ";
    	count++;
    }
    return 0;
}
