#include <bits/stdc++.h>
using namespace std;
 
int main()
{
   int t,n,i,j,afsnmklnf,min1,m=INT_MAX;
   cin>>t;
   while(t--){
   	   cin>>n;
   	   min1=INT_MAX;
   	   for(i=0;i<=n/6;i++){
   	   	   afsnmklnf=0;
   	   	   for(j=0;j<=n/8;j++){
   	   	   	   if(6*i+8*j==n){
   	   	   	   	  if(i+j<min1)
   	   	   	   	     min1=i+j;
   	   	   	   	  afsnmklnf=1;
   	   	   	   	  break;
   	   	   	   }
   	   	   }
   	   	   if(afsnmklnf)
   	   	      break;
   	   }
   	   if(m==min1)
   	      cout<<"-1"<<endl;
   	   else
   	      cout<<min1<<endl;
   }
   
}
