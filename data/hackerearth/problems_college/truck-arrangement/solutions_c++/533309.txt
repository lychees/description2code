#include<iostream>
#include<algorithm>
#include<math.h>
#define pii pair< int, int >
using namespace std;
int main()
{
	int t,n,a;
	cin>>t;
	while(t--)
	{
		cin>>n;
		pii arr[n];
		for(int i=0;i<n;i++)
		{
			cin>>a;
			arr[i]=pii(a,(i+1));
		}
		sort(arr,arr+n);
		int max=0;
		for(int i=0;i<n;i++)
		{
			if(fabs((arr[i].second)-(i+1))>max)
			{
				max=fabs(arr[i].second-(i+1));
			}
		}
		cout<<max<<endl;
		
	}
	return 0;
}
