#include<stdio.h>
#include<stdlib.h>

int cmpfunc(const void *a, const void *b){
	return (*(int *)a - *(int *)b);
}

int main(){
	int t, a, b, n; scanf("%d", &t);
	while(t-- && scanf("%d%d%d", &a, &b, &n)){
		int ar[n], i, cr = 0, cs = 0;
		for(i = 0; i < n; scanf("%d", &ar[i++]));
		qsort(ar, n, sizeof(int), cmpfunc);
		for(i = 0; i < n; i++){
			if(ar[i] <= a)	cr++, a -= ar[i];
			if(ar[i] <= b)	cs++, b -= ar[i];
		}
		if(cr < cs)	printf("Sayan Won\n");
			else if(cr > cs)	printf("Raghu Won\n");
				else	printf("Tie\n");
	}		
	return 0;
}
