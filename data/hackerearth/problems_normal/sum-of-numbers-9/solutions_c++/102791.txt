
#include <stdio.h>
#include <stdlib.h>

int N,flag=0;
long int *a,K;

void func(int j,long sum){
	if(j>N)
	{
		if(sum==K && flag==0)
		{
			flag = 1;
			printf("YES\n");
			
		}
		return;
	}
	func(j+1,sum);
	func(j+1,sum+a[j]);
}




int main()
{
   	int T;
   	scanf("%d",&T);
  // 	printf("%d\n",T);
   	while(T--){
   		flag=0;
   		scanf("%d",&N);
   	//	printf("%d\n",N);
   		a=(long *)malloc(N*sizeof(long));
   		for(int j=0;j<N;j++){
   			scanf("%ld",&a[j]);
   	//		printf("%ld ",a[j]);
   		}
   	//	printf("\n");
   		scanf("%ld",&K);
   	//	printf("%d\n",K);
   		
   		func(0,0);
   		if(flag==0){
   			printf("NO\n");
   		}
   		free(a);
   	}
   	return 0;
}
