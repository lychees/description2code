#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
ll caln(ll n)
{
    ll lo=0,hi=100000000LL;
    while(lo<hi-1)
    {
        ll mid=(lo+hi)/2;
        if(mid*(mid+1)==2*n)
            return mid;
        if(mid*(mid+1)>2*n)
            hi=mid-1;
        else
            lo=mid;
    }
    if(lo==hi)
        return lo;
    if(hi*(hi+1)<=2*n)
        return hi;
    return lo;
}
ll cal(ll n)
{
    ll term=caln(n);
   // cout<<term<<"\n";
    ll terms=(term+1)/2;
    ll sum=terms*terms;
    ll tot=term*(term+1)/2;
    ll left=n-tot;
    if(left==0)
        return sum;
    if(left%2==0)
        return sum;
    else
        return term+1+sum;
}
int main()
{
    int test;
    scanf("%d",&test);
    while(test--)
    {
        ll l,r;
        scanf("%lld%lld",&l,&r);
        ll ans=cal(r)-cal(l-1);
        printf("%lld\n",ans);
    }
}
