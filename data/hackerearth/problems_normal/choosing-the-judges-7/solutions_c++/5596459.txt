#include <bits/stdc++.h>
using namespace std;
typedef vector<long long> VI;
VI j,dp;
int t,n;

long long maxScore(int i){
	if(i>=n) return 0;
	if(dp[i]!=-1) return dp[i];
	return dp[i] = max(j[i]+maxScore(i+2),maxScore(i+1));
}

int main()
{
    cin>>t;
    for(int l=0;l<t;l++){
    	cin>>n;
    	j.clear();
    	dp.clear();
    	for(int i=0;i<n;i++){
    		int p;
    		cin>>p;
    		j.push_back(p);
    		dp.push_back(-1);
    	}
    	cout<<"Case "<<l+1<<": "<<maxScore(0)<<endl;
    }
    return 0;
}
