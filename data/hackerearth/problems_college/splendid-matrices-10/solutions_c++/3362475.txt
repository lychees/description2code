#include<bits/stdc++.h>
using namespace std;
int a[2010][2010]={0};
void make(int v,int n,int i,int j)
{
	if(n==1)
	a[i][j]=v;
	else
	{
		//cout<<"came"<<endl;
		if(!a[i][j])
		make(v,n/2,i,j);
		if(!a[i][j+n/2])
		make(v+(n*n)/4,n/2,i,j+n/2);
		if(!a[i+n/2][j])
		make(v+(n*n/2),n/2,i+n/2,j);
		if(!a[i+n/2][j+n/2])
		make(v+(3*n*n)/4,n/2,i+n/2,j+n/2);
	}
}
int main()
{
	int j,M,n,b[100],i;
	b[1]=2;
	for(i=2;i<=10;i++)
	b[i]=2*b[i-1];
	cin>>n;
	//cout<<b[n]<<endl;
	memset(a,0,sizeof(a));
	make(1,b[n],1,1);
	for(i=1;i<=b[n];i++,cout<<endl)
	for(j=1;j<=b[n];j++)
	cout<<a[i][j]<<" ";
	return(0);
}
