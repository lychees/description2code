#include<iostream>
#include<cstdio>
#define ll long long
#include<cstring>
#include<algorithm>
#include<cmath>
using namespace std;
int main()
{
	ll t,a,b,c,x,y,z;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld%lld%lld",&x,&y,&z);
		if(x>=y)
		{
			a=x;
			b=y;
		}
		else
		{
			b=x;
			a=y;
		}
		if(z>a)
		{
			b=a;
			a=z;
		}
		else if(z>b)
			b=z;
		if(a==0)
			cout<<"0\n";
		else if(b==0)
			cout<<"1\n";
		else if(a==b)
			cout<<2*b<<endl;
		else
			cout<<b+(b+1)<<endl;
	}
	return 0;
}
