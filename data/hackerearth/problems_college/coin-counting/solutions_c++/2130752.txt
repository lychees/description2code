#include<bits/stdc++.h>
using namespace std;

typedef long long ll;
int a[]={1,3,5};

ll fun(int n,int a[],int i)
{
    if(n!=0&&i==3)  return 0;
    if(n==0)    return 1;
    if(n<0)    return 0;
    return fun(n,a,i+1)+fun(n-a[i],a,i);
}


int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        ll n;
        cin>>n;
        cout<<fun(n,a,0)<<endl;
    }
    return 0;
}
