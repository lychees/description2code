from fractions import gcd
t=int(raw_input())
for qq in xrange(t):
	x,y,s=raw_input().split()
	x=int(x)
	y=int(y)
	g=gcd(x,y)
	p=0
	if s=='Daspal':
		p=1
	while x!=0 and y!=0:
		x-=g
		y-=g
		if x!=0 and y!=0:
			g=gcd(x,y)
		if p==0:
			p=1
		else:
			p=0
	if p==0:
		print 'Daspal'
	else:
		print 'Chandu'
