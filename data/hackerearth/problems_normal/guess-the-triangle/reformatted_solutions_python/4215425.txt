c, d = map(float, raw_input().split())
e, f = map(float, raw_input().split())

# x,y  m,n   r,s

x = (c+e-a)/2
m = (c+a-e)/2
r = (a+e-c)/2

y = (d+f-b)/2
n = (d+b-f)/2
s = (b+f-d)/2


arr = [[x, y], [m, n], [r, s]]
arr = sorted(arr)

for item in arr:
    for it in item:
        print("{0:.4f}".format(it)),
