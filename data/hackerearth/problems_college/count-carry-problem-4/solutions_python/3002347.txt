t = input()
while t > 0:
    a, b = map(str, raw_input().split())
    temp = min(int(a), int(b))
    lenght = len(str(temp))
    carry = 0
    for i in range(lenght):
        if (int(a[i]) + int(b[i])) >= 10:
            carry += 1
    if carry == 0:
        print "No carry operation"
    elif carry == 1:
        print "1 carry operation"
    else:
        print str(carry) + " carry operations"
    t -= 1
