#include <iostream>
using namespace std;

int main()
{
    int t,a1_s,a1_u,a2_s,a2_u,i,a1,a2;
    string in;
    bool mode;
    cin>>t;
    while(t--)
    {
    	cin>>in;
    	
    	//Matching with SU
    	mode=true;
    	a1_s=0;a1_u=0;
    	a2_s=0;a2_u=0;
    	for(i=0;i<in.length();i++,mode=!mode)
    	{
    		if(mode)
    		{
    			if(in[i]=='S')
    			a2_u++;
    			else
    			a1_s++;
    		}
    		else
    		{
    			if(in[i]=='U')
    			a2_s++;
    			else
    			a1_u++;
    		}
    	}
    	
    	a1=a1_s+a1_u;
    	a2=a2_s+a2_s;
    	
    	if(a1<a2)
    	{
    		if(a1_s==a1_u)
    		cout<<(a1/2)<<"\n";
    		else
    		if(a2_s==a2_u)
    		cout<<(a2/2)<<"\n";
    		else
    		cout<<"-1"<<"\n";
    	}
    	else
    	{
    		if(a2_s==a2_u)
    		cout<<(a2/2)<<"\n";
    		else
    		if(a1_s==a1_u)
    		cout<<(a1/2)<<"\n";
    		else
    		cout<<"-1"<<"\n";
    	}
    }
  
    return 0;
}
