def isBalancedString(string):
    sorted_str = sorted(string)
    last_count = 0
    before_last_count = 0
    prev_char = ''
    unique_chars = 0
    for s in sorted_str:
        if prev_char != s:
            unique_chars = unique_chars + 1
            if before_last_count > 0 and last_count != before_last_count:
                return False
            before_last_count = last_count
            last_count = 0
        last_count = last_count + 1
        prev_char = s

    return last_count == before_last_count and unique_chars == 26


if __name__ == "__main__":
    cases = int(raw_input())
    for i in range(cases):
        test = raw_input()
        balanced = isBalancedString(test)
        if balanced:
            print "Yes"
        else:
            print "No"
