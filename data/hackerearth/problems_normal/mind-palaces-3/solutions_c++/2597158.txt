#include <iostream>

using namespace std;

void binary(int * a, int n, int f, int & y) {
	int s = 0, e = n-1;
	while(s <= e) {
		int m = (s+e)/2;
		if(a[m] == f)  {
			y = m; return;
		}
		else if(a[m] > f) {
			e = m-1;
		}
		else
			s = m+1;
	}
	y = -1;
}

void find(int a, int n, int m, int arr[][1000], int & x, int  & y) {
	
	for(int i = 0; i < n; i++) {
		if(arr[i][0] <= a && arr[i][m-1] >= a) {
			binary(arr[i], m, a, y);
			if(y != -1) {
				x = i; return;
			}
		}
	}
	x = -1; y = -1;
}

int main(void) {
	int a[1000][1000];
	int n, m;
	cin >> n >> m;
	for(int i = 0; i < n; i++) 
		for(int j = 0; j < m; j++)
			cin >> a[i][j];
	int q;
	cin >> q;
	while(q--) {
		int x, y, t;
		cin >> t;
		find(t, n, m, a, x, y);
		cout << x << " " << y << endl;
	}
}
