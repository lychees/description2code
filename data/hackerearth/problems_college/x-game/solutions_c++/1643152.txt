#define MOD 1000000007
long long int res;
long long int fpower(long long int n, long long int k)
{
  long long int result, fpow,co;
  if(k==0)
    return 1;
  if(n==0)
    return 0;
  if(n==1)
    return 1;
  result=n%MOD;
  fpow=1;
  while(fpow*2<=k)
  {
   fpow=fpow*2;
    result=(result*result);
    result=result%MOD;
  }

  if(k-fpow!=0)
  result=result*fpower(n, k-fpow);
  result=result%MOD;

  return result;
}
