#define _USE_MATH_DEFINES
#include<bits/stdc++.h>
using namespace std;

#define pb push_back
#define For(i, begin, end) for (__typeof(end) i = (begin) - ((begin) > (end)); i != (end) - ((begin) > (end)); i += ((begin) > (end)?-1:1))
#define all( v ) (v).begin(),(v).end()
template<typename T> using V = vector<T>;
typedef long long int ll;
typedef unsigned long long int ull;
/***********************************************/
/*      ____________
 *     /            \
 *    /  /\      /\  \
 *   /  /  \    /  \  \
 *   \                /
 *    \     \___/    /
 *     \____________/
 */
const ll mod = 1000000007;
int main(){
	ios_base::sync_with_stdio(false);
	cin.tie(nullptr);
	size_t N;
	cin>>N;
	V<pair<ll,ll> >po(N);
	For(i,0,N)cin>>po[i].first>>po[i].second;
	ll out = 0;
	For(i,0,N-1){
		ll a = (po[i+1].second - po[i].second);
		ll b = (po[i].first - po[i+1].first);
		ll c = -(b * po[i].second + a * po[i].first);
		out++;
		For(j,i+2,N){
			ll res = a * po[j].first + b * po[j].second + c;
			if(res <= 0){
				out++;
				a = (po[j].second - po[i].second);
				b = (po[i].first - po[j].first);
				c = -(b * po[i].second + a * po[i].first);
			}
		}
	}
	cout<<out<<endl;
	return 0;
}
