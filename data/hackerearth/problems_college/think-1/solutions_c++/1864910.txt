#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,m,i,l;
    char a[50],b[50],sa[50],sb[50],tmp[50];
    cin>>t;
    while(t--)
    {
        cin>>n;
        sprintf(a,"%d",n);
        sprintf(sa,"%d",n*n);
        l=strlen(a);
        for(i=0;i<l;i++)
            b[i]=a[l-1-i];
        b[l]=0;
        l=strlen(sa);
        for(i=0;i<l;i++)
            sb[i]=sa[l-1-i];
        sb[l]=0;
        sscanf(b,"%d",&m);
        sprintf(sa,"%d",m*m);
        if(strcmp(sa,sb)==0 && m!=n)
            cout<<m<<"\n";
        else
            puts("No");
    }
    return 0;
}
