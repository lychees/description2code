#include <bits/stdc++.h>
using namespace std;

#define mod 1000000007 

long long arr[100005],dp[101][101];

long long power(long long a,long long b)
{
	if(b==0)
		return 1;
	long long x;
	x=power(a,b/2)%mod;
	x=(x*x)%mod;
	if(b%2==1)
		x=(x*a)%mod;
	return x;
}

long long facto[100002];

int main()
{
	long long i,j,n,k,t;
	
	facto[0]=1;
	for(i=1;i<100001;i++)
		facto[i]=(facto[i-1]*i)%mod;
	
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		for(i=0;i<n;i++)
		{
			cin>>arr[i];
			if(arr[i]>=mod)
				arr[i]%=mod;
		}
		sort(arr,arr+n);
		long long total=0;
		for(i=0;i<n-k+1;i++)
		{
			//cout<<facto[n-i-1]<<" "<<facto[k-1]<<" "<<facto[n-i-k]<<endl;
			total+=(((((arr[i]*facto[n-i-1])%mod)*power(facto[k-1],mod-2))%mod)*power(facto[n-i-k],mod-2))%mod;
			if(total>=mod)
			total%=mod;
			//cout<<"total "<<total<<endl;
		}
		cout<<total<<endl;
	}
	return 0;
}
