import math
t=input()
for _ in xrange(t):
    a=[int(x) for x in raw_input().split()]
    n,m=a[0],a[1]
    c,p,i,pv=1,1,1,1
    while c<n and p<m:
        p+=c
        c+=pv
        pv=p
        i+=1
    if n==1:
    	print 0
    elif c<n:
        i+=int(math.ceil((n-c)/(float(m))))
    	print i
    else:
    	print i
