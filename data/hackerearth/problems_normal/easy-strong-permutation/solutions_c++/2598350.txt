#include <iostream>
#include <functional>
#include <algorithm>
#define max 1000000007
using namespace std;
   int main ()
   {
      long long int array[100000],n,i,j,k,temp;
      unsigned long long sum=0;
      cin>>n;
      for ( i = 0; i < n; ++i)
      {
        cin>>array[i];
      }
      sort(array,array+n);
      j=n/2+n%2;
      k=n-1;
      for (int i = 0; i < j; ++i)
      {
        if(i==j-1){
          temp=abs(array[i]-array[k])+abs(array[k]-array[0]);
        }else{
         temp=abs(array[i]-array[k])+abs(array[k]-array[i+1]);
      }
       sum=sum+temp;
        k--;
      }
      cout<<sum<<"\n";
   }
