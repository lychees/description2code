#include <bits/stdc++.h>
using namespace std;


struct ans
{
	int freq;
	char c;
}d[26];


bool compare(const ans& a, const ans& b)
{
	if(a.freq == b.freq){
		return a.c < b.c;
	//	return a.batch < b.batch;
	}
	return a.freq > b.freq;
}


int main()
{
    int i,j,k,t,n,l,r;
    cin >>  t;
 
 	char f[100001];
    char ch;
	f[0] = 'a';
	f[1] = 'b';
	
	for(i=2;i<100001;i++)
	{
		ch = (f[i-1]+f[i-2] - 2*97) % 26;
		f[i] = 97+ ch;
		//cout<< f[i]<<endl;
	}
    
    while(t--)
    {
    	string s;
    	cin >> l >> r;
    	cin >> s;
    	int len = s.length();
    	
    	for(i=l;i<=r;i++)
    	{
    		d[f[i]-97].freq++;
    	}
    	
    	for(i=0;i<26;i++){
    		d[i].c = i+'a';
    		//cout << d[i].c <<" "<< d[i].freq<<endl;
    	}
    		
    		
    	sort(d,d+26,compare);
    	
    	for(i=0;i<len;i++)
    	{
    		cout << d[s[i]-97].c;
		}
    	
    	cout << endl;
    	
    	for(i=0;i<26;i++)
    		d[i].freq = 0;
    	
    }
    return 0;
}
