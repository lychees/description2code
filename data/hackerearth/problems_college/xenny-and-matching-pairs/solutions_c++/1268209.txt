#include <bits/stdc++.h>
using namespace std ;

#define MAXN 100001 

int BIT[MAXN] , T , N;

void update(int idx,int val){
	while(idx < MAXN){
		BIT[idx] = max(BIT[idx],val) ;
		idx += (idx & -idx) ;
	}
}


int query(int idx){

	int s = 0;
	while(idx){
		s = max(s,BIT[idx]) ;
		idx -= (idx & -idx) ;
	}
	return s ;
}


int main(){

	scanf("%d",&T) ;
	while(T--){

		scanf("%d",&N) ;
		vector<int> A(N+1) ;
		for(int i=1;i<=N;i++){
			scanf("%d",&A[i]) ;
		}
		int ans = 0 ;
		for(int i=1;i<=N;i++){
			int x = query(A[i]) ;
			ans = max(ans,x+1) ;
			update(A[i],x+1) ;
		}
		printf("%d\n",ans) ;
		memset(BIT,0,sizeof BIT) ;
	} 
	return 0 ;
}
