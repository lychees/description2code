#include <iostream>
using namespace std;

int main()
{int t;
cin>>t;
while(t--)
{
    int n,i,count=0;
  cin>>n;
  int a[n],b[n];
  for(i=1;i<=n;i++)
  cin>>a[i];
  for(i=1;i<=n;i++)
  b[a[i]]=i;
  for(i=1;i<=n;i++)
  if(a[i]==b[i])
  count++;
  if(count==n)
  cout<<"inverse"<<endl;
  else
  cout<<"not inverse"<<endl;
}
  
  
  return 0;
}
