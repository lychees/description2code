#include <iostream>
#include <stack>
#include <string>
using namespace std;

int main()
{
    int n;
    string p;
    stack<int> st;
    int idx = 0, next;
    string ans;
    bool possible;
    
    while (cin >> n)
    {
    	idx = 0;
    	ans = "";
    	possible = true;
    	
    	st.push(idx);
	    cin.ignore();
	    if (!getline(cin, p) || p == "")
	    {
	    	cout << endl;
	    	continue;
	    }
	    int pos, npos;
	    pos = 0;
	    npos = p.find(' ', pos);
	    
	    while (npos > 0)
	    {
	    	next = stoi(p.substr(pos, npos - pos));
	    	while (st.top() < next && idx <= next)
	    	{
	    		st.push(++idx);
	    		ans += 'C';
	    	}
	    	while (!st.empty() && st.top() > next)
	    	{
	    		st.pop();
	    		ans += 'K';
	    	}
	    	if (st.empty() || st.top() != next)
	    	{
	    		possible = false;
	    		break;
	    	}
	    	ans += 'K';
	    	st.pop();
	    	pos = npos + 1;
	    	npos = p.find(' ', pos);
	    }
	    next = stoi(p.substr(pos));
	    while (st.empty() || (st.top() < next && idx <= next))
    	{
    		st.push(++idx);
    		ans += 'C';
    	}
    	while (!st.empty() && st.top() > next)
    	{
    		st.pop();
    		ans += 'K';
    	}
    	if (st.empty() || st.top() != next)
    		possible = false;
    	else
    		ans += 'K';
    		
    	if (possible)
    		cout << ans << endl;
    	else
    		cout << -1 << endl;
    }
    
    return 0;
}
