#include <iostream>
#include<cstdio>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int sum=0,i;
		char a[2000]={0};
		scanf("%s",a);
		for(i=0;a[i];i++)
		{
			if(a[i]!='_')
			sum=(sum+(a[i]-'0'))%9;
			//cout<<sum<<endl;
		}
		if(sum==0)
		cout<<"0 or 9\n";
		else
		cout<<(9-sum)<<endl;
	}
    return 0;
}
