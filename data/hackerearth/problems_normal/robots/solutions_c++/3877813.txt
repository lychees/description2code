#include <iostream>
#include <set>
#include <vector>
#include <algorithm>

typedef unsigned long int ll;
#define LIMIT 1000000007
using namespace std;

int main()
{
	vector<ll> Max;
	ll T,N;
	ll x;
	ll result,elementsBehind,availChoices;
	cin>>T;
	while(T-- > 0)
	{
		cin>>N;
		for(int i = 1; i <= N; i++)
		{
			cin>>x;
			Max.push_back(x);
		}
		sort(Max.begin(), Max.end());
		result = 1;
		elementsBehind = 0;
		for(vector<ll>::iterator itr = Max.begin(); itr != Max.end(); itr++)
		{
			availChoices = *itr;
//			cout<<"  availChoices: "<<availChoices;
//			cout<<"  elementsBehind:  " <<elementsBehind;
//			availChoices = *itr;
			result = (result	*((availChoices - elementsBehind)%LIMIT))%LIMIT;
			elementsBehind++;
		}
		Max.resize(0);
		cout<<result<<endl;
	}
//    cout << "Hello World!" << endl;
    return 0;
}
