using namespace std;

int x[1000005],y[1000005];
int main()
{
	int n,k;
	long ans=0;
	scanf("%d%d",&n,&k);
	for(int i=0 ; i<k ; i++)
		scanf("%d",&x[i]);
	for(int i=0 ; i<k ; i++)
		scanf("%d",&y[i]);
	for(int i=0 ; i<k ; i++)
		ans+= min(x[i]-1,n-x[i]) + min(y[i]-1,n-y[i]);
	printf("%ld\n",ans);
	return 0;
}
