#include <iostream>
#include<algorithm>
using namespace std;
int main()
{
   long long t,m[1000],w[1000],i,n,sum=0;
   scanf("%lld",&t);
   while(t>0)
   {
   	scanf("%lld",&n);
   	for(i=0;i<n;i++)
   	{
   		scanf("%lld",&m[i]);
   	}
   	for(i=0;i<n;i++)
   	{
   		scanf("%lld",&w[i]);
   	}
   	sort(m,m+n);
   	sort(w,w+n);
   	for(i=0;i<n;i++)
   	{
   		sum=sum+m[i]*w[i];
   	}
   	printf("%lld\n",sum);
   	sum=0;
   	t--;
   }
    return 0;
}
