#include <bits/stdc++.h>
using namespace std;

int main()
{
	string s;
	cin >> s;
	string love = "love";
	bool found = false;
	
	for(int i = 0, j = 0; i < s.size(); ++i) {
		if(love[j] == s[i]) {
			++j;
		}
		
		if(j == 4) {
			found = true;
			break;
		}
	}
	
	cout << (found ? "I love you, too!" : "Let us breakup!") << endl;
	
}
