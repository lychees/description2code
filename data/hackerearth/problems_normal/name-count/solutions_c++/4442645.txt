#include <iostream>
#include <bits/stdc++.h>
using namespace std;
#define lli long long int
long int m=1e9+7;
lli f[100005];
lli mexp(lli a,lli b)
{
	if(b==0)
	{
		return 1;
	}
	lli t=mexp(a,b/2);
	if(b%2==0)
	{
		return (t%m*t%m)%m;
	}
	else
	{
		return ((t%m*t%m)*a%m)%m;
	}
}
void pre()
{
	f[0]=1;
	lli i;
	for(i=1;i<=100000;i++)
	{
		f[i]=mexp(i,m-2);
		f[i]*=f[i-1];
		f[i]%=m;
	}
}
int main()
{
	pre();
	ios_base::sync_with_stdio(false);
    lli t,n,k,i,ans,num,den;
    cin>>t;
    while(t--)
    {
    	cin>>n>>k;
    	num=1;
    	for(i=1;i<=k;i++)
    	{
    		num*=i;
    		num%=m;
    	}
    	den=f[k-n];
    	ans=(num*den)%m;
    	cout<<ans<<endl;
    }
    return 0;
}
