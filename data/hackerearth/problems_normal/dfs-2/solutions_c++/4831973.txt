#include <bits/stdc++.h>
using namespace std;

int p[1000000],n,q,temp;

int binaryFirst(int l,int r,int element)
{
	if(l<=r)
	{
		int mid = (l+r)/2;
		if(element>p[mid-1]&&p[mid]>=element)
			return mid;
		if(p[mid]<element)
			return binaryFirst(mid+1,r,element);
		return binaryFirst(l,mid-1,element);
	}
		return -1;
}
int binaryLast(int l,int r,int element)
{
	if(l<=r)
	{
		int mid = (l+r)/2;
		if(element<p[mid+1] && p[mid]<=element)
			return mid;
		if(p[mid]>element)
			return binaryLast(l,mid-1,element);
		return binaryLast(mid+1,r,element);
	}
		return -1;
}
int main()
{
   	scanf("%d",&n);
   	for(int i=1;i<=n;i++)
   		scanf("%d",&p[i]);
   	
   	sort(p+1,p+n+1);
   	
   	/*for(int i=0;i<n;i++)

   	p[0]=0;
	p[n+1]= INT_MAX;

   	{
   		scanf("%d %d",&l,&r);
			int ans1 = binaryFirst(0,n+1,l);
		int ans2 = binaryLast(0,n+1,r);

   	}
   	
    return 0;
}
