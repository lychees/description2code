#include<bits/stdc++.h>
using namespace std;
int main()
{
    int test,num;
    long long int arr[1001];
    arr[0]=0,arr[1]=1,arr[2]=2,arr[3]=3,arr[4]=5,arr[5]=8;
    for(int i=6;i<=1000;i++)
        arr[i]=arr[i-1]+arr[i-2];
    scanf("%lld",&test);
    while(test--)
    {
        scanf("%lld",&num);
        printf("%lld\n",arr[num]);
    }
}
