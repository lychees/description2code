#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t; cin >> t;
    while(t--)
    {
        string s; cin >> s;
        int ans=0,i;
        for(i=1;i<s.length();i++)
        {
            if(s[i]==s[i-1])
                ans++;
        }
        cout << ans << endl;
    }
    return 0;
}
