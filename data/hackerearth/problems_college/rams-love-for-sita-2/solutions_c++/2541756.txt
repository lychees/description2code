#include <iostream>
using namespace std;

int main()
{
    int t,n,c,m,choco,temp,r;
    cin>>t;
    while(t--){
    	cin>>n>>c>>m;
    	choco = n/c;
    	temp = choco;
    	while(temp>=m){
    		r = temp/m;
    		choco+=r;
    		temp=temp%m + r;
    	}
    	cout<<choco<<endl;
    }
    return 0;
}
