#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    scanf("%d",&n);
    vector <string> v;
    v.clear();int k,c=0;string s,s1;
    for(int i=0;i<n;i++)
    {
        k=0;
        cin>>s;
        if(s.length()>1)
        {
        s1=s.substr(1,s.length()-2);
        //cout<<s1<<endl;
        sort(s1.begin(),s1.end());
        //cout<<s1<<endl;
        s=s[0]+s1+s[s.size()-1];
        }
        for(int j=0;j<v.size();j++)
        {
            if(v[j]==s)
            {
                k=1;
                break;
            }
        }
        if(k==0)
        {
            c++;
            v.push_back(s);
        }
    }
    printf("%d\n",c);
    return 0;
}
