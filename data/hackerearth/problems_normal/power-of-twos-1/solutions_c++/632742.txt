#include <iostream>
#include <vector>
#include <string>
#include <queue>
#include <algorithm>
#include <cstdio>

using namespace std;
#define MAX 1000000

int func(int n){
	int ret = 1;
	int i=1;
	for(i=2; i<=n; i++){
		int x=1;
		while(n%i==0){
			x++;
			n/=i;
		}
		ret *= x;
	}
	if(n>1) ret++;
	return ret-1;
}

int main(){
	int T;
	cin >> T;
	vector<int> nums(MAX+1, 0);
	for(int i=1; i<=MAX; i++){
		for(int j=i; j<=MAX; j+=i){
			nums[j]++;
		}
	}
	for(int i=1; i<=MAX; i++){
		nums[i] += nums[i-1] - 1;
	}
	for(int t=0; t<T; t++){
		int N;
		scanf("%d", &N);
		long long ans = nums[N];
		printf("%lld\n", ans);
	}
	return 0;
}
