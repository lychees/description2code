#include<iostream>
using namespace std;
int main()
{
	int wlimit,mstrength,ws[1000],temp;
	cin>>wlimit;
	for( int i=0; i<wlimit;i++)
		cin>>ws[i];
	cin>>mstrength;
	for( int i=0; i<wlimit; i++)
		for(int j=0;j<wlimit-1;j++)
			if( ws[j]<ws[j+1])
			{
				temp=ws[j];
				ws[j]=ws[j+1];
				ws[j+1]=temp;
			}
	int count,sum=0;
	for( int i=0; i<wlimit;i++)
	{
		if( sum >= mstrength)
			break;
		sum+=ws[i];
		count++;
	}
	cout<<count;
	return 0;
}
