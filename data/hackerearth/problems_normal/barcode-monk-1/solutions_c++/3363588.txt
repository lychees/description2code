#include <iostream>
#include <string.h>
#define MOD 1000000007
#define ll long long int
using namespace std;

bool sizes[1004];
int dp[1003][2];
int main()
{
    //cout << "Hello World!" << endl;
    int n,w,wi;
    memset(dp,0,sizeof(dp));
    memset(sizes,false,sizeof(sizes));
    sizes[0]=true;
    cin >> n >> w;
    for(int i=0;i<w;i++){
    	cin >> wi;
    	if(wi > n)
    		continue;
    	for(int j=wi;j<=n;j++){
    		sizes[j]|=sizes[j-wi];
    	}
    }
 	dp[0][0]=1;
    for(int i=1;i<=n;i++){
		dp[i][0]=(dp[i-1][0]+dp[i-1][1])%MOD;
		for(int j=1;j<=i;j++){
			if(sizes[j]){
				dp[i][1]=(dp[i][1]%MOD+dp[i-j][0]%MOD)%MOD;
			}
		}
    }
    
    
    cout << (dp[n][0]%MOD+dp[n][1]%MOD)%MOD <<endl;
    return 0;
}
