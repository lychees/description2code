#include<iostream>
#define ll long long
using namespace std;
ll countFact(ll n, int p)
{
    ll k=0;
    while (n>0)
    {
        k+=n/p;
        n/=p;
    }
    return k;
}
ll x[1000001];
ll cum[1000001];

int main()
{
int T;
cin>>T;
cum[0]=0;
for(int i=1;i<=1000000;i++)
	{x[i]=countFact(i,5);
	cum[i]=cum[i-1]+x[i];}
while(T--)
{int a,b;
 cin>>a>>b;
 ll s=0;
 s=cum[b]-cum[a-1];
 //ll ans=countFact(s,5);	
 //printf("%lld\n",s);
 cout<<s<<endl;
}
 return 0;
}
