#include <bits/stdc++.h>
#define lli long long int 
#define size 105
using namespace std;
 
lli a[size][size];
lli n ,c;
lli dp[size][size];
lli temp_ans;
 
 
int main(){
 
	lli test;
	scanf("%lld",&test);
	
	while(test--){
		scanf("%lld%lld",&n,&c);
		temp_ans=0;
		memset(dp,0,sizeof(dp));
		
		for(lli i=0;i<n;++i){
			for(lli j=0;j<n;++j){
				scanf("%lld",&a[i][j]);
				dp[i][j]=a[i][j];
			}
		}
		dp[0][0]=0;
		
		for(lli i=0;i<n;++i){
 
			for(lli j=0;j<n;++j){
				lli up = 0;
				lli left =0;
				if(i-1>=0){
					up = dp[i-1][j];
 
				}
				if(j-1 >=0){
					left = dp[i][j-1];
				}
				dp[i][j]+=max(up , left);
			}
		}
 
		lli ans = dp[n-1][n-1];
		if(ans>=c){
			ans -= c;
			printf("%lld\n",ans);
		}else{
			printf("-1\n");
		}
	}
	return 0;
}
