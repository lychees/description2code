#include <iostream>
using namespace std;
int x[5000001];
long long o[5000001],e[5000001];
int main()
{
    long long t,n,i,j,so,se;
    ios_base::sync_with_stdio(false); cin.tie(0);
    cin>>t;
    o[0]=e[0]=0;
    
    for(j=1;j<=t;j++)
    {
    	cin>>n;
    	so=se=0;
    	
    	for(i=1;i<=n;i++)
    	cin>>x[i];
    	
    	for(i=1;i<=n;i++)
    	{
    		if(x[i]%2==0)
    		{
    			
    			e[i]=e[i-1]+1;
       			o[i]=o[i-1];
    		}
    		else
    		{
    			
    			e[i]=o[i-1];
    			o[i]=e[i-1]+1;
    		}
    		so=so+o[i];
    		se=se+e[i];
    	}
    	if(so>=se)
    	cout<<"BOB"<<endl;
    	else
    	cout<<"ANDI"<<endl;
    }
}
