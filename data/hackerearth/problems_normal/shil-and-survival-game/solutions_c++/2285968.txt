#include<bits/stdc++.h>
using namespace std;

#define PB push_back
#define MP make_pair
#define F first
#define S second
#define LL long long
const int MOD = 1e9 + 7;
const int INF = INT_MAX;

int main()
{
	int n,in[112345];
	cin >> n;
	int idx,ma = -1;
	for(int i=0;i<n;i++)
	{
		cin >> in[i];
		if(in[i] > ma)
		{
			ma = in[i];
			idx = i;
		}
	}
	
	vector<int> ans;
	ma = -1;
	for(int i=0;i<=idx;i++)
	{
		if(in[i] > ma)
		{
			ans.PB(i);
			ma = in[i];
		}
	}
	ma = -1;
	for(int i=n-1;i>idx;i--)
	{
		if(in[i] > ma)
		{
			ans.PB(i);
			ma = in[i];
		}
	}
	sort(ans.begin(),ans.end());

	for(int i=0;i<ans.size();i++)
		cout << ans[i] + 1 << " ";
	cout << endl; 
	
	return 0;
}
