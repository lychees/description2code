long long kadane(int a[], int size)
{
   int max_so_far = a[0], i;
   int curr_max = a[0];
 
   for (i = 1; i < size; i++)
   {
        curr_max = maxi(a[i], curr_max+a[i]);
        max_so_far = maxi(max_so_far, curr_max);
   }
   return max_so_far;
}
