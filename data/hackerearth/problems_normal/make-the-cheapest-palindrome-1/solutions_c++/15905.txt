#include<cstdio>
#include<string>
#include<algorithm>
#include<iostream>
using namespace std;

int main()
{
   int T;
   cin >> T;
   while(T--){
      string s;
      cin >> s;
      int a,b;
      cin >>a>>b;
      int left=0,right=s.size()-1;
      long long cost=0;
      for(;left<right;left++,right--){
         if(s[left]==s[right]) {
            if(s[left]=='/') cost+=2*min(a,b);
            continue;
         }
         if(s[left]=='a') {
            if(s[right]=='/') cost+=a;
            else if(s[right]=='b') {
               cost=-1;
               break;
            }
            continue;
         }
         if(s[left]=='b') {
            if(s[right]=='/') cost+=b;
            else if(s[right]=='a') {
               cost=-1;
               break;
            }
            continue;
         }
         if(s[left]=='/') {
            if(s[right]=='a') cost+=a;
            else if(s[right]=='b') cost+=b;
            continue;
         }
         if(s[right]=='/') {
            if(s[left]=='a') cost+=a;
            else if(s[left]=='b') cost+=b;
            continue;
         }



      }
      cout <<cost<<endl;
   }

   return 0;
}
