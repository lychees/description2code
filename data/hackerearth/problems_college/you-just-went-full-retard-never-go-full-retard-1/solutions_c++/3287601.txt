#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long int n,flag = 1;
		scanf("%lld",&n);
		long long int i,s = sqrt(n);
		if(n==1||n==0)
		{
			flag = 0;
			printf("NO\n");
		}
		
		for(i = 2; i <= s; i++)
		{
			if(n%i==0)
			{
				flag = 0;
				printf("NO\n");
				break;
			}
		 } 
		if(flag)
		printf("YES\n"); 
	}
	return 0;
}
