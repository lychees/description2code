#define ll long long int
#define maxN 100000
#define maxVal 100000000
#define minVal -100000000
#define mod 1000000007LL

#define gcd(a,b) __gcd(a,b)

using namespace std;

int n;
char a[maxN+5];

#define b1 33
#define m1 10000009
#define b2 163
#define m2 10000007

ll p1[maxN+5],p2[maxN+5];
ll h1[maxN+5],h2[maxN+5];

void preHash()
{
	int i;
	p1[0]=p2[0]=1;
	for(i=1;i<=n;i++)
	{
		p1[i]=(b1*p1[i-1])%m1;
		p2[i]=(b2*p2[i-1])%m2;
	}
	h1[0]=h2[0]=0;
	for(i=1;i<=n;i++)
	{
		h1[i]=(b1*h1[i-1]+a[i-1]-'a'+1)%m1;
		h2[i]=(b2*h2[i-1]+a[i-1]-'a'+1)%m2;
	}
}

ll hash1(int l,int r)
{
	int x=r-l+1;
	ll v;
	v=(h1[r]-(h1[l-1]*p1[x])%m1+m1)%m1;
	return v;
}

ll hash2(int l,int r)
{
	int x=r-l+1;
	ll v;
	v=(h2[r]-(h2[l-1]*p2[x])%m2+m2)%m2;
	return v;
}

int main()
{
    int i,q,l1,r1,l2,r2;
    ll h11,h12,h21,h22;
    scanf("%s%d",a,&q);
    n=strlen(a);
    preHash();
    while(q--)
    {
    	scanf("%d%d%d%d",&l1,&r1,&l2,&r2);
    	if((r1-l1)!=(r2-l2))
    	{
    		printf("No\n");
    		continue;
    	}
    	h11=hash1(l1,r1);
    	h12=hash2(l1,r1);
    	h21=hash1(l2,r2);
    	h22=hash2(l2,r2);
    	if(h11==h21&&h12==h22)
    		printf("Yes\n");
    	else
    		printf("No\n");
    }
    return 0;
}
