import string
alphabets = list(string.lowercase)
digits = [i for i in range(1, 27)]
usecase = {}
for i, j in zip(alphabets, digits):
    usecase[i] = j
n = input()
for i in range(n):
    product = 1
    user_input = raw_input()
    if list(user_input) == list(user_input)[::-1]:
        print "Palindrome"
    else:
        for j in list(user_input):
            product = product * usecase.get(j)
        print product
