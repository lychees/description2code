#include <bits/stdc++.h>
using namespace std;
#define MIN(a,b) ((a) < (b)) ? (a) : (b)
int out[20000004];
int main(){
	int t;
	scanf("%d",&t);
	out[0]=0;
        out[1] = 0;
        for(int i=2; i <= 20000000; i++) {
                out[i] = out[i-1] + 1;
                if (i%2 == 0) {
                        out[i] = MIN(1 + out[i/2], out[i]);
                }
                if (i%3 == 0) {
                        out[i] = MIN(1 + out[i/3], out[i]);
                }
        }
	while(t--){
        int num;
        //int i;
        scanf("%d", &num);
        printf("%d\n", out[num]);
	}
        return 0;
}
