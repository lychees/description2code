# Modify Sequence

n = input()
A = map(int, raw_input().split(" "))
s = sum(A)
k = n-1
p, f = 0, 0
if s % 2 == 1:
    print 'NO'
else:
    while k > 0:
        if A[k] > 0:
            p = p+(2*A[k])
            A[k-1] = A[k-1]-A[k]
            A[k] = 0
        if A[k-1] < 0:
            f = 1
            break
        if A[k] == 0:
            k -= 1
    if p == s and f == 0:
        print 'YES'
    else:
        print 'NO'
