import sys
t = raw_input()
t = int(t)
while (t):
    t = t-1
    num = map(int, sys.stdin.readline().split(' '))
    n0 = num[0]
    n1 = num[1]
    n2 = num[2]
    n3 = num[3]
    n4 = num[4]
    s = n0*n2+n1*n3
    i = min(n0, n1)
    n0_s = (n2*(n0-i))+(n3*(n1-i))+2*i*n4
    if (n0_s > s):
        s = n0_s
    print s
