#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>

int main()
{
       int t,i,j,n,a[100000],L,R,k;
       scanf("%d",&t);
       for(i=0;i<t;i++)
       {
              L = 0;
              R = 0;
              k = 0;
              scanf("%d",&n);
              for(j=0;j<n;j++)
              {
                     scanf("%d",&a[j]);
                     R = R + a[j];
              }
              for(j=0;j<n;j++)
              {
                     R = R - a[j];
                     if(R==L)
                     {
                           k++;
                           continue;
                     }
                     L = L + a[j];
              }
              if(k>0)
              {
                     printf("YES\n");
              }
              else
              {
              	printf("NO\n");
              }
       }
       return 0;
}
