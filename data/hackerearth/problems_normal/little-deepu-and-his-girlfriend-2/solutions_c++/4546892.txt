#include <stdio.h>
 
int main()
{
    //printf("Hello World!\n");
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n,m;
    	scanf("%d%d",&m,&n);
    	
    	int b[m+1];
    	for(int i=0;i<=m;i++)
    	b[i]=0;
    	
    	int a[n];
    	for(int i=0;i<n;i++){
    	scanf("%d",&a[i]);
    	b[a[i]]=1;
    	}
    	
    	
    	for(int i=0;i<=m;i++)
    	{
    		for(int j=0;j<n;j++)
    		{
    			if(i-a[j]>=0&&b[i-a[j]]==0){
    			b[i]=1;
    			break;}
    		}
    	}
    	if(b[m])
    	printf("Little Deepu\n");
    	else
    		printf("Kate\n");
    }
    return 0;
}
