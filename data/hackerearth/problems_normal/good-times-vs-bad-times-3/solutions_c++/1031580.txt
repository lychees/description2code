#include <iostream>
#include <vector>
#include <bits/stdc++.h>
using namespace std;
#define pb push_back

bool f[100];
vector<int> vec;

void pre()
{
	for(int i=2;i<=60;i++) {
		if(!f[i]) {
			vec.pb(i);
			for(int j=i*i;j<=60;j+=i) f[j] = 1;
		}
	}
}

int arr[86410];

int main()
{
	pre();
    for(int i=(24*60*60)-1;i>=0;i--) {
    	int h = i/3600;
    	int m = (i%3600)/60;
    	int s = i%60;
    	bool ff=0;
    	for(int j=0;j<vec.size();j++) {
    		ff |= (h%vec[j]==0)&&(m%vec[j]==0)&&(s%vec[j]==0);
    	}
    	arr[i] = ff + arr[i+1];
    }
    int t,h,m,s;
    cin >> t;
    while(t--) {
    	cin >> h >> m >> s;
    	int n = arr[h*3600+m*60+s];
    	int d = 86400 - (h*3600+m*60+s);
    	d -= n;
    	int gc = 1;
    	if(n && d) gc = __gcd(n,d);
    	cout << n/gc << ":" << d/gc << endl;
    }
    return 0;
}
