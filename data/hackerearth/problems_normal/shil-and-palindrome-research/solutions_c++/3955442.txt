#include<iostream>
#include<algorithm>

using namespace std;

long long fen[28][100005],n;

void update(char c,long long pos,long long val){

    for(;pos<=n;pos+= pos&-pos){
        fen[c-'a'][pos]+=val;
    }


}


long long getsum(long long pos,long long i){

    long long sum=0;
    for(;pos>0;pos-= pos&-pos){
        sum+=fen[i][pos];
    }
    return sum;


}

int main(){

    long long q,i;
    cin>>n>>q;
    string s;
    cin>>s;
    for(i=0;i<n;i++){
        update(s[i],i+1,1);
    }
    while(q--){
        long long x,l,r,odd=0,even=0,sum=0,i;
        char c;
        cin>>x;
        if(x==1){
            cin>>l>>c;
            update(s[l-1],l,-1);
            update(c,l,1);
            s[l-1]=c;
        }
        else {
            cin>>l>>r;
            for(i=0;i<26;i++){
                if(l==1)sum=getsum(r,i);
                else sum=getsum(r,i)-getsum(l-1,i);
                if(sum%2!=0)odd++;
                else even++;
            }
            if(odd>1)cout<<"no\n";
            else cout<<"yes\n";
        }
    }
    return 0;









}
