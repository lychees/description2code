#include <bits/stdc++.h>
using namespace std;

#define IOS	ios_base::sync_with_stdio(false); cin.tie(NULL);

int main()
{
    IOS
    int t;
    cin >> t;
    for(int i=0; i<=t; i++){

        string x,w;
        set <string> s;
        getline(cin,x);
        stringstream second(x);
        while(second >> w)
        	s.insert(w);
        if(i!=0)
        	cout << s.size()<< endl;
    }
    return 0;
}
