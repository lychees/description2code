#include<bits/stdc++.h>
using namespace std;

struct fan
{
    int index;
    long long int ready_time;
};

bool cmp(struct fan a , struct fan b)
{
    if(a.ready_time==b.ready_time)
    	return a.index<b.index;
    else
    	return a.ready_time<b.ready_time;
}



int main()
{

    int n;
    cin>>n;

    fan orders[n];

    for(int i=0;i<n;i++)
    {
        orders[i].index=i;
        long long int ti,di;
        cin>>ti>>di;

        orders[i].ready_time=ti+di;
    }

    sort(orders,orders+n,cmp);

    for(int i=0;i<n;i++)
        cout<<orders[i].index+1<<" ";
}
