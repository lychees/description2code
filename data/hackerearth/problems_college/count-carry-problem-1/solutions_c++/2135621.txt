#include <iostream>
using namespace std;
int main()
{
	int tc,i,count;
    long long j,x,y,a,b;
    cin>>tc;
    while(tc--)
    {
        cin>>x>>y;
        count=0;
        i=0;
        while(x>0 or y>0)
        {
            a=x%10;
            b=y%10;
            if(a+b+i>=10)
            {
                count++;
            }
            j=(a+b+i)/10;
            i=j;
            x=x/10;
            y=y/10;
        }
        if(count==0)
            cout<<"No carry operation\n";
            else if(count==1)
             cout<<"1 carry operation\n";
        else
            cout<<count<<" carry operations\n";
    }
 
    return 0;
}
