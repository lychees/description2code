#include <cstdio>
using namespace std;
int n;
char a[105];
int hash[26];
int main() {
	int i,j;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%s",a);
		for(j=0;a[j];j++)
		{
			if(hash[a[j]-'a']==i)
			hash[a[j]-'a']++;
		}
	}
	int cnt=0;
	for(i=0;i<26;i++)
	{
		if(hash[i]==n)
		cnt++;
	}
	printf("%d\n",cnt);
	return 0;
}
