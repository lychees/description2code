max = 10**6+1
p = [0]*max


def rev_seive():

    for i in range(2, max):
        if not p[i]:
            for j in range(2*i, max, i):
                p[j] = 1

    p[4] = 0
    p[1] = 1
    for k in range(1, max):
        p[k] += p[k-1]


rev_seive()


for _ in range(input()):
    l, r = map(int, raw_input().split())
    print p[r]-p[l-1]
