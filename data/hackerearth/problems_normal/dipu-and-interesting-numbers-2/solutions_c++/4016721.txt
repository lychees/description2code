#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	unsigned long long l,r;
    	cin>>l>>r;
    	
    	unsigned long long ans= (long long)sqrt(r)-(long long)sqrt(l);
    	unsigned long long a=sqrt(l),b=sqrt(r);
    	if(a*a==l)
    	ans+=1;
    	if(b*b>r)
    	ans-=1;
    	cout<<ans<<endl;
    }
    return 0;
}
