#include<bits/stdc++.h>
using namespace std;
int main()
{
	vector<string> p;
	int N;
	cin>>N;
	for(int i=0;i<N;i++)
	{
		string s;
		cin>>s;
		sort(s.begin(),s.end());
		p.push_back(s);
	}
	sort(p.begin(),p.end());
	int t=1;//it is because every string is present atleast once na that 's why
	int ans=-89080;
	for(int i=1;i<N;i++)
	{

		if(p[i]==p[i-1])t++;
		else t=1;
		ans=max(ans,t);


	}
	cout<<ans<<endl;







	return 0;
}
