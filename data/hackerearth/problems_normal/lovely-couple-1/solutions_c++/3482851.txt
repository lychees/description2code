#include <iostream>
using namespace std;

int gcd(int m, int n) {
	if(n == 0)return m;
	return gcd(n, m % n);
}
bool check[1000001];
unsigned int primes[1000001];

void sieve()
{
	register unsigned i,j;

	for(i=2;i<=1000000;i++)check[i] = true;

	for(i=2;i<=1000;i++)
	{
		if( check[i])
		{
			j = i * i;

			while( j <= 1000000 )
			{
				check[j] = false;
				j += i;
			}
		}
	}

	j=0;
	for(i=2;i<=1000000;i++)
		if(check[i])primes[j++] = i;
	primes[j]=1000001;
}
int main()
{
	sieve();
	int t;
	cin >> t;
	
	while(t-- > 0) {
		int a, b;
		cin >> a >> b;
		long long lcm = (long long)a * b / gcd(a, b);
		
		int cnt = 0;
		int i = 0;
		while(lcm >= 2) {
			if(lcm % primes[i] == 0) {
				cnt++;
				while(lcm % primes[i] == 0) lcm = lcm / primes[i];
			}
			i++;
		}
		bool res = check[cnt];
		if(res) cout << "Yes" << endl;
		else cout << "No" << endl;
	}
    return 0;
}
