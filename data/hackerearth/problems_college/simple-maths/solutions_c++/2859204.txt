#include <iostream>
using namespace std;
long long int mi(long long int a,long long int b)
{
	if ( a==0 ) return b;
  return mi( b%a, a );
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	long long int a,b,n,c,count=0;
    	scanf("%lld%lld%lld",&n,&a,&b);
    	//if(n<0) n*=-1;
    	if(a==0 || b==0) count= -1;
    	else 
    	{
	    	c=a*b;
	    	c/=mi(a,b);
	    	count=(((n/a)+(n/b))-(n/c));
    	}
    	printf("%lld\n",count);
    }
    return 0;
}
