#include<iostream>
using namespace std;
int palindrome(int);
int main()
{int t,a[10],b[10],count=0,d;
 cin>>t;
 for(int i=0;i<t;i++)
  cin>>a[i]>>b[i];
 for(int i=0;i<t;i++)
  {count=0;
   if(a[i]<b[i])
    {for(int j=a[i];j<=b[i];j++)
      {d=palindrome(j);
       if(d==1)
        count++;
	  }
	}
   else
    {for(int j=b[i];j<=a[i];j++)
      {d=palindrome(j);
       if(d==1)
        count++;
	  }
	}
   cout<<count<<endl;		
  }
 return 0;  
}
int palindrome(int n)
{int a,sum=0,dig;
 a=n;
 while(a>0)
  {dig=a%10;
   sum*=10;
   sum+=dig;
   a/=10;
  }
  if(sum==n)
   return 1;
  else
   return 0; 
}
