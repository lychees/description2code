#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
    cin>>t;char s[100000];
    while(t--)
    {
    	cin>>s;
    	int l=strlen(s);
    	int i=0;
    	int sum=0;
    	while(i<l)
    	{
    		int k=1;
    		while(s[i]==s[i+1]&&i<l) {k++;i++;}
    		sum+=k-1;
    		i++;
    	}
    	cout<<sum<<endl;
    }
}
