#include<bits/stdc++.h>
using namespace std;

#define mod 1000000007
typedef set<string> ss;
typedef vector<int> vs;
typedef map<int,char> msi;
typedef pair<int,int> pa;
typedef long long int ll;

ll dp[10005][6],i,j,k,l,temp,n,r;
char s[10005];
int main()
{
	//freopen("i.in", "r", stdin);
  	//freopen("o.out", "w", stdout);
  	int t;
	cin>>t;
	while(t--)
	{
		cin>>s;
		n=strlen(s);
		r=0;
		for(i=0;i<n;i++)
		{
			if(s[i]!='0')
			r++;
		}
		for(i=0;i<n;i++)
		{
			for(j=0;j<=5;j++)
			dp[i][j]=INT_MIN;
		}
		dp[0][1]=s[0]-'0';
		for(i=1;i<n;i++)
		{
			for(l=1;l<=5;l++)
			{
				if(l==1)
				{
					if(s[0]=='0')
					continue;
					temp=0;
					for(j=0;j<=i;j++)
					temp=temp*10+s[j]-'0';
					if(temp>1000000000000)
					continue;
					dp[i][1]=temp;
					//cout<<"i="<<i<<" l="<<l<<" "<<dp[i][l]<<"\n";
					continue;
				}
				for(j=max(1LL,i-12);j<=i;j++)
				{
					if(j!=i && s[j]=='0')
					continue;
					temp=0;
					for(k=j;k<=i;k++)
					{
						temp=temp*10+s[k]-'0';
					}
					if(temp>1000000000000)
					continue;
					if(dp[j-1][l-1]>=0)
					dp[i][l]=max(dp[i][l],dp[j-1][l-1]+temp);
					//cout<<"i="<<i<<" l="<<l<<" "<<dp[i][l]<<"\n";
				}
			}
		}
		if(dp[n-1][4]==INT_MIN)
		cout<<"unlucky\n";
		else
		cout<<dp[n-1][4]<<"\n";
	}
	return 0;
}
