#include<iostream>
using namespace std;
int main()
{
	int t,a;
	char ch;
	cin>>t;
	while(t>0)
	{
		cin>>a;
		int corx=0,cory=0,ctr=0;
		for(int x=0; x<a; x++)
		{
			for(int y=0; y<a; y++)
			{
				cin>>ch;
				if(ch == '*') {cory=y; corx=x;}
			}
		}
		while(corx!=a/2)
		{
			if(corx>a/2) corx--;
			else corx++;
			ctr++;
		}
		while(cory!=a/2)
		{
			if(cory>a/2) cory--;
			else cory++;
			ctr++;
		}
		cout<<ctr;
		if(--t) cout<<endl;
	}
	return 0;
}
