#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%lld",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
#define mod 1000000007
int main()
{
    char a[15];
    long long int T=0,i,j,ans,n;
    sd(T);
    while(T--)
    {
        ss(a);
        n=strlen(a);
        for(i=0,j=n-1;i<j;i++,j--)
        {
            if(a[i]!=a[j])
                break;
        }
        if(i>=j)
            printf("Palindrome\n");
        else
        {
            ans=1;
            for(i=0;i<n;i++)
            {
                ans=ans*(long long int)(a[i]-'a'+1);
            }
            printf("%lld\n",ans);
        }
    }
	return 0;
}
