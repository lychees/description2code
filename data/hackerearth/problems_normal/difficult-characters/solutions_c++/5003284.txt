#include <iostream>
#include <string>
#include <vector>
using namespace std;

void procesar(){
	string pal;
	vector<int> c(26,0);
	int mayor = 0;
	int pos;
	cin >> pal;
	for(int i = 0; i < pal.length(); i++){
		pos = pal.at(i) - 'a'; 
		c[pos]++;
		if(c[pos] > mayor) mayor = c[pos];
	}
	for(int j = 0; j <= mayor; j++)
		for(char i = 'z'; i >= 'a'; i--){
			pos = i -'a';
			if(c[pos] == 0) {
				cout << i << " ";
			}
			c[pos]--;
		}
	cout << endl;	
}
int main(){
	int t;
	cin >> t;
	for(int i = 0; i < t; i++){
		procesar();
	}
	return 0;
}
