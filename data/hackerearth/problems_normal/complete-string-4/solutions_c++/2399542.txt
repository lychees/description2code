#include <iostream>
using namespace std;

int main()
{
	int a[25],N,count;
	string str;
	cin>>N;
	while(N>0)
	{
		count=0;
		cin>>str;
		for(int i=0;i<26;i++)
		{
			a[i]=0;
		}
		for(int j=0;j<str.size();j++)
		{
			a[str[j]-97]=1;
		}
		for(int i=0;i<26;i++)
		{
			if(a[i]==0)
			{
				count=1;
			}
		}
		
		if(count==1)
		{
			cout<<"NO"<<endl;
		}
		
		else
		cout<<"YES"<<endl;
		
		
		N--;
	}
	
    return 0;
}
