#include<iostream>
using namespace std;

int power(int x, int y){
	if(y == 1)	return x;
	
	int temp = power(x, y / 2);
	if(y % 2 == 0)	return temp * temp;
		else	return x * temp * temp;
}

int main(){
	int a, n, i, sum = 0;
	cin >> a >> n;
	for(i = 1; i <= n; i++)	sum = sum + power(a, i);	
	cout << sum << endl;
	return 0;
}
