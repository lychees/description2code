#include <iostream>
using namespace std;

int main()
{
    long a,e,t,o,u,num;
    string s;
    a=0;e=0;t=0;o=0;u=0;
    cin>>num;
    cin>>s;
    for(long i=0;i<num;i++)
    {
    	if(s[i]=='a')
    	{
    		a++;
    		break;
    	}
    }
    for(long i=0;i<num;i++)
    {
    	if(s[i]=='e')
    	{
    		e++;
    		break;
    	}
    }
    for(int i=0;i<num;i++)
    {
    	if(s[i]=='i')
    	{
    		t++;
    		break;
    	}
    }
    for(long i=0;i<num;i++)
    {
    	if(s[i]=='o')
    	{
    		o++;
    		break;
    	}
    }
    for(long i=0;i<num;i++)
    {
    	if(s[i]=='u')
    	{
    		u++;
    		break;
    	}
    }
    if((a!=0)&&(e!=0)&&(t!=0)&&(o!=0)&&(u!=0))
    {
    cout<<"YES"<<endl;
    }
    else 
    {
    	cout<<"NO"<<endl;
    	}
    return 0;
}
