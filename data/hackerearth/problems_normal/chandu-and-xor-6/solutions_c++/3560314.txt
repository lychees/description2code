#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int n,m;
    cin>>n>>m;
    vector< vector<long> > v(n,vector<long>(m));
    for(int i=0;i<n;i++){
    	for(int j=0;j<m;j++)
    	cin>>v[i][j];
    }
    vector<long> row(n,0);
    vector<long> col(m,0);
    for(int i=0;i<n;i++){
    	for(int j=0;j<m;j++)
    	row[i]+=v[i][j];
    }
    for(int i=0;i<m;i++){
    	for(int j=0;j<n;j++)
    	col[i]+=v[j][i];
    }
    long max=0;
    for(int i=0;i<n;i++){
    	for(int j=0;j<m;j++){
    		long temp=row[i]^col[j];
    		if(temp>max){
    			max=temp;
    		}
    	}
    }
    cout<<max;
}
