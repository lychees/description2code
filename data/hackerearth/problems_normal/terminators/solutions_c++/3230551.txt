#include <bits/stdc++.h>
using namespace std;
struct node
{
	int x;
	int y;
};
node a[1005];
int comp(node m,node n)
{
	if(m.x<n.x)
	return 1;
	else 
	return 0;
	
}
int main()
{
	int t,n,x,y;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		for(int i=0;i<n;i++)
		{
			scanf("%d%d",&a[i].x,&a[i].y);
		}
		sort(a,a+n,comp);
		int lis[n];
		int ans=0;
		for(int i=0;i<n;i++)
		{
			lis[i]=1;
			for(int j=0;j<i;j++)
			{
				if(a[i].y>a[j].y&&lis[j]+1>lis[i])
				{
					lis[i]=lis[j]+1;
				}
			}
			ans=max(ans,lis[i]);
			
		}
		cout<<ans<<"\n";
	}
	return 0;
}
