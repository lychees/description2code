#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
using namespace std;
typedef long long ll;
ll dp[100010];
int a[100005];
int main() {
    int n;
    int t;
    scanf("%d",&t);
    while(t--) {
        memset(dp,0,sizeof(dp));
        memset(a,0,sizeof(a));
        scanf("%d",&n);
        for(int i=1;i<=n;i++) scanf("%d",&a[i]);
        for(int i=n;i>=1;i--) {
            dp[i]=max(dp[i],a[i]+min(dp[i+2],dp[i+3]));
            dp[i]=max(dp[i],a[i]+a[i+1]+min(dp[i+3],dp[i+4]));
        }
        ll sum=0;
        for(int i=1;i<=n;i++) sum+=ll(a[i]);
        ll temp=sum-dp[1];
        if(dp[1]>temp) printf("LOSE\012");
        else if(dp[1]<temp) printf("WIN\012");
        else printf("DRAW\012");
    }
    return 0;
}
