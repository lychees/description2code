#include <iostream>
#include <string.h>
using namespace std;

void decrypt(char[]);

int main()
{
    int N;
    char str[200];
    cin>>N;
    cin.ignore();
    for(int i=0;i<N;++i)
    {
    	cin.getline(str,sizeof(str));
    	decrypt(str);
    	cout<<str<<endl;
    }
    return 0;
}

void decrypt(char str[])
{
	int N,i=0;
	//Finding size of 2D matrices
	for(N=1;N*N<strlen(str);++N);
	//Generating Matrix
	char code[14][14];
	for(int j=0;j<N;++j)
		for(int k=0;k<N;++k)
			code[k][j]=str[i++];
	i=0;
	for(int k=0;k<N;++k)
		for(int j=0;j<N;++j)
			str[i++]=code[k][j];
	str[i]='\0';
}
