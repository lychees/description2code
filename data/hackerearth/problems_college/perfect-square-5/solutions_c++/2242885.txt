#include <iostream>
#include <cmath>
using namespace std;

int main()
{ 	int t;
    cin>>t;
    while(t--){
    	long unsigned int n;
    	cin>>n;
    	long unsigned int a=(int)sqrt(n);
    	if(pow(a,2)==n)
    		cout<<"YES"<<endl;
    	else{
    	if(sqrt(n)-a >=0.5)
    		cout<<"+"<<pow(a+1,2)-n<<endl;
    	else
    		cout<<"-"<<n-pow(a,2)<<endl;
    	}
    }
    return 0;
}
