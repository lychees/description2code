#include<stdio.h>
#include<bits/stdc++.h>

using namespace std;

long a[100000];
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,i,count=1;
        scanf("%d",&n);
        for(i=0;i<n;i++)
            scanf("%ld",&a[i]);
        sort(a,a+n);
        for(i=1;i<n;i++)
        {
            if(a[i]!=a[i-1])
                count++;
        }
        printf("%d\n",count);
    }
    return 0;
}
