#include<bits/stdc++.h>
using namespace std;
#define MOD	1000000007
long long int a[2][2]={{3,-1},{1,0}},d[2][2];
void mul(long long int b[2][2])
{
	long long int x=d[0][0]*b[0][0]+d[0][1]*b[1][0];
	long long int y=d[0][0]*b[0][1]+d[0][1]*b[1][1];
	long long int z=d[1][0]*b[0][0]+d[1][1]*b[1][0];
	long long int w=d[1][0]*b[0][1]+d[1][1]*b[1][1];
	d[0][0]=(x+MOD)%MOD;
	d[0][1]=(y+MOD)%MOD;
	d[1][0]=(z+MOD)%MOD;
	d[1][1]=(w+MOD)%MOD;
}
void pow(long long int n)
{
	if(n<2)
		return;
	pow(n/2);
	mul(d);
	if(n%2)
		mul(a);
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		d[0][0]=3;
		d[0][1]=-1;
		d[1][0]=1;
		d[1][1]=0;
		long long int n;
		cin>>n;
		if(n==1)
			cout<<1<<endl;
		else if(n==2)
			cout<<3<<endl;
		else
		{
			pow(n-2);
			//cout<<d[0][0]<<" "<<d[0][1]<<endl;
			//cout<<d[1][0]<<" "<<d[1][1]<<endl;
			cout<<(3*d[0][0]+d[0][1]+MOD)%MOD<<endl;
		}
	}
}
