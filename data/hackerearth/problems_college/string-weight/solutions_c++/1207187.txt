#include <iostream>
using namespace std;

int main()
{
  int t,i;
  cin>>t;
  string s;
  int n;
  bool a[26];
  char c;
  int sum;
  while(t--)
  {sum=0;
  	for(i=0;i<26;i++)
  	a[i]=false;
  	
  	cin>>s>>n;
  	
  	for(i=0;i<n;i++)
  	{
  		cin>>c;
  		a[c-97]=true;
  	}
  	for(i=0;i<s.length();i++)
  	{
  		char x = s.at(i);
  		x-=97;
  		if(a[x])
  		continue;
  		sum+=(x+1);
  	}
  	cout<<sum<<"\n";
  }
    return 0;
}
