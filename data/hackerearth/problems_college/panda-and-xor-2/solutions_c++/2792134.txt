#include <iostream>
using namespace std;

int countSetBits(long long n)
{
    unsigned int count = 0;
    while (n)
    {
      n &= (n-1) ;
      count++;
    }
    return count;
}
int main()
{
    long long t,i,j,n,m;
    cin>>t;
    while(t--)
    {
        cin>>n>>m;
        if(countSetBits(n^m)%2==0)
        cout<<"YES\n";
        else
        cout<<"NO\n";
    }
    return 0;
}
