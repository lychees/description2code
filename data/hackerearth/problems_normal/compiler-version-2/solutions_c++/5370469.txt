#include <iostream>
#include <vector>

using namespace std;

int main()
{
    vector<string> line;
    string temp;
    while(getline(cin,temp))
    	line.push_back(temp);
    for(unsigned i=0;i<line.size();i++){
    	string::size_type cpos,mpos;  
    	unsigned var = 0;
    	cpos = line[i].find("//");
    	mpos = line[i].find("->");
    	
    		while(mpos < cpos){
    		 	line[i].replace(mpos,2,".");
    			mpos = line[i].find("->");
    			cpos = line[i].find("//");
    	    }
    
    	
    	cout<<line[i]<<"\n";
    }	
    
    return 0;
}
