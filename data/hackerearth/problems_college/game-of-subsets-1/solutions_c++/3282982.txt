#include <iostream>
using namespace std;

long long int  foo(long long  base, long long  exp, long long  mod) {
  base %= mod;
  long long int  result = 1;
  while (exp > 0) {
    if (exp & 1) result = (result * base) % mod;
    base = (base * base) % mod;
    exp >>= 1;
  }
  return result;
}
int main()
{
    long long int t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	cout<<foo(3,n, 1000000007)<<endl;
    }
    
    
}
