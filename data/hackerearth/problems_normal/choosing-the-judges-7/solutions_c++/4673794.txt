#include<bits/stdc++.h>
#define ll long int
using namespace std;
ll max(ll a,ll b)
{
	return a>b?a:b;
}
int main()
{
	int t,n;
	cin>>t;
	for(int io=1;io<=t;io++)
	{
		cin>>n;
		ll ar[n],dp[n];
		for(int i=0;i<n;i++)
		cin>>ar[i];
		dp[0]=ar[0];
		if(n==1)
		{
			cout<<"Case "<<io<<": "<<dp[0]<<endl;
			continue;
		}
		dp[1]=ar[0]>ar[1]?ar[0]:ar[1];
		for(int i=2;i<n;i++)
			dp[i]=max(dp[i-1],dp[i-2]+ar[i]);
		cout<<"Case "<<io<<": "<<dp[n-1]<<endl;
	}
	return 0;
}
