#include <iostream>
#include<math.h>
using namespace std;

int main()
{
   int t;
   cin>>t;
   long long int p,n,sides=1;
   //cout<<ceil(float(5)/float(2));
   while(t--)
     {
     	cin>>p>>n;
        while(n--)
          {sides=p;
          	p=ceil((double)(p)/(double)(2.0))+1;
          if(sides==p)
            {  break;}
          	
          }
          cout<<p<<"\n";
     }
    return 0;
}
