#include <bits/stdc++.h>
#define ll long long int
#define s(x) scanf("%lld", &x)

using namespace std;

int main()
{
	ll test;
	s(test);

	while (test--) {
		ll n,k;
		s(n);s(k);
		ll a[n];
		ll dp[n];
		for (int i = 0; i < n; i++) {
			s(a[i]);
			dp[i] =0;
		}

		for (int i = 0; i < n; i++) {
			int d1 = i + k;
			int d2 = i - k;
			while (d2 >= 0 || d1 < n) {
				if (d2 >= 0 && a[d2] < a[i]) {
					dp[i] = abs(d2 - i);
					break;
				}
				if (d1 < n && a[d1] < a[i]) {
					dp[i] = abs(d1 - i);
					break;
				}
				d2 -= 1;
				d1 += 1;
			}
		}
		ll sum = 0;

		for (int i = 0; i < n; i++) {
			sum += dp[i];
		}

		cout << sum << endl;
	}

	return 0;
}
