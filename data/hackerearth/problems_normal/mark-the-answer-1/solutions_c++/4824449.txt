#include <iostream>
using namespace std;

int main()
{
	long n,x,*ai,i;
	long long sum=0,sum2=0,maxmarks=0;
	bool flag(false);
	cin>>n>>x;
	ai=new long[n];
	for(i=0;i<n;i++)
	{
		cin>>ai[i];
		if (ai[i]<=x)
		{
			sum++;
		}
		else if(ai[i]>x && (!flag))
		{
			flag=true;
		}
		else if(ai[i]>x && (flag) && ai[i-1]<=x)
		{
			if(maxmarks<sum)
			maxmarks=sum;
			sum=0;
		}
		else if(ai[i]>x && (flag) && ai[i-1]>x)
		break;
	}
	if(maxmarks<sum)
	maxmarks=sum;
	cout<<maxmarks<<endl;
	return 0;
}
