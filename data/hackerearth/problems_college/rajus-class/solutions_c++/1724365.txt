#include<stdio.h>
#include<iostream>
#include<algorithm>
#include <string.h>
#include <math.h>


using namespace std;

struct node {
	string str;
	int age;
	
};
bool cmp(struct node i,struct node j) {
	return i.age < j.age;
}





int main()
{
	node dog[10001];
	int t;
	cin>>t;
	while (t--) {
		int n,i,j,x,ans;
		string y;
		cin>>n;
		for(i = 0 ;i < n ;i++ ) {
			cin>>x>>y;
			dog[i].age = x;
			dog[i].str = y;
		}
		sort(dog,dog+n,cmp);
		cin>>ans;
		cout<<dog[ans-1].str<<endl;
	}


return 0;
}
