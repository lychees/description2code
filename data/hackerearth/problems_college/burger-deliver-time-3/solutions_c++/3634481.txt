#include <iostream>
#include <algorithm>

using namespace std;

struct burger
{
	int idx;
	int t;

	bool operator < (const burger &other) const
	{
		if (t != other.t)
			return t < other.t;

		return idx < other.idx;
	}
};

int main()
{
	int n, t, d;
	burger b[1001];

	cin >> n;
	
	for (int i = 0; i < n; i++)
	{
		b[i].idx = i+1;
		cin >> t >> d;
		b[i].t = t+d;
	}

	sort(b, b+n);

	for (int i = 0; i < n; i++)
		cout << b[i].idx << " ";

	cout << endl;

	return 0;
}
