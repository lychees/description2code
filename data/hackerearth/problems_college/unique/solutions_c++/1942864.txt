#include<iostream>
#include<map>
using namespace std;
//#define max 100000000
//int b[max];
int main()
{
	int n,t;
	scanf("%d",&t);
	while(t--)
	{
	map<int,int> mp;
	scanf("%d",&n);
	int a[n];
	for(int i=0;i<n;i++)
	scanf("%d",&a[i]);
	int max=0,num;
	for(int i=0;i<n;i++)
	{
	   mp[a[i]]++;
	   if(max<mp[a[i]])
	   {
	      max=mp[a[i]];
	      num=a[i];
	   }
	}
	int personality=0;
	for(int i=0;i<n;i++)
	{
		if(mp[a[i]]!=-1)
		{
		   mp[a[i]]=-1;
		   personality++;
	    }
	}
	printf("%d\n",personality);
    }
}
