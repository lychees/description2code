#include<bits/stdc++.h>
using namespace std;



int main(){
	int n,cnt=0,cnt2=0,M=0,M2=0;
    string s;
    cin >> n;
    while(n--){
	    cin >> s;
	    for(int i=0;i<s.size();i++){
	    	if(s[i]=='C'){
	    		cnt++;
	    		cnt2++;
	    	}
	    	else{
	    		M=max(M,cnt);
	    		M2=max(M2,cnt2);
	    		cnt=0;
	    		cnt2=0;
	    	}
	    }
	    if(cnt>0){
	    	M=max(M,cnt);
	    	cnt=0;
	    }
	    cnt=0;
    }
    if(cnt2>0){
    	M2=max(M2,cnt2);
    }
    
    cout << M << " " << M2 << endl;
    return 0;
}
