#include<bits/stdc++.h>
#define MOD 1000000007
using namespace std;
int main()
{
    int test;
    scanf("%d",&test);
    while(test--)
    {
        long long int n,m;
        scanf("%lld%lld",&n,&m);
        if(__gcd(m,n)==1)
            printf("Is a Co-Prime\n");
        else
            printf("Not a Co-Prime\n");
    }
    return 0;
}
