#include <iostream>

using namespace std;

int main(void)
{
	int tc,n,m;
	int discount;
	int min,a,b,c;
	cin>>tc;
	while(tc--)
	{
		cin>>n>>m;
		for(int i=0;i<n;i++)
		{
			min=1000000;
			int index=0;
			for(int j=0;j<m;j++)
			{
				cin>>a>>b>>c;
				discount=(100-a)*(100-b)*(100-c);
				if(discount<min)
				{
					min=discount;
					index=j;
				}
			}
			cout<<index+1<<" ";
		}
		cout<<endl;
	}
	return 0;
}
