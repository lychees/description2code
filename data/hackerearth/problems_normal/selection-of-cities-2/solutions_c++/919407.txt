#include <iostream>
using namespace std;
inline long long int modpow(long long int n, long long int p)
{
	long long int u=1, w=2, r;
	while(n!=0)
	{
		if((n&0x1) != 0)
		u=(u*w)%p;
		if((n>>=1) != 0)
		w=(w*w)%p;
	}
	r=u%p;
	return r;
}
int main()
{
    std::ios::sync_with_stdio(false);
    long long int t, n, p=1000000007;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	cout<<modpow(n, p)-1<<endl;
    }
    return 0;
}
