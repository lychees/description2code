#include <iostream>
#include <vector>

using namespace std;
long int outputs [61][100007];
int counters[61];

int main() {
    int t;
    scanf("%d",&t);
    while(t--){
        int n;
        scanf("%d",&n);

        for(int i=0;i<61;i++){
            counters[i]=0;
        }

        while(n--){
            long int x;
            scanf("%ld",&x);
            int no1s=__builtin_popcount(x);
            if(no1s==1){
                printf("%ld",x);
                printf(" ");
            }
            else{
                outputs[no1s][counters[no1s]]=x;
                counters[no1s]=counters[no1s]+1;
            }
        }
        for(int i=2;i<61;i++){
                for(int j=0;j<counters[i];j++){
                    printf("%ld",outputs[i][j]);
                    printf(" ");
                }
        }
        printf("\n");
    }
    return 0;
}
