#include <iostream>
using namespace std;
string substr(string s,int start,int end)
{
	string temp="";
	for(int i=start;i<=end;i++)
	{
		temp+=s.at(i);
	}
	return temp;
}
int main()
{
	int t;
	cin>>t;
	while(t>0)
	{
		string s;
		cin>>s;
		if(s.size()>=10)
			cout<<"YES\n";
		else if(s.size()<=3)
			cout<<"NO\n";
		else
		{
			int flag=0;
			for(int i=0;i<s.size()-3;i++)
			{
				for(int j=i+1;j<s.size()-2;j++)
				{
					for(int k=j+1;k<s.size()-1;k++)
					{
							string s1=substr(s,0,i);
							string s2=substr(s,i+1,j);
							string s3=substr(s,j+1,k);
							string s4=substr(s,k+1,s.size()-1);
							if(s1.size()>0 && s2.size()>0 && s3.size()>0 && s4.size()>0)
							{
								if(s1!=s2 && s1!=s3 && s1!=s4 && s2!=s3 && s2!=s4 && s3!=s4)
								{
									flag=1;
									break;
								}
							}
					}
					if(flag==1)
						break;
				}
				if(flag==1)
					break;
			}
			if(flag==1)
				cout<<"YES\n";
			else
				cout<<"NO\n";
				
		}
		t--;
	}
}
