#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while (t--)
    {
    	int n, p, ans = 0;
    	cin >> n;
    	for ( int i = 0; i < n; i++ )
    	{
    		cin >> p;
    		ans += p;
    	}
    	cout << fixed << setprecision(8) << (double)((ans*100.0)/(n*100.0)) << endl;
    }
    return 0;
}
