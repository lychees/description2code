#include <iostream>
#include <string.h>
using namespace std;

int main()
{
	int t=0, i=0, k,l;
	char a[102], b[102];
    cin>>t;
    while(t--)
    {
    	cin>>a;
    	cin>>b;
    	l = strlen(a);
    	i = 0;
    	while (i<l)
    	{
    		k=int(a[i])-int(b[i]);
    		k *= k;
    		i++;
    		cout<<k;
    	}
    	cout<<"\n";
    }
    return 0;
}
