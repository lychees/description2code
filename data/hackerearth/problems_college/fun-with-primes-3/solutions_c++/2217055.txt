#include <bits/stdc++.h>
using namespace std;

int main()
{
    long int i,j,t,n,sum,prime[100]={0};
    memset(prime,0,sizeof(prime));
    for(i=2;i<=100;i++)
    {
    	if(prime[i]==0)
    	{
    		j=i+i;
    		for(;j<=100;j=j+i)
    			prime[j]=1;
    	}
    }
    cin>>t;
    prime[0]=1;
    prime[1]=1;
    while(t--)
    {
    	cin>>n;
    	sum=0;
    	while(n)
    	{
    		sum=sum+n%10;
    		n=n/10;
    	}
    	if(prime[sum]==0)
    		cout<<"YES\n";
    		else
    		cout<<"NO\n";
    }
    return 0;
}
