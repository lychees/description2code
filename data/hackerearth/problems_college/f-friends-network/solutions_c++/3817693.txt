#include<bits/stdc++.h>
using namespace std;
int cycle=0;
int color[505];
void dfs(long long u,vector<vector<long long> > &adjlist);
int main()
{
    int t;
    std::ios::sync_with_stdio(false);
    cin>>t;
    while(t--)
    {
        long i,j,tmp,n;
        cin>>n;
        vector<vector<long long> > adjlist(n+1);
        for(i=1;i<=n;i++)
        {
            for(j=1;j<=n;j++)
            {
                cin>>tmp;
                if(tmp==1)
                    adjlist[i].push_back(j);
            }
        }
        cycle=0;
        memset(color,0,sizeof(color));
        dfs(1,adjlist);
        if(cycle==1)
            cout<<"Aye\n";
        else
            cout<<"Nae\n";
    }
    return 0;
}
void dfs(long long u,vector<vector<long long> > &adjlist)
{
    long long v;
    color[u]=1;
    for(long long j=0;j<adjlist[u].size();j++)
    {
        v=adjlist[u][j];
        if(color[v]==1)
        {
            cycle=1;
        }
        if(color[v]==0)
           dfs(v,adjlist);
    }
    color[u]=2;
    return;
}
