#include<cstdio>
#include<iostream>
using namespace std;
int main()
{
   int tc;
   long int n,*a,s,p;
   scanf("%d",&tc);
   while(tc--)
   {
     scanf("%ld",&n);
     a=new long int[100000000];
     s=0;
     while(n--)
     {
       scanf("%ld",&p);
       if(a[p]!=1)
       {
	     s++;
	     a[p]=1;
       }
    }
    printf("%ld\n",s);
    delete a;
   }
return 0;
}
