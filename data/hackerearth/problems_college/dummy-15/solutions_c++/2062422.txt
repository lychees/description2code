    #include <cstdio>
    #include <string>
    using namespace std;
     
    int a[100000];
     
     
    void updatelis(int x,int &len) {
    	if (x < 0) {
    		return;
    	}
    	int y = -1, left = 0, right = len - 1;
    	while (left <= right) {
    		int mid = (left +right) >> 1;
    		if (a[mid] <= x) {
    			left = mid + 1;
    		}
    		else {
    			y = mid;
    			right = mid - 1;
    		}
    	}
    	a[(y >= 0)?(y):(len++)] = x;
    }
     
     
    int main() {
    int z,i,n,len,x;
    	for (scanf("%d",&z);z;--z) {
    		scanf("%d",&n);
    		for (len = 0, i = 1; i <= n; ++i) {
    			scanf("%d", &x);
    			updatelis(x - i, len);
    		}
    		printf("%d \n",n - len);
    	}
    	return 0;
    }
