#include <iostream>
using namespace std;
long long int cnt1[91];
long long int cnt0[91];

int main()
{
//    cout << "Hello World!" << endl;
	cnt1[0] = 1;
	cnt1[1] = 0;
	cnt0[0] = 0;
	cnt0[1] = 1;
	for(int i=2;i<91;i++)
	{
		cnt1[i] = cnt1[i-1] + cnt1[i-2];
		cnt0[i] = cnt0[i-1] + cnt0[i-2];
	}
	int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		cout<<cnt1[n]<<" "<<cnt0[n]<<endl;
	}
    return 0;
}
