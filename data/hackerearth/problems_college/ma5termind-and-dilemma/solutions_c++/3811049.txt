#include<bits/stdc++.h>
using namespace std;
#define pii pair<int,int>
#define pb push_back
#define ft first
#define sd second

int main() {
	int t;
	cin>>t;
	while ( t-- ) {
		int n;
		cin>>n;
		string s;
		cin>>s;
		int l=s.length();
		int o=0,z=0;
		vector<int> ones,zeros;
		for ( int i=0;i<l;i++ ) {
            if ( s[i]=='0' ) {
                z++;
                zeros.pb(i);
            }
            else {
                o++;
                ones.pb(i);
            }
		}
        //111000 form
        int on=0,ze=0;
        int swaps1=0;
        int sz=ones.size()-1;
        string st1=s;
        for ( int i=0;i<l;i++ ) {
			if ( on==o ) {
				break;
			}
            if ( s[i]=='1' ) {
                on++;
            }
            else {
                swap(st1[i],st1[sz]);
                sz--;
                on++;
                swaps1++;
            }
        }
        //0000111
        sz=zeros.size()-1;
        string st2=s;
        int swaps2=0;
        for ( int i=0;i<l;i++ ) {
			if ( ze==z ) {
				break;
			}
            if ( s[i]=='0' ) {
                ze++;
            }
            else {
                swap(st2[i],st2[sz]);
                sz--;
                ze++;
                swaps2++;
            }
        }
        int ans=min(swaps1,swaps2);
        cout<<ans<<"\n";
	}
	return 0;
}
