#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    long t,x,y,i,c,b;
    cin>>t;
    long a[1000];
    a[0]=0,a[1]=1;
    for(i=2;i<1000;i++)
        {
            a[i]=a[i-1]+a[i-2];
        }
    while(t--)
    {
        long xi=0,yi=0,f1=0,f2=0;
        cin>>x>>y;
        for(i=0;i<1000;i++)
        {
            if(a[i]>=x && xi==0)
            {
                xi=1;
                f1=1;
                c=i;
            }
            if(a[i]>y && yi==0)
            {
                yi=1;
                f2=1;
                b=i-1;
            }
            if(f1==1 && f2==1)
                break;
        }
       // cout<<c<<" "<<b;
        if((b-c+1)%2==0)
            cout<<"VALID\n";
        else
            cout<<"INVALID\n";
    }
    return 0;
}
