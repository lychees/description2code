#include <iostream>
using namespace std;
int main()
{
    int t,n,m1,m2,rem;
    cin>>t;
    while(t--)
    {
    	cin>>n>>m1>>m2;
    	    	rem=(n%(m1+m2));
    	    	if(rem<m1)	printf("0\n");
  else if(rem==m1||rem>=m2)	printf("1\n");
  else
  {
  	rem=rem/m1;
  	if(rem&1)	printf("1\n");
  	else	printf("0\n");
  	}
  	}
  	return 0;
}
