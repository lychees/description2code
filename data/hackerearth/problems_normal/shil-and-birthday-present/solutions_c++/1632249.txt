#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
	long int n,i,j;
	scanf("%ld",&n);
	long int a[n+3],sum=0,count=0;
	for(i=0; i<n; i++)
		scanf("%ld",&a[i]);
	sort(a,a+n);
	for(i=0; i<n; i++)
	{
		if(a[i+1]!=a[i])
		{
			sum+=(i-count);
		}
		else
			count++;
	}
	printf("%ld\n",sum);
	return 0;
}
