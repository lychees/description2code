#include <iostream>
using namespace std;


int a[10][10000];

int main()
{
    
    int n,k,temp,i,j,r,yes,mul;
    
    cin>>n;
    cin>>k;
    
    for(i=0;i<n;i++)
    {
    	cin>>temp;
    	mul=0;
    	j=0;
    	do
    	{
    		mul=(mul+temp)%360;
    		a[i][j]=mul;
    		j++;
    		
    		
    		
    	}while(mul!=0);
    	
    	a[i][j]=-1;
    	
    	
    }
    
    
    for(i=0;i<k;i++)
    {
    	cin>>temp;
    	yes=0;
    	for(j=0;j<n;j++)
    	{
    		
    		r=0;
    		while(a[j][r]!=-1)
    		{
    			if(a[j][r]==temp)
    			{ yes=1; break;}
    			
    			r++;
    			
    		}
    		
    		
    		if(yes)
    		break;
    		
    	}
    	
    	if(yes)
    	cout<<"YES"<<endl;
    	else
    	cout<<"NO"<<endl;
    	
    	
    }
    
    
    
    
    
    return 0;
}
