#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll long long int
inline void inp( long long int &n )
{
    n=0; int ch = getchar(); int sign = 1;
    while(ch < '0' || ch > '9') { if(ch == '-') sign=-1; ch = getchar(); }
    while(ch >= '0' && ch <= '9') { n = (n << 3) + (n << 1) + ch - '0', ch = getchar(); }
    n = n * sign;
}
    inline void inp1( int &n )
{
    n=0; int ch = getchar(); int sign = 1;
    while(ch < '0' || ch > '9') { if(ch == '-') sign=-1; ch = getchar(); }
    while(ch >= '0' && ch <= '9') { n = (n << 3) + (n << 1) + ch - '0', ch = getchar(); }
    n = n * sign;
}
int a[1003];
int main()
{
    int t;
    inp1(t);
    while(t--)
    {
        int k,ans=0;
        inp1(k);
        int n;
        inp1(n);
        for(int i=0;i<n;i++)
            {
                inp1(a[i]);
            }
        sort(a,a+n);
        for(int i=n-1;i>=0;i--)
        {
            if(a[i]>0 && a[i]<=k)
            {
                for(int j=i-1;j>=0;j--)
                {
                    if(a[j]>=0 && a[i]+a[j]<=k)
                    {
                        ans+=1;
                        a[i]=-1;
                        a[j]=-1;
                        goto in;
                    }
                }
                ans+=1;
                a[i]=-1;
            }
            in:;
        }
 
        printf("%d\n",ans);
    }
}
