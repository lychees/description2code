using namespace std;
int dp[100005];
char str[100005];
int main(){
    int t;
    scanf("%d",&t);
    while(t--){
        scanf("%s",str);
        int l = strlen(str);
        dp[l]=1;
        for(int i=l-1;i>=0;i--){
            char c = str[i];
            dp[i] = 0;
            if(c == 'R'){
                dp[i] |= dp[i+1];
                if(i+1<l && str[i+1]=='Y') dp[i] |= dp[i+2];
                if(i+2<l && str[i+1]=='Y' && str[i+2]=='Y') dp[i] |= dp[i+3];
            }
        }
        puts(dp[0] ? "YES" : "NO");
    }
    return 0;
}
