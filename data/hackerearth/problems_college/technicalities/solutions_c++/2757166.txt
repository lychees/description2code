#include <bits/stdc++.h>
using namespace std;

typedef long long lint;

const int mod = 1e9+7 , maxn = 1e5+123;

char ar[maxn];

int main(){
	
	int t;
	
	scanf("%d",&t);
	
	int n;
	while( t-- ){
		
		scanf("%d %s",&n,ar);
		
		lint ans = 0 , mul = 1;
		
		reverse(ar,ar+n);
		
		for( int i=0 ; i < n ; i++ ){
			ans = (ans + mul * (ar[i]-'0')) % mod;
			mul = (mul * 10) % mod;
		}
		
		printf("%lld\n",ans);
	}
	
	return 0;
}
