#include <iostream>
#include<string.h>
using namespace std;

int main()
{
	char str[55],str1[55];
	int t,n,k,i;
    cin>>t;
    while(t--)
    {
    	cin>>str;
    	n=strlen(str);
    	k=0;
    	while(str[0]!='\0')
    	{
    		if(n%2==0)
    		{
    			if(str[n/2-1]<str[n/2])
    			{
    				str1[k]=str[n/2-1];
    				for(i=n/2;i<n;i++)
    				str[i-1]=str[i];
    			}
    			else
    			{
    				str1[k]=str[n/2];
    				for(i=n/2+1;i<n;i++)
    				str[i-1]=str[i];
    			}
    		}
    		else
    		{
    			str1[k]=str[n/2];
    			for(i=n/2+1;i<n;i++)
    				str[i-1]=str[i];
    		}
    		k++;
    		str[n-1]='\0';
    		n--;
    	}
    	str1[k]='\0';
    	cout<<str1<<endl;
    	}
    return 0;
}
