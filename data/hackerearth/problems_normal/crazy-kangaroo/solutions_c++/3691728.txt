#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t!=0){
        t--;
        long long a, b, m;
        cin>>a>>b>>m;
        long long start_pos, end_pos;
        start_pos = a%m==0? a: (a - a%m) + m;
        end_pos = b%m==0? b: b - b%m;
        //cout<<start_pos<<" "<<end_pos<<endl;
        long long dist = end_pos - start_pos;
        long long answer = dist/m + 1;
        if(answer<=0){
            cout<<0<<endl;
            continue;
        }
        cout<<answer<<endl;
    }
}
