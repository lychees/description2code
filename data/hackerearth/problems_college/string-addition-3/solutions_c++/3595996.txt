#include <bits/stdc++.h>
using namespace std;

typedef long long LL;

#define all(c)    c.begin(),c.end()
#define sz(c)     (int)c.size()



int val[1000];
char opp[100];

void pre()
{
	for(int i = 0; i <= 9; i++)
	{
		val[('0' + i)] = i;
		opp[i] = (char)('0' + i);
	}
	for(char i = 'a'; i <= 'z'; i++)
	{
		val[i] = val[i-32] = (int)(10 + i - 'a');
		opp[(int)(10 + i - 'a')] = i;
	}
}

string add(string &a, string &b)
{
	reverse(all(a));
	reverse(all(b));
	string ret = "";
	int carry = 0;
	int base = 36;
	int i = 0;
	for(i = 0; i < min(sz(a),sz(b)); i++)
	{
		int v = val[a[i]] + val[b[i]] + carry;
		carry = v / base;
		v %= base;
		ret += opp[v];
	}
	for(; i < sz(a); i++)
	{
		int v = val[a[i]] + carry;
		carry = v/base;
		v %= base;
		ret += opp[v];
	}
	for(; i < sz(b); i++)
	{
		int v = val[b[i]] + carry;
		carry = v / base;
		v %= base;
		ret += opp[v];
	}
	if(carry)
		ret += opp[carry];
	reverse(all(ret));
	return ret;
}

int main()
{
	pre();
	int t;
	cin >> t;
	while(t--)
	{
		int n;
		cin >> n;
		string ans;
		cin >> ans;
		for(int i = 1; i < n; i++)
		{
			string x;
			cin >> x;
			ans = add(ans , x);
		}
		cout << ans << endl;
	}
	return 0;
}
