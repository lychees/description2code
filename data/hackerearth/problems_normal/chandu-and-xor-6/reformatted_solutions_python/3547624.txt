n, m = map(int, raw_input().split())
mat = []
for i in range(n):
    k = list(map(int, raw_input().split()))
    mat.append(k)
row = []
col = []
for i in range(n):
    row.append(sum(mat[i]))
for i in range(m):
    sum1 = 0
    for j in range(n):
        sum1 += mat[j][i]
    col.append(sum1)
ans = 0
for i in row:
    for j in col:
        ans = max(ans, i ^ j)

print ans
