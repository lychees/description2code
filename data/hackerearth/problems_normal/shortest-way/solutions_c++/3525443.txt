#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long int t,a,b;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b;
    	long long int ans = 0;
    	while(a>1 and b>1)
    	{
    		if(a>b)
    		{
    			ans = ans + a/b;
    			a = a%b;
    		}
    		else
    		{
    			ans = ans + b/a;
    			b = b%a;
    		}
    	}
    	if(a==b and b==1)cout<<ans<<endl;
    	else 
    	{
    		ans = ans + abs(a-b);
    		cout<<ans<<endl;
    	}
    }
    return 0;
}
