#include<iostream>
using namespace std;
int gcd(int n1, int n2)
{
    if (n2!=0)
       return gcd(n2, n1%n2);
    else 
       return n1;
}
int main()
{
	int t;
	cin>>t;
	while(t>0)
	{
		int n,m;
		cin>>n>>m;
		int a[n];
		int i;
		for(i=0;i<n;i++)
			cin>>a[i];
		for(i=0;i<n;i++)
		{
			if(gcd(a[i],m)==1)
				cout<<a[i]<<" ";
		}
		cout<<endl;
		t--;
	}
}
