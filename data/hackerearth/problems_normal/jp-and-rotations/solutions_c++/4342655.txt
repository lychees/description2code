#include <iostream>
using namespace std;
int arr[100004],brr[100005];
int main()
{
	int n,m;
	cin>>n>>m;
	for(int i=0;i<n;i++)
		cin>>arr[i];
	for(int i=0;i<n;i++)
		cin>>brr[i];

	int start=arr[0];
	int i=0;
	while(brr[i]!=start)
	{
		i++;
	}
	int x;
	char c;
	int req=i;
//	cout<<req<<endl;
	int left=0;
	bool flag=false;
	int right=n-1;
	int j=0;
	while(j<m)
	{
		cin>>c>>x;
		if(req==left){
			flag=true;
			break;
		}
		if(c=='R')
		{
			left=(left+x)%n;
			right=n-left-1;
		//	cout<<left<<" "<<right<<endl;
		}
		else
		{
		//	cout<<right<<"  --  "<<x<<endl;
			right=(right+x)%n;
			left=n-right-1;

		//	cout<<left<<" "<<right<<endl;
		}
		j++;
	}
	if(flag)
		cout<<j<<endl;
	else
		cout<<"-1"<<endl;
	return 0;
}
