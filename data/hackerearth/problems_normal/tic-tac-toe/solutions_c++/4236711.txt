#include <iostream>
typedef long long ll;
using namespace std;
ll modInverse(ll a, ll m)
{
    ll m0 = m, t, q;
    ll x0 = 0, x1 = 1;

    if (m == 1)
      return 0;

    while (a > 1)
    {
        q = a / m;
        t = m;
        m = a % m, a = t;
        t = x0;
        x0 = x1 - q * x0;
        x1 = t;
    }
    if (x1 < 0)
       x1 += m0;

    return x1;
}
int main()
{
    ll t;
    ll M=1000000007;
    cin>>t;
    while(t--)
    {
        ll k;
        cin>>k;
        ll n = k-1,j,l;
        ll max,min;
        if(k%2==0)
        {
            j = (k-1)/2;
            l = j++;
            min = ((k/2)*(j*(j)%M)%M)%M;
            min = ((min%M) + ((k/2)*(l*(l)%M)%M)%M)%M;
        }
        else
        {
            j = ((k-1)/2)%M;
            min = (k*(j*(j)%M)%M)%M;
        }
        max = (n*((n+1)*((2*n+1)%M)%M)%M);
        max = (max*modInverse(6,M))%M;
        cout<<min%M<<" "<<max<<endl;
    }
}
