#include <iostream>
using namespace std;
static long a[1000001];

void sieveofera()
{
	long i,j;
	
	for(i=1;i<=1000000;i++)a[i]=i;
	
	for(i=2;i<=1000;i++)
	{
		if(a[i]!=0)
		{
			for(j=2*i;j<=1000000;j+=i)a[j]=0;
		}
	}
	return ;
}

int main()
{
	ios_base::sync_with_stdio(false);
    long t,i;
    cin>>t;
    static long val[1000001],l[1000001],r[1000001];
    
    for(i=0;i<t;i++)
    {
    	cin>>l[i]>>r[i];
    }
    val[0] = 0;
    val[1] = 1;
    val[2] = 1;
    val[3] = 1;
    val[4] = 1;
    val[5] = 1;
    sieveofera();
    
    for(i=6;i<=1000000;i++)
    {
		
    	if(a[i])
    	{
    		val[i] = val[i-1];
    	}
    	else
    	{
    		val[i] = val[i-1]+1;
    	}
    }
	
    for(i=0;i<t;i++)
    {
    	cout<<val[r[i]]-val[l[i]-1]<<"\n";
    }
    return 0;
}
