def findMonkeyness(a,n):

	start=0
	end=len(a)-1
	ans=-1

	while start<=end:
		mid=(start+end)/2
		if a[mid]>=n:
			ans=mid
			start=mid+1
		else:
			end=mid-1
	max=ans
	while max<end and a[max]==a[max+1]:
		max+=1
	return max


for tc in xrange(int(raw_input())):
	n=int(raw_input())

	a=map(int,raw_input().split())
	b=map(int,raw_input().split())

	monkeyness=0
	for x in xrange(n):
		m=findMonkeyness(b,a[x])
		if m-x>monkeyness:
			monkeyness=m-x
	print monkeyness
