#include <stdio.h>

using namespace std;
long long a [1000001];
long long int mod=1e9 +7;
void precompute()
{
    a[0]=0;
    a[1]=1;
    a[2]=2;
    int i;
    for(i=3;i<=1000000;i++)
    {
        a[i]=(a[i-1] + ((i-1)*a[i-2])%mod)%mod;
    }
}
int main()
{
    int n,t;
    long long int ans;
    precompute();
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        ans=a[n]%mod;
        printf("%lld\n",ans);
    }
    return 0;
}
