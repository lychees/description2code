int pr[50],ct[50];

	fast();

	ll n,k,total=1,n1,n2;
	sl(n)
	s(t)
	n2 = n1 = n;
	for(ll i=2; i*i<=n ; i++){
		if( (n%i) == 0){
			pr[ind] = i;
			countt = 0;
			while(n % i == 0 ){
				n/=i;
				countt++;
			}
			ct[ind++] = countt;
			total*=(countt+1);
		}
	}

	if( n > 1){
		pr[ind] = n;
		ct[ind++] = 1;
		total*=2;
	}

	//p1d(pr,ind);

    int ctr = 0;
	while(t--){

		s(tp) sl(k)
		n2 = k;
		if(tp == 1){
			f(i,0,ind){
					if(n2%pr[i] ==0){
						ctr = 0;
						while(n2 % pr[i] == 0){
							n2/=pr[i];
							ctr++;
						}
						ctr = min(ct[i],ctr);
						temp*=(ctr+1);
					}
			}
			pl(temp)

						temp*=(ct[i]+1);

							ctr++;
							n2/=pr[i];
						}

			f(i,0,ind){

						temp*=(ct[i]+1);

						ctr = 0;

							ctr++;
							n2/=pr[i];
						}				
							temp*=(ct[i] - ctr + 1);
						
					}
			}

			pl(total - temp)
