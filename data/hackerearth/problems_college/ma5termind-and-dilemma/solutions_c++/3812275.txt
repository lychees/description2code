int main()
{
    ll n,i,j,t,cnt1,cnt2;
    string s,s1;
    char temp;
    cin>>t;
    while(t--)
    {
        cin>>n>>s;
        s1.assign(s);
        cnt1=0;
        cnt2=0;
        i=0;
        j=n-1;
        while(i<j)
        {
            while(s[i]=='0' && i<j)
                i++;
            while(s[j]=='1' && i<j)
                j--;
            if(i<j)
            {
                //cout<<i<<" "<<j<<endl;
                cnt1++;
                temp=s[i];
                s[i]=s[j];
                s[j]=temp;
                i++;
                j--;
            }
        }
        i=0;
        j=n-1;
        while(i<j)
        {
            while(s1[i]=='1' && i<j)
                i++;
            while(s1[j]=='0' && i<j)
                j--;
            if(i<j)
            {
                //cout<<i<<" "<<j<<endl;
                cnt2++;
                temp=s1[i];
                s1[i]=s1[j];
                s1[j]=temp;
                i++;
                j--;
            }
        }
        //cout<<cnt1<<" "<<cnt2<<endl;
        cout<<min(cnt1,cnt2)<<endl;
        s.clear();
        s1.clear();
    }
}
