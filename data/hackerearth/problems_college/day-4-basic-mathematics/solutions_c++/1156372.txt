#include<iostream>
#include<cmath>
using namespace std;

int main(){
int t,i;
long int n,x;

cin>>t;

for(i=0;i<t;i++){
	cin>>n;

	if((float)sqrt(1+8*n)-(int)sqrt(1+8*n)==0){
		x = (-1 + sqrt(1+8*n))/2;
	}
	else{
		x = (-1 + sqrt(1+8*n))/2+1;
	}

	if(x%3==0)
		cout<<"SL"<<endl;
	else if(x%3==1)
		cout<<"LB"<<endl;
	else
		cout<<"BS"<<endl;
}

return 0;
}
