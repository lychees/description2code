#include <iostream>
#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define sc(a) scanf("%d",&a)
#define scl(a) scanf("%lld",&a)

int main()
{
    int t,i,j,k;
    ll n,p;
    sc(t);
    while(t--)
    {
    	scl(n);
    	p=(int)sqrt(n);
    	ll ans=(n-p);
    	printf("%lld\n",ans);
    }
    return 0;
}
