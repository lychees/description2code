#include <iostream>
using namespace std;

int rev(int x)
{
	int t=0;
	int y=x;
	int l=0;
	while(x!=0)
	{
		x=x/10;
		++l;
	}
	int u=1;
	int d=1;
	while(u<l)
	{
		d*=10;
		++u;
	}
	x=0;
	while(y!=0)
	{
		int p=y%10;
		y=y/10;
		x=x+d*p;
		d/=10;
	}
	return x;
}

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int a;
    	int b;
    	int c;
    	scanf("%d%d",&a,&b);
    	a=rev(a);
    	b=rev(b);
    	c=a+b;
    	c=rev(c);
    	printf("%d\n",c);
    }
    return 0;
}
