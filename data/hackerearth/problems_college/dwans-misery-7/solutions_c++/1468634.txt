#include <iostream>
using namespace std;

int main()
{
    unsigned long long a[25000],in,out,out_n,in1,out1,out_n1,max1,max2,ans;
    int test,n;
   scanf("%d",&test);
    while(test--)
    {
    	cin>>n;
   scanf("%lu",&a[0]);
    	in=a[0];
    	out=0;
   scanf("%lu",&a[1]);
    		
    		out_n=in>out?in:out;
    		in=out+a[1];
    		out=out_n;
    			in1=a[1];
    	out1=0;
    		
    	for(int i=2;i<n;++i)
    	
    	{scanf("%lu",&a[i]);
   	if(i<n-1){out_n=in>out?in:out;
    		in=out+a[i];
    		out=out_n;}
    		out_n1=in1>out1?in1:out1;
    		in1=out1+a[i];
    		out1=out_n1;
    		
    		
    	}
   max1= in>out?in:out;
max2= in1>out1?in1:out1;
ans=max1>max2?max1:max2;
printf("%llu\n",ans);
}
    
    return 0;
}
