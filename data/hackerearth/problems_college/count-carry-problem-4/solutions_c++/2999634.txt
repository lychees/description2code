#include <iostream>
using namespace std;
 
int main()
{
    int t,i,e;
    long long int n,m,a,b,c,d;
    cin>>t;
    for(i=1;i<=t;i++)
    {
    	e=0;
    	cin>>n>>m;
    	x:
    	if(n==0||m==0)
    	{
    		goto v;
    	}
    	a=n/10;
    	b+=n%10;
    	c=m/10;
    	d=m%10;
    	if(b+d>=10)
    	{
    		b=0;
    		e++;
    		n=a;
    		m=c;
    		b++;
    		goto x;
    	}
    	if(b+d<10)
    	{
    		b=0;
    		n=a;
    		m=c;
    		goto x;
    	}
    	v:
    	if(e==0)
    	cout<<"No carry operation\n";
    	if(e==1)
    	cout<<e<<" carry operation\n";
    	if(e!=0&&e!=1)
    	cout<<e<<" carry operations\n";
    }
    return 0;
}
