#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main() 
{
	ll n,i,i1;
	char str[1000090];
	int yes[1000090]={0};
	int ct[10]={0};
	scanf("%s",str);
	n=strlen(str);
	/*hackerearth =11
	a-2
	c-1
	e-2
	h-2
	k-1
	r-2
	t-1
	*/
	int ans[]={2,1,2,2,1,2,1};
	for(i=0;i<11;i++)
	{
		if((int)str[i]>=65 && (int)str[i]<=90)
		{
			str[i]=(int)str[i]+ 32;
		}
		if(str[i]=='a')
			ct[0]++;
		if (str[i]=='c')
			ct[1]++;
		if(str[i]=='e')
			ct[2]++;
		if(str[i]=='h')
			ct[3]++;
		if(str[i]=='k')
			ct[4]++;
		if(str[i]=='r')
			ct[5]++;
		if(str[i]=='t' )
			ct[6]++;
	}
	for(i=0;i<7;i++)
	{
		if(ct[i]!=ans[i])
		{
			break;
		}
	}
	if(i==7)
	{
		yes[0]=1;
	}
	//printf("%d ",yes[0]);
	for(i=11;i<n;i++)
	{
		if((int)str[i]>=65 && (int)str[i]<=90)
		{
			str[i]=(int)str[i]+ 32;
		}
		if(str[i-11]=='a')
			ct[0]--;
		if (str[i-11]=='c')
			ct[1]--;
		if(str[i-11]=='e')
			ct[2]--;
		if(str[i-11]=='h')
			ct[3]--;
		if(str[i-11]=='k')
			ct[4]--;
		if(str[i-11]=='r')
			ct[5]--;
		if(str[i-11]=='t' )
			ct[6]--;
			
		if(str[i]=='a')
			ct[0]++;
		if (str[i]=='c')
			ct[1]++;
		if(str[i]=='e')
			ct[2]++;
		if(str[i]=='h')
			ct[3]++;
		if(str[i]=='k')
			ct[4]++;
		if(str[i]=='r')
			ct[5]++;
		if(str[i]=='t' )
			ct[6]++;

		for(i1=0;i1<7;i1++)
		{
			if(ct[i1]!=ans[i1])
			{
				break;
			}
		}
		yes[i-10]=yes[i-11];
		if(i1==7)
		{
			yes[i-10]++;
		}
		//printf("%d ",yes[i-10]);
	}
	ll q;
//	printf("\nThe changed str is: %s\n",str);
//	printf("\nanswers:\n");
	scanf("%d",&q);
	for(i=0;i<q;i++)
	{
		ll low, high;
		scanf("%lld%lld",&low,&high);
		low--;
		high--;
		if(high-low<10)
		{
			printf("0\n");
			continue;
		}
		high-=10;
		
		if(low!=0)
			printf("%d\n",yes[high]-yes[low-1]);
		else
			printf("%d\n",yes[high]);
	}
	return 0;
}
