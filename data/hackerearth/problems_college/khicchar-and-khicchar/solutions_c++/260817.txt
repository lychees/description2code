#include<stdio.h>
#include<stdlib.h>
#include<iostream>
using namespace std;
int compare (const void * a, const void * b)
{
  return ( *(int*)a - *(int*)b );
}
int main()
{
    int t,e,i,num[4][251],l[4];
    char a[251],b[251],c[251],d[251];
    cin>>t;
    while(t--)
    {
        int prime[1000]={0};
        cin>>a>>b>>c>>d>>e;
        //cout<<a<<endl<<b<<endl<<c<<endl<<d<<endl<<e<<endl;
        for(i=0;a[i]!='\0';i++)
            num[0][i]=a[i]-48;
        l[0]=i;
        for(i=0;b[i]!='\0';i++)
            num[1][i]=b[i]-48;
        l[1]=i;
        for(i=0;c[i]!='\0';i++)
            num[2][i]=c[i]-48;
        l[2]=i;
        for(i=0;d[i]!='\0';i++)
            num[3][i]=d[i]-48;
        l[3]=i;
        //cout<<num[0][0]<<num[0][1]<<endl;
        int k=0;
        for(i=1;i*i<=e;i++)
        {
            if(e%i==0)
            {
                prime[k++]=i;
                prime[k++]=e/i;
            }
        }
        qsort (prime, k, sizeof(int), compare);
        /*for(i=0;i<k;i++)
            cout<<prime[i]<<" ";
        cout<<endl;*/
        int max=1,j,y=0;
        long long n=0,q;
        for(i=k-1;i>0;i--)
        {
            int x=prime[i];
            //cout<<x<<endl;
            y=0;
            qw:
            q=n=0;
            //cout<<"# "<<l[y]<<endl;
            for(j=0;j<l[y];j++)
            {
                n=n*10+num[y][j];
                q=q*10+n/x;
                n=n%x;
                //cout<<x<<" "<<n<<endl;
            }
            q=q*10+n/x;
            n=n%x;
            //cout<<x<<" "<<n<<endl;
            if(n==0 && y<3)
            {
                //cout<<"$ "<<x<<endl;
                y++;
                goto qw;
            }
            else if(y==3 && n==0)
            {
                //cout<<"$ "<<x<<endl;
                y=0;
                if(max<x)
                    max=x;
                //cout<<"# "<<max<<endl;
            }
            else if(n!=0)
            {
                y=0;
            }
        }
        cout<<max<<endl;
    }
    return 0;
}
