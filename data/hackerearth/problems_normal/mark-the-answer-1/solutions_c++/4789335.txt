#include <iostream>
using namespace std;

int main()
{
   long int n,x;
   cin>>n>>x;
   long int arr[n];
   for(long int i=0;i<n;i++)
   cin>>arr[i];
   long int count=0;
   bool skipped=false;
   for(long int i=0;i<n;i++)
   {
   		if(arr[i]>x&&!skipped)
   		skipped=true;
   		else if(arr[i]>x&&skipped==true)
   		break;
   		else if(arr[i]<=x)
   		count++;
   }
   cout<<count;
   return 0;
}
