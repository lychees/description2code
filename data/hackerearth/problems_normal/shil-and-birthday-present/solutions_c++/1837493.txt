#include <iostream>
#include <algorithm>

using namespace std;
#define na 1000000

int main()
{
    long long int a[na],i,n,ans,c=0;
    cin>>n;
    for(i=0;i<n;i++)
        cin>>a[i];
    sort(a,a+n);
    for(i=0;i<n-1;i++)
    {
        if(a[i]!=a[i+1])
            c++;
    }
    ans=(c*(c+1))/2;
    cout<<ans;
    return 0;
}
