string s[55];

char func(lli x, int i)
{
    //cout<<x<<" "<<i<<endl;
    if(i==0) return s[i][x];
    //if(x==len[i]-1) return s[i][x];
    if(x>=2*len[i-1]) return s[i][x-2*len[i-1]];
    if(x<len[i-1]) return func(x,i-1);
    return func(2*len[i-1]-x-1,i-1);
}
