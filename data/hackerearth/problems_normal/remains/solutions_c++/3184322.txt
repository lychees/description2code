#include"iostream"
#include"algorithm"
using namespace std;
int main()
{
	long long unsigned int t,x,y,n,sum,a,b,p,flag;
	cin>>t;
	while(t--)
	{
		flag=0;
		cin>>x>>y>>n;
		sum=x+y;
		p=2;
		while(p!=n)
		{
			if(x==y)
			{
				flag=1;
				break;
			}
			a=y;
			if(x>y)
				a=x;
			if(a==x)
				b=a-y;	
			else
				b=a-x;
			sum+=b;
			x=y;
			y=b;
			p++;
		}
		if(flag==1)
		{
			if((n-p)%3==0 || (n-p)%3==1)
				cout<<sum+((n-p)/3)*x*2<<endl;
			else if((n-p)%3==2)
				cout<<sum+((n-p)/3)*x*2+x<<endl;
		}
		else
		{
			cout<<sum<<endl;
		}
	}
	
	
	
}
