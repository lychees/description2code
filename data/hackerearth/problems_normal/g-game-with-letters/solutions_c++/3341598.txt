#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    long long n,t,i,len,a[100],j,max,sum,pos,lenm,it;
    char s[100000],st[100];
    cin>>t;
    while(t>0)
    {
    	max=-1,pos=-1,lenm=-1;
    	for(i=0;i<26;++i)
    	a[i]=0;
    	cin>>n;
    	cin>>s;
    	len=strlen(s);
    	for(i=0;i<len;++i)
    	{
    		++a[s[i]-97];
    	}
    	for(i=0;i<n;++i)
    	{
    		cin>>st;

    		len=strlen(st);
    		sum=0;
    		for(j=0;j<len;++j)
			sum=sum+a[st[j]-97];
    	if(sum==max)
    	{
    		if(len<lenm)
    		{
    			pos=i;
    			lenm=len;
    		}
    	}
    	else if(sum>max)
    	{
    		pos=i;
    		lenm=len;
    		max=sum;
    	}
    	}
    	cout<<pos+1<<endl;
    	--t;
    }
    return 0;
}
