#include <iostream>
using namespace std;

int main()
{
	int n,i,f;
    cin>>n;
    int a[n];
    f=0;
    for(i=0;i<n;i++)
    cin>>a[i];
    for(i=0;i<n-1;i++)
    {
    	if(a[i]<=0 || a[i+1]-a[i]<0)
    	{
    	  f=1;
    	  break;
    	}
    	if(a[i+1]-a[i]>0)
    	{
    		a[i+1]=a[i+1]-a[i];
    		a[i]=0;
    	}
    	if(a[i+1]-a[i]==0)
    	{
    	  a[i+1]=0;
    	  a[i]=0;
    	  i++;
    	}
    	
    }
    if(f==1)
    cout<<"NO";
    else
    {
    	if(a[n-1]==0)
    	cout<<"YES";
    	else
    	cout<<"NO";
    }
    return 0;
}
