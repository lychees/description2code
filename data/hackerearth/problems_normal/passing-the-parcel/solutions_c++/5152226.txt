#include <iostream>
using namespace std;
int next(int *a,int n,int cur){
	for ( int i = cur + 1 ; i < n ; i++ ) {
		if( a[i] != -1 )
			return i;
	}
	for ( int i = 0 ; i < cur ; i++ ) {
		if(a[i] != -1)
			return i;
	}
}
int main()
{
	int n;
	cin>>n;
	string s;
	cin>>s;
	int *a = (int *)malloc(n*sizeof(int));
	int count = n;
	for(int i=0;i<n;i++){
		a[i]=i+1;
	}
	int cur = 0;
	for(int i=0;i<s.size() ;i++){
		if(s[i] == 'a'){
			cur = next(a,n,cur);
		}
		else{
			a[cur] = -1;
			count --;
			cur = next(a,n,cur);
		}
		if(count == 1){
			printf("%d",a[cur]);
			break;
		}
//		printf(" %c %d ",s[i],a[cur]);
		if(i+1 == s.size())
			i=-1;
	}
    return 0;
}
