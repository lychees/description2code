#include<bits/stdc++.h>
using namespace std;
static int prime[100101],p=0;
static char mrk[1000101];
void seive(){
    int d=2;
    prime[p++]=2,prime[p++]=3;
   for(int i=5;i<=1000007;i+=d,d=6-d){
    if(mrk[i])
        continue;
    prime[p++]=i;
    int e=2;
    for(int j=5*i;j<=1000007;j+=e*i,e=6-e){
     mrk[j]=1;
    }
   }
}
static int ar[1000000];
int main(){
   seive();
   int t;
   scanf("%d",&t);
   for(int i=p-1;i>=1;i--){
   	   for(int j=prime[i]-1;j>=prime[i-1];j--)
   	     ar[j]=prime[i];
   }
   ar[1]=1;
   while(t--){
        int n;
    scanf("%d",&n);
    long long int sum=0;
    for(int i=0;i<n;i++){
        int x;
        scanf("%d",&x);
        sum+=ar[x];
    }
    printf("%lld\n",sum);
   }
   return 0;
}
