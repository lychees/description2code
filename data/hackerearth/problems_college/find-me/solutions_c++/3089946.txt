#include <iostream>

using namespace std;

int main()
{
    int t,i;
    cin>>t;
    for(i=0;i<t;i++)
    {
        long long int p,a[10000];
        int n,j,c=0;
        cin>>n>>p;
        for(j=0;j<n;j++)
        {
            cin>>a[j];
            if(a[j]!=p)
                c=c+1;
        }

        if(c==n)
        {
            cout<<"NO 0\n";
        }
        else
        {
            c=0;
            for(j=0;j<n;j++)
            {
                if(a[j]>p)
                    c=c+1;
            }
            cout<<"YES "<<c<<"\n";
        }
        
    }
}
