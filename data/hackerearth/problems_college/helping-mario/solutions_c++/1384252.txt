#include<bits/stdc++.h>
using namespace std;
#define gc getchar_unlocked
void scan(long long &x)
{
    register char c = gc();
    x = 0;
    int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}

long long gcd(long long a, long long b)
{
	while(b) b ^= a ^= b ^= a %= b;
	return a;
}
int main()
{
	long long t;
	scan(t);
	while(t--)
	{
		vector<long long>v;
		map<long long, long long>a, prev;
		long long size, j, l, r, g, ans, n, i, temp;
		ans=0;
		scan(n);
		for(i=0; i<n; i++)
		{
			scan(temp);
			size=v.size();
			for(j=0; j<size; j++)
			{
				g=gcd(v[j], temp);
				if(!a[g])
				v.push_back(g);
				a[g]+=prev[v[j]];	
			}
			if(!a[temp])
				v.push_back(temp);
			a[temp]++;
			prev=a;
		}
		scan(l);
		scan(r);
		for(i=l; i<=r; i++)
			ans+=a[i];
		printf("%lld\n", ans);
	}
	return 0;
}
