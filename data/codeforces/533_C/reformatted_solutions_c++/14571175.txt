#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>
#include <string>
using namespace std;
int xp, yp, xv, yv;
int main() {
  scanf("%d%d%d%d", &xp, &yp, &xv, &yv);
  if (xp + yp <= max(xv, yv) || (xp <= xv && yp <= yv))
    printf("Polycarp\n");
  else
    printf("Vasiliy\n");
  return 0;
}
