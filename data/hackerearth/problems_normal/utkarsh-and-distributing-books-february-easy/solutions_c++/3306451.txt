#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,m,i;
    int ar[10000];
    cin>>t;
    while(t--)
    {

    	cin>>n;
    	for(i=0;i<n;i++)
    	cin>>ar[i];

    	sort(ar,ar+n);
    	int s=0;
    	for(i=0;i<n;i++)
    	{
    	s+=ar[i]-1;
    	}
    	cout<<ar[0]-1<<" "<<s<<"\n";




    }


    return 0;
}
