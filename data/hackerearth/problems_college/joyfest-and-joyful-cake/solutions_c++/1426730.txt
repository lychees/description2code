#include<cstdio>
#include<algorithm>
#include<cstring>
 
 
using namespace std;
 
 
long long bin_search(long long a,long long b,long long n)
{
 
 
while(a<b)
{
    long long x=(a+b)/2;
    long long y=(x*(x+1))/2+1;
 
 
   // printf("%lld %lld %lld a==%lld b==%lld\n",y,n,x,a,b);
    if(y==n||a==b)
    return x;
 
    if(x==a||x==b)
    return b;
 
 
    if(n<y)
    {
        b=x;
    }
    else
    a=x; 
}
return b;  
}
long long t,n,ans;
int main()
{ 
scanf("%lld",&t);
while(t--)
{
scanf("%lld",&n);  
ans=bin_search(1,10000000,n);
if(n==1)
ans=0;
 
printf("%lld \n",ans);
}
 return 0;
}
