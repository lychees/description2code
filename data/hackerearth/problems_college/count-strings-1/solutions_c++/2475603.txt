#include<bits/stdc++.h>
using namespace std;
#define MOD 1000000007

int main()
{
    int test;
    cin>>test;
    while(test--)
    {
        string a,b;
        int sum=0;
        cin>>a>>b;
        int n=a.size();
        int m=b.size();
        string temp;
        long long pro=1;
        int i=0;
        int ha[26]={0},hb[26]={0};
        for(int i=0;i<n;i++)
            ha[a[i]-'a']++;
        for(int i=0;i<m;i++)
            hb[b[i]-'a']++;
        for(int i=0;i<m;i++)
        {
           pro*= ha[b[i]-'a'];
            ha[b[i]-'a']--;
            pro=pro % MOD;
        }

        cout<<pro<<endl;
    }
    return 0;
}
