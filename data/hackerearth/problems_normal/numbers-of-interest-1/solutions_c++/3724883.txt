#include <iostream>
using namespace std;

int main()
{
   long int t,x,y,n,b;
   long long int sum;
   cin>>t;
   while(t>0)
  {
   	cin>>x>>y>>n;
   	sum=y;
   	long long int* arr=new long long int[x];
   	for(int i=0;i<x;i++)
   	{
   		arr[i]=y;
   	}
   	
   	for(int j=x+1,b=0;j<=n;j++,b++)
   	{ sum=0;
   		for(int i=0;i<x;i++)
     	{
   	     sum+=arr[i];
   	    }
   	   if(b==x)
   	   b=0;
   	  arr[b]=sum;
    }
   cout<<sum<<"\n";
   t--;
   delete arr;
  }
  
  return 0;
}
