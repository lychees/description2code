#include <bits/stdc++.h>
using namespace std;
int main()
{
   int t;
	cin>>t;
	while(t--)
	{
		string x,y;
		cin>>x>>y;
		int a[26]={0},b[26]={0},flag=0;
		for(int i=0;i<x.size();i++)
			a[x[i]-'a']=1;
		for(int i=0;i<y.size();i++)
			b[y[i]-'a']=1;
		for(int i=0;i<26;i++)
			if(a[i] && b[i])
				flag=1;
		if(flag)
			cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;
	}
	return 0;
}
