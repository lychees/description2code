#include <bits/stdc++.h>
#define ll long long
#define ull unsigned long long
#define MOD 1000000007
#define INF 999999999
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	int n,m;
	cin >> n >> m;
	int a[1002][1002] = {};
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=m;j++)
		{
			cin >> a[i][j];
		}
	}
	int mat[1002][1002] = {};
	int ans = 1;
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=m;j++)
		{
			mat[i][j] = 1;
			if(a[i][j] == a[i-1][j] && a[i][j] != a[i][j-1])
			{
				mat[i][j] = mat[i-1][j]+1;
			}
			else if(a[i][j] != a[i-1][j] && a[i][j] == a[i][j-1])
			{
				mat[i][j] = mat[i][j-1]+1;
			}
			else if(a[i][j] == a[i-1][j] && a[i][j] == a[i][j-1])
			{
				mat[i][j] = max(mat[i][j-1], mat[i-1][j])+1;
			}
			ans = max(mat[i][j], ans);
		}
	}
	cout << ans << "\n";
	
}
