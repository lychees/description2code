#include<stdio.h>
#include<iostream>
using namespace std;
int main(){
	int t,n,m;
	cin >> t;
	for (int i = 0; i < t;i++){
		cin >> n;cin >> m;
		char **a = new char*[n];
		for (int j = 0; j < n;j++){
			a[j] = new char[m];
		}
		int c = 0; int x,y,z;
		for (int i = 0; i < n; i++)
		{
			for (int j = 0; j < m; j++)
			{
				cin >> a[i][j];
			}
		}
		for (int k = 0; k <= (n - 2);k++){
			x = k + 1; 
			for (int l = 0; l <= (m - 2);l++){
				y = l + 1;
				if ((a[k][l] == '/') && (a[k][y] == '\\') && (a[x][l] == '\\') && (a[x][y] == '/'))
				{
					c++;
				}
			}
			
		}
		cout << c << endl;
	}

}
