#include <iostream>
using namespace std;

int main()
{
	int times, j;
	cin >> times;
	int a[1000001];
	bool b[1000001];
	a[0]=1;
	b[1]=true;
	
	for(long long int i=1;i<1000001;i++)
	{
		if(b[i])
		{
			continue;
		}
		for(long long int j=i*i;j<1000001;j*=i)
		{
			b[j]=true;
		}
	}
	int count=0;
	for(long long int i=1;i<1000001;i++)
	{
		if(b[i])
		{
			a[count++]=i;
		}
	}

	while(times>0)
	{
		int n;
		cin >> n;
		int flag=0;
		for(int i=0;i<count && a[i]<n;i++)
		{
			if(b[n-a[i]])
			{
				cout << "Yes" << endl;
				flag=1;
				break;
			}
		}
		if(flag==0)
			cout << "No" << endl;

		times--;
	}

	return 0;
}
