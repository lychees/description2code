#include<iostream>
using namespace std;
typedef long long ll;
const ll m=1e9+7;
 ll modPow(long long a, long long x) {
    long long res = 1;
    while(x > 0) {
        if(x & 1)res = (res * a) % m;
        a = (a * a) % m;
        x >>= 1;
    }
    return res;
}
int main()
{
   int t;
   cin >> t;
   while(t--)
   {
       long long n;
       cin >> n;
       if(n<3)
        cout << 0 << "\n";
       else
        cout << ((modPow(3,n) -((3*(modPow(2,n)))%m) +m)%m + 3)%m << "\n";
   }
    return 0;
}
