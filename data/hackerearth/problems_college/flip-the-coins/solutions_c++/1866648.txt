#include<iostream>
#include<stdio.h>
#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,q;
	cin>>n>>q;
	int a[n],f[q],c[q],d[q],ct;
	memset(a,0,sizeof a);
	for(int i=0;i<q;i++)
		cin>>f[i]>>c[i]>>d[i];
	for(int i=c[0];i<=d[0];i++)
		a[i]=~a[i];
	for(int i=1;i<q;i++)
	{
		ct=0;
		for(int j=c[i];j<=d[i];j++)
		{
			if(a[j]) ct++;
		}
		cout<<ct<<endl;
	}
	return 0;
}
