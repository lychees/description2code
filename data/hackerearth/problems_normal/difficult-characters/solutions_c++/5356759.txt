#include <iostream>
#include<algorithm>
#include<string.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    char s[1000000];
    while(t--)
    {
    	cin>>s;
    	int count[123]={0},m=0,max=0,i=0;
    	while(s[i]!='\0')
    	{
    		count[s[i]]++;
    		i++;
    	}
      	for(int i=0;i<123;i++)
    	{
    		if(count[i]>max)
    		max=count[i];
    	}
    	while(m<=max)
    	{
    		for(int i=122;i>=97;i--)
    		{
    			if(count[i]==m)
    			{
    				char str=char(i);
    				cout<<str<<" ";
    			}
    		}
    		m++;
    	}
    	cout<<"\n";
    }
    return 0;
}
