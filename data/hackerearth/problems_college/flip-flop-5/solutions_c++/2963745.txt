#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int n,i;
    cin>>n;
   	char str[100005];
   while(n--)
    {
    	cin>>str;
    	int len = strlen(str);
    	int del = 0;
    	for(i=1;i<len;++i)
    	{
    		if(str[i]==str[i-1])del++;
    		
    	}
    	
    	cout<<del<<"\n";
   
    }
    return 0;
}
