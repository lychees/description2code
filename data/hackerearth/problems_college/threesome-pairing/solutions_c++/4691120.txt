#include <iostream>
using namespace std;

int main()
{
   int t,n;
   cin>>t;
   for(;t--;)
   {
   	cin>>n;
   	int i,j,count=0,a[n];
   	for(i=0;i<n;i++)
   	{
   		cin>>a[i];
   	}
   	for(i=0;i<n;i++)
   	{
   		for(j=i+1;j<n;j++)
   		{
   			if((a[i]+a[j])%3==0)
   			count++;
   		}
   	}
   	cout<<(count)<<endl;
   }// cout << "Hello World!" << endl;
   
    return 0;
}
