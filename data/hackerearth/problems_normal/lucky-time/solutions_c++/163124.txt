#include <iostream>
#include <string>
#include <cstring>
#include <map>
#include <vector>
using namespace std;
int A[10];
vector<string>v;
void rec(int in,string s)
{
     if(in==6)
     {
              int h=(s[0]-48)*10+(s[1]-48);
              int m=(s[2]-48)*10+(s[3]-48);
              int ss=(s[4]-48)*10+(s[5]-48);
              if(h<24 && m<60 && ss<60) v.push_back(s);
              return;
     }
     for(int i=0;i<10;i++)
     {
             if(A[i]==0)
             {
                        A[i]=1;
                        char ch=(char)(i+48);
                        rec(in+1,s+ch);
                        A[i]=0;
             }
     }
} 
int main()
{
    rec(0,"");
    //cout<<v[0]<<" "<<v[1]<<"\n";
    int t;
    cin >> t;
    string s;
    while ( t-- ) {
          cin >> s;
          string sp="";
          for(int i=0;i<s.length();i++)
                  if(s[i]!=':') sp+=s[i];
          
          int in=lower_bound(v.begin(),v.end(),sp)-v.begin();
          if(in==v.size()) cout<<"01:23:45\n";
          else
          {
              cout<<v[in][0]<<v[in][1]<<":"<<v[in][2]<<v[in][3]<<":"<<v[in][4]<<v[in][5]<<"\n";
          }
    }     
    return 0;
}
