#include <iostream>
#include <iomanip> 
#include <math.h>
using namespace std;

int main()
{
	int n;
	double cost;
	cin>>n;
	double s[n];
	double c[n];
	for(int i=0;i<n;i++)
	cin>>s[i];
	for(int i=0;i<n;i++)
	cin>>c[i];
	cin>>cost;
	double cal = 0.0;
	int index =0;
	double min = c[0]/s[0];
	for(int i =1;i<n;i++)
	{
		cal = c[i]/s[i];
		if(min>cal)
		{
			min = cal;
			index = i;
		}
	}
	double ans = (s[index]/c[index])*cost;
	ans = floor(ans*1000) / 1000;
	cout<<setprecision(3)<<fixed<<ans;
    return 0;
}
