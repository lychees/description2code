#include <stdio.h>

int main()
{
    long int N,*xi,*yi;
    int i,j,*color;
    scanf("%d",&N);
    xi=new long int[N];
    yi=new long int[N];
    color=new int[100000];
    for(i=0;i<N;i++){
    	scanf("%ld%ld",&xi[i],&yi[i]);
    	color[xi[i]]++;
    }
    for(i=0;i<N;i++){
    	int home=N-1,away=N-1;
			int temp=color[yi[i]];
			home+=temp;
			away-=temp;
 		printf("%d %d\n",home,away);	
    }
}
