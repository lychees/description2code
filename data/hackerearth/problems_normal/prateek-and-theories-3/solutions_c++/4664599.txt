#include <iostream>
#include <map>
using namespace std;
typedef long long int ll;
map <ll,ll> m;
map <ll,ll>:: iterator it;
int main()
{
	ll t;
	scanf("%lld", &t);
	while(t--){
		ll n;
		scanf("%lld", &n);
		ll i,a,b;
		for( i = 0; i < n; i++) {
			scanf("%lld %lld", &a,&b);
			m[a] = m[a]+ 1;
			m[b] = m[b] - 1;
		}
		/*for( it = m.begin(); it != m.end(); ++it) {
			printf("%lld %lld\n", it->first,it->second);
		}*/
		ll c = 0;
		ll ans = 0;
		for( it = m.begin(); it != m.end(); ++it) {
			//c = 0;
			c += (it->second);
			if ( c > ans) {
				ans = c;
			}
		}
		printf("%lld\n", ans);
		m.clear();
	}
    //cout << "Hello World!" << endl;
    return 0;
}
