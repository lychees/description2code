#include <iostream>
#include<vector>
#include<algorithm>

using namespace std;

int main()
{
//    cout << "Hello World!" << endl;
	int t;
	cin>>t;
	while(t--)
	{
		int n,i;
		long long q;
		cin>>n>>q;
		vector<int> a(n);
		for(i=0;i<n;i++)
		{
			cin>>a[i];
		}
		sort(a.begin(),a.end());
		long long ans=0;
		for(i=1;i<n;i++)
		{
			ans=ans+(a[i]-a[i-1])*q;
		}
		cout<<ans<<endl;
	}
  
    return 0;
}
