#include <iostream>
#include <cstdio>
using namespace std;
#define LL long long

int main()
{
    int tst;
    scanf("%d",&tst);
    while(tst--){
    	LL a,b,c;
    	scanf("%lld%lld%lld",&a,&b,&c);
    	a=min(a,min(b,c));
    	if(a%2==0){
    		printf("%lld.0\n",a/2);
    	}else{
    		printf("%lld.5\n",a/2);
    	}
    }
    return 0;
}
