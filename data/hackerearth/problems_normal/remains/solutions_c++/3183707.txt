#include<bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
	  int t;
	  cin>>t;
	  while(t--)
	  {
	  	  ll a,b;
	  	  ll n;
	  	  cin>>a>>b>>n;
	     
	      ll bild=0;
	      ll sum=a+b;
	      ll next=2;
		  while(1)
	      {
	      	     bild =max(a,b)-min(a,b);
	      	     if(bild==0 || next==n)
				   break;
				   next++;
	    //  	    cout<<bild<<endl;
	      	    sum+=bild;
	      	        a=b;
					  b=bild;    		  
			}
		 ll extra=0;
		 extra=((n-next)/3)*(2*a);
		 int rem=((n-next)%3);
		 if(rem==2)
		 extra+=a;
		// cout<<extra<<endl;
		 sum+=extra;
		 cout<<sum<<endl;
	  }
	  return 0;
}
