#include <iostream>
using namespace std;

int main()
{
    long long int t,q,cnt,i,j;
    cin>>t;
    while(t--)
    {
    	cin>>q;
    	cnt=0;
    	for(i=2;i<q;i++)
    	  for(j=i+1;j<=q;j++)
    	  {
    	  	 if((i*j)%(i+j)==0)
    	  	   cnt++;
    	  }
    
    	cout<<cnt<<"\n";
    }
    
    return 0;
}
