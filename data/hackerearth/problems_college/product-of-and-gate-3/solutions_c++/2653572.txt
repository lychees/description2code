#include <bits/stdc++.h>
using namespace std;

int len(long long int n){
	int count = 0;
	while(n!=0){
		n>>=1;
		count++;
	}
	return count;
}

void calc(long long int a, long long int A[]){
	int l = len(a);
	for (int i = 0; i < l; i++)
	{
		int ans = a & 1;
		A[i] = ans;
		a>>=1;
	}
}

long long int countContinous(long long int a, long long int b){
	long long int A[32];
	long long int B[32];
	calc(a,A);
	calc(b,B);
	int index = len(a)-1;
	int count=0;
	while( (A[index] == B[index]) && (index)){
		count++;
		index--;
	}
	int l = len(a);
	a>>=(l - count);
	a<<=(l - count);
	return a;
}
