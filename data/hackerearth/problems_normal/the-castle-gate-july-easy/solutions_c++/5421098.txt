#include <iostream>
#include <string>
using namespace std;

int main()
{
	string tString;
	string nString;
	int count = 0;
	int n;
	getline(cin,tString);
	int t = atoi(tString.c_str());
	
	for(int i = 0; i < t; i++){
		getline(cin,nString);
		n = atoi(nString.c_str());
		//cout << "n is currently: " << n << " and count is: " << count << endl;
		for( int j = 1; j < n; j++){
			for( int k = j+1; k <= n; k++){
				int res = j^k;
				//cout << "j is: " << j << " and k is: " << k << " their xor product is: " << res << endl;
				if( res <= n) {
					count++;
				}
			}
		}
		cout << count << endl;
		count = 0;
	}
	
    
    return 0;
}
