#include <bits/stdc++.h>
using namespace std;

int main() {
	int t;
	cin >> t;
	for (int i =0 ; i < t; ++i) {
		long int n;
		cin >> n;
		int k = int(sqrt(n));
		if (sqrt(n) != k) {
			int ans = 0;
			for (int j = 1; j <= k; ++j) {
				if (n % j == 0) {
					if (j % 2 == 0) {
						ans +=1;
						if (n/j % 2 == 0) ans += 1;
					}
					else {
						if (n/j % 2 == 0) ans += 1;
					}
				}
			}
			cout << ans << "\n";
		}
		else {
			int ans = 0;
			for (int j = 1; j < k; ++j) {
				if (n % j == 0) {
					if (j % 2 == 0) {
						ans +=1;
						if (n/j % 2 == 0) ans += 1;
					}
					else {
						if (n/j % 2 == 0) ans += 1;
					}
				}
			}
			if (k % 2 ==0) cout << ++ans << "\n";
			else cout << ans << "\n";
		}
	}
	return 0;
}
