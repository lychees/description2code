#include <bits/stdc++.h>
using namespace std;
int main()
{
   
    int t;
    string str;
    cin>>t;
    while(t--)
	{
		int u=0,s=0,sum=0;
		cin>>str;
		int n=str.size();
		for(int i=0;i<n;i++)
		{
			if(str[i]=='U')
			u++;
			else
			s++;
		}
		//cout<<n<<endl;
		if(n==1)
		cout<<"0"<<endl;
		else if((n%2==0)&&(u==s))
		{
			int temp=0;
			for(int i=0;i<n;i++)
				{
					if((i%2==0)&&str[i]!='U')
					temp++;
					else if((i%2==1)&&str[i]!='S')
					temp++;
					
				}
				
		
		
				for(int i=0;i<n;i++)
				{
					if((i%2==0)&&str[i]!='S')
					sum++;
					else if((i%2==1)&&str[i]!='U')
					sum++;
					
				}
				cout<<min(temp,sum)/2<<endl;
			
			
		}
		else if((n%2==1)&&(u==s+1||s==u+1))
		{
			if(s>u)
			{
				for(int i=0;i<n;i++)
				{
					if((i%2==0)&&str[i]!='S')
					sum++;
					else if((i%2==1)&&str[i]!='U')
					sum++;
					
				}
				cout<<sum/2<<endl;
			}
			else
			{
				for(int i=0;i<n;i++)
				{
					if((i%2==0)&&str[i]!='U')
					sum++;
					else if((i%2==1)&&str[i]!='S')
					sum++;
					
				}
				cout<<sum/2<<endl;
			}
		}
		else
		cout<<"-1"<<endl;
		
	}
    return 0;
}
