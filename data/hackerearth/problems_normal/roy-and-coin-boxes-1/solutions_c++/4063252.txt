#include <bits/stdc++.h>
using namespace std;
#define ll long long int
ll d[1111111];
ll b[1111111];
int main()
{
	ll n,m,i,c,k,q,a,g;
	cin >> n >>m;
	for(i=0;i<m;i++){
		cin >> a >> g;
		d[a-1]++;
		d[g]--;
	}
	for(i=1;i<n;i++){
		d[i]=d[i]+d[i-1];
	}
	sort(d,d+n);
	k=d[n-1];
	for(i=n-1;i>=0;i--){
		if(k==d[i]) {
			b[k]++;
		}
		else{
			i++;k--;
		}
		if(k==0) break;
	}
	for(i=d[n-1]-1;i>=0;i--){
		b[i]+=b[i+1];
	}
	cin >> q;
	while(q--){
		cin >> k;
		if(k==0) cout << n << endl;
		else
		cout << b[k] << endl;
	}
    return 0;
}
