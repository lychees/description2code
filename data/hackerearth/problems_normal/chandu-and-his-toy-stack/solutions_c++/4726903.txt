#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;
typedef long long int ll;
int main()
{
	ll t;
	cin >> t;
	while(t--) {
		ll n,x,y;
		cin >> n >> x >> y;
		ll a[n],b[n];
		ll i;
		for( i = 0; i < n; i++) {
			cin >> a[i] >> b[i];
		}
		sort(a,a+n);
		sort(b,b+n);
		ll sum = 0;
		for( i = 0; i < n; i++) {
			ll temp;
			temp = a[i] - b[i];
			if ( temp < 0) {
				sum+= abs(temp)*x;
			} else if ( temp  > 0) {
				sum+= temp*y;
			} else {
				continue;
			}
		}
		cout << sum << endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
