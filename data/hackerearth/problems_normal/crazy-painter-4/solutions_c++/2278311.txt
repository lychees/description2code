#include<bits/stdc++.h>

using namespace std;

int main()
{
    long long int a[26];
    long long int i,j,k,l,t,n,temp;
    long long int sum;
    scanf("%lld",&t);
    while(t--)
    {
        sum=0;
        scanf("%lld",&n);
        for(i=0;i<26;i++)
        {
            scanf("%lld",&a[i]);
            sum+=a[i];
        }
        temp=n*12;
        k=temp/26;
        long long int ans=0;
        if(k==0)
        {
            k=temp%26;
            for(i=0;i<k;i++)
                ans+=a[i];
        }
        else if(k==1)
        {
            ans=sum;
            k=temp%26;
            ans+=k;
            for(i=0;i<k;i++)
                ans+=a[i];
        }
        else
        {
            ans+=k*sum;
            ans+=((k-1)*(13*k));
            ans+=((temp%26)*k);
            k=temp%26;
            for(i=0;i<k;i++)
                ans+=a[i];
        }
        printf("%lld\n",ans);
    }
    return 0;
}
