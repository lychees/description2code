#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,m,i,x;
	cin>>n;
	int ctr[505] = {0}, mark[505] = {0};
	int g[100000];
	x = n;
	while(x--)
	{
		cin>>m;
		int v[505] = {0};
		for(i=0;i<m;i++)
			cin>>g[i];
		for(i=0;i<m;i++)
			ctr[g[i]]++;
		for(i=0;i<m;i++)
		{
			if(!v[g[i]])
			{
				mark[g[i]] += 1;		
				v[g[i]] = 1;
			}
		}
	}
	int sum=0, diff=0;
	for(i=0;i<505;i++)
	{
		if(ctr[i]>0)
			sum += i;
		if(mark[i]!=n && ctr[i]>0)
			diff += i;
	}
	cout<<diff-sum;
	return 0;
}
