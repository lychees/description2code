#include <iostream>
using namespace std;

int main()
{
    int n,i,q;
    char h[100005];
    cin>>n>>q;
    for(i=1;i<=n;i++)
    {
      cin>>h[i];
    }
   
    while(q--)
    {
    	int a,b,flag=0;
    	cin>>a>>b;
    	if(h[a]=='1'||h[b]=='1')
    	 	  { cout<<"No\n";
    	 	    continue;
    	 	  }
    	while(a!=b)
    	 {
    	 	if(h[a]=='1'||h[b]=='1')
    	 	  { flag=1;
    	 	  	break;
    	 	  }
    	 	 if(a>b)
    	 	  a=a/2;
    	 	 else 
    	 	   b=b/2;
    	 }
    	// cout<<flag<<"\n";
    	 if(flag)
    	   cout<<"No\n";
    	 else cout<<"Yes\n";
    }
    
    
    return 0;
}
