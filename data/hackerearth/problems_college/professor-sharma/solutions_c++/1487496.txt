#include <iostream>
using namespace std;

int main() {
	int t,cas=1;
	cin>>t;
	while(t--)
	{
		int a[100],b[100],x,y;
		cin>>x>>y;
		int temp=x,al=0,bl=0;
		for(int i=0;i<100;i++)
		a[i]=b[i]=0;
		while(temp>0)
		{
			a[al]=temp%10;
			temp/=10;
			al++;
		}
		temp=y;
		while(temp>0)
		{
			b[bl]=temp%10;
			temp/=10;
			bl++;
		}
		
		
		int f=0;
		int co=1,carry=0;
		while(f==0)
		{
			for(int i=0;i<al;i++)
			{
				a[i]=(a[i]*x)+carry;
				carry=a[i]/10;
				a[i]%=10;
			}
			while(carry>0)
			{
				a[al]=(carry%10);
				carry/=10;
				al++;
			}
			co++;
		
			if(a[al-1]==b[bl-1])
			{
				int i=1;
				
				f=1;
				while(i<=bl)
				{
					if(a[al-i]!=b[bl-i])
					f=0;
					i++;
				}
				
			}
		}
		cout<<"Case "<<cas<<": "<<co<<endl;
		cas++;
		
	}// your code goes here
	return 0;
}
