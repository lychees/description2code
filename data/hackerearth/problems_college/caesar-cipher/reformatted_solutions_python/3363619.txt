# print 'Hello World!'

t = int(input())
while (t > 0):
    t -= 1
    n = int(raw_input())
    s = raw_input()
    x = ''
    for i in range(len(s)):
        if ord(s[i]) >= 97 and ord(s[i]) <= 122:
            x += chr((ord(s[i])+int(n)-97) % 26+97)
        else:
            x += chr((ord(s[i])+int(n)-65) % 26+65)
    print(x)
