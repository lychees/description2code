#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		string str;
		cin>>str;
		string rev=str;
		int flag=0;
		reverse(rev.begin(),rev.end());
		for(int i=1;i<str.length();i++){
			if(abs(str[i]-str[i-1])!=abs(rev[i]-rev[i-1])){
				flag=1;
				break;
			}
		}
		if(flag){
			cout<<"Not Funny"<<endl;
		}
		else{
			cout<<"Funny"<<endl;
		}
	}
}
