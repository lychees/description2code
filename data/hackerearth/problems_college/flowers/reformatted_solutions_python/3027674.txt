n, k = raw_input().split(" ")
n, k = int(n), int(k)

costs = raw_input().split(" ")


def minimumMoney():
    integerCosts = []
    for c in costs:
        integerCosts.append(int(c))
    integerCosts = sorted(integerCosts)

    multiplyFactor = 1
    friendCount = k
    minMoney = 0
    for c in reversed(integerCosts):

        friendCount -= 1
        minMoney += c * multiplyFactor

        if (friendCount == 0):
            multiplyFactor += 1
            friendCount = k

    print minMoney


minimumMoney()
