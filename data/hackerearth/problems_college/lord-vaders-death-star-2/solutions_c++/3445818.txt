#include<iostream>
using namespace std;

int main()
{
    int arr[1000005];

    for(int i=0; i<1000005; i++)
        arr[i] = 0;

    int n;
    cin>>n;
    for(int i=0; i<n; i++)
    {
        int temp;
        cin>>temp;
        arr[temp]++;
    }

    int ans = -1;
    for(int i=1000005; i>=0; i--)
    {
        if(arr[i] == 1)
        {
            ans = i;
            break;
        }
    }

    cout<<ans<<endl;


    return 0;
}
