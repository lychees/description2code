#include <iostream>
#include<cstring>
using namespace std;

int main()
{
  string str;
	int i;
	cin>>str;
	for(i=0;i<str.length();i++)
	  {
	  	cout<<str[i];
	  	if(str.length()-1!=i)
	  	  cout<<"-";
	  }
	  cout<<endl;
	for(i=str.length()-1;i>=0;i--)
	  {
	  	cout<<str[i];
	  	if(i>0)
	  	  cout<<"-";
	  }
	return 0;
}
