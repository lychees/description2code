#include <iostream>
using namespace std;

int main()
{
	int k,m;
	cin>>k;
	cin>>m;
	int a,flag=0,b;
	int arr[105]={0};
	for (int i=0;i<k;i++)
	{
		cin>>a;
		for (int j=0;j<a;j++)
		{
			cin>>b;
			arr[b]++;
		}
	}
	for (int i=1;i<=m;i++)
	{
		if (arr[i]==0)
		{
			flag++;
			break;
		}
	}
	if (flag==0)
	cout<<"YES"<<endl;
	else
	cout<<"NO"<<endl;
	return 0;
}
