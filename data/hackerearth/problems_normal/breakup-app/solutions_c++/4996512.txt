#include <bits/stdc++.h>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   char str[10000];
   long long int t,n,x,d[31]={0},ans=0,i,g;
   cin>>t;
   while(t--){
   	scanf(" %[^\n]s",str);
   	n=strlen(str);
   	if(str[0]=='G')
   		x=1;
   	else
   		x=0;
   	for(i=0;i<n;i++){
   		if(i+1<n&&isdigit(str[i])&&isdigit(str[i+1])){
   			x=str[i]-48;
   			x=x*10+str[i+1]-48;
   			if(x)
   				d[x]+=2;
   			else
   				d[x]++;
   			if(d[ans]<d[x])
   				ans=x;
   		}
   		else if(isdigit(str[i])){
   			x=str[i]-48;
   			if(x)
   				d[x]+=2;
   			else
   				d[x]++;
   			if(d[ans]<d[x])
   				ans=x;
   		}
   	}
   //cout<<str<<endl;
   }
   	if(ans==19||ans==20)
   		cout<<"Date";
   	else
   		cout<<"No Date";
    return 0;
}
