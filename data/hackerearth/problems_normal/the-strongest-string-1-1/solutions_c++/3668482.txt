#include <iostream>
#include <string>
using namespace std;

bool done[26]={false};

int maxcharindex(string a)
{
	int maxcharindex = 0;
    int maxchar = 96;
    for(int i=0;i<a.length();i++)
    {
    	int k=(int)a.at(i);
    	if(maxchar<k && !done[k-97])
    	{
    		maxchar=k;
    		maxcharindex=i;
    	}
    }
    if(maxchar == 96)
    	return -1;
    return maxcharindex;
}

int main()
{
    int n;
    cin>>n;
    string a;
    cin>>a;
	int maxchaindex=maxcharindex(a);
	char p=a.at(maxchaindex);
	done[((int)p)-97]=true;
	cout<<p;
	a=a.substr(maxchaindex+1);
    while(1)
    {
    	maxchaindex=maxcharindex(a);
    	if(maxchaindex==-1)
    		break;
		p=a.at(maxchaindex);
		done[((int)p)-97]=true;
		cout<<p;
		a=a.substr(maxchaindex+1);
    }
    return 0;
}
