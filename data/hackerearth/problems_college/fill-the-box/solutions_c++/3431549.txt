#include <iostream>
using namespace std;

 int main()
{
    long int n,m;
    cin>>n>>m;
    long int box[n];
    for(long int i=0;i<n;i++)
    	box[i]=0;

    for(long int i=0;i<m;i++)
    {
    	long int a,b,k;
    	cin>>a>>b>>k;
    	if(a==b)
    		box[a-1]+=k;
    	if(a!=b)
    	{
    	    	for(long int j=a-1;j<=b-1;j++)
    	    	{
    	    		box[j]+=k;
    	    	}
    	}
    
    }
    
    long int sum=0;
    for(long int i=0;i<n;i++)
    {
    	sum+=box[i];
    }
    cout<<sum/n;
}
