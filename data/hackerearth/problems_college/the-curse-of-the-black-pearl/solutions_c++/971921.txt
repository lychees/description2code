#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<string>
using namespace std;
char str[36000];
int main() {
    char ch;
    int count=0;
    int t=882;
    while(t--) {
        scanf("%s",str);
        int len=strlen(str);
        ch='0';
        int carry=0;
        int ans=0;
        for(int i=len-1;i>=0;i--) {
            if(i==0&&carry==1) {ch='1';
                ans++;
                i--;
                continue;
            }
            else if(i==0) {
                i--;
                continue;
            }
            if(str[i]=='0'&&carry==1) {
                    str[i]='1';
                    carry=0;
                    ans++;
                    i++;
            }
            else if(str[i]=='0'){
                    ans++;
            }
            else if(str[i]=='1'&&carry==1) {
                    ans++;
            }
            else if(str[i]=='1'){
                    carry=1;
                    ans++;
            }
        }
        if(ch=='1') ans++;
        cout<<ans<<endl;
    }
    return 0;
}
