#include <iostream>
#include <cstdio>
#include <cmath>
#include <cstring>
using namespace std;

int main() {
	// your code goes here
	int t;
	cin >> t;
	getchar();
	for(int i = 0; i < t; i++){
		char a[20][20];
		char name[200];
		scanf("%[^\n]", name);
		//cout << name;
		int n = sqrt(strlen(name));
		int c = 0;
		for(int j = 0; j < n; j++){
			for(int k = 0; k < n; k++){
				a[k][j] = name[c++];
			}
		}
		for(int j = 0; j < n; j++){
			for(int k = 0; k < n; k++){
				cout << a[j][k];
			}
		}
		cout << endl;
		getchar();
	}
	return 0;
}
