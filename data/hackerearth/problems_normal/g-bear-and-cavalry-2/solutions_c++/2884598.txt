#include <iostream>
#include <cstdio>
#include<algorithm>
using namespace std;
int w[100],h[100];
int main() {

    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        
        for(int i=0;i<n;i++)
        cin>>w[i];
        
        for(int i=0;i<n;i++)
        cin>>h[i];
        
        int b=w[0];
        
        sort(w+1,w+n);
        sort(h,h+n);
        
        b=b*h[n-1];
        
        int mx=-1;
        for(int i=1;i<n;i++)
        mx=max(mx,w[i]*h[n-1-i]);
        if(mx<b)
         cout<<"YES"<<endl;
         else cout<<"NO"<<endl;
    }

	return 0;
}
