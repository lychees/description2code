#include<bits/stdc++.h>
using namespace std;

#define it(a,b,c) for(int a=b;a<c;a++)
#define mp make_pair

typedef pair<int, int> pii;

int main(){
	string s;
	cin >> s;
	int x=0,y=0;
	set<pii> S;
	S.insert(mp(0,0));
	int ans = 0;
	it(i,0,s.size()){
		if (s[i] == 'L') --y;
		if (s[i] == 'R') ++y;
		if (s[i] == 'U') --x;
		if (s[i] == 'D') ++x;
		if (S.count(mp(x,y))) ++ans;
		S.insert(mp(x,y));
	}
	cout << ans << endl;
    return 0;
}
