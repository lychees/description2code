#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s,s1="";
    char max;
    long long n,i,pos=-1,a[26]={0};
    cin>>n;
    cin>>s;
    while(1){
    	pos++;
    	while(pos<n && a[s[pos]-97]==1){
    		pos++;
    	}
    	if(pos==n){
    		break;
    	}
    	max=s[pos];
    	for(i=pos+1;i<n;i++){
    		if(s[i]>max && a[s[i]-97]==0){
    			pos=i;
    			max=s[i];
    		}
    	}
    	a[max-97]=1;
    	s1.push_back(max);
    }
    cout<<s1;
    return 0;
}
