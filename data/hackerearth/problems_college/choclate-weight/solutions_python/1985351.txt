n=input()
l=map(int,raw_input().split())
q=input()
while q:
	num=input()
	if num in l:
		print "I Like You"
	else:
		flag=0
		for i in range(0,n):
			for j in range(i+1,n):
				if (l[i]+l[j])==num:
					print "I Like You"
					flag=1
					break
			if flag==1:
				break
		if flag==0:
			print "I Hate You"
	q-=1
