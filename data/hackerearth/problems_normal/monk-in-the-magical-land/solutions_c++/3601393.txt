inline bool coprime(int a , int b) {
    return __gcd(a, b) == 1;
}

int main()
{
    int n, m, k;
    int x[20], c[100], z[100];
    int ans, temp;
    cases {
        ans = 0;
        cin >> n >> m >> k;
        rep(i,n) cin >> x[i];
        rep(i,m) cin >> c[i];
        rep(i,m) cin >> z[i];
        
        rep(i,m) {
            temp = 0;
            rep(j,n) if(!coprime(c[i], x[j])) temp += 1 << j;
            c[i] = temp;
        }
        
        rep(mask,(1<<n))
            if(__builtin_popcount(mask) <= k) {
            temp = 0;
            rep(j,m) if(c[j] & mask) temp += z[j];
            if(ans < temp) ans = temp;
        }
        
        cout << ans << endl;
    }
    return 0;
}
