#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
using namespace std;
int main() {
    int t;
    cin>>t;
    while(t--) {
        int moves;
        int P,x,y,u,v;
        cin>>P>>x>>y>>u>>v;
        int diff1=abs(u-x);
        int diff2=abs(v-y);
        if(diff1==0&&diff2==0) moves=0;
        else if(diff1==diff2) moves=1;
        else if((diff1+diff2)%2==0) moves=2;
        else moves=11;
        if(P>moves) cout<<"Counter-Terrorists Win !\012";
        else cout<<"Terrorists Win !\012";
    }
    return 0;
}
