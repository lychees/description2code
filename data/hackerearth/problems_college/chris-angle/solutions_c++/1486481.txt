using namespace std;

int gcd(int a, int b) {
	while (a) {
		int tmp = a;
		a=b%a;
		b=tmp;
	}
	
	return b;
}

int main() {
	int t;
	cin >> t;
	
	while (t--) {
		int n;
		cin >> n;
		
		n%=360;
		if (n==0) {
			puts("0$");
			continue;
		}
		
		int p = gcd(n, 180);
		int q=n/p;
		int r=180/p;
		if (q!=1) {
			cout << q;
		}
		cout << '$';
		
		if (r!=1) {
			cout << '/' << r;
		}
		cout << endl;
	}
	
	
	return 0;
}
