#include <bits/stdc++.h>
using namespace std;

string s, p;

int main()
{
	ios::sync_with_stdio(false);
	int t;
    cin >> t;
    while(t--) {
    	map <char, int> sm, pm;
    	cin >> s >> p;
    	int n = s.length();
    	int m = p.length();
    	for(int i = 0; i < n; i++) {
    		if(i < m)
    			pm[p[i]]++;
    		sm[s[i]]++;
    	}
    	int flag = 0;
    	for(int i = 0; i < m; i++) {
    		if(pm[p[i]] && sm[p[i]]) {
    			flag = 1;
    			break;
    		}
    	}
    	if(flag)
    		cout << "YES" << endl;
    	else
    		cout << "NO" << endl;
    } 
    return 0;
}
