#include <iostream>
using namespace std;

int main()
{
	int t,max,l,c_for,c_back,i,res;
	string s;
	cin>>t;
	while(t--)
	{
		cin>>s;
		l=s.length();
		max=1;
		for(i=1;i<l-1;i++)
		{
			res=1;
			if(s[i-1]==s[i+1])
			{
			   res=3;
			   c_back=i-2;
			   c_for=i+2;
			   while(c_back>=0 && c_for<l)
			   {
				   if(s[c_back]==s[c_for])
				   {
				      res+=2;
				      c_back--;
				      c_for++;
				   }
				   else
				   {
				      break;
				   }
			   }
			}
			max=(res>max?res:max);
		}
		cout<<max<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
