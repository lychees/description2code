#include <iostream>
#include <string>
#include <unordered_map>
using namespace std;

int main()
{
    int t;
    string s;
    cin >> t;
    
    while(t--){
    	cin >> s;
    	int insertions = 0;
    	unordered_map<char,int> palcount;
    	for(int i = 0; i < s.length();i++){
    		palcount[s[i]]++;
    	}
    	for(auto it=palcount.begin();it!=palcount.end();it++){
    		if(it->second % 2){ // oddcount
    			insertions++;
    		}
    	}
    	if(insertions)
    		cout << insertions-1 << endl;
    	else
    		cout << 0 << endl;
    }
    return 0;
}
