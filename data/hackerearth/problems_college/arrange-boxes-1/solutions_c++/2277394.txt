#include<bits/stdc++.h>
#define pc putchar//_unlocked
#define gc getchar//_unlocked
using namespace std;
inline int readInt()
{
    int cc = gc();
    for (;cc < '0' || cc > '9';)
            cc = gc();
    int ret = 0;
    for (;cc >= '0' && cc <= '9';)
    {
            ret = ret * 10 + cc - '0';
            cc = gc();
    }
    return ret;
}
inline string readString()
{
    char cc = gc();
    for (;cc==' '||cc=='\n';)
        cc = gc();
    string ret;
    for (;(cc !=' ')&&(cc!='\n');)
    {
            ret=ret+cc;
            cc =gc();
    }
    return ret;
}
    inline void writeInt (int n)
    {
        int N = n, rev, count = 0;
        rev = N;
        if (N == 0) { pc('0'); pc('\n'); return ;}
        while ((rev % 10) == 0) { count++; rev /= 10;} //obtain the count of the number of 0s
        rev = 0;
        while (N != 0) { rev = (rev<<3) + (rev<<1) + N % 10; N /= 10;}  //store reverse of N in rev
        while (rev != 0) { pc(rev % 10 + '0'); rev /= 10;}
        while (count--) pc('0');
        pc('\n');
    }
inline int writestring(string a)
