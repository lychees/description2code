#include <stdio.h>
#include <list>
#include <numeric>

using namespace std;

int main()
{
    list<int> n;
    int t,m,o,temp;
    scanf("%d",&t);
    for(int i=0;i<t;i++)
    {
        scanf("%d",&m);
        n.push_back(m);
    }
    n.sort();
    list<int> :: iterator w;
    for(w = n.begin();w!=n.end();w++)
        printf("%d\n",*w);

        return 0;
}
