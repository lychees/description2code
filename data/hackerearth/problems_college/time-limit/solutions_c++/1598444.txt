#include<stdio.h>
#include<iostream>
#include<algorithm>
#include <string.h>
#include <math.h>


using namespace std;

int main()
{
	int n,m,r[10001],w[10001],i,j,k;
	cin>>n>>m;
	for(i = 0 ;i < n ; i++) {
		cin>>r[i];
	}
	for(i = 0 ;i < m ; i++) {
		cin>>w[i];
	}
	sort(r,r+n);
	sort(w,w+m);
	if(w[0] <= r[n-1]) {
		cout<<"-1";
	}
	else {
		k = 2*r[0];
		if(k < w[0]) {
			if(k <= r[n-1]) 
			cout<<r[n-1];
			else 
			cout<<k;
		}
		else 
		cout<<"-1";
	}
	


return 0;
}
