#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;

const int N = (int) 2e5 + 100;
int a[N];

int main()
{
	
	int n, m;
	scanf("%d%d", &n, &m);
	for (int i = 0; i < 2 * n; i++)
		scanf("%d", &a[i] );
	sort(a, a + 2 * n);
	int cnt = 0;
	while (cnt < n && a[cnt] <= m)
	{
		m -= a[cnt];
		cnt++;
	}
	printf("%d\n", cnt);
	
    return 0;
}
