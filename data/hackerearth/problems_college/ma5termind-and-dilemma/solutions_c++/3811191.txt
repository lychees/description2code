#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;

int main(){
    string a;
    int t,i,j,x,y,m;
    cin>>t;
    while(t--){
    	cin>>m;
        cin>>a;
        
        x=0,y=0;
        i=0,j=m-1;
        while(i<j){
            while(a[i]=='0')
                i++;
            while(a[j]=='1')
                j--;
            if(i<j)
                x++,i++,j--;
        }
        i=0,j=m-1;
        while(i<j){
            while(a[i]=='1')
                i++;
            while(a[j]=='0')
                j--;
            if(i<j)
                y++,i++,j--;
        }
        cout<<min(x,y)<<endl;
    }
    return 0;
}
