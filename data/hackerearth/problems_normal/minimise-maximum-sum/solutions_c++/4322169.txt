#include <bits/stdc++.h>
using namespace std;
int a[1000000];
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n;
    	scanf("%d",&n);
    	for(int i=0;i<2*n;i++)
    		scanf("%d",&a[i]);
    	sort(a+0,a+2*n);	
    	int j=0,k=2*n-1,mini=INT_MIN;
    	//cout<<k;
    	while(j<k){
    		mini=max(mini,(a[j]+a[k]));
    		j++;
    		k--;
    	}
    	printf("%d\n",mini);
    }
    
    return 0;
}
