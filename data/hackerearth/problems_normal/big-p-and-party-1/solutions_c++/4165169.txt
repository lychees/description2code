#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int st[1005],d[1005];
vector <int> v[1005];
queue <int> q;
void bfs(int s,int n)
{
	int i,t;
	for(i=0;i<=n;i++)
	{
		st[i]=0;
		d[i]=1005;
	}
	q.push(s);
	st[s]=1;
	d[s]=0;
	while(!q.empty())
	{
		t=q.front();
		q.pop();
		for(i=0;i<v[t].size();i++)
		{
			if(st[v[t][i]]==0)
			{
				st[v[t][i]]=1;
				d[v[t][i]]=d[t]+1;
				q.push(v[t][i]);
			}
		}
		st[t]=2;
	}
}
int main()
{
    int a,b,i,x,y;
    cin>>a>>b;
    for(i=1;i<=b;i++)
    {
    	cin>>x>>y;
    	v[x].push_back(y);
    	v[y].push_back(x);
    }
    bfs(0,a-1);
    for(i=1;i<=a-1;i++)
    {
    	if(d[i]!=1005)
    	{
    		cout<<d[i]<<endl;
    	}
    	else
    	{
    		cout<<"-1"<<endl;
    	}
    }
    return 0;
}
