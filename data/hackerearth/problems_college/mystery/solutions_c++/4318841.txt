#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define MOD 1000000007
#define MAX 100005
ll modexp(ll a, ll b)
{
	ll res = 1;
	while(b)
	{
		res=(res*a)%MOD;
		a=(a*a)%MOD;
		b/=2;
	}
	return res;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		ll a,b,n;
		cin>>a>>b>>n;
		if(n==0) cout<<a<<endl;
		else if(n==1) cout<<b<<endl;
		else
		{
			int i;
			ll sum;
			for(i=2;i<=n;++i)
			{
				sum = a+b;
				a = b;
				b = sum;
			}
			cout<<b<<endl;
		}
		
	}
    return 0;
}
