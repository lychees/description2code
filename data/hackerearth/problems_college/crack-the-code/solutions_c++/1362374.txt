#include<bits/stdc++.h>
using namespace std;
long int f[2000002];
int main(){
	long int t,x,y,i,a,b,temp=0;
	a=1,b=2;
	f[0]=1;
	f[1]=1;
	f[2]=2;
	i=3;
	while(temp<=1000000){
		temp=a+b;
		for(i=b+1;i<temp;i++){
			f[i]=f[b];
		}
		f[temp]=f[b]+1;
		a=b;
		b=temp;
	}
	cin>>t;
	while(t--){
		cin>>x>>y;
		temp=f[y]-f[x-1];
		if(temp&1) cout<<"INVALID"<<endl;
		else cout<<"VALID"<<endl;
	}
	return 0;
}
