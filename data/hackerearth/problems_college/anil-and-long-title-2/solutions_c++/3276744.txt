#include<bits/stdc++.h>
using namespace std;
int main()
{
        string s;
        cin>>s;
       //32
       for(int i=0;i<s.length();i++)
       {
            if(i==0&&s[i]>='a')
            {
                  int k=s[0];
                   k-=32;
                   char c=char(k);
                   cout<<c;
            }
            else
            {
                  if(s[i-1]=='_'&&s[i]>='a')
                  {
                        int k=s[i];
                        k-=32;
                        char c=char(k);
                        cout<<c;
                  }

            else{

                cout<<s[i];
            }
            }
       }
       return 0;




}
