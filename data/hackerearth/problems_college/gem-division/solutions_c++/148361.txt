#include<stdio.h>
#include<string.h>
#include<iostream>
#include<algorithm>
using namespace std;
#define inf 99999999
int dp[55][5005],arr[55];

int rec(int i,int diff){
	//printf("%d %d %d\n",i,diff,arr[i]);
	if(arr[i] != -1 && i == 0)
		return abs(diff-arr[i]);
	if(i<=0)
	{
	
		return diff;
	}
	if(arr[i]==-1)
		return rec(i-1,diff);
	if(dp[i][diff]!=-1)
	{
		
		return dp[i][diff];
	}
	
	int a;
	if(diff>arr[i])
		a=rec(i-1,diff-arr[i]);
	else
		a=rec(i-1,arr[i]-diff);
	int b=rec(i-1,diff+arr[i]);
	//cout<<a<<" "<<b<<endl;
	dp[i][diff]=(a<b)?a:b;
	return dp[i][diff];
}

int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		int n;
		scanf("%d",&n);
		int i,tmp,j,mini=100000000,ans;
		for(i=0;i<n;i++)
			scanf("%d",&arr[i]);
		sort(arr,arr+n);
		for(i=0;i<n;i++){
			memset(dp,-1,sizeof(dp));
			tmp=arr[i];
			arr[i]=-1;
			ans=rec(n-1,0);
			//cout<<ans<<endl;
			if(mini>ans){
				mini=ans;
				j=i;
			}
			arr[i]=tmp;
		}
		printf("%d %d\n",mini,arr[j]);
	}
	return 0;
}
