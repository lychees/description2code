int f(int n,int k){
	if (n%2==0){
		if (k<=n/2) return 2*k-1;
		else return 2*f(n/2,k-n/2);
	}
	else{
		int temp = n/2 + 1;
		if (k<=temp) return 2*k-1;
		// else return (2*(f(n-temp , k-temp)-1) + 2)%n + 1; 
		else{
			int p = 2*f(n-temp,k-temp)+2;
			if (p<=n) return p;
			else return (p-1)%n+2;
		}
	}
}
