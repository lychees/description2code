#include<iostream>
using namespace std;
int main()
{
    int N,i;
    int sum=0;

    cin>>N;
     while(N--)
     { int temp=0;
         string s;
         cin>>s;
         for(i=0;i<s.length();i++)
         { temp+=s[i]-'0';

         }
         sum+=((temp%9)==0?9:(temp%9));

     }
     cout<<((sum%9)==0?9:(sum%9))<<endl;
}
