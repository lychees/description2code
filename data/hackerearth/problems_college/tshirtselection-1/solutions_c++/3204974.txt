#include<bits/stdc++.h>
#define ll long long 
using namespace std;
//vector <ll> v;
//vector <ll> :: iterator it;
set <ll> s;
set <ll> :: iterator it;
ll a[1000000];

int main()
{
ll sz,ans,l,r,lans,rans,n,i,t;

cin>>t;

while(t--)
{
    map <ll,bool> done;
    map <ll,ll> pos;
    ans=0;
    lans=1;
    
    cin>>n;
    rans=n;
    for(i=1;i<=n;i++)
        cin>>a[i];
    
    l=1,r=1;
    while(l<=n&&r<=n)
    {
      if(!done[a[r]])
      {
          done[a[r]]=true;
          pos[a[r]]=r;
          r++;
          continue;
    }
        sz=r-l;
        if(sz>ans)
        {
            ans=sz;
            lans=l;
            rans=r-1;
        }
            l=pos[a[r]]+1;
            r=l;
            done.clear();
    }
    sz=r-l;
    if(sz>ans)
    {
        ans=sz;
        lans=l;
        rans=r-1;
    }
    cout<<lans<<" "<<rans<<"\n";
    done.clear();
}
return 0;
}
