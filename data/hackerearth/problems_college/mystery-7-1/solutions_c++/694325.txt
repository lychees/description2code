#include <iostream>

int main()
{
	int t;
	std::cin >> t;
	std::string s, r;
	
	while (t--)
	{
		std::cin >> s;
		
		for (int i = 0; i < s.size(); ++i)
		{
			int x = (s.at(i) - 'a') / 3 + 2;
			x = x > 9 ? 9 : x; 
			x = (s.at(i) == 's') ? 7 : x;
			x = (s.at(i) >= 't' && s.at(i) <= 'v') ? 8 : x;
			x = (s.at(i) >= 'w' && s.at(i) <= 'z') ? 9 : x;
			
			std::cout << x;  
		}
		
		std::cout << std::endl;
	}
	
	return 0;
}
