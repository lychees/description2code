#include <string>
#include <algorithm>
using namespace std;
int main(){
	int t;
	string s1,s2;
	cin>>t;
	while(t--){
		cin>>s1;
		cin>>s2;
		while(s1.back()=='0'&& s1.length()>0)
			s1.pop_back();
		while(s2.back()=='0' && s2.length()>0)
			s2.pop_back();
		reverse(s1.begin(),s1.end());
		reverse(s2.begin(),s2.end());
		int n1 = stoi(s1);
		int n2 = stoi(s2);
		int n3 = n1+n2;
		string s3 = to_string(n3);
		while(s3.back()=='0'&& s3.length()>0)
			s3.pop_back();
		reverse(s3.begin(),s3.end());
		cout<<s3<<endl;
	}
}
