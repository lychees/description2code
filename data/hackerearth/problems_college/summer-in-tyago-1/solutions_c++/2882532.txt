#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n,flag=0;
    scanf("%d",&n);
    long long int arr[200005],t;
    for(int i=0;i<n;i++)
    {
    	scanf("%lld",&arr[i]);
    } 
    for(int i=0;i<n-1;i++)
    {
    	if(arr[i]>arr[i+1])
    	{
    		if(arr[i]!=0)
    		{
    			arr[i]--;
    			arr[i+1]++;
    			t=arr[i];
    			arr[i]=arr[i+1];
    			arr[i+1]=t;
    		}
    		else
    		{
    			flag=1;
    			break;
    		}
    	}
    }
    for(int i=0;i<n-1;i++)
    {
    	if(arr[i]<=arr[i+1])
    	{
    		continue;
    	}
    	else
    	{
    		flag=1;
    		break;
    	}
    }
    if(flag==0)
    {
    	for(int i=0;i<n;i++)
    	{
    		cout<<arr[i]<<" ";
    	}
    	cout<<endl;
    }
    else
    {
    	cout<<":("<<endl;
    }    
return 0;
}
