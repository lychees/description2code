#!/usr/bin/env python
# -*- coding: utf-8 -*-

def power(n, e):
  result = 1
  for i in xrange(0, e):
    result = result * n
  return result

N = int(raw_input())
two_power_N = power(2, N)
t = []
for i in xrange(1, two_power_N):
    t.append([int(x) for x in raw_input().split()])

participants = [i for i in xrange(0, two_power_N)]

while len(participants) > 1:
    new_participants = []
    for i in xrange(1, len(participants), 2):
        if t[participants[i] - 1][participants[i - 1]]:
            new_participants.append(participants[i])
        else:
            new_participants.append(participants[i - 1])
    participants = new_participants

print(participants[0] + 1)
