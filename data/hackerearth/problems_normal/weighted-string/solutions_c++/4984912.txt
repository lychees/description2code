#include<iostream>
#include<string.h>
using namespace std;
main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int k;
	cin>>k;
	string s;
	cin>>s;
	int n=s.length();
	int i=0;
	int curr=0,count=0,start=0;
	while(i<n)
	{
		
		//curr+=s[i]-'a'+1;

		while(start<i && curr>k)
		{
			curr-=s[start]-'a'+1;
			start++;
		}
		if(curr==k)
		count++;
		curr+=s[i]-'a'+1;
		i++;
	}
	if(curr>=k)
	{
		while(start<n && curr>k)
		{
			curr-=s[start]-'a'+1;
			start++;
		}
		if(curr==k)
		count++;
	}
	cout<<count<<endl;
	}
}
