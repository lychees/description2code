from heapq import heappush, heappop

numOfvalues = int(raw_input())
arry = []


def insert(numStr):
    num = int(numStr)
    heappush(arry, num)


map(insert, raw_input().split(" "))

prevcommand = heappop(arry)
print prevcommand,
for x in range(1, numOfvalues):
    minvalue = heappop(arry)
    diff = minvalue - (prevcommand + x)
    prevcommand = prevcommand + diff
    print prevcommand,
