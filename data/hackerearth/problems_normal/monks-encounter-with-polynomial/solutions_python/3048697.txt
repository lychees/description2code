from math import *
def main():
	t=int(raw_input())
	for i in range(t):
		a, b, c, k = map(int,raw_input().split())
		if(c>=k):
			print(0)
			continue
		d = (b*b) - 4*a*(c-k)
		if(d<=0):
			print(0)
			continue
		d = d**0.5
		d = ceil(d)
		d = int(d)
		x = d-b
		if(x%(2*a)==0):
			x=x//(2*a)
		else:
			x=x//(2*a)
			x=x+1
		print(x)
main()
