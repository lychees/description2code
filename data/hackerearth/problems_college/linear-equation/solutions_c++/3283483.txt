#include <bits/stdc++.h>
using namespace std;

typedef long long int ll;
int main()
{
	ll t, n, x;
	vector<ll> as;
	ll sum = 0;
	scanf("%lld",&t);
	while(t--) {
		as.clear();
		sum = 0;
		scanf("%lld",&n);
		for(int i = 0; i < n;i++) {
			cin >> x;
			as.push_back(x);
			sum += x;
		}
		sum /= (n-1);
		for(int i = 0;i < n; i++) {
			cout << sum - as[i] << " ";
		}
		cout << endl;
	}
    return 0;
}
