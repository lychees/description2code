#include <iostream>
using namespace std;


bool check(string str)
{
	int i, j;
	int len = str.size();
	j = len - 1;
	
	for(i = 0; i < len / 2; ++i) {
		if(str[i] != str[j])
			return 0;
		j--;
	}
	return 1;
}

int main()
{
    int t;
    cin >> t;
    string s1, s2, s3, s4;
    while(t--) {
    	cin >> s1 >> s2;
    	
    	s3 = s1 + s2;
    	
    	s4 = s2 + s1;
    	
    	if(check(s3) == 1 || check(s4) == 1)
    		cout <<"YES\n";
    	else 
    		cout <<"NO\n";
    	
    }
    return 0;
}
