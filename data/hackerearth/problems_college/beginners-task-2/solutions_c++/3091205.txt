#include <iostream>
using namespace std;

int main()
{
   int t;
   cin>>t;
   
   while(t--)
   {
   	   int n,mean;   cin>>n;
   	   long long int a[n],s=0;
   	   
   	   for(int i=0;i<n;i++)
   	       cin>>a[i];
   	       
   	     for(int i=0;i<n;i++)
   	      	 s+=a[i];
   	
   	   mean=s/n;
   	   cout<<"\n"<<mean;
   	
   }
   
   return 0;
}
