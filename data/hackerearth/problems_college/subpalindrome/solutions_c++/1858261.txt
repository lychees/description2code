#define MAX 500

using namespace std;

typedef pair<int,int>pii;
typedef long long LL;
 
int main()
{
	int t;
	scan(t);
	int a[MAX][MAX];
	LL dp[MAX][MAX];
	while(t--)
	{
		string s;
		cin>>s;
		int q;
		int l=s.length();
		for(int i=0;i<l;i++)
		{
			dp[i][i]=1;
			if(s[i]==s[i+1])
			{
				dp[i][i+1]=3;
			}
			else
			{
				dp[i][i+1]=2;
			}
		}
		for(int i=2;i<l;i++)
		{
			for(int j=0;j+i<l;j++)
			{
				int k=i+j;
				dp[j][k]=dp[j+1][k]+dp[j][k-1];
				if(s[j]==s[k])
				{
					dp[j][k]++;
				}
				else
				{
					dp[j][k]-=dp[j+1][k-1];
				}
			}
		}
		cout<<dp[0][l-1]<<endl;
	}
}
