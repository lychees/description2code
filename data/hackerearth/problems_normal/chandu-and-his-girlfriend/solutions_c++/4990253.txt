#include <iostream>
using namespace std;

int poration(int data[],int start,int end)
{
	int pivot=data[start];
	int pivotPos=start+1;
	for(int i=start+1;i<=end;i++)
	{
		if(data[i]>pivot)
		{
			int temp=data[i]^data[pivotPos];
			data[i]=temp^data[i];
			data[pivotPos]=temp^data[pivotPos];
			pivotPos++;
		}
	}
	
	int temp=data[start]^data[pivotPos-1];
	data[start]=temp^data[start];
	data[pivotPos-1]=temp^data[pivotPos-1];
	return pivotPos-1;
}

void quickSort(int data[],int start,int end)
{
	if(start<end)
	{
		int pivotPos=poration(data,start,end);
		quickSort(data,start,pivotPos-1);
		quickSort(data,pivotPos+1,end);
	}
}

int main()
{
    int testCase=0;
    cin>>testCase;
    for(int i=0;i<testCase;i++)
    {
    	int arrayNum=0;
    	cin>>arrayNum;
    	int array[arrayNum];
    	for(int i=0;i<arrayNum;i++)
    	{
    		cin>>array[i];
    	}
    	quickSort(array,0,arrayNum-1);
    
    	for(int k=0;k<arrayNum;k++)
    	{
    		cout<<array[k]<<" ";
    	}
    	cout<<endl;
    }
    return 0;
}
