#include <iostream>
using namespace std;
int A[111111] ;
int main()
{
    //cout << "Hello World!" << endl;
    int n , q , l , r ;
    cin >> n >> q;
    for(int i=1;i<=n;i++) cin >> A[i] ;
    while(q --){
    	cin >> l >> r ;
    	l ++ ;
    	r ++ ;
    	int ans = 4 ;
    	for(int i=l;i<=r;i++) ans = min(ans,A[i]) ;
    	cout << ans << endl ;
    }
    return 0;
}
