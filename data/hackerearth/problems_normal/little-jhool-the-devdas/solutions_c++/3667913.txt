#include<bits/stdc++.h>
using namespace std;
int a[10010];

int main()
{
    int i,t,n;
    long long sum=0LL;
    cin>>n;
    for(i=1;i<=n;i++)
    {
    	cin>>a[i];
    	sum+=a[i];
    }
    sort(a,a+n);
    
    if(sum%2==0)
    {
    	for(i=1;i<=n;i++)
    		if(a[i]%2==1)
    		{
    			sum-=a[i];
    			break;
    		}
    }
    if(sum%2==0)
    cout<<":(\n";
    else
    cout<<sum<<endl;
    return 0;
}
