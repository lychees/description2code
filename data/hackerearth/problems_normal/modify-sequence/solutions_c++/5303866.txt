#include <bits/stdc++.h>

#define REP(i,n) for(int i=0;i<(int)n;i++)
#define FOR(i,a,b) for(int i=(int)a;i<=(int)b;i++)
#define pb push_back
#define mp make_pair

using namespace std;

typedef long long int ll;

int main()
{
	int n;
	scanf("%d",&n);
	int arr[10005];
	REP(i,n){
		scanf("%d",&arr[i]);
		//printf("arr[%d]: %d\n",i,arr[i]);
	}
	
	bool bisa = true;
	REP(i,n-1){
		if(arr[i]>arr[i+1]){
			bisa = false;
			break;
		}else{
			arr[i+1]-=arr[i];
			arr[i]=0;
		}
	}
	if(arr[n-1]!=0) bisa = false;
	
	if(bisa) printf("YES\n");
	else printf("NO\n");
    return 0;
}
