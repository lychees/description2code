test_count = int(raw_input())
for line in range(test_count):
    line_input = raw_input()
    line_items = line_input.split(' ')
    start_index = int(line_items[0])
    num_swaps = int(line_items[1])

    if num_swaps is 0:
        print(start_index)
    else:
        if start_index is 1:
            if num_swaps % 2 is 0:
                print(1)
            else:
                print(0)
        else:
            if num_swaps % 2 is 0:
                print(0)
            else:
                print(1)
