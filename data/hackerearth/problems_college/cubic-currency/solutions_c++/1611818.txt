#include<iostream>
#include<cstring>
using namespace std;
int main()
{
      int arr[22],i;
      for( i =  1 ; i <= 21 ; ++i )
        arr[i] = i*i*i;
         int t;
         cin >> t;
         while(t--)
         {
               int  n;
               cin >> n;
                 long long  table[n+1];
                 memset(table,0,sizeof table);
                  table[0] = 1;
                  for(int i=1; i<=21; i++)
                 for(int j=arr[i]; j<=n; j++)
                      table[j] += table[j-arr[i]];
                      cout << table[n] << endl;
         }
         return 0;
}
