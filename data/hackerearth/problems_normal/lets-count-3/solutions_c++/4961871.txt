#include <bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
   	int testCases;
   	cin>>testCases;
   	while(testCases--) {
   		ll a[3], count = 0;
   		cin>>a[0]>>a[1]>>a[2];
   		sort(a,a+3);
   		for(ll i = 1; i*i <= a[0]; i++) {
   			if(a[0]%i == 0) {
   				if(a[1]%i == 0 && a[2]%i == 0)
   					count++;
   				if(a[0]/i != i) {
   					ll divisor = a[0]/i;
   					if(a[1]%divisor == 0 && a[2]%divisor == 0)
   						count++;
   				}
   			}
   		}
   		cout<<count<<endl;
   	}
    return 0;
}
