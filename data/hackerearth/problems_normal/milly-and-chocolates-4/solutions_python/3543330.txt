from __future__ import division

M=int(raw_input())
mlist=[[] for i in range(0,M)]
N=[0 for i in range(0,M)]

for i in range(0,M):
	N[i]=int(raw_input())
	line=raw_input().split(" ")
	for n in line:
		mlist[i].append(int(n))

for i in range(0,M):
	sum=0
	for j in range(0,N[i]):
		sum=sum+mlist[i][j]
	print("{0:.8f}".format(round(sum/float(N[i]),8)))
