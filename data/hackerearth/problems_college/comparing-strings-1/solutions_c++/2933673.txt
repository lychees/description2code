#include <iostream>
#include <stdio.h>
#include <string.h>
#define MAX 100001
using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		char A[MAX],B[MAX];
		int i,j,flag;
		scanf("%s",A);
		scanf("%s",B);
		flag=0;
		for(i=0;i<strlen(A);i++)
		{
			for(j=0;j<strlen(B);j++)
			{
				if(B[j]==A[i]){
				flag=1;
				break;}
			}
			if(flag==1)
			break;
		}
		if(flag==1)
		printf("YES\n");
		else
		printf("NO\n");
	}
	
	return 0;
}
