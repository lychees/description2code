cases=int(raw_input())

ans=[]
def permutations(head, tail=''):
    if len(head) == 0: ans.append(tail)
    else:
        for i in range(len(head)):
            permutations(head[0:i] + head[i+1:], tail+head[i])


for i in xrange(cases):
	val=raw_input()
	l=''
	for i in xrange(1,int(val)+1):
		l=l+str(i)
	permutations(l)
	
	for i in ans:
		print i ,
	print ""	
	ans=[]
