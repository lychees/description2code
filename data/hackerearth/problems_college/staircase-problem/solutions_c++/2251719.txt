#include <bits/stdc++.h>
using namespace std;

int dp[100005][455];
int best[100005];

const int inf = 12345678;

int main() {
	ios_base::sync_with_stdio(0);
	for(int i = 1; i <= 100000; i++)
		for(int j = 1; j <= 450; j++)
			dp[i][j] = inf;
	dp[1][1] = 1;
	dp[2][1] = 2;
	dp[2][2] = 1;
	for(int i = 3; i <= 100000; i++) {
		best[i] = inf;
		for(int j = 1; j < 450 && i - j > 0; j++) {
			dp[i][j] = min(dp[i][j], 1 + dp[i - j][j]);
			if(j > 1) dp[i][j] = min(dp[i][j], 1 + dp[i - j][j - 1]);
			best[i] = min(best[i], dp[i][j]);
		}
	}
	int t;
	cin >> t;
	while(t--) {
		int n;
		cin >> n;
		cout << best[n] << endl;
	}
	
}
