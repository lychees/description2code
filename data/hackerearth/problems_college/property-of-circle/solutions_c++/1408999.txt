#include <iostream>
#include <string>

struct Circle {
  Circle(int x, int y, int r, const std::string& name): x(x), y(y), r(r), name(name) {};
  Circle& operator=(const Circle& c) {
    x = c.x;
    y = c.y;
    r = c.r;
    name = c.name;
    return *this;
  };
  int x;
  int y;
  int r;
  std::string name;
};

int main() {
  int T;
  std::cin >> T;

  while (T--) {
    int x, y, r;
    std::cin >> x >> y >> r;
    Circle c1{x, y, r, "C1"};

    std::cin >> x >> y >> r;
    Circle c2{x, y, r, "C2"};

    if (c1.r < c2.r) std::swap(c1, c2);
    
    c2.x -= c1.x;
    c2.y -= c1.y;

    c1.x = 0;
    c1.y = 0;

    // special case
    if ( (c1.r==c2.r) && (c2.x==0) && (c2.y==0) ) {
      std::cout << "C2~C1" << '\n';
      continue;
    }

    if (c2.x*c2.x + c2.y*c2.y <= (c1.r - c2.r)*(c1.r - c2.r)) {
      std::cout << c1.name << "C" << c2.name << '\n';
      continue;
    }

    std::cout << "NOT SATISFIED" << '\n';
    continue;

  }
}
