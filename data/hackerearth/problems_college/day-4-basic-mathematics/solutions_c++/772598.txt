#include<cstdio>
#include<cstdlib>
#include<iostream>
#include<algorithm>
using namespace std;
long long int a[100002];
long long binary_search(long long int n){
	long long int initial=1;
	long long int last=100001;
	long long int m=(initial+last)/2;
	while(initial<=last){
		
		if(a[m]==n){
			return m;
		}else if(a[m+1]>=n && a[m]<n ){
			return m+1;
			
		}else if(a[m]>n){

			last=m-1;
		}else if(a[m]<n){

			initial=m+1;
		}
		m=(last+initial)/2;
	}
}
int main()
{
	a[0]=0;
	a[1]=1;
	for(long long int i=2;i<100002;i++){
		a[i]=a[i-1]+i;;
	}
//	cout<<a[100001];
	int t;
	cin>>t;
	while(t--){
		
		long long int n;
		cin>>n;
		long long int lo=binary_search(n);
		//cout<<lo<<endl;
		int mod=lo%3;
		if(mod==0){
			printf("SL\n");
		}else if(mod==1){
			printf("LB\n");			
		}else if(mod==2){
			printf("BS\n");			
		}
		
	}
}
