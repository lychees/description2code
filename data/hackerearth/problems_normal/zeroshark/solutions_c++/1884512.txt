#include<bits/stdc++.h>
using namespace std;
#define ll long long int
int m=1000000007;
ll arr[100005];
int main()
{
	 ll i;
	 arr[1]=1;
	 for(i=2;i<100005;i++)
	 {
	 	arr[i]=((arr[i-1]%m)+(arr[i-2]%m))%m;
	 }
     int t;
	 scanf("%d",&t);
	 while(t--)
	 {
	     ll n,ans=0;
		 scanf("%lld",&n);
		 if(n==1)
		 {
		 cout<<0<<endl;
		 continue;
		 }
		 if(n==2)
		 {
		 cout<<1<<endl;
		 continue;
		 }
		 
		 for(i=0;i<n-1;i++)
		 {
		    ans=(ans+(arr[i+1]*arr[n-i-1])%m)%m;
		 }
		 
		 cout<<ans<<endl;
	 }
	 return 0;
}
