#include <stdio.h>
using namespace std;
long long int dp[201][201];
void init()
{
    int i,j;
    for(i=0;i<201;i++)
    {
        for(j=0;j<201;j++)
        {
            dp[i][j]=-1;
        }
    }
}
long long int fun(int n,int k)
{
   // printf("n:%d ,k:%d\n",n,k);
    if(k<=0)
    {
        return 0;
    }
    if(k>n)
    {
        return 0;
    }
    if(n==k)
    {
        return 1;
    }
    if(dp[n][k]!=-1)
    {
        return dp[n][k];
    }
    dp[n][k]=fun(n,k+1)+ fun(n-k,k);
    return dp[n][k];
}
int main()
{
    int t,n,k;
    long long int ans;
    init();
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d %d",&n,&k);
        ans=fun(n,k);
        printf("%lld\n",ans);
    }
    return 0;
}
