#include <iostream>
#include <algorithm>

typedef unsigned long long ull;
#define MAX_N	100001

ull totalTime[MAX_N+1]; 

#define FORL(s,e,k)	for(int k =s ; k < e;++k)
using namespace std;

struct QuesData{
ull score;
ull time;
};

QuesData allQues[MAX_N]; 

bool CompQuesData(const QuesData &a, const QuesData &b)
{
	return a.score > b.score ;
}

int main()
{
	int N ,Q ;
	cin >> N >> Q ;

	FORL(0,N,i)
	{
		cin >> allQues[i].time ;
	}
	FORL(0,N,i)
	{
		cin >> allQues[i].score ;
	}
	std::sort(allQues,allQues+N,CompQuesData);
	totalTime[0] = 0 ;
	FORL(1,N+1,i)
	{	
		totalTime[i] = totalTime[i-1] + allQues[i-1].time ;
	}
	int K ;
	FORL(0,Q,i)
	{
		cin  >> K ;
		cout << totalTime[K] << endl;
	}
	return 0;
}
