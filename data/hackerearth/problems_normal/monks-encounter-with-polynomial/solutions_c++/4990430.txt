#define ll long long
#include <iostream>

using namespace std;

ll search(ll a, ll b, ll c, ll k, ll low, ll high) {
    ll mid = (low + high) / 2;
    ll current = (a * mid * mid) + (b * mid) + c;

    if (low == high) {
        return mid;
    }

    if (current < k) {
        return search(a, b, c, k, mid + 1, high);
    } else {
        return search(a, b, c, k, low, mid);
    }
}

int main()
{
    int cases;
    cin >> cases;

    while (cases--) {
        ll a, b, c, k;
        cin >> a >> b >> c >> k;
        cout << search(a, b, c, k, 0, 100000) << endl;
    }

    return 0;
}
