int main(){
	int T,len,value,ans;
	string doors;

	vector<int> arr;
	cin>>T;

	while(T--){
		arr.clear();
		cin>>doors;
		len = doors.length();

		value = 0;

		F(i,0,len){
			if(doors[i] == '_'){
				value++;
				continue;
			}

			else if(doors[i] == '|'){
					
				if(value) arr.pb(value);
				value = 0;
			}
		}

		if(value) arr.pb(value);

		if(arr.empty()){
			cout<<"Bob\n";
			continue;
		}

		if((arr.size() == 1) && (value == len)){
			cout<<"Bob\n";
			continue;
		}

		ans = arr[0];
		F(i,1,arr.size())	ans ^= arr[i];

		if(ans) cout<<"Alice\n";
		else cout<<"Bob\n";
	}
}
