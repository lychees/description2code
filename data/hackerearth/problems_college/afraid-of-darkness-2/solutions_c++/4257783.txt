#include <bits/stdc++.h>
using namespace std;
int A[25009],n;
int foo(int c)
{
	int count=0;
	while(1)
	{
		sort(A,A+n,greater<int>());
		for(int i=0;i<c;i++){
			if(A[i]>=1)
			A[i]--;
			else 
			return count;
		}
		count++;
		c++;
	}
	return count;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int i;
    	cin>>n;
    	for(i=0;i<n;i++) cin>>A[i];
        cout<<foo(1)<<endl;
        
    }
    return 0;
}
