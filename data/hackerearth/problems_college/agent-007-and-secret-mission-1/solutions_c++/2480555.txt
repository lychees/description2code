#include <iostream>
#include<string.h>
using namespace std;

int main()
{

	int x=0,y=0,N,T;
  cin>>T;
  while(T--)
  {
  		char binary[100002];
  	int x=0,y=0,sum=0;
  	cin>>binary;
  	cin>>N;
  	for(int i=0;i<strlen(binary);i++)
  	{
  		if(binary[i]=='1')
  		x++;
  		else
  		y++;
  	}
	for(int j=1;j<=N;j++)
	{
		sum+=j;
	}
  	if(sum%2!=0)
  	{
  		cout<<y<<endl;
  	}
  	else
  	cout<<x<<endl;
  }
    return 0;
}
