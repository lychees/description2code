#include <bits/stdc++.h>
#define lli long long
using namespace std;
lli A[200005];
int main()
{
    int t,n;
    lli k;
    cin >> t;
    while ( t--  ){
        cin >> n >> k;
        for ( int i = 0; i < n; i++ ) cin >> A[i];
        int idx = 0;
        lli ans = 0;
        while ( idx < n ) {
            if ( A[idx] > k ) idx++;
            else {
                lli val = 0;
                while ( idx < n && A[idx] <= k ) {
                    val += A[idx];
                    idx++;
                }
                ans = max(ans, val);
            }
        }
        cout << ans << endl;
    }
    return 0;
}
