def fact(n):
    fc = 1
    for i in range(1, n+1):
        fc *= i
    return fc


for i in range(int(raw_input())):
    n = int(raw_input())
    fct = str(fact(n))
    if len(fct) < 5:
        print '%05d' % int(fct)
    else:
        print fct[-1:-6:-1][::-1]
