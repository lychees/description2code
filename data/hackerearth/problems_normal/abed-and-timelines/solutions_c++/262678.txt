#include <iostream>
using namespace std;

int main()
{
    char x;
    int t;
    cin>>t;
    while(t--)
    {cin.get(x);
     int r,c;
     cin>>r>>c;
     int r1,c1,r2,c2;
     cin.get(x);
     cin>>r1>>c1;
     cin.get(x);
     cin>>r2>>c2;
     int p;
     cin.get(x);
     cin>>p;
     int y,z;
   if(r1<=r2)
   { y = r1+r-r2;
   if(y > (r2-r1))
   y = r2-r1;
   }
    else
     { y = r2+r-r1;
   if(y > (r1-r2))
   y = r1-r2;
   }
   
   
    if(c1<=c2)
   { z = c1+c-c2;
   if(z > (c2-c1))
   z = c2-c1;
   }
    else
     { z = c2+c-c1;
   if(z > (c1-c2))
   z = c1-c2;
   }
   int cost = (y+z)*p;
    cout<<cost<<"\n";
   if(cost <=1000)
    cout<<"You saved the group.\n";
else
cout<<"Let go of the group.\n";
}  
   
    return 0;
}
