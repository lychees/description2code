#include<bits/stdc++.h>
using namespace std;
int main()
	{
		int T;
		cin>>T;
		while(T--)
			{
				vector<int>time;
				//map<int,vector<int>>t;
				float num=0;
				for(int i=0;i<7;i++)
					{
						int clses,x,y;
						cin>>clses;
						num=num+clses;
						for(int j=0;j<clses;j++)
							{
								cin>>x;
								cin>>y;
								int diff=y-x;
								time.push_back(diff);
								
							}
						
					}
				sort(time.begin(),time.end());
				num=num*75;
				num=ceil(num/100);
			//	cout<<num<<endl;
				int sum=0;
				for(int i=0;i<num;i++)
					sum=sum+time[i];
				int res=604800-sum;
				cout<<res<<endl;
			}
		return 0;
	}
