from collections import defaultdict
t = int(raw_input())
d = defaultdict(list)
cs = []
rs = []
for _ in range(t):
    c,r = map(int, raw_input().split())
    cs.append(c)
    rs.append(r)
for i in range(len(cs)):
    d[cs[i]].append(rs[i])
for i in d:
    d[i].sort(reverse = True)
for i in d:
    for j in d[i]:
        print "%d %d"%(i,j)
