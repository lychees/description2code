#include <cstdlib>
#include <iostream>

using namespace std;
long T,n,a1,a2,p,i;
int main(int argc, char *argv[])
{   cin>>T;
    while(T)
    {  cin>>n;
       if(n==1)
          cout<<1<<endl;
       else
       if(n==2)
          cout<<2<<endl;
       else
       {   a1=1;
           a2=2; 
           for(i=3;i<=n;i++)
           {   p=a1+a2;
               a1=a2;
               a2=p;
           }
           cout<<p<<endl;
       }    
          
       T--;      
    }
   
    return EXIT_SUCCESS;
}
