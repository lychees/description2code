for _ in range(int(raw_input())):
    n, m, d = map(int, raw_input().split())
    l = [map(int, raw_input().split()) for __ in range(n)]
    t = [map(int, raw_input().split()) for __ in range(m)]
    ans = "NO"
    c = 0
    for x, y in t:
        for xx, yy in l:
            if abs(x-xx) + abs(y-yy) <= d:
                c += 1
                break
        if c > m / 2:
            ans = "YES"
            break
    print ans
