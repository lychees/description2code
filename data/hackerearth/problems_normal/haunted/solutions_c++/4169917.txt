#include<iostream>
#include<stdio.h>
#include<map>
using namespace std;
int main()
{ int n;
  long long int m;
  scanf("%d",&n);
  scanf("%lld",&m);
  std::map<long long int,int> mymap;
  std::pair<std::map<long long int,int>::iterator,bool> ret;
  int max=0,value=0,currmax=0,currval=0;
  while(n--)
  { long long int val;
    scanf("%lld",&val);
    ret = mymap.insert ( std::pair<int,int>(val,1) );
    if (ret.second==false)
      ret.first->second++;
   currval=ret.first->first;
   currmax=ret.first->second;
   if(currmax>max)
   { max=currmax;
     value=currval;
   }
   else if(currmax==max)
   { if(currval>value)
     { max=currmax;
       value=currval;
     }
   }
  cout<<value<<" "<<max<<"\n";
 }
return 0;
}
