#include <iostream>

typedef unsigned long long int ull;
using namespace std;

ull gcd(long long int a, long long int b)
{
	long long int r;
	while(a!=0)
	{
		r = b%a;
		b = a;
		a = r;
	}
	return b;
}


int main()
{
	ull a,b,c;
	ull x1, x2, y1, y2, ar, totar;
	x1 = x2 = y1 = y2 =100001;
	cin >> a >> b >> c;
	totar = 2*a*b;
	if(c > a)
		y2 = c -a;
	else
		x1 = c;

	if( c > b)
		x2 = c - b;
	else
		y1 = c;

	if(x1 <= a)
	{
		if(y1 <= b)
			ar = x1*y1;
		else
		{
			ar = b*(x1 + x2);
		}
	}
	else
	{
		if( y2 > b)
			ar = 2*a* b;
		else
		{
			if(y1 <= b)
			{
				ar = a*(y1 + y2);
			}
			else
				ar = 2*a*b - (a-x2)*(b-y2);
		}
	}
	ull gd = gcd(ar, totar);
	cout << ar/gd << "/"<< totar/gd;
	
	return 0;
}
