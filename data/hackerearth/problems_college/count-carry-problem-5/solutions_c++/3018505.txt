#include<iostream>
#include<string>
#include<cstdio>
using namespace std;

int main()
{
	int n,length,i,tot_carry,carry,num,diff;
	int arr1[1000],arr2[1000];
	string a1,a2;
	cin>>n;
	while(n--)
	{
		cin>>a1>>a2;

		//scanf("%s %s",&a1,&a2);
		if(a1.length()>a2.length())
			{
				length=a1.length();
				diff=(a1.length()-a2.length());
				for(i=0;i<diff;i++)
				{
				 arr1[i]=(int)a1[i]-(int)'0';
			     arr2[i]=0;
				}
				for(;i<length;i++)
				{
				 arr1[i]=(int)a1[i]-(int)'0';
				 arr2[i]=(int)a2[i-diff]-(int)'0';
				}
			}
		else
			{
				length=a2.length();
				diff=(a2.length()-a1.length());
				for(i=0;i<diff;i++)
				{
				 arr2[i]=(int)a2[i]-(int)'0';
				 arr1[i]=0;
				}
				for(;i<length;i++)
				{
				 arr2[i]=(int)a2[i]-(int)'0';
				 arr1[i]=(int)a1[i-diff]-(int)'0';
				}
			}
		//cout<<"len:"<<length;




		tot_carry=0;
		carry=0;
		/*for(i=0;i<length;i++)
		{
			cout<<arr1[i]<<"\t"<<arr2[i]<<"\n";
		}*/

		for(i=length-1;i>=0;i--)
		{
			num=arr1[i]+arr2[i]+carry;
			if(num>9)
			{
				carry=num/10;
				tot_carry++;
			}
			else
			{
				carry=0;
			}
		}
		if(tot_carry==0)
			cout<<"No carry operation"<<"\n";
		else
			if(tot_carry==1)
				cout<<"1 carry operation\n";
			else
		cout<<tot_carry<<" carry operations"<<"\n";
		//cout<<a1<<"\n"<<a2;
		//cout<<a1.length()<<"\t"<<a2.length();
	}
	return 0;
}
