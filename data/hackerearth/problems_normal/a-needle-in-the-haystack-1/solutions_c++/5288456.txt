#include<bits/stdc++.h>
using namespace std;

int main(){
	int t;
	bool ok;
	char c;
	string s,s2;
	cin >> t >> c;
	
	while(t--){
		cin >> s >> s2;
			
		int A[26]={}, B[26]={};
		bool ok2=false;
		
		for(int i=0;i<s.size();i++){
			A[s[i]-'a']++;
		}
		for(int i=0;i<s.size();i++){
			B[s2[i]-'a']++;
		}
		
		for(int i=s.size();i<s2.size();i++){
			ok=true;
			for(int k=0;k<26;k++){
				if(A[k]!=B[k]){
					ok=false;
					break;
				}
			}
			if(ok){
				ok2=true;
				break;
			}
			B[s2[i-s.size()]-'a']--;
			B[s2[i]-'a']++;
		}
		
		ok=true;
		for(int k=0;k<26 && !ok2;k++){
			if(A[k]!=B[k]){
				ok=false;
				break;
			}
		}
		if(ok) ok2=true;
		
		if(ok2) cout << "YES" << endl;
		else cout << "NO" << endl;
	}
	return 0;
}
