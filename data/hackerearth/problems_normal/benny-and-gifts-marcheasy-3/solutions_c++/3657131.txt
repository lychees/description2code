#include <iostream>
#include <cstring>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    int x[100005], y[100005], m = 1, res = 0;
    char str[100005];
    x[0] = y[0] = 0;
    cin >> str;
    int len = strlen(str);
    for(int i = 0; i < len; i++) {
    	if(str[i] == 'L') {
    		x[m] = x[m-1]; y[m] = y[m-1] - 1;
    	}
    	if(str[i] == 'R') {
    		x[m] = x[m-1]; y[m] = y[m-1] + 1;
    	}
    	if(str[i] == 'U') {
    		x[m] = x[m-1] - 1; y[m] = y[m-1];
    	}
    	if(str[i] == 'D') {
    		x[m] = x[m-1] + 1; y[m] = y[m-1];
    	}
    	m++;
    	
    	for(int j = 0; j < m-1; j++) {
    		if(x[m-1] == x[j] && y[m-1] == y[j]) {res++; break;}
    	}
    }
    cout << res << endl;
    return 0;
}
