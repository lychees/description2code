ll f[85];

int main(){
    f[0] = f[1] = 1 ;
    for(int i=2;i<85;++i) f[i] = f[i-1]+f[i-2];
    int t;cin>>t;while(t--){
        ll n;cin>>n;
        if(n==1) {cout<<4,el;continue;}
        ll ans;
        for(int i=1;i<85;++i){
            ll nn = f[i] - i;
            if(nn == n-1) {
                ans = f[i]+1;break;
            }
            if(nn < n-1) continue;
            ll aa = f[i-1]-(i-1) ;
            ans = f[i-1]+(n-aa);break;
        }
        cout<<ans;el;
    }
}
