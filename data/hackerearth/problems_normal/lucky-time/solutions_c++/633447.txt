#include <iostream>
#include <cstring>

#define UPDATE memset(occ, 0, sizeof(occ)); occ[hh/10]++; occ[hh%10]++;occ[mm/10]++; occ[mm%10]++;occ[ss/10]++; occ[ss%10]++

using namespace std;

bool check(int* occ) {
    for(int i = 0; i < 10; i++)
        if(occ[i] > 1) return true;
    return false;
}


void tick(int &h, int &m, int &s) {
    s++;
    m += s/60;
    s %= 60;
    h += m/60;
    m %= 60;
    h %= 24;
}


int main(void) {
    int occ[10];
    int t;
    cin >> t;
    while(t--) {
        int hh, mm, ss;
        char c;
        cin >> hh >> c >> mm >> c >> ss;
        UPDATE;
        while(check(occ)) {
            tick(hh, mm, ss);
            UPDATE;
        }
        if(hh/10 == 0)
            cout << 0;
        cout << hh << ":";
        if(mm/10 == 0)
            cout << 0;
        cout << mm << ":";
        if(ss/10 == 0)
            cout << 0;
        cout << ss << endl;
    }
    return 0;
}
