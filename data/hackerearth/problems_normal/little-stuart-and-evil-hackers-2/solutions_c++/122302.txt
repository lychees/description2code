#include <iostream>
#include <vector>
#define MOD 1000000006
#define N 100000
using namespace std;
typedef unsigned long long int ull;
ull fact[200005];
ull power(ull a, ull b)
{
    ull temp;
    if( b == 0)
        return 1;
    temp = (power(a, b/2))%MOD;
    if (b%2 == 0)
        return (temp*temp)%MOD;
    else
        return ((a%MOD)*((temp*temp)%MOD))%MOD;
}

ull modInverse(ull  a, ull m){
    return power(a,500000001);
}
ull getA(ull n)
{
	ull a=0;
	ull factN=fact[n],fact2N=fact[2*n];
	ull modIn=modInverse(factN,MOD);
	a=((fact2N)*(modIn))%MOD;
	a=((a)*(modIn))%MOD;
	return a;
}
int main()
{
	fact[0]=1;
	ull j=2*N+1;
    for(ull i=1;i<=j;i++)
	{
		fact[i]=(i*fact[i-1])%MOD;
	}
	//ull temp=power(931012898,100000);
	//cout<<"pow="<<temp<<endl;
	ull a,b,t,n;
	cin>>t;
	while(t--)
	{	
		cin>>b>>n;
		a=getA(n);
		//cout<<a<<" "<<b<<endl;
		ull tmp=power(a,b);
		tmp=tmp%MOD;

	}	
    return 0;
}
