    #include <bits/stdc++.h>
    #define mod 1000000007
    using namespace std;
     
     
    int main()
    {
    	int t;
    	long long int  i,ans=1;
    	scanf("%d",&t);
    	while(t--)
    	{
    		long long n,k;
    		scanf("%lld%lld",&n,&k);
    		ans=1;
    		for(i=0;i<n;i++)
    		{
    			ans=ans*(k-i)%mod;
    			ans%=mod;
    		}
    		printf("%lld\n",ans);  
    	}
    	
    	return 0;
    }
