#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	int a, b, c, d;
	while(t--)
	{
		cin >> a >> b >> c >> d;
		int lef = min(max(a, b), max(c, d));
		int righ = max(min(a, b), min(c, d));
		if (lef>righ)
			cout << "Nothing" << endl;
		else
		{
			if (lef==righ)
				cout << "Point" << endl;
			else
				cout << "Line" << endl;
		}
	}return 0;
}
