#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,a[101],i,j,sum;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	sort(a,a+n);
    	vector < int > v;
    	for(i=0;i<n/2;i++)
    	{
    		sum=a[i]+a[n-1-i];
    		v.push_back(sum);
    	}
    	sort(v.begin(),v.end());
    	int x=v.size();
    	cout<<abs(v[0]-v[x-1])<<endl;
    }
    return 0;
}
