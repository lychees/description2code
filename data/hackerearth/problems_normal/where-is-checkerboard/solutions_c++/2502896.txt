#include <iostream>
#include <stdio.h>
#define MAXN 2004

using namespace std;
int N,M,tmp[MAXN][MAXN];
char adj[MAXN][MAXN];

void Read(){
    scanf("%d %d" , &N , &M);
    for(int i=1;i<=N;i++){
            for(int j=1;j<=M;j++)
             	scanf(" %c",&adj[i][j]);
    }
}

void create(){
    for(int i=1;i<=N;i++)
        for(int j=1;j<=M-31;j++){
            int flag = 0;
            for(int k=1;k<=31;k++)
                if(adj[i][k+j] == adj[i][k+j-1]){
                    flag = 1;
                    break;
                }
            if(flag == 0){
                if(adj[i][j] == '_')
                    tmp[i][j] = 1;
                else
                    tmp[i][j]=2;
            }
            else
                tmp[i][j] = 0;
        }
}

int check(int a,int b){
    int idx = 0;
    for(int i=a;i<=a+31;i++)
        if(idx % 2 ==0){ /// _
            if(tmp[i][b] != 1)
                return 0;
            idx++;
        }
        else{
            if(tmp[i][b] != 2)
                return 0;
            idx++;
        }
    return 1;
}

int main(){
    Read();
    create();
    //for(int i=1;i<=N;i++){
    //    for(int j=1;j<=M;j++)
    //        cout<<tmp[i][j];
    //    cout<<'\n';
    // }
    int res = 0;
    for(int i=1;i<=N-31;i++)
        for(int j=1;j<=M-31;j++)
            if(check(i,j))
                res++;
	printf("%d" , res);
    return 0;
}
