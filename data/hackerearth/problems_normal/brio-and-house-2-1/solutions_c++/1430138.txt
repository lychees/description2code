#include<iostream>
#include<cstdio>
#define ll long long
#include<cstring>
#include<algorithm>
#include<cmath>
using namespace std;
int main()
{
	int t;
	double r,x,y,x1,x2,y1,y2,res;
	scanf("%d",&t);
	while(t--)
	{
		cin>>x1>>y1>>x2>>y2;
		cin>>r;
		x=sqrt(pow(x2-x1,2)+pow(y2-y1,2));
		y=fabs(y1*(x2-x1)-x1*(y2-y1));
		res=y/x;
		if(r<res)
			cout<<"SAFE"<<endl;
		else if(r==res)
			cout<<"JUST MISSED"<<endl;
		else
			cout<<"REPLANNING"<<endl;
	}
	return 0;
}
