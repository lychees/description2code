#include <iostream>

using namespace std;

int chandan[1001][1001] = {0};
int arijit[1001][1001] = {0};


int gcd(int a, int b){
	if(b==0) return a;
	return gcd(b,a%b);
}

void pre(){
	for(int i = 1; i<1001; i++){
		chandan[i][1] = 0; 
		arijit[i][1] = 1; 
		arijit[1][i] = 0;
		chandan[1][i] = 1;
	}
	for(int i = 2; i<1001; i++){
		for(int j = 2; j<1001; j++){
			int g = gcd(i,j);
			if(g==1){
				chandan[i][j] = !arijit[i-1][j];
				arijit[i][j] = !chandan[i][j-1];
			}else{
				chandan[i][j] = (!arijit[i/g][j]) || (!arijit[i-1][j]);
				arijit[i][j] = (!chandan[i][j/g]) || (!chandan[i][j-1]);
			}
		}
	}
}

int main(){
	pre();
	int t,a,b;
	cin>>t;
	while(t--){
		cin>>a>>b;
		if(a==1 && b==1){
			cout<<"Draw"<<endl;
		}else if(arijit[a][b]){
			cout<<"Arjit"<<endl;
		}else{
			cout<<"Chandu Don"<<endl;
		}
	}
	return 0;
}
