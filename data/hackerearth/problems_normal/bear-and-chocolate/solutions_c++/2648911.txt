#include <bits/stdc++.h>
#define ll long long int
#define ull unsigned long long int
using namespace std;
 
int main()
{
	int t, n;
	scanf("%d", &t);
	while(t--)
	{
		scanf("%d", &n);
		int a[n][n];
		int cherries_above = 0, cherries = 0;
		int rowsum[n], colsum[n];
		memset(rowsum, 0, sizeof(rowsum));
		memset(colsum, 0, sizeof(colsum));
		char c;
		for(int i = 0 ; i < n ; i++)
		{
			for(int j = 0 ; j < n ; j++)
			{
				cin >> c;
				if(c == '#')
				{
					a[i][j] = 1;
					cherries += 1;
				}
				else
					a[i][j] = 0;
				rowsum[i] += a[i][j];
				colsum[j] += a[i][j];
			}
		}
		//cout << rowsum[0] << " " << colsum[0] << endl;
		for(int i = 1 ; i < n ; i++)
		{	
			rowsum[i] += rowsum[i-1];
			colsum[i] += colsum[i-1];
			//cout << rowsum[i] << " " << colsum[i] << endl;
		}
		bool flag = false;
		if(cherries % 2 == 0)
		{
			for(int i = 0 ; i < n ; i++)
			{
				if(rowsum[i] == cherries/2 || colsum[i] == cherries/2)
				{
					flag = true;
					break;
				}
			}
		}
		if(flag)
			cout << "YES" << endl;
		else
			cout << "NO" << endl;
	}
	return 0;
}
