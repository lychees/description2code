#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while(t--) {
		int n;
		cin >> n;
		int a[n];
		for(int i = 0; i < n; ++i) {
			cin >> a[i];
		}
		
		sort(a, a+n);
		
		int h = -1, l = 1e7;
		for(int i = 0; i < n/2; ++i) {
			h = max(h, a[i] + a[n-1-i]);
			l = min(l, a[i] + a[n-1-i]);
		}
		
		cout << h-l << endl;
	}
}
