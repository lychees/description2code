#include <iostream>
using namespace std;

int A[1005];
bool OK[1005];

int main()
{
    int T; cin>>T;
    while(T--){
    	for(int i=0;i<1005;i++){ A[i] = 0; OK[i] = false;}
    	int P, M;
    	cin>>P>>M;
    	int valid;
    	OK[0] = true;
    	for(int i=0;i<M;i++){ 
    		cin>>valid;
    		for(int j = 1000; j>=valid;j--){
    			if(OK[j-valid]) OK[j] = true;
    		}
    	}
    	
    	int f;
    	for(int p=0;p<P;p++){
    		cin>>f; A[f]++;
    	}
    	int res = 0;
    	for(int i=0;i<1005;i++) if(A[i] > 0 && OK[A[i]]) res++;
    	cout<<res<<endl;
    }
    return 0;
}
