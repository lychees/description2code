#include<iostream>
#include<cstdio>
using namespace std;
long long int gcd(long long int a,long long int b)
{
	if(b==0)
	return a;
	return gcd(b,a%b);
}
int main()
{
	 long long int t,n,x;
	 scanf("%lld",&t);
	 while(t--)
	 {
	 	long long int mn=0,n1;
 		  scanf("%lld",&n);
 		  n1=n;
 		  while(n--)
 		  {
  		 	 scanf("%lld",&x);
  		 	 mn=gcd(mn,x);
  		 }
  		 printf("%lld\n",n1*mn);
 	}
 	return 0;
}
