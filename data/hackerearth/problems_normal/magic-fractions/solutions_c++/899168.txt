#include<iostream>
#include<vector>
#include<cmath>
using namespace std;
typedef long long int lint;

int main(){
    lint n, cnt=0, total=0;
    cin>>n;
    vector<int> p(n+1, -1);
    
    for(lint i=2;i<=n;i++){
        if(p[i]==-1){
            p[i]=1;
            
            for(lint j=i+i;j<=n;j=j+i){
                if(p[j]==-1)
                    p[j] = j;
                
                while(p[j]%i==0){
                    p[j]= p[j]/i;
                    p[i]++;
                }
            }
        }else if(p[i]==1)
            p[i]=0;
    }
    
    for(lint i=1;i<=n;i++){
        if(p[i]>0)
            cnt++;
        total = total + pow(2, cnt-1);
    }
    cout<<total<<endl;
    return 0;
}
