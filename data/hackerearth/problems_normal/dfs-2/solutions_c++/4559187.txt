#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int n; cin >> n;
    int a[n];
    for(int i=0;i<n;i++) cin >> a[i];
    sort(a,a+n);
    int q; cin >> q;
    while(q--)
    {
    	int x,y; cin >> x >> y;
    	int loc1 = lower_bound(a,a+n,x) - a;
    	int loc2 = upper_bound(a,a+n,y) - a;
    	cout << loc2-loc1 << endl;
    }
    return 0;
}
