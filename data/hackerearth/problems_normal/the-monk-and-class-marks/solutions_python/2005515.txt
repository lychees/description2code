from collections import OrderedDict, defaultdict, MutableSet                     
                                                                                 
n = int(raw_input())                                                             
d = defaultdict(list)                                                            
for i in xrange(n):                                                              
    m, n = [i.strip() for i in raw_input().split()]                              
    n = int(n)                                                                   
    d[n].append(m)                                                               
                                                                                                                                                 
od = OrderedDict(reversed(sorted(d.items())))                                    
for k, v in od.iteritems():                                                      
    for i in sorted(v):                                                          
        print '{} {}'.format(i, k)
