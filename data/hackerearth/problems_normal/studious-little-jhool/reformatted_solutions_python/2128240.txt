t = int(raw_input())
for _ in range(t):
    n = int(raw_input())
    ans = 1000000
    for i in range(0, 15):
        for j in range(0, 15):
            if i*10 + j*12 == n:
                ans = min(ans, i+j)
    if (ans == 1000000):
        print "-1"
    else:
        print ans
