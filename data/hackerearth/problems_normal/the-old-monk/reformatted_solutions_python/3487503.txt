tc = int(raw_input())
while tc > 0:
    tc -= 1
    Pmax = 0
    n = int(raw_input())
    a = map(int, raw_input().split())
    b = map(int, raw_input().split())
    for i in xrange(n):
        low = 0
        high = n-1
        pos = 0
        while low <= high:
            mid = low + (high - low) / 2
            if b[mid] >= a[i]:
                pos = mid
                low = mid + 1
            else:
                high = mid - 1

        Pmax = max(Pmax, pos-i)
    print Pmax
