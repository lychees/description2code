struct matrix{
    long long a[3][3];
    matrix(){
        FOR(i,0,2)
        FOR(j,0,2) a[i][j] = 0;
    }
};

int MOD = 1000000007;
matrix a, tg;
int test, n;
long long res;

void init(){
    a.a[0][0] = 0;
    a.a[0][1] = 1;
    a.a[0][2] = 0;
    a.a[1][0] = 4;
    a.a[1][1] = 2;
    a.a[1][2] = 0;
    a.a[2][0] = 4;
    a.a[2][1] = 2;
    a.a[2][2] = 1;
}

matrix nhan(matrix a, matrix b){
    matrix c;
    FOR(i,0,2)
    FOR(j,0,2)
    FOR(k,0,2)
        c.a[i][j] = (c.a[i][j] + a.a[i][k] * b.a[k][j]) % MOD;
    return c;
}

matrix mu(int k){
    if (k == 1) return a;
    matrix tg = mu(k/2);
    tg = nhan(tg,tg);
    if (k % 2 == 0) return tg;
    else return nhan(tg,a);
}

int main()
{
    ios::sync_with_stdio(false);
    //freopen("input.txt", "r", stdin);
    init();
    cin >> test;
    while (test--){
        cin >> n;
        if (n == 1) cout << "2\n";
        else if (n == 2) cout << "14\n";
        else {
            tg = mu(n-2);
            res = tg.a[2][0] * 2 + tg.a[2][1] * 12 + tg.a[2][2] * 14;
            res = res % MOD;
            cout << res << "\n";
        }
    }
    return 0;
}
