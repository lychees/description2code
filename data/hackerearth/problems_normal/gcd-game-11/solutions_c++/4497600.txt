#include <bits/stdc++.h>
using namespace std;
int dp[1005][1005][2],vis[1005][1005][2];
int gcd(int a,int b)
{
    while(a>0&&b>0)
    {
        int x=a;
        //cout<<x<<" ";
        a=b%a;
        //cout<<b<<" "<<
        b=x;
        //cout<<a<<" "<<b<<endl;
    }
    return b;
}
// 1 means that the one with the turn wins and 0 means vice versa.
int solve(int a,int b,int k)
{
    if(a==1&&k==0)
    {
        return 0;//a lose put 1 in previous state
    }
    if(b==1&&k==1)
    {
        return 0;// b lose
    }
    int ans=0;
    if(vis[a][b][k]==1)
        return dp[a][b][k];
    vis[a][b][k]=1;
    if(k==0)
    {
        ans|=!(solve(a,b-1,k^1));//previous state value ! have to be included.
        if(gcd(a,b)>1)
        {
            ans|=!(solve(a,b/gcd(a,b),k^1));
        }
    }
    else
    {
        ans|=!(solve(a-1,b,k^1));
        if(gcd(a,b)>1)
        {
            ans|=!(solve(a/gcd(a,b),b,k^1));
        }
    }
    dp[a][b][k]=ans;
    return ans;
}

}
