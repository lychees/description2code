typedef long long int ll;

using namespace std;

ll a[101][101];

struct point {
    ll pari;
    ll parj;
    ll mi;
    ll mj;
};

ll n,cnt=0;

bool valid(ll i,ll j) {
    if(i>=0 && j>=0 && i<n && j<n) {
        return true;
    }
    return false;
}

int vis[101][101];

void dfs(ll i, ll j) {
    vis[i][j] = 1;
    if(i == n-1 && j == n-1) cnt++;
    if(valid(i+1,j) && vis[i+1][j] == 0 && a[i+1][j] == a[i][j]) {
        dfs(i+1,j);
    }
    if(valid(i,j+1) && vis[i][j+1] == 0 && a[i][j+1] == a[i][j]) {
        dfs(i,j+1);
    }
    if(valid(i-1,j) && vis[i-1][j] == 0 && a[i-1][j] == a[i][j]) {
        dfs(i-1,j);
    }
    if(valid(i,j-1) && vis[i][j-1] == 0 && a[i][j-1] == a[i][j]) {
        dfs(i,j-1);
    }
    vis[i][j]=0;
}

int main()
{
    ios::sync_with_stdio(0);
    ll i,j,r,c;
    cin>>n;
    r=n;
    c=n;
    for(i=0;i<r;i++) {
        for(j=0;j<c;j++) {
            cin>>a[i][j];
        }
    }
    dfs(0,0);
    cout<<cnt<<endl;
    return 0;
}
