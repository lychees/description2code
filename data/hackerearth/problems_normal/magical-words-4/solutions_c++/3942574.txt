#include<bits/stdc++.h>
using namespace std;

int main()
{
   long long  int t,i,ans,a,b,x;
    string s;
    cin>>t;
    while(t--)
    {
        cin>>s;
        ans=0;
        for(i=0;i<s.length();i++)
        {
            a=i-1;
            b=i+1;
            //to find odd length
            x=1;
            ans++;
            while(a>=0&&b<s.length())
            {
                if(s[a]==s[b])
                    {
                        x=x+2;
                        ans=ans+x*x;
                         a--;
                    b++;
                    }
                    else
                    
                    break;
                   
            }

        }
        for(i=0;i<s.length()-1;i++)//for even length
        {
            if(s[i]==s[i+1])
            {
                x=2;
                ans=ans+4;
                a=i-1;
                b=i+2;
                while(a>=0&&b<s.length())
                {
                    if(s[a]==s[b])
                        {
                            x=x+2;
                            ans=ans+x*x;
                            a--;
                        b++;
                        }
                        else break;
                        
                }
            }
        }

        cout<<ans<<endl;
    }
    return 0;
}
