#include <stdio.h>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
	int n,l,r,q,x,i,low;
   scanf("%d%d%d%d",&n,&l,&r,&q);
   int a[r-l+1];
   int k=0;
   for(i=0;i<n;i++)
   {
   scanf("%d",&x);
   if(i>=l&&i<=r)
   a[k++]=x;
   }
   sort(a,a+k);
   while(q--)
   {
   scanf("%d",&x);
   if(x>a[k-1])
   printf("-1\n");
   else if(x<a[0])
    printf("%d\n",a[0]);
   else
   {
    vector<int> v(a,a+k);
    low=lower_bound (v.begin(), v.end(),x)-v.begin();
    if(v[low]==x)
        low++;
printf("%d\n",v[low]);
   }
   }

    return 0;
}
