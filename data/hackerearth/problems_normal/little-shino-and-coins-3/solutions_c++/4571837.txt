#include <bits/stdc++.h>

using namespace std;

int main()
{
    int k;
    string x;
   	cin >> k;
   	cin >> x;
   	int cnt=0;
   	set<char> s;
   	for(int i=0;i<x.length();i++)
   	{
   		for(int j=i;j<x.length();j++)
   		{
   			s.insert(x[j]);
   			if(s.size()==k)
   				cnt++;
   		}
   		s.clear();
   	}
   	cout << cnt << endl;
   	return 0;
}
