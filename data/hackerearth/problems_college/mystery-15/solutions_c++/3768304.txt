#include <iostream>
#include <math.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t>0)
    {
    	int a, b, c, d;
    	long int res=1;
    	cin>>a;
    	cin>>b>>c>>d;
    	for(int i=0; i<a; i++)
    	{
    		res = res *(b+c+d);
    	}
    	cout<<res<<"\n";
    	t--;
    }
    return 0;
}
