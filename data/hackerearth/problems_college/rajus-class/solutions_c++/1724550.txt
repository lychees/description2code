#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
using namespace std;

struct stud{
	int age;
	char name[20];
};

void solve(){
	int N,i,j,R;
	cin>>N;
	struct stud s[N],temp;
	for(i=0;i<N;i++){
		cin>>s[i].age;
		cin>>s[i].name;
	}
	cin>>R;
	
	for(i=0;i<N;i++){
		for(j=0;j<N-i-1;j++){
			if(s[j].age>s[j+1].age){
				temp=s[j];
				s[j]=s[j+1];
				s[j+1]=temp;
			}
		}
	}
	
	cout<<s[R-1].name<<endl;
}

int main()
{
    int T;
    cin>>T;
    while(T--){
    	solve();
    }
    return 0;
}
