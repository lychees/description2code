#include <bits/stdc++.h>

using namespace std;

int main() {
	int power[50001];
	power[0] = 1;
	for (int i = 1; i <= 50000; i++) {
		power[i] = (power[i-1] * 10) % 41;
	}
	
    int T;
    cin >> T;
    for (int t = 0; t < T; t++) {
    	int arr[50001];
    	int C, N;
    	cin >> arr[0] >> arr[1] >> C >> N;
    	if (N == 1) 
    		cout << ((arr[0] % 41 == 0) ? "YES": "NO") << endl;
    	else if (N == 2)
    		cout << (((10*arr[0] + arr[1]) % 41 == 0) ? "YES": "NO") << endl;
    	else {
    		int ans = 0;
    		ans += arr[0] * power[N-1], ans %= 41;
    		ans += arr[1] * power[N-2], ans %= 41;
    		for (int i = 2; i < N; i++) {
    			arr[i] = (arr[i-1]*C + arr[i-2]) % 10; 
    			ans += (arr[i] * power[N-(i+1)]), ans %= 41;
    		}
    		cout << ((ans % 41 == 0) ? "YES": "NO") << endl;
    	}
    }
    return 0;
}
