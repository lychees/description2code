
#include<bits/stdc++.h>
using namespace std;
#define ll long long int
ll gcd(ll a,ll b){
	if(b!=0)
	return gcd(b,a%b);
	else
	return(a);
	}
	
int main()
{
    ll testcase,n,K,x,i;
    scanf("%lld",&testcase);
    while(testcase--)
    {
    	ll k=0;
    	scanf("%lld",&n);
    	ll an=n*(n-1)*(n-2);
        scanf("%lld",&K);        
        while(n--)
        {
            scanf("%lld",&x);
            if(x==K)
                k++;
        }        
        ll ans=k*(k-1)*(k-2);
        ll g=gcd(ans,an);
        an/=g;
        ans/=g;
        printf("%lld/%lld\n",ans,an);
    }
    return 0;
}
