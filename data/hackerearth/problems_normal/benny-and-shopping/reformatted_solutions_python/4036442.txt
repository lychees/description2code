t = input()
list = []
for i in range(0, t):
    list.append(raw_input())
for item in list:
    item = item.replace(' ', '')
    i = item.index("$")
    k = 0
    item = item[i+1:]
    item = item.lower()
    for letter in item:
        if letter.isdigit():
            k = k+1
        else:
            break
    item = item[0:k]
    item = int(item)
    item = '$'+str(item)
    print item
