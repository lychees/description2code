#include <bits/stdc++.h>
using namespace std;
 
int sol[102][10004];
 
int main() {
    int t,n,p,a,c;
    scanf("%d", &t);
    while(t--) {
    	scanf("%d%d", &n,&p);
    	memset(sol,0,sizeof(sol));
    	int X[n],Y[n];
    	for(int i = 0; i < n; i++) {
    		scanf("%d%d", &X[i], &Y[i]);
    	}
    	for(int i = 1; i <= n; i++) {
    		for(int j = 1; j <= p; j++) {
    			if(Y[i-1] <= j) {
    				sol[i][j] = max(sol[i-1][j],sol[i-1][j-Y[i-1]] + X[i-1]-Y[i-1]);
    			} else {
    				sol[i][j] = sol[i-1][j];
    			}
    		//	printf("%d ", sol[i][j]);
    		}
    		//printf("\n");
    	}
    	c = 0;
    	for(int i = 0; i <= p; i++) {
    		if(sol[n][i] == sol[n][p]) {
    			c = i;
    			break;
    		}
    	}
    	printf("%d %d\n", c,p+sol[n][p]);
    }
    return 0;
}
