#include <iostream>
using namespace std;

int main()
{
	int T;
	cin >> T;
	for(int i = 0; i < T; ++i)
	{
		int K;
		cin >> K;
		string str;
		cin >> str;
		int N = str.size();
		int *A = new int[N+1];
		A[0] = 0;
		for(int j = 1; j <= N; ++j)
		{
			A[j] = A[j-1] + (str[j-1]-'a'+1);
		};
		int start = 0;
		int end = 1;
		int count = 0;
		while(end <= N)
		{
			if ((A[end] - A[start]) < K)
			{
				++end;
			}else if ((A[end] - A[start]) > K)
			{
				++start;
			}else
			{
				++count;
				++start;
				++end;
			}
		}
		cout << count << endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
