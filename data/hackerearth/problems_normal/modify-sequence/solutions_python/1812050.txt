import sys
n=int(sys.stdin.readline())
x=map(int,sys.stdin.readline().split())
flag=1
i=n-1
while i>=1:
	if x[i-1]<x[i]:
		flag=0
		print "NO"
		break
	else:
		x[i-1]=x[i-1]-x[i]
		x[i]=0
	i-=1
if n==1 and x[0]==0:
	print "YES"
elif n==1:
	print "NO"
elif flag==1 and x[0]!=0:
	print "NO"
elif flag==1:
	print "YES"
