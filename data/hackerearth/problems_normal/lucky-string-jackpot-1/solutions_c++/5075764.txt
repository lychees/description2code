// power function
int64 power(int64 a,int64 b){
    long long x=1,y=a; // long long is taken to avoid overflow of intermediate results
    while(b > 0){
        if(b%2 == 1){
           x*=y; 
        }
        y*=y; 
        b /= 2;
    }
    return x;
}

{

	while(flag==0){
		psum = sum;
		sum += count*power(2,count);
		if(n <= sum){
			int64 diff  = n-(psum);
			if(2*(count-1)-1<=diff){
				ans = 2*(count)-1;
				if(diff<ans){
					ans = diff;
				}
			}
			else{
				ans = 2*(count-1)-1;
			}
			flag=1;
		}
		count++;
	}
	
	cout<<ans<<"\n";
}
