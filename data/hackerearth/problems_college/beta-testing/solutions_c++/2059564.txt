#include <iostream>
#include<algorithm>
#include<map>
#include<string>
#include<vector>
using namespace std;

int main()
{
   int N;
   cin>>N;
   vector<string> v;
   string str;
   map<string,int> m;
   int i=0,temp=N;
   while(temp--)
   {
   	  cin>>str;
   	  v.push_back(str);
   	  if(!m.count(str))
   	        m[str]=1;
   	  else
   	  	 m[str]++;
   }
   
   int l=0;
   for(map<string,int>::iterator i=m.begin();i!=m.end();i=m.begin())
   {
   	  str=i->first;
   	   for(map<string,int>::iterator it=++i;it!=m.end();it++)
   	   {
   	   		if(m[str]==it->second)
   	   		{
   	   			if( (find(v.begin(),v.end(),str))>(find(v.begin(),v.end(),it->first)) )
   	   			    str=it->first;
   	   		}
   	   		if(m[str]<it->second)
   	   			str=it->first;
   	   			
   	   }
   	   cout<<str<<" "<<m[str]<<endl;
   	   m.erase(m.find(str));
   }
    return 0;
}
