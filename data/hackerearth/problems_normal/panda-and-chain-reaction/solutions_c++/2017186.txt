#include <bits/stdc++.h>
#define mod 1000003
#define ll long long
using namespace std;
ll int a[1000003];
int main()
{
	ll int i;
	a[1]=1;
	for(i=2;i<1000003;i++)
	{
		a[i]=(a[i-1]*i)%mod;
	}
	ll int test;
	scanf("%lld",&test);
	while(test--)
	{
		ll int n,x;
		ll int ans;
		scanf("%lld%lld",&n,&x);
		if(n>=mod)
		{
			ans=0;
		}
		else
		{
			ans=((x%mod)*(a[n]%mod))%mod;
		}
		printf("%lld\n",ans);
	}
	return 0;
}
