#include <bits/stdc++.h>

using namespace std;

#define pb push_back
#define mp make_pair
#define ll long long int
#define s(x) scanf("%d", &x)
#define sd(x) scanf("%lf", &x)
#define sl(x) scanf("%lld", &x)
#define mod 1000000007
#define get getchar_unlocked

int main()
{
    int a, b, q, l, r, ans, tmp;
    s(a);
    s(b);
    s(q);
    while (q--) {
        s(l);
        s(r);
        --l;
        ans = r/a + r/b - r/(a*b);
        tmp = l/a + l/b - l/(a*b);
        double p = ans-tmp;
        double q = r-l;
        printf("%.6lf\n", p/q);
    }
    return 0;
}
