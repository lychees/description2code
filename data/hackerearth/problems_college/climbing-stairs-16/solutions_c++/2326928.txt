#include<iostream>
#include<stdlib.h>
#include<stdio.h>
#include<string.h>
using namespace std;
long long int a[10000];
int main()
{
	int t;
	a[0]=1;
	a[1]=1;
	a[2]=2;
	for(int i=3;i<=1000;i++)
	{
		a[i]=a[i-1]+a[i-2];
	}
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		cout<<a[n]<<endl;
	}
}
