map<ll,list<ll> >adj;
map<ll,bool> visited;
map<ll,ll> level;
set<ll> vertex;
ll BFS(ll start);
int main(){
    t(){
        adj.clear();
        visited.clear();
        level.clear();
        vertex.clear();
        ll E,u,v;
        sl(E);
        wl(E){
            sl(u);sl(v);
            adj[u].pb(v);
            adj[v].pb(u);
            visited[u] = false;
            visited[v] = false;
            level[u] = 0;
            level[v] = 0;
            vertex.insert(u);
            vertex.insert(v);
        }
        ll ans = 0;
        for(set<ll> :: iterator itr = vertex.begin();itr!=vertex.end();itr++){
            if(!visited[(*itr)]){
                ll num = BFS((*itr));
                ans  += num;
            }
        }
        cout << "Case " << test << ": ";
        test++;
        cout << ans << endl;
    }
    return 0;
}
ll BFS(ll start){
    level[start] = 0;
    visited[start] = true;
    ll left = 1,right = 0;
    queue<ll> q;
    q.push(start);
    while(!q.empty()){
        ll u = q.front();
        q.pop();
        for(list<ll> :: iterator i = adj[u].begin();i!=adj[u].end();i++){
            ll v = (*i);
            if(!visited[v]){
                visited[v] = true;
                level[v] = level[u] + 1;
                if(level[v]%2==0)
                    left++;
                else
                    right++;
                q.push(v);
            }
        }
    }
    return max(left,right);
}
