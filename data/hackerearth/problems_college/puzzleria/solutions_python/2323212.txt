import math
for _ in range(input()):
	n = map(int, raw_input().split())
	r = n[1]
	n = n[0]
	if n == 7:
		ncrc = 1
	else:
		ncrc = math.factorial(7)/(math.factorial(7-n)*math.factorial(n))
	if r == 5:
		ncrv = 1
	else:
		ncrv = math.factorial(5)/(math.factorial(5-r)*math.factorial(r))
	
	print ncrv*ncrc*math.factorial(n+r)
