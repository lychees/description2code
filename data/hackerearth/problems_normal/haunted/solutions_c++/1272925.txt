#include<stdio.h>
#include<algorithm>
#include<string>
#include<queue>
#include<map>
#include<vector>
#include<math.h>
#define MAX 100
#define M 1000000007
#define PB push_back
#define scll(t) scanf("%lld",&t)
#define scl(t) scanf("%ld",&t)
#define max(a,b) a>b?a:b
#define min(a,b) a<b?a:b
#define ll long long
#define ll long long
using namespace std;
struct cmp
{
    bool operator()(const pair<ll,ll> &a,const pair<ll,ll> &b)
    {
        return a.second<b.second||(a.second==b.second&&a.first<b.first);
    }
};
map<ll,ll>mp;
priority_queue<pair<ll,ll>,vector<pair<ll,ll> >,cmp>q;
int main()
{
    ll n,m;
    scll(n);
    scll(m);
    for(long i=0;i<n;i++)
    {
        long r;
        scll(r);
        mp[r]++;
        q.push(make_pair(r,mp[r]));
        printf("%lld %lld\n",q.top().first,q.top().second);
    }
    return 0;
}
