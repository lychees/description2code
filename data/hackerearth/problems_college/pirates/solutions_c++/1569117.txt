int n;
#define MAX 100001
int a[MAX];
int main(){
	scanf("%d", &n);
	int val = -2;
	int countt = 0;
	int maxt = 0;
	for (int i = 0; i < n; i++){
		scanf("%d", &a[i]);
		if (val + 1 == a[i]||val-1==a[i]){
			countt++;
			maxt = max(maxt, countt);
		}
		else{
			countt = 1;
		}
		maxt = max(maxt, countt);
		val=a[i];
	}
	printf("%d\n", maxt);
	return 0;
}
