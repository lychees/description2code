#include<bits/stdc++.h>
#define len 1000005
typedef long long li;

int main(){
	int t;
	std::cin>>t;
	while(t--){
		int n;
		std::cin>>n;
		
		li sum=0;
		li arr[len];
		
		std::fill_n(arr,len,0);
		
		for(li i=0;i<n;i++){
			std::cin>>arr[i];
			sum+=arr[i];
		}
		
		li sum_l=sum;
		li sum_r=0;
		bool f=1;
		
		for(li i=n-1;i>=0;i--){
			sum_l-=arr[i];
			sum_r+=arr[i+1];
			if(sum_l==sum_r){
				f=0;
				printf("YES\n");
			}
		}
		if(f) printf("NO\n");
	}
}
