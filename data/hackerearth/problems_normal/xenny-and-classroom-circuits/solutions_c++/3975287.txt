#include <iostream>
using namespace std;

int main()
{
	int t, n, i, add;
	cin >> t;
	while (t--) {
	    string s;
	    cin >> n;
    	cin >> s;
    	add=0;
    	for (i=0;i<2*n;i=i+2) {
    		if (s[i]=='B') add++;
    	}
    	if (add<=n/2) cout << add;
    	else cout << n-add;
    	cout << endl;
	}
    return 0;
}
