#include <bits/stdc++.h>

using namespace std;

int k,m;

bool compare(string a,string b){
	string s1=a.substr(0,m),s2=b.substr(0,m);
	return s1<s2;
}
int main(){
	ios_base::sync_with_stdio(false);
	long long int g,n,t,i;
	string s[1005];
	cin>>t;
	while(t--){
		cin>>n>>k>>m;
		for(i=0;i<n;++i)
			cin>>s[i];
		sort(s,s+n,compare);
		cout<<s[k-1]<<endl;
	}

	return 0;
}
