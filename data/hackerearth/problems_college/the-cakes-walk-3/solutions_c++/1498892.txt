long long power(long long n,int m)
{
    if(m==0) return 1;
    long long x=power(n,m/2);
    if(m%2==0)
               return (x*x);
    else
               return (((x*x))*n);
}

int main()
{
    int t;
    scanf("%d", &t);
    while (t--) {
        long long x;
        scanf("%lld", &x);
        if (x == 1) {
            printf("1 0\n");
        } else {
            printf("%lld %lld\n", x-1, 1LL);
        }
    }
}
