for i in range(input()):
    l = [int(x) for x in raw_input().split()]
    l1 = [int(y) for y in raw_input().split()]
    s = l[1]-sum(l1)
    if (s == 0):
        print("1")
    else:
        if (s < 0):
            print("0")
        else:
            a = 1
            b = 1
            for m in range(1, l[0]):
                a *= (s+m)
                b *= m
                c = a/b
            print(c % 1000000007)
