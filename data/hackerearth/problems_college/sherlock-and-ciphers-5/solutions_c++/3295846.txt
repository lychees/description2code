#include <iostream>
#include<string>
#include<cmath>
using namespace std;

int main()
{
    int n;
    cin>>n;
    cin.ignore();
    while(n--)
    {
    	string c;
    	getline(cin,c);
    //	cout<<c<<endl;
    	int l=c.length();
    	int s=sqrt(l);
    	
		for(int i=0;i<s;i++)
    	{
    		for(int j=i;j<l;j+=s)
    		cout<<c[j];
    	}
    	cout<<endl;
    	
    	
    	
    	
    	
    	
    }
    return 0;
}
