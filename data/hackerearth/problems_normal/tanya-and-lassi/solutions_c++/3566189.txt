#include <bits/stdc++.h>
using namespace std;

long long a[10005];
long long dp[10005];

void solve() {
	int l;
	cin >> l;
	for(int i=1;i <=l;i++) {
		cin >> a[i];
	}
	dp[0] = 0;
	dp[1] = a[1];
	for(int i=2;i<=l;i++) {
		dp[i]=a[i];
		for(int j=1;j<=i;j++) {
			dp[i] = max(dp[i], dp[j] + dp[i-j]);
		}
	}
	cout << dp[l] << "\n";
}

int main()
{
	int tc;
	cin >> tc;
	while(tc--)
	{
		solve();
	}
    return 0;
}
