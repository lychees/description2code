#include <iostream>
using namespace std;
#include <algorithm> 
int main()
{

int t,i,k,l,n,m;
cin>>t;
while(t--){
	cin>>m;
	cin>>n;
	int arr[n];
   for(i=0;i<n;i++){
   	cin>>arr[i];
   }
	sort(arr,arr+n);
/*	for(i=0;i<n;i++){
		cout<<arr[i]<<" ";
	}*/
 k=0,l=n-1;
 int j=0;
    while(k<=l){
    	if(k==l){
    		if(arr[l]<=m)
    		j++;
    		break;
    	}
    	if(arr[k]+arr[l]>m){
        if(arr[l]<=m)
        j++;
        l--;
    	}else{
    		
    		k++,l--;
    		j++;
    	}
    }
	cout<<j<<endl;
	
}
    return 0;
}
