#include <bits/stdc++.h>
using namespace std;

int main()
{
    int i,j,k,l,m,n,t;
    scanf("%d",&t);
    long long p,b,g;
    int a[100001];
    while(t--)
    {
    	g=0;
    	j=0;
    	scanf("%d%d%lld",&n,&m,&p);
    	if(n%2==0)
    		j=1;
    	for(i=0;i<n;++i)
    	{
    		scanf("%d",&a[i]);
    		if(j==0)
    			g+=(long long)a[i];
    		else
    			g-=(long long)a[i];
    		j=1-j;
    	}
    	j=g%m;
    	j+=m;
    	j=j%m;
    	a[n]=j;
    	if(n&1)
    	{
    		k=(p%(n+1));
    		printf("%d\n",a[k]);
    	}
    	else
    	{
    		k=(p%(n+1));
    		j=(p/(n+1));
    		if(j&1)
    		{
    			printf("%d\n",m-a[k]);
    		}
    		else
    			printf("%d\n",a[k]);
    	}
    }
    return 0;
}
