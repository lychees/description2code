import sys

vals = {}
vals[0]=0
vals[1] = 1


def dist(n):
	global vals
	
	f = int(n/2)
	s = int(n/3)
	t = int(n/4)
	
	if f not in vals.keys():
		vals[f] = max(f,dist(f))
	
	if s not in vals.keys():
		vals[s] = max(s,dist(s))
		
	if t not in vals.keys():
		vals[t] = max(f,dist(t))
	
	return vals[f] + vals[s] + vals[t]


def solve(n):
	global vals
	if n not in vals.keys():
		vals[n] = max(n,dist(n))
	return vals[n]

for line in sys.stdin:
	print solve(int(line))
