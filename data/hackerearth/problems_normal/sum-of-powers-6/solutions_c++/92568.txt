#include<iostream>

#include<algorithm>

#include<stdio.h>
#include<math.h>
#include<string>

#include<stdlib.h>
#define MAX 1000010
using namespace std;

bool a[MAX+10]={false};
vector<long long> v[1010];

int main(){	
	v[1].push_back(1);
	for(long long i=2;i<1010;i++){
		for(long long j=i*i;;j*=i){
			if(j>MAX)	break;
			v[i].push_back(j);
		}
	}
		
	for(long long i=1;i<1010;i++){
		for(long long k=0;k<v[i].size();k++){					
			for(long long j=i;j<1010;j++){
				for(long long l=0;l<v[j].size();l++){	
					long long x=v[i][k] + v[j][l];
					if(x>MAX)	break;
					a[x]=true;
				}
			}		
		}
	}
		
	long long t,x;
	cin>>t;	
	
	while(t--){		
		cin>>x;
		if(a[x]==true)	cout<<"Yes";
		else			cout<<"No";
		cout<<endl;
	}
	
	return 0;
}
