for _ in range(int(raw_input())):
    n = int(raw_input().strip())
    i, j = divmod(n, 3)
    if i < 1 or j == 1 and i < 3:
        print -1 
        continue
    if j == 0:
        print "5" * i * 3
    elif j == 1:
        print "5" * (i - 3) * 3 + 10 * "3"
    else: 
        print "5" * (i - 1) * 3 + "33333"
