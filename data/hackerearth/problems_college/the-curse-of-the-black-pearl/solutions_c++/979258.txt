#include <string.h>
#include <iostream>
#include <stdio.h>
using namespace std;

int main()
{
 
  	int c=882,len,i,count;
  	while(c--)
  	{
  		char str[32800];
  		scanf("%s",str);
  		len=strlen(str);
  		count=0;
  		while(len!=1)
  		{
  			if(str[len-1]=='0')
  			{
  				len--;
  			}
  			else
  			{
  				str[len-1]='0';
  				for(i=len-2;i>=0;i--)
  				{
  					if(str[i]=='0')
  					{
  						str[i]='1';
  						break;
  					}
  					else
  					{
  						str[i]='0';
  					}
  				}
  				if(i==-1)
  				{
  					str[0]='1';  					
  					len=len+1;
  					str[len-1]='0';
  				}
  			}
  			count++;
  		}
  		printf("%d\n",count);
  		
  	}
    return 0;
}
