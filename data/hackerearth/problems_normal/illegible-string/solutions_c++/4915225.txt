#include <iostream>
#include <bits/stdc++.h> 
using namespace std ; 
#define ll long long 
int main()
{
	ll int n ; 
	cin >> n ; 
	string s ; 
	cin >> s ; 
	ll int cnt = 0  ;
ll 	int i ; 
	for ( i = 0; i <s.length() ; i++)
	{
		if ( s[i] == 'w')
		{
			cnt = cnt + 2 ; 
 		}
 		else
 		{
 			cnt = cnt +1 ; 
		 }
	}
	ll int chk = 0 ; 
 	ll int num = 0 ; 
	ll int rest = 0 ; 
	for ( i = 0 ; i< s.length() ; i++)
	{
		if ( s[i] == 'w')
		{
			chk = chk + 2 ;
		}
		else if ( s[i] == 'v')
		{
			chk = chk + 1 ; 
		}
		else
		{
			rest = rest + 1 ; 
			if ( chk%2 == 0 )
			{
				chk = chk / 2 ; 
			}
			else
			{
				chk = (chk/2)+1 ; 
			}
			num = num + chk ; 
			chk = 0 ; 
		}
		}
		if ( chk !=0)
		{
				if ( chk%2 == 0 )
			{
				chk = chk / 2 ; 
			}
			else
			{
				chk = (chk/2)+1 ; 
			}
			num = num + chk ;
		}
	
	cout << num + rest << " " << cnt <<endl; 
	return 0 ; 
}
