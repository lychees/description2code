#include <iostream>
using namespace std;
 
int main()
{
    int t,i;
    cin>>t;
    while(t--){
    	string s;
    	cin>>s;
    	int count=0;
    	for(i=0;i<s.size()-1;i++){
    		if(s[i]==s[i+1]){
    			count++;
    		}
    	}
    	cout<<count<<endl;
    }
    
    return 0;
}
