#include<bits/stdc++.h>
using namespace std;
unsigned long long fact(int n){
    if(n==0) return 1;
    if (n>0) return n*fact(n-1);
}
int main()
{
int t;
cin>>t;
while(t--)
{
int a,b;
cin>>a>>b;
 long long s;
unsigned long long ans;
ans=(120)/((fact(5-b)*fact(b)));
for(int i=7;i>7-a;i--)
                ans=ans*i;
ans=ans/fact(a);
ans=ans*fact(a+b);
cout<<ans<<endl;
}
return 0;
}
