#include<iostream>
using namespace std;
int main()
{
    long long int TC,N,ans,i,j;
   cin>>TC;
    while(TC--)
    {cin>>N;
    ans=0;
        if(N==1)
           {cout<<"1"<<endl;
           continue;
          }        i=1;j=N;
                  while(i+1<j)
       {
            ans+=j-i;
            j--;
            if(i+1<j)
           ans+=j-i;
                      i++;
 
        }
        ans+=N-i;
        cout<<ans<<endl;
    }
}
