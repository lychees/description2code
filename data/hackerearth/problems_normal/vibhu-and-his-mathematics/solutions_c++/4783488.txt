#include <iostream>
#include <vector>
#define mod 1000000007
using namespace std;

int main()
{
    int t;
    long n;
    cin>>t;
    long maxm=0;
    vector<long>arn;
    for(int i=0;i<t;i++)
    {
    	cin>>n;
    	maxm=max(n,maxm);
    	arn.push_back(n);
    }
    	long num;
    	vector <long> DP;
    	DP.push_back(0);
    	DP.push_back(1);
    	DP.push_back(2);
    	for(long i=3;i<=maxm;i++)
    	{
    		num=(DP[i-1]+((i-1)*DP[i-2])%mod)%mod;
    		DP.push_back(num);
    	}
    	for(int i=0;i<t;i++)
    	{
    		cout<<DP[arn[i]]<<"\n";
    	}
    return 0;
}
