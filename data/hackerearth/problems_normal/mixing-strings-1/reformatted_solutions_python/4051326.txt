
from itertools import permutations
n = int(raw_input())

stg = []

for i in range(n):
    stg.append(raw_input())

output = []

# finding all permutations

per = permutations(stg)

for st in per:
    opt = st[0]
    for j in range(1, n):
        indices = [i for i, x in enumerate(opt) if x == st[j][0]]

        flag = 1
        for ind in indices:
            if opt[ind:] == st[j][0:len(opt[ind:])]:
                opt = opt+st[j][len(opt[ind:]):]
                flag = 0
                break
        if flag:
            opt = opt+st[j]

    output.append(len(opt))

print min(output)
