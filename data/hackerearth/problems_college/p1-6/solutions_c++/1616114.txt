#include <bits/stdc++.h>
using namespace std;

typedef long long int ll;

typedef vector<int>vi;
typedef pair<int,int>ii;
typedef vector<ii>vii;
#define M 1000000007
const int INF = (int) 1e9;
const int MAX = (int) 1e6 + 10;
int dp[MAX];
int main()
{
    //freopen("input.txt","r",stdin);
    string s;
    cin>>s;
    dp[0]=0;
    for(int i=0;i<s.size();i++)
    {
        if(s[i]=='1')
        {
            dp[i+1]=dp[i]+1;
        }
        else dp[i+1]=dp[i];
    }
    ll ans=0;
    for(int i=0;i<s.size();i++)
    {
        if(s[i]=='0')
        {
            int m=dp[i];
            ll n=dp[s.size()]-dp[i];
            ans+=(m*n);
        }
    }
    cout<<ans<<endl;
    return 0;
}
