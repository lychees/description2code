#include <iostream>
#include <stdio.h>
#include <string.h>
using namespace std;

int main()
{
    int N,sum=0,ans=0;
    cin>>N;
   int rating[N];
    for(int j=0;j<N;j++){
    	cin>>rating[j];
    }
	
	for(int i=0;i<N;i++){
	ans = ans+rating[i];
	if(ans >sum){
	sum=ans;
	}
	if(ans<0){
	ans = 0;
	}
	}
	cout<<sum;
    return 0;
}
