#include<bits/stdc++.h>
using namespace std;
#define pii pair<long,long>
bool compare(const pii &p1, const pii &p2){
	if(p1.second < p2.second)
return true;
else if(p1.second == p2.second)
	return p1.first > p2.first;
else
	return false;
}
int main()
{
int t,n,i;
long long int m;
pair<long,long>pow[10002];
scanf("%d",&t);
while(t--){
bool flag = true;
scanf("%d %lld",&n,&m);
for(i=0;i<n;i++)
scanf("%ld %ld",&pow[i].first,&pow[i].second);
sort(pow,pow+n,compare);

for(i=0;i<n;i++)
{
	//cout<<pow[i].first<<" "<<pow[i].second<<endl;
	if(m < pow[i].second)
	{
		flag = false;
		break;
	}
	else
		m = m + pow[i].first;
}
if(flag == false)
	cout<<"NO"<<endl;
else
	cout<<"YES"<<endl;
}
}
