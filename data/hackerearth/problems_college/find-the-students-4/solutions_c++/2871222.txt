#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n;
    cin>>n;
    int x[n];
    int min;
    for(int i=0;i<n;i++)
   	{
   		cin>>x[i];
   		if(i==0)
   			min=x[i];
   		else if(x[i]<min)
   			min=x[i];
   	}
   	//cout<<min<<endl;
   	int m=n;
   	while(m!=0)
   	{
   		cout<<m<<endl;
   		int min1=1111;
   		for(int i=0;i<n;i++)
   		{
   			if(x[i]!=0)
   			{
   				x[i]=x[i]-min;
   				if(x[i]==0)
   					m--;
   				else if(x[i]<min1)
   					min1=x[i];
   			}
   			//cout<<x[i]<<" ";
   		}
   		min=min1;
   	}
    	
    return 0;
}
