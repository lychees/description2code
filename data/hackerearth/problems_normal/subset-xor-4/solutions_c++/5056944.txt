#include <iostream>
using namespace std;

int main() {
    int T;
    cin>>T;
    for(int testCase = 1; testCase <= T; testCase++)
    {
        int N;
        cin>>N;
        int num = 0;
        for (int i=0; i<N; i++) {
            cin>>num;
        }
        if(N == 1)
            cout<<num<<endl;
        else
            cout<<"0"<<endl;
    }
}
