#include <iostream>
using namespace std;

int main()
{int t;
cin>>t;
while(t--)
{
	int n,p;
	cin>>n>>p;
	int i,a[n],flag=0;
	int count=0;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
		if(a[i]==p)
		{
			flag=1;
		}
		if(a[i]>p)
		count++;
	}
	if(flag==0)
	{
		cout<<"NO 0"<<endl;
	}
	else
	{
		cout<<"YES "<<count<<endl;
	}
}
    
    return 0;
}
