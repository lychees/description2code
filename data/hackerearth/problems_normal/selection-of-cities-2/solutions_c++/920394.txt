long long power(long long base,long long exp,long long c)
          {
              long long x;
            if(exp==0)
            return 1;
            x=power(base,exp/2,c);
            x=x%c;
            x=(x*x)%c;
            if(exp%2==1)
            x=(x*base)%c;
            return x;
          }
