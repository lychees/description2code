#include <iostream>
using namespace std;

int main()
{
   long long int t,n,count,temp;
   
   cin>>t;
   while(t--)
   {
   	cin>>n;
   	count=0;
   	for(int i=1;i<=n;i++)
   	{
   		for(int j=1;j<=n;j++)
   	{
   		if(i<j)
   		{
   		temp=(i*j)%(i+j);
   		if(temp==0)
   		{
   			count++;
   		}
   	}
   	}
   	
   	}
   	
   	cout<<count<<endl;
   	
   }
}
