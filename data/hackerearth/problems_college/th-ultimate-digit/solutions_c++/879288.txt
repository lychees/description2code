#include <iostream>
#include <cmath>
int a[]={1,1,4,4,2,1,1,4,4,2};
void print_last(int n,long long k){
	long long temp;
	int last_digit=n;
	if(k==0){
		std::cout<<"1\n";
		return;
	}
	if(a[last_digit]==1){
		std::cout<<last_digit<<"\n";
		return;
	}

	temp=k%a[last_digit];
	if(temp==0){
		temp=std::pow(last_digit,a[last_digit]);
	}else{
		temp=std::pow(last_digit,temp);
	}
	std::cout<<temp%10<<"\n";		
}
int  main(){
	int t;
	std::cin>>t;
	std::string n;
	long long k;
	while(t--){
		std::cin>>n>>k;
		int p=n.length();
		int x=n.at(p-1)-'0';
		print_last(x,k);
	}
}
