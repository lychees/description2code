#include <iostream>
using namespace std;
int flag=0;
int findsum(int num,int sum,int i){
	if(sum==num){
	//	flag=1;
		return i;
	}
	if(sum>num){
		return i;
	}
	else{
		
		return min(findsum(num,sum+6,i+1),findsum(num,sum+8,i+1));
	}
}

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--){
    	int num,ans=0;
    	cin>>num;
    	if(num<12&&(num%6!=0&&num%8!=0)||num%2!=0){
    		cout<<-1<<endl;
    	}
    	else{
    		ans=findsum(num,0,0);
    		cout<<ans<<endl;
    	}
    	
    }
    return 0;
}
