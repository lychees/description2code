#include <iostream>
using namespace std;

int main()
{
    int n;
   cin>>n;
   while(n--)
   {
   	string s;
   	cin>>s;
   	int l=s.length();
   	char st=s[0];
   	int res=0;
   	for(int i=1;i<l;i++)
   	{
   		if(s[i]==st)
   		res++;
   		else
   		st=s[i];
   	}
   	cout<<res<<endl;
   }
    return 0;
}
