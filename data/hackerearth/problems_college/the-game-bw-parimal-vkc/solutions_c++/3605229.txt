#include<iostream>
#include<string>
#define MOD 1000000007
using namespace std;


int main(){


	string s;
	cin>>s;
	long long mult=1;
	long length=s.size();
	long start=1;
	long long count=0;
	long long mult10=10;
	for(int i=length-1;i>=0;--i){

		int no=s[i]-'0';
		count+=(mult*no)*(i+1);
		count=count % MOD;
		mult=mult+mult10;
		mult=mult % MOD;
		mult10*=10;
		mult10=mult10 % MOD;
	}
	cout<<count<<endl;



}
