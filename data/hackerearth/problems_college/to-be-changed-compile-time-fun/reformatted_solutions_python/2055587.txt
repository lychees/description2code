N = int(raw_input())
nums = [int(i) for i in raw_input().split()]
ans = 0
for j in range(1, N-1):
    max_ = 0
    for i in range(1, N-j):
        temp = nums[i-1]*nums[i+1]
        if temp > max_:
            max_ = temp
            x = i
        # print x
    del nums[x]
    ans += max_
print ans
