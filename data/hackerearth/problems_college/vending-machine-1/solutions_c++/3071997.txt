#include <bits/stdc++.h>

using namespace std;

typedef long long int ll;

int main()
{
	ll n;
	scanf("%lld",&n);
	ll data = (n-1)/5 + 1;
	data = log2(data);
	ll nx = data;
	ll sum = 5 * (pow(2,data) - 1);
	ll count = 0;
	ll adddata = pow(2,nx); 
//	cout << sum << endl;
//	cout << adddata << endl;
	while(sum < n) {
		sum = sum + adddata;
		count++;
	}
	if(count == 1) {
		cout << "Sheldon" << endl;
	} else if(count == 2) {
		cout << "Leonard" << endl;
	}  else if(count == 3) {
		cout << "Penny" << endl;
	} else if(count == 4) {
		cout << "Rajesh" << endl;
	} else if(count == 5) {
		cout << "Howard" << endl;
	}
	return 0;
}
