template<class T1> void deb(T1 e1) {
    cout << e1 << endl;
}

template<class T1,class T2> void deb(T1 e1, T2 e2) {
    cout << e1 << " " << e2 << endl;
}


int main() {
    //ios_base::sync_with_stdio(false);
    char s[MAX];
    scanf("%s",&s);
    bool flag;
    int q,l1,r1,l2,r2;
    cin>>q;
    while(q--) {
        flag=true;
        cin>>l1>>r1>>l2>>r2;
        l1--,l2--,r1--,r2--;
        if((r1-l1)!=(r2-l2))
            flag=false;
        if(!((memcmp(s+l1,s+l2,(sizeof(s[0]))*(r2-l2+1)))==0))
            flag=false;
        deb(flag?"Yes":"No");
    }
    return 0;
}
