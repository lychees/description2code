#include<bits/stdc++.h>

typedef long long int lli;
#define mod 1000000007
using namespace std;


lli fun(lli aval,lli tbu)
{
	lli ans=1,i=0;
	while(i<tbu)
	{
		ans=(ans*(aval-i))%mod;
		i++;
	}
	return ans;
}

int main(void)
{
	int t; scanf("%d",&t);
	while(t--)
	{
		int n; scanf("%d",&n);
		lli visited[1001]={0};
		
		for(int i=0;i<n;i++)
		{
			int temp; scanf("%d",&temp);
			visited[temp]++;
		}
		lli ans=1,used=0;
		for(int i=1;i<1001;i++)
		{
			if(visited[i]>0)
			{
				lli aval=i-used;
				lli tbu=visited[i];
				if(tbu>aval)
				{
					ans=0;
					break;
				}
				else
				{
					ans=(ans*fun(aval,tbu))%mod;
					used+=visited[i];
				}
			}
		}
		cout<<ans<<endl;
	}
}
