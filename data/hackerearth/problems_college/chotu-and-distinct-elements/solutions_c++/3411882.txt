#include<stdio.h>
#include<bits/stdc++.h>
using namespace std;
int main()
{
	set<long int>mt;
	set<long int>:: iterator at,bt;
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long int m,tem;
		long int i;
		scanf("%ld",&m);
		for(i=0;i<m;i++)
		{
			scanf("%ld",&tem);
			mt.insert(tem);
		}
		at=mt.begin();
		bt=mt.end();
		bt--;
		while(*at<*bt)
		{
			printf("%ld %ld ",*bt,*at);
			at++;
			bt--;
		}
		if(*at==*bt)
		printf("%ld",*at);	
		mt.clear();
		printf("\n");
	}
	return 0;
}
