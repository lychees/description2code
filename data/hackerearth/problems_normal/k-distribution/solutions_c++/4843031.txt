#include<bits/stdc++.h>
using namespace std;

long long int dp[1000000];
long long int a[110];

long long int mod=1e9+7;

int main() {
	long long int t,n,m,i,j,k,l,s;

		cin >> n >> k;
		s=0;
		for(i=0;i<n;i++) { 
			cin >> a[i];
			s+=a[i];
		}

		

		memset(dp,0,sizeof(dp));

	
		dp[0]=1;

		for(i=0;i<n;i++) {
			if(a[i]>=k)
				continue;
			else {
				for(j=k-a[i];j>=0;j--) {
					dp[j+a[i]]+=dp[j];
					if(dp[j+a[i]]>=mod)
						dp[j+a[i]]-=mod;
				}
			}
		}


		long long int ans=0;

		for(i=0;i<k;i++) {
			ans+=dp[i];
			if(ans>=mod)
				ans-=mod;

			if(s-i >= k){
				ans+=dp[i];
				if(ans>=mod)
				ans-=mod;
			} 
		}

		long long int total=1;
		for(i=0;i<n;i++) {
			total*=2;
			if(total>=mod)
				total-=mod;
		}
	
		ans=total-ans;

		if(ans<0)
			ans=ans+mod;

		cout << ans << endl;
	

	return 0;
}
