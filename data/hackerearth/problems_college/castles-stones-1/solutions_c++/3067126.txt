#include <iostream>
#include<algorithm>
using namespace std;
 
int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	int x,n;
    	cin>>x>>n; 
    	int a[n];
    	for(int i=0;i<n;i++)
    	cin>>a[i];
    	sort(a,a+n); 
        reverse(a,a+n);
    	int result=0;
    	for(int i=0;i<n;i++)
    	{
    		if(a[i]<=x)
    		{
    			if(a[i]==x)
    			{
    				++result ;
    				continue;
    			}
    			if(a[i]+a[n-1]<=x)
    			{ 
    				--n;
    				++result;
    			}
    			else
    			++result;
    		}
    	}
    	cout<<result<<endl;
    }
    return 0;
}
