inline int scan() {
    int n=0;
    char p=get();
    while((p<'0'||p>'9')&&p!=EOF&&p!='-') p=get();
    if(p=='-') p=get();
    while(p>='0'&&p<='9') { n = (n<< 3) + (n<< 1) + (p - '0'); p=get(); }
    return n;
}

int main() {
        
    ios::sync_with_stdio(false);
    
    int n,q,i;
    lli m = 0,x,y;
    cin >> n >> q;
    vector <lli> a;
    rep(i,0,n) {
    	cin >> x >> y;
    	m += min(x,y);
    	a.pb(x+y-2*min(x,y));
    }
    sort(a.begin(),a.end());
    bool flag = true;
    for (int i = 0; i < n; i += 2) {
		if (a[i] != a[i+1])
			flag = false;
		m += a[i];
    }
    string s = "YES", nn = "NO";
    while (q--) {
    	lli val;
    	cin >> val;
    	cout << ((val == m and flag)?s:nn) << endl;
    }

    
    
    return 0;
        
}
