#include <bits/stdc++.h>
using namespace std;
void push(int * top) {
		*top = *top + 1;
}
int main()
{
		string s;
		int c = 0, ca = 0, cb = 0, cac = 0,cbc = 0, bb = 0, t;
		int top = - 1;
		cin >> t;
		while(t--) {
			cin >> s;
			c = 0;
			top = -1;
			int l = s.length();
			push(&top);
			c = top + 1;
			while(c < l ) {
				if(top != -1) {
					if(s[c] == s[top]) {
							top--;
					} else {
							s[++top] = s[c];
							
					}
					c++;
					
				} else {
						s[++top] = s[c];
						c++;
				}
				
			}
			if(top  < 0) {
					bb++;
			}
		}		
			
				
		cout << bb;
    
}
