#include <iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;

int main()
{
    long int n,k,a[1000006],i;
    scanf("%ld%ld",&n,&k);i=0;
    while(i<n)
    {
    	scanf("%ld",&a[i]);i++;
    }
    sort(a,a+n);
    for(i=0;i<n-1;i++)
    {
    	if(a[i+1]-a[i]>k)
    	break;
    }
    printf("%ld",a[i]);
    return 0;
}
