#include <iostream>
#include<stdlib.h>
#include<cstdio>
#include<list>
using namespace std;


bool is_eulerian(int n,list<int>*adjlist)
{
	//if(is_connected(n)==false)
	//return false;
	int odd=0;
	for(int i=1;i<=n;i++)
	{
		if(adjlist[i].size()&1)
		{
			odd++;
		}
	}
	//cout<<odd<<endl;
	if(odd==2||odd==0)
	return true;
 else return false;
}

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,m;
		cin>>n>>m;
		
list<int>adjlist[100005];
		for(int i=1;i<=m;i++)
		{
			int x,y;
			cin>>x>>y;
			adjlist[x].push_back(y);
			adjlist[y].push_back(x);
		}
		if(is_eulerian(n,adjlist)==true)
		cout<<"Yes"<<endl;
		else
		cout<<"No"<<endl;

	}
    //cout << "Hello World!" << endl;
    return 0;
}
