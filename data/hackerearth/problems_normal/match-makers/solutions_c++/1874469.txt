#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int a[n],b[n];
    	for(int i=0;i<n;i++)
    	cin>>a[i];
    	for(int j=0;j<n;j++)
    	cin>>b[j];
    	sort(a,a+n);
    	sort(b,b+n);
    	int c=0;
    	for(int k=0;k<n;k++)
    	{
    		if(a[k]%b[n-k-1]==0||b[n-k-1]%a[k]==0)
    		c++;
    	}
    	cout<<c<<endl;
    }
    return 0;
}
