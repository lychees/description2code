#include <iostream>
#include <map>
using namespace std;
#define freq 1
int main()
{
   string s;
   getline(cin,s);
   map<int,int> m;
   for(int i=0;i<s.size();i++)
   {
   	    pair<map<int,int>::iterator,bool>rt;
   		rt=m.insert(make_pair((int)(s[i]),freq));
   		if(rt.second==false)rt.first->second+=1;
   }
   int max=-1;char op;
   for(auto i=m.begin();i!=m.end();i++)
   {
   		if(max<i->second)
   		{	
   			max=i->second;
   			op=i->first;
   		}
   }
   cout<<op<<" "<<max;
    return 0;
}
