#include <iostream>
#include<vector>
#include<algorithm>

using namespace std;

int main()
{
    int m,n;
    cin>>m>>n;
    bool success[m];
    string str[m];
    int count=0;
    for(int i=0;i<m;i++)
    {
    	cin>>str[i];
    	success[i]=false;
    }
    /*
    //debugging
    for(int i=0;i<n;i++)
    {
    	for(int j=0;j<m;j++)
    	{
    		cout<<str[i][j]-'0'<<"  ";
    	}
    	cout<<endl;
    }
    */
    for(int i=0;i<n;i++)
    {
    	int max=str[0][i]-'0';
    	vector<int> maxIndex;
    	maxIndex.push_back(0);
    	
    	for(int j=1;j<m;j++)
    	{
    		if(str[j][i]-'0'>max)
    		{
    			max=str[j][i]-'0';
    			maxIndex.clear();
    			maxIndex.push_back(j);
    		}
    		else if(str[j][i]-'0' == max)maxIndex.push_back(j);
    	}
    	
    	vector<int>::iterator it=maxIndex.begin();
    	while(it!=maxIndex.end())
    	{
    		if(success[*it] == false)count++;
    		success[*it]=true;
    		it++;	
    	}
    /*	
    	//debugging
    	it=maxIndex.begin();
    	while(it!=maxIndex.end())
    	{
    		cout<<*it<<"   ";
    		it++;	
    	}
    	cout<<endl;
    */
    	
    	
    	
    }
    /*
    //debugging
    for(int i=0;i<m;i++)cout<<success[i]<<"   ";
    	cout<<endl;
    */
    
    
    cout<<count<<endl;
    return 0;
}
