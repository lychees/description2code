#include <iostream>
using namespace std;

int main()
{
   int t,n,i,j,x,ma,k;
   cin>>t;
   while(t--)
   {
   	cin>>n;
   	for(i=0,k=0;i<n;i++)
   	{
   		for(j=0,ma=0;j<n;j++)
   		{
   			cin>>x;
   			if(x>ma)
   			ma=x;
   		}
   		if(k==1&&ma==n-1)
   	cout<<n<<" ";
   	else
   	{
   		cout<<ma<<" ";
   		if(ma==n-1)
   		k=1;
   	}
   
   	}
   	cout<<endl;
   }
   
    return 0;
}
