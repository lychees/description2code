#include<iostream>
#include<math.h>
#include<iomanip>
using namespace std;
const double Phi = (1+sqrt(5))/2;
const double phi = (1-sqrt(5))/2;
void noways(int N)
{
    long double p1,p2;
    p1 = pow(Phi,N+2);
    p2 = pow(phi,N+2);
    long double sum = (p1-p2)/sqrt(5);
    cout.setf(ios::fixed);
    cout<<setprecision(0)<<pow(sum,2)<<"\n";
}
int main()
{
    long int T,N;
    cin>>T;
    while(T--)
    {
        cin>>N;
        noways(N);
    }
}
