#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;


int main()
{
	long int T;
	cin>>T;
	while(T--){
		vector<unsigned long int> A;
		long int N,distance,sum=0,totalSum=0,k=0;
		int flag=0, i=0;
		cin>>N;
		while(N--) {
			unsigned long int temp;
			cin>>temp;
			A.push_back(temp);
			totalSum += temp;
			k++;
			if(temp>0)
			  k=0;
		}
		cin>>distance;
		vector<unsigned long int>::iterator it;
		long int rem = distance/totalSum;
		sum = totalSum * rem;
		if (sum >= distance){
			cout<<A.size()-k<<endl;
			flag = 1;
		}
		if(flag != 1) {
		  for(it = A.begin(); it != A.end() ; it++) {
		  	sum = sum + *it;
			i++;
			if (sum >= distance) {
				cout<<i<<endl;
				break;
			}
		  }
	    }
	}
    
    return 0;
}
