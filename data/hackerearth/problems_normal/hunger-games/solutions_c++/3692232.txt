#include <iostream>
#include <algorithm>
using namespace std;

int main()
{

		int n;
		cin>>n;
		int a[n];
		for(int i=0;i<n;i++)
		cin>>a[i];
		sort(a,a+n);
		int first=a[1]-a[0];
		int second=a[2]-a[0];
		int max_val=max(first,second);
		
		for(int i=1;i<n-2;i++)
		{
			if((a[i+2]-a[i])>max_val)
			  max_val=a[i+2]-a[i];
		}
		first=a[n-1]-a[n-2];
	    if(first>max_val)
	    max_val=first;
	    
	    cout<<max_val<<"\n";

    return 0;
}
