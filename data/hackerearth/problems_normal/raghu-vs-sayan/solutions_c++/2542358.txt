#include<iostream>
#include<stdlib.h>
using namespace std;
int compare(const void* a,const void *b)
{
    return (*(int *)a - *(int *)b);
}
int main()
{
    int t,m,n,k,count1,count2,i;
    cin>>t;
    while(t--)
    {
        count1=count2=0;
        cin>>m>>n>>k;
        int dishes[k];
        for(i=0;i<k;++i)
            cin>>dishes[i];
        qsort(dishes,k,sizeof(int),compare);
        i=0;
        while(m>=0 && n>=0)
        {
            m=m-dishes[i];
            n=n-dishes[i];
            if(m>=0)
                count1++;
            if(n>=0)
                count2++;
            ++i;
        }
        if(count1==count2)
            cout<<"Tie\n";
        else if(count1>count2)
            cout<<"Raghu Won\n";
        else
            cout<<"Sayan Won\n";
    }
return 0;
}
