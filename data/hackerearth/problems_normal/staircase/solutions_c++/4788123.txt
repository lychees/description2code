#include<bits/stdc++.h>
using namespace std;

int aJa[1000000];
int aJi[1000000];

long int countJack(int n){
	if( n < 1)
		return 0;
	if( aJi[n])
		return aJi[n];

	return countJack(n-1) + countJack(n-2) + countJack(n-3);
}
long int countJill(int n){
	if( n < 1 )
		return 0;
	if( aJa[n])
		return aJa[n];

	return countJill(n-1) + countJill(n-2);
}
int main(){
	int n;
	cin >> n;
	if( n >= 0){
		memset(aJa, 0, n);
		memset(aJi, 0, n);
		aJa[1] = 1, aJa[2] = 2;
		aJi[1] = 1, aJi[2] = 2, aJi[3] = 4;
	
		cout << "Jack-"<<countJack(n) << endl;
		cout << "Jill-"<<countJill(n) << endl;
	}
	else{
		cout << "ERROR"<<endl;
	}
	return 0;
}
