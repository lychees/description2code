#include <bits/stdc++.h>

using namespace std;

typedef long long int ll;


int main() {
    int t,n,k;
    ll r,c,x;
    pair<int,int>pi,pl;
    cin>>t;
    while(t--){
        cin>>n;
        n--;
        cin>>r>>c>>x;
        k=1;
        pi=make_pair((r+c)%2,x%2);
        while(n--){
            cin>>r>>c>>x;
            pl=make_pair((r+c)%2,x%2);
            if(pl.first==pi.first&&pl.second==pi.second||pl.first!=pi.first&&pl.second!=pi.second) continue;
            else {
                    k=0;
            }


        }

        if(k==1) cout<<"Yes"<<"\n";
        else  cout<<"No"<<"\n";
    }
    return 0;
}
