#include <iostream>
#include <cmath>
#include <cstdio>
using namespace std;
long long int pick(int i, int N, int K, int A[]){
	if(i >= N)
		return 0;
	return max(A[i] + pick(i + K + 1, N, K, A), pick(i + 1, N, K, A));
}
int main()
{
	int T;
	cin >> T;
	for(int t = 0; t < T; t++){
	    int N, K;
	    scanf("%d %d", &N, &K);
	    //printf("%d %d\n", N, K);
	    int A[N];
	    for(int n = 0; n < N; n++){
	    	scanf("%d", &A[n]);
	    }
	   	long long int Mem[N+1];
	   	Mem[N] = 0;
	   	for(int n = N - 1; n >= 0; n--){
	   		long long int t1 = A[n], t2 = 0;
	   		if(n + K + 1 < N){
	   			t1 += Mem[n + K + 1];
	   		}
	   		t2 = Mem[n + 1];
	   		Mem[n] = max(t1, t2);
	   	}
	   	printf("%lld\n", Mem[0]);
	   	
	}
    return 0;
}
