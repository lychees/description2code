#include <iostream>
#include <algorithm> //std::set_intersection, std::sort

using namespace std;

long long nC3(long n){
    if(n>=3){
        return n*(n-1)*(n-2)/6;
    }
    else{
        return 0;
    }
}


int main()
{
    long N;
    cin >> N;
    long long A[N];
    static long v2=0,v3=0,v5=0,v7=0;
    static long i23=0,i25=0,i27=0,i35=0,i37=0,i57=0,i235=0,i237=0,i257=0,i357=0,i2357=0;

    long i;
    for(i = 0; i< N; i++){
        cin >> A[i];
    }
    long long temp;
    for(i=0;i<N;i++){
        temp = A[i];

            int i2=0,i3=0,i5=0,i7=0;

            while(temp){  //assumption A[i] not equal to zero
                int j = temp%10;
                if(j==2 || j==3 || j==5 || j==7){
                        if(j==2 && i2==0){
                            v2++;
                            i2++;
                        }
                        else if(j==3 && i3==0){
                            v3++;
                            i3++;
                        }
                        else if(j==5 && i5==0){
                            v5++;
                            i5++;
                        }
                        else if(j==7 && i7==0){
                            v7++;
                            i7++;
                        }
                }

                temp = temp/10;
                }
                if(i2 && i3){
                    i23++;
                }
                if(i2 && i5){
                    i25++;
                }
                if(i2 && i7){
                    i27++;
                }
                if(i3 && i5){
                    i35++;
                }
                if(i3 && i7){
                    i37++;
                }
                if(i5 && i7){
                    i57++;
                }
                if(i2 && i3 && i5){
                    i235++;
                }
                if(i2 && i3 && i7){
                    i237++;
                }
                if(i2 && i5 && i7){
                    i257++;
                }
                if(i3 && i5 && i7){
                    i357++;
                }
                if(i2 && i3 && i5 && i7){
                    i2357++;
                }



     }

     long long result1 = nC3(v2) + nC3(v3) + nC3(v5) + nC3(v7);
     long long result2 = nC3(i23) + nC3(i25) + nC3(i27) + nC3(i35) + nC3(i37) + nC3(i57);
     long long result3 = nC3(i235) + nC3(i237) + nC3(i257) + nC3(i357);
     long long result4 = nC3(i2357);
     long long ans = result1 - result2 + result3 - result4;
     cout << ans << endl;
 }
