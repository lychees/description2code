n = int(raw_input())
i = 0
p = 1
while i < n:
    tc = raw_input()

    if tc == tc[::-1]:
        print "Palindrome"
    else:
        for x in tc:
            p *= ord(x)-96
        print p
        p = 1
    i += 1
