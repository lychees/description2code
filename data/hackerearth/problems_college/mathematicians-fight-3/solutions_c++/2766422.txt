#include <bits/stdc++.h>

using namespace std;
vector<long long> v;

long long josephus(long long n)
{
  for(int i=0;i<v.size();i++){
    if(v[i]<=n&&v[i+1]>n){
        n-=v[i];
        break;
    }
  }
  //cout<<n<<endl;
  return 2*n+1;
}
int main()
{
    long long t,y,n,p;
    long long f=1;
      while(f<1000000000){
        v.push_back(f);
        f*=2;
    }
    scanf("%lld",&t);
    while(t--){
        p=1;
       scanf("%lld",&n);
       long long x=josephus(n)+n;
       //cout<<x<<endl;
       x=(x+p-1)%n;
       if(x<=0)x+=n;
       printf("%d\n",x);
    }
    return 0;
}
