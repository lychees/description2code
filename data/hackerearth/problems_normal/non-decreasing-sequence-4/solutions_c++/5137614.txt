#include <bits/stdc++.h>
using namespace std;

int n, l;
vector<int> a;
void read() {
	scanf("%d", &n);
	a.resize(n);
	for (int i = 0; i < n; ++i) scanf("%d", &a[i]);
	scanf("%d", &l);
}

vector<int> b;
int solve() {
	b.clear();
	b.push_back(INT_MIN);
	for (int i = 0; i < n; ++i) {
		int p = lower_bound(b.begin(), b.end(), a[i]) - b.begin();
		if (p == b.size()) b.push_back(a[i]);
		else b[p] = a[i];
	}
	return l >= b.size() ? -1 : b[l];
}

int main() {
    int t; scanf("%d", &t);
    while (t--) {
    	read();
    	printf("%d\n", solve());
    }
    return 0;
}
