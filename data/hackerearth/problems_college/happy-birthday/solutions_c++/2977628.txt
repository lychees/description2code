#include<bits/stdc++.h>
using namespace std;
int main(){

    int hsh[5]={0};
    int n , x;
    cin >> n ;


    while(n--){

       cin >> x;
     hsh[x]++;


    }
    int ans = hsh[4];

    ans = ans + hsh[3];

    hsh[1] =  hsh[1] - hsh[3];

    ans = ans + hsh[2]/2;

    if(hsh[2]&1){
    ans++;
    hsh[1] = hsh[1] - 2;
    }
    if(hsh[1]>0){
     ans = ans + hsh[1]/4;
     if(hsh[1]%4!=0)
     ans++;
    }
    cout << ans << endl;


}
