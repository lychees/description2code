#include <iostream>
using namespace std;

int main()
{
    int t,i,f;
    string a,b;
    
    cin>>t;
    while(t--){
    	cin>>a>>b;
    	int c1[26]={0};
    	int c2[26]={0};
    	for(i=0;i<a.size();i++){
    		c1[a[i]-'a']++;
    	}
    	
    	for(i=0;i<b.size();i++){
    		c2[b[i]-'a']++;
    	}
    	
    	f=0;
    	for(i=0;i<26;i++){
    		if(c1[i] && c2[i]){
    			f=1;
    			break;
    		}
    	}
    	if(f){
    		cout<<"YES"<<endl;
    	}else{
    		cout<<"NO"<<endl;
    	}
    }
    return 0;
}
