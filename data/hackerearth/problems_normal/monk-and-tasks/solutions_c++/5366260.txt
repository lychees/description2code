#include<iostream>
#include<algorithm>

using namespace std;
#define ll long long

int main()
{
    int t;
    cin>>t;
    while(t--){
        int n;
        cin>>n;
        ll int a[n],i,cop[n],b[n],count,index=0;
        pair<ll int,ll int> p[100001];
        for(i=0;i<n;i++){
            cin>>a[i];
            cop[i]=a[i];
            count=0;
            
            while(cop[i]){
                cop[i]=cop[i]&(cop[i]-1);
                count++;
                }
               b[i]=count;
            p[i].first=b[i];
            p[i].second=i;
          }
        sort(p,p+n);
        
        for(i=0;i<n;i++){
            index=p[i].second;
            cout<<a[index]<<" ";

    }
    
     cout<<endl;
        
    }
    
    
    
        return 0;
}
