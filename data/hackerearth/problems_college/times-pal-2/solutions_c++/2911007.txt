#include<bits/stdc++.h>
using namespace std;

bool ispalin(string s)
{
    int i=0, j=s.length()-1;
    for(; i<=j; i++, j--)
        if(s[i]!=s[j])
            break;
    return (i>j);
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        string s1, s2, s3;
        cin>>s1>>s2;
        long int ctr= 0;
        for(int i=0; i<=s1.length(); i++)
        {
            s3 = s1;
            s3.insert(i, s2);
            if(ispalin(s3)==true)
                ctr++;
        }
        cout<<ctr<<endl;
    }
    return 0;
}
