#include<bits/stdc++.h>
#define ll long long
using namespace std;

ll LIS(vector<ll> & A)
{
    int N = A.size(),i;
    set<int> s;
    set<int>::iterator k;
    for (i=0;i<N;i++)
    {
        if (s.insert(A[i]).second)
        {
           k = s.find(A[i]);
           k++;
           if (k!=s.end())
              s.erase(k);
        }
    }
    return s.size();
}

int main(void)
{
	ll t,n,val;
	cin >> t;
	while(t--)
	{
		vector<ll> vec;
		cin >> n;
		for(ll i=0;i<n;i++)
		{
			cin >> val;
			vec.push_back(val);
		}
		cout << n-LIS(vec) << "\n";
	}
	return 0;
}
