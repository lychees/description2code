import string

test = int(raw_input())

for t in range(test):
    alice = raw_input()
    bob = raw_input()

    alice_dic = dict.fromkeys(string.ascii_lowercase, 0)
    bob_dic = dict.fromkeys(string.ascii_lowercase, 0)

    for i in alice:
        if i in string.ascii_lowercase:
            alice_dic[i] += 1

    for j in bob:
        if j in string.ascii_lowercase:
            bob_dic[j] += 1

    sum_alice = 0
    sum_bob = 0

    for k in string.ascii_lowercase:
        if alice_dic[k] >= bob_dic[k]:
            alice_dic[k] -= bob_dic[k]
            if bob_dic[k] != 0:
                bob_dic[k] = 0
        else:
            bob_dic[k] -= alice_dic[k]
            if alice_dic[k] != 0:
                alice_dic[k] = 0

        sum_alice += alice_dic[k]
        sum_bob += bob_dic[k]

    if sum_alice > sum_bob and sum_bob == 0:
        print "You win some."
    elif sum_alice < sum_bob and sum_alice == 0:
        print "You lose some."
    else:
        print "You draw some."
