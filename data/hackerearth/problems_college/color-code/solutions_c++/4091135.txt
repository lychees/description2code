#include <iostream>
#include <stdlib.h>
using namespace std;



void check(int i)
{
if(i<10)
    cout<<i;
else if(i==10)
    cout<<"U";
else if(i==11)
    cout<<"V";
else if(i==12)
    cout<<"W";
else if(i==13)
    cout<<"X";
else if(i==14)
    cout<<"Y";
else if(i==15)
    cout<<"Z";
else
    exit(0);
}



int main()
{
int t;
cin>>t;
int x=0,y=0,p,q;
int a[25][4]={};
 for(p=0;p<t;p++)
    {
    for(q=0;q<=2;q++)
       {cin>>a[p][q];}
    }
 for(p=0;p<t;p++)
    {cout<<"Case "<<p+1<<": ";
    for(q=0;q<=2;q++)
       {x=a[p][q]/16;
       //cout<<endl<<x<<" "<<i<<" "<<j<<" "<<a[i][j]<<endl;
        check(x);
        y=a[p][q]%16;
        //cout<<y<<" "<<endl;
        check(y);
        cout<<" ";
       }
       cout<<endl;
    }
}
