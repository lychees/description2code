#include <iostream>
using namespace std;

int main() {
	int n,m;
	cin>>n>>m;
	int i,j;
	int a[n][m];
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
			cin>>a[i][j];
	}
	int sum = 0;
	if(n==1)
	{
		for(i=0;i<m;i++)
			cout<<a[0][i]<<endl;
	}
	else if(m==1)
	{
		for(i=0;i<n;i++)
			cout<<a[i][0]<<endl;
	}
	else
	{
		while(sum<=(n)*(m))
		{
			for(i=0;i<n;i++)
			{
				for(j=0;j<m;j++)
				{
					if(i+j==sum)
						cout<<a[i][j]<<" ";
				}
			}
			cout<<endl;
			sum++;
		}
	}
	return 0;
}
