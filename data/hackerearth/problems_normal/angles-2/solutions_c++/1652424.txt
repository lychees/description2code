#include <bits/stdc++.h>
using namespace std;
int mark[10000007];
int main()
{
    int n,k,i,j,x;
    cin>>n>>k;
    
    for(i=0;i<n;i++)
    {
    	cin>>x;
    	for(j=x;j<=10000000;j+=x)
    	mark[j]=1;
    }
    
    for(i=0;i<k;i++)
    {
    	cin>>x;
    	for(j=x;j<=10000000;j+=360)
    	{
    		if(mark[j]==1)
    		{
    			cout<<"YES"<<endl;
    			goto in;
    		}
    	}
    	cout<<"NO"<<endl;
    	in:;
    }
    return 0;
}
