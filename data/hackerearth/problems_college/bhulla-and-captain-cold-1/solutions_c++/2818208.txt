#include <iostream>
using namespace std;

long long dp[1000005][3];
long long a[1000005][3];
int main()
{
	int tc;
	cin >> tc;
	while(tc--)
	{
		int n;
		cin >> n;
		for(int i=0;i<n;i++) cin >> a[i][0] >> a[i][1] >> a[i][2];
		
		dp[0][0] = a[0][0];
		dp[0][1] = a[0][1];
		dp[0][2] = a[0][2];
		for(int i=1;i<n;i++)
		{
			dp[i][0] = min(dp[i-1][1], dp[i-1][2]) + a[i][0];
			dp[i][1] = min(dp[i-1][0], dp[i-1][2]) + a[i][1];
			dp[i][2] = min(dp[i-1][0], dp[i-1][1]) + a[i][2];
		}
		cout << min(dp[n-1][0], min(dp[n-1][1], dp[n-1][2])) << "\n";
	}
}
