#include <iostream>
using namespace std;

int main()
{
	int n,a[10000],val,i,j,x,y,count;
	long int q;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
	}
	cin>>q;
	for(i=0;i<q;i++)
	{
		count=0;
		scanf("%d %d %d",&x,&y,&val);
		for(j=x;j<=y;j++)
		{
			if(a[j]==val)
			count++;
		}
		cout<<count<<endl;
	}
    return 0;
}
