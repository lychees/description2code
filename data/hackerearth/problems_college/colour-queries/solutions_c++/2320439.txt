using namespace std;

LL mat[105][105], visit[105][105];
LL n,m,d;
LL cnt ;

void dfs(LL x, LL y)
{
     visit[x][y] = 1;
     cnt++;
     if(x > 0 && abs(mat[x-1][y]-mat[x][y]) <= d && !visit[x-1][y]) {
          dfs(x-1,y);
     }
     
     if(x < n-1  && abs(mat[x+1][y]-mat[x][y]) <= d && !visit[x+1][y]) {
          dfs(x+1,y);
     }
     
     if(y > 0 && abs(mat[x][y-1]-mat[x][y]) <= d && !visit[x][y-1]) {
          dfs(x,y-1);
     }
     
     if(y < m-1 && abs(mat[x][y+1]-mat[x][y]) <= d && !visit[x][y+1]) {
          dfs(x,y+1);
     }
}
          
     


int main()
{
    LL i,j,k,l,t,q;
    
    cin >> n >> m >> q;
    
    for (i = 0; i < n; i++) {
        for (j = 0; j < m; j++) {
            cin >> mat[i][j];
        }
    }
    while (q--) {
          LL x,y;
          cnt = 0;
          cin >> x >> y >> d;
          x--;
          y--;
          memset(visit,0,sizeof visit);
          dfs(x, y); 
          cout << cnt << endl;
    }
    
     
    return 0;
}
