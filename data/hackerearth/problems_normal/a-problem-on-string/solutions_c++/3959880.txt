#include<iostream>

using namespace std;

#define N_MAX 1000000
#define Q_MAX 1000

int main()
{
    int N,Q;
    char s1[N_MAX];
    //cout << "Enter no" << endl;
    cin >> N;
    //cout << "N is " << N << endl;
    //cout << "Enter String" << endl;
    cin >> s1;
    //cout << "String is " << s1 << endl;
    //cout << "Enter Q " << endl;
    cin >> Q;
    //cout << "String is " << s1 << endl;
    int *a1;
    int *a2;
    a1 = new int [N];
    a2 = new int [N];
    int a1_len = 0;
    int a2_len = 0;
    int i,j;
    int freq[26];
    for(i = 0; i < N; i++)
    {
      	freq[s1[i]-'a']++;
    }
    while(Q > 0)
    {
        char first,last;
        cin >> first;
        cin >> last;
        //cout << "First is " << first << endl;
        //cout << "Last is " << last << endl;
        long int count = 0;
        a1_len = 0;
        a2_len = 0;
       //cout << "String is " << s1 << endl;
       if (first!=last)
       {
 
       		count = freq[first-'a']*freq[last-'a'];
       }
       else
       {
       		for(i = 0; i < N; i++)
        	{
            	//cout << "s1[" << i <<"] " << s1[i] << endl;
            	if (s1[i]==first)
            	{
                	a1_len++;
            	}
        	}
        	if (a1_len%2==0)
        	{
        		count = a1_len/2;
        		count = count*(a1_len-1);
        	}
        	else
        	{
        		count = (a1_len-1)/2;
        		count = count*(a1_len);
        	}
       	}
        cout << count << endl;
       	Q = Q - 1;
    }
return 0;
}
