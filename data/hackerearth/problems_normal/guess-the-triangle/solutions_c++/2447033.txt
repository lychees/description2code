#include <bits/stdc++.h>
using namespace std;

vector<pair <float,float > > v(3);
vector<pair <float,float > > m(3);

int main()
{
    for(int i=0;i<3;i++)
    {
    	cin>>m[i].first>>m[i].second;
    }
    for(int i=0;i<3;i++)
    {
    	float s=0,t=0;
    	for(int j=0;j<3;j++)
    	{
    		if(i==j)
    			{
    				s+= ((-1)*m[j].first);
    				t+= ((-1)*m[j].second);
    			}
    		else
    		{
    				s+= (m[j].first);
    				t+= (m[j].second);
    		}
    	}
    	v[i].first=s;
    	v[i].second=t;
    }
    sort(v.begin(),v.end());
    for(int i=0;i<3;i++)
    {
    	printf("%.4f %.4f\n",v[i].first,v[i].second);
    }
}
