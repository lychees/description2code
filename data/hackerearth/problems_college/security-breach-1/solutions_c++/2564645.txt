#include <iostream>
using namespace std;

int main()
{
	int t,i;
	char str[20];
    scanf("%d",&t);
    while(t--) {
    	scanf("%s",str);
    	i=0;
    	while(str[i]!='\0') {
    		if(str[i]=='2') {
    			printf("cde");
    		}
    		else if(str[i]=='6') {
    			printf("b3");
    		}
    		else if(str[i]=='9') {
    			printf("c6a");
    		}
    		else if(str[i]=='7') {
    			printf("acf");
    		}
    		i++;
    	}
    	printf("\n");
    }
    return 0;
}
