#include <iostream>
using namespace std;

  unsigned long long int  t,r,g,b,x,i;
  unsigned long long int  count[10];
   
   
int main()
{
 cin>>t>>r>>g>>b;
   for(i=0;i<t;i++)
  	{
  		x=0;
  		
  		if((i/r)%2)
  			x+=1;
  		
  		if((i/g)%2)
  			x+=2;
  			
  		if((i/b)%2)
  			x+=4;
  			
  		count[x]++;
  		
  	}
   
   
   cout<<count[1]<<" ";
   cout<<count[2]<<" ";
   cout<<count[4]<<" ";
   cout<<count[3]<<" ";
   cout<<count[6]<<" ";
   cout<<count[5]<<" ";
   cout<<count[7]<<" ";
   cout<<count[0]<<endl;
   
   
    return 0;
}
