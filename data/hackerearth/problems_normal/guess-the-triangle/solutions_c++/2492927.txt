#include <iostream>
#include <stdio.h>
#include <algorithm>


using namespace std;

class point
{
public:
	float x,y;
};

bool operator<(const point& p1, const point& p2)
{
    return p1.x<p2.x || (p1.x==p2.x && p1.y<p2.y);
}

int main()
{
	float a1,b1,a2,b2,a3,b3,x1,y1,x2,y2,x3,y3;
	cin>>a1>>b1;
	cin>>a2>>b2;
	cin>>a3>>b3;

	x1=a1-a2+a3;
	y1=b1-b2+b3;

	x2=a1+a2-a3;
	y2=b1+b2-b3;

	x3=-a1+a2+a3;
	y3=-b1+b2+b3;

	point p[3];
	p[0].x=x1;
	p[0].y=y1;

	p[1].x=x2;
	p[1].y=y2;

	p[2].x=x3;
	p[2].y=y3;

	sort(p,p+3);

	for(int i=0;i<3;i++)
    {

        printf("%.4f %.4f\n",p[i].x,p[i].y);
    }


}
