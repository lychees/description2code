#include <bits/stdc++.h>
using namespace std;
#define ll long long int
ll n,ans,d,a[10005];
int main(){
	int t;
	cin>>t;
	while(t--){
		ans=0;
		ll m=10000005;
		cin>>n;
		for(int i=1;i<=n;i++){
			cin>>a[i];
			m=min(a[i],m);
		}
		for(int i=1;i<=n;i++){
			int d=a[i]-m;
			ans+=d/5;
			d%=5;
			ans+=d/2;
			d%=2;
			ans+=d;
		}
		cout<<ans<<endl;
	}
}
