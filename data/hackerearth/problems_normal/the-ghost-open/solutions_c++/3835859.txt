using namespace std;
#include<iostream>
int main()
{
	int players,matches,i=1,n;
	cin >> n;
	while(i<=n)
	{
		cin >> players;
		if(players>1)
		{
			matches=players-1;
		}
		else
			if(players>=0&&players<=1)
			{
				matches=0;
			}
		cout << matches << endl;
		i++;
	}
	return 0;
}
