#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
#define gc getchar_unlocked
void readInt(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}
ll tree[1000005];
ll readIndex(int idx){
	ll sum=0;                  
	while(idx>0){
		sum=sum+tree[idx];
		idx-=idx & (-idx);
	}
    return sum;
}
void update(int l,int v,int max){
	while(l<=max){
		tree[l]+=v;
		l+=l & (-l);
	}
}
int main(){
	int t=1,n,test=1,u,l,r,v,i,j,q;
	ll res;
	//scanf("%lld",&t);
	while(t--){
		readInt(n);
		memset(tree,0,sizeof(tree));
		for(i=1;i<=n;i++){
			readInt(v);
			update(i,v,n);
			update(i+1,-v,n);
		}
		scanf("%lld",&q);
		while(q--){
			readInt(l);readInt(r);readInt(v);
			//scanf("%d%d%d",&l,&r,&v);
			update(l,v,n);
			update(r+1,-v,n);
		}
		for(i=1;i<=n;i++){
			res=readIndex(i);
			printf("%lld\n",res);
		}
	}
	return 0;
}
