#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int T, N;
    long long rangeNum = 1000000007;
    cin >> T;
    for(int i = 0; i < T; i++)
    {

        cin >> N;

        long long Max[N] ;

        for(int j = 0 ; j < N; j++)
        {
            cin >> Max[j];
        }

        sort(Max, Max + N);
        bool isSolutions = false;
        for(int i = 0; i < N; i++)
        {
            if(Max[i] < i+1)
            {
                cout << "0\n";
                isSolutions = true;
                break;
            }
        }
        if(isSolutions){
        continue;
        }
        unsigned long long ans = 1;
        for(int i = 0; i < N; i++){

            long long temp = Max[i] - i;

            ans = (temp*ans)%rangeNum;

        }

        cout << ans << "\n";



    }


}
