number = []
test = raw_input()
for i in range(int(test)):
    temp = raw_input()
    number.append(temp)
for x in number:
    if ('21' in x):
        print"The streak is broken!"
    elif ((int(x) % 21 == 0)):
        print"The streak is broken!"
    else:
        print"The streak lives still in our heart!"
