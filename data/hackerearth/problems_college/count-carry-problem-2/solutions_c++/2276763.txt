    #include <stdio.h>
     
    int main()
    {
        long int tes,num1,num2,count,sum;
        scanf("%ld",&tes);
        while(tes--)
        {
        	count=0;
        	scanf("%ld%ld",&num1,&num2);
        	while(num1>0&&num2>0)
        	{
        		sum=0;
        		sum=(num1%10)+(num2%10);
        		num1=num1/10;
        		num2=num2/10;
        		if(sum>=10)
        		{
        			count++;
        		}
        	}
        	if(count==0)
        	printf("No carry operation\n");
        	else if(count==1)
        	printf("1 carry operation\n");
        	else
        	printf("%ld carry operations\n",count);
        }
        return 0;
    }
