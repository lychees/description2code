#include <bits/stdc++.h>
using namespace std;

int a[100008],b[100008];

int find(int p,int n){
	
	int s=0,e=n-1,mid;
	
	while(e>=s){
		mid=(s+e)/2;
		if(b[mid]==p)
			return b[mid+1];
		else if(b[mid]>p)
			e=mid-1;
		else
			s=mid+1;
	}
	return b[s];
	
}

int main()
{
    int n,l,r,q,p,i,s;
    cin>>n>>l>>r>>q;
    for(i=0;i<n;i++)
    	cin>>a[i];
    p=0;
    for(i=l;i<=r;i++)
    	b[p++]=a[i];
    sort(b,b+p);
    
    for(i=0;i<q;i++){
    	cin>>s;
    	if(s>=b[p-1]){
    		cout<<"-1\n";
    	}
    	else
    		cout<<find(s,p)<<endl;
    }
    
    return 0;
}
