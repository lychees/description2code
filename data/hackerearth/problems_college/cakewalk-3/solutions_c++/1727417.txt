#include <iostream>
#include <string>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

#define MAX 100010

pair<int, string> a[MAX];
string s, t;
int n, k;

int main(){
    cin >> n;
    cin >> a[0].second;
    a[0].first = 0;
    for (int i = 1; i < n; i++)
        cin >> a[i].first >> a[i].second;
    s = a[n - 1].second;
    k = a[n - 1].first;
    n-=2;
    while (n >= 0 && k){
        if (k > a[n].first){
            s = (a[n].second).substr(0, k - a[n].first) + s;
            k = a[n].first;
        }
        n--;
    }
    cout << s;

    return 0;
}
