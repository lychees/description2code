

n, m = map(int, raw_input().split())

passengers1=[]
taxis = range(1,m+1)
not_available_taxis = []

def free_taxis(endtime,  not_available_taxis):
    for ind, (t,end) in enumerate(not_available_taxis):
        if end<=endtime:
            taxis.append(t)
            not_available_taxis[ind]=[]
    not_available_taxis= filter(lambda a: a!=[], not_available_taxis)
    return not_available_taxis



for i in xrange(n):
    passengers1.append([i]+map(int, raw_input().split()))

passengers = sorted(passengers1, key=lambda x:x[1])



res = [0 for k in xrange(n)]
for i in xrange(n):
    indd = passengers[i][0]
    not_available_taxis = free_taxis(passengers[i][1], not_available_taxis)
    taxis.sort()
    try:
        tt = taxis.pop(0)
        res[indd] = str(tt)
        not_available_taxis.append((tt, passengers[i][1]+passengers[i][2]))
    except IndexError:
        res[indd] =str(-1)

print ' '.join(res)
