#include <iostream>
using namespace std;

int main()
{
    int n;cin>>n;int a[n];
    for(int i=0;i<n;i++)cin>>a[i];
    int s=n;int m=0;int k=0;int ss=0;
    for(int j=0;j<n-2;j++)
    {m=0;
    for(int i=0;i<s-2;i++)
    {if(a[i]*a[i+2]>m){m=a[i]*a[i+2];k=i+1;}}
    ss=ss+m;s--;for(int i=k;i<s;i++)a[i]=a[i+1];
    }
    cout<<ss<<endl;
    return 0;
}
