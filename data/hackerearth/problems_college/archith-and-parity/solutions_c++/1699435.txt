#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long n;
    	cin>>n;
    	bitset<64> a=n;
    	int cu=0;
    	for(int i=0;i<64;i++)
    	{
    		if(a[i]==1)cu++;
    	}
    	if(cu%2==0)cout<<"even"<<endl;
    	else cout<<"odd"<<endl;
    }
    return 0;
}
