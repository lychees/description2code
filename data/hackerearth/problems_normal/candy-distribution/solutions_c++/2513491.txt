#include <iostream>
#include <cmath>
#include <map>
#include <vector>
#define M 1000000007
#define MAXN 10000007

using namespace std;
typedef long long ll;
map <ll,ll> dp;
ll stin(ll a,ll b){
    ll retvalue = 1;
    while(b!=0){
        if(b%2==1)
            retvalue = (retvalue*a)%M;
        a=(a*a)%M;
        b /= 2;
    }
    return retvalue;
}

ll solve(ll N){
    if(N <= 2) return 1;
    N--;
    ll ans = 0;
    for(ll i=1;i<=sqrt(N);i++){
        if(N%i == 0){
            if(i==1){
                ans++;
            }
            else if(i==sqrt(N) && N%i==0){
                if(dp[i] == 0)
                    dp[i] = solve(i);
                ans = (ans + stin(dp[i],N/i))%M;
            }
            else{
                if(dp[i] == 0)
                    dp[i] = solve(i);
                ans = (ans + stin(dp[i],N/i))%M;
                if(dp[N/i]==0)
                    dp[N/i] = solve(N/i);
                ans = (ans + stin(dp[N/i],i))%M;
            }
        }
    }
    return ans;
}

int main(){
    ll N;
    cin>>N;
    cout<<solve(N)<<'\n';
    return 0;
}
