#include <bits/stdc++.h>

using namespace std;
int t;
int n;
int A[1005];
int T[1005];

bool can(int k){
    for (int i = 0; i < n; i++) A[i] = T[i];
    A[0] = max(1, A[0] - k);
    for (int i = 1; i < n; i++){
        if (A[i] > A[i-1])
            A[i] = max(A[i-1] + 1, A[i] - k);
        else
            A[i] = min(A[i-1] + 1, A[i] + k);
        if (A[i] <= A[i-1])
            return false;
    }
    return true;
}

int main()
{
    cin >> t;
    while (t--){
        cin >> n;
        for (int i = 0; i < n; i++) cin >> A[i];
        for (int i = 0; i < n; i++) T[i] = A[i];
        int lo = 0, hi = 1000000000;
        while (lo < hi){
            int mid = (lo + hi) / 2;
            if (can(mid))
                hi = mid;
            else
                lo = mid + 1;
        }
        cout << lo << "\n";
    }
}
