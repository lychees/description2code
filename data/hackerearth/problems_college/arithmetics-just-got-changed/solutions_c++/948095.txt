# include <iostream>
# include <stdio.h>

using namespace std;

int rev(int x)
{
    int r=0;
    while(x>0)
    {
        r=r*10+x%10;
        x=x/10;
    }
    return r;
}
int main()
{
    int n,x,y;
    scanf("%d",&n);
    while(n--)
    {
        int sum=0;
        scanf("%d%d",&x,&y);
        sum=rev(x)+rev(y);
        sum=rev(sum);
        printf("%d\n",sum);
    }
    return 0;
}
