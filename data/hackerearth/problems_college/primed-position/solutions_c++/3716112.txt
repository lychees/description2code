#include<bits/stdc++.h>
using namespace std;
#define MOD 1000000007
bool a[10000005]={0};
long long b[10000005]={0};
int main()
{
	long long t,i,j,m,n;
	
	for(i=3;i*i<10000005;i=i+2)
	{
		if(a[i]==false)
		{
			for(j=2*i;j<10000005;j=j+i)
			{
				a[j]=true;
				
			}
		}
		
	}
	b[0]=2;
	for(i=3,j=1;i<10000005;i=i+2)
	{
		if(a[i]==false)
		{
			b[j++]=i;
		}
		
	}
    scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld%lld",&n,&m);
		printf("%lld\n",((b[b[n-1]-1]*b[b[m-1]-1])%MOD));
	
		
		
	}
	return 0;
}
