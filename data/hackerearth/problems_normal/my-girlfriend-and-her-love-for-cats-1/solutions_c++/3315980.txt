#include <cstdio>
#include <algorithm>
#include <vector>
using namespace std;
vector<long long> cats, cal;
int main(){
	int n;
	long long temp;
	scanf("%d", &n);
	for(int i=0;i<n;i++){
		scanf("%lld", &temp);
		cats.push_back(temp);
	}
	for(int i=0;i<n;i++){
		scanf("%lld", &temp);
		cal.push_back(temp);
	}
	sort(cats.begin(), cats.begin() + n);
	sort(cal.begin(), cal.begin() + n);
	temp = 0;
	for(int i=0;i<n;i++)
		temp = temp + cats[i]*cal[i];
	printf("%lld\n", temp);
	return 0;
}
