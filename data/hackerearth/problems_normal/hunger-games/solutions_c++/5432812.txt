#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n; cin >> n;
	int arr[n];
	for(int i=0;i<n;i++) cin >> arr[i];
	sort(arr, arr+n);
	int res = INT_MIN;
	
	for(int i=0;i<n;i++)
	{
		if(i == 0)
		{
			int x = abs(arr[i]-arr[i+1]);
			res =max(res, x);
			int y = abs(arr[i]-arr[i+2]);
			res = max(res, y);
		}
		else if(i == 1)
		{
			int x = abs(arr[i]-arr[i-1]);
			res =max(res, x);
			int y = abs(arr[i]-arr[i+2]);
			res = max(res, y);
		}
		else if(i == n-1)
		{
			int x = abs(arr[i]-arr[i-1]);
			res =max(res, x);
			int y = abs(arr[i]-arr[i-2]);
			res = max(res, y);
		}
		else if(i == n-2)
		{
			int x = abs(arr[i]-arr[i+1]);
			res =max(res, x);
			int y = abs(arr[i]-arr[i-2]);
			res = max(res, y);
		}
		else
		{
			int x = abs(arr[i]-arr[i+2]);
			res =max(res, x);
			int y = abs(arr[i]-arr[i-2]);
			res = max(res, y);
		}
	}
	cout << res;
}
