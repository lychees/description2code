#include<iostream>
#include<algorithm>

using namespace std;

int a[1000001];
int b[1000001];

int main(){
    int n;
    cin >> n;
    for(int i=0;i<n;i++){
        cin >> a[i];
    }
    for(int i=0;i<n;i++){
        cin >> b[i];
    }
    sort(a, a+n);
    sort(b, b+n);
    long long ans = 0;
    for(int i=0;i<n;i++){
        ans += ((long long)a[i]*b[i]);
    }
    cout << ans << endl;
}
