#include<bits/stdc++.h>
#include<algorithm>
using namespace std;
int main()
{
	long long int n,q,x,i,l,r,mid,flag=0;
	vector<long long int> v(100005);
	scanf("%lld %lld",&n,&q);
	for(i=0;i<n;i++)
		scanf("%lld",&v[i]);
	sort(&v[0],&v[n]);
/*	for(i=0;i<n;i++)
		printf("%lld ",v[i]);
	printf("\n");*/
	while(q--)
	{
		scanf("%lld",&x);
		l=0;r=n-1;flag=0;
		while(l<=r)
		{
			mid=l+(r-l)/2;
			if(v[mid] > x)
			{
				r=mid-1;
			}
			if(v[mid] < x)
			{
				l=mid+1;
			}
			if(v[mid] == x)
			{
				flag=1;
				break;
			}
		}
		if(flag==1)
			printf("YES\n");
		else
			printf("NO\n");
	}



	return 0;
}
