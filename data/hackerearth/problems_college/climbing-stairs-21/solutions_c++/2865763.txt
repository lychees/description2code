#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    unsigned long long int arr[1001];
    arr[1]=1;arr[2]=2;
    for(int i=3;i<1001;i++) arr[i]=arr[i-1]+arr[i-2];
    int tc;cin>>tc;
    while(tc--){
    	int n;cin>>n;
    	cout<<arr[n]<<endl;
    }
    return 0;
}
