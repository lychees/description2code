def string():
    n, s = raw_input().split()
    winner_s = raw_input()
    winner_i = 1
    string_d = {}
    winner_sum = 0
    for i in set(s):
        string_d[i] = s.count(i)
    tempsum = 0
    for j in winner_s:
        if j in string_d.iterkeys():
           tempsum += string_d[j]
    winner_sum = tempsum
    for i in xrange(int(n)-1):
        temp = raw_input()
        tempsum = 0
        for j in temp:
            if j in string_d.iterkeys():
                tempsum += string_d[j]
        if tempsum > winner_sum:
            winner_sum = tempsum
            winner_s = temp
            winner_i = i + 2
        elif tempsum == winner_sum and len(temp) < len(winner_s):
            winner_sum = tempsum
            winner_s = temp
            winner_i = i + 2

    return winner_i


for i in xrange(int(raw_input())):
    print string()
