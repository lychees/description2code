#include <iostream>
using namespace std;
int main(){
	long long oldSum = 0;
	long long sum = 0;
	string str = "";
	string out = "";
	while(cin >> str){

		for(int i = 0; i < str.length(); i++){
            sum += (long long)(str[i] - 48);
		}
		if(sum > oldSum){
			oldSum = sum;
            out = str;
		}

		sum = 0;
	}
	cout << out << endl;
	return 0;
}
