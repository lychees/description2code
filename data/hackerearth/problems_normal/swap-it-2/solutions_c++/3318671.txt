#include<iostream>
using namespace std;
long long a[100005];
int main(){
	int t;
	cin>>t;long n,k;
	for(int i=0;i<t;i++){cin>>n>>k;
	for(int j=0;j<n;j++)cin>>a[j];
	int g=0;int l=0;
	while(k!=0&&g<n){
		long min=a[g];int h=g;
		
		for(int j=g+1;j<=g+k&&j<n;j++){
			
			if(a[j]<min){h=j;min=a[j];
			}
		}
		if(h!=g){
		for(int j=h;j>=g+1;j--){
			a[j]=a[j-1];
		}
		a[g]=min;k=k-h+g;}g=g+1;
	}
	for(int j=0;j<n;j++)cout<<a[j]<<" ";
	cout<<endl;
	
	}
}
