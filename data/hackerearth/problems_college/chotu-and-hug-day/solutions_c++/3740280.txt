#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
     while(t--){
    int n,h;
    cin>>n>>h;
    int a[n+1];
  
    for(int i=0;i<n;i++){
    	cin>>a[i];
    }
    int k=INT_MAX;
    for(int j=0;j<n;j++){
    	if(abs(a[j]-h)<k){
    		k=abs(a[j]-h);
    	}
    }
    cout<<k<<endl;
}
    return 0;
}
