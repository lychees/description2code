#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
#define F(i,a,b) for(int i = (int)(a); i <= (int)(b); i++)
#define RF(i,a,b) for(int i = (int)(a); i >= (int)(b); i--)

int main()
{
    #ifndef ONLINE_JUDGE
        freopen("input.txt","r",stdin);
        freopen("output.txt","w",stdout);
    #endif
    int N,M;
    scanf("%d%d",&N,&M);
    int A[N];
    F(i,0,N-1) scanf("%d",&A[i]);
    sort(A,A+N);
    int ans = 1;
    while(binary_search(A,A+N,M*ans)) ans++;
    ans = ans-1;
    printf("%d",ans*M);
    return 0;
}
