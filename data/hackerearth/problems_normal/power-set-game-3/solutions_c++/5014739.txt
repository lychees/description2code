#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define MOD 1000000007
ll f[1010], invf[1010];

ll power(ll a, ll b){
	ll ret= 1;
	while(b){
		if(b&1)ret = ret*a %MOD;
		a = a*a %MOD;
		b>>=1;
	}
	return ret;
}
ll pw[1010];
int main(){
	f[0] = pw[0] = 1;
	invf[0] = power(1, MOD-2);
	for(int i=1; i<=1000; i++){
		f[i] = (i*f[i-1])%MOD;
		pw[i] = (pw[i-1]*2)%MOD;
		invf[i] = power(f[i], MOD-2);
	}
	ll t;cin>>t;
	
	while(t--){
	ll ans = 0;
		ll n; cin>>n;
		for(int i=1; i<=n; i++){
			ll ways = (f[n]*((invf[i]*invf[n-i])%MOD))%MOD;
			for(int j=i-1; j>=0; j--){
				ll diff = i-j;
				ll ways2 = (f[i]*((invf[j]*invf[i-j])%MOD))%MOD;
				ways2 = (ways*ways2)%MOD;
				ways2 = (ways2*pw[diff])%MOD;
				ans = (ans + ways2)%MOD;
			}
		}
		cout<<ans<<endl;
	}
	return 0;
}
