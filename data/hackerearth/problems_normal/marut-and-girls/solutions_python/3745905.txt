noOfQualities = int(raw_input())

qualities = raw_input().split(" ")
noOfQueries = int(raw_input())

count = 0

for n in range(1, noOfQueries + 1):
	values = raw_input().split(" ")
	m = {}
	for v in values:
		m[v] = 1
	shouldConsider = True	
	for q in qualities:
		if(m.has_key(q) == False):
			shouldConsider = False
	if(shouldConsider):
		count += 1
print count
