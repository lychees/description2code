#include <iostream>
#include<algorithm>

using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int nr,ng;
    	cin>>nr>>ng;
    	int w[nr];
    	for(int i=0;i<nr;i++)
    		cin>>w[i];
    		
    	int ur = w[0];
    	sort(w,w+nr,greater<int>());
    	
    	int count =0;
    	int i=0;
    	while(ur != w[i])i++;
    //	int md = i%ng;
    	
    	int j=i-ng;
    	while(j>=0)
    	{
    		if(ur<w[j]) count++;
    		j=j-ng;
    	}
    	
    	j = i+ng;
    	while(j<nr)
    	{
    		if(ur<w[j]) count++;
    		j = j+ng;
    	}
    	cout<<count<<endl;
    }
    return 0;
}
