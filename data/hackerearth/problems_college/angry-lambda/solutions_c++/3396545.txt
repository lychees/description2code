#include <bits/stdc++.h>
using namespace std;

int main()
{
   int t,k;
   int n;
   int a[1003];
   cin>>t;
   while(t--){
   	cin>>n>>k;
   	for(int i = 0; i < n; i++){
   		cin>>a[i];
   	}
   	int cp = 0;
   	for(int i = 0; i < n ;i++){
   		if(a[i] <= 0){
   			cp++;
   		}
   	}
   	if(cp < k){
   		cout<<"YES"<<endl;
   	}
   	else{
   		cout<<"NO"<<endl;
   	}
   }
    return 0;
}
