#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin >>n;
    int i,a[n];
    for(i=0;i<n;i++)
    {
        cin >>a[i];
    }
    int sum = a[0];
    cout << sum << " ";
    int r;
    for(i=1;i<n;i++)
    {
        r = ((i+1)*a[i])-sum;
        cout << r << " ";
        sum += r;

    }
    cout << endl;

}
