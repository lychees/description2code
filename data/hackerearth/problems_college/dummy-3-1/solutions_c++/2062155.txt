#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,c=0;
		cin>>n;
		int a[100001]={0},b[100001]={0};
		for(int i=0;i<n;i++)
		{
			int x;
			cin>>x;
			c++;
			a[x]=c; b[c]=x;
		}
		int f=0;
		for(int i=1;i<100001;i++) if(a[i]!=b[i])  {f=1; break;}
		if(f) cout<<"not inverse"<<endl;
		else cout<<"inverse"<<endl;
	}
	
	return 0;
}
