int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int i,j,k,n;
		cin>>n;
		int arr[4]={2,3,5,7};
		
		int dp[n+1],coins=INT_MAX;
		dp[0]=0;
		for(i=1;i<=n;i++)
		{
			coins=INT_MAX;
			for(j=0;j<4;j++)
			{
				if(i>=arr[j])
					coins=min(coins,dp[i-arr[j]]);
			}
			if(coins<INT_MAX)
				dp[i]=coins+1;
			else
				dp[i]=INT_MAX;
		}
		cout<<dp[n]<<"\n";
	}
	
	return 0;
}
