#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define mod 1000000007
#define N 10000001
int arr[N]={0};
vector<int> v;
int main() {
	ios_base::sync_with_stdio(0);cin.tie(0);
	int i,j;
   for(i=2;i<=N;i++)
   {   if(arr[i]==0)
    {  int g;j=2;g=j*i;
    	while(g<=N)
    	{
    		arr[g]=1;j++;g=j*i;
    	}
    }
   }
   
   for(i=2;i<=N;i++)
   {
   	if(arr[i]==0)
   	{
   	 v.push_back(i);	
   	}
   }
   ll ans[100001];
   ans[0]=1;j=0;
   for(i=1;i<100001;i++)
    {
    	ans[i]=(ans[i-1]*(ll)v[j++])%mod;
    }
     int t;
     cin>>t;
     while(t--)
     {  //ll ans=1;
     	int k;
     	cin>>k;
     	cout<<ans[k]<<endl;
     }
   
	return 0;
}
