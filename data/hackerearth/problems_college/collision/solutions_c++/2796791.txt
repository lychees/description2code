#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t,n,arr[10],a,sum;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	cin>>n;
    	sum = 0;
    	for(int j=0;j<10;j++)
    	arr[j] = -1;
    	for(int j=0;j<n;j++)
    	{
    		cin>>a;
    		arr[a%10]++;
    	}
    	for(int j=0;j<10;j++)
    	if(arr[j]!=-1) sum+=arr[j];
    	cout<<sum<<endl;
    }
    return 0;
}
