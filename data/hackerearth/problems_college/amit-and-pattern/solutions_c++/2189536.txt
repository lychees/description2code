#include<bits/stdc++.h>
using namespace std;

int main()
{
	int m,n,a[105][105],i,j,x,g,h;
	cin>>m>>n;
	for(i=1;i<=m;i++)
	{
		for(j=1;j<=n;j++)
		{
			cin>>a[i][j];	
		}
	}
	
	
	
	for(i=1;i<=m;i++)
	{
		g=i;
		if(i!=1)
		x=n;
		else
		x=1;	
		for(j=x;j<=n;j++)
		{
			h=j;g=i;	
			while(i<=m && j>=1)
			{
				cout<<a[i][j]<<" ";
				i++;j--;	
			}			
			cout<<endl;
			j=h;i=g;
		}
		i=g;
	
	}
	
	
	
	
	
	
	
	return 0;
}
