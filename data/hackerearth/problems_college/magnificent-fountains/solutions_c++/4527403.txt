#include <iostream>
using namespace std;
#define ll long long int
#define mod 1000000007;
int a[10002]={0};
void sieve(){
	for(int i=2;i*i<10001;i++){
		if(!a[i]){
			for(int j=i*2;j<10001;j+=i)
			a[j]=1;
		}
	}
}
int main()
{
	int T;
    cin>>T;
    
    sieve();
    while(T--)/*Number of test cases*/
    {
    	
        int N;
        long long ans=1;
        cin>>N;/*How many Numbers in Array*/
        ll A[N],i;
        for( i=0;i<N;++i)
        {
            cin>>A[i];//Input Array
        }
        for(int i=2;i<10001;i++){
        	ll j;
        	while(!a[i]){
        		int f=0;
        		for( j=0;j<N;j++){
        			if(A[j]%i==0){
        				A[j]=A[j]/i;
        				f++;
        			}
        		}
        		if(f>0){ans=ans*i%mod;}
        		else break;
        	}
        }
        cout<<ans<<endl;
    }
    return 0;
}
