#include <iostream>
#include <map>
using namespace std;
long long int prime = (long long int)(1e9)+7;

long long int modsq(long long int a){
    long long int r;
    r = a;
    r = (r*a);
    r = r>=prime ? r%prime : r;
    return r;
}

long long int modpow(long long int a, long long int b){
    long long int res=1;
    
    if(b==0) return 1;
    while(b>0){
        if(b&1){
            res  = (res * a) % prime;
        }
        b = b>>1;
        a = (a*a) % prime;
    }
    
    return res;
}

int main()
{
    int t,i;
    long long int n, min, max, a,mi;
   
    cin>>t;
    mi = modpow(6, prime-2) % prime;
    
    while(t>0){
        cin>>n;
        a = n-1;
        a = (a/2);
        min = modsq(a);
        min = (min * n); 
        min = min >=prime ? min % prime : min;
        
        n = n-1;
        max = n;
        max = (max * (n+1)) % prime;
        max = (max * (2*n +1)) % prime;
        max = (max * mi) % prime;
    
        cout<<min<<" "<<max<<endl;
        t--;
    }
    
    return 0;
}
