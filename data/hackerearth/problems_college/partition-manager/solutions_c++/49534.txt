#include <algorithm>
#include <iostream>
#include <vector>

void show_partitions(std::vector<int>& table, int remainder, int max_take)
{
    for (int take = std::min(remainder, max_take); take >= 1; --take)
    {
        table.push_back(take);
        if (remainder - take == 0)
        {
            for (std::vector<int>::const_iterator i = table.begin(); i != table.end(); ++i)
            {
                if (i != table.begin())
                    std::cout << ' ';
                std::cout << *i;
            }

            std::cout << std::endl;
        }
        else
            show_partitions(table, remainder - take, take);

        table.pop_back();
    }
}

int main()
{
    int size;
    std::vector<int> table;

    std::cin >> size;
    show_partitions(table, size, size);
}
