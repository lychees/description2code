#include<bits/stdc++.h>
using namespace std;
int gcd(int a,int b)
{
	if(a==0) return b;
	else return gcd(b%a,a);
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	long long n;
    	int a,b;
    	scanf("%lld%d%d",&n,&a,&b);
    	if(a==0 || b==0) printf("-1\n");
    	else{
    	long long lc=1ll*a*b/gcd(a,b);
    	long long ans=n/a+n/b-n/lc;
    	printf("%lld\n",ans);}
    }
    return 0;
}
