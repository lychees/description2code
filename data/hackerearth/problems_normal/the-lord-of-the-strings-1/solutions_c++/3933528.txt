#include <iostream>
#include <set>
using namespace std;
int mod(char c)
{
	return (c-'a');
}
int main()
{
    int t=0;
    int i=0,j=0,h=0;
    cin>>t;
    for(h=0;h<t;h++)
    {
    	string s="",op="";
    	set <int> hash;
    	cin>>s;
    	for(i=0;i<s.size();i++)
    	{
    	   if(!(hash.count(mod(s[i]))))
    	   {
    	   	  hash.insert(mod(s[i]));
    	      op+=s[i];
    	   }
    	}
    	cout<<op<<"\n";
    }
    return 0;
}
