#include <bits/stdc++.h>
using namespace std;

int mtrx[10001][101];


int main()
{
    int arr[10001],n,q,l,r,val;
    memset(mtrx,0,sizeof(mtrx));
    cin>>n;
    for(int i=1;i<=n;i++){
    	cin>>arr[i];
    	mtrx[i][arr[i]]=1;
    }
    for(int i=2;i<=n;i++){
    	for(int j=1;j<=100;j++){
    		mtrx[i][j] += mtrx[i-1][j];
    	}
    }
    cin>>q;
    while(q--){
    	cin>>l>>r>>val;
    	l++;r++;
    	cout<<mtrx[r][val]-mtrx[l-1][val]<<endl;
    }
    return 0;
}
