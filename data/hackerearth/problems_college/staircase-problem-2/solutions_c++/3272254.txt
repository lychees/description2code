#include <iostream>
using namespace std;
int main()
{
    int t,flag;
    long long a,n,d;
    cin>>t;
    while(t!=0)
    {
     cin>>n;
     a=1;
     int flag=0;
     d=0;
     while(flag!=1)
     {
       d++;
       if(n<=(a+d))
          flag=1;
       a=a+d+1;
     }
     cout<<d<<"\n";
     t--;
    }
    return 0;
}
