def out(x):
    bingo = False
    i = 1
    if x % 3 == 0:
        print '5'*x
        return
    while (i*5 <= x):
        if (x-i*5) % 3 == 0:
            bingo = True
            break
        else:
            i += 1
    if not bingo:
        print -1
    else:
        print '5'*(x-i*5)+'3'*(i*5)


t = input()
for i in range(t):
    x = input()
    out(x)
