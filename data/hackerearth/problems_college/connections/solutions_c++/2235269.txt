#include <iostream>
using namespace std;

int main()
{
    int n,c,u,v,q;
    cin>>n;
    int a[n+1][n+1];
    for(int i=0;i<=n;i++)
    	for(int j=0;j<=n;j++)
    		a[i][j]=99999;
    for(int j=0;j<=n;j++)
    		a[j][j]=0;
    cin>>c;
    for(int i=0;i<c;i++){
    	cin>>u>>v;
    	a[u][v]=1;
    	a[v][u]=1;
    }
    for(int k=0;k<=n;k++){
    	for(int i=0;i<=v;i++){
    		for(int j=0;j<=v;j++){
    			if(a[i][k]+a[k][j] < a[i][j])
    				a[i][j]=a[i][k]+a[k][j];
    		}
    	}
    }
    cin>>q;
    while(q--){
    	cin>>u>>v;
    	if(a[u][v]==1)
    		cout<<"Direct Connection"<<endl;
    	else if(a[u][v]==99999)
    		cout<<"No Connection"<<endl;
    	else
    		cout<<"Indirect Connection"<<endl;
    }
    return 0;
}
