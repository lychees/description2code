int a, b, c;
bool P(int x) {
  for (int i = 2; i * i <= x; i++) {
    if (x % i == 0)
      return false;
  }
  return true;
}
int F(int x, int y) {
  if (x < y || !P(y))
    return 0;
  int tmp, ans;
  tmp = ans = x / y;
  for (int i = 2; i < y && i <= tmp; i++)
    ans -= F(tmp, i);
  return ans;
}
int main() {
  scanf("%d%d%d", &a, &b, &c);
  printf("%d\n", F(b, c) - F(a - 1, c));
}
