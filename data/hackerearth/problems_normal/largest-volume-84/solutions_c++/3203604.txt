#include<iostream>
#include<math.h>
#include<bits/stdc++.h>
#define pi 3.141592653589793238
using namespace std;
int main()
{
    long double r,l,th,temp,max=0;
    cin>>r>>l;
    for(th=0;th<=90;th+=0.001)
    {
        temp=pi/3*l*cos(th*pi/180)*(3*r*r+l*l*sin(th*pi/180)*sin(th*pi/180)+3*r*l*sin(th*pi/180));
        if(max<temp)
            max=temp;

    }
    cout<<fixed;
    cout<<setprecision(2);
    cout<<max<<endl;
    return 0;
}
