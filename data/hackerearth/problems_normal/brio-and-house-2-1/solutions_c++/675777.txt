int main()
{

    int t;
    cin>>t;
    while(t--){
        int x1,x2,y1,y2;
        cin>>x1>>y1>>x2>>y2;
        int r;
        cin>>r;
        LL d= x1*y2-x2*y1;
        LL dr= sqrt((LL)(x2-x1)*(x2-x1)+ (LL)(y2-y1)*(y2-y1));
        LL det= r*r*dr*dr-d*d;
        if(det<0)
            cout<<"SAFE\n";
        else if(det==0)
            cout<<"JUST MISSED\n";
        else
            cout<<"REPLANNING\n";
    }
}
