#include<bits/stdc++.h>
using namespace std;
int main()
{
    int i,j,n,k;
    cin>>n>>k;
    vector<int> v;
    
    for(i=0;i<n;i++)
    {
        int x;
        cin>>x;
        v.push_back(x);
    }    
    sort(v.begin(),v.end());
    int sum=0,c=0;
    for(i=0;i<n;i++)
    {
        sum+=v[i];
        if(sum>k)
            break;
        else if(sum<=k)
            c++;
    }
    cout<<c<<endl;
    return 0;
}
