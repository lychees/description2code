#include<iostream>
#include<algorithm>
#include<algorithm>
using namespace std;
int main()
{
    string s;
    int n,a[27],count,i;
     cin>>s;
     cin>>n;
     
     
     while(s[0]!=48)
     {
         for(i=0;i<27;i++)a[i]=0;
         for(i=0;i<s.length();i++)
         {
             if(a[(s[i]%65)]==0 && n==0)
             {
             	count++;
             	a[(s[i]%65)]=2;
             }
             else if(a[(s[i]%65)]==0 && n)
             {
             	n--;
                a[(s[i]%65)]=1;
             }
             else if(a[(s[i]%65)]==1)
             {
                 n++;
                 a[(s[i]%65)]==0;
             }
             else if(a[(s[i]%65)]==2)
             {
             	a[(s[i]%65)]=0;
             }
         }
         cout<<count<<"\n";
         count=0;
         cin>>s;
         cin>>n;
     
     
     }
    return(0);
}
