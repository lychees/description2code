const int INF = 1000000000;
const int MAX = 1007;
const int MAX2 = 2000;
const int BASE = 1000000000;

double p[MAX][MAX];
double dp[MAX][MAX];

int main()
{
    //freopen("in.txt", "r", stdin);

    int t;
    cin >> t;
    FOR(tt,0,t)
    {
        FILL(dp,0);
        int n , m;
        cin >> n >> m;
        FOR(i,0,n)
        {
            FOR(j,0,m)
            {
                scanf("%lf" , &p[i][j]);
            }
        }
        dp[0][0] = 1;
        FOR(i,0,n)
        {
            FOR(j,0,m)
            {
                dp[i][j + 1] += dp[i][j] * p[i][j];
                dp[i + 1][j] += dp[i][j] * (1 - p[i][j]);
            }
        }
        double res = 0;
        FOR(j,0,m)
        {
            res += dp[n][j] * (n + j);
        }
        FOR(i,0,n)
        {
            res += dp[i][m] * (m + i);
        }
        printf("%.6f\n" , res);
    }

    return 0;
}
