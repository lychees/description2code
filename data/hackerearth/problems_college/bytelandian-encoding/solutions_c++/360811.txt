#include<iostream>
#include<algorithm>
#include<cstring>
#include<cstdio>

using namespace std;

main()
{
	
	int i=0,j,x,a,b,length,len;
	int t;
	
	cin>>t;
	while(t--)
	{
	char arr[50],k[50];
	
	cin>>arr;
	len=strlen(arr);
	for(j=0;j<len;j++)
	{
	length=strlen(arr);
	if(length==1)
	{
		k[i]=arr[length-1];
		i++;
		break;
	}
	
	else if(length%2==0)
	{
		a=(length/2);
		b=a+1;
		
		if((int)arr[a-1]<(int)arr[b-1])
		{
			k[i]=arr[a-1];
			memmove(&arr[a-1],&arr[a],length-(a-1));
			i++;
			//cout<<"arr "<<arr<<endl;
			//cout<<"k "<<k<<endl;
		}
		else
		{
			k[i]=arr[b-1];
			memmove(&arr[a],&arr[a+1],length-a);
			i++;
			//cout<<"arr "<<arr<<endl;
			//cout<<"k "<<k<<endl;
		}
	}
	else
	{
		x=(length/2);
		k[i]=arr[x];
		memmove(&arr[x],&arr[x+1],length-x);
		i++;
		//cout<<"arr "<<arr<<endl;
		//cout<<"k "<<k<<endl;
		
	}
	}
	printf("%.*s",i,k);
	cout<<endl;
	i=0;
	}
	return 0;
}
