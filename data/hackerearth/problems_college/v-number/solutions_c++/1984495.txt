#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    
    for(int i=0;i<t;i++)
    {
    	string s;
    	cin >> s;
    	
    	int s1=0,s2=0;
    	
    	for(int j=0;j<s.length()/2;j++)
    	{
    		s1+=s.at(j)-48;
    	}
    	for(int j=s.length()/2;j<s.length();j++)
    	{
    		s2+=s.at(j)-48;
    	}
    	if(s1==s2)
    	    cout << "YES" << endl;
    	else
    	    cout << "NO" << endl;
    }
}
