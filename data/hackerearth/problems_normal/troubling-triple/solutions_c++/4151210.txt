#include<bits/stdc++.h>
#define ll long long int
using namespace std;
int main()
{
		ll n,t,k,i,j,ans,ele;
		vector<ll> v;
		vector<ll>::iterator itr;
		cin>>n>>k;
		for(i=0;i<n;i++)
		{
			cin>>ele;
			v.push_back(ele);
		}
		sort(v.begin(),v.end());
		//cout<<"working"<<endl;
		ll cnt=0,val,tmp;
		ans=0;
		for(i=0;i<n;i++)
		{
			for(j=i+1;j<n;j++)
			{
				val=(v[i]*v[j]);
				if(val>k)
				{
					break;
				}
				ll trplt=(k/val);
				itr=upper_bound(v.begin(),v.end(),trplt);
				tmp=itr-v.begin();
				tmp--;
				if(tmp<j)
				{
					continue;
				}
				ans=ans+(tmp-j);
				//cout<<tmp<<" "<<ans<<endl;
			}
		}
		cout<<ans<<endl;
	return 0;
}
