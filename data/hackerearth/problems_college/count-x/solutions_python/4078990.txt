t=input("")
while t>0 and t<31:
    c=0
    k=[]
    s=raw_input("")
    l=len(s)
    for i in range(0,l,1):
        if s[i]=='X':
            c=c+1
            if s[i+1]=='-':
                k.append(c)
        elif s[i]=='-':
            c=0
            if i+1<l: 
                if s[i+1]=='-':
                    k.append(c)
    print ''.join(map(str,k))
    t=t-1
