#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
typedef long long ll;
ll a[1000009];
int main()
{
	ll n,i;
	scanf("%lld",&n);
	for(i=0;i<n;i++)
	scanf("%lld",&a[i]);
	sort(a,a+n);
	ll t=0,x;
	ll prev=a[0];
	ll count=1;
	for (i = 1; i < n; ++i)
	{
		if(prev==a[i])
			count++;
		else
		{
			x=(count*(count-1));
			t+=x;
			prev=a[i];
			count=1;
		}
	}
	x=n*(n-1);
	double ans= double(t)/x;
	ans=(1-ans)/2;
	printf("%.6f %.6f\n",ans,ans);
    return 0;
}
