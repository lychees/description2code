#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() 
{
    int t;
    long b,w,x,y,z,sum1,sum2,sum3;
    cin>>t;
    while(t--)
        {
        cin>>b>>w;
        cin>>x>>y>>z;
         sum1= b*x + w*y;
         sum2=x*(b+w)+ w*z;
         sum3=y*(b+w)+ b*z ;
        if(sum1<sum2 && sum1<sum3)
            cout<<sum1<<endl;
        else if (sum2<sum3)
            cout<<sum2<<endl;
       else         
            cout<<sum3<<endl;
    }
   
    return 0;
}
