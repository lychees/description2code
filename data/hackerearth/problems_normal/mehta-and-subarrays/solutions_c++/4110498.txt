#include<bits/stdc++.h>
using namespace std;
#define ll long long int
ll arr[1234567];
int main()
{
	//freopen("input.txt","r",stdin);
	//freopen("output1.txt","w",stdout);
	ll n,i,j,l,h,k,mid,ele;
	memset(arr,0,sizeof(arr));
	scanf("%lld",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%lld",&arr[i]);
		arr[i]+=arr[i-1];
	}
	/*for(i=0;i<=n;i++)
	{
		printf("%lld ",arr[i]);
	}*/
	l=1,h=n;
	ll maxi=0,val;	
	while(l<=h)
	{
		bool flag=0;
		mid=(l+h)/2;
		//printf("l =%lld h=%lld\n",l,h);
		for(i=1;i<=n-mid+1;i++)
		{
				j=i;
				k=i+mid-1;
				val=arr[k]-arr[j-1];
				if(val>=0)
				{
					flag=1;
					break;
				}
		}
		if(flag==1)
		{
			maxi=max(maxi,mid);
			//printf("max= %lld mid =%lld\n",maxi,mid);
			l=mid+1;
		}
		else
		{
			h=mid-1;
		}
	}
//	printf("\n%lld\n",maxi);
	if(maxi)
	{
		ll cnt=0;
		for(i=1;i<=n-maxi+1;i++)
		{
			j=i;
			k=i+maxi-1;
			val=arr[k]-arr[j-1];
			if(val>=0)
			{
				cnt++;
			}
		}
		printf("%lld %lld\n",maxi,cnt);
	}
	else
	{
		cout<<"-1"<<endl;
	}
//	fclose(stdout);
	return 0;
}
