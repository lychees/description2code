#include <iostream>
using namespace std;

int main()
{
    int t,n,i,j=0;
    cin >> t;
   	while(t--)
   	{
   		int ans=0; 
   		
   		cin >> n;
   		int a[n],inc[n],exc[n];
   		
   		for(i=0;i<n;i++)
   		{
   			cin  >> a[i];
   		}
   		
   		inc[0]=a[0];
   		exc[0]=0;
   		ans=max( max(inc[0],exc[0]) ,ans);
   		for(i=1;i<n;i++)
   		{
   			
   			inc[i]=	exc[i-1]+a[i];
   			exc[i]= max(inc[i-1],exc[i-1]);
   			
   			ans=max( max(inc[i],exc[i]) ,ans);
   		
   		}
   		cout << "Case "<< j+1 << ": " << ans <<endl;
   		j++;
   	}
    
}
