#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
    int n, ans, tmp;

    cin >> n;
    string s[n];
    ans = 0;
    for (int i = 0; i < n; i++) {
        cin >> s[i];
        ans += s[i].size();
    }

    sort(s, s + n);

    do {
        tmp = s[0].size();
        for (int i = 1; i < n; i++) {
            int j;
            for (j = min(s[i].size(), s[i - 1].size()); j > 0; j--) {
                bool b = true;
                for (int k = 0; k < j; k++) {
                    if (s[i][k] != s[i - 1][s[i - 1].size() - j + k]) {
                        b = false;
                        break;
                    }
                }
                if (b) {
                    break;
                }
            }
            tmp += s[i].size() - j;
        }
        ans = min(ans, tmp);
    } while (next_permutation(s, s + n));

    cout << ans << endl;
}
