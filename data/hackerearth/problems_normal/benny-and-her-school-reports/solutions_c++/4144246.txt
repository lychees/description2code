#include <iostream>
using namespace std;

typedef long long int ll;

int main()
{
    long T;
    cin>>T;
    while(T--){
    	ll a[100005],n,i,m,x,y,sum=0;
    	cin>>n>>m>>x;
    	for(i=0;i<n;i++){
    		cin>>a[i];
    		sum+=a[i];
    	}
    	y = x*(n+1)-sum;
    	
    	if(y<=0) cout<<"1"<<endl;
    	else if(y<=m) cout<<y<<endl;
    	else cout<<"Impossible"<<endl;
    }
    return 0;
}
