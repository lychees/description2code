noc = input()
while noc:
    noc-=1
    res = 0
    z,m,n = map(int,raw_input().split())
    for i in xrange(1, int(z** 0.5)+1):
        if z % i == 0:
            if i == z/i and ((1<= i <= m and 1<= z/i <= n) or ( 1<= i <= n and 1<= z/i <= m)):
                res+=1
            else:
                if 1<= i <= m and 1<= z/i <= n:
                    res = res +1
                if 1<= i <= n and 1<= z/i <= m:
                    res = res +1 
    print res
