#include <iostream>
using namespace std;

int main()
{
    int t,n,x;
    cin >> t;
    while(t--){
    	cin >> n >> x;
    	int a[n];
    	int sum = 0;
    	for(int i = 0; i < n; i++){
    		cin >> a[i];
    		sum += a[i];
    	}
    	if(sum <= x){
    		cout << "YES" << endl;
    		continue;
    	}
    	int dp[n][x+1];
    	for(int i = 0; i < n; i++){
    		for(int j = 0; j <= x; j++){
    			if(i == 0){
    				if(j >= a[i]){
    					dp[i][j] = a[i];
    				}else{
    					dp[i][j] = 0;
    				}
    				continue;
    			}
    			dp[i][j] = dp[i-1][j];
    			if(j >= a[i]){
    				dp[i][j] = max(dp[i][j], dp[i-1][j-a[i]] + a[i]);
    			}
    		}
    	}
    	if(dp[n-1][x] <= x && (sum - dp[n-1][x]) <= x){
    		cout << "YES" << endl;
    		continue;
    	}
    	cout << "NO" << endl;
    }
    return 0;
}
