#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin >> t;
	
	while (t--) {
		int n, k;
		cin >> n >> k;
		int i;
		int a[n];
		
		for (i = 0; i < n; i++) {
			cin >> a[i];
		}
		
		sort(a, a + n);
		
		int ans = 0;
		int mn = 999999999;
		int mx = -999999999;
		
		for (i = 0; i < n / 2; i++) {
			int x = a[i] + a[(n - 1) - i];
			if (x > mx)
				mx = x;
			if (x < mn)
				mn = x;
		}
		
		ans = abs(mx - mn);
		
		if (ans > k) {
			cout << ans << endl;
			cout << "No more girlfriends!\n";
		} else if (ans == k) {
			cout << ans << endl;
			cout << "Lucky chap!\n";
		} else {
			cout << ans << endl;
			cout << "Chick magnet Jhool!\n";
		}
	}
    return 0;
}
