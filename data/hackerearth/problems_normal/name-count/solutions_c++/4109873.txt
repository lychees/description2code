#include<bits/stdc++.h>

#define e 1000000007
#define MAX 1000006

#define pb push_back
using namespace std;

ll t,n,k,i,prd,ans;

ll pow(ll a, ll b, ll MOD) {
ll x = 1, y = a;
    while(b > 0) {
        if(b%2 == 1) {
            x=(x*y);
            if(x>MOD) x%=MOD;
        }
        y = (y*y);
        if(y>MOD) y%=MOD;
        b /= 2;
    }
    return x;
}
 
ll modInverse(ll a, ll m) {
    return pow(a,m-2,m);
}

ll kCn()
{
	if(n>k/2)
	n=k-n;
	prd=1;
	for(i=k;i>n;i--)
	{
		prd = prd * i % e * modInverse(k-i+1,e) % e;  
		//cout<<prd<<endl;
	}
	return prd;
}

ll fac(ll n)
{
	prd=1;
	rep(i,1,n+1)
	{
		prd*=i;
		prd%=e;
	}
	return prd;
}
