#include <bits/stdc++.h>

#define X first
#define Y second
#define INPUT freopen("jclash.inp","r",stdin)
#define OUTPUT freopen("jclash.out","w",stdout)
#define FOR(i,l,r) for(auto i=l;i<=r;i++)
#define REP(i,l,r) for(auto i=l;i<r;i++)
#define FORD(i,l,r) for(auto i=l;i>=r;i--)
#define REPD(i,l,r) for(auto i=l;i>r;i--)
#define ENDL printf("\n")
#define debug 1

typedef long long ll;
typedef pair<int,int> ii;

const int inf=1e9;
const int MOD=1e9+7;
const int N=1500005,M=5e5;

int n,m,x,b[N],pos[M+5];
char c;
void add(int x,int v){
    while (x<N){
        b[x]+=v;
        x+=x&-x;
    }
}
int get(int x){
    int ans=0;
    while (x){
        ans+=b[x];
        x-=x&-x;
    }
    return ans;
}
int main(){
    scanf("%d%d",&n,&m);
    FOR(i,1,n) {
        scanf("%d",&x);
        add(M+i,1);
        pos[x]=M+i;
    }
    int L=M+1,R=M+n;
    ll ans=0;
    while (m--){
        scanf("%d %c",&x,&c);
        int v=get(pos[x]);
        ans+=min(v-1,n-v);
        add(pos[x],-1);
        if (c=='l') pos[x]=--L;
        else pos[x]=++R;
        add(pos[x],1);
    }
    cout<<ans;
}
