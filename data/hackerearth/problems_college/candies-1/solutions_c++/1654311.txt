#include <iostream>
#define ll unsigned long long
using namespace std;

int main()
{
    ll t;
    cin>>t;
    while(t--)
    {
    	ll n,sum=0;
    	cin>>n;
    	for(ll i=0;i<n;i++)
    	{
    		ll a;
    		cin>>a;
    		sum+=a;
    		
    	}
    	if(sum%n==0) cout<<"YES"<<endl<<sum/n<<" "<<sum%n<<endl;
    	else cout<<"NO"<<endl<<sum/n<<" "<<sum%n<<endl;
    }
    
    
    return 0;
}
