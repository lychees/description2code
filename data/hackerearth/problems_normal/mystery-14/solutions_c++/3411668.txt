#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	
    	int n,temp,sum=0;
    	
    	cin >> n;
    	
    	for(int i=0;i<n;i++)
    	{
    		cin>>temp;
    		if(temp%2)sum+=temp;
    	}
    	
    	cout<<sum<<endl;
    	
    }
    
    
    return 0;
}
