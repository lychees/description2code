#include <bits/stdc++.h>
using namespace std;

const int size = 1e6 + 7;

int main(void) {

	int i = 0, j = 0;
	
	vector<long int> prefix(size);

	for(i = 2; i < size; i += 1) {
		if(!prefix[i]) {
			for(j = i; j < size; j += i) {
				int k = j;
				while (!(k % i)) {
					prefix[j] += 1;
					k /= i;
				}
			}
		}
	}

	for(i = 1; i < size; ++i)
		prefix[i] += prefix[i - 1];
	
	int T = 0;
	scanf("%d",&T);
	while(T--) {
		scanf("%d %d",&i,&j);
		printf("%d\n",prefix[j] - prefix[i - 1]);
	}

	return 0;
}
