#include<iostream>
#include<stdlib.h>
#include<algorithm>

using namespace std;

int main()
{
	int t, *sword, *animal, i, flag,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		flag = 0;
		sword = new int[n];
		animal = new int[n];
		for(i=0;i<n;i++)
		{
			cin>>sword[i];
		}
		for(i=0;i<n;i++)
		{
			cin>>animal[i];
		}
		sort(sword,sword+n);
		sort(animal,animal+n);
		
		for(i=0;i<n;i++)
		{
			if(sword[i]<=animal[i])
			{
				
				flag++;
				cout<<"NO"<<endl;
				break;
			}
		}
		if(flag==0)
		cout<<"YES"<<endl;		
	}
	return 0;
}
