#include <iostream>
#include <utility>
#include <algorithm>
#include <set>
using namespace std;
#define mp(l,r) make_pair((l),(r)) 
#define pb(x) push_back((x))
#define endl "\n"
int main()
    {
        int t;
        cin>>t;
    	while (t--)
        {
            int m, n;
            cin>>m>>n;
    	    vector <pair<int,int>> rng;
            for(int i=0;i<n;i++)
            {
                int l, r;
                cin>>l>>r;
                if(l<=r)
                {
                    rng.pb(mp(l,r));
                    rng.pb(mp(l+m,r+m));
                }
                else
                    rng.pb(mp(l,r+m));
            }
            sort(rng.begin(),rng.end());
            int i=0,valid=0;
           	set<int> s;
            bool flag = true;
            while(true)
            {
                if(s.empty())
                {
                    if(i==rng.size())
                        break;
                    else
                        valid = rng[i].first;
                }
    			while(i<rng.size() && rng[i].first==valid)
                {
                    s.insert(i);
                    i++;
                }
    			int ind=*s.begin();
                s.erase(s.begin());
    			if(!(valid>=rng[ind].first && valid<=rng[ind].second))
                {
                   flag = false;
                   break;
                }
                valid++;
            }
            if(flag)
            	 cout<<"YES"<<endl;
            else 
            	 cout<<"NO"<<endl;
        }
    }
