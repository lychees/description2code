#include <iostream>
using namespace std;

int main()
{
    int t=100000,n,i;
    while(t--)
    {
    	n=0;
    cin>>n;
    if(n)
    {
    	int a[n],flag=0;
    	for(i=0;i<n;i++) cin>>a[i];
    	for(i=n-1;i>=0;i--) {
    		if(a[i]%2==0) {
    			cout<<a[i]<<" ";
    			flag=1;
    		}
    	}
    	if(flag==0) cout<<"None to display";
    	
    }
    
    cout<<endl;
    }
    return 0;
}
