#include <iostream>
using namespace std;

#define LL long long int

bool df[10] = {false};
LL tp[20], sp[20];
int d[100];

int main()
{
    int t, k, i, j;
    LL ans, n;
    bool flag;
    scanf("%d", &t);
    df[2] = df[3] = df[5] = df[7] = true;
    tp[0] = sp[0] = 1;
    for(i = 1; i < 20; i++){
    	tp[i] = tp[i - 1] * 10;
    	sp[i] = sp[i - 1] * 6;
    }
    while(t--){
    	scanf("%lld", &n);
    	k = 0;
    	while(n > 0){
    		d[k] = n % 10;
    		n /= 10;
    		k++;
    	}
    	flag = false;
    	ans = 0;
    	while(k--){
    		for(j = 0; j < d[k]; j++){
    			if(flag | df[j]){
    				ans += tp[k];
    			}
    			else{
    				ans += tp[k] - sp[k];
    			}
    		}
    		flag |= df[d[k]];
    	}
    	if(flag == true){
    		ans++;
    	}
    	printf("%lld\n", ans);
    }
    return 0;
}
