#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
int main(){
     
        int t;
        cin >> t;
        while(t--)
      {
         int n;
         cin >> n;
           int a[n+1];
           for(int i = 1 ; i<= n ; ++i )
              cin >> a[i];
             long long  incl = a[1];
            long long  excl = 0;
              long long  excl_new;
              for(int i = 2 ; i<= n; ++i ){
               excl_new = (incl > excl)? incl: excl;
              incl = excl + a[i];
              excl = excl_new;
              
              }
              cout << max(incl,excl)<<endl;
      }
      
  }
