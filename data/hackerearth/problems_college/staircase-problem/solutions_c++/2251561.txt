#include <iostream>
using namespace std;

int main()
{
    int t,n,ans;
    cin>>t;
    while(t--)
    {
    	ans=0;
    	cin>>n;
    	if(n==1||n==2)
    		cout<<1<<endl;
    	else
    	{
    		int i=2;
    		while(n>0)
    		{
    			ans++;
    			n-=i;
    			i++;
    		}
    		cout<<ans<<endl;
    	}
    }
}
