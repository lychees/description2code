#include <iostream>
using namespace std;
int dp[1000000][10]={0};

int main() {

int t;
        cin>>t;
        for(int i=0;i<t;i++)
        { int n;
            cin>>n;
           
            
            for(int j=0;j<10;j++)
                dp[1][j] = 1;
            
            for (int i=2;i<=n;i++)  // no of digits
            {
                for(int j=0;j<10;j++) //ending digit
                {
                    if(j==0 || (j==1 && i==2))
                        dp[i][j] = dp[i-1][j+1];
                    else if(j==9)
                        dp[i][j] = dp[i-1][j-1];
                    else
                        dp[i][j] = (dp[i-1][j-1]+dp[i-1][j+1]) % 1000000007;
                    
                }
            }
            
            int sum =0;
           
                for(int j=0; j<=9; j++)
                {	sum+=dp[n][j];
                  //  cout<<dp[n][j]<<"\t"<<"for:"<<j<<"\n";
                    sum%=1000000007;}
            
            cout<<sum<<endl;
            
        }
        return 0;
    }
