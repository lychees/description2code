#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	string str[t+1];
	string str1;
	int i;
	for( i=0;i<t;i++)
	{
		cin>>str1;
		sort(str1.begin(),str1.end());
		str[i] = str1;
	}
	str[t]="999";
	sort(str,str+t);
	int c=1,max =0;
	for(i=0;i<t;i++)
	{
		if(str[i] == str[i+1])
		{
			c++;
		}
		else 
		{
			if(max<c)
			{
				max = c;
			
			}
			c=1;
		}
	}
	cout<<max<<"\n";
}
