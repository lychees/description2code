#include <iostream>
#include <math.h>
using namespace std;
#define MAX 1000008

bool isPrime(long n){
	long range = sqrt(n);
	
	for(long i=2;i<=range;i++)
	{
		if(n%i==0)
			return false;
	}
	
	return true;
}
int main()
{
	bool fib[MAX];
	for(int i=1;i<=MAX;i++)
		fib[i]=false;
	fib[1]=true;
	long f1=1,f2=1,f3;
	while(true){
		f3=f1+f2;
		if(f3<=MAX)
			fib[f3]=true;
		else
			break;
		f1=f2;
		f2=f3;
		//cout<<f3<<endl;
	}
    int T;
    cin>>T;
    
    while(T>0){
    	T--;
    	int num;
    	cin>>num;
    	long sum =0;
    	for(int i=1;i<num;i++){
    		if(fib[i]==false){
    			sum+=i;
    		}
    	}
    	if(isPrime(sum))
    		cout<<"Tic"<<endl;
    	else{
    		if(sum%2==0)
    			cout<<"Tac"<<endl;
    		else
    			cout<<"Toe"<<endl;
    	}
    }
    return 0;
}
