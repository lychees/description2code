#include<bits/stdc++.h>
using namespace std;
int cnt[26];
int main()
{
	int tc;
	cin>>tc;
	while(tc--)
	{
		string str1,str2;
		cin>>str1>>str2;
		if(str1.length()!=str2.length())
		{
			puts("No");
		}
		else
		{
			memset(cnt,0,sizeof cnt);
			int len=str1.length();
			for(int i=0;i<len;++i)
			{
				cnt[str1[i]-'a']++;
			}
			//for(int i=0;i<26;++i)
			//	printf("%d ",cnt[i]);
			//printf("\n");
			for(int i=0;i<len;++i)
			{
				cnt[str2[i]-'a']--;
			}
			bool flag=true;
			for(int i=0;i<26;++i)
			{
				if(cnt[i]!=0)
				{
					flag=false;
					break;
				}
			}
			(flag==false)?puts("No"):puts("Yes");
		}
	}
	return 0;
}
