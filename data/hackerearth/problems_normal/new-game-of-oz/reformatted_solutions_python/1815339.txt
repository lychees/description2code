for _ in range(input()):
    inplen = input()
    mylist = sorted(map(int, raw_input().split()))
    turn = 0
    index = 0
    while index < inplen:
        if index + 1 < inplen:
            if abs(mylist[index]-mylist[index + 1]) == 1:
                index += 1
        turn += 1
        index += 1
    print turn
