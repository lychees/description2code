#include <iostream>
#include<stdio.h>
#include<algorithm>
#define MOD 10000007
using namespace std;
long dp[2097159][2]={0},n,a[1005];
void copy(long big)
{
    long i=0;
    for(i=0;i<=big;i++)
    {
        dp[i][0]=dp[i][1];
    }
}
int main()
{
    long i,j,k,curbig,newbig,x;
    cin>>n>>x;
    for(i=0;i<n;i++)
    {
        scanf("%ld",&a[i]);
    }
    sort(a,a+n);
    curbig=a[0];
    dp[a[0]][1]=1;
    for(i=1;i<n;i++)
    {
        copy(curbig);
        newbig=curbig;
        dp[a[i]][1]=(dp[a[i]][0]+1)%MOD;
        if(a[i]>newbig)
        {
            newbig=a[i];
        }
        for(j=0;j<=curbig;j++)
        {
            if(dp[j][0]>0)
            {
                dp[j^a[i]][1]=(dp[j^a[i]][1]+dp[j][0])%MOD;
                if((j^a[i])>newbig)
                {
                    newbig=(j^a[i]);
                }
            }

        }
        curbig=newbig;
    }
    cout<<"\n"<<dp[x][1];
    
}
