#include <iostream>
using namespace std;

int main()
{
    int t;int n;
    cin>>t;int flag;
    for(int i =0;i<t;i++)
    {
    	cin>>n;int a[n];flag = 0;
    	for(int j=0;j<n;j++)
    	{
    		cin>>a[j];
    	}
    	int k = 0;int m;
    	while (!flag)
    	{
    		if(a[k] == -1)
    		{
    			flag = -1;	
    		}
            else
            {
    			if(a[k] == -2)
    			{
    				flag = -2;
    			}
    			else
    			{	
    				m = k;
    				k = a[m];
    				a[m] = -2;
    			}
            }
    	}
    	if(flag==-1)
    	{
    		cout<<"Win"<<endl;
    	}
    	else
    	{
    		cout<<"Lose"<<endl;
    	}
    }
    return 0;
}
