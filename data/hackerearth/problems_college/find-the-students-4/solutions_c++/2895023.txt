#include<bits/stdc++.h>

using namespace std;

int arr[1002],nor[1002];

int main()
{
   int n;
   scanf("%d",&n);
   int i,j;
   for(i=0;i<n;i++)
   {
   	  scanf("%d",&arr[i]);
   	  nor[i]=arr[i];
   }
   sort(nor,nor+n);
   
   int k,count ,m,l;
   
   i=0;
   while(i<n)
   {
   	   
   	  k=nor[i];
   	  j=i+1;
   	  while(j<n && nor[j]==k)
   	  {
   	  	  j++;
   	  	  
   	  }
   	  
   	  for(l=i;l<n;l++) nor[l]-=k;
   	  
   	  
   	  count=0;
   	  for(l=0;l<n;l++)
   	  {
   	  	 if(arr[l]>0) 
   	  	 {
   	  	 	count++;
   	  	 	arr[l]-=k;
   	  	 	
   	  	 }
   	  	 
   	  	 
   	  }
   	  
   	  printf("%d\n",count);
   	  i=j;
   	  
   	  
   	  
   }
   
   
   
}
