#include <bits/stdc++.h>
using namespace std;
#define ll long long
bool palin(string str)
{
	ll int n=str.size();
	ll int i,j;
	i=0;j=n-1;
	while(i<=j)
	{
		if(str[i++]!=str[j--])
		{
			return 0;
			break;
		}
	}
	return 1;
}
