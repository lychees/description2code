int t;
long long int sum(int a){
	if (a == 0){
		return 0;
	}
	return (1LL + a)*a / 2LL;
}
long long int summ(int a){
	if (a == 0){
		return 0;
	}
	long long int num = a / 2LL;
	num++;
	return (1LL + a)*num / 2LL;
}
long long int solve(long long int a){
	long long int mint = 0;
	long long int maxt = INT_MAX;
	while (mint + 1LL < maxt){
		long long int mid = (mint + maxt) / 2LL;
		if (sum(mid) >= a){
			maxt = mid - 1LL;
		}
		else{
			mint = mid;
		}
	}
	if (mint != maxt){
		if (sum(maxt) < a){
			mint = maxt;
		}
		else{
			maxt = mint;
		}
	}
	long long int val = mint;
	if (val % 2 == 0){
		val--;
	}
	long long int ans = summ(val);
	maxt = a - sum(maxt);
	mint++;
	if (maxt % 2){
		ans += mint;
	}
	return ans;
}
int main(){
	scanf("%d", &t);
	while (t--){
		long long int l, r;
		scanf("%lld%lld", &l, &r);
		long long int ans = solve(r);
		if (l){
			ans -= solve(l - 1LL);
		}
		printf("%lld\n", ans);
	}
	return 0;
}
