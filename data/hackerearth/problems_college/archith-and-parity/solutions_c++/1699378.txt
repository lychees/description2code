#include<iostream>
using namespace std;
int main()
{
    int j,n;
    cin>>n;
    for(j=0;j<n;j++)
    {
        long int i,v,a[32],f=0;
        cin>>v;
        for (i = 0; i < 32; ++i)
        {
        a[i] = v & (1 << i) ? 1 : 0;
        }
        for(i=0;i<32;i++)
        {
           if(a[i]==1)
                f++;
        }
        if((f%2)==0)
        {
            cout<<"even"<<endl;
        }
        else
            cout<<"odd"<<endl;

    }
    return 0;
}
