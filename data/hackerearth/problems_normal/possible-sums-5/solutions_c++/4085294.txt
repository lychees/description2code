#include <bits/stdc++.h>
using namespace std;
#define maxl 10005
int a[maxl];

int main()
{
	int t,n,i,j,k;
	cin >> t;
	while(t--){
		memset(a,-1,sizeof(a));
		cin >> n;
		int arr[n];
		for(i=0;i<n;i++){
			cin >> arr[i];
		}
		vector<int> v;
		a[0]=1;
		sort(arr,arr+n);
		for(i=0;i<n;i++){
			k=arr[i];
			for(j=0;j<maxl;j++){
				if(a[j]==1) v.push_back(k+j);
			}
			for(j=0;j<(int)v.size();j++){
				a[v[j]]=1;
			}
			v.clear();
		}
		int cnt=0;
		for(i=0;i<maxl;i++){
			if(a[i]==1) cnt++;
		}
		cout << cnt << endl;
	}
    return 0;
}
