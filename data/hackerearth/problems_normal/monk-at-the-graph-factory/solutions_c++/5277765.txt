#include<bits/stdc++.h>
#define lli long long int
#define MOD 1000000007
#define TEST lli t ; cin >> t ; while(t--)
using namespace std;

int main(){
	ios_base::sync_with_stdio(0);

	lli n ; cin >> n ;
	lli sumDeg = 0;
	
	for(int i = 0 ; i < n ; i++){
		lli x ; cin >> x;
		sumDeg += x;
	}

	lli edge = sumDeg/2;

	if(edge == n-1) cout << "Yes" << endl;
	else cout << "No" << endl;
	return 0;
}
