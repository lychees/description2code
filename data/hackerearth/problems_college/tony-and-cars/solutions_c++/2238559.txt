#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int t,c;
    long long int n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	int i=1;
    	long long k=0;
    	while(1)
    	{
    		int z=0;
    		long long int x=pow(10,i);
    		if(n>x)
    		k=k+9*pow(10,i-1)*i;
    		else
    		{
    			if(n==x && ++z)
    			n--;
    			k=k+i*(1+n%x - pow(10,i-1));
    			if(z)
    			k+=i+1;
    			break;
    		}
    		i++;
    	}
    	cout<<k<<endl;
    }
    return 0;
}
