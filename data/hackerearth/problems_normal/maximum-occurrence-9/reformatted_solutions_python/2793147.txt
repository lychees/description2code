r = raw_input()
count = {}
max_len = 0
max_char = ""
for i in r:
    if i not in count:
        count[i] = 1
    else:
        count[i] += 1
for i in count:
    if count[i] > max_len:
        max_char = i
        max_len = count[i]
    elif count[i] == max_len:
        if ord(i) < ord(max_char):
            max_char = i
print("%c %d" % (max_char, max_len))
