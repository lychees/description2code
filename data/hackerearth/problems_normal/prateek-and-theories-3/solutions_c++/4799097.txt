#include <bits/stdc++.h>
#define MAX 1000005
#define inf 10000000000000000
#define ll long long
#define MOD 1000000007
#define s(x) scanf("%d", &x);

using namespace std;

ll dp[MAX];
ll a[MAX];
string s;


int main() {
    int t;
    cin >> t;
    while(t--) {
      memset(a, 0 , sizeof(a));
      int n;
      cin >> n;
      vector<pair<int,int> >v;
      map<int, int > mp;
      for(int i = 0; i< n; i++) {
            int x, y;
            cin >> x >> y;
            v.push_back(make_pair(x , y));
            mp[x] = 0;
            mp[y] = 0;
      }
      map<int , int > :: iterator it = mp.begin();
      int k = 1;
      while(it != mp.end()) {
        it->second=k++; it++;
      }
      for(int i = 0; i < v.size(); i++) {
        int first = mp[v[i].first];
        int second = mp[v[i].second];
        a[first]++;
        a[second]--;
      }
      int mx = 0;
      for(int i = 1; i < MAX; i++) {
        a[i] += a[i-1];
        if( a[i] > mx) mx = a[i];
      }
      cout << mx << endl;
    }

}
