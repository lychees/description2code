int findsubstr(char str[], int n,char s,char e){
	 int r=0;

	if(s!=e){
	for(int i=0;i<n;i++){
		for(int j=i;j<n;j++){
			if(str[i]==s && str[j+1]==e)
			r++;
		}
	}

	for(int i=0;i<n;i++){
		for(int j=i;j<n;j++){
			if(str[i]==e && str[j+1]==s)
			r++;
		}
	}
	}
	else{
		for(int i=0;i<n;i++){
		for(int j=i;j<n;j++){
			if(str[i]==s && str[j+1]==e)
			r++;
		}
	}
	}
	return r;
}
