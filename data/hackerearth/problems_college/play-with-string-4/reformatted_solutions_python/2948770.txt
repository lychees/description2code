n = int(raw_input())
for j in range(n):
    s = raw_input()
    count = 0
    r = "".join(list(reversed(s)))
    for i in range(1, len(s)):
        if abs(ord(s[i])-ord(s[i-1])) != abs(ord(r[i])-ord(r[i-1])):
            print "Not Funny"
            count = 0
            break
        else:
            count = 1
            continue
    if count == 1:
        print "Funny"
