t = int(raw_input())
for i in range(t):
    n = int(raw_input())
    lst = [0]*26
    for j in range(n):
        s = list(set(raw_input()))
        for k in range(len(s)):
            a = ord(s[k])-97
            lst[a] = lst[a]+1
    m = max(lst)
    p = [x for x, y in enumerate(lst) if y == m]
    print(chr(p[0]+97))
