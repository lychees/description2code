#include<iostream>
using namespace std;
long long t,x,a,b;
long long fact(long long x)
{
    long long r=1;
    for(int i=1;i<=x;i++)
    {
        r=r*i;
    }
    return r;
}
int main()
{
    cin>>t;
    while(t--)
    {
        cin>>a>>b;
        x=0;
        for(int i=a;i<=b;i++)
        {
            x=x+fact(i);
        }
        int r=x%10;
        cout<<r<<endl;
    }


}
