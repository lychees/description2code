n=int(raw_input())
while n>0:
    s=str(raw_input())
    m=s[::-1]
    b=1
    for i in range(1,len(s)):
        d1=abs(ord(s[i])-ord(s[i-1]))
        d2=abs(ord(m[i])-ord(m[i-1]))
        if(d1!=d2):
            print "Not Funny"
            b=0
            break
    if(b==1):
        print "Funny"
    n-=1
