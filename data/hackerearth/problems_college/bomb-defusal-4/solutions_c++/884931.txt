#include <iostream>
using namespace std;
long mod(long x)
{
	if(x>0) return  x;
	return -x;
}
int main()
{
    int T,P,w;long x,y,u,v;
    cin>>T;
    for(int i=0;i<T;i++)
    {
    	cin>>P>>x>>y>>u>>v;
    	if((x-u+y-v)%2) w=1;
    	else
    	if(x==u && y==v ) {if(P>0) w=0; else w=1;}
    	else if(mod(x-u)==mod(y-v))
    	{
    		if(P>1) w=0; else w=1;
    	}
    	else
    	{
    		if(P>2) w=0; else w=1;
    	}
    	if(w==1) cout<<"Terrorists Win !"<<endl; else cout<<"Counter-Terrorists Win !"<<endl;
    }
    return 0;
}
