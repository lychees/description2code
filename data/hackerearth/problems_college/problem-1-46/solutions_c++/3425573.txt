#include <bits/stdc++.h>
using namespace std;
int dp[100001][26];
int check(char c,int j) {
	if(c-'a'==j) return 1;
	return 0;
}
int main()
{
    int n,i,j;
    cin>>n;
    getchar();
    char str[100002];
    cin>>str;
    memset(dp,0,sizeof(dp));
    for(i=0;i<=n;i++) {
    	for(j=0;j<26;j++) {
    		dp[i+1][j]=dp[i][j]+check(str[i],j);
    		//cout<<dp[i+1][j]<<" ";
    	}
    	//cout<<endl;
    }
    int q;
    cin>>q;
    while(q--) {
    	int l,r;
    	cin>>l>>r;
    	int flg=0;
    	for(j=0;j<26;j++) {
    		if((dp[r][j]-dp[l-1][j])%2!=0) {
    			flg++;
    			//char c='a'+j;
    			//cout<<c<<endl;
    		}
    	}
    	//cout<<flg<<"\n";
		if(r-l+1==1) {
			cout<<"NO\n";
			continue;
		}
    	if(flg==0 &&(r-l+1)%2==0) cout<<"YES\n";
    	else if(flg==1 &&(r-l+1)%2!=0) cout<<"YES\n";
    	else if(flg==2 && (r-l+1)%2==0) cout<<"YES\n";
    	else cout<<"NO\n";
    }
    return 0;
}
