#include <iostream>
using namespace std;

int main()
{
    int t, n, tot, len;
    string s1, s2;
    cin >> t;
    while (t--) {
    	cin >> s1 >> s2 >> n;
    	len = s1.length();
    	tot = 0;
    	for (int i=0;i<len;i++) {
    		if (s1[i]!=s2[i]) tot++;
    	}
    	if (n>=tot && (n-tot)%2==0) cout << "Yes" << endl;
    	else cout << "No" << endl;
    }
    
    return 0;
}
