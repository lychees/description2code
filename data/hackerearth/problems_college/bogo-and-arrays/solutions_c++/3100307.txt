#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,k,ff=0;
    	cin>>n>>k;
    	int a[n];
    	for(int i=0;i<n;i++)
    	cin>>a[i];
    	long long s=0;
    	for(int i=0;i<n&&ff==0;i++)
    	{
    		if(a[i]>k)
    		{
    			s+=(n-i);
    			//cout<<i<<" "<<s<<endl;
    		}
    		else
    		{
    			int j,f=0;
    			for(j=i+1;j<n&&f==0;j++)
    			{
    				if(a[j]>k)
    				f=1;
    			}
    			if(f==1)
    			{
    				j--;
    				s+=(n-j);
    			}
    			else
    			ff=1;
    			//cout<<i<<" "<<s<<endl;
    		}
    	}
    	cout<<s<<endl;
    }
    return 0;
}
