#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	int a[27]={0};
	string s;
	while(t--)
	{
		cin>>s;
		memset(a,0,sizeof(a));
		for(int i=0;s[i];i++)
		{
			a[s[i]-'a']++;
		}
		int flag=0;
		for (int i = 0; i < 26; ++i)
		{
			if(a[i]>=2)
				{
					cout<<"Yes"<<endl;
					flag=1;
					break;
				}
		}
		if(flag==0)cout<<"No"<<endl;
	}
}
