using namespace std;
int gcd2( int a , int b )
{
  if( b == 0 && a != 1 ) return 1000000000;
  if( !b ) return -1;
  return a/b + gcd2( b , a % b );
}
int main()
{
  int i , j , n ;
  int t ;
  cin >> t;
  while(t--)
  {
    cin >> n;
    int ans = 1000000000;
    if( n == 1 ) cout << "0\n";
    else
    {
      for( int i = 1 ; i < n ; i++ )
        ans = min(ans , gcd2(n,i) );
      cout << ans << endl;
    }
  }
  return 0;
}
