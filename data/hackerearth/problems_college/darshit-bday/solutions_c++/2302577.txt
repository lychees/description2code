#include <iostream>
using namespace std;

int main()
{
//    cout << "Hello World!" << endl;
	int t;
	cin >> t;
	while(t--){
		long long int b,w,x,y,z;
		cin >> b >> w >> x >> y >> z;
		if(x+z < y){
			y = x + z;
		}
		else if(y+z < x){
			x = y+z;
		}
		cout << b*x + w*y << endl;
	}
    return 0;
}
