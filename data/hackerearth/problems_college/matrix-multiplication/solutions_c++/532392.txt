#include<bits/stdc++.h>
using namespace std;
int main()
{
    int T;
    scanf("%d",&T);
    while(T--)
    {
        int x1,x2,y1,y2,ans;
        scanf("%d %d %d %d",&x1,&x2,&y1,&y2);
        if(x1==y2 && x2==y1)
            ans=2;
        else if(x2==y1 || x1==y2)
            ans=1;
        else
            ans=0;
        printf("%d\n",ans);
    }
    return 0;
}
