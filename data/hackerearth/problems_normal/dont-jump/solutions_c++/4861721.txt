#include<bits/stdc++.h>
using namespace std;

int main(){
	long int temp,n;

	while(scanf("%ld",&n)!=EOF) {
		string a="";
		string s;
		vector<long int> p;
		p.clear();
		long int temp,i;
		getchar();
		getline(cin ,s);
		stringstream ss(s);
		while(ss >> temp) {
			p.push_back(temp);
		}
		
		long int curr=1;
		stack<long int> st;
		while(!st.empty())
			st.pop();
		long int flag=0;
		for(i=0;i<p.size();i++) {
			 
			while(curr<=p[i]) {
				st.push(curr);
				curr++;
				a+='C';
			}

			if(st.empty() || st.top() < p[i])
			{
				flag=1;break;
			}

			while(!st.empty() && st.top() > p[i]) {
				st.pop();
				a+='K';
				if(st.empty() || st.top()<p[i]) {
					flag=1;
					break;
				}
			}

			if(flag==1)
				break;
			st.pop();
			a+='K';
		}
		if(flag==1)
			cout << "-1" << endl;
		else
			cout << a << endl;
	}

	return 0;
}
