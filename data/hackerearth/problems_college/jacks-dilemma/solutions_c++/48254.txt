#include <stdio.h>
#include<string.h>
int main() {
	int t,n,J,Lift;
	char C;
	scanf("%d",&t);
	while(t--)
	{
		int time=0;
		scanf("%d %d %d %c",&n,&J,&Lift,&C);
		int stairs=10*(J);
		if(C=='U')
		{
			time=5*(n-Lift)+5*n+20+10;
			if(J==n)
				time-=10;
		}
		else
		{
			if(Lift>=J)
			{
				time+=5*(Lift)+20;  	
			}
			else
			{
				time+=5*(Lift)+10+20+5*n+10+5*n; 
			}
		}
		int temp;
		if(time<=stairs)
			printf("Lift %d\n",time);	
		else
			printf("Stairs %d\n",stairs);
		getchar();
	}
	return 0;
}
