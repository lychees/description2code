#include<bits/stdc++.h>
#define ll long long
#define mod 1000000007
ll power(ll a, ll b)
{
	ll res = 1, tmp = a;
	while(b)
	{
		if(b&1)
		{
			res = res * tmp;
			if(res >= mod)
				res %= mod;
		}
		tmp = tmp * tmp;
		if(tmp >= mod)
			tmp %= mod;
		b = b >> 1;
	}
	return res;
}
using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		set<ll> s;
		ll n, i, a;
		scanf("%lld",&n);
		for(i=0;i<n;i++)
		{
			scanf("%lld",&a);
			s.insert(a);
		}
		ll ans  = power(n,s.size());
		printf("%lld\n",ans);
	}
	
	return 0;
}
