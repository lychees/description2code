N = int(raw_input())
calling = map(int,raw_input().split())
ideal = map(int,raw_input().split())
count = 0
while len(ideal) != 0:
    if calling[0] == ideal[0]:
        calling.pop(0)
        ideal.pop(0)
    else:
        count += 1
        temp = calling.pop(0)
        calling.append(temp)

print (count) + N
