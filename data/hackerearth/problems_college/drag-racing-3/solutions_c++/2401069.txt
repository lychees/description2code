#include <iostream>
#include <map>
#include <string>
#include <set>
using namespace std;

int main()
{
	
	
	int n;
	cin>>n;
	long long m = 1<<n;
	m=m-1;
	string a;
	string b;
	set<string> st;
	set<string> los;
	set<string>::iterator it;
	while(m--)
	{
		cin>>a;
		cin>>b;
		if(los.empty() || los.find(a)==los.end())
		{
			st.insert(a);
		}
		if(st.find(b)!=st.end())
		{
			
			st.erase(st.find(b));
			
			
		}
		los.insert(b);
	}
	it = st.begin();
	cout<<*it<<endl;
    return 0;
}
