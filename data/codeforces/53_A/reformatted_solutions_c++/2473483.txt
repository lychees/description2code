#include <algorithm>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>
#include <string>
using namespace std;
string x;
string y, z;
int n;
int main() {
  cin >> x >> n;
  bool f = 0;
  for (; n--;) {
    cin >> y;
    if (y.find(x) == 0) {
      if (!f)
        z = y, f = 1;
      if (z > y)
        z = y;
    }
  }
  cout << max(z, x) << endl;
}
