typedef long long ll;
const ll mod=1e9+7;
const int INF=0x7f7f7f7f;
const double eps = 1e-6;

int main() {
	string s; cin >> s;

	ll ans = 0,temp = 0,semi = 0;
	for (int i = 0; s[i]; i ++) {
		if(s[i] == ';') {
			semi ++;
			ans = (ans + temp) % mod;
		}
		else temp = (2ll * temp + semi) % mod;
	}
	cout << ans << endl;
	return 0;
}
