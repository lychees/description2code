#include <iostream>
using namespace std;

int main() {
	int t, n, three, five;
	cin >> t;
	for ( int i = 0; i < t; i++ )
	{
		cin >> n;
		five = n;
		three = 0;
		while ( three <= n)
		{
			if ( five % 3 == 0 && three % 5 == 0 )
			{
				if ( five )
				{
					for ( int j = 0 ; j < five; j++ )
				    {
					cout << 5;
				    }
				}
		        if ( three )
		        {
		        	for ( int j = 0; j < three; j++ )
				    {
					cout << 3;
				    }
		        }

				cout << endl;
				break;
			}
			else 
			{
				five --;
				three ++;
			}
		}
		if ( five % 3 || three % 5 )
		{
			cout << -1 << endl;
		}
	}
	return 0;
}
