#include<iostream>
using namespace std;
int main(){
	int t,r,g,b,i;
	bool p=true,q=true,s=true;
	int a[8];
	for(i=0;i<8;i++)
		a[i]=0;
	cin>>t>>r>>g>>b;
	for(i=0;i<t;i++){
		if(i%r==0)
		p=!p;
		if(i%g==0)
		q=!q;
		if(i%b==0)
		s=!s;
		if(p==true&&q==false&&s==false)
		++a[0];
		else if(p==false&&q==true&&s==false)
		++a[1];
		else if(p==false&&q==false&&s==true)
		++a[2];
		else if(p==true&&q==true&&s==false)
		++a[3];
		else if(p==false&&q==true&&s==true)
		++a[4];
		else if(p==true&&q==false&&s==true)
		++a[5];
		else if(p==true&&q==true&&s==true)
		++a[6];
		else
		++a[7];
	}
	for(i=0;i<8;i++)
	cout<<a[i]<<" ";
	return 0;
}
