int double_string(char * arr1,int len)
{	
	int count=0;
	for(int j=0;j<len-1;j++)
    	{
    		for(int k=j+1;k<len;k++)
    		{
    			if(arr1[j]==arr1[k]) count++;
    		}
    		
    	}
    		if(count>0)return 1;
    			else
    			return 0;
}
