#include <bits/stdc++.h>
using namespace std;
#define ll long long int
#define MOD 1000000007
int main(){
ll t,l,b,d,x,y,ans;
cin>>t;
while(t--){
    cin>>l>>b>>d;
    x = l/d;
    x = x%MOD;
    y = b/d;
    y = y%MOD;
    ans = (x*y)%MOD;
    cout<<ans<<endl;
}
return 0;
}
