int t , x , y , n , w , P1 , P2;
double dp[1005][1006];
int memo[1005][1005];
double prob1 , prob2;
double  solve(int idx , int points){
    if(idx == n || points >= w){
        return (points >= w);
    }
    if(memo[idx][points]!=-1)return dp[idx][points];
    memo[idx][points] = 1;
    double res1 = prob1*solve(idx+1,points+x) + (1-prob1)*solve(idx+1,points);
    double res2 = prob2*solve(idx+1,points+y) + (1-prob2)*solve(idx+1,points);
    return dp[idx][points] = max(res1,res2);
}
int main(){

	int i , j , k ;
    scanf("%d",&t);
    while(t--){
        scanf("%d %d %d %d %d %d",&x,&y,&n,&w,&P1,&P2);
        memset(dp,0,sizeof(dp));
        memset(memo,-1,sizeof(memo));
        prob1 = (P1/100.0);
        prob2 = (P2/100.0);
        double ans = solve(0,0)*100.0;
        cout << setprecision(6) << fixed << ans << endl;
    }
	return 0;
}
