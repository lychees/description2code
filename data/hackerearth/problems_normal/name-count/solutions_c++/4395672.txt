#include <iostream>
#define MAX 1000000007
using namespace std;

long long fact(long long number) {
	if (number == 0 || number == 1) {
		return 1;
	}
	else return (number*fact(number - 1)) % MAX;
}

long long power(long long x, long long n) {
	if (n == 0)
		return 1;
	else if (n == 1)
		return x;
	long long mid = power(x, n / 2);

	if (n % 2 == 0)
		return (mid*mid) % MAX;
	else
		return (((mid*mid) % MAX)*x) % MAX;
}

long long permute(long long n, long long r) {
	return ((fact(n) % MAX)*((power(fact(n - r), MAX - 2)) % MAX)) % MAX;
}

int main()
{
	int t, n, k;
	cin >> t;
	while (t--) {
		cin >> n;
		cin >> k;
		cout << permute(k, n) << endl;
	}

	getchar();
	return 0;
}
