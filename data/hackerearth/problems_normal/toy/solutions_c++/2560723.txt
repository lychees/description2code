#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int main()
{
    int N;
    cin>>N;
    int a;
    vector<int>m;
    for(int i = 0;i<N;i++)
    {
        cin>>a;
        m.push_back(a);
    }
    sort(m.begin(),m.end());

    int count = 1;
    int value = m[0];
    for(int i = 1 ; i < m.size() ;i++)
    {
        if(m[i] > value + 4)
        {
            count++;
            value = m[i];
        }
    }
    cout<<count<<endl;

    return 0;
}
