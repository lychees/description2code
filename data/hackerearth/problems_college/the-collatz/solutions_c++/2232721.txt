#include<bits/stdc++.h>
using namespace std;
int main()
{
long  long	int t;
	cin>>t;
	while(t--)
	{
		long long int n;
	long long int esum=0,osum=1;
		long long int k,k1,c=0;
		cin>>n;
		if(n==1)
		{
			cout<<"Odd Rules"<<'\n';
		}
		else if(n==2)
		{
			cout<<"Even Rules"<<'\n';
		}
		else{
		while(n!=1)
		{
			if(n%2==0)
			{
				esum=esum+n;
				n=n/2;
				c++;
			}
			else if(n%2==1)
			{
				osum=osum+n;
				n=3*n+1;
				c++;
			}
		}
		k=esum%c;
		k1=osum%c;
		if(k>k1)
		{
                      cout<<"Even Rules"<<'\n';
		}
		else if(k==k1)
		{
		  cout<<"Tie"<<'\n';
		}
		else
		{
                        cout<<"Odd Rules"<<'\n';
		}}
	}
	return 0;
}
