#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long int n,i,ca=0,cb=0,x;
    cin>>n;
    long long int a[n],b[n];
    for(i=0;i<n;i++)
    {
        a[i]=0;
        b[i]=0;
    }
    for(i=0;i<n;i++)
    {
       cin>>x;
       if(x<0)
       {
           b[i]=x*-1;
           
       }
       else if(x>0)
            {
            a[i]=x;
            }
    }
    sort(a,a+n);
    sort(b,b+n); 
    unsigned long long int ans1,ans2,ans;
    ans1=a[n-1]*a[n-2];
    ans2=b[n-1]*b[n-2];
    ans=max(ans1,ans2);
    cout<<ans;
}
