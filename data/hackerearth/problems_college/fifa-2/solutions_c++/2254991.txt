#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	int home[100005]={0},away[100005]={0};
	int ans[n+1];
	for(int i=1;i<=n;i++)
		ans[i]=(n-1);
	pair<int,int> inp[n+1];
	for(int i=1;i<=n;i++)
	{
		cin>>inp[i].first>>inp[i].second;
		home[inp[i].first]++;
		away[inp[i].second]++;
	}
	for(int i=1;i<=n;i++)
	{
		ans[i]+=(home[inp[i].second]);
	}
	for(int i=1;i<=n;i++)
	{
		cout<<ans[i]<<" "<<(2*(n-1))-ans[i]<<endl;
	}
	return 0;
}
