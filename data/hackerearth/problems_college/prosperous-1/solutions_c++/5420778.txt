#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
int dp[10][100005][2];
int degree;
int rec(int prev,int sum,int flag)
{
//cout<<"HELLO 2 "<<prev<<" "<<sum<<"\n";
  if(sum<0)
    return 0;
  else if(sum==0)
  {
  if(flag==1)
    return 1;
    else
    return 0;
    }
  if(dp[prev][sum][flag]!=-1)
    return dp[prev][sum][flag];
    int ans=0;
  for(int i=max(1,(prev-degree));i<=min(9,(prev+degree));i++)
  {
 // cout<<"HELLO "<<i<<" "<<sum<<endl;
   if(i<prev)
    continue;
  if((abs(prev-i))==degree)
  ans=(rec(i,sum-i,1)+ans)%mod;
    else
    ans=(rec(i,sum-i,flag)+ans)%mod;
  }
  return dp[prev][sum][flag]=ans;
}
int main()
{
  int t;
  cin>>t;
  int s;
  for(int i=1;i<=t;i++)
  {
    cin>>s>>degree;
  int sum=0;
  int a=0;

for(int i=1;i<=9;i++)
  {
      memset(dp,-1,sizeof(dp));
      if(degree!=0)
    a=rec(i,s-i,0);
    else
        a=rec(i,s-i,1);
   //if(i==1)
  //cout<<i<<" "<<a<<endl;
  sum=(sum+a)%mod;
  }


  cout<<"Case #"<<i<<": "<<sum<<"\n";
}
}
