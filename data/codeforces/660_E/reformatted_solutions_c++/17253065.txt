
#define Modulo (1000000007)

#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstring>
#include <deque>
#include <iostream>
#include <map>
#include <queue>
#include <set>
#include <stack>
#include <string>
#include <utility>
#include <vector>
using namespace std;

int n, m;
long long Ans, tmp, l;

int main() {
  Ans = 1ll;
  scanf("%d%d", &n, &m);
  for (int i = 0; i < n; i++) {
    l = Ans;
    Ans = ((2ll * l - tmp) % (long long)Modulo + (long long)Modulo) *
          (long long)m % (long long)Modulo;
    tmp = (l + tmp * ((long long)m - 1ll)) % (long long)Modulo;
  }
  printf("%I64d\n", Ans);
  return 0;
}
