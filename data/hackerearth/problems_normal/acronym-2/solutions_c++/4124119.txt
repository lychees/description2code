#include<bits/stdc++.h>
using namespace std;

int main()
{
    int k,n;
    string s;
    cin>>k;
    multiset<string> ms;

    for(int i=0;i<k;i++)
    {
        cin>>s;
        ms.insert(s);
    }

    cin>>n;
    vector<char> res;

    for(int i=0;i<n;i++)
    {
        cin>>s;

        if(ms.find(s)==ms.end())
        {
            //cout<<s<<endl;
            res.push_back(s[0]-32);
        }
    }

    for(int i=0;i<res.size()-1;i++)
        printf("%c.",res[i]);
    printf("%c\n",res[res.size()-1]);
}
