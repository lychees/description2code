#include <iostream>
#include <string>
#include <string.h>


using namespace std;
int main()
{
    int t,i,n;
    cin>>t;
    bool a[10009];
    a[1]=true;
    a[2]=false;
    a[3]=true;
    a[4]=true;
    int highest=4;
    while(t--)
    {
        cin>>n;
        if(n<=highest)
        {
            if(a[n]==true)
            cout<<"Yes"<<endl;

            else
            cout<<"No"<<endl;

        }

        else
        {
            for(i=highest+1;i<=n;i++)
            {
                a[i]= (a[i-1]==false) || (a[i-4]==false);
            }
            highest=n;
             if(a[n]==true)
            cout<<"Yes"<<endl;

            else
            cout<<"No"<<endl;


        }

    }
}
