#include<iostream>
using namespace std;
#define MOD 10000007
//-------------------------------------

long long dp[1000010];
int main()
{
	int t;
	cin>>t;
		 dp[0]=0;
		 dp[1]=1;
		 dp[2]=1;
		 dp[3]=2;
		 dp[4]=4;
		 for(int i=5;i<1000010;i++)
	 	{
	 		dp[i]=dp[i-1]+dp[i-3]+dp[i-4];
	 		dp[i]%=MOD;
		 }
	while(t--)
	{
		 int n;
		 cin>>n;
		 cout<<dp[n]<<endl;
	}
}
