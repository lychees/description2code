#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int i = 0;
    unsigned long long int L = 0,B = 0,H = 0,T = 0;
    unsigned long long int temp = 0;
    scanf("%d",&T);
    for(i = 0;i < T;i++)
    {
    	scanf("%llu %llu %llu",&L,&B,&H);
    	temp = min(L,min(B,H));
    	if(temp%2 == 0)
    		printf("%llu.0\n",(temp/2));
    	else
    		printf("%llu.5\n",(temp/2));
    	temp = L = B = H = 0;
    }
    return 0;
}
