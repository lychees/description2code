#include<bits/stdc++.h>
#define llu long long int

using namespace std;

struct frnd
{
    llu M,F;
};

bool cmp(const frnd &a,const frnd &b)
{
    if(a.M<=b.M) return 1;
    else return 0;
}

int main()
{
    int n,x,flag;
    vector<int> A;
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {
        scanf("%d",&x);
        A.push_back(x);
    }
    scanf("%d",&x);
    sort(A.begin(),A.end());
    for(int i=0;i<=n-3;i++)
    {
        for(int j=i+1;j<=n-2;j++)
        {
            flag=binary_search(A.begin()+j+1,A.end(),x-A[i]-A[j]);
            if(flag) break;
        }
        if(flag) break;
    }
   if(flag) printf("True");
    else printf("False");
    return 0;
}
