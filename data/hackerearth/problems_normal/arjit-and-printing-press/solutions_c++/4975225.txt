#include <bits/stdc++.h>

using namespace std;

int main() {
    int T;
    cin >> T;
    for (int t = 0; t < T; t++) {
    	string W, R;
    	cin >> W >> R;
    	sort(R.begin(), R.end());
    	int j = 0;
    	vector<char> ans;
    	for (int i = 0; i < W.length(); i++) {
    		if (j < R.length() && R[j] < W[i]) {
    			ans.push_back(R[j]);
    			j++;
    		}
    		else
    			ans.push_back(W[i]);
    	}
    	for (int i = 0; i < ans.size(); i++) {
    		cout << ans[i];
    	}
    	cout << endl;
    }
    return 0;
}
