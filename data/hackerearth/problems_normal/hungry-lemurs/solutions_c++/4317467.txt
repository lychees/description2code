#include<iostream>
#include<algorithm>
#include<math.h>
using namespace std;
int main()
{
	int a,b,mini,i,t;
	int num,first,second;
	cin>>a>>b;
	t=(a/b+1)*b-a;
	mini=min(t,a%b);
	if(mini==0 || mini==1)
	cout<<mini<<"\n";
	else
	{
		for(i=1;i<mini;i++)
		{
			num=b+i;
			first=a%num;
			second=(a/num+1)*num-a;
			t=min(first,second)+i;
			if(t<mini)
			mini=t;
		}
		for(i=1;i<mini;i++)
		{
			num=b-i;
			first=a%num;
			second=(a/num+1)*num-a;
			t=min(first,second)+i;
			if(t<mini)
			mini=t;
		}
		cout<<mini<<"\n";
	}
}
