#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<vector>
using namespace std;
int main()
{ int n;
  cin>>n;
  vector <int> a(n);
  int i;
  for(i=0;i<n;i++)
    cin>>a[i];
  sort(a.begin(),a.end());
  vector <int> b;
  b.push_back(a[0]);
  b.push_back(a[1]);
  for(i=3;i<n;i=i+2)
    b.push_back(a[i]);
  if(n%2==0)
   i=n-2;
  else
   i=n-1;
  for(;i>=2;i=i-2)
    b.push_back(a[i]);
  int diff=-1;
  for(i=0;i<n-1;i++)
  { int v=b[i+1]-b[i];
    if(v<0)
      v=v*-1;
    if(v>diff)
       diff=v;
  }
  int v1=b[n-1]-b[0];
  if(v1<0)
     v1=v1*-1;
  if(v1>diff)
     diff=v1;
  cout<<diff;
  return 0;
}
