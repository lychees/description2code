#include <iostream>
using namespace std;
#define MAX 1000005
char str[MAX];
long long int arr[MAX],brr[MAX];
int main()
{
    int tc,n,m,i;
    cin>>tc;
    while(tc--)
    {
    	long long int r=0,k=0,rstring=0,kstring=0;
    	cin>>n>>m;
    	cin>>str;
    	arr[0]=brr[0]=1;
    	for(i=1;i<MAX;i++) arr[i]=brr[i]=0;
    	for(i=0;str[i];i++)
    	{
    		if(str[i]=='R') r++;
    		if(str[i]=='K') k++;
    		if(r>=n) rstring+=arr[r-n];
    	    if(k>=m) kstring+=brr[k-m];
    	    arr[r]++;brr[k]++;
    	}
    	cout<<rstring<<" ";
    	cout<<kstring<<endl;
    }
    return 0;
}
