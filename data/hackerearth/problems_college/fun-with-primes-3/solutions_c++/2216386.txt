#include <bits/stdc++.h>
using namespace std;

int main()
{
	long long int t;
	scanf("%lld\n",&t);
	for(;t>0;t--)
	{
		long long int n,i;
		scanf("%lld",&n);
	
		long long int sum=0;
		while(n)
		{
			sum+=n%10;
			n/=10;
		}		
		
		//printf("%lld\n",sum);
		int f=0;
		for(int i=2;i<=sqrt(sum);i++)
		{
			if(sum%i==0)
			{
				f=1;
				break;
			}
		}
		
		if(f==0 && sum!=1)
		printf("YES\n");
		else
		printf("NO\n");
	}	
	return 0;
}
