#include<bits/stdc++.h>
using namespace std;
int a[120];
int main()
{
    int t;
    scanf("%d",&t);
    memset(a,-1,sizeof(a));
    a[10]=1;a[12]=1;
    for(int i=13;i<120;i++)
    {
        if(a[i-10]==-1&&a[i-12]==-1);
        else
        {
            if(a[i-10]!=-1&&a[i-12]==-1)
            {
                a[i]=a[i-10]+1;
            }
            else if(a[i-10]==-1&&a[i-12]!=-1)
            {
                a[i]=a[i-12]+1;
            }
            else
            {
                a[i]=min(a[i-10],a[i-12])+1;
            }
        }
    }
    int n;
    while(t--)
    {
        scanf("%d",&n);
        printf("%d\n",a[n]);
    }
    return 0;
}
