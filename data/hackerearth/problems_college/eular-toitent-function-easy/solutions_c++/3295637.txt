#include<bits/stdc++.h>
using namespace std;
int gcd(int a, int b)
{
    if (a == 0)
        return b;
    return gcd(b%a, a);
}
 
// A simple method to evaluate Euler Totient Function
int phi(unsigned long long int n)
{
    unsigned int result = 1;
    for (int i=2; i<n; i++)
        if (gcd(i, n) == 1)
            result++;
    return result;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		unsigned long long int n;
		cin>>n;
		cout<<phi(n)<<"\n";
	}
}
