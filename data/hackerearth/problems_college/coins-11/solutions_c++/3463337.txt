#include <bits/stdc++.h>

using namespace std;

bool vis [100000 + 10];
int siz = 0;

int N = 100000 + 10;

void sieve(){

    for(long long i = 2; i * i <= N; i++){
        if(!vis[i]){
            for( long long j = i * i; j <= N; j+= i){
                vis[j] = true;
            }
        }
    }
}


int main()
{

    int t , n;

    sieve();
    vis[1] = true;

    cin >> t;

    while(t--){

        cin >> n;
        if(n == 3) cout << "3\n";
        else if(vis[n]) cout << "1\n";
        else cout << "2\n";

    }

    return 0;
}
