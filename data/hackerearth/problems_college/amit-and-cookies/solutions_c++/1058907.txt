#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
typedef unsigned long long ll;
typedef unsigned long l;

int main()
{
   l N;ll K,max=0;
   l count=0;
   cin>>N>>K;
   vector<int>VI;
   while(N--)
   {
   	l a;
   	cin>>a;
    VI.push_back(a);
   }
   sort(VI.begin(),VI.end());
   for(vector<int>::iterator itr=VI.begin();itr!=VI.end();++itr)
   {
       max+=*itr;
       if(K>=max){count++;}
   }
   cout<<count<<endl;

    return 0;
}
