t=int(raw_input())

def fact(n):
	if n==1:
		return 1
	else:
		return n*fact(n-1)


for i in range(0,t):
	n=int(raw_input())
	n=fact(n)
	sum=0
	for i in str(n):
		sum+=int(i)
	print str(sum)
