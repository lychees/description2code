#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    pair<double,double> Ar[1005];
    int N;
    cin>>N;
    for(int i=0;i<N;i++)
    {
    	double x,y;
    	cin>>x>>y;
    	Ar[i].first=x;
    	Ar[i].second=y;
    }
    sort(Ar,Ar+N);
    int cnt=0;
    for(int i=0;i<N-1;i++)
    {
    	double m=(Ar[i].second-Ar[i+1].second)/(Ar[i+1].first-Ar[i].first);
    	cnt++;
    	for(int j=i+2;j<N;j++)
    	{
    		double m1=(Ar[i].second-Ar[j].second)/(Ar[j].first-Ar[i].first);
    		if(m1<=m)
    		{
    			cnt++;
    			m=m1;
    		}
    	}
    }
    cout<<cnt<<"\n";
    return 0;
}
