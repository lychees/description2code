#include <iostream>
using namespace std;
typedef long int li;
li getProfit(li *a,int n){
    li profit=0;
    li maxs[n];
    li max  = a[n-1];

    for(int i = n-1;i>=0;--i){
        if(a[i]>max){
            maxs[i] =a[i];
            max = a[i];
        }
        else{
            maxs[i] = max;
        }
    }

    for(int i = 0;i<n;i++){
        if(a[i]<maxs[i]){
            profit += maxs[i] - a[i];
        }
    }
    return profit;
}
int main(void){

    int t;
    cin>>t;

    while(t--){
        int n;
        cin>>n;
        li a[n];
        for(int i = 0;i<n;i++)
            cin>>a[i];
        cout<<getProfit(a,n)<<endl;
    }
return 0;
}
