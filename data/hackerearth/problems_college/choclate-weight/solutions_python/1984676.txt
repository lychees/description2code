n = int(input())

a = list(map(int,raw_input().split()))

q = int(input())

for i in range(q):
	x = int(input())
	flag = False
	for j in range(n):
		if(flag == True):
			break
		for k in range(n):
			if i != j and (a[j] + a[k] == x or a[j] == x or a[k] == x):
				print('I Like You')
				flag = True
				break
	if(flag == False):
		print('I Hate You')
