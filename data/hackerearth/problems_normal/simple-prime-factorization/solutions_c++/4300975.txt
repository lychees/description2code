#include <iostream>
using namespace std;

int p[1000001] = {0};
int primes[1000001],index = 0;

void sieve(){
	for(int i = 2; i*i <= 1000000; i++){
		if(!p[i]){
			primes[index++] = i;
			for(int j = i*i; j <= 1000000; j += i){
				p[j] = 1;
			}
		}
	}
}

int main()
{
	sieve();
	for(int i = 1000; i < 1000000; i++){
		if(!p[i]){
			primes[index++] = i;
		}
	}
	int t,n,count;
	cin >> t;
	while(t--){
		cout << "2^";
		cin >> n;
		if(!p[n]){
			cout << 0 << endl;
			continue;
		}
		count = 0;
		while(n%2 == 0){
			count++;
			n /= 2;
		}
		cout << count;
		for(int i = 1; i < index; i++){
			if(n == 1){
				break;
			}
			if(n%primes[i] == 0){
				count = 0;
				while(n%primes[i] == 0){
					count++;
					n /= primes[i];
				}
				cout << "*" << primes[i] << "^" << count;
			}
		}
		cout << endl;
	}
    return 0;
}
