#include <bits/stdc++.h>

using namespace std;

namespace stuff {
typedef long long ll;

void solve(int test_num) {
  string str;
  cin >> str;
  vector<pair<int, char> > cnt(26);
  for (int c = 0; c < 26; c++)
    cnt[c].second = c + 'a';
  for (const char& c : str)
    cnt[c - 'a'].first++;
  sort(cnt.begin(), cnt.end());
  for (int c = 0; c < 26; c++)
    for (int i = 0; i < cnt[c].first; i++)
      cout << cnt[c].second;
  cout << endl;
}
