#include<iostream>
using namespace std;

int FindNearestNumber(long int a,long int b,long int c,long long int k)
{
	for(unsigned int i=0;i<k;i++)
	{
		int ax2 = a * i * i; 
		if(ax2 >= k) return i;	
		int bx = (b * i) + ax2;
		if(bx >=k) return i;
		int cc = bx + c;
		if(cc >=k) return i;
	}
	
	return k;	
}

int main()
{
	long int t,a,b,c;
	long long int k = 0;
	
	cin>>t;
	while(t--)
	{
		cin>>a>>b>>c>>k;
		cout<<FindNearestNumber(a,b,c,k)<<endl;
	}
	
	return 0;
}
