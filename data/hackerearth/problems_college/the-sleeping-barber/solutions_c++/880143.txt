#include <bits/stdc++.h>
using namespace std;

int main()
{
    int D,G,M;
    scanf("%d",&D);
    while(D--)
    {
    	scanf("%d%d",&G,&M);
    	int a[G];
    	for(int i = 0; i<G;i++)
    		scanf("%d",&a[i]);
    	sort(a,a+G);
    	int i;
    	for(i =0; i<G;i++)
    	{
    		if(a[i]<=M) M -= a[i];
    		else break;
    	}
    	printf("%d\n",i);
    }
    return 0;
}
