#include <iostream>
#include <cstdlib>

using namespace std;

int main()
{
    ios::sync_with_stdio(false);
    int t;
    cin>>t;
    while(t--){
        int p, x, y, u, v;
        cin>>p>>x>>y>>u>>v;
        if(x == u && y == v)
            cout<<"Counter-Terrorists Win !\n";
        else if(abs(x - u) == abs(y - v)){
            if(p > 1)
                cout<<"Counter-Terrorists Win !\n";
            else
                cout<<"Terrorists Win !\n";
        }
        else if( (abs(x - u) == 0 && abs(y - v) % 2 == 0 ) || ( abs(y - v) == 0 && abs(x - u) % 2 == 0 ) ){
            if(p > 2)
                cout<<"Counter-Terrorists Win !\n";
            else
                cout<<"Terrorists Win !\n";
        }
        else if( ( abs(x - u) % 2 == 0 && abs(y - v) % 2 == 0 ) || ( abs(x - u) % 2 == 1 && abs(y - v) % 2 == 1 ) ){
            if(p > 2)
                cout<<"Counter-Terrorists Win !\n";
            else
                cout<<"Terrorists Win !\n";
        }
        else
            cout<<"Terrorists Win !\n";
    }
    return 0;
}
