#include <bits/stdc++.h>
using namespace std;
#define ll long long int
#define f(i,n)  for(i=0;i<n;i++)
int main()
{
	int t;
	cin >> t ;
	std::vector<ll> :: iterator it;
	ll min=1,max=1;
	int cpos=0;
	int cneg=0;
	int czer=0;
	int temp;
	int flag=1;
	int i , n,x;
	vector<ll> v;
	while(t--)
	{
		min=1;
		max=1;
		cin >> n ;
		flag=1;
		cpos=0;
		cneg=0;
		czer=0;
		v.clear();
		temp=n;
		while(temp--)
		{
			cin >> x;
			v.push_back(x);
			if(x>0)
				cpos++;
			else if(x<0)
				cneg++;
			else
				czer++;
        }
		it=v.end();
		sort(v.begin(),it);
        if(cneg==0)
			min=v[0];
		else
		{
			if(cneg%2==0)
				for(i=0;i<n;i++)
				{
                    if(v[i]==0 || i==cneg-1)
                        continue;
					min=min*v[i];
				}
			else
			{
				for(i=0;i<n;i++)
				{
				    if(v[i]==0)
						continue;
					min=min*v[i];
                }
			}
		}
        if(cneg%2==0)
		{
			for(i=0;i<n;i++)
            {
				if(v[i]==0)
				continue ;
				max=max*v[i];
            }
		}
		else
		{
			if(cpos==0)
			{
				if(cneg==1)
					max=v[n-1];
				else
				{
					for(i=0;i<n;i++)
					{
						if(i==cneg-1 || v[i]==0)
							continue ;
						max=max*v[i];
					}
				}
			}
			else
			{
			    for(i=0;i<n;i++)
				{
					if(i==cneg-1 || v[i]==0)
						continue;
					max=max*v[i];
				}
			}
        }
        if(cneg==0 && cpos==0)
		{
			min=0;
			max=0;
        }
        cout << max << " " << min << endl;
	}
}
