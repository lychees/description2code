#include <iostream>
#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	cin >> t;
	
	while(t--)
	{
		int n;
		cin >> n;
		
		vector<int> arrival(n);
		vector<int> departure(n);
		
		for(int i=0;i<n;i++)
		{
			cin >> arrival[i] >> departure[i];
			departure[i]-=1;
		}
		
		sort(arrival.begin(),arrival.end());
		sort(departure.begin(),departure.end());
		
		int i=0,j=0;
		int overlap=0;
		int maxie=0;
		
		while(i<n && j<n)
		{
			if(arrival[i] <= departure[j]) { overlap++; maxie = max(maxie,overlap); i++; }
			else
			{ overlap--; j++;}
		}
		
		cout << maxie << endl;
	}
    
    return 0;
}
