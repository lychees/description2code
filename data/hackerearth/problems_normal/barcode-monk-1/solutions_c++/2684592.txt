

#include <iostream>
#include <algorithm>
#define mod 1000000007
using namespace std;

int n,k,a[1001];
bool pos[1001];
int dp[1001][2];

int main()
{
	cin>>n>>k;
	pos[0]=true;
	for(int i=0;i<k;++i)
	{
		cin>>a[i];
		if(a[i]>n)
			continue;
		for(int j=a[i];j<=n;++j)
			pos[j]=pos[j]|pos[j-a[i]];
	}
	dp[0][0]=1;
	for(int i=1;i<=n;++i)
	{
		dp[i][0]=(dp[i-1][0]+dp[i-1][1])%mod;
		for(int j=1;j<=i;++j)
			if(pos[j])
				dp[i][1]=(dp[i][1]+dp[i-j][0])%mod;
	}
	cout<<((dp[n][1]+dp[n][0])%mod);
    return 0;
}
