#include<bits/stdc++.h>
using namespace std;
int main()
{
	string str;
	cin>>str;
	int i,j,len = str.length();
	for(i=0;i<len;i++)
	{
		if(i==0)
		{
			if(islower(str[i]))
			str[i] = toupper(str[i]);
		}
		else
		if(str[i-1]=='_')
		{
			if(islower(str[i]))
			str[i]=toupper(str[i]);
		}
	}
	cout<<str<<"\n";
}
