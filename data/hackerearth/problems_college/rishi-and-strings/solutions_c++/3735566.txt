#include<bits/stdc++.h>
using namespace std;
int dp[1001][1001];
#define inf 1000000
int main()
{
	  string a,b;
	  cin>>a>>b;
	  int n=a.length();
	  int m=b.length();
	  for(int i=1;i<=m;i++)
	  dp[0][i]=i;
	  for(int j=1;j<=n;j++)
	  dp[j][0]=j;
	  for(int i=1;i<=n;i++)
	  {
	  	  for(int j=1;j<=m;j++)
	  	  {
	  	  	dp[i][j]=inf;
	  	  	  if(a[i-1]==b[j-1])
	  	  	  {
	  	  	    dp[i][j]=dp[i-1][j-1];	
			  }
			  else  
			  {
			  dp[i][j]=min(dp[i][j],1+dp[i-1][j-1]);
			  dp[i][j]=min(dp[i][j],1+min(dp[i-1][j],dp[i][j-1]));
			   }
	      }
	  }
	  cout<<dp[n][m]<<endl;
	  return 0;
}
