#include <bits/stdc++.h>
using namespace std;

int res,n;
int mark[31][31];
char path[31][31];

int min(int a, int b)
{
	if(a<b)
	return a;
	else
	return b;
}


void findpath(int x, int y, int val)
{

	if(x<0 || y<0 || path[x][y]=='T' || mark[x][y]==1 || x>=n || y>=n)
		return;
	else if(path[x][y]=='E')
		{
			res= min(res,val);
			return;
		}
	else
	{
		mark[x][y]=1;
		findpath(x-1,y,val+1);
		findpath(x,y-1,val+1);
		findpath(x,y+1,val+1);
		findpath(x+1,y,val+1);
		mark[x][y]=0;
	}
		

}

int main()
{
	//int n;
	res=9999;
	scanf("%d",&n);

	int x,y;	
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
		{
			mark[i][j]=0;
		}
	}
	
	for(int i=0;i<n;i++)
	for(int j=0;j<n;j++)
	{
		cin>>path[i][j];
		if(path[i][j]=='S')
		{
			x=i;
			y=j;
		}
	}
    
    findpath(x,y,0);
    printf("%d\n",res);
    return 0;
    
}
