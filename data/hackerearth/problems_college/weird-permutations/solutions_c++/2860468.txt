#include <bits/stdc++.h>
using namespace std;

typedef long long LL;
#define PII       pair<int,int>
#define all(c)    c.begin(),c.end()
#define sz(c)     (int)c.size()
#define clr(c)    c.clear()
#define pb        push_back
#define mp        make_pair
#define cin(x)    scanf("%d",&x)
#define MOD       1000000007

LL dp[(1<<17)][17];
int n , k;

LL solve(int mask , int last)
{
	if(mask + 1 == (1 << n)) return 1;
	else if(dp[mask][last] != -1) return dp[mask][last];
	LL ret = 0;
	for(int i = 0; i < n; i++)
	{
		if(mask&(1<<i)) continue;
		if( abs(2 * (i + 1) - 3 * (last + 1)) <= k)
		{
			ret += solve(mask|(1<<i) , i);
		}
	}
	return (dp[mask][last] = ret);
}

int main()
{
	int t;
	cin(t);
	while(t--)
	{
		for(int i = 0; i < (1<<17); i++)
			for(int j = 0; j < 17; j++)
				dp[i][j] = -1;
		cin(n);
		cin(k);
		LL ans = 0;
		for(int i = 0; i < n; i++)
			ans += solve(1<<i , i);
		printf("%lld\n" , ans);
	}
	return 0;
}
