#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define pb push_back
#define mp make_pair
struct node
{
    ll val;
    ll course;
};
class compare
{
public:
    bool operator()(const struct node &t1, const struct node &t2)
    {
        if (t1.val!=t2.val)
        {
         return t1.val>t2.val;
        }
        return t1.course>t2.course;
    }
};

priority_queue<node,vector<node>, compare> pq;
int main()
{
    ll c,p,n;
    cin>>c>>p>>n;
    int arr[n+1];
    int fr[p+1];
    int result[p+1];int i;
    vector<int>cour[c+1];node temp;
    for(i=1;i<=n;i++)
    {
        cin>>arr[i];
        cour[i].pb(arr[i]);
        temp.val=arr[i];
        temp.course=i;
        pq.push(temp);
    }
    for(i=n+1;i<=c;i++)
    {
        temp.val=0;
        temp.course=i;
        pq.push(temp);
    }
    for(i=1;i<=p;i++)
    {
        cin>>fr[i];
    }

    for(i=1;i<=p;i++)
    {
        temp=pq.top();
        result[i]=temp.course;
        pq.pop();
        cour[temp.course].pb(fr[i]);
        if (cour[temp.course].size()==1)
        {
            temp.val=cour[temp.course][0];
            pq.push(temp);
        }
        else
        {
            temp.val=cour[temp.course].size()*(cour[temp.course][cour[temp.course].size()-1]+cour[temp.course][cour[temp.course].size()-2]);
            pq.push(temp);
        }
    }
    for(i=1;i<=p;i++)
    {
        cout<<result[i]<<" ";
    }
    cout<<"\n";
    return 0;
}//main
