#include<string.h>
#include <iostream>
using namespace std;

int main()
{
	int t,i,j,a[26],even,odd,len;
	char str[1002];
	cin>>t;
	while(t--)
	{
		even=odd=0;
		cin>>str;
		len=strlen(str);
		for(i=0;i<26;i++)
			a[i]=0;
		for(i=0;i<len;i++)
		{
			a[str[i]-'a']++;
		}
		for(i=0;i<26;i++)
		{
			if(a[i]%2==1)
				odd++;
			else
				even++;
		}
		if(odd==0)
		{
			cout<<"0"<<endl;
		}
		else if(odd==1)
			cout<<"0\n";
		else if(odd>1)
			cout<<(odd-1)<<endl;
	}
    // << "Hello World!" << endl;
    return 0;
}
