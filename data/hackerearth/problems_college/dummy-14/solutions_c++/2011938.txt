#include <bits/stdc++.h>
using namespace std;
#define MAX 100000000
int a[MAX]={0};
int main()
{
    int t;
    cin>>t;
    int n;
    
    	for(int i=1;i<=100;i++)
    	{
    		for(int j=i+1;j<=100;j++)
    		{
    			a[(i*i*i)+(j*j*j)]++;
    		}
    	}
    while(t--)
    {
    	cin>>n;
    	long long int i=0,j=0;
    	while(i<n)
    	{
    		if(a[j]>=2)
    		{
    			i++;
    		}
    		j++;
    	}
    	cout<<--j<<endl;	
    }  
}
