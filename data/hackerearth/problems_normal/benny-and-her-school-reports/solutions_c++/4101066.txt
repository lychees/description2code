#include <iostream>
#include <string>
#include <sstream>
using namespace std;

int main()
{
    long N,n,m,x,t,t1;
    string line;
    cin>>N;
    for(int i=0;i<N;i++){
    	cin>>n>>m>>x;
    	t=0;
    	for(int j=0;j<n;j++){
    		cin>>t1;
    		t+=t1;
    	}
    	if(t/(n+1)>=x){
    		cout<<"1\n";
    		continue;
    	}else{
    		t1=x*(n+1)-t;
    		if(t1>m)
    		cout<<"Impossible\n";
    		else
    		cout<<t1<<endl;
    	}
    }
    return 0;
}
