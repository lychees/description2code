#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll t[100001];
int n,m;
int days(ll d)
{
	int req=1;
	ll currsum=0;
	for(int i=1;i<=n;i++)
	{
		if(currsum+t[i]>d)
		{
			req++;
			currsum=t[i];
		}
		else
			currsum+=t[i];
	}
	return (req<=m)?1:0;
}
int main()
{
	cin >> n >> m;
	ll lo = 1,hi=1e17;
	for(int i=1;i<=n;i++)
	{
		cin >> t[i];
		lo=max(lo,t[i]);
	}
	while(lo<hi)
	{
		ll mid=(lo+hi)>>1;
		int p=days(mid);
		if(p)
		hi=mid;
		else
		lo=mid+1;
	}
	cout << lo << endl;
	return 0;
}
