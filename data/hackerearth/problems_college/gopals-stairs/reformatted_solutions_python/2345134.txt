a = 100010 * [0]
a[0], a[1], a[2] = 1, 1, 2
for i in range(3, 50010):
    a[i] = a[i-1] + a[i-2] + a[i-3]
done = False
for _ in xrange(int(raw_input())):
    n = int(raw_input())
    if n > 50000 and not done:
        for i in range(50000, 100010):
            a[i] = a[i-1] + a[i-2] + a[i-3]
        done = True
    print a[n] % 1000000007
