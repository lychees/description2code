#include<iostream>
using namespace std ;
int input[85][85];
int T , total=0;
void find(int , int ) ;
int main() {
	int i,j;
	cin >> T;
	for(i=0;i<T;i++)
	{
		for(j=0;j<T;j++)
		{
			cin >> input[i][j] ;
		}
	}find(0,0);
	cout << total << endl ;
	return 0 ;
}
void find(int x , int y )
{
	if(x>=T || y>=T || x<0 || y<0 )
	{
		return ;
	}
	if(x== (T-1) && y== (T-1) )
	{
		//cout << total << endl ;
		total++;
		return  ;
	}
	input[x][y] = 1;
	if( x+1 <T )
	{
		if(input[x+1][y] != 1 )	
		{
			find(x+1,y) ;
		}	
	}
	if( y+1 < T )
	{
		if(input[x][y+1] != 1)	 
		{
			find(x,y+1 ) ;
		}
	
	}
	if( x-1 >=0 )
	{
		if(input[x-1][y] != 1 )
		{
			find(x-1 , y ) ;
		}
	}
	if( y-1 >= 0 )
	{
		if(input[x][y-1] != 1)
		{
			find(x , y-1 ) ;
		}
	}
	input[x][y] = 0 ;
return  ;
}
