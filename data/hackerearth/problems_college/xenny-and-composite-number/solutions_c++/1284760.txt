#include <iostream>
#include<cstdio>
#include<math.h>
using namespace std;
int prime[100000];
int main()
{
    long int i,j,n;
    int t;
    for(i=2;i<=sqrt(100000);i++)
    {
        if(prime[i] == 0)
        {
            for(j=i*i;j<=100000;j+=i)
                prime[j] = 1;
        }
    }
    scanf("%d",&t);
    while(t--)
    {
        scanf("%ld",&n);
        n=n+1;
        while(prime[n] == 0)
            n++;
        printf("%ld\n",n);
    }
    return 0;
}
