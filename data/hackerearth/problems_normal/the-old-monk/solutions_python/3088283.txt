import sys
t=int(raw_input())
while t>0:
	t-=1
	n=int(raw_input())
	arr1=map(int,raw_input().split())
	arr2=map(int,raw_input().split())
	ans=0
	l=1

	for i in xrange(n):
		
		r=n-1
		
		while r>0 and l<r and l<n:
			
			mid=(l+r+1)/2
			#print "Checking %d %d %d"%(l,mid,r)
			if arr2[mid]>=arr1[i]:
				l=mid
				ans=max(ans,mid-i)
			else:
				r=mid-1
			#print l,mid,r
			sys.stdout.flush()
		#print ' '
	print ans
