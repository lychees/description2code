#include <iostream>
#include <cstdio>
using namespace std;

inline int max(int a , int b)
{
	if(a>b)
		return a ;
	else
		return b ;
}
inline int min(int a,int b)
{
	if(a<b)
		return a ;
	else
		return b ;	
}

void common(int *aold,int *bold,int anew, int bnew)
{
	*aold=max(*aold,anew);
	*bold=min(*bold,bnew);
}


int main()
{
 int T,N , i , j ;
 int  *Arry ;
 int  minprev = 0 , maxprev = 1000000000 ;
 int mincur = 0 , maxcur = 1000000000 ;
 int mean , prev ,next , mid;
 i = 1 ;
 scanf("%d",&T) ;
 while(i<=T)
 {
        minprev = 0 , maxprev = 1000000000 ;
 	scanf("%d",&N);
 	j=2;
 	scanf("%d",&prev);
 	while(j<=N)
 	{
 		scanf("%d",&next);	
 		if(next>prev)
 		{
 			if((next-prev)%2==0)
 				mid=(next-prev)/2;
 			else
 				mid=(next-prev)/2;
 				
 			maxcur=prev + mid;
 			mincur=0 ;
 			
 		}
 		else
 		{
 			if((prev-next)%2==0)
 				mid=(prev-next)/2;
 			else
 				mid=((prev-next)/2)+1;
 			maxcur = 1000000000 ;
 			mincur = next +mid ;
 			
 		}
 		common(&minprev,&maxprev,mincur,maxcur) ;
 		prev = next ;
 		j++ ;
 		
 	}
 	
 	if(maxprev - minprev >= 0)
 	{
 		printf("%d %d\n", minprev, maxprev);
 		
 	}else
 	{
 		printf("-1\n");
 		
 	}
 	
 	i++ ;
 }
 return 0 ;
}
