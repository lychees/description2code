#include <iostream>
#include <stdio.h>

void print_logo(int N){
	int charCount = 1;
	//prints first half
	for(int i = 1; i <= N; ++i){
 		for(int j = 1; j <= N - i; ++j)
			printf(" ");

 		for(int k = 1; k <= charCount; ++k){
 			printf("/");
 			if(charCount > 1 && k < charCount)
 				printf(" ");
 		}

		if(i%2 == 0)
			printf("  ");

 		for(int m = 1; m <= charCount; ++m){
 			printf("\\");
 			if(charCount > 1 && m < charCount)
 				printf(" ");
 		}

 		if (i % 2 == 0) ++charCount;
		printf("\n");
	}

	if (N%2 == 0)
		charCount -= 1;

	//prints second half in reverse
	for(int i = 1; i <= N; ++i){
 		for(int j = N - i; j <= N; ++j)
			printf(" ");

 		for(int k = 1; k <= charCount; ++k){
 			printf("\\");
 			if(charCount > 1 && k < charCount)
 				printf(" ");
 		}

		if((N-i-1)%2 == 0)
			printf("  ");

 		for(int m = 1; m <= charCount; ++m){
 			printf("/");
 			if(charCount > 1 && m < charCount)
 				printf(" ");
 		}

 		if ((N-i) % 2 == 0) --charCount;
		printf("\n");
	}
}

int main()
{
	int N;
	std::cin >> N;
	print_logo(N);
	print_logo(N+1);
	return 0;
}
