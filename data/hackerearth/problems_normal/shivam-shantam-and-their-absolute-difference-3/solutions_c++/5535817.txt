#include <iostream>
using namespace std;

int main()
{
    int n,q;
    cin>>n;
    int a[n];
    int b[n];
    for(int i=0;i<n;i++)cin>>a[i];
    for(int i=0;i<n;i++)cin>>b[i];
    cin>>q;
    long long mod = 1000000007;
    long long x[400001]={0};
    long long y[400001]={0};
    x[0]=1;y[0]=1;
    long long s=0;
    for(int j=0;j<n;j++)
     {  s+=a[j];
     	for(int i=s;i>=a[j];i--)
          x[i]=(x[i]+x[i-a[j]])%mod;
     }
    //for(int i=0;i<=s;i++)cout<<x[i]<<' ';    cout<<endl;
    long long s2=0;
    for(int j=0;j<n;j++)
     {  s2+=b[j];
     	for(int i=s2;i>=b[j];i--)
          y[i]=(y[i]+y[i-b[j]])%mod;
     }
    //for(int i=0;i<=s2;i++)cout<<y[i]<<' ';    cout<<endl;
    long long Y[400002]={0};
    Y[0]=0;
    for(int i=1;i<=400001;i++)
     {
     	Y[i]=Y[i-1]+y[i-1];
     }
    /*for(int i=0;i<=s2;i++)cout<<Y[i]<<' ';    cout<<endl;
     for(int i=0;i<=s;i++)cout<<X[i]<<' ';    cout<<endl;*/
    long long ans=0;
    for(int i=0;i<=s;i++)
     {long long t = (Y[min(400001,i+q+1)]-Y[max(0,i-q)]+mod)%mod;
      ans = (ans+(x[i]*t)%mod+mod)%mod;
      }
    cout<<ans<<endl;
    return 0;
}
