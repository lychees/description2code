#include<bits/stdc++.h>
#define ll long long
#define forn(i,n) for(int i=0;i<n;i++)
using namespace std;
int main(){
	ll t;
	cin>>t;
	while(t--)
	{
		ll n,flag=0,a,b,c;
		cin>>n;
		ll ar[n];
		forn(i,n){
			cin>>ar[i];
		}
		sort(ar,ar+n,std::greater<int>());
		forn(i,n-2){
			if(ar[i]<ar[i+1]+ar[i+2]){
				flag=1;
				a=ar[i+2];b=ar[i+1];c=ar[i];
				break;
			}
		}
		if(flag==1){
			cout<<a<<" "<<b<<" "<<c<<endl;
		}
		else
		cout<<-1<<endl;
	}
		
	return 0;
}
