#include<stdio.h>
#include<math.h>
#include<algorithm>
using namespace std;
void solve()
{
     int nof;
     int cost[100000][4];                //LET dp[i][1] represents the solution of the problem upto (i+1) th factories including the second shop
     int dp[100000][4];
     scanf("%d",&nof);
     for(int i=0;i<nof;i++)
     {
         scanf("%d %d %d",&cost[i][0],&cost[i][1],&cost[i][2]);
     }
     dp[0][0]=cost[0][0];
     dp[0][1]=cost[0][1];
     dp[0][2]=cost[0][2];
     for(int j=1;j<nof;j++)
     {
         dp[j][0]= cost[j][0] + min(dp[j-1][1],dp[j-1][2]);
         dp[j][1]= cost[j][1] + min(dp[j-1][0],dp[j-1][2]);
         dp[j][2]= cost[j][2] + min(dp[j-1][1],dp[j-1][0]);
     }
    int res= min(dp[nof-1][0],dp[nof-1][1]);
    res=min(res,dp[nof-1][2]);
    printf("%d\n",res);
}
int main()
{
    int tc;
    scanf("%d",&tc);
    for(int i=0;i<tc;i++)
        solve();
    return 0;
}
