#include<bits/stdc++.h>
using namespace std;
#define ll long long int 
ll ones(ll n)
{
	ll ans=0;
	while(n>0)
	{
		ans++;
		n=n&(n-1);
	}
	return(ans);
}
bool cmp(pair<ll,ll>a,pair<ll,ll>b)
{
	
 if(a.second!=b.second)
 return(a.second>b.second?false:true);
 else
 return(a.first>b.first?false:true);
}
int main()
{
	
	//cout<<ones(656478043)<<endl;
	ll t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		ll arr[100009];
		//vector<pair<value,pair<numOfOne,i> > >
		vector<pair<ll,ll> >res;
		for(ll i=0;i<n;i++)
		{
		 cin>>arr[i];
		 res.push_back(make_pair(i,ones(arr[i])));
		}
		sort(res.begin(),res.end(),cmp);
		for(ll i=0;i<res.size();i++)
		cout<<arr[res[i].first]<<" ";
		cout<<endl;
		
	    
	}
	return(0);
}
