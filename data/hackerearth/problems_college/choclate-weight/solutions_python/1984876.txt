def sub(x,n,m):
	x.sort()
	if m in x:
		return True
	else:
		i=0
		j=n-1
		while i<j:
			if x[i]+x[j]==m:
				return True
			elif x[i]+x[j]>m:
				j-=1
			else:
				i+=1
		return False
n=input()
x=map(int,raw_input().split())
m=input()
for _ in xrange(m):
	p=input()
	print "I Like You" if sub(x,n,p) else "I Hate You"
