#include <iostream>
#include<cstdio>
using namespace std;


int main()
{
	double x,y,cx,cy,n;
	long long t,i;	
	cin>>t;
	while(t--)
	{

		cin>>n;
		cx=cy=0.0;
		for(i=1;i<=n;i++)
		{
			cin>>x;
			cin>>y;
			cx=cx+x;
			cy=cy+y;
		}
		printf("(%.2f , %.2f)\n",1.0*cx/n,1.0*cy/n);
//		cout<<"("<<<<" , "<<1.0*cy/n<<")"<<endl;
		
	}
	return 0;
}
