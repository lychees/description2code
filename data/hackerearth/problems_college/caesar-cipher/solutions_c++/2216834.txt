#include <bits/stdc++.h>
using namespace std;

int main()
{
	long long int t;
	 
	cin>>t; 
	for(;t>0;t--)
	{
		long long int n;
		cin>>n;
		
		//string p;
		//cin>>p;
		
		string s;
		cin>>s;
		
		for(int i=0;i<s.size();i++)
		{
			int v=s[i];
			v+=n;
			
			if(v>=123 && s[i]>='a' && s[i]<='z')
			{
				int c=v-123;
				c%=26;
				v=97+c;	
			}
			else if(v>=91 && s[i]>='A' && s[i]<='Z')
			{
				int c=v-91;
				c%=26;
				v=65+c;
			}
			
			//cout<<v<<"\n";
			char b=v;
			cout<<b;
		}
		cout<<"\n";
	}
	
	return 0;
}
