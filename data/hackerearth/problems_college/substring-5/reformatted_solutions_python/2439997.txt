def get_all_substrings(input_string):
    length = len(input_string)
    return [input_string[i:j+1]
            for i in xrange(length) for j in xrange(i, length)]


s = raw_input()
count = 0
flag = 0
rs = get_all_substrings(s)
for w in rs:
    flag = 0
    if len(w) == 1:
        count += 1
    else:
        for i in xrange(1):
            for j in xrange(len(w)):
                if w[i] == w[j]:
                    flag += 1
        if flag == len(w):
            count += 1

print count
