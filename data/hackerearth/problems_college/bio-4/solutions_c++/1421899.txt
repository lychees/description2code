#include<bits/stdc++.h>
using namespace std;

int dp[1010][1010];
string s1,s2;

void solve() {
    memset(dp,0,sizeof(dp));
    int len1=s1.length();
    int len2=s2.length();
    int result=0;
    for(int i=1;i<=len1;i++) {
        if(i<len2) {
            continue;
        }
        int cnt=0;
        for(int j=i-len2+1,l=1;j<=i;j++,l++) {
            if(s1[j-1]==s2[l-1]) {
                cnt++;
            }
        }
        result=max(result,cnt);
    }
    printf("%d\012",len2-result);
    return;
}
int main() {
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        memset(dp,0,sizeof(dp));
        cin>>s1>>s2;
        solve();
    }
    return 0;
}
