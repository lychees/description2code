#include <iostream>
using namespace std;

int main()
{
	int n,k;
	cin>>n>>k;
	int a[n];
	
	for(int i=0;i<n;i++)
		cin>>a[i];
	int count=0;
	
	for(int i=0;i<n;i++)
	{
		int j=i+k;
		if(j>=n)
			break;
		int temp=abs(a[j]-a[i]);
		int temp2=a[j]+a[i];
		
		if(temp%k==0 || temp2%k==0)
		{
			count++;
		}	
		
	}
	cout<<count<<endl;
    return 0;
}
