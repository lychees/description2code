#include <bits/stdc++.h>
#include <stdio.h>
#define maxii 200005
using namespace std;

long long power[1000005];
long long a[1000005];
string s;
unordered_set<long long> mine;


bool tell(int m)
{
  mine.clear();
  for(int i=0;i+m<=s.size();i++)
  {
    long long xx=a[i+m]-a[i];
    xx*=power[1000000-i];
    if(mine.find(xx)!=mine.end())
      return true;
    mine.insert(xx);
  }
  return false;
}

int main()
{
    ios_base::sync_with_stdio(0);
    
    cin>>s;
    power[0]=1;
    for(int i=1;i<1000005;i++)
     power[i]=power[i-1]*173;
  
    int n=s.size();

    for(int i=1;i<=n;i++)
    {
     a[i]=a[i-1]+s[i-1]*power[i-1];
    }
    int l=0,r=n;
    while(l<r)
    {
      int mid=(l+r+1)/2;
      if(tell(mid))
        l=mid;
      else
        r=mid-1;
    }
  
    cout<<l;
    
    return 0;
}
