#include <iostream>
using namespace std;

int main()
{
    int t,n ; 
    scanf("%d",&t) ; 
    while(t--)
    {
    	long long a[100005] ; 
    	scanf("%d",&n) ; 
    	for(int i = 0  ; i < n ;i++ )
    	scanf("%d",&a[i]) ; 
    	if(n == 1)
    	printf("%d\n",a[0]) ; 
    	else
    	{
    		printf("0\n") ; 
    	}
    }
}
