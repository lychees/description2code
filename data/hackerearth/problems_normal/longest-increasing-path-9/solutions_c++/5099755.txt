#include <iostream>
using namespace std;

int maxx=0,n,m;
long M[101][101];
long A[101][101];


void fun(int i,int j,int c){
	if(c>maxx){
		maxx=c;
	}
	if(i+1<n&&A[i+1][j]>A[i][j]&&M[i+1][j]<c+1){
		M[i+1][j]=c+1;
		fun(i+1,j,c+1);
	}
	if(j+1<m&&A[i][j+1]>A[i][j]&&M[i][j+1]<c+1){
		M[i][j+1]=c+1;
		fun(i,j+1,c+1);
	}
	return;
}

int main()
{
 int t,i,j,ans;
 cin>>t;
 while(t--)
 {
  cin>>n>>m;
  for(i=0;i<n;i++)
   for(j=0;j<m;j++){
    cin>>A[i][j];
    M[i][j]=1;
   }
   maxx=0;
  fun(0,0,1);
  cout<<maxx<<"\n"; 
 }
 return 0;
}
