#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int n;string s;
    vector<int> v(26,0);
    cin >> n;
    while(n--){
    	cin >> s;
    	int i = 0;
    	while(i < s.size()){
    		v[s[i] - 'a']++; 
    		
    		i++;
    	}
   intern:
    	int min = 10000,maxi = -1;char tempchar = 'z';
    	for(int i = 0 ; i < 26 ; i++){
    		if(v[i] != 0){
    			if(v[i] < min){
    				min = v[i];
    			if(min == v[i]){
    				maxi = i;
    			    tempchar = v[i];	
    			}
    		  }
    		else if(v[i] == min && (char)v[i] < tempchar){
    			maxi = i;
    			tempchar = v[i];
    		}
    		
    		}
    	}
    	if(maxi >= 0){
    	v[maxi]=0;
    	}
    	
      if(min > 0 && min < 10000){
    	while(min--)cout << (char)('a'+maxi);
      
      }
    if(maxi != -1){
    goto intern;
    }
    cout << '\n';
    }
    
    
    return 0;
}
