#include <iostream>
#include<bits/stdc++.h>
using namespace std;



int main()
{
    int t;
    cin>>t;
    while(t--){
    	
    	int n,k;
    	int i;
    	int index;
    	cin>>n>>k;

    	vector<long long> a(n);
    	for(i=0;i<n;i++){
    		
    		cin>>a[i];
    	}
    	
    	for(int i=0;i<n-1;i++){
    	    index=i;	
    		for(int j=i+1;j<n&&(j-i)<=k;j++){
    			
    			if(a[index]>a[j])
    				index=j;
    			
    		}
    			
    			
    			int temp;
    			temp=a[index];
    			for(int x=index;x>i;x--){
    				a[x]=a[x-1];
    				
    			}
    			a[i]=temp;
    			k=k-(index-i);
    				
    			
    		}
    		
    		
    		
    		
    		
    	
    	
    	
    	
    	
   for(int h=0;h<n;h++){
   	
   	cout<<a[h]<<" ";
   	
   	
   } 	
   cout<<endl;
    	
    }
    return 0;
}
