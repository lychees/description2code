#include<bits/stdc++.h>
using  namespace std;
int main()
{
      int t;
      cin>>t;
      while(t--)
      {
            string s,p;
            cin>>s>>p;
            int  i=0;
            int j=0;
            while(i<s.length())
            {
                  if(s[i]==p[j])
                  {
                         i++;
                         j++;
                  }
                  else
                  {
                        i++;
                  }
            }

            if(j==p.length())
            {
                  cout<<"YES"<<endl;
            }
            else
            {
                 cout<<"NO"<<endl;
            }
      }
      return 0;

}
