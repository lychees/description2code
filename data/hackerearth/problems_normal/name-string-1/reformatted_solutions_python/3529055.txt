test = input()
for i in range(test):
    l = sorted(map(int, raw_input().split()))
    medium = l[1]
    maximum = l[2]
    count = 0
    while maximum != 0 and medium != 0:
        maximum -= 1
        medium -= 1
        count += 2
    if maximum != 0:
        print count+1
    else:
        print count
