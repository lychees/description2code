#include<iostream>
#include<cstdio>
#define ll long long
#include<cstring>
#include<algorithm>
#include<cmath>
using namespace std;
ll arr[1000005];
ll cal(ll n)
{
	ll ans=0;
	while(n)
	{
		ans+=n/5;
		n/=5;
	}
	return ans;
}
void preprocess()
{
	ll i;
	for(i=5;i<1000001;i++)
	{
		if(i%5==0)
			arr[i]=cal(i);
		else
			arr[i]=arr[i-1];
	}
}
int main()
{
	ll t,a,b,sum,i;
	preprocess();
	for(i=1;i<1000001;i++)
		arr[i]+=arr[i-1];
	scanf("%lld",&t);
	while(t--)
	{
		sum=0;
		scanf("%lld%lld",&a,&b);
		printf("%lld\n",arr[b]-arr[a-1]);
	}
	return 0;
}
