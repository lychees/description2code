import itertools

tst = int(raw_input())
i = 0
while i < tst:
    num = int(raw_input())
    ar = [int(x) for x in raw_input().split()]
    su = int(raw_input())

    res = False

    for j in range(1, num+1):
        for comb in itertools.combinations(ar, j):
            if sum(comb) == su:
                res = True
                break
        if res:
            break

    if res:
        print "YES"
    else:
        print "NO"

    i += 1
