#include<bits/stdc++.h>
using namespace std;
long long int power(long long int a, long long int b,long long int mod)
{
   if(b==0)
   return 1;
   long long int temp=power(a,b/2,mod);
   if(b&1)
   return (((temp*temp)%mod)*a)%mod;
   else
   return (temp*temp)%mod;
}
int main()
{
 int t;
 cin>>t;
 while(t--)
 {
   long long int a,m;
   cin>>a>>m;
   if(a==0 || m==2)
   {
    cout<<"YES"<<endl;
      continue;
   }
   long long int val=(m-1)/2;
   long long int ans=power(a,val,m);
   if(ans==1)
   cout<<"YES"<<endl;
   else cout<<"NO"<<endl;
   
}
return 0;
}
