#include <iostream>
#include <stdio.h>
#include <algorithm>
using namespace std;

int main()
{
    int t ; 
    scanf("%d",&t);
    while( t > 0 ){
    	int n ;
    	scanf("%d",&n);
    	int h[n];
    	int dp[n];
    	for(int i = 0 ; i < n ; i++){
    		h[i] = 0;
    		dp[i] = 0;
    	}
    	for(int i = 0 ; i < n ; i++){
    		int a ;
    		scanf("%d",&a);
    		h[i] = a;
    	}
    	
        if(h[0] > 0)
           dp[0] = h[0];
        int max = 0;
        for(int i = 1 ; i < n ; i++){
        	if(dp[i-1] + h[i] > 0){
        	     dp[i] = dp[i-1] + h[i];
        	     max = dp[i] > max ? dp[i] : max;
        	}
        	 else
        	      dp[i] = 0;
        }
        //sort(dp , dp + n);
        printf("%d\n",max);
    	t--;
    }
    return 0;
}
