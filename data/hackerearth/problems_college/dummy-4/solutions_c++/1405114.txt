#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    string str;
         int n,rem,sum = 0;
    while(t--){
        
        cin>>str;
        for(int i=0;i<str.length();i++)
        n+=str[i]-'0';
    }
      
      while(n > 0)
      {
                 while(n != 0)
                 {
                      rem = n%10;
                      sum = sum+rem;
                      n=n/10;
                 }
                 if(sum > 9)
                 {
                        n = sum;
                        sum = 0;
                 }
                  
      }
      cout<<sum<<endl;
    
}
