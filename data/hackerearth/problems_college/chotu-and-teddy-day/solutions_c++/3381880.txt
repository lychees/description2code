#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    ll t,n,k,sum;
    scanf("%lld", &t);
    while(t--){
        scanf("%lld %lld", &n, &k);
        sum = (n*(n+1))/2;
        if(sum>=k){
            cout<<"YES\n";
        }
        else cout<<"NO\n";
    }
    return 0;
}
