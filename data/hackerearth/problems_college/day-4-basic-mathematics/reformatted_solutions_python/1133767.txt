from math import sqrt
n = int(raw_input())
for i in range(n):
    x = int(raw_input())  # 12
    x = x << 1  # 24
    y = int(sqrt(x))  # 4
    x1 = y*(1 + y)  # 20
    if x1 < x:  # 20 < 24
        x = y + 1
    else:
        x = y
    x = x % 3
    if x == 0:
        print "SL"
    elif x == 1:
        print "LB"
    else:
        print "BS"
