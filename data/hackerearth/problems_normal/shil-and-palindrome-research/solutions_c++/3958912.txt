#include<bits/stdc++.h>
using namespace std;
int bit[27][100001];
int n,q;
string s;
inline void update(int idx,char c,int val)
{
    int t=(int)(c-'a');
    while(idx<=n)
    {
        bit[t][idx]+=val;
        idx+=(idx&(-idx));
    }
}
inline int sum(int idx,char c)
{
    int t=(int)(c-'a');
    int s=0;
    while(idx>0)
    {
       s+=bit[t][idx];
       idx-=(idx&(-idx));
    }
    return s;
}
void sol()
{
   cin>>n>>q;
   cin>>s;
   for(int i=1;i<=n;++i)
   {
       update(i,s[i-1],1);
   }
   while(q--)
   {
       int a;
       cin>>a;
       if(a==1)
       {
         int l;
         char c;
         cin>>l>>c;
         update(l,s[l-1],-1);
         s[l-1]=c;
         update(l,s[l-1],1);
       }
       else
       {
           int l,r;
           cin>>l>>r;
           int x=0;
           int odd=0;
           for(x=0;x<26;x++)
           {
               int total=sum(r,'a'+x)-sum(l-1,'a'+x);
               if(total&1) odd++;
           }
           if(odd>1) cout<<"no"<<endl;
           else cout<<"yes"<<endl;
       }
   }
}
int main()
{
    sol();
    return 0;
}
