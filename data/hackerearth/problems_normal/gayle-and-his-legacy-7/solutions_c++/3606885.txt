#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin >> t;
	while(t--){
		int n;
		long long m,g,p;
		cin >> n >> m;
		bool flag = true;
		vector<pair<long long, long long>> pi;
		for(long long i=1;i<=n; i++){
			cin >> g >> p;
			pi.push_back({p,g});
		}
		sort(pi.begin(), pi.end());
		
		for(auto &c: pi){
			if(m >= c.first) m+= c.second;
			else flag = false;
			
		}
		
		if(flag) cout << "YES\n";
		else cout << "NO\n";
	
	}
	
	return 0;
}
