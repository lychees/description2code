#include <bits/stdc++.h>

using namespace std;
long long solution[10000003];
void calculate() {
	solution[0] = 0;
	solution[1] = 1;
	for(long long  i = 2; i < 10000001; i++) {
		solution[i] = max(solution[i/2]+solution[i/3]+solution[i/4],i);
	}
	return;
}

long long coinExchange(long long n){
	long long  sol = 0;
	if(n > 10000000) {
		sol  = max(coinExchange(n/2)+coinExchange(n/3)+coinExchange(n/4),n);
	} else {
		sol = solution[n];
	}
	return sol;
}

int main() {
	long long n;
	calculate();
	while(scanf("%lld", &n) > 0) {
		printf("%lld\n", coinExchange(n));
	}
	return 0;
}
