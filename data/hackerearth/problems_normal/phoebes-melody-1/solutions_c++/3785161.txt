#include <iostream>
using namespace std;
int l[200005];
int r[200005];
int main()
{
	int t;
	scanf("%d",&t);
	while (t--)
	{
		int n,k;
		scanf("%d%d",&n,&k);
		int a[n+1];
		for (int i = 0; i < n; i++)
			scanf("%d",&a[i]);
		l[0] = -1;
		for (int i = 1; i < n; i++)
		{
			l[i] = i-1;
		}
		for (int i = 0; i < n-1; i++)
		{
			r[i] = i+1;
		}
		r[n-1] = n;
		for (int i = 1; i < n; i++)
		{
			while (l[i] >= 0 && a[i] <= a[l[i]])
			{
				l[i] = l[l[i]];
			}
		}
		for (int i = n; i >= 0; i--)
		{
			while (r[i] < n && a[i] <= a[r[i]])
			{
				r[i] = r[r[i]];
			}
		}
		int ans = 0;
		for (int i = 0; i < n; i++)
		{
			int x = i+k;
			int y = i-k;
			while (x < n && a[x] > a[i])
			{
				x = r[x];
			}
			while (y >= 0 && a[y] > a[i])
			{
				y = l[y];
			}
			int val1 = 987654321;
			int val2 = 987654321;
			if (x >= (i+k) && x < n)
			{
				val1 = abs(i-x);
			}
			if (y >= 0 && y <= (i-k))
			{
				val2 = abs(i-y);
			}
			if (val1 != 987654321 || val2 != 987654321)
				ans = ans+min(val1,val2);
		}
		cout << ans << endl;
	}
}
