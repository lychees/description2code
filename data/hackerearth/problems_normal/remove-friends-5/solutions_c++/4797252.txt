#include<bits/stdc++.h>
using namespace std;
struct node 
{
	int data;
	node * next;
	node *prev;
};
void push(node **head,node **last,int key)
{
	node *te=new node;
	te->data=key;
	te->next=NULL;
	te->prev=NULL;
	if(*last==NULL)
	{
		*last=te;
		*head=te;
		return ;
	}
	else
	(*last)->next=te;
	te->prev=*last;
	*last=te;
	return;
}
void prin(node *head)
{
	while(head!=NULL)
	{
		printf("%d ",head->data);
		head=head->next;
	}
}

						fl=1;

					cur=cur->next;
					te=te->next;
