#include <iostream>
#include<math.h>
using namespace std;
int gcd(int a,int b);
int prime(int a);

int main()
{
	int t,j,lcm,c,a,b,k;
	cin>>t;
	for(int i=0;i<t;i++)
	{
		c=0;
		cin>>a>>b;
		lcm=(a*b)/gcd(a,b);
		if(lcm%2==0)
		c++;
		while(lcm%2==0)
		lcm=lcm/2;
		for(j=3;j<lcm;j+=2)
		{
			if(lcm%j==0)
			c++;
			while(lcm%j==0)
			lcm=lcm/j;
		}
		if(lcm>2)
		c++;
		k=prime(c);
		if(k)
		cout<<"Yes\n";
		else
		cout<<"No\n";
	}
    return 0;
}
int gcd(int a,int b)
{
	if(b==0)
	return a;
	return gcd(b,a%b);
}
int prime(int a)
{
	if(a==1)
	return 0;
	for(int j=2;j<=sqrt(a);j++)
	{
		if(a%j==0)
		return 0;
	}
	return 1;
}
