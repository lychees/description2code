#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,q;
    cin>>n>>q;
    pair<long long int,long long int> p[n];
    for(int i=0;i<n;i++)
    	cin>>p[i].second;
    for(int i=0;i<n;i++)
    	cin>>p[i].first;
    sort(p,p+n);
    int ans[n];
    memset(ans,0,sizeof ans);
    ans[n-1] = p[n-1].second;
    for(int i = n-2;i>=0;i--){
    	ans[i] = ans[i+1] + p[i].second;
    }
    for(int i=0;i<q;i++){
    	int x;
    	cin>>x;
    	cout<<ans[n-x]<<endl;
    }
    return 0;
}
