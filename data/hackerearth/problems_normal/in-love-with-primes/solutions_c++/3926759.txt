#include<bits/stdc++.h>

using namespace std;

void sieve_of_eratosthenes(vector<int> &a,int n)
{
	for(int i=0;i<=n;i++)
		a.push_back(1);
	for(int i=2;i<=sqrt(n);i++)
	{	
		if(a[i] == 1)
		{
			for(int j=i*i;j<=n;j=j+i)
				a[j] =0;
		}
	}
}


int main()
{
	vector<int> v;
	sieve_of_eratosthenes(v,100000);
	int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		bool check = false;
		for(int i=2;i<=n/2;i++)
		{
			if(v[i]==1)
			{
				if(v[n-i]==1)
				{
					check = true;
					break;
				}
			}
		}
		if(check == false)
			cout<<"Arjit"<<endl;
		else
			cout<<"Deepa"<<endl;
			
	}
	return 0;
}
