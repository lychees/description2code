for t in xrange(input()):
    n, k = map(int, raw_input().split())
    x = map(int, raw_input().split())
    temp = 100000000000
    for i in x:
        cost = []
        for l in x:
            if i > l:
                cost.append((i-l)*3)
            else:
                cost.append((l-i)*5)
        cost.sort()
        temp = temp if temp < sum(cost[0:k]) else sum(cost[0:k])
    print temp
