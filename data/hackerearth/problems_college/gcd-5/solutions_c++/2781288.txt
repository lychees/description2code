    using namespace std;
    ll int a[100005];
    int main()
    {
        ll int t,n,i;
        cin >> t;

        while(t--){
            cin >> n;
            for (i = 1; i <= n; i++){
                cin >> a[i];
            }
            ll int prev = a[1];
            ll int flag = 0;
            for (i = 2; i <= n; i++){
                if(a[i] < prev){
                    flag = 1;
                    break;
                }
                else {
                    prev = a[i];
                }
            }
            if(flag){
                cout << "NO"<<endl;
            }
            else {
                cout << "YES"<<endl;
            }
        }
    }
