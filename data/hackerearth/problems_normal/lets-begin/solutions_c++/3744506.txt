llt min2 ( llt a , llt b )
{
	if ( a < b )
		return a ;
	return b ;
}

llt min4 ( llt a , llt b , llt c , llt d )
{
	llt x = min2 ( a , b ) ;
	llt y = min2 ( c , d ) ;
	return min2 ( x , y ) ;
}
