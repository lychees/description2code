mx = 1000001
l = []
ans = {}
for i in xrange(mx):
    l.append(0)
for i in xrange(2, mx, 1):
    ct = 0
    if l[i] == 0:
        for j in xrange(i*i, mx, i):
            if l[j] == 0:
                l[j] = 1
                ct += 1
    ans[i] = ct+1

tc = int(raw_input())

for i in xrange(tc):
    n = int(raw_input())
    print ans[n]
