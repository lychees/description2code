t = int(raw_input())
while (t > 0):
    r, c = map(int, raw_input().split(" "))
    res_odd = 0
    res_even = 0
    for i in range(1, r+1):
        if (i < c):
            if (i % 2 == 1):
                res_odd += i
            else:
                res_even += i
        else:
            if (i % 2 == 1):
                res_odd += c
            else:
                res_even += c
    print str(res_even)+" "+str(res_odd)
    t -= 1
