#include<iostream>

using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,i,j,a;
		cin>>n;
		for(i=n;;i++)
		{
			a = i;
			j = 0;
			while(a>0)
			{
				if(j==1 && a%2==1)
				{
					a = -100;
				}
				j = a%2;
				a = a/2;
			}
			if(a==0)
			{
				cout<<i<<endl;
				break;
			}
		}
	}
	return 0;
}
