	boost;

	int i,j,k,x,y,n,l,t;
	pii a[2000];
	cin>>t;
	while(t--){
		bool possible=false;
		cin>>n>>l;
		j=0;
		forp(i,1,n){
			cin>>x>>y;
			if(y-x<l)
			{
				a[j]=mp(x,y);
				j++;
			}
			else if(y-x==l)
			possible=true;
		}
		if(possible)
		{
			cout<<"Yes\n";
			continue;
		}
		sort(a,a+j);
		for(i=0;i<j;i++){
			int left=a[i].ff;
			int right=a[i].ss;
			for(k=i+1;k<j;k++){
				if(right >=a[k].ff and a[k].ss <= left+l)
				right = max(right,a[k].ss);
			}
			if(right==left+l)
			{
				possible=true;
				break;
			}
		}
		if(possible)
		cout<<"Yes\n";
		else
		cout<<"No\n";
	}
	return 0;
}
