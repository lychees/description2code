#include <bits/stdc++.h>
#include <algorithm>
using namespace std;

long max(long a, long b)
{
	if(a>b)
		return a;
	else 
		return b;
}
 
int main()
{
    long k,i,ans=0,j;
    scanf("%d",&k);
    long a[k];
    for(i=0;i<k;i++)
    {
    	scanf("%ld",&a[i]);
    }
    sort(a,a+k);
    for(i=k-1,j=0;i>=0;i--,j++)
    {
    	ans = max(ans, j+a[i]+1);
    }
    ans++;
    printf("%ld",ans);
    return 0;
}
