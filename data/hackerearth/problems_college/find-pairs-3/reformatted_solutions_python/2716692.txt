t = int(raw_input())


def result(val):
    a = map(int, raw_input().split())
    a = list(a)
    c = len(a)
    count = len(a)
    for i in range(c-1):
        for j in range(i+1, c):
            if a[i] == a[j]:
                count += 1

    print(count)


for _ in range(t):
    val = int(raw_input())
    result(val)
