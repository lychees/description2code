#include <iostream>
#include <string.h>

using namespace std;

int main()
{
	char a[26]={'q','w','e','r','t','y','u','i','o','p','a','s','d','f','g','h','j','k','l','z','x','c','v','b','n','m'};
	int t;
	char s[101];
	cin>>t;
	t=t+1;
	while(t--)
	{
		cin.getline(s,101);
		int l=strlen(s);
		for(int i=0;i<l;i++)
		{
			if(s[i]==' ')
			cout<<" ";
			else
			cout<<a[s[i]-'a'];
		}
		cout<<endl;
	}
	return 0;
}
