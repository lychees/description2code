#include <iostream>
using namespace std;

int main()
{
    int t,k;
    long long n, m, a, b, x, y;
    cin>>t;
    
    while(t--)
    {
    	cin>>n>>m;
    	long long count=0;
    	cin>>k;
    	x=y=1;
    	
    	for(int i=0; i<k; i++)
    	{
    		cin>>a>>b;
    		
    		while( x+a<=n && x+a>=1 && y+b<=m && y+b>=1)
    		{
    			x += a;
    			y += b;
    			count++;
    		}
    	}
    	cout<<count<<endl;
    }
    return 0;
}
