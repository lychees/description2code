#include<iostream>
#include<algorithm>
#include<utility>
#include<queue>
using namespace std;
void sol()
{
    vector<pair<long long ,long long > > v;
    priority_queue< long long ,vector<long long >,greater<long long > > p;
    long long int i,n,total=1,temp,x,arr[100001];
    cin>>n;
    for(i=0;i<n;i++)
        cin>>arr[i];
    for(i=0;i<n;i++)
    {
        cin>>x;
        v.push_back(make_pair(arr[i],arr[i]+x));
    }
    sort(v.begin(),v.end());
    p.push(v[0].second);
    for(i=1;i<n;i++)
    {
        x=v[i].first;
        temp=p.top();
        if(temp<=x)
        {
            p.pop();
            p.push(v[i].second);
        }
        else
        {
            total++;
            p.push(v[i].second);
        }
    }
    cout<<total<<endl;
}
int main()
{
    int t;
    cin>>t;
    while(t--) sol();
    return 0;
}
