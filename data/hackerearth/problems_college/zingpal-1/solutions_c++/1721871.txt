#include <bits/stdc++.h>
using namespace std;
const int maxN = 100321;
vector<long long > a[maxN];
long long  mark[maxN];
void dfs(long long  v)
{
  mark[v]=1;
  for(long long  i=0;i<a[v].size();i ++)
  	if(!mark[a[v][i]]) 
		dfs(a[v][i]);
}
int main()
{
  	long long  n,m; 
  	cin>>n>>m;
  	for(long long  i=0;i<m;i++){
      	long long  d,k; 
	  	cin>>d>>k; 
	  	d--; 
	  	k--;
      	a[d].push_back(k); 
	  	a[k].push_back(d);
	}
  	dfs(0);
  	if(m != n){ 
	  	cout<<"NO"<<endl; 
		return 0;
	}
  	for(long long  i = 0 ; i < n; i ++) 
	  if(!mark[i]){ 
	  	cout<<"NO"<<endl; 
		  return 0; 
	}
  	cout<<"FHTAGN!"<<endl;
}
