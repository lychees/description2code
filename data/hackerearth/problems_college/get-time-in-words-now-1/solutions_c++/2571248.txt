#include<iostream>
#include<cstdlib>
using namespace std;
void expand(int value)
{
    const char * const ones[20] = {"zero", "one", "two", "three","four","five","six","seven",
    "eight","nine","ten","eleven","twelve","thirteen","fourteen","fifteen","sixteen","seventeen",
    "eighteen","nineteen"};
    const char * const tens[10] = {"", "ten", "twenty", "thirty","forty","fifty","sixty","seventy",
    "eighty","ninety"};

    if(value<0)
    {
        cout<<"minus ";
        expand(-value);
    }
    else if(value>=1000)
    {
        expand(value/1000);
        cout<<" thousand";
        if(value % 1000)
        {
            if(value % 1000 < 100)
            {
                cout << " and";
            }
            cout << " " ;
            expand(value % 1000);
        }
    }
    else if(value >= 100)
    {
        expand(value / 100);
        cout<<" hundred";
        if(value % 100)
        {
            cout << " and ";
            expand (value % 100);
        }
    }
    else if(value >= 20)
    {
        cout << tens[value / 10];
        if(value % 10)
        {
            cout << " ";
            expand(value % 10);
        }
    }
    else
    {
        cout<<ones[value];
    }
    return;
}
main()
{
    int h,m;
    cin>>h;
    cin>>m;
    if(m==1)
    {
        expand(m);cout<<" minute past ";expand(h);
    }
    else if(m==0)
    {
        expand(h);cout<<" o' clock";
    }
    else if(m==15)
    {
        cout<<"quarter past ";expand(h);
    }
    else if(m==45)
    {
        cout<<"quarter to ";expand(h+1);
    }
    else if(m==30)
    {
        cout<<"half past ";expand(h);
    }
    else if(m<30)
    {
        expand(m);cout<<" minutes past ";expand(h);
    }
    else if(m>30)
    {
        expand(60-m);cout<<" minutes to ";expand(h+1);
    }

    else
        exit(0);
}
