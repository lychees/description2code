#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t; cin >> t;
	while(t--)
	{
		int n; cin >> n;
		int dp[n+1];
		
		dp[0] = 0;
		for(int i=1;i<=n;i++)
		{
			if( i< 10 && i < 12) dp[i] = -1;
			else
			{
				if( i == 10 || i == 12) dp[i] = 1;
				else if(i > 10 && i < 12)
				{
					dp[i] = -1;
				}
				else if(i > 12)
				{
					if(dp[i-12] == -1 && dp[i-10] == -1) dp[i] = -1;
					else if(dp[i-12] == -1) dp[i] = 1+dp[i-10];
					else if(dp[i-10] == -1) dp[i] = 1+dp[i-12];
					else dp[i] = 1+min(dp[i-10], dp[i-12]);
				}
			}
		}
		
		cout << dp[n] << endl;
	}
}
