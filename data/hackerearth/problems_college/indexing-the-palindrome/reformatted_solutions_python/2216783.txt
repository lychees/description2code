def is_pal(s):
    h = len(s) / 2
    return s[:h] == s[:-(h+1):-1]


for _ in range(int(raw_input())):
    s = raw_input().strip()
    if is_pal(s):
        print -1
        continue
    l = len(s) - 1
    i = 0
    while s[i] == s[l - i]:
        i += 1
    if is_pal(s[:i] + s[i+1:]):
        print i
    else:
        print l - i
