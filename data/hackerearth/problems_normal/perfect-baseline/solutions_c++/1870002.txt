#include <bits/stdc++.h>
using namespace std;
char bestchar(int j,string arr[],int n){
	int mini=1000000;
	char m;
	for(int c='a';c<='z';c++){
		int cost=0;
		for(int i=0;i<n;i++){
			cost+=abs(c-arr[i][j]);
		}
			
		if(cost<mini){
			mini=cost;
			m=c;	
		}	
	}
	return m;
}
int main(){
	int t,n, k,i,j;
	scanf("%d",&t);
	while(t--){
		scanf("%d %d",&n,&k);
		string arr[n];
		for(i=0;i<n;i++)	cin>>arr[i];
		string res="";
		for(i=0;i<k;i++)	res+=bestchar(i,arr,n);
		cout<<res<<endl;
	}
	return 0;
}
