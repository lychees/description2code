#include<iostream>
#include<string.h>
using namespace std;
int main(){
	char s[1001];
	cin.getline(s,1001,'\n');
	int l=0;
	char x;
	int hash[255]={0};
	for(int i=0;i<strlen(s);++i){
		hash[s[i]]++;
	}
	for(int i=0;i<strlen(s);++i){
		if(l<hash[s[i]]){
			l=hash[s[i]];
			x = s[i];
		}	
		else if(l==hash[s[i]]&&s[i]<x){
		l=hash[s[i]];
		x = s[i];
	}
	}
	cout<<x<<" "<<l<<endl;
}
