#include <bits/stdc++.h>
#include <string.h>
#define l long int
using namespace std;

int main(int argc, char const *argv[]) {
  char compNum[100000],userNum[100000];
  l diff=0,i;
  cin>>compNum;
  cin>>userNum;
  for(i=0;i<strlen(compNum);i++)
    if(compNum[i]!=userNum[i])
      ++diff;

  if(diff>1)
  {
    std::cout << "loses" << std::endl;
    exit(0);
  }
  else {
      std::cout << "wins" << std::endl;
  }
  return 0;
}
