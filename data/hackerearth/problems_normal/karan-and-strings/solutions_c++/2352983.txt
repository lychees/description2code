#include <iostream>
using namespace std;

int main()
{
 
 int t;
 cin>>t;
 int i=0;
 string str;
 while(i<t)
 {
 	cin>>str;
 	int n=str.size();
 	string ans;
 	ans=ans+str[0];
 	for(int i=1;i<n;i++)
 	{
 		if(str[i]!=str[i-1])
 		ans=ans+str[i];
 	}
 	
 	cout<<ans<<"\n";
 	
 	i++;
 }
 
 
 //   cout << "Hello World!" << endl;
    return 0;
}
