#include<bits/stdc++.h>
#define MOD 1000000007
using namespace std ;
int sum_a[400002] , sum_b[400002] , a[201] , b[201] ;
long long set_a[400002] , set_b[400002] ;
int main () {
    int n ;
    scanf ( "%d" , &n ) ;
    for ( int i = 0 ; i < n ; i++ ) {
        scanf ( "%d" , &a[i] ) ;
    }
    for ( int i = 0 ; i < n ; i++ ) {
        scanf ( "%d" , &b[i] ) ;
    }
    sum_a[0] = sum_b[0] = 1 ;
    set_a[0] = set_b[0] = 1 ;
    for ( int i = 0 ; i < n ; i++ ) {
        for ( int j = 400000 ; j >= 0 ; j-- ) {
            if ( sum_a[j] ) {
                sum_a[j+a[i]] = 1 ;
                set_a[j+a[i]] += set_a[j] ;
                set_a[j+a[i]] %= MOD ;
            }
            if ( sum_b[j] ) {
                sum_b[j+b[i]] = 1 ;
                set_b[j+b[i]] += set_b[j] ;
                set_b[j+b[i]] %= MOD ;
            }
        }
    }
    for ( int i = 1 ; i <= 400000 ; i++ ) {
        set_b[i] += set_b[i-1] ;
        set_b[i] %= MOD ;
    }
    int q ;
    scanf ( "%d" , &q ) ;
    q = min ( q , 400000 ) ;
    int ans = 0 ;
    for ( int i = 0 ; i <= 400000 ; i++ ) {
        if ( sum_a[i] ) {
            int k = set_b[ min ( 400000 , i+q ) ] ;
            if ( i-q-1 >= 0 )
                k = k - set_b[i-q-1] ;
            ans = ((ans%MOD)+(((set_a[i]%MOD)*(k%MOD))%MOD))%MOD ;
        }
    }
    if ( ans < 0 ) {
        ans = (ans + MOD)%MOD ;
    }
    printf ( "%d\n" , ans ) ;
    return 0 ;
}
