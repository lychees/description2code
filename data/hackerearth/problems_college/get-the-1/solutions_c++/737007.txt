#include<iostream>
using namespace std;
int get(int n){
    int r,s=0;
    while(n){
        r=n%10;
        s += r*r;
        n=n/10;
    }
    return s;
}

int main()
{
    int t,k,i,n;
    cin>>t;
    for(i=1;i<=t;i++){
        cin>>n;
        k=n;
        while(n/10 !=0){
            n=get(n);
            if(n==k || n==1 || n==19)
                break;
        }
        if(n==1)
            {cout<<"Case #"<<i<<":YES\n";}
        else
            {cout<<"Case #"<<i<<":NO\n";}
    }
    return 0;
}
