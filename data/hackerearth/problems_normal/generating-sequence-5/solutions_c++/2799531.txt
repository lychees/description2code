#include <iostream>
#define ll long long int
using namespace std;

int main()
{
    int t;
    ll g;
    int n;
    cin>>t;

    {
    	cin>>g>>n;
    	ll x=g*2;
    	ll y=g*3;
    	//cout<<x;
    	for(ll i=0;i<n-1;i++)
    	{
    		cout<<x<<" ";
    	}
    	cout<<y;
    	cout<<endl;
    }
    return 0;
}
