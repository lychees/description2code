#include <bits/stdc++.h>
using namespace std;
#define MIN -2
int dist[50005];
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	memset(dist,0,sizeof(dist));
    	for(int i=1;i<=n;i++)
    	{
    		int temp;
    		scanf("%d",&temp);
    		dist[temp]++;
    	}
    	int j=0;
    	int count=0;
    	int k=0;
    	int inc=0;
    	for(int i=1;i<=n;i++)
    	{
    		while(dist[j]==0)
    		j++;
    		dist[j]--;
    		if(j-inc<=0)
    		{
    			k=1;
    			break;
    		}
    		if(i%6==0&&i!=0)
    		{
    			inc++;
    		}
    		inc++;
    		count++;
    	}
    	if(k==0)
    	{
    		cout<<"Rick now go and save Carl and Judas"<<endl;
    	}
    	else
    	{
    		printf("Goodbye Rick\n");
    		printf("%d\n",count);
    	}
    }
    return 0;
}
