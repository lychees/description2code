#include <bits/stdc++.h>

using namespace std;

int main(){

    

    long T;
    cin>>T;
    assert(T>=1 && T<=25);

    while(T--){

        long N;
        cin>>N;
        assert(N>=3 && N<=30);

        vector< float > D(N), F(N);
        vector< float > ans(N);

        float p = 0.0;

        for(int i=0; i<N; i++)
            cin>>D[i];
        for(int i=0; i<N; i++)
            cin>>F[i];

        for(int i=0; i<N; i++)
            p += (F[i] * D[i]);

        for(int i=0; i<N; i++)
            ans[i] = (F[i] * D[i])/p;

        int index = -1;
        float m = 0.0;
        for(int i=0; i<N; i++)
            if(m < ans[i]){
                index = i;
                m = ans[i];
            }

        cout<<index<<" ";
        printf("%.3f", ans[index]);
        cout<<endl;
    }

    return 0;
}
