#include <iostream>
#include<string>
using namespace std;
int main()
{
    int N;
    cin>>N;
    string str;
    cin>>str;
    long int min=0,max=0;
    for(int i=0;i<str.length();i++){
    	if(str[i]=='v' || str[i]=='w' ){
    		int w_count=0,v_count=0;
    		while(str[i]=='v' || str[i]=='w'){
    			if(str[i]=='v')
    				v_count++;
    			else if(str[i]=='w')
    				w_count++;
       			i++;
       			if(i>=str.length())
       				break;
    		}
    		//cout<<min<<" "<<max<<endl;
    		//cout<<"W_count : "<<w_count<<" V_count : "<<v_count<<" "<<str[i]<<endl;
    		max+=(v_count+2*w_count);
    		min+=((w_count*2+v_count)/2)+(w_count*2+v_count)%2;
    		//cout<<str[i]<<" "<<min<<" "<<max<<endl;
    		i--;
    		continue;
    		}
    	max++;
    	min++;	
    }
   	cout<<min<<" "<<max<<endl;
    return 0;
}
