#include<bits/stdc++.h>
using namespace std;
bool comp(int hash1[],int hash2[])
{
    for(int i=0;i<=26;i++)
    {
        if(hash1[i]!=hash2[i])
        return 0;
    }
    return 1;
}
int main() {
	// your code goes here
	int T;
	cin>>T;
	while(T--)
	{
	    string A,B;
	    cin>>A;
	    cin>>B;
	    int n1=A.size();
	    int n2=B.size();
	    int hash1[27]={0};
	    int hash2[27]={0};
	    int fl=0;
	    for(int i=0;i<n1;i++)
	    {
	        hash1[A[i]-'a']++;
	        hash2[B[i]-'a']++;
	    }
	    if(comp(hash1,hash2))
	    {
	        cout<<"YES\n";
	    }
	    else
	    {
	        int j=0;
	        for(int i=n1;i<n2;i++,j++)
	        {
	            hash2[B[j]-'a']--;
	            hash2[B[i]-'a']++;
	            if(comp(hash1,hash2))
	            {
	                cout<<"YES\n";
	                fl=1;
	                break;
	            }
	        }
	        if(fl==0)
	        cout<<"NO\n";
	    }
	}
	return 0;
}
