#include <bits/stdc++.h>
#define fast ios_base::sync_with_stdio(0);cin.tie(0)
using namespace std;
vector<int> a[30];

void solve(string s)
{
	for(int i = 0; i <= ((int)(s.length())-1); i++)
	{
		a[s[i]-97].push_back(i);
	}
}

int binary_search_util(int m, int n)
{
	int low = 0, high = (((int)a[m].size())-1); 
	while (low != high) 
	{
    	int mid = (low + high) / 2; 
    	if (a[m][mid] <= n) 
    	{
        low = mid + 1;
    	}
    	else
    	{
        high = mid;
    	}
	}
	
	return a[m][high];
	
}


bool binary_search(string x)
{	
	if(((int)a[x[0]-97].size())==0)
		return false;
		
	int old_pos = a[x[0]-97][0];
	//cout<<old_pos<<endl;
	int new_pos = 0;
	for(int i = 1; i <= x.length()-1; i++)
	{	
		if(((int)a[x[i]-97].size())==0)
		return false;
		
		new_pos = binary_search_util(x[i]-97, old_pos);
		
		//cout<<new_pos<<endl;
		
		if(new_pos<=old_pos)
		return false;
		
		old_pos = new_pos;
	}
	
	return true;
	
}

int main()
{	fast;
	string s;
	cin>>s;
	solve(s);
	int q;
	cin>>q;
	for(int i = 0; i<= q-1; i++)
	{
		
		string a;
		cin>>a;
		if(binary_search(a))
		cout<<"Yes\n";
		else cout<<"No\n";
		
	}
    return 0;
}
