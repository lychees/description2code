#include <iostream>

#include <cmath>
using namespace std;
#define fi "INPUT.INP"

#define foru(i,l,r) for(int i=l;i<=r;++i)
int n,s;
int factor(int x){
	int ans=0;
	foru(i,2,int(sqrt(x))){
		while ((x%i)==0){
			++ans;
			x/=i;
		}
	}
	if (x>1) ++ans;
	return ans;
}
int main(){
    //freopen(fi,"r",stdin);
    //freopen(fo,"w",stdout);
	scanf("%d",&n); s=0;
	foru(i,1,n){
	    int x;
		scanf("%d",&x);
		int num=factor(x);
		s=s^num;
	}
	if (s==0) printf("BOB");
	else printf("ALICE");
	fclose(stdin);
	fclose(stdout);
}
