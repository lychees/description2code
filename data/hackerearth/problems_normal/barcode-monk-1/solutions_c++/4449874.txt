#include <bits/stdc++.h>
using namespace std;
const int MOD =1e9 +7;
int n,k,w[1003],i,j,super[1003],u[1003],p[1003];
int main()
{
	cin>>n>>k;
	for(i=0;i<k;i++){
		cin>>w[i];
	}
	super[0]=1;
	for(i=0;i<k;i++){
		for(j=w[i];j<=n;j++){
			super[j] |= super[j-w[i]];
		}
	}
	u[0]=1;
	for(i=1;i<=n;i++){
		u[i] = p[i-1] + u[i-1];
		u[i]%=MOD;
		for(j=1;j<=i;j++){
			if(super[j]==1){
				p[i] += u[i-j];
				p[i]%=MOD;
			}
		}
	}
	cout<<(p[n]+u[n])%MOD <<endl;
    return 0;
}
