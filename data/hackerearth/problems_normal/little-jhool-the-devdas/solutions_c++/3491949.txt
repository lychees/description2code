#include<iostream>
#include<algorithm>
using namespace std;
int main(){
    int n;
    cin>>n;
    int a[n];
    int count=0;
    int sum=0;
    for(int i=0;i<n;i++){
        cin>>a[i];
        sum=sum+a[i];
        if(a[i]%2==0)
            count++;            //if all numbers are even then its not possible so count==n then :(
    }
    sort(a,a+n);
    if(count==n)
        cout<<":("<<endl;
    else{
            if(sum%2!=0)
            cout<<sum<<endl;
            else{
                 for(int i=0;;i++){
                    sum=sum-a[i];
                    if(sum%2!=0)
                        break;
                    else
                        sum=sum+a[i];
                }
                cout<<sum<<endl;
            }
    }

return 0;
}
