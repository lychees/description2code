#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		long long m;
		int n1,n2;
		scanf("%lld %d %d",&m,&n1,&n2);
		long long int a[101],b[101];
		for(int i=0;i<n1;i++){
			scanf("%lld",&a[i]);
		}
		for(int i=0;i<n2;i++){
			scanf("%lld",&b[i]);
		}
		sort(a,a+n1);
		bool found=false;
		vector<long long int> ans;
		long long ma=0;
		for(int i=0;i<(1<<n2);i++){
			long long sum=0;
			for(int j=0;j<n2;j++){
				if((i>>j)&1){
					sum+=b[j];
				}
			}
			///vector<long long int> ans1;
			if(sum<=m){
				sum=m-sum;
				long long sum1=sum;
				vector<long long int> temp;
				for(int i=n1-1;i>=0;i--){
					if(a[i]<=sum){
						sum-=a[i];
						temp.push_back(a[i]);
					}
				}
				if(sum==0&&ma<sum1){
					found=true;
					ma=sum1;
					ans.clear();
					for(int i=0;i<temp.size();i++){
						ans.push_back(temp[i]);
					}
				}
			}
		}
		if(found){
		printf("YES\n");
		sort(ans.begin(),ans.end());
		for(int i=0;i<ans.size();i++)
		     printf("%lld ",ans[i]);
		   if(ans.size()>0)
		   printf("\n");
		}
		else
		printf("NO\n");
	}
	return 0;
}
