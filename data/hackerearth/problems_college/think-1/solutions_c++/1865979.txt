#include<iostream>
#include<string>
#include<math.h>
#include<algorithm>
using namespace std;

int main()
{
   int t,n;
   cin>>t;
   while(t--)
   {
     cin>>n;
     string s  = to_string(n*n);
     string r  = to_string(n*n);
     reverse(r.begin(),r.end());
     if(r.compare(s) == 0) 
     {
        cout<<"No";
     }
     else
     {
         int ri = stoi(r);
         int x  = (int)sqrt(ri);
         if(x*x == ri && x!=n) 
         {
         	 string p = to_string(x);
         	 string q = to_string(n);
         	 reverse(p.begin(),p.end());
         	 if(p.compare(q) ==0)
         	   cout<<x;
         	 else cout<<"No";
         }
         else cout<<"No";
     }
     cout<<endl;
   }
}
