#include <iostream>
using namespace std;

int n,dp1[1000000],dp2[1000000];

int solve1(int i)
{
   if(i>n)
   	return 0;
   if(i==n)
   	return 1;
   if(dp1[i]!=-1)
   	return dp1[i];
   int ans=solve1(i+1)+solve1(i+2)+solve1(i+3);
   dp1[i]=ans;
   return dp1[i];
}

int solve2(int i)
{
   if(i>n)
   	return 0;
   if(i==n)
   	return 1;
   if(dp2[i]!=-1)
   	return dp2[i];
   int ans=solve2(i+1)+solve2(i+2);
   dp2[i]=ans;
   return dp2[i];
}

int main()
{
    cin>>n;
    if(n<=0)
    {
    	cout<<"ERROR";
    	return 0;	
    }
    for(int i=0;i<=n;i++)
    {
    	dp1[i]=-1;
    	dp2[i]=-1;
    }
    cout<<"Jack-"<<solve1(0)<<endl;
    cout<<"Jill-"<<solve2(0);
    return 0;
}
