#include <iostream>
#include <algorithm>
using namespace std;
unsigned int numbers[100];
int main()
{
    //cout << "Hello World!" << endl;
    unsigned int numTestCases;
    cin>>numTestCases;
    for(unsigned int i = 0;i<numTestCases;i++)
    {
    	unsigned int num = 0;
    	cin>>num;
    	for(unsigned int j = 0;j<num;j++)
    	    cin>>numbers[j];
    	    
    	sort(numbers, numbers+num);
    	
    	long int count = 0;
    	for(unsigned int j = 0;j<num;j++)
    	{
    		if(j == num-1)
    		{
    			count++;
    			continue;
    		}
    		
    		if(numbers[j+1] == (numbers[j]+1))
    			j++;
    		count++;
    		
    	}
    	cout<<count<<endl;
    }
  
    return 0;
}
