#include<bits/stdc++.h>
using namespace std;

int setbits(int n)
{
    int counter=0;

    while(n)
    {
        counter += n&1;
        n=n>>1;
    }

    return counter;
}

bool cmp( pair<int,pair<int,int> > a , pair<int,pair<int,int> > b )
{

    if(a.first == b.first )
        return a.second.second < b.second.second;
    else
        return a.first>b.first;

}

int main()
{
    int t;
    cin>>t;

    while(t--)
    {
        int n,k;
        cin>>n>>k;

        vector< pair<int,pair<int,int> > > v;

        for(int i=0;i<n;i++)
        {
            int a;
            cin>>a;
            v.push_back( make_pair(setbits(a) ,make_pair(a,i)));
        }

        sort(v.begin(),v.end(),cmp);

        int counter=0;
        for(int i=0;i<k;i++)
            counter+=v[i].first;
        cout<<counter<<endl;
    }
}
