{

        return 0;
    if(v1[i]>v2[j])
    {
        summ= 100+fun(i+1,j+1,summ);
    }
    else if(v1[i]==v2[j])
    {
        summ=( max(50+fun(i+1,j+1,summ)  , fun(i,j+1,summ)) );

    }
    else
    {
        summ=0+fun(i,j+1,summ);
    }
    return summ;
}
