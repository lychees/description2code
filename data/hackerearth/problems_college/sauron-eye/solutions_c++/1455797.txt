#include<iostream>
using namespace std;
#define ll long long
#include <math.h>
ll mod = pow(10,9)+7;
void matmux(ll a[2][2],ll b[2][2])
{
    ll c[2][2];
    int i,j,k;
    int n=2;
	
     for(i=0;i<n;i++) {
        for(j=0;j<n;j++) {
        	c[i][j]=0;
        	
               for(k=0;k<n;k++) {
                    c[i][j]=((c[i][j]%mod)+((a[i][k]%mod)*(b[k][j]%mod))%mod)%mod;
                }
                
        }
     }

    for(i=0;i<n;i++) {
        for(j=0;j<n;j++){
            a[i][j]=c[i][j];
        }
    }

}
void power(ll a[2][2],ll b)
{
    ll m [2][2]={{3,-1},{1,0}};
    if(b==0|| b==1)
        return;

    power(a,b/2);
    matmux(a,a);

    if(b%2!=0)
        matmux(a,m);
}
ll fibo(ll n)
{
    ll a[2][2]={{3,-1},{1,0}};
    if(n==0)
        return 0;
    else
        if(n==1)
            return 1;
    else
        power(a,n-1);

    return (a[0][0]);
}
int main()
{
    int t,n;
    cin >> t;
    ll ans;
    while(t--){
        cin >> n;
        ans= fibo(n);
        if(fibo(n)<0)
           ans=ans+mod;
        cout << ans%mod<< endl;
    }
    return 0;
   }
