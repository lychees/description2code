#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	int flag=0;
	int count;
	for(int i=0; i<t; i++)
	{
		count=0;
		int n;
		cin>>n;
		
		for(int i=1;i<=n-1;i++)
   			
   			for(int j=i+1;j<=n;j++)
   				{   int d=i^j;
   					if(d<=n)
   					{
   						count++;
   					}
   				}
		cout<<count<<endl;
	} 
   
    return 0;
}
