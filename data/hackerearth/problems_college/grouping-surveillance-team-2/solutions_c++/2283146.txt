#include <bits/stdc++.h>
#define ll long long int

using namespace std;

ll a[1000][1000];

ll fun(int k,int n)
{
	if(k>n)
		return 0;
	else if(k==n)
		return 1;
	else if(a[k][n])
		return a[k][n];
	a[k][n]=fun(k+1,n)+fun(k,n-k);
	return a[k][n];
}

int main()
{
	ios_base::sync_with_stdio(0);
	int t,n,k;
	cin>>t;
	while(t--)
	{
		cin>>n;
		k=2;
		cout<<fun(k,n)<<"\n";
	}
	return 0;
}
