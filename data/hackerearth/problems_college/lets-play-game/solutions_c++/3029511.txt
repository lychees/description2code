#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long int n,it,t,k,i,ans;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld%lld",&n,&k);
        long long int a[n],b[n];
        for(i=0;i<n;i++)
            scanf("%lld",&a[i]);
        for(i=0;i<n;i++)
            scanf("%lld",&b[i]);
        ans=-1;
        for(i=0;i<n;i++)
        {
            it=(k/a[i])*b[i];
            if(it>ans)
                ans=it;
        }
        printf("%lld\n",ans);
    }
    return 0;
}
