#include <bits/stdc++.h>
#include <stdio.h>
using namespace std;
int main() {
  int t;
  scanf("%d", &t);
  while (t--) {
    string a;
    cin >> a;
    int lm = a.size();
    string b = "";
    b = b + a[0];
    int i;
    for (i = 1; i < lm; i++) {
      if (a[i] < b[0])
        b = b + a[i];
      else
        b = a[i] + b;
    }
    cout << b << endl;
  }
  return 0;
}
