#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
struct lottery
{
       int s,e,p; // s-> start time, e -> end time , p -> probability 
};
struct lottery a[109];
float dp[109];
bool my_func(struct lottery x,struct lottery y)
{
        return ( (x.s < y.s) || ( (x.s == y.s) && (x.e < y.e) ) );   
}
float max(float a,float b)
{
        return a>b?a:b;
}
int main()
{
        int i,j,k,t,n;
        float ans,m;
        scanf("%d",&t);
        while(t--)
        {
                scanf("%d",&n);
                for (i = 0; i < n; ++i)
                        scanf("%d%d%d",&a[i].s,&a[i].e,&a[i].p);
                sort(a,a+n,my_func);
                dp[0]=(float)a[0].p;
                ans=dp[0];
                for (i = 1; i < n; ++i)
                {
                        m=0.0;
                        dp[i]=(float)a[i].p;
                        for (j = i-1; j >= 0 ; --j)
                        {
                                if(a[j].e <= a[i].s)
                                        m=max(m,dp[j]);
                        }
                        dp[i]+=m;
                        ans=max(ans,dp[i]);
                }
                ans=ans/(float)100;
                printf("%.2f\n",ans);
        }
        return 0;
}
