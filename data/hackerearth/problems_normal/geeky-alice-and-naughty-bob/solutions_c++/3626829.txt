#include <iostream>
using namespace std;
typedef long long int ll;
int rsf[7]={0,1,2,6,6,3,9};
int Rsum(int n){
	int ans=0;
	while(1){
		while(n){
			
			ans+=n%10;
			n/=10;
		}
		if( ans>10 ){
		 n=ans;
		 ans=0;
		}
	}
	return ans;
}

int factRsum(int n){
	int f=9;
	if(n<=5)
	f=rsf[n];

	return f;
}
ll findSum(int a,int b){
	ll sum=0;
	for(int i=a;i<=b;i++){
		sum+= factRsum(i);
	}
	return sum;
}
int main()
{
 int t;
 cin>>t;
 while(t--)
 {
 	 int a,b;
 	 cin>>a>>b;
 	 ll ans;
 	 ans= findSum(a,b);
 	cout<<ans<<endl;
 }
    return 0;
}
