#include <bits/stdc++.h>

using namespace std;

#define ll long long
int main()
{
	ios_base::sync_with_stdio(0);
	ll n, m;
	cin >> n >> m;
	ll sum1, sum2, temp;
	sum1 = sum2 = 0;
	for(int i = 0; i < n; ++i){
		cin >> temp;
		sum1 += temp;
	}
	for(int i = 0; i < m; ++i){
		cin >> temp;
		sum2 += temp;
	}
	
	cout << sum1*sum2 << "\n";
}
