#include<stdio.h>

#include<string.h>
int main()
{
	int t;
	scanf("%d",&t);
	while(t>0)
	{
		long long n,i,del=0;
		char str[100000];
		//gets(str);
        scanf("%s",str);
		n=strlen(str);
		for(i=0;i<n-1;i++)
		{
			if(str[i]==str[i+1])
			{
				del++;
			}
		}
		printf("%lld\n",del);
        t--;
	}
    return 0;
}
