#include <iostream>
using namespace std;

int main()
{
    long int n, colour[100000][2], games[100000][2];
    cin>>n;
    for (int i=0; i<n; i++)
    	for (int j=0; j<2; j++)
    		cin>>colour[i][j];
    
    for (int i=0; i<n; i++)
    {
    	for (int j=0; j<n; j++)
    	{
    		if (j==i)
    			continue;
    		else
    		{
    			games[i][0]++;
    			if (colour[i][0]==colour[j][1])
    			{
    				games[j][0]++;
    			}
    			else
    			{
    				games[j][1]++;
    			}
    		}
    	}
    }
    for(int i=0; i<n; i++)
    {
    	for (int j=0; j<2; j++)
    		cout<<games[i][j]<<" ";
    	cout<<"\n";
    }
    		
    return 0;
}
