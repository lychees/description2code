#include <bits/stdc++.h>
using namespace std;
 
int main()
{
 long long n,i,j,a,ans;
 long long A[1000001]={0};
 ans=0;
 a=0;
 scanf("%lld",&n);
 for(i=0;i<n;i++)
  scanf("%lld",&A[i]);
 if(n>=1000)
 {
 for(i=n-1;i>=0;i--)
  A[i]=A[i+1]^A[i];
 for(i=0;i<n;i++)
  ans=ans|A[i];
 cout<<ans;
 }
 else
 {
  for(i=0;i<n-1;i++)
  {
  	for(j=i+1;j<n;j++)
  	{
  		a=A[i]^A[j];
  		ans=ans|a;
  	}
  }
  cout<<ans;
 }
 return 0;
}
