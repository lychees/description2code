#include <iostream>
using namespace std;

long long a[1001];

int main()
{
    int t;
    cin >> t;
    a[0] = 0;
	a[1] = 1;
	
	for(int i = 2; i < 1001 ;i++) {
		for(int j = 0 ;j < i-1 ; j++) {
			a[i] += 2*a[j];
			a[i] %= 1000000007;
		}
		a[i]++;
	} 
	
    while(t--) {
    	int n;
    	cin >> n;
    	cout << a[n] << endl;
    }
    return 0;
}
