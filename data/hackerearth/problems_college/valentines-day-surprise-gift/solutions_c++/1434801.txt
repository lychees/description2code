const int m=1000000007;
long long compute(float n)
{
	if(n==0||n==1||n==2)
	return 1;
	else if(n>=3&&n<5)
	return 3;
	else if(n>=5&&n<7)
	return 15;
	else if(n>=7&&n<=10)
	return 105;
	else
	{
		return (int((pow(2,ceil(n/4)))*compute(ceil(n/5))*compute(ceil(n/10))))%m;
	}
}
long long gcd(long long a, long long b)
{
  if (b == 0)
    return a;
  else
    return gcd(b, a%b);
}
long long power(long long a,long long b)
{
long long res=1;
a=a%m;
 while(b>0)
 {
 if(b&1)
  res=(res*a)%m;
  a=(a*a)%m;
  b>>=1;
 }
 return (res)%m;
}
