#include<bits/stdc++.h>

using namespace std;

int main()
{
	int n,i,j;
	cin>>n;
	int a;
	
	int b[n],c[n],l=0,k=0,s=0,s1=0;
	for(i=0;i<n;i++)
	{
		cin>>a;
		if(a%2==0)
		{
     	b[k]=a;
		k++;
	    }
		else {
	
		c[l]=a;
		l++;
	}
	}
	sort(b,b+k);
	sort(c,c+l);
	for(i=0;i<k;i++)
	{
		cout<<b[i]<<" ";
		s+=b[i];
	}
	cout<<s<<" ";
	for(i=0;i<l;i++)
	{
		cout<<c[i]<<" ";
		s1+=c[i];
	}
	cout<<s1<<" ";
	return 0;
	
}
