#include <iostream>
using namespace std;

int main()
{
    long long diff,sum,n,l,r,q,i,a[1000001];
    cin>>n>>q;
    for(i=0;i<n;i++)
     cin>>a[i];
     
    while(q--)
    {
    	cin>>l>>r;
    	diff=a[r-1];
    	sum=a[r-1];
    	r-=2;
    	while(r>=l-1)
    	{
    		sum+=a[r];
    		diff-=a[r];
    		r--;
    	}
    	cout<<sum<<" "<<diff<<endl;
    }
    return 0;
}
