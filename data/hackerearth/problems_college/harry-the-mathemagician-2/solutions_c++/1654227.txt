#include <iostream>
using namespace std;
typedef long long ll;
ll m,n;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    	scanf("%lld%lld",&n,&m);
    	if (m==0 || m==n) {
    		puts("odd");
    		continue;
    	}
    	if (m==1 || m==n-1){
    		if (n&1) puts("odd");
    		else puts("even");
    		continue;
    	}
    	ll m1 = n-m;
    	ll x1 = 0;
    	ll x2 = 0,x3 = 0;
    	ll tmp = 1;
    	for(int i=1;i<=26;i++){
    		tmp *= (1LL*2);
    		x1 += n/tmp;
    		x2 += m/tmp;
    		x3 += m1/tmp;
    	}
    	//cout << x1 << " " << x2 << " " << x3 << endl;
    	if (x1>x2+x3) puts("even");
    	else puts("odd");
    }
    return 0;
}
