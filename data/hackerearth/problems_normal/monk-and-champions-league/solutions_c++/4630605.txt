#include <iostream>
using namespace std;
void swap(long &a,long &b){
	long temp=a;a=b;b=temp;
}
void maxheap(long a[],long i,long n){
	long left=2*i+1,right=2*i+2,s;
	if(left<=n&&a[left]>a[i])s=left;else s=i;
	if(right<=n&&a[right]>a[s])s=right;
	if(s!=i){swap(a[s],a[i]);maxheap(a,s,n);}
}
int main()
{
    long m,n,i,s=0;cin>>m>>n;long a[m];for(i=0;i<m;i++)cin>>a[i];
    for(i=(m-2)/2;i>=0;i--)maxheap(a,i,m-1);
    while(n--){s+=a[0];a[0]--;maxheap(a,0,m-1);}
    cout<<s;
    return 0;
}
