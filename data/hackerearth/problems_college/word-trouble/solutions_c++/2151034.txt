#include<bits/stdc++.h>
using namespace std;

int main() {
   char a[11],b[11];
   int i,j,k,t,count;
   cin >> t;
   while(t--) {
     cin >> a >> b;
     i=strlen(a);
     j=strlen(b);
     if(i==j) {
     	count=0;
     	for(k=0;k<j;k++)
     		if(a[k]!=b[k])
     			count++;
         if(count)
         	cout << "Non Exact " << count << endl;
         else
         	cout << "Exact" << endl;
     }
     else if(i>j) {
     	count =0;
     	for(k=0;k<j;k++)
     		if(a[k]!=b[k])
     			count++;
     	cout << "Non Exact " << count + i-j << endl;
     }
     else {
     	  	count =0;
     	for(k=0;k<i;k++)
     		if(a[k]!=b[k])
     			count++;
     	cout << "Non Exact " << count + j-i << endl;
     }
   }
   return 0;
}
