#include <algorithm>
#include <cmath>
#include <cstdio>
#include <iostream>
using namespace std;
int n, a, b, c, d;
int main() {
  int sum = 0;
  scanf("%d", &n);
  scanf("%d%d%d%d", &a, &b, &c, &d);
  if ((a <= b && c <= d) || (a >= b && c >= d))
    printf("%d\n", abs(a - c) + abs(b - d));
  else
    printf("%d\n", min(4 * n - a - b - c - d, a + b + c + d));
  return 0;
}
