#include<bits/stdc++.h>
using namespace std;
typedef long int ll;
ll  quicksort(ll x[],ll first,ll last)
{
    ll pivot,j,temp,i,f;

     if(first<=last)
     {
         pivot=first;
       /// printf("wq");
         i=first;
         j=last;

        while(i<j)
        {
             while(x[i]<=x[pivot]&&i<last)
                 i++;
             while(x[j]>x[pivot]&&(j>=0))
                 j--;
            if(i<j)
             {
                  temp=x[i];
                  x[i]=x[j];
                  x[j]=temp;
             }
        }
         temp=x[pivot];
         x[pivot]=x[j];
         x[j]=temp;

         quicksort(x,first,j-1);
         quicksort(x,j+1,last);

    }
    return 00;
}

int main()
{
    ll t,x,i,h;
    cin>>t;
    ll n;
    ll ar[100000];
    while(t--)
    {
        ll count =0;

        cin>>n;
        for(i=0;i<n;i++)
        {
            cin>>ar[i];
        }
        quicksort(ar,0,n-1);

     for(i=0;i<n-1;i++)
     {
         if(ar[i+1]!=ar[i])
         {
             count++;

         }
     }
     cout<<count+1<<endl;


    }
    return 0;

}
