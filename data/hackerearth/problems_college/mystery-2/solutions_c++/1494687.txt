#include<stdio.h>
#include<stdlib.h>
int main()
{
	int t,k,a,b,c[1000],ci=0,i;
	scanf("%d",&t);
	for(k=0;k<t;k++,ci=0)
	{
		scanf("%d %d",&a,&b);
		if(a<b)
		{
			printf("%d",a);
		}
		else if(a==b)
		{
			a=a*2;
			printf("%d",a);
		}
		else if(a>b)
		{
			while(a!=0)
			{
				c[ci]=a%b;
				a=a/b;
				++ci;
				
			}
		}
		for(i=ci-1;i>=0;i--)
		{
			printf("%d",c[i]);
		}
		printf("\n");
 
}
}
