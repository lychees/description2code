#include <iostream>
#include <string>
using namespace std;
 
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string a,b;
		cin>>a>>b;
		int date1 = stoi(a.substr(0, 2));
		int date2 = stoi(b.substr(0, 2));
		int month1 = stoi(a.substr(3, 2));
		int month2 = stoi(b.substr(3, 2));
		int year1 = stoi(a.substr(6, 4));
		int year2 = stoi(b.substr(6, 4));
		int lucky=0;
		for(int i=year1+1;i<year2;i++)
		{
			if(i%100>=3 && i%100<=13)
				lucky++;
		}
		if(year1%100>=3 && year1%100<=13)
		{
			if((year1%100)-1>month1 && (year1<year2))
				lucky++;
			else if((year1%100)-1>month1 && year1==year2 && (year1%100)-1<month2)
				lucky++;
			else if((year1%100)-1>month1 && year1==year2 && (year1%100)-1==month2 && month2-1<=date2)
				lucky++;
			else if((year1%100)-1==month1 && date1+1<=month1 && year1<year2)
				lucky++;
			else if((year1%100)-1==month1 && date1+1<=month1 && year1==year2 && (year1%100)-1<month2)
				lucky++;
			else if((year1%100)-1==month1 && date1+1<=month1 && year1==year2 && (year1%100)-1==month2 && month2-1<=date2)
				lucky++;
			//else if(year1==year2 && month1==month2 && )
		}
		if(year2%100>=3 && year2%100<=13 && year1<year2)
		{
			if((year2%100)-1<month2)
				lucky++;
			else if((year2%100)-1==month2 && date2+1>=month2)
				lucky++;
		}
		cout<<lucky<<endl;
	}
    return 0;
}
