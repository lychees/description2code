#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    long long int n,k;
    cin>>n>>k;
    long long int a[n];
    for(long long int i=0;i<n;i++)
    cin>>a[i];
    sort(a,a+n);
    long long int prev=a[0];
    for(long long int i=1;i<n;i++)
    {
    	if((a[i]-prev)>k)
    	break;
    	prev=a[i];
    }
    cout<<prev<<endl;
    return 0;
}
