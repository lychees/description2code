#include <bits/stdc++.h>

#define LL long long
#define pii pair<int, int>
#define sd(x) scanf("%d", &x)

using namespace std;

int main()
{
	int t;
	cin >> t;

	for (int cas = 1; cas <= t; cas++) {
		long long n;
		cin >> n;

		bool z = true;

		while (true) {
			long long p = n;

			long long q = 1;
			while (p) {
				p >>= 1;
				q <<= 1;
			}

			if (2 * n == q) {
				break;
			}

			n = q - n;
			z = !z;
		}

		if (z) {
			puts("0");
		}
		else {
			puts("1");
		}

	}



	return 0;
}
