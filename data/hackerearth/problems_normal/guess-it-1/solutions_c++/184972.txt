#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <list>
#include <vector>
#include <queue>
#include <cstring>
using namespace std;

int t , x , ans=-1 ;
char a[210] ;

bool f()
{
  int rem=0 ;
  for(int i=0 ; a[i] ; i++) rem=(10*rem+a[i]-'0')%9 ; 
 return !rem ;
}

int main()
{
  scanf("%d",&t) ;
  while(t--)
  {
     scanf("%s",&a) ;
     for(int i=0 ; a[i] ; i++) if(a[i]=='_') x=i ;
     
     for(int i=0 ; i<=9 ; i++)
     {
        a[x]=i+'0' ;
        if(f()) 
        { 
           if(ans==0 && i==9) ans=10 ; 
           else ans=i ;
        } 
     }   
     
     printf(ans<=9?"%d\n":"0 or 9\n",ans) ;   
  }

return 0;
}
