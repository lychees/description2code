#include<bits/stdc++.h>
using namespace std;
#define lli long long int
#define f(i,a,b) for(i=a;i<b;i++)
#define si(a) scanf("%d",&a);
#define slli(a) scanf("%lld",&a);
#define INF 1000000007
#define MAX 100005
#define pb(a) push_back(a)
lli arr[MAX];
void fun(lli n,map<lli,lli>&prime)
{
	lli i,j,k,m,cnt=0;
	while(n%2==0)
	{
		n=n/2;
		cnt++;
	}
	if(cnt!=0)
		prime[2]+=cnt;

	for(i=3;i<=sqrt(n);i=i+2)
	{
		cnt=0;
		while(n%i==0)
		{
			n=n/i;
			cnt++;
		}
		if(cnt!=0)
			prime[i]+=cnt;
		if(n==1)
			break;
	}
	if(n!=1)
		prime[n]++;
}
int main()
{
	lli q,min,ans,i,j,k,m,n,t,current;
	map<lli,lli>mp1;
	map<lli,lli>::iterator it1,it2;
	slli(n);
	f(i,0,n)
	{
		slli(arr[i]);
		fun(arr[i],mp1);
	}
	/*for(it1=mp1.begin();it1!=mp1.end();it1++)

	slli(q);
	while(q--)
	{
		map<lli,lli>mp2;
                min=INF;
		slli(m);
		fun(m,mp2);
		if(mp2.size() > mp1.size())
			ans=0;

		else
		{
			//it2=mp2.begin();
			for(it2=mp2.begin();it2!=mp2.end();it2++)
			{
				current=mp1[it2->first]/it2->second;
				if(current < min && current > 0)
					min=current;
				if(current==0)
				{
					min=INF;
					break;
				}
			}
			if(min==INF)
				ans=0;
			else
				ans=min;
		}

	}
}
