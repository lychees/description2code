#include <bits/stdc++.h>
#include <stdlib.h>
using namespace std;
const long long mod=1e9+7;
 
#define llu long long unsigned
#define lld long long
#define ld long
#define sc(a) scanf("%lld",&a)
#define sc2(a,b) scanf("%lld%lld",&a,&b)
#define sc3(a,b,c) scanf("%lld%lld%lld",&a,&b,&c)

 
int main(){
    lld T,r,s,c,w,t;
    sc(T);
    for(s=1;s<=T;s++){
    	sc3(r,c,w);
    	t=(c/w)*(r-1)+(c-1)/w+w;
    	printf("Case #%lld: %lld\n",s,t);
    }
    return 0;
}
