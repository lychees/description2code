#include <bits/stdc++.h>
using namespace std;
unsigned long long int div(unsigned long long int n) {
		int f = 0, c = 0, k;
		unsigned long long int sum;
		sum = 1;
		while(n % 2 == 0) {
				f = 1;
				c++;
				n /= 2;
		}
		if(f == 1) {
			sum *= (c + 1);
			c++;
			f = 0;
			c = 0;
		}
		for(k = 3; k < sqrt(n); k += 2 ) {
				while(n % k == 0) {
					f = 1;
					n /= k;
					c++;
				}
				if(f == 1) {
					sum *= c + 1;
				}
				c = 0;
				
		}
		if(n > 2) {
			 sum *= 2;
		}
		return sum;
		
}
int main()
{
	 unsigned long long int n,x;
	 cin >> n;
	 while(n--) {
	 		cin >> x;
	 		if(div(x) < 4) {
	 			cout << "NO";
	 		} else {
	 				cout << "YES";
	 		}
	 		cout << "\n";
	 		
	 }
}
