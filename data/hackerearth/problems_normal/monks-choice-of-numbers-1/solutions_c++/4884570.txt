#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    int n,k,i,count,c,f,s=0,j;
    
    cin>>n>>k;
    int v[n];
    int a[n];
    for(i=0;i<n;i++)
    {
    	cin>>v[i];
    }
    	
	for(i=0;i<n;i++)
	{
		count=0;
		while(v[i])
		{
			v[i]=v[i]&(v[i]-1);
			count++;
		}
		a[i]=count;
		
	}
	for(i=0;i<k;i++)
	{
		c=0;
		for(j=0;j<n;j++)
		{
			if(a[j]>=c)
			{
				c=a[j];
				f=j;
			}
		}
		a[f]=0;
		s=s+c;
	}
	cout<<s<<endl;
    }
    return 0;
}
