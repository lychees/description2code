#include<bits/stdc++.h>
#define pb push_back
#define mp make_pair
#define f first
#define maxn 201000
#define s second
#define ll long long int
#define inf 1000000014
#define infl (ll)(1e18)
#define mod 1000000007
using namespace std;
ll pow2[maxn];
ll mulval[maxn];
int main(int argc, char const *argv[])
{
   pow2[0]=1ll;
   for(int i=1;i<=200010;i++)
   {
    pow2[i] = (pow2[i-1]*2)%mod;
   }
   mulval[0]=1ll;
   for (int i = 1; i <=200010; ++i)
   {
       mulval[i] = (pow2[i-1]*(i+2))%mod;
   }
   int n;
   ll val;
   cin>>n;
   ll ans=0;

   for (int i = 0; i < n; ++i)
   {
       scanf("%lld",&val);
       ans = ans+((val*mulval[i])%mod*pow2[n-1-i])%mod;
       ans = ans%mod;
   }
  cout<<ans%mod;
  return 0;
}
