#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int test;
    cin >> test;
    for (int i = 0; i < test; ++i)
    {
    	int k;
    	cin >> k;
    	long long int len = 1;
    	while (len < k)
    	{
    		len <<= 1;
    	}
    	--len;
    	bool done = false;
    	int bit = 0;
    	int toggle = 0;
    	while (len != 1 && done == false)
    	{
    		if ((k & (k - 1)) == 0)
    		{
    			done = true;
    		}
    		else if (k > len / 2)
    		{
    			k = len - k + 1;
    			++toggle;
    			len /= 2;
    		}
    		else
    		{
    			len /= 2;
    		}
    	}
    	bit = (toggle % 2) == 0 ? 0 : 1;
    	cout << bit << endl;
    }
    return 0;
}
