string=raw_input()
#list_of_char=string.split()
set_of_char=set(string)
#print set_of_char
char=''
max_count=0
for i in set_of_char:
	count=string.count(i)
	if count>max_count:
		max_count=count
		char=i
	elif count==max_count:
		max_count=count
		char=min(i,char)
	else:
		pass
print char,max_count
