#include<bits/stdc++.h>
using namespace std;
int main()
{
int t;
scanf("%d",&t);
while(t--)
{
char s[10005],p[10005];
scanf("%s",s);
//printf("%s\n",s);
int uu=0,ss=0,i;
for(i=0;s[i]!='\0';i++)
{
    if(s[i]=='U')
        uu++;
    else
        ss++;
}
//printf("%d %d\n",uu,ss);
if((uu-ss)>1||ss-uu>1)
{
printf("-1\n");
}
else
{
    int c=0;
    if(uu>=ss)
    {
for(i=0;c<uu;i+=2)
{
    p[i]='U';
    c++;
}
c=0;
for(i=1;c<ss;i+=2)
{
    p[i]='S';
    c++;
}
p[uu+ss]='\0';
    }
    else
    {
        for(i=0;c<ss;i+=2)
{
    p[i]='S';
    c++;
}
c=0;
for(i=1;c<uu;i+=2)
{
    p[i]='U';
    c++;
}
p[uu+ss]='\0';
    }

c=0;
for(i=0;i<uu+ss;i++)
{

    if(s[i]!=p[i])
        c++;
}
if(uu==ss)
    c=min(c,uu+ss-c);
c=c/2;
printf("%d\n",c);
}
}
return 0;
}
