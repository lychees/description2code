#include <iostream>
#include<cstdlib>
using namespace std;

int main()
{
	long int N;
    cin>>N;
    long int arr[N];
    for(int i=0;i<N;i++)
    {
   	cin>>arr[i];
    }
    for(int i=0;i<N;i++)
    {
    int flag=0;
    if(arr[i]!=-1)
    {
    for(int j=i+1;j<N;j++)
    	{
    	if(arr[i]==arr[j])
    		{
    		arr[j]=-1;
    		flag=1;
    		}
    	}
    if(flag==1)
    	arr[i]=-1;
    }
    }
    int temp=-1;
    for(int i=0;i<N;i++)
    {
   	if(arr[i]>temp)
   		temp=arr[i];
    }
    cout<<temp;
    return 0;
}
