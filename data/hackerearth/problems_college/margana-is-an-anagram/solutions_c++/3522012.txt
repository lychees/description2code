#include<bits/stdc++.h>
using namespace std;
  int main() {
  char s1[1000005], s2[1000005];
  int i, t;
  scanf("%d", &t);
  getchar();
  while(t--) {
  scanf("%s", s1);
  scanf("%s", s2);
  int h[30] = {0};
  for(i = 0 ; s1[i] != '\0' ; i++) 
      ++h[s1[i]-97];
  for(i = 0 ; s2[i] != '\0' ; i++) 
      --h[s2[i]-97];

  for(i = 0 ; i < 26 ; i++)
      if(h[i])
          break;
  if(i == 26)
      printf("Yes\n");
  else
      printf("No\n");
  }
  return 0;
  }
