int n, m;
int a[MAX_N], b[MAX_N];
int id[MAX_N * 10];

bool check(int x, int id) {
	assert(x % a[id] == 0);
	for (int i = id + 1; i <= m; i++)
		if (x % a[i] == 0)
			return 0;
	return 1;
}

		scanf("%d", &a[i]);
		for (int j = a[i]; j <= (int)1e6; j += a[i])
			id[j] = i;
	}
	int ans = 0;
	for (int i = 1; i <= n; i++) {
		scanf("%d", &b[i]);
		ans = max(ans + 1, a[b[i]]);
		if (ans % a[b[i]] != 0)
			ans = (ans / a[b[i]] + 1) * a[b[i]];
		while(!check(ans, b[i]))
			ans += a[b[i]];
	}
	printf("%d", ans);
	return 0;
}
