#include <iostream>
#include<string>
#include<map>
using namespace std;

int main()
{
	ios::sync_with_stdio(false);cin.tie(0);
   int tc;
   cin>>tc;
   cin.ignore();
   
   while(tc--)
 	{
 		int N,K,M,i=0;
 		cin>>N>>K>>M;
 		string s;
 		multimap<string,string> mymap;
 		
 		
 		while(N--)
 		{
 			cin>>s;
 			mymap.insert( std::pair<string,string>(s.substr(0,M),s ));
 			
 			
 		} 
 		std::multimap<string,string>::iterator it;
 		for (it=mymap.begin(); i<K-1; ++it,i++);
    std::cout << (*it).second << '\n';
 		
 		
 		
 	}
 	return 0;
}
