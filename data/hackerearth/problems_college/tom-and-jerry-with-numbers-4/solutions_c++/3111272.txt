#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
	bool  arr[1000009]={0};
		int n,a,b;
		scanf("%d %d %d",&n,&a,&b);
		arr[0]=0;
		for(int i=0;i<=n;i++)
		{
			if(i<a)
			arr[i]=0;
			else
			{
				if(i<b)
				arr[i]=(!arr[i-a]);
				else
				arr[i]=(!arr[i-a])||(!arr[i-b]);
			}
		}
	
		printf("%d\n",arr[n]);
	}
	return 0;
}
