#include<iostream>
#include<algorithm>
#include<string>
using namespace std;
int main()
{
	int t;
	cin>>t;
	getchar();
	while(t--)
	{   string s;
        getline(cin,s);
       reverse(s.begin(),s.end());
       string::iterator it,it1;
       for(it=s.begin();it!=s.end();it++){
       	it1=it;
       	while(it!=s.end()&&*it!=' ')it++;
       	reverse(it1,it);
       	if(*it=='\n'||it==s.end())break;  
       } 
       cout<<s<<endl;
	}
	return 0;
}
