#include <bits/stdc++.h>
#define ll long long
using namespace std;


int main()
{
    ll t;
    scanf("%lld", &t);
    while(t--) {
        ll n;
   		ll k;
 		scanf("%lld %lld", &n, &k);
	    ll a[n + 2];
        if(k > n / 2) {
            k = n - k;
        }
        a[k] = 8;
        for(int i = k + 1; i <= n - k; i++) {
           a[i] = 9;
        }
        for(int i = 1; i < k; i++) {
            a[i] = 9;
        }
        a[n] = 1;
        for(int i = n - k + 1; i < n; i++) {
           a[i] = 0;
        }
        for(int i = 1; i <= n; i++) {
        printf("%lld",a[i]);
        }
       printf("\n");
    }
    return 0;
}
