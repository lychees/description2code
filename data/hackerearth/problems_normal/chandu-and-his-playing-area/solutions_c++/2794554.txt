#include<iostream>
using namespace std;
int main()
{
	int test_cases;
	cin>>test_cases;
	long int straws,semi_peri,length,breadth,area;
	for(int i=0;i<test_cases;i++)
	{
		cin>>straws;
		if(straws%2==0)
		{
			semi_peri=straws/2;
			length=int(semi_peri/2);
			breadth=semi_peri-length;
			area=length*breadth;
		}
		else
		{
			straws=straws-1;
			semi_peri=straws/2;
			length=int(semi_peri/2);
			breadth=semi_peri-length;
			area=length*breadth;
		}
		cout<<area<<"\n";
	}
	
}
