#include <iostream>
#include <map>
using namespace std;

int main() {
	// your code goes here
	int t;
	cin>>t;
	map <string,int> m;
	while(t--)
	{
		string s;
		cin>>s;
		m[s]=1;
	}
	int n;
	cin>>n;
	string res="0";
	while(n--)
	{
		string s;
		cin>>s;
		if(!m[s])
		res=res+"."+char('A'+int(s[0]-'a'));
	}
	res=res.substr(2,res.size()-2);
	cout<<res<<endl;
	return 0;
}
