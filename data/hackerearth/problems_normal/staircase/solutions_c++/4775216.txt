#include <iostream>

using namespace std;

int main() {
    int n;
    cin>>n;
    if (n <= 0) {
        cout<<"ERROR";
        return 0;
    }

    long jack[n+1], jill[n+1];

    if (n == 1) {
        cout<<"Jack-1"<<"\n"<<"Jill-1";
        return 0;
    } else if (n == 2) {
        cout<<"Jack-2"<<"\n"<<"Jill-2";
        return 0;
    } else if (n == 3) {
        cout<<"Jack-4"<<"\n"<<"Jill-3";
        return 0;
    }

    jack[1] = 1;
    jack[2] = 2;
    jack[3] = 4;

    jill[1] = 1;
    jill[2] = 2;
    jill[3] = 3;


    for (int i = 4; i <= n; ++i) {
        jack[i] = jack[i-1] + jack[i-2] + jack[i-3];
        jill[i] = jill[i-1] + jill[i-2];
    }

    cout<<"Jack-"<<jack[n]<<"\n";
    cout<<"Jill-"<<jill[n];
    return 0;

}
