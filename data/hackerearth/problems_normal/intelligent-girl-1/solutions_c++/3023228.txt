#include <iostream>
#include<string>
using namespace std;

int main()
{
    string in;
    cin>>in;
    int arr[in.size()],i,len=in.size();
    arr[len-1]=0;
    if(in[len-1]%2==0) arr[len-1]=1;
    for(i=len-2;i>=0;i--) 
    {
    	if(in[i]%2==0) arr[i]=arr[i+1]+1;
    	else arr[i]=arr[i+1];
    }
    for(i=0;i<len-1;i++) cout<<arr[i]<<" ";
    cout<<arr[len-1]<<endl;
    return 0;
}
