#include <iostream>
using namespace std;

int main()
{
    int N,i,Q;
    cin>>N;
    long long A[N];
    
    for(i=0;i<N;i++)
    {
    	cin>>A[i];
    	if(i>0)
    		A[i]+=A[i-1];
    }
    cin>>Q;
    for(i=0;i<Q;i++)
    {
    	int s,e;
    	cin>>s>>e;
    	if(s!=1)	
	    	cout<<(A[e-1]-A[s-2])<<"\n";
	    else
	    	cout<<A[e-1]<<"\n";
    }
}
