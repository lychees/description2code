#include <iostream>
using namespace std;

int main()
{
	int t;
    cin >> t;
    for(int it=0; it<t; it++) {
    	long long int n;
    	cin >> n;
    	long long int a[n+2], b[n+2];
    	for(int i=1; i<=n; i++) {
    		cin >> a[i];
    		b[a[i]] = i;
    	}
    	int flag = 0;
    	for(int i=1; i<=n; i++) {
    		if(a[i] != b[i]) {
    			flag = 1;
    			break;
    		}
    	}
    	if(flag)
    		cout << "not inverse\n";
    	else
    		cout << "inverse\n";
    }
    return 0;
}
