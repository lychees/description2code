#include <bits/stdc++.h>
using namespace std;

int ans(int n,int m)
{
	int res[n];
	res[0]=1;
	res[1]=1;
	for(int i=2;i<n;i++)
	{
		res[i]=0;
		for(int j=1;j<=i && j<=m;j++)
		{
			res[i]+=res[i-j];
		}
	}
	return res[n-1];
}

int countways(int s,int m)
{
	return ans(s+1,m);
}


int main()
{
    int n;
    cin>>n;
    if(n<0) cout<<"ERROR\n";
    else
    {
    	int ans1,ans2;
    	ans1=countways(n,3);
    	ans2=countways(n,2);
    	cout<<"Jack-"<<ans1<<"\n";
    	cout<<"Jill-"<<ans2<<"\n";
    }
    return 0;
}
