#include <bits/stdc++.h>
using namespace std;

int res(string &s1, string &s2) {
	int count = 0;
	int len = s1.length() < s2.length() ? s1.length() : s2.length();
	//cout << len;
	for(int i = 0; i < len; ++i)
		if (s1[i] == s2[i])
			++count;
	return count;
}

int main(void) {
	int T = 0;
	scanf("%d",&T);
	while(T--) {
    	string s1,s2;
    	cin >> s1;
    	cin >> s2;
    	printf("%d\n",res(s1,s2));
    	s1.clear();
    	s2.clear();
	}
    return 0;
}
