#include <cstdio>
#include <cmath>
#include <iostream>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    	int p,x,y,u,v;
    	scanf("%d%d%d%d%d",&p,&x,&y,&u,&v);
    	int d=max(abs(u-x),abs(v-y))-min(abs(u-x),abs(v-y));
    	if((x-y)==(u-v)){
    		if(x==u&&y==v)printf("Counter-Terrorists Win !\n");
    		else if(p>1)printf("Counter-Terrorists Win !\n");
    		else printf("Terrorists Win !\n");
    	}
    	else if(x+y==u+v){
    		if(x==u&&y==v)printf("Counter-Terrorists Win !\n");
    		else if(p>1)printf("Counter-Terrorists Win !\n");
    		else printf("Terrorists Win !\n");
    	}
    	else{
    		if(p>2&&d%2==0)printf("Counter-Terrorists Win !\n");
    		else printf("Terrorists Win !\n");
    	}
    }
}
