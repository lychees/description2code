#include<bits/stdc++.h>
using namespace std;

#define mod 1000000007
typedef set<string> ss;
typedef vector<int> vs;
typedef map<int,char> msi;
typedef pair<int,int> pa;
typedef long long int ll;

int dp[705][155][55],val,c;
int solve(int n1,int n2,int n3)
{
	int k=(val-n1-n2*5-n3*10)/8;
	if(k==c)
	return 0;
	int &ret=dp[n1][n2][n3];
	if(ret!=-1)
	return ret;
	ret=INT_MAX;
	if(n3>=1)
	ret=min(ret,solve(n1+2,n2,n3-1)+1);
	if(n2>=2)
	ret=min(ret,solve(n1+2,n2-2,n3)+2);
	if(n2>=1 && n1>=3)
	ret=min(ret,solve(n1-3,n2-1,n3)+4);
	if(n1>=8)
	ret=min(ret,solve(n1-8,n2,n3)+8);
	if(n1>=3 && n3>=1)
	ret=min(ret,solve(n1-3,n2+1,n3-1)+4);
	return ret;
}

int n1,n2,n3;
int main()
{
	//freopen("i.in", "r", stdin);
  	//freopen("o.out", "w", stdout);
  	int t;
	cin>>t;
	while(t--)
	{
		memset(dp,-1,sizeof(dp));
		cin>>c>>n1>>n2>>n3;
		val=n1+n2*5+n3*10;
		if(val<c*8)
		cout<<-1<<"\n";
		else
		cout<<solve(n1,n2,n3)<<"\n";
	}
	return 0;
}
