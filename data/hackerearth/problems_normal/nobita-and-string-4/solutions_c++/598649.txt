#include <bits/stdc++.h>
using namespace std;
char s[100010];
int main()
{
	int t;
	cin>>t;
	getchar();
	while(t--)
	{
		gets(s);
		int n=strlen(s);
		reverse(s,s+n);
		int j=0;
		for(int i=0;i<n;i++)
		{
			if(s[i]==' ')
			{
					reverse(s+j,s+i);
					j=i+1;
			}
		}
		reverse(s+j,s+n);
		s[n]='\0';
		cout<<s<<endl;
	}
return 0;
}
