#include <iostream>
#include<stdlib.h>
#include<algorithm>
#include<bits/stdc++.h>
#include<string.h>
using namespace std;

int main()
{
	int t,i,j,k,flag=0,len;
	char str[103],temp;
	int small;
	cin>>t;
	while(t--)
	{
		flag=0;
		cin>>str;
		len=strlen(str);
		for(i=len-1;i>=0;i--)
		{
			if(str[i]>str[i-1])
				break;
		}
		if(i==0)
		{
			cout<<"no answer\n";
			continue;
		}
		
		small=i;
		int x=str[i-1];
		for(j=i+1;j<len;j++)
			if(str[j]>x && str[j]<str[small])
				small=j;
		temp=str[small];
		str[small]=str[i-1];
		str[i-1]=temp;
			sort(str+i,str+len);
			cout<<str<<endl;
	}
	
    //cout << "Hello World!" << endl;
    return 0;
}
