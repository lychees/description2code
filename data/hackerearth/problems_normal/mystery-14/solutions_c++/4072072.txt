#include <bits/stdc++.h>

using namespace std;

typedef long long ll;

int main()
{
	int t;
	scanf("%d", &t);
	
	while (t--) {
		int n, x;
		
		scanf("%d", &n);
		
		ll sum = 0;
		for (int i=0; i<n; i++) {
			scanf("%d", &x);
			if (x&1) {
				sum += x;
			}
		}
		
		printf("%lld\n", sum);
	}
	
    return 0;
}
