bool isPal(int b,int e,string s)
{
	bool x=true;
	while(b<=e)
	{
		if(s.at(b)==s.at(e))
		{
			b++;
			e--;
		}
		else
		{
			x=false;
			break;
		}
	}
	return x;
}
