#include <iostream>
#include <map>
#include <cmath>
using namespace std;
map<int,int> mp1;
int main()
{
    int t,n,p1,p2,r;
    mp1[1]=1;
    for(p1=2;p1<=1000;p1++)
    	for(p2=2;pow(p1,p2)<=1000000;p2++)
    		mp1[pow(p1,p2)]++;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	r=0;
    	for(p1=1;p1<=n/2;p1++)
    		if(mp1[p1]&&mp1[n-p1])
    		{
    			r=1;
    			break;
    		}
    	if(r)
    		cout<<"Yes\n";
    	else
    		cout<<"No\n";
    }
    return 0;
}
