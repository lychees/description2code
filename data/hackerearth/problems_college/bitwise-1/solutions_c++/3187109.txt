#include <bits/stdc++.h>
using namespace std;

int main()
{
    int T;
    long long i,n,k;
    cin >> T;
    for(int loop=1;loop<=T;loop++)
    {
    	cin >> n >> k;
    	for(i=1;k;i++)
    	{
    		//cout << n << " " << i << " " << n+i << " " << (n^i) << endl;
    		if(n+i==(n^i))
    			k--;
    	}
    	cout <<"Case #"<<loop << ": " <<  i-1 << endl;
    }
    return 0;
}
