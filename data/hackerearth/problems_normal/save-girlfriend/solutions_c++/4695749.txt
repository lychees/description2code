#include<bits/stdc++.h>
#define ll long long int
using namespace std;
bool arr[100005];
map<ll,ll> mp;
vector<ll> v;
void seive()
{
	ll i,j;
	arr[0]=arr[1]=1;
	for(i=2;i*i<=100005;i++)
	{
		if(!arr[i])
		{
			j=2;
			while(j*i<=100005)
			{
				arr[j*i]=1;
				j++;
			}
			v.push_back(i);
		}
	}
	for(j=i;j<=100005;j++)
	{
		if(!arr[j])
		{
			v.push_back(j);
		}
	}
}
int main()
{
	seive();
	ll t,a,b,i,j,val=1,ans;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld%lld",&a,&b);
		ll mx=0,cnt=0;
		for(i=a;i<=b;i++)
		{
			ll temp=i;
			ans=1;
			for(j=0;v[j]*v[j]<=temp;j++)
			{
				cnt=0;
				while(temp%v[j]==0)
				{
					temp=temp/v[j];
					cnt++;
				}
				if(cnt)
				{
					ans*=(cnt+1);
				}
			}
			if(temp>1)
			{
				ans*=2;
			}
			mx=max(mx,ans);
			mp[ans]++;
		}
		printf("%lld\n",mp[mx]+1);
		mp.clear();
	}
	return 0;
}
