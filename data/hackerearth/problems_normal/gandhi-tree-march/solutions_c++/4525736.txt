#include<bits/stdc++.h>
using namespace std;
int pos;
string str;
vector<char> L[10001],R[10001];
void buildtree(int c)
{
	if(str[pos]!='.')
	{
		if(abs(c)<=10000)
		{
			if(c>=0) L[c].push_back(str[pos]);
			else R[-c].push_back(str[pos]);
		}
		pos+=2;
		buildtree(c-1);
		buildtree(c+1);
	}
	pos++;
}
int main()
{
	int i,t,c,count; 
	cin>>t;
	while(t--)
	{
		pos=0;
		cin>>c>>str;
		for(int i=0;i<=10000;i++) L[i].clear(),R[i].clear();
		
		
		buildtree(0);
		if(c>=0)
		{
			if(count=L[c].size())
			{
				sort(L[c].begin(),L[c].end());
				for(int i=0;i<count;i++) cout<<L[c][i];
				cout<<endl;
			}
			else
			cout<<"Common Gandhijee!\n";
		}
		else
		{
			if(count=R[-c].size())
			{
				sort(R[-c].begin(),R[-c].end());
				for(int i=0;i<count;i++) cout<<R[-c][i];
				cout<<endl;
			}
			else cout<<"Common Gandhijee!\n";
		}
	}
}
