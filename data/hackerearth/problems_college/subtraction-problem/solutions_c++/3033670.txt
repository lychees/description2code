#include <iostream>

using namespace std;
int r;
int main()
{
    long long a,b;
    cin>>a>>b;
    long long x=a-b;
    int y[8];
    while(x!=0)
    {
        y[r++]=x%10;
        x=x/10;

    }
    for(int i=0;i<r;i++)
    {
        if(y[i]<9)
          { y[i]++;
          break;

          }

    }
    for(int i=r-1;i>=0;i--)
            cout<<y[i];
            cout<<endl;
    return 0;
}
