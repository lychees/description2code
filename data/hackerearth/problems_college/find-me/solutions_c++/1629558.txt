#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
	int t;
	cin >> t;
	
	while (t--) {
		int n, p;
		cin >> n >> p;
		int cnt = 0;
		bool there = false;
		
		while (n--) {
			int a;
			scanf("%d", &a);
			if (a==p)there = true;
			else if (a>p) ++cnt;
		}
		
		if (there) {
			printf ("YES %d\n", cnt);
		} else printf("NO 0\n");
	}
    return 0;
}
