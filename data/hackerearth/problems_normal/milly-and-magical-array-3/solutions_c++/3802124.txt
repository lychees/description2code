#include <iostream>
#include <climits>
using namespace std;

int main()
{
    int T;
    cin >> T;
    for(int i=0; i<T; i++){
    	int N, X;
    	cin >> N >> X;
    	int64_t p,n,ret;
    	ret = 0;
    	cin >> p;
    	for(int j=1; j<N; j++){
    		cin >> n;
    		//cout << "p="<<p<<"; n="<<n<<"; X=" << X << endl;
    		/*while(n<=p){
    			n+=X;
    			ret++;
    		}*/
    		if(n<=p){
    			int64_t div=((p-n)/X) + 1;
    			//cout << "div=" << div << endl;
    			n+=(div*X);
    			ret+=div;
    		}
    		p = n;
    	}
    	cout << ret << endl;
    }
    return 0;
}
