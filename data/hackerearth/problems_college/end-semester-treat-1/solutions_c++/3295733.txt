#include <iostream>
#include <vector>
#include <cmath>

using namespace std;

int main(){
	int T,N,M;
	int check;
	cin>>T;
	while(T--){
		cin>>N;

		vector<long long int> res(N);
		vector<bool> run(N);
		for(int i=0;i<N;i++){
			cin>>res[i];
			run[i]=true;
		}
		cin>>M;
		vector<vector<int> > have(M,vector<int>(N));
		vector<vector<int> > req(M,vector<int>(N));
		vector<vector<int> > need(M,vector<int>(N));

		for(int i=0;i<M;i++)
			for(int j=0;j<N;j++)
				cin>>have[i][j];
		for(int i=0;i<M;i++)
			for(int j=0;j<N;j++)
				cin>>req[i][j];

		for(int i=0;i<M;i++)
			for(int j=0;j<N;j++)
				need[i][j]=req[i][j]-have[i][j];
		

		for(int i=0;i<M;i++){
			check=-1;
			for(int j=0;j<M;j++){
				for(int k=0;k<N;k++){
					if(res[k]<need[j][k]){
						check=0;break;
					}
				}
				if(check==-1){
					for(int k=0;k<N;k++){
						res[k]+=have[j][k];
						run[j]=false;
					}
				}
			}
		}



		check=-1;
		for(int i=0;i<M;i++)
			if(run[i])check=0;
		if(check==-1)cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;

	}

	

	return 0;
}
