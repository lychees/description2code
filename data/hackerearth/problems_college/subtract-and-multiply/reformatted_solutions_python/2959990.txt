from math import sqrt
n = int(raw_input())
for j in range(n):
    d, p = map(float, raw_input().split())
    x = sqrt(d**2+4*p)
    z1 = (x+d)/2
    z2 = (x-d)/2
    if z1 == int(z1) and z2 == int(z2):
        b = p/z1
        if b != z1:
            print 4
        else:
            print 2
    else:
        print 0
