#include <iostream>
#include <algorithm>
#include <cstring>
#include <cstdio>
#include <stdlib.h>
#include <math.h>
#include <vector>

typedef long long int ll;
typedef long int l;
using namespace std;

l reverse_num(l a)
{
    l temp=0;
    while(a)
    {
        temp=temp*10+a%10;
        a=a/10;
    }
    return temp;
}

int main()
{
  int T;
  cin>>T;
  while(T--)
  {
      l a,b,res=0;
      cin>>a>>b;
       res=reverse_num(reverse_num(a)+reverse_num(b));
       cout<<res<<endl;
      }
    return 0;
}
