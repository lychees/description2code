#define INF 2147000000

using namespace std;

int T;
string str;

int main()
{
    ios_base::sync_with_stdio(0);

    cin >> T;
    for (int i = 1; i <= T; i++) {
        cin >> str;

        int deletions = 0;
        for (int j = 0 ; j < str.size() - 1; j++) {
            if (str[j + 1] == str[j]) {
                deletions++;
            }
        }
        cout << deletions << endl;
    }
