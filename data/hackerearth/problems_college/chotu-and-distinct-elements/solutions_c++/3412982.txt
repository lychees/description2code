#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t, n, i, x, flag;
    set<int>::iterator it1, it2;
    
    cin>>t;
    while(t--){
    	set<int> s;
    	cin>>n;
    	for(int i=0;i<n;i++){
			cin>>x;
    		s.insert(x);
    	}
    	flag = 0;
    	for(it1 = s.begin(), it2 = --s.end(); *it1<=*it2 && flag!=1; it1++, it2--)
    	{
    		if(*it1==*it2){
    			cout<<*it1;
    			flag = 1;
    		}
    		else
    		{
    			cout<<*it2<<" "<<*it1<<" ";
    		}
    	}
    	cout<<endl;
    }
    return 0;
}
