#include<bits/stdc++.h>
using namespace std;
bool fun(long int i,long int j){return i>j;}
int main()
{
    int T;
    long int i,j,k,A[3],max1,max2;
    cin>>T;
    while(T--)
    {
        cin>>A[0]>>A[1]>>A[2];
        sort(A,A+3,fun);
        max1=A[0];
        max2=A[1];
        long int c=0;
        if(max1==0 && max2==0)
        {
            cout<<"0\n";
        }
        else if(max1==0 || max2==0)
        {
            cout<<"1\n";
        }
        else
        {
            while(max2!=0)
            {
                c++;
                max1--;
                c++;
                max2--;
            }
            if(max1!=0)
                c++;
            cout<<c<<"\n";

        }


    }
}
