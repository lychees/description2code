from sys import stdin
from itertools import combinations


def main():
    for _ in xrange(int(raw_input())):
        n, k = map(int, raw_input().split())
        arrs = [set(raw_input().split()[1:]) for _ in xrange(n)]
        for x in xrange(1, n + 1):
            for comb in combinations(arrs, x):
                if len(reduce(set.union, comb)) >= k:
                    print x
                    break
            else:
                continue
            break
        else:
            print -1


main()
