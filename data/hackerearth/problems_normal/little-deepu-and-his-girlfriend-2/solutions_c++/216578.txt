using namespace std;

int a[10000];
int dp[10001], n;
int f ( int x )
{
	if ( x < 0 )
	return 1;
	if ( x == 0 )
	return 0;
	int &ans = dp[x];
	if ( ans != -1 )
	return ans;
	for ( int i = 0; i < n; i++ )
	{
		if ( f(x-a[i]) == 0)
		return dp[x] = 1;
	}
	return dp[x] = 0;
}
int main()
{
	int t;
	scanf("%d", &t );
	while ( t-- )
	{
		int  m;
		scanf("%d%d", &m, &n);
		for ( int i = 0; i < n; i++ )
		scanf("%d", &a[i]);
		sort(a,a+n);
		for ( int j = 0; j <= m; j++ )
		{
			dp[j] = 0;
			for ( int i = 0; i < n && j >= a[i]; i++ )
			{
				if ( j >= a[i]&& dp[j-a[i]] == 0 )
				{
				dp[j]=1;
				break;
				}
			}
		}
		dp[m]?puts("Little Deepu"):puts("Kate");
	}
}
