#include <iostream>
using namespace std;

int digsum(long long int n)
{
	int a=0;
	while(n)
	{
		a+=n%10;
		n/=10;
	}
	return a;
}
int main()
{
    int n,i;
    cin>>n;
    long long int sum=0;
    char str[2100];
    while(n--)
    {
    	cin>>str;
    	for(i=0;str[i];i++)
    	{
    		sum+=str[i]-'0';
    	}
    }
    long long int aa=sum;
    while(1)
    {
    	aa=digsum(aa);
    	if(aa<10) break;
	}
    cout<<aa;
    return 0;
}
