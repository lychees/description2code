#include<bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
	
     std::ios::sync_with_stdio(false);
     ll t;
	 cin>>t;
	 while(t--)
	 {
	     ll n,c,i;
	     c=0;
		 cin>>n;
		 ll s= (ll) sqrt(n) + 1;
		 for(i=1;i<s;i++)
		 {
		     if(n%i==0)
			 {
			     if(i%2==0)
				 c++;
				 ll d=n/i;
				 if(d%2==0&&d!=i)
				 c++;
			 }
		 }
		 cout<<c<<endl;
		 
	 }
	 return 0;
}
