#include<iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
   int girls[10005];
   int boys[10005];
   int tc;
   cin>>tc;
   while(tc--)
   {
       int n;
       cin>>n;
       int i,ans=0;
       for(i=0;i<n;i++) cin>>girls[i];
       for(i=0;i<n;i++) cin>>boys[i];
       sort(girls,girls+n);
       sort(boys,boys+n,greater<int>());
       for(i=0;i<n;i++)
         if(girls[i]%boys[i]==0 || boys[i]%girls[i]==0)
            ans++;
       cout<<ans<<endl;
   }
    return 0;
}
