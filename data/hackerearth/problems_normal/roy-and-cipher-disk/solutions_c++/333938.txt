#include <iostream>
#include <string>
#include <cmath>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string a;
    	cin>>a;
    	int l=a.length();
    	int curr=0;
    	int clock=0;
    	int anti=0;
    	int arr[26];
    	for(int i=0;i<26;i++)
    	arr[i]=i;
    	int num;
    	for(int i=0;i<l;i++)
    	{
    		num=a[i]-97;
    		//cout<<num<<" "<<curr<<" ";
    		if(num>=curr)
    		{
    			clock=arr[num]-arr[curr];
    			anti=clock-26;
    			if(abs(anti)<clock)
    			cout<<anti<<" ";
    			else
    			cout<<clock<<" ";
    			curr=num;
    			
    		}
    		else
    		{
    			anti=arr[num]-arr[curr];
    			clock=abs(anti+26);
    			if(abs(anti)<clock)
    			cout<<anti<<" ";
    			else
    			cout<<clock<<" ";
    			curr=num;
    			
    		}
    		
    	}
    	cout<<endl;
    }
    return 0;
}
