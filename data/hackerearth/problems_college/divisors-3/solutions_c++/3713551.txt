#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main()
{
	//ios_base::sync_with_stdio(false);
	//cin.tie(0);
	//cout.tie(0);
	ll n,t;
	cin >> n >> t;
	vector<ll>v;
	for(ll i=1;i*i<=n;i++)
	{
		if(n % i == 0)
		{
			v.push_back(i);
			if(i != n/i)
				v.push_back(n/i);
		}
	}
	//for(int i=0;i<v.size();i++) cout << v[i] << "\n";
	//assert(v.size() < 50);
	//cout << v.size() << endl;
	while(t--)
	{
		ll type, k, ans = 0;
		cin >> type >> k;
		if(type == 1)
		{
			for(int i=0;i<v.size();i++) if(k % v[i] == 0) ans++;
		}
		else if(type == 2)
		{
			for(int i=0;i<v.size();i++) if(v[i] % k == 0) ans++;
		}
		else
		{
			for(int i=0;i<v.size();i++) if(v[i] % k != 0) ans++;
		}
		cout << ans << "\n";
	}
    return 0;
}
