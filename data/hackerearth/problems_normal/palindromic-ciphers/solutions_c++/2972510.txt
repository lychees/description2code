#include <iostream>
#include<string>

using namespace std;

int main()
{
    
    int t;
    cin>>t;
    string s;
    for(int i=0;i<t;i++)
    {
    	bool palin=true;
    	cin>>s;
    	unsigned long long int ans=1;
    	for(int i=0;i<s.length();i++)
    	{
    		ans*=int(s[i]-96);
    		if(s[i]!=s[s.length()-1-i])palin=false;
    	}
    	if(!palin)cout<<ans<<endl;
    	else cout<<"Palindrome"<<endl;
    }
}
