#include <iostream>
#include <algorithm>
#include <stack>
using namespace std;

int main()
{
   
   int t,n;
   long long id,temp,top;
   char c,cc;
   stack <long long>ST,ST1;
   cin>>t;
   while(--t>=0)
  {
   	cin>>n>>id;
   	ST.push(id);
   	cc='P';
   	temp=0;
   	
   while(--n>=0)
   {
   
   	cin>>c;
   	if(c=='P')
   	{
   		cin>>id;
   		ST.push(id);
   		cc='P';
   	}
   	else if(c=='B')
   	{
   		if(cc=='B')
   		{
   			if(ST.top()!=temp)
   			{
   			ST.push(temp);
   			}
   			cc='P';
   		}
   		else
   		{
   		temp=ST.top();
   		ST.pop();
   		cc=c;
   		}
   	}
   }
   cout<<"Player "<<ST.top()<<endl;
   swap(ST,ST1);
   }
    return 0;
}
