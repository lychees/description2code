int pre[MAX];
int isprime(long long int n)
{
	long long int i,j,k;
	for(i=2;i<=sqrt(n);i++)
	{
		if(n%i==0)
			return 0;
	}
	return 1;
}
void fun() 
{ 
       int i,j; 
       for(i=0;i<MAX;i++) pre[i] = 1; 
       for(i=2;i<=(int)sqrt(MAX);i++) 
         if (pre[i]) 
           for(j=i;j*i<MAX;j++) pre[i*j] = 0; 
} 
int main()
{
	int t;
	cin>>t;
	fun();
	while(t--)
	{
		int i,m,n;

		for(i=m;i<=n;i++)
		{
			if(pre[i]==1&&i!=1)
			{
				//cout<<i<<"\n";
				sum+=i;
			}
		}

	}

}
