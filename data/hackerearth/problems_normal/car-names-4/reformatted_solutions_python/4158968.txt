N = int(raw_input())
for _ in xrange(N):
    s = raw_input()
    slen = len(s)
    seg_sz = slen//3
    flag = False
    if 3 * seg_sz == slen:
        charset = set(s)
        setlen = len(charset)
        if setlen == 3 and s[:seg_sz] == s[0] * seg_sz and s[seg_sz:2 *
                                                             seg_sz] == s[seg_sz] * seg_sz and s[2*seg_sz:] == s[2*seg_sz] * seg_sz:
            flag = True
    print "OK" if flag else "Not OK"
