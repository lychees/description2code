int main()
{
    int t;
    char s[100001];
    int l;
    bool ans;
    int r,y;
    scanf("%d\n",&t);
    while(t--)
    {
     ans=1;
     r=0;y=0;
     scanf("%s",&s);
     l=strlen(s);
     
     for(int i=0;i<l;i++)
       if(!(s[i]=='R' || s[i]=='Y'))
         ans=0;
      //cout<<ans<<endl;
     if(s[0]=='Y')
        ans=0;
     for(int i=2;i<l && ans==1;i++)
       if(s[i]=='Y' && s[i-1]=='Y' && s[i-2]=='Y')
             ans=0;
        
      if(ans ) 
        printf("YES\n");
      else
        printf("NO\n"); 
    }
   return 0;
}
