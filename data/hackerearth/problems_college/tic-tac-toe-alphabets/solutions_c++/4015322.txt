#include <iostream>
using namespace std;

int main()
{
    int n,temp,j,k,count=65,i;
    cin>>n;
    char cnst;
    while(n--)
    {
    
    string str;
    int c[26];
    for(i=0;i<26;i++)
    {
    	c[i]=0;
    }
    cin>>str;
    int len;
    len=str.size();
    for(i=0;i<len;i++)
    {
    	temp=str[i];
    	temp=temp-65;
    	c[temp]++;
    }
    for(j=0;j<26;j++)
    {
    	for(k=0;k<c[j];k++)
    	{
    		cnst=count;
    		cout<<cnst;
    	}
    	count++;
    }
    cout<<endl;
    count=65;
    
    }
    return 0;
    
}
