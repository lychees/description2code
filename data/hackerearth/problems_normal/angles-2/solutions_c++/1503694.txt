#include <iostream>
#include <cmath>
using namespace std;

int main()
{
  int can=0,to=0;
  int candraw[360]={0};
  int todraw[10]={0};
  cin >> can >> to ;
  for(int i=0;i<can;i++)
  {
    cin>>candraw[i];
  }
  for(int j=0;j<to;j++)
  {
    cin>>todraw[j];
  }
  for(int j=0;j<to;j++)
  {
    int check=todraw[j];
    int printYes=0;
    int addAngle=0,maddAngle=0;
    int subAngle=0,msubAngle=0;
    for(int i=0;i<can;i++)
    {
      subAngle=addAngle-candraw[i];
      addAngle+=candraw[i];
      if(check==candraw[i])
      {
        printYes=1;
        break;
      }
      if(check==abs(subAngle))
      {
        printYes=1;
        break;
      }
      if(check==addAngle)
      {
        printYes=1;
        break;
      }
      for(int k=1;k<100;k++)
      {
        msubAngle=addAngle-((candraw[i]*k)%360);
        maddAngle=addAngle+((candraw[i]*k)%360);
        int other=(candraw[i]*k)%360;
        if(other==check)
        {
          printYes=1;
          break;
        }
        if(abs(msubAngle)==check)
        {
          printYes=1;
          break;
        }
        if(maddAngle==check)
        {
          printYes=1;
          break;
        }
      }
    }
    if(printYes==1)
    {
      cout << "YES" << endl;
    }
    else
    {
      cout << "NO" << endl;
    }
  }
  return 0;
}
