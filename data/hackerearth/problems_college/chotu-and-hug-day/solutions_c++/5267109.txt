#include "bits/stdc++.h"
using namespace std;
long long abs(long long a,long long b)
{
	return a >b ?a-b :b-a;
}
int main()
{
	long int tc;
	cin>>tc;
	for(long int i1=1;i1<=tc;i1++)
	{
		long long n;
		cin>>n;
		long long h;
		cin>>h;
		long long ans=INT_MAX;
		for(long long i=1;i<=n;i++)
		{
			long long a;
			cin>>a;
			if(abs(h,a)<ans)
				ans=abs(h,a);
		}
		cout<<ans<<endl;	
	}
	return 0;
}
