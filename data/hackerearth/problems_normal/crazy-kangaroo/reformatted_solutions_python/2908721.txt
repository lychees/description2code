n = raw_input()
n = int(n)
for i in range(n):
    s_tmp = raw_input()
    s_tmp = s_tmp.split(" ")
    a = int(s_tmp[0])
    b = int(s_tmp[1])
    m = int(s_tmp[2])

    ans = 0
    if a % m != 0:
        a = a + m-(a % m)
    if b % m != 0:
        b -= (b % m)
    if b > a:
        ans = ((b-a)/m)+1
    print ans
