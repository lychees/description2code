#include<cstdio>
#include<iostream>
#include<cmath>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--){
		long long int n,count=0,x;
		long double m;
		scanf("%lld",&n);
		while(n>0){
			m=(log10l(n)/log10l(2));
			x= pow(2,(long long int)m);
			n = n%x;
			count++;
		}
		printf("%lld\n",count);
	}	
	return 0;
}
