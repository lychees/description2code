#include <iostream>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	
   	int hash[256]={0},c=0,i;
   	string a;
   	cin>>a;
   	for(i=0;i<a.size();i++)
   	{
   		hash[a[i]]++;
   	}
   	for(i=97;i<123;i++)
   	{
   		if(hash[i]==0)
   		break;
   		c++;
   	}
   	if(c==26)
   	cout<<"YES"<<endl;
   	else
   	cout<<"NO"<<endl;
   	
   	
   	
   	
   	
   	
   	
   	
   	
   }
   
   
   
   
   
    return 0;
}
