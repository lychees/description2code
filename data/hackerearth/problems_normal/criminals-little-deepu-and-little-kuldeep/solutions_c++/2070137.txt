#include <iostream>
#include <algorithm>
using namespace std;
#define LL long long

int main()
{
    int T;
    cin >> T;
    
    while(T--)
    {
    	int n;
    	
    	cin >> n;
    	LL a[n];
    	
    	int ans,ct;
    	
    	for(int i=0 ; i<n ; i++)
    	scanf("%lld",&a[i]);
    	
    	sort(a,a+n);
    	
    	ct=ans=1;
		for (int i = 1; i < n; i++)
		{	
			if (a[i] == a[i-1])
				ct++; 
			else
			{	
				if (ct > ans)
				ans=ct;
					
				ct=1;
			}
		}
		if (ct > ans)
		ans=ct;
    	
    	cout << ans << endl;
    	
    }
    return 0;
}
