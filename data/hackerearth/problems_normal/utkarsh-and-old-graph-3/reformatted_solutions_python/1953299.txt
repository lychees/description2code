N, M = map(int, raw_input().split())
hm = {}
for i in range(M):
    a, b = map(int, raw_input().split())
    if a in hm:
        if b not in hm[a]:
            hm[a].append(b)
    else:
        hm[a] = [b]
    if b in hm:
        if a not in hm[b]:
            hm[b].append(a)
    else:
        hm[b] = [a]

print N * (N - 1) / 2 - (N - 1)
