#include <iostream>
using namespace std;
int prime(int n)
{
	if(n==1||n==0)return 0;
	for(int i=2;i<=n/2;i++)
	{
		if(n%i==0)return 0;
	}
	return 1;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,cnt=0,temp;
    	cin>>n;
    	while(n--)
    	{
    		cin>>temp;
    		if(prime(temp))
    		cnt++;
    	}
    	cout<<cnt<<endl;
    }
    return 0;
}
