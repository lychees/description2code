#include<bits/stdc++.h>
#define ll long long int
using namespace std;
ll dp[1100];
int n;
ll rec(int x){
	if(x==n) return 1;
	if(x>n) return 0;
	if(dp[x]!=-1) return dp[x];
	ll ans=0ll;
	ans=rec(x+1);
	ans=ans+rec(x+2);
	return dp[x]=ans;
}
int main(){
	int t;
	cin>>t;
	while(t--){
		memset(dp,-1,sizeof dp);
		cin>>n;
		cout<<rec(0)<<endl;
	}	
}
