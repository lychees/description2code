using namespace std;
ll powe(ll a,ll n,ll m)
{
	ll an=1;
	while(n!=0)
	{
		if(n%2==1)
			an=(an*a)%m;
		a=(a*a)%m;
		n=n/2;
	}
	return an;
}
int main()
{
	int t;
	si(t);
	while(t--)
	{
		ll a,d,n,an;
		sll(a);sll(d);sll(n);
		a%=n;d%=n;
		if(d==0)
			an=-1;
		else if(a==0)
			an=0;
		else
		{
			ll x=powe(d,n-2,n);
			an=(x*(n-a))%n;
		}
		pll(an);
	}
	return 0;
}
