N = input()
nums = sorted(map(int, raw_input().split()))
values = [1 for i in range(N)]

for i in range(N - 1, -1, -1):
    temp = [1]
    num = nums[i]
    for j in range(i + 1, N):
        if nums[j] % num == 0:
            temp.append(values[j] + 1)
    values[i] = max(temp)

value = max(values)

if value == 1:
    print -1
else:
    print value
