#include <iostream>
#include <stdio.h>
#include <stack>
#include <vector>

using namespace std;

typedef long long int Lint;

int main() {
	
	Lint t, n, i, len, v1, v2;
	char ch;

	scanf("%lld", &t );

	while( t-- ) {

		string s, str = "", s2;
		cin >> s;
		stack<char> A;

		n = s.length();

		for( i = n - 1;i >= 0;i-- ) {

			if( s[ i ] >= '0' && s[ i ] <= '9' )
				str += s[ i ];
			else {

				while( !A.empty() ) {

					ch = A.top();
					s2 = ch;
					str += s2;
					A.pop();
				}
				A.push( s[ i ] );
			}
		}

		while( !A.empty() ) {

			ch = A.top();
			s2 = ch;
			str += s2;
			A.pop();
		}

		len = str.length();

		stack<Lint> B;

		for( i = 0;i < len;i++ ) {

			if( str[ i ] >= '0' && str[ i ] <= '9' ) 
				B.push( str[ i ] - '0' );
			else {

				v1 = B.top();
				B.pop();
				v2 = B.top();
				B.pop();
				switch( str[ i ] ) {

					case '+' : B.push( v2 + v1 );
					break;
					case '-' : B.push( v2 - v1 );
					break;
					case '*' : B.push( v2 * v1 );
					break;
				}
			}
		}

		printf("%lld\n", B.top() );
	}

	return 0;
}
