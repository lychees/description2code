#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,m;
		cin>>n>>m;
		for(int i=0;i<n;i++)
		{
		   int x,y,z,p=0;
		   int min=10000000;
		   //int *a=new int [m];
		   for(int i=0;i<m;i++)
		    {
		 	  cin>>x>>y>>z;
		 	    if((100-x)*(100-y)*(100-z)<min)
		 	     {
		 	     	min=(100-x)*(100-y)*(100-z);
		 	     	p=i;
			     }
		    }
		 cout<<p+1<<" ";
	   }
	   cout<<endl;
}
}
