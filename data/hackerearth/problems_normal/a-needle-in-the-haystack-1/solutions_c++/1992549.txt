#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;
int main()
{
 	int t;
 	cin>>t;
 	while(t--)
 	{
	 		  int f=0;
	 		  int count1[26]={0},count2[26]={0};
	 		  char s1[1001],s2[1000001];
	 		  cin>>s1>>s2;
			   int n=strlen(s1);
			   for(int i=0;i<n;i++)
			   count1[s1[i]-'a']++;
			   int n2=strlen(s2);
			   for(int j=0;j<n&&j<n2;j++)
			   {
			   		   count2[s2[j]-'a']++;
			   }
			   int i;
			   for(i=0;i<26;i++)
			   {
			   		   if(count1[i]!=count2[i])
			   		   break;
			   }
			   if(i==26)
			   f=1;
			   else
			   {
			   	   for(int j=n;j<n2;j++)
			   	   {
				   		   count2[s2[j-n]-'a']--;
				   		   count2[s2[j]-'a']++;
						   for(i=0;i<26;i++)
			   			   {
			   			      if(count1[i]!=count2[i])
			   				     break;
						   }
			   if(i==26)
			   {f=1;break;}
			   	   		   
				   }
			   }
 		   	   if(f==1)
 		   	   cout<<"YES\n";
 		   	   else
 		   	   cout<<"NO\n";
	}
 	
}
