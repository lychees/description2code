#include <algorithm>
#include <iostream>
#include <queue>
#include <stdio.h>
#include <vector>

using namespace std;
priority_queue<long long> q;
long long n, k, a, b, tmp, i;
main() {
  cin >> n >> k;
  for (i = 1; i <= k; i++)
    q.push(0);
  for (i = 1; i <= n; i++) {
    scanf("%d%d", &a, &b);
    tmp = q.top();
    q.pop();
    tmp = min(tmp, -a);
    printf("%I64d\n", b - tmp);
    tmp -= b;
    q.push(tmp);
  }
}
