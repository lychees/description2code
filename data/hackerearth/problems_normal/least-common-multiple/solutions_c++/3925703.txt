typedef unsigned long long int ull;

ll prime(ll n)
{
	if(n==1)return 0;
	if(n==2)return 1;
	if(n%2==0)return 0;
	for(ll i=3;i*i<=n;i+=2)
	{
		if(n%i==0)return 0;
		
	}
	return 1;
	
    
}
ll pow(ll a,ll b)
{
	ll res=1;
	while(b--) {
		res*=a;
		if(res > 1e16)return res;
	}
	return res;
}
