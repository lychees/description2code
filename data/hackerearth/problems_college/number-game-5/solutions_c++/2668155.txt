#include <iostream>
#include <bits/stdc++.h>

using namespace std;
#define ll long long

vector <ll> v;
int bsearc(long long int k)
{
	ll i , j;
	i = 1; j = 26000;
	while(i < j) {
		ll n = (i+j)/2;
		ll x = (n*(n+1))/2;
		ll x2 = pow(x,2);
		ll temp = x2 + (x*2*(2*n+1))/3 + x;
		//temp /= 6;
		//cout << temp << " " << i << " " << j <<  endl;
		if(temp == k || i +1 == j )
			return n;
		if(temp > k) {
			j = n;
		}
		else{
			i = n;
		}
	}
	return i;
}


int main()
{
    ll t,n;
   scanf("%lld",&t);
    
    for(int i = 0; i <= 26000 ; i++) {
    	ll n = i;
		ll x = (n*(n+1))/2;
		ll x2 = pow(x,2);
		ll temp = x2 + (x*2*(2*n+1))/3 + x;
    	v.push_back(temp);
    }
    while(t--) 
    {
    	scanf("%lld",&n);
    	if(n <= 3) {
    		puts("0");
    		puts("Draw");
    	}
    	else {
    		std::vector<ll>::iterator low;
    		low=std::lower_bound (v.begin(), v.end(), n); //  
    		ll no = low-v.begin();
    		if(v[no] > n)
    			no--;
    		printf("%lld\n",no);
    		if(no%2 == 1)
    			puts("Dhruv");
    		else
    			puts("Ketan");
    	}
    }
    return 0;
}
