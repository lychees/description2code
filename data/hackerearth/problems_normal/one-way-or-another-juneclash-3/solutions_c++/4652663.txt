#include <iostream>
using namespace std;
int main()
{  int n,i,t,k,o[2],x,y,flag,f[10];  long int r,c,a;
    cin>>t;
    for(k=0;k<t;k++)
 	{flag=1; cin>>n>>r>>c>>a;
  	x=(r+c)%2; y=(a%2);
 	o[x]=y; o[(!x)]=(!y);
 	for(i=0;i<n-1;i++){
 	cin>>r>>c>>a;
	 x=(r+c)%2; y=(a%2);
	 if(o[x]!=y) flag=0;	
	 }f[k]=flag;}
	 for(k=0;k<t;k++) { if(f[k]==0) cout<<"\nNo"; else if(f[k]==1) cout<<"\nYes";
	 } return 0;}
