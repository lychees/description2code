#include<iostream>

using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	int i;
       int c[26]={0};
    	for(i=0;i<s.size();i++)
    	{
    		int x=s[i]-'a';
    		c[x]++;
    	}
    	int max=0;
    for(i=0;i<26;i++)
    {
    	if(c[i]>max)
    	max=c[i];
    }
    cout<<max<<endl;
    }
    return 0;
}
