using namespace std;
const ll mod = (ll)1e9 + 7;
const int maxn = 3e5 + 5;
int n, m, k;
string a, b;
int dp[202][502][202];
int f(int i, int j, int l) {
  // std::cout << "start" << i << " " << l << "\n";
  if (i + m - 1 > n)
    return 0;
  int &res = dp[i][j][l];
  if (res != -1)
    return res;
  int cst = 0;
  rep(g, 1, m) {
    char z, y;
    if (l != 0 and (i + g - 1) <= l + m - 1) {
      if (a[i - l + g] != a[g])
        return res = f(i + 1, j, l);
    } else {
      z = max(a[g], b[i + g - 1]);
      y = min(a[g], b[i + g - 1]);
      cst += min(z - y, y - 97 + 'z' - z + 1);
    }
  }
  if (cst <= j)
    res = max(res, 1 + f(i + 1, j - cst, i));
  res = max(res, f(i + 1, j, l));
  return res;
}
int main() {
  ios::sync_with_stdio(false);
  std::cin >> m >> n >> k;
  std::cin >> a >> b;
  a = "@" + a;
  b = "@" + b;
  memset(dp, -1, sizeof(dp));
  std::cout << f(1, k, 0) << "\n";
}
