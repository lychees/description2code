#include<iostream>
#include<algorithm>

using namespace std;
int main()
{
	int t;
	cin>>t;
	long n;
	int count;
	long long a[100000];
	while(t--)
	{
		count =0;
		cin>>n;
		for(int i=0;i<n;i++)
			cin>>a[i];
		sort(a,a+n);
		for(int i=0;i<n-1;i++)
			{if(a[i]==a[i+1])
				count++;
			}
	  cout<<count<<endl;
	
	}
	
	return 0;
}
