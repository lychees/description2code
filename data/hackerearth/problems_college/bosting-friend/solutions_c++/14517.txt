#include<cstdio>
int coprime(int x)
    {
        int t=100;
            if(x%2==0) {t/=2;x/=2;}
            if(x%2==0) {t/=2;x/=2;}
            if(x%5==0) {t/=5;x/=5;}
            if(x%5==0) {t/=5;x/=5;}
            return t;
    }
main(void)
{
int t;
scanf("%d",&t);
    for(int i=1;i<=t;++i)
        {
            int flag=1;
            int PD,PG;
            long long N;
            scanf("%lld%d%d",&N,&PD,&PG);
            if( coprime(PD)> N ) flag=0;
            else if( PD!=100 && PG==100) flag=0;
            else if(PD!=0 && PG==0) flag=0;
        
            if(flag==1) printf("#%d:Possible\n",i);
            else printf("#%d:Broken\n",i);
            
        }
}
