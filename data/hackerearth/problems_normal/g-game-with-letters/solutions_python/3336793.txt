t = int(raw_input())

for ti in xrange(t): 
	a, b = raw_input().split()
	word = {}
	for c in b:
		if c in word:
			word[c] += 1
		else:
			word[c] = 1
	bestTeam = -1
	bestScore = -1
	minSize = -1
	for team in xrange(int(a)):
		s = raw_input()
		teamSize = len(s)
		score = 0
		for c in s:
			if c in word:
				score += word[c]
		if score > bestScore or score == bestScore and minSize > teamSize:
			bestScore = score
			minSize = teamSize
			bestTeam = team
	print bestTeam+1
