#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n, k;
    int raju[10], rani[10];
    bool rajuknow[15001];
    for(int i=0; i<15001; i++)
        rajuknow[i]=false;
    cin>>n>>k;

    for(int i=0; i<n; i++)
    {
        //cin>>raju[i];
        scanf("%d",&raju[i]);
        rajuknow[raju[i]]=true;
    }
    for(int i=0; i<k; i++)
        //cin>>rani[i];
        scanf("%d",&rani[i]);

    for(int i=0; i<15001; i++)
    {
        for(int j=0; j<n; j++)
        {
            if(i-raju[j]<0)
                continue;
            if(rajuknow[i-raju[j]]==true)
                rajuknow[i]=true;
        }
    }

    /*for(int i=0; i<3600; i++)
        if(rajuknow[i]==true)
            cout<<i<<endl;*/

    int flag;
    for(int i=0; i<k; i++)
    {
        flag=0;
        for(int j=0; j<35; j++)
        {
            int ang = 360*j + rani[i];
            //cout<<ang<<endl;
            if(rajuknow[ang]==true)
            {
                flag=1;
                break;
            }
        }
        if(flag==1)
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
    }

    return 0;
}
