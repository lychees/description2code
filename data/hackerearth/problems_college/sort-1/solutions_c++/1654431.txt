#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    long T;
    cin >> T;
    while(T--)
    {
    	long n;
    	cin >> n;
    	long a[n];
    	for(long j=0;j<n;j++)
    	{
    		cin >> a[j];
    	}
    	sort(a,a+n);
    	for(long i=0;i<n;i++)
    	{
    		cout << a[i]<<' ';
    	}
    	cout<<endl;
    }
    return 0;
}
