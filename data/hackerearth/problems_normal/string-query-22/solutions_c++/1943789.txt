#include <bits/stdc++.h>
using namespace std;
int main()
{
    string s;
    cin>>s;
    vector <int> v[150];
    for(int i=0;i<s.length();i++)
    {
    	v[s[i]].push_back(i);
    }
    int k;
    char ch;
    int m;
    cin>>m;
    while(m--)
    {
    	cin>>k;
    	cin>>ch;
    	s[v[ch][k-1]]='0';
    	v[ch].erase(v[ch].begin()+k-1);
    }
    for(int i=0;i<s.length();i++)
    {
 		if(s[i]!='0')
 		{
 			cout<<s[i];
 		}
    }
    cout<<endl;
 
}
