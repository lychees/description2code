    {
        int n; cin>>n; int arr[n][n];
        for (int i=0; i<n; i++)
        {
            for (int j=0; j<n; j++)
            {
                arr[i][j]=-1;
            }
        }
        int q; cin>>q; int a,b;
        while(q--)
        {
            cin>>a>>b; int a1=n-a-1,b1=b;
            for (int i=a1+1; i<n; i++)
            {
                for (int j=b1+1; j<n; j++)
                {
                    arr[i][j]=0;
                }
            }
            for (int i=0; i<a1; i++)
            {
                for (int j=0; j<b1; j++)
                {
                    arr[i][j]=0;
                }
            }
        }
        for (int i=0; i<n; i++)
        {
            if (arr[n-1][i]==-1)
            {
                arr[n-1][i]=1;
            }
        }
        for (int i=n-1; i>=0; i--)
        {
            if (arr[i][0]==-1)
            {
                arr[i][0]=1;
            }
        }
        arr[n-1][0]=0;
        for (int i=n-2; i>=0; i--)
        {
            for (int j=1; j<n; j++)
            {
                if (arr[i][j]==-1)
                {
                    ll k=(ll)arr[i][j-1]+(ll)arr[i+1][j];
                    k%=M;
                    arr[i][j]=k;
                }
            }
        }
        cout<<arr[0][n-1]<<endl;


    }
