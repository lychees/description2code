#include<bits/stdc++.h>
using namespace std;
bool check_divisor(long x)
{
	int count=0;
	for(int i=1;i*i<=x;i++)
	{
		if(x%i==0)
		{
			if(i*i==x)
			count++;
			else
			count+=2;
		}
		if(count>=4)
			break;
	}
	if(count<4)
	return 1;
	return 0;
}
int main()
{
	long n,i,p;
	cin>>n;
	long x[n];
	for(i=0;i<n;i++)
	{
		cin>>x[i];
		if(check_divisor(x[i]))
			cout<<"NO"<<endl;
		else
			cout<<"YES"<<endl;
	}
	return 0;
}
