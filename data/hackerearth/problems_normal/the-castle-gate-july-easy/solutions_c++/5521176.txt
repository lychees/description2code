#include <iostream>
using namespace std;

int main()
{
    int n,t,i,j,k;
    
    cin>>t;
    
    while(t--)
    {
    	cin>>n;
    	k=0;
    	
    	for(i=1;i<=n;i++)
    	{
    		for(j=i+1;j<=n;j++)
    		{
    			if((i^j)<=n)
    				k++;
    		}
    	}
    	cout<<k<<endl;
    }
    
    return 0;
}
