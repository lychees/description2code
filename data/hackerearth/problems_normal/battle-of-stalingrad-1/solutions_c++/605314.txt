#include <iostream>
#include <cmath>
using namespace std;

float area(int a,int b,int c,int d,int e,int f)
{
    return abs(0.5*((a*(d-f)) + (c*(f-b)) + (e*(b-d))));
}

int main()
{
    int tims;
    cin >> tims;
    int a[8];
    int i = 0;
    float a1,a2,a3,tot;
    while (i < tims)
    {
    	cin >> a[0] >> a[1] >> a[2] >> a[3] >> a[4] >> a[5] >> a[6] >> a[7];
    	for (int j = 0; j<9;j++)
        {
            tot = area(a[0],a[1],a[2],a[3],a[4],a[5]);
            a1 = area(a[6],a[7],a[2],a[3],a[4],a[5]);
            a2 = area(a[0],a[1],a[6],a[7],a[4],a[5]);
            a3 = area(a[0],a[1],a[2],a[3],a[6],a[7]);
        }
        if (tot == a1+a2+a3)
        {
            cout << "INSIDE" << endl;
        }
        else
            cout << "OUTSIDE" << endl;
        i++;
    }
}
