from collections import Counter

inp=[int(x) for x in raw_input().split()]

k=inp[1]

values=[int(x) for x in raw_input().split()]

final=Counter(values)
flag=True
for val in values:
    temp=k-val
    if final.has_key(temp):
        if (temp==val and final[temp]>1) or temp!=val:
            print "YES"
            flag=False
            break

if flag:
    print "NO"
