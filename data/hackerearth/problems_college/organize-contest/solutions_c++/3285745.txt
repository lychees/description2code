#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    string str;
    int point;
    map <string,int> mp;
    vector < pair<string,int> > v;
    while(t--) {
    	cin >> str >> point;
    	v.push_back({str,point});
    	if(point > 0)
    	mp[str] += point;
    	else {
    		mp[str] = -9999999;
    	}
    }
    int maxi = -1;
    for(auto it = mp.begin(); it != mp.end(); it++) {
    	if(it->second > maxi) {
    		maxi = it->second;
    	}
    }
    string ans;
    mp.clear();
    for(int i = 0; i < v.size(); i++) {
    	mp[v[i].first] += v[i].second;
    	if(mp[v[i].first] == maxi) {
    		ans = v[i].first;
    		break;
    	}
    }
    cout << ans << endl;
    return 0;
}
