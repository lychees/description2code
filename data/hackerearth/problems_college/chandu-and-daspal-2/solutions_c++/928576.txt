#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int t,x,y,temp,tempx,tempy,count;
    char a[7];
    
    cin>>t;
    while(t-->0)
    {
    	cin>>x>>y>>a;
    	if(y>x)
    	{
    		temp=x;
    		x=y;
    		x=temp;
    	}
    	count=0;
    	while(x>0&&y>0)
    	{
    		count++;
    	tempx=x;
    	tempy=y;
    	while(y!=0)
    	{
    	temp=y;
    	y=x%y;
    	x=temp;
    	}
    	y=tempy-x;
    	x=tempx-x;
    	}
    	if((count%2)!=0)
    	cout<<a<<endl;
    	else 
    	{
    		if(strcmp(a,"Chandu")==0)
    	cout<<"Daspal"<<endl;
    	else cout<<"Chandu"<<endl;
    	}
    }
    return 0;
}
