#include<cstdio>
#include<iostream>
#include<cmath>
using namespace std;
typedef long long int ll;
typedef unsigned long long ull;
ll binarysearch(ll a, ll b, ll c, ll d, ull k);

int main(){
		int t;
		scanf("%d",&t);
		while(t--){
				ll a,b,c,d;
				ull k;
				scanf("%lld %lld %lld %lld %llu",&a,&b,&c,&d,&k);
				cout<<binarysearch(a,b,c,d,k)<<endl;
			}
}
ll binarysearch(ll a, ll b, ll c, ll d, ull k){
		ll low = 1, high = 1000000, mid,ans = 0;
		while(low<=high){
				mid = low+((high-low)/2);
				ull foo;
				foo = (ull)(a*mid*mid*mid + b*mid*mid + c*mid + d);
				if(foo <= k){
						ans = mid;
						low = mid+1;
					}
				else{
						high = mid-1;
					}
			}
		return ans;
	}
