#include <bits/stdc++.h>
using namespace std;
//ETF topcoder tutorial
int fi(int n)
{          
		int result = n;          
		for(int i=2;i*i <= n;i++) 
		{            
			if (n % i == 0) result -= result / i;            
			while (n % i == 0) n /= i;          
		}          
		if (n > 1) result -= result / n;          
		return result;        
	}  

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int z,ans;
		scanf("%d",&z);
		ans=fi(z);
		cout<<ans<<endl;
	}
	return 0;
}
