#include<iostream>
#include<algorithm>
using namespace std;
int a[100007];
int prefmax[100007], suffmax[100007];
int main()
{
	int n;
	cin >> n;
	for (int i = 1; i <= n; i++)
	{
		cin >> a[i];
		prefmax[i] = max(prefmax[i - 1], a[i]);
	}
	for (int i = n; i >= 1; i--)
	{
		suffmax[i] = max(suffmax[i + 1], a[i]);
	}
	for (int i = 1; i <= n; i++)
	{
		if (prefmax[i] == a[i] || suffmax[i] == a[i])
		{
			cout << i << ' ';
		}
	}
	cout << endl;
}
