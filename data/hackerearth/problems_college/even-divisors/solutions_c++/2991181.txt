#include <iostream>
#include<math.h>
using namespace std;

int main()
{
	int t,c;
    long n;
    c=0;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	if(n%2!=0)
    	{
    		cout<<"0"<<endl;
    	}
    	else if(n==2)
    	{
    		cout<<"1"<<endl;
    	}
    	else
    	{
    	   c=n/2;
    		int c1=2;
    		
    		for(int i=2;i<=sqrt(c);i++)
    		{
    			if(c%i==0)
    			{
    				if(i == c/i)
    				{
    					c1+=1;
    				}else
    				{
    					c1+=2;
    				}
    			}
    		}
    		cout<<c1<<endl;
    	}
    
    }
    return 0;
}
