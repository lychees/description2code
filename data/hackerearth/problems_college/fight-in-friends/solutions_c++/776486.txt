#include<bits/stdc++.h>
using namespace std;
#define F first
bool matrix_array[1002][1002];
bool visitedarray[100002];

int breadth(int a,int b,int n)
{
    queue<pair<int,int> >q;
    q.push(make_pair(a,0));
    visitedarray[a] = 1;
    pair<int,int> p;

    while(!q.empty())
    {   p = q.front();
        q.pop();

        for(int i = 1; i<=n ; i++)
        {
            if(  matrix_array[p.F][i] == 0 && visitedarray[i] == 0 )
            {
                visitedarray[i] = 1;
                if(i == b)
                    return p.second+1;
                q.push(make_pair(i,p.second+1));
            }
        }
    }
    return -1;
}

main()
{
    int n,m,a,b,i;
    cin>>n>>m>>a>>b;
    for( i = 0 ; i<m ; i++)
    {
        int x,y;
        cin>>x>>y;
        matrix_array[x][y] = 1;
        matrix_array[y][x] = 1;
    }
    cout<<breadth(a,b,n)<<endl;




    return 0;
}
