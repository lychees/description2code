#include <iostream>
using namespace std;

int main()
{
	int n,b;
	cin>>n;
	int prateek[n],psum = 0;
	for(int k = 0;k<n;k++)
	{
		cin>>prateek[k];
		psum += prateek[k];
	}
	cin>>b;
	int chiku[b],diff[b];
	int sum=0;
	for(int i=0;i<b;i++)
	{
		int m,tot,ptot=0;
		cin>>m;
		int pro[m];
		for(int j=0;j<m;j++)
		{
			cin>>pro[j];
			ptot += prateek[pro[j]-1];
		}
		cin>>tot;
		diff[i] = tot - ptot;
		sum += tot;
	}
	if(sum <= psum)
	{
		sum = 0;
		for(int i=0;i<b;i++)
		{
			if(diff[i]>0)
			{
				sum += diff[i];
			}
		}
	}
	else
	{
		int sumdiff = 0,pos = 0;
		for(int i=0;i<b;i++)
		{
			sumdiff += diff[i];
			if(diff[i]>0)
				pos += diff[i];
		}
		if(sumdiff<(sum-psum))
		{
			sum = sum-psum;
		}
		else
		{
			sum = pos;
			
		}
	}
    cout<<sum;
}
