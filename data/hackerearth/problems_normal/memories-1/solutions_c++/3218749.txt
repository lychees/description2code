#include <bits/stdc++.h>
using namespace std;

int main() {
    int T;
    cin>>T;
    while(T--){
        int n,p;
        cin>>n>>p; 
        pair<int,int> pairs[p];
        for(int i=0 ; i<p;i++){
            int x ,y;
            cin>>x>>y;
            pairs[i] = make_pair(x,y);
        }
        int ans = 0;
        for(int i=1 ;i<(1<<n); i++){
            int flag =1;
            for(int j =0 ; j<p; j++){
                int x = pairs[j].first;
                int y = pairs[j].second ; 
                x-- , y--; 
                if(i>>x&1 && i>>y&1){
                    flag = 0;
                    break;
                }
                
            }
            if(flag ==1) { 
                ans++; 
                // cout<<i<<"\n";
                
            }
        }
      cout<<ans<<"\n";  
    }
    // cout<<"here "<<(2>>4);
	return 0;
}
