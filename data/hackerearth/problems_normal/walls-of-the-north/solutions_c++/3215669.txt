int n,i,a[100005],b[100005],ans,cnt;
int main()
{
	//freopen("i.in", "r", stdin);
  	//freopen("o.out", "w", stdout);
  	ios_base::sync_with_stdio(false);
  	cin.tie(0);
  	int t;
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(i=0;i<n;i++)
		cin>>a[i];
		for(i=0;i<n-1;i++)
		{
			if(a[i]>a[i+1])
				b[i]=0;
			else if(a[i]<a[i+1])
				b[i]=1;
			else
			{
				if(i==0)
					b[i]=0;
				else
					b[i]=1-b[i-1];
			}
		}
		ans=0;
		for(i=0;i<n-1;i++)
		{
			ans++;
			while(i<n-2 && b[i]==b[i+1])
				i++;
		}
		if(a[0]==a[1])
		{
			b[0]=1-b[0];
			cnt=0;
			for(i=0;i<n-1;i++)
			{
				cnt++;
				while(i<n-2 && b[i]==b[i+1])
					i++;
			}
		}
		cout<<max(ans+1,cnt+1)<<"\n";
	}
	return 0;
}
