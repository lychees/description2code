#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    int n,k;
    cin>>n>>k;
    ll sum=0;
    ll int ar[n];
    for(int i=0; i<n; i++)
        cin>>ar[i];
    sort(ar,ar+n);
    ll int f=k,temp=1;
    for(int i=n-1; i>=0; i--)
    {
        if(f)
        {
            sum+=temp*ar[i];
            f--;
        }
        else
        {
            f=k;
            temp++;
            i++;
        }
    }
cout<<sum<<endl;
}
