#include <bits/stdc++.h>
using namespace std;
int solve(vector<int> A,vector<int> B,int i, int j, int n) {
	if(i == n or j == n)
		return 0;
	if(j == n-1)
		return A[i] > B[j] ? 100 : A[i] == B[j] ? 50 : 0;
	int res = 0;
	for(int k=i;k<n;k++) {
		for(int l=j;l<n;l++) {
			res = max(res,(A[k] > B[l] ? 100 : A[k] == B[l] ? 50 : 0)+
							solve(A,B,k+1,l+1,n));
		}
	}
	return res;
}
int main()
{
    int n;
    cin>>n;
    vector<int> A(n),B(n);
    for(int i=0;i<n;i++)
    	cin>>A[i];
    for(int i=0;i<n;i++)
    	cin>>B[i];
    sort(A.rbegin(),A.rend());
    sort(B.rbegin(),B.rend());
    cout<<solve(A,B,0,0,n)<<endl;
    return 0;
}
