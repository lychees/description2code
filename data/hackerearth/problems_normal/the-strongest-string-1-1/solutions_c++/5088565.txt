using namespace std;
const int N=33;
int dis[N];
string ans;
vector<pair<char,int> > all;
inline bool com(pair<char,int> a,pair<char,int> b){
    if(a.first==b.first){
       return  a.second<b.second;
    }
    return a.first>b.first;
}
string s;
int sz;
int main(){
    cin>>sz>>s;
    for( int i=0;i<sz;i++){
        all.push_back(make_pair(s[i],i));
    }
    sort(all.begin(),all.end(),com);
    int pos=-1;
    ans="";
    for(int i=0;i<sz;i++){
        if(all[i].second>pos&&!dis[all[i].first-'a']){
            ans+=all[i].first;
            pos=all[i].second;
            dis[all[i].first-'a']=1;
        }
    }
    cout<<ans<<endl;
    return 0;
}
