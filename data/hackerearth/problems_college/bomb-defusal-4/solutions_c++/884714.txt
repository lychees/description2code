#include <iostream>
using namespace std;
int mod(int x)
{
    if(x>0)
        return x;
    return -x;
}
int main() {
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int p,x,y,u,v,temp_x,temp_y;
        scanf("%d %d %d %d %d",&p,&x,&y,&u,&v);
        temp_x=u-x;
        temp_y=v-y;
        if(mod(temp_y-temp_x)%2==1)
            printf("Terrorists Win !\n");
        else
        {
            int time_req;
            if(temp_x==temp_y && temp_x==0)
                time_req=0;
            else if (mod(temp_y-temp_x)==0)
                time_req=1;
            else
                time_req=2;
            if(time_req<p)
                printf("Counter-Terrorists Win !\n");
            else
                printf("Terrorists Win !\n");
        }
    }
    return 0;
}
