#include <iostream>
using namespace std;

int main()
{
    long int n,q,i,l,r,sum;
    cin>>n>>q;
    long int a[n];
    for(i=0;i<n;i++)
    cin>>a[i];
    while(q--){
   	sum=0;
    cin>>l>>r;
    for(i=l-1;i<=r-2;i++){
    	sum+=a[i];
    }
    cout<<(sum+a[r-1])<<" "<<(a[r-1]-sum)<<endl;
    }
}
