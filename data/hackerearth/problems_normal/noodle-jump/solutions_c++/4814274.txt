#include <bits/stdc++.h>
using namespace std;
void maxJumps(long long int *a,long long int n,long long int k)
{
	long long int i=0,count=0;
	sort(a,a+n);
	while((i+1)<n)
	{
		if((a[i+1]-a[i])<=k)
		{
			count++;
		}
		else{break;}
		i++;
	}

int main()
{
