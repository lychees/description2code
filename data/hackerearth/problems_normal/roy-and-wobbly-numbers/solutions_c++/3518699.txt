#include <iostream>

using namespace std;

int main()
{
    int n,t,k,num,m,add;
    cin>>t;
    for(int i=0;i<t;i++)
    {
        cin>>n>>k;
        if(k>81)
            cout<<-1<<endl;
        else
        {  add=(k+10)/10;
           if(k%10<2)
              add=add-1;
           num=k+9+add;
           m=n;
           if(n%2!=0)
             n=n-1;
           for(int j=0;j<n/2;j++)
              cout<<num;
           if(m%2!=0)
              cout<<num/10;
           cout<<endl;
        }

    }
    return 0;
}
