#include <iostream>
#include<vector>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int a,b,i=0;
    	vector<int> v;
    	cin>>a>>b;
    	while(a)
    	{
    		v.push_back(a%b);
    		a/=b;
    	}
    	for(i=v.size()-1;i>=0;i--)cout<<v[i];
    	cout<<endl;
    }
    return 0;
}
