int powr (int a, int b)
{
    if (b == 0)
        return 1;
    int x = powr(a, b/2);
    if (b % 2 == 0)
        return (x*x);
    else
        return (((x*x))*a);
}

string pr[79000];

void conv(int x, int i) {
    while(x > 0) {
        pr[i].pb(x%10 + 48);
        x/=10;
    }
    while(pr[i].size() < 6) pr[i].pb('0');
}

void pre() {
    int i,j;
    for(i = 2; i <= 1000000; i++) {
        if(mark[i] == 0) {
            conv(i,cnt);
            cnt++;
            for(j = i+i; j <= 1000000; j+=i) {
                mark[j] = 1;

    }
}
