#include<bits/stdc++.h>
using namespace std;

int main(){
    string str="";

    cin>>str;
    int x;

    vector<long long int> arr(26);
    int size = str.size(),i;

    for(i = 0; i < size; ){
        if(str[i] >= 'a' && str[i] <= 'z'){
            char temp = str[i];
            i++;
            string num = "";
            while(i < size && str[i] >= '0' && str[i] <= '9'){
                num = num + str[i];
                i++;
            }
            int x;
            stringstream convert(num);
            convert >> x;
            arr[temp - 'a'] += x;
        }
    }
    int q;
    cin>>q;
    while(q--){
        long long int k, temp = 0,i = 0;
        cin>>k;
        while((i < 26) && (temp < k)){
            temp = temp + arr[i];
            i++;
        }
        if(temp < k){
            cout<<-1<<endl;
            continue;
        }
        char ans = 'a' + i - 1;
        cout<<ans<<endl;
    }
}
