#include<stdlib.h>
#include<iostream>
using namespace std;
int main(){
    int T;
    cin >> T;
    while(T--){
        long long m,n;
        cin >> m >> n;
        long long mini = min(m, n);
        long long sum =0;
        long long gfgf = (mini*(mini-1))/2;
        long long a = gfgf*n;
        long long b = gfgf*m;
        sum = (mini*(mini+1));
        sum = sum +a + b;
        cout << sum << endl;
    }
    return 0;
}
