#include <cstdio>
#include <unordered_map>
#define ul unsigned long
using namespace std;
ul m=1000000007;
unordered_map<ul,ul> dp;
void initialize(ul n=100001)
{
	for(ul i=0;i<=n;i++)
		dp[n] = 0;
}
ul ppair(ul x)
{
	if(dp[x] != 0)
		return dp[x];
	else if(x==1)
		return 1;
	else if(x==2)
		return 2;
	else
	{
		dp[x] = (ppair(x-1)%m + ((x-1)%m)*(ppair(x-2)%m)%m)%m;
		return dp[x];
	}
}
int main()
{
	ul t;
	scanf("%lu",&t);
	while(t--)
	{
		ul n;
		scanf("%lu",&n);
		printf("%lu\n",ppair(n));
	}
	return 0;
}
