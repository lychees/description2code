#include <cstdio>
#include <cstdlib>
long long a[100000][2];
long long llb(long long a)
{
    if(a>0) return a;
    return -a;
}
int main()
{
	int n,t,i;
	long long y,x,btime,normaltime,fx,fy;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		for(i=0;i<n;i++) {scanf("%lld",&a[i][0]);}
		for(i=0;i<n;i++) {scanf("%lld",&a[i][1]);}
		scanf("%lld%lld",&fx,&fy);
		scanf("%lld%lld",&normaltime,&btime);
		x=(llb(fy)+llb(fx))*normaltime;
		//printf("%lld %lld %lld\n",normaltime,btime,x);
		for(i=0;i<n;i++)
		{
			y=(llb(a[i][0])+llb(a[i][1]))*normaltime;
			y=y+btime*(llb(a[i][0]-fx)+llb(a[i][1]-fy));
			if(x>y)
				x=y;
			//printf("%lld\n",x);
		}
		printf("%lld\n",x);
	}
    return 0;
}
