#include<cstring>
#include<vector>
#include<cassert>
#define mx 10009
using namespace std;
vector<bool> sieve(mx,false);
long long nos[mx];
void sv()
{
  int i,j;
  for(i=1;i<=mx;i++)
  nos[i]=i;
  for(i=2;i<mx;i++)
  {
    if(!sieve[i])
    {
      nos[i]--;
     for(j=i+i;j<mx;j+=i)
     {sieve[j]=true;
     nos[j]-=(nos[j]/i);}}}
     }
int main()
{
	
    sv();
    long long pre[mx];
    memset(pre,0,sizeof(pre));
    pre[1]=0;
    for(int i=2;i<mx;i++)
    pre[i]=pre[i-1]+nos[i];
    long long t,n;
    scanf("%lld",&t);
    while(t--)
    {
      scanf("%lld",&n);
      assert(n>=0);
      
      printf("%lld\n",pre[n]);
      }
      return 0;
      }
