#include <iostream>
using namespace std;

int main()
{	
	int t,l,r,min;
    cin>>t;
    while(t--)
    {
     cin>>l>>r;min=0;
     for(int i=l;i<=r;i++)
     {	
     	for(int j=i;j<=r;j++)
     	{
     		if((i^j)>min)
     		min=i^j;
     		
     		
     	}
     }
     cout<<min<<"\n";
     
    }
    return 0;
}
