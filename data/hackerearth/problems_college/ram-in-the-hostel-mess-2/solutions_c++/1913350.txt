#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define read(x) scanf("%lld", &x)
#define write(x) printf("%lld\n", x)

int main()
{
    ll t, n, Ans, fib[100001];
    fib[0]=fib[1]=1;
    for(int i=2; i<100000; i++) fib[i] = fib[i-1] + fib[i-2];
    read(t);
    while(t--)
    {
        read(n);
        write(fib[n+1]*fib[n+1]);
    }
    return 0;
}
