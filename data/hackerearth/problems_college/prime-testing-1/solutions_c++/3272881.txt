#define LIMIT 1000 /*size of integers array*/
int arr[10005];
void precompute()
{
    unsigned long long int i,j;
    int primes[1005];
    int z = 1;
    int ctr=0;
    //primes = malloc(sizeof(int)*LIMIT);

    for (i=2; i<LIMIT; i++)
        primes[i]=1;

    for (i=2; i<LIMIT; i++)
        if (primes[i])
            for (j=i; i*j<LIMIT; j++)
                primes[i*j]=0;

    for (i=2; i<LIMIT; i++)
        if (primes[i])
        {
            z++;
            arr[ctr]=i;
            ctr++;
        }
}
