import math
t = int(raw_input())
for i in range(0, t):
    m, n = [int(w) for w in raw_input().split()]
    m, n = m % 1000000007, n % 1000000007
    if (m >= n-1):
        print((math.factorial(m) % 1000000007)*(math.factorial(m+1) /
              math.factorial(m+1-n)) % 1000000007) % 1000000007
    else:
        print -1
