#include <iostream>
using namespace std;

string s[10];
string op[4][10];
int main()
{
    int t;
    cin >> t;
    while (t--) {
    	
    	for (int i = 0; i < 10; i++) {
    		cin >> s[i];
    		for (int j = 0; j < 10; j++) {
    			if (s[i][j] == 'X') s[i][j] = '_';
    			else s[i][j] = 'X';
    		}
    	}
    	
    	for (int i = 0; i < 4; i++) {
    		for (int  j= 0; j < 10; j++) {
    			cin >> op[i][j];
    		}
    	}
    	
    	for (int i = 0; i < 4; i++) {
    		int f = 0;
    		for (int j = 0; j < 10; j++) {
    			for (int k = 0; k < 10; k++) {
    				if (s[j][k] != op[i][j][k]) {
    					f = 1;
    					break;
    				}
    			}
    			if (f) break;
    		}
    		if (!f) {
    			cout << i + 1 << endl;
    			break;
    		}
    		f = 0;
    		for (int j = 0; j < 10; j++) {
    			for (int k = 0; k < 10; k++) {
    				if (s[j][k] != op[i][k][9-j]) {
    					f = 1;
    					break;
    				}
    			}
    			if (f) break;
    		}
    		if (!f) {
    			cout << i + 1 << endl;
    			break;
    		}
    		f = 0;
    		for (int j = 0; j < 10; j++) {
    			for (int k = 0; k < 10; k++) {
    				if (s[j][k] != op[i][9 - j][9 - k]) {
    					f = 1;break;
    				}
    			}
    			if (f) break;
    		}
    		if (!f) {
    			cout << i + 1 << endl;
    			break;
    		}
    		f = 0;
    		for (int j = 0; j < 10; j++) {
    			for (int k = 0; k < 10; k++) {
    				if (s[j][k] != op[i][9 - k][j]) {
    					f = 1;break;
    				}
    			}
    			if (f) break;
    		}
    		if (!f) {
    			cout << i + 1 << endl;
    			break;
    		}
    	}
    }
    return 0;
}
