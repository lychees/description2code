#include<bits/stdc++.h>
using namespace std;
long long _pow(int a, int b,int mod)
{
    if(b == 0)
        return 1ll;
    long long ret = _pow(a, b/2,mod);
    ret = (ret*ret) % mod;
 if(b%2 == 1)
        ret = (ret*a) % mod;
return ret;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int a,b,k;
        scanf("%d%d%d",&a,&b,&k);
        int mod=pow(10,k);
        //cout<<mod;
        long long ans=_pow(a,b,mod);
        int sum=0;
        while(ans>0)
        {
            sum=sum+ans%10;
            ans/=10;
        }
        printf("%d\n",sum);
    }
    return 0;
}
