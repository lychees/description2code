#include <bits/stdc++.h>
using namespace std;

bool match(int n)
{
	bool dp[n+1];
	dp[1]=true;
	dp[2]=false;
	dp[3]=true;
	dp[4]=true;
	if(n<5)
		return dp[n];
	for(int i=5;i<=n;i++)
	{
		if(dp[i-1] && dp[i-4])
			dp[i]=false;
		else
			dp[i]=true;
	}
	return dp[n];
}

int main()
{
   	int t,n;
   	cin>>t;
   	while(t--)
   	{
   		cin>>n;
   		if(match(n))
   			cout<<"Yes";
   		else
   			cout<<"No";
   		cout<<endl;
   	}
    return 0;
}
