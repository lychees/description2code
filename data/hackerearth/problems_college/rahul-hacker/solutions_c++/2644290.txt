#include <iostream>
using namespace std;
int reverse(int num)
{
	int sum=0;
	while(num)
	{
		sum*=10;
		sum+=(num%10);
		num/=10;
	}
	return sum;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int a,arev,sum=0;
    	scanf("%d%d",&a,&arev);
    //	printf("%d %d\n",a,reva);
    	a=reverse(a);
    	arev=reverse(arev);
    	sum=a+arev;
    	sum=reverse(sum);
    	
    	printf("%d\n",sum);
    }
    return 0;
}
