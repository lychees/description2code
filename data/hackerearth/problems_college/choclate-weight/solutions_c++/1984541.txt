#include<iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    int A[t];
    for(int i=0;i<t;i++)
        cin>>A[i];
    sort(A,A+t);
    int n;
    cin>>n;
    while(n--)
    {
        bool status=false;
        int check;
        cin>>check;
        for(int i=0;i<t;i++)
            if(A[i]==check)
                status=true;
        if(status==false)
        {
            int l, r;

            l = 0;
            r = t-1;
            while(l < r)
            {
                 if(A[l] + A[r] == check){
                      status=true;break;}
                 else if(A[l] + A[r] < check)
                      l++;
                 else // A[i] + A[j] > sum
                      r--;
            }
        }
        if(status==true)
            cout<<"I Like You"<<endl;
        else
            cout<<"I Hate You"<<endl;
    }
}
