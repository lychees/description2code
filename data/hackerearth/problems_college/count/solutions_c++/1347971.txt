#include <iostream>
using namespace std;

int main()
{
    int tr,test;
    cin>>test;
    for(tr=0;tr<test;tr++){
    	int r,c;
    	cin>>r>>c;
    	int even=0,odd=0;
    	for(int i=1;i<=r;i++){
    		if(i%2!=0){
    			if(c<i){
    				odd+=c;
    			}
    			else{
    				odd+=i;
    			}
    		}
    		else{
    			if(c<i){
    				even+=c;
    			}
    			else{
    				even+=i;
    			}
    		}
    	}
    	cout<<even<<" "<<odd<<endl;
    }
    
    return 0;
}
