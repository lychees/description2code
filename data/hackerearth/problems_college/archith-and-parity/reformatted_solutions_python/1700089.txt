def binary(decimal):
    ans = 0
    while decimal != 0:
        rem = (decimal % 2)
        if rem == 1:
            ans += 1
        decimal /= 2
    return ans


t = input()
while t:
    n = input()
    r = binary(n)
    if r % 2 == 0:
        print "even"
    else:
        print "odd"
    t -= 1
