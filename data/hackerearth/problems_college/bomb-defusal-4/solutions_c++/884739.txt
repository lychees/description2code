#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t,p,x,y,u,v;
	cin >> t;
	while(t--)
	{
		scanf("%d%d%d%d%d",&p,&x,&y,&u,&v);
		int dist = (abs(x-u)+abs(y-v));
		if((dist%2)==1)
			cout << "Terrorists Win !\n";
		else
		{
			int time;
			if((x==u)&&(y==v))
				time = 0;
			else if(abs(x-u)==abs(y-v))
				time = 1;
			else
				time = 2;
			if(time<p)
				cout << "Counter-Terrorists Win !\n";
			else
				cout << "Terrorists Win !\n";
		}
	}
	return 0;
}
