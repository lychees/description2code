#include <bits/stdc++.h>
#include <assert.h>
using namespace std;
typedef long long ll;
typedef long double ld;
#define PB push_back
#define MP make_pair
#define MOD 1000000007LL
#define endl "\n"
ll lim = 1e17;
ll a[1000];
ll g(ll n) {
	ll ans=2;
	while(1) {
		bool ok=false;
		for (ll x=2;x<n;x++) {
			if (n%x!=0) {
				n=x; ok=true;break;
			}
		}
		if(!ok)break;
		ans++;
	}
	return ans;
}

ll f(ll n) {
	if (n==1) return 1;
	if(n==2) return 2;
	ll d=1;
	ll limx=-1;
	for (ll x=1;;x++) {
		d = d*(x/__gcd(d,x));
		if (d>lim) break;
		limx=x;
		a[x] = n/d;
		//printf("%lld %lld %lld\n",x,d,a[x]);
	}
	ll take=0;
	ll ans=0;
	for (ll x=limx;x>=1;x--) {
		ll use = a[x]-take;
		ans += use*g(x+1);
		//printf("%lld: %lld*%lld\n",x,a[x]-take,g(x+1));
		take+=use;
	}
	return ans-3;
}

int main() {
		ll a,b; scanf("%lld %lld",&a,&b);
		ll final = f(b)-f(a-1);
		printf("%lld\n",final);
}
