#include <iostream>
using namespace std;

int main()
{
	int ts;
	long n,m,ti,pd,i,cov,temp;
	cin>>ts;
	
	
	for(i=0;i<ts;i++)
	{
		cin>>n>>m;
	pd=0;
	ti=0;
	cov=1;
	if(n==1){
		pd=m;
	}
	
	while((pd<m)&&(cov<n))
	{ti++;
		temp=cov;
		cov=cov+pd;
		if(pd<m)
		{
			
			pd=min(m,pd+temp);
		}

	}
	
	long x=(n-cov)/pd;
	if(x>=1){
	ti=ti+x;
	cov=cov+(pd*x);

	}	
	if(n-cov>0)
	{
	ti++;
	}
	cout<<ti<<endl;	
	}
  
    return 0;
}
