#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
	int t,i,n,a[107],c,ans;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	
    	sort(a,a+n);
    
        ans=0;
    	c = 1;
    	
    	for(i=1;i<n;i++)
    	 {
    	  	if(a[i]==a[i-1]+1)
    	  	 c++;
    	  	else
    	  	{
    	  		if(c%2==0)
    	  		 ans=ans + c/2;
    	  		else
    	  		 ans=ans +(c+1)/2;
    	  		c=1; 
    	  	}
    	  	
    	 }
    	       if(c%2==0)
    	  		 ans=ans + c/2;
    	  		else
    	  		 ans=ans +(c+1)/2;
    	  		 
    	cout<<ans<<endl;
    	
    }
    return 0;
}
