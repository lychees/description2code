#include<bits/stdc++.h>
using namespace std;
#define ll long long

ll arr[100010];
map<ll,int> m;

int main(){
    
    int t;
    cin>>t;
    while(t--){
    	m.clear();
    	int n;
    	scanf("%d",&n);
    	for(int i=1;i<=n;i++){
    		scanf("%lld",&arr[i]);
    	}
    	ll min_indx = 1;
    	ll max_ans = 1;
    	ll curr = 1;
    	ll curr_indx = 1;
    	m[arr[1]] = 1;
    	for(int i=2;i<=n;i++){
    		if(m[arr[i]] != 0 && m[arr[i]] >= curr_indx){
                curr = i-m[arr[i]];
                curr_indx = m[arr[i]]+1;
    		}
    		else {
                curr += 1;
    		}
    		if(curr > max_ans){
                max_ans = curr;
                min_indx = curr_indx;
    		}
    		m[arr[i]] = i;
    	}
    	//if(curr)
    	cout<<min_indx<<" "<<min_indx+max_ans-1<<endl;
    }
    return 0;
}
