#include <iostream>
using namespace std;
#include<stdio.h>

long long int fib(int n)
{

long long int f[n+1];
long long int i;


f[0] = 1;
f[1] = 1;

for (i = 2; i <= n; i++)
{
	f[i] = f[i-1] + f[i-2];
}

return f[n];
}

int main ()
{
long long int n,m;
cin>>m;
while(m--){
cin>>n;
printf("%lld\n", fib(n));
}
return 0;
}
