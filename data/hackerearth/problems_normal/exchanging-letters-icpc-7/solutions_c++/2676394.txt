ll mul(ll a,ll b){
    return (a*b)%hell;
}

template <class T,class U,class V> T fastexp(T a,U b,V mod){
    //Assumes b>=0
    a%=mod;
    T res = 1;
    while(b){
        if(b&1)
            res= (res * a)%mod;
        a= (a*a)%mod;
        b>>=1;
    }
    return res;
}

ll C(int n,int r){
    ll i;
    assert(n>=r);
    vector<ll> fac(n+1);
    fac[0]=1;
    rep(i,1,n+1){
        fac[i]=(fac[i-1]*i)%hell;
    }
    ll ans=fac[n];
    ans=mul(ans,fastexp<ll,ll,ll>(fac[r],hell-2,hell));
    ans=mul(ans,fastexp<ll,ll,ll>(fac[n-r],hell-2,hell));
    return ans;
}
