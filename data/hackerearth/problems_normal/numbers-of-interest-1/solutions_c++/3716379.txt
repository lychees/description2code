#include <iostream>
using namespace std;
int main()
{
   long long int t,x,y,n,i,j,arr[50],temp;
   cin>>t;
	while(t--)
	{
		cin>>x>>y>>n;
		for(i=0;i<x;i++)
		{
			arr[i]=y;
		}
		for(i=x;i<n;i++)
		{
			temp=0;
			for(j=0;j<x;j++)
			{
			temp=temp+arr[i-j-1];
			}
			arr[i]=temp;
		}
		cout<<arr[n-1]<<endl;
		
	}
    return 0;
}
