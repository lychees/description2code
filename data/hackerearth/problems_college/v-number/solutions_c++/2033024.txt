#include <iostream>
#include <stdbool.h>
#include <math.h>
#include <string>
#include <cstring>
using namespace std;
int main()
{
    int t,length,index,sumLeft,sumRight;
    char *n = new char[10000000];
    cin>>t;
    while(t--)
    {
    	sumLeft = 0;
    	sumRight = 0;
    	scanf("%s",n);
    	length = strlen(n);
    	index = length/2;
    	for(int i = 0; i < index; ++i)
    	{
    		sumLeft += (int)n[i];
    		sumRight += (int)n[length-i-1];
    	}
    	if(sumLeft == sumRight)
    		cout<<"YES\n";
    	else
    		cout<<"NO\n";
    }
    return 0;
}
