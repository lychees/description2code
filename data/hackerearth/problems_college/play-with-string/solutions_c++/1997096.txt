#include<iostream>
#include<cmath>
using namespace std;

int main()
{
    int T;
    cin>>T;
    string s;
    while(T--)
    {
        cin>>s;
        int l=s.length();
        int flag=0;
        for(int i=0,j=l-1;i<l-1;i++,j--)
        {
            int a=abs(int(s[i+1]-s[i]));
            int b=abs(int(s[j]-s[j-1]));
            if(a!=b)
            {
                flag=1;
                break;
            }
        }
        if(flag)
            cout<<"Not Funny\n";
        else cout<<"Funny\n";
    }
    return 0;
}
