using namespace std;

typedef long long LL;
typedef vector<int> VI;
typedef vector<VI> VVI;
typedef pair<int,int> PI;

const int inf=2000000000;
const LL INF=LL(inf)*inf;
const LL mod=1000000007LL;
double eps=1e-8;

#define F first
#define S second
#define mp make_pair
#define pb push_back	
#define sc(x) scanf("%d",&(x))
#define MAXN 100001

int main()
{
	int t,n,i,j,x,a,b,fact,factmax,d,expo,index;
	scanf("%d",&t);
	n=10020;
	int * A=new int[n];
	int root=sqrt(n);
	for(i=2;i<=root;i++)
	{
		if(!A[i])
		{
			for(j=2*i;j<n;j+=i)
				A[j]=1;
		}
	}	
	int primes=0;
	int p[10000];
	for(i=2;i<n;i++)
	{
		if(!A[i])
			p[primes++]=i;
	}
	while(t--)
	{
		sc(a);sc(b);
		int F[b-a+1];
		int N[b-a+1];
		int len=b-a+1;
		for(i=0;i<len;i++)
			F[i]=1,N[i]=a+i;
		factmax=2;
		root=sqrt((double)b+eps);
		for(i=0;p[i]<=root;i++)
		{
			x=p[i];
			if(a%x==0)
				index=0;
			else
				index=(a/x +1)*x-a; 
			d=p[i];     			
			for(j=index;j<len;j+=d)
			{
				x=N[j];
				expo=0;
				while(x%d==0)
				{
					expo++;
					x/=d;
				}
				F[j]*=(expo+1);
				N[j]=x;
			}
		}
		for(i=0;i<len;i++)
		{
			if(N[i]!=1)
				F[i]*=2;
			if(F[i]>factmax)
				factmax=F[i];
		}
		x=1;
		for(i=0;i<len;i++)
			if(F[i]==factmax)
				x++;
		printf("%d\n",x);		
	}
}
