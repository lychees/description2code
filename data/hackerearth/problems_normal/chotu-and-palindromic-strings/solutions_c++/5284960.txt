#include <iostream>
#include <string>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--){
		string s1,s2;
		cin>>s1>>s2;
		string s3 = s1+s2;
		string s4 = s2+s1;
		int l=0;
		int r=s3.length()-1;
		int flag=0;
		while(l<r){
			if(s3[l] != s3[r]){
				flag=1;
				break;
			}
			l++;
			r--;
		}
		if(flag!=1){
			cout<<"YES"<<"\n";
		}
		else{
			l=0;
			r=s4.length()-1;
			flag=0;
			while(l<r){
				if(s4[l] != s4[r]){
				  flag=1;
				  break;
			    }
			   l++;
			   r--;
			}
			if(flag!=1){
				cout<<"YES"<<"\n";
			}
			else{
				cout<<"NO"<<"\n";
			}
		}
	}
    return 0;
}
