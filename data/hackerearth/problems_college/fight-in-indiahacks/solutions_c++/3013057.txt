#include<bits/stdc++.h>
using namespace std;
static bool game[1000001];
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		int n,m1,m2;
		scanf("%d %d %d",&n,&m1,&m2);
		memset(game,false,sizeof(game));
		game[0]=true;
		game[m1]=game[m2]=true;
		for(int i=1;i<=n;i++){
			if(i-m1>=0&&game[i-m1]==false)
			   game[i]=true;
			if(i-m2>=0&&game[i-m2]==false)
			   game[i]=true;
		}
		if(game[n])
		printf("1\n");
		else
		printf("0\n");
	}
	return 0;
}
