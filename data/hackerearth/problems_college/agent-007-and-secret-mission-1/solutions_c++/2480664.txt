#include <iostream>
#include<string>
using namespace std;

int count_odds(int num)
{
	int odds = 0;
	for(int i=1; i<=num; i++)
	{
		if(i%2) ++odds;
	}
	return odds;
}

inline bool is_even(int num)
{
	return num%2?false:true;
}

int count_ones(string str)
{
	int length = str.length();
	int ones = 0;
	for(int i = 0; i<length; i++)
	{
		if(str[i] == '1') ++ones;
	}
	return ones;
}

int main()
{
    //cout << "Hello World!" << endl;
	int T, N;
	string in_str;
	cin >> T;
	while(T--)
	{
			cin.ignore();	
			getline(cin, in_str);
			if(!in_str.empty())
			{
				//cout << "Test: str = " << in_str << endl;
				cin.clear();
				//cin.ignore();
				cin >> N;
				//cout << "Test: N = " << N << endl;
				int odds = count_odds(N);
				//cout << "Test:odds = " << odds << endl;
				bool b_even = is_even(odds);
				//cout << "Test: b_eben = " << b_even << endl;
				int ones = count_ones(in_str);
				if(b_even)
					cout << ones << endl;
				else
					cout << in_str.length() - ones << endl;
			}

	}    
    return 0;
}
