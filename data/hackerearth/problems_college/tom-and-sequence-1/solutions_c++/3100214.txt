#include<iostream>
#include<string.h>
using namespace std;
int limit=1299828;
long long arr[100010]={};
    void simpleSieve(int limit)
{
    bool prime[limit];
    memset(prime, true, sizeof(prime));
    for (int p=2; p*p<limit; p++)
    {
        if (prime[p] == true)
        {
            for (int i=p*2; i<limit; i+=p)
                prime[i] = false;
        }
    }
    int k=0;
    for (int p=2; p<limit; p++)
        if (prime[p] == true)
            arr[k++]=p;
    //cout<<k<<'\n';
}

    int main(){
        int i,k;
        simpleSieve(limit);
        //cout<<arr[100009];
        long long ans[100002]={},r=1000000007;
        ans[0]=0;
        ans[1]=2;
        k=1;
        for(i=2;i<100002;i++){
            ans[i]=(ans[i-1]*arr[k])%r;
            k++;
        }
        int t,n;
        cin>>t;
        while(t--){
            cin>>n;
            cout<<ans[n]<<'\n';
        }
        return 0;
    }
