#include <iostream>
#include <algorithm>
using namespace std;
bool comp(int a,int b)
{
	return (a>b);
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,i;
    	int sum=0;
    	cin>>n;
    	int a[n];
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	sort(a,a+n,comp);
    	for(i=0;i<n;i++)
    	{
    		if(a[i]!=a[i+1])
    		sum++;
    	}
    	cout<<sum<<"\n";
    }
    return 0;
}
