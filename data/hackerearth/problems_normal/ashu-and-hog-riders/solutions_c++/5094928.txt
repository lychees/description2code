#include <bits/stdc++.h>
using namespace std;
int xx[4]={0,1,-1,0};
int yy[4]={1,0,0,-1};
int arr[1001][1001];
bool vis[1001][1001];

bool check(int x,int y,int n)
{
	if(x>=0 && x<n && y>=0 && y<n) return true;
	return false;
}
int val;

void findval(int n,int x,int y)
{
	for(int k=0;k<4;k++)
	{
		if(!check(x+xx[k],y+yy[k],n)) continue;
		if(arr[x+xx[k]][y+yy[k]]==0 && !vis[x+xx[k]][y+yy[k]]) 
		{
			vis[x+xx[k]][y+yy[k]]=1;
			continue;
		}
		else if(arr[x+xx[k]][y+yy[k]]==1 && !vis[x+xx[k]][y+yy[k]])
		{
			val++;
	//		cout<<val<<" "<<x<<" "<<y<<"\n";
			vis[x+xx[k]][y+yy[k]]=1;
			findval(n,x+xx[k],y+yy[k]);
		}
	}
}


int main()
{
	int n;
    cin>>n;
    for(int i=0;i<n;i++)
    {
    	for(int j=0;j<n;j++) cin>>arr[i][j];
    }
    memset(vis,0,sizeof(vis));
    int curr,maxi=0,temp;
    for(int i=0;i<n;i++)
    {
    	for(int j=0;j<n;j++)
    	{
    		if(!vis[i][j] && arr[i][j])
    		{
    			val=0;
    			findval(n,i,j);
    			maxi=max(val,maxi);
    		}
    	}
    }
    cout<<maxi;
    return 0;
}
