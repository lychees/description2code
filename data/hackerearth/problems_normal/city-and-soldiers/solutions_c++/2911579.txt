#include<bits/stdc++.h>
#define MAX 100005
#define sd(N) scanf("%lld",&N)
typedef long long ll;
using namespace std;
ll arr[MAX],siz[MAX],n,mini,maxi;
ll root(ll i)
{
    while(arr[i]!=i)
    {
        arr[i]=arr[arr[i]];
        i=arr[i];
    }
    return i;
}
void init()
{
    ll i;
    for(i=0;i<n;i++)
    {
        arr[i]=i;
        siz[i]=1;
    }
}
void dunion(ll a,ll b)
{
    ll ra,rb;
    ra=root(a);
    rb=root(b);
    if(ra!=rb)
    {
            arr[ra]=rb;
            siz[rb]+=siz[ra];
    }
}
int main()
{
    ll q,i,a,b,ans=0,ra,rb,ch;
    sd(n);
    sd(q);
    init();
    for(i=0;i<q;i++)
    {
        sd(ch);
        if(ch==1)
        {
        sd(a);
        sd(b);
        a--;
        b--;
        dunion(a,b);
        }
        else if(ch==2)
        {
            sd(a);
            a--;
            ra=root(a);
            arr[ra]=a;
            arr[a]=a;
            siz[a]=siz[ra];
        }
        else
        {
            sd(a);
            a--;
            ra=root(a);
            printf("%lld\n",ra+1);
        }
    }
    return 0;
}
