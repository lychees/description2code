#include<bits/stdc++.h>
using namespace std;
#define testcases int t;scanf("%d",&t);while(t--)
typedef long long int lld;
int main(){
	lld n,a[1000000],i,j,count=0,flag=1;
	cin>>n;
	for(i=1;i<=n;i++)	cin>>a[i];
	sort(a,a+n+1);
	for(i=1,j=1;;i++,j++){
		if(i*i<a[j]){
			count+=i*i;
			--j;
		}
		else{
			count=count+i*i-a[j];
			++flag;
		}
		if(flag==n+1)
			break;
	}
	cout<<count;
return 0;
}
