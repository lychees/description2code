#include<bits/stdc++.h>

using namespace std;

int main()
{
        int tc;
        cin >> tc;
        while(tc--)
        {
                unsigned long long int a, b, c, x;
                cin >> a >> b >> c;
                x = c * c;
                a += c;
                b += c;
                a *= b;
                cout << x << " / " << a << endl;
        }
        return 0;
}
