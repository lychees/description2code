#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin>>n;
    int ar[n];
    for(int i=0;i<n;i++)
    {
        scanf("%d",&ar[i]);
    }
    int q;
    cin>>q;
    int f;
    int temp=1;
    while(q--)
    {
        cin>>f;
        temp*=f;
        if(temp>1000000)
            break;
    }
    for(int i=0;i<n;i++)
        cout<<ar[i]/temp<<" ";


}
