#include<bits/stdc++.h>
using namespace std;
pair <int,int> Arr[10141];
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n;
        scanf("%d",&n);
        int i;
        for(i=0;i<n;i++)
        {
            int temp;
            scanf("%d",&temp);
            Arr[i]=make_pair(temp,i);
        }
        sort(Arr,Arr+n);
        int Aks=INT_MIN;
        for(i=0;i<n;i++)
        {
            int disp=Arr[i].second-i;
            if(disp<0) disp*=-1;
            if(disp>Aks)
                Aks=disp;
        }
        printf("%d\n",Aks);
    }
    return 0;
}
