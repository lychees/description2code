#include <iostream>
using namespace std;

int main()
{
    char s[500005],r[500005];
    int i,t;
    cin>>t;
    while(t--)
    {
    	int a[130]={0},c=0;
    	cin>>s>>r;
    	i=0;
    	while(s[i])a[s[i++]]+=1;
    	i=0;
    	while(r[i])
    	{
    		if(a[r[i]]!=0)
    		{
    			c+=1;
    			a[r[i]]-=1;
    		}
    		i++;
    	}
    	if(c>=2)cout<<"Yes\n";
    	else
    	cout<<"No\n";
    }
    return 0;
}
