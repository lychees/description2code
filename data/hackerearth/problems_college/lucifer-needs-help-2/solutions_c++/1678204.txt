#include <bits/stdc++.h>
using namespace std;

typedef	long long ll;
typedef vector<int> vi;
typedef vector<ll> vll;
typedef pair<int, int> pii;
#define pb push_back
#define mp make_pair
#define rep(i,n) for(int i = 0; i < n; i++)
#define sc(n) scanf("%d",&n)
#define scll(n) scanf("%lld",&n)
#define mod 1000000007
#define ll long long
template<class T>inline T gcd(T a,T b){return b?gcd(b,a%b):a;}
ll arr[1000005];
ll b[1000005];
int main()
{
	int n;
	sc(n);
	for(int i = 0;i<n;i++){
		scll(arr[i]);
	}
	memset(b,0,sizeof b);
	int k;
	sc(k);
	while(k--){
		int l,r;
		ll x;
		sc(l);
		sc(r);
		l--;
		scll(x);
		b[l]+=x;
		b[r]-=x;
	}
	ll cur = 0;
	for(int i = 0;i<n;i++){
		cur+=b[i];
		arr[i]+=cur;
		printf("%lld\n",arr[i]);
	}
}
