#include<bits/stdc++.h>

using namespace std;

int findmax(int arr[],int &s,int &e,int &l)
{
    int m=0,index;

    for(int i=s;i<=e;i++)
    {
        if(arr[i]>m)
        {
            m=arr[i];
            index=i;
        }
    }

    if(index==s)
    {
        s++;
        l--;
        return 0;
    }

    else
    {
        int cp=0,n=0;

        for(int i=s;i<=index;i++)
        {
            cp+=arr[i];
            n++;
        }
        s=index+1;
        l=e-index;

        return ((n*arr[index])-cp);
    }


}


int main()
{
    int arr[50000],t,s,e,l;
    cin>>t;
    while(t--)
    {
        cin>>l;

        for(int i=0;i<l;i++)
            cin>>arr[i];

        s=0;
        e=l-1;
        long int p=0;

        while(l!=0)
            p+=findmax(arr,s,e,l);



        cout<<p<<endl;

    }


    return 0;
}
