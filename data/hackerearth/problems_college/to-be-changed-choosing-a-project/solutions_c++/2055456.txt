#include <bits/stdc++.h>
using namespace std;

int main() {
	long long int n,r,ind,r1;
	cin>>n>>r;
	int sz=n;
	vector<int> vt;
	for(int i=1;i<=n;i++) vt.push_back(i);
	r1=r%sz;
	ind = 0;
	ind = (ind+r1-1+sz)%sz;
	vt.erase(vt.begin()+ind);
	sz--;
	
	while(sz>1){
		r1=r%sz;
		ind = (ind+r1-1+sz)%sz;
		vt.erase(vt.begin()+ind);
		sz--;
	}
	cout<<vt[0]<<endl;
	
	
	return 0;
}
