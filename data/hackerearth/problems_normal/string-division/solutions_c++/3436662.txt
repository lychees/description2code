#include <bits/stdc++.h>
using namespace std;
set<string> :: iterator it;
int ans(string str,int ind,int count1,set<string> ss){
//	cout<<"count=="<<count1<<endl;
	if(count1==4)
     return 1;
  //   cout<<"index="<<ind<<endl;
     
     for(int i=ind;i<str.size();i++)
       {
       	   int check;
       	    string s;
    //   	    cout<<"i->"<<i<<endl;
       	    s=str.substr(ind,i-ind+1);
      // 	    cout<<s<<endl;
       	    it=ss.find(s);
       	    if(it==ss.end()){
       	//    	cout<<" not exists"<<endl;
       	    	ss.insert(s);
       	    check=ans(str,i+1,count1+1,ss);
       	    
       	    if(check==1)
       	    return 1;
       	    else
       	    ss.erase(s);
       	    }
       	    else {
       	    	//it=ss.begin() + 4;
       	    //	cout<<"already exists"<<endl;
       	    	//ss.erase(s);
       	    	continue;
       	    }
       	  //  cout<<"check=="<<check<<endl;
       }
      // cout<<endl<<endl;
       return 0;
	
}

int main()
{
 int n;
 cin>>n;
 while(n--){
     //char ch[1001];
     string str;
 	cin>>str;
 	if(str.length()>=10)
 	{cout<<"YES"<<endl;
 	continue;
 	}
 	
 	set<string> ss;
 	
 	if(ans(str,0,0,ss)==1)
 	cout<<"YES"<<endl;
 	else
 	cout<<"NO"<<endl;
 }
    return 0;
}
