#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	int t;
	cin>>t;
	while(t--)
	{
		int arr[26]={0};
		string s;
		cin>>s;
		int l=s.length();
		for(int i=0;i<l;i++)
		{
			arr[s[i]-'a']++;
		}
		int flag=arr[0];
		for(int i=0;i<26;i++)
		{
			if(arr[i]!=flag)
			{
				flag=0;
				break;
			}
		}
		if(flag)
	cout<<"Yes\n";
	else cout<<"No\n";
	}
	
	return 0;
}
