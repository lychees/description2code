#include <bits/stdc++.h>
using namespace std;
long long int dp[205][205];
void solve(int n,int k){
	dp[0][0]=1;
	for(int i=1;i<=n;i++){
		dp[i][0]=1;
	}
	for(int j=1;j<=n;j++){
		dp[0][j]=0;
	}
	for(int i=1;i<k;i++){
		for(int j=1;j<=n;j++){
			dp[i][j]=dp[i-1][j];
		}
	}
	for(int i=k;i<=n;i++){
		for(int j=1;j<=n;j++){
			if(j<i){
				dp[i][j]=dp[i-1][j];
			}
			else{
				dp[i][j]=dp[i][j-i]+dp[i-1][j];
			}
		}
	}
	cout<<dp[n][n]<<endl;
}
int main() {
	int t;
	cin>>t;
	while(t--){
		int n,k;
		cin>>n>>k;
		solve(n,k);
	}
	return 0;
}
