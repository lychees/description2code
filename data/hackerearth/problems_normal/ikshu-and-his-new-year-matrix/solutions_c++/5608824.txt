
#include <iostream>
#include<string.h>
#include<set>
using namespace std;
bool b[100008];
int xa[10008];
set<int>mset;
int arr[1000][1000];
void sieve(){
	memset(b, true, sizeof(b));
	b[0]=false;
	b[1]=false;
	for(int i=2;i*i<=100008;i++){
		if(b[i])for(int j=i*i;j<=100008;j+=i)b[j]=false;

	}
	for(int i=2;i<10008;i++)if(b[i])mset.insert(i);
}

int main()
{
   int n,k;
   cin>>n>>k;
   int brr[n][n];
   sieve();
   for(int i=0;i<10008;i++)xa[i]=123456;
   for(auto it =mset.begin();it!=mset.end();it++){
   	int a=*it;
   //	cout<<"PRIME"<<a<<endl;
   	xa[a]=0;
   	while(a>10008)a-=k;
   	while(a>0){
   		xa[a]=min(xa[a+k]+1,xa[a]);
   		a-=k;
   	}
   }
   //cout<<xa[149]<<"adsda\n";
  //for(int i=0;i<100;i++)cout<<i<<" "<<x[i]<<endl;
   for(int i=0;i<n;i++){
   	for(int j=0;j<n;j++){
		int ele;
		cin>>ele;
		brr[i][j]=ele;
   		arr[i][j]=xa[ele];
   	}
   }
   int ans=123456,x,y,po=0;
  /* for(int i=8;i<n;i++){
   	for(int j=0;j<n;j++){cout<<brr[i][j]<<" ";}cout<<endl;}*/
   for(int j=1;j<n-1;j++){
   	for(int i=1;i<n-1;i++){
   		int sum=arr[j-1][i-1]+arr[j-1][i+1]+arr[j+1][i-1]+arr[j+1][i+1]+arr[j][i];
   		//cout<<sum<<endl;
   		if(sum<123456){
   			po=1;
   			if(sum<ans){ans=sum;x=j+1;y=i+1;}
   		}
   	}
   }
   if(po){
   	cout<<"yes\n";
   	cout<<ans<<endl;
   	cout<<x<<" "<<y<<endl;
   /*	int i=12-1,j=23-1;
   	cout<<"sdf\n";
   	cout<<arr[i][j]<<"  d "<<brr[i][j]<<endl;
   	cout<<arr[i+1][j+1]<<" i+1 j+1 "<<brr[i+1][j+1]<<endl;
   	cout<<arr[i+1][j-1]<<" j-1 "<<brr[i+1][j-1]<<endl;
   	cout<<arr[i-1][j+1]<<"i-1 j+1 "<<brr[i-1][j+1]<<endl;
   	cout<<arr[i-1][j-1]<<"j-1 "<<brr[i-1][j-1]<<endl;*/

   }
   else cout<<"no\n";

    return 0;
}
