inline ll mod(ll a,ll MOD){
	
	if(a>=MOD)
	a%=MOD;
    while(a<0)
	a+=MOD;
	
	return a;
}

ll exp(ll a,ll b,ll MOD){
    
    if(b==0) return 1;
    ll ans=1;
    if(b&1)
        ans=a;
    ll temp=exp(a,b/2,MOD);
    ans=(ans*temp)%MOD;
    ans=(ans*temp)%MOD;
    return ans;
}

ll modularInverse(ll a,ll m){
    
    return exp(a,m-2,m);
}
