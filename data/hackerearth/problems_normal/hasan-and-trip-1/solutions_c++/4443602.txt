#include<iostream>
#include<cmath>
#include<iomanip>
#include<climits>
using namespace std;
long double distance(long long int x1,long long int x2,long long int y1,long long int y2){
	long double d,d1,d2,di;
	d1=(x2-x1)*(x2-x1);
	d2=(y2-y1)*(y2-y1);
	d=d1+d2;
	di=sqrt(d);
return di;
}
int main(){
	long long int n,i,j;
	cin>>n;
	long long int x[n+1],y[n+1],f[n+1];
	long double max[n+1],max1;
	max[0]=0.0;
	x[0]=0;
	y[0]=0;
	f[0]=0;
	for(i=1;i<=n;i++){
		cin>>x[i];
		cin>>y[i];
		cin>>f[i];
	}
	max[1]=f[1];
	for(i=2;i<=n;i++){
		max1=INT_MIN;
		for(j=1;j<i;j++){
			if(f[i]-distance(x[i],x[j],y[i],y[j])+max[j]>max1)
				max1=f[i]-distance(x[i],x[j],y[i],y[j])+max[j];
		}
	max[i]=max1;
	}
	cout<<setprecision(6);
	cout<<fixed;
	cout<<max[n];
return 0;
}
