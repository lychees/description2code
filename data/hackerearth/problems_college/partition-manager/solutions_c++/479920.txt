#include<iostream>
#include<stdio.h>
#include<algorithm>
#define INF 30
using namespace std;
void recur_sol(int num,int *arr,int sz)
{
    //if(num<0)return;
    //cout<<num<<"\n";
    if(!num)
    {
        for(int i=0;i<sz;i++)
            printf("%d ",arr[i]);
        printf("\n");
        return;
    }
    int prev=INF;
    if(sz)
     prev=arr[sz-1];
    for(int i=min(prev,num);i;i--)
    {
        if(num>=i)
        {
            arr[sz]=i;
            recur_sol(num-i,arr,sz+1);
        }
    }
}
int main()
{
    int n,arr[300];
    scanf("%d",&n);
    recur_sol(n,arr,0);
}
