#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;

int main()
{
    int t,i,j,flag;
    long long cnt,cnt1;
    string s;
    scanf("%d",&t);
    while(t--)
    {
    	flag=0;
    	cnt=0;
    	cnt1=1;
    	cin>>s;
    	for(i=0;i<s.length();i++)
    	{
    		if(s[i]=='a' || s[i]=='z')
    		{
    			cnt+=(s.length()-i)*cnt1;
    			flag=1;
    			cnt1=1;
    		}
    		else
    		{
    			cnt1++;
    		}
    	}
    	if(flag==1)
    	printf("%lld\n",cnt);
    	else
    	printf("0\n");
    }
    return 0;
}
