unsigned long long int power_orig(unsigned long long base,unsigned long long power)
{
 unsigned long long result=1;
 base=base%m;
 while(power>0){
    if(power%2==1)
    {
        result=(result*base)%m;
    }
    power=power>>1;
    base=(base*base)%m;
 }
 return result%m;
}
