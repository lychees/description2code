#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	unsigned int n, k;
    	cin>>n>>k;
    	unsigned int flag = (1<<n)-1;
    	unsigned int test = k&flag;

    	if(test==flag) cout<<"YES"<<endl;
    	else cout<<"NO"<<endl;
    	
    	
    }
    return 0;
}
