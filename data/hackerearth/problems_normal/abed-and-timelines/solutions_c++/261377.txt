#include<bits/stdc++.h>
using namespace std;
int abs(int a)
{
	if(a<0)
		return -a;
	else
		return a;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long n,m,a,b,c,d,q;
		cin>>n>>m;
		cin>>a>>b>>c>>d>>q;
		long co=0;
		co=min(abs(a-c),abs(n-abs(a-c)))+min(abs(b-d),abs(m-abs(b-d)));
		cout<<co*q<<endl;
		if(co*q>1000)
			cout<<"Let go of the group."<<endl;
		else
			cout<<"You saved the group."<<endl;
	}
	return 0;
}
