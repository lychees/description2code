#include<cstdio>
using namespace std;
int main(){
	int n,i,sum=0,max;
	std::scanf("%d",&n);
	int a[n];
	for(i=0;i<n;i++){
		std::scanf("%d",&a[i]);
	}
	if(n==3){
		max=a[0];
		if(a[1]>max){
			max=a[1];
			if(a[2]>max){
				max=a[2];
			}
		}
		std::printf("%d\n",max);
		
	}
	else{
	for(i=0;i<n;i++){
		sum=sum+a[i];
	}
	if(sum%3==0){
		sum=sum/3;
		std::printf("%d\n",sum);
	}
	else{
		sum=sum/3;
		std::printf("%d\n",sum+1);
	}	
	}
	
	
}
