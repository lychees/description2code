#include <iostream>
#include<set>

using namespace std;


int main()
{	int t,n;
    for(cin>>t;t>0;--t){
    	set<int> lsi;
    	cin>>n;
    	for(int i=1;i<=n;i++){
    		int k;
    		cin>>k;
    		lsi.insert(k);
    	}
    	set<int>::iterator lt=lsi.end();
    	lt--;
    	set<int>::iterator it=lsi.begin();
    	if(lt!=it)
    	do{
    		cout<<*lt<<" ";
    		cout<<*it<<" ";
    		it++;
    		lt--;
    	}while((*lt)>(*it));
    	if((*it)==(*lt))
    	cout<<*it<<" ";
    	cout<<endl;
    }
    
    return 0;
}
