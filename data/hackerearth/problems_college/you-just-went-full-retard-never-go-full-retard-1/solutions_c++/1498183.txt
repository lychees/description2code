#include <iostream>
using namespace std;

int main()
{
   int t;
   scanf("%d",&t);
   while(t--){
   	long long int n;
   	scanf("%lld",&n);
   	bool ok=false;
   	for(long long int i=2;i*i<=n;i++){
   		if(n%i==0){
   			ok=true;
   			break;
   		}
   	}
   	if(ok||n<2){
   		puts("NO");
   	}
   	else{
   		puts("YES");
   	}
   }
    return 0;
}
