#include <iostream>
#include <stdio.h>

using namespace std;

bool bincon(int n)
{
	while(n)
	{
		int b=n;
		n>>=1;
		int c=n;
		if(b&1 and c&1) return true;
	}
	return false;
}
int main()
{
	int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		while(n)
		{
			if(bincon(n)) 
			{
				n++;
				continue;
			}
			printf("%d\n", n);
			break;
		}
	}	
	return 0;
}
