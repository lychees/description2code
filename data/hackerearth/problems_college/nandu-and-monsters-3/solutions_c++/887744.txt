const long MOD = 1000000007;

int main()
{
    //freopen("input.txt", "r", stdin);
    int n, k, i, j;
    read(n, k);
    string s;
    cin >> s;
    s = "0" + s;
    while(k--)
    {
        read(i, j);
        if(i > j) swap(i, j);
        bool flag;
        if(s[i] == '1' or s[j] == '1')
            flag = false;
        else flag = true;
        while(i != j)
        {
            if(j > i)
            {
                if(j % 2) j = (j-1)/2;
                else j /= 2;
                if(s[j] == '1') flag = false;
            }
            else
            {
                if(i % 2) i = (i-1)/2;
                else i /= 2;
                if(s[i] == '1') flag = false;
            }
        }
        //cout << i << ' ' << j << endl;
        if(i == j and flag == true)
            cout << "Yes" << endl;
        else cout << "No" << endl;
    }
    return 0;
}
