#include <iostream>
using namespace std;

using ull = unsigned long long;
constexpr ull mod = 1000000007;

ull powm(ull x, ull n) {
	ull res = 1;
	while (n) {
		if (n % 2)
			res = (res * x) % mod;
		x = (x * x) % mod;
		n /= 2;
	}
	return res;
}

int main() {
	ios_base::sync_with_stdio(0);
	cin.tie(0);
	
	int n;
    cin >> n;
    unsigned x = 0;
    for (int i = 0; i < n; ++i) {
    	unsigned a;
    	cin >> a;
    	x ^= a;
    }
    if (x) 
    	cout << '0' << '\n';
    else
    	cout << powm(2, n - 1) - 1 << '\n';
    return 0;
}
