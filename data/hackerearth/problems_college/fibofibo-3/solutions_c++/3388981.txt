#include<iostream>
#include<algorithm>

using namespace std;

int C=0,D=0;
int fibo(int n)
{
   if (n == 0)
{
  	C++;
   return 0;
}
if (n == 1)
{
   D++;
   return 1;
}
return fibo(n - 1) + fibo(n - 2);
}

int main(){
	
	int test,c=0,d=0;
	cin>>test;
	for(int i=0 ;i<test ;i++){
		int n;
		cin>>n;
		fibo(n);
		cout<<C-c<<" "<<D-d<<endl;
		c=C;
		d=D;
	}	
}
