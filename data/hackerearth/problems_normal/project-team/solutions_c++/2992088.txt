#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin >> n;
    while(n--){
    	int num;
  		cin >> num;
  		int st[num];
  		for(int i=0; i<num; i++)
  			cin >> st[i];
  		sort(st, st+num);
  		int low = 10001, high = 0;
  		for(int i=0,j=num-1; i<j; i++, j--){
  			int tmp = st[i]+st[j];
  			high = max(high, tmp);
  			low = min(low, tmp);
  		}
  		cout << high-low << endl;
    }
    return 0;
}
