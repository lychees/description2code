// const int INF = 1e9;
const int MAX_N = 500 + 1;
const ll INF = 1LL << 62;

int N, K;
int A[MAX_N];
ll S[MAX_N];
ll F[MAX_N][MAX_N];
inline ll sq(ll x){ return x * x; }

int main(){
  scanf("%d%d", &N, &K);
  for(int i = 1; i <= N; i++)
    scanf("%d", A + i);
  for(int i = 1; i <= N; i++)
    S[i] = A[i] + S[i - 1];
  for(int i = 0; i < MAX_N; i++)
    fill(F[i], F[i] + MAX_N, INF);
  F[0][0] = 0;
  for(int i = 1; i <= N; i++){
    for(int k = 1; k <= K; k++){
      for(int j = 0; j < i; j++){
        ll x = S[i] - S[j];
        ll na = F[j][k - 1] + (K + 1) * x * x - 2 * x * S[j] - 2 * x * x;
        F[i][k] = min(F[i][k], na);
      }
    }
  }
  printf("%lld\n", F[N][K]);
  return 0;
}
