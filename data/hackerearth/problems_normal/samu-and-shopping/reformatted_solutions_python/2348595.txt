T = int(raw_input())
for _ in xrange(T):
    N = int(raw_input())
    A = [[0]*3 for _ in xrange(N)]
    B = [[0]*3 for _ in xrange(N)]
    # print A
    for i in xrange(N):
        A[i] = map(int, raw_input().split())
    B[0] = A[0]
    for i in xrange(1, N):
        B[i][0] = A[i][0] + min(B[i-1][1], B[i-1][2])
        B[i][1] = A[i][1] + min(B[i-1][0], B[i-1][2])
        B[i][2] = A[i][2] + min(B[i-1][0], B[i-1][1])
    # print B
    print min(B[N-1])
