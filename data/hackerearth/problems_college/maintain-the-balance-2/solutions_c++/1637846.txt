template<class T>
inline void readInt( T &n )//fast input function
{
   n=0;
   T ch=getcx();int sign=1;
   while( ch < '0' || ch > '9' ){if(ch=='-')sign=-1; ch=getcx();}

   while(  ch >= '0' && ch <= '9' )
           n = (n<<3)+(n<<1) + ch-'0', ch=getcx();
   n=n*sign;
}

int main() {

    int T, n, f, res;

    readInt(T);
    while(T--) {
        getline(cin, s); n = s.length();
        stack<char> st; res = 0;

        rep(i, n) {
            if(s[i] == '}') {
                if(st.size() == 0) st.push(s[i]);
                else
                if(st.top() == '{') st.pop();
                else st.push(s[i]);
                res++;
            }
            else
            if(s[i] == '{') st.push(s[i]), res++;
        }

        if(res & 1) printf("-1\n");
        else {
            n = st.size();

            vector<char> v(n); res = n;

            while(n > 0) v[n - 1] = st.top(), st.pop(), --n;
            n = res;

            res = 0, f = 1;
            rep(i, n) {
                if(f && v[i] == '}') res++;
                else
                if(!f && v[i] == '{') res++;
                f = 1 - f;
            }

            printf("%d\n", res);
        }
    }




    return 0;
}
