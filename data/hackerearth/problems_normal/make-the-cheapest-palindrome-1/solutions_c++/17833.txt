#include <iostream>
#include<cstring>
#include<stdlib.h>
using namespace std;
int main()
{
    int te,a,b;
        string s;
        cin>>te;
        while(te--)
    {
        int i,j,l,c=0;
        cin>>s;
        l=(int)s.size()-1;
        cin>>a>>b;
        for(i=0,j=l;i<j;i++,j--)
        {
                if(s[i]==s[j] && s[i]=='/')
                 {
                        if(a<b)
                          c+=2*a;
                        else
                         c+=2*b;
                 }
        //      else if (s[i]==s[j] &&(s[i]=='a'||s[i]=='b'))
        //        continue;
                 else if(s[i]!=s[j] && (s[i]=='/'||s[j]=='/')) 
                  {
                        if(s[i]=='/'&& s[j]=='a')
                          c+=a;
                        else if(s[i]=='/'&& s[j]=='b')  
                          c+=b;
                        else if(s[j]=='/'&& s[i]=='a')  
                          c+=a;
                          else if(s[j]=='/'&& s[i]=='b')
                          c+=b;
                  }
              else if(s[i]!=s[j] && s[i]!='/' && s[j]!='/') 
               {
                        c=-1;
                        break;
                   }
                  
        }
        cout<<c<<endl;
    }
    return 0;
}
