#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int dp[100][100];
char str[101];
ll int mycheck(int i,int j)
{
	if(dp[i][j]!=-1)
	return dp[i][j];
	if(i==j)
	{
		dp[i][j]=1;
	}
	else if(j<i)
	{
		dp[i][j]=0;
	}
	else
	{
		dp[i][j]=mycheck(i,j-1)+mycheck(i+1,j)-mycheck(i+1,j-1);
		if(str[i]==str[j])
		{
			dp[i][j]+=1 + mycheck(i+1,j-1);
		}
	}
	return dp[i][j];
}
int main()
{
	ll int test;
	scanf("%lld",&test);
	while(test--)
	{
		scanf("%s",str);
		ll int l=strlen(str);
		ll int i,j;
		//Initialisation
		for(i=0;i<l;i++)
		{
			for(j=0;j<l;j++)
			{
				dp[i][j]=-1;
			}
		}
		mycheck(0,l-1);
		ll int ans=0;
		for(i=0;i<l;i++)
		{
			for(j=i;j<l;j++)
			{
				if(j<i)
				continue;
				else if(i==j)
				ans++;
				else
				{
					ans+=dp[i][j]-dp[i+1][j];
				}
			}
		}
		 //~ for(i=0;i<l;i++)
        //~ {
			//~ for(j=0;j<l;j++)
			//~ {
				//~ cout<<dp[i][j]<<" ";
			//~ }
			//~ cout<<endl;
		//~ }
		cout<<dp[0][l-1]<<endl;
		//~ printf("%lld\n",ans);
	}
	return 0;
}
