#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
ll dp[26][100005];
int main(){
	ll n;
	ll q;
	cin >> n >> q;
	string st;
	cin >> st;
	for(ll i=1;i<=st.size();i++)
	{
		for(ll j=0;j<26;j++)
		if(st[i-1] -'a' == j)
		dp[j][i] = dp[j][i-1]+1;
		else
		dp[j][i] = dp[j][i-1];
	}
	while(q--){
		ll l,r,k ;
		cin >> l >> r >> k ;
		ll a[26] = {0} ;
		// for(ll i = l-1; i < r; i++){
		// 	a[st[i]-97] += 1;
		// }
		for(ll j=0;j<26;j++)
		{
			a[j] = dp[j][r] - dp[j][l-1];
		}
		if(k > (r-l+1))
		cout << "Out of range" << endl ;
		else {
			for(ll i = 0; i < 26; i++){
				if(k <= 0){
					printf("%c\n",'a'+i-1);
					break ;
				}
				else {
					if(k <= a[i]){
						printf("%c\n",'a'+i);
						break ;
					}
					else
					k = k - a[i] ;
				}
			}
		}
	}
}
