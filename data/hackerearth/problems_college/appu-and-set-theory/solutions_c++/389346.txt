void fastLongLongRead(long long *a){
	register char c=0;
	while(c<33){
		c=getchar_unlocked();
	}
	*a=0;
	while(c>33){
		*a=*a*10+(c-'0');
		c=getchar_unlocked();
	}
}

//Code for fast evaluation of a^b%c
long long powmod(long long base,long long power,long long mod){
	long long result=1;
	base%=mod;
	while(power>0){
		if(power & 1){
			result=(result*base)%mod;
		}
		base=(base*base)%mod;
		power>>=1;
	}
	return result;
}

	int N;

	forall( i,0,N){

		results=((powmod(2,Data[i],MOD)*((1+results))%MOD)%MOD+results)%MOD;
