#include<iostream>
#include<cstdio>
#include<cstring>
#define gc getchar
using namespace std;
int readInt()
{
	int ret=0;
	char c;
	do{
		c=gc();
	}while(c<'0'||c>'9');
	do
	{
		ret=ret*10+c-'0';
		c=gc();
	}while(c>='0'&&c<='9');
	return ret;
}
int main()
{
int i,cnt=0,j,n;
  char str[100][100];
  char temp[100];
  n=readInt();
  for(i=0;i<n;i++)
      gets(str[i]);
  for(i=0;i<n;i++)
      for(j=i+1;j<=n;j++){
           if(strcmp(str[i],str[j])>0){
               strcpy(temp,str[i]);
              strcpy(str[i],str[j]);
              strcpy(str[j],temp);
           }
      }
  for(i=0;i<=n;i++)
  {
  
     if(strcmp(str[i],str[i+1])==0)
     cnt++;
     else
     {
     if(strlen(str[i]))
     cout<<str[i]<<" "<<cnt+1<<endl;
     cnt=0;
    }
}

	return 0;
}
