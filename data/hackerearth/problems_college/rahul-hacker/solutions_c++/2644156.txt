#include <iostream>
using namespace std;

int removeZero(int x);
int reverseNumber(int number);

int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	int x,y;
    	cin>>x>>y;
    	x = removeZero(x);
    	y = removeZero(y);
    	int sum = reverseNumber(x)+reverseNumber(y);
    	sum = removeZero(sum);
    	cout<<reverseNumber(sum)<<"\n";
    }
    return 0;
}

int removeZero(int x)
{
	while(x%10 == 0)
	{
		x = x/10;
	}
	return x;
}

int reverseNumber(int number)
{
	int temp = 0;
	while(number != 0)
	{
		
		temp = temp*10+ (number%10);
		number = number/10;
	}
	return temp;
}
