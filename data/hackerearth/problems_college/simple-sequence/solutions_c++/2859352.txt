#include <bits/stdc++.h>

using namespace std;

#define icin(x) scanf("%d",&x)
#define lcin(x) scanf("%lld",&x)

typedef long long LL;
typedef vector<int> vi;
typedef vector<vi> vvi; 
vi v;
vi tmp;

int LIS(vi& v) {
  vector<pair<int,int> > best;
  vi dad(v.size(), -1);  
  for (int i = 0; i < v.size(); i++) {
    pair<int,int> item = make_pair(v[i], i);
    vector<pair<int,int> >::iterator it = upper_bound(best.begin(), best.end(), item);
    if (it == best.end()) {
      dad[i] = (best.size() == 0 ? -1 : best.back().second);
      best.push_back(item);
    } else {
      dad[i] = dad[it->second];
      *it = item;
    }
  }
  return best.size();
}

int main(){
    int n;
    icin(n);
    v.resize(n);
    for(int i=0;i<n;i++){
      icin(v[i]);
      tmp.push_back(i-n);
    }
    tmp.push_back(0);
    for(int i=0;i<n;i++){
      tmp.push_back(v[i]);
    }
    for(int i=0;i<tmp.size();i++)
      tmp[i]=tmp[i]-i;
    cout<<(tmp.size()-(LIS(tmp)))<<endl; 
  return 0;
}
