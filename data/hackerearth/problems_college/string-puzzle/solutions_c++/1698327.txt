#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

void permute(char *, int, int);

void swap (char *x, char *y)
{
    char temp;
    temp = *x;
    *x = *y;
    *y = temp;
}

int main()
{
    char s[100000];
    cin >> s;
    int n, a, b;
    cin >> n;
    
    while(n--) {
        cin >> a >> b;
        
        
        permute(s, a, b);
        
        cout << endl;
    }
    
    return 0;
}

void permute(char *a, int i, int n)
{
    int j;
    if (i == n)
        cout << a << endl;
    
    else
    {
        for (j = i; j <= n; j++)
        {
            swap((a + i), (a + j));
            permute(a, i + 1, n);
            swap((a + i), (a + j));
        }
    }
}
