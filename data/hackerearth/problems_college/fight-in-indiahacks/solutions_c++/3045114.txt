#include <bits/stdc++.h>
#define ll long long

using namespace std;

int main() {
    ll n, m1, m2, t;

    cin >> t;

    while(t--) {
        cin >> n >> m1 >> m2;
        ll dp[n + 1];
        dp[0]= 0;
        //memset(dp, 0, (n + 1) * sizeof(ll));
        for (int i = 1; i <= n; i++) {
        	if (i >= m1 && !dp[i - m1]) dp[i] = 1;
        	else if(i >= m2 && !dp[i - m2]) dp[i] = 1;
        	else dp[i] = 0;
        }

        if(dp[n]) cout<<"1\n";
		else cout<<"0\n";
    }

    return 0;
}
