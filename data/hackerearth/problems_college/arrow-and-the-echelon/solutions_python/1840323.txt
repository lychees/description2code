#!/usr/bin/python
# -*- encoding: utf-8 -*-
# pylint: disable=invalid-name,missing-docstring,bad-builtin
from sys import stdin
from itertools import imap

def main():

    dstream = imap(int, stdin.read().split())
    for _ in xrange(next(dstream)):
        n = next(dstream)
        n5 = (n // 3)*3
        n %= 3
        if n == 0:
            print '5'*n5
        elif n == 1:
            if n5 >= 9:
                print '5'*(n5 - 9) + '3'*10
            elif (n5 + 1) % 5 == 0:
                print '3'*(n5 + 1)
            else:
                print -1
        else:
            if n5 >= 3:
                print '5'*(n5 - 3) + '3'*5
            elif (n5 + 2) % 5 == 0:
                print '3'*(n5 + 2)
            else:
                print -1


main()
