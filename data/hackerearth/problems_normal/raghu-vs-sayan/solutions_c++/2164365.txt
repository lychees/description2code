#include <bits/stdc++.h>
using namespace std;
 
int main()
{
    int t;
    cin>>t;
    for(int tt=0 ;tt<t;tt++){
    	long long int a ,b ,n, suma=0,sumb=0, counta=0 , countb=0;
    	cin>>a>>b>>n;
    	long long int arr[n];
    for(int i=0;i<n;i++){
    	cin>>arr[i];
    }
    sort(arr,arr+n);
    	for(int i =0;i<n;i++){
    		suma = suma+arr[i];
    		if(suma<=a){    			
    			counta++;
    		} 
    		sumb = sumb+arr[i];
    		if(sumb<=b) {
    			countb++;
    		} 
    	}
    	if(counta>countb) cout<<"Raghu Won"<<"\n";
    	else if(counta<countb) cout<<"Sayan Won"<<"\n";
    	else cout<<"Tie"<<"\n";
    }
    return 0;
}
