#include <bits/stdc++.h>

using namespace std;

int main() {
  int t;
  string a, b;
  cin >> t;
  while (t --> 0) {
    cin >> a >> b;
    int aa = 0, bb = 0;
    for (int i = 0; i < (int)a.length(); ++i) {
      aa = aa * 2 + a[i] - '0';
    }
    for (int i = 0; i < (int)b.length(); ++i) {
      bb = bb * 2 + b[i] - '0';
    }
    if (bb * log(aa * 1.0) > aa * log(bb * 1.0)) {
      cout << "The Mountain\n";
    } else {
      cout << "The Hound\n";
    }
  }
  return 0;
}
