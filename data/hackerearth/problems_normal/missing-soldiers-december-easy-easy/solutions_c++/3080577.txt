#include<bits/stdc++.h>
using namespace std;

class s
{
public:
    long int l, u;
}obj[1000005];

long int temp, ans;

bool mysort(s const &a, s const &b)
{
    return a.l < b.l;
}

int main()
{
    ans = 0;
    int n;
    cin>>n;
    for(int i=0; i<n; i++)
    {
        scanf("%ld %ld %ld", &obj[i].l, &temp, &obj[i].u);
        obj[i].u = obj[i].l + obj[i].u;
    }

    sort(&obj[0], &obj[n], mysort);
    long int ll = obj[0].l;
    long int uu = obj[0].u;
    ans = uu-ll+1;
    for(int i=1; i<n; i++)
    {
        if(obj[i].l>=ll && obj[i].l<=uu)
            obj[i].l = uu+1;
        if(obj[i].u>=ll && obj[i].u<=uu)
            obj[i].u = ll-1;
        if(obj[i].l>obj[i].u)
            continue;
        if(obj[i].l>uu)
            uu=obj[i].l;
        if(obj[i].u>uu)
            uu=obj[i].u;
        ans += obj[i].u-obj[i].l+1;
        //cout<<obj[i].u<<" "<<obj[i].l<<" "<<uu<<endl;
    }
    cout<<ans<<endl;
    return 0;
}
