t = input()
for i in range(t):
    n = input()
    d = {}
    s = map(int, raw_input().strip().split())
    for j in s:
        d[j] = d.get(j, 0)+1
    l = 0
    for key in d.keys():
        if d[key] > l:
            l = d[key]
    lst = []
    for j in d.keys():
        if d[j] == l:
            lst.append(j)
    lst.sort()
    lst.reverse()
    print " ".join(map(str, lst))
