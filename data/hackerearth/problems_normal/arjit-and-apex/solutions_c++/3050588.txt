#include <bits/stdc++.h>
using namespace std;
int main() {
	// your code goes here
	
	int t,m,n,i,a,b,f,great,good,r,e;
	cin>>t;
	while(t--){
		multimap<int,int> mymap;
		multimap<int,int>::iterator it;
		pair<multimap<int,int>::iterator,multimap<int,int>::iterator> ret;
		cin>>m>>n;
		great = 0;
		good = 0;
		int x[m],y[m];
		for(i=0;i<m;i++){
			cin>>x[i]>>y[i];
		}
		for(i=0;i<n;i++){
			cin>>a>>b;
			mymap.insert(make_pair(a,b));	
		}
		cin>>e>>r;
		for(i=0;i<m;i++){
			it = mymap.find(x[i]);
			if(it != mymap.end()){
				f = 0;
				ret = mymap.equal_range(x[i]);
				for(it = ret.first ; it != ret.second ; it++){
					if(it->second == y[i]){
						f = 1;
						break;
					}
				}
				if(f == 1){
					great++;
				}
				else 
				good++;
			}
		}
		if(good == e)
		cout<<"Good\n";
		else if(great == r)
		cout<<"Great\n";
		else
		cout<<":(\n";
	}
	return 0;
}
