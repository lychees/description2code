#include <iostream>
#include <bits/stdc++.h>
using namespace std;


bool visited[1000005];
int op[4];
int main() {
	// your code goes here
	int t;
	scanf("%d", &t);
	int src,dest;

	while ( t--) {
		memset( visited, false, sizeof( visited));
		scanf("%d %d", &src, &dest);
		bool flag = false;
		long long ans = -1;
		for ( int i = 0 ;i  < 4;i++)
			scanf("%d", op + i);
		visited[src] = true;
		
		queue<pair<long long , long long> > q;
		q.push( make_pair(src,0));
		while (q.size() > 0) {
			pair<int, int> p = q.front();
			q.pop();
			long long val;
			for ( int i = 0; i < 4; i++) {
				if ( i == 0)
				val = p.first * op[i];
				else if ( i == 1)
				 val = p.first / op[i];
				else if ( i == 2)
				 val = p.first + op[i];
				else 
				 val = p.first - op[i];
				
				if ( val <  2 || val > 1000000 )
				continue;
				if ( visited[val])
					continue;
				if ( val == dest) {
					flag = true;
					ans = p.second + 1;
					break;
				} else {
				 {
						visited[val] = true;
						q.push( make_pair(val, p.second + 1));
					}
				}
			}
			if ( flag)
				break;
			
		}
		printf("%lld\n", ans);
		
	}
	return 0;
}
