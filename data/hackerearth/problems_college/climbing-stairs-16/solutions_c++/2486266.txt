#include <iostream>
#include <stdio.h>
#include <cstring>
using namespace std;

int main()
{
    int T;
    scanf("%d", &T);
    long long dp[1001];
    memset(dp, 0, sizeof(dp));
    dp[0] = 1;
    for(int i=1; i<=1000; i++) {
    	if(i-1 >= 0)
    		dp[i] += dp[i-1];
    	if(i-2 >= 0)
    		dp[i] += dp[i-2];
    }
    while(T--) {
    	int n;
    	scanf("%d", &n);
    	printf("%lld\n", dp[n]);
    }
    return 0;
}
