#include <iostream>
using namespace std;

int main()
{
    int t,index,n;
    t=0;
    cin>>t;
    while(t--){
    	index=0; n=0;
    	cin>>index>>n;
    while(n!=0){
    		if(index>=1 && n!=0)
    		 {
    		   index--;
    		   n--;
    		 }
    		 else if(index==0 && n!=0)
    		   {
    		   index++;
    		   n--;
    		   }
    	
    	}
    	cout<<index<<endl;
    }    
}
