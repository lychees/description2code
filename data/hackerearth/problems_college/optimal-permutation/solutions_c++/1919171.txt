#include<bits/stdc++.h>
using namespace std;
long long int dp[200001];
int main()
{
   	dp[0]=0;
   	dp[1]=0;
   	dp[2]=1;
   	dp[3]=3;
   	int j=4;
   	for(int i=4;i<200001;i+=2)
   	{
   		dp[i]=dp[i-1]+j;
   		dp[i+1]=dp[i]+j;
   		j=j+2;
   	}
   	int t;
   	scanf("%d",&t);
   	while(t--)
   	{
   		int n;
   		scanf("%d",&n);
   		printf("%lld\n",dp[n]);
   	}
    return 0;
}
