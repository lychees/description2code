#include<bits/stdc++.h>
using namespace std;
main()
{
	int n,m, i,j,ct=1,max=0;
	cin>>n>>m;
	char a[n][m+1];
	int d[m];
	for(i=0;i<n;i++)
    scanf("%s",a[i]);
max=a[0][0];
        for(j=0;j<m;j++)
        {
        	max=0;
        	for(i=0;i<n;i++)
        	{
        		if(a[i][j]>=max){
        		max=a[i][j];
        		d[j]=i;
				}
        	}
        }
        for(i=0;i<m-1;i++)
        {
        	if(d[i]!=d[i+1])
        	ct++;
        }
        cout<<ct;
}
