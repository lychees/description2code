#include <iostream>

using namespace std;

int main()
{
    int t,i;
    cin>>t;
    int m,h,m1,h1,d,d2;
    for(i=0;i<t;i++)
    {
       cin>>h;
       cin>>m;
       m1=m*6;
       h1=h*30;
       d2=m/2;
       if(h==12)
       {
           h1=0;
       }
       if(h1>=m1)
       {
           d=h1-m1;
           d=d+d2;

       }
       else
       {
           d=m1-h1;
           if(d>d2)
           {
           		d=d-d2;
           }
           else
           {
           		d=d2-d;
           }
       }
       if(d>180)
       {
           d=360-d;
       }
       cout<<d<<"\n";

    }
    return 0;
}
