ll fr()
{
    char ch;
    ll t=0,flag=0;
    ch=gc();
    while(ch<33)
        ch=gc();
    if(ch=='-')
        {
            ch=gc();
            flag=1;
        }
    while(ch>33)
    {
        t=t*10+ch-'0';
        ch=gc();
    }
    if(flag)
        return -t;
    else
        return t;
}
int main()
{
    ll i,j,x,y,k,t,n,e,to,h,n1,n2,c,d,hd,star,z,ev,ans;
    n=fr();

    for(i=1,ans=0,x=0;i<=n;i++)
    {
        k=fr();
        ans+=(k-1)*i+1;
    }
    printf("%lld\n",ans);
        return 0;
}
