#include<bits/stdc++.h>
using namespace std;

#define ll long long int



   ll n,m;
    ll arr[2000];
    ll dp[2000][20];

    ll solve(ll curr,ll mover)
    {
    //  cout << curr << " " << mover  << "\n";
        if(dp[curr][mover] == -1)
        {
            ll i;
            ll ans = 1e18;

            if(mover == m)
            {
                ans = 0;
                for(i = curr;i < n;i++)
                {
                    ans += arr[i];
                }
            }
            else
            {
                ll sum = 0;
                ll rem = m - mover;
                for(i = curr;i < n - rem;i++)
                {
                //  cout << mover<< " " << i << "er\n";
                    sum += arr[i];
                    ans = min(ans, max(sum, solve(i + 1, mover + 1)));
                }
            }
            dp[curr][mover] = ans;
        }
        return dp[curr][mover];
    }

    int main()
    {
        cin>>m>>n;
        ll i;

        memset(dp , -1, sizeof(dp));

        for(i = 0;i < n;i++)
        {
            cin>>arr[i];
        }

        m = min(m,n);
        cout << solve(0, 1) << "\n";
    return 0;
    }
