#include <iostream>
#include <string>
using namespace std;

int main()
{
    int t;
    cin>>t;
    int n;
    string a,b;
    for(int k=0;k<t;k++)
    {
    	cin>>a>>b>>n;
    	int c=0;
    	for(int i=0;i<a.length();i++)
    	{
    		if(a[i]!=b[i])
    			c++;
    	}
    	int r=0;
    	if(n>c)
    	{
    		r=n-c;
    		if(r%2==0)
    			cout<<"Yes"<<endl;
    		else
    			cout<<"No"<<endl;
    	}
    	else if(n==c)
    		cout<<"Yes"<<endl;
    	else
    		cout<<"No"<<endl;
    }
    return 0;
}
