#include <iostream>
#include <algorithm>
#define LL long long

using namespace std;


LL arr[2000000];
LL sum[2000000];

int main()
{
	for (int i=1; i<2000000; i++) {
		arr[i] = arr[i-1]+i;
		if (i%2) sum[i]=i+sum[i-1];
		else sum[i]=sum[i-1];
	}
	
	int t;
	cin >> t;
	
	while (t--) {
		LL l, r;
		cin >> l >> r;
		
		int j = lower_bound(arr, arr+2000000, l) - arr;
		int k = lower_bound(arr, arr+2000000, r) - arr;
	//	cout << j << k;
		LL ans=0;
		if (j%2) {
			if ((arr[j]-l)%2)++j;
		} else {
			if ((arr[j]-l)%2==0)ans-=j;
		}
		if (k%2) {
			if ((r-arr[k-1])%2==0) --k;
		} else {
			if ((r-arr[k-1])%2) ans+=k;
		}
		cout << ans + sum[k]-sum[j-1] << endl;
	}
	
	
    return 0;
}
