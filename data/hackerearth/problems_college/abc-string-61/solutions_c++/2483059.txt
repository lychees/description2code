#include<bits/stdc++.h>
using namespace std;
const long long mod = 1000000007;
long long bp(long long x,long long y){
	if(!y) return 1;
	if(y%2) return x*bp(x,y-1)%mod;
	else return bp(x*x%mod,y/2);
}
int main(){
	int T;
	scanf("%d",&T);
	while(T--){
		long long n;
		scanf("%lld",&n);
		long long x = bp(3,n);
		x -= 3*bp(2,n) % mod;
		x += 3;
		x %= mod;
		x += mod;
		x %= mod;
		printf("%lld\n", x);
	}
	return 0;
}
