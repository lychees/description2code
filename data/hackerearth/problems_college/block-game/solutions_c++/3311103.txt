#include <bits/stdc++.h>
#include <iostream>

using namespace std;

#define s(x) scanf("%d", &x)
#define sl(x) scanf("%lld", &x)

typedef long long int lld;
int n;
struct build {
    lld h, c;
}a[100005];

bool operator <(struct build x, struct build y)
{
    if (x.h != y.h)
        return x.h < y.h;
    return x.c < y.c;
}

lld f(int m)
{
    int i;
    lld cost = 0;
    for (i = 0; i < n; i++) {
        cost += (a[i].c * ((a[i].h - a[m].h) < 0LL ? (a[m].h - a[i].h) : (a[i].h - a[m].h)));
    }

    return cost;
}

int main()
{

    //freopen("sl.txt", "r", stdin);
    //freopen("so.txt", "w", stdout);
    int t;
    s(t);

    while (t--) {
        int i;
        s(n);

        for (i = 0; i < n; i++)
            sl(a[i].h);
        for (i = 0; i < n; i++)
            sl(a[i].c);

        sort(a, a+n);

        int low = 0, high = n-1, l, r;

        while (high - low + 1 > 3) {
            l = (low * 2 + high) / 3;
            r = (low + high * 2) / 3;
            if (f(l) < f(r))
                high = r;
            else
                low = l;
        }

        lld minc = f(low);
        for (i = low+1; i <= high; i++)
            minc = min(minc, f(i));

        cout << minc << endl;
    }

    return 0;
}
