#include <iostream>
#include <bits/stdc++.h>
using namespace std;
#define N 100001
bool isPrime[N];
void preprocess() {
	memset(isPrime, true, sizeof(isPrime));
	isPrime[0]=false;
	isPrime[1]=false;
	for(int i=4;i<N;i+=2) {
		isPrime[i] = false;
	}
	for(int i=3;i*i<N;i+=2) {
		for(int j=3*i;j<N; j+=i) {
			isPrime[j] = false;
		}
	}
}

int main()
{
	preprocess();
    int t;
    cin>>t;
    while(t--) {
    	long long int n;
    	cin>>n;
    	long long int result = n;
    	
    	for(int i=2;i*i<=n;i++) {
    		if(n%i==0) {
    			while(n%i==0) {
    				n = n/i;
    			}
    			result -= result/i;
    		}
    	}
    	if(n>1) {
    		result -= result/n;
    	}
    	if(isPrime[result]==true) {
    		cout<<"TRUE"<<endl;
    	}else {
    		cout<<"FALSE\n";
    	}
    }
    return 0;
}
