#include<iostream>
using namespace std;

#define ull unsigned long long

int call (ull d)
{
    if(d<10)
        return d;
    else
    {
        int res = 0;
        while(d>0)
        {
            res = res + d%10;
            d =d/10;
        }
        return call(res);
    }
}

int main()
{
    int test;
    cin>>test;
    while(test--)
    {
        int n;
        cin>>n;;
        ull len,d,sum=0;
        while(n--)
        {
            cin>>len>>d;
            sum = sum+len*d;
            sum = call(sum);

        }
        sum = call(sum);
        cout<<sum<<endl;
    }
}
