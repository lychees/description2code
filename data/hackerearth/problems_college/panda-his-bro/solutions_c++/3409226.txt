#include<bits/stdc++.h>

using namespace std;

typedef long long ll;

const int max_n = 1e5 + 10;

vector<ll> v[max_n];
bool vis[max_n];

queue<pair<ll, ll> > q;

pair<ll, ll> bfs(ll node)
{
	ll dis, i, j, siz;

	memset(vis, 0, sizeof(vis));

	q.push(make_pair(node, 0));

	vis[node] = true;

	pair<ll, ll> pr;

	while(!q.empty())
	{
		pr = q.front();

		q.pop();

		node = pr.first;
		dis = pr.second;
		
		siz = v[node].size();

		for(j=0;j<siz;++j)
		{
			if(!vis[v[node][j]])
			{
				vis[v[node][j]] = true;

				pr.first = v[node][j];
				pr.second = dis + 1;

				q.push(pr);
			}
		}
	}

	return (make_pair(node, dis));
}

int main()
{
	ll t, i, j, n, a, b, node;

	scanf("%lld", &n);

	for(i=0;i<n-1;++i)
	{
		scanf("%lld %lld", &a, &b);
		
		v[a].push_back(b);
		v[b].push_back(a);
	}

	node = bfs(a).first;

	printf("%lld\n", bfs(node).second);

	return 0;
}
