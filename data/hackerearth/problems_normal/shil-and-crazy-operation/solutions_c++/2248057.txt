#include<iostream>
#include<vector>
using namespace std;
bool visited[100005];
vector<vector<int> >cycle;
int main()
{
ios_base::sync_with_stdio(false);
int n;
long long int t;
cin>>n>>t;
int a[n],p[n],b[n];
for(int i=1;i<=n;i++)
cin>>a[i];
for(int i=1;i<=n;i++)
cin>>p[i];
for(int i=1;i<=n;i++)
if(!visited[i])
{
int cur=i;
vector<int>v;
while(!visited[cur])
{
visited[cur]=true;
v.push_back(cur);
cur=p[cur];
}

cycle.push_back(v);
}
for(int i=0;i<cycle.size();i++)
{
int shift=t%cycle[i].size();
for(int j=0;j<cycle[i].size();j++)
b[cycle[i][j]]=a[cycle[i][(j+shift)%cycle[i].size()]];
}
for(int i=1;i<=n;i++)
cout<<b[i]<<" ";
cout<<endl;
return 0;
}
