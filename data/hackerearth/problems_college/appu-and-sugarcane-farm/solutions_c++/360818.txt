#include<iostream>
#include<algorithm>
#include<string.h>
using namespace std;

typedef long long ll;
ll a[55];
int main(){


    int n;
    ll max=0;
    int flag =0;
    cin>>n;
    for(int i=0;i<n;i++){
        cin>>a[i];
        max= max>a[i]?max:a[i];
    }
    sort(a,a+n);
    //cout<<max;
    //cout<<a[3];
    for(int i=0;i<n;i++){
         while(a[i]<max){
            a[i]*= 2;
            if(a[i]>max){
                flag = 1;
                break;
            }
        }
        if(flag)
            break;
    }
    if(flag)
        cout<<"NO"<<"\n";
    else
        cout<<"YES"<<"\n";
    return 0;

}
