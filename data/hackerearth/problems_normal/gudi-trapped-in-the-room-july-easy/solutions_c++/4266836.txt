      if(i%2==1)

   }

   if(visited.find(str1)==visited.end())
     dfs(str1);

   string str2=start;

   for(int i=0;i<n;++i){

     str2[(i+H)%n]=start[i];
   }

  // cout<<str2<<"\n";

     dfs(str2);

   return;
}
