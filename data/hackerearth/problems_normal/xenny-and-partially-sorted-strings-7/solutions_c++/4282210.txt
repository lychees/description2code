#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int m;
vector <string> v;
map <string,int> mp;
bool cmp(string a,string b)
{
	int i;
	for(i=0;i<m;i++)
	{
		if(a[i]!=b[i])
		{
			return a[i]<b[i];
		}
	}
	return mp[a]<mp[b];
}
int main()
{
   	int i,k,t,n,l;
   	string s;
   	cin>>t;
   	while(t--)
   	{
   		cin>>n>>k>>m;
   		for(i=0;i<n;i++)
   		{
   			cin>>s;
   			mp[s]=i+1;
   			v.push_back(s);
   		}
   		sort(v.begin(),v.end(),cmp);
   		cout<<v[k-1]<<endl;
   		v.clear();
   		mp.clear();
   	}
    return 0;
}
