#include <iostream>
#include<bits/stdc++.h>
using namespace std;
#define ll long long int

int main()
{
    ll n;
    ll ans=0;
    cin>>n;
    double x[n],y[n];
    double m, temp;
    for(int i=0;i<n;i++) cin>>x[i]>>y[i];
    for(int i=1;i<n;i++)
    {
    	m= INFINITY;
    	for(int j=i-1;j>=0;j--)
    	{
    		temp= ((y[j]-y[i])/(x[j]-x[i]));
    		if(temp<=m)
    		{
    			m=temp;
    			ans++;
    		}
    	}
    }
   cout<<ans;
    
    return 0;
}
