#include <iostream>
using namespace std;

int main()
{
    int t; 
    cin >> t; 
    while(t--){
    	long int count = 0;
    	long long int n, a, b;
    	
    	int array[1000001] = {};
    	cin >> n; 
    	for(long int i=0; i<n; i++){
    		cin >> a >> b; 
    		 array[a]++;
    		 if(array[b]){
    		 	array[b]--;
    		 }else{
    		 
    		 	count++;
    		 }
    	}
    
    	cout << count << endl;
    	
    	
    }
    return 0;
}
