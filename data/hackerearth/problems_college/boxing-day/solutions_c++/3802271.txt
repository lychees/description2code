#include<bits/stdc++.h>
using namespace std;

vector<long long >g[100001];
int v[100001];


void dfs(long i)
{
	if(v[i]==1)return ;
	v[i]=1;
	int l = g[i].size();
	for(long j=0;j<l;j++)
	{
		if(v[g[i][j]]==0)
			dfs(g[i][j]);
	}
}

int main()
{
	for(long i=0;i<100001;i++)v[i]=0;
	long n;
	cin>>n;
	long t;
	cin>>t;
	while(t--)
	{
		long long ax,bx;
		cin>>ax>>bx;
		g[ax].push_back(bx);
		g[bx].push_back(ax);
	}

	int cu=0;

	for(long i=1;i<=n;i++)
	{
		if(v[i]==0)
		{
			//cout<<i<<endl;
			cu++;
			dfs(i);
		}
	}
	cout<<cu<<endl;

}
