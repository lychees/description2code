#include <bits/stdc++.h>
using namespace std;
#define endl '\n';
typedef long long int LL;
LL dp[4][100005];
LL v[100005][4];
void cool(LL i,LL e,LL n)
{
    vector<LL> r;
  for(LL j=1;j<=3;j++)
  {
      if(j!=e&&dp[j][i+1]==-1)
            cool(i+1,j,n);
      if(j!=e)
            r.push_back(v[i][j]+dp[j][i+1]);
  }
  sort(r.begin(),r.end());
  r.clear();
  dp[e][i]=r[0];
}

int main()
{
      ios_base::sync_with_stdio(false);cin.tie(0);

      //freopen("input.in","r",stdin);
    
      LL t;
      cin>>t;
      while(t--)
      {
            LL n;cin>>n;

            for(LL i=1;i<=n;i++)
              {
                  cin>>v[i][1]>>v[i][2]>>v[i][3];
                  dp[0][i]=dp[1][i]=dp[2][i]=dp[3][i]=-1;
              }
            dp[0][n+1]=dp[1][n+1]=dp[2][n+1]=dp[3][n+1]=0;
            cool(1,0,n);
            cout<<dp[0][1]<<endl;
      }
      
    return 0;
}
