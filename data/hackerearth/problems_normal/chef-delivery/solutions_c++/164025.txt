#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream> 
using namespace std;
 
long long int memo[1001][1001],price[1001],fun[1001];
 
long long int  solve(long long int  b, long long int  n){
    if(n < 0) return 0;
    
    long long int  &ret = memo[b][n];
    
    if(ret == -1){
        ret = solve(b,n-1);
        if(price[n] <= b) ret = max(ret,fun[n] + solve(b-price[n],n-1));
    }
    
    return ret;
}

int  main(){
    long long int  n,m;
    long long int  t;
    cin>>t;
    while(t--){
        scanf("%lld %lld",&n,&m);

        
        for(long long int  i = 0;i < n;++i)
            {scanf("%lld",&price[i]);
            price[i] = (i+1) - price[i];}
            
        for(long long int  i = 0;i < n;++i)
            {scanf("%lld",&fun[i]);
            }
        
        memset(memo,-1,sizeof(memo));
        
        long long int  F = solve(m,n-1),lo = 0,hi = m,mi;
        
        while(lo < hi){
            mi = (lo + hi)>>1;
            
            if(solve(mi,n-1) < F) lo = mi+1;
            else hi = mi;
        }
        
        printf("%lld\n",F);
    }
    
    return 0;
}
