    #include <iostream>
    using namespace std;
     
    int main()
    {
        //cout << "Hello World!" << endl;
        int t; string s;
        cin >> t;
        while(t--){
        	cin >> s;
        	int r=0, u=0, b=0, y=0;
        	for(int i=0; i<s.length(); i++){
        		if(s[i]=='r')
        		r++;
        		if(s[i]=='u')
        		u++;
        		if(s[i]=='b')
        		b++;
        		if(s[i]=='y')
        		y++;
        	}
        	if(r<=u && r<=b && r<=y)
        	cout << r <<endl;
        	else if(u<=r && u<=b && u<=y)
        	cout << u << endl;
        	else if(b<=r && b<=u && b<=y)
        	cout << b << endl;
        	else if(y<=r && y<=u && y<=b)
        	cout << y << endl;
        	
        }
        return 0;
    }
