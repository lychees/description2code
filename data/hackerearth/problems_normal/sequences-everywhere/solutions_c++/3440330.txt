#include <bits/stdc++.h>
using namespace std;
#define forn(i, n) for(int i = 0; i < (int) (n); i++)
typedef long long tint;
const int N = 500;

int fact[N];
tint a, b, largo[N], f[N], ans;

int getMinNotDiv(int v){
    for(int i = 2; i <= v; i++){
        if(v % i) return i;
    }
    return 2;
}

tint divs(tint k){
    return b / k - (a - 1) / k;
}

int main(){
    scanf("%lld %lld", &a, &b);
    for(int i = 2; i <= 200; i++){
        int t = i;
        int cnt = 1;
        while(t != 2){
            cnt++;
            t = getMinNotDiv(t);
        }
        largo[i] = cnt;
    }
    while(a <= b && a <= 45){
        ans += largo[a];
        a++;
    }
    if(a == b){
        printf("%lld\n", ans);
        return 0;
    }

    tint val = 1;
    for(tint i = 1; i <= 42; i++){
        val = val * i / __gcd(val, i);
        f[i] = val;
    }

    tint soFar = 0;
    for(tint i = 42; i >= 1; i--){
        tint add = divs(f[i]) - soFar;
        soFar += add;
        ans += add * (largo[i + 1] + 1);
    }
    printf("%lld\n", ans);
    return 0;
}
