#include <iostream>

using namespace std;

int main() {
    int a,b;


    int max_serve=0;
    int max_col=0;
    int counting_temp=0;
    cin >> a >> b;
      int arr[a][b];
    int count_rows[a];
    int count_col[b];
    int pop[100000];
    for(int i=0;i<100000;i++)
        pop[i]=-10000;
    for(int i=0;i<a;i++)
        for(int j=0;j<b;j++)
            cin >> arr[i][j];



    for(int i=0;i<a;i++)
{
    for(int j=0;j<a;j++)
    {
        if(i!=j)
        {
            for(int k=0;k<b;k++)
            {
                if(arr[i][k]!=arr[j][k])
                {
                    counting_temp++;
                    break;
                }


            }
        }

    }

    count_rows[i]=a-counting_temp;
    counting_temp=0;

}

    counting_temp=0;
    for(int i=0;i<a;i++)
    {
        for(int j=0;j<b;j++)
            {
            if(arr[i][j]==0)
            {
                counting_temp++;
            }

            }
    count_col[i]=counting_temp;
    counting_temp=0;
    }


    max_col=count_rows[0];
    max_serve=count_col[0];

    for(int i=1;i<a;i++)
{

    if(max_col<=count_rows[i])
    {
        if(max_serve>=count_col[i])
        {
            max_col=count_rows[i];
            max_serve=count_col[i];
        }
    }

}


    cout << max_serve << " ";
    cout << max_col << endl;
    return 0;
}
