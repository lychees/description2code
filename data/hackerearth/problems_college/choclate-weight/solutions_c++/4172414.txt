#include<iostream>
#include<algorithm>
using namespace std;
int ab(int a)
{
	if(a>0)
	return a;
	else
	return -a;
}
int main()
{
		int n;
		cin>>n;
		int a[n];
		int i;
		for(i=0;i<n;i++)
			cin>>a[i];
		sort(a,a+n);
		int q;
		cin>>q;
		int b;
		while(q>0)
		{
			int flag=0;
			cin>>b;
			if(binary_search(a,a+n,b))
				cout<<"I Like You"<<endl;
			else
			{
				for(i=0;i<n;i++)
				{
					
					if(binary_search(a,a+n,ab(b-a[i])) && b!=2*a[i])
					{
						flag=1;
						break;
					}
				}
				if(flag==1)
					cout<<"I Like You"<<endl;
				else
					cout<<"I Hate You"<<endl;
			}
			q--;
		}
		
	
}
