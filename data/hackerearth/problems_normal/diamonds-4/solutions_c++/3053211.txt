#include <cstdio>
#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n,m;
    	scanf("%d%d",&n,&m);
    	char arr[1000][1000];
    	int i=0,j=0,sum,count=0;
    	for(i=0;i<n;i++)
    	{
    		for(j=0;j<m;j++)
    		{
    			cin>>arr[i][j];
    		}
    	}
    	for(i=0;i<n-1;i++)
    	{
    		for(j=0;j<m-1;j++)
    		{
    			if(arr[i][j]=='/' && arr[i][j+1]=='\\' && arr[i+1][j]=='\\' && arr[i+1][j+1]=='/')
    		   {	count++;}
    		}
    	}
    	printf("%d\n",count);
    }
    return 0;
}
