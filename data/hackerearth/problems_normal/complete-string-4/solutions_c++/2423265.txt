#include <iostream>
#include <string>
using namespace std;

int main()
{
    int n;
    cin >> n;
    
    for (int i = 0; i < n; i++) {
    	string s;
    	cin >> s;
    	bool marked[26] = {false};
    	int markeds = 0;
    	
    	for (int j = 0; j < s.size(); j++) {
    		if (!marked[s[j]-'a']) {
    			marked[s[j]-'a'] = true;
    			markeds++;
    		}
    	}
    	
    	if (markeds == 26) cout << "YES" << endl;
    	else               cout << "NO" << endl;
    }
    return 0;
}
