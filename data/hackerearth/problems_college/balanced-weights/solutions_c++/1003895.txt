#include<bits/stdc++.h>
using namespace std;
#define LL long long
long long gcd(long long a,long long b)
{
	long long temp;
while(b>0)
{
 temp=b;
  b=a%b;
 a=temp;
}
	return a;
}
int main()
{
	int t,n;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		LL a[n+5];
		for(int i=0;i<n;i++)
		scanf("%lld",&a[i]);
		LL g=a[0];
		for(int i=1;i<n;i++)
		{
			g=gcd(a[i],g);
		}
		printf("%lld\n",g*(LL)n);
	}
	return 0;
}
