#include <iostream>
using namespace std;

int main() {
  // cout << "Hello World!" << endl;
  int t;
  cin >> t;
  while (t--) {
    long long int n, ans = 0;
    cin >> n;
    while (n) {
      if (n & 1 == 1) {
        ans++;
      }
      n >>= 1;
    }
    cout << ans << endl;
  }
  return 0;
}
