#include<stdio.h>
#include<iostream>
#include<algorithm>
#include <string.h>
#include <math.h>


using namespace std;

int main()
{
	int n,m,a,ans =0;
	cin>>n>>m>>a;
	if(n > m) {
	
	if((m % a) == 0) {
		ans = ans + m/a;
	}
	else {
		ans = ans + m/a +1;
	}
	if((n-a)% a == 0) {
		ans = ans + ((n-a)/a) * ans;
	}
	else {
		ans = ans + ((n-a)/a +1)*ans;
	}
}
else {
	
	if(n % a == 0) {
		ans = ans + n/a;
	}
	else {
		ans = ans + n/a +1;
	}
	if( (m-a)% a == 0) {
		ans = ans + ((m-a)/a)*ans;
	}
	else {
		ans = ans + ((m-a)/a +1)*ans;
	}
	
	
}
	cout<<ans;
	
	


return 0;
}
