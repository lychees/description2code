#include<bits/stdc++.h>
using namespace std;
#define gc getchar
inline long long int read()
{
	register char c = gc();
	while(c<'0' || c>'9') c = gc();
	long long int ret = 0;
	while(c>='0' && c<='9')
	{
		ret = 10 * ret + c - 48;
		c = gc();
	}
	return ret;
}
int main()
{
    int t;
    t=read();
    while(t--)
    {
    	long long unsigned n,i,j,k,count=0,a[100001],d,max=0,b[100001]={0};
    	cin>>n;
    	cin>>d;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	for(i=0;i<n;i++)
    	{
    		if(b[i]==0)
    		for(j=i+1;j<n&&b[i]==0;j++)
    		{
    			if(b[j]==0)
    			{
    				if(a[j]-a[i]<=d)
    				{
    					b[j]=1;
    					b[i]=1;
    					count++;
    					break;
    				}
    			}
    			if(a[j]-a[i]>d)
    			{
    				break;
    			}
    		}
    	}
    	printf("%llu\n",count);
    }
	return 0;
}
