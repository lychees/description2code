#include<iostream>
#define max 1000005
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
    int n, num, i, h[max]={0}, h2[max]={0};
    long long int r=0;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>num;
		if(num<0)
		{
			h2[num-(2*num)]++;
			r+=h2[num-(2*num)];
		}
		else
		{
			h[num]++;
			r+=h[num];
		}
	}
	cout<<r<<endl;
	}
return 0;
}
