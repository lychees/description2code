#include <iostream>
#include<stdio.h>
using namespace std;
long long int  f[101];
void pre()
{
    f[0]=1;
    f[1]=1;
    for(int i=2;i<101;i++)
    {
        f[i]=f[i-1]+f[i-2];
    }
}
int main()
{
    pre();
    int tc;
    cin>>tc;
    while(tc--)
    {
        int n;
        cin>>n;
        printf("%lld\n",f[n]);
    }
}
