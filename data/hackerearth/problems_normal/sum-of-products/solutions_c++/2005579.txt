#include <iostream>
#include <algorithm>
#include <cmath>
#define m 1000000007
using namespace std;

int main()
{
    int t;
    long long int n, c, i, x, y, ans, z;
    scanf("%d", &t);
    while(t--)
    {
    	scanf("%lld", &n);c=0;
    	for(i=1;i*i<=n;i++)
    	{
    		c=(c+((n/i)*i)%m)%m;
    	}
    	i=1;
    	z=sqrt(n);
    	x=n/i;
    	y=max(n/(i+1), z);
    	while(x-y>0)
    	{
    		x=x%m;
    		y=y%m;
    		ans=((x*(x+1)/2)%m-(y*(y+1)/2)%m)%m;
    		if(ans<0)
    		ans+=m;
    		ans=(i%m*ans)%m;
    		c=(c+ans)%m;
    		i++;
    		x=n/i;
    		y=max(n/(i+1), z);
    	}
    	printf("%lld\n", c);
    }
    return 0;
}
