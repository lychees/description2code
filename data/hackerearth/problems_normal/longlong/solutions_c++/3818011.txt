string s ;
int t ;

bool find( int tam ){
	unordered_set<ll> c ;
	ll h = 0 , b = 27 , pot = 1 ;
	REP( i , tam ) h = h*b + v( s[ i ] ) ;
	REP( i , tam - 1 ) pot = pot*b ;
	c.insert( h ) ;
	ff( i , 1 , t - tam + 1 ){
		h = h - v( s[ i - 1 ] )*pot ;
		h = h*b + v( s[ i + tam - 1 ] ) ;
		if( c.find( h ) != c.end() ) return 1 ;
		c.insert( h ) ;
	}
	return 0 ;
}

	while( l < r ){
		mid = ( l + r + 1 ) >> 1 ;
		if( find( mid ) ) l = mid ;
		else r = mid - 1 ;
	}

}
