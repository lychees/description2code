#include <iostream>
#include <string>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	string a,b,r;
    	cin >> a;
    	cin >> b;
    	int l = a.length();
    	int c = 0,i;
    	for(i=0;i<l;i++)
    	{
    		r = a;
    		r.insert(i,b);
    		string rev(r.rbegin(),r.rend());
    		if(rev==r)
    		{
    			c++;
    		}
    	}
    	r = a;
    	r.append(b);
    	string rev(r.rbegin(),r.rend());
    	if(rev==r)
    	{
    		c++;
    	}
    	printf("%d\n",c);
    }
    return 0;
}
