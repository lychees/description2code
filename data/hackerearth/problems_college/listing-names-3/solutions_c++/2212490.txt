#include <iostream>
#include<map>
#include<unordered_map>
#include<algorithm>
#include<cstring>
using namespace std;

int main()
{
   int t;
   string s;
   unordered_map<string,int> m;
   vector<string> v;
   cin>>t;
   cin.ignore();
   for(int i=0;i<t;i++)
   {
   	getline(cin,s);
   	if(m[s]==0)
   	     v.push_back(s);
   	m[s]++;
   }
   sort(v.begin(),v.end());
   vector<string>::iterator it;
   for(it=v.begin();it!=v.end();it++)
   {
   	    string s=*it;
   		cout<<s<<" "<<m[s]<<endl;
   }
    return 0;
}
