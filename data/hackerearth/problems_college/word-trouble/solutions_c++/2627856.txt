#include <bits/stdc++.h>
using namespace std;

int main()
{
    int i,n1,n2,t,flag,ans;
    char s1[11],s2[11];
    scanf("%d",&t);
    while(t--)
    {
    	flag=ans=0;
    	cin>>s1;
    	cin>>s2;
    	n1=strlen(s1);
    	n2=strlen(s2);
    	if(n1!=n2)
    	{
    		if(n1>n2)
    		{
    			ans=n1-n2;
    			for(i=0;i<n2;i++)
    			{
    				if(s1[i]!=s2[i])
    					ans++;
    			}
    		}
    		else
    		{
    			ans=n2-n1;
    			for(i=0;i<n1;i++)
    			{
    				if(s1[i]!=s2[i])
    					ans++;
    			}
    		}
    	}
    	else
    	{
    		ans=0;
    		for(i=0;i<n1;i++)
    		{
    			if(s1[i]!=s2[i])
    				ans++;
    		}
    	}
    	if(ans==0)
    		printf("Exact\n");
    	else
    		printf("Non Exact %d\n",ans);
    }
    return 0;
}
