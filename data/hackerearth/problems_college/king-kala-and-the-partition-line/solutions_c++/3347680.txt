#include <iostream>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	long long n,c,x,y,a1=0,a2=0;
   	cin>>n;
   	int a,b;
   	cin>>a>>b>>c;
   	for(long long i=0;i<n;i++)
   	{
   		long long temp;
   		cin>>x>>y;
   		temp=(a*x)+(b*y)+c;
   		if(temp<0)
   		a1++;
   		else if(temp>0)
   		a2++;
   		
   	}
   	if(a1>a2)
   	cout<<a1<<endl;
   	else
   	cout<<a2<<endl;
   }
    return 0;
}
