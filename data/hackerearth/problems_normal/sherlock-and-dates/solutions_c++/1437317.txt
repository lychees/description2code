bool lucky(int d, int m, int y) {
	if(d+1 != m) return false;
	if(m+1 != y%100) return false;
	return true;
}
int d, m, y, d2, m2, y2;

bool between(int d1, int m1, int y1) {
	if(y1 < y || y1 > y2) return false;
	if(y1 == y && m1 < m) return false;
	if(y1 == y && m1 == m && d1 < d) return false;
	if(y1 == y2 && m1 > m2) return false;
	if(y1 == y2 && m1 == m2 && d1 > d2) return false;
	
	return true;
}

int main() {
	int T;
	scanf("%d", &T);
	while(T--) {
		int ans = 0;
		scanf("%02d:%02d:%04d %02d:%02d:%04d", &d, &m, &y, &d2, &m2, &y2);
		for(int D = 1; D < 12; D++) {
			int M = D+1;
			for(int mult = 0; mult*100 + M+1 <= y2; mult++) {
				if(between(D, M, mult*100+M+1))
					ans++;
			}
		}
		printf("%d\n", ans);
	}
	return 0;
}
