#include <bits/stdc++.h>
using namespace std;
long long DP[31][31];

long long calComb(int n, int open, int closed)
{
	if(DP[open][closed] != -1)
		return DP[open][closed];

	if(closed == n)
	{	
		DP[open][closed]=1;
		return 1;
	}
	
	DP[open][closed]=0;
	if(open > closed)
		DP[open][closed] +=  calComb(n, open, closed+1);
	if(open < n)
		DP[open][closed] += calComb(n, open+1, closed);

	return DP[open][closed];
}

int main()
{
    int T;
    cin>>T;
    while(T--)
    {
    	int n;
    	scanf("%d", &n);
		memset(DP,-1,sizeof(DP));
    	printf("%llu\n",calComb(n,0,0));
    }
    return 0;
}
