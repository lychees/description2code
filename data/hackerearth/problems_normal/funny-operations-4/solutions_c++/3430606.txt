#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <cmath>
using namespace std;
int getXor(int arr[],int p,int q){
	int result = arr[p];
	for(int i = p+1;i<=q;i++){
		result = result^arr[i];
	}
	return result;
}
int getAnd(int arr[],int p,int q){
	int result = arr[p];
	for(int i = p+1;i<=q;i++){
		// cout<<result<<" "<<arr[i]<<endl;
		result = result&arr[i];
	}
	return result;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    	int n;
    	scanf("%d",&n);
    	long long int arr[17];
    	for(int i = 0; i < n; i++){
    		cin>>arr[i];
    	}
    	int p = -1;
    	int maxLen = -1;
    	unsigned int setSize = pow(2,n);
    	for(int i=0;i<setSize;i++){
    		int tempA[17] = {0};
    		int counter = 0;
    		for(int j=0;j<n;j++){
    			if(i & (1<<j)){
    				tempA[counter++] = arr[j];	
    			}
    		}
    		int tempP = getXor(tempA,0,counter-1)|getAnd(tempA,0,counter-1);
    			// cout<<tempP<<" "<<p<<" "<<i<<" "<<j-1<<endl;
			if(tempP > p){
				p = tempP;
				maxLen = counter;
			}else if(tempP == p){
				if(maxLen<counter){
					maxLen = counter;
				}
			}
    	}
    	cout<<p<<" "<<maxLen<<endl;
    }
    return 0;
}
