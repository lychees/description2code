#include <bits/stdc++.h>
using namespace std;
#define pb push_back
#define lld I64d
#define mp make_pair
#define ff first
#define ss second
#define ll long long

ll fill(ll a,char type,ll x){

    if(type=='N') return -a;

    else if(type=='+') return a+x;

    else if(type=='-') return a-x;

    else if(type=='*') return a*x;

    else if(type=='/') return a/x;

}

int main(){
	

    int t;

    cin>>t;

    while(t--){

        int b,x;

        char ch;

        cin>>b;

        ll mi=1,ma=1;
 
      

        for(int i=0;i<b;++i){

             cin>>ch;

             if(ch!='N')
                cin>>x;

             vector<ll>v;
             
             v.pb(mi);

             v.pb(ma);

             v.pb(fill(mi,ch,x)); 
             v.pb(fill(ma,ch,x));
            
             sort(v.begin(),v.end());
            
             mi=v[0];
             ma=v.back();

        }   

        printf("%lld\n",ma);

        

    }

	return 0;
}
