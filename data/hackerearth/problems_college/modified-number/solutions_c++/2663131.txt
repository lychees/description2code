//KP Numbers
#include "bits/stdc++.h"
using namespace std;

int main(int argc, char const *argv[])
{
	int p, q;
	cin>> p >> q;
	int cnt = 0;
	for (int i = p; i <= q; ++i)
	{
		int d = log10(i)+1;
		int sep = pow(10, d);
		if(sep % 10 == 9)
			sep++;
		int mul = i*i;
		int r = mul % sep;
		int l = mul / sep;
		if(l+r == i){
			cout<< i << " ";
			cnt++;
		}
	}
	if(cnt == 0)
		cout<< "INVALID RANGE";
	return 0;
}
