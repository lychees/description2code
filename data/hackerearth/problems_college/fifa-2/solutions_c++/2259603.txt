#include <iostream>
#include <map>
#include <utility>
using namespace std;

int main()
{
 	int n;
 	map<pair<int,pair<int,int>>,pair<int,int> >my_map;
 	typedef map<int,int>mymap;
 	mymap hm,fn;
 	scanf("%d",&n);
 	for(int i =1;i<=n;i++)
 	{
 		int c,d;
 		scanf("%d%d",&c,&d);
 		my_map.insert(make_pair((make_pair(i,make_pair(c,d))),make_pair(0,0)));
 		if(hm.count(c) == 0)
 		hm[c] = 1;
 		else
 		hm[c] += 1;
 		if(fn.count(d) == 0)
 		fn[d] = 1;
 		else
 		fn[d] += 1;
 	}
 	//printf("%d\n",my_map.size());
    map<pair<int,pair<int,int>>,pair<int,int> >::iterator it;
    map<pair<int,pair<int,int>>,pair<int,int> >::iterator it1;
    int cnt1 = 0,cnt2 = 0;
    int sz = my_map.size();
    for(it = my_map.begin();it!=my_map.end();it++)
    {
    		
    	it->second.first += sz-1;
    	int x = (hm[it->first.second.second]);
    	it->second.first += x;
    	it->second.second += (sz-x-1);
    		
    	
    }
    //cout<<"cnt1 = "<<cnt1<<" cnt 2 = "<<cnt2<<endl;
    for(it = my_map.begin();it!=my_map.end();it++)
    {
    	cout<<it->second.first<<" "<<it->second.second<<endl;
    }
    return 0;
}
