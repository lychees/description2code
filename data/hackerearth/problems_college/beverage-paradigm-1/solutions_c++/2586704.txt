#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,i,j,k;
    bool found=false;
    scanf("%d",&n);
    int a[n];
    for(i=0;i<n;i++) scanf("%d",&a[i]);
    int tot; scanf("%d",&tot);
    sort(a,a+n);
    for(i=0;i<n;i++)
    {
    	j=i+1;
    	k=n-1;
    	while(j<k)
    	{
    		if((a[i]+a[j]+a[k])==tot) {printf("True");found=true;break;}
    		else if((a[i]+a[j]+a[k])<tot) j++;
    		else k--;
    	}
    	if(found) break;
    }
    if(i==n) printf("False");
    return 0;
}
