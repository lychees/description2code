#include<bits/stdc++.h>
using namespace std;
int main()
{
    int a[3],n,sum=0,i,j,k;
    cin>>n;
    if(n<=100&&n>=1)
    {
    	while(n--)
    	{
    		for(j=0;j<3;j++)
    		{
    			cin>>a[j];
    			if((a[j]>100)||(a[j]<-100))break;
			}
			k=a[1]*a[1]-4*a[0]*a[2];
			if(k>=0)sum++;
		}
		cout<<sum;
	}
	return 0;
}
