from bisect import bisect_left


def make_array(N, A):
    B, C = 1, [0] * N
    C[0] = A[0]
    for i in xrange(1, N):
        B = C[i - 1] + 1
        C[i] = B + A[i] - 1
    return C

if __name__ == '__main__':
    N = input()
    A = map(int, raw_input().split())
    C = make_array(N, A)
   

    for _ in xrange(input()):
        I = input()
        li = bisect_left(C, I)
        print li + 1
