#include<bits/stdc++.h>
using namespace std;
int main()
{
        int t;
        cin>>t;
 long long      int n,m,a,b,c,ans;
        while(t--)
        {
                ans=0;
                cin>>n>>m;
                cin>>a>>b>>c;
                ans=min(min((n*a+m*b),(n*a+m*a+m*c)),n*b+m*b+n*c);
                cout<<ans<<'\n';
        }
        return 0;
}
