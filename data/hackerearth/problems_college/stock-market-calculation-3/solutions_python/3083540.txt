for _ in range(input()):
    noc = input()
    ins = map(int, raw_input().split())
    tmp = ins[::]
    cnt = noc-2
    while cnt >= 0:
        tmp[cnt] = max(tmp[cnt+1], ins[cnt])        
        cnt-=1
    profit = 0    
    for i in range(noc):
        profit += tmp[i] - ins[i]
    print profit
