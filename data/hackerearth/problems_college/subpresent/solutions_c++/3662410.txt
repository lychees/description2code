#include<stdio.h>
#include<string.h>
#include<iostream>
using namespace std;
int main()
{
    string a,b;
    int k,i,j;
    int cnt = 0;
    cin >> a >> k;
    int t;
    scanf("%d" , &t);
    if ( k > t ) {
        printf ("UNLUCKY\n" );
        return 0;
    }
    int lm = a.size();
    while( t-- ) {
    cin >> b;
    int lem = b.size();
    i=0;
    j=0;
    while( i < lem && j < lm )
    {
        if ( a[j] == b[i] )
        {
            i++;
        }
        j++;
    }
    if ( i == lem )
        cnt++;
    }
    if ( cnt >= k )
        printf ( "%d LUCKY\n" , cnt );
    else
        printf ( "%d UNLUCKY\n" , cnt );
    return 0;
}
