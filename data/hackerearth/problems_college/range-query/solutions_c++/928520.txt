#include <bits/stdc++.h>
using namespace std;
#define MAX 10005
int arr[MAX];
struct node{
 int v[105];
};
node ST[4*MAX];
void build(int l,int r,int indx){
 if(l==r){
    ST[indx].v[arr[l]] = 1;
    return;
 }
 int mid = (l+r)/2;
 build(l,mid,indx*2);
 build(mid+1,r,indx*2+1);
    for(int i=1;i<=100;i++)
       ST[indx].v[i] = ST[2*indx].v[i] + ST[2*indx+1].v[i];
}
int query(int a,int b,int val,int l,int r,int indx){
   if(l>=a && r<=b){
    return ST[indx].v[val];
   }
   if(b<l || r<a){
    return 0;
   }
   int mid = (l+r)/2;
   int ans1 = query(a,b,val,l,mid,indx*2);
   int ans2 = query(a,b,val,mid+1,r,indx*2+1);
   return ans1+ans2;
}
int main(){
int n,a,b,val;
scanf("%d",&n);
for(int i=0;i<n;i++)
    scanf("%d",&arr[i]);
build(0,n-1,1);
int Q;
scanf("%d",&Q);
while(Q--){
    scanf("%d %d %d",&a,&b,&val);
    int ans = query(a,b,val,0,n-1,1);
    printf("%d\n",ans);
}
return 0;
}
