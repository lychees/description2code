#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main(){
	vector <int> vec;
	int t=0;
	
	cin >> t;
	
	for(int x=0; x<t; x++){
		int n=0;
		
		cin>> n;
		
		for(int y = 0; y < n; y++){
			int v = 0;
			cin >> v;
			vec.push_back(v);
		}
		// cout<< vec.size()<<endl;
		sort(vec.begin(), vec.end());
		for(int z=vec.size() - 1; z >=0; z--){
			cout<< vec.at(z) <<" ";
		}
		cout<<endl;
		vec.clear();
	}

	return 0;
}
