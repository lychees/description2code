#include <bits/stdc++.h>
int main()
{
   int test,a[]={1, 2, 5, 10, 20, 50, 100, 500, 1000};
   long int n,i;
   scanf("%d",&test);
   while(test--)
   {
   	int sum=0;
   	scanf("%ld",&n);
   	i=8;
   while(i>=0)
   {
   	if(n%a[i]!=n)
   	{
   		sum=sum+n/a[i];
   		n=n%a[i];
   	}
   	i--;
   }
   printf("%d\n",sum);
   }
    return 0;
}
