#include <iostream>
using namespace std;

int main()
{
    int t,n,r[1000];
    cin>>t;
    for(int l=0;l<t;l++)
    {int c=0;
    	cin>>n;
    	for(int k=0;k<n;k++)
    	{
    		cin>>r[k];
    	}
    	for(int i=0;i<n;i++)
    	{
    		for(int j=i;j<n;j++)
    		{
    			if(r[i]==r[j])
    			{
    			c=c+1;
    			}
    		}
    	}cout<<c<<"\n";
    }
    return 0;
}
