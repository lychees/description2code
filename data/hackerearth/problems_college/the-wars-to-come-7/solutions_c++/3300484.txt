#include <bits/stdc++.h>
#define ll long long int
#define gt(i) scanf("%lld", &i)

using namespace std;

int main()
{
	ll n, i, j, k;
	gt(n);
	ll a[n];
	for(i = 0; i < n; i++)	
		gt(a[i]);
	gt(k);
	sort(a, a + n);
	ll ct = 0, sum = 0;
	i = n - 1;
	while(sum < k) {
		sum += a[i];
		i--;
		ct++;
	}
	printf("%lld\n", ct);
	return 0;
}
