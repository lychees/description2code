    #include <iostream>
    using namespace std;
    bool checker(int N){
    	if(N<0) return false;
    	if(N==0) return true;
    	return (checker(N-6)||checker(N-9)||checker(N-20));
    }
    int main()
    {
    	int T, N;
    	cin>>T;
    	while(T--){
    		cin>>N;
    		if(checker(N))
    			cout<<"True"<<endl;
    		else
    			cout<<"False"<<endl;
    	}
    	return 0;
    }
