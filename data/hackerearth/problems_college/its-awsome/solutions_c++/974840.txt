#include <iostream>
using namespace std;
#include<cstdio>
#include<cstring>
int main()
{
    ////cout << "Hello World!" << endl;
    int t;
    char a[20];
    scanf("%d\n",&t);
    while(t--)
    {
    	scanf("%s\n",a);
    	int count=0;
    	int l=strlen(a);
    	for(int i=0;i<l;i++)
    	{
    		int k=a[i]-48;
    		if(k==4 || k==7)
    		count+=1;
    	}
    	if(count==4 || count==7 )
    	cout<<"YES"<<endl;
    	else
    	cout<<"NO"<<endl;
    }
    return 0;
}
