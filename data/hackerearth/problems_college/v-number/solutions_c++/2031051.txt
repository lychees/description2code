#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int T;
    cin>>T;
    for(int i=0;i<T;i++)
    {
    	string s;
    	cin>>s;
    	int sum1=0,sum2=0;
    	for(int j=0;j<s.length()/2;j++)
    	{
    		sum1=sum1+(int)s[j]-48;
    		sum2=sum2+(int)s[s.length()-j-1]-48;
    	}
    	if(sum1==sum2)
    	cout<<"YES"<<endl;
    	else
    	cout<<"NO"<<endl;
    	s.clear();
    }
    return 0;
}
