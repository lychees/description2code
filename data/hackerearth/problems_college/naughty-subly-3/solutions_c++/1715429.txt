#include<iostream>

using namespace std;
int main()
{
	int limit;
	int count=0;
	short int arr[100000];
	cin>>limit;
	for( int i=0; i < limit; i++)
		cin>>arr[i];
	for( int i = 0; i < limit; i++)
		for( int j=0 ; j<i; j++)
			if(arr[i]==arr[j])
			{
				count++;
				break;
			}
	cout<<count;
	return 0;
}
