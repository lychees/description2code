#include <stdio.h>
#include <iostream>
#include <string.h>
#define MAX 1000000
using namespace std;

bool a[MAX];
void mark(int i) {
    int j=i;
    i += j;
    while(i <= MAX){
        a[i] = false;
        i += j;
    }
}

int main(){
    int T,N,i;
    scanf("%d",&T);
    memset(a,true,sizeof(a));
    for(i=2; i<MAX; ++i){
        if(a[i] == true){
            mark(i);
        }
    }
    int m,c=1;
    bool f;
    while(T--){
        scanf("%d",&N);
        m = N/2;
        if(!(N&1)){
            m--;
        }
        //cout<<m<<endl;
        //c=0;
        f=false;
        while(m >= 2){
            //cout<<m<<" "<<N-m<<endl;
            if(a[m]==true && a[N-m]==true){
                f=true;
                break;
            }
            m--;
        }
        if(f){
            printf("%d %d\n",m,N-m);
        }
        else{
            printf("%d %d\n",-1,-1);
        }
    }
    return 0;
}
