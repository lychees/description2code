#include <iostream>
#include <bits/stdc++.h>

using namespace std;


int gcd(int a, int b){
	if(b==0) return a;
	return gcd(b, a%b);
}

int main()
{
   int t;
   cin>>t;
   while(t--){
   		int k, i, j;
   		
   		cin>>k;
   		int x[k], y[k];
   		
   		for(i = 0; i < k; i++){
   			scanf("%d/%d", &x[i], &y[i]);
   		}
   		
   		int fx[k], fy[k];
   		int index[k];
   		
   		for(i = k-1; i >= 0; i--){  			
   			fx[i] = x[i];
   			fy[i] = y[i];
   			index[i] = i;
   			
   			for(j = i+1; j < k; j = index[i]+1){
   				if((((float)fx[i])/fy[i]) >=  (((float)fx[j])/fy[j])){
   					break;
   				}
   				
   				fx[i] += fx[j];
   				fy[i] += fy[j];
   				index[i] = index[j];
   			}
   		}
   		
   		for(i = 0; i < k; i++){
   			int hcf = gcd(fx[i], fy[i]);
   			cout<<fx[i]/hcf<<"/"<<fy[i]/hcf<<"\n";
   		}
   		cout<<"\n";
   }
   return 0;
}
