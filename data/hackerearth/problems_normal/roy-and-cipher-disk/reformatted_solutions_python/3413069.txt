t = int(raw_input())

code = {}
for i in range(26):
    code[chr(ord('a')+i)] = i

for _ in range(t):
    s = raw_input()
    arr = []
    prev = 'a'
    for y in s:
        w1 = code[y] - code[prev]
        if w1 in range(-12, 14):
            arr.append(w1)
        elif w1 > 13:
            arr.append(w1-26)
        else:
            arr.append(w1+26)
        prev = y
    s = ""
    for p in arr:
        s += str(p)
        s += " "
    print(s)
