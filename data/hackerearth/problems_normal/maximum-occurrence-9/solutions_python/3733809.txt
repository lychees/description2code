import operator
s = raw_input()
d = dict()
for c in s:
	if not d.has_key(c):
		d[c] = 1
	else:
		d[c] += 1
t = sorted(d.items(), key=operator.itemgetter(1), reverse=True)

max_occurance = t[0][1]
new_dict = dict()
for ele in t:
	if ele[1] == max_occurance:
		new_dict[ele[0]] = ele[1]
t = sorted(new_dict.items(), key=operator.itemgetter(0), reverse=False)
print t[0][0],t[0][1]
