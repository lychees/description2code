#include <iostream>
using namespace std;

void next(char &ch)
{
    switch (ch)
    {
        case 'A': ch = 'B'; break;
        case 'B': ch = 'C'; break;
        case 'C': ch = 'D'; break;
        case 'D': ch = 'E'; break;
        case 'E': ch = 'F'; break;
        case 'F': ch = 'G'; break;
        case 'G': ch = 'I'; break;
        case 'H': ch = 'I'; break;
        case 'I': ch = 'J'; break;
        case 'J': ch = 'K'; break;
        case 'K': ch = 'L'; break;
        case 'L': ch = 'M'; break;
        case 'M': ch = 'N'; break;
        case 'N': ch = 'O'; break;
        case 'O': ch = 'P'; break;
        case 'P': ch = 'Q'; break;
        case 'Q': ch = 'R'; break;
        case 'R': ch = 'S'; break;
        case 'S': ch = 'T'; break;
        case 'T': ch = 'U'; break;
        case 'U': ch = 'V'; break;
        case 'V': ch = 'W'; break;
        case 'W': ch = 'X'; break;
        case 'X': ch = 'Y'; break;
        case 'Y': ch = 'Z'; break;
        
        case 'a': ch = 'b'; break;
        case 'b': ch = 'c'; break;
        case 'c': ch = 'd'; break;
        case 'd': ch = 'e'; break;
        case 'e': ch = 'f'; break;
        case 'f': ch = 'g'; break;
        case 'g': ch = 'h'; break;
        case 'h': ch = 'i'; break;
        case 'i': ch = 'j'; break;
        case 'j': ch = 'k'; break;
        case 'k': ch = 'l'; break;
        case 'l': ch = 'm'; break;
        case 'm': ch = 'n'; break;
        case 'n': ch = 'o'; break;
        case 'o': ch = 'p'; break;
        case 'p': ch = 'q'; break;
        case 'q': ch = 'r'; break;
        case 'r': ch = 's'; break;
        case 's': ch = 't'; break;
        case 't': ch = 'u'; break;
        case 'u': ch = 'v'; break;
        case 'v': ch = 'w'; break;
        case 'w': ch = 'x'; break;
        case 'x': ch = 'y'; break;
        case 'y': ch = 'z'; break;
    }
}

bool isValid(char ch, int h)
{
    string lCase = " abcdefghijklmnopqrstuvwxyz";
    string uCase = " ABCDEFGHIJKLMNOPQRSTUVWXYZ";
    for(int i = 1; i <= 26; i++)
    {
        if(lCase[i] == ch && h > (lCase.size() - i))
            return false;
        if(uCase[i] == ch && h > (uCase.size() - i))
            return false;
    }
    return true;
}

int main()
{
    int h;
    char ch;
    while(cin >> ch >> h)
    {
        if(h < 2 || h > 26)
            cout << "INVALID" << endl;
        else if(isValid(ch, h))
        {
            for(int i = 1; i <= h; i++)
            {
                for(int j = 0; j < i; j++)
                    cout << ch << " ";
                cout << endl;
                next(ch);
            }
        }
        else
            cout << "INVALID" << endl;
        cout << endl;
    }
    
    return 0;
}
