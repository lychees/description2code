#include <iostream>
#include<algorithm>
#include<vector>
using namespace std;

int main()
{
   int t;
   scanf("%d",&t);
   while(t--)
   {
   	int n;
   	scanf("%d",&n);
   	vector<int> level(n);
   	for(int i=0;i<n;i++)
   		scanf("%d",&level[i]);
   	sort(level.begin(),level.end());
   	vector<long long int > sum;
   	int k=n-1;
   	for(int i=0;i<n/2;i++)
   	{	
   		long long int temp = level[i] + level[k];
   		k--;
   		sum.push_back(temp);
   		
   	}
   	long long int maxlevel=sum[0],minlevel=sum[0];
   	for(int i=0;i<sum.size();i++)
   	{
   		maxlevel = max(maxlevel,sum[i]);
   		minlevel = min(minlevel,sum[i]);
   	}
   	cout<<maxlevel-minlevel<<endl;
   }
}
