#include<bits/stdc++.h>
using namespace std;
int main() {
	int t;
	cin>>t;
	while(t--) {
		long long x,y,n;
		scanf("%lld%lld%lld",&x,&y,&n);
		long long ans=x+y;
		long long g=__gcd(x,y);
		for(int i=3;i<=n;i++) {
			long long temp=abs(x-y);
			ans+=temp;
			if(y==g and temp==0) {
				n-=i;
				long long val=n/3;
				ans+=val*2*g;
				n%=3;
				if(n==1)
					ans+=g;
				if(n==2)
					ans+=2*g;
				break;
			}
			x=y;
			y=temp;
		}
		printf("%lld\n",ans);
	}
	return 0;
}
