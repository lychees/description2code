#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	char s[100000];
    	cin>>s;
    	int l=strlen(s),s1=0,s2=0;
    	for(int i=0;i<l/2;i++)
    	s1+=(s[i]-48);
    	for(int i=l/2;i<l;i++)
    	s2+=(s[i]-48);
    	if(s1==s2)
    	cout<<"YES"<<endl;
    	else
    	cout<<"NO"<<endl;
    }
    return 0;
}
