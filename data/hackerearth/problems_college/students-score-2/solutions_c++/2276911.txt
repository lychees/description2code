#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int a[n],i=-1,j,max=0,k;
    for(j=0;j<n;j++)
    cin>>a[j];
    while(n)
    if(a[--n] > 0)
    break;
    while(i<n)
    if(a[++i]>0)
    break;
    if(i > n || n<0)
    {
    	cout<<0;
    	return 0;
    }
    if(i==n)
    {
    	cout<<a[i];
    	return 0;
    }
    for(i=0;i<n;i++)
    {
    	k=a[i];
    	for(j=i+1;j<=n;j++)
    	{
    		k+=a[j];
    		if(k>max)
    		max=k;
    	}
    }
    cout<<max;
    return 0;
}
