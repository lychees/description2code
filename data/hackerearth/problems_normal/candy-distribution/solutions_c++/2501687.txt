#include<bits/stdc++.h>
#define mx 2005
#define mod 1000000007
#define ll long long
#define pb push_back
using namespace std;
vector<ll>factor(ll x)
{
    ll s=sqrt(x);
    vector<ll>t;
    for(ll i=2;i<=s;i++)
    {
        if(x%i==0){t.pb(i);
        if(i!=x/i)t.pb(x/i);}

    }
    return t;
}
ll pow(ll a,ll b)
{
    if(b==0)return 1;
    else
    {
        ll x=pow(a,b/2);
        if(b%2)return (((x*x)%mod)*a)%mod;
        else return (x*x)%mod;
    }
}
vector<ll>mm;
map<ll,ll>m;
ll make(ll x)
{
    if(x<=4)return 1;
    if(m[x-1]!=0)return mm[m[x-1]-2];
    x--;
    vector<ll>p=factor(x);
    ll ans=0;
    for(int i=0;i<p.size();i++)ans=(ans+pow(make(p[i]),x/p[i]))%mod;
    ans=(ans+1)%mod;
    mm.pb(ans);
    m[x]=mm.size()+1;
    return ans;

}

int main()
{
    ll n;
    scanf("%lld",&n);
    printf("%lld\n",make(n));


}
