#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int a[1000009];

int main()
{
    int n,t,i,x,m,k,j;
    cin>>t;
    while(t--)
    {
    	m = -1;
    	int flag = 1;
    	scanf("%d",&n);
	    for(i=0;i<n;i++)
	    {
	    	scanf("%d",a+i);
	    	if(i!=0)
	    	{
	    		if(a[i]<a[i-1])
	    			flag = 0;
	    	}
	    }
	    if(!flag)
	    {
	    	i=n-2;
	    	while(a[i]<a[i+1])
	    	{
	    		i--;
	    	}
	    	k = i;
	    	for(j=i+1;j<n;j++)
	    	{
	    		if(a[j]<a[i])
	    		{
	    			if(a[j]>m)
	    			{
	    				m = a[j];
	    				k = j;
	    			}
	    		}
	    	}
	    	m = a[i];
	    	a[i] = a[k];
	    	a[k] = m;
	    	sort(a+i+1,a+n,greater<int>());
	    }
    	for(i=0;i<n;i++)
    	{
    		printf("%d ",a[i]);
    	}
    	printf("\n");
    }
    
    
}
