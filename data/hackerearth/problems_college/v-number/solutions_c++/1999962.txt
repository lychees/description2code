#include <iostream>
using namespace std;

int main()
{
    int t,sum1,sum2;
    cin>>t;
    string s;
    while(t--){
    	cin>>s;
    	int i,t2;
    	sum1=0;
    	sum2=0;
    if(s.length()==2){
    	if(s[0]-'0'==s[1]-'0'){
    	cout<<"YES"<<"\n";
    	}else{
    		cout<<"NO"<<"\n";
    	}
    }else{
    for(i=0;i<=((s.length()/2)-1);i++){
    		sum1+=(s[i]-'0');
    	}
    
    	for(t2=i;t2<s.length();t2++){
    		sum2+=(s[t2]-'0');
    	}
    //	cout<<sum2<<" "<<sum1;
    	if(sum1==sum2){
    		cout<<"YES"<<"\n";
    	}else{
    		cout<<"NO"<<"\n";
    	}
      }
    }
    return 0;
}
