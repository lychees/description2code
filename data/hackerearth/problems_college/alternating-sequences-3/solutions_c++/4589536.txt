#include <iostream>
#include <vector>
#include <string>
#include <string.h>
#include <set>
#include <stdlib.h>
#include <stdio.h>
using namespace std;

int main() {
   int n,i,j;
   cin>>n;
   long long int a[n];
   for(i=0;i<n;i++)
    scanf("%lld",&a[i]);
   int dp[n];
   for(i=0;i<n;i++)
    dp[i]=1;
   for(i=1;i<n;i++)
   {
       for(j=0;j<i;j++)
       {
           if(((a[i]>0 && a[j]<0) || (a[i]<0 && a[j]>0)) && abs(a[j])<abs(a[i]) && (dp[i]<dp[j]+1))
           dp[i]=dp[j]+1;

       }
   }
   int m=0;
   for(i=0;i<n;i++)
    m=max(m,dp[i]);
    cout<<m;
	return 0;
}
