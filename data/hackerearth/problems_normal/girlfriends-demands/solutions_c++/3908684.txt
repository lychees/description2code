#include<iostream>
#include<string>
using namespace std;

#define MAX 100001

string S;
long Q, i, length;
unsigned long long a, b;

void check(int x, int y) {
	if (S[x] == S[y]) cout << "Yes" << endl;
	else cout << "No" << endl;
}

int main() {
	// freopen("sample_input.txt", "r", stdin);
	cin >> S;
	length = S.length();

	cin >> Q;
	for (i = 1; i <= Q; i++) {
		cin >> a >> b;
		a = a - 1;
		a = a % length;
		b = b - 1;
		b = b % length;
		check(a, b);
	}
	return 0;
}
