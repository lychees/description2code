using namespace std;
typedef long long int ll;

typedef pair<ll,ll>pii;

typedef vector<ll> vii;

const ll mod = 1000000000+7;
ll n;
ll x[200007],y[200007];
ll sumX[200007],sumY[200007];

int main(){
    //fr("travel.txt");
    //fw("out.txt");

    cin >> n;
    for(ll i=0;i<n;i++){
        cin >> x[i] >> y[i];
    }
    sort(x,x+n);
    sort(y,y+n);

    for(ll i=0;i<n;i++){
        if(i==0){
            sumX[i] = x[i];
            sumY[i] = y[i];
        }else{
            sumX[i] = (sumX[i-1] + x[i])%mod;
            sumY[i] = (sumY[i-1] + y[i])%mod;
        }
    }

    ll ans = 0;
    for(ll i=1;i<n;i++){
        ans = (ans + ((i*x[i])%mod-sumX[i-1]+mod)%mod)%mod;
        ans = (ans + ((i*y[i])%mod-sumY[i-1]+mod)%mod)%mod;
    }
    cout << ans << endl;


    return 0;
}
