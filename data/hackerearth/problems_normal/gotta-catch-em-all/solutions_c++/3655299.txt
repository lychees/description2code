#include <bits/stdc++.h>
using namespace std;

bool reversesort(int a,int b) {
	return a>b;
}

int main()
{
	ios::sync_with_stdio(false);
    int n; cin>>n;
    int data[n];
    for (int i = 0;i<n;i++) cin>>data[i];
    sort(data,data+n,reversesort);
    
    int currmax = 0;
    for (int i = 0;i<n;i++) {
    	//day = i+1
    	currmax = max(currmax,i+1+data[i]);
    }
    cout<<currmax+1;
    
    return 0;
}
