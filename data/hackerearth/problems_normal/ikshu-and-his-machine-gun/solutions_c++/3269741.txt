#include <iostream>
#include <algorithm>
using namespace std;

int main()
{	long long int n;
	cin>>n;
	long long int a[n];
	for(long long int i=0; i<n; i++)
	{ cin>>a[i];
	}
	sort(a,a+n);
	long long int f=0;
	for(long long int i=0; i<n; i++)
	{
		a[i]=a[i]-f;
		cout<<a[i]<<" ";
		f++;
	}
    return 0;
}
