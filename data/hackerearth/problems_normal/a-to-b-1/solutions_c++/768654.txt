#include<iostream>
using namespace std;

typedef long long ll;

ll mod=1000000007;

ll power(ll x,ll y)
{
	if(y==0)
	return 1;
	if(x==0)
	return 0;
	ll res = power(x,y/2);
	res = (res*res)%mod;
	if(y&1)
	res = (res *x)%mod;
	return res;
}

int main()
{
	int N;
	int zero_count=0;
	cin>>N;
	ll arr[N];
	ll res = 1;
	for(ll i=0;i<N;i++)
	{
		cin>>arr[i];
		if(arr[i]==0)
		zero_count++;
		else
		res = (res*arr[i])%mod;
	}
	
	int Q;
	cin>>Q;
	while(Q--)
	{
		int x;
		cin>>x;
		if(x==0)
		{
			int ind,val;
			cin>>ind>>val;
			if(val==0 && arr[ind-1]==0);
			else if(val==0)
			{
				res = (res*power(arr[ind-1],mod-2))%mod;
				zero_count++;
				arr[ind-1]=0;
			}
			else if(arr[ind-1]==0)
			{
				res = (res*val)%mod;
				arr[ind-1]=val;
				zero_count--;
				
			}
			else{
			ind--;
			int st=arr[ind];
			res = (res * val)%mod;
			ll mul=power(st,mod-2);
			res = (res *mul)%mod;
			arr[ind]=val;}
		}
		else
		{
			int ind;
			cin>>ind;
			ll ans;
			if(zero_count>1)
			ans = 0;
			else if(zero_count==1 && arr[ind-1]==0)
			ans=res;
			else if(zero_count==1)
			ans=0;
			else
			ans = (res * power(arr[ind-1],mod-2))%mod;
			cout << ans<<endl;
		}
	}
    
}
