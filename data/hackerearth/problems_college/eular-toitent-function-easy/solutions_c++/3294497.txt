#include<stdio.h>
int gcd(int i, int j) {
	int r=1;;
	while( j%i!=0 ) {
		r=j%i;
		j=i;
		i=r;
	}
	return i;
}
int main() {
	int t, i, n, x;
	scanf("%d", &t);
	for( ; t; t--) {
		scanf("%d", &n);
		for(x=1, i=2; i<n; i++) {
			if( n%i!=0 && gcd(n, i)==1 ) {
				x++;
			}
		}
		printf("%d\n", x);
	}
	return 0;
}
