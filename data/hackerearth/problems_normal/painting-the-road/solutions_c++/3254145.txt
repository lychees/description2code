ll k,l;
ll u[512];

ll c[512];
ll dp[257][257];
ll func(ll a,ll b){
if(dp[a][b]!=-1)
return dp[a][b];
vector<ll> V;
ll i,x,y;
rep(i,k){
ll x=u[i],y=v[i];
if(x<=a && y>=b)
V.pb(c[i]);
else if(x<=a && (y<b && y>a))
V.pb(c[i]+func(y,b));
else if( (x>a && x<b) && y>=b)
V.pb(c[i]+func(a,x));
else if((x>a && x<b) && (y<b && y>a))
V.pb(c[i]+func(a,x)+func(y,b));
}
ll ans=(*min_element(V.begin(),V.end()));
dp[a][b]=ans;
return ans;
}
