#include<bits/stdc++.h>
using namespace std;
#define MOD	1000000007
#define ll	long long int
ll num[1000005],den[1000005];
ll mypow(ll base,ll po)
{
	ll ret=1;
	while(po)
	{
		if(po&1)
			ret=(ret*base)%MOD;
		po/=2;
		base=(base*base)%MOD;
	}
	return ret;
}
void pre_cal() 
{
	num[0]=1;
	for(int i=1 ;i<=1000005;++i) 
		num[i]=(i* num[i-1])%MOD;
	den[0]=1;
	for (int i=1;i<=1000005;++i)
		den[i]=mypow(num[i],MOD-2);
}
int main()
{
	pre_cal();
	int t;
	scanf("%d",&t);
	while(t--)
	{
		string str;
		cin>>str;
		ll vov_cnt=0,con_cnt=0,ar[26];
		memset(ar,0,sizeof(ar));
		for(int i=0;i<str.length();++i)
		{
			if(str[i]=='a' || str[i]=='e' || str[i]=='i' || str[i]=='o' || str[i]=='u')
				vov_cnt++;
			else
				con_cnt++;
			ar[str[i]-'a']++;
		}
		if(vov_cnt>con_cnt+1)
			printf("-1\n");
		else
		{
			ll ans=1;
			ans=(ans*num[con_cnt])%MOD;
			ans=(ans*num[con_cnt+1])%MOD;
			ans=(ans*den[con_cnt+1-vov_cnt])%MOD;
			for(int i=0;i<26;++i)
				ans=(ans*den[ar[i]])%MOD;
			printf("%lld\n",ans);
		}
	}
	return 0;
}
