#include <iostream>
using namespace std;
#include<cstring>
#include<cstdio>
#include<cmath>
int main()
{
	int t,f,i,j,len;
	char s[1000001];
   cin>>t;
   while(t--)  {
   	f=0;
   	cin>>s;
   	len=strlen(s);
   	len-=1;
   	for(i=0,j=len-1;i<len-1 && j>=0;i++,j--) {
   		if((abs(s[i]-s[i+1]))!=(abs(s[j]-s[j+1]))) {
   			f=1;
   			break;
   		}
   	}
   	if(f==1)
   		printf("Not Funny\n");
   	else
   		printf("Funny\n");
   }
    return 0;
}
