#include <iostream>
using namespace std;

int main()
{
   int t ; 
   cin >> t ; 
   while ( t--)
   {
   	long long int n ; 
   	cin >> n ; 
   	long long int i ;
   	long long int p ; 
   	long long int sum = 0; 
   	for ( i = 0; i< n ; i++)
   	{
   		cin >> p ;
   		sum = sum + p ; 
   	}
   	cout << sum-n+1 <<endl; 
   }
    return 0;
}
