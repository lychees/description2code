#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    long long int n;
    long long int t;
    
    cin >> n;
    vector <long long int> array;
    while(n--){
    	cin >> t;
    	array.push_back(t);
    }
    sort(array.begin(),array.end());
    long long int last = array.size()-1;
    long long int pro_1 =  array[last]*array[last-1];
    long long int pro_2 = array[0]*array[1];
    
    if(pro_1 > pro_2){
    	cout << pro_1 << endl;
    }else{
    	cout << pro_2 << endl;
    }
    return 0;
}
