#include<iostream>
#include<cstdio>
#include<cmath>

#define FOR(i,j,k) for(i=j;i<k;i++)
#define MOD 1000000007
#define s(x) scanf("%d",&x)

typedef long long ll;
using namespace std;

int fast_pow(int base, int n,long long M)
{
    if(n==0)
       return 1;
    if(n==1)
    return base;
    long long halfn=fast_pow(base,n/2,M);
    if(n%2==0)
        return ( halfn * halfn ) % M;
    else
        return ( ( ( halfn * halfn ) % M ) * base ) % M;
}

int main()
{
int t,x,y;
cin>>t;
while(t--)
{
    cin>>x>>y;
    cout<<fast_pow(y,x,MOD)<<endl;
}
}
