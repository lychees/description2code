#include<bits/stdc++.h>
using namespace std;
typedef long long lld;
int main()
{
	lld n,t,i,j,x,y;
	cin>>n;
	lld ar[n+1];
	cin>>t;
	for(i=0;i<n+1;i++)
	{
		ar[i]=i;
	}
	while(t--)
	{
		cin>>x>>y;
		ar[y]=0;
	}
	lld count=0;
	for(i=1;i<n+1;i++)
	{
		if(ar[i]==i)
			count++;
	}
	cout<<count;
	return 0;
}
