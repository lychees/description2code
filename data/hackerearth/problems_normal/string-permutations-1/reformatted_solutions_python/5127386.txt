
from collections import Counter


def main():
    t = int(raw_input())
    names = []
    for i in range(t):
        names.append(raw_input().strip())
    names = Counter(names)
    max_value = 0
    prev_value = 0
    freq_list = names.values()
    freq_list.sort(reverse=True)
    for value in freq_list:
        if max_value == 0:
            max_value += value
            prev_value = value
        elif value == prev_value:
            max_value += value
        else:
            pass
    return max_value


if __name__ == '__main__':
    output = main()
    print output
