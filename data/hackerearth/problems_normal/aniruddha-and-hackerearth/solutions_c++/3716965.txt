#include<iostream>
#include<cstdio>
long long dp[1000002];
#define m  1000000007;
using namespace std;
int main(void)
{

	int t ,n;
	long long  nh,ne,temph,tempe;
	cin>>t;
		dp[1]= 2;
		nh = ne = 1;
		
		for(int i=2 ; i<=1000001;i++)
		{
			dp[i]  =(dp[i-1] + 2*nh + ne)%m;
			tempe = ne%m ;
			ne = nh%m ;
			nh = (nh + tempe)%m;
		}
		
	while(t--)
	{
		cin>>n;
		cout<<dp[n]<<endl;
	}


	return 0;

}
