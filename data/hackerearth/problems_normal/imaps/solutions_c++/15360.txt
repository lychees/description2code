#include<iostream>
#include<stdio.h>
#include<vector>
#include<string.h>

using namespace std;

vector<int> g[1000];

bool check[1000];

bool dfs(int x)
{
	check[x] = true;
	//printf("%d\n",x);
	bool curr = false;
	for(int i=0;i<g[x].size();i++)
	{
		if(check[g[x][i]] || dfs(g[x][i]))
		{
			curr = true;
			break;
		}
	}
	return curr;
}

int main()
{
	int t,n,m,i,j,x,k;
	scanf("%d",&t);
	for(k=1;k<=t;k++)
	{
		scanf("%d",&n);
		for(i=0;i<n;i++)
		{
			scanf("%d",&m);
			g[i].resize(m);
			for(j=0;j<m;j++)
			{
				scanf("%d",&x);
				g[i][j] = x-1;
			}
		}

		/*for(i=0;i<n;i++)
		{
			for(j=0;j<g[i].size();j++)
			{
				printf("%d ",g[i][j]);
			}
			printf("\n");
		}*/

		bool curr = false;
		for(i=0;i<n;i++)
		{
			memset(check,false,sizeof(check));
			if(dfs(i))
			{
				curr = true;
				break;
			}
		}
		printf("#%d:",k);
		if(curr)
			printf("Yes\n");
		else
			printf("No\n");
	}
	return 0;
}
