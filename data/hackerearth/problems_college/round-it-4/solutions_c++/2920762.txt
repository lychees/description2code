#include<bits/stdc++.h>
using namespace std;
int main()
{

long long tiya;
scanf("%lld",&tiya);

while(tiya--)
{

long long n,b1;
cin>>n>>b1;

long long x=n/b1;
long long y=x+1;

if(abs(b1*x-n)<abs(b1*y-n))
{cout<<b1*x<<endl;
continue;
}
if(abs(b1*x-n)==abs(b1*y-n))
{cout<<b1*y<<endl;
continue;
}
cout<<b1*y<<endl;
}
return 0;
}
