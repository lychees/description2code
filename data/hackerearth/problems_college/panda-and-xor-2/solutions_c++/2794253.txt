#include<bits/stdc++.h>
using  namespace std;
#define ll long long int
ll set_bits(ll num)
{
    ll count=0;
    while(num)
    {
        num&=(num-1);
        count++;
    }
    return count;
}
int main()
{
    ll a,b,test,ans;
    scanf("%lld",&test);
    while(test--)
    {
        scanf("%lld%lld",&a,&b);
        ans=set_bits(a^b);
        if(ans%2)
            cout<<"NO\n";
        else
            cout<<"YES\n";
    }
}
