#include <bits/stdc++.h>
using namespace std;

int visited[1111111];

int main() {
	int n, k, q, mx = 0;
	cin >> n;
	for(int i=2; i<=n; i++) {
		if(!visited[i]) {
			int j = i;
			while(j <= n) {
				visited[j] ++;
				j += i;
			}
		}
		mx = max(mx, visited[i]);
	}	
	int ans[50] = {0};
	for(int i=1; i<=n; i++) {
		ans[visited[i]] ++;
	}
	cin >> q;
	while( q-- ) {
		cin >> k;
		if(k > mx) cout << 0 << "\n";
		else cout << ans[k] << "\n";
	}
	return 0;
}
