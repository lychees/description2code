#include<stdio.h>
#include<iostream>
#include<algorithm>
using namespace std;
//#define gc() getchar()
#define gc() getchar_unlocked()
inline int fastread()
{
int i=0;
char c=gc();
while(c<48) c=gc();
while(c>47)
{
i=i*10+c-48;
c=gc();
}
return i;
}
int main()
{
    int t,n,i;
    t=fastread();
    while(t--)
    {
        n=fastread();
        int a[n];
        for(i=0;i<n;i++) a[i]=fastread();
        sort(a,a+n);
        for(i=0;i<n;i++) printf("%d ",a[i]);
        printf("\n");
    }
    return 0;
}
