#include<bits/stdc++.h>
#define ll long long int
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int x1,y1,x2,y2,xc,yc;
    	scanf("%d%d%d%d%d%d",&x1,&y1,&x2,&y2,&xc,&yc);
    	ll t1=x1*1ll*yc-y1*1ll*xc;
    	ll t2=x2*1ll*yc-y2*1ll*xc;
    	ll p1=(x2-x1)*1ll*y1+(y1-y2)*1ll*x1;
    	ll p2=(x1-x2)*1ll*(yc-y1)-(y1-y2)*1ll*(xc-x1);
    	if((t1>=0 && t2<=0) || (t1<=0 && t2>=0))
    	{
    		if((p1>=0 && p2<=0) || (p1<=0 && p2>=0))
    		{
				if(p1==0 && p2==0 && t1==0 && t2==0 && (xc<x1 && xc<x2))
					printf("YES\n");
				else
    				printf("NO\n");
    		}
    		else printf("YES\n");
    	}
    	else printf("YES\n");
    }
    return 0;
}
