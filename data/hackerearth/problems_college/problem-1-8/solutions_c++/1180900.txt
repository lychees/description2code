#include <iostream>
#include<cstring>
#include<cstdio>
using namespace std;

int main()
{
    char s[101];
    scanf("%s",&s);
int u=strlen(s);
int n,i=0,l,count=0;
    cin>>n;
    char c[n][100];
    char t[100];
    while(i<n)
    {
    scanf("%s",c[i]);
    
    for(l=0;l<i;l++)
    {if(strcmp(c[l],c[l+1])>0)
    {strcpy(t,c[l]);
    strcpy(c[l],c[l+1]);
    strcpy(c[l+1],t);
    }}
    	i++;
    }
    
    for(i=0;i<n;i++)
    {
    	strncpy(t,c[i],u);
    
    if(strncmp(t,s,u)!=0)
    count++;
    
    }
    if(count==n)
    printf("%s\n",s);
    else
    printf("%s\n",c[0]);
    
    return 0;
}
