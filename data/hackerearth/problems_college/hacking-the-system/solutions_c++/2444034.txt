#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s1,s2;
    cin>>s1>>s2;
    int c=0;
    for(int i=0;i<s1.length();i++)
     {
     	if(s1[i]!=s2[i])
       c++;
      if(c>1)
        break;
   }
    if(c==0 || c==1)
    cout<<"wins";
    else
    cout<<"loses";
    return 0;
}
