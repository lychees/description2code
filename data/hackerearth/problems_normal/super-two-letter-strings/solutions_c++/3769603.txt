#include<bits/stdc++.h>
using namespace std;
long long const  mod =1000000000+7;
int dp[2][10005];
int main()
{
//	freopen("in.txt","r",stdin);
    int t;
	cin>>t;//int co=1;
	while(t--)
	{
//		cout<<co++<<"  ";
		int n,p;
		cin>>n>>p;
		if(n==1)
		{
			cout<<"1"<<endl;
			continue;
		}
 	    for(int i=0;i<=n+1;i++)
 	    {
 	    	dp[0][i]=dp[1][i]=0;
 	    }
		dp[0][1]=0;
		dp[1][1]=1;
		for(int i=2;i<=p-1;i++)
		{
			dp[0][i]=dp[0][i-1]%mod+dp[1][i-1]%mod;
			dp[0][i]%=mod;
			dp[1][i]=dp[0][i-1]%mod+dp[1][i-1]%mod;
			dp[1][i]%=mod;

		}

//        for(int i=1;i<=p-1;i++)
  //      {
    //    	cout<<dp[0][i]<<" "<<dp[1][i]<<endl;
      //  }
        for(int i=p;i<=n;i++)
        {
        	for(int j=i-1;j>=i-p+1;j--)
        	{
        		dp[0][i]=dp[0][i]%mod+dp[1][j]%mod;
        				dp[0][i]%=mod;
			}
			if(i==1)
        	dp[1][i]=1;
			else
	     	dp[1][i]=dp[0][i-1]%mod+dp[1][i-1]%mod;
			dp[1][i]%=mod;

        }
//		cout<<dp[0][n]<<" "<<dp[1][n];

        cout<<(dp[0][n]+dp[1][n])%mod<<endl;
	} 
    return 0;
}
