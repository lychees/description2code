#include <bits/stdc++.h>
using namespace std;
#define ll long long int

int main()
{
	ll n;
	cin>>n;
	while(n){
		n--;
		bool ans=false;
		ll x,count=0;
		cin>>x;
		for(ll i=1;i*i<x;i++)
		{
			if(x%i==0)
			count=count+2;
			if(count>=4)
			{
				ans=true;
				break;
			}
		}

		if(ans){
			cout<<"YES"<<endl;
		}else{
			cout<<"NO"<<endl;
		}
	}
	return 0;
}
