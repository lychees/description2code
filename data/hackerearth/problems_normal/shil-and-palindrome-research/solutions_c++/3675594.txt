#include<bits/stdc++.h>
using namespace std;
struct node{
    int hash[26];
};
node arr[1000000];
int n,q;
void update(int index,char c,char p)
{
    index++;
    while(index<=n)
    {
        if(p>='a'&&p<='z')
        arr[index].hash[p-'a']--;
        arr[index].hash[c-'a']++;
        index+=index&(-index);
    }
}
void get(int index,int *has)
{
    index++;
    while(index>0)
    {
        for(int i=0;i<26;i++)
        {
            has[i]+=arr[index].hash[i];
        }
        index-=index&(-index);
    }
}
void create(char *s)
{
    for(int i=0;s[i]!='\0';i++)
    {
        update(i,s[i],'2');
    }
}
