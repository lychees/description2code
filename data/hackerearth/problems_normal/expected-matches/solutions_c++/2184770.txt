typedef unsigned long long int uint64;
typedef long long int int64;

double mat[1001][1001];
int n,m;
double dp[1001][2005];
double go(int p1,int p2,int match){
	if((p1==n)||(p2==m))
	return match;
	if(dp[p1][match]>=0.0)
	return dp[p1][match];
	double ret=0;
	ret+=mat[p1][p2]*go(p1,p2+1,match+1);
	ret+=(1-mat[p1][p2])*go(p1+1,p2,match+1);
	dp[p1][match]=ret;
	return ret;
}

int main(){
	int t,i;
	cin>>t;
	while(t--){
		cin>>n>>m;
		memset(dp,-1,sizeof(dp));
		for(i=0;i<n;i++)
		for(int j=0;j<m;j++)
		scanf("%lf",&mat[i][j]);
		
		printf("%.6lf\n",go(0,0,0));
	}
	return 0;
}
