#include <iostream>
#include <math.h>

using namespace std;

#define ll long long

ll arr[10002];

void solve(ll n) {

	ll prev = -1;
	ll start = -1, end = -1;

	ll i;
	for (i = 0; i < n; ++i)
		if (arr[i] > prev)
			prev = arr[i];
		else {
			start = i - 1;
			break;
		}

	if (i == n) {
		cout << -1 << " " << -1;
		return;
	}

	prev = arr[i];

	for (++i; i < n; ++i)
		if (arr[i] < prev)
			prev = arr[i];
		else {
			end = i - 1;
			break;
		}

	if (i == n) {
		end = i - 1;
	}

	ll leftmost, rightmost;

	if (start == 0)
		leftmost = -1;
	else
		leftmost = arr[start - 1];

	rightmost = end == n-1 ? 1000000000000 : arr[end+1];

	if (!(arr[end] > leftmost and arr[start] < rightmost)) {
		cout << -1 << " " << -1;
		return;
	}

	prev = i==n ? 0 : arr[i];
	for (++i; i < n; ++i)
		if (arr[i] > prev)
			prev = arr[i];
		else {
			cout << -1 << " " << -1;
			return;
		}

	cout << start+1 << " " << end+1;

}

int main() {
	ll T;

	cin >> T;

	for (ll ii = 0; ii < T; ++ii) {
		ll n;
		cin >> n;

		for (ll i = 0; i < n; ++i)
			cin >> arr[i];
		solve(n);
		cout<<endl;

	}

}
