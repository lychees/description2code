	#include <iostream>
	#include <vector>
	#include <algorithm>
	using namespace std;

	int main()
	{
		int N, len;
		bool flag;
		string str, sorted;
		vector < pair<string, string> > list;

		cin >> N;

		for (int n = 0; n < N; n++) {

			flag = false;
			cin >> str;
			
			len = str.size();
			sorted = str;
			sort(sorted.begin(), sorted.end());

			for (int j = 0; j < list.size(); j++) {

				if (list[j].second == sorted) {
					if (str[0] == list[j].first[0] && str[len - 1] == list[j].first[len - 1]) {
						flag = true;
					}
				}
			}
			if (!flag) {
				list.push_back(make_pair(str, sorted));
			}
		}
	 	cout << list.size();
	    return 0;
	}
