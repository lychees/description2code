typedef long long LL;
LL MOD = 1000000007;

#define MAXN 1000002
LL phi[MAXN + 1], prime[MAXN/10], sz;
bitset <MAXN + 1> mark;
 
int main()
{
	for (int i = 2; i <= MAXN; i++ ){
	if(!mark[i]){
		phi[i] = i-1;
		prime[sz++]= i;
	}
	for (int j=0; j<sz && prime[j]*i <= MAXN; j++ ){
		mark[prime[j]*i]=1;
		if(i%prime[j]==0){
			phi[i*prime[j]] = phi[i]*prime[j];
			break;
		}
		else phi[i*prime[j]] = phi[i]*(prime[j]-1 );
	}
	}
	phi[1] = 0;
	phi[0] = 0;
	for(int i = 2; i <=1000000; ++i){
		phi[i] = i - phi[i]-1;
		// if(i < 100) cout << phi[i] << " ";
		phi[i] += phi[i-1];
	}
	// cout << endl;

	cin >> T;

		int A, B;

		LL ans = phi[A];

	}

}
