#include <iostream>
#include <string.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    while(n--)
    {
    	int i,j,count=0;
    string s1,s2;
    cin>>s1>>s2;
    int n1=s1.length();
    int n2=s2.length();
    if(n1<n2)
    {
    for(i=0,j=0;i<s1.length();i++,j++)
    {
    	if(s1[i]==s2[j])
    	count++;
    }
    cout<<count<<endl;
    }
    else
    {
    for(i=0,j=0;j<s2.length();i++,j++)
    {
    	if(s1[i]==s2[j])
    	count++;
    }
    cout<<count<<endl;
    }
    }
return 0;
}
