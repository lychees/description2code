#include <bits/stdc++.h>
using namespace std;
#define lli long long int

lli fact[12];
int main()
{
	fact[0]=1;
	fact[1]=1;
	for(int i=2;i<=12;i++){
		fact[i] = i*fact[i-1];
	}
	
    lli t;
    cin>>t;
    while(t--){
    	lli c,v;
    	cin>>c>>v;
    	lli x=1;
    	x *= (fact[7]/(fact[c]*fact[7-c]));
    	x *= (fact[5]/(fact[v]*fact[5-v]));
    	x*= fact[c+v];
    	cout<<x<<endl;
    }
    return 0;
}
