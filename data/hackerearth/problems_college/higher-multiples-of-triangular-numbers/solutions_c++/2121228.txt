#include<bits/stdc++.h>
using namespace std;
long int a[100000]={0};
void init()
{
	a[1]=1;
	int n=2;
	int count=0;
	for(int i=3;i<100000;i=i+n)
	{
		count=0;
		for(int j=1;j<=sqrt(i);j++)
		{
			if(i%j==0)
			{
				count++;
			}
			if((i/j)!=j&&(i%(i/j))==0)
			{
				count++;
			}
		}
		a[i]=count;
		n++;
	}
}
int calldiv(int n)
{
	int count=0;
		for(int j=1;j<=sqrt(n);j++)
		{
			if(n%j==0)
			{
				count++;
			}
			if((n/j)!=j&&(n%(n/j))==0)
			{
				count++;
			}
		}
	return count;
}
int main()
{
    init();
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	//int flag=calldiv(n);
    	//cout<<calldiv(4)<<endl;
    	int p=1;
    	for(int i=1;i<100000;i=i+p)
    	{
    		if(n<a[i])
    		{
    			cout<<i<<endl;
    			break;
    		}
    		p++;
    	}
    }
}
