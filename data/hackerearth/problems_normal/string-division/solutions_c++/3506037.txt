#include <iostream>
#include <set>
#include <algorithm>
using namespace std;

int main()
{
int a,b,c,i,j,k,l,m,n,t;
string str,s1,s2,s3,s4;

cin >> t;

while(t--)
{

cin >> str;
n = str.size();
a = 1,b = 2,c =3;
if(str.size() == 4)
{
if(str[0]==str[1] || str[0] == str[2] || str[0]==str[3] || str[1] == str[2] || str[1] == str[3] || str[2] == str[3])
  cout << "NO\n";
else
  cout << "YES\n";
continue;
}

for(i=1;i<n-2;i++)
  for(j=i+1;j<n-1;j++)
    for(k=j+1;k<n;k++)
      {
        s1 = str.substr(0,i);
        s2 = str.substr(i,j-i);
        s3 = str.substr(j,k-j);
        s4 = str.substr(k,n - k);

        //cout<<s1<<" "<<s2<<" "<<s3 <<" " << s4 << endl;
        //cout << " " << i << " " << j << " " << k << endl;
        set<string> s;
        s.insert(s1);
        s.insert(s2);
        s.insert(s3);
        s.insert(s4);


        //cout << "size = " << s.size() << endl;
        if(s.size() == 4)
        {
          cout << "YES\n";
          goto finish;

        }
       s.clear();

      }
cout << "NO\n";
finish : ;
}
return 0;
}
