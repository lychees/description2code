#include <iostream>
#include <algorithm>
#include <cstdio>

using namespace std;
const int MAX = 200005;
const long long MOD = 1e9 + 7;
long long x[MAX], y[MAX];

long long labs(long long a)
{
    if(a < 0) return -a;
    else return a;
}

int main()
{
    //freopen("input.txt", "r", stdin);
    int n, a1 = 0, a2 = 0, k1 = 0, k2 = 0;
    long long x1 = 0, y1 = 0, x2 = 0, y2 = 0;
    scanf("%d", &n);
    for(int i = 0;i < n;++i)
    {
        scanf("%lld %lld", &x[i], &y[i]);
    }
    long long ans = 0;
    sort(x, x + n);
    sort(y, y + n);
    for(int i = 0;i < n;++i)
    {
        if(x[i] < 0) x1 -= x[i], k1++;
        else x2 += x[i], k2++;
        if(y[i] < 0) y1 -= y[i], a1++;
        else y2 += y[i], a2++;
    }

    for(int i = 0;i < n;++i)
    {
        if(x[i] < 0)
        {
            x1 += x[i];
            k1--;
            ans += labs(x1 + x[i]*k1);
            ans += labs(x2 - x[i]*k2);
            //cout << labs(x1 + x[i]*k1) << ' ' << labs(x2 - x[i]*k2) << endl;
        }
        else
        {
            x2 -= x[i];
            k2--;
            ans += labs(x1 + x[i]*k1);
            ans += labs(x2 - x[i]*k2);
            //cout << labs(x1 + x[i]*k1) << ' ' << labs(x2 - x[i]*k2) << endl;
        }
        if(y[i] < 0)
        {
            y1 += y[i];
            a1--;
            ans += labs(y1 + y[i]*a1);
            ans += labs(y2 - y[i]*a2);
            //cout << labs(y1 + y[i]*a1) << ' ' << labs(y2 - y[i]*a2) << endl;
        }
        else
        {
            y2 -= y[i];
            a2--;
            ans += labs(y1 + y[i]*a1);
            ans += labs(y2 - y[i]*a2);
            //cout << labs(y1 + y[i]*a1) << ' ' << labs(y2 - y[i]*a2) << endl;
        }
        //cout << ans << endl;
        ans %= MOD;
    }
    printf("%lld\n", ans);
    return 0;
}
