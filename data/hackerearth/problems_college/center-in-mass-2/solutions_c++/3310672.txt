#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    for(int i=1;i<=t;i++)
    {
        int n;
        cin>>n;
        double a1=0.0,a2=0.0,a3=0.0,b1=0.0,b2=0.0,b3=0.0;
        for(int j=0;j<n;j++)
        {
            double q1,q2,q3,p1,p2,p3;
            cin>>q1>>q2>>q3>>p1>>p2>>p3;
            a1+=q1;
            a2+=q2;
            a3+=q3;
            b1+=p1;
            b2+=p2;
            b3+=p3;
        }
        a1/=n;
        a2/=n;
        a3/=n;
        b1/=n;
        b2/=n;
        b3/=n;
        //cout<<b3<<endl;
        double t1=0,t2=0,t3=0;
        double ans=1000000.000;
        double time1=0.0;
        while(a1<=5000&&a2<=5000&&a3<=5000&&a1>=-5000&&a2>=-5000&&a3>=-5000)
        {
            if(ans>sqrt(a1*a1+a2*a2+a3*a3))
            {
             //   cout<<"ans: "<<ans<<endl;
                ans=sqrt(a1*a1+a2*a2+a3*a3);
                time1=max(max(t1,t2),t3);
            }
            a1+=b1,a2+=b2,a3+=b3;
            t1+=b1,t2+=b2,t3+=b3;
           // cout<<a1<<" "<<a2<<" "<<a3<<endl;
        }
        printf("Case #%d: %.8f %.8f\n",i,ans,time1);
    }
}
