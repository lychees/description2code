#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int calc(string s)
{
	int len = s.length();
	long total = 0;
	int x = 0;
	for(int i=0;i<len;++i)
	{
		if(s[i]=='A' || s[i]=='a' || s[i]=='B' || s[i]=='b'|| s[i]=='C' || s[i]=='c')
		{
			++x;
		}
		else
		{
			total+= (x*(x+1))/2;
			x=0;
		}
	}
	total+= (x*(x+1))/2;
	return total;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
    	string s;
    	cin>>s;
		cout<<calc(s)<<endl;
		
	}
    
    return 0;
}
