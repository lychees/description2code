
#include<bits/stdc++.h>

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n;
        scanf("%d",&n);
        long long int one=1,zero=0,i;
        for(i=0;i<n;i++)
        {
            long long int temp=zero;
            zero=zero+one;
            one=temp;
        }
        printf("%lld %lld\n",one,zero);
    }
    return 0;
}
