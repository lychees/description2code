#include <iostream>
#include<stdio.h>
using namespace std;

int arr[1024][1024];
int count = 1;

void print(int n, int i, int j) {
	if(n == 0) {
		arr[i][j] = count++;
		return;
	}
	int x = 1<<n;
	print(n-1, i, j);
	print(n-1, i, j+x/2);
	print(n-1, i+x/2, j);
	print(n-1, i+x/2, j+x/2);
}

int main()
{
    int n;
    scanf("%d", &n);
    print(n, 0, 0);
    int x = 1<<n;
    for(int i=0; i<x; i++) {
    	for(int j=0; j<x; j++)
    		printf("%d ", arr[i][j]);
    	printf("\n");
    }
    return 0;
}
