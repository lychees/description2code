#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <string>
#include <vector>

using namespace std;

long long mas[111111];

int main() {
    ios::sync_with_stdio(0);
    int t;
    cin >> t;
    while (t --> 0) {
        int n, k, p;
        cin >> n >> k >> p;
        for(int i = 0; i < n; ++i)
            cin >> mas[i];
        sort(mas, mas+n);
        bool f = 0;
        for(int i = 0; i+k-1 < n; ++i)
            if (mas[i+k-1] - mas[i] <= 2*p)
                f = 1;
        
        if (!f)
            cout << "YES\n";
        else
            cout << "NO\n";
    }
    return 0;
}
