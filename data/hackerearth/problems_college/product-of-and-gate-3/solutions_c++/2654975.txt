LL getans(LL st, LL en, LL A, LL B) {
    if (st > B || en < A) 
        return (1LL << 32) - 1;
    if (st >= A && en <= B) {
        return (st & en);
    }
    LL mid = (st + en) / 2;
    return (getans(st, mid, A, B) & getans(mid + 1, en, A, B));
}

    }

}
