for k in range(int(raw_input())):
    n = int(raw_input())
    table = [[0 for x in range(3)] for x in range(n)]
    table[0][0], table[0][1], table[0][2] = map(int, raw_input().split())
    for i in range(1, n):
        p, q, r = map(int, raw_input().split())
        table[i][0] = min(table[i-1][1], table[i-1][2])+p
        table[i][1] = min(table[i-1][0], table[i-1][2])+q
        table[i][2] = min(table[i-1][0], table[i-1][1])+r
    print(min(table[n-1][0], table[n-1][1], table[n-1][2]))
