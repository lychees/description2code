#include <iostream>
#include<string>
using namespace std;

int main()
{
    int t;
    cin>>t;
    string a[50];
    for(int i=0;i<t;i++)
    {
    	cin>>a[i];
    	
    	
    }
     
    for(int i=0;i<t;i++)
    {int triplecounter=0,vowelcounter=0,consonent=0;
    for (char & c : a[i])
{
   if(c=='a'||c=='o'||c=='i'||c=='e'||c=='u'||c=='y')
   {
   	triplecounter=0;
   	vowelcounter++;
   }
   else
   {triplecounter++;consonent++;}
   if(triplecounter==3)
   break;
}
 if(triplecounter==3||consonent>vowelcounter)
 cout<<"hard\n";
 else
 cout<<"easy\n";

    	
    }
    
    return 0;
}
