#include<stdio.h>
#include<stdlib.h>
#include<iostream>
using namespace std;

main()
{
	int T,N,i,j;
	scanf("%d",&T);
	for(j=0;j<T;j++)
	{
		scanf("%d",&N);
		bool ar[N];
		int x;
		int p=0;
		scanf("%d",&x);
		if(x%2==0)
		ar[0]=0;
		else
		ar[0]=1;
		for(i=1;i<N;i++)
		{
		scanf("%d",&x);
		if(x%2==0)
		{	
		ar[i]=ar[i-1];	
		}
		else
		{
		  ar[i]=1-ar[i-1];	
		} 
		}
		/*
		for(i=0;i<N;i++)
		{
			printf("%d",ar[i]);
		}
		*/
		int cot=0,cet=0,cep=0,cop=0,y;
		if(ar[0]==1)
		{
		cot=1;
		cop=1;
	   } 
		else
		{
		cet=1;
		cep=1;
		}
		for(i=1;i<N;i++)
		{
		  if(ar[i]==ar[i-1])
		  {
		  	cot+=cop;
		  	cet+=cep;
		  }
		  else if(ar[i]==ar[i-1])
		  {
		  	cot+=cop;
		  	cet+=cep;
		  }
		  else if(ar[i]!=ar[i-1])
		  {
		  	cot+=cep;
		  	cet+=cop;
		  	y=cop;
		  	cop=cep;
		  	cep=y;
		  }
		  	
		  if(ar[i]-ar[i-1]==0)
		  {
		  	cet++;
			 cep++;  	
		  }
		  else
		  {
		  cot++;
		  cop++;
	}
		  
		}
      if(cot>=cet)
      {
      	printf("BOB\n");
      }
      else
      printf("ANDI\n");
	}
	
}
