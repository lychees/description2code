
no_of_test_cases = int(raw_input())

for _ in range(no_of_test_cases):
    inp = raw_input().split()
    N, M, A, B, C = int(
        inp[0]), int(
        inp[1]), int(
            inp[2]), int(
                inp[3]), int(
                    inp[4])
    print max(N*A+M*B, min(N, M)*C*2+abs(N-M)*(A if (N > M) else B))
