#include <iostream>
using namespace std;

int main()
{
    int tc,n,i;
    bool a[10001];
    a[1]=true;a[2]=false;a[3]=true;a[4]=true;
    for(i=5;i<=10000;i++)
    	if(!(a[i-1] && a[i-4]))
    		a[i]=true;
    	else
    		a[i]=false;
    cin>>tc;
    while(tc--){
    	cin>>n;
    	if(a[n])
    		cout<<"Yes"<<endl;
    	else
    		cout<<"No"<<endl;
    }
    return 0;
}
