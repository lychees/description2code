using namespace std;

#define ll long long
#define vt vector
#define MOD 1000000007

ll powmod(ll a,ll b)
{
	if(b==0)
		return 1;
	if(b==1)
		return a%MOD;
	if(b%2!=0)
	{
		ll temp = a;
		ll temp2 = powmod(a,b-1);
		return (temp*temp2)%MOD;
	}
	else
	{
		ll temp = powmod(a,b/2);
		return (temp*temp)%MOD;
	}
}

ll invmod(ll n)
{
	return powmod(n,MOD-2)%MOD;
}

int main()
{
	int t;
	cin>>t;
	vector<ll> fact(1e6+101,0);
	fact[0]=1;
	for(ll i=1;i<=1e6+100;i++)
		fact[i] = (fact[i-1]*i)%MOD;
	while(t--)
	{
		string temp;
		cin>>temp;
		ll v=0;
		map<char,ll> freq;
		for(int i=0;i<temp.length();i++)
		{
			if((temp[i]=='a' || temp[i]=='e' ||temp[i]=='i' ||temp[i]=='o' ||temp[i]=='u'))
				v++;
			freq[temp[i]]++;
		}
		ll c = temp.length()-v;
		if(c+1<v)
		{
			cout << -1 << endl;
		}
		else
		{
			ll vow = (fact[c+1]*invmod(fact[c+1-v]))%MOD;
			ll cons = fact[c];
			for(int i=0;i<26;i++)
			{
				int j = i+97;
				if(j=='a'||j=='e'||j=='i'||j=='o'||j=='u')
				{
					vow = (vow*invmod(fact[freq[j]]))%MOD;
				}
				else
				{
					cons = (cons*invmod(fact[freq[j]]))%MOD;
				}
			}
			cout << (vow*cons)%MOD << endl;
		}
	}
	return 0;
}
