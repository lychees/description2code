using namespace std;
#include<iostream>
int main()
{
    int n,a,b;
    cin>>n;
    cin>>a;
    cin>>b;
    int i=1;
    int s=a;
    int sum=0;
    while(s<n)
    {
        sum=sum+s;
        i++;
        s=a*i;
    }
    //cout<<sum<<endl;
    s=b;
    i=1;

    while(s<n)
    {
        sum=sum+s;
        i++;
        s=b*i;
    }
    //cout<<sum<<endl;
    i=1;
    s=a*b;
    while(s<n)
    {
        sum=sum-s;
        i++;
        s=a*b*i;
    }
    cout<<sum<<endl;
}
