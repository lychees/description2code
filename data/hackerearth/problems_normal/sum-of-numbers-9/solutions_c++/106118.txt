#include <iostream>

using namespace std;

int main() 
{
	int t;
	cin >> t;
	while ( t-- ) {
		int n,s;
		cin >> n;
		int a[n];
		for ( int i = 0; i < n; i++ ) cin >> a[i];
		cin >> s;
		int lmt = (1 << n);
		int match = 0;
		for ( int i = 1; i < lmt; i++ ) {
			int sum = 0;
			for ( int j = 0; j < n; j++ ) {
				if ( ( ( 1 << j ) & i ) > 0 ) sum += a[j];
			}
			if ( sum == s ) {
				match = 1;
				break;
			}
		}
		if ( match ) cout << "YES\n";
		else cout << "NO\n";
	}
	return 0;
}
