#include<stdio.h>
#include<math.h>
#include<string.h>
#include<stdlib.h>
#include<ctype.h>
#include<vector>
#include<algorithm>
#include<map>
#include<queue>
#include<set>
#include<limits.h>
#include<iostream>

#define MOD 1000000007

using namespace std;

struct node
{
	int a,b,c;
}s[515];

bool compare(struct node const &p,struct node const &q)
{
	if(p.a<q.a)
	return 1;

	return 0;
}

int main()
{
int t,l,k;

cin>>t;

int dp[515];

while(t--)
{
	cin>>l>>k;
	
	for(int i=0;i<k;i++)
	cin>>s[i].a>>s[i].b>>s[i].c;
	
	sort(s,s+k,compare);
	for(int i=0;i<=l;i++)
	dp[i]=1000000;

	dp[0]=0;

	if(s[0].a!=0)
	{
		cout<<"-1"<<endl;
		continue;
	}


	for(int i=0;i<k;i++)
	{
		for(int j=s[i].a+1;j<=s[i].b;j++)
		{
			if(dp[j]>dp[s[i].a]+s[i].c)
			{
				for(int x=j;x<=s[i].b;x++)
				dp[x]=dp[s[i].a]+s[i].c;
				break;
			}
		}
	}



	if(dp[l]==1000000)
	cout<<"-1"<<endl;

	else
	cout<<dp[l]<<endl;

}

	return 0;
}
