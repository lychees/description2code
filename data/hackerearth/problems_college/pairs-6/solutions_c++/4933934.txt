#include <iostream>
using namespace std;

int main()
{
    int test,ct,t,diff,n;
    cin>>test;
    while(test--)
    {
    	cin>>n;
    	cin>>ct>>t;
    	diff=ct-t;
    	if(diff<0)
    	diff=(-1)*diff;
    	if(diff==0 || diff==1)
    	cout<<"Chotu"<<endl;
    	else
    	cout<<"Motu"<<endl;
    }
}
