#include <bits/stdc++.h>

#define REP(i, n)   for(ll i=0;i<n;i++)
#define REPN(i, n)  for(ll i=1;i<=n;i++)
#define SET(i, n)   memset(i, n, sizeof(i))

#define MAX         2050
#define INF         (1 << 29)
#define EPS         1e-9
#define pb          push_back 

using namespace std;

typedef long long ll;
typedef pair<ll, ll>    pii;

const ll MOD = ll(1e7 + 7);

ll n, nn, mm, A[MAX], B[MAX], X;
ll dp[MAX][MAX];
map<ll, ll>mp;

ll recur(int at, int zor){
	if(at > nn) return mp[zor^X];
	if(dp[at][zor] != -1) return dp[at][zor];
	ll ret = 0;
	ret += recur(at+1, zor ^ A[at]);
	ret %= MOD;
	ret += recur(at+1, zor);
	ret %= MOD;
	return dp[at][zor] = ret;  
}

ll solve(){
    for(ll i=0;i<(1 << mm);i++){
        ll zor = 0;
        REP(j, mm){
            if(i & (1 << j)){
                zor ^= B[j+1];
            }
        }
        mp[zor]++;
    }
    ll ret = recur(1, 0);
    return ret;
}

int main(){
    ll x; SET(dp, -1);
    scanf("%lld %lld", &n, &X);
    REPN(i, n){
        scanf("%lld", &x);
        if(x <= (1 << 10)) A[++nn] = x;
        else B[++mm] = x;
    }
    ll ret = solve();
    printf("%lld\n", ret);
    return 0;
}
