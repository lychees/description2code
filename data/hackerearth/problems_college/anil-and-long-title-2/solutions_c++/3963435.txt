#include <iostream>
using namespace std;

int main()
{
	string str ;
	cin>>str;
	int l =str.length() ;

	for(int i=0;i<l;i++)
	{
			if(i==0&&str[0]!='_')
	{str[0]=str[0]-32 ;
	}
		
		
		
		
		
		
		if(str[i]=='_'&&i!=l-1&&str[i+1]!='_')
		{
			str[i+1]=str[i+1]-32 ;
		}
		
		
		
	}
	cout<<str;
	
   // cout << "Hello World!" << endl;
    return 0;
}
