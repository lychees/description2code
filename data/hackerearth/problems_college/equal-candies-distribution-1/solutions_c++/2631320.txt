#include <bits/stdc++.h>
using namespace std;

int main()
{
   int i,t,n,ans,mini,diff;
   scanf("%d",&t);
   vector<int> cal(1001);
   cal[1]=1;
   cal[2]=1;
   cal[3]=2;
   cal[4]=2;
   cal[5]=1;
   for(i=6;i<=1000;i++)
   	cal[i]=1+cal[i-5];
   while(t--)
   {
   		ans=0;
   		scanf("%d",&n);
   		vector<int> v(n);
   		for(i=0;i<n;i++)
   			scanf("%d",&(v[i]));
   		sort(v.begin(),v.end());
   		mini=v[0];
   		for(i=1;i<n;i++)
   		{
   			if(v[i]!=mini)
   			{
   				diff=v[i]-mini;
   				ans+=cal[diff];
   			}
   		}
   		printf("%d\n",ans);
   }
    return 0;
}
