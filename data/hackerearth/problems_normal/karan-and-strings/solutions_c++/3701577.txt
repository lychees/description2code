#include <bits/stdc++.h>
using namespace std;
int main()
{
	

	
	long int t,x;
	string s,v;
	
	cin>>t;
	
	while(t--)
	
	{
		cin>>s;
		
		v="";
		
		for(x=0;x<s.size()-1;x++)
		
		{
			if(s[x] == s[x+1])
			continue;
			
			v=v+s[x];
			
		}
		
		cout<<v+s[s.size()-1]<<endl;
	}
	
	return 0;
}
