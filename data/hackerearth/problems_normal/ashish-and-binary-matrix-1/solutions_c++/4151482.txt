#include <bits/stdc++.h>
using namespace std;
#define lli long long int
int main()
{
	lli n,m,t;
	vector<string> v;
	cin>>t;
	while(t--)
	{
		cin>>n>>m;
		string s;
		for(lli i=0;i<n;i++)
		{
			cin>>s;
			v.push_back(s);
		}
		int flag=0;
		for(int i=0;i<n;i++)
		{
			if(flag==1)
			break;
			for(int j=i+1;j<n;j++)
			{
				if(v[i]==v[j])
				{
					flag=1;
					break;
				}
			}
		}
		if(flag==1)
		cout<<"No"<<endl;
		else
		cout<<"Yes"<<endl;
		v.clear();
		
	}
    
}
