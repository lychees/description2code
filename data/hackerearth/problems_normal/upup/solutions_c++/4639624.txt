#include <iostream>
#include <string>
using namespace std;

int main()
{
    std::ios::sync_with_stdio(false);
    string s;
    std::getline (cin,s);
    int l=s.length(),i=0;
    if(s[0]>=97)
    s[0]-=(97-65);
    while(i<l){
    	if(s[i]==' '&&s[i+1]>=97)
    	s[i+1]=s[i+1]-(97-65);
    	i++;
    }
    cout <<s<< endl;
    return 0;
}
