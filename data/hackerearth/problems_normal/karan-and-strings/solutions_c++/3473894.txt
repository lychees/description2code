#include <iostream>
using namespace std;

int main()
{
    
    long T;
    string s;
    char c;
    cin>>T;
    while(T--){
    	cin>>s;
    	c=s[0];
    	cout<<c;
    	for(long i=1;i<s.size();i++){
    		if(c!=s[i]){
    			c=s[i];
    			cout<<c;
    		}
    	}
    	cout<<endl;
    }
    return 0;
}
