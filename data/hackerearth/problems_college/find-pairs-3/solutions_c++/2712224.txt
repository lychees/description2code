#include<bits/stdc++.h>
using namespace std;
long long int arr[2000001];
int main()
{
ios_base::sync_with_stdio(false);
int t;
cin>>t;
while(t--)
{

    long long n;
    cin>>n;
    int i,x;
    for(i=0;i<2000001;i++)
        arr[i]=0;
    for(i=0;i<n;i++)
    {
        cin>>x;
        arr[x+1000000]++;
    }
    long long ans=0;
    for(i=0;i<2000001;i++)
    {
        ans+=(arr[i]*(arr[i]+1))/2;
    }
cout<<ans<<endl;
}
return 0;
}
