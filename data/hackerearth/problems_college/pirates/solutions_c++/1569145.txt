#include <bits/stdc++.h>
#define si(n) scanf("%d",&n);
#define pi(n) printf("%d\n",n);
#define pl(n) printf("%lld\n",n);
#define sl(n) scanf("%lld",&n);
#define pd(n) printf("%lf\n",n);
#define ss(s) scanf("%s",s);
#define ps(s) printf("%s\n",s);
#define ll long long
#define mod 1000000007
#define pb push_back
#define maxn 100005
int arr[maxn];
using namespace std;
int main()
{
    int n;
    si(n);
    for(int i=0;i<n;i++)
        si(arr[i]);
    int ans=1;
    int lans=1;
    for(int i=0;i<n-1;i++)
    {
        if(abs(arr[i]-arr[i+1])==1)
            lans++;
        else
            lans=1;
        ans=max(ans,lans);
        //pi(ans);
    }
    pi(ans);
    return 0;
}
