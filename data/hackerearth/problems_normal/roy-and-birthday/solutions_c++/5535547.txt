li inv(li a, li b) {
  if(a == 0) 
    return 0;  
  li ret = 1;
  while(b) {
    if(b&1) {
      ret = (ret*a)%mod;
    }
    b/=2;
    a = (a*a)%mod;
  }
  return ret;
}
