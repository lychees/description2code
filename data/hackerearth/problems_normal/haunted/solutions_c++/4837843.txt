
#include<bits/stdc++.h>
using namespace std;
int main()
{
    priority_queue<pair<long long,long long>> q;
    long long n;
    long long m;
    pair<long long,long long> x;
    scanf("%lld %lld",&n,&m);
    map<long long,long long> a;
    for(int i=0;i<n;i++){
    	long long aa,w,y;
    	scanf("%lld",&aa);
    	a[aa]++;
    	q.push(make_pair(a[aa],aa));
    	x=q.top();
    	w=x.second;
    	y=x.first;
    	printf("%lld %lld\n",w,y);
    }
    return 0;
}
