#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin >> n;
    vector <int> a(10005);
    for(int i = 0; i < n; i++) {
    	cin >> a[i];
    }
    
    sort(a.begin(), a.begin() + n);
    reverse(a.begin(), a.begin() + n);
    
    int sum;
    cin >> sum;
    int i = 0;
    while(sum > 0) {
    	sum = sum - a[i];
    	i++;
    }
    
    cout << i << endl;
    return 0;
}
