int dp[155][155][55];
bool vis[155][155][55];
int tot,cc;
int func(int c, int j, int k)
{
	if(c==0) return 0;
	int i=tot-8*(cc-c)-5*j-10*k;
	if(c*8>i+j*5+k*10) return 1000000;
	if(vis[c][j][k]==1) return dp[c][j][k];
	vis[c][j][k]=1;
	int ans=1000000;
	if(i>=8) ans=min(ans,func(c-1,j,k)+8);
	if(i>=3 and j>=1) ans=min(ans,func(c-1,j-1,k)+4);
	if(j>=2) ans=min(ans,func(c-1,j-2,k)+2);
	if(k>=1) ans=min(ans,func(c-1,j,k-1)+1);
	if(i>=3 and k>=1) ans=min(ans,func(c-1,j+1,k-1)+4);
	if(ans>=1000000) ans= 1000000;
	return dp[c][j][k]=ans;
}

int main()
{
	int t,c,n1,n5,n10;
	sc(t);
	while(t--)
	{
		sc(c,n1);
		sc(n5,n10);
		memset(vis,0,sizeof vis);
		int ans=0;
		cc=c;
		tot=n1+5*n5+10*n10;
		if(tot<c*8)
		{
			prl(-1); continue;
		}
		ans=func(c,n5,n10);
		if(ans==1000000) ans=-1;
		prl(ans);
	}
}
