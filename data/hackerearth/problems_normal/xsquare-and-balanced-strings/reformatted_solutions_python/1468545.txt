'''
# Read input from stdin and provide input before running code

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
t = input()
while t:
    s = raw_input()
    d = dict()
    for c in s:
        if c not in d:
            d[c] = 1
        else:
            d[c] += 1
    flag = False
    for k in d:
        if d[k] % 2 == 1:
            flag = True
    if flag:
        print -1
    else:
        print 1
    t -= 1
