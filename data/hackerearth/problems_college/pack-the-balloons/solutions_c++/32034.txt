#include<iostream>
#include<vector>
#include<algorithm>
#include<cstdio>
using namespace std;
int main()
{
    int test, i;
    long long int num, num1, num2, sol;
    vector<long long int> vec;
    cin>>test;
    for(i = 0; i<test; i++)
    {
        scanf("%lld", &num);
        vec.push_back(num);
        scanf("%lld", &num1);
        vec.push_back(num1);
        scanf("%lld", &num2);
        vec.push_back(num2);
        sort(vec.begin(), vec.end());
        sol = (vec[0]/2);
        if(vec[0]%2!=0)
        printf("%lld.5\n",sol);
        else
        printf("%lld.0\n",sol);
        vec.clear();
    }
}
