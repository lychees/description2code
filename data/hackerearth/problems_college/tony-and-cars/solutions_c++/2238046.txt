#include <iostream>
#define aa long long
using namespace std;
int main()
{
    ios_base::sync_with_stdio(false);
    aa int tc;
    cin>>tc;
    while(tc--)
    {
        aa int n;
        cin>>n;
        if(n<=9) cout<<n<<endl;
        else if(n>=10 && n<=99) cout<<(11+(2*(n-10)))<<endl;
        else if(n>=100 && n<=999) cout<<(192+(3*(n-100)))<<endl;
        else if(n>=1000 && n<= 9999) cout<<(2893+(4*(n-1000)))<<endl;
        else if(n>=10000 && n<= 99999) cout<<(38894+(5*(n-10000)))<<endl;
        else if(n>=100000 && n<= 999999) cout<<(488895+(6*(n-100000)))<<endl;
        else if(n>=1000000 && n<= 9999999) cout<<(5888896+(7*(n-1000000)))<<endl;
        else if(n>=10000000 && n<= 99999999) cout<<(68888897+(8*(n-10000000)))<<endl;
        else if(n>=100000000 && n<= 999999999) cout<<(788888898+(9*(n-100000000)))<<endl;
        else if(n>=1000000000 && n<= 9999999999) cout<<(8888888899+(10*(n-1000000000)))<<endl;
        else if(n>=10000000000 && n<= 99999999999) cout<<(98888888900+(11*(n-10000000000)))<<endl;
        else if(n>=100000000000 && n<= 999999999999) cout<<(1088888888901+(12*(n-100000000000)))<<endl;
        else if(n>=1000000000000 && n<= 9999999999999) cout<<(11888888888902+(13*(n-1000000000000)))<<endl;
    }
}
