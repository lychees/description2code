#include <iostream>
#include<cstdio>
#include <vector>
using namespace std;


int main(){
    int t,n,m,f,s,flag;
    scanf("%d",&t);
    while(t--){
        scanf("%d %d" ,&n,&m);
    	vector<vector<int> >vec(2,vector<int> (n));
    	for(int i=0;i<n;i++) scanf("%d",&vec[0][i]);
    	f=0;
    	s=1;
    	int flag=1;
    	while(m-- && flag==1)
    	{
    	flag=0;
    	for(int i=0;i<n;i++){
    			if(vec[f][i]==1)
    			flag=1;
       		}
    	if(vec[f][1]==1)
    	vec[s][0]=1;
    	else
    	vec[s][0]=0;
    	for(int i=1;i<(n-1);i++){
    		if(vec[f][i-1]==1 && vec[f][i+1]==1)
    		vec[s][i]=1;
    		else
    		vec[s][i]=0;
    	}
    	if(vec[f][n-2]==1)
    	vec[s][n-1]=1;
    	else
    	vec[s][n-1]=0;
    		if(f==0)
    		f=1;
    		else
    		f=0;
    		if(s==0)
    		s=1;
    		else
    		s=0;
    	}
    	for(int i=0;i<n;i++) printf("%d ",vec[f][i]);
    	printf("\n");
    }
    return 0;
}
