#include <bits/stdc++.h>
using namespace std;

#define mod 1000000007

int main()
{
    int t;
    
    cin>>t;
    
    while(t--)
    {
    	long long a[3];
    	
    	cin>>a[0]>>a[1]>>a[2];
    	
    	sort(a,a+3);
    	
    	long long ans=0;
    	
    	ans=a[0];
    	
    	ans=(ans%mod *(a[1]-1)%mod)%mod;
    	
    	ans=(ans%mod *(a[2]-2)%mod)%mod;
    	
    	cout<<ans<<endl;
    }
}
