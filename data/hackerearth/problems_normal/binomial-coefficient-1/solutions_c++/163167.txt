#include<iostream>
#define MOD 1000000007
#define MAX 1000005
using namespace std;
//------------------------------------

long long power(long long n,int m);
long long fact,ifactk,ifactn,ifac;
//-----------------------------------------------------------------------------
void preProcess(int n,int k)
{
	fact=1;
	ifac=1;
	int i;
	for(i=1;i<=n;i++)
	{
		fact=fact*i;
		fact%=MOD;
		ifac = (power(fact,MOD-2))%MOD;
		if(i==k)
		ifactk=ifac;
	    if(i==n-k)
		ifactn=ifac;
		}
}
//-----------------------------------------------------------------------------

long long int comb(int a,int b)
{
	return (((fact*ifactk)%MOD)*ifactn)%MOD;
}
//-----------------------------------------------------------------------------

long long power(long long n,int m)
{
		
    if(m==0) return 1;
    
    long long x=power(n,m/2);
    if(m%2==0) 
               return (x*x)%MOD;
    else
               return (((x*x)%MOD)*n)%MOD;
}
//-----------------------------------------------------------------------------
	
int main()
{
	int a,b,n,k;
	cin>>a>>b>>n>>k;
	preProcess(n,k);
	long long ans;
	ans=comb(n,k);
	//cout<<ans;
	ans%=MOD;
	ans=ans*(power(a,k)%MOD);
	ans%=MOD;
	ans=ans*(power(b,n-k)%MOD);
	ans%=MOD;
	cout<<ans<<endl;
	
}
