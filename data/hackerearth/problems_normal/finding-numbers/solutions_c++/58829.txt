#include<iostream>
#include<math.h>
using namespace std;
int main()
{
	unsigned long long t,z,m,n,j1,ans,i,j;
	cin>>t;
	for(i=0;i<t;i++)
	{
		ans=0;
		cin>>z;
		cin>>m;
		cin>>n;
		for(j=1;j<=sqrt(z);j++)
		{
			if(z%j==0)
			{
			if(j==sqrt(z))
			{
				j1=z/j;
				if(j<=m&&j1<=n && j<=n &&j1<=m)
				ans=ans+1;
				//else if(j<=m&&j1<=n || j<=n &&j1<=m)
				//ans=ans+1;
			
			}
			else
			{
				j1=z/j;
				if(j<=m&&j1<=n && j<=n &&j1<=m)
				ans=ans+2;
				else if(j<=m&&j1<=n || j<=n &&j1<=m)
				ans=ans+1;
			}
			}
		}
		cout<<ans<<endl;
	}
	return 0;
}
