#include<bits/stdc++.h>
using namespace std;
const int maxn=3e5+7;
double x[maxn],y[maxn],f[maxn];
double dp[maxn];
int main(){
	int n;
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>x[i]>>y[i]>>f[i];
	}
	dp[0]=f[0];
	for(int i=1;i<n;i++){		
		double temp=(x[0]-x[i]),temp1=y[0]-y[i];
		double temp2=((temp*temp)+(temp1*temp1));
		temp2=sqrt(temp2);
		dp[i]=dp[0]+f[i]-temp2;
		for(int j=1;j<i;j++){
			double temp=(x[j]-x[i]),temp1=y[j]-y[i];
			double temp2=((temp*temp)+(temp1*temp1));
			temp2=(double)sqrt(temp2);
			dp[i]=max(dp[i],dp[j]+f[i]-temp2);
		}
	}
	printf("%0.6lf\n",dp[n-1]);
}
