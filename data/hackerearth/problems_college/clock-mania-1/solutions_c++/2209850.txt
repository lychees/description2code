int calcAngle(double h, double m)
{
    if (h == 12) h = 0;
    if (m == 60) m = 0;
 	int hour_angle = 0.5 * (h*60 + m);
    int minute_angle = 6*m;
 	int angle = abs(hour_angle - minute_angle);
 	angle = min(360-angle, angle);
 	return angle;
}
int main()
{
    int testCases;
    scanf("%d", &testCases);
    for(int i=0;i<testCases;i++)
    {
    	int a, b;
    	scanf("%d%d", &a, &b);
    	printf("%d \n", calcAngle(a, b));
    }
    return 0;
}
