#include<iostream>
#include<stdio.h>
using namespace std;
void quicksort(int arr[],int beg,int last){
	if(beg<last)
	{
    int pivot=arr[last],count=beg,temp,i;
		//partiton
		for(i=beg;i<last;i++)
		{	if(arr[i]<=pivot)
			{
				temp=arr[i];
				arr[i]=arr[count];
				arr[count]=temp;
				count++;
			}
		}
				temp=arr[last];
				arr[last]=arr[count];
				arr[count]=temp;
		//recursion
			quicksort(arr,beg,count-1);
			quicksort(arr,count+1,last);
	}
}

int main(){
    int n,i,j;scanf("%d",&n);
    int arr[n];
    for(i=0;i<n;i++)
        scanf("%d",&arr[i]);
    quicksort(arr,0,n-1);
    int uplim=arr[0],res=0;
    for(i=2;i<=uplim+100000000;i++)
    {   res=arr[0]%i;
        for(j=1;j<n;j++)
        {   if(arr[j]%i!=res)
                break;
        }
        if(j==n)
            printf("%d ",i);
    }
}
