#include<iostream>
#include<sstream>
#include<stack>
#include<stdio.h>
#include<string>
using namespace std;
int main(){
	long int num[101],N;
	while(scanf("%d", &N) != EOF)	{
		int ele[100000] = {1};
		stack<int> st;
		st.push(0);
		string ps,out;
		getchar();
		getline(cin,ps);
		stringstream ss(ps);
		int temp,i=0;
		while(ss>>temp) {
			num[i] = temp;
			i++;
		}
		int flag=0;
		if(i == 0) out="";
		for(int j=0;j<i;j++) {	
			if (num[j] > N) {
				out = "-1";
				break;
			}
			if(st.empty() ? 0 : st.top() >= num[j]) {
				while(st.empty() ? 0 : st.top()>num[j]) {
					if(!st.empty()) {
						st.pop();
						out += "K";
					}
				}
			} else {
				flag = 0;
				for(int x=st.empty() ? 1 : st.top()+1; x<=num[j];x++) {
					if(ele[x] == -1){
						continue;
					}
					flag =1;
					st.push(x);
					ele[x] = -1;
					out += "C";
				}
			}
			if(!st.empty()) {
				 if(st.top() != num[j]) {
				 	out = "-1";
				 	break;
				 }
				st.pop();
				out += "K";
			}
			
		}
		if(flag == 0 && out != "") out = "-1";
		cout<<out<<endl;
	}
	return 0;
}
