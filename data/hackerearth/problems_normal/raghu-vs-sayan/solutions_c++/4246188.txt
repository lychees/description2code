#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
 	long long int T,i,j,N,A,B,R,S,num,RO,SO;
    string output;
    cin>>T;
    for(i=0;i<T;i++)
    {
        RO=SO=0;
        vector<long long int>V;
        cin>>A>>B>>N;
        for(j=0;j<N;j++)
        {
            cin>>num;
            V.push_back(num);
        }
        sort(V.begin(),V.end());
        for(j=0;j<N;j++)
        {
            if(V[j]<=A)
            {A=A-V[j];RO++;}
            if(V[j]<=B)
            {B=B-V[j];SO++;}
        }
        if(RO>SO)
            output+="Raghu Won\n";
        else if(RO<SO)
            output+="Sayan Won\n";
        else
            output+="Tie\n";
    }
    cout<<output;
    return 0;
}
