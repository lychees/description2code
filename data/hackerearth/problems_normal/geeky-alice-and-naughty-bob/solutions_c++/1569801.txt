#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    int t,s,i; ll a,b; cin>>t; int arr[6]={0,1,2,6,6,3};
    while(t--)
    {
        cin>>a>>b;
        if(a>=6 && b>=6)
            cout<<(b-a+1)*9<<endl;
        else if(a<6 && b>=6)
        {     s=0;
            for(i=a;i<=5;i++)
                s+=arr[i];
             cout<<s+(b-6+1)*9<<endl;}
             else if(a<6 && b<6)
             {
                  s=0;
                  for(i=a;i<=b;i++)
                    s+=arr[i];
                  cout<<s<<endl;
             }
    }
    return 0;
}
