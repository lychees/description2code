#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
ll n,k;
#define M 1000000007
ll powd(ll a,ll b, ll c)   // power function calculates power with mod in log n complexity
{
	if(b==0)
		return 1;
	else if(b%2==0)
		return powd((a*a)%c,b/2,c)%c;
	else
		return (a*(powd((a*a)%c,b/2,c)))%c;
}
int main()
{
	int t,i;
	scanf("%d",&t);
	while(t--)
	{
	
		scanf("%lld%lld",&n,&k);
		ll ans=n;
		ll sum=1;		
        for(i=1;i<=k;i++)
		{
			sum=(sum+ans)%M;
			ans=(  ((ans*(n-i))%M)*powd(i+1,M-2,M) )%M;	
						
		}
		printf("%lld\n",sum);
	}
	return 0;
}
