inline void in(int &n)
{
    n=0; int ch = getcx(); int sign = 1;
    while(ch < '0' || ch > '9') { if(ch == '-') sign=-1; ch = getcx(); }
    while(ch >= '0' && ch <= '9') { n = (n << 3) + (n << 1) + ch - '0', ch = getcx(); }
    n = n * sign;
}
int d[10004],state[10004];
int main()
{
    int t,n,i,j,cost,a[10004];
    vector< pii > adj[10004];
    in(t);
    while(t--)
    {
        in(n);
        for(i=1; i<=n; i++)
        {
            in(a[i]);
            state[i]=temp;
            d[i]=(2*INF)+5;
            adj[i].clear();
        }
        //printf("%d\n",d[1]);
        for(i=1; i<=n; i++)
        {
            for(j=1; j<=n; j++)
            {
                adj[i].push_back(pii(j,a[i]+a[j]));
                adj[j].push_back(pii(i,a[i]+a[j]));

            }
        }
        LL ans=0;
        int node,wt,v;
        priority_queue< pii, vector< pii >, comp > Q;
        Q.push(pii(0,1));
        d[1]=0;
        while(!(Q.empty()))
        {
            v=Q.top().second;
            cost=(Q.top().first);
            //printf("ans is %lld\n",ans);
            //printf("it is %d %d\n",v,d[v]);
            Q.pop();
            if(state[v]==perm)
                continue;
            state[v]=perm;
            ans=ans+cost;
            for(i=0; i<adj[v].size(); i++)
            {
                node=adj[v][i].first;
                wt=adj[v][i].second;
                if(wt<d[node])
                {
                    d[node]=wt;
                    Q.push(pii(d[node],node));
                }
            }
        }
        printf("%lld\n",ans);
    }
    return 0;
}
