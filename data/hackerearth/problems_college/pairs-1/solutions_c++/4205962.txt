#include <bits/stdc++.h>
int pair(int n);
int arr[500010];
int sarr[500010];
int main()
{
		arr[1]=1;
		int t,i,n,sum;
		sum=0;
		for(i=1;i<500010;i++)
		{
			t=pair(i);
			sum+=t;
			sarr[i]=sum;
		}		
 
		scanf("%d",&t);
		while(t--)
		{
				sum=0;
				scanf("%d",&n);
				printf("%d\n",sarr[n]);
 
		}
		return 0;
}
 
 
int pair(int n)
{
	
	if(arr[n]==0)
	{
		int i;
		int root;
		int count=2;
		root=sqrt(n);
		for(i=2;i<=root;i++)
		{
			if(n%i==0) count+=2;
			if(n==i*i) count--;
		}
		arr[n]=count;
	}
	return arr[n];
	
}
