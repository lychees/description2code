#include <iostream>
using namespace std;

int main()
{
    int T,K,N,M,i,j;
    cin>>T;
    while(T--)
    {
    	cin>>N;
    	cin>>K;
    	cin>>M;
    	string *str=new string[N];
    	for(i=0;i<N;i++)
    	{
    		cin>>str[i];
    	}
    	for(j=0;j<N;j++)
    	{
    		for(i=0;i<N-1;i++)
    		{
    			int r=str[i].compare(0,M-1,str[i+1],0,M-1);
    			if(r>0)
    				swap(str[i],str[i+1]);
    		}
    	}
    	cout<<str[K-1]<<"\n";
    }
    return 0;
}
