#include <iostream>
using namespace std;
int f(int n,int m){
	int arr[n+1];
	arr[0]=1;
	arr[1]=1;
	for(int i= 2;i<n+1;i++){
		arr[i]=0;
		for(int j=1;j<=m && j<=i;j++){
			arr[i]+=arr[i-j];
		}
	}
	return arr[n];
}
int main()
{
    int n;
    cin>>n;
    if(!(n>0)){
    	cout<<"ERROR"<<endl;
    }
    else{
    cout<<"Jack-"<<f(n,3)<<endl;
    cout<<"Jill-"<<f(n,2)<<endl;
	
    }
        return 0;
}
