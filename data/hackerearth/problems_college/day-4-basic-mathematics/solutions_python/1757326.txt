from math import ceil
from math import sqrt

def who_is_fighting(battle):
	if battle%3==0:
		print 'SL'
	if battle%3==1:
		print 'LB'
	if battle%3==2:
		print 'BS'

def from_day_to_battle(day):
	return ceil((-1+sqrt(1+8*day))/2)

t = int(raw_input())
iterations = 0
while iterations < t:
        iterations+=1
        day = int(raw_input())
        who_is_fighting(from_day_to_battle(day))
