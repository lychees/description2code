#include <stdio.h>

int main()
{
    int n;
    scanf("%d",&n);
    while(n--)
    {
    	int m;int co=0;
    	scanf("%d",&m);
    	int a[m];
    	for(int i=0;i<m;i++)
    	{
    		scanf("%d",&a[i]);
    	}
    	for(int i=0;i<m;i++)
    	{
    		for(int j=i+1;j<m;j++)
    		{
    			if(a[i]!=a[j])
    			{
    				if((a[i]+a[j])%2==0)
    				co++;
    			}
    		}
    	}
    	printf("%d\n",co);
    }
}
