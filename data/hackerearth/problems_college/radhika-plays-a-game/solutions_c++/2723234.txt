#include <iostream>
using namespace std;

int main() {
	// your code goes here
	int t;
	cin>>t;
	while(t--)
	{
		int n,k;
		cin>>n>>k;
		if(k==1)
		{
			if(n%2==0)cout<<"Rani"<<endl;
			else cout<<"Nandu"<<endl;
		}
		else if(n==1)
		cout<<"Nandu"<<endl;
		else
		{
			if(n<=k+1)cout<<"Rani"<<endl;
			else if(n==k+2)cout<<"Nandu"<<endl;
			else if((n-(k+2))%(k+1)==0)
			{
				cout<<"Nandu"<<endl;
			}
			else
			cout<<"Rani"<<endl;
		}
		
		
	}
	return 0;
}
