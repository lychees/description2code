#include<bits/stdc++.h>
using namespace std ;
int main()
{
	int t ;
	cin >> t ;
	int counter ;
	int x ;
	int y ;
	while ( t-- )
	{
		cin >> x ;
		counter = 0 ;
		while ( x-- )
		{
			cin >> y ;
			if( y % 2 == 0)
			{
				counter++ ;
			}
		}
		cout<<counter<<"\n" ;
	}
	return 0 ;
}
