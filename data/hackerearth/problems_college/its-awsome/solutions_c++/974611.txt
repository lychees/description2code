#include<iostream>
#include<string.h>
#include<stdio.h>
using namespace std;
char s[20];
int main()
{
     int t;
	 scanf("%d",&t);
	 while(t--)
	 {
	     scanf("%s",s);
		 int len=strlen(s);
		 int a=0;
		 int b=0;
		 for(int i=0;i<len;i++)
		 {
		     if(s[i]=='4')
			 a++;
			 else if(s[i]=='7')
			 b++;
			 
		 }
		 if((a+b==4)||(a+b==7))
		 cout<<"YES"<<endl;
		 else
		 cout<<"NO"<<endl;
	 }
	 return 0;
}
