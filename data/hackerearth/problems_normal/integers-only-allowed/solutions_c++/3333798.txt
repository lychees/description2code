#include <iostream>
#include <cstdio>
#include <vector>
#include <cmath>
using namespace std;
int gcd(int a, int b)
{
	if(b == 0) return a;
	else return gcd(b,a%b);
}
vector <long double> x;
vector <long double> y;
int main()
{
    int N;
    scanf("%d", &N);
    double X, Y;
    for(int i=0;i<N;i++)
    {
    	scanf("%lf %lf", &X, &Y);
    	x.push_back(X);
    	y.push_back(Y);
    }
   long double area = 0.0;
   long double b = 0.0;
    for(int i =0;i<N-1;i++)
    {
    	area += x[i]*y[i+1] - y[i]*x[i+1];
    	b += gcd(abs(x[i+1] - x[i]),abs( y[i+1] -y[i]));
    }
    area += x[N-1]*y[0] - y[N-1]*x[0];
    area = abs(area);
    area /= 2;
   
    b += gcd(abs(x[0] - x[N-1]),abs( y[0] - y[N-1]));
    long long int ans = area - b/2 + 1;
    printf("%lld", ans);
    return 0;
}
