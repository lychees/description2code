using namespace std;

const int maxk = 10000 + 15;
long long n, m, ggc, ans, xx, dd, x, y;
long long aa[maxk];
int kk;
long long gcd(long long a, long long b, long long &x, long long &y) {
  if (b == 0) {
    x = 1;
    y = 0;
    return a;
  }
  long long dd = gcd(b, a % b, x, y);
  long long tmp = x;
  x = y, y = tmp - a / b * y;
  return dd;
}
int main() {
  scanf("%I64d%I64d%d", &n, &m, &kk);
  scanf("%I64d", &ggc);
  aa[0] = ggc;
  ans = 0;
  if (ggc > n) {
    printf("NO\n");
    return 0;
  }
  for (int i = 1; i < kk; i++) {
    scanf("%I64d", &xx);
    aa[i] = xx;
    dd = gcd(ggc, xx, x, y);
    if ((-i - ans) % dd != 0) {
      printf("NO\n");
      return 0;
    }
    x *= (-i - ans) / dd;
    x %= (xx / dd);
    ans += x * ggc;
    ggc = ggc / dd * xx;
    if (ggc > n) {
      printf("NO\n");
      return 0;
    }
    ans %= ggc;
    ans = (ans + ggc) % ggc;
  }
  if (ans == 0)
    ans = ggc;
  if (ans + kk - 1 > m) {
    printf("NO\n");
    return 0;
  }
  for (int i = 0; i < kk; i++)
    if (gcd(ggc, ans + i, x, y) != aa[i]) {
      printf("NO\n");
      return 0;
    }
  printf("YES\n");
  return 0;
}
