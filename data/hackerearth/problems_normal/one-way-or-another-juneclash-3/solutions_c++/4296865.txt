#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long n;
		cin>>n;
		long long r,c,x;
		vector<bool> v0,v1;
		for(long i=0;i<n;i++)
		{
			cin>>r>>c>>x;
			if((r+c)%2==0)
			v0.push_back(x%2);
			else
			v1.push_back(x%2);
		}
		v0.erase(unique(v0.begin(),v0.end()),v0.end());
		v1.erase(unique(v1.begin(),v1.end()),v1.end());
		if(v0.size()==1 && v1.size()==1 && v0[0]!=v1[0])
		cout<<"Yes"<<endl;
		else
		cout<<"No"<<endl;
	}
	return 0;
}
