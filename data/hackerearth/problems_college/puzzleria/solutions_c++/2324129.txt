#include <iostream>
using namespace std;
int fact(int n)
{
	if(n==0)
		return 1;
	else
		return(n*fact(n-1));
}
int main()
{
    //cout << "Hello World!" << endl;
    int test,c,v,temp,temp1;
    cin>>test;
    for(;test;--test)
    {
    	int ct=5040;
    	int cv=120;
    	cin>>c>>v;
    	temp=fact(c);
    	temp1=fact(7-c);
    	temp*=temp1;
    	ct/=temp;
    	temp=fact(v);
    	temp1=fact(5-v);
    	temp*=temp1;
    	cv/=temp;
    	ct*=cv;
    	temp=fact(c+v);
    	ct*=temp;
    	cout<<ct<<endl;
    }
    return 0;
}
