#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int ans=0;
    	int a[50005],c,n;
    	cin>>a[0]>>a[1]>>c>>n;
    	if(n==1 && a[0]!=0)
    	{
    	cout<<"NO\n";
    	continue;
    	}
    	if(n==1 && a[0]==0)
    	{
    	cout<<"YES\n";
    	continue;
    	}
    	ans=(a[0]*10+a[1])%41;
    	for(int i=2;i<n;i++)
    	{
    		a[i]=((a[i-1]*c)+(a[i-2]))%10;
    		ans=(ans*10+a[i])%41;
    	}
    	if(ans==0)
    	{
    		cout<<"YES\n";
    	}
    	else
    	cout<<"NO\n";
    }
}
