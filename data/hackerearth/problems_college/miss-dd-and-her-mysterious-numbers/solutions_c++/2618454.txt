#include <bits/stdc++.h>
using namespace std;

long long hell[10000000];
// marks all mutiples of 'a' ( greater than 'a' but less than equal to 'n') as 1.
void markMultiples(bool arr[], int a, int n)
{
    int i = 2, num;
    while ( (num = i*a) <= n )
    {
        arr[ num-1 ] = 1; // minus 1 because index starts from 0.
        ++i;
    }
}

int main()
{
    long long  no,qo,ko;

    cin >> no >> qo;
  vector<int> size1;
  size1.resize(no+1, 0);

  for( long long i=2; i <= no; i+=2 )
    { size1[i]=1; }

  for( long long i=3; i <= no; i+=2 )  {
      if( size1[i]==0 )  {
          for( long long j=i; j <= no; j+=i ) { size1[j]+=1; }
      }
  }
  for( long long i=2; i <= no; ++i )  {
      hell[size1[i]]++;
  }
    while(qo--)
    {
    	cin >> ko;
    	if(ko>no)
    	cout << "0\n";
    	else
    	cout << hell[ko] << "\n";
    }
    return 0;

}
