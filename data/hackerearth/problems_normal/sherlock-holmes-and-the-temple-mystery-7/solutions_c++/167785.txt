#include <iostream>
#include <cmath>
#include <cstring>

#define MAXN 100

int vals[MAXN];

using namespace std;

int added;
int nadd;
int total;
int n;
int mid;
bool q[MAXN][10000];

int main()
{
	int t;
	scanf("%d",&t);

	while(t--)
	{
		total = 0;
		scanf("%d",&n);
		memset(q,0,sizeof(bool)*MAXN*10000);
		for(int i=0;i<n;i++)
		{
			scanf("%d",&vals[i]);
			total+=vals[i];
		}

		added = 0;
		nadd = 0;
		if((total&0x01)==0)
		{
			int mid = total/2;
			for(int i=1;i<=mid;i++)
				q[1][i] = (vals[0]==i);
			//if(vals[0]==mid)
			//	q[1][mid]=true;
			//else
			//	q[1][mid]=false;
			for(int i=2;i<=n;i++)
			{
				for(int k=1;k<=mid;k++)
				{
					q[i][k] = ((q[i-1][k]) || (vals[i-1]==k));
				       	if((k-vals[i-1])>=1)
						q[i][k] = (q[i][k] || (q[i-1][k-vals[i-1]]));

					//std::cout<<"q["<<i<<"]["<<k<<"] = "<<q[i][k]<<std::endl;
				}
			}
			
			if(q[n][mid])
				printf("YES\n");
			else
				printf("NO\n");
		}
		else
			printf("NO\n");
	}

	return 0;
}
