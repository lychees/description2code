#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
vector<long long> f(1000000 + 2,1);
char qq[1000006],aa[1000006];
inline void inp( long long  &n ) {
    n=0; int ch = getchar(); int sign = 1;
    while(ch < '0' || ch > '9') { if(ch == '-') sign=-1; ch = getchar(); }
    while(ch >= '0' && ch <= '9') { n = (n << 3) + (n << 1) + ch - '0', ch = getchar(); }
    n = n * sign;
    }
long long  pow1(long long  a,long long  b)
{
	if(b==0)
	return 1;
	if(b==1)
	return a;
	long long  temp=pow1(a,b/2);
	long long  x;
	if(b%2==0)
		x=(temp*temp)%mod;
	else
		x=(((temp*temp)%mod)*a)%mod;
	return x;
}

long long Cnr(long long  n, long long  r)
{
    return (f[n]*((pow1(f[r],mod-2) * pow1(f[n-r],mod-2)) % mod)) % mod;
}

int main()
{
    long long  i;
	for (int i=2; i<=1000000;i++)
        f[i]= (f[i-1]*i) % mod;
	long long  t;
	inp(t);
	while(t--)
	{
		qq[0]='\0';
		aa[0]='\0';
		scanf("%s",qq);
		scanf("%s",aa);
		vector<long long > c1(26,0);
		vector<long long > c2(26,0);
		long long  l1=strlen(qq),l2=strlen(aa);
		for(long long  i=0;i<l1;i++)
			c1[qq[i]-'a']++;
			
		for(long long  i=0;i<l2;i++)
			c2[aa[i]-'a']++;
		
		long long ans=1;
		for(long long  i=0;i<26;i++)
		{
			if(c2[i]>c1[i])
			{
				ans=0;
				goto in;
			}
			ans=ans*Cnr(c1[i],c2[i]);
			if(ans>mod)ans%=mod;
			ans=ans*f[c2[i]];
			if(ans>mod)ans%=mod;
		}
		in:;
		printf("%lld\n",ans);
	}
}
