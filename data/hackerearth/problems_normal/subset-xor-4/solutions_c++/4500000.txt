#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    if(t>=1 && t<=100){
    	int ans[t];
    	for(int c=0;c<t;c++){
    		int sz;
    		cin >> sz;
    		int a[sz],sum;
    		for(int i=0;i<sz;i++){
    			cin >> a[i];
    			if(i==0)
					sum = a[i];
				else
					sum^=a[i];
    		}
    		for(int i=0;i<sz;i++){
    			for(int k=i+1;k<sz;k++){
    				sum^=(a[i]^a[k]);
    			}
    		}
    		ans[c]=sum;
    	}
    	for(int c=0;c<t;c++){
    		cout << ans[c] << endl;	
    	}
    }
    return 0;
}
