#include<bits/stdc++.h>
#define lli long long int
using namespace std;
int count(lli n){
	int cnt=0;
	while(n){
		n&=n-1;
		cnt++;
	}
	return cnt;
}
int main(){
	int t;
	ios::sync_with_stdio(0);
	cin>>t;
	assert(t>=1 && t<=100);
	while(t--){
		int n;
		cin>>n;
		assert(n>=1 && n<=1000000);
		vector<lli>v[100];
		lli a[n];
		int maxi=-1;
		for(int i=0;i<n;i++){
			cin>>a[i];
			int tmp=count(a[i]);
			v[tmp].push_back(a[i]);
			if(maxi<tmp)maxi=tmp;
		}
		for(int i=1;i<=maxi;i++)
			for(auto j=0;j<v[i].size();j++){
				if(i==maxi && j==v[i].size()-1)cout<<v[i][j]<<endl;
				else cout<<v[i][j]<<" ";
			}
	}
	return 0;
}
