#include<stdio.h>
#include<iostream>
using namespace std;

int main()
{
    long long t, cas=1, pd, pg;
    long long n, i, tempd, tempg, glosses, dlosses;

    scanf("%lld", &t);

    while(cas<=t)
    {
        scanf("%lld %lld %lld", &n, &pd, &pg);
        
        if((pg==100&&pd<100)||(pg==0&&pd>0))
        {
            printf("#%lld:Broken\n", cas++);
            continue;
        }
        
        i=n;
        tempd = i*pd;

        while(tempd%100!=0&&i>0)
        {
            i--;            //'i' is set to actual number of games played today
            tempd = i*pd;
        }
        
        if(i==0)
        {
            printf("#%lld:Broken\n", cas++);
            continue;
        }

        printf("#%lld:Possible\n", cas++);
    }


    return 0;
}
