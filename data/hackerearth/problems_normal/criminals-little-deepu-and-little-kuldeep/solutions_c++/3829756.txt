#include<bits/stdc++.h>
#define ll long long
#define sc(n) scanf("%d",&n)
#define scl(n) scanf("%lld",&n)
#define prt(n) printf("%d\n",n)
using namespace std;
int main()
{
//	freopen("in.txt","r",stdin);
	int t,n,count,max,i;
	ll arr[100001];
	sc(t);
	while(t--)
	{
		sc(n);
		for(i=0;i<n;i++)
		scl(arr[i]);
		sort(arr,arr+n);
		max=1,count=1;
		for(i=1;i<n;i++)
		{
			if(arr[i]==arr[i-1])
			{
				count++;
				if(count>max)
				max=count;
			}
			else
			count=1;
		}
		prt(max);
	}
return 0;
}
