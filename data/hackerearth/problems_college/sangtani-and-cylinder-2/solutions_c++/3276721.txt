#include<bits/stdc++.h>

using namespace std;
#define PI  3.14159265358979323846 
int main(){
	int t;
	cin >> t;
	while(t--){
		double d,d1;
		cin >> d;
		d1 = d;
		d = d / 2;
		d = d / PI;
		double p = pow(d,(double)1/(double)3);
		//cout << d << "..." << p <<endl;
		d = (double)(3 * d1) / (double)p;
		printf("%.9f\n",d);
	}
	return 0;
}
