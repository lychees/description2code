#include<iostream>
#include<cstdio>

using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
       long long int n;
        cin>>n;
       long long  int ans,minc;
        if(n%2)
        {
          ans=(n%2+n/2)  ;
          minc=n/2;
        }
        else {ans= n/2+1;
        minc=n/2-1;
        }
        cout<<ans<<" "<<minc<<endl;

    }


    return 0;
}
