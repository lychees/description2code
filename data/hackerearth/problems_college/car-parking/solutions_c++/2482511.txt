#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int arr[10000009];
int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    string s;
    cin>> s;
    memset(arr,0,sizeof(arr));
    for(int i=0;i<s.length();i++)
    {
        if(s[i]=='D')
            arr[i]=1;
        if(s[i]=='B')
        {
            arr[i]=1;
            if(i-1>=0)
            arr[i-1]=1;
            if(i-2>=0)
            arr[i-2]=1;
        }
        if(s[i]=='S')
        {
            arr[i]=1;
            if(i-1>=0)
                arr[i-1]=1;
            if(i+1<s.length())
                arr[i+1]=1;
        }
    }
    int c=0;
    //cout<< s;
    for(int i=0;i<s.length();i++)
    {
        if(arr[i]==0)
        c++;
    }
    printf("%d\n",c);
    }
    return 0;
}
