#define MAX 1000000

bool mark[MAX+1];
uint64 score[MAX+1];

void sieve() {
    fill(mark, mark+MAX+1, true);
    fill(score, score+MAX+1, 1);
    int c, num;
    for(uint64 i=2; i<=MAX; i++) {
        if(mark[i]) {
            score[i] = i-1;
            for(uint64 j=i+i; j<=MAX; j+=i){
                mark[j] = false;
                num = j; c = 1;
                while(num%i==0)num/=i, c*=i;
                c/=i;
                score[j] *= ((c) * (i-1));
            }
        }
    }
}
