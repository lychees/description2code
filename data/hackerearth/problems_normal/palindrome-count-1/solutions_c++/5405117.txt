#include <bits/stdc++.h>
using namespace std;

int main()
{
    int i,j,l,k,n,maxe=0;
    string s,r,b;
    cin>>s;
    l=s.length();
    for(i=0;i<l;i++)
    {
    	k=l-i;
    	while(k>0)
    	{
    		r=s.substr(i,k);
    		if(r!="")
    		{
    		b=r;
    		reverse(r.begin(),r.end());
    		if(b==r)
    		maxe++;
    		}
    	
    		k--;
    	}
    }
	cout<<maxe<<endl;
	return 0;
}
