# define ll long long
ll coin_c(ll ar[],ll weight,int m)
{
    ll i,j,table[weight+1];
    for(i=1;i<=weight;i++)
        table[i]=INT_MAX;
    table[0]=0;
    for(i=1;i<=weight;i++)
    for(j=0;j<m;j++)
    if(ar[j]<=i)
    {
        ll s_r=table[i-ar[j]];
        if(s_r!=INT_MAX && s_r+1<table[i])
            table[i]=s_r+1;
    }
    return table[weight];
}
