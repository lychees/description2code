#include <iostream>
#include <map>
using namespace std;


void doRest(){
	
	int n,num=0,start=1,sz=0,ans=0;
	map<int,int> mp;
	cin>>n;
	for(int i=1;i<n+1;i++){
		cin>>num;
		
		if(mp[num]){
			start= max(start,mp[num]+1);
		}
		mp[num]=i;
		if(sz<(mp[num]-start+1)){
			ans = start;
			sz=(mp[num]-start+1);
		}
	}
	cout<<ans<< " "<<ans+sz-1<<endl;
}

int main()
{
	int t;
	cin>>t;
	while(t--){
		
		doRest();
		
	}
	
    return 0;
}
