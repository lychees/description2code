#include<iostream>
#include<cstdio>
using namespace std;

int main()
	{
		string a,b;
		int t,carry,x,y;
		cin>>t;
		while ( t-- )
			{
				carry = 0;
				cin>>a>>b;
				for( int i = 0 ; i < a.length() ; i++ )
					{
						x = (int)a[i] - 48;
						y = ( int)b[i] - 48;
						if ( x + y > 9) 	
							carry++;
					}
				if ( carry == 0 )
					printf("No carry operation\n");
					
					
					
				else if ( carry == 1 )
					printf("1 carry operation\n");
				else 
					printf("%d carry operations\n",carry);
			}
	}
