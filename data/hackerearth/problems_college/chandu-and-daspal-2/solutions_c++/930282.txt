inline int gcd(int x, int y){
    if(y == 0) return x;
    if(y > x) return gcd(y, x);
    return gcd(y, x % y);
}

inline void Solve(){
    int x, y, g;
    string s;
    sd(x); sd(y); cin>>s;
    string s1 = "Chandu", s2 = "Daspal";
    int i = 0;
    while(true){
        i++;
        g = gcd(x, y);
        x -= g, y -= g;
        if(x == 0 or y == 0){
            break;
        }
    }
    if(i & 1){
        if(s == s1){
            cout<<s1<<endl;
        }
        else{
            cout<<s2<<endl;
        }
    }
    else{
        if(s == s1){
            cout<<s2<<endl;
        }
        else{
            cout<<s1<<endl;
        }
    }
    return;
}

int main(){
    //freopen("input.txt", "r", stdin);
    int t = 1;
    sd(t);
    while(t--){
        Solve();
    }
    return 0;
}
