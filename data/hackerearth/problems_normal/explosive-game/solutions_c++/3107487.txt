int main() {
	int N, Q, a, b;
	vi diff;
	LL Min = 0, x, Max = 0;
	gi(N); gi(Q);
	rep(i, N) {
		gi(a); gi(b);
		if (a > b) swap(a, b);
		diff.pb(b - a);
		Min += a;
	}
	Max = Min;
	sort(all(diff));
	rep(i, N) {
		if (i & 1) Max += diff[i];
		else Min += diff[i];
	}
	while (Q--) {
		cin >> x;
		if (x != Max || Min != Max) {
			puts("NO");
			continue;
		}
		puts("YES");
	}
	return 0;
}
