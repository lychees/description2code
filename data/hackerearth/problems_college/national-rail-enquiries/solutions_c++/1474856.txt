#include<iostream>
#include<cstdio>
using namespace std;
long long gcd(long long a,long long b) 
{ 
	if(b==0)
	return a;
	else
	return gcd(b,a%b);
} 
long long fact(long long n) 
{ 
	if(n==0 || n==1) 
	return 1; 
	else 
	return n*fact(n-1); 
} 
int main() 
{ 
	int t;
	long long j; 
	scanf("%d",&t); 
	j=1LL;
	while(t--) 
	{ 
		long long n,res; 
		cin>>n;
		if(n>110) 
		res=0; 
		else 
		res=fact(n-1LL)%110LL; 
		if(res>0 && gcd(res,n)==1LL) 
		printf("Case %d:Yes\n",j++); 
		else 
		printf("Case %d:No\n",j++); 
	} 
	return 0; 
}
