#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int T;
    cin >> T;
    for(int t = 0; t < T; t++){
    	int N;
    	cin >> N;
    	vector<long> R;
    	R.reserve(N);
    	long min;
    	long sum = 0;
    	for(int n = 0; n < N; n++){
    		long r;
    		cin >> r;
    		sum += r;
    		R.push_back(r);
    		if(0==n) min = r;
    		if(min > r) min = r;
    	}
    	sum += ((N-2)*min);
    	cout << sum << endl;
    }
    return 0;
}
