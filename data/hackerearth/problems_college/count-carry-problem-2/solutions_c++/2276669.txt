#include <iostream>
#include <vector>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
	long long 	int a,b;
		cin>>a>>b;
	long long 	int n1=a;
		vector <int> v1;
		while(n1>0)
		{
			v1.push_back(n1%10);
			n1=n1/10;
			
		}
		long long int n2=b;
		vector <int> v2;
		while(n2>0)
		{
			v2.push_back(n2%10);
			n2=n2/10;
			
		}
			int count=0;
		if(v1.size()>v2.size())
		{
			int carry=0;
		for(int i=0;i<v2.size();i++)
		{
			if(v1[i]+v2[i]+carry>9)
			{
				count++;
				carry=1;
			}
			
		}
		if(carry==1)
		{
			int j=v2.size()-1;
			for(int i=j;i<v1.size();i++)
			{
					if(v1[i]+carry>9)
			{
				count++;
				carry=1;
			}
				
			}
		}
		}
		else
		{
			int carry=0;
		for(int i=0;i<v1.size();i++)
		{
			if(v1[i]+v2[i]+carry>9)
			{
				count++;
				carry=1;
			}
			
		}
		if(carry==1)
		{
			int j=v1.size()-1;
			for(int i=j;i<v2.size();i++)
			{
					if(v2[i]+carry>9)
			{
				count++;
				carry=1;
			}
				
			}
		}
		}
	switch(count)
	{
		case 0:cout<<"No carry operation"<<endl;
		break;
		case 1:cout<<"1 carry operation"<<endl;
		break;
		default :cout<<count<<" carry operations"<<endl;
	}
		
		
	}
    //cout << "Hello World!" << endl;
    return 0;
}
