#include <bits/stdc++.h>
using namespace std;
#define mod 10000007

int main()
{
	long long a[1000001],i,t,n;
	a[0]=1;
	a[1]=1;
	a[2]=1;
	a[3]=2;
	for(i=4;i<=1000000;i++)
	{
		a[i]=((a[i-1]%mod)+(a[i-3]%mod)+(a[i-4]%mod))%mod;
	}
	cin>>t;
	while(t--)
	{
		cin>>n;
		cout<<a[n]<<endl;
	}
    return 0;
}
