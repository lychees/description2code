#include <iostream>
using namespace std;

bool palin(string s)
{
	int len = s.length();
	for(int i=0;i<len/2;i++) if(s[i]!=s[len-1-i]) return false;
	return true;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	if(palin(s)) cout<<"Palindrome"<<endl;
    	else 
    	{
    	    long long prod=1;
    		for(int i=0;i<s.length();i++) prod*=s[i]-'a'+1;
    		cout<<prod<<endl;
    	}
    }
}
