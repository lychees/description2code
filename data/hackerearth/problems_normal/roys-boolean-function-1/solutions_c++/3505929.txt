#include <iostream>
using namespace std;
long gcd(long i,long n);
int main()
{
	int t,count,prime;
	long n,GCD;
	cin>>t;
	while(t--)
	{	
		cin>>n;
		count=0;
		for(long i=1;i<=n;i++)
		{	
			GCD=gcd(i,n);
			if(GCD==1)
				count++;
		}
		prime=0;
		for(int j=2;j*j<=count;j++)
		{
			if(count%j==0)
				prime=1;
		}
		if(prime==0 && count!=1)
			cout<<"TRUE"<<endl;
		else
			cout<<"FALSE"<<endl;
	}
	return 0;
	
}
long gcd(long n,long i)
{
	if(i==0)
		return n;
	else
		gcd(i,n%i);		
}
