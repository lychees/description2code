#include <iostream>
#include<cstring>
using namespace std;

char s[10001];
int main()
{
    int T,N;
    register int tc,p,q,lc;
    register int  ac,bc;
    int len;
    
    scanf("%d%*c",&T);
    
    while(T--)
    {
    	gets(s);
    	len=strlen(s);
    	
    	scanf("%d%*c%d%*c",&ac,&bc);
    	lc=(ac<bc)?ac:bc;
    	tc=0;
    	
    	q=len/2;
    	p=q-1;
    	
    	while(p>=0 && q<len)
    	{
    		char x=s[p],y=s[q];
    		
    		if(x==y)
    		{
    			if(x=='/')tc+=(lc+lc);
    		}
    		else if((x=='a' && y=='/')||(y=='a' && x=='/'))
    		{
    			tc+=ac;
    		}
     		else if((x=='b' && y=='/')||(x=='/' && y=='b'))
    		{
    			tc+=bc;
    		}
    		else
    		{
    			tc=-1;
    			break;
    		}
    		p--;
    		q++;
    	}
    	printf("%d\n",tc);
    }
    return 0;
    			
}
