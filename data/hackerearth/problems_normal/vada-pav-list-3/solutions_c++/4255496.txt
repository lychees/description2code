#include<bits/stdc++.h>
using namespace std;
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(false);
    set<string>v;
    int n;
    cin>>n;
    for(int i=0;i<n;i++)
    {
        string s;
       cin>>s;
       v.insert(s);
    }
    cout<<v.size()<<"\n";
    set<string>:: iterator it;
    for(it=v.begin();it!=v.end();it++)
        cout<<(*it)<<"\n";
    return 0;
}
