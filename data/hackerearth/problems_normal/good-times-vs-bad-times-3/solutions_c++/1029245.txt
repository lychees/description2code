#include <bits/stdc++.h>
using namespace std; int dpbad[25][65][65]; int dpgood[25][65][65]; 
int main()
{int curgood=0, curbad=0; 
    for (int g=23; g>=0; g--)
    {
    	for (int y=59; y>=0; y--)
    	{
    		for (int z=59; z>=0; z--)
    		{int r=0; 
    			for (int zz=2;zz<=59; zz++)
    			{
    				if (g%zz==0 && y%zz==0 && z%zz==0)
    				{r=1; 
    					break; 
    				}
    			}
    			if (r) curbad++; 
    			else curgood++;
    			dpbad[g][y][z]=curbad;
    			dpgood[g][y][z]=curgood;
    		}
    	}
    }
    int a; cin >> a;
    for (int g=0;g<a; g++)
    {
    	int b,c,d; cin >> b >> c >> d;
    	int t=dpgood[b][c][d];
    	int tt=dpbad[b][c][d];
    	if (!t || !tt)
    	cout << dpbad[b][c][d] << ":" << dpgood[b][c][d] << '\n'; 
    	else
    	{
    		int r=__gcd(t,tt);
    		cout << tt/r << ":" << t/r << '\n';
    	}
    	
    }
    return 0;
}
