#include <iostream>
using namespace std;

int main()
{
    long long int t;
    cin>>t;
    while(t--)
    {
    	long long int n,k,i,start=-1,end=-1,ans,max=0;
    	cin>>n>>k;
    	long long int a[n+100];
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	start=0;
    	i=0;
    	end=0;
    	while(i<n)
    	{
    	
    		if(a[i]>k && i<n)
    		{
    			i++;
    		}
    		else
    		{
    		ans=0;
    		while(a[i]<=k && i<n)
    		{
    			ans=ans+a[i];
    			if(ans>max)
    			max=ans;
    			i++;
    		}
    		
    	}
    	
    	}
    	cout<<max<<endl;
    	
    }
    return 0;
}
