#include<iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
	{
		int m;
		cin>>m;
		int a[10007]={0};
		int x;
		for(int i=0;i<m;++i)
			{
				cin>>x;
				++a[x];
			}
		//for(int j=0;j<=6;++j)
		//	cout<<a[j]<<" ";
		//cout<<endl;
		int n;
		cin>>n;
		char ch[10000];
		string w;
		int count=0;
		for(int i=0;i<n;++i)
			{
				int b[10007]={0};
				scanf(" %[^\n]s",ch);
				w="";
				int l=strlen(ch);
				for(int j=0;j<l;++j)
					{
						if(ch[j]==' ')
							{
								int y=stoi(w);
								++b[y];
								w="";
							}
						else
							w+=ch[j];
					}
				int y=stoi(w);
				++b[y];
				int no=0;
			//	for(int j=0;j<=6;++j)
			//		cout<<b[j]<<" ";
			//	cout<<endl;
				for(int j=0;j<10007;++j)
					{
						if(b[j]<a[j])
							{
								no=1;
								break;
							}
					}
				if(!(no))
					count++;
			}
		cout<<count<<endl;
		return 0;
	}
