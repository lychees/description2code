#include <bits/stdc++.h>

using namespace std;
int freq[27];
int main()
{
	int no;
	cin>>no;
	string str;
	cin>>str;
	int start=0,end=str.size();
	
	for(int i=0;i<27;i++)
		freq[i]=0;

	while(start<end)
	{
		int max_idx=-1;
		char maxc='A';
		for(int i=start;i<end;i++)
		{
			if(str[i]>maxc && freq[str[i]-'a']==0)
			{
				maxc=str[i];
				max_idx=i;
			}
		}
		if(max_idx==-1)
			break;
		cout<<str[max_idx];
		freq[str[max_idx]-'a']=1;

		start=max_idx+1;
	}
	cout<<endl;
}
