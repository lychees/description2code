#include<stdio.h>
#include<math.h>
int main()
{
    long long z,x,y,num2;
    int t,num1,loop=0,sqroot,cnt=0,ans[50],k=0,i;
    scanf("%d",&t);
    while(loop<t)
    {
    scanf("%llu%llu%llu",&z,&x,&y);
    sqroot=sqrt(z);
        for(i=1;i<=sqroot;i++) 
        {
            num2=z/i;
            if(num2*i==z)
            {
                if(num2<=y)
                    cnt++;
                if(num2<=x&&num2!=i)
                    cnt++;
            }
        }
        ans[k]=cnt;
        cnt=0;
        k++;
        loop++;
    }
    for(i=0;i<t;i++)
        printf("%d\n",ans[i]);
    return 0;
}
