#include<bits/stdc++.h>
using namespace std;
int arr[27],flag[27];
int main()
{
	int n,i,j,c,l,tmp;
	char str[101];
	scanf("%d",&n);
	memset(arr,0,sizeof(arr));
	for(j=0;j<n;j++)
	{
		scanf("%s",str);
		l=strlen(str);
		memset(flag,0,sizeof(flag));
		for(i=0;i<l;i++)
		{
			tmp=str[i];
			if(flag[tmp-97]==0)
			{
				flag[tmp-97]=1;
				arr[tmp-97]++;
			}
		}
	}
	c=0;
	for(i=0;i<26;i++)
	{
		if(arr[i]==n)
		{
			c++;
		}
	}
	printf("%d",c);
	return(0);
}
