#include <iostream>
using namespace std;

unsigned long long int dp[1001];

void construct_dp(){
	dp[0] = 0;
	for(unsigned long long int i = 1;i<1001;i++){
		dp[i] = dp[i-1] + (i*i*i);
		//cout<<dp[i]<<endl;
	}
}

int main(){
	//cout<<"started"<<endl;
	construct_dp();
	//cout<<"done"<<endl;
	int t;
	unsigned long long int n,k,m, bill;
	cin >> t;
	while(t--){
		cin >> n >> k >> m;
		bill = (dp[n] * k);
		if(bill > m) {
			cout<<bill-m<<endl;
		} else {
			cout<<0<<endl;
		}
		//cout<<- m<<endl;
	}
	return 0;
}
