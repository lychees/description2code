#include <iostream>
#include<cstdlib>
using namespace std;
int arr[100005];
int tree[100005];
int cmp(const void *a,const void *b)
{
return *(int *)a-*(int *)b;
}
int main()
{
 int t;
 cin>>t;
 while(t--)
 {
 	int n,i;
 	cin>>n;
 	for(i=0;i<n;i++)
 	{
 		cin>>arr[i];
 	}
 	for(i=0;i<n;i++)
 	{
 		cin>>tree[i];
 	}
 	qsort(arr,n,sizeof(int),cmp);
 	qsort(tree,n,sizeof(int),cmp);
 	int max=0,value;
 	for(i=0;i<n;i++)
 	{
 		value=abs(arr[i]-tree[i]);
 		if(max<value)
 		max=value;
 	}
 	cout<<max<<endl;
 	 value=0;
 	int ans=(n-1)/2;
 cout<<arr[ans]<<endl;
 }
 return 0;
}
