#include <iostream>
#include <climits>
using namespace std;

int main()
{
	long long int n, m, i, p, k, l, j, maxd=LLONG_MIN, maxb=LLONG_MIN;
    cin>>n;
    cin>>m;
    for(i=1;i<n;i++)
    {
    	cin>>p;
    	k=abs(m-p);
    	if(k>maxd)
    	{
    		maxd=k;
    	}
    	m=p;
    }
    cin>>l;
    for(j=1;j<n;j++)
    {
    	cin>>p;
    	k=abs(l-p);
    	if(k>maxb)
    	{
    		maxb=k;
    	}
    	l=p;
    }
    if(maxd>maxb)
    {
    	cout<<"Dom\n";
    	cout<<maxd;
    }
    else if(maxd<maxb)
    {
    	cout<<"Brian\n";
    	cout<<maxb;
    }
    else if(maxd==maxb)
    {
    	cout<<"Tie\n"<<maxd;
    }
    return 0;
}
