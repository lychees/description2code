#include <bits/stdc++.h>
using namespace std;
int a,b;
long long c;
map<long long, bool> f;

bool check(long long cur, int cof = 1){
	if (c/cof < cur) return false;
	cur = cur*cof;
	if (cur >= c) return false;
	if (f.count(cur)) return f[cur];

	bool res = false;
	res = res | !check(cur, a);
	res = res | !check(cur, b);

	f[cur] = res;
	return res;

}

void solve(){
	scanf("%d%d%lld",&a,&b,&c);
	if (a>b) swap(a,b);
	int cur=0;
	while (c>1){
		int d=a;
		if (cur==0) d=b;
		long long now = c/d;
		if (now*d < c) now++;
		c = now;
		cur ^= 1;
	}
	if (!cur) printf("Bob\n"); else printf("Adam\n");

}

int main(){
	int test; scanf("%d",&test);
	while (test--){
		solve();
	}
}
