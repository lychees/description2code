#include<iostream>
#include<map>
#include<cstdio>
using namespace std;
#define ull unsigned long long
ull arr[1000000];
int lou=0,upp=0;
int main()
{
	ull x=0,ans=0,p=0,a=1,b=2;
	for(;ans<=(ull)100000000000000000;x++)
	{
		ans+=a*b*b;
		b++,a++;
		arr[x]=ans;
	}
	upp=x;
	
	int test;
	scanf("%d",&test);
	while(test--)
	{
		ull xx;
		scanf("%llu",&xx);
		if(xx<arr[0])
		{
			printf("0\nDraw\n");
			continue;	
		}
		if(xx>arr[upp-1])
		{
			printf("%d\n%s\n",upp,((upp%2)?"Dhruv":"Ketan"));
			continue;
		}
		int l=0,u=upp-1;
		int ans=0;
		while(l<=u)
		{
			int mid=(l+u)/2;
			if(arr[mid]>xx)
			{
				u=mid-1;
			}
			else
			{
				l=mid+1;
				ans=l;
			}
		}
		printf("%d\n%s\n",ans,((ans%2)?"Dhruv":"Ketan"));
		
	}
}
