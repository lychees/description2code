for _ in range(int(raw_input())):
    P, x, y, u, v = map(int, raw_input().split())
    m = x-u
    n = y-v
    if m % 2 != n % 2:
        print "Terrorists Win !"
        continue
    if P > 2:
        print "Counter-Terrorists Win !"
        continue
    if P == 2:
        if abs(m) == abs(n):
            print "Counter-Terrorists Win !"
            continue
        else:
            print "Terrorists Win !"
            continue
    if m == 0 and n == 0:
        print "Counter-Terrorists Win !"
        continue
    print "Terrorists Win !"
