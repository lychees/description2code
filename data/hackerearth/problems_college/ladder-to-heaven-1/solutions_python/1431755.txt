def fib(n):
    f = [1,2]
    i=1
    while f[i]<n:
        i = i +1
        f.append(f[i-1]+f[i-2])
        
    return f
    
N =  4*(10**16)
arr = fib(N)
l = len(arr)

def num(n):
	global arr,l
	i = 1
	s = 0
	
	while arr[i]<=n:
		s += arr[i]
		i = i + 3
		
	return s
	
for _ in xrange(input()):
	print num(input())
