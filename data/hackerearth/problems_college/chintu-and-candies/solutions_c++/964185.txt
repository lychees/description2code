#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
using namespace std;
typedef long long ll;
bool isPrime[10005];
bool visited[10005];
ll arra[10005];
ll arrb[10005];
void calc()
{
    int i,j;
    for(i=0;i<10005;i++)
    {
        arra[i]=0;
        arrb[i]=i;
    }
    for(i=2;i<10005;i++)
    {
        if(arra[i]==0)
        {
            arra[i]=1;
            arrb[i]=i-1;
            for(j=2*i;j<10005;j=j+i)
            {
                arra[j]=arra[j]+1;
                arrb[j]=arrb[j]-(arrb[j]/i);
            }
        }
    }
}
int main() {
    calc();
    int t;
    cin>>t;
    while(t--) {
        int temp;
        bool flag=false;
        int a,b,c;
        cin>>a>>b>>c;
        for(int i=b+1;i<=a;i++) if(arrb[i]==c) {
            temp=i;
            flag=true;
        }
        if(!flag) cout<<"-1"<<endl;
        else cout<<temp<<endl;
    }
    return 0;
}
