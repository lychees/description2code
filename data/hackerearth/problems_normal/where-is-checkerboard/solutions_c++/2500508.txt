char s[2005][2005];
int val[6][2005][2005];
int main()
{
	int n,m;
	cin>>n>>m;
	rep(i,0,n)
	{
		rep(j,0,m)
		{
			cin>>s[i][j];
		}
	}
	int dx=1;
	rep(j,0,n)
	{
		rep(k,0,m)
		{
			if(j+dx<n&&k+dx<m)
			{
				if(s[j][k]=='_'&&s[j+dx][k+dx]=='_'&&s[j][k+dx]=='#'&&s[j+dx][k]=='#')
				{
					val[0][j][k]=1;
				}
			}
		}
	}
	for(int i=1;i<5;i++)
	{
		dx*=2;
		rep(j,0,n)
		{
			rep(k,0,m)
			{
				if(j+dx<n&&k+dx<m)
				{
					if(val[i-1][j][k]==1&&val[i-1][j+dx][k+dx]==1&&val[i-1][j][k+dx]==1&&val[i-1][j+dx][k]==1)
					{
						val[i][j][k]=1;
					}
				}
			}
		}

	}
	int sum=0;
	rep(i,0,n)
	{
		rep(j,0,m)
		{
			if(val[4][i][j]==1)
				sum++;
		}
	}
	cout<<sum<<"\n";
	return 0;
}
