#include<iostream>
#include<cstdio>
#include<vector>
using namespace std;
const int mod=1000000007;
long long power(long long a,long long n)
{
    long long temp;
    if(n==0)
        return 1;
    if(n==1)
        return a;
        temp=power(a,n/2);
        temp=(temp*temp)%mod;
        if(n%2)
            temp=(temp*a)%mod;
    return temp;
}
int main()
{
    long long t,n,i,sum,res,store;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld",&n);
        sum=(n*(n+1))/2;
        res=(power(2,n-1)*sum)%mod;
        printf("%lld\n",res);
    }
    return 0;
}
