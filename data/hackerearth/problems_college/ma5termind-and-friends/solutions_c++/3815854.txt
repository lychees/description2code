#include<bits/stdc++.h>
using namespace std;
#define pii pair<int,int>
#define pb push_back
#define ft first
#define sd second
#define ll long long

vector<ll> A(100005,0),B(100005,0);
vector< pair<pii, ll> > M(100005);

int main() {
	ios::sync_with_stdio(false);
	int t;
	cin>>t;
	while ( t-- ) {
		int n,m;
		cin>>n>>m;
		int lim=max(n,m);
		for ( int i=0;i<lim;i++ ) {
            A[i]=0;
            B[i]=0;
            M[i]={{0,0},0};
		}
		int l,r;
		ll c;
        for ( int i=0;i<m;i++ ) {
            cin>>l>>r>>c;
            M[i].ft.ft=l;
            M[i].ft.sd=r;
            M[i].sd=c;
        }
        int q;
        cin>>q;
        int x,y;
        while ( q-- ) {
            cin>>x>>y;
            x--;
            y--;
            A[x]+=1;
            if ( y+1 < m ) {
				A[y+1]-=1;
            }
		}
		for ( int i=0;i<m;i++ ) {
            if ( i!=0 ) A[i]+=A[i-1];
			l=M[i].ft.ft-1;
			r=M[i].ft.sd-1;
			c=M[i].sd;
			B[l]+=(c*A[i]);
			if ( r+1<n ) {
				B[r+1]+=(-1*c*A[i]);
			}
		}
		for ( int i=0;i<n;i++ ) {
            if ( i!=0 ) B[i]+=B[i-1];
			cout<<B[i]<<" ";
		}
		cout<<"\n";
	}
	return 0;
}
