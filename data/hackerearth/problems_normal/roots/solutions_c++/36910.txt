#include<iostream>
using namespace std;

int main()
{
int n;    
int count=0;
int a[100],b[100],c[100];
int d,d1,d2;
cin>>n;
for(int i=0;i<n;i++)    
{
cin>>a[i]>>b[i]>>c[i];        
d1=b[i]*b[i];
d2=4*a[i]*c[i];
d=d1-d2;

//cout<<d<<" "<<d1<<" "<<d2<<endl;

if(d>=0)
count++;        
}    
    
cout<<count;
}
