#include<iostream>
using namespace std;
int reverse(int x)
{
	int rem,rev=0;
    while (x > 0)
	{
      rem = x % 10;
      rev = rev * 10 + rem;
      x = x / 10; 
	}
	return rev;
}
int main()
{
	long long int N;
	int i,m;
	cin>>N;
	m=N;
	int *a=new int [N];
    int *b=new int [N];
	int *sum=new int [N];
	for(i=0;i<N;i++)
	{
		sum[i]=0;
	}
	i=0;
	while(N>0)
	{
		cin>>a[i]>>b[i];
	    N--;
		i++;
	}
	i=0;
	while(i<m)
	{
	sum[i]=sum[i]+reverse(a[i]);
	sum[i]=sum[i]+reverse(b[i]);
	cout<<reverse(sum[i])<<endl;
	i++;
	}
	return 0;
}
