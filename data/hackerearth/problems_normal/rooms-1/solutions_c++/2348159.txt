#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,i,j;

    	vector<pair<long,long> > v;
    	cin>>n;
    	long a[100005],d;
		for(i=0;i<n;i++)
		cin>>a[i];
		for(i=0;i<n;i++)
		{
		cin>>d;
		v.push_back(make_pair(a[i],a[i]+d));
		}
		priority_queue<long,vector<long>,greater<long> > p;
		sort(v.begin(),v.end());
		p.push(v[0].second);
		long count=1;
		for(i=1;i<n;i++)
		{
			long x=p.top();
			long y=v[i].first;
			if(y>=x)
			{
				p.pop();
				p.push(v[i].second);
			}
			else
			{
				count++;
				p.push(v[i].second);
			}
		}
		cout<<count<<endl;
		v.clear();
		//p=priority_queue<long,vector<int>,greater<int> >();
    }
    return 0;
}
