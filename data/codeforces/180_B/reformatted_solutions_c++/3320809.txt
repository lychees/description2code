const int maxn = 100005;

int gcd(int a, int b) { return b ? gcd(b, a % b) : a; }

int main() {
  int n, m, cnt = 0;
  scanf("%d %d", &n, &m);
  while (gcd(n, m) != 1) {
    m /= gcd(n, m);
    cnt++;
  }
  if (m == 1)
    printf("2-type\n%d\n", cnt);
  else if (!cnt && n % m == 1)
    printf("3-type\n");
  else if (!cnt && n % m == m - 1)
    printf("11-type\n");
  else if ((n * n - 1) / (n % 2 + 1) % m == 0)
    printf("6-type\n");
  else
    printf("7-type\n");
  return 0;
}
