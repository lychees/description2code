#include<iostream>
#define ll long 
#include<cmath>
using namespace std;
int main()
{
    ll b,c,n,i,j,t;
    cin>>n;
    if(n>=100000)
    {
    	cin>>t;
    	while(t--)
    	{
    		cin>>b;
    		cout<<0<<endl;
    	}
    }
    else
    {int a[n+1];
    for(i=2;i<=n;i++)
        a[i]=0;
    for(i=2;i<=n/2;i++)
    {
        if(a[i]==0)
        {
            for(j=2*i;j<=n;j=j+i)
            {
                a[j]++;
            }
        }
    }
    cin>>t;
    while(t--)
    {

        cin>>b;
        c=0;
        if(b==0)
            cout<<"1"<<endl;
        else if(b==1)
        {
            for(i=2;i<=n;i++)
            {
                if(a[i]<=1)
                    c++;
            }
        cout<<c<<endl;
        }
        else
        {
            for(i=6;i<=n;i++)
        {
            if(a[i]==b)
                c++;
        }
        cout<<c<<endl;
    }
    }}
    return 0;
}
