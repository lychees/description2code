#include<iostream>
#include<climits>

using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
	int n,k;
	cin>>k>>n;
	if(n==1)
	{
	cout<<k<<endl;
	continue;
	}
	int matrix[n+1][k+1];
	for(int i=0;i<=n;i++)
	{
	matrix[i][0]=0;
    }
    for(int i=1;i<=k;i++)
    matrix[1][i]=i;
  int i,j,m;
  for(int i=2;i<=n;i++)
  {
  	for(int j=1;j<=k;j++)
  	{
  		int mini,miny=INT_MAX;
  		for(m=1;m<=j;m++)
  	     {
  	     mini=1+max(matrix[i-1][m-1],matrix[i][j-m]);
  	     if(mini<miny)
  	     miny=mini;
  	     }
  	    matrix[i][j]=miny;
  	    /*cout<<matrix[i][j]<<" "<<i<<" "<<j<<endl;
  	    getch();*/
  	}
  	
   }
  cout<<matrix[n][k]<<endl;;
}
}
