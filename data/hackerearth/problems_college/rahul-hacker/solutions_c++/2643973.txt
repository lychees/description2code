#include <bits/stdc++.h>
using namespace std;
long int reverse(long int res)
{
    long int temp=0;
    while(res>0)
    {
      temp=temp*10+(res%10);
      res/=10;
     }
     return temp;
}
long int zero(long int temp)
{
    while(temp%10==0)
    {
       temp/=10;
    }
    return temp;
}
int main() {
                      int t;
                      long int a,b,result;
                      cin>>t;
                      while(t--)
                      {
                         cin>>a>>b;
                         a=zero(a);
                         b=zero(b);
                         a=reverse(a);
                         b=reverse(b);
                         result=a+b;
                         result=zero(result);
                         result=reverse(result);
                         cout<<result<<endl;
                     }
                         
	return 0;
}
