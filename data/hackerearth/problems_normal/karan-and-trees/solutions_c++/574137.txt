#include <iostream>
#include<cstdio>
using namespace std;
int chk(int a,int b)
{
if(a==b)
return a;
else if(a>b)
return chk(a/2,b);
else if(b>a)
return chk(a,b/2);
}

int main()
{
    long long int t,a,b,ans;
    scanf("%lld",&t);
    while(t--)
    {
    	scanf("%lld %lld",&a,&b);
    	ans=chk(a,b);
    	printf("%lld\n",ans);
    	
    }
    return 0;
}
