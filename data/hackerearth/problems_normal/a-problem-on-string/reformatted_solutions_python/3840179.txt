n = int(raw_input())
s = raw_input().strip()

counts = {}
for ch in s:
    counts[ch] = counts.get(ch, 0) + 1

q = int(raw_input())
while q > 0:
    q -= 1
    x, y = raw_input().split()

    if x != y:
        print counts.get(x, 0) * counts.get(y, 0)
    else:
        print counts.get(x, 0) * (counts.get(x, 0)-1) / 2
