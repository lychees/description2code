#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define d1(a)cout<<#a<<": "<<a<<"\n";
#define d2(a,b)cout<<#a<<": "<<a<<" , "<<#b<<": "<<b<<"\n";
#define d3(a,b,c)cout<<#a<<": "<<a<<" , "<<#b<<": "<<b<<" , "<<#c<<": "<<c<<"\n";
#define d4(a,b,c,d)cout<<#a<<": "<<a<<" , "<<#b<<": "<<b<<" , "<<#c<<": "<<c<<" , "<<#d<<": "<<d<<"\n";

ll dp[100];

int main() {
	string s;
	int n;
	cin>>n;
	cin>>s;
	memset(dp,0,sizeof(dp));
	for(int i=0;i<s.length();i++){
		dp[s[i]-'a']++;
	}
	int q;
	cin>>q;
	while(q--){
		char l,r;
		cin>>l>>r;
		ll ans = dp[r-'a']*dp[l-'a'];
		if(l == r){
			ans = dp[r-'a']*(dp[l-'a']-1);
			ans = ans / 2;
		}
		cout<<ans<<"\n";
	}
	return 0;
}
