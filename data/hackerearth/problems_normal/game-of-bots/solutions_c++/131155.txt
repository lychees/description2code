#include <iostream>
#include <stdio.h>
using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
    {
    	string a,b;
    	int i=0,j=0,n;
    	cin>>a>>b;
    	n=a.size();
    	bool flag=false;
    	while(i<n&&j<n)
    	{
    		if(b[i]=='A')
    		{
    			while(j<n&&a[j]=='#')
    			 j++;
    			if(a[j]=='B')
    			{
    				flag=true;
    				break;
    			}
    			i++;
    			j++;
    		}
    		else if(b[i]=='B')
    		{
    			while(j<n&&a[j]=='#')
    			 j++;
    			if(a[j]=='A'||j>i)
    			{
    				flag=true;
    				break;
    			}
    			i++;
    			j++;
    		}
    		else
    		 i++;
    	}
    	if(flag)
    	 printf("No\n");
    	else
    	 printf("Yes\n");
    }
    return 0;
}
