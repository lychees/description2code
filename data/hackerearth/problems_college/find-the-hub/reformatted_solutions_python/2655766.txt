import math
t = int(raw_input())
s = (raw_input())
u = map(int, s.split())
i = 0
q = []
while i < t:
    j = 2
    c = 0
    k = math.sqrt(u[i])
    while j <= (k):
        y = u[i] % j
        if y != 0:
            j = j+1
        else:
            c = c+1
            break
    if c == 0:
        q.append(u[i])
    i = i+1
x = max(q)
print x
print u.count(x)
