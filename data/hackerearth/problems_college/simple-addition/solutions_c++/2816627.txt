lld findP(lld x,lld y,lld no){
	if(x==1&&y==1) return no;
	if(x==y) return INT_MAX;
	lld t,a,b;
	a=max(x,y);
	b=x+y-a;
	t=(a-b)/b;
	if((a-b)%b!=0) t++;
	return findP(a-t*b,b,no+t);
}
