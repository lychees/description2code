#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int a,b,c;
		cin>>a>>b>>c;
		unsigned long long d = pow(a,b);
		long long div = d/c;
		long long int p = div*c;
		long long q = (div+1)*c;
		if(p==d )
		{
			cout<<d<<"\n";
		}
		else
		if(d-p<q-d)
		{
			cout<<p<<"\n";
		}
		else
		{
			cout<<q<<"\n";
		}
		
	}
}
