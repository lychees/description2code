#include<iostream>
typedef long long int ll;
using namespace std;
int main()
{
    ios::sync_with_stdio(0);
    int t;
    ll r,c,sum1=0,sum2=0;
    cin>>t;
    while(t--)
    {
        cin>>r>>c;
       if(r>=c)
       {
       sum1=c*(2+(c-1)*(r+1))/2;
       sum2=(c*(1+c*c))/2;
       cout<<sum1+sum2<<"\n";
       }
       else
       {
       sum1=(r*(1+r*r))/2;
       sum2=r*(2+(r-1)*(c+1))/2;
       cout<<sum1+sum2<<"\n";
       }
       /* for(int i=1;i<=c*c;i=i+(c+1))
        {
            sum2+=i;
        }
        cout<<sum2<<"\n";*/
    }
}
