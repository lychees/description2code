#include <iostream>
using namespace std;

int main()
{
    int k,m;
    cin>>k>>m;
    int a[m];
    for(int i=0;i<m;i++){
    	a[i]=0;
    }
    
    for(int i=0;i<k;i++){
    	int x=0;
    	cin>>x;
    	
    	while(x-->0){
    		int y;
    		cin>>y;
    		y--;
    		a[y]++;
    	}
    }
    
    int flag=0;
    for(int i=0;i<m;i++){
    	if(a[i]==0){
    		flag++;
    	}
    }
    if(flag==0){
    cout<<"YES";
    }
    else{
    	cout<<"NO";
    }
    
    return 0;
}
