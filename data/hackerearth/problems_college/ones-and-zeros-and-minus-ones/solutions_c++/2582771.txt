#include <iostream>
#include <vector>
using namespace std;

typedef long long LL;
#define FOR(i,p,n) for(int i=p; i<n; i++)
#define ROF(i,a,b) for(int i=a; i>b; --i)
const int maxn = 100100;
int a[maxn];
vector<int> two;

LL GCD(LL a, LL b)
{
	while (a > 0 && b > 0) {
		   if (a > b) a %= b;
		   else b %= a; 
	}
	return a + b;
}

int main()
{
   ios::sync_with_stdio(false);
   int cases;
   cin >> cases;
   while (cases--) {
   	      int n, x;
   	      cin >> n;
   	      two.clear();
   	      FOR (i, 0, n) {
   	      	   cin >> x;
   	      	   if (x == -1) {
   	      	   	   a[i] = 2;
   	      	   	   two.push_back(i);
   	      	   }
   	      	   else a[i] = x;
   	      }
   	      two.push_back(n);
   	      int ans = 0;
   	      int p = 0, q = 1;
   	      int id = 0, y;
   	      x = 0;
   	      while (true) {
   	      	     if (two[id] == n) break;
   	      	     y = two[id + 1];
   	      	     int cnt = 0;
   	      	     FOR (i, x, y) if (a[i] == 1) cnt++;
   	      	     if (cnt > ans) {
   	      	     	 ans = cnt;
   	      	     	 p = cnt;
   	      	     	 q = 1;
   	      	     }
   	      	     x = two[id] + 1;
   	      	     id++;
   	      }
   	      long long gcd = GCD(p, q);
		  
		  //cout << p << " " << q << endl;
   	      p /= gcd;
   	      q /= gcd;
   	      cout << p << "/" << q << endl;
   }
   return 0;
}
