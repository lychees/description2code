#include<bits/stdc++.h>
using namespace std;

bool p[1000005];

void sie() {
   p[1]=true;
   long int i,j;
   for(i=2;i<10000;i++) {
   	 if(!p[i])
   	 for(j=i*i;j<1000005;j=j+i) {
   	 	p[j]=true;
   	 }
   }
}


int main() {
    long int i,j,k,t;
    cin >> t;
    sie();
    while(t--) {
    	cin >> i;
    	j=0;
    	while(i!=0) {
    		j+=i%10;
    		i=i/10;
    	}
    
    	if(!p[j])
    		cout << "YES" << endl;
    	else
    		cout << "NO" << endl;
    }

	return 0;
}
