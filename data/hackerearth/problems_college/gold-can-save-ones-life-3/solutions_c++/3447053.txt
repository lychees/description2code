#include <bits/stdc++.h>
using namespace std;
 
long long int f[10005];
int a[10005];
int t, n, k;
 
long long int cnt(int ind) {
	if (ind >= n)	{
		return 0;
	}
	if (f[ind] != -1)	return f[ind];
	long long int r = -10000000000000000;
	r = max(r, cnt(ind + 1));
	r = max(r, a[ind] + cnt(ind + k + 1));
	f[ind] = r;
	return r;
}
 
int main()
{	
	scanf("%d", &t);
	while (t--) {
		scanf("%d%d", &n, &k);
		for (int i = 0; i < n; i++) {
			scanf("%d", &a[i]);
			f[i] = -1;
		}
		
		long long int mx = cnt(0);
		
		printf("%lld\n", mx);
	}
}
