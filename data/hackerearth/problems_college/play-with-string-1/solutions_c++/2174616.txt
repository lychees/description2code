#include <bits/stdc++.h>
using namespace std;

int main()
{
   int t;
   char str[1000];
   cin>>t;
   for(int tc=0;tc<t;tc++)
   {
   	cin>>str;
   	int len=strlen(str);
   	int flag=1,k=0;
   	char rev[len+1];
   	for(int j=len-1;j>=0;j--)
   		rev[k++]=str[j];
   	rev[k]='\0';
   	for(int i=1;str[i]!='\0';i++)
   	{
   		if(!((abs(str[i]-str[i-1]))==(abs(rev[i]-rev[i-1]))))
   			{
   				flag=0;
   				break;
   			}
   	}
   	if(flag==1)
   		cout<<"Funny"<<endl;
   	else
   		cout<<"Not Funny"<<endl;
   	
   }
    return 0;
}
