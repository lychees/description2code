#include <bits/stdc++.h>
#include<string.h>
using namespace std;

int m;
bool cmp(string s1,string s2)
{
	string t1=s1.substr(0,m);
	string t2=s2.substr(0,m);
	if(t1<=t2)
		return 1;
	else
		return 0;
}

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int n,k;
    	cin>>n>>k>>m;
    	vector <string> v(n);
    	for(int i=0;i<n;i++)
    		cin>>v[i];
    	sort(v.begin(),v.end(),cmp);
    	//for(int i=0;i<n;i++)
    		cout<<v[k-1]<<endl;
    }
    return 0;
}
