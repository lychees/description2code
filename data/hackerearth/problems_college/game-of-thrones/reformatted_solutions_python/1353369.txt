t = input()
while t:
    n = input()
    a = raw_input("").split()
    count = {}
    for n in a:
        if count.has_key(n) and n != ' ':
            count[n] += 1
        else:
            count[n] = 0
    tot = 0
    for key in count:
        if count[key] > 0:
            tot += count[key]

    print tot

    t -= 1
