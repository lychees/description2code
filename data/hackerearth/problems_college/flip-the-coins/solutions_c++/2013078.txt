#include<bits/stdc++.h>
using namespace std;
long long v[1000000];
int main()
{
    long long N,Q,A,B,C,i,c;

    scanf("%lld",&N);
    scanf("%lld",&Q);
    scanf("%lld %lld %lld",&A,&B,&C);
    for(i=B;i<=C;i++)
      v[i]=1;
    Q--;
    while(Q--)
    {
         c=0;
         scanf("%lld %lld %lld",&A,&B,&C);
         for(i=B;i<=C;i++)
         {
             if(v[i]==1)
                c++;
         }
         cout<<c<<"\n";
    }

}
