#include <bits/stdc++.h>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int a[26]={0},ans,i,n;
   char str[1000];
   cin>>str;
   n=strlen(str);
   for(i=0;i<n;i++)
   		a[str[i]-97]++;
   	ans=0;
   	for(i=0;i<26;i++)
   		ans+=(i+1)*a[i];
   	cout<<ans;
    return 0;
}
