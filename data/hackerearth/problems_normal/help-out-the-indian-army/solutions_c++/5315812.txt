#include <bits/stdc++.h>
using namespace std;

bool cmp(pair<long long,long long> a,pair<long long,long long> b){
	return (a.first-a.second<b.first-b.second);
}
int main(){
	int n;
	long long s,e;
	scanf("%d %lld %lld",&n,&s,&e);
	vector<pair<long long,long long> > vc;
	for(int i=0;i<n;i++){
		long long p,x;
		scanf("%lld %lld",&x,&p);
		vc.push_back(make_pair(x,p));
	}
	sort(vc.begin(),vc.end(),cmp);
	long long ans = 0;//max(0,vc[0].first-vc[0].second-S);
	long long last_unsafe = s;
	for(int i=0;i<n;i++){
		if(vc[i].first-vc[i].second>e)
			break;
	//	printf("%lld %lld\n",vc[i].first,vc[i].second);
		ans += max(0ll,vc[i].first-vc[i].second-last_unsafe);
	//	printf("snd=%lld\n",ans);
	//	printf("%lld %lld %lld\n",vc[i].first,vc[i].second,ans);
		last_unsafe = max(last_unsafe,vc[i].first+vc[i].second);
	}
	ans += max(0ll,e-last_unsafe);
	printf("%lld\n",ans);
	return 0;
}
