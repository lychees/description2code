#include <iostream>
using namespace std;

int main()
{
 long long int n;
 int arr[2000009];
 cin >> n;
 int i;
 for(i=2;i<=2000000;i++)
 arr[i]=1;
 long long int count=0;
 int j;
 for(i=2;i<=2*(n-1);i++)
 {
 	
 	if(arr[i]==1)
 	{
 		if(i<=n-1)
 		count+=i+1;
 		else
 		if(i>(n-1)&&i<=2*(n-1))
 		count+=2*n-i-1;
 		
 		int j;
 		for(j=2*i;j<=2*(n-1);j+=i)
 		arr[j]=0;
 	}
 }
 count=count%(1000000007);
 cout << count << endl;
    return 0;
}
