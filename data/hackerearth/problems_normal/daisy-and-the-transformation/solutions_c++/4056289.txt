#include<iostream>
#include<stdio.h>
using namespace std;
int main(){
	int cases,num,kount,i;
	scanf("%d",&cases);
	
	string s1,s2;
	while(cases--){

	cin>>s1>>s2;
	scanf("%d",&num);
	
	
	
	kount=0;
	for( i=0;i<s1.length();i++){
	if(s1[i]!=s2[i])
	kount=kount+1;	
	}	
		
	if(kount<=num){
		if((num-kount)%2==0)

		printf("Yes\n");
		else

	printf("No\n");
		
	}
	
	else

printf("No\n");	
	
	}
	
	
	
	
	return 0;
}
