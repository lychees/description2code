#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
#define ull unsigned long long
ll bs(ll a,ll b,ll c,ll d,ull k)
{
	ll low = 0,high = 1000000,mid,ans = 0;
	while(low<=high)
	{
		mid = low + ((high - low)/2);
		ull check = (ull)(a*pow(mid,3) + b*pow(mid,2) + c*mid + d);
		if(check <= k)
		{
			ans = mid;
			low = mid+1;
		}
		else
			high = mid - 1;
	}
	return ans;
}
int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	ll a,b,c,d;
    	ull k;
    	cin >> a >> b >> c >> d >> k;
    	cout << bs(a,b,c,d,k) << endl;
    }
    return 0;
}
