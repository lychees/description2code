#include <bits/stdc++.h>
using namespace std;

#define ll long long

int main()
{
	ll n,k;
	cin>>n>>k;

	priority_queue <ll> pq;

	while(n--)
	{
		int a;
		cin>>a;
		pq.push(a);
	}

	int sum=0;
	while(k--)
	{
		int x1 = pq.top();
		sum += x1;
		pq.pop();
		x1--;
		pq.push(x1);
	}
	cout<<sum<<endl;
	return 0;
}
