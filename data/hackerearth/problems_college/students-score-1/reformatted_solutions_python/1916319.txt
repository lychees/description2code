def consist():
    num = int(raw_input())
    i = 0
    score = {}
    a = raw_input().split()
    for a1 in a:
        score[i] = int(a1)
        i = i+1
    i = 0
    max = 0
    while i < num:
        j = i+1
        sum = score[i]
        while j < num:
            sum = sum+score[j]
            if sum > max:
                max = sum

            j = j+1
        i = i+1
    print max


if __name__ == "__main__":
    consist()
