using namespace std;
typedef pair<ll,ll> ii;
void rahul(ll n)
{
    ll till=0;
    //--------------------------------------UPPER-
    for(ll i=1;i<=n;i++)
    {
        for(ll sp=n-i;sp>0;sp--)
            cout<<" ";
        for(ll k=1;k<=till;k++)
            {
                cout<<"/ ";
            }
        if(i&1)
        {
          cout<<"/\\";
        }
        for(ll k=1;k<=till;k++)
            cout<<" \\";
        if(i&1)
            till++;
        cout<<"\n";
    }
    //----------------------------------------LOWER-
    if(n&1)
    {for(ll i=1;i<=n;i++)
    {
        for(ll sp=1;sp<=i+1;sp++)
            cout<<" ";
        if(i%2==0)
            till++;
        for(ll k=1;k<till;k++)
            cout<<"\\ ";
        if(i&1)
        {
            cout<<"\\/";
        }
        for(ll k=1;k<till;k++)
            cout<<" /";
        till--;
        cout<<"\n";
    }
    }
    else
    {
        for(ll i=1;i<=n;i++)
    {
        for(ll sp=1;sp<=i+1;sp++)
            cout<<" ";
        if(i%2)
            till++;
        for(ll k=1;k<till;k++)
            cout<<"\\ ";
        if(i%2==0)
        {
            cout<<"\\/";
        }
        for(ll k=1;k<till;k++)
            cout<<" /";
        till--;
        cout<<"\n";
    }
    }
}
int main()
{
        ll n;
        cin>>n;
        rahul(n);
        rahul(n+1);
}
