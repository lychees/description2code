#!/usr/bin/python
def anagramOrNot(a,b):
	if len(a)!=len(b):
		return False
	for i in a:
		if (i not in b) or b.count(i)!=a.count(i):
			return False
	for i in b:
		if (i not in a) or b.count(i)!=a.count(i):
			return False
	return True

def firstAndLast(a,b):
	if a[0]==b[0] and a[-1]==b[-1]:
		return True
	else:
		return False

def process(a,groupList):
	for word in groupList:
		if anagramOrNot(a,word) and firstAndLast(a,word):
			return True
	return False

number = int(input())
l=[]
for i in xrange(0,number):
	test = raw_input()
	l.append(test)
groupList = []
# print l
for word in l:
	if process(word,groupList):
		pass
	else:
		groupList.append(word)
