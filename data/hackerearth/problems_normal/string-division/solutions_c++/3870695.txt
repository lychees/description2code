#include <iostream>
using namespace std;

int main()
{
    int cases,l,tot;
    string s;
    cin>>cases;
    while(cases--){
    	tot=0;
    	cin>>s;
    	l=s.size();
    	if(l>9){
    		cout<<"YES\n";
    		continue;
    	}
    	int index[26]={0};
    	for(int i=0;i<l;i++){
    		index[s[i]-97]++;
    	}
    	for(int i=0;i<26;i++){
    		if(index[i])
    			tot++;
    	}
    	if(tot>3||(tot==3&&l>4)||(tot==2&&l>6)){
    		cout<<"YES\n";
    		continue;
    	}
    	cout<<"NO\n";
    }
    return 0;
}
