

total_candidates = int(raw_input())

scores = []
for i in range(1, total_candidates+1):
    score = int(raw_input())
    if score == 0:
        if len(scores) > 0:
            scores.pop()
        else:
            pass
    else:
        scores.append(score)

sum = 0
for _score in scores:
    sum += _score

print sum
