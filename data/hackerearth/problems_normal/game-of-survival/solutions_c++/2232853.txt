#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
   	int ts;
   	cin>>ts;
   	while(ts--)
   	{
   		vector<int> s;
   		vector<int> a;
   		int n;
   		int i;
   		int temp;
   		bool flag=true;
   		cin>>n;
   		for(i=0;i<n;i++)
   		{
   			cin>>temp;
   			s.push_back(temp);
   		}
   		for(i=0;i<n;i++)
   		{
   			cin>>temp;
   			a.push_back(temp);
   		}
   		sort(s.begin(),s.end());
   		sort(a.begin(),a.end());
   		for(i=0;i<n;i++)
   		{
   			if(s[i] <= a[i])
   			{
   				flag=false;
   				break;
   			}
   		}
   		if(flag)
   			cout<<"YES\n";
   		else
   			cout<<"NO\n";
   		
   	}
}
