#include<iostream>
using namespace std;
int n;
int ar[100002];
int dp[100002];

int cal(int i,int a)
{
   if(a==n)
   return 0;
   if(i==0&&dp[a]!=-1)
    return dp[a];
    if(i==0)
    {  dp[a]=max(ar[a]+cal(1,a+1),cal(0,a+1));
      return dp[a];
    }
    if(i==1)
    return cal(0,a+1);
}


main()
{
int t;
cin>>t;
for(int z=0;z<t;z++)
{
cin>>n;
for(int i=0;i<n;i++)
cin>>ar[i];
for(int i=0;i<n;i++)
  dp[i]=-1;
cout<<"Case "<<z+1<<": "<<cal(0,0)<<endl;
}
}
