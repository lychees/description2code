#include <bits/stdc++.h>

#define ll long long

using namespace std;

int main() {
    int t, n, x, m;

    cin >> t;
    while (t--) {
        cin >> n;

        m = floor(log2(n));
        m = pow(2, m);

        x = 2 * (n - m) + 1;
        cout << x << endl;
    }

    return 0;
}
