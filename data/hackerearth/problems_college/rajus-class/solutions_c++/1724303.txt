#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--) {
    	int n;
    	string s;
    	int no;
    	vector < pair < int,string> > v;
    	cin>>n;
    	while(n--){
    		cin>>no>>s;
    		v.push_back(make_pair(no,s));
    	}
    	sort(v.begin(),v.end());
    	cin>>no;
    	cout<<v[no-1].second<<endl;
    }
    return 0;
}
