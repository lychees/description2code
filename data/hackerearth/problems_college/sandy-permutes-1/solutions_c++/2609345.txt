#include <iostream>
using namespace std;

int main()
{
    long long int n,k,a[100000],max=0,ind,temp;
    
    cin>>n>>k;
    
    for(int i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    
    if(k>=n)
    {
    	for(int i=n;i>0;i--)
	    {
	    	cout<<i<<" ";
	    }
	    return 0;
    }
    
    for(int i=0;i<k&&i<n;i++)
    {
    	max=a[i];
    	ind =i;
    	for(int j=i+1;j<n;j++)
    	{
    		if(a[j]>max)
    		{
    			max=a[j];
    			ind=j;
    		}
    	}
    	if(a[i]<a[ind])
    	{
	    	temp=a[i];
	    	a[i]=a[ind];
	    	a[ind]=temp;
	    //	cout<<a[i]<<" ";
    	}
    	else
    	{
    	k++;
    	}
    	
    	
    }
    
    for(int i=0;i<n;i++)
    {
    		cout<<a[i]<<" ";
    }
    return 0;
}
