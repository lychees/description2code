#include <algorithm>
#include <cctype>
#include <cstdio>
#include <iostream>
using namespace std;
int main() {
  char c, temp;
  while (scanf("%c", &c) == 1)
    if (isalpha(c))
      temp = tolower(c);
  if (temp == 'a' || temp == 'e' || temp == 'i' || temp == 'o' || temp == 'u' ||
      temp == 'y')
    puts("YES");
  else
    puts("NO");
  exit(0);
}
