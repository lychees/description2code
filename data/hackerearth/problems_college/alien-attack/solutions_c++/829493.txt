#include <bits/stdc++.h>

using namespace std;

long long MOD = 1000000007;

long long powMod(long long a, long long b){
	if(b==0) return 1;
	if(b==1) return a%MOD;
	if(b%2) return (a*powMod(a*a%MOD, b/2))%MOD;
	else return powMod(a*a%MOD, b/2);
}

void solve(){
	long long n;
	cin>>n;
	cout<<((3*powMod(2,n-1))%MOD + MOD + (n%2?-1:1))%MOD<<endl;
}

int main(){
	int t;
	cin>>t;
	while(t--) solve();
	return 0;
}
