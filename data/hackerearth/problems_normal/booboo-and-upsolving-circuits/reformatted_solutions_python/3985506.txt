def fx(a, x, k):
    c = 0
    s = 0
    for i in range(len(a)):
        if s+a[i] <= x:
            s += a[i]
        else:
            c += 1
            s = a[i]
    if s > x:
        c += 1
    # print(c)
    if c <= k-1:
        return True
    return False
