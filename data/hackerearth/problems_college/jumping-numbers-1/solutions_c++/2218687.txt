#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<cstring>
using namespace std;
#define mod 4294967143
int cs,n,s,a;
int M[265][11];
int solve(int i,int j)
{
  int &res=M[i][j];
  if(res==-1)
  {
    if(i==0)
      res=abs(a-j)==1;
    else
    {
      res=0;
      if(j+1<=10)
        res=(res+solve(i-1,j+1))%mod;
      if(j-1>=0)
        res=(res+solve(i-1,j-1))%mod;
    }
    //printf("%d %d\n",i,j);
  }
  return res;
}
int main()
{
  scanf("%d",&cs);
  while(cs--)
  {
    scanf("%d",&n);
    s=0;
    for(a=1;a<=10;a++)
    {
      memset(M,-1,sizeof M);
      s=(s+solve(n-1,a))%mod;
    }
    printf("%d\n",s);
  }
  return 0;
}
