a = map(int, raw_input().split())
n = a[0]
x = a[1]
arr = map(int, raw_input().split())
i = 0
count = 0
score = 0
while i < n:
    if count < 2:
        if arr[i] <= x:
            score += 1
            i += 1
        else:
            count += 1
            i += 1
    else:
        break

print score
