#include <bits/stdc++.h>

using namespace std;
int A[1005], B[1005], C[1005], n, p, c, f;
bool a1;

void recur(int i, int s1, int s2, int s3)
{
    if(s1 == p and s2 == c and s3 == f)
        a1 = true;
    if(i >= n)
        return;
    recur(i+1, s1 + A[i], s2 + B[i], s3 + C[i]);
    recur(i+1, s1, s2, s3);
}

int main()
{
    int t, k = 1;
    for(scanf("%d", &t);t--;)
    {
        scanf("%d %d %d %d", &p, &c, &f, &n);
        for(int i = 0;i < n;++i)
            scanf("%d %d %d", &A[i], &B[i], &C[i]);
        a1 = false;
        recur(0, 0, 0, 0);
        if(a1) printf("Case #%d: yes\n", k++);
        else printf("Case #%d: no\n", k++);
    }
    return 0;
}
