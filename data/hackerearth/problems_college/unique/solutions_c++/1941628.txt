#include <bits/stdc++.h>
using namespace std;
long long ar[1000005];
int main() {
	int t;
	cin>>t;
	while(t--){
	    int n,cnt=1;
	    cin>>n;
	    for(int i=0;i<n;i++)
	    cin>>ar[i];
	    sort(ar,ar+n);
	    for(int i=1;i<n;i++){
	        if(ar[i]!=ar[i-1])
	        cnt++;
	    }
	    cout<<cnt<<endl;
	}
	return 0;
}
