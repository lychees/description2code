#include<iostream>
#include<algorithm>
#include<string.h>

using namespace std;

int main()
{
	int t,i,len,low,high,a,b,ans;
	string s;
	
	cin>>t;
	
	for(i=0;i<t;i++)
	{
		cin>>s;
		
		ans=0;
		
		len=s.length();
		
		low=0;
		high=len-1;
		
		while(low<high)
		{
			a=s[low];
			b=s[high];
			
			if(a<b)
				ans+=b-a;
			else
				ans+=a-b;
				
			low++;
			high--;
		}
		
		cout<<ans<<endl;
	}
	
	return 0;
	
}
