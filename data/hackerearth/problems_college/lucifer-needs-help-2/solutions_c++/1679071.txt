#include <iostream>
using namespace std;
long long int a[1000001],b[1000001];
int main()
{
	long long int n,temp=0,i,x,y,q,m;
    //cout << "Hello World!" << endl;
    scanf("%lld",&n);
    for(i=1;i<=n;i++) scanf("%lld",&a[i]);
    scanf("%lld",&q);
    while(q--)
    {
    	scanf("%lld%lld%lld",&x,&y,&m);
    	b[x]+=m;
    	if(y+1<=n) b[y+1]-=m;
    }
    for(i=1;i<=n;i++) { temp+=b[i]; printf("%lld\n",a[i]+temp); }
    return 0;
}
