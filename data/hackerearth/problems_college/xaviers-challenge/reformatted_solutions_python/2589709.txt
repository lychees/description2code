
# Read input from stdin and provide input before running code

import string
T = int(raw_input())
for i in xrange(T):
    string = raw_input()
    W = 0
    for letter in string:
        W += ord(letter)
    L = len(string)
    F = W/L
    if F % 2 != 0:
        string = string[::-1]
        print string
    elif F % 2 == 0:
        print string
