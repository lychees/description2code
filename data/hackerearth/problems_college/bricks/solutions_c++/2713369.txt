#include<bits/stdc++.h>
using namespace std;

const int N=123456;
const int mod=1e9+7;

int f[N], g[N], h[N];

inline void add(int &a, int b) {
    a+=b;
    if(a>=mod) a-=mod;
}

inline int mul(int a, int b) {
    return (long long)a*b%mod;
}

void prepare() {
    f[0]=1;
    for(int i=1;i<N;i++) {
        for(int j=1;j<=4;j++) {
            if((i-j)>=0) add(f[i], f[i-j]); 
        }
    }
}

int _pow(int a, int n) {
    int res=1;
    while(n) {
        if(n & 1) res=mul(res, a);
        a=mul(a, a);
        n>>=1;
    }
    return res;
}

int main() {
    prepare();
    int test;
    scanf("%d", &test);
    while(test--) {
        int n, m;
        scanf("%d %d", &n, &m);
        for(int i=1;i<=m;i++) {
            g[i]=_pow(f[i], n);
        }
        h[1]=1;
        for(int i=2;i<=m;i++) {
            h[i]=g[i];
            int temp=0;
            for(int j=1;j<i;j++) {
                add(temp, mul(h[j], g[i-j]));
            }
            add(h[i], mod-temp);
        }
        printf("%d\n", h[m]);
    }
}
