#include<bits/stdc++.h>
using namespace std;
int main()
{
	ios::sync_with_stdio(false);
	long long int  t,i,n,a,b;
	
	cin >> t;
	while(t--)
	{
		long long int res;
		cin >> n;
		
		cin >> a;
		
		res = a;
		for(i=1;i<n;i++)
		{
			cin >> b;
			res = __gcd(res,b);
			
		}
		
		cout << res * n << endl;
		
	}
	
	return 0;
}
