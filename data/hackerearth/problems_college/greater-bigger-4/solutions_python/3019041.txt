from itertools import *
t=int(raw_input())
assert(1<=t and t<=105)
for i in range(t):
    lis=[]
    s=raw_input()
    l=int(len(s))
    assert(1<=l and l<=100)
    for j in sorted(permutations(list(s),l)):
        lis.append("".join(j))
    for i in lis:
        if lis[0]==lis[1]:
            print "no answer"
            break
        elif i>s:
            print i
            break
