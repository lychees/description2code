#define ll long long

long long mod = 1e9 + 7;
long long power (long long base , long long po){

    if(po == 0) return 1;                   // base case
    if(po == 1) return base;                // base case
    long long ans;

    if(po % 2 == 0){                        // check if power is even

        ans = power(base , po / 2);
        return (ans % mod * ans % mod) % mod;

    } else {
        return (base % mod * power(base , po - 1) % mod) % mod;
    }
}
