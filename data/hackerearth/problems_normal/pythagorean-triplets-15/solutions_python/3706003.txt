tri = lambda m,n:(m**2-n**2, 2*m*n, m**2+n**2)

def coprime(x, y):
	while y:
		x, y = y, x%y
	return x==1

n = input()
while n:
	n -= 1
	
	m = input()
	c, i = 0, 2
	l = int(pow(m,0.5)) + 1
	while i<l:
		j = 1
		while j<i:
			if coprime(i,j) and (i**2+j**2)<=m and (i-j)%2:
				c += m/(i**2+j**2)
			j += 1
		i += 1
	print c
