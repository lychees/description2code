#include<iostream>
#include<cmath>
#include<cstring>
using namespace std;
int seive[1000001];
 
int main(){
    //rank();
    for(int i=1;i<=1000000;i++){seive[i]=i;}
    for(int i=2;i<=1000000;i++){
        if(seive[i]==i){
            for(int j=i;j<=1000000;j+=i){
                seive[j]=(seive[j]/i)*(i-1);
            }                 
        }    
    }    
    int t,n;
    cin>>t;
    while(t--){
        cin>>n;
        cout<<seive[n]<<endl;
    }    
    return 0;
}
