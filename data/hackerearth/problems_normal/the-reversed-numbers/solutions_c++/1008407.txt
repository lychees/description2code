#include <iostream>
using namespace std;

int main()
{
    int a,b,s,reverse,reverse1,reverse2,t;
    cin>>t;
    while(t>0)
    {
    cin>>a>>b;
    reverse=0;
    reverse1=0;
    reverse2=0;
    for( ; a!= 0 ; )
   {
      reverse = reverse * 10;
      reverse = reverse + a%10;
      a = a/10;
   }
   for( ; b!= 0 ; )
   {
      reverse1 = reverse1 * 10;
      reverse1 = reverse1 + b%10;
      b = b/10;
   }
   s=reverse+reverse1;
   for( ; s!= 0 ; )
   {
      reverse2 = reverse2 * 10;
      reverse2 = reverse2 + s%10;
      s = s/10;
   }
   cout<<"\n"<<reverse2;
   t--;
    }
    return 0;
}
