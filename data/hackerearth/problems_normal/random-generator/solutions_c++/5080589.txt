#include <bits/stdc++.h>

using namespace std;

const int N = 1e5 + 10;

int a[N];
int n, k, p;

bool ok() {
	for (int i = 0; i + k - 1 < n; i++) {
		if (a[i + k - 1] - a[i] <= 2 * p) {
			return false;
		} 
	}
	return true;
}

int main() {
	int t;
	scanf("%d", &t);
	for (int tt = 1; tt <= t; tt++) {
		scanf("%d %d %d", &n, &k, &p);
		for (int i = 0; i < n; i++) {
			scanf("%d", &a[i]);
		}
		sort(a, a + n);
		if (ok()) {
			printf("YES\n");
		} else {
			printf("NO\n");
		}
	}
	return 0;
}
