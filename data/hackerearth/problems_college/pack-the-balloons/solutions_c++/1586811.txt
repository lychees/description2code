#include <stdio.h>
int main()
{
	int N;
	long l,b,h,min;
	scanf("%d", &N);
	while(N--){
		scanf("%ld%ld%ld", &l, &b, &h);
		min = l;
		if(b<min) min = b;
		if(h<min) min = h;
		if(min%2 == 0){
			printf("%ld.0\n", min/2);
		}else{
			printf("%ld.5\n", min/2);
		}
	}
    return 0;
}
