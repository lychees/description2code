#include <iostream>
#include<math.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	long long a,b,c,ans,x,y,n,temp;
	while(t--)
	{
	ans=0;
	cin>>x>>y>>n;
	ans+=x+y;
	a=x,b=y;
	for(long long i=0;i<n-2;i++)
	{
		if(a-b>0)
			c=a-b;
		else if(a-b==0)
		{
			temp=n-i-1-2;
			if(temp%3==0)
			{
				ans+=((temp/3*(a+b)));
			}
			else if(temp%3==1)
			{
				ans+=((temp/3*(a+b))+a);
			}
			else
				ans+=((temp/3*(a+b))+a+b);
			break;
		}
		else
			c=b-a;
		ans+=c;
		a=b;
		b=c;
	}
	cout<<ans<<endl;
	}
    return 0;
}
