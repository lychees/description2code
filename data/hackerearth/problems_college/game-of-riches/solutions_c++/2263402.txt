#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t; 
	scanf("%d", &t);
	while(t--)
	{
		int ppl, gr;
		cin >> ppl >> gr;
		//int w[ppl];
		vector<int> w(ppl);
		for(int i=0; i<ppl; i++)
		{
			scanf("%d", &w[i]);
		}
		int mine=w[0];
		sort(w.begin(), w.end(), greater<int>());
		int x=0;
		for(int i=0; i<ppl; i++)
		{
			if(mine==w[i])
			{x=i; break;}
		}
		int ans=0;
		ans = x/gr;
		printf("%d\n", ans);
	}
	return 0;
}
