#include <iostream>
#include <cstdio>
#include <string>
#include <string.h>
using namespace std;

char s[1005];
int cnt[31];

int main()
{
    int n; scanf("%d", &n);
    for(int j = 0; j < n; j++) {
    	gets(s);
    	int len = strlen(s);
    	bool g = (s[0] == 'G');
    	for(int i = 0; i < len; i++) {
    		if(s[i] <='9' && s[i] >= '0') {
    			if(s[i+1] > '9' || s[i+1] < '0') {
    				cnt[s[i]-'0'] += (g ? 2:1);
    			}
    			if(s[i+1] <= '9' && s[i+1] >= '0') {
    				cnt[(s[i]-'0')*10+s[i+1]-'0'] += (g ? 2:1);
    				i++;
    			}
    		}
    	}
    }
    int ind = 0;
    for(int i = 1; i < 31; i++)
    	if(cnt[ind] < cnt[i]) 
    		ind = i;
    if(ind == 19 || ind == 20) printf("Date");
    else printf("No Date");
    return 0;
}
