#include<iostream>
using namespace std;
int countSetBits(unsigned int n)
{
  unsigned int count = 0;
  while(n)
  {
    count += n & 1;
    n >>= 1;
  }
  return count;
}
 
/* Program to test function countSetBits */
int main()
{
    int t,n;
    cin>>t;
    while(t--)
    {
    cin>>n;
    int max=0,a=0,b,i;
    for(i=0;i<=n;i++)
    {
      max=countSetBits(i);
      if(max>=a)
       {
       	a=max;
       	b=i;
       }
    }
    cout<<b<<endl;
    }
    
    return 0;
}
