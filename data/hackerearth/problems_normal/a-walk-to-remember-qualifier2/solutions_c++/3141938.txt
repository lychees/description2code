#include <cstdio>
#include <vector>
using namespace std;
int n,m,x,y,i,b[100100];
vector<int> g[100100],o[100100];
bool u[100100],r[100100];
void dfs(int i) {
  u[i]=true;
  for (int j=0; j<g[i].size(); j++) if (!u[g[i][j]]) dfs(g[i][j]);
  b[m++]=i;
}
void ofs(int i, int st) {
  u[i]=false;
  for (int j=0; j<o[i].size(); j++) if (u[o[i][j]]) ofs(o[i][j],st);
  if (i!=st) r[i]=r[st]=true;
}
int main() {
  scanf("%d%d",&n,&m);
  while (m--) {
    scanf("%d%d",&x,&y);
    g[x].push_back(y);
    o[y].push_back(x);
  }
  for (m=0, i=1; i<=n; i++) if (!u[i]) dfs(i);
  for (i=n-1; i>=0; i--) if (u[b[i]]) ofs(b[i],b[i]);
  for (i=1; i<=n; i++) printf("%d%c",int(r[i]),i==n?'\n':' ');
  return 0;
}
