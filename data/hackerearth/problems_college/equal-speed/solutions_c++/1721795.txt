#include<iostream>
#include<cstdlib>
using namespace std;
int main()
{
	ios::sync_with_stdio(false);
	long int n,k,i,j,a[1000];
	cin>>n>>k;
	for(i=0;i<n;i++)
	cin>>a[i];
	for(i=0;i<n;i++)
	{
		for(j=i+1;j<n;j++)
		{
			if(a[i]<a[j])
			{
				int t=a[i];
				a[i]=a[j];
				a[j]=t;
			}
		}
	}
	for(i=0;i<n;i++)
	{
		if(i==k-1)
		{
			cout<<a[i];
			exit(0);
		}
	}
	
	
	return 0;
}
