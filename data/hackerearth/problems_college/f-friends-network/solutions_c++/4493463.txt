#include<bits/stdc++.h>
using namespace std;
#define pb push_back
int visited[100004];
int rstack[100005];
vector<int> v[100005];
bool chk(int vertex)
{
    if(visited[vertex]==0)
    {
        int i;
        visited[vertex]=1;
        rstack[vertex]=1;
        for(i=0;i<v[vertex].size();i++)
        {
            if(visited[v[vertex][i]]==0)
            {
            bool ans=chk(v[vertex][i]);
            if(ans==true)
                return true;
            }
            if(rstack[v[vertex][i]]==1)
                return true;
        }


    }
    rstack[vertex]=0;
    return false;

}
int main()
{
   int t,a,b,c,i,j,k,num;
   cin>>t;
   while(t--)
{
    int vertex;
    cin>>vertex;
    for(i=0;i<=vertex;i++)
    {
        visited[i]=0;
        rstack[i]=0;
    }
    for(i=1;i<=vertex;i++)
    {
        for(j=1;j<=vertex;j++)
        {
            cin>>num;
            if(num==1)
            {
                v[i].pb(j);
               // v[j].pb(i);
            }
        }
    }
    bool ans;
    for(i=1;i<=vertex;i++)
    {
        if(visited[i]==0)
        {
            ans=chk(i);
            if(ans==true)
                break;
        }
    }
    if(ans==true)
        cout<<"Aye"<<endl;

    else
        cout<<"Nae"<<endl;
        for(i=0;i<=vertex;i++)
        {
            v[i].clear();
        }
}
