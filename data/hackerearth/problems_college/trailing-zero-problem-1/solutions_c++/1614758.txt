#include<cstdio>
#include<cmath>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		int x=0,i=1;
		while(n/pow(5,i)!=0)
		{
			x=x+n/pow(5,i);
			i++;
		}
		double z=0;
		for(int j=1;j<=n;j++)
			z=z+log10(j);
		printf("%d\n",(int)(ceil(z))-x);
	}
	return 0;
}
