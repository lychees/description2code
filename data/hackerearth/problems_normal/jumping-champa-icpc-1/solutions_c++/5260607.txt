#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n, q;
    	cin>>n>>q;
    	int a[n], max, min;
    	cin>>a[0];
    	max=min=a[0];
    	for(int i=1;i<n;i++){
    		cin>>a[i];
    		min=(min>a[i])?a[i]:min;
    		max=(max<a[i])?a[i]:max;
    	}
    	cout<<((long long)(max-min)*q)<<endl;
    }
    return 0;
}
