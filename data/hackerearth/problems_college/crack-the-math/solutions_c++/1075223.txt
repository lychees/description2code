{

	llu T;
	scanf("%lld",&T);
	
	while(T--)
	{
		memset(A,0,100);
		memset(B,0,100);
		memset(C,'0',100);
		scanf("%s",A);
		scanf("%s",B);

		for(int i=0;i<len;++i)

			if(A[i]!=B[i])
				C[i]='1';
			printf("%c",C[i]);
		}
		printf("\n");



	}
