T = int(raw_input())
a = [0.0]
for _ in range(T):
    x, y, n, w, p1, p2 = map(int, raw_input().split())
    p1 /= 100.0
    p2 /= 100.0
    d = [(w+1)*[0] for _ in range(n+1)]
    for i in range(n+1):
        d[i][0] = 1
    for i in range(1, n+1):
        for j in range(1, w+1):
            d[i][j] = max(d[i-1][max(j-x, 0)]*p1+d[i-1][j]*(1-p1),
                          d[i-1][max(j-y, 0)]*p2+d[i-1][j]*(1-p2))
    a.append(d[n][w])
for i in range(1, T+1):
    print "%.6f" % (100*a[i])
