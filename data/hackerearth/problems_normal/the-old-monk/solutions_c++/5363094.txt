#include <bits/stdc++.h>

using namespace std;

int T;
long long A[100005] = {0}, B[100005] = {0};

void M()
{
    int N, MaxMon = 0;
    cin >> N;
    //input
    for(int i = 0; i < N; i++)
        cin >> A[i];
    for(int i = 0; i < N; i++)
        cin >> B[i];

    int j;
    for(int i = 0; i < N; i++)
    {
        for(j = i; j < N; j++)
        {
        	if(A[i] > B[j])
        		break;
            if(B[j] >= A[i] &&  j >=i )
            {
                if(MaxMon < (j-i))
                    MaxMon = (j-i);
            }
        }
    }
    cout << MaxMon << endl;
}


int main()
{
    cin >> T;

    for(int i = 0; i < T; i++)
    {
         M();
    }
}
