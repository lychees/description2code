#include <iostream>
using namespace std;
int main()
{
    int n,p[100],sum,min=-1,i;
    cin>>n;
   for(i=0;i<n;i++)
    {
    	cin>>p[i];
    }
    sum=0;
    for(i=0;i<n;i++)
    {
    	sum=sum+p[i];
    }
    if(sum%2==0)
    {
    	for(i=0;i<n;i++)
    	{
    		if((min==-1)&&(p[i]%2==1))
    		{
    			min=p[i];
    		}
    		if((p[i]%2==1) && min>p[i])
    		{
    			min=p[i];
    		}
    	
    	}
    }
    if(sum%2==1)
    {
    	cout<<sum;
    }
    else if(min==-1)
    {
    	cout<<":(";
    }
    else
    {
    	cout<<(sum-min);
    	
    }
    return 0;
}
