#include <bits/stdc++.h>
#define ll long long int
using namespace std;

const ll mod=1000000009;

ll power (ll x, ll y)
{
	ll res=1;
	while (y>0)
	{
		if (y&1)
		{
			res=(res*x)%mod;
		}
		x=(x*x)%mod;
		y/=2;
	}
	return res;
}

int main()
{
    ll t,n,ans;
    scanf ("%lld",&t);
    while (t--)
    {
    	scanf ("%lld",&n);
    	ans=(power(10,n)-power(8,n));
    	if (ans<=0)
    		ans+=mod;
    	ans=(ans*power(2,mod-2))%mod;
    	printf ("%lld\n",abs(ans));
    }
    return 0;
}
