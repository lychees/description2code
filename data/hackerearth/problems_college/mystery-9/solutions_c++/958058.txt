#include <bits/stdc++.h>

using namespace std;

int a[] = {31,28,31,30,31,30,31,31,30,31,30,31};

int main()
{
    int t,n;
    cin >> t;
    while ( t-- ) {
        cin >> n;
        int sum = 0;
        for ( int i = 0; i < 12; i++ ) {
            sum += a[i];
            if ( sum >= n ) {
                cout << i+1 << endl;
                break;
            }
        }
    }
    return 0;
}
