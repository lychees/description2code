#include <bits/stdc++.h>
using namespace std;

typedef long long int ll;

char a[1001][1001];

void filli(ll c,ll b)
{
	ll i,j;
	
	for(i=1;i<=c;i++)
	{
		for(j=1;j<=b;j++)
		a[i][j]=(a[i][j]=='0')?'1':'0';
	}
}

int main()
{
	//ios::sync_with_stdio(false);
	
	ll t,n,m,i,j,k,x,y;
	char c;
	
    cin>>t;
    
    while(t--)
    {
        getchar();
    	cin>>n>>m;
    	
    	for(i=1;i<=n;i++)
    	{
    		//getchar();
    		for(j=1;j<=m;j++)
    		{
    			cin>>c;
    			a[i][j]=c;
    		}
    	}
    	
    	ll ctr=0;
    	i=m;
    	j=n;
    	while(i>=1 && j>=1)
    	{
    		for(k=j;k>=1;k--)
    		{
    			if(a[k][i]=='0')
    			{
	    			filli(k,i);
	    			ctr++;
	    		
    			}
    		}
    		for(k=i-1;k>=1;k--)
    		{
    			if(a[j][k]=='0')
    			{
	    			filli(j,k);
	    			ctr++;
	    			
    			}
    		}
    		i--;
    		j--;
    	}
    	
    	cout<<ctr<<endl;
    	
    }
    
    return 0;
}
