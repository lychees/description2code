from sys import stdin
t = int(stdin.readline())
for _ in xrange(t):
    n = int(stdin.readline())
    a = map(int, stdin.readline().split())
    di = {}
    for i in xrange(n):
        di[a[i]] = i + 1
    valid = 1
    # print di
    for i in xrange(n):
        if a[i] != di[i+1]:
            valid = 0
    if valid:
        print "inverse"
    else:
        print "not inverse"
