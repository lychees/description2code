

def func(n, x, t):
    n = len(x)

    count = 0
    for i in range(1, n):
        if (x[i] <= x[i-1]):
            temp = ((x[i-1]-x[i])/t)+1
            x[i] = x[i]+temp*t
            count = count+temp
    print count


test = int(raw_input())

while test > 0:

    n, x = map(int, raw_input().split(" "))
    list1 = map(int, raw_input().split(" "))
    func(n, list1[:], x)
    test = test-1
