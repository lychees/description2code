#include<iostream>
using namespace std;
int main()
{
	char ch;
	int a,ld=0, ls=0,tmpld=0,tmpls=0;
	cin>>a;
	for(a; a>0; a--)
	{
		for(int x=0; x<1440; x++)
		{
			cin>>ch;
			if(ch == 'C')
			{
				tmpls ++;
				tmpld ++;
			}
			else
			{
				if(tmpls > ls) ls = tmpls;
				if(tmpld > ld) ld = tmpld;
				tmpls = 0;
				tmpld = 0;
			}
		}
		if(tmpls > ls) ls = tmpls;
		if(tmpld > ld) ld = tmpld;
		tmpls = 0;
		if(ch != 'C') tmpld = 0;
	}
	cout<<ls<<' '<<ld;
	return 0;
}
