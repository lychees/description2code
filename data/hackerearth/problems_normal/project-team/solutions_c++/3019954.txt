#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,i;
		scanf("%d",&n);
		int arr[n];
		for(i=0;i<n;i++)
			scanf("%d",&arr[i]);
		
			
		sort(arr,arr+n);
		
		int ans[n];
		for(i=0;i<n/2;i++)
		{
			ans[i]=arr[i]+arr[n-i-1];
		}
	//	for(i=0;i<n/2;i++)
	//		printf("%d ",ans[i]);
		
		sort(ans,ans+n/2);
		
		printf("%d\n",ans[n/2-1]-ans[0]);
	}
	return 0;
}
