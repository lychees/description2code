#include <iostream>
using namespace std;
#define ll long long
#include <vector>

int main()
{	ll n,k,q,p,m=0,i;
	cin>>n>>k;
	vector<ll> r;
	for(i=0;i<=k+10;++i)
		r.push_back(0);
	for(i=0;i<n;++i)
	{	cin>>q;
		r[q]++;
	}
	for(i=0;i<=k+10;++i)
	{	if(r[i]>m)
		{	m=r[i];
			p=i;
		}
	}
	cout<<p<<"\n";
    return 0;
}
