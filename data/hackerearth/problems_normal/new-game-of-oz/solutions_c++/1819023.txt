#include<bits/stdc++.h>
using namespace std;
int main() {
    long t;
    scanf("%ld",&t);
    while(t--) {
        long long n , a[1005] , i , k , j , ans = 0;
        scanf("%lld",&n);
        for( i = 0 ; i<n ;i++) {
            scanf("%lld",&a[i]);
        }
        sort(a,a+n);


        for( i = 0 ; i< n ; i++) {
            if(a[i] == (a[i+1]-1)) {
                ans++;
                i++;
            }
            else {
                ans++;

            }
        }
        printf("%ld\n",ans);
    }
}
