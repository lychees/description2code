#include <iostream>
using namespace std;

int a[] = {1, 3, 5};

int solve(int m, int n) {
	if(n == 0)
		return 1;
	if(n < 0)
		return 0;
	if(m <= 0 && n >= 1)
		return 0;
	return solve(m, n - a[m - 1]) + solve(m - 1, n);
}
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin >> t;
    while(t--) {
    	int n;
    	cin >> n;
    	cout << solve(3, n) << endl;
    }
    return 0;
}
