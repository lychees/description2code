#include <iostream>
using namespace std;
 
int main()
{
    int t;
    cin >>t;
    while(t--)
    {
    	 long int n,a[100005],sell[100005]={0},ans=0;
    	cin >>n;
    	for(int i=0;i<n;i++)
    	 {cin >>a[i];}
    	 for(int i=n-1;i>=0;i--)
    	 {
    	 	if(i==n-1)
    	 	sell[i]=a[i];
    	 	else
    	 	{
    	 		if(a[i]<sell[i+1])
    	 			sell[i]=sell[i+1];
    	 		else
    	 		 sell[i]=a[i];
    	 	}
    	 }
    	  
    	 for(int i=0;i<n;i++)
    	 {
    	 	
    	 	ans+=sell[i]-a[i];
    	 	
    	 }
    	 cout << ans <<endl; 
    	 
    }
    return 0;
}
