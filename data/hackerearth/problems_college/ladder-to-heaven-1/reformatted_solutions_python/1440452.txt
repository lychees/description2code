def fib(n):
    ar = []
    a = 1
    b = 2
    while b <= n:
        ar.append(a)
        ar.append(b)
        a, b = b, a+b
    return ar


for i in range(int(raw_input())):
    n = int(raw_input())
    t = fib(n)
    t = list(set(t))
    s = 0
    for j in t:
        if j % 2 == 0:
            s = s+j
    print s
