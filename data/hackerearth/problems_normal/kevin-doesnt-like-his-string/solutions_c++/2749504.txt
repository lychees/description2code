#include <bits/stdc++.h>

using namespace std;

int N, K;
char S[100001];
vector<int> pos[26];
int spare[26];
int flag[100001];

int main()
{
    scanf("%d%d", &N, &K);
    scanf("%s", S);
    for(int i=0; i<N; i++)
        pos[S[i]-'a'].push_back(i);
    for(int i=0; i<N; i++)
    {
        for(int j=0; j<S[i]-'a'; j++)
        {
            if(spare[j])
            {
                if(!flag[i] && K==0)
                    continue;
                spare[j]--;
                if(!flag[i])
                    flag[i]=1, K--;
                spare[S[i]-'a']++;
                S[i]=j+'a';
                break;
            }
            if(!pos[j].empty() && pos[j].back()>i)
            {
                if(K+flag[i]<2)
                    continue;
                spare[S[i]-'a']++;
                if(!flag[i])
                    flag[i]=1, K--;
                S[i]=j+'a';
                flag[pos[j].back()]=1;
                K--;
                S[pos[j].back()]='z'+1;
                pos[j].pop_back();
            }
        }
    }
    printf("%s\n", S);
    return 0;
}
