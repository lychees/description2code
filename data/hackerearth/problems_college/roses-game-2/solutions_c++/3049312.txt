#include <bits/stdc++.h>

using namespace std;

map<int , int> mp;

int main()
{
	int n,t,x;
	scanf("%d",&t);
	while(t--) {
		scanf("%d",&n);
		for(int i = 0; i < n; i++) {
			scanf("%d",&x);
			mp[i] = x;
		}
		bool flag = false;
		int j = 0, count = 0;
		while(1) {
			if(j == -1) {
				flag = true;
				break;
			}
			j = mp[j];
//			cout << j << endl;
			count++;
			if(count > n) {
				break;
			}
		}
		if(flag == true) {
			cout << "Win" << endl;
		} else {
			cout << "Lose" << endl;
		}
	}
	return 0;
	
}
