#include<iostream>
using namespace std;

void count(int n, int k){
	long long int **p;
	p = new long long int*[n + 1];
	for (long long int i = 0; i < (n + 1); i++)
		p[i] = new long long int[k + 1];
	for (long long int i = 0; i < (n + 1); i++)
		p[i][0] = 1;
	for (long long int i = 0; i < (k + 1); i++)
		p[0][i] = 1;
	for (long long int i = 1; i < (n + 1); i++)
		for (long long int j = 1; j < (k + 1); j++)
			p[i][j] = (p[i - 1][j] + p[i][j - 1])%1000000007;
	cout << (p[n][k] - 1);
}
int main()
{
	int n, k;
	cin >> n >> k;
	count(n, k);
	return 0;
}
