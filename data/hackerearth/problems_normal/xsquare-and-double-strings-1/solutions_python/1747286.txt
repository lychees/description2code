T = int(raw_input())

for tc in range(T):
    S = raw_input().strip()

    NoMatch = True
    for i, chr1 in enumerate(S):
        for chr2 in S[i + 1 :]:
            if chr1 == chr2:
                NoMatch = False
                print 'Yes'
                break
        if not NoMatch:
            break
    if NoMatch:
        print 'No'
