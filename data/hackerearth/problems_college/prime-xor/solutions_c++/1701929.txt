#include<bits/stdc++.h>
using namespace std;

const int N=1000010;

bool isPrime[N];
bool prs[N];

void pre() {
    for(int i=2;1ll*i*i<=1e10;i++) {
        if(isPrime[i]==false) {
            for(int j=2*i;1ll*j*j<=1e10;j+=i) {
                isPrime[j]=true;
            }
        }
    }
    return;
}
int main() {
    pre();
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        int l,r;
        scanf("%d%d",&l,&r);
        memset(prs,0,sizeof(prs));
        for(int i=2;1ll*i*i<=r;i++) {
            if(isPrime[i]==false) {
                int rs=l/i;
                if(l%i) {
                    rs=(rs+1)*i;
                } else {
                    rs=rs*i;
                }
                if(rs==i) {
                    rs+=i;
                }
                for(int j=rs;j<=r;j+=i) {
                    prs[j-l]=true;
                }
            }
        }
        int odd=0;
        int even=0;
        for(int i=0;i<=(r-l);i++) {
            if(prs[i]==false&&i+l!=1) {
                int resq=i+l;
                if(resq&1) {
                  odd++;
                } else {
                  even++;
                }
            }
        }
        printf("%lld\012",1ll*odd*even);
    }
    return 0;
}
