#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
	long long int d,c,n,g[3000],sum=0,max=0;

	cin>>d;
	while(d--)
	{
	cin>>n>>c;
	for(int i=0;i<n;i++)
	{
		cin>>g[i];
	}
	sort(g,g+n);
	
	for(int i=0;i<n;i++)
	{
		//if(g[i]==c)cout<<max;
		//else 
		
		sum=sum+g[i];
		max++;
		if(sum>c)
		{
			break;
		}
		
	}
	cout<< max-1<<endl;
	sum=0;
	max=0;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
