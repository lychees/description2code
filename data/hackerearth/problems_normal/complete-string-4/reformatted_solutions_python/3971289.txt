from string import ascii_lowercase

num = int(raw_input())
for x in range(0, num):
    string = raw_input()
    flag = True
    if len(string) < 26:
        print 'NO'
    else:
        for c in ascii_lowercase:
            if c not in string:
                flag = False
                break
        if not flag:
            print 'NO'
        else:
            print 'YES'
