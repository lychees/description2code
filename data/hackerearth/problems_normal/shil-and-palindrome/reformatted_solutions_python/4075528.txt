b = ['a', 'b', 'c', 'd', 'e', 'f', 'g', 'h', 'i', 'j', 'k', 'l', 'm',
     'n', 'o', 'p', 'q', 'r', 's', 't', 'u', 'v', 'w', 'x', 'y', 'z']
s = raw_input()
a = [0]*26
for i in range(0, len(s)):
    a[ord(s[i])-ord('a')] = a[ord(s[i])-ord('a')]+1
s1 = ""
s2 = ""
for j in range(26):
    if (a[j] % 2 == 0):
        s1 += b[j]*(a[j]/2)
    else:
        s1 += b[j]*(a[j]/2)
        s2 += b[j]
s3 = s1+s2+s1[::-1]
if (s3 == s3[::-1]):
    print(s3)
else:
    print(-1)
