
#include<bits/stdc++.h>
using namespace std;
map<int,int> singles;
map<int,int> doubles;
int main(){
int n;
cin>>n;
int ar[n];
for(int i=0;i<n;i++){
	cin>>ar[i];
}
int googly=0;
vector<int> single;

//int doubles[n];
//memset(doubles,0,sizeof(double));

for(int i=0;i<n;i++){
       // cout<<i<<endl;
	if(abs(ar[i])%3==0&&singles[ar[i]/3]){googly++;}
	else{
		for(int k=0;k<single.size();k++){
			if(doubles[ar[i]-single[k]]){ googly++; break;}
		}
	}
	if(!singles[ar[i]]) single.push_back(ar[i]);
	singles[ar[i]]=1;
	for(int k=0;k<single.size();k++){
		doubles[ar[i]+single[k]]=1;
	}


}
cout<<googly;

}
