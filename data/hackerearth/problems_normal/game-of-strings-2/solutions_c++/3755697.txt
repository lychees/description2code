#include <bits/stdc++.h>
using namespace std;

int z[1000010];
string s;

void computez()
{
    int left,right,i;

    z[0] = 0;
    left = 0;
    right = 0;

    for (i = 1; i < s.size(); i++) {
            //cout <<left<<" "<<right<<endl;
            if (right < i) {
                left = i;
                right = i;
            } else {
                if (i + z[i-left] < right) {
                    z[i] = z[i-left];
                    continue;
                } else {
                    left = i;
                }
            }

            while (right < s.size() && s[right] == s[right-left])
                right++;
            z[i] = right - left;
            right--;
    }

}

int main()
{
    int t,i,k,si,ans,cur;
    string finans;
    cin >> t;

    while (t--) {
        memset(z,0, sizeof(z));
        cin >> s >> k;
        si = s.size();
        computez();
        ans = 0;

        for (i = 0; i < k; i++) {
            cur = 0;
            //cout <<z[i]<<" "<<(i+z[i])<<" "<<(si - z[i] + 1)<<" "<<z[si-z[i]+1]<<" ";
            if (z[i]) {
                if (i != 0) {
                    cur = min(k - i, z[i]);
                }
            }
            //cout <<cur<<endl;
            if (cur > ans) {
                ans = cur;
            }
        }
        int ma = -1;
        for (i = 0; i < si; i++) {
            if (i + z[i] == si && z[i] <= ans) {
                ma = max(ma, z[i]);
            }
        }

        ans = ma;
        ans = max(ans,0);

        if (ans != 0) {
            finans = s.substr(0,ans);
        } else {
            finans = "Puchi is a cheat!";
        }
        printf("%s\n", finans.c_str());
    }

    return 0;
}
