#include<iostream>
using namespace std;
int main()
{
	
	int t;
	cin>>t;
while(t--)
{
	long long int num;
	long long int m,rem,a,sum=0,last,no;
	cin>>num;
	if(num<5)
	cout<<num<<endl;
	else
	{

	rem=num%3;
	m=num/4;

	m=m-rem;
	if((m)%3==2)
		a=m-1;
	else
		a=m;

	sum=((a+1)*(a))/2 - ((a/3)*(4+(a/3 -1)*3))/2 + (a+1-a/3)*(rem);
	last=num-rem;
	no=(last-a)/3;
	sum+=((no)*(no-1)*2)/2;
	if((last-a)%3!=0)
		{
			sum+=no;
		}


sum+=num-(num/3);
if(rem==0)
sum+=1;
cout<<sum<<endl;
}
}

}
