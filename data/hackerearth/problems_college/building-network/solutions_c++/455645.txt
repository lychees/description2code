#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int t,n,k,i;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
		long long int sum=0,min=10000000000;
		for(i=0;i<n;i++)
		{
		   scanf("%d",&k);
		   if(k<min)
		   min=k;
		    sum+=k;
	    }
		sum+=(min*(n-2));
		printf("%lld\n",sum);
		
	}
	return 0;
}
