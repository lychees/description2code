//HackerEarth Problem
#include<stdio.h>
#include<string.h>
#include<math.h>
#define LL long long int
#define f_in(st) freopen(st,"r",stdin);
#define f_out(st) freopen(st,"w",stdout);
int main()
{
    //f_in("sample_input.txt");
    //f_out("sample_output.txt");
    int test;
    scanf("%d",&test);
    while(test--)
    {
        LL N;
        scanf("%lld",&N);
        if(N==0) printf("Worng Test Case!!!!");
        if(N==1)
            printf("Hacker\n");
        else if(N==2)
            printf("Earth\n");
        else
        {
            LL Sum=2,Count=1,power2=2,Prev_Sum=0;
            while(Sum<N)
            {
                Prev_Sum=Sum;
                power2=power2*2;
                Count++;
                Sum+=power2*Count;

            }
            LL Number;
            LL Local_Sum=N-Prev_Sum;
            LL Multiple=(Local_Sum)/(Count);
            if(Local_Sum%Count==0)
                Number=Local_Sum/Count;
            else
                Number=Local_Sum/Count+1;
            Number--;
            int arr[Count];
            memset(arr,0,sizeof(arr));
            int pos=Count-1;
            while(Number>0)
            {
                if(Number&1) arr[pos--]=1;
                else arr[pos--]=0;
                Number=Number/2;
            }
            int i;
            //printf("%lld %lld  ",Local_Sum,Multiple*Count);
            //for(i=0;i<Count;i++)
            //    printf("%d",arr[i]);
            LL Nth_Digit;
            if(Local_Sum%Count==0)
                Nth_Digit=Count-1;
            else
                Nth_Digit=Local_Sum-Multiple*Count-1;
            if(arr[Nth_Digit])
                printf("Earth\n");
            else
                printf("Hacker\n");
        }
    }
    return 0;
}
