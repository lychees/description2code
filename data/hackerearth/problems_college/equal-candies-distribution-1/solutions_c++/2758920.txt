#include <bits/stdc++.h>
using namespace std;
int a[10005];
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n;
    	scanf("%d",&n);
    	int min=INT_MAX;
    	for(int i=0;i<n;i++)
    	{
    	scanf("%d",&a[i]);
    	if(a[i]<min)
    	min=a[i];
    }
    int steps=0;
    	for(int i=0;i<n;i++)
    	{
    		if(a[i]>min)
    		{
    			int x=a[i]-min;
    			
    			steps+=(x/5)+(x%5)/2+((x%5)%2);
    		}
    	}
    	printf("%d\n",steps);
    	}
}
