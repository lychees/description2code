#include <bits/stdc++.h>
using namespace std;
int iss(int m)
{
//	printf("checking for %d\n",m);
	int j;
	for(j=2;j*j<=m;j++)
	if(m%j==0)return 0;
//	int m = 0;
int n = 0;
	int x= m;
	while(x)
	{
		n = n*10 + x%10;
		x/=10;
	}
	if(n==m)return 1;
	return 0;
}
int main()
{
   // printf("Hello World!\n");
   int t;
   scanf("%d",&t);
   while(t--)
   {
   	int n;
   	scanf("%d",&n);
   
   	while(1)
   	{
   		if(iss(n))break;
   		n++;
   		
   	}
   	printf("%d\n",n);
   }
    return 0;
}
