#include<iostream>
using namespace std;

int main()
{
	unsigned long long int i,j,t;
	cin>>t;
	string in,in2;
	while(t--)
	{
		cin>>in;
		in2=in;
		for(i=0;i<in.length();i++)
		{
			if(in[i]=='B')
			{
				if(i>1)
				{
					in2[i-1]='x';
					in2[i-2]='x';
				}
				else
				if(i==1)
				in2[0]='x';
			}
			else
				if(in[i]=='S')
				{
					if(i==in.length()-1)
					{
						in2[i-1]='x';
					}
					else
						if(i==0)
						{
							in2[1]='x';
						}
						else
						{
							in2[i-1]='x';
							in2[i+1]='x';
						}
				}
		}
		
		j=0;
		for(i=0;i<in.length();i++)
		{
			if(in2[i]=='-')
				j++;
		}
		cout<<j<<"\n";
	}
	return 0;
}
