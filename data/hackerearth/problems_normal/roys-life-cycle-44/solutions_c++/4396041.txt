#include <bits/stdc++.h>
using namespace std;

int find_longest(const string &s) {
	int ct = 0, res = 0;
	for (const char &c: s) {
		if (c == 'C') ct++;
		else {
			res = max(res, ct);
			ct = 0;
		}
	}
	return max(res, ct);
}

int main()
{
	ios_base::sync_with_stdio(false);
	int n;
	cin >> n;
	string s, a = "";
	int res = 0;
	while (n--) {
		cin >> s;
		res = max(res, find_longest(s));
		a += s;
	}
	cout << res << " " << find_longest(a) << endl;
   	return 0;
}
