#include <iostream>
#include <algorithm>
using namespace std;

int print(int a[],int n,int x)
{
    int i;
    for(i==n-1;i>=0;i--)
    {
        if(a[i]<x)
        {
            return i;
        }
    }
}

int main()
{
    int x,i,n;
    cin>>n;
    int a[n];
    for(i=0;i<n;i++)
    {
        cin>>a[i];
    }

    cin>>x;
    sort(a,a+n);

    print(a,n,x);

    int j,k;
    bool spam=false;

    for(i=0;i<=n-3;i++)
    {
        for(j=i+1;j<=n-2;j++)
        {
            for(k=j+1;k<=n-1;k++)
            {
                if((a[i]+a[j]+a[k] == x))
                {
                    spam = true;
                }
            }
        }
    }

    if(spam)
    {
        cout<<"True"<<endl;
    }
    else{
        cout<<"False"<<endl;
    }

    return 0;
}
