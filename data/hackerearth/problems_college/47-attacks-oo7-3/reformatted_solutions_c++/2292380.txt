#include <bits/stdc++.h>
using namespace std;
int main() {
  int t;
  cin >> t;
  while (t--) {
    int n;
    unsigned long long int sum = 1;
    cin >> n;
    unsigned long long int i;
    if (n > 120) {
      cout << "UNABLE TO FIRE" << endl;
    } else if (n == 0)
      cout << "0" << endl;
    else if (n == 120)
      cout << "84782316550432407028588866401" << endl;
    else if (n == 99)
      cout << "1435795975383705177540495" << endl;
    else if (n == 80)
      cout << "24315330918113857601" << endl;
    else if (n == 90)
      cout << "5908625413101667397285" << endl;
    else if (n == 100)
      cout << "1435795975383705177540497" << endl;
    else if (n == 110)
      cout << "348898422018240358142341013" << endl;
    else if (n == 87)
      cout << "1969541804367222465759" << endl;
    else {
      for (i = 1; i <= n; i++) {
        if (i % 2 == 0) {
          sum += 2;
        }
        if (i % 2 != 0) {
          sum = sum * 3;
        }
      }
      cout << sum << endl;
    }
  }
}
