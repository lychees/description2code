# Read carefully: "If *any sub-string* of the word"
t = int(raw_input().strip())

while t:
    t -= 1

    s = raw_input().strip()
    sub = raw_input().strip()

    d = {}

    for i in s:
        d[i] = d.get(i, 1)
    for i in sub:
        if d.get(i, None):
            d[i] = 0

    if 0 in d.values():
        print "YES"
    else:
        print "NO"
