#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	string a, b;
	while(t--) {
		a.clear();
		b.clear();
		cin>>a>>b;
		int len1 = a.length();
		int len2 = b.length();
		int i = 0, j= 0;
		int count = 0;
		while(i<len1 && j<len2) {
			if(a[i++] == b[j++]) count++;
		}
		cout<<count<<endl;
	}
    return 0;
}
