#include <bits/stdc++.h>
using namespace std;


    void maxheap(int arr[],int i,int N)
    {
    	int left = i*2;
    	int right =i*2 + 1;
    	int largest;
    	if(left<=N && arr[left]>arr[i])
	    {
	    	largest=left;
    	}
    	else 
    	largest=i;
    	
    	if(right<=N && arr[right]>arr[largest] )
    	{
    		largest=right;
    	}
    	if(largest!=i)
    	{
    		swap(arr[i],arr[largest]);
    		maxheap(arr,largest,N);
    	}
    }
    
    void buildmax(int arr[],int N)
    {
    	for(int i=N/2;i>=1;i--)
    	{maxheap(arr,i,N);}
    }
    
int main()
{
	int M,N,arr[1000000],sum=0;
	// N: no of seating row
	// M: no of fans
	cin>>N>>M;
	for(int i=1;i<=N;i++)
	{
		cin>>arr[i];
	}
	
	buildmax(arr,N);
	for(int i=1;i<=M;i++)
	{
		sum=sum+arr[1];
		arr[1]=arr[1]-1;
		maxheap(arr,1,N);
	}
	cout<<sum;
    return 0;
}
