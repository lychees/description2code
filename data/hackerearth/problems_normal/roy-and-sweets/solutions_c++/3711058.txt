#include <iostream>
using namespace std;

int S[1000001];

int main()
{
    int n;
    cin>>n;
    int maxs=0;
    for(int i=0; i<n; i++)
    {
    	int s,q;
    	cin>>s>>q;
    	S[s]=q;
    	maxs=max(maxs,s);
    	
    }
    
    for(int i=maxs-1; i>=1; i--)
    {
    	S[i]=max(S[i],S[i+1]);
    }
    int g;
    cin>>g;
    long long ans=0;
    for(int i=0; i<g; i++)
    {
    	int m;
    	cin>>m;
    	int p=S[m]/m;
    	ans=ans+100*p;
    }
    cout<<ans<<endl;
    
    
    return 0;
}
