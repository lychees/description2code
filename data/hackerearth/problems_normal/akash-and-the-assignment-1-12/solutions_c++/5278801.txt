#define maxlen 50001
#define ascii_a 97
using namespace std;



char lexSmall(string s, int l, int r, int k, int a[maxlen][26])
{
long cnt = 0;
	int r1 = r - 1, r2 = l - 2;

	for (int i = 0; i<26; i++)
	{
		if (a[r1][i] != 0)
		{
			if (r1 == r2)
				return i + ascii_a;
			else if (r2 < 0)
				cnt += a[r1][i];
			else
				cnt += (a[r1][i] - a[r2][i]);

			if (cnt >= k)
				return i + ascii_a;
		}
	}
	return 0;
}
 
int main()
{
	int n = 0, q = 0, l = 0, k = 0, r = 0;
	int a[maxlen][26];
	memset(a,0,sizeof(a));
	string s;
	cin >> n >> q;
 
	cin>>s;
 
 	for(int i=0; i<n; i++)
 	{
 		++a[i][s[i]- ascii_a];
 		if(i<(n-1))
 		memcpy(a[i+1],a[i],sizeof(a[i]));
 	}
 	
	for (int i = 0; i<q; i++)
	{
		cin >> l; cin >> r; cin >> k;
 
		if (k <= r - l + 1)
		{
			cout << lexSmall(s, l, r, k, a) << endl;
		}
		else
			cout << "Out of range" << endl;
	}
	return 0;
}
