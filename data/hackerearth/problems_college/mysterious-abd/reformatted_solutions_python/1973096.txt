def isp(x, n, m):
    if m == 0:
        return True
    if n <= 0 and m > 0:
        return False
    if x[n-1] > m:
        return isp(x, n-1, m)
    return isp(x, n-1, m) or isp(x, n-1, m-x[n-1])


for _ in xrange(input()):
    n, m = map(int, raw_input().split())
    x = map(int, raw_input().split())
    print "Yes" if isp(x, n, m) else "No"
