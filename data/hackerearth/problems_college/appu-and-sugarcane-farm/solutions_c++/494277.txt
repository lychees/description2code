#include <iostream>
using namespace std;



int numchk(long long int num)
{
	
	return (num&(num-1))!=0;
}

int main()
{
   int n;
   cin>>n;
   long long int a[51],min=1000000001;
   for(int i=0;i<n;i++)
   {
   cin>>a[i];
   min=min>a[i]?a[i]:min;
   }
   int check=0;
   for(int i=0;i<n;i++)
   {
   	if(a[i]%min!=0)
   	{
   		check=1;
   		break;
   	}
   	else
   	{
   		check = numchk(a[i]/min);
   		if(check==1)
   		break;
   	}
   }
   
   if(check==1)
   cout<<"NO";
   else
   cout<<"YES";
   
    return 0;
}
