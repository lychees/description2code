#include<cstdio>
#include<iostream>
#include<vector>
#include<algorithm>
#include<string>
#include<cstring>
using namespace std;

typedef long long LL;
typedef vector<int> VI;

#define REP(i,n) for(int i=0, i##_len=(n); i<i##_len; ++i)
#define EACH(i,c) for(__typeof((c).begin()) i=(c).begin(),i##_end=(c).end();i!=i##_end;++i)
#define eprintf(s...) fprintf(stderr, s)

template<class T> inline void amin(T &a, const T &b) { if (a>b) a=b; }
template<class T> inline void amax(T &a, const T &b) { if (a<b) a=b; }

int T;
int N, M;
string S[55];
char buf[111];
LL len[55];

char rec(int n, LL x) {
    if (2 * len[n] <= x) {
	return S[n][x - 2*len[n]];
    } else if (len[n] <= x) {
	return rec(n-1, 2*len[n]-x-1);
    } else {
	return rec(n-1, x);
    }
}
int main() {
    scanf("%d", &T);
    REP ($, T) {
	scanf("%d%d", &N, &M);
	REP (i, N) {
	    scanf("%s", buf);
	    S[i] = string(buf);
	    len[i+1] = 2 * len[i] + S[i].size();
	}
	REP ($$, M) {
	    LL X;
	    scanf("%lld", &X);
	    char c = rec(N-1, X);
	    putchar(c);
	}
	putchar('\n');
	
    }
    return 0;
}
