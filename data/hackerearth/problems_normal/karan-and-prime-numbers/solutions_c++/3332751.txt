#include<bits/stdc++.h>
#define ll long long int
#define MAX 10000009
#define MOD 1000000007
using namespace std;

bool isPrime[MAX+5];
ll prime[MAX+5];

void seive_N_logN(ll N)
{
    ///calculate prime upto N in NlogN time
    memset(isPrime,true,sizeof isPrime);

    isPrime[1]=false;
    for (ll i = 4; i<= N; i=i+2)
        isPrime[i]=false;

    for (ll i = 3; i * i<= N; i=i+2)
    {
        if (isPrime[i])
        {
            for (ll j = i * i; j <= N; j += i)
                isPrime[j] = false;
        }
    }
    prime[0]=0;
    for(ll i=1; i<MAX; i++)
    {
        if(isPrime[i])
        {
            prime[i]=prime[i-1]+i;
        }
        else
            prime[i]=prime[i-1];
    }
}

int main()
{
    ll t,T,n,m;
    seive_N_logN(MAX);
    scanf("%lld",&T);
    for(t=1; t<=T; t++)
    {
        scanf("%lld %lld",&n,&m);
        if(n>m)
            swap(n,m);
        printf("%lld\n",prime[m]-prime[n-1]);
    }
    return 0;
}
