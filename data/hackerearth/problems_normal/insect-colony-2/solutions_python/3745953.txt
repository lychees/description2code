T = int(raw_input())
for i in range(T):
	N = map(int, raw_input().split())
	n = N[0]
	#print N
	ans = 0
	for k in range(1,n+1):
		a = N[k]
		#print a
		if a % 2 != 0:
			ans += 1
	
	#print ans
	if ans % 2 == 0:
		print "Yes"
	else:
		print "No"
