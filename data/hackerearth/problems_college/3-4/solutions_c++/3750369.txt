#include <iostream>
using namespace std;

int main()
{
    int t,n,k,a[1000];
    cin>>t;
    for(int i=0;i<t;i++)
    { int count;
    	cin>>n>>k;
    	for(int i=0;i<k;i++)
    	{
    		cin>>a[i];
    	}
    	int j=1;
    	while(j<=n)
    {  count=0;
    	for(int i=0;i<k;i++)
    	{
    		if(a[i]==j)count++;
    	}
    	if(count%2!=0)
    	{
    		cout<<j<<endl;
    		break;
    	}
    	j++;
    }
    	
    }
    return 0;
}
