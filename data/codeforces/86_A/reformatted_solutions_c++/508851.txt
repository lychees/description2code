using namespace std;

typedef long long LL;
typedef pair<int, int> P;

int main() {

  string ln;
  long long A, B;

  getline(cin, ln);
  istringstream(ln) >> A >> B;

  long long b = 10;

  while (b <= B)
    b *= 10;

  long long p = max(A, min(b / 2, B));

  cout << p * (b - p - 1) << endl;

  return 0;
}
