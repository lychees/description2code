#include<bits/stdc++.h>
typedef long long ll;
using namespace std;
ll n,m,a[1000000],h,l,coun=0,r=1;
ll bs()
{
	ll mi;
	while(l<h)
	{mi=l+(h-l)/2;
	coun=0;
	r=1;
	 for(int i=0;i<=n;i++)
	 { if(coun+a[i]<=mi)
	    coun+=a[i];
	    else
	    {coun=a[i];
	     r++;
	    }
	 }
	 if(r<=m)
	  h=mi;
	  else
	  l=mi+1;
	}
	return l;
}
	

int main(){
	cin >> n >> m;
	for (int i = 1; i <=n;i++)
	{cin>>a[i];
	l=max(l,a[i]);
	
	}
	h=1e18;
	cout<<bs()<<endl;
	return 0;
}
