#include<bits/stdc++.h>
#define mp(a,b) make_pair(a,b)
#define pb(a) push_back(a)
#define scan(a) scanf("%d",&a)
#define scanl(a) scanf("%lld",&a)
#define print(a) printf("%d",a)
#define printl(a) printf("%lld",a)
#define MAX 10
using namespace std;

typedef pair<int,int>pii;
typedef long long LL;
int main()
{
	int t;
	scan(t);
	
	
	while(t--)
	{
		string s;
		cin>>s;
		int l=s.length();
		LL ans=s[l-1]-48;
		if(l==1)
		{
			cout<<s[0]<<endl;
			continue;
		}
		
		for(int i=l-2;i>=0;i--)
		{
			if(s[i]=='*')
			{
				ans*=(s[i-1]-48);
				i--;
			}
			else if(s[i]=='-')
			{
				ans-=(s[i-1]-48);
				i--;
			}
			else if(s[i]=='+')
			{
				ans+=(s[i-1]-48);
				i--;
			}
		}
		cout<<ans<<endl;
	}
}
