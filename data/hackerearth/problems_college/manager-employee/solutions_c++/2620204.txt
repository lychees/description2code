int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int i,j,k,n;
		cin>>n;
		int arr[n];
		int mini=INT_MAX,maxi=INT_MIN;
		for(i=0;i<n;i++)
			cin>>arr[i];
		sort(arr,arr+n);
		for(i=0,j=n-1;i<j;i++,j--)
		{
			mini=min(mini,arr[j]+arr[i]);
			maxi=max(maxi,arr[j]+arr[i]);
		}
		cout<<maxi-mini<<"\n";
	}
	return 0;
}
