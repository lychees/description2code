#include<bits/stdc++.h>
using namespace std;


int main()
{
	int t;
	cin>>t;
	
	while(t--)
	{
		int n;
		cin>>n;
		int i,id;
		float d[10005],f[10005];
		for(i=0;i<n;i++)
		{
			cin>>d[i];
			
		}
		
		for(i=0;i<n;i++)
		  cin>>f[i];
		 double mx=-99999; 
		 double sum=0;
		 for(i=0;i<n;i++)
		 {
		 	sum+=f[i]*d[i];
		 }
		 
		 
		 
		 for(i=0;i<n;i++)
		 {
		 	if(f[i]*d[i]>mx)
		 	 {
		 	 	mx=f[i]*d[i];
		 	 	id=i;
		 	 }
		 	
		 }
		 
		 printf("%d %.3f\n",id,mx/sum);
		
		
		
		
		
	}
	
	
	
	
	
	
}
