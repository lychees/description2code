int b[502][502];
void floydwarshal()
{
    int i, j, k;
    for (k = 0; k <= 501; k++)
    {
        for (i = 0; i <=501; i++)
        {
            for (j = 0; j <=501; j++)
            {
                if ((b[i][k] * b[k][j] != 0) && (i != j))
                {
                    if ((b[i][k] + b[k][j] < b[i][j]) || (b[i][j] == 0))
                    {
                        b[i][j] = b[i][k] + b[k][j];
                    }
                }
            }
        }
    }

}
