#include<bits/stdc++.h>
#define inf -99999999
using namespace std;
int main()
{
    int t,n;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        int a[n];
        for(int i=0;i<n;i++)
            scanf("%d",&a[i]);
        sort(a,a+n);
        int ans=0;
        while(1)
        {
            for(int i=0;i<n;i++)
            {
                if(a[i]!=inf)
                    goto con;
            }
            break;
            con:;
            if(a[0]!=inf && a[0]!=a[1]-1)
            {
                a[0]=inf;
                ans++;
                continue;
            }
            int f=0;
            for(int i=1;i<n-1;i++)
            {
                if(a[i]!=inf && a[i]!=a[i-1]+1 && a[i]!=a[i+1]-1)
                {
                    a[i]=inf;
                    f=1;
                    ans++;
                    break;
                }
            }
            if(f==1)
                continue;
            if(a[n-1]!=inf && a[n-1]!=a[n-2]+1)
            {
                a[n-1]=inf;
                ans++;
                continue;
            }
            if(a[0]!=inf)
            {
                a[0]=inf;
                a[1]=inf;
                ans++;
                continue;
            }
            f=0;
            for(int i=1;i<n-1;i++)
            {
                if(a[i]!=inf)
                {
                    if(a[i]!=a[i-1]+1)
                    {
                        a[i]=inf;
                        a[i+1]=inf;
                        ans++;
                        f=1;
                        break;
                    }
                    if(a[i]!=a[i-1]+1)
                    {
                        a[i]=inf;
                        a[i-1]=inf;
                        ans++;
                        f=1;
                        break;
                    }
                }
            }
            if(f==1)
                continue;
            if(a[n-1]!=inf)
            {
                a[n-1]=inf;
                a[n-2]=inf;
                ans++;
                continue;
            }
            f=0;
            for(int i=1;i<n;i++)
            {
                if(a[i]!=inf)
                {
                    a[i-1]=inf;
                    a[i]=inf;
                    a[i+1]=inf;
                    ans++;
                    f=1;
                    break;
                }
            }
        }
        printf("%d\n",ans);
    }
}
