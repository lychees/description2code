#include <iostream>
#include <cmath>
using namespace std;

 int cipher(char c)
    {
    	return c-'a';
    }
int main()
{
	int t;
	cin>>t;
	string str;
	while(t--)
	{
		cin>>str;
		int pos=0,diff,distance,n=str.length();
		for(int i=0; i<n; i++)
		{
			diff= cipher(str[i])-pos;
			if(abs(diff)<13)
			{
				distance=diff;
			}
			else
			{
				distance=26-abs(diff);
				if(diff>0 && diff!=13)
				{
					distance=distance*(-1);
				}
			}
			pos=cipher(str[i]);
			cout<<distance<<" ";
		}
		cout<<endl;
	}
   
    
    return 0;
}
