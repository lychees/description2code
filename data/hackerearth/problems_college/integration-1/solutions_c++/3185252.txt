#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,j;
	cin>>t;
	for(j=1;j<=t;j++)
	{
		int n ,p,u,i,d;
		cin>>n>>p>>u>>i>>d;
		while(n--)
		{
			if(p+i<=u)
			{
				p = p+i;
			}
			else
			if(p-d<=u && p-d>=0)
			{
				p = p-d;
			}
		
		}
		cout<<"Case #"<<j<<": "<<p<<"\n";
	}
}
