#include <iostream>
#include<stdio.h>
using namespace std;
long long int a[2005];
int main()
{
    long long int n,i,d,ans=0,inc,cur,tp;
    scanf("%lld",&n);
    scanf("%lld",&d);
    for(i=0;i<n;i++)
    	scanf("%lld",&a[i]);
    cur=a[0];
    for(i=1;i<n;i++)
    {
    	if(a[i]>cur)
    	{
    		cur=a[i];
    		continue;
    	}
    	else
    	{
    		inc=cur+1-a[i];
	    	if(inc%d==0)
	    	{
	    		tp=inc/d;
	    		ans+=tp;
	    		cur=a[i]+(d*tp);
	    	}
	    	else
	    	{
	    		tp=inc/d;
	    		tp++;
	    		ans+=tp;
	    		cur=a[i]+(d*tp);
	    	}
    	}
    }
    cout<<ans<<endl;
    return 0;
}
