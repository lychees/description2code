LL gcd(LL m, LL n)
{
	LL r;
	while (n != 0)
	{
		r=m%n;
		m=n;
		n=r;
	}
	return m;
}

LL expmod(LL a,LL b)
{
	LL x=1,y=a;
	while(b>0)
	{
		if(b&1) x=(x*y)%mod;
		y=(y*y)%mod;
		b>>=1;
	}
	return (x)%mod;
}
LL inv(LL n)
{
	return expmod(n,mod-2);
}
