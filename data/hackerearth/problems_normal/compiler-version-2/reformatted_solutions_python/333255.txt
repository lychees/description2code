while True:
    try:
        s = raw_input()
    except BaseException:
        break
    a = "."
    b = "->"

    temp = s.find("//")
    if (temp == -1):
        print s.replace(b, a)
    else:
        print s[:temp].replace(b, a) + s[temp:]
