
#include<iostream>
#define N 1000000
using namespace std;
int idx[N];
int a[N];
int main()
{
    int n, k;

    cin >> n >> k;

    for(int i = 1; i <= n; i ++)
    {
        cin >> a[i];
        idx[ a[i] ] = i;
    }

    int i = 1;
    while(i < n && k)
    {
        if(a[i] == n-i+1)
        {
            i ++;
            continue;
        }

        int tmp = idx[n-i+1];
        idx[ a[i] ] = idx[ n-i+1 ];
        idx[ n-i+1 ] = i;
        swap( a[ tmp ], a[i] );


        k --;
        i ++;
    }

    for(int j = 1; j <= n; j ++)
        cout << (j==1?"":" ") << a[j];
    cout << endl;

    return 0;
}
