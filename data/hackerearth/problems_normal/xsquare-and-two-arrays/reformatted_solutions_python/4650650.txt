n, q = [int(s) for s in raw_input().split()]
a = [int(s) for s in raw_input().split()]
b = [int(s) for s in raw_input().split()]
ab = [0]
ba = [0]
temp = [a, b]
for i in xrange(n):
    ab.append(ab[-1] + temp[i % 2][i])
    ba.append(ba[-1] + temp[1-i % 2][i])
for _ in xrange(q):
    t, l, r = [int(s) for s in raw_input().split()]
    sq = ab if (t == 1 and l % 2 == 1) or (t == 2 and l % 2 == 0) else ba
    print sq[r] - sq[l-1]
