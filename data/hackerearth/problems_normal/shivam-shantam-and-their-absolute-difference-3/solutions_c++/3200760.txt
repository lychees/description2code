#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007

long long int dp[400005],dp1[400005],sum1[400005],sum2[400005],ans;

long long int value(int x,int y)
{
    if(y>400000)
        y=400000;
	return (sum2[y]-((x>=0)?sum2[x]:0)+mod)%mod;
}

int suma[205],sumb[205],a[205],b[205],q;

int main()
{
	int i,n,j;
    scanf("%d",&n);

    for(i=0;i<n;i++)
    {
    scanf("%d",&a[i]);
    }

    for(i=0;i<n;i++)
    {
    scanf("%d",&b[i]);
    }

    scanf("%d",&q);

    sort(a,a+n);
    sort(b,b+n);

    for(i=0;i<n;i++)
    {
        suma[i]=a[i]+ ((i-1>=0)?suma[i-1]:0);
        sumb[i]=b[i]+ ((i-1>=0)?sumb[i-1]:0);
    }

    dp[0]=1;
    dp[a[0]]+=1;

    for(i=1;i<n;i++)
    {
    	for(j=suma[i];j>=0;j--)
            dp[j]=(dp[j]+((j-a[i]>=0)?dp[j-a[i]]:0))%mod;
    }

    dp1[0]=1;
    dp1[b[0]]+=1;

    for(i=1;i<n;i++)
    {
    	for(j=sumb[i];j>=0;j--)
    		dp1[j]=(dp1[j]+((j-b[i]>=0)?dp1[j-b[i]]:0))%mod;
    }


    for(i=0;i<=400002;i++)
    	sum1[i]=(dp[i]+((i-1>=0)?sum1[i-1]:0))%mod;

    for(i=0;i<=400002;i++)
    	sum2[i]=(dp1[i]+((i-1>=0)?sum2[i-1]:0))%mod;

    ans=0;

    for(i=0;i<=suma[n-1];i++)
    	ans=(ans+ (dp[i]*(value(i-q-1,i+q))%mod))%mod;

   	cout<<ans;

    return 0;
}
