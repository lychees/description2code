#include <iostream>
#include <cstdio>
#define ll long long
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	ll p,m;
    	scanf("%lld%lld",&p,&m);
    	ll k = p^m;
    	int c=0;
    	while(k){
    		k = k & (k-1);
    		c++;
    	}
    	printf("%d\n",c);
    }
    return 0;
}
