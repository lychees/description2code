#include<cstdio>
#include<iostream>
#include<vector>
using namespace std;
int func(long int n)
{
	int count=0;
	while(n>0)
	{
		if(n%2==1)
		count++;
		n/=2;
	}
	return count;
	
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,k;
		cin>>n>>k;
		vector<long int >arr(n);
		vector<int >bits(n);
		for(int i=0;i<n;i++)
		{
		cin>>arr[i];
		bits[i]=func(arr[i]);
		}
		long int temp,i,j;
		for(i=0;i<k;i++)
		for(j=i+1;j<n;j++)
		{
			if(bits[j]>bits[i])
			{
				temp=bits[j];
				bits[j]=bits[i];
				bits[i]=temp;
			}
			
			
		}
		long int ans=0;
		for(i=0;i<k;i++)
		ans+=bits[i];
		cout<<ans<<endl;
		
		
		
	}
	return 0;
	
	
}
