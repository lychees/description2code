#include<stdio.h>
#include<math.h>
int pat[1100][1100];
int val=1;
void printing(int i,int j,int m);
int main()
{
    int t,i,j,k,l,m,n;
    scanf("%d",&n);
    m=pow(2,n);
    printing(0,0,m);
    for(i=0;i<m;i++)
    {
     for(j=0;j<m;j++)
        printf("%d ",pat[i][j]);
     printf("\n");
    }
    return 0;
}
void printing(int i,int j,int m)
 {
    if(m==1)
        pat[i][j]=val++;
    else
    {

        printing(i,j,m/2);
        printing(i,j+m/2,m/2);
        printing(i+m/2,j,m/2);
        printing(i+m/2,j+m/2,m/2);
    }

 }
