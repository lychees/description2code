#include<cstdio>
using namespace std;
int main()
{
	int i;
	scanf("%d",&i);
	while(i!=1 && i!=2 && i!=3 && i!=4 && i!=5)
	{
		if(i%2==0)
			i=(i-4)/2;
		else
			i=(i-5)/2;
	}
	if(i==1)
		printf("Sheldon");
	else if(i==2)
		printf("Leonard");
	else if(i==3)
		printf("Penny");
	else if(i==4)
		printf("Rajesh");
	else
		printf("Howard");
	return 0;
}
