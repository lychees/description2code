#include<bits/stdc++.h>
using namespace std;
int main()
{
	int test;
	cin>>test;
 
	for(int i=0;i<test;i++)
	{
		int n,m;
 
		cin>>n>>m;
 
		char a[100][100]={};
 
		for(int j=0;j<n;j++)
		{
			for(int k=0;k<m;k++)
				cin>>a[j][k];
		}
 
		int count=0;
 
		for(int j=n-1;j>=0;j--)
		{
			for(int k=m-1;k>=0;k--)
				if(a[j][k]=='0')
				{
					count++;
 
 
					for(int v=0;v<=j;v++)
					{
						for(int b=0;b<=k;b++)
						if(a[v][b]=='0')
							a[v][b]='1';
						else
							a[v][b]='0';
					}
				}
		}
 
	cout<<count<<endl;
	
	
	}
 
	return 0;
}
