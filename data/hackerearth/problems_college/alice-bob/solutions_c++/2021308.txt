#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,i,j,count=0;
    scanf("%d",&n);
    set<int> s;
    vector<int> v;
    for(i=0;i<n;i++)
    {
    	int temp;
    	scanf("%d",&temp);
    	v.push_back(temp);
    	s.insert(temp);
    }
    for(i=0;i<n;i++)
    {
    	for(j=i+1;j<n;j++)
    	{
    		if(s.find(abs(v[i]-v[j]))==s.end())
    		{
    			s.insert(abs(v[i]-v[j]));
    			count++;
    		}
    	}
    }
    if(count%2==0)
    {
    	printf("Bob");
    }
    else
    {
    	printf("Alice");
    }
    return 0;
}
