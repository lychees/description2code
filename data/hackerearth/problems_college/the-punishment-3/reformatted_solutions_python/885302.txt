def power(n, exp):
    prod = 1
    while (exp > 0):
        if ((exp & 1) != 0):
            prod *= n
        n *= n
        exp >>= 1

    return prod


t = input()
while t > 0:
    t -= 1
    s = raw_input()
    n, k = [long(i) for i in s.split(' ')]
    print(power(10, n-k)-1)*(power(10, k)-1)
