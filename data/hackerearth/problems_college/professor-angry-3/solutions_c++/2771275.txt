#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
     {
     	
     	int n,k;
     	cin>>n>>k;
     	int g=0;
     	for(int i=0;i<n;i++)
     	 {
     	 	int test;
     	 	cin>>test;
     	   if(test<1){g++;}
     	 }
     	 if(g>=k){cout<<"NO"<<endl;}
     	 else{cout<<"YES"<<endl;}
     }
    return 0;
}
