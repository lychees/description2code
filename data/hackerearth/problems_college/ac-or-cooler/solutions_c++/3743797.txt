#include <bits/stdc++.h>
using namespace std;
typedef long long lld;
int main()
{
    lld m,n,x,count,sum,i;
    vector<lld> ar;
    cin>>n>>m;
    n=n*2;
    for(i=0;i<n;i++)
    {
    	cin>>x;
    	ar.push_back(x);
    }
    std::sort(ar.begin(),ar.end());
    count=0;
    sum=0;
    for(i=0;i<n;i++)
    {
    	sum+=ar[i];
    	if(sum>=m)
    		break;
    	count++;
    }
    cout<<count;
    return 0;
}
