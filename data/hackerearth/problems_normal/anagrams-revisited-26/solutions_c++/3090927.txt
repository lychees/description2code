using namespace std;

int main()
{
    int t ;
    cin >> t;
    string s;
    map<string,int> zone;
    map<string,int> :: iterator it;
    while(t--)
    {
    	cin >> s;
    	sort(s.begin(),s.end());
    	zone[s]++;
    }
    int maxi = 0;
    for(it = zone.begin() ;it != zone.end() ; it++)
    {
    	if(maxi < it->second)
    	{
    		maxi = it->second;
    	}
    }
    cout << maxi << endl;
    return 0;
}
