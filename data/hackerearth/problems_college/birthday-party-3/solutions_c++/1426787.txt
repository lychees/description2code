#include <bits/stdc++.h>
using namespace std;
#define MOD 179424673
int main()
{
    //cout << "Hello World!" << endl;
    int t,P,Q,X,Y;
    cin>>t;
    int DP[1001][1001][2]={0};
    while(t--)
    {
    	memset(DP,0,sizeof(DP));    	
    	DP[0][0][0]=DP[0][0][1]=1;    	
    	scanf("%d %d %d %d",&P,&Q,&X,&Y);
    	for(int i=0;i<=P;i+=1)
    	{
    		for(int j=0;j<=Q;j+=1)
    		{
    			for(int k=1;k<=X&&k<=i;k+=1)DP[i][j][0]=(DP[i][j][0]+DP[i-k][j][1])%MOD;
    			for(int k=1;k<=Y&&k<=j;k+=1)DP[i][j][1]=(DP[i][j][1]+DP[i][j-k][0])%MOD;
    		}
    	}
    	printf("%d\n",(DP[P][Q][0]+DP[P][Q][1])%MOD);
    }
    
    return 0;
}
