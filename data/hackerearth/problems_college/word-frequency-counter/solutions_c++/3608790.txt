#include <bits/stdc++.h>
using namespace std;
int main()
{
	map<int,pair<string,int> > m;
	map<int,pair<string,int> > ::iterator it;
	string s;
	int i,j=1;
	while(cin>>s)
	{
		int n=s.length();
		for(i=0;i<n;i++)
		s[i]=toupper(s[i]);
		int flag=1;
		for(it=m.begin();it!=m.end();it++)
		{
			if(it->second.first==s)
			{
				it->second.second++;
				flag=0;
			}
		}
		if(flag)
		{
			m.insert(pair<int,pair<string,int> >(j,pair<string,int>(s,1)));
			j++;
		}
	}
	for(it=m.begin();it!=m.end();it++)
	{
		cout<<it->second.first<<" "<<it->second.second<<"\n";
	}
    return 0;
}
