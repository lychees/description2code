#include<bits/stdc++.h>
using namespace std;
int main()
{
	ios::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		int n,c;
		c=0;
		cin>>n;
		long long a[n+1];
		for(int i=0;i<n;i++)
		{
			cin>>a[i];
		}
	     for(int i=0;i<n-1;i++)
	     {
	     	if(a[i]>=a[i+1])
	     	 c++;
	     }
	     cout<<c<<"\n";

}
return 0;
}
