#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string n;
    	cin>>n;
    	int i,f=0;
    	int nn;
    	nn = stoi(n);
    	for(i=0;i<(n.size()-1);i++)
    	{
    		if((n[i]=='2' && n[i+1]=='1') || (nn%21==0) )
    		{
    			cout<<"The streak is broken!"<<endl;
				f++;
    			break;
    		}
    	}
    	if(f==0)
    	cout<<"The streak lives still in our heart!"<<endl;
    	
    }
    return 0;
}
