#include <bits/stdc++.h>
using namespace std;


bool islucky(int n)
{
    int count =0;
    for(int i =2; i <=n; i++)
    {
		if( n % i ==0)
		{
		    count++;
		    while(n%i ==0 )
		    {
				n/= i;
		    }
		}
    }
    return count>2?true:false;
	
}
int main()
{
    int t,n;
    cin>>t;
    long int array[1001];
    int prime = 1;
    for(int i =30; i<10000000;i++)
    {
		if(prime > 1000)
		break;
		if(islucky(i))
		{
	    	array[prime] = i;
	    	prime++;
		}
    }
    while(t--)
    {
    	cin>>n;
    	cout<<array[n]<<endl;
    	
    }
    
}
