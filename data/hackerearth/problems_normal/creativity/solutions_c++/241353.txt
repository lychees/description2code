#include <iostream>
using namespace std;
#include<string.h>
int main()
{
 int t,i,j,f,la,lb,k,x=1;
 char A[1001],B[1001],X[1001];
 cin>>t;
 for(k=0;k<t;k++)
 {
 	if(x)
 	{
 	 cin.getline(X,1001);
 	 x=0;
 	}
 	cin.getline(A,1001);
 	cin.getline(B,1001);
 	
 	la=strlen(A);
 	lb=strlen(B);
 	if(la>lb)
 	 cout<<"not possible\n";
 	else
 	{
 		j=0;
 		f=0;
 		while(j<lb&&!f)
 		{
 			i=0;
 			if(A[i]==B[j])
 			{
 				f=1;
 				while(i<la&&f)
 				{
 				 if(A[i]==B[j])
 				 {
 				 	i++;
 				 	j++;
 				 	f=1;
 				 }
 				 else
 				  f=0;
 				}
 			}
 			j++;
 		}
 		if(f) 
 		 cout<<"possible\n";
 		else 
 		 cout<<"not possible\n";
 	}
 }
 return 0;
}
