#include <iostream>
#include <math.h>
#include <string>
#include <stdio.h>
using namespace std;

int main()
{
    int n;
    scanf("%d\n",&n);

    string s;
    while(n--)
    {
        getline(cin,s);
        
       int l=s.length(),r=sqrt(l);
        int z;
        for(int i=0;i<r;i++)
        {
            z=i;
            for(int j=0;j<r;j++)
            {
                cout<<s[z];
                z=(z+r)%l;
            }
        }
         cout<<endl;
    }
  return 0;
}
