import math

t = input()
for xx in range(0,t):
    n = input()
    s = math.sqrt(n)
    s += 1
    ans = 1
    s = int(s)
    for i in range(2,s):
        ct = 0
        while( (n%i) == 0 ):
            n /= i;
            ct += 1
        ans *= (ct+1)
    if(n != 1):
        ans *= 2
    print ans-1
