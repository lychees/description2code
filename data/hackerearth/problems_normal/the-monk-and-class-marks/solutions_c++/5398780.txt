#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    string s;
    int l,i;
  
    vector<pair<int,string>> v;
      v.clear();
    for(i=0;i<n;i++)
    {
    	cin>>s>>l;
    	v.push_back(make_pair(-l,s));
    }
    sort(v.begin(),v.end());
    for(i=0;i<v.size();i++)
    {
    	cout<<v[i].second<<" "<<-1*v[i].first<<endl;
    }
    	
    
    return 0;
}
