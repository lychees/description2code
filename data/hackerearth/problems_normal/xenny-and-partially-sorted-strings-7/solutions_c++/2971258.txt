#include<bits/stdc++.h>
using namespace std;

string s[1005];
string xyz;
int n,k,m;

bool mycomp(string x, string y)
{
    return x.substr(0,m) < y.substr(0,m);
}

int main()
{
	    ios_base::sync_with_stdio(false);

    int t, i;
	cin >> t;

    while (t--)
    {
		cin >> n >> k >> m;
		
		for ( i = 0 ; i <= n - 1 ; i++ )
		{
			cin >> xyz;
			s[i] = xyz;
		}

		sort(s,s+n,mycomp);

        cout << s[k-1] << "\n";
    }
    
    return 0;
}
