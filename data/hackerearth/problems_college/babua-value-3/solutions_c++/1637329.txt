ll gcd(ll a, ll b) {if (a == 0 || b == 0) return max(a,b); if (b % a == 0) return a; return gcd(b%a, a);}
ll modpow(ll a,ll b) {ll res=1;a%=MOD;for(;b;b>>=1){if(b&1)res=res*a%MOD;a=a*a%MOD;}return res;}
int64_t a[100005];

int main()
{
	int n;
	cin >> n;
	int64_t sum = 0;
	for (int i = 0; i < n; i++)
	{
		cin >> a[i];
		sum += a[i];
	}
	sort(a, a + n);
	int right = n - 1;
	sum -= (n * a[0]);
	int64_t ans = a[0] + 2 * sum;
	for (int i = 1; i < n; i++)
	{
		sum -= right * (a[i] - a[i - 1]);
		right--;
		ans += a[i] + 2 * sum;
	}
	int64_t g = gcd(ans, n);
	int64_t p = ans / g;
	int64_t q = n / g;
	cout << p << "/" << q << endl;
}
