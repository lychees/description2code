def check(a, i, j):
    while i <= j:
        if (a[i] != a[j]):
            return False
        i += 1
        j -= 1
    return True


t = input()
while t:
    t -= 1
    s = raw_input()
    s = s.strip()
    ans = 0
    for i in range(len(s)):
        for j in range(len(s)):
            if (check(s, i, j)):
                ans = max(ans, j-i+1)
    print ans
