#include <iostream>
#include<algorithm>
using namespace std;

struct order{
	int fan;
	long int t;
	long int d;
};

struct receiver{
	int no;
	long int tt;
};

bool compstart(const order &a,const order &b){
	return (a.t < b.t);
}

bool algo(const receiver &a,const receiver &b){
	return (a.tt <b.tt);
}

int main()
{
    int n;
    cin>>n;
    struct order o[n];
    
    for(int i=0;i<n;i++){
    	o[i].fan=i+1;
    	cin>>o[i].t>>o[i].d;
    }
    
    //sort(o,o+n,compstart);
    struct receiver res[n];
    for(int i=0;i<n;i++){
    	res[i].no=o[i].fan;
    	res[i].tt=o[i].t+o[i].d;
    }
    
    sort(res,res+n,algo);
    
    for(int i=0;i<n;i++){
    	cout<<res[i].no<<" ";
    }
    return 0;
}
