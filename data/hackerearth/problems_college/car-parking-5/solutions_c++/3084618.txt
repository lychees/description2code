#include <iostream>
using namespace std;
 
int main()
{
	int n;
    string a;
    cin>>n;
    while(n--)
    {
    cin>>a;
    int count=0;
    for(int i=0;i<a.length();i++)
    {
    	if(a[i]=='B' || a[i]=='D' || a[i]=='S' || a[i+1]=='B' || a[i+2]=='B'|| a[i+1]=='S' || a[i-1]=='S')
    	{}
    	else
    	{
    	count++;
    	}
     }
    cout<<count<<endl;
    }
}
