#include<iostream>
using namespace std;
#include<queue>

int main() {
	long t, i, val, first, second, third;
	std::priority_queue<long> pq;
	
	cin>>t;
	for(i=0;i<t;i++) {
		cin>>val;
		pq.push(val);
		
		if(pq.size() == 1 || pq.size() == 2)
			cout<<"-1";
		else {
			first = pq.top();
			pq.pop();
			second = pq.top();
			pq.pop();
			third = pq.top();
			pq.push(first);
			pq.push(second);
			cout<<first*second*third;
		}
		cout<<endl;
	}
	
	return 0;
}
