using namespace std;
typedef long long int llt;

vector<llt> v;

bool isItWorthIt(llt sum,int m)
{
	int count =1;
	llt s = 0;
	for(int i = 0 ; i < v.size() ; i++)
	{
		if(v[i] > sum)
			return false;
		if(v[i] + s > sum)
		{
			count++;
			s = v[i];
		}
		else
			s+=v[i];
	}
	if(count > m)
		return false;
	else
		return true;

}
