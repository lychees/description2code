#include<bits/stdc++.h>
using namespace std;

typedef long long int ll;

const int N = 5e6 + 5;
const ll mod = 1e9+7;

int tt;
int n,a,b,c,k;
//int arr[N];


  void solve(int a,int b,int c){
      string ret = "";
      float res = (float)1.0*((float)a/(float)b);
      cout << setprecision(c) << fixed << res << endl;
  }

bool composite[N];
ll primes[N];
ll dp[N];
int y;
 void sieve()
 {
     y = 0;
     composite[0] = composite[1] = true;
     for(int i = 2; i<N; i++){
        if(!composite[i]){
                primes[y++] = i;
            for(int j = i*2;j<N;j+=i)
                composite[j] = true;
        }
     }
 }


 void pre()
 {
     dp[0] = 0;
     dp[1] = 2;
     for(int i = 2; i<N; i++){
        dp[i] = (dp[i-1]%mod*primes[i-1]%mod)%mod;
     }
 }






int main(){

     sieve();
     pre();
     cin >> tt;
     while(tt--){
        cin >> n;
        cout << dp[n] << endl;
     }
     return 0;
}
