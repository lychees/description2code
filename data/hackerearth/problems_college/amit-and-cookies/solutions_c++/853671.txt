#include <iostream>
#include <algorithm>
using namespace std;

int a[100001];

int main()
{
    int n;
    cin >> n;
    long long k;
    cin >> k;
    
    int i;
    for(i=0;i<n;i++)
    {
    	cin >> a[i];
    }
    //cout << endl;
    sort(a,a+n);
//    for(i=0; i<n;i++)
//     cout << a[i] << " " ;
    i = 0 ;
    int count = 0;
    long long sum = 0; 
    while( i < n && sum < k)
     {
     	sum += a[i];
     	count++;
     	i++;
     }
     if(sum > k)
      count--;
    cout << count << endl;
    return 0;
}
