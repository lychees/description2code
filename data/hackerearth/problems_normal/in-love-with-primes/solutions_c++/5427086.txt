#include <iostream>
#include<math.h>
using namespace std;
int p(int no)
{
	
	for(int i=2;i<=sqrt(no);i++)
	if(no%i==0)
	return -1;
	
	return 0;
}
int main()
{
    
	int tc,no,i,c;
	cin>>tc;
	while(tc--)
	{
		cin>>no;
		c=0;
		for(i=2;i<=no/2;i++)
		if(p(i)==0&&p(no-i)==0)
		c++;
		if(c==0)
		cout<<"Arjit"<<endl;
		else
		cout<<"Deepa"<<endl;
	}

	
}
