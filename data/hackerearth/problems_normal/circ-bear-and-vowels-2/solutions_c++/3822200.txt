#include<bits/stdc++.h>
using namespace std;
main()
{
	int t,i,j,n;
	char arr[60],*p;
	cin>>t;
	while(t--)
	{
		int flag=0,c=0,v=0,flag2=0;
		cin>>arr;
		n=strlen(arr);
		p=arr;
		while(*p!='\0')
		{
			switch(*p)
			{
				case'a':
					{
						flag=0;
						v++;
						p++;
						break;
					}
				case'e':
					{
						flag=0;
						v++;p++;
						break;
					}
				case'i':
					{
						flag=0;
						v++;p++;
						break;
					}
				case'o':
					{
						flag=0;
						v++;p++;
						break;
					}
				case'u':
					{
						flag=0;
						v++;p++;
						break;
					}
				case'y':
					{
						flag=0;
						v++;p++;
						break;
					}
				default:{
					flag++;
					c++;p++;
					break;
				}
			}
			///cout<<c<<" "<<v<<" ";
			if(flag==3)
			{
				flag2=1;
				break;
			}
		}
		//cout<<c<<" "<<v<<" ";
		if(c>v)
			flag2=1;
		if(flag2==1)
			cout<<"hard\n";
		else
			cout<<"easy\n";
	}
}
