#include <bits/stdc++.h>
using namespace std;
int arr[1000];
void solve()
{
	for(int i=2;i<=1000;i++)
	{
		if(arr[i]==0)
		for(int j=i+i;j<=1000;j=j+i)
		{
			arr[j]=1;
		}
	}
}

int main()
{
 //   cout << "Hello World!" << endl;
 solve();
 int t;
 cin>>t;
 while(t--)
 {
 	int sum=0;
 	long long int n;
 	cin>>n;
 	while(n)
 	{
 		sum=sum+n%10;
 		n=n/10;
 	}
 	if(sum>=2)
 	{
 	if(arr[sum]==0)
 	cout<<"YES"<<endl;
 	else
 	cout<<"NO"<<endl;
   }
   else
   {
   	cout<<"NO"<<endl;
   }
 }
    return 0;
}
