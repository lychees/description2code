#include <iostream>
using namespace std;

//beingClashed()
//version 1
bool beingClashed(int* startTime, int* endTime, int n){

	for (int first = n-1; first > 0; first--)
		for (int second = 0; second < first; second++)
			if (startTime[second] > startTime[second+1]){
				swap(startTime[first], startTime[first+1]);
				swap(endTime[second], endTime[second+1]);
			}

	for (int i = 1; i < n; i++)
		if (startTime[i] < endTime[i-1])
			return true;

	return false;
}
//beingClashed2()
//version 2
bool beingClashed2(int* startTime, int* endTime, int n){

	for (int first = 0; first < n-1; first++)
		for (int second = first+1; second < n; second++)
			if (endTime[first] > startTime[second]
			    && endTime[second] > startTime[first])
					return true;
	return false;
}
//! main()
int main(int argc, char* argv[]){

	int n; //number of contests
	cin >> n;
	int* startTime = new int [n];
	int* endTime = new int [n];

	for (int i = 0; i < n; i++){
		int startHour, startMinute, endHour, endMinute;
		char chr;
		cin >> startHour >> chr >> startMinute; //input start time
		cin >> chr; //for "-"
		cin >> endHour >> chr >> endMinute; //input end time
		startTime[i] = startHour*60 + startMinute;
		endTime[i] = endHour*60 + endMinute;
	}

	if (beingClashed2(startTime, endTime, n))
		cout << "Will need a moderator!" << endl;
	else
		cout << "Who needs a moderator?" << endl;

	//system("pause");	// for VC++
	return 0;
}
