#include<iostream>
#include<stdio.h>
#include<math.h>
using namespace std;

int arr[4000005] = {0};

void prime(int b)
{
	arr[0] = arr[1] = 1;
	int i, j, index, flag;
	for(i=2;i <= sqrt(b);i++)
	{
		for(j=flag=2;j <= sqrt(i);j++)
			if(i % j == 0)
			{
				flag = 0;
				break;
			}
		if(flag != 0)
		{
			index = 2;
			while(index*i <= b)
				arr[((index++)*i)] = 1;
		}
	}
}

int main()
{
	int test, n, temp, i;
	long long int sum;
	scanf("%d", &n);
	prime(n);
	for(i=0, sum=0;i < n;i++)
		if(arr[i] == 0)
			sum = sum + i;
	printf("%lld\n", sum);
}
