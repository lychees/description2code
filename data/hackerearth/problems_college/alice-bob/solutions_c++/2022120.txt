#include<stdio.h>
#include<algorithm>
#include<vector>
#include<algorithm>
#define lli long long
using namespace std;
vector<lli> v;
lli hcf(lli a,lli b)
{
    if(b==0)
     return a;
    return hcf(b,a%b);
}
int main()
{
  lli n;
  lli max1=-1;
  v.clear();
  scanf("%lld",&n);
  for(lli i=0;i<n;i++)
  {
     lli temp;
     scanf("%lld",&temp);
     v.push_back(temp);
     if(max1<temp)
      max1=temp;
  }
  lli gcd=v[0];
  for(lli i=1;i<n;i++)
    gcd=hcf(gcd,v[i]);
  lli x=max1/gcd;
  v.clear();
  if((x-n)%2!=0)
    printf("Alice\n");
  else
    printf("Bob\n");
  return 0;
}
