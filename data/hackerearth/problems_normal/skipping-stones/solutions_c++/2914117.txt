#define MOD 1000000007
int d[1111];
double p[1111];
double re[1111];
int main(void){
	std::ios::sync_with_stdio(false);cin.tie(0);
	int i,j,n,m,L;
	cin>>n>>m>>L;
	d[0]=1;
	p[0]=1;
	re[0]=1;
	for(i=1;i<=n;i++)
		cin>>p[i];
	for(i=1;i<=n;i++)
		cin>>d[i];
	n++;
	d[n]=L;
	p[n]=1;
	for(i=1;i<=n;i++)
	{
		re[i]=0;
		for(j=i-1;j>=0&&d[i]-d[j]<=m;j--)
		{
			re[i]=max(re[i],p[i]*re[j]);
		}
	}
	printf("%.6lf\n",re[n]);
//	system("pause");
	return 0;
}
