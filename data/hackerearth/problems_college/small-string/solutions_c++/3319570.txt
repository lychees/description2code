int scan()
{
    int n = 0, ch = get();
    while (ch < '0' || ch > '9')
        ch = get();
    while (ch >= '0' && ch <= '9') {
        n = (n << 3) + (n << 1) + ch - '0';
        ch = get();
    }
    return n;
}

int main()
{
    int te, n, l, i, j, k;
    string a, s, t, res;
    cin >> te;
    while (te--) {
        cin >> n;
        cin >> s;
        res = s;
        l = s.length();
        for(i = 0; i < l; i++) {
            a = "";
            for(j = 0; j < l; j++) {
                if(i != j) {
                   a += s[j];
                }
            }

            for(j = 0; j < a.length() + 1; j++) {
                t = "";
                for(k = 0; k < j; k++) {
                    t += a[k];
                }
                t += s[i];
                for(k = j; k < a.length(); k++) {
                    t += a[k];
                }
                res = min(res, t);
            }
        }
        cout << res << endl;
    }

    return 0;
}
