struct ques{

}A[100005];
bool mycomp(ques A,ques B){
	if(A.S==B.S)
	   return A.T<B.T;
	 return A.S>B.S;
}

	LL i;

	for(i=0;i<N;i++)

	for(i=0;i<N;i++)

	 sort(A,A+N,mycomp);
	 SUM[0]=A[0].T;
