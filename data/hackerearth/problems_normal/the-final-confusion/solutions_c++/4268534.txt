typedef long long ll;
ll MOD=1000000007;
ll pw(ll a,ll b){
  ll r=1;
  while(b>0){
     if(b&1){
       r=(r*a)%MOD;
        }
     b=b>>1;
     a=(a*a)%MOD;
  }
return r;
}
ll v[100][100]={{}};
ll fun(int full,int half){
   if(v[full][half]>0){
      return v[full][half];
   }
   if(!full){
       return v[full][half]=1;
   }
   if(!half){
       return v[full][half]=fun(full-1,half+1);
   }

      return v[full][half]=fun(full-1,half+1)+fun(full,half-1);
}
