#include<stdio.h>
#include<cstring>
#include<vector>
#include<cassert>
#define mx 100009
using namespace std;
vector<bool> sieve(mx,false);
long long nos[mx];
void sv()
{
  int i,j;
  for(i=1;i<=mx;i++)
  nos[i]=i;
  for(i=2;i<mx;i++)
  {
    if(!sieve[i])
    {
      nos[i]--;
     for(j=i+i;j<mx;j+=i)
     {sieve[j]=true;
     nos[j]-=(nos[j]/i);}}}
     }
int main()
{
	
    sv();
    long long pre[mx];
    memset(pre,0,sizeof(pre));
    pre[1]=0;
    for(int i=2;i<mx;i++)
    pre[i]=pre[i-1]+nos[i];
    int t,n;
    scanf("%d",&t);
    while(t--)
    {
      scanf("%d",&n);
      //assert(n>0 && n<=10000);
      printf("%d\n",pre[n]);
      }
      return 0;
      }
