#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int sum=0,a[12];
    for(int i=0;i<n;i++)
    {
    	cin>>a[i];
    	sum+=a[i];
    }
    int dp[13][403];
    sum=sum/3+1;
    int max=0;
    for(int i=0;i<=n;i++)
    dp[i][0]=1;
    for(int i=1;i<=sum;i++)
    dp[0][i]=0;
    for(int i=1;i<=n;i++)
    for(int j=1;j<=sum;j++)
    {
        dp[i][j]=dp[i-1][j];
        if(j-a[i-1]>=0)
        dp[i][j]=dp[i][j]||dp[i-1][j-a[i-1]];
        if(dp[i][j]&&j>max)
        max=j;
    }
    for(int i=0;i<n;i++)
    if(a[i]>max)
    max=a[i];
    cout<<max<<endl;
}
