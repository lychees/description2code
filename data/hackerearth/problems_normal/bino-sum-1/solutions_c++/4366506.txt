#include <iostream>
using namespace std;

typedef long long int ll;

# define M 1000000007

ll mod_exp(ll base ,ll ex){
	ll res=1;
	while(ex>0){
		if(ex%2) res = (res*base)%M;
		base = (base*base)%M;
		ex/=2;
	}
	return res;
}

ll inverse(ll n){
	return mod_exp(n,M-2);
}

ll ncr(ll n,ll r){
    ll k;
	if(r==0) return 1;
	if(r==n) return 1;
	ll i,prod=1;
	for(i=n,k=0;k<r;i--,k++) prod = (prod*i)%M;
	while(r>0){
		prod = (prod*inverse(r))%M;
		r--;
	}
	return prod;
}

int main()
{
    int T;
    cin>>T;
    while(T--){
 		ll n,r,i,sum=1,temp=1;
    	cin>>n>>r;
    	for(i=0;i<r;i++){
    		temp = (((temp*(n-i))%M)*inverse(i+1))%M;
    		sum = (sum + temp)%M;
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
