#include <iostream>
#include <stdio.h>
#include <string.h>
#include <algorithm>
#define mod 1000000007
using namespace std;

long long power(int a,int b,int m)
{
	long long x = a,ans = 1;
	while(b>0)
	{
		if(b%2==1)
			ans = (ans*x)%m;
		b = b/2;
		x = (x*x)%m;
	}
	return ans;
}

long long inverseuler(long long a,int b)
{
	return power(a,b-2,b);
}

long long freq[100001];
void assign()
{
	freq[0] = 1;
	freq[1] = 1;
	for(int i=2;i<=100000;i++)
		freq[i] = (freq[i-1]*i)%mod;
}
long long ncr(int n,int r,int m)
{
	return (freq[n]*((inverseuler(freq[r],m)*inverseuler(freq[n-r],m))%m))%m;
}

int main()
{
	#ifndef ONLINE_JUDGE

	#endif
		assign();
	int t,n,k;
	long long sum;
	scanf("%d",&t);
	while(t--)
	{
		sum = 0;
		scanf("%d %d",&n,&k);
		int arr[n];
		for(int i=0;i<n;i++)
			scanf("%d",&arr[i]);
		sort(arr,arr+n);
		int i=0;
		while((n-1-i)>=(k-1))
		{
			sum+=(ncr(n-1-i,k-1,mod)*arr[i++])%mod;
			sum = sum%mod;
		}
		printf("%lld\n",sum);
	}
	return 0;
}
