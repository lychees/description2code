#include <bits/stdc++.h>

using namespace std;

const int N = 1e5 + 10;

pair <string, int> a[N];
string s[N];
int pos[N];

int main() {
	ios::sync_with_stdio(false);
	int n, q;
	cin >> n;
	for (int i = 0; i < n; i++) {
		string str;
		cin >> str;
		a[i] = make_pair(str, i + 1);
	}
	stable_sort(a, a + n);
	for (int i = 0; i < n; i++) {
		s[i] = a[i].first;
		pos[i] = a[i].second;
	}
	cin >> q;
	for (int qq = 1; qq <= q; qq++) {
		int l, r;
		string x;
		cin >> l >> r >> x;
		int ub = upper_bound(s, s + n, x) - s;
		int lb = lower_bound(s, s + n, x) - s;
		int ub2 = upper_bound(pos + lb, pos + ub, r) - (pos + lb);
		int lb2 = lower_bound(pos + lb, pos + ub, l) - (pos + lb);
		cout << (ub2 - lb2) << '\n';
	}
	return 0;
}
