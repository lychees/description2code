def not_divisible(num, lists):
	for n in lists:
		if num%n == 0:
			return False
	return True

def find_next_num(color, prev_ans, colors):
	ans = max(colors[color-1], prev_ans)
	while ans%colors[color-1] != 0 or not not_divisible(ans, colors[color:]) or ans <= prev_ans:
		ans += 1
	return ans
