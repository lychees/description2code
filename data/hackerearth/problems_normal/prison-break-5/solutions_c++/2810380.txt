#include<iostream>
#include<cstdio>
using namespace std;
int n,count_route;
int cell[20][20], mark[20][20];
void route(int i, int j)
{
    if(i == n-1 && j == n-1)
    {
        count_route++;
        return;
    }
    mark[i][j] = 1;
    if( (j+1) < n && cell[i][j+1] == 0 && mark[i][j+1] == 0)
        route(i,j+1);
    if( (j-1) >=0 && cell[i][j-1] == 0 && mark[i][j-1] == 0)
        route(i, j-1);
    if( (i+1) < n && cell[i+1][j] == 0 && mark[i+1][j] == 0)
        route(i+1,j);
    if( (i-1) >=0 && cell[i-1][j] == 0 && mark[i-1][j] == 0)
        route(i-1,j);
    mark[i][j] = 0;
}
int main()
{
   int t;
   scanf("%d",&t);
   while(t--)
   {
       scanf("%d",&n);
       for(int i=0; i<n; i++)
       {
           for(int j=0; j<n; j++)
           {
               scanf("%d",&cell[i][j]);
               mark[i][j] = 0;
           }
       }
       count_route = 0;
       route(0,0);
       printf("%d\n",count_route);
   }
   return 0;
}
