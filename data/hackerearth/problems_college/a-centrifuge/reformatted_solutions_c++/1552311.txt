#include <iostream>
using namespace std;

int main() {
  int t, n, m, x, y, z;
  cin >> t;
  while (t--) {
    cin >> m >> n;
    if (n == 0) {
      cout << "0\n";
      continue;
    }
    if (n >= m) {
      x = n / m;
      y = n % m;
      if (y > 0) {
        x++;
        n = n - (m - y);
        if (n <= 0)
          cout << 5 * x << endl;
        else
          cout << 5 * (x + (n - 1) / m + 1) << endl;
      } else
        cout << 2 * 5 * x << endl;
    } else
      cout << "10\n";
  }
  return 0;
}
