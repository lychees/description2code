#include<iostream>
#include<stdlib.h>
#include<stdio.h>
using namespace std;

int temp[100][100],d,a[100][100],n,m;

int abs(int x,int y)
{
    if(x>y)
        return x-y;
    return y-x;
}

void dfs(int x,int y,int visited[100][100])
{
    if(visited[x][y])
        return;
    visited[x][y] = 1;
    //cout<<x<<" "<<y<<endl;
    //system("pause");

   temp[x][y] = 1;
   if(x-1>=0 && abs(a[x-1][y],a[x][y]) <= d)
    {
        dfs(x-1,y,visited);
    }
    if(y-1>=0 && abs(a[x][y-1],a[x][y]) <=d)
    {
        dfs(x,y-1,visited);
    }
    if(y+1<m && abs(a[x][y+1],a[x][y]) <=d)
    {
        dfs(x,y+1,visited);
    }
    if(x+1<n && abs(a[x+1][y],a[x][y]) <=d)
    {
        dfs(x+1,y,visited);
    }
}

int func(int x,int y)
{
    int visited[100][100]={0};
    dfs(x-1,y-1,visited);

    int count=0;
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m;j++)
        {
            if(temp[i][j])
                count++;
        }
    }
    return count;
}

int main()
{
    int q;
    cin>>n>>m>>q;
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m;j++)
            {
                cin>>a[i][j];
            }
    }
    int x,y;
    for(int i=0;i<q;i++)
    {
        for(int l=0;l<n;l++)
        {
            for(int k=0;k<m;k++)
                temp[l][k] = 0;
        }
        cin>>x>>y>>d;
        cout<<func(x,y)<<endl;
    }

    //system("pause");
    return 0;
}
