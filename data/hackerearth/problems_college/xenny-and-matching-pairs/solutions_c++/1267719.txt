#include <iostream>
using namespace std;

int main()
{
    int t,n;
    cin>>t;
    while(t--){
    	cin>>n;
    	int k=0,d;
    int *b = new int[n+1];
    for(int i=0;i<n;i++){
        scanf("%d",&d);
        int x = lower_bound(b,b+k,d)-b;
        //cout << x << ".,";
        b[x] = d;
        k = max(k,x+1);
    }
    cout << k << "\n";
    }
    return 0;
}
