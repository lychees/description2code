#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;cin>>t;
    while(t--)
    {
    	int i,n,k;cin>>n>>k;
    	int a[n],b[n];
    	long long sum=0;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    		b[i]=__builtin_popcount(a[i]);
    	}
    	sort(b,b+n);
//    	for(i=0;i<n;i++)
//    		cout<<b[i]<<' ';
//    	cout<<'\n';
    	for(i=n-1;i>n-k-1;i--)
    		sum+=b[i];
    	cout<<sum<<'\n';
    }
    return 0;
}
