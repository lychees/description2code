#include <iostream>
using namespace std;

int main()
{
    
    
    int t;
    cin>>t;
    
    while(t--){
    	
    	int n;
    	cin>>n;
    	
    	char a[n][n];
    	
    	for(int i = 0;i<n;i++){
    		for(int j = 0;j<n;j++){
    			cin>>a[i][j];
    		}
    	}
    	
    	int r[n];
    	r[0] = 0;
    		for(int i = 0;i<n;i++)
    	       r[0]+=a[0][i]=='#'?1:0;
    	for(int i = 1;i<n;i++){
    		
    		r[i]=r[i-1];
    		for(int j = 0;j<n;j++){
    			r[i]+=a[i][j]=='#'?1:0;
    		}
    	}
    	
    	int c[n];
    	c[0]=0;
    	for(int i = 0;i<n;i++)
    	c[0]+=a[i][0]=='#'?1:0;
    	for(int i = 1;i<n;i++){
    		
    		c[i]=c[i-1];
    		for(int j = 0;j<n;j++){
    			c[i]+=a[j][i]=='#'?1:0;
    		}
    	}
    	
    	int flag = 0;
    	for(int i = 0;i<n-1;i++){
    		if(r[i]==r[n-1]-r[i]){
    			flag = 1;
    			cout<<"YES\n";
    			break;
    		}
    	}
    	
    	if(flag==0){
    		for(int i = 0;i<n-1;i++){
    		if(c[i]==c[n-1]-c[i]){
    			flag = 1;
    			cout<<"YES\n";
    			break;
    		}
    	}
    	}
    	
    	if(flag==0)
    	cout<<"NO\n";
    }
    return 0;
}
