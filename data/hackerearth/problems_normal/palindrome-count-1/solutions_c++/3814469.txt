#include <bits/stdc++.h>
using namespace std;
bool isPalindrome(string S){
	int N = S.length();
	for(int i = 0,j = N-1; i < j; i++,j--){
		if(S[i] != S[j]){
			return 0;
		}
	}
	return 1;
}
int main()
{
    string S;
    cin >> S;
    int N = S.length();
    int C = N;
    for(int i = 0; i < N; i++){
    	for(int j = i+1; j < N; j++){
    		if(isPalindrome(S.substr(i,j-i+1))){
    			C++;
    		}
    	}
    }
    printf("%d",C);
    return 0;
}
