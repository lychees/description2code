#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	long long a,x,d,n,m,ans;
	cin >> t;
	while ( t-- ) {
		cin >> n >> m;
		ans = 0;
		a = 1;
		x = min(n,m);
		d = m+1;
		ans += (x*(2*a + (x-1)*d))/2;
		d = n+1;
		ans += (x*(2*a + (x-1)*d))/2;
		cout << ans << endl;
	}
    return 0;
}
