#include <iostream>
#include <vector>
using namespace std;

int max(int a, int b)
{
	return a>b ? a : b;
}

int main()
{
    int n,m;
    cin>>n;
    vector<int> a;
    
    for(int i=0; i<n; i++){
    cin>>m;
    a.push_back(m);
    }
    
   int max_so_far = a[0], i;
   int curr_max = a[0];
 
   for (i = 1; i < n; i++)
   {
        curr_max = max(a[i], curr_max+a[i]);
        max_so_far = max(max_so_far, curr_max);
   }
   cout<<max_so_far<<"\n";
    return 0;
}
