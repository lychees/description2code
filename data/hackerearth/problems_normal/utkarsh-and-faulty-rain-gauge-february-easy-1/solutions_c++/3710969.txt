#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
typedef pair<int,int> ii;
typedef vector<int> vi;

typedef long double ld;
const int MOD = 1e9 + 7;

int n[1001];
int r[1001];

int main()
{
	ios_base::sync_with_stdio(false); cin.tie(0);
	int t; cin >> t;
	ld x = 1;
	for(int i = 0; i < t; i++)
	{
		cin >> n[i] >> r[i];
		x *= 1.5;
		x *= n[i];
		x *= (n[i]+1);
		x /= r[i];
		x /= (2*n[i]+1);
	}
	cout << fixed << setprecision(4) << x;
	return 0;
}
