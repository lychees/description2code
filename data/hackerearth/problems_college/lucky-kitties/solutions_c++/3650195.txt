#include<stdio.h>
int main()
{
	long long int t,k,prev=0,mod=1000000007,total;
	unsigned long long int i,n,h,lucky =0;
	scanf("%d%llu",&k,&n);
	for(i=1;i<=n;i++)
	{
		h=i;
		total = 0;
		while(h!=0)
		{
			t = h%10;
			if(t == 4 || t==7)
				total++;
			h = h/10;
		}
	        prev = total + prev;
		if(prev > k)
		{
			lucky++;
			if(lucky >= mod)
				lucky = lucky%mod;
			prev = 0;
		}
	}
	printf("%llu\n",lucky);
	return 0;
}
