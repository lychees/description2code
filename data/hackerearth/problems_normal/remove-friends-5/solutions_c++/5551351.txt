#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long int n,k,i;
        cin>>n>>k;
        long long int stck[1000005],top=-1,len=0,a;
        for(i=0;i<n;i++)
        {
            cin>>a;
            if(top==-1)
            {
                top++;
                stck[top]=a;
            }
            else if(a>stck[top])
            {
                if(len<k)
                {
                    while(stck[top]<a && len<k && top>-1)
                    {
                        top--;
                        len++;
                    }
                }
                top++;
                stck[top]=a;
            }
            else
            {
                top++;
                stck[top]=a;
            }
        }

        if(len<k)
        {
            long long int z;
            if(len==0)
            {
              z=top-k+1;
            }
            else
                {
                    z=top-len+1;
                }


            for(i=0;i<z;i++)
                cout<<stck[i]<<" ";
        }
        else
        {
        long long int y=0;
        while(y<=top)
        {
            cout<<stck[y]<<" ";
            y++;
        }
        }
        cout<<endl;
    }
}
