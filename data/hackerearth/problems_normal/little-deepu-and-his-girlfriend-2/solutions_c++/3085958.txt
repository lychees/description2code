#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
	int T,M,N;
    cin>>T;
    while(T) {
    	cin>>M>>N;
    	int arr[M+1],x;
    	vector<int> set;
    	for(int i=0;i<N;i++) {
    		cin>>x;
    		set.push_back(x);
    	}
    	arr[0] = 0;
    	sort(set.begin(),set.end());
    	for(int i=0;i<=M;i++) {
    		arr[i] = 0;
    	}
    	for(int i=0;i<=M;i++) {
    		if(arr[i]==0) {
    			int j=0;
    			while(j<N && i+set[j]<=M) {
    				arr[i+set[j]] = 1;
    				j++;
    			}
    		}
    	}
    	if(arr[M]) {
    		cout<<"Little Deepu\n";
    	}
    	else {
    		cout<<"Kate\n";
    	}
    	T--;
    }
    return 0;
}
