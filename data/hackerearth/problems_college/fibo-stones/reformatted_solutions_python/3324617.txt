t = raw_input()
t = t.split()
n = int(t[0])
m = int(t[1])
a = [0]*100

a[1] = 1
a[2] = 1
for i in range(3, 90):
    a[i] = a[i-1]+a[i-2]


x = raw_input()
x = x.split()
count = 0

for i in range(0, m):
    x[i] = int(x[i])
    if x[i] == 1:
        count += 1
sum1 = 0


for i in range(3, n+1):
    if a[i] not in x:
        sum1 += a[i]

if count == 1:
    sum1 += 1
if count == 0:
    sum1 += 2

print sum1
