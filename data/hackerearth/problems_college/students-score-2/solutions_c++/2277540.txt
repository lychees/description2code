#include <iostream>
#include <vector>

using namespace std;

typedef long long int Lint;

int main()
{
	ios_base::sync_with_stdio( false );
	Lint n, i, maxVal, fVal;
	
	cin >> n;
	
	vector<Lint> A( n );
	
	for( i =0 ;i < n;i++ )
		cin >> A[ i ];
	
	maxVal = A[ 0 ];
	fVal = A[ 0 ];
	
	for( i = 1;i < n;i++ ) {
		maxVal = max( maxVal + A[ i ], A[ i ] );
		fVal = max( fVal, maxVal );
	}
	
	if( fVal > 0 )
		cout << fVal << "\n";
	else
		cout << "0\n";
		
	return 0;
}
