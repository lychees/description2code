#include<bits/stdc++.h>
#define ll long long int
void power(ll F[2][2], ll n); 
ll fib(ll n)
{
  ll F[2][2] = {{1,1},{1,0}};
  if (n == 0)
      return 0;
  power(F, n-1);
 
  return F[0][0];
} 
void multiply(ll F[2][2], ll M[2][2])
{
  ll x =  F[0][0]*M[0][0] + F[0][1]*M[1][0];
  ll y =  F[0][0]*M[0][1] + F[0][1]*M[1][1];
  ll z =  F[1][0]*M[0][0] + F[1][1]*M[1][0];
  ll w =  F[1][0]*M[0][1] + F[1][1]*M[1][1];
 
  F[0][0] = x;
  F[0][1] = y;
  F[1][0] = z;
  F[1][1] = w;
} 
void power(ll F[2][2], ll n)
{
  ll i;
  ll M[2][2] = {{1,1},{1,0}};
  for (i = 2; i <= n; i++)
      multiply(F, M);
}
int main()
{
  ll n,t;
  scanf("%lld",&t);
  while(t--)
  {scanf("%lld",&n);
    printf("%lld\n", (fib(n+2)*fib(n+2)));    
  }
}
