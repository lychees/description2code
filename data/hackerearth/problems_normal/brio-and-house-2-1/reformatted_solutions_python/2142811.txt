import math

t = int(raw_input())
for i in range(t):
    data = raw_input().split(" ")
    r = int(raw_input())
    x1 = int(data[0])
    y1 = int(data[1])
    x2 = int(data[2])
    y2 = int(data[3])
    distance = (abs(y1*(x2 - x1) - x1*(y2 - y1))) / \
        (math.sqrt((x2-x1) ** 2 + (y2-y1)**2))
    if distance == r:
        print "JUST MISSED"
    elif distance > r:
        print "SAFE"
    else:
        print "REPLANNING"
