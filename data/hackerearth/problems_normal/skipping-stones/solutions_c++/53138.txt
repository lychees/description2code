#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
    int n, L, D;
    cin>>n;
    double p[n+2];
    cin>>L;
    cin>>D;

    for(int i=1;i<=n;i++)
        cin>>p[i];

    int d[n+2];

    for(int i=1;i<=n;i++)
        cin>>d[i];

    p[0] = 1.0;
    p[n+1] = 1.0;
    d[0] = 0;
    d[n+1] = D;

    double dp[n+2];
    for(int i=1;i<n+2;i++)
        dp[i] = 0.0;
    dp[0] = 1.0;

    for(int i=1;i<n+2;i++)
    {
        int j = i-1;
        while(d[j]>= d[i]-L && j>=0)
        {
            if(p[i]*dp[j] > dp[i])
                dp[i] = p[i] * dp[j];
            --j;
        }
    }
    printf("%.6lf\n", dp[n+1]);

    return 0;
}
