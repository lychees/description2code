#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
	int N;
	cin>>N;

	map<int,int> m;

	int temp;

	for(int i=0;i<N;i++) {
		cin>>temp;
		m.insert(make_pair(temp,i));
	}

	int count = 0;
	map<int,int>::iterator it = m.begin();

	while(it != m.end()) {
		int val = it->first;
		it++;

		while(it != m.end() && it->first <= (val + 4)) 
			it++;
		count++;
	}

	cout<<count<<endl;

	return 0;
}
