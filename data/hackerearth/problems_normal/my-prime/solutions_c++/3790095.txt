#include <iostream>
using namespace std;

int main()
{
    int n,i,j,count;
    long int s[1000000];
    cin>>n;
    for(i=1;i<=n;i++)
    {
    cin>>s[i];
    if(s[i]==1)
    {
    cout<<s[i]<<endl;
    return 0;
    }
    }
    for(i=1;i<=n;i++)
     {  
     	count=0;
     	for(j=1;j<=n;j++)
     	  { 
     	  	if(count>1)
     	  	break;
     	  	if(s[i]%s[j]==0)
     	  	count++;
     	  }
     	  if(count==1)
     	  cout<<s[i]<<" ";
     }	  
     
    return 0;
}
