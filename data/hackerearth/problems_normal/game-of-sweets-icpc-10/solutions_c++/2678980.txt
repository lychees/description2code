#include<bits/stdc++.h>
using namespace std;
int a[100005];
int b[100005];
int good[100005];
int off[100005];
set<int> s;
int main() {
	ios::sync_with_stdio(false);
//	freopen("/home/ahmed/testing.in", "r", stdin);
//	freopen("/home/ahmed/WA.out", "w", stdout);

	int t; t = 1;
//	int t; scanf("%d", &t);
	while (t--) {
		int n; scanf("%d", &n);
		clock_t begin = clock();
		for (int i = 0; i < n; i++)
			scanf("%lld", a + i);
		for (int i = 0; i <= 100000; i++)
			good[i] = 0, off[i] = 0;
		sort(a, a + n);
		int minn = 100000;
		int maxx = a[n - 1] - a[0];
		int ind = 0;
		b[ind++] = a[0];
		for (int i = 1; i < n; i++)
			if (a[i] != a[i - 1])
				b[ind++] = a[i];
		for (int i = 1; i < ind; i++)
			minn = min(minn, b[i] - b[i - 1]);
		n = ind;
		int ans = 0;

		for (int i = 0; ans < maxx - minn + 1 && i < n; i++) {
			for (int j = i + 1; ans < maxx - minn + 1 && j < n; j++)
				if (good[b[j] - b[i]] == 0)
					good[b[j] - b[i]] = 1, ans++;
			clock_t end = clock();
			double elapsed_secs = double(end - begin) / CLOCKS_PER_SEC;
			if (elapsed_secs > 1.9)
				break;
		}
		for (int i = 1; i < n; i++)
			if (a[i] == a[i - 1]) {
				ans++;
				break;
			}
		cout << ans << endl;
	}

	return 0;
}
