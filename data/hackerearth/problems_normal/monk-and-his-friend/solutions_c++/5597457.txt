#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll countBit(ll m,ll p)
{
    ll k,l,c=0;
    while(m || p)
    {
        k=m%2;
        l=p%2;
        if(k!=l)
            c++;
        m>>=1;
        p>>=1;
    }
    return c;
}
int main()
{
    ll t,m,p;
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cin>>t;
    assert(t>=1 && t<=10000);
    while(t--)
    {
        cin>>m>>p;
        assert(m>=0 && m<=10000000000000000);
        assert(p>=0 && p<=10000000000000000);
        cout<<countBit(m,p)<<endl;
    }
    return 0;
}
