#include <stdio.h>
#include <bits/stdc++.h>
#define MAXN 1000005

using namespace std;
int cnt[MAXN];

int main()
{
    int t; scanf("%d", &t);
    while(t--) {
    	for(int i = 0; i < MAXN; i++) cnt[i] = 0;
    	int n; scanf("%d", &n);
    	int rez = 0;
    	for(int i = 0; i < n; i++) {
    		int a, b; scanf("%d%d", &a, &b);
    		cnt[a]++; 
    		if(cnt[b] == 0) rez++;
    		else cnt[b]--;
    	}
    	printf("%d\n", rez);
    }
    return 0;
}
