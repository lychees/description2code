#define ll long long int
#define maxN 200
#define maxVal (ll)400000
#define minVal -100000000
#define mod 1000000007LL

#define gcd(a,b) __gcd(a,b)

using namespace std;

ll n;
ll a[maxN+5];
ll b[maxN+5];
ll da[maxVal+5];
ll db[maxVal+5];

ll powmod(ll a,ll b)
{
	ll x=1,y=a;
	while(b>0)
	{
		if(b%2)
			x=(x*y)%mod;
		y=(y*y)%mod;
		b=b/2;
	}
	return x;
}

int main()
{
    ll i,j,q,sa,sb,ans,x,y,p;
    scanf("%lld",&n);
    sa=0;
    for(i=0;i<n;i++)
    {
    	scanf("%lld",&a[i]);
    	sa+=a[i];
    }
    sb=0;
    for(i=0;i<n;i++)
    {
    	scanf("%lld",&b[i]);
    	sb+=b[i];
    }
    scanf("%lld",&q);
    da[0]=db[0]=1;
	for(j=0;j<n;j++)
    	for(i=sa;i>=1;i--)
    		if(a[j]<=i)
    		{
    			da[i]+=da[i-a[j]];
    			if(da[i]>=mod)
    				da[i]%=mod;
    		}
    for(i=1;i<=sa;i++)
    {
    	da[i]+=da[i-1];
		if(da[i]>=mod)
			da[i]%=mod;
	}
	for(j=0;j<n;j++)
    	for(i=sb;i>=1;i--)
    		if(b[j]<=i)
    		{
    			db[i]+=db[i-b[j]];
    			if(db[i]>=mod)
    				db[i]%=mod;
    		}
	for(i=1;i<=sb;i++)
    {
    	db[i]+=db[i-1];
    	if(db[i]>=mod)
    		db[i]%=mod;
    }
    //printf("%lld %lld\n",sa,sb);
	ans=powmod(2,2*n);
	x=0;
	for(i=0;i<=sa&&(i+q)<sb;i++)
	{
		p=da[i];
		if(i!=0)
			p=p-da[i-1];
		if(p<0)
			p+=mod;
		if(p)
		{
			//printf("%lld ",i);
			y=db[sb];
			y=y-db[i+q];
			if(y<0)
				y=y+mod;
			x=(x+(y*p)%mod)%mod;
		}
	}
	for(i=0;i<=sb&&(i+q)<sa;i++)
	{
		p=db[i];
		if(i!=0)
			p=p-db[i-1];
		if(p<0)
			p+=mod;
		if(p)
		{
			y=da[sa];
			y=y-da[i+q];
			if(y<0)
				y=y+mod;
			x=(x+(y*p)%mod)%mod;
		}
	}
	ans=ans-x;
	if(ans<0)
		ans+=mod;
	printf("%lld\n",ans);
    return 0;
}
