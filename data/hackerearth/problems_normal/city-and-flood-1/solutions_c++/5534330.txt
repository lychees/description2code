int arr[100005];
void initialize(int n)
{
	repeq(i,1,n)
		{
				arr[i]=i;
		}
}
int root(int x)
{
	while(x!=arr[x])
	{
		arr[x]=arr[arr[x]];
		x=arr[x];
	}
	return x;
}
void merge(int x,int y)
{
	arr[root(y)]=root(x);
}
int main()
{
	int n,q,s,d;
scanf("%d",&n);
	initialize(n);
	scanf("%d",&q);
	set<int>st;
	while(q--)
	{
			scanf("%d%d",&s,&d);
			merge(s,d);
	}

		st.insert(root(i));

}
