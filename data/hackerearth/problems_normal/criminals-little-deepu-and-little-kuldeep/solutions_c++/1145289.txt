#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<sstream>
#include<algorithm>
using namespace std;
int a[100010];
int main() {
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        int n;
        scanf("%d",&n);
        for(int i=0;i<n;i++) scanf("%d",&a[i]);
        int prevAns=1,ans=1;
        sort(a,a+n);
        for(int i=0;i<n-1;i++) {
            if(a[i]==a[i+1]) ans++;
            else {
                prevAns=max(prevAns,ans);
                ans=1;
            }
        }
        prevAns=max(ans,prevAns);
        printf("%d\012",prevAns);
    }
    return 0;
}
