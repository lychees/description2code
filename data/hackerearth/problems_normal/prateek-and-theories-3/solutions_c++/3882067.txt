#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	int T, N;
	cin>>T;
	for(int t=0;t<T;t++){
		cin>>N;
		long int *a = new long int[N];
		long int *e = new long int[N];
		for(int n=0;n<N;n++){
			cin>>a[n]>>e[n];
			e[n]--;
		}	
		sort(a,a+N);
		sort(e,e+N);
		int in=1, maxVal=1;
   		int i=1,j=0;
   		while((i<N) && (j<N)){
      		if(a[i]<=e[j]){
          		in++;
 				maxVal=max(maxVal,in);
          		i++;
      		}
      		else{
          		in--;
          		j++;
      		}
   		}
		cout <<maxVal<< endl;
	}
    return 0;
}
