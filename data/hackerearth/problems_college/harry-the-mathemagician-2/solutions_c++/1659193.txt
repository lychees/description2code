#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

bool even(ll N,ll M){
    ll t1,t3;
    if(N<M){
    	return false;
    }
    while(M){
        t1=N%2;
        N=N/2;
        t3=M%2;
        M=M/2;
        if(t1<t3){//0c1
            return true;
        }
    }
    return false;
}

int main(){
    ll M,N;
    int T;
    cin >> T;
    while(T--){
        cin >> N >> M;
        puts(even(N,M)?"even":"odd");
    }
    return 0;
}
