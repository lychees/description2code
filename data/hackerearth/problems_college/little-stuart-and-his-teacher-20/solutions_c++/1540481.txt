#define MAXX 1000000
struct node {
	ll x;
	ll y;
};
ll gcd(ll a, ll b)
{
	return b == 0 ? a : a > b ? gcd(b, a%b) : gcd(a, b%a);
}
bool myfunc(struct node a,struct node b)
{
	if(a.x == b.x) {
		return a.y < b.y;
	} else {
		return a.x < b.x;
	}
}
ll phi[1000010];
void pre()
{
	ll i;
	for(ll i=2; i<=MAXX; i++)
    {
        if(phi[i] == 0)
        {
            phi[i] = i-1;
 
            for(ll j=2*i; j<=MAXX; j+=i)
            {
                if(phi[j] == 0)
                {
                    phi[j] = j;
                }
                phi[j] -= (phi[j]/i);
            }
        }
    }
    for(i = 2; i <= MAXX;i++) {
    	phi[i]= phi[i-1] + phi[i];
    }
}
int main()
{
	pre();
	int t;
	cin >> t;
	while(t--) {
		ll n;
		cin >> n;
		ll ans1 = phi[n];
		ll ans2 = (n * (n-1)) / 2;
		ll ans = ans2 - ans1;
		cout << ans <<  endl;
	}

	return 0;
}
