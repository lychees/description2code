#include<bits/stdc++.h>
using namespace std;

int main()
{
	stack<int>s;
	int n, q_type, cost;
	cin >> n;

	while(n--)
	{
		cin >> q_type;
		
		//customer query
		if(q_type==1)
		{
			if(s.empty())
				cout<<"No Food"<<endl;
			else
			{
				cout<<s.top()<<endl;
				s.pop();
			}
		}

		//chef query
		else
		{
			cin >> cost;
			s.push(cost);
		}
	}
}
