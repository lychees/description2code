typedef long long ll;
typedef pair<int,int> pii;
int main()
{
    int t;
    gi(t);
    while(t--)
    {
        string s;
        cin>>s;
        ll ans = s[s.length()-1]-'0';
        for(int i=s.length()-3;i>=0;i-=2)
        {
            ll x = s[i]-'0';
            if(s[i+1]=='+') ans+=x;
            if(s[i+1]=='-') ans-=x;
            if(s[i+1]=='*') ans*=x;
        }
        pl(ans);
    }
    return 0;
}
