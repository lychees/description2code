#include<stdio.h>
inline int fastread()
{
int i=0;
char c=getchar_unlocked();
while(c<48) c=getchar_unlocked();
while(c>47)
{
i=i*10+c-48;
c=getchar_unlocked();
}
return i;
}
int main()
{
    int t;
    long long n;
    t=fastread();
    while(t--)
    {
        n=fastread();
        if(n%2==0) printf("%lld\n",n);
        else printf("%lld\n",n-1);
    }
    return 0;
}
