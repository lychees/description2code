#include <cstdio>
#include <memory.h>
using namespace std;
int main()
{
    char s[1001];
    int h[26],t;
    scanf("%d",&t);
    while(t--)
    {
	memset(h,0,sizeof(h));
	scanf("%s",s);
	for(int i=0;s[i]!='\0';i++)
	{
	    h[s[i]-'a']++;
	}
	int i=1;	
	int tmp=h[0];
	while(i<26 && h[i]==tmp)
	    i++;
	if(i<26)
	    printf("No\n");
	else
	    printf("Yes\n");
    }
    return 0;
}
