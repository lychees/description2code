#include <iostream>
#include <string> 
using namespace std;
int checkpalindrome(string k)
{ int flag=1,i;



for(i=0;i<=(k.length()-1)/2;i++)
{ if(k[i]!=k[k.length()-1-i])
	{
		flag=0;
		break;
		
	}
}
	
return flag;	
	
}




int main()
{
	int t;
	int count;
	cin>>t;
	while(t--)
	{ string a,b,c,d,e;
	int i;
	cin>>a;
	cin>>b;
	count=0;
	if(a.length()>=2)
	{
	for(i=0;i<=a.length()-1;i++)
	{ c = a.substr(0,i+1) + b + a.substr(i+1,a.length()-i-1); 
   
    if(checkpalindrome(c))
	count++;
	
	
	}}
	c=b+a;
	if(checkpalindrome(c))
	count++;
	

	
	cout<<count<<endl;
	
		
		
	}
	
    return 0;
}
