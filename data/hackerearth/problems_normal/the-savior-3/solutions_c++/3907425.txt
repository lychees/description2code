#include <iostream>
using namespace std;

int main()
{
    long long int t,n,i,j;
    cin>>t;
    while(t--)
    {
    	int count=0;
    	cin>>n;
    	long long int a[n];
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	for(i=0;i<n-1;i++)
    	{
    		for(j=i+1;j<n;j++)
    		{
    			if(a[i]!=a[j])
    			{
    				if((a[i]+a[j])%2==0)
    				count++;
    			}
    		}
    	}
    	cout<<count<<"\n";
    }
    return 0;
}
