#include <iostream>
using namespace std;

long int gcd(long int a, long int b)
{
  if (b == 0)
    return a;
  else
    return gcd(b, a%b);
}

long int func(long int a, long int b)
{
  return (a*b)/gcd(a,b);
}

int main()
{
    long int test_case,n,a,b,result,lcm;
    cin>>test_case;
    while(test_case--){
    	cin>>n;
    	cin>>a;
    	cin>>b;
    	if(a==0 || b==0){
    		cout<<"-1"<<"\n";
    		continue;
    	}
    	lcm=func(a,b);
    	result=(n/a)+(n/b)-(n/lcm);
    	cout<<result<<"\n";
    }
    return 0;
}
