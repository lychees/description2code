#include <bits/stdc++.h>
using namespace std;


int main(){
	int t,n,m;
	long long temp;
	scanf("%d",&t);
	for (int i = 0; i < t; ++i)
	{
		scanf("%d",&n);
		scanf("%d",&m);
		set<long long> s;
		for (int j = 0; j < n; ++j)
		{
			scanf("%lld", &temp);
			s.insert(temp);
		}
		for (int j = 0; j < m; ++j)
		{
			scanf("%lld",&temp);
			if(s.find(temp)!=s.end()){
				cout<<"YES"<<endl;
			}
			else{
				cout<<"NO"<<endl;
			}
			s.insert(temp);
		}
	}
	return 0;
}
