#include <iostream>
using namespace std;

int main()
{
            int t;
            string str;
            cin>>t;
            while(t--)
            {
            	cin>>str;
                int l=str.length();
                for(int i=0;i<l;i++)
                {
                	if(str[i]>=97 && str[i]<=99)
                	  cout<<2;
                	else if(str[i]>=100 && str[i]<=102)
                	  cout<<3;
                	else if(str[i]>=103 && str[i]<=105)
                	  cout<<4;
                   else if(str[i]>=106 && str[i]<=108)
                     cout<<5;
                   else if(str[i]>=109 && str[i]<=111)
                     cout<<6;
                   else if(str[i]>=112 && str[i]<=115)
                    cout<<7;
                   else if(str[i]>=116 && str[i]<=118)
                    cout<<8;
                   else
                    cout<<9;
                 }
                 cout<<endl;
            }
    return 0;
}
