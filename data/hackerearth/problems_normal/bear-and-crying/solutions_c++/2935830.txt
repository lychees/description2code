#include <iostream>
#include <string>
#include <vector>
#define M 1000000007
typedef long long LL;
using namespace std;

int main() {
	string s;
	LL stringSize, sColon, uScore, tmp, ans, i;
	
	cin>>s;
	stringSize = s.size();
	sColon = uScore = tmp = ans = 0;
	for (i=0; i<stringSize; i++) {
	    if(s[i]==';') {
	        ans = (ans + tmp) % M;
	        sColon++;
	    }
	    else {
	        tmp = (2*tmp + sColon) % M;
	    }
	}
	
	cout<<ans<<endl;

	return 0;
}
