a = input()
b = raw_input()
c = len(b)
l = []
for m in range(1, a+1):
    l.append(m)
i = 0
j = 0
while (len(l) > 1):
    if (b[i] == 'a'):
        j += 1
        if (j >= len(l)):
            j = 0
    else:
        l.remove(l[j])
        if (j >= len(l)):
            j = 0
    i += 1
    if (i >= c):
        i = 0
print(l[0])
