#include<bits/stdc++.h>
using namespace std;
long long dp[10000+10][22]; // Dp[i][j] defines number of ways of making value i using coins till jth indexz
long long A[22];
void k()
{
   
    for(int i=1;i<=22;i++)
        A[i]=i*i*i;
}
void pre_calculate()
{
    for(int i=0;i<=21;i++)
        dp[0][i]=1;
    for(int i=1;i<=10000;i++)
    {
        for(int j=1;j<=21;j++)
        {
            if((i-A[j])>=0) // coin can be included
                dp[i][j]+=dp[i-A[j]][j];
            //if(j>=1)
            dp[i][j]+=dp[i][j-1];
        }
    }
}
int main()
{ //freopen("tomar.txt","w",stdout);
     k(); pre_calculate();
  int t,n;
  scanf("%d",&t);
  while(t--)
  {
      scanf("%d",&n);
      printf("%lld\n",dp[n][21]);
  }
    return 0;
}
