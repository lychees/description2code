using namespace std;
bool *P;
int *cnt;
int *ans;
void seive(int n){
    rep(i,0,n){
        P[i]=1;
        cnt[i]=ans[i]=0;
    }
    P[0]=P[1]=0;
    rep(i,2,n){
        if(P[i]){
            cnt[i]=1;
            int j=2*i;
            while(j<n){
                P[j]=0;
                int x = j/i;
                if(x%i)cnt[j]=cnt[x]+1;
                else cnt[j]=cnt[x];
                j+=i;
            }
        }
    }
}
int main(){
    int n;
    cin>>n;
    n++;
    P = new bool[n+5];
    cnt = new int[n+5];
    ans = new int[n+5];
    seive(n+5);
    rep(i,0,n){
        ans[cnt[i]]++;
    }
    int q;
    cin>>q;
    while(q--){
        int k;
        cin>>k;
        if(k>n+4)cout<<0<<endl;
        else cout<<ans[k]<<endl;
    }
}
