#include<bits/stdc++.h>
using namespace std;
int a[1000000]={0};
int main()
{
	a[0]=1;
	a[1]=1;
	int t,n,ans,k;
	cin>>t;
	for(int i=2;i*i<1000000;i++)
	{
		for(int j=i ;j*i<1000000;j++ )
		{
			a[j*i]=1;
		}
		
	}
//	cout<<a[7]<<a[1]<<a[37]<<a[23]<<endl;
	//ans=0;
	while(t--)
	{
		ans=0;
		cin>>k;
		while(k--)
		{
			cin>>n;
		if(a[n]==0)
	    ans++;
		
		}
			cout<<ans<<endl;
	
	}	
	
	

	return 0;
}
