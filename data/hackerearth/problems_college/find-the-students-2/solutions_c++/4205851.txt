// Example program
#include <iostream>
#include<string.h>
using namespace std;
int main()
{
    int n,x[1000],i,j=0;
    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>x[i];
        if(x[i]!=0)
            j++;
    }
    while(j!=0)
    {
        cout<<j<<endl;
        j=0;
        for(i=0;i<n;i++)
        {
            x[i]=x[i]-2;
            if(x[i]<0)
                x[i]=0;
            if(x[i]!=0)
                j++;
        }
    }
    
    return 0;
}
