#include<stdio.h>
#include<math.h>
using namespace std;
int gcd(int a,int b)
{
     int c;
     while(b)
     {
          c=a%b;
          a=b;
          b=c;
     }
     return a;
}
int main()
{
    int T;
    scanf("%d",&T);
    while(T--)
    {
         int N;
         int i,j,k,c,counter=0;
         scanf("%d",&N);
         for(i=2;i<=sqrt((double)(N-1))+1;i++)
         {
                for(j=1;j<i;j++)
                {
                     if((gcd(i,j)==1)&&((i-j)%2)&&((i*i+j*j)<=N))
                     {
                           k=0;
                           c =i*i + j*j;
                           while ((++k)*c <= N)counter++;
                     }
                }
         }
         printf("%d\n",counter);
    }
    return 0;
}
