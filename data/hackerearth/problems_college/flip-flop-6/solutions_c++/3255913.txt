#include <iostream>
//#include<conio.h>
#include<string.h>
using namespace std;

int main()
{
	long long t,i,len,count;
	string s;
    //cout << "Hello World!" << endl;
    cin>>t;
    while(t--)
    {
    	cin>>s;
    	len=s.size();
    	count=0;
    	for(i=1;i<len;i++)
    	{
    		if(s[i]==s[i-1])
    		{
    			count=count+1;
    		}
    	}
    	cout<<count<<"\n";
    }
    return 0;
}
