#include<iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string s;
		cin>>s;
		int min=10;
		int n=s.length();
		for(int i=9;i>=1;i--)
		{
			bool done=false;
			string p=s;
			for(int j=0;j<n;j++)
			{
				if((p[j]-'0')%i==0)
				{
					p[j]=(p[j]-'0')/i+'0';
					if(j==n-1)
					{
						cout<<p<<endl;
						done=true;
					}
				}
				else break;
			}
			if(done) break;
		}
	}
}
