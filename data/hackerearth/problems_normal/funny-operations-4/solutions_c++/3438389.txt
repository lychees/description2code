#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll YOYO(vector<ll> v){
ll xr = v[0];
ll nd = v[0];
for(int i = 1  ; i < v.size() ; ++i ){
  xr = (xr ^ v[i]);
  nd = (nd & v[i]);
}
 return (xr|nd);
}
int main(){

    int t;
    cin >> t;
    while( t-- ){
        int n;
        cin >> n;
        ll a[n];
        for(int i = 0 ; i < n;  ++i )
           cin >> a[i];
        ll mx = 0;
        ll len = 0;
        for(int i = 1 ; i < (1<<n) ; ++i ){
          vector< ll > v;
          for(int j = 0 ; j < n;  ++j) {
            if((i>>j)&1)
            v.push_back(a[j]);
          }
            ll value = YOYO(v);
            if(value > mx ){
            mx = value;
            len = v.size();
            }
            else
            if(mx == value)
            len = max(len , (ll)v.size());
        }
        cout << mx << " "<< len << endl;

    }


return 0;
}
