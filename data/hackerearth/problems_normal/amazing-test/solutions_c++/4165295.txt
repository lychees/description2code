#include <vector>
#include <algorithm>
#include <iostream>
#include <cstring>
using namespace std;
int main()
{
	long t;
	cin>>t;
	while(t--)
	{
		vector<long long> v;
		int n;
	    long long a[100][100],tm,tp;
		cin>>n>>tm;
		for(long i=0;i<n;i++)
		{
			cin>>tp;
			v.push_back(tp);
		}
		if(n<=1)
		{
			if(v[0]<=tm)
			cout<<"YES"<<endl;
			else
			cout<<"NO"<<endl;
		}
		else
		{
			sort(v.begin(),v.end());
			int j=0;
			long long mx1=0,s=0;
			for(long i=0;i!=n;i++)
			{
				s+=v[i];
				if(v[i]>mx1 && v[i]<=tm)
				mx1=v[i];
				a[0][i]=v[i];
				if(s<=tm)
				{
					a[i][i]=s;
					j=i;
				}
			}
			if(a[j][j]>mx1)
			mx1=a[j][j];
			if(s>(tm*2))
			cout<<"NO"<<endl;
			else
			{
				long long mx=0;
				for(long i=1;i<=j;i++)
				{
					for(long k=i+1;k!=n;k++)
					{
						mx=0;
						for(long ii=i-1;ii<k;ii++)
						{
							tp=a[i-1][ii]+v[k];
							if(tp<=tm && tp>mx)
							mx=tp;
						}
						a[i][k]=mx;
						if(mx>mx1)
						mx1=mx;
					}
				}
				if((s-mx1)<=tm)
				cout<<"YES"<<endl;
				else
				cout<<"NO"<<endl;
			}
		}
	}
	return 0;
}
