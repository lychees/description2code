#include<bits/stdc++.h>
using namespace std;
int n,m,b[15][27]={0};
long long int dp[13][1030],ans,mod=1000000007;
char arr[15][7],s[15];
int func(int mask,int pos){
	if(pos==m)
		return 1;
	if(dp[pos][mask]!=-1)
		return dp[pos][mask];
	long long int val=0;
	for(int i=0;i<n;i++){
		if(mask&(1<<i))
			continue;
		int am=b[i][s[pos]-'a'];
		val+=(am*func(mask|(1<<i),pos+1))%mod;
		val%=mod;
	}
	dp[pos][mask]=val;
	return val;
}
int main(){
	memset(dp,-1, sizeof(dp));
	cin>>n>>m;
	for(int i=0;i<n;i++)
		for(int j=0;j<6;j++)
			cin>>arr[i][j];
	cin>>s;
	for(int i=0;i<n;i++)
		for(int j=0;j<6;j++)
			b[i][arr[i][j]-'a']++;
	cout<<func(0,0)<<endl;
	return 0;
}
