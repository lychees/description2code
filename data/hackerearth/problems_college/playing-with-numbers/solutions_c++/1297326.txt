#include<bits/stdc++.h>
using namespace std;

typedef long long ll;

int main() {
    int tt;
    scanf("%d",&tt);
    bool flag=true;
    while(tt--) {
        ll n;
        scanf("%lld",&n);
        int val=__builtin_popcountll(n);
        if(flag==true) {
            if(val&1) {
                puts("Sherlock");
            } else {
                puts("Watson");
                flag=!flag;
            }
        } else if(flag==false) {
            if(val&1) {
                puts("Watson");
            } else {
                puts("Sherlock");
                flag=!flag;
            }
        }
    }
    
    return 0;
}
