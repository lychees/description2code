#include <iostream>
using namespace std;

int main()
{
    int t,i,n,j=0,maxi=0;
    cin>>t;
    int ans[t],num[t];
    for(i=0;i<t;i++)
    {
    	cin>>num[i];
    	if(num[i]>maxi) maxi = num[i];
    }
    int dp[maxi+1];
    dp[0]=maxi;dp[1]=maxi;
    dp[2]=1;dp[3]=1;dp[5]=1;dp[7]=1;
    dp[4]=2;dp[6]=2;
    for(i=8;i<maxi+1;i++)
    {
    	dp[i]=min(dp[i-7],dp[i-5]);
    	dp[i]=min(dp[i],dp[i-3]);
    	dp[i]=min(dp[i],dp[i-2]);
    	dp[i]+=1;
    }
    for(i=0;i<t;i++)
    {
    	if(dp[num[i]]>=maxi) ans[i]=-1;
    	else ans[i]=dp[num[i]];
    	cout<<ans[i]<<endl;
    }
    return 0;
}
