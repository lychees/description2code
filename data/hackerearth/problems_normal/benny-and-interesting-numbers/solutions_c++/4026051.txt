vi p;

int f(ll x) {
  for (ll i : p) {
  	if (i * i > x) break;
  	if (x % i == 0) {
    	x /= i;
    	if (x % i == 0) return 0;
  	}
  }
  return 1;
}

int main() {
	vi erat(1000000, 1);
	for (int i = 2; i * i < erat.size(); ++i) if (erat[i]) {
		for (int j = i * i; j < erat.size(); j += i) erat[j] = 0;
	}
	for (int i = 2; i < erat.size(); ++i) if (erat[i]) p.push_back(i);
    int T;
    cin >> T;
    for (int test = 1; test <= T; ++test) {
    	ll n;
    	cin >> n;
    	if (f(n)) {
    		cout << "Yes\n";
    	} else cout << "No\n";
    }
    return 0;
}
