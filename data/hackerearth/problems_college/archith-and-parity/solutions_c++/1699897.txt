#include <iostream>
using namespace std;

int main()
{
    int n , arr[100] , a[100] , r , s = 0;

    cin >> n;

    for(int i = 0 ; i < n ; i++)
    {
        cin >> arr[i];
    }

    for(int i = 0 ; i < n ; i++)
    {
        while (arr[i] > 0)
        {
            r = arr[i] % 2;
            if (r == 1)
            {
                s++;
            }
            arr[i] = arr[i] / 2;
        }

        a[i] = s;
        s = 0;
    }

    for(int i = 0 ; i < n ; i++)
    {
        if(a[i] % 2 == 0)
        {
            cout << "even" <<  endl;
        }

        else
        {
            cout << "odd" <<  endl;
        }
    }

    return 0;
}
