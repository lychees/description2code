#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t ;
	cin >>t ;
	long long int arr[65];
	arr[0] = 1 ;
	for( int i = 1 ; i <= 64 ; i++ )
		arr[i] = arr[i-1]*  2 ;
	for( int j =1 ; j <= t ;j++ )
	{
		long long int n , k ;
		cin >> n >> k ;
		vector< int > tmp;
		long long int sum = 0 ;
		for( int i = 0 ;i <= 64 ; i++ )
		{
			if( ( arr[i] & n ) == 0)
			{
				tmp.push_back(i);
				sum += arr[i];
			}
		}
		assert( sum <= k );
		long long int ans = 0;
		//for( int i = 0 ; i < 5 ; i++ ) cout << tmp[i] << " ";
		for(  int i = 0 ; i <= 64 ; i++ )
		{
			if( ( arr[i] & k ) != 0  )
			{
				//puts("rishabh");
				ans += arr[ tmp[i] ];
			} 
		}
		printf("Case #%d: %lld\n", j , ans );
	}
}
