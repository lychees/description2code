#include<iostream>
#include<vector>

using namespace std;

int main()
{
    int n,m,ip;
    cin>>n;
    vector<int>vec;
    for(int i=0;i<n;i++)
    {
       cin>>ip;	
     	vec.push_back(ip);
    }
     vector<int>vec1(n);
     
     vec1[0]=vec[0];
     m=vec1[0];
     for(int i=1;i<n;i++)
     {
      vec1[i]=(i+1)*vec[i]-m;
       m=m+vec1[i];
     }
     
     for(auto i:vec1)
      cout<<i<<" ";
      return 0;
     
    
}
