#include <iostream>
using namespace std;

int main()
{
	int t, n, x;
	cin>>t;
	while(t--) {
		cin>>n>>x;
		long array[n];
		for(int i=0;i<n;i++) {
			cin>>array[i];
		}
		int i=0, j=0;
		long sum = 0;
		int flag = 0;
		//1 3 4 5 2
		while(i<n) {
			if(sum > x) {
				sum -= array[i];
				i++;
			}
			else if(sum == x) {
				flag = 1; 
				break;
			}
			else {
				sum = sum+array[j++];
			}
		}
		if(flag == 1) cout<<"YES\n";
		else cout<<"NO\n";
	}
    return 0;
}
