#include <cstdio>
#include <iostream>
#include <map>
#include <vector>
#include <utility>
#include <cstring>
using namespace std;
int main()
{
	int t;
	t = 882;
	while(t--)
	{
		char s[32811];
		scanf(" %s",s);
		int slen = strlen(s);
		int res = 0;
		int curr = slen-1;
		if(curr == 0)
		{
			printf("0\n");
		}
		else
		{
			while(1)
			{
				if(curr <= 0)
				{
					break;
				}
				if(curr >= 0 && s[curr] == '0')
				{
					while(curr >= 0 && s[curr] == '0')
					{
						res++;
						curr--;
					}
				}
				else
				{
					res++;
					while(curr >= 0 && s[curr] == '1')
					{
						res++;
						curr--;
					}
					if(curr < 0)
					{
						break;
					}
					else
					{
						s[curr] = '1';
					}
				}
			}
			printf("%d\n", res);
		}
	}
	return 0;
}
