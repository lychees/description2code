#include<bits/stdc++.h>
using namespace std;
long long int a[100][100];
long long int C(long long n,long long k)
{
	if(n==k || k==0) return 1;
	if(a[n][k]!=0) return a[n][k];
	return a[n][k]=C(n-1,k)+C(n-1,k-1);
}
long long fact(long long n)
{
	long long ans=1,i;
	for(i=1;i<=n;i++)
	{
		ans*=i;
	}
	return ans;
}
int main()
{
	long long int T,i,j,k,N;
	cin>>T;
	while(T--)
	{
		cin>>i>>j;
		k=fact(i+j)*C(7,i)*C(5,j);
		cout<<k<<"\n";
	}
}
