#include<bits/stdc++.h>
using namespace std;
int main()
{
    int test;
    scanf("%d",&test);
    while(test--)
    {
        int n,d;
        scanf("%d%d",&n,&d);
        int cnt=0,flag=0;
        while(cnt<d)
        {
            cnt+=2;
            n--;
            if(n<0)
            {
                flag=1;
                break;
            }
            if(cnt%6==0)
                n+=2;
        }
        if(flag==0)
            printf("Yes\n");
        else
            printf("No\n");
    }
}
