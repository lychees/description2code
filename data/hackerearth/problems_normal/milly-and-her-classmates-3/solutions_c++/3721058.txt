#include <iostream>
using namespace std;

int main()
{
	typedef long long int ulli;
	ulli T, n, a[10000];
	cin >> T;
	while(T--)
	{
		cin >> n;
		for(ulli i=0; i<n; ++i)
			cin >> a[i];
		ulli ans1=-1, ans2=-1;
		ulli flag=0, store1=0, store2=0;
		for(ulli i=1; i<n; ++i)
		{
			if(a[i]>a[i-1] && (flag==0 || flag==2))
				continue;
			else if(a[i]<a[i-1] && flag==0)
				store1=i, flag=1;
			else if(a[i]<a[i-1] && flag==1)
				continue;
			else if(a[i]>a[i-1] && flag==1)
				store2=i, flag=2;
			else if(a[i]<a[i-1] && flag==2)
			{
				flag=3;
				break;
			}
		}
		if(flag==3 || flag==0)
			ans1=-2, ans2=-2;
		if(flag==1)
		{
			ans1=store1-1, store2=n, ans2=store2-1;
			if(ans1>0 && a[ans1-1]>a[ans2])
				ans1=-2, ans2=-2;
		}
		else if(flag==2)
		{	
			ans1=store1-1, ans2=store2-1;
			if((ans1>0 && a[ans1-1]>a[ans2]) || (ans2<(n-1) && a[ans2+1]<a[ans1]))
				ans1=-2, ans2=-2;
		}
		/*if(flag==1 || flag==2)
		{
			ulli a1[10000];
			ulli c=0;
			cout << store1 << "," << store2 << endl;
			for(ulli i=store1-1; i<(store2-1); ++i, ++c)
			{
				a1[c]=a[i];
			}
			c--;
			for(ulli i=store1-1; i<(store2-1); ++i, --c)
			{
				a[i]=a1[c];
			}
			ulli i=1;
			for(; i<n; ++i)
			{
				if(a[i]<a[i-1])
				{
					ans1=-1, ans2=-1;
					break;
				}
			}
			if(i==n)
				ans1=store1, ans2=store2;
		}*/
		cout << ans1+1 << " " << ans2+1 << endl;
	}
    return 0;
}
