#include <iostream>
//#include<vector>
using namespace std;

int main()
{
    int t,l,zero=0,one=0,j,sum=0;
    
    cin>>t;int i,n[t];
    string s[t];
    for(i=0;i<t;i++)
    {
    	cin>>s[i];
    	cin>>n[i];
    }
    for(i=0;i<t;i++)
    {
    	
    	zero=0;one=0;
    	
    	for(j=0;j<s[i].length();j++)
    	{
    		if(s[i][j]=='1')
    		one++;
    		else
    		zero++;
    	}
    	sum=0;
    	for(j=1;j<=n[i];j++)
    	sum=sum+j;
    	if(sum%2==0)
    	cout<<one;
    	else
    	cout<<zero;
    	cout<<"\n";
    }

}
