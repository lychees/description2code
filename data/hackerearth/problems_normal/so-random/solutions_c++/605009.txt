#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <cmath>

using namespace std;

void solve(float P, float T)
{
	float pp = T / 1440.0;
	float ans = pow(1.0-P,pp);
	ans = 1 - ans;
	printf("%.4lf\n",ans);
	return;
}

int main()
{
	int t;
	cin >> t;
	while (t--)
	{
		float P,T;
		cin >> P >> T;
		solve(P,T);
	}
	return 0;
}
