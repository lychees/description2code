#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
	int n,i;
	cin>>n;
	long long int a[n],m,f=0,diff,ans;
	cin>>m;
	for(i=0;i<n;i++)
		cin>>a[i];
	sort(a,a+n);
	for(i=n-1;i>=1;i--)
	{
		f+=1;
		diff=f*(a[i]-a[i-1]);
		if(diff==m)
		{
			cout<<a[i-1]<<endl;
			return 0;
		}
		else if(diff<m)
			m-=diff;
		else
		{
			if(m%f==0)
				ans=a[i]-m/f;
			else
				ans=a[i]-m/f-1;
			cout<<ans<<endl;
			return 0;
		}
	}
	diff=(f+1)*(a[0]);
		if(diff==m)
			cout<<0<<endl;
		else if(diff<m)
			cout<<"-1"<<endl;
		else
		{
			if(m%f==0)
				ans=a[0]-m/f;
			else
				ans=a[0]-m/f-1;
			cout<<ans<<endl;
		}
		return 0;
}
