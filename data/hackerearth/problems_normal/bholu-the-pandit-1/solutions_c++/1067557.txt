#include <iostream>
#include <algorithm>
#define MOD 1000000007
using namespace std;

int gcd(int x,int y)
{
	int r;
	int a=max(x,y);
	int b=min(x,y);
	while(b!=0)
	{
		r=a%b;
		a=b;
		b=r;
	}
	return a;
}
int main()
{
    int t,a,b,c;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b>>c;
    	int h=gcd(gcd(a,b),c);
    	long long ans=((long long)a/h * b/h)%MOD;
    	ans=(ans*c/h)%MOD;
    	cout<<h<<" "<<ans<<endl;
    }
    return 0;
}
