

def main():
    N, A, B = raw_input().split(" ")
    N = int(N)
    A = int(A)
    B = int(B)
    i = A
    ans = 0
    count = 0
    diff = B - A
    while (i < N):
        ans = ans + i
        count = count + 1
        if ((i + count * diff) < N and (i + count * diff) % A != 0):
            ans = ans + (i + count * diff)
        i = A * (count + 1)
    print ans


main()
