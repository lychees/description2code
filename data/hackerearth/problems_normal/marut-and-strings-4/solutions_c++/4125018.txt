#include <iostream>
#include <cassert>
#include <string>
#include <algorithm>
#include <ctype.h>
using namespace std;

int req(string str){
    int c=0;
    int s=0;
    int l=0;
    int length=str.length();
    assert(1<=length and length<=100);
    for(int i=0;i<length;i++){
        if(str[i]>=65 && str[i]<=90){
            str[i]=str[i]+32;
            c++;
        }
        else if(str[i]>=97 && str[i]<=122){
            str[i]=str[i]-32;
            s++;
            
        }
        else if(!isalpha(str[i])){
            l++;
        }
        
    }
    
     if(l==length || length>100)
    {
        cout<<"Invalid Input"<<endl;
    }
    else if(s!=0 && c==0){
       cout<<0<<endl;
    }
    else if(c!=0 && s==0){
        cout<<0<<endl;
    }
    else {
        cout<<min(c,s)<<endl;
    }
  
    return 0;
}
