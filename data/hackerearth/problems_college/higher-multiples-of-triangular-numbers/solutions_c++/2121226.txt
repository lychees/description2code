#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    long long t;
    cin>>t;
    while(t--)
    {
        long long n,x=1,y=0,i,j,ans=0;
        cin>>n;
        for(i=1;;i+=x)
        {
          for(j=1;j*j<=i;j++)if(i%j==0){y++;if(i/j!=j)y++;}
          if(y>n){ans=i;break;}y=0;x++;
        }
        cout<<ans<<endl;
    }
    return 0;
}
