#include<bits/stdc++.h>
#define MOD 1000000007
using namespace std;
int main()
{
   int test;
   scanf("%d",&test);
   while(test--)
   {
      long long int l,b,d;
      scanf("%lld%lld%lld",&l,&b,&d);
      long long int a1=(l/d)%MOD;
      long long int a2=(b/d)%MOD;
      printf("%lld\n",a1*a2%MOD);
   }
   return 0;
}
