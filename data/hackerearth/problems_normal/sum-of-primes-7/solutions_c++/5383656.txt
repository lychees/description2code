#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int a[1000001]={0};
    a[1]=1;a[0]=1;
    long long int b[1000001];
    for(int i=2;i*i<=1000000;i++)
    {
    	for(int j=2*i;j<=1000000;j+=i)
    	{
    		a[j]=1;
    	}
    }
    b[0]=0;
    b[1]=0;
    long long int sum=0;
    for(int i=2;i<=1000000;i++)
    {
    	if(a[i]==0)
    	{
    		sum=sum+i;
    		b[i]=sum;
    	}
    	else
    	b[i]=sum;
    }
    int n;
    cin>>n;
    while(n--)
    {
    	int l,r;
    	cin>>l>>r;
    	cout<<(b[r]-b[l-1])<<endl;
    }
    
    return 0;
}
