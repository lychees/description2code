#include <iostream>
using namespace std;
int rev(int n)
{
	int s=0,r;
	while(n>0)
	{
		r=n%10;
		s=r+s*10;
		n=n/10;
	}
	return s;
}

int main() {
	// your code goes here
	int t,p,q;
	cin>>t;
	while(t--)
	{
		int s=0;
		cin>>p>>q;
		s=s+rev(p)+rev(q);
		cout<<rev(s)<<"\n";
	}
	return 0;
}
