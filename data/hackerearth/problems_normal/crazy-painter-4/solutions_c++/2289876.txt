#include<bits/stdc++.h>
using namespace std;
#define LL long long

int main()
{
    int t;
    cin>>t;

	while(t--)
    {
        LL n;
        int arr_alpha[30];

        int i,j;
        cin>>n;

        for(i = 0; i < 26; i++)
            cin>>arr_alpha[i];

        LL sum = 0;

        for(i=0; i < 26; i++)
            sum += arr_alpha[i];
        LL num_points = n*12;

        LL sets = num_points/26;

        LL total = 0;
        total = (sets*sum) + 13*(sets-1)*sets;
        int remains = num_points%26;
        for(i = 0; i < remains; i++)
            total += arr_alpha[i];

        total += (sets*remains);

        cout<<total<<endl;
    }
    return 0;
}
