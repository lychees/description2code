#include<iostream>
using namespace std;
int rev(int n){
   int  sum = 0;
   while (n != 0){
      sum = sum * 10;
      sum = sum + n%10;
      n = n/10;
   }
    return sum;
}
int main(){
    int n1,n2,t,f=1;
    cin>>t;
    for (int i=0;i<t;i++){
		f=0;
    	cin>>n1;
    	cin>>n2;
		for(int j=n1;j<=n2;j++){
			if(j==rev(j))
				f++;
		}
		cout<<f<<endl;
    }
    return 0;
}
