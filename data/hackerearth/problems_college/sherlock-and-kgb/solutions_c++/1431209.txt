#include <bits/stdc++.h>

using namespace std;

int main(int argc, char const *argv[])
{
	int t;
	cin>>t;
	while(t--) {
		long long int x, y, z, temp1, temp2, temp3;
		cin>>x>>y>>z;
		temp1 = x;
		temp2 = y;
		temp3 = z;
		while(temp1%3==0)
			temp1/=3;
		while(temp2%5==0)
			temp2/=5;
		while(temp3%2==0)
			temp3/=2;
		if (temp1==1) {	cout<<"K-";	}
		else { cout<<"X-"; }
		if (temp2==1) { cout<<"G-"; }
		else { cout<<"Y-"; }
		if(temp3==1) { cout<<"B "; }
		else { cout<<"Z"; }
		cout<<endl;
	}
	return 0;
}
