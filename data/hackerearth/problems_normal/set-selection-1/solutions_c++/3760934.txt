# include<iostream>
# include<cstring>
using namespace std;

int n;
int a[128];
int sum,sumi[100001];

void solve()
{
    for(int i=0; i<n; i++)
    {
        for(int j=sum; j+1>0; j--)
        {
            if(sumi[j]!=-1 && sumi[j+a[i]]==-1) sumi[j+a[i]]=1;
        }
    }
}

int main()
{
    cin>>n;
    for(int i=0; i<n; i++)
    {
        cin>>a[i];
        sum+=a[i];
    }
    memset(sumi,-1,sizeof(sumi));
    sumi[0]=0;
    solve();
    int t;
    cin>>t;
    for(int i=0; i<t; i++)
    {
        int q;
        cin>>q;
        if(sumi[q]!=-1) cout<<"Yes"<<endl;
        else cout<<"No"<<endl;

    }

}
