#include <bits/stdc++.h>

#define MAXN 1024

using namespace std;

char s[MAXN];
void read()
{
	scanf("%s", &s);
}

long long dp[MAXN][MAXN];

long long rec(int i, int j)
{
	if(i == j - 1) return 1;
	if(i >= j) return 0;

	if(dp[i][j] != -1) return dp[i][j];
	
	long long ans = rec(i + 1, j);
	for(int mid = i; mid < j; mid++)
		if(s[mid] == s[i]) ans = ans + 1 + rec(i + 1, mid);
	
	dp[i][j] = ans;
	return ans;

}

void solve()
{
	memset(dp, -1, sizeof(dp));
	int N = strlen(s);
	printf("%lld\n", rec(0, N));
}

int main()
{
	int T;
	scanf("%d", &T);

	for(int i = 0; i < T; i++)
	{
		read();
		solve();
	}

	return 0;
}
