#define wl(n) while(n--)
#define fl(i,a,b) for(i=a; i<b; i++)
#define rev(i,a,b) for(i=a; i>=b; i--)

#define si(n) scanf("%d", &n)
#define sll(l) scanf("%lld",&l)
#define ss(s) scanf("%s", s)

#define pi(n) printf("%d\n", n)
#define pll(l) printf("%lld\n", l)

#define pc(c) printf("%c\n", c)
#define pd(f) printf("%lf\n", f)

#define debug(x) cout<<"\n#("<<x<<")#\n"
#define nline printf("\n")

#define mem(a,i) memset(a,i,sizeof(a))

#define MOD 1000000007
#define ll long long int
#define u64 unsigned long long int

#define mclr(strn) strn.clear()
#define ignr cin.ignore()
#define PB push_back
#define SZ size
#define MP make_pair
#define fi first
#define sec second

char str1[10004], str2[10004];
int dont[10004];
std::vector<string> v;

int main()
{
	int i, j;

	int cases;

	si(cases);
	
//	getchar();
	wl(cases)
	{
		int c;
		int x = 0;
		c = getchar();
		
		while(c!=EOF && c!='\n')
		{
			str1[x++] = c;
			c = getchar();
		}
		str1[x]='\0';
		

		int limit = x;
		
	//	cout<<str1;nline;cout<<limit;nline;
		
		if(limit == 0)
		{
			cases++;
			continue;
		}

		v.clear();
		mem(dont,0);

		string temp;
		int ind = 0;

		int s = 0;

		fl(i,0,limit)
		{
			if(str1[i]>='0' && str1[i]<='9')
			{
				s += str1[i]-'0';
			}
			else
			{
				str2[ind++] = str1[i];
			}
		}
		
		str2[ind]='\0';
		
		//cout<<str2; nline;

		fl(i,0,ind)
		{
			if(str2[i] == ' ')
			{
				v.PB(temp);
				temp.clear();
			}
			else
			{
				temp += str2[i];

				if(i == ind-1)
				{
					v.PB(temp);
					temp.clear();
				}
			}
		}

		sort(v.begin(), v.end());
		
		limit = v.SZ();

		fl(i,0,limit)
		{
			cout<<v[i]<<" ";
			s -= v[i].SZ();
			cout<<abs(s)<<" ";
		}
		nline;
	}

	return 0;
}
