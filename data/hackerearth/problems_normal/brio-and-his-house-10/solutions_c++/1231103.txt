#include <iostream>
#include <cmath>
#include <iomanip>
#define pi 3.1415
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    
    int T;
    cin>>T;
    double a,b,c;
    for(int i=0;i<T;i++){
    	cin>>a>>b>>c;
    	double r = a*b*c/(sqrt((a+b+c)*(a+b-c)*(c+a-b)*(c+b-a)));
    	r = pi*r*r;
    	cout<<setprecision(4)<<fixed<<r<<endl;
    	//printf("%.4f\n",r);
    	
    }
    return 0;
}
