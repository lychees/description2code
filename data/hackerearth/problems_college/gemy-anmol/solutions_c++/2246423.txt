#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int m,cnt=0,sum=0;
		long long int n,arr[1000000],flag=0;
		cin>>n>>m;
		for(int i=0;i<n;i++)
			cin>>arr[i];
		sort(arr,arr+n);
		for(int i=n-1;i>=0;i--)
		{
			sum+=arr[i];
			cnt++;
			if(sum>=m){
				flag=1;
				break;
			}
		}
		if(flag==0)
			cout<<"-1"<<endl;
		else
		cout<<cnt<<endl;
		
	}
}
