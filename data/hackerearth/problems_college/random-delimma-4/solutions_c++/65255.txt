#include <iostream>
#include <cstdio>
#include <cstring>
#include <iomanip>

using namespace std;

int main()  {

    int t,n,k;scanf("%d",&t);
    while(t--)
    {
        scanf("%d %d",&k,&n);
        if(k>n)printf("1.000000000\n");
        else if(k==1)printf("0.000000000\n");
        else
        {
            long double val = 1.0;
            for(int i=1;i<k;++i)
            {
                val *= (long double)(n-i)/n;
            }
            cout<<fixed<<setprecision(9)<<1-val<<endl;
        }
    }
    return 0;
}
