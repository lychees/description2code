#include <bits/stdc++.h>
using namespace std;

int main()
{
	float x1,x2,x3,y1,y2,y3;
    //cout << "Hello World!" << endl;
    cin>>x1>>y1>>x2>>y2>>x3>>y3;
    vector < pair <float,float> >point;
    float a1=x1+x2-x3;
    float a2=-x1+x2+x3;
    float a3=x1-x2+x3;
    float b1=y1+y2-y3;
    float b2=-y1+y2+y3;
    float b3=y1-y2+y3;
    point.push_back(make_pair(a1,b1));
    point.push_back(make_pair(a2,b2));
    point.push_back(make_pair(a3,b3));
    sort(point.begin(),point.end());
    cout.setf(ios::fixed);
    for ( const pair<float, float> &edge : point )
    {
        cout <<setprecision(4)<<edge.first<<" "<<edge.second<<endl;
    }
    return 0;
}
