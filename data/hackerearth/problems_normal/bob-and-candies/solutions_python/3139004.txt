from itertools import combinations

string = raw_input()
nx = string.split(" ")
n = int(nx[0])
x = int(nx[1])

string = raw_input()
a = string.split(" ")
a = [int(i) for i in a[:n+1]]

if x in a:
	print a.index(x)+1
else:
	result = None
	for i in range(2, n+1):
		output = map(list, combinations(a, i))
		for combi in output:
			if sum(combi) == x:
				result = combi
				break
		if result:
			break
	
	if result:
		for r in result:
			m = a.index(r)
			a[m] = 0
			print m+1,
	else:
		print -1
