#include <bits/stdc++.h>

using namespace std;
int a[100005];
int b[100005];
void swap(int *x, int *y)
{
	int t = *x;
	*x = *y;
	*y = t;
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--) {
	int n;
	scanf("%d",&n);
	for(int i = 0; i <n; i++) {
		scanf("%d",&a[i]);
	}
	for(int i = 0; i <n; i++) {
		scanf("%d", &b[i]);
	}
	int j,count = 0;
	for(int i = n-1; i >= 0; i--) {
		for(j = i; j >= 0; j--) {
			if(a[i] == b[j]) {
				break;
			}
		}
		while(j != i) {
			swap(&b[j],&b[j+1]);
			j++;
			count++;
		}
	}
	cout << count << endl;
	}
	return 0;
}
