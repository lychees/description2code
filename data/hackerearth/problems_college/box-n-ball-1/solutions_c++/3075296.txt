#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    unsigned long int a,b,c,mn;
    int t;
    cin>>t;
    while(t--){
    	cin>>a>>b>>c;
    	mn = a;
    	if(mn > b){
    		mn = b;
    	}
    	
    	if (mn > c)	
    		mn = c;
    
    	//cout<<(float)mn/2.0<<endl;
    	printf("%.1f\n",mn/2.0);
    }
    return 0;
}
