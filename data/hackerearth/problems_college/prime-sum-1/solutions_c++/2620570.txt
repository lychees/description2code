#include <iostream>
#include<algorithm>
using namespace std;
int a[1000002]={0};
int min(int a,int b) { if(a<b) return a;return b;}
int main() {
	
	int t;
	scanf("%d",&t);
	a[0]=0,a[1]=0,a[2]=1,a[3]=1,a[4]=2,a[5]=1,a[6]=2,a[7]=1,a[8]=2;
	for(int i=9;i<1000002;i++)
	{
		a[i]=1+min(min(a[i-2],a[i-3]),min(a[i-5],a[i-7]));
	}
	while(t--)
	{
		int n;
		scanf("%d",&n);
		if(n<2) cout<<-1<<endl;
		else
		cout<<a[n]<<endl;
	}
		
	
	return 0;
}
