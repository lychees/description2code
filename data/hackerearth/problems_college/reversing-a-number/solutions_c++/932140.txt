#include<bits/stdc++.h>
using namespace std;
int main()
{
    std::ios_base::sync_with_stdio(false);
    int t;
    char str[102];
    cin>>t;
    while(t--)
    {
        cin>>str;
        if((str[0]-'0')%2==0)
            cout<<"EVEN"<<endl;
        else
            cout<<"ODD"<<endl;
    }
    return 0;
}
