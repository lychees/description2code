#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

#define MAX 100
#define ull unsigned long long
#define ll long long

ll f[MAX];
int t, n;

ll dp(int n){
    if (n <= 0)
        return 1;
    if (f[n] != -1)
        return f[n];
    f[n] = dp(n - 1) + dp(n - 2);
    return f[n];
}

int main(){
    memset(f, -1, sizeof(f));
    f[0] = 1;
    f[1] = 1;
    scanf("%d", &t);
    while (t--){
        scanf("%d", &n);
        printf("%lld\n", dp(n));
    }
    return 0;
}
