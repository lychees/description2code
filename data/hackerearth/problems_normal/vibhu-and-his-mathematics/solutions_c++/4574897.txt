#include <iostream>
using namespace std;
#define mod 1000000007
#define ll long long
int main()
{
	ll dp[1000007];
	dp[0]=0;
	dp[1]=1;
	dp[2]=2;
	for(long long i=3;i<1000001;i++)
	{
		long long v1 =((i-1) * dp[i-2]) % mod;
		v1=(v1+ dp[i-1]) % mod;
		dp[i]=v1;
	}
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		cout<<dp[n]<<endl;
	}
}
