#include <iostream>
using namespace std;
const int n=150;
int a[n],x[n];

void solve()
{
	int d,xr=0;
cin>>d;
for(int i=0;i<d;i++)
{
	cin>>a[i];
	
	
}
for(int i=0;i<d;i++)
{
	cin>>x[i];
	xr^=a[i]-x[i];
	xr^=x[i]-1;
	
}
if(xr==0)
{
	cout<<"Second"<<endl;
}
else
cout<<"First"<<endl;
	
}

int main()
{
	
	int T;
	cin>>T;
	int i=0;
	while(i<T)
	{
		solve();
		i++;
	}
	
    //cout << "Hello World!" << endl;
    return 0;
}
