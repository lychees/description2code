#include<bits/stdc++.h>
using namespace std;
int main()
{
   int n;
   cin>>n;
   int a[n], b[n];
   int i,j;
   for(i=0;i<n;i++)
   {
   	  cin>>a[i];
   	  b[i]=a[i];
   }
   sort(b,b+n);
   int k,count ,m,l;
   i=0;
   while(i<n)
   {
   	  k=b[i];
   	  j=i+1;
   	  while(j<n && b[j]==k)
   	  	j++;
   	  for(l=i;l<n;l++) b[l]-=k;
   	  count=0;
   	  for(l=0;l<n;l++)
   	  {
   	  	 if(a[l]>0) 
   	  	 {
   	  	 	count++;
   	  	 	a[l]-=k;
   	  	 }  	 
   	  }
   	  cout<<count<<endl;
   	  i=j;
   }
   return 0;
}
