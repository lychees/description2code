#include <iostream>
#include<vector>
#include <algorithm> 
using namespace std;

int main()
{
    int TC;
	cin>>TC;
	for(int tc =0 ; tc<TC; tc++)
	{
		vector<long> Values;
		int size, largeIndex, smallIndex;
		cin>>size;
		for(int i=0; i<size; i++)
		{
			long val;
			cin>>val;
			Values.push_back(val);
		}
		cin>>largeIndex>>smallIndex;
		sort(Values.begin(), Values.end());
		int sum = Values.at(Values.size()-largeIndex)+ Values.at(smallIndex-1);
		cout<<sum<<endl;
	}
    return 0;
}
