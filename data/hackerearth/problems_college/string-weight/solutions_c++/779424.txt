#include <iostream>
#define ll long long
using namespace std;
int main(){
   ll T,sum;
   int ia,N;
   char l;
   string W;
   cin>>T;
   while(T--){
      cin>>W;
      ll la[1000]={0};
      
      for (ll i=0;i<W.length();i++){
         ia=(int)W[i]-96;
         la[ia-1]+=ia;
      }
      cin>>N;
      while(N--){
         cin>>l;
         //sum=(int)l-97;
         for(int i=0;i<26;i++){
            if(i==(int)l-97)
              la[i]=0;
         } 
         sum=0; 
      }
      for(int i=0;i<26;i++){
         sum=sum+la[i];
      }
      cout<<sum<<endl;
      
   }

   return 0;
}
