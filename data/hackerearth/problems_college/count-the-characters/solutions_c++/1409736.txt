#include <bits/stdc++.h>
using namespace std;
int mark[150];
int cnt[150];
int main()
{
    int n,count=0;
    cin>>n;
    string str[n];
    for(int i=0;i<n;i++)
    cin>>str[i];
    for(int i=0;i<n;i++){
    	for(int j=0;j<str[i].length();j++){
    		if(!mark[str[i][j]]){
    		//	cout<<"df";
    			cnt[str[i][j]]+=1;
    		mark[str[i][j]]=1;
    		
    		}
    	}
    	fill( mark, mark + sizeof( mark ), 0 );
    	
}
    for(int i=97;i<123;i++){
    //	cout<<cnt[i]<<" ";
	if(cnt[i]==n)
	count++;}
	cout<<count<<endl;
    return 0;
}
