#include<iostream>
using namespace std;
#include<stack>

int main() {
	long t, i, sum = 0, val;
	std::stack<long> s;
	
	cin>>t;
	for(i=0;i<t;i++) {
		cin>>val;
		if(val != 0)
			s.push(val);
		else
			s.pop();
	}
	
	while(!s.empty()) {
		sum += s.top();
		s.pop();
	}
	cout<<sum;
	
	return 0;
}
