#include<bits/stdc++.h>
#define ll long long int
#define ld long double
using namespace std;
int main()
{
    int t,x,a,temp,five,three,flag,restrict,diff;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&x);
        flag=0;
        restrict=0;
        five=0;
        three=0;
        temp=x;
        a=1;
       if(x==1||x==2)
            printf("-1\n");
       else if(x%3==0)
            {
                for(int i=0;i<x;i++)
                    printf("5");
                    printf("\n");
            }
            else
            {
                while(flag!=1)
                {
                    diff=x-(5*a);
                    if(diff%3==0)
                        flag=1;
                    else
                        a++;
                    if(5*a>x)
                    {
                        restrict=1;
                        printf("-1\n");
                        break;
                    }

                }
                if(restrict==0)
                {
                    for(int i=0;i<diff;i++)
                        printf("5");
                    for(int i=0;i<a*5;i++)
                        printf("3");
                    printf("\n");

                }
            }

       }
return 0;
}
