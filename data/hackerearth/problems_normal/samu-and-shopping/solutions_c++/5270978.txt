#include<iostream>
using namespace std;

long cost;
int n;
int A[100009][3];
long B[100009][3];

int main(){
	int t, i, j;
	cin >> t;
	while (t--){
		cin >> n;
		for (i = 0; i < n; i++)
			for (j = 0; j < 3; j++){
				cin >> A[i][j];
			}
		for(i=0;i<n;i++){
			for(j=0;j<3;j++){
				if(i==0)
					B[i][j]=A[i][j];
				else{
					switch (j){
						case 0:
							B[i][j]=A[i][j]+min(B[i-1][1],B[i-1][2]);
							break;
						case 1:
							B[i][j]=A[i][j]+min(B[i-1][0],B[i-1][2]);
							break;
						case 2:
							B[i][j]=A[i][j]+min(B[i-1][0],B[i-1][1]);
							break;
					}
				}
			}
		}
		cost=min(B[n-1][0],B[n-1][1]);
		cost=min(cost,B[n-1][2]);
		cout << cost << endl;
	}
	return 0;
}
