#include <cstdio>
#include <iostream>
#include <algorithm>
#include <vector>
#include <cstring>
#include <string>

using namespace std;

vector<string> vec;

int main(){
	int N, k, j;
	scanf("%d", &N);
	for(int i=0;i<N;i++){
		string s;
		cin >> s;
		vec.push_back(s);
	}
	sort(vec.begin(), vec.begin() + N);
	long long ans = 1;
	ans = ans + vec[0].size();
	for(int i=1;i<N;i++){
		int l1 = vec[i-1].size();
		int l2 = vec[i].size();
		for(j=0,k=0; j<l1 and k<l2; ){
			if(vec[i-1][j]==vec[i][k])
				j++, k++;
			else
				break;
		}
		ans = ans + l2-k;
	}
	printf("%lld\n",ans);
	return 0;
}
