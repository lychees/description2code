#include <iostream>
using namespace std;

int main()
{
    int t,i,ctr[10];
    long a,b,rev,j,n;
    cin>>t;
    for(i=0;i<t;i++)
    {
        ctr[i]=0;
        cin>>a>>b;
        for(j=a;j<=b;j++)
        {
            rev=0;
            n=j;
            while(n)
            {
                rev=rev*10;
                rev=rev+n%10;
                n=n/10;
            }
            if(j==rev)
                ctr[i]++;
        }
    }
    for(i=0;i<t;i++)
        cout<<ctr[i]<<endl;
    return 0;
}
