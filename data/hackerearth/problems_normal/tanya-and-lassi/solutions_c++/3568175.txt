
#include<stdio.h>
#include<limits.h>

unsigned long max(unsigned long a, unsigned long b) { return (a > b)? a : b;}

unsigned long maxPrice(unsigned long price[], unsigned long n)
{
   unsigned long val[n+1];
   val[0] = 0;
   unsigned long i, j;
 

   for (i = 1; i<=n; i++)
   {
       unsigned long max_val = 0;
       for (j = 0; j < i; j++)
         max_val = max(max_val, price[j] + val[i-j-1]);
       val[i] = max_val;
   }
 
   return val[n];
}
 
/* Driver program to test above functions */
int main()
{
    unsigned long T;
    unsigned long L,arr[10000],i;
    scanf("%lu",&T);
    while (T!=0)
    {	
    	scanf("%lu",&L);
    	for(i=0;i<L;i++)
    	{
    		scanf("%lu",&arr[i]);
    	}
    	printf("%lu\n", maxPrice(arr, L));
    	getchar();
    	T--;
    }
}
