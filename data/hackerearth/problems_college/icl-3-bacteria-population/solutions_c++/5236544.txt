#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	int n,i,res=0;
	cin >> n;
	int arr[n+1][3];
	arr[0][0]=1;
	arr[0][1]=0;
	arr[0][2]=0;
	arr[1][0]=0;
	arr[1][1]=1;
	arr[1][2]=0;
	for(i=2;i<=n;i++)
	{
		arr[i][0]=arr[i-1][1]+arr[i-1][2];
		arr[i][1]=arr[i-1][0];
		arr[i][2]=arr[i-1][1];
	}
	res=res+arr[n][0]+arr[n][1]+arr[n][2];
	cout << res;
}
