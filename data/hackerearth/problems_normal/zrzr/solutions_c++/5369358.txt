#include<bits/stdc++.h>
using namespace std;
#define ll unsigned long long
int main(){
    int tc;cin>>tc;
    while(tc--){
        ll n;cin>>n;
        if(n<=4ULL)puts("0");
        else if(n>=5){
            ll ans=0;
            int  i=1;
            while(n/(ll)pow(5,i)>=1ULL ){
                ans+=(n/((ll)pow(5,i)));
                //n/=(ll)pow(5,i);
                i++;
            }
            cout<<ans<<"\n";
        }
    }
    return 0;
}
