#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string str1,str2;
		cin>>str1>>str2;
		int l1 = str1.length();
		int l2 = str2.length();
		int i,j=0;
		int flag = 0;
		for(i=0;i<l2;i++)
		{
			flag = 0;
			for(;j<l1;j++)
			{
				if(str2[i] == str1[j])
				{
					flag = 1;
					if(i == l2-1)
					{
						cout<<"YES"<<"\n";
					}
					j++;
					break;
				}
			}
			if(flag == 0)
			{
				cout<<"NO"<<"\n";
				break;
			}
		}
		

		
	}
}
