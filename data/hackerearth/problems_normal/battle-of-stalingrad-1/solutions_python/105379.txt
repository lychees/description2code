#!/usr/bin/env python

import math

def area (x1, y1, x2, y2, x3, y3):

	return math.fabs( ((x2-x1)*(y3-y1))- ( (x3-x1)*(y2-y1)) ) / 2 

tc = input()
for i in range(tc):
	number = raw_input().split()
	numbers = []
	for i in number:
		numbers.append(int(i))

	a = area(numbers[6], numbers[7], numbers[0], numbers[1] ,numbers[2] ,numbers[3])

	b = area(numbers[6], numbers[7], numbers[2], numbers[3] ,numbers[4] ,numbers[5])
	
	c = area(numbers[6], numbers[7], numbers[4], numbers[5] ,numbers[0] ,numbers[1])
	
	total = area( numbers[0], numbers[1] ,numbers[2] ,numbers[3],numbers[4] ,numbers[5])
	
	if a + b + c > total:
		print "OUTSIDE"
	else:
		print "INSIDE"
