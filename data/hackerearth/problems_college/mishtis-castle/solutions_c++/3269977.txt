#include<bits/stdc++.h>
using namespace std;
int n,a[105][105],c;
int dp[105][105];
int fun(int i,int j)
{
	
	if(i==n-1 && j==n-1)
	return a[i][j];
	else if(dp[i][j]!=-1)
	return dp[i][j];
	else
	return dp[i][j]=a[i][j]+max((j+1)<n?fun(i,j+1):0,(i+1)<n?fun(i+1,j):0);
}
int main()
{
	int t,i,j;
	cin>>t;
	while(t--)
	{
		cin>>n>>c;
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
			cin>>a[i][j];
		}
		memset(dp,-1,sizeof(dp));
		int ans=fun(0,0);
	//	cout<<ans;
		if(ans>=c)
		cout<<ans-c<<endl;
		else
		cout<<"-1"<<endl;
	}
	return 0;
}
