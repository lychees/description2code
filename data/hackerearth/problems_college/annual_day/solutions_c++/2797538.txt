#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--){
		int n,k;
		cin>>n>>k;
		int a[n];
		int b[n];
		for(int i=0;i<n;i++){
			cin>>a[i];
			b[i]=0;
		}int c=0;
		for(int i=0;i<n;i++){
			int no=a[i];
			while(no>0){
				if(no%2==1){b[i]=b[i]+1;}
				no=no/2;
			}
		}
		sort(b,b+n);
		for(int i=n-1,j=0;j<k;i--,j++){
			c=c+b[i];
		}
		cout<<c<<endl;
	}
   
    return 0;
}
