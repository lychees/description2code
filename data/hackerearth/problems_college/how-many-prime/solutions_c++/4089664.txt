#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;

long long isprime(long long n);

int main()
{
    long long t;
    cin>>t;
    while(t--)
    {
    	long long n,flag=0;
    	cin>>n;
    		for(long long i=(n-1)/2;i>=2;i--)
    		{
    			if(isprime(i)==1&&isprime(n-i)==1)
    			{
    				cout<<i<<" "<<n-i<<endl;
    				flag = 1;
    				break;
    			}
    		}
    		if(flag==0)
    			cout<<"-1 -1"<<endl;
    }
}

long long isprime(long long n)
{
	long long i=1, count=1;
	for(long long i=2;i<=sqrt(n);i++)
	{
		if(n%i==0)
			count = 0;
	}
	return count;
}
