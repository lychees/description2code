#include<iostream>
#include<algorithm>
#include <map>
#include <cstring>
using namespace std;

#define ll long long int
map<long long,long long> m;

int main()
{
    int t,i;
    cin>>t;
    int n;
    for(i=0;i<t;i++)
    {
        cin>>n;
        int j,t1,t2;
        m.clear();
        for(j=0;j<n;j++)
        {
            cin>>t1>>t2;
            m[t1]++;
            m[t2]--;
        }
        map<long long ,long long>::iterator it;
        ll k=0,maxtime=0;
        for(it=m.begin();it!=m.end();it++)
        {
            it->second=it->second+k;
            k=it->second;
            if(maxtime<k)
                maxtime=k;
            //cout<<it->first<<" "<<k<<endl;
        }
        cout<<maxtime<<endl;
    }
}
