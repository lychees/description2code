#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s;
    getline(cin, s);
    int l = s.size();
    int a[128] = {0};
    for(int i = 0; i < l; i++)
    {
    	a[(int)s[i]]++;
    }
    int mx = -1;
    char c = 150;
    for(int i = 0; i < 128; i++)
    {
    	if(mx < a[i])
    	{
    		mx = a[i];
    		c = (char)i;
    	}
    }
    cout<<c<<" "<<mx<<endl;
    return 0;
}
