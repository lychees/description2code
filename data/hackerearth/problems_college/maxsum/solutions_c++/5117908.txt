#include <bits/stdc++.h>
#define floop(x, p, n)	for(int x = p; x<n; ++x)

using namespace std;

typedef unsigned long long ull;
typedef unsigned long ul;
typedef unsigned int ui;
typedef vector<int> vi;
typedef vector<unsigned int> vui;
typedef vector<unsigned long> vul;
typedef vector<unsigned long long> vull;
typedef vector<char> vc;

ull max(ull a, ull b){
	return a<b ? b:a;
}

int main(){
    
    ull T, N;
    cin >> T;
    
    vull a(10000), choosen(10000), notChoosen(10000);

    floop(i, 0, 10000){
    	choosen[i] = 0;
    	notChoosen[i] = 0;
    }
    
    while(T--){
    	cin >> N;
    	
    	floop(i, 0, N){
    		cin >> a[i];
    	}
    	
    	choosen[0] = a[0];
    	notChoosen[0] = 0;

    	floop(i, 1, N){
    		choosen[i] = notChoosen[i-1]+a[i];
    		notChoosen[i] = max(choosen[i-1], notChoosen[i-1]);
    	}
    	
    	cout << max(choosen[N-1], notChoosen[N-1]) << endl;
    }
    
    return 0;
}
