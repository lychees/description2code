#include <bits/stdc++.h>
using namespace std;
#define MOD 1000000007
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		long long int a[n];
		for(int i=0;i<n;i++)
		{
			scanf("%lld",&a[i]);
		}
		sort(a,a+n);
		long long int ans=1;
		for(int i=0;i<n;i++)
		{
			ans=ans*(a[i]-i);
			if(ans>=MOD)ans=ans%MOD;
		}
		if(ans<0)ans=0;
		printf("%lld\n",ans);
	}
    return 0;
}
