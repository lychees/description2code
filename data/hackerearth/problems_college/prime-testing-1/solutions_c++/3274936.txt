#include <iostream>
using namespace std;

int a[1001]={0};

void sieve()
{
	a[0]=a[1]=1;
	for(int i=2;i<=1000;i++)
	{
		if(a[i]==0)
		{
			for(int j=i*2;j<=1000;j+=i)
			a[j]=1;
		}
	}
}


int main()
{
	ios_base::sync_with_stdio(false);
   int t;
   cin>>t;
   sieve();
   while(t--)
   {
   	 int n,c=0;
   	 cin>>n;
   	 while(n--)
   	 {
   	 	int x;
   	 	cin>>x;
   	 	if(a[x]==0) c++;
   	 }
   	 cout<<c<<endl;
   }
    return 0;
}
