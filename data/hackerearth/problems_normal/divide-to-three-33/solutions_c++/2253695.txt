#include <bits/stdc++.h>

using namespace std;

vector<int> temp;
int arr[100000];
int main()
{
   int n;
   scanf("%d",&n);
   
   for(int i=0;i<n;i++)
   scanf("%d",&arr[i]);
   int mini=INT_MAX;
   for(int i=0;i<(1<<n);i++)
   {
   	int sum=0;
   	  for(int j=0;j<n;j++)
   	  {
   	  	  if((1<<j)&i){
   	  	  	sum+=arr[j];
   	  	  }
   	  }
   	temp.clear();
   	  
   	  if(sum<=mini){
   	
   	  	for(int j=0;j<n;j++){
   	  		if(!((1<<j)&i)) {
   	  		  temp.push_back(arr[j]);
   	  		}
   	  	}
   	  	
   	  	for(int j=0;j<(1<<temp.size());j++)
   	  	{
   	  		int sum1=0,sum2=0;
   	  		for(int k=0;k<temp.size();k++){
   	  			if((1<<k)&j) {
   	  				sum1+=temp[k];
   	  			}
   	  			else
   	  			sum2+=temp[k];
   	  		}
   	  		if(sum>=(max(sum1,sum2))){
   	  		
   	  			  mini=sum;
   	  		}
   	  	}
   	  }
   }
   printf("%d",mini);
}
