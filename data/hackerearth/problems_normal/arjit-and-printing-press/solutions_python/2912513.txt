noOfTestCases = int(raw_input())

def lexSmallestString(string, reserve):
	
	reserve = ''.join(sorted(reserve))
	currentStart = 0
	reserveStart = 0
	newString = ''
	
	while currentStart < len(string) and reserveStart < len(reserve):
		if(reserve[reserveStart] < string[currentStart]):
			newString += reserve[reserveStart]
			reserveStart += 1
		else:
			newString += string[currentStart]		
		
		currentStart += 1
		
	while currentStart < len(string):
		newString += string[currentStart]
		currentStart += 1
		
	return newString	

for i in range(1, noOfTestCases + 1):
	print lexSmallestString(raw_input(), raw_input())
