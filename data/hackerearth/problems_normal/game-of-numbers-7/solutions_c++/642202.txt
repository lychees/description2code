#include<bits/stdc++.h>

using namespace std;

#define gc getchar//_unlocked

int read_int() {
	long long int num;
  scanf("%lld",&num);
  return num;
}

int arr[1000006];

int main()
{
	long long int t,x,y;
	t=read_int();

	while(t--)
	{
		x=read_int();
		y=read_int();

		long long int ans=0;
		memset(arr,0,sizeof(arr));
		
		for(long long int i=2;i*i<=y;i++)
		{
			long long int val=i*i;

			long long int base=(val-(x)%val)%val;
			base+=x;
			
			for(long long int k=base;k<=y;k+=val)
			arr[(int)(k-x)]=1;

		}

/*		printf("\nPrinting array : ");
		
		for(int i=0;i<y-x+1;i++)
		printf("%d ",arr[i]);
		printf("\n");*/
		
		for(int i=0;i<y-x+1;i++)
		if(!arr[i])
		ans++;

		cout<<ans<<endl;
		
	}
	
	return 0;
}
