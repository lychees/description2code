#include <bits/stdc++.h>
#define mod (int)((1e9)+7)
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    while(t--)
    {
    	int n;
    	cin>>n;
    	
    	string str;
    	cin>>str;
    	long long ans=0;
    	for(int i=0; i<n; i++)
    	 ans=(ans*10+str[i]-'0')%mod;
    	cout<<ans<<"\n";
    }
    return 0;
}
