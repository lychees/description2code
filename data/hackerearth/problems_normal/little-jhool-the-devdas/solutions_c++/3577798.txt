#include<climits>
#include <iostream>
using namespace std;

int main()
{
    long long int sum = 0,n,r,mini = 1000000,b = 0;
    cin>>n;
    while(n--)
    {
    	cin>>r;
    	sum = sum + r;
  		if(r%2!=0)
  		{
    		mini = min(mini,r);
    		b++;
  		}
    	
    }
    if(b==0)
    	cout<<":("<<endl;
    else
    {
    	if(b%2==0)
			sum = sum - mini;
		cout<<sum<<endl;
    }
    return 0;
}
