#include <iostream>
using namespace std;

long long a[1000007],dp[1000007];
int main(){
	long long n,q;
	cin>>n>>q;
	a[0]=0; dp[0]=0;
	for(int i=1;i<=n;i++){
		cin>>a[i];
		dp[i]=a[i]+dp[i-1];
	}
	long long x,y;
	for(int i=0;i<q;i++){
		cin>>x>>y;
		cout<<dp[y]-dp[x-1]<<" ";
		cout<<a[y]-(dp[y-1]-dp[x-1])<<"\n";
	}
}
