#include <iostream>
using namespace std;

int main() {
	int t;
	cin>>t;
	while(t--)
	{
	    string s;
	    int a[124];
	    for(int i=0;i<124;i++)
	    {
	        a[i]=0;
	    }
	    cin>>s;
	    for(int i=0;i<s.length();i++)
	    {
	        a[s[i]]++;
	    }
	    for(int i=0;i<124;i++)
	    {
	        char c;
	        for(int j=0;j<a[i];j++)
	        {
	            c=i;
	            cout<<c;
	        }
	    }
	    cout<<endl;
	}
	return 0;
}
