unsigned long long gcd(unsigned long long u, unsigned long long v)
{
    if (u == v)
        return u;
 
    if (u == 0)
        return v;
 
    if (v == 0)
        return u;
 
    
    if (~u & 1) 
    {
        if (v & 1)
            return gcd(u >> 1, v);
        else 
            return gcd(u >> 1, v >> 1) << 1;
    }
 
    if (~v & 1) 
        return gcd(u, v >> 1);
 
    if (u > v)
        return gcd((u - v) >> 1, v);
 
    return gcd((v - u) >> 1, u);
}
int main()
{
 int t;
 unsigned long long x,y,g;
 scanf("%d",&t);
 while(t--)
 {
     
     scanf("%llu%llu",&x,&y);
     g=gcd(x,y);
     if(g==1)
     {
         printf("Is a Co-Prime\n");
     }
     else
        printf("Not a Co-Prime\n");
 }
    return 0;
}
