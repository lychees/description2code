#include<cstdio>
#include<iostream>
#include<limits.h>


int main()
{
	int N;
	scanf("%d",&N);
	int a[N][N],ans[N];
	for(int i=0;i<N;i++)
		for(int j=0;j<N;j++)
			scanf("%d",&a[i][j]);
	for(int i=0;i<N;i++)
		ans[i]=a[i][0];
	for(int j=1;j<N;j++)
	{
		ans[0]+=a[0][j];
		for(int i=1;i<N;i++)
			ans[i]=std::min(ans[i-1]+a[i][j],ans[i]+a[i][j]);
		for(int i=N-2;i>=0;i--)
			ans[i]=std::min(ans[i],ans[i+1]+a[i][j]);
	}
	int min=INT_MAX;
	for(int i=0;i<N;i++)
		min=std::min(min,ans[i]);
	printf("%d",min);
	return 0;
}
