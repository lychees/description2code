'''
# Read input from stdin and provide input before running code

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
n = int(raw_input())
a = list(map(int, raw_input().split()))

for i in xrange(n-1):
    if a[i] > a[i+1]:
        a[i] = a[i] - a[i+1]
        a[i+1] = 0
    else:
        a[i+1] = a[i+1] - a[i]
        a[i] = 0

print 'YES' if all([a[i] == 0 for i in xrange(n)]) else 'NO'
