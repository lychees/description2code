#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int main()
{
    int test_cases,pos,i,count,l,fpos,x,m;
    cin>>test_cases;
    getchar();
    while(test_cases>0)
    {
        char str[101];
        gets(str);
        l=strlen(str);
        //printf("%d ",l);
        count=0;m=0;
        for(i=0;i<l;i++)
    	{
            if(str[i]>=97&&str[i]<=122)
            	   m++;
            if(str[i]=='*')
            {
            	pos=i;
            	break;
            }
        }
        for(i=pos+1;i<l;i++)
        {
        	if(str[i]>=97&&str[i]<=122||str[i]=='*')
               count++;
        }
        x=count;
        if(pos>count)
        {
           
           for(i=0;i<=pos;i++)
           {
           	   if(count==-1)
                  break;
           	   if(str[i]>=97&&str[i]<=122)
           	   {
           	   	  fpos=i;
           	   	  count--;
           	   }
           }
        }
        else
        {
        	for(i=l-1;i>=0;i--)
        	{
        		if(m==-1)
                  break;
        		if(str[i]>=97&&str[i]<=122)
        		{
        		   fpos=i;
        		   m--;
        		}   
        	}
        }
        //printf("%d %d %d\n",pos,x,fpos);
        printf("%c\n",str[fpos]);
        test_cases--;
    }
    return 0;
}
