#include<stdio.h>
#include<iostream>
#include<vector>
using namespace std;
long long arr[1000004];
vector <long long> xo[1000004];
long long result[1000004];
int main()
{
    long long i,n,j;
    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>arr[i];
    }
    for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++)
        {
            if(j!=i)
                xo[i].push_back(arr[i]^arr[j]);
        }
    }
    for(i=0;i<n;i++)
    {
        result[i]=xo[i][0];
        for(j=1;j<n-1;j++)
        {
            result[i]=result[i]|xo[i][j];
        }
    }
    long long res=result[0];
    for(i=1;i<n;i++)
    {
        res=res|result[i];
    }
    cout<<res<<endl;
    return 0;
}
