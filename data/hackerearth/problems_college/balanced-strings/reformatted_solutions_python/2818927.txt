
a = ['a', 'b', 'c', 'd', 'e', 'f', 'g', 'h', 'i', 'j', 'k', 'l', 'm',
     'n', 'o', 'p', 'q', 'r', 's', 't', 'u', 'v', 'w', 'x', 'y', 'z']
for i in range(int(input())):
    s = raw_input()
    s = sorted(s)
    "".join(s)
    f = 0
    for i in s:
        for j in a:
            if i == j:
                f += 1
    if f % 26 == 0:
        print("Yes")
    else:
        print("No")
