#include <iostream>
using namespace std;

int main()
{
	int number,no_of_rows,no_of_columns,counter,test_cases,col_start,col_end,row_start,row_end,price,row_travel,col_travel;
    cin>>test_cases;
    for(counter=0;counter<test_cases;counter++)
    {
    	cin>>no_of_rows>>no_of_columns;
    	cin>>row_start>>col_start;
    	cin>>row_end>>col_end;
    	cin>>price;
    	if(abs(row_start-row_end)<abs(no_of_rows-abs(row_start-row_end)))
    	{row_travel=abs(row_start-row_end);
    	}
    	else
    	{row_travel=abs(no_of_rows-abs(row_start-row_end));
    	}
    	if(abs(col_start-col_end)<abs(no_of_columns-abs(col_start-col_end)))
    	{col_travel=abs(col_start-col_end);
    	}
    	else
    	{col_travel=abs(no_of_columns-abs(col_start-col_end));
    	}
    	cout<<((col_travel+row_travel)*price)<<endl;
    	if(((col_travel+row_travel)*price)>1000)
    	{
    		cout<<"Let go of the group."<<endl;
    	}
    	else
    	cout<<"You saved the group."<<endl;	
    }
    return 0;
}
