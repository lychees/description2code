#include <iostream>
#include <math.h>
using namespace std;

int main()
{
    long int t;
    long long int a,b,c,d,e;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b>>c;
    	d=pow(a,b);
    	e=d%c;
    	if(e==0)
    	cout<<d<<endl;
    	else if(e>c/2)
    	cout<<d+(c-e)<<endl;
    	else
    	cout<<d-e<<endl;
    }
    return 0;
}
