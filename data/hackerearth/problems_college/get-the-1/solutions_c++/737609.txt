#include<iostream>
using namespace std;

int main()
{
	int i,b,d,c;
	cin>>b;
	for(i=0;i<b;i++)
	{
		
		cin>>c;
		for(c;c>9;)
		{
			d=0;
		    do
		    {
			 d=d+(c%10)*(c%10);
			 c=c/10;
		    }while(c!=0);
		    c=d;
		}
		if(c==1)
		cout<<"Case #"<<i+1<<":YES\n";
		else
		cout<<"Case #"<<i+1<<":NO\n";
	}
return 0;	
}
