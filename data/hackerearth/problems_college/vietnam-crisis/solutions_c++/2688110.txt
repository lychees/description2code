#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define gc getchar_unlocked

void scanint(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}
int arr[20000000];
int main()
{
	
    int n;
    while(true)
    {
        scanf("%d",&n);
        if(n==0)
            break;
        for(int i=0;i<n;++i)
            scanint(arr[i]);
        sort(arr,arr+n);
        for(int i=0;i<n;++i)
            printf("%d ",arr[i]);
        printf("\n");
    }
return 0;
}
