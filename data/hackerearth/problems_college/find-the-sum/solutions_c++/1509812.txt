#include <iostream>
#define LL long long
using namespace std;

LL sumless(LL n)
{
	if(n==1) return 0;
	LL ans=0,start=1,curr=1;
	while(start<n){
		if(curr&1) ans+=curr;
		curr++;
		start+=curr;
	}
	LL diff=n-(start-curr)-1;
	if(diff&1) ans+=curr;
	return ans;
}
LL sumeq(LL n)
{
	LL ans=0,start=1,curr=1;
	while(start<=n){
		if(curr&1) ans+=curr;
		curr++;
		start+=curr;
	}
	LL diff=n-(start-curr);
	if(diff&1) ans+=curr;
	return ans;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	LL l,r;
    	scanf("%lld %lld",&l,&r);
    	LL ans=sumeq(r)-sumless(l);
    	printf("%lld\n",ans);
    }
    return 0;
}
