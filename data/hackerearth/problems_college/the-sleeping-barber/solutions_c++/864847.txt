#include <iostream>
#include <stdio.h>
#include <vector>
#include <algorithm>

using namespace std;

typedef long long int Lint;

int main()
{
	
	Lint d, g, m, i, sum;
	Lint ct;
	
	scanf("%lld", &d );
	
	while( d-- ) {
		
		scanf("%lld %lld", &g, &m );
		vector<Lint> A( g );
		
		for( i = 0;i < g;i++ )
			scanf("%lld", &A[ i ] );
		
		sort( A.begin(), A.end() );
		
		for( i = 0, sum = 0, ct = 0;i < g;i++ ) {
			sum += A[ i ];
			if( sum <= m )
				ct++;
		}
		
		printf("%lld\n", ct );
	}
	
	return 0;
}
