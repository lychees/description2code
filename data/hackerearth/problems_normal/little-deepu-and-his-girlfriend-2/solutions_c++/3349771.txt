#include <bits/stdc++.h>
using namespace std;

vector<int>nums;

int main()
{
	int t, m, n, aux;
	for(cin>>t; t; --t){
		cin>>m>>n;
		nums.assign(n,0);
		vector<int>win(m+1, 0);
		for( int i=0; i<n; i++){
			cin>>nums[i];
			win[ nums[i] ] = 1;
		}
		
		for( int i=1; i<=m; i++){
			if( win[i]==0 ){
				for( int j=0; j<n; j++){
					if( i-nums[j]>=0 and win[ i-nums[j] ]==0){
						win[i]=1;
						break;
					}
				}
			}
		}
		
		if( win[ m] ){
			cout<<"Little Deepu\n";
		} else {
			cout<<"Kate\n";
		}
	}
    return 0;
}
