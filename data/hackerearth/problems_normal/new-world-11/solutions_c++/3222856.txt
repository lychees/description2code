#include <bits/stdc++.h>
using namespace std;
int t,n,k,a[100005];
int main(){
	scanf("%d",&t);
	while(t--){
		scanf("%d %d",&n,&k);
		for(int x=0;x<n;x++) scanf("%d",&a[x]);
		int s=1,e=1000000000;
		while(s<e){
			int m=(s+e)/2,c=a[0],fail=1;
			for(int y=0;y<k;y++){
				int i=upper_bound(a,a+n,c+m)-a-1;
				if(i==n-1){
					fail=0;
					break;
				}
				c=a[i];
			}
			if(fail) s=m+1;
			else e=m;
		}
		printf("%d\n",s);
	}
	return 0;
}
