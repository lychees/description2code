#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
    int x,y,q,n;
    long a[1000002]={1};
   
    scanf("%d%d%d",&x,&y,&q);
    for(int i=1;i<=1000000;i++)
        {a[i]=((a[i-1]*x)-y)%1000000007;}
    while(q--)
    {
        scanf("%d",&n);
        
        printf("%d\n",a[n]);
    }
}
