#include <iostream>
#include <algorithm>
#include <array>
#include <string.h>

using namespace std;

int main()
{
    int count = 0;
    int i;
    int arr[26];
    cin >> count;
    for (i = 0; i < count; i++){
    	for (int j = 0; j < 26; j++)
    		arr[j] = 0;
    	char str[100];
    	cin >> str;
    	for(int j = 0; j < strlen(str); j++){
    		if(arr[str[j] - 'a'] == 0){
    			arr[str[j] - 'a'] = 1;
    		}
    	}
    	int flag = 1;
    	for(int j = 0; j < 26; j++) {
    		if(arr[j] == 0){
    			flag = 0;
    			break;
    		}
    	}
    	if (flag == 0){
    		cout << "NO" << endl;
    	}
    	else {
    		cout << "YES" << endl;
    	}
    	
    	
    }
    return 0;
}
