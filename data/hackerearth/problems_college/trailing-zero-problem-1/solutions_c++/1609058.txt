#include<iostream>
#include<cmath>
using namespace std;   
int fun(int  n)
{
    // Initialize result
    int count = 0;

    // Keep dividing n by powers of 5 and update count
    for (int i=5; n/i>=1; i *= 5)
          count += n/i;

    return count;
}
int main()
{
int t;
cin>>t;
double n;
double pi = 2*acos(0);
long long int ans;

while(t--)
{
        cin>>n;
        if(n==0.00||n==1.00)
        ans=1;

        else{
        ans=(((log(2.00*pi*n)/2.00)+(n*(log(n)-1.00)))/log(10.0))+1;
        }
        
     //  while(ans)
        
        cout<<ans-fun(n)<<endl;
}
return 0;
}
