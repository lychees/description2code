#include <iostream>
#include<vector>
using namespace std;

typedef long long natural;

int main()
{
    natural length;
    cin>>length;
    vector<char> string;
    for(natural i=0;i<length;i++){
    	char value;
    	cin>>value;
    	string.push_back(value);
    }
    natural t;
    cin>>t;
	 natural maps[26];
	 for(int i=0;i<26;i++)
		 maps[i]=0;
     for(natural i=0;i<length;i++){
      	maps[string[i]-'0'-49]+=1;
	  }
    while(t--){
       char firstChar,secondChar;
       cin>>firstChar>>secondChar;
       natural subStrings=0;
       if(firstChar==secondChar){
       	     subStrings=(maps[firstChar-'0'-49]*(maps[firstChar-'0'-49]-1))/2;
       }
       else{
       	  subStrings=maps[firstChar-'0'-49] * maps[secondChar-'0'-49];
       }
       cout<<subStrings<<endl;
    }
   return 0;
}
