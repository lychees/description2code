#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int n,k;
    cin>>n>>k;
    long int c[n],s=1000000,sum=0,i;
    for(i=0;i<n;i++)
    {
    	cin>>c[i];
    	sum+=c[i];	
    }
    sort(c,c+n);
    if(n==k)
    {
    	cout<<sum<<endl;
    }
    else
    {
        int x=0,sum=0;
         loop:
         x++;
        for(i=n-1;i>n-k-1&&i>=0;i--)
        {
        	sum+=x*c[i];
        }
        n=n-k;
        if(n>0)
        goto loop;
        else
        cout<<sum<<endl;
    }
    return 0;
}
