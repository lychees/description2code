#include <iostream>
using namespace std;

double dp[1010][1010];
double getans(int x, int y, int n, int w, double p1, double p2) {
	
	
	if(w <= 0) return 1.00;
	if(n == 0) return 0;
	
	if(dp[n][w] != -1) return dp[n][w];

	
	double prob1 = p1*getans(x,y,n-1,w-x,p1,p2) + (1-p1)*getans(x,y,n-1,w,p1,p2);
	

	double prob2 = p2*getans(x,y,n-1,w-y,p1,p2) + (1-p2)*getans(x,y,n-1,w,p1,p2);
	
	dp[n][w] = max(prob1,prob2);
	return dp[n][w];
	
}
int main()
{
    int t;
    cin >> t;
    
    for(int i =0; i < t; i++) {
    	int x,y,n,w,p1,p2;
    	
    	cin >> x >> y >> n >> w >> p1 >> p2;
    	for(int j = 0; j <= n; j++){
    		for(int k = 0; k <= w; k++){
    			dp[j][k] = -1;
    		}
    	}
    	double p1d = p1/(double)100;
    	double p2d = p2/(double)100;
    	double ans = getans(x,y,n,w,p1d,p2d) * 100;
    	printf("%0.6lf\n",ans);
    }
    return 0;
}
