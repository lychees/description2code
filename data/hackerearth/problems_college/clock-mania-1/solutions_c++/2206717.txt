#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    int t,h,m;
    cin>>t;
    while(t--)
    {
    	cin>>h>>m;
 
    	if (h == 12)
    	h = 0;
    	if (m == 60)
    	m = 0;
 
  
     int hangle = 0.5 * (h*60 + m);
     int mangle = 6*m;
 
    int angle = abs(hangle-mangle);
    angle = min(360-angle, angle);
 
    cout<<angle<<"\n";
	}
    return 0;
}
