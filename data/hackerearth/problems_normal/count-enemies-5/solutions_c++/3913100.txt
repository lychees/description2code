#include<cstdio>
#include<iostream>
#include<algorithm>
#include<string>
#include<stack>

#define all(c) (c).begin(),(c).end()
#define ll long long 

using namespace  std;
stack<char> St;

int main()
{
	int t;
	cin>>t;
	string s;
	ll cnt,l,i;
	while(t--)
	{
		cin>>s;
		l = s.length();
		cnt = 0;
        for(i = 0;i<l;i++)
        {
           //if(s[i] == 'O') cnt++;

           if(!St.empty())
           {
             if(s[i] == '*')
             {
               while(!St.empty())
               {
               	 if(St.top() == 'O') cnt++;
               	 St.pop();
               }
             }
             else if(s[i] == 'X')
             {
             	if(St.top() == 'O')
             	{
             		while(!St.empty()) St.pop();
             		St.push(s[i]);
             	}
             }
             else
             	if(St.top() == 'O') St.push(s[i]);
           }
           else
           {
           	  if(s[i] != '*')
           	  	St.push(s[i]);
           }
        }
        while(!St.empty())
        {
        	if(St.top() == 'O') cnt++;
        	St.pop();
        }
        cout<<cnt<<"\n";
	}
	return 0;
}
