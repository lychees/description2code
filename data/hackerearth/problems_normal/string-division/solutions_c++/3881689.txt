#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string str;
		cin>>str;
		if(str.size()>=10)
		{
			cout<<"YES"<<endl;
			continue;
		}
		bool ok=0;
		for(int i=1;i<str.size();i++)
		{
			for(int j=i+1;j<str.size();j++)
			{
				for(int k=j+1;k<str.size();k++)
				{
					string s1 = str.substr(0,i);
					string s2 = str.substr(i, j - i);
					string s3 = str.substr(j, k - j);
					string s4 = str	.substr(k, str.size() - k);
					if (s1 != s2 && s1 != s3 && s1 != s4 && s2 != s3 && s2 != s4 && s3 != s4)
					{
						ok = 1;
					}
				}
			}
		}
		if(ok)
			cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;
	}
}
