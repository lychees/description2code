#include <iostream>
using namespace std;

int main()
{
 	
 	int t;
 	cin>>t;
 	while(t>0)
 	{
 		int n;
 		cin>>n;
 		int arr[10]={0};
 		int temp;
 		for(int i=0;i<n;i++)
 		{
 			cin>>temp;
 			arr[temp%10]++;
 		}
 		int sum=0;
 		for(int i=0;i<10;i++)
 		{
 			if(arr[i]>1)
 				sum+=arr[i]-1;
 		}
 		cout<<sum<<endl;
 		t--;
 	}
}
