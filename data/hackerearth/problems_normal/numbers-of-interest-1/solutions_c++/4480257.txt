#include <stdio.h>
#include <math.h>
#include <algorithm>
#include <iostream>
using namespace std;
int main()
{
	long long int t;
	scanf("%d",&t);
	while(t--)
	{
	long long int i,x,y,n,j,c;	
	scanf("%lld %lld %lld",&x,&y,&n);
	long long int a[50]={0};
	for(i=0;i<x;i++)
	{
		a[i]=y;
	}
	for(i=x;i<n;i++)
	{
		c=0;
		for(j=i-1;;j--)
		{
			c++;
			a[i]=a[i]+a[j];
			if(c==x)
				break;
		}
	}
	printf("%lld\n",a[n-1]);
	}
	return 0;
}
