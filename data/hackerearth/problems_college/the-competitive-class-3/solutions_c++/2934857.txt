#include <iostream>
using namespace std;

int main()
{
   int n;
   cin>>n;
   int a[n],pos,i;
   pos=1;
   for(i=0;i<n;i++)
   {
   	cin>>a[i];
   } 
  
    for(int j=0;j<n;j++)
  {
  	  int k=a[j];
    	pos=1;
  	 for(i=0;i<n;i++)
  	 {
  	 	if(a[i]>k)
  	 	pos++;
  	  }
  cout<<pos<<" ";
  }
    return 0;
}
