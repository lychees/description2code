#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long n,a,i,j=0,b,k;
	vector<int> v,v1;
	cin>>n;
	
	for(i=0;i<n;i++)
	{
		cin>>a;
		v.push_back(a);
	}
	
	b=*max_element(v.begin(),v.end());
	
	for(i=0;i<=b;i++)
	{
		v1.push_back(0);
	}
	
	for(i=0;i<n;i++)
	{
		v1[v[i]]++;
	}
	
	for(i=0;i<=b;i++)
	{
		if(v1[i]==1)
		{
			j=1;
			k=i;
		}
	}
	if(j==0)
	{
		cout<<"-1";
	}
	else
	{
		cout<<k;
		
	}
}
