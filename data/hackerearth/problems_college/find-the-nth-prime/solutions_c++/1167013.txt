#include <iostream>
using namespace std;
int chk(int number)
{	// Given:   num an integer > 1
	// Returns: true if num is prime
	// 			false otherwise.
	
	int i;
	
	for (i=2; i<number; i++)
	{
		if (number % i == 0)
		{
			return 0;
		}
	}
	
	return 1;	
}

int main()
{
    int i=2,k=0,n;
    cin>>n;
    while(1)
    {
 
    	if(chk(i)){k+=1;}
    	if(k==n)break;
    	i+=1;
    }cout<<i<<endl;
    return 0;
}
