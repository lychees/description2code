#include <iostream>
using namespace std;

int matrix[102][102]={0};

int main()
{
    int T, N,C;
    int maxSum=0;
    cin>>T;
    for( int i=0; i<T;i++){
    	cin>>N>>C;
    	for(int j=1;j<=N;j++){
    		for(int k=1;k<=N;k++){
    			cin>>matrix[j][k];
    		}
    	}
		
    	for( int l=1; l<= N;l++){
    		for(int m=1;m<=N;m++){
    			
    			if(matrix[l-1][m] > matrix[l][m-1]){
    				matrix[l][m]=matrix[l][m]+matrix[l-1][m];
    			}
    			else{
    				matrix[l][m]=matrix[l][m]+matrix[l][m-1];
    			}
    		}
    			
    		}
    		(matrix[N][N] >= C)&&cout<<matrix[N][N] - C<<endl;
    		(matrix[N][N] < C)&&cout<<-1<<endl;
    	}
    
    return 0;
}
