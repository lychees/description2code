#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    long long a[3],s;
    while(t--)
{
	cin>>a[0]>>a[1]>>a[2];
	sort(a,a+3);
	
	s=a[0]%1000000007;
	s=(s*(a[1]-1)%1000000007)%1000000007;
	s=(s*(a[2]-2)%1000000007)%1000000007;
	
	cout<<s<<endl;
	
}
	return 0;
}
