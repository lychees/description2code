from collections import OrderedDict as od

for i in range(input()):
    word = raw_input()
    temp = od.fromkeys(word).keys()
    # print temp
    lt = len(word)
    pt = len(temp)
    # print lt
    if lt % 3 != 0 or pt != 3:
        print "Not OK"
    else:
        m1 = lt/3
        # print m1
        new_name = temp[0]*m1 + temp[1]*m1 + temp[2]*m1
        # print new_name
        if new_name == word:
            print "OK"
        else:
            print "Not OK"
