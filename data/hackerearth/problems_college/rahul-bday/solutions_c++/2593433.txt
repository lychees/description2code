#include <bits/stdc++.h>
#define ll long long int
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
       ll b,w;
       cin>>b>>w;
       ll x,y,z;
       cin>>x>>y>>z;
         if(max(x,max(y,z))==z)
            {
                 cout<<b*x+w*y<<endl;
            }
            else if(min(x,y)==x)
             {
                 ll r=(b+w)*x+w*z;

                    cout<<min(b*x+w*y,r)<<endl;

             }
             else
             {
                ll r=(b+w)*y+b*z;
                cout<<min(b*x+w*y,r)<<endl;
             }
    }
    return 0;
}
