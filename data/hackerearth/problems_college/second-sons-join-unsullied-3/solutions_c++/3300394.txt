#include<bits/stdc++.h>
#include<string>
using namespace std;
 
string s;
 
int main() {
    int n,i,t;
    cin >> t;
    while(t--) {
        cin >> s;
    	int r1=0,u=0,r2=0,v=0;
    	n=s.length();
    	for(i=0;i<n;i++) {
    	     if(s[i]=='U')
    	      u++;
    	     else
    	      v++;
    	}
    	if(abs(u-v)>1) {
    	  cout << -1 << endl;
    	  continue;
    	}
 
    	if(u>v)  {
    	for(i=0;i<n;i=i+2) {
    	    if(s[i]=='S')
    	      r1++;
    	}
    	 	cout << r1 << endl;
    	}
        else if(u<v) {
    	for(i=0;i<n;i=i+2) {
    	    if(s[i]=='U')
    	      r1++;
    	}
    		cout << r1 << endl;}
    	else {
    	  for(i=0;i<n;i=i+2) {
    	     if(s[i]=='U')
    	       r1++;
    	     if(s[i]=='S')
    	       r2++;}
    	   cout << min(r1,r2) << endl;
    	     
    	}
 
    
 
    }
	return 0;
}
