#include <bits/stdc++.h>
using namespace std;

int lengthOfLongestSubstring(char *s)
{

    int maxi = 0;
    int counti = 0;
    int a[26];
    memset(a,-1,sizeof(a));

    for(int i=0;i<strlen(s);++i)
    {

        if(a[s[i]-'A'] >= 0)
        {
            int from = a[s[i] - 'A'];
            counti = i - a[s[i] - 'A'];
            memset(a,-1,sizeof(a));

            for(int j=from+1;j<=i;++j)
                a[s[j]-'A'] = j;
        }
        else {
            counti++;
            a[s[i]-'A'] = i;
        }

        maxi = max(maxi,counti);
    }

    return maxi;
 }
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        char s1[100005];
        cin>>s1;
        cout<<lengthOfLongestSubstring(s1)<<endl;
    }
    return 0;
}
