#include<iostream>
using namespace std;
int main()
{
	bool p[1000001];
	long int a,b,i,j,sum;
	int t;
	p[0]=p[1]=false;
	for(i=2;i<1000001;i++)
		p[i]=true;
	for(i=2;i<1000001;i++)
	{
		if(p[i]==true)
		{
			for(j=2*i;j<1000001;j+=i)
				p[j]=false;
		}
	}
	cin>>t;
	while(t--)
	{
		sum=0;
		cin>>a>>b;
		for(i=a;i<=b;i++)
		{
			if(p[i]==true)
				sum+=i;
		}
		cout<<sum<<endl;
	}
}
