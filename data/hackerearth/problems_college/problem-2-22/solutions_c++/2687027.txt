#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,i;
    cin>>t;
    while(t--)
    {
        int r,cx,cy;
        cin>>r>>cx>>cy;
        int x[4];
        int y[4];
        int flag=0;
        for(i=0; i<4; i++)
        {
            cin>>x[i]>>y[i];

            if(pow((x[i]-cx),2)+pow((y[i]-cy),2)<=pow(r,2))
            {
                flag++;
            }
        }
        if(flag==4)
            cout<<"Yes"<<endl;
        else
            cout<<"No"<<endl;

    }
}
