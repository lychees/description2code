#include<bits/stdc++.h>

using namespace std;
int dash(char a)
{
	
	if(a=='0')return 6;
	if(a=='1')return 2;
	if(a=='2')return 5;
	if(a=='3')return 5;
	if(a=='4')return 4;
	if(a=='5')return 5;
	if(a=='6')return 6;
	if(a=='7')return 3;
	if(a=='8')return 7;
	if(a=='9')return 6;
}
int main()
{
	string s;
	cin>>s;
	int l=s.length();
	int ans=0;
	for(int i=0;i<l;i++)
	{
		ans+=dash(s[i]);
	}
	cout<<ans;
	return 0;
}
