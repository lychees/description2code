#include<iostream>
#include<list>
#include<bitset>

using namespace std;

bitset<100000> visited;
list<unsigned long> connected[100000];
unsigned long n,e,i,j,total,t,a,b;


	void mark_circles(unsigned long x)
	{
		visited[x]=true;
		for(list<unsigned long>::iterator it=connected[x].begin();it!=connected[x].end();++it)
		{
			if(!visited[*it])
			{
				mark_circles(*it);
			}
		}
	}


int main()
{
	cin>>t;
	while(t--)
	{
		cin>>n>>e;
		for(i=0;i<n;i++)
		{
			connected[i].clear();
			visited[i]=false;
		}
		for(i=0;i<e;i++)
			{
				cin>>a>>b;
				connected[a].push_back(b);
				connected[b].push_back(a);
			}
		total=0;
		for(i=0;i<n;i++)
		{
			if(!visited[i])
			{
				total++;
				mark_circles(i);
			}
		}
		cout<<total<<"\n";
	}

	return 0;
}
