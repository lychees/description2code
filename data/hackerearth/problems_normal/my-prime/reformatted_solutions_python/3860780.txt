counts = input()
s = raw_input()
input_data = s.split(' ')
result = ''
for index in range(counts):
    count = 0
    for i in range(counts):

        if (int(input_data[index]) % int(input_data[i]) == 0):
            count = count + 1

    if (count == 1):
        result = result + input_data[index] + ' '
print result
