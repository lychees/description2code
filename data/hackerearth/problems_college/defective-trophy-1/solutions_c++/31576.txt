typedef long long ll;

int dp[10010];

int g(int x){
    if(x == 1) return 1;
    return 1 + (x%2);
}

int f(int n){
	if(n==0) return 0;
    if(n == 1) return 0;
    if(n == 2) return 1;
    if(dp[n] != -1) return dp[n];
    int ans = 1000000000;
    for(int i = 1; i <= n/2; i++){
        if((n-i)%2==0) {
            ans = min(ans, max(1 + f(i), 1+f((n-i)/2)));
            //ans = min(ans, g(i) + f(n-i));
        }
        if(i % 2 == 0){
        	ans = min(ans, max(1 + f(n-i), 1+f(i/2)));
        }
    }
    if(n % 2 == 0){
        ans = min(ans, 1 + f(n/2));
    }
    return dp[n] = ans;
}

int main(){
    memset(dp, -1, sizeof dp);
    int t; cin >> t;
    while(t--){
        int n;
        scanf("%d", &n);

        printf("%d\n", f(n));
    }

}
