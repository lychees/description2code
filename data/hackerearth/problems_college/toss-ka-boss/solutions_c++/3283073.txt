#include <iostream>
#include<string>
using namespace std;

int main()
{
    int t;
    long long int i;
    string s;
    cin>>t;
    while(t--)
    {
    	cin>>s;
    	long long int length=s.length();
    	long long int n=1;
    	long long int sum=0;
    	for(i=0;i<length;i++)
    	{
    		if(s.at(i)=='H')
    		{
    			sum=sum+n;
    			n++;
    		}
    		else
    		{
    			n=1;
    		}
    	}
    	cout<<sum<<"\n";
    }
    return 0;
}
