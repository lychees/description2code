#include <bits/stdc++.h>
using namespace std;

int main() {
	int n;
	cin>>n;
	while(n--) {
		string str;
		cin>>str;
		int i=0;
		int count = 0;
		int len = 0;
		while(i<str.length()) {
			if((str[i]>='a' && str[i]<='c') || (str[i]>='A' && str[i]<='C')) {
				len++;
			}else {
				count+= (len*(len+1))/2;
				len = 0;
			}
			i++;
		}
		count+=(len*(len+1))/2;
		cout<<count<<endl;
	}
	return 0;
}
