import math


def perm(n, k):
    return int(math.factorial(n)/math.factorial(n-k))


def ways():
    str = raw_input()
    m = int(str.split(" ")[0])
    n = int(str.split(" ")[1])
    if ((m+1) >= n):
        return ((math.factorial(m)*perm(m+1, n)) % 1000000007)
    else:
        return -1


t = int(raw_input())
while (t > 0):
    t = t-1
    print(ways())
