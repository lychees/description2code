#include<iostream>
#include<cstdio>
#include<stdlib.h>
#include<set>
#include<algorithm>
#include<vector>

using namespace std;

int main()
{
   int t;
   cin >> t;
   
   while(t--)
   {
      long long n;
      cin >> n;
      long long id, solve;     
      long long arr[1000000] = {0};
      
   
      long long maxi=-1, count = 0;
      for(int i=0; i<n; i++)
      {
          cin >> id >> solve;  
          if(solve == 1)
          {
             if(arr[id] == 0)
             {
                count++;
                arr[id] = 1;
             }
          }
          else
             count = 0;
          if(count > maxi)
             maxi = count;
      }
      
      cout << maxi << endl;
   }
   return 0;    
}
