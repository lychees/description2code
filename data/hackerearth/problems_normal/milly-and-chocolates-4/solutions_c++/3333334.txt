#include <iostream>
#include <iomanip>
using namespace std;
 
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		double sum=0.0;
		for(int i=0; i<n; i++)
		{
			double c = 0.0;
			cin>>c;
			sum += c;
		}
		double avg = sum/n;
		cout<<fixed<<setprecision(8)<<avg<<endl;
	}
    return 0;
}
