

t = input()
while t > 0:
    s = raw_input()
    d = {}
    f = 0
    for i in s:
        if i in d:
            d[i] = d[i]+1
        else:
            d[i] = 1
    # print d
    for i in d:
        if d[i] % 2 != 0:
            f = 1
            break
    if f == 1:
        print -1
    else:
        print 1
    t = t-1
