k = int(raw_input())
line = raw_input()
n = len(line)

sets = 0

for l in range(n):
	count = 0
	chars = ""
	for i, x in enumerate(line[l:]):
		if x not in chars:
			count += 1
			chars += x
		if count < k:
			continue
		elif count > k:
			break
		sets += 1
		# print line[l : l + i + 1]

print sets
