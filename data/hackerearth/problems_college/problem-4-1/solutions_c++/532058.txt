void fastRead_int(int &x)
{
	register int c=gc();
	x=0;int neg=0;
	for(;((c<48 || c>57) && c != '-');c=gc());
	if(c=='-')
	{
		neg=1;c=gc();
	}	
	for(; c>47 && c<58 ; c = getchar_unlocked())
	{
		x=(x<<1)+(x<<3)+c-48;
	}
	if(neg)
		x=-x;
}
int dp[101][101];
int a[103];

int fun(int l,int r)
{
	if(dp[l][r]!=-1)
		return dp[l][r];

	int i,j,k,m;

	int temp;
	int p,pp;
	

	m=0;

	for(i=l;i<=r;i++)
	{

		p=pp=0;
		if(i-2>=l)
			p=1;
		if(i+2<=r)
			pp=1;

		temp=a[i];
		if(p)
			temp+=fun(l,i-2);
		if(pp)
			temp+=fun(i+2,r);

		if(temp>m)
			m=temp;
	}

	dp[l][r]=m;
	return m;

}
