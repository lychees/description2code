#include <iostream>
using namespace std;

int main()
{
    int T,R,C,W;
    cin>>T;
    
    for(int t=1;t<=T;t++)
    {
    	cin>>R>>C>>W;
    	cout<<"Case #"<<t<<": ";
    	if(W==1)
    		cout<<R*C<<"\n";
    	else
    	{
    		int cnt=0;
    		if(C%W==0)
    		{
    			cnt+=(C/W-1)+(C/W)*(R-1);
    			cnt+=W;
    			cout<<cnt<<"\n";
    		}
    		else
    		{
    			cnt+=(C/W-1)+(C/W)*(R-1);
    			cnt+=W+1;
    			cout<<cnt<<"\n";
    		}
    	}
    }
    return 0;
}
