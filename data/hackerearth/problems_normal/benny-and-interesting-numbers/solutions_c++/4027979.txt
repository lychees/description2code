#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
template<class T>inline bool isPrime(T n){if(n<=1)return false;for(T i=2;i*i<=n;++i)if(n%i==0)return false;return true;}
#define SIZE 1000007
int pf[1000007];
void primeFactors()
 {
        pf[1]=1;
        for(ll i=2;i<SIZE;i+=2)
            pf[i]=2;
        for(ll i=3;i<SIZE;i+=2)
        {
            if(pf[i]==0)
            {
                pf[i]=i;
                for(ll j=i;i*j<SIZE;j+=2)
                    pf[i*j]=i;
            }
         }
}

int main()
{

    //primeFactors();
    int T;
    cin>>T;
    while(T--)
    {
        ll N;
        cin>>N;
        if(N<=3)
        {
           cout<<"Yes"<<endl;
           continue;
        }
        int c2=0;
        while(N%2==0)
        {
            ++c2;
            N>>=1;
        }
        if(c2>1)
        {
            cout<<"No"<<endl;
            continue;
        }
        if(isPrime(N))
        {
            cout<<"Yes"<<endl;
            continue;
        }
        map<int,int> mm;
        bool ans=true;

        for(ll i=2;i*i<=N;++i){
            while(N%i==0)
            {

                mm[i]++;
                if(mm[i]>1)
                {
                    ans=false;
                    break;
                }
                N/=i;
            }
        }
        if(ans)
            cout<<"Yes"<<endl;
        else
            cout<<"No"<<endl;
    }
    return 0;
}
