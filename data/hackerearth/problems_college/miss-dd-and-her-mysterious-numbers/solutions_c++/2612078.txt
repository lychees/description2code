#include<bits/stdc++.h>
using namespace std;
vector<int> s(1000010);
int arr[1000010];
void init()
{
int b=1000010,i,j;
for( i=2; i <= b; i+=2 )  { s[i]=1; }

  for( i=3; i <= b; i+=2 )  {
      if( s[i]==0 )  {
          for( j=i; j <= b; j+=i ) { s[j]+=1; }
      }
  }
}

int main()
{
init();
int n,t,cnt,k,i;
scanf("%d%d",&n,&t);
for(i=2;i<=n;i++)
    arr[s[i]]++;
for(;t>0;t--)
{
scanf("%d",&k);
cnt=0;
if(k<=1000000)
cnt=arr[k];
printf("%d\n",cnt);
}

}
