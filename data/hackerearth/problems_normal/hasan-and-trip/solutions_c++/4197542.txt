#include<bits/stdc++.h>
using namespace std;
#define ll long long int
pair<ll,ll> arr[1000100];
ll happy[1000100];
double ans[1000100];
double dist(ll i,ll j)
{
	double xval=(arr[i].first-arr[j].first)*(arr[i].first-arr[j].first);
	double yval=(arr[i].second-arr[j].second)*(arr[i].second-arr[j].second);
	return (sqrt(xval+yval));
}
int main()
{
	ll n;
	scanf("%lld",&n);
	ll i,x,y,f,j;
	for(i=1; i<=n; i++)
	{
		scanf("%lld %lld %lld",&x,&y,&f);
		arr[i]=make_pair(x,y);
		happy[i]=f;
	}
	for(i=1; i<=n; i++)
	{
		ans[i]=INT_MIN;
	}
	ans[1]=0;
	for(i=1; i<=n; i++)
	{
		ans[i]+=happy[i];
		for(j=i+1; j<=n; j++)
		{
			ans[j]=max(ans[j],ans[i]-dist(i,j));
		}
	}
	printf("%0.6lf\n",ans[n]);
	return 0;
}
