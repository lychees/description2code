using namespace std;

int idx,flag;
ll primes[1000005];
int t,it,res;
ll n;

void start(){
	idx=2;
	primes[1]=2;
	rep(i,3,1000000,2){
		flag=1;
		for(int j=1;primes[j]*primes[j]<=i;j++){
			if(i%primes[j]==0){
				flag=0;
				break;
			}
		}
		if(flag){
			primes[idx++]=i;
		}
	}
}

int main(){
	start();
	scanf("%d",&t);
	while(t--){
		it=1;
		flag=1;
		scanf("%lld",&n);
		while(n>1&&it<idx){
			res=0;
			while(n%primes[it]==0){
				n/=primes[it];
				res++;
			}
			if(res>1){
				flag=0;
				break;
			}
			it++;
		}
		if(flag)printf("Yes\n");
		else printf("No\n");
	}
	return 0;
}
