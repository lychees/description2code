#include<bits/stdc++.h>
#define md 1000000007
#define ll  long long
#define gc getchar//_unlocked
using namespace std;

ll a[1000];
ll b_search(ll no,int n){
	//cout<<n<<endl;
	if(no>=a[n-1])
		return n;
	if(no==1)
		return 1;
	int lb=1;
	int ub=n;
	while(1)
	{
			int mid=(lb+ub)/2;
			if(no==a[mid])
				return mid;
			else if(no>a[mid]){
				if(a[mid+1]>no)
				{	return mid;}
				lb=mid+1;
				
			}
			else{
				if(a[mid-1]<no)
				{	return mid-1;}
				ub=mid-1;
			}
	}
}
int main(){
    #ifndef ONLINE_JUDGE
        freopen("input.txt","r",stdin); 
    #endif
	ll t,n,x=1,y=1;
	cin>>t;
	a[1]=x;
	int id;
	for( id=2;x+y<=10000000000000000;id++)
	{
		a[id]=x+y;
		ll p=y;
		y=x+y;
		x=p;
	}
	while(t--){
		cin>>n;
		
		ll ct=0;
		ll pp=n;
		//cout<<a[73]<<endl;
		while(1){
			ct=pp-b_search(pp,id);
			//cout<<pp<<" "<<ct<<endl;
			if(ct>=n)
				break;
			
			pp+=n-ct;
		}
		cout<<pp<<endl;
		
	}
	return 0;
}
