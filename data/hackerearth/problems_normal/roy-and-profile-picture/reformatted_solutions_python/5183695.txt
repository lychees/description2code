L = int(raw_input())

N = int(raw_input())

for i in range(0, N):

    Xi = (raw_input())

    Xi = Xi.split(" ")

    Wi = int(Xi[0])
    Hi = int(Xi[1])

    if (Wi == Hi) and (Wi >= L and Hi >= L):

        print "ACCEPTED"

    elif Wi < L or Hi < L:

        print "UPLOAD ANOTHER"

    else:

        print "CROP IT"
