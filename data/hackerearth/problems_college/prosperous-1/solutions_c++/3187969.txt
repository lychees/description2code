#include<bits/stdc++.h>
using namespace std;

const int md=1e9+7;
const int N=1e5+10;

int dp[N][10];
int sum[N];

inline void add(int& a,int b) {
  a+=b;
  if(a>=md) {
    a-=md;
  }
  return;
}
inline void sub(int& a,int b) {
  a-=b;
  if(a<0) {
    a+=md;
  }
  return;
}
int solve(int s,int d) {
  memset(dp,0,sizeof(dp));
  for(int i=0;i<=9;i++) {
    dp[i][i]=1;
  }
  for(int i=1;i<=s;i++) {
    for(int j=1;j<=9;j++) {
      for(int k=0;k<=d;k++) {
        if(i!=j&&i-j>=0&&j-k>=0) {
          add(dp[i][j],dp[i-j][j-k]);
        }
      }
    }
  }
  int sum=0;
  for(int i=1;i<=9;i++) {
    add(sum,dp[s][i]);
  }
  return sum;
}
int main() {
  //freopen("a.cpp","r",stdin);
  int tt;
  cin>>tt;
  int cases=1;
  while(tt--) {
    cout<<"Case #"<<cases++<<": ";
    int s,d;
    cin>>s>>d;
    int c1=solve(s,d);
    int c2=0;
    if(d-1>=0) {
      c2=solve(s,d-1);
    }
    sub(c1,c2);
    cout<<c1<<endl;
  }
  return 0;
}
