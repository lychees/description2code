#include <iostream>
using namespace std;

int main()
{
	long long int sum=0;
    int a[1000001],t,i,x,y,j;
    
    for(i=2;i<1000001;i++)
    {
    	if(a[i]==0)
    	{
    		for(j=i+i;j<1000001;j=j+i)
    		{
    			a[j]=1;
    		}
    	}
    }
    scanf("%d",&t);
    while(t--)
    {
    	sum=0;
    	scanf("%d%d",&x,&y);
    	for(i=x;i<=y;i++)
    	{
    		if(a[i]==0 && i!=1)
    			sum=sum+i;
    	}
    	printf("%lld\n",sum);
    }
    return 0;
}
