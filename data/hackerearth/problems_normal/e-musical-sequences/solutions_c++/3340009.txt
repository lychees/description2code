#include<bits/stdc++.h>
#define ll long long
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		ll n,m,z;
		cin>>n>>m>>z;
		ll a[3*n+10];
		ll v=0,s=1;
		for(int i=0;i<n;i++)
		{	cin>>a[i];
		}
		
		for(int i=n-1;i>=0;i--)
		{
			v=(v+s*a[i]);
			if(v<0 || v>=m)
				v=(v%m + m)%m;
			s=s*(-1);
		}

		s=((n+1)%2);
		if(s==0) s=1;
		else s=-1;
		
		ll an[5*n];
		an[n]=(v%m+m)%m;
		for(int i=n+1,j=0;j<n;i++,j++){			
			an[j]=((s*a[j])%m + m)%m;
		}
		if(((z-n)/(n+1))%2 == 1)
			s=-1;
		else s=1;
		if(n%2==1) s=1;
		cout<<((s*an[(z)%(n+1)])%m + m)%m<<'\n';
	}
}
