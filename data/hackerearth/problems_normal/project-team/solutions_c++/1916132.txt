#include <iostream>
using namespace std;

int main()
{
    int cases,i=0,j=0,k=0;
    
    cin>>cases;
    
    int students[cases],levelArray[cases][100];
    
    for(i=0;i<cases;i++)
    {
        cin>>students[i];
        cout<<"\n";
        int count = students[i];
        for(j=0;j<count;j++)
        {
            cin>>levelArray[i][j];
        }
    }
    for(i=0;i<cases;i++)
    {
        int count = students[i];
        
        int array[count];
        for(j=0;j<count;j++)
        {
            array[j] = levelArray[i][j];
        }
        for(j=0;j<count-1;j++)
        {
            for(k=0;k<count-1-j;k++)
            {
                if(array[k]>array[k+1]){
                    
                    int temp = array[k];
                    array[k] = array[k+1];
                    array[k+1] = temp;
                }
            }
        }
//        cout<<"sorted array=";
//        for(j=0;j<count;j++)
//        {
//            cout<<array[j]<<" ";
//        }
        //cout<<"\n";
        int largeSide = count-1;
        int smallSide = 0;
        int teamArray[count/2];
        
        //cout<<"team sum = ";
        for(j=0;j<count/2;j++)
        {
            //cout<<array[largeSide] + array[smallSide]<<" ";
            teamArray[j] = array[largeSide] + array[smallSide];
            largeSide -= 1;
            smallSide += 1;
        }
        int lessKnow = 100000;
        int moreKnow = 0;
        for(j=0;j<count/2;j++)
        {
            if (teamArray[j]<lessKnow) {
                
                lessKnow = teamArray[j];
            }
            if (teamArray[j]>moreKnow) {
                
                moreKnow = teamArray[j];
            }
        }
//        cout<<"moreKnow="<<moreKnow<<"&lessKnow="<<lessKnow<<"\n";
        cout<<moreKnow-lessKnow<<"\n";
        
    }
}
