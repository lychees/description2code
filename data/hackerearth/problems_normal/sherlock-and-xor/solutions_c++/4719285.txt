#include <iostream>
#include<vector>
using namespace std;

int main()
{
unsigned int T,N;
long long A;
vector<long long> even, odd,result;

cin>>T;
for(unsigned int i=0;i<T;i++)
{
	even.clear();
	odd.clear();
	cin>>N;
	for(int j=0;j<N;j++)
	{
		cin>>A;
		if(A%2 == 0)
		even.push_back(A);
		else
		odd.push_back(A);
	}
	result.push_back(even.size()*odd.size());
}
for(vector<long long>::iterator it=result.begin();it!= result.end();it++)
cout<<*it<<endl;
    return 0;
}
