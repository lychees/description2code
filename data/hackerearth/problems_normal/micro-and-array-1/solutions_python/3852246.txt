def findValue(input_data=[],value=0):
		count = 1
		index = -1
		for i in input_data:
			if i >= value:
				index = count
				break
			count += 1
		return index

n,q = map(int,raw_input().split())
input_data = map(int,raw_input().split())

for i in range(q):
	input_operation = map(int,raw_input().split())
	if len(input_operation) == 2:
		print findValue(input_data = input_data, value = input_operation[1])
	else:
		input_data[input_operation[1]-1] = input_operation[2]
