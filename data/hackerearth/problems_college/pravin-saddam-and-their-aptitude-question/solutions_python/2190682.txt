def fact(num):
	if(num==1):
		return 1
	else:
		return num*fact(num-1)
		
		
t=input()
t=int(t)
for i in range(t):
	sum=0
	m=raw_input()
	x=[int(j) for j in m.split()]
	s=len(x)
	for k in range(x[0],x[s-1]):
		sum=sum+fact(k)
	sum=sum+fact(x[s-1])
	print sum%10
