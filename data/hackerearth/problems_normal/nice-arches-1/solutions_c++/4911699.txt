#include <iostream>
#include <string>
#include <stack>

using namespace std;

int main()
{
    int t, count = 0;
    cin >> t;
    stack<char> wordStack;
    while(t--)
    {
    	string str;
    	cin >> str;
    	while(!wordStack.empty())
    		wordStack.pop();
    	for(int i = 0; i < str.length(); i++)
    	{
    		if(wordStack.empty())
    			wordStack.push(str[i]);
    		else
    			if(wordStack.top() != str[i])
    				wordStack.push(str[i]);
    			else
    				wordStack.pop();
    		
    	}
    	if(wordStack.empty())
    		count += 1;
    }
    cout << count << endl;
}
