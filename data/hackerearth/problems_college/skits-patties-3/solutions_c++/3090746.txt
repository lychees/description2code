#include<iostream>  
#include<stdio.h>
using namespace std;


int result(int );

int main()
{
	int no_testcases,n,i;
	cin>>no_testcases;
	for(i=0;i<no_testcases;i++)
	{
		cin>>n;
		if(!result(n))
		{
			printf("False\n");
		}
		else if(result(n))
		{
			printf("True\n"); 
		}
	}
} 

int result(int x)
{	  
	if(x==0)
	{
            return 1;
        }
	else if(x<0)
	{
            return 0;
        }
	return (result(x-6) || result(x-9) || result(x-20));
}
