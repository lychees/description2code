#include <iostream>
#include <math.h>
using namespace std;

int main()
{
    int n, ans = 0;
    string s;
    int book[26] = {0};
    
    cin>>n;
    
    while(n--)
    {
    	cin>>s;	
    	book[s[0] - 'a']++;
    }
    
    for(int i = 0 ; i < 26 ; i++)
    {
    	ans = ans + ceil(book[i] / 10.0);
    }
    
    cout<<ans;
    return 0;
}
