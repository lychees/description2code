#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string s;
		int arr[26]={0},i,flag=1;
		cin>>s;
		for(i=0;i<s.length();i++)
		{
			arr[s[i]-97]++;
		}
		for(i=0;i<26;i++)
		{
			if(arr[i]>1)
			{
				cout<<"Yes"<<endl;
				flag=0;
				break;
			}
		}
		if(flag)
		{
			cout<<"No"<<endl;
		}
	}
    return 0;
}
