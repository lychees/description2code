#include <iostream>
using namespace std;

int main()
{
	int T;
	cin >> T;
	while(T--) {
		string s1, s2;
		cin >> s1 >> s2;
		if(s1.length() != s2.length()) {
			cout << "No" << endl;
			continue;
		}
		int target;
		cin >> target;
		int steps = 0;
		for(int i=0; i<s1.length(); i++) {
			if(s1[i] != s2[i])
				steps++;
		}
		
		if(steps <= target && (target-steps) % 2 == 0)
			cout << "Yes" <<endl;
		else
			cout << "No" <<endl;
	}
    return 0;
}
