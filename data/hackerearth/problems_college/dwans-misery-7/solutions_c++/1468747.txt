#include <iostream>
#include <stdio.h>
#define ll long long int
using namespace std;
ll no[25000];

int main()
{
	int t;
	ll n,num,i,j;
	ll included,excluded,new_tp,maxsum;
	bool fg=0;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%lld",&n);
		for(i=0;i<n;i++)
			scanf("%lld",&no[i]);

		maxsum=0;
		included=0;
		excluded=0;
		included=no[0];
		for(i=1;i<n-1;i++)
		{	
			new_tp=((excluded>included)?excluded:included);
			included=excluded+no[i];
			excluded=new_tp;
		}

		if(included>excluded)
			maxsum=included;
		else
			maxsum=excluded;
		
		included=0;
		excluded=0;
		included=no[1];
		for(i=2;i<n;i++)
		{	
			new_tp=((excluded>included)?excluded:included);
			included=excluded+no[i];
			excluded=new_tp;
		}

		if(included>excluded)
		{
			if(maxsum<included)
				maxsum=included;
		}
		else
		{
			if(maxsum<excluded)
				maxsum=excluded;
		}

		printf("%lld\n",maxsum);
	}
	return 0;
}
