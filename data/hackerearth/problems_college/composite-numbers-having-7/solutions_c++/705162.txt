#include<stdio.h>
#define max 1000007
int a[max]={0};
void sieve(){
    long i,j;
        for( i=2;i<=max;i++){
                if(a[i]!=0)
                    continue;
                for( j=2*i;j<=max;j+=i){
                        if(a[j]==0)
                        a[j]=i;
                }
        }
}

int main(){

    sieve();
    int t;
    scanf("%d",&t);
    while(t--)
    {

            long n,m,ans=0,k,i;
            int r;
            scanf("%ld%ld",&n,&m);
            for( i=n;i<=m;i++)
            {
                if(a[i]!=0)
                {
                    k=i;
                    while(k>0)
                    {
                        r=k%10;
                        k=k/10;
                        if(r==7)
                        {
                            ans++;
                            break;
                        }

                    }
                }

            }
            if(ans==0)
                printf("%d\n",-1);
            else
                printf("%d\n",ans);
}
return 0;
}
