#include<bits/stdc++.h>
using namespace std;
int possible(int arr[],int sum,int index,int n,int curr){
	if(curr==sum)return 1;
	else{
		for(int i=index;i<n;i++){
			
			if(possible(arr,sum,i+1,n,curr+arr[i]))return 1;
			if(possible(arr,sum,i+1,n,curr))return 1;
		}
	}
	return 0;
}

int main(){
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		int arr[n];
		for(int i=0;i<n;i++)cin>>arr[i];
		int sum;
		cin>>sum;
		if(possible(arr,sum,0,n,0))cout<<"YES"<<endl;
		else cout<<"NO"<<endl;
	}
	return 0;
}
