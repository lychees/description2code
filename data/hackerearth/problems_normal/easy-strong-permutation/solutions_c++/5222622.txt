#include "bits/stdc++.h"
using namespace std;
long long abs(long long a,long long b)
{
	return a > b ? a-b : b-a;
}
long long func(long long *arr,long long n)
{
	long long ans=0;
	ans = abs(arr[0],arr[n-1]);
	for(long long i=0;i<n-1;i++)
	{
		ans = ans + abs(arr[i],arr[i+1]);
	}
	return ans;
}
int main()
{
	long int n;
	cin >> n;
	
	long long arr[n];
	for(long long i=0;i<n;i++)
		cin >> arr[i];
	
	sort(arr,arr+n);
	
	long long brr[n];
	
	long long i=0,j=n-1;
	
	for(long long count=0;count<n;count++)
	{
		if(count%2 == 0)
			brr[count] = arr[i++];
		else
			brr[count] = arr[j--];
	}
	i=0,j=n-1;
	long long crr[n];
	for(long long count=0;count<n;count++)
	{
		if(count%2 == 0)
		{
			crr[count] = arr[j--];
		}
		else
		{
			crr[count]=arr[i++];
		}
	}
	
	cout << max(func(brr,n) , func(crr,n)) << endl;
	return 0;
}
