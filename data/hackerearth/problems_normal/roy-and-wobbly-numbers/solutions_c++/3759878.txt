#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int t=0,n=0,k=0,a=0,b=0,offset=0,flag=0;
    cin >> t;
    while(t--){
    	n=0,k=0,a=0,b=0,offset=0;
    	flag=0;
    	cin >> n >> k;
    	if(k>81){
    		cout << -1 << endl;
    		continue;
    	}
    	a=ceil(float(k/9.0));
    	offset=k-(9*(a-1));
    	if(offset <= a){
    		b=offset-1;
    		
    	}else {
    		b=offset;
    	}
    	for(int i=1;i<=n;i++){
    		if(i%2==0){
    			cout << b;
    			
    		}else{
    			cout << a;
    		}
    	}
    	cout << endl;
    	
    }
    return 0;
}
