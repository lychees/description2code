no_of_test_cases = int(raw_input())

for cases in range(no_of_test_cases):
    str_dict = {}
    for i in range(ord('a'), ord('z') + 1):
        str_dict[chr(i)] = -i
    input_text = raw_input()
    for item in input_text:
        str_dict[item] += 26
    sorted_dict = sorted(str_dict, key=(str_dict.get))
    print ' '.join(sorted_dict)
