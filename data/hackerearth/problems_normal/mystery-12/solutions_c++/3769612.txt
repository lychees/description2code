#include<iostream>
#include<algorithm>

using namespace std;

typedef long long ll;

ll pw(ll a, ll b, ll mod){
	a %= mod;
	ll ret = 1;
	while (b){
		if (b & 1)
			ret = (ret * a) % mod;
		b >>= 1;
		a = (a * a) % mod;
	}
	return ret;
}

int main(){
	ios::sync_with_stdio(false);
	cin.tie(0);
	ll te;	cin >> te;
	while (te--){
		ll a, d, n;	cin >> a >> d >> n;
		if (a % n == 0)
			cout << "0\n";
		else if (d % n == 0)
			cout << "-1\n";
		else{
			ll z = n - a % n;
			cout << (z * pw(d, n - 2, n)) % n << "\n";
		}
	}	
	return 0;
}
