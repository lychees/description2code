def find_xor_0(value):
    no = (value)%4
    xor = 0
    while no >=0:
        xor = xor^value
        value=value-1
        no = no-1
    return xor

T = input()
for x in xrange(T):
    L, R = map(int,raw_input().strip().split(' '))
    print (find_xor_0(L-1)^find_xor_0(R))
