#include <iostream>
using namespace std;

int main()
{
    long long n;
    string s;
    cin>>n;
    cin>>s;
    long long a[26];
    for(long long i=0;i<26;i++)
    a[i]=0;
    for(long long i=0;i<n;i++)
    a[s[i]-'a']++;
    long long q;
    cin>>q;
    while(q--)
    {
    	char x,y;
    	cin>>x>>y;
    	if(x==y)
    	cout<<((a[x-'a'])*(a[y-'a']-1))/2<<endl;
    	else
    	cout<<a[x-'a']*a[y-'a']<<endl;
    }
    return 0;
}
