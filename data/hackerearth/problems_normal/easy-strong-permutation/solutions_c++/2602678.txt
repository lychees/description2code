#include <iostream>
#include <list>
#include<vector>
#include <algorithm>
using namespace std;

int main()
{
    int N;
	cin >> N;
	//list<long> lstInput;
	vector<long> lstInput;

	for( int i = 0; i < N; i++ )
	{
		long var;
		cin >> var;

		lstInput.push_back(var);
	}

	std::sort(lstInput.begin(), lstInput.end());
	//lstInput.sort();

	bool bIsOdd;

	if( N%2 == 0 )
		bIsOdd = false;
	else
		bIsOdd = true;

	/*list<long>::iterator itStart, itEnd;

	itStart = lstInput.begin();
	itEnd = lstInput.end();*/
	long sum = 0;
	int left = 0;
	int right = N-1;

	for( int i = 0; i < N; i++ )
	{
		long diff = 0;

		if( left == right )
		{
			diff = abs(lstInput[0] - lstInput[left]);
		}
		else
		{
			diff = abs(lstInput[right] - lstInput[left]);	
		}

		sum += diff;

		if( i%2 == 0 )
			left++;
		else
			right--;


	}

	cout << sum;

	return 0;
}
