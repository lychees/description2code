int gcd(int a, int b) {
  return b == 0 ? a : gcd(b, a%b) ; 
}
int main()
{        
         int t;
         scanf("%d",&t);
         while(t--)
         {
                   int n,i,ans;
                   scanf("%d",&n);
                   int a[n];
                   for(i=0;i<n;i++)
                                   cin>>a[i];
                   sort(a,a+n);
                   ans=a[0];
                   for(i=1;i<n;i++)
                                     ans=gcd(ans,a[i]);
                   if(ans==1)
                             printf("YES\n");
                   else
                             printf("NO\n");
         }
         return 0;
}
