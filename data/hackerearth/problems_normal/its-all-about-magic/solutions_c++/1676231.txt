#include <iostream>
using namespace std;

int main()
{
    long int T=0,M=0,n,m,t=0;
    
    cin>>T;
    
    while(T--){
    	
    	cin>>M;
    	
    	if(M==1){
    		cout<<"0 1\n";
    	}
    	
    	else if(M==2){
    		cout<<"1 1\n";
    	}
    	
    	else{
    		n=1;
    		m=1;
    		t=0;
    		for(long int i=2;i<M;i++){
    			t=n;
    			n=m+n;
    			m=t;
    		}
    		
    		cout<<m<<" "<<n<<"\n";
    	}
    	
    }
    return 0;
}
