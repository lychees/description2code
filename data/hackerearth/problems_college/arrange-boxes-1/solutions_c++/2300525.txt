#include <iostream>
using namespace std;
#include<bits/stdc++.h>
void swap(long int *a,long int *b)
{
	int temp=*a;
	*a=*b;
	*b=temp;
}
long int solve(long int a[],long int b[],long int n)
{
	long int count=0;
	for(long int i=0;i<n;i++)
	{
		for(long int j=0;j<n;j++)
		{
			if(a[j]==b[i])
			{
				if(i==j)
				break;
				else
				{
					while(i!=j)
					{
						swap(&a[j],&a[j-1]);
						j--;
						count++;
					}
					break;
				}
			}
		}
	}
	return count;
}
int main()
{
	int t;
	cin>>t;
	long int n,a[100004],b[100005];
	while(t--)
	{
		cin>>n;
		for(long int i=0;i<n;i++)
		{
		cin>>a[i];
		}
		
		for(long int i=0;i<n;i++)
		{
		cin>>b[i];
		}
		
		cout<<solve(a,b,n)<<endl;
		
	}
	
    //cout << "Hello World!" << endl;
    return 0;
}
