#include <iostream>
#include <cstdio>

using namespace std;

int main()
{
        int test_cases;
        scanf("%d",&test_cases);
        while(test_cases > 0)
        {
                int N,i=0,m;
                char c;
                scanf("%d",&N);
                while(i<N)
                {
                	scanf("%d",&m);
                	c=(char)(~m);
                	printf("%c",c);
                	i++;
                }
                printf("\n");
                test_cases--;
        }
        return 0;
}
