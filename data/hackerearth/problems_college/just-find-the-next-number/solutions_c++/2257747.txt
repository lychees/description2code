#include<cstdio>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		char str[200];
		int j=-1;
		while(n!=0)
		{
			str[++j]=(n%2)+48;
			n=n/2;
		}
		//printf("%s\n",str);
		int i,k=0; 
		for(i=1;i<=j;i++)
		{
			if(str[i]=='1' && str[i-1]=='1')
			{
				for(;k<=i-1;k++)
					str[k]='0';
				while(i<=j && str[i]=='1')
				{
					str[i]='0';
					i++;
				}
				str[i]='1';
			}
		}
		//printf("%s\n",str);
		int an=0,a=1;
		for(int k=0;k<i;k++)
		{
			an=an+(str[k]-48)*a;
			a=a*2;
		}
		printf("%d\n",an);
	}
	return 0;
}
