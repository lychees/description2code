#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
long long fct[500001];
int main(){
	int t;
	scanf("%d",&t);
	fct[0]=1;
	for(long long int i=1;i<=500000;i++){
		fct[i]=(fct[i-1]*i)%mod;
	}
	while(t--){
		int f[500001];
		for(int i=0;i<500001;i++)
		  f[i]=0;
		int n,m;
		scanf("%d %d",&n,&m);
		for(int i=1;i<=n;i++){
			for(int j=1;j<=m;j++){
				f[(i|j)]++;
			}
		}
		long long ans=1;
		for(int i=1;i<500001;i++){
			if(!f[i])
			   continue;
			ans*=fct[f[i]];
			ans%=mod;
		}
		printf("%lld\n",ans);
	}
	return 0;
}
