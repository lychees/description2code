name = raw_input('What is your name?\n')
print 'Hi, %s.' % name

t = int(raw_input())
while t > 0:
    m, p = map(int, raw_input().split())
    temp = m ^ p
    count = 0
    while temp > 0:
        temp = temp & (temp - 1)
        count += 1
    print(count)
    t -= 1
