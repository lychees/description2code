#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        int a[n];
        for(int i=0;i<n;i++)
            cin>>a[i];
        int f=0,i=0,a1;
                while(1)
        {
            a1=a[i];
            if(a1==-1)
            {
                cout<<"Win"<<endl;
                f=1;
                break;
            }
            if(a1==-2)
            {
                f=2;break;
            }
            else
            {
                a[i]=-2;
                i=a1;
            }
        }
        if(f!=1)
            cout<<"Lose"<<endl;
    }
    return 0;
}
