#include<iostream>
#include<stdio.h>
using namespace std;
int t,n,m;
char a[25][25];
int toggle(int x,int y)
{
    for(int i=x;i>=0;i--)
    {
        for(int j=y;j>=0;j--)
        {
            if(a[i][j]=='1')a[i][j]='0';
            else a[i][j]='1';
        }
    }
}
int main()
{

    scanf("%d",&t);
    while(t--)
    {

        int count=0;
        scanf("%d %d",&n,&m);
        for(int i=0;i<n;i++)
        {
            scanf("%s",&a[i]);

        }


        for(int i=n-1;i>=0;i--)
        {
            for(int j=m-1;j>=0;j--)
            {

                if(a[i][j]=='0')
                {
                    count++;
                    toggle(i,j);
                }
            }
        }
        cout<<count<<"\n";
    }
    return 0;
}
