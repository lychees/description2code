#include<bits/stdc++.h>
#define ll long long
using namespace std;

int main()
{
    ll t,n,m,max;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	ll a[n],b[n];
    	max=0;
    	for(ll i=0;i<n;i++)
    	    cin>>a[i];
    	for(ll i=0;i<n;i++)
    	    cin>>b[i];
    	for(ll i=0;i<n;i++)
    	{
    		for(ll j=i;j<n;j++)
    		{
    			if(a[i]>b[j])
    			   break;
    			m = j-i;
    			if(m>=max)
    			   max=m;
    		}
    	}
    	cout<<max<<endl;
    }
    return 0;
}
