#include <bits/stdc++.h>
using namespace std;
typedef unsigned long long int ll;
#define F(i,a,b) for(int i = (int)(a); i <= (int)(b); i++)
#define RF(i,a,b) for(int i = (int)(a); i >= (int)(b); i--)
int main()
{
    #ifndef ONLINE_JUDGE
        freopen("input.txt","r",stdin);
        freopen("output.txt","w",stdout);
    #endif
    ll T,N;
    bool player = 0;
    scanf("%llu",&T);
    while(T--)
    {
        scanf("%llu",&N);
        int setbits = 0;
        while(N)
        {
            if(N&1) setbits++;
            N>>=1;
        }
        //cout<<setbits<<"\n";
        if(player == 0 && setbits%2==1)
        {
            printf("Sherlock\n");
        }
        else if(player == 1 && setbits%2==0)
        {
            player = 0;
            printf("Sherlock\n");
        }
        else if(player == 1 && setbits%2==1)
        {
            printf("Watson\n");
        }
        else
        {
            player = 1;
            printf("Watson\n");
        }
    }
    return 0;
}
