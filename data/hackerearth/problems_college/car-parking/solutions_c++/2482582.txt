#include<iostream>
using namespace std;

int main(){
    int test;
    cin>>test;

    while(test--){
        string x;
        cin>>x;

        for(int i=0;i<x.length();i++){
            if(x[i]=='B'){
                    if(i>0)
                x[i-1]='1';
                    if(i>1)
                x[i-2]='1';
            }
            else if(x[i]=='S'){
                    if(i>0)
                x[i-1]='1';
            if(i<x.length()-1 && x[i+1]=='-')
                x[i+1]='1';
            }
        }

        int count=0;
        for(int i=0;i<x.length();i++)
            if(x[i]=='-')
            count++;

        cout<<count<<endl;
    }
}
