def noOfFactors(n):
    result = set()
    for i in xrange(1, int(n ** 0.5) + 1):
        div, mod = divmod(n, i)
        if mod == 0:
            result |= {i, div}
    return len(result) - 1  # removing 1


def main():
    T = input()
    for i in xrange(T):
        N = input()
        print noOfFactors(N)


if __name__ == '__main__':
    main()
