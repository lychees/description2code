#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
	int n,*arr,i,s,c=0;
	long long int sum;
	cin >> n;
	arr = new int[n];
	for(i=0;i<n;i++)
		cin >> arr[i];
	sort(arr,arr+n);
	cin >> s;
	sum = 0;
	for(i=n-1;i>=0;i--)
	{
		sum += arr[i];
		c++;
		if(sum >= s)
			break;
		
	}
	cout << c;
	return 0;
}
