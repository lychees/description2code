cases = int(raw_input())

for case in range(cases):
	N = int(raw_input())
	paints = map(int, raw_input().split())
	x = set(paints)
	dic = {}
	max_count = 0
	element = paints[0]
	for i in x:
		dic[i] = paints.count(i)
		if paints.count(i) > max_count:
			element = i
			max_count = paints.count(i)
	if max_count - (sum(dic.values()) - dic[element]) > 1:
		print "bad luck"
	else:
		print "can do"
