#include <bits/stdc++.h>
using namespace std;

int m, n, k;
long long p[2000], t[2000], dp[2000][2001];
pair <long long , long long> x[2000];

long long rec (int i=0, int j=0, long long s=0)
{
	if(i==n-1)
		return p[i]*(s+k+t[i]);

	long long &ret=dp[i][j+1];

	if(ret!=-1)
		return ret;

	return ret=min(rec(i+1,j,s+t[i]),rec(i+1,i+1,0)+p[i]*(s+k+t[i]));
}

int main()
{
	ios::sync_with_stdio(false);cin.tie(NULL);cout.tie(NULL);


	cin >> m;

	while (m--)
	{
		memset(dp,-1,sizeof dp);
		cin >> n >> k;

		for(int i=0 ; i<n ; i++)
			cin >> x[i].first;

		for(int i=0 ; i<n ; i++)
			cin >> x[i].second;

		sort (x,x+n);

		for(int i=0 ; i<n ; i++)
			p[i]=x[i].first, t[i]=x[i].second;

		cout << rec() << '\n';
	}
	return 0;
}
