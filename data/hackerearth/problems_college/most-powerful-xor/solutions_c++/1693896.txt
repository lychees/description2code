#include <iostream>
using namespace std;

int main()
{
    int T;
    cin >> T;
    while(T--)
    {
    	int l,r,ans,max=0;
    	cin >> l >> r;
    	for(int i=l ; i<=r ; i++)
    	{
    		for(int j=l ; j<=r ; j++)
    		{
    			if(i<=j)
    			ans=i^j;
    			if(ans>max)
    			max=ans;
    			
    		}	
    	}
    	cout << max << endl;
    }
    return 0;
}
