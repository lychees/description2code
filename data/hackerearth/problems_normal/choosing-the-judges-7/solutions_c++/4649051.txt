#include <bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
	ios::sync_with_stdio(false);
	int t,cnt=1;
	scanf("%d",&t);
	while(t--)
	{
		int n,i,arr[10009];
		ll ans[10009];
		scanf("%d",&n);
		for(i=0;i<n;i++)scanf("%d",&arr[i]);
		ans[0]=arr[0];
		ans[1] = arr[1];
		ans[1]=max(ans[0],ans[1]);
		for(i=2;i<n;i++)
		{
			ans[i] = max(ans[i-1],arr[i]+ans[i-2]);
		}
		printf("Case %d: %llu\n",cnt++,ans[n-1]);
	}
    return 0;
}
