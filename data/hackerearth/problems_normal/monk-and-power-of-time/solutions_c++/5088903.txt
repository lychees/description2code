#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int N, i, lE, count = 0, tmp;
    vector<int> CO, EO;
    cin >> N;
    
    for(i=0; i<N; i++)
    {
    	cin >> tmp;
    	CO.push_back(tmp);
    }
    	
    for(i=0; i<N; i++)
    {
    	cin >> tmp;
    	EO.push_back(tmp);
    }

    lE = 0;	
    i=0;
    while(lE < N)
    {
    	if(CO[i] == -1)
    	{
    		i = (i+1) % N;
    		continue;
    	}
    	if(CO[i] != EO[lE])
    	{
    		i = (i+1) % N;
    		count++;
    	}
    	
    	else if(CO[i] == EO[lE])
    	{
    		CO[i] = -1;
    		lE++;
    		i = (i+1) % N;
    		count++;
    	}
    }
    cout << count << endl;
    return 0;
}
