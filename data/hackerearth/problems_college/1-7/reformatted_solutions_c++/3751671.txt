#include <bits/stdc++.h>
using namespace std;

int main() {
  int a[26] = {0}, i, j, k, l, t;
  string s, d;
  cin >> t;
  while (t--) {
    memset(a, 0, sizeof(a));
    cin >> s >> d;
    // cout<<s<<" "<<d;
    k = s.length();
    for (i = 0; i < k; i++)
      a[s[i] - 'a']++;

    //	cout<<d[0]<<" ";

    //	for(i=0;i<26;i++)
    //	cout<<a[i]<<" ";

    j = a[d[0] - 'a'];

    cout << j << "\n";
  }
}
