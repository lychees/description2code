#include<iostream>
using namespace std;
int func(int a[],int n){
int b[100001]={0};
for(int i=0;i<n;i++){
b[a[i]]++;
}
for(int i=0;i<100001;i++){
if(b[i]==1)
return i;
}
}
int main(){
int t;
cin>>t;
int n;
while(t--){
cin>>n;
int a[n];
for(int i=0;i<n;i++)
cin>>a[i];
cout<<func(a,n)<<"\n";
}

}
