#include<bits/stdc++.h>
using namespace std;
int main()
{
int i,t,n,count;
cin>>t;
while(t--)
{
cin>>n;
count=0;
while(n!=1)
{
if(n%2!=0&&n%3!=0)
{
n=n-1;
count++;
}
else if(n%3==0)
{
n=n/3;
count++;
}
else if(n%2==0)
{
n=n/2;
count++;
}
}
cout<<count<<endl;
}
return 0;
}
