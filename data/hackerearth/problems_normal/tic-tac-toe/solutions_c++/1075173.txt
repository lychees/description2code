int mod_pow(int x, ll y, int mod) {
    ll res = 1;
    ll a = x;
    while (y > 0) {
        if (y & 1) {
            res = (res * a) % mod;
        }
        a = (a * a) % mod;
        y >>= 1;
    }
    return (int) res;
}

        ll _min, _max;
        _max = (n * (n - 1) % MOD) * (2 * n - 1) % MOD;
        _max = (_max * mod_pow(6, MOD - 2, MOD)) % MOD;
        if (n % 2) {
            _min = (((n - 1) * (n - 1)) / 4) % MOD;
            _min = (_min * n) % MOD;
        }
        else {
            ll tn = n / 2;
            _min = (tn * tn) % MOD + (tn - 1) * (tn - 1) % MOD;
            _min = (_min * tn) % MOD;

    }
