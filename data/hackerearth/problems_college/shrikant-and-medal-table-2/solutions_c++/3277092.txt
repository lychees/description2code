#include <bits/stdc++.h>
using namespace std;
struct player
{
	string name;
	int G,S,B;
};

bool cmp(player a, player b) {
	if(a.G > b.G) return 1;
	if(a.G == b.G && a.S > b.S) return 1;
	if(a.G == b.G && a.S == b.S && a.B > b.B) return 1;
	if(a.G == b.G && a.S == b.S && a.B == b.B && a.name < b.name) return 1;
	return 0;
}
int main()
{
    int N;
    string a,b,c;
    map<string,int> M;
    cin >> N;
    player P[105];
    int ind = 1;
    for(int i = 0; i < N; i++) {
    	cin >> a >> b >> c;
    	//cout << "strings: " << a << " " << b << " " << c<< "\n";
    	if(M[a]) P[M[a]].G++;
    	else {
    		M[a] = ind++;
    		P[M[a]].name = a;
    		P[M[a]].G = 1;
    		P[M[a]].B = 0;
    		P[M[a]].S = 0;
    	}
    	if(M[b]) P[M[b]].S++;
    	else {
    		M[b] = ind++;
    		P[M[b]].name = b;
    		P[M[b]].S = 1;
    		P[M[b]].G = 0;
    		P[M[b]].B = 0;
    	}
    	if(M[c]) P[M[c]].B++;
    	else {
    		M[c] = ind++;
    		P[M[c]].name = c;
    		P[M[c]].B = 1;
    		P[M[c]].S = 0;
    		P[M[c]].G = 0;
    	}
    }
    sort(P+1,P+ind,cmp);
    for(int i = 1; i < ind; i++) {
    	cout << P[i].name << " " << P[i].G << " " << P[i].S << " " << P[i].B << "\n";
    }
    return 0;
}
