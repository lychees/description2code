#include<bits/stdc++.h>
#define ll long long
using namespace std;
void solve()
{
    int r,c,w;
    cin>>r>>c>>w;
    int res = (c/w)*r + w;
    if(c%w == 0)res--;
    cout<<res<<"\n";
}
int main(){


    int t;
    cin >> t;
     for(int i=1; i<=t; i++)
    {
        cout<<"Case #"<<i<<": ";
        solve();
    }

}
