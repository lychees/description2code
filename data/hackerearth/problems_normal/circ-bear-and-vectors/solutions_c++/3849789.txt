#include <bits/stdc++.h>
using namespace std;
typedef int ll;
ll x[3005],y[3005];
ll gcd(ll a, ll b)
{
	if(a<0)return gcd(-a,b);
	if(b<0)return gcd(a,-b);
	if(a==0)return b;
	if(b==0)return a;
	ll t;
	while(a%b)
	{
		t=a%b;a=b;b=t;
	}
	return b;
}
int main()
{
   // printf("Hello World!\n");
   int n,i,j,k;
   ll u,v,g;
   scanf("%d",&n);
   
   long long ans = 0;
   map< pair<ll,ll> ,ll> fir,sec;
   for(i=1;i<=n;i++)
   {
   	scanf("%d%d",&x[i],&y[i]);
   	if(x[i]==0)fir[make_pair(0,1)]++;
   	else if(y[i]==0)fir[make_pair(1,0)]++;
   	else {
   		 g = gcd(x[i],y[i]);
    	u= x[i];
   	v = y[i];
   	u/=g;
   	v/=g;
   	if(u<0){u=-u;v=-v;}
   	fir[make_pair(u,v)]++;
   	
   	}
   }
   pair<ll,ll> cur;
   map< pair<ll,ll> , ll>::iterator I;
   for(I=fir.begin();I!=fir.end();I++)
   {
   	cur = I->first;
   	//printf("%lld %lld comes %lld times\n",cur.first,cur.second,I->second);
   }
   for(i=1;i<=n;i++)
   for(j=1;j<=n;j++)
   {
   	if(i==j)continue;
   
   	u= x[i]+x[j];
   	v = y[i] + y[j];
   
   if(u==0 && v==0)continue;
   long long a1,a2,b1,b2;
   a1 = x[i];
   a2 = x[j];
   b1 = y[i];
   b2 = y[j];
   int mi = 0;
   if(a1*a1 + a1*a2 + b1*b1 + b1*b2==0){
  mi  = -2;
   }
   if(u==0)cur = make_pair(1,0);
   else if(v==0)cur = make_pair(0,1);
   else {
   		g=gcd(u,v);
   	u/=g;
   	v/=g;
   	u=-u;
   	if(v<0){u=-u;v=-v;}
   	cur = make_pair(v,u);
   }
   //printf("searching for %d %d %d %d\n",i,j,cur.first,cur.second);
  // printf("searching for %lld %lld\n",cur.first,cur.second);
   if(fir.find(cur)!=fir.end()){
   	ans+= fir[cur]+mi;
   	//printf("searching for %d %d %d %d\n",i,j,cur.first,cur.second);
   }
   	
   }
   printf("%lld\n",ans);
    return 0;
}
