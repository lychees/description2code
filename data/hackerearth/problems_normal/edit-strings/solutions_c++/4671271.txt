void count_replace(char str[],int l)
{
	int count =0,i=0;
	
	while(str[i])
	{
		if(str[i]=='.' &&i <l-1 )
		{
			if(str[i]==str[i+1] )
			{
				count++;
			}
		}
		i++;
	}
	printf("%d\n",count);
}
