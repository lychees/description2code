#!/usr/bin/env python

from sys import stdin, stderr


def main():
    TC = int(stdin.readline().strip())
    for tc in xrange(TC):
        N = int(stdin.readline().strip())
        odd = None
        even = None
        for i in xrange(N):
            r, c, x = map(int, stdin.readline().split())
            x = x & 1
            if ((r+c) & 1) == 1:
                if odd is None:
                    odd = x
                elif odd != x:
                    odd = -1
            else:
                if even is None:
                    even = x
                elif even != x:
                    even = -1
        if odd < 0 or even < 0:
            print 'No'
        elif odd is not None and even is not None and odd == even:
            print 'No'
        else:
            print 'Yes'
    return 0


if __name__ == '__main__':
    main()
