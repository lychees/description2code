#include<bits/stdc++.h>
using namespace std;
int main()
{
  int t,n,count1,i,l; 
  cin>>t;
  while(t--)
  {
  	count1=0;
  	string str1,str2; 
	cin>>str1>>str2; 
	cin>>n;
  	if(str1.length() != str2.length())
    cout<<"No";
  	else 
	{
		l=str1.size();
    	for (i = 0; i < l; i++)
    	{
        	if(str1[i] != str2[i])
			 count1++;
    	}
    	if((n-count1)>=0&&(n-count1)%2==0 )
    	 	cout<<"Yes";
   		else
    		cout<<"No";
  	}
  cout<<endl;
  }
  return 0;
}
