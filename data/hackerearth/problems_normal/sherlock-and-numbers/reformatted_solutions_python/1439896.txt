for _ in range(input()):
    n, k, p = map(int, raw_input().split())
    if (k == 0):
        r = []
    else:
        r = map(int, raw_input().split())
    a = p
    for i in range(k):
        if (r[i] <= a):
            # print i,a
            a += 1
    if (a < n):
        print a
    else:
        print -1
