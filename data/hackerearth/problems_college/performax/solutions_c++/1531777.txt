#include <iostream>
#include<cstdio>
using namespace std;

int main()
{
    long long int t,n,d,a[100004],b,c,i,temp;
    scanf("%lld",&n);
    
    	scanf("%lld",&d);
    	for(i=0;i<n;i++)
    	{
    		scanf("%lld",&a[i]);
    	}
    	temp=0;
    	for(i=1;i<n;i++)
    	{

    		{

    		}
    	}

}
