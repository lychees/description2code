#include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>
#include <functional>
#include <algorithm>

using namespace std;

int slist[100005];

int main()
{
	int n,temp,count;
	cin >> n;

	memset(slist,0,100005);
	temp = 0;
	count =1;
	
	for(int i =0; i<n; i++)
	{
		cin >> slist[i];
		temp = temp + slist[i];
	}
	
	if(temp % 4 ==0)
	{
		count = temp/4;
	}
	else
	{
		count = temp/4 + (temp%4) ;
	}
	
	cout << count << endl;

	return 0;
}
