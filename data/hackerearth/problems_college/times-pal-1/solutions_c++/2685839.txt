#include<cstdio>
#include<iostream>
#include<cstring>
using namespace std;
typedef int w;
typedef long long l;
string s1,s2;

bool check(w pos){
    string c="";
    for(int i=0;i<s1.size();++i){
        if(pos==i) c=c+s2;
        c=c+s1[i];
    }
    if(pos==s1.size()) c=c+s2;
    w st=0;
    //cout<<c<<endl;
    w en=c.size()-1;
    while(st<en){
        if(c[st]!=c[en]) return false;
        ++st;
        --en;
    }
    return true;
}

void solve(){
    cin>>s1>>s2;
    w res=0;
    for(int i=0;i<=s1.size();++i) if(check(i)) ++res;
    cout<<res<<endl;
}

int main(){
    ios_base::sync_with_stdio(false);
    //freopen("in","r",stdin);
    w T;
    cin>>T;
    while(T--) solve();
}
