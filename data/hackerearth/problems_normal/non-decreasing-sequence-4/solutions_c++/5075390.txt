int t,n,l,ans,lis_len,ind,maxLen;
int arr[1000007];
int lis[1000007];

void print();

int main(){
    //fr("seq.txt");
    //fw("out.txt");

    cin >> t;
    while(t--){
        cin >> n;
        for(int i=0;i<n;i++)
            cin >> arr[i];
        cin >> l;

        ans = INT_MAX;
        maxLen = 0;
        for(int i=0;i<n;i++){
            ind = lower_bound(lis,lis+maxLen,arr[i])-lis;
            lis[ind] = arr[i];
            maxLen = max(maxLen,ind+1);

            if(ind>=(l-1)) ans = min(ans,arr[i]);
            //print();
        }
        if(ans!=INT_MAX) cout << ans << endl;
        else cout << -1 << endl;
    }

    return 0;
}
