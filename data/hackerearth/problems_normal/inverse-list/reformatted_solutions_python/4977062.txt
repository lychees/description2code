'''Inverse List'''
T = input()
while (T):
    n = input()
    lst = [int(x) for x in raw_input().split()]
    inverse_lst = []
    for i in range(1, n+1):
        for j in range(n):
            if lst[j] == i:
                inverse_lst.append(j+1)
                break
    if lst == inverse_lst:
        print 'inverse'
    else:
        print 'not inverse'

    T -= 1
