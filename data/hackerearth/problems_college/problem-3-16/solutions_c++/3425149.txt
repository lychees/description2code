#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,k;
        int hash[11]={0};
        cin>>n>>k;
        int i,j;
        int max=-9999999;
        for(i=1;i<=n;i++)
        {
            int test;
            cin>>test;
            hash[test]++;
            if(hash[test]>max)
            {
                max=hash[test];
            }
        }
        cout<<n-max<<endl;
    }
    return 0;
}
