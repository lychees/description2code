#include<bits/stdc++.h>
using namespace std;
int str(string a,string b)
{
    int j,k;
    if(a.length()>b.length())
        return 0;
    int n=a.length();
    int n1=b.length();
    for(int i=0;i<=n1-n;i++)
    {
        for(j=i,k=0;k<n;j++,k++)
        {if(b[j]!=a[k])
                break;}
        if(k==n)
               return 1;
    }
    return 0;
}
int main()

{

            for(k=1;k<=n;k++)
