#include <bits/stdc++.h>
using namespace std;
vector<bool> p(1000001,1);
vector<long long> v,x,y,f,g;
void seive()
{
	long long i,j;
	p[0]=0;
	p[1]=0;
	for(i=2;i<=1000000;i++)
	{
		if(p[i]==1)
		{
			for(j=i*i;j<=1000000;j+=i)
			{
				p[j]=0;
			}
			v.push_back(i);
		}
	}
}
void pre()
{
	long long i,j;
	x.resize(1000001);
	f.resize(1000001);
	x[0]=x[1]=0;
	f[0]=f[1]=0;
	for(i=2;i<100001;i++)
	{
		if(x[i]==0)
		{
			for(j=1;j*i<1000001;j++)
			{
				if(x[i*j]==0)
					x[i*j]=i;
			}
		}
	}
	for(i=1;i<1000001;i++)
	{
		f[i]=f[i-1]+x[i];
	}
}
void pre2()

	{
		for(j=2;j*i<1000001;j++)
		{
				y[i*j]=y[i*j]+i;
		}
	}
	for(i=1;i<=1000000;i++)
	{
		g[i]=g[i-1]+y[i];
	}
}
int main()
