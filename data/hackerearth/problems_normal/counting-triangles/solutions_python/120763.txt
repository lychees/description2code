num=int(raw_input())
l=[]
for i in range(num):
  s=raw_input()
  s=s.split(" ",len(s))
  fi=s[0]
  se=s[1]
  th=s[2]
  samp=[]
  samp.append(fi)
  samp.append(se)
  samp.append(th)
  samp.sort()
  n=""
  for j in samp:
    n=n+j
  l.append(int(n))
l.sort()
unique=0
for i in range(1,len(l)-1):
  if l[i]!=l[i-1] and l[i]!=l[i+1]:
    unique=unique+1
if l[0]!=l[1]:
  unique=unique+1
if l[num-1]!=l[num-2]:
  unique=unique+1
print unique
