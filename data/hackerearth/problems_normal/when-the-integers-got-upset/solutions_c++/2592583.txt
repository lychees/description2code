{
	cin>>n;
	for (int i=0;i<n;i++)
		cin>>a[i];
	for (int i=0;i<n;i++)
		cin>>p[i];
	
	for (int i=0;i<=n;i++)
		for (int mask=0;mask<(1<<n);mask++)
			for (int l1=0;l1<n;l1++)
				for (int l2=0;l2<n;l2++)
					dp[i][mask][l1][l2]=1e9;
	
	for (int i=0;i<n;i++)
		for (int j=0;j<n;j++)
			if (i!=j)
				dp[2][(1<<i)+(1<<j)][i][j]=0;
	
	for (int i=2;i<n;i++)
		for (int mask=0;mask<(1<<n);mask++)
			for (int l1=0;l1<n;l1++)
				for (int l2=0;l2<n;l2++)
					if (dp[i][mask][l1][l2]<1e7)
						for (int ad=0;ad<n;ad++)
						{
							if (mask&(1<<ad))
								continue;
							int nmask=(mask|(1<<ad));
							int ncost=dp[i][mask][l1][l2]+p[i]*(a[l1]^a[l2]^a[ad]);
							if (dp[i+1][nmask][l2][ad]>ncost)
								dp[i+1][nmask][l2][ad]=ncost;
						}
	
	int ans=1e9;
	for (int i=0;i<n;i++)
		for (int j=0;j<n;j++)	
			ans=min(ans,dp[n][(1<<n)-1][i][j]);
	cout<<ans<<endl;
