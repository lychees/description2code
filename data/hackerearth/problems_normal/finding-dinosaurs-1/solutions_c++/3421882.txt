using namespace std;
#define MOD 1000000007
map<int,string> ma;
int main()
{
	std::ios::sync_with_stdio(false);cin.tie(0);
	int i,j,n,q,k,l;
	string s;
	int t,sum;
	cin>>n>>k>>l>>q;
	for(i=0;i<n;i++)
	{
		cin>>s;
		sum=0;
		for(j=0;j<k;j++)
		{
			cin>>t;
			sum=(sum<<4)+t;
		}
		//cout<<sum<<endl;
		ma[sum]=s;
	}
	while(q--)
	{
		sum=0;
		for(j=0;j<k;j++)
		{
			cin>>t;
			sum=(sum<<4)+t;
		}
		//cout<<sum<<endl;
		if(ma.find(sum)==ma.end())
			cout<<"You cant fool me :P\n";
		else
			cout<<ma[sum]<<"\n";
	}
//	system("pause");
	return 0;
}
