import sys

t=input()

while t>0:
    n=input()
    l=[]
    while n>0:
        l.append(sys.stdin.readline().strip())
        n=n-1
    cc=0
    mc=0
    mtx=""
    for i in xrange(97,123):
        cc=0
        for j in l:
            if chr(i) in j:
                cc=cc+1
        if cc>mc:
            mc=cc
            mtx=chr(i)
    print mtx
    t=t-1
