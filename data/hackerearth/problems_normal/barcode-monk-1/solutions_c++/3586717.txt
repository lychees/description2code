#include <cstdio>
#include <vector>
#include <cstring>
#include <algorithm>
using namespace std;

//--------------------
#define MAXN 1010
#define MOD 1000000007

int n, k;
int w[MAXN];
int dp[MAXN];
vector<int> pos;

bool mark[MAXN];
//--------------------

int solve(int x){
    
    if(x > n) return 1;
    if(dp[x] != -1) return dp[x];
    
    dp[x] = solve(x+1); // deixando o atual em branco
    dp[x] %= MOD;
    
    for(int i = 0;i < (int)pos.size();i++){
        if(x + pos[i] > n+1) break;
        dp[x] += solve(x + pos[i] + 1);
        dp[x] %= MOD;
    }
    
    return dp[x];
}

int main(){
    
    scanf("%d %d", &n, &k);
    for(int i = 1;i <= k;i++) scanf("%d", &w[i]);
    
    mark[0] = true;
    for(int i = 1;i <= k;i++)
    for(int j = 0;j <= n-w[i];j++)
        if(mark[j]) mark[j+w[i]] = true;
    
    for(int i = 1;i <= n;i++) if(mark[i]) pos.push_back(i);
    
    memset(dp, -1, sizeof dp);
    
    printf("%d\n", solve(1));
    
    return 0;
}
