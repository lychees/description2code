#include <iostream>
#include <string>
using namespace std;

int main()
{
    string input1;
    string input2;
    string a,b;
    int input, counter, sameWord;
    
    cin >> input;
    
    for(int i = 0; i<input; i++)
    {
    	cin >> input1;
    	cin >> input2;
    	sameWord = 0;
    	if(input1.size() > input2.size())
    	{
    		counter = input2.size();
    		a = input2;
    		b = input1;
    	}
    	else
    	{
    		counter = input1.size();
    		a = input1;
    		b = input2;
    	}
    	for(int k = 0; k<counter; k++)
    	{
    		if(a[k] == b[k])
    		{
    			sameWord += 1;
    		}
    	}
    	cout << sameWord << endl;
    	input1.clear();
    	input2.clear();
    }
    return 0;
}
