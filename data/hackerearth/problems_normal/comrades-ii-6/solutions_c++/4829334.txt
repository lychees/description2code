#include <bits/stdc++.h>
using namespace std;
int rec(vector<int> &a,vector<int> &dp,int j)
{
	if(dp[j]!=0) return dp[j];
	if(a[j]==0) return 0;
	dp[j]=1+rec(a,dp,a[j]);
	return dp[j];
}
int main()
{
    long long int t,x,handshakes,fist;
    long long int n;
    cin>>t;
    while(t--)
    {
    	handshakes=0;
    	vector<int> a;
    	cin>>n;
    	vector<int> dp(n+1);
    	a.push_back(-1);
    	for(int i=0;i<n;i++){
    		cin>>x;
    		a.push_back(x);
    	}
    	for(int i=1;i<=n;i++)
    	{
    		handshakes+=rec(a,dp,i);
    	}
    	fist=(n*(n-1))/2-handshakes;
    	cout<<handshakes<<" "<<fist<<endl;
    }
    return 0;
}
