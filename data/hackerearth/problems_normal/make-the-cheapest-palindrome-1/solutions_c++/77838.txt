#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    int t,ac,bc;
    cin>>t;
    while(t--)
    {
    	char str[10001];
    	cin>>str;
    	cin>>ac;
    	cin>>bc;
    	int l=strlen(str);

    	int cost=0,flag=0;

    	for(int i=0;i<l/2;i++)
    	{
    		if(str[i] == '/' && str[l-i-1] == '/')
            {
                if(ac>bc)
                    cost+=bc*2;
                else
                    cost+=ac*2;
            }
    		else if(str[i] == '/')
    		{
    			if(str[l-i-1] == 'a')
    				cost+=ac;
    			else
    				cost+=bc;
    		}
    		else if(str[l-i-1] == '/')
    		{
    			if(str[i] == 'a')
    				cost+=ac;
    			else
    				cost+=bc;
    		}
    		else if(str[i]!= str[l-i-1])
    		{
    			flag=1;
    			break;
    		}
    	}

    	if(flag == 0)
    		cout<<cost<<endl;
    	else
    		cout<<-1<<endl;
    }
    return 0;
}
