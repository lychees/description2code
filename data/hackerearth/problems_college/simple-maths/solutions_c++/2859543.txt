#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    while(t--)
    {
    	long long n,x,y;
    	cin>>n>>x>>y;
    	
    	if(x==0 || y==0) {
    		cout<<-1<<'\n';
    		continue;
    	}
    	long long lcm=x*y/(__gcd(x,y));
    	cout<<n/x+n/y-(n/lcm)<<'\n';
    }
    return 0;
}
