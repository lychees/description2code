#include <stdio.h>
int main()
{
    
    int t,x,p;
    scanf("%d",&t);
    while(t--)
    {
        int co=0;
        scanf("%d%d",&x,&p);
        while(x%p==0)
        {
            co++;
            x=x/p;
            if(x==0)
            break;
        }
        
        printf("%d %d\n",co,x==1?0:x);
    }
    return 0;
}
