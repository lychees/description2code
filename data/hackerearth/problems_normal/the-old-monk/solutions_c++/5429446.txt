#include <iostream>
#include <vector>
using namespace std;

vector<long long int>A;
long long int Bin_Search(long long int low,long long int high,long long int val)
{
    long long int mid;
    while(low<=high)
    {
        mid=(low+high)/2;
        if(A[mid]==val)
            high=mid-1;
        else if(val>A[mid])
            high=mid-1;
        else
            low=mid+1;
    }
    if(val>=A[mid-1])
        return mid-1;
    else
    	return low;
}
int main()
{
    long long int T,i,j,n,num;
    vector<long long int>answer;
    cin>>T;
    for(i=0;i<T;i++)
    {
        long long int Max=0;
        A.clear();
        cin>>n;
        for(j=0;j<n;j++)
        {
            cin>>num;
            A.push_back(num);
        }
        for(j=0;j<n;j++)
        {
            cin>>num;
            long long int pos=Bin_Search(0,j,num);
            if(pos>=0)
            if(Max<j-pos)
                Max=j-pos;
        }
        answer.push_back(Max);
    }
    for(i=0;i<T;i++)
        cout<<answer[i]<<endl;
}
