#include<bits/stdc++.h>
using namespace std;
int arr[100005];

long long int dp[100005],n;

long long solve(int i){

     if(i>=n) return 0;

     if(dp[i]!=-1)
           return dp[i];

    if(i==n-1)
       return dp[i]=arr[i];

    return dp[i]=max(solve(i+1),arr[i]+solve(i+2));

  }

  int main(){

      int t;

      cin>>t;

      while(t--){

         cin>>n;

         memset(dp,-1,sizeof(dp));

        for(int i=0;i<n;++i){

           cin>>arr[i];

        }

        printf("%lld\n",solve(0));

     }

  return 0;
}
