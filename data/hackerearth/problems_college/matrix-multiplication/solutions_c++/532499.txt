#include<iostream>
using namespace std;
#include<stdio.h>
int main()
{
  int t,x1,y1,x2,y2,ctr;
  scanf("%d",&t);
  while(t--)
  {
  	  ctr=0;
  	  scanf("%d %d %d %d",&x1,&y1,&x2,&y2);
      if(y1==x2) 
         ctr++;
      if(y2==x1)
	     ctr++;
      printf("%d\n",ctr);		    
  }
  return 0;
}
