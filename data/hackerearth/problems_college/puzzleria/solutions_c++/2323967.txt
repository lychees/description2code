#include <iostream>
using namespace std;
int fact(int n)
{
	int i,f=1;
	for(i=1;i<=n;i++)
	f=f*i;
	return f;
}
int com(int r,int n)
{
	int res=fact(n)/(fact(n-r)*fact(r));
	return res;
}

int main()
{
    int t,a,b,c,v,f,res;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b;
    	c=com(a,7);
    	v=com(b,5);
    	f=fact(a+b);
    	res=c*v*f;
    	cout<<res<<endl;
    }
    return 0;
}
