#include<iostream>

using namespace std;

int ascii(char letter)
{
    return (int)letter;
}

int main()
{
    string s;
    cin>>s;
    int i,j;
    long long int sum=0;
    j = s.length();
    int k[j];
    for(i=0;i<j;i++)
    {
        if(ascii(s[i])==97)
            k[i] = 1;
        else
            k[i] = ascii(s[i]) - 96 ;
    }
    for(i=0;i<j;i++)
    {
        sum+=k[i];
    }
    cout<<sum;
}
