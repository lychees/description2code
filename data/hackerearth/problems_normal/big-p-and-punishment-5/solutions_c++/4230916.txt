#include<bits/stdc++.h>
#define pii pair<long long,long long>
#define fi first
#define se second
#define mp make_pair
#define vpi vector< pii >
#define pb push_back
#define ll long long int
#define mod 1000000007
#define gcd(a,b) __gcd(a,b)
#define sf(n) scanf("%lld",&(n))
#define pf(n) printf("%lld\n",(n))
#define setbits(n) __builtin_popcount((n))
#define pq priority_queue
using namespace std;
int co=0;
bool visited[100005];
vector<int>v[100005];
void ini()
{
	memset(visited,0,sizeof(visited));
	for(int i=0;i<=100004;i++)
	v[i].clear();
	co=0;
}
void dfs(int node)
{
	visited[node]=true;
	co++;
	for(int i=0;i<v[node].size();i++)
	{
		if(!visited[v[node][i]])
			dfs(v[node][i]);
	}
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		ini();
		int n,f,s;
		cin>>n>>f>>s;
		for(int i=0;i<f;i++)
		{
			int a,b;
			cin>>a>>b;
			v[a].pb(b);
		}
		for(int i=0;i<s;i++)
		{
			int a;
			cin>>a;
			if(!visited[a])
				dfs(a);
		}
		cout<<co<<"\n";
	}
}
