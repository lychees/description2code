d = {}
n = input()
for _ in xrange(n):
    i, j = raw_input().split("-")
    i = int(i.replace(":", ""))
    j = int(j.replace(":", ""))
    d[i] = j
d = sorted(d.items(), key=lambda x: x[1])
flag = 0
for i in xrange(1, n):
    if d[i][0] < d[i-1][1]:
        flag = 1
        break
print "Will need a moderator!" if flag == 1 else "Who needs a moderator?"
