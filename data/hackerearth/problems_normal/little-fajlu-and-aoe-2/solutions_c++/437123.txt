#include <iostream>
#include <bits/stdc++.h>
#define ll long long
#define pb push_back
#define mp make_pair
#define vi vector <int>
#define pii pair <int, int>

using namespace std;

int qq = 1;
ll s, p, c;
ll dp[1604][1604];

ll fun (ll f, ll g)
{
	
	if (dp[f][g] != -1) return dp[f][g];
	//int x = 0; int y = 0; int z = 0;
	ll x = 0; ll y = 0; ll z = 0;
	if (f >= 4 * qq) x = s + fun (f - 4 * qq, g);
	if (f >= 5 * qq && g >= 2 * qq) y = p + fun (f - 5 * qq, g - 2 * qq);
	if (f >= 2 * qq && g >= 4 * qq) z = c + fun (f - 2 * qq, g - 4 * qq);
	
	dp[f][g] = max (x, max (y, z));
	return dp[f][g];
}
int main()
{
	int n, t, x;
	int mx = 0;
	cin >> t;
	while (t--) {
		//memset (dp, -1, sizeof dp);
		
		for (int i = 0; i <= 1600; i++) {
			for (int j = 0; j <= 1600; j++) {
				dp[i][j] = -1;
			}
		}
		
		ll f, g, pw;
		cin >> f >> g >> s >> p >> c >> pw;
		f /= 25;
		g /= 25;
		ll q = fun (f, g);
		
		if (pw >= q) {
			puts ("-1");
		} else {
			q-=pw;
			cout << q<< endl;	
			//printf ("%d\n", q - pw);
		}
	}
	return 0;
}
