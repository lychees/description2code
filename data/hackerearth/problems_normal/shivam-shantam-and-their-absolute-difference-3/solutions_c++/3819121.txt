#include <bits/stdc++.h>
using namespace std;

# define mod 1000000007

long long int dp1[1000001],dp2[1000001];

int main() {
	int n;
	scanf("%d",&n);
	
	int a[n],b[n],i;
	
	for(i=0;i<n;i++) {
		scanf("%d",&a[i]);
	}
	
	for(i=0;i<n;i++) {
		scanf("%d",&b[i]);
		}
		
	//	sort(a,a+n);
	//	sort(b,b+n);
		
		dp1[0]=1;
		dp2[0]=1;
		int j;
		
		for(i=0;i<n;i++) {
			for(j=400000;j>=0;j--) 
				if(dp1[j]>0&&j+a[i]<=400000) {
				dp1[a[i]+j]+=dp1[j];
				dp1[a[i]+j]%=mod;}
			
		}
		
		
		for(i=0;i<n;i++) {
			for(j=400000;j>=0;j--) {
				if(dp2[j]>0&&j+b[i]<=400000) {
				dp2[b[i]+j]+=dp2[j];
				dp2[b[i]+j]%=mod;
				}
			}
		}
		
		int q;
		scanf("%d",&q);
		long long int ans=0,k;
		
		
		
		for(i=1;i<=400000;i++) dp2[i]+=dp2[i-1];
		
		for(i=0;i<=400000;i++) {
			if(dp1[i]>0) {
		//		printf("%d %lld\n",i,dp1[i]);
		/*		int temp=i-q;
				if(temp<=0) temp=0;
				while(temp<=i) {
					ans=(ans+dp1[i]*dp2[temp])%mod;
					temp++;
				}
				
				temp=i+1;
				while(temp<=i+q&&temp<=400000) {
					ans=(ans+dp1[i]*dp2[temp])%mod;
					temp++;
				}
		*/
					k=dp2[min(i+q,400000)];
					if(i-q-1>=0) k=k-dp2[i-q-1]; 
					k=k%mod;
					ans=(ans+dp1[i]*k)%mod;
				
			}
		}
		printf("%lld",ans);
	
	return 0;
}
