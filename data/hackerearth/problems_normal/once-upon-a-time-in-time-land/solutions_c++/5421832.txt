#include <iostream>
using namespace std;

int main()
{
    int t,n,k;
    cin >> t;
    while(t--){
    	cin >> n >> k;
    	long long a[n];
    	for(int i = 0; i < n; i++){
    		cin >> a[i];
    	}
    	long long sum = 0, maxsum = 0;
    	pair<long long, long long> dp[n];
    	maxsum = a[0];
    	if(maxsum < 0){
    		maxsum = 0;
    	}
    	dp[0].first = a[0];
    	dp[0].second = maxsum;
    	for(int i = 1; i < n; i++){
    		sum = a[i];
    		if(i-k-1 >= 0){
    			sum += dp[i-k-1].second;
    		}
    		dp[i].first = sum;
    		if(sum > maxsum){
    			maxsum = sum;
    		}
    		dp[i].second = maxsum;
    	}
    	cout << dp[n-1].second << endl;
    }
    return 0;
}
