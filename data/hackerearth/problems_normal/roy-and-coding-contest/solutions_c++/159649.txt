#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,n,m,time;
	scanf("%d",&t);
	while(t--)
	{
		time=0;
		scanf("%d%d",&n,&m);
		if(n==1)
		{
			time=0;
		}
		else if(n<m)
		{
			time=ceil((log2(n)))+1;			
		}
		else
		{
			time=ceil(log2(m))+1;
			n=n-pow(2,time-1);
			time=time+ceil(((double)n)/m);
		}
		printf("%d\n",time);
	}
	return 0;
}
