#include <bits/stdc++.h>
using namespace std;
typedef struct sats{
    int odd;
    int even;
}sats;
int main()
{
    int t;  cin>>t; while(t--)
    {
        long int n,a[100001],ans=0;
        sats arr[100001];
        cin>>n;
        for(int i=1;i<=n;i++) cin>>a[i];
        int j=1,e=0,o=0;
        for(int i=1;i<=n;i++)
        {
            if(a[i]%2==0)
            {
                e++;
                arr[i].even=e;
                arr[i].odd=o;
            }
            else
            {
                o++;
                arr[i].odd=o;
                arr[i].even=e;
            }
        }
        /*for(int i=1;i<=n;i++)
        {
            cout<<arr[i].even<<" ";
        }
        cout<<endl;*/

        /*for(int i=1;i<=n;i++)
        {
            cout<<arr[i].odd<<" ";
        }
        cout<<endl;*/

        for(int i=n;i>=1;i--)
        {
            if(a[i]%2==0)
                ans+=arr[i].odd;
            else ans+=arr[i].even;
        }
        cout<<ans<<endl;

    }
    return 0;
}
