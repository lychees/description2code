#include<iostream>
#include<math.h>
#include<string.h>

using namespace std;

int main ()
{
	int test, len, temp, d;
	cin >> test;
	char str[ 1000 ], rev [ 1000 ];
	while ( test-- )
	{
		cin >> str;
		len = strlen( str );
		d = 0;
		for ( int c = len - 1; c >= 0; c--, d++)
      		rev [ d ] = str [ c ];
		rev[ d ] = '\0';
		temp = 0;
		for ( int i = 1; i < len; i++ )
		{
			if ( abs ( str[ i ] - str[ i - 1 ] ) != abs ( rev[ i ] - rev[ i - 1 ] ) )
			{
				temp = 1;
				break;
			}
		}
		if ( temp == 0 )
			cout << "Funny" << endl;
		else
			cout << "Not Funny" << endl;
	}
	return 0;
}
