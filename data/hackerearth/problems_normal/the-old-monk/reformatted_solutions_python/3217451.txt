from bisect import bisect
for _ in range(int(raw_input())):
    n = int(raw_input())
    a = map(int, raw_input().split())[::-1]
    b = map(int, raw_input().split())[::-1]
    c = []
    for i in range(n):
        x = bisect(a, b[i])
        if x > 0:
            c.append(x - i-1)
    if len(c):
        print max(c)
    else:
        print 0
