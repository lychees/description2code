#include<iostream>
#include<vector>
#include<algorithm>
#define ll long 
using namespace std;
int main()
{
    ll i,k,m,n;
    scanf("%ld",&n);
   // cin>>n;
    vector<pair<ll,ll>>a;
    //pair<ll,ll>p;
    for(i=0;i<n;i++)
    {
    	scanf("%ld%ld",&m,&k);
        //cin>>m>>k;
        //make_pair p(m,-k);
        a.push_back(make_pair(m,-k));
    }
    sort(a.begin(),a.end());
    for(auto pos=a.begin();pos!=a.end();++pos)
    {
        printf("%ld %ld\n",pos->first,-(pos->second));
    }
    return 0;
}
