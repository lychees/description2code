n = int(raw_input())
dom = map(int, raw_input().split())
brian = map(int, raw_input().split())

max_speed = []


def maxSpeed(arr):
    max_speed = []
    for i in xrange(n):
        if i != n - 1:
            max_speed.append(arr[i+1] - arr[i])
    max_speed = map(abs, max_speed)
    return max(max_speed)


domMax = maxSpeed(dom)
brianMax = maxSpeed(brian)


if domMax > brianMax:
    print 'Dom'
    print domMax
elif brianMax > domMax:
    print 'Brian'
    print brianMax
elif domMax == brianMax:
    print 'Tie'
    print domMax
