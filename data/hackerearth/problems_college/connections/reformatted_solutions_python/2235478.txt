def find(x, y):
    global graph
    if x in graph[y]:
        return True
    else:
        for i in graph[y]:
            if find(x, i):
                return True
        return False


graph = {}
n = input()
graph = {i: [] for i in range(1, n+1)}
c = input()
for _ in range(c):
    a, b = map(int, raw_input().split())
    graph[a].append(b)
    graph[b].append(a)
q = input()
for _ in range(q):
    a, b = map(int, raw_input().split())
    if b in graph[a]:
        print'Direct Connection'
    else:
        if find(a, b):
            print 'Indirect Connection'
        else:
            print 'No Connection'
