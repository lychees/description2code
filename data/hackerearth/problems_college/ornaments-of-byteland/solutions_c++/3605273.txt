int main(){
	int n;
	cin >> n;
	map<char, int>mm;
	for (int i = 0; i < n; ++i){
		string s;
		cin >> s;
		set<char>ss;
		for (int j = 0; j < s.length(); ++j){
			ss.insert(s[j]);
		}
		tr(ss, it){
			char ch = *it;
			++mm[ch];
		}
	}
	int ans = 0;
	tr(mm, it){
		ans += it->second == n;
	}
	out(ans); el;
	return 0;
}
