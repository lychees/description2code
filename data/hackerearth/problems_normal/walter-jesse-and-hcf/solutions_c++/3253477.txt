#include <bits/stdc++.h>
using namespace std;

#define MOD 1000000007LL
#define ll long long

int main()
{
	cin.tie(0);
	ios::sync_with_stdio(false);
	int d;
	ll x,y,k;
	cin >> d;
	while(d--)
	{
		cin >> x >> y >> k;
		x = __gcd(x,y);
		vector<ll> A, B;
		for(ll i=1;i*i<=x;++i)
			if(x%i==0)
			{
				A.push_back(i);
				if(i!=x/i)
					B.push_back(x/i);
			}
		while(B.size()!=0)
		{
			A.push_back(B.back());
			B.pop_back();
		}
		reverse(A.begin(),A.end());
		if(A.size()>=k)
			printf("%lld\n",A[k-1]);
		else
			printf("No crymeth today\n");
	}
	return 0;
}
