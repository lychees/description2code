#include<bits/stdc++.h>
using namespace std;
typedef  long long int llu;
#define od 1000000007

llu odin(int x, llu y, llu m)
{
    if (y == 0)
        return 1;
    llu p = odin(x, y/2, m) % m;
    p = (p * p) % m;
    return (y%2 == 0)? p : (x * p) % m;
}



int main()
{
	llu t,n,k,i,ans,ex;
	llu j;
	scanf("%llu",&t);
	while(t--)
	{
		scanf("%llu%llu",&n,&k);
		for(i=0;i<n;i++)
		{
			scanf("%llu",&ex);
			k-=ex;
		}
	if(k<0)
	{cout<<"0\n";
	continue;}
	if(k==0)
	{cout<<"1\n";
	continue;}
	
		
		i=n-1;
		n=n+k-1;
		
	ans=1;
	
		//	cout<<n<<" "<<i<<" "<<" "<<endl;
		for(j=n-i+1;j<=n;j++)
		{
			ans=(ans*j)%od;
		}
		for(j=1;j<=i;j++)
		{ans=(ans*odin(j,od-2,od))%1000000007;}
			
			
		
			ans=(ans)%od;
			printf("%llu\n",ans);
		
		
	}
	return 0;
}
