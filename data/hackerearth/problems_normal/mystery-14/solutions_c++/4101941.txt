#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int a[n];
    	for(int i=0;i<n;i++)
    		cin>>a[i];
    	int sum=0;
    	for(int i=0;i<n;i++)
    	{
    		if(a[i]%2==1)
    			sum+=a[i];
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
