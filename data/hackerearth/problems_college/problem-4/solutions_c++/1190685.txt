#include <iostream>
#include <algorithm> 
using namespace std;

int a[30];
int main()
{
    string s;
    cin >> s;
    int len = s.length();
    
    
    for(int i =0;i<len ;i++){
    	a[s[i]-'a']++;
    }
    char ch;
    for(int i=0;i< 27 ; i++) {
    	while(a[i]){
    		ch = 'a'+i;
    		cout<<ch;
    		a[i]--;
    	}
    }
    
    
    return 0;
}
