#include <bits/stdc++.h>
using namespace std;

#define MAXN 100000

bool win[MAXN];
int m, n;
int a[MAXN];

inline void solve(){
	int i, j;
	scanf("%d%d", &m, &n);
	for(i = 0; i < n; i++){
		scanf("%d", &a[i]);
	}
	sort(a, a + n);
	win[0] = false;
	for(i = 1; i <= m; i++){
		win[i] = false;
		for(j = 0; j < n; j++){
			if(a[j] > i){
				break;
			}
			if(win[i - a[j]] == false){
				win[i] = true;
				break;
			}
		}
	}
	if(win[m] == true){
		printf("Little Deepu\n");
	}
	else{
		printf("Kate\n");
	}
}

int main()
{
    int t;
    cin>>t;
    while(t--){
    	solve();
    }
    return 0;
}
