int tree[26][100003];

void update(int idx,char c,int val,int n)
{
    c-='a';
    for(int i=idx;i<=n;i+= (i&(-i)))
        tree[c][i]+=val;
}
int query(int idx,char c,int n)
{
    int sum=0;
    c-='a';
    for(int i=idx;i>0;i-= (i&(-i)))
        sum+=tree[c][i];
    return sum;
}

int main(void)
{
  int n,q;
  cin>>n>>q;

  char st[n+1];
  scanf("%s",st);

  for(int i=0;i<n;i++)
    update(i+1,st[i],1,n);

  while(q--)
  {
      int type;
      scan(type);

      if(type==1) //update
      {
          int idx;
          char c,now;
          scanf("%d %c",&idx,&c);
          now=st[idx-1];
          update(idx,now,-1,n);
          update(idx,c,1,n);
          st[idx-1]=c;
      }
      else //query
      {
          int l,r,odd=0,even=0;
          scan(l);scan(r);
          for(int i=0;i<26;i++)
          {
              int freq = query(r,i+'a',n)-query(l-1,i+'a',n);
              if(!freq)continue;
              if(freq%2)odd++;
              else even++;
          }
          if((r-l+1)%2)
          {
              if(odd==1)printf("yes\n");
              else      printf("no\n");
          }
          else
          {
              if(odd==0)printf("yes\n");
              else      printf("no\n");
          }
      }

  }

  return 0;
}
