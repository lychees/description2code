#include <iostream>
#include <stdio.h>
#include<string>
using namespace std;

int main()
{
	long Numbers;

	long long Sum;
	long *myArray = new long[1000000];

	cin >> Numbers >> Sum;


	for (long i = 0; i < Numbers; i++)
	{
		cin >> myArray[i];
	}

	bool BinMax[2000000] = { 0 };
	long long Temp=0;
	bool flag = false;
	for (long i = 0; i < Numbers; i++)
	{
		Temp = Sum - myArray[i];
		if (Temp >= 0 && BinMax[Temp] == 1)
		{
			flag = true; break;
		}
		BinMax[myArray[i]] = 1;
	}
	(flag == true) ? cout << "YES" : cout << "NO";
	return 0;
}
