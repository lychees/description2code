#include<iostream>
#include<cstdio>
#include<cmath>
#include<cstring>
#include<stdlib.h>
#include<algorithm>
using namespace std;
#define mod 1000000007
#define inf 2000000000
int n;
int low_limit = -inf;
int up_limit = inf;

int main()
{
	int m, i, x = 0, ans;
	scanf("%d", &n);
	for (i = 0; i<n; i++)
	{
		scanf("%d", &m);
		if (i % 2 == 0) 	// new max bound
		{
			x += m;
			if (up_limit > x)
				up_limit = x;

			x += m;
		}
		else 			// new min bound
		{
			x -= m;
			if (low_limit < x)
				low_limit = x;

			x -= m;
		}
	}

	ans = up_limit - low_limit + 1;
	if (ans>0)
		cout << ans << endl;
	else
		printf("no solution\n");

	return 0;
}
