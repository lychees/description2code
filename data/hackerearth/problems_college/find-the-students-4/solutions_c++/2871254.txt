#include <iostream>
using namespace std;

int main()
{
    int n,i;
    cin>>n;
    int a[n];
    for(i=0;i<n;i++)
    	cin>>a[i];
    while(1)
    {
    	int cnt=0;
    	for(i=0;i<n;i++)
    	{
    		if(a[i]>0)
    			cnt++;
    		a[i]-=2;
    	}
    	if(cnt<1)
    		break;
    	cout<<cnt<<"\n";
    }
    return 0;
}
