#include <bits/stdc++.h>
 
using namespace std;
 
vector <int> v[100010];
 
void dfs (int x) {
	
	if (x == 0)	return;
	cout << x << " ";
	if (v[x].size() >= 1)
		dfs (v[x][0]);
	if (v[x].size() >= 2)
		dfs (v[x][1]);
}
 
int main()
{
	int t;
	cin >> t;
	while (t--) {
		int n;
		cin >> n;
		
		int a, x, y;
		for (int i = 0; i < 100010; i++)
			v[i].clear();
		for (int i = 0; i < n; i++) {
			cin >> a >> x >> y;
			if (x != 0)
				v[a].push_back (x);
			if (y != 0)
				v[a].push_back (y);
		}
		
		dfs (1);
		cout << endl;
	}
	return 0;
}
