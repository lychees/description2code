#include <bits/stdc++.h>
using namespace std;

int a[1000004];
int main()
{
	long long int n,i,ans=0,sq;
    scanf("%lld",&n);
    for(i=0;i<n;i++){
    	scanf("%lld",&a[i]);
    }
    sort(a,a+n);
    //for(i=0;i<n;i++)
    //cout<<a[i]<<" ";
    //cout<<endl;
    sq=1;
    for(i=0;i<n;){
    	//cout<<ans<<endl;
    	if(a[i]>(sq*sq))
    	{
    		ans+=(sq*sq);
    		sq++;
    	}
    	else{
    	ans+=abs((sq*sq)-a[i]);
    	i++;
    	sq++;
    	}
    }
    printf("%lld",ans);
    return 0;
}
