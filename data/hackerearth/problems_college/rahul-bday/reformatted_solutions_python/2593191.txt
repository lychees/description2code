for _ in range(int(raw_input())):
    b, w = (int(i) for i in raw_input().split())
    x, y, z = (int(i) for i in raw_input().split())
    if x == y:
        res = (b+w)*x
    elif x > z+y:
        res = b*(y+z)+w*y
    elif y > z+x:
        res = b*x+w*(z+x)
    else:
        res = b*x+y*w
    print(res)
