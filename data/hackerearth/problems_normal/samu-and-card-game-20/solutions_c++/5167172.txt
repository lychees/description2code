#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin >> t;
    while(t--)
    {
        long long n, m, k;
        cin >> n >> m >> k;
        long long jump = 0;
        long long x = 1, y = 1;
        for(int i = 0; i < k; i++)
        {
            long long a, b;
            cin >> a >> b;
            while(x + a > 0 && y + b > 0 && x + a <= n && y + b <= m)
            {
                x = x + a;
                y = y + b;
                jump++;
            }
        }
        cout << jump <<"\n";
    }
    return 0;
}
