#include <iostream>
using namespace std;

int main()
{
   int t,j ;
   char a[200];
   cin>>t;
   while(t)
   {
   	cin>>a;
   	int num=0,flag=0;
   	j=4;
   	  while(a[j]!='.')
   	{ 
   		if(a[j]==97||a[j]==101||a[j]==105||a[j]==111||a[j]==117)
   		{
   			num++;
   		}
   		else
   		{
   			num++;
   			flag++;
   		}
   	 j++;
   	}
   	cout<<endl<<flag+4<<"/"<<num+8;
   	t--;
   }
    return 0;
}
