#include<bits/stdc++.h>
using namespace std;
static long long a[300001];
int binarySearch(int low,int high,long long n){
   int mid=(low+high)/2;
   if(mid==0){
    return -1;
   } else
   if(a[mid]>n&&a[mid-1]<=n){
    return mid;
   } else
   if(a[mid]<n)
    binarySearch(mid+1,high,n);
   else
    binarySearch(low,mid,n);
}
int main(){
   long long sum=0,n=1,p=0;
   while(sum<=1e17){
    sum+=n*(n+1)*(n+1);
    a[p++]=sum;
    n++;
   }
   int t;
   scanf("%d",&t);
   while(t--){
    long long n;
    scanf("%lld",&n);
    int x=binarySearch(0,p-1,n);
    if(x==-1){
        printf("0\nDraw\n");
        continue;
    } else
    printf("%d\n",x);
    if(x%2==0)
        printf("Ketan\n");
    else
        printf("Dhruv\n");
   }
   return 0;
}
