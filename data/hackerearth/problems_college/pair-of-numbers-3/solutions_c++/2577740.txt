#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int n,i,j,p,q,cnt=0;
    	cin>>n;
    	for(i=1;i<=n;i++)
    	{
    		for(j=i+1;j<=n;j++)
    		{
    			p=i*j;
    			q=i+j;
    			if(p%q==0)
    			cnt++;
    		}
    		
    	}
    	cout<<cnt<<endl;
    }
    return 0;
}
