#include <iostream>
#include<cstring>
using namespace std;
 
int main()
{
int n,count,i;
cin>>n;
count=n-2;
for(i=0;i<n;i++)
{
 cout<<"*";
 }
 cout<<endl;
 while(count>0)
 {
 for(int j=0;j<count;j++)
 {
  cout<<" ";
 }
 cout<<"*"<<endl;
 count--;
 }
 for(i=0;i<n;i++)
{
 cout<<"*";
 }
 
 }
