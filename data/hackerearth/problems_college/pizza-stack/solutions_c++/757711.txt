#include <iostream>
using namespace std;

int g(int a) {return (a*(a+1)+a+2);}

int main()
{
	int t;
	cin>>t;
	long output[1001];
	long result=g(1);
	long coffe[1001];
	int p=1001;
	while(p--)
		coffe[p]=0;
	coffe[0]=1;
	int x=1;
	output[3]=result;
	for(int yu=4;yu<=1000;yu++)
	{
		int label=2;
		for(int i=x-1;i>=0;i--)
		{
			result+=coffe[i]*g(label++);
			result=result%1000000007;
		}
		output[yu]=result;
		x++;
		label=2;
		for(int i=x-1;i>=1;i--)
		{
			coffe[i]+=label*coffe[i-1];
			coffe[i]=coffe[i]%1000000007;
			label++;
		}
	}
	while(t--)
	{
		int a;
		cin>>a;
		if(a<3)
			cout<<a<<endl;
		else
			cout<<output[a]<<endl;
	}
    return 0;
}
