#include<iostream>
#include<cmath>
using namespace std;
int main()
{
    int a,b,i,j,q,c;
    int t;
    cin>>t;
    while(t--)
    {
        c=0;
        cin>>a>>b;
        if(b>a)
        {
            cout<<0<<endl;
        }
        else
        {
            if(a==b)
                cout<<-1<<endl;
            else
            {
                q=a-b;
                j=sqrt(q);
                for(i=1;i<=j;i++)
                {
                	if(i>b)
                	{
                		if(q%i==0)
                		{
                			if(i*i!=q)
                			c=c+2;
                			else
                			c++;
                		}
                	}
                	else
                	{
                		if(q%i==0&&q/i>b)
                		{
                		    c++;
                		}
                	}
                }
                cout<<c<<endl;
            }
        }
    }
    return 0;
}
