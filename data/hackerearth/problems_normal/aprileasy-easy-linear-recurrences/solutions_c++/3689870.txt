void mult(ll A[5][5],ll B[5][5],ll C[5][5])
{
    memset(C,0,sizeof(C)) ;
    for(ll i=0;i<5;++i)for(ll j=0;j<5;++j)for(ll k=0;k<5;++k) C[i][j] = (C[i][j] + ( A[i][k]*B[k][j])%M )%M ;
}
void mult(ll A[5][5],ll B[5],ll C[5])
{

    for(ll i=0;i<5;++i)for(ll k=0;k<5;++k) C[i]=(C[i]+ ((A[i][k]*B[k])%M) )%M ;
}
