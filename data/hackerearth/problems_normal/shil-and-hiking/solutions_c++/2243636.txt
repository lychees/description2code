#include <bits/stdc++.h>
#define endl '\n'

using namespace std;

struct point {
  long long x,y;
};

long long s(point a, point b, point c) {
  return ((a.x-b.x)*(a.y+b.y) + (b.x-c.x)*(b.y+c.y) + (c.x-a.x)*(c.y+a.y));
}

int ans;
int n;
point p[1024];
point best;

int main() {
  ios_base::sync_with_stdio(false);
  cin.tie(NULL);
  //freopen("test.txt","r",stdin);
  int i,j;

  cin>>n;
  for(i=1;i<=n;i++) {
    cin>>p[i].x>>p[i].y;
  }

  for(i=1;i<n;i++) {
    ans++;
    best=p[i+1];
    for(j=i+2;j<=n;j++) {
      if(s(p[i],best,p[j])>=0) ans++,best=p[j];
    }
  }

  cout<<ans<<endl;
  
  return 0;
}
