#include <iostream>
using namespace std;

int main(){
	int N; // No of elements in array
	int Q; //No of queries
	cin>>N>>Q;
	int arr[N];
	for(int i=0; i<N; i++)
		cin>>arr[i];
	while(Q--){
		int queryType;
		cin>>queryType;
		if(queryType == 1){
			int no;
			cin>>no;
			bool success = false;
			for(int i=0; i<N; i++){
				if(arr[i] >= no){
					cout<<i+1<<"\n";
					success = true;
					break;
				}
			}
			if (!success)
				cout<<-1<<"\n";
			
		}else{
			int index, no;
			cin>>index>>no;
			arr[index-1] = no;
		}
	}
	return 0;
}
