#include <math.h>
#include <iostream>
#include <cstdio>
#include <algorithm>
#include <vector>
#include <stdio.h>
#include <limits.h>

using namespace std;

// Time complexity of log(n). Efficient way to calculate fibonacci number :)

#define MOD 1000000009

long long int fib(long long int n) {
	long long int i, j, k, h, t;
	i = h = 1;
	j = k = 0;
	while (n > 0) {
		if (n%2 == 1) {
			t = (j*h)%MOD;
			j = (i*h)%MOD + (j*k)%MOD + t;
			i = (i*k)%MOD + t;
		}
		t = (h*h)%MOD;
		h = (2*k*h)%MOD + t;
		k = (k*k)%MOD + t;
		n = n/2;
	}
	return j;
}

int main(){
    long long int t,n;
    scanf("%lld",&t);
while(t--) {
    scanf("%lld",&n);
    printf("%lld\n",(fib(n)*fib(n+1))%MOD);
}
    return 0;
}
