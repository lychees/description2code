N = map(int, raw_input().strip().split(' '))

for i in range(n):
    a = N[i]
    b = M[a:]
    j = ct
    while True:
        if j % M[a-1] == 0:
            temp = [j % rem for rem in b]
            if 0 not in temp:
                ct = j + 1
                break
