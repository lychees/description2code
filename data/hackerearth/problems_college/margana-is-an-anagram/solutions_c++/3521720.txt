#include <bits/stdc++.h>
using namespace std;

int main()
{
    int N;
    cin>>N;
    while(N--)
    {
    	string S1,S2;
    	cin>>S1>>S2;
    	sort(S1.begin(), S1.end());
    	sort(S2.begin(), S2.end());
    	if(S1==S2)
    		cout<<"Yes"<<endl;
    	else
    		cout<<"No"<<endl;
    }
    return 0;
}
