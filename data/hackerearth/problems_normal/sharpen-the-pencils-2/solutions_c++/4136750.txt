#include<iostream>

using namespace std;

int a[100001];

int main(){
    int t;
    cin >> t;
    while(t--){
        int n;
        cin >> n;
        int index1 = 0, index2 = n-1;
        long temp1 = 0, temp2 = 0;
        for(int i=0;i<n;i++){
            cin >> a[i];
        }
        while(index1 <= index2){
            if(index1 == index2 && temp1 == temp2){
                index1++;
            }
            else if(temp1 > temp2){
                temp2 += a[index2]*2;
                index2--;
            }
            else if(temp1 < temp2){
                temp1 += a[index1];
                index1++;
            }
            else{
                temp1 += a[index1];
                index1++;
                temp2 += a[index2]*2;
                index2--;
            }
        }
        cout << index1 << " " << n-1-index2 << endl;
    }
}
