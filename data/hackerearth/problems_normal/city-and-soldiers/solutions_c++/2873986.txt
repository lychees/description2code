#define MAX 100007
#define EPS 1e-9
int par[MAX];
int f_res(int u)
{
    if(par[u]==u)return u;
    return par[u]=f_res(par[u]);
}
void Union(int i,int j)
{
    int u=f_res(i);
    int v=f_res(j);
    if(u==v)return;
    par[u]=v;
}
void make_led(int u)
{
    int v=f_res(u);
    if(u==v)return;
    par[v]=u;
    par[u]=u;
}
int main()
{
    //READ("in.in");
    //WRITE("out.out");
    int n,q,u,v;
    scanf("%d%d",&n,&q);
    FOR(i,1,n)par[i]=i;
    FOR(i,1,q)
    {
        int type;
        scanf("%d",&type);
        if(type==1)
        {
            scanf("%d%d",&u,&v);
            Union(u,v);
        }
        else if(type==2)
        {
            scanf("%d",&u);
            make_led(u);
        }
        else
        {
            scanf("%d",&u);
            printf("%d\n",f_res(u));
        }
    }
    return 0;
}
