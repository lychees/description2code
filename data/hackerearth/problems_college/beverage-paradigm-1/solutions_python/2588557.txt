def possible(ar,n):
	ar.sort()
	for i in range(len(ar)):
		for j in range(i+1,len(ar)):
			for k in range(j+1,len(ar)):
				if ar[i]+ar[j]+ar[k] == n:
					return 1
	return 0
n = int(raw_input())
ar = map(int,raw_input().split())
k = int(raw_input())
re = possible(ar,k)
if re == 1:
	print "True"
else:
	print "False"
