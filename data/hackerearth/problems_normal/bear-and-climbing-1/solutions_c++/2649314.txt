#include <bits/stdc++.h>
using namespace std;

typedef pair<int,int> pii;
typedef long long LL;

const int N = 1000005;

pii stek[N];
int head,tail;

pii arr[N];
int n;

LL sqr(LL x){
	return x * x;
}

bool add(pii x,LL lim){
	while(head < tail && sqr(stek[head].first - x.first) + sqr(stek[head].second - x.second) > lim)
		head++;
	if(head == tail) return 0;
	if(stek[head].first - stek[head].second <= x.first - x.second) return 0;
	while(head < tail && stek[tail - 1].first - stek[tail - 1].second <= x.first - x.second)
		tail--;
	stek[tail++] = x;	
	return 1;
}

int main()
{
    ios_base::sync_with_stdio(0);
    cin >> n;
    for(int i = 1 ; i <= n ; i++)
    	cin >> arr[i].first >> arr[i].second;
    LL lo = 0;
    LL hi = (LL)1e15;
    while(lo < hi){
    	LL mid = (lo + hi) / 2LL;
    	head = tail = 0;
    	stek[tail++] = arr[n];
    	for(int i = n - 1 ; i > 1 ; i--){
    		bool temp = add(arr[i],mid);
    	}
    	if(add(arr[1],mid))
    		hi = mid;
    	else
    		lo = mid + 1;
    }
    if(lo > (LL)3e14)
    	lo = -1;
    cout << lo << endl;	
    return 0;
}
