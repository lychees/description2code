#include <iostream>
#include <stack>
using namespace std;


int main()
{
	stack <int> s;
	int n;
	cin>>n;
	while(n--)
	{
		int i;
		cin>>i;
		if (i== 0 && s.size() > 0)
		{
			s.pop();
		}
		else if(i != 0)
		{
			s.push(i);
		}
	}
	int sum = 0;
	while(!s.empty())
	{
		
		sum = sum + s.top();
		s.pop();
	}
	cout<<sum;
	return 0;
}
