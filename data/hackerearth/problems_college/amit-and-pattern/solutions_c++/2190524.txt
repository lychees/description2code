#include <iostream>
using namespace std;
int arr[101][101];
int n,m,k;
int print(int a,int b)
{
	if(k==m*n)
		return 0;
	int i=a,j=b;
	while(i>=0 && j<m && j>=0 &&i<n)
	{
		cout<<arr[i][j]<<" ";
		i++;
		j--;
		k++;
	}
	cout<<'\n';
	if(b==m-1)
		print(a+1,b);
	else
		print(0,b+1);
	return 0;
}
int main()
{
    cin>>n>>m;
    k=0;
    for(int i=0;i<n;i++)
    	for(int j=0;j<m;j++)
    		cin>>arr[i][j];
	print(0,0);
	return 0;
}
