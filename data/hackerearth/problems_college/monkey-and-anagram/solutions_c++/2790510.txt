#include<bits/stdc++.h>
using namespace std;

int main()
{
	int tc; scanf("%d",&tc);
	while(tc--){
		string s,t; cin>>s>>t;
		map<char,int> mp,mp1;
		map<char,int>::iterator itr,itr1;	
		int l=s.length();
		int l1=t.length(),cnt=0,flag=0;
		for(int i=0;i<l;i++) mp[s[i]]++;	
		for(int i=0;i<l1;i++) mp1[t[i]]++;
		for(int i=0;i<l;i++){
			itr=mp1.find(s[i]);
			if(itr==mp.end() || itr->second==0){
				flag=1; break;
            }       
            itr->second--;
	    }
	    if(flag==1) printf("-1\n");
	    else{
	    	mp1.clear();
	    	for(int i=0;i<l1;i++) mp1[t[i]]++;
	    	for(itr=mp1.begin();itr!=mp1.end();++itr){
	    		itr1=mp.find(itr->first);
	    		if(itr1==mp.end() || itr->second>itr1->second)  cnt++;
	        }	
	        printf("%d\n",cnt);
	    }
	}
    //cout<<"Hello World!\n";
    return 0;
}
