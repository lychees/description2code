#include <iostream>
#include <vector>
using namespace std;
typedef long long int ll;

ll dp[2000005];

int main()
{
	ll maxval=0,ans=0,x;
	int n;
	cin>>n;
	vector <ll> A;
	for(int i=0;i<n;++i)
	{
		cin>>x;
		ans=ans+x*(i+1);
		A.push_back(x);
	}
	for(int i=0;i<n;++i)
	{
		A.push_back(A[i]);
	}
	dp[0]=A[0];
	for(int i=1;i<A.size();++i)
	dp[i]=A[i]+dp[i-1];
	
	maxval=ans;
	for(int i=n;i<A.size();++i)
	{
		ans=ans-(dp[i-1]-((i-n-1)>=0?dp[i-n-1]:0))+n*A[i];
		maxval=max(maxval,ans);
	}

	cout<<maxval;
   // cout << "Hello World!" << endl;
    return 0;
}
