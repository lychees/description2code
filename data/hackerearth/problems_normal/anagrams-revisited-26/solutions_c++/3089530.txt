#include<bits/stdc++.h>
using namespace std;
int main()
{
	int N;
	map<string,int >mp;
	map<string,int>::iterator it;
	cin>>N;
	for(int i=0;i<N;i++)
	{
		string s;
		cin>>s;
		sort(s.begin(),s.end());
		mp[s]+=1;

	}

	int max=-900;
	for(it=mp.begin();it!=mp.end();it++)
	{
		if(max<it->second)max=it->second;
	}
	cout<<max<<endl;




	return 0;
}
