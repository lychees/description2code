 #include<iostream>
#include<cstdio>
#define ll long long
#include<cstring>
#include<algorithm>
#include<cmath>
using namespace std;
int arr[500001];
int main()
{
	int t,n,i,count,ans,j,flag,x;
	scanf("%d",&t);
	while(t--)
	{
		memset(arr,0,sizeof arr);
		count=ans=0;
		flag=1;
		scanf("%d",&n);
		for(i=0;i<n;i++)
		{
			scanf("%d",&x);
			arr[x]++;
		}
		j=0;
		for(i=1;i<=n;i++)
		{
			while(arr[j]==0)
				j++;
			arr[j]--;
			if(j-count<=0)
			{
				flag=0;
				break;
			}
			if(i%6==0&&i>0)
				count++;
			ans++;
			count++;
		}
		if(flag)
			cout<<"Rick now go and save Carl and Judas\n";
		else
			cout<<"Goodbye Rick\n"<<ans<<endl;
	}
	return 0;
}
