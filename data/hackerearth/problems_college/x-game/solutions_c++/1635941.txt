#include <iostream>
using namespace std;
#define MOD 1000000007
long long int ppow(int a, long long int b) {
	if(a==1 || b==0) return 1;
	if(b&1) {
		return (a*ppow(a,b-1))%MOD;
	}
	else {
		long long int t = ppow(a,b/2);
		return (t*t)%MOD;
	}
}
int main()
{
   
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	long long int n,x,ans =0;
    	scanf("%d",&n);
    	for(int i = 0; i < n; i++)
    	{
    		scanf("%lld",&x);
    		ans = ans | x;
    		if(ans >= MOD)  ans%=MOD;
    	}
    	ans = ans * ppow(2,n-1);
    	printf("%lld\n",ans%MOD);
    }
    return 0;
}
