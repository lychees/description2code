#include <bits/stdc++.h>
using namespace std;

int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
    	int n,m;
    	cin>>n>>m;
    	char a[n][m];
    	for(int i=0;i<n;i++)
    	{
    		for(int j=0;j<m;j++)
    			cin>>a[i][j];
    	}
    	int ct=0;
    	for(int i=0;i<n-1;i++)
    	{
    		for(int j=0;j<m-1;j++)
    		{
    			if(a[i][j]=='/')
    			{
    				if(a[i][j+1]==(char)(92) && a[i+1][j]==(char)(92) && a[i+1][j+1]==(char)(47))
    					{ct++;}
    			}
    		}
    	}
    	cout<<ct<<endl;
    }
    return 0;
}
