#include <bits/stdc++.h>
#include <iostream>
#include <algorithm>
#include <string>
#include <set>
#include <vector>
using namespace std;
#define MAX 10000006
#define LL long long
LL isprime[MAX];
 
void sieve()
{
	for(int i = 0; i < MAX; i++) {
		isprime[i] = 1;
	}
	isprime[0] = isprime[1] = 0;	
	for (LL i = 2; i * i <= MAX - 2; i++) {
		if (isprime[i]) {
			for (LL j = i * 2; j < MAX - 2; j += i) {
				isprime[j] = 0;
			}
		}
	}
}
 
int main()
{
	sieve();
	for (LL i = 1; i <=1e7; i++) {
		isprime[i] += isprime[i - 1];
	}
	int t;
	scanf("%d", &t);
	while(t--) {
		LL l, r;
		scanf("%lld%lld", &l, &r);
		LL ans = isprime[r] - isprime[l - 1];
		LL to = r - l + 1;
		float he = (float)(ans) / (float)to;
		printf("%.6f\n", he);
	}
	return 0;
}
