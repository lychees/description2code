#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t, n;
    cin >> t;
    while (t--) {
    	cin >> n;
    	long int a[102];
    	for (int i=0; i < n; i++) cin >> a[i];
    	sort(a,a+n);
    	vector<long> b;
    	for (int i=0; i < n/2; i++) {
    		b.push_back(a[n-i-1]+a[i]);
    	}
    	cout << *max_element(b.begin(),b.end())-*min_element(b.begin(),b.end()) << endl;
    }
    return 0;
}
