#include <iostream>
using namespace std;
#define IOS 		ios_base::sync_with_stdio(false); cin.tie(NULL);
#define IN          int t; scanf("%d",&t); while(t--)
int main()
{
   int i,n,hiring,ans;
   
   char s[25];
   IN
   {
   		scanf("%d",&n);
   		getc(stdin);
   		hiring=ans=0;
   		while(n--)
   		{
   			gets(s);
   			if(!hiring and s[0]=='N')//when there are no free developers(hiring = 0) and a new project is available 
   			ans++;//a project gets unassigned
   			else
   			{
   			if(s[0]=='1')//increment no of hirings done by its number given (1-5)
   			hiring++;
   			else if(s[0]=='2')
   			hiring+=2;
   			else if(s[0]=='3')
   		    hiring+=3;
   			else if(s[0]=='4')
   			hiring+=4;
   			else if(s[0]=='5')
   			hiring+=5;
   			else if(s[0]=='N')//new project in available and it gets assigned to one hired employee 
   			hiring--;
   			}
   		}
   		cout << ans << endl;//print the ans
   }
    return 0;
}
