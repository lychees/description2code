#include <bits/stdc++.h>
#define ll long long
#define sc(a) scanf("%lld", &a)
#define pr(a) printf("%lld", a)
#define MAX 1005

using namespace std;

ll pas[MAX][MAX];
void pascals(int n)
{
register ll i,j;
pas[0][0]=1;
pas[1][0]=pas[1][1]=1;
for(i=2;i<=n;i++)
{
pas[i][0]=1;
for(j=1;j<i;j++)
{
pas[i][j]= pas[i-1][j-1]+pas[i-1][j];
}
pas[i][j]=1;
}
}

int main()
{
	pascals(1000);
	ll n,coun=0;
	sc(n);
	for (ll i = 3; i <= n; i++) {
		//cout << pas[i-1][2] << " ";
		coun += pas[i-1][2];
	}
	//cout << " here ";
	cout << 2*coun << endl;
	
	return 0;
}
