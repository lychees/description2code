#include <bits/stdc++.h>
using namespace std;

int main(){
    long long n, tmp;
    cin >> tmp;
    n = tmp/2LL + 1;
    
    while(n < tmp){
        bool prime = true;
        long long tmp1 = n;
        for(long long x = 2; x <= sqrt(n); x++){
            if(n % x == 0){
                prime = false; 
                while(tmp1 % x == 0) tmp1 /= x;
                break;
            }
        }
        
        if(!prime && tmp1 != 1) n++; else break;
    }
    
    cout << n << '\n';
}
