T=int(raw_input())

while T>0:
    N,K=[int(x) for x in raw_input().split()[:2]]
    tList=[int(x) for x in raw_input().split()[:N]]

    if K>=N:
        print max(max(tList),0)

    DP=[0]*(N+1)

    for x in xrange(0,N):

        if x>K:
            DP[x]=max(DP[x-K-1]+tList[x],DP[x-1])
        else:
            DP[x]=max(DP[x-1],tList[x])

    print DP[N-1]
    T-=1
