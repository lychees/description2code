#include <iostream>
#include <stdio.h>
#include <algorithm>
using namespace std;

int main()
{
    int N, K, i;
    scanf("%d %d", &N, &K);
    int a[N];
    for(i=0;i<N;++i)
    {
    	scanf("%d", &a[i]);
    }
    sort(a, a + N);
    int sum = 0;
    if(K >= N)
    {
    	for(i=0;i<N;++i)
    	{
    		sum += a[i];
    	}
    }
    else
    {
    	int x = 1;
    	while(1)
    	{
		    for(i=0;i<K && N;++i)
		    {
		    	sum += (x * a[N - 1]);
		    	N--;
		    }
		    if(N == 0)
		    	break;
		    ++x;
    	}
    }
    printf("%d\n", sum);
    return 0;
}
