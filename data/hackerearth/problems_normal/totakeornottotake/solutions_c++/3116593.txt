#include<cstdio>
using namespace std;
int main()
{
    int T;
    scanf("%d",&T);
    while (T--)
    {
	int B,X;
	long long int maxp=1,maxn=1;
	char c;
	scanf("%d",&B);
	while (B--)
	{
	    scanf(" %c",&c);
 
	    if (c=='N')
	    {
		if ((-maxp)<maxn)
		    maxn = -maxp;
		if ((-maxn)>maxp)
		    maxp = -maxn;
	    }
	    else if (c=='+')
	    {
		scanf("%d",&X);
		maxp = maxp+X;
	    }
	    else if (c=='-')
	    {
		scanf("%d",&X);
		maxn = maxn-X;
	    }
	    else if (c=='*')
	    {
		scanf("%d",&X);
		maxp = maxp*X;
		maxn = maxn*X;
	    }
	    else if (c=='/')
	    {
		scanf("%d",&X);
		if (maxn>0)
		    maxn = maxn/X;
	    }
	    //printf("\nDEB %c %d %d\n",c,maxn,maxp);
	}
	printf("%lld\n",maxp);
 
    }
    return 0;
}
