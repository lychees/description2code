#include<iostream>
#include<stack>
#include<string.h>
#include<string>
using namespace std;

int main()
{
	int test, len;
	char ch;
	string temp;
	stack<char> str;
	stack<long int> ans;
	cin>> test;
	while(test--)
	{
		cin>> temp;
		for(int i=0;i < temp.size();i++)
			str.push(temp[i]);
		ans.push((long)(str.top()-'0'));
		str.pop();
		while(!str.empty())
		{
			ch = str.top();
			str.pop();
			switch(ch)
			{
				case '*':
					ans.top() *= (long)(str.top()-'0');
					break;
				case '+':
					ans.top() += (long)(str.top()-'0');
					break;
				case '-':
					ans.top() -= (long)(str.top()-'0');
					break;
			}
			str.pop();
		}
		cout<< ans.top() << endl;
	}
}
