tc=int(raw_input())
while tc:
    tc=tc-1
    N,E,V=map(int,raw_input().split())
    L=map(int,raw_input().split())
    #print N,E,V
    #print L
    count=0
    current_energy=E
    for I in L:
        if current_energy>I:
            current_energy=current_energy-I
        else:
            temp=I-current_energy
            to_add=temp/V +1
            current_energy=current_energy+(V*to_add)-I
            count+=to_add
    print count
