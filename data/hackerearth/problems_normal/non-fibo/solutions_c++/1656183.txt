template<class T>
inline void readInt( T &n )//fast input function
{
   n=0;
   T ch=getcx();int sign=1;
   while( ch < '0' || ch > '9' ){if(ch=='-')sign=-1; ch=getcx();}

   while(  ch >= '0' && ch <= '9' )
           n = (n<<3)+(n<<1) + ch-'0', ch=getcx();
   n=n*sign;
}

LL fib[maxn];

    rb(i, 3, maxn) {
        fib[i] = fib[i - 1] + fib[i - 2];
        if(fib[i] < 0 || fib[i] > maxL) {t = i + 1; break;}
    }

    cin >> T;

        cin >> n;

        num = 0; idx = 1;

            if(num + d >= n) break;

        }
