#include <stdio.h>
#define size 1000001
int main()
{
    int t,x,c;
    scanf("%d",&t);
    int arr[size]={0};
    int count[size];
    arr[0]=1;
    arr[1]=1;
    for(int i=2;i<size;i++)
    {
    c=0;
    if(arr[i]==0)
    {
    for(int j=1;i*j<size;j++)
    {
    if(arr[i*j]!=1)
    {
    arr[i*j]=1;
    c++;   
    }
    }
    }
    count[i]=c;
    }
    
    while(t--)
    {
    scanf("%d",&x);
    printf("%d\n",count[x]);
    }
    return 0;
}
