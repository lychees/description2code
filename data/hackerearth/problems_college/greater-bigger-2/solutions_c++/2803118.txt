#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t,flag,n,i;
    string s;
    scanf("%d",&t);
    while(t--)
    {
    	flag=0;
    	cin>>s;
    	n=s.length();
    	vector<char> v(n);
    	for(i=0;i<n;i++)
    	{
    		v[i]=s[i];
    	}
    	next_permutation(v.begin(),v.end());
    	for(i=0;i<n;i++)
    	{
    		if(s[i]!=v[i])
    		{
    			flag=1;
    			break;
    		}
    	}
    	if(flag)
    	for(i=0;i<n;i++)
    	{
    		printf("%c",v[i]);
    	}
    	else
    		printf("no answer");
    	printf("\n");
    }
    return 0;
}
