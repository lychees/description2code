#include<iostream>
#include<map>
using namespace std;

map<long int,bool> prime;

void preComp(){
	prime[0]=true;
	prime[1]=true;
	prime[2]=false;
	for(long int i=2;i<100010;i++){
		if(!prime[i]){
			for(long int j=i*i;j<100010;j+=i){
				prime[j]=true;
			}
		}
	}
}

int main(){
	int t;
	cin>>t;
	preComp();
	while(t--){
		int n;
		cin>>n;
		if(n<2)
			cout<<4<<endl;
		else{
			while(!prime[++n]);
			cout<<n<<endl;
		}
	}
	return 0;
}
