#include <bits/stdc++.h>
using namespace std;
#define gc getchar_unlocked
#define sc(x) scanf("%d",&x)
inline void sf(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc())   
        {x = (x<<1) + (x<<3) + c - 48;}
}
int main()
{
	int t;
	sc(t);
	while(t--)
	{
		int *fx;
		fx=(int*)calloc(100005,sizeof(int));
		int *fy;
		fy=(int*)calloc(100005,sizeof(int));
		int n,q;
		sc(n);
		sc(q);
		for(int i=0;i<n;i++)
		{
			int x1,y1,x2,y2;
			sc(x1);
			sc(y1);
			sc(x2);
			sc(y2);
			int l1,r1,l2,r2;
			l1=min(x1,x2);
			r1=max(x1,x2);
			l2=min(y1,y2);
			r2=max(y1,y2);
			fx[l1]++;
			fx[r1+1]--;
			fy[l2]++;
			fy[r2+1]--;
		}
		for(int i=1;i<=100003;i++){
			fx[i]+=fx[i-1];
			fy[i]+=fy[i-1];
		}
		while(q--)
		{
			int a,b;
			sc(a);
			sc(b);
			if(b==0)
			{
				printf("%d\n",fx[a]);
			}
			else
			{
				printf("%d\n",fy[b]);				
			}
		}
	}
	return 0;
}
