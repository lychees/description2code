using namespace std;
int main()
{
    int t,i,l;
    char s[100000];
    cin>>t;
    while(t--)
    {
        i=0;
        scanf("%s",s);
        l=strlen(s);
        while(i<l)
        {
           printf("%c",s[i]);
           if(s[i]==s[i+1])
                while(s[i]==s[i+1]&&i<l)
           {
               i++;
           }
            i++;
        }
        printf("\n");
    }
    return 0;
}
