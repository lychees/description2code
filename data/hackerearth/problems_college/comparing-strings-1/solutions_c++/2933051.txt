#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int tc,i;string a,b;
    scanf("%d",&tc);
    while(tc--)
    {
    	bool flag=0;
    	cin>>a>>b;
    	int hash[26]={0};
    	for( i=0;i<a.length();i++)
    	{
    		hash[a[i]-97]++;
    	}
    	for(i=0;i<b.length();i++)
    	{
    		if(hash[b[i]-97]>0)
    		{flag=1;break;}
    	}
    	if(flag==1)
    	{
    		cout<<"YES\n";
    	}
    	else
    	cout<<"NO\n";
    }
    return 0;
}
