#include <iostream>
using namespace std;
#include<cstdio>

int A[1000000];
int main()
{
	int n;
	cin>>n;
	long long int min=1000000001,max=1;
	for(int i=0;i<n;i++){
		scanf("%d",&A[i]);
		if(A[i]>max)
			max=A[i];
		if(A[i]<min&&A[i]!=1)
			min=A[i];
	}
	long long int ans=min*max;
    printf("%lld\n",ans);
    return 0;
}
