#include <iostream>
using namespace std;

int josephus(int n,int k)
{
    if(n==1)
    return 1;

    else
    return ((josephus(n-1,k)+k)%n+1);
}


int main()
{
    int cases,kase,n,k;
    
    cin>>cases;
    
    for(kase=0;kase<cases;kase++)
    {
    	cin>>n;
    	cin>>k;
    	
    	cout<<josephus(n,k)<<"\n";
    }
    return 0;
}
