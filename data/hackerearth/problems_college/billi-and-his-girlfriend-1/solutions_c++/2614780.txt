#include<bits/stdc++.h>
using namespace std;
long long int power(long long int a, long long int b, long long int MOD) {
	long long int res=1;
	while(b) {
		if(b&1) {
			res=((res%MOD)*(a%MOD))%MOD;
		}
		a=((a%MOD)*(a%MOD))%MOD;
		b>>=1;
	}
	long long int sum=0;
	while(res) {
		sum+=res%10;
		res/=10;
	}
	return sum;
}
		
int main() {
	int t,k;
	long long int a,b,MOD;
	scanf("%d",&t);
	while(t--) {
		scanf("%lld%lld%d",&a,&b,&k);
		MOD=1;
		for(int i=1;i<=k;i++)
			MOD*=10;
		printf("%lld\n",power(a,b,MOD));
	}
	return 0;
}
