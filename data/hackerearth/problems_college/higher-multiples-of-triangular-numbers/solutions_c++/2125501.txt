#include <iostream>
#include <vector>
using namespace std;
vector<int> tri;

void init()
{
	for(int i=1 ; i<=1000 ; i++){
		tri.push_back((i*(i+1))/2);
	}
}

int divi(int n){
	if(n==1) return 1;
	int ans=0;
	for(int i=1 ; i*i<=n ; i++){
		if(n%i==0)
		ans+=2;
	}
	if(ans==0)
	return 2;
	return ans;
}
int main()
{
	init();
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	
    	for(int i=0 ;i<tri.size() ; i++){
    		if(divi(tri[i])>n){
    			cout<<tri[i]<<endl;
    			break;
    		}
    	}
    }
    return 0;
}
