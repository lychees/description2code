#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int sum,dif,a=0,b=0;
    	cin>>sum>>dif;
    	if(dif>sum)
    	cout<<"impossible"<<endl;
    	else
    	{
    		 
    		a=(sum+dif)/2;
    		b=(sum-dif)/2;
    		if(a+b!=sum)
    		cout<<"impossible"<<endl;
    		else{
    		if(a>b)
    	cout<<a<<" "<<b<<endl;
    	else
    	cout<<b<<" "<<a<<endl;
    		}
    	}
    	
    	
    	}
    return 0;
}
