#include <iostream>
using namespace std;

int main()
{
   long long int t;
   cin>>t;
   while(t--)
   {
   	long long int n,b,n1,n2,n3;
   	cin>>n>>b;
   	n1=n/b;
   	n2=b*n1;
   	n3=b*(n1+1);
   	if((n3-n)==(n-n2))
   	{
   		cout<<n3<<endl;
   	}
   	else if((n3-n)<(n-n2))
   	cout<<n3<<endl;
   	else
   	cout<<n2<<endl;
   }
}
