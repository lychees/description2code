#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    for(;t--;)
    {
    	long long int a,b,sum=0,count=0;
    	cin>>a>>b;
    	while(a!=0 && b!=0)
    	{
    		sum=(a%10)+(b%10);
    		if(sum>=10)
    		count++;
    		a/=10;
    		b/=10;
    		
    	}
    	if(count==0)
    	cout<<("No carry operation\n");
    	else if(count==1)
    	cout<<("1 carry operation\n");
    	else
    	cout<<count<<(" carry operations\n");
    }
    return 0;
}
