#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    int t;
    cin >> t;
    for(int k = 0; k < t; k++)
    {
        int c = 0;
        int n;
        cin >> n;
        int arr[10000];
        bool visit[n];
        for(int i=0; i < n; i++)
        {
            cin >> arr[i];
        }
        sort(arr,arr+n);
        for(int i = 0; i < n; i++)
        {
            visit[i] = false;
        }
        for(int i = 0; i < n; i++)
        {
            if(visit[i] == false){
                    c++;
                    visit[i] = true;
                if(arr[i]+ 1 == arr[i+1] && arr[i]-1 == arr[i-1] && visit[i+1] == false && visit[i-1] == false)
                {
                    c++;
                    visit[i+1] = true;
                    visit[i-1] = true;
                }
                else if(arr[i] + 1 == arr[i+1] && visit[i+1] == false)
                {
                   visit[i+1] = true;
                }
                else if(arr[i] - 1 == arr[i-1] && visit[i-1] == false)
                {
                   visit[i-1] = true;
                }
            }
        }
        cout << c << endl;

    }
    return 0;
}
