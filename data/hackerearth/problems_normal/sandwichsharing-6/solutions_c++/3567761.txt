#include <iostream>
using namespace std;
#define ll long long int
int main()
{
    ios::sync_with_stdio(0);
    ll i,j,k,t,n,p;
    cin>>t;
    while(t--)
    {
    	cin>>n>>p;
    	while(p--)
    	{
    		if(n==3)
    		break;
    		if(n%2==0)
    		{
    			n=n/2+1;
    		}
    		else
    		{
    			n=n/2+2;
    		}
    	}
    	cout<<n<<"\n";
    }
    return 0;
}
