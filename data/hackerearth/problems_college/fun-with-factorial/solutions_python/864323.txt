import math
fact=[]
for i in range(25):
    fact.append(math.factorial(i)%100000)
t=int(input())
while(t):
    n=int(input())
    if(n<25):
        x=fact[n]
        y=str(fact[n])
        if(x<10):
            print '0000'+y
        elif(x<100):
            print '000'+y
        elif(x<1000):
            print '00'+y
        elif(x<10000):
            print '0'+y
        else:
            print y

    else:
        print('00000')
    t=t-1
