#include <iostream>
using namespace std;

int main()
{
    int t,cnt;
    long long n;
    cin>>t;
    while(t--)
    {
    	cnt=0;
    	cin>>n;
    	while(n!=0)
    	{
    	  if(n&1)
    	  cnt++;
    	  n=n>>1;
    	}
    	if(cnt&1)
    	cout<<"odd\n";
    	else
    	cout<<"even\n";
    }
    return 0;
}
