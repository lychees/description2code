#include <bits/stdc++.h>
using namespace std;
map<long long, long long> dp;
long long solve(long long n)
{
    if(n==0 || n==1) return n;
    if(dp[n]) return dp[n];

     dp[n]=n;
        return dp[n]=(max(dp[n],solve(n/2)+solve(n/3)+solve(n/4)));


}
