inline void sc(int &x)
{
	register int c = gc();
	x = 0;
	int neg = 0;
    for(; ((c<48 || c>57) && c != '-');c=gc());
	if(c=='-')
	{
		neg = 1;
    	c = gc();
    }
    for(; c>47 && c<58 ; c = gc()) {
    	x = (x<<1) + (x<<3) + c - 48;
    }
    if(neg)
       	x = -x;
}
 
inline void scl(long long *a) {
	register char c=0;
	while (c < 33)
		c = gc();
	*a = 0;	
	while (c > 33) {
		*a = *a * 10 + c - '0';
		c = gc();
	}
}
 
 
long long modpow(long long a, long long b) {
    a %= mod;
 
    long long r = 1;
 
    while (b) {
        if (b & 1) {
            r = (r * a) % mod;
        }
 
        a = (a * a) % mod;
 
        b >>= 1;
    }
 
    return r;
}
