#include <bits/stdc++.h>

using namespace std;

const int MOD=1000000007;
vector<int> primes;
bool isprime[1000001];
int A, B;

int work(int N)
{
    int ret=0;
    int ptr0=-1;
    for(size_t i=0; i<primes.size(); i++)
    {
        while(ptr0+1<(int)i && primes[ptr0+1]*primes[i]<=N)
            ptr0++;
        while(ptr0>=0 && primes[ptr0]*primes[i]>N)
            ptr0--;
        ret+=ptr0+1;
    }
    return ret;
}

void solve()
{
    scanf("%d%d", &A, &B);
    printf("%d\n", work(B)-work(A-1));
}

int main()
{
    memset(isprime, 1, sizeof isprime);
    for(int i=2; i<=1000000; i++) if(isprime[i])
    {
        primes.push_back(i);
        for(int j=i*2; j<=1000000; j+=i)
            isprime[j]=false;
    }
    int T;
    scanf("%d", &T);
    while(T--)
        solve();
    return 0;
}
