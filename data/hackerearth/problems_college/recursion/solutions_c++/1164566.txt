#include<bits/stdc++.h>
using namespace std;
map<long long int,int> mp1;
int k=1;
void swap(int *a,int *b)
{
    int temp=*a;
    *a=*b;
    *b=temp;
}
void permute(int *a, int i, int n)
{
   int j;
   if (i == n)
     {
         long long int ans=0;
         for(int i=0;i<n;i++)
         {
             ans=ans+(pow(10.0,n-1-i))*a[i];
         }
        // printf("%lld\n",ans);
         mp1[ans]=k;
         k++;
     }
   else
   {
        for (j = i; j < n; j++)
       {
          swap((a+i), (a+j));
          permute(a, i+1, n);
          swap((a+i), (a+j));
       }
   }
}

return 0;}
