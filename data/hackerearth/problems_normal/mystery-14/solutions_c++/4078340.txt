#include <iostream>

using namespace std;

int main()
{
    int t;
    cin>>t;

    while(t--){
        int n;
        cin>>n;
        int sum = 0;
        int* arr = new int[n];
        
        for(int i=0; i<n; i++){
            cin>>arr[i];
        }
        for(int i=0; i<n; i++){
            if(arr[i]%2 != 0){
                sum = sum + arr[i];
            }
        }

        cout<<sum<<endl;
    }
    return 0;
}
