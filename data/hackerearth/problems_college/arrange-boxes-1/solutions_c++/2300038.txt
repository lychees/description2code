#include <iostream>
#include <vector> 

using namespace std;

int find (vector<int> , vector<int> , int);
int main()
{
	int t,n; 
	cin >> t;
	vector<int> giv,req;
	while (t)
	{
		cin >> n;
		vector<int> giv(n),req(n);
		for (int i = 0; i < n; i++)
			cin >> giv[i];
		for (int i = 0; i < n; i++)
			cin >> req[i];
		cout << find(giv,req,n) << "\n";
		t--;
	}
	return 0;
}


int find (vector<int> giv , vector<int> req , int n)
{
	int count = 0;
	for (int i = 0; i < n; i++)
		{
			for (int j = i; j < n; j++)
			{
				if (req[i] == giv[j])
				{
					if (i == j)
					{
						break;
					}
					else  
					{
						while (i != j)
						{
							swap(giv[j],giv[j-1]);
							count++;
							j = j-1;
						}
						break;
					}
				}
			}
		}
		
	return count;
}
