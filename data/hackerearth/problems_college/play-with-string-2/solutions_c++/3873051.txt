#include <iostream>
#include<string.h>
using namespace std;
int main()
{
	int t,temp,i;
	cin>>t;
	string s;
	while(t--)
	{
		temp=1;
		cin>>s;
		for(i=0;i<s.length()/2;i++)
		{
			if(abs(s[i+1]-s[i])!=abs(s[s.length()-1-i]-s[s.length()-2-i]))
			{
				temp=0;
				break;
			}
		}
		if(temp)
		cout<<"Funny\n";
		else
		cout<<"Not Funny\n";;
	}
    return 0;
}
