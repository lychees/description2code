#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int z=0;z<t;z++)
    {
    	string s;
    	cin>>s;
    	int chk=0;
    	for(int i=0;i<s.size()/2;i++)
    	{
    		if(s[i]!=s[s.size()-i-1])
    		{
    			chk=1;
    			cout<<"NO"<<endl;
    			break;
    		}
    	}
    	if(chk==0)
    	cout<<"YES"<<endl;
    }
    return 0;
}
