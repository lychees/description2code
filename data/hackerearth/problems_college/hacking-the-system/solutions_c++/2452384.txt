#include <iostream>
#include<cstdlib>
#include<sstream>

using namespace std;

int main()
{
  long int comp,player;
  int count;
  cin>>comp;
  cin>>player;
string Result1,Result2;//string which will contain the result

stringstream convert; // stringstream used for the conversion
stringstream convert1;
convert <<comp;//add the value of Number to the characters in the stream
convert1<<player;
Result1 = convert.str();//set Result to the content of the stream
Result2=convert1.str();
int len=Result1.length();
int len1=Result2.length();
if(len!=len1)
cout<<"loses";
else
{
	for(int i=0;i<len1;i++)
	{
	if(Result1[i]==Result2[i])
	continue;
	else
	count++;
	}
	if(count==1)
	cout<<"wins";
	else if(count==0)
	cout<<"wins";
	else
	cout<<"loses";
}
    return 0;
}
