#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int dp[1002][1003];
int a[1003];
int main()
{
    int t;
    cin>>t;int n;ll sum=0;
    while(t--)
    {sum=0;
        cin>>n;
        for(int i=0;i<n;i++)
       {

       cin>>a[i];sum+=a[i];}
        for(int i=n-1;i>=0;i--)
        {
            for(int j=i;j<=n-1;j++)
            {
                if(j==i)dp[i][j]=a[i];
                else if(j==i+1)dp[i][j]=max(a[i],a[j]);
                else
                dp[i][j]=max(a[i]+min(dp[i+2][j],dp[i+1][j-1]),a[j]+min(dp[i][j-2],dp[i+1][j-1]));
            }
        }
        if(dp[0][n-1]>(sum-dp[0][n-1]))
        cout<<"Ravi"<<" "<<dp[0][n-1]<<endl;
        else if(dp[0][n-1]<(sum-dp[0][n-1]))
        cout<<"Abhinav"<<" "<<sum-dp[0][n-1]<<endl;
        else
        cout<<"Tie"<<endl;
    }
    return 0;

}
