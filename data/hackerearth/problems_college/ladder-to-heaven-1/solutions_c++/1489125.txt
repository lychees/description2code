#include <iostream>
using namespace std;

int main()
{
	long int t,n,i,j,prev,cur,next,sum;
	cin>>t;
	while(t)
	{
		cin>>n;
		prev=1;
		cur=2;
		next=3;
		sum=2;
		while(next<n)
		{
			if(!(next&1))
			sum+=next;
			
			prev=cur;
			cur=next;
			next=cur+prev;
		}
		cout<<sum<<endl;
		t--;
		
	}
    //cout << "Hello World!" << endl;
    return 0;
}
