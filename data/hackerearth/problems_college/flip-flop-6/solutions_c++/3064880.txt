#include<bits/stdc++.h>
#include<algorithm>
#include<cstring>
using namespace std;



main()
{
	long long t,i;
	long long count,l;
	char a[100001];
	scanf("%lld",&t);
	while(t--)
	{
		count=0;
		scanf("%s",a);
		l=strlen(a);
	//	sort(a,a+l);
		for(i=1;i<l;i++)
		{
			if(a[i]==a[i-1])
			
			count++;
		}
		printf("%lld\n",count);
	}
}
