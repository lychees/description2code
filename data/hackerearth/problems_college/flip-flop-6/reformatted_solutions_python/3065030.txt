'''
# Read input from stdin and provide input before running code

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
cases = input()
for case in range(cases):
    S = raw_input()
    deletion = 0
    length = len(S)
    A = ''
    for char in S:
        if char == 'X':
            A += '0'
        else:
            A += '1'
    for count in range(length - 1):
        if int(A[count]) ^ int(A[count + 1]) == 0:
            deletion += 1
    print deletion
