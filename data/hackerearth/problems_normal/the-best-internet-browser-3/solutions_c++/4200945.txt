#include <iostream>
using namespace std;
#include <string>

int main()
{
	int tc = 0, n = 0;
	const char* st =(const char*) malloc(201*sizeof(char));
	scanf("%d", &tc);
	while (n<tc) {
		scanf("%s", st);
		string site = st;
		int len = site.length();
		int pos = len - 8;
		string subsite = site.substr(4, pos);
		int vow=0;
		for (int i = 0; i < pos; i++)
		{
			if (subsite.at(i) == 'a' || subsite.at(i) == 'e' || subsite.at(i) == 'i' || subsite.at(i) == 'o' || subsite.at(i) == 'u')
			vow++;
		}
		int num = pos - vow + 4;
		int deno = len;
		cout << num << "/" << deno << endl;
		n++;
	}
	return 0;
}
