#include<iostream>
#include<cstdlib>
#include<cmath>
using namespace std;
int main()
{
	int ceo,coo,cto,minimum;
	cin>>ceo>>coo>>cto;
	int n;
	cin>>n;
	int arr[n];
	for(int i=0;i<n;i++)
	cin>>arr[i];
	for(int i=0;i<n;i++)
	{
		minimum=i;
		for(int j=i+1;j<n;j++)
		{
			if(arr[j]<arr[minimum])
			minimum=j;
		}
		int temp=arr[i];
		arr[i]=arr[minimum];
		arr[minimum]=temp;
	}
	int total[n+3];
	int mid=(n+3)/2;
	total[mid]=ceo;
	if(coo<cto)
	{
		total[n+2]=cto;
		total[0]=coo;
	}
	else
	{
		total[n+2]=coo;
		total[0]=cto;
	}
	int j=0;
	for(int i=0;i<n+3;i++)
	{
		if(i==0 || i==mid || i==n+2)
		continue;
		else
		total[i]=arr[j++];
	}
	j=0;
	int diff[n+2];
	for(int i=0;i<n+2;i++)
	diff[j++]=abs(total[i+1]-total[i]);
	int sum=0;
	for(int i=0;i<n+2;i++)
	sum=sum+diff[i];
	cout<<sum;
}
