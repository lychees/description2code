#include <iostream>
#include <set>
using namespace std;

int main()
{
    set<pair<int, int> > uni;
    string s;
    cin>>s;
    int x = 0;
    int y = 0;
    int slip = 0;
    uni.insert(make_pair(x,y));
    for(int i=0; i<s.length(); i++)
    {
    	if(s[i]=='L')
    	{
    		y--;
    	}
    	else if(s[i]=='R')
    	{
    		y++;
    	}
    	else if(s[i]=='U')
    	{
    		x--;
    	}
    	else if(s[i]=='D')
    	{
    		x++;
    	}
    	if(uni.count(make_pair(x,y))==0)
    	{
    		uni.insert(make_pair(x,y));
    	}
    	else
    	{
    		slip++;
    	}
    }
    cout<<slip<<endl;
    return 0;
}
