#include <iostream>

using namespace std;

int main()
{
   int t;
   cin >> t;
   while(t--)
   {
       long long int l,b,d,ans;
       cin >> l >> b >> d;
       if(d>l || d>b)
       ans=0;
       else
       {
           long long int n1=(l/d)%1000000007;
           long long int n2=(b/d)%1000000007;
           ans = (n1*n2)%1000000007;
       }
       cout << ans << "\n";
   }
   
   
   
   return 0;
}
