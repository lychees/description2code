#include <iostream>
using namespace std;

int main()
{
    int t,i,temp;
    int a[26];
    string str;
    cin>>t;
    while(t--)
    {
    	cin>>str;
   	    for(i=0;i<26;i++)
            a[i]=0;
    	for(i=0;str[i];i++)
    	{
    		a[str[i]-'a']++;
    	}
    	temp=a[0];
    	for(i=1;i<26;i++)
    	{
    		if(a[i]!=temp)
    		     break;
    	}
    	if(i==26)
   		    cout<<"Yes\n";
   		else
   		    cout<<"No\n";
    }
    return 0;
}
