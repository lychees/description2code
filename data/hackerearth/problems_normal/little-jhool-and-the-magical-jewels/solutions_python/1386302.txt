if __name__ == '__main__':
	cases = int(raw_input())
	start = 1
	while start <= cases:
		string = raw_input()
		rcount = 0
		ucount = 0
		bcount = 0
		ycount = 0
		for char in string:
			if char == 'r':
				rcount = rcount + 1
			elif char == 'u':
				ucount = ucount + 1
			elif char == 'b':
				bcount = bcount + 1
			elif char == 'y':
				ycount = ycount + 1
			else:
				continue
		print min(rcount, ucount, bcount, ycount)
		start = start + 1
