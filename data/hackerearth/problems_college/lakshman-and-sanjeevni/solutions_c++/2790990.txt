#include<bits/stdc++.h>
using namespace std;
#define ll long long int
ll arr[2000005];
ll p[10000004];
void markMultiples(ll a,ll n)
{
    ll i = 2, num;
    while ( (num = i*a) <= n )
    {
        p[ num-1 ]++; 
        ++i;
    }
}
 
void SieveOfEratosthenes(ll n)
{
	 
    
    if (n >= 2)
    {
        for (ll i=1; i<n; ++i)
        {
            if ( p[i] == 0 )
            {
                p[i]++;
                markMultiples( i+1, n);
            }
        }
    }
}
int main()
{
	 SieveOfEratosthenes(10000004);
     ll n, sum=0,i,rad;
	 scanf("%lld",&n);
	 for(i=0;i<n;i++)
	 scanf("%lld",&arr[i]);
	 for(i=0;i<n;i++)
	 {
	 	 if(arr[i]==1)
	 	 rad=0;
	 	 else
	     rad=p[arr[i]-1];
		 sum+=pow(2,rad);
		 //cout<<sum<<endl;
	 }
	 printf("%lld",sum);
	 return 0;
}
