#include<bits/stdc++.h>
using namespace std;
vector<string>vec;
int main()
{
	int i,j,n,t;
	cin>>t;
	getchar();
	while(t--)
	{
		string s,s1;
		getline(cin,s);
		n=s.size();
		for(i=0;i<n;i++)
		{
			s1="";
			for(j=i;j<n&&s[j]!=32;j++)
			{
				s1=s1+s[j];
			}
			vec.push_back(s1);
			i=j;
		}
		n=vec.size();
		for(i=0,j=n-1;i<n&&j>=0&&i<j;i++,j--)
		swap(vec[i],vec[j]);
			for(i=0;i<vec.size();i++)
		cout<<vec[i]<<" ";
		vec.clear();
		cout<<endl;
	}
	return 0;
}
