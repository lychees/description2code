#include<iostream>
#include<string>
#include<vector>
using namespace std;
int main()
{
	int n;
	string s,p="as";
    cin>>s>>n;
    vector<string> a;    
    for( int i=0; i < n; i++) {
        string temp;
        cin >> temp;
        a.push_back(temp);   
    }
    for(int i=0;i<n;i++)
    {
    	if(s==a[i].substr(0,s.length()))
    	{ if(p=="as") p=a[i];
    	  else if(p.length()>a[i].length()) p=a[i];
    	}
    }
    if(p=="as") cout<<s;
    else cout<<p;
    return 0;
}
