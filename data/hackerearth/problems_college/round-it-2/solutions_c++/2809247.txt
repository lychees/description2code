#include<bits/stdc++.h>

using namespace std;

void sol(int n,int b)
{
    int quo=n/b;
    if((n-(quo*b))<((quo+1)*b)-n)
        cout<<quo*b<<endl;
    else if((n-(quo*b))>=((quo+1)*b)-n)
        cout<<(quo+1)*b<<endl;
}
int main()
{
    int t;
    cin>>t;

    while(t--)
    {
        int n,b;
        cin>>n>>b;
        sol(n,b);


    }
}
