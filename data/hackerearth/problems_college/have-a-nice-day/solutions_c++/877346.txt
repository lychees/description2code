#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t,n;
    cin>>t;
    while(t--){
    	cin>>n;
    	int m[n],p[n];
    	for(int i=0;i<n;i++)scanf("%d",&m[i]);//cin>>m[i];
    	for(int i=0;i<n;i++)scanf("%d",&p[i]);//cin>>p[i];
    	sort(m,m+n);
    	sort(p,p+n);
    	long long int ans=0;
    	for(int i=0;i<n;i++)ans+=m[i]*p[i];
    	//cout<<ans<<endl;
    	printf("%lld\n",ans);
    }
    return 0;
}
