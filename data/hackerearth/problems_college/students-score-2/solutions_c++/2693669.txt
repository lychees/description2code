#include <iostream>
using namespace std;

int main()
{
    int n,i,maxEndHere,maxSoFar;
    int *a;
    
    cin>>n;
    
    	a = (int *)malloc(sizeof(int)*n);
    	for(i=0;i<n;i++){
    		cin>>a[i];
    	}
    	
    	maxEndHere = maxSoFar = 0;
    	for(i=0;i<n;i++){
    		maxEndHere += a[i];
    		if(maxSoFar<maxEndHere){
    			maxSoFar = maxEndHere;
    		}else if(maxEndHere<0){
    			maxEndHere = 0;
    		}
    	}
    	
    	if(maxSoFar>0){
    		cout<<maxSoFar<<endl;
    	}else{
    		cout<<0<<endl;
    	}
    
    
    return 0;
}
