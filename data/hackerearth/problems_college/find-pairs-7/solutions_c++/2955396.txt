#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--> 0)
	{
    long long int n;
    cin>>n;
    long long int a[n];
    for(int i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    long long int c=0;
    for(int i=0;i<n;i++)
    {
    	for(int j=i;j<n;j++)
    	{
    		if(a[i]==a[j])
    		{
    			c++;
    		}
    	}
    }
    cout<<c<<endl;
	}
}
