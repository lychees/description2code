#include <bits/stdc++.h>
using namespace std;

bool compare(int a, int b) 
{
    int d = abs(a) - abs(b);
    if (d == 0)
        return a < b;
    else
        return d < 0;
}

struct visitor {
    int start;
    int stay;
    int end;
};

int main()
{
    visitor arr[100005];
    int t;
    cin>>t;
    while(t--)
    {
    	vector<int> vec;
    	int N;
    	cin>>N;
    	for (int i = 0; i < N; i++) {
    		cin>>arr[i].start;
    	}
    	for (int i = 0; i < N; i++) {
    		cin>>arr[i].stay;
    	}
    	for (int i = 0; i < N; i++) {
    		arr[i].end=arr[i].start+arr[i].stay;
    	}
    	for (int i = 0; i < N; i++) {
    		vec.push_back(arr[i].start);
        	vec.push_back(-arr[i].end);   
    	}
    
    sort(vec.begin(), vec.end(), compare);
    
    long rooms = 0, maxRooms = 0;
    
    for (int i : vec) {
        if (i > 0) {
            rooms++;
            if (rooms > maxRooms)
                maxRooms = rooms;
        }
        else
            rooms--;
    }
    
    cout << maxRooms << endl;
    }
}
