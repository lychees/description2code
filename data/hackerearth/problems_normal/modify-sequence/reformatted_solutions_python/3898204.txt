N = int(raw_input())
arr = map(int, raw_input().split())
i = 0

for i in range(N-1):
    while arr[i] > 0 and arr[i+1] > 0:
        if arr[i] > arr[i+1]:
            arr[i] = arr[i] - arr[i+1]
            arr[i+1] = arr[i+1] - arr[i+1]
        else:
            arr[i+1] = arr[i+1] - arr[i]
            arr[i] = arr[i] - arr[i]

if all(x == 0 for x in arr):
    print 'YES'
else:
    print 'NO'
