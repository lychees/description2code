#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,i,n;
	cin>>t;

	for(i=0;i<t;i++)
	{

		cin>>n;

		if(n==1 || n==2)
			cout<<"1"<<endl;
		else
		{

		    int sum=2;
		    int count=1;
		    int steps=2;

		    while(sum<n)
		    {
			    steps++;
			    count++;
			    sum+=steps;
		    }

		    cout<<count<<endl;
		}

		//return 0;
	}
	return 0;
}
