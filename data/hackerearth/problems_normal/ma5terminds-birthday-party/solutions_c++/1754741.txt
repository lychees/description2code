typedef long long ll;

int not_prime[1000010];
int dp[1000010];

void markMultiples(int a){
    int i=2, num;
    while((num=i*a)<=1000001){
        i++;
        not_prime[num] = 1;
    }
}

void precalc(){
    for(int i=2;i<=1000001;i++){
        if(not_prime[i]==0)
            markMultiples(i);
    }
    not_prime[4] = 0;
    not_prime[1] = 1;
    dp[0] = 0;
    for(int i=1;i<=1000001;i++){
        dp[i] = dp[i-1];
        if(not_prime[i])
            dp[i]++;
    }
}

int main(){
    ios::sync_with_stdio(false);
    #ifdef DEBUG
    freopen("input.txt", "r", stdin);
    #endif // DEBUG

    int T,L,R;
    precalc();
    // for(int i=1;i<=20;i++)
    //     cout<<i<<" "<<not_prime[i]<<endl;
    scanf("%d", &T);
    while(T--){
        scanf("%d %d", &L, &R);
        printf("%d\n", dp[R] - dp[L-1]);
    }

    return 0;
}
