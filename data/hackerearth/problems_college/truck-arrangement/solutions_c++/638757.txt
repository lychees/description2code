#include<bits/stdc++.h>
using namespace std;
int main(){

	int t;
	cin >> t;
	while(t--){

		int n;
		cin >> n;
		vector<pair<int,int> > A(n);		
		for(int i=0;i<n;i++){
			cin >> A[i].first;
			A[i].second=i;
		}
		sort(A.begin(),A.end());
		int maxx=0;		
		for(int i=0;i<n;i++){
			maxx=max(maxx,abs(i-A[i].second));
		}
		cout << maxx << endl;
	}
	return 0;
}
