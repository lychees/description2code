#include <iostream>
using namespace std;

int main()
{
    long int i,j,k,l,n,a[100005];
    long long int c=0;
    cin>>n;
    
    for(i=0;i<n;i++)cin>>a[i];
    
    for(i=0;i<n;i++){
    	for(j=i+1;j<n;j++){
    		if(a[i]==a[j]){
    			for(k=i+1;k<j;k++){
    				for(l=j+1;l<n;l++){
    					if(a[k]==a[l] && a[k]!=a[i]) c++;
    				}
    			}
    		}
    	}
    }
    
    cout<<c<<endl;
    return 0;
}
