using namespace std;
typedef long int ll;
ll scan()
{
  char c = gc();
  while(c<'0' || c>'9') c = gc();
  ll ret = 0;
  while(c>='0' && c<='9') {
    ret = 10 * ret + c - 48;
    c = gc();
  }
  return ret;
}
int main()
{
	ll t = scan();
	while(t--)
	{
	ll n = scan();
	ll ans = 0, x;
	while(n--)
	{
		x = scan();
		ans ^= x;
	}
	cout<<ans<<"\n";
	}
	return 0;
}
