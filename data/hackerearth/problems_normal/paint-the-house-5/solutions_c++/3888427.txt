using namespace std;
struct node
{
    ll ti,xi,yi;
};
bool comp(node a,node b)
{
    if(a.ti<b.ti)
        return true;
    else if(a.ti==b.ti)
    {
        if(a.yi>b.yi)
            return true;
        else if(a.yi==b.yi)
        {
            if(a.xi<b.xi)
                return true;
            else
                return false;
        }
        else
            return false;
    }
    else
        return false;
}
int main()
{
    ll n,d;
    cin>>n>>d;
    node arr[n];
    for(ll i=0;i<n;i++)
    {
        cin>>arr[i].ti>>arr[i].xi>>arr[i].yi;
    }
    sort(arr,arr+n,comp);
    ll sp=0,prev_time=arr[0].ti,ar_cov=0,ans=0;
    for(ll i=0;i<n;i++)
    {
        ll cov=sp*(arr[i].ti-prev_time);
        ar_cov+=cov;
        if(ar_cov>=d)
        break;
        if(arr[i].yi>sp)
        {
            sp=arr[i].yi;
            ans+=arr[i].xi;
        }
        prev_time=arr[i].ti;

    }
    cout<<ans<<endl;
}
