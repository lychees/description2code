#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	int graph[n+1][n+1];
    	int color[n+1];
    	for(int i=1;i<=n;i++){
    		color[i] = 0;
    		for(int j=1;j<=n;j++)
    			graph[i][j] = 0;
    	}
    	
    	int edges;
    	cin>>edges;
    	for(int i=0;i<edges;i++ ){
    		int s , d;
    		cin>>s>>d;
    		s++;
    		d++;
    		graph[s][d] = 1;
    		graph[d][s] = 1;
    	}
    	
    	queue<int> q;
    	int col = 1;
    	color[1] = 1;
    	q.push(1);
    	while(!q.empty()){
    		int s = q.front();
    		q.pop();
    		for(int i=1;i<=n;i++){
    			if(graph[s][i]==1 && color[i]==0 && s!=i ){
    				q.push(i);
    				if(col==1)
    					color[i] = 2;
    				else
    					color[i] = 1;
    			}    			
    		}
    		if(col==1) col = 2;
    		else col = 1;
    	}
    	bool flag = false;
    	for(int i=1;i<=n;i++){
    		flag = false;
    		for(int j=1;j<=n;j++){
    			if(color[i]==color[j] && graph[i][j]==1 && i!=j){
    				flag = true;
    				break;
    			}    				
    		}
    		if(flag)
    			break;
    	}
    	
    	if(flag) cout<<"NOT BICOLORABLE."<<endl;
    	else cout<<"BICOLORABLE."<<endl;
    	
    	
    }
    return 0;
}
