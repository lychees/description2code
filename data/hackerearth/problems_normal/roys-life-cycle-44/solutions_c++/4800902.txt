#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	char a[900000];
	int k=0;
	int max_day=0,max_day_final=0;
    int max_tot=0,max_tot_final=0;
	while(n--)
	{
		char s[1445];
		cin>>s;
		int l=strlen(s);
		max_day=0;
		for(int i=0;i<l;i++)
		{
			a[k++]=s[i];
			if(s[i]=='C')
			{
				max_day+=1;
			}
			else
			{	
				if(max_day>max_day_final)
				max_day_final=max_day;
				max_day=0;
			}
			if(max_day>max_day_final)
			max_day_final=max_day;
		}
	}
	int co=0,ans=0;
	for(int i=0;i<k;i++)
	{
		if(a[i]=='C')
		{
			while(a[i]=='C'&&i<k)
			{
				co++;
				i++;
			}
		}
		if(co>ans)
		ans=co;
		co=0;
	}
	cout<<max_day_final<<" "<<ans;
	return 0;
}
