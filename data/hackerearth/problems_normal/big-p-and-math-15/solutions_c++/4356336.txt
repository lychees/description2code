#include <iostream>
#include <vector>
using namespace std;
int main()
{
	int x,z;
	cin >> x >> z;
	vector <int> a,aa,b,bb;
	
	while(x!=0)
	{
		int y= x%10;
		if(y==5||y==6)
		{
			a.push_back(5);
			aa.push_back(6);
		}
		else 
		{
			a.push_back(y);
			aa.push_back(y);
		}
		x/=10;
		
	}
	while(z!=0)
	{
		int y= z%10;
		if(y==5||y==6)
		{
			b.push_back(5);
			bb.push_back(6);
		}
		else 
		{
			b.push_back(y);
			bb.push_back(y);
		}
		z/=10;
	}
	
	long long q=0,w=0,e=0,r=0;
	long long t,m;
	for(int d=0,g=1;d<a.size();d++)
	{
		q+=(a[d]*g);
		g*=10;
	}
	
	for(int d=0,g=1;d<b.size();d++)
	{
		w+=(b[d]*g);
		g*=10;
	}
	m=q+w;
	for(int d=0,g=1;d<aa.size();d++)
	{
		e+=(aa[d]*g);
		g*=10;
	}
	for(int d=0,g=1;d<bb.size();d++)
	{
		r+=(bb[d]*g);
		g*=10;
	}
	t=e+r;
	cout << m <<" " << t << endl;
	
	
}
