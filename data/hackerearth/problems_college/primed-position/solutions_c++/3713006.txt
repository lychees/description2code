#include<bits/stdc++.h>
#define maxx 10000000
#define lim 1000000007
using namespace std;
long long a[maxx+1],b[maxx+1],c[maxx+1];
void seive()
{
    long  long i,j;
    a[0]=a[1]=-1;
    for(i=2;i<=sqrt(maxx);i++)
    {
        if(a[i]==0)
        {
            for(j=i*i;j<=maxx;j=j+i)
            a[j]=-1;
        }
    }
    j=0;
    for(i=2;i<=maxx;i++)
    {
        if(a[i]==0)
        {
            b[j++]=i;
        }
    }
    //cout<<j<<endl<<b[j-1]<<endl;
    long long l=0;
    for(i=0;i<j;i++)
    {
        if(a[i+1]==0)
        {
            c[l++]=b[i]%lim;
        }
    }
    //cout<<l<<endl<<c[l-1];
    /*for(i=0;i<l;i++)
        cout<<b[i]<<" ";
    cout<<endl;
    l=c.size();
    for(i=0;i<l;i++)
        cout<<c[i]<<" ";
    cout<<endl;*/
}
int main()
{
    seive();
     long  t;
     long long m,n,p;
     scanf("%ld",&t);
    while(t--)
    {
    	p=0;
    	scanf("%lld%lld",&m,&n);
        p=(c[m-1]*c[n-1])%lim;
        printf("%lld\n",p);
    }
    return 0;
}
