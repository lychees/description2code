def prime(p):
    count = 0
    for i in range(p):
        if i != 0 and i != 1:
            if p % i == 0:
                return 0
    return 1


b = []

for i in range(501):
    if i == 0:
        b.append(0)
    elif i == 1:
        b.append(0)
    else:
        if prime(i) == 1:
            count = count+1
        b.append(b[i-1]+pow(2, count-1))
