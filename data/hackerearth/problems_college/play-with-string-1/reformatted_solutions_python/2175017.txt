m = int(raw_input())
for j in xrange(m):

    s = raw_input()
    n = len(s)
    r = s[::-1]
    flag = 0
    for i in range(1, n):

        x = abs(ord(s[i])-ord(s[i-1]))
        y = abs(ord(r[i])-ord(r[i-1]))
        if x != y:

            flag = 1

    if (flag == 0):

        print "Funny"
    else:

        print "Not Funny"
