#include<bits/stdc++.h>
using namespace std;
#define FOR(i,a,b) for(int i = (a); i <= (b); ++i)
#define REP(i,n) FOR(i,0,(n)-1)
#define RI(i,n) FOR(i,1,(n))
typedef long long ll;
const int K = 25;
ll ways[1 << K], s[1 << K];
ll solve(int n) {
	REP(i, 1 << n) ways[i] = s[i] = 0;
	ways[0] = 1;
	REP(mask, 1 << n) REP(i, n)
		if(!(mask & (1 << i)))
			if(__builtin_popcount(mask) != i) {
				int bigger = 0;
				FOR(j, i + 1, n - 1) if(mask & (1 << j)) ++bigger;
				int m2 = mask | (1 << i);
				ways[m2] += ways[mask];
				s[m2] += s[mask] + bigger * ways[mask];
			}
	return s[(1 << n) - 1];
}
int main() {
	int z;
	scanf("%d", &z);
	while(z--) {
		int n;
		scanf("%d", &n);
		printf("%lld\n", solve(n));
	}
	return 0;
}
