#include<iostream>
#include<math.h>
#include<string.h>
#include<stdio.h>
#include<stdlib.h>
#include<iomanip>
#include<algorithm>
typedef long long int lli;
using namespace std;
int main() {

lli sum,t,n,i;
string s1,s2;
cin>>t;
while(t--)
{
    s1.clear();
    s2.clear();
    cin>>s1>>s2;sum=0;
   n=min(s1.size(),s2.size());
   for(i=0;i<n;i++)
    sum+=(s1[i]==s2[i]);
   cout<<sum<<'\n';

}





}
