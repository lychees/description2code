#include <iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t>0)
	{
		int n,done=0,count=0,cnt=0;
		cin>>n;
		int p=n;
		while(n>0 && done==0)
		{
			if(n%3==0)
			done=1;
			else 
			{
				n=n-5;
				count++;
			}
		}
		
		if(done)
		{
			int k=n;
			for(int i=0;i<k;i++)
			cout<<"5";
			for(int i=0;i<5*count+cnt;i++)
			cout<<"3";
			cout<<endl;
		}
		else
		{
			if(p%5==0)
			{
				for(int i=0;i<p;i++)
				cout<<"3";
				cout<<endl;
			}
			else
			cout<<"-1"<<endl;
		}
		t--;
	}
    return 0;
}
