#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		int n,m;
		cin >> n >> m;
		long long int tmp;
		set<long long int> s;
		for(int i=0;i<n;i++)
		{
			cin >> tmp;
			s.insert(tmp);
		}
		int val;
		for(int i=0;i<m;i++)
		{
			cin >> tmp;
			val=s.size();
			s.insert(tmp);
			if(s.size()==val)
				cout << "YES" << endl;
			else
				cout << "NO" << endl;
		}
	}
	return 0;
}
