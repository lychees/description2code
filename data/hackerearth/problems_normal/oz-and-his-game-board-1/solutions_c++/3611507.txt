#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
    long long a,b,c,n,m,t,z,p;
    scanf("%lld\n",&t);
    while(t--)
    {
    	scanf("%lld %lld %lld %lld %lld\n",&n,&m,&a,&b,&c);
    	if(2*c-(a+b)>0)
    	{
    		if(n>m)
    		z=m;
    		else
    		z=n;
    		p=n*a+m*b+(2*c-(a+b))*z;
    	}
    	else
    	p=n*a+m*b;
    	cout<<p<<endl;
    }
    return 0;
}
