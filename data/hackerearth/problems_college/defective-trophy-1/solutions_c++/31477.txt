#include<iostream>
#include<cstdio>
#include<string>
#include<cmath>
#include<vector>
#include<algorithm>
#include<sstream>
#include<cctype>

typedef long long LL;

#define FOR(i, a, b)   for(LL i=a;i<b;i++)
#define RFOR(i, a, b)  for(LL i=a-1;i>=b;i--)
#define REP(i, a)      FOR(i, 0, a)
#define RREP(i, a)     RFOR(i, a, 0)
#define PB(x)          push_back(x)
#define BP()           pop_back()
#define SZ()           size()
#define LEN()          length()
#define BG             begin()
#define ED             end()

using namespace std;

int main()
{
	LL t, n, ans;
	
	scanf("%lld", &t);
	
	while(t--){
		scanf("%lld", &n);
		
		ans = 0;
		LL x;
		
		while(n > 1){
			x = n % 3;
			if(x != 0)
				n += (3 - x);
				
			n = n/3;
			ans += 1;
		}	
		
		printf("%lld\n", ans);
	}
	
    return 0;
}
