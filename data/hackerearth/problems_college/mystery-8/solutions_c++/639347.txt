#include<iostream>
#include<string.h>
#include<cstdio>
using namespace std;
int main()
{
    int test_cases;
    cin>>test_cases;
    getchar();
    while(test_cases>0)
    {
        char s[101],str[101];
        int i,j,ind,l;
        gets(str);
        l = strlen(str);
        i=0;j=0;
        while(i<l)
        {
        	if((str[i]>='a'&&str[i]<='z')||str[i]=='*')
        	{
        		if(str[i]=='*')
        			ind = j;
        		s[j]=str[i];
        		j++;        		
        	}
        	i++ ;
        }
        s[j] = '\0';
        l = strlen(s);
        cout<<s[l-ind-1]<<endl;
        test_cases--;
    }
    return 0;
}
