int POW(int Base, int Exp)
{
	int y,ret=1;
	y=Base;
	while(Exp)
	{
		if(Exp&1)
				ret=(ret*y)%MOD;
		y = (y*y)%MOD;
		Exp/=2;
	}
	return ret%MOD;
}
