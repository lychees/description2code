#include <iostream>
#include <string.h>
#include <math.h>
#include <stdio.h>
using namespace std;
#define ll long long int
#define S(n) scanf("%d",&n)
#define Sll(n) scanf("%lld",&n)
ll a[1000010];
ll sum[1000010];
ll su(int  a, int b)
{
	if(a == 0) {
		return sum[b];
	} else {
		return sum[b] - sum[a-1];
	}
}
int main()
{
	
		int n;
		
		memset(sum,0,sizeof(sum));
		int i;
		S(n);
		int q;
		S(q);
		for(i = 0; i < n; i++) {
			Sll(a[i]);
		}
		sum[0] = a[0];
		for(i = 1;i < n; i++) {
			sum[i] = sum[i-1] + a[i];
		}
		for(i = 1;i <= q; i++) {
			int g1,g2,p1,p2;
			S(g1);
			S(p1);
			S(g2);
			S(p2);
			ll ans = 0;
			int a1,a2,b1,b2;
			if(g1 < g2) {
				a1 = g1;
				a2 = g2;
				
			} else {
				a1 = g2;
				a2 = g1;
			}
			if(p1 < p2) {
				b1 = p1;
				b2 = p2;
				
			} else {
				b1 = p2;
				b2 = p1;
			}
			int st;
			int en;
			if(a1 < b1) {
				st = a1;
				if(a2 < b1) {
					ans = su(st,a2) + su(b1,b2);
					//cout << st << " " << a2 << " " << b1 << " "<< b2 << endl;
				} else {
					if(a2 < b2) {
						ans = su(st,b2);
						//cout << st << " " << b2 << endl;
					} else {
						ans = su(st,a2);
						//cout << st << " " << a2 << endl;
					}
				}
			} else {
				st = b1;
				if(b2 < a1) {
					ans = su(st,b2) +  su(a1,a2);
				} else {
					if(b2 < a2) {
						ans = su(st,a2);
					} else {
						ans = su(st,b2);
					}
				}
			}
			printf("%lld\n",ans);
		
	}
}
