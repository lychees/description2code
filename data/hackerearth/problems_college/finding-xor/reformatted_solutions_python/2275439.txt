for y in range(input()):
    y = raw_input()
    z = raw_input()
    for x in range(len(y)):
        if ((y[x] is '0' and z[x] is '0') or (y[x] is '1' and z[x] is '1')):
            sys.stdout.write("0")
        else:
            sys.stdout.write("1")
    print ""
