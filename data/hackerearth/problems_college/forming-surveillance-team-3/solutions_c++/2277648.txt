#include <iostream>
using namespace std;

int main()
{
    int t,n,a[100002],x,c;
    cin>>t;
    while(t--)
    {
    	c=0;
    	for(int i=0;i<100002;i++)
    	a[i]=0;
    	cin>>n;
    	for(int i=0;i<n;i++)
    	{
    		cin>>x;
    		a[x]++;
    		
    	}
    	for(int i=0;i<100002;i++)
    	if(a[i]>0)
    	c++;
    	cout<<c<<"\n";
    }
    }
