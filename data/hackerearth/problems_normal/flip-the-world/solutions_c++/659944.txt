#include<stdio.h>
int main(){
	int n,m,t,count[60][60];
	char a[60][60];
	scanf("%d",&t);

	while(t--){
		scanf("%d %d",&n,&m);
		getchar();
		for(int i=0;i<n;i++){
				gets(a[i]);
		}
		/*
		for(int i=0;i<n;i++){
			for(int j=0;j<m;j++){
				printf("%c",a[i][j]);
			}		
			puts("");
		}
		*/

		for(int i=n-1;i>=0;i--){
			for(int j=m-1;j>=0;j--){
				if(i==n-1 && j==m-1)
					count[i][j]=0;
				else if(i==n-1)
					count[i][j]=count[i][j+1];
				else if(j==m-1)
					count[i][j]=count[i+1][j];
				else
					count[i][j]=count[i+1][j]+count[i][j+1]-count[i+1][j+1];
				if((count[i][j]%2==1 && a[i][j]=='1')||(count[i][j]%2==0 && a[i][j]=='0'))
					count[i][j]++;
			}
		}
		printf("%d\n",count[0][0]);
	}
	return 0;
}
