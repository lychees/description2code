#include <bits/stdc++.h>
using namespace std;
int main()
{
    int lenghtOfString;
    cin>>lenghtOfString;
    string String;
    cin>>String;
    long long int occurenceOfChars[26] = {0};
    for(int i=0; i<String.size(); i++) {
        occurenceOfChars[String[i]-'a']++;
    }
    int numberOfQueries;
    cin>>numberOfQueries;
    while(numberOfQueries--) {
        char X,Y;
        cin>>X>>Y;
        if(X == Y) {
            long long int answer = occurenceOfChars[X-'a']*(occurenceOfChars[X-'a']-1)/2;
            cout<<answer<<endl;
        }
        else {
            long long int answer = occurenceOfChars[X-'a']*occurenceOfChars[Y-'a'];
            cout<<answer<<endl;
        }
    }
    return 0;
}
