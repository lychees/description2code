#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
   int t,n,m,fg;
   //cin >> t;
   scanf("%d",&t);
   while(t--)
   {
   		//cin  >> n >> m;
   		scanf("%d %d",&n,&m);
   		int a[n],newa[n];
   		
   		for(int i=0;i<n;i++)
   		{
   			//cin >> a[i];
   			scanf("%d",&a[i]);
   		}
   		
   		for(int j=0;j<m;j++)
   		{
   			fg=0;
   			for(int i=0;i<n;i++)
   			{
   				
   				if(i==0)
   				{
   					if(a[i+1])
   					{
   						newa[i]=1;
   						fg=1;
   					}
   					else
   					newa[i]=0;
   				}
   				if(i>0 && i < n-1)
   				{
   					if(a[i-1] && a[i+1])
   					{
   						newa[i]=1;
   						fg=1;
   					}
   					else
   					newa[i]=0;
   				}
   				if(i==n-1)
   				{
   					if(a[i-1])
   					{
   						newa[i]=1;
   						fg=1;
   					}
   					else
   					newa[i]=0;
   				}
   			}	
   			
   		
//   				cout << endl;
   				
	   			for(int i=0;i<n;i++)
	   			a[i]=newa[i];
	   			if(!fg)
   				break;
   	
   		}
   	
		   		for(int i=0;i<n;i++)
		   		{
		   			//cout << a[i] << " "  ; 
		   			printf("%d ",a[i]);
		   		}
   		printf("\n");
   		//cout << endl ;
   }
}
