#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll val[40000];
int main()
{
    int t;
    ll n,ans,i,j,mod=1000000007;
    cin>>t;
    while(t--)
    {
        cin>>n;
        ans=0;
        j=sqrt(n);
        for(i=1;i<=j;i++)
        {
            val[i]=n/i;
            ans+=(i*val[i]);
            if(ans>=mod)
                ans%=mod;
        }
        for(i=1;i<j;i++)
        {
            ans+=i*(((val[i]-val[i+1])*(1+val[i]+val[i+1]))/2);
            if(ans>=mod)
                ans%=mod;
        }
        ans+=j*(((val[j]-j)*(1+val[j]+j))/2);
        ans%=1000000007;
        printf("%lld\n",ans);
    }
    return 0;
}
