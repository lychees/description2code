
import sys
# from copy import deepcopy

lines = sys.stdin.readlines()
# print(lines)

l = lines[1].rstrip()
data = map(int, l.split())


def is_even(n):
    return n % 2 == 0


def is_odd(n):
    return n % 2 != 0


zombies = filter(is_even, data)
vampires = filter(is_odd, data)

result = []


if zombies != []:
    zombies.sort()
    result = zombies
    result.append(sum(zombies))
if vampires != []:
    vampires.sort()
    result.extend(vampires)
    result.append(sum(vampires))

if result != []:
    str_result = map(str, result)
    print(" ".join(str_result))
else:
    print("")
