#include<bits/stdc++.h>
using namespace std;


int main() {
    int t,x,y,z,i,j,k;
    cin >> t;
    while(t--)
   {
      cin >> k;
      if(k%3==0) {
      	for(i=0;i<k;i++)
      		cout << "5";
      	cout << endl;
      }
      else {
      	 x=k%3;
         y=k-x;
      	
      	while(y>=0) {
      	  if(y%3==0 && x%5==0) {
      	  	for(i=0;i<y;i++)
      	  		cout << "5" ;
      	  	for(i=0;i<x;i++)
      	  		cout << "3";
      	  	cout << endl;
      	  	break;
      	  }
      	  y--;
      	  x++;
        }
        if(y<0)
      	cout << "-1" << endl;
      }

      
   } 
	return 0;
}
