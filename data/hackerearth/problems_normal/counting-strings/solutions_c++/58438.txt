#include<iostream>
#define l1 long long
#include<cstdio>
#include<cstring>
using namespace std;

int x[1000000];
int main()
{
	char a[1000001];
	l1 t;
	cin>>t;
	while(t--){
	      scanf("%s",a);
	      l1 l = strlen(a);
	      l1 count = 0;
	      l1 p = 1;
	      for(l1 i = l-1; i >=0; i--){
	      	if(a[i] == 'a' || a[i] == 'z'){
	      		x[count++] = p;
	      	}
	      	p++;
	      }
	      l1 ans = 0;
	      l1 m1 = l+1;
	      l1 x2;
	      for(l1 i = count-1; i >= 0; i--){
	      	x2 = m1-x[i];
	      	ans += x2*x[i];
	      	m1 = x[i];
	      }
	      cout<<ans<<"\n";
	}
	return 0;
}
