# print 'Hello World!'

n = input()
i = 0
while i < n:
    s = raw_input()
    p = s.split()
    j = 0
    while j < (len(p)/2):
        k = p[j]
        p[j] = p[len(p)-1-j]
        p[len(p)-1-j] = k
        j += 1
    print " ".join(p)
    i += 1
