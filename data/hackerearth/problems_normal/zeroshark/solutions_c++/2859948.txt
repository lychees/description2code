#include<bits/stdc++.h>
using namespace std;
long long int dp[4][100001],mod=1000000007;
int main()
{
    int i;
    dp[0][1]=1;
    dp[1][1]=1;
    dp[2][1]=0;
    dp[3][1]=0;
    for(i=2;i<=100000;i++)
    {
        dp[0][i]=dp[1][i-1];
        dp[1][i]=(dp[1][i-1]+dp[0][i-1])%mod;
        dp[2][i]=(dp[0][i-1]+dp[3][i-1])%mod;
        dp[3][i]=(dp[2][i-1]+dp[3][i-1])%mod;
    }
    long long int T,N;
    scanf("%lld",&T);
    while(T--)
    {
        scanf("%lld",&N);
        printf("%lld\n",(dp[2][N]+dp[3][N])%mod);
    }
}
