
for _ in xrange(int(raw_input())):
	ns, ks = raw_input().split()
	n, k = int(ns), int(ks)
	slips = map(int, raw_input().split())
	slips[0] = max(0, slips[0])
	for x in xrange(1, n):
		if x > k:
			slips[x] = max(slips[x - 1], slips[x - k - 1] + slips[x])
		else:
			slips[x] = max(slips[x - 1], slips[x])
	print slips[-1]
