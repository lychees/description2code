#include <iostream>
using namespace std;
main()
{
	int T,N,M,wealth[100],i,j,t,my;
	cin>>T;
	while(T--)
	{
		cin>>N>>M;
		for(i=0;i<N;i++)	cin>>wealth[i];
		my=wealth[0];
		for(i=0;i<N-1;i++)
			for(j=0;j<N-i-1;j++)
				if(wealth[j]<wealth[j+1])
				{
					t=wealth[j];
					wealth[j]=wealth[j+1];
					wealth[j+1]=t;
				}
		for(i=0;i<N&&wealth[i]-my;i++);
		cout<<i/M<<endl;
	}
}
