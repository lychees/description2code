#define mod 1000000007
#define ll long long int

ll power(ll a,ll b)
{
	ll ans=1,temp=a;
	while(b)
	{
		if(b&1)
		ans=(ans*temp)%mod;
		b/=2;
		temp=(temp*temp)%mod;
	}
	return ans;
}
