#include <iostream>
#include<math.h>
using namespace std;

int countbits(int n)
{
	int count=0;
	while(n)
	{
		n&=(n-1);
		count+=1;
	}
	return count;
}

int main()
{
   long long int t,a,b;
   cin>>t;
   while(t--)
   {
   		cin>>a>>b;
   		long long int ans= a^b;
   		ans=countbits(ans);
   		if((ans)%2==0)
   		{
   			cout<<"YES"<<endl;
   		}
   		else
   			cout<<"NO"<<endl;
   }
    return 0;
}
