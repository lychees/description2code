#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int r,c;
		cin>>r>>c;
		long o,e;
		if(r<=c)
		{
			e=o=0;
			for(int i=1;i<=r;i++)
			{
				if((i&1)==0)
				e+=i;
				else
				o+=i;
			}
			cout<<e<<' '<<o<<endl;
		}
		else
		{
			e=o=0;
			for(int i=1;i<=c;i++)
			{
				if((i&1)==0)
				e+=i;
				else
				o+=i;
			}
			for(int i=c+1;i<=r;i++)
			{
				if((i&1)==0)
				e+=c;
				else
				o+=c;
			}
			cout<<e<<' '<<o<<endl;
		}
	}
	return 0;
}
