#include<bits/stdc++.h>
#define Sameer_Gupta

using namespace std;


int main()
{
	int i,t,l,count;

	scanf("%lld",&t);
	while(t--)
	{
		char a[1000]={'\0'},b[1000]={'\0'};
		count=0;
		scanf("%s",&a);
		scanf("%s",&b);
		l=strlen(a);
		for(i=0;i<l;i++){
					if(a[i]=='\0' || b[i]=='\0')	break;
					else{
						if(a[i]==b[i])	count++;
					}
		}
		printf("%d\n",count);
	}
	return 0;
}
