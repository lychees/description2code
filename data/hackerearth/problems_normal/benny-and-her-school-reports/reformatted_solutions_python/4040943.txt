T = int(raw_input())
for test_case in range(T):
    N, M, X = [int(token) for token in raw_input().split()]
    required = (N + 1) * X

    scores = [int(token) for token in raw_input().split()]
    total = sum(scores)

    diff = abs(total - required)

    if total >= required:
        print 1
    elif diff > M:
        print "Impossible"
    else:
        print diff
