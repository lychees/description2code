#include<bits/stdc++.h>
using namespace std;
#define MAXX 25005
#define MAX(a,b) (((a)>(b))?(a):(b))
int a[MAXX];
int b[MAXX];
long long int dp[MAXX];
int main()
{
	int t, i, n;
	scanf("%d", &t);
	while (t--)
	{
		scanf("%d", &n);
		for (i = 1; i <= n; i++)
			scanf("%d", &a[i]);
		dp[0] = 0;
		dp[1] = a[1];
		for (i = 2; i < n; i++)
		{
			dp[i] = MAX(dp[i - 2] + a[i], dp[i - 1]);
		}
		long long ans1 = dp[n - 1];
		memset(dp, 0, sizeof(dp));
		dp[2] = a[2];
		for (int i = 3; i <= n; i++) {
			dp[i] = MAX(dp[i - 2] + a[i], dp[i - 1]);
		}
		long long ans2 = dp[n];
		printf("%lld\n", MAX(ans1, ans2));
	}
	return 0;
}
