#include <iostream>
#include<cmath>
using namespace std;
int A[1024][1024];
void func(int fd,int s,int a,int b){
  if(fd>1){
      int p=pow(2,fd-1)*pow(2,fd-1);
      int k=pow(2,fd)/2;
      func(fd-1,s,a,b);
      func(fd-1,s+p,a,b+k);
      func(fd-1,s+p*2,a+k,b);
      func(fd-1,s+p*3,a+k,b+k);
  }
   else if(fd==1){
       A[a][b]=s;
       A[a][b+1]=s+1;
       A[a+1][b]=s+2;
      A[a+1][b+1]=s+3;
   }
  }
int main()
{
  int fd;
  cin>>fd;
  int g=pow(2,fd);
  func(fd,1,0,0);
  for(int i=0;i<g;i++){
   for(int j=0;j<g;j++){
       cout<<A[i][j]<<" ";
   }
   cout<<endl;
  }
}
