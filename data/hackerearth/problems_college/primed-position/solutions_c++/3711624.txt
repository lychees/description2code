#include <bits/stdc++.h>

using namespace std;
const int N = 1e6 + 5e5, mod = 1e9 + 7;
bool is[N];

int32_t main() {
    vector<int> p, pp;
    p.push_back(1);
    pp.push_back(1);
    for (int i = 2; i < N; i++) if(!is[i]){
        p.push_back(i);
        for (int j = i + i; j < N; j += i)
            is[j] = 1;
    }
    for (int i = 2; i < p.size(); i++) if (!is[i])
        pp.push_back(p[i]);
    int t; scanf("%d", &t);
    while (t--) {
        int x, y; scanf("%d%d", &x, &y);
        printf("%d\n", (pp[x] *1ll* pp[y]) % mod);
    }
}
