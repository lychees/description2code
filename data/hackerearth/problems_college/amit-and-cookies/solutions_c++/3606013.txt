#include <iostream>
using namespace std;
int main()
{
    int N,K,sub,temp,maxSub,lastSwap,purchase=0;
    char sentinel = 'Y';
    cin >> N >> K;
    int cookies[N];
    for (int i=0;i<N;i++)
    			cin >> cookies[i];
    maxSub = N-1;
    while (sentinel == 'Y')
    {
    	sentinel = 'N';
    	sub = 0;
    	while (sub < maxSub)
    	{
    		if (cookies[sub]>cookies[sub+1])
    		{
    			temp = cookies[sub];
    			cookies[sub] = cookies[sub+1];
    			cookies[sub+1] = temp;
    			sentinel = 'Y';
    			lastSwap = sub;
    		}
    		sub++;
    	}
    	maxSub = lastSwap;
    }
    sub = 0;
    while (K>= cookies[sub])
    {
    	K=K-cookies[sub];
    	sub++;
    	purchase++;
    }
    cout << purchase;
    return 0;
}
