#include <iostream>
#include<string>
using namespace std;
bool Contains21(long int N)
{
	string s=to_string(N);
	int count=0;
	for(int i=0; i<s.length(); i++)
	{	
			if(s.substr(i,2)=="21")
				count++;
	}
	if(count==0)	return false;
	return true;
}
int main()
{
    int T;
    cin>>T;
    long int N[T];
    for(int i=0; i<T; i++)
    	cin>>N[i];
    for(int i=0; i<T; i++)
    {
    	if(N[i]%21==0 || Contains21(N[i]))
    		cout<<"The streak is broken!"<<endl;
    	else
    		cout<<"The streak lives still in our heart!"<<endl;
    }
    return 0;
}
