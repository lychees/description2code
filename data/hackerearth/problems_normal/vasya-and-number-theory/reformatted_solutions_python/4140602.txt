N = int(raw_input())
arr = sorted([int(elem) for elem in raw_input().split()])
countArr = [1 for i in xrange(N)]
out = 1
for i in xrange(len(arr)):
    maxCount = 0
    for j in xrange(i):
        if (arr[i] % arr[j] == 0):
            maxCount = max(maxCount, countArr[j])
    countArr[i] = maxCount+1
    out = max(out, countArr[i])
if (out == 1):
    print -1
else:
    print out
