#include <iostream>
#include <math.h>
using namespace std;

int main() 
{
	long long int n, t, sum=0, sum1=0, i=1, l=1, x, y;
	cin>>t;
	while(t--)
	{
		cin>>n;
		while(sum<n)
		{
			sum1=sum;
			sum+=(i*pow(2,i));
			i++;
		}
		sum=n-sum1;
		i--;
		x=sum/i;
		y=(i-sum%i);
		if(y==i)
		{
			x--;
			y=0;
		}
		l=l<<y;
		if((l&x)==0)
		{
			cout<<"Hacker\n";
		}
		else
		{
			cout<<"Earth\n";
		}
		l=i=1;
		sum=sum1=0;
	}
	return 0;
}
