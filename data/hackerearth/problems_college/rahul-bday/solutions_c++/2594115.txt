#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;
int main() {
   long long int b,w,x,y,z,tc;
    cin>>tc;
    long long int res=0;
    while(tc--)
    {   cin>>b>>w;
        cin>>x>>y>>z;
        if(x+z<y)
        {   res=((b*x)+(w*(x+z)));
            cout<<res<<endl;
        }
        else if(y+z<x)
        {   res=((w*y)+(b*(y+z)));
            cout<<res<<endl;
        }
        else
        {   res=((b*x)+(w*y));
          cout<<res<<endl;
        }
        
    }
    return 0;
}
