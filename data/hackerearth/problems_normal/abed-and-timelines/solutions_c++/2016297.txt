#include<stdio.h>
#include<stdlib.h>
int main()
{
    int t,p;
    scanf("%d",&t);
    long m,n,x1,x2,y1,y2,a,b,c;
    while(t--)
    {
    	c=0;
    	scanf("%ld%ld%ld%ld%ld%ld%d",&m,&n,&x1,&y1,&x2,&y2,&p);
    	a=abs(x1-x2);
    	b=m-abs(x1-x2);
    	if(a<b)
    		c=c+a;
    	else
    		c=c+b;
    	a=abs(y1-y2);
    	b=n-abs(y1-y2);
    	if(a<b)
    		c=c+a;
    	else
    		c=c+b;
    	c=c*p;
    	printf("%ld\n",c);
    	if(c<=1000)
    		printf("You saved the group.\n");
    	else
    		printf("Let go of the group.\n");
    }
    return 0;
}
