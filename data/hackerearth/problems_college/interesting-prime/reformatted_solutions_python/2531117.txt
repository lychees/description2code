import sys

T = int(sys.stdin.readline().strip('\n'))
while T:
    T -= 1
    N = int(sys.stdin.readline().strip('\n'))
    n = int(N ** 0.5) + 1

    fact1 = 0
    fact2 = 0
    for i in range(2, n):
        if i*i == N:
            fact1 += 1
            continue
        if N % i == 0:
            fact2 += 1
    print(fact1 + (fact2 << 1) + 1)
