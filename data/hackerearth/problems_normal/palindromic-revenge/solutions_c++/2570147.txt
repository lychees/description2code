#include <iostream>
#include <cstring>
#include <cmath>
using namespace std;

int main()
{
    int t;
    cin>>t;
    char str[100000];
    while(t--)
    {
    	cin>>str;
    	int len = strlen(str);
    	int cnt=0;
    	for(int i =0;i<len/2;i++)
    		{
    			cnt += abs(int(str[i])-int(str[len-i-1]));
    		}
    	cout<<cnt<<endl;
    }
}
