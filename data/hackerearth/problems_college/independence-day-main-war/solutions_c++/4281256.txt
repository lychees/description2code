#include<iostream>
#include<iomanip>
using namespace std;
int count =0;
int main()
{
	int n,num;
	float x,y,x1=0.0,y1=0.0,div;
	cin>>n;
	while(n--)
	{
		cin>>num;
		div=num;
		while(num--)
		{
			cin>>x>>y;
			x1 = x1+x;
			y1 = y1+y;
		}
		cout<<"("<<setprecision(2)<<fixed<<x1/div<<" , "<<setprecision(2)<<fixed<<y1/div<<")"<<endl;
		x1=y1=div=0.0;
		
	}
}
