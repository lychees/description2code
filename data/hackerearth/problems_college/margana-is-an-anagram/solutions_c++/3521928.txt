#include<bits/stdc++.h>
using namespace std;


#define FOR(i,a,b) for(long long i=a;i<b;i++)
#define cin(a) scanf("%d",%a);
#define cout(a)scanf("%lld\n",a);
#define DRT() int t;cin>>t;while(t--)


short int a[9999999];
int main()
{
	

	DRT()
	{
		string a,b;
		cin>>a>>b;
		int l = a.size();
		int l1 = b.size();

		int has[200]={0};

		FOR(i,0,l)has[a[i]]++;
		FOR(i,0,l1)has[b[i]]--;

		int f=0;
		FOR(i,0,200)
		{
			if(has[i]!=0)
			{
				f=1;
				break;
			}
		}
		if(f)cout<<"No"<<endl;
		else cout<<"Yes"<<endl;
	}











	return 0;
}
