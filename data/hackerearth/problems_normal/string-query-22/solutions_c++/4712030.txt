#include <bits/stdc++.h>
using namespace std;

string s;
int q,x;
char c;
map<char, vector<int>>m;
set<pair<int,char>> se;

int main(){
	cin>>s;
	for(int i=0;i<s.size();i++)
		m[s[i]].push_back(i);
	cin>>q;
	while(q--){
		cin>>x>>c;
		auto it=m[c].begin();
		m[c].erase(it+x-1);
	}
	for(auto i:m)
		for(auto j:i.second)
			se.insert(make_pair(j,i.first));
	for(auto i:se)
		cout<<i.second;
		return 0;
}
