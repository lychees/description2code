#include<bits/stdc++.h>
#define lim 2000000
using namespace std;
long long hsh[2000001]={0};
int main(){

               vector<long long> vec ;
               int i , j;
                vec.push_back(1);
                map<long long,long long> mymap;
            for( i = 2  ; i <= 1000  ; ++i ){
                       long long val = i , x = i;
                 for( j = 2 ; j <= 20 ; ++j )
                 {
                       x = x*val;
                      if(x <= lim)
                      {
                      	  
                       if(mymap[x] == 0)
                       vec.push_back(x);
                       mymap[x] = 1;
                      }
                       else
                       break;
                 }

            }
            for(int i  = 0 ; i < vec.size() ; ++i ){
               for(int j=  i ; j < vec.size() ; ++j){
                  long long val  = vec[i]+vec[j];
                    if(val <= lim)
                      hsh[val]=1;
                      else
                      break;
               }
            }
           int t;
              cin >> t;
              while(t--)
              {

                 int n;
                 cin >> n ;
                   if(hsh[n] == 1)
                   cout <<"Yes" <<endl;
                   else
                   cout <<"No"<<endl;
              }
                return  0;

}
