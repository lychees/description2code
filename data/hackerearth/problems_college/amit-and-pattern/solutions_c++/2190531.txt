

#include<bits/stdc++.h>
#include<iostream>
//#include<stdbool>
using namespace std;

bool isvalid(int g, int s,int Row,int Col)
{
    if (g < 0 || g >= Row || s >= Col || s < 0) return false;
    return true;
}

void print_DG(int Array[][100],int Row,int Col)
{
    for (int w = 0; w < Col; w++)
    {
        cout << Array[0][w] << " ";
        int g = 1;    int s = w-1;

        while (isvalid(g,s,Row,Col))
        {
            cout << Array[g][s] << " ";
            g++;
            s--;
        }
        cout << endl;
    }

   for (int w = 1; w < Row; w++)
    {
        cout << Array[w][Col-1] << " ";
        int g = w+1; // set row index for next point in diagonal
        int s = Col-2; // set column index for next point in diagonal

        while (isvalid(g,s,Row,Col))
        {
            cout << Array[g][s] << " ";
            g++;
            s--; // move in upright direction
        }
        cout << endl;
    }
}

// Driver program to test above
int main()
{

    int k,l,row,col;
    cin>>k>>l;
    int Array[100][100];
    for(int t=0;t<k;t++)
        for(int u=0;u<l;u++)
        cin>>Array[t][u];
    row=k;col=l;
    print_DG(Array,row,col);
    return 0;
}
