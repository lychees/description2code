#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,n,m,x,a,b;
    scanf("%d",&t);
    pair<int,int> p[40000];
    set<int> s;
    while(t--)
    {
        s.clear();
        scanf("%d%d",&n,&m);
        for(int i=0;i<m;i++)
        {
            scanf("%d%d",&a,&b);
            p[i].first=a;
            p[i].second=b;
        }
        for(int i=0;i<n-1;i++)
        {
            scanf("%d",&x);
            a=p[x-1].first;
            b=p[x-1].second;
            s.insert(a);
            s.insert(b);
        }
        if(s.size()==n)
            printf("YES\n");
        else
            printf("NO\n");
    }
    return 0;
}
