pair<int,pii> p[N];
struct comp{
	bool operator()(pi3 a, pi3 b){
		if(a.F<b.F) return true;
		else if(a.F==b.F){
			if(a.S.S>b.S.S) return true;
			if(a.S.S==b.S.S) return a.S.F<b.S.F;
			return false;
		}
		return false;
	}
};

struct comp3{
	bool operator()(pi3 a, pi3 b){
		if(a.S.S>b.S.S) return true;
		else if(a.S.S==b.S.S) return a.S.F<b.S.F;
		else return false;
	}
};

bool comp2(pi3 a, pi3 b)
{
	if(a.F<b.F) return true;
	else if(a.F==b.F){
		if(a.S.S>b.S.S) return true;
		if(a.S.S==b.S.S) return a.S.F<b.S.F;
		return false;
	}
	return false;
}

set<pi3,comp> st;
set<pi3,comp3> st2;
int main()
{
	int t,x,y,last=0;
	lli n,d;
	sc(n,d);
	Fore(i,1,n){
		sc(p[i].F,p[i].S.F,p[i].S.S);
	}
	sort(p+1,p+n+1,comp2);
	lli ans=0;
	Fore(i,1,n){
		if(p[i].F!=last){
			st.insert(p[i]);
			last=p[i].F;
		}
	}
	//cout<<"a"<<endl;
	Fore(i,1,N-5){
		if(d<=0) break;
		if(!st.empty()){
			auto it=st.begin();
			if(it->F<=i){
				pi3 x=*it;
				st2.insert(x);
				st.erase(it);
			}
		}
		if(!st2.empty()){
			auto it2=st2.begin();			
			pi3 x=*it2;
			d-=x.S.S;
			ans+=x.S.F;
