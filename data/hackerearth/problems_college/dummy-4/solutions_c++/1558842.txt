#include <bits/stdc++.h>
using namespace std;

int sum_of_digit(long long int n)
{
	int a=0;
	while(n)
	{
		a+=n%10;
		n/=10;
	}
	return a;
}
int main()
{
    int n;
    cin>>n;
    long long int sum=0;
    char str[400];
    while(n--)
    {
    	cin>>str;
    	for(int i=0;str[i];i++)
    	{
    		sum+=str[i]-'0';
    	}
    }
    long long int temp=sum;
    while(5)
    {
    	temp=sum_of_digit(temp);
    	if(temp<10) 
    		break;
	}
    cout<<temp;
    return 0;
}
