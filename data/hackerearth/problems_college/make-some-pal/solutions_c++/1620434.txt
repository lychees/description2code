#include <iostream>
using namespace std;

bool palin(const string& s,int l,int r)
{
	if(l==r) return true;
	while(l<r)
	{
		if(s[l]!=s[r]) return false;
		l++;r--;
	}
	return true;
}
int main()
{
    int t;
    string s;
    scanf("%d",&t);
    while(t--)
    {
    	cin>>s;
    	int l=s.length();
    	int ans=0;
    	for(int i=0;i<l;++i)
    	{
    		if(palin(s,i,l-1)){ ans+=l-i; break;}
    		ans+=2;
    		//printf("%d\n",ans);
    	}
    	printf("%d\n",ans);
    }
    return 0;
}
