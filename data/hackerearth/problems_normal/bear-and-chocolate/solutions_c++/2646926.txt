#include <bits/stdc++.h>
#define endl '\n'

using namespace std;

int tests;
int n,a[1024][1024];
int all;

int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int i,j;
	char c;
	
	cin>>tests;
	while(tests--) {
		cin>>n;
		all=0;
		for(i=1;i<=n;i++) {
			for(j=1;j<=n;j++) {
				cin>>c;
				if(c=='#') a[i][j]=1;
				else a[i][j]=0;
				all+=a[i][j];
				a[i][j]+=a[i-1][j];
				a[i][j]+=a[i][j-1];
				a[i][j]-=a[i-1][j-1];
			}
		}
		
		if(all&1) {
			cout<<"NO"<<endl;
			continue;
		}
		
		bool f=false;
		for(i=1;i<=n && f==false;i++) {
			if(a[i][n]==(all/2)) f=true;
		}
		
		for(j=1;j<=n && f==false;j++) {
			if(a[n][j]==(all/2)) f=true;
		}
		
		if(f==true) cout<<"YES"<<endl;
		else cout<<"NO"<<endl;
	}
	
	return 0;
}
