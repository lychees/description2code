#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{int t,i,j,n,we,dp[1001][1001];
int w[1001],v[1001];
    cin>>t;
    while(t--){
    	cin>>n>>we;
    	for(i=1;i<=n;i++)
    	cin>>w[i];
    	for(i=1;i<=n;i++)
    	cin>>v[i];
    
    	for(i=1;i<=n;i++)	
    	dp[i][0]=0;
    	for(i=0;i<=we;i++)
    	dp[0][i]=0;
    	
    	
    	for(j=1;j<=we;j++)
    	{
    		for(i=1;i<=n;i++)
    		{
    			if(w[i]<=j)
    			dp[i][j]=max(dp[i-1][j],dp[i-1][j-w[i]]+v[i]);
    			else 
    			dp[i][j]=dp[i-1][j];
    		}
    	
    	}
    cout<<dp[n][we]<<endl;	
    	
    }
    return 0;
}
