#include <iostream>
using namespace std;

int main()
{
    int t, n;
    cin >> t;
    while(t--)
    {
    	cin >> n;
    	int c = 0;
    	for(int i = 1; i <= n; i++)
    	{
    		for(int j = i; j <= n; j++)
    		{
    			if(i != j)
    			{
    				if((i ^ j) <= n)
    				  c++;
    				 
    			}
    		}
    	}
    	cout << c << endl;
    }
    
    return 0;
}
