#include <iostream>
#include <stdio.h>
#include <vector>
using namespace std;

#define MAX 1000007

vector<int> v;

int gcd(int a, int b)
{
	if(b == 0)
	{
		return a;
	}
	else
	{
		return gcd(b, a % b);
	}
}

int prime[MAX];

void Seive()
{
	int i, j;
	for(i=0;i<MAX;++i)
	{
		prime[i] = 1;
	}
	prime[0] = 0;
	prime[1] = 0;
	for(i=2;i*i<=MAX;++i)
	{
		if(prime[i])
		{
			for(j=i*i;j<=MAX;j+=i)
			{
				prime[j] = 0;
			}
		}
	}
	for(i=2;i<MAX;++i)
	{
		if(prime[i])
		{
			v.push_back(i);
		}
	}
}

int main()
{
	Seive();
    int T;
    scanf("%d", &T);
    while(T--)
    {
    	int i, n, count = 0;
    	scanf("%d", &n);
    	double p = 1;
    	for(i=0;v[i] <= n;++i)
    	{
    		if(n % v[i] == 0)
    		{
    			p = p * (1 - (1.00 / v[i]));
    		}
    	}
    	printf("%.0lf\n", n * p);
    }
    return 0;
}
