#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long int t,d1,d2;cin>>t;
    while(t--)
    {
        long long int n1,n2,c=0;
        cin>>n1>>n2;
        long long int x=n1,y=n2;
        while(x!=0 and y!=0)
        {
             d1=x%10;d2=y%10;
             if((d1+d2)>=10)
             {
                c++;
             }
             x=x/10;y=y/10;
        }
        if(c==0)
        {
            cout<<"No carry operation"<<endl;
        }
         else if(c==1)
         {
             cout<<c<<" carry operation"<<endl;
         }
         else
         {
             cout<<c<<" carry operations"<<endl;
         }
    }
     return 0;
}
