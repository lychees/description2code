#include <bits/stdc++.h>
using namespace std;

int main() {

   int t;
   cin>>t;
   while(t--)
   {
   long long int n, i=1,r=0;
   scanf("%lld",&n);
   while(i <= sqrt(n))
    {
        if(n%i==0) {
        	
        	if(i%2==0)
        	{ r+=1;
        //	cout<<i<<"\n";
        	}
        	
            if (i != (n / i) && (n/i)%2 == 0) 
            { 
            //	cout<<n/i<<"\n";
                r+=1;
            }
        } 

        i++;
    }
    cout<<r<<"\n";
    }
    
	return 0;
}
