#include <cstdio>
#include <algorithm>

const uint C = 1000000;
uint lst[C + 1];
uint ans[C + 1];

int main()
{
    std::fill_n(lst, C + 1, 0);
    std::fill_n(ans, C + 1, 0);
    lst[0] = 0, lst[1] = 1;
    for(uint i = 2; i <= C; ++i){
    	if(!lst[i]){
    		ans[i] = 1; lst[i] = i;
    		uint j;
    		if (C/i < i) continue;
    		j = i * i;
    		while(j <= C){
    			if(!lst[j]){
    				lst[j] = i;
    				++ans[i];
    			}
    			j += i;
    		}
    	}
    }
    int T; scanf("%d", &T);
    while(T-->0){
    	uint n; scanf("%u", &n);
    	printf("%u\n", ans[n]);
    }
    return 0;
}
