#include <iostream>
#define mod 1000000009
using namespace std;
long long int fastpow(long long int a,long long int b)
{
long long int res=1;
while(b)
{
	if(b&1)
	res=(res*a)%mod;
	a=(a*a)%mod;
	b=b/2;
}
	return res%mod;
}	
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	long long int n;
    	cin>>n;
    //	if(n==1||n==2)
    //	cout<<26<<endl;
    //	else
    	cout<<fastpow(26,(n+1)/2)<<endl;
    	
    	
    }

}
