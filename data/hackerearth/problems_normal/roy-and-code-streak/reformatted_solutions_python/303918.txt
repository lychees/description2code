# print 'Hello World!'
t = int(raw_input())
for i in range(t):
    n = int(raw_input())
    maxi = 0
    count = 0
    presid = set()
    for j in range(n):
        q = raw_input().split()
        sid, result = int(q[0]), int(q[1])
        if sid not in presid and result == 1:
            count += 1
            presid.add(sid)
            if count > maxi:
                maxi = count
        if result == 0:
            if maxi < count:
                maxi = count
            count = 0
    print maxi
