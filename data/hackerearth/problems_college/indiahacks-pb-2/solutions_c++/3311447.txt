#include<bits/stdc++.h>
using namespace std;
void rev(char str[])
{
	int i,j,temp;
	i=0;
   	j=strlen(str)- 1; 
   while(i<j){
      temp=str[i];
      str[i]=str[j];
      str[j]=temp;
      i++;
      j--;
   }
}
int main()
{
	int l,t,i,j,flag=0,p;
	char x;
	scanf("%d", &t);
	char s[t][25];
	for(i=0;i<t;i++)
		scanf("%s", s[i]);
	for(i=0;i<t;i++)
	{
		for(j=i+1;j<t;j++)
		{
			rev(s[j]);
			p=strlen(s[i])/2;
			if(strcmp(s[i],s[j])==0){
				x=s[i][p];
				flag=1;
				l=strlen(s[i]);
				break;
			}
		}
		if(flag==1)
			break;		
	}
	printf("%d %c",l, x);
	return 0;	
}
