#include<cstdio>
#include<cmath>
#define ll long long
int main()
{
    ll num,nd,sum,tenpo,d,t;
    scanf("%lld",&t);
    while(t--)
    {
      scanf("%lld",&num);
      nd=1+(ll)log10(num);
      sum=0;
      tenpo=1;
      for(d=1;d<nd;++d)
      {
        sum+=9*tenpo*d;
        tenpo*=10;
      }
      sum+=nd*(num+1-tenpo);
      printf("%lld\n",sum);	
    }
    return 0;
}
