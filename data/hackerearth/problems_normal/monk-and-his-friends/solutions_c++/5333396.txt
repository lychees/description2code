#include<iostream>
#include<map>
using namespace std;
int main()
{
	map<long long int,long int> ma;
	long long int t,n,m,a;
	cin>>t;
	while(t--)
	{
		cin>>n>>m;
		for(int i=0;i<n;i++)
		{
			cin>>a;
			ma[a]++;
		}
		for(int i=0;i<m;i++)
		{
			cin>>a;
			if(ma[a]==0)
				cout<<"NO\n";
			else
				cout<<"YES\n";
			ma[a]++;
		}
		ma.clear();
	}
	return 0;
}
