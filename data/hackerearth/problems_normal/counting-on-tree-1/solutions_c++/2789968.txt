#include<bits/stdc++.h>
using namespace std;

#define REP(i,a,b) for(int i=a;i<b;i++)
#define rep(i,n) REP(i,0,n)

#define ll long long
#define ull unsigned ll
ll mod =1000000007;

vector<int> g[50009];
int n,k;
int a[50009];
ll dp[50009][110];
ll ans=0;
void norm(ll &num)
{
  if(num>=mod)
    num%=mod;
  while(num<0)
    num+=mod;
}
void dfs(int u,int p)
{
    if(a[u]==1)
    dp[u][1]=1;
    else
    dp[u][0]=1;
    ll temp[110];

    for(int i=0;i<g[u].size();i++)
    {
        int v = g[u][i];
        if(v==p)continue;
        dfs(v,u);
        for(int j=0;j<=k;j++)
        {
            temp[j]=0;
            for(int l=0;l<=j;l++)
            {
                temp[j] = (temp[j]+(1LL*dp[u][l]*dp[v][j-l]));
                norm(temp[j]);
            }
        }
        for(int j=0;j<=k;j++)
        {
            dp[u][j] = (dp[u][j]+temp[j]);
            norm(dp[u][j]);
        }
    }
    ans = (ans+dp[u][k]);
    norm(ans);
}
int main(){
    //#ifndef ONLINE_JUDGE
    //freopen("input.txt","r",stdin);
    //#endif // ONLINE_JUDGE
    int t;
    scanf("%d",&t);
    while(t--)
    {
        ans=0;
        scanf("%d%d",&n,&k);
        for(int i=0;i<=n;i++)
            for(int j=0;j<=100;j++)
            dp[i][j]=0;
        for(int i=0;i<n;i++)
        {
            g[i].clear();
            scanf("%d",&a[i]);
        }
        for(int i=0;i<n-1;i++)
        {
            int u,v;
            scanf("%d%d",&u,&v);
            u--;v--;
            g[u].push_back(v);
            g[v].push_back(u);
        }
        dfs(0,-1);

        //for(int i=0;i<n;i++)
          //  ans = (ans+dp[i][k])%mod;
        printf("%lld\n",ans%mod);
    }

    return 0;
}
