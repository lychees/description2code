#include<bits/stdc++.h>
using namespace std;
int main()
{
 int t;
 cin>>t;
 while(t--)
 {
    int n;
    cin>>n;
    int arr[n][n];
    for(int i = 0; i < n;i++)
    {
        for(int j = 0; j < n;j++)
        {
            cin>>arr[i][j];
        }
    }
    int mark[51][51] = {0};
    if(arr[0][0])
    mark[0][0] = 1;
    for(int i = 1;i<n;i++)
    {
        if(mark[i-1][0] && arr[i][0])
        mark[i][0] = 1;
    }
    for(int i = 1;i<n;i++)
    {
        if(mark[0][i-1] && arr[0][i])
        mark[0][i] = 1;
    }
    for(int i = 1;i<n;i++)
    {
        for(int j = 1;j<n;j++)
        {
            if( (mark[i-1][j] || mark[i][j-1]) && arr[i][j])
            mark[i][j] = 1;
        }
    }
    if(mark[n-1][n-1])
    cout<<"POSSIBLE\n";
    else
    cout<<"NOT POSSIBLE\n";
 }
}
