 
            #include<bits/stdc++.h>
            #include<iostream>
            #define ll long long int
            using namespace std;
            int main()
            {
            ll t;
            cin>>t;
            for(ll i=1;i<=t;i++)
            {
            ll a,b,c;
            cin>>a>>b>>c;
            ll w[]={a,b,c};
            for(int j=0;j<3;j++)
            {
            for(int k=j+1;k<3;k++)
            {
            if(w[j]<w[k])
            {
            ll p=w[j];
            w[j]=w[k];
            w[k]=p;
            }
            }
            }
            if(w[0]!=w[1])
            {
            ll r=(w[1]*2)+1;
            cout<<r<<endl;
            }
            else if(w[0]==w[1])
            cout<<2*(w[0])<<endl;
            else if(w[0]==0 && w[1]==0 && w[2]==0)
            cout<<"0"<<endl;
        }
           return 0;
           }
