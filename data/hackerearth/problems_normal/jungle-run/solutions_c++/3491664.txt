int shortest_dis(vector<int> p[],int src,int des,int visited[])
{
	int m=INF;

	if(src == des)
		return 0;

	visited[src]=1;

	vector<int>::iterator it;

	for(it=p[src].begin();it!=p[src].end();it++)
	{
		if(visited[*it]==0)
			m=min(m,1+shortest_dis(p,*it,des,visited));
	}
	visited[src]=0;
	return m;
}
