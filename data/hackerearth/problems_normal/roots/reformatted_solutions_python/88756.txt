def main():
    N = input()
    a = {}
    b = {}
    c = {}
    count = 0
    for i in range(N):
        w = raw_input()
        w = w.split()
        w = map(int, w)
        a[i] = w[0]
        b[i] = w[1]
        c[i] = w[2]
    for i in range(N):
        if (b[i]**2 - 4*a[i]*c[i]) >= 0:
            count += 1
    print count


if __name__ == '__main__':
    main()
