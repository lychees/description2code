#include <bits/stdc++.h>
 
using namespace std;
 
int main()
{
    int N,T,arr[105],temp;
    scanf("%d",&T);
    while (T--){
        scanf("%d",&N);
        for (int i=0;i<N;i++)scanf("%d",&arr[i]);
        sort(arr,arr+N);
        int maxn=0,minn=2e9;
        for (int i=0;i<N/2;i++){
            temp=arr[i]+arr[N-1-i];
            maxn=max(maxn,temp);
            minn=min(minn,temp);
        }
        printf("%d\n",maxn-minn);
    }
    
}
