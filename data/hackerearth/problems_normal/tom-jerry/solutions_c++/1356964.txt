#include<bits/stdc++.h>
using namespace std;

typedef long long ll;
const int N=101;
int n;

int grid[N][N];
bool flag[N][N];

ll calc(int a,int b) {
    if(a==0&&b==0) {
        return 1;
    }
    if(a<0||b<0||a>=n||b>=n) {
        return 0;
    }
    if(grid[a][b]||flag[a][b]) {
    	return 0;
    }
    flag[a][b]=true;
   	ll res=calc(a-1,b)+calc(a,b-1)+calc(a,b+1)+calc(a+1,b);
    flag[a][b]=false;
    return res;
}
int main() {
    scanf("%d",&n);
    for(int i=0;i<n;i++) {
        for(int j=0;j<n;j++) {
            scanf("%d",&grid[i][j]);
        }
    }
    printf("%lld\012",calc(n-1,n-1));
    return 0;
}
