#include<bits/stdc++.h>

using namespace std;

int main()
{
	vector<int>e,o;
	int n,i,s=0,s1=0;
	cin>>n;
	int a;
	for(i=0;i<n;i++)
	{
		cin>>a;
		if(a%2==0)
		{
			e.push_back(a);
		}
		else
		o.push_back(a);
		
	}
	sort(e.begin(),e.end());
	sort(o.begin(),o.end());
	
	vector<int>::iterator it;
	for(it=e.begin();it!=e.end();it++)
	{
		cout<<*it<<" ";
		s+=*it;
	}
	cout<<s<<" ";
	for(it=o.begin();it!=o.end();it++)
	{
		cout<<*it<<" ";
		s1+=*it;
	}
	cout<<s1<<" ";
	return 0;
	
	
}
