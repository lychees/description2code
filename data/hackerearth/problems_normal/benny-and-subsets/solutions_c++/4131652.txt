#include<bits/stdc++.h>
#define mod 10000007
using namespace std;
long long cnt[1<<22];
long long dp[2][1<<12];
int main()
{
    long i,j,n,x;
    cin>>n>>x;
    vector<long long> small,big;
    for(i=0;i<n;i++)
    {
        cin>>j;
        if(j>(1<<10))
            big.push_back(j);
        else
            small.push_back(j);
    }
    long big_size;
    big_size=big.size();
    for(i=0;i<(1<<big_size);i++)
    {
        long long val=0;
        for(j=0;j<big_size;j++)
        {
            if(i&(1<<j))
                val=val^big[j];
        }
        cnt[val]++;
    }
    for(i=0;i<(1<<12);i++)
    {
        dp[0][i]=dp[1][i]=0;
    }
    dp[0][0]=1;
    for(i=0;i<small.size();i++)
    {
        for(j=0;j<(1<<12);j++)
        {
            if(dp[0][j]>0)
            {
                dp[1][j]=(dp[1][j]+dp[0][j])%mod;
                dp[1][j^small[i]]=(dp[1][j^small[i]]+dp[0][j])%mod;
            }
        }
        for(j=0;j<(1<<12);j++)
        {
            dp[0][j]=dp[1][j];
            dp[1][j]=0;
        }
    }
    long long ans=0;
    for(j=0;j<(1<<12);j++)
    {
        ans=(ans+(dp[0][j]*cnt[j^x])%mod)%mod;
    }
    cout<<ans;
    return 0;
}
