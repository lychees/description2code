#include <iostream>
using namespace std;

long long int a[1003];
long long int b[1003];


void init()
{
	a[1] = b[1] = 1;
	
	for(int i=2 ; i<=1000 ; i++)
	{
		b[i] = a[i-1];
		a[i] = a[i-1] + b[i-1] ;
	}
	
}


int main()
{
	init();
	int t ; cin>>t ;
	while(t--)
	{
		int n;
		cin >> n ;
		
		cout << (a[n]+b[n])*(a[n]+b[n]) << endl;
	}
	return 0;
}
