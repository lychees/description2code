#include <iostream>
#include<algorithm>
using namespace std;
int alpha(string s)
{
	int c=0;
	for(int i=0;i<s.length();i++)
	{
		if((s[i]>=65 && s[i]<=90)||(s[i]>=97 && s[i]<=122))
		{
			c+=1;
		}
	}
	return c;
}
int main()
{
   int t,i,j,k,n,m;string p;
   cin>>t;
   if(t>10 || t<1)
   {
   	cout<<"Invalid Test\n";
   	return 0;
   }
   else
   {
   	for(i=0;i<t;i++)
   	{
   		cin>>p;
   		int h=p.length();
   		if(!(h>=1 && h<=100)||alpha(p)==0)
   		{
   			cout<<"Invalid Input\n";
   		continue;
   		}
   		int g=alpha(p);
   		if(g==1)
   		{
   		cout<<0;
   		continue;
   		}
   		else
   		{
   			int c1=0,c2=0;
   			for(j=0;j<p.length();j++)
   			{
   				if(p[j]>=65 && p[j]<=90)
   				{
   					c1+=1;
   				}
   				else if(p[j]>=97 && p[j]<=122)
   				{
   					c2+=1;
   				}
   			}
   			cout<<min(c1,c2)<<"\n";
   		}
   		
   	}
   }
    return 0;
}
