#include <iostream>
using namespace std;

long long dp[200][200];
long long modd = 10000007;

long long rek(int n, int r) {
	if (n < 0) return -1;
	if (dp[n][r] != 0) return dp[n][r];
	if (n == 0) return dp[n][r] = 1;
	if (r == 1) return dp[n][r] = 1;
	long long res = 0;
	for (int i = 0; i <= n; i++)
		res = (res + rek(n - i, r - 1)) % modd;
	return dp[n][r] = res;
}

int main()
{
    int n, r;
    cin >> n >> r;
    cout << rek(n - r, r) << endl;
    return 0;
}
