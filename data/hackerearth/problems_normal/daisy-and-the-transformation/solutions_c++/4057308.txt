#include <iostream>
#include <string>
using namespace std;

int main()
{
    int t,n,ctr,i;
    string s1,s2;
    scanf("%d",&t);
    while (t--){
    	ctr=0;
    	cin>>s1>>s2>>n;
    	for (i=0;i<s1.length();i++){
    		if(s1[i]!=s2[i])
    		ctr++;
    	}
    	//printf("%d",ctr);
    	if(ctr==n)
    	printf("Yes\n");
    	else if(ctr<n && (n-ctr)%2==0)
    	printf("Yes\n");
    	else
    	printf("No\n");
    }
    return 0;
}
