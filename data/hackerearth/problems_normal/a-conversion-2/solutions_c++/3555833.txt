void func(int id, int sm) {
	char c,d;
	if(sm>=cnt) return;
	if(id>=n) {
		if(sm<cnt) cnt=sm;
		return;
	}
	c=s1[id];
	if(c==s2[id]) {
		func(id+1,sm);
		return;
	}
	if(c<s2[id]) {
		s1[id]=s2[id];
		func(id+1,(sm+s2[id]-c));
		s1[id]=c;
	}
	if(c>s2[id]) {
		s1[id]=s2[id];
		func(id+1,(sm+c-s2[id]));
		s1[id]=c;
	}
	rep(i,id+1,n) {
		d=s1[i];
		s1[i]=s1[id];
		s1[id]=d;
		c=s1[id];
		if(c==s2[id]) {
			func(id+1,sm+1);

			func(id+1,(sm+1+s2[id]-c));

		if(c>s2[id]) {
			s1[id]=s2[id];
			func(id+1,(sm+1+c-s2[id]));
			s1[id]=c;
		}
		d=s1[id];
		s1[id]=s1[i];
		s1[i]=d;
	}
}
