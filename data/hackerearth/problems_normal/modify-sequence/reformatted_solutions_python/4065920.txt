n = int(raw_input())

x = map(int, raw_input().split())

possible = True

for i in range(len(x)-1):
    if x[i+1] - x[i] < 0:
        possible = False
        break
    else:
        x[i+1] = x[i+1] - x[i]
        x[i] = 0

if x[-1] != 0:
    possible = False

if possible is True:
    print "YES"
else:
    print "NO"
