#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;

int main()
{
    ll t, n, k, ctr = 0;
    cin>>t;
    while (t--) {
    	cin>>n>>k;
    	ctr = 0;
    	ll arr;
    	while (n--) {
    		cin>>arr;
    		if (arr <= 0) ctr += 1;
    	}
    	if (ctr >= k) cout<<"NO"<<endl;
    	else cout<<"YES"<<endl;
    }
    return 0;
}
