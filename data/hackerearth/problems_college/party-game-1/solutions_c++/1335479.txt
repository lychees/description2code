int a[10009];
int main()
{

    int n,t,i;
    ll c,j;
    cin>>t;
    while(t--)
    {
        cin>>c>>j>>n;
        for(i=0;i<n;i++)
            cin>>a[i];
        sort(a,a+n);
        //for(i=0;i<n;i++)
            //cout<<a[i]<<" -- ";
        ll sumc=0,sumj=0;
        i=0;
        int cc=0,cj=0;
        while((sumc+a[i]<=c || sumj+a[i]<=j) && i<n)
        {
            if(sumc+a[i]<=c)
              {
                   sumc+=a[i];
                   cc++;
              }

            if(sumj+a[i]<=j)
                {
                    sumj+=a[i];
                    cj++;
                }
            i++;

        }
        if(cc==cj)
            cout<<"Tie\n";
        else if(cc>cj)
            cout<<"Chef Won\n";
        else  cout<<"Jhool Won\n";
    }

    return 0;
}
