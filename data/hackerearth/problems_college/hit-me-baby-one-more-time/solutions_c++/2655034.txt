#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
int main()
{
	ll n,sum,i,a,b,c;
	cin>>n;
	i=1;
	sum=0;
	a=n/3;
	if(a*3==n)
		a--;
	b=n/5;
	if(b*5==n)
		b--;
	c=n/15;
	if(c*15==n)
		c--;
	if(a>0)
		sum+=(3*a*(a+1))/2;
	if(b>0)
		sum+=(5*b*(b+1))/2;
	if(c>0)
		sum-=(15*c*(c+1))/2;
	cout<<sum<<endl;
	return 0;
}
