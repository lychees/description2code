#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	int a[1000];
	
    int T;cin>>T;
    while(T--){
    	int sum=0;
    	int N;cin>>N;
    	for(int i=0;i<N;i++){
    		cin>>a[i];
    		a[i]=a[i]-1;
    		sum+=a[i];
    	}
    	sort(a,a+N);
    	cout<<a[0]<<" "<<sum<<"\n";
    	
    	
    }
    return 0;
}
