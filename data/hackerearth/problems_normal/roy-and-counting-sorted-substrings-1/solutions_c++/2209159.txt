#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	string s;
    	cin>>s;
    	long long ans=1;
    	long long val=1;
    	for(int i=1;s[i]!='\0';i++){
    		if(s[i]>=s[i-1]){
    			val++;
    			
    		}
    		else{
    			val=1;
    			
    		}
    		ans+=val;
    	}
    	cout<<ans<<endl;
    	ans=0;
    }
}
