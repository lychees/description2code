#include<bits/stdc++.h>
using namespace std;
  long long int temp[1000000];
long long fun(long long int n)
{
	if(n==0|| n==1 || n==2)
	   return temp[n]=n;
	   else if(temp[n]!=0)
	         return temp[n];
	   else  return temp[n]=fun(n-1)+fun(n-2);
}
int main()
{
	long long int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		cout<<fun(n)<<endl;
	}
}
