def pages(n):
    ans = 1000000
    for i in xrange(113):
        for j in xrange(113):
            if (i*10 + j*12 == n):
                ans = min(ans, i+j)
    if (ans == 1000000):
        return -1
    else:
        return ans


def main():
    tc = int(raw_input())
    while (tc > 0):
        tc = tc - 1
        n = int(raw_input())
        print pages(n)


if __name__ == "__main__":
    main()
