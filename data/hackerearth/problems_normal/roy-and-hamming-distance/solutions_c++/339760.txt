#include <iostream>
#include<stdio.h>
using namespace std;
#define MOD 1000000007
#define LL long long int
int main() {
    int N,i;
	char s[100];
	scanf("%d %s",&N,s);
	LL a[100];
	a[0]=1,a[1]=0;
	for(i=2;i<100;i++)
	{
		if(i&1)
			a[i]=(i*a[i-1]-1)%MOD;
		else
			a[i]=(i*a[i-1]+1)%MOD;
	}
	if(N==1)
		printf("1\n");
	else
	printf("%lld\n",a[N]);
	return 0;
}
