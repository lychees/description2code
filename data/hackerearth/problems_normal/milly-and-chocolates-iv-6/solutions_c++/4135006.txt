using namespace std;
#define ll long long
#define INF (int)1e9

int main() {
	int t;
	cin >> t;
	while (t--) {
		int n, k, m;
		cin >> n >> k >> m;
		vector<int> p(n);
		vector<vector<int>> t(n);
		for (int i = 0; i < n; ++i) {
			cin >> p[i];
		}

		for (int i = 0; i < n; ++i) {
			for (int j = 0; j < p[i]; ++j) {
				int k;
				cin >> k;
				t[i].push_back(k);
			}
		}

		pair<int, ll> result{ -1, numeric_limits<long long>::max() };
		ll curr_result = numeric_limits<long long>::max();
		for (int i = 0; i < n; ++i) {			
			long long total_time = t[i][0] + k;
			for (int j = 1; j < p[i]; ++j) {
				total_time += t[i][j] + m + k;
			}

			curr_result = min(curr_result, total_time);
			if (result.second != curr_result) { result = { i + 1, curr_result }; }
		}

		cout << result.first << " " << result.second << '\n';
	}

	return 0;
}
