#include<cstdio>
#include<iostream>


int main()
{

	int t;
	std::cin>>t;
	while (t--)
	{
		char str[20]={'\0'};
		scanf("%s",str);
		for(int i=0;str[i]!='\0';i++)
		{
			if(str[i]=='2')
				printf("cde");
			else if(str[i]=='7')
				printf("acf");
			else if(str[i]=='6')
				printf("b3");
			else if(str[i]=='9')
				printf("c6a");
		}
		printf("\n");
	}
	return 0;
}
