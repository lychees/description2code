#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    map<string,int> mymap;
    int tot = pow(2,n) - 1;
    map<string,int>::iterator it1,it2;
    for(int i=0;i<tot;i++)
    {
    	string a,b;
    	cin>>a>>b;
    	it1 = mymap.find(a);
    	it2 = mymap.find(b);
    	if(it1 == mymap.end())
    	{
    		mymap.insert(make_pair(a,1));
    	}
    	if(it2 == mymap.end())
    	{
    		mymap.insert(make_pair(b,0));
    	}
    	else if(it2 != mymap.end() && it2->second == 1)
    	{
    		mymap[b] = 0;
    	}
    }
    string ans;
    for(it1=mymap.begin();it1!=mymap.end();it1++)
    {
    	//cout<<it1->first<<" "<<it1->second<<endl;
    	if((it1->second) == 1)
    	{
    		ans = it1->first;
    		//cout<<ans<<endl;
    		break;
    	}
    }
    cout<<ans<<endl;
    return 0;
}
