#include<iostream>
using namespace std;
int main()
{
int n,count=0;
cin>>n;

cin.ignore();

for(int j=0;j<n;j++)
{
string str;
 string str1;
 getline(cin,str);
getline(cin,str1);
 if(str.length()>str1.length())
 cout<<"not possible"<<endl;
else{
 for(int i=0;i<=str1.length()-str.length();i++){
 if(str1.substr(i,str.length())==str){
   count++;
   break;
 }}
if(count>0)
cout<<"possible"<<endl;
else
cout<<"not possible"<<endl;

count=0;
}}

return 0;
}
