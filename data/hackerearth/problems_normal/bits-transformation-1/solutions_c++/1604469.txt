#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		int N,X,Y,Z,T;
		cin>>N>>X>>Y>>Z>>T;
		string A,B;
		cin>>A>>B;
		vector<vector<int>> DP(N+1,vector<int>(2*N+1));
		for(int i=0;i<DP.size();i++){
			for(int j=0;j<DP[i].size();j++){
				DP[i][j]=1000000000;
			}
		}
		DP[0][N]=0;
		for(int i=0;i<N;i++){
			for(int j=-N;j<=N;j++){
				int currentValue=j-(B[i]-'0');
                if(currentValue < -N || currentValue>N)
                        continue;
                int penalty;
                if(A[i]=='?')
                {
                	penalty=Z;
                }
                else if(A[i]=='0'){
                	penalty=0;
                }
                else if(A[i]=='1'){
                	penalty=Y;
                }
                else{
                	penalty=X;
                }
                int currentCost=DP[i][j+N]+T*abs(j);
				currentCost+=penalty;
                DP[i+1][currentValue+N]=min(DP[i+1][currentValue+N],currentCost);
				currentValue=j+(1-(B[i]-'0'));
                if(currentValue < -N || currentValue>N)
                        continue;
                if(A[i]=='?')
                {
                	penalty=Z;
                }
                else if(A[i]=='1'){
                	penalty=0;
                }
                else{
                	penalty=X;
                }
                currentCost=DP[i][j+N]+T*abs(j);
                currentCost+=penalty;
                DP[i+1][currentValue+N]=min(DP[i+1][currentValue+N],currentCost); 
                //cout<<DP[i][j]<<" ";
			}
		}
		cout<<DP[N][N]<<"\n";
	}
}
