int sec(char c,int *curbut)
{
	for(int i = 0;i < buttons.size();i++)
	{
		int t = 0;
		for(int j = 0;j < buttons[i].size();j++)
		{
			t++;
			if (c == buttons[i][j]) {
				if (i != *curbut)
					t++;
				*curbut = i;
				return t;
			}
		}
	}
}

long long total_time(char *str)
{
	int curbut = 1;
	long long t = 0;
	for(;*str != '\0' && *str != '\n';str++) {
		t += sec(*str,&curbut);
	}
	return t;
		
}
