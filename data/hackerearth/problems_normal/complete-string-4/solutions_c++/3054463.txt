#include <iostream>
#include <string.h>
using namespace std;

int main()
{
char s[100];
int i,l,test,j,c;
cin>>test;
while(test--)
{
cin>>s;
l=strlen(s);
c=0;
for(i=97;i<=122;i++)
{
for(j=0;j<l;j++)
{
if(i==s[j])
{
c++;
break;
}
}
}
if(c==26)
cout<<"YES"<<endl;
else
cout<<"NO"<<endl;
}
return 0;
}
