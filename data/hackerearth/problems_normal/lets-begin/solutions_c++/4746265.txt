#include <bits/stdc++.h>
using namespace std;

int x,prime[4],dp[4][1000005];

int solve(int i,int sum)
{
	if(i<0 || sum<0)
		return 1e9;
	if(sum==0)
		return 0;
	if(dp[i][sum]!=-1)
		return dp[i][sum];
	int ans=0;	
	if(prime[i]>sum)
		ans=solve(i-1,sum);
	else
		ans=min({solve(i-1,sum),solve(i-1,sum-prime[i])+1,solve(i,sum-prime[i])+1});
	dp[i][sum]=ans;	
	return dp[i][sum];	
}

int main()
{
    int t,i,j;
    prime[0]=2; prime[1]=3; prime[2]=5; prime[3]=7;
    cin>>t;
    while(t--)
    {
    	cin>>x;
    	for(i=0;i<4;i++)
    		for(j=0;j<1000005;j++)
    			dp[i][j]=-1;
    	int ans=solve(3,x);
    	if(ans>1e7)
    		cout<<"-1\n";
    	else
    		cout<<ans<<endl;
    }
    return 0;
}
