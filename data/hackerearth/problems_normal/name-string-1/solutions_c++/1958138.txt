#include <iostream>
using namespace std;

int main()
{
   int t;
   long long r,k,v,great,great2,c;
   cin>>t;
   while(t!=0)
   {
   	cin>>r>>k>>v;
   	great=r;
   	r=0;
   	if(k>=great)
   	{
   		r=great;
   	great=k;
   	k=0;
   	}
   	if(v>=great)
   	{  
   		k=great;
   		great=v;
   		v=0;
   	}
   	if(great>=r)
   	great2=r;
    if(great2<=k)
   	great2=k;
   	if(great2<=v)
   	great2=v;
   	c=0;
 while(great!=0)
  {
  	c++;
  	if(great2!=0)
  	{
  		c++;
  		great2--;
  	}
  	else
  	break;
  	great--;
  }
  cout<<c<<endl;
  t--;
   }
    return 0;
}
