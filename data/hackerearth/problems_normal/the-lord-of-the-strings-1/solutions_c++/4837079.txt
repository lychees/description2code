#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin >> t;
    string s;
    while (t--) {
    	cin >> s;
    	int count[26] = {0};
        for (int i = 0, n = s.length(); i < n; ++i)
            count[s[i] -'a']++;
        for (int i = 0, n = s.length(); i < n; ++i)
             if(count[s[i] -'a']) {
             	cout << s[i];
             	count[s[i]-'a'] = 0;
             }
        cout << endl;
    }
    return 0;
}
