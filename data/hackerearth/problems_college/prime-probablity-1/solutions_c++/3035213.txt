#include<stdio.h>

typedef long long ll;

const ll SIZE = (1e+6)+5;
const ll mill = (1e+6);

ll sieve[SIZE]={0};

void seives()
{
    for(ll i=2;i<SIZE;++i)
        sieve[i] = 1;

    for(ll i=2;i<SIZE;++i)
    {
        for(ll j=i*2;j<SIZE; j+=i)
        {
            sieve[j] = 0;
        }
    }
    for(ll i=2;i<SIZE;++i)
        sieve[i] += sieve[i-1];
}

int main()
{
    seives();
    ll t ,l,r;
    ll primes,tot;

    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld%lld",&l,&r);
        primes = sieve[r] - sieve[l-1];

        tot = r-l+1;

        /*if(prob == mill)
            printf("1.000000\n");

            printf("0.%lld\n",prob);
        else if(prob>=(1e+4))

        else if(prob>=(1e+3))
            printf("0.00%lld\n",prob);
        else if(prob>=(100))
            printf("0.000%lld\n",prob);
        else if(prob>=(10))
            printf("0.0000%lld\n",prob);
        else if(prob>=1)
            printf("0.00000%lld\n",prob);
        else
            printf("0.000000\n");*/
        printf("%.6f\n",(float)primes/(float)tot);
    }
    return 0;
}
