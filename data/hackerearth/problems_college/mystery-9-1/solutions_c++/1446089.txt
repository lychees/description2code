#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    int t,l,m,j;
    char str[100];
    cin>>t;
    while(t--)
    {
    	cin>>str;
    	l=strlen(str);
    	m=0;
    	for(j=0;j<strlen(str);j++)
    	{
    		if(str[j]==str[l-j-1])
    		{m=0;}
    	    else
    	    {m=1;break;}
    	}
    	if(m==0)
    	cout<<(int)str[l/2]<<endl;
    	else
    	cout<<-1<<endl;
    	}
    return 0;
}
