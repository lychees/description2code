#include <iostream>
using namespace std;

int main()
{
    int r,c,t;
    cin>>t;
    while(t--)
    {
    	cin>>r>>c;
    	int sum=0,su=0;
    	for(int i=1;i<=r;i++)
    	{
    		if(i%2)
    		{
    			if(i>c) sum+=c;
    			else sum+=i;
    		}
    		else
    		{
    			if(i>c) su+=c;
    			else su+=i;
    		}
    	}
    	cout<<su<<" "<<sum<<endl;
    }
    return 0;
}
