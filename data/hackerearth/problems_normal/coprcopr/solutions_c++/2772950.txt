#include <iostream>
#include<vector>
#include<cmath>
long long int tot[10000005];
long long int p[10000005];
using namespace std;

int main() {
	// your code goes here
	long long int i,j,z,k,n,t;
	vector<bool> v(10000005,0);
	
	for(i=3;i<=sqrt(10000005);i=i+2)
	{
		while(v[i]==1)
		{
			i=i+2;
		}
		
		j=i*(i-2);
		z=i<<1;

	while((j=j+z)<=10000005)
	{
		v[j]=1;
	}
	
	}
	
	p[0]=2;
	
	for(i=3,j=1;i<=10000005;i=i+2)
	{
		if(v[i]==0)
		{
			p[j++]=i;
		}
	}
	
	for(i=2;i<=10000005;i++)
	{
		tot[i]=i;
	}
	
	tot[1]=1;
	
	for(i=0;i<j;i++)
	{
		tot[p[i]]=tot[p[i]]-1;
		
		for(k=2;p[i]*k<=10000005;k++)
		{
			tot[p[i]*k]=(tot[p[i]*k]*(p[i]-1))/p[i];
		}
		
	}

		cin>>n;
		while(n--)
		{
		long long int ans=0;
		scanf("%lld",&t);
		for(i=1;i<=t;i++)
		ans+=tot[i];
		cout<<ans<<endl;
		}
	
	return 0;
}
