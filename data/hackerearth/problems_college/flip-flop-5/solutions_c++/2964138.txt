#include <bits/stdc++.h>
using namespace std;
char ch[100005];
int main()
{
    int n;
    cin>>n;
    while(n--){
    	scanf("%s",&ch);
    	int len = strlen(ch);
        stack<int> stk;
    	int ans = 0;
    	for(int i=0;i<len;i++){
	    	if(stk.empty() || stk.top() != ch[i]){
	    		stk.push(ch[i]);
	    	}
	    	else{
	    	   ++ans;	
	    	}
    	}
    	printf("%d\n",ans);
    	
    }
    return 0;
}
