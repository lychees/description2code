#include<bits/stdc++.h>
using namespace std;
int bin(long long int a[],int f,int l,long long int x)
{
        int m;
        while(f<=l)
        {
                int m=(f+l)/2;
                if(a[m]==x)
                        return m;
                else if(a[m]>x)
                        l=m-1;
                else
                        f=m+1;
        }
        return -1;
}

int main()
{
        int n;
        scanf("%d",&n);
        long long int j,s,i,num,ar[n];
        scanf("%lld",&num);
        for(i=0;i<n;i++)
                scanf("%lld",&ar[i]);

        sort(ar,ar+n);
        for(i=0;i<n-2;i++)
        {
                int f=0;
                for(j=i+1;j<n-1;j++)
                {
                        s=ar[i]+ar[j];

                        if(bin(ar,j+1,n,num-s)!=-1)
                        {printf("YES\n");
                                f=1;
                                break;
                        }
                }
                if(f==1)
                        break;
        }

        if(i==n-2)
                printf("NO\n");
        return 0;
}
