#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t; cin >> t;
	if(t<1 || t >10) cout << "Invalid Test";
	else
	while(t--)
	{
		string s; cin >> s;
		int v = s.length();
		if(v <1 || v > 100) cout << "Invalid Input";
		else
		{
			int l=0, h=0;
			for(int i=0;i<s.length();i++)
			{
				if(s[i] >=97 && s[i] <=122) l++;
				if(s[i] >=65 && s[i] <= 90) h++;
			}
			if(l == 0 && h==0) cout << "Invalid Input";
			else cout << min(l, h);
		}
		cout << endl;
	}
}
