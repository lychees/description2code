#include<bits/stdc++.h>
using namespace std;
void print(char arr[],long int arrC)
{
	for(long int i=arrC-1;i>=0;i--)
		cout<<arr[i];
}
int main()
{
	long int tc;
	cin>>tc;
	char p[1];
	gets(p);
	while(tc--)
	{
		char arr[100001];
		gets(arr);
		
		long int len;
		for(len=0;arr[len]!='\0';len++);
		
		for(long int i=len-1;i>=0;)
		{
			long int newarrC=0;
			char newarr[100001];
			long int j=i;
			while(arr[j]!=' ' && j>=0)
			{
			//	cout<<"I = "<<i<<"       "<<arr[j];
				newarr[newarrC++]=arr[j];
				j--;
			}
			print(newarr,newarrC);
			while(arr[j]==' ' && j>=0)
			{
			//	cout<<"I = "<<i<<"       "<<arr[j];
				cout<<" ";
				j--;
			}
			i=j;
		}
		cout<<endl;
	}
	return 0;
}
