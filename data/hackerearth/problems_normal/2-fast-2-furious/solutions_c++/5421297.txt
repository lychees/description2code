#include <iostream>
using namespace std;

int main()
{
    int N,domChange=0,brainChange=0,currentChange;
    long int dom[1000000],brain[1000000];
    cin>>N;
    
    for(int i = 0;i<N;i++){
    	cin>>dom[i];
    }
    for(int i = 0;i<N;i++){
    	cin>>brain[i];
    }
    
    for(int i=0;i<N-1;i++){
    	int currentChange1 = abs(dom[i] - dom[i+1]);
    	int currentChange2 = abs(brain[i] - brain[i+1]);
    	if(currentChange1 > domChange){
    		domChange = currentChange1;
    	}
    	if(currentChange2 > brainChange){
    		brainChange = currentChange2;
    	}
    }
    
    if(brainChange > domChange){
    	cout<<"Brian"<<endl;
    	cout<<brainChange;
    }
    else if(brainChange < domChange){
    	cout<<"Dom"<<endl;
    	cout<<domChange;
    }
    else{
    	cout<<"Tie";
    }
    return 0;
}
