#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define mod 1000000007
ll a[1000005];
inline int inp( )
{
    int n = 0 , s = 1 ;
    char p = getchar_unlocked( ) ;
    if( p == '-' )
        s = -1 ;
    while( ( p < '0' || p > '9' ) && p != EOF && p != '-' )
        p = getchar_unlocked( ) ;
    if( p == '-' )
        s = -1 , p = getchar_unlocked( ) ;
    while( p >= '0' && p <= '9' )
    {
        n = ( n << 3 ) + ( n << 1 ) + ( p - '0' ) ;
        p = getchar_unlocked( ) ;
    }
    return n * s ;
}
ll sam(ll p) {
	ll sn;
	sn=(p*(p+1)/2);
	return sn%mod;
}
int main() {
	int t;
	long long n,sum,res,ans,i,j,k;
	scanf("%d",&t);
	while(t--) {
		scanf("%lld",&n);
		sum=res=ans=k=0;
		for(i=1;i*i<=n;i++) {
			res=n/i;
			a[i]=res;
			sum=(sum+(i*(res%mod))%mod)%mod;
			k=i;
			//printf("aa %lld\n",sum);
		}
		a[k+1]=0;
		for(i=1;i<=k;i++) {
			if(i*i>=n||((n/a[i])==a[i])) break;
			if(a[i+1]==0) {//puts("akshay");
				//ans=a[i];
				j=n/a[i];
				res=sam(a[i]);
			    ans=sam(j);
			    ans=res-ans;
			}
			else {
			res=sam(a[i]);
			ans=sam(a[i+1]);
			ans=res-ans;
			}
			sum=(sum+((i%mod)*(ans%mod))%mod)%mod;
			//printf("ad %lld\n",sum);
		}
		printf("%lld\n",sum%mod);
	}
	return 0;
}
