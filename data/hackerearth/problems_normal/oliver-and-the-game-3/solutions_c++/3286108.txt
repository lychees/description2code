vector<int> g[maxN+5];
int visit[maxN+5];
int start[maxN+5];
int ender[maxN+5];

int timer;

void dfs(int s)
{
	visit[s]=1;
	start[s]=timer++;
	for(int i=0;i<(int)g[s].size();i++)
	{
		if(!visit[g[s][i]])
			dfs(g[s][i]);
	}
	ender[s]=timer++;
}
