int main()
{
	//ios_base::sync_with_stdio(false);
	//cin.tie(0);
	int n,i,j,maxx=0;
	cin>>n;
	int a[n],b[n];
	f(i,n) cin>>a[i];
	f(i,n) cin>>b[i];
	sort(a,a+n);
	sort(b,b+n);
	for(i=0;i<n;i++)
	{
		int ans=0,k=0;
		if(a[i]>b[k]) ans+=100;
		else if(a[i]==b[k]) ans+=50;
		k=1;
		for(j=(i+1)%n;j!=i;)
		{
			if(a[j]>b[k]) ans+=100;
			else if(a[j]==b[k]) ans+=50;
			k++;
			j=(j+1)%n;
		}
		if(ans>maxx) maxx=ans;
	}
	cout<<maxx;

return 0;
}
