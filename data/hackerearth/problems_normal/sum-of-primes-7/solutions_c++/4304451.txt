#include<iostream>
#include<cmath>
#include<cstdio>
using namespace std;

typedef long long int ll;

ll PRIME[1000005];
ll SUM[1000005];

void Prime_Sieve(ll n){
for(int i = 0 ; i < n ; i++){
    PRIME[i] = 1;
}
PRIME[0] = 0 ;
PRIME[1] = 0;

for(int i = 2 ; i*i <= n ; i++){
    if(PRIME[i]){
        for(int j = i*i ; j <= n ; j += i){
            PRIME[j] = 0;
        }
    }
}
SUM[0] = 0;SUM[1] = 0;SUM[2] = 2;
for(int i = 3 ; i < n ; i++){
    if(PRIME[i]){
        SUM[i] = SUM[i-1]+i;
    }else{
     SUM[i] = SUM[i-1];
    }
   }
}

int main(){
Prime_Sieve(1000005);
int t;
scanf("%d",&t);
while(t--){
    ll a,b;
    scanf("%lld %lld",&a,&b);
    printf("%lld\n",(SUM[b]-SUM[a-1]));
  }
  return 0;
}
