const int MAXM = 2e2+5 ;
ll dp[MAXM][MAXM];
 
ll solve(int f,int h){
	if(dp[f][h]!=0)		return dp[f][h];
	if(f==0)	return dp[f][h]=1;
	if(h==0)	return dp[f][h]=solve(f-1,1);
	return dp[f][h]=solve(f-1,h+1)+solve(f,h-1);
}

{
	mset(dp,0);
	int t,p,n,i,j,k;

		cin>>n;
		cout<<solve(n,0)<<endl;
	}	
    return 0;
}
