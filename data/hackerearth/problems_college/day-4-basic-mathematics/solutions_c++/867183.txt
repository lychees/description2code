#include <iostream>
using namespace std;
#define f(x) x*x+x

int findRoot(int s){
	float h=1,k=2*s;
	while((f(h)-k)<0){
		h*=2;
	}
	float l=h/2;
	while((h-l)>1){
		int m = (l+h)/2;
		if((f(m)-k)==0)
			return m%3;
		if((f(m)-k) <0)
			l=m;
		else
			h=m;
	}
	
	return ((int)h)%3;
	
	
	
}

int main()
{
    int T;
    cin>>T;
    while(T>0){
    	int n;
    	cin>>n;
    	switch(findRoot(n)){
    	case 0:
    		cout<<"SL"<<endl;
    		break;
    	case 1:
    		cout<<"LB"<<endl;
    		break;
    	case 2:
    		cout<<"BS"<<endl;
    		break;
   	 	}
   	 	T--;
    }
    return 0;
}
