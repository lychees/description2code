#include <iostream>
#include <bits/stdc++.h>
#define ll long long int
using namespace std;

int main()
{
    ll i,k,n;
    int t;
    cin>>t;
    while(t--)
    {
        k=0;
        cin>>n;
        vector <ll> v(n);
        for(i=0;i<n;i++)
        {
            scanf("%lld",&v[i]);
            k=k+v[i];
        }
        k=k/(n-1);
        for(i=0;i<n;i++)
        {
            printf("%lld ",(k-v[i]));
        }
        cout<<endl;

    }

    return 0;
}
