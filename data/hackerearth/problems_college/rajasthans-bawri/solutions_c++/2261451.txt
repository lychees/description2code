#include <iostream>
using namespace std;

int main()
{	
	int t;
    cin >> t;
    while(t--){
    	long long int a,b,c,d;
    	cin >>a >> b >> c >> d;
    	long long int t = max(a,b);
    	long long int s = min(a,b);
    	a = s; b = t ;
    	t = max(c,d);
    	 s = min(c,d);
    	c = s; d = t ;
    	if(a >= c  && b >= d){
    		cout << "Possible\n";
    	}
    	else{
    		cout << "Not Possible\n";
    	}
    }
    return 0;
}
