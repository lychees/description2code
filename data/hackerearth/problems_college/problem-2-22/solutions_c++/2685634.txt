#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    long long int test,i,j,k,a,b,r,c,d,e,f;
    cin>>test;
    while(test--){
    	cin>>r>>a>>b;
    	int f=0;
    	for(i=0;i<4;i++){
    		cin>>d>>e;
    		if(f==0){
    			if((((d-a)*(d-a))+((e-b)*(e-b)))>(r*r))
    			{
    				f=1;
    			}
    		}
    	}
    	if(f==1){
    		cout<<"No"<<endl;
    	}
    	else
    	cout<<"Yes"<<endl;
    }
    return 0;
}
