#include<iostream>
#include<stdio.h>
using namespace std;

bool isEven(int n)
{
    if((n%2)== 0)
        return true;
    else
        return false;
}

int main()
{
    long long t;
    scanf("%lld",&t);
    while(t--)
    {
        long long countOdd=0, countEven=0;
        long long r,c;
        scanf("%lld",&r);
        scanf("%lld",&c);
        if(r <= c)
        {
            long long lastEven,oddTerms;
            if(isEven(r))
            {
                lastEven = r;
                oddTerms = r/2;
            }
            else
            {
                lastEven = r-1;
                oddTerms = (r+1)/2;
            }
            long long n = ((2 + lastEven)/2)-1;
           	countOdd = oddTerms*oddTerms;
            countEven = n*(n+1);
        }
        else if(r > c)
        {
            long long lastEven,oddTerms;
            if(isEven(r))
            {
                lastEven = r;
                oddTerms = r/2;
            }
            else
            {
                lastEven = r-1;
                oddTerms = (r+1)/2;
            }
            long long n = ((2 + lastEven)/2)-1;
           	countOdd = oddTerms*oddTerms;
            countEven = n*(n+1);

            if(isEven(c))
            {
                for(long long i=c+1;i<=r;i=i+2)
                {
                    countOdd = countOdd - (i-c);
                }
                for(long long i=c+2;i<=r;i=i+2)
                {
                    countEven = countEven - (i-c);
                }
            }
            else
            {
                for(long long i=c+2;i<=r;i=i+2)
                {
                    countOdd = countOdd - (i-c);
                }
                for(long long i=c+1;i<=r;i=i+2)
                {
                    countEven = countEven - (i-c);
                }
            }
        }
        printf("%lld %lld\n",countEven, countOdd);
    }
    return 0;
}
