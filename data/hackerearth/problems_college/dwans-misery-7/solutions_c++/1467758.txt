#include<iostream>
#include<cstdio>
#include<cmath>
#include<vector>
#include<algorithm>
#include<cstring>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int a[25001]={0};
		int n;
		scanf("%d",&n);
		for(int i=0;i<n;i++)
			scanf("%d",&a[i]);
		long long int incl=a[0],excl=0;
		for(int i=1;i<(n-1);i++)
		{
			long long int excl1=max(incl,excl);
			long long int incl1=excl+a[i];
			incl=incl1;
			excl=excl1;
		}
		long long int ans=max(incl,excl);
		incl=a[1],excl=0;
		for(int i=2;i<(n);i++)
		{
			long long int excl1=max(incl,excl);
			long long int incl1=excl+a[i];
			incl=incl1;
			excl=excl1;
		}
		ans=max(ans,max(incl,excl));
		printf("%lld\n",ans);
	}
	
	return 0;
}
