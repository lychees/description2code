#include <bits/stdc++.h>

using namespace std;
vector < long long int > a(100001,-1);
vector < long long int > b(100001,-1);
long long int n;


long long int dp(int j,long long int k)
{
   if(j==1)
   {

    if(a[k]!=-1)
    {
        return a[k];
    }

   else return (dp(1,k-1)+dp(1,k-2)+dp(1,k-3));
   }
   if(j==0)
   {

    if(b[k]!=-1)
    {
        return b[k];
    }

   else return (dp(0,k-1)+dp(0,k-2));
   }


}

int main()
{
    cin>>n;
    if(n<=0)
    {
        cout<<"ERROR";
        return 0;
    }
    long long int j1,j2;
   a[0]=0;
   a[1]=1;
   a[2]=2;
   a[3]=4;
   b[0]=0;
   b[1]=1;
   b[2]=2;
    j1=dp(1,n);
    cout<<"Jack-"<<j1<<"\n";
    j2=dp(0,n);
    cout<<"Jill-"<<j2<<"\n";

}
