#include <iostream>
#include <string>
using namespace std;
int main()
{
    int t;
    cin >> t;
    string a,b;
    while(t--)
    {
        cin >> a >> b;
        for(int i=0;i<a.length();i++)
        {
            if((a[i]=='1' && b[i]=='1') || (a[i]=='0' && b[i]=='0'))
                cout << "0";
            else
                cout << "1";
        }
        cout << "\n";
    }
    return 0;
}
