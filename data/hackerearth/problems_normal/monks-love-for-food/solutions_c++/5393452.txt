#include <iostream>
#include <stack>
using namespace std;
int main(int argc, char const *argv[])
{
	int n,v,cost=0;
	stack<int> a;
	cin >> n;
	while(n){
		cin>>v;
		if(v==1){
			if(a.empty()) cout << "No Food" <<endl;
			else {
				cout << a.top() << endl;
				a.pop();
			}
		}else {
				cin >> cost;
				a.push(cost);
		}
		--n;
	}
	
	return 0;
}
