#include <iostream>
#include <stack>
#include <cmath>
#include <vector>

using namespace std;

int main(){
	int n, val;
	cin >> n;
	
	int players = 1;
	for(int i=0; i<n; i++)
		players *= 2;

	int grid[1025][1025];

	
	// initialization, just for good practice
	for(int i=0; i<11; i++)
		for(int j=0; j<11; j++)
			grid[i][j] = 0;

	/*
	Input like :
	A21
	A31 A32
	A41 A42 A43
	...
	*/
	for(int i=2; i <= players; i++){
		for(int j=1; j < i; j++){
			cin >> val;
			grid[i][j] = val;
		}
	}

	// ans vector
	vector <int> ans;
	for(int i=1; i<=players; i++)
		ans.push_back(i);
	
	// current winners list for conducting matches for the next round
	vector <int> temp;

	while(ans.size() > 1){
		for(int i=0; i< ans.size(); i+=2){
			int mmax = max(ans[i], ans[i+1]);
			int mmin = min(ans[i], ans[i+1]);
			if(grid[mmax][mmin] == 0){
				temp.push_back(mmin);
			}
			else
				temp.push_back(mmax);
		}
		/*
		// DEBUG
		// print all
		for(int i=0; i<ans.size(); i++)
			cout << ans[i] << " ";
		cout << endl;
		// print winners
		for(int i=0; i<temp.size(); i++)
			cout << temp[i] << " ";
		cout << endl;
		*/
		ans.clear();
		for(int i=0; i<temp.size(); i++)
			ans.push_back(temp[i]);
		temp.clear();
	}
	cout << ans[0] << endl;
	return 0;
}
