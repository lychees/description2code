#include <bits/stdc++.h>

using namespace std;

int gcd (int a,int b) {
    if (a % b == 0) {
        return b;
    } else {
        return gcd(b,a % b);
    }
}

int main()
{
    int t;
    scanf("%d",&t);
    while (t --) {
        int x,y,g,cnt;
        char s[15];
        scanf("%d%d",&x,&y);
        scanf("%s",s);
        cnt = 0;
        while (x > 0 && y > 0) {
            g = gcd(x,y);
            x = x - g;
            y = y - g;
            cnt += 1;
        }
//        printf("%d\n",cnt);
        if (cnt % 2 == 1) {
            printf("%s\n",s);
        } else {
            if (s[0] == 'C') {
                printf("Daspal\n");
            } else {
                printf("Chandu\n");
            }
        }
    }
    return 0;
    
}
