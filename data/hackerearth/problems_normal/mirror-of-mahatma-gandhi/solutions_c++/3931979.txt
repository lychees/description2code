#include<bits/stdc++.h>
using namespace std;

int main()
{
	int tcase;
	string s;
	cin>>tcase;
	while(tcase--)
	{
		bool flag=1;
		cin>>s;
		int l=s.size();
		for(int i=0;i<l/2;i++)
		{
			if(s[i]==s[l-i-1] && (s[i]=='1' || s[i]== '0' || s[i]=='8' )) continue;
			else {flag=0;break;}
		}
		if(flag)
		{
			if(l%2!=0 && (s[l/2]!='0' &&  s[l/2]!='1'&& s[l/2]!='8'))flag=0;
		}
		if(flag==1)cout<<"YES"<<endl;
		else cout<<"NO"<<endl;
	}
}
