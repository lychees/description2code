#include <iostream>
using namespace std;
int main(){
    int T,N,i;
    long long int M,sum,n_sum,c;
    cin>>T;
    while(T--){
        cin>>N;
        int a[N];
        for(i=0;i<N;++i) cin>>a[i];
        cin>>M;
        if(M == 0){
            cout<<0<<endl;
            continue;
        }
        sum=0;
        c=0;
        for(i=0;i<N;){
            sum += a[i];
            if(sum >= M){
                break;
            }
            i++;
            if(i == N){
                c++;
                if(c == 1) n_sum = sum;
//                cout<<"Twice-->";
//                cout<<"sum="<<sum<<" ";
                while(sum < M/2){
                    sum = 2*sum;
//                    cout<<sum<<" ";
                }
//                cout<<"\n\n\nRest-->";
//                cout<<"sum="<<sum<<" ";
                c++;
                while(sum+n_sum*c < M){
                    sum = sum+n_sum*c;
                    c++;
                }
                while(sum+n_sum < M){
                    sum += n_sum;
//                    cout<<sum<<" ";
                }
                i=0;
            }
        }
        cout<<i+1<<endl;
    }
    return 0;
}
