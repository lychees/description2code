#include <bits/stdc++.h>
using namespace std;
#define m 1000000007
typedef unsigned long long int ll;
ll ans,i,j,t,l,s,n,k;
ll d,f[51][51],e[51][51];
void matrix_multi(ll a[][51],ll b[][51])
{
    ll c[51][51];
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=n;j++)
		{
			c[i][j]=0;
			for(k=1;k<=n;k++)
			{
				c[i][j]=(c[i][j]%m+(a[i][k]%m*b[k][j]%m)%m)%m;
			}
		}
    }
    for(i=1;i<=n;i++)
    {
    	for(j=1;j<=n;j++)
    	{
    		a[i][j]=c[i][j]%m;
    	}
    }
}
ll matrix_exponentaition(ll l,ll s,ll d)
{
    for(i=1;i<=n;i++)
        for(j=1;j<=n;j++)
           {
            if(i==j)
            f[i][j]=1;
            else
            f[i][j]=0;
           }
	while(l>0)
	{
		if(l%2==1)
		matrix_multi(f,e);
		if(l!=1)
		matrix_multi(e,e);
		l=l/2;
	}
	return f[s][d]%m;
}
int main()
{
    cin>>t;
    while(t--)
    {
        cin>>n>>l>>s>>d;
    for(i=1;i<=n;i++)
    {
    	for(j=1;j<=n;j++)
    	{
    		cin>>e[i][j];
    	}
    }
    cout<<matrix_exponentaition(l,s+1,d+1)<<endl;;
    }
    return 0;
}
