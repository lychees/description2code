#include<bits/stdc++.h>
using namespace std;
#define ll long long int
ll z,len;
ll dp[200][200];
char s[200];
ll fun(ll idx,ll n)
{
    if(dp[idx][n]!=-2)
        return dp[idx][n];

    if(idx==len && n==0)
        return (dp[idx][n]=0);

    if(idx==len || n==0)
       return (dp[idx][n]=-1);

    ll k1=0,ans=-1,i;
        for(i=idx;i<len;i++)
        {
            k1=k1*10+(s[i]-'0');
            if(k1<=z)
            {
                ll k3=n-1;
                ll k4=i+1;
                ll k2=fun(k4,k3);
                if(k2!=-1)
                {
                    ans=max(ans,k1+k2);
                }
            }
        }
    return (dp[idx][n]=ans);
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        ll k,i,j;
        for(i=0;i<=100;i++)
        for(j=0;j<=100;j++)
        dp[i][j]=-2;
        scanf("%lld%lld",&k,&z);
        scanf("%s",s);
        len=strlen(s);
        ll k1=0,ans=-1;
        for(i=0;i<len;i++)
        {
            k1=k1*10+(s[i]-'0');
            if(k1<=z)
            {
                ll k3=k-1;
                ll k4=i+1;
                ll k2=fun(k4,k3);
                if(k2!=-1)
                {
                    ans=max(ans,k1+k2);
                }
            }
        }
        printf("%lld\n",ans);
    }
}
