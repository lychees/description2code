t = int(raw_input())
while (t):
    n = int(raw_input())
    a = map(int, raw_input().split())
    count = 0
    for i in range(n):
        for j in range(i+1, n):
            if (a[i]+a[j]) % 2 == 0 and a[i] != a[j]:
                count += 1
    print(count)
    t -= 1
