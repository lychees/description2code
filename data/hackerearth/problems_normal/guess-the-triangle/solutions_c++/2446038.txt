#include <bits/stdc++.h>
using namespace std;

int main()
{
    pair<float,float>p[3];
    float x1,y1,x2,y2,x3,y3;
    cout.precision(4);
    cout<<fixed;
    cin>>x1>>y1>>x2>>y2>>x3>>y3;
    p[0]=make_pair(x1+x2-x3,y1+y2-y3);
    p[1]=make_pair(x2+x3-x1,y2+y3-y1);
    p[2]=make_pair(x3+x1-x2,y3+y1-y2);
    sort(p,p+3);
    cout<<p[0].first<<" "<<p[0].second<<endl;
    cout<<p[1].first<<" "<<p[1].second<<endl;
    cout<<p[2].first<<" "<<p[2].second<<endl;
    return 0;
}
