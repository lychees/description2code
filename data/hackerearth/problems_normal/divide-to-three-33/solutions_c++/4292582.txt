#include <bits/stdc++.h>
using namespace std;
int a[20];
int ans;
void fun(int index,int sum1,int sum2,int sum3,int n)
{
	if(index==n)
	{
		if(sum1>=sum2&&sum2>=sum3)
		{
			ans=min(ans,sum1);
		}
		return;
	}
	fun(index+1,sum1+a[index],sum2,sum3,n);
	fun(index+1,sum1,sum2+a[index],sum3,n);
	fun(index+1,sum1,sum2,sum3+a[index],n);
}
int main()
{
	int n;
	scanf("%d",&n);
	for(int i=0;i<n;i++)
	scanf("%d",&a[i]);
	ans=1300;
	fun(0,0,0,0,n);
	cout<<ans;
	return 0;
	
}
