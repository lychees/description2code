#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define MOD 1000000007
#define MAX 1005
ll fact[MAX];
ll ifact[MAX];
ll modexp(ll a, ll b)
{
	ll res =1;
	while(b)
	{
		if(b&1) res = (res*a)%MOD;
		b/=2;
		a=(a*a)%MOD;
	}
	return res;
}
int main()
{
    //cout << "Hello World!" << endl;
    int i;
    int j;
    fact[0]=1;
    for(i=1;i<MAX;++i)
    	fact[i] = (fact[i-1]*i)%MOD;
    for(i=0;i<MAX;++i)
    	ifact[i] = modexp(fact[i],MOD-2);
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	int p;
    	ll sum = 0;
    	ll tmp;
    	cin>>n>>p;
    	for(i=0;i<=p;++i)
    	{
    		tmp = (ifact[i]*ifact[n-i])%MOD;
    		tmp = (tmp*fact[n])%MOD;
    		sum = (sum + tmp)%MOD;
    	}
    	if(sum<0) sum+=MOD;
    	cout<<sum<<endl;
    }
    return 0;
}
