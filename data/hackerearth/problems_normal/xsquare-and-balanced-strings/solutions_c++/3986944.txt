#include<bits/stdc++.h>
using namespace std;

int main()
{
   int n,i;
   cin>>n;
   while(n--)
   {
   	int flag=0;
   	char arr[100010];
   	cin>>arr;
   	int len=strlen(arr);
   	map<char,int> m;
   	for(i=0;i<len;i++)
   	{
   		m[arr[i]]++;
   	}
   	for(i=0;i<len;i++)
   	{
   		if(m[arr[i]]%2==1)
   		{
   			flag=1;
   			cout<<"-1\n";
   			break;
   		}
   	}
   	if(!flag)
   	cout<<"1\n";
   }
}
