#include <cstdio>
#define MOD 1000000007
inline int next_int() {
 int n = 0;
 char c = getchar_unlocked();
 while (!('0' <= c && c <= '9')) {
  c = getchar_unlocked();
 }
 while ('0' <= c && c <= '9') {
  n = n * 10 + c - '0';
  c = getchar_unlocked();
 }
 return n;
}
int a[1000001]={0},arr[1000001]={0};
void sieve()
{
    int i,j;
    a[1]=1;
    for(i=2;i*i<=1000000;i++)

        if(a[i]==0){
        for(j=2*i;j<=1000000;j+=i){
        if(a[j]==0)
        a[j]=i;
        }
        }
    }
}
