#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
    int N;
    cin >> N;
    
    int t[N], d[N], C[N];
    for(int i=0; i<N; i++) {
    	cin >> t[i] >> d[i];
    	C[i] = t[i] + d[i];
    }
    sort(C, C+N);
    
    for(int i=0; i<N; i++) {
    	for(int j=0; j<N; j++) {
    		if((t[j] + d[j]) == C[i]) {
    			cout << j+1 << " ";
    			t[j] = 0;
    			d[j] = 0;
    			break;
    		}
    	}
    }
    return 0;
}
