using namespace std;
map<int, int> dp;
long long int a[1000001];
int main(){
	int n, k;
	scanf("%d%d", &n, &k);
	for (int i = 0; i < n; i++){
		scanf("%lld", &a[i]);
	}
	sort(a, a + n);
	long long int countt = 0;
	for (int i = 0; i < n; i++){
		if (a[i] <= k){
			for (int j = i + 1; j < n; j++){
				long long int tmp = a[i] * a[j];
				if (tmp <= k){
					long long int want = k / tmp;
					int ind = upper_bound(a, a + n, want) - a;
					if (j < ind){
						countt += (long long int)(ind - j - 1);
					}
				}
			}
		}
	}
	printf("%lld\n", countt);
	return 0;
}
