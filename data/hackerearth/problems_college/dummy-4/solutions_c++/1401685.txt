#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,ans=0,tmp;cin>>n;
    string s;
    while(n--){
    	cin>>s;
    	for(int i=0;i<s.size();i++)ans+=s[i]-'0';
    	while(ans>=10){
    		tmp=ans;
    		ans=0;
    		while(tmp){
    			ans+=tmp%10;
    			tmp/=10;
    		}
    	}
    }
    cout<<ans<<endl;
    return 0;
}
