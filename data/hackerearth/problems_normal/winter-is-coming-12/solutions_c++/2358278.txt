#include <iostream>
using namespace std;
#include<bits/stdc++.h>
int a[100005] , arr[100005];

int L = -5 ;
int R = 1000005 ;
int main()
{
   ;long long int i , j ,k ,prev , count , t , n ;
   long long sum  = 0 ;
   cin>>t ;
   unordered_map< long long int , long long int > m ;
   while(t--)
   {
   		m.clear() ;
   		sum = 0 ;
   		m[0] = - 1 ; 
   		cin>>n ;
   		int mini = INT_MAX ;
   		for(i = 0 ; i< n  ;i++)
   		{
   			 cin>>a[i] ;
   			 sum = sum + a[i] ;
   			 arr[i] = sum % n ;
   			 if(m[arr[i]] == 0)
   			   m[arr[i]] = i ; 
   			 else
   			 {
   			 	prev = 1 +  m[arr[i]] ;
   			 	if(i - prev < mini)
   			 	{
   			 		  L = prev ;
   			 		  R = i ;
   			 		  mini = i - prev ;
   			 	}
   			 	 m[arr[i]] = i ; 
   			 	
   			 }
   			 
   		}
   		cout<<L + 1 <<" "<<R + 1 <<endl;
   		
   		
   		
   	
   }
   	 
   	 
   	 return 0;
}
