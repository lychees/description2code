n=int(raw_input())

strings=[]
i=0
while i < n:

	strings.append(raw_input())
strings.sort();
hashs={s:0 for s in strings}
for s in strings:
	hashs[s]+=1;
a=hashs.keys()
a.sort()
for s in a:
	print s,hashs[s]
