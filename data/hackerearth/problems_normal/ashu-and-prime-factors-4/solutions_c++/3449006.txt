#include <iostream>
using namespace std ;
int a[1000010] ; 
int sie[1000010]  ;
 void init()
 {
 	int  i ; 
 	sie[0] = 1 ; 
 	sie[1] = 1 ; 
 	for ( i = 2 ; i<= 1000000 ; i++)
 	{
 		 if ( sie[i] == 0 )
 		 {
 		 	int c = 1 ; 
 		 	
 		 	for ( int j = i+i ; j<=1000000 ; j = j+i )
 		 	{
 		 		if ( sie[j] == 0 )
 		 		{
 		 			
 		 			c++ ; 
 		 			
				  }
				  sie[j] = 1  ;
			  }
			  a[i] = c  ; 
		  }
 		
	 }
 	
 }
int main()
{
	init() ; 
	int t ; 
	cin >> t ; 
	while ( t--)
	{
		int y ; 
		cin >> y ; 
		cout << a[y] <<endl ;
		
		
	}
	return 0 ; 
}
