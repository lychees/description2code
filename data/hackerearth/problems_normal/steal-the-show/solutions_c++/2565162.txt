#include<bits/stdc++.h>
using namespace std;
bool fun(int i,int j){return i>j;}
int main()
{
	int ceo,cto,coo,i,j,k,n,a[50];
	cin>>ceo>>cto>>coo;
	cin>>n;
	for(i=0;i<n;i++)
	cin>>a[i];
	sort(a,a+n,fun);

	int M[n+3];
	if(cto>coo)
	{
		M[0]=cto;
		M[n+2]=coo;
	}
	else
	{
		M[0]=coo;
		M[n+2]=cto;
	}
	k=(n+3)/2;
	j=0;
	M[k]=ceo;
	for(i=1;i<(n+3)-1;i++)
	{
		if(i==k)
            continue;
        else
			M[i]=a[j++];
	}

    int M1[n+3],ans1=0,ans=0;
    for(i=0;i<(n+3)-1;i++)
	{
		ans+=abs(M[i+1]-M[i]);
	}

    if((n+3)%2==0)
    {
        M1[k-1]=ceo;
        j=0;
        for(i=1;i<(n+3)-1;i++)
        {
            if(i==k-1)
                continue;
            else
                M[i]=a[j++];
        }
        for(i=0;i<(n+3)-1;i++)
        {
            ans1+=abs(M[i+1]-M[i]);
        }
        cout<<min(ans1,ans)<<"\n";
    }
    else
        cout<<ans<<"\n";

}
