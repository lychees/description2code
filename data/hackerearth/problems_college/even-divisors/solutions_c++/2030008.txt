#include <iostream>
#include <math.h>
using namespace std;
#define LL long long

int main()
{
    int T;
    cin >> T;
    while(T--)
    {
    	LL n,count=0;
    	
    	cin >> n;
    	
    	if(n%2!=0)
    	cout << 0 << endl;
    	else
    	{
    		int x=int(sqrt(n));
    		for(int i=1 ; i<=x ; i++)
    		{	
    			if(n%i==0)
    			{
    				if(i%2==0)
    				count++;
    				
    				int rem=n/i;
    				
    				if(i*i!=n && rem%2==0)
    				count++;
    				
    			}
    		}
    		cout << count << endl;
    	}
    }
    return 0;
}
