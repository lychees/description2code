#include <iostream>
using namespace std;

int main()
{
    int t;
     signed long int n;
    cin>>t;
    int a[t],k=0,i,j,count;
    while(t--)
    {
    	cin>>n;
    	count=0;
    	for(i=1;i<=n;i++)
    	{
    		for(j=i+1;j<=n;j++)
    			if((i^j)<=n)
    				count++;
    		
    	}
    	a[k++]=count;
    	
    }
    for(i=0;i<k;i++)
    	cout<<a[i]<<endl;
    	
    return 0;
}
