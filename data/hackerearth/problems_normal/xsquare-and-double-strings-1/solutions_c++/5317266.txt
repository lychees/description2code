#include <iostream>
#include <stdio.h>
#include <string>
#include <algorithm>
#include <string.h>

using namespace std;

int h[27];

int main()
{
    int t;
    scanf("%d",&t);
    for(int tt=1;tt<=t;++tt)
    {
        string s;
        memset(h,0,sizeof(h));
        cin>>s;
        int ln=s.size();
        bool ok=true;
        for(int i=0;i<ln && ok;++i)
        {
            h[s[i]-'a']++;
            if(h[s[i]-'a']>1)
            {
                printf("Yes\n");
                ok=false;
            }
        }
        if(ok) printf("No\n");
    }
    return 0;
}
