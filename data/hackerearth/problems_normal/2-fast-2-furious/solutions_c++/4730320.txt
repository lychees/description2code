#include<iostream>
using namespace std;
main()
{
   int i,n,m,k,max1,max2;
   long long a[1000001],b[1000001];

 cin>>k;
 for(i=0;i<k;i++) cin>>a[i];
 for(i=0;i<k;i++) cin>>b[i];
 max1=max2=m=n=0;
 for(i=0;i<k-1;i++)
 {
    (a[i+1]-a[i])<0?(m=a[i]-a[i+1]):(m=a[i+1]-a[i]);
  // cout<<m<<endl;
    if(m>max1) max1=m;

    (b[i+1]-b[i])<0?(n=b[i]-b[i+1]):(n=b[i+1]-b[i]);
    if(n>max2) max2=n;
   //cout<<n<<endl;
 }
if(max1>max2) cout<<("Dom\n")<<max1;
else if(max1<max2) cout<<("Brian\n")<<max2;
else cout<<("Tie\n")<<max1;
}
