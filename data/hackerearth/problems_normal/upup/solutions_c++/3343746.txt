#include <iostream>
#include<stdio.h>
using namespace std;

int main() {
	string s;
	int spaceFlag=0;
	std::getline (std::cin,s);
	s[0] = toupper(s[0]);
	for(int i=0;i<s.length();i++){
		if(s[i]==' '){
			spaceFlag = 1;
			
		}
		if(spaceFlag ==1 && isalnum(s[i])){
			s[i] = toupper(s[i]);
			spaceFlag=0;
		}
	}
	cout<<s;
	return 0;
}
