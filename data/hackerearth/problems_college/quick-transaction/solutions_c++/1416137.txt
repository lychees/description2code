#include <iostream>
#include<cstdio>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t,n;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	int x=0;
    	x=n/1000;
    	n=n%1000;
    	x+=n/500;
    	n=n%500;
    	x+=n/100;
    	n=n%100;
    	x+=n/50;
    	n=n%50;
    	x+=n/20;
    	n=n%20;
    	x+=n/10;
    	n=n%10;
    	x+=n/5;
    	n=n%5;
    	x+=n/2;
    	n=n%2;
    	x+=n;
    	printf("%d\n",x);
    }
    return 0;
}
