#include <bits/stdc++.h>

#define s(x) scanf("%d", &x)
#define MAX 1000000
#define pb push_back

using namespace std;

int p[1000005], ppo[MAX];
vector<int> pr;

void sieve()
{
    int i;
    p[0] = 1;
    p[1] = 1;
    for (i = 2; i <= MAX; i++) {
        if (!p[i]) {
            pr.pb(i);
            for (int j = 1; i*j <= MAX; j++)
                p[i*j] = i;
        }
    }
}

int main()
{
    int n, i, x;
    sieve();

    s(n);

    for (i = 0; i < n; i++) {
        s(x);
        while (x > 1) {
            ppo[p[x]]++;
            x /= p[x];
        }
    }

    int q, minn = INT_MAX, cnt;
    s(q);

    while (q--) {
        s(x);
        minn = INT_MAX;
        while (x > 1) {
            cnt = 0;
            int tmp = p[x];
            //cout << x << " ";
            while (tmp == p[x]) {
                cnt++;
                x /= p[x];
            }
            minn = min(minn, ppo[tmp] / cnt);
        }
        cout << minn << endl;
    }

    return 0;
}
