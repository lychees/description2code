#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin >> n;
    vector<vector<bool>> v(n, vector<bool>(26));
    for (int i = 0; i < n; ++i) {
    	string s;
    	cin >> s;
    	for (int j = 0; j < s.size(); ++j) {
    		v[i][s[j] - 'a'] = true;
    	}
    }
    int ans = 0;
    for (int i = 0; i < 26; ++i) {
    	bool is_common = true;
    	for (int j = 0; j < n && is_common; ++j) {
    		if (!v[j][i]) {
    			is_common = false;
    		}
    	}
    	if (is_common) {
    		++ans;
    	}
    }
    cout << ans;
    return 0;
}
