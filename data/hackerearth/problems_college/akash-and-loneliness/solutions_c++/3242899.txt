#include <iostream>
using namespace std;
int maxn = 1e6+1;
const int mod = 1e9+7;
int main()
{
	
    int t ;
    cin >> t ;
    long long po[maxn] ;
    po[1] = 1 ; po[2] = 2 ; po[3] = 4 ;
    for( int i=4 ; i<maxn ; ++i )
    po[i] = ( po[i-1] + po[i-2] + po[i-3] + mod ) % mod ;
    while(t--)
    {
    	int n ;
    	cin >> n ;
    	cout << po[n] << endl ;
    	
    }
    return 0;
}
