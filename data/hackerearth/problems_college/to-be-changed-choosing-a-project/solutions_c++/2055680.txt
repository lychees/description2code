#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
    vector<int> ideas;
    int n;
    cin>>n;
    for(int i=1; i<=n; i++)
        ideas.push_back(i);

    long long r;
    cin>>r;

    long long start = -1;
    while(n != 1)
    {
        long long index = (start+r)%n ;
        start = index-1;
        ideas.erase(ideas.begin() + index);
        n--;
    }

    cout<<ideas[0];


    return 0;
}
