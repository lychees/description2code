ostream & operator<<(ostream & ost,const vector<int> & v){
	for(int i = 0 ; i < v.size(); i++){
		ost<<v[i]<<" ";
	}
	return ost;
}




int main(){
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int m, n;
	cin>>m>>n;
	priority_queue<int> p;
	int input;
	while(m--){
		cin>>input;
		p.push(input);
	}
	long long answer = 0;
	while(n--){
		int highest = p.top();
		answer += highest;
		p.pop();
		//cout<<highest<<"\n";
		p.push(--highest);
	}
	cout<<answer<<endl;
	return 0;
}
