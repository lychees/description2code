#include <bits/stdc++.h>
using namespace std;
 
int main()
{
    int n,t,l,i;
	cin>>t;
	while(t--)
	{
		cin>>n;
		int a[n];
		for(i=0;i<n;i++)
		{
			cin>>a[i];
		}
		cin>>l;
		int min[n],len=1;
		min[0]=a[0];
		for(i=1;i<n;i++)
		{
			if(a[i]<min[0])
			{
				min[0]=a[i];
			}
			else if(a[i]>min[len-1])
			{
				if(len<l)
				{
					min[len]=a[i];
					len++;
				}
			}
			else
			{
				int low=-1,high=len-1;
			
				while(low<high)
				{
					int mid=low+(high-low)/2;
					
					if(min[mid]<a[i])
					{
						low=mid+1;
					}
					else
					{
						high=mid;
					}
					
				}
				
				min[high]=a[i];
			}
		}
		if(len<l)
		{
			cout<<"-1"<<endl;
		}
		else
		{
			cout<<min[l-1]<<endl;
		}
	}
	return 0;
}
