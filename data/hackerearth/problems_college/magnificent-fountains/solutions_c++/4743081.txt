#include <boost/multiprecision/cpp_int.hpp>
#include <iostream>
namespace mp = boost::multiprecision;
using namespace std;
int gcd(int a,int b)
{
	if(b==0)
	return a;
	else
	return gcd(b,a%b);
}
int lcm(int a,int b)
{
	return (((a*b))/gcd(a,b));
}
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int a[n];
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	 mp::cpp_int l = 1;
    	for(int i=0;i<n;i++)
    	{
    		l=lcm(l,a[i]);
    
    	}
    	cout<<l%1000000007<<"\n";
    }
    return 0;
}
