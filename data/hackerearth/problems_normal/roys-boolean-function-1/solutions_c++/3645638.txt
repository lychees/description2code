#include <iostream>
#include<math.h>
using namespace std;

int gcd(int a,int b)
{
	int g=0,r=0;
	while(a%b!=0)
	{
		r=a%b;
		a=b;
		b=r;
	}
	g=b;
	return g;
	
}

int main()
{
	int n,count=0,flag=1,t,g;
	cin>>t;
	for(int k=1;k<=t;k++)
	{
		cin>>n;
		
		for(int i=1;i<=n;i++)
		{
			g=gcd(n,i);
			if(g==1)
				count++;
			
		}
		
		if(count==1)
			flag=0;
		else
		{
			for(int j=2;j<count;j++)
			{
				if(count%j!=0)
				{
					flag=1;
				}
				else
				{
					flag=0;
					break;
				}
			
			}
		}
		if(flag==1)
		{
			cout<<"TRUE"<<endl;
		}
		else
		{
			cout<<"FALSE"<<endl;
		}
		count=0;
		flag=1;
	
	}

	return 0;
}
