#include <bits/stdc++.h>
using namespace std;

int Test, N, K;
vector <int> factor;
multiset <int> S;

main()
{
    for (scanf("%d", &Test); Test--;)
    {
        scanf("%d %d", &N, &K);

        factor.clear();
        for (int i = 2; i * i <= K; i++)
            for (; K % i == 0; K /= i)
                factor.push_back(i);
        if (K != 1) factor.push_back(K);
        reverse(factor.begin(), factor.end());

        S.clear();
        for (int i = 1; i <= N; i++) S.insert(1);
        for (vector <int> :: iterator it = factor.begin(); it != factor.end(); it++)
        {
            int x = *S.begin();
            S.erase(S.begin());
            S.insert(x * (*it));
        }

        int ans = 0;
        for (multiset <int> :: iterator it = S.begin(); it != S.end(); it++)
            ans += *it;
        printf("%d\n", ans);
    }
}
