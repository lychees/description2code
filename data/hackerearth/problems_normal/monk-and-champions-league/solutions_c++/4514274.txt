#include <stdio.h>
#include <queue>
#include <iostream>
#include <set>

using namespace std ; 

int n ,m , t , sum , v;
priority_queue<int , vector<int> , less<int> > q;  

int main(int argc, char **argv)
{
    cin >> n >> m  ; 
    for(int i = 1 ; i <= n ; ++i){
        cin >> v ; 
        q.push(v) ; 
    }
    sum = 0 ; 
    while(m--){
        t = q.top() ; 
        sum += t ; 
        q.pop() , t--;
        q.push(t) ;
    }
    cout << sum ; 
	return 0;
}
