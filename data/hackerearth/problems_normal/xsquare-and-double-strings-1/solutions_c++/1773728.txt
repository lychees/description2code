#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    string st;
    string dummy;
    int arr[26];
    bool isd;
    getline(cin,dummy);
    while(t-->0)
    {
    	isd=false;
    	for(int i=0;i<26;i++)
    	    	arr[i]=0;
    	getline(cin,st);
    	for(int j=0;j<st.length();j++)
    	{
    		//cout<<"st[j]-a"<<st[j]-'a'<<endl;
    		arr[st[j]-'a']++;
    		if(arr[st[j]-'a'] == 2)
    		{
    			isd=true;
    			break;
    		}
    	}
    	if(isd == true)
    		cout<<"Yes";
    	else
    		cout<<"No";
    	cout<<"\n";
    }
    return 0;
}
