#include <iostream>
#include <cmath>

using namespace std;

int gcdFun(int a, int b)
{
    while(a!=b)
    {
        if(a>b)
            a-=b;
        else
            b-=a;
    }
	return a;
}

int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		int limit,m,n,sqrtValue,c,count = 0,x;
		cin >> limit;
		sqrtValue = sqrt((float)limit);
		for(m=1;m<sqrtValue;m++)
		{
			for(n=m+1;n<=sqrtValue;n=n+2)
			{
				if(gcdFun(m,n)==1)
				{
					x = 0;
					c = m*m+n*n;
					while(x<=limit)
					{
						x += c;
						if(x<=limit)
							count++;
						else 
							break;
					}
				}
			}
		}
		cout << count << "\n";
	}
}
