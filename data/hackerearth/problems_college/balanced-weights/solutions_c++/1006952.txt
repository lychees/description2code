#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,s;
		scanf("%d",&n);
		int a[n];
		for(int i=0;i<n;i++)
		scanf("%d",&a[i]);
		do
		{
			sort(a,a+n);
			 s=a[0];
			for(int i=1;i<n;i++)
			{
				if(a[i]%s==0)
				{
					a[i]=a[i]-(((a[i]/s)-1)*s);
				}
				else
				a[i]=a[i]%s;
			}
			sort(a,a+n);
		}while(s!=a[0]);
		long long sum=0;
		for(int j=0;j<n;j++)
		sum+=a[j];
		printf("%llu\n",sum);
	}
	return 0;
}
