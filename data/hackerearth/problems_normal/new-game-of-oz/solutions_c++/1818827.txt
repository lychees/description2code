using namespace std;

int cnt[1001];

int main()
{
    //freopen("C:\\Users\\lovelotus\\Desktop\\input.txt","r",stdin);
    //freopen("C:\\Users\\lovelotus\\Desktop\\output.txt","w",stdout);
    int t,i,n,x;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        memset(cnt,0,sizeof(cnt));
        for(i=0;i<n;i++)
        {
            scanf("%d",&x);
            cnt[x]++;
        }
        int ans=0,i=1000;
        while(i>=0)
        {
            if(cnt[i])
            {
                cnt[i]--;
                if(i>0 && cnt[i-1]) cnt[i-1]--;
                ans++;
            }
            else i--;
        }
        printf("%d\n",ans);
    }
    return 0;
}
