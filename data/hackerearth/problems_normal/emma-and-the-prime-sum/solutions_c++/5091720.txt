#include <bits/stdc++.h>
#define ll long long int
using namespace std;
  
ll primes[1000001];
ll psum[1000001];
void prime(){
  ll i,j,k,n,m;
  for (i=0; i <1000001; i++)
  {
     primes[i]=i;
  }

  for (i=2; i <1000; i++)
  {
     if(primes[i]!=0){
      for (j= 2; j*i<1000001; j++)
      {
         primes[j*i]=0;
      }
     }
  }

   primes[1]=0;
   ll a=0;
   for (i = 0; i <1000001; i++)
   {
      a= a+ primes[i];
      psum[i]=a;
   }
} 
   
int main()
{  ll t,j;
    cin>>t;
    prime();
     for(j=0;j<t;j++)
     {
     ll a,b;
       cin>>a>>b;
       ll ans=0;
       ans=psum[b]- psum[a-1];
       cout<<ans<<endl;
     }
    return 0;
}
