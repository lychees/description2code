ll p[2][2],tmp[2][2],ans[2][2],tans[2][2];

void mul(ll a[2][2], ll b[2][2])
{
	int i,j,k;
	ll c=0;
	fl(i,0,2)
	{
		
		fl(j,0,2)
		{
		
			fl(k,0,2)
			{
				c = (c+(a[i][k]*b[k][j]))%MOD;
			}
			tmp[i][j] = c%MOD;
			//cout<<i<<"--"<<j<<"=="<<tmp[i][j]<<endl;
			c=0;
		}
	}


	fl(i,0,2)
	{
		fl(j,0,2)
		{
			ans[i][j] = tmp[i][j];
			tmp[i][j] = 0;
		}
		//nline;
	}
	//nline;nline;nline;

}


void power(ll n)
{
	if (n==0 || n==1)
		return;
	power(n/2);
	mul(ans,ans);

	if(n%2)
		mul(ans,p);
}

void reset()
{
	int i,j;
	p[0][0] = 1;
	p[0][1] = 1;
	p[1][0] = 1;
	p[1][1] = 0;

	fl(i,0,2)
	{
		fl(j,0,2)
		{
			tmp[i][j] = 0;
			ans[i][j] = p[i][j];
		}
	}
	
}


int main()
{
	
	ll n;
	scanll(n);
	reset();
	ll x,i,y=0,res=0;
	fl(i,0,n)
	{
		scanll(x);
		y = __gcd(y,x);
	}

	if(y>=3)
	{
		power(y-2);
		res = (ans[0][0] + ans[0][1])%MOD;

	}
	else
		res = 1;

	//nline;
	printll(res);

	return 0;
}
