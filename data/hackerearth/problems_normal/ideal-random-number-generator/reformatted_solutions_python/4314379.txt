from fractions import gcd
a, b, c = map(int, raw_input().split())
if (a+b <= c):
    m, n = 1, 1
elif (a <= c and b <= c):
    m, n = (2*a*b-(b-c+a)*(a-c+b)), (2*a*b)
elif (a <= c):
    m, n = (2*a*c-a*a), (2*a*b)
elif (b <= c):
    m, n = (2*b*c-b*b), (2*a*b)
else:
    m, n = c*c, 2*a*b
f = gcd(m, n)
print(str(m/f)+"/"+str(n/f))
