#include <iostream>
#include<vector>
using namespace std;

#define MQ 1000000007

int gcd(int a, int b){
    if (a == 0)
        return b;
    return gcd(b%a, a);
}

int main()
{
    long long N;
    cin>>N;
    vector<long long> v(N);
    for(long long i = 0 ; i < N ; i++ ){
    	cin>>v[i];
    }
    long long g = v[0];
    for(long long i = 1 ; i < N ; i++ ){
    	g = gcd(g, v[i]);
    }
    long long value = 1;
    for(long long i = 0 ; i < N ; i++ ){
    	long long ans = 1;
    	for(long long j = 0 ; j < g ; j++ ){
    	ans = ( ( ans % MQ ) * ( v[i] % MQ ) )%MQ;
    	}
    	value = ( ( value % MQ ) * ( ans % MQ ) )%MQ;
    }
    cout<<value;
    return 0;
}
