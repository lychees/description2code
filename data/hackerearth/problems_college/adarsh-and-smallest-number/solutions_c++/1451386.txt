#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
long long int a[500002];
void sieve(){
	int i,j;
	for (i = 2; i <500002 ; ++i) {
		if (a[i] != 1) {
			for (j = 2 * i; j <500002; j += i) {
				if (a[j] % a[i] == 0) {
					a[j] /= a[i];
				}
			}
		}
		a[i] = ((a[i] % mod) * (a[i - 1] % mod)) % mod;
	}
}
int main(){
	a[0]=0;
	for(int i=1;i<500002;i++)
	a[i]=i;
	sieve();
	int t;
	scanf("%lld",&t);
	while(t--){
		long long int n;
		scanf("%lld",&n);
		int i,j=1;
	//	for(i=1;i<=n;i++)
	//	cout<<a[i]<<" ";
	
		
		cout<<a[n]<<endl;
	}
	return 0;
}
