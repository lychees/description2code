#include <bits/stdc++.h>
using namespace std;
string s;
bool ispalindrome(int sr, int e) {
	if(sr == e) {
		return true;
	}
	if((sr + 1) == e) {
		if(s[sr] == s[e]) {
			return true;
		}
		return false;
	}
	bool ans = false;
	
	if(s[sr] == s[e]) {
		ans = ispalindrome(sr+1,e-1);
	}
	return ans;
	
}
int main()
{
    int t;
    cin >> t;
    while(t--) {
    	//string s;
    	cin >> s;
    	if(ispalindrome(0,s.length()-1)) {
    		cout << "Palindrome\n";
    	}
    	else {
    		long long int ans = 1;
    		for(int i =0; i < s.length(); i++) {
    			ans = ans*(int)(s[i] - 96);
    		}
    		cout << ans << endl;
    	}
    }
    return 0;
}
