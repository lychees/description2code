#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t;
    long long int n,r;
    cin>>t;
    while(t--)
    {
        cin>>n;
        r=sqrt(n);
    //    cout<<r;
    if(n==r*r)
        cout<<r+r-1<<"\n";
    else
        if(n>=r*r && n<=((r*r)+r))
        cout<<r+r<<"\n";
    else
        cout<<r+r+1<<"\n";
    }
    return 0;
}
