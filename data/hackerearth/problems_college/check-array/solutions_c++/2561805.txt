#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int a[n];
    	int i;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	for(i=1;i<n;i++)
    	{
    		a[i]=a[i]+a[i-1];
    	}
    	int s1,s2,flag=0;
    	for(i=1;i<n-1;i++)
    	{
    		s1=a[i-1];
    		s2=a[n-1]-a[i];
    		if(s1==s2)
    		{
    			flag=1;
    			break;
    		}
    	}
    	if(flag==1)
    		cout<<"YES"<<endl;
    	else
    		cout<<"NO"<<endl;
    }
}
