#include <iostream>
using namespace std;

int main()
{
    int t,n,r=2,p=1,s=0,i;
    cin>>t;
    while(t--){
    	cin>>n;
    		for(i=1;i<n;i++){
    			s=r+p;
    			p=r;
    			r=s;
    		}
    		if(s>0)
    			cout<<(s*s)<<"\n";
    		else
    			cout<<4<<"\n";
    		r=2;
    		p=1;
    		s=0;
    }
    return 0;
}
