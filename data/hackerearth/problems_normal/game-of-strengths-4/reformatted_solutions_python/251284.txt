t = int(raw_input())
for asfas in xrange(t):
    n = int(raw_input())
    arr = sorted([int(e) for e in raw_input().split()])
    b = [0]*n
    c = [0]*n
    ab = 0
    for i in xrange(1, n):
        b[i] = b[i-1]+arr[i-1]
        c[n-i-1] = c[n-i]+arr[n-i]
        ab = ab-(b[i]-arr[i]*i) + (c[n-i-1]-arr[n-i-1]*i)
    ab /= 2
    print(ab*max(arr)) % 1000000007
