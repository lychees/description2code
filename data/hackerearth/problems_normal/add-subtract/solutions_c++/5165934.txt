#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	int t,i,j,k,l,v,a[10000],b[10000],mx,mn,n;
	cin>>t;
	while (t--)
	{
		cin>>n>>k;
		cin>>a[1];
		mx=a[1];
		mn=a[1];
		for (i=2;i<=n;i++)
		{
			cin>>a[i];
			if (a[i]>mx) mx=a[i];
			if (a[i]<mn) mn=a[i];
		}
		for (i=mn;i<=mx;i++)
		{
			l=0;
			for (j=1;j<=n;j++)
			{
				if (a[j]>i) b[j]=(a[j]-i)*5;
				else b[j]=(i-a[j])*3;
			}
			sort(b+1,b+n+1);
			for (j=1;j<=k;j++)
			l+=b[j];
			if (i==mn) v=l;
			else if (l<v) v=l;
		}
		cout<<v<<endl;
	}
}
