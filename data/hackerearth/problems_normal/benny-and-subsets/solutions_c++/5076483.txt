#include <bits/stdc++.h>
#define mp make_pair
#define pb push_back
#define ll long long
#define debug(x) cout<<">value ("<<#x<<") : "<<x<<endl;
#define fr(i,beg,end) for(i=beg;i<end;i++)
#define rf(i,beg,end) for(i=beg;i>=end;i--)
#define itfr(it,stl) for(it=stl.begin();it!=stl.end();it++)
#define PII pair<int,int>
#define init(x,val) memset(x,val,sizeof(x))
#define fst first
#define snd second
#define vi vector<int>
#define MOD 10000007
using namespace std;
ll dp[1001][1<<11];
int a[1001];
vi l,g;
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int i,j,n;
    ll x;
    cin>>n>>x;
    fr(i,1,n+1)
    {
        cin>>a[i];
        if(a[i]>(1<<10))
            g.pb(a[i]);
        else
            l.pb(a[i]);
    }
    dp[0][0]=1;
    fr(i,1,l.size()+1)
    {
        fr(j,0,1<<11)
            dp[i][j]=(dp[i][j]+dp[i-1][j]+dp[i-1][j^l[i-1]])%MOD;
    }
//    fr(j,0,11)
//    {
//        fr(i,0,l.size()+1)
//            cout<<dp[i][j]<<" ";
//        cout<<endl;
//    }
    int m=g.size();
    vector<ll> v;
    fr(i,0,1<<m)
    {
        ll ans=0;
        fr(j,0,m)
        {
            if(i&(1<<j))
                ans^=g[j];
        }
        v.pb(ans);
    }
    ll ans=0;
    fr(i,0,v.size())
    {
        ll xr=x^v[i];
        if(xr>=(1<<11))
            continue;
        ans=(ans+dp[l.size()][xr])%MOD;
    }
    cout<<ans<<endl;
    return 0;
}
