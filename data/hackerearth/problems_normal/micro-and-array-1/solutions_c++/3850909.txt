#include <iostream>
using namespace std;

int main()
{
    int n,q;
    cin>>n>>q;
    int A[n+1],a,b,op;
    for(int i=1;i<=n;i++)
    	cin>>A[i];
    while(q--)
    {
    	cin>>op;
    	if(op==0)
    	{
    		cin>>a>>b;
    		A[a]=b;
    	}
    	else
    	{
    		int flag=0,i;
    		cin>>a;
    		for(i=1;i<=n;i++)
    		{
    			if(A[i]>=a)
    			{
    				flag=i;
    				break;
    			}
    		}
    		if(flag)
    			cout<<i<<endl;
    		else
    		cout<<"-1\n";
    	}
    }
    return 0;
}
