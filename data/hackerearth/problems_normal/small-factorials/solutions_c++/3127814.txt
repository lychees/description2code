#include <iostream>
using namespace std;

int main()
{
    int t,n;
    cin>>t;
    while(t--){
        cin>>n;
        //int m=1;
        int a[200];
        for(int i=0;i<200;i++){
            a[i]=0;
        }
        int num = n;
        int k =  0;
      	while(n){
      		a[k++] = n%10;
      		n /= 10;
      	}
      	int tot = 0;
		for(int i = num-1;i > 1;i--){
			int carry = 0;
			for(int j= 0;j < 200;j++){
				a[j] = a[j]*i+carry;
				carry = a[j] / 10;
				a[j] %= 10;
				if(a[j]){
					tot = max(j,tot);
				}
			}
		}
		for(int i = tot;i >= 0;i--){
			cout << a[i];
		}
		cout << "\n";
    }
return 0;
}
