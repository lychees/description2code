#include<bits/stdc++.h>
using namespace std;

int main()
{
	int n;
	scanf("%d",&n);
	if(n==1) {
		printf("YES\n");
		printf("0 0\n");
	}
	else {
		if(n==2) {
			printf("YES\n0 0\n1 0\n");
		}
		else {
			if(n==4) {
				printf("YES\n0 0\n0 1\n1 1\n1 0\n");
			}
			else printf("NO\n");
		}
	}
	return 0;
}
