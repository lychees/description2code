#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

// 5 10 25 12 4 1 
// 3 16 28 100

int main() {
	
	int n,t;
	vector<int> vec;
	while(cin >> t) {
		while(t--) {
			cin >> n;
			if(n%2==0)
				vec.push_back(n);
		}
		
		if(vec.size()==0)
			cout << "None to display" << endl;
		else {
			for(int i=vec.size()-1; i>=0; i--)
				cout << vec[i] << ' ';
			cout << endl;
		}
		
		vec.clear();
	}
	

	
	
	return 0;
}
