#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    long int N,j;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	cin>>N;
    	long int a,s;
    	priority_queue<long int> num;
    	vector<long int> x,y;
    	for(j=0;j<N;j++)
    	{
    		cin>>a;
    		num.push(a);
    	}
    	j=0;
	    long int max=0,count=0;;
	     while(j<N)
	    {
	    	a=num.top();
	    	count=0;
	    	s=num.top();
	    	y.push_back(s);
	    	while(a==s && j<N)
	    	{
	    		count++;
	    		num.pop();
	    		s=num.top();
	    		j++;
	    	}
	    	x.push_back(count);
	    	if(max<count)
	    		max=count;
	    	
	    }
	    j=0;
   		while(j!=x.size())
	    {
	    	if(x[j]==max)
	    		cout<<y[j]<<" ";
	    	j++;
	    }
	    cout<<"\n";
    	
    }
    return 0;
}
