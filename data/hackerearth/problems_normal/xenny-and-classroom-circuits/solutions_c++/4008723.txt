#include<iostream>
#include<string>
using namespace std;

string s;
int n;

int func()
{
int c=0,d=0;
for(int i=0;i<s.size();i++)
if(s[i]=='G'&&i%2==0)
c++;
else if(s[i]=='G')
d++;
return min(c,d);
}

int main()
{
ios_base::sync_with_stdio(false);
int t;
cin>>t;
while(t--)
{
cin>>n>>s;
cout<<func()<<endl;
}
return 0;
}
