#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,i,j,count=0;;
		cin>>n;
		int arr[n];
		for(i=0;i<n;i++)
		cin>>arr[i];
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
			{
				if(arr[i]!=arr[j])
				{
					if((arr[i]+arr[j])%2==0)
					count++;
				}
			}
		}
		cout<<count/2<<"\n";
	}
	
	
	return 0;
}
