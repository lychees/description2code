#include <bits/stdc++.h> // Firstest code for length.
using namespace std;
long int a[100002];
int main()
{
    long int t,n,by,s,inc,k;
    scanf("%ld",&t);
    for(int j=1;j<=t;j++)
    {
        cin >> n;
        inc=0;
        for(int i=1; i<=n; i++)
        {
            scanf("%ld", &a[i]);
        }
        a[0]=0;
        for(int i=n;i>1;i--)
        {
            k=i;
            s=0,by=0;
            if(a[k]>a[i-1])
            {
                while(a[k]>a[i-1]&&i>=2)
                {
                    by=by+a[i-1];
                    s++;
                    i--;
                }
                i++;
            }

            inc=inc+s*a[k]-by;
        }
        if(inc>0)
            printf("%ld\n",inc);
        else printf("0\n");
    }

}
