#include<bits/stdc++.h>
using namespace std;

int main()
{
    int T,n,m,a,b,c,S,A,H,sum;
    cin>>T;
    while(T-->0)
    {
        cin>>n>>m;
        int adj[n+1][n+1];
        for(int j=1;j<=n;j++)
        {
            for(int k=1;k<=n;k++)
            {
                adj[j][k] = 1000000009;
            }
            adj[j][j] = 0;
        }
        for(int i=0;i<m;i++)
        {
            cin>>a>>b>>c;
            adj[a][b] = min(adj[a][b],c);
            adj[b][a] = adj[a][b];
        }
        cin>>S>>A>>H;
        for(int i=1;i<=n;i++)
        {
          for(int j=1;j<=n;j++)
          {
            for(int k=1;k<=n;k++)
            {
              adj[j][k]=min(adj[j][i]+adj[i][k],adj[j][k]);
            }
          }
        }
        int ans = 0;
        for(int i=1;i<=n;i++)
        {
        	if(i==S || i==H || i==A)
        		continue;
            ans=max( adj[S][i]+ 2*adj[i][A] + adj[i][H],ans);
        }
        cout<<ans<<endl;
    }
}
