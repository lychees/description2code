t = int(raw_input())
 
for i in range(t):
    i1 = raw_input().split()
    n = int(i1[0])
    k = int(i1[1])
    if k > n:
    	k = n
    nums = [int(x) for x in raw_input().split()]
    nums.sort()
    s = 0
    j = n - 1
    while k > 0:
        s = s + nums[j]
        j = j - 1
        k = k - 1
    print s
