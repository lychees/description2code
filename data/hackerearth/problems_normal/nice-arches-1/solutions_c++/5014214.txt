#include <iostream>
#include<stack>
#include<string>
using namespace std;

int main()
{
    
    string input;
    int tes_cases=0,count=0;
    cin>>tes_cases;
    while(tes_cases--)
    {
    	stack<char> check;
    	//cout<<"here"<<endl;
    	cin>>input;
    	for(int i=0;i<input.size();i++)
    	{
    		char ch=input[i];
    		//cout<<"here"<<endl;
    		if(check.empty())
    		  check.push(ch);
    		else if(check.top()==ch)
    		   check.pop();
    		else
    		  check.push(ch);
    	}
    	if(check.empty())
         count++;
    }
     cout<<count;
    return 0;
}
