#include <bits/stdc++.h>
using namespace std;
const int mod= 1e9+7;
int dp[512][108];
int a[512];
int main(int argc, char const *argv[])
{
	int n;
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		scanf("%d",&a[i]);
		dp[i][a[i]] = 1;
	}
	for(int i=1;i<n;i++){
		for(int j=i-1;j>=0;j--){
			if(a[i] > a[j]){
				for(int k=1;k<=100;k++){
					int gcd = __gcd(a[i],k);
					dp[i][gcd] += dp[j][k];
					if(dp[i][gcd] >= mod)
						dp[i][gcd] -= mod;
				}
			}
		}
	}

	int sum = 0;
	for(int i=0;i<n;i++){
		sum += dp[i][1];
		if(sum >= mod)
			sum -= mod;
	}
	printf("%d",sum);
	return 0;
}
