#include<bits/stdc++.h>
using namespace std;
int n,m;
int mat[1001][1001];
int ans=0;
int visited[1001][1001];

int fx[]={1,-1,0,0};
int fy[]={0,0,1,-1};

bool check(int x,int y)
{
       if((x>=0&&x<n)&&(y>=0&&y<m))
       {
             return true;
       }
       else
       {
            return false;
       }
}








void dfs(int k,int kk)
{
             if(check(k,kk)==false)
             {
                   return;
             }

          for(int i=0;i<4;i++)
          {
                 int x=fx[i]+k;
                 int y=fy[i]+kk;
                  if((visited[x][y]==0)&&(mat[x][y]==1))
                     {
                              visited[x][y]=1;
                              //cout<<x+1<<" "<<y+1<<endl;
                              ans--;
                            dfs(x,y);

                     }
          }


}




int main()
{
      scanf("%d%d",&n,&m);
     int q;
    scanf("%d",&q);
     for(int i=0;i<n;i++)
     {
          for(int j=0;j<m;j++)
          {
                 int s;
                   scanf("%d",&s);
                 mat[i][j]=s;
                 if(s==1)
                 {
                      ans++;
                 }

          }
     }

    // cout<<ans<<endl;
      int a,b;
     while(q--)
      {

              scanf("%d%d",&a,&b);
              a--;
              b--;
              if(mat[a][b]==1&&visited[a][b]==0){
                    visited[a][b]=1;
                    ans--;
              dfs(a,b);
              }
              printf("%d\n",ans);
             // printf("\n");

     }




     return 0;
}
