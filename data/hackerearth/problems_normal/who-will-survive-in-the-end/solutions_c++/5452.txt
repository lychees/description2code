int dp(int n,bool flip)
{
  if(n==1)
    return 1;
  if(n==2 && !flip)
    return 2;
  if(n==2 && flip)
    return 1;
  if(!flip)
  {
    if(n&1)
      return 2*(dp(n/2,!flip));
    else
      return 2*(dp(n/2,flip));
  }
  else
  {
    if(n&1)
      return 2*(dp((n+1)/2,!flip))-1;
    else
      return 2*(dp(n/2,flip))-1;
  }
}
