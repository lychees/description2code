#include <bits/stdc++.h>

#define MOD 1000000007

using namespace std;

typedef long long ll;

ll N;
map < ll, ll > memo;

ll f(ll base, ll p) {
	if (p == 0LL) return 1LL;
	if (p == 1LL) return (base%MOD);
	ll retVal = f(base, p>>1);
	retVal = (retVal*retVal)%MOD;
	if (p&1) retVal = (retVal*base)%MOD;
	retVal = retVal%MOD;
	return retVal;
}

ll dp(ll x) {
	if (x == 1LL || x == 2LL) return 1LL;
	
	if (memo.find(x) != memo.end())
		return memo[x];
	
	ll retVal = 0LL;
	
	x --;
	ll limit = floor(sqrt(x));
	for (ll i=2LL; i<=limit; i++) {
		if ((x%i) == 0LL) {
			ll y = x/i;
			retVal = (retVal + f(dp(i), y))%MOD;
			if (y != i) retVal = (retVal + f(dp(y), i))%MOD;
		}
	}
	
	retVal = (retVal + 1)%MOD; /// in one way
	retVal = (retVal%MOD);
	memo[x+1] = retVal;
	return retVal;
}

int main()
{
    cin >> N;
    
    for (ll i=1; i<1000LL; i++)
    dp(i);
    
    cout << dp(N) << "\n";
    return 0;
}
