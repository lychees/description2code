#include<iostream>
#include<stdio.h>
#include<algorithm>

using namespace std;

long long int freq[1000015];

int main()
{
 long long int n;
 scanf("%d",&n);
 long long int x;
 long long int t=0;
 for(int i=1;i<=n;i++)
 {
    scanf("%d",&x);
    freq[x]++;
    if(freq[x]==1){
        t++;
    }
 }
 long long int ans=t*(t-1);
 ans/=2;
 cout<<ans;
 
 return 0;
}
