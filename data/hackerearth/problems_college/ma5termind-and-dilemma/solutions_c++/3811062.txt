using namespace std;
int arr[100001];
int compare(int a,int b)
{
	if(a>b)
	return 1;
	return 0;
}
int main()
{
//	freopen("in.txt","r",stdin);
	int n,t,i;
	char s[100001];
	sc(t);
	while(t--)
	{
		sc(n);		
		cin>>s;
		int count=0,ans,count0=0,count1=0;
		for(i=0;i<n;i++)
		{
			arr[i]=s[i]-48;
		}
		sort(arr,arr+n);
		for(i=0;i<n;i++)
		{
			if(arr[i]!=s[i]-48)
			count0++;
		}		
		sort(arr,arr+n,compare);
		for(i=0;i<n;i++)
		{
			if(arr[i]!=s[i]-48)
			count1++;
		}
		count=min(count0,count1);

		cout<<ans<<"\n";
