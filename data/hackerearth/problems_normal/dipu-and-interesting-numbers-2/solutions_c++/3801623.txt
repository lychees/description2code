#include <iostream>
#include <math.h>
using namespace std;
typedef long long int ll;
int main()
{
    ll t;
    cin >> t;
    while(t--) {
    	ll a, b, ans;
    	cin >> a >> b;
    	ll p, q;
    	p = (sqrt(a));	
    	q = (sqrt(b));
    	ans = q- p;
    	if(q == 1000000000) {
    		if(b != q * q) {
    			ans--;
    		}
    	}
    	if((p * p) == a) {
    		ans++;
    	}
    	cout << ans << endl; 
    }
    return 0;
}
