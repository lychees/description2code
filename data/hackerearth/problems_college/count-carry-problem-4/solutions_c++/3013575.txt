#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    while(t--){
    	long int a,b;
    	int c=0,n=0;
    	cin>>a>>b;
    	
    	while(a&&b){
    		if((a%10)+(b%10)+n > 9){
    			c++;
    			n=1;
    		}
    		else n=0;
    		a=a/10;
    		b=b/10;
    	}
    	if(!c)cout<<"No carry operation"<<endl;
    	else if(c==1) cout<<c<<" carry operation"<<endl;
    	else cout<<c<<" carry operations"<<endl;
    }
    return 0;
}
