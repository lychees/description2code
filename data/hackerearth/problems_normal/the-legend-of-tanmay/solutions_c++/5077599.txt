#include <bits/stdc++.h>

using namespace std;

int main() {
	int T;
	cin >> T;
	for (int t = 0; t < T; t++) {
		int N;
		cin >> N;
		int arr[N];
		for (int i = 0; i < N; i++) {
			cin >> arr[i];
		}
		sort(arr, arr + N);
		
		int pos = 0, neg = 0, zero = 0;
		for (int i = 0; i < N; i++) {
			if (arr[i] > 0)
				pos++;
			if (arr[i] < 0)
				neg++;
			if (arr[i] == 0)
				zero++;
		}
		
		long long maxAns=1, minAns=1;
		if (neg % 2 == 1) {
			if (neg == 1 && pos == 0) {
				if(zero > 0)
					maxAns=0;
				else
					maxAns=arr[0];
			}
			else {
				for (int i = 0; i < neg-1; i++) {
					maxAns *= arr[i];
				}
				for (int i = neg-1; i < N; i++) {
					if (arr[i] > 0)
						maxAns *= arr[i];
				}	
			}
			
			for (int i = 0; i < N; i++) {
				if (arr[i] != 0)
					minAns *= arr[i];
			}
		}
		else {
			if (neg == 0 && zero == 0) {
				minAns = arr[0];
				for (int i = 0; i < N; i++) {
					maxAns *= arr[i];
				}
			}
			else if (neg == 0 && zero > 0) {
				minAns = 0;
				if (pos == 0)
					maxAns=0;
				else {
					for (int i = 0; i < N; i++) {
						if (arr[i] != 0)
							maxAns *= arr[i];
					}	
				}
			}
			else {
				if (pos==0) {
					for(int i = 0; i < neg-1; i++) {
						minAns *= arr[i];
					}
					for (int i = 0; i < neg; i++) {
						maxAns *= arr[i];
					}
				}
				else {
					for (int i = 0; i < N; i++) {
						if (arr[i] != 0)
							maxAns *= arr[i];
					}
					for (int i = 0; i < neg-1; i++) {
						minAns *= arr[i];
					}
					for (int i = neg-1; i < N; i++) {
						if (arr[i] > 0)
							minAns *= arr[i];
					}
				}
			}
		}
		cout << maxAns << " " << minAns << endl;
	}
	
	return 0;
}
