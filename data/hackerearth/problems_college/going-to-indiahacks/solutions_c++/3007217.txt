#include <iostream>
using namespace std;

#define M 1000000007

long long int count[1005][1005];
long long int numberOfPaths(int m, int n)
{
    for (int i = 0; i < m; i++)
        count[i][0] = 1;
    for (int j = 0; j < n; j++)
        count[0][j] = 1;
    for (int i = 1; i < m; i++)
    {
        for (int j = 1; j < n; j++)
            count[i][j] =  (( count[i-1][j])%M + (count[i][j-1])%M + (count[i-1][j-1])%M)%M ;
    }
    return count[m-1][n-1];
}

int main()
{
    long int T;
    cin>>T;
    numberOfPaths(1004,1004);
    
    while(T--)
    {
        int a,b,c,d;
        cin>>a>>b>>c>>d;
        long long ans = count[c-a][d-b];
        cout <<ans%1000000007<<endl;
    }
    
    return 0;
}
