#include <bits/stdc++.h>

using namespace std;

int main()
{
    long long N,Q,a,b,M=0;
    cin>>N>>Q;
    map<long long,long long> m;
    while(N--)
    {
        cin>>a>>b;
        m[abs(a-b)]++;
        M+=min(a,b);
    }
    bool flag=true;
    a=0;
    for(auto it=m.begin();it!=m.end();it++)
    {
        if(it->second%2==1)
        {
            flag=false;
            break;
        }
        a+=it->first*it->second;
    }
    M+=a/2;
    while(Q--)
    {
        cin>>b;
        if(flag && M==b)
            cout<<"YES\n";
        else
            cout<<"NO\n";
    }
    return 0;
}
