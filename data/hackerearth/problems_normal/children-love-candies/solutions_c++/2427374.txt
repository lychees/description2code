#include <bits/stdc++.h>
using namespace std;
#define ll long long

int main()
{
	int q, t;
	ll n;
	scanf("%d", &q);
	while (q--) {
		scanf("%lld %d", &n, &t);
		for (auto i = t; t > 0;  t--) {
			if (n % 2 == 0)
				n = n - n / 2;
			else n = n - (n + 1) / 2;
			n = n - n / 4;
		}		
		printf("%lld\n", n);
	}
	
	return 0;
}
