#include <iostream>
typedef long long ll;

using namespace std;

int getInvCount(ll arr[], ll n)
{
  ll inv_count = 0;
  ll i, j;
 
  for(i = 0; i <= n - 1; i++)
    for(j = i; j < n; j++)
      if(arr[i] == arr[j])
        inv_count++;
 
  return inv_count;
}
ll a[1000000] = {0};
int main()
{
    ll n = 0, nn = 0, tc = 0;
    cin >> n;
    while (n--){
    	cin >> nn;
    	//ll a[nn] = {0};
    	for (ll i = 0; i < nn; i++){
    		cin >> tc;
    		a[i] = tc;
    	}
    	cout << getInvCount(a,nn) << endl;
    }
    
    return 0;
}
