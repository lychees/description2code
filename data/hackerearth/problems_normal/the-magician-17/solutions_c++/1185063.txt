#include <bits/stdc++.h>
using namespace std;

int solve(int n, int k) {
    if (n == 1) return 1;
    if (n & 1) {
        if (k <= (n + 1) / 2) return 2 * k - 1;
        k -= (n + 1) / 2;
        int g = solve(n / 2, k);
        g = (g == n / 2) ? 1 : g + 1;
        return g * 2;
    }
    else {
        if (k <= n / 2) return 2 * k - 1;
        else return 2 * solve(n / 2, k - n / 2);
    }
}

int main() {
    int T; scanf("%d", &T);
    while (T --) {
        int n, k; scanf("%d%d", &n, &k);
        printf("%d\n", solve(n, k));
    }
    return 0;
}
