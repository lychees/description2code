#include <bits/stdc++.h>
using namespace std;

int main()
{ priority_queue <int> q;
   long long int n,a,m;
   int i;
   long long int sum,x;
   cin>>n>>m;
   
   for(i=0;i<n;i++)
   {
   	cin>>a;
   	q.push(a);
   }
   
   while(m--)
   {
   	x=q.top();
   	sum+=x;
   	q.pop();
   	q.push(x-1);
   }
   
   cout<<sum<<endl;
    return 0;
}
