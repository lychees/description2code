N = input()
a = []
hash = [0]*10005
a = map(int, raw_input().split())
for i in a:
    hash[i] += 1
sum = 0
for i in range(0, 10005):
    sum += hash[i]
    hash[i] = sum
a.sort()
ans = 1000000
for i in range(1, a[N-1]):
    if (3*i < 10004):
        ans = min(ans, N-(hash[3*i]-hash[i-1]))
print ans
