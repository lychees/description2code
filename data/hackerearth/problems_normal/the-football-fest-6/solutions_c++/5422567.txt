#include<iostream>
#include<vector>
using namespace std;

vector<long int>pass;
vector<long int>last;
void case1(int n)
{
if(!last.empty())
  last.pop_back();
last.push_back(pass.back());
pass.pop_back();
pass.push_back(n);
return;
}//end of case1

void case2(){
int x=pass.back();
pass.pop_back();
pass.push_back(last.back());
last.pop_back();
last.push_back(x);
return;
}//end of case2


int main()
{
int T,N,id;//test cases;
char ch;
cin>>T;
while(T--)
{
pass.clear();
last.clear();
cin>>N>>id;
pass.push_back(id);
while(N--)
{
cin>>ch;
if(ch=='P')
{
cin>>id;
case1(id);
}//end of if
else
 case2();

}//end of while
cout<<"Player "<<pass.back()<<endl;
}//end of T
return 0;
}
