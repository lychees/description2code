#include <iostream>
#include<cstring>
using namespace std;
int main()
{
    string str;
    int count=1;
    long long int temp=0;
    cin>>str;
    if(str.size()==1)
      temp=1;
    else
    {
    for(int i=1;i<=str.size();i++)
    {
    	if(str[i-1]==str[i])
    	  count++;
    	else{
    		temp+=(count*(count+1))/2;
    		  count=1;
    	}
    }
    }
    cout<<temp;
    return 0;
}
