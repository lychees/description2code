#include <iostream>
#include<stdio.h>
#include<string.h>
#include<stdlib.h>
#define max 1000000
using namespace std;

int main()
{
char str1[max],str2[max],curr,c1[max];
int t,len1,len2,i1[max],flag;
cin>>t;
while(t--)
{
	cin>>str1>>str2;

	len1=strlen(str1),len2=strlen(str2);
	flag=1;
	if(len1!=len2)
	{
		cout<<"No"<<endl;
        continue;
	}
	
for(int i=0,y=0;i<len1;i++)
{
	if(str1[i]!='#'){
		c1[y]=str1[i];
		i1[y]=i;
	    y++;
	}}
for(int i=0,y=0;i<len1;i++)
{
	if(str2[i]!='#')
	{
		if(str2[i]!=c1[y])
		 {
		 	
		 	flag=0;
		 	break;
		 }
		if(str2[i]=='A'&&i>i1[y])
		{
		 	flag=0;
		 	break;
			
		}
		if(str2[i]=='B'&&i<i1[y])
		{
			
		 	flag=0;
		 	break;
		}
	y++;
	}

}
if(flag==0)
cout<<"No"<<endl;
else
cout<<"Yes"<<endl;

}


}
