#include<stdio.h>
#include<algorithm>
#include<vector>
#include<math.h>
using namespace std;

int gcd(int a,int b){
	if(b == 0)
		return a;
	else
		return gcd(b,a%b);
}

int lcm(int a,int b){
	return (a*b)/gcd(a,b);
}

int max(int a,int b){
	if(a>b)
		return a;
		else
			return b;
}

int main(){
	long long high,low,mid,factors;
	int a,b,n,flag,t;
	scanf("%lld",&t);
	while(t--){
		flag = 0;
		scanf("%d%d%d",&a,&b,&n);
		low = 1;
		high = 100000000000000000LL;
		int temp = lcm(a,b);
		while(low <= high) {
		mid = (low + high)/2;
		factors = mid/a + mid/b - mid/temp;
		if(factors == n){
			flag = 1;
			break;
		}
		if(factors > n) {
			high = mid -1;
		} else {
			low = mid + 1;
		} 
		
		
		}
		printf("%lld\n",max(mid/a * a , mid/b * b));
	
	}
}
