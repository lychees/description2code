#include <iostream>
using namespace std;

int main()
{
    long long int dp[31][31]={0};
    for(int i=0;i<31;i++)
    dp[0][i]=1;
    for(int i=1;i<31;i++)
    dp[i][0]=dp[i-1][1];
    for(int i=0;i<31;i++)
    {
    	for(int j=0;j<31;j++)
    	{
    		if(i>=1&&j<30)
    		dp[i][j]=dp[i-1][j+1];
    		if(j>=1)
    		dp[i][j]+=dp[i][j-1];
    	}
    }
    
    
    
    int tc,n;
    cin>>tc;
    while(tc--)
    {
    	
    	cin>>n;
    	cout<<dp[n-1][0]<<endl;
    	
    }
    return 0;
}
