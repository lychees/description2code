int res,k;
int dfs(int n, vector<vi> &nb) {
  int sum = 0;
  for (int i = 0; i < nb[n].sz; i++)
    sum += dfs(nb[n][i], nb);
  if (sum + 1 >= k) {
    res++;
    return 0;
  } else {
    return sum + 1;
  }
}
