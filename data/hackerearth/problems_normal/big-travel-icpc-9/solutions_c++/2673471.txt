#include<bits/stdc++.h>
using namespace std;
#define M 1000000007
int main()
{
    long long ans = 0, i, n;
    cin >> n;
    long long x[n], y[n];
    for(i = 0; i < n; i++)
        cin >> x[i] >> y[i];
    sort(x, x+n);
    sort(y, y+n);
    if(n > 1)
    {
        for(i = n-1; i >= 0; i--)
        {
            ans = (ans + x[i]*i - x[i]*(n-i-1)) % M;
            ans = (ans + y[i]*i - y[i]*(n-i-1)) % M;
        }
    }
    cout << ans%M;
    return 0;
}
