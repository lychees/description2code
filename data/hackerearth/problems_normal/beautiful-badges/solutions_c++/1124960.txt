#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <math.h>

#include <queue>

using namespace std;

#define INF 1000000007
#define MAXD 1010

int x[MAXD], y[MAXD];
vector<int> cima;
vector<int> baixo;

void calcular() {
    int i, j;
    for (i=1; i<MAXD; i++) {
        for (j=1; j<MAXD; j++) {
            if (2*j*j>i*i && 2*i*i>j*j) {
                x[i]=j;
                break;
            }
        }
        for (j=MAXD-1; j>=1; j--) {
            if (2*j*j>i*i && 2*i*i>j*j) {
                y[i]=j;
                break;
            }
        }
    }
}

int main() {
    calcular();
    int n;
    scanf("%d", &n);
    int i, j;
    int c, d;
    for (i=1; i<=n; i++) {
        scanf("%d %d", &c, &d);
        if (c==1) {
            cima.push_back(d);
        }
        else {
            baixo.push_back(d);
        }
    }
    sort(cima.begin(), cima.end());
    sort(baixo.begin(), baixo.end());
    int res=0;
    i=0;
    j=0;
    while (i<cima.size() && j<baixo.size()) {
        if (y[cima[i]]<baixo[j]) {
            i++;
        }
        else if (x[cima[i]]>baixo[j]) {
            j++;
        }
        else {
            res++;
            i++;
            j++;
        }
    }
    printf("%d\n", res);
    return 0;
}
