import sys                                                                      
import string                                                                   
for line in sys.stdin:                                                          
        line=line.rstrip()                                                      
        loc=line.find("//")                                                     
        match="->"                                                              
        prefix=line                                                             
        if loc!=-1:                                                             
            prefix=line[0:loc]                                                  
        #print prefix                                                           
        new_str=string.replace(prefix,match,".")                                
        if(loc!=-1):                                                            
            msg=new_str+line[loc:]                                              
        else:                                                                   
            msg=new_str                                                         
        print msg
