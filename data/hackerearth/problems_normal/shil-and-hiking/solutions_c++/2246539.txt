#include<bits/stdc++.h>
using namespace std;
int main(){
	int n; cin>>n;
	long double a[n+1], b[n+1];
	for(int i=0; i<n; i++){
		cin>>a[i]>>b[i];
	}
	int ans=0;
	for(int i=0; i<n-1; i++){
		long double maxi=b[i+1]-b[i];
		maxi/=(a[i+1]-a[i]);
		ans++;
		for(int j=i+2; j<n; j++){
			long double m=b[j]-b[i]; m/=(a[j]-a[i]);
			if(m>=maxi) ans++;
			maxi=max(maxi,m);
		}
	}
	cout<<ans<<endl;
}
