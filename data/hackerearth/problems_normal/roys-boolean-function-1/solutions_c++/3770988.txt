#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;

int main()
{
	int t,N,p=0,i;
    cin>>t;
    while(t--)
    {
        p=0;
    	cin>>N;
    	if(N==1 || N==2)
        {
            cout<<"FALSE\n";
            continue;
        }
    	for(i=2; i<N; i++)
    	{
    		if(N%i!=0)
    			if(__gcd(i,N)==1)
    				p++;
    	}
    	if( p==1 || p==2 || p==4)
        {
            cout<<"TRUE\n";
            continue;
        }
        if( p==3)
        {
            cout<<"FALSE\n";
            continue;
        }
    	for(i=2; i<=sqrt(p+1); i++)
        {
            if((p+1)%i==0)
            {
                cout<<"FALSE\n"; break;
            }
        }
        if(i==sqrt(p+1))
            cout<<"TRUE\n";
    }
    return 0;
}
