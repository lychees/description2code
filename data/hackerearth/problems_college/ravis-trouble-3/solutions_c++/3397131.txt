#include<iostream>
#include<string>
#include<cstring>
using namespace std;
int main()
{
	string s,p;
	int t,i,ans,j,start;
	char c;
	cin>>t;
	while(t--)
	{
		cin>>s>>p;
		start=0;
		ans=0;
		for(i=0;i<p.length();i++)
		{
			for(j=start;j<s.length();j++)
			{
				//cout<<p[i]<<" "<<s[j]<<endl;
				if(p[i]==s[j])
				{
					start=j+1;
					break;
				}
			}
			if(j==s.length())
			{
				ans=-1;
			}
		}
		if(ans==-1)
			cout<<"NO"<<endl;
		else cout<<"YES"<<endl;
		
	}
}
