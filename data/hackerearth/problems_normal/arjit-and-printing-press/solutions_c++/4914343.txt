#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	string s,p;
    	cin>>s>>p;
    		int k=0;
    	sort(p.begin(),p.end());
    	for(int i=p.size();i<s.size();i++)
    		p+='z';
    
    	for(int i=0;i<s.size();i++){
    		if(s[i]>p[k]){
    		s[i]=p[k];
    		k++;
    		}
    		
    	}
    	cout<<s<<endl;
    }
    return 0;
}
