#include<bits/stdc++.h>
using namespace std;
int main()
{
		long long int a,b,gcd,l,r,t,i;
		vector<long long int> vt;
		scanf("%lld %lld",&a,&b);
		gcd=__gcd(a,b);
		for(i=1; i*i<gcd; ++i )
		{
				if(gcd%i==0)
				{
						vt.push_back(i);
						vt.push_back(gcd/i);
				}
		}
		if(i*i==gcd) vt.push_back(i);
		sort(vt.begin(),vt.end());
		scanf("%d",&t);
		while(t--)
		{
				scanf("%lld %lld",&l,&r);
				i=upper_bound(vt.begin(),vt.end(),r)-vt.begin();
				if(i==0)
				{
						printf("-1\n");
				}
				else
				{
						if(vt[i-1]>=l && vt[i-1]<=r) printf("%lld\n",vt[i-1]);
						else printf("-1\n");
				}
		}
		return 0;
}
