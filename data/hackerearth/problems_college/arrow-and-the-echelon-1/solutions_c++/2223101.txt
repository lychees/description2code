#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main()
{	ios::sync_with_stdio(0);
	cin.tie(0);
	int t;
	cin>>t;
	while(t--){
		ll x,i;
		string an;
		cin>>x;
		if(x==1)
		{
			cout<<-1<<endl;
			continue;
		}else
		{
			if(x%5==0){
				for(i=0;i<x;i++)
				an[i]='3';
			}else if(x%3==0){
				for(i=0;i<x;i++)
				an[i]='5';
			}
			else{
				ll tmp=0;
				for(i=x;i>0;i--)
				{
					if(i%5==0)
					tmp=i;
					
				}
				for(i=0;i<(x-tmp);i++)
				an[i]='5';
				for(i=(x-tmp);i<x;i++)
				an[i]='3';
				
			}
		}
		for(i=0;i<x;i++)
		cout<<an[i];
		cout<<endl;
		
	}
    return 0;
}
