#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,s,q,g,m;
	cin>>n;
	int ar[1000005];
	memset(ar,0,sizeof(ar));
	for(int i=0;i<n;++i)
	{
		cin>>s>>q;
		ar[s]=q;
	}
	int maxx[1000005],temp=0;
	maxx[0]=0;
	for(int i=1000000;i>0;--i)
	{
		temp=max(temp,ar[i]);
		maxx[i]=temp;
	}
	cin>>g;
	long long int ans=0;
	while(g--)
	{
		cin>>m;
		ans=ans+(maxx[m]/m)*100;
	}
	cout<<ans<<endl;
	return 0;
}
