#include <iostream>
using namespace std;

int main()
{
   long long int t,n,ans;
   cin>>t;
   while(t--){
   	cin>>n;
   	if(n==1)cout<<"1\n";
   	else{
   	if(n&1){
   		ans=(2*(n/2)*((n/2)+1))-1;
   		cout<<ans<<endl;
   	}
   	else{
   		ans=(2*((n-2)/2)*(((n-2)/2)+1))+n-1;
   		cout<<ans<<endl;
   	}
   }}
    return 0;
}
