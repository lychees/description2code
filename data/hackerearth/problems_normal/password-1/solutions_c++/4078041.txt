#include <iostream>
using namespace std;

bool match(string s1, string s2)
{
	int i, j;
	if(s1.length() !=s2.length()) return false;
	else
	{
		int l=s1.length();
		for(i=0, j=l-1; i<l; i++, j--)
			if(s1[i] !=s2[j])
				return false;
		
	}
	return true;
}

char mid(string s)
{
	int l = s.length();
	return s[l/2];
}

int main()
{
    //cout << "Hello World!" << endl;
    int t, N;
    
    	string s[100];
    	bool flag=false;
    	cin>>N;
    	for(int i=0; i<N; i++)
    	{
    		cin>>s[i];
    		for(int j=i-1; j>=0; j--)
    			if(match(s[i],s[j]))
    			{
    				flag= true; break;
    			}
    		if(flag ==true)
    		{
    			cout<<s[i].length()<<" "<<mid(s[i])<<"\n";
    			break;
    		}
    	}
    
    return 0;
}
