#include <iostream>
using namespace std;
 
int main()
{
    long t,a,b;
    cin>>t;
    while(t!=0)
   {
   int a1,b1,c1,cnt=0,sum;
   c1=0;
   cin>>a>>b;
   while(a!=0 || b!=0)
   {
	a1=a%10;
	b1=b%10;
	sum=a1+b1+c1;
	c1=(sum/10);
	if(c1!=0)
	  cnt++;
	a=a/10;
	b=b/10;
  }
  if(cnt==0)
	cout<<"\nNo carry operation";
  else if(cnt==1)
	cout<<"\n1 carry operation";
  else
	cout<<"\n"<<cnt<<" carry operations";
  t--;
 }
    return 0;
}
