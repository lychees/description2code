#include <iostream>
using namespace std;

int main()
{
    int t;
    int a,b,c,i;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b>>c;
    	cout<<a/b;
    	if(c!=0)
    	{
    		a=a%b;
    		cout<<'.';
    		for(i=0;i<c;i++)
    		{
    			a*=10;
    			cout<<a/b;
    			a=a%b;
    		}
    	}
    	cout<<endl;
    }
    return 0;
}
