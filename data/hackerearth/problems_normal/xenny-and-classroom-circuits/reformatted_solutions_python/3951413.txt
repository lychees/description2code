q = input()
for _ in xrange(q):
    x = input()
    t = raw_input()
    a, b, n, m = 0, 0, 0, 0
    for y in xrange(2*x):
        if y % 2 == 0:
            if t[y] == 'B':
                a += 1
            else:
                b += 1
        else:
            if t[y] == 'B':
                n += 1
            else:
                m += 1
    print min(a, b, n, m)
