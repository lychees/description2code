#include <iostream>
#include<vector>
#include<algorithm>
#include<unordered_map>
using namespace std;

int main()
{
    int n;
    cin>>n;
    vector<vector<long long int>> vec;
    vec.resize(n);
    for(int i=0;i<n;i++)
    {
    	vec[i].resize(3);
    	for(int j=0;j<3;j++)
    		cin>>vec[i][j];
    	sort(vec[i].begin(),vec[i].end());	
    }
    sort(vec.begin(),vec.end());
	int cnt=1;
	vector<long long int> fin;
	fin=vec[0];
	int sum=0;
	for(int i=1;i<n;i++)
    {
    	if(vec[i]==fin)
    		cnt++;
    	else
    	{
    		if(cnt==1)	
    			sum++;
    		fin=vec[i];
    		cnt=1;
    	}
    }
   	if(cnt==1)
   		sum++;
    cout<<sum<<endl;	
}
