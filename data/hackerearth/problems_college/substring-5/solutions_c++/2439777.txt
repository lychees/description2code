#include <iostream>
using namespace std;

int main()
{
    string s;
    cin >> s;
    int count =1;
    int ans=0;
    for(int i=1;i<s.size();i++)
    {
    	if(s[i]==s[i-1])
  			count++;
  		else
  		{
  			ans += (count * (count+1))/2;
  			count = 1;
  		}
    }
    ans += (count * (count+1))/2;
    cout << ans;
    return 0;
}
