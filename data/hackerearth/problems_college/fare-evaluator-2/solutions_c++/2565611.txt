#include<bits/stdc++.h>
using namespace std;
int main()
{
    int T;
    cin>>T;
    while(T--)
    {
    	int WT,D,maxi=0;
      cin>>D>>WT;
     
      if(D<=60)
         {
           if(D==1)
           {
           maxi=50+WT*2;
           }
           else{
          maxi=50+(D-2)*12+WT*2;
           }
            cout<<maxi<<endl;
           
      }
      else
      cout<<"NA"<<endl;
    }
    return 0;
}
