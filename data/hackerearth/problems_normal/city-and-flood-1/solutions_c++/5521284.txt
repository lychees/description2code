#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll int a[100000];
ll int s[100000],k,x,y;
ll int root(ll int a[],ll int i)
{
	
	while(i!=a[i])
	{
	i=a[i];
	
	}
return a[i];
}
bool find(ll int a[],ll int i,ll int j)

{

    x=root(a,i);
    y=	root(a,j);
    if(x==y)
    return true;
    return false;
}
void uni(ll int a[],ll int i,ll int j)
{
	
	if(!find(a,i,j))
	{
	  x=root(a,i);
	  y=root(a,j);
	  if(s[x]>s[y])
	  {


	  a[y]=x;
	  s[x]+=s[y];
	  }
	  else
	  
	  {
	  	a[x]=y;
	  s[y]+=s[x];
	}
	k--;
    
}
}
int main()
{
	ll int n,q;
	cin>>n>>q;
	k = n;
	for(ll int i = 1; i <= n; i++) {
		a[i] = i;
		s[i] = 1;
	}
	while(q--)
	{
      cin>>x>>y;
    uni (a,x,y);

	}
     cout<<k<<endl;
return 0;
}
