#include <iostream>
using namespace std;
#define MOD 1000000007

long long dp[505][505];
long long alls;

long long countbst(long long n, long long st, long long h)
{
	
	if(h==0) return 0;
	if(n<=1) return 1;
	if(dp[n][h]!=-2) return dp[n][h];
	long long count=0;
	long long sum=0;
	for(int i=st;i<st+n;i++)
	{
		long long cnt=0;
		cnt=(cnt+((countbst(i-st,st,h-1)%MOD)*(countbst(n+st-i-1,i+1,h-1)%MOD))%MOD);
		sum=((sum%MOD) + ((cnt%MOD) * (i%MOD)%MOD)%MOD);
		count = (count%MOD + cnt%MOD)%MOD;
	}
	alls=sum%MOD;
	dp[n][h]=count%MOD;
	return dp[n][h];
}

int main()
{
    long long n,h;
    cin>>n>>h;
    for(int i=0;i<=n;i++)
    {
    	for(int j=0;j<=h;j++)
    	{
    		dp[i][j]=-2;
    	}
    }
    countbst(n,1,h);
    cout<<alls<<endl;
}
