#include<iostream>
#include<algorithm>
#include<cstdio>
#define MIN(a,b) (a<b?a:b)
using namespace std;
int main()
{
    int t,len,k;
    cin>>t;
    while(t--)
    {
        cin>>len>>k;
        int i=0;
        int a,b,c,j,i2;
        int grid[500][500]={0};
        while(i<k)
        {
            cin>>a>>b>>c;

                for(i2=a;i2<=b;i2++)
                {
                    for(j=a+1;j<=b;j++)
                    {
                         if(grid[i2][j]==0)
                            grid[i2][j]=c;
                        grid[i2][j]=MIN(grid[i2][j],c);
                    }
                }

              //  cout<<a<<" "<<b<<" = "<<grid[a][b]<<endl;


            i++;
        }

        for(int l=2;l<=len+1;l++)
        {

            for(int i=0;i<=len-l+1;i++)
            {
             //   cout<<"calculating length "<<l<<" from "<<i<<" to "<<i+l-1<<endl;
                for(k=1;k<l-1;k++)
                {
                    int temp=grid[i][i+k]+grid[i+k][i+l-1];
                    if(grid[i][i+k]==0 || grid[i+k][i+l-1]==0)
                        continue;
                    if(grid[i][i+l-1]==0)
                        grid[i][i+l-1]=temp;
                    grid[i][i+l-1]=MIN(grid[i][i+l-1],temp);
                }
            }
        }
        /*for(int i=0;i<=len;i++)
        {
            for(int j=0;j<=len;j++)
            {
                printf("%5d ",grid[i][j]);
            }
            cout<<endl;
        }*/
        if(grid[0][len]!=0)
        cout<<grid[0][len]<<endl;
        else
        cout<<"-1"<<endl;
    }
    return 0;
}
