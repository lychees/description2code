#include <iostream>
using namespace std;

int main()
{
    //cout << Hello World!" << endl;
    
    int t;
    cin>>t;
    while(t--){
    	int n,b;
    	cin>>n>>b;
    	if(b&1){
    		if( n%b <= b/2 ) cout<<n-n%b<<endl;
    		else cout<< n + b - n%b <<endl;
    		continue;
    	}
    	if( n%b < b/2 ) cout<<n-n%b<<endl;
    	else cout<< n + b - n%b <<endl;
    }
    return 0;
}
