typedef long long ll;
typedef vector<int> VI;
typedef pair<int, int> PII;
typedef vector<VI> matrix;
const ll mod = 1000000007LL;

ll power(ll a, ll n)
{
	if(n==0)
		return 1;
	if(n==1)
		return a;
	ll b=power(a, n/2);
	ll c=(b*b)%mod;
	if(n%2==0)
		return c;
	return (a*c)%mod;
}

ll fact[1000100];
void precal()
{
	fact[0]=1;
	for(int i=1;i<1000100;i++){
		fact[i]=(fact[i-1]*i)%mod;
	}
}

ll comb(ll n, ll r)
{
	ll nr=fact[n];
	ll dr=(fact[r]*fact[n-r])%mod;
	dr=power(dr, mod-2);	//fermat
	return (nr*dr)%mod;
}
int main()
{
    //freopen("input.txt", "r", stdin);
    //freopen("output.txt", "w", stdout);
    int t;
    precal();
    cin>>t;
    while(t--){
        ll k, m;
        cin>>k>>m;
        for(int i=0; i<k; i++){
            ll x;
            cin>>x;
            m -= x;
        }
        if(m < 0){
            cout<<"0\n";
        }
        else{
            cout<<comb(m+k-1, k-1)<<"\n";
        }
    }
    return 0;
}
