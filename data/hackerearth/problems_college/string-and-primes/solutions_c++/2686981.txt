#include <iostream>
using namespace std;

int main()
{
    string s;
    int a[256],t,i,flag;
    a[0]=a[1]=1;
    for(i=2;i<256;i++)
    	a[i]=0;
    for(i=2;i<256;i++)
    {	if(a[i]==0)
    	{
    	for(int j=i+i;j<256;j+=i)
    		a[j]=1;
    	}
    }
    cin>>t;
    while(t--)
    {	
    	flag=1;
    	cin>>s;
    	for(i=0;s[i]!='\0';i++)
    	{
    		if(a[s[i]]==1)
    		{
    			cout<<s[i];
    			flag=0;
    		}
    	}
    	if(flag)
    		cout<<"NULL";
    	cout<<"\n";
    }
    
    return 0;
}
