#include <iostream>
using namespace std;

int main()
{
    long long int test,i,j,k,l,m,n;
    cin>>test;
    string str,str1;
    while(test--){
    	int arr[26]={};
    	cin>>str>>str1;
    	for(i=0;i<str.length();i++){
    		arr[str[i]-'a']++;
    	}
    	for(i=0;i<str1.length();i++){
    		arr[str1[i]-'a']--;
    	}
    	k=0;
    	for(i=0;i<26;i++){
    		if(arr[i]!=0){
    			k=1;
    			break;
    		}
    	}
    	if(k==0)
    	cout<<"Yes"<<endl;
    	else
    	cout<<"No"<<endl;
    }
    return 0;
}
