int main()
{
    int t;
    int n,i,j,p;
    int a[1000000]={0};
    a[1] = 1;
    fr(i,2,1001){
        for(j=2;pow(i,j)<=1000000;j++){
            p=pow(i,j);
            a[p] = 1;
        }
    }
    cin>>t;
    while(t--){
        cin>>n;
        bool c=false;
        fr(i,1,(n/2)+1){
            if(a[i] && a[n-i]) {
                c=true;
                break;
            }
        }
        if(c)
        	cout<<"Yes"<<endl;
        else
        	cout<<"No"<<endl;
        }
return 0;
