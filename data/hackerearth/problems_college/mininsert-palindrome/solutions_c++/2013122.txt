#include <bits/stdc++.h>

typedef long long ll;
#define MOD 1000000007
#define pb push_back
#define mp make_pair
#define ff first
#define ss second
#define MAXN 1005

using namespace std;

int main()
{
    ios_base::sync_with_stdio(0);cin.tie(0);
    char str[25],rev[25];
    int i,j,l,t;
    str[0]='a';
    cin>>t;
    while(t--){

    cin>>l;
    cin>>str+1;

    for(i=1;i<=l;i++)
    {
        rev[i]=str[l-i+1];
    }
    int dp[25][25];
    for(i=0;i<=l;i++)
       dp[i][0]=0;

    for(i=0;i<=l;i++)
       dp[0][i]=0;

    for(i=1;i<=l;i++)
    {
        for(j=1;j<=l;j++)
        {
            if(str[i]==rev[j])
                dp[i][j]=dp[i-1][j-1]+1;
            else
                dp[i][j]=max(dp[i-1][j],dp[i][j-1]);
        }
    }
    cout<<l-dp[l][l]<<'\n';
    }







    return 0;
}
