#include <bits/stdc++.h>

using namespace std;

#define int long long

int n, m;

inline int getA()
{
    int temp = min(n, m);
    return (m + 1) * ((temp - 1) * temp / 2) + temp;
}

inline int getB()
{
    int temp = min(n, m);
    return (n + 1) * ((temp - 1) * temp / 2) + temp;
}

inline int solve()
{
    return getA() + getB();
}

main()
{
    int t; cin >> t;
    while (t--)
    {
        cin >> n >> m;
        cout << solve() << endl;
    }
}
