#include<bits/stdc++.h>
using namespace std;
int t,n,m,a,b;
int edges[40005][2];
int visited[5003];
int main()
{
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d",&n,&m);
        memset(visited,0,sizeof visited);
        for(int i=0;i<m;i++)
        {
            scanf("%d%d",&a,&b);
            edges[i][0]=a;
            edges[i][1]=b;
        }
        for(int i=0;i<n-1;i++)
        {
            scanf("%d",&a);
            visited[edges[a-1][0]]=1;
            visited[edges[a-1][1]]=1;
        }
        sort(visited,visited+n+1);
        if(visited[1]==0)printf("NO\n");
        else printf("YES\n");
    }

    return 0;
}
