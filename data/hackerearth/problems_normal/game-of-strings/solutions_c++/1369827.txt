#include <iostream>
#include <stdio.h>
#include <string>
#include <string.h>
#include <math.h>
using namespace std;
 
int a[5001][5001];
int main()
{
	string x, y;
	cin>>x>>y;
	for(int i=0;i<=x.size();i++)
		a[i][0] = 0;
	for(int j=0;j<=y.size();j++)
		a[0][j] = 0;
		
	for(int i=1;i<=x.size();i++)
	{
		for(int j=1;j<=y.size();j++)
		{
			if(x[i-1]==y[j-1] && x[i-1]!='a' && x[i-1]!='e' && x[i-1]!='i' &&
			x[i-1]!='o' && x[i-1]!='u')
				a[i][j] = 1+a[i-1][j-1];
			else
				a[i][j] = max(a[i-1][j], a[i][j-1]);
		}
	}
	printf("%d\n",a[x.size()][y.size()]);
	return 0;
}
