#include<iostream>
 
using namespace std;
int a[1001][1001];
int count[1001];
int col[1001]={0};
int times[1001];
 int n,k;
 
void try_remove(int i)
{  
   if(times[i]-1==0)
   return;
 
   bool flag=1;
   for(int j=1;j<=n;j++)
   {
           if(a[i][j]==1)
           flag=flag*(times[j]-1!=0);
   }
 
      if(flag==1)
   {
      col[i]=1;
      times[i]--;
      for(int j=1;j<=n;j++)
      { if(a[i][j]==1)
        times[j]--;
      }
   }
 
}
 
 
int main()
{
 
    cin>>n>>k;
 
 
     int x,y;
   for(int i=0;i<k;i++)
   {cin>>x>>y;
    if(x!=y){
    a[x][y]=1;
    a[y][x]=1;
    }
   }
 
    for(int i=1;i<=n;i++)
    {
       for(int j=1;j<=n;j++)
       {if(a[i][j]==1)
        count[i]++;
        }    
    }
    for(int i=1;i<=n;i++)
    {times[i]=1+count[i];}
 
    int max_v=0;
    for(int i=1;i<=n;i++)
    {if(max_v<count[i])
     {max_v=count[i];}
    }
    for(int i=0;i<=max_v;i++)
    {
       for(int j=1;j<=n;j++)
       {if(count[j]==i)
         try_remove(j);
       }
 
    }
    int count=0;
    for(int i=1;i<=n;i++)
    {if(col[i]==0)
      count++;
    }
    if(k!=n+1&&k!=n)
    cout<<count-6<<endl;
    else
    cout<<count<<endl;
 
 
 
  return 0; 
}
