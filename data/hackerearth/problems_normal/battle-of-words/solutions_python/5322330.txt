t = int(raw_input())
for kappa in range(t):
    alice = raw_input()
    bob = raw_input()
    alpha = 'abcdefghijklmnopqrstuvwxyz'
    alicel = []
    bobl = []
    for letter in alpha:
        alicel.append(alice.count(letter))
        bobl.append(bob.count(letter))
    lst = []    
    for i in range(26):
        diff = alicel[i]-bobl[i]
        lst.append(diff)
    if max(lst)>0 and min(lst)>=0:
        print 'You win some.'
    elif max(lst)<=0 and min(lst)<0:
        print 'You lose some.'
    else:
        print 'You draw some.'
