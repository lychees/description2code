#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
   long long  int t,a,b,ans;
    cin>>t;
    while(t--)
        {
        cin>>a>>b;
        ans=0;
        for(int i=0;i<32;++i)
            {
            long long int k=(a + (1ll << i)) >> i;
            long long int x = (k << i)-1;
            if(x>=b)
                {
                if((k&1)==0)ans|=1ll << i;
            }                
        }
        cout<<ans<<endl;
    }
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */   
    return 0;
}
