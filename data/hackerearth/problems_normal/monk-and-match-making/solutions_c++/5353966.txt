#include<bits/stdc++.h>
#define ll long long
#define MOD 1000000007
using namespace std;
int find_hash(int l,int r,ll *h,ll *pow)
{
	int length=(r-l+1);
	return ((h[r]+MOD-(h[l-1]*pow[length])%MOD)%MOD);
}
bool compare(char *s,int l1,int r1,int l2,int r2)
{
	if(r1-l1 != r2-l2)
		return false;
	int i=l1,j=l2;
	while(i<=r1)
	{
		if(s[i]!=s[j])
			return false;
		i++;
		j++;
	}
	return true;
}
