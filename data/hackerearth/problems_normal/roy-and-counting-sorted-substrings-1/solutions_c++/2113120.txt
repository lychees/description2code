#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ll int test;
	scanf("%lld",&test);
	while(test--)
	{
		ll int n;
		scanf("%lld",&n);
		char str[1000010];
		scanf("%s",str);
		ll int ans=0;
		ll int mc;
		ll int i;
		mc=1;
		for(i=1;i<n;i++)
		{
			if(str[i]>=str[i-1])
			{
				mc++;
			}
			else
			{
				ans+=((mc *(mc+1))/2);
				mc=1;
			}
		}
		ans+=((mc *(mc+1))/2);
		printf("%lld\n",ans);
	}
	return 0;
}
