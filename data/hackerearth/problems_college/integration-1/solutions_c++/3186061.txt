#include<iostream>
using namespace std;
int main()
{
	int n,i,u,r,s,j,k,t,test,in;
	scanf("%d",&t);
	for(test=1;test<=t;test++)
	{
	
	scanf("%d",&n);
	scanf("%d",&i);
	scanf("%d",&u);
	scanf("%d",&r);
	scanf("%d",&s);
	for(in=0;in<n;in++)
	{
	//	printf("%d %d %d\n",i,r,s);
		if(i+r<=u)
		{
			i=i+r;
		}
		else if(i-s>=0)
		{
		//	printf("%d %d %delse\n",i,r,s);
			i=i-s;
		}
		else
		{
			break;
		}
	}
	printf("Case #%d: %d\n",test,i);
}
	
	return 0;
}
