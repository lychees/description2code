#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long t;
    cin>>t;
    while(t--)
    {
    	long long n;
    	long long i,j,cont=0;
    	cin>>n;
    	for(i=1;i<=n;i++)
    	{
    		for(j=i;i*j<=n;j++)
    		cont++;
    	}
    	cout<<cont<<endl;
    }
    return 0;
}
