#include<iostream>
#include<bits/stdc++.h>
using namespace std;
long int a[1000000];
main()
{
	long int n,i;
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long int max=-1,count=0;
		scanf("%ld",&n);
		for(i=0;i<n;i++)
		scanf("%ld",&a[i]);
		for(i=n-1;i>=0;i--)
		{
			if(a[i]>max)
			{
				count++;
				max=a[i];
			}
		}
		printf("%ld\n",count);
	}
}
