#include<bits/stdc++.h>
#define ll long long
using namespace std;
struct node
{
	ll cc;
	ll r;
};
node no[1000008];
int compare(node a,node b)
{
	if(a.cc<b.cc)
	return 1;
	if(a.cc==b.cc)
	{
		if(a.r>b.r)
		return 1;
	}
	return 0;
}
int main()
{
//	freopen("in.txt","r",stdin);
	int n,i;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>no[i].cc>>no[i].r;
	}
	sort(no,no+n,compare);
	for(i=0;i<n;i++)
	{
		cout<<no[i].cc<<" "<<no[i].r<<"\n";
	}


return 0;
}
