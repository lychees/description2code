#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    string arr="abcdefghijklmnopqrstuvwxyz";
    while(t--){
    	string s,t;
    	cin>>s;
    	cin>>t;
    	int slen=s.length();
int tlen=t.length();
int i,index,sum=0,temp=0;
int freqs[26]={0};
int freqt[26]={0};
char c;
int x=0;
for(i=0;i<slen;i++){
	c=s[i];
	index=arr.find(c);
	freqs[index]++;
}
for(i=0;i<tlen;i++){
	c=t[i];
	index=arr.find(c);
	freqt[index]++;
}

for(i=0;i<26;i++){
	if(freqs[i]>freqt[i]){
		temp=1;
		break;
	}
}
if(temp==0){
	for(i=0;i<26;i++){
		x=freqt[i]-freqs[i];
		if(x>0)
		sum++;
	}
	cout<<sum<<endl;
}
else if(temp==1)
cout<<-1<<endl;
    }
    return 0;
}
