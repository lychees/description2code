#include <bits/stdc++.h>
using namespace std;
int bit[100005][26],n;
char s[100005];

void update(int idx, int z, int val) {
	while (idx <= n) {
		bit[idx][z] += val;
		idx += (idx & -idx);
	}
}

int query(int idx, int z) {
	int sum=0;
	while (idx) {
		sum += bit[idx][z];
		idx -= (idx & -idx);
	}
	return sum;
}

int main()
{
    int q,i,j,k;
    scanf("%d%d",&n,&q);
    getchar();
    scanf("%s",s+1);
    for (i=1 ; i <= n ; i++) {
    	update(i,s[i]-'a',1);
    }
    while (q--) {
    	scanf("%d",&k);
    	if (k==1) {
    		int l; char c;
    		scanf("%d",&l);
    		getchar(); c = getchar();
    		if (s[l] != c){
    			update(l,s[l]-'a',-1);
    			update(l,c-'a',1);
    		}
    		s[l]=c;
    	}
    	if (k==2) {
    		int l,r ;
    		scanf("%d%d",&l,&r);
    		int odd=0;
    		for (i=0 ; i < 26; i++) {
    			if ((query(r,i)-query(l-1,i))%2 == 1)
    				odd++;
    		}
    		if (odd<2) {
    			printf("yes\n");
    		} else printf("no\n");
    	}
    }
    return 0;
}
