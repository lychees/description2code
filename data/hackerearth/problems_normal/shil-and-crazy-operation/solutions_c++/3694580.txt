using namespace std;
#include<bits/stdc++.h>

#define BG begin()
#define ED end()
#define st first
#define nd second
#define PB push_back
#define PF push_front
#define FOR(i,a,b) for (long long i=a;i<b;i++)
#define FORE(i,a,b) for (long long i=a;i<=b;i++)
#define FORD(i,a,b) for (long long i=a;i>=b; i--)
#define TR(c, it) for(typeof((c).begin()) it=(c).begin(); it!=(c).end(); it++)
#define PI 2 * acos(0)
#define ri(n)({\
    int neg=0;\
    n=0;\
    char ch;\
    for(ch=getchar(); ch<'0' || ch>'9'; ch=getchar()) if (ch=='-') neg=1-neg;\
    n=ch-48;\
    for(ch=getchar(); ch>='0' && ch<='9'; ch=getchar()) n=(n<<3)+(n<<1)+ch-48;\
    if (neg) n=-n;\
})

typedef long long ll;

int dem,n,a[100000+10],p[100000+10],chep,ans[100000+10],vt,value;
bool f[100000+10];
ll t,tt,tt2;
vector<int> res[100000+10];

void dfs(int u){
    if (!f[u]) return;
    dem++;
    f[u]=false;
    res[chep].PB( u );
    dfs(p[u]);
}

int main(){
 //   freopen("shil-and-crazy-operation.inp", "r", stdin);
 //   freopen("shil-and-crazy-operation.out", "w", stdout);
    ios_base::sync_with_stdio (false);
    cin.tie();
    cin >> n >> t;
    tt=t;
    FORE(i,1,n) cin >> a[i];
    FORE(i,1,n) cin >> p[i];
    memset(f,true,sizeof(f));
    chep=0;
    tt2=1;
    FORE(i,1,n)
    if (f[i]){
        dem=0;
        chep++;
        dfs(p[i]);
        tt/=dem;
        tt2*=dem;
    }
    if (tt>1) t%=tt2;

    FORE(i,1,chep){
        value=res[i].size();
        FOR(j,0,value){
            vt=(j+ (t%value))%value;
            ans[ res[i][j] ]=a[res[i][vt]];
        }
    }
    FORE(i,1,n) cout << ans[i] << " ";
}
