#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
int t;
cin>>t;
while(t--){
	
int n;
cin>>n;
int a[n];
int flag=0;
int zombie;
for(int i=0;i<n;i++){cin>>a[i];}
int count=0;
sort(a+0,a+n);
for(int i=1;i<n;i++){
if(a[i]-(count+i)<=0){flag=1;zombie=i;break;}
if((i+1)%6==0){count++;}
}
if(flag==0){cout<<"Rick now go and save Carl and Judas"<<endl;}
else{cout<<"Goodbye Rick"<<endl;
cout<<zombie<<endl;}
}
return 0;
}
