#include<stdio.h>
int main()
{
	int t,i,max,hash[256]={0};
	char a[10000];
	scanf("%d",&t);
	while(t--)
	{
		scanf("%s",a);
		//printf("str=%s\n",a);
		for(i=0;a[i]!='\0';i++)
			hash[a[i]]++;
		max=0;
		for(i=0;i<256;i++)
		{
			if(hash[i]>max)
				max=hash[i];
		}
		printf("%d\n",max);
		for(i=0;i<256;i++)
		{
			hash[i]=0;
		}		
	}
	
	return 0;
}
