#include <bits/stdc++.h>
using namespace std;

#define ll long long
ll exponent(ll a, ll b, ll c) {
	ll x = 1, y = a;
	while (b > 0) {
		if (b & 1) {
			x = (x * y) % c;
		}
		y = (y * y) % c;
		b = (b >> 1);
	}
	return x % c;
}

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	int n;
    	cin >> n;
    	set<int> st;
    	int x;
    	for(int i=0;i<n;i++){
    		cin >> x;
    		st.insert(x);
    	}
    	cout << exponent(n,st.size(),1000000007) << "\n";
    }
}
