#include<bits/stdc++.h>
using namespace std;

long long int t,b,w,x,y,z,sum;

int main()
{
    cin>>t;
    
    while(t--)
    { sum=0;
        cin>>b>>w;
        cin>>x>>y>>z;
        
        
        
            if(y>x && y>(x+z))
            
                sum=(b+w)*x+(w*z);
            
            else if(x>y && x>(y+z))
            
                sum=(w+b)*y+(b*z);
        
        else
        sum=(b*x)+(w*y);
        
        cout<<sum<<endl;
    }
    return 0;
}
