#include <iostream>
using namespace std;

int main()
{
    int a,b,c,d,t;
    cin>>t; while(t--) {
    	cin>>a>>b>>c;
    	if(a!=9 and b!=9 and c!=9) {
    		cout<<"0\n";
    		continue;
    	}
    	d=0;
    	if(a>0 and a<7) {
    		d+=1;
    	} else if(a>6 and a<9) {
    		d+=2;
    	} else if(a==9) {
			d+=10;
			cout<<d<<endl;
			continue;
    	}
    	if(b>0 and b<7) {
			d+=1;
    	} else if(b>6 and b<9) {
			d+=2;
    	} else if(b==9) {
			d+=7;
			cout<<d<<endl;
			continue;
    	}
    	if(c>0 and c<7) {
    		d+=1;
    	} else if(c>6 and c<9) {
    		d+=2;
    	} else if(c==9) {
    		d+=4;
    		cout<<d<<endl;
			continue;
    	}
    	cout<<d<<endl;
    }
}
