t = int(input())
c = []
for i in range(t):
    c.append(raw_input())
for i in range(t):
    count = 0
    st = c[i]
    rev = st[::-1]
    for j in range((len(st)//2)):
        while ord(st[j]) != ord(rev[j]):
            if st[j] > rev[j]:
                n = ord(st[j])-1
                st = st[:j]+chr(n)+st[j+1:]
            else:
                n = ord(rev[j])-1
                rev = rev[:j]+chr(n)+rev[j+1:]
            count += 1
    print(count)
