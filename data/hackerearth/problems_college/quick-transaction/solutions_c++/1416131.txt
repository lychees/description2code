#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
	int arr[]={1,2,5,10,20,50,100,500,1000};
	int t,x,i,count,test;
	scanf("%d",&t);
	while(t--){
		scanf("%d",&x);
		test=x;
		count=0;
		for(i=8;i>=0;i--){
			if(test>=arr[i]){
				count+=(test/arr[i]);
				test=test%arr[i];
			}
		}
		printf("%d\n",count);
	}
    return 0;
}
