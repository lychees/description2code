#include <bits/stdc++.h>
using namespace std;

int n;
long long X[1005], Y[1005];

const long long inf = 10000000000LL;

int main() {
	scanf("%d", &n);
	for(int i = 1; i <= n; i++)
		scanf("%lld %lld", &X[i], &Y[i]);
	
	int ans = 0;
	for(int i = 1; i <= n; i++) {
		long long a = -inf, b = 1; // a / b
		for(int j = i + 1; j <= n; j++) {
			long long c = Y[j] - Y[i];
			long long d = X[j] - X[i];
			if(a * d <= b * c) {
				a = c;
				b = d;
				ans++;
			}
		}
		
	}
	printf("%d\n", ans);
	
	
}
