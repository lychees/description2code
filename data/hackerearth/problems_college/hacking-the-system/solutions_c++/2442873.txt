#include<vector>
#include<iostream>
using namespace std;
int main(){
	int x;
	int y;
	cin >> x;
	cin >> y;
	vector<int>a;
	vector<int>b;
	while(x>0){
		a.push_back(x%10);
		x=x/10;
	}
	while(y>0){
		b.push_back(y%10);
		y=y/10;
	}
	if(a.size()!=b.size())cout << "loses" << endl;
	else{
		bool failed = false;
		bool tmp_failed = false;
		for(int i=0;i<a.size();++i){
			if(a[i]!=b[i]){
				if(failed==true) tmp_failed = true;
				failed = true;
			}
		}
		if(tmp_failed==true) cout <<"loses" << endl;
		else cout <<"wins" << endl;
	}
}
