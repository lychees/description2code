#include<iostream>
#define MAX 100000
using namespace std;



int Partition(unsigned int a[], int beg, int end)          
{
int p=beg, pivot=a[beg], loc;

for(loc=beg+1;loc<=end;loc++)
{
if(pivot>a[loc])
{
a[p]=a[loc];
a[loc]=a[p+1];
a[p+1]=pivot;

p=p+1;
}
}
return p;
}


void QuickSort(unsigned int a[], int beg, int end)
{
if(beg<end)
{
int p=Partition(a,beg,end);                       
QuickSort(a,beg,p-1);                             
QuickSort(a,p+1,end);			          
}
}




int main()
{
  int i,j,t,n,beg,end;
  unsigned int a[MAX];

  cin>>t;

  for(i=0;i<t;i++)
    {
      int count=0;
      cin>>n;
      for(j=1;j<=n;j++)
	{
	  cin>>a[j];
	}
      beg=1;
      end=n;
      QuickSort(a,beg,end);
      /*    for(j=1;j<=n;j++)
	cout<<" "<<a[j]<<endl;
*/
      for(j=2;j<=n;j++)
	{
	  if(a[j]==a[j-1])
	    count++;
	}
      cout<<count<<endl;
    }

  return 0;
}
