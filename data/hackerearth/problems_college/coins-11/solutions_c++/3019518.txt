#include <iostream>
#include<bits/stdc++.h>
#define ll long long
using namespace std;
bool checkprime(ll n)
{
	ll i;
	if(n==1)
	{
		return true;
	}
	for(i=2;i<=n/2;i++)
	{
		if(n%i==0)
		{
			return true;
		}
	}
	return false;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	ll n,i,cnt=1,ch=0;
    	cin>>n;
    	if(checkprime(n))
    	{
    		cout<<1<<endl;
    	}
    	else
    	{
    		loop:
    		for(i=1;i<=n/2;i++)
    		{
    		if(checkprime(n-i)&&checkprime(i))
    		{
    			ch=1;
    			cnt++;
    			break;
    		}
    		}
    		if(n!=0)
    		{
    		if(ch==0)
    		{
    			n=n/2;
    			cnt++;
    			goto loop;
    		}
    		}
    		cout<<cnt<<endl;
    	}
    }
    return 0;
}
