t=input()
while t:
	t-=1
	s1=raw_input()
	s2=raw_input()
	a=[]
	b=[]
	for i in xrange(26):
		a.append(0)
		b.append(0)
	for i in xrange(len(s1)):
		if s1[i]!=' ':
			a[ord(s1[i])-97]+=1
	
	for i in xrange(len(s2)):
		if s2[i]!=' ':
			b[ord(s2[i])-97]+=1
	cnta=0
	cntb=0
	for i in xrange(26):
		if a[i]>b[i]:
			cnta+=1
		elif b[i]>a[i]:
			cntb+=1
			
	if cntb==0:
		print "You win some."
	elif cnta==0:
		print "You lose some."
	else:
		print "You draw some."
