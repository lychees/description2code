#include <bits/stdc++.h>
using namespace std;

int S[100005], A[100005], B[100005];

int main()
{
    int N;
    scanf("%d", &N);
    
    for(int i=1; i<=N; i++)
    	scanf("%d", &S[i]);
    
    A[1] = S[1];
    
    for(int i=2; i<=N; i++)
    	A[i] = max(A[i-1], S[i]);
    
    B[N] = S[N];
    
    for(int i=N-1; i>=1; i--)
    	B[i] = max(B[i+1], S[i]);
    
    for(int i=1; i<=N; i++)
    	if(S[i]==A[i] || S[i]==B[i])
    		printf("%d ", i);
    
    return 0;
}
