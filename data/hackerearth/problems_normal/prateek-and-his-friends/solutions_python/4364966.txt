T = int(raw_input())
while T>0:
	T -= 1
	N, X = map(int, raw_input().split(" "))
	nums = []
	while N>0:
		nums.append( int(raw_input()))
		N -= 1

	_sum = 0
	start = 0
	ptr = 0
	yes = False
	length = len(nums)
	while start < length:
		if _sum == X:
			yes = True
			break
		elif _sum < X and ptr<length:
			_sum += nums[ptr]
			ptr += 1
		else:
			_sum -= nums[start]
			start += 1
		#print "start, ptr, sum", start, ptr, _sum
	if yes or _sum == X:
		print 'YES'
	else:
		print 'NO'
