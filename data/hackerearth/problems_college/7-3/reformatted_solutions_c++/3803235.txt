#include <cstdlib>
#include <iostream>
using namespace std;
struct data {
  int s;
  int c;
  int id;
  int in;
};
int comp(const void *a, const void *b) {
  data *i = (data *)a;
  data *j = (data *)b;
  if (i->s < j->s) {
    return 1;
  } else {
    if (i->s == j->s) {
      if (i->c > j->c) {
        return 1;
      } else {
        if (i->c == j->c && i->id > j->id) {
          return 1;
        } else {
          return 0;
        }
      }
    } else {
      return 0;
    }
  }
}
int comp1(const void *a, const void *b) {
  data *i = (data *)a;
  data *j = (data *)b;
  if (i->id > j->id) {
    return 1;
  } else {
    return 0;
  }
}
int main() {
  int i, j, k, n, q, t;
  cin >> n >> q;
  data arr[n];
  for (i = 0; i < n; ++i) {
    cin >> arr[i].id >> arr[i].s >> arr[i].c;
  }
  qsort(arr, n, sizeof(data), comp);
  for (i = 0; i < n; ++i) {
    arr[i].in = i + 1;
  }
  qsort(arr, n, sizeof(data), comp1);
  for (; q; --q) {
    cin >> t;
    for (i = 0, j = n - 1; i <= j;) {
      k = (i + j) / 2;
      if (arr[k].id == t) {
        cout << arr[k].in << endl;
        break;
      } else if (arr[k].id > t) {
        j = k - 1;
      } else {
        i = k + 1;
      }
    }
  }
  return 0;
}
