#include <iostream>
#include <map>
#include <algorithm>
#include <vector>
using namespace std;

int main()
{
    int t;
    cin>>t;
    map <string,int> m;
    while(t--){
    	string s;
    	cin>>s;
    	m[s]++;
    }
    
    vector<pair<int,string>> v;
  	map <string,int> :: iterator mit;
  	for(mit = m.begin();mit!=m.end();mit++){
  		v.push_back(make_pair(mit->second,mit->first));
  	}
    
    vector <pair<int,string>> :: reverse_iterator vit;
    sort(v.begin(),v.end());
    for(vit=v.rbegin();vit!=v.rend();vit++){
    	cout<<vit->second<<" "<<vit->first<<endl;
    }
    return 0;
}
