#include <iostream>
#include <map>
#include <cstring>
#include <climits>
using namespace std;

int main()
{
    int i, N, big=INT_MIN;
    string playerWin, playerLost, winner;
    map<string, int> players;
    cin>>N;
    for(i=1; i<(1<<N); i++)
    {
    	cin>>playerWin>>playerLost;
    	players[playerWin]++;
    	players[playerLost]--;
    	if(big<players[playerWin])
    	{
    		big = players[playerWin];
    		winner = playerWin;
    	}
    }
    cout<<winner;
    return 0;
}
