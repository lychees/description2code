#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    for(int i=0;i<n;i++)
    {
    	
    	int m;
    	cin>>m;
    	int *a=new int[m];
    


    	for(int j=0;j<m;j++)
    	{
			cin>>a[j];

    	}
    	
    	int count=0;
  for(int k=0;k<m;k++)
  {
  	for(int k1=k;k1<m;k1++)
  	{
  		if(a[k1]==a[k])
  		{
  			count++;
  			
  			
  		}
  		
  		
  	}
  	
  	
 }
 
 cout<<count<<endl;
    	
    	
    	
    	
    	
    	
    	
    	
    	
    	
    }
    return 0;
}
