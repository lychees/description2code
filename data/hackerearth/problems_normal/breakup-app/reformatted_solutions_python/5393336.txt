n = input()
days = [0] * 31
for i in range(n):
    msg = raw_input()
    if msg.startswith('G:'):
        weight = 2
    else:
        weight = 1
    msg = msg.split()
    for j in msg:
        try:
            j = int(j)
            days[j] += weight
        except BaseException:
            pass
maxweight = 0
dday = 0
count = 0
for i in range(31):
    if days[i] == maxweight:
        count += 1

    if days[i] > maxweight:
        dday = i
        maxweight = days[i]
        count = 0

# print count , days
if count > 0:
    print "No Date"
elif dday == 19 or dday == 20:
    print "Date"
else:
    print "No Date"
