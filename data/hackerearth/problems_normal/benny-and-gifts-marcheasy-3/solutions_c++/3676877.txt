#include <iostream>
#include <set>
#include <stdio.h>
#include <string.h>
using namespace std;


#define mp make_pair
const int N = 100500;
int n;
int x, y;
char s[N];

set< pair<int, int> > was;
int main() {
    gets(s+1);
    n = (int)strlen(s+1) + 1;
    
    x = y = 0;
    int result = 0;
    for (int i = 1; i <= n; i++) {
        
        if (was.find(mp(x, y)) != was.end()) {
            ++result;
        }
        was.insert(mp(x, y));
        if (s[i] == 'L') {
            --y;
        } else if (s[i] == 'R') {
            ++y;
        } else if (s[i] == 'U') {
            --x;
        } else if (s[i] == 'D') {
            ++x;
        }
    }
    printf("%d\n", result);
    return 0;
}
