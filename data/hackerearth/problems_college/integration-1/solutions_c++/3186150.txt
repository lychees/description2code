#include <cstdio>
#include <cmath>
using namespace std;


int main()
{
	int T;
	scanf("%d", &T);
	for(int i = 1; i <= T; i++)
	{
		long long N, P, U, R, S;
		scanf("%lld %lld %lld %lld %lld", &N, &P, &U, &R, &S);
		for(int i = 0; i < N; ++i)
		{
			if(P + R <= U) { P += R; }
			else if(P - S >= 0) { P -= S; }
			else { break; }
		}
		printf("Case #%d: %lld\n", i, P);
	}
	return 0;
}
