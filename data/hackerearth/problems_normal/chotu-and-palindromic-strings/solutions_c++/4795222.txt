#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s1,s2,s3,s4;
    	int flag=0;
    	string r1,r2;
    	cin>>s1;
    	cin>>s2;
    	s3=s1+s2;
    	r1=s3;
    	reverse(s3.begin(),s3.end());
    	
    	if(s3==r1)
    	flag=1;
    	s4=s2+s1;
    	r2=s4;
    	
    	reverse(s4.begin(),s4.end());
    	
    	if(s4==r2)
    	flag=1;
    	if(flag==1)
    	cout<<"YES"<<endl;
    	else
    	cout<<"NO"<<endl;
    }

	return 0;
}
