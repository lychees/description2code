#include <bits/stdc++.h>
using namespace std;
#define f first
#define s second
#define rep(i,N) for(int i=0;i<N;i++)
#define ll long long int
int main()
{
    int t,n,k;
    cin>>t;
    while(t--){
    	cin>>n>>k;
    	pair<ll,ll>p[n];
    	ll cu[n],dp[n];
    	memset(cu,0,sizeof cu);
    	rep(i,n) cin>>p[i].f;
    	rep(i,n) cin>>p[i].s;
    	sort(p,p+n);
    	rep(i,n) (i==0)?cu[i]=p[i].s:cu[i]+=cu[i-1]+p[i].s;
    	for(int i=0;i<n;i++){
    		dp[i]=(k+cu[i])*p[i].f;
    		for(int j=0;j<i;j++){
    			dp[i]=min(dp[i],dp[j]+(k+cu[i]-cu[j])*p[i].f);
    		}
    	}
    	cout<<dp[n-1]<<endl;
    }
    return 0;
}
