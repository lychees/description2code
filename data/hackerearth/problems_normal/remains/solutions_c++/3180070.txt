#include <bits/stdc++.h>
using namespace std;

void solve()
{
	long long x, y; int n;
	scanf("%lld%lld%d", &x, &y, &n);
	long long ans = x+y;
	int i = 2;
	for (;i < n;i++)
	{
		long long z = abs(y-x);
		ans += z;
		x = y; y = z;
		if (y == 0) break;
	}
	int rem = n-i;
	ans += rem/3*2*x;
	if (rem%3 == 2) ans += x;
	printf("%lld\n", ans);
}

int main()
{
	int t; scanf("%d", &t);
	while (t--)
		solve();
	return 0;
}
