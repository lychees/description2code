#include <bits/stdc++.h>
using namespace std;

int cnt = 0;
bool a[30][30];


int main() {
    int tc;
    cin >> tc;

    while(tc--) {
        int n,m,last_0 = -1;
        cin >> n >> m;
        cnt = 0;
        for(int i = 0; i < n; i++){
            string s;
            cin >> s;
            for(int j = 0; j < m; j++){
                a[i][j] = (int)s[j] - 48;
                if(a[i][j] == 0){
                    last_0 = (i*10) + j;

                }
            }
        }

        for(int i = n-1; i >= 0; i--){
            for(int j = m - 1; j >= 0; j--){
                if(a[i][j] == 0){
                    cnt++;
                    //cout << i << " ---- " << j<<endl;
                    
                    for(int k = 0; k <= i; k++){
                        for(int l = 0; l <= j; l++){
                            a[k][l] = 1 - a[k][l];
                        }
                    }
                }
                //cout << a[i][j];
            }
            //cout << endl;
        }
       cout << cnt << endl;
    }
}
