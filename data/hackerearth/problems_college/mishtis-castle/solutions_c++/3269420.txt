#include <iostream>
using namespace std;

int main()
{
    int T;
    cin>>T;
    while(T!=0){
    	int n,c;
    	cin>>n>>c;
    	int A[n][n];
    	for(int i=0;i<n;i++){
    		for(int j=0;j<n;j++) {
    			cin>>A[i][j];
    		}
    	}
    	
    	for(int i=0;i<n;i++){
    		if(i==0){
    		for(int j=1;j<n;j++) {
    			A[i][j]=A[i][j]+A[i][j-1];
    		}
    		}
    		else {
    			A[i][0]=A[i][0]+A[i-1][0];
    			for(int j=1;j<n;j++) {
    				if(A [i-1][j]<A[i][j-1]) A[i][j]=A[i][j]+A[i][j-1];
    				else A[i][j]=A[i][j]+A[i-1][j];
    			}
    		}
    		
    	}
    	if(A[n-1][n-1]-c>=0)
    	cout<<A[n-1][n-1]-c<<endl;
    	else cout<<-1<<endl;
    	T--;
    }
    return 0;
}
