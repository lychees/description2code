#include<iostream>
#include<algorithm>
using namespace std;
int main(){
    
        int n;
        cin >> n;
       int a[n+1];
         for(int i = 0 ; i < n ;++i )
          cin >> a[i];
          int ans = 1;
          int mx = 1;
          for(int i = 1 ; i < n ; ++i ){
          
             if(abs(a[i]-a[i-1])==1)
               ans++;
               else {
                mx = max(mx,ans);
             ans = 1;
                }
                mx = max(mx,ans);
          }       
          cout << mx << endl;
         
  

}
