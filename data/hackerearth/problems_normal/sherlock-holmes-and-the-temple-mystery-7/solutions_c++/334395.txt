using namespace std;
#include<iostream>
int ar[100];
char kr[100][5000];
char recur(int,int);
int n;
int main()
{
    int t;
    cin>>t;
    while(t>0)
    {

        cin>>n;
        int i=0;
        int sum=0;
        while(i<n)
        {
            cin>>ar[i];
            sum=sum+ar[i];
            i++;
        }

        i=0;
        while(i<n)
        {
           int j=0;
           while(j<=5000)
           {
              kr[i][j]='2';
              j++;
           }
           i++;
        }
        if(sum%2!=0)
            cout<<"NO"<<endl;
        else{
        if(recur(sum/2,0)=='1')
            cout<<"YES"<<endl;
        else cout<<"NO"<<endl;
        }
        t--;
    }
}
char recur(int c,int i)
{
    if(i==n)
    {
        if(c==0)
            return '1';
        else return '0';
    }
    if(kr[i][c]=='2')
    {
        if(recur(c-ar[i],i+1)=='1'||recur(c,i+1)=='1')
            kr[i][c]='1';
        else kr[i][c]='0';
    }
    return kr[i][c];
}
