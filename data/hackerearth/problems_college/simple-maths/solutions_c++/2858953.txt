#include <iostream>
#include<stdio.h>
using namespace std;

long long gcd(long long a,long long b)
{
	if(b%a==0)
	return a;
	return gcd(b%a,a);
}

int main()
{
	
	long long a,b,count,n,t,i,x;
	
	scanf("%lld",&t);
	
	while(t--)
	{
		count=0;
		
		scanf("%lld%lld%lld",&n,&a,&b);
		
		
		if(a==0 || b==0)
		count=-1;
		
		else
		{
		count+=n/a;
		count+=n/b;
		
		x=a*b/gcd(a,b);
		
		count-=n/x;
		}
		
		
		
		
		printf("%lld\n",count);
		
	}
	
	
	
	
    //cout << "Hello World!" << endl;
    return 0;
}
