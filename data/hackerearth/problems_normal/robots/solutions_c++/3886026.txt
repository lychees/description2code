#include<iostream>
#include<algorithm>
using namespace std;


void Test()
{
	int N,i;
	long long int curWays;
	long long int Modulus = 1000000007LL;
	int MAX[1000];
	
	cin >> N;
	for(i = 0; i < N; i++)
		cin >> MAX[i];
	curWays = 1;
	sort(MAX, MAX+N);
	for(i = 0; i < N; i++)
	{
		if(i >= MAX[i])
		{
			curWays = 0;
			break;
		}
		
		curWays = curWays * (MAX[i] - i);
		
		if(curWays > Modulus)
		{
			
			while(curWays >= Modulus)
				curWays = curWays - Modulus;
		}
	}
	cout << curWays;
	cout << "\n";
	return;
}

int main()
{
	int T,i;
	cin >> T;
	for(i = 0; i < T; i++)
	{
		Test();
	}
	return 0;
}
