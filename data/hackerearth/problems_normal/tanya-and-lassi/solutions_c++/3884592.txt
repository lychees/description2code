#include <iostream>
#include <bits/stdc++.h>
using namespace std;
long long dp[10003];
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int l;
    	scanf("%d",&l);
    	int arr[l+1];
    	memset(dp,0,sizeof(dp));
    	for(int i=1;i<=l;i++)
    	{
    		scanf("%d",&arr[i]);
    	}
    	for(int i=1;i<=l;i++)
    	{
    		for(int j=i;j<=l;j++)
    		{
    			if(j-i>=0)
    			dp[j]=max(dp[j],arr[i]+dp[j-i]);
    		}
    	}
    	cout<<dp[l]<<endl;
    }
    return 0;
}
