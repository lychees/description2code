#include <iostream>
#include <queue>
using namespace std;

int main()
{
    int numOfSeats,numOfPeople,seatNo,maxSum=0;
    cin>>numOfSeats>>numOfPeople;
    priority_queue<int>pq;
    while(numOfSeats--)
    {
    	cin>>seatNo;
    	pq.push(seatNo);
    }
    while(numOfPeople--)
    {
    	int pq_top = pq.top();
    	maxSum += pq_top;
    	pq.pop();
    	pq.push(pq_top-1);
    }
    cout<<maxSum<<endl;
    return 0;
}
