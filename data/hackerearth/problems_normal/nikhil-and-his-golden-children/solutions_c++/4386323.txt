#include <iostream>
#include<bits/stdc++.h>
using namespace std;
#define MAX 1000000007

long long power(long long x, long long n){
	if(n==0)
	 return 1;
	else if(n==1)
	 return x;
	long long mid=power(x,n/2);
	
	if(n%2==0)
	 return (mid*mid)%MAX;
	else 
	 return (((mid*mid)%MAX)*x)%MAX;
}

long long fact(long long n){
	if(n==0||n==1){
		return 1;
	}
	else return (n*fact(n-1))%MAX;
}

long long combination(long long n, long long r){
	return ((fact(n)%MAX)*power(((fact(r)%MAX)*(fact(n-r)%MAX) )%MAX,MAX-2))%MAX;
}
int main()
{
	int tests;
	scanf("%d",&tests);
	long long children,tablets,min_total=0,count=0,min,n,r;
	while(tests--){
		min_total = 0;
		scanf("%lld %lld",&children,&tablets);
		count=children;
		for(int i=0;i<children;i++){
			scanf("%lld",&min);
			min_total+= min;
		}
		if(tablets<min_total){
			printf("%d\n",0);
		}else {
			n = tablets-min_total;
			r = children-1;
			//cout<<nCrModpLucas(n,r,1000000007)<<endl;
			printf("%lld\n",combination(n+r,r));
		}
		
	}
    return 0;
}
