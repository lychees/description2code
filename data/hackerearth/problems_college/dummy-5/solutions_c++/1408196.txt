#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <iostream>
#include <algorithm>
#include <string>
#include <cstring>
using namespace std;
int a[25],b[25],c[25],T,A,B,C,_,n,i;
bool FLAG;
void dfs(int x,int A,int B,int C)
{
    if (A==0 && B==0 && C==0) FLAG=true;
    if (x==n+1) return;
    dfs(x+1,A,B,C);
    dfs(x+1,A-a[x],B-b[x],C-c[x]);
}
int main()
{
    scanf("%d",&T);
    for (_=1; _<=T; _++)
    {
        scanf("%d%d%d",&A,&B,&C);
        scanf("%d",&n);
        for (i=1; i<=n; i++)
            scanf("%d%d%d",&a[i],&b[i],&c[i]);
        FLAG=false;
        dfs(1,A,B,C);
        printf("Case #%d: ",_);
        if (FLAG) printf("yes\n"); else printf("no\n");
    }
    return 0;
}
