#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    long long n,sum1=0,sum2=0,i,j;
  cin>>n;
  long long ar[n][n];
  for(i=0;i<n;i++)
  {
  	for(j=0;j<n;j++)
  	cin>>ar[i][j];
  }
  for(i=0;i<n;i++)
  {
  	for(j=0;j<n;j++)
  	{
  		if(i==j)
  	sum1+=ar[i][j];
  	if(i+j==n-1){
  		sum2+=ar[i][j];
  	}
  	}
  }
  cout<<abs(sum1-sum2);
}
