#include<iostream>
#include<cstdio>
#include<vector>
using namespace std;
vector<long long> f;
void precomp()
{
	f.push_back(1);
	f.push_back(1);
	int i=2;
	while(1)
	{
		long long temp=f[i-1]+f[i-2];
		if(temp>10000000000000000ll)
			break;
		f.push_back(temp);
		i++;
	}
}
int main()
{
	precomp();
	int t;
	scanf("%d",&t);
	long long n;
	while(t--)
	{
		scanf("%lld",&n);
		for(int i=1;i<f.size();i++)
			if(n>=f[i])
				n++;
		printf("%lld\n",n);
	}
	return 0;
}
