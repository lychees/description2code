a = input()
for i in xrange(a):
    c = raw_input()
    d = [0 for i in xrange(26)]
    for j in xrange(len(c)):
        k = ord(c[j:j+1])-97
        d[k] = d[k] + 1

    doesExist = False
    for i in d:
        if (i > 1):
            doesExist = True
    if (doesExist):
        print "YES"
    else:
        print "NO"
