#include <iostream>
using namespace std;

#define MAX 1001

int N, answer;
char s[MAX];
bool ot[MAX];

int main()
{
 //   freopen("Passing the Parcel.txt", "r", stdin);
	// setbuf(stdout, NULL);

	cin >> N;
	cin >> s;
	int length = 0;
	while(s[length] != '\0')
		length++;
    int num = 0, cur = 1;
	for(int i = 1; i <= N; i++)
		ot[i] = true;

	while(num < N)
	{
		for(int i = 0; i < length; i++)
		{
			if(cur > N)
				cur = 1;
			if(ot[cur] == false)
				while(ot[cur] == false)
				{
					cur++;
					if(cur > N)
						cur = 1;
				}
			if(s[i] == 'b')
			{
				ot[cur] = false;
				num++;
				answer = cur;
				if(num == N)
					break;
			}
			cur++;
		}
	}
	cout << answer << endl;
    return 0;
}
