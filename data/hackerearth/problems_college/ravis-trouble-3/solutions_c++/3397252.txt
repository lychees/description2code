#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n;
    cin >> n;
    string a,b;

    while(n--)
    {
        cin >> a >> b;
        int k = 0,count = 0,i = 0;

        for (int k = 0;k < a.size();k++)
        {
            if (b[i] == a[k])
            {
                i++;
                //k++;
                count++;
            }
        }

        if (count == b.size())
            cout << "YES" << endl;
        else
            cout << "NO" << endl;
    }

    return 0;
}
