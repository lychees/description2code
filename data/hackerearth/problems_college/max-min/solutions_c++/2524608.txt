#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n,k;
    scanf("%d %d",&n,&k);
    int a[n+4];
    for(int i=0;i<n;i++){
        scanf("%d",&a[i]);
    }
    sort(a,a+n);
    reverse(a,a+n);
    int min=INT_MAX,dif;
    for(int i=0;i<n-k+1;i++){
        dif=a[i]-a[i+k-1];
        if(dif<min)min=dif;
    }
    printf("%d\n",min);
    return 0;
}
