#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,i,n;
    bool visited[27];
    char arr[50];
    cin>>t;
    while(t--)
    {
    	for(i=0;i<27;i++)
    	visited[i]=false;
     scanf("%s",arr);
     n=strlen(arr);
     for(i=0;i<n;i++)
     {
     	if(!visited[arr[i]-'a'])
     	{
     	printf("%c",arr[i]);
     	visited[arr[i]-'a']=true;
     	}
     }
     printf("\n");
    }
    return 0;
}
