#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    int a[100];
    int i;
    int m = 0;

    cin>>n;
    for(i = 0; i < n; i++) {
        cin>>a[i];
        if(a[i] > m) {
            m = a[i];
        }
    }
    int ans = m - n;
    if(ans % 2 == 0) {
        cout<<"Bob";
    }
    else {
        cout<<"Alice";
    }
}
