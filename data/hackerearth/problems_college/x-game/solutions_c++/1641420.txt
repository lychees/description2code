LL modpow(LL a, LL b)
{
	LL x = 1, y = a;
	while(b) {
		if(b & 1) {
			x = x * y;
			if(x >= mod) {
				x %= mod;
			}
		}
		y = y * y;
		if(y >= mod) {
			 y %= mod;
		}
		b >>= 1;
	}
	return x;
}
