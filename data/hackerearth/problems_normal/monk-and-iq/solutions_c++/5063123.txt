#include <bits/stdc++.h>
 
using namespace std;
 
priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
long long second[1000001];
long long size[1000001]; 
 
int main() {
	int C, P, N;
	cin >> C >> P >> N;
 
	for (int i = 1; i <= N; i++) {
		long long x;
		cin >> x;
		pq.push(make_pair(x, i));
		second[i] = 0;
		size[i] = 1;
	}
	for (int i = N+1; i <= C; i++) {
		pq.push(make_pair(0, i));
		second[i] = 0;
		size[i] = 0;
	}
	
	for (int i = 1; i <= P; i++) {
		long long x;
		if (!(cin >> x))
			x = 0;
			
		pair<long long, int> p = pq.top();
		pq.pop();
		
		cout << p.second << " ";
		if (size[p.second] > 0)
			second[p.second] = p.first/size[p.second] - second[p.second];
		size[p.second]++;
		pq.push({(second[p.second]+x) * size[p.second], p.second});
	}
    
    return 0;
}
