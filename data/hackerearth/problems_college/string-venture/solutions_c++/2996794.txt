#include<bits/stdc++.h>
#include<iostream>
#include<string.h>
using namespace std;

int main()
{
	string str;
	getline(cin, str);
	long long sum = 0;
	int map[256];
	int l = str.length();
	for(int i = 0; i < 256; i++)
		map[i] = 0;
	for(int i = 0; i < l; i++){
		sum += str[i];
		map[str[i]]++;
	}
	int w = floor(sum/l);
	if(w % 2 == 0){
		for(int i = l-1; i >= 0; i--)
			cout << str[i];
	}else{
		int m = 0, index;
		for(int i =0; i < l; i++){
			if(map[str[i]] > m){
				m = map[str[i]];
				index = i;
			}
		}
		cout << str[index];
	}
	cout << endl;
	return 0;
}
