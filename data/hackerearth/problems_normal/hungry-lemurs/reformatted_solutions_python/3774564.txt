n, k = map(int, raw_input().split())


def diff(x, k, n):

    ans = n-k
    i = 2
    while i*i <= x:
        if x % i == 0:
            ans = min(ans, abs(n-x)+abs(k-i))
            z = x/i
            if x % z == 0:
                ans = min(ans, abs(n-x)+abs(k-z))
        i += 1

    return ans
