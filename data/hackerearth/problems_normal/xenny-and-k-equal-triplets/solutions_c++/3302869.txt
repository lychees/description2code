#include <iostream>

using namespace std;
ll a[1000007];
ll gcd(ll x,ll y)
{
	return (y==0?x:gcd(y,x%y));
}
ll compute(ll x)
{
	ll res;
	res=x*(x-1);
	res/=2;
	res=res*(x-2);
	res=res/3;
	return res;
}
int main()
{
	ll t,i,n,c,ans,tot,val,k;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld %lld",&n,&k);
		for(i=0;i<n;i++)
		scanf("%lld",&a[i]);
		sort(a,a+n);
		c=1;
		ans=0;
		for(i=1;i<n;i++)
		{
			if(a[i]==a[i-1] && a[i]==k)
			{
				c++;
			}
			else
			{
				if(c>=3)
				ans+=compute(c);
				c=1;
			}
		}
		if(c>=3)
		ans+=compute(c);
		tot=compute(n);
		val=gcd(tot,ans);
		printf("%lld/%lld\n",ans/val,tot/val);
	}
	return 0;
}
