#include"iostream"
#include"algorithm"
#include"queue"
using namespace std;

struct compare
{
	private :
		int *m;	
	public :
		compare(int *a)
		{
				m=a;
		}
	bool operator()(const int &i,const int &j)
	{
		return m[i]<m[j];
	}
};
int main()
{
	int n,t;
	cin>>t;
	while(t--)
	{
	
	cin>>n;
	int a[n];
	int b[n];
	int pos[n];
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
		pos[i]=i;
	}
	for(int i=0;i<n;i++)
	{
		cin>>b[i];
	}
	sort(pos,pos+n,compare(a));
	priority_queue <int, vector<int>, greater<int> > q;
	q.push(a[pos[0]]+b[pos[0]]);
	int count=1;
	int k=1;
	while(k<n)
	{
		if(a[pos[k]]<q.top() || q.empty())
		{
			q.push(a[pos[k]]+b[pos[k]]);
			count++;
			k++;
		}		
		else
		{
			q.pop();
			q.push(a[pos[k]]+b[pos[k]]);
			k++;	
			
		}
		
	}
	cout<<count<<endl;
	
	}
	return 0;
}
