//second solution without sets
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		int flag[1000005]={0};
		int m=0,temp=0;
		for(int i=0;i<n;i++)
		{
			int a,b;
			scanf("%d %d",&a,&b);
			if(b==0)
			{
				m=max(m,temp);
				temp=0;
			}
			else if(b==1)
			{
				if(flag[a])
				continue;
				else
				{
					temp++;
					flag[a]=1;
						m=max(m,temp);
				}
			}
		}
		cout<<m<<endl;
	}
	return 0;
}
