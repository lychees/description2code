#include <iostream>
#include <cmath>
#define aa long long
using namespace std;
int main()
{
    ios_base::sync_with_stdio(false);
    aa int tc;
    cin>>tc;
    while(tc--)
    {
        aa int one=0,zero=0;
        string s;
        cin>>s;
        for(int i=0;i<s.length();i++)
        {
            if(s[i] == '1')
                one++;
            else zero++;
        }
        cout<<abs(one-zero)<<endl;
    }
}
