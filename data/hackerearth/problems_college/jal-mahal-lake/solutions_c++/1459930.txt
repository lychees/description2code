#include <iostream>
#include<cstdio>
using namespace std;

long long *planks;
long long *length;

long long energy;
long long diff;
int T;
int N;
int main()
{
	cin>>T;
	while(T--){
		energy=0;
		scanf("%d",&N);
		planks = new long long[N];
		length = new long long[N];
		
		for(int i=0;i<N;i++){
			scanf("%lld",&planks[i]);
		}
		for(int i=0;i<N;i++){
			scanf("%lld",&length[i]);
			length[i]+= planks[i];
		}
		
		// Sort the planks according to its position
		
		for(int i=0;i<N;i++){
			for(int j=0;j<N-1;j++){
				if(planks[j]>planks[j+1]){
					long long tempP,tempL;
					swap(planks[j],planks[j+1]);
					swap(length[j],length[j+1]);
				}
			}
		}

		int i;
		for(i=0;i<N/2;i++){
			diff = planks[i+1] - length[i];
			if(diff>0){
				for(int j=0;j<=i;j++){
					planks[j]+= diff;
					length[j]+= diff;
					energy+= diff;		
				}
			}
		}
		for(i=0;i<N-1;i++){
			diff = planks[i+1] - length[i];
			if(diff>0){
					planks[i+1] -= diff;
					length[i+1] -= diff;
					energy += diff;		
				}
			}
			cout<<energy<<endl;
		}

    return 0;
}
