#include <iostream>
#include <stdlib.h>
using namespace std;

int main()
{
	int N,i;
	long long int maxpos=0,maxneg=0,secmaxpos=0,secmaxneg=0;
	cin >> N;
	//cout << N;
	long long int A;
	for(i=0;i<N;i++){
		cin >> A;
		if(A>secmaxpos){
			secmaxpos = A;
			if(A>=maxpos){
			secmaxpos=maxpos;
			maxpos = A;
			}
			//cout << "maxpos " <<maxpos << " i " << i << endl;
		}
		else if(A<secmaxneg){
			secmaxneg = A;
			if(A<=maxneg){
			secmaxneg=maxneg;
			maxneg = A;
			}
			//cout << "maxneg " <<maxneg << " i " << i << endl;
		}
	}
	//	cout << "i " << i;
	//cout << maxneg << " ";
	//cout << secmaxneg << " ";
	//cout << maxpos << " ";
	//cout << secmaxpos << " " << endl;
	if((maxneg*secmaxneg)>(maxpos*secmaxpos)){
		cout << maxneg*secmaxneg << endl;
	}
	else
		cout << maxpos*secmaxpos << endl;
    return 0;
}
