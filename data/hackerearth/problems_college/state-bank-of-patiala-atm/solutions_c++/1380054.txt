#include<bits/stdc++.h>
using namespace std;
int main()
{
    int tc,i,n;
    size_t found;
    cin>>tc;
    vector<string> pin;
    string p;
    while(tc--)
    {
        cin>>n;
        pin.clear();
        for(i=0;i<n;i++)
        {
            cin>>p;
            pin.push_back(p);
        }
        sort(pin.begin(),pin.end());
        p=pin[0];
        for(i=1;i<pin.size();i++)
        {
            found=pin[i].find(p);
            if(found==0)
            {
                break;
            }
            p=pin[i];
        }
        if(found==0)
        cout<<"NO"<<endl;
        else
        cout<<"YES"<<endl;
    }
    return 0;
}
