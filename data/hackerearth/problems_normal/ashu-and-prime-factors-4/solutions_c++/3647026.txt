#include <stdio.h>
#define max 1000001

int a[max] = {0};
int ans[max];

void sieve ()
{
	int i;
	for (i = 0; i < max ; i++)
		ans[i] = 1;
	for (i = 2 ; i*i <= max ; i++)
	{
		if (a[i] == 0)
		{
			int j;
			for (j = i*2 ; j < max ; j += i)
			{
				if (a[j] == 0)
				{
					a[j] = i;
					ans[i] += 1;
				}
			}
		}
	}
}

int main ()
{
	sieve ();
	int t;
	scanf ("%d",&t);
	while (t--)
	{
		int n;
		scanf ("%d",&n);
		printf ("%d\n",ans[n]);
	}

	return 0;
}
