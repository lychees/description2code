#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s; int t;
    cin>>t;
    int i ,l;
    while(t--)
    {
    	cin>>s; int flag=0;
    l=s.length();
    for(i=0;i<l-1;i++)
    {

    	int a=s[i+1]>s[i]?s[i+1]-s[i]:s[i]-s[i+1];
    	int b=s[l-i-1]>s[l-i-2]?s[l-i-1]-s[l-i-2]:s[l-i-2]-s[l-i-1];
    	if(a!=b) {flag=1; break;}
    }
    if(!flag) printf("Funny\n");
    else printf("Not Funny\n");
    }
    return 0;
}
