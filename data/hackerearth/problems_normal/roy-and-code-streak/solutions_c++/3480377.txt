#include <bits/stdc++.h>

using namespace std;

int flag[1000001];
int s[1000001];
int r[1000001];

int count_streak(int n)
{
	int count = 0;
	int max_count = -1;
	memset(flag, 0, sizeof(flag));
	for (int i = 0; i < n; i++)
	{
		if (r[i] == 1 && flag[s[i]] == 0)
		{
			flag[s[i]] = 1;
			count++;
		}
		else if (r[i] == 0)
		{	
			max_count = max(max_count, count);
			count = 0;
		}
	}
	max_count = max(max_count, count);
	return max_count;
}

int main()
{
	int t, n, streak;
	cin >> t;
	while (t--)
	{
		cin >> n;
		for (int i = 0; i < n; i++)
			cin >> s[i] >> r[i];
		streak = count_streak(n);
		cout << streak << endl;
	}
	return 0;
}
