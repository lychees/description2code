#include<iostream>
using namespace std ;
int main(){
	int t ;
	cin >> t ;
	while( t-- ){
	         int n ;
		 cin >> n ;
		int carry = 0;
		long long count ;
		long long  a[1000005] ;
		int i , j , temp ;
		int len = 1 ;
		count = 0 ;
		//cin >> n ;
		a[0] = 1 ;
		for( i  = 2 ; i <= n ; i++ ){
			for( j  = 0 ; j < len ; j++ ){
				temp = carry + a[j] * i ;
				carry = temp / 10 ;
				a[j] = temp % 10 ;
			}
			while(carry != 0 ){
				a[j] = carry % 10 ;
				 j++ ;
				 carry = carry/ 10 ;
			}
			len = j ;
		}
		for ( i = len - 1 ; i >= 0 ; i-- ){
			//cout<<a[i] ;
			if(( a[i] == 7 ) || ( a[i] == 4 ))
			{
				count++ ;
			}
		}
		cout<<count<<"\n" ;
	}
	return 0 ;
}
