#include <iostream>
using namespace std;

int main()
{
    int T;
    cin>>T;
    for(int i=0;i<T;i++)
    {
    	string s;
    	cin>>s;
    	for(int j=1;j<s.length();j++)
    	{
    		if(((int)s[j]==111&&(int)s[j-1]==120)||((int)s[j]==111&&(int)s[j-1]==0))
    		s[j]=0;
    	}
    	for(int j=s.length()-2;j>=0;j--)
    	{
    		if(((int)s[j]==111&&(int)s[j+1]==120)||((int)s[j]==111&&(int)s[j+1]==0))
    		s[j]=0;
    	}
    	for(int j=0;j<s.length();j++)
    	{
    		if(s[j]!=NULL)
    		cout<<s[j];
    	}
    	cout<<endl;
    }
    return 0;
}
