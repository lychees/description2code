//Header files used...
#include<bits/stdc++.h>

using namespace std;
int main()
{
	int t,n;
	long int i;
	scanf("%d",&t);
	while(t--)
	{
		char b;
		long int odd=0,even=0;
		scanf("%d",&n);
		for(i=0;i<2*n+1;i++)
		{
		
			scanf("%c",&b);
			//cout<<i<<" "<<b<<"\n";
			if(b=='B' && i%2==1)
				{//cout<<i<<"odd ";
				odd++;}
			if(b=='B' && i%2==0)
			{//cout<<i<<"even ";
				even++;	}	
		}
	//	cout<<odd<<" "<<even<<"\n";
	//	std::cout<<min(odd,even)<<"\n";
		printf("%ld \n",min(odd,even));
	}
}
