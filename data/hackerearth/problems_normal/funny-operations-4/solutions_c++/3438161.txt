#include <bits/stdc++.h>
using namespace std;

int main()
{
   int t;
   cin>>t;
   
   while( t-- ){
   		unsigned int n;
   		cin>>n;
   		vector<unsigned int> vect( n );
   		unsigned int len = 0, p = 0;
   		
   		for( unsigned int i = 0 ; i < n; i++ )	cin>>vect[i];
   		
   		for( unsigned int i = 1; i < ( 1 << n ); i++ ){
 			unsigned int a = 0, b = ~0u, cnt = 0;
 			
 			for( unsigned int j = 0; j <= n; j++ )
 				if( ( 1 << j ) & i ){
 					cnt++;
 					a = a ^ vect[j];
 					b = b & vect[j];
 				}
 			
 			
 			if( ( a | b ) > p ){
 				p = a | b;
 				len = cnt;
 			} else if( ( a | b ) == p )	len = max( len, cnt );
   		}
   		cout << p << " " << len << endl;
   }
   return 0;
}
