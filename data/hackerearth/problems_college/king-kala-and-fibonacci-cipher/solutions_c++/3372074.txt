#include<bits/stdc++.h>
using namespace std;

int f[100005][26],a[100005];

void init(){
    a[0]=0;a[1]=1;
    f[0][0]=1;f[1][1]=1;f[1][0]=1;
    for(int i=2;i<100005;i++){
        a[i]=(a[i-1]+a[i-2])%26;
        for(int j=0;j<26;j++) f[i][j]=f[i-1][j];
        f[i][a[i]]++;
    }
}

int cmp(pair<int,int>a,pair<int,int>b){
    if(a.second!=b.second) return a.second > b.second;
    else{
        if(a.first < b.first) return 1;
        else return 0;
    }
}

int main()
{
    init();
    int t;
    cin >> t;
    while(t--){
        int l,r;
        string s;
        vector<pair<int,int> >v;
        cin >> l >> r;
        for(int i=0;i<26;i++){
            if(l==0) v.push_back(make_pair(i,f[r][i]));
            else v.push_back(make_pair(i,f[r][i]-f[l-1][i]));
        }
        sort(v.begin(),v.end(),cmp);

        cin >> s;
        for(int i=0;i<s.length();i++){
            printf("%c",'a'+v[s[i]-'a'].first);
        }
        cout << endl;
    }
    return 0;
}
