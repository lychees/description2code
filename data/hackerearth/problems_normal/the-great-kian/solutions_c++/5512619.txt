#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++){
    	cin>>a[i];
    }
    if(n>=3){
    	for(int i=0;i<3;i++){
    		long int s=a[i];
    		for(int j=i+3;j<n;j+=3){
    			s+=a[j];
    		}
    		cout<<s<<" ";
    	}
    }
    else{
    	for(int i=0;i<n;i++){
    		cout<<a[i]<<" ";
    	}
    	for(int i=0;i<3-n;i++){
    		cout<<0<<" ";
    	}
    }
    
    return 0;
}
