#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t=0,i=0;
    set<string> c;
    set<string>::iterator j;
    vector<string> v;
    scanf("%d",&t);
    for(i=0;i<t;i++)
    {
    	string s;
    	cin>>s;
    	c.insert(s);
    	v.push_back(s);
    }
    for(j=c.begin();j!=c.end();j++)
    {
    	cout<<*j<<" "<<count(v.begin(),v.end(),*j)<<endl;
    }
    
    return 0;
}
