#include <iostream>
using namespace std;

int main()
{
    int a[26];
    for(int i = 0; i < 26; i++){
    	a[i] = i+1;
    }
    int t;
	cin>>t;
	for(int u = 0; u < t; u++){
		string s;
		int k;
		cin>>k;
		cin>>s;
		int l = s.size();
	   	int sum = 0;
	   	int start = 0;
	   	int end;
	   	int i = -1,count = 0;
	   	while(i < l){
	   	
	   		while(i < l && sum < k){
	   			i++;
	   			sum = sum + a[(s[i]-'a')];
	   		}
	   		
	   		end = i;
	   		while(start <= end && sum > k){
	   			sum = sum - a[s[start]-'a'];
	   			start++;
	   		}
	   		if(sum == k){
	   			count++;
	   		}
	   		if(i+1 < l){
	   			i++;
	   			sum = sum + a[s[i]-'a'];   		
	   		}
	   		else if(i+1 >= l){
	   			i++;
	   		}
	   	}
	   	cout<<count<<endl;
		
	}
    return 0;
}
