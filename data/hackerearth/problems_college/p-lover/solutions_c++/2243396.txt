bool comp(const int& a, const int& b)

          return a<b?false:true;//min_heap
       }
using namespace std;
bool prime[maxn];
int ans[1000009],ans_[1000099];
int reverse_(int x)
{int y=0,k;
    while(x)
    {
        k=x%10;
        y=y*10+k;
        x=x/10;
    }
    return y;
}
int z=0;
void prime_()
{int i,j,c=0;
    for(i=1;i<=maxn;i++)
    {
        prime[i]=true;
    }
    for(i=2;i*i<=maxn;i++)
    {
        if(prime[i]==true)
        for(j=i*i;j<=maxn;j=j+i)
        {
            prime[j]=false;
        }
    }

    for(i=2;i<=maxn;i++)
    {
        if(prime[i])
        {
            ans[c]=i;
            c++;
        }
    }
    for(i=0;i<c;i++)
    {
        int y=reverse_(ans[i]);
        if(y==ans[i])
        {
            ans_[z]=y;
            z++;
        }
    }
}
int main()
{
    prime_();
    int t,x;
    s(t);
    w(t)
    {
        s(x);
        cout<<*lower_bound(ans_,ans_+z,x)<<endl;
    }
	return 0;
}
