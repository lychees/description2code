#include <iostream>
using namespace std;

#define MAX 1001
char s[MAX];
int ch[26];

int main()
{
 //   freopen("Palindrome Permutation.txt", "r", stdin);
	// setbuf(stdout, NULL);
    
	cin >> s;
	int length = 0;
	while(s[length] != '\0')
	{
		int c = s[length] - 'a';
		ch[c]++;
		length++;
	}

	bool palin = true;
	if(length % 2 == 0)
	{
		for(int i = 0; i < 26; i++)
			if(ch[i] % 2 == 1)
			{
				palin = false;
				break;
			}
		cout << (palin == true ? "YES" : "NO") << endl;
	}
	else
	{
		int count = 0;
		for(int i = 0; i < 26; i++)
			if(ch[i] % 2 == 1)
			{
				count++;
			}
		if(count > 1)
			cout << "NO" << endl;
		else
			cout << "YES" << endl;
	}
    return 0;
}
