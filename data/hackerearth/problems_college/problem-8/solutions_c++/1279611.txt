#include <iostream>
using namespace std;

int main()
{
    int n, a, b, x;
    while(cin>>n)
    {
    	if ( n == 1)
    		cout<<1<<endl;
    	else
    	{
    		a = 1; b = 1;
    		cout<<1<<endl<<1<<endl;
    		n -= 2;
    		while(--n >= 0)
    		{
    			x = b;
    			b = a + b;
    			a = x;
    			cout<<b<<endl;
    		}
    	}
    	
    	
    }
    return 0;
}
