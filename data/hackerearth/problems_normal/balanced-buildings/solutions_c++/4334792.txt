using namespace std;

int p, n, i, j, k, m, l , s;
int a[5000];

ll dp[3000][3000];

ll get(ll x, ll y)
{
	if (x < y)
		return (y - x) * p;
	else
		return (x - y) * m;
}

int main()
{
	//freopen("input.txt" , "r" , stdin);
	//freopen("output.txt" , "w" , stdout);
	cin >> n >> s >> m >> p;

	for (i = 0; i < n; i++)
		cin >> a[i];

	for (i = 0; i < n; i++)
	{
		dp[0][i] = -get(a[0], a[i]);
	}

	for (i = 1; i < n; i++)
	{
		ll mx = 0;
		for (j = 0; j < n; j++)
			mx = max(mx, dp[i - 1][j]);

		for (j = 0; j < n; j++)
		{
			ll val = get(a[i], a[j]);

			dp[i][j] = -val + mx;
			dp[i][j] = max(dp[i][j] , -val + dp[i - 1][j] + s);
		}
	}

	ll ans = 0;
	for (i = 0; i < n; i++)
		ans = max(ans, dp[n - 1][i]);

	cout << ans << endl;

	return 0;
}
