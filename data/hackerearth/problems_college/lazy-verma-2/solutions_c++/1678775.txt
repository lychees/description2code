#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   long long n,m;
   cin>>n>>m;
   long long a[n],b[m];
   for(long long i=0;i<n;i++)
   cin>>a[i];
   for(long long j=0;j<m;j++)
   cin>>b[j];
   long long c[n];
   c[0]=a[0];
   for(long long i=1;i<n;i++)
   c[i]=c[i-1]+a[i];
   long long temp=0;
   for(long long j=0;j<m;j++)
   temp+=(c[n-1]*b[j]);
   cout<<temp<<endl;
    return 0;
}
