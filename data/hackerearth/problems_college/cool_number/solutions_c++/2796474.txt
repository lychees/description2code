#include <iostream>
#include<string.h>
using namespace std;
int call(int n)
{
	string str="";
	int x,i,count;
	count=0;
	while(n>0)
	{
	   x=n%2;
	   str+=char(x+'0');
	   n=n/2;
	}
	  int l=str.length();
	  for(i=0;i<=l-3;i++)
	  {
	  	 if(str.substr(i,3)=="101")
	  	  count++;
	  }
	   return count;
}	   
int main()
{
                 int t,r,k;
                 cin>>t;
                 while(t--)
                 {
                 	 cin>>r>>k;
                 	 int count=0,i,cool;
                 	 for(i=5;i<=r;i++)
                 	 {
                 	 	cool=call(i);
                 	 	if(cool>=k)
                 	 	 count++;
                 	 }
                 	  cout<<count<<endl;
                 }
    return 0;
}
