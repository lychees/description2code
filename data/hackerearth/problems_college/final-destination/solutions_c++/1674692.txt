#include<bits/stdc++.h>
#define ll long long int
using namespace std;
int main()
{
	int a,b,t;
	cin >> t;
	while(t--)
	{
		bool visit[1000000];
		memset(visit,false,sizeof(visit));
        int distance[1000000],mul,di,add,sub;
        memset(distance,-1,sizeof(distance));
		ll tmp;
		cin >> a >> b;
		cin >> mul >> di >> add >> sub;
		queue<int> q;
		q.push(a);
		visit[a]=true;
		distance[a]=0;
		while(!q.empty())
		{
			int t=q.front();
			q.pop();
			if(t==b)
			{
				//int ans=distance[t];
				break;
			}
			tmp=t*mul;
			if(tmp >=2 && tmp<=1000000 && visit[tmp]!=true)
			{
				q.push(tmp);
				visit[tmp]=true;
				distance[tmp]=distance[t]+1;
			}
			tmp=t/di;
			if(tmp >=2 && tmp<=1000000 && visit[tmp]!=true)
			{
				q.push(tmp);
				visit[tmp]=true;
				distance[tmp]=distance[t]+1;
			}
			tmp=t+add;
			if(tmp >=2 && tmp<=1000000 && visit[tmp]!=true)
			{
				q.push(tmp);
				visit[tmp]=true;
				distance[tmp]=distance[t]+1;
			}
			tmp=t-sub;
			if(tmp >=2 && tmp<=1000000 && visit[tmp]!=true)
			{
				q.push(tmp);
				visit[tmp]=true;
				distance[tmp]=distance[t]+1;
			}
		}
		printf("%d\n",distance[b]);
	}
}
