#include<bits/stdc++.h>
#include <iostream>
using namespace std;
#define ll long
int main()
{
    ll t,n,a,ct,ct1,temp,tmp,i;
    cin>>t;
    while(t--)
    {
    	ct=ct1=0;
   		cin>>n;
   		for(i=0; i<n; i++)
   		{
   			cin>>a;
   			if(a==0)
   			++ct;
   			else
   			++ct1;
   		}
   		if(ct1&1)
   		{
   			cout<<ct1<<endl;
   			continue;
   		}
   		cout<<ct<<endl;
    }
    return 0;
}
