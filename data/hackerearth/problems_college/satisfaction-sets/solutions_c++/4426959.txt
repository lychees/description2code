#include <iostream>
#define M 1000000000
typedef long long int ll;
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int G[101][501]={0};int n,x;ll m,i,j,sum=0;
    cin>>n;
    for(i=1;i<=n;i++)
    {
       cin>>m;	
       for(j=1;j<=m;j++)
       {
       	cin>>x;G[i][x]++;
       }
    }
    for(i=0;i<=500;i++)
    {
    	int flag=0;
    	for(j=1;j<=n;j++)
    	{
    		if(G[j][i]>0) flag=1;
    		else {flag=0;break;}
    	}
    	if(flag==1) {sum=(sum+i)%M;}
    }
    cout<<-1*sum<<endl;
    return 0;
}
