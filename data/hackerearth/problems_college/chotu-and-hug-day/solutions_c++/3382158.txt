#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,h,flag=0;
		scanf("%d %d",&n,&h);
		int a[n];
		for(int i=0;i<n;i++)
			scanf("%d",&a[i]);
		sort(a,a+n);
		int i;
		for(i=0;i<n-1;i++)
		{
			if(abs(a[i]-h)<abs(a[i+1]-h))
			{
				flag=1;
				break;
			}	
		}
		if(flag==1)
		{
			int y=abs(a[i]-h);
			printf("%d\n",y);
		}
		else
		{
			int y=abs(a[n-1]-h);
			printf("%d\n",y);	
		}
	}
	return 0;
}
