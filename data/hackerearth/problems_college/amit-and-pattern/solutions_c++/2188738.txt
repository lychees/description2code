#include <bits/stdc++.h>
#define LL long long int
#define mod 1000000007

using namespace std;

int a[101][101],vis[101][101] = {0},n,m;

void rec(int r,int c){

    if(r>=n || c<0 || c>=m || r<0){
        cout << endl;
        return;
    }

    cout << a[r][c] << " ";
    vis[r][c] = 1;

    if(!vis[r+1][c-1])  rec(r+1,c-1);
}

int main () {

    cin >> n >> m;


    for(int i=0;i<n;i++)
        for(int j=0;j<m;j++)
            cin >> a[i][j];

      for(int i=0;i<n;i++)
        for(int j=0;j<m;j++)
            if(!vis[i][j])
                rec(i,j);

    return 0;
}
