#include <iostream>
#include <string>
#include <algorithm>

template <typename T>
std::string to_joey_base(T input)
{
	std::string n;
	
	while (input != 0) {
		n.push_back(static_cast<char>('2' + ((input - 1) % 3)));
		
		input = (input - 1) / 3;
	}
	
	std::reverse(n.begin(), n.end());
	
	return n;
}

int main()
{
	int tests;
	std::cin >> tests;
	
	while (tests--) {
		long long int input;
		std::cin >> input;
		
		std::cout << to_joey_base(input) << std::endl;
	}
	
    return 0;
}
