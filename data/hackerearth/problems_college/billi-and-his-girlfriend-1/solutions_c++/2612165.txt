#include<stdio.h>
#include<bits/stdc++.h>
using namespace std;
long long int X=1000000000;
int power(long long int x,long long int n)
{
	if(n==0)
	return 1;
	long long int tmp=power(x,n/2);
	if(n%2==0)
	return (tmp*tmp) % X;
	else
	return (((x*tmp)%X)*tmp) % X;
}
int main()
{
	long long int t,a,b,k;
	cin >> t;
	while(t--)
	{
		scanf("%lld%lld%lld",&a,&b,&k);
		int p=power(a,b);
		int sum=0;
		while(p > 0 && k-- > 0)
		{
			sum+=p%10;
			p=p/10;
		}
	   printf("%d\n",sum);
	}
}
