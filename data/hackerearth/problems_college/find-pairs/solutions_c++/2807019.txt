#include <iostream>
using namespace std;

int main()
{int t,n,a[1000];
    //cout << "Hello World!" << endl;
    cin>>t;
    while(t--){int flag;
    	cin>>n;
    	flag=n;
    	for(int i=1;i<=n;i++){
    		cin>>a[i];
    	}
    	    	for(int i=1;i<=n;i++){
    	    		for(int j=i+1;j<=n;j++){
    	    			
    	    				if(a[i]==a[j]){
    	    					flag++;
    	    				
    	    			}
    	    			
    	    		                 }
								}
								cout<<flag<<endl;
    }
    return 0;
}
