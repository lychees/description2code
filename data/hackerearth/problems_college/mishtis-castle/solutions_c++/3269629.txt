#include <bits/stdc++.h>
using namespace std;
int n,c;
int a[102][102];
long long a1[102][102];
bool vis[10005];
long long int f=0;
int dfs(int s,vector<int>* v)
{
	vis[s]=true;
	f+=a[s%n][s/n];
	for(int i=0;i<v[s].size();i++)
	{
		if(!vis[v[s][i]])
		{
			dfs(v[s][i],v);
			if(!vis[n*n-1]) f=0;
		}
	}
}
int main()
{
   int t;
   cin>>t;
   while(t--)
    {
        cin>>n>>c;
        memset(a,0,sizeof a);
        memset(a1,0LL,sizeof a1);
        for(int i=1;i<=n;i++)
        {
        	for(int j=1;j<=n;j++)
        		scanf("%d",&a[i][j]);
        }
        for(int i=1;i<=n;i++)
        {
        	for(int j=1;j<=n;j++)
        	{
        		a1[i][j]=a[i][j]+max(a1[i-1][j],a1[i][j-1]);
        	}
        }
 
		if(a1[n][n]>=c)
	        cout<<a1[n][n]-c<<endl;
	    else
        	cout<<"-1\n";
    }
    return 0;
}
