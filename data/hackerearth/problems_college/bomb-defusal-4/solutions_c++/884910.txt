#include<cstdio>
#include<iostream>
#include<math.h>
using namespace std;

int main()
{
    int   t,p,x,y,u,v,d;

    scanf("%d",&t);

    while(t--)
    {
            scanf("%d%d%d%d%d",&p,&x,&y,&u,&v);

            d=(int)(fabs(u-x)+fabs(v-y))%2;
            if(d)
                printf("Terrorists Win !\n");
            else
            {
                if(x==u&&y==v||p>=3)
                printf("Counter-Terrorists Win !\n");
                else
                if(fabs(u-x)==fabs(v-y)&&p>=2)
                printf("Counter-Terrorists Win !\n");
                else
                printf("Terrorists Win !\n");

            }

    }
    return 0;
}
