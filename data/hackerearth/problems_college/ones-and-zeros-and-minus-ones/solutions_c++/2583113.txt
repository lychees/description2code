#include<iostream>
using namespace std;
int main()
{
	int tc;
	cin>>tc;
	while(tc--)
	{
		int ar[100002],n,minus[100002],k=0;
		cin>>n;
		for(int i=0;i<n;i++)
		{
			cin>>ar[i];
			if(ar[i]==-1) minus[k++]=i;
		}
		int max=0;
		for(int i=0;i<k;i++)
		{
			int l=minus[i],count=0;
			l--;
			while(l>=0)
			{
				if(ar[l]==-1)
					break;
				if(ar[l]==1) count++;
				l--;
			}
			l=minus[i];
			l++;
			while(l<n)
			{
				if(ar[l]==-1)
					break;
				if(ar[l]==1) count++;
				l++;
			}	
			//cout<<count<<" ";
			if(count>max) max=count;
		}
		cout<<max<<"/"<<"1"<<endl;
	}
	return 0;
}
