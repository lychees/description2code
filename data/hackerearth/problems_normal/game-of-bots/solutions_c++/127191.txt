vector<int> relevant(string s)
{
    vector<int> x;
    for (int i=0; i<s.size(); i++) {
        if (s[i] != '#') {
            x.push_back(i);
        }
    }
    return x;
}
string go(string begin, string target)
{

    vector<int> x = relevant(begin), y = relevant(target);

    if (x.size() != y.size()) {

        return "No";
    }

    for (int i=0; i<x.size(); i++) {
        if ( begin[x[i]] != target[y[i]] ) {

            return "No";
        }
        if ( (begin[x[i]] == 'B') && (y[i] < x[i]) ) {

            return "No";
        }
        if ( (begin[x[i]] == 'A') && (y[i] > x[i]) ) {

            return "No";
        }
    }
    return "Yes";

}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        string a,b;
        cin>>a>>b;
        cout<<go(a,b)<<endl;
    }
    return 0;
}
