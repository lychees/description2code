#include<iostream>
using namespace std;
int main()
{
	int t,r,g,b,d,e,f;
	cin>>t>>r>>g>>b;
	int a[8]={0};
	for(int i=0;i<t;i++)
	{
		if((i/r)%2==0)
		d=0;
		else
		d=1;
		if((i/g)%2==0)
		e=0;
		else
		e=1;
		if((i/b)%2==0)
		f=0;
		else
		f=1;
		if(d==1)
		{
			if(e==1)
			{
				if(f==1)
				a[6]++;
				else
				a[3]++;
			}
			else
			{
				if(f==1)
				a[5]++;
				else
				a[0]++;
			}
		}
		else
		{
			if(e==1)
			{
				if(f==1)
				a[4]++;
				else
				a[1]++;
			}
			else
			{
				if(f==1)
				a[2]++;
				else
				a[7]++;
			}
		}
	}
	for(int i=0;i<8;i++)
	cout<<a[i]<<" ";
	return 0;
}
