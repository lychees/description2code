p = [1]*(1000001)
primes = []
p[0] = 0
p[1] = 0
cnt = 0
ind = [0]*1000001
cntr = [0]*(1000001)
for i in range(2, 1000001):
    if p[i] == 1:
        c = 1
        for j in range(2*i, 1000001, i):
            if p[j] != 0:
                c += 1
            p[j] = 0
        cntr[i] = c
        primes.append(i)
        ind[i] = cnt
        cnt += 1

t = input()
while t > 0:
    t -= 1
    n = input()
    print str(cntr[n])
