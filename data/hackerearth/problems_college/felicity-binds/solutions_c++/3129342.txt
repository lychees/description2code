#include <iostream>
using namespace std ;
#include <algorithm>
int main()
{

    int t ;
    cin >> t ;
    while ( t--)
    {
        int n ;
        cin >> n ;
        int i ;
        int man[10000] ;
        int w[10000] ;
        for ( i = 0; i < n ; i++)
        {

            cin >> man[i] ;
        }
        for ( i = 0; i < n ; i++)
        {
            cin >> w[i] ;
        }
        sort (man , man+n ) ;
        sort ( w , w+n) ;

     long long    sum = 0 ;
       for ( i = 0 ; i< n ; i++)
       {
           sum = sum + man[i]*w[i] ;
       }
       cout << sum << endl;

    }
    return 0 ;
}
