#include <iostream>
#include <cmath>
#define Max 1000050
using namespace std;

int A[Max],Z[Max];
void sieve()
{
	for(int i=3;i<=Max;i+=2)
	{
		A[i]=1;
	}
	int s=sqrt(Max);
	for(int i=3;i<=s;i+=2)
	{
		if(A[i])
		{
			for(int j=i*i;j<=Max;j+=2*i)
				A[j]=0;
		}
	}
	Z[2]=1;	Z[3]=2;	Z[4]=3;
	for(int i=5;i<Max;i++)
		Z[i]=Z[i-1]+A[i];
}
 
int main()
{
	sieve();	int t,a,b;
	ios::sync_with_stdio(false);
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d",&a,&b);
		printf("%d\n",(b-a+1)-(Z[b]-Z[a-1]));
	}
	return 0;
}
