#include<stdio.h>
#include<iostream>
#include<algorithm>
#include <string.h>
#include <math.h>


using namespace std;

int main()
{
	int t;
	cin>>t;
	while (t--) {
		int n,m;
		cin>>n>>m;
		int max = 0;
		for(int i = n;i<=m;i++) {
			for(int j = n;j<=m;j++) {
				if((i ^ j) > max) {
					max = i^j;
				}
				
			}
		}
		cout<<max<<endl;
	}


return 0;
}
