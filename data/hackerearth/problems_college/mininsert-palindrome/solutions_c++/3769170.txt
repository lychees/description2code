int fun(string s, int l, int h)
{
    if(l > h) return INT_MAX;
    if(l == h) return 0;
    if(l == h-1) return(s[l] == s[h] ? 0 : 1);
    return(s[l] == s[h] ? fun(s, l+1, h-1) : (min(fun(s, l+1, h), fun(s, l, h-1)) + 1));
}

int main()
{
    int t, n, i;
    string s;
    cin >> t;
    while(t--) {
        cin >> n;
        cin >> s;
        cout << fun(s, 0, s.length()-1) << endl;
    }

    return 0;
}
