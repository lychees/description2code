#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    long int m,x,val,i,j,sm,root,flag;
    scanf("%ld",&m);
    long int a[m],b[500]={0};
    for(i=0;i<m;i++)
    scanf("%ld",&a[i]);
    sort(a,a+m);
    sm=a[m-1]/m-1;
    sm=sm*2;
    if(m==2)
    sm=a[1]-a[0];
    for(i=sm;i>=2;i--)
    {
    	x=a[0]%i;
    	flag=1;
    	for(j=1;j<m;j++)
    	{
    		if(a[j]%i!=x)
    		{
    			flag=0;
    			break;
    		}
    	}
    	if(flag==1)
    	{
    		val=i;
    		break;
    	}
    }
    root=sqrt(val);
    j=0;
    for(i=1;i<=root;i++)
    {
    	if(val%i==0)
    	{
    		b[j++]=i;
    		if(i*i!=val)
    		b[j++]=val/i;
    	}
    }
    sort(b,b+500);
    for(i=0;i<500;i++)
    {
    	if(b[i]!=0 && b[i]!=1)
    	printf("%ld ",b[i]);
    }
    return 0;
}
