#include<bits/stdc++.h>
using namespace std;
int main()
{
	map <char,int> m;
	char c;
	for(int i=65,j=26;i<=90;i++)
	{
		c=(char)i;
		m[c]=j;
		j--;
	}
	int t;
	scanf("%d ",&t);
	while(t--)
	{
		int cnt[26]={0},x;
		char s[100001];
		scanf("%s",s);
		//printf("%s\n",s);
		for(int i=0;i<strlen(s);i++)
		{
			x=(int)(s[i]-65);
			cnt[x]++;
		}
		int an=0;
		for(int i=0;i<26;i++)
		{
			c=65+i;
			if(cnt[i]==0)
				continue;
			else if(cnt[i]==m[c])
				continue;
			else
			{
				an=1;
				break;
			}
		}
		if(an==0)
			printf("Yes\n");
		else
			printf("No\n");
	}
	return 0;
}
