#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define maxi 201
long long q,n;
long long a[maxi],b[maxi];
int dp[2][800001];

long long solve(long long i,long long diff)
{
	 if(diff>400000||diff<-400000)return 0;
 if(i==n)
 {
     return (abs(diff)<=q);
 }
 
 int &ans = dp[i&1][diff+400000];
    if(ans!=-1)
        return ans;
        ans=0;

 ans+=solve(i+1,diff);
 ans=ans%mod;
 ans+=solve(i+1,diff+a[i]);
  ans=ans%mod;
 ans+=solve(i+1,diff-b[i]);
  ans=ans%mod;

 ans+=solve(i+1,diff+a[i]-b[i]);
  ans=ans%mod;
 return ans;

}
int main(){
long long i;
cin>>n;

for(i=0;i<n;i++)
cin>>a[i];
for(i=0;i<n;i++)
cin>>b[i];

cin>>q;
for(int i=n;i>=0;i--){
        memset(dp[i&1],-1,sizeof dp[i&1]);
        for(int j=-400000;j<=400000;j++){
            solve(i,j);
        }
    }
cout<<solve(0,0)<<endl;

}
