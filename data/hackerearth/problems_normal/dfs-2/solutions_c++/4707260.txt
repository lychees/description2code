#include <bits/stdc++.h>
using namespace std;

typedef long long int ll;

ll lbis(ll a[],ll x,ll l,ll r)
{
	if(r>=l)
	{
		ll mid=l+(r-l)/2;
		
		if(a[mid]==x && a[mid-1]<x) return mid;
		
		if(a[mid]>x && a[mid-1]<x) return mid;
			
		if(a[mid]<x) return lbis(a,x,mid+1,r);
		
		return lbis(a,x,l,mid-1);
	
	}
	
	return -1;
}

ll rbis(ll a[],ll x,ll l,ll r)
{
	if(r>=l)
	{
		ll mid=l+(r-l)/2;
		
		if(a[mid]==x && a[mid+1]>x) return mid;
		
		if(a[mid]<x && a[mid+1]>x) return mid;
		
		if(a[mid]>x) return rbis(a,x,l,mid-1);
		
		return rbis(a,x,mid+1,r);
		
	}
	
	return -1;
}
