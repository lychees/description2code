#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	string a,b;
    	cin>>a>>b;
    	string c;
    
    for(int i=0;i<a.length();i++){
    		if(a[i]==b[i])c.push_back('0');
    		else c.push_back('1');
    	}
    	cout<<c<<endl;
    	
    }
    
    
    
    return 0;
}
