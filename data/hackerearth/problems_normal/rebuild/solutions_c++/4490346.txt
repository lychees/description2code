void fs(int &x)
{
    register int c = gc();
    x = 0;
    int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}

int id[100002];
int deg[100002];

void init(int n)
{
    repn(i,n)
    {
        id[i]=i;
        deg[i]=0;
    }
}

int root(int x)
{
    while(id[x]!=x)
    {
        id[x]=id[id[x]];
        x=id[x];
    }
    return x;
}

void unionn(int a, int b)
{
    int u=root(a);
    int v=root(b);
    id[u]=v;
}

ll kruskal(viii e)
{
    ll ans=0;
    foreach(it, e)
    {
        int c=it->fi;
        int a= -((it->se).fi);
        int b= -((it->se).se);
        if(root(a)!=root(b))
        {
            unionn(a,b);
            ans+=c;
            deg[a]++;
            deg[b]++;
        }
    }
    return ans;
}
