long long int power(long long int a,long long int b) 
{
   long long int x = 1, y = a;
    while(b > 0) 
    {
        if(b%2 == 1) 
        {
            x=x*y;
            
        }
        y = y*y;
       
        b =b/2;
    }
    return x;
}
