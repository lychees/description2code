#include<bits/stdc++.h>
#define ll int
#define ld long double
#define f first
#define s second
#define pb push_back
#define M 1000000007
#define MX 2000005
#define PI 3.141593
using namespace std;
typedef pair<ll,ll> ii;
ll bucket_sum[30][500]={0};
ll query[40005][3],flag[30][200005];
char ans[200005];
void bucket(ll ch,ll occ,ll b_size,ll n)
{
  ll x=0,i,found=0;
  for(i=1;i*b_size<n;i++)
  {
    x+=bucket_sum[ch][i-1];
    //cout<<occ<<" "<<x<<"\n";
    if(x>=occ)
        {
            x-=bucket_sum[ch][i-1];
            found=1;
            break;
        }
  }
  occ-=x;
  //cout<<"Range:"<<(i-1)*b_size<<" "<<min((i)*b_size,n)<<"\n";
  ll ul=min((i)*b_size,n);
  for(ll j=(i-1)*b_size;j<ul;j++)
  {
      if(flag[ch][j]==1)
        {
            //cout<<j<<"\n";
            occ--;
        }
      if(!occ)
      {
          //cout<<"Removal: "<<j<<"\n";
          flag[ch][j]=2;
          bucket_sum[ch][i-1]--;
          break;
      }

  }


}
int main()
{
    string s;
    cin>>s;
    ll q,n=s.size(),bucket_size;
    //SQRT-DECOMPOSITION
    bucket_size=(ll)sqrt((ld)n);
    if(bucket_size*bucket_size<n)
        bucket_size++;
    for(ll i=0;i<n;i++)
    {
        flag[s[i]-'a'][i]=1;
    }
    for(ll i=0;i<26;i++)
    {
        ll j;
        for(j=1;j*bucket_size<n;j++)
            {
                ll sz=0;
                for(ll k=(j-1)*bucket_size;k<j*bucket_size;k++)
                    if(flag[i][k]==1)
                    sz++;
                bucket_sum[i][j-1]=sz;
            }
        for(ll k=(j-1)*bucket_size;k<n;k++)
           if(flag[i][k]==1)
             bucket_sum[i][j-1]++;
    }
    /*for(ll i=0;i<26;i++)
    {
        cout<<(char)(i+(int)'a');
        ll j;
        for(j=1;j*bucket_size<n;j++)
            cout<<bucket_sum[i][j-1]<<" ";
        cout<<bucket_sum[i][j-1]<<"\n";
    }*/
    cin>>q;
    for(ll i=0;i<q;i++)
    {
        string str;
        cin>>query[i][0]>>str;
        //cout<<str<<"\n";
        //query[i][1]=str[0]-'a';
        bucket(str[0]-'a',query[i][0],bucket_size,n);
    }
    for(ll i=0;i<26;i++)
    {
        for(ll j=0;j<n;j++)
        {
            if(flag[i][j]==1)
                ans[j]=(char)(i+(ll)'a');
        }
    }
    for(ll i=0;i<n;i++)
    {
        if(ans[i]>='a'&&ans[i]<='z')
            cout<<ans[i];
    }
    cout<<"\n";
}
