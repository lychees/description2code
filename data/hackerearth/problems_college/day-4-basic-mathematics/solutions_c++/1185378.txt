#include <iostream>
using namespace std;

int main()
{
    int T,days;
    cin>>T;
    for(int i=0;i<T;++i)
    {
    	cin>>days;
    	int x=1;
    	while(days>0)
    	{
    		days-=x;
    		++x;
    	}
    	--x;
    	if(x%3==0) cout<<"SL"<<endl;
    	else if(x%3==1) cout<<"LB"<<endl;
    	else cout<<"BS"<<endl;
    }
}
