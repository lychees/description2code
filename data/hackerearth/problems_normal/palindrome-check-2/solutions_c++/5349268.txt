#include <iostream>
using namespace std;

int main()
{
    string s1,s2;
    int i,k=0,j,count=0;
    cin>>s1;
    for(i=s1.size()-1;i>=0;i--)
    {
    	s2[k] = s1[i];
    	k++;
    }
    s2[k] = '\0';
    for(j=0;s1[j]!='\0';j++)
    {
    	if(s1[j]!=s2[j])
    	{
    		count++;
    		break;
    	}
    }
    if(count!=0)
    cout<<"NO";
    else
    cout<<"YES";
    return 0;
}
