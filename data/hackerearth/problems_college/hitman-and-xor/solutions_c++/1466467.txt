#include <iostream>
using namespace std;

int main()
{
    long long int t;
    cin>>t;
    
    while(t--)
    {
    	 long long int n,i,k,ev=0,od=0;
    	 cin>>n;
    	 
    	 for(i=0;i<n;i++)
    	 {
    	 	 cin>>k;
    	 	 if(k & 1)
    	 	   od++;
    	 	 else
    	 	   ev++;
    	 }
    	 cout<<ev*od<<endl;
    	 
    }
    return 0;
}
