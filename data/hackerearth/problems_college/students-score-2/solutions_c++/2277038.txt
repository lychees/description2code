#include <iostream>
   using namespace std;
    int main()
    {
	int n;
	cin>>n;
	int max_sofar=0,max=0,arr[n];
	
	for(int i=0;i<n;i++)
	cin>>arr[i];
	for(int i=0;i<n;i++)
	{   max_sofar=max_sofar+arr[i];
		if(max_sofar<0)
		max_sofar=0;
		if(max_sofar>max)
		max=max_sofar;
	}
	
	
	cout<<max<<endl;
	  return 0;
   }
