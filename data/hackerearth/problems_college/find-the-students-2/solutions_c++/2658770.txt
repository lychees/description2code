#include <bits/stdc++.h>
using namespace std;

int main()
{
ios_base::sync_with_stdio(false);
  	
	long long int n,i,a[100000],k,z,min,b[100000];
    cin>>n;
    for(i=0;i<n;i++)
    cin>>a[i];
    
    sort(a,a+n);
    k=n;
    while(k>0)
    {
    	z=0;
    	min=99999999999;
    for(i=0;i<k;i++)
    {
    	if(a[i]<min)
    	min=a[i];
    }
     cout<<k<<endl;
    for(i=0;i<k;i++)
    {
      a[i]=a[i]-min;
    if(a[i]>0)
    b[z++]=a[i];
    }
    for(i=0;i<z;i++)
    a[i]=b[i];
    k=z;
   
   
   
    }
    return 0;
}
