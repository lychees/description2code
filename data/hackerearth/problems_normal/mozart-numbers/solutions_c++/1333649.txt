#include <iostream>
#define lli long long int
#define s(a) scanf("%lld",&a)

using namespace std;

lli fib[1000001]={0};
lli sum[1000001];
lli mod=1000000007;

int main()
{
    lli i,n,t,a,b,ans;
    fib[0]=0;
    fib[1]=1;
    fib[2]=1;
    
    for(i=3;i<=1000000;i++) {
        fib[i]=fib[i-1]+fib[i-2];
        //cout<<i<<"   ";check(fibo[i]);
        fib[i]%=mod;
    }
    
    s(t);
    
    while(t--) {
        ans=0;
        s(a);
        s(b);
        --a;
        --b;
        
        if (b > 12)
        	b = 12;
        
        if(a<=6&&b>=6) ans+=8;
        if(a<=12&&b>=12) ans+=144;
        printf("%lld\n",ans);
    }
    return 0;
}
