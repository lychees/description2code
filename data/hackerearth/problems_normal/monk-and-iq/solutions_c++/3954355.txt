#include <bits/stdc++.h>
#define ll long long
using namespace std;
typedef struct node
{
	ll int id,cc,p1,p2;//course-id,course-count of students,marks of last & 2nd last stuent enrolled
}node;
node a[100010];
ll int monk[100010];
struct my
{
	bool operator()(const node a,const node b)//z = x*c where c is the number of students already
	{
		ll int x1,x2;
		x1=(a.p1+a.p2)*a.cc;
		x2=(b.p1+b.p2)*b.cc;
		if(x1<x2)
		{
			return false;
		}
		if(x1==x2)
		{
			if(a.id<b.id)
			return false;
		}
		return true;
	}
};
int main()
{
	priority_queue<node,vector<node>,my> q;
	ll int c,p,n;
	scanf("%lld%lld%lld",&c,&p,&n);//course,mank & friends,n students already enrolled
	node t;
	ll int i,j;
	for(i=1;i<=c;i++)
	{
		a[i].id=i;
		a[i].cc=0;
		a[i].p1=0;
		a[i].p2=0;
	}
	for(i=1;i<=n;i++)
	{
		scanf("%lld",&j);
		a[i].cc++;
		a[i].p1=j;
	}
	for(i=1;i<=c;i++)
	{
		q.push(a[i]);
	}
	for(i=0;i<p;i++)
	{
		scanf("%lld",(monk+i));
	}
	for(i=0;i<p;i++)
	{
		t=q.top();
		q.pop();
		t.cc++;
		j=1;
		t.p2=t.p1;
		t.p1=monk[i];
		q.push(t);
		printf("%lld ",t.id);
	}
	printf("\n");
	return 0;
}
