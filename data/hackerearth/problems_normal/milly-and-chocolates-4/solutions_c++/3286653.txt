#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long int t,n,temp;
    double ans=0.0;
    cin>>t;
    while(t--)
    {
    	ans=0.00000000;
    	cin>>n;
    	for(int i=0;i<n;i++)
    	{
    		cin>>temp;
    		temp*=1.00000000000;
    		ans+=(temp*1.00000000/n)*1.00000000;
    	}
    	printf("%0.8f\n",ans);
    }
    return 0;
}
