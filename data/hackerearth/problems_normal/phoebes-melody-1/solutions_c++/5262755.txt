#include <bits/stdc++.h>
#define ll long long int
#define M 1000000007
#define pb push_back
#define mp make_pair

using namespace std;
int v[200005];
int l[200005], r[200005];

int main()
{
	int t, n, k;
	
	cin >> t;
	while (t--) {
		cin >> n >> k;
		for (int i = 1; i <= n;i++) {
			scanf("%d", &v[i]);
			l[i] = i - 1;
			r[i] = i + 1;
		}
		
		v[0] = v[n + 1] = 0;
		l[0]= 0;
		r[n+1] = n+1;
		
		for (int i = 1; i <= n; i++) {
			while (l[i] >= 1 && v[i] <= v[l[i]]) {
				l[i] = l[l[i]];
			} 
		}
		
		for (int i = n; i >= 1; i--) {
			while (r[i] <= n && v[i] <= v[r[i]]) {
				r[i] = r[r[i]];
			}
		}
		int ans = 0;
		for (int i = 1; i <= n; i++) {
			int x = i - k;
			int y = i + k;
			int tmp = INT_MAX;
			while (x >= 1 && v[i] <= v[x]) {
				x = l[x];
			}
			
			while (y <= n && v[i] <= v[y]) {
				y = r[y];
			}
			
			if (x >= 1 && x <= i - k) {
				tmp = min(tmp, abs(i-x));
			}
			
			if (y <= n && y >= i+k) {
				tmp = min(tmp, abs(i-y));
			}
			
			if (tmp != INT_MAX) {
				ans += tmp;
			}
		}
		
		cout << ans << endl;
	}
	
	return 0;
}
