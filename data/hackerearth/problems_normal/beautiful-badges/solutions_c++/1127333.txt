#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;

vector <int> y, b;

int main()
{
	int n;
	int r = 0;
    cin >> n;
    for(int i=0;i<n;i++) {
    	int c, d;
    	scanf("%d%d", &c, &d);
    	if(c == 1) b.push_back(d);
    	else y.push_back(d);
    }
    sort(b.begin(),b.end());
    sort(y.begin(),y.end());
    int j = 0, i = 0;
    while(i < y.size() && j < b.size()) {
    	if(y[i]*y[i] >= 2*b[j]*b[j]) j++;
    	else if(2*y[i]*y[i] <= b[j]*b[j]) i++;
    	else { i++; j++; r++;}
    }
    cout << r << endl;
    return 0;
}
