    #include <bits/stdc++.h>
    using namespace std;
    int main()
    {
    	int t;
    	scanf("%d",&t);
    	assert(t >= 1 && t <= 50000);
    	while(t--)
    	{
    		long long int n,m;
    		scanf("%lld %lld",&n,&m);
    		assert(n >= 1 && n <= 10000000000000000);
    		assert(m >= 1 && m <= 10000000000000000);
    		int a[100] = {0};
    		int len = 0;
    		int l1 = -1;
    		while(m > 0)
    		{
    			a[len] = m%2;
    			if(a[len] == 1 && l1 == -1)
    				l1 = len;
    			m/=2;
    			len++;
    		}/*
    		for (int i = 0; i < len; ++i)
    		{
    			printf("%d", a[i]);
    		}
    		printf("\n");*/
    		for (int i = len-1; i > l1; --i)
    		{
    			if(a[i] == 1)
    				printf("(%lld<<%d) + ", n,i);
    		}
    		printf("(%lld<<%d)\n", n,l1);
    	}
    	return 0;
    }
