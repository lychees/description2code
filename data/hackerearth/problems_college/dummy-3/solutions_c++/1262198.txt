#include<cstring>

typedef long long LL;
typedef vector<int> VI;

#define REP(i,n) for(int i=0, i##_len=(n); i<i##_len; ++i)

int T;
char F[16][16];

bool rec(int z) {
    if (z == 81) return true;
    int x = z % 9, y = z / 9;
    if (F[y][x] != '0') return rec(z+1);
    
    for (int d=1; d<=9; d++) {
	bool ok = true;
	REP (i, 9) {
	    if (F[y][i] == d + '0') ok = false;
	    if (F[i][x] == d + '0') ok = false;
	    if (F[z/27*3 + i/3][z%9/3*3 + i%3] == d + '0') ok = false;
	}
	F[y][x] = d + '0';
	if (ok && rec(z+1)) return true;
	F[y][x] = '0';
    }
    return false;
}

int main() {
    scanf("%d", &T);
    int sum = 0;
    REP ($, T) {
	REP (i, 9) scanf("%s", F[i]);
	rec(0);
	REP (i, 9) sum += F[i][i] - '0';
    }
    printf("%d\n", sum);
	 
    return 0;
}
