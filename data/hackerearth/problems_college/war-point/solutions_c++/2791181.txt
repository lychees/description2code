#include <iostream>
#include<cstdio>
using namespace std;

int main()
{
    int T;
    cin>>T;
    while(T--){
    	int N;
    	scanf("%d", &N);
    	int *a = new int[N];
    	long long sum =0;
    	for(int i =0; i <N; i++) {
    		scanf("%d", &a[i]);
    		sum+= a[i];
    	}
    	long long min = 100000000000;
    	long long sum2 =0;
    	for(int i =0; i <N; i++) {
    		sum-= a[i];
    		sum2 +=a[i];
    		long long temp = sum - sum2;
    		if(abs(temp) < min) min = abs(temp);
    	}
    	printf("%d\n",min);
    }
    return 0;
}
