#include <bits/stdc++.h>
using namespace std;
void dfs(int n);


void dfs(int n,map<int,vector<int> > &tree)
{

	for(vector<int>:: iterator it = tree[n].begin();it!=tree[n].end();it++)
	{
		if(*it!=0)
		{
			cout<<*it<<" ";
			dfs(*it,tree);
		}
	}
}
int main()
{
    int T;
    //freopen("file1.txt","r",stdin);
    cin>>T;
    while(T--)
    {
    	int n,X,Y,Z;
    	cin>>n;
    	map<int,vector<int> > tree;
    	for(int i=1;i<=n;i++)
    	{
    		cin>>X>>Y>>Z;
    		tree[X].push_back(Y);
    		tree[X].push_back(Z);
    	}
    	cout<<"1"<<" ";
    	dfs(1,tree);
    	cout<<endl;
    }
}
