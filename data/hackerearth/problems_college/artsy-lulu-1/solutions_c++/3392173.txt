#include <iostream>
using namespace std;

int main()
{
  char letter[1];
  int height;
  while(cin >> letter)
  {
    cin >> height;
    int letterCode = (int) letter[0];
   	
   
   	if((
   		(letterCode <= (int) 'Z' && letterCode >= (int) 'A')
   	    ||
   	    (letterCode <= (int) 'z' && letterCode >= (int) 'a')
   	   )
   	   &&
   	   (
   	    ((letterCode + (height - 1)) <= 'Z')
   	   	||
   	   	((letterCode + (height - 1)) <= 'z')
   	   )
   	   && 
   	   (
   	    height >=  2 
   	    &&
   	    height <= 26
   	   )
   	  )
   	{
		for(int i = 0; i < height; i++)
			{
				for(int j = 0; j <= i; j++)
				{ 
					 cout << (char)(letterCode + i) << " ";
				}
			cout << endl;
			}
		cout << endl;
	}
	else
	  cout << "INVALID" << endl << endl;
  }
  return 0;
}
