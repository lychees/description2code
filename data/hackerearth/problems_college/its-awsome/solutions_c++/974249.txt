#include <iostream>
using namespace std;

int main()
{	long long int T,n[1000],i,a,c,e,count,d,y;
	cin>>T;
    for(i=0;i<T;++i)
    	cin>>n[i];
    for(i=0;i<T;++i)
    {	count=0;
    	y=0;
    	a=n[i];
    	while(a>0)
    	{	c=a%10;
    		if(c==4||c==7)
    			++count;
    		a=a/10;
    	}
    	e=count;
    	if(e==0)
    		cout<<"NO \n";
    	while(e>0)
    	{	d=e%10;
    		if(d!=4&&d!=7)
    		{	cout<<"NO \n";
    			break;
    		}
    		y*=10;
    		y+=d;
    		if(y==e)
    		{	cout<<"YES \n";
    			break;
    		}
    		e=e/10;
    	}
    }
    return 0;
}
