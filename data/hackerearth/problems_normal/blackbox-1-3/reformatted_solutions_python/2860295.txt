from fractions import gcd
n = int(raw_input())
for i in range(n):
    a, b, c = map(int, raw_input().split())
    val = gcd(a, gcd(b, c))
    if (a**2 == b**2+c**2 and val == 1):
        print "YES"
    elif (b**2 == a**2+c**2 and val == 1):
        print "YES"
    elif (c**2 == a**2+b**2 and val == 1):
        print "YES"
    else:
        print "NO"
