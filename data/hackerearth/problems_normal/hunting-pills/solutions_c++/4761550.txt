stack<long long> s;
vector<long long> visited ;

long long DFSutil2(long long vertex, vector<long long> adj[],vector<long long> cnt){
    long long ans = cnt[vertex] ;
    visited[vertex]=1 ;
    FOR(i,adj[vertex].size()){
        if(!visited[adj[vertex][i]]){
            ans+= DFSutil2(adj[vertex][i], adj,cnt) ;
        }
    }
    return ans ;
}
void DFSutil(long long vertex,long long v,vector<long long> adj[]){
    visited[vertex]=1 ;
    FOR(i,adj[vertex].size()){
        if(visited[adj[vertex][i]])
            continue ;
        
        DFSutil(adj[vertex][i],v,adj) ;
    }
    s.push(vertex) ;

}

void DFS(vector<long long> adj[],vector<long long> adjr[],vector<long long> cnt, long long v, long long e) 
{
    
    visited.resize(v) ;
    fill(visited.begin(),visited.end(),0) ;
    while(!s.empty())
    s.pop() ;
    for(int vertex=0;vertex<v;vertex++){
        if(!visited[vertex]){
            DFSutil(vertex,v,adj) ;
        }
    }
    fill(visited.begin(),visited.end(),0) ;
    long long ans= 0 ;
    long long curmax= 0 ;
    while(!s.empty()){
        long long vertex = s.top() ;
        s.pop() ;
        if(visited[vertex])
            continue  ;
        curmax=DFSutil2(vertex,adjr,cnt) ;
        ans = max(ans,curmax) ;
    }
    plldn(ans) ;
}


int main(){
    std::ios_base::sync_with_stdio (false);
    long long t,v,e,x,y ;
    slld(t) ;
    while(t--){
        slld(v) ;
        slld(e) ;
        vector<long long> adj[v] , adjr[v] ;
        vector <long long> cnt(v) ;
        FOR(i,v)
            slld(cnt[i]) ;
        FOR(i,e){
            slld(x) ;
            slld(y) ;
            adj[x].pb(y) ;
            adjr[y].pb(x) ;
        }
        DFS(adj,adjr,cnt, v , e) ;

    }
    return 0 ;
}
