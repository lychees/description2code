#include <iostream>
#include<map>
#include<algorithm>
#include<cstdio>
using namespace std;
#define tr(c,it) \
for(typeof(c.begin())it=c.begin();it!=c.end();it++)
int main()
{
	char str[1000];
    int t,i,count[26],flag=1;
    scanf("%d",&t);
    while(t--)
    {
    	flag=1;
    	map<char,int> m;
    	scanf("%s",str);
    	for(i=0;str[i]!='\0';i++)
    	{
    		m[str[i]]++;
    	}
    	int j=0;
    	for(map<char,int>::iterator it=m.begin();it!=m.end();it++)
    	{
    		count[j]=it->second;
    		j++;
    	}
    	if(j==26)
    	{
			sort(count,count+26);
			for(i=1;i<26;i++)
			{
				if(count[i]!=count[i-1])
				{
					flag=0;
					break;
				}
			}
			if(flag==0)
				printf("No\n");
			else
				printf("Yes\n");
    	}
    	else
    		printf("No\n");
    }
    return 0;
}
