#include <iostream>
#include <stdio.h>
#include <string.h>
using namespace std;

int main()
{
    char str[105];
    scanf("%s", str);
    int n = strlen(str);
    int arr[n];
    arr[0] = 1;
    for(int i=1; i<n; i++) {
    	if(str[i] == str[i-1])
    		arr[i] = arr[i-1]+1;
    	else
    		arr[i] = 1;
    }
    int sum = 0;
    for(int i=0; i<n; i++)
    	sum += arr[i];
    printf("%d", sum);
    return 0;
}
