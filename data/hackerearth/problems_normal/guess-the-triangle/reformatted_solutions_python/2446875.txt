a1, a2 = map(float, raw_input().split())
b1, b2 = map(float, raw_input().split())
c1, c2 = map(float, raw_input().split())

x1 = (c1-b1+a1)
x2 = (c1+b1-a1)
x3 = (b1-c1+a1)

y1 = (c2-b2+a2)
y2 = (c2+b2-a2)
y3 = (b2-c2+a2)

arr = []
arr.append((x1, y1))
arr.append((x2, y2))
arr.append((x3, y3))

arr.sort()

for i in arr:
    x, y = i
