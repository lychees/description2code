#include <iostream>
using namespace std;

int main()
{
    int n, t;
    cin >> n >> t;
    while (t--)
    {
    	int m;    	
    	cin >> m;
    	bool chf[10];
    	for (int i = 0; i < 10; ++i)
    		chf[i] = false;
    		
    	for (int i = 0; i < m; ++i)
    	{
    		int temp;
    		cin >> temp;
    		chf[temp]++;
    	}
    	
    	int cnt = 0;
    	for (int i = 1; i <= n; ++i)
    	{
    		int j = i;
    		while (j)
    		{
    			if (chf[j % 10])
    			{
    				cnt++;
    				break;
    			}
    			j /= 10;
    		}
    	}
    	
    	cout << cnt << endl;
    }
    return 0;
}
