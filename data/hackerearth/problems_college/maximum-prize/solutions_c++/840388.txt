#include <iostream>
#include<bits/stdc++.h>
#include <algorithm>
using namespace std;

int main()
{
    int T;
    scanf("%d", &T);
    for(int i=0; i<T; i++){
    	long N, K;
    	scanf("%ld %ld", &N, &K);
    	long long *a = new long long[N];
    	
    	for(int j =0; j<N; j++)scanf("%lld", &a[j]);
  
    	sort(a,a+N);
    	
    	long long total=0;
    	if(N<K) K = N;
    	if(K>0){
    		for(int l=1; l<=K; l++){
    			total = total+a[N-l];
    		}
    		printf("%lld\n", total);
    	}
    	else printf("%lld\n", total);
    	
    	delete [] a;
 	
    }
    
    return 0;
}
