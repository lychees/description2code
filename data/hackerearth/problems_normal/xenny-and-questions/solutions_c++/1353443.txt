#include <iostream>
using namespace std;
void countSort(long a[],long n,long m)
{
    // The output character array that will have sorted str
    long *output,i;
    // Create a count array to store count of inidividul characters and
    // initialize count array as 0
    int *count;
 	count=(int*)calloc(m+1,sizeof(int));
 	output=(long*)calloc(n,sizeof(long));
    // Store count of each character
    for(i-0;i<n;i++)
        count[a[i]]++;
 
    // Change count[i] so that count[i] now contains actual position of
    // this character in output array
    for(i=1;i<=m;i++)
        count[i]+=count[i-1];
 
    // Build the output character array
    for (i=0;i<n;i++)
    {
        output[count[a[i]]-1]=a[i];
        count[a[i]]--;
    }
    // Copy the output array to str, so that str now
    // contains sorted characters
    for(i=0;i<n;i++)
        a[i]=output[i];
}
 
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	long int n,k,i,j,m=0;
    	long long sum=0;
    	scanf("%ld%ld",&n,&k);
    	long a[n+1];
    	for(i=0;i<n;i++)
    	{
    		scanf("%ld",&a[i]);
    		if(a[i]>m)
    			m=a[i];
    	}
    	countSort(a,n,m);
    	for(i=n-1;i>=0&&k>0;i--,k--)
    		sum+=a[i];
    	printf("%lld\n",sum);
    }
    return 0;
}
