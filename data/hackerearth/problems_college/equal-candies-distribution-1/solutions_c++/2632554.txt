#include<bits/stdc++.h>


using namespace std;

int main()
{
	int t;
	cin>>t;
	
	while(t--)
	{
		long long int n,i,candies=0,sum;
		cin>>n;
		
		long long int arr[n],arr1[n];
		
		
		for(i=0;i<n;i++)
		{
			cin>>arr[i];	
		}
		sort(arr,arr+n);
		
		for(i=0;i<n;i++)
		{
			arr1[i]=arr[i]- arr[0];
		}
		for(i=0;i<n;i++)
		{
			sum=arr1[i];
			candies+=sum/5;
			sum%=5;
			candies+=sum/2;
			sum%=2;
			candies+=sum/1;
		}
		cout<<candies<<endl;
	}
	
	
	
}
