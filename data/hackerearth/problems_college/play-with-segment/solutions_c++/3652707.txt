#include <iostream>
#include <cstring>
#include <cstdio>
#include <cstdlib>
#include <vector>
#include <stack>
#include <queue>
#include <algorithm>
#include <cmath>
#include <climits>

using namespace std;
int arr[500008];
int dp[500007][2];
int main()
{
	int t;
	scanf("%d",&t);
	while(t--){
    int N;
    scanf("%d",&N);

    //int arr[N];
    for(int i=0; i<N; i++)
        scanf("%d",&arr[i]);

    //int dp[N][2];
    for(int i=0; i<N; i++)
        dp[i][0] = dp[i][1] = 1;

    for(int i=1; i<N; i++)
    {
        if ( arr[i] == arr[i-1] )
        {
            dp[i][0] += dp[i-1][0];
            dp[i][1] += dp[i-1][1];
        }
        else if ( arr[i] < arr[i-1]  )
        {
            dp[i][1] += dp[i-1][0];
        }
        else
        {
            dp[i][0] += dp[i-1][1];
        }
    }

    int ret = INT_MIN;
    for(int i=0; i<N; i++)
        for(int j=0; j<2; j++)
            ret = max(ret, dp[i][j]);

    printf("%d\n",ret);}

    return 0;
}
