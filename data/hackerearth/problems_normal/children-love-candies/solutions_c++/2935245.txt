#include <iostream>
using namespace std;
int main()
{
	long long int x;
	cin >> x;
	while(x--)
	{
		unsigned long long int y,z;
		cin >> y >> z;
		while(z--)
		{
			long long int q;

			if(y%2!=0)
			{
				q=(y+1)/2;
			}
			else
				q=y/2;
			y-=q;

			long long int t= y/4;
			y-=t;

		}
		cout << y<<endl;



	}
}
