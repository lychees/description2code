#define pb push_back

#define debug(x) cout<<">value ("<<#x<<") : "<<x<<endl;
#define fr(i,beg,end) for(i=beg;i<end;i++)
#define itfr(it,stl) for(it=stl.begin();it!=stl.end();it++)
#define PII pair<int,int>

using namespace std;
ll dp[1000001];
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    ll i,j,t,n;
    fr(i,1,1000001)
    {
        for(j=2;i*j<1000001;j++)
            dp[i*j]++;
    }
    fr(i,1,1000001)
    {
        dp[i]+=dp[i-1];
    }
    cin>>t;
    while(t--)
    {
        cin>>n;
        cout<<dp[n]<<endl;
    }
    return 0;
}
