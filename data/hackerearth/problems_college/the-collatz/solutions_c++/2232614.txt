#include <iostream>
using namespace std;
#define lli long long int
int main(){
    int t,n;
    cin>>t;
    while(t--){
    	cin>>n;
    	lli odd=0,even=0,steps=0;
    	while(n!=1){
    		if(n&1){
    			odd+=n;
    			n=3*n+1;
    		}
    		else{
    			even+=n;
    			n>>=1;
    		}
    		steps++;
    	}
    	odd++;
//    	cout<<odd<< " "<<even<<" "<<steps<<"\n";
    	if(odd%steps > even%steps){
    		puts("Odd Rules");
    	}
    	else if(odd%steps <even%steps){
    		puts("Even Rules");
    	}
    	else puts("Tie");
    }
    return 0;
}
