#include<bits/stdc++.h>
#define ll long long
#define sc(n) scanf("%d",&n)
#define scl(n) scanf("%lld",&n)
#define prt(n) printf("%d ",n)
#define mod 1000000007
using namespace std;
struct st
{
	int l;
	int r;
	ll c;
};
st myst[100001];
int main()
{
//	freopen("in.txt","r",stdin);
	int m,n,t,i,x,y,q,j;
	ll a;
	sc(t);
	while(t--)
	{
		ll arr[100001]={0};
		int temp[100001]={0};		
		sc(n);		
		sc(m);
		for(i=1;i<=m;i++)
		{
			sc(myst[i].l);
			sc(myst[i].r);
			scl(myst[i].c);
		}
		sc(q);
		for(i=0;i<q;i++)
		{
			sc(x);
			sc(y);
			temp[x]++;
			temp[y+1]--;
		}
		for(i=2;i<=m;i++)
		temp[i]+=temp[i-1];
		for(j=1;j<=m;j++)
		{
			arr[myst[j].l]+=temp[j]*myst[j].c;
			arr[myst[j].r+1]-=temp[j]*myst[j].c;
		}
		for(j=2;j<=n;j++)
		arr[j]=arr[j]+arr[j-1];
		for(j=1;j<=n;j++)
		cout<<arr[j]<<" ";
		cout<<"\n";
	}
return 0;
}
