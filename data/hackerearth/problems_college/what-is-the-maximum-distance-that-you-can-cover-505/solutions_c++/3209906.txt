#include <iostream>
#include<iomanip>
#include<math.h>
using namespace std;

int main()
{
    int t=0,i=0,j=0;
    cin>>t;
    double a[2][t];
    while(i<2)
    {
    	while(j<t)
    	{
    		
    		cin>>a[i][j];
    		
    		j++;
    	}
    	i++;
   		j=0;
}
    double fuel;
    cin>>fuel;
    double max_dist=0;
    i=0,j=0;
    double temp_max=0;
    while(j<t)
    {
    	temp_max=(fuel/a[1][j])*a[0][j];
    	if(temp_max>max_dist)
    		max_dist=temp_max;
    	j++;
    }
   // if(max_dist-0.000)
   max_dist=max_dist*1000;
   max_dist=floor(max_dist);
   max_dist=max_dist/1000;
    cout<<fixed<<setprecision(3)<<max_dist<<endl;
        return 0;
}
