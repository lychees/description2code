
#include<iostream>
#include<string.h>
using namespace std;
char s[1000];
int main()
{
    int i,j,t,n;
    cin>>t;
    while(t--)
    {
        cin>>s;
        n=strlen(s);
        int d=0;
        for(i=0;i<n;++i)
            if(s[i]>='0'&&s[i]<='9')
            d+=s[i]-'0';
        if(d%9==0)
            cout<<"0 or 9\n";
        else
            cout<<9-d%9<<"\n";
    }
}
