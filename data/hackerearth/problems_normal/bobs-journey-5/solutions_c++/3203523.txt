#include<bits/stdc++.h>
using namespace std;
int main()
{
       int t;
       cin>>t;
       while(t--)
       {
              map<char,int>mp;
              string s;
              int n;
              cin>>n;
              bool f=false;
              for(int i=0;i<n;i++)
              {
                    cin>>s;
                   if(mp[s[0]])
                   {
 
                        f=true;
                   }
                   else
                   {
                       mp[s[0]]++;
                   }
              }
              if(f==true)
              {
                  cout<<"NO"<<endl;
              }
              else
              {
                  cout<<"YES"<<endl;
              }
       }
       return 0;
 
}
