#include<bits/stdc++.h>
#define LL long long

using namespace std;
#define loop(i,a,b) for(i=a;i<b;++i)
#define plln(n) printf("%lld\n",n)
#define s3ll(n1,n2,n3) scanf("%lld%lld%lld",&n1,&n2,&n3)
#define s2ll(n1,n2) scanf("%lld%lld",&n1,&n2)
#define sll(n) scanf("%lld",&n)



#define MOD 1000000007
int main()
{
    LL T;
    sll(T);
    while(T--)
    {
        LL X,y,z;
        sll(X);
        if(X%8==0)
            plln(X/8);
        else if(X%2!=0)
            printf("-1\n");
        else
        {
            y=X/8;
            ++y;
            z=(y*8-X)/2;
            if(y<z)
                 printf("-1\n");
            else
                plln(y);
        }

    }
    return 0;
}
