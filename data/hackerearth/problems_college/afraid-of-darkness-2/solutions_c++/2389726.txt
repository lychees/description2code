#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		vector<int> ar(n);
		for(int i=0;i<n;i++)
			cin>>ar[i];
		int day=0;
		bool flag=1;
		while(1)
		{
			day++;
			sort(ar.rbegin(),ar.rend());
			for(int i=0;i<day;i++)
			{
				if(ar[i] == 0)
				{
					cout<<day-1<<endl;
					flag = 0;
					break;
				}
				ar[i]--;
			}
			if(flag == 0)
				break;
		}
	}
}
