#include<iostream>
#include<cstdio>
#include<cstring>
int main()
{
          int t;
          scanf("%d",&t);
          while(t--)
          {
          int i,j,n,c=0;
          scanf("%d",&n);
          int a[n];
          for(i=0;i<n;i++)
                    scanf("%d",&a[i]);
          for(i=0;i<n;i++)
          {
          for(j=i;j<n;j++)
          {
                    if(i<=j && a[i]==a[j])
                              c++;
          }
          }
          printf("%d\n",c);
          }
          return 0;
}
