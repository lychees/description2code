#include <cstdio>
#include <iostream>
using namespace std;

int main()
{
    int n, a[100], i, flag=0;
    //cout << "Hello World!" << endl;
    while(cin>>n){
    	//cout<<n<<endl;
    	for(i=0;i<n;i++)
    		scanf("%d",&a[i]);
    	for(i=n-1;i>=0;i--)
    	{
    		if(a[i]%2==0)
    		{printf("%d ",a[i]);
    		flag=1;}
    		
    	}
    	
    	if(flag==0)
    		cout<<"None to display";
    	flag=0;
    	printf("\n");
    }
    
    
    return 0;
}
