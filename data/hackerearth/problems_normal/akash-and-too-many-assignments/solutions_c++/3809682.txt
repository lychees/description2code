#include<bits/stdc++.h>
using namespace std;
int bit[1000001][31];
void update(int idx1,int idx2,int val,int n)
{
    idx1=idx1+1;
    while(idx1<=n)
    {
        bit[idx1][idx2]+=val;
        idx1+=idx1&(-idx1);
    }
}
long get_sum(int idx1,int idx2)
{
    long sum=0;

    while(idx1>0)
    {
        sum+=bit[idx1][idx2];
        idx1-=idx1&(-idx1);
    }
    return sum;
}
int main()
{
    int i,n,q,c;
    char s[1000001];
   cin>>n>>q;
   fflush(stdin);
   cin>>s;
  // cout<<q<<endl;
    for(i=0;i<n;i++) update(i,s[i]-'a',1,n);
    while(q--)
    {
        fflush(stdin);
        scanf("%d",&c);
        if(c==1)
        {
            int l,r,k;
            int ans=0;
            int j;
            fflush(stdin);
            cin>>l>>r>>k;
            for(j=0;j<26;j++)
            {
                ans+=get_sum(r,j)-get_sum(l-1,j);
                if(ans>=k) break;
            }
            if(ans>=k) printf("%c\n",j+'a');
            else cout<<"Out of range"<<endl;
        }
        else
        {
            int idx;
            char a;
            fflush(stdin);
            cin>>idx;
            fflush(stdin);
            cin>>a;
            update(idx-1,s[idx-1]-'a',-1,n);
            s[idx-1]=a;
            update(idx-1,s[idx-1]-'a',1,n);
        }
       // cout<<q<<endl;
    }
    return 0;
}
