def bs(arr, q):
	start = 0
	end = len(arr) - 1
	while True:
		i = (start + end) / 2
		if arr[i] == q:
			return True
		elif arr[i] < q:
			start = i + 1
		else:
			end = i - 1
		if start > end:
			return False

len_arr, len_q = map(int, raw_input().strip().split(" "))

arr.sort()
for i in range(len_q):

	if bs(arr, q):
