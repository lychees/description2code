#include <iostream>
#include <cstring>
using namespace std;
int dp[1010][1010];
int w[1010], v[1010];

int main()
{
	int T;
	cin >> T;
	while (T--)
	{
		memset (dp, 0, sizeof (dp));
		memset (w, 0, sizeof (w));
		memset (v, 0, sizeof (v));
		int N, W;
		cin >> N >> W;
		for (int i = 0; i < N; i++)
			cin >> w[i];
		for (int j = 0; j < N; j++)
			cin >> v[j];
		
		for (int i = 1; i <= N; i++)
		{
			for (int j = 1; j <= W; j++)
			{
				if (j < w[i-1])
					dp[i][j] = dp[i-1][j];
				else
					dp[i][j] = max(dp[i-1][j], v[i-1] + dp[i-1][j-w[i-1]]);
			}
		}
		cout << dp[N][W] << "\n";
		
	}
    return 0;
}
