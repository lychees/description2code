#include <cmath>
#include <iostream>
 
using namespace std;
 
int main()
{
  unsigned int nd, nz;   
  unsigned char *ca;     
  unsigned int j, n, q, temp,t;
  int i,flag;
  double p;
   cin>>t;
  while(t--)
  {
    cin >> n;
    if(n==0){cout<<1<<endl;continue;}
    n=2*n-1;
 
    if (n == 0) break;
 
    p = 0.0;
    for(j = 2; j <= n; j++)
    {
      p += log10((double)j);   
    }
 
    nd = (int)p + 1;
    ca = new unsigned char[nd];
    
    for (i = 1; i < nd; i++)
    {
      ca[i] = 0;
    }
    ca[0] = 1;
 
    p = 0.0;
    for (j = 3; j <= n; j+=2)
    {
      p += log10((double)j);   
      nz = (int)p + 1;         
      q = 0;                   
      for (i = 0; i <= nz; i++)
      {
        temp = (ca[i] * j) + q;
        q = (temp / 10);
        ca[i] = (char)(temp % 10);
      }
    }
 
    for( i = nd - 1,flag=0; i >= 0; i--)
    { if(int(ca[i])|flag)
      {cout << (int)ca[i];flag=1;}
    }
    cout << endl;
 
    delete []ca;    
  } 
  return 0;
}
