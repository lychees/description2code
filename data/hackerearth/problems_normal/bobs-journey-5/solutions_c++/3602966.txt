#include<iostream>
#include<bits/stdc++.h>
typedef long long ll;
using namespace std;
int main()
{
	
	ll t;
	cin>>t;
	while(t--)
	{
		ll n;
		cin>>n;
		ll hash[100005]={0};
		while(n--)
		{
	      string s;
	      cin>>s;
	      hash[int(s[0])]++;	
	    }
	    ll len;
	    int flag=0;
	    len=(sizeof(hash)/sizeof(hash[0]));
	    for(ll i=0;i<len;i++)
	    {
	    	
	    	if(hash[i]>1)
	    	{
	    		flag=0;	
			  cout<<"NO"<<endl;
			  break;
		    }
	    	else
	    	{
	    		flag=1;
			}
		}
		if(flag==1)
		cout<<"YES\n";
	} 
}
