#include<iostream>
#include<algorithm>
#include<vector>
#include<limits.h>

using namespace std;

int main(){
	int t,N,K,i,j,a[1000],cost,ans;
	
	cin>>t;
	while(t--){
		cin>>N>>K;
	
		for(i=0;i<N;i++){
			cin>>a[i];
		}
		ans = INT_MAX;
		for(i=1;i<=100;i++)
		{vector<int> v;
			for(j=0;j<N;j++){
				if(i>=a[j]){
					cost = (i-a[j])*3;
				}
				else{
					cost = (a[j]-i)*5;
				}
				v.push_back(cost);
			}
			sort(v.begin(),v.end());
			cost = 0;
			for(j=0;j<K;j++){
				cost += v[j];
			}
			ans = min(ans,cost);
		}
		cout<<ans<<endl;
	}
	
}
