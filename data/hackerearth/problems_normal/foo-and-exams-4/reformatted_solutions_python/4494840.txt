lines = input()


def find(x, mini, maxi):
    if maxi == mini:
        print maxi
    else:
        mid = (mini+maxi+(mini+maxi) % 2)/2
        val = x[0]*(mid**3)+x[1]*(mid**2)+x[2]*mid+x[3]
        if val > x[-1]:
            find(x, mini, mid-1)
        elif val < x[-1]:
            find(x, mid, maxi)
        else:
            print mid


for i in range(lines):
    x = map(int, raw_input().split())
    maxi = int(x[-1]**(1/3.0))
    mini = 0
    find(x, mini, maxi)
