#include<iostream>
#include<cstdio>
#include<cmath>
using namespace std;


int r_lt_10[] = {1, 1, 1, 3, 3, 15, 15, 105, 105, 105, 105};
int mod = 1000000007;
int POW(long long r, long long n)
{
    int ans = 1;
    while(n>0)
    {
        if(n&1)
            ans = (ans*r)%mod;
        n >>= 1;
        r = (r*r)%mod;
    }
    return ans;
}

int R(int n){
    if(n <= 10)return r_lt_10[n];
    int ret = pow(2, n/4+bool(n%4));
    ret = (ret*1LL*R(n/5+bool(n%5)))%mod;
    ret = (ret*1LL*R(n/10+bool(n%10)))%mod;
    return ret;
}

long long GCD(long long a,long long b)
{
    while(b^=a^=b^=a%=b);
    return a;
}
void solve(int n){
    int r_n = R(n);
    int maxi = 0;
    for(int i = 1;i<=5;i++)
        for(int j = 1;j<=5;j++)
            if(GCD(i, j) == 1)
                maxi = max(maxi, (int)POW(r_n*i, r_n*j));
    printf("%d\n", maxi);
}
int main()
{
	int t,n;
	scanf("%d",&t);    
	while(t--)
	{
        	scanf("%d",&n);
        	solve(n);	    
	}
	return 0;
}
