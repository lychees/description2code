#include <bits/stdc++.h>
#define maxi 100005
using namespace std;
long long v[maxi]={0};
void pre()
{
	v[0]=v[1]=0;
for(long long i=2;i<maxi;i++)
{
	if(v[i]==0)
	{ 
		for(long long j=2*i;j<maxi;j+=i)
		{
			v[j]=1;
		}
	}
}
}
int main()
{
	pre();
	long long t;
	cin>>t;
	while(t--)
	{
		long long x;
		cin>>x;
		do{
			x++;
		}while(v[x]==0);
		cout<<x<<endl;
	}
}
