#include <bits/stdc++.h>
using namespace std;

int main() {
	unsigned int m = 1000000007,ans;
    vector<vector<unsigned int> > pt(1002);
    pt[0].push_back(1);
    pt[1].push_back(1);
    pt[1].push_back(1);
    for(int i = 2; i < 1002; i++) {
    	pt[i].push_back(1);
    	for(int j = 1; j < i; j++) {
    		pt[i].push_back((pt[i-1][j]+pt[i-1][j-1])%m);
    	}
    	pt[i].push_back(1);
    }
    int t,n,p;
    scanf("%d", &t);
    while(t--) {
    	scanf("%d%d", &n,&p);
    	ans = 0;
    	for(int i = 0; i <= p; i++) {
    		ans = (ans + pt[n][i])%m;
    	}
    	printf("%u\n", ans);
    }
    return 0;
}
