#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ll int h[1000010];
	memset(h,0,sizeof(h));
	vector<ll int> a;
	ll int n,k;
	scanf("%lld%lld",&n,&k);
	ll int i,j,mc=0;
	map<ll int,ll int> m;
	for(i=0;i<n;i++)
	{
		scanf("%lld",&j);
		if(m[j]==1)//Removing duplicates
		continue;
		else
		{
			a.push_back(j);
			mc++;
		}
	}
	n=mc;
	ll int ans=0;
	sort(a.begin(),a.end(),less<ll int>());
	for(i=0;i<(n-2);i++)
	{
		for(j=(i+1);j<(n-1);j++)
		{
			//~ cout<<a[i]<<" "<<a[j]<<" "<<endl;
			vector<ll int>::iterator it;
			it=upper_bound(a.begin()+j+1,a.end(),k/(a[i]*a[j]));
			ans+=it-(a.begin()+j+1);
		}
	}
	printf("%lld\n",ans);
	return 0;
}
