#include <iostream>
using namespace std;
int main(){
	ios::sync_with_stdio(false);
	int T=0;
	
	cin >> T;
	
	while(T--)
	{
	 int N=0;
	 cin >> N;
	 long arr[N];
	 long sum=0;
	 
	 for(long i=0;i<N;i++)
	  arr[i] = 0;
	 
	 for(long i=0;i<N;i++)
	 {
	  cin >> arr[i];
	  sum+=arr[i];
	 }
	 
	 if(sum%N==0)
	  cout << "YES\n" << sum/N << " 0\n";
	 else
	  cout << "NO\n" << sum/N << " " << sum%N << endl;
	}
	
	return 0;
}
