#include <bits/stdc++.h>
using namespace std;

int main()
{
   int t,i,u,s,count,x,y;
   int len;
   cin>>t;
   
   while(t--)
   {
   	count=0;
   	x=0;y=0;
   	char arr[10001];
   	cin>>arr;
   	
   	u=0,s=0;
   	len=strlen(arr);
   	
   	for(i=0;i<len;i++)
   	{
   		if(arr[i]=='U')
   		u++;
   		else
   		s++;
   		
   	}
   	
   	if(s>=u+2 || u>=s+2)
   	{
   		cout<<"-1\n";
   		continue;
   	}
   	if(s>u)
   	{for(i=1;i<len;)
   		{
   			if(arr[i]!='U')
   			count++;
   			i=i+2;
   		}
   	}
   	else if(s<u)
   	{
   		for(i=1;i<len;)
   		{
   			if(arr[i]!='S')
   			count++;
   			i+=2;
   		}
   	}
   	else
   	{
   		for(i=0;i<len;)
   		{
   			if(arr[i]!='S')
   			x++;
   			i+=2;
   		}
   		for(i=0;i<len;)
   		{
   			if(arr[i]!='U')
   			y++;
   			i+=2;
   		}
   		if(x<y)
   		count=x;
   		else
   		count=y;
   	}
   	cout<<count<<"\n";
   }
   
    return 0;
}
