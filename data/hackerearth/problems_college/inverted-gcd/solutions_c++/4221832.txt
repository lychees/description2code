#include <iostream>
using namespace std;
#include <vector>
#define ll long long

ll cop(ll a,ll b)
{	while(b) 
		b^=a^=b^=a%=b;
	if(a==1)
    	return 1;
    return 0;
}

int main()
{   ll n,q,i,j,s=0;
    scanf("%lld",&n);
    ll z[n];
    for(i=0;i<n;++i)
    	scanf("%lld",&z[i]);
    for(i=0;i<n;++i)
        for(j=i+1;j<n;++j)
        {   if(z[i]>z[j])
            {   if(cop(z[i],z[j]))
                    s++;
            }
        }
    printf("%lld\n",s);
    return 0;
}
