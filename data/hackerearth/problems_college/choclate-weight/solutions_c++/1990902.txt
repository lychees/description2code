#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int N;
    cin>>N;
    int A[N];
    int i,j;
    vector<int> v;
    for(i=0;i<N;++i)
    cin>>A[i];
    for(i=0;i<N-1;++i)
    for(j=i+1;j<N;++j)
    v.push_back(A[i]+A[j]);
    int Q;
    cin>>Q;
    vector<int>::iterator it;
    for(;Q!=0;--Q)
    {
    	int k;
    	cin>>k;
    	for(i=0;i<N;++i)
    	if(A[i]==k){cout<<"I Like You"<<endl;break;}
    	if(A[i]==k){continue;}
    	for(it=v.begin();it!=v.end();++it)
    	if(k==*it){cout<<"I Like You"<<endl;break;}
    	if(k==*it){continue;}
    	cout<<"I Hate You"<<endl;
    }
}
