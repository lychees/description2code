#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int n,d,i,f=1;
    	cin>>n>>d;
    	for(i=0;i<20;i++)
    	{
    		if(d>=6)
    		{
    			if(n>=3)
    			{
    				d-=6;
    				n-=1;
    			}
    			else
    			{
    				f=0;
    				break;
    			}
    		}
    		else if(n*2<d)
    		{
    			f=0;
    			break;
    		}
    		else if(d<=0)
    			break;
    	}
    	if(f==0)
    		cout<<"No"<<endl;
    	else
    		cout<<"Yes"<<endl;
    }
    return 0;
}
