#include<iostream>
using namespace std;
int main()
{int t,i,n;
cin>>t;
while(t--)
{
cin>>n;
int a[n],b[n];
for(i=1;i<=n;i++)
{
    cin>>a[i];
    b[a[i]]=i;
}
int flag=0;
for(i=1;i<=n;i++)
{
    if(b[i]!=a[i])
    {
        flag=1; break;
    }
}
 
if(flag==0)
    cout<<"inverse"<<endl;
else cout<<"not inverse"<<endl;
}
 
 
}
