
#include<bits/stdc++.h>

using namespace std;

int main(){
int t;
cin>>t;

while(t--){
    vector< vector<int> > v(100005);
    int n;
    long long sum=0;
    cin>>n;
    for(int i=0;i<n;i++){
        long long num;
        cin>>num;
        sum+=num;
        sum%=n;
        //cout<<sum<<endl;
        v[sum].push_back(i);
    }
    int len=INT_MAX,l=0,r=0;
    if(v[0].size()!=0){
    //cout<<"here1";
        l=0;
        r=v[0][0];
        len=r-l;
    }
    for(int i=1;i<n;i++){//cout<<"wtf";
    if(v[i].size()==0) continue;
        for(int j=0;j<v[i].size()-1;j++){
            int diff=v[i][j+1]-v[i][j];
            if(diff<len){
                  // cout<<"here2";
                len=diff;
                l=v[i][j]+1;
                r=v[i][j+1];

            }
            else if(diff==len)
            {
                if(v[i][j]+1<l){
                      //  cout<<"here3";
                    l=v[i][j]+1;r=v[i][j+1];
                }
            }
        }
    }
    cout<<l+1<<" "<<r+1<<endl;//1 based indexing
    sum=0;//v.clear();
}

}
