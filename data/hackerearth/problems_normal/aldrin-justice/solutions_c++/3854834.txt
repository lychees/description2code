#include <bits/stdc++.h>
using namespace std;

int main()
{
	int tcases;
   
    cin>>tcases;
    while(tcases--){ 
    	int a,b,c,d,left,righ;
    	cin>>a>>b>>c>>d;
    	left=min(max(a,b),max(c,d));
    	righ=max(min(a,b),min(c,d));
    	if(left==righ){
    		cout<<"Point"<<endl;
    	}
    	else if(left<righ){
    		cout<<"Line"<<endl;
    	}
    	else{
    		cout<<"Nothing"<<endl;
    	}
    	
    	
    }
    return 0;
}
