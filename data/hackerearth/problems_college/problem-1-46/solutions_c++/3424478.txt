#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	vector<int> arr[26];
	string a;
	cin>>a;
	for(int i=0;i<a.length();i++)
	{
		arr[a[i]-'a'].push_back(i);
	}
	int q;
	cin>>q;
	while(q--)
	{
		int l,r;
		int ans=0;
		
		cin>>l>>r;
		l--;r--;
		int asd=(r-l+1);
		if(asd==1)
		{
			cout<<"NO"<<endl;
			continue;
		}
		bool checked = 1;
        for(int i=0;i<26;i++)
        {
        	if(arr[i].size()==0)
        	    continue;
        	int count=0;
        	int r1,l1;
            int index = upper_bound(arr[i].begin(),arr[i].end(),r) - arr[i].begin();
            if(index==0)
            {
            	 count=0;
            	 continue;
            }
            else
            	r1=index-1;
            index = lower_bound(arr[i].begin(),arr[i].end(),l) - arr[i].begin();
            if(index==arr[i].size())
            {
            	count = 0;
            	continue;
            }
            else
            {
            	l1=index;
            	if(r1<l1)
            	   continue;
            	count = r1-l1+1;
            }
            
            //count = max(0,count);
            //cout<<count<<endl;
            if((count%2)!=0)
                ans++;

        }
        if(asd%2==0)
        {
        	if(ans==0 || ans==2)
        		checked=1;
        	else
        		checked=0;
        }
        else
        {
        	if(ans==1)
        		checked=1;
        	else
        		checked=0;
        }
        if(checked)
          cout<<"YES"<<endl;
        else
        	cout<<"NO"<<endl;
	}
	return 0;
}
