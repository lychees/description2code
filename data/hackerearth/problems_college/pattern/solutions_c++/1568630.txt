#include <iostream>
#include <cstdlib>
using namespace std;

int main () {
	
	//int m;
	int n;
	cin >> n;
	if  (n > 2 && n <= 20)
	 {
 		for (int i = 0; i < n; i++) {
		 	cout << "*"; 
 		}
 			cout << endl;
 			
		int x = n-2;
 			
		while (x) 
			{
		for (int j = x; j > 0; j--)
		 	cout << " ";
			cout << "*" << endl;
		 	x--;
			 }
		for (int k = 1; k <= n; k++)
			cout << "*";
	 }
	 cout << endl;
	return 0;
}
