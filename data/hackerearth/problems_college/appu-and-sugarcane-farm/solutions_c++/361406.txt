#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int n;
	cin>>n;
	int ar[n];
	for(int i=0;i<n;i++)
		cin>>ar[i];
	for(int i=0;i<n-1;i++)
	{
		sort(ar,ar+n);
//		for(int f=0;f<n;f++)
//		cout<<ar[f]<<" ";
//		cout<<"\n";
		for(;ar[0]*2<=ar[n-1];ar[0]*=2);
		
	}
	for(int i=0;i<n-1;i++)
	{
		if(ar[i]!=ar[i+1])
		{
			cout<<"NO";
			return 0;
		}
	}
	cout<<"YES";
	return 0;
}
