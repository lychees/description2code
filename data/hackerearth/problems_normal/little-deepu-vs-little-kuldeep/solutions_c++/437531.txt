#include<bits/stdc++.h>
using namespace std;
#define MOD 1000000007
long long fact[2000002];

long long pow(long long a,int b)
{
	if(b==0)
	return 1;
	if(b==1)
	return a%MOD;
	long long tmp=pow(a,b/2);
	tmp=(tmp*tmp)%MOD;
	if(b%2==1)
		tmp=(tmp*a)%MOD;
	return tmp;
}

void preprocess()
{
	int i=0;
	fact[0]=1;
	fact[1]=1;

	for(i=2;i<=2000000;i++)
	{
		fact[i]=(fact[i-1]*i)%MOD;
	}
}

long long catalannumber(int n)
{
	return ((((fact[2*n]*pow(fact[n],MOD-2))%MOD)*pow(fact[n+1],MOD-2))%MOD)%MOD;
}
int main()
{
	int tc,n;
	preprocess();
	scanf("%d",&tc);
	while(tc-->0)
	{
		scanf("%d",&n);
		printf("%lld\n",catalannumber(n));
	}
	return 0;
}
