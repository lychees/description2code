#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long arr[50001];
		int a0,a1,c,n;
		cin>>a0>>a1>>c>>n;
		arr[0]=a0,arr[1]=a1;
		if(n==1)
		{arr[1]=0;}
		long long ans=(arr[0]*10+arr[1])%41;
		for(int i=2;i<n;i++)
		{
			arr[i]=(arr[i-1]*c+arr[i-2])%10;
			ans=(ans*10+arr[i])%41;
		}
		if(ans==0)
		cout<<"YES"<<endl;
		else
		cout<<"NO"<<endl;
	}
}
