def GCD(p,q):
	while q:
		if p<q:
			p,q=q,p
		p,q=q,p%q
	return p
def not_coprime(a,b):
	if a == b:
		return False
	x=max(a,b)
	y=min(a,b)
	if x%y == 0:
		return False
	if not GCD(y,x) == 1:
		return False
	else:
		return True
