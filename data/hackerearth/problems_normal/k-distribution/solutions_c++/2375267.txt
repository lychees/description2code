#include<iostream>
#include<algorithm>
#include<vector>

using namespace std;

int const N = 1e5 + 41;
int const MOD = 1e9 + 7;

int n, k, mn[N], dp[N], ans, v, a[N];
long long sum;

int bp(int d){
	int r = 1;
	for(int i=1;i<=d;++i){
		r *= 2;
		r %= MOD;
	}
	return r;
}

int main(){
	scanf("%d%d",&n,&k);
	ans = bp(n);
	for(int i=0;i<n;++i){
		scanf("%d",&a[i]);
		sum += a[i];
	}
	dp[0] = 1;
	for(int i=0;i<n;++i){
		for(int j=k-1;j>=0;--j){
			if(a[i] + j < k){
				dp[a[i] + j] += dp[j];
				dp[a[i] + j] %= MOD;
			}
		}
	}
	for(int i=0;i<k;++i){
		mn[i] = 2;
		if(sum - i < k){
			mn[i] = 1;
		}
	}
	for(int i=0;i<k;++i){
		ans -= dp[i] * mn[i] % MOD;
		ans %= MOD;
	}
	if(ans < 0){
		ans += MOD;
	}
	printf("%d\n",ans);

	return 0;
}
