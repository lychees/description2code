from itertools import product
from operator import xor
m,n=map(int,raw_input().split())
l=[[0 for x in range(n)] for y in range(m)]
for i in range(m):
	l[i]=map(int,raw_input().split())

# to get sum of all rows

rsum=map(sum,l)
csum=map(sum,zip(*l))
maxx=0
for x in product(rsum,csum):
	maxx=max(maxx,reduce(xor,x))

print maxx
