tc = int(raw_input())
for _ in range(tc):
    n, m = map(int, raw_input().split())
    lst = map(int, raw_input().split())
    s1 = set(lst[:n])
    s2 = lst[n:n+m]
    for i in s2:
        if i in s1:
            print "YES"
        else:
            print "NO"
            s1 |= {i}
