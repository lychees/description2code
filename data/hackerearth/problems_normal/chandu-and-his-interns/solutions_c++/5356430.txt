#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long a,b;
	cin>>a;
	while(a--)
	{
		cin>>b;
		int count=0;
		for(long long i=1;i*i<b;i++)
		{
			if(b%i==0)
			count+=2;
			if(count>=4)
			break;
		}
		if(count>=4)
		cout<<"YES\n";
		else
		cout<<"NO\n";
		
	}
	return 0;
}
