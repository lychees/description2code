    using namespace std;
    int main(){
    	int n,k;
    	scanf("%d%d",&n,&k);
    	long long int sum=0;
    	int arrx[k],arry[k];
    	for(int i=0;i<k;i++){
    		scanf("%d",(arrx+i));
    		if(arrx[i]<=(n/2))
    			sum+=(arrx[i]-1);
    		else
    			sum+=(n-arrx[i]);
    	}
    	for(int i=0;i<k;i++){
    		scanf("%d",(arry+i));
    		if(arry[i]<=(n/2))
    			sum+=(arry[i]-1);
    		else
    			sum+=(n-arry[i]);
    	}
    	printf("%lld\n",sum);
    	return 0;	
    }
