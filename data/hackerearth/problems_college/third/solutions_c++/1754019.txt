#include <iostream>
#include <cstring>
using namespace std;
int main ()
{
	long long T,no,a[10000];
	cin>>T;
while (T--)
{
	
	cin>>no;
	for (long i=0;i<no;i++)
		cin>>a[i];
	long long dp[no];
	memset (dp,0,sizeof dp);
	dp[0]=a[0];
	dp[1]=a[1];
	for (long i=2;i<no;i++)
		dp[i]=max (dp[i-1],a[i]+dp[i-2]);
	cout<<dp[no-1]<<"\n";
}
}
