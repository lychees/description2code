T = int(raw_input().strip())
while T > 0:
    line = map(int, raw_input().strip().split(" "))
    B = line[0]
    W = line[1]
    line = map(int, raw_input().strip().split(" "))
    X = line[0]
    Y = line[1]
    Z = line[2]
    if X < Y and Y > X+Z:
        print(B*X)+(W*(X+Z))
    else:
        if Y < X and X > Y+Z:
            print(W*Y)+(B*(Y+Z))
        else:
            print(B*X)+(W*Y)

    T = T-1
