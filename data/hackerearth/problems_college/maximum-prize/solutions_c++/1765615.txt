#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t;
   long long int n,k,i,cnt;
   cin>>t;
   while(t--)
   {
   	cin>>n>>k;
   	cnt=0;
   	long long int a[n],j=0;
   	for(i=0;i<n;i++)
   		cin>>a[i];
   	sort(a,a+n);
   	for(i=n-1;i>=0;i--)
   	{
   		if(j==k)
   		break;
   		cnt+=a[i];
   		j++;
   		
   		
   	}
   	cout<<cnt<<endl;
   }
    return 0;
}
