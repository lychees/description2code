void pt(int n) {
  char stack[30];
  int top = 0;
  if(n == 0) {
    putchar('0');
  } else {
    while(n > 0) {
      stack[top++] = n % 10 + '0';
      n /= 10;
    }
    while(top > 0) {
      putchar(stack[--top]);
    }
  }
  putchar('\n');
}
