#include <iostream>
#include <string>
#include <vector>
#include <map>

using namespace std;

void procesar(){
	int n, k, num, num1;
	bool existe = false;
	cin >> n >> k;
	int cant = 0;
	map<int, int> numeros;
	if(k == 2000000) 
		for(int i = 1 ; i <= n and !existe; i++){	
			cin >> num;
			if(num == 1000000) 
				cant ++; 
			if(cant > 1)
				existe = true;
		}
	else{
		for(int i = 1 ; i <= n and !existe; i++){
			cin >> num;
			if(num < k){
				num1 = k - num;
		
				if(numeros[num] == 0) numeros[num] = 1;
				else numeros[num] = numeros[num] + 1;
			
				if(num1 == num){
					if (numeros[num] > 1)
						existe = true;
					}else{
					if(numeros[num1] != 0){ 
						existe = true;
					}
				}
			}
		}
	}

	//cout << "\n n " << n << " k " << k << "\n";
	
	if(existe) cout << "YES\n";
	else cout << "NO\n";
}

int main(){
	procesar();
	return 0;
}
