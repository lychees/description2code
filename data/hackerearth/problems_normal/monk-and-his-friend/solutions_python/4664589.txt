t = input()
while t:
    count = 0
    l1, l2 = map(int, raw_input().split())
    n1, n2 = bin(l1), bin(l2)
    r1, r2 = n1[2:], n2[2:]
    if len(r1) > len(r2):
        k = len(r1) - len(r2)
        c = "0"
        c = c * k
        r2 = c + r2
    elif len(r2) > len(r1):
        k = len(r2) - len(r1)
        c = "0"
        c = c * k
        r1 = c + r1
    for i in range(len(r1)):
        if r1[i] != r2[i]:
           count = count + 1
    print count
    t = t - 1
