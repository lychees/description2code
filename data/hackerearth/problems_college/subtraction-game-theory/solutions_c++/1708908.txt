#include <iostream>
using namespace std;
 
int sod(long long a) {
 
	int x = 0;
	while(a > 0) {
		x += a % 10;
		a /= 10;
	}
	return x;
}
 
 
 
int main()
{
    int n, sum1 = 0, sum2 = 0;
    cin >> n;
    long long a;
    
   
    while(n--) {
    	
    	cin >> a;
    	sum1 += sod(a);
    	sum2 += a % 9;
    	
    }
    
    cout << (sum1 % 9) - (sum2 % 9) << endl;
    
    return 0;
}
