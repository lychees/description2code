#include <iostream>
using namespace std;

const int Maxn = 105;

int n, m;
char B[Maxn][Maxn];
int res;

int main()
{
	cin >> n >> m;
	for (int i = 0; i < n; i++)
		for (int j = 0; j < m; j++)
			cin >> B[i][j];
	for (int i = 0; i < n; i++) {
		bool is = false;
		for (int j = 0; j < m && !is; j++) {
			bool best = true;
			for (int l = 0; l < n && best; l++)
				best = B[i][j] >= B[l][j];
			is = best;
		}
		if (is) res++;
	}
    cout << res << endl;
    return 0;
}
