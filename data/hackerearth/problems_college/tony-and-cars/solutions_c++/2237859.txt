#include <iostream>
using namespace std;

int main()
{
    int i,d;
    long long int n,c,t,k;
    cin>>i;
    while(i--){
    	cin>>n;
    	c=0;d=1;t=k=9;
    	while(t<=n){
    		c+=k*d;
    		k=k*10;
    		t=(t+1)*10-1;
    		d++;
    	}
    	t=(t+1)/10-1;
    	t=n-t;
    	c+=t*d;
    	cout<<c<<"\n";
    }
    return 0;
}
