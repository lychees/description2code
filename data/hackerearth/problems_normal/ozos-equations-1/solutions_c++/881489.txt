#include<math.h>
#include<iostream>
using namespace std;
int s(long long i)
{
	int sum=0;
	long long  t=i;
	while(t)
	{
		sum+=t%10;
		t/=10;
	}
	return sum;
}
int main()
{
   	int t;
	cin>>t;
	while(t--)
	{
		long long l,r,n,tmp,ans;
		
		
			int f=0;
		cin>>n;
		l=(long long)sqrt(n)-82;
		r=(long long)sqrt(n)+1;
		for(long long i=l;i<=r;i++)
		{
			if(i>=0)
			{
		
				if(i*(i+s(i))==n)
				{
					ans=i;
					f=1;
					break;
				} 
			}
		}
		if(f)
			printf("%lld\n",ans);
		else
			printf("-1\n");
	}
	return 0;
}
