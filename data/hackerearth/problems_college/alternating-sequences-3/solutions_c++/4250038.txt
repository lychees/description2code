	#include <iostream>
	using namespace std;

	int absolute(int a)
	{
		if(a<0)
			return -a;
		else
			return a;
	}

	int main()
	{
		int n,a[5000],len[5000],i,j;
		cin >> n;
		for(i=0;i<n;i++)
		{
			cin >> a[i];
			len[i]=1;
			if(i!=0)
			{
				for(j=i-1;j>=0;j--)
				{
					if(((a[i]>0 && a[j]<0) || (a[i]<0 && a[j]>0)) && (absolute(a[j])<absolute(a[i])) && 1+len[j]>len[i])
					{
						len[i]=1+len[j];
					}
				}
			}
		}
		int max_len=1;
		for(i=0;i<n;i++)
		{
			//cout << len[i] << endl;
			if(len[i]>max_len)
				max_len=len[i];
		}
		cout << max_len;

		return 0;
	}
