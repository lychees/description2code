fib.append(1)
for i in xrange(2, 1001):
    fib.append(fib[i - 1] + fib[i - 2])
dp = []
for i in range(1001):
    temp = []
    cnt = 0
    for j in range(1001):
        if fib[j] % fib[i] == 0:
            cnt += 1
        temp.append(cnt)
    dp.append(temp)
for t in range(int(raw_input())):
    n = int(raw_input())
    ans = 0
    for i in range(n):
        ans = ans + dp[i][n - 1]
