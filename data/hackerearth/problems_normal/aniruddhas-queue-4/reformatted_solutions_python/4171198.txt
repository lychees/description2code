for i in range(input()):
    a = input()
    l = [int(x) for x in raw_input().split()]
    b = input()
    s = sum(l)
    b = b % s
    if (b == 0):
        for j in range((a-1), 0, -1):
            if (l[j] != 0):
                print(j+1)
                break
    else:
        c = 0
        for k in range(0, len(l)):
            if (l[k] > 0):
                c += l[k]
            if (c >= b):
                print(k+1)
                break
