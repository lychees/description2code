def binarysearch(a, t):
    low = 0
    high = len(a)-1
    while low <= high:
        mid = (low+high)/2
        if a[mid] >= t:
            low = mid+1
        else:
            high = mid-1
    return high


for _ in range(input()):
    n = input()
    ar = map(int, raw_input().split())
    b = map(int, raw_input().split())
    final = 0
    for i in range(len(ar)):
        x = binarysearch(b, ar[i])
        if x-i > final:
            final = x-i
