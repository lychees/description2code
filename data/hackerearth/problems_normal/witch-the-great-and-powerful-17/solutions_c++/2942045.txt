#include<iostream>
#include<algorithm>
using namespace std;

int main()
{
    long long int t;
    cin>>t;
    
    while(t--)
    {
        long long int a[3];
        cin>>a[0]>>a[1]>>a[2];
        
        sort(a,a+3);
        
        long long int p=a[2]-a[1];
        long long int q=a[1]-a[0];
        
        if(p>q)
            cout<<p-1<<endl;
        else
            cout<<q-1<<endl;
        
        
    }
}
