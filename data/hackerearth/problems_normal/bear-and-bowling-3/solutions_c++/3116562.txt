#include <bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;
ll int a[200010],f[200010];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int n;
	cin>>n;
	ll int i,t1,t2,ms=0,ans=0;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
	}
	f[0]=1;
	for(i=1;i<n;i++)
	f[i]=(f[i-1]*2)%mod; //Factorial
	for(i=0;i<=(n-1);i++)
	{
		t1=(n-(i+1));
		if(i==0)
		ans=(ans + (a[i]*f[t1])%mod)%mod;
		else
		{
			t2=(f[i-1]*(2ll+(i))%mod)%mod;
			ans=(ans+((t2*a[i])%mod * f[t1])%mod)%mod;
		}
	}
	cout<<ans<<endl;
	return 0;
}
