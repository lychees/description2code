#include <iostream>
using namespace std;

long A[50005], B[50005];
long Result[100010];
int main()
{
	int testcase;
	cin >> testcase;
	
	while(testcase)
	{
		int N, M;
		cin >> N >> M;
	
		for(int i=0;i<N; i++)
		{
			cin >> A[i];
		}
	
		for(int i=0;i<M; i++)
		{
			cin >> B[i];
		}

    	int i=0;
    	int j=0;
    	int k = 0;
    	while(i<N && j<M)
    	{
    		if(A[i]>=B[j])
    		{
    			//printf("%ld===\n", A[i]);
    			Result[k++] = A[i++];
    		}
    		else
    		{
    			//printf("%ld===\n", A[i]);
    			Result[k++] = B[j++]; 
    		}
    	}

    	while(i<N)
    	{
    		Result[k++] = A[i++];
    	}
    
	    while(j<M)
	    {
	    	Result[k++] = B[j++]; 
	    }
	
		for(int z=0; z<(N+M); z++)
		{
			cout << Result[z] << " " ;
		}
		cout << endl;
		testcase--;
	}
    return 0;
}
