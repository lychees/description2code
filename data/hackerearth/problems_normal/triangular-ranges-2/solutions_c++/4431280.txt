int main(){
    int q;
    sd(q);
    while(q--){
        lli l,r,i=1,sum=0LL,ans=0LL;
        slld(l),slld(r);
        vlli v;
        while(sum<r){
            sum+=(i++);
            v.pb(sum);
        }
        vlli::iterator ptr=v.begin();
        rep(i,0,v.sz){
            ans+=(ub(ptr,v.end(),r-v[i])-lb(ptr,v.end(),l-v[i]));

        }
        plld(ans);
    }
