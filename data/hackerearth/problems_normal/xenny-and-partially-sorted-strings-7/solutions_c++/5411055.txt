#include <bits/stdc++.h>
using namespace std;

#define ll long long int

ll t,n,k,m,i;

bool compare(string a,string b)
{
	for(i=0;i<m;i++)
	{
		if(a[i]<b[i])
		 return true;
		else if(a[i]>b[i])
		 return false;
	}
	
	return true;
	
	
}

int main()
{
	cin>>t;
	while(t--)
	{
		cin>>n>>k>>m;
		vector<string> v(n);
		for(i=0;i<n;i++)
		 cin>>v[i];
		
		sort(v.begin(),v.end(),compare);
		
		cout<<v[k-1]<<endl;
	}
    
    return 0;
}
