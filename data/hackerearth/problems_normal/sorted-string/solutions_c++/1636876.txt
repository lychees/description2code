using namespace std;

char s[101];
int cnt[26];
vector<pii>v;
int main()
{
    int t,i;
    scanf("%d",&t);
    while(t--)
    {
        v.clear();
        memset(cnt,0,sizeof(cnt));
        scanf("%s",s);
        int n=strlen(s);
        for(i=0;i<n;i++)
        {
            cnt[s[i]-97]++;
        }
        for(i=0;i<26;i++)
        {
            //printf("%d %d\n",cnt[i],i);
            v.push_back(pii(cnt[i],i));
        }
        sort(v.begin(),v.end());
        for(i=0;i<26;i++)
        {
            int x=v[i].F;
            int y=v[i].S;
            //printf("%d %d\n",x,y);
            while(x--)
            {
                printf("%c",y+97);
            }
        }
        printf("\n");
    }
    return 0;
}
