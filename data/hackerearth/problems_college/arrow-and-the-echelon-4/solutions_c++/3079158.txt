#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,no3,no5,n,i;
	cin>>t;
	while(t--)
	{
		cin>>n;
		if(n==1 ||n==2)
		{
			cout<<"-1\n";
		}
		else
		{
			no3=n-n%3;
				no5=n-no3;
			while((no3 % 3!=0 || no5 % 5!=0) && no3>=0)
			{
				no3=no3-3;
				no5=n-no3;
				//cout<<"fd\n";
			}
			if(no3 % 3 ==0 && no5% 5==0)
			{
				for(i=0;i<no3;i++)
				{
					cout<<"5";
				}
				for(i=1;i<=no5;i++)
				{
					cout<<"3";
				}
				//cout<<"fdd\n";
			}
			else
			cout<<"-1";
			cout<<"\n";		
			
		}
	}
	return 0;
}
