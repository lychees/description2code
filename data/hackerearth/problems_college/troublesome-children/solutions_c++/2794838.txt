#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	bool done[n];
	vector<int> arr;
	queue<int> queue1;
	for(int i=0;i<n;i++)
		queue1.push(i);
	memset(done,0,n*sizeof(bool));
	for(int i=0;i<n;i++)
	{
		int temp;
		cin>>temp;
		arr.push_back(temp);
	}
	int pos[n+1];
	for(int i=n;i>=1;i--)
	{
       int number=(arr[n-i]-1)%i;
       for(int j=0;j<number;j++)
       {
    	   int numb=queue1.front();
    	   queue1.pop();
    	   queue1.push(numb);
       }
       int numb1=queue1.front();
       queue1.pop();
       pos[n-i]=numb1+1;
	}
	for(int i=0;i<n;i++)
		cout<<pos[i]<<endl;
	return 0;
}
