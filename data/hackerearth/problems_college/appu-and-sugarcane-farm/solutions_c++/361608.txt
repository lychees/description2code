#include <iostream>
using namespace std;

int main() {
	int N;
	scanf("%d",&N);
	long long a[N];
	for(int i = 0; i<N; i++){
		scanf("%lld",&a[i]);
	}
	for(int i = 0; i<N; i++){
		while(a[i]>1){
			if(a[i]%2==0){
				a[i] = a[i]/2;
			}
			else{
				break;
			}
		}
	}
	int k = 0;
	for(int i = 0; i<N-1; i++){
		if(a[i]==a[i+1]){
			k++;
		}
		else{
			printf("NO");
			break;
		}
	}
	
	if(k==N-1){
		printf("YES");
	}
	
	return 0;
}
