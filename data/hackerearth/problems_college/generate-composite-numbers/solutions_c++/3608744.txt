#include<bits/stdc++.h>

using namespace std;

const int MAX = 1e4;
vector<bool> seen(MAX+5, false);

void seive(){
    seen[0] = seen[1] = true;
    for(int i =4;i<MAX+1;i+=2)
        seen[i] = true;

    for(int i = 3;i<MAX+1;i+=2){
        if(seen[i]==false){
            int j = 2;
            while(i*j<=MAX){
                seen[i*j] = true;
                j++;
            }
        }
    }
}

int main(){

	seive();
	int n;
	cin>>n;
	for(int i=2; i<=n; ++i)
	   if(seen[i]==true)
        cout<<i<<" ";
}
