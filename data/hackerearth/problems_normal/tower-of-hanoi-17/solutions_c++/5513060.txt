#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		int n;
		cin >> n;
		vector<pair<int, int>> v;
		for(int i=0;i<n;i++)
		{
			int r, h; 
			cin >> r >> h;
			v.push_back({r,h});
		}
		sort(v.begin(), v.end());
		
		long long int dp[n];
		dp[0] = v[0].second;
		long long int ans = dp[0];
		
		for(int i=1;i<n;i++)
		{
			dp[i] = v[i].second;
			for(int j=0;j<i;j++)
			{
				if(v[i].second > v[j].second && v[i].first > v[j].first)
					dp[i] = max(dp[i], dp[j]+v[i].second);
			}
			ans = max(ans, dp[i]);
		}
		cout << ans << endl;
	}
    return 0;
}
