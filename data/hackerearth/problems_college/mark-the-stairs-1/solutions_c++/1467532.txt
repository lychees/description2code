#include <iostream>
using namespace std;

int main()
{
    int t,x,y,a,b,c;
    cin>>t;
    while(t--)
    {
    	cin>>x>>y;
    	a=x%2;b=y%2;c=x-y;
    	if(a==0 && b==0 && (c==0 || c==2))
    	cout<<x+y<<endl;
       else if(a!=0 && b!=0 && (c==0 || c==2))
    	cout<<x+y-1<<endl;
    	else
    	cout<<"No Number"<<endl;
    }
    return 0;
}
