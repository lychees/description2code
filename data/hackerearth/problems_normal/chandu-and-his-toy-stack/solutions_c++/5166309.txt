#include <iostream>
#include <algorithm>
#define ll long long int
#define size 1000002
using namespace std;
ll a[size],b[size];
int main()
{
    ll i,t,n,x,y,sum;
    cin>>t;
    while(t--)
    {
    	sum=0;
    	cin>>n>>x>>y;
    	for(i=0; i<n; i++)
    	{
    		cin>>a[i]>>b[i];
    	}
    	sort(a,a+n);
    	sort(b,b+n);
    	for(i=0; i<n; i++)
    	{
    		if(a[i] < b[i])
    		{
    			sum = sum + abs(a[i]-b[i])*x;		
    		}
    		else if(a[i] > b[i])
    		{
    			sum = sum + abs(a[i]-b[i])*y;
    		}
    	}
    	cout<<sum<<"\n";
    }
    return 0;
}
