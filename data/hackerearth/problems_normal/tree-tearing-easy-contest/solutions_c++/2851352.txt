#include <bits/stdc++.h>

using namespace std;

vector<int> adj[10010];
int sub[10010], k, ans;

int dfs(int start, int parent)
{
	int i;
	sub[start] = 1;
	for(i=0; i<adj[start].size(); i++){
		if(adj[start][i]!=parent){
			sub[start]+=dfs(adj[start][i], start);
		}
	}
	if(sub[start]>=k){
		ans++;
		sub[start] = 0;
	}
	return sub[start];
}

int main()
{
	int n, i, j;
	scanf("%d%d", &k, &n);
	ans = 0;
	for(i=2; i<=n; i++){
		scanf("%d", &j);
		adj[j].push_back(i);
		adj[i].push_back(j);
	}
	dfs(1, -1);
	printf("%d\n", ans);
	return 0;
}
