#include <iostream>
using namespace std;

int main()
{
   long long int n;
    cin>>n;
    long long int a,b;
    long long int i,max=0,pos=0;
    for(i=0;i<n;i++)
    {
    	cin>>a>>b;
    	if(a>max)
    	{
    		max=a;
    		pos=b;
    	}
    }
    if(pos<0)
    cout<<"Right Corner";
    else if(pos>0)
    cout<<"Left Corner";
    else
    cout<<"Either Corner";
    return 0;
}
