#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	string str;
    	cin>>str;
    	int n,x,y;
    	cin>>n;
    	cin>>x>>y;
    	x--;
    	y--;
    	int count=0;
    	int c,i,j;
    	for(i=x;i<=y;i++)
    	{
    		c=0;
    		for(j=i;j<=y;j++)
    		{
    			c=(c*10 + str[j]-'0');
    			if(c%n==0)
    			count++;
    			c=c%n;
    		}
    	}
    	cout<<count<<endl;
    }
    return 0;
}
