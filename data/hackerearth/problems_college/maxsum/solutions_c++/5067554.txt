#include <bits/stdc++.h>
using namespace std;

long long int dp[10000001];
long long int a[10000001];

long long int f(long long int i,long long int n)
{
	if(dp[i] != -1)
		return dp[i];
		
	long long int ret = 0;
	
	if( i >= n)
	{
		dp[i] = 0;
		return dp[i];
	}
	
	if( i == n-1)
	{
		dp[i] = a[i];
		return dp[i];
	}
	
	
	ret = max(a[i] + f(i+2,n),f(i+1,n));
	
	dp[i] = ret;
	
	return ret;
		
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	memset(dp,-1,sizeof(dp));
    	long long int n;
    	cin>>n;
    	
    	for(int i = 0;i<n;i++)
    		cin>>a[i];
    	
    	cout<<f(0,n)<<endl;
    	
    }
    return 0;
}
