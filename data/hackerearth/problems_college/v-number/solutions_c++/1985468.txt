#include<bits/stdc++.h>
using namespace std;
int main()
{
	int test;
	cin>>test;
	while(test--)
	{
		string str;
		cin>>str;
		int n=str.length();
		if(n % 2!=0)
		cout<<"NO"<<endl;
		else
		{
			int s1=0,s2=0;
			for(int i=0;i<n/2;i++)
			s1+=str[i]-48;
			for(int i=n/2;i<n;i++)
			s2+=str[i]-48;
		//	cout<<s1<<" "<<s2;
			if(s1==s2)
			cout<<"YES"<<endl;
			else
			cout<<"NO"<<endl;
		}
	}
}
