#include <iostream>
#include <set>
#include <algorithm>
using namespace std;

set<int> Idl;
set<int> lc;
typedef set<int> :: iterator It;
int main(){
	int test,id,res;
	cin>>test;
	while(test--){
		int N,maxx=0;
		cin>>N;
		while(N--){
			cin>>id>>res;

			if(res){
				It itr = Idl.find(id);
				if(itr==Idl.end()){
				     lc.insert(id);
				     Idl.insert(id);
				}
			}

			else{
                int t = lc.size();
                maxx = max(maxx,t);
				lc.clear();
			}
		}
        int t = lc.size();
		maxx = max(maxx,t);
		lc.clear();
		Idl.clear();
		cout<<maxx<<'\n';
	}
	return 0;
}
