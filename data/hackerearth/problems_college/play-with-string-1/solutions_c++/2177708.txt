
#include <iostream>
#include <string>
#include <stdlib.h>

using namespace std;

int main()
{
    int test,len,f,x,y;

    cin >> test;

    string s;

    while(test--)
    {
        cin >> s;

        len=s.length();
        f=0;

        for(int i=0;i<len-1;i++)
        {
            x=abs(s[i]-s[i+1]);
            y=abs(s[len-i-1]-s[len-i-2]);

            if(x!=y)
            {
                f=1;
                break;
            }
        }

        if(f==0)
            cout << "Funny\n";
        else
            cout << "Not Funny\n";

    }
}
