#include <iostream>
using namespace std;

int main()
{
    int T;
    string n,m;
    cin>>T;
    while(T--)
    {
    	cin>>n>>m;
    	if(n==m)
    	{
    		cout<<"YES"<<endl;
    	}
    	else if((n=="2"&&m=="4")||(n=="4"&&m=="2"))
    	cout<<"YES"<<endl;
    	else
    	cout<<"NO"<<endl;
    }
    return 0;
}
