from collections import OrderedDict
t = int(raw_input())

while t:
	t-=1
	n = int(raw_input())
	ni = OrderedDict([])
	for i in range(n):
		item = raw_input()
		ni[item] = True
		
	l = ni.keys()
	l.sort()
	for item in l:
		print item
