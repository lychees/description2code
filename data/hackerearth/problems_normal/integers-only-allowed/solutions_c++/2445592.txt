#include <bits/stdc++.h>
using namespace std;

long long int findarea(long long int *X, long long int *Y, long long int points) {

  long long int  itisanswer = 0;
  long long int  i, j=points-1  ;

  for (i=0; i<points; i++) {
    itisanswer += (X[j]+X[i])*(Y[j]-Y[i]);
    	j = i;
   }

  return abs(itisanswer*.5);
}

int main()
{
	long long int n, i, j, itisanswer;
	cin >> n;
	long long int x[n], y[n];
	for(i = 0; i < n; i++) {
		cin >> x[i] >> y[i];
	}
	itisanswer = findarea(x, y, n);
	long long int findsum = 0;
	for(i = 0; i < n - 1; i++) {
			findsum += __gcd(abs(x[i] - x[i+1]), abs(y[i] - y[i+1]));

	}
		findsum += __gcd(abs(x[n-1] - x[0]), abs(y[n-1] - y[0]));
	itisanswer = itisanswer + 1 - findsum / 2;
	cout << itisanswer << endl;
    return 0;
}
