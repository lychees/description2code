#include<bits/stdc++.h>
using namespace std;

typedef long long lld;

lld dp_soln(lld price[],lld n)
{
	lld dp[100005];
	dp[0]=0;
	lld l;
	for(lld i=1;i<=n;i++)
	{   l=INT_MIN;
		for(lld j=0;j<i;j++)
		{
			 l=max(l,price[j]+dp[i-j-1]);
		}
		dp[i]=l;
		
	}
	return dp[n];
}

int main()
{
	lld t;
	cin>>t;
	
	while(t--)
	{
		lld n;
		cin>>n;
		lld price[100005];
		for(lld i=0;i<n;i++)
		cin>>price[i];
		
		cout<<dp_soln(price,n)<<"\n";
		
		
	}
	
	
	
}
