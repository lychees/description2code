#include <stdio.h>
#include <vector>
#include <string.h>
#include <math.h>

#include <iostream>
#include <map>

#include <set>
#include <string>
#include <queue>
using namespace std;
#define s(t) scanf("%d",&t)
#define ll long long int
#define mod 1000000007


int main()
{

	int n, i, j;
	int A[10005];

	int output_one[10006];
	int output_two[10006];

	s(n);
	for( i = 0; i< n ; i++)
	{
		s(A[i]);
	}

	sort(A, A + n);


	i = 0, j = n - 1;
	int  c = 0;

	while( i < j)
	{
		output_one[c] = A[i];
		output_one[c+1] = A[j];


		output_two[c] = A[j];
		output_two[c+1] = A[i];

		i++;
		j--;

		c += 2;
	}

	output_two[c] = output_one[c] = A[i];

		
	int sum_one = 0;	
	for( i = 1; i< n ; i++)
	{
		int diff = output_one[i] - output_one[i-1];

		if( diff < 1)
			diff *= -1;

		sum_one += diff;
	}


	int sum_two = 0;	
	for( i = 1; i< n ; i++)
	{
		int diff = output_two[i] - output_two[i-1];

		if( diff < 1)
			diff *= -1;

		sum_two += diff;
	}

	printf("%d\n",max(sum_two, sum_one) );
	return 0;
}
