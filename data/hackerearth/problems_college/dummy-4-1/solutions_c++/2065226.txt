#include <bits/stdc++.h>
using namespace std;
int dp[200][200];
int vis[200][200];
int path(int a[][100],int n,int m,int r,int c)
{

    if(n == r - 1 && m == c-1) {
        return 1;
    }
    if(n >= r || m >= c || n < 0 || m < 0 || a[n][m] == 1) {
        return 0;
    }

     if(vis[n][m] == 1) {
        return 0;
    }
    else {
        vis[n][m] = 1;
        int x = path(a,n+1,m,r,c);
        int y = path(a,n,m+1,r,c);
        int u = path(a,n-1,m,r,c);
        int v = path(a,n,m-1,r,c);
        vis[n][m] = 0;
        return (x+y+u+v);
    }

}






int main()
{
    int a[100][100];
    int i;
    int j;
    int r;
    int c;
    cin>>r;

    for(i = 0; i < r; i++) {
        for(j = 0; j < r; j++) {
            cin>>a[i][j];
        }
    }
    cout<<path(a,0,0,r,r);


}
