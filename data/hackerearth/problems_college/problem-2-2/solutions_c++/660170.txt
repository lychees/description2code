#include <iostream>
using namespace std;
 
int main() 
{
	int g ,b,i,k,j,min,max;
	cin>>g>>b;
	if(g==0)
		cout<<b<<endl;
		if(b==0 && g!=0)
		cout<<g<<endl;
		if(g!=0 && b!=0)
		{	max=g>b?g:b;
			min=g>=b?b:g;
			min++;
			i=max/min;
			if(max%min!=0)
			i++;
			cout<<i<<endl;
		}	
	return 0;
}
