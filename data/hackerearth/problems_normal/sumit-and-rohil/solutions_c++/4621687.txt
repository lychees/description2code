#include <bits/stdc++.h>
using namespace std;
 
bool check_func(string a, string b){
    int h=a.length();
    if(h!=b.length()) return false;
    if(a[0]!=b[0] || a[h-1]!=b[h-1]) return false;
    int alpha[26];
    for(int i=0;i<26;i++)alpha[i]=0;
    for(int i=0; i<h; i++){
        alpha[a[i]-'a']++;
        alpha[b[i]-'a']--;
    }
    for(int i=0; i<26; i++){
        if(alpha[i]!=0) return false;
    }
    return true;
}
 
int main() {
	string s;
	vector<string> vec;
	int n, ans; cin>>n;
	ans=n;
	for(int i=0; i<n; i++){
	    cin>>s; vec.push_back(s);
	    for(int j=0; j<i; j++){
	        if(check_func(vec[i], vec[j])){ ans--; break; }
	    }
	}cout<<ans<<endl;
	return 0;
}
