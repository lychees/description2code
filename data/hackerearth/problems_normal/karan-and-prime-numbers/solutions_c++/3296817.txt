using namespace std;

//ll sieve(ll n)
int checkprime(int N){
int count = 0;
for( int i = 1;i * i <=N;++i ) {
if( N % i == 0) {
if( i * i == N )
count++;
else
count += 2;
}
}
if(count == 2)
  return 1;
//cout << N << “ is a prime number.” << endl;
else
return 0;
}

//cout << N << “ is not a prime number.” << endl;
int main()
{
	int t;
	cin >>t;
	while(t--)
	{
	
	int a,n,i,ans=0;
	cin>>a >>n;
	for(i=a;i<=n;i++)
	{
		if(checkprime(i)==1)
		{
			ans+=i;
			
		}
	}
	cout<<ans<<endl;
}
}
