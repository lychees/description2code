#include <iostream>
#include <string.h>
using namespace std;
int a[100001] = {0};
int fun(int n) {
  int c = 0, x;
  for (int i = 0; i < n; i++) {
    x = (a[i] * 3) + c;
    a[i] = x % 10;
    c = x / 10;
  }
  if (c) {
    while (c) {
      a[n++] = c % 10;
      c = c / 10;
    }
  }
  return n;
}
int sum(int n) {
  int c = 0, x, i;
  x = a[0] + 2;
  a[0] = x % 10;
  c = x / 10;
  i = 1;
  while (c) {
    x = a[i] + c;
    a[i] = x % 10;
    c = x / 10;
    i++;
  }
  if (i > n)
    return i;
  else
    return n;
}
int main() {
  int t, n;
  cin >> t;
  while (t--) {
    int x = 1;
    cin >> n;
    memset(a, 0, sizeof(a));
    a[0] = 1;
    if (n > 120) {
      cout << "UNABLE TO FIRE\n";
      continue;
    }
    for (int i = 1; i <= n; i++) {
      if (i % 2)
        x = fun(x);
      else
        x = sum(x);
    }

    for (int i = x - 1; i >= 0; i--)
      printf("%d", a[i]);
    printf("\n");
  }
  return 0;
}
