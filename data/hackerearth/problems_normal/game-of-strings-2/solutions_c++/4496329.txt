#include <bits/stdc++.h>
using namespace std;
string s,p;
vector < long long int > T;
long long int create_table(long long int y)
{
	T.clear();
	long long int c=0;
	long long int pos=0;
	long long int l=p.length();
	long long int maxi=-1;
	T.push_back(0);
	int k;
	for(long long int i = 1;i<l;)
	{
		while(p[i]==p[pos]&&i<l)
		{   
			//d++;
			c++;
			T.push_back(c);
			if(c>maxi&&i<y)
			maxi=c;
			i++;pos++;	
		}
		if(i<l)
		{
		    c=0;
		    if(pos>0)
		    k=T[pos-1];
		    else
		    k=0;
			while(p[i]!=p[k])
			{
					if(k==0)
					break;
					k=T[k-1];
			}
			if(p[i]==p[k])
			{
				c=k;
				c++;
				T.push_back(c);
				if(c>maxi&&i<y)
				maxi=c;
				i++;
				pos=k+1;
			}					
			else
			{
				T.push_back(c);
				if(c>maxi&&i<y)
				maxi=c;
				i++;
				pos=0;
			}
		}
	}
	return maxi;
}							
int main()
{
	int t;long long int k,maxi,l;
	cin>>t;
	while(t--)
	{
		cin>>p>>k;
		l=p.length();
		maxi=create_table(k);
		if(maxi>=T[l-1])
		{
			if(T[l-1]!=0)
			for(long long int i=0;i<T[l-1];i++)
			{
				cout<<p[i];
			}
			else
			cout<<"Puchi is a cheat!";
			cout<<endl;
		}
		else
		{
			k=T[l-1];
			while(maxi<k)
			{
				k=T[k-1];
				if(k==0)
				break;
			}
			//cout<<maxi<<" "<<k<<endl;
			if(k!=0)
			for(long long int i=0;i<k;i++)
			{
				cout<<p[i];
			}
			else
			cout<<"Puchi is a cheat!";
			cout<<endl;
		}
	}
	return 0;
}
