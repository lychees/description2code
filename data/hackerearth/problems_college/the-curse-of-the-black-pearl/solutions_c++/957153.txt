#include<bits/stdc++.h>
using namespace std;
int main()
{
	ios::sync_with_stdio(false);
	int t;
//	cin>>t;
    t=882;
	while(t--) 
	{
	    char str[32801];
	    cin>>str;
	    int i=0,len=strlen(str);
	    i=len-1;
	    if(len==1)
	    {
	        cout<<"0"<<endl;
	        continue;
	    }
	        int ans=0;
	    while(str[i]!='1')
	        i--,ans++;
	    if(i!=0)
	    {
		    int con=1;
		    i--;
		    while(i>=0)
		    {
		      
		        if(i==0 && con==0)
		            break;
		        if(str[i]=='1')
		            con++;
		        else
		        {
		           
		            ans+=(con+1);
		             con=1;
		        }
		   //       cout<<"at "<<i<<" ans= "<<ans<<"  con "<<con<<endl;
		        i--;
		    }
		    if(con!=0)
		    {
		        ans+=(con+1);
		    }
	    }
	    cout<<ans<<endl;
	}
	return 0;
}
