int solvemin(int st1, int en1, int st2, int en2, int idx, int st3, vector<int>v,int n)
{
	vector<int>val;
	int arr[20];
	clr(arr);
	int k=idx,ans=0,temp=0;
	for(int i=st1;i<en1;i++)
	{
		arr[k]=v[i];
		k-=2;
	}
	for(int i=st2;i<en2;i++)
		val.pb(v[i]);
	do
	{
		int p=0;
		for(int i=st3;i>=1;i-=2)
			arr[i]=val[p++];
		for(int i=2;i<=n;i++)
			temp+=(i*abs(arr[i]-arr[i-1]));
		ans=max(ans,temp);
		temp=0;
	}while(next_permutation(val.begin(),val.end()));
	return ans;
}
int solvemax(int st1, int en1, int st2, int en2, int idx, int st3, vector<int>v,int n)
{
	vector<int>val;
	int arr[20];
	clr(arr);
	int k=idx,ans=0,temp=0;
	for(int i=st1;i>=en1;i--)
	{
		arr[k]=v[i];
		k-=2;
	}
	for(int i=st2;i<en2;i++)
		val.pb(v[i]);
	do
	{
		int p=0;
		for(int i=st3;i>=1;i-=2)
			arr[i]=val[p++];
		for(int i=2;i<=n;i++)
			temp+=(i*abs(arr[i]-arr[i-1]));
		ans=max(ans,temp);
		temp=0;
	}while(next_permutation(val.begin(),val.end()));
	return ans;
}
