#include<bits/stdc++.h>
using namespace std;
long long int t,n,i,j,k,l,m,s;
char a[500005];
int main()
{
    cin>>t;
    while(t--)
    {
    	scanf("%s",a);
    	scanf("%lld",&m);
    	n=strlen(a);
    	l=0;i=m;
    	while(i>0)
    	{
    		l++;
    		i/=10;
    	}
    	i=0;
    	k=0;
        while((i<(l+1))&&(i<n))
        {
        	k=k*10+(a[i]-48);
        	i++;
        }
        //cout<<k<" ";
        if(k>=m)
        k%=m;
        for(i=l+1;i<n;i++)
        {
        	k=k*10+(a[i]-48);
        	if(k>=m)
        	k%=m;
        }
    	cout<<k<<endl;
    }
    return 0;
}
