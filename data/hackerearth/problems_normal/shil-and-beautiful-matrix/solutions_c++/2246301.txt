template < typename T > T abs(T x)
{
	return x > 0 ? x : -x;
}

template < typename T > T sqr(T x)
{
	return x * x;
}

int mod = 1000000007, k[5005];
int fact[1000005];

int main()
{	
	int n, m;
	
	cin >> n >> m;
	
	for (int i = 1; i <= n; i++)
		for (int j = 1; j <= m; j++) {
			k[i ^ j]++;
		}
	
	int ans = 1;
	
	fact[0] = 1;
	
	for (int i = 1; i <= n * m; i++) {
		fact[i] = ((ll)fact[i - 1] * (ll)i) % mod;
	} 
	
	for (int i = 0; i <= 5000; i++) {
		ans = ((ll)ans * fact[k[i]]) % mod;
	}
	
	cout << ans;
	
	return 0;	
}
