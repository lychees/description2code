#include <bits/stdc++.h>

using namespace std;

#define ll long long
#define MOD 1000000007
#define M 100005

ll dp[M][2], f[M][2];

int main() {
	memset(dp, 0, sizeof(dp));
	memset(f, 0, sizeof(f));
	
    dp[0][0] = 1;
    dp[0][1] = 1;
    for (int i=1; i<=M; i++) {
    	dp[i][0] = dp[i - 1][1] % MOD;
    	dp[i][1] = ((dp[i - 1][1] % MOD) + (dp[i - 1][0] % MOD)) % MOD;
    }
    f[M][0] = 1;
    f[M][1] = 1;
    for (int i=M - 1; i>=0; i--) {
    	f[i][0] = f[i + 1][1] % MOD;
    	f[i][1] = ((f[i + 1][1] % MOD) + (f[i + 1][0] % MOD)) % MOD;
    }
    int T,N;
    cin >> T;
    while (T -- ) {
    	cin >> N;
    	ll res, ans = 0;
    	for (int i=0; i<N - 1; i++) {
    		ll res = 1;
    		if (i - 1 >= 0) res = ((res % MOD) * (dp[i - 1][1] % MOD)) % MOD;
    		if (i + 2 < N) 
    			res = ((res % MOD) * (f[M - (N - i - 2) + 1][1] % MOD)) % MOD;
    		ans = ((ans % MOD) + (res % MOD)) % MOD;
    	}
    	cout << ans << "\n";
    }
    return 0;
}
