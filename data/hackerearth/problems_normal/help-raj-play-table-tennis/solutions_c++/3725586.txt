# include<iostream>
# include<cstring>
# include<cmath>
using namespace std;

const int MOD=1e9+9;

int f1,f2;
int dp[32][32];
int solve(int x, int y)
{
	if(x==f1 && y==f2) return 1;
	if(x>f1) return 0;
	if(y>f2) return 0;
	if((x>=11 || y>=11) && abs(x-y)>=2) return 0;
	if(dp[x][y]) return dp[x][y];
	int ans=0;
	ans=(ans+solve(x+1,y))%MOD;
	ans=(ans+solve(x,y+1))%MOD;

	return dp[x][y]=ans;
}

int main()
{
	int t;
	cin>>t;
	for(int tc=1; tc<=t; tc++)
	{
		int n1,n2;
		cin>>n1>>n2>>f1>>f2;
		if(abs(f1-f2)<2 || (f1<11 && f2<11))cout<<"Case "<<tc<<": "<<-1<<endl;
		else
		{
			int ans=solve(n1,n2)%MOD;
			if(ans==0) ans=-1;
			cout<<"Case "<<tc<<": "<<ans<<endl;
		}
		memset(dp,0,sizeof(dp));
	}

    return 0;
}
