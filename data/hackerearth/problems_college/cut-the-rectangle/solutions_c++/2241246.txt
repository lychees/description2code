
#include <iostream>
#include <cstdio>
#include <queue>
#include <algorithm>
using namespace std;
long long n,m,k;
int main() {

    int t;
    cin>>t;
    while(t--){
	cin>>n>>m>>k;
	if (k>(n+m-2)) {
		cout<<-1<<endl;
		continue;
	}
	if (k>n-1 && k>m-1) {
		long long a = m/(k-n+2),b = n/(k-m+2);
		cout<<max(a,b)<<endl;
	} else if (k>n-1) {
		cout<< (m/(k+1))*n << endl;
	} else if (k>m-1) {
		cout<< (n/(k+1))*m << endl;
	} else {
		long long a = (m/(k+1))*n,b = (n/(k+1))*m;
		cout<<max(a,b)<<endl;
	}
    }
}
