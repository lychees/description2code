#include <iostream>
#include <algorithm>
#include <cstdio>

using namespace std;

int main()
{
    long long int z, t, m, n, x, ans;
    
    cin >> t;
    
    while (t--) {
    	cin >> z >> m >> n;
    	x = min((long long int)sqrt(z), max(m, n));
    	ans = 0;
    	
    	for (long long int i = 1; i <= x; i++) {
    		if (z % i == 0) {
    			if (i * i == z) {
    				if (i <= m && i <= n) {
    					ans++;
    				}
    			} else {
	    			if (i <= m && (z / i <= n)) {
	    				ans++;
	    			}
	    			if (i <= n && (z / i <= m)) {
	    				ans++;
	    			}
    			}
    		}
    	}
    	
    	cout << ans << endl;
    }
    return 0;
}
