#include <bits/stdc++.h>
using namespace std;

#define MOD 1000000007

void mini(long long int &a,long long int &b)
{
	long long int temp;
	if(a>b)
	{
		temp = a;
		a = b;
		b = temp;
	}
}

int main() 
{
	int t;
	long long int n1,n2,n3,ans;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%lld%lld%lld",&n1,&n2,&n3);
		mini(n1,n2);
		mini(n1,n3);
		mini(n2,n3);
		ans = (n1 * (n2-1)) % MOD;
		ans = (ans * (n3 -2))% MOD;
		printf("%lld\n",ans);
	}
	return 0;
}
