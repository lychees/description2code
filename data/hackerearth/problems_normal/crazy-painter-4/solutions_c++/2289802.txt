#include <iostream>
using namespace std;
long long int sumofn(long long int x,long long int y)
{
	long long int a,b,c,d;
	a=x+y;
	b=x-1;
	c=a*(a+1)/2;
	d=b*(b+1)/2;
	return(c-d);
}
int main()
{
  //  cout << "Hello World!" << endl;
   long int t;
    cin>>t;
  while(t--)
  {
   	  long long int n,m,d=0,rem=0,sum=0,sum1=0;
   	   cin>>n;
          long long int arr[26],i;
        	for(i=0;i<26;i++)
    		{
   	 		cin>>arr[i];
    		}
    	if(n==0)	
    	cout<<0<<endl;
    	else
    	{
    		m=n*12;
    		d=m/26;
    		rem=m%26;
      		if(d>0)
    		for(i=0;i<26;i++)
    		{
   	         sum=sum+sumofn(arr[i],d-1);
        	}
      		for(i=0;i<rem;i++)
      		 {
   	    		sum1=sum1+arr[i]+d;
       		}
           cout<<sum+sum1<<endl;
    	}
  }
    return 0;
}
