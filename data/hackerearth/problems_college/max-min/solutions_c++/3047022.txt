#include <bits/stdc++.h>
using namespace std;

int main()
{
    long int n,i;
    long long int k,a[100005];
    
    cin>>n>>k;
    for(i=0;i<n;i++){
    	cin>>a[i];
    }
    
    sort(a,a+n);
    
    i=0;
    long long int min=a[i+k-1]-a[i];
    
    for(i=1;i<=n-k;i++){
    	if(min>a[i+k-1]-a[i]) min= a[i+k-1]-a[i];
    }
    cout<<min<<endl;
    
    return 0;
}
