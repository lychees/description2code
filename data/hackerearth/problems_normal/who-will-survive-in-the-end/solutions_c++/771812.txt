#include<iostream>
#include<stdio.h>
#include<math.h>
using namespace std;
int main()
{
    int ar[28],i,j=1,t,n;
    for(i=0;i<28;i++)
    {
    ar[i]=j;
    j=j*2;
    }
    
    scanf("%d",&t);
    while(t--)
    {
    int l;
    scanf("%d",&n);
    for(i=27;i>=0;i--)
    {
    if(n-ar[i]>=0)
    {l=n-ar[i];break;}
    }
    printf("%d\n",2*l+1);
    }
    
    return 0;
}
