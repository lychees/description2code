#include <algorithm>
#include <iostream>
using namespace std;
int main() {
  int t, a[8] = {2, 3, 5, 7, 11, 13, 17, 19}, p, k, i, l, z;
  cin >> t;
  while (t--) {
    int n;
    k = 0;
    cin >> n;
    k = n / a[0];
    l = k;
    while (l != 0) {
      l = l / a[0];
      k = k + l;
    }

    p = k;
    z = 0;
    for (i = 1; i < 8; i++) {
      k = 0;
      k = n / a[i];
      l = k;
      while (l != 0) {
        l = l / a[i];
        k = k + l;
      }
      if (k >= p) {
        p = k;
        z = i;
      }
    }
    cout << a[z] << "\n";
  }
}
