#include <bits/stdc++.h>
using namespace std;

bool isSum(int arr[], int n)
{

	int *left = new int[n];
	int *right = new int[n];
	left[0]=0;
	int sum = arr[0];
	for(int i=1; i<n; i++)
	{

		left[i]=sum;
		sum+=arr[i];
		
	}
	right[n-1]=0;
	sum=arr[n-1];
	for(int i=n-2; i>=0; i--)
	{

		right[i]=sum;
		sum+=arr[i];
	}
	for(int i=0; i<n; i++)
	{

		if(left[i]==right[i]){delete left; delete right;return true;}
	}
		return false;
}

int main()
{

	int t;
	cin>>t;
	while(t--)
	{

		int n;
		cin>>n;
		int *arr = new int [n];
		for(int i=0; i<n; i++)cin>>arr[i];
		if(isSum(arr,n))cout<<"YES"<<endl;
		else cout<<"NO"<<endl;
		delete arr;
	}
	return 0;
}
