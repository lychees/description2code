#include <iostream>
#include<algorithm>
#include<string.h>
#define m 1000000007
using namespace std;

int main()
{
    int t,i,j,n;
    long long ma,s,p,ans;
    cin>>t;
    while(t--)
    {
    	cin>>n;
   	    s=0;
   	    long long a[100004];
        long long di[100004];
        memset(a,0,sizeof(a));
        memset(di,0,sizeof(di));
   	    ans=ma=0;
   	    for(i=0;i<n;i++)
   	    scanf("%lld",&a[i]);
   	    sort(a,a+n);
  	    for(i=n-2;i>=0;i--)
  	    di[i]=a[n-1]-a[i];
  	    for(i=0;i<n-1;i++)
  	    s=(s+di[i])%m;
    	for(i=n-2;i>=0;i--)
		{ ans=(ans+s)%m;
		  p=(ma+((i+1)*di[i+1])%m)%m;
		  ans=(ans-p+m)%m;
		  ma=(ma+di[i])%m;
		  p=1;
		  //cout<<ma<<" ";
        } 
        //cout<<ans<<" ";
		p=0;
		for(i=0;i<n;i++)
		p=max(a[i],p);
    	cout<<(p*ans)%m<<"\n";   
    }
}
