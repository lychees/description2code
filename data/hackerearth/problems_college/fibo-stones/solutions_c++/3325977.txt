#include<bits/stdc++.h>
using namespace std;
long long int fib(int n)
{
  /* Declare an array to store Fibonacci numbers. */
  long long int f[n+1];
  int i;

  /* 0th and 1st number of the series are 0 and 1*/
  f[0] = 0;
  f[1] = 1;

  for (i = 2; i <= n; i++)
  {
      /* Add the previous 2 numbers in the series
         and store it */
      f[i] = f[i-1] + f[i-2];
  }

  return f[n];
}

int main ()
{
   // cout << 10 % 19 << endl;

  int m;
  cin >> m;
  long long int sum = 0;
  for(int i = 0; i < m; i++) {
    long long int temp;
    cin >> temp;
    sum = sum + temp;
  }
 /// cout << fib(n + 2) - 1 << endl;
  cout << fib(n + 2) - 1 -  sum << endl;
  return 0;
}
