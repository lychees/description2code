#include <iostream>
using namespace std;
const int numL = 26;
int main()
{
	int K;
	cin >> K;
	string str;
	cin >> str;
	int size = str.size();
	long total = 0;
	bool *isPresent = new bool[numL];

	for(int i = 0; i < size; ++i)
	{
		for(int i = 0; i < numL; ++i)
		{
			isPresent[i] = false;
		}
		int count = 0;
		for(int j = i; j < size; ++j)
		{
			if (isPresent[str[j]-'a'] == false)
			{
				++count;
				isPresent[str[j]-'a'] = true;
			}
			if (count == K)
			{
				++total;
			}else if (count > K)
			{
				break;
			}
		}
	}
	cout << total << endl;
    //cout << "Hello World!" << endl;
    return 0;
}
