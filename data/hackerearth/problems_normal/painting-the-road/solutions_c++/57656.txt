{
    int i,j,k,l,m,n;
    int T,M,N,K,t1,t2,t3,t4,t5,q,p,r;
    long long ans,out;

	{
		cin>>L>>N;
		map<pair<int,int>, int >map1;
		map<pair<int,int>, int >::iterator it;
		
		int arr[260][260]={0};
		vector<vector<int> > idx(L+2,vector<int>());
		while(N--)
		{
			scanf("%d%d%d",&t1,&t2,&t3);
			//if(map1.find(make_pair(t1,t2))!)
			if(arr[t1][t2]==0)
			{
				arr[t1][t2]=t3;
				idx[t1].push_back(t2);
			}
			else
			{
				arr[t1][t2]=min(t3,arr[t1][t2]);
			}
		}
		for(i=1;i<=L;++i)
		{
			for(j=0;j<idx[i].size();++j)
			{
				for(k=i;k<idx[i][j];++k)
				{
					if(arr[0][k]!=0)
					{
						if(arr[0][idx[i][j]]!=0)
							arr[0][idx[i][j]]=min(arr[0][idx[i][j]],arr[0][k]+arr[i][idx[i][j]]);
						else
							arr[0][idx[i][j]]=arr[0][k]+arr[i][idx[i][j]];
					}
				}
			}
		}
		if(arr[0][L]==0)
			cout<<-1<<endl;
		else
			cout<<arr[0][L]<<endl;
	}

}
