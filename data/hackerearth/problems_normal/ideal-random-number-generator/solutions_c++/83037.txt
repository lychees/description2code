#define FOR(i,a,b) for(int i= (int)a; i< (int)b; ++i)
#define REP(i,n) FOR(i,0,n)
#define fill(a, x) memset(a, x, sizeof(a))

#define LL long long int 
#define PII pair<int,int> 
#define PB push_back
#define MP make_pair
#define INF 1000000000
#define mod 1000000007

int main()
{
   LL a,b,c; 
   cin>>a>>b>>c;

   LL d = a*b;
   LL num = 0;
   LL mn = min(a,b);
   LL mx = max(a,b);
   if( c <= mn ) {
      num = c*c;
      d = d*2;
   }
   else if(c > mn && c <= mx) {
      LL y = abs(c - mn);
      num = c*c - y*y;
      d = d*2;
   }
   else if(c > mn && c > mx){
      if(c >= a+b) num = d;
      else {
         LL y1 = abs(c-mx);
         LL y2 = abs(c-mn);

         num = c*c - y1*y1 - y2*y2;
         d = d*2;
      }
   } 
   {
      LL g = __gcd(num,d);
      num /= g;
      d /= g;
      cout << num << "/" << d << endl;
   }

   return 0;
}
