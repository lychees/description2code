#include <bits/stdc++.h>
using namespace std ;

#define MAXN 111111
#define MOD 1000000007

int n , s , t , a , b , dp[MAXN];
int solve(int n){
	if(n == 0)
		return 1 ;
	if(n <= 0)
		return 0 ;
	int &ret = dp[n] ;
	if(ret != -1)
		return ret ;
	ret = 0 ;
	ret = solve(n-1) + solve(n-s) ;
	if(ret >= MOD)
		ret -= MOD ;
	return ret ;
}

int main(){
	scanf("%d%d",&t,&s) ;
	memset(dp,-1,sizeof dp) ;
	for(int i=1;i<MAXN;i++){
		dp[i] = solve(i) ;
	}
	for(int i=1;i<MAXN;i++){
		dp[i] += dp[i-1] ;
		if(dp[i] >= MOD)
			dp[i] -= MOD ;
	}
	while(t --){
		scanf("%d%d",&a,&b) ;
		int ans = dp[b] - dp[a-1] ;
		if(ans < 0)
			ans += MOD ;
		printf("%d\n",ans) ;
	}
	return 0 ;
}
