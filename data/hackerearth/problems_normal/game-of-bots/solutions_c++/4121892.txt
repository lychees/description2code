#include <bits/stdc++.h>
using namespace std;
#define DEB(x) cout<< #x << " = " << x <<endl;
int main ()
{
    //freopen("input.txt","r",stdin);
    int t;
    scanf("%d",&t);
    while (t--){
        string ini,fi;
        cin>>ini;
        cin>>fi;
        int l=ini.size(),i,j;
        int cai=0,caf=0,cbi=0,cbf=0;
        vector < pair<char,int> > iv;
        for (i=0;i<l;i++){
            if (ini[i]=='A'){
                cai++;
                iv.push_back(make_pair('A',i));
            }
            if (ini[i]=='B'){
                cbi++;
                iv.push_back(make_pair('B',i));
            }
            if (fi[i]=='A')
                caf++;
            if (fi[i]=='B')
                cbf++;
        }
        int flag=1;
        if (cai!=caf || cbi!=cbf)
            flag=0;
        for (i=0,j=0;i<l && j<iv.size() && flag;i++){
            if (fi[i]=='A'){
                if (iv[j].first!='A' || iv[j].second<i)
                    flag=0;
                else
                    j++;
            }
            if (fi[i]=='B'){
                if (iv[j].first!='B' || iv[j].second>i)
                    flag=0;
                else
                    j++;
            }
        }
        if (flag==0)
            cout<<"No"<<endl;
        else
            cout<<"Yes"<<endl;
    }
    return 0;
}
