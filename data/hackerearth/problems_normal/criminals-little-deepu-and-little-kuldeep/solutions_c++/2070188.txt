#include <iostream>
#include <algorithm>
using namespace std;
#define LL long long
 
int main()
{
    int T;
    cin >> T;
    
    while(T--)
    {
    	int n;
    	
    	cin >> n;
    	LL a[n];
    	
    	int ans=n;
    	
    	for(int i=0 ; i<n ; i++)
    	scanf("%lld",&a[i]);
    	
    	sort(a,a+n);
    	
    	int j=1;
    	
    	for(int i=0 ; i<n-1 ; i++)
    	{
    		for( ; j<n ; )
    		{
    			if(a[i]<a[j])
    			{
    				ans--;
    				j++;
    				break;
    			}
    			j++;
    		}
    	}
    	
    	cout << ans << endl;
    	
    }
    return 0;
}
