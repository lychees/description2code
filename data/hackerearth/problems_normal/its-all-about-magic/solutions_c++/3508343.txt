#include <iostream>

using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int m;
    	cin>>m;
    	long long one = 1;
    	long long zero = 0;
    	long long temp = 1;
    	while(m--)
    	{
    		temp = one;
    		one = zero;
    		zero = zero + temp;
    	}
    	cout<<one<<" "<<zero<<endl;
    }
}
