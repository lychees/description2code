long long a, b, c, T;
int main()
{
    cin >> T;
    while (T--)
    {
        cin >> a >> b >> c;
        int stare = 0;
        long long lower = c;
        while (lower > 1)
        {
            if (stare == 0) {stare = 1; while (lower%b) lower++; lower/=b; }
            else  { stare = 0; while (lower%a) lower++; lower/=a; }
        }
        cout << (stare ? "Adam" : "Bob") << endl;
    }
    return 0;
}
