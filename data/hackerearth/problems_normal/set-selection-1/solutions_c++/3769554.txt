#include <bits/stdc++.h>
using namespace std;
#define s(n)  scanf("%d",&n)
int main()
{

    int n,sumarr[2][100001]={0},num,t,cur=0;
    sumarr[0][0]=1;sumarr[1][0]=1;
    // freopen("1input.txt", "r", stdin);
    s(n);

    for (int i = 0; i < n; i++) {
        s(num);
        for (int s = 0; s <= 100000; s++)
            if (sumarr[1-cur][s]) {
                sumarr[cur][s] = max(sumarr[cur][s],sumarr[1-cur][s]);
                sumarr[cur][s+num]=sumarr[1-cur][s+num]+1;
            }

        cur = !cur;
    }
    s(t);
    while (t--) {
        s(num);
        if (num > 100000){cout << "No" << endl;continue;}
        cout << (sumarr[1-cur][num]?"Yes":"No") << endl;
    }

    return 0;
}
