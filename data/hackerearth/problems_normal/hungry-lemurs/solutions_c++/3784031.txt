#include <iostream>
#include<limits.h>
using namespace std;
int main()
{
    int n,k;
cin>>n>>k;

if(n<=k)
{
	cout<<min(n,k-n);
}
else
{
	int b=INT_MAX;
	for(int i=1;i<=n;i++)
	{
		int tm;
		if(k-i>0)
			tm=k-i;
			else
			tm=-(k-i);
		int mb=(n/i)*i;
		int ma=mb+i;
		tm+=min(n-mb,ma-n);
		if(tm<b)
			b=tm;
	}
	cout<<b;
}
 
}
