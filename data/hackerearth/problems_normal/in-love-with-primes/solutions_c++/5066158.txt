#include <iostream>
#include <vector>
using namespace std;

bool p(int n){
	if(n==1)
		return 0;
	else if(n<=3)
		return 1;
	else if(n%2==0||n%3==0)
		return 0;
	int i=5;
	for(;i*i<=n&&n%i!=0;i+=2);
	if(i*i>n)
		return 1;
	return 0;
}
bool check(int n){
	vector<int> v(0);
	for(int i=1;i<n;i++)
		if(p(i)){
			v.resize(v.size()+1);
			v[v.size()-1]=i;
		}
	for(int i=0;i<v.size();i++)
		for(int j=i;j<v.size();j++){
			int temp=v[i]+v[j];
			if(temp==n)
				return 1;
			if(temp>n)
				break;
		}
		return 0;
}
int main()
{
	int T;
	cin>>T;
	while(T>0){
		int c;
		cin>>c;
		if(check(c))
			cout<<"Deepa"<<endl;
		else
			cout<<"Arjit"<<endl;
		T--;
	}
    return 0;
}
