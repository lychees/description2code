#include <iostream>
#include <vector>

using namespace std;
void display(char ch, int h);
int main(){
	char ch;
	int h;
	
	while(cin>>ch>>h){
		if(h>1){
			if(((int)ch>=65&&(int)ch+h<=91)||((int)ch>=97&&(int)ch+h<=123)){
				display(ch,h);
			}else
				cout<<"INVALID"<<endl;
		}else
			cout<<"INVALID"<<endl;
		cout<<endl;
	}
	return 0;
}

void display(char ch, int h){
	for(int i=1;i<=h;i++){
		int j=0;
		while(j<i){
			cout<<ch<<" ";
			j++;
		}
		ch+=1;
		cout<<endl;
	}
}
