#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;
int t;
string a , b;
int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */   
    cin >> t;
    while(t --){
        int car = 0;
        cin >> a >> b;
        int sz = min(a.size() , b.size());
        int one = 0;
        for(int i = sz - 1;i >= 0;i --){
            int x = (a[i] - '0'+ b[i] - '0' + one);
            one = (x > 9);
            car += one;
        }
        if(!car)	cout << "No";
        else	cout << car;
        cout << " carry operation";
        if(car > 1) cout << 's';
        cout << endl;
    }
    return 0;
}
