#include <iostream>
using namespace std;

int main()
{   ios::sync_with_stdio(0);
    int t;
    cin>>t;
    while(t--){
    	string s;
    	cin>>s;
    	int a[s.size()];
    		for(int i=0;i<s.size();++i)
    		  a[i]=1;
    	
    	for(int i=0;i<s.size();++i){
    		if(s[i]=='S')
    		  { a[i]=0;
    		  	if(i+1<s.size())
    		  	  a[i+1]=0;
    		  	 if(i-1>=0)
    		  	  a[i-1]=0;
    		  }
    		else if(s[i]=='B'){
    			a[i]=0;
    		    if(i-1>=0)
    		  	  a[i-1]=0;
    		  	 if(i-2>=0)
    		  	  a[i-2]=0;
    		}
    	  else if(s[i]=='D')
    	     a[i]=0;
    	}
    	int ans=0;
    	for(int i=0;i<s.size();++i)
    	  {ans+=a[i];}
    	printf("%d\n",ans);
    }
    return 0;
}
