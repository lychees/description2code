#include <bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
    ll t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	if(!n)
    	{
    		cout<<0<<endl;
    		continue;
    	}
    	ll i,arr[n];
    	for(i=0;i<n;i++) cin>>arr[i];
    	ll days=0,f=1;
    	while(days<=n)
    	{
    		sort(arr,arr+n,greater<ll>());
    		for(i=0;i<=days;i++)
    		{
    			arr[i]-=1;
    			if(arr[i]<0)
    			{
    				f=0;
    				break;
    			}
    		}
    		if(!f) break;
    		days+=1;
    	}
    	cout<<days<<endl;
    }
    return 0;
}
