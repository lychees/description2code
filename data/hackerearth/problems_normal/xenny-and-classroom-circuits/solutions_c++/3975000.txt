#include<stdio.h>
int main()
{
  int t,n,flag,cnt,cnt1,i;
  char s[1000001];
  scanf("%d",&t);
  while(t--)
  {
  cnt=cnt1=0;
    scanf("%d",&n);
    scanf("%s",s);
    flag=0;
    for(i=0;i<2*n;i++)
    {
      if(flag)
      {
        if(s[i]=='G')
         cnt++;
      }
      else
      {
        if(s[i]=='B')
         cnt++;
      }
      flag=!flag;
    }
    flag=1;
    for(i=0;i<2*n;i++)
    {
      if(flag)
      {
        if(s[i]=='G')
         cnt1++;
      }
      else
      {
        if(s[i]=='B')
         cnt1++;
      }
      flag=!flag;
    }
    cnt=cnt<cnt1?cnt:cnt1;
    printf("%d\n",cnt/2);
  }
  return 0;
}
