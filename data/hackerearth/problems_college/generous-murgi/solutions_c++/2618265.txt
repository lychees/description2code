#include<bits/stdc++.h>
using namespace std;
int main() {
	int t;
	long long int x,y,c,temp,i;
	scanf("%d",&t);
	while(t--) {
		scanf("%lld%lld",&x,&y);
		temp=x-y;
		c=0;
		if(temp==0) {
			printf("-1\n");
			continue;
		}
		if(y>x) {
			printf("0\n");
			continue;
		}
		for(i=1;i<sqrt(temp);i++) {
			if(x%i==y)
				c++;
			if(x%(temp/i)==y)
				c++;
		}
		if(i*i==temp&&x%i==y)
			c++;
		printf("%lld\n",c);
	}
	return 0;
}
