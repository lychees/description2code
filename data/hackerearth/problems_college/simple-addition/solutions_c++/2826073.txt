#include <stdio.h>
#include <iostream>
#include <queue>
#include <algorithm>
#define ll long long int
using namespace std;

ll dfs(int x, int y){
if(x < 1 || y < 1)
return 99999999;
if(x == 1)
return y - 1;
if(y == 1)
return x - 1;
if(x > y){
return (dfs(x - y, y) + 1);
}
else
return (dfs(y - x, x) + 1);
}

int main(){
ios_base::sync_with_stdio(false);
cin.tie(NULL);
int t;
ll l;
scanf("%d", &t);
ll ans = 99999999;
while(t--){
ans = 99999999;
scanf("%lld", &l);
if(l == 1){
printf("0\n");
continue;
}
for(int j=1;j<= l;j++){
ans = min(ans , 1 + dfs(j, l - j));
}
printf("%lld\n", ans);
}
return 0;
}
