#include <iostream>
using namespace std;
int main()
{
    int i, j, k, n;
    cin >> n;
    k = n + 10;
    int arr[k];
    for (i = 2; i < k; i++) {
    	arr[i] = 0;
    }
    arr[0] = arr[1] = 1;
    for (i = 2; i < k; i++) {
    	if (arr[i] == 0) {
    		for (j = i*i; j < k; j += i) {
    			arr[j] = 1;
    		}
    	}
    }
    for (i = 4; i <= n; i++) {
    	if (arr[i]) {
    		cout << i << " " ;
    	}
    }
    return 0;
}
