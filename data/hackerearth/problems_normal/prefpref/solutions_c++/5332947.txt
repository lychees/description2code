#include <bits/stdc++.h>
using namespace std;

int main()
{
    string a,b;
    cin>>a>>b;
    int na=a.length();
    int nb=b.length();
    int ans=0;
    int cur=0;
    for(int i=0;i<na;i++)
    {
    	int pos=b.find(a[i],cur);
    	// cout<<pos<<endl;
    	if(pos==-1) break;
    	cur=pos+1;
    	ans++;
    }
    cout<<ans<<endl;
}
