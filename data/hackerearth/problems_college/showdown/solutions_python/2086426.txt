#Max score between football players

FinanceTeamSkill = []
MarketingTeamSkill = []

def MaxScore():
	temp = 0
	total = 0
	for fp_skill in range(0, len(FinanceTeamSkill)+1):
		if(temp >= len(MarketingTeamSkill) or fp_skill >= len(FinanceTeamSkill)):
			break
		while FinanceTeamSkill[fp_skill] <= MarketingTeamSkill[temp]:
			temp += 1
			if(temp >= len(MarketingTeamSkill)):
				break
		while FinanceTeamSkill[fp_skill] >= MarketingTeamSkill[temp] and fp_skill < len(FinanceTeamSkill):
			if(FinanceTeamSkill[fp_skill] == MarketingTeamSkill[temp]):
				FinanceTeamSkill.pop(fp_skill)
				MarketingTeamSkill.pop(temp)
				total += 50
			elif (FinanceTeamSkill[fp_skill] > MarketingTeamSkill[temp]):
				FinanceTeamSkill.pop(fp_skill)
				MarketingTeamSkill.pop(temp)
				total += 100
			else:
				temp += 1
			if(temp >= len(MarketingTeamSkill) or fp_skill >= len(FinanceTeamSkill)):
				break
		if(fp_skill >= len(FinanceTeamSkill)):
				break
		fp_skill -= 1
	return total
