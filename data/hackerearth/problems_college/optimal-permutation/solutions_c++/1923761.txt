#include <iostream>
#include <stdlib.h>
using namespace std;


void n_even(long long  n)
{
    cout<<(n*n)/2-1;
    cout<<"\n";
}
void n_odd(long long n)
{
    cout<<2*(n/2)*(n/2+1)-1;
    cout<<"\n";
}
int main()
{
    long long T,N;
    cin>>T;
    while(T--)
    {
        cin>>N;  
        if(N==1)
        {
           	cout<<"0\n";
            continue;
        }
        if(N%2==0)
            n_even(N);
        else
            n_odd(N);
    }
    return 0;
}
