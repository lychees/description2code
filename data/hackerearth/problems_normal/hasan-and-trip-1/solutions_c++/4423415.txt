#include<bits/stdc++.h>
using namespace std;
long double dp[3003];
pair< pair<long double,long double> ,long double >arr[3003];
long double dist(pair<long double,long double> p1,pair<long double,long double> p2)
{
    long double ans;
    long double f = p1.first-p2.first;
    long double s = p1.second-p2.second;
    ans = f*f + s*s;
    return sqrt(ans);
}
int main()
{
    ios::sync_with_stdio(false);
    cout<<fixed;
    int i,j,n;
    long double x,y,f;
    cin>>n;
    for(i=1;i<=n;i++)
    {
        cin>>x>>y>>f;
        arr[i]=make_pair(make_pair(x,y),f);
    }
    dp[1]=arr[1].second;
    for(i=2;i<=n;i++)
    {
        dp[i]=LONG_MIN;
        for(j=1;j<i;j++){
            dp[i]=max(dp[i],dp[j]-dist(arr[i].first,arr[j].first));
        }
        dp[i]+=arr[i].second;
    }
    cout<<setprecision(6)<<dp[n]<<"\n";
    return 0;
}
