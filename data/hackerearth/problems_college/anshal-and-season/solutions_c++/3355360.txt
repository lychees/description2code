#include <iostream>
using namespace std;

int main()
{	string a,b;
	long long x[26]={0},z[26]={0},i;
    cin>>a>>b;
    if(a.length()!=b.length())
    {	cout<<"WRONG\n";
    	return 0;
	}
    for(i=0;i<a.length();++i)
    	x[a[i]-65]++;
    for(i=0;i<b.length();++i)
    	z[b[i]-65]++;
    for(i=0;i<26;++i)
    	if(x[i]!=z[i])
    	{	cout<<"WRONG\n";
    		return 0;
		}
	cout<<"CORRECT\n";
    return 0;
}
