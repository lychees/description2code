#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll long long int
#define pb push_back
#define mk make_pair
ll power(ll a, ll b) {
ll x = 1, y = a;
    while(b > 0) {
        if(b%2 == 1) {
            x=(x*y);
            if(x>mod) x%=mod;
        }
        y = (y*y);
        if(y>mod) y%=mod;
        b /= 2;
    }
    return x;
}
int mem[100002][11][2];
ll sum = 0;

void func(int s, int d)
{
	int i,j,l,x,y;
	sum = 0;
	for(i = 1; i <= 9; i++) {
		if(d == 0) {
			mem[i][i][1] = 1;
		}
		else {
			mem[i][i][0] = 1;
		}
	}
	for(i = 0; i <= s; i++) {
		for(j = 1; j <= 9; j++) {
			y = min(9,j+d);
			for(l = j; l <= y && i+l <= s; l++) {
				x = (d==(l-j))?1:0;
				mem[i+l][l][0||x] += mem[i][j][0];
				mem[i+l][l][0||x] %= mod;
				mem[i+l][l][1] += mem[i][j][1];
				mem[i+l][l][1] %= mod;
			}
		}
	}
	for(i = 1; i <= 9; i++) {
		sum  += mem[s][i][1];
		if(sum >= mod) sum %= mod;
	}
}

int main()
{
	int t,tc=1,i,s,d;
	scanf("%d",&t);
	while(t--) {
		memset(mem,0,sizeof(mem));
		scanf("%d%d",&s,&d);
		func(s,d);
		printf("Case #%d: %Ld\n",tc++,sum);
	}
	return 0;
}
