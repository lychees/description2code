#include <iostream>
#include<string.h>
using namespace std;

int main()
{
	int test, i, len, max; 
	int ind[26];
	char string[10005];
	cin>> test;
	while(test--)
	{
		cin>> string;
		len = strlen(string);
		for(i=0;i < 26;i++)
			ind[i] = 0;
		for(i=max=0;i < len;i++)
		{
			ind[string[i]-'a']++;
			if(max < ind[string[i]-'a'])
				max = ind[string[i]-'a'];
		}
		cout<< max << endl;
	}
    return 0;
}
