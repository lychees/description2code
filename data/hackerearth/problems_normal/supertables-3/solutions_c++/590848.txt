#include <iostream>
using namespace std;

long long hcf(int a, int b)
{int tmp;
	if(b<a){tmp=b;b=a;a=tmp;}
	if(b%a==0)return a;
	else {tmp=a;
	a=b%a;
	b=tmp;}
	return hcf(b,a);
}
	


long long llcm(int a , int b)
{
	return a*b/hcf(a,b);
}

int main()
{
   int t,a,b;
   long long int lcm,va,vb,v,n,ti,res,ans,te;
   cin>>t;
   int i,j;
   for(i=1;i<=t;i++)
   {ans=0;
   	cin>>a>>b;cin>>n;
   	lcm=llcm(a,b);
   	va=lcm/a;
   	vb=lcm/b;
   	v=va+vb-1;
   	ti=n/v;te=n%v;
   	ans=ti*lcm;
   	va=0;vb=0;
   	for(v=1;v<=te;v++){if((va+1)*a<(vb+1)*b)va++;
   	else vb++;}
   	if((vb)*b>(va)*a)ans=ans+(vb)*b;
   	else ans=ans+(va)*a;
   	cout<<ans<<endl;
   	
   }
}
