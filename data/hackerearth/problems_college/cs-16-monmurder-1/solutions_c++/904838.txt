#include <iostream>
#define MAX 1000000
using namespace std;

bool arr[1000001];
int ans[1000001];
void sieve()
{
	for(int i=2;i<=MAX;++i)
	{
		if(arr[i]==0)
		{
			for(int j=i;j<=MAX;j+=i)
			{
				arr[j]=1;
				int n=0,c=j;
				while(c%i==0)
				{
					n++;
					c/=i;
				}
				ans[j]^=n;
			}
		}
	}
}
int main()
{
    sieve();
    int t,h;
    cin>>t;
    while(t--)
    {
    	cin>>h;
    	if(ans[h]>0) cout<<"Printf{}"<<endl;
    	else cout<<"Pappu Bhujia"<<endl;
    }
    return 0;
}
