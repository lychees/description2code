#include <iostream>

#define BIGINT unsigned long long
#define mod 1000000007
#define MAX(x,y) ((x)>(y)?(x):(y))

using namespace std;

BIGINT dp[31][31];

BIGINT fun(int f, int h)
{

	if (dp[f][h] > 1) return dp[f][h];

	if (f == 0)
		return 1;

	if (h == 0)
	{
		return dp[f][h] = fun(f - 1, h + 1);
	}

	return dp[f][h] = fun(f - 1, h + 1) + fun(f, h - 1);
}

int main()
{
	int t;

	cin >> t;

	while (t--)  
	{
		int n;
		cin >> n;

		cout << fun(n,0) << endl;
	}

	return 0;
}
