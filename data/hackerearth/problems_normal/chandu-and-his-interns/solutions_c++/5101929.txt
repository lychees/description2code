#include<iostream>
#include<cmath>
using namespace std;
int main()
{
    long long n,x,i,c;
    cin>>n;
    while(n--)
    {
        cin>>x;
        c=0;
        int f=0;
        long long sq= (int)sqrt(x)+1;
        for(i=1;i<sq;i++)
        {
            if(x%i==0 && (i*i)!=x)
                c+=2;
            if(c==4)
            {
                cout<<"YES"<<endl;
                f=1;
                break;
            }
        }
        if(f==0)
            cout<<"NO"<<endl;
    }
    return 0;
}
