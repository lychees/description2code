#include <bits/stdc++.h>

using namespace std;

int main()
{

    int n , m , k , t;

    cin >> t;

    while(t--){

        cin >> n >> k;

        if(k > 81){

            cout << "-1\n";
            continue;
        }

        int start = k / 9 + (k % 9 != 0);

        int order = k % 9;
        if(order == 0) order = 9;

        int second = order - 1;

        if(second >= start) second++;

        for(int i = 0, j = 1; i < n; i++, j *= -1){

            if(j == 1) cout << start;
            else cout << second;
        }

        cout << "\n";
    }

    return 0;
}
