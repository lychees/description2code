from fractions import gcd
n = int(raw_input())
arr = [int(x) for x in raw_input().split()]

reqgcd = arr[0]

i = 1
while i < n:
    reqgcd = gcd(reqgcd, arr[i])
    i = i+1

ans = pow(arr[0], reqgcd, 1000000007)
i = 1

while i < n:
    ans = (ans*pow(arr[i], reqgcd, 1000000007)) % 1000000007
    i = i+1
print ans
