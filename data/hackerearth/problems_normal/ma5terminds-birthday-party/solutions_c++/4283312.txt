#include<bits/stdc++.h>
using namespace std;
#define MAX 1000006
int lp[MAX];
void siever(void)
{
		int i,j;
		for(i=2;i<MAX;++i)
		{
				if(lp[i] == 0)
				{
						for(j=i;j<MAX;j+=i)
						{
								lp[j]=1;
						}
						lp[i]=0;
				}
		}
		lp[1]=1;
		lp[4]=0;
		for(i=1;i<MAX;++i)
				lp[i]+=lp[i-1];
}

int main()
{
		siever();
		//cout<<"sivee-done\n";
		int t;
		scanf("%d",&t);
		while(t--)
		{
				int a,b;
				scanf("%d %d",&a,&b);
				int ans = lp[b]-lp[a-1];
				printf("%d\n",lp[b]-lp[a-1]);
		}
		return 0;
}
