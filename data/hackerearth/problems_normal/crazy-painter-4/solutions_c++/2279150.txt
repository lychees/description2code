#include <iostream>
using namespace std;

int main()
{
	long long int T,N;
	long long int arr[26];
	cin>>T;
	while(T--)
	{
		cin>>N;
		
		for(int i= 0;i<26;i++)
			cin>>arr[i];
			
		long long points = 12*N;
		
		long long int cycles = points / 26;
		long long int sum1 = (cycles*(cycles-1))/2;
		long long int sum2 = (cycles*(cycles+1))/2;
		
		int extra = points % 26;
		long long sum = 0;
		
		for(int i=0;i<extra;i++)
			sum+=(arr[i]*(cycles+1) + sum2);
		
		for(int i=extra;i<26;i++)
		{
			sum+=(arr[i]*cycles+sum1);
		}
		cout<<sum<<"\n";
	}
    return 0;
}
