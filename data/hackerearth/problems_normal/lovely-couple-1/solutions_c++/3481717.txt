bool isPrime[inf];
ll distPrimeFact[inf];
//seive of ero
void seive2(){
	isPrime[1]=false;
	for (int p=2; p<=inf; p++)
    {
        if (isPrime[p] == true)
        {
            for (int i=p*2; i<=inf; i += p){
                isPrime[i] = false;
            }
        }
    }
}
//seive of ero
void seive(){
	for (int p=2; p<=inf; p++)
    {
        if (distPrimeFact[p] == 0)
        {
            for (int i=p; i<=inf; i += p){
                distPrimeFact[i]++;
            }
        }
    }
}
int main(){
	std::ios::sync_with_stdio(false);
	ll t,a,b;
	fill(isPrime,inf,true);
	seive();
	seive2();
	in(t);
	for (int i = 0; i < t; ++i)
	{
		in2(a,b);
		ll x=(a*b)/(__gcd(a,b));
		ll y=distPrimeFact[x];
		//pr(y);

		}
		else{
			pr("No");
		}
	}
}
