#include <bits/stdc++.h>

typedef long long int li;

#define min(a,b) a<=b?a:b
#define max(a,b) a>=b?a:b
#define Fi(i, st, ft) for(i = st;i <= ft; ++i)
#define Fd(i, st, ft) for(i = st;i >= ft; --i)

using namespace std;

li t;
li arr[30];
string s;

int main()
{
	li i , j , a , b , c;
	cin >> t;
	while(t--) {
		cin >> s;
		memset(arr , 0 , sizeof(arr));
		li len = s.length();
		Fi(i , 0 , len-1) {
			arr[s[i]-'a'+1]++;
		}
		a = arr[1];
		b = 0;
		Fi(i , 2 , 26) {
			if(arr[i] != a) {
				b = 1;

			}
		}
		if(b == 0) {
			cout << "Yes\n";
		} else {
			cout << "No\n";
		}
	}
	return 0;
}
