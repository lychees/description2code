const int inf = (int)1e9;
const int mod = inf + 7;
const double eps = 1e-9;
const double pi = acos(-1.0);

string a[20000];
int t, n, k;
int cnt[30][26];

void solve() {
    cin >> n >> k;
    for(int i = 0; i < n; i++) cin >> a[i];
    memset(cnt, 0, sizeof cnt);
    for(int i = 0; i < n; i++) {
        for(int j = 0; j < k; j++) {
            cnt[j][a[i][j]-'a']++;
        }
    }
    string ans = "";
    for(int i = 0; i < k; i++) {
        int mn = inf;
        for(int c = 0; c < 26; c++) {
            int cur = 0;
            for(int t = 0; t < 26; t++) {
                cur += cnt[i][t] * abs(t-c);
            }
            mn = min(mn, cur);
        }
        for(int c = 0; c < 26; c++) {
            int cur = 0;
            for(int t = 0; t < 26; t++) {
                cur += cnt[i][t] * abs(t-c);
            }
            if(mn == cur) {
                ans += (char)(c+'a');
                break;
            }
        }
    }
    cout << ans << endl;
}

int main(){

    cin >> t;
    
    while(t--) solve();
    
    return 0;
}
