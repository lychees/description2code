#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	int h,m;
    	cin >> h >> m;
    	if(h>=12) h-=12;
    	int ma=m*6;
    	int ha=(h*60+m)/2.0;
    	//cout << ha << " " << ma << endl; 
    	cout << (int)(min(fabs(ha-ma),360-fabs(ha-ma))) << endl;
    }
    return 0;
}
