#include <iostream>
#include <string>
#include <cstring>
using namespace std;

typedef long long ll;

int main()
{
    ll T;
    cin >> T;
    
    for(ll t = 1; t <= T; t++)
    {
    	ll f[26];
    	memset(f, 0, sizeof(f));
    	
    	string s;
    	cin >> s;
    	for(auto c : s)
    		f[c - 'a']++;
    	
    	ll ans = 1;	
    	for(ll i = 0; i < 26; i++)
    	{
    		if(f[i] % 2 == 1)
    		{
    			ans = -1;
    			break;
    		}
    	}
    	
    	cout << ans << endl;
    }
    return 0;
}
