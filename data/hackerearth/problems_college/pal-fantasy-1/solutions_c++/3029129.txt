#include <iostream>
#include<stdio.h>
#include<string.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    while(t--){
    	char a[51];
    	int i;
    	scanf("%s",a);
    	int n=strlen(a);
    	
    	int d=0;
    	for(i=0;i<n/2;i++){
    		if(a[i]!=a[n-i-1]){
    			d=1;
    			break;
    		}
    	}
    	
    	if(d==0){
    		cout<<n<<endl;
    	}
    	else{
    		int l=n,j=0,te=-1;
    		for(i=0;i<n;i++){
    			if(a[i]!=a[n-j-1]){
    				l=l+i-te;
    				te=i;
    				j=0;
    			}
    			else j++;
    		}
    		cout<<l<<endl;
    	}
    	
    }
    return 0;
}
