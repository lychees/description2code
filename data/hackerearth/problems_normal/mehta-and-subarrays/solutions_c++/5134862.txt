#include <bits/stdc++.h>
using namespace std;
long long n,maxi,i,a[1000000],sum[1000000],ans,ans2,c,b[1000000],low,high,mid,flag;
int check(int x){
	flag=0;

	for(i=x;i<=n;i++){
			ans=sum[i];
     	ans2=sum[i-x];
     		if((ans-ans2) >=0)
     		{flag=1;break;}
	}
	if(flag==1)return 1;
	else return 0;
}
int main()
{
cin>>n;
maxi=0;
sum[0]=0;
for(i=1;i<=n;i++){
	cin>>a[i];
	sum[i]=sum[i-1]+a[i];
}
low=0;
high=n;
//ans=check(4);
//cout<<ans<<endl;
while(low<=high){
	mid=(low+high);
	mid=mid/2;
//	cout<<mid<<endl;
	ans=check(mid);
	if(ans==1){
		maxi=mid;
		low=mid+1;
	}
	else
	high=mid-1;
}

for(i=maxi;i<=n;i++){
	ans=sum[i];
	ans2=sum[i-maxi];
	if((ans-ans2) >=0)
	c++;
}
if(maxi==0)cout<<"-1";
else
cout<<maxi<<" "<<c;
    return 0;
}
