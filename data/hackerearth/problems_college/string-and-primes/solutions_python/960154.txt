def prime_check(n):
    i = 2
    c = 1
    while i <= n/2:
        if n % i == 0:
            c = 0
            break
        i += 1
    return c

t = int(raw_input())
while t > 0:
    s = raw_input()
    p=''
    l = len(s)
    i = 0
    while i < l:
        ans = prime_check(ord(s[i]))
        if ans != 1:
            p = p + s[i]
        i += 1
    if len(p) == 0:
        print 'NULL'
    else:
        print p
    t -= 1
