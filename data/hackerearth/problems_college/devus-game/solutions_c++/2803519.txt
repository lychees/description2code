#include <iostream>
using namespace std;

int main()
{
    int t, n;
    
    cin>>t;
    
    while(t--)
    {
    	cin>>n;
    	
    	int a[n];
    	bool visited[n];
    	for(int i = 0; i<n; i++)
    	{
    		cin>>a[i];
    		visited[i] = false;
    	}
    		
    	int i = 0;
    	while(i<n && visited[i] == false)
    	{
    		if(a[i] == -1)
    			break;
    		else
    		{
    			visited[i] = true;
    			i = a[i];
    		}
    	}
    	
    	if(a[i] == -1)
    		cout<<"Win"<<endl;
    	else
    		cout<<"Lose"<<endl;
    }
    
    return 0;
}
