#include<bits/stdc++.h>
using namespace std;
#define MAX 65
long long T;
long long dp[MAX];
long long n;
long long nn;
		


long long solve(long long a, long long b)
{
    long long sz = dp[a - 1];
    if (b < sz)
        return solve(a - 1, b);
    else if (b == sz)
        return 0;
    else
    {
        b -= sz + 1;
        return !solve(a - 1, sz - b - 1);
    }
}

int main()
{
    scanf("%d", &T);

    dp[1] = 1;
    for (long long i = 2; i < MAX; ++i)
	{
		dp[i] = dp[i - 1] * 2 + 1;
		//cout << i << " " << dp[i] << endl;
	}
        

    while(T--)
    {
        cin >> nn >> n;
		n--;
		if(dp[nn] < n)
		{
			cout << "-1\n";
			continue;
		}
		else
		{
			long long val = solve(nn, n);
			char ans;
			if(val == 0) ans = 'G';
			else ans = 'C';
			cout << ans << "\n";
		}
	}
}
