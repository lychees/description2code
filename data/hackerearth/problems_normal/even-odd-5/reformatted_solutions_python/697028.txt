#! /usr/bin/python

t = int(raw_input())

for i in xrange(t):
    s = raw_input()
    l = s.split()
    r = int(l[0])
    c = int(l[1])
    etotal = 0
    ototal = 0
    tmp1 = 0
    if (r > c):
        tmp1 = c
    else:
        tmp1 = r
    n1 = tmp1 / 2
    n2 = (tmp1 + 1)/2
    etotal = n1 * ((4 + (n1 - 1)*2)/2)
    ototal = n2 * ((2 + (n2 - 1)*2)/2)

    tmp2 = r - c
    if (tmp2 > 0):
        if (r % 2 != 0):
            n1 = tmp2/2
            n2 = (tmp2 + 1)/2
        else:
            n1 = (tmp2 + 1) / 2
            n2 = tmp2 / 2
        etotal += n1 * c
        ototal += n2 * c

    print etotal, ototal
