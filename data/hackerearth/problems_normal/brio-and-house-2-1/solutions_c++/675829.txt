#include <iostream>
#include <cmath>

int main()
{
    int T;
    int x1,x2,y1,y2,r;

    std::cin>>T;

    while(T--)
    {
        std::cin>>x1>>y1>>x2>>y2;
        std::cin>>r;

        //std::cout<<r<<std::endl;
        int d_x = x2-x1;
        int d_y = y2-y1;
        long d_r =sqrt(d_x*d_x + d_y*d_y);
        long D = x1*y2-x2*y1;

        long delta = (r*r)*(d_r*d_r) - (D*D);

        if(delta<0)
            std::cout<<"SAFE";
        else if(delta == 0)
            std::cout<<"JUST MISSED";
        else
            std::cout<<"REPLANNING";

        std::cout<<std::endl;
    }
    return 0;
}
