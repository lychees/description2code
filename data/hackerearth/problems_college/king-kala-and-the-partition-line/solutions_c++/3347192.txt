#include <bits/stdc++.h>
using namespace std;

int main()
{
	long long int t,i,j,k,l,m,n,a,b,c,p,first,second;
	cin>>t;
	while(t--)
	{
		cin>>n;
		p=0,first=0,second=0;
		cin>>a>>b>>c;
		for(i=0;i<n;i++)
		{
			cin>>j>>k;
			m=(a*j)+(b*k)+c;
			if(m>0)
			first++;
			else if(m<0)
			second++;
		}
		k=first;
		if(second>first)
		k=second;
		cout<<k<<"\n";
	}
}
