int main(){
	std::ios::sync_with_stdio(false);
	ll t,n;
	in(t);
	for (int i = 0; i < t; ++i)
	{
		in(n);
		ll a[n],b[n];
		for (int j = 0; j <n; ++j)
		{
			in(a[j]);
			b[j]=a[j];
		}
		sort(b,b+n);
		ll s=0,e=n-1;
		while(s<e){
			if(b[s]==a[s]){
				s++;
				if(b[e]==a[e]){
					e--;
				}
			}
			else{
				if(b[e]==a[e]){
					e--;
				}
				else{
					break;
				}
			}
			//pr2(s,e);
		}
		//pr2(s,e);
	    for (int j = 0; j <=(e-s)/2; ++j)
		{
			ll tmp=a[j+s];
			a[j+s]=a[e-j];
			a[e-j]=tmp;
		}
		bool areSame=true;
		for (int j = 0; j <n; ++j)
		{
			if(a[j]!=b[j]) areSame=false;
		}
		if(areSame && s!=e){
			pr2(s+1,e+1);
		}
		else{
			pr2(-1,-1);
		}
	}
    return 0;
}
