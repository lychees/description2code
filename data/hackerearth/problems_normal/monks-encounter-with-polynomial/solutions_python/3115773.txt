def func(a,b,c,x):
	return (a*x*x)+(b*x)+c
	
def search(a,b,c,k):
    left=0
    right=k
    ans=-1
    mid = -1
    while left<=right:
        mid = (right+left)/2
        val = func(a,b,c,mid)
        #print val
        if val >= k:
            ans = mid
            right = mid-1
        else:
            left = mid+1
    return ans

for t in xrange(input()):
	a,b,c,k = map(int, raw_input().split())
	print search(a,b,c,k)
