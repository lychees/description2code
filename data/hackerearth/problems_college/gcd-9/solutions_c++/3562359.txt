
#include <iostream>
#include <bits/stdc++.h>
using namespace std;
 
int gcdm(int a,int b){
    if(b==0)
        return a;
    else return gcdm(b,a%b);
}
 
int main() {
    int n,k,a[500000],i,curr=-1,max=-1;
	scanf("%d %d",&n,&k);
	for(i=0;i<n;i++)
	scanf("%d",&a[i]);
	for(int i=0;i<n;i++)
    {
        if(a[i]<k) 
        continue;
        int gcd=a[i],j=0;
        curr=1;
        for( j=i+1;j<n;j++)
        {    
            gcd=gcdm(gcd,a[j]);
            if(gcd <k)
            break;
            else
            curr++;
        }
            
        max=curr>max?curr:max;            
        if(max==n-i) break;
    }
        
    if(max==-1)
    printf("0");
    else
    printf("%d",max);
	return 0;
}
