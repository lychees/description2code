#include <bits/stdc++.h>
using namespace std;

int main(void) {

    int T = 0;
    scanf("%d", &T);
    while(T--) {

        string s;
        cin >> s;
        int count_z = 0, count_o = 0;
        for(int i = 0; i < s.length(); ++i) {
        	if(s[i] == '0')
        		++count_z;
        	else
        		++count_o;
        }
		int res = min(count_z, count_o);
		printf("%d\n", s.length() - 2 * res);
    }
    return 0;
}
