#include <iostream>
#include <string>
using namespace std;

void findIt(string str)
{
    long int ans = 0,count = 0;
    bool flag = false;
    for(long int i = 0 ;i < str.size(); i++)
    {
        if(str[i] == '.')
        {
            if(flag == false)
            {
                flag = true;
            }
            count++;
        }
        else
        {
            if(flag == true)
            {
                ans+=count-1;
                flag = false;
            }
            count = 0;
        }
    }
    if(count>0)
    {
        ans+=count-1;
    }
    cout << ans << endl;
}
int main()
{
    long int size, q;
    char ch;
    cin >> size >> q;
    string str;

    cin >> str;
    while(q--)
    {
        cin >> size >> ch;
        str[size-1] = ch;
        findIt(str);
    }
}
