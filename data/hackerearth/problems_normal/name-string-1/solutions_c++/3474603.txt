#include <iostream>
using namespace std;

int main() {
    int T;
    cin >> T;
    while(T--) {
        int R, K, V, fmax, smax, tmax, temp;
        cin >> R >> K >> V;
        if(R>K) {
            if(R>V) {
                fmax = R;
                smax = K;
                tmax = V;
            } else {
                fmax = V;
                smax = K;
                tmax = R;
            }
        } else {
            if(K>V) {
                fmax = K;
                smax = R;
                tmax = V;
            } else {
                fmax = V;
                smax = K;
                tmax = R;
            }
        }

        if(tmax>smax) {
            temp = tmax;
            tmax = smax;
            smax = temp;
        }

        if(fmax == smax) {
            cout << 2*fmax << endl;
        } else {
            cout << smax+smax+1 << endl;
        }
    }
}
