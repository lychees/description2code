#include <bits/stdc++.h>
using namespace std;
set<string> mp;
int shift,add;
void dfs(string s){
    //cout<<s<<endl;
	string temp=s;
	mp.insert(s);
	for(int i=0;i<s.length();++i){
		temp[(i+shift)%s.length()]=s[i];
	}
	if(mp.find(temp)==mp.end())
		dfs(temp);
	temp=s;
	for(int i=1;i<s.length();i+=2){
		temp[i]=(s[i]-'0'+add)%10+'0';
	}
	if(mp.find(temp)==mp.end())
		dfs(temp);
}
int main()
{
    int t;
    string s;
    cin>>t;
    while(t--){
    	cin>>s>>add>>shift;
    	mp.clear();
    	dfs(s);
    	cout<<*(mp.begin())<<endl;
    }
    return 0;
}
