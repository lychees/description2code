#include<stdio.h>
#include<stdlib.h>
#include<bits/stdc++.h>
#define maxi 5555555
#define mod 1000000007
long long int a[maxi],arr[maxi];
using namespace std;
void seive()
{
    long long int i,j;
    a[0]=1;
    a[1]=1;
    for(i=2; i<maxi; i++)
    {
        for(j=i; i*j<maxi; j++)
        {
            a[i*j]=1;
        }
    }
}
int main()
{
    int t;
  scanf("%d",&t);
    seive();
    int k=1,l=1,i,j;
    arr[0]=0;
    for(i=2;i<=maxi;i++)
    {
        if(a[i]==0)
        {
            if(a[k]==0)
            {
            arr[l]=i;
                l++;
            }
           k++;
        }
    }

    while(t--) {
       long long int n,m;
       scanf("%llu%llu",&n,&m);
       // cout<<arr[1]<<" "<<arr[2]<<endl;
     printf("%llu\n",((arr[n]%mod)*(arr[m]%mod))%mod);
       /* for(i=1;i<10000;i++)
        {
            cout<<arr[i]<<endl;
        }*/

    }
    return 0;
}
