#include <iostream>
using namespace std;
int hashFunc(int x)
{
	return x%10;
}
int main()
{
    bool a[10];
    int i;
   
    int t;
    cin>>t;
    while(t--)
    {
         int n,index,collision=0,x;
         cin>>n;
          for(i=0;i<10;i++)
    {
    	a[i]=false;
    }
         for(i=0;i<n;i++)
         {
         	cin>>x;
         	index=hashFunc(x);
         	if(a[index]==true)
         	{
         		collision++;
         	}
         	else a[index]=true;
         }
         cout<<collision<<"\n";
    }
    return 0;
}
