#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int a[n][2];
    	for(int i=0;i<n;i++)cin>>a[i][0]>>a[i][1];
    	int k[n][2];
    	for(int i=0;i<n;i++){k[a[i][0]-1][0] = i;k[a[i][1]-1][1]=i;}
    	vector<vector<int> > b;
    	for(int i=0;i<1001;i++)
        {  vector<int> p(1001);
           b.push_back(p);}
    	for(int i=0;i<n+1;i++)
    	 for(int j=0;j<n+1;j++)
    	  {if (i == 0 || j == 0)
              b[i][j] = 0;
           else if (k[i-1][0] == k[j-1][1])
              b[i][j] = b[i-1][j-1] + 1;
           else
              b[i][j] = max(b[i-1][j], b[i][j-1]);
    	  }
    	cout<<b[n][n]<<endl;
    }
    return 0;
}
