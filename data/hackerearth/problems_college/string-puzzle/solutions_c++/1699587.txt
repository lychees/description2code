#include<iostream>
#include<cstring>
using namespace std;
void swap (char *x, char *y)
{
    char temp;
    temp = *x;
    *x = *y;
    *y = temp;
}
void permutations(char *str,int i,int n)
{
    int j;
    if(i==n)
        cout<<str<<endl;
    else
    {
        for(j=i;j<=n;j++)
        {
            swap(str+i,str+j);
            permutations(str,i+1,n);
            swap(str+i,str+j);
        }
    }
}
int main()
{
    string str;
    int no,n,i;
    cin>>str;
    char *s;
    int len=str.size();
    s=new char[len];
    for(i=0;i<len;i++)
        s[i]=str[i];

    cin>>no;
    while(no--)
    {
        cin>>i>>n;
        permutations(s,i,n);
        cout<<endl;
    }
}
