a=[0]*1000001
c=[0]*1000001
i=2
while i<1000001:
    if p[i]==1:
        a[i]=i
        j=2
        while j*i<1000001:
            if a[j*i]==0:
                a[j*i]=i
            p[j*i]=0
            j+=1
    i+=1
i=0
while i<1000001:
    c[a[i]]+=1
    i+=1
