#include <iostream>
#include <math.h>
using namespace std;

int main()
{
    int n,k,count=0;
  	cin>>n>>k;
  	int mat[n][n];
	if (k > n) 
	 	return 0;

	for(int i=0;i<n;i++)
	{
	 	for(int j=0;j<n;j++)
	 	{
	    	cin>>mat[i][j];
	 	}
	}
 
   int stripSum[n][n];
   
   for (int j=0; j<n; j++)
   {
       int sum = 0;
       for (int i=0; i<k; i++)
          sum += mat[i][j];
       stripSum[0][j] = sum;
       for (int i=1; i<n-k+1; i++)
       {
            sum += (mat[i+k-1][j] - mat[i-1][j]);
            stripSum[i][j] = sum;
       }
   }
   for (int i=0; i<n-k+1; i++)
   {
      int sum = 0;
      for (int j = 0; j<k; j++)
           sum += stripSum[i][j];
      
     	int temp=sum;
    
    	int root(round(cbrt(temp)))	;
    	if(temp == root*root*root)
    	count++;
    	

      for (int j=1; j<n-k+1; j++)
      {
        sum += (stripSum[i][j+k-1] - stripSum[i][j-1]);
        temp=sum;
	
    	int root(round(cbrt(temp)))	;
    	if(temp == root*root*root)
    	count++;    
      }
   
}
	cout<<count;
	return 0;
}
