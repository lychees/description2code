#include <stdio.h>
#include <stdlib.h>
#include<iostream>
using namespace std;
 
void preprocess(int k, int Table[][2])
{
    int trans0, trans1;
    for (int state=0; state<k; ++state)
    {
        trans0 = state<<1;
	cout<<state<<" ";
        Table[state][0] = (trans0 < k)? trans0: trans0-k;	
	cout<<Table[state][0]<<" ";
        trans1 = (state<<1) + 1;
        Table[state][1] = (trans1 < k)? trans1: trans1-k;
	cout<<Table[state][1]<<endl;
    }
}

int main()
{
    int k;
    cin>>k;
    int (*Table)[2] = (int (*)[2])malloc(k*sizeof(*Table));
    preprocess(k, Table);
    return 0;
}
