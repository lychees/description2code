#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    string a;
    cin>>a;
    int i;
    pair <int,int> current;
    current={0,0};
    set <pair<int,int>> cord;
    cord.insert(current);
    int ans=0;
    
    for(i=0;i<a.length();i++){
    	if(a[i]=='D'){current.first-=1;}
    	if(a[i]=='U'){current.first+=1;}
    	if(a[i]=='L'){current.second-=1;}
    	if(a[i]=='R'){current.second+=1;}
    	//cout<<current.first<<" "<<current.second<<"\n";
    	if(cord.find(current)!=cord.end()){ans++;}
    	else{cord.insert(current);}
    	
    }
    cout<<ans;
    return 0;
}
