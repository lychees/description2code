#include<iostream>
#include<algorithm>
#include<cstdio>
using namespace std;
int main()
{
	int m, g, z, s, e, v;
	int i, j, k;
	int ans, tc;
	cin >> tc;
	while(tc--) 
        {
		scanf("%d %d %d %d %d %d", &m, &g, &z, &s, &e, &v);
		ans=0;
		for(i=0;i*100<=m;i++) 
                {
			for(j=0;i*100+j*125<=m && j*50<=g;j++) 
                        {
				k=min((m-i*100-j*125)/50, (g-j*50)/100);
				ans=max(ans, i*z+j*s+k*e);
			}
		}
		if (v>=ans) 
                {
		cout<<"-1"<<endl;
		}
		else 
                {
		cout << ans-v << endl;
		}
	}
return 0;
}
