#include<iostream>
#include <bits/stdc++.h>
#include<cstdio>
#define MAX_N 1000
#define MAX_M 1000
#include<cmath>
using namespace std;
void shot(int n,int m,long int D){
long int x[n],y[n];
long int v[m],w[m];
long int c=0;
for(int i=1;i<=n;i++){
cin>>x[i]>>y[i];
assert(-50000<=x[i] and x[i]<=50000);
assert(-50000<=y[i] and y[i]<=50000);
}
for(int j=1;j<=m;j++){
cin>>v[j]>>w[j];
assert(-50000<=v[j] and v[j]<=50000);
assert(-50000<=w[j] and w[j]<=50000);
}
for(int i=1;i<=n;i++){
for(int j=1;j<=m;j++){
if((abs(x[i]-v[j])+abs(y[i]-w[j]))<=D){
c++;
break;
}

}
}
if(c>(m/2)){

cout<<"YES"<<endl;


}
else{
cout<<"NO"<<endl;
}



}
int main(){
	ios_base::sync_with_stdio(0);
int t;
cin>>t;
assert(1<=t and t<=6);
int n,m;
long int d;
for(int i=1;i<=t;i++){
cin>>n>>m>>d;
assert(1<=n and n<=1000);
assert(1<=m and m<=1000);
assert(0<=d and d<=50000);
shot(n,m,d);
}
return 0;
}
