#include <iostream>
#include<vector>
#include <algorithm>
#include <cstdio>
using namespace std;

int main()
{ 
	vector<pair<int,int>>v;
	int n;
	cin>>n;
	while(n--)
	{
		int hs,hm,eh,em,th,te;
		scanf("%d:%d-%d:%d",&hs,&hm,&eh,&em);
		th=60*hs+hm;
		te=60*eh+em;
		v.push_back(make_pair(th,te));
	}
	bool flag=true;
		sort(v.begin(),v.end());
		for(int i=0;i<v.size()-1;i++)
		if(v[i].second>v[i+1].first)
		flag=false;
		if(flag)
		cout<<"Who needs a moderator?"<<endl;
		else
		cout<<"Will need a moderator!"<<endl;
		
	
   
    return 0;
}
