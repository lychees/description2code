#include<iostream>
#include<algorithm>
#include<vector>
#include<utility>
#include<map>

using namespace std;

typedef long long ll;

const ll MAXN = 1e5 + 10;

ll n, q;
ll a[MAXN], b[MAXN], d[MAXN];
map<ll, int>	cnt;

int main(){
	ios::sync_with_stdio(false);
	cin.tie(0);
	cin >> n >> q;
	ll sum = 0;
	for (ll i = 0; i < n; i++){
		cin >> a[i] >> b[i];
		d[i] = abs(a[i] - b[i]);
		sum += a[i];
		sum += b[i];
		cnt[d[i]]++;
	}
	sort(d, d + n);
	int m = n;
	m = unique(d, d + n) - d;
	bool fl = 0;
	for (int i = 0; i < m; i++)
		if (cnt[d[i]] % 2)	fl = 1;
	
	while (q--){
		ll x;	cin >> x;
		if (fl)
			cout << "NO\n";
		else{
			if (sum == x * 2)
				cout << "YES\n";
			else
				cout << "NO\n";
		}
	}
	return 0;
}
