#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cmath>

using namespace std;
long long arr[10004];

int main()
{
	int N;
	long long i,med,k;
	long long sum=0;
	scanf("%d",&N);
	for(i=0;i<N;i++)
	{
		scanf("%lld",&arr[i]);
		sum+=arr[i];
	}
	long long mean = sum/N;
	arr[0]-=mean;
	sum=0;
	for(i=1;i<N;i++)
	{
		arr[i]-=mean;
		arr[i]+=arr[i-1];
	}
	sort(arr,arr+N);
	
	if(N%2==0)
		k = (arr[(N/2)-1]+arr[N/2])/2;
	else
		k=arr[N/2];
	k=k*-1;
	for(i=0;i<N;i++)
	{
		sum+=(abs(k+arr[i]));
	}
	printf("%lld",sum);
}
