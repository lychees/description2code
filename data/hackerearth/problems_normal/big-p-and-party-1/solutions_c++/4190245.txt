#include<bits/stdc++.h>
#define pii pair<long long,long long>
#define fi first
#define se second
#define mp make_pair
#define vpi vector< pii >
#define pb push_back
#define ll long long int
#define mod 1000000007
#define gcd(a,b) __gcd(a,b)
#define sf(n) scanf("%lld",&(n))
#define pf(n) printf("%lld\n",(n))
#define setbits(n) __builtin_popcount((n))
#define pq priority_queue
using namespace std;
ll dist[1005];
vector< pii >v[1005];
void ini()
{
    fill(dist,dist+1004,mod);
}
void dij(int s)
{
    ini();
    dist[s]=0;
    pq< pii,vpi,greater<pii> >q;
    q.push(mp(s,0));
    while(!q.empty())
    {
        pii k=q.top();
        q.pop();
        int x=k.fi;
        int y=k.se;
        for(int i=0;i<v[x].size();i++)
        {
            int node=v[x][i].fi;
            if(dist[x]+v[x][i].se<dist[node])
            {
                dist[node]=dist[x]+v[x][i].se;
                q.push(mp(node,dist[node]));
            }
        }
    }
}
int main()
{
    int n,m;
    cin>>n>>m;
    for(int i=0;i<m;i++)
    {
        int a,b;
        cin>>a>>b;
        v[a].pb(mp(b,1));
        v[b].pb(mp(a,1));
    }
    dij(0);
    for(int i=1;i<n;i++)
    {
        if(dist[i]==mod)
            cout<<-1<<"\n";
        else
            cout<<dist[i]<<"\n";
    }
}
