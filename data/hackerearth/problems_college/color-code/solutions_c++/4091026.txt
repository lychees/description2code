#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
   
   int test,a[26][3],i,j,t[26][3],u[26][3];
   scanf("%d",&test);
   for(j=0;j<test;j++)
   {
   	 scanf("%d %d %d",&a[j][0],&a[j][1],&a[j][2]);
   	 for(i=0;i<3;i++)
   	 {  t[j][i]=a[j][i]/16;
   	    if(t[j][i]>9)
   	     t[j][i]=(t[j][i]%10)+85;
   	    else
   	     t[j][i]=t[j][i]+48;
   	     
   	    u[j][i]=a[j][i]%16;
   	    if(u[j][i]>9)
   	     u[j][i]=(u[j][i]%10)+85;
   	    else
   	     u[j][i]=u[j][i]+48;
   	      
   	 	 }
 
   }
   
   
   for(j=0;j<test;j++)
    { printf("Case %d:",j+1);
    	for(i=0;i<3;i++)
          printf(" %c%c",t[j][i],u[j][i]);
     printf("\n");     
     
   
    }
   
   
    return 0;
}
