int getString(char s[]) {

    char ch;
    int i=0;

    while( (ch = getchar()) != '\n'   &&   ch != EOF ) {
        if(ch!=' ' || ch!=32)
        {s[i] = ch;
        ++i;}
    }

    s[i] = '\0';

    fflush(stdin);

    return i;
}
