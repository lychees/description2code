#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	long long int n,ans;
    	cin>>n;
    	if(n==1||n==2||n==3){
    		cout<<1<<endl;
    		continue;
    	}
    	if(n==0){
    		cout<<0<<endl;
    		continue;
    	}
    	if(n%3==0) ans = n/3;
    	else if(n%3==1) ans = n/3 + (n/3)*2+1;
    	else ans = n/3 + ((n+1)/3)*2;
    	cout<<ans<<endl;
    }
    return 0;
}
