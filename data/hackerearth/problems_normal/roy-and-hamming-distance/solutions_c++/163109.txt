#include <iostream>
#include <string>
#define M 1000000007

using namespace std;

long long dp[102];

void pre()
{
     dp[0] = 0;
     dp[1] = 1;
     for ( long long i = 2; i <= 94; i++ ) {
         long long val = (dp[i-1] + dp[i-2])%M;
         val = (val*(i))%M;
         dp[i] = val;
     }
     return;
}

int main()
{
    
    pre();
    int n;
    cin >> n;
    string s;
    getline(cin,s);
    if ( n == 0 ) cout << "0" << endl;
    else if ( n == 1 ) cout << "1" << endl;
    else cout << dp[n-1] << endl;
    return 0;
}
