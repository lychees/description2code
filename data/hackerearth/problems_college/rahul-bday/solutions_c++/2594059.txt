#include <iostream>
#include <vector>

using namespace std;

void solve(){
	long long b, w, x, y, z;
	cin>>b>>w>>x>>y>>z;
	long long cost1= min(b*(y+z), b*x);
	long long cost2= min(w*(x+z), w*y);
	cout<<cost1+cost2<<endl;
}

int main()
{
    int t;
    cin>>t;
    while(t--) solve();
    return 0;
}
