#include<iostream>
using namespace std;
int main(){
int t;
cin>>t;
while(t--){
long long a,b;
cin>>a>>b;
int carry=0,count=0;
while(a>0&&b>0){
    if((a%10+b%10+carry)>9){
        count++;
        carry=(a%10+b%10+carry)/10;
    }
    else
        carry=0;
    a/=10;
    b/=10;
}
    if(a!=0){
        while(a>0){
            if((a%10+carry)>9){
                count++;
                carry=(a%10+carry)/10;
            }
            else
                carry=0;
            a/=10;
        }
 
    }
    else if(b!=0){
        while(b>0){
            if((b%10+carry)>9){
                count++;
                carry=(b%10+carry)/10;
            }
            else
                carry=0;
            b/=10;
        }
 
 
    }
    if(count==0)
    cout<<"No carry operation\n";
    else if(count==1)
        cout<<"1 carry operation\n";
    else
    cout<<count<<" carry operations\n";
}
}
