            #include <iostream>
                using namespace std;
                #include<bits/stdc++.h>
                int main()
                {
                    int L,R,max,i,j,c;
                    cin>>L>>R;
                    max=0;
                    for(i=L;i<=R;i++)
                    	for(j=L;j<=R;j++)
                    	{
                    		c=i^j;//Xor opertor for calculating Xor of 2 numbers
                    		if(c>max){max=c;}//Stores max value
                    	}
                    cout<<max;
                    return 0;
                }
