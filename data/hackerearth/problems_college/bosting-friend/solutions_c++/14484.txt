#include<cstdio>
#include<iostream>

using namespace std;
typedef long long ll;

ll gcd(ll a, ll b){
	
	ll r;
		
	r=a%b;
	while(r!=0){
		
		a=b;
		b=r;
		r=a%b;}
		
	return b;}
	
int main(){
	
	int t;
	ll n,pd,pg,i,temp,j;
	
	scanf("%d",&t);
	
	for(j=1;j<=t;j++)
	{
		
		scanf("%lld%lld%lld\n",&n,&pd,&pg);
		
		temp=gcd(pd,100);
		temp=100/temp;
		
		if(temp>n || (pg==100 && pd<100) || (pg==0 && pd>0) || n==0)
			printf("#%lld:Broken\n",j);
			
		else{
			printf("#%lld:Possible\n",j);
			}}
			
	return(0);}
