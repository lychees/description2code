#include <iostream>
#include <stdio.h>
#include <vector>
#define DIVI 1000000007
#define sz 1000001

using namespace std;

typedef long long int Lint;

int main( ) {
	
	Lint t, l, r, i, j, val, val2, sum;
	
	vector<bool> A( sz, true );
	vector<Lint> B( sz, 0 );
	vector<Lint> C( sz );

	A[ 0 ] = A[ 1 ] = false;

	for( i = 2;( i * i ) < sz;i++ ) {

		for( j = ( i * i );j < sz;j += i ) 
			if( A[ j ] ) {

				A[ j ] = false;
				B[ j ] = i;
			}
	}	
	
	sum = 0;
	for( i = 0;i < sz;i++ ) {
		sum = ( sum + B[ i ] ) % DIVI;
		C[ i ] = sum;
	}

	scanf("%lld", &t );

	while( t-- ) {

		scanf("%lld %lld", &l, &r );

		val = ( C[ r ] - C[ l ] + DIVI ) % DIVI;
		val2 = ( val + B[ l ] ) % DIVI;

		printf("%lld\n", val2 );
	}
	return 0;
}
