// This program is to find the factorial of the given number.
// Date     :       18 Aug 2016

# include <iostream>
using namespace std;
int factorial (int);
int main()
{
        int number = 0;
        //cout << "Enter the number you want to find the factorial for:" << endl;
        cin >> number;
        //cout << number;
        if (number < 0) cout << "You have entered a negative number" << endl;
        int result = factorial (number);
        cout << result << endl;
        return 0;
}

int factorial (int number)
{
        if (number == 0 || number == 1)
        {
                return 1;
        }
        else return number * factorial (number - 1);
}
