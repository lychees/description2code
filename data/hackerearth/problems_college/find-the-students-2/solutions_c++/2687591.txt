#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n;
    cin >> n;
    int a[n];
    for (int i = 0; i < n; i++)
    	cin >> a[i];
    sort(a, a + n);
    for (int i = 0; i < n; i++)
    {
    	int c = 0;
    	int min = a[i];
    	if (min > 0)
    	{	for (int j = i; j < n; j++)
    		{
    			a[j] = a[j] - min;
    			if (a[j] <= 0)
    				a[j] = -1;
    			c++;
    		}
    		cout << c << endl;
    	}
    }
    return 0;
}
