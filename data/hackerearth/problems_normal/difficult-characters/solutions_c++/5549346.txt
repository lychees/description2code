#include <bits/stdc++.h>

using namespace std;

int freq[26 + 1] = {0};
bool visited[26 + 1] = {false};

int main() {

	int T, n;
  	string s;

  cin >> T;

  while(T--) {

      memset(freq, 0 , sizeof freq);
      memset(visited , false, sizeof visited);


	    cin >> s;

	    n = s.size();

	    for (int i = 0; i < n; ++i)
		    freq[s[i] - 'a']++;

      for (int i = 0; i < 26; i++) {
        int mini = 1e9;
        char miniChar = '/';

        for (int j = 25; j >= 0; j--) {
          if(freq[j] < mini && !visited[j]) {
            mini = freq[j];
            miniChar = j + 'a';
          }
        }
        visited[miniChar - 'a'] = true;
        std::cout << miniChar << " ";
      }
      cout << endl;
  }
  return 0;
}
