#include<iostream>
using namespace std;
long method1(long y[],long n,long i)
{
    long height;
    height=2*i;
    return height;
}

long method2(long y[],long n,long i)
{
    long height;
    height=y[i-1]+1;
    return height;
}

int main()
{
    long w,n,i,x;
    long t;
    cin>>t;
    while(t--)
    {
        x=0;
        long length=0;
        cin>>w>>n;
        long a[n],y[n];
        y[0]=0;
        for(i=0;i<n;i++)
        {
            cin>>a[i];
        }
        for(i=1;i<=n;i++)
        {
            if(a[i-1]==1)
            {
                x=method1(y,n,i);
                y[i]=x;
            }
            if(a[i-1]==2)
            {
                x=method2(y,n,i);
                y[i]=x;
            }

            length=length+w+(y[i]-y[i-1]);
        }
        cout<<length<<endl;
    }
    return 0;
}
