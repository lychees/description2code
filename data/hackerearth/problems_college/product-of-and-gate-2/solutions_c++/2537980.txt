#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long t;
    scanf("%lld",&t);
    while(t--)
    {
        long long  a,b;
        scanf("%lld%lld",&a,&b);
        int x,i;
    x=a&a;
    for(i=a+1;i<=b;i++)
    {
    	 x=x&i;
    }
    printf("%lld\n",x);
    }
    return 0;
}
