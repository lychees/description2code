#include<stdio.h>
#include<string.h>
using namespace std;
int count_[26];
char str[10000];
int main()
{
    int cases;
    scanf("%d",&cases);
    for(int i=0;i<cases;i++)
    {
            for(int j=0;j<26;j++)
                count_[j]=0;
            scanf("%s",str);
            int length = strlen(str);
            for(int j=0;j<length;j++)
            {
                count_[str[j]-97]++;

            }
            int max=0;
            for(int j=0;j<26;j++)
            {
                if(count_[j] > max)
                    max=count_[j];

            }
            printf("%d\n",max);
    }
	return 0;
}
