#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	int i,a[n];
	for(i=0;i<n;i++)
	{
		cin>>a[i];
	}
	sort(a,a+n);
	int q;
	cin>>q;
	while(q--)
	{
		int x,y;
		cin>>x>>y;
		int ind1 = lower_bound(a,a+n,x)-a;
		int ind2 = upper_bound(a,a+n,y)-a;
		cout<<ind2-ind1<<"\n";
	}
}
