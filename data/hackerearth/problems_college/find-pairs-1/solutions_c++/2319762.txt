#include <iostream>
#include <cstdio>
#include <cstdlib>
using namespace std;
int main(){
    int T,limit=1000002,n,i,*arr,*c;
    long long int result;
    scanf("%d",&T);
    while(T--){
        scanf("%d",&n);
        arr=new int[n];
        c=new int[2*limit];
        for(i=0;i<2*limit;i++){
                c[i]=0;
        }
        for(i=0;i<n;i++){
            scanf("%d",&arr[i]);
            c[limit+arr[i]]++;
        }
        result=0;
        for(i=1;i<2*limit;i++){
            result+=((c[i]*(c[i]+1))/2);
        }
        printf("%lld\n",result);
    }
return 0;
}
