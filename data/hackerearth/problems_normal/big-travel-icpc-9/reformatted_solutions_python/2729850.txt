sum = 0
q = 1000000007
n = int(raw_input())
a1 = []
a2 = []
for i in range(n):
    n1, n2 = raw_input().split()
    a1.append(int(n1))
    a2.append(int(n2))

a1.sort()
a2.sort()

for i in range(n-1, -1, -1):
    sum = (sum+(n-1-2*(n-1-i))*a1[i]) % q
    sum = (sum+(n-1-2*(n-1-i))*a2[i]) % q

print(sum) % q
