#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    vector<vector<int>> v(n);
    int x,y,d;
    for(int i=0;i<n;i++)
    {
    	scanf("%d %d %d",&x,&y,&d);
    	v[i].push_back(x);
    	v[i].push_back(x+d);
    }
    sort(v.begin(),v.end());
    /*for(int i=0;i<n;i++)
    {
    	cout<<v[i][0]<<" "<<v[i][1]<<"\n";
    }*/
    int cleft=v[0][0];
    int cright=v[0][1];
    int ans=0;
    for(int i=0;i<n;i++)
    {
    	int flag=0;
    	int l=v[i][0];
    	int r=v[i][1];
    	if(l<=cright && r<=cright) continue;
    	if(l<=cright && r>cright)
    	{
    		cright=r;
    	}
    	if(l>cright)
    	{
    		ans+=cright-cleft+1;
    		cleft=l;
    		cright=r;
    		flag=1;
    	}
    }
    ans+=cright-cleft+1;
    cout<<ans;
    return 0;
}
