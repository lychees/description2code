#include<bits/stdc++.h>
using namespace std;
#define ll long long 
#define mod 10000007
ll int dp[1000004];
int main()
{
    int t,i;
    dp[0]=1;
    dp[1]=1;
    dp[2]=1;
    dp[3]=2;
    dp[4]=3;
    for(i=4;i<=1000000;i++)
    {
        dp[i]=(dp[i-1]%mod+dp[i-3]%mod+dp[i-4]%mod)%mod;
    }
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        cout<<dp[n]%mod<<endl;
    }

}
