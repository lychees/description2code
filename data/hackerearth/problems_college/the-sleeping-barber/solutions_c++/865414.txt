#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
   int d;
   cin>>d;
   while(d--)
    {
    	int64_t g,n,arr[3000];
    	cin>>g>>n;
    	int i=0;
    	for(i=0;i<g;i++)
    	 cin>>arr[i];
    	 sort(arr,arr+g);
    	 int64_t sum=0;
    	 for(i=0;i<g;i++)
    	   {
    	   	sum+=arr[i];
    	   	if(sum>n)
    	   	  break;
    	   }
    	   cout<<i<<"\n";
    	 
    }
    return 0;
}
