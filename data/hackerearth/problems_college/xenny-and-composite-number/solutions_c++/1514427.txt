#include <iostream>
#include<cstdio>
#include<stdlib.h>
#include<algorithm>
#include <bits/stdc++.h>
using namespace std;
int arr[100005];
void markmultiples(int a)
{
	int i=2,num;
	while((num=i*a)<=100005)
	{
		arr[ num-1 ] = 1; // minus 1 because index starts from 0.
        ++i;
	}
	return;
}
void cal()
{

    memset(arr,0,sizeof(arr));
	for(int i=1;i<100005;i++)
	{
		if(arr[i]==0)
		{
			markmultiples(i+1);
		}//means i+1 is prime
	}
	return ;
}
int main()
{
	int t;
	cin>>t;
	cal();
	while(t--)
    {
        int x;
        cin>>x;
        if(x!=0 && x!=1 && x!=2)
        {
            if(arr[x]==0)//x+ is prime
        {
            cout<<x+2<<endl;
