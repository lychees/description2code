#include <bits/stdc++.h>
using namespace std;

bitset<100005>bs;
bitset<100005>de;
bitset<100005>ans;
int ar[100005];

int main()
{
    //freopen("0.in","r",stdin);

    int a,b,c,d,e,x,y,z,n;

    scanf("%d",&n);

    for(a=0;a<n;a++) scanf("%d",&ar[a]);

    sort(ar,ar+n);

    bs.reset();
    de.reset();
    ans.reset();

    for(a=1;a<n;a++)
    {
        d=ar[a]-ar[a-1];

        de=(bs<<d);
        bs=de;
        bs[d]=1;
        ans=(ans|bs);
    }

    cout<<ans.count()<<endl;

    return 0;
}
