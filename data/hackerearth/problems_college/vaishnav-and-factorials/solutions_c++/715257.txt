 #include<iostream>
 
 
 
using namespace std;
 
int main()
{
	int n,n2;
	cin>>n;
     while(n>0)
	 {
	 	int tot=0;
		 int a[1000]={0},x=0,temp1;
		 cin>>n2;
		 temp1=n2;
		 while(n2>0)
		 {
			 a[x]=n2%10;
			 x++;
			 n2=n2/10;
		 }
		 n2=temp1-1;
		 int temp=0;
		 while(n2>0)
		 {
			 int cnt=0,mul=0;
			 while(cnt<x)
			 {
				 mul=a[cnt]*n2+temp;
				 a[cnt]=mul%10;
				 temp=mul/10;
				 cnt++;
 
			 }
			 while(temp>0)
			 {
				 a[cnt]=temp%10;
				 temp=temp/10;
				 cnt++;
			 }
			 x=cnt;
			 n2--;
		 }
	//	 cout<<x<<endl;
		 for(int i=x-1;i>=0;i--)
		 {
			 if(a[i]==7 || a[i]==4)
			 tot++;
		 }
		 cout<<tot<<endl;
		 n--;
	 }
	return 0;
}
