#include <iostream>
using namespace std;
char str[10004],ch;
int main()
{
	int tc,n,i;
    cin>>tc;
    while(tc--)
    {
    	int h[26];
    	for(i=0;i<26;i++)
    	 h[i]=0;
    	cin>>str;
    	for(i=0;str[i];i++)
    	{
    		h[str[i]-'a']++;
    	}
    	cin>>n;
    	for(i=0;i<n;i++)
    	{
    		cin>>ch;
    		h[ch-'a']=0;
    	}
    	int ans=0;
    	for(i=0;i<26;i++)
    	{
    		ans+=(h[i]*(i+1));
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
