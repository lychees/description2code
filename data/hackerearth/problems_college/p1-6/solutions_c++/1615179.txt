#include <iostream>
using namespace std;

#define LL long long

string str;

int main()
{
	cin >> str;
	int l = str.size();
	
	int b4[l], a[l];
	if (str[0]=='1') b4[0] = 1;
	else b4[0] = 0;
	if (str[l-1]=='1') a[l-1] = 1;
	else a[l-1] = 0;
	
	for (int i=1; i<l; i++) {
		b4[i] = b4[i-1];
		if (str[i]=='1') ++b4[i];
	}
	
	for (int i= l-2; i>=0; i--) {
		a[i] = a[i+1];
		if (str[i]=='1') ++a[i];
	}
	
	LL ans=0;
	for (int i=0; i<l; i++) {
		if (str[i] == '0') {
			ans += b4[i]*(LL)a[i];
		}
	}
	
	
	cout << ans << endl;
	
    return 0;
}
