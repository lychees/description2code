ll gcd(ll a,ll b){return (!a)?b:(!b)?a:gcd(b,a%b);}

ll pow(ll a,ll b,ll MOD) {
ll x = 1, y = a;
    while(b > 0) {
        if(b%2 == 1) {
            x=(x*y);
            if(x>MOD) x%=MOD;
        }
        y = (y*y);
        if(y>MOD) y%=MOD;
        b /= 2;
    }
    return x;
}
