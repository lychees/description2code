#include<stdio.h>
int n,arr[100][100];
int func(int a,int b)
{
	if(a==n-1&&b==n-1)
	return 1;
	if(arr[a][b]==0)
	return 0;
	if(a==n-1)
	return func(a,b+1);
	if(b==n-1)
	return func(a+1,b);
	int ans;
	ans=func(a+1,b)||func(a,b+1);
	return ans;
}
int main()
{
	int test;
	scanf("%d",&test);
	while(test--)
	{
		scanf("%d",&n);
		int i,j;
		for(i=0;i<n;i++)
		for(j=0;j<n;j++)
		scanf("%d",&arr[i][j]);
		int k=func(0,0);
		if(k)
		printf("POSSIBLE\n");
		else
		printf("NOT POSSIBLE\n");
	}
	return 0;
}
