def main():
    s = "abcdefghijklmnopqrstuvwxyz"
    n = input()

    for i in range(n):
        count = 0
        t = raw_input()
        for x in s:
            if (x not in t):
                count = 1
                print("NO")
                break
        if (count == 0):
            print("YES")


main()
