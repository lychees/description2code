#include <bits/stdc++.h>
using namespace std;
int sz[100005];
vector <int> g[100005];
void dfs(int u , int p) {
	sz[u] = 1;
	for(int i = 0 ; i < g[u].size() ; i ++)
		if(g[u][i] != p) {
			dfs(g[u][i] , u);
			sz[u] += sz[g[u][i]];
		}
}
int main()
{
	int n; cin >> n;
	for(int i =  1 ; i < n ; i ++) {
		int u , v;  cin >> u >> v;
		g[u].push_back(v);
		g[v].push_back(u);
	}
	dfs(1,0);
	cout << sz[1]-1 << " ";
	for(int i = 2 ; i <= n ; i ++)
		cout << sz[i] << " ";
	cout << '\n';
    return 0;
}
