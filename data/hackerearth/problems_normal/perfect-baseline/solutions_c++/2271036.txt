#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,k;
		char str[10000][100];
		char r[1000];
		cin>>n>>k;
		for(int i=0;i<n;i++)
			cin>>str[i];
		for(int i=0;i<k;i++)
		{
			vector<int>temp;
			for(int j=0;j<n;j++)
			{
				temp.push_back((str[j][i]-'a'));
			}
			sort(temp.begin(),temp.end());
			r[i]=temp[(n-1)/2]+'a';	
		}
		for(int i=0;i<k;i++)
		cout<<r[i];
		cout<<endl;
	}
}
