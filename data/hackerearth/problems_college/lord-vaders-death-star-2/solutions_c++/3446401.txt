#include <iostream>
#include <map>
#include <algorithm>
using namespace std;

int main()
{
	int x,n;
	bool flag=false;
	cin>>n;
	map<int, int, std::greater<int>>mymap;
	map<int, int>::iterator it;
	
	while(n--)
	{
		cin>>x;
		it = mymap.find(x);
		if(it != mymap.end())
			mymap[x]++;
		else
			mymap.insert(pair<int, int>(x,1));
	}

	for(it = mymap.begin(); it!=mymap.end(); ++it)
	{
		if(it->second == 1)
		{
			cout<<it->first<<endl;
			flag = true;
			break;
		}
	}
	if(!flag)
		cout<<"-1\n";
    return 0;
}
