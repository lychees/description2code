#define FOR(i,a,b) for (long long i=a;i<b;i++)
#define FORE(i,a,b) for (long long i=a;i<=b;i++)
#define FORD(i,a,b) for (long long i=a;i>=b; i--)
#define ri(n)({\
    int neg=0;\
    n=0;\
    char ch;\
    for(ch=getchar(); ch<'0' || ch>'9'; ch=getchar()) if (ch=='-') neg=1-neg;\
    n=ch-48;\
    for(ch=getchar(); ch>='0' && ch<='9'; ch=getchar()) n=(n<<3)+(n<<1)+ch-48;\
    if (neg) n=-n;\
})

typedef long long ll;

ll power(ll a, ll n, ll m)
{
    if (n==0) return 1;
    ll tmp,c;
    tmp=c=a;
    n--;
    while (n)
    {
        if (n%2) tmp=tmp*c%m;
        c=c*c%m;
        n/=2;
    }
    return tmp;
}

int test,a[100000+10],b[100000+10],n,ans,vt;

int main()
{
 //   freopen("codeforces.inp", "r", stdin);
 //   freopen("arrange-boxes-1.out", "w", stdout);
    ri(test);
    while (test--)
    {
        ri(n);
        FORE(i,1,n) ri(a[i]);
        FORE(i,1,n) ri(b[i]);
        ans=0;
        FORE(i,1,n)
        if (a[i]!=b[i])
        {
            FORE(j,i+1,n)
            if (a[j]==b[i])
            {
                vt=j;
                break;
            }
            FORD(j,vt,i+1)
            {
                swap(a[j],a[j-1]);
                ans++;
            }
        }
        cout << ans << "\n";
    }
}
