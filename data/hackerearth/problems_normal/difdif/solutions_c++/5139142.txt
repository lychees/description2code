#include <bits/stdc++.h>
using namespace std;

int main()
{
	string str;
	cin >> str;

	set<string> st;	
	for (int i = 0; i < str.length(); i++) {
		for (int j = i; j < str.length(); j++) {
			st.insert(str.substr(i, j - i + 1));
		}
	}
	cout << st.size() << endl;
    return 0;
}
