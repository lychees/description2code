#include <bits/stdc++.h>
using namespace std;

int main(){
	int t;
	scanf("%d",&t);
	while (t--){
		int x;
		scanf("%d",&x);
		int a,b;
		int i = 0;
		int flag = 0;
		while (i<x){
			a = 5*i;
			b = x -a;
			if (b<0) break;
			if (b%3==0){
				flag = 1;
				break;
			}
			i++;
		}
		if (flag==0){
			printf("-1\n");
			continue;
		}
		for (int i=0;i<b;i++) printf("5");
		for (int i=0;i<a;i++) printf("3");
		printf("\n");
	}
	return 0;
}
