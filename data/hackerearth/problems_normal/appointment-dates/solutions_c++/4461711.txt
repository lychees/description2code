#include <stdio.h>
using namespace std;

int calendar[3000][366];

int leap(int y){
	if ((y%4)!=0)
		return 0;
	else if (y%100!=0)
		return 1;
	else if (y%400!=0)
		return 0;
	else
		return 1;
}

int main(){
	int yea=2000;
	int date=6;
	for (int y=0; y<3000; y++){
		int maxd=365+leap(yea);
		for (int d=0; d<maxd; d++){
			calendar[y][d]=date;
			date++;
			if (date>7)
				date=1;
		}
		yea++;
	}
	
	int t;
	scanf ("%d", &t);
	while (t--){
		int month, year;
		scanf ("%d %d", &month, &year);
		int start, end;
		if (month==1)
			start=0, end=30;
		else if (month==2)
			start=31, end=58+leap(year);
		else if (month==3)
			start=59+leap(year), end=89+leap(year);
		else if (month==4)
			start=90+leap(year), end=119+leap(year);
		else if (month==5)
			start=120+leap(year), end=150+leap(year);
		else if (month==6)
			start=151+leap(year), end=180+leap(year);
		else if (month==7)
			start=181+leap(year), end=211+leap(year);
		else if (month==8)
			start=212+leap(year), end=242+leap(year);
		else if (month==9)
			start=243+leap(year), end=272+leap(year);
		else if (month==10)
			start=273+leap(year), end=303+leap(year);
		else if (month==11)
			start=304+leap(year), end=333+leap(year);
		else if (month==12)
			start=334+leap(year), end=364+leap(year);
			
		year-=2000;
		for (int i=start; i<=end; i++)
			if (calendar[year][i]==7)
				printf ("%d ", i-(start-1));
		printf ("\n");
	}
	
    return 0;
}
