#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    long long int n,m;
    cin>>n>>m;
    long long int *arr;
    arr=(long long int *)calloc(n,sizeof(long long int));
    for(int i=1;i<=m;i++)
    {
    	long long int a,b,k;
    	cin>>a>>b>>k;
    	a-=1;b-=1;
    	for(int j=a;j<=b;j++)
    	{
    		arr[j]+=k;
    	}
    }
    long long int sum=0;
    for(int i=0;i<n;i++)
    {
    	sum+=arr[i];
    }
    cout<<sum/n;
}
