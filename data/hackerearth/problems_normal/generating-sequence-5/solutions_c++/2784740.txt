#include<bits/stdc++.h>
using namespace std;

int main()
{
    long long int g,f,s;
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,i;
        scanf("%lld %d",&g,&n);
        f = g*2;
        s = f + g;
        if(n==2)
            printf("%lld %lld\n",f, s);
        else
        {
            for(i=0; i<n-1; i++)
                printf("%lld ",f);
            printf("%lld\n",s);
        }
    }
    return 0;
}
