void sort(int *arr,int i,int j)
{
	if(i<j)
	{
		int mid=partition(arr,i,j);
		sort(arr,i,mid-1);
		sort(arr,mid+1,j);
	}
}

int partition(int *arr,int i,int j)
{
	int x=arr[j];
	int k,pre=i-1;
	for(k=i;k<j;k++)
	{
		if(arr[k]>=x)
		{
			swap(arr,++pre,k);
		}
	}
	swap(arr,++pre,j);
	return pre;
}

void swap(int *arr,int i,int j)
{
	int temp;
	temp=arr[i];
	arr[i]=arr[j];
	arr[j]=temp;
}

void printArray(int *arr,int n)
{
	for(int i=0;i<n;i++)
	{
		cout<<arr[i]<<"  ";
	}
	cout<<endl;
}
