#include <cstdio>
#include <iostream>
#include <cmath>
using namespace std;

long long t, n;

int main()
{
    
    cin >> t;
    
    long long powStored[10000];
    
    powStored[0] = 1;
    for( int i = 1; i < 1000; i++ )
    	powStored[i] = 2 * powStored[i-1];
    
    while ( t-- ) {
    	
    	scanf("%lld",&n);
    	long long sofar = 1, i = 1;
    	
    	
    	while( sofar + i*powStored[i] <= n ) {
    		sofar += i*powStored[i];
    		i++;
    	}
    	
    	long long minResult = (i-1)*2-1;
    	long long maxResult = minResult+2;
    	long long inclusiveDistance = n - sofar + 1;
    	
    	if( inclusiveDistance == minResult+1 )
    		cout << minResult+1;
		else if( inclusiveDistance <= minResult )
    		cout << minResult;
		else
			cout << maxResult;
    	
    	cout << endl;
    	    	
    }
    
    return 0;
}
