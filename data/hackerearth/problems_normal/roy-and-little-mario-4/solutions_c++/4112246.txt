#include <bits/stdc++.h>

using namespace std;
long long int MOD=1000000007;
int main()
{
	int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		long long int a[n+5];
		a[1]=1,a[2]=2,a[3]=4;
		for(int i=4;i<=n;i++)
		{
			a[i]=(a[i-3]%MOD + a[i-2]%MOD + a[i-1]%MOD)%MOD;
		}
		cout<<a[n]<<"\n";
	}
	return 0;
}
