#include <iostream>

using namespace std;

int main()
{
        int test_cases;
        cin>>test_cases;
        while(test_cases > 0)
        {
                int X1,X2,Y1,Y2;
                cin>>X1>>Y1>>X2>>Y2;
                if(X1==Y2&&X2==Y1)
                	printf("2\n");
                else if(X2==Y1||X1==Y2)
                	printf("1\n");
                else
                	printf("0\n");
                
                test_cases --;
        }
        return 0;
}
