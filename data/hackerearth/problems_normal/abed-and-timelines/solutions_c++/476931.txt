#include<iostream>
#include<math.h>
#include<stdlib.h>
using namespace std;
int main()
{
	int t;
	cin >> t;
 	while(t--)
 	{
 		int n,m,x1,x2,y1,y2,p;
 		cin >> n>>m;
 		cin >> x1 >> y1;
 		cin >> x2>>y2;
 		cin >> p;
 		int r1,r2,c1,c2;
 		r1 = abs(x1-x2);
 		c1 = abs(y1-y2);
 		if(y1 < y2)
 			c2 = y1 + (m-y2);
 		else
 			c2 = y2 + (m-y1);
 		if(x1 > x2)
 			r2 = x2 + (n-x1);
 		else
 			r2 = x1 + (n-x2);
 		int res1;
 		res1 = min(r1,r2) + min(c1,c2);
 		cout << res1*p<<endl;
 		if((res1*p)<=1000)
 			cout <<"You saved the group."<<endl;
 		else
 			cout <<"Let go of the group."<<endl;
 	}
}
