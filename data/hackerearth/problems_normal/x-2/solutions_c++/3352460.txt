#include <bits/stdc++.h>

using namespace std;

#define pb push_back
#define mp make_pair
#define ll long long int
#define s(x) scanf("%d", &x)
#define sd(x) scanf("%lf", &x)
#define sl(x) scanf("%lld", &x)
#define mod 1000000007
#define get getchar_unlocked

ll sum[1000005];
ll a[1000005];
int arr[1000005];

int b_search(int i, int n, ll k)
{
    int l = i, r = n, m;
    while (r-l > 1) {
        m = (l+r)/2;
        if (sum[m]-sum[i-1] > k)
            r = m-1;
        else
            l = m;
    }
    if (sum[r]-sum[i-1] <= k)
        return r;
    if (sum[l]-sum[i-1] <= k)
        return l;
    return -1;
}

int main()
{
    int n, i, t, j;
    ll k;
    s(t);
    while (t--) {
        s(n);
        sl(k);
        for (i = 1; i <= n; ++i) {
            sl(a[i]);
            sum[i] = sum[i-1] + a[i];
            arr[i] = 0;
        }
        arr[i] = 0;
        for (i = 1; i <= n; ++i) {
            j = b_search(i, n, k);
            if (j != -1)
                ++arr[j-i+1];
            //cout << i << " " << j << endl;
        }
        for (i = n; i > 0; --i)
            arr[i] += arr[i+1];
        for (i = 1; i <= n; ++i)
            printf("%d ", arr[i]);
        printf("\n");
    }
    return 0;
}
