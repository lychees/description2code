#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;

typedef long long int LL;

#define REP(i,n) for(int i=0;i<n;i++)

LL A[2][2],T[2][2],R[2][2];
LL mod;

void mulRtoA()
{
        int i,j,k;
        memset(T,0,sizeof(T));
        for(i=0;i<2;i++)
                for(j=0;j<2;j++)
                {
                        LL ans=0;
                        for(k=0;k<2;k++)
                        {
                                ans+=(R[i][k] * A[k][j])%mod;
                        }
                        T[i][j]=ans;
                }
        for(i=0;i<2;i++)
                for(j=0;j<2;j++)
                        R[i][j]=T[i][j];
}

void mulAtoA()
{
        memset(T,0,sizeof(T));
        int i,j,k;
        for(i=0;i<2;i++)
                for(j=0;j<2;j++)
                {
                        LL ans=0;
                        for(k=0;k<2;k++)
                        {
                                ans+=(A[i][k] * A[k][j])%mod;
                        }
                        T[i][j]=ans;
                }
        for(i=0;i<2;i++)
                for(j=0;j<2;j++)
                        A[i][j]=T[i][j];
}


LL compute(LL n)
{
        A[0][0] = 1; A[0][1] = 1;
        A[1][0] = 1; A[1][1] = 0;
        R[0][0] = 1; R[0][1] = 0;
        R[1][0] = 0; R[1][1] = 1;
        while(n)
        {
                if(n&1)
                        mulRtoA();
                mulAtoA();
                n>>=1;
        }
        return (R[1][0])%mod;
}

int main()
{
      int T;
      scanf("%d",&T);
        mod=1000000007;
        while(T--)
        {
          int n;
          scanf("%d",&n);
              LL ans=compute(n);
                printf("%lld\n",ans);
        }
        return 0;
}
