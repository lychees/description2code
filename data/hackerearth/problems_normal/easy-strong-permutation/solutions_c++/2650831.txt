#include <bits/stdc++.h>
using namespace std;

int main()
{
    
    int n;
    long long int ans=0;
    cin>>n;
    vector<int> A(n);
    
    for(int i=0; i<n; i++)
    	cin>>A[i];
    
    int i=0,j=n-1;
    sort(A.begin(),A.end());
    
    while(i<j)
    {
    	ans+=abs(A[i]-A[j]);
    	
    	i++;
    	if(i!=j)
    		ans+=abs(A[i]-A[j]);
    	
    	j--;
    }
    if(j+1<n) ans+=abs(A[j+1]-A[0]);
    cout<<ans;
    
    return 0;
}
