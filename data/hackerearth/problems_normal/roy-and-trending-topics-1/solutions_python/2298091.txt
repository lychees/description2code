import heapq
dic={}
pic={}
h=[]
heapq.heapify(h)
for _ in range(int(raw_input())):
	i,old,p,l,c,s=map(int,raw_input().split())
	new=p*50+l*5+c*10+s*20
	
	change=(new-old)+i*1.0/1000000000
	dic[change]=i
	pic[change]=new
	heapq.heappush(h,-change)
	

hi=heapq.nsmallest(5, h)
for i in hi:
	print dic[-i],pic[-i]
