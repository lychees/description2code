#include<iostream>
#include<stdio.h>
#include<stdlib.h>
#include<math.h>
using namespace std;
typedef long long int lint;
template<typename t>
t input(t in)
{
    cin>>in;
    return in;
}
int main()
{
    //freopen("input","r",stdin);
    std::ios_base::sync_with_stdio(false);
    int t=input(t);
    while(t-->0)
    {
        float n=input(n);
        float b=input(b);
        lint ceilAns = ceil(n/b);
        lint floorAns = floor(n/b);
        //cout<<ceilAns<<" "<<floorAns<<endl;
        lint mulCeil = b*ceilAns;
        lint mulFloor = b*floorAns;
        lint mul = mulCeil-n;
        lint absCeil = abs(mul);
        mul = mulFloor-n;
        lint absFloor = abs(mul);

        if(absCeil>absFloor)
        cout<<mulFloor;
        else
        cout<<mulCeil;
        cout<<endl;
    }
}
