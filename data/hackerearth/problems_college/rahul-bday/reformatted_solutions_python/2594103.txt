t = int(raw_input())
while t > 0:
    t = t-1
    gifts = map(int, raw_input().split())
    b = gifts[0]
    w = gifts[1]
    costs = map(int, raw_input().split())
    x = costs[0]
    y = costs[1]
    z = costs[2]
    tot = 0
    if x < y:
        tot = tot+b*x
        if y > (x+z):
            tot = tot+(x+z)*w
        else:
            tot = tot+y*w
    else:
        tot = tot+w*y
        if x > (y+z):
            tot = tot+(y+z)*b
        else:
            tot = tot+x*b
    print tot
