#include <iostream>
using namespace std;
#define ll long long

ll partition(ll arr[],ll l,ll h)
{   ll x=arr[h],i=(l-1),t;
	for(int j=l;j<=h-1;j++)
	{	if(arr[j]<=x)
		{	i++;
			t=arr[i];
			arr[i]=arr[j];
			arr[j]=t;
		}
	}
	t=arr[i+1];
	arr[i+1]=arr[h];
	arr[h]=t;
	return i+1;
}

void quickSort(ll arr[],ll l,ll h)
{	if(l<h)
	{	ll p=partition(arr,l,h);
		quickSort(arr,l,p-1);
		quickSort(arr,p+1,h);
	}
}

void printArray(ll arr[],ll size)
{	ll i;
	for(i=0;i<size;i++)
		cout<<arr[i]<<" ";
}

int main()
{   ll t;
	cin>>t;
	while(t--)
	{	ll n;
    	cin>>n;
    	ll a[n],i;
    	for(i=0;i<n;++i)
        	cin>>a[i];
		quickSort(a,0,n);
		printArray(a,n);
		cout<<"\n";
	}
	return 0;
}
