#include <iostream>
#include <stack>
#include <bitset>
#include <cmath>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>
#include <map>
#include <climits>

#define MOD 4294967296
#define pb push_back
#define mp make_pair

typedef long long int ll;

using namespace std;

int main()
{
    ios::sync_with_stdio(0);
    ll n,x,i,j,loc;
    cin>>n;
    vector <ll> v;
    for(i=0;i<n;i++) {
        cin>>x;
        v.pb(x);
    }
    ll ans,sum = 0;
    for(i=0;i<n-2;i++) {
        ans = INT_MIN;
        for(j=1;j<=v.size()-2;j++) {
            if(ans < v[j-1]*v[j+1]) {
                ans = v[j-1]*v[j+1];
                loc = j;
            } else if(ans == v[j-1]*v[j+1] && v[loc] > v[j]) {
                loc = j;
            }
        }
        v.erase(v.begin()+loc);
        sum+=ans;
    }
    cout<<sum<<endl;
    return 0;
}
