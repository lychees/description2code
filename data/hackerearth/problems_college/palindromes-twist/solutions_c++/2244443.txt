#include<cstdio>
#include<cstring>
#include<cstdlib>
using namespace std;
int main()
{
	int t;
	scanf("%d ",&t);
	while(t--)
	{
		char str[10001];
		scanf("%s",str);
		int ac,bc,cst=0;
		scanf("%d%d",&ac,&bc);
		//char ch=getchar();
		fflush(stdin);
		int l=strlen(str);
		int j,flag=0;
		for(int i=0;i<=(l-1)/2;i++)
		{
			j=l-i-1;
			if(str[i]!=str[j] && !(str[i]=='/' || str[j]=='/'))
			{
				printf("-1\n");
				flag=1;
				break;
			}
			else
			{
				if(str[i]=='/' && str[j]=='/')
				{
					if(ac<=bc)
						cst=cst+2*ac;
					else
						cst=cst+2*bc;
				}
				else if(str[i]=='/')
				{
					if(str[j]=='a')
						cst=cst+ac;
					else if(str[j]=='b')
						cst=cst+bc;
				}
				else if(str[j]=='/')
				{
					if(str[i]=='a')
						cst=cst+ac;
					else if(str[i]=='b')
						cst=cst+bc;
				}
			}
		}
		if(flag==0)
			printf("%d\n",cst);
	}
	return 0;
}
