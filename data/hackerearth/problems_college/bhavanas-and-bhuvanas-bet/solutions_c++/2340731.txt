#include <iostream>
#include <cmath>
#include <bits/stdc++.h>

using namespace std;

typedef unsigned long long ull;

int ret(ull number){
	return (int)log2(number);
}

int main(void)
{
	int t;
	ull number;
	cin>>t;
	while(t--)
	{
		cin >> number;
		cout<<ret(number)<<endl;
	}
	return 0;
}
