#include <bits/stdc++.h>
using namespace std;


int main() {
	int h;
	char ans;
	while (cin >> ans >> h) {
		if(h > 26 or h < 2) {
			puts("INVALID");
			cout << endl;
			continue;
		}
		
		if(tolower(ans)+h-1 > 'z') {
			puts("INVALID");
			cout << endl;
			continue;
		}
		
		for(int i = 1; i<= h; i++) {
			for(int j = 0; j<i; j++) {
				cout << ans << " ";
			}
			cout << endl;
			ans++;
		}
		cout << endl;
	}
	return 0;
}
