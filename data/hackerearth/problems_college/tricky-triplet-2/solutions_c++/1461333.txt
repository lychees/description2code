#include <iostream>
#include<map>

int a[1000005];

int main()
{
    int n,ctr=0,s;
    cin>>n;
    for(int i=0;i<n;i++)
    cin>>a[i];
	for(int i=0;i<n-2;i++)
	{
		for(int j=i+1;j<n-1;j++)
		{
			s=a[j]+a[j]-a[i];
			for(int k=j+1;k<n;k++)

			}
		}
	}

}
