#include <bits/stdc++.h>
using namespace std;
int gcd(int a,int b)
{
	if(b==0)
		return a;
	return(gcd(b,a%b));
}
int main()
{
    int a,b,t;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d%d",&a,&b);
    	if((a%2==1)&&(b%2==1))
    	{
    		printf("Nilesh\n");
    	}
    	else
    	{
    		printf("Shekhar\n");
    	}
    }
    return 0;
}
