#include <iostream>
using namespace std;

int main()
{
    unsigned int t,k;
    int i,j,ctr[5],n,m;
    char a[1000][1000];
    cin>>t;
    for(k=0;k<t;k++)
    {
    	cin>>n>>m;
    	for(i=0;i<n;i++)
    	{
    		for(j=0;j<m;j++)
    		   cin>>a[i][j];
    		cout<<endl;   
    	}
    	ctr[k]=0;
    	for(i=0;i<(n-1);i++)
    		for(j=0;j<(m-1);j++)
    		    if(a[i][j]=='/'&&a[i][j+1]=='\\' &&a[i+1][j]=='\\'&&a[i+1][j+1]=='/')
    		       ctr[k]++;
    }
    for(k=0;k<t;k++)
    {
    	cout<<ctr[k]<<endl;
    }
    return 0;
}
