using namespace std;
long long dp[257];
int main()
{
	//freopen("input.txt", "r", stdin);
	//freopen("output.txt", "w", stdout);
	dp[0] = 1;
	int n, m;
	scanf("%d%d", &m, &n);
	for (int i = 1; i <= n; i++)
	{
		int x;
		scanf("%d", &x);
		for (int j = x; j <= m; j++)
		{
			dp[j] += dp[j - x];
		}
	}
	printf("%lld\n", dp[m]);
}
