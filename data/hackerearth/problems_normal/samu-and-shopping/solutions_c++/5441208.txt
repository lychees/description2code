#include <bits/stdc++.h>
using namespace std;
int dpa[100005];
int dpb[100005];
int dpc[100005];
int main() {
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		for(int i=0;i<n;i++){
			int a,b,c;
			cin>>a>>b>>c;
			if(i==0){
				dpa[0]=a;dpb[0]=b;dpc[0]=c;
			}
			else{
				dpa[i]=a+min(dpb[i-1],dpc[i-1]);
				dpb[i]=b+min(dpc[i-1],dpa[i-1]);
				dpc[i]=c+min(dpb[i-1],dpa[i-1]);
			}
		}
		int ans=min(dpa[n-1],min(dpb[n-1],dpc[n-1]));
		cout<<ans<<endl;
	}
	return 0;
}
