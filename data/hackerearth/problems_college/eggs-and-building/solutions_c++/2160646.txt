#include <bits/stdc++.h>
#define ll long long int

using namespace std;

int dp[1010][20];

int main()
{
	ios_base::sync_with_stdio(0);
	int q,n,k;
	cin>>q;
	while(q--)
	{
		cin>>n>>k;
		for(int i=0;i<=n;i++)
			dp[i][1]=i;
		for(int i=1;i<=k;i++)
		{
			dp[1][i]=1;
			dp[0][i]=0;
		}
		for(int i=2;i<=k;i++)
			for(int j=2;j<=n;j++)
			{
				if(i>j)
					dp[j][i]=dp[j][i-1];
				else
				{
					dp[j][i]=INT_MAX;
					for(int k=1;k<=j;k++)
						dp[j][i]=min(dp[j][i],1+max(dp[k-1][i-1],dp[j-k][i]));
				}
			}
		cout<<dp[n][k]<<"\n";
	}
	return 0;
}
