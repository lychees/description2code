#include <iostream>
#include <string>
using namespace std;

int main()
{
	string str;
	cin >> str;
	int len = str.length();
	bool flag = true;
	
	for(int i=0;i<=len/2;i++)
	{
		if(str[i] != str[len-i-1])
		{
		flag = false;
		break;
		}
		else
			continue;
		
	}
	if(flag  == true)
    	cout << "YES" << endl;
    else
    	cout << "NO" << endl;
    return 0;
}
