t = input()
for _ in range(0, t):
    n = input()
    case = map(int, raw_input().split())
    cnt = 0
    for x in range(0, n-1):
        mx = case[x]
        i = x+1
        while case[i] < mx:
            i += 1
            if i == n:
                cnt += 1
#                print mx
                break
    print cnt+1
