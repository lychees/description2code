#include <bits/stdc++.h>
using namespace std;
int main()
{
	long int a,b,c,n,t,ans=0,x;
	cin>>t;
	
	while(t--)
	{
		
		
		cin>>a>>b>>c>>n;
		long int arr[n];
		memset(arr,0,sizeof(arr));
		ans=0;
		arr[0]=a;
		arr[1]=b;
		
		for(x=2;x<n;x++)
		arr[x]=(arr[x-1]*c+arr[x-2])%10;
		
		for(x=0;x<n;x++)
		ans=(ans*10+arr[x])%41;
		
		
		if(ans == 0)
		cout<<"YES"<<endl;
		else
		cout<<"NO"<<endl;
		
	}
	
	return 0;
}
