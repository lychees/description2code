#include <bits/stdc++.h>
using namespace std;
map<string,int>mp;
long long int cost[1002][1002];
string s;
int main()
{
	cin.sync_with_stdio(false);cin.tie(0);
    int n;
    cin>>n;
    for(int i=1;i<=n;i++)
    {
    	cin>>s;
    	mp[s]=i;
    }
    for(int i=1;i<=n;i++)
    	for(int j=1;j<=n;j++)
    		cin>>cost[i][j];
    long long int tot=0;
    int curr=1;
    int p,tmp;
    cin>>p;
    while(p--)
    {
    	cin>>s;
    	tmp=mp[s];
    	tot+=cost[curr][tmp];
    	curr=tmp;
    }
    cout<<tot;
    return 0;
}
