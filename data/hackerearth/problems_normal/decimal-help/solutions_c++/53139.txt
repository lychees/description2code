#include <iostream>
using namespace std;


int  divide(int num,int dem,int c,int t,int h){
	while(1){
	    if(c != 0){
	        if(num/dem == 0){
	            num = num*10;
	            if(t==1){
	                c-=1;
	                h= 0;
	            }
	            else{
	                t=1;
	            }
	        }
	 
	        else{
	            h =  num/dem;
	            c-=1;
	            t=0;
	            num=num%dem;
	        }
	    }
	    else{
	        return  h;
	    }
	}
}

void test(){
	int lst[3],i,num,dem,c;
	for(int i=0;i<3;i++){
		cin>>lst[i];
	}
	num = lst[0];
	dem = lst[1];
	c = lst[2];
	cout<<divide(num,dem,c,0,NULL)<<endl;
}


int main(){
	int n,i;
	cin>>n;
	for(i=0;i<n;i++){
		test();
	}
	return 0;
}
