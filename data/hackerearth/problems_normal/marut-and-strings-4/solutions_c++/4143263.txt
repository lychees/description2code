#include <iostream>
using namespace std;

int main()
{
	int t;
	cin >> t;
	if(t <=0 || t>10)
	{
		cout << "Invalid Test" << endl;
		return 0;
	}
	while(t--)
	{
		int i,u=0,l=0;
		string s;
		cin >> s;
		if(s.size() > 100)
		{
			cout << "Invalid Input" << endl;
			continue;
		}
		for(i=0 ; i < s.size();i++)
		{
			if(s[i] >=65 && s[i] <= 91)
			{
				u++;
			}
			if(s[i] >=97 && s[i] <= 122)
			{
				l++;
			}
		}
		if(l == 0 && u == 0)
		{
			cout << "Invalid Input" << endl;
		}
		else if((l > 0 && u == 0) || (u > 0 && l == 0))
		{
			cout << 0 << endl;
		}
		else
		{
			if(l >= u)
			{
				cout << u << endl;
			}
			else
			{
				cout << l << endl;
			}
		}
		
	}
    
    return 0;
}
