#include<bits/stdc++.h>
using namespace std;
map<int,int> m;
int main(){
	int n;
	scanf("%d",&n);
	int a[n+1000];
	for(int i=0;i<n;i++)
	scanf("%d",&a[i]);
	vector<pair<int,int> > ans;
	int s=0,c=1;
	for(int i=0;i<n;i++){
		if(m[a[i]]!=c)
		   m[a[i]]=c;
		else {
			ans.push_back(make_pair(s+1,i+1));
			c++;
			s=i+1;
		}
	}
	if(ans.size()==0)
	printf("-1\n");
	else
	{
		printf("%d\n",ans.size());
		for(int i=0;i<ans.size()-1;i++){
			printf("%d %d\n",ans[i].first,ans[i].second);
		}
		printf("%d %d\n",ans[ans.size()-1].first,max(ans[ans.size()-1].second,n));
	}
	return 0;
}
