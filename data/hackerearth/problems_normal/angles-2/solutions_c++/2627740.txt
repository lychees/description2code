#include <iostream>
using namespace std;

int main()
{
    int n,k,i,j;
    cin>>n>>k;
    int arr[361]={0},a[n];
    for(i=0;i<n;i++)
    cin>>a[i];
    for(i=0;i<n;i++){
    	for(j=1;j<361;j++){
    		arr[a[i]*j%360]=1;
    	}
    }
    for(i=0;i<k;i++){
    	int t;
    	cin>>t;
    	if(arr[t])
    	cout<<"YES"<<endl;
    	else
    	cout<<"NO"<<endl;
    }
}
