#include<iostream>
#include<stdio.h>
#include<string.h>

using namespace std;
int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
        char a[100];
        cin>>a;
        int len=strlen(a);
        for(int p=0;p<len;p++)
        {
            int temp = a[p]+p;
            if(temp>122)
            {
                temp=temp-122+96;
            }
            printf("%c",temp);
        }
        cout<<"\n";
    }
}
