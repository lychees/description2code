#include<iostream>
#include<cstring>
#include<string.h>
#define ll long long int
using namespace std;
int main()
{
	int tc=0;
	cin>>tc;
	while(tc-->0)
	{
		string str;
		cin>>str;
		int len=str.length();
		ll count=0;
		ll sub_str=0;
		int j=0;
		while(j<len)
		{
			count=0;
			while(j<len&&(str[j]=='A'||str[j]=='B'||str[j]=='C'||str[j]=='a'||str[j]=='b'||str[j]=='c'))
			{
				count++;
				j++;
			}
			sub_str+=(count*(count+1)/2);
			j++;
		}
		cout<<sub_str<<endl;
	}
	return 0;
}
