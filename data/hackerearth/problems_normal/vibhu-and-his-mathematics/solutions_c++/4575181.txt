#include <bits/stdc++.h>
using namespace std;
int main()
{int t;
	long long dp[1234567];
	memset(dp,0,sizeof(dp));
	dp[1]=1;
	dp[2]=2;
	for(long long i=3;i<=1000000;i++)
	{
		dp[i]=dp[i-1];
		dp[i]+= ((i-1)*dp[i-2]);
		dp[i]=dp[i]%1000000007;
	}
cin>>t;
while(t--)
{
	int n;
	cin>>n;
 	cout<<dp[n]<<endl;
	
}
    return 0;
}
