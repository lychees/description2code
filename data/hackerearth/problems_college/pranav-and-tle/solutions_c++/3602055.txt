#include<bits/stdc++.h>
using namespace std;
int main(){
    int t;
    scanf("%d",&t);
    while(t--){
    	char m[500001];
    	long long int m1;
    	scanf("%s %lld",m,&m1);
    	int x=strlen(m)-1;
    	long long f=1;
    	long long ans=0;
    	for(int i=x;i>=0;i--){
    		ans+=(f*(m[i]-'0'))%m1;
    		f*=10;
    		f%=m1;
    		ans%=m1;
    	}
    	printf("%lld\n",ans);
    }
	return 0;
}
