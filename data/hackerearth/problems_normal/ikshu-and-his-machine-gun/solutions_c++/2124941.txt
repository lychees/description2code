#include <cstdio>
#include <vector>
#include <algorithm>

int main() {
	int s, x;
	scanf("%d", &s);
	std::vector<int> a;
	while (s--) {
		scanf("%d", &x);
		a.push_back(x);
	}
	sort(a.begin(), a.end());
	std::vector<int>::iterator i = a.begin();
	printf("%d ", (*i));
	int n = 1;
	++i;
	for (i; i != a.end(); ++i) {
		printf("%d ", (*i)-n);
		++n;
	}
	printf("\n");
	return 0;
}
