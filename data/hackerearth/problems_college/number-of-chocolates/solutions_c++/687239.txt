#include <iostream>
#include<cstdio>
using namespace std;

int main()
{
    long long n,*arr,*arr1,g1,p1,g2,p2,q;
    
    	cin>>n>>q;
    	arr=new long long[n+1];
    	arr1=new long long[n+1];
    	for(int i=0;i<n;i++)
    	scanf("%lld",&arr[i]);
    	
    	arr1[0]=arr[0];
    	for(int i=1;i<n;i++)
    	arr1[i]=arr1[i-1]+arr[i];
    	cout<<endl;
    	for(int i=0;i<q;i++)
    	{
    		long long t;
    		scanf("%lld%lld%lld%lld",&p1,&g1,&p2,&g2);
    		if(p1>g1)
    		{
    			t=p1;p1=g1;g1=t;
    			t=p2;p2=g2;g2=t;
    		}
    		if(g1<=p2&&g2>=p2)
    		{
    			cout<<(arr1[g2]-arr1[p1]+arr[p1])<<endl;
    		}
    		else if(g1<=p2&&g2<=p2)
    		cout<<(arr1[p2]-arr1[p1]+arr[p1])<<endl;
    		else
    		{
    			cout<<(arr1[p2]-arr1[p1]+arr[p1]+arr1[g2]-arr1[g1]+arr[g1])<<endl;
    		}
    		
    		
    	}
    return 0;
}
