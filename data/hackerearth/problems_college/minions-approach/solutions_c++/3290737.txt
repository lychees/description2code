#include<bits/stdc++.h>
using namespace std;
long long int a[1000000]={0};
void multiply(long long int arr[],long long int j,long long int n)
{
	long long int num,k=2;
	while((num=j*k)<=n)
	{
		a[num]=1;
		k++;
	}
}
int main()
{
	int i,k=0,j;
		for(i=2;i<=100000;i++)
		{
			if(a[i]==0)
			{
			multiply(a,i,100000);
			}
		}
		j=0;
		a[1]=1;
		int t,n;
		scanf("%d",&t);
		while(t--)
		{
			int count=0;
			scanf("%d",&n);
			for(i=1;i<=n;i++)
			{
				if(__gcd(n,i)==1)
				count++;
			}
			if(a[count]!=0)
			printf("Incorrect\n");
			else
			printf("Correct\n");
		}
		return 0;
}
