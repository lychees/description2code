#include <iostream>
#include <string.h>
#include <stdio.h>
using namespace std;

int main()
{
    char arr[10000];
    int t,acost,bcost,i,j,length,cost,flag=0;
    char c;
    cin>>t;
    getchar();
    for(i=0;i<t;i++)
    {
        length=0;
        cost=0;
        flag=0;
        c=getchar();
        while(c!='\n')
        {
                arr[length]=c;
                length++;
                c=getchar();
        }

        cin>>acost;
        getchar();
        cin>>bcost;
        getchar();
        for(j=0;j<length/2;j++)
        {
            if(arr[j]=='/' && arr[length-j-1]=='/')
                {
                    if(acost>=bcost)
                    {
                        cost=cost+2*bcost;

                    }
                    else
                    {
                       cost=cost+2*acost;

                    }
                }
            else if(arr[j]!=arr[length-j-1])
            {

                if(arr[j]=='/')
                {
                    if(arr[length-j-1]=='a')
                    {
                        cost=cost+acost;


                    }
                    else
                    {
                        cost=cost+bcost;

                    }
                }
                else if(arr[length-j-1]=='/')
                {
                    if(arr[j]=='a')
                    {
                        cost=cost+acost;


                    }
                    else
                    {
                        cost=cost+bcost;

                    }
                }
                else
                {
                    flag=1;
                    cout<<"-1"<<endl;
                    break;
                }




            }
        }
    if(flag==0)
    {
        cout<<cost<<endl;
    }
    }

}
