#include<iostream>
#define mod 1000000007
using namespace std;
long long int f[2000000];
long long int inv[2000000];
void fn()
{
	long long int i,result,exp,base;
	f[0]=1;
	for(i=1;i<=2000000;i++)
	{
		f[i]=i*f[i-1]%mod;
	}
inv[0]=1;
	for(i=1;i<=1000000;i++)
	{
	result=1;
	exp=mod-2;
	base =f[i];
	while(exp>0)
	{
		if(exp&1)
		result=(result*base)%mod;
		exp=exp>>1;
		base=(base*base)%mod;
	}
	inv[i]=result;
}
}
int main(){
fn();
int t,n,r,x;
cin>>t;

while(t--){

    cin>>n>>r;
    int num=n;
    while(n--){cin>>x;r-=x;}
   // cout<<r<<endl;
    long long ans=f[num+r-1]%mod;
    ans=(ans*inv[num-1])%mod;
    ans=(ans*inv[r])%mod;
    cout<<ans<<endl;
}

return 0;
}
