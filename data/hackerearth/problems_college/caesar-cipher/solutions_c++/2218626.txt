#include <iostream>
using namespace std;

int main()
{
	int T; cin>>T;
	while(T--){
		int z;
		string s;
		cin>>z>>s;
		for(int i=0;i<s.size();i++){
			char c=s[i];
			if('a'<=c && c<='z'){
				int y=c-'a';
				c=(y+z)%26+'a';
				s [i]=c;
			}
			else{
				int y=c-'A';
				c=(y+z)%26+'A';
				s [i]=c;	
			}
		}
		cout<<s<<endl;
	}
    
    return 0;
}
