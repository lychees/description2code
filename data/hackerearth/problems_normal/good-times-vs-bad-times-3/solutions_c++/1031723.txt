#include <bits/stdc++.h>
#define pb push_back
using namespace std;

bool f[100];
vector<int> vec;

void pre()
{
	for(int i=2;i<=60;i++) {
		if(!f[i]) {
			vec.pb(i);
			for(int j=i*i;j<=60;j+=i) f[j]=1;
		}
	}
}

int arr[86410];

int main()
{
	int t,h,m,s,gc;
    pre();
    for(int i=86359;i>=0;i--) {
    	h=i/3600,m=(i%3600)/60,s=i%60;
    	bool ff=0;
    	for(int j=0;j<vec.size();j++) ff|=(h%vec[j]==0)&&(m%vec[j]==0)&&(s%vec[j]==0);
    	arr[i]=arr[i+1]+ff;
    }
    cin >> t;
    while(t--) {
    	cin >> h >> m >> s;
    	gc=1;
    	if(arr[h*3600+m*60+s]) gc=__gcd(arr[h*3600+m*60+s],86400-arr[h*3600+m*60+s]-h*3600-m*60-s);
    	cout << arr[h*3600+m*60+s]/gc << ":" << (86400-arr[h*3600+m*60+s]-h*3600-m*60-s)/gc << endl;
    }
    return 0;
}
