#include <bits/stdc++.h>
#define pdd pair<double, double>
using namespace std;

int main()
{
    vector< pdd > in, s;
    double a,b,x=0,y=0;
    int i;
    for(i=0;i<3;i++){
    	cin>>a>>b;
    	in.push_back(pdd(a,b));
    	x+=a;
    	y+=b;
    }
    s.resize(3);
    for(i=0;i<3;i++){
    	s[i].first = x - 2*in[i].first;
    	s[i].second = y - 2*in[i].second;
    }
    sort(s.begin(),s.end());
    cout.precision(4);
    for(i=0;i<3;i++){
    	cout<<fixed<<s[i].first<<" "<<s[i].second<<endl;
    }
    return 0;
}
