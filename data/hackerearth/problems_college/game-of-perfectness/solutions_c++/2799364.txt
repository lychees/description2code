#include <cstdio>
#include <iostream>
#include <vector>
#include <algorithm>
#define DEBUG if(0)
using namespace std;
typedef long long int int32;
typedef struct Data
{
    int32 T;
    int32 P;
    int32 index;
}Data;

bool CMP(Data A,Data B)
{
    if( A.P > B.P )
        return true;
    if( A.T < B.T )
            return false;
     if( A.index > B.index )
          return false;
    
}

void Process(vector<Data> &input,int N,int I)
{
    sort(input.begin(),input.end(),&CMP);
    DEBUG
    {
        for(int i = 0; i < input.size(); ++i)
        {
            cout << input[i].index << " " << input[i]. T << " " << input[i].P << endl;
        }
    }

    cout << "Case #"<< I << ":";
    for(int i = 0; i < N; ++i)
        cout << " " << input[i].index;
    cout << endl;
}

int main()
{
    int T,N,I=1;
    cin >> T;
    while(T--)
    {
        cin >> N;
        vector<Data> input;
        input.resize(N);

        for(int i = 0; i < N; ++i)
        {
            cin >> input[i].T;
            input[i].index = i;
        }


        for(int i = 0; i < N; ++i)
            cin >> input[i].P;

        Process(input,N,I++);
    }
    return 0;
}
