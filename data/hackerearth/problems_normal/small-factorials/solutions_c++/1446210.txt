#include <iostream>
using namespace std;
	
int main()
{
   
    int k=100;
   int n;
   int in[100];
   //cout<<"\n Enter Number of Terms :";
   cin>>n;
   //cout<<"\n Enter Input:";

   for(int i=0;i<n;i++){
    int o;
    cin>>o;
    in[i]=o;
   }
   for(int y=0;y<n;y++){
        int a[200];
        int l=1;
        a[1]=1;
   for(int i=1;i<=in[y];i++){
        int temp=0;
    for(int j=1;j<=l;j++){
        int x=(a[j]*i)+temp;
        a[j]=x%10;
        temp=x/10;
    }
    while(temp>0){
        l++;
        a[l]=temp%10;
        temp=temp/10;
    }
   }
   //cout<<"\n Factorial is :\n";
   for(int i=l;i>=1;i--){
    cout<<a[i];
   }
   cout<<"\n";
   }
    return 0;
}
