#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int n,k;
    cin>>n>>k;
    int arr[100022];
    for(int i =0;i<n;i++)
    cin>>arr[i];
    sort(arr,arr+n);
    //cout<<arr[k-1]-arr[0]<<endl;
    int min = arr[k-1] - arr[0];
    for(int i =0;i<n-k+1;i++)
    {
    	if(arr[k-1+i]-arr[i] <min)
    	min = arr[k-1+i] - arr[i];
    }
    cout<<min<<endl;
    return 0;
}
