#include <iostream>
using namespace std;

int main()
{
    long long int p1,p2,n,m,a,b,c,result,t;
    cin>>t;
    while(t--)
    {
    	cin>>n>>m>>a>>b>>c;
    	result=0;
    	if(n>=m)
    	{
    		result+=max(2*c*m,m*a+b*m);
    		result+=(n-m)*a;
    	}
    	else
    	{
    		result+=max(2*c*n,n*a+b*n);
    		result+=(m-n)*b;
    	}
    	cout<<result<<endl;
    }
    return 0;
}
