#include <iostream>
#include <string.h>
using namespace std;

int main()
{
	int t,n;
	cin>>t;
	string s1,s2,s3;
	while(t--)
	{
		cin>>s1>>s2;
		s3 = s1 + s2;
		s1 = s2 + s1;
		n = s1.size();
		for(int i = 0; i <= n/2; i++)
		{
			if(i==n/2)
				cout<<"YES"<<endl;
			if(s1[i]!=s1[n-i-1]&&s3[i]!=s3[n-i-1])
			{
				cout<<"NO"<<endl;
				break;
			}
		}
	}
    return 0;
}
