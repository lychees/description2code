#include<bits/stdc++.h>
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int i,n;
	cin>>n;
	int a[n];
	for(i=0;i<n;i++)
	  cin>>a[i];
	sort(a,a+n);
	int b[n],k=0;
	for(i=0;i<n;i++)
	{
		if(i==0)
		{
			b[k++]=abs(a[i]-a[i+2]);
			b[k++]=abs(a[i]-a[i+1]);
		}
		else if(i>0 && i<n-2)
		{
			b[k++]=abs(a[i]-a[i+2]);
		}
		else if(i==n-2)
		    b[k++]=abs(a[i]-a[i+1]);
	}
	sort(b,b+k);
	cout<<b[k-1];
	return 0;
}
