#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int denom = 0;
    	denom += n/1000;
    	n %= 1000;
    	denom += n/500;
    	n %= 500;
    	denom += n/100;
    	n %= 100;
    	denom += n/50;
    	n %= 50;
    	denom += n/20;
    	n %= 20;
    	denom += n/10;
    	n %= 10;
    	denom += n/5;
    	n %= 5;
    	denom += n/2;
    	n %= 2;
    	denom += n/1;
    	n %= 1;
    	cout<<denom<<endl;
    }
    return 0;
}
