#include <iostream>
#include<algorithm>
using namespace std;
#define m 1000000007
long long int pr(long long int a,long long int n)
{
	if(n==0)
	 return 1;
	 
	 long long int ans=pr(a,n/2),t;
	 t=ans*ans;
	 t=t%m;
	 if(n%2)
	 t=t*a;
	 t=t%m;
	 return t;
	 
}
int main()
{
   int t;
   cin>>t;
   long long int fact[100000];
   fact[0]=fact[1]=1;
   for(long int i=2;i<=100000;i++)
   {
   	   fact[i]=(i*(fact[i-1]%1000000007))%1000000007;
   }
   while(t--)
   {
   	   long int n,k;
   	   cin>>n>>k;
   	   long long int arr[n];
   	   for(int i=0;i<n;i++)
           cin>>arr[i];
   	      sort(arr,arr+n);
   	      unsigned long  long int res=0;
   	      for(long int i=0;i<=n-k;i++)
   	      {
   	      	  long long int ans=fact[k-1]*fact[n-i-k],t;
   	      	    ans=ans%m;
   	      	    long long int p=pr(ans,m-2);
   	      	    p=p%m;
   	      	    t=(fact[n-i-1]*arr[i]);
   	      	    t=t%m;
   	      	    t=t*p;
   	      	    t=t%m;
   	      	    res+=t;
   	      	    res=res%m;
   	      }
  cout<<res<<endl;
   	   
   	   
   	   
   }
    return 0;
}
