using namespace std;
int reverse(int n){
int r=0,r1=0;
	while(n)
	{
		r=n%10;
		r1=r1*10+r;
		n=n/10;
	}
	return r1;}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
	int n1,n2;
	int r2,r1;
	cin>>n1;
	cin>>n2;
	r1=reverse(n1);
	r2=reverse(n2);

}
