#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    
    {
    	int count=0;
    	int n;
    	cin>>n;
    	while(n)
    	{
    		if(n%2==0)
    		{
    			;
    			n/=2;
    		}
    		else
    		{
    			count++;
    			n=n/2;
    		}
    	}
    	if(count%2==0)
    	cout<<"even\n";
    	else cout<<"odd\n";
    }
}
