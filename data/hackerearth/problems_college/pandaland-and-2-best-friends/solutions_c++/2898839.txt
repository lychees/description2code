#include <iostream>
using namespace std;

int main()
{
	long long i,n,t,x,a[100005];
	
	for(i=0;i<=100000;i++)
	{
		if(i%2==0)
		a[i] = ((i+3)/2)*((i+3)/2);
		else
		a[i] = ((i+1)/2)*((i+3)/2);
	}
	cin>>t;
	while(t--)
	{
		cin>>n;
		x = lower_bound(a,a+100001,n)-a;
		cout << x+1 << endl;
	}
    
    return 0;
}
