#include <bits/stdc++.h>
using namespace std;
 
#define MOD 1000000007
#define mp make_pair
 
int N, K, T;
 
int A[2002][2];
map< pair<int, int>, long long> M;
 
long long solve(int i, int sum)
{
	if(sum>T)
		return 0;
	
	if(i>K)
	{
		if(sum==T)
			return 1;
		
		return 0;
	}
	
	if(M.count(mp(i, sum)))
		return M[mp(i, sum)];
	
	long long ans = solve(i+1, sum+A[i][0]) + solve(i+1, sum+A[i][1]);	
	
	if(ans>=MOD)
		ans%= MOD;
	
	M[mp(i, sum)] = ans;
	
	return M[mp(i, sum)];
}
 
int main()
{
    scanf("%d %d %d", &N, &K, &T);
    
    for(int i=1; i<=K; i++)
    {
    	int x;
    	scanf("%d", &x);
    	
    	A[i][0] = x-1;
    	A[i][1] = x+1;
    }
    
    printf("%lld\n", solve(1, 0));
    
    return 0;
}
