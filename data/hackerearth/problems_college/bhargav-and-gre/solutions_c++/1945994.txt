#include<iostream>
#include<string.h>
using namespace std;
int main(){
	char a[100000];
	cin>>a;
	long long int n=strlen(a),t;
	cin>>t;
	long long int syn=0,ant=0;
	while(t--){
		char s[100000];
		cin>>s;
		if(strlen(s) == n){
			if(s[0] == a[0]){
				syn++;
			}
			if((s[0]-a[0])==3){
				ant++;
			}
		}
	}
	cout<<syn<<" "<<ant;
	return 0;
	
}
