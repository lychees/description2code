#include <iostream>
#include <math.h>
#include <algorithm>
using namespace std;

int main()
{
    unsigned short int T;
    cin >> T;
    while(T--)
    {
    	unsigned int a, b;
    	cin >> a >> b;
    	
    	while(a != b)
    	{
    		if(a > b)
    		{
    			a = a/2;
    		}
    		else
    		{
    			b = b/2;
    		}
    	}
    	
    	cout << a << endl;
    	
    }
    return 0;
}
