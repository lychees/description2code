#include<bits/stdc++.h>
#include<iostream>
using namespace std;
#define fre 	freopen("0.in","r",stdin),freopen("0.out","w",stdout)
#define abs(x) ((x)>0?(x):-(x))
#define MOD 100000000000000000LL
#define lld signed long long int
#define pp pop_back()
#define ps(x) push_back(x)
#define mpa make_pair
#define pii pair<int,int>
#define fi first
#define se second
#define scan(x) scanf("%d",&x)
#define print(x) printf("%d\n",x)
#define scanll(x) scanf("%lld",&x)
#define printll(x) printf("%lld\n",x)
#define boost ios_base::sync_with_stdio(0)
//vector<int> g[2*100000+5];int par[2*100000+5];
lld dp[503][305];
lld data[505];
int main()
{
	//fre;
	int N,K;
	cin >>N>>K;
	lld S=0;
	for(int i=1;i<=N;++i)
	{
		scanll(data[i]);
		S += data[i];
	}
	for(int i=0;i<=N;++i)
		for(int k=0;k<=K;++k)
			dp[i][k] = MOD;
	dp[0][0] = 0;
	for(int i=1;i<=N;++i)
	{
		for(int k=1;k<=K;++k)
		{
			lld cur = 0;
			for(int j=i;j>=1;--j)
			{
				cur += data[j];
				dp[i][k] = min(dp[i][k], dp[j-1][k-1] + cur * cur);
			}
		}
	}
	printll(dp[N][K] * K - S*S);
}
