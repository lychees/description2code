#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int n;
	cin>>n;
	ll int i,j,l,ans=0;
	string s;
	map<vector<ll int>,ll int> mp;
	for(i=0;i<n;i++)
	{
		cin>>s;
		l=s.length();
		vector<ll int> v(26,0);
		for(j=0;j<l;j++)
		{
			v[s.at(j)-'a']++;
		}
		mp[v]++;
		ans=max(ans,mp[v]);
	}
	cout<<ans<<endl;
	return 0;
}
