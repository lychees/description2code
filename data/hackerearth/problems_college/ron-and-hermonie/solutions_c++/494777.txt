#include<iostream>
#include <cstdio>

using namespace std;
#define gc getchar_unlocked

typedef long long ll;

void scanint(ll &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}


int main(){
	ll t, a, b, carry, lastd, mul;
	int data[10] = {1, 1, 4, 4, 2, 1, 1, 4, 4, 2};
	scanint(t);
	while(t--){
		scanint(a);
		scanint(b);
		lastd = a%10;
		if(data[lastd] == 1)
			printf("%llu\n",lastd);
		else{
			carry = b % data[lastd];
			mul = lastd;
			if(carry == 0){
				for(int i=1; i<data[lastd]; i++)
					mul = mul*lastd;
			}else if(carry == 2)
				mul = mul * mul ;
			else if(carry == 3)
				mul = mul * mul * mul;
			printf("%llu\n",mul%10);
		}
	}
	
	
	return 0;
}
