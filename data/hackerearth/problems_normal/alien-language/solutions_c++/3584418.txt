#include <iostream>
using namespace std;

int main()
{
    int t; cin>>t;
    while(t--)
    {
    	string a,b;
    	cin>>a>>b;
    	int arr[26];
    	int brr[26];
    	for(int i=0;i<26;i++) arr[i]=0;
    	for(int i=0;i<26;i++) brr[i]=0;
    	for(int i=0;i<a.length();i++) arr[a[i]-'a']++;
    	for(int i=0;i<b.length();i++) brr[b[i]-'a']++;
    	int c=0;
    	for(int i=0;i<26;i++) if(arr[i]>0&&brr[i]>0) c++;
    	if(c>0) cout<<"YES"<<endl;
    	else cout<<"NO"<<endl;
    }
}
