#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int T, N, CT, TT;
    
    cin >> T;
    
    while (T--)
    {
    	cin >> N >> CT >> TT;
    	
    	if (CT == TT)
    		cout << "Chotu" << endl;
    	else if  (abs(CT-TT) == 1)
    		cout << "Chotu" << endl;
    	else
    		cout << "Motu" << endl;
    		
    }
    return 0;
}
