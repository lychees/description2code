#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <cmath>

using namespace std;

bool solve(int W[45], int H[45], int brave, int N){
    for(int i=1; i<N; i++){
        if(W[i]*H[N-1-i] >= brave){
            return false;
        }
    }
    return true;
}

int main()
{
    int T;
    scanf("%d", &T);
    while(T-- > 0){
        int N;
        scanf("%d", &N);
        int W[45];
        for(int i=0; i<N; i++) { scanf("%d", &W[i]); }
        sort(W+1, W+N);

        int H[45];
        for(int i=0; i<N; i++) { scanf("%d", &H[i]); }
        sort(H, H+N);

        long brave = W[0] * H[N-1];
        bool res = solve(W, H, brave, N);
        if(res){
            printf("YES\n");
        }else{
            printf("NO\n");
        }
    }

    return 0;
}
