#include <iostream>
using namespace std;

int main()
{
	int t;
	long long int a,b,c,sum1,sum2,sum3,diff1,diff2,diff3;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b>>c;
    	sum1=a+b;
        sum2=b+c;
        sum3=c+a;
        diff1=a-b;
        diff2=b-c;
        diff3=c-a;
        if((sum1 <= c)||(sum2 <= a)||(sum3 <= b)||(diff1>c)||(diff2>a)||(diff3>b))
        {
        	cout<<"NO";
        }
    	else if(a==b && b==c )
    	{
    		cout<<"YES. It is a equilateral triangle.";
    	}
    	else if((a==b)||(b==c)||(c==a))
    	{
    		cout<<"YES. It is a isosceles triangle.";
    	}
    	else
    	{
           cout<<"YES. It is a scalene triangle.";
    	}
    	cout<<endl;
    }
    
    
    
    return 0;
}
