#include <bits/stdc++.h>
using namespace std;

int main()
{
    int N;
    cin >> N;
    
    long long a = 0, b = 0, c = 0;
    
    for(int i=1; i<=N; i++)
   	{
   		int x;
   		cin >> x;
   		
   		if(i%3==1)
   			a+= x;
   		else if(i%3==2)
   			b+= x;
   		else
   			c+= x;
   	}
   	
   	cout << a << " " << b << " " << c << endl;
    
    return 0;
}
