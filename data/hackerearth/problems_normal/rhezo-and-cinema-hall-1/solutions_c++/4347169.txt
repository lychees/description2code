#include <iostream>
#include <vector>
#include <cstring>
#include <string> 
#include <cmath>
#include <algorithm>
#include <cstdlib>
#define mod 1000000007
using namespace std;
vector<vector<int> > v(10, vector<int>(5,0));
long long dp[501][11][501];

long long solve(int curr,int seats,int n,int m,int k){
	//cout<<"curr ->"<<curr<<" seats left ->"<<k<<endl;
	if(curr >= n-1){
		if(k <= 0)return 1;
		return 0;
	}
	else if(dp[curr][seats][k] != -1)return dp[curr][seats][k];
	else{
		long long ans = 0;
		for(int i = 2;i <= 5;i++ ){
			//cout<<"consumed ->"<<i<<endl;
			ans = (ans + ((solve(curr+1,i,n,m,k-i))%mod * v[m-1][i-1])%mod)%mod;
		}
		dp[curr][seats][k] = ans;
		return dp[curr][seats][k];
	}
}

int main() {
	int n,m,k;
	cin>>n>>m>>k;
	for(int i=0;i<10;i++){
		v[i][0] = i+1;
	}
	for(int i=0;i<10;i++){
		for(int j=1;j<5;j++){
			int sm = 0;
			for(int k=i-2;k>=0;k--){
				sm += v[k][j-1];
			}
			v[i][j] = sm;
		}
	}
	
	memset(dp,-1,sizeof(dp));
	long long ans = 0;
	for(int i = 2;i <= 5;i++ ){
		int temp = ((solve(0,i,n,m,k-i)%mod )*v[m-1][i-1])%mod;
		ans = (ans + temp)%mod;
	}
	cout<<ans<<endl;
	return 0;
}
