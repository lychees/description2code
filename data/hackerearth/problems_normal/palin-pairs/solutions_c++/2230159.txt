

    #include<iostream>
    #include<map>
    #include<ctype.h>
    #include<algorithm>
    using namespace std;
    //---------------------------
    int main()
    {
    	int n;
    	cin>>n;
    	map<string,int> M;
    	long long count=0;
    	string s;
    	while(n--)
    	{
    		cin>>s;
    		count+=M[s];
    		reverse(s.begin(),s.end());
    		M[s]++;
    		 
    	}
    	cout<<count<<endl;
    }
