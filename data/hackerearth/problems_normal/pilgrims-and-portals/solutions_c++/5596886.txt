#include <bits/stdc++.h>
using namespace std;
#define ii pair<int,int>
#define zz pair<int,ii>
int T,n,m,d[110][110],k,pr[110];
vector<zz> q;
int parent(int u){
	if(pr[u]==0) return u;
	return pr[u] = parent(pr[u]);
}
int main(){
	//freopen("in.in","r",stdin);
	cin >> T;
	while(T--){
		cin >> n >> m >> k;
		q.clear();
		memset(pr,0,sizeof(pr));
		for(int i = 1; i <= n; i++){
			for(int j = i+1; j <= n; j++){
				d[i][j] = 1e9+1;
				d[j][i] = 1e9+1;
			}
		}
		for(int i = 1; i <= m; i++){
			int u,v,w;
			cin >> u >> v >> w;
			d[u][v] = w;
			d[v][u] = w;
		}
		for(int x = 1; x <= n; x++){
			for(int u = 1; u <= n; u++){
				for(int v = 1; v <= n; v++){
					d[u][v] = min(d[u][v],d[u][x]+d[x][v]);
				}
			}
		}
		for(int i = 1; i <= k; i++){
			for(int j = i+1; j <= k; j++){
				q.push_back(zz(d[i][j],ii(i,j)));
			}
		}
		sort(q.begin(),q.end());
		int ans = 0;
		for(int i = 0; i < q.size(); i++){
			int w = q[i].first;
			int u = q[i].second.first;
			int v = q[i].second.second;
			int pu = parent(u);
			int pv = parent(v);
			if(pu==pv) continue;
			ans += w;
			pr[pu] = pv;
		}
		cout << ans << "\n";
	}
}
