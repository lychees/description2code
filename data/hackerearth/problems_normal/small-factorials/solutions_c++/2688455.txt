int f[105][205];
int sz[105];

void init()
{
	f[1][0] = 1; sz[1] = 1;
	f[0][0] = 1; sz[0] = 1;
	for (int i=2 ; i<=100 ; i++)
	{
		int j = i;
		int z = 0;
		while (j)
		{
			int r = j%10;
			
			int t[205] , k=0 , kk=0 , c=0;
			for ( ; k<z ; k++) t[k]=0;
			for ( ; kk<sz[i-1] ; k++,kk++)
				t[k] = r*f[i-1][kk] + c , c=t[k]/10 , t[k]%=10;
			if (c) t[k++] = c;
			
			if (sz[i]==0) sz[i]=1;
			int p=0,q=0;
			c = 0;
			while (p<sz[i] || q<k)
				f[i][p] = f[i][p]+t[q]+c , c=f[i][p]/10 , f[i][p]%=10 , p++ , q++;
			if (c) f[i][p++] = c;
			sz[i] = p;
			
			j/=10;
			z++;
		}
		
		//cout<<i<<"\t";
		//for (int k=0 ; k<sz[i] ; k++) cout<<f[i][k]; cout<<"\n";
	}
}

int main()
{
	init();
	
    int t; int n;
	cin>>t;
    while (t--) { cin>>n; for (int i=sz[n]-1 ; i>=0 ; i--) cout<<f[n][i]; cout<<"\n"; }
    
    return 0;
}
