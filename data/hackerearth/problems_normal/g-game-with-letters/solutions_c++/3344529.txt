#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		map<int,int> m;
		string str;
		cin>>str;
		
		for(int i=0;i<str.length();i++)
		{
			if(m.find(str[i]-'a')==m.end())
				m[str[i]-'a']=1;
			else
				m[str[i]-'a']++;
		}
		string s;
		int res,ans=-1,len=-1,maxi=-1;
		for(int i=0;i<n;i++)
		{
			cin>>s;
			res=0;
			for(int j=0;j<s.length();j++)
				res+=m[s[j]-'a'];
			if(res>maxi)
			{
				maxi=res;
				ans=i+1;
				len=s.length();
			}
			else if(res==maxi)
			{
				if(s.length()<len)
				{
					ans=i+1;
					len=s.length();
				}
			}
		}
		cout<<ans<<endl;
	}
    return 0;
}
