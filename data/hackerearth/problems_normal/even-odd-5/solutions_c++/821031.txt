#include <iostream>
using namespace std;
int main()
{
	int t;
	//cin>>t;
	scanf("%d",&t);
	while(t--)
	{
		unsigned long long r,c,o=0,e=0;
		//cin>>r>>c;
		scanf("%llu %llu",&r,&c);
		if(r<=c)
		{
			if(r%2)
			{
				o=(((r+1)/2)*(1+r))/2;
				e=((r/2)*(1+r))/2;
			}
			else
			{
				o=((r/2)*(r))/2;
				e=((r/2)*(2+r))/2;
			}
		}
		else
		{
			if(c%2)
			{
				o=(((c+1)/2)*(1+c))/2;
				e=((c/2)*(1+c))/2;
				
				if((r-c)%2)
				{
					e+=(((r-c+1)/2)*c);
					o+=(((r-c)/2)*c);
				}
				else
				{
					e+=(((r-c)/2)*c);
					o+=(((r-c)/2)*c);
				}
			}
			else
			{
				o=((c/2)*(c))/2;
				e=((c/2)*(2+c))/2;
				
				if((r-c)%2)
				{
					e+=(((r-c)/2)*c);
					o+=(((r-c+1)/2)*c);
				}
				else
				{
					e+=(((r-c)/2)*c);
					o+=(((r-c)/2)*c);
				}
			}
			
		}
		printf("%llu %llu\n",e,o);
		//cout<<e<<" "<<o<<endl;
	}
	return(0);
}
