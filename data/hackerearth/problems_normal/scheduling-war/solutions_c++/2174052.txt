#include <bits/stdc++.h>
using namespace std;


int main()
{
    int sys,nob,nop,ch_pro=0,ch_sys,pr_pro=0,b_sys,j,b_pro,tot_pro;
    scanf("%d",&sys);
    int i;
    int pra[sys];
    
    for(i=0;i<sys;i++){
    	cin>>pra[i];
    	pr_pro += pra[i];
    }
    tot_pro = pr_pro;
    scanf("%d",&nob);
    for(i=0;i<nob;i++){
    	scanf("%d",&ch_sys);
    	nop = 0;
    	for(j=0;j<ch_sys;j++){
    		scanf("%d",&b_sys);
    		nop += pra[b_sys-1];
    	}
    	scanf("%d",&b_pro);
    	ch_pro += b_pro;
    	if(nop < b_pro)
    	    tot_pro += b_pro-nop;
    }
    int res = max(tot_pro-pr_pro,ch_pro-pr_pro);
    printf("%d",res);
    return 0;
}
