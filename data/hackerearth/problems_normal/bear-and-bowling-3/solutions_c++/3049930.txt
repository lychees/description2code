#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define DBG 0
const int N = 2e5 + 5, MOD = 1e9 + 7;

ll power[N];

int main(){

    if(DBG){
        freopen("input.in", "r", stdin);
        freopen("output.out", "w", stdout);
    }

    power[0] = 1;

    for(int i = 1; i < N; i++)  power[i] = (power[i - 1] * 2) % MOD;

    int n;
    ll temp;
    ll tot = 0;
    cin>>n;

    if(n == 1){
        cin>>temp;
        cout<<temp<<endl;
        return 0;
    }

    for(int i = 1; i <= n; i++){
        cin>>temp;
        ll lol = (power[n - 2] * temp) % MOD;
        lol = (lol * (i + 1)) % MOD;
        tot = (tot + lol) % MOD;
    }
    cout<<tot<<endl;
}
