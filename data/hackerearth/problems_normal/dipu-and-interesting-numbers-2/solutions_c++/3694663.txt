#include <iostream>
#include <math.h>
#include <stdio.h>
#define ll long long
using namespace std;

int main()
{
	ll int T, L, R, sqrtL, sqrtR, i_sqrtL, i_sqrtR, n;
    long double d_sqrtL, d_sqrtR;
    scanf("%lld", &T);
    while(T--)
    {
    	scanf("%lld %lld", &L, &R);

        //printf("%Lf %Lf %lf %lf \n", (long double)L, (long double)R, sqrt((long double)L), sqrt((long double)R));
        d_sqrtL = sqrt((long double)L);
        d_sqrtR = sqrt((long double)R);
        i_sqrtL = d_sqrtL;
        i_sqrtR = d_sqrtR;
        
        if (d_sqrtL==i_sqrtL && d_sqrtL*d_sqrtL > L)
            d_sqrtL -= 0.000001;
        else if (d_sqrtL==i_sqrtL && d_sqrtL*d_sqrtL < L)
            d_sqrtL += 0.000001;

        if (d_sqrtR==i_sqrtR && d_sqrtR*d_sqrtR > R)
            d_sqrtR -= 0.000001;
        else if (d_sqrtR==i_sqrtR && d_sqrtR*d_sqrtR < R)
            d_sqrtR += 0.000001;        
        
        //printf("%Lf %Lf \n", d_sqrtL, d_sqrtR);
    	sqrtL = ceil(d_sqrtL);
    	sqrtR = floor(d_sqrtR);
    	n = sqrtR-sqrtL+1;
        //printf("%lld %lld \n", sqrtL, sqrtR);
    	printf("%lld\n", n);
    }
    return 0;
}
