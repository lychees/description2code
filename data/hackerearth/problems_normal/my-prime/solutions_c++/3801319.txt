#include<iostream>
using namespace std;
int main()
{ unsigned short int n,j,k,h,i=0;
  cin>>n;
  int A[100];
  k=n;
  cout<<"\n";
  while(n>0)
  { cin>>A[i++];
    cout<<" ";
    n--;
  }
  cout<<"\n";
  for(i=0;i<k;i++)
  { h=0;
  	if(i!=0)
    {for(j=0;j<=(i-1);j++)
      {if(A[i]%A[j]==0)
        {h=1;
         break;
        }
      }
    }
    if(i!=k-1)
    { if(h==0)
     {for(j=i+1;j<k;j++)
      { if(A[i]%A[j]==0)
        { h=1;
          break;}
      }
    }
    }
    if(h==0)
     { cout<<A[i];
      cout<<" ";
     }
   }   
  return 0;
}
