#include <cstdio>

using namespace std;

int main()
{
	int n;
	scanf("%d", &n);
	int x, y, a, b, t;
	scanf("%d%d", &x, &y);
	a = x;
	b = y;
	
	while (b != 0) {
		t = b;
		b = a % b;
		a = t;
	}
	//printf("hello\n");
	long int lcm = 1;
	lcm = (x * y) / a;
	//printf("%ld\n", lcm);
	a = n / x;
	b = n / y;
	t = n / lcm;
	
	if (n % x == 0) {
		a--;
	}
	if (n % y == 0) {
		b--;
	}
	if (n % lcm == 0) {
		lcm--;
	}
	//printf("%d	%d	%d\n", a,b,t);
	long long int res = 1;
	res = x * ((a * (a + 1)) / 2) + y * ((b * (b + 1)) / 2) - lcm * ((t * (t + 1)) / 2) ;
	
	printf("%lld\n", res); 
	
	return 0;
}
