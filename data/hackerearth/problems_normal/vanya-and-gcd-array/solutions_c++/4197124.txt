#include<iostream>
// #include<conio.h>
#include <algorithm>
using namespace std;
// #define MOD 1000000007
int findGcd(int x, int y){
	return __gcd(x,y);
}
int main()
{
	int n;
	cin>>n;
	int arr[n];
	for(int i=0;i<n;i++)
		cin>>arr[i];
	long long result[n][101];
	for(int i=0;i<=n-1;i++)	{
		for(int j=0;j<=100;j++)
			result[i][j]=0;
	}
	result[0][arr[0]]=1;
	long long ans = result[0][1];
	int  temp=-1;
	for(int i=1;i<n;i++) {
   		for(int j=i-1;j>=0;j--) {
   			if(arr[i]>arr[j]){
   				for(int k=1;k<=100;k++){
   					result[i][findGcd(arr[i],k)] += result[j][k];
					result[i][findGcd(arr[i],k)] %= 1000000007;
				   }
			   }
    		}
    	result[i][arr[i]]++;
		ans += result[i][1];
		ans %= 1000000007;
	}
	cout<<ans<<endl;
	// getch();
	return 0;
}
