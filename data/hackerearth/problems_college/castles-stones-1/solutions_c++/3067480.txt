#include<cstdio>
#include<cstdlib>
#include<iostream>
#include<algorithm>
using namespace std;

int a[1010],visited[1010];

bool cmp(int a,int b)
{
	return (a>=b);
}


int main()
{
	int t,n,m,i,j,rem,count;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d",&m,&n);
		for(i=0;i<n;i++)
		scanf("%d",&a[i]);
		sort(a,a+n,cmp);
		count=0;
		for(i=0;i<n;i++)
		{
			if(a[i]<=m)
			if(!visited[i])
			{
				visited[i]=1;
				rem=m-a[i];
				count++;
				for(j=i+1;j<n;j++)
				{
					if(a[j]<=rem && !visited[j])
					{
						visited[j]=1;
						break;
					}
					
				}
				
			}
		}
		for(i=0;i<n;i++)
		visited[i]=0;
		printf("%d\n",count);
	}
}
