#include<iostream>
#include<string>
#include<sstream>
using namespace std;
inline void scanint(int *a)
{
	register char c=0;
	while(c<33)
	c=getchar();
	*a=0;
	while(c>33)
	{
	  *a=*a*10+c-'0';
	  c=getchar();	
	}
}
bool F(bool dp[],int n)
{	
  ostringstream ss;
  ss<<n;
  string s=ss.str();
  int i=0,temp;
  while(s[i]!='\0')
  {
  	temp=n-(s[i]-48)*(s[i]-48);
    if(temp>0)
  	{
      if(dp[temp])
      return true;  
	}
	i++;
  }
  return false;
}
int main()
{
	bool dp[1000001];
	int A[]={1,4,27,256,3125,46656,823543,0};
	int k=1,s=1,t;
