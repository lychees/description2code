#include <iostream>
#include <stdlib.h>
#include <algorithm>
using namespace std;

int main()
{
    long long int n,min;
    min=10000000000;
    cin>>n;
    int k,a[n];
    cin>>k;
    for(int i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    sort(a,a+n);
    for(int i=k-1;i<n;i++)
    {
    	if(min>(a[i]-a[i-k+1]))
    	{
    		min=(a[i]-a[i-k+1]);
    	}
    }
    cout<<min;
    return 0;
}
