#include <iostream>
using namespace std;

int main()
{
    int a,b,n,t;
    cin>>a>>b>>n;
    
    if(n==1)
    	cout<<a<<endl;
    else if(n==2)
    	cout<<b<<endl;
    else if(n==3)
    	cout<<a+b<<endl;
    else
    {
    	n-=2;
    	while(n>0)
    	{
    		t=a+b;
    		a=b;
    		b=t;
    		n--;
    	}
    	cout<<b<<endl;
    }
    
    return 0;
}
