#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int i,a,b,sum,d,s1=0,e,a1,b1,a2,b2;
    for(i=0;i<n;i++)
    {
    	s1=0,a1=0,b1=0;
    	cin>>a>>b;
    	while(a>0)
    	{
    		a2=a%10;
    		a=a/10;
    		a1=a1*10+a2;
    	};
    	while(b>0)
    	{
    		b2=b%10;
    		b=b/10;
    		b1=b1*10+b2;
    	};
    	sum=a1+b1;
    	d=sum;
    	while(d>0)
    	{
    		e=d%10;
    		d=d/10;
    		s1=s1*10+ e;
    	};
    	cout<<s1<<endl;
    }
    return 0;
}
