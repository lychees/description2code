#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main(){
	int n;
	cin>>n;
	set<string> s;
	map<string,int> mp;
	pair<map<string,int>::iterator,bool> ck;
	while(n--){
		string str;
		cin>>str;
		
		ck= mp.insert(make_pair(str,1));
		if(!(ck.second))
			mp[str]=mp[str]+1;
		s.insert(str);
	}
	set<string>::iterator it=s.begin();
	while(it!=s.end()){
		cout<<*it<<" "<<mp[*it]<<endl;
		it++;
	}
}
