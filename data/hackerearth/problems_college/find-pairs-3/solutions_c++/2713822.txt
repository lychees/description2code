#include <iostream>
using namespace std;
//konoha's yellow flash
int a[1000002];
int main()
{
	int i;
	int j;
	int t;
	cin >>t ;
	int n;
	int counti = 0;
	while(t--)
	{
		counti = 0;
		cin >>n ;
	for( i = 0 ; i < n ; i++ )
	{cin >> a[i];
	}
	for(i = 0 ; i <n ; i++)
	{for(j = i ; j < n ; j++)
	{if(a[i] == a[j])
	{
		counti++;
	}
	}
	}
	cout << counti << endl;
	}
	return 0;
}
