#include <iostream>
using namespace std;

int main()
{
    int n,q;
    cin>>n>>q;
    int a[n+1],b[n+1];
    for(int i=1;i<=n;i++){
    	cin>>a[i];
    }
    b[0]=0;
    b[1]=a[1];
    for(int i=2;i<=n;i++){
    	b[i]=a[i]+b[i-1];
    }
    int l,r,ansa,ansd,x,y;
    while(q--){
    	cin>>l>>r;
    	ansa=b[r]-b[l-1];
    	x=b[r]-b[r-1];//a[r]
    	y=b[r-1]-b[l-1];
    	ansd=x-y;
    	cout<<ansa<<" "<<ansd<<"\n";
    }
    return 0;
}
