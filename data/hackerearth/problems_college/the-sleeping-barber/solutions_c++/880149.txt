#include <bits/stdc++.h>
using namespace std;
int cmp(const void *a, const void *b)
{
	return (*(int*)a-*(int*)b);
}
int main()
{
    int D,G,M;
    scanf("%d",&D);
    while(D--)
    {
    	scanf("%d%d",&G,&M);
    	int a[G];
    	for(int i = 0; i<G;i++)
    		scanf("%d",&a[i]);
    	qsort(a,G,sizeof(int),cmp);
    	int i;
    	for(i =0; i<G;i++)
    	{
    		if(a[i]<=M) M -= a[i];
    		else break;
    	}
    	printf("%d\n",i);
    }
    return 0;
}
