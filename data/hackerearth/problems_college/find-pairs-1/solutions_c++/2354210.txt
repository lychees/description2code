#include<iostream>
#define max 100000
using namespace std;
int main()
{
	int tc,i,j,a[max],n,count;
	cin>>tc;
	while(tc--)
	{
		count=0;
		cin>>n;
		for(i=0;i<n;i++)
			cin>>a[i];
		for(i=0;i<n;i++)
			for(j=i;j<n;j++)
				if(a[i]==a[j])
					count++;
		cout<<count<<endl;
	}
	return 0;
}
