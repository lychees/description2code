#include<iostream>
#include<cstdio>
using namespace std;
typedef struct
{
    int n;
    int ctr;
}stair;
stair arr[10005];
void precompute()
{
    int sum=0;
    for(int i=0; i<500; i++)
    {
        sum=sum+i;
        arr[i].n=sum;
        arr[i].ctr=i;
    }

}

int main()
{
    int i,sum=0,low,high,key,t;
    #ifndef ONLINE_JUDGE
    freopen("i_p","r",stdin);
#endif
    cin >> t;
    precompute();
    while(t--)
    {
       cin >> key;

    low=0;
    high=499;
    while(low<=high)
    {
    int mid=(low+high)/2;
        if(key>=arr[mid].n && key<arr[mid+1].n)
        {
            cout << arr[mid].ctr << "\n";
            break;
        }
        else if(key>arr[mid].n)
        low=mid+1;
        else
        high=mid-1;
    }
    }

    return 0;
}
