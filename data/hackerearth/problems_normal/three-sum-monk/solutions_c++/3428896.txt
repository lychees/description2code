#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t,n;
	cin >> t;
	vector<bool> cube(10000,false);
	for(int i=0;i<=20;i++){
		cube[i*i*i] = true;
	}
	while(t--){
		cin >> n;
		double ans=0;
		vector<int> a(n);
		for(int i=0;i<n;i++){
			scanf("%d",&a[i]);
		}
		for(int i=0;i<n;i++){
			for(int j=i+1;j<n;j++){
				for(int k=j+1;k<n;k++){
					if(cube[a[i] + a[j] + a[k]]) ans++;
				}
			}
		}
		printf("%.9f\n",(ans*6.0000000)/(1.000*n*(n-1)*(n-2)));
	}
    return 0;
}
