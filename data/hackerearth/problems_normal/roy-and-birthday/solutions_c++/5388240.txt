#include <bits/stdc++.h>
#define ll long long int
const int MOD = 1000000007;
using namespace std;
const int N = 1000001;
ll f[N+1], rf[N+1];

ll pow_mod(ll a, ll b){
	ll ans = 1;
	while(b){
		if(b&1) ans = (ans*a)%MOD;
		a = (a*a)%MOD;
		b >>= 1;
	}
	return ans%MOD;
}

ll inv(ll a){
	return pow_mod(a,MOD-2);
}

void fact() {
	f[0] = rf[0] = 1;
	for(ll i=1; i<=N; ++i){
		f[i] = (f[i-1]*i)%MOD;
	}
	rf[N] = inv(f[N]);
	for(ll i=N-1; i>=0; --i){
		rf[i] = ((i+1)*rf[i+1])%MOD;
	}
}

ll C(ll n, ll r) {
	// cout<<f[n]<<" "<<rf[r]<<" "<<rf[n-r]<<"khatam"<<endl;
	//cout<<((rf[r]*rf[n-r])%MOD);
	if(r < 0 || r > n) return 0;
	return ((f[n]*rf[r])%MOD*rf[n-r])%MOD;
}

int main(){
	fact();
	int t;
	string s;
	int a[26];
	cin>>t;
	while(t--){
		cin>>s;
		int len = s.size();
		memset(a,0,sizeof(a));
		for(int i=0; i<len; ++i){
			a[s[i]-'a']++;
		}
		int vow=0, tot_inv=1;
		for(int i=0; i<26; ++i) {
			if(i == 0 || i == 4 || i == 8 || i == 14 || i == 20) {
				vow += a[i];
			}
			tot_inv = (tot_inv*rf[a[i]])%MOD;
		}
		int cons = len - vow;
		ll a,b;
		a = (C(cons+1, vow)*f[vow])%MOD;
		 //cout<<a<<" "<<f[vow]<<endl;
		b = (f[cons]*tot_inv)%MOD;
		 //cout<<a<<" "<<b<<endl;
		if(vow > cons+1) printf("-1\n");
		else printf("%lld\n", (a*b)%MOD);
	}
    return 0;
}
