def isCoprime(a, b):
	upperLimit = min(a,b)
	commonF = 1
	
	for j in range(2, upperLimit + 1):
		if(a % j == 0 and b % j == 0):
			commonF = j
		
	if(commonF != 1):
		return False
	else:
		return True
			

def friendCount(MaxNo, otherNumbers):
	friendCount = 0
	for n in otherNumbers:
		if(isCoprime(MaxNo, int(n)) == False):
			friendCount += 1
	return 	friendCount
