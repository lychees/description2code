#include<iostream>

using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long x,y,count=0;
        cin>>x>>y;
        long long n=x^y;
        while(n)
        {
            count++;
            n&=n-1;
        }
        cout<<count<<"\n";
    }
    return 0;
}
