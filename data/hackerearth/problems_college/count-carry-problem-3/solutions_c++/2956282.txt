#include <iostream>
#include <cstdio>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	long long int k1,k2;
    	scanf("%lld %lld",&k1,&k2);
    	int c = 0,c1 = 0;
    	while(k1!=0||k2!=0)
    	{
    		int t1 = k1%10;
    		int t2 = k2%10;
    		if((t1+t2+c1)/10>0)
    		{
    			c1 = 1;
    			c++;
    		}
    		else
    		{
    			c1 = 0;
    		}
    		k1 = k1/10;
    		k2 = k2/10;
    	}
    	if(c==0)
    	{
    		printf("No carry operation\n");
    	}
    	else if(c==1)
    	{
    		printf("1 carry operation\n");
    	}
    	else
    	{
    		printf("%d carry operations\n",c);
    	}
    }
    return 0;
}
