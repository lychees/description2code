#include <bits/stdc++.h>
using namespace std;
#define ll long long

// O(n) phi sieve
#define MAXN 1000000
ll phi[MAXN + 1], prime[MAXN/10], sz;
bitset <MAXN + 1> mark;
 

int main() {
  ios_base::sync_with_stdio(0);
  
for (int i = 2; i <= MAXN; i++ ){
	if(!mark[i]){
		phi[i] = i-1;
		prime[sz++]= i;
	}
	for (int j=0; j<sz && prime[j]*i <= MAXN; j++ ){
		mark[prime[j]*i]=1;
		if(i%prime[j]==0){
			phi[i*prime[j]] = phi[i]*prime[j];
			break;
		}
		else phi[i*prime[j]] = phi[i]*(prime[j]-1 );
	}
}

  for(int i=1; i<=MAXN; i++) {
    phi[i]+=phi[i-1];
  }
  int t;
  cin>>t;
  while(t-->0) {
    int n;
    cin>>n;
    cout<<(n*1LL*(n-1))/2-phi[n]<<'\n';
  }
  
  return 0;
}
