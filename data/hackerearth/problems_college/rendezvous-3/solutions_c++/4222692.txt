int mult(int x,int y)
{
  ll ans,x1=(ll)x,y1=(ll)y;
  ans=(x1*y1)%mi;
  return (int)ans;
}
int gcd(int a,int b) { return b==0 ? a : gcd (b,a%b);}
int lcm(int a,int b) { return a*(b/gcd(a,b));}

ll pow1(ll a,ll b)
{
  ll ans=1;
  while(b>0)
    {
      if(b&1) ans=(ans*a)%mi;	
      a=(a*a)%mi; b>>=1;
    }	
  return ans;
}
int mina(int arr[],int n)
{
  int x=arr[0],i,pos=0;
  rep(i,1,n){ if(arr[i]<x) { x=arr[i]; pos=i; } }
  return x;
}

ll p[300005],r[300005];
ll finda(ll i)
{
  if(p[i]==i)
    return i;
  else
    return p[i]=finda(p[i]);
}
void makeunion(ll a,ll b)
{
  ll a1=finda(a);
  ll a2=finda(b);
  if(r[a1]<r[a2])
    p[a1]=a2;
  else if(r[a2]<r[a1])
    p[a2]=a1;
  else
    {
      r[a1]++;
      p[a2]=a1;
    }
}
bool cmp(pair< pair<ll,ll>,pair<ll,ll> > a,pair< pair<ll,ll>,pair<ll,ll> > b)
{
  return a.first.first<b.first.first;
}
