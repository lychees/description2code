#include "bits/stdc++.h"
using namespace std;
long int abs(long int a,long int b){
	return a>b ? a-b : b-a;
}
int main()
{
	long int tc;
	cin>>tc;
	while(tc--)
	{
		string s;
		cin>>s;
		long int ans=0;
		for(long int i=0,j=s.length()-1;i<j;i++,j--)
		{
			long int a=s[i];
			long int b=s[j];
			ans=ans+abs(a,b);
		}
		cout<<ans<<endl;
	}
	return 0;
}
