#include<bits/stdc++.h>


using namespace std;

int main(){


    int T,N;
    cin>>T;
    for (int i=0;i <T;i++){

        cin>>N;

        if(N==1 || N==2){
            cout<<"0"<<endl;
        }else{
            int bits=0;
            N+=1;
            while(N){   N/=2; bits++; }

            cout<<bits-2<<endl;

        }
    }

    return 0;
}
