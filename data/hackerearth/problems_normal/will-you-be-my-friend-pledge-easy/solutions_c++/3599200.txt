#include <iostream>
using namespace std;

int main()
{
    int n,t,i,j,count=0;
    cin>>n>>t;
    int a[t];
    for(i=0;i<t;i++)
    {
    	cin>>a[i];
    }
    for(i=0;i<t;i++)
    {
    	for(j=min(n,a[i]);j>=1;j--)
    	{
    		if(a[i]%j==0&&n%j==0)
    			break;	
    	}
    	if(j!=1)
    	count++;
    }
    cout<<count<<endl;
    return 0;
}
