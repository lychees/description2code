#include<bits/stdc++.h>
using namespace std;
#define ll unsigned long long

int fac(ll n)
{
	ll i,max=1,c=0;
	for(i=2;i<=sqrt(n);i++)
	{
		while(n%i == 0)
		{
			n = n/i;
			c++;
		}
		max = max*(c+1);
		c=0;
	}
	if(n>1)
	{
		max = max*2;
	}
	return max;
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		ll n;
		scanf("%llu",&n);
		ll max = fac(n);
		cout<<max<<"\n";
	}
}
