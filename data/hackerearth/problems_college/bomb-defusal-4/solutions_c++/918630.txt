#include <iostream>
using namespace std;
#define INF 12
int solve(int a,int b,int u, int v)
{
	a=abs(a);
	b=abs(b);
	u=abs(u);
	v=abs(v);
	if(a==u && b==v) return 0;
	
	if((a%2==0 && b%2==0)||(a%2==1 && b%2==1))
	{
		if((u%2==0 && v%2==1) || (u%2==1 && v%2==0))
		return INF;
		else 
		{
			if(abs(b-v)!=abs(a-u)) return 2;
			else return 1;
		}
	}
	else 
	{
		if((u%2==0 && v%2==0) || (u%2==1 && v%2==1))
		return INF;
		else
		{
			if(abs(b-v)!=abs(a-u)) return 2;
			else return 1;
		}
	}
}


int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    int P,a,b,u,v;
    while(t--)
    {
    	scanf("%d%d%d%d%d",&P,&a,&b,&u,&v);
    	int ans=solve(a,b,u,v);
    	if(ans<P)printf("Counter-Terrorists Win !\n");
    	else printf("Terrorists Win !\n");
    }
    return 0;
}
