#include <iostream>
#include <stdio.h>
using namespace std;
int main() {
  char a[1100], c;
  int t;
  cin >> t;
  while (t > 0) {
    cin >> a;
    cin >> c;
    int p = 0;
    for (int i = 0; a[i] != '\0'; i++) {
      if (c == a[i]) {
        ++p;
      }
    }
    cout << p << endl;
    t--;
  }
  return 0;
}
