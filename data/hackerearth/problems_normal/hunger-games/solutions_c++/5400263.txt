#include<bits/stdc++.h>

using namespace std;

int main() {

    int n;
    cin>>n;
    int arr[n];
    for(int i=0;i<n;i++) {
        cin>>arr[i];
    }
    sort(arr,arr+n);
    int m=arr[n-1]-arr[n-3];
    for(int i=n-2;i>1;i--) {
        m=max(m,arr[i]-arr[i-2]);
    }
    cout<<m;
    return 0;
}
