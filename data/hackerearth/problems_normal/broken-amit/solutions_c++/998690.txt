#include<cstdio>
#include<iostream>
#include<vector>
#include<algorithm>
#include<string>
#include<cstring>
using namespace std;

typedef long long LL;
typedef vector<int> VI;

#define REP(i,n) for(int i=0; i<int(n); i++)
#define EACH(i,c) for(__typeof((c).begin()) i=(c).begin();i!=(c).end();i++)

LL tri(int n) { return n * (n+1) / 2; }

int N;
int D[21];
LL sum, Z ;

int main() {

    scanf("%d", &N);
    REP (i, N) {
	int x;
	scanf("%d", &x);
	sum += x;
    }
    REP (i,N) {
	scanf("%d", D+i);
	D[i] *= i;
    }
    sort(D, D+N);

    sum *= N;

    REP (i, N) {
	sum -= tri(i) * D[N-1-i] + Z;
	Z += LL(i) * D[N-1-i];
    }

    printf("%lld\n", sum);

    return 0;
}
