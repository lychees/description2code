n, m = map(int, raw_input().split())
c = map(int, raw_input().split())
nums = map(int, raw_input().split())
i = 0
num = 0
while i < n:
    div = c[nums[i] - 1]
    lst = c[nums[i]:]
    x = num + (div - num % div)
    while True:
        for item in lst:
            if x % item == 0:
                break
        else:
            break
        x += div
    num = x
    i += 1
print num
