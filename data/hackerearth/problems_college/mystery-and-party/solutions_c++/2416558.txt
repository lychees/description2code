#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	int t;
    cin>>t;
    while(t>0)
    {
    	long int n, k;
    	cin>>n>>k;
    	long int A[n];
    	for(int i=0; i<n; i++)
    		cin>>A[i];
    	std::sort(A, A+n);
    	printf("%ld\n", A[k-1]);
    	t--;
    }
    return 0;
}
