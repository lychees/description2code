n = input()
l = []
no = 0
remo = []
while n > 0:
    n = n-1
    name = raw_input()
    l.append(name)
for i in l:
    c = l.count(i)
    if c > 1:
        remo.append(i)
        while c > 1:
            c = c-1
            l.remove(i)
        no = no+1
    # print i
remo.sort()
