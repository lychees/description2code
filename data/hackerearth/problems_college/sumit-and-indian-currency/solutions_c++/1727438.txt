#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main() 
{
    int i,l,t;
    cin>>t;
    string str;
    while(t--) {
    	cin>>str;
    	l = str.size();
    	int c = 0;
    	for(i = l-1; i >= 0; i--) {
    		if(i == 0) continue;
    		if(c == 2) {
    			str.insert(i,",");
    		}
    		else if(c > 3 && c%2 == 0) {
    			str.insert(i,",");
    		}
    		c++;
    	}
    	cout<<str<<endl;
    }
	return 0;
}
