#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#include <bits/stdc++.h>
using namespace std;


int main() {
    int n,i,j,k,sum =0,max =-1,max_index =0,x;
    vector< int>v;
    
    cin>>n;
    for(i = 0; i  < n ; i ++) {
        
        cin>>x;
        v.push_back(x);
        
    }
    while (v.size() != 2) {
        vector<int>::iterator it;
        max = -1;
        max_index =-1;
        for(i = 1;i<v.size()-1;i++) {
            
            if(v[i-1]*v[i+1] > max) {
                max= v[i-1]*v[i+1];
                max_index = i;
            }
        }
       // cout<<max_index<<endl;
        sum =  sum + max;
        v.erase(v.begin()+max_index);
    }
    cout<<sum<<endl;
    
    
}
