#include <bits/stdc++.h>
using namespace std;

int main()
{
  //  cout << "Hello World!" << endl;
    string a,b;
    cin>>a>>b;
    string c=a;
    string d=b;
    for(int i=0;i<a.length();i++){
    	if(a[i]=='6')a[i]--;
    }
    for(int i=0;i<b.size();i++)
    	if(b[i]=='6')b[i]--;
    for(int i=0;i<c.size();i++)
    	if(c[i]=='5')c[i]++;
    for(int i=0;i<d.size();i++)
    	if(d[i]=='5')d[i]++;
    //	cout<<d<<" ";
    printf("%d %d",atoi(a.c_str())+atoi(b.c_str()),atoi(c.c_str())+atoi(d.c_str()));
    return 0;
}
