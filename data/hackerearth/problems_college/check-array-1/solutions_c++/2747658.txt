#include<stdio.h>

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		int arr[n+1];
		for(int i=0;i<n;i++)
		scanf("%d",&arr[i]);
		int sumleft=0,sumright=0;
		int flag=0;
		for(int i=0;i<n;i++)
		sumright=sumright+arr[i];
		for(int i=0;i<n;i++)
		{
			if(i>0)
			sumleft=sumleft+arr[i-1];
			else
			sumleft=0;
			if(i==n-1)
			sumright=0;
			else
			{
				sumright=sumright-arr[i];
			}
			
			/*for(int j=i-1;j>=0;j--)
			{
				sumleft=sumleft+arr[j];
				if(sumleft>sumright)
				break;
			}*/
			if(sumleft==sumright)
			{
				flag=1;
				break;
			}
		}
		if(flag==1)
		printf("YES\n");
		else
		printf("NO\n");
	}
	return 0;
}
