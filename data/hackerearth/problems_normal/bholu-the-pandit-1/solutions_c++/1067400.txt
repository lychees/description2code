#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll hi = 1000000007;
ll gcd(ll a,ll b)
    {
    ll t;
    while(a%b){
        t=a%b; a=b ;b = t;
    }
    return b;
}

int main()
    {
    int t,m,n;
    ll ans ;
    scanf("%d",&t);
    ll a,b,c;
    while(t--)
        {
        scanf("%lld%lld%lld",&a,&b,&c);
        ll g = gcd(a,b);
        g = gcd(g,c);
        a/=g;
        b/=g;
        c/=g;
        ll v = a;
        v = (v*b)%hi;
        v = (v*c)%hi;
        printf("%lld %lld\n",g,v);
    }
    return 0;
}
