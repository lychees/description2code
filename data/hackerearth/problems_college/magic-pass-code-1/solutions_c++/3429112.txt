#include <iostream>
#include <cstdlib>
#include <vector>
#include <sstream>
#define ll long long

using namespace std;

ll convert(string x) {
		int res;
		stringstream(x) >> res;
	return res;
}


int main() {
	
	string str;
	ll ctr = 0;
	ll ctr_2 = 0;
	ll max;
	
	while(getline(cin,str)) {
		
		for(ll i=0; i<str.length(); i++) {
			ctr_2 += str[i]-'0';
		}
		
		if(ctr_2>ctr) {
			ctr = ctr_2; 
			max = convert(str);
		}
		
		
		
		ctr_2 = 0;
	}
	cout << max << endl;
	
	return 0;
}
