#include <bits/stdc++.h>

using namespace std;

void min_heapify(int *a,int i,int n)

{

    int j, temp;

    temp = a[i];

    j = 2 * i+1;

    while (j <= n)

    {

        if (j < n && a[j+1] < a[j])

            j = j + 1;

        if (temp < a[j])

            break;

        else if (temp >= a[j])

        {

            a[j/2] = a[j];

            j = 2 * j;

        }
    }

    a[j/2] = temp;

    return;

}

void build_minheap(int *a, int n)
{

    int i;

    for(i = n/2; i >= 1; i--)
    {

        min_heapify(a,i,n);

    }

}

int main()
{
    //freopen("in.txt","r",stdin);
    int n;
    int arr[1000];
    int a = 0;
    while(cin >> n)
    {
        arr[a] = n;
        a++;
    }
    //cout << a << endl;
    sort(arr,arr+a);
    build_minheap(arr, n);
    cout << arr[0] << endl;

    return 0;
}
