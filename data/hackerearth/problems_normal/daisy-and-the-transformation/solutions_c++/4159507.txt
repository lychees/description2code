#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t,n,l,i;
    char s1[50],s2[50];
    string s;
    cin>>t;
    while(t--)
    	{
    		cin>>s1;
    		//cout<<s1<<" ";
    		l=strlen(s1);
    		
    		for(i=0;i<l;i++)
    			cin>>s2[i];
    		s2[l]='\0';
    	//	cout<<s2;
    		//getline(cin,s);
    		
    		cin>>n;
	    		int j=0;
	    		for(i=0;(i<l);i++)
	    		{
	    			if(s1[i]!=s2[i])
	    			{
	    				j++;
	    			}
	    		}	
	    	
	    	//	cout<<s1<<" "<<s2;
	    		if(j>n)
	    			cout<<"No"<<endl;
	    		else if(j==n || (n-j)%2==0)
	    			cout<<"Yes"<<endl;
	    		else
	    			cout<<"No"<<endl;
    	
    	//	}    			
    	}
    return 0;
}
