#include <iostream>
#include <vector>

using namespace std;

int main() {
    int t;
    scanf("%d",&t);
    while(t--){
        int n;
        scanf("%d",&n);
        vector <long int> outputs[61];
        while(n--){
            long int x;
            scanf("%ld",&x);
            int no1s=__builtin_popcount(x);
            if(no1s==1){
                printf("%ld",x);
                printf(" ");
            }
            else{
                outputs[no1s].push_back(x);
            }
        }
        for(int i=2;i<61;i++){
                for(vector <long int>::iterator it=outputs[i].begin();it!=outputs[i].end();it++){
                    printf("%ld",*it);
                    printf(" ");
                }
        }
        printf("\n");
    }
    return 0;
}
