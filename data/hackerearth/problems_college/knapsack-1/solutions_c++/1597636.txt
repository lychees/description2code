#include<bits/stdc++.h>
using namespace std;

int main() {
    int T,N,K;
    int item[2000];
    int opt[2001];
    cin>>T;
    while(T--){
        cin>>N>>K;
        for(int i=0;i<N;i++)
            cin>>item[i];
        opt[0]=0; int m=0;
        for(int i=1;i<=K;i++){
            for(int j=0;j<N;j++){
                if(i-item[j]>=0){
                    m=max(m,opt[i-item[j]]+item[j]);
                }
            }
            opt[i]=m;
            m=0;
        }
        cout<<opt[K]<<"\n";
    }
    return 0;
}
