#include <bits/stdc++.h>
#define ll long long
using namespace std;
int marks[110][110];
string s[110];
int t[110],b[110];//best or not
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	memset(b,0,sizeof(b));
	int n,m;
	cin>>n>>m;
	int i,j;
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			marks[i][j]=0;
		}
	}
	for(i=0;i<n;i++)
	{
		cin>>s[i];
		for(j=0;j<m;j++)
		{
			marks[i][j]=s[i].at(j)-'0';
			t[j]=max(t[j],marks[i][j]);
		}
	}
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			if(marks[i][j]==t[j])
			b[i]=1;
		}
	}
	int ans=0;
	for(i=0;i<n;i++)
	ans+=b[i];
	cout<<ans<<endl;
	return 0;
}
