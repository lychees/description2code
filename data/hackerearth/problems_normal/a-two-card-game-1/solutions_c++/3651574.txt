#include<bits/stdc++.h>
using namespace std;
#define ll	long long int
int main()
{
	ll n,tmp;
	scanf("%d",&n);
	int a[n];
	for(int i=0;i<n;++i)
	{
		scanf("%d",&a[i]);
	}
	ll win=0,cnt=0;
	sort(a,a+n);
	tmp=a[0];
	cnt=1;
	for(int i=1;i<n;++i)
	{
		if(a[i]==tmp)
		{
			cnt++;
			continue;
		}
		else
		{
			tmp=a[i];
			win+=cnt*(n-i);
			cnt=1;
		}
	}
	double winp=(win*1.0)/(n*(n-1));
	printf("%0.6f %0.6f\n",winp,winp);
	return 0;
}
