#include<stdio.h>
#include<iostream>
using namespace std;
int main()
{
    int n,i,s=2,k,j;
    cin>>n;
    int b[n],a[n];
    for(i=0;i<n;i++)
    {
        cin>>b[i];
    }
    a[0]=b[0];
    for(i=1;i<n;i++)
    {
        k=0;
        for(j=0;j<i;j++)
        {
            k=k+a[j];
        }
        a[i]=(b[i]*s)-k;
        s++;
    }
    for(i=0;i<n;i++)
    {
        cout<<a[i]<<" ";
    }
    return 0;
}
