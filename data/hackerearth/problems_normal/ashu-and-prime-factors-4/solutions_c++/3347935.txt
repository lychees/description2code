#include <iostream>
#include<algorithm>
#define max 1000000
using namespace std;

int primes[max+1];

void sieve()
{
	int count;
	for(int i=2;i<=sqrt(max);i++)
	{
	count=0;	
		for(int j=1;i*j<=max;j++)
		{
			if(primes[i*j]==0)
			{
				primes[i*j]=i;
				count++;
			}
		}
		primes[i]=count;
	}
	
	for(int i=sqrt(max);i<=max;i++)
	{
		if(primes[i]==0)
		primes[i]=1;
	}
}

int main()
{
    int t,x;
    cin>>t;
    sieve();
    while(t--)
    {
    	cin>>x;
    	cout<<primes[x]<<endl;
    }
    return 0;
}
