#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n,x;
    	cin>>n>>x;
    	long long int arr[n+4],sum=0;
    	for(int i=1;i<=n;i++){
    		cin>>arr[i];
    		sum += arr[i];
    	}
    	if(sum <= x)
    		cout<<"YES\n";
    	else{
    		long long int dp[2][x+4];
    		for(int i=0;i<=x;i++){
    			dp[0][i] = 0;
    		}
    		for(int i=1;i<=n;i++){
    			for(int j=0;j<=x;j++){
    				dp[i%2][j] = dp[(i-1)%2][j];
    				if(j>=arr[i]){
    					dp[i%2][j] = max(dp[i%2][j],dp[(i-1)%2][j-arr[i]] + arr[i]);
    				}
    			}
    		}
    		if(sum - dp[n%2][x] <= x)
    			cout<<"YES\n";
    		else
    			cout<<"NO\n";
    	}
    }
    return 0;
}
