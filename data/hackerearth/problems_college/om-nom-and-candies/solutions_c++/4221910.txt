#include <iostream>
using namespace std;

int main()
{
    int *maxAt,*arr;
    int n;
    cin>>n;
    maxAt = new int[n];
    arr = new int[n];
    for(int i=0;i<n;++i)
    {
    	cin>>arr[i];
    	maxAt[i]=arr[i];
    }
    for(int i=1;i<=n;++i)
    {
    	for(int j=2*i;j<=n;j+=i)
    	{
    		if(maxAt[j-1]<maxAt[i-1]+arr[j-1])
    			maxAt[j-1]=maxAt[i-1]+arr[j-1];
    	}
    }
    for(int i=0;i<n;++i)
    cout<<maxAt[i]<<" ";
    return 0;
}
