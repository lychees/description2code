#include <iostream>
#include <cmath>
using namespace std;

bool isprime(int n)
{
	bool ret = 1;
	for(int i=2; i<=int(sqrt(n)); i++)
	{
		if(i != n)
		{
			if(n % i == 0)
			{
				ret = 0;
				break;
			}
		}
	}
	return ret;
}


int main()
{
    int t;
    cin >> t;
    while(t > 0)
    {
    	string str;
    	cin >> str;
    	bool allprimes = 1;
    	for(int i=0; i<str.length(); i++)
    	{
    		char c = str[i];
    		int a = c;
    		if(!isprime(a))
    		{
    			allprimes = 0;
    			cout << c;
    		}
    	}
    	if(allprimes)
    	{
    		cout << "NULL";
    	}
    	cout << endl;
    	t--;
    }
    return 0;
}
