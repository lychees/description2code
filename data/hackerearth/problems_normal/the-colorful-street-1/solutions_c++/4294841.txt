#include<bits/stdc++.h>
using namespace std;
#define lli long long int
#define maxx 25;
int main()
{
	lli t;
	cin>>t;
	while(t--)
	{
		lli n;
		cin>>n;
		lli arr[n][3];
		lli i,j;
		for(i=0;i<n;++i)
		{
			for(j=0;j<3;++j)
			{
				cin>>arr[i][j];
			}
		}
		lli dp[n][3];
		memset(dp,0,sizeof dp);
		for(i=0;i<3;++i)
		dp[0][i]=arr[0][i];
		for(i=1;i<n;++i)
		{
			for(j=0;j<3;++j)
			{
				if(j==1)
				dp[i][j]=arr[i][j]+min(dp[i-1][j-1],dp[i-1][j+1]);
				 if(j==2)
				dp[i][j]=arr[i][j]+min(dp[i-1][j-1],dp[i-1][j-2]);
				if(j==0)
				dp[i][j]=arr[i][j]+min(dp[i-1][j+1],dp[i-1][j+2]);
			}
		}
		cout<<min(dp[n-1][0],min(dp[n-1][1],dp[n-1][2]))<<endl;
	}
	return 0;
}
