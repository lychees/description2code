#include <iostream>
using namespace std;

int check_palindrome(string str)
{
	int start = 0, end = str.length()-1;
	while(start<end)
	{
		if(str[start]!=str[end])
		return 0;
        start++;
        end--;
	}
	return 1;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string a,b;
    	cin>>a>>b;
    	int count = 0;
    	int n1 = a.length();
    	int n2 = b.length();
    	string str;
    	str.append(b);
    	str.append(a);
    	if(check_palindrome(str))
    	count++;
    	for(int i=0; i<n1; i++)
    	{
    		string str;
    		str.append(a,0,i+1);
    		str.append(b);
    		str.append(a,i+1,n1-i-1);
    		if(check_palindrome(str))
    		count++;
    	}
    	cout<<count<<"\n";
    }
    return 0;
}
