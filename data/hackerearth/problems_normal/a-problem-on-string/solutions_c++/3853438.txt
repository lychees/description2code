#include <iostream>
#include<string.h>
using namespace std;

int main()
{
//    cout << "Hello World!" << endl;
  	long long int n,q,cx,cy,count[50],i,j,k;
  	char s[1000001],x,y;
  	cin>>n;
  	cin>>s;
  	cin>>q;
  	
  	memset(count,0,sizeof(count));
  	for(i=0;i<n;i++)
  	{
  		count[s[i]-97]++;
  	}
  	for(j=1;j<=q;j++)
  	{
  		cin>>x>>y;
  		 cx=count[x-97];
  		 cy=count[y-97];
  		 if(x!=y)
  		 cout<<(cx*cy)<<endl;
  		 else {
  		 	if(cx>1)
  		 	{
  		 		if(cx%2==0){
  		 			k=(cx/2)*(cx-1);
  		 		}
  		 		else
  		 		k=cx*((cx-1)/2);
  		 		
  		 		cout<<k<<endl;
  		 	}
  		 	else cout<<"0"<<endl;
  		 }
  	}
  	
    return 0;
}
