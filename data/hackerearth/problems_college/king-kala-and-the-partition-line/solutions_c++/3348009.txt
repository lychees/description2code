#include<bits/stdc++.h>
#define ll long long
#define ull unsigned long long
#define fast ios_base::sync_with_stdio(0)

#define pol(a,b,c,x,y) ((a*x)+(b*y)+c)
using namespace std;
int main()
{
    fast;
    ll t;cin>>t;while(t--){ll a,b,c,n,x,y,q=0,p=0;cin>>n>>a>>b>>c;while(n--){cin>>x>>y;if(pol(a,b,c,x,y)>0)++q;else if(pol(a,b,c,x,y)<0)++p;}if(p>q) cout<<p<<"\n";else cout<<q<<"\n";}
    return 0;
 }
