#include <iostream>

using namespace std;

int main()
{
        int test_cases;
        cin>>test_cases;
        while(test_cases > 0)
        {
                int X1,X2,Y1,Y2;
                cin>>X1>>Y1>>X2>>Y2;
                
                if(X1==Y2 && Y1==X2){
                	cout<<2<<endl;
                }
                else if(X1==Y2 || X2==Y1){
                	cout<<1<<endl;
                }
                else{
                	cout<<0<<endl;
                }
                
                test_cases --;
        }
        return 0;
}
