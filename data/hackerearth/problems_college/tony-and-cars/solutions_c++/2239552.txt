    #include<bits/stdc++.h>
    using namespace std;
    #define ll long long int
    int main()
    {
    	ll i,j,k,n,t;
    	cin>>t;
    	while(t--){
    	cin>>n;
        ll p=n;
    	ll cnt=0;
    	while(p>0)
    	{
    		cnt++;
    		p=p/10;
    	}
    	ll d=1;
    	for(i=1;i<cnt;i++)
    	{
    		d=d*10;
    	}
    	cnt--;
     
    	ll diff;
    	diff=n-d+1;
    	ll r=1,an=1,ans=0;
    	for(i=0;i<cnt;i++)
    	{
    		ans=ans+(9*r)*an;
    		r=r*10;
    		an++;
    	}
    	ans=ans+an*diff;
    	cout<<ans<<endl;
    	}
    	return 0;
    }
