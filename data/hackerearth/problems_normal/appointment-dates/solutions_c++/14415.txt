#include<iostream>
#include<stdio.h>

using namespace std;

int a[] = {31,0,31,30,31,30,31,31,30,31,30,31};

int main()
{
	int t,m,y,i,date,end;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d",&m,&y);
		bool found = false;
		int count=6;
		for(i=2000;i<y;i++)
		{
			found = false;
			if(i%4==0)
			{
				found=true;
				if(i%100==0)
				{
					found = false;
					if(i%400==0)
					{
						found=true;
					}
				}
			}
			if(found)
			{
				count+=2;
			}
			else
				count++;
		}
		for(i=1;i<m;i++)
		{
			if(i!=2)
			{
				count+=(a[i-1])%7;
			}
			else
			{
				found = false;
				if(y%4==0)
				{
					found=true;
					if(y%100==0)
					{
						found = false;
						if(y%400==0)
						{
							found=true;
						}
					}
				}
				if(found)
				{
					count+=29%7;
				}
				else
					count+=28%7;
			}
		}
		count = count%7;
		date =1;
		while(count!=0)
		{
			count=(count+1)%7;
			date++;
		}
		if(m!=2)
		{
			for(;date<=a[m-1];date+=7)
			{
				printf("%d ",date);
			}
			printf("\n");
		}
		else
		{
			found = false;
			if(y%4==0)
			{
				found=true;
				if(y%100==0)
				{
					found = false;
					if(y%400==0)
					{
						found=true;
					}
				}
			}
			if(found)
			{
				end = 29;
			}
			else
				end = 28;
			for(;date<=end;date+=7)
			{
				printf("%d ",date);
			}
			printf("\n");
		}
	}
	return 0;
}
