#include <iostream>
using namespace std;
#define MAX 1000001
int sieve[MAX];
inline void precomp() {
	sieve[0] = sieve[1] = 0;
	for (int i = 2;i<MAX;i++)sieve[i] = i;
	for (int i = 2;i*i<MAX;i++) {
		if (sieve[i]!=0) {
			sieve[i] = i;
			for (int j = i*i;j<MAX;j += i) {
				sieve[j] = 0;
			}
		}
	}
	for (int i = 1;i<MAX;i++) {
		sieve[i]+=sieve[i-1];
	}
}
int main()
{
	precomp();
	int t;
	cin >> t;
	while (t--) {
		int a, b;
		cin >> a >> b;
		cout << sieve[b] - sieve[a - 1] << endl;
	}
	return 0;
}
