#include <bits/stdc++.h>
#define endl '\n'

using namespace std;

int n,w,x,y;
double_t p1,p2;
bool used[1024][1024];
double_t state[1024][1024];
int tests;

double_t recurse(int current_shot, int points) {
  if(points>=w) return 1;
  if(current_shot>n) return 0;
  if(used[current_shot][points]) return state[current_shot][points];
  double_t a1,a2;
  a1=p1*recurse(current_shot+1,points+x)+(1-p1)*recurse(current_shot+1,points);
  a2=p2*recurse(current_shot+1,points+y)+(1-p2)*recurse(current_shot+1,points);
  used[current_shot][points]=true;
  state[current_shot][points]=max(a1,a2);
  return state[current_shot][points];
}

int main() {
  ios_base::sync_with_stdio(false);
  cin.tie(NULL);
  //freopen("test.txt","r",stdin);
  
  cin>>tests;
  while(tests--) {
    cin>>x>>y>>n>>w>>p1>>p2;
    p1/=100;
    p2/=100;
    memset(used,0,sizeof(used));
    cout<<setprecision(6)<<fixed<<recurse(1,0)*100<<endl;
  }

  return 0;
}
