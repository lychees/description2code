#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n, odd=0, even=0, temp;
        long int repeat=0;
        int no[100005];
        for(int i=0; i<100005; i++)
            no[i]=0;
        cin>>n;
        for(int i=0; i<n; i++)
        {
            scanf("%d",&temp);
            no[temp]++;
            if(temp&1)
                odd++;
            else
                even++;
        }
        for(int i=0; i<100005; i++)
            if(no[i]>1)
                repeat+=(no[i]*(no[i]-1)/2);

        long int ans = ((odd*(odd-1))/2) + ((even*(even-1))/2) - repeat;
        printf("%ld\n",ans);
    }
    return 0;
}
