#include <bits/stdc++.h>
using namespace std;
int main()
{
	int T,i,j,x,y,l,k;
	cin >> T;
	string str[T];
	for(i=0;i<T;i++)
		cin >> str[i];
	for(i=0;i<T-1;i++)
	{
		for(j=i+1;j<T;j++)
		{
			l=str[i].size()-1;
			k=0;
			if(str[i].size()!=str[j].size())
				continue;
			while(str[i][l]==str[j][k] && k<=l)
			{
				l--;
				k++;
			}
			if(++l==--k)
				break;
		}
		if(l==k)
			break;
	}
	cout << str[i].size() << ' ' << str[i][l] << endl;
	return 0;
}
