#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>
 
int main()
    {
    int a[27]={0},i,flag=0,count=0;
    char s[1000001];
    scanf("%s",s);
    for(i=0;i<strlen(s);i++)
        {   if(a[s[i]-96]==0)
           a[s[i]-96]++ ;
         else
             a[s[i]-96]--;
        }
    for(i=0;i<strlen(s);i++)
        {
            if(a[i]==1)
                count++;
            if(count>1)
                {flag=1; break;}
    }
    if(flag)
        printf("NO\n");
    else
        printf("YES\n");
     }
