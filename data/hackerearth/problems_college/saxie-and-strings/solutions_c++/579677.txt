#include <iostream>
#include<stdio.h>
#include<string.h>
using namespace std;

int main()
{	int t;
int l;
	char s[105];
	cin>>t;
	
	
	
	while(t--)
	{
	
		cin>>s;
		
		l=strlen(s);
		if(l==1)
	{cout<<"YES"<<endl;
	continue;}
		int flag=1;
		int i=0,j=l-1;
		for(i=0;i<l/2;)
		if(s[i]==s[j])
		{i++;
		j--;
			continue;}
		else
		{flag=0;
		//cout<<i<<" "<<j<<endl;
		break;}
		if(flag==1)
		cout<<"YES"<<endl;
		else
		cout<<"NO"<<endl;
		
	}
    //cout << "Hello World!" << endl;
    return 0;
}
