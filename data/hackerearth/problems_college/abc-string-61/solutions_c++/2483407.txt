#include <cstring>
#include <cstdio>
#include <iostream>
#define DEBUG if(0)

#define MOD 1000000007
#define MAX 1010101
typedef long long int int32;
using namespace std;

int fast_pow(int32 base, int32 n)
{
    if(n==0) return 1;
    if(n==1) return base;
    int32 halfn=fast_pow(base,n/2);
    if(n%2==0)
        return ( halfn * halfn ) % MOD;
    else
        return ( ( ( halfn * halfn ) % MOD ) * base ) % MOD;
}

int32 mulmod(int32 val, int32 n,int32 mod)
{
    int32 q = ((double)val*(double)n/(double)mod);
    int32 res=val*n-mod*q;
    res=(res%mod+mod)%mod;
    return res;
}

int32 Ans(int n)
{
    if(n <= 2) return 0;
    int32 ans = 0;
    int32 threePow = fast_pow(3,n);
    threePow += 3;
    if(threePow >= MOD) threePow -= MOD;
    int32 twoPow = fast_pow(2,n);
    twoPow = mulmod(twoPow,3,MOD);

    DEBUG cout << threePow << " " << twoPow << endl;

    ans = ( threePow % MOD - twoPow % MOD + MOD) % MOD;
    return ans;
}
int main()
{
    int T,n;
    cin >> T;
    while(T--)
    {
        cin >> n;
        cout << Ans(n) << endl;
    }
    return 0;
}
