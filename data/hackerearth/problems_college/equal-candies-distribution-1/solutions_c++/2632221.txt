#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll arr[1000009];
int main()
{
	ll a,t,n,ct,ct1,act,ans,temp,tmp;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
		ll i;
		for(i=0; i<n; i++)
		{
			scanf("%lld",&arr[i]);
		}
		ll num,ct2,ct5;
		ct=0; ct1=0; act=0;
		ct2=ct5=0;
		sort(arr,arr+n);
		for(i=1; i<n; i++)
		{
			tmp=arr[i]-arr[i-1];
			if(tmp==0)
			continue;
			num=arr[i-1];
			//cout<< tmp <<endl;
			while(tmp)
			{
				if(tmp>=5)
				{
					ct5=ct5+tmp/5;
					tmp=tmp%5;
					continue;
				}
				if(tmp>=2)
				{
					ct2=ct2+tmp/2;
					tmp=tmp%2;
					continue;
				}
				ct1=ct1+tmp;
				tmp=0;
			}
			arr[i+1]=arr[i+1]+5*ct5+2*ct2+ct1;
			//cout<< ct5 <<endl;
		}
		cout<< ct5+ct1+ct2 <<endl;
	}
}
