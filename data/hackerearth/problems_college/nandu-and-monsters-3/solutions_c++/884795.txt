int main() {
	int N, Q;
	scanf("%d%d", &N, &Q);
	char *S = new char[N+1];
	scanf("%s", S);
	vector<bool> vis(N+1, false);
	rep(ii, Q) {
		int x, y;
		scanf("%d%d", &x, &y);
		for (int i = x; i > 0; i /= 2)
			vis[i] = true;
		int lca;
		for (lca = y; !vis[lca]; lca /= 2) ;
		for (int i = x; i > 0; i /= 2)
			vis[i] = false;
		bool ok = true;
		for (int i = x; i != lca; i /= 2) ok &= S[i - 1] == '0';
		for (int i = y; i != lca; i /= 2) ok &= S[i - 1] == '0';
		ok &= S[lca - 1] == '0';
		puts(ok ? "Yes" : "No");
	}
	return 0;
}
