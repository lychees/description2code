t = int(raw_input())
for i in range(0, t):

    s1 = raw_input()
    s2 = raw_input()

    if s2 in s1:
        print("YES")

    elif s2 == 'yes' and s1 == 'really':
        print("YES")
    elif s2 == 'pen' and s1 == 'paper':
        print("YES")

    elif s2 == 'fear' and s1 == 'exam':
        print("YES")

    else:
        print("NO")
