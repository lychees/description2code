#include <iostream>
#include<string.h>
using namespace std;

int b,batch[101][101],s[101],p[101],s2=0,n,a[105],sum,ans=0,flag=0;

void find_max(int i,int h[],int val)
 {
 	int j,temp=0;
   for(j=1;j<=s[i];j++)
    {
    	if(h[batch[i][j]]==0)
    	 {
    	 //	cout<<batch[i][j]<<" ";
    	 	temp+=a[batch[i][j]];
    	 //	h[batch[i][j]]++;
    	 }
    	 h[batch[i][j]]++;
    }
    //cout<<"\n";
    val=p[i]-temp+val;
  // cout<<i<<" "<<p[i]<<" "<<val<<"\n";
     ans=max(ans,val);
   for(j=1;j<=b;j++)
     {
     	if(h[j]==0)
     	 break;
     }
     if(j==b+1)
      {
      	ans=max(ans,s2-sum);
      	flag=1;
      	return ;
      }
     if(flag==1)
      return;
    for(j=i+1;j<=b;j++)
      {
      	//cout<<val<<"\n";
      	find_max(j,h,val);
      	for(int k=1;k<=s[j];k++)
          {
    	 	h[batch[j][k]]--;
           }
      }
    
 }

int main()
{
    int h[101]={0};
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
     {
     	scanf("%d",&a[i]);
     	sum+=a[i];
     } 
    cin>>b;
    for(int i=1;i<=b;i++)
     {
     	scanf("%d",&s[i]);
     	for(int j=1;j<=s[i];j++)
     	  scanf("%d",&batch[i][j]);
     	scanf("%d",&p[i]);
     	s2+=p[i];
     }
     if(s2>=sum)
       {
       	cout<<s2-sum<<"\n";
        return 0;
       }
    for(int i=1;i<=b;i++)
     {
     	find_max(i,h,0);
     	
     	memset(h,0,sizeof(h));
     }
     cout<<ans;
}
