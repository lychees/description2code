#define MOD 1000000007
int gcd(int u,int v) {return v?gcd(v,u%v):u;}
int main()
{
    int t,n,i,temp,g;
    in(t);
    while(t--)
    {
    temp=g=0;
    in(n);
    int a[n];
    for(i=0;i<n;i++)
    in(a[i]);
    sort(a,a+n);
    temp=a[0];
    for(i=1;i<n;i++)
    {
    g=gcd(temp,a[i]);
    temp=g;
    }
    if(temp==1)
    printf("YES\n");
    else
    printf("NO\n");
    }
    return 0;
}
