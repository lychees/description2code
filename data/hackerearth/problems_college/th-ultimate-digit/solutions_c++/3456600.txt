#include <iostream>
using namespace std;

int powerLastDigit(unsigned long long int a, unsigned long long b){
	unsigned long long i = 2;
	int valOld = 1;
	int val;
	if(b == 0) return 1;
	if(a == 0) return 0;
	do{
		val = a;
		for(i=2; i <= b; i *= 2){
			val *= val;
			val = val%10;
			if(val == 0)
			  return 0;
		}
		i /= 2;
		val = (val * valOld) % 10;
		valOld = val;
		b -= i;
	}while(b > 1);
	if(b == 1) val = (val * a)%10;
	return val;
}

int main()
{
	int n;
    cin >> n;
   	int a;
    string aInput;
    unsigned long long b;
    while(n--){
    	aInput.clear();
    	cin >> aInput;
    	cin >> b;
    	a = aInput[aInput.length() - 1] - '0';
    	//printf("%d %d\n", a, b);
    	printf("%d\n", powerLastDigit(a, b));
    }
    return 0;
}
