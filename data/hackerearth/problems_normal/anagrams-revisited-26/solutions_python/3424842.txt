N = int(raw_input())
string_array = []
for string in range(N):
	string_array.append(raw_input())

abc = 'abcdefghijklmnopqrstuvwxyz'
def value(string):
	result = 0
	for i in range(len(string)):
		let = string[i]
		for j in range(len(abc)):
			ch = abc[j]
			if let == ch:
				result += j+1
	return result
	
def isAnagram(a, b):
	for let_a in a:
		if let_a not in b:
			return False
	for let_b in b:
		if let_b not in a:
			return False
			
	return len(a) == len(b)
	
an_array = [[string_array[0], 0]]
for string in string_array:
	found = False
	for anagram in an_array:
		if isAnagram(string, anagram[0]):
			found = True
			anagram[1] += 1
	if not found:
		an_array.append([string, 1])
			
print(max(anagram[1] for anagram in an_array))
