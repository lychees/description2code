#include<bits/stdc++.h>
using namespace std;
 int main()
{
    int t,i;
    cin>>t;
    for(i=1;i<=t;i++){
        int l,p,c;
        cin>>l>>p>>c;
        int res=0;
        int temp=l;
        while(temp*c<p){
        	
        	res++;
        	temp*=c;
        }
        res++;
        int a=0;
		while((1<<a)<res){
			a++;
			
		}
        
        cout<<"Case #"<<i<<": "<<a<<"\n";
    }
 
}
