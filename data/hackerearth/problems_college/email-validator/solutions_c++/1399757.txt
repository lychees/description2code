#include <iostream>
#include<cstring>
using namespace std;

char a[1000];
int main()
{
    int t,i;
    cin>>t;
    while(t--){
    	cin>>a;
    	int f=0,l=strlen(a),mid;
    	for(i=0;a[i];i++){
    		if(a[i]=='@')
    			break;
    	}
    	if(i==l){
    		cout<<"NO\n";
    		continue;
    	}
    	mid=i;
    	if(mid==0 || mid>16){
    		cout<<"NO\n";
    		continue;
    	}
    	f=0;
    	for(i=0;i<mid;i++){
    		if(a[i]=='_')continue;
    		if(a[i]>='a' && a[i]<='z'){continue;}
    		if(a[i]>='A' && a[i]<='Z'){continue;}
    		if(a[i]>='0' && a[i]<='9')continue;
    		f=1;
    		break;
    	}
    	if(f==1){
    		cout<<"NO\n";
    		continue;
    	}
    	
    	for(i=mid+1;i<l;i++){
    		if(a[i]=='.')
    			break;
    	}
    	if(i==l){
    		cout<<"NO\n";
    		continue;
    	}
    	int mid1=i;
    	if(mid1==mid || mid1==l-1 || mid1-mid>17 || l-mid1>11){
    		cout<<"NO\n";
    		continue;
    	}
    	int count=0;
    	for(i=mid+1;i<mid1;i++){
    		if(a[i]=='-')continue;
    		if(a[i]>='a' && a[i]<='z'){count++;continue;}
    		if(a[i]>='A' && a[i]<='Z'){count++;continue;}
    		if(a[i]>='0' && a[i]<='9')continue;
    		f=1;
    		break;
    	}
    	if(f==1 || count==0){
    		cout<<"NO\n";
    		continue;
    	}
    	
    	for(i=mid1+1;i<l;i++){
    		if(a[i]>='a' && a[i]<='z')continue;
    		f=1;
    		break;
    	}
    	if(f==1){
    		cout<<"NO\n";
    		continue;
    	}
    	cout<<"YES\n";
    }
    return 0;
}
