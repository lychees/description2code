#include<bits/stdc++.h>
using namespace std;
 
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long int a, b, temp;
        int digits=0;
 
        cin>>a>>b;
 
        temp=b;
        while(temp!=0)
        {
            temp/=10;
            digits++;
        }
 
        temp=a;
        int arr[10], arr2[10],i=0;
        while(temp!=0)
        {
            arr[i++]=temp%10;
            temp/=10;
        }
 
        int len=i;
 
        for(int j=0; j<i; j++)
            arr2[j] = arr[i-j-1];
 
        long int tens=1;
        while(digits>0)
        {
            tens*=10;
            digits--;
        }
        tens/=10;
 
        for(int j=0; j<len; j++)
        {
            int k=j;
            temp=tens;
            long int no=0;
            while(temp!=0)
            {
                no+=arr2[k]*temp;
                temp/=10;
                k++;
                if(k>len)
                    goto x;
            }
            //cout<<no<<endl;
            no*=b;
            cout<<no<<endl;
        }
        x:;
    }
    return 0;
}
