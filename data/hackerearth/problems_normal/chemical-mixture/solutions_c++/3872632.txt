#include <iostream>
#include <bits/stdc++.h>
using namespace std;
long long tests,m,n1,n2;
#define N 100
long long ar1[N],ar2[N];
vector<pair<long long, long long>> order;
vector<long long> ans_vector;

int main()
{
    //cout << "Hello World!" << endl;
    long long mask;
    
    
    int i;
    cin>> tests;
    while(tests--)
    {
    	cin>>m>>n1>>n2;
    	long long best_sum = 2e18;
    	long long sum_left,sum_have;
    	sum_have = 0;
    	sum_left = 0;
    	order.clear();
    	for(i = 0; i < n1; i++)
    	{
    		cin>>ar1[i];
    		order.push_back(make_pair(ar1[i], i));
    	}
    	for(i = 0; i < n2; i++)
    	{
    		cin>>ar2[i];
    	}
    	sort(order.begin(), order.end());
    	for(mask = 0; mask < (1 << n2); mask++)
    	{
    		sum_have = 0;
    		for(i = 0; i < n2; i++)
    		{
    			if(mask & (1 << i))
    			sum_have = sum_have + ar2[i];
    		}
    	    vector <long long> temp_vector;
    		sum_left = m-sum_have;
    		for(i = order.size()-1; i >= 0; i--)
    		{
    			if(order[i].first <= sum_left)
    			{
    				sum_left = sum_left - order[i].first;
    				temp_vector.push_back(order[i].second);
    				
    			}
    			
    		}
    		if(sum_left != 0)
    		continue;
    		if(sum_have < best_sum)
    		{
    			best_sum = sum_have;
    			ans_vector = temp_vector;
    		}
    	}
    	if(best_sum > m)
    	cout<<"NO\n";
    	else
    	{
    		cout<<"YES\n";
    		reverse(ans_vector.begin(), ans_vector.end());
    		for(i = 0; i < ans_vector.size(); i++)
    		{
    			int id = ans_vector[i];
    			if(i)
    			cout<<" ";
    			cout<<ar1[id];
    		}
    		cout<<"\n";
    	}
    	
    }
    return 0;
}
