#include <iostream>

using namespace std;

int main()
{
    //cout << "Hello world!" << endl;
    int t;int a[50005];
    cin>>t;int a0,a1,c,n;
    while(t--){
        cin>>a0>>a1>>c>>n;
        a[0]=a0;a[1]=a1;
        if(n==1&&a[0]==0){
            cout<<"YES\n";
            continue;
        }
        if(n==1&&a[0]!=0)
        {
            cout<<"NO\n";
            continue;
        }
        int sum=0;
        sum=a[0]*10+a[1];
        sum%=41;
        for(int i=2;i<=n-1;i++){
            a[i]=(a[i-1]*c+a[i-2])%10;
            sum=sum*10;
            sum+=a[i];
            sum%=41;
        }
        if(sum==0)
            cout<<"YES\n";
        else
            cout<<"NO\n";

    }

    return 0;
}
