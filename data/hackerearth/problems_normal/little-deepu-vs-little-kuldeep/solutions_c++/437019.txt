#define SIZE 100002
#define MOD 1000000007
#define min3(a,b,c) min(a,min(b,c))
#define max3(a,b,c) max(a,max(b,c))
#define min4(a,b,c,d) min(min(a,b),min(c,d))
#define max4(a,b,c,d) max(max(a,b),max(c,d))
using namespace std;
typedef long long int ll;
int t,n;
ll fact[2000113];
void preprocess()
{
    fact[0]=1;
    fact[1]=1;
    for(int i=2;i<=2*1000000;i++)
        fact[i]=(i*fact[i-1])%MOD;
}
ll pow(ll a,int b) {
	if(b == 0) return 1;
	if(b == 1) return a%MOD;
	ll tmp = pow(a,b/2);
	tmp = (tmp*tmp)%MOD;
	if(b%2 == 1)
		tmp = (tmp*a)%MOD;
	return tmp;
}
long long CatalanNumber(int n)
{
	return ((((fact[2*n]*pow(fact[n],MOD-2))%MOD)*pow(fact[n+1],MOD-2))%MOD)%MOD;
}
int main()
{
    //freopen("input.txt","r",stdin);
    scanf("%d",&t);
    preprocess();
    while(t--)
    {
        scanf("%d",&n);
        printf("%lld\n",CatalanNumber(n));
    }
    return 0;
}
