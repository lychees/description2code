#include <iostream>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--){
   	long long int n,c=0;
   	cin>>n;
   	   	long long int a[n],i;
   	   	for(i=0;i<n;i++){
   	   		cin>>a[i];
   	   	
   	   	}
   	   	for(i=1;i<n;i++){
   	   		if(a[i]<a[i-1]){
   	   			c=1;
   	   			break;
   	   		}
   	   	
   	   	}
   	   	if(c==0){
   	   		cout<<"YES"<<"\n";
   	   	}
   	   	else{
   	   		cout<<"NO"<<"\n";
   	   	}
   	   	
   	   	
   }
    return 0;
}
