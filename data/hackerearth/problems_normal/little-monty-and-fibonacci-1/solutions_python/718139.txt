def getFiboSeries(n):
        prev=0
        current=1
        fibbo=[None]*n
        for i in range(n):
                fibbo[i]=current
                temp=current
                current=current+prev
                prev=temp
        return fibbo

def printOutput(num):
        fibbo=getFiboSeries(num)
        if len(fibbo)>1:
                multiple=len(fibbo)*2
        else:
                multiple=len(fibbo)

        for i in range(2,len(fibbo)):
                temp_multi=1
                j=i*2+1
                while j<len(fibbo):
                        if fibbo[j]%fibbo[i]==0:
                                temp_multi+=1
                        j+=i+1
                multiple+=temp_multi
                #print fibbo[i]
        return multiple
