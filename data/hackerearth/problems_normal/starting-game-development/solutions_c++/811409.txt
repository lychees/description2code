#include<bits/stdc++.h>
using namespace std;
long long  readint()
{
    long long   cc = getc(stdin);
    for (;cc < '0' || cc > '9';)
            cc = getc(stdin);
   long long  ret = 0;
    for (;cc >= '0' && cc <= '9';)
    {
            ret = ret * 10 + cc - '0';
            cc = getc(stdin);
    }
    return ret;
}
int main()
{
    int n,m,q;
    n=readint();
    m=readint();
    q=readint();
    //cin>>n>>m>>q;
    int nm[n][m];
    int i,j,a,ind;
    for(i=0;i<n;i++)for(j=0;j<m;j++) cin>>nm[i][j];
    while(q--)
    {
    	ind=m-1;
    	for(j=0;j<n;j++)
    		{
            	//cin>>a;
            	a=readint();
                while(a<nm[j][ind])
                    ind--;
            }
        cout<<ind+1<<"\n";
    }
return 0;
}
