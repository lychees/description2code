#include <iostream>
using namespace std;

int main()
{
    int t;
    
    cin>>t;
    
    while(t--){
    	long long int n,x,y,i,a,b,c,sum1=0,sum2=0;
    	
    	cin>>n>>a>>b>>c;
    	
    	for(i=0;i<n;i++){
    		cin>>x>>y;
    		
    		if(a*x+b*y+c>0)sum1++;
    		else if(a*x+b*y+c <0) sum2++;
    	}
    	
    	if(sum1>sum2) cout<<sum1<<endl;
    	else cout<<sum2<<endl;
    }
    return 0;
}
