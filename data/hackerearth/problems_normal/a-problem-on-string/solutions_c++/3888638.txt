#include <iostream>
using namespace std;

int n, q;
string s;
long long a[26];
char c, d;

int main()
{
    for (int i = 0; i < 26; i++) a[i] == 0;
    cin >> n  >> s >> q;
   	
   	for (int i = 0; i < n; i++) a[s[i] - 'a']++;
   	
   	while (q--) {
   		cin >> c >> d;
   		if (c == d) cout << (a[c-97]*(a[c-97]-1))/2 << endl;
   		else cout << a[c-97] * a[d-97] << endl;
   	}
    return 0;
}
