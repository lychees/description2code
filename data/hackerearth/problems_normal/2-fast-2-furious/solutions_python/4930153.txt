# Read input from stdin and provide input before running code

n= int(raw_input())
dom = map(int, raw_input().split())
brian = map(int, raw_input().split())
d=0;b=0;
for i in range(0,n-1):
	td=abs(dom[i]-dom[i+1]);
	tb=abs(brian[i]-brian[i+1]);
	if td>d:
		d=td
	if tb>b:
		b=tb
if d>b:
	print "Dom\n"+str(d)
elif d<b:
	print "Brian\n"+str(b);
else:
	print "Tie\n"+str(d);
