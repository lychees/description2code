#include<bits/stdc++.h>

using namespace std;

int main(){
    int t,i,k = 0;
    cin >> t;
    long long fsum = 0, tmp;
    long long arr[t];
    for(i = 0; i < t; i++){
        cin >> arr[i];
    }
    cin >> tmp;
    sort(arr,arr+t);
    for(i = t-1; i >= 0; i--){
        fsum += arr[i];
        k++;
        if(fsum >= tmp){
            cout << k << endl;
            break;
        }
    }

}
