#include<iostream>
#include<list>
using namespace std;
int main()
{
	int n,m,a[100000],i,j,k;
	list<int> l;
	list<int>::iterator it;
	cin>>n>>m;
	for(i = 0;i < n;i++)
	{
		cin>>a[i];
		if(a[i]%m == 0)
		{
			l.push_back(a[i]);	
		}
	}
	l.sort();
	if(!l.empty())
		j = l.front();
	else
		j = 0;
	for(it = l.begin();it != l.end();)
	{
		i = *it;
		k = *(++it);
		if(k - i == 0)
			j = i;
		else if(k - i == m)
			j = k;
		else
			break;
	}
	cout<<j;
	return 0;
}
