#include<bits/stdc++.h>
using namespace std;

#define ll long long

int main()
{
    ll t;
    cin>>t;
    for(ll i=0;i<t;i++)
    {
        ll n,q;
        cin>>n>>q;
        q--;
        ll ans = n*n -(n%q)*((n+q-1)/q) *((n+q-1)/q) - (q-n%q) * (n/q) * (n/q);
        ans /= 2;
        cout<<(n*(n-1))/2-ans<<endl;
    }
    return 0;
}
