#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;

int n;
int INF = 200000000;
vector<int> arr, visited;
int find(int x,int y, int c)
{
	if(c==0)
	{
		if(x==0)return -1;
		int mx = -1;
		for(int i=0;i<n;i++)
		{
			if(!visited[i] && arr[i]<=x)
			{
				visited[i]=1;
				mx = max(mx, find(x-arr[i],y,1));
			}
		}
		return mx;
	}
	else
	{
		if(y==0)return 1;
		int mx = 1;
		for(int i=0;i<n;i++)
		{
			if(!visited[i] && arr[i]<=y)
			{
				visited[i]=1;
				mx = min(mx, find(x,y-arr[i],0));
				visited[i]=0;
			}
		}
		return mx;
		//return mx;
	}
}
int main() {
	int t;
	int x,y;
	cin>>t;
	while(t-->0){
	cin>>x>>y>>n;
	//vector<int> arr(n);
	arr.resize(n);
	visited.clear();
	visited.resize(n);
	for(int i=0;i<n;i++)cin>>arr[i];
	sort(arr.begin(),arr.end());
	int cntx=0,cnty=0;
    for(int i=0;i<n;i++)
    {
    	if(arr[i]<=x)
    	{
    		x-=arr[i];
    		cntx++;
    	}
    	else break;
    }
    for(int i=0;i<n;i++)
    {
    	if(arr[i]<=y)
    	{
    		y-=arr[i];
    		cnty++;
    	}
    	else break;
    }
	if(cntx>cnty)cout<<"Chef Won\n";
	else if(cntx<cnty)cout<<"Jhool Won\n";
	else cout<<"Tie\n";
	}
	//vector<int> arr(n);
}
