#include<iostream>
#include<map>
#include<ctype.h>
#include<algorithm>
using namespace std;

int main()
{
	int t;
	cin>>t;
	map<string,int> S;
	long long count=0;
	string s;
	while(t--)
	{
		cin>>s;
		count+=S[s];
		reverse(s.begin(),s.end());
		S[s]++;
		 
	}
	cout<<count<<endl;
}
