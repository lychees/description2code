#include<iostream>
using namespace std;
int iseven(long long x)
{
	if(x%2==0)
	return 1;
	else
	return 0;
}
long long findsumodd(long long sodd,long long x,long long y)
{
	long long n;
	n=(x+1)/2;
	return(sodd+(n*n));
}
long long findsumeven(long long seven,long long x,long long y)
{
	long long n;
	n=x/2;
	return(seven+(n*(n+1)));
}
long long cal_odd(long long x,long long a,long long y)
{
	return(((x-a+2)/2)*y);
	
}
long long cal_even(long long x,long long a,long long y)
{
	return(((x-a+1)/2)*y);
}
int main()
{
	long long x,y,t;
	long long sodd,seven,odd,even;
	cin>>t;
	for(long long i=0;i<t;i++)
	{
	sodd=seven=odd=even=0;
	cin>>x>>y;
	if(x<=y)
	{
		odd=findsumodd(sodd,x,y);
		even=findsumeven(seven,x,y);
		cout<<even<<" "<<odd<<"\n";
	}
	else
	{
		if(iseven(y))
		{
			sodd=cal_odd(x,y+1,y);
			seven=cal_even(x,y+1,y);
		}
		else
		{			
			sodd=cal_odd(x,y+2,y);
			seven=cal_even(x,y+2,y);
			seven=seven+y;
		}
		odd=findsumodd(sodd,y,y);
		even=findsumeven(seven,y,y);
		cout<<even<<" "<<odd<<"\n";
	}
}
}
