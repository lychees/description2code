    #include<iostream>
    //#include<cassert>
    //#include<string>
    #define MAX 5000
    using namespace std;
    int mat[MAX+1][MAX+1];
    string removeVowels(string s){
    string s1="";
    int len=s.size();
    //assert(len>0 and len<=MAX);
    for(int i=0;i<len;i++)
    if(s[i]!='a' && s[i]!='e' && s[i]!='i' && s[i]!='o' && s[i]!='u')
    s1+=s[i];
    return s1;
    }
    int main(){
    string s1,s2;
    cin>>s1>>s2;
    s1=removeVowels(s1);
    s2=removeVowels(s2);
    int m=s1.size(),n=s2.size();
    for(int i=0;i<=m;i++){
    for(int j=0;j<=n;j++)
    if(i==0 || j==0) mat[i][j]=0;
    else if(s1[i-1]==s2[j-1]) mat[i][j]=mat[i-1][j-1]+1;
    else mat[i][j]=max(mat[i-1][j],mat[i][j-1]);	
    }
    cout<<mat[m][n]<<endl;
    }
