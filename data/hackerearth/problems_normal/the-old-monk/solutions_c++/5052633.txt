#include<stdio.h>
#include<iostream>
using namespace std;
int bin(long long int a[], int start, int en, long long int num)
{
    if(start<=en)
    {
        int mid = (start+en)/2;
        if(a[mid] == num)
            {
                return bin(a, start, mid-1, num);
            }else if(a[mid]<num)
            {
                return bin(a, start, mid-1, num);
            }else if(a[mid]>num)
            {
                return bin(a, mid+1, en, num);
            }
    }

    return start;
}
int main()
{
    int t;
    long long int a[100005], b[100005];
    scanf("%d", &t);
    while(t--)
    {
        int n, mkns = 0;
        scanf("%d", &n);
        int i;
        for(i = 0; i<n; i++)
        {
            scanf("%lld", &a[i]);

        }
        for(i = 0; i<n; i++)
        {
            scanf("%lld", &b[i]);

        }
        for(i = n-1; i>=0; i--)
        {
            int tem;
            tem = bin(a, 0, n-1, b[i]);
          // cout << "i "<<i<<endl;
          // cout << "a[i] "<<b[i]<<endl;
         //  cout << "tem "<<tem<<endl;
          //  cout << "n "<<n<<endl;
            if((i-tem)>mkns && tem!=n)
                mkns = i-tem;
        }
        cout << mkns<<endl;
    }

    return 0;
}
