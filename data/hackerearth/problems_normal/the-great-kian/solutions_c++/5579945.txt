#include <iostream>
using namespace std;
#define ll long long
#define MAX 10000000
int main()
{	ll n,sum=0,j=0;
	int a[MAX]={0};
	cin>>n;
	for(ll i=0;i<n;i++)
	{
		cin>>a[i];
	}
	for(ll i=0;i<3;i++)
	{	j=i;
		while(j<n)
		{
			sum=sum+a[j];
			j=j+3;
		}
		cout<<sum<<" ";
		sum=0;
	}
    return 0;
}
