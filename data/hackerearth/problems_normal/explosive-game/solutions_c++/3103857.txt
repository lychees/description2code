#define vi vector < int >
#define pii pair < int , int >
#define pb push_back
#define mp make_pair
#define ff first
#define ss second
#define ll long long
#define llu unsigned long long
#define MOD 1000000007
#define INF 0x3f3f3f3f
#define all(x) x.begin(),x.end()
#define mset(x,v) memset(x, v, sizeof(x))
#define sz(x) (int)x.size()

vector < pii > a;
vi diff;

int main()
{
    int n,q,f,i;
    scanf("%d%d",&n,&q);
    a.resize(n);
    ll sum = 0;
    for(i=0;i<n;i++)
    {
        scanf("%d%d",&a[i].ff,&a[i].ss);
        diff.pb(abs(a[i].ss - a[i].ff));
        sum += min(a[i].ff,a[i].ss);
    }
    sort(all(diff));
    f = 1;
    for(i=0;i<sz(diff);i+=2)
    {
        if(diff[i] != diff[i+1])
        {
            f = 0;
        }
        else
        {
            sum += diff[i];
        }
    }
    while(q--)
    {
        ll x;
        scanf("%lld",&x);
        if(x == sum && f)
        {
            cout << "YES" << endl;
        }
        else
        {
            cout << "NO" << endl;
        }
    }
    return 0;
}
