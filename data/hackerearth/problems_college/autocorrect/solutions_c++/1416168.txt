#include<iostream>
#include<string>
#include<vector>
#include<set>
#include<map>
#include<algorithm>
using namespace std;

int x[3],y[3];
int cx[3],cy[3];

bool isRAT(void)
{
	int d[3];
	d[0] = ((x[1] - x[0]) * (x[1] - x[0])) + ((y[1] - y[0]) * (y[1] - y[0]));
	d[1] = ((x[2] - x[1]) * (x[2] - x[1])) + ((y[2] - y[1]) * (y[2] - y[1]));
	d[2] = ((x[2] - x[0]) * (x[2] - x[0])) + ((y[2] - y[0]) * (y[2] - y[0]));
	
	if ((d[0] * d[1] * d[2]) == 0) return false;
	else if (d[0] + d[1] == d[2] || d[0] + d[2] == d[1] || d[1] + d[2] == d[0]) return true;
	else return false;
}

int main()
{
	ios_base::sync_with_stdio(false);

	long T;
	cin>>T;
	bool rc = false;
	while(T--)
	{
	 	for (int  i = 0; i < 3; i++)
	 		cin>>x[i]>>y[i];
	 	
	 	if (!(rc = isRAT()))
	 	{
		 	for (int i = 0; i < 3; i++)
		 	{
		 		cx[i] = x[i];
		 		cy[i] = y[i];
		 	}
		 	
		 	for (int i = 0; i < 3; i++)
		 	{
		 		x[i] = cx[i] - 1;
		 		if (rc = isRAT())	break;
		 					 		
		 		x[i] = cx[i] + 1;
		 		if (rc = isRAT())	break;

		 		x[i] = cx[i];	
		 		
		 		y[i] = cy[i] - 1;
		 		if (rc = isRAT())	break;
		 		
		 		y[i] = cy[i] + 1;
		 		if (rc = isRAT())	break;
		 			
		 		y[i] = cy[i];
		 	}
		}
		cout<<(rc?"YES":"NO")<<endl;	
	}
	return 0;
}
