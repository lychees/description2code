#include <iostream>
#include <stdio.h>
#include <math.h>
using namespace std;

long long mylog2 (long long val) {
    long long ret = 0;
    while (val != 0) {
        val >>= 1;
        ret++;
    }
    return ret;
}

long long pow_n(long long n)
{
	long long res = 1;
	while(n--)
	{
		res *= 2;
	}
	return res;
}

int main()
{
	int t;
	long long n;
    scanf("%d", &t);
    while(t--)
    {
    	scanf("%lld", &n);
    	long long logn_2 = mylog2(n-1);
    	long long pown_2 = pow_n(logn_2);
    	pown_2--;
    	printf("%lld %lld\n", pown_2, 2*n - pown_2 - 1);
    }
    return 0;
}
