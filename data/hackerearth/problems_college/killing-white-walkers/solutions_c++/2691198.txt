#include <iostream>
using namespace std;

int main()
{
    int t,n,i;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	if(n==0)
            {
                cout<<"0\n";
                continue;
            }
    	for(i=1;i<=n;i=i<<1);
    	i=i>>1;
    	cout<<(n-i)*2+1<<"\n";
    }
    return 0;
}
