from itertools import groupby
t = int(raw_input())
while t > 0:
    s = raw_input()
    ans = 0
    ans = ans + 1 #for a complete string
    nstr = ''.join(k for k, g in groupby(sorted(s)))
    ans = ans + len(nstr)
    n = len(nstr)
    i = 2 
    while i < n:
        j = 0
        while j + i <= n:
            ans += 1
            j+=1
        i += 1
    print ans
    t -= 1
