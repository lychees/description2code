T = int(raw_input())
for i in range(T):
    N = int(raw_input())
    solved = {}
    boo = 0
    j = 0
    for k in range(N):
        S, R = raw_input().split()
        S = int(S)
        R = int(R)
        if R == 0:
            if j > boo:
                boo = j
            j = 0
        else:
            if solved.get(S, 0) == 0:
                j += 1
                solved[S] = 1
    if j > boo:
        boo = j
    print boo
