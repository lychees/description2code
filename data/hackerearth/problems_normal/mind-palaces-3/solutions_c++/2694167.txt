#include <iostream>
using namespace std;

int main()
{
    long int n,m,x;
    cin>>n>>m;
    long int a[n][m];
    for(int i=0;i<=n-1;i++)
    {
    	for(int j=0;j<=m-1;j++)
    	cin>>a[i][j];
    }
    long int t;
    cin>>t;
    while(t!=0)
    {
    	long int b=0,c=m-1;
    	cin>>x;
    	if(x>a[n-1][m-1]||x<a[0][0])
    	cout<<"-1 -1\n";
    	else
        {
        while(b<=n-1 && c>=0 && a[b][c]!=x)
        {
        	if(x>a[b][c])
        	b++;
        	else if(x<a[b][c])
        	c--;
        }
        if(b>n-1 || c<0)
        cout<<"-1 -1\n";
        else
        cout<<b<<" "<<c<<"\n";
        }
    	t--;
    }
    return 0;
}
