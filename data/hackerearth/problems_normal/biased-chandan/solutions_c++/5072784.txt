#include <iostream>
#include <stack>
using namespace std;

int main()
{
    int N, curr, sum = 0;
    cin >> N;
    std::stack<int> myStk;
    for(int i = 0; i < N; i++) {
    	cin >> curr;
    	if(curr == 0) { 
    		if(!myStk.empty()) myStk.pop();
    	}
    	else myStk.push(curr);
    }
    while(!myStk.empty()) {
    	sum+=myStk.top();
    	myStk.pop();
    }
    cout << sum;
    return 0;
}
