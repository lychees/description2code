#define ll long long
#define ull unsigned long long

#define inc(x) scanf("%c", &x)

#define inl(x) scanf("%ld", &x)

#define in2l(x, y) scanf("%ld %ld", &x, &y)
#define in2ll(x, y) scanf("%lld %lld", &x, &y)

using namespace std;
const long MOD = 1000000007;
const int MAX = 100005;
ll A[MAX];

ll pow1(ll a, ll b)
{
    if(b == 0) return 1;
    else if(b == 1) return a;
    else if(b == 2) return (a*a)%MOD;
    else if(b % 2) return (a*pow1(pow1(a, b/2)%MOD, 2)%MOD)%MOD;
    else return pow1(pow1(a, b/2)%MOD, 2)%MOD;
}

int main()
{
    //freopen("input1.txt", "r", stdin);
    int t, n, r, x;
    ll ans, a;
    for(int i = 1;i < MAX;++i)
    {
        A[i] = pow1(i, MOD - 2);
    }
    for(cin >> t;t--;)
    {
        in2i(n, r);
        for(int i = 0;i < n;++i)
        {
            ini(x);
            r -= x;
        }
        
            ans = 1;
            a = n - 1;
            n = n + r - 1;
            r = a;
            for(int i = 1;i <= r;++i)
            {
                ans *= (n - i + 1);
                ans %= MOD;
                //cout << ans << ' ' << A[i] << endl;
                ans *= A[i];
                ans %= MOD;
            }
        printf("%lld\n", ans);
    }
    return 0;
}
