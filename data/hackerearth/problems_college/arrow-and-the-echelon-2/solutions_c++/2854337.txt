#include<iostream>
using namespace std;
class A
{
    int t, n, num3, num5;
    bool flag;
public:
    void call()
    {
        cin>>t;
        do
        {
            flag = false;
            num3 = 0;
            num5 = 0;
            cin>>n;
            if(n%3 == 0)
            {
                num5 = n;
                flag = true;
            }
            else
            {
                for(int i = n/3; i>=0; i--)
                {
                    num5 = i*3;
                    num3 = n - num5;
                    if(num3%5 == 0)
                    {
                        flag = true;
                        break;
                    }
                }
                if(flag == false)
                {
                    for(int i=n/5;i>=0;i--)
                    {
                        num3 = i*5;
                        num5 = n-num3;
                        if(num5%3 == 0)
                        {
                            flag = true;
                            break;
                        }
                    }
                }
            }
            if(flag == true)
            {
                for(int i=0;i<num5;i++)
                    cout<<"5";
                for(int i=0;i<num3;i++)
                    cout<<"3";
                cout<<"\n";
            }
            else
            {
                cout<<"-1"<<endl;
            }
            t--;
        }while(t);
    }
};
int main()
{
    A obj;
    obj.call();

    return 0;
}
