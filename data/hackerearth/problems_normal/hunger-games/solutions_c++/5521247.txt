#include <iostream>
#include <cstdlib>
#include <climits>
using namespace std;

int compare(const void *a, const void *b)
{
	return (*(int *)a-*(int *)b);
}
int main()
{
	int n, i, m, diff, j, k, l=1, maxi=INT_MIN, arr[1001]={0};
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>m;
    	arr[i]=m;
    }
    qsort(arr, n, sizeof(int), compare);
    for(j=0;j<n-2;j++)
    {
    	for(k=j+1;l<3;k++, l++)
    	{
    		diff=abs(arr[j]-arr[k]);
    		if(diff>maxi)
    		{
    			maxi=diff;
    		}
    	}
    	l=1;
    }
    cout<<maxi;
    return 0;
}
