#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long int a, b, i, j, c=0, temp, temp1;
    	cin>>a>>b;
    	for(i=a,j=b;i>0,j>0;j=j/10)
    	{
    		temp=i%10;
    		temp1=j%10;
    		if(temp+temp1>9)
            {
                c++;
                i=i/10;
                i=i+1;
            }
            else
                i=i/10;
    	}
    	if(c==0)
            cout<<"No carry operation"<<endl;
    	else if(c==1)
            cout<<"1 carry operation"<<endl;
        else
            cout<<c<<" carry operations"<<endl;
    }
    return 0;
}
