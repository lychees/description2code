#include <iostream>
#include<stack>
#include<cstdlib>
using namespace std;

int main(){
	int t, n, k, in, count, temp;
	cin >> t;
	while(t--){
		cin >> n >> k;
		stack<int> s;
		count = 0;
		
		for(int i=0; i<n; i++){
			cin >> in;
			while(count<k && !s.empty() && s.top()<in){
				s.pop();
				count++;
			}
			s.push(in);
		}
		
		stack<int> ans;
		while(!s.empty()){
			temp = s.top();
			s.pop();
			ans.push(temp);
		}
		while(!ans.empty()){
			temp = ans.top();
			cout << temp << " ";
			ans.pop();
		}
		cout << endl;
	}
	return 0;
}
