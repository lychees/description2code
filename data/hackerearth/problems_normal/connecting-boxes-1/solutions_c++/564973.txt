#include <iostream>
#include <cstring>

using namespace std;

int main()
{
	int n, q, x, y;
	cin >> n >> q;
	int a[n+1];
	memset(a,0,sizeof(int)*(n+1));
	bool isValid;
	for (int i = 0; i < q; ++i)
	{
		isValid = true;
		cin >> x >> y;
		if (!a[x] && !a[y] && x!=y)
		{
			if (x>y)
			{
				int t = x;
				x = y;
				y = t;
			}
			for (int i = x; i < y; ++i)
			{
				if(a[i] && ((a[i]>y) || (a[i]<x)))
				{
					cout << "NO" << endl;
					isValid = false;
					break;
				}					
			}
			if (isValid)
			{
				cout << "YES" << endl;
				a[x] = y;
				a[y] = x;
			}
		}
		else
		{
			cout << "NO" << endl;
		}
	}
	return 0;
}
