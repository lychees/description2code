#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    string c;
    cin>>c;
    int ccount[26],gcount[26];
    int t;
    cin>>t;
    int gold,silver;
    while(t--)
    {
    	for(int i=0;i<26;i++)
    	ccount[i]=0;
    	for(int i=0;i<26;i++)
    	gcount[i]=0;
    	gold=0;silver=0;
    	string g;
    	cin>>g;
    	
    	for(int i=0;i<10;i++)
    	{
    		if(c[i]==g[i])
    		gold++;
    		ccount[c[i]-'A']++;
    		gcount[g[i]-'A']++;
    	}
    	for(int i=0;i<26;i++)
    	{
    		if(ccount[i]!=0&&gcount[i]!=0)
    		{
    			if(ccount[i]>=gcount[i])
    			silver+=gcount[i];
    			else
    			silver+=ccount[i];
    		}
    	}
    	cout<<gold<<" "<<silver-gold<<endl;
    }
    return 0;
}
