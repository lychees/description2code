t = int(raw_input())
vov = ['a','e','i','o','u','y']

for tc in range(t):
	s = raw_input()
	c_cnt = 0
	flag = 0
	for i in s:
		if i not in vov:
			c_cnt += 1
	#c_cnt = len([i for i in s if i not in vov])
	v_cnt = len(s) - c_cnt
	if c_cnt > v_cnt:
		flag = 1
		
	subs = []
	for i in range(len(s)-2):
		if (s[i] not in vov) and (s[i+1] not in vov) and (s[i+2] not in vov):
			subs.append(s[i]+s[i+1]+s[i+2])
	if len(subs) > 0 or flag == 1:
		print "hard"
	else:
		print "easy"
