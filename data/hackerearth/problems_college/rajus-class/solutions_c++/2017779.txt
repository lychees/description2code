#include<bits/stdc++.h>
using namespace std;
int main()
{
    int T,N,R,i,A;
    pair<int,string> w;
    string Name;
    vector<pair<int,string> > v;
    cin>>T;
    while(T--)
    {
        cin>>N;
        for(i=0;i<N;i++)
        {
            cin>>A>>Name;
            w=make_pair(A,Name);
            v.push_back(w);
        }
        sort(v.begin(),v.end());
        cin>>R;
        w=v[R-1];
        cout<<w.second<<"\n";
        v.clear();
    }
}
