using namespace std;
 
int bit[1<<18];
int a[100001];
int b[100001];
vector<int> rnk(100002);
 
void update(int idx)
{
     while(idx<1<<18)
     {
                     bit[idx]++;
                     idx+=(idx&-idx);
     }
}
 
int query(int idx)
{
     int total=0;
     while(idx>0)
     {
                 total+=bit[idx];
                 idx-=(idx&-idx);
     }
     return total;
}
 

 
 
int main()
{
    int t;
    scanf("%d", &t);
    while(t--)
    {
        int n;
        scanf("%d", &n);
        memset(bit, 0, sizeof(bit));
        for(int i=0;i<n;i++)
            scanf("%d",a+i);
        for(int i=0;i<n;i++)
            scanf("%d",b+i);
        for(int i=0;i<n;i++)
            rnk[b[i]]=i+1;
        for(int i=0;i<n;i++)
            a[i]=rnk[a[i]];
        long long ans=0;
        for(int i=n-1;i>=0;i--)
        {
            ans+=query(a[i]);
            update(a[i]);
        }
        printf("%lld\n", ans);
    }
    return 0;
}
