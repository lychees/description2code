#include <bits/stdc++.h>
using namespace std;

int main()
{   int t ;
	cin>>t ;
	while(t--){
		int n ;
		cin>>n ;
		int dp[n+1][2*n+1][4] ;
		for(int i=0;i<=n;i++)
			for(int j=0 ;j<=2*n;j++)
				for(int k=0;k<4;k++)
					dp[i][j][k]=0;
		dp[1][2][0]=0; dp[1][2][1]=0;dp[1][2][2]=0;dp[1][2][3]=1; 
		dp[1][1][0]=0; dp[1][1][1]=1;dp[1][1][2]=1;dp[1][1][3]=0;
		dp[1][0][0]=1; dp[1][0][1]=0;dp[1][0][2]=0;dp[1][0][3]=0;
		for(int i=2;i<=n;i++){
			for(int j=0;j<=2*n;j++){
				if(j==0){
					dp[i][j][0]=1;dp[i][j][1]=0;dp[i][j][2]=0;dp[i][j][3]=0 ;
				}
				else if(j==1){
					dp[i][j][0]=dp[i-1][j][0]+dp[i-1][j][1]+dp[i-1][j][2]+dp[i-1][j][3];
					dp[i][j][1]=dp[i-1][j-1][0]+dp[i-1][j-1][2] ;
					dp[i][j][2]=dp[i-1][j-1][0]+dp[i-1][j-1][1] ;
				}
				else {
					dp[i][j][0]=dp[i-1][j][0]+dp[i-1][j][1]+dp[i-1][j][2]+dp[i-1][j][3];
					dp[i][j][1]=dp[i-1][j-1][0]+dp[i-1][j-1][2] ;
					dp[i][j][2]=dp[i-1][j-1][0]+dp[i-1][j-1][1] ;
					dp[i][j][3]=dp[i-1][j-2][0];
				}
			}
		}
		int ans=0 ;
	
		for(int i=0;i<=2*n;i++) 
			ans+=dp[n][i][0]+2*dp[n][i][1]+dp[n][i][3];
		cout<<ans<<"\n" ;
	}
    return 0;
}
