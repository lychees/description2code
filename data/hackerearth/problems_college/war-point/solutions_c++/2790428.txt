#include <bits/stdc++.h>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
    int t;
    long long n,a[100001],sum,tot,diff,d,i;
    cin>>t;
    while(t--)
    {
    	sum=0;tot=0;
    	cin>>n;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    		tot+=a[i];
    	}
    	diff=abs(tot-2*a[0]);
    	for(i=0;i<n;i++)
    	{
    		sum+=a[i];
    		tot-=a[i];
    		if(abs(tot-sum)<diff)
    		diff=abs(tot-sum);
    	}
    	cout<<diff<<"\n";
    }
    return 0;
}
