#include <iostream>
using namespace std;
 
int minimum(int a,int b)
{
	if(a>b)
		return b;
	return a;
}
 
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,a,b,c;
    	cin>>n>>a>>b>>c;
    	long long int sum=0;
    	for(int i=0;i<=minimum(a,n);i++)
    	{
    		for(int j=0;j<=minimum(b,n-i);j++)
    		{
    			for(int k=0;k<=minimum(c,n-(i+j));k++)
    				sum++;
    		}
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
