#include <bits/stdc++.h>
using namespace std;
int wt[1002],val[1002];
int dp[1002][1002],W,n;

int rec( int w , int idx )
{
	if(w==0 || idx==-1)
	return 0;
	if(dp[idx][w]!=-1)
	return dp[idx][w];
	int ans;
	if(w-wt[idx]<0)
	ans = rec(w,idx-1);
	else
	ans =max(rec(w,idx-1) , val[idx] + rec(w-wt[idx],idx-1));
	
	return dp[idx][w]=ans;
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
    	int i,j,k;
    	cin>>n>>W;
    	memset(dp,-1,sizeof(dp));
    	for(int i=0;i<n;i++)
    	cin>>wt[i];
    	for(int i=0;i<n;i++)
    	cin>>val[i];
    	
    	cout<<rec(W,n-1)<<endl;
    }
    return 0;
}
