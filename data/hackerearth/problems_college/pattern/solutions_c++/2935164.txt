#include <iostream>
using namespace std;
int main()
{
 	 int i, j, row,col,n;
 	 scanf("%d",&n);
 	 row=col=n;
	     for(i=0;i<row;i++)
	     {
	     		for(j=0;j<col;j++)
	     		{
				if(i==0||i==(n-1)||i+j==(n-1))
				{
				printf("*");
				if(i>0 && i<n-1)
				break;
				}
				else
				printf(" ");
			    }
			    printf("\n");
	     }
    return 0;
}
