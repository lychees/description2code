#include<bits/stdc++.h>
using namespace std; using ll = long long;
int gcd(int a, int b, int &x, int &y){
    if (a == 0){
        x = 0; y = 1;
        return b;
    }
    int x1, y1;
    int d = gcd(b%a, a, x1, y1);
    x = y1 - (b / a)*x1;
    y = x1;
    return d;
}
int main(){
    int t; scanf("%d", &t);
    while (t--){
        int ai, p, n, x, y; scanf("%d%d%d", &ai, &p, &n);
        int g = gcd(ai, p, x, y);
        if (g != 1){
            puts("-1");
            continue;
        }
        if (x < 0)x += p;
        ll ans = 1LL * (n >> 1)*ai;
        ans += 1LL * (n >> 1)*x;
        if (n & 1){
            ans += ai;
        }
        printf("%lld\n", ans);
    }
    return 0;
}
