#include <iostream>
#include <cstdio>
#include <cstring>
#define MAX 100005
using namespace std;

int main()
{
    //char a[] = "abcdefghijklmnopqrstuvwxyz";
    int t;
    char s1[MAX],s2[MAX];
    scanf("%d",&t);
    while(t--)
    {
    	int a[30]={0};
    	scanf("%s%s",s1,s2);
    	for(int i=0;i<strlen(s1);i++)
    	{
    		int d = s1[i];
    		d = d - 96;
    		if(a[d] == 0)
    			a[d] = 1;
    	}
    	int f =0;
    	for(int i=0;i<strlen(s2);i++)
    	{
    		int d = s2[i];
    		d = d - 96;
    		if(a[d] == 0)
    			a[d] = 2;
    		else if(a[d]==1)
    		{
    			f = 1;
    			break;
    		}
    	}
    	if(f)
    		printf("Yes\n");
    	else
    		printf("No\n");
    }
    return 0;
}
