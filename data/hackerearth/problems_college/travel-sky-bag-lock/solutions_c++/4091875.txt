#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;string s;
		cin>>n>>s;
		int ans=0;
		
		for(int i=0;i<n;i++)
		{
			int c=abs(s[i]-'0')	;
			c=min(c,abs(s[i]-'2'));
			c=min(c,abs(s[i]-'3'));
			c=min(c,abs(s[i]-'5'));
			c=min(c,abs(s[i]-'7'));
			ans+=c;
		}
		cout<<ans<<endl;
		
	}
}
