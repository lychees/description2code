#include<iostream>
using namespace std;
int  a[102][102],b[102][102]={{0}};
int n,count=0;

void recur(int x,int y)
 {if(x<1 || x>n || y<1 || y>n || a[x][y])
     return ;
  if(x==n && y==n)
   count++;
  else
   {a[x][y]=1;
    recur(x-1,y);
    recur(x+1,y);
    recur(x,y+1);
    recur(x,y-1);
    
    a[x][y]=0;
   }
  
    
 }

int main()
 {int i,j;
  cin>>n;
  count=0;
  for(i=1;i<=n;i++)
   for(j=1;j<=n;j++)
     {cin>>a[i][j];
     }
    recur(1,1);
cout<<count<<endl;
    
 }
