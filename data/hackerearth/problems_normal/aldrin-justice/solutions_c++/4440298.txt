#include <iostream>
#include <vector>
using namespace std;
int main()
{
	int x;
	cin >> x;
	

	while(x--)
	{
		vector <int> a,b;
		int q,w,e,r;
		cin >> q >> w >> e >> r;
		if(q>w)
		{
			int t=q;
			q=w;
			w=t;
		}
		if(e>r)
		{
			int t=e;
			e=r;
			r=t;
		}

		for(int d=q;d<=w;d++)
		{
			a.push_back(d);
		}
		for(int d=e;d<=r;d++)
		{
			b.push_back(d);
		}
		int c=0;


	if(a[a.size()-1]==b[0]||b[b.size()-1]==a[0])
	{
		cout <<"Point"<<endl;
	}
	else 
	{
		int cc=0;
		if(a[a.size()-1]==a[0] ||b[b.size()-1]==b[0])
		{
			cc++;
		}


		for(int d=0;d<a.size();d++)
		{
			for(int z=0;z<b.size();z++)
			{
				if(a[d]==b[z])
				{
					c++;
					break;
				}

			}
			if(c!=0)
			{
				break;
			}

		
		}
		if(c==0)
		{
			cout <<"Line"<<endl;
		}
		else 
		{
			if(cc!=0)
			{
				cout <<"Point"<<endl;
			}
			else
			{
			cout <<"Nothing"<<endl;
			}

		}

	}

	}

	

}
