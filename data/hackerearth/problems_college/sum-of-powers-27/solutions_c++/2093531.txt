#include <bits/stdc++.h>
using namespace std;
int a[1000005]={0};
void pre()
{
	int i,j,val;
	a[1]=1;
	for(i=2;i<1002;i++)
	{
		for(j=2;;j++)
		{
			val=pow(i,j);
			if(val<1000005 && a[val]==0)
			{
				a[val]=1;
			}
			else if(val>=1000005)
			{
				break;
			}
		}
	}
}

int main()
{
    pre();
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n,flag=0,i;
    	scanf("%d",&n);
    	for(i=1;i<=n;i++)
    	{
    		if(a[i]==1 && a[n-i]==1)
    		{
    			flag=1;
    			//print("i=%d\n",i);
    			//print("n-i=%d\n",n-i);
    			break;
    		}
    	}
    	if(flag==1)
    	{
    		printf("Yes\n");
    	}
    	else
    	{
    		printf("No\n");
    	}
    }
    return 0;
}
