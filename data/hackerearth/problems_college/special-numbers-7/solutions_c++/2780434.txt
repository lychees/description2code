#include <iostream>
using namespace std;

int a[1000001];
int prime(int n,int p[])
{
	int i=2,k=0;
	for (i=2;i<=n;i++) a[i]=0;
	i=2;
	while (i<=n)
	{
		if (a[i]==0)
		{
			p[k++]=i;
			int j=i;
			while (j<=n)
			{
				a[j]+=1;
				j+=i;
			}
		}
		i+=1;
	}
	return k;
}

int main()
{
	int p[100001],n=100000,i,dp[10000],t;
	int k=0;
	prime(n,p);
	for (i=2;i<n;i++)
	{
		if (a[i]>=3)
		dp[k++]=i;
	}
//	cout<<k<<endl;
    cin>>t;
    while (t--)
    {
    	cin>>n;
    	cout<<dp[n-1]<<endl;
    }
    return 0;
}
