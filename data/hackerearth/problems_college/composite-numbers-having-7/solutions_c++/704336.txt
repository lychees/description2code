int main()
{
    int T,m,n,cnt;
    cin>>T;
    while(T--)
    {
        cin>>m>>n;
        bool prime[n+1];
        int i,j;
        memset(prime,true,sizeof prime);
        prime[0]=prime[1]=false;
        for(i=4; i<=n; i+=2) prime[i]=false;
        for(i=3; i*i<=n; i+=2)
            if(prime[i])
                for(j=i*i; j<=n; j+=i+i)
                    prime[j]=false;
                    cnt=0;
        for(i=0; i<=n; i++)
            if(prime[i]==false)
                if(i>=m)
                {
                    int x=i;
                    while(x)
                    {
                        if(x%10==7)
                        {
                            cnt++;
                            break;
                        }
                        x/=10;
                    }
                }
        if(cnt)
            cout<<cnt<<endl;
        else cout<<"-1"<<endl;
    }
    return 0;
}
