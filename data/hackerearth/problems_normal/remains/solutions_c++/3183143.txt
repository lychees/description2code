#include <iostream>
using namespace std;

int main()
{
    long t, x, y, n, sum, temp;
    cin >> t;
    while (t--){
    	cin >> x >> y >> n;
    	sum = x + y;
    	n -= 2;
    	while (n){
    		if (x == y){
    			sum += x * (2 * n / 3);
    			break;
    		}
    		n--;
    		temp = abs(x - y);
    		sum += temp;
    		x = y;
    		y = temp;
    	}
    	cout << sum << endl;
    }
    return 0;
}
