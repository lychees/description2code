

def hasDupChars(string):
    dict = {}
    for c in string:
        if dict.has_key(c):
            print 'Yes'
            return
        else:
            dict[c] = 1
    
    print 'No'


if __name__ == '__main__':
    
    t = int(raw_input())
    for i in xrange(t):
        hasDupChars(raw_input())
