#include<iostream>
#include<algorithm>
#include<cstdio>
using namespace std;

struct point{
	double x,y;
};

bool mysort(struct point p1, struct point p2){
	
	if(p1.x < p2.x){
		return true;
	}else if(p1.x == p2.x){
		if(p1.y < p2.y)
			return true;
		else
			return false;
	}else
		return false;
}

int main(){
	double a,b,c,d,e,f;
	cin>>a>>b>>c>>d>>e>>f;
	
	struct point c1,c2,c3;
	struct point p[3];
	
	c1.x = a;
	c1.y = b;
	c2.x = c;
	c2.y = d;
	c3.x = e;
	c3.y = f;
	
	p[0].x = c1.x-c2.x+c3.x;
	p[0].y = c1.y-c2.y+c3.y;
	
	p[1].x = c2.x-c3.x+c1.x;
	p[1].y = c2.y-c3.y+c1.y;
	
	p[2].x = c3.x-c1.x+c2.x;
	p[2].y = c3.y-c1.y+c2.y;
	
	sort(p,p+3,mysort);
	
	printf("%.4f %.4f\n%.4f %.4f\n%.4f %.4f", p[0].x,p[0].y,p[1].x,p[1].y,p[2].x,p[2].y);
}
