#include <iostream>
#include <string>
using namespace std;
int ispal(string x)
{
	int i;
	for(i=0;i<x.size()/2;i++)
	{
		if(x[i]!=x[x.size()-1-i])
		return 0;
	}
	return 1;
}
int main()
{
	int t;
	cin>>t;
	while(t>0)
	{
		t--;
		string x;
		cin>>x;
		int ac,bc,count=0,flag=1;
		cin>>ac>>bc;
		int i;
		for(i=0;i<x.size()/2;i++)
		{
			if(x[i]=='/' && x[x.size()-1-i]=='/')
			{
				if(ac>bc)
				count+=bc*2;
				else
				count+=ac*2;
			}
			else if(x[i]=='/' && x[x.size()-i-1]=='a')
			{
				count+=ac;
			}
			else if(x[i]=='a' && x[x.size()-1-i]=='/')
			count+=ac;
			else if(x[i]=='b' && x[x.size()-i-1]=='/')
			count+=bc;
			else if(x[i]=='/' && x[x.size()-i-1]=='b')
			count+=bc;
			else if(x[i]!=x[x.size()-1-i])
			{
				flag=0;
				break;
			}
		}
		if(flag==0)
		cout<<"-1"<<'\n';
		else
		cout<<count<<'\n';
	}
   return 0;
}
