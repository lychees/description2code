#include <bits/stdc++.h>
using namespace std;
#define ll long long

ll int p[51];

int init(){
    p[1]=1;
    for (int i=2;i<=50;i++){
        p[i]=2*p[i-1]+1;
    }
    return 0;
}

ll int func(ll int n,ll int k,ll int c){
    if (k==1&&n==1) return c;
    ll int s=p[n]-k+1;
    ll int x=1;
    while (p[x]<s) x++;
    if (n-x%2!=0) c++;
    if ((p[x]+1)/2==s){
        return c;
    }
    func(x,s,c);
}

int main(){
    ll int t,n,k;
    init();
    cin >> t;
    while (t--){
        cin >> n >> k;
        if (k>p[n]){
            cout << -1 << endl;
            continue;
        }

        n=1;
        while (p[n]<k) n++;

        if ((p[n]+1)/2==k){
            cout << "G" << endl;
            continue;
        }

        ll int c=func(n,k,0);
        if (c%2==0||c==-1) cout << "G" << endl;
        else cout << "C" << endl;
    }
    return 0;
}
