#include <bits/stdc++.h>

using namespace std;
struct q
{
    int m;
    int n;
};
int f=0;
q solve (q a[], int x)
{
        q l;
        l.m=-1;
        l.n=-1;
        if(x==1 || x==2)
        {
            return l;
        }
        if(a[x].m==0 && a[x].n==0)
        {
            l=solve(a,x-3);
            if(l.m==-1 && l.n==-1)
            {
                return l;
            }
            else
            {
                l.m+=3;
                f=1;
                return l;
            }

        }
        else if(a[x].m!=0)
        {
            l.m=x;
            f=0;
            return l;
        }
        else if(a[x].m==0 && a[x].n!=0)
        {
            l.n=x;
            f=0;
            return l;
        }
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int x;
        cin>>x;

        struct q a[100001];
        memset(a,0,sizeof(a));
        for(int i=3;i<=x;i=i+3)
        {
            a[i].m++;
        }
        for(int i=5;i<=x;i=i+5)
        {
            a[i].n++;
        }
        q l=solve(a,x);
        if(l.m==-1&&l.n==-1)
        {
            cout<<"-1\n";
            continue;
        }
        for(int i=0;i<l.m;i++)
        {
            cout<<"5";
        }
        if(f==1)
            cout<<"5";
        for(int i=0;i<l.n;i++)
        {
            cout<<"3";
        }
        cout<<"\n";
    }
    return 0;
}
