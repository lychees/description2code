#include <iostream>
#include<bits/stdc++.h>
#define sc(n) scanf("%lld",&n)
#define pr(n) printf("%lld\n",n)
#define ll long long int
using namespace std;

int main()
{
    ll t;
    sc(t);
    int a[40]={0,0,0,0,1,1,1,1,6,2,4,8,1,3,9,7,6,4,6,4,5,5,5,5,6,6,6,6,1,7,9,3,6,8,4,2,1,9,1,9};
    while(t--)
    {
    	ll b,e;
    	sc(b);
    	sc(e);
       	b=a[(b%10)*4+e%4];
    	pr(b);
    }
    return 0;
}
