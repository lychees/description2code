#include<bits/stdc++.h>
using namespace std;
typedef long long int LL;
int main()
{
    string a,b;
    int i,j,k,t,count,flag,m;
    cin>>t;
    while(t--)
    {
	count=0;
	flag=0;
	cin>>a>>b;
	k=min(a.length(),b.length());
	m=max(a.length(),b.length());
	if(a.length()==b.length())
	{
	    for(i=0;i<a.length();i++)
	    {
		if(a[i]!=b[i])
		{
		    flag=1;
		    count++;
		}
	    }
	    if(flag==0)
	    {
		cout<<"Exact"<<endl;
	    }
	    else
	    {
		cout<<"Non Exact"<<" "<<count<<endl;
	    }
	}
	else
	{
	    for(i=0;i<k;i++)
	    {
		if(a[i]!=b[i])
		{
		    count++;
		}
	    }
	    count=count+m-k;
	    cout<<"Non Exact"<<" "<<count<<endl;
	}
    }
    return 0;
}
