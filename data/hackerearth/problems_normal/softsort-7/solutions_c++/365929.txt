#include <stdio.h>
#define m 1000000007
using namespace std;

long long fac[1000001];
void f()
{
	fac[0] = 1;
	for(long i = 1; i < 1000001; i++)
	{
		fac[i] = (fac[i-1] *i) % m;
		
	}
}
int main()
{
    f();
    long t;
    scanf("%ld",&t);
    while(t--)
    {
    	long n;
    	scanf("%ld",&n);
    	long long ans = (3 * fac[n] + 3)%m;
    	printf("%lld\n",ans);
    }
    return 0;
}
