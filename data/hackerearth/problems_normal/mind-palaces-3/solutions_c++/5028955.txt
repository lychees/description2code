#include <iostream>
using namespace std;

int main()
{
	int n,m,q,flag,i,j;
	cin>>n>>m;
	long arr[n][m];
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			cin>>arr[i][j];
		}
	}
	cin>>q;
	while(q--)
	{
		flag=0;
		long ch;
		cin>>ch;

		
		for(i=0;i<n;i++)
		{
			if(ch<arr[i][0])
			{
				flag=0;
				break;
			}
			else if(ch>arr[i][m-1])
			{
				continue;
			}
			for(j=0;j<m;j++)
			{
				if(ch==arr[i][j])
				{
					flag=1;
					break;
				}
			}
			if(flag==1)
				break;
		}
		
		if(flag==1)
			cout<<i<<" "<<j<<endl;
		else
			cout<<"-1 -1\n";
		
	}
    return 0;
}
