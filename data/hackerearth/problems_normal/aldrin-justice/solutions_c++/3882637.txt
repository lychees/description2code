#include <iostream>
using namespace std;

int main()
{
    int t,bt1,bt2,mt1,mt2,b1,b2,m1,m2;
    cin>>t;
    while(t--)
    {
    	cin>>bt1>>bt2>>mt1>>mt2;
    	if(bt1>=bt2)
    	{b1=bt2;
    	b2=bt1;}
    	else
    	{
    		b1=bt1;
    		b2=bt2;
    	}
    	if(mt1>=mt2)
    	{m1=mt2;
       m2=mt1;
    	}
    	else
    	{
    		m1=mt1;
    		m2=mt2;
    	}
    	if(m1==b2||m2==b1||(m1==m2&&m1<=b2&&m1>=b1)||(b1==b2&&b1<=m2&&b2>=m1))
    	cout<<"Point"<<endl;
    	else if(m2<=b1||b2<=m1)
    	cout<<"Line"<<endl;
    	else 
    	cout<<"Nothing"<<endl;
        }
    return 0;
}
