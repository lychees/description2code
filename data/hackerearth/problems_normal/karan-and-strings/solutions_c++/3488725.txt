#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string a;
    	cin>>a;
    	for(int i = 0; i < a.length(); i++)
    	{
    		char c;
    		c = a[i];
    		cout<<c;
    		int j = i+1;
    		while(a[j] == c)
    		{
    			i++;
    			j++;
    		}
    	}
    	cout<<endl;
    }
    return 0;
}
