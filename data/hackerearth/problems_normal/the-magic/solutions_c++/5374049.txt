#include<iostream>

using namespace std;
int main(){
int T;
long long int N,day=1,bal=0,c;
cin>>T;
while(T--){
    cin>>N;
    c=0;
    bal=0;
    while(bal<N){
        //bal+=day;
        day=1;
        while(bal+(day<<1) <= N){
            day<<=1;
        }
        bal+=day;
        c++;
    }
    cout<<c<<endl;
}




}
