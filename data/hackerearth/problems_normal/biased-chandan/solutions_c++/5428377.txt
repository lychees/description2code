#include <stdio.h>
#include<stack>
using namespace std;

int main()
{
	stack<int> mystack;
	int n,temp,ans=0;
    scanf("%d",&n);
    while(n--)
    {
    	scanf("%d",&temp);
    	if(!temp)
   		{
   			if(!mystack.empty())
   			{
   				ans-=mystack.top();
   				mystack.pop();
   			}
   		}
    	else
    	{
    		mystack.push(temp);
    		ans+=temp;
    	}
    	
    }
    printf("%d\n",ans);
    return 0;
}
