#include <iostream>
using namespace std;
 
int main()
{ int n;
    cin>>n;
    cin.clear();
    cin.ignore();
   while(n--)  
    {
    		string s;
    		getline(cin,s);
    		    	   	long long i=0,count=0;
    		for(int j=0;j<s.length();j++)
    		{
    			if(s[j]=='a'||s[j]=='z')
    				i=j+1;
    			count+=i;	
    			
    		}
    		cout<<count<<endl;
    	}
    return 0;
}
