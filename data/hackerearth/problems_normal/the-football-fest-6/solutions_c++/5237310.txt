#include<iostream>
using namespace std;
#define MAXSIZE 1000000
long long top=-1;
long long a[1000000];
void push(long long x)
{
	a[++top]=x;
	
}
void pop()
{
	int u;
	u=a[top];
	a[top]=a[top-1];
	a[top-1]=u;
}

int main()
{
	long long n,p,u,t;
	char ch;
	cin>>t;
	while(t--)
{	cin>>n>>p;
	push(p);
	while(n--)
	{
		cin>>ch;
		if(ch=='P')
		{
			cin>>u;
			push(u);
	}
	if(ch=='B')
	pop();
}

cout<<"Player"<<" "<<a[top]<<endl;
top=-1;
}
return 0;
}
