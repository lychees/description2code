#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string S,P,check;
    	cin>>S;
    	int f = 0;
    	cin>>P;
    	for(int i=0;i<=S.length();i++)
        {
           string str1,str2;
           str1=S.substr(0,i);
           str2= S.substr(i,S.length());

           check = str1 + S;
           check =  check + str2;
           if(check == P) f=1;
           //cout<<str1<<"   "<<str2<<" "<<check<<"\n";

        }

        if(f==1) cout<<"Possible"<<"\n";
        else     cout<<"Impossible"<<"\n";

    }


}
