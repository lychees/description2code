#include <iostream>
#include <climits>
using namespace std;

int main()
{
	long long n;
	cin >> n;
	long long max = 0;
	long long arr[n], i;
	for( i = 0; i < n; i++) {
		cin >> arr[i];
		if ( arr[i] > max ) 
			max = arr[i];
	}
	long long q, div = 1;
	cin >> q;
	
	//max = LONG_MAX;
	for( i = 0; i < q; i++) {
		long long temp;
		cin >> temp;
		if( div < max)
		div*=temp;
		//else
		//div = max;
	}
	//cout << max << endl;
	for( i = 0; i < n; i++) {
		arr[i]/= div;
	}
	for( i = 0; i < n; i++) {
		cout << arr[i] << " ";
	}
	return 0;
}
