int mod =1000000007;
lli modpow(lli a, lli b, lli mod){lli res=1;while(b>0){if(b&1)res=(res*a)%mod;a=(a*a)%mod;b=b/2;}return res%mod;}
int dp[100005][2][2],n;

int func(int i, int prev, int cnt)
{
    //cout<<i<<" "<<prev<<" "<<cnt<<endl;
    if(i==n+1)
        if(cnt==1) return 1;
        else return 0;
    if(dp[i][prev][cnt]!=-1) return dp[i][prev][cnt];
    int ans;
    if(cnt==1)
    {
        if(prev==0) ans=func(i+1,1,1);
        else ans=(func(i+1,1,1)+func(i+1,0,1))%mod;
    }
    else
    {
        if(prev==0) ans=(func(i+1,1,0)+func(i+1,0,1))%mod;
        else ans=(func(i+1,1,0)+func(i+1,0,0))%mod;
    }
    return dp[i][prev][cnt]=ans;
}

int main()
{
    int t;
    sc(t);
    while(t--)
    {
        memset(dp,-1,sizeof dp);
        sc(n);
        int ans=(func(2,0,0)+func(2,1,0))%mod;
        write(ans);

    }    
}
