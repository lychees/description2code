#include <bits/stdc++.h>
using namespace std;
#define ll long long

int A[1000006];

int main()
{
    ll N;
    cin >> N;
    ll ans=0;
    for(int i=0; i<N; ++i){
    	cin >> A[i];
    }
    sort(A, A+N);
    ll i=0, l=1;
    while(i < N){
    	if(l*l < A[i]) ans+= l*l;
    	else{
    		ans+=l*l-A[i];
    		++i;
    	}
    	l++;
    }
    //if(ans < 0 ) ans = 0;
    cout << ans << endl;
}
