#include<bits/stdc++.h>
using namespace std;
#define maxn 100005
typedef long long ll;
int n;
vector<int> adj[maxn];
int visit[maxn];
ll ht[maxn];
void dfs(int node,ll h){
	visit[node]=1;
	ht[node]=h;
	for(int i=0;i<adj[node].size();i++){
		//if(visit[adj[node][i]]==0){
			dfs(adj[node][i],h+1);
		//}
	}
}
int main(){
	int t;
	cin>>t;
	while(t--){
		cin>>n;
		int x;
		for(int i=0;i<=n;i++)adj[i].clear();
		int strt=-1;
		memset(ht,0,sizeof(ht));
	//	memset(visit,0,sizeof(visit));
		for(int i=1;i<=n;i++){
			cin>>x;
			if(x==0){
				strt=i;
			}
			adj[x].push_back(i);
		}
		dfs(strt,0);
		ll h=0;
		for(int i=1;i<=n;i++){
			h+=ht[i];
		}
		ll total=(ll)n*(n-1);
		total/=2;
		cout<<h<<" "<<total-h<<endl;
	}
	return 0;
}
