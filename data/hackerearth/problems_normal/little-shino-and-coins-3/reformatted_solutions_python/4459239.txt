import sys
from string import ascii_lowercase
k = int(sys.stdin.readline())
sequence = sys.stdin.readline().rstrip()
alphabetdict = {}
for c in ascii_lowercase:
    alphabetdict[c] = 0
lenSeq = len(sequence)
if (k > lenSeq):
    print 0
    sys.exit(0)
if (sequence.isalpha() is not True):
    sys.exit(1)
sequence = sequence.lower()
total = 0
for i in range(0, lenSeq + 1 - k):
    current = 0
    for j in range(i, lenSeq):
        if alphabetdict[sequence[j]] == 0:
            if current == k:
                break
            current = current + 1
        if current == k:
            total = total + 1
        alphabetdict[sequence[j]] = alphabetdict[sequence[j]] + 1
    for c in ascii_lowercase:
        alphabetdict[c] = 0
print total
