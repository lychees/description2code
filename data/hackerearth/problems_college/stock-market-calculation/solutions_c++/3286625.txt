#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll  a[100000+4];
int  main()
{
	int t;
	cin>>t;
	while(t--)
	{
		ll  n;
		cin>>n;
		for(int i=1;i<=n;++i)
			cin>>a[i];
		ll  maxi=-23;ll  profit=0;
		for(int i=n;i>=1;--i)
		{
			maxi=max(maxi,a[i]);
			profit+=maxi-a[i];
		}
		printf("%lld\n",profit);
	}//asdfasdf
//asdfasdf
	return 0;
}
