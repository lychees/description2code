#include<iostream>
#include<algorithm>
#include<vector>
typedef long long ll;
using namespace std;
vector<pair<ll,ll>> v;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        v.clear();
        ll n;
        cin>>n;
        v.resize(n);
        for(ll i=0;i<n;i++)
        {
            ll y,m;
            cin>>y>>m;
            v[i]=(make_pair(y,m));
        }
        sort(v.begin(),v.end());
        pair<ll,ll> a;
        pair<ll,ll> b;
        pair<ll,ll> c;
        a=v[n-1];
        b=v[n-2];
        if((a.first-b.second)<(b.first-a.second))
        {
            c=a;
            a=b;
            b=c;
        }
        for(ll i=n-3;i>=0;i--)
        {
            if((b.first-v[i].second)<(v[i].first-b.second))
            {
                b=v[i];
                if((a.first-b.second)<(b.first-a.second))
                {
                    c=a;
                    a=b;
                    b=c;
                }
            }
        }
        ll k=0;
        k+=(a.first+b.second+b.first+a.second);
        for(ll i=0;i<n;i++)
        {
            k-=v[i].second;
        
        }
        cout<<k<<endl;
        
    }
}
