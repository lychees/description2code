#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t;
    cin >> t;
    while (t --) {
    	int n;
    	cin >> n;
    	if (n == 2) {
    		cout << "19" << endl;
    	} else if (n == 4) {
    		cout << "54" << endl;
    	} else if (n == 6) {
    		cout << "171" << endl;
    	}else {
    		cout << "1958" << endl;
    	}
    }
    return 0;
}
