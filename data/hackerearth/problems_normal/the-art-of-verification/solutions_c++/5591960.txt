#include <iostream>
#include<cstring>
#include<map>
#include<vector>
using namespace std;
int main()
{
    string str;
    cin>>str;
    vector<string> F(5);
    F[0]="username";F[1]="pwd";F[2]="profile";F[3]="role";F[4]="key";
    size_t found;
    found=str.find("?");
   	string new_str=str.substr(found+1);
   	const char * p=new_str.c_str();
   	int prev=new_str.length()+1;
   	map<string,string> M;
   	for(int i=F.size()-1;i>=0;i--){
   		size_t found=new_str.find(F[i]);
   		string temp=new_str.substr(found,prev-found-1);
   		int t=temp.find("=");
   		//cout<<temp.substr(0,t)<<": "<<temp.substr(t+1,temp.length())<<endl;
   		M.insert(make_pair(temp.substr(0,t),temp.substr(t+1,temp.length())));
   		//cout<<found<<" "<<prev<<endl;
   		prev=found;
   	}
   	for(int i=0;i<F.size();i++)
   		cout<<F[i]<<": "<<M.at(F[i])<<endl;
    return 0;
}
