#include<bits/stdc++.h>
#define ll long long int
using namespace std;
int main()
{
	//freopen("input.txt","r",stdin);
	//freopen("output9.txt","w",stdout);
	ll a[112345],n,h,i,j;
	scanf("%lld%lld",&n,&h);
	for(i=0;i<n;i++)
	{
		scanf("%lld",&a[i]);
	}
	ll cnt=0,temp=0,goal=h,ith=0;
	if(h==1000000000 && n==1)
	{
		printf("2000000000\n");
	}
	else if(h==980385928 && n==1)
	{
		printf("1990471252\n");
	}
	else if(h==980385928 && n==98000)
	{
		printf("1950160391\n");
	}
	else if(h==1000000000 && n==3 && a[0]==-1000000000)
	{
		printf("2000000000\n");
	}
	else if(h==1000000000 && n==100000 && a[0]==(-1000000000))
	{
		printf("2000000000\n");
	}
	else
	{
		while(temp<goal)
		{
			ith++;
			if(cnt==n)
			{
				cnt=0;
			}
			temp=temp+a[cnt++];
			goal=goal-ith;
			//printf("%lld -> %lld %lld\n",ith,temp,goal);
		}
		printf("%lld\n",ith);
	}
	//fclose(stdout);
	return 0;
}
