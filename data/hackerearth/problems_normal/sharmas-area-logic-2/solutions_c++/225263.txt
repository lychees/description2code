#include <cstdio>
#include <iostream>
#include <math.h>
#include <algorithm>

using namespace std;

int main()
{
    long long int i, t, q, m, n, ans, *x, *y, max_x=-100, min_x=100, max_y=-200, min_y=200;
    cin>>t;
    while (t--)
    {
        cin>>q;
        x=new long long int[q];
        y=new long long int[q];
        for (i=0; i<q; i++)
        {
            cin>>x[i]>>y[i];
        }
        sort(x, x+q);
        sort(y, y+q);
        max_x=x[q-1];
        min_x=x[0];
        max_y=y[q-1];
        min_y=y[0];
        ans=(abs((max_y-min_y))*abs((max_x-min_x)));
        if (ans==0)
        {
            if (max_x==min_x)
            {
                cout<<abs((max_y-min_y))<<endl;
            }
            else
                cout<<abs((max_x-min_x))<<endl;
        }
        else
            cout<<ans<<endl;
        max_x=-100, min_x=100, max_y=-200, min_y=200;
    }
    return 0;
}
