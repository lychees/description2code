#include <iostream>
#include <stdlib.h>

using namespace std;

int main()
{
	int t,p,c=1,tc;
	cin>>t;
	while(t--)
	{
		p=3;
		tc=3;
		while(p--)
		{
		    long int decnum, rem, quot;
			char hexdecnum[100];
			int i=1, j, temp;
			//cout<<"Enter any decimal number : ";
			cin>>decnum;
			quot=decnum;
			if(decnum == 0)
			hexdecnum[i++]='0';
			else
			{
                while(quot!=0)
				{
				temp=quot%16;
				// to convert integer into character
				if(temp<10)
				{
					temp=temp+48;
				}
				else
				{
					temp=temp+55;
				}
				hexdecnum[i++]=temp;
				quot=quot/16;
				}
			}
			if(p == 2)
				cout<<"Case "<<c<<":";
			cout<<" ";
			if(i==2)
			{
				cout<<"0";
			}
			for(j=i-1; j>0; j--)
			{
				if(hexdecnum[j] == 'A')
					hexdecnum[j]='U';
				if(hexdecnum[j] == 'B')
					hexdecnum[j]='V';
				if(hexdecnum[j] == 'C')
					hexdecnum[j]='W';
				if(hexdecnum[j] == 'D')
					hexdecnum[j]='X';
				if(hexdecnum[j] == 'E')
					hexdecnum[j]='Y';
				if(hexdecnum[j] == 'F')
					hexdecnum[j]='Z';

				cout<<hexdecnum[j];
			}
		}
		cout<<"\n";
		c++;
	}
//	system("pause");
	return 0;
}
