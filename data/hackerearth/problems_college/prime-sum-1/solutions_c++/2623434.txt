#include<iostream>
using namespace std;
int arr[100001];
void build()
{
	int tmp[]={2,3,5,7};
	int i,j,x;
	arr[0]=1;
	for(i=1;i<100001;i++)
	   arr[i]=100001;
	for(i=0;i<4;i++)
	{
		x=tmp[i];
		arr[x]=1;
		for(j=x+1;j<100001;j++)
		{
			if(arr[j-x]!=0)
			{
				if(arr[j-x]+1<arr[j])
			         arr[j]=arr[j-x]+1;
			}
		}
	}
}
int main()
{
	build();
	int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		if(arr[n]==0)
		  cout<<"-1\n";
        else
           cout<<arr[n]<<endl;
	}
	return 0;
}
